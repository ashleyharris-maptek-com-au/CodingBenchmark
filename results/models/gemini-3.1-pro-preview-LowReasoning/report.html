uint64_t hilbert_index(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int s = 31; s >= 0; s--) {
        uint32_t rx = (x >> s) & 1;
        uint32_t ry = (y >> s) & 1;
        d = (d << 2) | ((rx * 3) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x;
            x = y;
            y = t;
        }
    }
    return d;
}

inline double distance_nodes(int a, int b, const float* x, const float* y) {
    double dx = x[a] - x[b];
    double dy = y[a] - y[b];
    return std::sqrt(dx * dx + dy * dy);
}

void reverse_segment(vector<int>& tour, int start, int end) {
    int n = tour.size();
    while (start != end) {
        int next_start = (start + 1 < n) ? start + 1 : start + 1 - n;
        if (next_start == end) {
            swap(tour[start], tour[end]);
            break;
        }
        swap(tour[start], tour[end]);
        start = next_start;
        end = (end - 1 >= 0) ? end - 1 : end - 1 + n;
    }
}

struct Point {

    Point operator+(const Point& o) const { return Point(x + o.x, y + o.y, z + o.z); }
    Point operator*(double s) const { return Point(x * s, y * s, z * s); }
    double length() const { return sqrt(x*x + y*y + z*z); }
    void normalize() {
        double l = length();
        if (l > 1e-12) { x /= l; y /= l; z /= l; }
    }
};

double dot(const Point& a, const Point& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Point cross(const Point& a, const Point& b) {
    return Point(a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x);
}

    }

    int build_recursive(int start, int end) {

        node.start = start; node.end = end;

        }

            node.is_leaf = true; node.left = node.right = -1;
            nodes.push_back(node);

        }

        node.is_leaf = false;
        Vec3 size = node.aabb.max_p - node.aabb.min_p;
        int axis = 0;
        if (size.y > size.x && size.y > size.z) axis = 1;
        if (size.z > size.x && size.z > size.y) axis = 2;

            Vec3 ca = ta.v0 + ta.v1 + ta.v2; Vec3 cb = tb.v0 + tb.v1 + tb.v2;
            if (axis == 0) return ca.x < cb.x;
            if (axis == 1) return ca.y < cb.y;
            return ca.z < cb.z;
        };

        int mid = start + (end - start) / 2;
        nth_element(tri_indices.begin() + start, tri_indices.begin() + mid, tri_indices.begin() + end, comp);

        int left = build_recursive(start, mid);
        int right = build_recursive(mid, end);
        nodes[curr_idx].left = left; nodes[curr_idx].right = right;

    }

            for(int i=nodes[node_idx].start; i<nodes[node_idx].end; ++i) out.push_back(tri_indices[i]);
        } else {
            query_aabb(nodes[node_idx].left, box, out);
            query_aabb(nodes[node_idx].right, box, out);
        }
    }
};

bool sat_test(const Vec3& axis, const Vec3 p1[3], const Vec3 p2[3]) {
    double l2 = length_squared(axis);
    if (l2 < 1e-14) return false;
    double l = sqrt(l2);
    Vec3 naxis = axis / l;
    double min1 = 1e30, max1 = -1e30, min2 = 1e30, max2 = -1e30;
    for(int i=0; i<3; ++i) {
        double p = dot(p1[i], naxis);
        if (p < min1) min1 = p; if (p > max1) max1 = p;
        double q = dot(p2[i], naxis);
        if (q < min2) min2 = q; if (q > max2) max2 = q;
    }
    return (max1 < min2 - 1e-7 || max2 < min1 - 1e-7);
}

bool tri_tri_intersect(const Triangle& t1, const Triangle& t2) {
    Vec3 p1[3] = {t1.v0, t1.v1, t1.v2};
    Vec3 p2[3] = {t2.v0, t2.v1, t2.v2};
    Vec3 e1[3] = {p1[1]-p1[0], p1[2]-p1[1], p1[0]-p1[2]};
    Vec3 e2[3] = {p2[1]-p2[0], p2[2]-p2[1], p2[0]-p2[2]};
    Vec3 n1 = cross(e1[0], e1[1]);
    Vec3 n2 = cross(e2[0], e2[1]);
    
    if (sat_test(n1, p1, p2)) return false;
    if (sat_test(n2, p1, p2)) return false;
    for(int i=0; i<3; ++i) {
        if (sat_test(cross(e1[i], n1), p1, p2)) return false;
        if (sat_test(cross(e2[i], n2), p1, p2)) return false;

        }
    }
    return true;
}

void split_polygon(const Polygon& poly, const Plane& plane, Polygon& front, Polygon& back) {
    int n = poly.vertices.size();
    vector<double> dists(n);
    for(int i=0; i<n; ++i) dists[i] = dot(poly.vertices[i], plane.n) - plane.d;
    
    for(int i=0; i<n; ++i) {
        int j = (i + 1) % n;
        double d1 = dists[i], d2 = dists[j];

        if (d1 <= 1e-9) back.vertices.push_back(poly.vertices[i]);

bool ray_tri(const Vec3& ro, const Vec3& rd, const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    Vec3 edge1 = v1 - v0, edge2 = v2 - v0;
    Vec3 h = cross(rd, edge2);
    double a = dot(edge1, h);
    if (a > -1e-8 && a < 1e-8) return false;
    double f = 1.0 / a;
    Vec3 s = ro - v0;
    double u = f * dot(s, h);
    if (u < 0.0 || u > 1.0) return false;
    Vec3 q = cross(s, edge1);
    double v = f * dot(rd, q);
    if (v < 0.0 || u + v > 1.0) return false;
    double t = f * dot(edge2, q);
    return t > 1e-8;
}

bool is_inside(const BVH& bvh, const Mesh& mesh, const Vec3& c) {
    Vec3 dir = random_dir();
    Vec3 rd_inv = {1.0/(abs(dir.x)<1e-9?1e-9:dir.x), 1.0/(abs(dir.y)<1e-9?1e-9:dir.y), 1.0/(abs(dir.z)<1e-9?1e-9:dir.z)};
    int count = 0;
    
    vector<int> stack; stack.push_back(0);
    while (!stack.empty()) {
        int node = stack.back(); stack.pop_back();
        if (!bvh.nodes[node].aabb.intersect_ray(c, rd_inv)) continue;
        if (bvh.nodes[node].is_leaf) {

        }
    }
    return (count % 2) != 0;
}

bool point_on_segment(const Vec3& p, const Vec3& a, const Vec3& b) {
    Vec3 ab = b - a;
    double l2 = length_squared(ab);
    if (l2 < 1e-14) return false;

    if (t <= 1e-5 || t >= 1.0 - 1e-5) return false;

    return length_squared(p - proj) < 1e-12;
}

    int orig_idx;
    int w, h;
    int board_idx;
    int x, y;
    int rotated;
};

struct FreeRect {
    int x, y, w, h;
};

struct Board {
    vector<FreeRect> free_rects;
    long long area_left;
    int max_w, max_h;
};

int N;
int BOARD_W, BOARD_H;

struct Solution {
    int num_boards = 1e9;

inline void recalc_board(Board& b) {
    b.max_w = 0;
    b.max_h = 0;
    for (const auto& fr : b.free_rects) {
        if (fr.w > b.max_w) b.max_w = fr.w;
        if (fr.h > b.max_h) b.max_h = fr.h;
    }
}

inline void calculate_score(int fw, int fh, int w, int h, int place_rule, long long& score1, long long& score2) {
    int dw = fw - w;
    int dh = fh - h;
    if (place_rule == 0) { // BSSF
        score1 = std::min(dw, dh);
        score2 = std::max(dw, dh);
    } else if (place_rule == 1) { // BAF
        score1 = (long long)fw * fh;
        score2 = std::min(dw, dh);
    } else if (place_rule == 2) { // BLSF
        score1 = std::max(dw, dh);
        score2 = std::min(dw, dh);
    }
}

bool pack_item(Rect& item, vector<Board>& boards, int place_rule, int split_rule) {
    long long best_score1 = 3e18;
    long long best_score2 = 3e18;
    int best_board = -1;
    int best_fr_idx = -1;
    int best_rot = 0;
    
    long long item_area = (long long)item.w * item.h;
    
    for (int b = 0; b < (int)boards.size(); b++) {
        if (boards[b].area_left < item_area) continue;
        
        bool can_fit = (item.w <= boards[b].max_w && item.h <= boards[b].max_h) ||
                       (item.h <= boards[b].max_w && item.w <= boards[b].max_h);
        if (!can_fit) continue;
        
        for (int i = 0; i < (int)boards[b].free_rects.size(); i++) {
            const auto& fr = boards[b].free_rects[i];
            
            // Unrotated
            if (item.w <= fr.w && item.h <= fr.h) {
                long long s1, s2;
                calculate_score(fr.w, fr.h, item.w, item.h, place_rule, s1, s2);
                if (s1 < best_score1 || (s1 == best_score1 && s2 < best_score2)) {
                    best_score1 = s1; best_score2 = s2;
                    best_board = b; best_fr_idx = i; best_rot = 0;
                    if (s1 == 0 && s2 == 0) goto found_perfect;
                }
            }
            
            // Rotated
            if (item.h <= fr.w && item.w <= fr.h && item.w != item.h) {
                long long s1, s2;
                calculate_score(fr.w, fr.h, item.h, item.w, place_rule, s1, s2);
                if (s1 < best_score1 || (s1 == best_score1 && s2 < best_score2)) {
                    best_score1 = s1; best_score2 = s2;
                    best_board = b; best_fr_idx = i; best_rot = 1;
                    if (s1 == 0 && s2 == 0) goto found_perfect;
                }
            }
        }
    }
    
found_perfect:
    if (best_board == -1) return false;
    
    auto fr = boards[best_board].free_rects[best_fr_idx];
    boards[best_board].free_rects[best_fr_idx] = boards[best_board].free_rects.back();
    boards[best_board].free_rects.pop_back();
    
    int fit_w = best_rot ? item.h : item.w;
    int fit_h = best_rot ? item.w : item.h;
    
    item.board_idx = best_board;
    item.x = fr.x;
    item.y = fr.y;
    item.rotated = best_rot;
    
    FreeRect r1, r2;
    int dw = fr.w - fit_w;
    int dh = fr.h - fit_h;
    bool split_horizontal = false;
    
    if (split_rule == 0) { // SLLA
        split_horizontal = (dw >= dh);
    } else if (split_rule == 1) { // SSLA
        split_horizontal = (dw < dh);
    } else if (split_rule == 2) { // MAXAS
        long long area_horiz = std::max((long long)dw * fit_h, (long long)fr.w * dh);
        long long area_vert  = std::max((long long)fit_w * dh, (long long)dw * fr.h);
        split_horizontal = (area_horiz > area_vert);
    } else { // MINAS
        long long area_horiz = std::max((long long)dw * fit_h, (long long)fr.w * dh);
        long long area_vert  = std::max((long long)fit_w * dh, (long long)dw * fr.h);
        split_horizontal = (area_horiz <= area_vert);
    }
    
    if (split_horizontal) {
        r1 = {fr.x + fit_w, fr.y, dw, fit_h};
        r2 = {fr.x, fr.y + fit_h, fr.w, dh};
    } else {
        r1 = {fr.x, fr.y + fit_h, fit_w, dh};
        r2 = {fr.x + fit_w, fr.y, dw, fr.h};
    }
    
    if (r1.w > 0 && r1.h > 0) boards[best_board].free_rects.push_back(r1);
    if (r2.w > 0 && r2.h > 0) boards[best_board].free_rects.push_back(r2);
    
    boards[best_board].area_left -= item_area;
    recalc_board(boards[best_board]);
    
    return true;
}

template<typename TimePoint>
bool pack_all(vector<Rect>& rects, int place_rule, int split_rule, int& out_boards, long long& out_tie, TimePoint start_time, bool is_first_pass) {
    vector<Board> boards;
    boards.reserve(rects.size());
    out_boards = 0;
    out_tie = 0;
    
    for (int i = 0; i < (int)rects.size(); i++) {
        if (!is_first_pass && (i & 1023) == 0) {
            auto curr_time = high_resolution_clock::now();
            double elapsed = duration_cast<duration<double>>(curr_time - start_time).count();
            if (elapsed > 28.5) {
                return false; 
            }
        }
        
        if (!pack_item(rects[i], boards, place_rule, split_rule)) {
            Board new_board;
            new_board.free_rects.reserve(64);
            new_board.free_rects.push_back({0, 0, BOARD_W, BOARD_H});
            new_board.area_left = (long long)BOARD_W * BOARD_H;
            new_board.max_w = BOARD_W;
            new_board.max_h = BOARD_H;
            boards.push_back(std::move(new_board));
            
            bool ok = pack_item(rects[i], boards, place_rule, split_rule);
            if (!ok) {
                // Should logically never happen with valid constraints
            }
        }
    }
    
    out_boards = boards.size();
    if (out_boards > 0) {
        out_tie = boards.back().area_left;
    }
    return true;
}

void apply_sort(vector<Rect>& rects, int sort_rule, const vector<double>& noise) {
    if (sort_rule == 0) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double s_a = (long long)a.w * a.h * noise[a.orig_idx];
            double s_b = (long long)b.w * b.h * noise[b.orig_idx];
            return s_a > s_b;
        });
    } else if (sort_rule == 1) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double max_a = std::max(a.w, a.h) * noise[a.orig_idx];
            double max_b = std::max(b.w, b.h) * noise[b.orig_idx];
            if (max_a != max_b) return max_a > max_b;
            double min_a = std::min(a.w, a.h) * noise[a.orig_idx];
            double min_b = std::min(b.w, b.h) * noise[b.orig_idx];
            return min_a > min_b;
        });
    } else if (sort_rule == 2) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double s_a = (a.w + a.h) * noise[a.orig_idx];
            double s_b = (b.w + b.h) * noise[b.orig_idx];
            return s_a > s_b;
        });
    } else if (sort_rule == 3) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double min_a = std::min(a.w, a.h) * noise[a.orig_idx];
            double min_b = std::min(b.w, b.h) * noise[b.orig_idx];
            if (min_a != min_b) return min_a > min_b;
            double max_a = std::max(a.w, a.h) * noise[a.orig_idx];
            double max_b = std::max(b.w, b.h) * noise[b.orig_idx];
            return max_a > max_b;
        });
    } else if (sort_rule == 4) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double min_a = std::min(a.w, a.h);
            double min_b = std::min(b.w, b.h);
            double ratio_a = (min_a == 0 ? 0 : ((double)std::max(a.w, a.h) / min_a)) * noise[a.orig_idx];
            double ratio_b = (min_b == 0 ? 0 : ((double)std::max(b.w, b.h) / min_b)) * noise[b.orig_idx];
            return ratio_a > ratio_b;
        });
    } else if (sort_rule == 5) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double area_a = (long long)a.w * a.h * noise[a.orig_idx];
            double area_b = (long long)b.w * b.h * noise[b.orig_idx];
            if (area_a != area_b) return area_a > area_b;
            double max_a = std::max(a.w, a.h) * noise[a.orig_idx];
            double max_b = std::max(b.w, b.h) * noise[b.orig_idx];
            return max_a > max_b;
        });
    }
}

fn make_ccw(vertices: &mut Vec<(f64, f64)>) {
    let mut area = 0.0;
    let n = vertices.len();
    for i in 0..n {
        let j = (i + 1) % n;
        area += vertices[i].0 * vertices[j].1 - vertices[j].0 * vertices[i].1;

fn point_on_segment(px: f64, py: f64, x1: f64, y1: f64, x2: f64, y2: f64) -> bool {

fn polygon_inside(poly_a: &Polygon, poly_b: &Polygon) -> bool {

fn rotate_polygon(poly: &Polygon, rad: f64) -> Polygon {

                if (nr >= 0 && nr < H && nc >= 0 && nc < W) {
                    nIdx[count++] = nr * W + nc;
                }
            }
            return count;
        }

        static bool HasUnknowns(int clueIdx) {
            int r = clueIdx / W;
            int c = clueIdx % W;
            for(int i=0; i<8; i++) {
                int nr = r + dr[i];
                int nc = c + dc[i];
                if (nr >= 0 && nr < H && nc >= 0 && nc < W) {
                    if (board[nr * W + nc] == UNKNOWN) return true;
                }
            }
            return false;
        }

        static void WakeUpClues(int cellIdx) {
            int[] nIdx = new int[8];
            int nans = GetNeighbors(cellIdx, nIdx);
            for(int j=0; j<nans; j++) {
                int ni = nIdx[j];
                if (board[ni] <= 8) {
                    if (!inQueue[ni]) {
                        activeClues.Enqueue(ni);
                        inQueue[ni] = true;
                    }
                    if (!inPairwiseQueue[ni]) {
                        pairwiseQueue.Enqueue(ni);
                        inPairwiseQueue[ni] = true;
                    }
                }
            }
        }
        
        static bool RunSimpleRules() {
            bool changed = false;
            int[] nIdx = new int[8];
            
            while(activeClues.Count > 0) {

                int nans = GetNeighbors(clueIdx, nIdx);
                int nUnknowns = 0;
                int nMines = 0;
                
                for(int j=0; j<nans; j++) {
                    byte v = board[nIdx[j]];
                    if (v == UNKNOWN) nUnknowns++;
                    else if (v == FLAG) nMines++;
                }
                
                if (nUnknowns == 0) continue;
                
                int clueVal = board[clueIdx];
                if (clueVal == nMines) {
                    for(int j=0; j<nans; j++) {
                        int ni = nIdx[j];
                        if (board[ni] == UNKNOWN) {
                            board[ni] = SAFE;
                            safeCellsOut.Add(ni);
                            changed = true;
                            WakeUpClues(ni);
                        }
                    }
                } else if (clueVal == nMines + nUnknowns) {
                    for(int j=0; j<nans; j++) {

                        if (board[ni] == UNKNOWN) {
                            board[ni] = FLAG;
                            changed = true;

                        }
                    }
                }
            }

        }

        static bool RunPairwiseRules() {
            bool changed = false;

            int[] nIdx2 = new int[8];
            int[] adjClues = new int[64];
            int[] uNeighbors = new int[8];

                int unk1 = 0, mines1 = 0;
                int uMask1 = 0;
                for(int j=0; j<nans1; j++) {
                    if (board[nIdx1[j]] == UNKNOWN) { unk1++; uMask1 |= (1 << j); }
                    else if (board[nIdx1[j]] == FLAG) { mines1++; }

                if (unk1 == 0) continue;
                int V1 = board[c1] - mines1;

                for(int j=0; j<nans1; j++) {

                        int unans = GetNeighbors(uIdx, uNeighbors);
                        for(int k=0; k<unans; k++) {
                            int cand = uNeighbors[k];
                            if (cand != c1 && board[cand] <= 8) {
                                bool found = false;
                                for(int m=0; m<adjCount; m++) {
                                    if (adjClues[m] == cand) { found = true; break; }
                                }
                                if (!found) adjClues[adjCount++] = cand;
                            }
                        }
                    }
                }
                
                bool localChanged = false;
                for(int m=0; m<adjCount; m++) {
                    int c2 = adjClues[m];
                    if (!HasUnknowns(c2)) continue;
                    
                    int nans2 = GetNeighbors(c2, nIdx2);
                    int unk2 = 0, mines2 = 0;
                    for(int j=0; j<nans2; j++) {
                        if (board[nIdx2[j]] == UNKNOWN) unk2++;
                        else if (board[nIdx2[j]] == FLAG) mines2++;
                    }
                    if (unk2 == 0) continue;
                    int V2 = board[c2] - mines2;
                    
                    int O_count = 0, U1_only = 0, U2_only = 0;
                    for(int j=0; j<nans1; j++) {
                        if (board[nIdx1[j]] == UNKNOWN) {
                            bool in2 = false;
                            for(int k=0; k<nans2; k++) {
                                if (nIdx1[j] == nIdx2[k]) { in2 = true; break; }
                            }
                            if (in2) O_count++;
                            else U1_only++;
                        }
                    }
                    for(int k=0; k<nans2; k++) {
                        if (board[nIdx2[k]] == UNKNOWN) {
                            bool in1 = false;
                            for(int j=0; j<nans1; j++) {
                                if (nIdx2[k] == nIdx1[j]) { in1 = true; break; }
                            }
                            if (!in1) U2_only++;
                        }
                    }
                    
                    if (O_count == 0) continue;
                    
                    int max_O_from_1 = Math.Min(O_count, V1);
                    int min_O_from_1 = Math.Max(0, V1 - U1_only);
                    
                    if (V2 - max_O_from_1 == U2_only && U2_only > 0) {
                        for(int k=0; k<nans2; k++) {
                            int idx = nIdx2[k];
                            if (board[idx] == UNKNOWN) {
                                bool in1 = false;
                                for(int j=0; j<nans1; j++) if (idx == nIdx1[j]) { in1 = true; break; }
                                if (!in1) { board[idx] = FLAG; localChanged = true; WakeUpClues(idx); }
                            }
                        }
                    }
                    if (V2 - min_O_from_1 == 0 && U2_only > 0) {
                        for(int k=0; k<nans2; k++) {
                            int idx = nIdx2[k];
                            if (board[idx] == UNKNOWN) {
                                bool in1 = false;
                                for(int j=0; j<nans1; j++) if (idx == nIdx1[j]) { in1 = true; break; }
                                if (!in1) { board[idx] = SAFE; safeCellsOut.Add(idx); localChanged = true; WakeUpClues(idx); }
                            }
                        }
                    }
                    
                    int max_O_from_2 = Math.Min(O_count, V2);
                    int min_O_from_2 = Math.Max(0, V2 - U2_only);
                    
                    if (V1 - max_O_from_2 == U1_only && U1_only > 0) {
                        for(int j=0; j<nans1; j++) {
                            int idx = nIdx1[j];
                            if (board[idx] == UNKNOWN) {
                                bool in2 = false;
                                for(int k=0; k<nans2; k++) if (idx == nIdx2[k]) { in2 = true; break; }
                                if (!in2) { board[idx] = FLAG; localChanged = true; WakeUpClues(idx); }
                            }
                        }
                    }
                    if (V1 - min_O_from_2 == 0 && U1_only > 0) {
                        for(int j=0; j<nans1; j++) {
                            int idx = nIdx1[j];
                            if (board[idx] == UNKNOWN) {
                                bool in2 = false;
                                for(int k=0; k<nans2; k++) if (idx == nIdx2[k]) { in2 = true; break; }
                                if (!in2) { board[idx] = SAFE; safeCellsOut.Add(idx); localChanged = true; WakeUpClues(idx); }
                            }
                        }
                    }
                    
                    if (localChanged) {
                        changed = true;
                        break; 
                    }
                }
                
                if (localChanged) return true;
            }
            return changed;
        }

        static bool RunCSPRules() {
            bool overallChanged = false;
            
            List<int> activeCSPClues = new List<int>();
            foreach(int c in allClues) {
                if (HasUnknowns(c)) activeCSPClues.Add(c);
            }
            if (activeCSPClues.Count == 0) return false;
            
            Dictionary<int, List<int>> clueUnknowns = new Dictionary<int, List<int>>();
            foreach(int c in activeCSPClues) {
                clueUnknowns[c] = new List<int>();
                int[] nIdx = new int[8];
                int nans = GetNeighbors(c, nIdx);
                for(int i=0; i<nans; i++) {
                    if (board[nIdx[i]] == UNKNOWN) clueUnknowns[c].Add(nIdx[i]);
                }
            }
            
            List<List<int>> components = new List<List<int>>();
            HashSet<int> visitedClues = new HashSet<int>();
            
            foreach(int startClue in activeCSPClues) {
                if (visitedClues.Contains(startClue)) continue;
                
                List<int> compClues = new List<int>();
                Queue<int> q = new Queue<int>();
                q.Enqueue(startClue);
                visitedClues.Add(startClue);
                
                while(q.Count > 0) {
                    int curr = q.Dequeue();
                    compClues.Add(curr);
                    
                    foreach(int other in activeCSPClues) {
                        if (!visitedClues.Contains(other)) {
                            bool share = false;
                            foreach(int u in clueUnknowns[curr]) {
                                if (clueUnknowns[other].Contains(u)) { share = true; break; }
                            }
                            if (share) {
                                visitedClues.Add(other);
                                q.Enqueue(other);
                            }
                        }
                    }
                }
                components.Add(compClues);
            }
            
            foreach(List<int> compClues in components) {
                HashSet<int> compUnknownsSet = new HashSet<int>();
                foreach(int c in compClues) {
                    foreach(int u in clueUnknowns[c]) compUnknownsSet.Add(u);
                }
                
                List<int> compUnknowns = new List<int>(compUnknownsSet);
                
                // Adaptive logic: only run full CSP if component is reasonably small
                if (compUnknowns.Count > 22) continue; 
                
                int[] validAssignments = new int[compUnknowns.Count];
                int totalValid = 0;
                bool[] currentAssigned = new bool[compUnknowns.Count];
                
                int[] cIdxs = compClues.ToArray();
                int[] cTarget = new int[cIdxs.Length];
                for(int i=0; i<cIdxs.Length; i++) {
                    int mines = 0;
                    int[] nIdx = new int[8];
                    int nans = GetNeighbors(cIdxs[i], nIdx);
                    for(int j=0; j<nans; j++) if (board[nIdx[j]] == FLAG) mines++;
                    cTarget[i] = board[cIdxs[i]] - mines;
                }
                
                List<int>[] cUnkIdxs = new List<int>[cIdxs.Length];
                for(int i=0; i<cIdxs.Length; i++) {
                    cUnkIdxs[i] = new List<int>();
                    foreach(int u in clueUnknowns[cIdxs[i]]) {
                        cUnkIdxs[i].Add(compUnknowns.IndexOf(u));
                    }
                }
                
                void Solve(int uIdx) {
                    for(int i=0; i<cIdxs.Length; i++) {
                        int assignedMines = 0;
                        int unassigned = 0;
                        foreach(int cu in cUnkIdxs[i]) {
                            if (cu < uIdx) {
                                if (currentAssigned[cu]) assignedMines++;
                            } else {
                                unassigned++;
                            }
                        }
                        if (assignedMines > cTarget[i]) return;
                        if (assignedMines + unassigned < cTarget[i]) return;
                    }
                    
                    if (uIdx == compUnknowns.Count) {
                        totalValid++;
                        for(int i=0; i<compUnknowns.Count; i++) {
                            if (currentAssigned[i]) validAssignments[i]++;
                        }
                        return;
                    }
                    
                    currentAssigned[uIdx] = false;
                    Solve(uIdx + 1);
                    currentAssigned[uIdx] = true;
                    Solve(uIdx + 1);
                }
                
                Solve(0);
                
                if (totalValid > 0) {
                    for(int i=0; i<compUnknowns.Count; i++) {
                        if (validAssignments[i] == 0) {
                            int u = compUnknowns[i];
                            if (board[u] == UNKNOWN) {
                                board[u] = SAFE;
                                safeCellsOut.Add(u);
                                overallChanged = true;
                                WakeUpClues(u);
                            }
                        } else if (validAssignments[i] == totalValid) {
                            int u = compUnknowns[i];
                            if (board[u] == UNKNOWN) {
                                board[u] = FLAG;
                                overallChanged = true;
                                WakeUpClues(u);
                            }
                        }
                    }
                }
                
                if (overallChanged) return true; 
            }
            
            return overallChanged;
        }

        static void Main(string[] args)
        {
            string line = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line)) return;
            string[] parts = line.Split(new char[]{' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
            H = int.Parse(parts[0]);
            W = int.Parse(parts[1]);
            
            board = new byte[H * W];
            inQueue = new bool[H * W];
            inPairwiseQueue = new bool[H * W];
            
            for (int r = 0; r < H; r++) {
                string rowLine = Console.ReadLine();
                for (int c = 0; c < W; c++) {
                    char ch = rowLine[c];
                    int idx = r * W + c;
                    if (ch == ' ') board[idx] = UNKNOWN;
                    else if (ch == '*') board[idx] = FLAG;
                    else if (ch >= '0' && ch <= '8') {
                        board[idx] = (byte)(ch - '0');
                        allClues.Add(idx);
                        
                        activeClues.Enqueue(idx);
                        inQueue[idx] = true;
                        
                        pairwiseQueue.Enqueue(idx);
                        inPairwiseQueue[idx] = true;
                    }
                }
            }
            
            bool changed = true;
            while(changed) {
                changed = false;
                
                if (RunSimpleRules()) { changed = true; continue; }
                if (RunPairwiseRules()) { changed = true; continue; }
                if (RunCSPRules()) { changed = true; continue; }
            }
            
            Console.WriteLine(safeCellsOut.Count);
            using (StreamWriter sw = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536)) {
                foreach(int idx in safeCellsOut) {
                    int r = idx / W;
                    int c = idx % W;
                    sw.Write(c);
                    sw.Write(' ');
                    sw.WriteLine(r);
                }
            }
        }
