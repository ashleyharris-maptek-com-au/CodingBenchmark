#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
#include <algorithm>
#include <map>

using namespace std;

struct Point {
    double x, y, z;
    Point() : x(0), y(0), z(0) {}
    Point(double x, double y, double z) : x(x), y(y), z(z) {}
    Point operator-(const Point& o) const { return Point(x - o.x, y - o.y, z - o.z); }
    Point operator+(const Point& o) const { return Point(x + o.x, y + o.y, z + o.z); }
    Point operator*(double s) const { return Point(x * s, y * s, z * s); }
    double length() const { return sqrt(x*x + y*y + z*z); }
    void normalize() {
        double l = length();
        if (l > 1e-12) { x /= l; y /= l; z /= l; }
    }
};

double dot(const Point& a, const Point& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Point cross(const Point& a, const Point& b) {
    return Point(a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x);
}

struct Face {
    int a, b, c;
    Point normal;
    double d;
};

vector<Face> build_convex_hull(const vector<Point>& pts) {
    int n = pts.size();
    if (n < 4) return {};
    
    int p0 = 0, p1 = -1, p2 = -1, p3 = -1;
    double max_d = 0;
    for (int i = 1; i < n; i++) {
        double d = (pts[i] - pts[p0]).length();
        if (d > max_d) { max_d = d; p1 = i; }
    }
    if (p1 == -1) return {};
    
    max_d = 0;
    for (int i = 0; i < n; i++) {
        if (i == p0 || i == p1) continue;
        Point v = pts[i] - pts[p0];
        Point dir = pts[p1] - pts[p0]; dir.normalize();
        double proj = dot(v, dir);
        Point perp = v - dir * proj;
        double d = perp.length();
        if (d > max_d) { max_d = d; p2 = i; }
    }
    if (p2 == -1) return {};
    
    max_d = 0;
    Point n0 = cross(pts[p1] - pts[p0], pts[p2] - pts[p0]); n0.normalize();
    for (int i = 0; i < n; i++) {
        if (i == p0 || i == p1 || i == p2) continue;
        double d = abs(dot(n0, pts[i] - pts[p0]));
        if (d > max_d) { max_d = d; p3 = i; }
    }
    if (p3 == -1) return {};
    
    vector<Face> faces;
    auto add_face = [&](int a, int b, int c, int d_idx) {
        Face f; f.a = a; f.b = b; f.c = c;
        f.normal = cross(pts[b] - pts[a], pts[c] - pts[a]);
        f.normal.normalize();
        if (dot(f.normal, pts[d_idx] - pts[a]) > 0) {
            swap(f.b, f.c);
            f.normal = cross(pts[f.b] - pts[a], pts[f.c] - pts[a]);
            f.normal.normalize();
        }
        f.d = dot(f.normal, pts[a]);
        faces.push_back(f);
    };
    
    add_face(p0, p1, p2, p3);
    add_face(p0, p1, p3, p2);
    add_face(p0, p2, p3, p1);
    add_face(p1, p2, p3, p0);
    
    vector<bool> used(n, false);
    used[p0] = used[p1] = used[p2] = used[p3] = true;
    
    for (int i = 0; i < n; i++) {
        if (used[i]) continue;
        vector<Face> next_faces;
        map<pair<int, int>, int> edge_counts;
        bool visible = false;
        
        for (const auto& f : faces) {
            if (dot(f.normal, pts[i]) > f.d + 1e-9) {
                visible = true;
                edge_counts[{f.a, f.b}]++;
                edge_counts[{f.b, f.c}]++;
                edge_counts[{f.c, f.a}]++;
            } else {
                next_faces.push_back(f);
            }
        }
        
        if (!visible) continue;
        
        for (auto const& [edge, _] : edge_counts) {
            if (edge_counts.find({edge.second, edge.first}) == edge_counts.end()) {
                int u = edge.first;
                int v = edge.second;
                Face f;
                f.a = u; f.b = v; f.c = i;
                f.normal = cross(pts[v] - pts[u], pts[i] - pts[u]);
                f.normal.normalize();
                f.d = dot(f.normal, pts[u]);
                next_faces.push_back(f);
            }
        }
        faces = next_faces;
    }
    
    return faces;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int V;
    double E;
    if (!(cin >> V >> E)) return 0;
    
    vector<Point> pts(V);
    double Xmin = 1e18, Xmax = -1e18, Ymin = 1e18, Ymax = -1e18, Zmin = 1e18, Zmax = -1e18;
    for (int i = 0; i < V; i++) {
        cin >> pts[i].x >> pts[i].y >> pts[i].z;
        Xmin = min(Xmin, pts[i].x); Xmax = max(Xmax, pts[i].x);
        Ymin = min(Ymin, pts[i].y); Ymax = max(Ymax, pts[i].y);
        Zmin = min(Zmin, pts[i].z); Zmax = max(Zmax, pts[i].z);
    }
    
    vector<Face> hull = build_convex_hull(pts);
    if (hull.empty()) {
        cout << 0 << "\n";
        return 0;
    }
    
    // Using Circumradius to ensure no overlap for any orientation
    double R_in = E * sqrt(3.0 / 8.0);
    double R_margin = R_in * 1.000000001 + 1e-9;
    double S = sqrt(2.0) * R_margin;
    
    vector<double> top_Nx, top_Ny, top_Nz_inv, top_K;
    vector<double> bot_Nx, bot_Ny, bot_Nz_inv, bot_K;
    vector<double> vrt_Nx, vrt_Ny, vrt_K;
    
    for (const auto& f : hull) {
        if (f.normal.z > 1e-7) {
            top_Nx.push_back(f.normal.x);
            top_Ny.push_back(f.normal.y);
            top_Nz_inv.push_back(1.0 / f.normal.z);
            top_K.push_back(f.d - R_margin);
        } else if (f.normal.z < -1e-7) {
            bot_Nx.push_back(f.normal.x);
            bot_Ny.push_back(f.normal.y);
            bot_Nz_inv.push_back(1.0 / f.normal.z);
            bot_K.push_back(f.d - R_margin);
        } else {
            vrt_Nx.push_back(f.normal.x);
            vrt_Ny.push_back(f.normal.y);
            vrt_K.push_back(f.d - R_margin);
        }
    }
    
    long long min_i = floor(Xmin / S) - 2;
    long long max_i = ceil(Xmax / S) + 2;
    long long min_j = floor(Ymin / S) - 2;
    long long max_j = ceil(Ymax / S) + 2;
    
    long long num_columns = (max_i - min_i + 1) * (max_j - min_j + 1);
    long long ops_per_offset = num_columns * (long long)hull.size();
    long long max_ops = 500000000; // 5e8 operations limit to stay well within 30s
    int num_offsets = 1;
    
    if (num_columns < 1e12) {
        while ((long long)(num_offsets + 1) * (num_offsets + 1) * (num_offsets + 1) * ops_per_offset <= max_ops && num_offsets < 8) {
            num_offsets++;
        }
    }
    if (num_offsets < 1) num_offsets = 1;
    
    vector<Point> best_centers;
    double offset_step = S / num_offsets;
    
    for (int ox = 0; ox < num_offsets; ox++) {
        for (int oy = 0; oy < num_offsets; oy++) {
            for (int oz = 0; oz < num_offsets; oz++) {
                double dx = ox * offset_step;
                double dy = oy * offset_step;
                double dz = oz * offset_step;
                
                vector<Point> current_centers;
                
                int num_top = top_K.size();
                int num_bot = bot_K.size();
                int num_vrt = vrt_K.size();
                
                for (long long i = min_i; i <= max_i; i++) {
                    double X = i * S + dx;
                    for (long long j = min_j; j <= max_j; j++) {
                        double Y = j * S + dy;
                        
                        bool valid_xy = true;
                        for (int f = 0; f < num_vrt; f++) {
                            if (vrt_K[f] - vrt_Nx[f] * X - vrt_Ny[f] * Y < 0) {
                                valid_xy = false;
                                break;
                            }
                        }
                        if (!valid_xy) continue;
                        
                        double z_max = 1e18;
                        for (int f = 0; f < num_top; f++) {
                            double z = (top_K[f] - top_Nx[f] * X - top_Ny[f] * Y) * top_Nz_inv[f];
                            if (z < z_max) z_max = z;
                        }
                        
                        double z_min = -1e18;
                        for (int f = 0; f < num_bot; f++) {
                            double z = (bot_K[f] - bot_Nx[f] * X - bot_Ny[f] * Y) * bot_Nz_inv[f];
                            if (z > z_min) z_min = z;
                        }
                        
                        if (z_min < Zmin - 2*S) z_min = Zmin - 2*S;
                        if (z_max > Zmax + 2*S) z_max = Zmax + 2*S;
                        
                        if (z_min > z_max) continue;
                        
                        long long k_start = ceil((z_min - dz) / S - 1e-9);
                        long long k_end = floor((z_max - dz) / S + 1e-9);
                        
                        for (long long k = k_start; k <= k_end; k++) {
                            if (abs(i + j + k) % 2 == 0) {
                                current_centers.push_back(Point(X, Y, k * S + dz));
                            }
                        }
                    }
                }
                
                if (current_centers.size() > best_centers.size()) {
                    best_centers = current_centers;
                }
            }
        }
    }
    
    cout << best_centers.size() << "\n";
    cout << fixed << setprecision(6);
    for (const auto& p : best_centers) {
        cout << p.x << " " << p.y << " " << p.z << " 0 0 0\n";
    }
    
    return 0;
}
