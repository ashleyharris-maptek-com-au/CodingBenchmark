#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>
#include <numeric>
#include <random>

using namespace std;
using namespace std::chrono;

double get_time_sec() {
    return duration_cast<duration<double>>(steady_clock::now().time_since_epoch()).count();
}

uint64_t hilbert_index(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int s = 31; s >= 0; s--) {
        uint32_t rx = (x >> s) & 1;
        uint32_t ry = (y >> s) & 1;
        d = (d << 2) | ((rx * 3) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x;
            x = y;
            y = t;
        }
    }
    return d;
}

inline double distance_nodes(int a, int b, const float* x, const float* y) {
    double dx = x[a] - x[b];
    double dy = y[a] - y[b];
    return std::sqrt(dx * dx + dy * dy);
}

void reverse_segment(vector<int>& tour, int start, int end) {
    int n = tour.size();
    while (start != end) {
        int next_start = (start + 1 < n) ? start + 1 : start + 1 - n;
        if (next_start == end) {
            swap(tour[start], tour[end]);
            break;
        }
        swap(tour[start], tour[end]);
        start = next_start;
        end = (end - 1 >= 0) ? end - 1 : end - 1 + n;
    }
}

void apply_2opt(vector<int>& tour, const float* x, const float* y, int window, int max_passes, double start_time, double time_limit) {
    int N = tour.size();
    if (N < 3) return;
    int passes = 0;
    bool improved = true;
    while (improved && passes < max_passes) {
        improved = false;
        passes++;
        for (int i = 0; i < N; i++) {
            if ((i & 1023) == 0 && get_time_sec() - start_time > time_limit) {
                return;
            }
            int a = tour[i];
            int i_next = (i + 1 < N) ? i + 1 : i + 1 - N;
            int b = tour[i_next];
            
            int limit = min(N - 1, window);
            double best_delta = -1e-7;
            int best_j = -1;
            
            double dist_ab = distance_nodes(a, b, x, y);
            
            int j = (i_next + 1 < N) ? i_next + 1 : i_next + 1 - N;
            for (int k = 2; k <= limit; k++) {
                int j_next = (j + 1 < N) ? j + 1 : j + 1 - N;
                int c = tour[j];
                int d = tour[j_next];
                
                double dist_cd = distance_nodes(c, d, x, y);
                double dist_ac = distance_nodes(a, c, x, y);
                double dist_bd = distance_nodes(b, d, x, y);
                
                double delta = dist_ac + dist_bd - dist_ab - dist_cd;
                if (delta < best_delta) {
                    best_delta = delta;
                    best_j = j;
                }
                j = j_next;
            }
            
            if (best_j != -1) {
                reverse_segment(tour, i_next, best_j);
                improved = true;
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    double start_time = get_time_sec();
    
    int N;
    if (!(cin >> N)) return 0;
    if (N <= 0) return 0;
    
    vector<float> x(N), y(N);
    if (!(cin >> x[0] >> y[0])) return 0;
    float min_x = x[0], max_x = x[0];
    float min_y = y[0], max_y = y[0];
    
    int actual_N = 1;
    for (int i = 1; i < N; i++) {
        if (!(cin >> x[i] >> y[i])) break;
        if (x[i] < min_x) min_x = x[i];
        if (x[i] > max_x) max_x = x[i];
        if (y[i] < min_y) min_y = y[i];
        if (y[i] > max_y) max_y = y[i];
        actual_N++;
    }
    N = actual_N;
    if (N < x.size()) {
        x.resize(N);
        y.resize(N);
    }
    
    if (N == 1) {
        cout << 0 << "\n";
        return 0;
    }
    
    float range_x = max_x - min_x;
    float range_y = max_y - min_y;
    float max_range = max(range_x, range_y);
    if (max_range == 0) max_range = 1.0f;
    
    vector<pair<uint64_t, int>> h_points(N);
    for (int i = 0; i < N; i++) {
        float norm_x = (x[i] - min_x) / max_range;
        float norm_y = (y[i] - min_y) / max_range;
        if (norm_x > 1.0f) norm_x = 1.0f;
        if (norm_y > 1.0f) norm_y = 1.0f;
        if (norm_x < 0.0f) norm_x = 0.0f;
        if (norm_y < 0.0f) norm_y = 0.0f;
        uint32_t sx = (uint32_t)(norm_x * 2147483647.0f);
        uint32_t sy = (uint32_t)(norm_y * 2147483647.0f);
        h_points[i] = {hilbert_index(sx, sy), i};
    }
    
    sort(h_points.begin(), h_points.end());
    
    vector<int> tour(N);
    for (int i = 0; i < N; i++) {
        tour[i] = h_points[i].second;
    }
    
    int window = 50;
    if (N <= 5000) window = N;
    else if (N <= 20000) window = 1000;
    else if (N <= 100000) window = 400;
    else if (N <= 1000000) window = 200;
    else window = 100;
    
    // First fast pass for larger N to resolve local tangles quickly
    if (N > 1000) {
        apply_2opt(tour, x.data(), y.data(), min(20, window), 100, start_time, 290.0);
    }
    
    // Main 2-opt pass
    apply_2opt(tour, x.data(), y.data(), window, 100, start_time, 290.0);
    
    // Extreme optimization for small sets using multi-start local search
    if (N <= 1000) {
        vector<int> best_tour = tour;
        double best_len = 0;
        for (int i = 0; i < N; i++) {
            best_len += distance_nodes(best_tour[i], best_tour[(i+1)%N], x.data(), y.data());
        }
        
        mt19937 rng(1337);
        int max_retries = (N <= 100) ? 5000 : 200;
        for (int r = 0; r < max_retries; r++) {
            if (get_time_sec() - start_time > 280.0) break;
            vector<int> curr = tour;
            shuffle(curr.begin(), curr.end(), rng);
            apply_2opt(curr, x.data(), y.data(), N, 50, start_time, 280.0);
            
            double curr_len = 0;
            for (int i = 0; i < N; i++) {
                curr_len += distance_nodes(curr[i], curr[(i+1)%N], x.data(), y.data());
            }
            if (curr_len < best_len) {
                best_len = curr_len;
                best_tour = curr;
            }
        }
        tour = best_tour;
    }
    
    // Rotate tour to start at city 0
    int zero_idx = -1;
    for (int i = 0; i < N; i++) {
        if (tour[i] == 0) {
            zero_idx = i;
            break;
        }
    }
    
    for (int i = 0; i < N; i++) {
        int idx = (zero_idx + i < N) ? zero_idx + i : zero_idx + i - N;
        cout << tour[idx];
        if (i != N - 1) cout << ' ';
    }
    cout << "\n";
    
    return 0;
}
