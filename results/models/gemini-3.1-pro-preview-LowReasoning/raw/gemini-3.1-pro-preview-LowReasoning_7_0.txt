using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    Vec3 operator/(double s) const { return {x/s, y/s, z/s}; }
};

double dot(const Vec3& a, const Vec3& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}
double length_squared(const Vec3& v) { return dot(v, v); }
double length(const Vec3& v) { return sqrt(length_squared(v)); }
Vec3 normalize(const Vec3& v) { double l = length(v); return l > 0 ? v/l : v; }

struct Triangle { Vec3 v0, v1, v2; };

struct AABB {
    Vec3 min_p, max_p;
    AABB() : min_p{1e30, 1e30, 1e30}, max_p{-1e30, -1e30, -1e30} {}
    void expand(const Vec3& p) {
        min_p.x = min(min_p.x, p.x); min_p.y = min(min_p.y, p.y); min_p.z = min(min_p.z, p.z);
        max_p.x = max(max_p.x, p.x); max_p.y = max(max_p.y, p.y); max_p.z = max(max_p.z, p.z);
    }
    bool intersects(const AABB& b) const {
        if (max_p.x < b.min_p.x || min_p.x > b.max_p.x) return false;
        if (max_p.y < b.min_p.y || min_p.y > b.max_p.y) return false;
        if (max_p.z < b.min_p.z || min_p.z > b.max_p.z) return false;
        return true;
    }
    bool intersect_ray(const Vec3& ro, const Vec3& rd_inv) const {
        double t1 = (min_p.x - ro.x) * rd_inv.x;
        double t2 = (max_p.x - ro.x) * rd_inv.x;
        double tmin = min(t1, t2); double tmax = max(t1, t2);
        
        t1 = (min_p.y - ro.y) * rd_inv.y;
        t2 = (max_p.y - ro.y) * rd_inv.y;
        tmin = max(tmin, min(t1, t2)); tmax = min(tmax, max(t1, t2));
        
        t1 = (min_p.z - ro.z) * rd_inv.z;
        t2 = (max_p.z - ro.z) * rd_inv.z;
        tmin = max(tmin, min(t1, t2)); tmax = min(tmax, max(t1, t2));
        
        return tmax >= max(0.0, tmin);
    }
};

struct Mesh {
    vector<Vec3> vertices;
    vector<int> indices;
    Triangle get_triangle(int idx) const {
        return {vertices[indices[idx*3]], vertices[indices[idx*3+1]], vertices[indices[idx*3+2]]};
    }
    AABB get_aabb(int idx) const {
        AABB b; Triangle t = get_triangle(idx);
        b.expand(t.v0); b.expand(t.v1); b.expand(t.v2);
        return b;
    }
};

struct BVHNode {
    AABB aabb;
    int left, right;
    int start, end;
    bool is_leaf;
};

struct BVH {
    vector<BVHNode> nodes;
    vector<int> tri_indices;
    const Mesh* mesh;

    void build(const Mesh& m) {
        mesh = &m;
        int num_tris = m.indices.size() / 3;
        tri_indices.resize(num_tris);
        for(int i=0; i<num_tris; ++i) tri_indices[i] = i;
        nodes.reserve(num_tris * 2);
        build_recursive(0, num_tris);
    }

    int build_recursive(int start, int end) {
        BVHNode node;
        node.start = start; node.end = end;
        
        for(int i=start; i<end; ++i) {
            Triangle t = mesh->get_triangle(tri_indices[i]);
            node.aabb.expand(t.v0); node.aabb.expand(t.v1); node.aabb.expand(t.v2);
        }

        if (end - start <= 4) {
            node.is_leaf = true; node.left = node.right = -1;
            nodes.push_back(node);
            return nodes.size() - 1;
        }

        node.is_leaf = false;
        Vec3 size = node.aabb.max_p - node.aabb.min_p;
        int axis = 0;
        if (size.y > size.x && size.y > size.z) axis = 1;
        if (size.z > size.x && size.z > size.y) axis = 2;

        auto comp = [&](int a, int b) {
            Triangle ta = mesh->get_triangle(a); Triangle tb = mesh->get_triangle(b);
            Vec3 ca = ta.v0 + ta.v1 + ta.v2; Vec3 cb = tb.v0 + tb.v1 + tb.v2;
            if (axis == 0) return ca.x < cb.x;
            if (axis == 1) return ca.y < cb.y;
            return ca.z < cb.z;
        };

        int mid = start + (end - start) / 2;
        nth_element(tri_indices.begin() + start, tri_indices.begin() + mid, tri_indices.begin() + end, comp);

        int curr_idx = nodes.size();
        nodes.push_back(node);
        int left = build_recursive(start, mid);
        int right = build_recursive(mid, end);
        nodes[curr_idx].left = left; nodes[curr_idx].right = right;
        return curr_idx;
    }

    void query_aabb(int node_idx, const AABB& box, vector<int>& out) const {
        if (!nodes[node_idx].aabb.intersects(box)) return;
        if (nodes[node_idx].is_leaf) {
            for(int i=nodes[node_idx].start; i<nodes[node_idx].end; ++i) out.push_back(tri_indices[i]);
        } else {
            query_aabb(nodes[node_idx].left, box, out);
            query_aabb(nodes[node_idx].right, box, out);
        }
    }
};

bool sat_test(const Vec3& axis, const Vec3 p1[3], const Vec3 p2[3]) {
    double l2 = length_squared(axis);
    if (l2 < 1e-14) return false;
    double l = sqrt(l2);
    Vec3 naxis = axis / l;
    double min1 = 1e30, max1 = -1e30, min2 = 1e30, max2 = -1e30;
    for(int i=0; i<3; ++i) {
        double p = dot(p1[i], naxis);
        if (p < min1) min1 = p; if (p > max1) max1 = p;
        double q = dot(p2[i], naxis);
        if (q < min2) min2 = q; if (q > max2) max2 = q;
    }
    return (max1 < min2 - 1e-7 || max2 < min1 - 1e-7);
}

bool tri_tri_intersect(const Triangle& t1, const Triangle& t2) {
    Vec3 p1[3] = {t1.v0, t1.v1, t1.v2};
    Vec3 p2[3] = {t2.v0, t2.v1, t2.v2};
    Vec3 e1[3] = {p1[1]-p1[0], p1[2]-p1[1], p1[0]-p1[2]};
    Vec3 e2[3] = {p2[1]-p2[0], p2[2]-p2[1], p2[0]-p2[2]};
    Vec3 n1 = cross(e1[0], e1[1]);
    Vec3 n2 = cross(e2[0], e2[1]);
    
    if (sat_test(n1, p1, p2)) return false;
    if (sat_test(n2, p1, p2)) return false;
    for(int i=0; i<3; ++i) {
        if (sat_test(cross(e1[i], n1), p1, p2)) return false;
        if (sat_test(cross(e2[i], n2), p1, p2)) return false;
        for(int j=0; j<3; ++j) {
            if (sat_test(cross(e1[i], e2[j]), p1, p2)) return false;
        }
    }
    return true;
}

struct Plane {
    Vec3 n; double d;
    Plane(Vec3 normal, double dist) {
        double l = length(normal);
        n = normal / l; d = dist / l;
    }
};

struct Polygon { vector<Vec3> vertices; };

void clean_polygon(Polygon& p) {
    if (p.vertices.empty()) return;
    vector<Vec3> clean;
    clean.push_back(p.vertices[0]);
    for(size_t i=1; i<p.vertices.size(); ++i) {
        if (length_squared(p.vertices[i] - clean.back()) > 1e-12) clean.push_back(p.vertices[i]);
    }
    if (clean.size() > 1 && length_squared(clean.front() - clean.back()) <= 1e-12) clean.pop_back();
    p.vertices = clean;
}

void split_polygon(const Polygon& poly, const Plane& plane, Polygon& front, Polygon& back) {
    int n = poly.vertices.size();
    vector<double> dists(n);
    for(int i=0; i<n; ++i) dists[i] = dot(poly.vertices[i], plane.n) - plane.d;
    
    for(int i=0; i<n; ++i) {
        int j = (i + 1) % n;
        double d1 = dists[i], d2 = dists[j];
        
        if (d1 >= -1e-9) front.vertices.push_back(poly.vertices[i]);
        if (d1 <= 1e-9) back.vertices.push_back(poly.vertices[i]);
        
        if ((d1 > 1e-9 && d2 < -1e-9) || (d1 < -1e-9 && d2 > 1e-9)) {
            double t = d1 / (d1 - d2);
            Vec3 p = poly.vertices[i] + (poly.vertices[j] - poly.vertices[i]) * t;
            front.vertices.push_back(p);
            back.vertices.push_back(p);
        }
    }
    clean_polygon(front); clean_polygon(back);
}

bool plane_intersects_polygon(const Polygon& poly, const Plane& plane) {
    bool has_front = false, has_back = false;
    for(const Vec3& v : poly.vertices) {
        double d = dot(v, plane.n) - plane.d;
        if (d > 1e-9) has_front = true;
        if (d < -1e-9) has_back = true;
        if (has_front && has_back) return true;
    }
    return false;
}

bool ray_tri(const Vec3& ro, const Vec3& rd, const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    Vec3 edge1 = v1 - v0, edge2 = v2 - v0;
    Vec3 h = cross(rd, edge2);
    double a = dot(edge1, h);
    if (a > -1e-8 && a < 1e-8) return false;
    double f = 1.0 / a;
    Vec3 s = ro - v0;
    double u = f * dot(s, h);
    if (u < 0.0 || u > 1.0) return false;
    Vec3 q = cross(s, edge1);
    double v = f * dot(rd, q);
    if (v < 0.0 || u + v > 1.0) return false;
    double t = f * dot(edge2, q);
    return t > 1e-8;
}

Vec3 random_dir() {
    static mt19937 g(12345);
    uniform_real_distribution<double> dist(-1.0, 1.0);
    while (true) {
        Vec3 v = {dist(g), dist(g), dist(g)};
        double l2 = length_squared(v);
        if (l2 > 0.1 && l2 <= 1.0) return v / sqrt(l2);
    }
}

bool is_inside(const BVH& bvh, const Mesh& mesh, const Vec3& c) {
    Vec3 dir = random_dir();
    Vec3 rd_inv = {1.0/(abs(dir.x)<1e-9?1e-9:dir.x), 1.0/(abs(dir.y)<1e-9?1e-9:dir.y), 1.0/(abs(dir.z)<1e-9?1e-9:dir.z)};
    int count = 0;
    
    vector<int> stack; stack.push_back(0);
    while (!stack.empty()) {
        int node = stack.back(); stack.pop_back();
        if (!bvh.nodes[node].aabb.intersect_ray(c, rd_inv)) continue;
        if (bvh.nodes[node].is_leaf) {
            for(int i=bvh.nodes[node].start; i<bvh.nodes[node].end; ++i) {
                Triangle tri = mesh.get_triangle(bvh.tri_indices[i]);
                if (ray_tri(c, dir, tri.v0, tri.v1, tri.v2)) count++;
            }
        } else {
            stack.push_back(bvh.nodes[node].left);
            stack.push_back(bvh.nodes[node].right);
        }
    }
    return (count % 2) != 0;
}

struct GridCell {
    int cx, cy, cz, index;
    bool operator<(const GridCell& o) const {
        if (cx != o.cx) return cx < o.cx;
        if (cy != o.cy) return cy < o.cy;
        return cz < o.cz;
    }
};

bool point_on_segment(const Vec3& p, const Vec3& a, const Vec3& b) {
    Vec3 ab = b - a;
    double l2 = length_squared(ab);
    if (l2 < 1e-14) return false;
    double t = dot(p - a, ab) / l2;
    if (t <= 1e-5 || t >= 1.0 - 1e-5) return false;
    Vec3 proj = a + ab * t;
    return length_squared(p - proj) < 1e-12;
}

struct TriData {
    int i0, i1, i2, c0, c1, c2;
    bool operator<(const TriData& o) const {
        if (c0 != o.c0) return c0 < o.c0;
        if (c1 != o.c1) return c1 < o.c1;
        return c2 < o.c2;
    }
    bool operator==(const TriData& o) const { return c0 == o.c0 && c1 == o.c1 && c2 == o.c2; }
};

TriData make_tridata(int i0, int i1, int i2) {
    TriData d; d.i0 = i0; d.i1 = i1; d.i2 = i2;
    if (i0 < i1 && i0 < i2) { d.c0 = i0; d.c1 = i1; d.c2 = i2; }
    else if (i1 < i0 && i1 < i2) { d.c0 = i1; d.c1 = i2; d.c2 = i0; }
    else { d.c0 = i2; d.c1 = i0; d.c2 = i1; }
    return d;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    size_t VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh meshA, meshB;
    meshA.vertices.resize(VA);
    for(size_t i=0; i<VA; ++i) cin >> meshA.vertices[i].x >> meshA.vertices[i].y >> meshA.vertices[i].z;
    meshA.indices.resize(FA * 3);
    for(size_t i=0; i<FA*3; ++i) cin >> meshA.indices[i];

    meshB.vertices.resize(VB);
    for(size_t i=0; i<VB; ++i) cin >> meshB.vertices[i].x >> meshB.vertices[i].y >> meshB.vertices[i].z;
    meshB.indices.resize(FB * 3);
    for(size_t i=0; i<FB*3; ++i) cin >> meshB.indices[i];

    BVH bvhA, bvhB;
    bvhA.build(meshA); bvhB.build(meshB);

    mt19937 g_rand(42);
    vector<Polygon> all_polys;

    auto process_mesh = [&](const Mesh& mesh1, const BVH& b1, const Mesh& mesh2, const BVH& b2, bool is_A) {
        for(size_t i=0; i<mesh1.indices.size()/3; ++i) {
            Triangle t1 = mesh1.get_triangle(i);
            Vec3 n1 = normalize(cross(t1.v1 - t1.v0, t1.v2 - t1.v0));
            AABB box = mesh1.get_aabb(i);
            
            vector<int> cands;
            b2.query_aabb(0, box, cands);
            
            vector<Plane> split_planes;
            for(int idx2 : cands) {
                Triangle t2 = mesh2.get_triangle(idx2);
                if (tri_tri_intersect(t1, t2)) {
                    Vec3 n2 = normalize(cross(t2.v1 - t2.v0, t2.v2 - t2.v0));
                    if (abs(dot(n1, n2)) > 1.0 - 1e-6) {
                        Vec3 d0 = normalize(t2.v1 - t2.v0);
                        Vec3 d1 = normalize(t2.v2 - t2.v1);
                        Vec3 d2 = normalize(t2.v0 - t2.v2);
                        split_planes.push_back(Plane(cross(d0, n2), dot(cross(d0, n2), t2.v0)));
                        split_planes.push_back(Plane(cross(d1, n2), dot(cross(d1, n2), t2.v1)));
                        split_planes.push_back(Plane(cross(d2, n2), dot(cross(d2, n2), t2.v2)));
                    } else {
                        split_planes.push_back(Plane(n2, dot(n2, t2.v0)));
                    }
                }
            }
            
            shuffle(split_planes.begin(), split_planes.end(), g_rand);
            vector<Polygon> fragments = { Polygon{{t1.v0, t1.v1, t1.v2}} };
            
            for(const auto& pl : split_planes) {
                vector<Polygon> next_frag;
                for(const auto& poly : fragments) {
                    if (!plane_intersects_polygon(poly, pl)) {
                        next_frag.push_back(poly);
                        continue;
                    }
                    Polygon front, back;
                    split_polygon(poly, pl, front, back);
                    if (front.vertices.size() >= 3) next_frag.push_back(front);
                    if (back.vertices.size() >= 3) next_frag.push_back(back);
                }
                fragments = next_frag;
            }
            
            for(const auto& poly : fragments) {
                Vec3 c = {0,0,0};
                for(const auto& v : poly.vertices) c = c + v;
                c = c / poly.vertices.size();
                Vec3 c_test = c + n1 * 1e-5;
                if (!is_inside(b2, mesh2, c_test)) all_polys.push_back(poly);
            }
        }
    };

    process_mesh(meshA, bvhA, meshB, bvhB, true);
    process_mesh(meshB, bvhB, meshA, bvhA, false);

    vector<Vec3> raw_v;
    for(const auto& p : all_polys) for(const auto& v : p.vertices) raw_v.push_back(v);

    double eps = 1e-5;
    vector<GridCell> grid(raw_v.size());
    for(int i=0; i<raw_v.size(); ++i) {
        grid[i] = {(int)floor(raw_v[i].x/eps), (int)floor(raw_v[i].y/eps), (int)floor(raw_v[i].z/eps), i};
    }
    sort(grid.begin(), grid.end());

    vector<Vec3> out_v;
    vector<int> remap(raw_v.size(), -1);

    for(int i=0; i<grid.size(); ++i) {
        int orig_idx = grid[i].index;
        if (remap[orig_idx] != -1) continue;
        
        int new_idx = out_v.size();
        out_v.push_back(raw_v[orig_idx]);
        remap[orig_idx] = new_idx;
        
        int cx = grid[i].cx, cy = grid[i].cy, cz = grid[i].cz;
        for(int dx=-1; dx<=1; ++dx) {
            for(int dy=-1; dy<=1; ++dy) {
                for(int dz=-1; dz<=1; ++dz) {
                    GridCell dummy{cx+dx, cy+dy, cz+dz, 0};
                    auto it = lower_bound(grid.begin(), grid.end(), dummy);
                    while(it != grid.end() && it->cx == cx+dx && it->cy == cy+dy && it->cz == cz+dz) {
                        if (remap[it->index] == -1) {
                            if (length_squared(raw_v[it->index] - raw_v[orig_idx]) < eps*eps) {
                                remap[it->index] = new_idx;
                            }
                        }
                        it++;
                    }
                }
            }
        }
    }

    struct PolyIdx { vector<int> idxs; };
    vector<PolyIdx> all_pidxs;
    int raw_idx = 0;
    for(const auto& p : all_polys) {
        PolyIdx pi;
        for(size_t i=0; i<p.vertices.size(); ++i) pi.idxs.push_back(remap[raw_idx++]);
        vector<int> clean;
        for(int idx : pi.idxs) {
            if (clean.empty() || clean.back() != idx) clean.push_back(idx);
        }
        if (clean.size() > 1 && clean.front() == clean.back()) clean.pop_back();
        if (clean.size() >= 3) all_pidxs.push_back({clean});
    }

    vector<GridCell> vgrid(out_v.size());
    for(int i=0; i<out_v.size(); ++i) {
        vgrid[i] = {(int)floor(out_v[i].x/eps), (int)floor(out_v[i].y/eps), (int)floor(out_v[i].z/eps), i};
    }
    sort(vgrid.begin(), vgrid.end());

    for(auto& pi : all_pidxs) {
        vector<int> new_idxs;
        for(size_t i=0; i<pi.idxs.size(); ++i) {
            int a = pi.idxs[i];
            int b = pi.idxs[(i+1)%pi.idxs.size()];
            new_idxs.push_back(a);
            
            Vec3 va = out_v[a], vb = out_v[b];
            Vec3 min_p = {min(va.x, vb.x), min(va.y, vb.y), min(va.z, vb.z)};
            Vec3 max_p = {max(va.x, vb.x), max(va.y, vb.y), max(va.z, vb.z)};
            
            int min_cx = floor(min_p.x/eps)-1, max_cx = floor(max_p.x/eps)+1;
            int min_cy = floor(min_p.y/eps)-1, max_cy = floor(max_p.y/eps)+1;
            int min_cz = floor(min_p.z/eps)-1, max_cz = floor(max_p.z/eps)+1;
            
            vector<pair<double, int>> splits;
            for(int cx=min_cx; cx<=max_cx; ++cx) {
                for(int cy=min_cy; cy<=max_cy; ++cy) {
                    for(int cz=min_cz; cz<=max_cz; ++cz) {
                        GridCell dummy{cx, cy, cz, 0};
                        auto it = lower_bound(vgrid.begin(), vgrid.end(), dummy);
                        while(it != vgrid.end() && it->cx == cx && it->cy == cy && it->cz == cz) {
                            int idx = it->index;
                            if (idx != a && idx != b && point_on_segment(out_v[idx], va, vb)) {
                                double t = dot(out_v[idx] - va, vb - va) / length_squared(vb - va);
                                splits.push_back({t, idx});
                            }
                            it++;
                        }
                    }
                }
            }
            sort(splits.begin(), splits.end());
            for(const auto& sp : splits) {
                if (new_idxs.back() != sp.second) new_idxs.push_back(sp.second);
            }
        }
        pi.idxs = new_idxs;
    }

    vector<TriData> final_tris;
    for(const auto& pi : all_pidxs) {
        for(size_t i=1; i+1<pi.idxs.size(); ++i) {
            int i0 = pi.idxs[0];
            int i1 = pi.idxs[i];
            int i2 = pi.idxs[i+1];
            if (i0 == i1 || i1 == i2 || i2 == i0) continue;
            if (length_squared(cross(out_v[i1]-out_v[i0], out_v[i2]-out_v[i0])) < 1e-12) continue;
            final_tris.push_back(make_tridata(i0, i1, i2));
        }
    }

    sort(final_tris.begin(), final_tris.end());
    vector<TriData> unique_tris;
    for(size_t i=0; i<final_tris.size(); ++i) {
        if (i > 0 && final_tris[i] == final_tris[i-1]) continue;
        unique_tris.push_back(final_tris[i]);
    }

    cout << out_v.size() << " " << unique_tris.size() << "\n";
    cout << fixed << setprecision(6);
    for(const auto& v : out_v) cout << v.x << " " << v.y << " " << v.z << "\n";
    for(const auto& t : unique_tris) cout << t.i0 << " " << t.i1 << " " << t.i2 << "\n";

    return 0;
}
