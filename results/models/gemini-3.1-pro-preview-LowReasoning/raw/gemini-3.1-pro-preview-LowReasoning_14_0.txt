using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

namespace MinesweeperSolver
{
    class Program
    {
        static int H;
        static int W;
        static byte[] board;
        
        const byte UNKNOWN = 9;
        const byte FLAG = 10;
        const byte SAFE = 11;
        
        static int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
        static int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };
        
        static Queue<int> activeClues = new Queue<int>();
        static bool[] inQueue = Array.Empty<bool>();
        
        static Queue<int> pairwiseQueue = new Queue<int>();
        static bool[] inPairwiseQueue = Array.Empty<bool>();
        
        static List<int> allClues = new List<int>();
        static List<int> safeCellsOut = new List<int>();
        
        static int GetNeighbors(int idx, int[] nIdx) {
            int count = 0;
            int r = idx / W;
            int c = idx % W;
            for(int i=0; i<8; i++) {
                int nr = r + dr[i];
                int nc = c + dc[i];
                if (nr >= 0 && nr < H && nc >= 0 && nc < W) {
                    nIdx[count++] = nr * W + nc;
                }
            }
            return count;
        }

        static bool HasUnknowns(int clueIdx) {
            int r = clueIdx / W;
            int c = clueIdx % W;
            for(int i=0; i<8; i++) {
                int nr = r + dr[i];
                int nc = c + dc[i];
                if (nr >= 0 && nr < H && nc >= 0 && nc < W) {
                    if (board[nr * W + nc] == UNKNOWN) return true;
                }
            }
            return false;
        }

        static void WakeUpClues(int cellIdx) {
            int[] nIdx = new int[8];
            int nans = GetNeighbors(cellIdx, nIdx);
            for(int j=0; j<nans; j++) {
                int ni = nIdx[j];
                if (board[ni] <= 8) {
                    if (!inQueue[ni]) {
                        activeClues.Enqueue(ni);
                        inQueue[ni] = true;
                    }
                    if (!inPairwiseQueue[ni]) {
                        pairwiseQueue.Enqueue(ni);
                        inPairwiseQueue[ni] = true;
                    }
                }
            }
        }
        
        static bool RunSimpleRules() {
            bool changed = false;
            int[] nIdx = new int[8];
            
            while(activeClues.Count > 0) {
                int clueIdx = activeClues.Dequeue();
                inQueue[clueIdx] = false;
                
                int nans = GetNeighbors(clueIdx, nIdx);
                int nUnknowns = 0;
                int nMines = 0;
                
                for(int j=0; j<nans; j++) {
                    byte v = board[nIdx[j]];
                    if (v == UNKNOWN) nUnknowns++;
                    else if (v == FLAG) nMines++;
                }
                
                if (nUnknowns == 0) continue;
                
                int clueVal = board[clueIdx];
                if (clueVal == nMines) {
                    for(int j=0; j<nans; j++) {
                        int ni = nIdx[j];
                        if (board[ni] == UNKNOWN) {
                            board[ni] = SAFE;
                            safeCellsOut.Add(ni);
                            changed = true;
                            WakeUpClues(ni);
                        }
                    }
                } else if (clueVal == nMines + nUnknowns) {
                    for(int j=0; j<nans; j++) {
                        int ni = nIdx[j];
                        if (board[ni] == UNKNOWN) {
                            board[ni] = FLAG;
                            changed = true;
                            WakeUpClues(ni);
                        }
                    }
                }
            }
            return changed;
        }

        static bool RunPairwiseRules() {
            bool changed = false;
            int[] nIdx1 = new int[8];
            int[] nIdx2 = new int[8];
            int[] adjClues = new int[64];
            int[] uNeighbors = new int[8];
            
            while(pairwiseQueue.Count > 0) {
                int c1 = pairwiseQueue.Dequeue();
                inPairwiseQueue[c1] = false;
                
                if (!HasUnknowns(c1)) continue;
                
                int nans1 = GetNeighbors(c1, nIdx1);
                int unk1 = 0, mines1 = 0;
                int uMask1 = 0;
                for(int j=0; j<nans1; j++) {
                    if (board[nIdx1[j]] == UNKNOWN) { unk1++; uMask1 |= (1 << j); }
                    else if (board[nIdx1[j]] == FLAG) { mines1++; }
                }
                
                if (unk1 == 0) continue;
                int V1 = board[c1] - mines1;
                
                int adjCount = 0;
                for(int j=0; j<nans1; j++) {
                    if ((uMask1 & (1 << j)) != 0) {
                        int uIdx = nIdx1[j];
                        int unans = GetNeighbors(uIdx, uNeighbors);
                        for(int k=0; k<unans; k++) {
                            int cand = uNeighbors[k];
                            if (cand != c1 && board[cand] <= 8) {
                                bool found = false;
                                for(int m=0; m<adjCount; m++) {
                                    if (adjClues[m] == cand) { found = true; break; }
                                }
                                if (!found) adjClues[adjCount++] = cand;
                            }
                        }
                    }
                }
                
                bool localChanged = false;
                for(int m=0; m<adjCount; m++) {
                    int c2 = adjClues[m];
                    if (!HasUnknowns(c2)) continue;
                    
                    int nans2 = GetNeighbors(c2, nIdx2);
                    int unk2 = 0, mines2 = 0;
                    for(int j=0; j<nans2; j++) {
                        if (board[nIdx2[j]] == UNKNOWN) unk2++;
                        else if (board[nIdx2[j]] == FLAG) mines2++;
                    }
                    if (unk2 == 0) continue;
                    int V2 = board[c2] - mines2;
                    
                    int O_count = 0, U1_only = 0, U2_only = 0;
                    for(int j=0; j<nans1; j++) {
                        if (board[nIdx1[j]] == UNKNOWN) {
                            bool in2 = false;
                            for(int k=0; k<nans2; k++) {
                                if (nIdx1[j] == nIdx2[k]) { in2 = true; break; }
                            }
                            if (in2) O_count++;
                            else U1_only++;
                        }
                    }
                    for(int k=0; k<nans2; k++) {
                        if (board[nIdx2[k]] == UNKNOWN) {
                            bool in1 = false;
                            for(int j=0; j<nans1; j++) {
                                if (nIdx2[k] == nIdx1[j]) { in1 = true; break; }
                            }
                            if (!in1) U2_only++;
                        }
                    }
                    
                    if (O_count == 0) continue;
                    
                    int max_O_from_1 = Math.Min(O_count, V1);
                    int min_O_from_1 = Math.Max(0, V1 - U1_only);
                    
                    if (V2 - max_O_from_1 == U2_only && U2_only > 0) {
                        for(int k=0; k<nans2; k++) {
                            int idx = nIdx2[k];
                            if (board[idx] == UNKNOWN) {
                                bool in1 = false;
                                for(int j=0; j<nans1; j++) if (idx == nIdx1[j]) { in1 = true; break; }
                                if (!in1) { board[idx] = FLAG; localChanged = true; WakeUpClues(idx); }
                            }
                        }
                    }
                    if (V2 - min_O_from_1 == 0 && U2_only > 0) {
                        for(int k=0; k<nans2; k++) {
                            int idx = nIdx2[k];
                            if (board[idx] == UNKNOWN) {
                                bool in1 = false;
                                for(int j=0; j<nans1; j++) if (idx == nIdx1[j]) { in1 = true; break; }
                                if (!in1) { board[idx] = SAFE; safeCellsOut.Add(idx); localChanged = true; WakeUpClues(idx); }
                            }
                        }
                    }
                    
                    int max_O_from_2 = Math.Min(O_count, V2);
                    int min_O_from_2 = Math.Max(0, V2 - U2_only);
                    
                    if (V1 - max_O_from_2 == U1_only && U1_only > 0) {
                        for(int j=0; j<nans1; j++) {
                            int idx = nIdx1[j];
                            if (board[idx] == UNKNOWN) {
                                bool in2 = false;
                                for(int k=0; k<nans2; k++) if (idx == nIdx2[k]) { in2 = true; break; }
                                if (!in2) { board[idx] = FLAG; localChanged = true; WakeUpClues(idx); }
                            }
                        }
                    }
                    if (V1 - min_O_from_2 == 0 && U1_only > 0) {
                        for(int j=0; j<nans1; j++) {
                            int idx = nIdx1[j];
                            if (board[idx] == UNKNOWN) {
                                bool in2 = false;
                                for(int k=0; k<nans2; k++) if (idx == nIdx2[k]) { in2 = true; break; }
                                if (!in2) { board[idx] = SAFE; safeCellsOut.Add(idx); localChanged = true; WakeUpClues(idx); }
                            }
                        }
                    }
                    
                    if (localChanged) {
                        changed = true;
                        break; 
                    }
                }
                
                if (localChanged) return true;
            }
            return changed;
        }

        static bool RunCSPRules() {
            bool overallChanged = false;
            
            List<int> activeCSPClues = new List<int>();
            foreach(int c in allClues) {
                if (HasUnknowns(c)) activeCSPClues.Add(c);
            }
            if (activeCSPClues.Count == 0) return false;
            
            Dictionary<int, List<int>> clueUnknowns = new Dictionary<int, List<int>>();
            foreach(int c in activeCSPClues) {
                clueUnknowns[c] = new List<int>();
                int[] nIdx = new int[8];
                int nans = GetNeighbors(c, nIdx);
                for(int i=0; i<nans; i++) {
                    if (board[nIdx[i]] == UNKNOWN) clueUnknowns[c].Add(nIdx[i]);
                }
            }
            
            List<List<int>> components = new List<List<int>>();
            HashSet<int> visitedClues = new HashSet<int>();
            
            foreach(int startClue in activeCSPClues) {
                if (visitedClues.Contains(startClue)) continue;
                
                List<int> compClues = new List<int>();
                Queue<int> q = new Queue<int>();
                q.Enqueue(startClue);
                visitedClues.Add(startClue);
                
                while(q.Count > 0) {
                    int curr = q.Dequeue();
                    compClues.Add(curr);
                    
                    foreach(int other in activeCSPClues) {
                        if (!visitedClues.Contains(other)) {
                            bool share = false;
                            foreach(int u in clueUnknowns[curr]) {
                                if (clueUnknowns[other].Contains(u)) { share = true; break; }
                            }
                            if (share) {
                                visitedClues.Add(other);
                                q.Enqueue(other);
                            }
                        }
                    }
                }
                components.Add(compClues);
            }
            
            foreach(List<int> compClues in components) {
                HashSet<int> compUnknownsSet = new HashSet<int>();
                foreach(int c in compClues) {
                    foreach(int u in clueUnknowns[c]) compUnknownsSet.Add(u);
                }
                
                List<int> compUnknowns = new List<int>(compUnknownsSet);
                
                // Adaptive logic: only run full CSP if component is reasonably small
                if (compUnknowns.Count > 22) continue; 
                
                int[] validAssignments = new int[compUnknowns.Count];
                int totalValid = 0;
                bool[] currentAssigned = new bool[compUnknowns.Count];
                
                int[] cIdxs = compClues.ToArray();
                int[] cTarget = new int[cIdxs.Length];
                for(int i=0; i<cIdxs.Length; i++) {
                    int mines = 0;
                    int[] nIdx = new int[8];
                    int nans = GetNeighbors(cIdxs[i], nIdx);
                    for(int j=0; j<nans; j++) if (board[nIdx[j]] == FLAG) mines++;
                    cTarget[i] = board[cIdxs[i]] - mines;
                }
                
                List<int>[] cUnkIdxs = new List<int>[cIdxs.Length];
                for(int i=0; i<cIdxs.Length; i++) {
                    cUnkIdxs[i] = new List<int>();
                    foreach(int u in clueUnknowns[cIdxs[i]]) {
                        cUnkIdxs[i].Add(compUnknowns.IndexOf(u));
                    }
                }
                
                void Solve(int uIdx) {
                    for(int i=0; i<cIdxs.Length; i++) {
                        int assignedMines = 0;
                        int unassigned = 0;
                        foreach(int cu in cUnkIdxs[i]) {
                            if (cu < uIdx) {
                                if (currentAssigned[cu]) assignedMines++;
                            } else {
                                unassigned++;
                            }
                        }
                        if (assignedMines > cTarget[i]) return;
                        if (assignedMines + unassigned < cTarget[i]) return;
                    }
                    
                    if (uIdx == compUnknowns.Count) {
                        totalValid++;
                        for(int i=0; i<compUnknowns.Count; i++) {
                            if (currentAssigned[i]) validAssignments[i]++;
                        }
                        return;
                    }
                    
                    currentAssigned[uIdx] = false;
                    Solve(uIdx + 1);
                    currentAssigned[uIdx] = true;
                    Solve(uIdx + 1);
                }
                
                Solve(0);
                
                if (totalValid > 0) {
                    for(int i=0; i<compUnknowns.Count; i++) {
                        if (validAssignments[i] == 0) {
                            int u = compUnknowns[i];
                            if (board[u] == UNKNOWN) {
                                board[u] = SAFE;
                                safeCellsOut.Add(u);
                                overallChanged = true;
                                WakeUpClues(u);
                            }
                        } else if (validAssignments[i] == totalValid) {
                            int u = compUnknowns[i];
                            if (board[u] == UNKNOWN) {
                                board[u] = FLAG;
                                overallChanged = true;
                                WakeUpClues(u);
                            }
                        }
                    }
                }
                
                if (overallChanged) return true; 
            }
            
            return overallChanged;
        }

        static void Main(string[] args)
        {
            string line = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line)) return;
            string[] parts = line.Split(new char[]{' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
            H = int.Parse(parts[0]);
            W = int.Parse(parts[1]);
            
            board = new byte[H * W];
            inQueue = new bool[H * W];
            inPairwiseQueue = new bool[H * W];
            
            for (int r = 0; r < H; r++) {
                string rowLine = Console.ReadLine();
                for (int c = 0; c < W; c++) {
                    char ch = rowLine[c];
                    int idx = r * W + c;
                    if (ch == ' ') board[idx] = UNKNOWN;
                    else if (ch == '*') board[idx] = FLAG;
                    else if (ch >= '0' && ch <= '8') {
                        board[idx] = (byte)(ch - '0');
                        allClues.Add(idx);
                        
                        activeClues.Enqueue(idx);
                        inQueue[idx] = true;
                        
                        pairwiseQueue.Enqueue(idx);
                        inPairwiseQueue[idx] = true;
                    }
                }
            }
            
            bool changed = true;
            while(changed) {
                changed = false;
                
                if (RunSimpleRules()) { changed = true; continue; }
                if (RunPairwiseRules()) { changed = true; continue; }
                if (RunCSPRules()) { changed = true; continue; }
            }
            
            Console.WriteLine(safeCellsOut.Count);
            using (StreamWriter sw = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536)) {
                foreach(int idx in safeCellsOut) {
                    int r = idx / W;
                    int c = idx % W;
                    sw.Write(c);
                    sw.Write(' ');
                    sw.WriteLine(r);
                }
            }
        }
    }
}
