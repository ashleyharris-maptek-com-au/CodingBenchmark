using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program {
    static long bestSum = -1;
    static int[] bestPattern;
    static int[] currentPattern;
    static int iterations = 0;
    const int MAX_ITERATIONS = 5000;

    class Pattern {
        public int[] counts;
        public int times;
    }

    struct Cut : IComparable<Cut> {
        public long length;
        public int index;
        public int CompareTo(Cut other) {
            int cmp = other.length.CompareTo(length);
            if (cmp == 0) return index.CompareTo(other.index);
            return cmp;
        }
    }

    static void Main(string[] args) {
        using var stream = Console.OpenStandardInput();
        var scanner = new FastScanner(stream);

        int N = scanner.ReadInt();
        if (N <= 0) {
            Console.WriteLine(0);
            return;
        }
        long stockLength = scanner.ReadLong();

        var cuts = new Cut[N];
        var originalIndices = new Dictionary<long, Stack<int>>();

        for (int i = 0; i < N; i++) {
            long len = scanner.ReadLong();
            cuts[i] = new Cut { length = len, index = i };
            if (!originalIndices.TryGetValue(len, out var stack)) {
                stack = new Stack<int>();
                originalIndices[len] = stack;
            }
            stack.Push(i);
        }

        int m = originalIndices.Count;

        using var writer = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536);

        if (m <= 5000) {
            // Pattern-based greedy algorithm (very strong heuristic / exact for many cases)
            long[] L = originalIndices.Keys.OrderByDescending(x => x).ToArray();
            int[] available = new int[m];
            for (int i = 0; i < m; i++) available[i] = originalIndices[L[i]].Count;

            var appliedPatterns = new List<Pattern>();
            long[] suffixSum = new long[m];
            bestPattern = new int[m];
            currentPattern = new int[m];

            while (true) {
                int firstIdx = -1;
                for (int i = 0; i < m; i++) {
                    if (available[i] > 0) {
                        firstIdx = i;
                        break;
                    }
                }
                if (firstIdx == -1) break;

                long target = stockLength - L[firstIdx];
                long currentSuffix = 0;
                for (int i = m - 1; i > firstIdx; i--) {
                    long add = L[i] * available[i];
                    if (currentSuffix + add < 0 || currentSuffix + add > target) {
                        currentSuffix = target + 1;
                    } else {
                        currentSuffix += add;
                    }
                    suffixSum[i] = currentSuffix;
                }

                bestSum = -1;
                Array.Clear(bestPattern, 0, m);
                Array.Clear(currentPattern, 0, m);
                iterations = 0;

                SolveKnapsack(firstIdx + 1, 0, target, L, available, suffixSum, m);

                bestPattern[firstIdx]++;

                int times = available[firstIdx];
                for (int i = firstIdx + 1; i < m; i++) {
                    if (bestPattern[i] > 0) {
                        times = Math.Min(times, available[i] / bestPattern[i]);
                    }
                }

                for (int i = firstIdx; i < m; i++) {
                    available[i] -= bestPattern[i] * times;
                }

                appliedPatterns.Add(new Pattern { counts = (int[])bestPattern.Clone(), times = times });
            }

            int totalBins = 0;
            foreach (var p in appliedPatterns) totalBins += p.times;
            writer.WriteLine(totalBins);

            foreach (var p in appliedPatterns) {
                for (int t = 0; t < p.times; t++) {
                    bool first = true;
                    for (int i = 0; i < m; i++) {
                        for (int c = 0; c < p.counts[i]; c++) {
                            if (!first) writer.Write(" ");
                            writer.Write(originalIndices[L[i]].Pop());
                            first = false;
                        }
                    }
                    writer.WriteLine();
                }
            }
        } else {
            // Best Fit Decreasing for very large number of distinct sizes
            Array.Sort(cuts);
            var bst = new SortedSet<(long cap, int id)>();
            var bins = new List<List<int>>();

            for (int i = 0; i < N; i++) {
                long val = cuts[i].length;
                var view = bst.GetViewBetween((val, -1), (long.MaxValue, int.MaxValue));
                var min = view.Min;
                if (min.cap >= val) {
                    bst.Remove(min);
                    min.cap -= val;
                    bins[min.id].Add(cuts[i].index);
                    bst.Add(min);
                } else {
                    int newId = bins.Count;
                    bins.Add(new List<int> { cuts[i].index });
                    bst.Add((stockLength - val, newId));
                }
            }

            writer.WriteLine(bins.Count);
            foreach (var bin in bins) {
                bool first = true;
                foreach (var idx in bin) {
                    if (!first) writer.Write(" ");
                    writer.Write(idx);
                    first = false;
                }
                writer.WriteLine();
            }
        }
    }

    static void SolveKnapsack(int itemIdx, long currentSum, long target, long[] L, int[] available, long[] suffixSum, int m) {
        if (currentSum == target) {
            bestSum = currentSum;
            Array.Copy(currentPattern, bestPattern, m);
            return;
        }
        if (itemIdx == m) {
            if (currentSum > bestSum) {
                bestSum = currentSum;
                Array.Copy(currentPattern, bestPattern, m);
            }
            return;
        }
        
        iterations++;
        if (iterations > MAX_ITERATIONS) return;

        long maxPossible = currentSum + suffixSum[itemIdx];
        if (maxPossible <= bestSum) return;

        long remainingCap = target - currentSum;
        long maxTakeLong = remainingCap / L[itemIdx];
        int maxTake = (int)Math.Min(maxTakeLong, available[itemIdx]);
        
        for (int take = maxTake; take >= 0; take--) {
            currentPattern[itemIdx] = take;
            SolveKnapsack(itemIdx + 1, currentSum + take * L[itemIdx], target, L, available, suffixSum, m);
            if (bestSum == target) return; 
            if (iterations > MAX_ITERATIONS) return;
        }
    }

    class FastScanner {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 1024];
        private int head, tail;

        public FastScanner(Stream stream) {
            this.stream = stream;
        }

        public int ReadInt() {
            int c = Read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = Read();
            }
            int res = 0;
            while (c > 32) {
                res = res * 10 + c - '0';
                c = Read();
            }
            return res;
        }

        public long ReadLong() {
            int c = Read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = Read();
            }
            long res = 0;
            while (c > 32) {
                res = res * 10 + c - '0';
                c = Read();
            }
            return res;
        }

        private int Read() {
            if (head >= tail) {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }
}
