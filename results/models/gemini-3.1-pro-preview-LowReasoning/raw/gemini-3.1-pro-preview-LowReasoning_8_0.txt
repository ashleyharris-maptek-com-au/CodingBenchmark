use std::io::{self, BufRead, Read, Write};
use std::collections::VecDeque;

const UNVISITED: u8 = 0;
const WALL: u8 = 1;
const PARENT_UP: u8 = 2;
const PARENT_RIGHT: u8 = 3;
const PARENT_DOWN: u8 = 4;
const PARENT_LEFT: u8 = 5;
const START: u8 = 6;

const NEXT_UP: u8 = 10;
const NEXT_RIGHT: u8 = 11;
const NEXT_DOWN: u8 = 12;
const NEXT_LEFT: u8 = 13;
const GOAL_MARK: u8 = 14;

struct FastWriter<W: Write> {
    writer: W,
    buf: Vec<u8>,
}

impl<W: Write> FastWriter<W> {
    fn new(writer: W) -> Self {
        Self {
            writer,
            buf: Vec::with_capacity(64 * 1024),
        }
    }

    fn write_u32(&mut self, mut n: u32) {
        if n == 0 {
            self.push(b'0');
            return;
        }
        let mut temp = [0u8; 10];
        let mut len = 0;
        while n > 0 {
            temp[len] = b'0' + (n % 10) as u8;
            n /= 10;
            len += 1;
        }
        for i in (0..len).rev() {
            self.push(temp[i]);
        }
    }

    fn push(&mut self, b: u8) {
        if self.buf.len() == self.buf.capacity() {
            self.flush();
        }
        self.buf.push(b);
    }

    fn flush(&mut self) {
        if !self.buf.is_empty() {
            self.writer.write_all(&self.buf).unwrap();
            self.buf.clear();
        }
    }
}

impl<W: Write> Drop for FastWriter<W> {
    fn drop(&mut self) {
        self.flush();
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lock = stdin.lock();

    let mut first_line = String::new();
    if lock.read_line(&mut first_line).is_err() {
        println!("0");
        return;
    }
    let parts: Vec<u32> = first_line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
    if parts.len() < 2 {
        println!("0");
        return;
    }
    let h = parts[0] as u32;
    let w = parts[1] as u32;

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    let mut grid = vec![UNVISITED; (w as usize) * (h as usize)];
    let mut start = None;
    let mut goal = None;

    let mut r: u32 = 0;
    let mut c: u32 = 0;

    loop {
        let buffer = lock.fill_buf().unwrap();
        let n = buffer.len();
        if n == 0 { break; }

        let mut consumed = 0;
        for &b in buffer {
            consumed += 1;
            if b == b'\n' {
                c = 0;
                r += 1;
                continue;
            }
            if b == b'\r' {
                continue;
            }
            if r >= h {
                break;
            }

            if c < w {
                let idx = (r as usize) * (w as usize) + (c as usize);
                if b == b'A' {
                    start = Some((c, r));
                } else if b == b'B' {
                    goal = Some((c, r));
                } else if b == b'#' {
                    grid[idx] = WALL;
                }
                c += 1;
            }
        }
        lock.consume(consumed);
        if r >= h { break; }
    }

    let (start_x, start_y) = match start {
        Some(s) => s,
        None => { println!("0"); return; }
    };
    let (goal_x, goal_y) = match goal {
        Some(g) => g,
        None => { println!("0"); return; }
    };

    let mut end_found = start_x == goal_x && start_y == goal_y;
    let goal_idx = (goal_y as usize) * (w as usize) + (goal_x as usize);
    let start_idx = (start_y as usize) * (w as usize) + (start_x as usize);
    grid[start_idx] = START;

    if !end_found {
        let mut q = VecDeque::new();
        q.push_back((start_x, start_y));

        while let Some((x, y)) = q.pop_front() {
            let idx = (y as usize) * (w as usize) + (x as usize);

            // UP
            if y > 0 {
                let nidx = idx - (w as usize);
                if grid[nidx] == UNVISITED {
                    grid[nidx] = PARENT_DOWN;
                    if nidx == goal_idx {
                        end_found = true;
                        break;
                    }
                    q.push_back((x, y - 1));
                }
            }
            // RIGHT
            if x + 1 < w {
                let nidx = idx + 1;
                if grid[nidx] == UNVISITED {
                    grid[nidx] = PARENT_LEFT;
                    if nidx == goal_idx {
                        end_found = true;
                        break;
                    }
                    q.push_back((x + 1, y));
                }
            }
            // DOWN
            if y + 1 < h {
                let nidx = idx + (w as usize);
                if grid[nidx] == UNVISITED {
                    grid[nidx] = PARENT_UP;
                    if nidx == goal_idx {
                        end_found = true;
                        break;
                    }
                    q.push_back((x, y + 1));
                }
            }
            // LEFT
            if x > 0 {
                let nidx = idx - 1;
                if grid[nidx] == UNVISITED {
                    grid[nidx] = PARENT_RIGHT;
                    if nidx == goal_idx {
                        end_found = true;
                        break;
                    }
                    q.push_back((x - 1, y));
                }
            }
        }
    }

    if end_found {
        let mut len = 0;
        let mut cx = goal_x;
        let mut cy = goal_y;

        let mut prev_dir = GOAL_MARK;

        loop {
            len += 1;
            let idx = (cy as usize) * (w as usize) + (cx as usize);
            let val = grid[idx];

            grid[idx] = prev_dir;

            if val == START {
                break;
            } else if val == PARENT_UP {
                cy -= 1;
                prev_dir = NEXT_DOWN;
            } else if val == PARENT_RIGHT {
                cx += 1;
                prev_dir = NEXT_LEFT;
            } else if val == PARENT_DOWN {
                cy += 1;
                prev_dir = NEXT_UP;
            } else if val == PARENT_LEFT {
                cx -= 1;
                prev_dir = NEXT_RIGHT;
            } else {
                println!("0");
                return;
            }
        }

        let stdout = io::stdout();
        let mut out = FastWriter::new(stdout.lock());
        out.write_u32(len);
        out.push(b'\n');

        let mut cx = start_x;
        let mut cy = start_y;

        loop {
            out.write_u32(cx);
            out.push(b' ');
            out.write_u32(cy);
            out.push(b'\n');

            let idx = (cy as usize) * (w as usize) + (cx as usize);
            let next_dir = grid[idx];

            if next_dir == GOAL_MARK {
                break;
            } else if next_dir == NEXT_UP {
                cy -= 1;
            } else if next_dir == NEXT_RIGHT {
                cx += 1;
            } else if next_dir == NEXT_DOWN {
                cy += 1;
            } else if next_dir == NEXT_LEFT {
                cx -= 1;
            } else {
                break;
            }
        }
    } else {
        println!("0");
    }
}
