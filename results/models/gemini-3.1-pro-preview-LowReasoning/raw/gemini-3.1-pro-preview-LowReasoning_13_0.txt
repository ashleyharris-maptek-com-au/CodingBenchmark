#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

void solve_trivial(const vector<string>& S) {
    string best = "";
    int n = S[0].length();
    for (int i = 0; i < n; i++) {
        for (int j = 1; i + j <= n; j++) {
            string sub = S[0].substr(i, j);
            if (sub.length() <= best.length()) continue;
            bool ok = true;
            for (size_t k = 1; k < S.size(); k++) {
                if (S[k].find(sub) == string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) best = sub;
        }
    }
    cout << best << "\n";
}

void solve_binary_search(const vector<string>& S) {
    int low = 1, high = S[0].length();
    string best = "";
    while (low <= high) {
        int mid = low + (high - low) / 2;
        bool found_mid = false;
        string current_best = "";
        
        for (int i = 0; i <= (int)S[0].length() - mid; i++) {
            string sub = S[0].substr(i, mid);
            bool ok = true;
            for (size_t k = 1; k < S.size(); k++) {
                if (S[k].find(sub) == string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                found_mid = true;
                current_best = sub;
                break;
            }
        }
        if (found_mid) {
            best = current_best;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    cout << best << "\n";
}

void solve_sam(const vector<string>& S) {
    int n = S[0].length();
    if (n == 0) {
        cout << "\n";
        return;
    }
    
    int max_states = 2 * n + 2;
    vector<int> st_len(max_states);
    vector<int> st_link(max_states);
    vector<int> st_head(max_states, -1);
    vector<int> st_first_pos(max_states);
    
    vector<int> edge_next;
    vector<int> edge_to;
    vector<char> edge_char;
    edge_next.reserve(4 * n + 4);
    edge_to.reserve(4 * n + 4);
    edge_char.reserve(4 * n + 4);
    
    int sz = 1;
    int last = 0;
    int edges_cnt = 0;
    
    st_len[0] = 0;
    st_link[0] = -1;
    st_first_pos[0] = -1;
    
    auto add_edge = [&](int from, char c, int to) {
        edge_to.push_back(to);
        edge_char.push_back(c);
        edge_next.push_back(st_head[from]);
        st_head[from] = edges_cnt++;
    };
    
    auto get_edge = [&](int from, char c) -> int {
        for (int e = st_head[from]; e != -1; e = edge_next[e]) {
            if (edge_char[e] == c) return edge_to[e];
        }
        return -1;
    };
    
    auto set_edge = [&](int from, char c, int to) {
        for (int e = st_head[from]; e != -1; e = edge_next[e]) {
            if (edge_char[e] == c) {
                edge_to[e] = to;
                return;
            }
        }
        add_edge(from, c, to);
    };
    
    for (char c : S[0]) {
        int cur = sz++;
        st_len[cur] = st_len[last] + 1;
        st_first_pos[cur] = st_len[cur] - 1;
        st_head[cur] = -1;
        
        int p = last;
        while (p != -1 && get_edge(p, c) == -1) {
            add_edge(p, c, cur);
            p = st_link[p];
        }
        
        if (p == -1) {
            st_link[cur] = 0;
        } else {
            int q = get_edge(p, c);
            if (st_len[p] + 1 == st_len[q]) {
                st_link[cur] = q;
            } else {
                int clone = sz++;
                st_len[clone] = st_len[p] + 1;
                st_head[clone] = -1;
                st_first_pos[clone] = st_first_pos[q];
                
                for (int e = st_head[q]; e != -1; e = edge_next[e]) {
                    add_edge(clone, edge_char[e], edge_to[e]);
                }
                
                st_link[clone] = st_link[q];
                
                while (p != -1 && get_edge(p, c) == q) {
                    set_edge(p, c, clone);
                    p = st_link[p];
                }
                
                st_link[q] = st_link[cur] = clone;
            }
        }
        last = cur;
    }
    
    vector<int> cnt(n + 1, 0);
    for (int i = 0; i < sz; i++) cnt[st_len[i]]++;
    for (int i = 1; i <= n; i++) cnt[i] += cnt[i - 1];
    vector<int> sorted_states(sz);
    for (int i = sz - 1; i >= 0; i--) {
        sorted_states[--cnt[st_len[i]]] = i;
    }
    
    vector<int> min_len(sz);
    for (int i = 0; i < sz; i++) min_len[i] = st_len[i];
    
    vector<int> curr_match(sz, 0);
    
    for (size_t i = 1; i < S.size(); i++) {
        fill(curr_match.begin(), curr_match.end(), 0);
        int v = 0;
        int l = 0;
        for (char c : S[i]) {
            while (v != -1 && get_edge(v, c) == -1) {
                v = st_link[v];
                if (v != -1) l = st_len[v];
                else l = 0;
            }
            if (v == -1) {
                v = 0;
                l = 0;
            } else {
                v = get_edge(v, c);
                l++;
            }
            curr_match[v] = max(curr_match[v], l);
        }
        
        for (int j = sz - 1; j >= 0; j--) {
            int u = sorted_states[j];
            min_len[u] = min(min_len[u], curr_match[u]);
            if (curr_match[u] > 0 && st_link[u] != -1) {
                curr_match[st_link[u]] = max(curr_match[st_link[u]], st_len[st_link[u]]);
            }
        }
    }
    
    int best_len = 0;
    int best_pos = -1;
    for (int i = 0; i < sz; i++) {
        if (min_len[i] > best_len) {
            best_len = min_len[i];
            best_pos = st_first_pos[i];
        }
    }
    
    if (best_len > 0 && best_pos != -1) {
        cout << S[0].substr(best_pos - best_len + 1, best_len) << "\n";
    } else {
        cout << "\n";
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);
    
    vector<string> S(N);
    for (int i = 0; i < N; i++) {
        getline(cin, S[i]);
    }
    
    if (N == 0) {
        cout << "\n";
        return 0;
    }
    
    for (const auto& s : S) {
        if (s.empty()) {
            cout << "\n";
            return 0;
        }
    }
    
    int min_idx = 0;
    for (int i = 1; i < N; i++) {
        if (S[i].length() < S[min_idx].length()) {
            min_idx = i;
        }
    }
    if (min_idx != 0) {
        swap(S[0], S[min_idx]);
    }
    
    size_t max_len = 0;
    for (const auto& s : S) {
        max_len = max(max_len, s.length());
    }
    
    if (max_len <= 50) {
        solve_trivial(S);
    } else if (max_len <= 500) {
        solve_binary_search(S);
    } else {
        solve_sam(S);
    }
    
    return 0;
}
