using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.Linq;

namespace OrbitalTSP
{
    public struct Vector3D
    {
        public double X, Y, Z;
        public Vector3D(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3D operator -(Vector3D a, Vector3D b) => new Vector3D(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3D operator +(Vector3D a, Vector3D b) => new Vector3D(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3D operator *(Vector3D a, double d) => new Vector3D(a.X * d, a.Y * d, a.Z * d);
        public static Vector3D operator /(Vector3D a, double d) => new Vector3D(a.X / d, a.Y / d, a.Z / d);
        public static Vector3D Cross(Vector3D a, Vector3D b) => new Vector3D(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
        public double Dot(Vector3D b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Magnitude() => Math.Sqrt(X * X + Y * Y + Z * Z);
    }

    class Orbit
    {
        public Vector3D r, v, h, e;
        public double a, eps;
        public Orbit(Vector3D r, Vector3D v, double mu)
        {
            this.r = r; this.v = v;
            h = Vector3D.Cross(r, v);
            eps = v.Magnitude() * v.Magnitude() / 2.0 - mu / r.Magnitude();
            eps = Math.Min(-1e-9, eps); // Ensure bound orbit for heuristic approximation
            a = -mu / (2.0 * eps);
            e = Vector3D.Cross(v, h) / mu - r / r.Magnitude();
        }
    }

    class Program
    {
        static double ComputeCost(Orbit o1, Orbit o2, double mu)
        {
            double v1 = Math.Sqrt(mu / o1.a);
            double v2 = Math.Sqrt(mu / o2.a);
            double atx = (o1.a + o2.a) / 2.0;
            double vtx1 = Math.Sqrt(Math.Max(0, mu * (2.0 / o1.a - 1.0 / atx)));
            double vtx2 = Math.Sqrt(Math.Max(0, mu * (2.0 / o2.a - 1.0 / atx)));

            double h1Mag = o1.h.Magnitude();
            double h2Mag = o2.h.Magnitude();
            double cosTheta = (h1Mag > 1e-9 && h2Mag > 1e-9) ? o1.h.Dot(o2.h) / (h1Mag * h2Mag) : 1.0;
            double theta = Math.Acos(Math.Max(-1.0, Math.Min(1.0, cosTheta)));

            double dv_combined, dv_other;
            if (o1.a > o2.a)
            {
                dv_combined = Math.Sqrt(Math.Max(0, vtx1 * vtx1 + v1 * v1 - 2 * vtx1 * v1 * Math.Cos(theta)));
                dv_other = Math.Abs(vtx2 - v2);
            }
            else
            {
                dv_combined = Math.Sqrt(Math.Max(0, vtx2 * vtx2 + v2 * v2 - 2 * vtx2 * v2 * Math.Cos(theta)));
                dv_other = Math.Abs(vtx1 - v1);
            }

            double dv = dv_combined + dv_other;
            double de = (o1.e - o2.e).Magnitude();
            double ecc_penalty = Math.Max(v1, v2) * de;

            return dv + ecc_penalty;
        }

        static void Main(string[] args)
        {
            string firstLine = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(firstLine)) return;
            string[] parts1 = firstLine.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int N = int.Parse(parts1[0]);
            double mu = double.Parse(parts1[1], CultureInfo.InvariantCulture);

            double[] ReadVector()
            {
                string line = Console.ReadLine();
                string[] parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                return parts.Select(p => double.Parse(p, CultureInfo.InvariantCulture)).ToArray();
            }

            double[] startParts = ReadVector();
            Orbit startOrbit = new Orbit(
                new Vector3D(startParts[0], startParts[1], startParts[2]),
                new Vector3D(startParts[3], startParts[4], startParts[5]),
                mu
            );

            Orbit[] stations = new Orbit[N];
            for (int i = 0; i < N; i++)
            {
                double[] s = ReadVector();
                stations[i] = new Orbit(
                    new Vector3D(s[0], s[1], s[2]),
                    new Vector3D(s[3], s[4], s[5]),
                    mu
                );
            }

            int numNodes = N + 1;
            double[,] dist = new double[numNodes, numNodes];
            Orbit[] allOrbits = new Orbit[numNodes];
            allOrbits[0] = startOrbit;
            for (int i = 0; i < N; i++) allOrbits[i + 1] = stations[i];

            for (int i = 0; i < numNodes; i++)
            {
                for (int j = i + 1; j < numNodes; j++)
                {
                    double c = ComputeCost(allOrbits[i], allOrbits[j], mu);
                    dist[i, j] = c;
                    dist[j, i] = c;
                }
            }

            int[] bestTour = null;
            double bestCost = double.MaxValue;
            Random rnd = new Random(42);

            Stopwatch sw = Stopwatch.StartNew();

            int iterations = 0;
            // Adaptive time limit: 4.5 seconds to guarantee we finish well within 5 seconds.
            while (sw.ElapsedMilliseconds < 4500 && iterations < 200000)
            {
                iterations++;
                int[] tour = new int[numNodes];
                tour[0] = 0;
                List<int> unvisited = new List<int>(N);
                for (int i = 1; i <= N; i++) unvisited.Add(i);

                int current = 0;
                for (int i = 1; i < numNodes; i++)
                {
                    int best1 = -1, best2 = -1, best3 = -1;
                    double d1 = double.MaxValue, d2 = double.MaxValue, d3 = double.MaxValue;

                    foreach (int n in unvisited)
                    {
                        double d = dist[current, n];
                        if (d < d1)
                        {
                            d3 = d2; best3 = best2;
                            d2 = d1; best2 = best1;
                            d1 = d; best1 = n;
                        }
                        else if (d < d2)
                        {
                            d3 = d2; best3 = best2;
                            d2 = d; best2 = n;
                        }
                        else if (d < d3)
                        {
                            d3 = d; best3 = n;
                        }
                    }

                    List<int> candidates = new List<int>();
                    if (best1 != -1) candidates.Add(best1);
                    if (best2 != -1) candidates.Add(best2);
                    if (best3 != -1) candidates.Add(best3);

                    int next = candidates[rnd.Next(candidates.Count)];
                    tour[i] = next;
                    unvisited.Remove(next);
                    current = next;
                }

                double currentCost = Evaluate(tour, dist);

                bool improved = true;
                while (improved)
                {
                    improved = false;
                    for (int i = 1; i < numNodes - 1; i++)
                    {
                        for (int j = i + 1; j < numNodes; j++)
                        {
                            double oldEdgeCost = dist[tour[i - 1], tour[i]];
                            if (j < numNodes - 1)
                                oldEdgeCost += dist[tour[j], tour[j + 1]];

                            double newEdgeCost = dist[tour[i - 1], tour[j]];
                            if (j < numNodes - 1)
                                newEdgeCost += dist[tour[i], tour[j + 1]];

                            if (newEdgeCost < oldEdgeCost - 1e-9)
                            {
                                Array.Reverse(tour, i, j - i + 1);
                                currentCost = currentCost - oldEdgeCost + newEdgeCost;
                                improved = true;
                            }
                        }
                    }
                }

                if (currentCost < bestCost)
                {
                    bestCost = currentCost;
                    bestTour = (int[])tour.Clone();
                }
            }

            int[] resultIndices = new int[N];
            for (int i = 0; i < N; i++)
            {
                resultIndices[i] = bestTour[i + 1] - 1;
            }

            Console.WriteLine(string.Join(" ", resultIndices));
        }

        static double Evaluate(int[] tour, double[,] dist)
        {
            double cost = 0;
            for (int i = 0; i < tour.Length - 1; i++)
            {
                cost += dist[tour[i], tour[i + 1]];
            }
            return cost;
        }
    }
}
