use std::io::{self, Read};
use std::collections::VecDeque;
use std::time::Instant;

#[derive(Clone, Copy)]
struct Point {
    x: f32,
    y: f32,
}

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

struct Node {
    mass: f32,
    com_x: f32,
    com_y: f32,
    min_x: f32,
    min_y: f32,
    max_x: f32,
    max_y: f32,
    child: usize,
    point_idx: usize,
}

fn get_child(cx: f32, cy: f32, p: &Point) -> usize {
    let dx = if p.x < cx { 0 } else { 1 };
    let dy = if p.y < cy { 0 } else { 1 };
    dx + (dy << 1)
}

fn insert(tree: &mut Vec<Node>, node_idx: usize, p_idx: usize, pos: &[Point], depth: usize) {
    if depth > 60 { return; }
    
    if tree[node_idx].child == 0 {
        if tree[node_idx].point_idx == usize::MAX {
            tree[node_idx].point_idx = p_idx;
            return;
        }
        
        let existing_idx = tree[node_idx].point_idx;
        tree[node_idx].point_idx = usize::MAX;
        
        let cx = (tree[node_idx].min_x + tree[node_idx].max_x) * 0.5;
        let cy = (tree[node_idx].min_y + tree[node_idx].max_y) * 0.5;
        let first_child = tree.len();
        tree[node_idx].child = first_child;
        
        for i in 0..4 {
            let dx = i & 1;
            let dy = (i >> 1) & 1;
            tree.push(Node {
                mass: 0.0, com_x: 0.0, com_y: 0.0,
                min_x: if dx == 0 { tree[node_idx].min_x } else { cx },
                min_y: if dy == 0 { tree[node_idx].min_y } else { cy },
                max_x: if dx == 0 { cx } else { tree[node_idx].max_x },
                max_y: if dy == 0 { cy } else { tree[node_idx].max_y },
                child: 0, point_idx: usize::MAX,
            });
        }
        
        let c1 = get_child(cx, cy, &pos[existing_idx]);
        insert(tree, first_child + c1, existing_idx, pos, depth + 1);
        let c2 = get_child(cx, cy, &pos[p_idx]);
        insert(tree, first_child + c2, p_idx, pos, depth + 1);
        return;
    }
    
    let cx = (tree[node_idx].min_x + tree[node_idx].max_x) * 0.5;
    let cy = (tree[node_idx].min_y + tree[node_idx].max_y) * 0.5;
    let c = get_child(cx, cy, &pos[p_idx]);
    let first_child = tree[node_idx].child;
    insert(tree, first_child + c, p_idx, pos, depth + 1);
}

fn compute_mass(tree: &mut Vec<Node>, node_idx: usize, pos: &[Point]) {
    if tree[node_idx].child == 0 {
        if tree[node_idx].point_idx != usize::MAX {
            tree[node_idx].mass = 1.0;
            let p = pos[tree[node_idx].point_idx];
            tree[node_idx].com_x = p.x;
            tree[node_idx].com_y = p.y;
        } else {
            tree[node_idx].mass = 0.0;
            tree[node_idx].com_x = 0.0;
            tree[node_idx].com_y = 0.0;
        }
        return;
    }
    
    let first_child = tree[node_idx].child;
    let mut mass = 0.0;
    let mut com_x = 0.0;
    let mut com_y = 0.0;
    for i in 0..4 {
        compute_mass(tree, first_child + i, pos);
        let child_mass = tree[first_child + i].mass;
        mass += child_mass;
        com_x += tree[first_child + i].com_x * child_mass;
        com_y += tree[first_child + i].com_y * child_mass;
    }
    
    tree[node_idx].mass = mass;
    if mass > 0.0 {
        tree[node_idx].com_x = com_x / mass;
        tree[node_idx].com_y = com_y / mass;
    }
}

fn build_tree(pos: &[Point]) -> Vec<Node> {
    let mut tree = Vec::with_capacity(pos.len() * 4);
    let mut min_x = f32::MAX;
    let mut min_y = f32::MAX;
    let mut max_x = f32::MIN;
    let mut max_y = f32::MIN;
    for p in pos {
        if p.x < min_x { min_x = p.x; }
        if p.y < min_y { min_y = p.y; }
        if p.x > max_x { max_x = p.x; }
        if p.y > max_y { max_y = p.y; }
    }
    
    let cx = (min_x + max_x) * 0.5;
    let cy = (min_y + max_y) * 0.5;
    let size = ((max_x - min_x).max(max_y - min_y) * 0.5 + 0.1).max(1.0);
    
    tree.push(Node {
        mass: 0.0, com_x: 0.0, com_y: 0.0,
        min_x: cx - size, min_y: cy - size,
        max_x: cx + size, max_y: cy + size,
        child: 0, point_idx: usize::MAX,
    });
    
    for (i, _) in pos.iter().enumerate() {
        insert(&mut tree, 0, i, pos, 0);
    }
    compute_mass(&mut tree, 0, pos);
    tree
}

fn compute_force(tree: &[Node], node_idx: usize, p: &Point, theta: f32, k: f32, fx: &mut f32, fy: &mut f32) {
    let node = &tree[node_idx];
    if node.mass == 0.0 { return; }
    
    let mut dx = node.com_x - p.x;
    let mut dy = node.com_y - p.y;
    let mut dist_sq = dx * dx + dy * dy;
    
    if dist_sq < 1e-6 {
        dx = 0.01;
        dy = 0.01;
        dist_sq = dx * dx + dy * dy;
    }
    
    let size = node.max_x - node.min_x;
    if node.child == 0 || size * size < theta * theta * dist_sq {
        let factor = node.mass * (k * k) / dist_sq;
        *fx -= dx * factor;
        *fy -= dy * factor;
    } else {
        let first_child = node.child;
        for i in 0..4 {
            compute_force(tree, first_child + i, p, theta, k, fx, fy);
        }
    }
}

fn initialize_positions(n: usize, adj: &[Vec<usize>]) -> Vec<Point> {
    let mut pos = vec![Point { x: 0.0, y: 0.0 }; n];
    let mut visited = vec![false; n];
    let mut components = Vec::new();
    
    for i in 0..n {
        if !visited[i] {
            let mut comp = Vec::new();
            let mut q = VecDeque::new();
            q.push_back(i);
            visited[i] = true;
            
            while let Some(u) = q.pop_front() {
                comp.push(u);
                for &v in &adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        q.push_back(v);
                    }
                }
            }
            components.push(comp);
        }
    }
    
    let box_size = (n as f32).sqrt() * 1.5;
    let mut rng_seed = 12345u32;
    let mut rand_f32 = || -> f32 {
        rng_seed = rng_seed.wrapping_mul(1664525).wrapping_add(1013904223);
        (rng_seed as f32) / (u32::MAX as f32)
    };
    
    let mut dist = vec![usize::MAX; n];
    
    for comp in components {
        if comp.is_empty() { continue; }
        
        let cx = (rand_f32() - 0.5) * box_size;
        let cy = (rand_f32() - 0.5) * box_size;
        
        let mut best_u = comp[0];
        let mut max_deg = adj[best_u].len();
        for &u in &comp {
            if adj[u].len() > max_deg {
                max_deg = adj[u].len();
                best_u = u;
            }
        }
        
        let mut q = VecDeque::new();
        q.push_back(best_u);
        dist[best_u] = 0;
        
        let mut layers: Vec<Vec<usize>> = Vec::new();
        
        while let Some(u) = q.pop_front() {
            let d = dist[u];
            if d >= layers.len() {
                layers.resize(d + 1, Vec::new());
            }
            layers[d].push(u);
            for &v in &adj[u] {
                if dist[v] == usize::MAX {
                    dist[v] = dist[u] + 1;
                    q.push_back(v);
                }
            }
        }
        
        for &u in &comp {
            dist[u] = usize::MAX;
        }
        
        let layer_dist = 1.0; 
        for (d, nodes) in layers.iter().enumerate() {
            let r = d as f32 * layer_dist;
            let count = nodes.len();
            if count == 1 {
                pos[nodes[0]] = Point { x: cx, y: cy };
            } else {
                let angle_step = 2.0 * std::f32::consts::PI / count as f32;
                for (i, &u) in nodes.iter().enumerate() {
                    let angle = i as f32 * angle_step;
                    pos[u] = Point {
                        x: cx + r * angle.cos(),
                        y: cy + r * angle.sin(),
                    };
                }
            }
        }
    }
    
    for (i, p) in pos.iter_mut().enumerate() {
        p.x += (i as f32 * 0.001).fract();
        p.y += (i as f32 * 0.0013).fract();
    }
    
    pos
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() { return; }
    let mut tokens = input.split_whitespace();
    
    let n_str = tokens.next();
    if n_str.is_none() { return; }
    let n: usize = match n_str.unwrap().parse() {
        Ok(v) => v,
        Err(_) => return,
    };
    let m: usize = match tokens.next().unwrap_or("0").parse() {
        Ok(v) => v,
        Err(_) => return,
    };
    
    let mut adj = vec![Vec::new(); n];
    let mut edges = Vec::with_capacity(m);
    
    for _ in 0..m {
        if let (Some(u_str), Some(v_str)) = (tokens.next(), tokens.next()) {
            let u: usize = u_str.parse().unwrap_or(n);
            let v: usize = v_str.parse().unwrap_or(n);
            if u < n && v < n {
                adj[u].push(v);
                adj[v].push(u);
                edges.push(Edge { u, v });
            }
        }
    }
    
    if n == 0 { return; }
    
    let k = 1.0;
    let time_limit = 28.0; 
    let max_iters = 800;
    
    let mut pos = initialize_positions(n, &adj);
    let start_time = Instant::now();
    
    let mut temp = (n as f32).sqrt() * 0.5;
    if temp < 5.0 { temp = 5.0; }
    let initial_temp = temp;
    
    let num_threads = std::thread::available_parallelism().map(|n| n.get()).unwrap_or(4).max(1);
    
    let mut disp = vec![Point { x: 0.0, y: 0.0 }; n];
    
    for iter in 0..max_iters {
        if iter % 5 == 0 && start_time.elapsed().as_secs_f64() > time_limit {
            break;
        }
        
        let tree = build_tree(&pos);
        
        let chunk_size = (pos.len() + num_threads - 1) / num_threads;
        let mut pos_chunks = pos.chunks(chunk_size);
        let mut disp_chunks = disp.chunks_mut(chunk_size);
        
        std::thread::scope(|s| {
            for (p_chunk, d_chunk) in pos_chunks.by_ref().zip(disp_chunks.by_ref()) {
                let tree_ref = &tree;
                s.spawn(move || {
                    for i in 0..p_chunk.len() {
                        let mut fx = 0.0;
                        let mut fy = 0.0;
                        compute_force(tree_ref, 0, &p_chunk[i], 1.2, k, &mut fx, &mut fy);
                        d_chunk[i].x = fx;
                        d_chunk[i].y = fy;
                    }
                });
            }
        });
        
        for e in &edges {
            let mut dx = pos[e.v].x - pos[e.u].x;
            let mut dy = pos[e.v].y - pos[e.u].y;
            let mut dist_sq = dx * dx + dy * dy;
            if dist_sq < 1e-6 {
                dx = 0.01;
                dy = 0.01;
                dist_sq = dx * dx + dy * dy;
            }
            let dist = dist_sq.sqrt();
            
            let factor = dist / k;
            let fx = dx * factor;
            let fy = dy * factor;
            
            disp[e.u].x += fx;
            disp[e.u].y += fy;
            disp[e.v].x -= fx;
            disp[e.v].y -= fy;
        }
        
        for i in 0..n {
            let dx = 0.0 - pos[i].x;
            let dy = 0.0 - pos[i].y;
            let factor = 0.05 * k;
            disp[i].x += dx * factor;
            disp[i].y += dy * factor;
        }
        
        for i in 0..n {
            let dx = disp[i].x;
            let dy = disp[i].y;
            let d_mag = (dx * dx + dy * dy).sqrt();
            
            if d_mag > 1e-6 {
                let move_dist = d_mag.min(temp);
                pos[i].x += dx / d_mag * move_dist;
                pos[i].y += dy / d_mag * move_dist;
            }
        }
        
        let progress = (iter as f32) / (max_iters as f32);
        temp = initial_temp * (1.0 - progress).max(0.01);
    }
    
    for p in pos {
        println!("{:.4} {:.4}", p.x, p.y);
    }
}
