use std::io::{self, BufRead};

struct State {
    w: usize,
    h: usize,
    chunks_per_row: usize,
    require_cycle: bool,
    u: Vec<u64>,
    visited: Vec<u64>,
    cnt: [usize; 2],
    degree: Vec<Vec<u8>>,
    deg_count: [usize; 5],
    unvisited_start_neighbors: usize,
    start_neighbors: Vec<Vec<bool>>,
    adj: Vec<Vec<Vec<(usize, usize)>>>,
}

impl State {
    #[inline(always)]
    fn bit_idx(&self, x: usize, y: usize) -> (usize, u64) {
        (y * self.chunks_per_row + (x / 64), 1 << (x % 64))
    }
    
    #[inline(always)]
    fn is_u_set(&self, x: usize, y: usize) -> bool {
        let (idx, mask) = self.bit_idx(x, y);
        (self.u[idx] & mask) != 0
    }
    
    #[inline(always)]
    fn clear_u(&mut self, x: usize, y: usize) {
        let (idx, mask) = self.bit_idx(x, y);
        self.u[idx] &= !mask;
    }
    
    #[inline(always)]
    fn set_u(&mut self, x: usize, y: usize) {
        let (idx, mask) = self.bit_idx(x, y);
        self.u[idx] |= mask;
    }

    fn do_move(&mut self, x: usize, y: usize) {
        self.clear_u(x, y);
        let c = (x + y) % 2;
        self.cnt[c] -= 1;
        let d = self.degree[x][y] as usize;
        self.deg_count[d] -= 1;
        
        if self.require_cycle && self.start_neighbors[x][y] {
            self.unvisited_start_neighbors -= 1;
        }
        
        for i in 0..self.adj[x][y].len() {
            let (nx, ny) = self.adj[x][y][i];
            if self.is_u_set(nx, ny) {
                let nd = self.degree[nx][ny] as usize;
                self.deg_count[nd] -= 1;
                self.degree[nx][ny] -= 1;
                self.deg_count[nd - 1] += 1;
            }
        }
    }

    fn undo_move(&mut self, x: usize, y: usize) {
        for i in 0..self.adj[x][y].len() {
            let (nx, ny) = self.adj[x][y][i];
            if self.is_u_set(nx, ny) {
                let nd = self.degree[nx][ny] as usize;
                self.deg_count[nd] -= 1;
                self.degree[nx][ny] += 1;
                self.deg_count[nd + 1] += 1;
            }
        }
        
        let d = self.degree[x][y] as usize;
        self.deg_count[d] += 1;
        
        if self.require_cycle && self.start_neighbors[x][y] {
            self.unvisited_start_neighbors += 1;
        }
        
        let c = (x + y) % 2;
        self.cnt[c] += 1;
        self.set_u(x, y);
    }
    
    fn is_connected(&mut self) -> bool {
        let mut start_idx = usize::MAX;
        for i in 0..self.u.len() {
            if self.u[i] != 0 {
                start_idx = i;
                break;
            }
        }
        if start_idx == usize::MAX { return true; }
        
        self.visited.fill(0);
        self.visited[start_idx] = self.u[start_idx] & !(self.u[start_idx] - 1);
        
        let mut changed = true;
        let chunks = self.chunks_per_row;
        
        while changed {
            changed = false;
            
            for y in 0..self.h {
                for c in 0..chunks {
                    let idx = y * chunks + c;
                    let mut v = self.visited[idx];
                    let orig = v;
                    
                    let mut left_shifted = v << 1;
                    if c > 0 {
                        left_shifted |= self.visited[idx - 1] >> 63;
                    }
                    
                    let mut right_shifted = v >> 1;
                    if c + 1 < chunks {
                        right_shifted |= self.visited[idx + 1] << 63;
                    }
                    
                    v |= left_shifted | right_shifted;
                    if y > 0 { v |= self.visited[idx - chunks]; }
                    if y + 1 < self.h { v |= self.visited[idx + chunks]; }
                    
                    v &= self.u[idx];
                    if v != orig {
                        self.visited[idx] = v;
                        changed = true;
                    }
                }
            }
            
            for y in (0..self.h).rev() {
                for c in (0..chunks).rev() {
                    let idx = y * chunks + c;
                    let mut v = self.visited[idx];
                    let orig = v;
                    
                    let mut left_shifted = v << 1;
                    if c > 0 {
                        left_shifted |= self.visited[idx - 1] >> 63;
                    }
                    
                    let mut right_shifted = v >> 1;
                    if c + 1 < chunks {
                        right_shifted |= self.visited[idx + 1] << 63;
                    }
                    
                    v |= left_shifted | right_shifted;
                    if y > 0 { v |= self.visited[idx - chunks]; }
                    if y + 1 < self.h { v |= self.visited[idx + chunks]; }
                    
                    v &= self.u[idx];
                    if v != orig {
                        self.visited[idx] = v;
                        changed = true;
                    }
                }
            }
        }
        
        for i in 0..self.u.len() {
            if self.visited[i] != self.u[i] { return false; }
        }
        true
    }

    fn solve(&mut self, cur_x: usize, cur_y: usize, path: &mut Vec<(usize, usize)>) -> bool {
        let u_size = self.cnt[0] + self.cnt[1];
        
        if u_size == 0 {
            if self.require_cycle {
                return self.start_neighbors[cur_x][cur_y];
            } else {
                return true;
            }
        }
        
        if self.require_cycle && self.unvisited_start_neighbors == 0 {
            return false;
        }
        
        let c = (cur_x + cur_y) % 2;
        let expected_diff = (u_size % 2) as i32;
        let actual_diff = self.cnt[1 - c] as i32 - self.cnt[c] as i32;
        if actual_diff != expected_diff {
            return false;
        }
        
        if u_size > 1 && self.deg_count[0] > 0 {
            return false;
        }
        
        if self.deg_count[1] > 2 {
            return false;
        }
        
        if !self.is_connected() {
            return false;
        }
        
        let mut neighbors = [(0, 0, 0); 4];
        let mut n_len = 0;
        
        for i in 0..self.adj[cur_x][cur_y].len() {
            let (nx, ny) = self.adj[cur_x][cur_y][i];
            if self.is_u_set(nx, ny) {
                neighbors[n_len] = (nx, ny, self.degree[nx][ny]);
                n_len += 1;
            }
        }
        
        let slice = &mut neighbors[..n_len];
        slice.sort_unstable_by_key(|k| k.2);
        
        for i in 0..n_len {
            let (nx, ny, _) = slice[i];
            self.do_move(nx, ny);
            path.push((nx, ny));
            
            if self.solve(nx, ny, path) {
                return true;
            }
            
            path.pop();
            self.undo_move(nx, ny);
        }
        
        false
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().filter_map(Result::ok);
    
    let first_line = match lines.next() {
        Some(l) => l,
        None => return,
    };
    
    let parts: Vec<usize> = first_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    if parts.len() < 4 {
        println!("0");
        return;
    }
    
    let mut w = parts[0];
    let mut h = parts[1];
    let obs_count = parts[2];
    let require_cycle = parts[3] != 0;
    
    let transposed = w > h;
    if transposed {
        std::mem::swap(&mut w, &mut h);
    }
    
    let mut obstacles = vec![vec![false; h]; w];
    for _ in 0..obs_count {
        if let Some(line) = lines.next() {
            let mut parts = line.split_whitespace().map(|s| s.parse::<usize>().unwrap());
            let mut x = parts.next().unwrap();
            let mut y = parts.next().unwrap();
            if transposed {
                std::mem::swap(&mut x, &mut y);
            }
            obstacles[x][y] = true;
        }
    }
    
    if obstacles[0][0] {
        println!("0");
        return;
    }

    let chunks_per_row = (w + 63) / 64;
    let mut state = State {
        w,
        h,
        chunks_per_row,
        require_cycle,
        u: vec![0; h * chunks_per_row],
        visited: vec![0; h * chunks_per_row],
        cnt: [0; 2],
        degree: vec![vec![0; h]; w],
        deg_count: [0; 5],
        unvisited_start_neighbors: 0,
        start_neighbors: vec![vec![false; h]; w],
        adj: vec![vec![Vec::new(); h]; w],
    };
    
    let mut total_cells = 0;
    for x in 0..w {
        for y in 0..h {
            if obstacles[x][y] { continue; }
            
            state.set_u(x, y);
            let c = (x + y) % 2;
            state.cnt[c] += 1;
            total_cells += 1;
            
            let mut nbors = Vec::with_capacity(4);
            if x > 0 && !obstacles[x-1][y] { nbors.push((x-1, y)); }
            if x + 1 < w && !obstacles[x+1][y] { nbors.push((x+1, y)); }
            if y > 0 && !obstacles[x][y-1] { nbors.push((x, y-1)); }
            if y + 1 < h && !obstacles[x][y+1] { nbors.push((x, y+1)); }
            
            state.degree[x][y] = nbors.len() as u8;
            state.deg_count[nbors.len()] += 1;
            state.adj[x][y] = nbors;
            
            if (x == 1 && y == 0) || (x == 0 && y == 1) {
                state.start_neighbors[x][y] = true;
                state.unvisited_start_neighbors += 1;
            }
        }
    }
    
    if require_cycle && total_cells % 2 != 0 {
        println!("0");
        return;
    }
    
    let expected_b = total_cells / 2 + total_cells % 2;
    let expected_w = total_cells / 2;
    if state.cnt[0] != expected_b || state.cnt[1] != expected_w {
        println!("0");
        return;
    }
    
    let mut path = Vec::with_capacity(total_cells);
    path.push((0, 0));
    state.do_move(0, 0);
    
    if state.solve(0, 0, &mut path) {
        println!("{}", path.len());
        for &(x, y) in &path {
            if transposed {
                println!("{} {}", y, x);
            } else {
                println!("{} {}", x, y);
            }
        }
    } else {
        println!("0");
    }
}
