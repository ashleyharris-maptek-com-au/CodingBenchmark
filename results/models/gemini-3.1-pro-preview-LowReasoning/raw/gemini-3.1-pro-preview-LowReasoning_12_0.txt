    public struct Vector3 {
        public double x, y, z;
        public Vector3(double x, double y, double z) { this.x = x; this.y = y; this.z = z; }
        public static Vector3 Cross(Vector3 a, Vector3 b) {
            return new Vector3(
                a.y * b.z - a.z * b.y,
                a.z * b.x - a.x * b.z,
                a.x * b.y - a.y * b.x
            );
        }
        public static double Dot(Vector3 a, Vector3 b) {
            return a.x * b.x + a.y * b.y + a.z * b.z;
        }
        public double Length() {
            return Math.Sqrt(x*x + y*y + z*z);
        }
        public Vector3 Normalized() {
            double len = Length();
            if (len < 1e-9) return new Vector3(0,0,0);
            return new Vector3(x/len, y/len, z/len);
        }
    }

    public struct Quat {
        public double w, x, y, z;
        public Quat(double w, double x, double y, double z) {
            this.w = w; this.x = x; this.y = y; this.z = z;
        }
        public static Quat Multiply(Quat q1, Quat q2) {
            return new Quat(
                q1.w * q2.w - q1.x * q2.x - q1.y * q2.y - q1.z * q2.z,
                q1.w * q2.x + q1.x * q2.w + q1.y * q2.z - q1.z * q2.y,
                q1.w * q2.y - q1.x * q2.z + q1.y * q2.w + q1.z * q2.x,
                q1.w * q2.z + q1.x * q2.y - q1.y * q2.x + q1.z * q2.w
            );
        }
        public static Vector3 Rotate(Quat q, Vector3 v) {
            Quat qv = new Quat(0, v.x, v.y, v.z);
            Quat qInv = new Quat(q.w, -q.x, -q.y, -q.z);
            Quat qres = Multiply(Multiply(q, qv), qInv);
            return new Vector3(qres.x, qres.y, qres.z);
        }
    }

    class PackingConfig {
        public long Count;
        public long nx, ny, nz;
        public double min_x, min_y, min_z;
        public double bw, bh, bd;
        public Quat q;
    }

    enum Scale { Trivial, Medium, Large, Extreme }

    static class QuatHelper {
        public static List<Quat> GetOrthogonalRotations() {
            List<Quat> list = new List<Quat>();
            Queue<Quat> q = new Queue<Quat>();
            Quat id = new Quat(1, 0, 0, 0);
            list.Add(id);
            q.Enqueue(id);
            
            double root2 = Math.Sqrt(0.5);
            Quat[] axes = new Quat[] {
                new Quat(root2, root2, 0, 0),
                new Quat(root2, 0, root2, 0),
                new Quat(root2, 0, 0, root2)
            };
            
            while(q.Count > 0) {
                Quat curr = q.Dequeue();
                foreach (Quat axis in axes) {
                    Quat next = Quat.Multiply(curr, axis);
                    bool found = false;
                    foreach (Quat existing in list) {
                        double dot = existing.w * next.w + existing.x * next.x + existing.y * next.y + existing.z * next.z;
                        if (Math.Abs(dot) > 0.999) { found = true; break; }
                    }
                    if (!found) {
                        list.Add(next);
                        q.Enqueue(next);
                    }
                }
            }
            return list;
        }

        public static List<Quat> GetRandomRotations(int count, int seed = 42) {
            List<Quat> list = new List<Quat>();
            Random rnd = new Random(seed);
            for (int i=0; i<count; i++) {
                double u1 = rnd.NextDouble();
                double u2 = rnd.NextDouble();
                double u3 = rnd.NextDouble();
                double w = Math.Sqrt(1 - u1) * Math.Sin(2 * Math.PI * u2);
                double x = Math.Sqrt(1 - u1) * Math.Cos(2 * Math.PI * u2);
                double y = Math.Sqrt(u1) * Math.Sin(2 * Math.PI * u3);
                double z = Math.Sqrt(u1) * Math.Cos(2 * Math.PI * u3);
                double len = Math.Sqrt(w*w + x*x + y*y + z*z);
                list.Add(new Quat(w/len, x/len, y/len, z/len));
            }
            return list;
        }

        public static List<Quat> GetFaceAlignments(List<Vector3> vertices, List<int[]> faces) {
            List<Quat> rotations = new List<Quat>();
            rotations.Add(new Quat(1, 0, 0, 0));
            
            var faceAreas = new List<Tuple<int[], double>>();
            foreach (var face in faces) {
                if (face.Length >= 3) {
                    Vector3 v0 = vertices[face[0]];
                    Vector3 v1 = vertices[face[1]];
                    Vector3 v2 = vertices[face[2]];
                    Vector3 edge1 = new Vector3(v1.x - v0.x, v1.y - v0.y, v1.z - v0.z);
                    Vector3 edge2 = new Vector3(v2.x - v0.x, v2.y - v0.y, v2.z - v0.z);
                    double area = Vector3.Cross(edge1, edge2).Length();
                    faceAreas.Add(Tuple.Create(face, area));
                }
            }
            
            faceAreas.Sort((a, b) => b.Item2.CompareTo(a.Item2));
            int maxFaces = Math.Min(100, faceAreas.Count);
            
            for (int i = 0; i < maxFaces; i++) {
                var face = faceAreas[i].Item1;
                Vector3 v0 = vertices[face[0]];
                Vector3 v1 = vertices[face[1]];
                Vector3 v2 = vertices[face[2]];
                
                Vector3 edge1 = new Vector3(v1.x - v0.x, v1.y - v0.y, v1.z - v0.z);
                Vector3 edge2 = new Vector3(v2.x - v0.x, v2.y - v0.y, v2.z - v0.z);
                Vector3 normal = Vector3.Cross(edge1, edge2).Normalized();
                
                if (normal.Length() < 0.1) continue;
                
                Vector3 target = new Vector3(0, 0, 1);
                Vector3 axis = Vector3.Cross(normal, target);
                double dot = Vector3.Dot(normal, target);
                
                Quat alignZ;
                if (axis.Length() < 1e-5) {
                    if (dot > 0) {
                        alignZ = new Quat(1, 0, 0, 0);
                    } else {
                        alignZ = new Quat(0, 1, 0, 0); 
                    }
                } else {
                    axis = axis.Normalized();
                    double angle = Math.Acos(Math.Max(-1.0, Math.Min(1.0, dot)));
                    double halfAngle = angle / 2.0;
                    double sin = Math.Sin(halfAngle);
                    alignZ = new Quat(Math.Cos(halfAngle), axis.x * sin, axis.y * sin, axis.z * sin);
                }
                
                Quat z0 = new Quat(1, 0, 0, 0);
                Quat z90 = new Quat(Math.Sqrt(0.5), 0, 0, Math.Sqrt(0.5));
                Quat z180 = new Quat(0, 0, 0, 1);
                Quat z270 = new Quat(Math.Sqrt(0.5), 0, 0, -Math.Sqrt(0.5));
                
                rotations.Add(Quat.Multiply(z0, alignZ));
                rotations.Add(Quat.Multiply(z90, alignZ));
                rotations.Add(Quat.Multiply(z180, alignZ));
                rotations.Add(Quat.Multiply(z270, alignZ));
            }
            return rotations;
        }

        public static List<Quat> UniqueRotations(List<Quat> input) {
            List<Quat> res = new List<Quat>();
            foreach (var q in input) {
                bool duplicate = false;
                foreach (var e in res) {
                    double dot = q.w * e.w + q.x * e.x + q.y * e.y + q.z * e.z;
                    if (Math.Abs(dot) > 0.999) { duplicate = true; break; }
                }
                if (!duplicate) res.Add(q);
            }
            return res;
        }
    }

    abstract class PackingStrategy {
        public abstract PackingConfig Pack(List<Vector3> vertices, List<int[]> faces, double W, double H, double D);
        
        protected PackingConfig FindBest(List<Vector3> vertices, IEnumerable<Quat> rotations, double W, double H, double D) {
            PackingConfig best = new PackingConfig { Count = -1 };
            object lockObj = new object();
            
            Parallel.ForEach(rotations, q => {
                var config = EvaluateRotation(vertices, q, W, H, D);
                lock (lockObj) {
                    if (config.Count > best.Count) {
                        best = config;
                    }
                }
            });
            
            return best;
        }
        
        protected PackingConfig EvaluateRotation(List<Vector3> vertices, Quat q, double W, double H, double D) {
            double min_x = double.MaxValue, min_y = double.MaxValue, min_z = double.MaxValue;
            double max_x = double.MinValue, max_y = double.MinValue, max_z = double.MinValue;
            
            foreach (var v in vertices) {
                Vector3 rv = Quat.Rotate(q, v);
                if (rv.x < min_x) min_x = rv.x;
                if (rv.x > max_x) max_x = rv.x;
                if (rv.y < min_y) min_y = rv.y;
                if (rv.y > max_y) max_y = rv.y;
                if (rv.z < min_z) min_z = rv.z;
                if (rv.z > max_z) max_z = rv.z;
            }
            
            double bw = max_x - min_x;
            double bh = max_y - min_y;
            double bd = max_z - min_z;
            
            long nx = bw > 1e-9 ? (long)Math.Floor(W / bw + 1e-9) : 0;
            long ny = bh > 1e-9 ? (long)Math.Floor(H / bh + 1e-9) : 0;
            long nz = bd > 1e-9 ? (long)Math.Floor(D / bd + 1e-9) : 0;
            
            if (nx <= 0 || ny <= 0 || nz <= 0) {
                return new PackingConfig { Count = 0 };
            }
            
            return new PackingConfig { 
                Count = nx * ny * nz, 
                nx = nx, ny = ny, nz = nz,
                min_x = min_x, min_y = min_y, min_z = min_z,
                bw = bw, bh = bh, bd = bd,
                q = q
            };
        }
    }

    class HybridStrategy : PackingStrategy {
        public override PackingConfig Pack(List<Vector3> vertices, List<int[]> faces, double W, double H, double D) {
            List<Quat> rots = new List<Quat>();
            rots.AddRange(QuatHelper.GetOrthogonalRotations());
            rots.AddRange(QuatHelper.GetFaceAlignments(vertices, faces));
            rots.AddRange(QuatHelper.GetRandomRotations(10000, 42));
            return FindBest(vertices, QuatHelper.UniqueRotations(rots), W, H, D);
        }
    }

    class MediumStrategy : PackingStrategy {
        public override PackingConfig Pack(List<Vector3> vertices, List<int[]> faces, double W, double H, double D) {
            List<Quat> rots = new List<Quat>();
            rots.AddRange(QuatHelper.GetOrthogonalRotations());
            rots.AddRange(QuatHelper.GetFaceAlignments(vertices, faces));
            rots.AddRange(QuatHelper.GetRandomRotations(1000, 42));
            return FindBest(vertices, QuatHelper.UniqueRotations(rots), W, H, D);
        }
    }

    class LargeStrategy : PackingStrategy {
        public override PackingConfig Pack(List<Vector3> vertices, List<int[]> faces, double W, double H, double D) {
            List<Quat> rots = new List<Quat>();
            rots.AddRange(QuatHelper.GetOrthogonalRotations());
            rots.AddRange(QuatHelper.GetFaceAlignments(vertices, faces));
            return FindBest(vertices, QuatHelper.UniqueRotations(rots), W, H, D);
        }
    }

    class ExtremeStrategy : PackingStrategy {
        public override PackingConfig Pack(List<Vector3> vertices, List<int[]> faces, double W, double H, double D) {
            List<Quat> rots = new List<Quat>();
            rots.AddRange(QuatHelper.GetOrthogonalRotations());
            return FindBest(vertices, QuatHelper.UniqueRotations(rots), W, H, D);
        }
    }

    class Scanner {
        private readonly StreamReader _reader;
        private string[] _tokens = new string[0];
        private int _pos = 0;

        public Scanner(Stream stream) {
            _reader = new StreamReader(stream, System.Text.Encoding.UTF8, true, 65536);
        }

        public string Next() {
            while (_pos >= _tokens.Length) {
                string line = _reader.ReadLine();
                if (line == null) return null;
                _tokens = line.Split(new char[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
                _pos = 0;
            }
            return _tokens[_pos++];
        }

        public double NextDouble() {
            return double.Parse(Next(), CultureInfo.InvariantCulture);
        }

        public int NextInt() {
            return int.Parse(Next(), CultureInfo.InvariantCulture);
        }
    }

    class Program {
        static Scale GetScale(double W, double H, double D) {
            double m = Math.Max(W, Math.Max(H, D));
            if (m <= 5.0) return Scale.Trivial;
            if (m <= 20.0) return Scale.Medium;
            if (m <= 100.0) return Scale.Large;
            return Scale.Extreme;
        }

        static PackingStrategy GetStrategy(Scale scale) {
            switch (scale) {
                case Scale.Trivial: return new HybridStrategy();
                case Scale.Medium: return new MediumStrategy();
                case Scale.Large: return new LargeStrategy();
                case Scale.Extreme: default: return new ExtremeStrategy();
            }
        }

        static void PrintConfig(PackingConfig config) {
            if (config == null || config.Count <= 0) {
                Console.WriteLine("0");
                return;
            }
            
            const long MAX_OUTPUT_LINES = 10_000_000;
            long printCount = Math.Min(config.Count, MAX_OUTPUT_LINES);
            
            using (StreamWriter sw = new StreamWriter(Console.OpenStandardOutput(), new System.Text.UTF8Encoding(false), 65536)) {
                sw.WriteLine(printCount);
                long printed = 0;
                for (long i = 0; i < config.nx && printed < printCount; i++) {
                    for (long j = 0; j < config.ny && printed < printCount; j++) {
                        for (long k = 0; k < config.nz && printed < printCount; k++) {
                            double tx = i * config.bw - config.min_x;
                            double ty = j * config.bh - config.min_y;
                            double tz = k * config.bd - config.min_z;
                            sw.WriteLine($"{tx:F5} {ty:F5} {tz:F5} {config.q.w:F5} {config.q.x:F5} {config.q.y:F5} {config.q.z:F5}");
                            printed++;
                        }
                    }
                }
            }
        }

        static void Main(string[] args) {
            Thread.CurrentThread.CurrentCulture = CultureInfo.InvariantCulture;
            
            Scanner scanner = new Scanner(Console.OpenStandardInput());
            string vStr = scanner.Next();
            if (vStr == null) return;
            
            int V = int.Parse(vStr, CultureInfo.InvariantCulture);
            int F = scanner.NextInt();
            
            List<Vector3> vertices = new List<Vector3>(V);
            for (int i = 0; i < V; i++) {
                vertices.Add(new Vector3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble()));
            }
            
            List<int[]> faces = new List<int[]>(F);
            for (int i = 0; i < F; i++) {
                faces.Add(new int[] { scanner.NextInt(), scanner.NextInt(), scanner.NextInt() });
            }
            
            double W = scanner.NextDouble();
            double H = scanner.NextDouble();
            double D = scanner.NextDouble();
            
            Scale scale = GetScale(W, H, D);
            PackingStrategy strategy = GetStrategy(scale);
            PackingConfig bestConfig = strategy.Pack(vertices, faces, W, H, D);
            
            PrintConfig(bestConfig);
        }
    }
