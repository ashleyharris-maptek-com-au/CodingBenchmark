#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>
#include <random>
#include <cmath>

using namespace std;
using namespace std::chrono;

struct Rect {
    int orig_idx;
    int w, h;
    int board_idx;
    int x, y;
    int rotated;
};

struct FreeRect {
    int x, y, w, h;
};

struct Board {
    vector<FreeRect> free_rects;
    long long area_left;
    int max_w, max_h;
};

int N;
int BOARD_W, BOARD_H;
vector<Rect> orig_rects;

struct Solution {
    int num_boards = 1e9;
    long long tie_breaker = -1;
    vector<Rect> rects;
};

Solution best_solution;

inline void recalc_board(Board& b) {
    b.max_w = 0;
    b.max_h = 0;
    for (const auto& fr : b.free_rects) {
        if (fr.w > b.max_w) b.max_w = fr.w;
        if (fr.h > b.max_h) b.max_h = fr.h;
    }
}

inline void calculate_score(int fw, int fh, int w, int h, int place_rule, long long& score1, long long& score2) {
    int dw = fw - w;
    int dh = fh - h;
    if (place_rule == 0) { // BSSF
        score1 = std::min(dw, dh);
        score2 = std::max(dw, dh);
    } else if (place_rule == 1) { // BAF
        score1 = (long long)fw * fh;
        score2 = std::min(dw, dh);
    } else if (place_rule == 2) { // BLSF
        score1 = std::max(dw, dh);
        score2 = std::min(dw, dh);
    }
}

bool pack_item(Rect& item, vector<Board>& boards, int place_rule, int split_rule) {
    long long best_score1 = 3e18;
    long long best_score2 = 3e18;
    int best_board = -1;
    int best_fr_idx = -1;
    int best_rot = 0;
    
    long long item_area = (long long)item.w * item.h;
    
    for (int b = 0; b < (int)boards.size(); b++) {
        if (boards[b].area_left < item_area) continue;
        
        bool can_fit = (item.w <= boards[b].max_w && item.h <= boards[b].max_h) ||
                       (item.h <= boards[b].max_w && item.w <= boards[b].max_h);
        if (!can_fit) continue;
        
        for (int i = 0; i < (int)boards[b].free_rects.size(); i++) {
            const auto& fr = boards[b].free_rects[i];
            
            // Unrotated
            if (item.w <= fr.w && item.h <= fr.h) {
                long long s1, s2;
                calculate_score(fr.w, fr.h, item.w, item.h, place_rule, s1, s2);
                if (s1 < best_score1 || (s1 == best_score1 && s2 < best_score2)) {
                    best_score1 = s1; best_score2 = s2;
                    best_board = b; best_fr_idx = i; best_rot = 0;
                    if (s1 == 0 && s2 == 0) goto found_perfect;
                }
            }
            
            // Rotated
            if (item.h <= fr.w && item.w <= fr.h && item.w != item.h) {
                long long s1, s2;
                calculate_score(fr.w, fr.h, item.h, item.w, place_rule, s1, s2);
                if (s1 < best_score1 || (s1 == best_score1 && s2 < best_score2)) {
                    best_score1 = s1; best_score2 = s2;
                    best_board = b; best_fr_idx = i; best_rot = 1;
                    if (s1 == 0 && s2 == 0) goto found_perfect;
                }
            }
        }
    }
    
found_perfect:
    if (best_board == -1) return false;
    
    auto fr = boards[best_board].free_rects[best_fr_idx];
    boards[best_board].free_rects[best_fr_idx] = boards[best_board].free_rects.back();
    boards[best_board].free_rects.pop_back();
    
    int fit_w = best_rot ? item.h : item.w;
    int fit_h = best_rot ? item.w : item.h;
    
    item.board_idx = best_board;
    item.x = fr.x;
    item.y = fr.y;
    item.rotated = best_rot;
    
    FreeRect r1, r2;
    int dw = fr.w - fit_w;
    int dh = fr.h - fit_h;
    bool split_horizontal = false;
    
    if (split_rule == 0) { // SLLA
        split_horizontal = (dw >= dh);
    } else if (split_rule == 1) { // SSLA
        split_horizontal = (dw < dh);
    } else if (split_rule == 2) { // MAXAS
        long long area_horiz = std::max((long long)dw * fit_h, (long long)fr.w * dh);
        long long area_vert  = std::max((long long)fit_w * dh, (long long)dw * fr.h);
        split_horizontal = (area_horiz > area_vert);
    } else { // MINAS
        long long area_horiz = std::max((long long)dw * fit_h, (long long)fr.w * dh);
        long long area_vert  = std::max((long long)fit_w * dh, (long long)dw * fr.h);
        split_horizontal = (area_horiz <= area_vert);
    }
    
    if (split_horizontal) {
        r1 = {fr.x + fit_w, fr.y, dw, fit_h};
        r2 = {fr.x, fr.y + fit_h, fr.w, dh};
    } else {
        r1 = {fr.x, fr.y + fit_h, fit_w, dh};
        r2 = {fr.x + fit_w, fr.y, dw, fr.h};
    }
    
    if (r1.w > 0 && r1.h > 0) boards[best_board].free_rects.push_back(r1);
    if (r2.w > 0 && r2.h > 0) boards[best_board].free_rects.push_back(r2);
    
    boards[best_board].area_left -= item_area;
    recalc_board(boards[best_board]);
    
    return true;
}

template<typename TimePoint>
bool pack_all(vector<Rect>& rects, int place_rule, int split_rule, int& out_boards, long long& out_tie, TimePoint start_time, bool is_first_pass) {
    vector<Board> boards;
    boards.reserve(rects.size());
    out_boards = 0;
    out_tie = 0;
    
    for (int i = 0; i < (int)rects.size(); i++) {
        if (!is_first_pass && (i & 1023) == 0) {
            auto curr_time = high_resolution_clock::now();
            double elapsed = duration_cast<duration<double>>(curr_time - start_time).count();
            if (elapsed > 28.5) {
                return false; 
            }
        }
        
        if (!pack_item(rects[i], boards, place_rule, split_rule)) {
            Board new_board;
            new_board.free_rects.reserve(64);
            new_board.free_rects.push_back({0, 0, BOARD_W, BOARD_H});
            new_board.area_left = (long long)BOARD_W * BOARD_H;
            new_board.max_w = BOARD_W;
            new_board.max_h = BOARD_H;
            boards.push_back(std::move(new_board));
            
            bool ok = pack_item(rects[i], boards, place_rule, split_rule);
            if (!ok) {
                // Should logically never happen with valid constraints
            }
        }
    }
    
    out_boards = boards.size();
    if (out_boards > 0) {
        out_tie = boards.back().area_left;
    }
    return true;
}

void apply_sort(vector<Rect>& rects, int sort_rule, const vector<double>& noise) {
    if (sort_rule == 0) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double s_a = (long long)a.w * a.h * noise[a.orig_idx];
            double s_b = (long long)b.w * b.h * noise[b.orig_idx];
            return s_a > s_b;
        });
    } else if (sort_rule == 1) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double max_a = std::max(a.w, a.h) * noise[a.orig_idx];
            double max_b = std::max(b.w, b.h) * noise[b.orig_idx];
            if (max_a != max_b) return max_a > max_b;
            double min_a = std::min(a.w, a.h) * noise[a.orig_idx];
            double min_b = std::min(b.w, b.h) * noise[b.orig_idx];
            return min_a > min_b;
        });
    } else if (sort_rule == 2) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double s_a = (a.w + a.h) * noise[a.orig_idx];
            double s_b = (b.w + b.h) * noise[b.orig_idx];
            return s_a > s_b;
        });
    } else if (sort_rule == 3) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double min_a = std::min(a.w, a.h) * noise[a.orig_idx];
            double min_b = std::min(b.w, b.h) * noise[b.orig_idx];
            if (min_a != min_b) return min_a > min_b;
            double max_a = std::max(a.w, a.h) * noise[a.orig_idx];
            double max_b = std::max(b.w, b.h) * noise[b.orig_idx];
            return max_a > max_b;
        });
    } else if (sort_rule == 4) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double min_a = std::min(a.w, a.h);
            double min_b = std::min(b.w, b.h);
            double ratio_a = (min_a == 0 ? 0 : ((double)std::max(a.w, a.h) / min_a)) * noise[a.orig_idx];
            double ratio_b = (min_b == 0 ? 0 : ((double)std::max(b.w, b.h) / min_b)) * noise[b.orig_idx];
            return ratio_a > ratio_b;
        });
    } else if (sort_rule == 5) {
        std::sort(rects.begin(), rects.end(), [&](const Rect& a, const Rect& b) {
            double area_a = (long long)a.w * a.h * noise[a.orig_idx];
            double area_b = (long long)b.w * b.h * noise[b.orig_idx];
            if (area_a != area_b) return area_a > area_b;
            double max_a = std::max(a.w, a.h) * noise[a.orig_idx];
            double max_b = std::max(b.w, b.h) * noise[b.orig_idx];
            return max_a > max_b;
        });
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    auto start_time = high_resolution_clock::now();
    
    if (!(cin >> N >> BOARD_W >> BOARD_H)) return 0;
    
    orig_rects.resize(N);
    for (int i = 0; i < N; i++) {
        orig_rects[i].orig_idx = i;
        orig_rects[i].board_idx = -1;
        orig_rects[i].x = 0;
        orig_rects[i].y = 0;
        orig_rects[i].rotated = 0;
        cin >> orig_rects[i].w >> orig_rects[i].h;
    }
    
    int global_best_sort = 0;
    int global_best_place = 0;
    int global_best_split = 0;
    
    vector<double> no_noise(N, 1.0);
    
    for (int sort_rule = 0; sort_rule < 6; sort_rule++) {
        for (int place_rule = 0; place_rule < 3; place_rule++) {
            for (int split_rule = 0; split_rule < 4; split_rule++) {
                auto curr_time = high_resolution_clock::now();
                if (duration_cast<duration<double>>(curr_time - start_time).count() > 28.5) {
                    goto end_deterministic;
                }
                
                vector<Rect> rects = orig_rects;
                apply_sort(rects, sort_rule, no_noise);
                
                int out_boards = 0;
                long long out_tie = 0;
                
                bool is_first = (best_solution.num_boards == 1e9);
                if (pack_all(rects, place_rule, split_rule, out_boards, out_tie, start_time, is_first)) {
                    if (out_boards < best_solution.num_boards || 
                       (out_boards == best_solution.num_boards && out_tie > best_solution.tie_breaker)) {
                        best_solution.num_boards = out_boards;
                        best_solution.tie_breaker = out_tie;
                        best_solution.rects = rects; // copy
                        global_best_sort = sort_rule;
                        global_best_place = place_rule;
                        global_best_split = split_rule;
                    }
                }
            }
        }
    }
    
end_deterministic:
    
    mt19937 rng(42);
    vector<double> noise(N);
    
    while (true) {
        auto curr_time = high_resolution_clock::now();
        if (duration_cast<duration<double>>(curr_time - start_time).count() > 29.0) {
            break;
        }
        
        double noise_level = 0.01 + (rng() % 100) / 1000.0;
        for (int i = 0; i < N; i++) {
            noise[i] = 1.0 + (rng() % 10000 - 5000) / 5000.0 * noise_level;
        }
        
        vector<Rect> rects = orig_rects;
        apply_sort(rects, global_best_sort, noise);
        
        int out_boards = 0;
        long long out_tie = 0;
        
        bool is_first = (best_solution.num_boards == 1e9);
        if (pack_all(rects, global_best_place, global_best_split, out_boards, out_tie, start_time, is_first)) {
            if (out_boards < best_solution.num_boards || 
               (out_boards == best_solution.num_boards && out_tie > best_solution.tie_breaker)) {
                best_solution.num_boards = out_boards;
                best_solution.tie_breaker = out_tie;
                best_solution.rects = std::move(rects);
            }
        }
    }
    
    std::sort(best_solution.rects.begin(), best_solution.rects.end(), [](const Rect& a, const Rect& b) {
        return a.orig_idx < b.orig_idx;
    });
    
    if (best_solution.num_boards == 1e9) {
        // Fallback for safety, should never be reached unless input is totally empty
        cout << "0\n";
        return 0;
    }
    
    cout << best_solution.num_boards << "\n";
    for (int i = 0; i < N; i++) {
        cout << best_solution.rects[i].board_idx << " " 
             << best_solution.rects[i].x << " " 
             << best_solution.rects[i].y << " " 
             << best_solution.rects[i].rotated << "\n";
    }
    
    return 0;
}
