using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    struct Edge
    {
        public int to;
        public int weight;
        public int id;
    }

    static int nextEdgeId;
    static bool[] edgeUsed;
    static List<Edge>[] adj;

    static void AddDuplicateEdge(int u, int v, int w)
    {
        adj[u].Add(new Edge { to = v, weight = w, id = nextEdgeId });
        adj[v].Add(new Edge { to = u, weight = w, id = nextEdgeId });
        if (nextEdgeId >= edgeUsed.Length)
        {
            Array.Resize(ref edgeUsed, edgeUsed.Length * 2 + 1000);
        }
        nextEdgeId++;
    }

    static void Main(string[] args)
    {
        Scanner scanner = new Scanner(Console.OpenStandardInput());
        int N = scanner.NextInt();
        if (N == -1) return;
        int M = scanner.NextInt();

        adj = new List<Edge>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<Edge>();

        for (int i = 0; i < M; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            int w = scanner.NextInt();
            adj[u].Add(new Edge { to = v, weight = w, id = i });
            adj[v].Add(new Edge { to = u, weight = w, id = i });
        }

        if (M == 0)
        {
            Console.WriteLine("0");
            return;
        }

        int[] degree = new int[N];
        for (int i = 0; i < N; i++) degree[i] = adj[i].Count;

        List<int> oddList = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (degree[i] % 2 != 0) oddList.Add(i);
        }
        int[] oddVertices = oddList.ToArray();
        int k = oddVertices.Length;

        edgeUsed = new bool[Math.Max(M, 1)];
        nextEdgeId = M;

        int[] visitedGen = new int[N];
        int currentGen = 0;

        if (k > 0)
        {
            if (k <= 20)
            {
                // Exact Minimum Weight Perfect Matching using DP for small scale
                long[][] oddDist = new long[k][];
                int[][] oddParentNode = new int[k][];

                for (int i = 0; i < k; i++)
                {
                    oddDist[i] = new long[N];
                    oddParentNode[i] = new int[N];
                    Array.Fill(oddDist[i], long.MaxValue / 2);

                    PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
                    pq.Enqueue(oddVertices[i], 0);
                    oddDist[i][oddVertices[i]] = 0;

                    currentGen++;
                    visitedGen[oddVertices[i]] = currentGen;

                    while (pq.Count > 0)
                    {
                        pq.TryDequeue(out int curr, out long d);
                        if (d > oddDist[i][curr]) continue;

                        foreach (var edge in adj[curr])
                        {
                            int nxt = edge.to;
                            long nxt_d = d + edge.weight;
                            if (visitedGen[nxt] != currentGen || nxt_d < oddDist[i][nxt])
                            {
                                oddDist[i][nxt] = nxt_d;
                                visitedGen[nxt] = currentGen;
                                oddParentNode[i][nxt] = curr;
                                pq.Enqueue(nxt, nxt_d);
                            }
                        }
                    }
                }

                long[] dp = new long[1 << k];
                int[] dpPrev = new int[1 << k];
                Array.Fill(dp, long.MaxValue / 2);
                dp[0] = 0;

                for (int mask = 0; mask < (1 << k); mask++)
                {
                    if (dp[mask] == long.MaxValue / 2) continue;

                    int u = -1;
                    for (int i = 0; i < k; i++)
                    {
                        if ((mask & (1 << i)) == 0)
                        {
                            u = i;
                            break;
                        }
                    }
                    if (u == -1) continue;

                    for (int v = u + 1; v < k; v++)
                    {
                        if ((mask & (1 << v)) == 0)
                        {
                            int nextMask = mask | (1 << u) | (1 << v);
                            long cost = dp[mask] + oddDist[u][oddVertices[v]];
                            if (cost < dp[nextMask])
                            {
                                dp[nextMask] = cost;
                                dpPrev[nextMask] = mask;
                            }
                        }
                    }
                }

                int currMask = (1 << k) - 1;
                while (currMask > 0)
                {
                    int prevMask = dpPrev[currMask];
                    int diff = currMask ^ prevMask;
                    int u = -1, v = -1;
                    for (int i = 0; i < k; i++)
                    {
                        if ((diff & (1 << i)) != 0)
                        {
                            if (u == -1) u = i;
                            else v = i;
                        }
                    }

                    int currNode = oddVertices[v];
                    while (currNode != oddVertices[u])
                    {
                        int pNode = oddParentNode[u][currNode];
                        long w = oddDist[u][currNode] - oddDist[u][pNode];
                        AddDuplicateEdge(pNode, currNode, (int)w);
                        currNode = pNode;
                    }

                    currMask = prevMask;
                }
            }
            else
            {
                // Greedy Matching for highly optimized large scale graphs
                bool[] unmatchedNode = new bool[N];
                foreach (var o in oddVertices) unmatchedNode[o] = true;

                long[] dist = new long[N];
                int[] parentNode = new int[N];
                PriorityQueue<int, long> pq = new PriorityQueue<int, long>();

                for (int i = 0; i < k; i++)
                {
                    int u = oddVertices[i];
                    if (!unmatchedNode[u]) continue;

                    unmatchedNode[u] = false;

                    currentGen++;
                    pq.Enqueue(u, 0);
                    dist[u] = 0;
                    visitedGen[u] = currentGen;

                    int v = -1;
                    while (pq.Count > 0)
                    {
                        pq.TryDequeue(out int curr, out long d);
                        if (d > dist[curr]) continue;

                        if (unmatchedNode[curr])
                        {
                            v = curr;
                            break;
                        }

                        foreach (var edge in adj[curr])
                        {
                            int nxt = edge.to;
                            long nxt_d = d + edge.weight;
                            if (visitedGen[nxt] != currentGen || nxt_d < dist[nxt])
                            {
                                dist[nxt] = nxt_d;
                                visitedGen[nxt] = currentGen;
                                parentNode[nxt] = curr;
                                pq.Enqueue(nxt, nxt_d);
                            }
                        }
                    }

                    if (v != -1)
                    {
                        unmatchedNode[v] = false;
                        int currNode = v;
                        while (currNode != u)
                        {
                            int pNode = parentNode[currNode];
                            long w = dist[currNode] - dist[pNode];
                            AddDuplicateEdge(pNode, currNode, (int)w);
                            currNode = pNode;
                        }
                    }

                    pq.Clear();
                }
            }
        }

        // Hierholzer's algorithm to find the Eulerian Circuit
        List<int> circuit = new List<int>();
        Stack<int> stack = new Stack<int>();
        stack.Push(0);

        int[] head = new int[N];

        while (stack.Count > 0)
        {
            int curr = stack.Peek();
            bool foundUnused = false;

            while (head[curr] < adj[curr].Count)
            {
                var edge = adj[curr][head[curr]];
                head[curr]++;
                if (!edgeUsed[edge.id])
                {
                    edgeUsed[edge.id] = true;
                    stack.Push(edge.to);
                    foundUnused = true;
                    break;
                }
            }

            if (!foundUnused)
            {
                circuit.Add(curr);
                stack.Pop();
            }
        }

        // Output formatting: High throughput stream writer
        using (StreamWriter sw = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536))
        {
            for (int i = circuit.Count - 1; i >= 0; i--)
            {
                sw.Write(circuit[i]);
                if (i > 0) sw.Write(" ");
            }
            sw.WriteLine();
        }
    }
}

class Scanner
{
    Stream stream;
    byte[] buffer = new byte[1024 * 64];
    int head = 0;
    int tail = 0;

    public Scanner(Stream stream)
    {
        this.stream = stream;
    }

    public int NextInt()
    {
        int c = Read();
        while (c <= 32)
        {
            if (c == -1) return -1;
            c = Read();
        }
        bool neg = false;
        if (c == '-')
        {
            neg = true;
            c = Read();
        }
        int res = 0;
        while (c > 32)
        {
            res = res * 10 + (c - '0');
            c = Read();
        }
        return neg ? -res : res;
    }

    int Read()
    {
        if (head >= tail)
        {
            head = 0;
            tail = stream.Read(buffer, 0, buffer.Length);
            if (tail == 0) return -1;
        }
        return buffer[head++];
    }
}
