use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone)]
struct BBox {
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
}

#[derive(Clone)]
struct Polygon {
    vertices: Vec<(f64, f64)>,
    bbox: BBox,
    area: f64,
}

fn make_ccw(vertices: &mut Vec<(f64, f64)>) {
    let mut area = 0.0;
    let n = vertices.len();
    for i in 0..n {
        let j = (i + 1) % n;
        area += vertices[i].0 * vertices[j].1 - vertices[j].0 * vertices[i].1;
    }
    if area < 0.0 {
        vertices.reverse();
    }
}

impl Polygon {
    fn new(mut vertices: Vec<(f64, f64)>) -> Self {
        make_ccw(&mut vertices);
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        let mut area = 0.0;
        let n = vertices.len();
        for i in 0..n {
            let p = vertices[i];
            if p.0 < min_x { min_x = p.0; }
            if p.0 > max_x { max_x = p.0; }
            if p.1 < min_y { min_y = p.1; }
            if p.1 > max_y { max_y = p.1; }
            
            let j = (i + 1) % n;
            area += vertices[i].0 * vertices[j].1 - vertices[j].0 * vertices[i].1;
        }
        let bbox = BBox { min_x, max_x, min_y, max_y };
        Polygon { vertices, bbox, area: (area / 2.0).abs() }
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let new_verts = self.vertices.iter().map(|&(x, y)| (x + dx, y + dy)).collect();
        Polygon::new(new_verts)
    }
}

struct PieceInput {
    id: usize,
    poly: Polygon,
}

fn bbox_overlap(a: &BBox, b: &BBox) -> bool {
    if a.max_x < b.min_x - 1e-4 || b.max_x < a.min_x - 1e-4 { return false; }
    if a.max_y < b.min_y - 1e-4 || b.max_y < a.min_y - 1e-4 { return false; }
    true
}

fn point_on_segment(px: f64, py: f64, x1: f64, y1: f64, x2: f64, y2: f64) -> bool {
    let cross = (x2 - x1) * (py - y1) - (px - x1) * (y2 - y1);
    let len_sq = (x2 - x1).powi(2) + (y2 - y1).powi(2);
    if len_sq == 0.0 {
        return (px - x1).hypot(py - y1) < 1e-6;
    }
    let dist = cross.abs() / len_sq.sqrt();
    if dist > 1e-4 {
        return false;
    }
    let dot = (px - x1) * (x2 - x1) + (py - y1) * (y2 - y1);
    if dot < -1e-4 || dot > len_sq + 1e-4 {
        return false;
    }
    true
}

fn point_in_polygon(x: f64, y: f64, poly: &[(f64, f64)]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        let (xi, yi) = poly[i];
        let (xj, yj) = poly[j];
        
        if point_on_segment(x, y, xi, yi, xj, yj) {
            return true;
        }

        if ((yi > y) != (yj > y)) && (x < (xj - xi) * (y - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn point_strictly_inside(x: f64, y: f64, poly: &[(f64, f64)]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        let (xi, yi) = poly[i];
        let (xj, yj) = poly[j];
        
        if point_on_segment(x, y, xi, yi, xj, yj) {
            return false; 
        }

        if ((yi > y) != (yj > y)) && (x < (xj - xi) * (y - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn segments_intersect(p1: (f64, f64), p2: (f64, f64), q1: (f64, f64), q2: (f64, f64)) -> bool {
    fn ccw(a: (f64, f64), b: (f64, f64), c: (f64, f64)) -> f64 {
        (c.1 - a.1) * (b.0 - a.0) - (b.1 - a.1) * (c.0 - a.0)
    }
    let ccw1 = ccw(p1, p2, q1);
    let ccw2 = ccw(p1, p2, q2);
    let ccw3 = ccw(q1, q2, p1);
    let ccw4 = ccw(q1, q2, p2);
    
    if ccw1 * ccw2 < -1e-6 && ccw3 * ccw4 < -1e-6 {
        return true;
    }
    false
}

fn collinear_overlap(p1: (f64, f64), p2: (f64, f64), q1: (f64, f64), q2: (f64, f64)) -> bool {
    let cross1 = (p2.0 - p1.0) * (q1.1 - p1.1) - (p2.1 - p1.1) * (q1.0 - p1.0);
    let cross2 = (p2.0 - p1.0) * (q2.1 - p1.1) - (p2.1 - p1.1) * (q2.0 - p1.0);
    
    let len_sq = (p2.0 - p1.0).powi(2) + (p2.1 - p1.1).powi(2);
    if len_sq < 1e-7 { return false; }
    let dist1 = cross1.abs() / len_sq.sqrt();
    let dist2 = cross2.abs() / len_sq.sqrt();
    
    if dist1 > 1e-4 || dist2 > 1e-4 {
        return false;
    }
    
    let dot = |a: (f64, f64), b: (f64, f64)| a.0 * b.0 + a.1 * b.1;
    let dir = (p2.0 - p1.0, p2.1 - p1.1);
    
    let t_p1 = dot(p1, dir);
    let t_p2 = dot(p2, dir);
    let t_q1 = dot(q1, dir);
    let t_q2 = dot(q2, dir);
    
    let min_p = t_p1.min(t_p2);
    let max_p = t_p1.max(t_p2);
    let min_q = t_q1.min(t_q2);
    let max_q = t_q1.max(t_q2);
    
    if max_p > min_q + 1e-4 && max_q > min_p + 1e-4 {
        let dot_dir = (p2.0 - p1.0) * (q2.0 - q1.0) + (p2.1 - p1.1) * (q2.1 - q1.1);
        if dot_dir > 0.0 {
            return true;
        }
    }
    false
}

fn polygons_overlap(poly_a: &Polygon, poly_b: &Polygon) -> bool {
    if !bbox_overlap(&poly_a.bbox, &poly_b.bbox) {
        return false;
    }

    for i in 0..poly_a.vertices.len() {
        let p1 = poly_a.vertices[i];
        let p2 = poly_a.vertices[(i + 1) % poly_a.vertices.len()];
        
        let mid = ((p1.0 + p2.0) / 2.0, (p1.1 + p2.1) / 2.0);
        if point_strictly_inside(mid.0, mid.1, &poly_b.vertices) {
            return true;
        }
        if point_strictly_inside(p1.0, p1.1, &poly_b.vertices) {
            return true;
        }

        for j in 0..poly_b.vertices.len() {
            let q1 = poly_b.vertices[j];
            let q2 = poly_b.vertices[(j + 1) % poly_b.vertices.len()];
            
            if segments_intersect(p1, p2, q1, q2) {
                return true; 
            }
            if collinear_overlap(p1, p2, q1, q2) {
                return true;
            }
        }
    }

    for i in 0..poly_b.vertices.len() {
        let p1 = poly_b.vertices[i];
        let p2 = poly_b.vertices[(i + 1) % poly_b.vertices.len()];
        let mid = ((p1.0 + p2.0) / 2.0, (p1.1 + p2.1) / 2.0);
        if point_strictly_inside(mid.0, mid.1, &poly_a.vertices) {
            return true;
        }
        if point_strictly_inside(p1.0, p1.1, &poly_a.vertices) {
            return true;
        }
    }
    
    false
}

fn polygon_inside(poly_a: &Polygon, poly_b: &Polygon) -> bool {
    if poly_a.bbox.min_x < poly_b.bbox.min_x - 1e-4 ||
       poly_a.bbox.max_x > poly_b.bbox.max_x + 1e-4 ||
       poly_a.bbox.min_y < poly_b.bbox.min_y - 1e-4 ||
       poly_a.bbox.max_y > poly_b.bbox.max_y + 1e-4 {
        return false;
    }

    for &p in &poly_a.vertices {
        if !point_in_polygon(p.0, p.1, &poly_b.vertices) {
            return false;
        }
    }
    
    for i in 0..poly_a.vertices.len() {
        let p1 = poly_a.vertices[i];
        let p2 = poly_a.vertices[(i + 1) % poly_a.vertices.len()];
        let mid = ((p1.0 + p2.0) / 2.0, (p1.1 + p2.1) / 2.0);
        if !point_in_polygon(mid.0, mid.1, &poly_b.vertices) {
            return false;
        }
        
        for j in 0..poly_b.vertices.len() {
            let q1 = poly_b.vertices[j];
            let q2 = poly_b.vertices[(j + 1) % poly_b.vertices.len()];
            
            if segments_intersect(p1, p2, q1, q2) {
                return false;
            }
            
            let cross1 = (p2.0 - p1.0) * (q1.1 - p1.1) - (p2.1 - p1.1) * (q1.0 - p1.0);
            let cross2 = (p2.0 - p1.0) * (q2.1 - p1.1) - (p2.1 - p1.1) * (q2.0 - p1.0);
            
            let len_sq = (p2.0 - p1.0).powi(2) + (p2.1 - p1.1).powi(2);
            if len_sq > 1e-7 {
                let dist1 = cross1.abs() / len_sq.sqrt();
                let dist2 = cross2.abs() / len_sq.sqrt();
                
                if dist1 < 1e-4 && dist2 < 1e-4 {
                    let dot = |a: (f64, f64), b: (f64, f64)| a.0 * b.0 + a.1 * b.1;
                    let dir = (p2.0 - p1.0, p2.1 - p1.1);
                    
                    let t_p1 = dot(p1, dir);
                    let t_p2 = dot(p2, dir);
                    let t_q1 = dot(q1, dir);
                    let t_q2 = dot(q2, dir);
                    
                    let min_p = t_p1.min(t_p2);
                    let max_p = t_p1.max(t_p2);
                    let min_q = t_q1.min(t_q2);
                    let max_q = t_q1.max(t_q2);
                    
                    if max_p > min_q + 1e-4 && max_q > min_p + 1e-4 {
                        let dot_dir = (p2.0 - p1.0) * (q2.0 - q1.0) + (p2.1 - p1.1) * (q2.1 - q1.1);
                        if dot_dir < 0.0 {
                            return false; 
                        }
                    }
                }
            }
        }
    }
    true
}

fn rotate_polygon(poly: &Polygon, rad: f64) -> Polygon {
    let cos = rad.cos();
    let sin = rad.sin();
    
    let mut new_verts = Vec::new();
    for &(x, y) in &poly.vertices {
        new_verts.push((x * cos - y * sin, x * sin + y * cos));
    }
    
    let mut p = Polygon::new(new_verts);
    make_ccw(&mut p.vertices);
    p
}

fn enrich_vertices(vertices: &[(f64, f64)]) -> Vec<(f64, f64)> {
    let mut enriched = Vec::new();
    for i in 0..vertices.len() {
        let p1 = vertices[i];
        let p2 = vertices[(i + 1) % vertices.len()];
        enriched.push(p1);
        enriched.push(((p1.0 * 2.0 + p2.0) / 3.0, (p1.1 * 2.0 + p2.1) / 3.0));
        enriched.push(((p1.0 + p2.0 * 2.0) / 3.0, (p1.1 + p2.1 * 2.0) / 3.0));
    }
    enriched
}

fn get_potential_translations(piece: &Polygon, stock: &Polygon, placed_pieces: &[Polygon]) -> Vec<(f64, f64)> {
    let mut translations = Vec::new();
    
    let piece_enrich = enrich_vertices(&piece.vertices);
    let stock_enrich = enrich_vertices(&stock.vertices);
    
    for vp in &piece_enrich {
        for vs in &stock_enrich {
            translations.push((vs.0 - vp.0, vs.1 - vp.1));
        }
    }
    
    let recent_placed: Vec<_> = placed_pieces.iter().rev().take(50).collect();
    for p in recent_placed {
        let p_enrich = enrich_vertices(&p.vertices);
        for vp in &piece_enrich {
            for vo in &p_enrich {
                translations.push((vo.0 - vp.0, vo.1 - vp.1));
            }
        }
    }
    
    let bbox = &stock.bbox;
    let w = bbox.max_x - bbox.min_x;
    let h = bbox.max_y - bbox.min_y;
    
    let step = if placed_pieces.len() > 2000 {
        (w.max(h)) / 2.0
    } else if placed_pieces.len() > 200 {
        (w.max(h)) / 5.0
    } else {
        (w.max(h)) / 20.0
    };
    
    if step > 0.1 {
        let mut x = bbox.min_x;
        while x <= bbox.max_x {
            let mut y = bbox.min_y;
            while y <= bbox.max_y {
                translations.push((x - piece.bbox.min_x, y - piece.bbox.min_y));
                y += step;
            }
            x += step;
        }
    }
    
    translations.push((0.0, 0.0));
    
    translations.sort_by(|a, b| {
        let y_cmp = a.1.partial_cmp(&b.1).unwrap_or(std::cmp::Ordering::Equal);
        if y_cmp == std::cmp::Ordering::Equal {
            a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal)
        } else {
            y_cmp
        }
    });
    
    translations.dedup_by(|a, b| (a.0 - b.0).abs() < 1e-4 && (a.1 - b.1).abs() < 1e-4);
    
    translations.retain(|&(tx, ty)| {
        let p_min_x = piece.bbox.min_x + tx;
        let p_max_x = piece.bbox.max_x + tx;
        let p_min_y = piece.bbox.min_y + ty;
        let p_max_y = piece.bbox.max_y + ty;
        
        p_min_x >= stock.bbox.min_x - 1e-2 &&
        p_max_x <= stock.bbox.max_x + 1e-2 &&
        p_min_y >= stock.bbox.min_y - 1e-2 &&
        p_max_y <= stock.bbox.max_y + 1e-2
    });
    
    translations
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().filter_map(Result::ok);
    
    let first_line = match lines.next() {
        Some(l) => l,
        None => return,
    };
    let parts: Vec<usize> = first_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let sv = parts[0];
    let np = parts[1];
    
    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap();
        let parts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        stock_vertices.push((parts[0], parts[1]));
    }
    let stock_poly = Polygon::new(stock_vertices);
    
    let mut pieces = Vec::new();
    for i in 0..np {
        let pv: usize = lines.next().unwrap().trim().parse().unwrap();
        let mut p_verts = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap();
            let parts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
            p_verts.push((parts[0], parts[1]));
        }
        pieces.push(PieceInput {
            id: i,
            poly: Polygon::new(p_verts),
        });
    }
    
    pieces.sort_by(|a, b| b.poly.area.partial_cmp(&a.poly.area).unwrap());
    
    let mut stocks: Vec<Vec<Polygon>> = Vec::new();
    let mut placements = vec![None; np];
    
    let angles = vec![0.0, PI/2.0, PI, 3.0*PI/2.0];

    for piece_in in pieces {
        let mut placed = false;
        
        let mut rotated_pieces = Vec::new();
        for &angle in &angles {
            rotated_pieces.push((angle, rotate_polygon(&piece_in.poly, angle)));
        }
        
        for stock_idx in 0..stocks.len() {
            let placed_in_stock = &stocks[stock_idx];
            
            for (angle, r_piece) in &rotated_pieces {
                let translations = get_potential_translations(r_piece, &stock_poly, placed_in_stock);
                for &(tx, ty) in &translations {
                    let translated = r_piece.translate(tx, ty);
                    
                    if polygon_inside(&translated, &stock_poly) {
                        let mut overlap = false;
                        for p in placed_in_stock {
                            if polygons_overlap(&translated, p) {
                                overlap = true;
                                break;
                            }
                        }
                        if !overlap {
                            placements[piece_in.id] = Some((stock_idx, tx, ty, *angle));
                            stocks[stock_idx].push(translated);
                            placed = true;
                            break;
                        }
                    }
                }
                if placed { break; }
            }
            if placed { break; }
        }
        
        if !placed {
            let stock_idx = stocks.len();
            stocks.push(Vec::new());
            
            for (angle, r_piece) in &rotated_pieces {
                let translations = get_potential_translations(r_piece, &stock_poly, &stocks[stock_idx]);
                for &(tx, ty) in &translations {
                    let translated = r_piece.translate(tx, ty);
                    if polygon_inside(&translated, &stock_poly) {
                        placements[piece_in.id] = Some((stock_idx, tx, ty, *angle));
                        stocks[stock_idx].push(translated);
                        placed = true;
                        break;
                    }
                }
                if placed { break; }
            }
        }
    }
    
    println!("{}", stocks.len());
    for i in 0..np {
        let (s_idx, tx, ty, rot) = placements[i].unwrap();
        println!("{} {} {} {}", s_idx, tx, ty, rot);
    }
}
