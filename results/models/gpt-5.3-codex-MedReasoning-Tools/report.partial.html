#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iomanip>

#include <limits>
#include <numeric>
#include <unordered_map>

struct City {
    double x, y;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = a.x - b.x;
    const double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline double dist(const City& a, const City& b) {
    return std::sqrt(dist2(a, b));
}

std::vector<int> solve_exact_bruteforce(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    std::vector<int> perm;
    perm.reserve(n - 1);
    for (int i = 1; i < n; ++i) perm.push_back(i);

    std::vector<int> best = perm;
    double best_len = std::numeric_limits<double>::infinity();

    do {
        double len = 0.0;
        int prev = 0;
        for (int id : perm) {
            len += dist(c[prev], c[id]);
            prev = id;
        }
        len += dist(c[prev], c[0]); // implicit return edge
        if (len < best_len) {
            best_len = len;
            best = perm;
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    std::vector<int> order;
    order.reserve(n);
    order.push_back(0);
    order.insert(order.end(), best.begin(), best.end());
    return order;
}

void limited_2opt(std::vector<int>& route, const std::vector<City>& c, int max_window = 80, int max_passes = 2) {
    const int n = static_cast<int>(route.size());
    if (n < 5) return;

    for (int pass = 0; pass < max_passes; ++pass) {
        bool improved = false;
        for (int i = 1; i + 2 < n; ++i) { // keep route[0] fixed
            const int a = route[i - 1];
            const int b = route[i];
            const int j_end = std::min(n - 1, i + max_window);
            for (int j = i + 1; j <= j_end; ++j) {
                const int c1 = route[j];
                const int d = (j + 1 < n) ? route[j + 1] : route[0];

                const double before = dist2(c[a], c[b]) + dist2(c[c1], c[d]);
                const double after = dist2(c[a], c[c1]) + dist2(c[b], c[d]);

                if (after + 1e-15 < before) {
                    std::reverse(route.begin() + i, route.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

std::vector<int> solve_angle_then_2opt(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    std::vector<int> ids;
    ids.reserve(n - 1);
    for (int i = 1; i < n; ++i) ids.push_back(i);

    const City o = c[0];
    std::sort(ids.begin(), ids.end(), [&](int a, int b) {
        const double aa = std::atan2(c[a].y - o.y, c[a].x - o.x);
        const double bb = std::atan2(c[b].y - o.y, c[b].x - o.x);
        if (aa != bb) return aa < bb;
        return dist2(c[a], o) < dist2(c[b], o);
    });

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    route.insert(route.end(), ids.begin(), ids.end());

    limited_2opt(route, c);
    return route;
}

static inline uint64_t splitmix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

static inline uint64_t cell_key(int cx, int cy) {
    uint64_t ux = static_cast<uint32_t>(cx);
    uint64_t uy = static_cast<uint32_t>(cy);
    return (ux << 32) | uy;
}

std::vector<int> solve_grid_nearest_neighbor(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());

    double minx = c[0].x, maxx = c[0].x, miny = c[0].y, maxy = c[0].y;
    for (const auto& p : c) {
        minx = std::min(minx, p.x);
        maxx = std::max(maxx, p.x);
        miny = std::min(miny, p.y);
        maxy = std::max(maxy, p.y);
    }

    const double span = std::max(maxx - minx, maxy - miny);
    const double cell = std::max(1e-12, span / std::sqrt(static_cast<double>(n)));

    std::unordered_map<uint64_t, std::vector<int>> buckets;
    buckets.reserve(static_cast<size_t>(n * 1.3));

    auto to_cell = [&](const City& p) {
        int cx = static_cast<int>(std::floor((p.x - minx) / cell));
        int cy = static_cast<int>(std::floor((p.y - miny) / cell));
        return std::pair<int, int>(cx, cy);
    };

    for (int i = 1; i < n; ++i) {
        auto [cx, cy] = to_cell(c[i]);
        buckets[cell_key(cx, cy)].push_back(i);
    }

    std::vector<uint8_t> used(n, 0);
    used[0] = 1;

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);

    int curr = 0;
    int remaining = n - 1;

    while (remaining > 0) {
        auto [ccx, ccy] = to_cell(c[curr]);
        int best = -1;
        double bestd = std::numeric_limits<double>::infinity();

        const int max_r = 4096;
        for (int r = 0; r <= max_r; ++r) {
            bool looked = false;
            int x0 = ccx - r, x1 = ccx + r;
            int y0 = ccy - r, y1 = ccy + r;

            for (int x = x0; x <= x1; ++x) {
                for (int y : {y0, y1}) {
                    if (y0 == y1 && y != y0) continue;
                    auto it = buckets.find(cell_key(x, y));
                    if (it == buckets.end()) continue;
                    looked = true;
                    for (int id : it->second) {
                        if (used[id]) continue;
                        double d = dist2(c[curr], c[id]);
                        if (d < bestd) {
                            bestd = d;
                            best = id;
                        }
                    }
                }
            }

            for (int y = y0 + 1; y <= y1 - 1; ++y) {
                for (int x : {x0, x1}) {
                    if (x0 == x1 && x != x0) continue;
                    auto it = buckets.find(cell_key(x, y));
                    if (it == buckets.end()) continue;
                    looked = true;
                    for (int id : it->second) {
                        if (used[id]) continue;
                        double d = dist2(c[curr], c[id]);
                        if (d < bestd) {
                            bestd = d;
                            best = id;
                        }
                    }
                }
            }

            if (best != -1 && (r >= 1 || looked)) break;
        }

        if (best == -1) {
            for (int i = 1; i < n; ++i) {
                if (!used[i]) {
                    best = i;
                    break;
                }
            }
        }

        used[best] = 1;
        route.push_back(best);
        curr = best;
        --remaining;
    }

    return route;
}

static inline uint32_t part1by1(uint32_t x) {
    x &= 0x0000ffffu;
    x = (x ^ (x << 8)) & 0x00FF00FFu;
    x = (x ^ (x << 4)) & 0x0F0F0F0Fu;
    x = (x ^ (x << 2)) & 0x33333333u;
    x = (x ^ (x << 1)) & 0x55555555u;
    return x;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return (static_cast<uint64_t>(part1by1(y)) << 1) | part1by1(x);
}

std::vector<int> solve_morton(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    double minx = c[0].x, maxx = c[0].x, miny = c[0].y, maxy = c[0].y;
    for (const auto& p : c) {
        minx = std::min(minx, p.x);
        maxx = std::max(maxx, p.x);
        miny = std::min(miny, p.y);
        maxy = std::max(maxy, p.y);
    }

    const double dx = std::max(1e-12, maxx - minx);
    const double dy = std::max(1e-12, maxy - miny);

    struct Item { uint64_t key; int id; };
    std::vector<Item> items;
    items.reserve(n - 1);

    for (int i = 1; i < n; ++i) {
        double fx = (c[i].x - minx) / dx;
        double fy = (c[i].y - miny) / dy;
        fx = std::clamp(fx, 0.0, 1.0);
        fy = std::clamp(fy, 0.0, 1.0);
        uint32_t qx = static_cast<uint32_t>(fx * 65535.0 + 0.5);
        uint32_t qy = static_cast<uint32_t>(fy * 65535.0 + 0.5);
        uint64_t key = morton2D(qx, qy);
        key ^= splitmix64(static_cast<uint64_t>(i) * 0x9e3779b97f4a7c15ULL) & 0x3FFULL;
        items.push_back({key, i});
    }

    std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        if (a.key != b.key) return a.key < b.key;
        return a.id < b.id;
    });

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    for (const auto& it : items) route.push_back(it.id);
    return route;
}

}
