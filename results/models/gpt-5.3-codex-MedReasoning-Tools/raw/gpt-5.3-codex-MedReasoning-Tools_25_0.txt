using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    const double G = 6.67430e-11;
    const double Day = 86400.0;

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Normalized()
        {
            double n = Norm();
            return n > 0 ? this / n : new Vec3(0, 0, 0);
        }
    }

    class Body
    {
        public string Name = "";
        public double Mass;
        public Vec3 R;
        public Vec3 V;
    }

    class Ephem
    {
        public double DtDays;
        public int Steps;
        public Dictionary<string, Vec3[]> Pos = new Dictionary<string, Vec3[]>(StringComparer.OrdinalIgnoreCase);
        public Dictionary<string, Vec3[]> Vel = new Dictionary<string, Vec3[]>(StringComparer.OrdinalIgnoreCase);

        public Vec3 GetPos(string name, double tDays)
        {
            var a = Pos[name];
            return Lerp(a, tDays);
        }

        public Vec3 GetVel(string name, double tDays)
        {
            var a = Vel[name];
            return Lerp(a, tDays);
        }

        Vec3 Lerp(Vec3[] arr, double tDays)
        {
            if (tDays <= 0) return arr[0];
            double maxT = (Steps - 1) * DtDays;
            if (tDays >= maxT) return arr[Steps - 1];
            double u = tDays / DtDays;
            int i = (int)Math.Floor(u);
            double f = u - i;
            Vec3 a = arr[i], b = arr[i + 1];
            return a * (1.0 - f) + b * f;
        }
    }

    class Plan
    {
        public double Tl, Tm, Ti;
        public Vec3 Dv1, Dv2, Dv3;
        public double Score;
    }

    public static void Main()
    {
        string input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
        {
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        var tokens = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;

        int numBodies = ParseInt(tokens[p++]);
        double warningDays = ParseDouble(tokens[p++]);
        double impactorMass = ParseDouble(tokens[p++]);
        double dvBudget = ParseDouble(tokens[p++]);
        double asteroidMass = ParseDouble(tokens[p++]);
        double t0Unix = ParseDouble(tokens[p++]);

        var bodies = new List<Body>(numBodies);
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = tokens[p++];
            b.Mass = ParseDouble(tokens[p++]);
            b.R = new Vec3(ParseDouble(tokens[p++]), ParseDouble(tokens[p++]), ParseDouble(tokens[p++]));
            b.V = new Vec3(ParseDouble(tokens[p++]), ParseDouble(tokens[p++]), ParseDouble(tokens[p++]));
            bodies.Add(b);
        }

        string bspPath = p < tokens.Length ? tokens[p] : "";

        int earthIdx = FindBody(bodies, "Earth");
        int asteroidIdx = FindBody(bodies, "Asteroid");
        int sunIdx = FindBody(bodies, "Sun");

        if (earthIdx < 0 || asteroidIdx < 0)
        {
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        if (sunIdx < 0)
        {
            sunIdx = HeaviestBodyIndex(bodies);
        }

        double dtDays = warningDays > 8000 ? 2.0 : (warningDays > 2000 ? 1.0 : 0.5);
        var ephem = BuildEphemeris(bodies, warningDays, dtDays);

        double impactEstimate = EstimateImpactTime(ephem, "Earth", "Asteroid", warningDays);
        double tLatestIntercept = Math.Max(2.0, Math.Min(warningDays, impactEstimate - 0.5));

        Plan best = SearchPlan(ephem, bodies[sunIdx].Mass, dvBudget, warningDays, tLatestIntercept, impactEstimate, impactorMass, asteroidMass);

        if (best == null)
        {
            // Safe fallback: immediate launch toward asteroid with capped dv.
            double tl = 0.0;
            double ti = Math.Max(1.0, Math.Min(warningDays, impactEstimate > 1.0 ? impactEstimate * 0.7 : warningDays * 0.7));
            Vec3 rE = ephem.GetPos("Earth", tl);
            Vec3 vE = ephem.GetVel("Earth", tl);
            Vec3 rA = ephem.GetPos("Asteroid", ti);
            Vec3 vReq = (rA - rE) / (Math.Max(0.1, ti - tl) * Day);
            Vec3 dv1 = vReq - vE;
            double n1 = dv1.Norm();
            if (n1 > dvBudget && n1 > 0) dv1 = dv1 * (dvBudget / n1);
            Vec3 relDir = (vReq - ephem.GetVel("Asteroid", ti)).Normalized();
            Vec3 dv3 = relDir * Math.Max(0, dvBudget - dv1.Norm());

            Console.WriteLine("2");
            PrintBurn(tl, dv1);
            PrintBurn(ti, dv3);
            return;
        }

        Console.WriteLine("3");
        PrintBurn(best.Tl, best.Dv1);
        PrintBurn(best.Tm, best.Dv2);
        PrintBurn(best.Ti, best.Dv3);
    }

    static Plan SearchPlan(Ephem ephem, double sunMass, double dvBudget, double warningDays, double tLatestIntercept, double impactEstimate, double impactorMass, double asteroidMass)
    {
        double launchMax = Math.Max(0.0, Math.Min(warningDays, tLatestIntercept - 1.0));
        if (launchMax <= 0.0) return null;

        Plan best = null;
        double launchStep = Math.Max(1.0, warningDays / 40.0);
        double tofMin = 5.0;
        double tofMax = Math.Max(tofMin + 1.0, tLatestIntercept);

        for (double tl = 0.0; tl <= launchMax + 1e-9; tl += launchStep)
        {
            double localTofMax = Math.Min(tofMax, tLatestIntercept - tl);
            if (localTofMax < tofMin) continue;

            int tofSamples = 18;
            for (int k = 0; k <= tofSamples; k++)
            {
                double tof = tofMin + (localTofMax - tofMin) * k / Math.Max(1, tofSamples);
                double ti = tl + tof;
                if (ti > tLatestIntercept) continue;

                Plan c = BuildCandidate(ephem, sunMass, dvBudget, tl, ti, impactEstimate, impactorMass, asteroidMass);
                if (c == null) continue;
                if (best == null || c.Score > best.Score) best = c;
            }
        }

        // Refine around best if found.
        if (best != null)
        {
            Plan refined = null;
            double tl0 = Math.Max(0.0, best.Tl - launchStep);
            double tl1 = Math.Min(launchMax, best.Tl + launchStep);
            double ti0 = Math.Max(best.Tl + 2.0, best.Ti - 10.0);
            double ti1 = Math.Min(tLatestIntercept, best.Ti + 10.0);

            for (double tl = tl0; tl <= tl1 + 1e-9; tl += Math.Max(0.5, launchStep / 6.0))
            {
                for (double ti = ti0; ti <= ti1 + 1e-9; ti += 0.5)
                {
                    if (ti <= tl + 1.0) continue;
                    Plan c = BuildCandidate(ephem, sunMass, dvBudget, tl, ti, impactEstimate, impactorMass, asteroidMass);
                    if (c == null) continue;
                    if (refined == null || c.Score > refined.Score) refined = c;
                }
            }
            if (refined != null) best = refined;
        }

        return best;
    }

    static Plan BuildCandidate(Ephem ephem, double sunMass, double dvBudget, double tl, double ti, double impactEstimate, double impactorMass, double asteroidMass)
    {
        if (ti <= tl + 0.2) return null;

        Vec3 rE = ephem.GetPos("Earth", tl);
        Vec3 vE = ephem.GetVel("Earth", tl);
        Vec3 rA = ephem.GetPos("Asteroid", ti);
        Vec3 vA = ephem.GetVel("Asteroid", ti);

        // First guess: straight transfer in inertial frame.
        Vec3 vReq = (rA - rE) / ((ti - tl) * Day);
        Vec3 dv1 = vReq - vE;

        // Mid-course correction at halfway after propagating under Sun gravity.
        double tm = 0.5 * (tl + ti);
        var scMid = PropagateSun(rE, vE + dv1, tl, tm, ephem, sunMass);
        Vec3 rM = scMid.r;
        Vec3 vM = scMid.v;

        Vec3 rA2 = ephem.GetPos("Asteroid", ti);
        Vec3 vReq2 = (rA2 - rM) / ((ti - tm) * Day);
        Vec3 dv2 = vReq2 - vM;

        double used = dv1.Norm() + dv2.Norm();
        if (used > dvBudget * 1.0001) return null;

        Vec3 rel = vReq2 - vA;
        double rem = Math.Max(0.0, dvBudget - used);

        // Spend remaining dv to maximize impact speed.
        Vec3 dir = rel.Norm() > 1e-9 ? rel.Normalized() : (rA - ephem.GetPos("Earth", ti)).Normalized();
        Vec3 dv3 = dir * rem;

        double vImpact = (rel + dv3).Norm();
        double leadTime = Math.Max(0.0, impactEstimate - ti);

        // Deflection proxy: delivered momentum * leverage time.
        double momentum = impactorMass * vImpact;
        double massFactor = impactorMass / Math.Max(1.0, asteroidMass);
        double score = momentum * (1.0 + leadTime / 365.25) * (1.0 + Math.Sqrt(Math.Max(0.0, massFactor)));

        return new Plan
        {
            Tl = tl,
            Tm = tm,
            Ti = ti,
            Dv1 = dv1,
            Dv2 = dv2,
            Dv3 = dv3,
            Score = score
        };
    }

    static (Vec3 r, Vec3 v) PropagateSun(Vec3 r0, Vec3 v0, double t0Days, double t1Days, Ephem ephem, double sunMass)
    {
        if (t1Days <= t0Days) return (r0, v0);

        double totalSec = (t1Days - t0Days) * Day;
        int n = Math.Max(1, (int)Math.Ceiling((t1Days - t0Days) / 0.25));
        double h = totalSec / n;

        Vec3 r = r0, v = v0;
        double tSec = t0Days * Day;

        for (int i = 0; i < n; i++)
        {
            double tA = tSec / Day;
            double tB = (tSec + 0.5 * h) / Day;
            double tC = (tSec + h) / Day;

            Vec3 sunA = ephem.GetPos("Sun", tA);
            Vec3 sunB = ephem.GetPos("Sun", tB);
            Vec3 sunC = ephem.GetPos("Sun", tC);

            Vec3 a1 = AccFromSun(r, sunA, sunMass);
            Vec3 k1r = v;
            Vec3 k1v = a1;

            Vec3 r2 = r + k1r * (0.5 * h);
            Vec3 v2 = v + k1v * (0.5 * h);
            Vec3 a2 = AccFromSun(r2, sunB, sunMass);
            Vec3 k2r = v2;
            Vec3 k2v = a2;

            Vec3 r3 = r + k2r * (0.5 * h);
            Vec3 v3 = v + k2v * (0.5 * h);
            Vec3 a3 = AccFromSun(r3, sunB, sunMass);
            Vec3 k3r = v3;
            Vec3 k3v = a3;

            Vec3 r4 = r + k3r * h;
            Vec3 v4 = v + k3v * h;
            Vec3 a4 = AccFromSun(r4, sunC, sunMass);
            Vec3 k4r = v4;
            Vec3 k4v = a4;

            r = r + (k1r + k2r * 2.0 + k3r * 2.0 + k4r) * (h / 6.0);
            v = v + (k1v + k2v * 2.0 + k3v * 2.0 + k4v) * (h / 6.0);

            tSec += h;
        }

        return (r, v);
    }

    static Vec3 AccFromSun(Vec3 r, Vec3 sunPos, double sunMass)
    {
        Vec3 dr = sunPos - r;
        double d2 = dr.X * dr.X + dr.Y * dr.Y + dr.Z * dr.Z + 1e-9;
        double invR3 = 1.0 / (Math.Sqrt(d2) * d2);
        return dr * (G * sunMass * invR3);
    }

    static Ephem BuildEphemeris(List<Body> initialBodies, double warningDays, double dtDays)
    {
        int n = initialBodies.Count;
        int steps = Math.Max(2, (int)Math.Ceiling(warningDays / dtDays) + 1);

        var names = new string[n];
        var masses = new double[n];
        var r = new Vec3[n];
        var v = new Vec3[n];

        var e = new Ephem { DtDays = dtDays, Steps = steps };

        for (int i = 0; i < n; i++)
        {
            names[i] = initialBodies[i].Name;
            masses[i] = initialBodies[i].Mass;
            r[i] = initialBodies[i].R;
            v[i] = initialBodies[i].V;
            e.Pos[names[i]] = new Vec3[steps];
            e.Vel[names[i]] = new Vec3[steps];
            e.Pos[names[i]][0] = r[i];
            e.Vel[names[i]][0] = v[i];
        }

        double h = dtDays * Day;
        var a = new Vec3[n];

        for (int s = 1; s < steps; s++)
        {
            ComputeAcc(r, masses, a);

            // Velocity-Verlet / KDK-like step.
            for (int i = 0; i < n; i++)
            {
                v[i] = v[i] + a[i] * (0.5 * h);
                r[i] = r[i] + v[i] * h;
            }

            ComputeAcc(r, masses, a);
            for (int i = 0; i < n; i++)
            {
                v[i] = v[i] + a[i] * (0.5 * h);
                e.Pos[names[i]][s] = r[i];
                e.Vel[names[i]][s] = v[i];
            }
        }

        return e;
    }

    static void ComputeAcc(Vec3[] r, double[] m, Vec3[] a)
    {
        int n = r.Length;
        for (int i = 0; i < n; i++) a[i] = new Vec3(0, 0, 0);

        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                Vec3 d = r[j] - r[i];
                double d2 = d.X * d.X + d.Y * d.Y + d.Z * d.Z + 1e-9;
                double invR = 1.0 / Math.Sqrt(d2);
                double invR3 = invR / d2;
                Vec3 f = d * (G * invR3);
                a[i] = a[i] + f * m[j];
                a[j] = a[j] - f * m[i];
            }
        }
    }

    static double EstimateImpactTime(Ephem ephem, string earthName, string asteroidName, double warningDays)
    {
        double bestT = warningDays;
        double bestD = double.MaxValue;

        double step = Math.Max(0.25, Math.Min(10.0, warningDays / 300.0));
        for (double t = 0.0; t <= warningDays + 1e-9; t += step)
        {
            Vec3 re = ephem.GetPos(earthName, t);
            Vec3 ra = ephem.GetPos(asteroidName, t);
            double d = (ra - re).Norm();
            if (d < bestD)
            {
                bestD = d;
                bestT = t;
            }
        }

        return bestT;
    }

    static int FindBody(List<Body> bodies, string name)
    {
        for (int i = 0; i < bodies.Count; i++)
        {
            if (string.Equals(bodies[i].Name, name, StringComparison.OrdinalIgnoreCase)) return i;
        }
        return -1;
    }

    static int HeaviestBodyIndex(List<Body> bodies)
    {
        int idx = 0;
        double maxMass = double.MinValue;
        for (int i = 0; i < bodies.Count; i++)
        {
            if (bodies[i].Mass > maxMass)
            {
                maxMass = bodies[i].Mass;
                idx = i;
            }
        }
        return idx;
    }

    static int ParseInt(string s) => int.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);
    static double ParseDouble(string s) => double.Parse(s, NumberStyles.Float | NumberStyles.AllowThousands, CultureInfo.InvariantCulture);

    static void PrintBurn(double tDays, Vec3 dv)
    {
        Console.WriteLine(FormattableString.Invariant($"{tDays:F9} {dv.X:E17} {dv.Y:E17} {dv.Z:E17}"));
    }
}
