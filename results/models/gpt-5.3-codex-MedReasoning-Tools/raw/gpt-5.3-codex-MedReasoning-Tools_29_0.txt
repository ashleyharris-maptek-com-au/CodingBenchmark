using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buf = new byte[1 << 20];
        private int _len;
        private int _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sign;
        }
    }

    private sealed class Graph
    {
        public readonly int N;
        public readonly int M;
        public readonly int[][] Adj;
        public readonly int[] Degree;

        public Graph(int n, int m, int[][] adj, int[] degree)
        {
            N = n;
            M = m;
            Adj = adj;
            Degree = degree;
        }
    }

    private sealed class MaxCliqueSolver
    {
        private readonly Graph _g;
        private readonly long _deadlineTicks;

        public MaxCliqueSolver(Graph g, TimeSpan budget)
        {
            _g = g;
            _deadlineTicks = DateTime.UtcNow.Add(budget).Ticks;
        }

        private bool TimeExceeded() => DateTime.UtcNow.Ticks >= _deadlineTicks;

        public int[] Solve()
        {
            if (_g.N == 0) return Array.Empty<int>();

            int[] heuristic = ParallelGreedySearch();

            // Exact search is only practical on smaller instances.
            // We still use it aggressively for medium sizes when time allows.
            if (_g.N <= 1400 && !TimeExceeded())
            {
                int[] exact = SolveExactWithBranchAndBound(heuristic);
                if (exact.Length >= heuristic.Length) return exact;
            }

            return heuristic;
        }

        private int[] ParallelGreedySearch()
        {
            int n = _g.N;
            int[] deg = _g.Degree;
            int[][] adj = _g.Adj;

            int[] order = Enumerable.Range(0, n).ToArray();
            Array.Sort(order, (a, b) =>
            {
                int c = deg[b].CompareTo(deg[a]);
                return c != 0 ? c : a.CompareTo(b);
            });

            var globalBest = new List<int>();
            object gate = new object();

            int seedCount;
            if (n <= 2000) seedCount = n;
            else if (n <= 50000) seedCount = Math.Min(n, 4000);
            else seedCount = Math.Min(n, 12000);

            int[] seedVertices = new int[seedCount];
            Array.Copy(order, seedVertices, seedCount);

            Parallel.ForEach(
                Partitioner(seedVertices.Length),
                new ParallelOptions { MaxDegreeOfParallelism = Math.Max(1, Environment.ProcessorCount) },
                () => new WorkerState(n),
                (range, _, state) =>
                {
                    for (int i = range.start; i < range.end; i++)
                    {
                        if (TimeExceeded()) break;
                        int seed = seedVertices[i];
                        BuildGreedyCliqueFromSeed(seed, state, ref globalBest, gate);

                        // Randomized perturbation per seed to diversify local maxima.
                        if ((i & 3) == 0 && adj[seed].Length > 0)
                        {
                            int alt = adj[seed][state.Rng.Next(adj[seed].Length)];
                            BuildGreedyCliqueFromSeed(alt, state, ref globalBest, gate);
                        }
                    }
                    return state;
                },
                _ => { }
            );

            if (globalBest.Count == 0)
            {
                int v = order.Length > 0 ? order[0] : 0;
                return new[] { v };
            }

            globalBest.Sort();
            return globalBest.ToArray();
        }

        private static IEnumerable<(int start, int end)> Partitioner(int total)
        {
            int threads = Math.Max(1, Environment.ProcessorCount);
            int block = Math.Max(64, total / (threads * 8 + 1));
            for (int i = 0; i < total; i += block)
            {
                int e = i + block;
                if (e > total) e = total;
                yield return (i, e);
            }
        }

        private sealed class WorkerState
        {
            public readonly int[] Mark;
            public int Stamp;
            public readonly List<int> Cand;
            public readonly List<int> Next;
            public readonly List<int> Clique;
            public readonly Random Rng;

            public WorkerState(int n)
            {
                Mark = new int[n];
                Stamp = 1;
                Cand = new List<int>(256);
                Next = new List<int>(256);
                Clique = new List<int>(256);
                Rng = new Random(unchecked(Environment.TickCount * 31 + Thread.CurrentThread.ManagedThreadId));
            }

            public int NewStamp()
            {
                Stamp++;
                if (Stamp == int.MaxValue)
                {
                    Array.Clear(Mark, 0, Mark.Length);
                    Stamp = 1;
                }
                return Stamp;
            }
        }

        private void BuildGreedyCliqueFromSeed(int seed, WorkerState st, ref List<int> globalBest, object gate)
        {
            int[][] adj = _g.Adj;
            int[] deg = _g.Degree;

            st.Clique.Clear();
            st.Cand.Clear();
            st.Next.Clear();

            st.Clique.Add(seed);

            foreach (int u in adj[seed]) st.Cand.Add(u);

            while (st.Cand.Count > 0 && !TimeExceeded())
            {
                int bestV = -1;
                int bestScore = -1;

                // Pick candidate with max degree; randomized tie-break to diversify.
                for (int i = 0; i < st.Cand.Count; i++)
                {
                    int v = st.Cand[i];
                    int score = deg[v];
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestV = v;
                    }
                    else if (score == bestScore && st.Rng.Next(2) == 0)
                    {
                        bestV = v;
                    }
                }

                if (bestV < 0) break;
                st.Clique.Add(bestV);

                int stamp = st.NewStamp();
                int[] mark = st.Mark;
                int[] nbest = adj[bestV];
                for (int i = 0; i < nbest.Length; i++) mark[nbest[i]] = stamp;

                st.Next.Clear();
                for (int i = 0; i < st.Cand.Count; i++)
                {
                    int v = st.Cand[i];
                    if (v != bestV && mark[v] == stamp)
                        st.Next.Add(v);
                }

                var temp = st.Cand;
                st.Cand.Clear();
                st.Cand.AddRange(st.Next);
                st.Next.Clear();
            }

            lock (gate)
            {
                if (st.Clique.Count > globalBest.Count)
                {
                    globalBest = new List<int>(st.Clique);
                }
            }
        }

        private int[] SolveExactWithBranchAndBound(int[] incumbent)
        {
            int n = _g.N;
            int words = (n + 63) >> 6;
            ulong[][] adjBits = new ulong[n][];
            for (int i = 0; i < n; i++)
            {
                adjBits[i] = new ulong[words];
                foreach (int v in _g.Adj[i])
                    adjBits[i][v >> 6] |= 1UL << (v & 63);
            }

            var solver = new ExactBnB(adjBits, _g.Degree, _deadlineTicks, incumbent);
            return solver.Run();
        }

        private sealed class ExactBnB
        {
            private readonly ulong[][] _adj;
            private readonly int[] _deg;
            private readonly long _deadlineTicks;
            private readonly int _n;
            private List<int> _best;
            private readonly List<int> _cur;

            public ExactBnB(ulong[][] adj, int[] deg, long deadlineTicks, int[] incumbent)
            {
                _adj = adj;
                _deg = deg;
                _deadlineTicks = deadlineTicks;
                _n = adj.Length;
                _best = new List<int>(incumbent);
                _cur = new List<int>(Math.Max(incumbent.Length + 16, 64));
            }

            private bool TimeExceeded() => DateTime.UtcNow.Ticks >= _deadlineTicks;

            public int[] Run()
            {
                var initial = Enumerable.Range(0, _n).ToList();
                initial.Sort((a, b) =>
                {
                    int c = _deg[b].CompareTo(_deg[a]);
                    return c != 0 ? c : a.CompareTo(b);
                });

                Expand(initial);
                _best.Sort();
                return _best.ToArray();
            }

            private void Expand(List<int> cand)
            {
                if (TimeExceeded() || cand.Count == 0)
                {
                    if (_cur.Count > _best.Count)
                        _best = new List<int>(_cur);
                    return;
                }

                ColorSort(cand, out List<int> order, out List<int> bounds);

                for (int i = order.Count - 1; i >= 0; i--)
                {
                    if (TimeExceeded()) return;
                    if (_cur.Count + bounds[i] <= _best.Count) return;

                    int v = order[i];
                    _cur.Add(v);

                    var next = new List<int>();
                    for (int j = 0; j < i; j++)
                    {
                        int u = order[j];
                        if (IsAdjacent(v, u)) next.Add(u);
                    }

                    if (next.Count == 0)
                    {
                        if (_cur.Count > _best.Count)
                            _best = new List<int>(_cur);
                    }
                    else if (_cur.Count + next.Count > _best.Count)
                    {
                        Expand(next);
                    }

                    _cur.RemoveAt(_cur.Count - 1);
                }
            }

            private bool IsAdjacent(int a, int b)
            {
                return ((_adj[a][b >> 6] >> (b & 63)) & 1UL) != 0UL;
            }

            private bool NonAdjacentToColorClass(int v, ulong[] colorMask)
            {
                ulong[] row = _adj[v];
                for (int w = 0; w < row.Length; w++)
                {
                    if ((row[w] & colorMask[w]) != 0UL) return false;
                }
                return true;
            }

            private void ColorSort(List<int> cand, out List<int> order, out List<int> bounds)
            {
                order = new List<int>(cand.Count);
                bounds = new List<int>(cand.Count);

                var rem = new List<int>(cand);
                rem.Sort((a, b) =>
                {
                    int c = _deg[b].CompareTo(_deg[a]);
                    return c != 0 ? c : a.CompareTo(b);
                });

                int words = _adj[0].Length;
                int color = 0;

                while (rem.Count > 0)
                {
                    color++;
                    var colorMask = new ulong[words];
                    var nextRem = new List<int>(rem.Count);

                    for (int i = 0; i < rem.Count; i++)
                    {
                        int v = rem[i];
                        if (NonAdjacentToColorClass(v, colorMask))
                        {
                            order.Add(v);
                            bounds.Add(color);
                            colorMask[v >> 6] |= 1UL << (v & 63);
                        }
                        else
                        {
                            nextRem.Add(v);
                        }
                    }

                    rem = nextRem;
                }
            }
        }
    }

    private static Graph ReadGraph()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        if (n == int.MinValue || m == int.MinValue || n < 0 || m < 0)
            return new Graph(0, 0, Array.Empty<int[]>(), Array.Empty<int>());

        int[] deg = new int[n];
        var edgesU = new int[m];
        var edgesV = new int[m];
        int realM = 0;

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u == int.MinValue || v == int.MinValue) break;
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            edgesU[realM] = u;
            edgesV[realM] = v;
            deg[u]++;
            deg[v]++;
            realM++;
        }

        int[][] adj = new int[n][];
        for (int i = 0; i < n; i++) adj[i] = new int[deg[i]];
        int[] ptr = new int[n];

        for (int i = 0; i < realM; i++)
        {
            int u = edgesU[i];
            int v = edgesV[i];
            adj[u][ptr[u]++] = v;
            adj[v][ptr[v]++] = u;
        }

        // Deduplicate adjacency (robust against duplicate edges in input).
        for (int i = 0; i < n; i++)
        {
            Array.Sort(adj[i]);
            int w = 0;
            int prev = -1;
            for (int j = 0; j < adj[i].Length; j++)
            {
                int x = adj[i][j];
                if (x != prev)
                {
                    adj[i][w++] = x;
                    prev = x;
                }
            }

            if (w != adj[i].Length)
            {
                var shrunk = new int[w];
                Array.Copy(adj[i], shrunk, w);
                adj[i] = shrunk;
                deg[i] = w;
            }
        }

        long edgesCount = 0;
        for (int i = 0; i < n; i++) edgesCount += deg[i];
        int undirectedEdges = (int)(edgesCount / 2);
        return new Graph(n, undirectedEdges, adj, deg);
    }

    public static void Main()
    {
        Graph g = ReadGraph();

        // Keep a strict runtime reserve for printing/output flush.
        TimeSpan budget = TimeSpan.FromMinutes(4.8);
        var solver = new MaxCliqueSolver(g, budget);
        int[] clique = solver.Solve();

        // Safety: ensure non-empty output when graph has vertices.
        if (g.N > 0 && clique.Length == 0) clique = new[] { 0 };

        Console.WriteLine(clique.Length);
        if (clique.Length > 0)
            Console.WriteLine(string.Join(" ", clique));
        else
            Console.WriteLine();
    }
}
