#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <tuple>
#include <utility>
#include <vector>

struct Vec3 {
    double x{}, y{}, z{};
    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double norm(const Vec3& v) {
    return std::sqrt(dot(v, v));
}

struct Plane {
    Vec3 n; // unit normal
    double d; // n.x * x + n.y * y + n.z * z + d <= 0 inside
};

static bool nearPlane(const Plane& a, const Plane& b, double angEps = 1e-6, double dEps = 1e-6) {
    return std::fabs(a.n.x - b.n.x) <= angEps &&
           std::fabs(a.n.y - b.n.y) <= angEps &&
           std::fabs(a.n.z - b.n.z) <= angEps &&
           std::fabs(a.d - b.d) <= dEps;
}

static std::vector<Plane> buildConvexHullPlanesBruteforce(const std::vector<Vec3>& p) {
    std::vector<Plane> planes;
    const int n = static_cast<int>(p.size());
    if (n < 4) return planes;

    const double colEps = 1e-12;
    const double sideEps = 1e-9;

    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 a = p[i], b = p[j], c = p[k];
                Vec3 nn = cross(b - a, c - a);
                double nnorm = norm(nn);
                if (nnorm <= colEps) continue;

                double d = -dot(nn, a);
                bool hasPos = false, hasNeg = false;
                for (int t = 0; t < n; ++t) {
                    double s = dot(nn, p[t]) + d;
                    if (s > sideEps) hasPos = true;
                    if (s < -sideEps) hasNeg = true;
                    if (hasPos && hasNeg) break;
                }
                if (hasPos && hasNeg) continue;

                // Normalize and orient so all points are <= 0.
                nn = nn * (1.0 / nnorm);
                d /= nnorm;

                bool anyPos = false;
                for (int t = 0; t < n; ++t) {
                    if (dot(nn, p[t]) + d > sideEps) {
                        anyPos = true;
                        break;
                    }
                }
                if (anyPos) {
                    nn = nn * -1.0;
                    d = -d;
                }

                Plane pl{nn, d};
                bool dup = false;
                for (const auto& q : planes) {
                    if (nearPlane(pl, q)) {
                        dup = true;
                        break;
                    }
                }
                if (!dup) planes.push_back(pl);
            }
        }
    }

    return planes;
}

static bool insideHalfspacesWithMargin(const std::vector<Plane>& planes, const Vec3& c, double margin) {
    for (const auto& pl : planes) {
        if (dot(pl.n, c) + pl.d > -margin) return false;
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V;
    double edge;
    if (!(std::cin >> V >> edge)) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::vector<Vec3> pts;
    pts.reserve(std::max(V, 0));
    for (int i = 0; i < V; ++i) {
        double x, y, z;
        std::cin >> x >> y >> z;
        pts.emplace_back(x, y, z);
    }

    if (pts.empty() || edge <= 0.0) {
        std::cout << 0 << '\n';
        return 0;
    }

    constexpr double SQRT2 = 1.4142135623730950488;
    constexpr double SQRT3_8 = 0.6123724356957945245; // sqrt(3/8)
    const double R = edge * SQRT3_8; // tetra circumradius

    Vec3 mn = pts[0], mx = pts[0];
    for (const auto& p : pts) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }

    // Adaptive geometry model:
    // - Small/medium V: infer convex hull halfspaces by brute force planes.
    // - Large V: fallback to AABB-only, to guarantee speed.
    std::vector<Plane> hullPlanes;
    bool useHull = (V >= 4 && V <= 80);
    if (useHull) {
        hullPlanes = buildConvexHullPlanesBruteforce(pts);
        if (hullPlanes.empty()) useHull = false;
    }

    const double minX = mn.x + R, minY = mn.y + R, minZ = mn.z + R;
    const double maxX = mx.x - R, maxY = mx.y - R, maxZ = mx.z - R;
    if (minX > maxX || minY > maxY || minZ > maxZ) {
        std::cout << 0 << '\n';
        return 0;
    }

    auto validCenter = [&](const Vec3& c) {
        if (c.x < minX || c.x > maxX || c.y < minY || c.y > maxY || c.z < minZ || c.z > maxZ) return false;
        if (useHull) return insideHalfspacesWithMargin(hullPlanes, c, R);
        return true;
    };

    const std::array<std::array<double, 3>, 4> fccOffsets = {{
        {{0.0, 0.0, 0.0}},
        {{0.0, 0.5, 0.5}},
        {{0.5, 0.0, 0.5}},
        {{0.5, 0.5, 0.0}}
    }};

    const double stepFCC = 2.0 * R * SQRT2;
    const double stepCubic = 2.0 * R;

    const std::size_t MAX_OUT = 600000; // keep runtime and output bounded on extreme cases

    std::vector<std::array<double, 6>> best;
    best.reserve(200000);

    auto tryFCC = [&]() {
        std::vector<std::array<double, 6>> out;
        out.reserve(best.capacity());

        const int nx = std::max(1, static_cast<int>(std::floor((maxX - minX) / stepFCC)) + 1);
        const int ny = std::max(1, static_cast<int>(std::floor((maxY - minY) / stepFCC)) + 1);
        const int nz = std::max(1, static_cast<int>(std::floor((maxZ - minZ) / stepFCC)) + 1);

        for (int iz = 0; iz < nz; ++iz) {
            for (int iy = 0; iy < ny; ++iy) {
                for (int ix = 0; ix < nx; ++ix) {
                    const double bx = minX + ix * stepFCC;
                    const double by = minY + iy * stepFCC;
                    const double bz = minZ + iz * stepFCC;
                    for (const auto& off : fccOffsets) {
                        Vec3 c{bx + off[0] * stepFCC, by + off[1] * stepFCC, bz + off[2] * stepFCC};
                        if (!validCenter(c)) continue;
                        out.push_back({c.x, c.y, c.z, 0.0, 0.0, 0.0});
                        if (out.size() >= MAX_OUT) return out;
                    }
                }
            }
        }
        return out;
    };

    auto tryCubic = [&]() {
        std::vector<std::array<double, 6>> out;
        out.reserve(best.capacity());

        const int nx = std::max(1, static_cast<int>(std::floor((maxX - minX) / stepCubic)) + 1);
        const int ny = std::max(1, static_cast<int>(std::floor((maxY - minY) / stepCubic)) + 1);
        const int nz = std::max(1, static_cast<int>(std::floor((maxZ - minZ) / stepCubic)) + 1);

        for (int iz = 0; iz < nz; ++iz) {
            for (int iy = 0; iy < ny; ++iy) {
                for (int ix = 0; ix < nx; ++ix) {
                    Vec3 c{minX + ix * stepCubic, minY + iy * stepCubic, minZ + iz * stepCubic};
                    if (!validCenter(c)) continue;
                    out.push_back({c.x, c.y, c.z, 0.0, 0.0, 0.0});
                    if (out.size() >= MAX_OUT) return out;
                }
            }
        }
        return out;
    };

    auto a = tryFCC();
    auto b = tryCubic();
    best = (a.size() >= b.size()) ? std::move(a) : std::move(b);

    std::cout << best.size() << '\n';
    std::cout << std::setprecision(10);
    for (const auto& t : best) {
        std::cout << t[0] << ' ' << t[1] << ' ' << t[2] << ' ' << t[3] << ' ' << t[4] << ' ' << t[5] << '\n';
    }

    return 0;
}
