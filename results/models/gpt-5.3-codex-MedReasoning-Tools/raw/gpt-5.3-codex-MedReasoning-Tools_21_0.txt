#include <array>
#include <cmath>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>

#include <queue>
#include <random>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        uint64_t h = 1469598103934665603ULL;
        for (int x : v) {
            uint64_t y = static_cast<uint64_t>(static_cast<int64_t>(x)) ^ 0x9e3779b97f4a7c15ULL;
            h ^= y + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        }
        return static_cast<size_t>(h);
    }
};

static inline int manhattan(const vector<int>& a, const vector<int>& b) {
    int s = 0;
    for (size_t i = 0; i < a.size(); ++i) s += abs(a[i] - b[i]);
    return s;
}

struct Move {
    int axis;
    int dir;
};

struct Node {
    int f;
    int g;
    vector<int> p;
};

struct NodeCmp {
    bool operator()(const Node& a, const Node& b) const {
        if (a.f != b.f) return a.f > b.f;
        return a.g < b.g;
    }
};

class SnakeSolver {
public:
    void run() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);

        if (!(cin >> D >> maxTurns)) return;
        bounds.resize(D);
        for (int i = 0; i < D; ++i) cin >> bounds[i];

        int snakeLen;
        cin >> snakeLen;
        vector<int> head(D);
        for (int i = 0; i < D; ++i) cin >> head[i];

        int F;
        cin >> F;
        vector<vector<int>> foods;
        foods.reserve(F);
        for (int i = 0; i < F; ++i) {
            vector<int> f(D);
            for (int j = 0; j < D; ++j) cin >> f[j];
            foods.push_back(move(f));
        }

        int O;
        cin >> O;
        for (int i = 0; i < O; ++i) {
            vector<int> o(D);
            for (int j = 0; j < D; ++j) cin >> o[j];
            obstacles.insert(move(o));
        }

        initSnake(head, snakeLen);

        for (const auto& f : foods) foodSet.insert(f);

        for (int turn = 0; turn < maxTurns; ++turn) {
            if (foodSet.empty()) break;

            Move m = chooseMove();
            cout << m.axis << ' ' << m.dir << '\n';

            if (!applyMove(m)) {
                // If the chosen move is invalid under our internal simulation,
                // stop emitting further turns.
                break;
            }
        }
    }

private:
    int D = 0;
    int maxTurns = 0;
    vector<int> bounds;

    deque<vector<int>> snake;
    unordered_set<vector<int>, VecHash> snakeSet;
    unordered_set<vector<int>, VecHash> obstacles;
    unordered_set<vector<int>, VecHash> foodSet;

    mt19937 rng{123456789};

    bool inBounds(const vector<int>& p) const {
        for (int i = 0; i < D; ++i) {
            if (p[i] < 0 || p[i] >= bounds[i]) return false;
        }
        return true;
    }

    vector<int> step(const vector<int>& p, int axis, int dir) const {
        vector<int> q = p;
        q[axis] += dir;
        return q;
    }

    void initSnake(const vector<int>& head, int len) {
        snake.clear();
        snakeSet.clear();

        snake.push_front(head);
        snakeSet.insert(head);
        if (len <= 1) return;

        int axis = -1;
        int dir = -1;

        for (int a = 0; a < D; ++a) {
            if (head[a] - (len - 1) >= 0) {
                axis = a;
                dir = -1;
                break;
            }
        }
        if (axis == -1) {
            for (int a = 0; a < D; ++a) {
                if (head[a] + (len - 1) < bounds[a]) {
                    axis = a;
                    dir = +1;
                    break;
                }
            }
        }

        if (axis == -1) {
            // Fallback: keep length 1 if no valid deterministic reconstruction is possible.
            return;
        }

        vector<int> cur = head;
        for (int i = 1; i < len; ++i) {
            cur[axis] += dir;
            if (!inBounds(cur) || obstacles.count(cur) || snakeSet.count(cur)) break;
            snake.push_back(cur);
            snakeSet.insert(cur);
        }
    }

    optional<vector<int>> nearestFood(const vector<int>& from) const {
        if (foodSet.empty()) return nullopt;
        int bestDist = numeric_limits<int>::max();
        const vector<int>* best = nullptr;
        for (const auto& f : foodSet) {
            int d = manhattan(from, f);
            if (d < bestDist) {
                bestDist = d;
                best = &f;
            }
        }
        if (!best) return nullopt;
        return *best;
    }

    vector<Move> legalMoves(bool assumeGrow) const {
        vector<Move> out;
        const vector<int>& h = snake.front();
        const vector<int>& tail = snake.back();

        for (int a = 0; a < D; ++a) {
            for (int d : {-1, 1}) {
                vector<int> nh = step(h, a, d);
                if (!inBounds(nh)) continue;
                if (obstacles.count(nh)) continue;

                bool occupied = snakeSet.count(nh) > 0;
                if (!occupied) {
                    out.push_back({a, d});
                    continue;
                }

                // Moving into current tail is valid only when tail moves away this turn.
                if (!assumeGrow && nh == tail) {
                    out.push_back({a, d});
                }
            }
        }
        return out;
    }

    bool applyMove(const Move& m) {
        vector<int> nh = step(snake.front(), m.axis, m.dir);
        if (!inBounds(nh)) return false;
        if (obstacles.count(nh)) return false;

        bool grow = (foodSet.count(nh) > 0);
        vector<int> oldTail = snake.back();
        if (!grow) {
            snakeSet.erase(oldTail);
        }
        if (snakeSet.count(nh)) {
            if (!grow) snakeSet.insert(oldTail);
            return false;
        }

        snake.push_front(nh);
        snakeSet.insert(nh);

        if (grow) {
            foodSet.erase(nh);
        } else {
            snake.pop_back();
        }
        return true;
    }

    bool reachableAStar(
        const vector<int>& start,
        const vector<int>& goal,
        const unordered_set<vector<int>, VecHash>& blocked,
        int nodeCap
    ) const {
        if (start == goal) return true;

        priority_queue<Node, vector<Node>, NodeCmp> pq;
        unordered_map<vector<int>, int, VecHash> bestG;
        pq.push({manhattan(start, goal), 0, start});
        bestG[start] = 0;

        int expanded = 0;

        while (!pq.empty()) {
            Node cur = pq.top();
            pq.pop();

            auto itg = bestG.find(cur.p);
            if (itg == bestG.end() || itg->second != cur.g) continue;

            if (cur.p == goal) return true;
            if (++expanded > nodeCap) break;

            for (int a = 0; a < D; ++a) {
                for (int d : {-1, 1}) {
                    vector<int> np = cur.p;
                    np[a] += d;
                    if (!inBounds(np)) continue;
                    if (blocked.count(np)) continue;

                    int ng = cur.g + 1;
                    auto it = bestG.find(np);
                    if (it == bestG.end() || ng < it->second) {
                        bestG[np] = ng;
                        int nf = ng + manhattan(np, goal);
                        pq.push({nf, ng, move(np)});
                    }
                }
            }
        }

        // If we hit the cap, return optimistic true in huge open spaces only if direct line is not blocked.
        return false;
    }

    bool moveLooksSafe(const Move& m) const {
        vector<int> nh = step(snake.front(), m.axis, m.dir);
        bool grow = (foodSet.count(nh) > 0);

        unordered_set<vector<int>, VecHash> blocked = snakeSet;
        if (!grow) blocked.erase(snake.back());
        blocked.erase(nh); // start cell should not block search
        for (const auto& o : obstacles) blocked.insert(o);

        // Prefer maintaining path to tail after the move to avoid dead-end traps.
        vector<int> futureTail = grow ? snake.back() : snake[snake.size() - 2];
        int cap = (D <= 4 ? 50000 : (D <= 8 ? 20000 : 8000));
        return reachableAStar(nh, futureTail, blocked, cap);
    }

    optional<Move> bestFoodDirectedMove() const {
        auto targetOpt = nearestFood(snake.front());
        if (!targetOpt.has_value()) return nullopt;
        const vector<int>& target = *targetOpt;

        struct Cand {
            Move m;
            int score;
            bool toward;
        };
        vector<Cand> cands;

        for (int a = 0; a < D; ++a) {
            for (int d : {-1, 1}) {
                vector<int> nh = step(snake.front(), a, d);
                bool grow = foodSet.count(nh) > 0;

                // Fast legality check.
                if (!inBounds(nh) || obstacles.count(nh)) continue;
                bool occ = snakeSet.count(nh) > 0;
                if (occ) {
                    if (grow) continue;
                    if (nh != snake.back()) continue;
                }

                int oldDist = manhattan(snake.front(), target);
                int newDist = manhattan(nh, target);
                bool toward = (newDist < oldDist);

                // Local freedom estimate.
                int freeNbr = 0;
                for (int a2 = 0; a2 < D; ++a2) {
                    for (int d2 : {-1, 1}) {
                        vector<int> nn = step(nh, a2, d2);
                        if (!inBounds(nn) || obstacles.count(nn)) continue;
                        bool occ2 = snakeSet.count(nn) > 0;
                        if (!occ2 || nn == snake.back()) ++freeNbr;
                    }
                }

                int score = (oldDist - newDist) * 100 + freeNbr * 3;
                cands.push_back({{a, d}, score, toward});
            }
        }

        if (cands.empty()) return nullopt;

        sort(cands.begin(), cands.end(), [](const Cand& x, const Cand& y) {
            if (x.toward != y.toward) return x.toward > y.toward;
            return x.score > y.score;
        });

        // Check safety for top candidates first.
        int checkCount = min<int>(static_cast<int>(cands.size()), 12);
        for (int i = 0; i < checkCount; ++i) {
            if (moveLooksSafe(cands[i].m)) return cands[i].m;
        }

        return cands[0].m;
    }

    Move chooseMove() const {
        // 1) Prefer food-directed safe move.
        if (auto m = bestFoodDirectedMove(); m.has_value()) {
            return *m;
        }

        // 2) Any legal move that appears safe.
        auto moves = legalMoves(false);
        for (const auto& m : moves) {
            if (moveLooksSafe(m)) return m;
        }

        // 3) Fallback: any legal move.
        if (!moves.empty()) return moves[0];

        // 4) No legal moves.
        return {0, 1};
    }
};

int main() {
    SnakeSolver solver;
    solver.run();
    return 0;
}
