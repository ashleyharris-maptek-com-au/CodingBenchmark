using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X
        );
        public Vec3 Unit()
        {
            double n = Norm();
            return n > 1e-12 ? this * (1.0 / n) : new Vec3(0, 0, 1);
        }
    }

    struct OrbitFeature
    {
        public double R;
        public double V;
        public Vec3 Rhat;
        public Vec3 Nhat;
        public double Ecc;
    }

    static OrbitFeature BuildFeature(Vec3 r, Vec3 v, double mu)
    {
        double rMag = r.Norm();
        double vMag = v.Norm();

        Vec3 h = Vec3.Cross(r, v);
        Vec3 nHat = h.Unit();
        Vec3 rHat = rMag > 1e-12 ? r * (1.0 / rMag) : new Vec3(1, 0, 0);

        Vec3 eVec = (Vec3.Cross(v, h) * (1.0 / mu)) - (r * (1.0 / Math.Max(1e-12, rMag)));
        double ecc = eVec.Norm();

        return new OrbitFeature
        {
            R = rMag,
            V = vMag,
            Rhat = rHat,
            Nhat = nHat,
            Ecc = ecc
        };
    }

    static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);

    static double TransferCost(in OrbitFeature a, in OrbitFeature b, double mu)
    {
        double r1 = Math.Max(1.0, a.R);
        double r2 = Math.Max(1.0, b.R);

        double term1 = Math.Sqrt(mu / r1);
        double term2 = Math.Sqrt(mu / r2);
        double dvH1 = Math.Abs(term1 * (Math.Sqrt(2.0 * r2 / (r1 + r2)) - 1.0));
        double dvH2 = Math.Abs(term2 * (1.0 - Math.Sqrt(2.0 * r1 / (r1 + r2))));
        double dvH = dvH1 + dvH2;

        double cosPlane = Clamp(Vec3.Dot(a.Nhat, b.Nhat), -1.0, 1.0);
        double planeAngle = Math.Acos(cosPlane);
        double vRef = Math.Sqrt(mu / ((r1 + r2) * 0.5));
        double dvPlane = 2.0 * vRef * Math.Sin(0.5 * planeAngle);

        double dvEcc = Math.Abs(a.Ecc - b.Ecc) * vRef;

        double cosPhase = Clamp(Vec3.Dot(a.Rhat, b.Rhat), -1.0, 1.0);
        double phase = Math.Acos(cosPhase);
        double dvPhase = 0.20 * vRef * (phase / Math.PI);

        return dvH + 0.9 * dvPlane + 0.35 * dvEcc + dvPhase;
    }

    static double RouteCost(int[] route, double[] startTo, double[,] c)
    {
        if (route.Length == 0) return 0;
        double total = startTo[route[0]];
        for (int i = 1; i < route.Length; i++) total += c[route[i - 1], route[i]];
        return total;
    }

    static int[] SolveExactDP(int n, double[] startTo, double[,] c)
    {
        int full = 1 << n;
        double[,] dp = new double[full, n];
        int[,] prev = new int[full, n];
        double INF = 1e100;

        for (int m = 0; m < full; m++)
            for (int j = 0; j < n; j++)
            {
                dp[m, j] = INF;
                prev[m, j] = -1;
            }

        for (int j = 0; j < n; j++)
        {
            int mask = 1 << j;
            dp[mask, j] = startTo[j];
        }

        for (int mask = 1; mask < full; mask++)
        {
            for (int j = 0; j < n; j++)
            {
                if ((mask & (1 << j)) == 0) continue;
                double cur = dp[mask, j];
                if (cur >= INF * 0.5) continue;

                int rem = (~mask) & (full - 1);
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int k = BitOperations.TrailingZeroCount((uint)bit);
                    int nmask = mask | bit;
                    double cand = cur + c[j, k];
                    if (cand < dp[nmask, k])
                    {
                        dp[nmask, k] = cand;
                        prev[nmask, k] = j;
                    }
                    rem ^= bit;
                }
            }
        }

        int all = full - 1;
        int last = 0;
        double best = INF;
        for (int j = 0; j < n; j++)
        {
            if (dp[all, j] < best)
            {
                best = dp[all, j];
                last = j;
            }
        }

        int[] route = new int[n];
        int maskBack = all;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = last;
            int p = prev[maskBack, last];
            maskBack ^= (1 << last);
            last = p;
        }

        return route;
    }

    static int[] BuildGreedyRoute(int n, double[] startTo, double[,] c, Random rng, bool randomized)
    {
        bool[] used = new bool[n];
        int[] route = new int[n];

        int first;
        if (!randomized)
        {
            first = 0;
            for (int i = 1; i < n; i++) if (startTo[i] < startTo[first]) first = i;
        }
        else
        {
            int k = Math.Min(5, n);
            var cand = Enumerable.Range(0, n).OrderBy(i => startTo[i]).Take(k).ToArray();
            first = cand[rng.Next(cand.Length)];
        }

        route[0] = first;
        used[first] = true;
        int cur = first;

        for (int pos = 1; pos < n; pos++)
        {
            int next = -1;
            if (!randomized)
            {
                double best = double.MaxValue;
                for (int j = 0; j < n; j++)
                {
                    if (used[j]) continue;
                    double v = c[cur, j];
                    if (v < best) { best = v; next = j; }
                }
            }
            else
            {
                int candCount = Math.Min(6, n - pos);
                var list = new List<(double cost, int idx)>();
                for (int j = 0; j < n; j++)
                    if (!used[j]) list.Add((c[cur, j], j));
                list.Sort((a, b) => a.cost.CompareTo(b.cost));
                int pick = rng.Next(candCount);
                next = list[pick].idx;
            }

            route[pos] = next;
            used[next] = true;
            cur = next;
        }

        return route;
    }

    static void TwoOptPath(int[] route, double[] startTo, double[,] c, int maxPasses)
    {
        int n = route.Length;
        if (n < 4) return;

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n - 1; i++)
            {
                for (int k = i + 1; k < n; k++)
                {
                    if (k == i + 1) continue;

                    int a = (i == 0) ? -1 : route[i - 1];
                    int b = route[i];
                    int c1 = route[k];
                    int d = (k + 1 < n) ? route[k + 1] : -1;

                    double oldEdge1 = (a == -1) ? startTo[b] : c[a, b];
                    double newEdge1 = (a == -1) ? startTo[c1] : c[a, c1];

                    double oldEdge2 = (d == -1) ? 0.0 : c[c1, d];
                    double newEdge2 = (d == -1) ? 0.0 : c[b, d];

                    double delta = (newEdge1 + newEdge2) - (oldEdge1 + oldEdge2);
                    if (delta < -1e-9)
                    {
                        Array.Reverse(route, i, k - i + 1);
                        improved = true;
                    }
                }
            }

            if (!improved) break;
        }
    }

    static int[] SolveHeuristic(int n, double[] startTo, double[,] c)
    {
        var rng = new Random(1234567);
        int tries = n <= 20 ? 24 : (n <= 50 ? 18 : 12);
        int maxPasses = n <= 30 ? 8 : 5;

        int[] bestRoute = null;
        double bestCost = double.MaxValue;

        for (int t = 0; t < tries; t++)
        {
            bool randomized = t > 0;
            int[] route = BuildGreedyRoute(n, startTo, c, rng, randomized);
            TwoOptPath(route, startTo, c, maxPasses);
            double cost = RouteCost(route, startTo, c);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestRoute = route;
            }
        }

        return bestRoute ?? Enumerable.Range(0, n).ToArray();
    }

    static class BitOperations
    {
        public static int TrailingZeroCount(uint value)
        {
            int c = 0;
            while ((value & 1u) == 0u) { value >>= 1; c++; }
            return c;
        }
    }

    public static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        string all = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(all)) return;

        var toks = all.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;

        int n = int.Parse(toks[p++], ci);
        double mu = double.Parse(toks[p++], ci);

        Vec3 startR = new Vec3(
            double.Parse(toks[p++], ci),
            double.Parse(toks[p++], ci),
            double.Parse(toks[p++], ci));
        Vec3 startV = new Vec3(
            double.Parse(toks[p++], ci),
            double.Parse(toks[p++], ci),
            double.Parse(toks[p++], ci));

        OrbitFeature startF = BuildFeature(startR, startV, mu);
        OrbitFeature[] st = new OrbitFeature[n];

        for (int i = 0; i < n; i++)
        {
            Vec3 r = new Vec3(
                double.Parse(toks[p++], ci),
                double.Parse(toks[p++], ci),
                double.Parse(toks[p++], ci));
            Vec3 v = new Vec3(
                double.Parse(toks[p++], ci),
                double.Parse(toks[p++], ci),
                double.Parse(toks[p++], ci));
            st[i] = BuildFeature(r, v, mu);
        }

        double[] startTo = new double[n];
        for (int i = 0; i < n; i++) startTo[i] = TransferCost(startF, st[i], mu);

        double[,] c = new double[n, n];
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                c[i, j] = (i == j) ? 0.0 : TransferCost(st[i], st[j], mu);

        int[] ans = (n <= 12) ? SolveExactDP(n, startTo, c) : SolveHeuristic(n, startTo, c);
        Console.WriteLine(string.Join(" ", ans));
    }
}
