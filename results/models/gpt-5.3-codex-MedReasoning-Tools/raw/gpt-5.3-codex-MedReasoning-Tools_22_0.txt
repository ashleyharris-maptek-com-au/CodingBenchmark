using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    private readonly struct Cell
    {
        public readonly int X;
        public readonly int Y;
        public Cell(int x, int y) { X = x; Y = y; }
    }

    private sealed class Shape
    {
        public readonly Cell[] Cells;
        public readonly int Width;
        public readonly int Height;

        public Shape(Cell[] cells)
        {
            Cells = cells;
            int maxX = 0, maxY = 0;
            for (int i = 0; i < cells.Length; i++)
            {
                if (cells[i].X > maxX) maxX = cells[i].X;
                if (cells[i].Y > maxY) maxY = cells[i].Y;
            }
            Width = maxX + 1;
            Height = maxY + 1;
        }
    }

    private sealed class Board
    {
        public readonly int Width;
        public readonly int Height;
        public readonly bool[,] Grid;
        public readonly int[] ColHeights;

        public Board(int width, int height)
        {
            Width = width;
            Height = height;
            Grid = new bool[height, width];
            ColHeights = new int[width];
        }

        private Board(int width, int height, bool[,] grid, int[] colHeights)
        {
            Width = width;
            Height = height;
            Grid = grid;
            ColHeights = colHeights;
        }

        public Board Clone()
        {
            var g = (bool[,])Grid.Clone();
            var h = new int[Width];
            Array.Copy(ColHeights, h, Width);
            return new Board(Width, Height, g, h);
        }

        public bool CanPlaceAt(Shape s, int x, int y)
        {
            for (int i = 0; i < s.Cells.Length; i++)
            {
                int px = x + s.Cells[i].X;
                int py = y + s.Cells[i].Y;
                if ((uint)px >= (uint)Width || py < 0 || py >= Height) return false;
                if (Grid[py, px]) return false;
            }
            return true;
        }

        public bool TryFindDropY(Shape s, int x, out int y)
        {
            y = Height - s.Height;
            if (x < 0 || x + s.Width > Width) return false;

            while (y >= 0 && !CanPlaceAt(s, x, y)) y--;
            if (y < 0) return false;
            while (y > 0 && CanPlaceAt(s, x, y - 1)) y--;
            return true;
        }

        public int PlaceAndClear(Shape s, int x, int y)
        {
            for (int i = 0; i < s.Cells.Length; i++)
            {
                int px = x + s.Cells[i].X;
                int py = y + s.Cells[i].Y;
                Grid[py, px] = true;
            }

            int cleared = 0;
            int writeRow = 0;
            for (int readRow = 0; readRow < Height; readRow++)
            {
                bool full = true;
                for (int c = 0; c < Width; c++)
                {
                    if (!Grid[readRow, c]) { full = false; break; }
                }

                if (full)
                {
                    cleared++;
                    continue;
                }

                if (writeRow != readRow)
                {
                    for (int c = 0; c < Width; c++) Grid[writeRow, c] = Grid[readRow, c];
                }
                writeRow++;
            }

            for (int r = writeRow; r < Height; r++)
            {
                for (int c = 0; c < Width; c++) Grid[r, c] = false;
            }

            RebuildHeights();
            return cleared;
        }

        private void RebuildHeights()
        {
            for (int c = 0; c < Width; c++)
            {
                int h = 0;
                for (int r = Height - 1; r >= 0; r--)
                {
                    if (Grid[r, c]) { h = r + 1; break; }
                }
                ColHeights[c] = h;
            }
        }

        public Metrics EvaluateMetrics()
        {
            int aggregateHeight = 0;
            int maxHeight = 0;
            int holes = 0;
            int rowTransitions = 0;
            int colTransitions = 0;
            int bumpiness = 0;
            int wells = 0;

            for (int c = 0; c < Width; c++)
            {
                int h = ColHeights[c];
                aggregateHeight += h;
                if (h > maxHeight) maxHeight = h;
            }

            for (int c = 0; c < Width; c++)
            {
                bool blockSeen = false;
                for (int r = Height - 1; r >= 0; r--)
                {
                    bool filled = Grid[r, c];
                    if (filled) blockSeen = true;
                    else if (blockSeen) holes++;
                }
            }

            for (int r = 0; r < Height; r++)
            {
                bool prev = true;
                for (int c = 0; c < Width; c++)
                {
                    bool cur = Grid[r, c];
                    if (cur != prev) rowTransitions++;
                    prev = cur;
                }
                if (!prev) rowTransitions++;
            }

            for (int c = 0; c < Width; c++)
            {
                bool prev = true;
                for (int r = Height - 1; r >= 0; r--)
                {
                    bool cur = Grid[r, c];
                    if (cur != prev) colTransitions++;
                    prev = cur;
                }
                if (!prev) colTransitions++;
            }

            for (int c = 0; c < Width - 1; c++)
            {
                bumpiness += Math.Abs(ColHeights[c] - ColHeights[c + 1]);
            }

            for (int c = 0; c < Width; c++)
            {
                int left = c == 0 ? Height : ColHeights[c - 1];
                int right = c == Width - 1 ? Height : ColHeights[c + 1];
                int minLR = left < right ? left : right;
                if (minLR > ColHeights[c])
                {
                    wells += minLR - ColHeights[c];
                }
            }

            return new Metrics(aggregateHeight, maxHeight, holes, rowTransitions, colTransitions, bumpiness, wells);
        }
    }

    private readonly struct Metrics
    {
        public readonly int AggregateHeight;
        public readonly int MaxHeight;
        public readonly int Holes;
        public readonly int RowTransitions;
        public readonly int ColTransitions;
        public readonly int Bumpiness;
        public readonly int Wells;

        public Metrics(int aggregateHeight, int maxHeight, int holes, int rowTransitions, int colTransitions, int bumpiness, int wells)
        {
            AggregateHeight = aggregateHeight;
            MaxHeight = maxHeight;
            Holes = holes;
            RowTransitions = rowTransitions;
            ColTransitions = colTransitions;
            Bumpiness = bumpiness;
            Wells = wells;
        }
    }

    private readonly struct Move
    {
        public readonly int Rotation;
        public readonly int Column;
        public Move(int rotation, int column)
        {
            Rotation = rotation;
            Column = column;
        }
    }

    private static readonly Dictionary<char, Shape[]> PieceRotations = BuildPieceRotations();

    public static void Main()
    {
        var scanner = new FastScanner(Console.In);
        int width = scanner.NextInt();
        int height = scanner.NextInt();
        int numPieces = scanner.NextInt();

        var board = new Board(width, height);
        int placed = 0;

        for (int i = 0; i < numPieces; i++)
        {
            string token = scanner.NextToken();
            if (string.IsNullOrEmpty(token)) break;
            char pieceType = token[0];

            Move move = ChooseMove(board, pieceType, placed, numPieces);
            Console.WriteLine(string.Create(CultureInfo.InvariantCulture, $"{move.Rotation} {move.Column}"));
            Console.Out.Flush();

            if (!ApplyOurMove(board, pieceType, move))
            {
                Move fallback = FindAnyLegalMove(board, pieceType);
                if (fallback.Column >= 0) ApplyOurMove(board, pieceType, fallback);
            }

            string result = scanner.NextToken();
            if (string.IsNullOrEmpty(result)) break;
            if (result == "gameover") break;
            placed++;
        }
    }

    private static Move ChooseMove(Board board, char piece, int placed, int total)
    {
        Shape[] rots;
        if (!PieceRotations.TryGetValue(piece, out rots)) return new Move(0, 0);

        bool marathonPhase = total > 0 && placed > total / 2;
        bool tallStack = false;
        int maxH = 0;
        for (int c = 0; c < board.Width; c++) if (board.ColHeights[c] > maxH) maxH = board.ColHeights[c];
        if (maxH >= (board.Height * 3) / 5) tallStack = true;

        int complexity = board.Width * rots.Length;
        bool deepSearch = (complexity <= 80 && !tallStack) || (marathonPhase && complexity <= 60);

        double bestScore = double.NegativeInfinity;
        Move bestMove = new Move(0, -1);

        for (int r = 0; r < rots.Length; r++)
        {
            var shape = rots[r];
            int maxCol = board.Width - shape.Width;
            for (int col = 0; col <= maxCol; col++)
            {
                if (!board.TryFindDropY(shape, col, out int y)) continue;

                var sim = board.Clone();
                int cleared = sim.PlaceAndClear(shape, col, y);
                Metrics m = sim.EvaluateMetrics();

                double score = ScoreMetrics(m, cleared, board, marathonPhase, tallStack);

                if (deepSearch)
                {
                    score += LookaheadOne(sim, board.Width <= 12 ? 3 : 2);
                }

                if (score > bestScore)
                {
                    bestScore = score;
                    bestMove = new Move(r, col);
                }
            }
        }

        if (bestMove.Column < 0)
        {
            return FindAnyLegalMove(board, piece);
        }

        return bestMove;
    }

    private static double LookaheadOne(Board board, int sampleCount)
    {
        char[] bag = new[] { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };
        double total = 0;
        int considered = 0;

        for (int i = 0; i < bag.Length && considered < sampleCount; i++)
        {
            char p = bag[(i * 3) % bag.Length];
            Move m = FindBestGreedy(board, p);
            if (m.Column < 0) { total -= 10000; considered++; continue; }

            var sim = board.Clone();
            if (!ApplyOurMove(sim, p, m)) { total -= 10000; considered++; continue; }
            Metrics mm = sim.EvaluateMetrics();
            total += ScoreMetrics(mm, 0, sim, false, false);
            considered++;
        }

        return considered == 0 ? 0 : total / considered * 0.15;
    }

    private static Move FindBestGreedy(Board board, char piece)
    {
        if (!PieceRotations.TryGetValue(piece, out var rots)) return new Move(0, -1);

        double best = double.NegativeInfinity;
        Move bestMove = new Move(0, -1);
        for (int r = 0; r < rots.Length; r++)
        {
            Shape s = rots[r];
            int maxCol = board.Width - s.Width;
            for (int col = 0; col <= maxCol; col++)
            {
                if (!board.TryFindDropY(s, col, out int y)) continue;
                var sim = board.Clone();
                int cleared = sim.PlaceAndClear(s, col, y);
                Metrics m = sim.EvaluateMetrics();
                double score = ScoreMetrics(m, cleared, board, false, false);
                if (score > best)
                {
                    best = score;
                    bestMove = new Move(r, col);
                }
            }
        }
        return bestMove;
    }

    private static double ScoreMetrics(Metrics m, int cleared, Board board, bool marathon, bool emergency)
    {
        double lineWeight = marathon ? 5.5 : 4.2;
        double holeWeight = emergency ? 10.5 : 8.2;
        double heightWeight = emergency ? 0.85 : 0.55;
        double bumpWeight = marathon ? 0.23 : 0.30;

        double danger = 0;
        if (m.MaxHeight > board.Height * 0.75) danger += (m.MaxHeight - board.Height * 0.75) * 3.0;

        double score = 0;
        score += cleared * lineWeight;
        score -= m.Holes * holeWeight;
        score -= m.AggregateHeight * heightWeight;
        score -= m.Bumpiness * bumpWeight;
        score -= m.RowTransitions * 0.35;
        score -= m.ColTransitions * 0.22;
        score += m.Wells * 0.08;
        score -= danger;

        if (cleared >= 2) score += cleared * 1.3;
        return score;
    }

    private static bool ApplyOurMove(Board board, char piece, Move move)
    {
        if (!PieceRotations.TryGetValue(piece, out var rots)) return false;
        if ((uint)move.Rotation >= (uint)rots.Length) return false;
        Shape s = rots[move.Rotation];
        if (!board.TryFindDropY(s, move.Column, out int y)) return false;
        board.PlaceAndClear(s, move.Column, y);
        return true;
    }

    private static Move FindAnyLegalMove(Board board, char piece)
    {
        if (!PieceRotations.TryGetValue(piece, out var rots)) return new Move(0, 0);
        for (int r = 0; r < rots.Length; r++)
        {
            Shape s = rots[r];
            int maxCol = board.Width - s.Width;
            for (int col = 0; col <= maxCol; col++)
            {
                if (board.TryFindDropY(s, col, out _)) return new Move(r, col);
            }
        }
        return new Move(0, 0);
    }

    private static Dictionary<char, Shape[]> BuildPieceRotations()
    {
        return new Dictionary<char, Shape[]>
        {
            ['I'] = UniqueRotations(new[]
            {
                new Shape(new[] { new Cell(0,0), new Cell(1,0), new Cell(2,0), new Cell(3,0) }),
                new Shape(new[] { new Cell(0,0), new Cell(0,1), new Cell(0,2), new Cell(0,3) }),
            }),
            ['O'] = UniqueRotations(new[]
            {
                new Shape(new[] { new Cell(0,0), new Cell(1,0), new Cell(0,1), new Cell(1,1) }),
            }),
            ['T'] = UniqueRotations(new[]
            {
                new Shape(new[] { new Cell(0,0), new Cell(1,0), new Cell(2,0), new Cell(1,1) }),
                new Shape(new[] { new Cell(0,0), new Cell(0,1), new Cell(0,2), new Cell(1,1) }),
                new Shape(new[] { new Cell(1,0), new Cell(0,1), new Cell(1,1), new Cell(2,1) }),
                new Shape(new[] { new Cell(1,0), new Cell(0,1), new Cell(1,1), new Cell(1,2) }),
            }),
            ['S'] = UniqueRotations(new[]
            {
                new Shape(new[] { new Cell(1,0), new Cell(2,0), new Cell(0,1), new Cell(1,1) }),
                new Shape(new[] { new Cell(0,0), new Cell(0,1), new Cell(1,1), new Cell(1,2) }),
            }),
            ['Z'] = UniqueRotations(new[]
            {
                new Shape(new[] { new Cell(0,0), new Cell(1,0), new Cell(1,1), new Cell(2,1) }),
                new Shape(new[] { new Cell(1,0), new Cell(0,1), new Cell(1,1), new Cell(0,2) }),
            }),
            ['J'] = UniqueRotations(new[]
            {
                new Shape(new[] { new Cell(0,0), new Cell(0,1), new Cell(1,1), new Cell(2,1) }),
                new Shape(new[] { new Cell(0,0), new Cell(1,0), new Cell(0,1), new Cell(0,2) }),
                new Shape(new[] { new Cell(0,0), new Cell(1,0), new Cell(2,0), new Cell(2,1) }),
                new Shape(new[] { new Cell(1,0), new Cell(1,1), new Cell(0,2), new Cell(1,2) }),
            }),
            ['L'] = UniqueRotations(new[]
            {
                new Shape(new[] { new Cell(2,0), new Cell(0,1), new Cell(1,1), new Cell(2,1) }),
                new Shape(new[] { new Cell(0,0), new Cell(0,1), new Cell(0,2), new Cell(1,2) }),
                new Shape(new[] { new Cell(0,0), new Cell(1,0), new Cell(2,0), new Cell(0,1) }),
                new Shape(new[] { new Cell(0,0), new Cell(1,0), new Cell(1,1), new Cell(1,2) }),
            }),
        };
    }

    private static Shape[] UniqueRotations(Shape[] shapes)
    {
        var seen = new HashSet<string>(StringComparer.Ordinal);
        var result = new List<Shape>(4);

        for (int i = 0; i < shapes.Length; i++)
        {
            string key = ShapeKey(shapes[i]);
            if (seen.Add(key)) result.Add(shapes[i]);
        }

        return result.ToArray();
    }

    private static string ShapeKey(Shape s)
    {
        var cells = new (int x, int y)[s.Cells.Length];
        for (int i = 0; i < s.Cells.Length; i++) cells[i] = (s.Cells[i].X, s.Cells[i].Y);
        Array.Sort(cells, (a, b) => a.y != b.y ? a.y.CompareTo(b.y) : a.x.CompareTo(b.x));

        var sw = new StringWriter(CultureInfo.InvariantCulture);
        for (int i = 0; i < cells.Length; i++)
        {
            sw.Write(cells[i].x);
            sw.Write(',');
            sw.Write(cells[i].y);
            sw.Write(';');
        }
        return sw.ToString();
    }

    private sealed class FastScanner
    {
        private readonly TextReader _reader;
        private readonly char[] _buffer = new char[1 << 14];
        private int _len;
        private int _pos;

        public FastScanner(TextReader reader)
        {
            _reader = reader;
        }

        public string NextToken()
        {
            SkipWhite();
            if (!EnsureData()) return string.Empty;

            var sw = new StringWriter(CultureInfo.InvariantCulture);
            while (true)
            {
                if (!EnsureData()) break;
                char ch = _buffer[_pos];
                if (char.IsWhiteSpace(ch)) break;
                sw.Write(ch);
                _pos++;
            }
            return sw.ToString();
        }

        public int NextInt()
        {
            string t = NextToken();
            return t.Length == 0 ? 0 : int.Parse(t, CultureInfo.InvariantCulture);
        }

        private void SkipWhite()
        {
            while (true)
            {
                if (!EnsureData()) return;
                if (!char.IsWhiteSpace(_buffer[_pos])) return;
                _pos++;
            }
        }

        private bool EnsureData()
        {
            if (_pos < _len) return true;
            _len = _reader.Read(_buffer, 0, _buffer.Length);
            _pos = 0;
            return _len > 0;
        }
    }
}
