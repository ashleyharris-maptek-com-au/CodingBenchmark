using System;
using System.Collections.Generic;
using System.Text;

public static class Program
{
    private const byte Unknown = 0;
    private const byte Number = 1;
    private const byte Mine = 2;
    private const byte Safe = 3;

    private static int _h;
    private static int _w;
    private static byte[] _type = Array.Empty<byte>();
    private static byte[] _value = Array.Empty<byte>();
    private static bool[] _queued = Array.Empty<bool>();
    private static readonly Queue<int> WorkQueue = new Queue<int>(1024);
    private static readonly List<int> NumberCells = new List<int>(1024);
    private static readonly List<int> SafeCells = new List<int>(1024);

    public static void Main()
    {
        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.WriteLine("0");
            return;
        }

        string[] hw = first.Split((char[])null!, StringSplitOptions.RemoveEmptyEntries);
        if (hw.Length < 2)
        {
            Console.WriteLine("0");
            return;
        }

        _h = int.Parse(hw[0]);
        _w = int.Parse(hw[1]);
        int n = checked(_h * _w);

        _type = new byte[n];
        _value = new byte[n];
        _queued = new bool[n];

        for (int y = 0; y < _h; y++)
        {
            string? raw = Console.ReadLine();
            raw ??= string.Empty;
            if (raw.Length < _w) raw = raw.PadRight(_w, ' ');
            else if (raw.Length > _w) raw = raw.Substring(0, _w);

            int rowBase = y * _w;
            for (int x = 0; x < _w; x++)
            {
                char c = raw[x];
                int idx = rowBase + x;
                if (c == ' ')
                {
                    _type[idx] = Unknown;
                }
                else if (c == '*')
                {
                    _type[idx] = Mine;
                }
                else if (c >= '0' && c <= '8')
                {
                    _type[idx] = Number;
                    _value[idx] = (byte)(c - '0');
                    NumberCells.Add(idx);
                }
                else
                {
                    _type[idx] = Unknown;
                }
            }
        }

        foreach (int idx in NumberCells) Enqueue(idx);

        long boardArea = (long)_h * _w;
        bool enableAdvanced = boardArea <= 120_000_000L && NumberCells.Count > 0;

        bool changed;
        do
        {
            changed = RunDeterministicQueue();
            if (enableAdvanced)
            {
                bool advancedChanged = RunLocalSubsetPass(boardArea);
                changed |= advancedChanged;
            }
        } while (changed);

        var sb = new StringBuilder();
        sb.Append(SafeCells.Count).Append('\n');
        foreach (int idx in SafeCells)
        {
            int y = idx / _w;
            int x = idx - y * _w;
            sb.Append(x).Append(' ').Append(y).Append('\n');
        }

        Console.Write(sb.ToString());
    }

    private static void Enqueue(int idx)
    {
        if (_type[idx] != Number || _queued[idx]) return;
        _queued[idx] = true;
        WorkQueue.Enqueue(idx);
    }

    private static bool RunDeterministicQueue()
    {
        bool changed = false;
        int[] unknownBuf = new int[8];

        while (WorkQueue.Count > 0)
        {
            int idx = WorkQueue.Dequeue();
            _queued[idx] = false;
            if (_type[idx] != Number) continue;

            Analyze(idx, unknownBuf, out int unknownCount, out int remaining);
            if (unknownCount == 0) continue;

            if (remaining == 0)
            {
                for (int i = 0; i < unknownCount; i++) changed |= MarkSafe(unknownBuf[i]);
            }
            else if (remaining == unknownCount)
            {
                for (int i = 0; i < unknownCount; i++) changed |= MarkMine(unknownBuf[i]);
            }
        }

        return changed;
    }

    private static bool RunLocalSubsetPass(long boardArea)
    {
        // Adaptive cap: tighter for huge boards to maintain throughput.
        int maxComparisons = boardArea <= 2_000_000L ? 2_000_000 : 500_000;
        int comparisons = 0;
        bool changed = false;

        int[] aBuf = new int[8];
        int[] bBuf = new int[8];
        int[] diffBuf = new int[8];

        for (int ni = 0; ni < NumberCells.Count; ni++)
        {
            int aIdx = NumberCells[ni];
            if (_type[aIdx] != Number) continue;

            Analyze(aIdx, aBuf, out int aCount, out int aRemaining);
            if (aCount == 0) continue;

            int ay = aIdx / _w;
            int ax = aIdx - ay * _w;

            int y0 = ay > 2 ? ay - 2 : 0;
            int y1 = ay + 2 < _h ? ay + 2 : _h - 1;
            int x0 = ax > 2 ? ax - 2 : 0;
            int x1 = ax + 2 < _w ? ax + 2 : _w - 1;

            for (int y = y0; y <= y1; y++)
            {
                int rowBase = y * _w;
                for (int x = x0; x <= x1; x++)
                {
                    int bIdx = rowBase + x;
                    if (bIdx == aIdx || _type[bIdx] != Number) continue;

                    Analyze(bIdx, bBuf, out int bCount, out int bRemaining);
                    if (bCount == 0) continue;

                    comparisons++;
                    if (comparisons > maxComparisons) return changed;

                    if (TryInferFromSubset(aBuf, aCount, aRemaining, bBuf, bCount, bRemaining, diffBuf, out int diffCount, out int diffMines))
                    {
                        if (diffMines == 0)
                        {
                            for (int i = 0; i < diffCount; i++) changed |= MarkSafe(diffBuf[i]);
                        }
                        else if (diffMines == diffCount)
                        {
                            for (int i = 0; i < diffCount; i++) changed |= MarkMine(diffBuf[i]);
                        }
                    }

                    if (TryInferFromSubset(bBuf, bCount, bRemaining, aBuf, aCount, aRemaining, diffBuf, out diffCount, out diffMines))
                    {
                        if (diffMines == 0)
                        {
                            for (int i = 0; i < diffCount; i++) changed |= MarkSafe(diffBuf[i]);
                        }
                        else if (diffMines == diffCount)
                        {
                            for (int i = 0; i < diffCount; i++) changed |= MarkMine(diffBuf[i]);
                        }
                    }
                }
            }
        }

        return changed;
    }

    private static bool TryInferFromSubset(
        int[] subSet,
        int subCount,
        int subMines,
        int[] superSet,
        int superCount,
        int superMines,
        int[] diffBuf,
        out int diffCount,
        out int diffMines)
    {
        diffCount = 0;
        diffMines = 0;

        if (subCount > superCount || subMines > superMines) return false;

        for (int i = 0; i < subCount; i++)
        {
            if (!Contains(superSet, superCount, subSet[i])) return false;
        }

        for (int i = 0; i < superCount; i++)
        {
            int v = superSet[i];
            if (!Contains(subSet, subCount, v)) diffBuf[diffCount++] = v;
        }

        diffMines = superMines - subMines;
        if (diffCount == 0) return false;
        if (diffMines < 0 || diffMines > diffCount) return false;

        return diffMines == 0 || diffMines == diffCount;
    }

    private static bool Contains(int[] arr, int count, int v)
    {
        for (int i = 0; i < count; i++)
        {
            if (arr[i] == v) return true;
        }
        return false;
    }

    private static void Analyze(int idx, int[] unknownBuf, out int unknownCount, out int remaining)
    {
        unknownCount = 0;
        int knownMines = 0;

        int y = idx / _w;
        int x = idx - y * _w;

        int y0 = y > 0 ? y - 1 : 0;
        int y1 = y + 1 < _h ? y + 1 : _h - 1;
        int x0 = x > 0 ? x - 1 : 0;
        int x1 = x + 1 < _w ? x + 1 : _w - 1;

        for (int ny = y0; ny <= y1; ny++)
        {
            int rowBase = ny * _w;
            for (int nx = x0; nx <= x1; nx++)
            {
                if (nx == x && ny == y) continue;
                int nIdx = rowBase + nx;
                byte t = _type[nIdx];
                if (t == Mine) knownMines++;
                else if (t == Unknown) unknownBuf[unknownCount++] = nIdx;
            }
        }

        remaining = _value[idx] - knownMines;
        if (remaining < 0) remaining = 0;
        if (remaining > unknownCount) remaining = unknownCount;
    }

    private static bool MarkMine(int idx)
    {
        if (_type[idx] != Unknown) return false;
        _type[idx] = Mine;
        EnqueueAdjacentNumbers(idx);
        return true;
    }

    private static bool MarkSafe(int idx)
    {
        if (_type[idx] != Unknown) return false;
        _type[idx] = Safe;
        SafeCells.Add(idx);
        EnqueueAdjacentNumbers(idx);
        return true;
    }

    private static void EnqueueAdjacentNumbers(int idx)
    {
        int y = idx / _w;
        int x = idx - y * _w;

        int y0 = y > 0 ? y - 1 : 0;
        int y1 = y + 1 < _h ? y + 1 : _h - 1;
        int x0 = x > 0 ? x - 1 : 0;
        int x1 = x + 1 < _w ? x + 1 : _w - 1;

        for (int ny = y0; ny <= y1; ny++)
        {
            int rowBase = ny * _w;
            for (int nx = x0; nx <= x1; nx++)
            {
                int nIdx = rowBase + nx;
                if (_type[nIdx] == Number) Enqueue(nIdx);
            }
        }
    }
}
