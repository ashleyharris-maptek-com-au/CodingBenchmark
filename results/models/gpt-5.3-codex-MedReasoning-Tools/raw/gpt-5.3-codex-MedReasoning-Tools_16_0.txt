#include <algorithm>
#include <cstdint>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct Task {
    int machine;
    int duration;
};

struct Instance {
    int J = 0;
    int M = 0;
    vector<vector<Task>> jobs;
    vector<long long> totalJobDur;
    vector<long long> totalMachDur;
    int N = 0;
};

struct Schedule {
    vector<vector<long long>> start;
};

static inline long long llmax(long long a, long long b) { return a > b ? a : b; }

static Instance readInstance() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Instance ins;
    if (!(cin >> ins.J >> ins.M)) return ins;
    ins.jobs.assign(ins.J, {});
    ins.totalJobDur.assign(ins.J, 0);
    ins.totalMachDur.assign(ins.M, 0);

    for (int j = 0; j < ins.J; ++j) {
        int T;
        cin >> T;
        ins.jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int m, d;
            cin >> m >> d;
            if (m < 0) m = 0;
            if (m >= ins.M) m = ins.M - 1;
            if (d < 0) d = 0;
            ins.jobs[j][t] = {m, d};
            ins.totalJobDur[j] += d;
            ins.totalMachDur[m] += d;
            ins.N++;
        }
    }
    return ins;
}

class TinyExactSolver {
public:
    explicit TinyExactSolver(const Instance& in) : ins(in) {
        J = ins.J;
        M = ins.M;
        nextTask.assign(J, 0);
        jobReady.assign(J, 0);
        machReady.assign(M, 0);

        curStart.resize(J);
        bestStart.resize(J);
        remJob.assign(J, 0);
        remMach.assign(M, 0);
        jobSuffix.assign(J, {});
        for (int j = 0; j < J; ++j) {
            int T = (int)ins.jobs[j].size();
            curStart[j].assign(T, -1);
            bestStart[j].assign(T, 0);
            jobSuffix[j].assign(T + 1, 0);
            for (int t = T - 1; t >= 0; --t) {
                jobSuffix[j][t] = jobSuffix[j][t + 1] + ins.jobs[j][t].duration;
            }
            remJob[j] = jobSuffix[j][0];
        }
        for (int m = 0; m < M; ++m) remMach[m] = ins.totalMachDur[m];

        bestMakespan = numeric_limits<long long>::max();
    }

    Schedule solve() {
        dfs(0);
        return Schedule{bestStart};
    }

private:
    const Instance& ins;
    int J = 0, M = 0;
    vector<int> nextTask;
    vector<long long> jobReady;
    vector<long long> machReady;
    vector<vector<long long>> curStart;
    vector<vector<long long>> bestStart;
    vector<long long> remJob;
    vector<long long> remMach;
    vector<vector<long long>> jobSuffix;
    long long bestMakespan;

    long long lowerBoundCurrent() {
        long long lb = 0;
        for (int j = 0; j < J; ++j) {
            lb = max(lb, jobReady[j] + remJob[j]);
        }
        for (int m = 0; m < M; ++m) {
            lb = max(lb, machReady[m] + remMach[m]);
        }
        return lb;
    }

    void dfs(int done) {
        if (done == ins.N) {
            long long mk = 0;
            for (int j = 0; j < J; ++j) mk = max(mk, jobReady[j]);
            if (mk < bestMakespan) {
                bestMakespan = mk;
                bestStart = curStart;
            }
            return;
        }

        if (lowerBoundCurrent() >= bestMakespan) return;

        vector<int> candidates;
        candidates.reserve(J);
        for (int j = 0; j < J; ++j) {
            if (nextTask[j] < (int)ins.jobs[j].size()) candidates.push_back(j);
        }

        stable_sort(candidates.begin(), candidates.end(), [&](int a, int b) {
            const Task& ta = ins.jobs[a][nextTask[a]];
            const Task& tb = ins.jobs[b][nextTask[b]];
            long long ea = llmax(jobReady[a], machReady[ta.machine]);
            long long eb = llmax(jobReady[b], machReady[tb.machine]);
            if (ea != eb) return ea < eb;
            if (ta.duration != tb.duration) return ta.duration < tb.duration;
            return a < b;
        });

        for (int j : candidates) {
            int t = nextTask[j];
            const Task& op = ins.jobs[j][t];
            int m = op.machine;
            int d = op.duration;

            long long oldJR = jobReady[j];
            long long oldMR = machReady[m];
            long long st = llmax(oldJR, oldMR);
            long long en = st + d;

            curStart[j][t] = st;
            nextTask[j]++;
            jobReady[j] = en;
            machReady[m] = en;
            remJob[j] -= d;
            remMach[m] -= d;

            dfs(done + 1);

            remMach[m] += d;
            remJob[j] += d;
            machReady[m] = oldMR;
            jobReady[j] = oldJR;
            nextTask[j]--;
            curStart[j][t] = -1;
        }
    }
};

struct HeapNode {
    long long est;
    long long score;
    int job;
    int ver;
    bool operator<(const HeapNode& other) const {
        if (est != other.est) return est > other.est;
        if (score != other.score) return score > other.score;
        return job > other.job;
    }
};

class AdaptiveHeuristicSolver {
public:
    explicit AdaptiveHeuristicSolver(const Instance& in) : ins(in) {
        J = ins.J;
        M = ins.M;
        nextTask.assign(J, 0);
        version.assign(J, 0);
        jobReady.assign(J, 0);
        machReady.assign(M, 0);
        remJob = ins.totalJobDur;
        start.resize(J);
        for (int j = 0; j < J; ++j) start[j].assign(ins.jobs[j].size(), 0);
    }

    Schedule solve(bool mediumMode) {
        priority_queue<HeapNode> pq;
        for (int j = 0; j < J; ++j) {
            if (!ins.jobs[j].empty()) pushCandidate(pq, j, mediumMode);
        }

        int done = 0;
        while (done < ins.N) {
            if (pq.empty()) break;

            HeapNode cur = pq.top();
            pq.pop();

            int j = cur.job;
            if (cur.ver != version[j]) continue;
            if (nextTask[j] >= (int)ins.jobs[j].size()) continue;

            int t = nextTask[j];
            const Task& op = ins.jobs[j][t];
            int m = op.machine;
            long long estNow = llmax(jobReady[j], machReady[m]);
            long long scoreNow = computeScore(j, op.duration, mediumMode);

            if (estNow != cur.est || scoreNow != cur.score) {
                pushCandidate(pq, j, mediumMode);
                continue;
            }

            long long st = estNow;
            long long en = st + op.duration;
            start[j][t] = st;
            nextTask[j]++;
            jobReady[j] = en;
            machReady[m] = en;
            remJob[j] -= op.duration;
            done++;

            version[j]++;
            if (nextTask[j] < (int)ins.jobs[j].size()) pushCandidate(pq, j, mediumMode);
        }

        return Schedule{start};
    }

private:
    const Instance& ins;
    int J = 0, M = 0;
    vector<int> nextTask;
    vector<int> version;
    vector<long long> jobReady;
    vector<long long> machReady;
    vector<long long> remJob;
    vector<vector<long long>> start;

    long long computeScore(int j, int d, bool mediumMode) const {
        if (!mediumMode) return d;
        return (long long)d * 1024 - remJob[j];
    }

    void pushCandidate(priority_queue<HeapNode>& pq, int j, bool mediumMode) {
        int t = nextTask[j];
        const Task& op = ins.jobs[j][t];
        long long est = llmax(jobReady[j], machReady[op.machine]);
        long long score = computeScore(j, op.duration, mediumMode);
        pq.push(HeapNode{est, score, j, version[j]});
    }
};

static Schedule solveAdaptive(const Instance& ins) {
    const int J = ins.J;
    const int M = ins.M;
    const int N = ins.N;

    bool tiny = (N <= 14 && J <= 8 && M <= 8);
    bool medium = (N <= 20000 && J <= 400 && M <= 40);

    if (tiny) {
        TinyExactSolver solver(ins);
        return solver.solve();
    }

    AdaptiveHeuristicSolver solver(ins);
    return solver.solve(medium);
}

int main() {
    Instance ins = readInstance();
    if (ins.J <= 0 || ins.M <= 0) return 0;

    Schedule sol = solveAdaptive(ins);

    for (int j = 0; j < ins.J; ++j) {
        const int T = (int)ins.jobs[j].size();
        for (int t = 0; t < T; ++t) {
            if (t) cout << ' ';
            cout << sol.start[j][t] << ' ' << ins.jobs[j][t].machine;
        }
        cout << '\n';
    }
    return 0;
}
