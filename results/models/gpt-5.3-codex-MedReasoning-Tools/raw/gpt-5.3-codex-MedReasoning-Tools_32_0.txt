#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct InputEdge {
    int u;
    int v;
    long long w;
};

struct AdjEdge {
    int to;
    long long w;
    int id;
};

struct DSU {
    vector<int> p, r;
    explicit DSU(int n) : p(n), r(n, 0) { iota(p.begin(), p.end(), 0); }
    int find(int x) { return p[x] == x ? x : (p[x] = find(p[x])); }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

struct ReconInfo {
    uint8_t type; // 0 none, 1 base, 2 merge, 3 path
    int a;        // submask (merge) or prev vertex (path)
    int edgeId;   // edge for path
};

static const long long INF = numeric_limits<long long>::max() / 4;

static vector<long long> dijkstra_dist(int n, const vector<vector<AdjEdge>>& g, int src, vector<int>* parentV = nullptr, vector<int>* parentE = nullptr) {
    vector<long long> dist(n, INF);
    if (parentV) parentV->assign(n, -1);
    if (parentE) parentE->assign(n, -1);

    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) continue;
        for (const auto& e : g[u]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                if (parentV) (*parentV)[e.to] = u;
                if (parentE) (*parentE)[e.to] = e.id;
                pq.push({nd, e.to});
            }
        }
    }
    return dist;
}

static vector<int> kruskal_filter(int n, const vector<InputEdge>& edges, const vector<int>& edgeIds) {
    vector<int> ids = edgeIds;
    sort(ids.begin(), ids.end(), [&](int a, int b) {
        if (edges[a].w != edges[b].w) return edges[a].w < edges[b].w;
        return a < b;
    });
    DSU dsu(n);
    vector<int> out;
    out.reserve(ids.size());
    for (int id : ids) {
        if (dsu.unite(edges[id].u, edges[id].v)) out.push_back(id);
    }
    return out;
}

static vector<int> prune_nonterminal_leaves(int n, const vector<InputEdge>& edges, const vector<int>& edgeIds, const vector<char>& isTerminal) {
    vector<vector<pair<int, int>>> adj(n);
    adj.assign(n, {});
    vector<char> alive(edgeIds.size(), 1);
    vector<int> deg(n, 0);

    for (int i = 0; i < (int)edgeIds.size(); ++i) {
        int id = edgeIds[i];
        int u = edges[id].u, v = edges[id].v;
        adj[u].push_back({v, i});
        adj[v].push_back({u, i});
        ++deg[u];
        ++deg[v];
    }

    queue<int> q;
    for (int v = 0; v < n; ++v) {
        if (!isTerminal[v] && deg[v] <= 1) q.push(v);
    }

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        if (isTerminal[u] || deg[u] > 1) continue;
        for (auto [to, idx] : adj[u]) {
            if (!alive[idx]) continue;
            alive[idx] = 0;
            --deg[u];
            --deg[to];
            if (!isTerminal[to] && deg[to] <= 1) q.push(to);
            break;
        }
    }

    vector<int> out;
    for (int i = 0; i < (int)edgeIds.size(); ++i) {
        if (alive[i]) out.push_back(edgeIds[i]);
    }
    return out;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<InputEdge> edges(m);
    vector<vector<AdjEdge>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges[i] = {u, v, w};
        g[u].push_back({v, w, i});
        g[v].push_back({u, w, i});
    }

    vector<int> terminals(k);
    vector<char> isTerminal(n, 0);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
        if (0 <= terminals[i] && terminals[i] < n) isTerminal[terminals[i]] = 1;
    }

    if (k == 0) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }
    if (k == 1) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    vector<int> selectedEdgeIds;

    const int EXACT_K_LIMIT = 18;
    bool canExact = (k <= EXACT_K_LIMIT);
    if (canExact) {
        long long states = (1LL << k) * 1LL * n;
        if (states > 12000000LL) canExact = false;
    }

    if (canExact) {
        int fullMask = (1 << k) - 1;
        int totalMasks = 1 << k;
        vector<long long> dp((size_t)totalMasks * n, INF);
        vector<ReconInfo> rec((size_t)totalMasks * n, {0, -1, -1});

        auto idx = [n](int mask, int v) -> size_t {
            return (size_t)mask * n + v;
        };

        for (int i = 0; i < k; ++i) {
            int mask = 1 << i;
            dp[idx(mask, terminals[i])] = 0;
            rec[idx(mask, terminals[i])] = {1, -1, -1};
        }

        for (int mask = 1; mask <= fullMask; ++mask) {
            if ((mask & (mask - 1)) != 0) {
                int firstBit = mask & -mask;
                for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
                    if ((sub & firstBit) == 0) continue;
                    int other = mask ^ sub;
                    if (other == 0) continue;
                    for (int v = 0; v < n; ++v) {
                        long long a = dp[idx(sub, v)];
                        long long b = dp[idx(other, v)];
                        if (a == INF || b == INF) continue;
                        long long cand = a + b;
                        size_t id = idx(mask, v);
                        if (cand < dp[id]) {
                            dp[id] = cand;
                            rec[id] = {2, sub, -1};
                        }
                    }
                }
            }

            priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
            for (int v = 0; v < n; ++v) {
                long long d = dp[idx(mask, v)];
                if (d < INF) pq.push({d, v});
            }

            while (!pq.empty()) {
                auto [d, u] = pq.top();
                pq.pop();
                if (d != dp[idx(mask, u)]) continue;
                for (const auto& e : g[u]) {
                    long long nd = d + e.w;
                    size_t toId = idx(mask, e.to);
                    if (nd < dp[toId]) {
                        dp[toId] = nd;
                        rec[toId] = {3, u, e.id};
                        pq.push({nd, e.to});
                    }
                }
            }
        }

        int root = -1;
        long long best = INF;
        for (int v = 0; v < n; ++v) {
            long long d = dp[idx(fullMask, v)];
            if (d < best) {
                best = d;
                root = v;
            }
        }

        if (root != -1 && best < INF) {
            vector<char> usedEdge(m, 0);
            vector<char> seen((size_t)totalMasks * n, 0);

            vector<pair<int, int>> st;
            st.push_back({fullMask, root});
            while (!st.empty()) {
                auto [mask, v] = st.back();
                st.pop_back();
                size_t id = idx(mask, v);
                if (seen[id]) continue;
                seen[id] = 1;
                ReconInfo ri = rec[id];
                if (ri.type == 0) continue;
                if (ri.type == 1) {
                    continue;
                } else if (ri.type == 2) {
                    int sub = ri.a;
                    st.push_back({sub, v});
                    st.push_back({mask ^ sub, v});
                } else if (ri.type == 3) {
                    int prevV = ri.a;
                    int eId = ri.edgeId;
                    if (eId >= 0) usedEdge[eId] = 1;
                    st.push_back({mask, prevV});
                }
            }

            for (int i = 0; i < m; ++i) if (usedEdge[i]) selectedEdgeIds.push_back(i);
        }
    }

    if (selectedEdgeIds.empty()) {
        vector<vector<long long>> termDist(k);
        vector<vector<int>> parentV(k), parentE(k);
        termDist.reserve(k);

        for (int i = 0; i < k; ++i) {
            termDist[i] = dijkstra_dist(n, g, terminals[i], &parentV[i], &parentE[i]);
        }

        vector<long long> minCost(k, INF);
        vector<int> parent(k, -1);
        vector<char> inTree(k, 0);
        minCost[0] = 0;

        for (int it = 0; it < k; ++it) {
            int u = -1;
            for (int i = 0; i < k; ++i) {
                if (!inTree[i] && (u == -1 || minCost[i] < minCost[u])) u = i;
            }
            if (u == -1 || minCost[u] == INF) break;
            inTree[u] = 1;
            for (int v = 0; v < k; ++v) {
                if (!inTree[v] && termDist[u][terminals[v]] < minCost[v]) {
                    minCost[v] = termDist[u][terminals[v]];
                    parent[v] = u;
                }
            }
        }

        vector<char> usedEdge(m, 0);
        for (int v = 1; v < k; ++v) {
            int u = parent[v];
            if (u < 0) continue;
            int cur = terminals[v];
            int src = terminals[u];
            while (cur != src && cur != -1) {
                int eId = parentE[u][cur];
                int pv = parentV[u][cur];
                if (eId < 0 || pv < 0) break;
                usedEdge[eId] = 1;
                cur = pv;
            }
        }

        for (int i = 0; i < m; ++i) if (usedEdge[i]) selectedEdgeIds.push_back(i);
    }

    if (selectedEdgeIds.empty()) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    selectedEdgeIds = kruskal_filter(n, edges, selectedEdgeIds);
    selectedEdgeIds = prune_nonterminal_leaves(n, edges, selectedEdgeIds, isTerminal);

    long long totalWeight = 0;
    for (int id : selectedEdgeIds) totalWeight += edges[id].w;

    cout << totalWeight << ' ' << selectedEdgeIds.size() << '\n';
    for (int id : selectedEdgeIds) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }

    return 0;
}
