using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public static class Program
{
    private struct Cut
    {
        public int Length;
        public int Index;
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastInput(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    // Treap keyed by remaining length; each node holds a stack of stock ids with that remainder.
    private sealed class RemTreap
    {
        private sealed class Node
        {
            public int Key;
            public uint Pri;
            public Node L;
            public Node R;
            public int[] Stocks;
            public int Count;

            public Node(int key, uint pri, int stock)
            {
                Key = key;
                Pri = pri;
                Stocks = new int[4];
                Stocks[0] = stock;
                Count = 1;
            }

            public void Push(int stock)
            {
                if (Count == Stocks.Length) Array.Resize(ref Stocks, Count << 1);
                Stocks[Count++] = stock;
            }

            public int Pop()
            {
                return Stocks[--Count];
            }
        }

        private Node _root;
        private uint _rng = 2463534242u;

        private uint NextRand()
        {
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return x;
        }

        private static Node RotateRight(Node p)
        {
            Node q = p.L;
            p.L = q.R;
            q.R = p;
            return q;
        }

        private static Node RotateLeft(Node p)
        {
            Node q = p.R;
            p.R = q.L;
            q.L = p;
            return q;
        }

        private Node Insert(Node n, int key, int stock)
        {
            if (n == null) return new Node(key, NextRand(), stock);

            if (key == n.Key)
            {
                n.Push(stock);
                return n;
            }

            if (key < n.Key)
            {
                n.L = Insert(n.L, key, stock);
                if (n.L.Pri < n.Pri) n = RotateRight(n);
            }
            else
            {
                n.R = Insert(n.R, key, stock);
                if (n.R.Pri < n.Pri) n = RotateLeft(n);
            }

            return n;
        }

        public void Add(int key, int stock)
        {
            _root = Insert(_root, key, stock);
        }

        private static Node Merge(Node a, Node b)
        {
            if (a == null) return b;
            if (b == null) return a;

            if (a.Pri < b.Pri)
            {
                a.R = Merge(a.R, b);
                return a;
            }
            else
            {
                b.L = Merge(a, b.L);
                return b;
            }
        }

        // Pop one stock id from the smallest key >= need. Returns false if none exists.
        public bool TryTakeCeiling(int need, out int key, out int stock)
        {
            key = 0;
            stock = -1;
            if (_root == null) return false;

            Node cur = _root;
            Node cand = null;
            while (cur != null)
            {
                if (cur.Key >= need)
                {
                    cand = cur;
                    cur = cur.L;
                }
                else
                {
                    cur = cur.R;
                }
            }

            if (cand == null) return false;

            key = cand.Key;

            _root = DeleteOne(_root, key, out stock);
            return true;
        }

        private Node DeleteOne(Node n, int key, out int stock)
        {
            if (n == null)
            {
                stock = -1;
                return null;
            }

            if (key < n.Key)
            {
                n.L = DeleteOne(n.L, key, out stock);
                return n;
            }

            if (key > n.Key)
            {
                n.R = DeleteOne(n.R, key, out stock);
                return n;
            }

            // Found key
            stock = n.Pop();
            if (n.Count > 0) return n;
            return Merge(n.L, n.R);
        }
    }

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());
        int n = input.NextInt();
        int stockLen = input.NextInt();

        if (n <= 0 || stockLen <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int len = input.NextInt();
            if (len == int.MinValue) len = 0;
            if (len > stockLen)
            {
                // Invalid input per constraints; still emit empty feasible output.
                Console.WriteLine(0);
                return;
            }
            cuts[i] = new Cut { Length = len, Index = i };
        }

        Array.Sort(cuts, (a, b) => b.Length.CompareTo(a.Length));

        // Linked-list style assignment storage to avoid per-stock List overhead.
        int[] nextInStock = new int[n];
        Array.Fill(nextInStock, -1);

        int capacity = Math.Max(4, n / 8);
        int[] stockHead = new int[capacity];
        int[] stockTail = new int[capacity];
        Array.Fill(stockHead, -1);
        Array.Fill(stockTail, -1);

        int stockCount = 0;

        var remTree = new RemTreap();

        for (int i = 0; i < n; i++)
        {
            int len = cuts[i].Length;
            int cutIdx = cuts[i].Index;

            int remKey;
            int stockId;
            if (remTree.TryTakeCeiling(len, out remKey, out stockId))
            {
                int newRem = remKey - len;

                int tail = stockTail[stockId];
                nextInStock[tail] = cutIdx;
                stockTail[stockId] = cutIdx;

                remTree.Add(newRem, stockId);
            }
            else
            {
                if (stockCount == stockHead.Length)
                {
                    int newCap = stockHead.Length << 1;
                    Array.Resize(ref stockHead, newCap);
                    Array.Resize(ref stockTail, newCap);
                    for (int k = stockCount; k < newCap; k++)
                    {
                        stockHead[k] = -1;
                        stockTail[k] = -1;
                    }
                }

                stockId = stockCount++;
                stockHead[stockId] = cutIdx;
                stockTail[stockId] = cutIdx;

                int rem = stockLen - len;
                remTree.Add(rem, stockId);
            }
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        writer.WriteLine(stockCount);
        for (int s = 0; s < stockCount; s++)
        {
            int cur = stockHead[s];
            bool first = true;
            while (cur != -1)
            {
                if (!first) writer.Write(' ');
                writer.Write(cur);
                first = false;
                cur = nextInStock[cur];
            }
            writer.WriteLine();
        }
    }
}
