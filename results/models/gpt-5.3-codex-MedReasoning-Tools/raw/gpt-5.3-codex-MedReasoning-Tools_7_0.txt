struct Vec3 {
    double x = 0, y = 0, z = 0;
    Vec3() = default;
    Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}

struct Tri { int a, b, c; };

struct AABB {
    Vec3 mn{DBL_MAX, DBL_MAX, DBL_MAX};
    Vec3 mx{-DBL_MAX, -DBL_MAX, -DBL_MAX};
    void expand(const Vec3& p) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }
    void expand(const AABB& b) { expand(b.mn); expand(b.mx); }
};

static inline bool overlap(const AABB& a, const AABB& b) {
    return !(a.mx.x < b.mn.x || b.mx.x < a.mn.x || a.mx.y < b.mn.y || b.mx.y < a.mn.y || a.mx.z < b.mn.z || b.mx.z < a.mn.z);
}

struct Mesh {
    std::vector<Vec3> v;
    std::vector<Tri> f;
    AABB box;
};

struct BVHNode {
    AABB box;
    int l = -1, r = -1;
    int start = 0, count = 0;
    bool leaf() const { return l < 0; }
};

struct BVH {
    const Mesh* m = nullptr;
    std::vector<int> idx;
    std::vector<BVHNode> nodes;

    AABB triBox(int ti) const {
        AABB b;
        const auto& t = m->f[ti];
        b.expand(m->v[t.a]); b.expand(m->v[t.b]); b.expand(m->v[t.c]);
        return b;
    }

    Vec3 triCentroid(int ti) const {
        const auto& t = m->f[ti];
        const Vec3& A = m->v[t.a]; const Vec3& B = m->v[t.b]; const Vec3& C = m->v[t.c];
        return {(A.x + B.x + C.x) / 3.0, (A.y + B.y + C.y) / 3.0, (A.z + B.z + C.z) / 3.0};
    }

    int buildRec(int s, int e) {
        BVHNode n;
        for (int i = s; i < e; ++i) n.box.expand(triBox(idx[i]));
        int id = (int)nodes.size();
        nodes.push_back(n);

        int cnt = e - s;
        if (cnt <= 8) {
            nodes[id].start = s;
            nodes[id].count = cnt;
            return id;
        }

        AABB cb;
        for (int i = s; i < e; ++i) cb.expand(triCentroid(idx[i]));
        Vec3 ext = cb.mx - cb.mn;
        int axis = (ext.y > ext.x) ? 1 : 0;
        if ((axis == 0 ? ext.x : ext.y) < ext.z) axis = 2;

        int mid = (s + e) / 2;
        std::nth_element(idx.begin() + s, idx.begin() + mid, idx.begin() + e,
            [&](int a, int b) {
                Vec3 ca = triCentroid(a), cb2 = triCentroid(b);
                if (axis == 0) return ca.x < cb2.x;
                if (axis == 1) return ca.y < cb2.y;
                return ca.z < cb2.z;
            });

        nodes[id].l = buildRec(s, mid);
        nodes[id].r = buildRec(mid, e);
        return id;
    }

    void build(const Mesh& mesh) {
        m = &mesh;
        idx.resize(m->f.size());
        std::iota(idx.begin(), idx.end(), 0);
        nodes.clear();
        if (!idx.empty()) buildRec(0, (int)idx.size());
    }

    static bool rayAabbX(const Vec3& o, const AABB& b) {
        if (o.y < b.mn.y || o.y > b.mx.y || o.z < b.mn.z || o.z > b.mx.z) return false;
        return o.x <= b.mx.x;
    }

    bool rayTriPosX(const Vec3& o, const Tri& t) const {
        const Vec3& A = m->v[t.a];
        const Vec3& B = m->v[t.b];
        const Vec3& C = m->v[t.c];

        const double eps = 1e-12;
        Vec3 e1 = B - A, e2 = C - A;
        Vec3 dir{1.0, 0.0, 0.0};
        Vec3 p = cross(dir, e2);
        double det = dot(e1, p);
        if (std::abs(det) < eps) return false;
        double inv = 1.0 / det;
        Vec3 s = o - A;
        double u = dot(s, p) * inv;
        if (u < -eps || u > 1.0 + eps) return false;
        Vec3 q = cross(s, e1);
        double v = dot(dir, q) * inv;
        if (v < -eps || u + v > 1.0 + eps) return false;
        double tHit = dot(e2, q) * inv;
        return tHit > eps;
    }

    bool inside(const Vec3& p) const {
        if (nodes.empty()) return false;
        int hits = 0;
        std::vector<int> st;
        st.push_back(0);
        while (!st.empty()) {
            int id = st.back(); st.pop_back();
            const BVHNode& n = nodes[id];
            if (!rayAabbX(p, n.box)) continue;
            if (n.leaf()) {
                for (int i = 0; i < n.count; ++i) {
                    if (rayTriPosX(p, m->f[idx[n.start + i]])) ++hits;
                }
            } else {
                st.push_back(n.l);
                st.push_back(n.r);
            }
        }
        return (hits & 1) != 0;
    }
};

struct Key3 {
    int x, y, z;
    bool operator==(const Key3& o) const { return x == o.x && y == o.y && z == o.z; }
};
struct KeyHash {
    std::size_t operator()(const Key3& k) const {
        std::uint64_t h = 1469598103934665603ull;
        auto mix = [&](std::uint64_t v){ h ^= v; h *= 1099511628211ull; };
        mix((std::uint32_t)k.x); mix((std::uint32_t)k.y); mix((std::uint32_t)k.z);
        return (std::size_t)h;
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.v.resize(VA); A.f.resize(FA);
    B.v.resize(VB); B.f.resize(FB);

    for (auto& p : A.v) { std::cin >> p.x >> p.y >> p.z; A.box.expand(p); }
    for (auto& t : A.f) std::cin >> t.a >> t.b >> t.c;
    for (auto& p : B.v) { std::cin >> p.x >> p.y >> p.z; B.box.expand(p); }
    for (auto& t : B.f) std::cin >> t.a >> t.b >> t.c;

    if (!overlap(A.box, B.box)) {
        std::cout << (VA + VB) << ' ' << (FA + FB) << '\n';
        for (const auto& p : A.v) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
        for (const auto& p : B.v) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
        for (const auto& t : A.f) std::cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
        for (const auto& t : B.f) std::cout << (t.a + VA) << ' ' << (t.b + VA) << ' ' << (t.c + VA) << '\n';
        return 0;
    }

    BVH bvhA, bvhB;
    bvhA.build(A);
    bvhB.build(B);

    AABB U = A.box;
    U.expand(B.box);

    std::size_t triCount = A.f.size() + B.f.size();
    int baseRes = 64;
    if (triCount < 5000) baseRes = 96;
    if (triCount < 1000) baseRes = 128;
    if (triCount > 100000) baseRes = 56;
    if (triCount > 500000) baseRes = 44;
    if (triCount > 1500000) baseRes = 36;

    Vec3 ext = U.mx - U.mn;
    double maxExt = std::max({ext.x, ext.y, ext.z});
    if (maxExt <= 0) {
        std::cout << "0 0\n";
        return 0;
    }

    int nx = std::max(2, (int)std::lround(baseRes * ext.x / maxExt));
    int ny = std::max(2, (int)std::lround(baseRes * ext.y / maxExt));
    int nz = std::max(2, (int)std::lround(baseRes * ext.z / maxExt));

    std::size_t voxCount = (std::size_t)nx * ny * nz;
    while (voxCount > 120000000ull) {
        nx = std::max(2, nx * 9 / 10);
        ny = std::max(2, ny * 9 / 10);
        nz = std::max(2, nz * 9 / 10);
        voxCount = (std::size_t)nx * ny * nz;
    }

    Vec3 cell{ext.x / nx, ext.y / ny, ext.z / nz};
    std::vector<std::uint8_t> occ(voxCount, 0);

    auto idx3 = [=](int i, int j, int k) -> std::size_t { return ((std::size_t)k * ny + j) * nx + i; };

    for (int k = 0; k < nz; ++k) {
        for (int j = 0; j < ny; ++j) {
            for (int i = 0; i < nx; ++i) {
                Vec3 p{U.mn.x + (i + 0.5) * cell.x, U.mn.y + (j + 0.5) * cell.y, U.mn.z + (k + 0.5) * cell.z};
                bool in = bvhA.inside(p) || bvhB.inside(p);
                occ[idx3(i, j, k)] = (std::uint8_t)(in ? 1 : 0);
            }
        }
    }

    std::vector<Vec3> outV;
    std::vector<Tri> outF;
    std::unordered_map<Key3, int, KeyHash> vmap;
    vmap.reserve(1 << 20);

    auto getV = [&](int gx, int gy, int gz) -> int {
        Key3 k{gx, gy, gz};
        auto it = vmap.find(k);
        if (it != vmap.end()) return it->second;
        Vec3 p{U.mn.x + gx * cell.x, U.mn.y + gy * cell.y, U.mn.z + gz * cell.z};
        int id = (int)outV.size();
        outV.push_back(p);
        vmap.emplace(k, id);
        return id;
    };

    auto inside = [&](int i, int j, int k) -> bool {
        if (i < 0 || i >= nx || j < 0 || j >= ny || k < 0 || k >= nz) return false;
        return occ[idx3(i, j, k)] != 0;
    };

    auto addQuad = [&](int v0, int v1, int v2, int v3) {
        outF.push_back({v0, v1, v2});
        outF.push_back({v0, v2, v3});
    };

    for (int k = 0; k < nz; ++k) {
        for (int j = 0; j < ny; ++j) {
            for (int i = 0; i < nx; ++i) {
                if (!inside(i, j, k)) continue;

                if (!inside(i + 1, j, k)) {
                    int a = getV(i + 1, j, k), b = getV(i + 1, j + 1, k), c = getV(i + 1, j + 1, k + 1), d = getV(i + 1, j, k + 1);
                    addQuad(a, b, c, d);
                }
                if (!inside(i - 1, j, k)) {
                    int a = getV(i, j, k), b = getV(i, j, k + 1), c = getV(i, j + 1, k + 1), d = getV(i, j + 1, k);
                    addQuad(a, b, c, d);
                }
                if (!inside(i, j + 1, k)) {
                    int a = getV(i, j + 1, k), b = getV(i, j + 1, k + 1), c = getV(i + 1, j + 1, k + 1), d = getV(i + 1, j + 1, k);
                    addQuad(a, b, c, d);
                }
                if (!inside(i, j - 1, k)) {
                    int a = getV(i, j, k), b = getV(i + 1, j, k), c = getV(i + 1, j, k + 1), d = getV(i, j, k + 1);
                    addQuad(a, b, c, d);
                }
                if (!inside(i, j, k + 1)) {
                    int a = getV(i, j, k + 1), b = getV(i + 1, j, k + 1), c = getV(i + 1, j + 1, k + 1), d = getV(i, j + 1, k + 1);
                    addQuad(a, b, c, d);
                }
                if (!inside(i, j, k - 1)) {
                    int a = getV(i, j, k), b = getV(i, j + 1, k), c = getV(i + 1, j + 1, k), d = getV(i + 1, j, k);
                    addQuad(a, b, c, d);
                }
            }
        }
    }

    std::cout << outV.size() << ' ' << outF.size() << '\n';
    for (const auto& p : outV) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const auto& t : outF) std::cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
    return 0;
}
