using namespace std;

struct Graph {
    int n = 0;
    int m = 0;
    int k = 0;
    vector<pair<int, int>> edges;
    vector<int> degree;
    vector<int> offset;
    vector<int> adj;
};

static inline uint64_t now_seed() {
    return static_cast<uint64_t>(chrono::high_resolution_clock::now().time_since_epoch().count());
}

Graph read_graph() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Graph g;
    cin >> g.n >> g.m >> g.k;
    g.edges.reserve(g.m);
    g.degree.assign(g.n, 0);

    for (int i = 0; i < g.m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= g.n || v >= g.n || u == v) continue;
        g.edges.emplace_back(u, v);
        g.degree[u]++;
        g.degree[v]++;
    }
    g.m = static_cast<int>(g.edges.size());

    g.offset.assign(g.n + 1, 0);
    for (int i = 0; i < g.n; ++i) g.offset[i + 1] = g.offset[i] + g.degree[i];
    g.adj.assign(g.offset[g.n], -1);

    vector<int> cur = g.offset;
    for (const auto& e : g.edges) {
        int u = e.first;
        int v = e.second;
        g.adj[cur[u]++] = v;
        g.adj[cur[v]++] = u;
    }

    return g;
}

struct DsaturSolver {
    const Graph& g;
    int n;
    int k;
    vector<int> color;
    vector<uint64_t> neigh_used;
    vector<int> sat;
    vector<int> uncolored;
    uint64_t all_mask;
    chrono::steady_clock::time_point deadline;
    bool timed_out = false;

    DsaturSolver(const Graph& graph, chrono::steady_clock::time_point dl)
        : g(graph), n(graph.n), k(graph.k), color(n, -1), neigh_used(n, 0), sat(n, 0), uncolored(n, 1), deadline(dl) {
        all_mask = (k >= 64) ? ~0ULL : ((1ULL << k) - 1ULL);
    }

    int choose_vertex() {
        int best = -1;
        int best_sat = -1;
        int best_deg = -1;
        for (int v = 0; v < n; ++v) {
            if (!uncolored[v]) continue;
            if (sat[v] > best_sat || (sat[v] == best_sat && g.degree[v] > best_deg)) {
                best = v;
                best_sat = sat[v];
                best_deg = g.degree[v];
            }
        }
        return best;
    }

    bool dfs(int left) {
        if ((left & 255) == 0 && chrono::steady_clock::now() > deadline) {
            timed_out = true;
            return false;
        }
        if (left == 0) return true;

        int v = choose_vertex();
        if (v < 0) return true;

        uint64_t used = neigh_used[v];
        uint64_t avail = all_mask & ~used;
        if (avail == 0) return false;

        array<int, 64> candidates{};
        int cc = 0;
        for (int c = 0; c < k; ++c) {
            if (avail & (1ULL << c)) candidates[cc++] = c;
        }

        for (int i = 0; i < cc; ++i) {
            int c = candidates[i];
            color[v] = c;
            uncolored[v] = 0;

            vector<pair<int, uint64_t>> touched;
            touched.reserve(g.degree[v]);
            uint64_t bit = (1ULL << c);

            for (int it = g.offset[v]; it < g.offset[v + 1]; ++it) {
                int u = g.adj[it];
                if (!uncolored[u]) continue;
                uint64_t before = neigh_used[u];
                uint64_t after = before | bit;
                if (after != before) {
                    neigh_used[u] = after;
                    sat[u] += 1;
                    touched.emplace_back(u, before);
                }
            }

            if (dfs(left - 1)) return true;
            if (timed_out) return false;

            for (const auto& t : touched) {
                int u = t.first;
                uint64_t before = t.second;
                neigh_used[u] = before;
                sat[u] -= 1;
            }

            uncolored[v] = 1;
            color[v] = -1;
        }

        return false;
    }

    bool solve(vector<int>& out) {
        if (k <= 0) return false;
        if (k == 1) {
            if (g.m == 0) {
                out.assign(n, 0);
                return true;
            }
            return false;
        }

        bool ok = dfs(n);
        if (ok) out = color;
        return ok;
    }
};

static inline bool conflict_free(const Graph& g, const vector<int>& color) {
    for (const auto& e : g.edges) {
        if (color[e.first] == color[e.second]) return false;
    }
    return true;
}

vector<int> greedy_initial_coloring(const Graph& g, mt19937_64& rng, bool shuffle_ties) {
    vector<int> order(g.n);
    iota(order.begin(), order.end(), 0);

    if (shuffle_ties) {
        shuffle(order.begin(), order.end(), rng);
        stable_sort(order.begin(), order.end(), [&](int a, int b) {
            return g.degree[a] > g.degree[b];
        });
    } else {
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (g.degree[a] != g.degree[b]) return g.degree[a] > g.degree[b];
            return a < b;
        });
    }

    vector<int> color(g.n, -1);
    vector<int> stamp(g.k, -1);
    vector<int> cnt(g.k, 0);

    for (int idx = 0; idx < g.n; ++idx) {
        int v = order[idx];

        for (int it = g.offset[v]; it < g.offset[v + 1]; ++it) {
            int u = g.adj[it];
            int c = color[u];
            if (c >= 0) {
                stamp[c] = v;
                cnt[c]++;
            }
        }

        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (stamp[c] != v) {
                chosen = c;
                break;
            }
        }

        if (chosen < 0) {
            int best_conf = cnt[0];
            chosen = 0;
            for (int c = 1; c < g.k; ++c) {
                if (cnt[c] < best_conf) {
                    best_conf = cnt[c];
                    chosen = c;
                }
            }
        }

        color[v] = chosen;

        for (int it = g.offset[v]; it < g.offset[v + 1]; ++it) {
            int u = g.adj[it];
            int c = color[u];
            if (c >= 0) cnt[c]--;
        }
    }

    return color;
}

long long compute_conflicts(const Graph& g, const vector<int>& color, vector<int>& conf) {
    conf.assign(g.n, 0);
    long long total = 0;
    for (const auto& e : g.edges) {
        int u = e.first;
        int v = e.second;
        if (color[u] == color[v]) {
            conf[u]++;
            conf[v]++;
            total++;
        }
    }
    return total;
}

bool repair_min_conflicts(
    const Graph& g,
    vector<int>& color,
    mt19937_64& rng,
    chrono::steady_clock::time_point deadline
) {
    vector<int> conf;
    long long total_conf = compute_conflicts(g, color, conf);
    if (total_conf == 0) return true;

    vector<int> pos(g.n, -1);
    vector<int> conflicted;
    conflicted.reserve(g.n / 8 + 1);

    auto add_conflicted = [&](int v) {
        if (conf[v] > 0 && pos[v] == -1) {
            pos[v] = static_cast<int>(conflicted.size());
            conflicted.push_back(v);
        }
    };
    auto remove_conflicted = [&](int v) {
        int p = pos[v];
        if (p == -1) return;
        int back = conflicted.back();
        conflicted[p] = back;
        pos[back] = p;
        conflicted.pop_back();
        pos[v] = -1;
    };

    for (int v = 0; v < g.n; ++v) add_conflicted(v);

    vector<int> cnt(g.k, 0);
    int stall = 0;

    while (!conflicted.empty() && chrono::steady_clock::now() < deadline) {
        int pick = static_cast<int>(rng() % conflicted.size());
        int v = conflicted[pick];

        for (int it = g.offset[v]; it < g.offset[v + 1]; ++it) {
            int u = g.adj[it];
            cnt[color[u]]++;
        }

        int old_c = color[v];
        int old_conf_v = cnt[old_c];

        int best_c = old_c;
        int best_conf = old_conf_v;

        for (int c = 0; c < g.k; ++c) {
            int val = cnt[c];
            if (val < best_conf) {
                best_conf = val;
                best_c = c;
            }
        }

        bool random_move = false;
        if (best_c == old_c) {
            if ((rng() % 100ULL) < 20ULL) {
                best_c = static_cast<int>(rng() % g.k);
                best_conf = cnt[best_c];
                random_move = true;
            } else {
                stall++;
            }
        } else {
            stall = 0;
        }

        if (best_c != old_c) {
            color[v] = best_c;

            for (int it = g.offset[v]; it < g.offset[v + 1]; ++it) {
                int u = g.adj[it];
                if (color[u] == old_c) {
                    conf[u]--;
                    total_conf--;
                    if (conf[u] == 0) remove_conflicted(u);
                }
                if (color[u] == best_c) {
                    conf[u]++;
                    total_conf++;
                    if (conf[u] == 1) add_conflicted(u);
                }
            }

            conf[v] = best_conf;
            if (conf[v] == 0) remove_conflicted(v);
            else add_conflicted(v);
        }

        for (int it = g.offset[v]; it < g.offset[v + 1]; ++it) {
            int u = g.adj[it];
            cnt[color[u]]--;
        }

        if (total_conf == 0) return true;

        if (stall > 20000 || random_move) {
            stall = 0;
            int r = static_cast<int>(rng() % g.n);
            int rc = static_cast<int>(rng() % g.k);
            int prev = color[r];
            if (prev != rc) {
                color[r] = rc;
                // Do a cheap full recompute occasionally to keep state consistent.
                total_conf = compute_conflicts(g, color, conf);
                fill(pos.begin(), pos.end(), -1);
                conflicted.clear();
                for (int i = 0; i < g.n; ++i) add_conflicted(i);
            }
        }
    }

    return total_conf == 0;
}

vector<int> solve_graph_coloring(const Graph& g) {
    vector<int> color(g.n, 0);
    if (g.n == 0) return color;

    auto start = chrono::steady_clock::now();
    auto deadline = start + chrono::minutes(4) + chrono::seconds(45);

    // Adaptive exact strategy for smaller instances.
    bool try_exact = (g.n <= 220 && g.k <= 60 && g.m <= 20000);
    if (try_exact) {
        auto exact_deadline = start + chrono::seconds(20);
        if (exact_deadline > deadline) exact_deadline = deadline;
        DsaturSolver solver(g, exact_deadline);
        vector<int> exact;
        if (solver.solve(exact)) return exact;
    }

    mt19937_64 rng(now_seed());

    // Adaptive heuristic strategy for medium/large instances.
    vector<int> best = greedy_initial_coloring(g, rng, false);
    if (conflict_free(g, best)) return best;

    long long best_conf_count = LLONG_MAX;
    {
        vector<int> conf;
        best_conf_count = compute_conflicts(g, best, conf);
    }

    int restarts = (g.n < 200000) ? 10 : 4;
    for (int r = 0; r < restarts && chrono::steady_clock::now() < deadline; ++r) {
        vector<int> cur = greedy_initial_coloring(g, rng, true);

        if (repair_min_conflicts(g, cur, rng, deadline)) {
            return cur;
        }

        vector<int> conf;
        long long c = compute_conflicts(g, cur, conf);
        if (c < best_conf_count) {
            best_conf_count = c;
            best.swap(cur);
        }
    }

    // Final repair attempt on best-so-far.
    repair_min_conflicts(g, best, rng, deadline);

    // Last deterministic cleanup: choose least-conflicting color per vertex in degree order.
    vector<int> order(g.n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (g.degree[a] != g.degree[b]) return g.degree[a] > g.degree[b];
        return a < b;
    });

    vector<int> cnt(g.k, 0);
    for (int pass = 0; pass < 2; ++pass) {
        for (int v : order) {
            for (int it = g.offset[v]; it < g.offset[v + 1]; ++it) {
                cnt[best[g.adj[it]]]++;
            }
            int chosen = best[v];
            int best_val = cnt[chosen];
            for (int c = 0; c < g.k; ++c) {
                if (cnt[c] < best_val) {
                    best_val = cnt[c];
                    chosen = c;
                }
            }
            best[v] = chosen;
            for (int it = g.offset[v]; it < g.offset[v + 1]; ++it) {
                cnt[best[g.adj[it]]]--;
            }
        }
    }

    // Ensure valid color range and completeness.
    for (int v = 0; v < g.n; ++v) {
        if (best[v] < 0 || best[v] >= g.k) best[v] = ((best[v] % g.k) + g.k) % g.k;
    }

    return best;
}

int main() {
    Graph g = read_graph();

    if (g.k <= 0) {
        return 0;
    }

    vector<int> color = solve_graph_coloring(g);

    for (int i = 0; i < g.n; ++i) {
        if (i) cout << ' ';
        int c = color[i];
        if (c < 0 || c >= g.k) c = ((c % g.k) + g.k) % g.k;
        cout << c;
    }
    cout << '\n';

    return 0;
}
