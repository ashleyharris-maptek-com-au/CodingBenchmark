using System;
using System.IO;
using System.Collections.Generic;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int m = fs.NextInt();

        if (n <= 0 || m < 0)
        {
            Console.WriteLine("0");
            return;
        }

        // Duplicate each input edge once, so every vertex degree becomes even.
        // Then run Euler tour on this multigraph.
        int undirectedEdges = checked(m * 2);
        int directedArcs = checked(undirectedEdges * 2);

        int[] head = new int[n];
        Array.Fill(head, -1);

        int[] to = new int[directedArcs];
        int[] next = new int[directedArcs];
        int[] edgeId = new int[directedArcs];
        bool[] used = new bool[undirectedEdges];

        int arcPtr = 0;
        int eid = 0;

        void AddUndirected(int u, int v, int id)
        {
            to[arcPtr] = v;
            edgeId[arcPtr] = id;
            next[arcPtr] = head[u];
            head[u] = arcPtr++;

            to[arcPtr] = u;
            edgeId[arcPtr] = id;
            next[arcPtr] = head[v];
            head[v] = arcPtr++;
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            _ = fs.NextInt(); // weight is irrelevant for route feasibility

            // Original copy
            AddUndirected(u, v, eid++);
            // Duplicated copy
            AddUndirected(u, v, eid++);
        }

        // No edges: trivial tour at node 0.
        if (m == 0)
        {
            Console.WriteLine("0");
            return;
        }

        int start = 0;
        if (start < 0 || start >= n)
        {
            Console.WriteLine("0");
            return;
        }

        int[] it = new int[n];
        Array.Copy(head, it, n);

        var stack = new Stack<int>(Math.Max(4, undirectedEdges / 4));
        var circuit = new List<int>(undirectedEdges + 1);

        stack.Push(start);

        while (stack.Count > 0)
        {
            int v = stack.Peek();
            int e = it[v];

            while (e != -1 && used[edgeId[e]])
            {
                e = next[e];
            }

            it[v] = e;

            if (e == -1)
            {
                circuit.Add(v);
                stack.Pop();
            }
            else
            {
                int id = edgeId[e];
                used[id] = true;
                it[v] = next[e];
                stack.Push(to[e]);
            }
        }

        var output = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1 << 20);

        for (int i = circuit.Count - 1; i >= 0; i--)
        {
            if (i != circuit.Count - 1) output.Write(' ');
            output.Write(circuit[i]);
        }
        output.WriteLine();
        output.Flush();
    }
}
