using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Numerics;
using System.Text;

public static class Program
{
    private sealed class FastReader
    {
        private readonly TextReader _r;
        private string[] _buf = Array.Empty<string>();
        private int _idx;

        public FastReader(TextReader r) => _r = r;

        public string Next()
        {
            while (_idx >= _buf.Length)
            {
                string? line = _r.ReadLine();
                if (line == null) return string.Empty;
                _buf = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
                _idx = 0;
            }
            return _buf[_idx++];
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);
    }

    private readonly struct Candidate
    {
        public readonly Quaternion Q;
        public readonly Vector3 Min;
        public readonly Vector3 Ext;
        public readonly long Count;

        public Candidate(Quaternion q, Vector3 min, Vector3 ext, long count)
        {
            Q = q;
            Min = min;
            Ext = ext;
            Count = count;
        }
    }

    public static void Main()
    {
        var fr = new FastReader(Console.In);
        string first = fr.Next();
        if (string.IsNullOrEmpty(first))
        {
            Console.WriteLine("0");
            return;
        }

        int v = int.Parse(first, CultureInfo.InvariantCulture);
        int f = fr.NextInt();

        var vertices = new Vector3[v];
        for (int i = 0; i < v; i++)
        {
            float x = (float)fr.NextDouble();
            float y = (float)fr.NextDouble();
            float z = (float)fr.NextDouble();
            vertices[i] = new Vector3(x, y, z);
        }

        int maxFaceIndex = -1;
        var faces = new (int A, int B, int C)[f];
        for (int i = 0; i < f; i++)
        {
            int a = fr.NextInt();
            int b = fr.NextInt();
            int c = fr.NextInt();
            faces[i] = (a, b, c);
            if (a > maxFaceIndex) maxFaceIndex = a;
            if (b > maxFaceIndex) maxFaceIndex = b;
            if (c > maxFaceIndex) maxFaceIndex = c;
        }

        float W = (float)fr.NextDouble();
        float H = (float)fr.NextDouble();
        float D = (float)fr.NextDouble();

        if (v <= 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        if (f > 0 && maxFaceIndex >= v)
        {
            for (int i = 0; i < f; i++)
            {
                faces[i] = (faces[i].A - 1, faces[i].B - 1, faces[i].C - 1);
            }
        }

        int maxDim = (int)Math.Ceiling(Math.Max(W, Math.Max(H, D)));
        long voxels = (long)Math.Max(1, Math.Floor((double)W * H * D));
        int complexity = v + f;

        int rotationBudget;
        long emitCap;
        if (maxDim <= 8 && complexity <= 128)
        {
            rotationBudget = 120;
            emitCap = 20_000_000;
        }
        else if (maxDim <= 25 && complexity <= 2_000)
        {
            rotationBudget = 64;
            emitCap = 5_000_000;
        }
        else if (maxDim <= 120 && complexity <= 20_000)
        {
            rotationBudget = 28;
            emitCap = 2_000_000;
        }
        else
        {
            rotationBudget = 10;
            emitCap = 500_000;
        }

        if (voxels > 30_000_000) emitCap = Math.Min(emitCap, 500_000);

        List<Quaternion> rotations = BuildRotationSet(rotationBudget);

        Candidate best = new Candidate(Quaternion.Identity, Vector3.Zero, Vector3.Zero, 0);
        foreach (Quaternion q in rotations)
        {
            if (!TryEvaluate(vertices, q, W, H, D, out Candidate c))
                continue;

            if (c.Count > best.Count)
                best = c;
        }

        if (best.Count <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        long outCount = Math.Min(best.Count, emitCap);
        Console.WriteLine(outCount.ToString(CultureInfo.InvariantCulture));

        var inv = CultureInfo.InvariantCulture;
        float tx0 = -best.Min.X;
        float ty0 = -best.Min.Y;
        float tz0 = -best.Min.Z;

        long nx = (long)Math.Floor(W / best.Ext.X + 1e-7f);
        long ny = (long)Math.Floor(H / best.Ext.Y + 1e-7f);
        long nz = (long)Math.Floor(D / best.Ext.Z + 1e-7f);

        long emitted = 0;
        var sb = new StringBuilder(1 << 20);

        for (long iz = 0; iz < nz && emitted < outCount; iz++)
        {
            float z = tz0 + iz * best.Ext.Z;
            for (long iy = 0; iy < ny && emitted < outCount; iy++)
            {
                float y = ty0 + iy * best.Ext.Y;
                for (long ix = 0; ix < nx && emitted < outCount; ix++)
                {
                    float x = tx0 + ix * best.Ext.X;
                    sb.Append(x.ToString("G9", inv)).Append(' ')
                      .Append(y.ToString("G9", inv)).Append(' ')
                      .Append(z.ToString("G9", inv)).Append(' ')
                      .Append(best.Q.W.ToString("G9", inv)).Append(' ')
                      .Append(best.Q.X.ToString("G9", inv)).Append(' ')
                      .Append(best.Q.Y.ToString("G9", inv)).Append(' ')
                      .Append(best.Q.Z.ToString("G9", inv)).Append('\n');

                    emitted++;
                    if (sb.Length >= (1 << 20))
                    {
                        Console.Write(sb.ToString());
                        sb.Clear();
                    }
                }
            }
        }

        if (sb.Length > 0)
            Console.Write(sb.ToString());
    }

    private static bool TryEvaluate(Vector3[] vertices, Quaternion q, float W, float H, float D, out Candidate c)
    {
        Vector3 min = new Vector3(float.PositiveInfinity, float.PositiveInfinity, float.PositiveInfinity);
        Vector3 max = new Vector3(float.NegativeInfinity, float.NegativeInfinity, float.NegativeInfinity);

        for (int i = 0; i < vertices.Length; i++)
        {
            Vector3 p = Vector3.Transform(vertices[i], q);
            min = Vector3.Min(min, p);
            max = Vector3.Max(max, p);
        }

        Vector3 ext = max - min;
        const float eps = 1e-6f;
        if (ext.X <= eps || ext.Y <= eps || ext.Z <= eps)
        {
            c = default;
            return false;
        }

        long nx = (long)Math.Floor(W / ext.X + 1e-7f);
        long ny = (long)Math.Floor(H / ext.Y + 1e-7f);
        long nz = (long)Math.Floor(D / ext.Z + 1e-7f);

        if (nx <= 0 || ny <= 0 || nz <= 0)
        {
            c = default;
            return false;
        }

        long cnt;
        try
        {
            checked { cnt = nx * ny * nz; }
        }
        catch
        {
            cnt = long.MaxValue;
        }

        c = new Candidate(q, min, ext, cnt);
        return true;
    }

    private static List<Quaternion> BuildRotationSet(int budget)
    {
        var list = new List<Quaternion>(Math.Max(24, budget));
        var seen = new HashSet<string>(StringComparer.Ordinal);

        void AddQ(Quaternion q)
        {
            q = Quaternion.Normalize(q);
            if (q.W < 0f) q = new Quaternion(-q.X, -q.Y, -q.Z, -q.W);
            string key =
                MathF.Round(q.W, 5).ToString(CultureInfo.InvariantCulture) + "|" +
                MathF.Round(q.X, 5).ToString(CultureInfo.InvariantCulture) + "|" +
                MathF.Round(q.Y, 5).ToString(CultureInfo.InvariantCulture) + "|" +
                MathF.Round(q.Z, 5).ToString(CultureInfo.InvariantCulture);
            if (seen.Add(key)) list.Add(q);
        }

        AddQ(Quaternion.Identity);

        var qx90 = Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI / 2f);
        var qy90 = Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI / 2f);
        var qz90 = Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI / 2f);

        var queue = new Queue<Quaternion>();
        queue.Enqueue(Quaternion.Identity);

        while (queue.Count > 0 && list.Count < 24)
        {
            Quaternion cur = queue.Dequeue();
            Quaternion[] nxt =
            {
                Quaternion.Normalize(Quaternion.Concatenate(cur, qx90)),
                Quaternion.Normalize(Quaternion.Concatenate(cur, qy90)),
                Quaternion.Normalize(Quaternion.Concatenate(cur, qz90))
            };

            for (int i = 0; i < nxt.Length; i++)
            {
                int before = list.Count;
                AddQ(nxt[i]);
                if (list.Count > before && list.Count < 24)
                    queue.Enqueue(nxt[i]);
            }
        }

        if (budget <= list.Count)
            return list.GetRange(0, budget);

        int extra = budget - list.Count;
        for (int i = 0; i < extra * 3 && list.Count < budget; i++)
        {
            float a = RadicalInverse(i + 1, 2) * MathF.PI * 2f;
            float b = RadicalInverse(i + 1, 3) * MathF.PI;
            float c = RadicalInverse(i + 1, 5) * MathF.PI * 2f;
            Quaternion q = Quaternion.CreateFromYawPitchRoll(a, b, c);
            AddQ(q);
        }

        return list;
    }

    private static float RadicalInverse(int n, int b)
    {
        float invBase = 1f / b;
        float invBi = invBase;
        float x = 0f;
        while (n > 0)
        {
            int d = n % b;
            x += d * invBi;
            n /= b;
            invBi *= invBase;
        }
        return x;
    }
}
