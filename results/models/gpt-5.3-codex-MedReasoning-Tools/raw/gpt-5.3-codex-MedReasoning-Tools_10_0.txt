#include <algorithm>
#include <array>
#include <cstdint>
#include <iostream>
#include <limits>
#include <set>
#include <tuple>
#include <utility>
#include <vector>

struct Rect {
    int w = 0;
    int h = 0;
    int id = -1;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct FreeRect {
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
};

struct Candidate {
    bool ok = false;
    int free_idx = -1;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int rotated = 0;
    int short_fit = 0;
    int long_fit = 0;
    int area_fit = 0;
};

struct Board {
    std::vector<FreeRect> free_rects;
    long long free_area = 0;
};

static inline bool better_candidate(const Candidate& a, const Candidate& b) {
    if (!a.ok) return false;
    if (!b.ok) return true;
    if (a.short_fit != b.short_fit) return a.short_fit < b.short_fit;
    if (a.long_fit != b.long_fit) return a.long_fit < b.long_fit;
    if (a.area_fit != b.area_fit) return a.area_fit < b.area_fit;
    if (a.y != b.y) return a.y < b.y;
    if (a.x != b.x) return a.x < b.x;
    return a.rotated < b.rotated;
}

static Candidate find_best_in_board(const Board& b, int rw, int rh) {
    Candidate best;
    for (int i = 0; i < static_cast<int>(b.free_rects.size()); ++i) {
        const FreeRect& fr = b.free_rects[i];
        if (rw <= fr.w && rh <= fr.h) {
            const int dw = fr.w - rw;
            const int dh = fr.h - rh;
            Candidate cur;
            cur.ok = true;
            cur.free_idx = i;
            cur.x = fr.x;
            cur.y = fr.y;
            cur.w = rw;
            cur.h = rh;
            cur.rotated = 0;
            cur.short_fit = std::min(dw, dh);
            cur.long_fit = std::max(dw, dh);
            cur.area_fit = fr.w * fr.h - rw * rh;
            if (better_candidate(cur, best)) best = cur;
        }
    }
    return best;
}

static void prune_free_rects(std::vector<FreeRect>& free_rects) {
    for (size_t i = 0; i < free_rects.size(); ++i) {
        if (free_rects[i].w <= 0 || free_rects[i].h <= 0) {
            free_rects.erase(free_rects.begin() + static_cast<std::ptrdiff_t>(i));
            --i;
            continue;
        }
    }

    for (size_t i = 0; i < free_rects.size(); ++i) {
        bool removed_i = false;
        for (size_t j = i + 1; j < free_rects.size(); ++j) {
            const FreeRect& a = free_rects[i];
            const FreeRect& b = free_rects[j];
            const bool a_in_b = (a.x >= b.x && a.y >= b.y && a.x + a.w <= b.x + b.w && a.y + a.h <= b.y + b.h);
            const bool b_in_a = (b.x >= a.x && b.y >= a.y && b.x + b.w <= a.x + a.w && b.y + b.h <= a.y + a.h);

            if (a_in_b) {
                free_rects.erase(free_rects.begin() + static_cast<std::ptrdiff_t>(i));
                --i;
                removed_i = true;
                break;
            }
            if (b_in_a) {
                free_rects.erase(free_rects.begin() + static_cast<std::ptrdiff_t>(j));
                --j;
            }
        }
        if (removed_i) continue;
    }
}

static void commit_to_board(Board& b, const Candidate& c) {
    const FreeRect fr = b.free_rects[c.free_idx];
    b.free_rects[c.free_idx] = b.free_rects.back();
    b.free_rects.pop_back();

    const int used_w = c.w;
    const int used_h = c.h;
    const int rem_w = fr.w - used_w;
    const int rem_h = fr.h - used_h;

    const bool split_vertical = (rem_w > rem_h);

    if (split_vertical) {
        if (rem_w > 0) {
            b.free_rects.push_back({fr.x + used_w, fr.y, rem_w, fr.h});
        }
        if (rem_h > 0) {
            b.free_rects.push_back({fr.x, fr.y + used_h, used_w, rem_h});
        }
    } else {
        if (rem_h > 0) {
            b.free_rects.push_back({fr.x, fr.y + used_h, fr.w, rem_h});
        }
        if (rem_w > 0) {
            b.free_rects.push_back({fr.x + used_w, fr.y, rem_w, used_h});
        }
    }

    // Keep free list compact for speed and quality.
    if (b.free_rects.size() > 80) {
        prune_free_rects(b.free_rects);
    }

    b.free_area -= static_cast<long long>(used_w) * static_cast<long long>(used_h);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N = 0;
    int BW = 0;
    int BH = 0;
    if (!(std::cin >> N >> BW >> BH)) {
        return 0;
    }

    std::vector<Rect> rects;
    rects.reserve(N);
    long long total_area = 0;

    for (int i = 0; i < N; ++i) {
        int w = 0;
        int h = 0;
        std::cin >> w >> h;
        rects.push_back({w, h, i});
        total_area += static_cast<long long>(w) * static_cast<long long>(h);
    }

    for (const Rect& r : rects) {
        const bool fit_normal = (r.w <= BW && r.h <= BH);
        const bool fit_rot = (r.h <= BW && r.w <= BH);
        if (!fit_normal && !fit_rot) {
            // Infeasible instance. Emit minimal syntactically-valid output.
            std::cout << 0 << '\n';
            for (int i = 0; i < N; ++i) {
                std::cout << "0 0 0 0\n";
            }
            return 0;
        }
    }

    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        const int amax = std::max(a.w, a.h);
        const int bmax = std::max(b.w, b.h);
        if (amax != bmax) return amax > bmax;
        const int amin = std::min(a.w, a.h);
        const int bmin = std::min(b.w, b.h);
        if (amin != bmin) return amin > bmin;
        const long long aa = static_cast<long long>(a.w) * a.h;
        const long long ba = static_cast<long long>(b.w) * b.h;
        if (aa != ba) return aa > ba;
        return a.id < b.id;
    });

    std::vector<Placement> out(N);
    std::vector<Board> boards;
    boards.reserve(std::max(4, N / 8));

    // Ranking by free area descending to pick promising boards quickly.
    std::set<std::pair<long long, int>, std::greater<std::pair<long long, int>>> area_rank;

    const long long board_area = static_cast<long long>(BW) * static_cast<long long>(BH);
    const int candidate_limit = (N <= 2000 ? 96 : (N <= 10000 ? 40 : 24));

    auto create_board = [&]() -> int {
        Board b;
        b.free_rects.push_back({0, 0, BW, BH});
        b.free_area = board_area;
        const int idx = static_cast<int>(boards.size());
        boards.push_back(std::move(b));
        area_rank.insert({boards[idx].free_area, idx});
        return idx;
    };

    // Start with one board.
    create_board();

    for (const Rect& r : rects) {
        std::vector<int> candidates;
        candidates.reserve(candidate_limit + 2);

        int taken = 0;
        for (auto it = area_rank.begin(); it != area_rank.end() && taken < candidate_limit; ++it) {
            if (it->first <= 0) break;
            candidates.push_back(it->second);
            ++taken;
        }

        // Always include newest board for locality.
        if (!boards.empty()) {
            const int last = static_cast<int>(boards.size()) - 1;
            bool exists = false;
            for (int id : candidates) {
                if (id == last) {
                    exists = true;
                    break;
                }
            }
            if (!exists) candidates.push_back(last);
        }

        Candidate best_overall;
        int best_board = -1;

        for (int bid : candidates) {
            const Board& b = boards[bid];
            if (b.free_area < static_cast<long long>(r.w) * r.h) continue;

            Candidate c1 = find_best_in_board(b, r.w, r.h);
            Candidate c2 = find_best_in_board(b, r.h, r.w);
            if (c2.ok) c2.rotated = 1;

            Candidate local_best;
            if (better_candidate(c1, local_best)) local_best = c1;
            if (better_candidate(c2, local_best)) local_best = c2;

            if (local_best.ok) {
                if (best_board == -1 || better_candidate(local_best, best_overall)) {
                    best_overall = local_best;
                    best_board = bid;
                }
            }
        }

        if (best_board == -1) {
            best_board = create_board();
            Candidate c1 = find_best_in_board(boards[best_board], r.w, r.h);
            Candidate c2 = find_best_in_board(boards[best_board], r.h, r.w);
            if (c2.ok) c2.rotated = 1;
            if (better_candidate(c1, best_overall)) best_overall = c1;
            if (better_candidate(c2, best_overall)) best_overall = c2;

            if (!best_overall.ok) {
                // Should not happen due to pre-check, but keep output valid.
                std::cout << 0 << '\n';
                for (int i = 0; i < N; ++i) {
                    std::cout << "0 0 0 0\n";
                }
                return 0;
            }
        }

        area_rank.erase({boards[best_board].free_area, best_board});
        commit_to_board(boards[best_board], best_overall);
        area_rank.insert({boards[best_board].free_area, best_board});

        out[r.id] = {best_board, best_overall.x, best_overall.y, best_overall.rotated};
    }

    int used_boards = static_cast<int>(boards.size());
    while (used_boards > 0 && boards[used_boards - 1].free_area == board_area) {
        --used_boards;
    }

    std::cout << used_boards << '\n';
    for (int i = 0; i < N; ++i) {
        const Placement& p = out[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
