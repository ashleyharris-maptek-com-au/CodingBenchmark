use std::cmp::Ordering;
use std::collections::{HashMap, VecDeque};
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Default)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    #[inline]
    fn add(self, o: Pt) -> Pt {
        Pt { x: self.x + o.x, y: self.y + o.y }
    }
    #[inline]
    fn sub(self, o: Pt) -> Pt {
        Pt { x: self.x - o.x, y: self.y - o.y }
    }
    #[inline]
    fn mul(self, k: f64) -> Pt {
        Pt { x: self.x * k, y: self.y * k }
    }
    #[inline]
    fn norm2(self) -> f64 {
        self.x * self.x + self.y * self.y
    }
    #[inline]
    fn norm(self) -> f64 {
        self.norm2().sqrt()
    }
}

fn splitmix64(mut x: u64) -> u64 {
    x = x.wrapping_add(0x9E3779B97F4A7C15);
    let mut z = x;
    z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
    z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
    z ^ (z >> 31)
}

fn rand01(state: u64) -> f64 {
    let v = splitmix64(state);
    ((v >> 11) as f64) * (1.0 / ((1u64 << 53) as f64))
}

fn bfs_levels(nodes: &[usize], adj: &[Vec<usize>], deg: &[usize]) -> (Vec<usize>, Vec<Vec<usize>>) {
    if nodes.is_empty() {
        return (Vec::new(), Vec::new());
    }
    let &start = nodes
        .iter()
        .max_by_key(|&&u| deg[u])
        .unwrap_or(&nodes[0]);

    let mut idx = HashMap::with_capacity(nodes.len() * 2);
    for (i, &u) in nodes.iter().enumerate() {
        idx.insert(u, i);
    }

    let mut dist = vec![usize::MAX; nodes.len()];
    let mut q = VecDeque::new();
    dist[*idx.get(&start).unwrap()] = 0;
    q.push_back(start);

    while let Some(u) = q.pop_front() {
        let du = dist[*idx.get(&u).unwrap()];
        for &v in &adj[u] {
            if let Some(&iv) = idx.get(&v) {
                if dist[iv] == usize::MAX {
                    dist[iv] = du + 1;
                    q.push_back(v);
                }
            }
        }
    }

    let maxd = dist
        .iter()
        .filter(|&&d| d != usize::MAX)
        .copied()
        .max()
        .unwrap_or(0);

    let mut levels = vec![Vec::<usize>::new(); maxd + 1];
    for (i, &u) in nodes.iter().enumerate() {
        let d = if dist[i] == usize::MAX { maxd } else { dist[i] };
        levels[d].push(u);
    }

    (dist, levels)
}

fn barycentric_order(levels: &mut [Vec<usize>], adj: &[Vec<usize>], rounds: usize) {
    if levels.is_empty() {
        return;
    }
    let mut pos = HashMap::<usize, f64>::new();
    for lev in levels.iter() {
        for (i, &u) in lev.iter().enumerate() {
            pos.insert(u, i as f64);
        }
    }

    for _ in 0..rounds {
        // Downward sweep
        for l in 1..levels.len() {
            let mut scored: Vec<(usize, f64)> = levels[l]
                .iter()
                .map(|&u| {
                    let mut s = 0.0;
                    let mut c = 0usize;
                    for &v in &adj[u] {
                        if let Some(&p) = pos.get(&v) {
                            s += p;
                            c += 1;
                        }
                    }
                    let key = if c == 0 { pos.get(&u).copied().unwrap_or(0.0) } else { s / c as f64 };
                    (u, key)
                })
                .collect();
            scored.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(Ordering::Equal));
            levels[l].clear();
            levels[l].extend(scored.iter().map(|x| x.0));
            for (i, &u) in levels[l].iter().enumerate() {
                pos.insert(u, i as f64);
            }
        }

        // Upward sweep
        if levels.len() > 1 {
            for l in (0..levels.len() - 1).rev() {
                let mut scored: Vec<(usize, f64)> = levels[l]
                    .iter()
                    .map(|&u| {
                        let mut s = 0.0;
                        let mut c = 0usize;
                        for &v in &adj[u] {
                            if let Some(&p) = pos.get(&v) {
                                s += p;
                                c += 1;
                            }
                        }
                        let key = if c == 0 { pos.get(&u).copied().unwrap_or(0.0) } else { s / c as f64 };
                        (u, key)
                    })
                    .collect();
                scored.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(Ordering::Equal));
                levels[l].clear();
                levels[l].extend(scored.iter().map(|x| x.0));
                for (i, &u) in levels[l].iter().enumerate() {
                    pos.insert(u, i as f64);
                }
            }
        }
    }
}

fn layered_radial_layout(
    nodes: &[usize],
    levels: &[Vec<usize>],
    out: &mut [Pt],
    scale: f64,
    jitter_seed: u64,
) {
    if nodes.is_empty() {
        return;
    }
    if nodes.len() == 1 {
        out[nodes[0]] = Pt { x: 0.0, y: 0.0 };
        return;
    }

    let lcnt = levels.len().max(1);
    let ring_step = scale / (lcnt as f64).sqrt().max(1.0);

    for (li, lev) in levels.iter().enumerate() {
        let r = ring_step * (li as f64 + 1.0);
        let k = lev.len();
        if k == 1 {
            let u = lev[0];
            out[u] = Pt { x: r, y: 0.0 };
            continue;
        }
        for (i, &u) in lev.iter().enumerate() {
            let base = 2.0 * PI * (i as f64) / (k as f64);
            let j = (rand01(jitter_seed ^ ((u as u64) << 1)) - 0.5) * (PI / (k as f64).max(6.0));
            let a = base + j;
            out[u] = Pt {
                x: r * a.cos(),
                y: r * a.sin(),
            };
        }
    }
}

fn force_refine(
    comp_nodes: &[usize],
    edges: &[(usize, usize)],
    pos: &mut [Pt],
    n_total: usize,
) {
    let n = comp_nodes.len();
    if n <= 1 {
        return;
    }

    let area = (n as f64).max(4.0) * 400.0;
    let k = area.sqrt() / (n as f64).sqrt();

    let (iters, use_full_repulsion) = if n <= 200 {
        (140, true)
    } else if n <= 800 {
        (100, true)
    } else if n <= 3000 {
        (60, false)
    } else {
        (0, false)
    };

    if iters == 0 {
        return;
    }

    let comp_index: HashMap<usize, usize> = comp_nodes.iter().enumerate().map(|(i, &u)| (u, i)).collect();

    let mut disp = vec![Pt::default(); n];
    let mut t = (area.sqrt()) * 0.15;

    for it in 0..iters {
        disp.fill(Pt::default());

        if use_full_repulsion {
            for i in 0..n {
                let u = comp_nodes[i];
                for j in (i + 1)..n {
                    let v = comp_nodes[j];
                    let mut d = pos[u].sub(pos[v]);
                    let mut dist2 = d.norm2();
                    if dist2 < 1e-9 {
                        let r = rand01(((u as u64) << 32) ^ (v as u64) ^ (it as u64));
                        let a = 2.0 * PI * r;
                        d = Pt { x: a.cos(), y: a.sin() };
                        dist2 = 1e-6;
                    }
                    let dist = dist2.sqrt();
                    let f = (k * k) / dist.max(1e-6);
                    let vec = d.mul(f / dist.max(1e-6));
                    disp[i] = disp[i].add(vec);
                    disp[j] = disp[j].sub(vec);
                }
            }
        } else {
            let samples = 24usize;
            for i in 0..n {
                let u = comp_nodes[i];
                for s in 0..samples {
                    let seed = ((u as u64) << 32) ^ ((it as u64) << 16) ^ (s as u64) ^ (n_total as u64);
                    let j = (splitmix64(seed) as usize) % n;
                    if i == j {
                        continue;
                    }
                    let v = comp_nodes[j];
                    let d = pos[u].sub(pos[v]);
                    let dist2 = d.norm2().max(1e-6);
                    let dist = dist2.sqrt();
                    let f = (k * k) / dist;
                    disp[i] = disp[i].add(d.mul(f / dist));
                }
            }
        }

        for &(u, v) in edges {
            if u == v {
                continue;
            }
            let (Some(&iu), Some(&iv)) = (comp_index.get(&u), comp_index.get(&v)) else {
                continue;
            };
            let d = pos[u].sub(pos[v]);
            let dist = d.norm().max(1e-6);
            let f = (dist * dist) / k.max(1e-6);
            let vec = d.mul(f / dist);
            disp[iu] = disp[iu].sub(vec);
            disp[iv] = disp[iv].add(vec);
        }

        let gravity = 0.01;
        for (i, &u) in comp_nodes.iter().enumerate() {
            disp[i] = disp[i].sub(pos[u].mul(gravity));
        }

        for (i, &u) in comp_nodes.iter().enumerate() {
            let d = disp[i];
            let dn = d.norm().max(1e-9);
            let step = dn.min(t);
            pos[u] = pos[u].add(d.mul(step / dn));
        }

        t *= 0.95;
    }
}

fn component_bbox(nodes: &[usize], pos: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &u in nodes {
        let p = pos[u];
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (minx, miny, maxx, maxy)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap_or(0),
        None => return,
    };
    let m: usize = it.next().and_then(|x| x.parse().ok()).unwrap_or(0);

    let mut adj = vec![Vec::<usize>::new(); n];
    let mut edges = Vec::<(usize, usize)>::with_capacity(m);

    for _ in 0..m {
        let (Some(a), Some(b)) = (it.next(), it.next()) else { break };
        let mut u: usize = a.parse().unwrap_or(0);
        let mut v: usize = b.parse().unwrap_or(0);
        if n == 0 {
            continue;
        }
        if u >= n {
            u %= n;
        }
        if v >= n {
            v %= n;
        }

        // Keep all edges for spring attraction; avoid self-loop in adjacency traversal.
        edges.push((u, v));
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    if n == 0 {
        return;
    }

    let deg: Vec<usize> = adj.iter().map(|v| v.len()).collect();

    // Connected components
    let mut comp_id = vec![usize::MAX; n];
    let mut components: Vec<Vec<usize>> = Vec::new();
    let mut cid = 0usize;
    for s in 0..n {
        if comp_id[s] != usize::MAX {
            continue;
        }
        let mut q = VecDeque::new();
        let mut comp = Vec::new();
        comp_id[s] = cid;
        q.push_back(s);
        while let Some(u) = q.pop_front() {
            comp.push(u);
            for &v in &adj[u] {
                if comp_id[v] == usize::MAX {
                    comp_id[v] = cid;
                    q.push_back(v);
                }
            }
        }
        // Isolated node (no adjacency) still forms component.
        if comp.is_empty() {
            comp.push(s);
        }
        components.push(comp);
        cid += 1;
    }

    let mut pos = vec![Pt::default(); n];

    for (i, comp) in components.iter().enumerate() {
        let size = comp.len();

        if size == 1 {
            pos[comp[0]] = Pt { x: 0.0, y: 0.0 };
            continue;
        }

        let (_, mut levels) = bfs_levels(comp, &adj, &deg);
        let rounds = if size <= 1000 { 6 } else { 3 };
        barycentric_order(&mut levels, &adj, rounds);

        let base_scale = (size as f64).sqrt() * 12.0 + 20.0;
        layered_radial_layout(comp, &levels, &mut pos, base_scale, (i as u64) * 0xA5A5_1234);

        let comp_edges: Vec<(usize, usize)> = edges
            .iter()
            .copied()
            .filter(|&(u, v)| comp_id[u] == comp_id[v])
            .collect();

        force_refine(comp, &comp_edges, &mut pos, n);
    }

    // Pack components on a coarse grid to avoid overlaps.
    let mut order: Vec<usize> = (0..components.len()).collect();
    order.sort_by_key(|&i| std::cmp::Reverse(components[i].len()));

    let mut packed = vec![Pt::default(); components.len()];
    let mut cursor = Pt { x: 0.0, y: 0.0 };
    let mut row_h = 0.0;
    let target_w = (n as f64).sqrt() * 40.0 + 200.0;

    for &ci in &order {
        let comp = &components[ci];
        let (minx, miny, maxx, maxy) = component_bbox(comp, &pos);
        let w = (maxx - minx).max(1.0) + 40.0;
        let h = (maxy - miny).max(1.0) + 40.0;

        if cursor.x + w > target_w && cursor.x > 0.0 {
            cursor.x = 0.0;
            cursor.y += row_h;
            row_h = 0.0;
        }

        packed[ci] = Pt {
            x: cursor.x - minx + 20.0,
            y: cursor.y - miny + 20.0,
        };

        cursor.x += w;
        row_h = row_h.max(h);
    }

    for (ci, comp) in components.iter().enumerate() {
        let off = packed[ci];
        for &u in comp {
            pos[u] = pos[u].add(off);
        }
    }

    // Normalize to a compact positive coordinate space.
    let (mut minx, mut miny) = (f64::INFINITY, f64::INFINITY);
    for p in &pos {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
    }
    for p in &mut pos {
        p.x -= minx;
        p.y -= miny;
    }

    let mut out = String::with_capacity(n * 24);
    for p in &pos {
        out.push_str(&format!("{:.6} {:.6}\n", p.x, p.y));
    }
    print!("{}", out);
}
