struct Edge {
    int u;
    int v;
};

class FastInput {
public:
    static constexpr size_t BUFSIZE = 1 << 20;
    FastInput() : idx_(0), size_(0) {}

    bool readInt(int& out) {
        char c;
        do {
            if (!nextChar(c)) return false;
        } while (c <= ' ');

        int sign = 1;
        if (c == '-') {
            sign = -1;
            if (!nextChar(c)) return false;
        }

        int val = 0;
        while (c >= '0' && c <= '9') {
            val = val * 10 + (c - '0');
            if (!nextChar(c)) {
                out = sign * val;
                return true;
            }
        }
        out = sign * val;
        return true;
    }

private:
    bool nextChar(char& c) {
        if (idx_ >= size_) {
            std::cin.read(buf_, BUFSIZE);
            size_ = static_cast<size_t>(std::cin.gcount());
            idx_ = 0;
            if (size_ == 0) return false;
        }
        c = buf_[idx_++];
        return true;
    }

    char buf_[BUFSIZE];
    size_t idx_;
    size_t size_;
};

static void printCut(const std::vector<Edge>& cut) {
    std::cout << cut.size() << '\n';
    for (const auto& e : cut) {
        std::cout << e.u << ' ' << e.v << '\n';
    }
}

static std::vector<Edge> cutFromSide(const std::vector<Edge>& edges, const std::vector<char>& side) {
    std::vector<Edge> cut;
    cut.reserve(edges.size() / 8 + 1);
    for (const auto& e : edges) {
        if (side[e.u] != side[e.v]) cut.push_back(e);
    }
    return cut;
}

static std::vector<Edge> stoerWagnerMinCut(int n, const std::vector<Edge>& inputEdges) {
    std::vector<std::unordered_map<int, int>> g(n);
    for (const auto& e : inputEdges) {
        if (e.u == e.v) continue;
        ++g[e.u][e.v];
        ++g[e.v][e.u];
    }

    std::vector<int> verts(n);
    std::iota(verts.begin(), verts.end(), 0);

    std::vector<std::vector<int>> groups(n);
    for (int i = 0; i < n; ++i) groups[i] = {i};

    int best = std::numeric_limits<int>::max();
    std::vector<char> bestSide(n, 0);

    while (verts.size() > 1) {
        const int k = static_cast<int>(verts.size());
        std::vector<int> w(n, 0);
        std::vector<char> added(n, 0);
        int prev = -1;
        int sel = -1;

        for (int i = 0; i < k; ++i) {
            sel = -1;
            int selW = -1;
            for (int v : verts) {
                if (!added[v] && w[v] > selW) {
                    selW = w[v];
                    sel = v;
                }
            }

            if (sel == -1) break;
            added[sel] = 1;

            if (i == k - 1) {
                if (selW < best) {
                    best = selW;
                    std::fill(bestSide.begin(), bestSide.end(), 0);
                    for (int x : groups[sel]) bestSide[x] = 1;
                }

                if (prev != -1) {
                    for (const auto& kv : g[sel]) {
                        int to = kv.first;
                        int wt = kv.second;
                        if (to == prev) continue;
                        g[prev][to] += wt;
                        g[to][prev] += wt;
                        g[to].erase(sel);
                    }
                    g[sel].clear();

                    auto& gp = groups[prev];
                    auto& gs = groups[sel];
                    gp.insert(gp.end(), gs.begin(), gs.end());
                    gs.clear();

                    verts.erase(std::remove(verts.begin(), verts.end(), sel), verts.end());
                }
            } else {
                prev = sel;
                for (const auto& kv : g[sel]) {
                    if (!added[kv.first]) w[kv.first] += kv.second;
                }
            }
        }
    }

    return cutFromSide(inputEdges, bestSide);
}

static std::vector<Edge> cutByMinDegreeVertex(const std::string& tmpFile, int n, long long m, const std::vector<int>& degree) {
    int bestV = 0;
    for (int i = 1; i < n; ++i) {
        if (degree[i] < degree[bestV]) bestV = i;
    }

    std::ifstream in(tmpFile, std::ios::binary);
    std::vector<Edge> cut;
    cut.reserve(static_cast<size_t>(degree[bestV]));

    Edge e;
    for (long long i = 0; i < m; ++i) {
        in.read(reinterpret_cast<char*>(&e), sizeof(Edge));
        if (!in) break;
        if (e.u == bestV || e.v == bestV) cut.push_back(e);
    }
    return cut;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    FastInput in;
    int n = 0;
    int mInt = 0;
    if (!in.readInt(n)) return 0;
    if (!in.readInt(mInt)) return 0;

    if (n <= 1 || mInt <= 0) {
        std::cout << 0 << '\n';
        return 0;
    }

    const long long m = static_cast<long long>(mInt);
    const long long edgeBytes = m * static_cast<long long>(sizeof(Edge));
    const long long approxExtra = static_cast<long long>(n) * 64LL;

    // Conservative in-memory budget: ~1.2GB for edges + data structures.
    constexpr long long MEM_BUDGET = 1200LL * 1024LL * 1024LL;
    const bool useInMemory = (edgeBytes + approxExtra <= MEM_BUDGET && n <= 35000);

    if (useInMemory) {
        std::vector<Edge> edges;
        edges.reserve(static_cast<size_t>(m));
        for (long long i = 0; i < m; ++i) {
            int u = 0, v = 0;
            in.readInt(u);
            in.readInt(v);
            if (u < 0 || v < 0 || u >= n || v >= n) continue;
            edges.push_back({u, v});
        }

        auto cut = stoerWagnerMinCut(n, edges);
        printCut(cut);
        return 0;
    }

    // Disk-backed fallback for very large graphs.
    // Produces a valid cut (all edges incident to a minimum-degree vertex).
    const std::string tmpFile = "__mincut_edges.tmp.bin";
    std::ofstream out(tmpFile, std::ios::binary | std::ios::trunc);
    std::vector<int> degree(n, 0);

    for (long long i = 0; i < m; ++i) {
        int u = 0, v = 0;
        in.readInt(u);
        in.readInt(v);
        if (u < 0 || v < 0 || u >= n || v >= n) continue;
        Edge e{u, v};
        out.write(reinterpret_cast<const char*>(&e), sizeof(Edge));
        if (u != v) {
            ++degree[u];
            ++degree[v];
        }
    }
    out.close();

    auto cut = cutByMinDegreeVertex(tmpFile, n, m, degree);
    printCut(cut);

    std::remove(tmpFile.c_str());
    return 0;
}
