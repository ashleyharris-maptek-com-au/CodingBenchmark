use std::collections::{HashSet, VecDeque};
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash)]
struct Pt {
    x: i32,
    y: i32,
}

fn neighbors4(x: i32, y: i32, w: i32, h: i32) -> impl Iterator<Item = (i32, i32)> {
    const D: [(i32, i32); 4] = [(1, 0), (-1, 0), (0, 1), (0, -1)];
    D.into_iter().filter_map(move |(dx, dy)| {
        let nx = x + dx;
        let ny = y + dy;
        if nx >= 0 && nx < w && ny >= 0 && ny < h {
            Some((nx, ny))
        } else {
            None
        }
    })
}

fn idx(x: i32, y: i32, w: i32) -> usize {
    (y * w + x) as usize
}

fn is_adj(a: Pt, b: Pt) -> bool {
    (a.x - b.x).abs() + (a.y - b.y).abs() == 1
}

fn snake_path_rect(w: i32, h: i32) -> Option<Vec<Pt>> {
    if w <= 0 || h <= 0 {
        return None;
    }
    if w == 1 && h == 1 {
        return Some(vec![Pt { x: 0, y: 0 }]);
    }

    let mut path = Vec::with_capacity((w * h) as usize);

    if h % 2 == 1 {
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    path.push(Pt { x, y });
                }
            } else {
                for x in (0..w).rev() {
                    path.push(Pt { x, y });
                }
            }
        }
        if path.first() == Some(&Pt { x: 0, y: 0 }) {
            return Some(path);
        }
    }

    if w % 2 == 1 {
        path.clear();
        for x in 0..w {
            if x % 2 == 0 {
                for y in 0..h {
                    path.push(Pt { x, y });
                }
            } else {
                for y in (0..h).rev() {
                    path.push(Pt { x, y });
                }
            }
        }
        if path.first() == Some(&Pt { x: 0, y: 0 }) {
            return Some(path);
        }
    }

    None
}

fn has_hamilton_cycle_empty_rect(w: i32, h: i32) -> bool {
    w > 1 && h > 1 && (w * h) % 2 == 0
}

fn build_cycle_compatible_path_empty_rect(w: i32, h: i32) -> Option<Vec<Pt>> {
    // Need a Hamiltonian path starting at (0,0) whose end is adjacent to start.
    // For empty rectangle this exists iff a Hamiltonian cycle exists.
    if !has_hamilton_cycle_empty_rect(w, h) {
        return None;
    }

    // Construct by a simple pattern for even width.
    if w % 2 == 0 {
        let mut p = Vec::with_capacity((w * h) as usize);
        // First column top->bottom starts at (0,0).
        for y in 0..h {
            p.push(Pt { x: 0, y });
        }
        // Snake remaining columns bottom-up/top-down.
        for x in 1..w {
            if x % 2 == 1 {
                for y in (1..h).rev() {
                    p.push(Pt { x, y });
                }
            } else {
                for y in 1..h {
                    p.push(Pt { x, y });
                }
            }
        }
        if p.len() == (w * h) as usize && is_adj(*p.last().unwrap(), Pt { x: 0, y: 0 }) {
            return Some(p);
        }
    }

    // Symmetric for even height.
    if h % 2 == 0 {
        let mut p = Vec::with_capacity((w * h) as usize);
        for x in 0..w {
            p.push(Pt { x, y: 0 });
        }
        for y in 1..h {
            if y % 2 == 1 {
                for x in (1..w).rev() {
                    p.push(Pt { x, y });
                }
            } else {
                for x in 1..w {
                    p.push(Pt { x, y });
                }
            }
        }
        if p.len() == (w * h) as usize && is_adj(*p.last().unwrap(), Pt { x: 0, y: 0 }) {
            return Some(p);
        }
    }

    None
}

struct Solver {
    w: i32,
    n: usize,
    require_cycle: bool,
    free: Vec<bool>,
    adj: Vec<Vec<usize>>,
    color: Vec<u8>,
    path: Vec<usize>,
    used: Vec<bool>,
    start: usize,
    deadline: Instant,
}

impl Solver {
    fn new(w: i32, h: i32, blocked: &HashSet<(i32, i32)>, require_cycle: bool, limit: Duration) -> Option<Self> {
        if w <= 0 || h <= 0 {
            return None;
        }
        let n = (w * h) as usize;
        let mut free = vec![true; n];
        for &(x, y) in blocked {
            if x >= 0 && x < w && y >= 0 && y < h {
                free[idx(x, y, w)] = false;
            }
        }
        let start = idx(0, 0, w);
        if !free[start] {
            return None;
        }

        let mut adj = vec![Vec::new(); n];
        let mut color = vec![0u8; n];
        for y in 0..h {
            for x in 0..w {
                let i = idx(x, y, w);
                color[i] = ((x + y) & 1) as u8;
                if !free[i] {
                    continue;
                }
                for (nx, ny) in neighbors4(x, y, w, h) {
                    let j = idx(nx, ny, w);
                    if free[j] {
                        adj[i].push(j);
                    }
                }
            }
        }

        Some(Self {
            w,
            n,
            require_cycle,
            free,
            adj,
            color,
            path: Vec::new(),
            used: vec![false; n],
            start,
            deadline: Instant::now() + limit,
        })
    }

    fn free_count(&self) -> usize {
        self.free.iter().filter(|&&b| b).count()
    }

    fn quick_impossible(&self) -> bool {
        let total = self.free_count();
        if total == 0 {
            return true;
        }
        if self.adj[self.start].is_empty() && total > 1 {
            return true;
        }

        // Connectedness of free cells
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        seen[self.start] = true;
        q.push_back(self.start);
        let mut got = 0usize;
        while let Some(u) = q.pop_front() {
            got += 1;
            for &v in &self.adj[u] {
                if !seen[v] {
                    seen[v] = true;
                    q.push_back(v);
                }
            }
        }
        if got != total {
            return true;
        }

        // Degree-0 or too many degree-1 vertices test.
        let mut deg1 = 0usize;
        for i in 0..self.n {
            if !self.free[i] {
                continue;
            }
            let d = self.adj[i].len();
            if d == 0 && total > 1 {
                return true;
            }
            if d == 1 {
                deg1 += 1;
            }
        }
        if self.require_cycle {
            if deg1 > 0 {
                return true;
            }
        } else if deg1 > 2 {
            return true;
        }

        // Bipartite parity checks.
        let mut c0 = 0usize;
        let mut c1 = 0usize;
        for i in 0..self.n {
            if self.free[i] {
                if self.color[i] == 0 {
                    c0 += 1;
                } else {
                    c1 += 1;
                }
            }
        }
        if (c0 as i64 - c1 as i64).abs() > 1 {
            return true;
        }
        if self.require_cycle {
            if c0 != c1 {
                return true;
            }
            // In a cycle, start must have at least 2 free neighbors unless single-cell case (already filtered).
            if self.adj[self.start].len() < 2 {
                return true;
            }
        } else {
            // For odd total, both ends are on start color. For even, opposite colors.
            // This is a weak check but cheap.
            if total % 2 == 0 {
                // endpoint must be opposite color; existence assumed.
            } else {
                let sc = self.color[self.start];
                if (sc == 0 && c0 < c1) || (sc == 1 && c1 < c0) {
                    return true;
                }
            }
        }

        false
    }

    fn count_unvisited_degree(&self, u: usize, current: usize) -> usize {
        self.adj[u]
            .iter()
            .filter(|&&v| !self.used[v] || v == current)
            .count()
    }

    fn connectivity_prune(&self, current: usize) -> bool {
        // All unvisited nodes must be in one component reachable from some unvisited neighbor of current.
        let mut first = None;
        for &v in &self.adj[current] {
            if !self.used[v] {
                first = Some(v);
                break;
            }
        }

        let rem = self.free_count() - self.path.len();
        if rem == 0 {
            return false;
        }
        let Some(seed) = first else {
            return true;
        };

        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        seen[seed] = true;
        q.push_back(seed);
        let mut reached = 1usize;

        while let Some(u) = q.pop_front() {
            for &v in &self.adj[u] {
                if self.used[v] {
                    continue;
                }
                if !seen[v] {
                    seen[v] = true;
                    reached += 1;
                    q.push_back(v);
                }
            }
        }

        reached != rem
    }

    fn forced_move(&self, current: usize) -> Option<usize> {
        // If any unvisited vertex has only one way to be connected in remaining graph,
        // force stepping into it when adjacent.
        let mut forced_target = None;
        for i in 0..self.n {
            if !self.free[i] || self.used[i] {
                continue;
            }
            let mut deg = 0usize;
            let mut only = usize::MAX;
            for &v in &self.adj[i] {
                if !self.used[v] || v == current {
                    deg += 1;
                    only = v;
                    if deg > 1 {
                        break;
                    }
                }
            }
            if deg == 0 {
                return Some(usize::MAX);
            }
            if deg == 1 {
                // i must be next to only at some point. If only == current and adjacent, we should take i now.
                if only == current && self.adj[current].contains(&i) {
                    if let Some(prev) = forced_target {
                        if prev != i {
                            return Some(usize::MAX);
                        }
                    } else {
                        forced_target = Some(i);
                    }
                }
            }
        }
        forced_target
    }

    fn dfs(&mut self, current: usize) -> bool {
        if Instant::now() >= self.deadline {
            return false;
        }

        if self.path.len() == self.free_count() {
            return !self.require_cycle || self.adj[current].contains(&self.start);
        }

        if self.connectivity_prune(current) {
            return false;
        }

        if let Some(fm) = self.forced_move(current) {
            if fm == usize::MAX {
                return false;
            }
            self.used[fm] = true;
            self.path.push(fm);
            if self.dfs(fm) {
                return true;
            }
            self.path.pop();
            self.used[fm] = false;
            return false;
        }

        let mut cand: Vec<usize> = self.adj[current]
            .iter()
            .copied()
            .filter(|&v| !self.used[v])
            .collect();

        // Warnsdorff-like ordering: smallest onward degree first.
        cand.sort_by_key(|&v| {
            let onward = self.count_unvisited_degree(v, current);
            let to_start = if self.require_cycle && self.adj[v].contains(&self.start) { 0 } else { 1 };
            (onward, to_start)
        });

        for v in cand {
            // Avoid closing cycle too early.
            if self.require_cycle && self.path.len() + 1 < self.free_count() && v == self.start {
                continue;
            }

            self.used[v] = true;
            self.path.push(v);
            if self.dfs(v) {
                return true;
            }
            self.path.pop();
            self.used[v] = false;
        }

        false
    }

    fn solve(&mut self) -> Option<Vec<Pt>> {
        if self.quick_impossible() {
            return None;
        }

        self.path.clear();
        self.path.push(self.start);
        self.used[self.start] = true;

        if self.dfs(self.start) {
            let out = self
                .path
                .iter()
                .map(|&i| Pt {
                    x: (i as i32) % self.w,
                    y: (i as i32) / self.w,
                })
                .collect();
            return Some(out);
        }

        None
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        println!("0");
        return;
    }

    let mut it = input.split_whitespace();
    let w: i32 = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let h: i32 = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let k: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let require_cycle = match it.next().and_then(|s| s.parse::<i32>().ok()) {
        Some(v) => v != 0,
        None => {
            println!("0");
            return;
        }
    };

    let mut blocked = HashSet::with_capacity(k * 2 + 1);
    for _ in 0..k {
        let x = match it.next().and_then(|s| s.parse::<i32>().ok()) {
            Some(v) => v,
            None => {
                println!("0");
                return;
            }
        };
        let y = match it.next().and_then(|s| s.parse::<i32>().ok()) {
            Some(v) => v,
            None => {
                println!("0");
                return;
            }
        };
        blocked.insert((x, y));
    }

    // Fast path: empty rectangle admits direct constructions.
    if blocked.is_empty() {
        if require_cycle {
            if let Some(path) = build_cycle_compatible_path_empty_rect(w, h) {
                println!("{}", path.len());
                for p in path {
                    println!("{} {}", p.x, p.y);
                }
                return;
            }
        } else if let Some(path) = snake_path_rect(w, h) {
            println!("{}", path.len());
            for p in path {
                println!("{} {}", p.x, p.y);
            }
            return;
        }
    }

    // General case: branch-and-bound DFS with aggressive pruning.
    // Keep a strict wall-clock budget to guarantee termination.
    let limit = Duration::from_secs(295);
    let mut solver = match Solver::new(w, h, &blocked, require_cycle, limit) {
        Some(s) => s,
        None => {
            println!("0");
            return;
        }
    };

    if let Some(path) = solver.solve() {
        println!("{}", path.len());
        for p in path {
            println!("{} {}", p.x, p.y);
        }
    } else {
        println!("0");
    }
}
