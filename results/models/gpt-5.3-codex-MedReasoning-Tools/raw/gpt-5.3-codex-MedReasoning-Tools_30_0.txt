import sys
import time
import random
from array import array


def ints_from_stdin():
    data = sys.stdin.buffer.read()
    n = len(data)
    i = 0
    while i < n:
        while i < n and data[i] <= 32:
            i += 1
        if i >= n:
            break
        sign = 1
        if data[i] == 45:  # '-'
            sign = -1
            i += 1
        x = 0
        while i < n:
            c = data[i]
            if 48 <= c <= 57:
                x = x * 10 + (c - 48)
                i += 1
            else:
                break
        yield sign * x


def eval_lit(lit, assign):
    v = lit if lit > 0 else -lit
    val = assign[v]
    return val if lit > 0 else (1 - val)


def main():
    start = time.time()
    it = ints_from_stdin()

    try:
        n = next(it)
        m = next(it)
    except StopIteration:
        return

    if n <= 0:
        sys.stdout.write("SAT\n\n")
        return

    total_lits = 3 * m
    clauses = array('i', [0]) * total_lits
    head = array('i', [-1]) * (n + 1)
    next_occ = array('i', [0]) * total_lits

    p = 0
    for cid in range(m):
        for _ in range(3):
            lit = next(it)
            clauses[p] = lit
            v = lit if lit > 0 else -lit
            next_occ[p] = head[v]
            head[v] = p
            p += 1

    rng = random.Random((n * 1000003 + m) ^ 0x9E3779B97F4A7C15)
    assign = bytearray(n + 1)
    for i in range(1, n + 1):
        assign[i] = rng.getrandbits(1)

    sat_count = bytearray(m)
    breakcount = array('i', [0]) * (n + 1)
    unsat_flag = bytearray(m)
    unsat = []

    def clause_true_count_and_unique(cid, flip_var=0, flip_to=0):
        base = 3 * cid
        cnt = 0
        uniq = 0
        for k in (base, base + 1, base + 2):
            lit = clauses[k]
            v = lit if lit > 0 else -lit
            val = flip_to if v == flip_var else assign[v]
            t = val if lit > 0 else (1 - val)
            if t:
                cnt += 1
                uniq = v
        return cnt, uniq

    for cid in range(m):
        cnt, uniq = clause_true_count_and_unique(cid)
        sat_count[cid] = cnt
        if cnt == 0:
            unsat_flag[cid] = 1
            unsat.append(cid)
        elif cnt == 1:
            breakcount[uniq] += 1

    if not unsat:
        out = ['SAT', '\n']
        out.append(' '.join('1' if assign[i] else '0' for i in range(1, n + 1)))
        sys.stdout.write(''.join(out))
        return

    time_budget = 295.0
    max_restarts = 3
    noise = 0.35
    max_flips = max(200000, min(120000000, 20 * m + 1000))

    for _ in range(max_restarts):
        flips = 0
        while flips < max_flips and (time.time() - start) < time_budget:
            while unsat and not unsat_flag[unsat[-1]]:
                unsat.pop()
            if not unsat:
                out = ['SAT', '\n']
                out.append(' '.join('1' if assign[i] else '0' for i in range(1, n + 1)))
                sys.stdout.write(''.join(out))
                return

            # Random unsatisfied clause (retry stale indices if needed)
            for _try in range(16):
                cid = unsat[rng.randrange(len(unsat))]
                if unsat_flag[cid]:
                    break
            else:
                while unsat and not unsat_flag[unsat[-1]]:
                    unsat.pop()
                if not unsat:
                    break
                cid = unsat[-1]

            b = 3 * cid
            l1, l2, l3 = clauses[b], clauses[b + 1], clauses[b + 2]
            v1 = l1 if l1 > 0 else -l1
            v2 = l2 if l2 > 0 else -l2
            v3 = l3 if l3 > 0 else -l3

            if rng.random() < noise:
                flip_v = (v1, v2, v3)[rng.randrange(3)]
            else:
                b1 = breakcount[v1]
                b2 = breakcount[v2]
                b3 = breakcount[v3]
                best = b1
                flip_v = v1
                if b2 < best or (b2 == best and rng.getrandbits(1)):
                    best = b2
                    flip_v = v2
                if b3 < best or (b3 == best and rng.getrandbits(1)):
                    flip_v = v3

            old = assign[flip_v]
            new = 1 - old

            occ = head[flip_v]
            while occ != -1:
                c = occ // 3
                before = sat_count[c]
                base = 3 * c

                # Remove old unit-sat ownership
                if before == 1:
                    _, uniq_before = clause_true_count_and_unique(c)
                    breakcount[uniq_before] -= 1

                after, uniq_after = clause_true_count_and_unique(c, flip_var=flip_v, flip_to=new)
                sat_count[c] = after

                if before == 0 and after > 0:
                    unsat_flag[c] = 0
                elif before > 0 and after == 0:
                    if not unsat_flag[c]:
                        unsat_flag[c] = 1
                        unsat.append(c)

                if after == 1:
                    breakcount[uniq_after] += 1

                occ = next_occ[occ]

            assign[flip_v] = new
            flips += 1

        # Restart with a fresh random assignment if needed
        for i in range(1, n + 1):
            assign[i] = rng.getrandbits(1)
        sat_count[:] = b'\x00' * m
        breakcount[:] = array('i', [0]) * (n + 1)
        unsat_flag[:] = b'\x00' * m
        unsat.clear()

        for cid in range(m):
            cnt, uniq = clause_true_count_and_unique(cid)
            sat_count[cid] = cnt
            if cnt == 0:
                unsat_flag[cid] = 1
                unsat.append(cid)
            elif cnt == 1:
                breakcount[uniq] += 1

        if not unsat:
            out = ['SAT', '\n']
            out.append(' '.join('1' if assign[i] else '0' for i in range(1, n + 1)))
            sys.stdout.write(''.join(out))
            return

    # Guaranteed satisfiable by prompt; fallback outputs best effort format.
    sys.stdout.write('SAT\n')
    sys.stdout.write(' '.join('1' if assign[i] else '0' for i in range(1, n + 1)))


if __name__ == '__main__':
    main()
