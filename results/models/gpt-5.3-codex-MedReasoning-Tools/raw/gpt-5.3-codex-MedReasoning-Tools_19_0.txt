using System;
using System.Buffers;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private struct Candidate
    {
        public int X;
        public int Y;
        public int Z;
        public byte Voxel;
        public long Priority;
    }

    private sealed class CandidateComparer : IComparer<Candidate>
    {
        public int Compare(Candidate a, Candidate b)
        {
            int c = a.Priority.CompareTo(b.Priority);
            if (c != 0) return c;
            c = a.Z.CompareTo(b.Z);
            if (c != 0) return c;
            c = a.X.CompareTo(b.X);
            if (c != 0) return c;
            return a.Y.CompareTo(b.Y);
        }
    }

    private sealed class FastInput
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _len;
        private int _pos;

        public FastInput(Stream s, int bufferSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufferSize];
        }

        private int ReadByteRaw()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len == 0) return -1;
            }
            return _buf[_pos++];
        }

        public string ReadToken()
        {
            int c;
            do
            {
                c = ReadByteRaw();
                if (c < 0) return string.Empty;
            } while (c <= 32);

            var sb = new StringBuilder(16);
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByteRaw();
                if (c < 0) break;
            }
            return sb.ToString();
        }

        public byte ReadVoxelByte()
        {
            int c = ReadByteRaw();
            if (c < 0) return 0;
            return (byte)c;
        }
    }

    private sealed class Operation
    {
        public int DigX;
        public int DigY;
        public int DigZ;
        public int[] Steps = Array.Empty<int>(); // x0,y0,x1,y1,...
    }

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());

        int xCount = ParseInt(input.ReadToken());
        int yCount = ParseInt(input.ReadToken());
        int zCount = ParseInt(input.ReadToken());
        _ = ParseDouble(input.ReadToken());
        _ = ParseDouble(input.ReadToken());
        _ = ParseDouble(input.ReadToken());

        if (xCount <= 0 || yCount <= 0 || zCount <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        long xyLong = (long)xCount * yCount;
        if (xyLong > int.MaxValue)
        {
            Console.WriteLine(0);
            return;
        }

        int xy = (int)xyLong;
        var heights = new int[xy];
        Array.Fill(heights, -1);

        long totalVoxels = xyLong * zCount;
        bool smallMode = totalVoxels <= 80_000_000L;

        byte[]? voxels = null;
        if (smallMode)
        {
            voxels = new byte[(int)totalVoxels];
        }

        var initialCandidates = new List<Candidate>(Math.Min(1_000_000, xy));

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int colIndex = x * yCount + y;
                int topZ = -1;
                byte topByte = 0;
                long baseIndex = ((long)colIndex) * zCount;

                for (int z = 0; z < zCount; z++)
                {
                    byte b = input.ReadVoxelByte();
                    if (smallMode)
                    {
                        voxels![(int)(baseIndex + z)] = b;
                    }
                    if (b != 0)
                    {
                        topZ = z;
                        topByte = b;
                    }
                }

                heights[colIndex] = topZ;

                if (topZ >= 0)
                {
                    int mass = topByte & 15;
                    int value = topByte >> 4;
                    if (mass > 0 && value > 0)
                    {
                        long d = x + y;
                        long gross = 1000L * mass * value;
                        long optimisticFuel = mass * d;
                        long optimistic = gross - optimisticFuel;
                        if (optimistic > 0)
                        {
                            initialCandidates.Add(new Candidate
                            {
                                X = x,
                                Y = y,
                                Z = topZ,
                                Voxel = topByte,
                                Priority = optimistic
                            });
                        }
                    }
                }
            }
        }

        List<Operation> ops = smallMode
            ? SolveSmall(xCount, yCount, zCount, heights, voxels!, initialCandidates)
            : SolveLarge(xCount, yCount, heights, initialCandidates);

        WriteOutput(ops);
    }

    private static List<Operation> SolveLarge(int xCount, int yCount, int[] heights, List<Candidate> candidates)
    {
        const int maxOps = 50_000;
        candidates.Sort((a, b) => b.Priority.CompareTo(a.Priority));

        var ops = new List<Operation>(Math.Min(maxOps, candidates.Count));
        int take = Math.Min(maxOps * 4, candidates.Count);

        for (int i = 0; i < take && ops.Count < maxOps; i++)
        {
            var c = candidates[i];
            int idx = c.X * yCount + c.Y;
            if (heights[idx] != c.Z) continue;

            int mass = c.Voxel & 15;
            int value = c.Voxel >> 4;
            if (mass <= 0 || value <= 0) continue;

            if (!TryBuildPathToPlant(c.X, c.Y, yCount, heights, out int[] steps, out long fuel))
                continue;

            long gross = 1000L * mass * value;
            long net = gross - fuel;
            if (net <= 0) continue;

            ops.Add(new Operation
            {
                DigX = c.X,
                DigY = c.Y,
                DigZ = c.Z,
                Steps = steps
            });

            heights[idx] = c.Z - 1;
        }

        return ops;
    }

    private static List<Operation> SolveSmall(int xCount, int yCount, int zCount, int[] heights, byte[] voxels, List<Candidate> seed)
    {
        const int maxOps = 200_000;
        var ops = new List<Operation>(Math.Min(maxOps, seed.Count));

        var pq = new PriorityQueue<Candidate, Candidate>(new CandidateComparer());
        for (int i = 0; i < seed.Count; i++)
        {
            pq.Enqueue(seed[i], seed[i]);
        }

        while (pq.Count > 0 && ops.Count < maxOps)
        {
            Candidate c = pq.Dequeue();
            int idx = c.X * yCount + c.Y;
            int h = heights[idx];
            if (h != c.Z || h < 0) continue;

            long voxelIndex = ((long)idx) * zCount + h;
            byte b = voxels[(int)voxelIndex];
            if (b == 0) continue;

            int mass = b & 15;
            int value = b >> 4;
            if (mass <= 0 || value <= 0) continue;

            if (!TryBuildPathToPlant(c.X, c.Y, yCount, heights, out int[] steps, out long fuel))
                continue;

            long gross = 1000L * mass * value;
            long net = gross - fuel;
            if (net <= 0) continue;

            ops.Add(new Operation
            {
                DigX = c.X,
                DigY = c.Y,
                DigZ = h,
                Steps = steps
            });

            voxels[(int)voxelIndex] = 0;
            h--;
            while (h >= 0)
            {
                byte nb = voxels[(int)(((long)idx) * zCount + h)];
                if (nb != 0) break;
                h--;
            }
            heights[idx] = h;

            if (h >= 0)
            {
                byte nb = voxels[(int)(((long)idx) * zCount + h)];
                int nm = nb & 15;
                int nv = nb >> 4;
                if (nm > 0 && nv > 0)
                {
                    long d = c.X + c.Y;
                    long optimistic = 1000L * nm * nv - nm * d;
                    if (optimistic > 0)
                    {
                        var nc = new Candidate
                        {
                            X = c.X,
                            Y = c.Y,
                            Z = h,
                            Voxel = nb,
                            Priority = optimistic
                        };
                        pq.Enqueue(nc, nc);
                    }
                }
            }
        }

        return ops;
    }

    private static bool TryBuildPathToPlant(int sx, int sy, int yCount, int[] heights, out int[] steps, out long fuelCost)
    {
        if (sx == 0 && sy == 0)
        {
            steps = Array.Empty<int>();
            fuelCost = 0;
            return true;
        }

        if (TryBuildPath(sx, sy, 0, 0, yCount, heights, true, out steps, out fuelCost))
            return true;

        return TryBuildPath(sx, sy, 0, 0, yCount, heights, false, out steps, out fuelCost);
    }

    private static bool TryBuildPath(int sx, int sy, int dx, int dy, int yCount, int[] heights, bool xFirst, out int[] steps, out long fuelCost)
    {
        int len = Math.Abs(dx - sx) + Math.Abs(dy - sy);
        steps = new int[len * 2];
        fuelCost = 0;

        int cx = sx;
        int cy = sy;
        int ci = cx * yCount + cy;
        int ch = heights[ci];
        int w = 0;

        if (xFirst)
        {
            if (!WalkAxis(ref cx, ref cy, ref ch, dx, true, yCount, heights, steps, ref w, ref fuelCost)) return false;
            if (!WalkAxis(ref cx, ref cy, ref ch, dy, false, yCount, heights, steps, ref w, ref fuelCost)) return false;
        }
        else
        {
            if (!WalkAxis(ref cx, ref cy, ref ch, dy, false, yCount, heights, steps, ref w, ref fuelCost)) return false;
            if (!WalkAxis(ref cx, ref cy, ref ch, dx, true, yCount, heights, steps, ref w, ref fuelCost)) return false;
        }

        return true;
    }

    private static bool WalkAxis(ref int cx, ref int cy, ref int ch, int target, bool isX, int yCount, int[] heights, int[] steps, ref int w, ref long fuelCost)
    {
        int delta = isX ? target - cx : target - cy;
        if (delta == 0) return true;

        int step = delta > 0 ? 1 : -1;
        int count = Math.Abs(delta);

        for (int i = 0; i < count; i++)
        {
            int nx = isX ? cx + step : cx;
            int ny = isX ? cy : cy + step;
            int nh = heights[nx * yCount + ny];
            int up = nh - ch;
            if (up > 1) return false;

            if (up > 0) fuelCost += 5;
            else if (up == 0) fuelCost += 1;

            cx = nx;
            cy = ny;
            ch = nh;
            steps[w++] = cx;
            steps[w++] = cy;
        }

        return true;
    }

    private static void WriteOutput(List<Operation> ops)
    {
        var sb = new StringBuilder(Math.Max(64, ops.Count * 32));
        sb.Append(ops.Count).Append('\n');

        for (int i = 0; i < ops.Count; i++)
        {
            var op = ops[i];
            int n = op.Steps.Length / 2;
            sb.Append(n).Append(' ')
              .Append(op.DigX).Append(' ')
              .Append(op.DigY).Append(' ')
              .Append(op.DigZ);

            for (int j = 0; j < op.Steps.Length; j += 2)
            {
                sb.Append(' ').Append(op.Steps[j]).Append(' ').Append(op.Steps[j + 1]);
            }

            sb.Append(' ').Append(0).Append(' ').Append(0).Append('\n');
        }

        Console.Write(sb.ToString());
    }

    private static int ParseInt(string s)
    {
        if (!int.TryParse(s, out int v)) return 0;
        return v;
    }

    private static double ParseDouble(string s)
    {
        if (!double.TryParse(s, System.Globalization.NumberStyles.Float, System.Globalization.CultureInfo.InvariantCulture, out double v))
            return 0.0;
        return v;
    }
}
