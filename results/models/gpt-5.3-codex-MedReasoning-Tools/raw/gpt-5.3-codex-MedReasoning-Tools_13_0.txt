#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <vector>

class SuffixAutomaton {
public:
    struct State {
        int len = 0;
        int link = -1;
        int head = -1;
        int first_pos = -1;
        int best = 0;
        int tmp = 0;
    };

    struct Edge {
        int to = -1;
        int next = -1;
        unsigned char ch = 0;
    };

    explicit SuffixAutomaton(size_t max_len = 0) {
        states.reserve(max_len ? (2 * max_len) : 2);
        edges.reserve(max_len ? (2 * max_len) : 2);
        states.push_back(State{});
        states[0].link = -1;
        last = 0;
    }

    void extend(unsigned char c, int pos) {
        int cur = static_cast<int>(states.size());
        states.push_back(State{});
        states[cur].len = states[last].len + 1;
        states[cur].first_pos = pos;

        int p = last;
        while (p != -1 && get_transition(p, c) == -1) {
            add_transition(p, c, cur);
            p = states[p].link;
        }

        if (p == -1) {
            states[cur].link = 0;
        } else {
            int q = get_transition(p, c);
            if (states[p].len + 1 == states[q].len) {
                states[cur].link = q;
            } else {
                int clone = static_cast<int>(states.size());
                states.push_back(states[q]);
                states[clone].len = states[p].len + 1;
                states[clone].tmp = 0;
                states[clone].best = 0;

                while (p != -1 && get_transition(p, c) == q) {
                    set_transition(p, c, clone);
                    p = states[p].link;
                }

                states[q].link = clone;
                states[cur].link = clone;
            }
        }

        last = cur;
    }

    void finalize_best_init() {
        for (State &s : states) {
            s.best = s.len;
            s.tmp = 0;
        }
    }

    void build_order() {
        int max_len = 0;
        for (const State &s : states) {
            if (s.len > max_len) max_len = s.len;
        }

        std::vector<int> cnt(static_cast<size_t>(max_len) + 1, 0);
        for (const State &s : states) {
            ++cnt[s.len];
        }
        for (size_t i = 1; i < cnt.size(); ++i) {
            cnt[i] += cnt[i - 1];
        }

        order.assign(states.size(), 0);
        for (int i = static_cast<int>(states.size()) - 1; i >= 0; --i) {
            order[--cnt[states[i].len]] = i;
        }
    }

    void process_string(const std::string &s) {
        int v = 0;
        int l = 0;

        for (unsigned char c : s) {
            int to = get_transition(v, c);
            while (v != 0 && to == -1) {
                v = states[v].link;
                l = states[v].len;
                to = get_transition(v, c);
            }

            if (to != -1) {
                v = to;
                ++l;
            } else {
                v = 0;
                l = 0;
            }

            if (l > states[v].tmp) states[v].tmp = l;
        }

        for (int i = static_cast<int>(order.size()) - 1; i > 0; --i) {
            int state_id = order[i];
            int parent = states[state_id].link;
            if (parent >= 0) {
                int candidate = std::min(states[state_id].tmp, states[parent].len);
                if (candidate > states[parent].tmp) states[parent].tmp = candidate;
            }
        }

        for (State &st : states) {
            st.best = std::min(st.best, st.tmp);
            st.tmp = 0;
        }
    }

    std::pair<int, int> best_answer() const {
        int best_len = 0;
        int best_pos = -1;

        for (const State &st : states) {
            if (st.best > best_len) {
                best_len = st.best;
                best_pos = st.first_pos;
            }
        }

        return {best_len, best_pos};
    }

private:
    std::vector<State> states;
    std::vector<Edge> edges;
    std::vector<int> order;
    int last = 0;

    int get_transition(int state_id, unsigned char c) const {
        int e = states[state_id].head;
        while (e != -1) {
            if (edges[e].ch == c) return edges[e].to;
            e = edges[e].next;
        }
        return -1;
    }

    void add_transition(int state_id, unsigned char c, int to) {
        Edge e;
        e.to = to;
        e.ch = c;
        e.next = states[state_id].head;
        edges.push_back(e);
        states[state_id].head = static_cast<int>(edges.size()) - 1;
    }

    void set_transition(int state_id, unsigned char c, int to) {
        int e = states[state_id].head;
        while (e != -1) {
            if (edges[e].ch == c) {
                edges[e].to = to;
                return;
            }
            e = edges[e].next;
        }
        add_transition(state_id, c, to);
    }
};

static std::string brute_force_small(const std::vector<std::string> &strings, size_t pivot_idx) {
    const std::string &base = strings[pivot_idx];
    const int m = static_cast<int>(base.size());

    for (int len = m; len >= 1; --len) {
        for (int i = 0; i + len <= m; ++i) {
            const std::string sub = base.substr(i, len);
            bool ok = true;
            for (size_t k = 0; k < strings.size(); ++k) {
                if (k == pivot_idx) continue;
                if (strings[k].find(sub) == std::string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) return sub;
        }
    }

    return "";
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    if (!(std::cin >> n)) {
        return 0;
    }

    std::string dummy;
    std::getline(std::cin, dummy);

    if (n <= 0) {
        std::cout << '\n';
        return 0;
    }

    std::vector<std::string> strings;
    strings.reserve(static_cast<size_t>(n));

    for (int i = 0; i < n; ++i) {
        std::string s;
        std::getline(std::cin, s);
        strings.push_back(std::move(s));
    }

    size_t pivot_idx = 0;
    for (size_t i = 1; i < strings.size(); ++i) {
        if (strings[i].size() < strings[pivot_idx].size()) {
            pivot_idx = i;
        }
    }

    const size_t min_len = strings[pivot_idx].size();
    size_t total_len = 0;
    for (const std::string &s : strings) total_len += s.size();

    const bool trivial_case = (strings.size() <= 3 && min_len <= 200 && total_len <= 2000);

    if (trivial_case) {
        std::cout << brute_force_small(strings, pivot_idx) << '\n';
        return 0;
    }

    std::swap(strings[0], strings[pivot_idx]);

    SuffixAutomaton sam(strings[0].size());
    for (int i = 0; i < static_cast<int>(strings[0].size()); ++i) {
        sam.extend(static_cast<unsigned char>(strings[0][i]), i);
    }

    sam.finalize_best_init();
    sam.build_order();

    for (size_t i = 1; i < strings.size(); ++i) {
        sam.process_string(strings[i]);
    }

    auto [best_len, best_pos] = sam.best_answer();
    if (best_len <= 0 || best_pos < 0) {
        std::cout << '\n';
        return 0;
    }

    const int start = best_pos - best_len + 1;
    std::cout << strings[0].substr(static_cast<size_t>(start), static_cast<size_t>(best_len)) << '\n';
    return 0;
}
