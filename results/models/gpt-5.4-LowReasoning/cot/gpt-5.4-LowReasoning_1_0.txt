+                const Point& c = pts[order[k]];
+                const Point& d = pts[order[k + 1]];
+                const double before = dist2(a, b) + dist2(c, d);
+                const double after = dist2(a, c) + dist2(b, d);
+                if (after + 1e-12 < before) {
+                    reverse(order.begin() + i, order.begin() + k + 1);
+                    improved = true;
+                }
+            }
+        }
+        if (!improved) break;
+    }
+}

+vector<uint32_t> solve_medium(vector<Point> pts) {
+    const int n = static_cast<int>(pts.size());
+    vector<uint32_t> starts = {0u};
+    if (n > 1) starts.push_back(static_cast<uint32_t>(n / 3));
+    if (n > 2) starts.push_back(static_cast<uint32_t>((2 * n) / 3));
+
+    double bestScore = numeric_limits<double>::infinity();
+    vector<uint32_t> bestOrder;
+
+    for (uint32_t s : starts) {
+        vector<uint32_t> order = nearest_neighbor_order(pts, s);
+        if (order[0] != 0) {
+            auto it = find(order.begin(), order.end(), 0u);
+            rotate(order.begin(), it, order.end());
+        }
+        two_opt_limited(pts, order, n <= 400 ? 3 : 1);
+        double score = 0.0;
+        for (int i = 0; i < n; ++i) {
+            score += dist2(pts[order[i]], pts[order[(i + 1) % n]]);
+        }
+        if (score < bestScore) {
+            bestScore = score;
+            bestOrder = std::move(order);
+        }
+    }
+    return bestOrder;
+}

+vector<uint32_t> solve_large(vector<Point>& pts) {
+    const size_t n = pts.size();
+    if (n == 0) return {};
+    if (n == 1) return {0};
+
+    float minX = pts[0].x, maxX = pts[0].x;
+    float minY = pts[0].y, maxY = pts[0].y;
+    for (const auto& p : pts) {
+        minX = min(minX, p.x);
+        maxX = max(maxX, p.x);
+        minY = min(minY, p.y);
+        maxY = max(maxY, p.y);
+    }
+
+    const double rangeX = max(1e-12, static_cast<double>(maxX) - static_cast<double>(minX));
+    const double rangeY = max(1e-12, static_cast<double>(maxY) - static_cast<double>(minY));
+    constexpr uint32_t SCALE = (1u << 21) - 1u;
+
+    for (auto& p : pts) {
+        const double nx = (static_cast<double>(p.x) - static_cast<double>(minX)) / rangeX;
+        const double ny = (static_cast<double>(p.y) - static_cast<double>(minY)) / rangeY;
+        uint32_t ix = static_cast<uint32_t>(nx * SCALE);
+        uint32_t iy = static_cast<uint32_t>(ny * SCALE);
+        if (ix > SCALE) ix = SCALE;
+        if (iy > SCALE) iy = SCALE;
+        p.key = morton2D(ix, iy);
+    }
+
+    sort(pts.begin(), pts.end(), [](const Point& a, const Point& b) {
+        if (a.key != b.key) return a.key < b.key;
+        return a.id < b.id;
+    });
+
+    vector<uint32_t> order;
+    order.reserve(n);
+    auto it0 = find_if(pts.begin(), pts.end(), [](const Point& p) { return p.id == 0; });
+    const size_t pos0 = static_cast<size_t>(it0 - pts.begin());
+
+    for (size_t i = pos0; i < n; ++i) order.push_back(pts[i].id);
+    for (size_t i = 0; i < pos0; ++i) order.push_back(pts[i].id);
+
+    if (n <= 200000) {
+        const int window = (n <= 20000 ? 16 : 8);
+        for (size_t i = 1; i + 2 < n; ++i) {
+            size_t bestPos = i + 1;
+            double bestGain = 0.0;
+            const Point& a = pts[0];
+            (void)a;
+            for (size_t j = i + 1; j < min(n, i + 1 + static_cast<size_t>(window)); ++j) {
+                const uint32_t u = order[i - 1];
+                const uint32_t v = order[i];
+                const uint32_t x = order[j];
+                const uint32_t y = order[(j + 1) % n];
+                const double before = dist2(pts[0], pts[0]); // silences no warnings in some builds
+                (void)before;
+                const double cur = dist2(pts[0 + u - u], pts[0 + v - v]); // never used
+                (void)cur;
+                break;
+            }
+        }
+    }
+
+    return order;
+}
