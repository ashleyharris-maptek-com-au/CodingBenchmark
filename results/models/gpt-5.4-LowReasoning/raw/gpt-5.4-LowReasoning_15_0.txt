use std::f64::consts::PI;
use std::io::{self, Read};

const MAX_SCALE: f64 = 10.0;
const EPS: f64 = 1.0e-9;

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn cross(self, other: Self) -> f64 {
        self.x * other.y - self.y * other.x
    }

}

impl std::ops::Add for Vec2 {
    type Output = Self;

    fn add(self, rhs: Self) -> Self::Output {
        Self::new(self.x + rhs.x, self.y + rhs.y)
    }
}

impl std::ops::Sub for Vec2 {
    type Output = Self;

    fn sub(self, rhs: Self) -> Self::Output {
        Self::new(self.x - rhs.x, self.y - rhs.y)
    }
}

impl std::ops::Mul<f64> for Vec2 {
    type Output = Self;

    fn mul(self, rhs: f64) -> Self::Output {
        Self::new(self.x * rhs, self.y * rhs)
    }
}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }

    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(self, other: Self) -> Self {
        Self::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }

    fn len(self) -> f64 {
        self.dot(self).sqrt()
    }

    fn normalized(self) -> Self {
        let len = self.len();
        if len <= EPS {
            Self::new(0.0, 0.0, 0.0)
        } else {
            self * (1.0 / len)
        }
    }
}

impl std::ops::Add for Vec3 {
    type Output = Self;

    fn add(self, rhs: Self) -> Self::Output {
        Self::new(self.x + rhs.x, self.y + rhs.y, self.z + rhs.z)
    }
}

impl std::ops::Sub for Vec3 {
    type Output = Self;

    fn sub(self, rhs: Self) -> Self::Output {
        Self::new(self.x - rhs.x, self.y - rhs.y, self.z - rhs.z)
    }
}

impl std::ops::Mul<f64> for Vec3 {
    type Output = Self;

    fn mul(self, rhs: f64) -> Self::Output {
        Self::new(self.x * rhs, self.y * rhs, self.z * rhs)
    }
}

#[derive(Clone, Copy)]
struct Basis2 {
    c0: Vec2,
    c1: Vec2,
    inv_det: f64,
}

impl Basis2 {
    fn new(c0: Vec2, c1: Vec2) -> Option<Self> {
        let det = c0.cross(c1);
        if det.abs() <= EPS {
            None
        } else {
            Some(Self {
                c0,
                c1,
                inv_det: 1.0 / det,
            })
        }
    }

    fn solve(&self, p: Vec2) -> Vec2 {
        Vec2::new(
            p.cross(self.c1) * self.inv_det,
            self.c0.cross(p) * self.inv_det,
        )
    }
}

#[derive(Clone, Copy)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone)]
struct Candidate {
    quat: Quat,
    a: Vec2,
    b: Vec2,
    t1: Vec2,
    t2: Vec2,
    min_i: i32,
    max_i: i32,
    min_j: i32,
    max_j: i32,
    score: i64,
}

fn project_along_sun(p: Vec3, sun: Vec3) -> Vec2 {
    let t = -p.z / sun.z;
    Vec2::new(p.x + sun.x * t, p.y + sun.y * t)
}

fn build_frame(sun_dir: Vec3) -> (Vec3, Vec3, Vec3) {
    let helper = if sun_dir.z.abs() < 0.9 {
        Vec3::new(0.0, 0.0, 1.0)
    } else {
        Vec3::new(1.0, 0.0, 0.0)
    };
    let u0 = helper.cross(sun_dir).normalized();
    let v0 = sun_dir.cross(u0).normalized();
    (u0, v0, sun_dir)
}

fn rotate_about_axis(u0: Vec3, v0: Vec3, angle: f64) -> (Vec3, Vec3) {
    let c = angle.cos();
    let s = angle.sin();
    let u = u0 * c + v0 * s;
    let v = v0 * c - u0 * s;
    (u, v)
}

fn matrix_to_quat(u: Vec3, v: Vec3, w: Vec3) -> Quat {
    let m00 = u.x;
    let m01 = v.x;
    let m02 = w.x;
    let m10 = u.y;
    let m11 = v.y;
    let m12 = w.y;
    let m20 = u.z;
    let m21 = v.z;
    let m22 = w.z;

    let trace = m00 + m11 + m22;
    if trace > 0.0 {
        let s = (trace + 1.0).sqrt() * 2.0;
        Quat {
            w: 0.25 * s,
            x: (m21 - m12) / s,
            y: (m02 - m20) / s,
            z: (m10 - m01) / s,
        }
    } else if m00 > m11 && m00 > m22 {
        let s = (1.0 + m00 - m11 - m22).sqrt() * 2.0;
        Quat {
            w: (m21 - m12) / s,
            x: 0.25 * s,
            y: (m01 + m10) / s,
            z: (m02 + m20) / s,
        }
    } else if m11 > m22 {
        let s = (1.0 + m11 - m00 - m22).sqrt() * 2.0;
        Quat {
            w: (m02 - m20) / s,
            x: (m01 + m10) / s,
            y: 0.25 * s,
            z: (m12 + m21) / s,
        }
    } else {
        let s = (1.0 + m22 - m00 - m11).sqrt() * 2.0;
        Quat {
            w: (m10 - m01) / s,
            x: (m02 + m20) / s,
            y: (m12 + m21) / s,
            z: 0.25 * s,
        }
    }
}

fn bbox(points: &[Vec2]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for &p in points {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }
    (min_x, max_x, min_y, max_y)
}

fn point_in_polygon(point: Vec2, polygon: &[Vec2]) -> bool {
    let mut inside = false;
    let n = polygon.len();
    for i in 0..n {
        let a = polygon[i];
        let b = polygon[(i + 1) % n];
        let ab = b - a;
        let ap = point - a;
        if ab.cross(ap).abs() <= EPS && ap.dot(point - b) <= EPS {
            return true;
        }
        let intersects = ((a.y > point.y) != (b.y > point.y))
            && (point.x < (b.x - a.x) * (point.y - a.y) / (b.y - a.y) + a.x);
        if intersects {
            inside = !inside;
        }
    }
    inside
}

fn point_in_convex(point: Vec2, poly: &[Vec2]) -> bool {
    let n = poly.len();
    let mut prev: f64 = 0.0;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let cross = (b - a).cross(point - a);
        if cross.abs() <= EPS {
            continue;
        }
        if prev.abs() <= EPS {
            prev = cross;
        } else if cross * prev < -EPS {
            return false;
        }
    }
    true
}

fn on_segment(a: Vec2, b: Vec2, p: Vec2) -> bool {
    (b - a).cross(p - a).abs() <= EPS
        && (p.x - a.x).min(p.x - b.x) <= EPS
        && (p.x - a.x).max(p.x - b.x) >= -EPS
        && (p.y - a.y).min(p.y - b.y) <= EPS
        && (p.y - a.y).max(p.y - b.y) >= -EPS
}

fn orient(a: Vec2, b: Vec2, c: Vec2) -> f64 {
    (b - a).cross(c - a)
}

fn segments_intersect(a1: Vec2, a2: Vec2, b1: Vec2, b2: Vec2) -> bool {
    let o1 = orient(a1, a2, b1);
    let o2 = orient(a1, a2, b2);
    let o3 = orient(b1, b2, a1);
    let o4 = orient(b1, b2, a2);

    if o1.abs() <= EPS && on_segment(a1, a2, b1) {
        return true;
    }
    if o2.abs() <= EPS && on_segment(a1, a2, b2) {
        return true;
    }
    if o3.abs() <= EPS && on_segment(b1, b2, a1) {
        return true;
    }
    if o4.abs() <= EPS && on_segment(b1, b2, a2) {
        return true;
    }

    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn rhombus_vertices(center: Vec2, a: Vec2, b: Vec2) -> [Vec2; 4] {
    [center + a, center + b, center - a, center - b]
}

fn polygon_intersects_rhombus(polygon: &[Vec2], rhombus: &[Vec2; 4]) -> bool {
    for &v in rhombus {
        if point_in_polygon(v, polygon) {
            return true;
        }
    }
    for &v in polygon {
        if point_in_convex(v, rhombus) {
            return true;
        }
    }
    for i in 0..polygon.len() {
        let a1 = polygon[i];
        let a2 = polygon[(i + 1) % polygon.len()];
        for j in 0..4 {
            let b1 = rhombus[j];
            let b2 = rhombus[(j + 1) % 4];
            if segments_intersect(a1, a2, b1, b2) {
                return true;
            }
        }
    }
    false
}

fn estimate_index_range(basis: Basis2, bb: (f64, f64, f64, f64)) -> (i32, i32, i32, i32, i64) {
    let (min_x, max_x, min_y, max_y) = bb;
    let corners = [
        Vec2::new(min_x, min_y),
        Vec2::new(min_x, max_y),
        Vec2::new(max_x, min_y),
        Vec2::new(max_x, max_y),
    ];
    let mut min_i = f64::INFINITY;
    let mut max_i = f64::NEG_INFINITY;
    let mut min_j = f64::INFINITY;
    let mut max_j = f64::NEG_INFINITY;
    for corner in corners {
        let ij = basis.solve(corner);
        min_i = min_i.min(ij.x);
        max_i = max_i.max(ij.x);
        min_j = min_j.min(ij.y);
        max_j = max_j.max(ij.y);
    }
    let pad = 2.0;
    let lo_i = (min_i - pad).floor() as i32;
    let hi_i = (max_i + pad).ceil() as i32;
    let lo_j = (min_j - pad).floor() as i32;
    let hi_j = (max_j + pad).ceil() as i32;
    let width = i64::from(hi_i - lo_i + 1);
    let height = i64::from(hi_j - lo_j + 1);
    (lo_i, hi_i, lo_j, hi_j, width * height)
}

fn choose_candidate(polygon: &[Vec2], sun: Vec3) -> Candidate {
    let sun_dir = sun.normalized();
    let (u0, v0, w) = build_frame(sun_dir);
    let bb = bbox(polygon);
    let search_steps = if polygon.len() <= 64 { 24 } else if polygon.len() <= 512 { 12 } else { 6 };

    let mut best: Option<Candidate> = None;
    for step in 0..search_steps {
        let angle = (step as f64) * PI / (search_steps as f64);
        let (u, v) = rotate_about_axis(u0, v0, angle);
        let a = project_along_sun(u * MAX_SCALE, sun);
        let b = project_along_sun(v * MAX_SCALE, sun);
        let t1 = a + b;
        let t2 = a - b;
        let Some(basis) = Basis2::new(t1, t2) else {
            continue;
        };
        let (min_i, max_i, min_j, max_j, score) = estimate_index_range(basis, bb);
        let quat = matrix_to_quat(u, v, w);
        let candidate = Candidate {
            quat,
            a,
            b,
            t1,
            t2,
            min_i,
            max_i,
            min_j,
            max_j,
            score,
        };
        match &best {
            Some(current) if current.score <= candidate.score => {}
            _ => best = Some(candidate),
        }
    }
    best.expect("failed to build a valid candidate")
}

fn dedup_centers(centers: &mut Vec<Vec2>) {
    centers.sort_by(|a, b| {
        a.x.partial_cmp(&b.x)
            .unwrap_or(std::cmp::Ordering::Equal)
            .then(a.y.partial_cmp(&b.y).unwrap_or(std::cmp::Ordering::Equal))
    });
    centers.dedup_by(|a, b| (a.x - b.x).abs() <= 1.0e-8 && (a.y - b.y).abs() <= 1.0e-8);
}

fn solve(polygon: &[Vec2], sun: Vec3) -> String {
    let candidate = choose_candidate(polygon, sun);

    let mut centers = Vec::new();
    for i in candidate.min_i..=candidate.max_i {
        for j in candidate.min_j..=candidate.max_j {
            let center = candidate.t1 * (i as f64) + candidate.t2 * (j as f64);
            let rhombus = rhombus_vertices(center, candidate.a, candidate.b);
            if polygon_intersects_rhombus(polygon, &rhombus) {
                centers.push(center);
            }
        }
    }
    dedup_centers(&mut centers);

    let mut out = String::new();
    out.push_str(&format!("{}\n", centers.len()));
    for center in centers {
        out.push_str(&format!(
            "{:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9}\n",
            center.x,
            center.y,
            0.0,
            candidate.quat.w,
            candidate.quat.x,
            candidate.quat.y,
            candidate.quat.z,
            MAX_SCALE,
        ));
    }
    out
}

fn parse_input(input: &str) -> (Vec<Vec2>, Vec3) {
    let mut it = input.split_whitespace();
    let n: usize = it
        .next()
        .expect("missing vertex count")
        .parse()
        .expect("invalid vertex count");
    let mut polygon = Vec::with_capacity(n);
    for _ in 0..n {
        let x: f64 = it.next().expect("missing x").parse().expect("invalid x");
        let y: f64 = it.next().expect("missing y").parse().expect("invalid y");
        polygon.push(Vec2::new(x, y));
    }
    let sx: f64 = it.next().expect("missing sx").parse().expect("invalid sx");
    let sy: f64 = it.next().expect("missing sy").parse().expect("invalid sy");
    let sz: f64 = it.next().expect("missing sz").parse().expect("invalid sz");
    assert!(sz.abs() > EPS, "sun direction must not be parallel to the ground");
    (polygon, Vec3::new(sx, sy, sz))
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("failed to read stdin");
    let (polygon, sun) = parse_input(&input);
    let answer = solve(&polygon, sun);
    print!("{}", answer);
}
