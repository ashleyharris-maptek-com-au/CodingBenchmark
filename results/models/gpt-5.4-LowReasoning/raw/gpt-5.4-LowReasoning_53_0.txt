import math


MU_EARTH = 3.986004418e14
EARTH_RADIUS_M = 6371000.0
MAX_THRUST_CMD = 1.0
MAX_ATT_RATE = 0.035

_memory = {
    "pos": [0.0, 0.0, 0.0],
    "vel": [0.0, 0.0, 0.0],
    "att": [0.0, 0.0, 0.0],  # roll, pitch, yaw
    "rates": [0.0, 0.0, 0.0],
    "range": None,
    "phase": "acquire",
    "time": 0.0,
    "last_good": 0.0,
    "bias_force": [0.0, 0.0, 0.0],
    "last_cmd": [0.0, 0.0, 0.0],
    "last_meas_vel": [0.0, 0.0, 0.0],
}


def _clamp(v, lo, hi):
    return lo if v < lo else hi if v > hi else v


def _median(values):
    if not values:
        return None
    s = sorted(values)
    n = len(s)
    m = n // 2
    if n % 2:
        return s[m]
    return 0.5 * (s[m - 1] + s[m])


def _fuse_axis(state, prefixes, component, fallback):
    vals = []
    for prefix in prefixes:
        value_key = f"{prefix}_{component}"
        valid_key = f"{prefix}_{component.split('_')[0]}_valid" if "vel" not in component else f"{prefix}_{component.replace('vel_', 'v_').split('_ms')[0]}_valid"
        if state.get(valid_key, False):
            vals.append(float(state.get(value_key, 0.0)))
    if not vals:
        return fallback, False
    med = _median(vals)
    consistent = [v for v in vals if abs(v - med) <= max(2.0, 0.25 * max(1.0, abs(med)))]
    return _median(consistent or vals), True


def _fuse_rel_state(state):
    pos = [0.0, 0.0, 0.0]
    vel = [0.0, 0.0, 0.0]
    valid_pos = [False, False, False]
    valid_vel = [False, False, False]
    pos_map = ["rel_pos_x_m", "rel_pos_y_m", "rel_pos_z_m"]
    vel_map = ["rel_vel_x_ms", "rel_vel_y_ms", "rel_vel_z_ms"]

    for i, comp in enumerate(pos_map):
        pos[i], valid_pos[i] = _fuse_axis(state, ["radar", "lidar"], comp, _memory["pos"][i])
    for i, comp in enumerate(vel_map):
        vel[i], valid_vel[i] = _fuse_axis(state, ["radar", "lidar"], comp, _memory["vel"][i])

    ranges = [
        float(state[f"range_{i}_m"])
        for i in (1, 2, 3)
        if state.get(f"range_{i}_valid", False)
    ]
    scalar_range = _median(ranges) if ranges else None

    if scalar_range is not None:
        vec_norm = math.sqrt(sum(p * p for p in pos))
        if vec_norm > 1e-6 and abs(vec_norm - scalar_range) > max(10.0, 0.15 * scalar_range):
            scale = scalar_range / vec_norm
            pos = [p * scale for p in pos]
        elif vec_norm <= 1e-6:
            pos = [0.0, -scalar_range, 0.0]

    return pos, vel, scalar_range, any(valid_pos), any(valid_vel)


def _fuse_rates(state):
    axes = [("roll", 0), ("pitch", 1), ("yaw", 2)]
    out = [0.0, 0.0, 0.0]
    for axis, idx in axes:
        vals = []
        for i in (1, 2, 3):
            if state.get(f"gyro_{i}_{axis}_valid", False):
                vals.append(math.radians(float(state.get(f"gyro_{i}_{axis}_dps", 0.0))))
        out[idx] = _median(vals) if vals else _memory["rates"][idx]
    return out


def _fuse_attitude(state):
    axes = [("roll", 0), ("pitch", 1), ("yaw", 2)]
    out = [0.0, 0.0, 0.0]
    for axis, idx in axes:
        vals = []
        for i in (1, 2):
            if state.get(f"star_tracker_{i}_{axis}_valid", False):
                vals.append(math.radians(float(state.get(f"star_tracker_{i}_{axis}_deg", 0.0))))
        out[idx] = _median(vals) if vals else _memory["att"][idx]
    return out


def _mean_motion(state):
    a = None
    if state.get("ground_semi_major_axis_m_valid", False):
        a = float(state.get("ground_semi_major_axis_m", 0.0))
    elif state.get("ground_altitude_km_valid", False):
        a = EARTH_RADIUS_M + 1000.0 * float(state.get("ground_altitude_km", 0.0))
    if not a or a < EARTH_RADIUS_M:
        a = EARTH_RADIUS_M + 400000.0
    return math.sqrt(MU_EARTH / (a ** 3))


def _estimate_disturbance(measured_vel, dt):
    if dt <= 1e-5:
        return
    expected = _memory["last_cmd"]
    obs = [(measured_vel[i] - _memory["last_meas_vel"][i]) / dt for i in range(3)]
    alpha = 0.08
    for i in range(3):
        residual = _clamp(obs[i] - expected[i], -0.1, 0.1)
        _memory["bias_force"][i] = (1.0 - alpha) * _memory["bias_force"][i] + alpha * residual
    _memory["last_meas_vel"] = measured_vel[:]


def _target_speed(range_m, fuel_pct):
    if range_m > 800.0:
        base = 6.0
    elif range_m > 300.0:
        base = 3.5
    elif range_m > 100.0:
        base = 1.5
    elif range_m > 30.0:
        base = 0.6
    elif range_m > 10.0:
        base = 0.25
    else:
        base = 0.08
    if fuel_pct < 15.0:
        base *= 0.65
    return base


def _update_phase(range_m):
    if range_m > 200.0:
        _memory["phase"] = "far"
    elif range_m > 50.0:
        _memory["phase"] = "mid"
    elif range_m > 10.0:
        _memory["phase"] = "close"
    else:
        _memory["phase"] = "final"


def _guidance(pos, vel, range_m, n, fuel_pct):
    x, y, z = pos
    vx, vy, vz = vel
    phase = _memory["phase"]
    cmd = [0.0, 0.0, 0.0]
    vmag = math.sqrt(vx * vx + vy * vy + vz * vz)
    approach_speed = _target_speed(range_m, fuel_pct)

    if phase == "far":
        desired_vy = 0.0
        if y < -20.0:
            desired_vy = min(approach_speed, 0.0045 * abs(y) + 0.25)
        elif y > 20.0:
            desired_vy = -min(approach_speed, 0.0045 * abs(y) + 0.25)

        # Counter-intuitive orbital capture:
        # behind target (negative y) => thrust retrograde (negative y command) to drop lower/faster orbit
        cmd[1] = _clamp(-0.16 * desired_vy - 0.35 * vy - 0.002 * y, -0.65, 0.65)

        desired_x = _clamp(-0.10 * y, -80.0, 80.0)
        cmd[0] = _clamp(-0.010 * (x - desired_x) - 0.20 * vx + 1.2 * n * vy, -0.30, 0.30)

        desired_z = 0.0
        if abs(z) > 15.0:
            desired_z = 0.0
        cmd[2] = _clamp(-0.010 * (z - desired_z) - 0.22 * vz, -0.25, 0.25)

    elif phase == "mid":
        desired_y_rate = 0.0
        if abs(y) > 3.0:
            desired_y_rate = _clamp(0.018 * (-y), -approach_speed, approach_speed)
        cmd[1] = _clamp(-0.50 * (vy - desired_y_rate) - 0.004 * y, -0.35, 0.35)
        cmd[0] = _clamp(-0.030 * x - 0.45 * vx + 1.6 * n * vy, -0.25, 0.25)
        cmd[2] = _clamp(-0.030 * z - 0.45 * vz, -0.25, 0.25)

    elif phase == "close":
        desired_y = -1.5
        desired_y_rate = _clamp(0.03 * (desired_y - y), -approach_speed, approach_speed)
        cmd[1] = _clamp(-0.75 * (vy - desired_y_rate) - 0.010 * (y - desired_y), -0.22, 0.22)
        cmd[0] = _clamp(-0.060 * x - 0.65 * vx, -0.18, 0.18)
        cmd[2] = _clamp(-0.060 * z - 0.65 * vz, -0.18, 0.18)

    else:
        desired_y = -0.8
        desired_y_rate = _clamp(0.05 * (desired_y - y), -0.10, 0.10)
        cmd[1] = _clamp(-0.90 * (vy - desired_y_rate) - 0.020 * (y - desired_y), -0.12, 0.12)
        cmd[0] = _clamp(-0.10 * x - 0.90 * vx, -0.10, 0.10)
        cmd[2] = _clamp(-0.10 * z - 0.90 * vz, -0.10, 0.10)

        if range_m < 2.0 and vmag < 0.3:
            cmd = [-0.6 * vx, -0.6 * vy, -0.6 * vz]

    if vmag > max(approach_speed * 1.4, 0.35):
        cmd[0] += -0.50 * vx
        cmd[1] += -0.50 * vy
        cmd[2] += -0.50 * vz

    return [_clamp(cmd[i] - _memory["bias_force"][i], -MAX_THRUST_CMD, MAX_THRUST_CMD) for i in range(3)]


def _camera_refine(state, pos):
    if not state.get("dock_camera_active", False):
        return pos
    bx_valid = state.get("dock_camera_bearing_x_valid", False)
    by_valid = state.get("dock_camera_bearing_y_valid", False)
    bz_valid = state.get("dock_camera_bearing_z_valid", False)
    if not (bx_valid and by_valid and bz_valid):
        return pos
    rng = float(state.get("dock_camera_range_m", 0.0))
    if rng <= 0.0 or rng > 500.0:
        return pos
    bearing = [
        float(state.get("dock_camera_bearing_x", 0.0)),
        float(state.get("dock_camera_bearing_y", 0.0)),
        float(state.get("dock_camera_bearing_z", 0.0)),
    ]
    norm = math.sqrt(sum(b * b for b in bearing))
    if norm < 1e-6:
        return pos
    cam_pos = [rng * b / norm for b in bearing]
    blend = 0.7 if rng < 50.0 else 0.35
    return [(1.0 - blend) * pos[i] + blend * cam_pos[i] for i in range(3)]


def autopilot_step(state: dict, dt: float) -> dict:
    dt = max(0.02, min(float(dt), 2.0))
    _memory["time"] += dt

    pos, vel, scalar_range, have_pos, have_vel = _fuse_rel_state(state)
    pos = _camera_refine(state, pos)

    if not have_pos:
        pos = [_memory["pos"][i] + _memory["vel"][i] * dt for i in range(3)]
    if not have_vel:
        vel = [(pos[i] - _memory["pos"][i]) / dt for i in range(3)]

    if scalar_range is None:
        scalar_range = math.sqrt(sum(p * p for p in pos))
    else:
        scalar_range = 0.6 * scalar_range + 0.4 * math.sqrt(sum(p * p for p in pos))

    _update_phase(scalar_range)
    n = _mean_motion(state)

    rates = _fuse_rates(state)
    att = _fuse_attitude(state)

    fuel_pct = float(state.get("fuel_remaining_pct", 100.0))
    fuel_kg = float(state.get("fuel_remaining_kg", 500.0))
    low_fuel = fuel_pct < 12.0 or fuel_kg < 60.0

    _estimate_disturbance(vel, dt)
    cmd = _guidance(pos, vel, scalar_range, n, fuel_pct)

    if low_fuel:
        cmd = [0.7 * c for c in cmd]
        if scalar_range > 100.0 and abs(pos[1]) < 20.0:
            cmd[0] *= 0.7
            cmd[2] *= 0.7

    desired_att = [0.0, 0.0, 0.0]
    if scalar_range < 500.0 and state.get("dock_camera_active", False):
        bx = float(state.get("dock_camera_bearing_x", 0.0)) if state.get("dock_camera_bearing_x_valid", False) else 0.0
        bz = float(state.get("dock_camera_bearing_z", 0.0)) if state.get("dock_camera_bearing_z_valid", False) else 0.0
        desired_att[1] = _clamp(-0.15 * bz, -0.08, 0.08)
        desired_att[2] = _clamp(0.15 * bx, -0.08, 0.08)

    att_rate_roll = _clamp(-1.2 * att[0] - 0.6 * rates[0], -MAX_ATT_RATE, MAX_ATT_RATE)
    att_rate_pitch = _clamp(2.0 * (desired_att[1] - att[1]) - 0.7 * rates[1], -MAX_ATT_RATE, MAX_ATT_RATE)
    att_rate_yaw = _clamp(2.0 * (desired_att[2] - att[2]) - 0.7 * rates[2], -MAX_ATT_RATE, MAX_ATT_RATE)

    # Deadband near target to avoid chatter and save fuel.
    if scalar_range < 5.0:
        for i in range(3):
            if abs(pos[i]) < 0.15 and abs(vel[i]) < 0.03:
                cmd[i] = 0.0

    cmd = [_clamp(c, -MAX_THRUST_CMD, MAX_THRUST_CMD) for c in cmd]

    _memory["pos"] = pos[:]
    _memory["vel"] = vel[:]
    _memory["att"] = att[:]
    _memory["rates"] = rates[:]
    _memory["range"] = scalar_range
    _memory["last_cmd"] = cmd[:]
    _memory["last_good"] = _memory["time"]

    return {
        "thrust_x": cmd[0],
        "thrust_y": cmd[1],
        "thrust_z": cmd[2],
        "att_rate_roll": att_rate_roll,
        "att_rate_pitch": att_rate_pitch,
        "att_rate_yaw": att_rate_yaw,
    }
