from math import atan2, cos, radians, sin
from statistics import median


_MEM = {
    "speed_kmh": 0.0,
    "yaw_rate_dps": 0.0,
    "lane_offset_m": 0.0,
    "curvature_1pm": 0.0,
    "heading_deg": 0.0,
    "x_m": 0.0,
    "y_m": 0.0,
    "front_obstacle_m": 200.0,
    "front_rel_speed_kmh": 0.0,
    "speed_limit_kmh": 130.0,
    "lane_width_m": 3.6,
    "integral_speed": 0.0,
    "integral_lane": 0.0,
    "target_lane_hold": 1,
    "lane_change_timer_s": 0.0,
    "steer_prev": 0.0,
    "throttle_prev": 0.0,
    "brake_prev": 0.0,
    "vision_dropout_s": 0.0,
    "front_blocked_s": 0.0,
}


def _clamp(value, lo, hi):
    return max(lo, min(hi, value))


def _median_if_any(values, default):
    return median(values) if values else default


def _pick_numeric(values, prev, hard_min=None, hard_max=None, max_step=None):
    filtered = []
    for value in values:
        if value is None:
            continue
        if hard_min is not None and value < hard_min:
            continue
        if hard_max is not None and value > hard_max:
            continue
        if max_step is not None and abs(value - prev) > max_step:
            continue
        filtered.append(value)
    if filtered:
        return median(filtered)
    fallback = []
    for value in values:
        if value is None:
            continue
        if hard_min is not None and value < hard_min:
            continue
        if hard_max is not None and value > hard_max:
            continue
        fallback.append(value)
    return median(fallback) if fallback else prev


def _rate_limit(target, prev, rise_per_s, fall_per_s, dt):
    delta = target - prev
    if delta >= 0.0:
        return prev + min(delta, rise_per_s * dt)
    return prev + max(delta, -fall_per_s * dt)


def _ema(prev, value, alpha):
    return prev + alpha * (value - prev)


def autopilot_step(state: dict, dt: float) -> dict:
    global _MEM

    dt = state.get("dt", dt)
    if dt is None or dt <= 0.0:
        dt = 0.05
    dt = _clamp(dt, 0.01, 0.2)

    mu = _clamp(state.get("estimated_mu", 1.0), 0.05, 1.2)
    rain = 1.0 if state.get("rain_sensor", False) else 0.0

    wheel_speeds = []
    for corner in ("fl", "fr", "rl", "rr"):
        value = state.get(f"wheel_speed_{corner}_kmh")
        valid = state.get(f"wheel_speed_{corner}_valid", True)
        if isinstance(value, (int, float)):
            plausible = 0.0 <= value <= 300.0
            if plausible and (valid or abs(value - _MEM["speed_kmh"]) < 20.0):
                wheel_speeds.append(float(value))

    vehicle_speeds = []
    for idx in (1, 2):
        value = state.get(f"vehicle_speed_{idx}_kmh")
        if isinstance(value, (int, float)) and 0.0 <= value <= 300.0:
            vehicle_speeds.append(float(value))

    gps_speeds = []
    gps_positions = []
    gps_headings = []
    for idx in (1, 2):
        speed = state.get(f"gps_{idx}_speed_kmh")
        heading = state.get(f"gps_{idx}_heading_deg")
        x_m = state.get(f"gps_{idx}_x_m")
        y_m = state.get(f"gps_{idx}_y_m")
        valid = state.get(f"gps_{idx}_valid", True)
        if isinstance(speed, (int, float)) and 0.0 <= speed <= 300.0:
            if valid or abs(speed - _MEM["speed_kmh"]) < 15.0:
                gps_speeds.append(float(speed))
        if all(isinstance(v, (int, float)) for v in (x_m, y_m)):
            if valid or abs(x_m - _MEM["x_m"]) < 20.0:
                gps_positions.append((float(x_m), float(y_m)))
        if isinstance(heading, (int, float)):
            if valid or abs(heading - _MEM["heading_deg"]) < 45.0:
                gps_headings.append(float(heading))

    fused_speed = _pick_numeric(
        wheel_speeds + vehicle_speeds + gps_speeds,
        _MEM["speed_kmh"],
        hard_min=0.0,
        hard_max=300.0,
        max_step=max(10.0, dt * 30.0),
    )
    _MEM["speed_kmh"] = _ema(_MEM["speed_kmh"], fused_speed, 0.35)
    speed_ms = _MEM["speed_kmh"] / 3.6

    yaw_candidates = []
    for idx in (1, 2, 3):
        yaw = state.get(f"gyro_{idx}_yaw_rate_dps")
        valid = state.get(f"gyro_{idx}_valid", True)
        if isinstance(yaw, (int, float)) and abs(yaw) <= 80.0:
            if valid or abs(yaw - _MEM["yaw_rate_dps"]) < 10.0:
                yaw_candidates.append(float(yaw))
    _MEM["yaw_rate_dps"] = _ema(
        _MEM["yaw_rate_dps"],
        _pick_numeric(yaw_candidates, _MEM["yaw_rate_dps"], hard_min=-80.0, hard_max=80.0, max_step=15.0),
        0.35,
    )

    accel_lat = []
    accel_lon = []
    for idx in (1, 2, 3):
        lat = state.get(f"accel_{idx}_lateral_ms2")
        lon = state.get(f"accel_{idx}_longitudinal_ms2")
        valid = state.get(f"accel_{idx}_valid", True)
        if isinstance(lat, (int, float)) and abs(lat) <= 20.0 and (valid or abs(lat) < 8.0):
            accel_lat.append(float(lat))
        if isinstance(lon, (int, float)) and abs(lon) <= 20.0 and (valid or abs(lon) < 8.0):
            accel_lon.append(float(lon))
    lat_acc_ms2 = _median_if_any(accel_lat, 0.0)
    lon_acc_ms2 = _median_if_any(accel_lon, 0.0)

    if gps_positions:
        x_vals = [p[0] for p in gps_positions]
        y_vals = [p[1] for p in gps_positions]
        _MEM["x_m"] = _ema(_MEM["x_m"], median(x_vals), 0.4)
        _MEM["y_m"] = _ema(_MEM["y_m"], median(y_vals), 0.4)
    else:
        hdg = radians(_MEM["heading_deg"])
        _MEM["x_m"] += speed_ms * cos(hdg) * dt
        _MEM["y_m"] += speed_ms * sin(hdg) * dt

    if gps_headings:
        sin_sum = sum(sin(radians(h)) for h in gps_headings)
        cos_sum = sum(cos(radians(h)) for h in gps_headings)
        _MEM["heading_deg"] = atan2(sin_sum, cos_sum) * 180.0 / 3.141592653589793
    else:
        _MEM["heading_deg"] += _MEM["yaw_rate_dps"] * dt

    steering_angles = []
    for idx in (1, 2):
        angle = state.get(f"steering_angle_{idx}_deg")
        valid = state.get(f"steering_angle_{idx}_valid", True)
        if isinstance(angle, (int, float)) and abs(angle) <= 720.0:
            if valid or abs(angle) <= 90.0:
                steering_angles.append(float(angle))
    fused_steering_deg = _median_if_any(steering_angles, _MEM["steer_prev"] * 25.0)

    lidar = {}
    for ang in (-90, -60, -45, -30, -15, -5, 5, 15, 30, 45, 60, 90):
        key = f"lidar_{ang:+04d}_deg_range_m"
        valid_key = f"lidar_{ang:+04d}_deg_valid"
        rng = state.get(key)
        valid = state.get(valid_key, True)
        if isinstance(rng, (int, float)) and 0.0 < rng <= 250.0 and (valid or rng < 120.0):
            lidar[ang] = float(rng)

    vision_offsets = []
    vision_curvatures = []
    speed_limits = []
    lane_width_estimates = []
    obstacle_votes = []
    left_solid = 0
    right_solid = 0
    good_vision = 0
    for idx in (1, 2, 3):
        offset = state.get(f"vision_{idx}_lane_offset_m")
        left_dist = state.get(f"vision_{idx}_left_line_dist_m")
        right_dist = state.get(f"vision_{idx}_right_line_dist_m")
        left_valid = state.get(f"vision_{idx}_left_line_valid", True)
        right_valid = state.get(f"vision_{idx}_right_line_valid", True)
        left_type = str(state.get(f"vision_{idx}_left_line_type", "")).lower()
        right_type = str(state.get(f"vision_{idx}_right_line_type", "")).lower()
        curvature = state.get(f"vision_{idx}_road_curvature_1pm")
        spd_lim = state.get(f"vision_{idx}_speed_limit_kmh")
        spd_lim_valid = state.get(f"vision_{idx}_speed_limit_valid", True)
        obs_valid = state.get(f"vision_{idx}_obstacle_valid", True)
        obs_detected = bool(state.get(f"vision_{idx}_obstacle_detected", False))
        obs_dist = state.get(f"vision_{idx}_obstacle_dist_m")
        obs_rel = state.get(f"vision_{idx}_obstacle_rel_speed_kmh")

        local_good = False
        if isinstance(offset, (int, float)) and abs(offset) <= 4.5:
            vision_offsets.append(float(offset))
            local_good = True
        if isinstance(curvature, (int, float)) and abs(curvature) <= 0.2:
            vision_curvatures.append(float(curvature))
            local_good = True
        if isinstance(left_dist, (int, float)) and isinstance(right_dist, (int, float)):
            if 0.5 <= left_dist <= 6.0 and 0.5 <= right_dist <= 6.0 and (left_valid or right_valid):
                lane_width_estimates.append(float(left_dist + right_dist))
                local_good = True
        if isinstance(spd_lim, (int, float)) and 20.0 <= spd_lim <= 140.0 and (spd_lim_valid or spd_lim < 130.0):
            speed_limits.append(float(spd_lim))
        if "solid" in left_type:
            left_solid += 1
        if "solid" in right_type:
            right_solid += 1
        if obs_detected and isinstance(obs_dist, (int, float)) and 0.0 < obs_dist <= 250.0:
            rel = float(obs_rel) if isinstance(obs_rel, (int, float)) and abs(obs_rel) <= 200.0 else 0.0
            if obs_valid or obs_dist < 120.0:
                obstacle_votes.append((float(obs_dist), rel))
                local_good = True
        if local_good:
            good_vision += 1

    if good_vision == 0:
        _MEM["vision_dropout_s"] += dt
    else:
        _MEM["vision_dropout_s"] = 0.0

    if lane_width_estimates:
        _MEM["lane_width_m"] = _ema(_MEM["lane_width_m"], _clamp(median(lane_width_estimates), 3.0, 4.5), 0.2)

    if vision_offsets:
        _MEM["lane_offset_m"] = _ema(_MEM["lane_offset_m"], _clamp(median(vision_offsets), -2.5, 2.5), 0.3)
    else:
        side_balance = None
        if 90 in lidar and -90 in lidar:
            side_balance = (lidar[90] - lidar[-90]) * 0.05
        elif 60 in lidar and -60 in lidar:
            side_balance = (lidar[60] - lidar[-60]) * 0.04
        if side_balance is not None:
            _MEM["lane_offset_m"] = _ema(_MEM["lane_offset_m"], _clamp(side_balance, -1.5, 1.5), 0.15)
        else:
            _MEM["lane_offset_m"] *= 0.98

    if vision_curvatures:
        _MEM["curvature_1pm"] = _ema(_MEM["curvature_1pm"], median(vision_curvatures), 0.25)
    else:
        yaw_rad = radians(_MEM["yaw_rate_dps"])
        curvature_from_yaw = yaw_rad / max(speed_ms, 3.0)
        _MEM["curvature_1pm"] = _ema(_MEM["curvature_1pm"], _clamp(curvature_from_yaw, -0.05, 0.05), 0.15)

    if speed_limits:
        _MEM["speed_limit_kmh"] = _ema(_MEM["speed_limit_kmh"], median(speed_limits), 0.25)

    lidar_front = [lidar[a] for a in (-15, -5, 5, 15) if a in lidar]
    lidar_front_wide = [lidar[a] for a in (-30, -15, -5, 5, 15, 30) if a in lidar]
    front_lidar_m = min(lidar_front) if lidar_front else (min(lidar_front_wide) if lidar_front_wide else 200.0)

    if obstacle_votes:
        front_obs_m = min(v[0] for v in obstacle_votes)
        rel_candidates = [v[1] for v in obstacle_votes if abs(v[0] - front_obs_m) < 8.0]
        front_rel_kmh = median(rel_candidates) if rel_candidates else 0.0
        fused_front_m = min(front_obs_m, front_lidar_m + 5.0) if lidar_front_wide else front_obs_m
        _MEM["front_obstacle_m"] = _ema(_MEM["front_obstacle_m"], fused_front_m, 0.5)
        _MEM["front_rel_speed_kmh"] = _ema(_MEM["front_rel_speed_kmh"], front_rel_kmh, 0.4)
    else:
        inferred_rel = 0.0
        if front_lidar_m < 120.0:
            _MEM["front_obstacle_m"] = _ema(_MEM["front_obstacle_m"], front_lidar_m, 0.35)
        else:
            _MEM["front_obstacle_m"] = _ema(_MEM["front_obstacle_m"], 200.0, 0.1)
        _MEM["front_rel_speed_kmh"] = _ema(_MEM["front_rel_speed_kmh"], inferred_rel, 0.2)

    left_sector = min([lidar[a] for a in (30, 45, 60, 90) if a in lidar] or [80.0])
    right_sector = min([lidar[a] for a in (-30, -45, -60, -90) if a in lidar] or [80.0])

    target_lane = int(_clamp(state.get("target_lane", state.get("current_lane", _MEM["target_lane_hold"])), 0, 2))
    current_lane = int(_clamp(state.get("current_lane", _MEM["target_lane_hold"]), 0, 2))

    lane_change_allowed_left = current_lane < 2 and left_sector > 10.0 and left_solid < 2
    lane_change_allowed_right = current_lane > 0 and right_sector > 10.0 and right_solid < 2

    rel_speed_ms = _MEM["front_rel_speed_kmh"] / 3.6
    closing_speed_ms = max(0.0, -rel_speed_ms)
    time_gap_s = _MEM["front_obstacle_m"] / max(speed_ms, 1.0)
    ttc_s = _MEM["front_obstacle_m"] / max(closing_speed_ms, 0.1) if closing_speed_ms > 0.1 else 999.0

    traction_speed_cap = 130.0
    if mu < 0.75:
        traction_speed_cap -= (0.75 - mu) * 80.0
    if mu < 0.4:
        traction_speed_cap -= (0.4 - mu) * 120.0
    traction_speed_cap -= rain * 10.0
    traction_speed_cap = _clamp(traction_speed_cap, 30.0, 130.0)

    curve_cap = 130.0
    curve_mag = abs(_MEM["curvature_1pm"])
    if curve_mag > 1e-4:
        safe_lat = _clamp(2.5 * mu + 0.5, 0.8, 4.0)
        curve_cap = (safe_lat / curve_mag) ** 0.5 * 3.6
    curve_cap = _clamp(curve_cap, 25.0, 130.0)

    commanded_speed_kmh = min(
        float(state.get("target_speed_kmh", _MEM["speed_limit_kmh"])),
        _MEM["speed_limit_kmh"] + 2.0,
        traction_speed_cap,
        curve_cap,
    )

    if time_gap_s < 2.0:
        commanded_speed_kmh = min(commanded_speed_kmh, max(0.0, _MEM["speed_kmh"] + _MEM["front_rel_speed_kmh"] - 8.0))
    if ttc_s < 4.0:
        commanded_speed_kmh = min(commanded_speed_kmh, max(0.0, speed_ms * 3.6 - (4.0 - ttc_s) * 18.0))

    front_blocked = _MEM["front_obstacle_m"] < 45.0 and (ttc_s < 6.0 or time_gap_s < 1.8)
    if front_blocked:
        _MEM["front_blocked_s"] += dt
    else:
        _MEM["front_blocked_s"] = 0.0

    chosen_lane = target_lane
    if _MEM["lane_change_timer_s"] > 0.0:
        chosen_lane = _MEM["target_lane_hold"]
        _MEM["lane_change_timer_s"] = max(0.0, _MEM["lane_change_timer_s"] - dt)
    else:
        if front_blocked and _MEM["front_blocked_s"] > 0.3:
            prefer_left = left_sector >= right_sector
            if prefer_left and lane_change_allowed_left:
                chosen_lane = current_lane + 1
            elif lane_change_allowed_right:
                chosen_lane = current_lane - 1
            elif lane_change_allowed_left:
                chosen_lane = current_lane + 1
            else:
                chosen_lane = current_lane
        elif abs(target_lane - current_lane) <= 1:
            if target_lane > current_lane and lane_change_allowed_left:
                chosen_lane = current_lane + 1
            elif target_lane < current_lane and lane_change_allowed_right:
                chosen_lane = current_lane - 1
            else:
                chosen_lane = current_lane
        else:
            chosen_lane = current_lane
        if chosen_lane != current_lane:
            _MEM["target_lane_hold"] = chosen_lane
            _MEM["lane_change_timer_s"] = 1.8 if mu > 0.5 else 2.6
        else:
            _MEM["target_lane_hold"] = chosen_lane

    lane_error_m = _MEM["lane_offset_m"] + (chosen_lane - current_lane) * _MEM["lane_width_m"]
    lookahead_gain = _clamp(0.015 * speed_ms + 0.35, 0.35, 1.0)
    yaw_term = radians(_MEM["yaw_rate_dps"]) * lookahead_gain
    curvature_term = _MEM["curvature_1pm"] * (6.0 + 0.15 * speed_ms * speed_ms)

    _MEM["integral_lane"] = _clamp(_MEM["integral_lane"] + lane_error_m * dt, -2.0, 2.0)
    lane_kp = 0.32 if mu > 0.5 else 0.22
    lane_ki = 0.05 if good_vision > 0 else 0.01
    lane_kd = 0.18 if good_vision > 0 else 0.10
    raw_steer = (
        lane_kp * lane_error_m
        + lane_ki * _MEM["integral_lane"]
        + lane_kd * yaw_term
        + 0.9 * curvature_term
    )

    if abs(lat_acc_ms2) > 4.0 * max(mu, 0.3):
        raw_steer *= 0.85
    if abs(fused_steering_deg - _MEM["steer_prev"] * 25.0) > 12.0 and speed_ms > 20.0:
        raw_steer *= 0.9
    if _MEM["vision_dropout_s"] > 0.5:
        raw_steer *= 0.75

    steer_limit = 0.85 if mu > 0.5 else max(0.25, 0.65 * mu + 0.2)
    steer_cmd = _clamp(raw_steer, -steer_limit, steer_limit)
    steer_rate = 1.8 if mu > 0.5 else 0.9
    steer_cmd = _rate_limit(steer_cmd, _MEM["steer_prev"], steer_rate, steer_rate, dt)

    speed_error = commanded_speed_kmh - _MEM["speed_kmh"]
    if abs(speed_error) < 8.0:
        _MEM["integral_speed"] = _clamp(_MEM["integral_speed"] + speed_error * dt, -20.0, 20.0)
    else:
        _MEM["integral_speed"] *= 0.9

    kp_speed = 0.030 if mu > 0.5 else 0.020
    ki_speed = 0.006 if mu > 0.5 else 0.003
    throttle_cmd = kp_speed * speed_error + ki_speed * _MEM["integral_speed"]
    brake_cmd = 0.0

    desired_decel = 0.0
    if ttc_s < 6.0:
        desired_decel = max(desired_decel, _clamp((6.0 - ttc_s) * (1.2 if mu > 0.4 else 0.8), 0.0, 8.0 * mu + 1.0))
    if _MEM["front_obstacle_m"] < max(8.0, speed_ms * 1.2):
        desired_decel = max(desired_decel, 8.5 * mu + 1.2)
    if time_gap_s < 1.0:
        desired_decel = max(desired_decel, 5.0 * mu + 1.0)
    if mu < 0.25:
        desired_decel = min(desired_decel, 2.2)

    if desired_decel > 0.0:
        throttle_cmd = min(throttle_cmd, 0.0)
        brake_cmd = _clamp(desired_decel / max(2.0, 8.0 * mu + 1.0), 0.0, 1.0)
    elif speed_error < -2.0:
        brake_cmd = _clamp((-speed_error) * (0.018 if mu > 0.5 else 0.012), 0.0, 0.5 if mu > 0.5 else 0.3)

    throttle_cmd = _clamp(throttle_cmd, 0.0, 1.0)

    if abs(steer_cmd) > 0.35:
        throttle_cmd *= 0.85
    if curve_cap < _MEM["speed_kmh"] - 5.0:
        throttle_cmd *= 0.6
    if mu < 0.35:
        throttle_cmd = min(throttle_cmd, 0.25)
    elif mu < 0.55:
        throttle_cmd = min(throttle_cmd, 0.45)

    brake_sensors = []
    for idx in (1, 2):
        value = state.get(f"brake_pressure_{idx}_pct")
        valid = state.get(f"brake_pressure_{idx}_valid", True)
        if isinstance(value, (int, float)) and 0.0 <= value <= 100.0 and (valid or value < 60.0):
            brake_sensors.append(float(value))
    brake_feedback = median(brake_sensors) / 100.0 if brake_sensors else None
    if brake_cmd > 0.2 and brake_feedback is not None and brake_feedback < brake_cmd * 0.25:
        throttle_cmd = 0.0
        brake_cmd = 1.0
        commanded_speed_kmh = 0.0

    throttle_cmd = _rate_limit(throttle_cmd, _MEM["throttle_prev"], 1.0 if mu > 0.5 else 0.5, 2.5, dt)
    brake_cmd = _rate_limit(brake_cmd, _MEM["brake_prev"], 2.5, 3.5, dt)

    if brake_cmd > 0.05:
        throttle_cmd = min(throttle_cmd, 0.05)
    if _MEM["front_obstacle_m"] < 5.0:
        brake_cmd = 1.0
        throttle_cmd = 0.0
    if _MEM["vision_dropout_s"] > 1.5 and not lidar_front_wide:
        throttle_cmd = min(throttle_cmd, 0.2)
        commanded_speed_kmh = min(commanded_speed_kmh, 60.0)

    _MEM["steer_prev"] = steer_cmd
    _MEM["throttle_prev"] = throttle_cmd
    _MEM["brake_prev"] = brake_cmd

    return {
        "steering": _clamp(steer_cmd, -1.0, 1.0),
        "throttle": _clamp(throttle_cmd, 0.0, 1.0),
        "brake": _clamp(brake_cmd, 0.0, 1.0),
    }
