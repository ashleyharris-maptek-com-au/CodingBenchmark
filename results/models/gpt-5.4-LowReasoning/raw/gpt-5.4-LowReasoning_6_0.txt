using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double NormSq() => X * X + Y * Y + Z * Z;
        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec3 Cross(Vec3 a, Vec3 b) =>
            new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X
            );
        public Vec3 Normalized()
        {
            double n = Norm();
            if (n < 1e-12) return new Vec3(0, 0, 0);
            return this / n;
        }
    }

    sealed class Node
    {
        public Vec3 R;
        public Vec3 V;
        public double Rmag;
        public double Vmag;
        public Vec3 Hhat = new Vec3(0, 0, 1);
        public Vec3 Evec = new Vec3(0, 0, 0);
        public double Ecc;
        public double A;
        public double CircSpeed;
        public double Energy;
        public double Phase;
    }

    static readonly CultureInfo CI = CultureInfo.InvariantCulture;

    public static void Main()
    {
        string? header = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(header)) return;
        var h = header.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(h[0], CI);
        double mu = double.Parse(h[1], CI);

        Node start = ReadNode(mu);
        Node[] stations = new Node[n];
        for (int i = 0; i < n; i++) stations[i] = ReadNode(mu);

        double[,] cost = BuildCostMatrix(start, stations, mu);
        int[] order = Solve(cost, n);
        Console.WriteLine(string.Join(" ", order));
    }

    static Node ReadNode(double mu)
    {
        string? line = Console.ReadLine();
        while (line != null && line.Trim().Length == 0) line = Console.ReadLine();
        var p = line!.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        Vec3 r = new Vec3(
            double.Parse(p[0], CI),
            double.Parse(p[1], CI),
            double.Parse(p[2], CI));
        Vec3 v = new Vec3(
            double.Parse(p[3], CI),
            double.Parse(p[4], CI),
            double.Parse(p[5], CI));
        return BuildNode(r, v, mu);
    }

    static Node BuildNode(Vec3 r, Vec3 v, double mu)
    {
        var node = new Node();
        node.R = r;
        node.V = v;
        node.Rmag = r.Norm();
        node.Vmag = v.Norm();
        Vec3 h = Vec3.Cross(r, v);
        double hmag = h.Norm();
        node.Hhat = hmag > 1e-12 ? h / hmag : new Vec3(0, 0, 1);
        Vec3 e = (Vec3.Cross(v, h) / mu) - (r / Math.Max(node.Rmag, 1e-9));
        node.Evec = e;
        node.Ecc = e.Norm();
        node.Energy = 0.5 * node.Vmag * node.Vmag - mu / Math.Max(node.Rmag, 1e-9);
        double denom = -2.0 * node.Energy;
        node.A = Math.Abs(denom) > 1e-12 ? mu / denom : 1e12;
        if (double.IsNaN(node.A) || double.IsInfinity(node.A) || node.A <= 0) node.A = Math.Max(node.Rmag, 1.0);
        node.CircSpeed = Math.Sqrt(mu / Math.Max(node.Rmag, 1.0));
        node.Phase = Math.Atan2(r.Y, r.X);
        return node;
    }

    static double[,] BuildCostMatrix(Node start, Node[] stations, double mu)
    {
        int n = stations.Length;
        Node[] all = new Node[n + 1];
        all[0] = start;
        for (int i = 0; i < n; i++) all[i + 1] = stations[i];

        double[,] cost = new double[n + 1, n + 1];
        for (int i = 0; i <= n; i++)
        {
            for (int j = i + 1; j <= n; j++)
            {
                double c = TransferProxy(all[i], all[j], mu);
                cost[i, j] = c;
                cost[j, i] = c;
            }
        }
        return cost;
    }

    static double TransferProxy(Node a, Node b, double mu)
    {
        double ra = Math.Max(a.Rmag, 1.0);
        double rb = Math.Max(b.Rmag, 1.0);
        double va = a.CircSpeed;
        double vb = b.CircSpeed;
        double vavg = 0.5 * (va + vb);

        double planeDot = Clamp(Vec3.Dot(a.Hhat, b.Hhat), -1.0, 1.0);
        double planeAngle = Math.Acos(planeDot);
        double planeCost = 2.0 * vavg * Math.Sin(0.5 * planeAngle);

        double transferA = 0.5 * (ra + rb);
        double h1 = Math.Sqrt(mu / ra) * (Math.Sqrt((2.0 * rb) / (ra + rb)) - 1.0);
        double h2 = Math.Sqrt(mu / rb) * (1.0 - Math.Sqrt((2.0 * ra) / (ra + rb)));
        double hohmann = Math.Abs(h1) + Math.Abs(h2);
        if (double.IsNaN(hohmann) || double.IsInfinity(hohmann)) hohmann = Math.Abs(va - vb);

        double eDiff = (a.Evec - b.Evec).Norm();
        double eccCost = 0.5 * vavg * eDiff;

        double radialRatio = Math.Abs(ra - rb) / Math.Max(1.0, 0.5 * (ra + rb));
        double speedRatio = Math.Abs(a.Vmag - b.Vmag) / Math.Max(1.0, vavg);

        double phase = Math.Abs(NormalizeAngle(a.Phase - b.Phase));
        double phasePenalty = 0.12 * vavg * phase;

        double energyGap = Math.Abs((1.0 / Math.Max(a.A, 1.0)) - (1.0 / Math.Max(b.A, 1.0))) * Math.Sqrt(mu * transferA);

        return hohmann
            + 1.35 * planeCost
            + eccCost
            + 0.35 * vavg * radialRatio
            + 0.20 * vavg * speedRatio
            + phasePenalty
            + 0.15 * energyGap;
    }

    static int[] Solve(double[,] cost, int n)
    {
        if (n <= 8) return SolveBruteForce(cost, n);
        if (n <= 15) return SolveHeldKarp(cost, n);

        int[] best = BuildGreedyPath(cost, n, startStation: -1);
        double bestScore = PathCost(best, cost);

        int candidates = n <= 30 ? Math.Min(n, 8) : Math.Min(n, 16);
        int[] seeds = Enumerable.Range(0, n)
            .OrderBy(i => cost[0, i + 1])
            .Take(candidates)
            .ToArray();

        foreach (int seed in seeds)
        {
            int[] cand = BuildGreedyPath(cost, n, seed);
            ImprovePath(cand, cost, n);
            double score = PathCost(cand, cost);
            if (score < bestScore)
            {
                bestScore = score;
                best = cand;
            }
        }

        ImprovePath(best, cost, n);
        return best;
    }

    static int[] SolveBruteForce(double[,] cost, int n)
    {
        int[] perm = Enumerable.Range(0, n).ToArray();
        int[] best = (int[])perm.Clone();
        bool[] used = new bool[n];
        int[] cur = new int[n];
        double bestScore = double.PositiveInfinity;

        void Dfs(int depth, int prev, double acc)
        {
            if (acc >= bestScore) return;
            if (depth == n)
            {
                bestScore = acc;
                Array.Copy(cur, best, n);
                return;
            }
            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                used[i] = true;
                cur[depth] = i;
                Dfs(depth + 1, i, acc + cost[prev, i + 1]);
                used[i] = false;
            }
        }

        Dfs(0, 0, 0.0);
        return best;
    }

    static int[] SolveHeldKarp(double[,] cost, int n)
    {
        int m = 1 << n;
        double[,] dp = new double[m, n];
        int[,] parent = new int[m, n];
        double inf = 1e100;

        for (int mask = 0; mask < m; mask++)
            for (int j = 0; j < n; j++)
            {
                dp[mask, j] = inf;
                parent[mask, j] = -1;
            }

        for (int j = 0; j < n; j++)
        {
            int mask = 1 << j;
            dp[mask, j] = cost[0, j + 1];
        }

        for (int mask = 1; mask < m; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0) continue;
                double baseCost = dp[mask, last];
                if (baseCost >= inf) continue;
                int rem = ((m - 1) ^ mask);
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int nxt = BitIndex(bit);
                    int nm = mask | bit;
                    double cand = baseCost + cost[last + 1, nxt + 1];
                    if (cand < dp[nm, nxt])
                    {
                        dp[nm, nxt] = cand;
                        parent[nm, nxt] = last;
                    }
                    rem ^= bit;
                }
            }
        }

        int full = m - 1;
        double bestScore = inf;
        int bestLast = 0;
        for (int j = 0; j < n; j++)
        {
            if (dp[full, j] < bestScore)
            {
                bestScore = dp[full, j];
                bestLast = j;
            }
        }

        int[] path = new int[n];
        int mask2 = full;
        int cur = bestLast;
        for (int idx = n - 1; idx >= 0; idx--)
        {
            path[idx] = cur;
            int p = parent[mask2, cur];
            mask2 ^= 1 << cur;
            cur = p;
        }
        return path;
    }

    static int BitIndex(int bit)
    {
        int idx = 0;
        while ((bit >>= 1) != 0) idx++;
        return idx;
    }

    static int[] BuildGreedyPath(double[,] cost, int n, int startStation)
    {
        bool[] used = new bool[n];
        int[] path = new int[n];
        int first;
        if (startStation >= 0)
        {
            first = startStation;
        }
        else
        {
            first = 0;
            double best = double.PositiveInfinity;
            for (int i = 0; i < n; i++)
            {
                double c = cost[0, i + 1];
                if (c < best)
                {
                    best = c;
                    first = i;
                }
            }
        }

        path[0] = first;
        used[first] = true;
        for (int pos = 1; pos < n; pos++)
        {
            int prev = path[pos - 1];
            int bestJ = -1;
            double bestCost = double.PositiveInfinity;
            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double c = cost[prev + 1, j + 1] + 0.08 * cost[0, j + 1];
                if (c < bestCost)
                {
                    bestCost = c;
                    bestJ = j;
                }
            }
            path[pos] = bestJ;
            used[bestJ] = true;
        }
        return path;
    }

    static void ImprovePath(int[] path, double[,] cost, int n)
    {
        TwoOpt(path, cost, n);
        Relocate(path, cost, n);
        TwoOpt(path, cost, n);
    }

    static void TwoOpt(int[] path, double[,] cost, int n)
    {
        int maxPasses = n <= 40 ? 12 : 6;
        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                int a = (i == 0) ? -1 : path[i - 1];
                int b = path[i];
                for (int j = i + 1; j < n; j++)
                {
                    int c = path[j];
                    int d = (j == n - 1) ? -1 : path[j + 1];

                    double oldCost = EdgeCost(a, b, cost) + EdgeCost(c, d, cost);
                    double newCost = EdgeCost(a, c, cost) + EdgeCost(b, d, cost);
                    if (newCost + 1e-9 < oldCost)
                    {
                        Array.Reverse(path, i, j - i + 1);
                        improved = true;
                    }
                }
            }
            if (!improved) break;
        }
    }

    static void Relocate(int[] path, double[,] cost, int n)
    {
        int passes = n <= 40 ? 8 : 4;
        for (int pass = 0; pass < passes; pass++)
        {
            bool improved = false;
            for (int i = 0; i < n; i++)
            {
                int node = path[i];
                double removeGain = EdgeCost(i == 0 ? -1 : path[i - 1], node, cost)
                    + EdgeCost(node, i == n - 1 ? -1 : path[i + 1], cost)
                    - EdgeCost(i == 0 ? -1 : path[i - 1], i == n - 1 ? -1 : path[i + 1], cost);

                int bestPos = i;
                double bestDelta = 0.0;
                for (int pos = 0; pos <= n; pos++)
                {
                    if (pos == i || pos == i + 1) continue;
                    int left = pos == 0 ? -1 : path[pos - 1];
                    int right = pos == n ? -1 : path[pos];
                    if (left == node || right == node) continue;
                    double insertCost = EdgeCost(left, node, cost)
                        + EdgeCost(node, right, cost)
                        - EdgeCost(left, right, cost);
                    double delta = insertCost - removeGain;
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestPos = pos;
                    }
                }

                if (bestPos != i)
                {
                    Move(path, i, bestPos);
                    improved = true;
                }
            }
            if (!improved) break;
        }
    }

    static void Move(int[] path, int from, int to)
    {
        int val = path[from];
        if (to > from)
        {
            for (int k = from; k < to - 1; k++) path[k] = path[k + 1];
            path[to - 1] = val;
        }
        else
        {
            for (int k = from; k > to; k--) path[k] = path[k - 1];
            path[to] = val;
        }
    }

    static double EdgeCost(int a, int b, double[,] cost)
    {
        if (b < 0) return 0.0;
        int ia = a < 0 ? 0 : a + 1;
        int ib = b + 1;
        return cost[ia, ib];
    }

    static double PathCost(int[] path, double[,] cost)
    {
        double total = 0.0;
        int prev = -1;
        for (int i = 0; i < path.Length; i++)
        {
            total += EdgeCost(prev, path[i], cost);
            prev = path[i];
        }
        return total;
    }

    static double NormalizeAngle(double a)
    {
        while (a <= -Math.PI) a += 2.0 * Math.PI;
        while (a > Math.PI) a -= 2.0 * Math.PI;
        return a;
    }

    static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }
}
