use std::collections::VecDeque;
use std::io::{self, Read};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input)?;
        Ok(Self { buf: input, idx: 0 })
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut value = 0usize;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            value = value.saturating_mul(10).saturating_add((b - b'0') as usize);
            self.idx += 1;
        }
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        Some(value)
    }
}

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

fn build_csr(n: usize, edges: &[Edge]) -> (Vec<usize>, Vec<usize>, Vec<usize>) {
    let mut deg = vec![0usize; n];
    for &Edge { u, v } in edges {
        deg[u] += 1;
        deg[v] += 1;
    }

    let mut offset = vec![0usize; n + 1];
    for i in 0..n {
        offset[i + 1] = offset[i] + deg[i];
    }

    let mut next = offset[..n].to_vec();
    let mut nbr = vec![0usize; edges.len() * 2];
    let mut eid = vec![0usize; edges.len() * 2];
    for (i, &Edge { u, v }) in edges.iter().enumerate() {
        let pu = next[u];
        nbr[pu] = v;
        eid[pu] = i;
        next[u] += 1;

        let pv = next[v];
        nbr[pv] = u;
        eid[pv] = i;
        next[v] += 1;
    }
    (offset, nbr, eid)
}

fn exact_vertex_cover_small(vertices: &[usize], edges: &[Edge]) -> Vec<usize> {
    let k = vertices.len();
    let mut local_index = vec![usize::MAX; vertices.iter().copied().max().unwrap_or(0) + 1];
    for (i, &v) in vertices.iter().enumerate() {
        local_index[v] = i;
    }

    let mut adj = vec![0u64; k];
    for &Edge { u, v } in edges {
        let a = local_index[u];
        let b = local_index[v];
        if a != usize::MAX && b != usize::MAX && a != b {
            adj[a] |= 1u64 << b;
            adj[b] |= 1u64 << a;
        }
    }

    fn greedy_lb(mask: u64, adj: &[u64]) -> usize {
        let mut tmp = mask;
        let mut lb = 0usize;
        while tmp != 0 {
            let u = tmp.trailing_zeros() as usize;
            let um = 1u64 << u;
            let nbrs = adj[u] & tmp;
            if nbrs != 0 {
                let v = nbrs.trailing_zeros() as usize;
                tmp &= !((1u64 << v) | um);
                lb += 1;
            } else {
                tmp &= !um;
            }
        }
        lb
    }

    fn find_edge(mask: u64, adj: &[u64]) -> Option<(usize, usize)> {
        let mut tmp = mask;
        while tmp != 0 {
            let u = tmp.trailing_zeros() as usize;
            let nbrs = adj[u] & mask;
            if nbrs != 0 {
                let v = nbrs.trailing_zeros() as usize;
                return Some((u, v));
            }
            tmp &= !(1u64 << u);
        }
        None
    }

    fn dfs(mask: u64, chosen: u64, used: usize, best: &mut usize, best_chosen: &mut u64, adj: &[u64]) {
        if used >= *best {
            return;
        }
        let lb = greedy_lb(mask, adj);
        if used + lb >= *best {
            return;
        }
        if let Some((u, v)) = find_edge(mask, adj) {
            dfs(mask & !(1u64 << u), chosen | (1u64 << u), used + 1, best, best_chosen, adj);
            dfs(mask & !(1u64 << v), chosen | (1u64 << v), used + 1, best, best_chosen, adj);
        } else {
            *best = used;
            *best_chosen = chosen;
        }
    }

    let full_mask = if k == 64 { u64::MAX } else { (1u64 << k) - 1 };
    let mut best = k + 1;
    let mut best_mask = 0u64;
    dfs(full_mask, 0, 0, &mut best, &mut best_mask, &adj);

    let mut out = Vec::with_capacity(best);
    for (i, &v) in vertices.iter().enumerate() {
        if ((best_mask >> i) & 1) != 0 {
            out.push(v);
        }
    }
    out
}

fn main() -> io::Result<()> {
    let mut sc = Scanner::new()?;
    let n = match sc.next_usize() {
        Some(v) => v,
        None => return Ok(()),
    };
    let m = sc.next_usize().unwrap_or(0);

    let mut cover = vec![false; n];
    let mut edges = Vec::with_capacity(m);

    for _ in 0..m {
        let u = match sc.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match sc.next_usize() {
            Some(vv) => vv,
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            cover[u] = true;
        } else {
            edges.push(Edge { u, v });
        }
    }

    if edges.is_empty() {
        let result: Vec<usize> = (0..n).filter(|&v| cover[v]).collect();
        println!("{}", result.len());
        if !result.is_empty() {
            for (i, v) in result.iter().enumerate() {
                if i > 0 {
                    print!(" ");
                }
                print!("{v}");
            }
            println!();
        } else {
            println!();
        }
        return Ok(());
    }

    let (offset, nbr, eid) = build_csr(n, &edges);
    let mut degree = vec![0usize; n];
    for v in 0..n {
        degree[v] = offset[v + 1] - offset[v];
    }
    let mut active_vertex = vec![true; n];
    let mut active_edge = vec![true; edges.len()];
    let mut q = VecDeque::new();
    for v in 0..n {
        if cover[v] {
            q.push_back(v);
        } else if degree[v] <= 1 {
            q.push_back(v);
        }
    }

    while let Some(v) = q.pop_front() {
        if !active_vertex[v] {
            continue;
        }
        if cover[v] {
            active_vertex[v] = false;
            for p in offset[v]..offset[v + 1] {
                let e = eid[p];
                if !active_edge[e] {
                    continue;
                }
                active_edge[e] = false;
                let w = nbr[p];
                if active_vertex[w] && degree[w] > 0 {
                    degree[w] -= 1;
                    if degree[w] <= 1 {
                        q.push_back(w);
                    }
                }
            }
            degree[v] = 0;
            continue;
        }

        if degree[v] == 0 {
            active_vertex[v] = false;
            continue;
        }

        if degree[v] == 1 {
            let mut only_neighbor = None;
            for p in offset[v]..offset[v + 1] {
                let e = eid[p];
                if active_edge[e] {
                    only_neighbor = Some(nbr[p]);
                    break;
                }
            }
            if let Some(u) = only_neighbor {
                if active_vertex[u] && !cover[u] {
                    cover[u] = true;
                    q.push_back(u);
                }
            }
        }
    }

    let mut rem_vertices = Vec::new();
    let mut in_rem = vec![false; n];
    for v in 0..n {
        if active_vertex[v] && !cover[v] && degree[v] > 0 {
            in_rem[v] = true;
            rem_vertices.push(v);
        }
    }

    let mut rem_edges = Vec::new();
    rem_edges.reserve(edges.len());
    for (i, &Edge { u, v }) in edges.iter().enumerate() {
        if active_edge[i] && in_rem[u] && in_rem[v] {
            rem_edges.push(Edge { u, v });
        }
    }

    if !rem_edges.is_empty() {
        if rem_vertices.len() <= 64 {
            for v in exact_vertex_cover_small(&rem_vertices, &rem_edges) {
                cover[v] = true;
            }
        } else {
            let mut matched = vec![false; n];
            for &Edge { u, v } in &rem_edges {
                if !matched[u] && !matched[v] {
                    matched[u] = true;
                    matched[v] = true;
                    cover[u] = true;
                    cover[v] = true;
                }
            }
            for &Edge { u, v } in &rem_edges {
                if !cover[u] && !cover[v] {
                    if degree[u] >= degree[v] {
                        cover[u] = true;
                    } else {
                        cover[v] = true;
                    }
                }
            }
        }
    }

    let mut changed = true;
    while changed {
        changed = false;
        for v in 0..n {
            if !cover[v] {
                continue;
            }
            let mut removable = true;
            for p in offset[v]..offset[v + 1] {
                let u = nbr[p];
                if !cover[u] {
                    removable = false;
                    break;
                }
            }
            if removable {
                cover[v] = false;
                changed = true;
            }
        }
    }

    let mut result = Vec::new();
    result.reserve(n.min(edges.len() * 2));
    for (v, &take) in cover.iter().enumerate() {
        if take {
            result.push(v);
        }
    }

    println!("{}", result.len());
    if !result.is_empty() {
        for (i, v) in result.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{v}");
        }
    }
    println!();
    Ok(())
}
