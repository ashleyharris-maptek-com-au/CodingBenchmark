import sys


def main():
    lines = [line.strip() for line in sys.stdin.read().splitlines() if line.strip()]
    if not lines:
        print("NO SOLUTION")
        return

    first = lines[0].split()
    if len(first) != 2:
        print("NO SOLUTION")
        return

    u, s = map(int, first)
    if len(lines) < s + 1:
        print("NO SOLUTION")
        return

    sets = []
    for i in range(s):
        parts = lines[i + 1].split()
        elems = []
        seen = set()
        valid = True
        for part in parts:
            try:
                x = int(part)
            except ValueError:
                valid = False
                break
            if x < 0 or x >= u or x in seen:
                valid = False
                break
            seen.add(x)
            elems.append(x)
        if not valid:
            print("NO SOLUTION")
            return
        sets.append(tuple(elems))

    elem_to_sets = [set() for _ in range(u)]
    for idx, subset in enumerate(sets):
        for elem in subset:
            elem_to_sets[elem].add(idx)

    active_sets = set(range(s))
    uncovered = set(range(u))
    chosen = []

    def search():
        if not uncovered:
            return True

        target = min(uncovered, key=lambda e: (len(elem_to_sets[e] & active_sets), e))
        candidates = sorted(elem_to_sets[target] & active_sets)
        if not candidates:
            return False

        for set_idx in candidates:
            subset = sets[set_idx]

            if any(elem not in uncovered for elem in subset):
                continue

            removed_sets = set()
            removed_elems = []

            for elem in subset:
                if elem in uncovered:
                    uncovered.remove(elem)
                    removed_elems.append(elem)
                removed_sets.update(elem_to_sets[elem] & active_sets)

            active_sets.difference_update(removed_sets)
            chosen.append(set_idx)

            if search():
                return True

            chosen.pop()
            active_sets.update(removed_sets)
            uncovered.update(removed_elems)

        return False

    if search():
        chosen.sort()
        print("SOLUTION")
        print(" ".join(map(str, chosen)))
    else:
        print("NO SOLUTION")


if __name__ == "__main__":
    main()
