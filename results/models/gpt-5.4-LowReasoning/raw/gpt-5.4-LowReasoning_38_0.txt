import sys


def parse_input():
    lines = sys.stdin.read().splitlines()
    if not lines:
        return 0, 0, []

    first = lines[0].split()
    if len(first) != 2:
        raise ValueError("Invalid header")

    u = int(first[0])
    s = int(first[1])

    sets = []
    for i in range(s):
        if i + 1 < len(lines):
            text = lines[i + 1].strip()
            if text:
                elems = [int(x) for x in text.split()]
            else:
                elems = []
        else:
            elems = []
        sets.append(elems)

    return u, s, sets


def solve_exact_cover(u, sets):
    if u == 0:
        return []

    set_masks = []
    for elems in sets:
        mask = 0
        valid = True
        seen = set()
        for e in elems:
            if e < 0 or e >= u or e in seen:
                valid = False
                break
            seen.add(e)
            mask |= 1 << e
        set_masks.append(mask if valid else -1)

    elem_to_sets = [[] for _ in range(u)]
    for idx, mask in enumerate(set_masks):
        if mask < 0:
            continue
        m = mask
        while m:
            bit = m & -m
            elem = bit.bit_length() - 1
            elem_to_sets[elem].append(idx)
            m ^= bit

    full_mask = (1 << u) - 1
    chosen = []

    def search(covered_mask, used_mask):
        if covered_mask == full_mask:
            return True

        remaining = full_mask ^ covered_mask
        best_elem = -1
        best_candidates = None

        m = remaining
        while m:
            bit = m & -m
            elem = bit.bit_length() - 1
            candidates = []
            for idx in elem_to_sets[elem]:
                set_mask = set_masks[idx]
                if set_mask >= 0 and (set_mask & used_mask) == 0:
                    candidates.append(idx)
            if not candidates:
                return False
            if best_candidates is None or len(candidates) < len(best_candidates):
                best_elem = elem
                best_candidates = candidates
                if len(best_candidates) == 1:
                    break
            m ^= bit

        for idx in best_candidates:
            set_mask = set_masks[idx]
            chosen.append(idx)
            if search(covered_mask | set_mask, used_mask | set_mask):
                return True
            chosen.pop()

        return False

    if search(0, 0):
        return chosen[:]
    return None


def main():
    try:
        u, _s, sets = parse_input()
        solution = solve_exact_cover(u, sets)
    except Exception:
        solution = None

    out = sys.stdout
    if solution is None:
        out.write("NO SOLUTION\n")
    else:
        out.write("SOLUTION\n")
        out.write(" ".join(str(i) for i in solution) + "\n")


if __name__ == "__main__":
    main()
