type Cell = [number, number];

declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Town = { id: string; x: number; y: number; value: number };

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = data.grid;
const ignitions: Cell[] = data.ignitions || [];
const towns: Town[] = data.towns || [];
const cutBudget: number = Math.max(0, data.cutBudget | 0);

const cellCount = width * height;
const dirs = [1, 0, -1, 0, 1];

function idx(x: number, y: number): number {
  return y * width + x;
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

const burnable = new Uint8Array(cellCount);
for (let y = 0; y < height; y++) {
  const row = grid[y] || "";
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) burnable[idx(x, y)] = 1; // '.'
  }
}

const ignitionMark = new Uint8Array(cellCount);
for (const [x, y] of ignitions) {
  if (inBounds(x, y)) ignitionMark[idx(x, y)] = 1;
}

const townAt = new Int32Array(cellCount);
townAt.fill(-1);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (inBounds(t.x, t.y)) townAt[idx(t.x, t.y)] = i;
}

function simulate(cutMark: Uint8Array, needParents: boolean) {
  const visited = new Uint8Array(cellCount);
  const dist = needParents ? new Int32Array(cellCount) : null;
  const parent = needParents ? new Int32Array(cellCount) : null;
  if (dist) dist.fill(-1);
  if (parent) parent.fill(-1);

  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;

  for (const [x, y] of ignitions) {
    if (!inBounds(x, y)) continue;
    const start = idx(x, y);
    if (!burnable[start] || cutMark[start]) continue;
    if (visited[start]) continue;
    visited[start] = 1;
    if (dist) dist[start] = 0;
    queue[tail++] = start;
  }

  while (head < tail) {
    const cur = queue[head++];
    const cx = cur % width;
    const cy = (cur / width) | 0;
    for (let k = 0; k < 4; k++) {
      const nx = cx + dirs[k];
      const ny = cy + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (!burnable[ni] || cutMark[ni] || visited[ni]) continue;
      visited[ni] = 1;
      if (dist) dist[ni] = (dist[cur] + 1) | 0;
      if (parent) parent[ni] = cur;
      queue[tail++] = ni;
    }
  }

  let saved = 0;
  let burned = 0;
  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    if (!inBounds(t.x, t.y)) continue;
    const ti = idx(t.x, t.y);
    if (!burnable[ti] || cutMark[ti] || !visited[ti]) saved += t.value;
    else burned += t.value;
  }

  return { visited, dist, parent, saved, burned };
}

function collectCandidates(
  cutMark: Uint8Array,
  visited: Uint8Array,
  dist: Int32Array,
  parent: Int32Array
): number[] {
  const score = new Float64Array(cellCount);
  const threatened: { idx: number; value: number; dist: number }[] = [];

  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    if (!inBounds(t.x, t.y)) continue;
    const ti = idx(t.x, t.y);
    if (!burnable[ti] || cutMark[ti] || !visited[ti]) continue;
    threatened.push({ idx: ti, value: t.value, dist: dist[ti] });
  }

  threatened.sort((a, b) => {
    if (b.value !== a.value) return b.value - a.value;
    return a.dist - b.dist;
  });

  const sampleLimit = threatened.length <= 200 ? threatened.length : 200;
  for (let s = 0; s < sampleLimit; s++) {
    const item = threatened[s];
    let cur = item.idx;
    let steps = 0;
    while (cur >= 0) {
      if (!ignitionMark[cur] && townAt[cur] < 0 && !cutMark[cur]) {
        score[cur] += item.value / (1 + steps * 0.35);
      }
      cur = parent[cur];
      steps++;
    }
  }

  for (let i = 0; i < cellCount; i++) {
    if (!visited[i] || cutMark[i] || !burnable[i] || ignitionMark[i] || townAt[i] >= 0) continue;
    let blockedSides = 0;
    const x = i % width;
    const y = (i / width) | 0;
    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k];
      const ny = y + dirs[k + 1];
      if (!inBounds(nx, ny)) {
        blockedSides++;
        continue;
      }
      const ni = idx(nx, ny);
      if (!burnable[ni] || cutMark[ni]) blockedSides++;
    }
    if (blockedSides > 0 && score[i] > 0) {
      score[i] *= 1 + blockedSides * 0.2;
    }
  }

  const all: { i: number; s: number }[] = [];
  for (let i = 0; i < cellCount; i++) {
    if (score[i] > 0) all.push({ i, s: score[i] });
  }
  all.sort((a, b) => b.s - a.s);

  const limit = Math.min(all.length, Math.max(60, Math.min(500, 40 + cutBudget * 20)));
  const result: number[] = [];
  for (let i = 0; i < limit; i++) result.push(all[i].i);

  if (result.length === 0) {
    for (const item of threatened.slice(0, Math.min(threatened.length, 40))) {
      let cur = parent[item.idx];
      let taken = 0;
      while (cur >= 0 && taken < 3) {
        if (!ignitionMark[cur] && townAt[cur] < 0 && !cutMark[cur]) {
          result.push(cur);
          taken++;
        }
        cur = parent[cur];
      }
    }
  }

  const seen = new Uint8Array(cellCount);
  const dedup: number[] = [];
  for (const c of result) {
    if (!seen[c]) {
      seen[c] = 1;
      dedup.push(c);
    }
  }
  return dedup;
}

const cutMark = new Uint8Array(cellCount);
const chosen: number[] = [];

let current = simulate(cutMark, true);
if (cutBudget > 0 && current.burned > 0) {
  while (chosen.length < cutBudget) {
    current = simulate(cutMark, true);
    if (current.burned <= 0 || !current.dist || !current.parent) break;

    const candidates = collectCandidates(cutMark, current.visited, current.dist, current.parent);
    if (candidates.length === 0) break;

    let bestGain = 0;
    let bestSet: number[] = [];

    const baselineSaved = current.saved;

    const singleLimit = Math.min(candidates.length, 80);
    for (let i = 0; i < singleLimit; i++) {
      const c = candidates[i];
      cutMark[c] = 1;
      const sim = simulate(cutMark, false);
      cutMark[c] = 0;
      const gain = sim.saved - baselineSaved;
      if (gain > bestGain) {
        bestGain = gain;
        bestSet = [c];
      }
    }

    const remaining = cutBudget - chosen.length;
    if (remaining >= 2) {
      const pairLimit = Math.min(candidates.length, 18);
      for (let i = 0; i < pairLimit; i++) {
        const a = candidates[i];
        cutMark[a] = 1;
        for (let j = i + 1; j < pairLimit; j++) {
          const b = candidates[j];
          cutMark[b] = 1;
          const sim = simulate(cutMark, false);
          cutMark[b] = 0;
          const gain = sim.saved - baselineSaved;
          if (gain > bestGain) {
            bestGain = gain;
            bestSet = [a, b];
          }
        }
        cutMark[a] = 0;
      }
    }

    if (bestGain <= 0 || bestSet.length === 0) break;
    for (const c of bestSet) {
      if (!cutMark[c] && chosen.length < cutBudget) {
        cutMark[c] = 1;
        chosen.push(c);
      }
    }
  }
}

const firebreaks: Cell[] = chosen.map((i) => [i % width, (i / width) | 0]);
process.stdout.write(JSON.stringify({ firebreaks }));
