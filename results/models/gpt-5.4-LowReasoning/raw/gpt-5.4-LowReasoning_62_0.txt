declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Point = [number, number];
type Town = { id: string; x: number; y: number; value: number };

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = data.grid;
const ignitions: Point[] = data.ignitions || [];
const towns: Town[] = data.towns || [];
const cutBudget: number = Math.max(0, data.cutBudget | 0);

const n = width * height;
const burnable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) burnable[y * width + x] = 1;
  }
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function idxOf(x: number, y: number): number {
  return y * width + x;
}

function xOf(idx: number): number {
  return idx % width;
}

function yOf(idx: number): number {
  return (idx / width) | 0;
}

const ignitionIdx: number[] = [];
for (const p of ignitions) {
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (inBounds(x, y)) {
    const idx = idxOf(x, y);
    if (burnable[idx]) ignitionIdx.push(idx);
  }
}

const townIdx = new Int32Array(towns.length);
const townValue = new Float64Array(towns.length);
for (let i = 0; i < towns.length; i++) {
  townIdx[i] = idxOf(towns[i].x | 0, towns[i].y | 0);
  townValue[i] = Number(towns[i].value) || 0;
}

const cut = new Uint8Array(n);
const visitMark = new Int32Array(n);
let visitToken = 1;
const queue = new Int32Array(n);

function nextToken(): number {
  visitToken++;
  if (visitToken === 0x7fffffff) {
    visitMark.fill(0);
    visitToken = 1;
  }
  return visitToken;
}

function computeReachable(collectDist: boolean, dist?: Int32Array): number {
  const token = nextToken();
  let head = 0;
  let tail = 0;
  if (collectDist && dist) dist.fill(-1);

  for (let i = 0; i < ignitionIdx.length; i++) {
    const s = ignitionIdx[i];
    if (!burnable[s] || cut[s] || visitMark[s] === token) continue;
    visitMark[s] = token;
    if (collectDist && dist) dist[s] = 0;
    queue[tail++] = s;
  }

  while (head < tail) {
    const v = queue[head++];
    const d = collectDist && dist ? dist[v] : 0;
    const x = v % width;

    let u = v - width;
    if (u >= 0 && burnable[u] && !cut[u] && visitMark[u] !== token) {
      visitMark[u] = token;
      if (collectDist && dist) dist[u] = d + 1;
      queue[tail++] = u;
    }
    u = v + width;
    if (u < n && burnable[u] && !cut[u] && visitMark[u] !== token) {
      visitMark[u] = token;
      if (collectDist && dist) dist[u] = d + 1;
      queue[tail++] = u;
    }
    if (x > 0) {
      u = v - 1;
      if (burnable[u] && !cut[u] && visitMark[u] !== token) {
        visitMark[u] = token;
        if (collectDist && dist) dist[u] = d + 1;
        queue[tail++] = u;
      }
    }
    if (x + 1 < width) {
      u = v + 1;
      if (burnable[u] && !cut[u] && visitMark[u] !== token) {
        visitMark[u] = token;
        if (collectDist && dist) dist[u] = d + 1;
        queue[tail++] = u;
      }
    }
  }
  return token;
}

function savedValueFromToken(token: number): number {
  let saved = 0;
  for (let i = 0; i < towns.length; i++) {
    const idx = townIdx[i];
    if (!inBounds(towns[i].x | 0, towns[i].y | 0)) continue;
    if (!burnable[idx] || cut[idx] || visitMark[idx] !== token) saved += townValue[i];
  }
  return saved;
}

const baselineToken = computeReachable(false);
let baselineBurnedValue = 0;
for (let i = 0; i < towns.length; i++) {
  const idx = townIdx[i];
  if (inBounds(towns[i].x | 0, towns[i].y | 0) && burnable[idx] && visitMark[idx] === baselineToken) {
    baselineBurnedValue += townValue[i];
  }
}

if (cutBudget === 0 || ignitionIdx.length === 0 || baselineBurnedValue <= 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const dist = new Int32Array(n);
const score = new Float64Array(n);
const seenCandidate = new Uint8Array(n);
const candidates: number[] = [];

function addCandidate(idx: number, weight: number): void {
  if (!burnable[idx] || cut[idx]) return;
  const isIgnition = ignitionIdx.indexOf(idx) >= 0;
  if (isIgnition) return;
  score[idx] += weight;
  if (!seenCandidate[idx]) {
    seenCandidate[idx] = 1;
    candidates.push(idx);
  }
}

function traceTownPaths(reachableToken: number): void {
  score.fill(0);
  seenCandidate.fill(0);
  candidates.length = 0;
  computeReachable(true, dist);

  for (let i = 0; i < towns.length; i++) {
    const tIdx = townIdx[i];
    const val = townValue[i];
    if (!burnable[tIdx] || cut[tIdx] || visitMark[tIdx] !== reachableToken) continue;
    let cur = tIdx;
    let d = dist[cur];
    if (d <= 0) continue;

    const pathWeight = val / Math.max(1, d);
    while (d > 0) {
      const x = cur % width;
      let best = -1;
      let bestScore = -1;

      const up = cur - width;
      if (up >= 0 && dist[up] === d - 1) {
        const s = score[up];
        if (s > bestScore) {
          bestScore = s;
          best = up;
        }
      }
      const down = cur + width;
      if (down < n && dist[down] === d - 1) {
        const s = score[down];
        if (s > bestScore) {
          bestScore = s;
          best = down;
        }
      }
      if (x > 0) {
        const left = cur - 1;
        if (dist[left] === d - 1) {
          const s = score[left];
          if (s > bestScore) {
            bestScore = s;
            best = left;
          }
        }
      }
      if (x + 1 < width) {
        const right = cur + 1;
        if (dist[right] === d - 1) {
          const s = score[right];
          if (s > bestScore) {
            bestScore = s;
            best = right;
          }
        }
      }
      if (best < 0) break;
      if (dist[best] > 0) addCandidate(best, pathWeight + val * 0.05);
      cur = best;
      d--;
    }

    const tx = tIdx % width;
    const ty = (tIdx / width) | 0;
    const neighbors = [
      ty > 0 ? tIdx - width : -1,
      ty + 1 < height ? tIdx + width : -1,
      tx > 0 ? tIdx - 1 : -1,
      tx + 1 < width ? tIdx + 1 : -1,
    ];
    for (let k = 0; k < 4; k++) {
      const nb = neighbors[k];
      if (nb >= 0 && burnable[nb] && !cut[nb]) addCandidate(nb, val * 0.2);
    }
  }
}

function addFrontierCandidates(reachableToken: number): void {
  for (let v = 0; v < n; v++) {
    if (visitMark[v] !== reachableToken || !burnable[v] || cut[v]) continue;
    const x = v % width;
    let blockedSides = 0;
    let fireSide = 0;

    const up = v - width;
    if (up < 0 || !burnable[up] || cut[up]) blockedSides++;
    else if (visitMark[up] === reachableToken) fireSide++;

    const down = v + width;
    if (down >= n || !burnable[down] || cut[down]) blockedSides++;
    else if (visitMark[down] === reachableToken) fireSide++;

    if (x === 0 || !burnable[v - 1] || cut[v - 1]) blockedSides++;
    else if (visitMark[v - 1] === reachableToken) fireSide++;

    if (x + 1 >= width || !burnable[v + 1] || cut[v + 1]) blockedSides++;
    else if (visitMark[v + 1] === reachableToken) fireSide++;

    if (blockedSides >= 2 && fireSide >= 1) {
      addCandidate(v, 0.5 + blockedSides * 0.35);
    }
  }
}

function sortTopCandidates(limit: number): number[] {
  candidates.sort((a, b) => {
    const diff = score[b] - score[a];
    return diff !== 0 ? diff : a - b;
  });
  if (candidates.length > limit) candidates.length = limit;
  return candidates;
}

function evaluateCutGain(cell: number, currentSaved: number): number {
  if (!burnable[cell] || cut[cell]) return -1;
  cut[cell] = 1;
  const token = computeReachable(false);
  const saved = savedValueFromToken(token);
  cut[cell] = 0;
  return saved - currentSaved;
}

const chosen: number[] = [];
let currentToken = baselineToken;
let currentSaved = 0;

for (let step = 0; step < cutBudget; step++) {
  traceTownPaths(currentToken);
  addFrontierCandidates(currentToken);
  const candidateLimit = n <= 20000 ? 160 : n <= 100000 ? 96 : 64;
  const evalLimit = n <= 20000 ? 56 : n <= 100000 ? 40 : 28;
  const pool = sortTopCandidates(candidateLimit);

  let bestCell = -1;
  let bestGain = 0;
  const evaluateCount = Math.min(evalLimit, pool.length);
  for (let i = 0; i < evaluateCount; i++) {
    const cell = pool[i];
    const gain = evaluateCutGain(cell, currentSaved);
    if (gain > bestGain) {
      bestGain = gain;
      bestCell = cell;
    }
  }

  if (bestCell < 0 || bestGain <= 0) break;
  cut[bestCell] = 1;
  chosen.push(bestCell);
  currentToken = computeReachable(false);
  currentSaved = savedValueFromToken(currentToken);
  if (currentSaved >= baselineBurnedValue) break;
}

const firebreaks: Point[] = chosen.map((idx) => [xOf(idx), yOf(idx)]);
process.stdout.write(JSON.stringify({ firebreaks }));
