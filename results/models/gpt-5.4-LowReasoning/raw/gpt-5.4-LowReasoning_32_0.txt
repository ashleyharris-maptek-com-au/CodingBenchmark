using namespace std;

struct Edge {
    int to;
    long long w;
    int id;
};

struct InputEdge {
    int u;
    int v;
    long long w;
};

static constexpr long long INF = numeric_limits<long long>::max() / 4;

struct DijkstraResult {
    vector<long long> dist;
    vector<int> parent_vertex;
    vector<int> parent_edge;
};

struct PathInfo {
    long long dist = INF;
    vector<int> vertices;
};

static long long edge_key(int a, int b) {
    if (a > b) swap(a, b);
    return (static_cast<long long>(a) << 32) ^ static_cast<unsigned int>(b);
}

static DijkstraResult run_dijkstra(int src, const vector<vector<Edge>>& graph) {
    const int n = static_cast<int>(graph.size());
    vector<long long> dist(n, INF);
    vector<int> parent_vertex(n, -1);
    vector<int> parent_edge(n, -1);
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [cd, u] = pq.top();
        pq.pop();
        if (cd != dist[u]) {
            continue;
        }
        for (const auto& e : graph[u]) {
            long long nd = cd + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parent_vertex[e.to] = u;
                parent_edge[e.to] = e.id;
                pq.push({nd, e.to});
            }
        }
    }
    return {move(dist), move(parent_vertex), move(parent_edge)};
}

static vector<int> restore_vertex_path(int src, int dst, const vector<int>& parent_vertex) {
    vector<int> rev;
    int cur = dst;
    while (cur != -1 && cur != src) {
        rev.push_back(cur);
        cur = parent_vertex[cur];
    }
    if (cur == -1) {
        return {};
    }
    rev.push_back(src);
    reverse(rev.begin(), rev.end());
    return rev;
}

static void add_path_edges_from_parents(
    int src,
    int dst,
    const vector<int>& parent_vertex,
    const unordered_map<long long, int>& edge_lookup,
    vector<char>& used_edges
) {
    int cur = dst;
    while (cur != src) {
        int p = parent_vertex[cur];
        if (p == -1) {
            return;
        }
        auto it = edge_lookup.find(edge_key(cur, p));
        if (it != edge_lookup.end()) {
            used_edges[it->second] = 1;
        }
        cur = p;
    }
}

static pair<long long, vector<int>> build_tree_from_used_edges(
    int n,
    const vector<InputEdge>& edges,
    const vector<char>& used_edges,
    const vector<int>& terminals
) {
    vector<vector<pair<int, int>>> tree_adj(n);
    long long total_weight = 0;
    vector<int> selected;
    selected.reserve(edges.size());
    for (int id = 0; id < static_cast<int>(edges.size()); ++id) {
        if (!used_edges[id]) {
            continue;
        }
        const auto& e = edges[id];
        total_weight += e.w;
        selected.push_back(id);
        tree_adj[e.u].push_back({e.v, id});
        tree_adj[e.v].push_back({e.u, id});
    }

    if (selected.empty()) {
        return {0, {}};
    }

    vector<char> is_terminal(n, 0);
    for (int t : terminals) {
        is_terminal[t] = 1;
    }

    vector<int> degree(n, 0);
    for (int v = 0; v < n; ++v) {
        degree[v] = static_cast<int>(tree_adj[v].size());
    }

    queue<int> q;
    vector<char> removed_vertex(n, 0);
    vector<char> removed_edge(edges.size(), 0);
    for (int v = 0; v < n; ++v) {
        if (!is_terminal[v] && degree[v] <= 1) {
            q.push(v);
        }
    }

    while (!q.empty()) {
        int v = q.front();
        q.pop();
        if (removed_vertex[v] || is_terminal[v] || degree[v] > 1) {
            continue;
        }
        removed_vertex[v] = 1;
        for (auto [to, id] : tree_adj[v]) {
            if (removed_edge[id]) {
                continue;
            }
            removed_edge[id] = 1;
            --degree[to];
            if (!is_terminal[to] && degree[to] <= 1) {
                q.push(to);
            }
        }
    }

    vector<int> pruned;
    pruned.reserve(selected.size());
    long long pruned_weight = 0;
    for (int id : selected) {
        if (!removed_edge[id]) {
            pruned.push_back(id);
            pruned_weight += edges[id].w;
        }
    }
    return {pruned_weight, move(pruned)};
}

static pair<long long, vector<int>> heuristic_steiner(
    int n,
    const vector<vector<Edge>>& graph,
    const vector<InputEdge>& edges,
    const vector<int>& terminals,
    const unordered_map<long long, int>& edge_lookup
) {
    const int k = static_cast<int>(terminals.size());
    vector<DijkstraResult> dij(k);
    for (int i = 0; i < k; ++i) {
        dij[i] = run_dijkstra(terminals[i], graph);
    }

    vector<vector<PathInfo>> metric(k, vector<PathInfo>(k));
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            metric[i][j].dist = metric[j][i].dist = dij[i].dist[terminals[j]];
            vector<int> path = restore_vertex_path(terminals[i], terminals[j], dij[i].parent_vertex);
            metric[i][j].vertices = path;
            vector<int> rev = path;
            reverse(rev.begin(), rev.end());
            metric[j][i].vertices = move(rev);
        }
    }

    vector<long long> best(k, INF);
    vector<int> parent(k, -1);
    vector<char> in_mst(k, 0);
    best[0] = 0;
    for (int it = 0; it < k; ++it) {
        int v = -1;
        for (int i = 0; i < k; ++i) {
            if (!in_mst[i] && (v == -1 || best[i] < best[v])) {
                v = i;
            }
        }
        if (v == -1 || best[v] >= INF) {
            break;
        }
        in_mst[v] = 1;
        for (int to = 0; to < k; ++to) {
            if (!in_mst[to] && metric[v][to].dist < best[to]) {
                best[to] = metric[v][to].dist;
                parent[to] = v;
            }
        }
    }

    vector<char> used_edges(edges.size(), 0);
    for (int v = 1; v < k; ++v) {
        if (parent[v] == -1) {
            continue;
        }
        const auto& path = metric[v][parent[v]].vertices;
        for (int i = 1; i < static_cast<int>(path.size()); ++i) {
            auto it = edge_lookup.find(edge_key(path[i - 1], path[i]));
            if (it != edge_lookup.end()) {
                used_edges[it->second] = 1;
            }
        }
    }
    return build_tree_from_used_edges(n, edges, used_edges, terminals);
}

struct ExactState {
    long long cost = INF;
    int split = -1;
    int prev_vertex = -1;
    bool via_relax = false;
};

static pair<long long, vector<int>> exact_steiner(
    int n,
    const vector<vector<Edge>>& graph,
    const vector<InputEdge>& edges,
    const vector<int>& terminals,
    const unordered_map<long long, int>& edge_lookup
) {
    const int k = static_cast<int>(terminals.size());
    const int full = 1 << k;

    vector<vector<ExactState>> dp(full, vector<ExactState>(n));
    for (int i = 0; i < k; ++i) {
        dp[1 << i][terminals[i]].cost = 0;
    }

    for (int mask = 1; mask < full; ++mask) {
        if ((mask & (mask - 1)) != 0) {
            int first_bit = mask & -mask;
            for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                if ((sub & first_bit) == 0) {
                    continue;
                }
                int other = mask ^ sub;
                if (other == 0) {
                    continue;
                }
                for (int v = 0; v < n; ++v) {
                    long long a = dp[sub][v].cost;
                    long long b = dp[other][v].cost;
                    if (a >= INF || b >= INF) {
                        continue;
                    }
                    long long cand = a + b;
                    if (cand < dp[mask][v].cost) {
                        dp[mask][v].cost = cand;
                        dp[mask][v].split = sub;
                        dp[mask][v].via_relax = false;
                        dp[mask][v].prev_vertex = -1;
                    }
                }
            }
        }

        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v].cost < INF) {
                pq.push({dp[mask][v].cost, v});
            }
        }
        while (!pq.empty()) {
            auto [cd, u] = pq.top();
            pq.pop();
            if (cd != dp[mask][u].cost) {
                continue;
            }
            for (const auto& e : graph[u]) {
                long long nd = cd + e.w;
                if (nd < dp[mask][e.to].cost) {
                    dp[mask][e.to].cost = nd;
                    dp[mask][e.to].split = -1;
                    dp[mask][e.to].via_relax = true;
                    dp[mask][e.to].prev_vertex = u;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    int all = full - 1;
    int root = -1;
    long long best = INF;
    for (int v = 0; v < n; ++v) {
        if (dp[all][v].cost < best) {
            best = dp[all][v].cost;
            root = v;
        }
    }

    vector<char> used_edges(edges.size(), 0);
    vector<pair<int, int>> stack;
    stack.push_back({all, root});
    while (!stack.empty()) {
        auto [mask, v] = stack.back();
        stack.pop_back();
        const auto& st = dp[mask][v];
        if (st.via_relax) {
            int p = st.prev_vertex;
            auto it = edge_lookup.find(edge_key(v, p));
            if (it != edge_lookup.end()) {
                used_edges[it->second] = 1;
            }
            stack.push_back({mask, p});
        } else if (st.split != -1) {
            int a = st.split;
            int b = mask ^ a;
            stack.push_back({a, v});
            stack.push_back({b, v});
        }
    }

    return build_tree_from_used_edges(n, edges, used_edges, terminals);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    vector<InputEdge> edges;
    edges.reserve(m);
    vector<vector<Edge>> graph(n);
    unordered_map<long long, int> edge_lookup;
    edge_lookup.reserve(static_cast<size_t>(m) * 2 + 1);

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges.push_back({u, v, w});
        graph[u].push_back({v, w, i});
        graph[v].push_back({u, w, i});

        long long key = edge_key(u, v);
        auto it = edge_lookup.find(key);
        if (it == edge_lookup.end() || w < edges[it->second].w) {
            edge_lookup[key] = i;
        }
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
    }

    sort(terminals.begin(), terminals.end());
    terminals.erase(unique(terminals.begin(), terminals.end()), terminals.end());
    k = static_cast<int>(terminals.size());

    if (k <= 1) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    pair<long long, vector<int>> ans;
    if (k <= 18) {
        ans = exact_steiner(n, graph, edges, terminals, edge_lookup);
    } else {
        ans = heuristic_steiner(n, graph, edges, terminals, edge_lookup);
    }

    cout << ans.first << ' ' << ans.second.size() << '\n';
    for (int id : ans.second) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }
    return 0;
}
