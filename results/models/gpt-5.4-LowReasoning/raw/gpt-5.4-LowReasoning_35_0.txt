use std::cmp::Reverse;
use std::collections::BinaryHeap;
use std::io::{self, Read};

#[derive(Clone)]
struct Graph {
    n: usize,
    adj: Vec<Vec<usize>>,
    closed: Vec<Vec<usize>>,
}

impl Graph {
    fn from_input(input: &str) -> Self {
        let mut it = input.split_whitespace();
        let n: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
        let m: usize = it.next().unwrap_or("0").parse().unwrap_or(0);

        let mut adj = vec![Vec::new(); n];
        for _ in 0..m {
            let Some(us) = it.next() else { break };
            let Some(vs) = it.next() else { break };
            let mut u: usize = us.parse().unwrap_or(0);
            let mut v: usize = vs.parse().unwrap_or(0);
            if n == 0 {
                continue;
            }
            if u >= n {
                u %= n;
            }
            if v >= n {
                v %= n;
            }
            if u == v {
                continue;
            }
            adj[u].push(v);
            adj[v].push(u);
        }

        for neighbors in &mut adj {
            neighbors.sort_unstable();
            neighbors.dedup();
        }

        let mut closed = vec![Vec::new(); n];
        for v in 0..n {
            let mut list = Vec::with_capacity(adj[v].len() + 1);
            list.push(v);
            list.extend_from_slice(&adj[v]);
            closed[v] = list;
        }

        Self { n, adj, closed }
    }
}

fn add_vertex(
    v: usize,
    graph: &Graph,
    chosen: &mut [bool],
    dominated: &mut [bool],
    undominated_count: &mut usize,
    cover_count: &mut [u32],
    solution: &mut Vec<usize>,
) {
    if chosen[v] {
        return;
    }
    chosen[v] = true;
    solution.push(v);

    for &u in &graph.closed[v] {
        if !dominated[u] {
            dominated[u] = true;
            *undominated_count -= 1;
        }
        for &w in &graph.closed[u] {
            cover_count[w] = cover_count[w].saturating_sub(1);
        }
    }
}

fn forced_reductions(graph: &Graph) -> Vec<usize> {
    let n = graph.n;
    let mut chosen = vec![false; n];
    let mut dominated = vec![false; n];
    let mut undominated_count = n;
    let mut cover_count = vec![0u32; n];
    let mut solution = Vec::new();

    for v in 0..n {
        cover_count[v] = graph.closed[v].len() as u32;
    }

    // Isolated vertices must be chosen.
    for v in 0..n {
        if graph.adj[v].is_empty() {
            add_vertex(
                v,
                graph,
                &mut chosen,
                &mut dominated,
                &mut undominated_count,
                &mut cover_count,
                &mut solution,
            );
        }
    }

    // If a vertex has exactly one dominator candidate remaining, that candidate is forced.
    let mut changed = true;
    while changed && undominated_count > 0 {
        changed = false;
        for u in 0..n {
            if dominated[u] {
                continue;
            }
            let mut only = usize::MAX;
            let mut cnt = 0usize;
            for &cand in &graph.closed[u] {
                if !chosen[cand] && cover_count[cand] > 0 {
                    only = cand;
                    cnt += 1;
                    if cnt > 1 {
                        break;
                    }
                }
            }
            if cnt == 1 {
                add_vertex(
                    only,
                    graph,
                    &mut chosen,
                    &mut dominated,
                    &mut undominated_count,
                    &mut cover_count,
                    &mut solution,
                );
                changed = true;
            }
        }
    }

    solution
}

fn greedy_dominating_set(graph: &Graph) -> Vec<usize> {
    let n = graph.n;
    if n == 0 {
        return Vec::new();
    }

    let forced = forced_reductions(graph);
    let mut chosen = vec![false; n];
    let mut dominated = vec![false; n];
    let mut undominated_count = n;
    let mut cover_count = vec![0u32; n];
    let mut solution = Vec::with_capacity(forced.len() + n / 4 + 1);

    for v in 0..n {
        cover_count[v] = graph.closed[v].len() as u32;
    }

    for &v in &forced {
        add_vertex(
            v,
            graph,
            &mut chosen,
            &mut dominated,
            &mut undominated_count,
            &mut cover_count,
            &mut solution,
        );
    }

    let mut heap = BinaryHeap::new();
    for v in 0..n {
        heap.push((cover_count[v], Reverse(graph.closed[v].len()), Reverse(v)));
    }

    while undominated_count > 0 {
        let mut picked = None;
        while let Some((gain, _, Reverse(v))) = heap.pop() {
            if chosen[v] {
                continue;
            }
            let current = cover_count[v];
            if gain != current {
                heap.push((current, Reverse(graph.closed[v].len()), Reverse(v)));
                continue;
            }
            picked = Some(v);
            break;
        }

        let v = if let Some(v) = picked {
            v
        } else {
            let mut fallback = None;
            for u in 0..n {
                if !dominated[u] {
                    fallback = Some(u);
                    break;
                }
            }
            fallback.unwrap_or(0)
        };

        add_vertex(
            v,
            graph,
            &mut chosen,
            &mut dominated,
            &mut undominated_count,
            &mut cover_count,
            &mut solution,
        );

        for &u in &graph.closed[v] {
            for &w in &graph.closed[u] {
                if !chosen[w] {
                    heap.push((cover_count[w], Reverse(graph.closed[w].len()), Reverse(w)));
                }
            }
        }
    }

    solution
}

fn prune_solution(graph: &Graph, solution: &mut Vec<usize>) {
    if solution.is_empty() {
        return;
    }

    let n = graph.n;
    let mut in_solution = vec![false; n];
    let mut dominated_by = vec![0u32; n];

    for &v in solution.iter() {
        in_solution[v] = true;
        for &u in &graph.closed[v] {
            dominated_by[u] += 1;
        }
    }

    solution.sort_unstable_by_key(|&v| graph.closed[v].len());
    solution.reverse();

    let mut kept = Vec::with_capacity(solution.len());
    for &v in solution.iter() {
        let removable = graph.closed[v]
            .iter()
            .all(|&u| dominated_by[u] >= 2);

        if removable {
            in_solution[v] = false;
            for &u in &graph.closed[v] {
                dominated_by[u] -= 1;
            }
        } else {
            kept.push(v);
        }
    }

    kept.sort_unstable();
    *solution = kept;
}

fn improve_by_redundant_swap(graph: &Graph, solution: &mut Vec<usize>) {
    let n = graph.n;
    if solution.len() < 2 || n == 0 {
        return;
    }

    let mut in_solution = vec![false; n];
    let mut dominated_by = vec![0u32; n];
    for &v in solution.iter() {
        in_solution[v] = true;
        for &u in &graph.closed[v] {
            dominated_by[u] += 1;
        }
    }

    let current = solution.clone();
    for &v in &current {
        let uniquely_covered: Vec<usize> = graph.closed[v]
            .iter()
            .copied()
            .filter(|&u| dominated_by[u] == 1)
            .collect();

        if uniquely_covered.is_empty() {
            continue;
        }

        let mut candidate = None;
        let mut best_gain = 0usize;
        for &w in &uniquely_covered {
            for &x in &graph.closed[w] {
                if in_solution[x] || x == v {
                    continue;
                }
                let mut covers_all = true;
                let mut gain = 0usize;
                for &u in &uniquely_covered {
                    if !graph.closed[x].contains(&u) {
                        covers_all = false;
                        break;
                    }
                }
                if !covers_all {
                    continue;
                }
                for &u in &graph.closed[x] {
                    if dominated_by[u] == 0 {
                        gain += 1;
                    }
                }
                if candidate.is_none() || gain > best_gain {
                    candidate = Some(x);
                    best_gain = gain;
                }
            }
        }

        if let Some(x) = candidate {
            in_solution[v] = false;
            for &u in &graph.closed[v] {
                dominated_by[u] -= 1;
            }

            in_solution[x] = true;
            for &u in &graph.closed[x] {
                dominated_by[u] += 1;
            }

            if dominated_by.iter().all(|&c| c > 0) {
                if let Some(pos) = solution.iter().position(|&y| y == v) {
                    solution[pos] = x;
                }
            } else {
                in_solution[x] = false;
                for &u in &graph.closed[x] {
                    dominated_by[u] -= 1;
                }
                in_solution[v] = true;
                for &u in &graph.closed[v] {
                    dominated_by[u] += 1;
                }
            }
        }
    }

    solution.sort_unstable();
    solution.dedup();
}

fn is_valid_dominating_set(graph: &Graph, solution: &[usize]) -> bool {
    let mut dominated = vec![false; graph.n];
    for &v in solution {
        if v >= graph.n {
            return false;
        }
        for &u in &graph.closed[v] {
            dominated[u] = true;
        }
    }
    dominated.into_iter().all(|x| x)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let graph = Graph::from_input(&input);

    let mut solution = greedy_dominating_set(&graph);
    prune_solution(&graph, &mut solution);
    improve_by_redundant_swap(&graph, &mut solution);
    prune_solution(&graph, &mut solution);

    if !is_valid_dominating_set(&graph, &solution) {
        solution.clear();
        for v in 0..graph.n {
            solution.push(v);
        }
    }

    println!("{}", solution.len());
    if solution.is_empty() {
        println!();
    } else {
        for (i, v) in solution.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{v}");
        }
        println!();
    }
}
