use std::cmp::Reverse;
use std::collections::BinaryHeap;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap_or(0),
        None => {
            println!("0");
            println!();
            return;
        }
    };
    let m: usize = it.next().and_then(|x| x.parse().ok()).unwrap_or(0);

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u = match it.next() {
            Some(x) => x.parse::<usize>().unwrap_or(0),
            None => break,
        };
        let v = match it.next() {
            Some(x) => x.parse::<usize>().unwrap_or(0),
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            adj[u].push(v);
        } else {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    for nbrs in &mut adj {
        nbrs.sort_unstable();
        nbrs.dedup();
    }

    let solution = minimum_dominating_set_heuristic(&adj);

    println!("{}", solution.len());
    if solution.is_empty() {
        println!();
    } else {
        for (i, v) in solution.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    }
}

fn minimum_dominating_set_heuristic(adj: &[Vec<usize>]) -> Vec<usize> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }

    let closed = build_closed_neighborhoods(adj);
    let mut selected = vec![false; n];
    let mut dominated = vec![false; n];
    let mut undominated_count = n;

    for v in 0..n {
        if adj[v].is_empty() {
            add_vertex(v, &closed, &mut selected, &mut dominated, &mut undominated_count);
        }
    }

    let mut heap = BinaryHeap::<(usize, Reverse<usize>)>::new();
    for v in 0..n {
        let gain = count_undominated(&closed[v], &dominated);
        heap.push((gain, Reverse(v)));
    }

    while undominated_count > 0 {
        let mut chosen = None;
        while let Some((gain_snapshot, Reverse(v))) = heap.pop() {
            if selected[v] {
                continue;
            }
            let actual_gain = count_undominated(&closed[v], &dominated);
            if actual_gain != gain_snapshot {
                heap.push((actual_gain, Reverse(v)));
                continue;
            }
            chosen = Some(v);
            break;
        }

        let v = if let Some(v) = chosen {
            v
        } else {
            let mut best_v = 0;
            let mut best_gain = 0;
            let mut best_deg = 0;
            for u in 0..n {
                if selected[u] {
                    continue;
                }
                let gain = count_undominated(&closed[u], &dominated);
                let deg = closed[u].len();
                if gain > best_gain || (gain == best_gain && deg > best_deg) {
                    best_gain = gain;
                    best_deg = deg;
                    best_v = u;
                }
            }
            best_v
        };

        add_vertex(v, &closed, &mut selected, &mut dominated, &mut undominated_count);
    }

    prune_redundant(&closed, &mut selected);

    let mut answer: Vec<usize> = (0..n).filter(|&v| selected[v]).collect();
    answer.sort_unstable();
    answer
}

fn build_closed_neighborhoods(adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let n = adj.len();
    let mut closed = vec![Vec::new(); n];
    for v in 0..n {
        let mut list = Vec::with_capacity(adj[v].len() + 1);
        list.push(v);
        list.extend_from_slice(&adj[v]);
        list.sort_unstable();
        list.dedup();
        closed[v] = list;
    }
    closed
}

fn count_undominated(neighborhood: &[usize], dominated: &[bool]) -> usize {
    neighborhood.iter().filter(|&&u| !dominated[u]).count()
}

fn add_vertex(
    v: usize,
    closed: &[Vec<usize>],
    selected: &mut [bool],
    dominated: &mut [bool],
    undominated_count: &mut usize,
) {
    if selected[v] {
        return;
    }
    selected[v] = true;
    for &u in &closed[v] {
        if !dominated[u] {
            dominated[u] = true;
            *undominated_count -= 1;
        }
    }
}

fn prune_redundant(closed: &[Vec<usize>], selected: &mut [bool]) {
    let n = selected.len();
    let mut cover_count = vec![0usize; n];
    for v in 0..n {
        if selected[v] {
            for &u in &closed[v] {
                cover_count[u] += 1;
            }
        }
    }

    let mut order: Vec<usize> = (0..n).filter(|&v| selected[v]).collect();
    order.sort_unstable_by_key(|&v| (closed[v].len(), v));

    for v in order {
        if !selected[v] {
            continue;
        }
        let removable = closed[v].iter().all(|&u| cover_count[u] >= 2);
        if removable {
            selected[v] = false;
            for &u in &closed[v] {
                cover_count[u] -= 1;
            }
        }
    }
}
