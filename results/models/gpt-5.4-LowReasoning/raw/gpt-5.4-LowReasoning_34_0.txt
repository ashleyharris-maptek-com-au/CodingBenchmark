import math
import sys


def solve():
    data = sys.stdin.buffer.read().split()
    if not data:
        return

    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = int(next(it))

    depot = (float(next(it)), float(next(it)))
    customers = []
    demands = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        customers.append((x, y))
        demands.append(d)

    if n == 0:
        sys.stdout.write("0.0\n0\n")
        return

    if any(d > capacity for d in demands):
        # No valid solution exists if a single customer exceeds vehicle capacity.
        sys.stdout.write("0.0\n0\n")
        return

    points = [depot] + customers
    dist = [[0.0] * (n + 1) for _ in range(n + 1)]
    for i in range(n + 1):
        xi, yi = points[i]
        for j in range(i + 1, n + 1):
            xj, yj = points[j]
            d = math.hypot(xi - xj, yi - yj)
            dist[i][j] = d
            dist[j][i] = d

    routes = {i: [i] for i in range(n)}
    route_load = {i: demands[i] for i in range(n)}
    belongs = list(range(n))
    active = set(range(n))

    savings = []
    for i in range(n):
        for j in range(i + 1, n):
            s = dist[0][i + 1] + dist[0][j + 1] - dist[i + 1][j + 1]
            savings.append((s, i, j))
    savings.sort(reverse=True)

    def try_merge(route_a_id, route_b_id):
        if route_a_id == route_b_id:
            return False
        if route_a_id not in active or route_b_id not in active:
            return False

        ra = routes[route_a_id]
        rb = routes[route_b_id]
        if route_load[route_a_id] + route_load[route_b_id] > capacity:
            return False

        candidates = []
        if ra[-1] == ra[-1] and rb[0] == rb[0]:
            candidates.append(ra + rb)
        candidates.append(ra + list(reversed(rb)))
        candidates.append(list(reversed(ra)) + rb)
        candidates.append(list(reversed(ra)) + list(reversed(rb)))

        best = None
        best_cost = float("inf")
        for merged in candidates:
            cost = route_cost(merged, dist)
            if cost < best_cost:
                best_cost = cost
                best = merged

        routes[route_a_id] = best
        route_load[route_a_id] += route_load[route_b_id]
        for c in routes[route_b_id]:
            belongs[c] = route_a_id
        active.remove(route_b_id)
        del routes[route_b_id]
        del route_load[route_b_id]
        return True

    for _, i, j in savings:
        ri = belongs[i]
        rj = belongs[j]
        if ri == rj:
            continue

        a = routes[ri]
        b = routes[rj]
        if i not in (a[0], a[-1]) or j not in (b[0], b[-1]):
            continue

        opts = []
        if a[-1] == i and b[0] == j:
            opts.append((a + b, ri, rj))
        if a[0] == i and b[-1] == j:
            opts.append((b + a, rj, ri))
        if a[0] == i and b[0] == j:
            opts.append((list(reversed(a)) + b, ri, rj))
        if a[-1] == i and b[-1] == j:
            opts.append((a + list(reversed(b)), ri, rj))

        best = None
        best_cost = float("inf")
        for merged, keep_id, drop_id in opts:
            if route_load[keep_id] + route_load[drop_id] > capacity:
                continue
            cost = route_cost(merged, dist)
            if cost < best_cost:
                best_cost = cost
                best = (merged, keep_id, drop_id)

        if best is None:
            continue

        merged, keep_id, drop_id = best
        routes[keep_id] = merged
        route_load[keep_id] += route_load[drop_id]
        for c in routes[drop_id]:
            belongs[c] = keep_id
        active.remove(drop_id)
        del routes[drop_id]
        del route_load[drop_id]

    route_list = [routes[rid][:] for rid in active]

    improved = []
    for route in route_list:
        improved.append(two_opt(route, dist))
    route_list = improved

    route_list = reduce_route_count(route_list, demands, capacity, k, dist)

    total = 0.0
    for route in route_list:
        total += route_cost(route, dist)

    out = [f"{total:.6f}", str(len(route_list))]
    for route in route_list:
        out.append(" ".join(map(str, route)))
    sys.stdout.write("\n".join(out))


def route_cost(route, dist):
    if not route:
        return 0.0
    total = dist[0][route[0] + 1]
    for i in range(len(route) - 1):
        total += dist[route[i] + 1][route[i + 1] + 1]
    total += dist[route[-1] + 1][0]
    return total


def two_opt(route, dist):
    m = len(route)
    if m < 4:
        return route[:]

    best = route[:]
    best_cost = route_cost(best, dist)
    improved = True
    while improved:
        improved = False
        for i in range(m - 2):
            for j in range(i + 2, m):
                if i == 0 and j == m - 1:
                    continue
                cand = best[:i + 1] + list(reversed(best[i + 1:j + 1])) + best[j + 1:]
                cand_cost = route_cost(cand, dist)
                if cand_cost + 1e-12 < best_cost:
                    best = cand
                    best_cost = cand_cost
                    improved = True
                    break
            if improved:
                break
    return best


def merge_cost(a, b, dist):
    options = [
        a + b,
        a + list(reversed(b)),
        list(reversed(a)) + b,
        list(reversed(a)) + list(reversed(b)),
    ]
    best_route = None
    best_cost = float("inf")
    for route in options:
        cost = route_cost(route, dist)
        if cost < best_cost:
            best_cost = cost
            best_route = route
    added = best_cost - route_cost(a, dist) - route_cost(b, dist)
    return added, best_route


def reduce_route_count(routes, demands, capacity, k, dist):
    loads = [sum(demands[c] for c in route) for route in routes]

    while len(routes) > k:
        best_pair = None
        best_added = float("inf")
        best_route = None
        for i in range(len(routes)):
            for j in range(i + 1, len(routes)):
                if loads[i] + loads[j] > capacity:
                    continue
                added, merged = merge_cost(routes[i], routes[j], dist)
                if added < best_added:
                    best_added = added
                    best_pair = (i, j)
                    best_route = merged

        if best_pair is None:
            break

        i, j = best_pair
        routes[i] = two_opt(best_route, dist)
        loads[i] += loads[j]
        routes.pop(j)
        loads.pop(j)

    return routes


if __name__ == "__main__":
    solve()
