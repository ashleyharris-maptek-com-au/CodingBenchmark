import math
import sys


def euclidean(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_distance(route, dist):
    if not route:
        return 0.0
    total = dist[0][route[0]]
    for i in range(len(route) - 1):
        total += dist[route[i]][route[i + 1]]
    total += dist[route[-1]][0]
    return total


def two_opt(route, dist):
    if len(route) < 4:
        return route
    improved = True
    best = route[:]
    while improved:
        improved = False
        for i in range(len(best) - 2):
            a = 0 if i == 0 else best[i - 1]
            b = best[i]
            for j in range(i + 1, len(best) - 1):
                c = best[j]
                d = best[j + 1]
                before = dist[a][b] + dist[c][d]
                after = dist[a][c] + dist[b][d]
                if after + 1e-12 < before:
                    best[i : j + 1] = reversed(best[i : j + 1])
                    improved = True
    return best


def nearest_neighbor_partition(n, k, capacity, demands, dist):
    customers = list(range(1, n + 1))
    remaining = set(customers)
    routes = []

    while remaining:
        if len(routes) == k - 1:
            leftover = list(remaining)
            leftover.sort(key=lambda c: math.atan2(coords[c][1] - coords[0][1], coords[c][0] - coords[0][0]))
            route = []
            load = 0
            next_remaining = set()
            for c in leftover:
                if load + demands[c] <= capacity:
                    route.append(c)
                    load += demands[c]
                else:
                    next_remaining.add(c)
            if not route:
                route.append(min(remaining, key=lambda c: dist[0][c]))
                next_remaining = set(remaining)
                next_remaining.remove(route[0])
            routes.append(route)
            remaining = next_remaining
            continue

        current = 0
        load = 0
        route = []
        while True:
            candidates = [c for c in remaining if load + demands[c] <= capacity]
            if not candidates:
                break
            next_customer = min(candidates, key=lambda c: (dist[current][c], dist[0][c]))
            route.append(next_customer)
            remaining.remove(next_customer)
            load += demands[next_customer]
            current = next_customer
        if not route:
            next_customer = min(remaining, key=lambda c: demands[c])
            route = [next_customer]
            remaining.remove(next_customer)
        routes.append(route)

    return routes


def clarke_wright(n, k, capacity, demands, dist):
    routes = {i: [i] for i in range(1, n + 1)}
    route_of = {i: i for i in range(1, n + 1)}
    load = {i: demands[i] for i in range(1, n + 1)}

    savings = []
    for i in range(1, n + 1):
        for j in range(i + 1, n + 1):
            s = dist[0][i] + dist[0][j] - dist[i][j]
            savings.append((s, i, j))
    savings.sort(reverse=True)

    for _, i, j in savings:
        ri = route_of[i]
        rj = route_of[j]
        if ri == rj:
            continue
        if load[ri] + load[rj] > capacity:
            continue
        left = routes[ri]
        right = routes[rj]

        merged = None
        if left[-1] == i and right[0] == j:
            merged = left + right
        elif left[0] == i and right[-1] == j:
            merged = right + left
        elif left[0] == i and right[0] == j:
            merged = list(reversed(left)) + right
        elif left[-1] == i and right[-1] == j:
            merged = left + list(reversed(right))

        if merged is None:
            continue

        new_id = ri
        old_id = rj
        routes[new_id] = merged
        load[new_id] += load[old_id]
        for c in merged:
            route_of[c] = new_id
        del routes[old_id]
        del load[old_id]

    result = list(routes.values())

    # If there are too many routes, greedily concatenate feasible pairs with best increase.
    while len(result) > k:
        best_pair = None
        best_delta = None
        for a in range(len(result)):
            load_a = sum(demands[c] for c in result[a])
            for b in range(a + 1, len(result)):
                load_b = sum(demands[c] for c in result[b])
                if load_a + load_b > capacity:
                    continue
                variants = [
                    result[a] + result[b],
                    result[a] + list(reversed(result[b])),
                    list(reversed(result[a])) + result[b],
                    list(reversed(result[a])) + list(reversed(result[b])),
                ]
                base = route_distance(result[a], dist) + route_distance(result[b], dist)
                for merged in variants:
                    delta = route_distance(merged, dist) - base
                    if best_delta is None or delta < best_delta:
                        best_delta = delta
                        best_pair = (a, b, merged)
        if best_pair is None:
            break
        a, b, merged = best_pair
        new_result = []
        for idx, route in enumerate(result):
            if idx not in (a, b):
                new_result.append(route)
        new_result.append(merged)
        result = new_result

    return result


def fix_route_count(routes, k, capacity, demands, dist):
    result = [r[:] for r in routes if r]
    while len(result) > k:
        best_pair = None
        best_delta = None
        for i in range(len(result)):
            load_i = sum(demands[c] for c in result[i])
            for j in range(i + 1, len(result)):
                load_j = sum(demands[c] for c in result[j])
                if load_i + load_j > capacity:
                    continue
                base = route_distance(result[i], dist) + route_distance(result[j], dist)
                variants = [
                    result[i] + result[j],
                    result[i] + list(reversed(result[j])),
                    list(reversed(result[i])) + result[j],
                    list(reversed(result[i])) + list(reversed(result[j])),
                ]
                for merged in variants:
                    delta = route_distance(merged, dist) - base
                    if best_delta is None or delta < best_delta:
                        best_delta = delta
                        best_pair = (i, j, merged)
        if best_pair is None:
            break
        i, j, merged = best_pair
        result = [route for idx, route in enumerate(result) if idx not in (i, j)]
        result.append(merged)
    return result


def improve_by_reinsertion(routes, capacity, demands, dist):
    route_loads = [sum(demands[c] for c in route) for route in routes]
    improved = True
    while improved:
        improved = False
        for i in range(len(routes)):
            if improved:
                break
            for pos in range(len(routes[i])):
                customer = routes[i][pos]
                old_route = routes[i]
                old_cost_i = route_distance(old_route, dist)
                reduced_route = old_route[:pos] + old_route[pos + 1 :]
                reduced_cost_i = route_distance(reduced_route, dist)
                for j in range(len(routes)):
                    if i == j:
                        continue
                    if route_loads[j] + demands[customer] > capacity:
                        continue
                    base_cost_j = route_distance(routes[j], dist)
                    best_insert_cost = None
                    best_insert_pos = None
                    for ins in range(len(routes[j]) + 1):
                        cand = routes[j][:ins] + [customer] + routes[j][ins:]
                        cand_cost = route_distance(cand, dist)
                        delta = (reduced_cost_i + cand_cost) - (old_cost_i + base_cost_j)
                        if best_insert_cost is None or delta < best_insert_cost:
                            best_insert_cost = delta
                            best_insert_pos = ins
                    if best_insert_cost is not None and best_insert_cost < -1e-12:
                        routes[i] = reduced_route
                        routes[j] = routes[j][:best_insert_pos] + [customer] + routes[j][best_insert_pos:]
                        route_loads[i] -= demands[customer]
                        route_loads[j] += demands[customer]
                        if not routes[i]:
                            del routes[i]
                            del route_loads[i]
                        improved = True
                        break
                if improved:
                    break
    return routes


def validate(routes, n, k, capacity, demands):
    if len(routes) > k:
        return False
    seen = []
    for route in routes:
        load = sum(demands[c] for c in route)
        if load > capacity:
            return False
        seen.extend(route)
    if len(seen) != n:
        return False
    if sorted(seen) != list(range(1, n + 1)):
        return False
    return True


data = sys.stdin.read().strip().split()
if not data:
    sys.exit(0)

it = iter(data)
n = int(next(it))
k = int(next(it))
capacity = int(next(it))

coords = []
demands = [0]

depot_x = float(next(it))
depot_y = float(next(it))
coords.append((depot_x, depot_y))

for _ in range(n):
    x = float(next(it))
    y = float(next(it))
    d = int(next(it))
    coords.append((x, y))
    demands.append(d)

if any(d > capacity for d in demands[1:]):
    raise ValueError("Infeasible instance: a customer demand exceeds vehicle capacity")

total_demand = sum(demands)
min_needed = (total_demand + capacity - 1) // capacity
if min_needed > k:
    raise ValueError("Infeasible instance: not enough vehicles to satisfy capacity")

dist = [[0.0] * (n + 1) for _ in range(n + 1)]
for i in range(n + 1):
    for j in range(i + 1, n + 1):
        d = euclidean(coords[i], coords[j])
        dist[i][j] = d
        dist[j][i] = d

routes_a = clarke_wright(n, k, capacity, demands, dist)
routes_b = nearest_neighbor_partition(n, k, capacity, demands, dist)

candidates = []
for base in (routes_a, routes_b):
    fixed = fix_route_count(base, k, capacity, demands, dist)
    optimized = [two_opt(route, dist) for route in fixed]
    optimized = improve_by_reinsertion(optimized, capacity, demands, dist)
    optimized = [two_opt(route, dist) for route in optimized]
    if validate(optimized, n, k, capacity, demands):
        candidates.append(optimized)

if not candidates:
    # Conservative fallback: sweep and split by angle.
    ordered = list(range(1, n + 1))
    ordered.sort(key=lambda c: math.atan2(coords[c][1] - coords[0][1], coords[c][0] - coords[0][0]))
    fallback = []
    current = []
    load = 0
    for c in ordered:
        if load + demands[c] > capacity:
            fallback.append(current)
            current = [c]
            load = demands[c]
        else:
            current.append(c)
            load += demands[c]
    if current:
        fallback.append(current)
    candidates.append([two_opt(route, dist) for route in fallback])

best_routes = min(candidates, key=lambda rs: sum(route_distance(route, dist) for route in rs))
best_total = sum(route_distance(route, dist) for route in best_routes)

out = [f"{best_total:.10f}", str(len(best_routes))]
for route in best_routes:
    out.append(" ".join(str(c - 1) for c in route))

sys.stdout.write("\n".join(out))
