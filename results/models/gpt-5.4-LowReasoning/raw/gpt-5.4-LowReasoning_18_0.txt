struct FastScanner {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline char nextChar() {
        if (idx >= size) {
            size = fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T& out) {
        char c;
        do {
            c = nextChar();
            if (!c) return false;
        } while (c <= ' ');
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = nextChar();
        }
        T val = 0;
        while (c > ' ') {
            val = val * 10 + static_cast<T>(c - '0');
            c = nextChar();
        }
        out = neg ? -val : val;
        return true;
    }
};

struct Edge {
    int u;
    int v;
};

struct CutResult {
    int value = numeric_limits<int>::max();
    vector<char> side;
};

static vector<int> buildCutEdges(const vector<Edge>& edges, const vector<char>& side) {
    vector<int> cut;
    cut.reserve(edges.size() / 16 + 1);
    for (int i = 0; i < static_cast<int>(edges.size()); ++i) {
        const auto& e = edges[i];
        if (e.u == e.v) continue;
        if (side[e.u] != side[e.v]) cut.push_back(i);
    }
    return cut;
}

static int findBridge(const vector<vector<pair<int, int>>>& adj, const vector<Edge>& edges) {
    const int n = static_cast<int>(adj.size());
    vector<int> tin(n, -1), low(n, -1), parentEdge(n, -1), it(n, 0);
    int timer = 0;
    for (int root = 0; root < n; ++root) {
        if (tin[root] != -1) continue;
        vector<int> st;
        st.push_back(root);
        while (!st.empty()) {
            int v = st.back();
            if (tin[v] == -1) {
                tin[v] = low[v] = timer++;
            }
            if (it[v] < static_cast<int>(adj[v].size())) {
                auto [to, id] = adj[v][it[v]++];
                if (id == parentEdge[v]) continue;
                if (tin[to] == -1) {
                    parentEdge[to] = id;
                    st.push_back(to);
                } else {
                    low[v] = min(low[v], tin[to]);
                }
            } else {
                st.pop_back();
                if (!st.empty()) {
                    int p = st.back();
                    low[p] = min(low[p], low[v]);
                    if (low[v] > tin[p]) {
                        return parentEdge[v];
                    }
                }
            }
        }
    }
    return -1;
}

static CutResult stoerWagner(int n, const vector<Edge>& edges) {
    vector<unordered_map<int, int>> g(n);
    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        ++g[e.u][e.v];
        ++g[e.v][e.u];
    }

    vector<vector<int>> members(n);
    for (int i = 0; i < n; ++i) members[i].push_back(i);
    vector<int> verts(n);
    iota(verts.begin(), verts.end(), 0);

    CutResult best;
    best.value = numeric_limits<int>::max();
    best.side.assign(n, 0);

    while (verts.size() > 1) {
        vector<int> w(n, 0);
        vector<char> added(n, 0);
        int prev = -1;
        int sel = -1;

        for (size_t step = 0; step < verts.size(); ++step) {
            sel = -1;
            int bestW = -1;
            for (int v : verts) {
                if (!added[v] && w[v] > bestW) {
                    bestW = w[v];
                    sel = v;
                }
            }
            if (sel == -1) break;
            added[sel] = 1;
            if (step + 1 == verts.size()) {
                if (bestW < best.value) {
                    best.value = bestW;
                    fill(best.side.begin(), best.side.end(), 0);
                    for (int x : members[sel]) best.side[x] = 1;
                }
                if (prev == -1) break;

                if (g[sel].size() > g[prev].size()) {
                    swap(g[sel], g[prev]);
                    swap(members[sel], members[prev]);
                }
                for (const auto& kv : g[sel]) {
                    int to = kv.first;
                    int cap = kv.second;
                    if (to == prev) continue;
                    g[prev][to] += cap;
                    g[to][prev] += cap;
                    g[to].erase(sel);
                }
                g[prev].erase(sel);
                g[sel].clear();
                members[prev].insert(members[prev].end(), members[sel].begin(), members[sel].end());
                members[sel].clear();
                verts.erase(remove(verts.begin(), verts.end(), sel), verts.end());
                break;
            }
            prev = sel;
            for (const auto& kv : g[sel]) {
                if (!added[kv.first]) w[kv.first] += kv.second;
            }
        }
    }
    return best;
}

static vector<Edge> buildCertificate(int n, const vector<Edge>& edges, int k) {
    vector<vector<pair<int, int>>> adj(n);
    adj.assign(n, {});
    adj.reserve(n);
    for (int i = 0; i < static_cast<int>(edges.size()); ++i) {
        const auto& e = edges[i];
        if (e.u == e.v) continue;
        adj[e.u].push_back({e.v, i});
        adj[e.v].push_back({e.u, i});
    }

    vector<int> weight(n, 0);
    vector<char> used(n, 0);
    priority_queue<pair<int, int>> pq;
    pq.push({0, 0});
    vector<int> order;
    order.reserve(n);

    for (int iter = 0; iter < n; ++iter) {
        while (!pq.empty() && used[pq.top().second]) pq.pop();
        int v = -1;
        if (!pq.empty()) {
            v = pq.top().second;
            pq.pop();
        } else {
            for (int i = 0; i < n; ++i) {
                if (!used[i]) {
                    v = i;
                    break;
                }
            }
        }
        used[v] = 1;
        order.push_back(v);
        for (const auto& [to, id] : adj[v]) {
            (void)id;
            if (!used[to]) {
                ++weight[to];
                pq.push({weight[to], to});
            }
        }
    }

    vector<int> pos(n, -1);
    for (int i = 0; i < n; ++i) pos[order[i]] = i;

    vector<int> keepCount(n, 0);
    vector<char> keep(edges.size(), 0);
    for (int idx = static_cast<int>(order.size()) - 1; idx >= 0; --idx) {
        int v = order[idx];
        for (const auto& [to, id] : adj[v]) {
            if (pos[to] < pos[v] && keepCount[v] < k) {
                keep[id] = 1;
                ++keepCount[v];
            }
        }
    }

    vector<Edge> cert;
    cert.reserve(static_cast<size_t>(k) * static_cast<size_t>(max(0, n - 1)));
    for (int i = 0; i < static_cast<int>(edges.size()); ++i) {
        if (keep[i]) cert.push_back(edges[i]);
    }
    return cert;
}

struct Dinic {
    struct Arc { int to, rev, cap; };
    int n;
    vector<vector<Arc>> g;
    vector<int> level, it;

    explicit Dinic(int n_) : n(n_), g(n_), level(n_), it(n_) {}

    void addEdge(int u, int v, int c) {
        Arc a{v, static_cast<int>(g[v].size()), c};
        Arc b{u, static_cast<int>(g[u].size()), c};
        g[u].push_back(a);
        g[v].push_back(b);
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (const auto& e : g[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    int dfs(int v, int t, int f) {
        if (v == t) return f;
        for (int& i = it[v]; i < static_cast<int>(g[v].size()); ++i) {
            Arc& e = g[v][i];
            if (e.cap <= 0 || level[e.to] != level[v] + 1) continue;
            int got = dfs(e.to, t, min(f, e.cap));
            if (got > 0) {
                e.cap -= got;
                g[e.to][e.rev].cap += got;
                return got;
            }
        }
        return 0;
    }

    int maxflow(int s, int t, int limit) {
        int flow = 0;
        while (flow < limit && bfs(s, t)) {
            fill(it.begin(), it.end(), 0);
            while (flow < limit) {
                int pushed = dfs(s, t, limit - flow);
                if (!pushed) break;
                flow += pushed;
            }
        }
        return flow;
    }

    vector<char> minSide(int s) {
        vector<char> vis(n, 0);
        queue<int> q;
        q.push(s);
        vis[s] = 1;
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (const auto& e : g[v]) {
                if (e.cap > 0 && !vis[e.to]) {
                    vis[e.to] = 1;
                    q.push(e.to);
                }
            }
        }
        return vis;
    }
};

static CutResult sourceBoundedFlows(int n, const vector<Edge>& edges, int source, int upperBound) {
    CutResult best;
    best.value = upperBound;
    best.side.assign(n, 0);
    best.side[source] = 1;

    for (int t = 0; t < n; ++t) {
        if (t == source) continue;
        Dinic dinic(n);
        for (const auto& e : edges) {
            if (e.u == e.v) continue;
            dinic.addEdge(e.u, e.v, 1);
        }
        int flow = dinic.maxflow(source, t, best.value - 1);
        if (flow < best.value) {
            best.value = flow;
            best.side = dinic.minSide(source);
            if (best.value <= 1) break;
        }
    }
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    FastScanner fs;
    int n = 0;
    long long mll = 0;
    if (!fs.readInt(n) || !fs.readInt(mll)) return 0;
    if (n <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    vector<Edge> edges;
    edges.reserve(static_cast<size_t>(max<long long>(0, mll)));
    vector<vector<pair<int, int>>> adj(n);
    vector<int> degree(n, 0);

    for (long long i = 0; i < mll; ++i) {
        int u, v;
        fs.readInt(u);
        fs.readInt(v);
        edges.push_back({u, v});
        int id = static_cast<int>(edges.size()) - 1;
        if (u != v) {
            adj[u].push_back({v, id});
            adj[v].push_back({u, id});
            ++degree[u];
            ++degree[v];
        }
    }

    int minDegV = 0;
    for (int i = 1; i < n; ++i) {
        if (degree[i] < degree[minDegV]) minDegV = i;
    }
    int upperBound = degree[minDegV];

    if (upperBound <= 1) {
        cout << upperBound << '\n';
        int printed = 0;
        for (const auto& [to, id] : adj[minDegV]) {
            (void)to;
            const auto& e = edges[id];
            cout << e.u << ' ' << e.v << '\n';
            if (++printed == upperBound) break;
        }
        return 0;
    }

    int bridgeId = findBridge(adj, edges);
    if (bridgeId >= 0) {
        cout << 1 << '\n';
        cout << edges[bridgeId].u << ' ' << edges[bridgeId].v << '\n';
        return 0;
    }

    CutResult best;
    best.value = upperBound;
    best.side.assign(n, 0);
    best.side[minDegV] = 1;

    const long long m = static_cast<long long>(edges.size());
    bool ranExact = false;

    if (n <= 4000 || m <= 200000) {
        best = stoerWagner(n, edges);
        ranExact = true;
    } else if (upperBound <= 64 && n <= 30000) {
        vector<Edge> cert = buildCertificate(n, edges, upperBound);
        CutResult certBest = stoerWagner(n, cert);
        if (certBest.value < upperBound) {
            best = certBest;
            ranExact = true;
        }
    }

    if (!ranExact && upperBound <= 8 && n <= 2500 && m <= 150000) {
        CutResult flowBest = sourceBoundedFlows(n, edges, minDegV, upperBound);
        if (flowBest.value <= best.value) {
            best = flowBest;
            ranExact = true;
        }
    }

    vector<int> cut = buildCutEdges(edges, best.side);
    if (static_cast<int>(cut.size()) != best.value) {
        vector<char> fallbackSide(n, 0);
        fallbackSide[minDegV] = 1;
        best.side.swap(fallbackSide);
        cut.clear();
        for (const auto& [to, id] : adj[minDegV]) {
            (void)to;
            cut.push_back(id);
        }
        sort(cut.begin(), cut.end());
        cut.erase(unique(cut.begin(), cut.end()), cut.end());
        best.value = static_cast<int>(cut.size());
    }

    cout << best.value << '\n';
    for (int id : cut) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }
    return 0;
}
