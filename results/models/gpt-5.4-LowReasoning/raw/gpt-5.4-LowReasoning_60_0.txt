const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : { machineCount: 1, tasks: [] };

const machineCount = Math.max(1, data.machineCount | 0);
const tasksInput = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasksInput.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const tasks = new Array(n);
const idToIndex = new Map();

for (let i = 0; i < n; i++) {
  const t = tasksInput[i] || {};
  const id = t.id;
  idToIndex.set(id, i);
  tasks[i] = {
    id,
    duration: Math.max(0, t.duration | 0),
    depsRaw: Array.isArray(t.deps) ? t.deps : [],
    deadline: Number.isFinite(t.deadline) ? t.deadline : Number.MAX_SAFE_INTEGER / 4,
    weight: Number.isFinite(t.weight) ? t.weight : 1,
    deps: [],
    succ: [],
  };
}

const indegree = new Array(n).fill(0);
for (let i = 0; i < n; i++) {
  const deps = tasks[i].depsRaw;
  const seen = new Set();
  for (let j = 0; j < deps.length; j++) {
    const depIdx = idToIndex.get(deps[j]);
    if (depIdx === undefined || depIdx === i || seen.has(depIdx)) continue;
    seen.add(depIdx);
    tasks[i].deps.push(depIdx);
    tasks[depIdx].succ.push(i);
    indegree[i]++;
  }
}

function topoSort() {
  const queue = new Array(n);
  let head = 0;
  let tail = 0;
  const indeg = indegree.slice();
  for (let i = 0; i < n; i++) {
    if (indeg[i] === 0) queue[tail++] = i;
  }
  const order = new Array(n);
  let count = 0;
  while (head < tail) {
    const u = queue[head++];
    order[count++] = u;
    const succ = tasks[u].succ;
    for (let k = 0; k < succ.length; k++) {
      const v = succ[k];
      if (--indeg[v] === 0) queue[tail++] = v;
    }
  }
  if (count !== n) return null;
  return order;
}

const topo = topoSort();
if (!topo) {
  process.stdout.write(JSON.stringify({ order: tasks.map((t) => t.id) }));
  process.exit(0);
}

const revTopo = topo.slice().reverse();
const cpLen = new Array(n).fill(0);
const succWeight = new Array(n).fill(0);
const descCount = new Array(n).fill(0);

for (let r = 0; r < n; r++) {
  const u = revTopo[r];
  let bestChild = 0;
  let weightSum = 0;
  let desc = 0;
  const succ = tasks[u].succ;
  for (let k = 0; k < succ.length; k++) {
    const v = succ[k];
    if (cpLen[v] > bestChild) bestChild = cpLen[v];
    weightSum += tasks[v].weight + succWeight[v];
    desc += 1 + descCount[v];
  }
  cpLen[u] = tasks[u].duration + bestChild;
  succWeight[u] = weightSum;
  descCount[u] = desc;
}

const ready = [];
const remainingDeps = indegree.slice();
const depFinishMax = new Array(n).fill(0);
const machineAvail = new Array(machineCount).fill(0);
const scheduled = new Array(n);
let readyCount = 0;

function priorityScore(idx, est) {
  const t = tasks[idx];
  const slack = t.deadline - est - cpLen[idx];
  const urgency = slack < 0 ? -slack : 0;
  return (
    cpLen[idx] * 1000000 +
    urgency * 20000 +
    t.weight * 5000 +
    succWeight[idx] * 5 +
    descCount[idx]
  );
}

for (let i = 0; i < n; i++) {
  if (remainingDeps[i] === 0) ready[readyCount++] = i;
}

function popBestReady() {
  let bestPos = 0;
  let bestIdx = ready[0];
  let bestStart = Infinity;
  let bestMachine = 0;
  let bestScore = -Infinity;

  for (let pos = 0; pos < readyCount; pos++) {
    const idx = ready[pos];
    let earliest = depFinishMax[idx];
    let chosenMachine = 0;

    if (machineCount > 1) {
      let minAvail = machineAvail[0];
      let minMachine = 0;
      for (let m = 1; m < machineCount; m++) {
        if (machineAvail[m] < minAvail) {
          minAvail = machineAvail[m];
          minMachine = m;
        }
      }
      earliest = Math.max(earliest, minAvail);
      chosenMachine = minMachine;
    } else {
      earliest = Math.max(earliest, machineAvail[0]);
    }

    const score = priorityScore(idx, earliest);
    if (
      score > bestScore ||
      (score === bestScore && earliest < bestStart) ||
      (score === bestScore && earliest === bestStart && tasks[idx].duration > tasks[bestIdx].duration)
    ) {
      bestScore = score;
      bestPos = pos;
      bestIdx = idx;
      bestStart = earliest;
      bestMachine = chosenMachine;
    }
  }

  readyCount--;
  ready[bestPos] = ready[readyCount];
  ready.length = readyCount;
  return { idx: bestIdx, start: bestStart, machine: bestMachine };
}

for (let placed = 0; placed < n; placed++) {
  const pick = popBestReady();
  const idx = pick.idx;
  const t = tasks[idx];
  const start = pick.start;
  const finish = start + t.duration;

  machineAvail[pick.machine] = finish;
  scheduled[idx] = { id: t.id, start, machine: pick.machine };

  const succ = t.succ;
  for (let k = 0; k < succ.length; k++) {
    const v = succ[k];
    if (finish > depFinishMax[v]) depFinishMax[v] = finish;
    if (--remainingDeps[v] === 0) {
      ready[readyCount++] = v;
    }
  }
}

scheduled.sort((a, b) => {
  if (a.start !== b.start) return a.start - b.start;
  if (a.machine !== b.machine) return a.machine - b.machine;
  const as = String(a.id);
  const bs = String(b.id);
  return as < bs ? -1 : as > bs ? 1 : 0;
});

process.stdout.write(JSON.stringify({ schedule: scheduled }));
