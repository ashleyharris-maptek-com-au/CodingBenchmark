const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input || '{}');

const machineCount = Math.max(1, data.machineCount | 0);
const rawTasks = Array.isArray(data.tasks) ? data.tasks : [];

class Heap {
  constructor(compare) {
    this.compare = compare;
    this.data = [];
  }

  size() {
    return this.data.length;
  }

  peek() {
    return this.data[0];
  }

  push(value) {
    const arr = this.data;
    arr.push(value);
    let i = arr.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.compare(arr[i], arr[p]) >= 0) break;
      [arr[i], arr[p]] = [arr[p], arr[i]];
      i = p;
    }
  }

  pop() {
    const arr = this.data;
    if (arr.length === 0) return undefined;
    const top = arr[0];
    const last = arr.pop();
    if (arr.length > 0) {
      arr[0] = last;
      let i = 0;
      for (;;) {
        const l = i * 2 + 1;
        const r = l + 1;
        let best = i;
        if (l < arr.length && this.compare(arr[l], arr[best]) < 0) best = l;
        if (r < arr.length && this.compare(arr[r], arr[best]) < 0) best = r;
        if (best === i) break;
        [arr[i], arr[best]] = [arr[best], arr[i]];
        i = best;
      }
    }
    return top;
  }
}

function cmpId(a, b) {
  const sa = String(a);
  const sb = String(b);
  return sa < sb ? -1 : sa > sb ? 1 : 0;
}

const n = rawTasks.length;
const indexById = new Map();
const tasks = new Array(n);

for (let i = 0; i < n; i++) {
  const src = rawTasks[i] || {};
  const id = src.id;
  indexById.set(id, i);
  tasks[i] = {
    id,
    duration: Math.max(0, src.duration | 0),
    deps: Array.isArray(src.deps) ? src.deps.slice() : [],
    deadline: Number.isFinite(src.deadline) ? src.deadline : Number.MAX_SAFE_INTEGER / 4,
    weight: Number.isFinite(src.weight) ? src.weight : 1,
    succ: [],
    indegree: 0,
    maxDepFinish: 0,
    cp: 0,
    succCount: 0,
    downWeight: 0,
  };
}

for (let i = 0; i < n; i++) {
  const t = tasks[i];
  const seen = new Set();
  for (const depId of t.deps) {
    const depIndex = indexById.get(depId);
    if (depIndex === undefined || depIndex === i || seen.has(depIndex)) continue;
    seen.add(depIndex);
    tasks[depIndex].succ.push(i);
    t.indegree++;
  }
}

const indegreeForTopo = tasks.map((t) => t.indegree);
const topo = [];
const topoHeap = new Heap((a, b) => cmpId(tasks[a].id, tasks[b].id));
for (let i = 0; i < n; i++) {
  if (indegreeForTopo[i] === 0) topoHeap.push(i);
}
while (topoHeap.size() > 0) {
  const u = topoHeap.pop();
  topo.push(u);
  const succ = tasks[u].succ;
  for (let k = 0; k < succ.length; k++) {
    const v = succ[k];
    indegreeForTopo[v]--;
    if (indegreeForTopo[v] === 0) topoHeap.push(v);
  }
}

if (topo.length !== n) {
  const order = rawTasks
    .map((t) => t.id)
    .slice()
    .sort(cmpId);
  process.stdout.write(JSON.stringify({ order }));
  process.exit(0);
}

for (let i = n - 1; i >= 0; i--) {
  const idx = topo[i];
  const t = tasks[idx];
  let bestCp = 0;
  let succCount = 0;
  let downWeight = 0;
  for (let j = 0; j < t.succ.length; j++) {
    const s = tasks[t.succ[j]];
    if (s.cp > bestCp) bestCp = s.cp;
    succCount += 1 + s.succCount;
    downWeight += s.weight + s.downWeight;
  }
  t.cp = t.duration + bestCp;
  t.succCount = succCount;
  t.downWeight = downWeight;
}

function compareReady(a, b) {
  const ta = tasks[a];
  const tb = tasks[b];

  const slackA = ta.deadline - ta.cp;
  const slackB = tb.deadline - tb.cp;
  if (slackA !== slackB) return slackA - slackB;

  if (ta.cp !== tb.cp) return tb.cp - ta.cp;

  const wa = ta.weight;
  const wb = tb.weight;
  const da = Math.max(1, ta.duration);
  const db = Math.max(1, tb.duration);
  const densityCmp = wb * da - wa * db;
  if (densityCmp !== 0) return densityCmp;

  const impactA = ta.weight + ta.downWeight;
  const impactB = tb.weight + tb.downWeight;
  if (impactA !== impactB) return impactB - impactA;

  if (ta.deadline !== tb.deadline) return ta.deadline - tb.deadline;
  if (ta.duration !== tb.duration) return tb.duration - ta.duration;
  return cmpId(ta.id, tb.id);
}

const futureReady = new Heap((a, b) => {
  if (a.time !== b.time) return a.time - b.time;
  return compareReady(a.idx, b.idx);
});
const ready = new Heap(compareReady);
const machines = new Heap((a, b) => (a.time !== b.time ? a.time - b.time : a.machine - b.machine));

for (let m = 0; m < machineCount; m++) {
  machines.push({ time: 0, machine: m });
}

for (let i = 0; i < n; i++) {
  if (tasks[i].indegree === 0) {
    futureReady.push({ time: 0, idx: i });
  }
}

const schedule = [];
let scheduled = 0;
let currentTime = 0;

while (scheduled < n) {
  let nextMachineTime = machines.peek() ? machines.peek().time : Number.MAX_SAFE_INTEGER;
  let nextTaskTime = futureReady.peek() ? futureReady.peek().time : Number.MAX_SAFE_INTEGER;

  if (ready.size() === 0) {
    if (nextTaskTime === Number.MAX_SAFE_INTEGER) break;
    currentTime = Math.max(currentTime, nextTaskTime);
  } else {
    currentTime = Math.max(currentTime, nextMachineTime);
  }

  while (futureReady.size() > 0 && futureReady.peek().time <= currentTime) {
    ready.push(futureReady.pop().idx);
  }

  if (ready.size() === 0) continue;

  while (machines.size() > 0 && machines.peek().time <= currentTime && ready.size() > 0) {
    const machine = machines.pop();
    const idx = ready.pop();
    const t = tasks[idx];
    const start = currentTime;
    const finish = start + t.duration;
    schedule.push({ id: t.id, start, machine: machine.machine });
    machines.push({ time: finish, machine: machine.machine });
    scheduled++;

    for (let j = 0; j < t.succ.length; j++) {
      const sIdx = t.succ[j];
      const s = tasks[sIdx];
      if (finish > s.maxDepFinish) s.maxDepFinish = finish;
      s.indegree--;
      if (s.indegree === 0) {
        futureReady.push({ time: s.maxDepFinish, idx: sIdx });
      }
    }
  }
}

if (schedule.length !== n) {
  const order = topo.map((idx) => tasks[idx].id);
  process.stdout.write(JSON.stringify({ order }));
} else {
  process.stdout.write(JSON.stringify({ schedule }));
}
