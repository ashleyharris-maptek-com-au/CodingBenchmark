#include <algorithm>
#include <array>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct City {
    float x;
    float y;
    std::uint32_t id;
    std::uint64_t key;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline std::uint64_t split_by_1(std::uint32_t x) {
    std::uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

static inline std::uint64_t morton_code(std::uint32_t x, std::uint32_t y) {
    return split_by_1(x) | (split_by_1(y) << 1);
}

static std::vector<int> solve_tiny_exact(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());
    if (n == 1) return {0};

    const int m = n - 1;
    const int full = 1 << m;
    const double INF = std::numeric_limits<double>::infinity();

    std::vector<double> dp(static_cast<std::size_t>(full) * m, INF);
    std::vector<int> parent(static_cast<std::size_t>(full) * m, -1);

    auto idx = [m](int mask, int last) -> std::size_t {
        return static_cast<std::size_t>(mask) * static_cast<std::size_t>(m) + static_cast<std::size_t>(last);
    };

    for (int i = 0; i < m; ++i) {
        dp[idx(1 << i, i)] = dist2(cities[0], cities[i + 1]);
    }

    for (int mask = 1; mask < full; ++mask) {
        for (int last = 0; last < m; ++last) {
            if ((mask & (1 << last)) == 0) continue;
            const double cur = dp[idx(mask, last)];
            if (cur == INF) continue;
            for (int nxt = 0; nxt < m; ++nxt) {
                if (mask & (1 << nxt)) continue;
                const int nmask = mask | (1 << nxt);
                const double cand = cur + dist2(cities[last + 1], cities[nxt + 1]);
                const std::size_t at = idx(nmask, nxt);
                if (cand < dp[at]) {
                    dp[at] = cand;
                    parent[at] = last;
                }
            }
        }
    }

    int best_last = 0;
    double best_cost = INF;
    const int all = full - 1;
    for (int last = 0; last < m; ++last) {
        const double cand = dp[idx(all, last)];
        if (cand < best_cost) {
            best_cost = cand;
            best_last = last;
        }
    }

    std::vector<int> order(n);
    order[0] = 0;
    int mask = all;
    int last = best_last;
    for (int pos = n - 1; pos >= 1; --pos) {
        order[pos] = last + 1;
        const int prev = parent[idx(mask, last)];
        mask ^= 1 << last;
        last = prev;
    }
    return order;
}

static std::vector<int> greedy_nearest(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());
    std::vector<int> order;
    order.reserve(n);
    std::vector<unsigned char> used(n, 0);
    order.push_back(0);
    used[0] = 1;

    for (int step = 1; step < n; ++step) {
        const City& cur = cities[order.back()];
        int best = -1;
        double best_d = std::numeric_limits<double>::infinity();
        for (int i = 1; i < n; ++i) {
            if (used[i]) continue;
            const double d = dist2(cur, cities[i]);
            if (d < best_d) {
                best_d = d;
                best = i;
            }
        }
        used[best] = 1;
        order.push_back(best);
    }
    return order;
}

static inline bool should_swap(const std::vector<City>& cities, const std::vector<int>& order, int i, int k) {
    const int n = static_cast<int>(order.size());
    const City& a = cities[order[i]];
    const City& b = cities[order[i + 1]];
    const City& c = cities[order[k]];
    const City& d = cities[order[(k + 1) % n]];
    const double old_cost = dist2(a, b) + dist2(c, d);
    const double new_cost = dist2(a, c) + dist2(b, d);
    return new_cost + 1e-12 < old_cost;
}

static void two_opt_limited(const std::vector<City>& cities, std::vector<int>& order, int passes, int window) {
    const int n = static_cast<int>(order.size());
    if (n < 4) return;

    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;
        for (int i = 0; i + 2 < n; ++i) {
            const int max_k = std::min(n - 1, i + window);
            for (int k = i + 2; k <= max_k; ++k) {
                if (i == 0 && k == n - 1) continue;
                if (should_swap(cities, order, i, k)) {
                    std::reverse(order.begin() + i + 1, order.begin() + k + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

static std::vector<int> solve_medium(std::vector<City> cities) {
    std::vector<int> order = greedy_nearest(cities);
    const int n = static_cast<int>(cities.size());
    int passes = 1;
    int window = 24;
    if (n <= 400) {
        passes = 4;
        window = 80;
    } else if (n <= 1000) {
        passes = 2;
        window = 48;
    }
    two_opt_limited(cities, order, passes, window);
    return order;
}

static std::vector<int> solve_by_morton(const std::vector<City>& original, std::vector<City> cities, bool refine) {
    const std::size_t n = cities.size();
    if (n == 1) return {0};

    float min_x = cities[0].x, max_x = cities[0].x;
    float min_y = cities[0].y, max_y = cities[0].y;
    for (const City& c : cities) {
        if (c.x < min_x) min_x = c.x;
        if (c.x > max_x) max_x = c.x;
        if (c.y < min_y) min_y = c.y;
        if (c.y > max_y) max_y = c.y;
    }

    const double dx = std::max(1e-30, static_cast<double>(max_x) - static_cast<double>(min_x));
    const double dy = std::max(1e-30, static_cast<double>(max_y) - static_cast<double>(min_y));
    constexpr double scale = 4294967295.0;

    for (City& c : cities) {
        const double nx = (static_cast<double>(c.x) - static_cast<double>(min_x)) / dx;
        const double ny = (static_cast<double>(c.y) - static_cast<double>(min_y)) / dy;
        const std::uint32_t xi = static_cast<std::uint32_t>(std::clamp(nx, 0.0, 1.0) * scale);
        const std::uint32_t yi = static_cast<std::uint32_t>(std::clamp(ny, 0.0, 1.0) * scale);
        c.key = morton_code(xi, yi);
    }

    std::sort(cities.begin(), cities.end(), [](const City& a, const City& b) {
        if (a.key != b.key) return a.key < b.key;
        return a.id < b.id;
    });

    std::vector<int> order;
    order.reserve(n);
    std::size_t pos0 = 0;
    for (std::size_t i = 0; i < n; ++i) {
        if (cities[i].id == 0) {
            pos0 = i;
            break;
        }
    }

    for (std::size_t k = 0; k < n; ++k) {
        order.push_back(static_cast<int>(cities[(pos0 + k) % n].id));
    }

    if (refine) {
        int passes = (n <= 5000 ? 2 : 1);
        int window = (n <= 5000 ? 64 : 24);
        two_opt_limited(original, order, passes, window);
    }

    return order;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::size_t n;
    if (!(std::cin >> n)) {
        return 0;
    }

    std::vector<City> cities(n);
    for (std::size_t i = 0; i < n; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
        cities[i].id = static_cast<std::uint32_t>(i);
        cities[i].key = 0;
    }

    std::vector<int> order;
    if (n == 0) {
        order.clear();
    } else if (n == 1) {
        order = {0};
    } else if (n <= 20) {
        order = solve_tiny_exact(cities);
    } else if (n <= 1200) {
        order = solve_medium(cities);
    } else if (n <= 50000) {
        order = solve_by_morton(cities, cities, true);
    } else {
        order = solve_by_morton(cities, cities, false);
    }

    if (!order.empty() && order[0] != 0) {
        auto it = std::find(order.begin(), order.end(), 0);
        if (it != order.end()) {
            std::rotate(order.begin(), it, order.end());
        }
    }

    for (std::size_t i = 0; i < order.size(); ++i) {
        if (i) std::cout << ' ';
        std::cout << order[i];
    }
    std::cout << '\n';
    return 0;
}
