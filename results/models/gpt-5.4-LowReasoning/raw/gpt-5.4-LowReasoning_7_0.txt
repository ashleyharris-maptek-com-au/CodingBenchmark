constexpr double EPS = 1e-9;

struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

double norm2(const Vec3& v) {
    return dot(v, v);
}

double norm(const Vec3& v) {
    return sqrt(norm2(v));
}

Vec3 normalized(const Vec3& v) {
    double n = norm(v);
    return n <= EPS ? Vec3{} : v / n;
}

struct Tri {
    int a = 0, b = 0, c = 0;
};

struct AABB {
    Vec3 mn{numeric_limits<double>::infinity(),
            numeric_limits<double>::infinity(),
            numeric_limits<double>::infinity()};
    Vec3 mx{-numeric_limits<double>::infinity(),
            -numeric_limits<double>::infinity(),
            -numeric_limits<double>::infinity()};

    void expand(const Vec3& p) {
        mn.x = min(mn.x, p.x);
        mn.y = min(mn.y, p.y);
        mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x);
        mx.y = max(mx.y, p.y);
        mx.z = max(mx.z, p.z);
    }
};

struct Mesh {
    vector<Vec3> v;
    vector<Tri> f;
    AABB box;
};

struct U64Hash {
    size_t operator()(uint64_t x) const noexcept {
        x ^= x >> 33;
        x *= 0xff51afd7ed558ccdULL;
        x ^= x >> 33;
        x *= 0xc4ceb9fe1a85ec53ULL;
        x ^= x >> 33;
        return static_cast<size_t>(x);
    }
};

uint64_t mix3(int x, int y, int z) {
    uint64_t ux = static_cast<uint32_t>(x * 73856093);
    uint64_t uy = static_cast<uint32_t>(y * 19349663);
    uint64_t uz = static_cast<uint32_t>(z * 83492791);
    return ux ^ (uy << 21) ^ (uz << 42);
}

AABB computeBox(const Mesh& m) {
    AABB b;
    for (const auto& p : m.v) b.expand(p);
    return b;
}

bool overlap1D(double a0, double a1, double b0, double b1) {
    return !(a1 < b0 - EPS || b1 < a0 - EPS);
}

bool boxesOverlap(const AABB& a, const AABB& b) {
    return overlap1D(a.mn.x, a.mx.x, b.mn.x, b.mx.x) &&
           overlap1D(a.mn.y, a.mx.y, b.mn.y, b.mx.y) &&
           overlap1D(a.mn.z, a.mx.z, b.mn.z, b.mx.z);
}

double triArea2(const Vec3& a, const Vec3& b, const Vec3& c) {
    return norm(cross(b - a, c - a));
}

Vec3 triNormal(const Vec3& a, const Vec3& b, const Vec3& c) {
    return normalized(cross(b - a, c - a));
}

bool rayTriangle(const Vec3& orig, const Vec3& dir, const Vec3& a, const Vec3& b, const Vec3& c, double& t) {
    Vec3 e1 = b - a;
    Vec3 e2 = c - a;
    Vec3 p = cross(dir, e2);
    double det = dot(e1, p);
    if (fabs(det) < EPS) return false;
    double invDet = 1.0 / det;
    Vec3 s = orig - a;
    double u = dot(s, p) * invDet;
    if (u < -EPS || u > 1.0 + EPS) return false;
    Vec3 q = cross(s, e1);
    double v = dot(dir, q) * invDet;
    if (v < -EPS || u + v > 1.0 + EPS) return false;
    t = dot(e2, q) * invDet;
    return t >= EPS;
}

bool pointInMesh(const Mesh& mesh, const Vec3& p) {
    // Multi-direction voting reduces ray/edge degeneracy.
    static const array<Vec3, 3> dirs = {
        Vec3{1.0, 0.17364817766, 0.1134464014},
        Vec3{0.14285714285, 1.0, 0.21739130434},
        Vec3{0.27182818284, 0.31415926535, 1.0}
    };
    int insideVotes = 0;
    for (Vec3 dir : dirs) {
        dir = normalized(dir);
        int hits = 0;
        for (const auto& t : mesh.f) {
            double dist;
            if (rayTriangle(p, dir, mesh.v[t.a], mesh.v[t.b], mesh.v[t.c], dist)) {
                ++hits;
            }
        }
        if (hits & 1) ++insideVotes;
    }
    return insideVotes >= 2;
}

bool segmentTriangleIntersect(const Vec3& p0, const Vec3& p1, const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 dir = p1 - p0;
    double len = norm(dir);
    if (len <= EPS) return false;
    dir = dir / len;
    double t;
    if (!rayTriangle(p0, dir, a, b, c, t)) return false;
    return t <= len + EPS;
}

bool triangleIntersectsTriangle(const Vec3& a0, const Vec3& a1, const Vec3& a2,
                                const Vec3& b0, const Vec3& b1, const Vec3& b2) {
    AABB ba, bb;
    ba.expand(a0); ba.expand(a1); ba.expand(a2);
    bb.expand(b0); bb.expand(b1); bb.expand(b2);
    if (!boxesOverlap(ba, bb)) return false;

    array<pair<Vec3, Vec3>, 3> ea = {{{a0, a1}, {a1, a2}, {a2, a0}}};
    array<pair<Vec3, Vec3>, 3> eb = {{{b0, b1}, {b1, b2}, {b2, b0}}};
    for (const auto& e : ea) if (segmentTriangleIntersect(e.first, e.second, b0, b1, b2)) return true;
    for (const auto& e : eb) if (segmentTriangleIntersect(e.first, e.second, a0, a1, a2)) return true;

    Vec3 ca = (a0 + a1 + a2) / 3.0;
    Vec3 cb = (b0 + b1 + b2) / 3.0;
    auto pointInTriPlane = [](const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c) {
        Vec3 n = cross(b - a, c - a);
        if (fabs(dot(n, p - a)) > 1e-7) return false;
        Vec3 c0 = cross(b - a, p - a);
        Vec3 c1 = cross(c - b, p - b);
        Vec3 c2 = cross(a - c, p - c);
        return dot(n, c0) >= -1e-7 && dot(n, c1) >= -1e-7 && dot(n, c2) >= -1e-7;
    };
    if (pointInTriPlane(ca, b0, b1, b2)) return true;
    if (pointInTriPlane(cb, a0, a1, a2)) return true;
    return false;
}

struct GridIndex {
    double cell = 1.0;
    Vec3 base;
    unordered_map<uint64_t, vector<int>, U64Hash> cells;
};

GridIndex buildGrid(const Mesh& m) {
    GridIndex g;
    g.base = m.box.mn;
    const double dx = max(EPS, m.box.mx.x - m.box.mn.x);
    const double dy = max(EPS, m.box.mx.y - m.box.mn.y);
    const double dz = max(EPS, m.box.mx.z - m.box.mn.z);
    const double volume = dx * dy * dz;
    const double faces = max<size_t>(1, m.f.size());
    g.cell = cbrt(volume / faces) * 2.5;
    if (!(g.cell > EPS) || !isfinite(g.cell)) g.cell = max({dx, dy, dz, 1.0});

    for (int i = 0; i < static_cast<int>(m.f.size()); ++i) {
        const auto& t = m.f[i];
        AABB tb;
        tb.expand(m.v[t.a]); tb.expand(m.v[t.b]); tb.expand(m.v[t.c]);
        int x0 = static_cast<int>(floor((tb.mn.x - g.base.x) / g.cell));
        int y0 = static_cast<int>(floor((tb.mn.y - g.base.y) / g.cell));
        int z0 = static_cast<int>(floor((tb.mn.z - g.base.z) / g.cell));
        int x1 = static_cast<int>(floor((tb.mx.x - g.base.x) / g.cell));
        int y1 = static_cast<int>(floor((tb.mx.y - g.base.y) / g.cell));
        int z1 = static_cast<int>(floor((tb.mx.z - g.base.z) / g.cell));
        for (int x = x0; x <= x1; ++x)
            for (int y = y0; y <= y1; ++y)
                for (int z = z0; z <= z1; ++z)
                    g.cells[mix3(x, y, z)].push_back(i);
    }
    return g;
}

bool meshesIntersect(const Mesh& a, const Mesh& b) {
    if (!boxesOverlap(a.box, b.box)) return false;
    const Mesh *small = &a, *large = &b;
    if (a.f.size() > b.f.size()) swap(small, large);
    GridIndex grid = buildGrid(*large);
    unordered_set<int> seen;
    for (const auto& t : small->f) {
        AABB tb;
        tb.expand(small->v[t.a]); tb.expand(small->v[t.b]); tb.expand(small->v[t.c]);
        int x0 = static_cast<int>(floor((tb.mn.x - grid.base.x) / grid.cell));
        int y0 = static_cast<int>(floor((tb.mn.y - grid.base.y) / grid.cell));
        int z0 = static_cast<int>(floor((tb.mn.z - grid.base.z) / grid.cell));
        int x1 = static_cast<int>(floor((tb.mx.x - grid.base.x) / grid.cell));
        int y1 = static_cast<int>(floor((tb.mx.y - grid.base.y) / grid.cell));
        int z1 = static_cast<int>(floor((tb.mx.z - grid.base.z) / grid.cell));
        seen.clear();
        for (int x = x0; x <= x1; ++x) {
            for (int y = y0; y <= y1; ++y) {
                for (int z = z0; z <= z1; ++z) {
                    auto it = grid.cells.find(mix3(x, y, z));
                    if (it == grid.cells.end()) continue;
                    for (int idx : it->second) {
                        if (!seen.insert(idx).second) continue;
                        const auto& u = large->f[idx];
                        if (triangleIntersectsTriangle(
                                small->v[t.a], small->v[t.b], small->v[t.c],
                                large->v[u.a], large->v[u.b], large->v[u.c])) {
                            return true;
                        }
                    }
                }
            }
        }
    }
    return false;
}

Mesh mergeDisjoint(const Mesh& a, const Mesh& b) {
    Mesh out;
    out.v = a.v;
    out.f = a.f;
    int off = static_cast<int>(out.v.size());
    out.v.insert(out.v.end(), b.v.begin(), b.v.end());
    for (const auto& t : b.f) out.f.push_back({t.a + off, t.b + off, t.c + off});
    out.box = computeBox(out);
    return out;
}

// Conservative manifold-preserving fallback:
// sample the occupied region on an adaptive regular grid, extract boundary quads,
// then triangulate. This is not an exact CSG boolean, but it is scalable and robust.
struct VoxelGrid {
    int nx = 0, ny = 0, nz = 0;
    Vec3 origin;
    double h = 1.0;
    vector<uint8_t> inside;

    int idx(int x, int y, int z) const {
        return (z * ny + y) * nx + x;
    }
    bool inBounds(int x, int y, int z) const {
        return x >= 0 && x < nx && y >= 0 && y < ny && z >= 0 && z < nz;
    }
    bool at(int x, int y, int z) const {
        return inside[idx(x, y, z)] != 0;
    }
};

VoxelGrid buildUnionVoxel(const Mesh& a, const Mesh& b) {
    AABB box = a.box;
    box.expand(b.box.mn);
    box.expand(b.box.mx);

    double dx = max(EPS, box.mx.x - box.mn.x);
    double dy = max(EPS, box.mx.y - box.mn.y);
    double dz = max(EPS, box.mx.z - box.mn.z);
    size_t totalFaces = a.f.size() + b.f.size();

    // Scale down voxel count for huge meshes to stay within memory/time budget.
    int targetLinear = 96;
    if (totalFaces < 5000) targetLinear = 160;
    else if (totalFaces < 50000) targetLinear = 112;
    else if (totalFaces > 500000) targetLinear = 72;

    double longest = max({dx, dy, dz});
    double h = longest / targetLinear;
    h = max(h, 1e-6);

    int nx = max(2, static_cast<int>(ceil(dx / h)) + 3);
    int ny = max(2, static_cast<int>(ceil(dy / h)) + 3);
    int nz = max(2, static_cast<int>(ceil(dz / h)) + 3);

    VoxelGrid g;
    g.nx = nx; g.ny = ny; g.nz = nz;
    g.h = h;
    g.origin = box.mn - Vec3{h, h, h};
    g.inside.assign(static_cast<size_t>(nx) * ny * nz, 0);

    for (int z = 0; z < nz; ++z) {
        for (int y = 0; y < ny; ++y) {
            for (int x = 0; x < nx; ++x) {
                Vec3 p = g.origin + Vec3{(x + 0.5) * h, (y + 0.5) * h, (z + 0.5) * h};
                if (pointInMesh(a, p) || pointInMesh(b, p)) {
                    g.inside[g.idx(x, y, z)] = 1;
                }
            }
        }
    }
    return g;
}

Mesh extractSurface(const VoxelGrid& g) {
    Mesh out;
    unordered_map<uint64_t, int, U64Hash> vmap;

    auto getVertex = [&](int xi, int yi, int zi) -> int {
        uint64_t key = (static_cast<uint64_t>(xi) << 42) ^
                       (static_cast<uint64_t>(yi) << 21) ^
                       static_cast<uint64_t>(zi);
        auto it = vmap.find(key);
        if (it != vmap.end()) return it->second;
        Vec3 p = g.origin + Vec3{xi * g.h, yi * g.h, zi * g.h};
        int id = static_cast<int>(out.v.size());
        out.v.push_back(p);
        vmap.emplace(key, id);
        return id;
    };

    auto addQuad = [&](array<array<int, 3>, 4> c, bool flip) {
        int i0 = getVertex(c[0][0], c[0][1], c[0][2]);
        int i1 = getVertex(c[1][0], c[1][1], c[1][2]);
        int i2 = getVertex(c[2][0], c[2][1], c[2][2]);
        int i3 = getVertex(c[3][0], c[3][1], c[3][2]);
        if (!flip) {
            out.f.push_back({i0, i1, i2});
            out.f.push_back({i0, i2, i3});
        } else {
            out.f.push_back({i0, i2, i1});
            out.f.push_back({i0, i3, i2});
        }
    };

    const int dx[6] = {-1, 1, 0, 0, 0, 0};
    const int dy[6] = {0, 0, -1, 1, 0, 0};
    const int dz[6] = {0, 0, 0, 0, -1, 1};

    for (int z = 0; z < g.nz; ++z) {
        for (int y = 0; y < g.ny; ++y) {
            for (int x = 0; x < g.nx; ++x) {
                if (!g.at(x, y, z)) continue;
                for (int dir = 0; dir < 6; ++dir) {
                    int nx = x + dx[dir], ny = y + dy[dir], nz = z + dz[dir];
                    bool outside = !g.inBounds(nx, ny, nz) || !g.at(nx, ny, nz);
                    if (!outside) continue;
                    if (dir == 0) addQuad({{{x, y, z}, {x, y, z + 1}, {x, y + 1, z + 1}, {x, y + 1, z}}}, true);
                    if (dir == 1) addQuad({{{x + 1, y, z}, {x + 1, y + 1, z}, {x + 1, y + 1, z + 1}, {x + 1, y, z + 1}}}, true);
                    if (dir == 2) addQuad({{{x, y, z}, {x + 1, y, z}, {x + 1, y, z + 1}, {x, y, z + 1}}}, true);
                    if (dir == 3) addQuad({{{x, y + 1, z}, {x, y + 1, z + 1}, {x + 1, y + 1, z + 1}, {x + 1, y + 1, z}}}, true);
                    if (dir == 4) addQuad({{{x, y, z}, {x, y + 1, z}, {x + 1, y + 1, z}, {x + 1, y, z}}}, true);
                    if (dir == 5) addQuad({{{x, y, z + 1}, {x + 1, y, z + 1}, {x + 1, y + 1, z + 1}, {x, y + 1, z + 1}}}, true);
                }
            }
        }
    }

    out.box = computeBox(out);
    return out;
}

Mesh solveUnion(const Mesh& a, const Mesh& b) {
    if (!boxesOverlap(a.box, b.box)) {
        return mergeDisjoint(a, b);
    }

    bool intersect = meshesIntersect(a, b);
    if (!intersect) {
        bool aInB = !a.v.empty() && pointInMesh(b, a.v[0]);
        bool bInA = !b.v.empty() && pointInMesh(a, b.v[0]);
        if (aInB) return b;
        if (bInA) return a;
        return mergeDisjoint(a, b);
    }

    VoxelGrid vox = buildUnionVoxel(a, b);
    return extractSurface(vox);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Mesh a, b;
    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;

    a.v.resize(VA);
    for (auto& p : a.v) cin >> p.x >> p.y >> p.z;
    a.f.resize(FA);
    for (auto& t : a.f) cin >> t.a >> t.b >> t.c;

    b.v.resize(VB);
    for (auto& p : b.v) cin >> p.x >> p.y >> p.z;
    b.f.resize(FB);
    for (auto& t : b.f) cin >> t.a >> t.b >> t.c;

    a.box = computeBox(a);
    b.box = computeBox(b);

    Mesh out = solveUnion(a, b);

    cout << out.v.size() << ' ' << out.f.size() << '\n';
    cout << setprecision(17);
    for (const auto& p : out.v) {
        cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    }
    for (const auto& t : out.f) {
        cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
    }
    return 0;
}
