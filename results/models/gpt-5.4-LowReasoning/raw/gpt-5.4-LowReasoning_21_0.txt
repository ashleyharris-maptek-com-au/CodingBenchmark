using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        uint64_t h = 1469598103934665603ull;
        for (int x : v) {
            uint64_t y = static_cast<uint64_t>(static_cast<int64_t>(x) + 0x9e3779b97f4a7c15ull);
            h ^= y + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        }
        return static_cast<size_t>(h);
    }
};

struct Move {
    int axis = 0;
    int dir = 1;
};

struct State {
    int D = 0;
    int max_turns = 0;
    vector<int> bounds;
    deque<vector<int>> body;
    unordered_set<vector<int>, VecHash> foods;
    unordered_set<vector<int>, VecHash> obstacles;
};

static long long cappedCellCount(const vector<int>& bounds, long long cap) {
    long long total = 1;
    for (int b : bounds) {
        if (b <= 0) {
            return 0;
        }
        if (total > cap / b) {
            return cap + 1;
        }
        total *= b;
    }
    return total;
}

static bool inBounds(const vector<int>& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) {
            return false;
        }
    }
    return true;
}

static long long manhattan(const vector<int>& a, const vector<int>& b) {
    long long d = 0;
    for (size_t i = 0; i < a.size(); ++i) {
        d += llabs(static_cast<long long>(a[i]) - static_cast<long long>(b[i]));
    }
    return d;
}

static vector<vector<int>> generateNeighbors(const vector<int>& p, const vector<int>& bounds) {
    vector<vector<int>> out;
    out.reserve(p.size() * 2);
    for (int axis = 0; axis < static_cast<int>(p.size()); ++axis) {
        if (p[axis] > 0) {
            vector<int> q = p;
            --q[axis];
            out.push_back(std::move(q));
        }
        if (p[axis] + 1 < bounds[axis]) {
            vector<int> q = p;
            ++q[axis];
            out.push_back(std::move(q));
        }
    }
    return out;
}

static vector<Move> allMoves(int D) {
    vector<Move> moves;
    moves.reserve(D * 2);
    for (int axis = 0; axis < D; ++axis) {
        moves.push_back({axis, -1});
        moves.push_back({axis, 1});
    }
    return moves;
}

static vector<int> applyMove(const vector<int>& p, const Move& m) {
    vector<int> q = p;
    q[m.axis] += m.dir;
    return q;
}

static unordered_set<vector<int>, VecHash> occupiedSet(const deque<vector<int>>& body) {
    unordered_set<vector<int>, VecHash> occ;
    occ.reserve(body.size() * 2 + 1);
    for (const auto& c : body) {
        occ.insert(c);
    }
    return occ;
}

static deque<vector<int>> initializeBody(int length,
                                         const vector<int>& head,
                                         const vector<int>& bounds,
                                         const unordered_set<vector<int>, VecHash>& obstacles) {
    deque<vector<int>> body;
    body.push_front(head);
    unordered_set<vector<int>, VecHash> used;
    used.insert(head);
    vector<Move> moves = allMoves(static_cast<int>(head.size()));
    while (static_cast<int>(body.size()) < length) {
        bool extended = false;
        const vector<int>& tail = body.back();
        for (const auto& m : moves) {
            vector<int> candidate = tail;
            candidate[m.axis] -= m.dir;
            if (!inBounds(candidate, bounds)) {
                continue;
            }
            if (obstacles.count(candidate) || used.count(candidate)) {
                continue;
            }
            body.push_back(candidate);
            used.insert(candidate);
            extended = true;
            break;
        }
        if (!extended) {
            break;
        }
    }
    return body;
}

static bool isBlocked(const vector<int>& p,
                      const unordered_set<vector<int>, VecHash>& blocked,
                      const unordered_set<vector<int>, VecHash>& obstacles) {
    return obstacles.count(p) || blocked.count(p);
}

static optional<vector<vector<int>>> bfsPath(const vector<int>& start,
                                             const vector<int>& goal,
                                             const vector<int>& bounds,
                                             const unordered_set<vector<int>, VecHash>& blocked,
                                             const unordered_set<vector<int>, VecHash>& obstacles,
                                             size_t expansion_limit) {
    if (start == goal) {
        return vector<vector<int>>{start};
    }
    queue<vector<int>> q;
    unordered_map<vector<int>, vector<int>, VecHash> parent;
    parent.reserve(min(expansion_limit * 2, static_cast<size_t>(1000003)));
    q.push(start);
    parent.emplace(start, vector<int>());
    size_t expanded = 0;
    while (!q.empty()) {
        vector<int> cur = q.front();
        q.pop();
        if (++expanded > expansion_limit) {
            return nullopt;
        }
        for (vector<int> nxt : generateNeighbors(cur, bounds)) {
            if (nxt != goal && isBlocked(nxt, blocked, obstacles)) {
                continue;
            }
            if (parent.find(nxt) != parent.end()) {
                continue;
            }
            parent.emplace(nxt, cur);
            if (nxt == goal) {
                vector<vector<int>> path;
                vector<int> walk = nxt;
                while (!walk.empty()) {
                    path.push_back(walk);
                    auto it = parent.find(walk);
                    if (it == parent.end() || it->second.empty()) {
                        break;
                    }
                    walk = it->second;
                }
                reverse(path.begin(), path.end());
                return path;
            }
            q.push(std::move(nxt));
        }
    }
    return nullopt;
}

static size_t reachableArea(const vector<int>& start,
                            const vector<int>& bounds,
                            const unordered_set<vector<int>, VecHash>& blocked,
                            const unordered_set<vector<int>, VecHash>& obstacles,
                            size_t limit) {
    if (isBlocked(start, blocked, obstacles)) {
        return 0;
    }
    queue<vector<int>> q;
    unordered_set<vector<int>, VecHash> seen;
    seen.reserve(limit * 2 + 1);
    q.push(start);
    seen.insert(start);
    while (!q.empty() && seen.size() < limit) {
        vector<int> cur = q.front();
        q.pop();
        for (vector<int> nxt : generateNeighbors(cur, bounds)) {
            if (isBlocked(nxt, blocked, obstacles) || seen.count(nxt)) {
                continue;
            }
            seen.insert(nxt);
            q.push(std::move(nxt));
            if (seen.size() >= limit) {
                break;
            }
        }
    }
    return seen.size();
}

static optional<Move> pathFirstMove(const vector<int>& head, const vector<int>& next) {
    for (int axis = 0; axis < static_cast<int>(head.size()); ++axis) {
        int diff = next[axis] - head[axis];
        if (diff == 1 || diff == -1) {
            return Move{axis, diff};
        }
    }
    return nullopt;
}

static vector<int> nearestFoodByManhattan(const vector<int>& head,
                                          const unordered_set<vector<int>, VecHash>& foods) {
    long long best = numeric_limits<long long>::max();
    vector<int> target;
    for (const auto& f : foods) {
        long long d = manhattan(head, f);
        if (d < best) {
            best = d;
            target = f;
        }
    }
    return target;
}

static bool simulateMove(const State& s, const Move& m, State& out, bool& grows) {
    out = s;
    const vector<int>& head = s.body.front();
    vector<int> nh = applyMove(head, m);
    if (!inBounds(nh, s.bounds) || s.obstacles.count(nh)) {
        return false;
    }

    bool eat = s.foods.count(nh) > 0;
    unordered_set<vector<int>, VecHash> occ = occupiedSet(s.body);
    if (!eat && !s.body.empty()) {
        occ.erase(s.body.back());
    }
    if (occ.count(nh)) {
        return false;
    }

    out.body.push_front(nh);
    if (eat) {
        out.foods.erase(nh);
    } else {
        out.body.pop_back();
    }
    grows = eat;
    return true;
}

static bool hasEscapeRoute(const State& s, size_t bfs_limit) {
    if (s.body.empty()) {
        return true;
    }
    unordered_set<vector<int>, VecHash> blocked = occupiedSet(s.body);
    blocked.erase(s.body.front());
    if (!s.body.empty()) {
        blocked.erase(s.body.back());
    }
    const vector<int>& tail = s.body.back();
    auto path = bfsPath(s.body.front(), tail, s.bounds, blocked, s.obstacles, bfs_limit);
    if (path.has_value()) {
        return true;
    }
    size_t space = reachableArea(s.body.front(), s.bounds, blocked, s.obstacles, 4096);
    return space >= min<size_t>(s.body.size() + 2, 64);
}

static Move chooseGreedyMove(const State& s) {
    const vector<int>& head = s.body.front();
    vector<int> target = s.foods.empty() ? s.body.back() : nearestFoodByManhattan(head, s.foods);
    vector<Move> moves = allMoves(s.D);

    long long best_score = numeric_limits<long long>::min();
    Move best_move = moves.front();
    bool found = false;

    for (const auto& m : moves) {
        State ns;
        bool grows = false;
        if (!simulateMove(s, m, ns, grows)) {
            continue;
        }

        long long dist = ns.foods.empty() ? manhattan(ns.body.front(), ns.body.back())
                                          : manhattan(ns.body.front(), target);
        unordered_set<vector<int>, VecHash> blocked = occupiedSet(ns.body);
        blocked.erase(ns.body.front());
        if (!ns.body.empty()) {
            blocked.erase(ns.body.back());
        }
        size_t area = reachableArea(ns.body.front(), ns.bounds, blocked, ns.obstacles, 512);
        int onward = 0;
        for (const auto& m2 : moves) {
            vector<int> p2 = applyMove(ns.body.front(), m2);
            if (!inBounds(p2, ns.bounds) || ns.obstacles.count(p2)) {
                continue;
            }
            if (blocked.count(p2)) {
                continue;
            }
            ++onward;
        }
        long long score = -dist * 1000000LL + static_cast<long long>(area) * 1024LL + onward * 64LL - (grows ? 1 : 0);
        if (!hasEscapeRoute(ns, 25000)) {
            score -= 1000000000LL;
        }
        if (!found || score > best_score) {
            found = true;
            best_score = score;
            best_move = m;
        }
    }

    if (found) {
        return best_move;
    }
    return moves.front();
}

static optional<Move> chooseSearchMove(const State& s, size_t bfs_limit) {
    if (s.foods.empty()) {
        return nullopt;
    }
    const vector<int>& head = s.body.front();
    vector<int> target = nearestFoodByManhattan(head, s.foods);

    unordered_set<vector<int>, VecHash> blocked = occupiedSet(s.body);
    blocked.erase(head);
    if (!s.body.empty()) {
        blocked.erase(s.body.back());
    }

    auto path = bfsPath(head, target, s.bounds, blocked, s.obstacles, bfs_limit);
    if (!path.has_value() || path->size() < 2) {
        return nullopt;
    }

    auto move = pathFirstMove(head, (*path)[1]);
    if (!move.has_value()) {
        return nullopt;
    }

    State ns;
    bool grows = false;
    if (!simulateMove(s, *move, ns, grows)) {
        return nullopt;
    }
    if (!hasEscapeRoute(ns, bfs_limit)) {
        return nullopt;
    }
    return move;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    State s;
    if (!(cin >> s.D >> s.max_turns)) {
        return 0;
    }

    s.bounds.resize(s.D);
    for (int i = 0; i < s.D; ++i) {
        cin >> s.bounds[i];
    }

    int snake_length = 0;
    vector<int> head(s.D);
    cin >> snake_length;
    for (int i = 0; i < s.D; ++i) {
        cin >> head[i];
    }

    int F = 0;
    cin >> F;
    for (int i = 0; i < F; ++i) {
        vector<int> food(s.D);
        for (int j = 0; j < s.D; ++j) {
            cin >> food[j];
        }
        s.foods.insert(food);
    }

    int O = 0;
    cin >> O;
    for (int i = 0; i < O; ++i) {
        vector<int> obstacle(s.D);
        for (int j = 0; j < s.D; ++j) {
            cin >> obstacle[j];
        }
        s.obstacles.insert(obstacle);
    }

    s.body = initializeBody(max(1, snake_length), head, s.bounds, s.obstacles);
    if (s.body.empty()) {
        s.body.push_front(head);
    }

    const long long total_cells = cappedCellCount(s.bounds, 2000000LL);
    const bool search_friendly = total_cells <= 2000000LL;
    const size_t bfs_limit = search_friendly ? 300000 : 40000;

    vector<Move> output;
    output.reserve(s.max_turns);

    for (int turn = 0; turn < s.max_turns && !s.foods.empty(); ++turn) {
        optional<Move> chosen;
        if (search_friendly) {
            chosen = chooseSearchMove(s, bfs_limit);
        }
        if (!chosen.has_value()) {
            chosen = chooseGreedyMove(s);
        }

        State ns;
        bool grows = false;
        if (!simulateMove(s, *chosen, ns, grows)) {
            bool found = false;
            for (const auto& m : allMoves(s.D)) {
                if (simulateMove(s, m, ns, grows)) {
                    chosen = m;
                    found = true;
                    break;
                }
            }
            if (!found) {
                break;
            }
        }

        output.push_back(*chosen);
        s = std::move(ns);
    }

    for (const auto& m : output) {
        cout << m.axis << ' ' << m.dir << '\n';
    }
    return 0;
}
