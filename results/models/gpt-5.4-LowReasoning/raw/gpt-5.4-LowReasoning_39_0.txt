use std::io::{self, Read};

struct Solver {
    n: usize,
    target: usize,
    adj: Vec<Vec<usize>>,
    degree: Vec<usize>,
    order: Vec<usize>,
    assign: Vec<i8>, // -1 = unassigned, 0 = B, 1 = A
    to_a: Vec<i32>,
    to_b: Vec<i32>,
    best_cut: i32,
    best_part: Vec<usize>,
}

impl Solver {
    fn new(n: usize, adj: Vec<Vec<usize>>) -> Self {
        let degree = adj.iter().map(|v| v.len()).collect::<Vec<_>>();
        let mut order = (1..n).collect::<Vec<_>>();
        order.sort_by_key(|&v| std::cmp::Reverse(degree[v]));
        Self {
            n,
            target: n / 2,
            adj,
            degree,
            order,
            assign: vec![-1; n],
            to_a: vec![0; n],
            to_b: vec![0; n],
            best_cut: i32::MAX,
            best_part: Vec::new(),
        }
    }

    fn compute_cut_of_partition(&self, in_a: &[bool]) -> i32 {
        let mut cut = 0i32;
        for u in 0..self.n {
            for &v in &self.adj[u] {
                if u < v && in_a[u] != in_a[v] {
                    cut += 1;
                }
            }
        }
        cut
    }

    fn seed_upper_bound(&mut self) {
        let mut verts = (0..self.n).collect::<Vec<_>>();
        verts.sort_by_key(|&v| self.degree[v]);

        let mut in_a = vec![false; self.n];
        in_a[0] = true;
        let mut chosen = 1usize;

        for &v in &verts {
            if v == 0 {
                continue;
            }
            if chosen < self.target {
                in_a[v] = true;
                chosen += 1;
            }
        }

        let mut improved = true;
        while improved {
            improved = false;
            let a_side = (0..self.n).filter(|&v| in_a[v]).collect::<Vec<_>>();
            let b_side = (0..self.n).filter(|&v| !in_a[v]).collect::<Vec<_>>();
            let current = self.compute_cut_of_partition(&in_a);

            'outer: for &a in &a_side {
                if a == 0 {
                    continue;
                }
                for &b in &b_side {
                    in_a[a] = false;
                    in_a[b] = true;
                    let next = self.compute_cut_of_partition(&in_a);
                    if next < current {
                        improved = true;
                        break 'outer;
                    }
                    in_a[a] = true;
                    in_a[b] = false;
                }
            }
        }

        self.best_cut = self.compute_cut_of_partition(&in_a);
        self.best_part = (0..self.n).filter(|&v| in_a[v]).collect();
        self.best_part.sort_unstable();
    }

    fn lower_bound(&self, a_count: usize, current_cut: i32) -> i32 {
        let need_a = self.target - a_count;
        let mut base = 0i32;
        let mut deltas = Vec::with_capacity(self.n);

        for v in 0..self.n {
            if self.assign[v] == -1 {
                base += self.to_a[v];
                deltas.push(self.to_b[v] - self.to_a[v]);
            }
        }

        deltas.sort_unstable();
        let extra = deltas.into_iter().take(need_a).sum::<i32>();
        current_cut + base + extra
    }

    fn choose_vertex(&self) -> Option<usize> {
        self.order.iter().copied().find(|&v| self.assign[v] == -1)
    }

    fn record_solution(&mut self, cut: i32) {
        let mut part = Vec::with_capacity(self.target);
        for v in 0..self.n {
            if self.assign[v] == 1 {
                part.push(v);
            }
        }
        part.sort_unstable();
        self.best_cut = cut;
        self.best_part = part;
    }

    fn branch_order(&self, v: usize) -> [i8; 2] {
        if self.to_b[v] <= self.to_a[v] {
            [1, 0]
        } else {
            [0, 1]
        }
    }

    fn dfs(&mut self, a_count: usize, b_count: usize, current_cut: i32) {
        if current_cut >= self.best_cut {
            return;
        }
        if a_count > self.target || b_count > self.target {
            return;
        }

        let unassigned = self.n - a_count - b_count;
        let need_a = self.target - a_count;
        let need_b = self.target - b_count;
        if need_a > unassigned || need_b > unassigned {
            return;
        }

        let lb = self.lower_bound(a_count, current_cut);
        if lb >= self.best_cut {
            return;
        }

        if unassigned == 0 {
            self.record_solution(current_cut);
            return;
        }

        let v = self.choose_vertex().unwrap();
        let branch_order = self.branch_order(v);

        for side in branch_order {
            if side == 1 && a_count == self.target {
                continue;
            }
            if side == 0 && b_count == self.target {
                continue;
            }

            self.assign[v] = side;
            let added = if side == 1 { self.to_b[v] } else { self.to_a[v] };

            for &u in &self.adj[v] {
                if self.assign[u] == -1 {
                    if side == 1 {
                        self.to_a[u] += 1;
                    } else {
                        self.to_b[u] += 1;
                    }
                }
            }

            if side == 1 {
                self.dfs(a_count + 1, b_count, current_cut + added);
            } else {
                self.dfs(a_count, b_count + 1, current_cut + added);
            }

            for &u in &self.adj[v] {
                if self.assign[u] == -1 {
                    if side == 1 {
                        self.to_a[u] -= 1;
                    } else {
                        self.to_b[u] -= 1;
                    }
                }
            }
            self.assign[v] = -1;
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut solver = Solver::new(n, adj);
    solver.seed_upper_bound();

    solver.assign[0] = 1;
    for &u in &solver.adj[0] {
        solver.to_a[u] += 1;
    }

    solver.dfs(1, 0, 0);

    println!("{}", solver.best_cut);
    for (i, v) in solver.best_part.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
