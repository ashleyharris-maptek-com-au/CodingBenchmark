use std::cmp::Ordering;
use std::io::{self, Read};

#[derive(Clone)]
struct Solver {
    n: usize,
    target_a: usize,
    adj: Vec<Vec<usize>>,
    side: Vec<i8>, // -1 = unassigned, 0 = A, 1 = B
    to_a: Vec<i32>,
    to_b: Vec<i32>,
    a_vertices: Vec<usize>,
    best_cut: i32,
    best_a: Vec<usize>,
}

impl Solver {
    fn new(n: usize, adj: Vec<Vec<usize>>) -> Self {
        Self {
            n,
            target_a: n / 2,
            adj,
            side: vec![-1; n],
            to_a: vec![0; n],
            to_b: vec![0; n],
            a_vertices: Vec::with_capacity(n / 2),
            best_cut: i32::MAX,
            best_a: Vec::new(),
        }
    }

    fn solve(&mut self) {
        self.greedy_upper_bound();
        self.assign(0, 0);
        self.dfs(1, 0, 0);
    }

    fn assign(&mut self, v: usize, s: i8) {
        self.side[v] = s;
        if s == 0 {
            self.a_vertices.push(v);
            for &u in &self.adj[v] {
                self.to_a[u] += 1;
            }
        } else {
            for &u in &self.adj[v] {
                self.to_b[u] += 1;
            }
        }
    }

    fn unassign(&mut self, v: usize, s: i8) {
        self.side[v] = -1;
        if s == 0 {
            self.a_vertices.pop();
            for &u in &self.adj[v] {
                self.to_a[u] -= 1;
            }
        } else {
            for &u in &self.adj[v] {
                self.to_b[u] -= 1;
            }
        }
    }

    fn choose_vertex(&self) -> usize {
        let mut best_v = usize::MAX;
        let mut best_key = (i32::MIN, i32::MIN, i32::MIN);
        for v in 0..self.n {
            if self.side[v] != -1 {
                continue;
            }
            let decided_neighbors = self.to_a[v] + self.to_b[v];
            let bias = (self.to_a[v] - self.to_b[v]).abs();
            let key = (decided_neighbors, bias, self.adj[v].len() as i32);
            if key > best_key {
                best_key = key;
                best_v = v;
            }
        }
        best_v
    }

    fn lower_bound(&self, cut_so_far: i32, a_count: usize, assigned: usize) -> i32 {
        let rem = self.n - assigned;
        let need_a = self.target_a - a_count;
        if need_a > rem {
            return i32::MAX / 4;
        }
        let need_b = rem - need_a;
        if need_b > rem {
            return i32::MAX / 4;
        }

        let mut base = 0i32;
        let mut diffs = Vec::with_capacity(rem);
        for v in 0..self.n {
            if self.side[v] == -1 {
                let cost_a = self.to_b[v];
                let cost_b = self.to_a[v];
                base += cost_b;
                diffs.push(cost_a - cost_b);
            }
        }
        diffs.sort_unstable();
        let mut extra = 0i32;
        for &d in diffs.iter().take(need_a) {
            extra += d;
        }
        cut_so_far + base + extra
    }

    fn record_solution(&mut self, cut: i32) {
        if cut < self.best_cut {
            self.best_cut = cut;
            self.best_a = self.a_vertices.clone();
            self.best_a.sort_unstable();
        }
    }

    fn dfs(&mut self, assigned: usize, a_count: usize, cut_so_far: i32) {
        if assigned == self.n {
            if a_count == self.target_a {
                self.record_solution(cut_so_far);
            }
            return;
        }

        let lb = self.lower_bound(cut_so_far, a_count, assigned);
        if lb >= self.best_cut {
            return;
        }

        let rem = self.n - assigned;
        let need_a = self.target_a - a_count;
        if need_a > rem {
            return;
        }
        if need_a == rem {
            let verts: Vec<usize> = (0..self.n).filter(|&v| self.side[v] == -1).collect();
            let mut cut = cut_so_far;
            for v in verts {
                cut += self.to_b[v];
                self.assign(v, 0);
            }
            self.record_solution(cut);
            for &v in self.a_vertices.clone()[a_count..].iter().rev() {
                self.unassign(v, 0);
            }
            return;
        }
        if need_a == 0 {
            let verts: Vec<usize> = (0..self.n).filter(|&v| self.side[v] == -1).collect();
            let mut cut = cut_so_far;
            for &v in &verts {
                cut += self.to_a[v];
                self.assign(v, 1);
            }
            self.record_solution(cut);
            for &v in verts.iter().rev() {
                self.unassign(v, 1);
            }
            return;
        }

        let v = self.choose_vertex();
        let cost_a = self.to_b[v];
        let cost_b = self.to_a[v];

        let mut order = [(0i8, cost_a), (1i8, cost_b)];
        order.sort_by(|x, y| {
            let primary = x.1.cmp(&y.1);
            if primary != Ordering::Equal {
                return primary;
            }
            x.0.cmp(&y.0)
        });

        for (s, add) in order {
            if s == 0 && a_count == self.target_a {
                continue;
            }
            if s == 1 && assigned - a_count == self.target_a {
                continue;
            }
            self.assign(v, s);
            self.dfs(assigned + 1, a_count + (s == 0) as usize, cut_so_far + add);
            self.unassign(v, s);
        }
    }

    fn greedy_upper_bound(&mut self) {
        let mut side = vec![-1i8; self.n];
        side[0] = 0;
        let mut a_count = 1usize;
        let mut assigned = 1usize;
        let mut to_a = vec![0i32; self.n];
        let mut to_b = vec![0i32; self.n];
        for &u in &self.adj[0] {
            to_a[u] += 1;
        }

        while assigned < self.n {
            let rem = self.n - assigned;
            let need_a = self.target_a - a_count;
            let mut best_v = usize::MAX;
            let mut best_s = 0i8;
            let mut best_score = i32::MAX;

            for v in 0..self.n {
                if side[v] != -1 {
                    continue;
                }
                let can_a = need_a > 0;
                let can_b = rem > need_a;

                if can_a {
                    let score = to_b[v];
                    if score < best_score {
                        best_score = score;
                        best_v = v;
                        best_s = 0;
                    }
                }
                if can_b {
                    let score = to_a[v];
                    if score < best_score {
                        best_score = score;
                        best_v = v;
                        best_s = 1;
                    }
                }
            }

            side[best_v] = best_s;
            if best_s == 0 {
                a_count += 1;
                for &u in &self.adj[best_v] {
                    to_a[u] += 1;
                }
            } else {
                for &u in &self.adj[best_v] {
                    to_b[u] += 1;
                }
            }
            assigned += 1;
        }

        let mut cut = 0i32;
        let mut part_a = Vec::with_capacity(self.target_a);
        for v in 0..self.n {
            if side[v] == 0 {
                part_a.push(v);
                for &u in &self.adj[v] {
                    if side[u] == 1 {
                        cut += 1;
                    }
                }
            }
        }

        self.best_cut = cut;
        self.best_a = part_a;
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }

    let mut solver = Solver::new(n, adj);
    solver.solve();

    println!("{}", solver.best_cut);
    for (i, v) in solver.best_a.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
