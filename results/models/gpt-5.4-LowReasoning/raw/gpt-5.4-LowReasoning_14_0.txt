using System;
using System.Collections.Generic;
using System.Linq;

public static class Program
{
    const byte STATE_UNKNOWN = 0;
    const byte STATE_SAFE = 1;
    const byte STATE_MINE = 2;

    static readonly int[] DX = { -1, 0, 1, -1, 1, -1, 0, 1 };
    static readonly int[] DY = { -1, -1, -1, 0, 0, 1, 1, 1 };

    static int H, W;
    static char[][] board = Array.Empty<char[]>();
    static byte[] state = Array.Empty<byte>();

    public static void Main()
    {
        var first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.WriteLine(0);
            return;
        }

        var parts = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        board = new char[H][];
        state = new byte[H * W];

        for (int y = 0; y < H; y++)
        {
            var line = Console.ReadLine() ?? string.Empty;
            if (line.Length < W) line = line.PadRight(W, ' ');
            if (line.Length > W) line = line.Substring(0, W);
            board[y] = line.ToCharArray();

            for (int x = 0; x < W; x++)
            {
                char c = board[y][x];
                if (c == '*') state[Id(x, y)] = STATE_MINE;
            }
        }

        Solve();

        var safeCells = new List<(int x, int y)>();
        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                int id = Id(x, y);
                if (board[y][x] == ' ' && state[id] == STATE_SAFE)
                    safeCells.Add((x, y));
            }
        }

        Console.WriteLine(safeCells.Count);
        foreach (var (x, y) in safeCells)
            Console.WriteLine($"{x} {y}");
    }

    static void Solve()
    {
        long cells = (long)H * W;
        int exactVarLimit = cells <= 10_000 ? 28 : cells <= 1_000_000 ? 24 : 20;
        int exactConstraintLimit = 80;

        while (true)
        {
            var context = BuildConstraints();
            bool progress = false;

            progress |= ApplyBasicRules(context.Constraints);
            progress |= ApplySubsetRules(context.Constraints, context.ConstraintCellsByVar);

            if (!progress)
                progress |= ApplyExactComponentRules(context, exactVarLimit, exactConstraintLimit);

            if (!progress)
                break;
        }
    }

    static SolverContext BuildConstraints()
    {
        var constraints = new List<Constraint>();
        var constraintCellsByVar = new Dictionary<int, List<int>>();
        var frontierVars = new HashSet<int>();

        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                char c = board[y][x];
                if (c < '0' || c > '8') continue;

                int clue = c - '0';
                int knownMines = 0;
                var unknowns = new List<int>(8);

                for (int d = 0; d < 8; d++)
                {
                    int nx = x + DX[d];
                    int ny = y + DY[d];
                    if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;

                    int nid = Id(nx, ny);
                    byte st = state[nid];
                    if (st == STATE_MINE) knownMines++;
                    else if (board[ny][nx] == ' ' && st == STATE_UNKNOWN) unknowns.Add(nid);
                }

                int need = clue - knownMines;
                if (unknowns.Count == 0) continue;
                if (need < 0) continue;

                unknowns.Sort();
                var cons = new Constraint(unknowns.ToArray(), need);
                int ci = constraints.Count;
                constraints.Add(cons);

                foreach (int v in cons.Vars)
                {
                    frontierVars.Add(v);
                    if (!constraintCellsByVar.TryGetValue(v, out var list))
                    {
                        list = new List<int>(4);
                        constraintCellsByVar[v] = list;
                    }
                    list.Add(ci);
                }
            }
        }

        return new SolverContext(constraints, constraintCellsByVar, frontierVars);
    }

    static bool ApplyBasicRules(List<Constraint> constraints)
    {
        bool progress = false;
        foreach (var c in constraints)
        {
            if (c.Need == 0)
            {
                foreach (int v in c.Vars)
                    progress |= MarkSafe(v);
            }
            else if (c.Need == c.Vars.Length)
            {
                foreach (int v in c.Vars)
                    progress |= MarkMine(v);
            }
        }
        return progress;
    }

    static bool ApplySubsetRules(List<Constraint> constraints, Dictionary<int, List<int>> byVar)
    {
        bool progress = false;
        var seenPairs = new HashSet<long>();

        for (int i = 0; i < constraints.Count; i++)
        {
            var a = constraints[i];
            if (a.Vars.Length == 0) continue;

            var related = new HashSet<int>();
            foreach (int v in a.Vars)
            {
                if (!byVar.TryGetValue(v, out var list)) continue;
                foreach (int j in list)
                    if (j != i) related.Add(j);
            }

            foreach (int j in related)
            {
                long key = ((long)Math.Min(i, j) << 32) | (uint)Math.Max(i, j);
                if (!seenPairs.Add(key)) continue;

                var b = constraints[j];
                progress |= ApplySubsetPair(a, b);
                progress |= ApplySubsetPair(b, a);
            }
        }

        return progress;
    }

    static bool ApplySubsetPair(Constraint sub, Constraint sup)
    {
        if (sub.Vars.Length >= sup.Vars.Length) return false;

        int i = 0, j = 0;
        var diff = new List<int>(sup.Vars.Length);
        while (i < sub.Vars.Length && j < sup.Vars.Length)
        {
            int a = sub.Vars[i], b = sup.Vars[j];
            if (a == b) { i++; j++; }
            else if (a > b) { diff.Add(b); j++; }
            else return false;
        }
        if (i != sub.Vars.Length) return false;
        while (j < sup.Vars.Length) diff.Add(sup.Vars[j++]);

        int needDiff = sup.Need - sub.Need;
        if (needDiff < 0 || needDiff > diff.Count) return false;

        bool progress = false;
        if (needDiff == 0)
        {
            foreach (int v in diff) progress |= MarkSafe(v);
        }
        else if (needDiff == diff.Count)
        {
            foreach (int v in diff) progress |= MarkMine(v);
        }
        return progress;
    }

    static bool ApplyExactComponentRules(SolverContext context, int exactVarLimit, int exactConstraintLimit)
    {
        bool progress = false;
        var visitedVars = new HashSet<int>();
        var visitedConstraints = new bool[context.Constraints.Count];

        foreach (int startVar in context.FrontierVars)
        {
            if (!visitedVars.Add(startVar)) continue;

            var compVars = new List<int>();
            var compCons = new List<int>();
            var q = new Queue<int>();
            q.Enqueue(startVar);

            while (q.Count > 0)
            {
                int v = q.Dequeue();
                compVars.Add(v);

                if (!context.ConstraintCellsByVar.TryGetValue(v, out var cons)) continue;
                foreach (int ci in cons)
                {
                    if (!visitedConstraints[ci])
                    {
                        visitedConstraints[ci] = true;
                        compCons.Add(ci);
                    }

                    foreach (int nv in context.Constraints[ci].Vars)
                    {
                        if (visitedVars.Add(nv))
                            q.Enqueue(nv);
                    }
                }
            }

            if (compVars.Count == 0) continue;
            if (compVars.Count > exactVarLimit || compCons.Count > exactConstraintLimit) continue;

            progress |= SolveExactComponent(context.Constraints, compVars, compCons);
        }

        return progress;
    }

    static bool SolveExactComponent(List<Constraint> allConstraints, List<int> compVars, List<int> compCons)
    {
        int n = compVars.Count;
        var localIndex = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++) localIndex[compVars[i]] = i;

        var equations = new List<Equation>(compCons.Count);
        var varToEq = new List<int>[n];
        for (int i = 0; i < n; i++) varToEq[i] = new List<int>(4);

        for (int ei = 0; ei < compCons.Count; ei++)
        {
            var c = allConstraints[compCons[ei]];
            var vars = new int[c.Vars.Length];
            for (int k = 0; k < c.Vars.Length; k++)
                vars[k] = localIndex[c.Vars[k]];

            Array.Sort(vars);
            equations.Add(new Equation(vars, c.Need));
            foreach (int v in vars) varToEq[v].Add(ei);
        }

        int[] values = new int[n];
        Array.Fill(values, -1);

        int[] assigned = new int[equations.Count];
        int[] mines = new int[equations.Count];

        long solutions = 0;
        long[] mineCounts = new long[n];

        bool Dfs()
        {
            if (solutions > 2_000_000) return true;

            int nextVar = -1;
            int bestScore = -1;
            for (int v = 0; v < n; v++)
            {
                if (values[v] != -1) continue;
                int score = varToEq[v].Count;
                if (score > bestScore)
                {
                    bestScore = score;
                    nextVar = v;
                }
            }

            if (nextVar == -1)
            {
                for (int ei = 0; ei < equations.Count; ei++)
                    if (mines[ei] != equations[ei].Need) return false;

                solutions++;
                for (int v = 0; v < n; v++)
                    if (values[v] == 1) mineCounts[v]++;
                return false;
            }

            bool shouldStop = false;
            TryAssign(nextVar, 0, ref shouldStop);
            if (!shouldStop) TryAssign(nextVar, 1, ref shouldStop);
            return shouldStop;
        }

        void TryAssign(int var, int val, ref bool shouldStop)
        {
            values[var] = val;
            var touched = varToEq[var];
            foreach (int ei in touched)
            {
                assigned[ei]++;
                if (val == 1) mines[ei]++;
            }

            bool ok = true;
            foreach (int ei in touched)
            {
                int minPossible = mines[ei];
                int maxPossible = mines[ei] + (equations[ei].Vars.Length - assigned[ei]);
                int need = equations[ei].Need;
                if (need < minPossible || need > maxPossible)
                {
                    ok = false;
                    break;
                }
            }

            if (ok) shouldStop = Dfs();

            foreach (int ei in touched)
            {
                assigned[ei]--;
                if (val == 1) mines[ei]--;
            }
            values[var] = -1;
        }

        Dfs();

        if (solutions == 0) return false;

        bool progress = false;
        for (int i = 0; i < n; i++)
        {
            if (mineCounts[i] == 0)
                progress |= MarkSafe(compVars[i]);
            else if (mineCounts[i] == solutions)
                progress |= MarkMine(compVars[i]);
        }

        return progress;
    }

    static bool MarkSafe(int id)
    {
        if (state[id] != STATE_UNKNOWN) return false;
        state[id] = STATE_SAFE;
        return true;
    }

    static bool MarkMine(int id)
    {
        if (state[id] == STATE_MINE) return false;
        if (state[id] == STATE_SAFE) return false;
        state[id] = STATE_MINE;
        return true;
    }

    static int Id(int x, int y) => y * W + x;

    readonly struct Constraint
    {
        public readonly int[] Vars;
        public readonly int Need;

        public Constraint(int[] vars, int need)
        {
            Vars = vars;
            Need = need;
        }
    }

    readonly struct Equation
    {
        public readonly int[] Vars;
        public readonly int Need;

        public Equation(int[] vars, int need)
        {
            Vars = vars;
            Need = need;
        }
    }

    sealed class SolverContext
    {
        public readonly List<Constraint> Constraints;
        public readonly Dictionary<int, List<int>> ConstraintCellsByVar;
        public readonly HashSet<int> FrontierVars;

        public SolverContext(
            List<Constraint> constraints,
            Dictionary<int, List<int>> constraintCellsByVar,
            HashSet<int> frontierVars)
        {
            Constraints = constraints;
            ConstraintCellsByVar = constraintCellsByVar;
            FrontierVars = frontierVars;
        }
    }
}
