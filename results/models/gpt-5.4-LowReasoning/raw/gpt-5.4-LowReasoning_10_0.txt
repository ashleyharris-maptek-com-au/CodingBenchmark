using namespace std;

struct RectInput {
    int id = -1;
    int w = 0;
    int h = 0;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
    bool placed = false;
};

struct PackedResult {
    int boards = numeric_limits<int>::max();
    vector<Placement> placements;
    bool feasible = false;
};

static inline long long rect_area(int w, int h) {
    return 1LL * w * h;
}

static vector<RectInput> make_order(const vector<RectInput>& rects, int mode) {
    vector<RectInput> order = rects;
    auto by_area = [](const RectInput& a, const RectInput& b) {
        long long aa = rect_area(a.w, a.h);
        long long ab = rect_area(b.w, b.h);
        if (aa != ab) return aa > ab;
        if (max(a.w, a.h) != max(b.w, b.h)) return max(a.w, a.h) > max(b.w, b.h);
        if (min(a.w, a.h) != min(b.w, b.h)) return min(a.w, a.h) > min(b.w, b.h);
        return a.id < b.id;
    };
    auto by_height = [](const RectInput& a, const RectInput& b) {
        if (max(a.h, a.w) != max(b.h, b.w)) return max(a.h, a.w) > max(b.h, b.w);
        if (min(a.h, a.w) != min(b.h, b.w)) return min(a.h, a.w) > min(b.h, b.w);
        long long aa = rect_area(a.w, a.h);
        long long ab = rect_area(b.w, b.h);
        if (aa != ab) return aa > ab;
        return a.id < b.id;
    };
    auto by_perimeter = [](const RectInput& a, const RectInput& b) {
        int pa = a.w + a.h;
        int pb = b.w + b.h;
        if (pa != pb) return pa > pb;
        long long aa = rect_area(a.w, a.h);
        long long ab = rect_area(b.w, b.h);
        if (aa != ab) return aa > ab;
        return a.id < b.id;
    };

    if (mode == 0) sort(order.begin(), order.end(), by_area);
    else if (mode == 1) sort(order.begin(), order.end(), by_height);
    else sort(order.begin(), order.end(), by_perimeter);
    return order;
}

struct FreeRect {
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
};

struct GuillotineBoard {
    vector<FreeRect> freeRects;
};

struct GuillotineTryResult {
    bool feasible = false;
    vector<Placement> placements;
};

static void prune_free_list(vector<FreeRect>& freeRects) {
    vector<FreeRect> pruned;
    pruned.reserve(freeRects.size());
    for (size_t i = 0; i < freeRects.size(); ++i) {
        bool contained = false;
        for (size_t j = 0; j < freeRects.size(); ++j) {
            if (i == j) continue;
            const auto& a = freeRects[i];
            const auto& b = freeRects[j];
            if (a.x >= b.x && a.y >= b.y &&
                a.x + a.w <= b.x + b.w &&
                a.y + a.h <= b.y + b.h) {
                contained = true;
                break;
            }
        }
        if (!contained && freeRects[i].w > 0 && freeRects[i].h > 0) {
            pruned.push_back(freeRects[i]);
        }
    }
    freeRects.swap(pruned);
}

static GuillotineTryResult try_pack_guillotine(
    const vector<RectInput>& order,
    int boardW,
    int boardH,
    int boardCount
) {
    vector<GuillotineBoard> boards(boardCount);
    for (int i = 0; i < boardCount; ++i) {
        boards[i].freeRects.push_back({0, 0, boardW, boardH});
    }

    vector<Placement> placements(order.size());
    for (const auto& rect : order) {
        long long bestScore0 = numeric_limits<long long>::max();
        long long bestScore1 = numeric_limits<long long>::max();
        int bestBoard = -1;
        int bestFree = -1;
        int bestRot = 0;
        int bestW = 0;
        int bestH = 0;

        for (int bi = 0; bi < boardCount; ++bi) {
            auto& freeRects = boards[bi].freeRects;
            for (int fi = 0; fi < static_cast<int>(freeRects.size()); ++fi) {
                const auto& fr = freeRects[fi];
                for (int rot = 0; rot < 2; ++rot) {
                    int rw = rot ? rect.h : rect.w;
                    int rh = rot ? rect.w : rect.h;
                    if (rw > fr.w || rh > fr.h) continue;

                    int shortSide = min(fr.w - rw, fr.h - rh);
                    int longSide = max(fr.w - rw, fr.h - rh);
                    long long waste = 1LL * fr.w * fr.h - 1LL * rw * rh;
                    long long score0 = 1LL * shortSide * 1000000000LL + longSide;
                    long long score1 = waste;

                    if (score0 < bestScore0 || (score0 == bestScore0 && score1 < bestScore1)) {
                        bestScore0 = score0;
                        bestScore1 = score1;
                        bestBoard = bi;
                        bestFree = fi;
                        bestRot = rot;
                        bestW = rw;
                        bestH = rh;
                    }
                }
            }
        }

        if (bestBoard < 0) {
            return {false, {}};
        }

        auto& freeRects = boards[bestBoard].freeRects;
        FreeRect chosen = freeRects[bestFree];
        freeRects[bestFree] = freeRects.back();
        freeRects.pop_back();

        placements[rect.id] = {bestBoard, chosen.x, chosen.y, bestRot, true};

        int remW = chosen.w - bestW;
        int remH = chosen.h - bestH;

        bool splitHorizontal = (remW <= remH);
        if (splitHorizontal) {
            if (remW > 0) freeRects.push_back({chosen.x + bestW, chosen.y, remW, bestH});
            if (remH > 0) freeRects.push_back({chosen.x, chosen.y + bestH, chosen.w, remH});
        } else {
            if (remW > 0) freeRects.push_back({chosen.x + bestW, chosen.y, remW, chosen.h});
            if (remH > 0) freeRects.push_back({chosen.x, chosen.y + bestH, bestW, remH});
        }

        if (static_cast<int>(freeRects.size()) <= 256) {
            prune_free_list(freeRects);
        }
    }

    return {true, placements};
}

struct Shelf {
    int y = 0;
    int height = 0;
    int usedWidth = 0;
};

struct ShelfBoard {
    vector<Shelf> shelves;
    int usedHeight = 0;
};

static PackedResult pack_shelf(
    const vector<RectInput>& order,
    int boardW,
    int boardH
) {
    vector<ShelfBoard> boards;
    boards.reserve(max(1, static_cast<int>(order.size() / 16)));
    vector<Placement> placements(order.size());

    for (const auto& rect : order) {
        int bestBoard = -1;
        int bestShelf = -1;
        int bestRot = 0;
        long long bestScore = numeric_limits<long long>::max();

        for (int bi = 0; bi < static_cast<int>(boards.size()); ++bi) {
            auto& board = boards[bi];
            for (int si = 0; si < static_cast<int>(board.shelves.size()); ++si) {
                const auto& shelf = board.shelves[si];
                for (int rot = 0; rot < 2; ++rot) {
                    int rw = rot ? rect.h : rect.w;
                    int rh = rot ? rect.w : rect.h;
                    if (rh > shelf.height) continue;
                    if (shelf.usedWidth + rw > boardW) continue;
                    int widthSlack = boardW - (shelf.usedWidth + rw);
                    int heightSlack = shelf.height - rh;
                    long long score = 1LL * widthSlack * 1000000LL + heightSlack;
                    if (score < bestScore) {
                        bestScore = score;
                        bestBoard = bi;
                        bestShelf = si;
                        bestRot = rot;
                    }
                }
            }
        }

        if (bestBoard >= 0) {
            auto& shelf = boards[bestBoard].shelves[bestShelf];
            int rw = bestRot ? rect.h : rect.w;
            int rh = bestRot ? rect.w : rect.h;
            placements[rect.id] = {bestBoard, shelf.usedWidth, shelf.y, bestRot, true};
            shelf.usedWidth += rw;
            (void)rh;
            continue;
        }

        bestBoard = -1;
        bestRot = 0;
        bestScore = numeric_limits<long long>::max();

        for (int bi = 0; bi < static_cast<int>(boards.size()); ++bi) {
            auto& board = boards[bi];
            for (int rot = 0; rot < 2; ++rot) {
                int rw = rot ? rect.h : rect.w;
                int rh = rot ? rect.w : rect.h;
                if (rw > boardW || rh > boardH) continue;
                if (board.usedHeight + rh > boardH) continue;
                long long score = 1LL * (boardH - (board.usedHeight + rh)) * 1000000LL + (boardW - rw);
                if (score < bestScore) {
                    bestScore = score;
                    bestBoard = bi;
                    bestRot = rot;
                }
            }
        }

        if (bestBoard < 0) {
            ShelfBoard newBoard;
            boards.push_back(newBoard);
            bestBoard = static_cast<int>(boards.size()) - 1;

            int area0 = (rect.w <= boardW && rect.h <= boardH) ? (boardW - rect.w) + (boardH - rect.h) : numeric_limits<int>::max();
            int area1 = (rect.h <= boardW && rect.w <= boardH) ? (boardW - rect.h) + (boardH - rect.w) : numeric_limits<int>::max();
            if (area1 < area0) bestRot = 1;
            else bestRot = 0;
        }

        auto& board = boards[bestBoard];
        int rw = bestRot ? rect.h : rect.w;
        int rh = bestRot ? rect.w : rect.h;
        if (rw > boardW || rh > boardH || board.usedHeight + rh > boardH) {
            return {};
        }

        int y = board.usedHeight;
        board.shelves.push_back({y, rh, rw});
        board.usedHeight += rh;
        placements[rect.id] = {bestBoard, 0, y, bestRot, true};
    }

    PackedResult result;
    result.boards = static_cast<int>(boards.size());
    result.placements = std::move(placements);
    result.feasible = true;
    return result;
}

static PackedResult solve_instance(const vector<RectInput>& rects, int boardW, int boardH) {
    PackedResult best;
    best.feasible = false;

    long long totalArea = 0;
    for (const auto& r : rects) totalArea += rect_area(r.w, r.h);
    long long boardArea = rect_area(boardW, boardH);
    int lowerBound = static_cast<int>((totalArea + boardArea - 1) / boardArea);
    lowerBound = max(lowerBound, 1);

    const int n = static_cast<int>(rects.size());
    bool useGuillotine = (n <= 2500);

    if (useGuillotine) {
        for (int mode = 0; mode < 3; ++mode) {
            vector<RectInput> order = make_order(rects, mode);
            PackedResult upper = pack_shelf(order, boardW, boardH);
            if (!upper.feasible) continue;

            int lo = lowerBound;
            int hi = upper.boards;
            vector<Placement> bestPlacement = upper.placements;
            int bestBoards = upper.boards;

            while (lo <= hi) {
                int mid = lo + (hi - lo) / 2;
                auto attempt = try_pack_guillotine(order, boardW, boardH, mid);
                if (attempt.feasible) {
                    bestBoards = mid;
                    bestPlacement = std::move(attempt.placements);
                    hi = mid - 1;
                } else {
                    lo = mid + 1;
                }
            }

            if (!best.feasible || bestBoards < best.boards) {
                best.feasible = true;
                best.boards = bestBoards;
                best.placements = std::move(bestPlacement);
            }
        }
    } else {
        for (int mode = 0; mode < 2; ++mode) {
            vector<RectInput> order = make_order(rects, mode);
            PackedResult candidate = pack_shelf(order, boardW, boardH);
            if (!candidate.feasible) continue;
            if (!best.feasible || candidate.boards < best.boards) {
                best = std::move(candidate);
            }
        }
    }

    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n = 0;
    int boardW = 0;
    int boardH = 0;
    if (!(cin >> n >> boardW >> boardH)) {
        return 0;
    }

    vector<RectInput> rects;
    rects.reserve(n);
    for (int i = 0; i < n; ++i) {
        int w = 0, h = 0;
        cin >> w >> h;
        rects.push_back({i, w, h});
    }

    for (const auto& r : rects) {
        bool fits = (r.w <= boardW && r.h <= boardH) || (r.h <= boardW && r.w <= boardH);
        if (!fits) {
            cout << -1 << '\n';
            return 0;
        }
    }

    PackedResult result = solve_instance(rects, boardW, boardH);
    if (!result.feasible) {
        cout << -1 << '\n';
        return 0;
    }

    cout << result.boards << '\n';
    for (int i = 0; i < n; ++i) {
        const auto& p = result.placements[i];
        cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }
    return 0;
}
