use std::cmp::Reverse;
use std::collections::{HashMap, VecDeque};
use std::io::{self, Read};
use std::time::{Duration, Instant};

const DX: [i32; 4] = [1, -1, 0, 0];
const DY: [i32; 4] = [0, 0, 1, -1];

#[derive(Clone)]
struct Solver {
    width: i32,
    height: i32,
    require_cycle: bool,
    n: usize,
    coords: Vec<(i32, i32)>,
    adj: Vec<Vec<usize>>,
    color: Vec<u8>,
    start: usize,
    deadline: Instant,
    best: Vec<usize>,
}

impl Solver {
    fn solve(&mut self) -> Option<Vec<usize>> {
        if self.n == 0 {
            return None;
        }
        if self.n == 1 {
            return if self.require_cycle { None } else { Some(vec![self.start]) };
        }
        if !self.basic_feasible() {
            return None;
        }
        if let Some(path) = self.try_full_rectangle() {
            return Some(path);
        }

        let mut visited = vec![false; self.n];
        let mut path = Vec::with_capacity(self.n);
        visited[self.start] = true;
        path.push(self.start);
        self.best = path.clone();

        if self.dfs(self.start, &mut visited, &mut path, 0) {
            Some(path)
        } else {
            None
        }
    }

    fn basic_feasible(&self) -> bool {
        if self.start >= self.n {
            return false;
        }
        if !self.connected_with_subset(&vec![true; self.n], self.start) {
            return false;
        }
        if self.n > 1 && self.adj[self.start].is_empty() {
            return false;
        }

        let mut black = 0usize;
        let mut white = 0usize;
        for &c in &self.color {
            if c == 0 { black += 1; } else { white += 1; }
        }
        if self.require_cycle {
            if self.adj[self.start].is_empty() {
                return false;
            }
            if black != white {
                return false;
            }
        } else {
            let diff = black.abs_diff(white);
            if diff > 1 {
                return false;
            }
            let start_color = self.color[self.start] as usize;
            if black > white && start_color != 0 {
                return false;
            }
            if white > black && start_color != 1 {
                return false;
            }
        }

        let deg1 = self.adj.iter().filter(|v| v.len() == 1).count();
        if self.require_cycle {
            if self.adj.iter().any(|v| v.len() < 2) {
                return false;
            }
        } else if deg1 > 2 {
            return false;
        } else if deg1 == 2 && self.adj[self.start].len() == 1 {
            let only_deg1_is_start = self.adj.iter().enumerate().all(|(i, v)| v.len() != 1 || i == self.start || deg1 == 2);
            if only_deg1_is_start {
                return false;
            }
        }
        true
    }

    fn try_full_rectangle(&self) -> Option<Vec<usize>> {
        if self.n != (self.width as usize) * (self.height as usize) {
            return None;
        }
        if self.start != 0 {
            return None;
        }

        if self.require_cycle {
            if self.width > 1 && self.height > 1 {
                if self.width % 2 == 0 {
                    return Some(self.rectangle_cycle(false));
                }
                if self.height % 2 == 0 {
                    return Some(self.rectangle_cycle(true));
                }
            }
            return None;
        }

        Some(self.rectangle_path())
    }

    fn rectangle_path(&self) -> Vec<usize> {
        let mut order = Vec::with_capacity(self.n);
        if self.height == 1 {
            for x in 0..self.width {
                order.push((x, 0));
            }
        } else if self.width == 1 {
            for y in 0..self.height {
                order.push((0, y));
            }
        } else if self.height % 2 == 0 {
            for y in 0..self.height {
                if y % 2 == 0 {
                    for x in 0..self.width {
                        order.push((x, y));
                    }
                } else {
                    for x in (0..self.width).rev() {
                        order.push((x, y));
                    }
                }
            }
        } else {
            for x in 0..self.width {
                if x % 2 == 0 {
                    for y in 0..self.height {
                        order.push((x, y));
                    }
                } else {
                    for y in (0..self.height).rev() {
                        order.push((x, y));
                    }
                }
            }
        }
        self.coords_to_indices(&order)
    }

    fn rectangle_cycle(&self, transpose: bool) -> Vec<usize> {
        let (w, h) = if transpose { (self.height, self.width) } else { (self.width, self.height) };
        let mut order = Vec::with_capacity(self.n);
        for x in 0..w {
            if x == 0 {
                order.push((0, 0));
                for xx in 1..w {
                    order.push((xx, 0));
                }
            }
        }
        for x in (1..w).rev() {
            if ((w - 1) - x) % 2 == 0 {
                for y in 1..h {
                    order.push((x, y));
                }
            } else {
                for y in (1..h).rev() {
                    order.push((x, y));
                }
            }
        }
        for y in (1..h).rev() {
            order.push((0, y));
        }

        let mapped: Vec<(i32, i32)> = if transpose {
            order.into_iter().map(|(x, y)| (y, x)).collect()
        } else {
            order
        };
        self.coords_to_indices(&mapped)
    }

    fn coords_to_indices(&self, pts: &[(i32, i32)]) -> Vec<usize> {
        let mut pos = HashMap::with_capacity(self.coords.len());
        for (i, &p) in self.coords.iter().enumerate() {
            pos.insert(p, i);
        }
        pts.iter().map(|p| pos[p]).collect()
    }

    fn dfs(&mut self, current: usize, visited: &mut [bool], path: &mut Vec<usize>, depth_hint: usize) -> bool {
        if Instant::now() >= self.deadline {
            return false;
        }
        if path.len() > self.best.len() {
            self.best = path.clone();
        }
        if path.len() == self.n {
            return !self.require_cycle || self.adj[current].contains(&self.start);
        }

        if !self.remaining_feasible(current, visited, path.len(), depth_hint) {
            return false;
        }

        let mut nexts: Vec<usize> = self.adj[current].iter().copied().filter(|&v| !visited[v]).collect();
        nexts.sort_by_key(|&v| {
            let onward = self.adj[v].iter().filter(|&&u| !visited[u]).count();
            let manhattan = (self.coords[v].0 - self.coords[self.start].0).abs() + (self.coords[v].1 - self.coords[self.start].1).abs();
            (onward, Reverse(manhattan))
        });

        if nexts.len() > 1 {
            nexts.sort_by_key(|&v| self.forced_score(v, current, visited));
        }

        for v in nexts {
            visited[v] = true;
            path.push(v);
            if self.dfs(v, visited, path, depth_hint + 1) {
                return true;
            }
            path.pop();
            visited[v] = false;
        }
        false
    }

    fn forced_score(&self, v: usize, current: usize, visited: &[bool]) -> (usize, usize, i32) {
        let mut deg = 0usize;
        let mut forced = 0usize;
        for &u in &self.adj[v] {
            if !visited[u] && u != current {
                deg += 1;
                let rem = self.adj[u].iter().filter(|&&w| !visited[w] && w != v).count();
                if rem <= 1 {
                    forced += 1;
                }
            }
        }
        let dist = (self.coords[v].0 - self.coords[self.start].0).abs() + (self.coords[v].1 - self.coords[self.start].1).abs();
        (forced, deg, -dist)
    }

    fn remaining_feasible(&self, current: usize, visited: &[bool], used: usize, depth_hint: usize) -> bool {
        let remaining = self.n - used;
        if remaining == 0 {
            return true;
        }

        let mut allowed = vec![false; self.n];
        for i in 0..self.n {
            allowed[i] = !visited[i];
        }
        if !self.connected_with_subset(&allowed, current) {
            return false;
        }

        let mut deg1 = 0usize;
        for i in 0..self.n {
            if visited[i] {
                continue;
            }
            let avail = self.adj[i]
                .iter()
                .filter(|&&v| !visited[v] || v == current || (self.require_cycle && v == self.start && remaining == 1))
                .count();
            if avail == 0 {
                return false;
            }
            if avail == 1 {
                deg1 += 1;
                if self.require_cycle {
                    return false;
                }
            }
        }
        if !self.require_cycle && deg1 > 1 {
            return false;
        }

        if depth_hint % 8 == 0 {
            let mut black = 0usize;
            let mut white = 0usize;
            for i in 0..self.n {
                if !visited[i] {
                    if self.color[i] == 0 { black += 1; } else { white += 1; }
                }
            }
            let diff = black.abs_diff(white);
            if diff > 1 {
                return false;
            }
        }
        true
    }

    fn connected_with_subset(&self, allowed: &[bool], current: usize) -> bool {
        let start = if allowed[current] {
            current
        } else {
            match allowed.iter().position(|&b| b) {
                Some(v) => v,
                None => return true,
            }
        };
        let need = allowed.iter().filter(|&&b| b).count();
        if need == 0 {
            return true;
        }
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        seen[start] = true;
        q.push_back(start);
        let mut count = 0usize;
        while let Some(v) = q.pop_front() {
            count += 1;
            for &u in &self.adj[v] {
                if allowed[u] && !seen[u] {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        count == need
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        println!("0");
        return;
    }
    let mut it = input.split_whitespace();
    let width: i32 = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let height: i32 = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let obstacle_count: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let require_cycle = match it.next().and_then(|s| s.parse::<i32>().ok()) {
        Some(v) => v != 0,
        None => {
            println!("0");
            return;
        }
    };

    if width <= 0 || height <= 0 {
        println!("0");
        return;
    }

    let mut blocked = vec![false; (width as usize) * (height as usize)];
    let idx = |x: i32, y: i32| -> usize { (y as usize) * (width as usize) + (x as usize) };
    for _ in 0..obstacle_count {
        let x: i32 = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => {
                println!("0");
                return;
            }
        };
        let y: i32 = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => {
                println!("0");
                return;
            }
        };
        if x < 0 || x >= width || y < 0 || y >= height {
            println!("0");
            return;
        }
        blocked[idx(x, y)] = true;
    }

    if blocked[idx(0, 0)] {
        println!("0");
        return;
    }

    let mut map = HashMap::new();
    let mut coords = Vec::new();
    for y in 0..height {
        for x in 0..width {
            if !blocked[idx(x, y)] {
                let id = coords.len();
                coords.push((x, y));
                map.insert((x, y), id);
            }
        }
    }

    let n = coords.len();
    if n == 0 {
        println!("0");
        return;
    }
    let start = match map.get(&(0, 0)).copied() {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut adj = vec![Vec::new(); n];
    let mut color = vec![0u8; n];
    for (i, &(x, y)) in coords.iter().enumerate() {
        color[i] = ((x + y) & 1) as u8;
        for d in 0..4 {
            let nx = x + DX[d];
            let ny = y + DY[d];
            if nx >= 0 && nx < width && ny >= 0 && ny < height && !blocked[idx(nx, ny)] {
                adj[i].push(map[&(nx, ny)]);
            }
        }
    }

    let seconds = if n <= 80 { 295 } else if n <= 200 { 220 } else { 90 };
    let deadline = Instant::now() + Duration::from_secs(seconds);
    let mut solver = Solver {
        width,
        height,
        require_cycle,
        n,
        coords,
        adj,
        color,
        start,
        deadline,
        best: Vec::new(),
    };

    if let Some(path) = solver.solve() {
        println!("{}", path.len());
        for v in path {
            let (x, y) = solver.coords[v];
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}
