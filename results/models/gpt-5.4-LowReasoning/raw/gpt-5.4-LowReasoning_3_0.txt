use std::cmp::Ordering;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

fn splitmix64(mut x: u64) -> u64 {
    x = x.wrapping_add(0x9E3779B97F4A7C15);
    x = (x ^ (x >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
    x = (x ^ (x >> 27)).wrapping_mul(0x94D049BB133111EB);
    x ^ (x >> 31)
}

fn hash01(seed: u64) -> f64 {
    let v = splitmix64(seed);
    ((v >> 11) as f64) * (1.0 / ((1u64 << 53) as f64))
}

fn bfs_farthest(start: usize, nodes: &[usize], adj: &[Vec<usize>], mark: &[bool]) -> (usize, Vec<i32>) {
    let n = adj.len();
    let mut dist = vec![-1i32; n];
    let mut q = VecDeque::new();
    dist[start] = 0;
    q.push_back(start);
    let mut best = start;
    while let Some(u) = q.pop_front() {
        if dist[u] > dist[best] || (dist[u] == dist[best] && adj[u].len() > adj[best].len()) {
            best = u;
        }
        for &v in &adj[u] {
            if mark[v] && dist[v] == -1 {
                dist[v] = dist[u] + 1;
                q.push_back(v);
            }
        }
    }
    debug_assert!(nodes.iter().all(|&x| dist[x] >= 0));
    (best, dist)
}

fn build_component_order(nodes: &[usize], adj: &[Vec<usize>]) -> (Vec<usize>, Vec<i32>) {
    let n = adj.len();
    let mut mark = vec![false; n];
    for &u in nodes {
        mark[u] = true;
    }
    let seed = *nodes
        .iter()
        .max_by_key(|&&u| (adj[u].len(), std::cmp::Reverse(u)))
        .unwrap();
    let (a, _) = bfs_farthest(seed, nodes, adj, &mark);
    let (root, dist) = bfs_farthest(a, nodes, adj, &mark);

    let mut parent = vec![usize::MAX; n];
    let mut q = VecDeque::new();
    let mut seen = vec![false; n];
    seen[root] = true;
    q.push_back(root);
    while let Some(u) = q.pop_front() {
        for &v in &adj[u] {
            if mark[v] && !seen[v] {
                seen[v] = true;
                parent[v] = u;
                q.push_back(v);
            }
        }
    }

    let mut order = nodes.to_vec();
    order.sort_by(|&u, &v| {
        let du = dist[u];
        let dv = dist[v];
        if du != dv {
            return du.cmp(&dv);
        }
        let pu = if parent[u] == usize::MAX { root } else { parent[u] };
        let pv = if parent[v] == usize::MAX { root } else { parent[v] };
        if pu != pv {
            return pu.cmp(&pv);
        }
        let deg_cmp = adj[v].len().cmp(&adj[u].len());
        if deg_cmp != Ordering::Equal {
            return deg_cmp;
        }
        u.cmp(&v)
    });
    (order, dist)
}

fn init_layered_positions(
    nodes: &[usize],
    order: &[usize],
    dist: &[i32],
    global_to_local: &[usize],
) -> (Vec<f64>, Vec<f64>) {
    let k = nodes.len();
    let mut pos = vec![(0.0f64, 0.0f64); k];
    let max_d = order.iter().map(|&u| dist[u]).max().unwrap_or(0).max(0) as usize;
    let mut layers = vec![Vec::<usize>::new(); max_d + 1];
    for &u in order {
        layers[dist[u] as usize].push(u);
    }

    let radial_step = 2.5;
    for (d, layer) in layers.iter().enumerate() {
        if layer.is_empty() {
            continue;
        }
        let radius = if d == 0 { 0.0 } else { d as f64 * radial_step };
        let span = (layer.len() as f64).sqrt().clamp(1.0, 32.0);
        let angle_span = (2.0 * PI / span).clamp(0.6, 2.0 * PI);
        let base = (d as f64 * 0.61803398875) % 1.0 * 2.0 * PI;
        for (i, &u) in layer.iter().enumerate() {
            let ang = if layer.len() == 1 {
                base
            } else {
                let t = i as f64 / (layer.len() - 1) as f64;
                base + (t - 0.5) * angle_span
            };
            let idx = global_to_local[u];
            let jitter = ((i * 2 + d + 1) as f64).ln_1p() * 0.03;
            pos[idx] = (
                radius * ang.cos() + jitter * (1.7 * ang).sin(),
                radius * ang.sin() + jitter * (1.3 * ang).cos(),
            );
        }
    }
    let (xs, ys): (Vec<_>, Vec<_>) = pos.into_iter().unzip();
    (xs, ys)
}

fn fruchterman_reingold(
    nodes: &[usize],
    comp_edges: &[(usize, usize)],
    global_to_local: &[usize],
    order: &[usize],
    dist: &[i32],
) -> (Vec<f64>, Vec<f64>) {
    let k = nodes.len();
    let (mut x, mut y) = init_layered_positions(nodes, order, dist, global_to_local);
    let area = (k.max(2) as f64) * 25.0;
    let kk = (area / k.max(1) as f64).sqrt();
    let mut dx = vec![0.0f64; k];
    let mut dy = vec![0.0f64; k];
    let iters = if k <= 16 { 200 } else if k <= 32 { 140 } else { 100 };
    let mut temp = (k as f64).sqrt() * 2.0 + 1.0;

    for _ in 0..iters {
        dx.fill(0.0);
        dy.fill(0.0);

        for i in 0..k {
            for j in (i + 1)..k {
                let vx = x[i] - x[j];
                let vy = y[i] - y[j];
                let dist2 = vx * vx + vy * vy + 1e-9;
                let dist = dist2.sqrt();
                let force = kk * kk / dist;
                let fx = vx / dist * force;
                let fy = vy / dist * force;
                dx[i] += fx;
                dy[i] += fy;
                dx[j] -= fx;
                dy[j] -= fy;
            }
        }

        for &(u, v) in comp_edges {
            let i = global_to_local[u];
            let j = global_to_local[v];
            if i == j {
                continue;
            }
            let vx = x[i] - x[j];
            let vy = y[i] - y[j];
            let dist2 = vx * vx + vy * vy + 1e-9;
            let dist = dist2.sqrt();
            let force = dist2 / kk.max(1e-6);
            let fx = vx / dist * force;
            let fy = vy / dist * force;
            dx[i] -= fx;
            dy[i] -= fy;
            dx[j] += fx;
            dy[j] += fy;
        }

        for i in 0..k {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt();
            if disp > 0.0 {
                let scale = temp.min(disp) / disp;
                x[i] += dx[i] * scale;
                y[i] += dy[i] * scale;
            }
        }
        temp *= 0.94;
    }
    (x, y)
}

fn sampled_force_layout(
    nodes: &[usize],
    comp_edges: &[(usize, usize)],
    global_to_local: &[usize],
    order: &[usize],
    dist: &[i32],
    degrees: &[usize],
) -> (Vec<f64>, Vec<f64>) {
    let k = nodes.len();
    let (mut x, mut y) = init_layered_positions(nodes, order, dist, global_to_local);
    let mut dx = vec![0.0f64; k];
    let mut dy = vec![0.0f64; k];

    let samples = if k <= 256 { 16 } else if k <= 1024 { 12 } else { 8 };
    let iters = if k <= 512 { 80 } else if k <= 2000 { 56 } else { 32 };
    let edge_stride = (comp_edges.len() / (k * 6).max(1)).max(1);
    let natural = (k as f64).sqrt().max(4.0) * 0.9;
    let mut step = natural * 0.14 + 0.3;

    for iter in 0..iters {
        dx.fill(0.0);
        dy.fill(0.0);

        for i in 0..k {
            let seed = (nodes[i] as u64) ^ ((iter as u64) << 32);
            for s in 0..samples {
                let j = (splitmix64(seed ^ (s as u64 * 0x9E3779B9)) as usize) % k;
                if i == j {
                    continue;
                }
                let vx = x[i] - x[j];
                let vy = y[i] - y[j];
                let dist2 = vx * vx + vy * vy + 1e-6;
                let inv = 1.0 / dist2;
                let force = natural * natural * inv;
                dx[i] += vx * force;
                dy[i] += vy * force;
            }
        }

        for (ei, &(u, v)) in comp_edges.iter().enumerate() {
            if ei % edge_stride != 0 {
                continue;
            }
            let i = global_to_local[u];
            let j = global_to_local[v];
            if i == j {
                continue;
            }
            let vx = x[i] - x[j];
            let vy = y[i] - y[j];
            let dist2 = vx * vx + vy * vy + 1e-6;
            let dist = dist2.sqrt();
            let target = natural * (0.85 + 0.05 * ((degrees[u] + degrees[v]) as f64).ln_1p());
            let force = (dist - target) * 0.12;
            let fx = vx / dist * force;
            let fy = vy / dist * force;
            dx[i] -= fx;
            dy[i] -= fy;
            dx[j] += fx;
            dy[j] += fy;
        }

        for i in 0..k {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt();
            if disp > 0.0 {
                let scale = step.min(disp) / disp;
                x[i] += dx[i] * scale;
                y[i] += dy[i] * scale;
            }
        }
        step *= 0.96;
    }
    (x, y)
}

fn huge_component_layout(
    nodes: &[usize],
    order: &[usize],
    dist: &[i32],
    global_to_local: &[usize],
    degrees: &[usize],
) -> (Vec<f64>, Vec<f64>) {
    let k = nodes.len();
    let mut x = vec![0.0f64; k];
    let mut y = vec![0.0f64; k];
    let max_d = order.iter().map(|&u| dist[u]).max().unwrap_or(0).max(0) as usize;
    let mut layers = vec![Vec::<usize>::new(); max_d + 1];
    for &u in order {
        layers[dist[u] as usize].push(u);
    }

    let radial_step = 3.0;
    for (d, layer) in layers.iter_mut().enumerate() {
        if layer.is_empty() {
            continue;
        }
        layer.sort_by(|&a, &b| {
            degrees[b]
                .cmp(&degrees[a])
                .then_with(|| a.cmp(&b))
        });
        let radius = if d == 0 { 0.0 } else { d as f64 * radial_step };
        let base = ((d as f64) * 0.7548776662466927) % 1.0 * 2.0 * PI;
        for (i, &u) in layer.iter().enumerate() {
            let idx = global_to_local[u];
            let ang = base + 2.0 * PI * (i as f64) / (layer.len() as f64);
            let wobble = 0.18 * hash01((u as u64) ^ ((d as u64) << 20));
            x[idx] = (radius + wobble) * ang.cos();
            y[idx] = (radius + wobble) * ang.sin();
        }
    }
    (x, y)
}

fn normalize_component(x: &mut [f64], y: &mut [f64]) -> (f64, f64) {
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for i in 0..x.len() {
        min_x = min_x.min(x[i]);
        max_x = max_x.max(x[i]);
        min_y = min_y.min(y[i]);
        max_y = max_y.max(y[i]);
    }
    let cx = 0.5 * (min_x + max_x);
    let cy = 0.5 * (min_y + max_y);
    for i in 0..x.len() {
        x[i] -= cx;
        y[i] -= cy;
    }
    (max_x - min_x, max_y - min_y)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut edges = Vec::with_capacity(m);
    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        edges.push(Edge { u, v });
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let degrees: Vec<usize> = adj.iter().map(|v| v.len()).collect();

    let mut comp_id = vec![usize::MAX; n];
    let mut components: Vec<Vec<usize>> = Vec::new();
    for s in 0..n {
        if comp_id[s] != usize::MAX {
            continue;
        }
        let cid = components.len();
        let mut nodes = Vec::new();
        let mut q = VecDeque::new();
        q.push_back(s);
        comp_id[s] = cid;
        while let Some(u) = q.pop_front() {
            nodes.push(u);
            for &v in &adj[u] {
                if comp_id[v] == usize::MAX {
                    comp_id[v] = cid;
                    q.push_back(v);
                }
            }
        }
        components.push(nodes);
    }

    let mut comp_edges = vec![Vec::<(usize, usize)>::new(); components.len()];
    for e in &edges {
        if e.u >= n || e.v >= n || e.u == e.v {
            continue;
        }
        let cid = comp_id[e.u];
        if cid == comp_id[e.v] {
            comp_edges[cid].push((e.u, e.v));
        }
    }

    let mut ans_x = vec![0.0f64; n];
    let mut ans_y = vec![0.0f64; n];
    let mut boxes = vec![(0.0f64, 0.0f64); components.len()];

    let mut global_to_local = vec![usize::MAX; n];
    for (cid, nodes) in components.iter().enumerate() {
        for (i, &u) in nodes.iter().enumerate() {
            global_to_local[u] = i;
        }
        let (order, dist) = build_component_order(nodes, &adj);
        let k = nodes.len();
        let ecount = comp_edges[cid].len();
        let (mut x, mut y) = if k <= 64 {
            fruchterman_reingold(nodes, &comp_edges[cid], &global_to_local, &order, &dist)
        } else if k <= 4000 {
            sampled_force_layout(
                nodes,
                &comp_edges[cid],
                &global_to_local,
                &order,
                &dist,
                &degrees,
            )
        } else if ecount <= k.saturating_mul(3) && k <= 20000 {
            sampled_force_layout(
                nodes,
                &comp_edges[cid],
                &global_to_local,
                &order,
                &dist,
                &degrees,
            )
        } else {
            huge_component_layout(nodes, &order, &dist, &global_to_local, &degrees)
        };
        let (w, h) = normalize_component(&mut x, &mut y);
        boxes[cid] = (w.max(1.0), h.max(1.0));
        for (i, &u) in nodes.iter().enumerate() {
            ans_x[u] = x[i];
            ans_y[u] = y[i];
        }
    }

    let total_area: f64 = boxes.iter().map(|(w, h)| (w + 8.0) * (h + 8.0)).sum();
    let target_row = total_area.sqrt().max(1.0);
    let mut order_cids: Vec<usize> = (0..components.len()).collect();
    order_cids.sort_by(|&a, &b| {
        let aa = boxes[a].0 * boxes[a].1;
        let bb = boxes[b].0 * boxes[b].1;
        bb.partial_cmp(&aa).unwrap_or(Ordering::Equal)
    });

    let mut placements = vec![(0.0f64, 0.0f64); components.len()];
    let mut cur_x = 0.0;
    let mut cur_y = 0.0;
    let mut row_h = 0.0;
    for cid in order_cids {
        let (w, h) = boxes[cid];
        let pw = w + 8.0;
        let ph = h + 8.0;
        if cur_x > 0.0 && cur_x + pw > target_row {
            cur_x = 0.0;
            cur_y += row_h;
            row_h = 0.0;
        }
        placements[cid] = (cur_x + w * 0.5, cur_y + h * 0.5);
        cur_x += pw;
        row_h = row_h.max(ph);
    }

    let mut out = String::new();
    for cid in 0..components.len() {
        let (ox, oy) = placements[cid];
        for &u in &components[cid] {
            out.push_str(&format!("{:.6} {:.6}\n", ans_x[u] + ox, ans_y[u] + oy));
        }
    }

    // Output must be in node order, not component order.
    out.clear();
    for u in 0..n {
        let (ox, oy) = placements[comp_id[u]];
        out.push_str(&format!("{:.6} {:.6}\n", ans_x[u] + ox, ans_y[u] + oy));
    }
    print!("{out}");
}
