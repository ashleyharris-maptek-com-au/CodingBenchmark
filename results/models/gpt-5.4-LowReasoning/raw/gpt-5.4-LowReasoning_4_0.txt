constexpr double EPS = 1e-9;

struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
    Vec3& operator+=(const Vec3& o) { x += o.x; y += o.y; z += o.z; return *this; }
};

double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

double norm2(const Vec3& v) { return dot(v, v); }
double norm(const Vec3& v) { return std::sqrt(norm2(v)); }

struct Plane {
    Vec3 n;
    double d = 0.0; // inside if dot(n, p) + d <= 0
};

struct AABB {
    Vec3 lo, hi;
};

struct Hull {
    vector<Plane> planes;
    AABB box;
    bool exact = false;
};

Vec3 computeCentroid(const vector<Vec3>& pts) {
    Vec3 c;
    for (const auto& p : pts) c += p;
    if (!pts.empty()) c = c / static_cast<double>(pts.size());
    return c;
}

AABB computeAABB(const vector<Vec3>& pts) {
    AABB box;
    box.lo = {numeric_limits<double>::infinity(),
              numeric_limits<double>::infinity(),
              numeric_limits<double>::infinity()};
    box.hi = {-numeric_limits<double>::infinity(),
              -numeric_limits<double>::infinity(),
              -numeric_limits<double>::infinity()};
    for (const auto& p : pts) {
        box.lo.x = min(box.lo.x, p.x);
        box.lo.y = min(box.lo.y, p.y);
        box.lo.z = min(box.lo.z, p.z);
        box.hi.x = max(box.hi.x, p.x);
        box.hi.y = max(box.hi.y, p.y);
        box.hi.z = max(box.hi.z, p.z);
    }
    return box;
}

Hull makeAABBHull(const vector<Vec3>& pts) {
    Hull h;
    h.box = computeAABB(pts);
    const auto& lo = h.box.lo;
    const auto& hi = h.box.hi;
    h.planes = {
        {{ 1, 0, 0}, -hi.x},
        {{-1, 0, 0},  lo.x},
        {{ 0, 1, 0}, -hi.y},
        {{ 0,-1, 0},  lo.y},
        {{ 0, 0, 1}, -hi.z},
        {{ 0, 0,-1},  lo.z},
    };
    h.exact = false;
    return h;
}

bool pointInsidePlanes(const vector<Plane>& planes, const Vec3& p, double shrink) {
    for (const auto& pl : planes) {
        if (dot(pl.n, p) + pl.d + shrink > EPS) return false;
    }
    return true;
}

Hull buildConvexHullBrute(const vector<Vec3>& pts) {
    Hull hull;
    hull.box = computeAABB(pts);
    const int n = static_cast<int>(pts.size());
    if (n < 4) return makeAABBHull(pts);

    Vec3 center = computeCentroid(pts);
    set<tuple<int, int, int>> used;
    vector<Plane> planes;

    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 normal = cross(pts[j] - pts[i], pts[k] - pts[i]);
                double len = norm(normal);
                if (len < 1e-10) continue;
                normal = normal / len;
                double d = -dot(normal, pts[i]);

                int pos = 0, neg = 0;
                for (int t = 0; t < n; ++t) {
                    double s = dot(normal, pts[t]) + d;
                    if (s > 1e-8) ++pos;
                    else if (s < -1e-8) ++neg;
                    if (pos && neg) break;
                }
                if (pos && neg) continue;

                if (dot(normal, center) + d > 0.0) {
                    normal = normal * -1.0;
                    d = -d;
                }

                vector<int> coplanar;
                coplanar.reserve(n);
                for (int t = 0; t < n; ++t) {
                    double s = dot(normal, pts[t]) + d;
                    if (std::abs(s) <= 1e-7) coplanar.push_back(t);
                }
                if (coplanar.size() < 3) continue;
                sort(coplanar.begin(), coplanar.end());

                bool inserted = false;
                for (size_t a = 0; a < coplanar.size() && !inserted; ++a) {
                    for (size_t b = a + 1; b < coplanar.size() && !inserted; ++b) {
                        for (size_t c = b + 1; c < coplanar.size() && !inserted; ++c) {
                            auto key = make_tuple(coplanar[a], coplanar[b], coplanar[c]);
                            if (used.insert(key).second) {
                                planes.push_back({normal, d});
                                inserted = true;
                            }
                        }
                    }
                }
            }
        }
    }

    if (planes.empty()) return makeAABBHull(pts);

    hull.planes = std::move(planes);
    hull.exact = true;
    return hull;
}

Hull buildAdaptiveHull(const vector<Vec3>& pts) {
    const int n = static_cast<int>(pts.size());
    if (n <= 70) return buildConvexHullBrute(pts);
    return makeAABBHull(pts);
}

double estimateBoxVolume(const AABB& box) {
    return max(0.0, box.hi.x - box.lo.x) *
           max(0.0, box.hi.y - box.lo.y) *
           max(0.0, box.hi.z - box.lo.z);
}

vector<Vec3> generateFCC(const Hull& hull, double radius, const Vec3& shift, size_t hardCap) {
    const double nn = 2.0 * radius;
    const double a = nn * std::sqrt(2.0); // FCC cubic cell edge
    const array<Vec3, 4> basis = {
        Vec3{0.0, 0.0, 0.0},
        Vec3{0.0, 0.5 * a, 0.5 * a},
        Vec3{0.5 * a, 0.0, 0.5 * a},
        Vec3{0.5 * a, 0.5 * a, 0.0}
    };

    AABB box = hull.box;
    box.lo = box.lo + Vec3{radius, radius, radius};
    box.hi = box.hi - Vec3{radius, radius, radius};
    if (box.lo.x > box.hi.x || box.lo.y > box.hi.y || box.lo.z > box.hi.z) return {};

    const int ix0 = static_cast<int>(std::floor((box.lo.x - shift.x) / a)) - 1;
    const int iy0 = static_cast<int>(std::floor((box.lo.y - shift.y) / a)) - 1;
    const int iz0 = static_cast<int>(std::floor((box.lo.z - shift.z) / a)) - 1;
    const int ix1 = static_cast<int>(std::ceil((box.hi.x - shift.x) / a)) + 1;
    const int iy1 = static_cast<int>(std::ceil((box.hi.y - shift.y) / a)) + 1;
    const int iz1 = static_cast<int>(std::ceil((box.hi.z - shift.z) / a)) + 1;

    vector<Vec3> out;
    out.reserve(min<size_t>(hardCap, 1u << 20));

    for (int ix = ix0; ix <= ix1; ++ix) {
        double bx = shift.x + ix * a;
        for (int iy = iy0; iy <= iy1; ++iy) {
            double by = shift.y + iy * a;
            for (int iz = iz0; iz <= iz1; ++iz) {
                double bz = shift.z + iz * a;
                Vec3 base{bx, by, bz};
                for (const auto& b : basis) {
                    Vec3 p = base + b;
                    if (p.x + EPS < box.lo.x || p.x - EPS > box.hi.x ||
                        p.y + EPS < box.lo.y || p.y - EPS > box.hi.y ||
                        p.z + EPS < box.lo.z || p.z - EPS > box.hi.z) {
                        continue;
                    }
                    if (pointInsidePlanes(hull.planes, p, radius)) {
                        out.push_back(p);
                        if (out.size() >= hardCap) return out;
                    }
                }
            }
        }
    }
    return out;
}

vector<Vec3> solvePacking(const vector<Vec3>& pts, double edge) {
    const double radius = edge * std::sqrt(3.0 / 8.0);
    Hull hull = buildAdaptiveHull(pts);

    const double boxVolume = estimateBoxVolume(hull.box);
    const double tetraVolume = edge * edge * edge / (6.0 * std::sqrt(2.0));
    const double roughCount = (tetraVolume > 0.0) ? (boxVolume / tetraVolume) : 0.0;

    vector<Vec3> shifts;
    const double nn = 2.0 * radius;
    const double a = nn * std::sqrt(2.0);

    if (roughCount < 200000.0) {
        shifts = {
            {0.00 * a, 0.00 * a, 0.00 * a},
            {0.25 * a, 0.25 * a, 0.25 * a},
            {0.50 * a, 0.00 * a, 0.00 * a},
            {0.00 * a, 0.50 * a, 0.00 * a},
            {0.00 * a, 0.00 * a, 0.50 * a},
            {0.50 * a, 0.50 * a, 0.00 * a},
            {0.50 * a, 0.00 * a, 0.50 * a},
            {0.00 * a, 0.50 * a, 0.50 * a}
        };
    } else {
        shifts = {
            {0.00 * a, 0.00 * a, 0.00 * a},
            {0.25 * a, 0.25 * a, 0.25 * a}
        };
    }

    size_t hardCap;
    if (roughCount < 2e6) hardCap = static_cast<size_t>(max(roughCount * 1.2, 1024.0));
    else hardCap = static_cast<size_t>(min(roughCount * 1.05, 5000000.0));

    vector<Vec3> best;
    for (const auto& shift : shifts) {
        auto cur = generateFCC(hull, radius, shift, hardCap);
        if (cur.size() > best.size()) best = std::move(cur);
    }
    return best;
}

} // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge;
    if (!(cin >> V >> edge)) {
        cout << 0 << '\n';
        return 0;
    }

    vector<Vec3> pts(V);
    for (int i = 0; i < V; ++i) {
        cin >> pts[i].x >> pts[i].y >> pts[i].z;
    }

    if (V <= 0 || edge <= 0.0) {
        cout << 0 << '\n';
        return 0;
    }

    vector<Vec3> centers = solvePacking(pts, edge);

    cout << centers.size() << '\n';
    cout << std::fixed << std::setprecision(9);
    for (const auto& c : centers) {
        cout << c.x << ' ' << c.y << ' ' << c.z
             << ' ' << 0.0 << ' ' << 0.0 << ' ' << 0.0 << '\n';
    }
    return 0;
}
