use std::cmp::Reverse;
use std::collections::{BTreeMap, BinaryHeap, HashMap};
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Item {
    value: i64,
    index: usize,
}

struct FastInput {
    data: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data)?;
        Ok(Self { data, pos: 0 })
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.data.len() {
            return None;
        }
        let mut sign = 1_i64;
        if self.data[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        }
        let mut value = 0_i64;
        let mut saw_digit = false;
        while self.pos < self.data.len() {
            let b = self.data[self.pos];
            if !b.is_ascii_digit() {
                break;
            }
            saw_digit = true;
            value = value.saturating_mul(10).saturating_add((b - b'0') as i64);
            self.pos += 1;
        }
        if saw_digit {
            Some(value.saturating_mul(sign))
        } else {
            None
        }
    }
}

fn gcd_i64(mut a: i64, mut b: i64) -> i64 {
    a = a.abs();
    b = b.abs();
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

fn emit(indices: &[usize]) {
    println!("YES");
    if indices.is_empty() {
        println!();
        return;
    }
    let mut out = String::new();
    for (i, idx) in indices.iter().enumerate() {
        if i > 0 {
            out.push(' ');
        }
        out.push_str(&idx.to_string());
    }
    println!("{out}");
}

fn solve_meet_in_middle(values: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = values.len();
    let left_n = n / 2;
    let right_n = n - left_n;
    let mut left = Vec::with_capacity(1usize << left_n);
    let mut mask = 0usize;
    while mask < (1usize << left_n) {
        let mut sum = 0_i64;
        let mut i = 0usize;
        while i < left_n {
            if ((mask >> i) & 1) != 0 {
                sum = sum.saturating_add(values[i]);
            }
            i += 1;
        }
        left.push((sum, mask));
        mask += 1;
    }
    let mut right = Vec::with_capacity(1usize << right_n);
    mask = 0usize;
    while mask < (1usize << right_n) {
        let mut sum = 0_i64;
        let mut i = 0usize;
        while i < right_n {
            if ((mask >> i) & 1) != 0 {
                sum = sum.saturating_add(values[left_n + i]);
            }
            i += 1;
        }
        right.push((sum, mask));
        mask += 1;
    }
    right.sort_unstable_by_key(|x| x.0);
    for (lsum, lmask) in left {
        let need = target.saturating_sub(lsum);
        if let Ok(pos) = right.binary_search_by_key(&need, |x| x.0) {
            let mut result = Vec::new();
            let rmask = right[pos].1;
            let mut i = 0usize;
            while i < left_n {
                if ((lmask >> i) & 1) != 0 {
                    result.push(i);
                }
                i += 1;
            }
            i = 0;
            while i < right_n {
                if ((rmask >> i) & 1) != 0 {
                    result.push(left_n + i);
                }
                i += 1;
            }
            return Some(result);
        }
    }
    None
}

fn solve_signed_dp(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let sum_abs: i64 = items.iter().map(|x| x.value.abs()).sum();
    if sum_abs > 2_000_000 {
        return None;
    }
    let offset = usize::try_from(sum_abs).ok()?;
    let width = offset.checked_mul(2)?.checked_add(1)?;
    let mut seen = vec![false; width];
    let mut parent_item = vec![usize::MAX; width];
    let mut parent_prev = vec![usize::MAX; width];
    seen[offset] = true;
    for (item_pos, item) in items.iter().enumerate() {
        let mut next = seen.clone();
        let mut sum = -(sum_abs);
        while sum <= sum_abs {
            let cur_idx = usize::try_from(sum + sum_abs).ok()?;
            if seen[cur_idx] {
                let next_sum = sum.saturating_add(item.value);
                if next_sum >= -sum_abs && next_sum <= sum_abs {
                    let next_idx = usize::try_from(next_sum + sum_abs).ok()?;
                    if !next[next_idx] {
                        next[next_idx] = true;
                        parent_item[next_idx] = item_pos;
                        parent_prev[next_idx] = cur_idx;
                    }
                }
            }
            sum += 1;
        }
        seen = next;
    }
    if target < -sum_abs || target > sum_abs {
        return None;
    }
    let mut idx = usize::try_from(target + sum_abs).ok()?;
    if !seen[idx] {
        return None;
    }
    let mut result = Vec::new();
    while idx != offset {
        let item_pos = parent_item[idx];
        if item_pos == usize::MAX {
            break;
        }
        result.push(items[item_pos].index);
        idx = parent_prev[idx];
    }
    result.reverse();
    Some(result)
}

fn solve_positive_count_dp(items: &[Item], target: usize) -> Option<Vec<usize>> {
    let mut groups: BTreeMap<usize, Vec<usize>> = BTreeMap::new();
    for item in items {
        let v = usize::try_from(item.value).ok()?;
        if v == 0 || v > target {
            continue;
        }
        groups.entry(v).or_default().push(item.index);
    }
    let mut chunks: Vec<(usize, Vec<usize>)> = Vec::new();
    for (value, idxs) in groups {
        let mut taken = 0usize;
        let mut step = 1usize;
        while taken < idxs.len() {
            let cnt = step.min(idxs.len() - taken);
            let mut chunk_indices = Vec::with_capacity(cnt);
            let end = taken + cnt;
            let mut i = taken;
            while i < end {
                chunk_indices.push(idxs[i]);
                i += 1;
            }
            chunks.push((value.saturating_mul(cnt), chunk_indices));
            taken = end;
            step = step.saturating_mul(2);
        }
    }

    let mut seen = vec![false; target + 1];
    let mut parent_item = vec![usize::MAX; target + 1];
    let mut parent_prev = vec![usize::MAX; target + 1];
    seen[0] = true;

    for (chunk_id, (weight, _)) in chunks.iter().enumerate() {
        let w = *weight;
        if w > target {
            continue;
        }
        let mut sum = target - w;
        loop {
            if seen[sum] && !seen[sum + w] {
                seen[sum + w] = true;
                parent_item[sum + w] = chunk_id;
                parent_prev[sum + w] = sum;
            }
            if sum == 0 {
                break;
            }
            sum -= 1;
        }
        if seen[target] {
            break;
        }
    }

    if !seen[target] {
        return None;
    }
    let mut result = Vec::new();
    let mut cur = target;
    while cur != 0 {
        let chunk_id = parent_item[cur];
        if chunk_id == usize::MAX {
            return None;
        }
        for idx in &chunks[chunk_id].1 {
            result.push(*idx);
        }
        cur = parent_prev[cur];
    }
    result.reverse();
    Some(result)
}

fn greedy_fill(sorted: &[Item], target: i64) -> Option<Vec<usize>> {
    let mut remain = target;
    let mut result = Vec::new();
    for item in sorted {
        if item.value <= remain {
            remain -= item.value;
            result.push(item.index);
            if remain == 0 {
                return Some(result);
            }
        }
    }
    None
}

fn solve_positive_best_first(items: &[Item], target: i64) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let mut filtered: Vec<Item> = items
        .iter()
        .copied()
        .filter(|x| x.value > 0 && x.value <= target)
        .collect();
    filtered.sort_unstable_by_key(|x| Reverse(x.value));

    if let Some(ans) = greedy_fill(&filtered, target) {
        return Some(ans);
    }

    let beam_width = 4096usize;
    let depth_limit = filtered.len().min(80);
    let mut heap: BinaryHeap<(i64, usize, Vec<usize>)> = BinaryHeap::new();
    let mut best_seen: HashMap<(usize, i64), usize> = HashMap::new();
    heap.push((target, 0, Vec::new()));
    best_seen.insert((0, target), 0);

    while let Some((remain, pos, chosen)) = heap.pop() {
        if remain == 0 {
            return Some(chosen);
        }
        if pos >= depth_limit {
            continue;
        }
        let item = filtered[pos];
        if item.value <= remain {
            let next_remain = remain - item.value;
            let mut next_chosen = chosen.clone();
            next_chosen.push(item.index);
            let key = (pos + 1, next_remain);
            if best_seen.get(&key).is_none_or(|v| next_chosen.len() < *v) {
                best_seen.insert(key, next_chosen.len());
                heap.push((next_remain, pos + 1, next_chosen));
            }
        }
        let key = (pos + 1, remain);
        if best_seen.get(&key).is_none_or(|v| chosen.len() < *v) {
            best_seen.insert(key, chosen.len());
            heap.push((remain, pos + 1, chosen));
        }
        if heap.len() > beam_width {
            let mut trimmed = BinaryHeap::new();
            let mut kept = 0usize;
            while kept < beam_width / 2 {
                if let Some(state) = heap.pop() {
                    trimmed.push(state);
                } else {
                    break;
                }
                kept += 1;
            }
            heap = trimmed;
        }
    }
    None
}

fn remap_solution(local_indices: Vec<usize>, items: &[Item]) -> Vec<usize> {
    let mut mapped = Vec::with_capacity(local_indices.len());
    for idx in local_indices {
        mapped.push(items[idx].index);
    }
    mapped
}

fn solve(items: &[Item], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }

    for item in items {
        if item.value == target {
            return Some(vec![item.index]);
        }
    }

    if items.len() <= 46 {
        let values: Vec<i64> = items.iter().map(|x| x.value).collect();
        if let Some(local) = solve_meet_in_middle(&values, target) {
            return Some(remap_solution(local, items));
        }
    }

    let all_non_negative = items.iter().all(|x| x.value >= 0);
    if !all_non_negative {
        return solve_signed_dp(items, target);
    }

    let total: i64 = items.iter().map(|x| x.value).sum();
    if target < 0 || total < target {
        return None;
    }

    let mut g = 0_i64;
    for item in items {
        g = gcd_i64(g, item.value);
    }
    if g > 0 && target % g != 0 {
        return None;
    }

    let positive: Vec<Item> = items.iter().copied().filter(|x| x.value > 0).collect();
    if positive.is_empty() {
        return None;
    }

    if let Ok(tu) = usize::try_from(target) {
        let distinctish = positive.len() <= 300_000;
        if tu <= 2_000_000 && distinctish {
            if let Some(ans) = solve_positive_count_dp(&positive, tu) {
                return Some(ans);
            }
        }
    }

    solve_positive_best_first(&positive, target)
}

fn main() -> io::Result<()> {
    let mut input = FastInput::new()?;
    let n_raw = match input.next_i64() {
        Some(v) => v,
        None => {
            println!("NO");
            return Ok(());
        }
    };
    let target = match input.next_i64() {
        Some(v) => v,
        None => {
            println!("NO");
            return Ok(());
        }
    };

    let n = match usize::try_from(n_raw) {
        Ok(v) => v,
        Err(_) => {
            println!("NO");
            return Ok(());
        }
    };

    let mut items = Vec::with_capacity(n);
    let mut i = 0usize;
    while i < n {
        if let Some(value) = input.next_i64() {
            items.push(Item { value, index: i });
        } else {
            break;
        }
        i += 1;
    }

    if items.len() != n {
        println!("NO");
        return Ok(());
    }

    match solve(&items, target) {
        Some(mut indices) => {
            indices.sort_unstable();
            emit(&indices);
        }
        None => {
            println!("NO");
        }
    }

    Ok(())
}
