#include <algorithm>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <queue>
#include <vector>

using namespace std;

class MaxCliqueSolver {
public:
    explicit MaxCliqueSolver(vector<vector<uint64_t>> adj_bits)
        : adj(move(adj_bits)), n(static_cast<int>(adj.size())),
          words((n + 63) >> 6), best_size(0) {}

    vector<int> solve() {
        vector<int> cand(n);
        iota(cand.begin(), cand.end(), 0);
        sort(cand.begin(), cand.end(), [&](int a, int b) {
            int da = degree(a), db = degree(b);
            if (da != db) return da > db;
            return a < b;
        });
        expand(cand);
        return best;
    }

private:
    vector<vector<uint64_t>> adj;
    int n;
    int words;
    int best_size;
    vector<int> best;
    vector<int> current;

    static int popcount64(uint64_t x) {
        int c = 0;
        while (x) {
            x &= x - 1;
            ++c;
        }
        return c;
    }

    int degree(int v) const {
        int sum = 0;
        for (uint64_t w : adj[v]) sum += popcount64(w);
        return sum;
    }

    bool has_edge(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    void color_sort(const vector<int>& cand, vector<int>& order, vector<int>& colors) {
        order.clear();
        colors.clear();

        vector<int> remaining = cand;
        vector<uint64_t> forbidden(words);
        int color = 0;

        while (!remaining.empty()) {
            ++color;
            fill(forbidden.begin(), forbidden.end(), 0ULL);
            vector<int> next_remaining;

            for (int v : remaining) {
                if (((forbidden[v >> 6] >> (v & 63)) & 1ULL) == 0ULL) {
                    order.push_back(v);
                    colors.push_back(color);
                    for (int i = 0; i < words; ++i) forbidden[i] |= adj[v][i];
                } else {
                    next_remaining.push_back(v);
                }
            }

            remaining.swap(next_remaining);
        }
    }

    void expand(const vector<int>& cand) {
        if (cand.empty()) {
            if (static_cast<int>(current.size()) > best_size) {
                best_size = static_cast<int>(current.size());
                best = current;
            }
            return;
        }

        vector<int> order, colors;
        color_sort(cand, order, colors);

        for (int i = static_cast<int>(order.size()) - 1; i >= 0; --i) {
            if (static_cast<int>(current.size()) + colors[i] <= best_size) return;

            int v = order[i];
            current.push_back(v);

            vector<int> next_cand;
            next_cand.reserve(i);
            for (int j = 0; j < i; ++j) {
                int u = order[j];
                if (has_edge(v, u)) next_cand.push_back(u);
            }

            if (next_cand.empty()) {
                if (static_cast<int>(current.size()) > best_size) {
                    best_size = static_cast<int>(current.size());
                    best = current;
                }
            } else {
                expand(next_cand);
            }

            current.pop_back();
        }
    }
};

struct HopcroftKarp {
    int n_left, n_right;
    vector<vector<int>> g;
    vector<int> dist, pair_u, pair_v;

    HopcroftKarp(int nl, int nr) : n_left(nl), n_right(nr), g(nl) {}

    void add_edge(int u, int v) {
        g[u].push_back(v);
    }

    bool bfs() {
        queue<int> q;
        dist.assign(n_left, -1);
        for (int u = 0; u < n_left; ++u) {
            if (pair_u[u] == -1) {
                dist[u] = 0;
                q.push(u);
            }
        }

        bool found = false;
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : g[u]) {
                int pu = pair_v[v];
                if (pu == -1) {
                    found = true;
                } else if (dist[pu] == -1) {
                    dist[pu] = dist[u] + 1;
                    q.push(pu);
                }
            }
        }
        return found;
    }

    bool dfs(int u) {
        for (int v : g[u]) {
            int pu = pair_v[v];
            if (pu == -1 || (dist[pu] == dist[u] + 1 && dfs(pu))) {
                pair_u[u] = v;
                pair_v[v] = u;
                return true;
            }
        }
        dist[u] = -1;
        return false;
    }

    int max_matching() {
        pair_u.assign(n_left, -1);
        pair_v.assign(n_right, -1);
        int matching = 0;
        while (bfs()) {
            for (int u = 0; u < n_left; ++u) {
                if (pair_u[u] == -1 && dfs(u)) ++matching;
            }
        }
        return matching;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<vector<int>> g(n);
    vector<vector<uint64_t>> adj_bits(n, vector<uint64_t>((n + 63) >> 6, 0ULL));

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        g[u].push_back(v);
        g[v].push_back(u);
        adj_bits[u][v >> 6] |= 1ULL << (v & 63);
        adj_bits[v][u >> 6] |= 1ULL << (u & 63);
    }

    vector<int> comp_id(n, -1);
    vector<vector<int>> components;
    for (int s = 0; s < n; ++s) {
        if (comp_id[s] != -1) continue;
        queue<int> q;
        q.push(s);
        comp_id[s] = static_cast<int>(components.size());
        components.push_back({});
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            components.back().push_back(u);
            for (int v : g[u]) {
                if (comp_id[v] == -1) {
                    comp_id[v] = comp_id[s];
                    q.push(v);
                }
            }
        }
    }

    vector<int> answer;
    answer.reserve(n);

    for (const auto& comp : components) {
        int sz = static_cast<int>(comp.size());
        if (sz == 1) {
            answer.push_back(comp[0]);
            continue;
        }

        vector<int> local_index(n, -1);
        for (int i = 0; i < sz; ++i) local_index[comp[i]] = i;

        vector<int> color(sz, -1);
        bool bipartite = true;
        for (int i = 0; i < sz && bipartite; ++i) {
            if (color[i] != -1) continue;
            queue<int> q;
            color[i] = 0;
            q.push(i);
            while (!q.empty() && bipartite) {
                int li = q.front();
                q.pop();
                int u = comp[li];
                for (int v : g[u]) {
                    if (comp_id[v] != comp_id[u]) continue;
                    int lv = local_index[v];
                    if (color[lv] == -1) {
                        color[lv] = color[li] ^ 1;
                        q.push(lv);
                    } else if (color[lv] == color[li]) {
                        bipartite = false;
                        break;
                    }
                }
            }
        }

        if (bipartite) {
            vector<int> left_nodes, right_nodes, left_id(sz, -1), right_id(sz, -1);
            for (int i = 0; i < sz; ++i) {
                if (color[i] == 0) {
                    left_id[i] = static_cast<int>(left_nodes.size());
                    left_nodes.push_back(i);
                } else {
                    right_id[i] = static_cast<int>(right_nodes.size());
                    right_nodes.push_back(i);
                }
            }

            HopcroftKarp hk(static_cast<int>(left_nodes.size()), static_cast<int>(right_nodes.size()));
            for (int li : left_nodes) {
                int u = comp[li];
                for (int v : g[u]) {
                    if (comp_id[v] != comp_id[u]) continue;
                    int lv = local_index[v];
                    if (color[lv] == 1) hk.add_edge(left_id[li], right_id[lv]);
                }
            }
            hk.max_matching();

            vector<char> vis_left(left_nodes.size(), false), vis_right(right_nodes.size(), false);
            queue<int> q;
            for (int u = 0; u < static_cast<int>(left_nodes.size()); ++u) {
                if (hk.pair_u[u] == -1) {
                    vis_left[u] = true;
                    q.push(u);
                }
            }
            while (!q.empty()) {
                int u = q.front();
                q.pop();
                for (int v : hk.g[u]) {
                    if (hk.pair_u[u] != v && !vis_right[v]) {
                        vis_right[v] = true;
                        int mu = hk.pair_v[v];
                        if (mu != -1 && !vis_left[mu]) {
                            vis_left[mu] = true;
                            q.push(mu);
                        }
                    }
                }
            }

            for (int i = 0; i < static_cast<int>(left_nodes.size()); ++i) {
                if (vis_left[i]) answer.push_back(comp[left_nodes[i]]);
            }
            for (int i = 0; i < static_cast<int>(right_nodes.size()); ++i) {
                if (!vis_right[i]) answer.push_back(comp[right_nodes[i]]);
            }
            continue;
        }

        int words = (sz + 63) >> 6;
        vector<vector<uint64_t>> comp_complement(sz, vector<uint64_t>(words, 0ULL));
        for (int i = 0; i < sz; ++i) {
            int u = comp[i];
            for (int j = 0; j < sz; ++j) {
                if (i == j) continue;
                int v = comp[j];
                bool edge = (adj_bits[u][v >> 6] >> (v & 63)) & 1ULL;
                if (!edge) comp_complement[i][j >> 6] |= 1ULL << (j & 63);
            }
        }

        MaxCliqueSolver solver(move(comp_complement));
        vector<int> clique = solver.solve();
        for (int v : clique) answer.push_back(comp[v]);
    }

    sort(answer.begin(), answer.end());
    cout << answer.size() << '\n';
    for (size_t i = 0; i < answer.size(); ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';
    return 0;
}
