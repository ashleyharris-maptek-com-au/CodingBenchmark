#include <algorithm>
#include <bit>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <vector>

using namespace std;

class MinVertexCoverSolver {
public:
    explicit MinVertexCoverSolver(vector<vector<int>> graph)
        : g_(std::move(graph)),
          n_(static_cast<int>(g_.size())),
          words_((n_ + 63) >> 6),
          adj_(n_, vector<uint64_t>(words_, 0)),
          best_size_(numeric_limits<int>::max()),
          best_cover_(n_, 0) {
        for (int v = 0; v < n_; ++v) {
            for (int u : g_[v]) {
                adj_[v][u >> 6] |= 1ULL << (u & 63);
            }
        }
    }

    vector<int> solve() {
        vector<uint64_t> active(words_, 0);
        for (int v = 0; v < n_; ++v) {
            active[v >> 6] |= 1ULL << (v & 63);
        }
        vector<char> chosen(n_, 0);

        vector<char> heuristic = greedy_cover(active, chosen);
        best_size_ = count_chosen(heuristic);
        best_cover_ = heuristic;

        dfs(active, chosen, 0);

        vector<int> cover;
        for (int v = 0; v < n_; ++v) {
            if (best_cover_[v]) cover.push_back(v);
        }
        return cover;
    }

private:
    const vector<vector<int>> g_;
    const int n_;
    const int words_;
    vector<vector<uint64_t>> adj_;
    int best_size_;
    vector<char> best_cover_;

    static int popcount64(uint64_t x) {
        return static_cast<int>(std::popcount(x));
    }

    static int first_set_bit(uint64_t x) {
        return static_cast<int>(std::countr_zero(x));
    }

    int count_chosen(const vector<char>& chosen) const {
        int total = 0;
        for (char x : chosen) total += x != 0;
        return total;
    }

    bool is_active(const vector<uint64_t>& active, int v) const {
        return (active[v >> 6] >> (v & 63)) & 1ULL;
    }

    void clear_active(vector<uint64_t>& active, int v) const {
        active[v >> 6] &= ~(1ULL << (v & 63));
    }

    int degree_in_active(const vector<uint64_t>& active, int v) const {
        int degree = 0;
        for (int i = 0; i < words_; ++i) {
            degree += popcount64(active[i] & adj_[v][i]);
        }
        return degree;
    }

    int first_neighbor_in_active(const vector<uint64_t>& active, int v) const {
        for (int i = 0; i < words_; ++i) {
            uint64_t w = active[i] & adj_[v][i];
            if (w) {
                return (i << 6) + first_set_bit(w);
            }
        }
        return -1;
    }

    vector<int> neighbors_in_active(const vector<uint64_t>& active, int v) const {
        vector<int> out;
        for (int i = 0; i < words_; ++i) {
            uint64_t w = active[i] & adj_[v][i];
            while (w) {
                int b = first_set_bit(w);
                out.push_back((i << 6) + b);
                w &= w - 1;
            }
        }
        return out;
    }

    bool has_edges(const vector<uint64_t>& active) const {
        for (int v = 0; v < n_; ++v) {
            if (is_active(active, v) && first_neighbor_in_active(active, v) != -1) {
                return true;
            }
        }
        return false;
    }

    void reduce(vector<uint64_t>& active, vector<char>& chosen, int& chosen_count) const {
        bool changed = true;
        while (changed) {
            changed = false;
            for (int v = 0; v < n_; ++v) {
                if (!is_active(active, v)) continue;
                int deg = degree_in_active(active, v);
                if (deg == 0) {
                    clear_active(active, v);
                    changed = true;
                } else if (deg == 1) {
                    int u = first_neighbor_in_active(active, v);
                    if (u != -1 && !chosen[u]) {
                        chosen[u] = 1;
                        ++chosen_count;
                    }
                    clear_active(active, u);
                    changed = true;
                }
            }
        }
    }

    int greedy_matching_lower_bound(const vector<uint64_t>& active) const {
        vector<char> used(n_, 0);
        int matched = 0;
        for (int v = 0; v < n_; ++v) {
            if (!is_active(active, v) || used[v]) continue;
            for (int u : g_[v]) {
                if (is_active(active, u) && !used[u]) {
                    used[v] = used[u] = 1;
                    ++matched;
                    break;
                }
            }
        }
        return matched;
    }

    vector<char> greedy_cover(vector<uint64_t> active, vector<char> chosen) const {
        int chosen_count = count_chosen(chosen);
        while (has_edges(active)) {
            int best_v = -1;
            int best_deg = -1;
            for (int v = 0; v < n_; ++v) {
                if (!is_active(active, v)) continue;
                int deg = degree_in_active(active, v);
                if (deg > best_deg) {
                    best_deg = deg;
                    best_v = v;
                }
            }
            if (best_v == -1 || best_deg <= 0) break;
            if (!chosen[best_v]) {
                chosen[best_v] = 1;
                ++chosen_count;
            }
            clear_active(active, best_v);
            reduce(active, chosen, chosen_count);
        }
        return chosen;
    }

    vector<vector<int>> split_components(const vector<uint64_t>& active) const {
        vector<char> seen(n_, 0);
        vector<vector<int>> components;
        queue<int> q;

        for (int s = 0; s < n_; ++s) {
            if (!is_active(active, s) || seen[s]) continue;
            seen[s] = 1;
            q.push(s);
            components.push_back({});
            auto& comp = components.back();

            while (!q.empty()) {
                int v = q.front();
                q.pop();
                comp.push_back(v);
                for (int u : g_[v]) {
                    if (is_active(active, u) && !seen[u]) {
                        seen[u] = 1;
                        q.push(u);
                    }
                }
            }
        }
        return components;
    }

    vector<vector<int>> extract_subgraph(const vector<int>& comp) const {
        int sz = static_cast<int>(comp.size());
        vector<int> where(n_, -1);
        for (int i = 0; i < sz; ++i) where[comp[i]] = i;
        vector<vector<int>> sub(sz);
        for (int i = 0; i < sz; ++i) {
            int v = comp[i];
            for (int u : g_[v]) {
                int j = where[u];
                if (j != -1) sub[i].push_back(j);
            }
        }
        return sub;
    }

    void dfs(vector<uint64_t> active, vector<char> chosen, int chosen_count) {
        reduce(active, chosen, chosen_count);

        if (chosen_count >= best_size_) return;
        if (!has_edges(active)) {
            if (chosen_count < best_size_) {
                best_size_ = chosen_count;
                best_cover_ = chosen;
            }
            return;
        }

        auto comps = split_components(active);
        if (comps.size() > 1) {
            vector<char> aggregate = chosen;
            int aggregate_count = chosen_count;
            for (const auto& comp : comps) {
                MinVertexCoverSolver subsolver(extract_subgraph(comp));
                vector<int> subcover = subsolver.solve();
                vector<int> map = comp;
                for (int idx : subcover) {
                    int gv = map[idx];
                    if (!aggregate[gv]) {
                        aggregate[gv] = 1;
                        ++aggregate_count;
                    }
                }
                if (aggregate_count >= best_size_) return;
            }
            if (aggregate_count < best_size_) {
                best_size_ = aggregate_count;
                best_cover_ = aggregate;
            }
            return;
        }

        int lb = greedy_matching_lower_bound(active);
        if (chosen_count + lb >= best_size_) return;

        vector<char> heuristic = greedy_cover(active, chosen);
        int heuristic_size = count_chosen(heuristic);
        if (heuristic_size < best_size_) {
            best_size_ = heuristic_size;
            best_cover_ = heuristic;
        }

        int branch_v = -1;
        int branch_deg = -1;
        for (int v = 0; v < n_; ++v) {
            if (!is_active(active, v)) continue;
            int deg = degree_in_active(active, v);
            if (deg > branch_deg) {
                branch_deg = deg;
                branch_v = v;
            }
        }
        if (branch_v == -1) return;

        {
            vector<uint64_t> next_active = active;
            vector<char> next_chosen = chosen;
            int next_count = chosen_count;
            if (!next_chosen[branch_v]) {
                next_chosen[branch_v] = 1;
                ++next_count;
            }
            clear_active(next_active, branch_v);
            dfs(std::move(next_active), std::move(next_chosen), next_count);
        }

        {
            vector<uint64_t> next_active = active;
            vector<char> next_chosen = chosen;
            int next_count = chosen_count;
            vector<int> nbrs = neighbors_in_active(active, branch_v);
            for (int u : nbrs) {
                if (!next_chosen[u]) {
                    next_chosen[u] = 1;
                    ++next_count;
                }
                clear_active(next_active, u);
            }
            clear_active(next_active, branch_v);
            dfs(std::move(next_active), std::move(next_chosen), next_count);
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    vector<vector<int>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    for (int v = 0; v < n; ++v) {
        auto& adj = g[v];
        sort(adj.begin(), adj.end());
        adj.erase(unique(adj.begin(), adj.end()), adj.end());
    }

    MinVertexCoverSolver solver(g);
    vector<int> cover = solver.solve();
    vector<char> in_cover(n, 0);
    for (int v : cover) in_cover[v] = 1;

    vector<int> independent;
    independent.reserve(n - static_cast<int>(cover.size()));
    for (int v = 0; v < n; ++v) {
        if (!in_cover[v]) independent.push_back(v);
    }

    cout << independent.size() << '\n';
    for (int i = 0; i < static_cast<int>(independent.size()); ++i) {
        if (i) cout << ' ';
        cout << independent[i];
    }
    cout << '\n';
    return 0;
}
