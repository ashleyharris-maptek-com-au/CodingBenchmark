const fs = require('fs'); const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

function solve(data) {
  const depot = data.depot || [0, 0];
  const depotX = Number(depot[0]) || 0;
  const depotY = Number(depot[1]) || 0;
  const packages = Array.isArray(data.packages) ? data.packages : [];
  const n = packages.length;
  const truckLimitRaw = Number(data.truckCount);

  if (n === 0) {
    return { routes: [] };
  }

  const truckCount = Math.max(1, Math.min(n, Number.isFinite(truckLimitRaw) ? Math.floor(truckLimitRaw) : n));

  const pts = new Array(n);
  for (let i = 0; i < n; i++) {
    const p = packages[i];
    const dx = Number(p.x) - depotX;
    const dy = Number(p.y) - depotY;
    pts[i] = {
      index: i,
      id: p.id,
      x: Number(p.x),
      y: Number(p.y),
      dx,
      dy,
      angle: Math.atan2(dy, dx),
      radius2: dx * dx + dy * dy
    };
  }

  pts.sort((a, b) => {
    if (a.angle !== b.angle) return a.angle - b.angle;
    if (a.radius2 !== b.radius2) return a.radius2 - b.radius2;
    return a.index - b.index;
  });

  let bestRoutes = buildRoutesFromSweep(pts, truckCount, depotX, depotY);
  let bestCost = totalCost(bestRoutes, depotX, depotY);

  const candidateStarts = buildCandidateStarts(n, truckCount);
  for (let i = 0; i < candidateStarts.length; i++) {
    const shifted = rotateArray(pts, candidateStarts[i]);
    const routes = buildRoutesFromSweep(shifted, truckCount, depotX, depotY);
    const cost = totalCost(routes, depotX, depotY);
    if (cost < bestCost) {
      bestCost = cost;
      bestRoutes = routes;
    }
  }

  rebalanceAdjacent(bestRoutes, depotX, depotY);
  for (let i = 0; i < bestRoutes.length; i++) {
    bestRoutes[i] = improveRoute(bestRoutes[i], depotX, depotY);
  }

  return {
    routes: bestRoutes.map(route => route.map(p => p.id))
  };
}

function buildCandidateStarts(n, truckCount) {
  const set = new Set([0]);
  const attempts = Math.min(12, Math.max(1, truckCount * 2, Math.floor(Math.sqrt(n))));
  for (let i = 1; i < attempts; i++) {
    set.add(Math.floor((i * n) / attempts));
  }
  return Array.from(set).sort((a, b) => a - b);
}

function rotateArray(arr, offset) {
  if (offset <= 0 || offset >= arr.length) return arr.slice();
  return arr.slice(offset).concat(arr.slice(0, offset));
}

function buildRoutesFromSweep(sortedPts, truckCount, depotX, depotY) {
  const n = sortedPts.length;
  const routes = [];
  let start = 0;

  for (let r = 0; r < truckCount; r++) {
    const remainingPts = n - start;
    const remainingRoutes = truckCount - r;
    if (remainingPts <= 0) break;

    let size = Math.floor(remainingPts / remainingRoutes);
    if (remainingPts % remainingRoutes !== 0) size++;
    const chunk = sortedPts.slice(start, start + size);
    start += size;

    routes.push(orderCluster(chunk, depotX, depotY));
  }

  return routes.filter(route => route.length > 0);
}

function orderCluster(points, depotX, depotY) {
  const m = points.length;
  if (m <= 2) return points.slice();

  const unvisited = points.slice();
  let seedIndex = 0;
  let bestSeed = -1;
  for (let i = 0; i < unvisited.length; i++) {
    if (unvisited[i].radius2 > bestSeed) {
      bestSeed = unvisited[i].radius2;
      seedIndex = i;
    }
  }

  const route = [unvisited[seedIndex]];
  unvisited.splice(seedIndex, 1);

  let farthestIndex = 0;
  let farthestDist = -1;
  for (let i = 0; i < unvisited.length; i++) {
    const d = dist2(route[0], unvisited[i]);
    if (d > farthestDist) {
      farthestDist = d;
      farthestIndex = i;
    }
  }
  if (unvisited.length > 0) {
    route.push(unvisited[farthestIndex]);
    unvisited.splice(farthestIndex, 1);
  }

  while (unvisited.length > 0) {
    let bestPointIndex = 0;
    let bestInsertPos = 0;
    let bestDelta = Infinity;

    for (let i = 0; i < unvisited.length; i++) {
      const p = unvisited[i];
      for (let pos = 0; pos <= route.length; pos++) {
        const delta = insertionDelta(route, pos, p, depotX, depotY);
        if (delta < bestDelta - 1e-12) {
          bestDelta = delta;
          bestPointIndex = i;
          bestInsertPos = pos;
        }
      }
    }

    route.splice(bestInsertPos, 0, unvisited[bestPointIndex]);
    unvisited.splice(bestPointIndex, 1);
  }

  return improveRoute(route, depotX, depotY);
}

function improveRoute(route, depotX, depotY) {
  const m = route.length;
  if (m <= 2) return route.slice();

  const arr = route.slice();
  const maxPasses = m <= 80 ? 6 : m <= 300 ? 4 : 2;

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;

    for (let i = 0; i < m - 1; i++) {
      for (let k = i + 1; k < m; k++) {
        const gain = twoOptGain(arr, i, k, depotX, depotY);
        if (gain > 1e-9) {
          reverseSegment(arr, i, k);
          improved = true;
        }
      }
    }

    if (m <= 120) {
      for (let i = 0; i < m; i++) {
        let bestPos = i;
        let bestGain = 0;
        for (let pos = 0; pos < m; pos++) {
          if (pos === i || pos === i + 1) continue;
          const gain = relocateGain(arr, i, pos, depotX, depotY);
          if (gain > bestGain + 1e-9) {
            bestGain = gain;
            bestPos = pos;
          }
        }
        if (bestPos !== i) {
          relocate(arr, i, bestPos);
          improved = true;
        }
      }
    }

    if (!improved) break;
  }

  return arr;
}

function rebalanceAdjacent(routes, depotX, depotY) {
  if (routes.length <= 1) return;
  const rounds = routes.length <= 8 ? 3 : 1;

  for (let round = 0; round < rounds; round++) {
    let changed = false;
    for (let i = 0; i < routes.length - 1; i++) {
      if (tryBoundaryMove(routes, i, i + 1, depotX, depotY)) changed = true;
      if (tryBoundaryMove(routes, i + 1, i, depotX, depotY)) changed = true;
    }
    if (!changed) break;
  }
}

function tryBoundaryMove(routes, fromIdx, toIdx, depotX, depotY) {
  const from = routes[fromIdx];
  const to = routes[toIdx];
  if (from.length <= 1) return false;

  const candidates = [0, from.length - 1];
  let best = null;

  for (let ci = 0; ci < candidates.length; ci++) {
    const removeIdx = candidates[ci];
    const point = from[removeIdx];
    const removalGain = removalDelta(from, removeIdx, depotX, depotY);

    const positions = [0, to.length];
    for (let pi = 0; pi < positions.length; pi++) {
      const insertPos = positions[pi];
      const addCost = insertionDelta(to, insertPos, point, depotX, depotY);
      const netGain = removalGain - addCost;
      if (netGain > 1e-9 && (!best || netGain > best.netGain)) {
        best = { removeIdx, insertPos, netGain };
      }
    }
  }

  if (!best) return false;

  const moved = from.splice(best.removeIdx, 1)[0];
  to.splice(best.insertPos, 0, moved);
  routes[fromIdx] = improveRoute(from, depotX, depotY);
  routes[toIdx] = improveRoute(to, depotX, depotY);
  return true;
}

function totalCost(routes, depotX, depotY) {
  let sum = 0;
  for (let i = 0; i < routes.length; i++) {
    sum += routeCost(routes[i], depotX, depotY);
  }
  return sum;
}

function routeCost(route, depotX, depotY) {
  if (route.length === 0) return 0;
  let total = depotDist(route[0], depotX, depotY);
  for (let i = 1; i < route.length; i++) {
    total += dist(route[i - 1], route[i]);
  }
  total += depotDist(route[route.length - 1], depotX, depotY);
  return total;
}

function insertionDelta(route, pos, point, depotX, depotY) {
  const prev = pos === 0 ? null : route[pos - 1];
  const next = pos === route.length ? null : route[pos];
  const removed = edgeCost(prev, next, depotX, depotY);
  const added = edgeCost(prev, point, depotX, depotY) + edgeCost(point, next, depotX, depotY);
  return added - removed;
}

function removalDelta(route, idx, depotX, depotY) {
  const prev = idx === 0 ? null : route[idx - 1];
  const curr = route[idx];
  const next = idx === route.length - 1 ? null : route[idx + 1];
  const removed = edgeCost(prev, curr, depotX, depotY) + edgeCost(curr, next, depotX, depotY);
  const added = edgeCost(prev, next, depotX, depotY);
  return removed - added;
}

function twoOptGain(route, i, k, depotX, depotY) {
  const a = i === 0 ? null : route[i - 1];
  const b = route[i];
  const c = route[k];
  const d = k === route.length - 1 ? null : route[k + 1];
  const current = edgeCost(a, b, depotX, depotY) + edgeCost(c, d, depotX, depotY);
  const swapped = edgeCost(a, c, depotX, depotY) + edgeCost(b, d, depotX, depotY);
  return current - swapped;
}

function relocateGain(route, fromIdx, toPos, depotX, depotY) {
  const point = route[fromIdx];
  const afterRemoval = route.slice(0, fromIdx).concat(route.slice(fromIdx + 1));
  const removalGain = removalDelta(route, fromIdx, depotX, depotY);
  let adjustedPos = toPos;
  if (toPos > fromIdx) adjustedPos--;
  const addCost = insertionDelta(afterRemoval, adjustedPos, point, depotX, depotY);
  return removalGain - addCost;
}

function relocate(route, fromIdx, toPos) {
  const point = route.splice(fromIdx, 1)[0];
  let adjusted = toPos;
  if (toPos > fromIdx) adjusted--;
  route.splice(adjusted, 0, point);
}

function reverseSegment(arr, start, end) {
  while (start < end) {
    const tmp = arr[start];
    arr[start] = arr[end];
    arr[end] = tmp;
    start++;
    end--;
  }
}

function edgeCost(a, b, depotX, depotY) {
  if (a === null && b === null) return 0;
  if (a === null) return depotDist(b, depotX, depotY);
  if (b === null) return depotDist(a, depotX, depotY);
  return dist(a, b);
}

function depotDist(p, depotX, depotY) {
  const dx = p.x - depotX;
  const dy = p.y - depotY;
  return Math.hypot(dx, dy);
}

function dist(a, b) {
  return Math.hypot(a.x - b.x, a.y - b.y);
}

function dist2(a, b) {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return dx * dx + dy * dy;
}

const result = solve(data);
process.stdout.write(JSON.stringify(result));
