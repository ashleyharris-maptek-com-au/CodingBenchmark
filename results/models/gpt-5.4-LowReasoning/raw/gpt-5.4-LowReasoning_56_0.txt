const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const depot = data.depot || [0, 0];
const depotX = depot[0];
const depotY = depot[1];
const packages = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const requestedTruckCount = Number.isInteger(data.truckCount) ? data.truckCount : 1;
const truckCount = Math.max(1, Math.min(requestedTruckCount, n));

function dist(a, b) {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
}

function distDepot(p) {
  const dx = p.x - depotX;
  const dy = p.y - depotY;
  return Math.hypot(dx, dy);
}

function routeDistance(route) {
  if (route.length === 0) return 0;
  let total = distDepot(route[0]) + distDepot(route[route.length - 1]);
  for (let i = 1; i < route.length; i++) total += dist(route[i - 1], route[i]);
  return total;
}

function totalDistance(routes) {
  let total = 0;
  for (const route of routes) total += routeDistance(route);
  return total;
}

const enriched = packages.map((p, idx) => {
  const dx = p.x - depotX;
  const dy = p.y - depotY;
  let angle = Math.atan2(dy, dx);
  if (angle < 0) angle += Math.PI * 2;
  return {
    id: p.id,
    x: p.x,
    y: p.y,
    angle,
    radius: Math.hypot(dx, dy),
    idx
  };
});

enriched.sort((a, b) => {
  if (a.angle !== b.angle) return a.angle - b.angle;
  if (a.radius !== b.radius) return a.radius - b.radius;
  return a.idx - b.idx;
});

function buildRoute(points) {
  const m = points.length;
  if (m <= 1) return points.slice();

  const remaining = points.slice();
  let bestStartIndex = 0;
  let bestStartRadius = Infinity;
  for (let i = 0; i < remaining.length; i++) {
    if (remaining[i].radius < bestStartRadius) {
      bestStartRadius = remaining[i].radius;
      bestStartIndex = i;
    }
  }

  const route = [remaining.splice(bestStartIndex, 1)[0]];

  while (remaining.length > 0) {
    const last = route[route.length - 1];
    let bestIndex = 0;
    let bestD = Infinity;
    for (let i = 0; i < remaining.length; i++) {
      const d = dist(last, remaining[i]);
      if (d < bestD) {
        bestD = d;
        bestIndex = i;
      }
    }
    route.push(remaining.splice(bestIndex, 1)[0]);
  }

  improveRoute2Opt(route);
  return route;
}

function improveRoute2Opt(route) {
  const m = route.length;
  if (m < 4) return;

  // Keep large instances scalable.
  const maxNodes = 220;
  if (m > maxNodes) return;

  let improved = true;
  let passes = 0;
  while (improved && passes < 3) {
    improved = false;
    passes++;
    for (let i = 0; i < m - 2; i++) {
      const a = i === 0 ? null : route[i - 1];
      const b = route[i];
      for (let j = i + 1; j < m - 1; j++) {
        const c = route[j];
        const d = route[j + 1];

        const before = (a ? dist(a, b) : distDepot(b)) + dist(c, d);
        const after = (a ? dist(a, c) : distDepot(c)) + dist(b, d);

        if (after + 1e-9 < before) {
          reverseSegment(route, i, j);
          improved = true;
        }
      }
    }

    for (let i = 0; i < m - 1; i++) {
      const a = i === 0 ? null : route[i - 1];
      const b = route[i];
      const c = route[i + 1];
      const before = (a ? dist(a, b) : distDepot(b)) + dist(b, c) + distDepot(route[m - 1]);
      const after = (a ? dist(a, c) : distDepot(c)) + dist(c, b) + distDepot(route[m - 1]);
      if (after + 1e-9 < before) {
        reverseSegment(route, i, i + 1);
        improved = true;
      }
    }
  }
}

function reverseSegment(arr, left, right) {
  while (left < right) {
    const tmp = arr[left];
    arr[left] = arr[right];
    arr[right] = tmp;
    left++;
    right--;
  }
}

function splitIntoGroups(rotated, k) {
  const groups = Array.from({ length: k }, () => []);
  const target = rotated.length / k;
  let cursor = 0;

  for (let g = 0; g < k; g++) {
    const remainingGroups = k - g;
    const remainingItems = rotated.length - cursor;
    let size = Math.round(target);
    const minSize = Math.floor(remainingItems / remainingGroups);
    const maxSize = Math.ceil(remainingItems / remainingGroups);
    if (size < minSize) size = minSize;
    if (size > maxSize) size = maxSize;
    groups[g] = rotated.slice(cursor, cursor + size);
    cursor += size;
  }

  if (cursor < rotated.length) {
    groups[groups.length - 1].push(...rotated.slice(cursor));
  }

  return groups.filter(group => group.length > 0);
}

function solveForRotation(offset) {
  const rotated = enriched.slice(offset).concat(enriched.slice(0, offset));
  const groups = splitIntoGroups(rotated, truckCount);
  const routes = groups.map(buildRoute);
  return {
    routes,
    score: totalDistance(routes)
  };
}

let best = null;
const rotationCount = Math.min(n, Math.max(1, truckCount * 4, 8));
for (let r = 0; r < rotationCount; r++) {
  const offset = Math.floor((r * n) / rotationCount);
  const candidate = solveForRotation(offset);
  if (!best || candidate.score < best.score) best = candidate;
}

const output = {
  routes: best.routes.map(route => route.map(p => p.id))
};

process.stdout.write(JSON.stringify(output));
