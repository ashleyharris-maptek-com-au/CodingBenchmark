using namespace std;

struct U64Pair {
    uint64_t a = 0;
    uint64_t b = 0;

    bool operator==(const U64Pair& other) const {
        return a == other.a && b == other.b;
    }
};

struct U64PairHash {
    size_t operator()(const U64Pair& x) const {
        uint64_t h = x.a ^ (x.b + 0x9e3779b97f4a7c15ULL + (x.a << 6) + (x.a >> 2));
        return static_cast<size_t>(h);
    }
};

struct RollingHash {
    static constexpr uint64_t BASE1 = 1469598103934665603ULL;
    static constexpr uint64_t BASE2 = 1099511628211ULL;
    static constexpr uint64_t MUL1 = 11400714819323198485ULL;
    static constexpr uint64_t MUL2 = 14029467366897019727ULL;

    vector<uint64_t> pref1, pref2, pow1, pow2;

    explicit RollingHash(const string& s) {
        size_t n = s.size();
        pref1.assign(n + 1, 0);
        pref2.assign(n + 1, 0);
        pow1.assign(n + 1, 1);
        pow2.assign(n + 1, 1);
        for (size_t i = 0; i < n; ++i) {
            uint64_t c = static_cast<unsigned char>(s[i]) + 1ULL;
            pref1[i + 1] = pref1[i] * MUL1 + (c ^ BASE1);
            pref2[i + 1] = pref2[i] * MUL2 + (c ^ BASE2);
            pow1[i + 1] = pow1[i] * MUL1;
            pow2[i + 1] = pow2[i] * MUL2;
        }
    }

    U64Pair get(size_t l, size_t r) const {
        return {
            pref1[r] - pref1[l] * pow1[r - l],
            pref2[r] - pref2[l] * pow2[r - l]
        };
    }
};

struct SuffixAutomaton {
    struct State {
        int link = -1;
        int len = 0;
        int first_pos = -1;
        unordered_map<unsigned char, int> next;
    };

    vector<State> st;
    int last = 0;

    explicit SuffixAutomaton(size_t reserve_len = 0) {
        st.reserve(reserve_len ? reserve_len * 2 : 2);
        st.push_back(State{});
    }

    void extend(unsigned char c, int pos) {
        int cur = static_cast<int>(st.size());
        st.push_back(State{});
        st[cur].len = st[last].len + 1;
        st[cur].first_pos = pos;

        int p = last;
        while (p != -1 && !st[p].next.count(c)) {
            st[p].next[c] = cur;
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = static_cast<int>(st.size());
                st.push_back(st[q]);
                st[clone].len = st[p].len + 1;
                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = clone;
                st[cur].link = clone;
            }
        }
        last = cur;
    }
};

static string solve_tiny(const vector<string>& strings, size_t shortest_idx) {
    const string& base = strings[shortest_idx];
    for (size_t len = base.size(); len > 0; --len) {
        unordered_set<string> seen;
        for (size_t i = 0; i + len <= base.size(); ++i) {
            string sub = base.substr(i, len);
            if (!seen.insert(sub).second) {
                continue;
            }
            bool ok = true;
            for (size_t j = 0; j < strings.size(); ++j) {
                if (j == shortest_idx) {
                    continue;
                }
                if (strings[j].find(sub) == string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                return sub;
            }
        }
    }
    return "";
}

static string solve_sam(const vector<string>& strings, size_t shortest_idx) {
    const string& base = strings[shortest_idx];
    if (base.empty()) {
        return "";
    }

    SuffixAutomaton sam(base.size());
    for (int i = 0; i < static_cast<int>(base.size()); ++i) {
        sam.extend(static_cast<unsigned char>(base[i]), i);
    }

    const int states = static_cast<int>(sam.st.size());
    vector<int> best(states);
    for (int i = 0; i < states; ++i) {
        best[i] = sam.st[i].len;
    }

    vector<int> order(states);
    for (int i = 0; i < states; ++i) {
        order[i] = i;
    }
    sort(order.begin(), order.end(), [&](int lhs, int rhs) {
        return sam.st[lhs].len > sam.st[rhs].len;
    });

    vector<int> cur_best(states);
    for (size_t idx = 0; idx < strings.size(); ++idx) {
        if (idx == shortest_idx) {
            continue;
        }

        fill(cur_best.begin(), cur_best.end(), 0);
        int v = 0;
        int l = 0;

        for (unsigned char c : strings[idx]) {
            while (v != 0 && !sam.st[v].next.count(c)) {
                v = sam.st[v].link;
                l = sam.st[v].len;
            }

            auto it = sam.st[v].next.find(c);
            if (it != sam.st[v].next.end()) {
                v = it->second;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            cur_best[v] = max(cur_best[v], l);
        }

        for (int state : order) {
            int parent = sam.st[state].link;
            if (parent >= 0) {
                cur_best[parent] = max(cur_best[parent], min(cur_best[state], sam.st[parent].len));
            }
        }

        for (int i = 0; i < states; ++i) {
            best[i] = min(best[i], cur_best[i]);
        }
    }

    int best_len = 0;
    int best_state = 0;
    for (int i = 0; i < states; ++i) {
        if (best[i] > best_len) {
            best_len = best[i];
            best_state = i;
        }
    }

    if (best_len == 0) {
        return "";
    }
    int end_pos = sam.st[best_state].first_pos;
    return base.substr(end_pos - best_len + 1, best_len);
}

static bool exists_common_of_length(
    const vector<string>& strings,
    size_t shortest_idx,
    const vector<RollingHash>& hashes,
    size_t len,
    string& answer
) {
    if (len == 0) {
        answer.clear();
        return true;
    }

    const string& base = strings[shortest_idx];
    unordered_map<U64Pair, int, U64PairHash> candidates;
    candidates.reserve(base.size() + 1);

    for (size_t i = 0; i + len <= base.size(); ++i) {
        candidates.emplace(hashes[shortest_idx].get(i, i + len), static_cast<int>(i));
    }

    for (size_t idx = 0; idx < strings.size(); ++idx) {
        if (idx == shortest_idx) {
            continue;
        }
        const string& s = strings[idx];
        if (s.size() < len) {
            return false;
        }

        unordered_map<U64Pair, int, U64PairHash> next_candidates;
        next_candidates.reserve(candidates.size());

        for (size_t i = 0; i + len <= s.size(); ++i) {
            U64Pair h = hashes[idx].get(i, i + len);
            auto it = candidates.find(h);
            if (it != candidates.end()) {
                next_candidates.emplace(h, it->second);
            }
        }

        candidates.swap(next_candidates);
        if (candidates.empty()) {
            return false;
        }
    }

    const auto& chosen = *candidates.begin();
    answer = base.substr(static_cast<size_t>(chosen.second), len);
    return true;
}

static string solve_hash(const vector<string>& strings, size_t shortest_idx) {
    vector<RollingHash> hashes;
    hashes.reserve(strings.size());
    for (const string& s : strings) {
        hashes.emplace_back(s);
    }

    size_t lo = 0;
    size_t hi = strings[shortest_idx].size();
    string best_answer;

    while (lo <= hi) {
        size_t mid = lo + (hi - lo) / 2;
        string candidate;
        if (exists_common_of_length(strings, shortest_idx, hashes, mid, candidate)) {
            best_answer.swap(candidate);
            lo = mid + 1;
        } else {
            if (mid == 0) {
                break;
            }
            hi = mid - 1;
        }
    }

    return best_answer;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    size_t n = 0;
    if (!(cin >> n)) {
        return 0;
    }
    string dummy;
    getline(cin, dummy);

    vector<string> strings(n);
    for (size_t i = 0; i < n; ++i) {
        getline(cin, strings[i]);
    }

    if (strings.empty()) {
        cout << '\n';
        return 0;
    }
    if (strings.size() == 1) {
        cout << strings[0] << '\n';
        return 0;
    }

    size_t shortest_idx = 0;
    size_t total_len = 0;
    for (size_t i = 0; i < strings.size(); ++i) {
        total_len += strings[i].size();
        if (strings[i].size() < strings[shortest_idx].size()) {
            shortest_idx = i;
        }
    }

    const size_t shortest_len = strings[shortest_idx].size();
    string result;

    if (shortest_len == 0) {
        result = "";
    } else if (strings.size() <= 5 && total_len <= 20000) {
        result = solve_tiny(strings, shortest_idx);
    } else if (shortest_len <= 2000000 && total_len <= 20000000) {
        result = solve_sam(strings, shortest_idx);
    } else {
        result = solve_hash(strings, shortest_idx);
    }

    cout << result << '\n';
    return 0;
}
