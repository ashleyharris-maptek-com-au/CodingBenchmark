from math import copysign


_MEM = {
    "mode": "approach",
    "goaround": False,
    "goaround_latched": False,
    "flare": False,
    "last_loc": 0.0,
    "last_gs": 0.0,
    "integral_loc": 0.0,
    "integral_gs": 0.0,
    "integral_speed": 0.0,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _median(values):
    if not values:
        return None
    s = sorted(values)
    n = len(s)
    m = n // 2
    if n % 2:
        return s[m]
    return 0.5 * (s[m - 1] + s[m])


def _best_value(state, prefixes, suffix, valid_suffix="_valid"):
    vals = []
    for p in prefixes:
        vk = f"{p}{valid_suffix}"
        xk = f"{p}{suffix}"
        if state.get(vk, True):
            v = state.get(xk)
            if v is not None:
                vals.append(v)
    return _median(vals)


def _choose_altitude_agl(state):
    ra = state.get("radio_alt_ft")
    if ra is not None and ra >= 0:
        return ra
    baro = _best_value(state, ["baro_alt_1", "baro_alt_2", "baro_alt_3"], "_ft")
    target_alt = state.get("target_alt_ft")
    if baro is not None and target_alt is not None:
        return max(0.0, baro - max(0.0, target_alt - 1500.0))
    return max(0.0, baro or 0.0)


def _fused_ias(state):
    ias = _best_value(state, ["ias_1", "ias_2", "ias_3"], "_kt")
    if ias is not None:
        return ias
    vals = [state.get(k) for k in ("gps_1_gs_kt", "gps_2_gs_kt", "gps_3_gs_kt", "groundspeed_kt") if state.get(k) is not None]
    return _median(vals) or 140.0


def _fused_vs(state):
    vals = []
    for i in (1, 2, 3):
        if state.get(f"irs_{i}_valid", True):
            v = state.get(f"irs_{i}_vs_fpm")
            if v is not None:
                vals.append(v)
    return _median(vals) or state.get("target_vs_fpm", -700.0) or -700.0


def _fused_roll(state):
    vals = []
    for i in (1, 2, 3):
        if state.get(f"irs_{i}_valid", True):
            v = state.get(f"irs_{i}_roll_deg")
            if v is not None:
                vals.append(v)
    return _median(vals) or 0.0


def _fused_pitch(state):
    vals = []
    for i in (1, 2, 3):
        if state.get(f"irs_{i}_valid", True):
            v = state.get(f"irs_{i}_pitch_deg")
            if v is not None:
                vals.append(v)
    return _median(vals) or 0.0


def _ils_value(state, name):
    vals = []
    for i in (1, 2, 3):
        if state.get(f"{name}_{i}_valid", True):
            v = state.get(f"{name}_{i}_dots")
            if v is not None:
                vals.append(v)
    if not vals:
        return None
    med = _median(vals)
    good = [v for v in vals if abs(v - med) < 1.2]
    return _median(good) if good else med


def _engine_ok(state, n):
    if state.get(f"engine_{n}_fire"):
        return False
    if state.get(f"engine_{n}_running") is False:
        return False
    n1 = state.get(f"engine_{n}_n1_pct")
    thrust = state.get(f"engine_{n}_thrust_pct")
    if n1 is not None and n1 < 20:
        return False
    if thrust is not None and thrust < 10:
        return False
    return True


def autopilot_step(state: dict, dt: float) -> dict:
    dt = _clamp(dt or 0.02, 0.01, 0.2)

    alt = _choose_altitude_agl(state)
    ias = _fused_ias(state)
    vs = _fused_vs(state)
    roll = _fused_roll(state)
    pitch = _fused_pitch(state)
    loc = _ils_value(state, "localizer")
    gs = _ils_value(state, "glideslope")
    dme = _median([v for v in (state.get("dme_1_nm"), state.get("dme_2_nm")) if v is not None])
    target_vref = state.get("target_vref_kt") or 137.0
    target_ias = state.get("target_ias_kt") or (target_vref + 5.0)
    on_ground = bool(state.get("on_ground") or state.get("weight_on_wheels"))
    eng1 = _engine_ok(state, 1)
    eng2 = _engine_ok(state, 2)
    engine_count = int(eng1) + int(eng2)

    if loc is None:
        loc = _MEM["last_loc"]
    else:
        _MEM["last_loc"] = loc
    if gs is None:
        gs = _MEM["last_gs"]
    else:
        _MEM["last_gs"] = gs

    if on_ground:
        _MEM["mode"] = "rollout"
        _MEM["flare"] = False
        return {
            "elevator": _clamp(-0.08 - 0.01 * pitch, -0.3, 0.2),
            "aileron": _clamp(-0.08 * roll, -0.4, 0.4),
            "rudder": _clamp(-0.15 * loc, -0.5, 0.5),
            "throttle_1": 0.0,
            "throttle_2": 0.0,
            "speedbrake": 1.0,
            "spoiler": 1.0,
            "gear": 1.0,
            "flaps": 30,
            "autobrake": 1.0 if ias > 60 else 0.6,
            "reverse_1": 1.0 if eng1 and ias > 50 else 0.0,
            "reverse_2": 1.0 if eng2 and ias > 50 else 0.0,
        }

    if alt < 50:
        _MEM["flare"] = True
        _MEM["mode"] = "flare"
    elif alt < 200:
        _MEM["mode"] = "approach"
    else:
        _MEM["mode"] = "intercept"

    stable_speed_err = ias - (target_vref + (10 if alt > 1000 else 5))
    unstable = (
        (alt < 1200 and abs(loc) > 2.2) or
        (alt < 1200 and abs(gs) > 2.0) or
        (alt < 800 and stable_speed_err > 25) or
        (alt < 500 and vs < -1400) or
        (alt < 400 and engine_count == 0)
    )

    if unstable and alt > 150 and not _MEM["goaround_latched"]:
        _MEM["goaround"] = True
        _MEM["goaround_latched"] = True

    if _MEM["goaround"]:
        if alt > 1500:
            _MEM["goaround"] = False
        return {
            "elevator": 0.55,
            "aileron": _clamp(-0.18 * loc - 0.05 * roll, -0.5, 0.5),
            "rudder": _clamp(-0.15 * loc, -0.4, 0.4),
            "throttle_1": 1.0 if eng1 else 0.0,
            "throttle_2": 1.0 if eng2 else 0.0,
            "elevator_trim": 0.2,
            "flaps": 15,
            "gear": 1.0 if alt < 400 else 0.0,
            "speedbrake": 0.0,
            "spoiler": 0.0,
            "autobrake": 0.0,
            "reverse_1": 0.0,
            "reverse_2": 0.0,
        }

    flare_bias = 0.0
    desired_vs = -700.0
    if alt > 1000:
        desired_vs = -800.0
    if alt < 300:
        desired_vs = -550.0
    if alt < 100:
        desired_vs = -350.0
    if alt < 50:
        desired_vs = -180.0
        flare_bias = 0.20
    if alt < 20:
        desired_vs = -80.0
        flare_bias = 0.32

    speed_ref = target_vref + 5.0
    if alt > 1000:
        speed_ref = max(speed_ref + 10.0, target_ias)
    elif alt > 200:
        speed_ref = max(speed_ref, min(target_ias, target_vref + 8.0))
    if engine_count == 1:
        speed_ref += 5.0
    elif engine_count == 0:
        speed_ref = max(target_vref + 15.0, min(ias, target_vref + 25.0))

    loc_p = 0.24 if alt > 200 else 0.18
    gs_p = 0.20 if alt > 200 else 0.14
    _MEM["integral_loc"] = _clamp(_MEM["integral_loc"] + loc * dt, -3.0, 3.0)
    _MEM["integral_gs"] = _clamp(_MEM["integral_gs"] + gs * dt, -3.0, 3.0)

    aileron = _clamp(-(loc_p * loc + 0.030 * _MEM["integral_loc"]) - 0.035 * roll, -1.0, 1.0)
    rudder = _clamp(-(0.10 * loc) - 0.02 * roll, -0.6, 0.6)

    vs_err = desired_vs - vs
    gs_term = -gs_p * gs
    elevator = 0.00060 * vs_err + gs_term - 0.020 * pitch + flare_bias
    elevator = _clamp(elevator, -1.0, 1.0)

    speed_err = speed_ref - ias
    _MEM["integral_speed"] = _clamp(_MEM["integral_speed"] + speed_err * dt, -20.0, 20.0)

    base_throttle = 0.52
    if alt < 50:
        base_throttle = 0.18
    if alt < 20:
        base_throttle = 0.02
    if engine_count == 0:
        base_throttle = 0.0

    throttle_cmd = base_throttle + 0.018 * speed_err - 0.00022 * min(vs, 0.0) + 0.006 * _MEM["integral_speed"]
    throttle_cmd = _clamp(throttle_cmd, 0.0, 1.0)

    if alt < 10:
        throttle_cmd = 0.0

    if eng1 and eng2:
        t1 = t2 = throttle_cmd
    elif eng1 and not eng2:
        t1, t2 = _clamp(throttle_cmd + 0.08, 0.0, 1.0), 0.0
        rudder += 0.12
        aileron += 0.06
    elif eng2 and not eng1:
        t1, t2 = 0.0, _clamp(throttle_cmd + 0.08, 0.0, 1.0)
        rudder -= 0.12
        aileron -= 0.06
    else:
        t1 = t2 = 0.0

    if alt < 5 and vs < -300:
        elevator = max(elevator, 0.35)

    flaps = 30
    if alt > 1500:
        flaps = 15
    elif alt > 800:
        flaps = 20
    if engine_count == 0:
        flaps = 20 if alt > 300 else 30

    gear = 1.0 if alt < 2000 or (dme is not None and dme < 8.0) else 0.0
    speedbrake = 0.0

    return {
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": t1,
        "throttle_2": t2,
        "elevator_trim": _clamp(0.18 * elevator, -0.4, 0.4),
        "flaps": flaps,
        "gear": gear,
        "speedbrake": speedbrake,
        "spoiler": 0.0,
        "autobrake": 0.0,
        "reverse_1": 0.0,
        "reverse_2": 0.0,
    }
