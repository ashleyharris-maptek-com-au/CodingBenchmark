from math import isnan


_MEMORY = {
    "mode": "APPROACH",
    "flare_timer": 0.0,
    "rollout_timer": 0.0,
    "last_loc": 0.0,
    "last_gs": 0.0,
    "last_ra": 1500.0,
    "last_vs": -700.0,
    "speed_err_i": 0.0,
    "loc_err_i": 0.0,
    "gs_err_i": 0.0,
    "trim": 0.0,
    "ga_latched": False,
    "ga_reason": "",
}


def _clamp(value, lo, hi):
    return lo if value < lo else hi if value > hi else value


def _pick(values, default=0.0):
    clean = []
    for value in values:
        if value is None:
            continue
        try:
            if isnan(value):
                continue
        except TypeError:
            continue
        clean.append(float(value))
    if not clean:
        return float(default)
    clean.sort()
    mid = len(clean) // 2
    if len(clean) % 2:
        return clean[mid]
    return 0.5 * (clean[mid - 1] + clean[mid])


def _voted_sensor(state, stem, suffix, count=3, default=0.0, require_valid=True):
    values = []
    for i in range(1, count + 1):
        valid_key = f"{stem}_{i}_valid"
        value_key = f"{stem}_{i}_{suffix}"
        if require_valid and not state.get(valid_key, False):
            continue
        if value_key in state:
            values.append(state.get(value_key))
    return _pick(values, default)


def _best_radio_alt(state):
    radio_alt = state.get("radio_alt_ft")
    if isinstance(radio_alt, (int, float)) and radio_alt > -50.0:
        return float(radio_alt)
    baro_alt = _voted_sensor(state, "baro_alt", "ft", default=1500.0)
    target_alt = state.get("target_alt_ft", baro_alt)
    agl_guess = baro_alt - max(0.0, float(target_alt) - 50.0)
    return _clamp(agl_guess, 0.0, 2500.0)


def _safe_glideslope(state):
    raw = []
    for i in range(1, 4):
        if state.get(f"glideslope_{i}_valid", False):
            raw.append(float(state.get(f"glideslope_{i}_dots", 0.0)))
    if not raw:
        return _MEMORY["last_gs"]
    voted = _pick(raw, _MEMORY["last_gs"])
    filtered = []
    for value in raw:
        if abs(value - voted) <= 1.2:
            filtered.append(value)
    return _pick(filtered or raw, voted)


def _safe_localizer(state):
    raw = []
    for i in range(1, 4):
        if state.get(f"localizer_{i}_valid", False):
            raw.append(float(state.get(f"localizer_{i}_dots", 0.0)))
    if not raw:
        return _MEMORY["last_loc"]
    voted = _pick(raw, _MEMORY["last_loc"])
    filtered = []
    for value in raw:
        if abs(value - voted) <= 1.2:
            filtered.append(value)
    return _pick(filtered or raw, voted)


def _vertical_speed(state, ra, dt):
    vs_irs = _voted_sensor(state, "irs", "vs_fpm", default=_MEMORY["last_vs"], require_valid=True)
    if abs(vs_irs) > 50.0:
        _MEMORY["last_vs"] = vs_irs
        return vs_irs
    dt_used = max(dt, 0.02)
    derived = (ra - _MEMORY["last_ra"]) / dt_used * 60.0
    _MEMORY["last_vs"] = derived
    return derived


def _airspeed(state):
    ias = _voted_sensor(state, "ias", "kt", default=state.get("target_vref_kt", 140.0))
    if ias > 30.0:
        return ias
    return _pick(
        [
            state.get("groundspeed_kt"),
            state.get("gps_1_gs_kt") if state.get("gps_1_valid", False) else None,
            state.get("gps_2_gs_kt") if state.get("gps_2_valid", False) else None,
            state.get("gps_3_gs_kt") if state.get("gps_3_valid", False) else None,
            state.get("irs_1_gs_kt") if state.get("irs_1_valid", False) else None,
            state.get("irs_2_gs_kt") if state.get("irs_2_valid", False) else None,
            state.get("irs_3_gs_kt") if state.get("irs_3_valid", False) else None,
        ],
        state.get("target_vref_kt", 140.0),
    )


def _attitude(state):
    pitch = _voted_sensor(state, "irs", "pitch_deg", default=0.0)
    roll = _voted_sensor(state, "irs", "roll_deg", default=0.0)
    heading = _voted_sensor(state, "irs", "heading_deg", default=state.get("target_hdg_deg", 0.0))
    track = _voted_sensor(state, "irs", "track_deg", default=heading)
    return pitch, roll, heading, track


def _engine_alive(state, engine_index):
    running = bool(state.get(f"engine_{engine_index}_running", False))
    n1 = float(state.get(f"engine_{engine_index}_n1_pct", 0.0) or 0.0)
    thrust = float(state.get(f"engine_{engine_index}_thrust_pct", 0.0) or 0.0)
    fire = bool(state.get(f"engine_{engine_index}_fire", False))
    return running and not fire and (n1 > 20.0 or thrust > 15.0)


def _go_around_command():
    _MEMORY["mode"] = "GO_AROUND"
    _MEMORY["ga_latched"] = True
    return {
        "elevator": 0.55,
        "aileron": 0.0,
        "rudder": 0.0,
        "throttle_1": 1.0,
        "throttle_2": 1.0,
        "elevator_trim": _clamp(_MEMORY["trim"] + 0.08, -1.0, 1.0),
        "flaps": 20,
        "gear": 1.0,
        "speedbrake": 0.0,
        "spoiler": 0.0,
        "autobrake": 0.0,
        "reverse_1": 0.0,
        "reverse_2": 0.0,
    }


def autopilot_step(state: dict, dt: float) -> dict:
    dt = max(0.02, float(dt or 0.02))

    ra = _best_radio_alt(state)
    vs = _vertical_speed(state, ra, dt)
    ias = _airspeed(state)
    gs_dev = _safe_glideslope(state)
    loc_dev = _safe_localizer(state)
    pitch, roll, heading, track = _attitude(state)
    target_vref = float(state.get("target_vref_kt", 137.0) or 137.0)
    target_speed = max(115.0, target_vref + (5.0 if ra < 500.0 else 8.0))
    gs_kt = _pick([state.get("groundspeed_kt"), state.get("gps_1_gs_kt"), state.get("gps_2_gs_kt"), state.get("gps_3_gs_kt")], ias)
    eng1_alive = _engine_alive(state, 1)
    eng2_alive = _engine_alive(state, 2)
    both_failed = not eng1_alive and not eng2_alive
    wow = bool(state.get("weight_on_wheels", False) or state.get("on_ground", False))

    if wow:
        _MEMORY["mode"] = "ROLLOUT"
    elif _MEMORY["mode"] != "GO_AROUND":
        if ra < 50.0:
            _MEMORY["mode"] = "FLARE"
        elif ra < 1000.0:
            _MEMORY["mode"] = "APPROACH"
        else:
            _MEMORY["mode"] = "INTERCEPT"

    loc_rate = (loc_dev - _MEMORY["last_loc"]) / dt
    gs_rate = (gs_dev - _MEMORY["last_gs"]) / dt
    _MEMORY["last_loc"] = loc_dev
    _MEMORY["last_gs"] = gs_dev

    unstable = False
    if not both_failed and ra < 900.0:
        unstable = (
            abs(loc_dev) > 2.5
            or abs(gs_dev) > 2.0
            or (ra < 500.0 and ias > target_vref + 25.0)
            or (ra < 300.0 and vs < -1400.0)
            or (ra < 200.0 and abs(loc_dev) > 1.3)
            or (ra < 200.0 and abs(gs_dev) > 1.0)
        )
    if state.get("below_decision_height", False) and ra < 220.0:
        need_visual = abs(loc_dev) > 1.0 or abs(gs_dev) > 1.0
        if need_visual and not both_failed:
            unstable = True

    if unstable and not wow and _MEMORY["mode"] != "GO_AROUND":
        return _go_around_command()

    if _MEMORY["mode"] == "GO_AROUND":
        if ra > 1400.0:
            _MEMORY["mode"] = "INTERCEPT"
            _MEMORY["ga_latched"] = False
        else:
            return _go_around_command()

    flaps = 30
    if ra > 1800.0:
        flaps = 5
    elif ra > 1200.0:
        flaps = 15
    elif ra > 700.0:
        flaps = 20
    elif ias < target_vref + 8.0:
        flaps = 30
    if ra < 300.0 or both_failed:
        flaps = 30

    gear = 1.0 if ra < 1800.0 or gs_kt < 190.0 else 0.0

    # Localizer and heading control. Dots are primary; heading-track damps crab and turbulence.
    heading_error = ((heading - track + 540.0) % 360.0) - 180.0
    loc_p = -0.22 * loc_dev
    loc_d = -0.020 * loc_rate
    track_damp = -0.015 * heading_error
    roll_hold = -0.030 * roll
    _MEMORY["loc_err_i"] = _clamp(_MEMORY["loc_err_i"] - loc_dev * dt * 0.025, -0.18, 0.18)
    aileron = _clamp(loc_p + loc_d + track_damp + roll_hold + _MEMORY["loc_err_i"], -1.0, 1.0)
    rudder = _clamp(-0.09 * heading_error - 0.05 * roll, -1.0, 1.0)

    desired_vs = -700.0 - 260.0 * gs_dev
    desired_vs = _clamp(desired_vs, -1050.0, -250.0)
    if ra > 1200.0 and abs(gs_dev) > 1.5:
        desired_vs = _clamp(desired_vs - 100.0 * gs_dev, -1400.0, -200.0)
    if both_failed:
        desired_vs = _clamp(-550.0 - 220.0 * gs_dev, -950.0, -250.0)

    if _MEMORY["mode"] == "FLARE":
        _MEMORY["flare_timer"] += dt
        desired_vs = -120.0 - 2.5 * max(0.0, ra - 5.0)
    else:
        _MEMORY["flare_timer"] = 0.0

    speed_error = target_speed - ias
    vs_error = desired_vs - vs
    gs_pitch = -0.10 * gs_dev - 0.00065 * vs_error
    flare_bias = 0.0
    if _MEMORY["mode"] == "FLARE":
        flare_bias = 0.18 + 0.003 * min(50.0, ra)
    pitch_damp = -0.035 * pitch
    _MEMORY["gs_err_i"] = _clamp(_MEMORY["gs_err_i"] - gs_dev * dt * 0.02, -0.12, 0.12)
    elevator = _clamp(gs_pitch + flare_bias + pitch_damp + _MEMORY["gs_err_i"], -1.0, 1.0)

    throttle_base = 0.52
    if both_failed:
        throttle_base = 0.0
    elif not eng1_alive or not eng2_alive:
        throttle_base = 0.62

    _MEMORY["speed_err_i"] = _clamp(_MEMORY["speed_err_i"] + speed_error * dt * 0.004, -0.25, 0.25)
    throttle_cmd = throttle_base + 0.020 * speed_error - 0.00011 * vs + _MEMORY["speed_err_i"]

    if _MEMORY["mode"] == "FLARE":
        throttle_cmd -= 0.35 + 0.010 * _MEMORY["flare_timer"]
    throttle_cmd = _clamp(throttle_cmd, 0.0, 1.0)

    throttle_1 = throttle_cmd if eng1_alive else 0.0
    throttle_2 = throttle_cmd if eng2_alive else 0.0
    if eng1_alive and not eng2_alive:
        rudder += 0.18
        throttle_1 = _clamp(throttle_cmd + 0.10, 0.0, 1.0)
    elif eng2_alive and not eng1_alive:
        rudder -= 0.18
        throttle_2 = _clamp(throttle_cmd + 0.10, 0.0, 1.0)

    if _MEMORY["mode"] == "FLARE":
        _MEMORY["trim"] = _clamp(_MEMORY["trim"] + 0.015 * dt, -1.0, 1.0)
    else:
        _MEMORY["trim"] = _clamp(_MEMORY["trim"] + (pitch * -0.003) * dt, -1.0, 1.0)

    speedbrake = 0.0
    spoiler = 0.0
    autobrake = 0.0
    reverse_1 = 0.0
    reverse_2 = 0.0

    if wow:
        _MEMORY["rollout_timer"] += dt
        speedbrake = 1.0
        spoiler = 1.0
        autobrake = 1.0 if gs_kt > 80.0 else 0.6 if gs_kt > 40.0 else 0.2
        if eng1_alive:
            reverse_1 = 1.0 if gs_kt > 70.0 else 0.4 if gs_kt > 40.0 else 0.0
        if eng2_alive:
            reverse_2 = 1.0 if gs_kt > 70.0 else 0.4 if gs_kt > 40.0 else 0.0
        throttle_1 = 0.0
        throttle_2 = 0.0
        elevator = 0.15
        aileron = _clamp(-0.04 * roll, -0.4, 0.4)
        rudder = _clamp(-0.025 * heading_error, -0.6, 0.6)
    else:
        _MEMORY["rollout_timer"] = 0.0

    if ra < 20.0 and not wow:
        throttle_1 = min(throttle_1, 0.08 if eng1_alive else 0.0)
        throttle_2 = min(throttle_2, 0.08 if eng2_alive else 0.0)

    _MEMORY["last_ra"] = ra

    return {
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": _clamp(throttle_1, 0.0, 1.0),
        "throttle_2": _clamp(throttle_2, 0.0, 1.0),
        "elevator_trim": _clamp(_MEMORY["trim"], -1.0, 1.0),
        "flaps": flaps,
        "gear": gear,
        "speedbrake": speedbrake,
        "spoiler": spoiler,
        "autobrake": autobrake,
        "reverse_1": reverse_1,
        "reverse_2": reverse_2,
    }
