#include <algorithm>
#include <array>
#include <chrono>
#include <cstdint>
#include <cstdlib>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct Graph {
    int n = 0;
    int m = 0;
    int k = 0;
    vector<int> degree;
    vector<int> offsets;
    vector<int> adj;
    vector<pair<int, int>> edges;
};

static inline uint64_t nowMillis() {
    using namespace std::chrono;
    return duration_cast<milliseconds>(steady_clock::now().time_since_epoch()).count();
}

struct Solver {
    Graph g;
    mt19937_64 rng;
    uint64_t deadline_ms;

    explicit Solver(Graph graph, uint64_t time_limit_ms = 295000)
        : g(std::move(graph)),
          rng((uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count() ^
              0x9E3779B97F4A7C15ULL),
          deadline_ms(nowMillis() + time_limit_ms) {}

    bool timedOut() const { return nowMillis() >= deadline_ms; }

    bool verify(const vector<int>& color) const {
        if ((int)color.size() != g.n) return false;
        for (int i = 0; i < g.n; ++i) {
            if (color[i] < 0 || color[i] >= g.k) return false;
        }
        for (const auto& e : g.edges) {
            if (color[e.first] == color[e.second]) return false;
        }
        return true;
    }

    vector<int> degreeOrder() const {
        vector<int> ord(g.n);
        iota(ord.begin(), ord.end(), 0);
        sort(ord.begin(), ord.end(), [&](int a, int b) {
            if (g.degree[a] != g.degree[b]) return g.degree[a] > g.degree[b];
            return a < b;
        });
        return ord;
    }

    vector<int> degeneracyOrder() const {
        int max_deg = 0;
        for (int d : g.degree) max_deg = max(max_deg, d);
        vector<vector<int>> buckets(max_deg + 1);
        vector<int> deg = g.degree;
        vector<char> removed(g.n, 0);
        for (int v = 0; v < g.n; ++v) buckets[deg[v]].push_back(v);
        vector<int> order;
        order.reserve(g.n);
        int current = 0;
        for (int it = 0; it < g.n; ++it) {
            while (current <= max_deg) {
                while (!buckets[current].empty() && removed[buckets[current].back()]) {
                    buckets[current].pop_back();
                }
                if (!buckets[current].empty()) break;
                ++current;
            }
            int v = -1;
            if (current <= max_deg && !buckets[current].empty()) {
                v = buckets[current].back();
                buckets[current].pop_back();
            } else {
                for (int u = 0; u < g.n; ++u) {
                    if (!removed[u]) {
                        v = u;
                        break;
                    }
                }
            }
            if (v == -1) break;
            removed[v] = 1;
            order.push_back(v);
            for (int ei = g.offsets[v]; ei < g.offsets[v + 1]; ++ei) {
                int u = g.adj[ei];
                if (!removed[u]) {
                    int du = deg[u]--;
                    if (du >= 0) buckets[max(0, du - 1)].push_back(u);
                }
            }
        }
        reverse(order.begin(), order.end());
        return order;
    }

    vector<int> greedyColorWithOrder(const vector<int>& order) const {
        vector<int> color(g.n, -1);
        vector<int> seen(g.k, -1);
        for (int idx = 0; idx < (int)order.size(); ++idx) {
            int v = order[idx];
            for (int ei = g.offsets[v]; ei < g.offsets[v + 1]; ++ei) {
                int u = g.adj[ei];
                if (color[u] != -1) seen[color[u]] = v;
            }
            int pick = -1;
            for (int c = 0; c < g.k; ++c) {
                if (seen[c] != v) {
                    pick = c;
                    break;
                }
            }
            if (pick == -1) return {};
            color[v] = pick;
        }
        return color;
    }

    bool twoColorBipartite(vector<int>& out) const {
        if (g.k < 2) return false;
        out.assign(g.n, -1);
        queue<int> q;
        for (int s = 0; s < g.n; ++s) {
            if (out[s] != -1) continue;
            out[s] = 0;
            q.push(s);
            while (!q.empty()) {
                int v = q.front();
                q.pop();
                for (int ei = g.offsets[v]; ei < g.offsets[v + 1]; ++ei) {
                    int u = g.adj[ei];
                    if (out[u] == -1) {
                        out[u] = out[v] ^ 1;
                        q.push(u);
                    } else if (out[u] == out[v]) {
                        return false;
                    }
                }
            }
        }
        return true;
    }

    bool dsaturExact(vector<int>& answer) {
        if (g.n > 140) return false;
        vector<int> color(g.n, -1);
        vector<unordered_set<int>> sat(g.n);
        vector<int> uncolored_deg = g.degree;
        int colored = 0;

        function<bool()> dfs = [&]() -> bool {
            if (timedOut()) return false;
            if (colored == g.n) {
                answer = color;
                return true;
            }
            int best = -1;
            int best_sat = -1;
            int best_deg = -1;
            for (int v = 0; v < g.n; ++v) {
                if (color[v] != -1) continue;
                int sv = (int)sat[v].size();
                if (sv > best_sat || (sv == best_sat && uncolored_deg[v] > best_deg)) {
                    best = v;
                    best_sat = sv;
                    best_deg = uncolored_deg[v];
                }
            }
            vector<int> forbidden(g.k, 0);
            for (int ei = g.offsets[best]; ei < g.offsets[best + 1]; ++ei) {
                int u = g.adj[ei];
                if (color[u] != -1) forbidden[color[u]] = 1;
            }
            vector<int> choices;
            choices.reserve(g.k);
            for (int c = 0; c < g.k; ++c) {
                if (!forbidden[c]) choices.push_back(c);
            }
            sort(choices.begin(), choices.end(), [&](int a, int b) {
                int ca = 0, cb = 0;
                for (int ei = g.offsets[best]; ei < g.offsets[best + 1]; ++ei) {
                    int u = g.adj[ei];
                    if (color[u] == -1) {
                        ca += !sat[u].count(a);
                        cb += !sat[u].count(b);
                    }
                }
                return ca < cb;
            });

            vector<int> changed_nodes;
            changed_nodes.reserve(g.degree[best]);
            for (int c : choices) {
                color[best] = c;
                ++colored;
                vector<int> touched;
                for (int ei = g.offsets[best]; ei < g.offsets[best + 1]; ++ei) {
                    int u = g.adj[ei];
                    if (color[u] == -1) {
                        if (sat[u].insert(c).second) touched.push_back(u);
                        --uncolored_deg[u];
                    }
                }
                if (dfs()) return true;
                for (int ei = g.offsets[best]; ei < g.offsets[best + 1]; ++ei) {
                    int u = g.adj[ei];
                    if (color[u] == -1) ++uncolored_deg[u];
                }
                for (int u : touched) sat[u].erase(c);
                --colored;
                color[best] = -1;
            }
            return false;
        };

        return dfs();
    }

    bool recolorNeighborhood(int v, vector<int>& color, vector<int>& mark) const {
        fill(mark.begin(), mark.end(), 0);
        for (int ei = g.offsets[v]; ei < g.offsets[v + 1]; ++ei) {
            int u = g.adj[ei];
            if (color[u] >= 0) mark[color[u]] = 1;
        }
        for (int c = 0; c < g.k; ++c) {
            if (!mark[c]) {
                color[v] = c;
                return true;
            }
        }
        for (int ei = g.offsets[v]; ei < g.offsets[v + 1]; ++ei) {
            int u = g.adj[ei];
            int old = color[u];
            fill(mark.begin(), mark.end(), 0);
            for (int ej = g.offsets[u]; ej < g.offsets[u + 1]; ++ej) {
                int w = g.adj[ej];
                if (w != v && color[w] >= 0) mark[color[w]] = 1;
            }
            for (int alt = 0; alt < g.k; ++alt) {
                if (!mark[alt]) {
                    color[u] = alt;
                    fill(mark.begin(), mark.end(), 0);
                    for (int ej = g.offsets[v]; ej < g.offsets[v + 1]; ++ej) {
                        int w = g.adj[ej];
                        if (color[w] >= 0) mark[color[w]] = 1;
                    }
                    for (int c = 0; c < g.k; ++c) {
                        if (!mark[c]) {
                            color[v] = c;
                            return true;
                        }
                    }
                    color[u] = old;
                    break;
                }
            }
        }
        return false;
    }

    bool greedyRepair(vector<int>& color, const vector<int>& order, int rounds) const {
        vector<int> mark(g.k, 0);
        for (int round = 0; round < rounds; ++round) {
            bool changed = false;
            for (int v : order) {
                if (color[v] != -1) continue;
                if (recolorNeighborhood(v, color, mark)) {
                    changed = true;
                }
            }
            if (!changed) break;
        }
        return find(color.begin(), color.end(), -1) == color.end();
    }

    vector<int> randomizedOrder() {
        vector<int> ord(g.n);
        iota(ord.begin(), ord.end(), 0);
        shuffle(ord.begin(), ord.end(), rng);
        stable_sort(ord.begin(), ord.end(), [&](int a, int b) {
            int sa = g.degree[a] + int(rng() % 8);
            int sb = g.degree[b] + int(rng() % 8);
            if (sa != sb) return sa > sb;
            return a < b;
        });
        return ord;
    }

    vector<int> adaptiveHeuristic() {
        vector<int> best;

        vector<int> ord1 = degreeOrder();
        best = greedyColorWithOrder(ord1);
        if (!best.empty()) return best;

        vector<int> ord2 = degeneracyOrder();
        best = greedyColorWithOrder(ord2);
        if (!best.empty()) return best;

        if (g.n <= 200000) {
            for (int iter = 0; iter < 24 && !timedOut(); ++iter) {
                vector<int> ord = (iter % 2 == 0) ? randomizedOrder() : ord2;
                if (iter % 3 == 1) reverse(ord.begin(), ord.end());
                vector<int> cur = greedyColorWithOrder(ord);
                if (!cur.empty()) return cur;
                cur.assign(g.n, -1);
                vector<int> seen(g.k, -1);
                for (int idx = 0; idx < g.n; ++idx) {
                    int v = ord[idx];
                    for (int ei = g.offsets[v]; ei < g.offsets[v + 1]; ++ei) {
                        int u = g.adj[ei];
                        if (cur[u] != -1) seen[cur[u]] = v;
                    }
                    int pick = -1;
                    for (int c = 0; c < g.k; ++c) {
                        if (seen[c] != v) {
                            pick = c;
                            break;
                        }
                    }
                    cur[v] = pick;
                }
                if (greedyRepair(cur, ord, 2) && verify(cur)) return cur;
            }
        } else {
            vector<int> cur(g.n, -1);
            vector<int> seen(g.k, -1);
            for (int idx = 0; idx < g.n; ++idx) {
                int v = ord2[idx];
                for (int ei = g.offsets[v]; ei < g.offsets[v + 1]; ++ei) {
                    int u = g.adj[ei];
                    if (cur[u] != -1) seen[cur[u]] = v;
                }
                for (int c = 0; c < g.k; ++c) {
                    if (seen[c] != v) {
                        cur[v] = c;
                        break;
                    }
                }
            }
            if (greedyRepair(cur, ord2, 1) && verify(cur)) return cur;
        }

        return {};
    }

    vector<int> solve() {
        if (g.n == 0) return {};
        if (g.k <= 0) return {};
        if (g.m == 0) return vector<int>(g.n, 0);
        if (g.k == 1) {
            if (g.m == 0) return vector<int>(g.n, 0);
            return {};
        }

        if (g.k == 2) {
            vector<int> bip;
            if (twoColorBipartite(bip)) return bip;
        }

        if (g.n <= 140) {
            vector<int> exact;
            if (dsaturExact(exact)) return exact;
        }

        vector<int> heuristic = adaptiveHeuristic();
        if (!heuristic.empty() && verify(heuristic)) return heuristic;

        if (g.k >= g.n) {
            vector<int> trivial(g.n);
            iota(trivial.begin(), trivial.end(), 0);
            return trivial;
        }

        return {};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Graph g;
    if (!(cin >> g.n >> g.m >> g.k)) {
        return 0;
    }

    g.degree.assign(g.n, 0);
    g.edges.reserve(g.m);

    for (int i = 0; i < g.m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= g.n || v < 0 || v >= g.n || u == v) continue;
        g.edges.push_back({u, v});
        ++g.degree[u];
        ++g.degree[v];
    }
    g.m = (int)g.edges.size();

    g.offsets.assign(g.n + 1, 0);
    for (int i = 0; i < g.n; ++i) g.offsets[i + 1] = g.offsets[i] + g.degree[i];
    g.adj.assign(g.offsets.back(), -1);
    vector<int> ptr = g.offsets;
    for (const auto& e : g.edges) {
        g.adj[ptr[e.first]++] = e.second;
        g.adj[ptr[e.second]++] = e.first;
    }

    Solver solver(std::move(g));
    vector<int> ans = solver.solve();

    if (ans.empty()) {
        return 0;
    }

    for (int i = 0; i < (int)ans.size(); ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
