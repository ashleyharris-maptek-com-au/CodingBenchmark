declare const require: any;
const fs = require('fs');
const stdout = require('process').stdout;

type Robot = {
  id: number;
  x: number;
  y: number;
};

type Job = {
  id: number;
  x: number;
  y: number;
  priority: number;
};

type Input = {
  grid: string[];
  robots: Robot[];
  jobs: Job[];
};

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);

const grid = data.grid;
const robots = data.robots.slice().sort((a, b) => a.id - b.id);
const jobs = data.jobs.slice();

const height = grid.length;
const width = height === 0 ? 0 : grid[0].length;
const cellCount = width * height;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function cellIndex(x: number, y: number): number {
  return y * width + x;
}

function isOpen(x: number, y: number): boolean {
  return inBounds(x, y) && grid[y].charCodeAt(x) !== 35;
}

type Point = {
  x: number;
  y: number;
};

const points: Point[] = [];
for (const robot of robots) {
  points.push({ x: robot.x, y: robot.y });
}
for (const job of jobs) {
  points.push({ x: job.x, y: job.y });
}

const pointCount = points.length;
const pointCells = new Int32Array(pointCount);
for (let i = 0; i < pointCount; i++) {
  const p = points[i];
  if (!isOpen(p.x, p.y)) {
    throw new Error('Robot or job is placed on a blocked cell');
  }
  pointCells[i] = cellIndex(p.x, p.y);
}

const distanceMatrix: Int32Array[] = new Array(pointCount);

function bfsFromPoint(sourcePointIndex: number): Int32Array {
  const dist = new Int32Array(cellCount);
  dist.fill(-1);

  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;

  const startCell = pointCells[sourcePointIndex];
  dist[startCell] = 0;
  queue[tail++] = startCell;

  while (head < tail) {
    const current = queue[head++];
    const currentDist = dist[current] + 1;
    const x = current % width;
    const y = (current - x) / width;

    let nx = x + 1;
    let ny = y;
    if (nx < width) {
      const next = current + 1;
      if (grid[ny].charCodeAt(nx) !== 35 && dist[next] === -1) {
        dist[next] = currentDist;
        queue[tail++] = next;
      }
    }

    nx = x - 1;
    ny = y;
    if (nx >= 0) {
      const next = current - 1;
      if (grid[ny].charCodeAt(nx) !== 35 && dist[next] === -1) {
        dist[next] = currentDist;
        queue[tail++] = next;
      }
    }

    nx = x;
    ny = y + 1;
    if (ny < height) {
      const next = current + width;
      if (grid[ny].charCodeAt(nx) !== 35 && dist[next] === -1) {
        dist[next] = currentDist;
        queue[tail++] = next;
      }
    }

    nx = x;
    ny = y - 1;
    if (ny >= 0) {
      const next = current - width;
      if (grid[ny].charCodeAt(nx) !== 35 && dist[next] === -1) {
        dist[next] = currentDist;
        queue[tail++] = next;
      }
    }
  }

  const row = new Int32Array(pointCount);
  for (let i = 0; i < pointCount; i++) {
    row[i] = dist[pointCells[i]];
  }
  return row;
}

for (let i = 0; i < pointCount; i++) {
  distanceMatrix[i] = bfsFromPoint(i);
}

function compareJobOrder(a: Job, b: Job): number {
  if (a.priority !== b.priority) {
    return b.priority - a.priority;
  }
  return a.id - b.id;
}

jobs.sort(compareJobOrder);

const robotCount = robots.length;
const jobPointIndexById = new Map<number, number>();
for (let i = 0; i < jobs.length; i++) {
  jobPointIndexById.set(jobs[i].id, robotCount + i);
}

const currentPointIndex = new Int32Array(robotCount);
for (let i = 0; i < robotCount; i++) {
  currentPointIndex[i] = i;
}

const maxRobotId = robots.reduce((max, robot) => Math.max(max, robot.id), -1);
const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

for (const job of jobs) {
  const jobPointIndex = jobPointIndexById.get(job.id)!;
  let bestRobotIdx = 0;
  let bestDistance = Number.MAX_SAFE_INTEGER;

  for (let robotIdx = 0; robotIdx < robotCount; robotIdx++) {
    const sourceIndex = currentPointIndex[robotIdx];
    const distance = distanceMatrix[sourceIndex][jobPointIndex];
    const effectiveDistance = distance >= 0 ? distance : Number.MAX_SAFE_INTEGER;

    if (
      effectiveDistance < bestDistance ||
      (effectiveDistance === bestDistance && robots[robotIdx].id < robots[bestRobotIdx].id)
    ) {
      bestDistance = effectiveDistance;
      bestRobotIdx = robotIdx;
    }
  }

  assignments[robots[bestRobotIdx].id].push(job.id);
  currentPointIndex[bestRobotIdx] = jobPointIndex;
}

stdout.write(JSON.stringify({ assignments }));
