use std::io::{self, Read, Write};

const DIR_UP: u8 = 0;
const DIR_DOWN: u8 = 1;
const DIR_LEFT: u8 = 2;
const DIR_RIGHT: u8 = 3;

struct BitSet {
    data: Vec<u64>,
}

impl BitSet {
    fn new(bits: usize) -> Self {
        Self {
            data: vec![0; bits.div_ceil(64)],
        }
    }

    #[inline]
    fn get(&self, idx: usize) -> bool {
        ((self.data[idx >> 6] >> (idx & 63)) & 1) != 0
    }

    #[inline]
    fn set(&mut self, idx: usize) {
        self.data[idx >> 6] |= 1u64 << (idx & 63);
    }
}

struct Packed2 {
    data: Vec<u64>,
}

impl Packed2 {
    fn new(len: usize) -> Self {
        let bits = len * 2;
        Self {
            data: vec![0; bits.div_ceil(64)],
        }
    }

    #[inline]
    fn get(&self, idx: usize) -> u8 {
        let bit = idx * 2;
        ((self.data[bit >> 6] >> (bit & 63)) & 3) as u8
    }

    #[inline]
    fn set(&mut self, idx: usize, value: u8) {
        let bit = idx * 2;
        let word = bit >> 6;
        let shift = bit & 63;
        let mask = 3u64 << shift;
        self.data[word] = (self.data[word] & !mask) | ((value as u64) << shift);
    }
}

#[inline]
fn step(idx: usize, dir: u8, width: usize) -> usize {
    match dir {
        DIR_UP => idx - width,
        DIR_DOWN => idx + width,
        DIR_LEFT => idx - 1,
        DIR_RIGHT => idx + 1,
        _ => unreachable!(),
    }
}

fn expand_frontier(
    frontier: &[usize],
    next: &mut Vec<usize>,
    passable: &BitSet,
    my_visited: &mut BitSet,
    my_parent: &mut Packed2,
    other_visited: &BitSet,
    height: usize,
    width: usize,
) -> Option<usize> {
    next.clear();

    for &idx in frontier {
        let r = idx / width;
        let c = idx % width;

        if r > 0 {
            let n = idx - width;
            if passable.get(n) && !my_visited.get(n) {
                my_visited.set(n);
                my_parent.set(n, DIR_DOWN);
                if other_visited.get(n) {
                    return Some(n);
                }
                next.push(n);
            }
        }

        if r + 1 < height {
            let n = idx + width;
            if passable.get(n) && !my_visited.get(n) {
                my_visited.set(n);
                my_parent.set(n, DIR_UP);
                if other_visited.get(n) {
                    return Some(n);
                }
                next.push(n);
            }
        }

        if c > 0 {
            let n = idx - 1;
            if passable.get(n) && !my_visited.get(n) {
                my_visited.set(n);
                my_parent.set(n, DIR_RIGHT);
                if other_visited.get(n) {
                    return Some(n);
                }
                next.push(n);
            }
        }

        if c + 1 < width {
            let n = idx + 1;
            if passable.get(n) && !my_visited.get(n) {
                my_visited.set(n);
                my_parent.set(n, DIR_LEFT);
                if other_visited.get(n) {
                    return Some(n);
                }
                next.push(n);
            }
        }
    }

    None
}

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut p = 0usize;

    fn skip_ws(input: &[u8], p: &mut usize) {
        while *p < input.len() && matches!(input[*p], b' ' | b'\n' | b'\r' | b'\t') {
            *p += 1;
        }
    }

    fn read_usize(input: &[u8], p: &mut usize) -> usize {
        skip_ws(input, p);
        let mut value = 0usize;
        while *p < input.len() && input[*p].is_ascii_digit() {
            value = value * 10 + (input[*p] - b'0') as usize;
            *p += 1;
        }
        value
    }

    let height = read_usize(&input, &mut p);
    let width = read_usize(&input, &mut p);
    while p < input.len() && (input[p] == b'\n' || input[p] == b'\r') {
        p += 1;
    }

    if height == 0 || width == 0 {
        print!("0");
        return;
    }

    let cells = height.saturating_mul(width);
    let mut passable = BitSet::new(cells);
    let mut start = None;
    let mut goal = None;

    for r in 0..height {
        for c in 0..width {
            if p >= input.len() {
                print!("0");
                return;
            }
            let ch = input[p];
            p += 1;
            let idx = r * width + c;
            match ch {
                b'#' => {}
                b' ' => passable.set(idx),
                b'A' => {
                    passable.set(idx);
                    start = Some(idx);
                }
                b'B' => {
                    passable.set(idx);
                    goal = Some(idx);
                }
                _ => {
                    print!("0");
                    return;
                }
            }
        }
        if p < input.len() && input[p] == b'\r' {
            p += 1;
        }
        if p < input.len() && input[p] == b'\n' {
            p += 1;
        }
    }

    let start = match start {
        Some(v) => v,
        None => {
            print!("0");
            return;
        }
    };
    let goal = match goal {
        Some(v) => v,
        None => {
            print!("0");
            return;
        }
    };

    if start == goal {
        let mut out = io::BufWriter::new(io::stdout().lock());
        let _ = writeln!(out, "1");
        let _ = writeln!(out, "{} {}", start / width, start % width);
        return;
    }

    let mut vis_a = BitSet::new(cells);
    let mut vis_b = BitSet::new(cells);
    let mut parent_a = Packed2::new(cells);
    let mut parent_b = Packed2::new(cells);
    vis_a.set(start);
    vis_b.set(goal);

    let mut frontier_a = vec![start];
    let mut frontier_b = vec![goal];
    let mut next_a = Vec::new();
    let mut next_b = Vec::new();
    let mut meet = None;

    while !frontier_a.is_empty() && !frontier_b.is_empty() {
        if frontier_a.len() <= frontier_b.len() {
            meet = expand_frontier(
                &frontier_a,
                &mut next_a,
                &passable,
                &mut vis_a,
                &mut parent_a,
                &vis_b,
                height,
                width,
            );
            std::mem::swap(&mut frontier_a, &mut next_a);
        } else {
            meet = expand_frontier(
                &frontier_b,
                &mut next_b,
                &passable,
                &mut vis_b,
                &mut parent_b,
                &vis_a,
                height,
                width,
            );
            std::mem::swap(&mut frontier_b, &mut next_b);
        }

        if meet.is_some() {
            break;
        }
    }

    let meet = match meet {
        Some(v) => v,
        None => {
            print!("0");
            return;
        }
    };

    let mut left = Vec::new();
    let mut cur = meet;
    left.push(cur);
    while cur != start {
        let dir = parent_a.get(cur);
        cur = step(cur, dir, width);
        left.push(cur);
    }
    left.reverse();

    let mut right = Vec::new();
    let mut cur = meet;
    while cur != goal {
        let dir = parent_b.get(cur);
        cur = step(cur, dir, width);
        right.push(cur);
    }

    let mut out = io::BufWriter::new(io::stdout().lock());
    let total = left.len() + right.len();
    let _ = writeln!(out, "{}", total);
    for idx in left.into_iter().chain(right.into_iter()) {
        let _ = writeln!(out, "{} {}", idx / width, idx % width);
    }
}
