declare const require: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Cell = [number, number];

interface DroneInput {
  id: string | number;
  x: number;
  y: number;
}

interface State {
  x: number;
  y: number;
  stepsLeft: number;
}

const probability: number[][] = data.probability;
const drones: DroneInput[] = data.drones;
const maxStepsPerDrone: number = data.maxStepsPerDrone | 0;

const height = Array.isArray(probability) ? probability.length : 0;
const width = height > 0 && Array.isArray(probability[0]) ? probability[0].length : 0;

const DIRS: ReadonlyArray<readonly [number, number]> = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

function inBounds(x: number, y: number): boolean {
  return y >= 0 && y < height && x >= 0 && x < width;
}

function isOpen(x: number, y: number): boolean {
  return inBounds(x, y) && probability[y][x] !== -1;
}

function cellId(x: number, y: number): number {
  return y * width + x;
}

function idToCell(id: number): Cell {
  return [id % width, Math.floor(id / width)];
}

const totalCells = width * height;
const openCells: number[] = [];
const positiveCells: number[] = [];
const cellValue = new Float64Array(totalCells);

for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const v = probability[y][x];
    if (v !== -1) {
      const id = cellId(x, y);
      openCells.push(id);
      if (v > 0) {
        positiveCells.push(id);
        cellValue[id] = v;
      }
    }
  }
}

function bfsFrom(startX: number, startY: number): Int32Array {
  const dist = new Int32Array(totalCells);
  dist.fill(-1);
  if (!isOpen(startX, startY)) {
    return dist;
  }

  const queue = new Int32Array(totalCells);
  let head = 0;
  let tail = 0;
  const startId = cellId(startX, startY);
  dist[startId] = 0;
  queue[tail++] = startId;

  while (head < tail) {
    const cur = queue[head++];
    const cx = cur % width;
    const cy = (cur / width) | 0;
    const base = dist[cur] + 1;

    for (let i = 0; i < 4; i++) {
      const nx = cx + DIRS[i][0];
      const ny = cy + DIRS[i][1];
      if (!isOpen(nx, ny)) continue;
      const nid = cellId(nx, ny);
      if (dist[nid] !== -1) continue;
      dist[nid] = base;
      queue[tail++] = nid;
    }
  }

  return dist;
}

const droneStartIds: number[] = new Array(drones.length);
const droneDists: Int32Array[] = new Array(drones.length);

for (let i = 0; i < drones.length; i++) {
  const d = drones[i];
  const sx = d.x | 0;
  const sy = d.y | 0;
  droneStartIds[i] = isOpen(sx, sy) ? cellId(sx, sy) : -1;
  droneDists[i] = bfsFrom(sx, sy);
}

const assignedTo = new Int32Array(totalCells);
assignedTo.fill(-1);

interface Candidate {
  droneIndex: number;
  cell: number;
  gain: number;
}

const droneTargets: number[][] = Array.from({ length: drones.length }, () => []);

function selectAssignments(): void {
  const reachableCounts = new Int32Array(drones.length);
  for (let di = 0; di < drones.length; di++) {
    const dist = droneDists[di];
    let count = 0;
    for (let i = 0; i < positiveCells.length; i++) {
      const cid = positiveCells[i];
      const d = dist[cid];
      if (d >= 0 && d <= maxStepsPerDrone) count++;
    }
    reachableCounts[di] = count;
  }

  const scarcityWeight = new Float64Array(drones.length);
  for (let di = 0; di < drones.length; di++) {
    scarcityWeight[di] = 1 / Math.max(1, reachableCounts[di]);
  }

  const allCandidates: Candidate[] = [];
  for (let i = 0; i < positiveCells.length; i++) {
    const cid = positiveCells[i];
    const baseValue = cellValue[cid];
    for (let di = 0; di < drones.length; di++) {
      const dist = droneDists[di][cid];
      if (dist < 0 || dist > maxStepsPerDrone) continue;
      const closeness = 1 / (1 + dist);
      const gain = baseValue * (0.70 * closeness + 0.30 * scarcityWeight[di]);
      allCandidates.push({ droneIndex: di, cell: cid, gain });
    }
  }

  allCandidates.sort((a, b) => b.gain - a.gain);

  const maxTargetsPerDrone = Math.max(8, Math.min(48, Math.floor(Math.sqrt(Math.max(1, positiveCells.length))) + 8));
  const droneQuota = new Int32Array(drones.length);

  for (let i = 0; i < allCandidates.length; i++) {
    const c = allCandidates[i];
    if (assignedTo[c.cell] !== -1) continue;
    if (droneQuota[c.droneIndex] >= maxTargetsPerDrone) continue;
    assignedTo[c.cell] = c.droneIndex;
    droneTargets[c.droneIndex].push(c.cell);
    droneQuota[c.droneIndex]++;
  }

  for (let i = 0; i < positiveCells.length; i++) {
    const cid = positiveCells[i];
    if (assignedTo[cid] !== -1) continue;
    let bestDrone = -1;
    let bestScore = -1;
    for (let di = 0; di < drones.length; di++) {
      const dist = droneDists[di][cid];
      if (dist < 0 || dist > maxStepsPerDrone) continue;
      const score = cellValue[cid] / (1 + dist + droneTargets[di].length * 0.7);
      if (score > bestScore) {
        bestScore = score;
        bestDrone = di;
      }
    }
    if (bestDrone >= 0) {
      assignedTo[cid] = bestDrone;
      droneTargets[bestDrone].push(cid);
    }
  }
}

function reconstructPath(dist: Int32Array, startId: number, goalId: number): number[] {
  if (startId === goalId) return [startId];
  const goalDist = dist[goalId];
  if (goalDist < 0) return [startId];

  const path = new Array<number>(goalDist + 1);
  let idx = goalDist;
  let cur = goalId;
  path[idx--] = cur;

  while (cur !== startId && idx >= 0) {
    const cx = cur % width;
    const cy = (cur / width) | 0;
    const curDist = dist[cur];
    let moved = false;
    for (let i = 0; i < 4; i++) {
      const nx = cx + DIRS[i][0];
      const ny = cy + DIRS[i][1];
      if (!isOpen(nx, ny)) continue;
      const nid = cellId(nx, ny);
      if (dist[nid] === curDist - 1) {
        cur = nid;
        path[idx--] = cur;
        moved = true;
        break;
      }
    }
    if (!moved) {
      return [startId];
    }
  }

  return path;
}

function pickBestNextTarget(currentId: number, stepsLeft: number, availableTargets: number[]): number {
  if (availableTargets.length === 0) return -1;
  const [cx, cy] = idToCell(currentId);
  const distFromCurrent = bfsFrom(cx, cy);

  let bestCell = -1;
  let bestScore = -Infinity;

  for (let i = 0; i < availableTargets.length; i++) {
    const cid = availableTargets[i];
    const d = distFromCurrent[cid];
    if (d < 0 || d > stepsLeft) continue;
    const immediate = cellValue[cid];
    const lookahead = Math.max(0, stepsLeft - d);
    const score = immediate / (1 + d) + Math.min(lookahead, 6) * 1e-6;
    if (score > bestScore) {
      bestScore = score;
      bestCell = cid;
    }
  }

  return bestCell;
}

function planDronePath(droneIndex: number): Cell[] {
  const startId = droneStartIds[droneIndex];
  const startDrone = drones[droneIndex];

  if (startId < 0) {
    return [[startDrone.x | 0, startDrone.y | 0]];
  }

  const pathIds: number[] = [startId];
  let currentId = startId;
  let stepsUsed = 0;

  const availableTargets = droneTargets[droneIndex]
    .filter((cid) => droneDists[droneIndex][cid] >= 0 && droneDists[droneIndex][cid] <= maxStepsPerDrone)
    .sort((a, b) => {
      const va = cellValue[a];
      const vb = cellValue[b];
      if (vb !== va) return vb - va;
      return droneDists[droneIndex][a] - droneDists[droneIndex][b];
    });

  while (stepsUsed < maxStepsPerDrone && availableTargets.length > 0) {
    const stepsLeft = maxStepsPerDrone - stepsUsed;
    const nextTarget = pickBestNextTarget(currentId, stepsLeft, availableTargets);
    if (nextTarget < 0) break;

    const [cx, cy] = idToCell(currentId);
    const distMap = bfsFrom(cx, cy);
    const segment = reconstructPath(distMap, currentId, nextTarget);
    if (segment.length <= 1) {
      const idx = availableTargets.indexOf(nextTarget);
      if (idx >= 0) availableTargets.splice(idx, 1);
      continue;
    }

    for (let i = 1; i < segment.length && stepsUsed < maxStepsPerDrone; i++) {
      pathIds.push(segment[i]);
      stepsUsed++;
    }
    currentId = pathIds[pathIds.length - 1];

    for (let i = availableTargets.length - 1; i >= 0; i--) {
      if (availableTargets[i] === currentId) {
        availableTargets.splice(i, 1);
      }
    }
  }

  while (stepsUsed < maxStepsPerDrone) {
    pathIds.push(currentId);
    stepsUsed++;
  }

  const result: Cell[] = new Array(pathIds.length);
  for (let i = 0; i < pathIds.length; i++) {
    result[i] = idToCell(pathIds[i]);
  }
  return result;
}

function fallbackSweepPath(droneIndex: number): Cell[] {
  const startId = droneStartIds[droneIndex];
  const startDrone = drones[droneIndex];
  if (startId < 0) {
    return [[startDrone.x | 0, startDrone.y | 0]];
  }

  const pathIds: number[] = [startId];
  let currentId = startId;
  let stepsUsed = 0;

  const candidates = positiveCells.length > 0 ? positiveCells : openCells;
  const ranked = candidates
    .filter((cid) => droneDists[droneIndex][cid] >= 0 && droneDists[droneIndex][cid] <= maxStepsPerDrone)
    .sort((a, b) => {
      const sa = (cellValue[a] || 0.000001) / (1 + droneDists[droneIndex][a]);
      const sb = (cellValue[b] || 0.000001) / (1 + droneDists[droneIndex][b]);
      return sb - sa;
    });

  for (let r = 0; r < ranked.length && stepsUsed < maxStepsPerDrone; r++) {
    const target = ranked[r];
    if (target === currentId) continue;
    const [cx, cy] = idToCell(currentId);
    const distMap = bfsFrom(cx, cy);
    const d = distMap[target];
    if (d < 0 || stepsUsed + d > maxStepsPerDrone) continue;
    const segment = reconstructPath(distMap, currentId, target);
    for (let i = 1; i < segment.length && stepsUsed < maxStepsPerDrone; i++) {
      pathIds.push(segment[i]);
      stepsUsed++;
    }
    currentId = target;
  }

  while (stepsUsed < maxStepsPerDrone) {
    pathIds.push(currentId);
    stepsUsed++;
  }

  return pathIds.map(idToCell);
}

selectAssignments();

const paths: Cell[][] = new Array(drones.length);
for (let i = 0; i < drones.length; i++) {
  const path = planDronePath(i);
  const usefulCoverage = new Set(path.map(([x, y]) => cellId(x, y))).size;
  if (usefulCoverage <= 1 && maxStepsPerDrone > 0) {
    paths[i] = fallbackSweepPath(i);
  } else {
    paths[i] = path;
  }
}

process.stdout.write(JSON.stringify({ paths }));
