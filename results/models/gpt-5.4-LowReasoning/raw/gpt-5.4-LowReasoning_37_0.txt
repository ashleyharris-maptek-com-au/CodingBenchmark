using System;
using System.Collections.Generic;
using System.Linq;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        var edges = new List<int>[n];
        var rev = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            edges[i] = new List<int>();
            rev[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            edges[u].Add(v);
            rev[v].Add(u);
        }

        var solver = new DirectedFeedbackVertexSetSolver(edges, rev);
        var answer = solver.Solve();
        answer.Sort();

        Console.WriteLine(answer.Count);
        if (answer.Count > 0)
        {
            Console.WriteLine(string.Join(" ", answer));
        }
        else
        {
            Console.WriteLine();
        }
    }
}

internal sealed class DirectedFeedbackVertexSetSolver
{
    private readonly List<int>[] _g;
    private readonly List<int>[] _rg;
    private readonly int _n;

    public DirectedFeedbackVertexSetSolver(List<int>[] g, List<int>[] rg)
    {
        _g = g;
        _rg = rg;
        _n = g.Length;
    }

    public List<int> Solve()
    {
        var alive = new bool[_n];
        Array.Fill(alive, true);
        return SolveState(alive);
    }

    private List<int> SolveState(bool[] alive)
    {
        alive = Reduce(alive);
        var active = GetActiveVertices(alive);
        if (active.Count == 0) return new List<int>();

        var sccs = GetNonTrivialSccs(alive, active);
        if (sccs.Count == 0) return new List<int>();
        if (sccs.Count > 1)
        {
            var result = new List<int>();
            foreach (var comp in sccs)
            {
                var subAlive = new bool[_n];
                foreach (int v in comp) subAlive[v] = true;
                result.AddRange(SolveState(subAlive));
            }
            return result;
        }

        var only = sccs[0];
        if (only.Count == 1)
        {
            return new List<int> { only[0] };
        }

        var best = HeuristicUpperBound(alive);
        int lowerBound = LowerBound(alive);
        if (lowerBound >= best.Count) return best;

        var cycle = FindCycle(alive);
        cycle.Sort((a, b) => VertexScore(alive, b).CompareTo(VertexScore(alive, a)));

        foreach (int v in cycle)
        {
            var nextAlive = (bool[])alive.Clone();
            nextAlive[v] = false;
            var candidate = SolveState(nextAlive);
            candidate.Add(v);
            if (candidate.Count < best.Count)
            {
                best = candidate;
                if (lowerBound >= best.Count) break;
            }
        }

        return best;
    }

    private bool[] Reduce(bool[] startAlive)
    {
        var alive = (bool[])startAlive.Clone();
        var indeg = new int[_n];
        var outdeg = new int[_n];
        var queue = new Queue<int>();

        for (int v = 0; v < _n; v++)
        {
            if (!alive[v]) continue;
            foreach (int to in _g[v])
            {
                if (alive[to]) outdeg[v]++;
            }
            foreach (int from in _rg[v])
            {
                if (alive[from]) indeg[v]++;
            }
        }

        for (int v = 0; v < _n; v++)
        {
            if (alive[v] && (indeg[v] == 0 || outdeg[v] == 0))
            {
                queue.Enqueue(v);
            }
        }

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (!alive[v]) continue;
            if (indeg[v] > 0 && outdeg[v] > 0) continue;

            alive[v] = false;
            foreach (int to in _g[v])
            {
                if (!alive[to]) continue;
                indeg[to]--;
                if (indeg[to] == 0 || outdeg[to] == 0) queue.Enqueue(to);
            }
            foreach (int from in _rg[v])
            {
                if (!alive[from]) continue;
                outdeg[from]--;
                if (indeg[from] == 0 || outdeg[from] == 0) queue.Enqueue(from);
            }
        }

        return alive;
    }

    private List<int> HeuristicUpperBound(bool[] alive)
    {
        var current = (bool[])alive.Clone();
        var removed = new List<int>();

        while (true)
        {
            current = Reduce(current);
            var active = GetActiveVertices(current);
            if (active.Count == 0) break;

            var sccs = GetNonTrivialSccs(current, active);
            if (sccs.Count == 0) break;

            int bestVertex = -1;
            long bestScore = long.MinValue;
            foreach (var comp in sccs)
            {
                foreach (int v in comp)
                {
                    long score = VertexScore(current, v);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestVertex = v;
                    }
                }
            }

            current[bestVertex] = false;
            removed.Add(bestVertex);
        }

        return removed;
    }

    private int LowerBound(bool[] alive)
    {
        var available = (bool[])alive.Clone();
        int bound = 0;
        while (true)
        {
            var cycle = FindCycle(available);
            if (cycle.Count == 0) break;
            bound++;
            foreach (int v in cycle) available[v] = false;
            available = Reduce(available);
        }
        return bound;
    }

    private long VertexScore(bool[] alive, int v)
    {
        long inCount = 0;
        long outCount = 0;
        foreach (int to in _g[v]) if (alive[to]) outCount++;
        foreach (int from in _rg[v]) if (alive[from]) inCount++;
        return (inCount + 1) * (outCount + 1);
    }

    private List<int> GetActiveVertices(bool[] alive)
    {
        var active = new List<int>();
        for (int i = 0; i < _n; i++)
        {
            if (alive[i]) active.Add(i);
        }
        return active;
    }

    private List<List<int>> GetNonTrivialSccs(bool[] alive, List<int> active)
    {
        var order = new List<int>(active.Count);
        var seen = new bool[_n];
        foreach (int v in active)
        {
            if (!seen[v]) DfsOrder(v, alive, seen, order);
        }

        Array.Fill(seen, false);
        var result = new List<List<int>>();
        for (int i = order.Count - 1; i >= 0; i--)
        {
            int v = order[i];
            if (seen[v]) continue;
            var comp = new List<int>();
            DfsRev(v, alive, seen, comp);
            if (comp.Count > 1 || HasSelfLoop(comp[0], alive))
            {
                result.Add(comp);
            }
        }
        return result;
    }

    private void DfsOrder(int start, bool[] alive, bool[] seen, List<int> order)
    {
        var stack = new Stack<(int v, int idx)>();
        stack.Push((start, 0));
        seen[start] = true;
        while (stack.Count > 0)
        {
            var top = stack.Pop();
            int v = top.v;
            int idx = top.idx;
            if (idx < _g[v].Count)
            {
                stack.Push((v, idx + 1));
                int to = _g[v][idx];
                if (alive[to] && !seen[to])
                {
                    seen[to] = true;
                    stack.Push((to, 0));
                }
            }
            else
            {
                order.Add(v);
            }
        }
    }

    private void DfsRev(int start, bool[] alive, bool[] seen, List<int> comp)
    {
        var stack = new Stack<int>();
        stack.Push(start);
        seen[start] = true;
        while (stack.Count > 0)
        {
            int v = stack.Pop();
            comp.Add(v);
            foreach (int from in _rg[v])
            {
                if (alive[from] && !seen[from])
                {
                    seen[from] = true;
                    stack.Push(from);
                }
            }
        }
    }

    private bool HasSelfLoop(int v, bool[] alive)
    {
        foreach (int to in _g[v])
        {
            if (to == v && alive[v]) return true;
        }
        return false;
    }

    private List<int> FindCycle(bool[] alive)
    {
        var color = new byte[_n];
        var parent = new int[_n];
        Array.Fill(parent, -1);

        for (int s = 0; s < _n; s++)
        {
            if (!alive[s] || color[s] != 0) continue;

            var stack = new Stack<(int v, int idx)>();
            stack.Push((s, 0));
            color[s] = 1;

            while (stack.Count > 0)
            {
                var top = stack.Pop();
                int v = top.v;
                int idx = top.idx;

                if (idx < _g[v].Count)
                {
                    stack.Push((v, idx + 1));
                    int to = _g[v][idx];
                    if (!alive[to]) continue;

                    if (to == v)
                    {
                        return new List<int> { v };
                    }

                    if (color[to] == 0)
                    {
                        color[to] = 1;
                        parent[to] = v;
                        stack.Push((to, 0));
                    }
                    else if (color[to] == 1)
                    {
                        var cycle = new List<int> { to };
                        int cur = v;
                        while (cur != to && cur != -1)
                        {
                            cycle.Add(cur);
                            cur = parent[cur];
                        }
                        return cycle;
                    }
                }
                else
                {
                    color[v] = 2;
                }
            }
        }

        return new List<int>();
    }
}

internal sealed class FastScanner
{
    private readonly System.IO.Stream _input = Console.OpenStandardInput();
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _len;
    private int _ptr;

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
        } while (c <= 32 && c >= 0);

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int value = 0;
        while (c > 32)
        {
            value = value * 10 + (c - '0');
            c = Read();
        }
        return value * sign;
    }

    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _input.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len <= 0) return -1;
        }
        return _buffer[_ptr++];
    }
}
