using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        var edges = new List<int>[n];
        var redges = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            edges[i] = new List<int>();
            redges[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            edges[u].Add(v);
            redges[v].Add(u);
        }

        var solver = new DirectedFvsSolver(n, edges, redges);
        var answer = solver.Solve();
        answer.Sort();

        var sw = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        sw.WriteLine(answer.Count);
        if (answer.Count > 0) sw.WriteLine(string.Join(" ", answer));
        else sw.WriteLine();
        sw.Flush();
    }

    private sealed class DirectedFvsSolver
    {
        private readonly int n;
        private readonly List<int>[] g;
        private readonly List<int>[] rg;

        public DirectedFvsSolver(int n, List<int>[] g, List<int>[] rg)
        {
            this.n = n;
            this.g = g;
            this.rg = rg;
        }

        public List<int> Solve()
        {
            var alive = new bool[n];
            Array.Fill(alive, true);
            var removed = new bool[n];
            var answer = new List<int>();

            ReduceGlobal(alive, removed, answer);
            SolveRecursive(alive, removed, answer);
            return answer;
        }

        private void SolveRecursive(bool[] alive, bool[] removed, List<int> answer)
        {
            while (true)
            {
                ReduceGlobal(alive, removed, answer);
                var sccs = GetNonTrivialSccs(alive);
                if (sccs.Count == 0) return;
                if (sccs.Count == 1) break;

                foreach (var scc in sccs)
                {
                    var localRemoved = SolveComponent(scc);
                    foreach (int v in localRemoved)
                    {
                        if (!removed[v])
                        {
                            removed[v] = true;
                            alive[v] = false;
                            answer.Add(v);
                        }
                    }
                }
            }

            var only = GetNonTrivialSccs(alive);
            if (only.Count == 0) return;
            var best = SolveComponent(only[0]);
            foreach (int v in best)
            {
                if (!removed[v])
                {
                    removed[v] = true;
                    alive[v] = false;
                    answer.Add(v);
                }
            }
        }

        private List<int> SolveComponent(List<int> vertices)
        {
            int cn = vertices.Count;
            var map = new Dictionary<int, int>(cn);
            for (int i = 0; i < cn; i++) map[vertices[i]] = i;

            var cg = new List<int>[cn];
            var crg = new List<int>[cn];
            var selfLoop = new bool[cn];
            for (int i = 0; i < cn; i++)
            {
                cg[i] = new List<int>();
                crg[i] = new List<int>();
            }

            for (int i = 0; i < cn; i++)
            {
                int ov = vertices[i];
                foreach (int to in g[ov])
                {
                    if (!map.TryGetValue(to, out int j)) continue;
                    if (i == j) selfLoop[i] = true;
                    else
                    {
                        cg[i].Add(j);
                        crg[j].Add(i);
                    }
                }
            }

            var state = new ComponentState(vertices.ToArray(), cg, crg, selfLoop);
            return state.Solve();
        }

        private void ReduceGlobal(bool[] alive, bool[] removed, List<int> answer)
        {
            while (true)
            {
                bool changed = false;
                var indeg = new int[n];
                var outdeg = new int[n];
                var selfLoop = new bool[n];

                for (int v = 0; v < n; v++)
                {
                    if (!alive[v]) continue;
                    foreach (int to in g[v])
                    {
                        if (!alive[to]) continue;
                        if (v == to) selfLoop[v] = true;
                        else
                        {
                            outdeg[v]++;
                            indeg[to]++;
                        }
                    }
                }

                for (int v = 0; v < n; v++)
                {
                    if (!alive[v]) continue;
                    if (selfLoop[v])
                    {
                        alive[v] = false;
                        removed[v] = true;
                        answer.Add(v);
                        changed = true;
                    }
                }
                if (changed) continue;

                var q = new Queue<int>();
                var dead = new bool[n];
                for (int v = 0; v < n; v++)
                {
                    if (alive[v] && indeg[v] == 0 || alive[v] && outdeg[v] == 0)
                        q.Enqueue(v);
                }

                while (q.Count > 0)
                {
                    int v = q.Dequeue();
                    if (!alive[v] || dead[v]) continue;
                    if (indeg[v] > 0 && outdeg[v] > 0) continue;
                    dead[v] = true;
                    changed = true;
                    foreach (int to in g[v])
                    {
                        if (!alive[to] || dead[to] || v == to) continue;
                        indeg[to]--;
                        if (indeg[to] == 0 || outdeg[to] == 0) q.Enqueue(to);
                    }
                    foreach (int from in rg[v])
                    {
                        if (!alive[from] || dead[from] || from == v) continue;
                        outdeg[from]--;
                        if (indeg[from] == 0 || outdeg[from] == 0) q.Enqueue(from);
                    }
                }

                if (changed)
                {
                    for (int v = 0; v < n; v++)
                    {
                        if (dead[v]) alive[v] = false;
                    }
                    continue;
                }

                break;
            }
        }

        private List<List<int>> GetNonTrivialSccs(bool[] alive)
        {
            var order = new List<int>();
            var vis = new bool[n];
            for (int s = 0; s < n; s++)
            {
                if (!alive[s] || vis[s]) continue;
                var stack = new Stack<(int v, int idx)>();
                stack.Push((s, 0));
                vis[s] = true;
                while (stack.Count > 0)
                {
                    var cur = stack.Pop();
                    int v = cur.v;
                    int idx = cur.idx;
                    if (idx < g[v].Count)
                    {
                        stack.Push((v, idx + 1));
                        int to = g[v][idx];
                        if (alive[to] && !vis[to])
                        {
                            vis[to] = true;
                            stack.Push((to, 0));
                        }
                    }
                    else
                    {
                        order.Add(v);
                    }
                }
            }

            var compVis = new bool[n];
            var result = new List<List<int>>();
            for (int i = order.Count - 1; i >= 0; i--)
            {
                int s = order[i];
                if (!alive[s] || compVis[s]) continue;
                var comp = new List<int>();
                var st = new Stack<int>();
                st.Push(s);
                compVis[s] = true;
                while (st.Count > 0)
                {
                    int v = st.Pop();
                    comp.Add(v);
                    foreach (int to in rg[v])
                    {
                        if (alive[to] && !compVis[to])
                        {
                            compVis[to] = true;
                            st.Push(to);
                        }
                    }
                }

                if (comp.Count > 1)
                {
                    result.Add(comp);
                }
                else
                {
                    int v = comp[0];
                    foreach (int to in g[v])
                    {
                        if (alive[to] && to == v)
                        {
                            result.Add(comp);
                            break;
                        }
                    }
                }
            }

            return result;
        }
    }

    private sealed class ComponentState
    {
        private readonly int n;
        private readonly int[] original;
        private readonly List<int>[] g;
        private readonly List<int>[] rg;
        private readonly bool[] baseSelfLoop;
        private List<int>? best;

        public ComponentState(int[] original, List<int>[] g, List<int>[] rg, bool[] selfLoop)
        {
            this.original = original;
            this.g = g;
            this.rg = rg;
            this.baseSelfLoop = selfLoop;
            n = original.Length;
        }

        public List<int> Solve()
        {
            var alive = new bool[n];
            Array.Fill(alive, true);

            var forced = new List<int>();
            Reduce(alive, forced);
            if (CountAlive(alive) == 0) return Map(forced);

            best = GreedyUpperBound(alive, forced);
            Search(alive, forced);
            return Map(best ?? forced);
        }

        private void Search(bool[] alive, List<int> removed)
        {
            if (best != null && removed.Count >= best.Count) return;

            var localAlive = (bool[])alive.Clone();
            var localRemoved = new List<int>(removed);
            Reduce(localAlive, localRemoved);

            if (best != null && localRemoved.Count >= best.Count) return;
            if (CountAlive(localAlive) == 0)
            {
                if (best == null || localRemoved.Count < best.Count) best = localRemoved;
                return;
            }

            int lb = LowerBound(localAlive);
            if (best != null && localRemoved.Count + lb >= best.Count) return;

            var cycle = FindCycle(localAlive);
            if (cycle == null || cycle.Count == 0)
            {
                if (best == null || localRemoved.Count < best.Count) best = localRemoved;
                return;
            }

            cycle = cycle.Distinct().OrderByDescending(v => Score(localAlive, v)).ToList();
            foreach (int v in cycle)
            {
                var nextAlive = (bool[])localAlive.Clone();
                var nextRemoved = new List<int>(localRemoved) { v };
                nextAlive[v] = false;
                Search(nextAlive, nextRemoved);
            }
        }

        private int LowerBound(bool[] alive)
        {
            var used = new bool[n];
            int count = 0;
            while (true)
            {
                var cycle = FindCycleAvoidingUsed(alive, used);
                if (cycle == null) break;
                count++;
                foreach (int v in cycle) used[v] = true;
            }
            return count;
        }

        private List<int> GreedyUpperBound(bool[] alive, List<int> baseRemoved)
        {
            var workAlive = (bool[])alive.Clone();
            var removed = new List<int>(baseRemoved);
            Reduce(workAlive, removed);

            while (true)
            {
                var cycle = FindCycle(workAlive);
                if (cycle == null || cycle.Count == 0) break;
                int bestV = cycle[0];
                int bestScore = Score(workAlive, bestV);
                for (int i = 1; i < cycle.Count; i++)
                {
                    int v = cycle[i];
                    int sc = Score(workAlive, v);
                    if (sc > bestScore)
                    {
                        bestScore = sc;
                        bestV = v;
                    }
                }
                workAlive[bestV] = false;
                removed.Add(bestV);
                Reduce(workAlive, removed);
            }

            return removed;
        }

        private int Score(bool[] alive, int v)
        {
            int inDeg = 0, outDeg = 0;
            foreach (int to in g[v]) if (alive[to]) outDeg++;
            foreach (int from in rg[v]) if (alive[from]) inDeg++;
            return inDeg * outDeg + inDeg + outDeg;
        }

        private void Reduce(bool[] alive, List<int> removed)
        {
            while (true)
            {
                bool changed = false;
                var indeg = new int[n];
                var outdeg = new int[n];
                var selfLoop = new bool[n];

                for (int i = 0; i < n; i++)
                {
                    if (!alive[i]) continue;
                    if (baseSelfLoop[i]) selfLoop[i] = true;
                    foreach (int to in g[i])
                    {
                        if (!alive[to]) continue;
                        outdeg[i]++;
                        indeg[to]++;
                    }
                }

                for (int i = 0; i < n; i++)
                {
                    if (!alive[i]) continue;
                    if (selfLoop[i])
                    {
                        alive[i] = false;
                        removed.Add(i);
                        changed = true;
                    }
                }
                if (changed) continue;

                var q = new Queue<int>();
                var peel = new bool[n];
                for (int i = 0; i < n; i++)
                {
                    if (alive[i] && (indeg[i] == 0 || outdeg[i] == 0)) q.Enqueue(i);
                }

                while (q.Count > 0)
                {
                    int v = q.Dequeue();
                    if (!alive[v] || peel[v]) continue;
                    if (indeg[v] > 0 && outdeg[v] > 0) continue;
                    peel[v] = true;
                    changed = true;
                    foreach (int to in g[v])
                    {
                        if (!alive[to] || peel[to]) continue;
                        indeg[to]--;
                        if (indeg[to] == 0 || outdeg[to] == 0) q.Enqueue(to);
                    }
                    foreach (int from in rg[v])
                    {
                        if (!alive[from] || peel[from]) continue;
                        outdeg[from]--;
                        if (indeg[from] == 0 || outdeg[from] == 0) q.Enqueue(from);
                    }
                }

                if (changed)
                {
                    for (int i = 0; i < n; i++) if (peel[i]) alive[i] = false;
                    continue;
                }

                break;
            }
        }

        private List<int>? FindCycle(bool[] alive)
        {
            return FindCycleAvoidingUsed(alive, null);
        }

        private List<int>? FindCycleAvoidingUsed(bool[] alive, bool[]? forbidden)
        {
            var state = new byte[n];
            var parent = new int[n];
            Array.Fill(parent, -1);

            for (int s = 0; s < n; s++)
            {
                if (!alive[s] || state[s] != 0 || (forbidden != null && forbidden[s])) continue;
                var stack = new Stack<(int v, int idx)>();
                stack.Push((s, 0));
                state[s] = 1;

                while (stack.Count > 0)
                {
                    var cur = stack.Pop();
                    int v = cur.v;
                    int idx = cur.idx;
                    if (idx < g[v].Count)
                    {
                        stack.Push((v, idx + 1));
                        int to = g[v][idx];
                        if (!alive[to] || (forbidden != null && forbidden[to])) continue;
                        if (state[to] == 0)
                        {
                            parent[to] = v;
                            state[to] = 1;
                            stack.Push((to, 0));
                        }
                        else if (state[to] == 1)
                        {
                            var cycle = new List<int> { to };
                            int x = v;
                            while (x != to)
                            {
                                cycle.Add(x);
                                x = parent[x];
                            }
                            return cycle;
                        }
                    }
                    else
                    {
                        state[v] = 2;
                    }
                }
            }
            return null;
        }

        private int CountAlive(bool[] alive)
        {
            int cnt = 0;
            for (int i = 0; i < n; i++) if (alive[i]) cnt++;
            return cnt;
        }

        private List<int> Map(List<int> local)
        {
            var result = new List<int>(local.Count);
            foreach (int v in local) result.Add(original[v]);
            return result;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream input;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len, ptr;

        public FastScanner(Stream input)
        {
            this.input = input;
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = input.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sign;
        }
    }
}
