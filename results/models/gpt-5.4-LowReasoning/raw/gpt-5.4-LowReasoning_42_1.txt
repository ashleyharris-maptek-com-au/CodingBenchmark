OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%float_0_05 = OpConstant %float 0.05
%float_n0_05 = OpConstant %float -0.05
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_1 = OpConstant %uint 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_1 = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_gl_PerVertex_in_1 = OpTypePointer Input %arr_gl_PerVertex_in_1
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_float = OpTypePointer Input %float
%arr_v3float_1 = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3float_1 = OpTypePointer Input %arr_v3float_1
%ptr_Input_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_1 Input
%worldPos = OpVariable %ptr_Input_arr_v3float_1 Input
%normal = OpVariable %ptr_Input_arr_v3float_1 Input
%color = OpVariable %ptr_Input_arr_v3float_1 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %func
%label = OpLabel
%color_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%inColor = OpLoad %v3float %color_ptr
%x_ptr = OpAccessChain %ptr_Input_float %gl_in %int_0 %int_0 %int_0
%x = OpLoad %float %x_ptr
%y_ptr = OpAccessChain %ptr_Input_float %gl_in %int_0 %int_0 %int_1
%y = OpLoad %float %y_ptr
%z_ptr = OpAccessChain %ptr_Input_float %gl_in %int_0 %int_0 %int_2
%z = OpLoad %float %z_ptr
%w_ptr = OpAccessChain %ptr_Input_float %gl_in %int_0 %int_0 %int_3
%w = OpLoad %float %w_ptr

%top_y = OpFAdd %float %y %float_0_05
%top = OpCompositeConstruct %v4float %x %top_y %z %w
%top_out = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %top_out %top
OpStore %outColor %inColor
OpEmitVertex

%right_x = OpFAdd %float %x %float_0_05
%right = OpCompositeConstruct %v4float %right_x %y %z %w
%right_out = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %right_out %right
OpStore %outColor %inColor
OpEmitVertex

%left_x = OpFAdd %float %x %float_n0_05
%left = OpCompositeConstruct %v4float %left_x %y %z %w
%left_out = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %left_out %left
OpStore %outColor %inColor
OpEmitVertex

%bottom_y = OpFAdd %float %y %float_n0_05
%bottom = OpCompositeConstruct %v4float %x %bottom_y %z %w
%bottom_out = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %bottom_out %bottom
OpStore %outColor %inColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
