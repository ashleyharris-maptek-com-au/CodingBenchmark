enum Piece : int {
    EMPTY = 0,
    WP = 1, WN, WB, WR, WQ, WK,
    BP = 7, BN, BB, BR, BQ, BK
};

enum Color : int { WHITE = 0, BLACK = 1 };

struct Move {
    int from = -1;
    int to = -1;
    int promotion = EMPTY;
    bool isCapture = false;
    bool isEnPassant = false;
    bool isCastleKing = false;
    bool isCastleQueen = false;
};

struct Undo {
    Move move;
    int captured = EMPTY;
    int movedPiece = EMPTY;
    int prevEnPassant = -1;
    int prevCastling = 0;
    int prevHalfmove = 0;
    int prevFullmove = 1;
};

inline bool isWhite(int p) { return p >= WP && p <= WK; }
inline bool isBlack(int p) { return p >= BP && p <= BK; }
inline int colorOf(int p) { return isBlack(p) ? BLACK : WHITE; }
inline int rankOf(int sq) { return sq / 8; }
inline int fileOf(int sq) { return sq % 8; }
inline bool onBoard(int sq) { return sq >= 0 && sq < 64; }

string squareToString(int sq) {
    string s = "a1";
    s[0] = char('a' + fileOf(sq));
    s[1] = char('1' + rankOf(sq));
    return s;
}

char notationPieceLetter(int piece) {
    switch (piece) {
        case WP:
        case BP: return 'p';
        case WN:
        case BN: return 'N';
        case WB:
        case BB: return 'B';
        case WR:
        case BR: return 'R';
        case WQ:
        case BQ: return 'Q';
        case WK:
        case BK: return 'K';
        default: return '?';
    }
}

int promotedPiece(int side, char c) {
    switch (c) {
        case 'q':
        case 'Q': return side == WHITE ? WQ : BQ;
        case 'r':
        case 'R': return side == WHITE ? WR : BR;
        case 'b':
        case 'B': return side == WHITE ? WB : BB;
        case 'n':
        case 'N': return side == WHITE ? WN : BN;
        default: return EMPTY;
    }
}

struct Position {
    array<int, 64> board{};
    int sideToMove = WHITE;
    int castlingRights = 0; // 1 WK, 2 WQ, 4 BK, 8 BQ
    int enPassant = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
    int kingSq[2] = {4, 60};

    bool parseFEN(const string& fen) {
        board.fill(EMPTY);
        sideToMove = WHITE;
        castlingRights = 0;
        enPassant = -1;
        halfmoveClock = 0;
        fullmoveNumber = 1;

        istringstream iss(fen);
        string pieces, stm, castling, ep;
        if (!(iss >> pieces >> stm >> castling >> ep >> halfmoveClock >> fullmoveNumber)) return false;

        int sq = 56;
        for (char c : pieces) {
            if (c == '/') {
                sq -= 16;
                continue;
            }
            if (isdigit(static_cast<unsigned char>(c))) {
                sq += c - '0';
                continue;
            }
            int p = EMPTY;
            switch (c) {
                case 'P': p = WP; break;
                case 'N': p = WN; break;
                case 'B': p = WB; break;
                case 'R': p = WR; break;
                case 'Q': p = WQ; break;
                case 'K': p = WK; break;
                case 'p': p = BP; break;
                case 'n': p = BN; break;
                case 'b': p = BB; break;
                case 'r': p = BR; break;
                case 'q': p = BQ; break;
                case 'k': p = BK; break;
                default: return false;
            }
            if (!onBoard(sq)) return false;
            board[sq] = p;
            if (p == WK) kingSq[WHITE] = sq;
            if (p == BK) kingSq[BLACK] = sq;
            ++sq;
        }

        sideToMove = (stm == "w") ? WHITE : BLACK;

        if (castling.find('K') != string::npos) castlingRights |= 1;
        if (castling.find('Q') != string::npos) castlingRights |= 2;
        if (castling.find('k') != string::npos) castlingRights |= 4;
        if (castling.find('q') != string::npos) castlingRights |= 8;

        if (ep != "-") {
            if (ep.size() != 2) return false;
            int file = ep[0] - 'a';
            int rank = ep[1] - '1';
            if (file < 0 || file >= 8 || rank < 0 || rank >= 8) return false;
            enPassant = rank * 8 + file;
        }
        return true;
    }
};

bool isSquareAttacked(const Position& pos, int sq, int bySide) {
    const auto& b = pos.board;
    const int f = fileOf(sq);
    const int r = rankOf(sq);

    if (bySide == WHITE) {
        if (f > 0 && r > 0 && b[sq - 9] == WP) return true;
        if (f < 7 && r > 0 && b[sq - 7] == WP) return true;
    } else {
        if (f > 0 && r < 7 && b[sq + 7] == BP) return true;
        if (f < 7 && r < 7 && b[sq + 9] == BP) return true;
    }

    static constexpr int knightDf[8] = {1, 2, 2, 1, -1, -2, -2, -1};
    static constexpr int knightDr[8] = {2, 1, -1, -2, -2, -1, 1, 2};
    for (int i = 0; i < 8; ++i) {
        int nf = f + knightDf[i];
        int nr = r + knightDr[i];
        if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) continue;
        int p = b[nr * 8 + nf];
        if (bySide == WHITE && p == WN) return true;
        if (bySide == BLACK && p == BN) return true;
    }

    static constexpr int bishopDf[4] = {1, 1, -1, -1};
    static constexpr int bishopDr[4] = {1, -1, 1, -1};
    for (int d = 0; d < 4; ++d) {
        int nf = f + bishopDf[d], nr = r + bishopDr[d];
        while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            int p = b[nr * 8 + nf];
            if (p != EMPTY) {
                if (bySide == WHITE && (p == WB || p == WQ)) return true;
                if (bySide == BLACK && (p == BB || p == BQ)) return true;
                break;
            }
            nf += bishopDf[d];
            nr += bishopDr[d];
        }
    }

    static constexpr int rookDf[4] = {1, -1, 0, 0};
    static constexpr int rookDr[4] = {0, 0, 1, -1};
    for (int d = 0; d < 4; ++d) {
        int nf = f + rookDf[d], nr = r + rookDr[d];
        while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            int p = b[nr * 8 + nf];
            if (p != EMPTY) {
                if (bySide == WHITE && (p == WR || p == WQ)) return true;
                if (bySide == BLACK && (p == BR || p == BQ)) return true;
                break;
            }
            nf += rookDf[d];
            nr += rookDr[d];
        }
    }

    for (int df = -1; df <= 1; ++df) {
        for (int dr = -1; dr <= 1; ++dr) {
            if (df == 0 && dr == 0) continue;
            int nf = f + df, nr = r + dr;
            if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) continue;
            int p = b[nr * 8 + nf];
            if (bySide == WHITE && p == WK) return true;
            if (bySide == BLACK && p == BK) return true;
        }
    }
    return false;
}

bool makeMove(Position& pos, const Move& m, Undo& u) {
    u.move = m;
    u.captured = EMPTY;
    u.movedPiece = pos.board[m.from];
    u.prevEnPassant = pos.enPassant;
    u.prevCastling = pos.castlingRights;
    u.prevHalfmove = pos.halfmoveClock;
    u.prevFullmove = pos.fullmoveNumber;

    int piece = pos.board[m.from];
    if (piece == EMPTY) return false;
    int side = pos.sideToMove;
    int capturedSq = m.to;

    if (m.isEnPassant) {
        capturedSq = (side == WHITE) ? (m.to - 8) : (m.to + 8);
    }
    u.captured = pos.board[capturedSq];

    pos.board[m.from] = EMPTY;
    if (m.isEnPassant) pos.board[capturedSq] = EMPTY;

    int placedPiece = (m.promotion != EMPTY) ? m.promotion : piece;
    pos.board[m.to] = placedPiece;

    if (m.isCastleKing) {
        if (side == WHITE) {
            pos.board[7] = EMPTY;
            pos.board[5] = WR;
        } else {
            pos.board[63] = EMPTY;
            pos.board[61] = BR;
        }
    } else if (m.isCastleQueen) {
        if (side == WHITE) {
            pos.board[0] = EMPTY;
            pos.board[3] = WR;
        } else {
            pos.board[56] = EMPTY;
            pos.board[59] = BR;
        }
    }

    if (piece == WK) pos.kingSq[WHITE] = m.to;
    if (piece == BK) pos.kingSq[BLACK] = m.to;

    pos.enPassant = -1;
    if (piece == WP && rankOf(m.from) == 1 && rankOf(m.to) == 3) pos.enPassant = m.from + 8;
    if (piece == BP && rankOf(m.from) == 6 && rankOf(m.to) == 4) pos.enPassant = m.from - 8;

    if (piece == WK) pos.castlingRights &= ~3;
    if (piece == BK) pos.castlingRights &= ~12;
    if (m.from == 0 || m.to == 0) pos.castlingRights &= ~2;
    if (m.from == 7 || m.to == 7) pos.castlingRights &= ~1;
    if (m.from == 56 || m.to == 56) pos.castlingRights &= ~8;
    if (m.from == 63 || m.to == 63) pos.castlingRights &= ~4;

    pos.halfmoveClock = (piece == WP || piece == BP || u.captured != EMPTY) ? 0 : pos.halfmoveClock + 1;
    if (side == BLACK) ++pos.fullmoveNumber;
    pos.sideToMove ^= 1;

    if (isSquareAttacked(pos, pos.kingSq[side], pos.sideToMove)) {
        pos.sideToMove ^= 1;
        pos.fullmoveNumber = u.prevFullmove;
        pos.halfmoveClock = u.prevHalfmove;
        pos.castlingRights = u.prevCastling;
        pos.enPassant = u.prevEnPassant;
        pos.board[m.from] = piece;
        pos.board[m.to] = (m.isEnPassant ? EMPTY : u.captured);
        if (m.isEnPassant) pos.board[capturedSq] = u.captured;
        if (m.isCastleKing) {
            if (side == WHITE) {
                pos.board[7] = WR;
                pos.board[5] = EMPTY;
            } else {
                pos.board[63] = BR;
                pos.board[61] = EMPTY;
            }
        } else if (m.isCastleQueen) {
            if (side == WHITE) {
                pos.board[0] = WR;
                pos.board[3] = EMPTY;
            } else {
                pos.board[56] = BR;
                pos.board[59] = EMPTY;
            }
        }
        pos.kingSq[WHITE] = -1;
        pos.kingSq[BLACK] = -1;
        for (int sq = 0; sq < 64; ++sq) {
            if (pos.board[sq] == WK) pos.kingSq[WHITE] = sq;
            if (pos.board[sq] == BK) pos.kingSq[BLACK] = sq;
        }
        return false;
    }
    return true;
}

void undoMove(Position& pos, const Undo& u) {
    const Move& m = u.move;
    pos.sideToMove ^= 1;
    pos.enPassant = u.prevEnPassant;
    pos.castlingRights = u.prevCastling;
    pos.halfmoveClock = u.prevHalfmove;
    pos.fullmoveNumber = u.prevFullmove;

    pos.board[m.from] = u.movedPiece;
    pos.board[m.to] = (m.isEnPassant ? EMPTY : u.captured);

    if (m.isEnPassant) {
        int capSq = (pos.sideToMove == WHITE) ? (m.to - 8) : (m.to + 8);
        pos.board[capSq] = u.captured;
    }

    if (m.isCastleKing) {
        if (pos.sideToMove == WHITE) {
            pos.board[7] = WR;
            pos.board[5] = EMPTY;
        } else {
            pos.board[63] = BR;
            pos.board[61] = EMPTY;
        }
    } else if (m.isCastleQueen) {
        if (pos.sideToMove == WHITE) {
            pos.board[0] = WR;
            pos.board[3] = EMPTY;
        } else {
            pos.board[56] = BR;
            pos.board[59] = EMPTY;
        }
    }

    if (u.movedPiece == WK) pos.kingSq[WHITE] = m.from;
    else if (u.movedPiece == BK) pos.kingSq[BLACK] = m.from;
    else if (m.promotion != EMPTY) {
        // Kings did not move; nothing special.
    }
}

void addMove(vector<Move>& moves, int from, int to, bool capture = false, int promo = EMPTY, bool ep = false, bool ck = false, bool cq = false) {
    Move m;
    m.from = from;
    m.to = to;
    m.promotion = promo;
    m.isCapture = capture;
    m.isEnPassant = ep;
    m.isCastleKing = ck;
    m.isCastleQueen = cq;
    moves.push_back(m);
}

vector<Move> generatePseudoLegal(const Position& pos) {
    vector<Move> moves;
    const auto& b = pos.board;
    int side = pos.sideToMove;

    for (int sq = 0; sq < 64; ++sq) {
        int p = b[sq];
        if (p == EMPTY || colorOf(p) != side) continue;
        int f = fileOf(sq), r = rankOf(sq);

        if (p == WP || p == BP) {
            int dir = (p == WP) ? 1 : -1;
            int startRank = (p == WP) ? 1 : 6;
            int promoRank = (p == WP) ? 6 : 1;
            int next = sq + dir * 8;
            if (onBoard(next) && b[next] == EMPTY) {
                if (r == promoRank) {
                    addMove(moves, sq, next, false, promotedPiece(side, 'Q'));
                    addMove(moves, sq, next, false, promotedPiece(side, 'R'));
                    addMove(moves, sq, next, false, promotedPiece(side, 'B'));
                    addMove(moves, sq, next, false, promotedPiece(side, 'N'));
                } else {
                    addMove(moves, sq, next);
                    int jump = sq + dir * 16;
                    if (r == startRank && b[jump] == EMPTY) addMove(moves, sq, jump);
                }
            }
            for (int df : {-1, 1}) {
                int nf = f + df;
                int nr = r + dir;
                if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) continue;
                int to = nr * 8 + nf;
                if (b[to] != EMPTY && colorOf(b[to]) != side) {
                    if (r == promoRank) {
                        addMove(moves, sq, to, true, promotedPiece(side, 'Q'));
                        addMove(moves, sq, to, true, promotedPiece(side, 'R'));
                        addMove(moves, sq, to, true, promotedPiece(side, 'B'));
                        addMove(moves, sq, to, true, promotedPiece(side, 'N'));
                    } else {
                        addMove(moves, sq, to, true);
                    }
                }
                if (to == pos.enPassant) addMove(moves, sq, to, true, EMPTY, true);
            }
            continue;
        }

        if (p == WN || p == BN) {
            static constexpr int kDf[8] = {1, 2, 2, 1, -1, -2, -2, -1};
            static constexpr int kDr[8] = {2, 1, -1, -2, -2, -1, 1, 2};
            for (int i = 0; i < 8; ++i) {
                int nf = f + kDf[i], nr = r + kDr[i];
                if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) continue;
                int to = nr * 8 + nf;
                if (b[to] == EMPTY) addMove(moves, sq, to);
                else if (colorOf(b[to]) != side) addMove(moves, sq, to, true);
            }
            continue;
        }

        auto slide = [&](const int* dfs, const int* drs, int n) {
            for (int i = 0; i < n; ++i) {
                int nf = f + dfs[i], nr = r + drs[i];
                while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                    int to = nr * 8 + nf;
                    if (b[to] == EMPTY) {
                        addMove(moves, sq, to);
                    } else {
                        if (colorOf(b[to]) != side) addMove(moves, sq, to, true);
                        break;
                    }
                    nf += dfs[i];
                    nr += drs[i];
                }
            }
        };

        if (p == WB || p == BB) {
            static constexpr int dfs[4] = {1, 1, -1, -1};
            static constexpr int drs[4] = {1, -1, 1, -1};
            slide(dfs, drs, 4);
            continue;
        }
        if (p == WR || p == BR) {
            static constexpr int dfs[4] = {1, -1, 0, 0};
            static constexpr int drs[4] = {0, 0, 1, -1};
            slide(dfs, drs, 4);
            continue;
        }
        if (p == WQ || p == BQ) {
            static constexpr int dfs[8] = {1, 1, -1, -1, 1, -1, 0, 0};
            static constexpr int drs[8] = {1, -1, 1, -1, 0, 0, 1, -1};
            slide(dfs, drs, 8);
            continue;
        }
        if (p == WK || p == BK) {
            for (int df = -1; df <= 1; ++df) {
                for (int dr = -1; dr <= 1; ++dr) {
                    if (df == 0 && dr == 0) continue;
                    int nf = f + df, nr = r + dr;
                    if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) continue;
                    int to = nr * 8 + nf;
                    if (b[to] == EMPTY) addMove(moves, sq, to);
                    else if (colorOf(b[to]) != side) addMove(moves, sq, to, true);
                }
            }
            if (side == WHITE) {
                if ((pos.castlingRights & 1) && b[5] == EMPTY && b[6] == EMPTY &&
                    !isSquareAttacked(pos, 4, BLACK) && !isSquareAttacked(pos, 5, BLACK) && !isSquareAttacked(pos, 6, BLACK)) {
                    addMove(moves, 4, 6, false, EMPTY, false, true, false);
                }
                if ((pos.castlingRights & 2) && b[3] == EMPTY && b[2] == EMPTY && b[1] == EMPTY &&
                    !isSquareAttacked(pos, 4, BLACK) && !isSquareAttacked(pos, 3, BLACK) && !isSquareAttacked(pos, 2, BLACK)) {
                    addMove(moves, 4, 2, false, EMPTY, false, false, true);
                }
            } else {
                if ((pos.castlingRights & 4) && b[61] == EMPTY && b[62] == EMPTY &&
                    !isSquareAttacked(pos, 60, WHITE) && !isSquareAttacked(pos, 61, WHITE) && !isSquareAttacked(pos, 62, WHITE)) {
                    addMove(moves, 60, 62, false, EMPTY, false, true, false);
                }
                if ((pos.castlingRights & 8) && b[59] == EMPTY && b[58] == EMPTY && b[57] == EMPTY &&
                    !isSquareAttacked(pos, 60, WHITE) && !isSquareAttacked(pos, 59, WHITE) && !isSquareAttacked(pos, 58, WHITE)) {
                    addMove(moves, 60, 58, false, EMPTY, false, false, true);
                }
            }
        }
    }
    return moves;
}

vector<Move> generateLegalMoves(Position& pos) {
    vector<Move> legal;
    vector<Move> pseudo = generatePseudoLegal(pos);
    legal.reserve(pseudo.size());
    for (const Move& m : pseudo) {
        Undo u;
        if (makeMove(pos, m, u)) {
            legal.push_back(m);
            undoMove(pos, u);
        }
    }
    return legal;
}

    -50, -40, -30, -20, -20, -30, -40, -50,
    -30, -20, -10,   0,   0, -10, -20, -30,
    -30, -10,  20,  30,  30,  20, -10, -30,
    -30, -10,  30,  40,  40,  30, -10, -30,
    -30, -10,  30,  40,  40,  30, -10, -30,
    -30, -10,  20,  30,  30,  20, -10, -30,
    -30, -30,   0,   0,   0,   0, -30, -30,
    -50, -30, -30, -30, -30, -30, -30, -50
};

int mirrorSq(int sq) {
    return (7 - rankOf(sq)) * 8 + fileOf(sq);
}

int evaluate(const Position& pos) {
    int whiteMat = 0, blackMat = 0;
    int score = 0;

    for (int sq = 0; sq < 64; ++sq) {
        int p = pos.board[sq];
        if (p == EMPTY) continue;

        int baseType = ((p - 1) % 6) + 1;
        int value = pieceValue[baseType];
        int pst = 0;
        int idx = isWhite(p) ? sq : mirrorSq(sq);
        switch (baseType) {
            case 1: pst = pawnTable[idx]; break;
            case 2: pst = knightTable[idx]; break;
            case 3: pst = bishopTable[idx]; break;
            case 4: pst = rookTable[idx]; break;
            case 5: pst = queenTable[idx]; break;
            default: break;
        }

        if (isWhite(p)) {
            whiteMat += value;
            score += value + pst;
        } else {
            blackMat += value;
            score -= value + pst;
        }
    }

    bool endgame = (whiteMat + blackMat) <= 2600;
    int wkIdx = pos.kingSq[WHITE];
    int bkIdx = mirrorSq(pos.kingSq[BLACK]);
    score += endgame ? kingEndTable[wkIdx] : kingMidTable[wkIdx];
    score -= endgame ? kingEndTable[bkIdx] : kingMidTable[bkIdx];

    return (pos.sideToMove == WHITE) ? score : -score;
}

int moveScoreHeuristic(const Position& pos, const Move& m) {
    if (m.isCastleKing || m.isCastleQueen) return 50;
    int moving = pos.board[m.from];
    int captured = m.isEnPassant ? (pos.sideToMove == WHITE ? BP : WP) : pos.board[m.to];
    int score = 0;
    if (captured != EMPTY) {
        int movingType = ((moving - 1) % 6) + 1;
        int capturedType = ((captured - 1) % 6) + 1;
        score += 10 * pieceValue[capturedType] - pieceValue[movingType];
    }
    if (m.promotion != EMPTY) score += 8000 + pieceValue[((m.promotion - 1) % 6) + 1];
    return score;
}

struct Searcher {
    chrono::steady_clock::time_point deadline;
    bool timedOut = false;
    uint64_t nodes = 0;

    bool timeUp() {
        return chrono::steady_clock::now() >= deadline;
    }

    int quiescence(Position& pos, int alpha, int beta) {
        if ((++nodes & 2047ULL) == 0 && timeUp()) {
            timedOut = true;
            return 0;
        }

        int standPat = evaluate(pos);
        if (standPat >= beta) return beta;
        if (standPat > alpha) alpha = standPat;

        vector<Move> moves = generatePseudoLegal(pos);
        stable_sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            return moveScoreHeuristic(pos, a) > moveScoreHeuristic(pos, b);
        });

        for (const Move& m : moves) {
            if (!m.isCapture && !m.isEnPassant && m.promotion == EMPTY) continue;
            Undo u;
            if (!makeMove(pos, m, u)) continue;
            int score = -quiescence(pos, -beta, -alpha);
            undoMove(pos, u);
            if (timedOut) return 0;
            if (score >= beta) return beta;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }

    int negamax(Position& pos, int depth, int alpha, int beta) {
        if ((++nodes & 2047ULL) == 0 && timeUp()) {
            timedOut = true;
            return 0;
        }

        if (depth <= 0) return quiescence(pos, alpha, beta);

        vector<Move> moves = generatePseudoLegal(pos);
        stable_sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            return moveScoreHeuristic(pos, a) > moveScoreHeuristic(pos, b);
        });

        bool anyLegal = false;
        for (const Move& m : moves) {
            Undo u;
            if (!makeMove(pos, m, u)) continue;
            anyLegal = true;
            int score = -negamax(pos, depth - 1, -beta, -alpha);
            undoMove(pos, u);
            if (timedOut) return 0;
            if (score >= beta) return beta;
            if (score > alpha) alpha = score;
        }

        if (!anyLegal) {
            if (isSquareAttacked(pos, pos.kingSq[pos.sideToMove], pos.sideToMove ^ 1)) {
                return -MATE_SCORE + depth;
            }
            return 0;
        }
        return alpha;
    }

    Move search(Position root, double timeSeconds) {
        vector<Move> legal = generateLegalMoves(root);
        if (legal.empty()) return Move{};
        if (legal.size() == 1) return legal[0];

        deadline = chrono::steady_clock::now() + chrono::milliseconds(max(1, int(timeSeconds * 1000.0) - 50));
        timedOut = false;
        nodes = 0;

        stable_sort(legal.begin(), legal.end(), [&](const Move& a, const Move& b) {
            return moveScoreHeuristic(root, a) > moveScoreHeuristic(root, b);
        });

        Move bestMove = legal[0];
        int bestScore = -INF;

        for (int depth = 1; depth <= 64; ++depth) {
            if (timeUp()) break;
            Move iterationBest = bestMove;
            int iterationScore = -INF;

            for (const Move& m : legal) {
                Undo u;
                if (!makeMove(root, m, u)) continue;
                int score = -negamax(root, depth - 1, -INF, INF);
                undoMove(root, u);
                if (timedOut) break;
                if (score > iterationScore) {
                    iterationScore = score;
                    iterationBest = m;
                }
            }

            if (timedOut) break;
            bestMove = iterationBest;
            bestScore = iterationScore;

            stable_sort(legal.begin(), legal.end(), [&](const Move& a, const Move& b) {
                if (a.from == bestMove.from && a.to == bestMove.to && a.promotion == bestMove.promotion) return true;
                if (b.from == bestMove.from && b.to == bestMove.to && b.promotion == bestMove.promotion) return false;
                return moveScoreHeuristic(root, a) > moveScoreHeuristic(root, b);
            });

            if (bestScore > MATE_SCORE / 2 || bestScore < -MATE_SCORE / 2) break;
        }
        return bestMove;
    }
};

string moveToNotation(Position& pos, const Move& m) {
    if (m.isCastleKing) return "O-O";
    if (m.isCastleQueen) return "O-O-O";

    string s;
    s += notationPieceLetter(pos.board[m.from]);
    s += squareToString(m.from);
    s += (m.isCapture ? "x" : "");
    s += squareToString(m.to);
    if (m.isEnPassant) s += "ep";
    if (m.promotion != EMPTY) {
        s += "/";
        s += notationPieceLetter(m.promotion);
    }

    Undo u;
    if (makeMove(pos, m, u)) {
        vector<Move> replies = generateLegalMoves(pos);
        bool inCheck = isSquareAttacked(pos, pos.kingSq[pos.sideToMove], pos.sideToMove ^ 1);
        if (inCheck && replies.empty()) s += "#";
        else if (inCheck) s += "+";
        undoMove(pos, u);
    }
    return s;
}

} // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSeconds;
    string fen;
    if (!(cin >> timeoutSeconds)) return 0;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    getline(cin, fen);
    if (fen.empty()) return 0;

    Position pos;
    if (!pos.parseFEN(fen)) return 0;

    Searcher searcher;
    Move best = searcher.search(pos, timeoutSeconds);
    vector<Move> legal = generateLegalMoves(pos);
    if (best.from == -1 && !legal.empty()) best = legal[0];

    cout << moveToNotation(pos, best) << '\n';
    return 0;
}
