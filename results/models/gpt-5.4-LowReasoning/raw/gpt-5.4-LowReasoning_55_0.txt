struct Position {
    array<int, 64> board{};
    bool whiteToMove = true;
    int castling = 0;
    int enPassant = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;

    static bool inBounds(int sq) { return sq >= 0 && sq < 64; }
};

int pieceValue(int piece) {
    switch (abs(piece)) {
        case 1: return 100;
        case 2: return 320;
        case 3: return 330;
        case 4: return 500;
        case 5: return 900;
        case 6: return 0;
        default: return 0;
    }
}

bool isWhite(int p) { return p > 0; }
bool isBlack(int p) { return p < 0; }

string squareToString(int sq) {
    string s;
    s.push_back(static_cast<char>('a' + (sq % 8)));
    s.push_back(static_cast<char>('1' + (sq / 8)));
    return s;
}

int parseSquare(const string& s) {
    if (s.size() != 2) return -1;
    if (s[0] < 'a' || s[0] > 'h' || s[1] < '1' || s[1] > '8') return -1;
    return (s[1] - '1') * 8 + (s[0] - 'a');
}

char promotionLetter(int piece) {
    switch (abs(piece)) {
        case 2: return 'N';
        case 3: return 'B';
        case 4: return 'R';
        case 5: return 'Q';
        default: return 'Q';
    }
}

char movePieceLetter(int piece) {
    switch (abs(piece)) {
        case 1: return 'p';
        case 2: return 'N';
        case 3: return 'B';
        case 4: return 'R';
        case 5: return 'Q';
        case 6: return 'K';
        default: return '?';
    }
}

uint64_t mix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ull;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ull;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebull;
    return x ^ (x >> 31);
}

uint64_t hashPosition(const Position& pos) {
    uint64_t h = 0xcbf29ce484222325ull;
    for (int sq = 0; sq < 64; ++sq) {
        uint64_t v = static_cast<uint64_t>(pos.board[sq] + 8) * 67ull + static_cast<uint64_t>(sq);
        h ^= mix64(v);
        h *= 1099511628211ull;
    }
    h ^= mix64(static_cast<uint64_t>(pos.whiteToMove));
    h ^= mix64(static_cast<uint64_t>(pos.castling + 17));
    h ^= mix64(static_cast<uint64_t>(pos.enPassant + 3));
    return h;
}

bool parseFEN(const string& fen, Position& pos) {
    istringstream iss(fen);
    string boardPart, sidePart, castlePart, epPart;
    if (!(iss >> boardPart >> sidePart >> castlePart >> epPart >> pos.halfmoveClock >> pos.fullmoveNumber)) {
        return false;
    }

    pos = Position{};
    pos.halfmoveClock = 0;
    pos.fullmoveNumber = 1;
    istringstream iss2(fen);
    if (!(iss2 >> boardPart >> sidePart >> castlePart >> epPart >> pos.halfmoveClock >> pos.fullmoveNumber)) {
        return false;
    }

    int rank = 7;
    int file = 0;
    for (char c : boardPart) {
        if (c == '/') {
            --rank;
            file = 0;
            continue;
        }
        if (isdigit(static_cast<unsigned char>(c))) {
            file += c - '0';
            continue;
        }
        if (rank < 0 || file > 7) return false;
        int piece = 0;
        switch (c) {
            case 'P': piece = WPAWN; break;
            case 'N': piece = WKNIGHT; break;
            case 'B': piece = WBISHOP; break;
            case 'R': piece = WROOK; break;
            case 'Q': piece = WQUEEN; break;
            case 'K': piece = WKING; break;
            case 'p': piece = BPAWN; break;
            case 'n': piece = BKNIGHT; break;
            case 'b': piece = BBISHOP; break;
            case 'r': piece = BROOK; break;
            case 'q': piece = BQUEEN; break;
            case 'k': piece = BKING; break;
            default: return false;
        }
        pos.board[rank * 8 + file] = piece;
        ++file;
    }

    pos.whiteToMove = (sidePart == "w");
    pos.castling = 0;
    if (castlePart.find('K') != string::npos) pos.castling |= 1;
    if (castlePart.find('Q') != string::npos) pos.castling |= 2;
    if (castlePart.find('k') != string::npos) pos.castling |= 4;
    if (castlePart.find('q') != string::npos) pos.castling |= 8;
    pos.enPassant = (epPart == "-") ? -1 : parseSquare(epPart);
    return true;
}

bool sameColor(int a, int b) {
    return a != EMPTY && b != EMPTY && ((a > 0) == (b > 0));
}

bool isSquareAttacked(const Position& pos, int sq, bool byWhite) {
    int file = sq % 8;
    int rank = sq / 8;

    const int pawnDir = byWhite ? -1 : 1;
    for (int df : {-1, 1}) {
        int nf = file + df;
        int nr = rank + pawnDir;
        if (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            int p = pos.board[nr * 8 + nf];
            if (p == (byWhite ? WPAWN : BPAWN)) return true;
        }
    }

    static const int knightOffsets[8][2] = {
        {1, 2}, {2, 1}, {2, -1}, {1, -2},
        {-1, -2}, {-2, -1}, {-2, 1}, {-1, 2}
    };
    for (const auto& off : knightOffsets) {
        int nf = file + off[0], nr = rank + off[1];
        if (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            int p = pos.board[nr * 8 + nf];
            if (p == (byWhite ? WKNIGHT : BKNIGHT)) return true;
        }
    }

    static const int bishopDirs[4][2] = {{1,1},{1,-1},{-1,1},{-1,-1}};
    for (const auto& dir : bishopDirs) {
        int nf = file + dir[0], nr = rank + dir[1];
        while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            int p = pos.board[nr * 8 + nf];
            if (p != EMPTY) {
                if (p == (byWhite ? WBISHOP : BBISHOP) || p == (byWhite ? WQUEEN : BQUEEN)) return true;
                break;
            }
            nf += dir[0];
            nr += dir[1];
        }
    }

    static const int rookDirs[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for (const auto& dir : rookDirs) {
        int nf = file + dir[0], nr = rank + dir[1];
        while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            int p = pos.board[nr * 8 + nf];
            if (p != EMPTY) {
                if (p == (byWhite ? WROOK : BROOK) || p == (byWhite ? WQUEEN : BQUEEN)) return true;
                break;
            }
            nf += dir[0];
            nr += dir[1];
        }
    }

    for (int df = -1; df <= 1; ++df) {
        for (int dr = -1; dr <= 1; ++dr) {
            if (df == 0 && dr == 0) continue;
            int nf = file + df, nr = rank + dr;
            if (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                int p = pos.board[nr * 8 + nf];
                if (p == (byWhite ? WKING : BKING)) return true;
            }
        }
    }

    return false;
}

int findKing(const Position& pos, bool white) {
    int king = white ? WKING : BKING;
    for (int i = 0; i < 64; ++i) {
        if (pos.board[i] == king) return i;
    }
    return -1;
}

bool inCheck(const Position& pos, bool white) {
    int kingSq = findKing(pos, white);
    if (kingSq < 0) return false;
    return isSquareAttacked(pos, kingSq, !white);
}

void addMove(vector<Move>& moves, int from, int to, int flags = 0, int promotion = 0) {
    moves.push_back(Move{from, to, promotion, flags, 0});
}

void generatePseudoMoves(const Position& pos, vector<Move>& moves, bool capturesOnly) {
    moves.clear();
    const bool white = pos.whiteToMove;
    for (int sq = 0; sq < 64; ++sq) {
        int p = pos.board[sq];
        if (p == EMPTY || isWhite(p) != white) continue;
        int file = sq % 8;
        int rank = sq / 8;
        int absP = abs(p);

        if (absP == 1) {
            int forward = white ? 8 : -8;
            int startRank = white ? 1 : 6;
            int promoRank = white ? 6 : 1;
            int next = sq + forward;
            if (!capturesOnly && rank != promoRank && Position::inBounds(next) && pos.board[next] == EMPTY) {
                addMove(moves, sq, next);
                int dbl = sq + 2 * forward;
                if (rank == startRank && pos.board[dbl] == EMPTY) addMove(moves, sq, dbl);
            }
            if (!capturesOnly && rank == promoRank && Position::inBounds(next) && pos.board[next] == EMPTY) {
                for (int promo : {white ? WQUEEN : BQUEEN, white ? WROOK : BROOK, white ? WBISHOP : BBISHOP, white ? WKNIGHT : BKNIGHT}) {
                    addMove(moves, sq, next, FLAG_PROMOTION, promo);
                }
            }
            for (int df : {-1, 1}) {
                int nf = file + df;
                if (nf < 0 || nf > 7) continue;
                int to = sq + forward + df;
                if (!Position::inBounds(to)) continue;
                int target = pos.board[to];
                if (target != EMPTY && isWhite(target) != white) {
                    if (rank == promoRank) {
                        for (int promo : {white ? WQUEEN : BQUEEN, white ? WROOK : BROOK, white ? WBISHOP : BBISHOP, white ? WKNIGHT : BKNIGHT}) {
                            addMove(moves, sq, to, FLAG_CAPTURE | FLAG_PROMOTION, promo);
                        }
                    } else {
                        addMove(moves, sq, to, FLAG_CAPTURE);
                    }
                }
                if (to == pos.enPassant) addMove(moves, sq, to, FLAG_CAPTURE | FLAG_EN_PASSANT);
            }
        } else if (absP == 2) {
            static const int offs[8][2] = {
                {1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}
            };
            for (const auto& off : offs) {
                int nf = file + off[0], nr = rank + off[1];
                if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
                int to = nr * 8 + nf;
                int target = pos.board[to];
                if (target == EMPTY) {
                    if (!capturesOnly) addMove(moves, sq, to);
                } else if (isWhite(target) != white) {
                    addMove(moves, sq, to, FLAG_CAPTURE);
                }
            }
        } else if (absP == 3 || absP == 4 || absP == 5) {
            static const int bishopDirs[4][2] = {{1,1},{1,-1},{-1,1},{-1,-1}};
            static const int rookDirs[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
            const int (*dirs)[2] = (absP == 3) ? bishopDirs : rookDirs;
            int dirCount = (absP == 5) ? 8 : 4;
            int combined[8][2];
            if (absP == 5) {
                for (int i = 0; i < 4; ++i) {
                    combined[i][0] = bishopDirs[i][0];
                    combined[i][1] = bishopDirs[i][1];
                    combined[i + 4][0] = rookDirs[i][0];
                    combined[i + 4][1] = rookDirs[i][1];
                }
                dirs = combined;
            }
            for (int i = 0; i < dirCount; ++i) {
                int nf = file + dirs[i][0], nr = rank + dirs[i][1];
                while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                    int to = nr * 8 + nf;
                    int target = pos.board[to];
                    if (target == EMPTY) {
                        if (!capturesOnly) addMove(moves, sq, to);
                    } else {
                        if (isWhite(target) != white) addMove(moves, sq, to, FLAG_CAPTURE);
                        break;
                    }
                    nf += dirs[i][0];
                    nr += dirs[i][1];
                }
            }
        } else if (absP == 6) {
            for (int df = -1; df <= 1; ++df) {
                for (int dr = -1; dr <= 1; ++dr) {
                    if (df == 0 && dr == 0) continue;
                    int nf = file + df, nr = rank + dr;
                    if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
                    int to = nr * 8 + nf;
                    int target = pos.board[to];
                    if (target == EMPTY) {
                        if (!capturesOnly) addMove(moves, sq, to);
                    } else if (isWhite(target) != white) {
                        addMove(moves, sq, to, FLAG_CAPTURE);
                    }
                }
            }
            if (!capturesOnly && white && sq == 4) {
                if ((pos.castling & 1) && pos.board[5] == EMPTY && pos.board[6] == EMPTY &&
                    !isSquareAttacked(pos, 4, false) && !isSquareAttacked(pos, 5, false) && !isSquareAttacked(pos, 6, false)) {
                    addMove(moves, 4, 6, FLAG_CASTLE_KING);
                }
                if ((pos.castling & 2) && pos.board[1] == EMPTY && pos.board[2] == EMPTY && pos.board[3] == EMPTY &&
                    !isSquareAttacked(pos, 4, false) && !isSquareAttacked(pos, 3, false) && !isSquareAttacked(pos, 2, false)) {
                    addMove(moves, 4, 2, FLAG_CASTLE_QUEEN);
                }
            }
            if (!capturesOnly && !white && sq == 60) {
                if ((pos.castling & 4) && pos.board[61] == EMPTY && pos.board[62] == EMPTY &&
                    !isSquareAttacked(pos, 60, true) && !isSquareAttacked(pos, 61, true) && !isSquareAttacked(pos, 62, true)) {
                    addMove(moves, 60, 62, FLAG_CASTLE_KING);
                }
                if ((pos.castling & 8) && pos.board[57] == EMPTY && pos.board[58] == EMPTY && pos.board[59] == EMPTY &&
                    !isSquareAttacked(pos, 60, true) && !isSquareAttacked(pos, 59, true) && !isSquareAttacked(pos, 58, true)) {
                    addMove(moves, 60, 58, FLAG_CASTLE_QUEEN);
                }
            }
        }
    }
}

bool makeMove(Position& pos, const Move& move, Undo& undo) {
    undo.move = move;
    undo.prevEnPassant = pos.enPassant;
    undo.prevCastling = pos.castling;
    undo.prevHalfmove = pos.halfmoveClock;
    undo.prevFullmove = pos.fullmoveNumber;
    undo.movedPiece = pos.board[move.from];
    undo.capturedPiece = (move.flags & FLAG_EN_PASSANT) ? pos.board[pos.whiteToMove ? move.to - 8 : move.to + 8] : pos.board[move.to];

    int piece = pos.board[move.from];
    pos.board[move.from] = EMPTY;
    pos.enPassant = -1;

    if (abs(piece) == 1 || (move.flags & FLAG_CAPTURE)) pos.halfmoveClock = 0;
    else ++pos.halfmoveClock;

    if (move.flags & FLAG_EN_PASSANT) {
        int capSq = pos.whiteToMove ? move.to - 8 : move.to + 8;
        pos.board[capSq] = EMPTY;
    }

    if (move.flags & FLAG_CASTLE_KING) {
        pos.board[move.to] = piece;
        if (piece == WKING) {
            pos.board[7] = EMPTY;
            pos.board[5] = WROOK;
        } else {
            pos.board[63] = EMPTY;
            pos.board[61] = BROOK;
        }
    } else if (move.flags & FLAG_CASTLE_QUEEN) {
        pos.board[move.to] = piece;
        if (piece == WKING) {
            pos.board[0] = EMPTY;
            pos.board[3] = WROOK;
        } else {
            pos.board[56] = EMPTY;
            pos.board[59] = BROOK;
        }
    } else if (move.flags & FLAG_PROMOTION) {
        pos.board[move.to] = move.promotion;
    } else {
        pos.board[move.to] = piece;
    }

    if (piece == WKING) pos.castling &= ~3;
    if (piece == BKING) pos.castling &= ~12;
    if (move.from == 0 || move.to == 0) pos.castling &= ~2;
    if (move.from == 7 || move.to == 7) pos.castling &= ~1;
    if (move.from == 56 || move.to == 56) pos.castling &= ~8;
    if (move.from == 63 || move.to == 63) pos.castling &= ~4;

    if (abs(piece) == 1 && abs(move.to - move.from) == 16) {
        pos.enPassant = (move.from + move.to) / 2;
    }

    bool movingWhite = pos.whiteToMove;
    pos.whiteToMove = !pos.whiteToMove;
    if (pos.whiteToMove) ++pos.fullmoveNumber;

    if (inCheck(pos, movingWhite)) {
        pos.whiteToMove = movingWhite;
        pos.enPassant = undo.prevEnPassant;
        pos.castling = undo.prevCastling;
        pos.halfmoveClock = undo.prevHalfmove;
        pos.fullmoveNumber = undo.prevFullmove;
        pos.board[move.from] = piece;
        if (move.flags & FLAG_EN_PASSANT) {
            pos.board[move.to] = EMPTY;
            int capSq = movingWhite ? move.to - 8 : move.to + 8;
            pos.board[capSq] = undo.capturedPiece;
        } else if (move.flags & FLAG_CASTLE_KING) {
            pos.board[move.to] = EMPTY;
            if (piece == WKING) {
                pos.board[7] = WROOK;
                pos.board[5] = EMPTY;
            } else {
                pos.board[63] = BROOK;
                pos.board[61] = EMPTY;
            }
        } else if (move.flags & FLAG_CASTLE_QUEEN) {
            pos.board[move.to] = EMPTY;
            if (piece == WKING) {
                pos.board[0] = WROOK;
                pos.board[3] = EMPTY;
            } else {
                pos.board[56] = BROOK;
                pos.board[59] = EMPTY;
            }
        } else {
            pos.board[move.to] = undo.capturedPiece;
        }
        return false;
    }

    return true;
}

void unmakeMove(Position& pos, const Undo& undo) {
    const Move& move = undo.move;
    pos.whiteToMove = !pos.whiteToMove;
    pos.enPassant = undo.prevEnPassant;
    pos.castling = undo.prevCastling;
    pos.halfmoveClock = undo.prevHalfmove;
    pos.fullmoveNumber = undo.prevFullmove;

    pos.board[move.from] = undo.movedPiece;
    if (move.flags & FLAG_EN_PASSANT) {
        pos.board[move.to] = EMPTY;
        int capSq = pos.whiteToMove ? move.to - 8 : move.to + 8;
        pos.board[capSq] = undo.capturedPiece;
    } else if (move.flags & FLAG_CASTLE_KING) {
        pos.board[move.to] = EMPTY;
        if (undo.movedPiece == WKING) {
            pos.board[7] = WROOK;
            pos.board[5] = EMPTY;
        } else {
            pos.board[63] = BROOK;
            pos.board[61] = EMPTY;
        }
    } else if (move.flags & FLAG_CASTLE_QUEEN) {
        pos.board[move.to] = EMPTY;
        if (undo.movedPiece == WKING) {
            pos.board[0] = WROOK;
            pos.board[3] = EMPTY;
        } else {
            pos.board[56] = BROOK;
            pos.board[59] = EMPTY;
        }
    } else {
        pos.board[move.to] = undo.capturedPiece;
    }
}

vector<Move> generateLegalMoves(Position& pos, bool capturesOnly = false) {
    vector<Move> pseudo;
    vector<Move> legal;
    generatePseudoMoves(pos, pseudo, capturesOnly);
    legal.reserve(pseudo.size());
    for (const Move& mv : pseudo) {
        Undo u;
        if (makeMove(pos, mv, u)) {
            legal.push_back(mv);
            unmakeMove(pos, u);
        }
    }
    return legal;
}

const int kingTableEnd[64] = {
    -50,-40,-30,-20,-20,-30,-40,-50,
    -30,-20,-10,0,0,-10,-20,-30,
    -30,-10,20,30,30,20,-10,-30,
    -30,-10,30,40,40,30,-10,-30,
    -30,-10,30,40,40,30,-10,-30,
    -30,-10,20,30,30,20,-10,-30,
    -30,-30,0,0,0,0,-30,-30,
    -50,-30,-30,-30,-30,-30,-30,-50
};

int pstValue(int piece, int sq, bool endgame) {
    int idx = (piece > 0) ? sq : (56 ^ sq);
    switch (abs(piece)) {
        case 1: return pawnTable[idx];
        case 2: return knightTable[idx];
        case 3: return bishopTable[idx];
        case 4: return rookTable[idx];
        case 5: return queenTable[idx];
        case 6: return endgame ? kingTableEnd[idx] : kingTableMid[idx];
        default: return 0;
    }
}

int evaluate(const Position& pos) {
    int score = 0;
    int materialNoPawns = 0;
    for (int sq = 0; sq < 64; ++sq) {
        int p = pos.board[sq];
        if (p == EMPTY) continue;
        int val = pieceValue(p);
        if (abs(p) != 1) materialNoPawns += val;
    }
    bool endgame = materialNoPawns < 2600;

    for (int sq = 0; sq < 64; ++sq) {
        int p = pos.board[sq];
        if (p == EMPTY) continue;
        int val = pieceValue(p) + pstValue(p, sq, endgame);
        score += (p > 0) ? val : -val;
    }

    Position whitePos = pos;
    whitePos.whiteToMove = true;
    Position blackPos = pos;
    blackPos.whiteToMove = false;
    auto whiteMoves = generateLegalMoves(whitePos, false);
    auto blackMoves = generateLegalMoves(blackPos, false);
    score += static_cast<int>(whiteMoves.size()) * 4;
    score -= static_cast<int>(blackMoves.size()) * 4;
    return pos.whiteToMove ? score : -score;
}

struct Searcher {
    chrono::steady_clock::time_point deadline;
    unordered_map<uint64_t, TTEntry> tt;
    atomic<bool> stop{false};
    Move bestRootMove{};

    bool timeUp() {
        return chrono::steady_clock::now() >= deadline;
    }

    int scoreMove(const Position& pos, const Move& mv) {
        int victim = (mv.flags & FLAG_EN_PASSANT) ? (pos.whiteToMove ? BPAWN : WPAWN) : pos.board[mv.to];
        int attacker = pos.board[mv.from];
        int score = 0;
        if (mv.flags & FLAG_CAPTURE) score += 10 * pieceValue(victim) - pieceValue(attacker);
        if (mv.flags & FLAG_PROMOTION) score += pieceValue(mv.promotion) + 800;
        if (mv.flags & (FLAG_CASTLE_KING | FLAG_CASTLE_QUEEN)) score += 50;
        return score;
    }

    int quiescence(Position& pos, int alpha, int beta, int ply) {
        if ((ply & 127) == 0 && timeUp()) {
            stop = true;
            return evaluate(pos);
        }
        int standPat = evaluate(pos);
        if (standPat >= beta) return beta;
        if (standPat > alpha) alpha = standPat;

        auto moves = generateLegalMoves(pos, true);
        for (Move& mv : moves) mv.score = scoreMove(pos, mv);
        sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) { return a.score > b.score; });

        for (const Move& mv : moves) {
            Undo u;
            if (!makeMove(pos, mv, u)) continue;
            int score = -quiescence(pos, -beta, -alpha, ply + 1);
            unmakeMove(pos, u);
            if (stop) return alpha;
            if (score >= beta) return beta;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }

    int negamax(Position& pos, int depth, int alpha, int beta, int ply) {
        if ((ply & 127) == 0 && timeUp()) {
            stop = true;
            return evaluate(pos);
        }

        const int alphaOrig = alpha;
        uint64_t key = hashPosition(pos);
        auto it = tt.find(key);
        if (it != tt.end() && it->second.depth >= depth) {
            const TTEntry& e = it->second;
            if (e.flag == 0) return e.score;
            if (e.flag == 1) alpha = max(alpha, e.score);
            else if (e.flag == 2) beta = min(beta, e.score);
            if (alpha >= beta) return e.score;
        }

        if (depth <= 0) return quiescence(pos, alpha, beta, ply);

        auto moves = generateLegalMoves(pos, false);
        if (moves.empty()) {
            if (inCheck(pos, pos.whiteToMove)) return -MATE_SCORE + ply;
            return 0;
        }

        Move ttMove{};
        bool hasTTMove = false;
        if (it != tt.end()) {
            ttMove = it->second.bestMove;
            hasTTMove = true;
        }
        for (Move& mv : moves) {
            mv.score = scoreMove(pos, mv);
            if (hasTTMove && mv.from == ttMove.from && mv.to == ttMove.to &&
                mv.promotion == ttMove.promotion && mv.flags == ttMove.flags) {
                mv.score += 1000000;
            }
        }
        sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) { return a.score > b.score; });

        Move bestLocal = moves.front();
        int bestScore = -INF;
        for (const Move& mv : moves) {
            Undo u;
            if (!makeMove(pos, mv, u)) continue;
            int score = -negamax(pos, depth - 1, -beta, -alpha, ply + 1);
            unmakeMove(pos, u);
            if (stop) return alpha;
            if (score > bestScore) {
                bestScore = score;
                bestLocal = mv;
            }
            if (score > alpha) {
                alpha = score;
                if (ply == 0) bestRootMove = mv;
            }
            if (alpha >= beta) break;
        }

        TTEntry entry;
        entry.depth = depth;
        entry.score = bestScore;
        entry.bestMove = bestLocal;
        entry.flag = 0;
        if (bestScore <= alphaOrig) entry.flag = 2;
        else if (bestScore >= beta) entry.flag = 1;
        tt[key] = entry;
        return bestScore;
    }

    Move search(Position pos, double seconds) {
        double usable = max(0.1, seconds - 0.05);
        deadline = chrono::steady_clock::now() + chrono::milliseconds(static_cast<int>(usable * 1000.0));
        bestRootMove = Move{};
        auto fallbackMoves = generateLegalMoves(pos, false);
        if (!fallbackMoves.empty()) bestRootMove = fallbackMoves.front();

        for (int depth = 1; depth <= 64; ++depth) {
            int score = negamax(pos, depth, -INF, INF, 0);
            (void)score;
            if (stop || timeUp()) break;
        }
        return bestRootMove;
    }
};

string formatMove(Position& pos, const Move& mv) {
    if (mv.flags & FLAG_CASTLE_KING) return "O-O";
    if (mv.flags & FLAG_CASTLE_QUEEN) return "O-O-O";

    int movingPiece = pos.board[mv.from];
    string out;
    out.push_back(movePieceLetter(movingPiece));
    out += squareToString(mv.from);
    if (mv.flags & FLAG_CAPTURE) out.push_back('x');
    out += squareToString(mv.to);
    if (mv.flags & FLAG_EN_PASSANT) out += "ep";
    if (mv.flags & FLAG_PROMOTION) {
        out.push_back('/');
        out.push_back(promotionLetter(mv.promotion));
    }

    Undo u;
    if (makeMove(pos, mv, u)) {
        auto replies = generateLegalMoves(pos, false);
        bool check = inCheck(pos, pos.whiteToMove);
        if (check) out.push_back(replies.empty() ? '#' : '+');
        unmakeMove(pos, u);
    }
    return out;
}

}  // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSeconds = 0.0;
    string fen;
    if (!(cin >> timeoutSeconds)) return 0;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    if (!getline(cin, fen)) return 0;

    Position pos;
    if (!parseFEN(fen, pos)) {
        cout << "invalid\n";
        return 0;
    }

    Searcher searcher;
    Move best = searcher.search(pos, timeoutSeconds);
    auto legalMoves = generateLegalMoves(pos, false);
    if (legalMoves.empty()) {
        cout << "resign\n";
        return 0;
    }

    bool found = false;
    for (const Move& mv : legalMoves) {
        if (mv.from == best.from && mv.to == best.to && mv.promotion == best.promotion && mv.flags == best.flags) {
            found = true;
            break;
        }
    }
    if (!found) best = legalMoves.front();

    cout << formatMove(pos, best) << '\n';
    return 0;
}
