using System;
using System.Globalization;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        long[,] flow = new long[n, n];
        long[,] dist = new long[n, n];

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                flow[i, j] = fs.NextLong();
            }
        }

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                dist[i, j] = fs.NextLong();
            }
        }

        int[] perm;
        long bestCost;

        if (n <= 10)
        {
            perm = SolveExactDp(flow, dist, n, out bestCost);
        }
        else
        {
            perm = BuildInitialSolution(flow, dist, n);
            bestCost = Evaluate(flow, dist, perm, n);
            ImproveBySwaps(flow, dist, perm, ref bestCost, n);
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        writer.WriteLine(bestCost.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < n; i++)
        {
            if (i > 0) writer.Write(' ');
            writer.Write(perm[i]);
        }
        writer.WriteLine();
        writer.Flush();
    }

    private static int[] SolveExactDp(long[,] flow, long[,] dist, int n, out long bestCost)
    {
        int[] current = new int[n];
        int[] best = new int[n];
        bool[] used = new bool[n];
        bestCost = long.MaxValue;

        void Dfs(int facility, long partialCost)
        {
            if (facility == n)
            {
                if (partialCost < bestCost)
                {
                    bestCost = partialCost;
                    Array.Copy(current, best, n);
                }
                return;
            }

            for (int loc = 0; loc < n; loc++)
            {
                if (used[loc]) continue;

                long add = 0;
                for (int prev = 0; prev < facility; prev++)
                {
                    add += flow[facility, prev] * dist[loc, current[prev]];
                    add += flow[prev, facility] * dist[current[prev], loc];
                }

                long nextCost = partialCost + add;
                if (nextCost >= bestCost) continue;

                used[loc] = true;
                current[facility] = loc;
                Dfs(facility + 1, nextCost);
                used[loc] = false;
            }
        }

        Dfs(0, 0);
        return best;
    }

    private static int[] BuildInitialSolution(long[,] flow, long[,] dist, int n)
    {
        long[] flowPotential = new long[n];
        long[] distPotential = new long[n];

        for (int i = 0; i < n; i++)
        {
            long fs = 0;
            long ds = 0;
            for (int j = 0; j < n; j++)
            {
                fs += flow[i, j] + flow[j, i];
                ds += dist[i, j] + dist[j, i];
            }
            flowPotential[i] = fs;
            distPotential[i] = ds;
        }

        int[] facilities = CreateOrderDescending(flowPotential);
        int[] locations = CreateOrderAscending(distPotential);

        int[] perm = new int[n];
        for (int i = 0; i < n; i++)
        {
            perm[facilities[i]] = locations[i];
        }

        return perm;
    }

    private static void ImproveBySwaps(long[,] flow, long[,] dist, int[] perm, ref long bestCost, int n)
    {
        int[] inv = new int[n];
        for (int i = 0; i < n; i++) inv[perm[i]] = i;

        long[] interaction = ComputeInteraction(flow, perm, dist, n);
        int candidateCount = Math.Min(n, n <= 200 ? 32 : (n <= 800 ? 20 : 12));
        int[] activeFacilities = SelectTopIndices(interaction, candidateCount);
        int[] activeLocations = SelectTopIndicesByLocation(interaction, perm, candidateCount);

        bool improved = true;
        int passes = 0;
        int maxPasses = n <= 200 ? 20 : (n <= 800 ? 12 : 8);

        while (improved && passes < maxPasses)
        {
            improved = false;
            passes++;

            long bestDelta = 0;
            int bestA = -1;
            int bestB = -1;

            for (int ia = 0; ia < activeFacilities.Length; ia++)
            {
                int a = activeFacilities[ia];
                for (int ib = ia + 1; ib < activeFacilities.Length; ib++)
                {
                    int b = activeFacilities[ib];
                    long delta = ComputeSwapDelta(flow, dist, perm, n, a, b);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestA = a;
                        bestB = b;
                    }
                }
            }

            for (int la = 0; la < activeLocations.Length; la++)
            {
                int locA = activeLocations[la];
                int a = inv[locA];
                for (int lb = la + 1; lb < activeLocations.Length; lb++)
                {
                    int locB = activeLocations[lb];
                    int b = inv[locB];
                    if (a == b) continue;
                    long delta = ComputeSwapDelta(flow, dist, perm, n, a, b);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestA = a;
                        bestB = b;
                    }
                }
            }

            if (bestDelta < 0)
            {
                int tmp = perm[bestA];
                perm[bestA] = perm[bestB];
                perm[bestB] = tmp;
                inv[perm[bestA]] = bestA;
                inv[perm[bestB]] = bestB;
                bestCost += bestDelta;

                interaction = ComputeInteraction(flow, perm, dist, n);
                activeFacilities = SelectTopIndices(interaction, candidateCount);
                activeLocations = SelectTopIndicesByLocation(interaction, perm, candidateCount);
                improved = true;
            }
        }
    }

    private static long[] ComputeInteraction(long[,] flow, int[] perm, long[,] dist, int n)
    {
        long[] interaction = new long[n];
        for (int i = 0; i < n; i++)
        {
            long sum = 0;
            int pi = perm[i];
            for (int j = 0; j < n; j++)
            {
                sum += flow[i, j] * dist[pi, perm[j]];
                sum += flow[j, i] * dist[perm[j], pi];
            }
            interaction[i] = sum;
        }
        return interaction;
    }

    private static int[] SelectTopIndices(long[] values, int count)
    {
        int n = values.Length;
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Array.Sort(idx, (a, b) =>
        {
            int cmp = values[b].CompareTo(values[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        int m = Math.Min(count, n);
        int[] result = new int[m];
        Array.Copy(idx, result, m);
        return result;
    }

    private static int[] SelectTopIndicesByLocation(long[] facilityInteraction, int[] perm, int count)
    {
        int n = facilityInteraction.Length;
        long[] locationScores = new long[n];
        for (int f = 0; f < n; f++)
        {
            locationScores[perm[f]] = facilityInteraction[f];
        }
        return SelectTopIndices(locationScores, count);
    }

    private static int[] CreateOrderDescending(long[] values)
    {
        int n = values.Length;
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Array.Sort(idx, (a, b) =>
        {
            int cmp = values[b].CompareTo(values[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });
        return idx;
    }

    private static int[] CreateOrderAscending(long[] values)
    {
        int n = values.Length;
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Array.Sort(idx, (a, b) =>
        {
            int cmp = values[a].CompareTo(values[b]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });
        return idx;
    }

    private static long ComputeSwapDelta(long[,] flow, long[,] dist, int[] perm, int n, int a, int b)
    {
        if (a == b) return 0;

        int pa = perm[a];
        int pb = perm[b];
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];

            delta += flow[a, k] * (dist[pb, pk] - dist[pa, pk]);
            delta += flow[b, k] * (dist[pa, pk] - dist[pb, pk]);
            delta += flow[k, a] * (dist[pk, pb] - dist[pk, pa]);
            delta += flow[k, b] * (dist[pk, pa] - dist[pk, pb]);
        }

        delta += flow[a, a] * (dist[pb, pb] - dist[pa, pa]);
        delta += flow[b, b] * (dist[pa, pa] - dist[pb, pb]);
        delta += flow[a, b] * (dist[pb, pa] - dist[pa, pb]);
        delta += flow[b, a] * (dist[pa, pb] - dist[pb, pa]);

        return delta;
    }

    private static long Evaluate(long[,] flow, long[,] dist, int[] perm, int n)
    {
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            for (int j = 0; j < n; j++)
            {
                total += flow[i, j] * dist[pi, perm[j]];
            }
        }
        return total;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            return (int)NextLong();
        }

        public long NextLong()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }

            return _buffer[_ptr++];
        }
    }
}
