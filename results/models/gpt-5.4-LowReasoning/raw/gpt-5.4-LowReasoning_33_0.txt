using System;
using System.Collections.Generic;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int[][] flow = new int[n][];
        int[][] dist = new int[n][];

        for (int i = 0; i < n; i++)
        {
            flow[i] = new int[n];
            for (int j = 0; j < n; j++) flow[i][j] = fs.NextInt();
        }

        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            for (int j = 0; j < n; j++) dist[i][j] = fs.NextInt();
        }

        int[] perm;
        long bestCost;

        if (n <= 10)
        {
            perm = SolveExactSmall(flow, dist);
            bestCost = ComputeCost(flow, dist, perm);
        }
        else
        {
            perm = SolveHeuristic(flow, dist);
            bestCost = ComputeCost(flow, dist, perm);
        }

        var sb = new StringBuilder();
        sb.Append(bestCost).Append('\n');
        for (int i = 0; i < perm.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        Console.Write(sb.ToString());
    }

    private static int[] SolveExactSmall(int[][] flow, int[][] dist)
    {
        int n = flow.Length;
        var used = new bool[n];
        var cur = new int[n];
        var best = new int[n];
        long bestCost = long.MaxValue;

        void Dfs(int idx, long partial)
        {
            if (partial >= bestCost) return;

            if (idx == n)
            {
                bestCost = partial;
                Array.Copy(cur, best, n);
                return;
            }

            for (int loc = 0; loc < n; loc++)
            {
                if (used[loc]) continue;

                long add = 0;
                for (int i = 0; i < idx; i++)
                {
                    int li = cur[i];
                    add += (long)flow[idx][i] * dist[loc][li];
                    add += (long)flow[i][idx] * dist[li][loc];
                }
                add += (long)flow[idx][idx] * dist[loc][loc];

                used[loc] = true;
                cur[idx] = loc;
                Dfs(idx + 1, partial + add);
                used[loc] = false;
            }
        }

        Dfs(0, 0);
        return best;
    }

    private static int[] SolveHeuristic(int[][] flow, int[][] dist)
    {
        int n = flow.Length;
        long[] facilityScore = new long[n];
        long[] locationScore = new long[n];
        long[] flowOut = new long[n];
        long[] flowIn = new long[n];
        long[] distOut = new long[n];
        long[] distIn = new long[n];

        for (int i = 0; i < n; i++)
        {
            long fo = 0, fi = 0, do_ = 0, di = 0;
            for (int j = 0; j < n; j++)
            {
                fo += flow[i][j];
                fi += flow[j][i];
                do_ += dist[i][j];
                di += dist[j][i];
            }
            flowOut[i] = fo;
            flowIn[i] = fi;
            distOut[i] = do_;
            distIn[i] = di;
            facilityScore[i] = fo + fi;
            locationScore[i] = do_ + di;
        }

        int[] byFacility = CreateOrder(n, i => -facilityScore[i]);
        int[] byLocation = CreateOrder(n, i => locationScore[i]);

        int[] perm1 = new int[n];
        for (int rank = 0; rank < n; rank++) perm1[byFacility[rank]] = byLocation[rank];
        long cost1 = ComputeCost(flow, dist, perm1);

        int[] perm2 = GreedyConstruct(flow, dist, facilityScore, locationScore, flowOut, flowIn, distOut, distIn);
        long cost2 = ComputeCost(flow, dist, perm2);

        int[] bestPerm = cost2 < cost1 ? perm2 : perm1;
        long bestCost = cost2 < cost1 ? cost2 : cost1;

        ImproveBySwaps(flow, dist, facilityScore, locationScore, bestPerm, ref bestCost);
        return bestPerm;
    }

    private static int[] GreedyConstruct(
        int[][] flow,
        int[][] dist,
        long[] facilityScore,
        long[] locationScore,
        long[] flowOut,
        long[] flowIn,
        long[] distOut,
        long[] distIn)
    {
        int n = flow.Length;
        int[] facilities = CreateOrder(n, i => -facilityScore[i]);
        int[] locations = CreateOrder(n, i => locationScore[i]);

        var assignedFacility = new bool[n];
        var assignedLocation = new bool[n];
        var perm = new int[n];
        Array.Fill(perm, -1);

        int seedCount = Math.Min(4, n);
        for (int t = 0; t < seedCount; t++)
        {
            int f = facilities[t];
            int l = locations[t];
            perm[f] = l;
            assignedFacility[f] = true;
            assignedLocation[l] = true;
        }

        for (int step = seedCount; step < n; step++)
        {
            int bestFacility = -1;
            long bestFacilityWeight = long.MinValue;
            for (int f = 0; f < n; f++)
            {
                if (assignedFacility[f]) continue;
                long w = 0;
                for (int g = 0; g < n; g++)
                {
                    if (assignedFacility[g]) w += flow[f][g] + flow[g][f];
                }
                if (w > bestFacilityWeight)
                {
                    bestFacilityWeight = w;
                    bestFacility = f;
                }
            }

            int bestLocation = -1;
            long bestLocationScore = long.MaxValue;
            for (int l = 0; l < n; l++)
            {
                if (assignedLocation[l]) continue;
                long score = 0;
                for (int x = 0; x < n; x++)
                {
                    if (assignedLocation[x]) score += dist[l][x] + dist[x][l];
                }
                if (score < bestLocationScore)
                {
                    bestLocationScore = score;
                    bestLocation = l;
                }
            }

            long bestIncrement = long.MaxValue;
            int chosenLocation = bestLocation;
            for (int l = 0; l < n; l++)
            {
                if (assignedLocation[l]) continue;
                long inc = (long)flow[bestFacility][bestFacility] * dist[l][l];
                for (int g = 0; g < n; g++)
                {
                    if (!assignedFacility[g]) continue;
                    int lg = perm[g];
                    inc += (long)flow[bestFacility][g] * dist[l][lg];
                    inc += (long)flow[g][bestFacility] * dist[lg][l];
                }

                long bias =
                    Math.Abs((flowOut[bestFacility] + flowIn[bestFacility]) - facilityScore[bestFacility]) +
                    Math.Abs((distOut[l] + distIn[l]) - locationScore[l]);

                long combined = inc + (bias >> 4);
                if (combined < bestIncrement)
                {
                    bestIncrement = combined;
                    chosenLocation = l;
                }
            }

            perm[bestFacility] = chosenLocation;
            assignedFacility[bestFacility] = true;
            assignedLocation[chosenLocation] = true;
        }

        return perm;
    }

    private static void ImproveBySwaps(
        int[][] flow,
        int[][] dist,
        long[] facilityScore,
        long[] locationScore,
        int[] perm,
        ref long cost)
    {
        int n = perm.Length;
        int maxCandidates = n <= 300 ? n : (n <= 1000 ? 96 : 64);
        int maxPasses = n <= 200 ? 12 : (n <= 800 ? 8 : 5);

        int[] posToFacility = new int[n];
        for (int f = 0; f < n; f++) posToFacility[perm[f]] = f;

        for (int pass = 0; pass < maxPasses; pass++)
        {
            long[] mismatch = new long[n];
            for (int f = 0; f < n; f++)
            {
                mismatch[f] = facilityScore[f] * locationScore[perm[f]];
            }

            int[] candidates = TopIndices(mismatch, maxCandidates, descending: true);
            bool improved = false;

            for (int ia = 0; ia < candidates.Length; ia++)
            {
                int a = candidates[ia];
                long bestDelta = 0;
                int bestB = -1;

                for (int ib = 0; ib < candidates.Length; ib++)
                {
                    int b = candidates[ib];
                    if (a >= b) continue;

                    long delta = SwapDelta(flow, dist, perm, a, b);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestB = b;
                    }
                }

                if (bestB != -1)
                {
                    int la = perm[a];
                    int lb = perm[bestB];
                    perm[a] = lb;
                    perm[bestB] = la;
                    posToFacility[la] = bestB;
                    posToFacility[lb] = a;
                    cost += bestDelta;
                    improved = true;
                }
            }

            if (!improved) break;
        }
    }

    private static long SwapDelta(int[][] flow, int[][] dist, int[] perm, int a, int b)
    {
        int n = perm.Length;
        int la = perm[a];
        int lb = perm[b];

        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int lk = perm[k];
            delta += (long)flow[a][k] * (dist[lb][lk] - dist[la][lk]);
            delta += (long)flow[b][k] * (dist[la][lk] - dist[lb][lk]);
            delta += (long)flow[k][a] * (dist[lk][lb] - dist[lk][la]);
            delta += (long)flow[k][b] * (dist[lk][la] - dist[lk][lb]);
        }

        delta += (long)flow[a][a] * (dist[lb][lb] - dist[la][la]);
        delta += (long)flow[b][b] * (dist[la][la] - dist[lb][lb]);
        delta += (long)flow[a][b] * (dist[lb][la] - dist[la][lb]);
        delta += (long)flow[b][a] * (dist[la][lb] - dist[lb][la]);

        return delta;
    }

    private static long ComputeCost(int[][] flow, int[][] dist, int[] perm)
    {
        int n = perm.Length;
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            long row = 0;
            for (int j = 0; j < n; j++)
            {
                row += (long)flow[i][j] * dist[pi][perm[j]];
            }
            total += row;
        }
        return total;
    }

    private static int[] CreateOrder(int n, Func<int, long> keySelector)
    {
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) =>
        {
            long ka = keySelector(a);
            long kb = keySelector(b);
            int cmp = ka.CompareTo(kb);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });
        return order;
    }

    private static int[] TopIndices(long[] values, int take, bool descending)
    {
        int n = values.Length;
        int[] idx = new int[n];
        for (int i = 0; i < n; i++) idx[i] = i;
        Array.Sort(idx, (a, b) =>
        {
            int cmp = values[a].CompareTo(values[b]);
            if (descending) cmp = -cmp;
            return cmp != 0 ? cmp : a.CompareTo(b);
        });
        if (take >= n) return idx;
        int[] res = new int[take];
        Array.Copy(idx, res, take);
        return res;
    }

    private sealed class FastScanner
    {
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = Console.OpenStandardInput().Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }
}
