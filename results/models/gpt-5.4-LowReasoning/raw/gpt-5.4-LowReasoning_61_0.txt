declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Point = { x: number; y: number };
type Package = { id: string | number; x: number; y: number; angle: number; radius2: number };

function dist(a: Point, b: Point): number {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
}

function edgeGain(prev: Point, a: Point, b: Point, next: Point): number {
  return dist(prev, a) + dist(b, next) - dist(prev, b) - dist(a, next);
}

function twoOptWindow(route: Package[], depot: Point, window: number, maxPasses: number): void {
  const n = route.length;
  if (n < 4) return;

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;

    for (let i = 0; i < n - 2; i++) {
      const a = i === 0 ? depot : route[i - 1];
      const b = route[i];
      const maxJ = Math.min(n - 2, i + window);

      for (let j = i + 1; j <= maxJ; j++) {
        const c = route[j];
        const d = route[j + 1];
        const gain = dist(a, b) + dist(c, d) - dist(a, c) - dist(b, d);
        if (gain > 1e-9) {
          let left = i;
          let right = j;
          while (left < right) {
            const tmp = route[left];
            route[left] = route[right];
            route[right] = tmp;
            left++;
            right--;
          }
          improved = true;
        }
      }

      const c = route[n - 1];
      const gainToDepot = dist(a, b) + dist(c, depot) - dist(a, c) - dist(b, depot);
      if (gainToDepot > 1e-9) {
        let left = i;
        let right = n - 1;
        while (left < right) {
          const tmp = route[left];
          route[left] = route[right];
          route[right] = tmp;
          left++;
          right--;
        }
        improved = true;
      }
    }

    if (!improved) break;
  }
}

function relocateWindow(route: Package[], depot: Point, window: number, maxPasses: number): void {
  const n = route.length;
  if (n < 3) return;

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;

    for (let i = 0; i < n; i++) {
      const prev = i === 0 ? depot : route[i - 1];
      const node = route[i];
      const next = i === n - 1 ? depot : route[i + 1];
      const removalCost = dist(prev, node) + dist(node, next) - dist(prev, next);

      const startJ = Math.max(0, i - window);
      const endJ = Math.min(n - 1, i + window);

      for (let j = startJ; j <= endJ; j++) {
        if (j === i || j === i - 1) continue;

        const insertPrev = route[j];
        const insertNext = j === n - 1 ? depot : route[j + 1];
        const insertionCost = dist(insertPrev, node) + dist(node, insertNext) - dist(insertPrev, insertNext);
        const gain = removalCost - insertionCost;

        if (gain > 1e-9) {
          route.splice(i, 1);
          const insertAt = j < i ? j + 1 : j;
          route.splice(insertAt, 0, node);
          improved = true;
          break;
        }
      }

      if (improved) break;
    }

    if (!improved) break;
  }
}

function routeDistance(route: Package[], depot: Point): number {
  if (route.length === 0) return 0;
  let total = dist(depot, route[0]);
  for (let i = 1; i < route.length; i++) total += dist(route[i - 1], route[i]);
  total += dist(route[route.length - 1], depot);
  return total;
}

function buildShiftedOrder(sorted: Package[], start: number, reversed: boolean): Package[] {
  const n = sorted.length;
  const order = new Array<Package>(n);
  if (!reversed) {
    for (let i = 0; i < n; i++) order[i] = sorted[(start + i) % n];
  } else {
    for (let i = 0; i < n; i++) {
      let idx = start - i;
      if (idx < 0) idx = ((idx % n) + n) % n;
      order[i] = sorted[idx];
    }
  }
  return order;
}

function candidateStarts(sorted: Package[]): number[] {
  const n = sorted.length;
  if (n <= 1) return [0];

  const gaps: Array<{ index: number; value: number }> = [];
  for (let i = 0; i < n; i++) {
    const a = sorted[i];
    const b = sorted[(i + 1) % n];
    const da = Math.atan2(Math.sin(b.angle - a.angle), Math.cos(b.angle - a.angle));
    const angularGap = da >= 0 ? da : da + Math.PI * 2;
    const spatialGap = dist(a, b);
    gaps.push({ index: (i + 1) % n, value: angularGap * 1000 + spatialGap });
  }
  gaps.sort((lhs, rhs) => rhs.value - lhs.value || lhs.index - rhs.index);

  const starts: number[] = [0];
  const limit = Math.min(n, 12);
  for (let i = 0; i < limit; i++) starts.push(gaps[i].index);

  const seen = new Set<number>();
  const unique: number[] = [];
  for (const start of starts) {
    if (!seen.has(start)) {
      seen.add(start);
      unique.push(start);
    }
  }
  return unique;
}

function solveSingleRoute(depot: Point, packages: Package[]): Array<string | number> {
  const n = packages.length;
  if (n === 0) return [];
  if (n === 1) return [packages[0].id];

  const sorted = packages.slice().sort((a, b) => {
    if (a.angle !== b.angle) return a.angle - b.angle;
    if (a.radius2 !== b.radius2) return a.radius2 - b.radius2;
    if (a.x !== b.x) return a.x - b.x;
    if (a.y !== b.y) return a.y - b.y;
    return String(a.id).localeCompare(String(b.id));
  });

  const starts = candidateStarts(sorted);
  let bestRoute = sorted.slice();
  let bestDistance = Number.POSITIVE_INFINITY;

  for (const reversed of [false, true]) {
    for (const start of starts) {
      const route = buildShiftedOrder(sorted, start, reversed);
      const sizeWindow = n <= 200 ? 48 : n <= 2000 ? 32 : 20;
      const passes = n <= 200 ? 3 : n <= 2000 ? 2 : 1;
      twoOptWindow(route, depot, sizeWindow, passes);
      relocateWindow(route, depot, Math.max(8, Math.floor(sizeWindow / 2)), 1);
      const total = routeDistance(route, depot);
      if (total < bestDistance) {
        bestDistance = total;
        bestRoute = route;
      }
    }
  }

  return bestRoute.map((pkg) => pkg.id);
}

const depot: Point = { x: Number(data.depot[0]), y: Number(data.depot[1]) };
const rawPackages: Array<{ id: string | number; x: number; y: number }> = Array.isArray(data.packages) ? data.packages : [];
const packages: Package[] = rawPackages.map((pkg) => {
  const x = Number(pkg.x);
  const y = Number(pkg.y);
  const dx = x - depot.x;
  const dy = y - depot.y;
  return {
    id: pkg.id,
    x,
    y,
    angle: Math.atan2(dy, dx),
    radius2: dx * dx + dy * dy,
  };
});

const route = solveSingleRoute(depot, packages);
process.stdout.write(JSON.stringify({ routes: route.length === 0 ? [] : [route] }));
