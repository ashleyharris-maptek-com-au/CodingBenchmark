type Point = { x: number; y: number };
type Package = { id: string | number; x: number; y: number };
type InputData = {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
};

declare const require: any;
const fs = require("fs");

const input: string = fs.readFileSync(0, "utf8");
const data: InputData = JSON.parse(input);

const depot: Point = { x: data.depot[0], y: data.depot[1] };
const packages: Package[] = Array.isArray(data.packages) ? data.packages.slice() : [];
const maxRoutes = Math.max(1, Math.floor(Number.isFinite(data.truckCount) ? data.truckCount : 1));

function sqDist(a: Point, b: Point): number {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return dx * dx + dy * dy;
}

function dist(a: Point, b: Point): number {
  return Math.sqrt(sqDist(a, b));
}

function routeDistance(route: Package[]): number {
  if (route.length === 0) return 0;
  let total = dist(depot, route[0]);
  for (let i = 1; i < route.length; i++) {
    total += dist(route[i - 1], route[i]);
  }
  total += dist(route[route.length - 1], depot);
  return total;
}

function polarAngle(p: Package): number {
  return Math.atan2(p.y - depot.y, p.x - depot.x);
}

function radialDistance(p: Package): number {
  return sqDist(p, depot);
}

function seededOrder(items: Package[]): Package[] {
  return items.sort((a, b) => {
    const aa = polarAngle(a);
    const ab = polarAngle(b);
    if (aa !== ab) return aa - ab;
    const ra = radialDistance(a);
    const rb = radialDistance(b);
    if (ra !== rb) return ra - rb;
    return String(a.id).localeCompare(String(b.id));
  });
}

function buildInitialRoutes(items: Package[], truckCount: number): Package[][] {
  const n = items.length;
  if (n === 0) return [];

  const routeCount = Math.min(truckCount, n);
  const ordered = seededOrder(items.slice());

  const routes: Package[][] = Array.from({ length: routeCount }, () => []);
  for (let i = 0; i < n; i++) {
    const idx = Math.floor((i * routeCount) / n);
    routes[idx].push(ordered[i]);
  }
  return routes.filter((r) => r.length > 0);
}

function nearestNeighborOrder(route: Package[]): Package[] {
  const remaining = route.slice();
  if (remaining.length <= 2) return remaining;

  let bestStart = 0;
  let bestStartDist = Infinity;
  for (let i = 0; i < remaining.length; i++) {
    const d = sqDist(remaining[i], depot);
    if (d < bestStartDist) {
      bestStartDist = d;
      bestStart = i;
    }
  }

  const ordered: Package[] = [];
  let current = remaining.splice(bestStart, 1)[0];
  ordered.push(current);

  while (remaining.length > 0) {
    let bestIdx = 0;
    let best = Infinity;
    for (let i = 0; i < remaining.length; i++) {
      const d = sqDist(current, remaining[i]);
      if (d < best) {
        best = d;
        bestIdx = i;
      }
    }
    current = remaining.splice(bestIdx, 1)[0];
    ordered.push(current);
  }

  return ordered;
}

function twoOpt(route: Package[]): Package[] {
  const n = route.length;
  if (n < 4) return route;

  let improved = true;
  let passes = 0;
  const maxPasses = n <= 80 ? 8 : n <= 250 ? 4 : 2;

  while (improved && passes < maxPasses) {
    improved = false;
    passes++;

    for (let i = 0; i < n - 2; i++) {
      const a: Point = i === 0 ? depot : route[i - 1];
      const b: Point = route[i];

      for (let k = i + 1; k < n - 1; k++) {
        const c: Point = route[k];
        const d: Point = route[k + 1];

        const current = dist(a, b) + dist(c, d);
        const swapped = dist(a, c) + dist(b, d);

        if (swapped + 1e-9 < current) {
          let left = i;
          let right = k;
          while (left < right) {
            const tmp = route[left];
            route[left] = route[right];
            route[right] = tmp;
            left++;
            right--;
          }
          improved = true;
        }
      }
    }

    const lastBefore = route[n - 2];
    const last = route[n - 1];
    if (dist(lastBefore, last) + dist(last, depot) > dist(lastBefore, depot) + dist(last, depot) + 1e-9) {
      route.splice(n - 1, 1);
      route.splice(n - 2, 0, last);
      improved = true;
    }
  }

  return route;
}

function improveEachRoute(routes: Package[][]): Package[][] {
  return routes.map((route) => twoOpt(nearestNeighborOrder(route)));
}

function bestInsertionDelta(route: Package[], pkg: Package): { delta: number; pos: number } {
  if (route.length === 0) {
    return { delta: 2 * dist(depot, pkg), pos: 0 };
  }

  let bestDelta = Infinity;
  let bestPos = 0;

  for (let pos = 0; pos <= route.length; pos++) {
    const prev: Point = pos === 0 ? depot : route[pos - 1];
    const next: Point = pos === route.length ? depot : route[pos];
    const delta = dist(prev, pkg) + dist(pkg, next) - dist(prev, next);
    if (delta < bestDelta) {
      bestDelta = delta;
      bestPos = pos;
    }
  }

  return { delta: bestDelta, pos: bestPos };
}

function rebalanceRoutes(routes: Package[][]): Package[][] {
  if (routes.length <= 1) return routes;

  let totalPackages = 0;
  for (const route of routes) totalPackages += route.length;
  const target = totalPackages / routes.length;

  const passes = totalPackages <= 300 ? 5 : totalPackages <= 1500 ? 3 : 2;

  for (let pass = 0; pass < passes; pass++) {
    let changed = false;

    const lengths = routes.map((r) => routeDistance(r));
    const order = routes
      .map((_, idx) => idx)
      .sort((a, b) => lengths[b] - lengths[a] || routes[b].length - routes[a].length);

    for (const fromIdx of order) {
      const from = routes[fromIdx];
      if (from.length <= 1) continue;
      if (from.length <= Math.floor(target)) continue;

      const fromDist = lengths[fromIdx];
      let bestMove: {
        pkgIndex: number;
        toIdx: number;
        insertPos: number;
        gain: number;
      } | null = null;

      const candidateIndices =
        from.length <= 6
          ? from.map((_, idx) => idx)
          : [0, 1, Math.floor(from.length / 2), from.length - 2, from.length - 1].filter(
              (v, idx, arr) => v >= 0 && v < from.length && arr.indexOf(v) === idx
            );

      for (const pkgIndex of candidateIndices) {
        const pkg = from[pkgIndex];
        const prev: Point = pkgIndex === 0 ? depot : from[pkgIndex - 1];
        const next: Point = pkgIndex === from.length - 1 ? depot : from[pkgIndex + 1];
        const removalGain = dist(prev, pkg) + dist(pkg, next) - dist(prev, next);

        for (let toIdx = 0; toIdx < routes.length; toIdx++) {
          if (toIdx === fromIdx) continue;
          const to = routes[toIdx];
          if (to.length > Math.ceil(target) + 1 && lengths[toIdx] >= fromDist) continue;

          const insertion = bestInsertionDelta(to, pkg);
          const gain = removalGain - insertion.delta;

          if (gain > 1e-9 && (!bestMove || gain > bestMove.gain)) {
            bestMove = {
              pkgIndex,
              toIdx,
              insertPos: insertion.pos,
              gain,
            };
          }
        }
      }

      if (bestMove) {
        const [pkg] = from.splice(bestMove.pkgIndex, 1);
        routes[bestMove.toIdx].splice(bestMove.insertPos, 0, pkg);
        lengths[fromIdx] = routeDistance(from);
        lengths[bestMove.toIdx] = routeDistance(routes[bestMove.toIdx]);
        changed = true;
      }
    }

    if (!changed) break;
  }

  return routes;
}

function solveVRP(items: Package[], truckCount: number): Array<Array<string | number>> {
  if (items.length === 0) return [];

  let routes = buildInitialRoutes(items, truckCount);
  routes = improveEachRoute(routes);
  routes = rebalanceRoutes(routes);
  routes = improveEachRoute(routes);

  return routes.map((route) => route.map((pkg) => pkg.id));
}

const result = { routes: solveVRP(packages, maxRoutes) };
process.stdout.write(JSON.stringify(result));
