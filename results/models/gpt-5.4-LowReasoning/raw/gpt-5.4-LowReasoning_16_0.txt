#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

struct Task {
    int machine = 0;
    int duration = 0;
};

struct Job {
    vector<Task> tasks;
    long long totalDuration = 0;
};

enum class Strategy {
    Trivial,
    Small,
    Large,
    Extreme
};

struct Candidate {
    long long earliestStart = 0;
    long long priorityA = 0;
    long long priorityB = 0;
    int job = 0;
    int task = 0;
};

struct CandidateCmp {
    bool operator()(const Candidate& a, const Candidate& b) const {
        if (a.earliestStart != b.earliestStart) return a.earliestStart > b.earliestStart;
        if (a.priorityA != b.priorityA) return a.priorityA > b.priorityA;
        if (a.priorityB != b.priorityB) return a.priorityB > b.priorityB;
        if (a.job != b.job) return a.job > b.job;
        return a.task > b.task;
    }
};

static Strategy chooseStrategy(int jobs, int machines, long long totalTasks) {
    if (totalTasks <= 32 && jobs <= 6 && machines <= 6) return Strategy::Trivial;
    if (totalTasks <= 5000 && jobs <= 200 && machines <= 30) return Strategy::Small;
    if (totalTasks <= 50000 && jobs <= 2000 && machines <= 80) return Strategy::Large;
    return Strategy::Extreme;
}

static void scheduleByPriorityQueue(
    const vector<Job>& jobs,
    vector<vector<long long>>& starts,
    bool favorShortTasks
) {
    const int J = static_cast<int>(jobs.size());
    int M = 0;
    for (const auto& job : jobs) {
        for (const auto& task : job.tasks) {
            M = max(M, task.machine + 1);
        }
    }

    vector<long long> machineReady(M, 0);
    vector<long long> jobReady(J, 0);
    vector<int> nextTask(J, 0);
    vector<long long> remainingWork(J, 0);

    for (int j = 0; j < J; ++j) {
        starts[j].assign(jobs[j].tasks.size(), 0);
        remainingWork[j] = jobs[j].totalDuration;
    }

    priority_queue<Candidate, vector<Candidate>, CandidateCmp> pq;

    auto pushCandidate = [&](int j) {
        if (nextTask[j] >= static_cast<int>(jobs[j].tasks.size())) return;
        int t = nextTask[j];
        const Task& task = jobs[j].tasks[t];
        long long est = max(jobReady[j], machineReady[task.machine]);
        long long pA = 0;
        long long pB = 0;
        if (favorShortTasks) {
            pA = task.duration;
            pB = -remainingWork[j];
        } else {
            pA = -remainingWork[j];
            pB = -task.duration;
        }
        pq.push(Candidate{est, pA, pB, j, t});
    };

    for (int j = 0; j < J; ++j) {
        pushCandidate(j);
    }

    long long scheduled = 0;
    long long totalTasks = 0;
    for (const auto& job : jobs) totalTasks += static_cast<long long>(job.tasks.size());

    while (scheduled < totalTasks) {
        Candidate cur = pq.top();
        pq.pop();

        int j = cur.job;
        int t = cur.task;
        if (t != nextTask[j]) continue;

        const Task& task = jobs[j].tasks[t];
        long long start = max(jobReady[j], machineReady[task.machine]);
        long long finish = start + task.duration;

        starts[j][t] = start;
        jobReady[j] = finish;
        machineReady[task.machine] = finish;
        remainingWork[j] -= task.duration;
        ++nextTask[j];
        ++scheduled;

        pushCandidate(j);
    }
}

static void scheduleByJobOrder(
    const vector<Job>& jobs,
    vector<vector<long long>>& starts,
    const vector<int>& order
) {
    int M = 0;
    for (const auto& job : jobs) {
        for (const auto& task : job.tasks) {
            M = max(M, task.machine + 1);
        }
    }

    vector<long long> machineReady(M, 0);
    vector<long long> jobReady(jobs.size(), 0);

    for (size_t j = 0; j < jobs.size(); ++j) {
        starts[j].assign(jobs[j].tasks.size(), 0);
    }

    for (int j : order) {
        long long ready = 0;
        for (size_t t = 0; t < jobs[j].tasks.size(); ++t) {
            const Task& task = jobs[j].tasks[t];
            long long start = max(ready, machineReady[task.machine]);
            long long finish = start + task.duration;
            starts[j][t] = start;
            ready = finish;
            machineReady[task.machine] = finish;
        }
        jobReady[j] = ready;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) {
        return 0;
    }

    vector<Job> jobs(J);
    long long totalTasks = 0;
    vector<long long> machineLoad(M, 0);

    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].tasks.resize(T);
        for (int t = 0; t < T; ++t) {
            int machine, duration;
            cin >> machine >> duration;
            jobs[j].tasks[t] = Task{machine, duration};
            jobs[j].totalDuration += duration;
            ++totalTasks;
            if (machine >= 0 && machine < M) {
                machineLoad[machine] += duration;
            }
        }
    }

    Strategy strategy = chooseStrategy(J, M, totalTasks);
    vector<vector<long long>> starts(J);

    if (strategy == Strategy::Trivial) {
        scheduleByPriorityQueue(jobs, starts, true);
    } else if (strategy == Strategy::Small) {
        scheduleByPriorityQueue(jobs, starts, false);
    } else {
        vector<int> order(J);
        iota(order.begin(), order.end(), 0);

        vector<int> firstMachinePressure(J, 0);
        for (int j = 0; j < J; ++j) {
            if (!jobs[j].tasks.empty()) {
                firstMachinePressure[j] = static_cast<int>(machineLoad[jobs[j].tasks.front().machine]);
            }
        }

        stable_sort(order.begin(), order.end(), [&](int a, int b) {
            if (strategy == Strategy::Large) {
                if (jobs[a].totalDuration != jobs[b].totalDuration) {
                    return jobs[a].totalDuration > jobs[b].totalDuration;
                }
                if (jobs[a].tasks.size() != jobs[b].tasks.size()) {
                    return jobs[a].tasks.size() > jobs[b].tasks.size();
                }
            } else {
                if (firstMachinePressure[a] != firstMachinePressure[b]) {
                    return firstMachinePressure[a] > firstMachinePressure[b];
                }
                if (jobs[a].tasks.size() != jobs[b].tasks.size()) {
                    return jobs[a].tasks.size() > jobs[b].tasks.size();
                }
                if (jobs[a].totalDuration != jobs[b].totalDuration) {
                    return jobs[a].totalDuration > jobs[b].totalDuration;
                }
            }
            return a < b;
        });

        scheduleByJobOrder(jobs, starts, order);
    }

    for (int j = 0; j < J; ++j) {
        for (size_t t = 0; t < jobs[j].tasks.size(); ++t) {
            if (t) cout << ' ';
            cout << starts[j][t] << ' ' << jobs[j].tasks[t].machine;
        }
        cout << '\n';
    }

    return 0;
}
