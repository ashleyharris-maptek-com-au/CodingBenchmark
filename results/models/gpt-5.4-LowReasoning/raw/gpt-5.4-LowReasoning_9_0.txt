using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());
        int n = reader.NextInt();
        int stockLength = reader.NextInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int length = reader.NextInt();
            if (length > stockLength)
            {
                throw new InvalidOperationException("Cut length exceeds stock length.");
            }

            cuts[i] = new Cut(length, i);
        }

        Array.Sort(cuts, static (a, b) =>
        {
            int cmp = b.Length.CompareTo(a.Length);
            return cmp != 0 ? cmp : a.Index.CompareTo(b.Index);
        });

        var remainingByStock = new List<int>(Math.Max(4, n / 4));
        var assignedCuts = new List<List<int>>(Math.Max(4, n / 4));
        var bins = new RemainingCapacityIndex();

        for (int i = 0; i < cuts.Length; i++)
        {
            int cutLength = cuts[i].Length;
            int originalIndex = cuts[i].Index;

            int stockId = bins.TakeFirstFit(cutLength);
            if (stockId < 0)
            {
                stockId = remainingByStock.Count;
                remainingByStock.Add(stockLength);
                assignedCuts.Add(new List<int>(4));
            }

            int remaining = remainingByStock[stockId] - cutLength;
            remainingByStock[stockId] = remaining;
            assignedCuts[stockId].Add(originalIndex);

            if (remaining > 0)
            {
                bins.Add(remaining, stockId);
            }
        }

        var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        writer.WriteLine(assignedCuts.Count);
        for (int i = 0; i < assignedCuts.Count; i++)
        {
            List<int> stockCuts = assignedCuts[i];
            if (stockCuts.Count == 0)
            {
                writer.WriteLine();
                continue;
            }

            writer.Write(stockCuts[0]);
            for (int j = 1; j < stockCuts.Count; j++)
            {
                writer.Write(' ');
                writer.Write(stockCuts[j]);
            }
            writer.WriteLine();
        }
        writer.Flush();
    }

    private readonly struct Cut
    {
        public readonly int Length;
        public readonly int Index;

        public Cut(int length, int index)
        {
            Length = length;
            Index = index;
        }
    }

    private sealed class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _position;

        public FastReader(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }
    }

    private sealed class RemainingCapacityIndex
    {
        private sealed class Node
        {
            public int Key;
            public int Priority;
            public int StockId;
            public Node? Left;
            public Node? Right;

            public Node(int key, int priority, int stockId)
            {
                Key = key;
                Priority = priority;
                StockId = stockId;
            }
        }

        private Node? _root;
        private uint _rng = 2463534242u;

        public void Add(int remainingCapacity, int stockId)
        {
            _root = Insert(_root, new Node(remainingCapacity, unchecked((int)NextPriority()), stockId));
        }

        public int TakeFirstFit(int required)
        {
            if (_root == null)
            {
                return -1;
            }

            Node? target = null;
            Node? current = _root;
            while (current != null)
            {
                if (current.Key >= required)
                {
                    target = current;
                    current = current.Left;
                }
                else
                {
                    current = current.Right;
                }
            }

            if (target == null)
            {
                return -1;
            }

            int stockId = target.StockId;
            _root = Remove(_root, target.Key, stockId);
            return stockId;
        }

        private uint NextPriority()
        {
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return x;
        }

        private static Node Insert(Node? root, Node node)
        {
            if (root == null)
            {
                return node;
            }

            if (node.Key < root.Key || (node.Key == root.Key && node.StockId < root.StockId))
            {
                root.Left = Insert(root.Left, node);
                if (root.Left != null && root.Left.Priority > root.Priority)
                {
                    root = RotateRight(root);
                }
            }
            else
            {
                root.Right = Insert(root.Right, node);
                if (root.Right != null && root.Right.Priority > root.Priority)
                {
                    root = RotateLeft(root);
                }
            }

            return root;
        }

        private static Node? Remove(Node? root, int key, int stockId)
        {
            if (root == null)
            {
                return null;
            }

            if (key < root.Key || (key == root.Key && stockId < root.StockId))
            {
                root.Left = Remove(root.Left, key, stockId);
                return root;
            }

            if (key > root.Key || (key == root.Key && stockId > root.StockId))
            {
                root.Right = Remove(root.Right, key, stockId);
                return root;
            }

            return Merge(root.Left, root.Right);
        }

        private static Node? Merge(Node? left, Node? right)
        {
            if (left == null) return right;
            if (right == null) return left;

            if (left.Priority > right.Priority)
            {
                left.Right = Merge(left.Right, right);
                return left;
            }

            right.Left = Merge(left, right.Left);
            return right;
        }

        private static Node RotateRight(Node root)
        {
            Node pivot = root.Left!;
            root.Left = pivot.Right;
            pivot.Right = root;
            return pivot;
        }

        private static Node RotateLeft(Node root)
        {
            Node pivot = root.Right!;
            root.Right = pivot.Left;
            pivot.Left = root;
            return pivot;
        }
    }
}
