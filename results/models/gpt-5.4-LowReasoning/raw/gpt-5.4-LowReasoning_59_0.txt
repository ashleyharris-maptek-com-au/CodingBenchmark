const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = Array.isArray(data.probability) ? data.probability : [];
const drones = Array.isArray(data.drones) ? data.drones : [];
const maxStepsPerDrone = Math.max(0, data.maxStepsPerDrone | 0);

const height = grid.length;
const width = height > 0 && Array.isArray(grid[0]) ? grid[0].length : 0;
const cellCount = width * height;

function idx(x, y) {
  return y * width + x;
}

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < width && y < height;
}

const values = new Float64Array(cellCount);
const blocked = new Uint8Array(cellCount);

for (let y = 0; y < height; y++) {
  const row = Array.isArray(grid[y]) ? grid[y] : [];
  for (let x = 0; x < width; x++) {
    const v = Number(row[x]);
    const id = idx(x, y);
    if (!Number.isFinite(v) || v < 0) {
      blocked[id] = 1;
      values[id] = 0;
    } else {
      values[id] = v > 0 ? v : 0;
    }
  }
}

const staticPotential = new Float64Array(cellCount);
const dirs = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const id = idx(x, y);
    if (blocked[id]) continue;
    let score = values[id];
    for (let dy = -2; dy <= 2; dy++) {
      const ny = y + dy;
      if (ny < 0 || ny >= height) continue;
      for (let dx = -2; dx <= 2; dx++) {
        const nx = x + dx;
        if (nx < 0 || nx >= width) continue;
        const md = Math.abs(dx) + Math.abs(dy);
        if (md === 0 || md > 2) continue;
        const nid = idx(nx, ny);
        if (blocked[nid]) continue;
        const w = md === 1 ? 0.35 : 0.12;
        score += values[nid] * w;
      }
    }
    staticPotential[id] = score;
  }
}

const covered = new Uint8Array(cellCount);

const dist = new Int32Array(cellCount);
const parent = new Int32Array(cellCount);
const seen = new Int32Array(cellCount);
let bfsMark = 1;

const q = new Int32Array(Math.max(1, cellCount));

function localUncoveredBonus(cell) {
  const x = cell % width;
  const y = (cell / width) | 0;
  let bonus = 0;
  for (let dy = -2; dy <= 2; dy++) {
    const ny = y + dy;
    if (ny < 0 || ny >= height) continue;
    for (let dx = -2; dx <= 2; dx++) {
      const nx = x + dx;
      if (nx < 0 || nx >= width) continue;
      const md = Math.abs(dx) + Math.abs(dy);
      if (md === 0 || md > 2) continue;
      const nid = idx(nx, ny);
      if (blocked[nid] || covered[nid]) continue;
      bonus += values[nid] * (md === 1 ? 0.30 : 0.10);
    }
  }
  return bonus;
}

function chooseTarget(start, remainingSteps) {
  bfsMark++;
  let head = 0;
  let tail = 0;
  q[tail++] = start;
  seen[start] = bfsMark;
  dist[start] = 0;
  parent[start] = -1;

  let bestCell = start;
  let bestScore = covered[start] ? -1e100 : values[start] + staticPotential[start] * 0.4;

  while (head < tail) {
    const cur = q[head++];
    const d = dist[cur];
    const remAfter = remainingSteps - d;

    let score = 0;
    if (!covered[cur]) score += values[cur] * 2.4;
    score += staticPotential[cur] * 0.45;
    score += localUncoveredBonus(cur);
    score += remAfter * 0.015;
    score /= Math.pow(d + 1, 0.92);

    if (
      score > bestScore + 1e-9 ||
      (Math.abs(score - bestScore) <= 1e-9 &&
        (!covered[cur] && covered[bestCell]))
    ) {
      bestScore = score;
      bestCell = cur;
    }

    if (d >= remainingSteps) continue;

    const x = cur % width;
    const y = (cur / width) | 0;
    for (let i = 0; i < 4; i++) {
      const nx = x + dirs[i][0];
      const ny = y + dirs[i][1];
      if (!inBounds(nx, ny)) continue;
      const nid = idx(nx, ny);
      if (blocked[nid] || seen[nid] === bfsMark) continue;
      seen[nid] = bfsMark;
      dist[nid] = d + 1;
      parent[nid] = cur;
      q[tail++] = nid;
    }
  }

  return bestScore > 0 ? bestCell : start;
}

function reconstructPath(target) {
  const rev = [];
  let cur = target;
  while (cur !== -1) {
    rev.push(cur);
    cur = parent[cur];
  }
  rev.reverse();
  return rev;
}

function greedyFinish(current, remainingSteps, pathOut) {
  let cur = current;
  for (let step = 0; step < remainingSteps; step++) {
    const x = cur % width;
    const y = (cur / width) | 0;
    let best = cur;
    let bestScore = -1e100;

    for (let i = -1; i < 4; i++) {
      const nx = i === -1 ? x : x + dirs[i][0];
      const ny = i === -1 ? y : y + dirs[i][1];
      if (!inBounds(nx, ny)) continue;
      const nid = idx(nx, ny);
      if (blocked[nid]) continue;

      let score = 0;
      if (!covered[nid]) score += values[nid] * 3.0;
      score += staticPotential[nid] * 0.30;
      score += localUncoveredBonus(nid) * 0.75;
      if (nid === cur) score -= 0.02;

      if (score > bestScore) {
        bestScore = score;
        best = nid;
      }
    }

    cur = best;
    covered[cur] = 1;
    pathOut.push([cur % width, (cur / width) | 0]);
  }
}

const paths = [];

for (let di = 0; di < drones.length; di++) {
  const drone = drones[di] || {};
  let x = drone.x | 0;
  let y = drone.y | 0;

  if (!inBounds(x, y) || blocked[idx(x, y)]) {
    let found = false;
    for (let yy = 0; yy < height && !found; yy++) {
      for (let xx = 0; xx < width; xx++) {
        const id = idx(xx, yy);
        if (!blocked[id]) {
          x = xx;
          y = yy;
          found = true;
          break;
        }
      }
    }
    if (!found) {
      paths.push([]);
      continue;
    }
  }

  let current = idx(x, y);
  const path = [[x, y]];
  covered[current] = 1;
  let remaining = maxStepsPerDrone;

  while (remaining > 0) {
    const target = chooseTarget(current, remaining);
    if (target === current) break;

    const segment = reconstructPath(target);
    if (segment.length <= 1) break;

    for (let i = 1; i < segment.length; i++) {
      const cell = segment[i];
      current = cell;
      covered[cell] = 1;
      path.push([cell % width, (cell / width) | 0]);
      remaining--;
      if (remaining === 0) break;
    }
  }

  if (remaining > 0) {
    greedyFinish(current, remaining, path);
  }

  paths.push(path);
}

process.stdout.write(JSON.stringify({ paths }));
