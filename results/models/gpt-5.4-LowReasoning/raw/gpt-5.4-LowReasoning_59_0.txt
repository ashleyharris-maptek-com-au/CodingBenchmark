const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const grid = data.probability || [];
const drones = Array.isArray(data.drones) ? data.drones : [];
const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const cellCount = width * height;

function indexOf(x, y) {
  return y * width + x;
}

function coordOf(index) {
  return [index % width, Math.floor(index / width)];
}

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < width && y < height;
}

const walkable = new Uint8Array(cellCount);
const valueAt = new Float64Array(cellCount);
const positiveCells = [];

for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const v = grid[y][x];
    const idx = indexOf(x, y);
    if (v !== -1) {
      walkable[idx] = 1;
      if (v > 0) {
        valueAt[idx] = v;
        positiveCells.push(idx);
      }
    }
  }
}

const neighbors = Array.from({ length: cellCount }, () => []);
for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const idx = indexOf(x, y);
    if (!walkable[idx]) continue;
    if (x > 0) {
      const n = idx - 1;
      if (walkable[n]) neighbors[idx].push(n);
    }
    if (x + 1 < width) {
      const n = idx + 1;
      if (walkable[n]) neighbors[idx].push(n);
    }
    if (y > 0) {
      const n = idx - width;
      if (walkable[n]) neighbors[idx].push(n);
    }
    if (y + 1 < height) {
      const n = idx + width;
      if (walkable[n]) neighbors[idx].push(n);
    }
  }
}

const localPotential = new Float64Array(cellCount);
const radius = 2;
for (const idx of positiveCells) {
  const x = idx % width;
  const y = Math.floor(idx / width);
  let sum = valueAt[idx];
  for (let dy = -radius; dy <= radius; dy++) {
    for (let dx = -radius; dx <= radius; dx++) {
      const dist = Math.abs(dx) + Math.abs(dy);
      if (dist === 0 || dist > radius) continue;
      const nx = x + dx;
      const ny = y + dy;
      if (!inBounds(nx, ny)) continue;
      const nidx = indexOf(nx, ny);
      if (!walkable[nidx] || valueAt[nidx] <= 0) continue;
      sum += valueAt[nidx] / (dist + 1);
    }
  }
  localPotential[idx] = sum;
}

const covered = new Uint8Array(cellCount);

function markCovered(idx) {
  if (idx >= 0 && idx < cellCount && valueAt[idx] > 0) covered[idx] = 1;
}

function bfs(startIdx, maxDist) {
  const dist = new Int32Array(cellCount);
  dist.fill(-1);
  const parent = new Int32Array(cellCount);
  parent.fill(-1);
  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;
  dist[startIdx] = 0;
  queue[tail++] = startIdx;

  while (head < tail) {
    const cur = queue[head++];
    const d = dist[cur];
    if (d >= maxDist) continue;
    const adj = neighbors[cur];
    for (let i = 0; i < adj.length; i++) {
      const next = adj[i];
      if (dist[next] !== -1) continue;
      dist[next] = d + 1;
      parent[next] = cur;
      queue[tail++] = next;
    }
  }
  return { dist, parent };
}

function reconstructNextStep(targetIdx, parent, startIdx) {
  let cur = targetIdx;
  let prev = -1;
  while (cur !== -1 && cur !== startIdx) {
    prev = cur;
    cur = parent[cur];
  }
  return prev === -1 ? startIdx : prev;
}

function chooseTarget(startIdx, remainingSteps) {
  const { dist, parent } = bfs(startIdx, remainingSteps);

  let bestIdx = startIdx;
  let bestScore = covered[startIdx] || valueAt[startIdx] <= 0 ? -Infinity : valueAt[startIdx] + 0.35 * localPotential[startIdx];

  for (const idx of positiveCells) {
    if (covered[idx]) continue;
    const d = dist[idx];
    if (d < 0 || d > remainingSteps) continue;

    const immediate = valueAt[idx];
    const cluster = localPotential[idx];
    const score =
      immediate / Math.pow(d + 1, 1.15) +
      0.35 * cluster / Math.pow(d + 1, 1.35) +
      (d === 0 ? 0.5 : 0);

    if (score > bestScore) {
      bestScore = score;
      bestIdx = idx;
    }
  }

  return { targetIdx: bestIdx, dist, parent };
}

function chooseExplorationStep(startIdx, remainingSteps) {
  if (remainingSteps <= 0) return startIdx;
  const { dist, parent } = bfs(startIdx, Math.min(remainingSteps, 12));

  let bestIdx = startIdx;
  let bestScore = -Infinity;

  for (let idx = 0; idx < cellCount; idx++) {
    const d = dist[idx];
    if (d <= 0) continue;
    let score = 0;

    if (!covered[idx] && valueAt[idx] > 0) score += valueAt[idx] * 2.0;

    const x = idx % width;
    const y = Math.floor(idx / width);
    for (let dy = -2; dy <= 2; dy++) {
      for (let dx = -2; dx <= 2; dx++) {
        const md = Math.abs(dx) + Math.abs(dy);
        if (md === 0 || md > 2) continue;
        const nx = x + dx;
        const ny = y + dy;
        if (!inBounds(nx, ny)) continue;
        const nidx = indexOf(nx, ny);
        if (!covered[nidx] && valueAt[nidx] > 0) {
          score += valueAt[nidx] / (md + 1);
        }
      }
    }

    score /= Math.pow(d + 1, 1.2);
    if (score > bestScore) {
      bestScore = score;
      bestIdx = idx;
    }
  }

  return reconstructNextStep(bestIdx, parent, startIdx);
}

const states = drones.map((drone, i) => {
  const x = drone.x | 0;
  const y = drone.y | 0;
  const valid = inBounds(x, y) && walkable[indexOf(x, y)];
  const idx = valid ? indexOf(x, y) : -1;
  if (idx !== -1) markCovered(idx);
  return {
    id: drone.id ?? i,
    idx,
    stepsLeft: maxSteps,
    path: idx === -1 ? [] : [[x, y]],
    route: []
  };
});

for (let round = 0; round < maxSteps; round++) {
  let anyMoved = false;

  for (const state of states) {
    if (state.idx === -1 || state.stepsLeft <= 0) continue;

    if (state.route.length === 0) {
      const plan = chooseTarget(state.idx, state.stepsLeft);
      if (plan.targetIdx !== state.idx) {
        const next = reconstructNextStep(plan.targetIdx, plan.parent, state.idx);
        if (next !== -1 && next !== state.idx) {
          state.route.push(next);
        }
      } else {
        const next = chooseExplorationStep(state.idx, state.stepsLeft);
        if (next !== state.idx) state.route.push(next);
      }
    }

    let nextIdx = state.idx;
    if (state.route.length > 0) {
      nextIdx = state.route.shift();
    }

    state.idx = nextIdx;
    const [nx, ny] = coordOf(nextIdx);
    state.path.push([nx, ny]);
    markCovered(nextIdx);
    state.stepsLeft--;
    anyMoved = true;
  }

  if (!anyMoved) break;
}

const result = {
  paths: states.map((state) => state.path)
};

process.stdout.write(JSON.stringify(result));
