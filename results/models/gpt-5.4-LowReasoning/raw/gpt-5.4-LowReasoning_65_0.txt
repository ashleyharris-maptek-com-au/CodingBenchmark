declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type TaskInput = {
  id: string | number;
  duration: number;
  deps?: Array<string | number>;
  deadline?: number;
  weight?: number;
};

type Task = {
  index: number;
  id: string | number;
  duration: number;
  deadline: number;
  weight: number;
  deps: number[];
  outs: number[];
  indegree: number;
  remainingDeps: number;
  readyTime: number;
  criticality: number;
  descWeight: number;
};

class BinaryHeap<T> {
  private readonly data: T[] = [];
  private readonly compare: (a: T, b: T) => number;

  constructor(compare: (a: T, b: T) => number) {
    this.compare = compare;
  }

  size(): number {
    return this.data.length;
  }

  peek(): T | undefined {
    return this.data[0];
  }

  push(value: T): void {
    const arr = this.data;
    arr.push(value);
    let i = arr.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.compare(arr[i], arr[p]) >= 0) break;
      [arr[i], arr[p]] = [arr[p], arr[i]];
      i = p;
    }
  }

  pop(): T | undefined {
    const arr = this.data;
    if (arr.length === 0) return undefined;
    const top = arr[0];
    const last = arr.pop()!;
    if (arr.length > 0) {
      arr[0] = last;
      let i = 0;
      for (;;) {
        let best = i;
        const l = i * 2 + 1;
        const r = l + 1;
        if (l < arr.length && this.compare(arr[l], arr[best]) < 0) best = l;
        if (r < arr.length && this.compare(arr[r], arr[best]) < 0) best = r;
        if (best === i) break;
        [arr[i], arr[best]] = [arr[best], arr[i]];
        i = best;
      }
    }
    return top;
  }
}

const machineCountRaw = Number(data.machineCount);
const machineCount = Number.isFinite(machineCountRaw) && machineCountRaw > 0 ? Math.floor(machineCountRaw) : 1;
const taskInputs: TaskInput[] = Array.isArray(data.tasks) ? data.tasks : [];
const n = taskInputs.length;

const idToIndex = new Map<string | number, number>();
for (let i = 0; i < n; i++) {
  idToIndex.set(taskInputs[i].id, i);
}

const tasks: Task[] = new Array(n);
for (let i = 0; i < n; i++) {
  const src = taskInputs[i];
  tasks[i] = {
    index: i,
    id: src.id,
    duration: Math.max(0, Math.floor(Number(src.duration) || 0)),
    deadline: Number.isFinite(Number(src.deadline)) ? Number(src.deadline) : Number.POSITIVE_INFINITY,
    weight: Math.max(0, Number(src.weight) || 0),
    deps: [],
    outs: [],
    indegree: 0,
    remainingDeps: 0,
    readyTime: 0,
    criticality: 0,
    descWeight: 0,
  };
}

for (let i = 0; i < n; i++) {
  const srcDeps = Array.isArray(taskInputs[i].deps) ? taskInputs[i].deps! : [];
  const seen = new Set<number>();
  const deps: number[] = [];
  for (const depId of srcDeps) {
    const depIndex = idToIndex.get(depId);
    if (depIndex === undefined || depIndex === i || seen.has(depIndex)) continue;
    seen.add(depIndex);
    deps.push(depIndex);
  }
  tasks[i].deps = deps;
  tasks[i].indegree = deps.length;
  tasks[i].remainingDeps = deps.length;
  for (const depIndex of deps) {
    tasks[depIndex].outs.push(i);
  }
}

const topo: number[] = [];
const q: number[] = [];
for (let i = 0; i < n; i++) {
  if (tasks[i].indegree === 0) q.push(i);
}
for (let head = 0; head < q.length; head++) {
  const u = q[head];
  topo.push(u);
  for (const v of tasks[u].outs) {
    tasks[v].indegree--;
    if (tasks[v].indegree === 0) q.push(v);
  }
}

if (topo.length !== n) {
  const fallbackOrder = taskInputs.map((t) => t.id);
  process.stdout.write(JSON.stringify({ order: fallbackOrder }));
} else {
  for (let p = n - 1; p >= 0; p--) {
    const u = topo[p];
    let bestTail = 0;
    let weightSum = tasks[u].weight;
    for (const v of tasks[u].outs) {
      if (tasks[v].criticality > bestTail) bestTail = tasks[v].criticality;
      weightSum += tasks[v].descWeight;
    }
    tasks[u].criticality = tasks[u].duration + bestTail;
    tasks[u].descWeight = weightSum;
  }

  type ReadyEntry = { task: number; seq: number };
  type RunningEntry = { finish: number; machine: number; task: number };
  type MachineEntry = { available: number; machine: number };

  let readySeq = 0;
  const readyHeap = new BinaryHeap<ReadyEntry>((a, b) => {
    const ta = tasks[a.task];
    const tb = tasks[b.task];

    const slackA = ta.deadline - ta.duration;
    const slackB = tb.deadline - tb.duration;
    if (slackA !== slackB) return slackA - slackB;

    const densA = ta.weight / Math.max(1, ta.duration);
    const densB = tb.weight / Math.max(1, tb.duration);
    if (densA !== densB) return densB - densA;

    if (ta.criticality !== tb.criticality) return tb.criticality - ta.criticality;
    if (ta.descWeight !== tb.descWeight) return tb.descWeight - ta.descWeight;
    if (ta.duration !== tb.duration) return tb.duration - ta.duration;
    if (a.seq !== b.seq) return a.seq - b.seq;
    return ta.index - tb.index;
  });

  const runningHeap = new BinaryHeap<RunningEntry>((a, b) => {
    if (a.finish !== b.finish) return a.finish - b.finish;
    return a.machine - b.machine;
  });

  const machineHeap = new BinaryHeap<MachineEntry>((a, b) => {
    if (a.available !== b.available) return a.available - b.available;
    return a.machine - b.machine;
  });

  for (let m = 0; m < machineCount; m++) {
    machineHeap.push({ available: 0, machine: m });
  }

  for (let i = 0; i < n; i++) {
    tasks[i].remainingDeps = tasks[i].deps.length;
    tasks[i].readyTime = 0;
    if (tasks[i].remainingDeps === 0) {
      readyHeap.push({ task: i, seq: readySeq++ });
    }
  }

  const schedule: Array<{ id: string | number; start: number; machine: number }> = [];
  let currentTime = 0;
  let completed = 0;

  const releaseFinishedTasks = (upToTime: number): void => {
    while (runningHeap.size() > 0) {
      const next = runningHeap.peek()!;
      if (next.finish > upToTime) break;
      runningHeap.pop();
      machineHeap.push({ available: next.finish, machine: next.machine });
      const finishedTask = tasks[next.task];
      completed++;
      for (const child of finishedTask.outs) {
        const depTask = tasks[child];
        if (next.finish > depTask.readyTime) depTask.readyTime = next.finish;
        depTask.remainingDeps--;
        if (depTask.remainingDeps === 0) {
          readyHeap.push({ task: child, seq: readySeq++ });
        }
      }
    }
  };

  while (completed < n) {
    releaseFinishedTasks(currentTime);

    let assigned = false;
    while (readyHeap.size() > 0 && machineHeap.size() > 0) {
      const machine = machineHeap.peek()!;
      const candidate = readyHeap.peek()!;
      const task = tasks[candidate.task];
      const start = Math.max(machine.available, task.readyTime, currentTime);
      if (start > currentTime) break;

      readyHeap.pop();
      machineHeap.pop();

      schedule.push({ id: task.id, start, machine: machine.machine });
      runningHeap.push({ finish: start + task.duration, machine: machine.machine, task: task.index });
      assigned = true;
    }

    if (assigned) continue;

    const nextFinish = runningHeap.peek()?.finish ?? Number.POSITIVE_INFINITY;
    const nextMachine = machineHeap.peek()?.available ?? Number.POSITIVE_INFINITY;
    const nextReadyTask = readyHeap.peek();
    const nextReadyTime = nextReadyTask ? Math.max(tasks[nextReadyTask.task].readyTime, nextMachine) : Number.POSITIVE_INFINITY;
    const nextTime = Math.min(nextFinish, nextReadyTime, nextMachine);

    if (!Number.isFinite(nextTime)) break;
    if (nextTime <= currentTime) {
      currentTime++;
    } else {
      currentTime = nextTime;
    }
  }

  if (schedule.length !== n) {
    const order = topo.map((i) => tasks[i].id);
    process.stdout.write(JSON.stringify({ order }));
  } else {
    process.stdout.write(JSON.stringify({ schedule }));
  }
}
