use std::cmp::Ordering;
use std::collections::HashMap;
use std::f64::consts::{FRAC_PI_2, PI};
use std::hash::{Hash, Hasher};
use std::io::{self, Read};

const EPS: f64 = 1.0e-9;

#[derive(Clone, Copy, Debug, Default)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn cross(self, other: Self) -> f64 {
        self.x * other.y - self.y * other.x
    }

    fn norm2(self) -> f64 {
        self.dot(self)
    }

    fn norm(self) -> f64 {
        self.norm2().sqrt()
    }

    fn rotate(self, ang: f64) -> Self {
        let (s, c) = ang.sin_cos();
        Pt::new(c * self.x - s * self.y, s * self.x + c * self.y)
    }
}

impl std::ops::Add for Pt {
    type Output = Pt;
    fn add(self, rhs: Pt) -> Pt {
        Pt::new(self.x + rhs.x, self.y + rhs.y)
    }
}

impl std::ops::Sub for Pt {
    type Output = Pt;
    fn sub(self, rhs: Pt) -> Pt {
        Pt::new(self.x - rhs.x, self.y - rhs.y)
    }
}

impl std::ops::Mul<f64> for Pt {
    type Output = Pt;
    fn mul(self, rhs: f64) -> Pt {
        Pt::new(self.x * rhs, self.y * rhs)
    }
}

impl std::ops::Div<f64> for Pt {
    type Output = Pt;
    fn div(self, rhs: f64) -> Pt {
        Pt::new(self.x / rhs, self.y / rhs)
    }
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    min: Pt,
    max: Pt,
}

impl BBox {
    fn empty() -> Self {
        Self {
            min: Pt::new(f64::INFINITY, f64::INFINITY),
            max: Pt::new(f64::NEG_INFINITY, f64::NEG_INFINITY),
        }
    }

    fn include(&mut self, p: Pt) {
        self.min.x = self.min.x.min(p.x);
        self.min.y = self.min.y.min(p.y);
        self.max.x = self.max.x.max(p.x);
        self.max.y = self.max.y.max(p.y);
    }

    fn width(&self) -> f64 {
        self.max.x - self.min.x
    }

    fn height(&self) -> f64 {
        self.max.y - self.min.y
    }

    fn area(&self) -> f64 {
        self.width().max(0.0) * self.height().max(0.0)
    }

    fn center(&self) -> Pt {
        (self.min + self.max) * 0.5
    }

    fn intersects(&self, other: &Self) -> bool {
        !(self.max.x < other.min.x - EPS
            || other.max.x < self.min.x - EPS
            || self.max.y < other.min.y - EPS
            || other.max.y < self.min.y - EPS)
    }
}

#[derive(Clone)]
struct PolyData {
    verts: Vec<Pt>,
    bbox: BBox,
    area: f64,
    centroid: Pt,
    radius: f64,
    edge_angles: Vec<f64>,
}

#[derive(Clone)]
struct Piece {
    id: usize,
    poly: PolyData,
}

#[derive(Clone)]
struct Placement {
    piece_id: usize,
    stock_id: usize,
    trans: Pt,
    angle: f64,
    verts: Vec<Pt>,
    bbox: BBox,
    center: Pt,
    radius: f64,
}

#[derive(Clone, Copy, Eq)]
struct Cell(i64, i64);

impl PartialEq for Cell {
    fn eq(&self, other: &Self) -> bool {
        self.0 == other.0 && self.1 == other.1
    }
}

impl Hash for Cell {
    fn hash<H: Hasher>(&self, state: &mut H) {
        self.0.hash(state);
        self.1.hash(state);
    }
}

struct SpatialHash {
    cell: f64,
    map: HashMap<Cell, Vec<usize>>,
}

impl SpatialHash {
    fn new(cell: f64) -> Self {
        Self {
            cell: cell.max(1.0),
            map: HashMap::new(),
        }
    }

    fn coord(&self, v: f64) -> i64 {
        (v / self.cell).floor() as i64
    }

    fn cells_for_bbox(&self, bb: &BBox) -> Vec<Cell> {
        let x0 = self.coord(bb.min.x);
        let x1 = self.coord(bb.max.x);
        let y0 = self.coord(bb.min.y);
        let y1 = self.coord(bb.max.y);
        let mut out = Vec::new();
        for x in x0..=x1 {
            for y in y0..=y1 {
                out.push(Cell(x, y));
            }
        }
        out
    }

    fn insert(&mut self, idx: usize, bb: &BBox) {
        for c in self.cells_for_bbox(bb) {
            self.map.entry(c).or_default().push(idx);
        }
    }

    fn query(&self, bb: &BBox) -> Vec<usize> {
        let mut vals = Vec::new();
        for c in self.cells_for_bbox(bb) {
            if let Some(v) = self.map.get(&c) {
                vals.extend_from_slice(v);
            }
        }
        vals.sort_unstable();
        vals.dedup();
        vals
    }
}

struct StockInstance {
    placements: Vec<Placement>,
    hash: SpatialHash,
    used_area: f64,
    anchors: Vec<Pt>,
}

struct SearchCfg {
    max_rot: usize,
    max_stock_probe: usize,
    seed_grid: usize,
    local_rings: usize,
    local_dirs: usize,
    random_jitter: usize,
}

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b - a).cross(c - a)
}

fn signed_area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        s += poly[i].cross(poly[(i + 1) % poly.len()]);
    }
    0.5 * s
}

fn polygon_centroid(poly: &[Pt]) -> Pt {
    let mut c = Pt::new(0.0, 0.0);
    let mut acc = 0.0;
    for i in 0..poly.len() {
        let p = poly[i];
        let q = poly[(i + 1) % poly.len()];
        let cr = p.cross(q);
        acc += cr;
        c = c + (p + q) * cr;
    }
    if acc.abs() < EPS {
        let mut avg = Pt::new(0.0, 0.0);
        for &p in poly {
            avg = avg + p;
        }
        return avg / poly.len() as f64;
    }
    c / (3.0 * acc)
}

fn normalize_polygon(mut poly: Vec<Pt>) -> Vec<Pt> {
    if signed_area(&poly) < 0.0 {
        poly.reverse();
    }
    poly
}

fn build_poly_data(poly: Vec<Pt>) -> PolyData {
    let verts = normalize_polygon(poly);
    let mut bbox = BBox::empty();
    for &p in &verts {
        bbox.include(p);
    }
    let centroid = polygon_centroid(&verts);
    let area = signed_area(&verts).abs();
    let mut radius = 0.0;
    let mut edge_angles = Vec::new();
    for i in 0..verts.len() {
        let p = verts[i];
        let q = verts[(i + 1) % verts.len()];
        radius = radius.max((p - centroid).norm());
        let d = q - p;
        if d.norm2() > EPS {
            edge_angles.push(d.y.atan2(d.x));
        }
    }
    edge_angles.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    edge_angles.dedup_by(|a, b| (*a - *b).abs() < 1.0e-8);
    PolyData {
        verts,
        bbox,
        area,
        centroid,
        radius,
        edge_angles,
    }
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    cross(a, b, p).abs() <= EPS
        && p.x >= a.x.min(b.x) - EPS
        && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS
        && p.y <= a.y.max(b.y) + EPS
}

fn segments_proper_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let ab1 = cross(a, b, c);
    let ab2 = cross(a, b, d);
    let cd1 = cross(c, d, a);
    let cd2 = cross(c, d, b);
    if ab1.abs() <= EPS && on_segment(a, b, c) {
        return false;
    }
    if ab2.abs() <= EPS && on_segment(a, b, d) {
        return false;
    }
    if cd1.abs() <= EPS && on_segment(c, d, a) {
        return false;
    }
    if cd2.abs() <= EPS && on_segment(c, d, b) {
        return false;
    }
    (ab1 > EPS && ab2 < -EPS || ab1 < -EPS && ab2 > EPS)
        && (cd1 > EPS && cd2 < -EPS || cd1 < -EPS && cd2 > EPS)
}

fn point_in_polygon(poly: &[Pt], p: Pt) -> i32 {
    let mut inside = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        if on_segment(a, b, p) {
            return 0;
        }
        let intersects = ((a.y > p.y) != (b.y > p.y))
            && (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x);
        if intersects {
            inside = !inside;
        }
    }
    if inside { 1 } else { -1 }
}

fn polygons_overlap(a: &[Pt], b: &[Pt]) -> bool {
    let bba = bbox_of(a);
    let bbb = bbox_of(b);
    if !bba.intersects(&bbb) {
        return false;
    }
    for i in 0..a.len() {
        let p = a[i];
        let q = a[(i + 1) % a.len()];
        for j in 0..b.len() {
            let r = b[j];
            let s = b[(j + 1) % b.len()];
            if segments_proper_intersect(p, q, r, s) {
                return true;
            }
        }
    }
    for &p in a {
        if point_in_polygon(b, p) > 0 {
            return true;
        }
    }
    for &p in b {
        if point_in_polygon(a, p) > 0 {
            return true;
        }
    }
    false
}

fn bbox_of(poly: &[Pt]) -> BBox {
    let mut bb = BBox::empty();
    for &p in poly {
        bb.include(p);
    }
    bb
}

fn polygon_within_stock(stock: &[Pt], piece: &[Pt], stock_bb: &BBox) -> bool {
    let bb = bbox_of(piece);
    if bb.min.x < stock_bb.min.x - EPS
        || bb.max.x > stock_bb.max.x + EPS
        || bb.min.y < stock_bb.min.y - EPS
        || bb.max.y > stock_bb.max.y + EPS
    {
        return false;
    }
    for &p in piece {
        if point_in_polygon(stock, p) < 0 {
            return false;
        }
    }
    for i in 0..piece.len() {
        let a = piece[i];
        let b = piece[(i + 1) % piece.len()];
        let mid = (a + b) * 0.5;
        if point_in_polygon(stock, mid) < 0 {
            return false;
        }
        for j in 0..stock.len() {
            let c = stock[j];
            let d = stock[(j + 1) % stock.len()];
            if segments_proper_intersect(a, b, c, d) {
                return false;
            }
        }
    }
    point_in_polygon(stock, polygon_centroid(piece)) >= 0
}

fn transform(poly: &[Pt], ang: f64, trans: Pt) -> Vec<Pt> {
    poly.iter().map(|&p| p.rotate(ang) + trans).collect()
}

fn reduce_angle(mut a: f64) -> f64 {
    while a <= -PI {
        a += 2.0 * PI;
    }
    while a > PI {
        a -= 2.0 * PI;
    }
    a
}

fn dedup_angles(mut angles: Vec<f64>, max_keep: usize) -> Vec<f64> {
    for a in &mut angles {
        *a = reduce_angle(*a);
    }
    angles.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mut out = Vec::new();
    for a in angles {
        if out.iter().all(|&x| (x - a).abs() > 1.0e-7) {
            out.push(a);
            if out.len() >= max_keep {
                break;
            }
        }
    }
    out
}

fn candidate_rotations(piece: &PolyData, stock: &PolyData, cfg: &SearchCfg) -> Vec<f64> {
    let mut angs = vec![0.0, FRAC_PI_2, PI, -FRAC_PI_2];
    let mut pairs = Vec::new();
    for &sa in &stock.edge_angles {
        for &pa in &piece.edge_angles {
            pairs.push(sa - pa);
        }
    }
    pairs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    if !pairs.is_empty() {
        let stride = (pairs.len() / cfg.max_rot.max(1)).max(1);
        for i in (0..pairs.len()).step_by(stride) {
            angs.push(pairs[i]);
        }
    }
    dedup_angles(angs, cfg.max_rot.max(1))
}

fn make_search_cfg(np: usize, complexity: usize) -> SearchCfg {
    if np <= 32 && complexity <= 32 {
        SearchCfg {
            max_rot: 28,
            max_stock_probe: 64,
            seed_grid: 5,
            local_rings: 8,
            local_dirs: 16,
            random_jitter: 24,
        }
    } else if np <= 256 {
        SearchCfg {
            max_rot: 16,
            max_stock_probe: 32,
            seed_grid: 4,
            local_rings: 5,
            local_dirs: 12,
            random_jitter: 12,
        }
    } else if np <= 1500 {
        SearchCfg {
            max_rot: 8,
            max_stock_probe: 16,
            seed_grid: 3,
            local_rings: 3,
            local_dirs: 8,
            random_jitter: 4,
        }
    } else {
        SearchCfg {
            max_rot: 4,
            max_stock_probe: 8,
            seed_grid: 2,
            local_rings: 2,
            local_dirs: 6,
            random_jitter: 0,
        }
    }
}

fn halton(mut idx: usize, base: usize) -> f64 {
    let mut f = 1.0 / base as f64;
    let mut r = 0.0;
    while idx > 0 {
        r += f * (idx % base) as f64;
        idx /= base;
        f /= base as f64;
    }
    r
}

fn base_seed_points(stock: &PolyData, cfg: &SearchCfg) -> Vec<Pt> {
    let mut seeds = Vec::new();
    seeds.push(stock.centroid);
    seeds.push(stock.bbox.center());
    for &v in &stock.verts {
        seeds.push(v * 0.15 + stock.centroid * 0.85);
    }
    let gx = cfg.seed_grid.max(1);
    let gy = cfg.seed_grid.max(1);
    for ix in 0..gx {
        for iy in 0..gy {
            let tx = (ix as f64 + 0.5) / gx as f64;
            let ty = (iy as f64 + 0.5) / gy as f64;
            seeds.push(Pt::new(
                stock.bbox.min.x + tx * stock.bbox.width(),
                stock.bbox.min.y + ty * stock.bbox.height(),
            ));
        }
    }
    seeds
}

fn stock_specific_seeds(
    stock: &PolyData,
    inst: &StockInstance,
    piece_radius: f64,
    cfg: &SearchCfg,
) -> Vec<Pt> {
    let mut seeds = Vec::new();
    seeds.extend(inst.anchors.iter().copied());
    seeds.push(stock.centroid);
    for plc in &inst.placements {
        let dx = piece_radius + plc.radius;
        seeds.push(plc.center + Pt::new(dx, 0.0));
        seeds.push(plc.center + Pt::new(-dx, 0.0));
        seeds.push(plc.center + Pt::new(0.0, dx));
        seeds.push(plc.center + Pt::new(0.0, -dx));
        seeds.push(Pt::new(plc.bbox.max.x + piece_radius, plc.center.y));
        seeds.push(Pt::new(plc.center.x, plc.bbox.max.y + piece_radius));
    }
    let mut base = base_seed_points(stock, cfg);
    seeds.append(&mut base);
    seeds
}

fn make_offsets(step: f64, cfg: &SearchCfg) -> Vec<Pt> {
    let mut out = vec![Pt::new(0.0, 0.0)];
    for ring in 1..=cfg.local_rings {
        let r = step * ring as f64;
        for k in 0..cfg.local_dirs {
            let a = 2.0 * PI * (k as f64) / cfg.local_dirs as f64;
            out.push(Pt::new(r * a.cos(), r * a.sin()));
        }
    }
    out
}

fn stock_score(stock: &StockInstance, new_area: f64, stock_area: f64) -> f64 {
    let fill = (stock.used_area + new_area) / stock_area.max(1.0);
    let slack = (1.0 - fill).max(0.0);
    stock.placements.len() as f64 * 1000.0 - slack
}

fn try_place_in_stock(
    stock_poly: &PolyData,
    inst: &StockInstance,
    piece: &PolyData,
    angle: f64,
    cfg: &SearchCfg,
) -> Option<(Pt, Vec<Pt>, BBox)> {
    let rotated_center = piece.centroid.rotate(angle);
    let step = ((piece.bbox.width().max(piece.bbox.height()) + stock_poly.bbox.width().max(stock_poly.bbox.height()))
        / 60.0)
        .max(1.0);
    let offsets = make_offsets(step, cfg);
    let seeds = stock_specific_seeds(stock_poly, inst, piece.radius, cfg);
    for seed in seeds {
        for off in &offsets {
            let trans = seed + *off - rotated_center;
            let verts = transform(&piece.verts, angle, trans);
            if !polygon_within_stock(&stock_poly.verts, &verts, &stock_poly.bbox) {
                continue;
            }
            let bb = bbox_of(&verts);
            let candidates = inst.hash.query(&bb);
            let center = polygon_centroid(&verts);
            let mut ok = true;
            for idx in candidates {
                let other = &inst.placements[idx];
                let dist2 = (center - other.center).norm2();
                let lim = piece.radius + other.radius + 1.0e-7;
                if dist2 > lim * lim {
                    continue;
                }
                if polygons_overlap(&verts, &other.verts) {
                    ok = false;
                    break;
                }
            }
            if ok {
                return Some((trans, verts, bb));
            }
        }
    }
    if cfg.random_jitter > 0 {
        for i in 1..=cfg.random_jitter {
            let seed = Pt::new(
                stock_poly.bbox.min.x + halton(i, 2) * stock_poly.bbox.width(),
                stock_poly.bbox.min.y + halton(i, 3) * stock_poly.bbox.height(),
            );
            let trans = seed - rotated_center;
            let verts = transform(&piece.verts, angle, trans);
            if !polygon_within_stock(&stock_poly.verts, &verts, &stock_poly.bbox) {
                continue;
            }
            let bb = bbox_of(&verts);
            let candidates = inst.hash.query(&bb);
            let mut ok = true;
            for idx in candidates {
                if polygons_overlap(&verts, &inst.placements[idx].verts) {
                    ok = false;
                    break;
                }
            }
            if ok {
                return Some((trans, verts, bb));
            }
        }
    }
    None
}

fn create_stock_instance(stock_bb: &BBox) -> StockInstance {
    let cell = (stock_bb.width().max(stock_bb.height()) / 24.0).max(1.0);
    StockInstance {
        placements: Vec::new(),
        hash: SpatialHash::new(cell),
        used_area: 0.0,
        anchors: Vec::new(),
    }
}

fn place_piece(
    stock_poly: &PolyData,
    stocks: &mut Vec<StockInstance>,
    piece: &Piece,
    cfg: &SearchCfg,
) -> Placement {
    let rotations = candidate_rotations(&piece.poly, stock_poly, cfg);
    let mut stock_order: Vec<usize> = (0..stocks.len()).collect();
    stock_order.sort_by(|&a, &b| {
        stock_score(&stocks[b], piece.poly.area, stock_poly.area)
            .partial_cmp(&stock_score(&stocks[a], piece.poly.area, stock_poly.area))
            .unwrap_or(Ordering::Equal)
    });
    stock_order.truncate(cfg.max_stock_probe.min(stock_order.len()));
    for &sid in &stock_order {
        for &ang in &rotations {
            if let Some((trans, verts, bb)) = try_place_in_stock(stock_poly, &stocks[sid], &piece.poly, ang, cfg) {
                let center = polygon_centroid(&verts);
                let plc = Placement {
                    piece_id: piece.id,
                    stock_id: sid,
                    trans,
                    angle: ang,
                    verts,
                    bbox: bb,
                    center,
                    radius: piece.poly.radius,
                };
                return plc;
            }
        }
    }

    let sid = stocks.len();
    stocks.push(create_stock_instance(&stock_poly.bbox));
    for &ang in &rotations {
        if let Some((trans, verts, bb)) = try_place_in_stock(stock_poly, &stocks[sid], &piece.poly, ang, cfg) {
            let center = polygon_centroid(&verts);
            return Placement {
                piece_id: piece.id,
                stock_id: sid,
                trans,
                angle: ang,
                verts,
                bbox: bb,
                center,
                radius: piece.poly.radius,
            };
        }
    }

    for &ang in &[0.0, FRAC_PI_2, PI, -FRAC_PI_2] {
        if let Some((trans, verts, bb)) = try_place_in_stock(stock_poly, &stocks[sid], &piece.poly, ang, cfg) {
            let center = polygon_centroid(&verts);
            return Placement {
                piece_id: piece.id,
                stock_id: sid,
                trans,
                angle: ang,
                verts,
                bbox: bb,
                center,
                radius: piece.poly.radius,
            };
        }
    }

    let fallback_trans = stock_poly.centroid - piece.poly.centroid;
    let verts = transform(&piece.poly.verts, 0.0, fallback_trans);
    let bb = bbox_of(&verts);
    Placement {
        piece_id: piece.id,
        stock_id: sid,
        trans: fallback_trans,
        angle: 0.0,
        center: polygon_centroid(&verts),
        radius: piece.poly.radius,
        verts,
        bbox: bb,
    }
}

fn commit_placement(stock: &mut StockInstance, plc: Placement, piece_area: f64) {
    let idx = stock.placements.len();
    stock.hash.insert(idx, &plc.bbox);
    stock.used_area += piece_area;
    stock.anchors.push(plc.center);
    stock.anchors.push(Pt::new(plc.bbox.max.x, plc.center.y));
    stock.anchors.push(Pt::new(plc.center.x, plc.bbox.max.y));
    stock.placements.push(plc);
}

struct Scanner {
    data: Vec<String>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        let data = input.split_whitespace().map(|s| s.to_string()).collect();
        Self { data, idx: 0 }
    }

    fn next<T: std::str::FromStr>(&mut self) -> T {
        let s = &self.data[self.idx];
        self.idx += 1;
        s.parse::<T>().ok().unwrap()
    }
}

fn main() {
    let mut sc = Scanner::new();
    let sv: usize = sc.next();
    let np: usize = sc.next();

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = sc.next();
        let y: f64 = sc.next();
        stock.push(Pt::new(x, y));
    }
    let stock_poly = build_poly_data(stock);

    let mut pieces = Vec::with_capacity(np);
    let mut complexity = 0usize;
    for id in 0..np {
        let pv: usize = sc.next();
        complexity = complexity.max(pv);
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = sc.next();
            let y: f64 = sc.next();
            poly.push(Pt::new(x, y));
        }
        pieces.push(Piece {
            id,
            poly: build_poly_data(poly),
        });
    }

    let cfg = make_search_cfg(np, complexity);

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        let pa = &pieces[a].poly;
        let pb = &pieces[b].poly;
        let sa = pa.area * 10.0 + pa.bbox.area();
        let sb = pb.area * 10.0 + pb.bbox.area();
        sb.partial_cmp(&sa).unwrap_or(Ordering::Equal)
    });

    let mut stocks = vec![create_stock_instance(&stock_poly.bbox)];
    let mut answer: Vec<Option<(usize, Pt, f64)>> = vec![None; np];

    for idx in order {
        let piece = &pieces[idx];
        let plc = place_piece(&stock_poly, &mut stocks, piece, &cfg);
        let sid = plc.stock_id;
        answer[piece.id] = Some((sid, plc.trans, plc.angle));
        commit_placement(&mut stocks[sid], plc, piece.poly.area);
    }

    println!("{}", stocks.len());
    for item in answer {
        let (sid, trans, ang) = item.unwrap();
        println!("{} {:.10} {:.10} {:.10}", sid, trans.x, trans.y, ang);
    }
}
