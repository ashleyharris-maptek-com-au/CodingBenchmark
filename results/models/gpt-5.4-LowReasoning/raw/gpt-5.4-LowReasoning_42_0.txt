OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04
%float_n0_04 = OpConstant %float -0.04
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_1 = OpTypeArray %gl_PerVertex_in %uint_1
%_ptr_Input_arr_gl_PerVertex_in_1 = OpTypePointer Input %arr_gl_PerVertex_in_1
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in_1 Input
%arr_v3float_1 = OpTypeArray %v3float %uint_1
%_ptr_Input_arr_v3float_1 = OpTypePointer Input %arr_v3float_1
%worldPos = OpVariable %_ptr_Input_arr_v3float_1 Input
%normal = OpVariable %_ptr_Input_arr_v3float_1 Input
%color = OpVariable %_ptr_Input_arr_v3float_1 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %_ptr_Output_v3float Output
%main = OpFunction %void None %func
%entry = OpLabel
%pos_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%color_ptr = OpAccessChain %_ptr_Input_v3float %color %int_0
%in_color = OpLoad %v3float %color_ptr
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%x0 = OpFAdd %float %x %float_n0_04
%x1 = OpFAdd %float %x %float_0_04
%y0 = OpFAdd %float %y %float_n0_04
%y1 = OpFAdd %float %y %float_0_04
%gl_out_pos_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
%p0 = OpCompositeConstruct %v4float %x0 %y0 %float_0 %float_1
OpStore %gl_out_pos_ptr %p0
OpStore %outColor %in_color

%p1 = OpCompositeConstruct %v4float %x1 %y0 %float_0 %float_1
OpStore %gl_out_pos_ptr %p1
OpStore %outColor %in_color
OpEmitVertex
%p2 = OpCompositeConstruct %v4float %x0 %y1 %float_0 %float_1
OpStore %gl_out_pos_ptr %p2
OpStore %outColor %in_color
OpEmitVertex
%p3 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
OpStore %gl_out_pos_ptr %p3
OpStore %outColor %in_color
OpEmitVertex
OpEndPrimitive
OpReturn
