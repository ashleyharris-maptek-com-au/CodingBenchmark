const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const n = width * height;
const grid = data.grid || [];
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const towns = Array.isArray(data.towns) ? data.towns : [];
const cutBudget = Math.max(0, data.cutBudget | 0);

const dirs = [1, 0, -1, 0, 1];
const INF = 0x3fffffff;
const PATH_CAP = 1e18;

function idx(x, y) {
  return y * width + x;
}

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

const burnable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) burnable[idx(x, y)] = 1; // '.'
  }
}

const townAt = new Int32Array(n);
const townValueAt = new Float64Array(n);
for (let i = 0; i < n; i++) townAt[i] = -1;
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (!t || !inBounds(t.x, t.y)) continue;
  const id = idx(t.x | 0, t.y | 0);
  townAt[id] = i;
  townValueAt[id] += Number(t.value) || 0;
}

function multiSourceDistances(blocked) {
  const dist = new Int32Array(n);
  dist.fill(-1);
  const queue = new Int32Array(n);
  let head = 0;
  let tail = 0;

  for (let i = 0; i < ignitions.length; i++) {
    const p = ignitions[i];
    if (!p || !inBounds(p[0], p[1])) continue;
    const id = idx(p[0] | 0, p[1] | 0);
    if (!burnable[id]) continue;
    if (blocked && blocked[id]) continue;
    if (dist[id] !== -1) continue;
    dist[id] = 0;
    queue[tail++] = id;
  }

  while (head < tail) {
    const cur = queue[head++];
    const x = cur % width;
    const y = (cur / width) | 0;
    const nd = dist[cur] + 1;
    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k];
      const ny = y + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (!burnable[ni]) continue;
      if (blocked && blocked[ni]) continue;
      if (dist[ni] !== -1) continue;
      dist[ni] = nd;
      queue[tail++] = ni;
    }
  }

  return { dist, order: queue.subarray(0, tail) };
}

function computeBurnedValue(dist) {
  let total = 0;
  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    if (!t || !inBounds(t.x, t.y)) continue;
    const id = idx(t.x | 0, t.y | 0);
    if (dist[id] !== -1) total += Number(t.value) || 0;
  }
  return total;
}

function buildCandidates(baseDist, order) {
  const pathCount = new Float64Array(n);
  const townFlow = new Float64Array(n);
  const score = new Float64Array(n);

  for (let i = 0; i < order.length; i++) {
    const id = order[i];
    if (baseDist[id] === 0) {
      pathCount[id] = 1;
      continue;
    }
    const x = id % width;
    const y = (id / width) | 0;
    let sum = 0;
    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k];
      const ny = y + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (baseDist[ni] === baseDist[id] - 1) {
        sum += pathCount[ni];
        if (sum > PATH_CAP) sum = PATH_CAP;
      }
    }
    pathCount[id] = sum;
  }

  for (let i = 0; i < n; i++) {
    townFlow[i] = townValueAt[i];
  }

  for (let oi = order.length - 1; oi >= 0; oi--) {
    const id = order[oi];
    const d = baseDist[id];
    if (d <= 0) continue;
    const flow = townFlow[id];
    const totalPaths = pathCount[id];
    if (flow <= 0 || totalPaths <= 0) continue;
    const x = id % width;
    const y = (id / width) | 0;
    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k];
      const ny = y + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (baseDist[ni] !== d - 1) continue;
      const share = flow * (pathCount[ni] / totalPaths);
      townFlow[ni] += share;
    }
  }

  const list = [];
  for (let i = 0; i < order.length; i++) {
    const id = order[i];
    const d = baseDist[id];
    if (d <= 0) continue;
    if (townAt[id] !== -1) continue;
    const x = id % width;
    const y = (id / width) | 0;
    let branching = 0;
    let frontier = 0;
    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k];
      const ny = y + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (!burnable[ni]) continue;
      if (baseDist[ni] === d - 1) branching++;
      if (baseDist[ni] === d + 1) frontier++;
    }
    const bottleneck = frontier / Math.max(1, branching);
    score[id] = townFlow[id] * (1 + Math.min(3, bottleneck)) / (d + 1);
    if (score[id] > 0) list.push(id);
  }

  list.sort((a, b) => score[b] - score[a]);

  const chosen = [];
  const blockedNear = new Uint8Array(n);
  const limit = Math.min(list.length, Math.max(cutBudget * 30, 200));
  for (let i = 0; i < limit; i++) {
    const id = list[i];
    if (blockedNear[id]) continue;
    chosen.push(id);
    const x = id % width;
    const y = (id / width) | 0;
    for (let dy = -1; dy <= 1; dy++) {
      for (let dx = -1; dx <= 1; dx++) {
        const nx = x + dx;
        const ny = y + dy;
        if (inBounds(nx, ny)) blockedNear[idx(nx, ny)] = 1;
      }
    }
    if (chosen.length >= Math.max(50, cutBudget * 10)) break;
  }

  return { candidates: chosen, score };
}

function savedValueWithCuts(cutList) {
  const blocked = new Uint8Array(n);
  for (let i = 0; i < cutList.length; i++) blocked[cutList[i]] = 1;
  const dist = multiSourceDistances(blocked).dist;
  let saved = 0;
  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    if (!t || !inBounds(t.x, t.y)) continue;
    const id = idx(t.x | 0, t.y | 0);
    if (dist[id] === -1) saved += Number(t.value) || 0;
  }
  return saved;
}

function solve() {
  if (cutBudget === 0 || n === 0) return [];

  const base = multiSourceDistances(null);
  const baseBurnedValue = computeBurnedValue(base.dist);
  if (baseBurnedValue <= 0) return [];

  const { candidates, score } = buildCandidates(base.dist, base.order);
  if (candidates.length === 0) return [];

  const result = [];
  const used = new Uint8Array(n);
  let currentSaved = 0;

  const exactPool = Math.min(candidates.length, cutBudget <= 12 ? 80 : 48);
  for (let step = 0; step < cutBudget && step < exactPool; step++) {
    let bestId = -1;
    let bestSaved = currentSaved;
    for (let i = 0; i < exactPool; i++) {
      const id = candidates[i];
      if (used[id]) continue;
      result.push(id);
      const saved = savedValueWithCuts(result);
      result.pop();
      if (saved > bestSaved || (saved === bestSaved && bestId !== -1 && score[id] > score[bestId])) {
        bestSaved = saved;
        bestId = id;
      }
    }
    if (bestId === -1 || bestSaved < currentSaved) break;
    used[bestId] = 1;
    result.push(bestId);
    currentSaved = bestSaved;
  }

  if (result.length < cutBudget) {
    for (let i = 0; i < candidates.length && result.length < cutBudget; i++) {
      const id = candidates[i];
      if (used[id]) continue;
      used[id] = 1;
      result.push(id);
    }
  }

  return result;
}

const firebreaks = solve().map((id) => [id % width, (id / width) | 0]);
process.stdout.write(JSON.stringify({ firebreaks }));
