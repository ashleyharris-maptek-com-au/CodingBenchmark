const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid || [];
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const towns = Array.isArray(data.towns) ? data.towns : [];
const cutBudget = Math.max(0, data.cutBudget | 0);

const cellCount = width * height;
const dirs = [1, 0, -1, 0, 1];

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function idx(x, y) {
  return y * width + x;
}

function isBurnableIndex(i) {
  const y = (i / width) | 0;
  const x = i - y * width;
  return grid[y].charCodeAt(x) === 46;
}

const burnable = new Uint8Array(cellCount);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) burnable[idx(x, y)] = 1;
  }
}

function computeDistances(blocked) {
  const dist = new Int32Array(cellCount);
  dist.fill(-1);
  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;

  for (let i = 0; i < ignitions.length; i++) {
    const p = ignitions[i];
    const x = p[0] | 0;
    const y = p[1] | 0;
    if (!inBounds(x, y)) continue;
    const id = idx(x, y);
    if (!burnable[id]) continue;
    if (blocked && blocked.has(id)) continue;
    if (dist[id] !== -1) continue;
    dist[id] = 0;
    queue[tail++] = id;
  }

  while (head < tail) {
    const cur = queue[head++];
    const cy = (cur / width) | 0;
    const cx = cur - cy * width;
    const nextDist = dist[cur] + 1;

    for (let k = 0; k < 4; k++) {
      const nx = cx + dirs[k];
      const ny = cy + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const nid = idx(nx, ny);
      if (!burnable[nid]) continue;
      if (dist[nid] !== -1) continue;
      if (blocked && blocked.has(nid)) continue;
      dist[nid] = nextDist;
      queue[tail++] = nid;
    }
  }

  return dist;
}

const baseDist = computeDistances(null);

const townInfos = [];
const townValueByCell = new Float64Array(cellCount);
let maxThreatenedTownDist = 0;
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0;
  const y = t.y | 0;
  const value = Number(t.value) || 0;
  if (!inBounds(x, y)) continue;
  const id = idx(x, y);
  const threatened = burnable[id] && baseDist[id] >= 0;
  if (threatened) {
    townInfos.push({ index: i, id, x, y, value, dist: baseDist[id] });
    townValueByCell[id] += value;
    if (baseDist[id] > maxThreatenedTownDist) maxThreatenedTownDist = baseDist[id];
  }
}

if (cutBudget === 0 || townInfos.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

townInfos.sort((a, b) => b.value - a.value || b.dist - a.dist);

function buildThresholds() {
  const set = new Set();
  set.add(1);

  for (let i = 0; i < townInfos.length; i++) {
    const d = townInfos[i].dist | 0;
    if (d > 0) {
      set.add(d);
      if (d > 1) set.add(d - 1);
    }
  }

  if (maxThreatenedTownDist > 0) {
    const samples = Math.min(24, maxThreatenedTownDist);
    for (let i = 1; i <= samples; i++) {
      const t = Math.max(1, Math.floor((i * maxThreatenedTownDist) / samples));
      set.add(t);
    }
  }

  return Array.from(set).filter((t) => t > 0).sort((a, b) => b - a);
}

function gatherCandidates(thresholds) {
  const candidates = [];

  for (let ti = 0; ti < thresholds.length; ti++) {
    const threshold = thresholds[ti];
    const seen = new Uint8Array(cellCount);
    const queue = new Int32Array(cellCount);

    for (let i = 0; i < townInfos.length; i++) {
      const seed = townInfos[i];
      if (seed.dist < threshold) continue;
      if (seen[seed.id]) continue;

      let head = 0;
      let tail = 0;
      queue[tail++] = seed.id;
      seen[seed.id] = 1;

      const cells = [];
      const frontier = [];
      const frontierMark = new Uint8Array(cellCount);
      const townCells = [];
      let townValue = 0;

      while (head < tail) {
        const cur = queue[head++];
        cells.push(cur);
        if (townValueByCell[cur] > 0) {
          townValue += townValueByCell[cur];
          townCells.push(cur);
        }

        const cy = (cur / width) | 0;
        const cx = cur - cy * width;
        for (let k = 0; k < 4; k++) {
          const nx = cx + dirs[k];
          const ny = cy + dirs[k + 1];
          if (!inBounds(nx, ny)) continue;
          const nid = idx(nx, ny);
          if (!burnable[nid]) continue;
          const d = baseDist[nid];
          if (d < 0) continue;
          if (d >= threshold) {
            if (!seen[nid]) {
              seen[nid] = 1;
              queue[tail++] = nid;
            }
          } else if (!frontierMark[nid]) {
            frontierMark[nid] = 1;
            frontier.push(nid);
          }
        }
      }

      if (townValue <= 0 || frontier.length === 0 || frontier.length > cutBudget) continue;

      candidates.push({
        threshold,
        townValue,
        cost: frontier.length,
        score: townValue / frontier.length,
        towns: townCells,
        cuts: frontier.slice(),
      });
    }
  }

  candidates.sort((a, b) => b.score - a.score || b.townValue - a.townValue || a.cost - b.cost);
  return candidates;
}

const thresholds = buildThresholds();
const candidates = gatherCandidates(thresholds);

const selectedCuts = new Set();
const protectedTownCells = new Set();
let remainingBudget = cutBudget;

for (let i = 0; i < candidates.length; i++) {
  const cand = candidates[i];
  let addedCost = 0;
  for (let j = 0; j < cand.cuts.length; j++) {
    if (!selectedCuts.has(cand.cuts[j])) addedCost++;
  }
  if (addedCost === 0 || addedCost > remainingBudget) continue;

  let marginalValue = 0;
  for (let j = 0; j < cand.towns.length; j++) {
    const townCell = cand.towns[j];
    if (!protectedTownCells.has(townCell)) marginalValue += townValueByCell[townCell];
  }

  if (marginalValue <= 0) continue;
  if (marginalValue / addedCost < cand.score * 0.35) continue;

  for (let j = 0; j < cand.cuts.length; j++) selectedCuts.add(cand.cuts[j]);
  remainingBudget -= addedCost;

  for (let j = 0; j < cand.towns.length; j++) protectedTownCells.add(cand.towns[j]);

  if (remainingBudget === 0) break;
}

function burnedTownValue(blocked) {
  const dist = computeDistances(blocked);
  let burned = 0;
  for (let i = 0; i < townInfos.length; i++) {
    const t = townInfos[i];
    if (dist[t.id] >= 0) burned += t.value;
  }
  return burned;
}

function topPathCandidates(limit) {
  const scores = new Float64Array(cellCount);
  const maxTowns = Math.min(townInfos.length, 64);

  for (let i = 0; i < maxTowns; i++) {
    const t = townInfos[i];
    if (selectedCuts.has(t.id)) continue;
    let cur = t.id;
    let d = baseDist[cur];
    if (d <= 0) continue;
    const weight = t.value / (d + 1);
    while (d > 0) {
      scores[cur] += weight;
      const cy = (cur / width) | 0;
      const cx = cur - cy * width;
      let best = -1;
      let bestDist = d;
      for (let k = 0; k < 4; k++) {
        const nx = cx + dirs[k];
        const ny = cy + dirs[k + 1];
        if (!inBounds(nx, ny)) continue;
        const nid = idx(nx, ny);
        if (baseDist[nid] >= 0 && baseDist[nid] < bestDist && burnable[nid]) {
          bestDist = baseDist[nid];
          best = nid;
        }
      }
      if (best < 0) break;
      cur = best;
      d = bestDist;
    }
  }

  const list = [];
  for (let id = 0; id < cellCount; id++) {
    if (!burnable[id]) continue;
    if (baseDist[id] <= 0) continue;
    if (selectedCuts.has(id)) continue;
    const s = scores[id];
    if (s > 0) list.push([s, id]);
  }
  list.sort((a, b) => b[0] - a[0]);
  return list.slice(0, limit).map((x) => x[1]);
}

if (remainingBudget > 0) {
  let currentBurned = burnedTownValue(selectedCuts);
  const extraCandidates = topPathCandidates(Math.min(256, cellCount));

  for (let step = 0; step < remainingBudget; step++) {
    let bestCell = -1;
    let bestGain = 0;

    for (let i = 0; i < extraCandidates.length; i++) {
      const id = extraCandidates[i];
      if (selectedCuts.has(id)) continue;
      selectedCuts.add(id);
      const burned = burnedTownValue(selectedCuts);
      selectedCuts.delete(id);
      const gain = currentBurned - burned;
      if (gain > bestGain) {
        bestGain = gain;
        bestCell = id;
      }
    }

    if (bestCell < 0 || bestGain <= 0) break;
    selectedCuts.add(bestCell);
    currentBurned -= bestGain;
  }
}

if (selectedCuts.size > 1 && selectedCuts.size <= 200) {
  let currentBurned = burnedTownValue(selectedCuts);
  const cutList = Array.from(selectedCuts);
  for (let i = 0; i < cutList.length; i++) {
    const id = cutList[i];
    if (!selectedCuts.has(id)) continue;
    selectedCuts.delete(id);
    const burned = burnedTownValue(selectedCuts);
    if (burned === currentBurned) {
      currentBurned = burned;
    } else {
      selectedCuts.add(id);
    }
  }
}

const firebreaks = [];
for (const id of selectedCuts) {
  const y = (id / width) | 0;
  const x = id - y * width;
  firebreaks.push([x, y]);
}

firebreaks.sort((a, b) => a[1] - b[1] || a[0] - b[0]);
if (firebreaks.length > cutBudget) firebreaks.length = cutBudget;

process.stdout.write(JSON.stringify({ firebreaks }));
