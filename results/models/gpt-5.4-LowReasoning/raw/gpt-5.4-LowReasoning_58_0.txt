const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = Array.isArray(data.grid) ? data.grid : [];
const robotsInput = Array.isArray(data.robots) ? data.robots : [];
const jobsInput = Array.isArray(data.jobs) ? data.jobs : [];

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const cellCount = width * height;

function indexOf(x, y) {
  return y * width + x;
}

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

const open = new Uint8Array(cellCount);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) open[indexOf(x, y)] = 1;
  }
}

function isOpen(x, y) {
  return inBounds(x, y) && open[indexOf(x, y)] === 1;
}

const maxRobotId = robotsInput.reduce((m, r) => Math.max(m, r.id | 0), -1);
const assignments = Array.from({ length: maxRobotId + 1 }, () => []);
const robotStates = [];

for (const robot of robotsInput) {
  const id = robot.id | 0;
  robotStates.push({
    id,
    x: robot.x | 0,
    y: robot.y | 0,
    pos: indexOf(robot.x | 0, robot.y | 0),
    travel: 0,
  });
}

robotStates.sort((a, b) => a.id - b.id);

const distCache = new Map();
const queue = new Int32Array(Math.max(cellCount, 1));

function bfsFromTarget(targetPos) {
  let cached = distCache.get(targetPos);
  if (cached) return cached;

  const dist = new Int32Array(cellCount);
  dist.fill(-1);

  if (targetPos < 0 || targetPos >= cellCount || open[targetPos] === 0) {
    distCache.set(targetPos, dist);
    return dist;
  }

  let head = 0;
  let tail = 0;
  dist[targetPos] = 0;
  queue[tail++] = targetPos;

  while (head < tail) {
    const pos = queue[head++];
    const d = dist[pos] + 1;
    const x = pos % width;
    const y = (pos / width) | 0;

    let np;
    if (x > 0) {
      np = pos - 1;
      if (open[np] && dist[np] === -1) {
        dist[np] = d;
        queue[tail++] = np;
      }
    }
    if (x + 1 < width) {
      np = pos + 1;
      if (open[np] && dist[np] === -1) {
        dist[np] = d;
        queue[tail++] = np;
      }
    }
    if (y > 0) {
      np = pos - width;
      if (open[np] && dist[np] === -1) {
        dist[np] = d;
        queue[tail++] = np;
      }
    }
    if (y + 1 < height) {
      np = pos + width;
      if (open[np] && dist[np] === -1) {
        dist[np] = d;
        queue[tail++] = np;
      }
    }
  }

  distCache.set(targetPos, dist);
  return dist;
}

function fallbackDistance(ax, ay, bx, by) {
  return Math.abs(ax - bx) + Math.abs(ay - by) + width + height + 1;
}

const jobs = jobsInput.map((job, inputIndex) => ({
  id: job.id | 0,
  x: job.x | 0,
  y: job.y | 0,
  pos: indexOf(job.x | 0, job.y | 0),
  priority: Number(job.priority) || 0,
  inputIndex,
}));

jobs.sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  if (a.y !== b.y) return a.y - b.y;
  if (a.x !== b.x) return a.x - b.x;
  if (a.id !== b.id) return a.id - b.id;
  return a.inputIndex - b.inputIndex;
});

for (const job of jobs) {
  const targetOpen = isOpen(job.x, job.y);
  const dist = targetOpen ? bfsFromTarget(job.pos) : null;
  let bestRobot = null;
  let bestScore = Infinity;
  let bestReachable = 0;

  for (const robot of robotStates) {
    let stepCost = -1;
    if (targetOpen && robot.pos >= 0 && robot.pos < cellCount) {
      stepCost = dist[robot.pos];
    }
    const reachable = stepCost >= 0 ? 1 : 0;
    const score = reachable ? stepCost : fallbackDistance(robot.x, robot.y, job.x, job.y);

    if (
      bestRobot === null ||
      reachable > bestReachable ||
      score < bestScore ||
      (score === bestScore && robot.travel < bestRobot.travel) ||
      (score === bestScore && robot.travel === bestRobot.travel && robot.id < bestRobot.id)
    ) {
      bestRobot = robot;
      bestScore = score;
      bestReachable = reachable;
    }
  }

  assignments[bestRobot.id].push(job.id);
  bestRobot.travel += bestScore;
  bestRobot.x = job.x;
  bestRobot.y = job.y;
  bestRobot.pos = job.pos;
}

process.stdout.write(JSON.stringify({ assignments }));
