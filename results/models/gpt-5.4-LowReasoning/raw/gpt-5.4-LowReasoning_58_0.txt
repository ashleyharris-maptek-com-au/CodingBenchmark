const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const grid = Array.isArray(data.grid) ? data.grid : [];
const robotsInput = Array.isArray(data.robots) ? data.robots : [];
const jobsInput = Array.isArray(data.jobs) ? data.jobs : [];

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const cellCount = width * height;

function inBounds(x, y) {
  return y >= 0 && y < height && x >= 0 && x < width;
}

function cellOpen(x, y) {
  return inBounds(x, y) && grid[y][x] === '.';
}

function toIndex(x, y) {
  return y * width + x;
}

const openMask = new Uint8Array(cellCount);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    if (row[x] === '.') openMask[toIndex(x, y)] = 1;
  }
}

const distanceCache = new Map();
const queue = new Int32Array(Math.max(1, cellCount));

function bfsFrom(x, y) {
  const key = x + ',' + y;
  let cached = distanceCache.get(key);
  if (cached) return cached;

  const dist = new Int32Array(Math.max(1, cellCount));
  dist.fill(-1);

  if (!cellOpen(x, y)) {
    distanceCache.set(key, dist);
    return dist;
  }

  let head = 0;
  let tail = 0;
  const start = toIndex(x, y);
  dist[start] = 0;
  queue[tail++] = start;

  while (head < tail) {
    const cur = queue[head++];
    const curDist = dist[cur] + 1;
    const cx = cur % width;
    const cy = (cur / width) | 0;

    if (cx > 0) {
      const n = cur - 1;
      if (openMask[n] && dist[n] === -1) {
        dist[n] = curDist;
        queue[tail++] = n;
      }
    }
    if (cx + 1 < width) {
      const n = cur + 1;
      if (openMask[n] && dist[n] === -1) {
        dist[n] = curDist;
        queue[tail++] = n;
      }
    }
    if (cy > 0) {
      const n = cur - width;
      if (openMask[n] && dist[n] === -1) {
        dist[n] = curDist;
        queue[tail++] = n;
      }
    }
    if (cy + 1 < height) {
      const n = cur + width;
      if (openMask[n] && dist[n] === -1) {
        dist[n] = curDist;
        queue[tail++] = n;
      }
    }
  }

  distanceCache.set(key, dist);
  return dist;
}

function getDistance(fromX, fromY, toX, toY) {
  if (!cellOpen(toX, toY)) return Number.MAX_SAFE_INTEGER;
  const dist = bfsFrom(fromX, fromY);
  const value = dist[toIndex(toX, toY)];
  return value >= 0 ? value : Number.MAX_SAFE_INTEGER;
}

let maxRobotId = -1;
for (const robot of robotsInput) {
  if (robot && Number.isInteger(robot.id) && robot.id > maxRobotId) {
    maxRobotId = robot.id;
  }
}

const robotCount = maxRobotId + 1;
const assignments = Array.from({ length: Math.max(0, robotCount) }, () => []);
const robots = new Array(Math.max(0, robotCount));

for (let i = 0; i < robotCount; i++) {
  robots[i] = {
    id: i,
    x: 0,
    y: 0,
    totalTravel: 0,
    jobCount: 0,
    valid: false
  };
}

for (const robot of robotsInput) {
  if (!robot || !Number.isInteger(robot.id) || robot.id < 0 || robot.id >= robotCount) continue;
  robots[robot.id] = {
    id: robot.id,
    x: robot.x | 0,
    y: robot.y | 0,
    totalTravel: 0,
    jobCount: 0,
    valid: true
  };
}

const jobs = jobsInput
  .filter((job) => job && Number.isInteger(job.id))
  .map((job, index) => ({
    id: job.id,
    x: job.x | 0,
    y: job.y | 0,
    priority: Number.isFinite(job.priority) ? job.priority : 0,
    inputOrder: index
  }));

jobs.sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  if (a.id !== b.id) return a.id - b.id;
  return a.inputOrder - b.inputOrder;
});

for (const job of jobs) {
  let bestRobot = -1;
  let bestDistance = Number.MAX_SAFE_INTEGER;
  let bestTotalTravel = Number.MAX_SAFE_INTEGER;
  let bestJobCount = Number.MAX_SAFE_INTEGER;

  for (let robotId = 0; robotId < robots.length; robotId++) {
    const robot = robots[robotId];
    if (!robot || !robot.valid) continue;

    const distance = getDistance(robot.x, robot.y, job.x, job.y);
    if (distance < bestDistance) {
      bestDistance = distance;
      bestTotalTravel = robot.totalTravel;
      bestJobCount = robot.jobCount;
      bestRobot = robotId;
      continue;
    }
    if (distance !== bestDistance) continue;

    if (robot.totalTravel < bestTotalTravel) {
      bestTotalTravel = robot.totalTravel;
      bestJobCount = robot.jobCount;
      bestRobot = robotId;
      continue;
    }
    if (robot.totalTravel !== bestTotalTravel) continue;

    if (robot.jobCount < bestJobCount) {
      bestJobCount = robot.jobCount;
      bestRobot = robotId;
      continue;
    }
    if (robot.jobCount !== bestJobCount) continue;

    if (robotId < bestRobot) bestRobot = robotId;
  }

  if (bestRobot === -1) {
    continue;
  }

  assignments[bestRobot].push(job.id);
  const robot = robots[bestRobot];
  if (bestDistance !== Number.MAX_SAFE_INTEGER) {
    robot.totalTravel += bestDistance;
  }
  robot.x = job.x;
  robot.y = job.y;
  robot.jobCount++;
}

process.stdout.write(JSON.stringify({ assignments }));
