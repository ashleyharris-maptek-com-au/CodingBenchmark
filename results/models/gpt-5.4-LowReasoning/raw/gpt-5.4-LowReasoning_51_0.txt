from math import atan2, cos, sin, sqrt


_AP_STATE = {
    "alt_i": 0.0,
    "vs_i": 0.0,
    "hdg_i": 0.0,
    "ias_i": 0.0,
    "trim": 0.0,
    "last_alt": None,
    "last_hdg_err": 0.0,
    "last_pitch": 0.0,
    "last_roll": 0.0,
}


def autopilot_step(state: dict, dt: float) -> dict:
    def clamp(x, lo, hi):
        return lo if x < lo else hi if x > hi else x

    def wrap180(deg):
        while deg > 180.0:
            deg -= 360.0
        while deg < -180.0:
            deg += 360.0
        return deg

    def median(values):
        vals = sorted(values)
        n = len(vals)
        if n == 0:
            return None
        if n % 2:
            return vals[n // 2]
        return 0.5 * (vals[n // 2 - 1] + vals[n // 2])

    def mean(values):
        return sum(values) / len(values) if values else None

    def pick_triplet(prefix, field, valid_suffix="_valid", max_spread=None):
        samples = []
        for i in (1, 2, 3):
            value = state.get(f"{prefix}_{i}_{field}")
            valid = bool(state.get(f"{prefix}_{i}{valid_suffix}", True))
            if value is None:
                continue
            samples.append((float(value), valid, i))
        if not samples:
            return None, 0.0
        values = [v for v, valid, _ in samples if valid]
        if not values:
            values = [v for v, _, _ in samples]
        if len(values) == 1:
            return values[0], 0.35
        med = median(values)
        if max_spread is not None:
            close = [v for v in values if abs(v - med) <= max_spread]
            if close:
                values = close
                med = median(values)
        spread = max(values) - min(values)
        conf = 1.0 / (1.0 + spread / max(1.0, max_spread or max(abs(med), 1.0)))
        return med, clamp(conf, 0.2, 1.0)

    def pick_angle(prefix, field):
        xs = []
        for i in (1, 2, 3):
            ang = state.get(f"{prefix}_{i}_{field}")
            valid = bool(state.get(f"{prefix}_{i}_valid", True))
            if ang is not None and valid:
                r = float(ang) * 3.141592653589793 / 180.0
                xs.append((cos(r), sin(r)))
        if not xs:
            return None, 0.0
        cx = sum(a for a, _ in xs) / len(xs)
        sx = sum(b for _, b in xs) / len(xs)
        mag = sqrt(cx * cx + sx * sx)
        ang = atan2(sx, cx) * 180.0 / 3.141592653589793
        return ang % 360.0, clamp(mag, 0.2, 1.0)

    def blend(primary, secondary, w_primary=0.7):
        if primary is None:
            return secondary
        if secondary is None:
            return primary
        return primary * w_primary + secondary * (1.0 - w_primary)

    dt = max(0.01, min(0.2, float(dt if dt is not None else state.get("dt", 0.05))))

    ias, ias_conf = pick_triplet("ias", "kt", max_spread=12.0)
    aoa, aoa_conf = pick_triplet("aoa", "deg", max_spread=3.0)
    pitch, pitch_conf = pick_triplet("irs", "pitch_deg", max_spread=4.0)
    roll, roll_conf = pick_triplet("irs", "roll_deg", max_spread=6.0)
    vs_irs, vs_conf = pick_triplet("irs", "vs_fpm", max_spread=600.0)
    baro_alt, baro_conf = pick_triplet("baro_alt", "ft", valid_suffix="_valid", max_spread=120.0)
    radar_alt, radar_conf = pick_triplet("radar_alt", "ft", valid_suffix="_valid", max_spread=40.0)
    gps_alt, gps_alt_conf = pick_triplet("gps", "alt_ft", max_spread=180.0)
    gs_gps, _ = pick_triplet("gps", "gs_kt", max_spread=25.0)
    gs_irs, _ = pick_triplet("irs", "gs_kt", max_spread=25.0)
    yaw_rate, _ = pick_triplet("gyro", "yaw_rate_dps", max_spread=2.5)
    roll_rate, _ = pick_triplet("gyro", "roll_rate_dps", max_spread=3.0)
    track, track_conf = pick_angle("gps", "track_deg")
    if track is None:
        track, track_conf = pick_angle("irs", "track_deg")
    heading, hdg_conf = pick_angle("irs", "heading_deg")
    if heading is None:
        heading = float(state.get("magnetic_heading_deg", state.get("true_heading_deg", 0.0))) % 360.0
        hdg_conf = 0.4

    if vs_irs is None:
        vs_irs = float(state.get("vertical_speed_fpm", state.get("target_vs_fpm", 0.0)))
    if pitch is None:
        pitch = _AP_STATE["last_pitch"]
    if roll is None:
        roll = _AP_STATE["last_roll"]
    if ias is None:
        ias = float(state.get("target_ias_kt", 250.0))
        ias_conf = 0.2

    pressure_alt = []
    for i in (1, 2, 3):
        if bool(state.get(f"adc_{i}_valid", True)) and state.get(f"adc_{i}_pressure_alt_ft") is not None:
            pressure_alt.append(float(state[f"adc_{i}_pressure_alt_ft"]))
    adc_alt = median(pressure_alt) if pressure_alt else None

    alt = blend(baro_alt, adc_alt, 0.75)
    if radar_alt is not None and (alt is None or radar_alt < 2500.0):
        alt = blend(radar_alt, alt, 0.75 if radar_alt < 1000.0 else 0.55)
    if alt is None:
        alt = gps_alt if gps_alt is not None else 0.0

    if _AP_STATE["last_alt"] is None:
        calc_vs = vs_irs
    else:
        calc_vs = (alt - _AP_STATE["last_alt"]) * 60.0 / dt
    vs = blend(vs_irs, calc_vs, 0.65)

    flaps_l = float(state.get("flap_pos_left_deg", 0.0))
    flaps_r = float(state.get("flap_pos_right_deg", flaps_l))
    flaps_now = 0.5 * (flaps_l + flaps_r)
    slats = 0.5 * (float(state.get("slat_pos_left_deg", 0.0)) + float(state.get("slat_pos_right_deg", 0.0)))
    gross_weight = float(state.get("gross_weight_kg", 64000.0))
    turbulence = float(state.get("turbulence_intensity", 0.0))
    wind_speed = float(state.get("wind_speed_kt", 0.0))
    throttle1_now = float(state.get("throttle_1_pos", 0.5))
    throttle2_now = float(state.get("throttle_2_pos", 0.5))
    eng1_ok = bool(state.get("engine_1_running", True)) and not bool(state.get("engine_1_fire", False))
    eng2_ok = bool(state.get("engine_2_running", True)) and not bool(state.get("engine_2_fire", False))
    wow = any(bool(state.get(k, False)) for k in ("weight_on_wheels_left", "weight_on_wheels_right", "weight_on_wheels_nose"))

    target_alt = float(state.get("target_alt_ft", alt))
    target_hdg = float(state.get("target_hdg_deg", heading))
    target_ias = float(state.get("target_ias_kt", ias))
    target_vs = float(state.get("target_vs_fpm", 0.0))

    clean_stall = 118.0 + max(0.0, (gross_weight - 60000.0) / 12000.0) * 8.0
    flap_bonus = min(30.0, flaps_now * 0.8 + slats * 0.15)
    stall_speed = max(95.0, clean_stall - flap_bonus)
    low_alt = radar_alt is not None and radar_alt < 1500.0 or alt < 1500.0
    approach = low_alt or flaps_now > 1.0 or state.get("gear_handle_down", False)
    min_speed = stall_speed + (20.0 if approach else 35.0) + min(10.0, turbulence * 8.0)
    max_speed = 340.0 if flaps_now > 0.5 else 350.0
    if flaps_now >= 5.0:
        max_speed = min(max_speed, 230.0)
    if flaps_now >= 15.0:
        max_speed = min(max_speed, 200.0)

    commanded_speed = clamp(target_ias, min_speed, max_speed)
    aoa_high = aoa is not None and aoa > 11.0
    stall_risk = ias < min_speed + 8.0 or aoa_high
    overspeed = ias > max_speed or float(mean([state.get("mach_1", 0.0), state.get("mach_2", 0.0), state.get("mach_3", 0.0)])) > 0.815

    alt_err = clamp(target_alt - alt, -4000.0, 4000.0)
    vs_cmd = clamp(alt_err * 3.0, -2200.0, 2200.0)
    if abs(alt_err) < 250.0:
        vs_cmd = clamp(target_vs, -1200.0, 1200.0)
    if stall_risk:
        vs_cmd = min(vs_cmd, -300.0 if low_alt else 0.0)
    if overspeed:
        vs_cmd = max(vs_cmd, -1500.0)

    _AP_STATE["alt_i"] = clamp(_AP_STATE["alt_i"] + alt_err * dt * 0.0008, -0.25, 0.25)
    _AP_STATE["vs_i"] = clamp(_AP_STATE["vs_i"] + (vs_cmd - vs) * dt * 0.00025, -0.2, 0.2)
    pitch_cmd = (
        0.0016 * (vs_cmd - vs)
        + 0.00012 * alt_err
        - 0.035 * pitch
        + _AP_STATE["alt_i"]
        + _AP_STATE["vs_i"]
    )
    if stall_risk:
        pitch_cmd = min(pitch_cmd, -0.12 if aoa_high else 0.02)
    if overspeed:
        pitch_cmd = max(pitch_cmd, -0.35)
    pitch_cmd = clamp(pitch_cmd, -0.35, 0.35)

    hdg_err = wrap180(target_hdg - heading)
    track_err = wrap180(target_hdg - track) if track is not None else hdg_err
    nav_err = 0.65 * hdg_err + 0.35 * track_err
    bank_limit = 12.0 if low_alt else 25.0
    if abs(nav_err) < 5.0:
        _AP_STATE["hdg_i"] = clamp(_AP_STATE["hdg_i"] + nav_err * dt * 0.02, -6.0, 6.0)
    else:
        _AP_STATE["hdg_i"] *= 0.98
    bank_cmd = clamp(0.85 * nav_err + _AP_STATE["hdg_i"] - 0.18 * (yaw_rate or 0.0), -bank_limit, bank_limit)
    roll_err = bank_cmd - roll
    aileron = clamp(0.045 * roll_err - 0.02 * (roll_rate or 0.0), -1.0, 1.0)
    rudder = clamp(-0.08 * (yaw_rate or 0.0) + 0.015 * roll + 0.25 * ((throttle2_now - throttle1_now) if eng1_ok and eng2_ok else 0.0), -1.0, 1.0)

    speed_err = commanded_speed - ias
    _AP_STATE["ias_i"] = clamp(_AP_STATE["ias_i"] + speed_err * dt * 0.01, -0.35, 0.35)
    base_throttle = 0.42 + 0.0025 * speed_err - 0.00010 * vs_cmd + _AP_STATE["ias_i"]
    if stall_risk:
        base_throttle = max(base_throttle, 0.82)
    if overspeed:
        base_throttle = min(base_throttle, 0.08)
    if not eng1_ok and eng2_ok:
        throttle_1 = 0.0
        throttle_2 = 1.0 if stall_risk or not wow else clamp(base_throttle, 0.0, 1.0)
        rudder += 0.35
    elif not eng2_ok and eng1_ok:
        throttle_2 = 0.0
        throttle_1 = 1.0 if stall_risk or not wow else clamp(base_throttle, 0.0, 1.0)
        rudder -= 0.35
    else:
        throttle_1 = clamp(base_throttle, 0.0, 1.0)
        throttle_2 = clamp(base_throttle, 0.0, 1.0)

    if wow:
        gear_cmd = 1
    else:
        gear_cmd = 1 if approach and alt < 2200.0 else 0

    if wow or (approach and commanded_speed < 190.0):
        if commanded_speed < 135.0:
            flaps_cmd = 30
        elif commanded_speed < 155.0:
            flaps_cmd = 15
        elif commanded_speed < 185.0:
            flaps_cmd = 5
        else:
            flaps_cmd = 0
    else:
        flaps_cmd = 0
    if overspeed:
        flaps_cmd = 0

    speedbrake = 0.0
    if overspeed and not low_alt and flaps_cmd == 0:
        speedbrake = clamp((ias - max_speed) / 25.0, 0.0, 0.8)
    if stall_risk:
        speedbrake = 0.0

    trim_target = clamp(pitch_cmd * 0.85, -0.6, 0.6)
    _AP_STATE["trim"] += clamp(trim_target - _AP_STATE["trim"], -0.12 * dt, 0.12 * dt)
    elevator = clamp((pitch_cmd - _AP_STATE["trim"]) * 3.0 - 0.03 * (state.get("gyro_1_pitch_rate_dps", 0.0) or 0.0), -1.0, 1.0)

    if not state.get("ap_engaged", True):
        elevator = 0.0
        aileron = 0.0
        rudder = 0.0
        _AP_STATE["alt_i"] *= 0.95
        _AP_STATE["vs_i"] *= 0.95
        _AP_STATE["hdg_i"] *= 0.95
    if not state.get("at_engaged", True):
        throttle_1 = throttle1_now
        throttle_2 = throttle2_now

    rudder = clamp(rudder, -1.0, 1.0)
    _AP_STATE["last_alt"] = alt
    _AP_STATE["last_hdg_err"] = hdg_err
    _AP_STATE["last_pitch"] = pitch
    _AP_STATE["last_roll"] = roll

    return {
        "elevator": clamp(elevator, -1.0, 1.0),
        "aileron": aileron,
        "rudder": rudder,
        "throttle_1": clamp(throttle_1, 0.0, 1.0),
        "throttle_2": clamp(throttle_2, 0.0, 1.0),
        "elevator_trim": clamp(_AP_STATE["trim"], -1.0, 1.0),
        "flaps": int(flaps_cmd),
        "speedbrake": clamp(speedbrake, 0.0, 1.0),
        "gear": int(gear_cmd),
    }
