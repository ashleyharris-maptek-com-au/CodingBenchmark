import math


_MEM = {
    "initialized": False,
    "last_time": None,
    "last_alt_ft": None,
    "last_track_deg": None,
    "last_ias_kt": None,
    "last_pitch_deg": 0.0,
    "last_roll_deg": 0.0,
    "alt_int": 0.0,
    "vs_int": 0.0,
    "hdg_int": 0.0,
    "ias_int": 0.0,
    "trim_int": 0.0,
    "flare_bias": 0.0,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _sign(x):
    return -1.0 if x < 0.0 else 1.0


def _ang_norm(deg):
    while deg <= -180.0:
        deg += 360.0
    while deg > 180.0:
        deg -= 360.0
    return deg


def _median(values):
    vals = sorted(values)
    n = len(vals)
    if n == 0:
        return None
    if n % 2:
        return vals[n // 2]
    return 0.5 * (vals[n // 2 - 1] + vals[n // 2])


def _circular_mean(degs):
    if not degs:
        return None
    s = sum(math.sin(math.radians(d)) for d in degs)
    c = sum(math.cos(math.radians(d)) for d in degs)
    if abs(s) < 1e-9 and abs(c) < 1e-9:
        return degs[0]
    return (math.degrees(math.atan2(s, c)) + 360.0) % 360.0


def _trimmed_mean(values, tol):
    if not values:
        return None
    if len(values) == 1:
        return values[0]
    med = _median(values)
    close = [v for v in values if abs(v - med) <= tol]
    if close:
        return sum(close) / len(close)
    return med


def _voted_sensor_triplet(state, prefix, suffix, valid_suffix="_valid", tol=0.0, circular=False):
    candidates = []
    for i in (1, 2, 3):
        value = state.get(f"{prefix}_{i}{suffix}")
        valid = state.get(f"{prefix}_{i}{valid_suffix}", True)
        if value is None:
            continue
        if isinstance(valid, bool) and not valid:
            continue
        candidates.append(float(value))
    if not candidates:
        return None
    if circular:
        if len(candidates) == 1:
            return candidates[0] % 360.0
        best = None
        best_support = -1
        for cand in candidates:
            support = sum(1 for other in candidates if abs(_ang_norm(other - cand)) <= tol)
            if support > best_support:
                best_support = support
                best = cand
        inliers = [v for v in candidates if abs(_ang_norm(v - best)) <= tol]
        return _circular_mean(inliers if inliers else [best])
    return _trimmed_mean(candidates, tol)


def _best_altitude(state):
    radar = _voted_sensor_triplet(state, "radar_alt", "_ft", tol=75.0)
    baro = _voted_sensor_triplet(state, "baro_alt", "_ft", tol=200.0)
    gps = _voted_sensor_triplet(state, "gps", "_alt_ft", tol=250.0)
    pressure = _voted_sensor_triplet(state, "adc", "_pressure_alt_ft", tol=250.0)

    airborne = not any(
        bool(state.get(k, False))
        for k in ("weight_on_wheels_left", "weight_on_wheels_right", "weight_on_wheels_nose")
    )
    if radar is not None and radar < 2500.0:
        return radar
    candidates = []
    if baro is not None:
        candidates.append(baro)
    if pressure is not None:
        candidates.append(pressure)
    if gps is not None and (not airborne or radar is None or radar > 2000.0):
        candidates.append(gps)
    if candidates:
        return _trimmed_mean(candidates, 300.0)
    return radar if radar is not None else 0.0


def _best_vs(state, dt, alt_ft):
    irs_vs = _voted_sensor_triplet(state, "irs", "_vs_fpm", tol=500.0)
    nav_vs = state.get("vertical_speed_fpm")
    derived = None
    if _MEM["last_alt_ft"] is not None and dt > 1e-3:
        derived = (alt_ft - _MEM["last_alt_ft"]) * 60.0 / dt
    values = []
    if irs_vs is not None:
        values.append(float(irs_vs))
    if nav_vs is not None:
        values.append(float(nav_vs))
    if derived is not None:
        values.append(float(derived))
    if values:
        return _trimmed_mean(values, 800.0)
    return 0.0


def _best_ias(state):
    ias = _voted_sensor_triplet(state, "ias", "_kt", tol=12.0)
    if ias is not None:
        return max(ias, 30.0)
    tas = _voted_sensor_triplet(state, "tas", "_kt", tol=18.0, valid_suffix="")
    return max(tas or 150.0, 30.0)


def _best_mach(state):
    vals = [state.get(f"mach_{i}") for i in (1, 2, 3)]
    vals = [float(v) for v in vals if v is not None and 0.0 < float(v) < 1.5]
    return _trimmed_mean(vals, 0.03) if vals else 0.0


def _best_aoa(state):
    aoa = _voted_sensor_triplet(state, "aoa", "_deg", tol=2.5)
    return aoa if aoa is not None else 5.0


def _best_pitch_roll(state):
    pitch = _voted_sensor_triplet(state, "irs", "_pitch_deg", tol=3.0)
    roll = _voted_sensor_triplet(state, "irs", "_roll_deg", tol=5.0)
    if pitch is None:
        pitch = _MEM["last_pitch_deg"]
    if roll is None:
        roll = _MEM["last_roll_deg"]
    return pitch, roll


def _best_heading_track(state):
    hdg_values = []
    trk_values = []
    for i in (1, 2, 3):
        if state.get(f"irs_{i}_valid", True):
            h = state.get(f"irs_{i}_heading_deg")
            t = state.get(f"irs_{i}_track_deg")
            if h is not None:
                hdg_values.append(float(h) % 360.0)
            if t is not None:
                trk_values.append(float(t) % 360.0)
        if state.get(f"gps_{i}_valid", True):
            gt = state.get(f"gps_{i}_track_deg")
            if gt is not None:
                trk_values.append(float(gt) % 360.0)
    nav_track = state.get("track_deg")
    if nav_track is not None:
        trk_values.append(float(nav_track) % 360.0)
    nav_hdg = state.get("true_heading_deg")
    if nav_hdg is not None:
        hdg_values.append(float(nav_hdg) % 360.0)

    hdg = _circular_mean(hdg_values) if hdg_values else 0.0
    trk = _circular_mean(trk_values) if trk_values else hdg
    return hdg, trk


def _best_rates(state):
    pitch_rate = _voted_sensor_triplet(state, "gyro", "_pitch_rate_dps", tol=1.5)
    roll_rate = _voted_sensor_triplet(state, "gyro", "_roll_rate_dps", tol=2.5)
    yaw_rate = _voted_sensor_triplet(state, "gyro", "_yaw_rate_dps", tol=2.5)
    return pitch_rate or 0.0, roll_rate or 0.0, yaw_rate or 0.0


def _engine_ok(state, idx):
    running = bool(state.get(f"engine_{idx}_running", False))
    fire = bool(state.get(f"engine_{idx}_fire", False))
    n1 = float(state.get(f"engine_{idx}_n1_pct", 0.0) or 0.0)
    oil = float(state.get(f"engine_{idx}_oil_press_psi", 0.0) or 0.0)
    vib = float(state.get(f"engine_{idx}_vibration", 0.0) or 0.0)
    return running and not fire and n1 > 25.0 and oil > 10.0 and vib < 8.0


def _pid_int(name, error, dt, limit):
    _MEM[name] = _clamp(_MEM[name] + error * dt, -limit, limit)
    return _MEM[name]


def autopilot_step(state: dict, dt: float) -> dict:
    dt = float(dt or state.get("dt") or 0.05)
    dt = _clamp(dt, 0.01, 0.2)

    sim_time = float(state.get("sim_time_s", 0.0) or 0.0)
    if _MEM["last_time"] is not None and sim_time + 1.0 < _MEM["last_time"]:
        for key in list(_MEM):
            if isinstance(_MEM[key], bool):
                _MEM[key] = False
            elif _MEM[key] is None:
                pass
            else:
                _MEM[key] = 0.0
        _MEM["last_time"] = None
        _MEM["last_alt_ft"] = None
        _MEM["last_track_deg"] = None
        _MEM["last_ias_kt"] = None

    alt_ft = _best_altitude(state)
    vs_fpm = _best_vs(state, dt, alt_ft)
    ias_kt = _best_ias(state)
    mach = _best_mach(state)
    aoa_deg = _best_aoa(state)
    pitch_deg, roll_deg = _best_pitch_roll(state)
    hdg_deg, track_deg = _best_heading_track(state)
    pitch_rate_dps, roll_rate_dps, yaw_rate_dps = _best_rates(state)

    gross_weight = float(state.get("gross_weight_kg", 65000.0) or 65000.0)
    weight_factor = _clamp((gross_weight - 55000.0) / 25000.0, 0.0, 1.0)

    target_alt = float(state.get("target_alt_ft", alt_ft) or alt_ft)
    target_hdg = float(state.get("target_hdg_deg", hdg_deg) or hdg_deg) % 360.0
    target_ias = float(state.get("target_ias_kt", ias_kt) or ias_kt)
    target_vs = float(state.get("target_vs_fpm", 0.0) or 0.0)

    on_ground = any(
        bool(state.get(k, False))
        for k in ("weight_on_wheels_left", "weight_on_wheels_right", "weight_on_wheels_nose")
    )
    radar_agl = _voted_sensor_triplet(state, "radar_alt", "_ft", tol=75.0)
    gear_down_required = on_ground or target_alt < 1800.0 or (radar_agl is not None and radar_agl < 1500.0)

    flap_actual = max(
        float(state.get("flap_pos_left_deg", 0.0) or 0.0),
        float(state.get("flap_pos_right_deg", 0.0) or 0.0),
    )
    slat_actual = max(
        float(state.get("slat_pos_left_deg", 0.0) or 0.0),
        float(state.get("slat_pos_right_deg", 0.0) or 0.0),
    )

    engine1_ok = _engine_ok(state, 1)
    engine2_ok = _engine_ok(state, 2)
    thrust1 = float(state.get("engine_1_thrust_pct", 0.0) or 0.0)
    thrust2 = float(state.get("engine_2_thrust_pct", 0.0) or 0.0)
    thrust_asym = (thrust1 - thrust2) / 100.0

    alpha_prot = 10.0 + 1.0 * weight_factor
    alpha_max = 13.5 + 1.5 * weight_factor
    stall_margin = min(ias_kt - (128.0 + 12.0 * weight_factor + 0.25 * flap_actual), alpha_prot - aoa_deg)
    overspeed = ias_kt > 350.0 or mach > 0.80
    severe_overspeed = ias_kt > 362.0 or mach > 0.82

    alt_error = target_alt - alt_ft
    hdg_error = _ang_norm(target_hdg - hdg_deg)
    track_error = _ang_norm(target_hdg - track_deg)
    ias_error = target_ias - ias_kt

    if target_alt - alt_ft > 250.0:
        commanded_vs = min(target_vs if target_vs > 100.0 else 1800.0, 2500.0)
    elif alt_ft - target_alt > 250.0:
        commanded_vs = max(target_vs if target_vs < -100.0 else -1800.0, -2500.0)
    else:
        commanded_vs = _clamp(target_vs + 0.18 * alt_error, -1200.0, 1200.0)

    if stall_margin < 12.0:
        commanded_vs = min(commanded_vs, 500.0)
    if stall_margin < 6.0:
        commanded_vs = min(commanded_vs, 0.0)
    if severe_overspeed:
        commanded_vs = max(commanded_vs, -2500.0)

    max_bank = 25.0
    if gear_down_required or flap_actual > 5.0 or slat_actual > 5.0:
        max_bank = 18.0
    if stall_margin < 8.0:
        max_bank = min(max_bank, 12.0)
    if not engine1_ok or not engine2_ok:
        max_bank = min(max_bank, 15.0)

    bank_cmd = _clamp(0.85 * track_error + 0.35 * hdg_error, -max_bank, max_bank)
    if abs(track_error) < 2.0 and abs(hdg_error) < 4.0:
        bank_cmd *= 0.6

    hdg_int = _pid_int("hdg_int", _clamp(track_error / 30.0, -1.0, 1.0), dt, 10.0)
    aileron = 0.045 * (bank_cmd - roll_deg) - 0.035 * roll_rate_dps + 0.015 * hdg_int
    aileron = _clamp(aileron, -1.0, 1.0)

    vs_error = commanded_vs - vs_fpm
    vs_int = _pid_int("vs_int", _clamp(vs_error / 1500.0, -1.0, 1.0), dt, 8.0)
    target_pitch = _clamp(
        2.5 + 0.0042 * commanded_vs + 0.0022 * alt_error - 0.012 * max(-ias_error, 0.0),
        -8.0,
        16.0,
    )

    if overspeed:
        target_pitch = max(target_pitch, 1.5 if severe_overspeed else 0.0)
    if stall_margin < 10.0:
        target_pitch = min(target_pitch, 7.0)
    if aoa_deg > alpha_prot:
        target_pitch = min(target_pitch, 3.0 - 0.6 * max(0.0, aoa_deg - alpha_prot))
    if aoa_deg > alpha_max:
        target_pitch = -2.0
    if radar_agl is not None and radar_agl < 60.0 and target_alt < alt_ft + 200.0:
        flare = _clamp((60.0 - radar_agl) / 60.0, 0.0, 1.0)
        _MEM["flare_bias"] = flare
        target_pitch = min(target_pitch, 4.0 - 2.0 * flare)
        commanded_vs = max(commanded_vs, -700.0 + 500.0 * flare)
    else:
        _MEM["flare_bias"] *= 0.92

    pitch_error = target_pitch - pitch_deg
    elevator = 0.09 * pitch_error + 0.00038 * vs_error - 0.05 * pitch_rate_dps + 0.015 * vs_int
    if aoa_deg > alpha_prot:
        elevator -= 0.10 * (aoa_deg - alpha_prot)
    if severe_overspeed:
        elevator += 0.16
    elevator = _clamp(elevator, -1.0, 1.0)

    trim_int = _pid_int("trim_int", _clamp(pitch_error / 8.0, -1.0, 1.0), dt, 12.0)
    elevator_trim = _clamp(0.06 * trim_int - 0.015 * pitch_rate_dps, -1.0, 1.0)

    yaw_damper = -0.06 * yaw_rate_dps - 0.18 * thrust_asym
    if not engine1_ok and engine2_ok:
        yaw_damper += 0.22
    if not engine2_ok and engine1_ok:
        yaw_damper -= 0.22
    rudder = _clamp(yaw_damper + 0.04 * _clamp(roll_deg, -10.0, 10.0), -1.0, 1.0)

    ias_int = _pid_int("ias_int", _clamp(ias_error / 40.0, -1.0, 1.0), dt, 10.0)
    throttle_base = 0.52 + 0.0016 * alt_error + 0.00055 * commanded_vs + 0.018 * ias_error + 0.025 * ias_int
    throttle_base += 0.06 * weight_factor
    if overspeed:
        throttle_base -= 0.35 if severe_overspeed else 0.18
    if stall_margin < 10.0:
        throttle_base += 0.18
    if aoa_deg > alpha_prot:
        throttle_base += 0.12

    throttle_base = _clamp(throttle_base, 0.0, 1.0)
    throttle_1 = throttle_base
    throttle_2 = throttle_base
    if not engine1_ok and engine2_ok:
        throttle_1 = 0.0
        throttle_2 = max(throttle_2, 0.72)
    elif not engine2_ok and engine1_ok:
        throttle_2 = 0.0
        throttle_1 = max(throttle_1, 0.72)

    if severe_overspeed:
        speedbrake = 1.0 if alt_ft > 1000.0 and not gear_down_required else 0.5
    elif overspeed:
        speedbrake = 0.45 if alt_ft > 1000.0 and not gear_down_required else 0.2
    else:
        speedbrake = 0.0
    if stall_margin < 12.0 or aoa_deg > alpha_prot or target_alt > alt_ft + 100.0:
        speedbrake = 0.0

    if on_ground:
        if ias_kt < 145.0:
            flaps = 15
        elif ias_kt < 175.0:
            flaps = 5
        else:
            flaps = 0
    elif gear_down_required:
        if ias_kt < 165.0:
            flaps = 20
        elif ias_kt < 190.0:
            flaps = 10
        elif ias_kt < 215.0:
            flaps = 5
        else:
            flaps = 0
    elif stall_margin < 8.0 and ias_kt < 210.0:
        flaps = 5
    else:
        flaps = 0

    if ias_kt > 230.0:
        flaps = 0
    if severe_overspeed:
        flaps = 0

    gear = 1 if gear_down_required else 0

    _MEM["last_time"] = sim_time
    _MEM["last_alt_ft"] = alt_ft
    _MEM["last_track_deg"] = track_deg
    _MEM["last_ias_kt"] = ias_kt
    _MEM["last_pitch_deg"] = pitch_deg
    _MEM["last_roll_deg"] = roll_deg
    _MEM["initialized"] = True

    return {
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": _clamp(throttle_1, 0.0, 1.0),
        "throttle_2": _clamp(throttle_2, 0.0, 1.0),
        "elevator_trim": _clamp(elevator_trim, -1.0, 1.0),
        "flaps": int(_clamp(flaps, 0, 40)),
        "speedbrake": _clamp(speedbrake, 0.0, 1.0),
        "gear": int(gear),
    }
