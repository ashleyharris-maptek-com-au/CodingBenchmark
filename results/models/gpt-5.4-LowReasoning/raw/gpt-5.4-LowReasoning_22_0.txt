using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.In);
        int width = scanner.NextInt();
        int height = scanner.NextInt();
        int numPieces = scanner.NextInt();

        var game = new TetrisAI(width, height, numPieces);

        for (int i = 0; i < numPieces; i++)
        {
            string pieceToken = scanner.NextToken();
            if (pieceToken == null)
            {
                break;
            }

            char pieceType = pieceToken[0];
            Move move = game.ChooseMove(pieceType);

            Console.WriteLine(
                move.Rotation.ToString(CultureInfo.InvariantCulture) + " " +
                move.Column.ToString(CultureInfo.InvariantCulture));
            Console.Out.Flush();

            string result = scanner.NextToken();
            if (result == null)
            {
                break;
            }

            game.ApplyReportedResult(pieceType, move, result);
            if (result.Equals("gameover", StringComparison.OrdinalIgnoreCase))
            {
                break;
            }
        }
    }
}

internal sealed class TetrisAI
{
    private readonly int width;
    private readonly int height;
    private readonly int totalPieces;
    private readonly ulong[] rows;
    private readonly Dictionary<char, PieceOrientation[]> pieceMap;
    private int piecesPlayed;

    public TetrisAI(int width, int height, int totalPieces)
    {
        this.width = width;
        this.height = height;
        this.totalPieces = totalPieces;
        rows = new ulong[height];
        pieceMap = PieceLibrary.Build();
    }

    public Move ChooseMove(char pieceType)
    {
        PieceOrientation[] orientations = pieceMap[pieceType];
        int[] skyline = ComputeHeights(rows);
        bool danger = IsDangerous(skyline);
        bool wideBoard = width >= 14;
        bool tallBoard = height >= 28;
        bool lateGame = totalPieces > 0 && piecesPlayed > (totalPieces * 3) / 4;

        int candidateCount = danger ? 8 : (wideBoard || tallBoard ? 4 : 6);
        bool useBeam = width >= 12 || totalPieces >= 400 || lateGame;

        Move bestMove = default;
        bool found = false;
        double bestScore = double.NegativeInfinity;
        var candidates = new List<ScoredMove>(32);

        foreach (PieceOrientation orientation in orientations)
        {
            int maxColumn = width - orientation.Width;
            for (int column = 0; column <= maxColumn; column++)
            {
                SimulationResult sim = Simulate(rows, orientation, column);
                if (!sim.Success)
                {
                    continue;
                }

                double score = EvaluateBoard(sim.Rows, sim.LinesCleared, danger, useBeam: false);
                var move = new Move(orientation.Rotation, column);
                candidates.Add(new ScoredMove(move, score, sim));

                if (score > bestScore)
                {
                    bestScore = score;
                    bestMove = move;
                    found = true;
                }
            }
        }

        if (!found)
        {
            return FallbackMove(pieceType);
        }

        if (ShouldUseLookahead(danger, wideBoard, tallBoard))
        {
            candidates.Sort((a, b) => b.Score.CompareTo(a.Score));
            int limit = Math.Min(candidateCount, candidates.Count);
            double bestDeepScore = double.NegativeInfinity;
            Move bestDeepMove = bestMove;

            for (int i = 0; i < limit; i++)
            {
                var candidate = candidates[i];
                double deepScore = CandidateFutureScore(candidate.Simulation.Rows, danger, useBeam);
                double combined = candidate.Score + deepScore;

                if (combined > bestDeepScore)
                {
                    bestDeepScore = combined;
                    bestDeepMove = candidate.Move;
                }
            }

            bestMove = bestDeepMove;
        }

        return bestMove;
    }

    public void ApplyReportedResult(char pieceType, Move move, string result)
    {
        PieceOrientation orientation = GetOrientation(pieceType, move.Rotation);
        SimulationResult sim = Simulate(rows, orientation, move.Column);
        if (!sim.Success)
        {
            piecesPlayed++;
            return;
        }

        Array.Copy(sim.Rows, rows, height);
        piecesPlayed++;
    }

    private double CandidateFutureScore(ulong[] board, bool danger, bool useBeam)
    {
        double total = 0.0;
        char[] priority = danger
            ? new[] { 'I', 'T', 'L', 'J', 'O', 'S', 'Z' }
            : new[] { 'I', 'O', 'T', 'L', 'J', 'S', 'Z' };

        int sampleCount = useBeam ? 4 : 7;
        for (int i = 0; i < sampleCount; i++)
        {
            char piece = priority[i];
            double best = double.NegativeInfinity;
            foreach (PieceOrientation orientation in pieceMap[piece])
            {
                int maxColumn = width - orientation.Width;
                for (int column = 0; column <= maxColumn; column++)
                {
                    SimulationResult sim = Simulate(board, orientation, column);
                    if (!sim.Success)
                    {
                        continue;
                    }

                    double score = EvaluateBoard(sim.Rows, sim.LinesCleared, danger, useBeam: true);
                    if (score > best)
                    {
                        best = score;
                    }
                }
            }

            if (double.IsNegativeInfinity(best))
            {
                best = -100000.0;
            }

            total += best;
        }

        return total / sampleCount * 0.35;
    }

    private bool ShouldUseLookahead(bool danger, bool wideBoard, bool tallBoard)
    {
        if (danger)
        {
            return true;
        }

        if (wideBoard && tallBoard)
        {
            return false;
        }

        if (width * height >= 350)
        {
            return false;
        }

        return true;
    }

    private bool IsDangerous(int[] heights)
    {
        int maxHeight = 0;
        int roughness = 0;
        for (int i = 0; i < heights.Length; i++)
        {
            if (heights[i] > maxHeight)
            {
                maxHeight = heights[i];
            }

            if (i + 1 < heights.Length)
            {
                roughness += Math.Abs(heights[i] - heights[i + 1]);
            }
        }

        return maxHeight >= height - 6 || roughness >= width * 3;
    }

    private Move FallbackMove(char pieceType)
    {
        PieceOrientation orientation = pieceMap[pieceType][0];
        int column = Math.Max(0, Math.Min(width - orientation.Width, width / 2 - orientation.Width / 2));
        return new Move(orientation.Rotation, column);
    }

    private PieceOrientation GetOrientation(char pieceType, int rotation)
    {
        foreach (PieceOrientation orientation in pieceMap[pieceType])
        {
            if (orientation.Rotation == rotation)
            {
                return orientation;
            }
        }

        return pieceMap[pieceType][0];
    }

    private int[] ComputeHeights(ulong[] board)
    {
        int[] heights = new int[width];
        for (int c = 0; c < width; c++)
        {
            int h = 0;
            for (int r = height - 1; r >= 0; r--)
            {
                if (((board[r] >> c) & 1UL) != 0)
                {
                    h = r + 1;
                    break;
                }
            }

            heights[c] = h;
        }

        return heights;
    }

    private double EvaluateBoard(ulong[] board, int linesCleared, bool danger, bool useBeam)
    {
        int[] heights = ComputeHeights(board);
        int aggregateHeight = 0;
        int maxHeight = 0;
        int bumpiness = 0;
        int holes = 0;
        int coveredHoles = 0;
        int wells = 0;

        for (int c = 0; c < width; c++)
        {
            int h = heights[c];
            aggregateHeight += h;
            if (h > maxHeight)
            {
                maxHeight = h;
            }

            bool blockSeen = false;
            for (int r = height - 1; r >= 0; r--)
            {
                bool filled = ((board[r] >> c) & 1UL) != 0;
                if (filled)
                {
                    blockSeen = true;
                }
                else if (blockSeen)
                {
                    holes++;
                    coveredHoles += r + 1;
                }
            }

            int left = c == 0 ? height : heights[c - 1];
            int right = c == width - 1 ? height : heights[c + 1];
            int minSide = Math.Min(left, right);
            if (minSide > h)
            {
                wells += minSide - h;
            }

            if (c + 1 < width)
            {
                bumpiness += Math.Abs(heights[c] - heights[c + 1]);
            }
        }

        int rowTransitions = 0;
        int columnTransitions = 0;
        for (int r = 0; r < height; r++)
        {
            bool prev = true;
            for (int c = 0; c < width; c++)
            {
                bool filled = ((board[r] >> c) & 1UL) != 0;
                if (filled != prev)
                {
                    rowTransitions++;
                }

                prev = filled;
            }

            if (!prev)
            {
                rowTransitions++;
            }
        }

        for (int c = 0; c < width; c++)
        {
            bool prev = true;
            for (int r = height - 1; r >= 0; r--)
            {
                bool filled = ((board[r] >> c) & 1UL) != 0;
                if (filled != prev)
                {
                    columnTransitions++;
                }

                prev = filled;
            }

            if (!prev)
            {
                columnTransitions++;
            }
        }

        double lineWeight = danger ? 12.0 : 9.5;
        double heightWeight = danger ? -1.15 : -0.72;
        double holeWeight = danger ? -8.8 : -10.5;
        double coveredHoleWeight = danger ? -0.16 : -0.20;
        double bumpWeight = danger ? -0.48 : -0.34;
        double wellWeight = danger ? -0.28 : -0.18;
        double maxHeightWeight = danger ? -2.6 : -1.1;
        double transitionWeight = useBeam ? -0.08 : -0.12;

        return
            linesCleared * lineWeight +
            aggregateHeight * heightWeight +
            holes * holeWeight +
            coveredHoles * coveredHoleWeight +
            bumpiness * bumpWeight +
            wells * wellWeight +
            maxHeight * maxHeightWeight +
            (rowTransitions + columnTransitions) * transitionWeight;
    }

    private SimulationResult Simulate(ulong[] board, PieceOrientation orientation, int column)
    {
        int landingRow = FindLandingRow(board, orientation, column);
        if (landingRow < 0)
        {
            return SimulationResult.Failure;
        }

        ulong[] next = new ulong[height];
        Array.Copy(board, next, height);

        foreach (Cell cell in orientation.Cells)
        {
            int x = column + cell.X;
            int y = landingRow + cell.Y;
            if (y < 0 || y >= height || x < 0 || x >= width)
            {
                return SimulationResult.Failure;
            }

            next[y] |= 1UL << x;
        }

        int linesCleared = ClearLines(next);
        return new SimulationResult(true, next, linesCleared);
    }

    private int FindLandingRow(ulong[] board, PieceOrientation orientation, int column)
    {
        int lastValid = -1;
        int maxRow = height - orientation.Height;
        for (int row = maxRow; row >= 0; row--)
        {
            if (Collides(board, orientation, column, row))
            {
                continue;
            }

            lastValid = row;
            if (row == 0 || Collides(board, orientation, column, row - 1))
            {
                return row;
            }
        }

        return lastValid;
    }

    private bool Collides(ulong[] board, PieceOrientation orientation, int column, int row)
    {
        foreach (Cell cell in orientation.Cells)
        {
            int x = column + cell.X;
            int y = row + cell.Y;
            if (x < 0 || x >= width || y < 0 || y >= height)
            {
                return true;
            }

            if (((board[y] >> x) & 1UL) != 0)
            {
                return true;
            }
        }

        return false;
    }

    private int ClearLines(ulong[] board)
    {
        ulong fullMask = width == 64 ? ulong.MaxValue : ((1UL << width) - 1UL);
        int write = 0;
        int cleared = 0;

        for (int read = 0; read < height; read++)
        {
            if ((board[read] & fullMask) == fullMask)
            {
                cleared++;
                continue;
            }

            board[write++] = board[read];
        }

        while (write < height)
        {
            board[write++] = 0UL;
        }

        return cleared;
    }
}

internal readonly struct Move
{
    public Move(int rotation, int column)
    {
        Rotation = rotation;
        Column = column;
    }

    public int Rotation { get; }
    public int Column { get; }
}

internal readonly struct Cell
{
    public Cell(int x, int y)
    {
        X = x;
        Y = y;
    }

    public int X { get; }
    public int Y { get; }
}

internal readonly struct PieceOrientation
{
    public PieceOrientation(int rotation, Cell[] cells, int width, int height)
    {
        Rotation = rotation;
        Cells = cells;
        Width = width;
        Height = height;
    }

    public int Rotation { get; }
    public Cell[] Cells { get; }
    public int Width { get; }
    public int Height { get; }
}

internal readonly struct SimulationResult
{
    public static readonly SimulationResult Failure = new SimulationResult(false, null, 0);

    public SimulationResult(bool success, ulong[] rows, int linesCleared)
    {
        Success = success;
        Rows = rows;
        LinesCleared = linesCleared;
    }

    public bool Success { get; }
    public ulong[] Rows { get; }
    public int LinesCleared { get; }
}

internal readonly struct ScoredMove
{
    public ScoredMove(Move move, double score, SimulationResult simulation)
    {
        Move = move;
        Score = score;
        Simulation = simulation;
    }

    public Move Move { get; }
    public double Score { get; }
    public SimulationResult Simulation { get; }
}

internal static class PieceLibrary
{
    public static Dictionary<char, PieceOrientation[]> Build()
    {
        return new Dictionary<char, PieceOrientation[]>
        {
            ['I'] = new[]
            {
                Make(0, new[] { (0,0), (1,0), (2,0), (3,0) }),
                Make(1, new[] { (0,0), (0,1), (0,2), (0,3) })
            },
            ['O'] = new[]
            {
                Make(0, new[] { (0,0), (1,0), (0,1), (1,1) })
            },
            ['T'] = new[]
            {
                Make(0, new[] { (0,0), (1,0), (2,0), (1,1) }),
                Make(1, new[] { (0,0), (0,1), (0,2), (1,1) }),
                Make(2, new[] { (1,0), (0,1), (1,1), (2,1) }),
                Make(3, new[] { (1,0), (0,1), (1,1), (1,2) })
            },
            ['S'] = new[]
            {
                Make(0, new[] { (1,0), (2,0), (0,1), (1,1) }),
                Make(1, new[] { (0,0), (0,1), (1,1), (1,2) })
            },
            ['Z'] = new[]
            {
                Make(0, new[] { (0,0), (1,0), (1,1), (2,1) }),
                Make(1, new[] { (1,0), (0,1), (1,1), (0,2) })
            },
            ['J'] = new[]
            {
                Make(0, new[] { (0,0), (0,1), (1,1), (2,1) }),
                Make(1, new[] { (0,0), (1,0), (0,1), (0,2) }),
                Make(2, new[] { (0,0), (1,0), (2,0), (2,1) }),
                Make(3, new[] { (1,0), (1,1), (0,2), (1,2) })
            },
            ['L'] = new[]
            {
                Make(0, new[] { (2,0), (0,1), (1,1), (2,1) }),
                Make(1, new[] { (0,0), (0,1), (0,2), (1,2) }),
                Make(2, new[] { (0,0), (1,0), (2,0), (0,1) }),
                Make(3, new[] { (0,0), (1,0), (1,1), (1,2) })
            }
        };
    }

    private static PieceOrientation Make(int rotation, (int x, int y)[] coords)
    {
        int maxX = 0;
        int maxY = 0;
        var cells = new Cell[coords.Length];
        for (int i = 0; i < coords.Length; i++)
        {
            cells[i] = new Cell(coords[i].x, coords[i].y);
            if (coords[i].x > maxX)
            {
                maxX = coords[i].x;
            }

            if (coords[i].y > maxY)
            {
                maxY = coords[i].y;
            }
        }

        return new PieceOrientation(rotation, cells, maxX + 1, maxY + 1);
    }
}

internal sealed class FastScanner
{
    private readonly TextReader reader;
    private readonly char[] buffer;
    private int length;
    private int index;

    public FastScanner(TextReader reader)
    {
        this.reader = reader;
        buffer = new char[1 << 15];
    }

    public string NextToken()
    {
        int ch;
        do
        {
            ch = Read();
            if (ch < 0)
            {
                return null;
            }
        }
        while (char.IsWhiteSpace((char)ch));

        var chars = new List<char>(16);
        do
        {
            chars.Add((char)ch);
            ch = Read();
        }
        while (ch >= 0 && !char.IsWhiteSpace((char)ch));

        return new string(chars.ToArray());
    }

    public int NextInt()
    {
        string token = NextToken();
        return int.Parse(token, CultureInfo.InvariantCulture);
    }

    private int Read()
    {
        if (index >= length)
        {
            length = reader.Read(buffer, 0, buffer.Length);
            index = 0;
            if (length <= 0)
            {
                return -1;
            }
        }

        return buffer[index++];
    }
}
