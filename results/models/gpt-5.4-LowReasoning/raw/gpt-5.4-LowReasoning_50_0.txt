#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint n = counts.x;
    if (i >= n) {
        return;
    }

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    vec3 pos = inData[i * 2u].xyz;
    vec3 vel = inData[i * 2u + 1u].xyz;

    float sepRadius2 = sepRadius * sepRadius;
    float alignRadius2 = alignRadius * alignRadius;
    float cohRadius2 = cohRadius * cohRadius;

    vec3 sepSum = vec3(0.0);
    vec3 alignSum = vec3(0.0);
    vec3 cohSum = vec3(0.0);

    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < n; ++j) {
        if (j == i) {
            continue;
        }

        vec3 otherPos = inData[j * 2u].xyz;
        vec3 otherVel = inData[j * 2u + 1u].xyz;
        vec3 delta = otherPos - pos;
        float dist2 = dot(delta, delta);

        if (dist2 <= 0.0) {
            continue;
        }

        if (dist2 < sepRadius2) {
            float dist = sqrt(dist2);
            sepSum -= delta / dist;
            sepCount++;
        }

        if (dist2 < alignRadius2) {
            alignSum += otherVel;
            alignCount++;
        }

        if (dist2 < cohRadius2) {
            cohSum += otherPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        accel += (sepSum / float(sepCount)) * sepWeight;
    }

    if (alignCount > 0u) {
        vec3 avgVel = alignSum / float(alignCount);
        accel += (avgVel - vel) * alignWeight;
    }

    if (cohCount > 0u) {
        vec3 center = cohSum / float(cohCount);
        accel += (center - pos) * cohWeight;
    }

    float boundThreshold = boundSize * 0.4;
    float boundScale = boundSize * 0.5;

    if (pos.x > boundThreshold) {
        accel.x -= ((pos.x - boundThreshold) / boundScale) * boundForce;
    } else if (pos.x < -boundThreshold) {
        accel.x += ((-boundThreshold - pos.x) / boundScale) * boundForce;
    }

    if (pos.y > boundThreshold) {
        accel.y -= ((pos.y - boundThreshold) / boundScale) * boundForce;
    } else if (pos.y < -boundThreshold) {
        accel.y += ((-boundThreshold - pos.y) / boundScale) * boundForce;
    }

    if (pos.z > boundThreshold) {
        accel.z -= ((pos.z - boundThreshold) / boundScale) * boundForce;
    } else if (pos.z < -boundThreshold) {
        accel.z += ((-boundThreshold - pos.z) / boundScale) * boundForce;
    }

    vel += accel * dt;

    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = (vel / speed) * maxSpeed;
    }

    pos += vel * dt;

    outData[i * 2u] = vec4(pos, 0.0);
    outData[i * 2u + 1u] = vec4(vel, 0.0);
}
