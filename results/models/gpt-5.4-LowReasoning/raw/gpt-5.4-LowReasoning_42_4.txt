OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpSource GLSL 450
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.0399999991
%float_0_005 = OpConstant %float 0.00499999989
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_uint_2 = OpTypeArray %gl_PerVertex_in %uint_2
%_ptr_Input_arr_gl_PerVertex_in_uint_2 = OpTypePointer Input %arr_gl_PerVertex_in_uint_2
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in_uint_2 Input
%arr_v3float_uint_2 = OpTypeArray %v3float %uint_2
%_ptr_Input_arr_v3float_uint_2 = OpTypePointer Input %arr_v3float_uint_2
%worldPos = OpVariable %_ptr_Input_arr_v3float_uint_2 Input
%normal = OpVariable %_ptr_Input_arr_v3float_uint_2 Input
%color = OpVariable %_ptr_Input_arr_v3float_uint_2 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %_ptr_Output_v3float Output
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v4float = OpTypePointer Input %v4float
%main = OpFunction %void None %3
%5 = OpLabel
%p0_ptr = OpAccessChain %_ptr_Input_v3float %worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p1_ptr = OpAccessChain %_ptr_Input_v3float %worldPos %int_1
%p1 = OpLoad %v3float %p1_ptr
%c0_ptr = OpAccessChain %_ptr_Input_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %_ptr_Input_v3float %color %int_1
%c1 = OpLoad %v3float %c1_ptr
%gp0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%gp0 = OpLoad %v4float %gp0_ptr
%gp1_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%gp1 = OpLoad %v4float %gp1_ptr
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%delta = OpFSub %v2float %p1_xy %p0_xy
%dir = OpExtInst %v2float %1 Normalize %delta
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_start_xy = OpVectorTimesScalar %v2float %perp %float_0_04
%perp_end_xy = OpVectorTimesScalar %v2float %perp %float_0_005
%perp_start = OpCompositeConstruct %v3float %perp_start_xy %float_0
%perp_end = OpCompositeConstruct %v3float %perp_end_xy %float_0
%p0_plus = OpFAdd %v3float %p0 %perp_start
%p0_minus = OpFSub %v3float %p0 %perp_start
%p1_plus = OpFAdd %v3float %p1 %perp_end
%p1_minus = OpFSub %v3float %p1 %perp_end
%gp0_z = OpCompositeExtract %float %gp0 2
%gp0_w = OpCompositeExtract %float %gp0 3
%gp1_z = OpCompositeExtract %float %gp1 2
%gp1_w = OpCompositeExtract %float %gp1 3
%p0_plus_x = OpCompositeExtract %float %p0_plus 0
%p0_plus_y = OpCompositeExtract %float %p0_plus 1
%out_p0_plus = OpCompositeConstruct %v4float %p0_plus_x %p0_plus_y %gp0_z %gp0_w
%p0_minus_x = OpCompositeExtract %float %p0_minus 0
%p0_minus_y = OpCompositeExtract %float %p0_minus 1
%out_p0_minus = OpCompositeConstruct %v4float %p0_minus_x %p0_minus_y %gp0_z %gp0_w
%p1_plus_x = OpCompositeExtract %float %p1_plus 0
%p1_plus_y = OpCompositeExtract %float %p1_plus 1
%out_p1_plus = OpCompositeConstruct %v4float %p1_plus_x %p1_plus_y %gp1_z %gp1_w
%p1_minus_x = OpCompositeExtract %float %p1_minus 0
%p1_minus_y = OpCompositeExtract %float %p1_minus 1
%out_p1_minus = OpCompositeConstruct %v4float %p1_minus_x %p1_minus_y %gp1_z %gp1_w
%gl_out_pos = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_pos %out_p0_plus
OpStore %outColor %c0
OpEmitVertex
OpStore %gl_out_pos %out_p0_minus
OpStore %outColor %c0
OpEmitVertex
OpStore %gl_out_pos %out_p1_plus
OpStore %outColor %c1
OpEmitVertex
OpStore %gl_out_pos %out_p1_minus
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
