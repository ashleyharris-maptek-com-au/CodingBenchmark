use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone)]
struct HoleSample {
    hole_idx: usize,
    hole_id: String,
    sample_idx: usize,
    pos: [f64; 3],
    props: Vec<f64>,
}

#[derive(Clone)]
struct Suspicion {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

#[derive(Clone, Copy)]
struct GlobalStats {
    median: f64,
    mad: f64,
}

#[derive(Clone, Copy)]
enum SearchStrategy {
    Exact,
    Grid,
}

struct SpatialGrid {
    cell_size: f64,
    cells: HashMap<(i32, i32, i32), Vec<usize>>,
}

impl SpatialGrid {
    fn new(samples: &[HoleSample], cell_size: f64) -> Self {
        let mut cells: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
        for (idx, sample) in samples.iter().enumerate() {
            cells.entry(Self::cell_of(sample.pos, cell_size)).or_default().push(idx);
        }
        Self { cell_size, cells }
    }

    fn cell_of(pos: [f64; 3], cell_size: f64) -> (i32, i32, i32) {
        (
            (pos[0] / cell_size).floor() as i32,
            (pos[1] / cell_size).floor() as i32,
            (pos[2] / cell_size).floor() as i32,
        )
    }

    fn gather_neighbors(
        &self,
        samples: &[HoleSample],
        center_idx: usize,
        min_neighbors: usize,
        max_radius: i32,
    ) -> Vec<usize> {
        let center_cell = Self::cell_of(samples[center_idx].pos, self.cell_size);
        let mut found = Vec::new();

        for radius in 0..=max_radius {
            for dx in -radius..=radius {
                for dy in -radius..=radius {
                    for dz in -radius..=radius {
                        if radius > 0
                            && dx.abs() != radius
                            && dy.abs() != radius
                            && dz.abs() != radius
                        {
                            continue;
                        }
                        let key = (center_cell.0 + dx, center_cell.1 + dy, center_cell.2 + dz);
                        if let Some(bucket) = self.cells.get(&key) {
                            for &idx in bucket {
                                if idx != center_idx {
                                    found.push(idx);
                                }
                            }
                        }
                    }
                }
            }
            if found.len() >= min_neighbors {
                break;
            }
        }
        found
    }
}

fn parse_f64(token: Option<&str>) -> f64 {
    token
        .and_then(|t| t.parse::<f64>().ok())
        .filter(|v| v.is_finite())
        .unwrap_or(f64::NAN)
}

fn cmp_f64(a: &f64, b: &f64) -> Ordering {
    a.partial_cmp(b).unwrap_or(Ordering::Equal)
}

fn median_in_place(values: &mut [f64]) -> f64 {
    if values.is_empty() {
        return f64::NAN;
    }
    let mid = values.len() / 2;
    values.select_nth_unstable_by(mid, cmp_f64);
    let hi = values[mid];
    if values.len() % 2 == 1 {
        hi
    } else {
        let lo = values[..mid]
            .iter()
            .copied()
            .max_by(cmp_f64)
            .unwrap_or(hi);
        0.5 * (lo + hi)
    }
}

fn median_of(values: &[f64]) -> f64 {
    let mut tmp = values.to_vec();
    median_in_place(&mut tmp)
}

fn mad_of(values: &[f64], median: f64) -> f64 {
    if values.is_empty() || !median.is_finite() {
        return f64::NAN;
    }
    let mut devs = Vec::with_capacity(values.len());
    for &v in values {
        if v.is_finite() {
            devs.push((v - median).abs());
        }
    }
    median_in_place(&mut devs)
}

fn distance2(a: [f64; 3], b: [f64; 3]) -> f64 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

fn estimate_cell_size(samples: &[HoleSample]) -> f64 {
    if samples.len() < 2 {
        return 1.0;
    }

    let mut per_hole: HashMap<usize, Vec<(usize, [f64; 3])>> = HashMap::new();
    for s in samples {
        per_hole.entry(s.hole_idx).or_default().push((s.sample_idx, s.pos));
    }

    let mut spacings = Vec::new();
    for (_hole, pts) in per_hole.iter_mut() {
        pts.sort_by_key(|x| x.0);
        for w in pts.windows(2).take(64) {
            let d2 = distance2(w[0].1, w[1].1);
            if d2.is_finite() && d2 > 0.0 {
                spacings.push(d2.sqrt());
            }
        }
    }

    if !spacings.is_empty() {
        let m = median_of(&spacings);
        if m.is_finite() && m > 1e-9 {
            return m * 2.5;
        }
    }

    let mut minp = [f64::INFINITY; 3];
    let mut maxp = [f64::NEG_INFINITY; 3];
    for s in samples {
        for k in 0..3 {
            minp[k] = minp[k].min(s.pos[k]);
            maxp[k] = maxp[k].max(s.pos[k]);
        }
    }
    let dx = (maxp[0] - minp[0]).max(1.0);
    let dy = (maxp[1] - minp[1]).max(1.0);
    let dz = (maxp[2] - minp[2]).max(1.0);
    let volume = dx * dy * dz;
    let n = samples.len().max(1) as f64;
    (volume / n).cbrt().max(1.0)
}

fn k_nearest_exact(samples: &[HoleSample], center_idx: usize, k: usize) -> Vec<usize> {
    let center = samples[center_idx].pos;
    let mut pairs = Vec::with_capacity(samples.len().saturating_sub(1));
    for (idx, s) in samples.iter().enumerate() {
        if idx == center_idx {
            continue;
        }
        pairs.push((distance2(center, s.pos), idx));
    }
    pairs.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
    pairs.into_iter().take(k).map(|x| x.1).collect()
}

fn local_score(
    samples: &[HoleSample],
    property_idx: usize,
    target_idx: usize,
    neighbor_indices: &[usize],
    global: GlobalStats,
) -> Option<f64> {
    let target = samples[target_idx].props[property_idx];
    if !target.is_finite() {
        return None;
    }

    let mut vals = Vec::with_capacity(neighbor_indices.len());
    for &idx in neighbor_indices {
        let v = samples[idx].props[property_idx];
        if v.is_finite() {
            vals.push(v);
        }
    }
    if vals.len() < 4 {
        return None;
    }

    let local_med = median_of(&vals);
    let local_mad = mad_of(&vals, local_med);
    let scale = (local_mad * 1.4826)
        .max(global.mad * 1.4826 * 0.35)
        .max(1e-9);
    Some((target - local_med).abs() / scale)
}

fn along_hole_score(
    hole_samples: &[usize],
    samples: &[HoleSample],
    property_idx: usize,
    pos_in_hole: usize,
    global: GlobalStats,
) -> Option<f64> {
    let target_idx = hole_samples[pos_in_hole];
    let target = samples[target_idx].props[property_idx];
    if !target.is_finite() {
        return None;
    }

    let mut vals = Vec::with_capacity(4);
    let start = pos_in_hole.saturating_sub(2);
    let end = (pos_in_hole + 3).min(hole_samples.len());
    for i in start..end {
        if i == pos_in_hole {
            continue;
        }
        let v = samples[hole_samples[i]].props[property_idx];
        if v.is_finite() {
            vals.push(v);
        }
    }
    if vals.len() < 2 {
        return None;
    }
    let med = median_of(&vals);
    let mad = mad_of(&vals, med);
    let scale = (mad * 1.4826).max(global.mad * 1.4826 * 0.25).max(1e-9);
    Some((target - med).abs() / scale)
}

fn confidence_from_score(score: f64) -> f64 {
    let x = ((score - 3.0) / 3.5).clamp(-20.0, 20.0);
    1.0 / (1.0 + (-x).exp())
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let tokens: Vec<&str> = input.split_whitespace().collect();
    let mut at = 0usize;

    let num_holes = tokens.get(at).and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    at += 1;
    let num_properties = tokens.get(at).and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    at += 1;

    let mut property_names = Vec::with_capacity(num_properties);
    for i in 0..num_properties {
        property_names.push(
            tokens
                .get(at + i)
                .map(|s| s.to_string())
                .unwrap_or_else(|| format!("property_{}", i + 1)),
        );
    }
    at += num_properties;

    let mut samples: Vec<HoleSample> = Vec::new();
    let mut hole_to_samples: HashMap<usize, Vec<usize>> = HashMap::new();

    for hole_idx in 0..num_holes {
        let Some(hole_id_tok) = tokens.get(at) else { break };
        let hole_id = (*hole_id_tok).to_string();
        at += 1;

        let start = [
            parse_f64(tokens.get(at).copied()),
            parse_f64(tokens.get(at + 1).copied()),
            parse_f64(tokens.get(at + 2).copied()),
        ];
        at += 3;
        let dir = [
            parse_f64(tokens.get(at).copied()),
            parse_f64(tokens.get(at + 1).copied()),
            parse_f64(tokens.get(at + 2).copied()),
        ];
        at += 3;
        let length = parse_f64(tokens.get(at).copied());
        at += 1;
        let num_samples = tokens.get(at).and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
        at += 1;

        let norm = (dir[0] * dir[0] + dir[1] * dir[1] + dir[2] * dir[2]).sqrt();
        let unit = if norm.is_finite() && norm > 1e-12 {
            [dir[0] / norm, dir[1] / norm, dir[2] / norm]
        } else {
            [0.0, 0.0, 1.0]
        };

        for sample_idx in 0..num_samples {
            let depth = parse_f64(tokens.get(at).copied());
            at += 1;
            let d = if depth.is_finite() {
                if length.is_finite() && length > 0.0 {
                    depth.clamp(0.0, length)
                } else {
                    depth.max(0.0)
                }
            } else {
                0.0
            };
            let pos = [
                start[0] + unit[0] * d,
                start[1] + unit[1] * d,
                start[2] + unit[2] * d,
            ];

            let mut props = Vec::with_capacity(num_properties);
            for _ in 0..num_properties {
                props.push(parse_f64(tokens.get(at).copied()));
                at += 1;
            }

            let global_idx = samples.len();
            samples.push(HoleSample {
                hole_idx,
                hole_id: hole_id.clone(),
                sample_idx,
                pos,
                props,
            });
            hole_to_samples.entry(hole_idx).or_default().push(global_idx);
        }
    }

    if samples.is_empty() || num_properties == 0 {
        println!("0");
        return;
    }

    for idxs in hole_to_samples.values_mut() {
        idxs.sort_by_key(|&idx| samples[idx].sample_idx);
    }

    let strategy = if samples.len() <= 4000 {
        SearchStrategy::Exact
    } else {
        SearchStrategy::Grid
    };

    let grid = match strategy {
        SearchStrategy::Exact => None,
        SearchStrategy::Grid => Some(SpatialGrid::new(&samples, estimate_cell_size(&samples))),
    };

    let exact_k = if samples.len() <= 500 { 12 } else { 16 };
    let grid_min_neighbors = if samples.len() <= 20000 { 14 } else { 10 };
    let max_results = (samples.len() / 50).clamp(32, 20_000);

    let mut global_stats = Vec::with_capacity(num_properties);
    for p in 0..num_properties {
        let mut vals = Vec::new();
        for s in &samples {
            let v = s.props[p];
            if v.is_finite() {
                vals.push(v);
            }
        }
        if vals.len() < 3 {
            global_stats.push(GlobalStats { median: 0.0, mad: 1.0 });
            continue;
        }
        let med = median_of(&vals);
        let mad = mad_of(&vals, med);
        global_stats.push(GlobalStats {
            median: med,
            mad: if mad.is_finite() && mad > 1e-12 { mad } else { 1.0 },
        });
    }

    let mut suspicions = Vec::new();

    for p in 0..num_properties {
        let global = global_stats[p];
        let mut per_prop = Vec::new();

        for idx in 0..samples.len() {
            let value = samples[idx].props[p];
            if !value.is_finite() {
                continue;
            }

            let spatial_neighbors = match strategy {
                SearchStrategy::Exact => k_nearest_exact(&samples, idx, exact_k),
                SearchStrategy::Grid => {
                    let mut n = grid
                        .as_ref()
                        .unwrap()
                        .gather_neighbors(&samples, idx, grid_min_neighbors, 3);
                    if n.len() > 48 {
                        let center = samples[idx].pos;
                        n.sort_by(|&a, &b| {
                            distance2(center, samples[a].pos)
                                .partial_cmp(&distance2(center, samples[b].pos))
                                .unwrap_or(Ordering::Equal)
                        });
                        n.truncate(48);
                    }
                    n
                }
            };

            let spatial_score = local_score(&samples, p, idx, &spatial_neighbors, global);

            let hole_samples = &hole_to_samples[&samples[idx].hole_idx];
            let pos_in_hole = hole_samples
                .binary_search(&idx)
                .unwrap_or_else(|ins| ins.min(hole_samples.len().saturating_sub(1)));
            let continuity_score = along_hole_score(hole_samples, &samples, p, pos_in_hole, global);

            let global_z = ((value - global.median).abs() / (global.mad * 1.4826).max(1e-9)).min(1000.0);

            let combined = match (spatial_score, continuity_score) {
                (Some(a), Some(b)) => a.max(b * 0.9),
                (Some(a), None) => a,
                (None, Some(b)) => b,
                (None, None) => global_z * 0.6,
            };

            let support = spatial_neighbors.len() as f64;
            let support_factor = if support >= 10.0 {
                1.0
            } else {
                (support / 10.0).max(0.55)
            };
            let final_score = combined * support_factor + global_z * 0.15;

            if final_score >= 4.5 || (final_score >= 3.8 && global_z >= 6.0) {
                per_prop.push(Suspicion {
                    hole_id: samples[idx].hole_id.clone(),
                    sample_idx: samples[idx].sample_idx,
                    property_name: property_names[p].clone(),
                    confidence: confidence_from_score(final_score),
                });
            }
        }

        per_prop.sort_by(|a, b| {
            b.confidence
                .partial_cmp(&a.confidence)
                .unwrap_or(Ordering::Equal)
                .then_with(|| a.hole_id.cmp(&b.hole_id))
                .then_with(|| a.sample_idx.cmp(&b.sample_idx))
                .then_with(|| a.property_name.cmp(&b.property_name))
        });

        let keep = if samples.len() < 200 {
            per_prop.len()
        } else {
            per_prop.len().min(max_results / num_properties.max(1) + 8)
        };
        suspicions.extend(per_prop.into_iter().take(keep));
    }

    suspicions.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.property_name.cmp(&b.property_name))
    });

    let mut dedup: HashMap<(String, usize, String), f64> = HashMap::new();
    for s in suspicions {
        let key = (s.hole_id.clone(), s.sample_idx, s.property_name.clone());
        dedup
            .entry(key)
            .and_modify(|c| *c = c.max(s.confidence))
            .or_insert(s.confidence);
    }

    let mut final_list: Vec<Suspicion> = dedup
        .into_iter()
        .map(|((hole_id, sample_idx, property_name), confidence)| Suspicion {
            hole_id,
            sample_idx,
            property_name,
            confidence,
        })
        .collect();

    final_list.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.property_name.cmp(&b.property_name))
    });

    println!("{}", final_list.len());
    for s in final_list {
        println!(
            "{} {} {} {:.6}",
            s.hole_id,
            s.sample_idx,
            s.property_name,
            s.confidence.clamp(0.0, 0.999999)
        );
    }
}
