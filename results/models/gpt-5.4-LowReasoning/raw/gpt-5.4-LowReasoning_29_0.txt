using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var edgesU = new int[m];
        var edgesV = new int[m];
        var degree = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            edgesU[i] = u;
            edgesV[i] = v;
            if (u == v) continue;
            degree[u]++;
            degree[v]++;
        }

        var adj = new int[n][];
        for (int i = 0; i < n; i++) adj[i] = new int[degree[i]];
        var ptr = new int[n];
        for (int i = 0; i < m; i++)
        {
            int u = edgesU[i];
            int v = edgesV[i];
            if (u == v) continue;
            adj[u][ptr[u]++] = v;
            adj[v][ptr[v]++] = u;
        }
        for (int i = 0; i < n; i++) Array.Sort(adj[i]);

        var solver = new MaximumCliqueSolver(n, adj, degree);
        var answer = solver.Solve();

        Console.WriteLine(answer.Count);
        if (answer.Count > 0) Console.WriteLine(string.Join(" ", answer));
        else Console.WriteLine();
    }
}

public sealed class MaximumCliqueSolver
{
    private readonly int n;
    private readonly int[][] adj;
    private readonly int[] degree;

    public MaximumCliqueSolver(int n, int[][] adj, int[] degree)
    {
        this.n = n;
        this.adj = adj;
        this.degree = degree;
    }

    public List<int> Solve()
    {
        var best = BuildGreedyCliqueAllVertices();
        int lowerBound = best.Count;

        if (n <= 6000)
        {
            var exact = SolveExactOnWholeGraph(lowerBound);
            if (exact.Count > best.Count) best = exact;
            return best;
        }

        var core = ComputeCoreNumbers(out int maxCore);
        if (maxCore + 1 <= best.Count) return best;

        var kernel = BuildKernel(core, best.Count);
        if (kernel.Length > 0)
        {
            var improved = ImproveInsideKernel(kernel, best.Count);
            if (improved.Count > best.Count) best = improved;
        }

        return best;
    }

    private List<int> BuildGreedyCliqueAllVertices()
    {
        var order = Enumerable.Range(0, n).ToArray();
        Array.Sort(order, (a, b) =>
        {
            int cmp = degree[b].CompareTo(degree[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        var best = new List<int>();
        int[] marks = new int[n];
        int token = 1;

        int seeds = Math.Min(order.Length, n <= 100000 ? order.Length : Math.Min(order.Length, 4096));
        for (int si = 0; si < seeds; si++)
        {
            int start = order[si];
            var clique = new List<int> { start };
            token++;
            foreach (int x in adj[start]) marks[x] = token;

            var cand = new List<int>();
            foreach (int x in adj[start])
            {
                if (degree[x] + 1 >= best.Count) cand.Add(x);
            }
            cand.Sort((a, b) =>
            {
                int cmp = degree[b].CompareTo(degree[a]);
                return cmp != 0 ? cmp : a.CompareTo(b);
            });

            foreach (int v in cand)
            {
                if (marks[v] != token) continue;
                bool ok = true;
                foreach (int u in clique)
                {
                    if (!HasEdge(u, v))
                    {
                        ok = false;
                        break;
                    }
                }
                if (!ok) continue;
                clique.Add(v);
                token++;
                foreach (int x in adj[v])
                {
                    if (marks[x] == token - 1) marks[x] = token;
                }
                for (int i = 0; i < clique.Count - 1; i++) marks[clique[i]] = token;
            }

            if (clique.Count > best.Count) best = clique;
        }

        if (best.Count == 0) best.Add(order[0]);
        return best;
    }

    private int[] ComputeCoreNumbers(out int maxCore)
    {
        int[] deg = new int[n];
        Array.Copy(degree, deg, n);
        int maxDeg = 0;
        for (int i = 0; i < n; i++) if (deg[i] > maxDeg) maxDeg = deg[i];

        int[] bin = new int[maxDeg + 1];
        for (int i = 0; i < n; i++) bin[deg[i]]++;
        int start = 0;
        for (int d = 0; d <= maxDeg; d++)
        {
            int num = bin[d];
            bin[d] = start;
            start += num;
        }

        int[] pos = new int[n];
        int[] vert = new int[n];
        for (int v = 0; v < n; v++)
        {
            pos[v] = bin[deg[v]];
            vert[pos[v]] = v;
            bin[deg[v]]++;
        }
        for (int d = maxDeg; d >= 1; d--) bin[d] = bin[d - 1];
        bin[0] = 0;

        int[] core = new int[n];
        maxCore = 0;
        for (int i = 0; i < n; i++)
        {
            int v = vert[i];
            core[v] = deg[v];
            if (core[v] > maxCore) maxCore = core[v];
            foreach (int u in adj[v])
            {
                if (deg[u] > deg[v])
                {
                    int du = deg[u];
                    int pu = pos[u];
                    int pw = bin[du];
                    int w = vert[pw];
                    if (u != w)
                    {
                        pos[u] = pw;
                        pos[w] = pu;
                        vert[pu] = w;
                        vert[pw] = u;
                    }
                    bin[du]++;
                    deg[u]--;
                }
            }
        }
        return core;
    }

    private int[] BuildKernel(int[] core, int bestSize)
    {
        var list = new List<int>();
        int needed = bestSize - 1;
        for (int v = 0; v < n; v++)
        {
            if (core[v] >= needed) list.Add(v);
        }

        if (list.Count <= 1) return list.ToArray();
        if (list.Count > 7000)
        {
            list.Sort((a, b) =>
            {
                int cmp = core[b].CompareTo(core[a]);
                if (cmp != 0) return cmp;
                cmp = degree[b].CompareTo(degree[a]);
                return cmp != 0 ? cmp : a.CompareTo(b);
            });
            list.RemoveRange(7000, list.Count - 7000);
        }
        return list.ToArray();
    }

    private List<int> ImproveInsideKernel(int[] kernel, int lowerBound)
    {
        var id = new Dictionary<int, int>(kernel.Length);
        for (int i = 0; i < kernel.Length; i++) id[kernel[i]] = i;

        var subAdjList = new List<int>[kernel.Length];
        for (int i = 0; i < kernel.Length; i++) subAdjList[i] = new List<int>();
        for (int i = 0; i < kernel.Length; i++)
        {
            int v = kernel[i];
            foreach (int u in adj[v])
            {
                if (id.TryGetValue(u, out int j)) subAdjList[i].Add(j);
            }
        }

        if (kernel.Length <= 4096)
        {
            var exact = SolveExactSubgraph(kernel, subAdjList, lowerBound);
            if (exact.Count > lowerBound) return exact;
        }

        return BuildGreedyCliqueInSubgraph(kernel, subAdjList);
    }

    private List<int> BuildGreedyCliqueInSubgraph(int[] kernel, List<int>[] subAdjList)
    {
        int k = kernel.Length;
        var order = Enumerable.Range(0, k).ToArray();
        Array.Sort(order, (a, b) =>
        {
            int cmp = subAdjList[b].Count.CompareTo(subAdjList[a].Count);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        var best = new List<int>();
        bool[] mark = new bool[k];

        int seeds = Math.Min(k, 512);
        for (int si = 0; si < seeds; si++)
        {
            int s = order[si];
            Array.Clear(mark, 0, k);
            foreach (int x in subAdjList[s]) mark[x] = true;
            var clique = new List<int> { s };

            var cand = subAdjList[s].ToArray();
            Array.Sort(cand, (a, b) =>
            {
                int cmp = subAdjList[b].Count.CompareTo(subAdjList[a].Count);
                return cmp != 0 ? cmp : a.CompareTo(b);
            });

            foreach (int v in cand)
            {
                if (!mark[v]) continue;
                bool ok = true;
                foreach (int u in clique)
                {
                    if (!subAdjList[u].Contains(v) && u != v)
                    {
                        ok = false;
                        break;
                    }
                }
                if (!ok) continue;
                clique.Add(v);
                var nextMark = new bool[k];
                foreach (int x in subAdjList[v]) if (mark[x]) nextMark[x] = true;
                mark = nextMark;
            }

            if (clique.Count > best.Count)
            {
                best.Clear();
                foreach (int x in clique) best.Add(kernel[x]);
            }
        }

        return best;
    }

    private List<int> SolveExactOnWholeGraph(int lowerBound)
    {
        int[] vertices = Enumerable.Range(0, n).ToArray();
        var subAdjList = new List<int>[n];
        for (int i = 0; i < n; i++) subAdjList[i] = new List<int>(adj[i]);
        return SolveExactSubgraph(vertices, subAdjList, lowerBound);
    }

    private List<int> SolveExactSubgraph(int[] originalVertices, List<int>[] subAdjList, int lowerBound)
    {
        int k = originalVertices.Length;
        var neighborBits = new ulong[k][];
        int words = (k + 63) >> 6;
        for (int i = 0; i < k; i++) neighborBits[i] = new ulong[words];

        for (int i = 0; i < k; i++)
        {
            foreach (int j in subAdjList[i])
            {
                neighborBits[i][j >> 6] |= 1UL << (j & 63);
            }
        }

        var solver = new ExactCliqueBitset(k, neighborBits, originalVertices, lowerBound);
        return solver.Run();
    }

    private bool HasEdge(int u, int v)
    {
        var arr = adj[u];
        int lo = 0, hi = arr.Length - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            int x = arr[mid];
            if (x == v) return true;
            if (x < v) lo = mid + 1;
            else hi = mid - 1;
        }
        return false;
    }
}

public sealed class ExactCliqueBitset
{
    private readonly int n;
    private readonly ulong[][] neighbors;
    private readonly int[] original;
    private readonly int words;
    private int bestSize;
    private int[] bestClique;
    private readonly int[] current;
    private readonly ulong[] allMask;

    public ExactCliqueBitset(int n, ulong[][] neighbors, int[] originalVertices, int lowerBound)
    {
        this.n = n;
        this.neighbors = neighbors;
        original = originalVertices;
        words = (n + 63) >> 6;
        bestSize = lowerBound;
        bestClique = new int[Math.Max(1, n)];
        current = new int[Math.Max(1, n)];
        allMask = new ulong[words];
        for (int i = 0; i < words; i++) allMask[i] = ulong.MaxValue;
        int rem = n & 63;
        if (rem != 0) allMask[words - 1] = (1UL << rem) - 1;
    }

    public List<int> Run()
    {
        var candidates = new ulong[words];
        Array.Copy(allMask, candidates, words);
        Expand(candidates, 0);

        var ans = new List<int>(bestSize);
        for (int i = 0; i < bestSize; i++) ans.Add(original[bestClique[i]]);
        ans.Sort();
        return ans;
    }

    private void Expand(ulong[] cand, int size)
    {
        int candCount = PopCount(cand);
        if (size + candCount <= bestSize) return;
        if (candCount == 0)
        {
            if (size > bestSize)
            {
                bestSize = size;
                Array.Copy(current, bestClique, size);
            }
            return;
        }

        int[] order, colors;
        ColorSort(cand, out order, out colors);

        for (int idx = order.Length - 1; idx >= 0; idx--)
        {
            if (size + colors[idx] <= bestSize) return;
            int v = order[idx];
            if (!GetBit(cand, v)) continue;

            current[size] = v;
            var next = Intersect(cand, neighbors[v]);
            ClearBit(cand, v);

            if (size + 1 > bestSize && PopCount(next) == 0)
            {
                bestSize = size + 1;
                Array.Copy(current, bestClique, size + 1);
            }
            else
            {
                Expand(next, size + 1);
            }
        }
    }

    private void ColorSort(ulong[] cand, out int[] order, out int[] colors)
    {
        var left = (ulong[])cand.Clone();
        var ord = new List<int>();
        var col = new List<int>();
        int color = 0;

        while (PopCount(left) > 0)
        {
            color++;
            var available = (ulong[])left.Clone();
            while (PopCount(available) > 0)
            {
                int v = FirstBit(available);
                ord.Add(v);
                col.Add(color);
                ClearBit(left, v);
                ClearBit(available, v);

                for (int w = 0; w < words; w++)
                {
                    available[w] &= ~neighbors[v][w];
                }
            }
        }

        order = ord.ToArray();
        colors = col.ToArray();
    }

    private ulong[] Intersect(ulong[] a, ulong[] b)
    {
        var r = new ulong[words];
        for (int i = 0; i < words; i++) r[i] = a[i] & b[i];
        return r;
    }

    private static bool GetBit(ulong[] bits, int idx)
    {
        return (bits[idx >> 6] & (1UL << (idx & 63))) != 0;
    }

    private static void ClearBit(ulong[] bits, int idx)
    {
        bits[idx >> 6] &= ~(1UL << (idx & 63));
    }

    private static int FirstBit(ulong[] bits)
    {
        for (int i = 0; i < bits.Length; i++)
        {
            ulong w = bits[i];
            if (w != 0)
            {
                int bit = BitOperationsEx.TrailingZeroCount(w);
                return (i << 6) + bit;
            }
        }
        return -1;
    }

    private static int PopCount(ulong[] bits)
    {
        int c = 0;
        for (int i = 0; i < bits.Length; i++) c += BitOperationsEx.PopCount(bits[i]);
        return c;
    }
}

public static class BitOperationsEx
{
    public static int PopCount(ulong x)
    {
        x -= (x >> 1) & 0x5555555555555555UL;
        x = (x & 0x3333333333333333UL) + ((x >> 2) & 0x3333333333333333UL);
        x = (x + (x >> 4)) & 0x0F0F0F0F0F0F0F0FUL;
        return (int)((x * 0x0101010101010101UL) >> 56);
    }

    public static int TrailingZeroCount(ulong x)
    {
        if (x == 0) return 64;
        int n = 0;
        while ((x & 1UL) == 0)
        {
            n++;
            x >>= 1;
        }
        return n;
    }
}

public sealed class FastScanner
{
    private readonly Stream stream;
    private readonly byte[] buffer = new byte[1 << 16];
    private int ptr;
    private int len;

    public FastScanner(Stream stream)
    {
        this.stream = stream;
    }

    private int Read()
    {
        if (ptr >= len)
        {
            len = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (len <= 0) return -1;
        }
        return buffer[ptr++];
    }

    public int NextInt()
    {
        int c;
        do c = Read(); while (c <= 32 && c >= 0);
        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }
        int val = 0;
        while (c > 32)
        {
            val = val * 10 + (c - '0');
            c = Read();
        }
        return val * sign;
    }
}
