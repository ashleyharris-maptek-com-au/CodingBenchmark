using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    struct Box
    {
        public int Index;
        public int W;
        public int H;
        public int D;
        public long Volume;
        public int MaxDim;
        public int MinDim;
    }

    struct Placement
    {
        public int BoxIndex;
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
    }

    struct Space
    {
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;

        public long Volume => (long)W * H * D;
    }

    sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }

    static int N;
    static int CW;
    static int CH;
    static int CD;
    static Box[] Boxes = Array.Empty<Box>();

    public static void Main()
    {
        var fs = new FastScanner();
        N = fs.NextInt();
        CW = fs.NextInt();
        CH = fs.NextInt();
        CD = fs.NextInt();

        Boxes = new Box[N];
        for (int i = 0; i < N; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();
            Boxes[i] = new Box
            {
                Index = i,
                W = w,
                H = h,
                D = d,
                Volume = (long)w * h * d,
                MaxDim = Math.Max(w, Math.Max(h, d)),
                MinDim = Math.Min(w, Math.Min(h, d))
            };
        }

        List<Placement> best;
        long containerVolume = (long)CW * CH * CD;
        if (N <= 12 && containerVolume <= 250000)
        {
            best = SolveSmallExactish();
        }
        else
        {
            best = new List<Placement>();
        }

        var greedy = SolveGreedyEnsemble(best.Count);
        if (greedy.Count > best.Count)
            best = greedy;

        var sw = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        sw.WriteLine(best.Count);
        foreach (var p in best)
        {
            sw.WriteLine($"{p.BoxIndex} {p.X} {p.Y} {p.Z}");
        }
        sw.Flush();
    }

    static List<Placement> SolveGreedyEnsemble(int currentBest)
    {
        var strategies = new List<(Box[] order, int mode)>
        {
            (Boxes.OrderBy(b => b.Volume).ThenBy(b => b.MaxDim).ThenBy(b => b.MinDim).ToArray(), 0),
            (Boxes.OrderBy(b => b.MaxDim).ThenBy(b => b.Volume).ThenBy(b => b.MinDim).ToArray(), 1),
            (Boxes.OrderBy(b => (long)b.MaxDim * b.MinDim).ThenBy(b => b.Volume).ToArray(), 2),
        };

        if (N <= 500)
        {
            strategies.Add((Boxes.OrderByDescending(b => b.MinDim).ThenBy(b => b.Volume).ToArray(), 3));
            strategies.Add((Boxes.OrderByDescending(b => b.Volume).ThenBy(b => b.MaxDim).ToArray(), 4));
        }

        List<Placement> best = new List<Placement>();
        foreach (var (order, mode) in strategies)
        {
            var packed = PackGreedy(order, mode);
            if (packed.Count > best.Count)
                best = packed;
            if (best.Count == N || best.Count > currentBest && best.Count >= N * 0.995)
                break;
        }

        return best;
    }

    static List<Placement> PackGreedy(Box[] order, int mode)
    {
        var spaces = new List<Space>(4096)
        {
            new Space { X = 0, Y = 0, Z = 0, W = CW, H = CH, D = CD }
        };
        var placements = new List<Placement>(order.Length);

        foreach (var box in order)
        {
            int bestSpace = -1;
            int bestW = 0, bestH = 0, bestD = 0;
            long bestScore = long.MaxValue;

            var orientations = GetOrientations(box);
            for (int si = 0; si < spaces.Count; si++)
            {
                var s = spaces[si];
                foreach (var o in orientations)
                {
                    if (o.w > s.W || o.h > s.H || o.d > s.D)
                        continue;

                    long score = ScorePlacement(s, o.w, o.h, o.d, mode);
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestSpace = si;
                        bestW = o.w;
                        bestH = o.h;
                        bestD = o.d;
                    }
                }
            }

            if (bestSpace < 0)
                continue;

            var chosen = spaces[bestSpace];
            placements.Add(new Placement
            {
                BoxIndex = box.Index,
                X = chosen.X,
                Y = chosen.Y,
                Z = chosen.Z,
                W = bestW,
                H = bestH,
                D = bestD
            });

            SplitSpace(spaces, bestSpace, chosen, bestW, bestH, bestD);
            PruneSpaces(spaces);
        }

        return placements;
    }

    static long ScorePlacement(Space s, int w, int h, int d, int mode)
    {
        int rx = s.W - w;
        int ry = s.H - h;
        int rz = s.D - d;
        long waste = (long)rx * s.H * s.D + (long)w * ry * s.D + (long)w * h * rz;

        switch (mode)
        {
            case 0:
                return ((long)s.Z << 50) + ((long)s.Y << 34) + ((long)s.X << 18) + waste;
            case 1:
                return ((long)(rx + ry + rz) << 48) + ((long)s.Z << 32) + ((long)s.Y << 16) + s.X;
            case 2:
                return (waste << 8) + (long)Math.Max(rx, Math.Max(ry, rz));
            case 3:
                return ((long)Math.Max(rx, Math.Max(ry, rz)) << 48) + ((long)s.Z << 32) + waste;
            default:
                return (waste << 12) + ((long)s.Z << 8) + ((long)s.Y << 4) + s.X;
        }
    }

    static void SplitSpace(List<Space> spaces, int index, Space s, int w, int h, int d)
    {
        spaces[index] = spaces[spaces.Count - 1];
        spaces.RemoveAt(spaces.Count - 1);

        int rw = s.W - w;
        int rh = s.H - h;
        int rd = s.D - d;

        if (rw > 0)
        {
            spaces.Add(new Space
            {
                X = s.X + w,
                Y = s.Y,
                Z = s.Z,
                W = rw,
                H = s.H,
                D = s.D
            });
        }

        if (rh > 0)
        {
            spaces.Add(new Space
            {
                X = s.X,
                Y = s.Y + h,
                Z = s.Z,
                W = w,
                H = rh,
                D = s.D
            });
        }

        if (rd > 0)
        {
            spaces.Add(new Space
            {
                X = s.X,
                Y = s.Y,
                Z = s.Z + d,
                W = w,
                H = h,
                D = rd
            });
        }
    }

    static void PruneSpaces(List<Space> spaces)
    {
        if (spaces.Count <= 1) return;

        spaces.Sort((a, b) =>
        {
            int c = a.X.CompareTo(b.X);
            if (c != 0) return c;
            c = a.Y.CompareTo(b.Y);
            if (c != 0) return c;
            c = a.Z.CompareTo(b.Z);
            if (c != 0) return c;
            c = b.W.CompareTo(a.W);
            if (c != 0) return c;
            c = b.H.CompareTo(a.H);
            if (c != 0) return c;
            return b.D.CompareTo(a.D);
        });

        var pruned = new List<Space>(spaces.Count);
        for (int i = 0; i < spaces.Count; i++)
        {
            var s = spaces[i];
            bool contained = false;
            for (int j = 0; j < pruned.Count; j++)
            {
                var t = pruned[j];
                if (Contains(t, s))
                {
                    contained = true;
                    break;
                }
            }

            if (!contained)
                pruned.Add(s);
        }

        const int maxSpacesLarge = 12000;
        const int maxSpacesHuge = 6000;
        int limit = N > 4000 ? maxSpacesHuge : maxSpacesLarge;
        if (pruned.Count > limit)
        {
            pruned.Sort((a, b) =>
            {
                int c = a.Z.CompareTo(b.Z);
                if (c != 0) return c;
                c = b.Volume.CompareTo(a.Volume);
                if (c != 0) return c;
                c = a.Y.CompareTo(b.Y);
                if (c != 0) return c;
                return a.X.CompareTo(b.X);
            });
            pruned.RemoveRange(limit, pruned.Count - limit);
        }

        spaces.Clear();
        spaces.AddRange(pruned);
    }

    static bool Contains(Space outer, Space inner)
    {
        return outer.X <= inner.X &&
               outer.Y <= inner.Y &&
               outer.Z <= inner.Z &&
               outer.X + outer.W >= inner.X + inner.W &&
               outer.Y + outer.H >= inner.Y + inner.H &&
               outer.Z + outer.D >= inner.Z + inner.D;
    }

    static (int w, int h, int d)[] GetOrientations(Box b)
    {
        int a = b.W, c = b.H, d = b.D;
        var set = new HashSet<long>();
        var list = new List<(int, int, int)>(6);

        void Add(int x, int y, int z)
        {
            if (x > CW || y > CH || z > CD) return;
            long key = ((long)x << 42) ^ ((long)y << 21) ^ (uint)z;
            if (set.Add(key))
                list.Add((x, y, z));
        }

        Add(a, c, d);
        Add(a, d, c);
        Add(c, a, d);
        Add(c, d, a);
        Add(d, a, c);
        Add(d, c, a);
        return list.ToArray();
    }

    static List<Placement> SolveSmallExactish()
    {
        var order = Boxes.OrderBy(b => b.Volume).ThenBy(b => b.MaxDim).ToArray();
        var used = new bool[N];
        var placed = new List<Placement>(N);
        var best = new List<Placement>(N);
        var candidates = new List<(int x, int y, int z)> { (0, 0, 0) };

        void Dfs()
        {
            if (placed.Count > best.Count)
                best = new List<Placement>(placed);

            int remaining = N - placed.Count;
            if (placed.Count + remaining <= best.Count)
                return;

            candidates.Sort((p1, p2) =>
            {
                int c = p1.z.CompareTo(p2.z);
                if (c != 0) return c;
                c = p1.y.CompareTo(p2.y);
                if (c != 0) return c;
                return p1.x.CompareTo(p2.x);
            });
            DedupCandidates(candidates);

            for (int ci = 0; ci < candidates.Count; ci++)
            {
                var c = candidates[ci];
                bool progressed = false;

                for (int oi = 0; oi < order.Length; oi++)
                {
                    var box = order[oi];
                    if (used[box.Index]) continue;

                    foreach (var o in GetOrientations(box))
                    {
                        if (CanPlace(c.x, c.y, c.z, o.w, o.h, o.d, placed))
                        {
                            used[box.Index] = true;
                            var p = new Placement
                            {
                                BoxIndex = box.Index,
                                X = c.x,
                                Y = c.y,
                                Z = c.z,
                                W = o.w,
                                H = o.h,
                                D = o.d
                            };
                            placed.Add(p);

                            int oldCount = candidates.Count;
                            candidates.Add((c.x + o.w, c.y, c.z));
                            candidates.Add((c.x, c.y + o.h, c.z));
                            candidates.Add((c.x, c.y, c.z + o.d));
                            Dfs();
                            candidates.RemoveRange(oldCount, candidates.Count - oldCount);

                            placed.RemoveAt(placed.Count - 1);
                            used[box.Index] = false;
                            progressed = true;
                        }
                    }
                }

                if (!progressed)
                    continue;

                if (best.Count == N)
                    return;
            }
        }

        Dfs();
        return best;
    }

    static void DedupCandidates(List<(int x, int y, int z)> candidates)
    {
        int write = 0;
        int lastX = int.MinValue, lastY = int.MinValue, lastZ = int.MinValue;
        for (int i = 0; i < candidates.Count; i++)
        {
            var c = candidates[i];
            if (c.x == lastX && c.y == lastY && c.z == lastZ)
                continue;
            candidates[write++] = c;
            lastX = c.x;
            lastY = c.y;
            lastZ = c.z;
        }
        if (write < candidates.Count)
            candidates.RemoveRange(write, candidates.Count - write);
    }

    static bool CanPlace(int x, int y, int z, int w, int h, int d, List<Placement> placed)
    {
        if (x < 0 || y < 0 || z < 0) return false;
        if (x + w > CW || y + h > CH || z + d > CD) return false;

        for (int i = 0; i < placed.Count; i++)
        {
            var p = placed[i];
            bool overlap =
                x < p.X + p.W && x + w > p.X &&
                y < p.Y + p.H && y + h > p.Y &&
                z < p.Z + p.D && z + d > p.Z;
            if (overlap) return false;
        }

        return true;
    }
}
