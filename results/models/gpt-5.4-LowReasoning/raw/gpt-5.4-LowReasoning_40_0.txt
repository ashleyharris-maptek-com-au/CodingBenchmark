#include <algorithm>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

class MWISSolver {
public:
    explicit MWISSolver(int n)
        : n_(n),
          weight_(n, 0),
          adj_(n),
          chosen_(n, 0),
          best_(n, 0),
          removed_(n, 0),
          active_(n, 1),
          degree_(n, 0),
          active_count_(n),
          current_value_(0),
          best_value_(0) {}

    void set_weight(int v, long long w) {
        weight_[v] = max(0LL, w);
    }

    void add_edge(int u, int v) {
        if (u == v) return;
        adj_[u].push_back(v);
        adj_[v].push_back(u);
    }

    pair<long long, vector<int>> solve() {
        for (int v = 0; v < n_; ++v) {
            sort(adj_[v].begin(), adj_[v].end());
            adj_[v].erase(unique(adj_[v].begin(), adj_[v].end()), adj_[v].end());
            degree_[v] = static_cast<int>(adj_[v].size());
        }
        branch();
        return {best_value_, best_};
    }

private:
    struct Change {
        int v;
        int prev_removed;
        int prev_active;
        int prev_degree;
    };

    int n_;
    vector<long long> weight_;
    vector<vector<int>> adj_;
    vector<int> chosen_;
    vector<int> best_;
    vector<int> removed_;
    vector<int> active_;
    vector<int> degree_;
    int active_count_;
    long long current_value_;
    long long best_value_;
    vector<Change> history_;

    void save_and_remove(int v) {
        history_.push_back({v, removed_[v], active_[v], degree_[v]});
        if (!removed_[v]) {
            removed_[v] = 1;
            if (active_[v]) {
                active_[v] = 0;
                --active_count_;
                for (int u : adj_[v]) {
                    if (!removed_[u] && active_[u]) {
                        history_.push_back({u, removed_[u], active_[u], degree_[u]});
                        --degree_[u];
                    }
                }
            }
        }
    }

    void restore(size_t checkpoint) {
        while (history_.size() > checkpoint) {
            Change ch = history_.back();
            history_.pop_back();
            if (active_[ch.v] != ch.prev_active && ch.prev_active) {
                ++active_count_;
            } else if (active_[ch.v] != ch.prev_active && !ch.prev_active) {
                --active_count_;
            }
            removed_[ch.v] = ch.prev_removed;
            active_[ch.v] = ch.prev_active;
            degree_[ch.v] = ch.prev_degree;
        }
    }

    bool adjacent(int u, int v) const {
        return binary_search(adj_[u].begin(), adj_[u].end(), v);
    }

    long long component_bound(const vector<int>& comp) {
        vector<int> order = comp;
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (degree_[a] != degree_[b]) return degree_[a] > degree_[b];
            if (weight_[a] != weight_[b]) return weight_[a] > weight_[b];
            return a < b;
        });

        vector<long long> clique_max_weight;
        vector<vector<int>> cliques;

        for (int v : order) {
            int idx = -1;
            for (int i = 0; i < static_cast<int>(cliques.size()); ++i) {
                bool fits = true;
                for (int u : cliques[i]) {
                    if (!adjacent(u, v)) {
                        fits = false;
                        break;
                    }
                }
                if (fits) {
                    idx = i;
                    break;
                }
            }

            if (idx == -1) {
                cliques.push_back({v});
                clique_max_weight.push_back(weight_[v]);
            } else {
                cliques[idx].push_back(v);
                clique_max_weight[idx] = max(clique_max_weight[idx], weight_[v]);
            }
        }

        long long ub = 0;
        for (long long w : clique_max_weight) ub += w;
        return ub;
    }

    long long upper_bound() {
        vector<int> seen(n_, 0);
        queue<int> q;
        long long ub = current_value_;

        for (int start = 0; start < n_; ++start) {
            if (!active_[start] || seen[start]) continue;
            vector<int> comp;
            seen[start] = 1;
            q.push(start);
            while (!q.empty()) {
                int v = q.front();
                q.pop();
                comp.push_back(v);
                for (int u : adj_[v]) {
                    if (active_[u] && !seen[u]) {
                        seen[u] = 1;
                        q.push(u);
                    }
                }
            }
            ub += component_bound(comp);
        }
        return ub;
    }

    bool reduce_forced() {
        bool changed = true;
        while (changed) {
            changed = false;
            for (int v = 0; v < n_; ++v) {
                if (!active_[v]) continue;
                if (weight_[v] == 0) {
                    save_and_remove(v);
                    changed = true;
                    continue;
                }
                if (degree_[v] == 0) {
                    chosen_[v] = 1;
                    current_value_ += weight_[v];
                    size_t checkpoint = history_.size();
                    save_and_remove(v);
                    for (int u : adj_[v]) save_and_remove(u);
                    (void)checkpoint;
                    changed = true;
                }
            }
        }
        return true;
    }

    int choose_branch_vertex() {
        int best_v = -1;
        long double best_score = -1;
        for (int v = 0; v < n_; ++v) {
            if (!active_[v]) continue;
            long double score = static_cast<long double>(weight_[v]) * (degree_[v] + 1);
            if (score > best_score || (score == best_score && weight_[v] > weight_[best_v])) {
                best_score = score;
                best_v = v;
            }
        }
        return best_v;
    }

    void branch() {
        size_t checkpoint = history_.size();
        vector<int> chosen_checkpoint = chosen_;
        long long value_checkpoint = current_value_;

        reduce_forced();

        if (active_count_ == 0) {
            if (current_value_ > best_value_) {
                best_value_ = current_value_;
                best_ = chosen_;
            }
            restore(checkpoint);
            chosen_ = std::move(chosen_checkpoint);
            current_value_ = value_checkpoint;
            return;
        }

        if (upper_bound() <= best_value_) {
            restore(checkpoint);
            chosen_ = std::move(chosen_checkpoint);
            current_value_ = value_checkpoint;
            return;
        }

        int v = choose_branch_vertex();

        {
            size_t cp = history_.size();
            vector<int> chosen_cp = chosen_;
            long long value_cp = current_value_;

            chosen_[v] = 1;
            current_value_ += weight_[v];
            save_and_remove(v);
            for (int u : adj_[v]) save_and_remove(u);
            branch();

            restore(cp);
            chosen_ = std::move(chosen_cp);
            current_value_ = value_cp;
        }

        {
            size_t cp = history_.size();
            vector<int> chosen_cp = chosen_;
            long long value_cp = current_value_;

            save_and_remove(v);
            branch();

            restore(cp);
            chosen_ = std::move(chosen_cp);
            current_value_ = value_cp;
        }

        restore(checkpoint);
        chosen_ = std::move(chosen_checkpoint);
        current_value_ = value_checkpoint;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    MWISSolver solver(n);
    for (int i = 0; i < n; ++i) {
        long long c;
        cin >> c;
        solver.set_weight(i, c);
    }

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (0 <= u && u < n && 0 <= v && v < n) {
            solver.add_edge(u, v);
        }
    }

    auto [objective, x] = solver.solve();
    cout << objective << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << x[i];
    }
    cout << '\n';
    return 0;
}
