#include <algorithm>
#include <cstdint>
#include <functional>
#include <iostream>
#include <numeric>
#include <queue>
#include <vector>

using namespace std;

struct MaxWeightCliqueSolver {
    int n = 0;
    vector<long long> w;
    vector<vector<uint64_t>> adj;
    long long best_weight = 0;
    vector<int> best_set;
    vector<int> current_set;

    explicit MaxWeightCliqueSolver(const vector<long long>& weights,
                                   const vector<vector<uint64_t>>& adjacency)
        : n((int)weights.size()), w(weights), adj(adjacency) {}

    bool connected(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    vector<int> sort_candidates(const vector<int>& cand) const {
        vector<int> out = cand;
        sort(out.begin(), out.end(), [&](int a, int b) {
            if (w[a] != w[b]) return w[a] > w[b];
            return a < b;
        });
        return out;
    }

    void color_sort(const vector<int>& cand, vector<int>& order, vector<long long>& bound) const {
        vector<int> remaining = sort_candidates(cand);
        order.clear();
        bound.clear();
        long long sum_max = 0;

        while (!remaining.empty()) {
            vector<int> color_class;
            vector<int> next_remaining;
            long long class_max = 0;

            for (int v : remaining) {
                bool fits = true;
                for (int u : color_class) {
                    if (connected(u, v)) {
                        fits = false;
                        break;
                    }
                }
                if (fits) {
                    color_class.push_back(v);
                    class_max = max(class_max, w[v]);
                } else {
                    next_remaining.push_back(v);
                }
            }

            sum_max += class_max;
            for (int v : color_class) {
                order.push_back(v);
                bound.push_back(sum_max);
            }

            remaining.swap(next_remaining);
        }
    }

    void expand(const vector<int>& cand, long long current_weight) {
        if (cand.empty()) {
            if (current_weight > best_weight) {
                best_weight = current_weight;
                best_set = current_set;
            }
            return;
        }

        vector<int> order;
        vector<long long> bound;
        color_sort(cand, order, bound);

        for (int i = (int)order.size() - 1; i >= 0; --i) {
            int v = order[i];
            if (current_weight + bound[i] <= best_weight) return;

            current_set.push_back(v);

            vector<int> next_cand;
            next_cand.reserve(i);
            for (int j = 0; j < i; ++j) {
                int u = order[j];
                if (connected(u, v)) next_cand.push_back(u);
            }

            if (next_cand.empty()) {
                long long total = current_weight + w[v];
                if (total > best_weight) {
                    best_weight = total;
                    best_set = current_set;
                }
            } else {
                expand(next_cand, current_weight + w[v]);
            }

            current_set.pop_back();
        }
    }

    pair<long long, vector<int>> solve() {
        vector<int> all(n);
        iota(all.begin(), all.end(), 0);
        expand(all, 0);
        return {best_weight, best_set};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<vector<int>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    vector<int> answer(n, 0);
    long long objective = 0;

    vector<int> active_vertices;
    active_vertices.reserve(n);
    for (int i = 0; i < n; ++i) {
        if (c[i] > 0) active_vertices.push_back(i);
    }

    vector<char> active(n, false), seen(n, false);
    for (int v : active_vertices) active[v] = true;

    for (int start : active_vertices) {
        if (seen[start]) continue;

        vector<int> component;
        queue<int> q;
        q.push(start);
        seen[start] = true;

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            component.push_back(v);
            for (int to : g[v]) {
                if (active[to] && !seen[to]) {
                    seen[to] = true;
                    q.push(to);
                }
            }
        }

        int k = (int)component.size();
        if (k == 1) {
            int v = component[0];
            answer[v] = 1;
            objective += c[v];
            continue;
        }

        vector<int> local_id(n, -1);
        for (int i = 0; i < k; ++i) local_id[component[i]] = i;

        int words = (k + 63) >> 6;
        vector<vector<uint64_t>> conflict(k, vector<uint64_t>(words, 0));
        for (int i = 0; i < k; ++i) {
            int v = component[i];
            for (int to : g[v]) {
                int j = local_id[to];
                if (j != -1) conflict[i][j >> 6] |= 1ULL << (j & 63);
            }
        }

        vector<vector<uint64_t>> clique_adj(k, vector<uint64_t>(words, 0));
        for (int i = 0; i < k; ++i) {
            for (int j = 0; j < k; ++j) {
                if (i == j) continue;
                bool has_conflict = (conflict[i][j >> 6] >> (j & 63)) & 1ULL;
                if (!has_conflict) clique_adj[i][j >> 6] |= 1ULL << (j & 63);
            }
        }

        vector<long long> weights(k);
        for (int i = 0; i < k; ++i) weights[i] = c[component[i]];

        MaxWeightCliqueSolver solver(weights, clique_adj);
        auto [best_weight, chosen_local] = solver.solve();
        objective += best_weight;
        for (int id : chosen_local) answer[component[id]] = 1;
    }

    cout << objective << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';
    return 0;
}
