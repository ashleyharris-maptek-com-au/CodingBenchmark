using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Numerics;
using System.Text;

public static class Program
{
    private const double Eps = 1e-9;

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int vertexCount = scanner.NextInt();
        int faceCount = scanner.NextInt();

        var vertices = new Vector3[vertexCount];
        for (int i = 0; i < vertexCount; i++)
        {
            vertices[i] = new Vector3(
                (float)scanner.NextDouble(),
                (float)scanner.NextDouble(),
                (float)scanner.NextDouble());
        }

        var faces = new Face[faceCount];
        for (int i = 0; i < faceCount; i++)
        {
            faces[i] = new Face(scanner.NextInt(), scanner.NextInt(), scanner.NextInt());
        }

        double containerW = scanner.NextDouble();
        double containerH = scanner.NextDouble();
        double containerD = scanner.NextDouble();

        var mesh = new Mesh(vertices, faces);
        var solver = new AdaptivePacker(mesh, containerW, containerH, containerD);
        solver.Solve();
    }

    private readonly struct Face
    {
        public readonly int A;
        public readonly int B;
        public readonly int C;

        public Face(int a, int b, int c)
        {
            A = a;
            B = b;
            C = c;
        }
    }

    private sealed class Mesh
    {
        public Vector3[] Vertices { get; }
        public Face[] Faces { get; }
        public Vector3 Centroid { get; }
        public double RawVolumeEstimate { get; }
        public double ExtentProduct { get; }
        public Bounds OriginalBounds { get; }

        public Mesh(Vector3[] vertices, Face[] faces)
        {
            Vertices = vertices;
            Faces = faces;

            Vector3 sum = Vector3.Zero;
            foreach (var v in vertices)
            {
                sum += v;
            }

            Centroid = vertices.Length == 0 ? Vector3.Zero : sum / vertices.Length;
            OriginalBounds = ComputeBounds(vertices, Quaternion.Identity);
            ExtentProduct = OriginalBounds.SizeX * OriginalBounds.SizeY * OriginalBounds.SizeZ;
            RawVolumeEstimate = EstimateSignedVolume(vertices, faces);
        }

        private static double EstimateSignedVolume(Vector3[] vertices, Face[] faces)
        {
            double sum = 0.0;
            foreach (var f in faces)
            {
                var a = vertices[f.A];
                var b = vertices[f.B];
                var c = vertices[f.C];
                sum += Vector3.Dot(a, Vector3.Cross(b, c));
            }

            return Math.Abs(sum) / 6.0;
        }

        private static Bounds ComputeBounds(Vector3[] vertices, Quaternion rotation)
        {
            double minX = double.PositiveInfinity;
            double minY = double.PositiveInfinity;
            double minZ = double.PositiveInfinity;
            double maxX = double.NegativeInfinity;
            double maxY = double.NegativeInfinity;
            double maxZ = double.NegativeInfinity;

            foreach (var v in vertices)
            {
                Vector3 p = Vector3.Transform(v, rotation);
                if (p.X < minX) minX = p.X;
                if (p.Y < minY) minY = p.Y;
                if (p.Z < minZ) minZ = p.Z;
                if (p.X > maxX) maxX = p.X;
                if (p.Y > maxY) maxY = p.Y;
                if (p.Z > maxZ) maxZ = p.Z;
            }

            return new Bounds(minX, minY, minZ, maxX, maxY, maxZ);
        }
    }

    private sealed class AdaptivePacker
    {
        private readonly Mesh _mesh;
        private readonly double _w;
        private readonly double _h;
        private readonly double _d;

        public AdaptivePacker(Mesh mesh, double w, double h, double d)
        {
            _mesh = mesh;
            _w = w;
            _h = h;
            _d = d;
        }

        public void Solve()
        {
            var candidates = BuildCandidateRotations();
            PlacementPlan best = BuildEmptyPlan();

            foreach (var q in candidates)
            {
                var plan = EvaluateRotation(q);
                if (plan.IsBetterThan(best))
                {
                    best = plan;
                }
            }

            WritePlan(best);
        }

        private PlacementPlan BuildEmptyPlan()
        {
            return new PlacementPlan
            {
                Count = 0,
                Rotation = Quaternion.Identity,
                Bounds = ComputeBounds(_mesh.Vertices, Quaternion.Identity),
                BaseTranslation = Vector3.Zero,
                NX = 0,
                NY = 0,
                NZ = 0
            };
        }

        private List<Quaternion> BuildCandidateRotations()
        {
            double containerVolume = _w * _h * _d;
            double characteristicSize = Math.Max(_w, Math.Max(_h, _d));
            int complexity = _mesh.Vertices.Length + (_mesh.Faces.Length * 2);

            int faceBudget;
            if (characteristicSize <= 25.0 && complexity <= 2000)
            {
                faceBudget = 192;
            }
            else if (characteristicSize <= 100.0 && complexity <= 15000)
            {
                faceBudget = 96;
            }
            else if (containerVolume <= 8_000_000.0 && complexity <= 50000)
            {
                faceBudget = 48;
            }
            else
            {
                faceBudget = 16;
            }

            var list = new List<Quaternion>();
            var seen = new HashSet<string>(StringComparer.Ordinal);

            foreach (var q in GenerateAxisAlignedCubeRotations())
            {
                AddCandidate(list, seen, q);
            }

            int sampled = 0;
            int stride = Math.Max(1, _mesh.Faces.Length / Math.Max(1, faceBudget));
            for (int i = 0; i < _mesh.Faces.Length && sampled < faceBudget; i += stride)
            {
                var f = _mesh.Faces[i];
                Vector3 a = _mesh.Vertices[f.A];
                Vector3 b = _mesh.Vertices[f.B];
                Vector3 c = _mesh.Vertices[f.C];
                Vector3 normal = Vector3.Cross(b - a, c - a);
                if (normal.LengthSquared() < 1e-12f)
                {
                    continue;
                }

                normal = Vector3.Normalize(normal);
                Quaternion alignNormal = RotationBetween(normal, Vector3.UnitZ);
                Vector3 rotatedEdge = Vector3.Transform(Vector3.Normalize(b - a), alignNormal);
                Vector3 edgeXY = new Vector3(rotatedEdge.X, rotatedEdge.Y, 0f);
                if (edgeXY.LengthSquared() < 1e-12f)
                {
                    AddCandidate(list, seen, Quaternion.Normalize(alignNormal));
                    sampled++;
                    continue;
                }

                edgeXY = Vector3.Normalize(edgeXY);
                double yaw = Math.Atan2(edgeXY.Y, edgeXY.X);
                Quaternion alignEdge = Quaternion.CreateFromAxisAngle(Vector3.UnitZ, (float)(-yaw));
                AddCandidate(list, seen, Quaternion.Normalize(alignEdge * alignNormal));
                sampled++;
            }

            return list;
        }

        private PlacementPlan EvaluateRotation(Quaternion rotation)
        {
            Bounds bounds = ComputeBounds(_mesh.Vertices, rotation);
            double sx = bounds.SizeX;
            double sy = bounds.SizeY;
            double sz = bounds.SizeZ;

            if (sx <= Eps || sy <= Eps || sz <= Eps)
            {
                return BuildEmptyPlan();
            }

            long nx = SafeFloorDiv(_w, sx);
            long ny = SafeFloorDiv(_h, sy);
            long nz = SafeFloorDiv(_d, sz);

            if (nx <= 0 || ny <= 0 || nz <= 0)
            {
                return BuildEmptyPlan();
            }

            double countAsDouble = (double)nx * ny * nz;
            long count = countAsDouble >= long.MaxValue ? long.MaxValue : (long)countAsDouble;

            return new PlacementPlan
            {
                Count = count,
                Rotation = rotation,
                Bounds = bounds,
                BaseTranslation = new Vector3(
                    (float)(-bounds.MinX),
                    (float)(-bounds.MinY),
                    (float)(-bounds.MinZ)),
                NX = nx,
                NY = ny,
                NZ = nz
            };
        }

        private void WritePlan(PlacementPlan plan)
        {
            using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
            writer.WriteLine(plan.Count.ToString(CultureInfo.InvariantCulture));

            if (plan.Count == 0)
            {
                return;
            }

            var q = Quaternion.Normalize(plan.Rotation);
            string qw = Format(q.W);
            string qx = Format(q.X);
            string qy = Format(q.Y);
            string qz = Format(q.Z);

            double stepX = plan.Bounds.SizeX;
            double stepY = plan.Bounds.SizeY;
            double stepZ = plan.Bounds.SizeZ;

            for (long iz = 0; iz < plan.NZ; iz++)
            {
                double baseZ = plan.BaseTranslation.Z + (iz * stepZ);
                for (long iy = 0; iy < plan.NY; iy++)
                {
                    double baseY = plan.BaseTranslation.Y + (iy * stepY);
                    for (long ix = 0; ix < plan.NX; ix++)
                    {
                        double baseX = plan.BaseTranslation.X + (ix * stepX);
                        writer.Write(Format(baseX));
                        writer.Write(' ');
                        writer.Write(Format(baseY));
                        writer.Write(' ');
                        writer.Write(Format(baseZ));
                        writer.Write(' ');
                        writer.Write(qw);
                        writer.Write(' ');
                        writer.Write(qx);
                        writer.Write(' ');
                        writer.Write(qy);
                        writer.Write(' ');
                        writer.WriteLine(qz);
                    }
                }
            }
        }

        private static long SafeFloorDiv(double total, double step)
        {
            if (step <= Eps)
            {
                return 0;
            }

            double ratio = total / step;
            if (double.IsNaN(ratio) || double.IsInfinity(ratio) || ratio < 1.0 - Eps)
            {
                return 0;
            }

            return (long)Math.Floor(ratio + Eps);
        }

        private static IEnumerable<Quaternion> GenerateAxisAlignedCubeRotations()
        {
            Vector3[] dirs =
            {
                Vector3.UnitX, -Vector3.UnitX,
                Vector3.UnitY, -Vector3.UnitY,
                Vector3.UnitZ, -Vector3.UnitZ
            };

            foreach (var xAxis in dirs)
            {
                foreach (var yAxis in dirs)
                {
                    if (Math.Abs(Vector3.Dot(xAxis, yAxis)) > 1e-6f)
                    {
                        continue;
                    }

                    Vector3 zAxis = Vector3.Cross(xAxis, yAxis);
                    if (zAxis.LengthSquared() < 0.5f)
                    {
                        continue;
                    }

                    var m = new Matrix4x4(
                        xAxis.X, xAxis.Y, xAxis.Z, 0f,
                        yAxis.X, yAxis.Y, yAxis.Z, 0f,
                        zAxis.X, zAxis.Y, zAxis.Z, 0f,
                        0f, 0f, 0f, 1f);
                    yield return Quaternion.Normalize(Quaternion.CreateFromRotationMatrix(m));
                }
            }
        }

        private static void AddCandidate(List<Quaternion> list, HashSet<string> seen, Quaternion q)
        {
            q = Quaternion.Normalize(q);
            if (q.W < 0f)
            {
                q = new Quaternion(-q.X, -q.Y, -q.Z, -q.W);
            }

            string key = string.Create(
                CultureInfo.InvariantCulture,
                $"{Math.Round(q.W, 6)},{Math.Round(q.X, 6)},{Math.Round(q.Y, 6)},{Math.Round(q.Z, 6)}");

            if (seen.Add(key))
            {
                list.Add(q);
            }
        }

        private static Quaternion RotationBetween(Vector3 from, Vector3 to)
        {
            from = Vector3.Normalize(from);
            to = Vector3.Normalize(to);
            float dot = Vector3.Dot(from, to);

            if (dot > 1f - 1e-6f)
            {
                return Quaternion.Identity;
            }

            if (dot < -1f + 1e-6f)
            {
                Vector3 axis = Vector3.Cross(from, Vector3.UnitX);
                if (axis.LengthSquared() < 1e-8f)
                {
                    axis = Vector3.Cross(from, Vector3.UnitY);
                }

                axis = Vector3.Normalize(axis);
                return Quaternion.CreateFromAxisAngle(axis, MathF.PI);
            }

            Vector3 cross = Vector3.Cross(from, to);
            float s = MathF.Sqrt((1f + dot) * 2f);
            float invS = 1f / s;
            return Quaternion.Normalize(new Quaternion(cross * invS, s * 0.5f));
        }

        private static Bounds ComputeBounds(Vector3[] vertices, Quaternion rotation)
        {
            double minX = double.PositiveInfinity;
            double minY = double.PositiveInfinity;
            double minZ = double.PositiveInfinity;
            double maxX = double.NegativeInfinity;
            double maxY = double.NegativeInfinity;
            double maxZ = double.NegativeInfinity;

            foreach (var v in vertices)
            {
                Vector3 p = Vector3.Transform(v, rotation);
                if (p.X < minX) minX = p.X;
                if (p.Y < minY) minY = p.Y;
                if (p.Z < minZ) minZ = p.Z;
                if (p.X > maxX) maxX = p.X;
                if (p.Y > maxY) maxY = p.Y;
                if (p.Z > maxZ) maxZ = p.Z;
            }

            return new Bounds(minX, minY, minZ, maxX, maxY, maxZ);
        }

        private static string Format(double value)
        {
            if (Math.Abs(value) < 5e-10)
            {
                value = 0.0;
            }

            return value.ToString("G9", CultureInfo.InvariantCulture);
        }
    }

    private readonly struct Bounds
    {
        public readonly double MinX;
        public readonly double MinY;
        public readonly double MinZ;
        public readonly double MaxX;
        public readonly double MaxY;
        public readonly double MaxZ;

        public Bounds(double minX, double minY, double minZ, double maxX, double maxY, double maxZ)
        {
            MinX = minX;
            MinY = minY;
            MinZ = minZ;
            MaxX = maxX;
            MaxY = maxY;
            MaxZ = maxZ;
        }

        public double SizeX => MaxX - MinX;
        public double SizeY => MaxY - MinY;
        public double SizeZ => MaxZ - MinZ;
    }

    private struct PlacementPlan
    {
        public long Count;
        public Quaternion Rotation;
        public Bounds Bounds;
        public Vector3 BaseTranslation;
        public long NX;
        public long NY;
        public long NZ;

        public bool IsBetterThan(PlacementPlan other)
        {
            if (Count != other.Count)
            {
                return Count > other.Count;
            }

            double myCell = Bounds.SizeX * Bounds.SizeY * Bounds.SizeZ;
            double otherCell = other.Bounds.SizeX * other.Bounds.SizeY * other.Bounds.SizeZ;
            if (Math.Abs(myCell - otherCell) > 1e-9)
            {
                return myCell < otherCell;
            }

            double myPerimeter = Bounds.SizeX + Bounds.SizeY + Bounds.SizeZ;
            double otherPerimeter = other.Bounds.SizeX + other.Bounds.SizeY + other.Bounds.SizeZ;
            return myPerimeter < otherPerimeter;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            return int.Parse(NextToken(), CultureInfo.InvariantCulture);
        }

        public double NextDouble()
        {
            return double.Parse(NextToken(), CultureInfo.InvariantCulture);
        }

        private string NextToken()
        {
            var sb = new StringBuilder();
            int c;
            do
            {
                c = Read();
            } while (c >= 0 && char.IsWhiteSpace((char)c));

            while (c >= 0 && !char.IsWhiteSpace((char)c))
            {
                sb.Append((char)c);
                c = Read();
            }

            return sb.ToString();
        }

        private int Read()
        {
            if (_ptr >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_ptr++];
        }
    }
}
