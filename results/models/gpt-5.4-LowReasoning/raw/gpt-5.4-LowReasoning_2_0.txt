using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        if (n <= 0)
        {
            return;
        }

        if (m == 0)
        {
            Console.WriteLine(1);
            Console.WriteLine(0);
            return;
        }

        int duplicatedUndirectedEdges = checked(m * 2);
        int arcCapacity = checked(duplicatedUndirectedEdges * 2);

        int[] head = new int[n];
        Array.Fill(head, -1);

        int[] to = new int[arcCapacity];
        int[] next = new int[arcCapacity];
        int[] edgeId = new int[arcCapacity];
        int[] degree = new int[n];

        int arcCount = 0;
        int logicalEdgeId = 0;

        void AddArc(int u, int v, int id)
        {
            to[arcCount] = v;
            edgeId[arcCount] = id;
            next[arcCount] = head[u];
            head[u] = arcCount++;
        }

        void AddUndirected(int u, int v)
        {
            int id = logicalEdgeId++;
            AddArc(u, v, id);
            AddArc(v, u, id);
            degree[u]++;
            degree[v]++;
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            _ = fs.NextInt(); // Weight is irrelevant for constructing a valid route.
            AddUndirected(u, v);
            AddUndirected(u, v);
        }

        int start = 0;
        if (degree[start] == 0)
        {
            Console.WriteLine(1);
            Console.WriteLine(0);
            return;
        }

        bool[] seen = new bool[n];
        var queue = new Queue<int>();
        queue.Enqueue(start);
        seen[start] = true;

        while (queue.Count > 0)
        {
            int u = queue.Dequeue();
            for (int e = head[u]; e != -1; e = next[e])
            {
                int v = to[e];
                if (!seen[v])
                {
                    seen[v] = true;
                    queue.Enqueue(v);
                }
            }
        }

        for (int i = 0; i < n; i++)
        {
            if (degree[i] > 0 && !seen[i])
            {
                Console.WriteLine(0);
                return;
            }
        }

        int[] iter = new int[n];
        Array.Copy(head, iter, n);

        bool[] used = new bool[logicalEdgeId];
        int totalEulerEdges = logicalEdgeId;

        int[] stack = new int[totalEulerEdges + 1];
        int[] route = new int[totalEulerEdges + 1];
        int stackSize = 0;
        int routeSize = 0;

        stack[stackSize++] = start;

        while (stackSize > 0)
        {
            int u = stack[stackSize - 1];
            int e = iter[u];

            while (e != -1 && used[edgeId[e]])
            {
                e = next[e];
            }

            iter[u] = e;

            if (e == -1)
            {
                route[routeSize++] = u;
                stackSize--;
            }
            else
            {
                used[edgeId[e]] = true;
                iter[u] = next[e];
                stack[stackSize++] = to[e];
            }
        }

        if (routeSize != totalEulerEdges + 1 || route[routeSize - 1] != start)
        {
            Console.WriteLine(0);
            return;
        }

        var sb = new StringBuilder(Math.Max(32, routeSize * 3));
        sb.Append(routeSize).Append('\n');
        for (int i = routeSize - 1; i >= 0; i--)
        {
            if (i != routeSize - 1)
            {
                sb.Append(' ');
            }
            sb.Append(route[i]);
        }
        sb.Append('\n');
        Console.Write(sb.ToString());
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _pos;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_pos >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_pos++];
        }
    }
}
