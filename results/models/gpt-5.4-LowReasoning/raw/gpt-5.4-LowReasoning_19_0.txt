using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private const long RevenuePerValueMass = 1000L;
    private const int SmallWorldVoxelLimit = 8_000_000;
    private const int SmallWorldColumnLimit = 2_000_000;

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());

        int xCount = input.NextInt();
        int yCount = input.NextInt();
        int zCount = input.NextInt();
        double xBlockSize = input.NextDouble();
        double yBlockSize = input.NextDouble();
        double zBlockSize = input.NextDouble();

        long totalVoxels = (long)xCount * yCount * zCount;
        long totalColumns = (long)xCount * yCount;

        bool smallWorld = totalVoxels <= SmallWorldVoxelLimit && totalColumns <= SmallWorldColumnLimit;

        var solver = smallWorld
            ? (ISolver)new SmallWorldSolver(xCount, yCount, zCount, xBlockSize, yBlockSize, zBlockSize)
            : new LargeWorldSolver(xCount, yCount, zCount, xBlockSize, yBlockSize, zBlockSize);

        solver.ReadWorld(input);
        solver.Plan();
        solver.WriteOutput(Console.Out);
    }

    private interface ISolver
    {
        void ReadWorld(FastInput input);
        void Plan();
        void WriteOutput(TextWriter writer);
    }

    private sealed class SmallWorldSolver : ISolver
    {
        private readonly int _xCount;
        private readonly int _yCount;
        private readonly int _zCount;
        private readonly Column[] _columns;
        private readonly int[] _surface;
        private readonly List<Operation> _operations = new List<Operation>();

        private int _plantSurface;
        private int _spoilIndex;
        private int _spoilSurface;

        public SmallWorldSolver(int xCount, int yCount, int zCount, double _, double __, double ___)
        {
            _xCount = xCount;
            _yCount = yCount;
            _zCount = zCount;
            _columns = new Column[xCount * yCount];
            _surface = new int[xCount * yCount];
            Array.Fill(_surface, -1);
        }

        public void ReadWorld(FastInput input)
        {
            for (int x = 0; x < _xCount; x++)
            {
                for (int y = 0; y < _yCount; y++)
                {
                    int index = IndexOf(x, y);
                    var voxels = new byte[_zCount];
                    int top = -1;
                    for (int z = 0; z < _zCount; z++)
                    {
                        byte d = input.NextByte();
                        voxels[z] = d;
                        if ((d & 15) > 0)
                        {
                            top = z;
                        }
                    }

                    _columns[index] = new Column(voxels, top);
                    _surface[index] = top;
                }
            }

            _plantSurface = _surface[0];
            ChooseSpoilSite();
        }

        public void Plan()
        {
            var candidates = new List<ColumnPlan>();

            for (int x = 0; x < _xCount; x++)
            {
                for (int y = 0; y < _yCount; y++)
                {
                    int index = IndexOf(x, y);
                    if (index == 0 || index == _spoilIndex)
                    {
                        continue;
                    }

                    var plan = EvaluateColumn(x, y, _columns[index]);
                    if (plan.TargetDepth >= 0 && plan.EstimatedProfit > 0)
                    {
                        candidates.Add(plan);
                    }
                }
            }

            candidates.Sort((a, b) =>
            {
                int byProfit = b.EstimatedProfit.CompareTo(a.EstimatedProfit);
                if (byProfit != 0) return byProfit;
                return a.DistanceToPlant.CompareTo(b.DistanceToPlant);
            });

            foreach (var plan in candidates)
            {
                MineColumn(plan.X, plan.Y, plan.TargetDepth);
            }
        }

        public void WriteOutput(TextWriter writer)
        {
            writer.WriteLine(_operations.Count);
            foreach (var op in _operations)
            {
                writer.WriteLine(op.ToOutputLine());
            }
        }

        private ColumnPlan EvaluateColumn(int x, int y, Column column)
        {
            if (column.Top < 0)
            {
                return default;
            }

            int distanceToPlant = Manhattan(x, y, 0, 0);
            int distanceToSpoil = Manhattan(x, y, SpoilX(), SpoilY());

            long bestProfit = long.MinValue;
            int bestDepth = -1;
            long cumulativeRevenue = 0;
            long cumulativeCost = 0;

            for (int z = column.Top; z >= 0; z--)
            {
                byte d = column.Voxels[z];
                int mass = d & 15;
                int value = d >> 4;
                if (mass == 0)
                {
                    continue;
                }

                if (value > 0)
                {
                    cumulativeRevenue += RevenuePerValueMass * mass * value;
                    cumulativeCost += EstimateMoveCost(mass, distanceToPlant, z, _plantSurface);
                }
                else
                {
                    cumulativeCost += EstimateMoveCost(mass, distanceToSpoil, z, _spoilSurface);
                }

                long profit = cumulativeRevenue - cumulativeCost;
                if (profit > bestProfit && CanReachPlantFromHeight(z, distanceToPlant))
                {
                    bestProfit = profit;
                    bestDepth = z;
                }
            }

            return new ColumnPlan
            {
                X = x,
                Y = y,
                TargetDepth = bestDepth,
                EstimatedProfit = bestProfit,
                DistanceToPlant = distanceToPlant
            };
        }

        private void MineColumn(int x, int y, int targetDepth)
        {
            int index = IndexOf(x, y);
            var column = _columns[index];

            while (column.Top >= targetDepth && column.Top >= 0)
            {
                int z = column.Top;
                byte d = column.Voxels[z];
                int mass = d & 15;
                int value = d >> 4;
                column.Voxels[z] = 0;
                column.Top = FindNextTop(column.Voxels, z - 1);
                _surface[index] = column.Top;

                if (mass == 0)
                {
                    continue;
                }

                if (value > 0 && CanReachPlantFromHeight(z, Manhattan(x, y, 0, 0)))
                {
                    _operations.Add(BuildOperation(x, y, z, 0, 0));
                }
                else
                {
                    int spoilX = SpoilX();
                    int spoilY = SpoilY();
                    _operations.Add(BuildOperation(x, y, z, spoilX, spoilY));
                    _spoilSurface++;
                    _surface[_spoilIndex] = _spoilSurface;
                }
            }
        }

        private Operation BuildOperation(int sx, int sy, int sz, int dx, int dy)
        {
            var points = new List<Point>();
            int x = sx;
            int y = sy;

            while (x != dx)
            {
                x += x < dx ? 1 : -1;
                points.Add(new Point(x, y));
            }

            while (y != dy)
            {
                y += y < dy ? 1 : -1;
                points.Add(new Point(x, y));
            }

            int intermediateCount = Math.Max(0, points.Count - 1);
            return new Operation(sx, sy, sz, intermediateCount, points, dx, dy);
        }

        private bool CanReachPlantFromHeight(int sourceZ, int distance)
        {
            return sourceZ <= _plantSurface + distance;
        }

        internal static long EstimateMoveCost(int mass, int distance, int sourceZ, int destSurface)
        {
            if (distance == 0)
            {
                return 0;
            }

            int climb = Math.Max(0, destSurface - sourceZ);
            int flatOrDown = Math.Max(0, distance - climb);
            return (long)mass * (5L * climb + flatOrDown);
        }

        private void ChooseSpoilSite()
        {
            int bestIndex = _xCount * _yCount - 1;
            _spoilIndex = bestIndex;
            _spoilSurface = _surface[bestIndex];
        }

        private int SpoilX() => _spoilIndex / _yCount;
        private int SpoilY() => _spoilIndex % _yCount;

        private int IndexOf(int x, int y) => x * _yCount + y;

        private static int Manhattan(int ax, int ay, int bx, int by) => Math.Abs(ax - bx) + Math.Abs(ay - by);

        private static int FindNextTop(byte[] voxels, int start)
        {
            for (int z = start; z >= 0; z--)
            {
                if ((voxels[z] & 15) > 0)
                {
                    return z;
                }
            }

            return -1;
        }
    }

    private sealed class LargeWorldSolver : ISolver
    {
        private readonly int _xCount;
        private readonly int _yCount;
        private readonly int _zCount;
        private readonly List<SurfaceVoxel> _surfaceValuables = new List<SurfaceVoxel>();
        private readonly List<Operation> _operations = new List<Operation>();

        private int _plantSurface = -1;
        private int _spoilX;
        private int _spoilY;
        private int _spoilSurface = -1;

        public LargeWorldSolver(int xCount, int yCount, int zCount, double _, double __, double ___)
        {
            _xCount = xCount;
            _yCount = yCount;
            _zCount = zCount;
            _spoilX = xCount - 1;
            _spoilY = yCount - 1;
        }

        public void ReadWorld(FastInput input)
        {
            for (int x = 0; x < _xCount; x++)
            {
                for (int y = 0; y < _yCount; y++)
                {
                    int topZ = -1;
                    byte topByte = 0;
                    for (int z = 0; z < _zCount; z++)
                    {
                        byte d = input.NextByte();
                        if ((d & 15) > 0)
                        {
                            topZ = z;
                            topByte = d;
                        }
                    }

                    if (x == 0 && y == 0)
                    {
                        _plantSurface = topZ;
                    }

                    if (x == _spoilX && y == _spoilY)
                    {
                        _spoilSurface = topZ;
                    }

                    if (topZ >= 0)
                    {
                        int mass = topByte & 15;
                        int value = topByte >> 4;
                        if (value > 0 && mass > 0)
                        {
                            _surfaceValuables.Add(new SurfaceVoxel(x, y, topZ, mass, value));
                        }
                    }
                }
            }
        }

        public void Plan()
        {
            _surfaceValuables.Sort((a, b) =>
            {
                long pa = ProfitScore(a);
                long pb = ProfitScore(b);
                return pb.CompareTo(pa);
            });

            foreach (var voxel in _surfaceValuables)
            {
                int distance = Math.Abs(voxel.X) + Math.Abs(voxel.Y);
                if (voxel.Z > _plantSurface + distance)
                {
                    continue;
                }

                long revenue = RevenuePerValueMass * voxel.Mass * voxel.Value;
                long cost = SmallWorldSolver.EstimateMoveCost(voxel.Mass, distance, voxel.Z, _plantSurface);
                if (revenue <= cost)
                {
                    continue;
                }

                _operations.Add(BuildOperation(voxel.X, voxel.Y, voxel.Z, 0, 0));
            }
        }

        public void WriteOutput(TextWriter writer)
        {
            writer.WriteLine(_operations.Count);
            foreach (var op in _operations)
            {
                writer.WriteLine(op.ToOutputLine());
            }
        }

        private long ProfitScore(SurfaceVoxel voxel)
        {
            int distance = voxel.X + voxel.Y;
            long revenue = RevenuePerValueMass * voxel.Mass * voxel.Value;
            long cost = SmallWorldSolver.EstimateMoveCost(voxel.Mass, distance, voxel.Z, _plantSurface);
            return revenue - cost;
        }

        private static Operation BuildOperation(int sx, int sy, int sz, int dx, int dy)
        {
            var points = new List<Point>();
            int x = sx;
            int y = sy;

            while (x != dx)
            {
                x += x < dx ? 1 : -1;
                points.Add(new Point(x, y));
            }

            while (y != dy)
            {
                y += y < dy ? 1 : -1;
                points.Add(new Point(x, y));
            }

            int intermediateCount = Math.Max(0, points.Count - 1);
            return new Operation(sx, sy, sz, intermediateCount, points, dx, dy);
        }
    }

    private struct ColumnPlan
    {
        public int X;
        public int Y;
        public int TargetDepth;
        public long EstimatedProfit;
        public int DistanceToPlant;
    }

    private sealed class Column
    {
        public Column(byte[] voxels, int top)
        {
            Voxels = voxels;
            Top = top;
        }

        public byte[] Voxels { get; }
        public int Top { get; set; }
    }

    private readonly struct SurfaceVoxel
    {
        public SurfaceVoxel(int x, int y, int z, int mass, int value)
        {
            X = x;
            Y = y;
            Z = z;
            Mass = mass;
            Value = value;
        }

        public int X { get; }
        public int Y { get; }
        public int Z { get; }
        public int Mass { get; }
        public int Value { get; }
    }

    private readonly struct Point
    {
        public Point(int x, int y)
        {
            X = x;
            Y = y;
        }

        public int X { get; }
        public int Y { get; }
    }

    private sealed class Operation
    {
        private readonly int _digX;
        private readonly int _digY;
        private readonly int _digZ;
        private readonly int _stepCount;
        private readonly List<Point> _points;
        private readonly int _destX;
        private readonly int _destY;

        public Operation(int digX, int digY, int digZ, int stepCount, List<Point> points, int destX, int destY)
        {
            _digX = digX;
            _digY = digY;
            _digZ = digZ;
            _stepCount = stepCount;
            _points = points;
            _destX = destX;
            _destY = destY;
        }

        public string ToOutputLine()
        {
            var sb = new StringBuilder();
            sb.Append(_stepCount);
            sb.Append(' ');
            sb.Append(_digX);
            sb.Append(' ');
            sb.Append(_digY);
            sb.Append(' ');
            sb.Append(_digZ);

            for (int i = 0; i < _stepCount; i++)
            {
                sb.Append(' ');
                sb.Append(_points[i].X);
                sb.Append(' ');
                sb.Append(_points[i].Y);
            }

            sb.Append(' ');
            sb.Append(_destX);
            sb.Append(' ');
            sb.Append(_destY);
            return sb.ToString();
        }
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _position;

        public FastInput(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            SkipWhitespace();
            int sign = 1;
            int c = Read();
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        public double NextDouble()
        {
            SkipWhitespace();
            int c = Read();
            var sb = new StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = Read();
            }

            return double.Parse(sb.ToString(), System.Globalization.CultureInfo.InvariantCulture);
        }

        public byte NextByte()
        {
            int c = Read();
            if (c < 0)
            {
                throw new EndOfStreamException();
            }

            return (byte)c;
        }

        private void SkipWhitespace()
        {
            int c;
            do
            {
                c = Read();
            } while (c >= 0 && c <= 32);

            if (c >= 0)
            {
                _position--;
            }
        }

        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }
    }
}
