        if (poly.normal.dot(p_n) > 0) cp.coplanar_same = 1;
        else cp.coplanar_opp = 1;
        out.push_back(cp);
        return;
    }

        out.push_back(poly);
        return;
    }

    int n = poly.verts.size();

        Vector3 vi = poly.verts[i], vj = poly.verts[j];
        
        if (di >= 0) f.push_back(vi);
        if (di <= 0) b.push_back(vi);
        
        if ((di > 0 && dj < 0) || (di < 0 && dj > 0)) {
            double t = di / (di - dj);
            Vector3 v = vi + (vj - vi) * t;
            f.push_back(v);
            b.push_back(v);
        }
    }
    
    if (f.size() >= 3) {
        Polygon pf; pf.verts = f; pf.normal = poly.normal; 
        pf.coplanar_same = poly.coplanar_same; pf.coplanar_opp = poly.coplanar_opp;

fn try_place(
    piece: &[(f64, f64)], 
    stock_poly: &[(f64, f64)], 
    placed: &[PlacedPiece], 
    grid_res: usize, 
    rotations: &[f64],
    stock_is_rect: bool
) -> Option<(f64, f64, f64, Vec<(f64, f64)>)> {
    let (s_min_x, s_max_x, s_min_y, s_max_y) = bounding_box(stock_poly);
    
    let mut candidates = Vec::new();
    let mut rotated_polys = Vec::new();
    
    for (rot_idx, &rot) in rotations.iter().enumerate() {
        let rotated = rotate_polygon(piece, rot);
        let (p_min_x, p_max_x, p_min_y, p_max_y) = bounding_box(&rotated);
        
        rotated_polys.push(rotated);
        
        let dx_min = s_min_x - p_min_x;
        let dx_max = s_max_x - p_max_x;
        let dy_min = s_min_y - p_min_y;
        let dy_max = s_max_y - p_max_y;
        
        if dx_min > dx_max + 1e-7 || dy_min > dy_max + 1e-7 {
            continue;
        }
        
        let dx_max = dx_max.max(dx_min);
        let dy_max = dy_max.max(dy_min);
        
        for i in 0..=grid_res {
            let dx = if grid_res == 0 { dx_min } else { dx_min + (dx_max - dx_min) * (i as f64 / grid_res as f64) };
            for j in 0..=grid_res {
                let dy = if grid_res == 0 { dy_min } else { dy_min + (dy_max - dy_min) * (j as f64 / grid_res as f64) };
                candidates.push((dx, dy, rot_idx));
            }
        }
    }
    
    candidates.sort_by(|a, b| {
        a.1.partial_cmp(&b.1).unwrap().then(a.0.partial_cmp(&b.0).unwrap())
    });
    
    for (dx, dy, rot_idx) in candidates {
        let rot = rotations[rot_idx];
        let rotated = &rotated_polys[rot_idx];
        
        let mut inside_stock = true;
        let mut translated = Vec::new();
        
        if !stock_is_rect {
            translated = translate_polygon(rotated, dx, dy);
            inside_stock = polygon_inside(&translated, stock_poly);
        }
        
        if inside_stock {
            if translated.is_empty() {
                translated = translate_polygon(rotated, dx, dy);
            }
            let bb_trans = bounding_box(&translated);
            let mut overlap = false;
            for pl in placed {
                if polygons_overlap_fast(&translated, bb_trans, pl) {
                    overlap = true;
                    break;
                }
            }
            if !overlap {
                return Some((dx, dy, rot, translated));
            }
        }
    }
    None
}

fn cross_product(p0: (f64, f64), p1: (f64, f64), p2: (f64, f64)) -> f64 {
    (p1.0 - p0.0) * (p2.1 - p0.1) - (p1.1 - p0.1) * (p2.0 - p0.0)
}

fn segments_intersect(p1: (f64, f64), p2: (f64, f64), p3: (f64, f64), p4: (f64, f64)) -> bool {
    let cp1 = cross_product(p3, p4, p1);
    let cp2 = cross_product(p3, p4, p2);
    let cp3 = cross_product(p1, p2, p3);
    let cp4 = cross_product(p1, p2, p4);

    let eps = 1e-7;

    if ((cp1 > eps && cp2 < -eps) || (cp1 < -eps && cp2 > eps)) &&
       ((cp3 > eps && cp4 < -eps) || (cp3 < -eps && cp4 > eps)) {
        return true;
    }
    false
}

fn point_in_or_on_polygon(p: (f64, f64), poly: &[(f64, f64)]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    let eps = 1e-7;

    for i in 0..poly.len() {
        let pi = poly[i];
        let pj = poly[j];

        let min_x = pi.0.min(pj.0);
        let max_x = pi.0.max(pj.0);
        let min_y = pi.1.min(pj.1);
        let max_y = pi.1.max(pj.1);
        
        if p.0 >= min_x - eps && p.0 <= max_x + eps && p.1 >= min_y - eps && p.1 <= max_y + eps {
            let cp = (p.0 - pi.0) * (pj.1 - pi.1) - (p.1 - pi.1) * (pj.0 - pi.0);
            if cp.abs() < eps {
                return true;
            }
        }

        if (pi.1 > p.1) != (pj.1 > p.1) {
            let xinters = (pj.0 - pi.0) * (p.1 - pi.1) / (pj.1 - pi.1) + pi.0;
            if p.0 < xinters {
                inside = !inside;
            }
        }
        j = i;
    }
    inside
}

fn polygons_overlap_fast(poly1: &[(f64, f64)], bb1: (f64, f64, f64, f64), placed: &PlacedPiece) -> bool {
    let bb2 = placed.bbox;
    let eps = 1e-7;
    if bb1.0 > bb2.1 - eps || bb1.1 < bb2.0 + eps || bb1.2 > bb2.3 - eps || bb1.3 < bb2.2 + eps {
        return false;

    let poly2 = &placed.polygon;
    for i in 0..poly1.len() {
        let p1 = poly1[i];
        let p2 = poly1[(i + 1) % poly1.len()];
        for j in 0..poly2.len() {
            let p3 = poly2[j];
            let p4 = poly2[(j + 1) % poly2.len()];
            if segments_intersect(p1, p2, p3, p4) {
                return true;
            }
        }

    if any_point_strictly_inside(poly1, poly2) {
        return true;

}

fn polygon_inside(inner: &[(f64, f64)], outer: &[(f64, f64)]) -> bool {
    let bbi = bounding_box(inner);
    let bbo = bounding_box(outer);
    let eps = 1e-7;
    if bbi.0 < bbo.0 - eps || bbi.1 > bbo.1 + eps || bbi.2 < bbo.2 - eps || bbi.3 > bbo.3 + eps {
        return false;
    }

    for p in inner {
        if !point_in_or_on_polygon(*p, outer) {
            return false;
        }
    }
    
    for i in 0..inner.len() {
        let p1 = inner[i];
        let p2 = inner[(i + 1) % inner.len()];
        for j in 0..outer.len() {
            let p3 = outer[j];
            let p4 = outer[(j + 1) % outer.len()];
            if segments_intersect(p1, p2, p3, p4) {
                return false;
            }
        }
    }
    true
}

fn bounding_box(poly: &[(f64, f64)]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;
    for &(x, y) in poly {
        if x < min_x { min_x = x; }
        if x > max_x { max_x = x; }
        if y < min_y { min_y = y; }
        if y > max_y { max_y = y; }
    }
    (min_x, max_x, min_y, max_y)
}

fn polygon_area(poly: &[(f64, f64)]) -> f64 {
    let mut area = 0.0;
    for i in 0..poly.len() {
        let p1 = poly[i];
        let p2 = poly[(i + 1) % poly.len()];
        area += p1.0 * p2.1 - p2.0 * p1.1;
    }
    (area / 2.0).abs()
}

fn is_axis_aligned_rectangle(poly: &[(f64, f64)]) -> bool {
    if poly.len() != 4 {

    let mut xs = vec![poly[0].0, poly[1].0, poly[2].0, poly[3].0];
    let mut ys = vec![poly[0].1, poly[1].1, poly[2].1, poly[3].1];

}

fn rotate_polygon(poly: &[(f64, f64)], rot: f64) -> Vec<(f64, f64)> {
    let cos_r = rot.cos();
    let sin_r = rot.sin();
    poly.iter().map(|&(x, y)| {
        (x * cos_r - y * sin_r, x * sin_r + y * cos_r)
    }).collect()
}

fn translate_polygon(poly: &[(f64, f64)], dx: f64, dy: f64) -> Vec<(f64, f64)> {
    poly.iter().map(|&(x, y)| (x + dx, y + dy)).collect()
}

fn try_place(
    piece: &[(f64, f64)], 
    stock_poly: &[(f64, f64)], 
    placed: &[PlacedPiece], 
    grid_res: usize, 
    rotations: &[f64],
    stock_is_rect: bool
) -> Option<(f64, f64, f64, Vec<(f64, f64)>)> {
    let (s_min_x, s_max_x, s_min_y, s_max_y) = bounding_box(stock_poly);
    
    let mut candidates = Vec::new();
    let mut rotated_polys = Vec::new();
    
    for (rot_idx, &rot) in rotations.iter().enumerate() {
        let rotated = rotate_polygon(piece, rot);
        let (p_min_x, p_max_x, p_min_y, p_max_y) = bounding_box(&rotated);
        
        rotated_polys.push(rotated);
        
        let dx_min = s_min_x - p_min_x;
        let dx_max = s_max_x - p_max_x;
        let dy_min = s_min_y - p_min_y;
        let dy_max = s_max_y - p_max_y;
        
        if dx_min > dx_max + 1e-7 || dy_min > dy_max + 1e-7 {
            continue;
        }
        
        let dx_max = dx_max.max(dx_min);
        let dy_max = dy_max.max(dy_min);
        
        for i in 0..=grid_res {
            let dx = if grid_res == 0 { dx_min } else { dx_min + (dx_max - dx_min) * (i as f64 / grid_res as f64) };
            for j in 0..=grid_res {
                let dy = if grid_res == 0 { dy_min } else { dy_min + (dy_max - dy_min) * (j as f64 / grid_res as f64) };
                candidates.push((dx, dy, rot_idx));
            }
        }
    }
    
    candidates.sort_by(|a, b| {
        a.1.partial_cmp(&b.1).unwrap().then(a.0.partial_cmp(&b.0).unwrap())
    });
    
    for (dx, dy, rot_idx) in candidates {
        let rot = rotations[rot_idx];
        let rotated = &rotated_polys[rot_idx];
        
        let mut inside_stock = true;
        let mut translated = Vec::new();
        
        if !stock_is_rect {
            translated = translate_polygon(rotated, dx, dy);
            inside_stock = polygon_inside(&translated, stock_poly);

            if translated.is_empty() {
                translated = translate_polygon(rotated, dx, dy);
            }
            let bb_trans = bounding_box(&translated);
            let mut overlap = false;
            for pl in placed {
                if polygons_overlap_fast(&translated, bb_trans, pl) {
                    overlap = true;
                    break;
                }
            }

                return Some((dx, dy, rot, translated));
            }
        }
    }
    None
}

        public Vector3 Rotate(Vector3 v)
        {
            double tx = 2 * (Y * v.Z - Z * v.Y);

void compute_work_remaining() {
    work_remaining.assign(J, vector<int>());
    for (int j = 0; j < J; ++j) {
        work_remaining[j].resize(jobs[j].size());
        int rem = 0;
        for (int t = (int)jobs[j].size() - 1; t >= 0; --t) {

            work_remaining[j][t] = rem;
        }
    }

using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program
{
    struct Dimensions
    {
        public long w, h, d;
        public Dimensions(long w, long h, long d)
        {
            this.w = w; this.h = h; this.d = d;
        }
    }

    struct Space
    {
        public long x, y, z, w, h, d;
        public Space(long x, long y, long z, long w, long h, long d)
        {
            this.x = x; this.y = y; this.z = z;
            this.w = w; this.h = h; this.d = d;
        }
        public double Volume => (double)w * h * d;
    }

    class Box
    {
        public int id;
        public long w, h, d;
        public double Volume => (double)w * h * d;
        public Dimensions[] rotations;
    }

    class PlacedBox
    {
        public int id;
        public long x, y, z;
    }

    static Dimensions[] GetRotations(long w, long h, long d)
    {
        var set = new HashSet<(long, long, long)>
        {
            (w, h, d),
            (w, d, h),
            (h, w, d),
            (h, d, w),
            (d, w, h),
            (d, h, w)
        };

        var list = new List<Dimensions>();
        foreach (var item in set)
        {
            list.Add(new Dimensions(item.Item1, item.Item2, item.Item3));
        }
        // Pack as flat as possible (minimize Z, then Y, then X)
        return list.OrderBy(r => r.d).ThenBy(r => r.h).ThenBy(r => r.w).ToArray();
    }

    static Space[] ChooseBestSplit(Space s, long bw, long bh, long bd)
    {
        Space[][] splits = new Space[6][];
        
        long dx = s.w - bw;
        long dy = s.h - bh;
        long dz = s.d - bd;

        splits[0] = new[] {
            new Space(s.x + bw, s.y, s.z, dx, s.h, s.d),
            new Space(s.x, s.y + bh, s.z, bw, dy, s.d),
            new Space(s.x, s.y, s.z + bd, bw, bh, dz)
        };
        splits[1] = new[] {
            new Space(s.x + bw, s.y, s.z, dx, s.h, s.d),
            new Space(s.x, s.y, s.z + bd, bw, s.h, dz),
            new Space(s.x, s.y + bh, s.z, bw, dy, bd)
        };
        splits[2] = new[] {
            new Space(s.x, s.y + bh, s.z, s.w, dy, s.d),
            new Space(s.x + bw, s.y, s.z, dx, bh, s.d),
            new Space(s.x, s.y, s.z + bd, bw, bh, dz)
        };
        splits[3] = new[] {
            new Space(s.x, s.y + bh, s.z, s.w, dy, s.d),
            new Space(s.x, s.y, s.z + bd, s.w, bh, dz),
            new Space(s.x + bw, s.y, s.z, dx, bh, bd)
        };
        splits[4] = new[] {
            new Space(s.x, s.y, s.z + bd, s.w, s.h, dz),
            new Space(s.x + bw, s.y, s.z, dx, s.h, bd),
            new Space(s.x, s.y + bh, s.z, bw, dy, bd)
        };
        splits[5] = new[] {
            new Space(s.x, s.y, s.z + bd, s.w, s.h, dz),
            new Space(s.x, s.y + bh, s.z, s.w, dy, bd),
            new Space(s.x + bw, s.y, s.z, dx, bh, bd)
        };

        int bestSplit = 0;
        double bestScore = -1;
        
        // Pick the split that maximizes the volume of its largest single space
        for (int i = 0; i < 6; i++)
        {
            double maxVol = 0;
            foreach (var space in splits[i])
            {
                double vol = space.Volume;
                if (vol > maxVol) maxVol = vol;
            }
            if (maxVol > bestScore)
            {
                bestScore = maxVol;
                bestSplit = i;
            }
        }
        
        return splits[bestSplit];
    }

    static void Main()
    {
        using var reader = new StreamReader(Console.OpenStandardInput(), bufferSize: 65536);
        string line = reader.ReadLine();
        if (string.IsNullOrWhiteSpace(line)) return;
        
        string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts[0]);
        long W = long.Parse(parts[1]);
        long H = long.Parse(parts[2]);
        long D = long.Parse(parts[3]);

        var boxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            line = reader.ReadLine();
            if (string.IsNullOrWhiteSpace(line))
            {
                i--;
                continue;
            }
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            long w = long.Parse(parts[0]);
            long h = long.Parse(parts[1]);
            long d = long.Parse(parts[2]);
            
            boxes.Add(new Box {
                id = i,
                w = w,
                h = h,
                d = d,
                rotations = GetRotations(w, h, d)
            });
        }

        // Sort by volume ascending to maximize number of boxes packed.
        // Tie-break by maximum dimension to favor more cubic boxes first.
        boxes = boxes.OrderBy(b => b.Volume).ThenBy(b => Math.Max(b.w, Math.Max(b.h, b.d))).ToList();

        var spaces = new List<Space>();
        spaces.Add(new Space(0, 0, 0, W, H, D));

        var placedBoxes = new List<PlacedBox>();

        foreach (var box in boxes)
        {
            double minVolumeRemaining = box.Volume;

            // Purge spaces that are too small for this box and all future boxes
            for (int i = spaces.Count - 1; i >= 0; i--)
            {
                if (spaces[i].Volume < minVolumeRemaining)
                {
                    spaces[i] = spaces[spaces.Count - 1];
                    spaces.RemoveAt(spaces.Count - 1);
                }
            }

            Space bestSpace = default;
            bool found = false;
            Dimensions bestRot = default;
            double bestScore = double.MaxValue;
            int bestSpaceIndex = -1;

            // Find best fit space
            for (int i = 0; i < spaces.Count; i++)
            {
                Space s = spaces[i];
                foreach (var rot in box.rotations)
                {
                    if (rot.w <= s.w && rot.h <= s.h && rot.d <= s.d)
                    {
                        double score = s.Volume;
                        if (score < bestScore)
                        {
                            bestScore = score;
                            bestSpace = s;
                            bestRot = rot;
                            bestSpaceIndex = i;
                            found = true;
                        }
                        else if (score == bestScore && found)
                        {
                            // Tie-breaker: Bottom-Left-Back
                            if (s.z < bestSpace.z || 
                               (s.z == bestSpace.z && s.y < bestSpace.y) || 
                               (s.z == bestSpace.z && s.y == bestSpace.y && s.x < bestSpace.x))
                            {
                                bestSpace = s;
                                bestRot = rot;
                                bestSpaceIndex = i;
                            }
                        }
                    }
                }
            }

            if (found)
            {
                placedBoxes.Add(new PlacedBox { id = box.id, x = bestSpace.x, y = bestSpace.y, z = bestSpace.z });

                // Remove used space in O(1)
                spaces[bestSpaceIndex] = spaces[spaces.Count - 1];
                spaces.RemoveAt(spaces.Count - 1);

                // Add partitioned remaining spaces
                Space[] newSpaces = ChooseBestSplit(bestSpace, bestRot.w, bestRot.h, bestRot.d);

                foreach (var ns in newSpaces)
                {
                    if (ns.w > 0 && ns.h > 0 && ns.d > 0 && ns.Volume >= minVolumeRemaining)
                    {
                        spaces.Add(ns);
                    }
                }
            }
        }

        // Restore original index ordering for deterministic output
        placedBoxes = placedBoxes.OrderBy(pb => pb.id).ToList();

        using var writer = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536);
        writer.WriteLine(placedBoxes.Count);
        foreach (var pb in placedBoxes)
        {
            writer.WriteLine($"{pb.id} {pb.x} {pb.y} {pb.z}");
        }
    }
