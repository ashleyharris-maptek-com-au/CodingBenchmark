using System;
using System.IO;
using System.Collections.Generic;

class Program
{
    struct EdgeInfo {
        public int u, v, w, id;
    }

    struct MultiEdge {
        public int to, id;
    }

    struct Pair : IComparable<Pair> {
        public int i, j;
        public long dist;
        public int CompareTo(Pair other) {
            int c = dist.CompareTo(other.dist);
            if (c != 0) return c;
            c = i.CompareTo(other.i);
            if (c != 0) return c;
            return j.CompareTo(other.j);
        }
    }

    class FastScanner
    {
        private Stream stream;
        private byte[] buffer = new byte[32768];
        private int head = 0;
        private int tail = 0;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        private int Read()
        {
            if (head >= tail)
            {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }

        public int NextInt()
        {
            int c = Read();
            while (c <= 32)
            {
                if (c == -1) return -1;
                c = Read();
            }
            bool minus = false;
            if (c == '-')
            {
                minus = true;
                c = Read();
            }
            int res = 0;
            while (c > 32)
            {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return minus ? -res : res;
        }
    }

    static void Main()
    {
        using Stream inputStream = Console.OpenStandardInput();
        FastScanner scanner = new FastScanner(inputStream);

        int N = scanner.NextInt();
        if (N == -1) return; // EOF
        int M = scanner.NextInt();

        EdgeInfo[] originalEdges = new EdgeInfo[M];
        int[] degree = new int[N];

        for (int i = 0; i < M; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            int w = scanner.NextInt();
            originalEdges[i] = new EdgeInfo { u = u, v = v, w = w, id = i };
            degree[u]++;
            degree[v]++;
        }

        List<int> oddNodes = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (degree[i] % 2 != 0)
            {
                oddNodes.Add(i);
            }
        }

        int K = oddNodes.Count;
        List<EdgeInfo> duplicatedEdges = new List<EdgeInfo>();

        if (K > 0)
        {
            if (K <= 20)
            {
                RunExactDP(N, M, K, originalEdges, oddNodes, duplicatedEdges);
            }
            else if ((long)K * M <= 20_000_000)
            {
                RunGreedy(N, M, K, originalEdges, oddNodes, duplicatedEdges);
            }
            else
            {
                RunMST(N, M, originalEdges, degree, duplicatedEdges);
            }
        }

        RunHierholzer(N, M, originalEdges, duplicatedEdges);
    }

    static void BuildAdj(int N, int M, EdgeInfo[] originalEdges, out List<EdgeInfo>[] adj)
    {
        adj = new List<EdgeInfo>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<EdgeInfo>();
        foreach (var e in originalEdges)
        {
            adj[e.u].Add(new EdgeInfo { u = e.u, v = e.v, w = e.w, id = e.id });
            adj[e.v].Add(new EdgeInfo { u = e.v, v = e.u, w = e.w, id = e.id });
        }
    }

    static void RunDijkstra(int N, int startNode, List<EdgeInfo>[] adj, out long[] dist, out int[] prevEdge)
    {
        dist = new long[N];
        Array.Fill(dist, long.MaxValue);
        prevEdge = new int[N];
        Array.Fill(prevEdge, -1);
        
        var pq = new PriorityQueue<int, long>();
        dist[startNode] = 0;
        pq.Enqueue(startNode, 0);
        
        while (pq.Count > 0)
        {
            pq.TryDequeue(out int u, out long d);
            if (d > dist[u]) continue;
            
            foreach (var edge in adj[u])
            {
                if (dist[u] + edge.w < dist[edge.to])
                {
                    dist[edge.to] = dist[u] + edge.w;
                    prevEdge[edge.to] = edge.id;
                    pq.Enqueue(edge.to, dist[edge.to]);
                }
            }
        }
    }

    static void RunExactDP(int N, int M, int K, EdgeInfo[] originalEdges, List<int> oddNodes, List<EdgeInfo> duplicatedEdges)
    {
        BuildAdj(N, M, originalEdges, out var adj);
        long[][] distFromOdd = new long[K][];
        int[][] prevFromOdd = new int[K][];
        
        for (int i = 0; i < K; i++) {
            RunDijkstra(N, oddNodes[i], adj, out distFromOdd[i], out prevFromOdd[i]);
        }
        
        long[] dp = new long[1 << K];
        Array.Fill(dp, long.MaxValue);
        dp[0] = 0;
        
        int[] parentMask = new int[1 << K];
        int[] matchI = new int[1 << K];
        int[] matchJ = new int[1 << K];
        
        for (int mask = 0; mask < (1 << K); mask++) {
            if (dp[mask] == long.MaxValue) continue;
            
            int i = -1;
            for (int bit = 0; bit < K; bit++) {
                if ((mask & (1 << bit)) == 0) {
                    i = bit;
                    break;
                }
            }
            if (i == -1) continue;
            
            for (int j = i + 1; j < K; j++) {
                if ((mask & (1 << j)) == 0) {
                    if (distFromOdd[i][oddNodes[j]] != long.MaxValue) {
                        int newMask = mask | (1 << i) | (1 << j);
                        long newCost = dp[mask] + distFromOdd[i][oddNodes[j]];
                        if (newCost < dp[newMask]) {
                            dp[newMask] = newCost;
                            parentMask[newMask] = mask;
                            matchI[newMask] = i;
                            matchJ[newMask] = j;
                        }
                    }
                }
            }
        }
        
        int currMask = (1 << K) - 1;
        while (currMask > 0) {
            int i = matchI[currMask];
            int j = matchJ[currMask];
            
            int currNode = oddNodes[j];
            int startNode = oddNodes[i];
            while (currNode != startNode) {
                int edgeId = prevFromOdd[i][currNode];
                EdgeInfo e = originalEdges[edgeId];
                duplicatedEdges.Add(e);
                currNode = (e.u == currNode) ? e.v : e.u;
            }
            
            currMask = parentMask[currMask];
        }
    }

    static void RunGreedy(int N, int M, int K, EdgeInfo[] originalEdges, List<int> oddNodes, List<EdgeInfo> duplicatedEdges)
    {
        BuildAdj(N, M, originalEdges, out var adj);
        long[][] distFromOdd = new long[K][];
        int[][] prevFromOdd = new int[K][];
        
        for (int i = 0; i < K; i++) {
            RunDijkstra(N, oddNodes[i], adj, out distFromOdd[i], out prevFromOdd[i]);
        }
        
        List<Pair> pairs = new List<Pair>();
        for (int i = 0; i < K; i++) {
            for (int j = i + 1; j < K; j++) {
                if (distFromOdd[i][oddNodes[j]] != long.MaxValue) {
                    pairs.Add(new Pair { i = i, j = j, dist = distFromOdd[i][oddNodes[j]] });
                }
            }
        }
        pairs.Sort();
        
        bool[] matched = new bool[K];
        foreach (var pair in pairs) {
            if (!matched[pair.i] && !matched[pair.j]) {
                matched[pair.i] = true;
                matched[pair.j] = true;
                
                int currNode = oddNodes[pair.j];
                int startNode = oddNodes[pair.i];
                while (currNode != startNode) {
                    int edgeId = prevFromOdd[pair.i][currNode];
                    EdgeInfo e = originalEdges[edgeId];
                    duplicatedEdges.Add(e);
                    currNode = (e.u == currNode) ? e.v : e.u;
                }
            }
        }
    }

    static void RunMST(int N, int M, EdgeInfo[] originalEdges, int[] degree, List<EdgeInfo> duplicatedEdges)
    {
        int[] weights = new int[M];
        int[] edgeIndices = new int[M];
        for (int i = 0; i < M; i++) {
            weights[i] = originalEdges[i].w;
            edgeIndices[i] = i;
        }
        
        Array.Sort(weights, edgeIndices);
        
        int[] parentUF = new int[N];
        for (int i = 0; i < N; i++) parentUF[i] = i;
        
        int Find(int i) {
            int root = i;
            while (root != parentUF[root]) root = parentUF[root];
            int curr = i;
            while (curr != root) {
                int nxt = parentUF[curr];
                parentUF[curr] = root;
                curr = nxt;
            }
            return root;
        }
        
        void Union(int i, int j) {
            int rootI = Find(i);
            int rootJ = Find(j);
            if (rootI != rootJ) parentUF[rootI] = rootJ;
        }
        
        List<EdgeInfo>[] treeAdj = new List<EdgeInfo>[N];
        for (int i = 0; i < N; i++) treeAdj[i] = new List<EdgeInfo>();
        
        foreach (int id in edgeIndices) {
            EdgeInfo e = originalEdges[id];
            if (Find(e.u) != Find(e.v)) {
                Union(e.u, e.v);
                treeAdj[e.u].Add(e);
                treeAdj[e.v].Add(e);
            }
        }
        
        bool[] parity = new bool[N];
        for (int i = 0; i < N; i++) parity[i] = (degree[i] % 2 != 0);
        
        bool[] visited = new bool[N];
        int[] parent = new int[N];
        int[] order = new int[N];
        int orderIdx = 0;
        int[] q = new int[N];
        
        for (int i = 0; i < N; i++) {
            if (!visited[i] && treeAdj[i].Count > 0) {
                int head = 0, tail = 0;
                q[tail++] = i;
                visited[i] = true;
                parent[i] = -1;
                
                while (head < tail) {
                    int u = q[head++];
                    order[orderIdx++] = u;
                    
                    foreach (var e in treeAdj[u]) {
                        int v = (e.u == u) ? e.v : e.u;
                        if (!visited[v]) {
                            visited[v] = true;
                            parent[v] = u;
                            q[tail++] = v;
                        }
                    }
                }
            }
        }
        
        for (int i = orderIdx - 1; i >= 0; i--) {
            int u = order[i];
            if (parent[u] != -1 && parity[u]) {
                EdgeInfo pEdge = default;
                foreach (var e in treeAdj[u]) {
                    int v = (e.u == u) ? e.v : e.u;
                    if (v == parent[u]) {
                        pEdge = e;
                        break;
                    }
                }
                duplicatedEdges.Add(pEdge);
                parity[u] = !parity[u];
                parity[parent[u]] = !parity[parent[u]];
            }
        }
    }

    static void RunHierholzer(int N, int M, EdgeInfo[] originalEdges, List<EdgeInfo> duplicatedEdges)
    {
        if (M == 0) {
            Console.WriteLine(0);
            return;
        }
        
        List<MultiEdge>[] multiAdj = new List<MultiEdge>[N];
        for (int i = 0; i < N; i++) multiAdj[i] = new List<MultiEdge>();
        
        int nextEdgeId = 0;
        void AddMulti(int u, int v) {
            multiAdj[u].Add(new MultiEdge { to = v, id = nextEdgeId });
            multiAdj[v].Add(new MultiEdge { to = u, id = nextEdgeId });
            nextEdgeId++;
        }
        
        foreach (var e in originalEdges) AddMulti(e.u, e.v);
        foreach (var e in duplicatedEdges) AddMulti(e.u, e.v);
        
        bool[] used = new bool[nextEdgeId];
        int[] head = new int[N];
        
        Stack<int> currPath = new Stack<int>();
        List<int> circuit = new List<int>();
        
        currPath.Push(0);
        while (currPath.Count > 0)
        {
            int u = currPath.Peek();
            bool found = false;
            while (head[u] < multiAdj[u].Count)
            {
                var edge = multiAdj[u][head[u]];
                head[u]++;
                if (!used[edge.id])
                {
                    used[edge.id] = true;
                    currPath.Push(edge.to);
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                circuit.Add(u);
                currPath.Pop();
            }
        }
        
        circuit.Reverse();
        
        using (StreamWriter sw = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536))
        {
            for (int i = 0; i < circuit.Count; i++)
            {
                sw.Write(circuit[i]);
                if (i < circuit.Count - 1) sw.Write(' ');
            }
            sw.WriteLine();
        }
    }
}
