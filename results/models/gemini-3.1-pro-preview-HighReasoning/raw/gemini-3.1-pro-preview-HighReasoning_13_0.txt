#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

#ifdef _MSC_VER
#include <intrin.h>
#pragma intrinsic(_umul128)
#endif

using namespace std;

// We use two hashes to simulate a collision-free 128-bit hash.
// Hash 1 uses modulo 2^61 - 1, Hash 2 uses 64-bit unsigned integer overflow.
const uint64_t MOD = (1ULL << 61) - 1;

// Fast modular multiplication using MSVC intrinsic if available.
inline uint64_t mul_mod(uint64_t a, uint64_t b) {
#ifdef _MSC_VER
    uint64_t high;
    uint64_t low = _umul128(a, b, &high);
    uint64_t c = (low & MOD) + (low >> 61) + (high << 3);
    c = (c & MOD) + (c >> 61);
    if (c >= MOD) c -= MOD;
    return c;
#else
    unsigned __int128 p = (unsigned __int128)a * b;
    uint64_t c = (uint64_t)(p >> 61) + (uint64_t)(p & MOD);
    if (c >= MOD) c -= MOD;
    return c;
#endif
}

inline uint64_t add_mod(uint64_t a, uint64_t b) {
    uint64_t c = a + b;
    if (c >= MOD) c -= MOD;
    return c;
}

inline uint64_t sub_mod(uint64_t a, uint64_t b) {
    uint64_t c = a + MOD - b;
    if (c >= MOD) c -= MOD;
    return c;
}

struct Hash128 {
    uint64_t h1; 
    uint64_t h2; 

    bool operator<(const Hash128& o) const {
        if (h1 != o.h1) return h1 < o.h1;
        return h2 < o.h2;
    }
    bool operator==(const Hash128& o) const {
        return h1 == o.h1 && h2 == o.h2;
    }
};

struct HashPos {
    Hash128 hash;
    uint32_t pos;
};

const uint64_t BASE1 = 313;
const uint64_t BASE2 = 317;

struct Solver {
    // 2^28 bits = 32MB Bloom filter to drastically reduce binary search lookups
    // This provides our adaptive logic: it gracefully scales down operations when
    // comparing against massive 10MB+ strings by pre-filtering candidate hashes.
    static constexpr uint32_t FILTER_BITS = 28;
    std::vector<HashPos> candidates;
    std::vector<HashPos> next_cands;
    std::vector<bool> found;
    std::vector<uint32_t> bloom;

    Solver() : bloom(1 << (FILTER_BITS - 5), 0) {}

    int Test(int L, const std::vector<std::string>& S) {
        if (L == 0) return -1;
        
        uint64_t P1 = 1, P2 = 1;
        for (int i = 0; i < L; ++i) {
            P1 = mul_mod(P1, BASE1);
            P2 = P2 * BASE2;
        }

        candidates.clear();
        if (S[0].length() < (size_t)L) return -1;
        
        uint64_t h1 = 0, h2 = 0;
        for (int i = 0; i < L; ++i) {
            uint8_t c = (uint8_t)S[0][i];
            h1 = add_mod(mul_mod(h1, BASE1), c);
            h2 = h2 * BASE2 + c;
        }
        candidates.push_back({{h1, h2}, 0});

        for (int i = L; i < (int)S[0].length(); ++i) {
            uint8_t c_in = (uint8_t)S[0][i];
            uint8_t c_out = (uint8_t)S[0][i-L];

            h1 = add_mod(mul_mod(h1, BASE1), c_in);
            uint64_t drop1 = mul_mod(c_out, P1);
            h1 = sub_mod(h1, drop1);

            h2 = h2 * BASE2 + c_in;
            uint64_t drop2 = c_out * P2;
            h2 = h2 - drop2;

            candidates.push_back({{h1, h2}, (uint32_t)(i - L + 1)});
        }

        std::sort(candidates.begin(), candidates.end(), [](const HashPos& a, const HashPos& b) {
            return a.hash < b.hash;
        });

        auto it = std::unique(candidates.begin(), candidates.end(), [](const HashPos& a, const HashPos& b) {
            return a.hash == b.hash;
        });
        candidates.erase(it, candidates.end());

        for (size_t k = 1; k < S.size(); ++k) {
            if (candidates.empty()) return -1;

            std::fill(bloom.begin(), bloom.end(), 0);
            for (const auto& c : candidates) {
                uint32_t idx_bloom = c.hash.h1 & ((1 << FILTER_BITS) - 1);
                bloom[idx_bloom >> 5] |= (1U << (idx_bloom & 31));
            }

            found.assign(candidates.size(), false);
            int num_found = 0;
            
            if (S[k].length() < (size_t)L) return -1;

            uint64_t ch1 = 0, ch2 = 0;
            for (int i = 0; i < L; ++i) {
                uint8_t c = (uint8_t)S[k][i];
                ch1 = add_mod(mul_mod(ch1, BASE1), c);
                ch2 = ch2 * BASE2 + c;
            }
            
            auto check_and_mark = [&]() {
                uint32_t idx_bloom = ch1 & ((1 << FILTER_BITS) - 1);
                if (bloom[idx_bloom >> 5] & (1U << (idx_bloom & 31))) {
                    Hash128 cur = {ch1, ch2};
                    auto it = std::lower_bound(candidates.begin(), candidates.end(), HashPos{cur, 0}, [](const HashPos& a, const HashPos& b) {
                        return a.hash < b.hash;
                    });
                    if (it != candidates.end() && it->hash == cur) {
                        size_t idx = it - candidates.begin();
                        if (!found[idx]) {
                            found[idx] = true;
                            num_found++;
                        }
                    }
                }
            };
            
            check_and_mark();

            for (int i = L; i < (int)S[k].length(); ++i) {
                // Adaptive early exit: if all candidates found, skip rest of string
                if ((size_t)num_found == candidates.size()) break;

                uint8_t c_in = (uint8_t)S[k][i];
                uint8_t c_out = (uint8_t)S[k][i-L];

                ch1 = add_mod(mul_mod(ch1, BASE1), c_in);
                uint64_t drop1 = mul_mod(c_out, P1);
                ch1 = sub_mod(ch1, drop1);

                ch2 = ch2 * BASE2 + c_in;
                uint64_t drop2 = c_out * P2;
                ch2 = ch2 - drop2;

                check_and_mark();
            }

            next_cands.clear();
            for (size_t i = 0; i < candidates.size(); ++i) {
                if (found[i]) {
                    next_cands.push_back(candidates[i]);
                }
            }
            candidates = next_cands;
        }

        if (candidates.empty()) return -1;
        return candidates[0].pos;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy); // consume newline

    vector<string> S(N);
    for(int i = 0; i < N; ++i) {
        getline(cin, S[i]);
        // Handle Windows \r\n safely
        if (!S[i].empty() && S[i].back() == '\r') {
            S[i].pop_back();
        }
    }
    
    if (N == 0) {
        cout << "\n";
        return 0;
    }
    if (N == 1) {
        cout << S[0] << "\n";
        return 0;
    }

    // Sort to optimize candidate extraction from the shortest string
    std::sort(S.begin(), S.end(), [](const string& a, const string& b) {
        return a.length() < b.length();
    });

    Solver solver;
    int low = 1, high = (int)S[0].length();
    int best_L = 0;
    int best_pos = -1;

    // Binary search over possible lengths
    while (low <= high) {
        int mid = low + (high - low) / 2;
        int pos = solver.Test(mid, S);
        if (pos != -1) {
            best_L = mid;
            best_pos = pos;
            low = mid + 1; // Try longer
        } else {
            high = mid - 1; // Try shorter
        }
    }

    if (best_L > 0) {
        cout << S[0].substr(best_pos, best_L) << "\n";
    } else {
        cout << "\n";
    }

    return 0;
}
