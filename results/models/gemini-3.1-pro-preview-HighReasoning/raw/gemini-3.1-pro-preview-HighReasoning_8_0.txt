use std::io::{self, BufRead, Read, Write};
use std::collections::VecDeque;

struct FastOut<W: Write> {
    writer: W,
    buf: Vec<u8>,
}

impl<W: Write> FastOut<W> {
    fn new(writer: W) -> Self {
        Self {
            writer,
            buf: Vec::with_capacity(128 * 1024),
        }
    }

    fn write_u32(&mut self, mut n: u32) {
        if n == 0 {
            self.push(b'0');
            return;
        }
        let mut temp = [0u8; 10];
        let mut idx = 10;
        while n > 0 {
            idx -= 1;
            temp[idx] = b'0' + (n % 10) as u8;
            n /= 10;
        }
        let slice = &temp[idx..];
        if self.buf.len() + slice.len() > self.buf.capacity() {
            self.writer.write_all(&self.buf).unwrap();
            self.buf.clear();
        }
        self.buf.extend_from_slice(slice);
    }

    fn push(&mut self, b: u8) {
        if self.buf.len() == self.buf.capacity() {
            self.writer.write_all(&self.buf).unwrap();
            self.buf.clear();
        }
        self.buf.push(b);
    }

    fn flush(&mut self) {
        if !self.buf.is_empty() {
            self.writer.write_all(&self.buf).unwrap();
            self.buf.clear();
        }
    }
}

fn main() {
    let stdin = io::stdin();
    let mut handle = stdin.lock();

    let mut header = String::new();
    handle.read_line(&mut header).unwrap();
    let parts: Vec<usize> = header.trim().split_whitespace().map(|s| s.parse().unwrap()).collect();
    if parts.len() < 2 { return; }
    let h = parts[0];
    let w = parts[1];

    let total_cells = h * w;
    let mut grid = Vec::with_capacity(total_cells);
    let mut start_idx = u32::MAX;
    let mut end_idx = u32::MAX;

    // Use a heap-allocated buffer to prevent stack overflow on Windows (MSVC default stack size is 1MB)
    let mut buffer = vec![0u8; 1024 * 1024];
    while let Ok(n) = handle.read(&mut buffer) {
        if n == 0 { break; }
        for i in 0..n {
            let b = buffer[i];
            if b != b'\n' && b != b'\r' {
                if b == b'A' {
                    start_idx = grid.len() as u32;
                } else if b == b'B' {
                    end_idx = grid.len() as u32;
                }
                grid.push(b);
                if grid.len() == total_cells {
                    break;
                }
            }
        }
        if grid.len() == total_cells {
            break;
        }
    }

    if start_idx == u32::MAX || end_idx == u32::MAX {
        println!("0");
        return;
    }
    if start_idx == end_idx {
        println!("1");
        println!("{} {}", start_idx % (w as u32), start_idx / (w as u32));
        return;
    }

    let w32 = w as u32;
    let h32 = h as u32;

    let mut queue = VecDeque::new();
    queue.push_back(start_idx);

    let mut found = false;
    let mut before_end = u32::MAX;

    // BFS loop using grid for storing parent directions to save massive amounts of memory
    while let Some(curr) = queue.pop_front() {
        let r = curr / w32;
        let c = curr % w32;

        // Up
        if r > 0 {
            let next = curr - w32;
            let b = unsafe { *grid.get_unchecked(next as usize) };
            if b == b'B' {
                found = true;
                before_end = curr;
                break;
            } else if b == b' ' {
                unsafe { *grid.get_unchecked_mut(next as usize) = b'D' };
                queue.push_back(next);
            }
        }
        // Down
        if r + 1 < h32 {
            let next = curr + w32;
            let b = unsafe { *grid.get_unchecked(next as usize) };
            if b == b'B' {
                found = true;
                before_end = curr;
                break;
            } else if b == b' ' {
                unsafe { *grid.get_unchecked_mut(next as usize) = b'U' };
                queue.push_back(next);
            }
        }
        // Left
        if c > 0 {
            let next = curr - 1;
            let b = unsafe { *grid.get_unchecked(next as usize) };
            if b == b'B' {
                found = true;
                before_end = curr;
                break;
            } else if b == b' ' {
                unsafe { *grid.get_unchecked_mut(next as usize) = b'R' };
                queue.push_back(next);
            }
        }
        // Right
        if c + 1 < w32 {
            let next = curr + 1;
            let b = unsafe { *grid.get_unchecked(next as usize) };
            if b == b'B' {
                found = true;
                before_end = curr;
                break;
            } else if b == b' ' {
                unsafe { *grid.get_unchecked_mut(next as usize) = b'L' };
                queue.push_back(next);
            }
        }
    }

    let stdout = io::stdout();
    let mut out = FastOut::new(stdout.lock());

    if found {
        let mut path = Vec::new();
        path.push(end_idx);
        let mut current = before_end;
        while current != start_idx {
            path.push(current);
            let back_dir = unsafe { *grid.get_unchecked(current as usize) };
            if back_dir == b'U' {
                current -= w32;
            } else if back_dir == b'D' {
                current += w32;
            } else if back_dir == b'L' {
                current -= 1;
            } else if back_dir == b'R' {
                current += 1;
            } else {
                panic!("Broken path");
            }
        }
        path.push(start_idx);

        out.write_u32(path.len() as u32);
        out.push(b'\n');
        // Reconstruct path in correct order natively
        for &node in path.iter().rev() {
            out.write_u32(node % w32); // x
            out.push(b' ');
            out.write_u32(node / w32); // y
            out.push(b'\n');
        }
    } else {
        out.write_u32(0);
        out.push(b'\n');
    }
    out.flush();
}
