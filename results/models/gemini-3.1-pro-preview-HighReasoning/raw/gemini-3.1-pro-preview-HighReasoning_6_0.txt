using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator *(Vector3 a, double d) => new Vector3(a.X * d, a.Y * d, a.Z * d);
        public static Vector3 operator /(Vector3 a, double d) => new Vector3(a.X / d, a.Y / d, a.Z / d);
        public double Dot(Vector3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public Vector3 Cross(Vector3 b) => new Vector3(Y * b.Z - Z * b.Y, Z * b.X - X * b.Z, X * b.Y - Y * b.X);
        public double Length() => Math.Sqrt(Dot(this));
    }

    struct Orbit
    {
        public Vector3 R, V;
        public Vector3 H;
        public Vector3 E_vec;
        public double E;
        public double I, RAAN, AOP;
        
        public Orbit(double x, double y, double z, double vx, double vy, double vz, double mu)
        {
            R = new Vector3(x, y, z);
            V = new Vector3(vx, vy, vz);
            H = R.Cross(V);
            E = V.Dot(V) / 2.0 - mu / R.Length();
            E_vec = V.Cross(H) / mu - R / R.Length();
            
            double h_len = H.Length();
            I = h_len > 0 ? Math.Acos(H.Z / h_len) : 0;
            
            Vector3 N_vec = new Vector3(0, 0, 1).Cross(H);
            double n_len = N_vec.Length();
            if (n_len != 0)
            {
                RAAN = Math.Acos(N_vec.X / n_len);
                if (N_vec.Y < 0) RAAN = 2 * Math.PI - RAAN;
            }
            else
            {
                RAAN = 0;
            }
            
            double e_len = E_vec.Length();
            if (n_len != 0 && e_len != 0)
            {
                double dot = N_vec.Dot(E_vec) / (n_len * e_len);
                dot = Math.Max(-1.0, Math.Min(1.0, dot));
                AOP = Math.Acos(dot);
                if (E_vec.Z < 0) AOP = 2 * Math.PI - AOP;
            }
            else
            {
                AOP = 0;
            }
        }
    }

    static double Distance(Orbit a, Orbit b, double mu)
    {
        double dI = Math.Abs(a.I - b.I);
        double dRAAN = Math.Abs(a.RAAN - b.RAAN);
        if (dRAAN > Math.PI) dRAAN = 2 * Math.PI - dRAAN;
        
        double cosDTheta = Math.Cos(a.I) * Math.Cos(b.I) + Math.Sin(a.I) * Math.Sin(b.I) * Math.Cos(dRAAN);
        cosDTheta = Math.Max(-1.0, Math.Min(1.0, cosDTheta));
        double dTheta = Math.Acos(cosDTheta);
        
        double v_avg = (a.V.Length() + b.V.Length()) / 2.0;
        double dv_plane = 2 * v_avg * Math.Sin(dTheta / 2.0);
        
        double val = a.V.Dot(a.V) + 2 * (b.E - a.E);
        double dv_energy = Math.Abs(Math.Sqrt(Math.Max(0, val)) - a.V.Length());
        
        double dEvec = (a.E_vec - b.E_vec).Length();
        double dv_shape = dEvec * v_avg * 0.5;
        
        double dPos = (a.R - b.R).Length() / Math.Max(a.R.Length(), b.R.Length());
        double dv_pos = dPos * v_avg * 0.2;
        
        return dv_plane + dv_energy + dv_shape + dv_pos;
    }

    static void Main()
    {
        string line1Str = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line1Str)) return;
        string[] line1 = line1Str.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(line1[0]);
        double mu = double.Parse(line1[1]);

        string[] line2 = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        Orbit startOrbit = new Orbit(
            double.Parse(line2[0]), double.Parse(line2[1]), double.Parse(line2[2]),
            double.Parse(line2[3]), double.Parse(line2[4]), double.Parse(line2[5]), mu);

        Orbit[] stations = new Orbit[N];
        for (int i = 0; i < N; i++)
        {
            string[] line = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            stations[i] = new Orbit(
                double.Parse(line[0]), double.Parse(line[1]), double.Parse(line[2]),
                double.Parse(line[3]), double.Parse(line[4]), double.Parse(line[5]), mu);
        }

        double[,] dist = new double[N + 1, N + 1];
        for (int i = 0; i < N; i++)
        {
            dist[0, i + 1] = Distance(startOrbit, stations[i], mu);
            dist[i + 1, 0] = dist[0, i + 1];
            for (int j = 0; j < N; j++)
            {
                dist[i + 1, j + 1] = Distance(stations[i], stations[j], mu);
            }
        }

        List<int> bestPath = null;
        double bestDist = double.MaxValue;
        
        Random rnd = new Random(42);
        int numRestarts = 1000;
        
        for (int restart = 0; restart < numRestarts; restart++)
        {
            List<int> path = new List<int>();
            List<int> unvisited = new List<int>();
            for (int i = 1; i <= N; i++) unvisited.Add(i);
            
            int curr = 0;
            if (restart > 0)
            {
                int idx = rnd.Next(unvisited.Count);
                int next = unvisited[idx];
                path.Add(next);
                unvisited.RemoveAt(idx);
                curr = next;
            }
            
            while (unvisited.Count > 0)
            {
                int next = -1;
                double minDist = double.MaxValue;
                foreach (int cand in unvisited)
                {
                    double d = dist[curr, cand];
                    if (restart > 0) d *= (1.0 + (rnd.NextDouble() - 0.5) * 0.1);
                    
                    if (d < minDist)
                    {
                        minDist = d;
                        next = cand;
                    }
                }
                path.Add(next);
                unvisited.Remove(next);
                curr = next;
            }

            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < path.Count - 1; i++)
                {
                    for (int j = i + 1; j < path.Count; j++)
                    {
                        int node1 = i == 0 ? 0 : path[i - 1];
                        int node2 = path[i];
                        int node3 = path[j];
                        int node4 = j == path.Count - 1 ? -1 : path[j + 1];

                        double currentDist = dist[node1, node2] + (node4 == -1 ? 0 : dist[node3, node4]);
                        double newDist = dist[node1, node3] + (node4 == -1 ? 0 : dist[node2, node4]);

                        if (newDist < currentDist - 1e-9)
                        {
                            path.Reverse(i, j - i + 1);
                            improved = true;
                        }
                    }
                }
            }

            double totalDist = 0;
            int prev = 0;
            foreach(int node in path)
            {
                totalDist += dist[prev, node];
                prev = node;
            }
            
            if (totalDist < bestDist)
            {
                bestDist = totalDist;
                bestPath = new List<int>(path);
            }
        }

        for (int i = 0; i < bestPath.Count; i++)
        {
            bestPath[i]--;
        }

        Console.WriteLine(string.Join(" ", bestPath));
    }
}
