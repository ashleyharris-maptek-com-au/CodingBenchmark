#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>

using namespace std;

struct RectInfo {
    int id;
    int w, h;
    int max_dim;
    int min_dim;
    long long area;
    
    int board_idx;
    int x, y;
    int rotated;
};

struct FreeRect {
    int x, y, w, h;
};

struct Placement {
    int board_idx;
    int x, y;
    int rotated;
};

struct Solution {
    int num_boards;
    vector<Placement> placements;
};

class Board {
public:
    int id;
    int W, H;
    vector<FreeRect> free_rects;
    int max_w_free;
    int max_h_free;
    long long max_area_free;

    Board(int id, int W, int H) : id(id), W(W), H(H) {
        free_rects.push_back({0, 0, W, H});
        max_w_free = max(W, H);
        max_h_free = min(W, H);
        max_area_free = (long long)W * H;
    }

    bool can_fit(int item_w, int item_h) const {
        if (max_area_free < (long long)item_w * item_h) return false;
        int i_max = max(item_w, item_h);
        int i_min = min(item_w, item_h);
        return i_max <= max_w_free && i_min <= max_h_free;
    }

    void update_max_dims() {
        max_w_free = 0;
        max_h_free = 0;
        max_area_free = 0;
        for (const auto& fr : free_rects) {
            max_w_free = max(max_w_free, max(fr.w, fr.h));
            max_h_free = max(max_h_free, min(fr.w, fr.h));
            max_area_free = max(max_area_free, (long long)fr.w * fr.h);
        }
    }

    void merge_free_rects() {
        bool merged = true;
        while (merged) {
            merged = false;
            for (int i = 0; i < (int)free_rects.size(); ++i) {
                for (int j = i + 1; j < (int)free_rects.size(); ++j) {
                    if (free_rects[i].w == free_rects[j].w && free_rects[i].x == free_rects[j].x) {
                        if (free_rects[i].y + free_rects[i].h == free_rects[j].y) {
                            free_rects[i].h += free_rects[j].h;
                            free_rects.erase(free_rects.begin() + j);
                            merged = true; break;
                        }
                        if (free_rects[j].y + free_rects[j].h == free_rects[i].y) {
                            free_rects[i].y = free_rects[j].y;
                            free_rects[i].h += free_rects[j].h;
                            free_rects.erase(free_rects.begin() + j);
                            merged = true; break;
                        }
                    }
                    else if (free_rects[i].h == free_rects[j].h && free_rects[i].y == free_rects[j].y) {
                        if (free_rects[i].x + free_rects[i].w == free_rects[j].x) {
                            free_rects[i].w += free_rects[j].w;
                            free_rects.erase(free_rects.begin() + j);
                            merged = true; break;
                        }
                        if (free_rects[j].x + free_rects[j].w == free_rects[i].x) {
                            free_rects[i].x = free_rects[j].x;
                            free_rects[i].w += free_rects[j].w;
                            free_rects.erase(free_rects.begin() + j);
                            merged = true; break;
                        }
                    }
                }
                if (merged) break;
            }
        }
    }

    bool try_pack(int item_w, int item_h, int& out_x, int& out_y, int& out_rot, 
                  long long min_item_area, int min_item_max_dim, int min_item_min_dim, 
                  int fit_strategy, int split_strategy) {
        int best_i = -1;
        long long best_score1 = 3e18;
        long long best_score2 = 3e18;
        int best_rot = 0;
        
        for (int i = 0; i < (int)free_rects.size(); ++i) {
            const auto& fr = free_rects[i];
            
            for (int rot = 0; rot <= 1; ++rot) {
                int w = rot ? item_h : item_w;
                int h = rot ? item_w : item_h;
                
                if (w <= fr.w && h <= fr.h) {
                    long long score1, score2;
                    if (fit_strategy == 0) { // BSSF
                        score1 = min(fr.w - w, fr.h - h);
                        score2 = max(fr.w - w, fr.h - h);
                    } else { // BAF
                        score1 = (long long)fr.w * fr.h - (long long)w * h;
                        score2 = min(fr.w - w, fr.h - h);
                    }
                    
                    if (score1 < best_score1 || (score1 == best_score1 && score2 < best_score2)) {
                        best_score1 = score1;
                        best_score2 = score2;
                        best_i = i;
                        best_rot = rot;
                    }
                }
            }
        }
        
        if (best_i == -1) return false;
        
        FreeRect best_fr = free_rects[best_i];
        out_x = best_fr.x;
        out_y = best_fr.y;
        out_rot = best_rot;
        
        int w = best_rot ? item_h : item_w;
        int h = best_rot ? item_w : item_h;
        
        free_rects.erase(free_rects.begin() + best_i);
        
        long long s1_f1_area = (long long)(best_fr.w - w) * h;
        long long s1_f2_area = (long long)best_fr.w * (best_fr.h - h);
        
        long long s2_f1_area = (long long)(best_fr.w - w) * best_fr.h;
        long long s2_f2_area = (long long)w * (best_fr.h - h);
        
        FreeRect new1, new2;
        bool split_horizontal;
        
        if (split_strategy == 0) { // MAXAS
            long long s1_max = max(s1_f1_area, s1_f2_area);
            long long s2_max = max(s2_f1_area, s2_f2_area);
            split_horizontal = (s1_max > s2_max);
        } else { // MINAS
            long long s1_min = min(s1_f1_area, s1_f2_area);
            long long s2_min = min(s2_f1_area, s2_f2_area);
            split_horizontal = (s1_min < s2_min);
        }
        
        if (split_horizontal) {
            new1 = {best_fr.x + w, best_fr.y, best_fr.w - w, h};
            new2 = {best_fr.x, best_fr.y + h, best_fr.w, best_fr.h - h};
        } else {
            new1 = {best_fr.x + w, best_fr.y, best_fr.w - w, best_fr.h};
            new2 = {best_fr.x, best_fr.y + h, w, best_fr.h - h};
        }
        
        auto add_if_valid = [&](const FreeRect& fr) {
            if (fr.w > 0 && fr.h > 0) {
                long long area = (long long)fr.w * fr.h;
                if (area >= min_item_area && max(fr.w, fr.h) >= min_item_max_dim && min(fr.w, fr.h) >= min_item_min_dim) {
                    free_rects.push_back(fr);
                }
            }
        };
        
        add_if_valid(new1);
        add_if_valid(new2);
        
        merge_free_rects();
        update_max_dims();
        return true;
    }
};

Solution solve(const vector<RectInfo>& original_items, int board_width, int board_height, 
               long long min_area, int min_max_dim, int min_min_dim, 
               int sort_strategy, int fit_strategy, int split_strategy) {
    vector<RectInfo> items = original_items;
    
    if (sort_strategy == 0) {
        sort(items.begin(), items.end(), [](const RectInfo& a, const RectInfo& b) {
            if (a.max_dim != b.max_dim) return a.max_dim > b.max_dim;
            if (a.min_dim != b.min_dim) return a.min_dim > b.min_dim;
            return a.id < b.id;
        });
    } else if (sort_strategy == 1) {
        sort(items.begin(), items.end(), [](const RectInfo& a, const RectInfo& b) {
            if (a.area != b.area) return a.area > b.area;
            if (a.max_dim != b.max_dim) return a.max_dim > b.max_dim;
            return a.id < b.id;
        });
    } else if (sort_strategy == 2) {
        sort(items.begin(), items.end(), [](const RectInfo& a, const RectInfo& b) {
            long long p1 = (long long)a.w + a.h;
            long long p2 = (long long)b.w + b.h;
            if (p1 != p2) return p1 > p2;
            if (a.max_dim != b.max_dim) return a.max_dim > b.max_dim;
            return a.id < b.id;
        });
    } else if (sort_strategy == 3) {
        sort(items.begin(), items.end(), [](const RectInfo& a, const RectInfo& b) {
            if (a.min_dim != b.min_dim) return a.min_dim > b.min_dim;
            if (a.max_dim != b.max_dim) return a.max_dim > b.max_dim;
            return a.id < b.id;
        });
    }
    
    vector<Board> boards;
    boards.reserve(items.size());
    vector<int> active_boards;
    active_boards.reserve(items.size());

    for (int i = 0; i < (int)items.size(); ++i) {
        RectInfo& item = items[i];
        bool placed = false;
        
        for (int b_idx : active_boards) {
            if (boards[b_idx].can_fit(item.w, item.h)) {
                if (boards[b_idx].try_pack(item.w, item.h, item.x, item.y, item.rotated, 
                                           min_area, min_max_dim, min_min_dim, 
                                           fit_strategy, split_strategy)) {
                    item.board_idx = b_idx;
                    placed = true;
                    break;
                }
            }
        }
        
        if (!placed) {
            boards.emplace_back(boards.size(), board_width, board_height);
            active_boards.push_back(boards.size() - 1);
            
            boards.back().try_pack(item.w, item.h, item.x, item.y, item.rotated, 
                                   min_area, min_max_dim, min_min_dim, 
                                   fit_strategy, split_strategy);
            item.board_idx = boards.size() - 1;
        }
    }
    
    Solution sol;
    sol.num_boards = boards.size();
    sol.placements.resize(items.size());
    for (const auto& it : items) {
        sol.placements[it.id] = {it.board_idx, it.x, it.y, it.rotated};
    }
    return sol;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, board_width, board_height;
    if (!(cin >> N >> board_width >> board_height)) return 0;

    vector<RectInfo> items(N);
    long long min_item_area = 2e18;
    int min_item_max_dim = 2e9;
    int min_item_min_dim = 2e9;

    for (int i = 0; i < N; ++i) {
        items[i].id = i;
        cin >> items[i].w >> items[i].h;
        items[i].max_dim = max(items[i].w, items[i].h);
        items[i].min_dim = min(items[i].w, items[i].h);
        items[i].area = (long long)items[i].w * items[i].h;
        
        min_item_area = min(min_item_area, items[i].area);
        min_item_max_dim = min(min_item_max_dim, items[i].max_dim);
        min_item_min_dim = min(min_item_min_dim, items[i].min_dim);
    }

    auto start_time = chrono::steady_clock::now();
    Solution best_sol;
    best_sol.num_boards = 2e9;

    for (int sort_strat = 0; sort_strat < 4; ++sort_strat) {
        for (int fit_strat = 0; fit_strat < 2; ++fit_strat) {
            for (int split_strat = 0; split_strat < 2; ++split_strat) {
                auto now = chrono::steady_clock::now();
                double elapsed = chrono::duration<double>(now - start_time).count();
                if (elapsed > 25.0) goto end_search;
                
                Solution sol = solve(items, board_width, board_height, 
                                     min_item_area, min_item_max_dim, min_item_min_dim, 
                                     sort_strat, fit_strat, split_strat);
                if (sol.num_boards < best_sol.num_boards) {
                    best_sol = sol;
                }
            }
        }
    }
end_search:

    cout << best_sol.num_boards << "\n";
    for (int i = 0; i < N; ++i) {
        const auto& p = best_sol.placements[i];
        cout << p.board_idx << " " << p.x << " " << p.y << " " << p.rotated << "\n";
    }

    return 0;
}
