using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program
{
    struct Dimensions
    {
        public long w, h, d;
        public Dimensions(long w, long h, long d)
        {
            this.w = w; this.h = h; this.d = d;
        }
    }

    struct Space
    {
        public long x, y, z, w, h, d;
        public Space(long x, long y, long z, long w, long h, long d)
        {
            this.x = x; this.y = y; this.z = z;
            this.w = w; this.h = h; this.d = d;
        }
        public double Volume => (double)w * h * d;
    }

    class Box
    {
        public int id;
        public long w, h, d;
        public double Volume => (double)w * h * d;
        public Dimensions[] rotations;
    }

    class PlacedBox
    {
        public int id;
        public long x, y, z;
    }

    static Dimensions[] GetRotations(long w, long h, long d)
    {
        var set = new HashSet<(long, long, long)>
        {
            (w, h, d),
            (w, d, h),
            (h, w, d),
            (h, d, w),
            (d, w, h),
            (d, h, w)
        };

        var list = new List<Dimensions>();
        foreach (var item in set)
        {
            list.Add(new Dimensions(item.Item1, item.Item2, item.Item3));
        }
        // Pack as flat as possible (minimize Z, then Y, then X)
        return list.OrderBy(r => r.d).ThenBy(r => r.h).ThenBy(r => r.w).ToArray();
    }

    static Space[] ChooseBestSplit(Space s, long bw, long bh, long bd)
    {
        Space[][] splits = new Space[6][];
        
        long dx = s.w - bw;
        long dy = s.h - bh;
        long dz = s.d - bd;

        splits[0] = new[] {
            new Space(s.x + bw, s.y, s.z, dx, s.h, s.d),
            new Space(s.x, s.y + bh, s.z, bw, dy, s.d),
            new Space(s.x, s.y, s.z + bd, bw, bh, dz)
        };
        splits[1] = new[] {
            new Space(s.x + bw, s.y, s.z, dx, s.h, s.d),
            new Space(s.x, s.y, s.z + bd, bw, s.h, dz),
            new Space(s.x, s.y + bh, s.z, bw, dy, bd)
        };
        splits[2] = new[] {
            new Space(s.x, s.y + bh, s.z, s.w, dy, s.d),
            new Space(s.x + bw, s.y, s.z, dx, bh, s.d),
            new Space(s.x, s.y, s.z + bd, bw, bh, dz)
        };
        splits[3] = new[] {
            new Space(s.x, s.y + bh, s.z, s.w, dy, s.d),
            new Space(s.x, s.y, s.z + bd, s.w, bh, dz),
            new Space(s.x + bw, s.y, s.z, dx, bh, bd)
        };
        splits[4] = new[] {
            new Space(s.x, s.y, s.z + bd, s.w, s.h, dz),
            new Space(s.x + bw, s.y, s.z, dx, s.h, bd),
            new Space(s.x, s.y + bh, s.z, bw, dy, bd)
        };
        splits[5] = new[] {
            new Space(s.x, s.y, s.z + bd, s.w, s.h, dz),
            new Space(s.x, s.y + bh, s.z, s.w, dy, bd),
            new Space(s.x + bw, s.y, s.z, dx, bh, bd)
        };

        int bestSplit = 0;
        double bestScore = -1;
        
        // Pick the split that maximizes the volume of its largest single space
        for (int i = 0; i < 6; i++)
        {
            double maxVol = 0;
            foreach (var space in splits[i])
            {
                double vol = space.Volume;
                if (vol > maxVol) maxVol = vol;
            }
            if (maxVol > bestScore)
            {
                bestScore = maxVol;
                bestSplit = i;
            }
        }
        
        return splits[bestSplit];
    }

    static void Main()
    {
        using var reader = new StreamReader(Console.OpenStandardInput(), bufferSize: 65536);
        string line = reader.ReadLine();
        if (string.IsNullOrWhiteSpace(line)) return;
        
        string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts[0]);
        long W = long.Parse(parts[1]);
        long H = long.Parse(parts[2]);
        long D = long.Parse(parts[3]);

        var boxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            line = reader.ReadLine();
            if (string.IsNullOrWhiteSpace(line))
            {
                i--;
                continue;
            }
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            long w = long.Parse(parts[0]);
            long h = long.Parse(parts[1]);
            long d = long.Parse(parts[2]);
            
            boxes.Add(new Box {
                id = i,
                w = w,
                h = h,
                d = d,
                rotations = GetRotations(w, h, d)
            });
        }

        // Sort by volume ascending to maximize number of boxes packed.
        // Tie-break by maximum dimension to favor more cubic boxes first.
        boxes = boxes.OrderBy(b => b.Volume).ThenBy(b => Math.Max(b.w, Math.Max(b.h, b.d))).ToList();

        var spaces = new List<Space>();
        spaces.Add(new Space(0, 0, 0, W, H, D));

        var placedBoxes = new List<PlacedBox>();

        foreach (var box in boxes)
        {
            double minVolumeRemaining = box.Volume;

            // Purge spaces that are too small for this box and all future boxes
            for (int i = spaces.Count - 1; i >= 0; i--)
            {
                if (spaces[i].Volume < minVolumeRemaining)
                {
                    spaces[i] = spaces[spaces.Count - 1];
                    spaces.RemoveAt(spaces.Count - 1);
                }
            }

            Space bestSpace = default;
            bool found = false;
            Dimensions bestRot = default;
            double bestScore = double.MaxValue;
            int bestSpaceIndex = -1;

            // Find best fit space
            for (int i = 0; i < spaces.Count; i++)
            {
                Space s = spaces[i];
                foreach (var rot in box.rotations)
                {
                    if (rot.w <= s.w && rot.h <= s.h && rot.d <= s.d)
                    {
                        double score = s.Volume;
                        if (score < bestScore)
                        {
                            bestScore = score;
                            bestSpace = s;
                            bestRot = rot;
                            bestSpaceIndex = i;
                            found = true;
                        }
                        else if (score == bestScore && found)
                        {
                            // Tie-breaker: Bottom-Left-Back
                            if (s.z < bestSpace.z || 
                               (s.z == bestSpace.z && s.y < bestSpace.y) || 
                               (s.z == bestSpace.z && s.y == bestSpace.y && s.x < bestSpace.x))
                            {
                                bestSpace = s;
                                bestRot = rot;
                                bestSpaceIndex = i;
                            }
                        }
                    }
                }
            }

            if (found)
            {
                placedBoxes.Add(new PlacedBox { id = box.id, x = bestSpace.x, y = bestSpace.y, z = bestSpace.z });

                // Remove used space in O(1)
                spaces[bestSpaceIndex] = spaces[spaces.Count - 1];
                spaces.RemoveAt(spaces.Count - 1);

                // Add partitioned remaining spaces
                Space[] newSpaces = ChooseBestSplit(bestSpace, bestRot.w, bestRot.h, bestRot.d);

                foreach (var ns in newSpaces)
                {
                    if (ns.w > 0 && ns.h > 0 && ns.d > 0 && ns.Volume >= minVolumeRemaining)
                    {
                        spaces.Add(ns);
                    }
                }
            }
        }

        // Restore original index ordering for deterministic output
        placedBoxes = placedBoxes.OrderBy(pb => pb.id).ToList();

        using var writer = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536);
        writer.WriteLine(placedBoxes.Count);
        foreach (var pb in placedBoxes)
        {
            writer.WriteLine($"{pb.id} {pb.x} {pb.y} {pb.z}");
        }
    }
}
