use std::io::{self, Read};
use std::collections::{HashSet, HashMap, VecDeque};

struct FastBFS {
    visited: Vec<u32>,
    generation: u32,
    queue: Vec<usize>,
}

impl FastBFS {
    fn new(v: usize) -> Self {
        FastBFS {
            visited: vec![0; v],
            generation: 0,
            queue: Vec::with_capacity(v),
        }
    }
    fn is_connected(&mut self, start: usize, expected_count: usize, adj: &[Vec<usize>], node_visited: &[bool]) -> bool {
        if expected_count <= 1 {
            return true;
        }
        self.generation += 1;
        self.queue.clear();
        self.queue.push(start);
        self.visited[start] = self.generation;
        let mut count = 1;

        let mut head = 0;
        while head < self.queue.len() {
            let u = self.queue[head];
            head += 1;

            for i in 0..adj[u].len() {
                let v = adj[u][i];
                if !node_visited[v] && self.visited[v] != self.generation {
                    self.visited[v] = self.generation;
                    self.queue.push(v);
                    count += 1;
                }
            }
        }
        count == expected_count
    }
}

struct Solver {
    adj: Vec<Vec<usize>>,
    color: Vec<usize>,
    visited: Vec<bool>,
    unvisited_count: usize,
    unvisited_colors: [usize; 2],
    deg: Vec<usize>,
    deg1_count: usize,
    deg1_xor: usize,
    deg0_count: usize,
    start_node: usize,
    require_cycle: bool,
    is_adj_to_start: Vec<bool>,
    path: Vec<usize>,
}

impl Solver {
    fn move_forward(&mut self, w: usize) {
        self.visited[w] = true;
        self.unvisited_count -= 1;
        self.unvisited_colors[self.color[w]] -= 1;

        let d_w = self.deg[w];
        if d_w == 1 {
            self.deg1_count -= 1;
            self.deg1_xor ^= w;
        } else if d_w == 0 {
            self.deg0_count -= 1;
        }

        for i in 0..self.adj[w].len() {
            let n = self.adj[w][i];
            if !self.visited[n] {
                let d = self.deg[n];
                if d == 1 {
                    self.deg1_count -= 1;
                    self.deg1_xor ^= n;
                    self.deg0_count += 1;
                } else if d == 2 {
                    self.deg1_count += 1;
                    self.deg1_xor ^= n;
                }
                self.deg[n] -= 1;
            }
        }
    }

    fn move_backward(&mut self, w: usize) {
        for i in 0..self.adj[w].len() {
            let n = self.adj[w][i];
            if !self.visited[n] {
                self.deg[n] += 1;
                let d = self.deg[n];
                if d == 1 {
                    self.deg0_count -= 1;
                    self.deg1_count += 1;
                    self.deg1_xor ^= n;
                } else if d == 2 {
                    self.deg1_count -= 1;
                    self.deg1_xor ^= n;
                }
            }
        }

        let d_w = self.deg[w];
        if d_w == 1 {
            self.deg1_count += 1;
            self.deg1_xor ^= w;
        } else if d_w == 0 {
            self.deg0_count += 1;
        }

        self.visited[w] = false;
        self.unvisited_count += 1;
        self.unvisited_colors[self.color[w]] += 1;
    }

    fn solve(&mut self, u: usize, bfs: &mut FastBFS) -> bool {
        if self.unvisited_count == 0 {
            if self.require_cycle {
                return self.is_adj_to_start[u];
            }
            return true;
        }

        let next_c = 1 - self.color[u];
        let req_next = (self.unvisited_count + 1) / 2;
        let req_other = self.unvisited_count / 2;
        if self.unvisited_colors[next_c] != req_next || self.unvisited_colors[1 - next_c] != req_other {
            return false;
        }

        if self.deg0_count > 0 {
            if self.unvisited_count > 1 {
                return false;
            }
        }

        if self.deg1_count > 2 {
            return false;
        }

        if self.deg1_count == 1 && self.require_cycle {
            if !self.is_adj_to_start[self.deg1_xor] {
                return false;
            }
        }

        let mut moves = Vec::new();
        for &v in &self.adj[u] {
            if !self.visited[v] {
                moves.push(v);
            }
        }

        if moves.is_empty() {
            return false;
        }

        if !bfs.is_connected(moves[0], self.unvisited_count, &self.adj, &self.visited) {
            return false;
        }

        moves.sort_by_key(|&v| self.deg[v]);

        if self.deg1_count == 2 {
            moves.retain(|&v| {
                if self.deg[v] == 1 {
                    if self.require_cycle {
                        let other_d1 = self.deg1_xor ^ v;
                        self.is_adj_to_start[other_d1]
                    } else {
                        true
                    }
                } else {
                    false
                }
            });
            if moves.is_empty() {
                return false;
            }
        }

        for i in 0..moves.len() {
            let v = moves[i];
            self.move_forward(v);
            self.path.push(v);
            
            if self.solve(v, bfs) {
                return true;
            }
            
            self.path.pop();
            self.move_backward(v);
        }

        false
    }
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut tokens = input.split_whitespace();
    
    let w_str = tokens.next();
    if w_str.is_none() { return; }
    let W: usize = w_str.unwrap().parse().unwrap();
    let H: usize = tokens.next().unwrap().parse().unwrap();
    let obs_count: usize = tokens.next().unwrap().parse().unwrap();
    let req_cycle_num: usize = tokens.next().unwrap().parse().unwrap();
    let require_cycle = req_cycle_num == 1;

    let total_cells = (W as u64).saturating_mul(H as u64);
    let expected_visits = total_cells.saturating_sub(obs_count as u64);

    let use_dense = total_cells <= 100_000_000;
    let mut dense_grid = Vec::new();
    let mut obs_set = HashSet::new();

    if use_dense {
        dense_grid.resize(total_cells as usize, false);
    }

    for _ in 0..obs_count {
        let x: usize = tokens.next().unwrap().parse().unwrap();
        let y: usize = tokens.next().unwrap().parse().unwrap();
        if use_dense {
            dense_grid[y * W + x] = true;
        } else {
            obs_set.insert((x, y));
        }
    }

    let is_obs = |x: usize, y: usize| -> bool {
        if use_dense {
            dense_grid[y * W + x]
        } else {
            obs_set.contains(&(x, y))
        }
    };

    if expected_visits == 0 || is_obs(0, 0) {
        println!("0");
        return;
    }

    let mut reachable = Vec::new();
    let mut node_idx = HashMap::new();
    let mut dense_idx = Vec::new();
    if use_dense {
        dense_idx.resize(total_cells as usize, usize::MAX);
    }

    let mut queue = VecDeque::new();
    queue.push_back((0, 0));

    if use_dense {
        dense_idx[0] = 0;
    } else {
        node_idx.insert((0, 0), 0);
    }
    reachable.push((0, 0));

    let dx = [0, 1, 0, -1];
    let dy = [1, 0, -1, 0];

    let mut head = 0;
    while head < reachable.len() {
        let (x, y) = reachable[head];
        head += 1;

        for i in 0..4 {
            let nx = x as i32 + dx[i];
            let ny = y as i32 + dy[i];
            if nx >= 0 && nx < W as i32 && ny >= 0 && ny < H as i32 {
                let nux = nx as usize;
                let nuy = ny as usize;
                if !is_obs(nux, nuy) {
                    let visited = if use_dense {
                        dense_idx[nuy * W + nux] != usize::MAX
                    } else {
                        node_idx.contains_key(&(nux, nuy))
                    };
                    
                    if !visited {
                        let id = reachable.len();
                        if use_dense {
                            dense_idx[nuy * W + nux] = id;
                        } else {
                            node_idx.insert((nux, nuy), id);
                        }
                        reachable.push((nux, nuy));
                    }
                }
            }
        }
    }

    if (reachable.len() as u64) != expected_visits {
        println!("0");
        return;
    }

    let V = reachable.len();
    let mut adj = vec![Vec::new(); V];
    for i in 0..V {
        let (x, y) = reachable[i];
        for dir in 0..4 {
            let nx = x as i32 + dx[dir];
            let ny = y as i32 + dy[dir];
            if nx >= 0 && nx < W as i32 && ny >= 0 && ny < H as i32 {
                let nux = nx as usize;
                let nuy = ny as usize;
                if use_dense {
                    let id = dense_idx[nuy * W + nux];
                    if id != usize::MAX {
                        adj[i].push(id);
                    }
                } else {
                    if let Some(&id) = node_idx.get(&(nux, nuy)) {
                        adj[i].push(id);
                    }
                }
            }
        }
    }

    let mut solver = Solver {
        adj: adj.clone(),
        color: vec![0; V],
        visited: vec![false; V],
        unvisited_count: V,
        unvisited_colors: [0, 0],
        deg: vec![0; V],
        deg1_count: 0,
        deg1_xor: 0,
        deg0_count: 0,
        start_node: 0,
        require_cycle,
        is_adj_to_start: vec![false; V],
        path: Vec::with_capacity(V),
    };

    for i in 0..V {
        solver.deg[i] = solver.adj[i].len();
        let (x, y) = reachable[i];
        solver.color[i] = (x + y) % 2;
        solver.unvisited_colors[solver.color[i]] += 1;
    }

    for i in 0..V {
        if solver.deg[i] == 1 {
            solver.deg1_count += 1;
            solver.deg1_xor ^= i;
        } else if solver.deg[i] == 0 {
            solver.deg0_count += 1;
        }
    }

    for &v in &solver.adj[0] {
        solver.is_adj_to_start[v] = true;
    }

    if solver.unvisited_count > 1 && solver.deg0_count > 0 {
        println!("0");
        return;
    }

    if require_cycle && solver.unvisited_colors[0] != solver.unvisited_colors[1] {
        println!("0");
        return;
    }

    let diff = (solver.unvisited_colors[0] as i32 - solver.unvisited_colors[1] as i32).abs();
    if !require_cycle && diff > 1 {
        println!("0");
        return;
    }

    solver.move_forward(0);
    solver.path.push(0);

    let mut bfs = FastBFS::new(V);
    
    // Fast initial connectivity check if V > 1
    if solver.unvisited_count > 0 {
        let mut any_unvisited = usize::MAX;
        for i in 0..V {
            if !solver.visited[i] {
                any_unvisited = i;
                break;
            }
        }
        if any_unvisited != usize::MAX {
            if !bfs.is_connected(any_unvisited, solver.unvisited_count, &solver.adj, &solver.visited) {
                println!("0");
                return;
            }
        }
    }

    if !solver.solve(0, &mut bfs) {
        println!("0");
    } else {
        println!("{}", solver.path.len());
        for &u in &solver.path {
            let (x, y) = reachable[u];
            println!("{} {}", x, y);
        }
    }
}
