using namespace std;

const double EPS = 1e-7;

struct Vector3 {
    double x, y, z;
    Vector3() : x(0), y(0), z(0) {}
    Vector3(double _x, double _y, double _z) : x(_x), y(_y), z(_z) {}
    Vector3 operator+(const Vector3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vector3 operator-(const Vector3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vector3 operator*(double s) const { return {x*s, y*s, z*s}; }
    double dot(const Vector3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vector3 cross(const Vector3& o) const { return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x}; }
    double length() const { return sqrt(x*x + y*y + z*z); }
    Vector3 normalized() const { double l = length(); return l > 0 ? (*this) * (1.0/l) : *this; }
};

struct AABB {
    Vector3 min_pt, max_pt;
    AABB() { min_pt = {1e30, 1e30, 1e30}; max_pt = {-1e30, -1e30, -1e30}; }
    void expand(const Vector3& p) {
        min_pt.x = min(min_pt.x, p.x); min_pt.y = min(min_pt.y, p.y); min_pt.z = min(min_pt.z, p.z);
        max_pt.x = max(max_pt.x, p.x); max_pt.y = max(max_pt.y, p.y); max_pt.z = max(max_pt.z, p.z);
    }
    void expand(const AABB& o) { expand(o.min_pt); expand(o.max_pt); }
    bool intersects(const AABB& o) const {
        return (min_pt.x <= o.max_pt.x && max_pt.x >= o.min_pt.x &&
                min_pt.y <= o.max_pt.y && max_pt.y >= o.min_pt.y &&
                min_pt.z <= o.max_pt.z && max_pt.z >= o.min_pt.z);
    }
    bool intersectRay(const Vector3& ro, const Vector3& rd, double& t_out) const {
        double tmin = -1e30, tmax = 1e30;
        for (int i=0; i<3; ++i) {
            double invD = 1.0 / (&rd.x)[i];
            double t0 = ((&min_pt.x)[i] - (&ro.x)[i]) * invD;
            double t1 = ((&max_pt.x)[i] - (&ro.x)[i]) * invD;
            if (invD < 0.0) std::swap(t0, t1);
            tmin = t0 > tmin ? t0 : tmin;
            tmax = t1 < tmax ? t1 : tmax;
            if (tmax < tmin) return false;
        }
        if (tmax < 0) return false;
        t_out = max(0.0, tmin);
        return true;
    }
};

struct Triangle {
    Vector3 v[3];
    Vector3 normal;
    double d;
    AABB aabb;
    void compute() {
        normal = (v[1]-v[0]).cross(v[2]-v[0]).normalized();
        d = normal.dot(v[0]);
        aabb = AABB();
        aabb.expand(v[0]); aabb.expand(v[1]); aabb.expand(v[2]);
        aabb.min_pt = aabb.min_pt - Vector3(EPS, EPS, EPS);
        aabb.max_pt = aabb.max_pt + Vector3(EPS, EPS, EPS);
    }
};

struct BVHNode {
    AABB aabb;
    int left, right;
    int first_tri, num_tris;
};

class BVH {
public:
    vector<BVHNode> nodes;
    vector<Triangle> tris;
    
    int build(int start, int count) {
        int idx = nodes.size();
        nodes.push_back(BVHNode());
        AABB box;
        for (int i=0; i<count; ++i) box.expand(tris[start+i].aabb);
        nodes[idx].aabb = box;
        
        if (count <= 4) {
            nodes[idx].left = nodes[idx].right = -1;
            nodes[idx].first_tri = start;
            nodes[idx].num_tris = count;
            return idx;
        }
        
        Vector3 extent = box.max_pt - box.min_pt;
        int axis = 0;
        if (extent.y > extent.x && extent.y > extent.z) axis = 1;
        else if (extent.z > extent.x && extent.z > extent.y) axis = 2;
        
        auto cmp = [axis](const Triangle& a, const Triangle& b) {
            return (&a.aabb.min_pt.x)[axis] + (&a.aabb.max_pt.x)[axis] < 
                   (&b.aabb.min_pt.x)[axis] + (&b.aabb.max_pt.x)[axis];
        };
        std::nth_element(tris.begin() + start, tris.begin() + start + count/2, tris.begin() + start + count, cmp);
        
        int mid = count / 2;
        nodes[idx].left = build(start, mid);
        nodes[idx].right = build(start + mid, count - mid);
        nodes[idx].num_tris = 0;
        return idx;
    }
    
    void findOverlapping(int nodeIdx, const AABB& box, vector<int>& out_indices) const {
        if (!nodes[nodeIdx].aabb.intersects(box)) return;
        if (nodes[nodeIdx].num_tris > 0) {
            for (int i=0; i<nodes[nodeIdx].num_tris; ++i) {
                if (tris[nodes[nodeIdx].first_tri + i].aabb.intersects(box)) {
                    out_indices.push_back(nodes[nodeIdx].first_tri + i);
                }
            }
        } else {
            findOverlapping(nodes[nodeIdx].left, box, out_indices);
            findOverlapping(nodes[nodeIdx].right, box, out_indices);
        }
    }
    
    void findClosest(int nodeIdx, const Vector3& ro, const Vector3& rd, double& min_t, const Triangle*& best_tri, double& best_u, double& best_v) const {
        double t_box;
        if (!nodes[nodeIdx].aabb.intersectRay(ro, rd, t_box)) return;
        if (t_box > min_t) return;
        
        if (nodes[nodeIdx].num_tris > 0) {
            for (int i=0; i<nodes[nodeIdx].num_tris; ++i) {
                const Triangle& tri = tris[nodes[nodeIdx].first_tri + i];
                Vector3 e1 = tri.v[1] - tri.v[0];
                Vector3 e2 = tri.v[2] - tri.v[0];
                Vector3 h = rd.cross(e2);
                double a = e1.dot(h);
                if (std::abs(a) < 1e-12) continue;
                double f = 1.0 / a;
                Vector3 s = ro - tri.v[0];
                double u = f * s.dot(h);
                if (u < 0.0 || u > 1.0) continue;
                Vector3 q = s.cross(e1);
                double v = f * rd.dot(q);
                if (v < 0.0 || u + v > 1.0) continue;
                double t = f * e2.dot(q);
                if (t > 1e-8 && t < min_t) {
                    min_t = t;
                    best_tri = &tri;
                    best_u = u;
                    best_v = v;
                }
            }
        } else {
            double tl = 1e30, tr = 1e30;
            bool hl = nodes[nodes[nodeIdx].left].aabb.intersectRay(ro, rd, tl);
            bool hr = nodes[nodes[nodeIdx].right].aabb.intersectRay(ro, rd, tr);
            int first = nodes[nodeIdx].left, second = nodes[nodeIdx].right;
            if (hl && hr && tr < tl) std::swap(first, second);
            if (hl || hr) {
                findClosest(first, ro, rd, min_t, best_tri, best_u, best_v);
                findClosest(second, ro, rd, min_t, best_tri, best_u, best_v);
            }
        }
    }
    
    bool isInside(const Vector3& p) const {
        if (tris.empty()) return false;
        int attempts = 0;
        thread_local mt19937 rng(1337 + std::hash<std::thread::id>{}(std::this_thread::get_id()));
        uniform_real_distribution<double> dist(-1.0, 1.0);
        while (attempts++ < 50) {
            Vector3 rd(dist(rng), dist(rng), dist(rng));
            if (rd.length() < 0.1) continue;
            rd = rd.normalized();
            
            double min_t = 1e30;
            const Triangle* best_tri = nullptr;
            double best_u = 0, best_v = 0;
            
            findClosest(0, p, rd, min_t, best_tri, best_u, best_v);
            
            if (!best_tri) return false;
            
            if (best_u < 1e-3 || best_v < 1e-3 || (1.0 - best_u - best_v) < 1e-3) {
                if (attempts == 50) return rd.dot(best_tri->normal) > 0;
                continue;
            }
            if (std::abs(rd.dot(best_tri->normal)) < 1e-3) {
                if (attempts == 50) return rd.dot(best_tri->normal) > 0;
                continue;
            }
            
            return rd.dot(best_tri->normal) > 0;
        }
        return false;
    }
};

struct Polygon {
    vector<Vector3> verts;
    Vector3 normal;
    int coplanar_same = 0;
    int coplanar_opp = 0;
};

void split(const Polygon& poly, const Vector3& p_n, double p_d, vector<Polygon>& out) {
    vector<double> dists;
    int front = 0, back = 0, coplanar = 0;
    for (const auto& v : poly.verts) {
        double d = p_n.dot(v) - p_d;
        if (std::abs(d) < EPS) { d = 0; coplanar++; }
        else if (d > 0) front++;
        else back++;
        dists.push_back(d);
    }
    
    if (front == 0 && back == 0) {
        Polygon cp = poly;
        if (poly.normal.dot(p_n) > 0) cp.coplanar_same = 1;
        else cp.coplanar_opp = 1;
        out.push_back(cp);
        return;
    }
    if (front == 0 || back == 0) {
        out.push_back(poly);
        return;
    }
    
    vector<Vector3> f, b;
    int n = poly.verts.size();
    for (int i=0; i<n; ++i) {
        int j = (i + 1) % n;
        double di = dists[i], dj = dists[j];
        Vector3 vi = poly.verts[i], vj = poly.verts[j];
        
        if (di >= 0) f.push_back(vi);
        if (di <= 0) b.push_back(vi);
        
        if ((di > 0 && dj < 0) || (di < 0 && dj > 0)) {
            double t = di / (di - dj);
            Vector3 v = vi + (vj - vi) * t;
            f.push_back(v);
            b.push_back(v);
        }
    }
    
    if (f.size() >= 3) {
        Polygon pf; pf.verts = f; pf.normal = poly.normal; 
        pf.coplanar_same = poly.coplanar_same; pf.coplanar_opp = poly.coplanar_opp;
        out.push_back(pf);
    }
    if (b.size() >= 3) {
        Polygon pb; pb.verts = b; pb.normal = poly.normal;
        pb.coplanar_same = poly.coplanar_same; pb.coplanar_opp = poly.coplanar_opp;
        out.push_back(pb);
    }
}

bool potential_intersect(const Triangle& t1, const Triangle& t2) {
    if (!t1.aabb.intersects(t2.aabb)) return false;
    double d1 = t2.normal.dot(t1.v[0]) - t2.d;
    double d2 = t2.normal.dot(t1.v[1]) - t2.d;
    double d3 = t2.normal.dot(t1.v[2]) - t2.d;
    if ((d1 > EPS && d2 > EPS && d3 > EPS) || 
        (d1 < -EPS && d2 < -EPS && d3 < -EPS)) return false;
        
    double e1 = t1.normal.dot(t2.v[0]) - t1.d;
    double e2 = t1.normal.dot(t2.v[1]) - t1.d;
    double e3 = t1.normal.dot(t2.v[2]) - t1.d;
    if ((e1 > EPS && e2 > EPS && e3 > EPS) || 
        (e1 < -EPS && e2 < -EPS && e3 < -EPS)) return false;
        
    return true;
}

Vector3 computeCentroid(const Polygon& p) {
    Vector3 c(0,0,0);
    for (const auto& v : p.verts) c = c + v;
    return c * (1.0 / p.verts.size());
}

struct Mesh {
    vector<Vector3> verts;
    vector<Triangle> tris;
};

struct Key3D {
    long long x, y, z;
    bool operator==(const Key3D& o) const { return x==o.x && y==o.y && z==o.z; }
};
struct KeyHash {
    size_t operator()(const Key3D& k) const {
        size_t h1 = std::hash<long long>()(k.x);
        size_t h2 = std::hash<long long>()(k.y);
        size_t h3 = std::hash<long long>()(k.z);
        return h1 ^ (h2 << 1) ^ (h3 << 2);
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;
    
    Mesh meshA, meshB;
    meshA.verts.resize(VA);
    for (int i=0; i<VA; ++i) cin >> meshA.verts[i].x >> meshA.verts[i].y >> meshA.verts[i].z;
    meshA.tris.resize(FA);
    for (int i=0; i<FA; ++i) {
        int a, b, c; cin >> a >> b >> c;
        meshA.tris[i].v[0] = meshA.verts[a];
        meshA.tris[i].v[1] = meshA.verts[b];
        meshA.tris[i].v[2] = meshA.verts[c];
    }
    
    meshB.verts.resize(VB);
    for (int i=0; i<VB; ++i) cin >> meshB.verts[i].x >> meshB.verts[i].y >> meshB.verts[i].z;
    meshB.tris.resize(FB);
    for (int i=0; i<FB; ++i) {
        int a, b, c; cin >> a >> b >> c;
        meshB.tris[i].v[0] = meshB.verts[a];
        meshB.tris[i].v[1] = meshB.verts[b];
        meshB.tris[i].v[2] = meshB.verts[c];
    }
    
    Vector3 min_ext = (VA > 0) ? meshA.verts[0] : ((VB > 0) ? meshB.verts[0] : Vector3(0,0,0));
    Vector3 max_ext = min_ext;
    for(auto* m : {&meshA, &meshB}) {
        for(auto& v : m->verts) {
            min_ext.x = min(min_ext.x, v.x); min_ext.y = min(min_ext.y, v.y); min_ext.z = min(min_ext.z, v.z);
            max_ext.x = max(max_ext.x, v.x); max_ext.y = max(max_ext.y, v.y); max_ext.z = max(max_ext.z, v.z);
        }
    }
    Vector3 center = (min_ext + max_ext) * 0.5;
    Vector3 extent = max_ext - min_ext;
    double max_dim = max({extent.x, extent.y, extent.z});
    if (max_dim < 1e-12) max_dim = 1.0;
    double scale = 1.0 / max_dim;
    
    for(auto* m : {&meshA, &meshB}) {
        for(auto& t : m->tris) {
            for(int i=0; i<3; ++i) t.v[i] = (t.v[i] - center) * scale;
            t.compute();
        }
    }
    
    BVH bvhA, bvhB;
    bvhA.tris = meshA.tris;
    if (FA > 0) bvhA.build(0, FA);
    bvhB.tris = meshB.tris;
    if (FB > 0) bvhB.build(0, FB);
    
    int num_threads = thread::hardware_concurrency();
    if (num_threads == 0) num_threads = 8;
    
    vector<Polygon> all_polys;
    mutex out_mtx;
    
    auto workerA = [&](int start, int end) {
        vector<Polygon> local_p;
        for (int i=start; i<end; ++i) {
            const Triangle& tri = meshA.tris[i];
            vector<int> b_idx;
            if (FB > 0) bvhB.findOverlapping(0, tri.aabb, b_idx);
            vector<int> exact_b;
            for (int idx : b_idx) {
                if (potential_intersect(tri, bvhB.tris[idx])) exact_b.push_back(idx);
            }
            
            vector<Polygon> polys = { { {tri.v[0], tri.v[1], tri.v[2]}, tri.normal, 0, 0 } };
            for (int idx : exact_b) {
                const Triangle& tb = bvhB.tris[idx];
                vector<Polygon> next;
                for (const auto& p : polys) split(p, tb.normal, tb.d, next);
                polys = next;
            }
            
            for (const auto& p : polys) {
                if (p.coplanar_opp) continue;
                Vector3 c = computeCentroid(p);
                bool is_same = p.coplanar_same > 0;
                Vector3 test_p = is_same ? c + p.normal * 1e-5 : c;
                if (!bvhB.isInside(test_p)) {
                    local_p.push_back(p);
                }
            }
        }
        lock_guard<mutex> lk(out_mtx);
        all_polys.insert(all_polys.end(), local_p.begin(), local_p.end());
    };
    
    auto workerB = [&](int start, int end) {
        vector<Polygon> local_p;
        for (int i=start; i<end; ++i) {
            const Triangle& tri = meshB.tris[i];
            vector<int> a_idx;
            if (FA > 0) bvhA.findOverlapping(0, tri.aabb, a_idx);
            vector<int> exact_a;
            for (int idx : a_idx) {
                if (potential_intersect(tri, bvhA.tris[idx])) exact_a.push_back(idx);
            }
            
            vector<Polygon> polys = { { {tri.v[0], tri.v[1], tri.v[2]}, tri.normal, 0, 0 } };
            for (int idx : exact_a) {
                const Triangle& ta = bvhA.tris[idx];
                vector<Polygon> next;
                for (const auto& p : polys) split(p, ta.normal, ta.d, next);
                polys = next;
            }
            
            for (const auto& p : polys) {
                if (p.coplanar_opp) continue;
                if (p.coplanar_same) continue; 
                Vector3 c = computeCentroid(p);
                if (!bvhA.isInside(c)) {
                    local_p.push_back(p);
                }
            }
        }
        lock_guard<mutex> lk(out_mtx);
        all_polys.insert(all_polys.end(), local_p.begin(), local_p.end());
    };
    
    vector<thread> threads;
    int chunkA = (FA + num_threads - 1) / num_threads;
    for (int i=0; i<num_threads; ++i) {
        int s = i * chunkA;
        int e = min(FA, s + chunkA);
        if (s < e) threads.push_back(thread(workerA, s, e));
    }
    for (auto& t : threads) t.join();
    threads.clear();
    
    int chunkB = (FB + num_threads - 1) / num_threads;
    for (int i=0; i<num_threads; ++i) {
        int s = i * chunkB;
        int e = min(FB, s + chunkB);
        if (s < e) threads.push_back(thread(workerB, s, e));
    }
    for (auto& t : threads) t.join();
    
    double inv_scale = max_dim;
    vector<Vector3> out_verts;
    vector<vector<int>> out_faces;
    unordered_map<Key3D, int, KeyHash> vmap;
    
    for (const auto& p : all_polys) {
        if (p.verts.size() < 3) continue;
        vector<int> idxs;
        for (const auto& v : p.verts) {
            Key3D k = { (long long)std::round(v.x * 1e6), (long long)std::round(v.y * 1e6), (long long)std::round(v.z * 1e6) };
            auto it = vmap.find(k);
            int idx;
            if (it != vmap.end()) idx = it->second;
            else {
                idx = out_verts.size();
                out_verts.push_back(v * inv_scale + center);
                vmap[k] = idx;
            }
            if (idxs.empty() || idxs.back() != idx) idxs.push_back(idx);
        }
        if (idxs.size() > 1 && idxs.back() == idxs.front()) idxs.pop_back();
        
        if (idxs.size() >= 3) {
            int i0 = idxs[0];
            for (size_t i = 1; i < idxs.size() - 1; ++i) {
                out_faces.push_back({i0, idxs[i], idxs[i+1]});
            }
        }
    }
    
    cout << out_verts.size() << " " << out_faces.size() << "\n";
    cout << fixed << setprecision(6);
    for (const auto& v : out_verts) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& f : out_faces) {
        cout << f[0] << " " << f[1] << " " << f[2] << "\n";
    }
    
    return 0;
}
