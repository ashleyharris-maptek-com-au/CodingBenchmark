use std::io::{self, Read, Write, BufWriter};

#[derive(Clone, Default)]
struct QuadNode {
    mass: f64,
    cx: f64,
    cy: f64,
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
    children: [usize; 4],
}

fn partition4(points: &mut [usize], pos: &[(f64, f64)], mid_x: f64, mid_y: f64) -> [usize; 5] {
    let mut bounds = [0; 5];
    bounds[0] = 0;
    
    let mut i = 0;
    let mut j = points.len();
    while i < j {
        if pos[points[i]].1 < mid_y { i += 1; }
        else { j -= 1; points.swap(i, j); }
    }
    let mid_split = i; 
    
    let mut i = 0;
    let mut j = mid_split;
    while i < j {
        if pos[points[i]].0 < mid_x { i += 1; }
        else { j -= 1; points.swap(i, j); }
    }
    bounds[1] = i; 
    bounds[2] = mid_split; 
    
    let mut i = mid_split;
    let mut j = points.len();
    while i < j {
        if pos[points[i]].0 < mid_x { i += 1; }
        else { j -= 1; points.swap(i, j); }
    }
    bounds[3] = i; 
    bounds[4] = points.len(); 
    
    bounds
}

fn split_4<'a>(points: &'a mut [usize], bounds: &[usize; 5]) -> [&'a mut [usize]; 4] {
    let (p0, rest) = points.split_at_mut(bounds[1]);
    let (p1, rest) = rest.split_at_mut(bounds[2] - bounds[1]);
    let (p2, p3) = rest.split_at_mut(bounds[3] - bounds[2]);
    [p0, p1, p2, p3]
}

fn build_tree(
    points: &mut [usize], 
    pos: &[(f64, f64)], 
    min_x: f64, max_x: f64, min_y: f64, max_y: f64, 
    nodes: &mut Vec<QuadNode>,
    depth: usize
) -> Option<usize> {
    if points.is_empty() {
        return None;
    }
    
    let mut coincident = true;
    let first_p = pos[points[0]];
    for &p in points.iter().skip(1) {
        if (pos[p].0 - first_p.0).abs() > 1e-5 || (pos[p].1 - first_p.1).abs() > 1e-5 {
            coincident = false;
            break;
        }
    }
    
    if points.len() == 1 || depth > 40 || coincident {
        let mut sum_x = 0.0;
        let mut sum_y = 0.0;
        for &p in points.iter() {
            sum_x += pos[p].0;
            sum_y += pos[p].1;
        }
        let mass = points.len() as f64;
        let node_idx = nodes.len();
        nodes.push(QuadNode {
            mass,
            cx: sum_x / mass,
            cy: sum_y / mass,
            min_x, max_x, min_y, max_y,
            children: [usize::MAX; 4],
        });
        return Some(node_idx);
    }
    
    let mid_x = (min_x + max_x) / 2.0;
    let mid_y = (min_y + max_y) / 2.0;
    
    let bounds = partition4(points, pos, mid_x, mid_y);
    let mut subs = split_4(points, &bounds);
    
    let mut children = [usize::MAX; 4];
    let mut mass = 0.0;
    let mut cx_sum = 0.0;
    let mut cy_sum = 0.0;
    
    for c in 0..4 {
        let (nx1, nx2) = if c % 2 == 0 { (min_x, mid_x) } else { (mid_x, max_x) };
        let (ny1, ny2) = if c / 2 == 0 { (min_y, mid_y) } else { (mid_y, max_y) };
        if let Some(child_idx) = build_tree(subs[c], pos, nx1, nx2, ny1, ny2, nodes, depth + 1) {
            children[c] = child_idx;
            let m = nodes[child_idx].mass;
            mass += m;
            cx_sum += nodes[child_idx].cx * m;
            cy_sum += nodes[child_idx].cy * m;
        }
    }
    
    let node_idx = nodes.len();
    nodes.push(QuadNode {
        mass,
        cx: cx_sum / mass,
        cy: cy_sum / mass,
        min_x, max_x, min_y, max_y,
        children,
    });
    
    Some(node_idx)
}

fn compute_repulsion(
    node_idx: usize, 
    p: usize, 
    pos: &[(f64, f64)], 
    nodes: &[QuadNode], 
    theta: f64,
    k: f64
) -> (f64, f64) {
    let node = &nodes[node_idx];
    let dx = node.cx - pos[p].0;
    let dy = node.cy - pos[p].1;
    let mut dist_sq = dx * dx + dy * dy;
    
    let is_leaf = node.children == [usize::MAX; 4];
    let width = node.max_x - node.min_x;
    
    if is_leaf || (width / dist_sq.sqrt().max(1e-5)) < theta {
        let mut m = node.mass;
        if is_leaf && dist_sq < 1e-5 {
            m -= 1.0;
            if m <= 0.0 {
                return (0.0, 0.0);
            }
            let angle = (p as f64).sin() * 100.0;
            return (k * angle.cos(), k * angle.sin());
        }
        
        let dist = dist_sq.sqrt().max(1e-5);
        let f = (k * k) * m / (dist * dist);
        return (-dx * f, -dy * f);
    }
    
    let mut fx = 0.0;
    let mut fy = 0.0;
    for &child_idx in &node.children {
        if child_idx != usize::MAX {
            let (cfx, cfy) = compute_repulsion(child_idx, p, pos, nodes, theta, k);
            fx += cfx;
            fy += cfy;
        }
    }
    (fx, fy)
}

fn find_components(n: usize, edges: &[(usize, usize)]) -> Vec<Vec<usize>> {
    let mut adj = vec![vec![]; n];
    for &(u, v) in edges {
        adj[u].push(v);
        adj[v].push(u);
    }
    
    let mut visited = vec![false; n];
    let mut components = Vec::new();
    
    for i in 0..n {
        if !visited[i] {
            let mut comp = Vec::new();
            let mut q = vec![i];
            visited[i] = true;
            
            while let Some(u) = q.pop() {
                comp.push(u);
                for &v in &adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        q.push(v);
                    }
                }
            }
            components.push(comp);
        }
    }
    components
}

fn layout_single_component(n: usize, edges: &[(usize, usize)], k: f64) -> Vec<(f64, f64)> {
    if n == 1 {
        return vec![(0.0, 0.0)];
    }
    
    let mut adj = vec![vec![]; n];
    for &(u, v) in edges {
        adj[u].push(v);
        adj[v].push(u);
    }
    
    let mut pos = vec![(0.0, 0.0); n];
    let mut visited = vec![false; n];
    let mut layers = Vec::new();
    
    let start_node = (0..n).max_by_key(|&i| adj[i].len()).unwrap_or(0);
    
    let mut current_layer = vec![start_node];
    visited[start_node] = true;
    let mut next_layer = Vec::new();
    
    while !current_layer.is_empty() {
        layers.push(current_layer.clone());
        for &u in &current_layer {
            for &v in &adj[u] {
                if !visited[v] {
                    visited[v] = true;
                    next_layer.push(v);
                }
            }
        }
        current_layer = next_layer;
        next_layer = Vec::new();
    }
    
    let mut current_radius = 0.0;
    for layer in layers {
        let count = layer.len();
        if count == 1 {
            pos[layer[0]] = (current_radius, 0.0);
        } else {
            let angle_step = std::f64::consts::TAU / (count as f64);
            for (i, &u) in layer.iter().enumerate() {
                let angle = i as f64 * angle_step;
                pos[u] = (current_radius * angle.cos(), current_radius * angle.sin());
            }
        }
        current_radius += k;
    }
    
    let theta = if n > 50000 { 1.5 } else if n > 10000 { 1.2 } else { 0.9 };
    let iters = if n > 100000 { 60 } else if n > 50000 { 80 } else if n > 10000 { 120 } else { 200 };
    
    let mut temp = 5.0 * k;
    let temp_step = temp / (iters as f64);
    
    let mut disp = vec![(0.0, 0.0); n];
    let mut points: Vec<usize> = (0..n).collect();
    let mut nodes = Vec::with_capacity(n * 4);
    
    for _iter in 0..iters {
        disp.fill((0.0, 0.0));
        
        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        
        for &(x, y) in &pos {
            if x < min_x { min_x = x; }
            if x > max_x { max_x = x; }
            if y < min_y { min_y = y; }
            if y > max_y { max_y = y; }
        }
        
        min_x -= 1.0; max_x += 1.0;
        min_y -= 1.0; max_y += 1.0;
        
        nodes.clear();
        for i in 0..n { points[i] = i; }
        
        let root = build_tree(&mut points, &pos, min_x, max_x, min_y, max_y, &mut nodes, 0);
        
        if let Some(root_idx) = root {
            for i in 0..n {
                let (mut fx, mut fy) = compute_repulsion(root_idx, i, &pos, &nodes, theta, k);
                
                let gx = 0.0 - pos[i].0;
                let gy = 0.0 - pos[i].1;
                fx += 0.01 * gx;
                fy += 0.01 * gy;
                
                disp[i].0 += fx;
                disp[i].1 += fy;
            }
        }
        
        for &(u, v) in edges {
            let dx = pos[v].0 - pos[u].0;
            let dy = pos[v].1 - pos[u].1;
            let dist_sq = dx * dx + dy * dy;
            let dist = dist_sq.sqrt().max(1e-5);
            let f = dist / k;
            let fx = dx * f;
            let fy = dy * f;
            
            disp[u].0 += fx;
            disp[u].1 += fy;
            disp[v].0 -= fx;
            disp[v].1 -= fy;
        }
        
        for i in 0..n {
            let dx = disp[i].0;
            let dy = disp[i].1;
            let d_len = (dx * dx + dy * dy).sqrt().max(1e-5);
            let limit = temp.min(d_len);
            pos[i].0 += (dx / d_len) * limit;
            pos[i].1 += (dy / d_len) * limit;
        }
        
        temp -= temp_step;
        if temp < 0.1 { temp = 0.1; }
    }
    
    pos
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut tokens = input.split_whitespace();
    
    let n_token = tokens.next();
    if n_token.is_none() {
        return;
    }
    let n: usize = n_token.unwrap().parse().unwrap();
    let m: usize = tokens.next().unwrap().parse().unwrap();
    
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = tokens.next().unwrap().parse().unwrap();
        let v: usize = tokens.next().unwrap().parse().unwrap();
        edges.push((u, v));
    }
    
    let components = find_components(n, &edges);
    
    let mut comp_id = vec![0; n];
    for (id, comp) in components.iter().enumerate() {
        for &u in comp {
            comp_id[u] = id;
        }
    }
    
    let mut comp_edges = vec![vec![]; components.len()];
    for &(u, v) in &edges {
        if comp_id[u] == comp_id[v] {
            comp_edges[comp_id[u]].push((u, v));
        }
    }
    
    let mut pos = vec![(0.0, 0.0); n];
    let k = 10.0;
    
    let mut global_to_local = vec![usize::MAX; n];
    
    for (id, comp) in components.iter().enumerate() {
        for (local_u, &global_u) in comp.iter().enumerate() {
            global_to_local[global_u] = local_u;
        }
        
        let edges_local = comp_edges[id].iter()
            .map(|&(u, v)| (global_to_local[u], global_to_local[v]))
            .collect::<Vec<_>>();
            
        let pos_local = layout_single_component(comp.len(), &edges_local, k);
        
        for (local_u, &global_u) in comp.iter().enumerate() {
            pos[global_u] = pos_local[local_u];
        }
    }
    
    let mut component_boxes = Vec::new();
    for (id, comp) in components.iter().enumerate() {
        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        for &u in comp {
            let (x, y) = pos[u];
            if x < min_x { min_x = x; }
            if x > max_x { max_x = x; }
            if y < min_y { min_y = y; }
            if y > max_y { max_y = y; }
        }
        
        if min_x > max_x {
            min_x = 0.0; max_x = 0.0;
            min_y = 0.0; max_y = 0.0;
        }
        
        let width = max_x - min_x + k;
        let height = max_y - min_y + k;
        component_boxes.push((id, width, height, min_x, max_x, min_y, max_y));
    }
    
    component_boxes.sort_by(|a, b| b.2.partial_cmp(&a.2).unwrap());
    
    let max_width = 100.0 * k * (components.len() as f64).sqrt().max(1.0);
    let mut current_x = 0.0;
    let mut current_y = 0.0;
    let mut row_height = 0.0;
    
    let mut offsets = vec![(0.0, 0.0); components.len()];
    
    for &(id, width, height, min_x, max_x, min_y, max_y) in &component_boxes {
        if current_x + width > max_width && current_x > 0.0 {
            current_x = 0.0;
            current_y += row_height;
            row_height = 0.0;
        }
        
        offsets[id] = (current_x - min_x + k/2.0, current_y - min_y + k/2.0);
        
        current_x += width;
        if height > row_height {
            row_height = height;
        }
    }
    
    for (id, comp) in components.iter().enumerate() {
        let (ox, oy) = offsets[id];
        for &u in comp {
            pos[u].0 += ox;
            pos[u].1 += oy;
        }
    }
    
    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    for &(x, y) in &pos {
        writeln!(writer, "{:.6} {:.6}", x, y).unwrap();
    }
}
