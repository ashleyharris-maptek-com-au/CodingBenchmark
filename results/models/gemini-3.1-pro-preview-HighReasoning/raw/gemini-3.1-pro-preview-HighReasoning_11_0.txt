use std::io::{self, BufRead};
use std::time::Instant;

struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Rng { state: seed }
    }
    fn next_f64(&mut self) -> f64 {
        self.state ^= self.state << 13;
        self.state ^= self.state >> 7;
        self.state ^= self.state << 17;
        let res = self.state.wrapping_mul(0x2545F4914F6CDD1D);
        (res >> 11) as f64 / (1u64 << 53) as f64
    }
}

struct PlacedPiece {
    polygon: Vec<(f64, f64)>,
    bbox: (f64, f64, f64, f64),
}

struct Stock {
    placed: Vec<PlacedPiece>,
    used_area: f64,
}

fn cross_product(p0: (f64, f64), p1: (f64, f64), p2: (f64, f64)) -> f64 {
    (p1.0 - p0.0) * (p2.1 - p0.1) - (p1.1 - p0.1) * (p2.0 - p0.0)
}

fn segments_intersect(p1: (f64, f64), p2: (f64, f64), p3: (f64, f64), p4: (f64, f64)) -> bool {
    let cp1 = cross_product(p3, p4, p1);
    let cp2 = cross_product(p3, p4, p2);
    let cp3 = cross_product(p1, p2, p3);
    let cp4 = cross_product(p1, p2, p4);

    let eps = 1e-7;

    if ((cp1 > eps && cp2 < -eps) || (cp1 < -eps && cp2 > eps)) &&
       ((cp3 > eps && cp4 < -eps) || (cp3 < -eps && cp4 > eps)) {
        return true;
    }
    false
}

fn point_in_or_on_polygon(p: (f64, f64), poly: &[(f64, f64)]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    let eps = 1e-7;

    for i in 0..poly.len() {
        let pi = poly[i];
        let pj = poly[j];

        let min_x = pi.0.min(pj.0);
        let max_x = pi.0.max(pj.0);
        let min_y = pi.1.min(pj.1);
        let max_y = pi.1.max(pj.1);
        
        if p.0 >= min_x - eps && p.0 <= max_x + eps && p.1 >= min_y - eps && p.1 <= max_y + eps {
            let cp = (p.0 - pi.0) * (pj.1 - pi.1) - (p.1 - pi.1) * (pj.0 - pi.0);
            if cp.abs() < eps {
                return true;
            }
        }

        if (pi.1 > p.1) != (pj.1 > p.1) {
            let xinters = (pj.0 - pi.0) * (p.1 - pi.1) / (pj.1 - pi.1) + pi.0;
            if p.0 < xinters {
                inside = !inside;
            }
        }
        j = i;
    }
    inside
}

fn point_strictly_inside_polygon(p: (f64, f64), poly: &[(f64, f64)]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    let eps = 1e-7;

    for i in 0..poly.len() {
        let pi = poly[i];
        let pj = poly[j];

        let min_x = pi.0.min(pj.0);
        let max_x = pi.0.max(pj.0);
        let min_y = pi.1.min(pj.1);
        let max_y = pi.1.max(pj.1);
        
        if p.0 >= min_x - eps && p.0 <= max_x + eps && p.1 >= min_y - eps && p.1 <= max_y + eps {
            let cp = (p.0 - pi.0) * (pj.1 - pi.1) - (p.1 - pi.1) * (pj.0 - pi.0);
            if cp.abs() < eps {
                return false;
            }
        }

        if (pi.1 > p.1) != (pj.1 > p.1) {
            let xinters = (pj.0 - pi.0) * (p.1 - pi.1) / (pj.1 - pi.1) + pi.0;
            if p.0 < xinters {
                inside = !inside;
            }
        }
        j = i;
    }
    inside
}

fn any_point_strictly_inside(poly1: &[(f64, f64)], poly2: &[(f64, f64)]) -> bool {
    for p in poly1 {
        if point_strictly_inside_polygon(*p, poly2) {
            return true;
        }
    }
    false
}

fn polygons_overlap_fast(poly1: &[(f64, f64)], bb1: (f64, f64, f64, f64), placed: &PlacedPiece) -> bool {
    let bb2 = placed.bbox;
    let eps = 1e-7;
    if bb1.0 > bb2.1 - eps || bb1.1 < bb2.0 + eps || bb1.2 > bb2.3 - eps || bb1.3 < bb2.2 + eps {
        return false;
    }

    let poly2 = &placed.polygon;
    for i in 0..poly1.len() {
        let p1 = poly1[i];
        let p2 = poly1[(i + 1) % poly1.len()];
        for j in 0..poly2.len() {
            let p3 = poly2[j];
            let p4 = poly2[(j + 1) % poly2.len()];
            if segments_intersect(p1, p2, p3, p4) {
                return true;
            }
        }
    }
    
    if any_point_strictly_inside(poly1, poly2) {
        return true;
    }
    if any_point_strictly_inside(poly2, poly1) {
        return true;
    }
    
    false
}

fn polygon_inside(inner: &[(f64, f64)], outer: &[(f64, f64)]) -> bool {
    let bbi = bounding_box(inner);
    let bbo = bounding_box(outer);
    let eps = 1e-7;
    if bbi.0 < bbo.0 - eps || bbi.1 > bbo.1 + eps || bbi.2 < bbo.2 - eps || bbi.3 > bbo.3 + eps {
        return false;
    }

    for p in inner {
        if !point_in_or_on_polygon(*p, outer) {
            return false;
        }
    }
    
    for i in 0..inner.len() {
        let p1 = inner[i];
        let p2 = inner[(i + 1) % inner.len()];
        for j in 0..outer.len() {
            let p3 = outer[j];
            let p4 = outer[(j + 1) % outer.len()];
            if segments_intersect(p1, p2, p3, p4) {
                return false;
            }
        }
    }
    true
}

fn bounding_box(poly: &[(f64, f64)]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;
    for &(x, y) in poly {
        if x < min_x { min_x = x; }
        if x > max_x { max_x = x; }
        if y < min_y { min_y = y; }
        if y > max_y { max_y = y; }
    }
    (min_x, max_x, min_y, max_y)
}

fn polygon_area(poly: &[(f64, f64)]) -> f64 {
    let mut area = 0.0;
    for i in 0..poly.len() {
        let p1 = poly[i];
        let p2 = poly[(i + 1) % poly.len()];
        area += p1.0 * p2.1 - p2.0 * p1.1;
    }
    (area / 2.0).abs()
}

fn is_axis_aligned_rectangle(poly: &[(f64, f64)]) -> bool {
    if poly.len() != 4 {
        return false;
    }
    let mut xs = vec![poly[0].0, poly[1].0, poly[2].0, poly[3].0];
    let mut ys = vec![poly[0].1, poly[1].1, poly[2].1, poly[3].1];
    xs.sort_by(|a, b| a.partial_cmp(b).unwrap());
    ys.sort_by(|a, b| a.partial_cmp(b).unwrap());
    
    (xs[0] - xs[1]).abs() < 1e-7 && (xs[2] - xs[3]).abs() < 1e-7 &&
    (ys[0] - ys[1]).abs() < 1e-7 && (ys[2] - ys[3]).abs() < 1e-7
}

fn rotate_polygon(poly: &[(f64, f64)], rot: f64) -> Vec<(f64, f64)> {
    let cos_r = rot.cos();
    let sin_r = rot.sin();
    poly.iter().map(|&(x, y)| {
        (x * cos_r - y * sin_r, x * sin_r + y * cos_r)
    }).collect()
}

fn translate_polygon(poly: &[(f64, f64)], dx: f64, dy: f64) -> Vec<(f64, f64)> {
    poly.iter().map(|&(x, y)| (x + dx, y + dy)).collect()
}

fn try_place(
    piece: &[(f64, f64)], 
    stock_poly: &[(f64, f64)], 
    placed: &[PlacedPiece], 
    grid_res: usize, 
    rotations: &[f64],
    stock_is_rect: bool
) -> Option<(f64, f64, f64, Vec<(f64, f64)>)> {
    let (s_min_x, s_max_x, s_min_y, s_max_y) = bounding_box(stock_poly);
    
    let mut candidates = Vec::new();
    let mut rotated_polys = Vec::new();
    
    for (rot_idx, &rot) in rotations.iter().enumerate() {
        let rotated = rotate_polygon(piece, rot);
        let (p_min_x, p_max_x, p_min_y, p_max_y) = bounding_box(&rotated);
        
        rotated_polys.push(rotated);
        
        let dx_min = s_min_x - p_min_x;
        let dx_max = s_max_x - p_max_x;
        let dy_min = s_min_y - p_min_y;
        let dy_max = s_max_y - p_max_y;
        
        if dx_min > dx_max + 1e-7 || dy_min > dy_max + 1e-7 {
            continue;
        }
        
        let dx_max = dx_max.max(dx_min);
        let dy_max = dy_max.max(dy_min);
        
        for i in 0..=grid_res {
            let dx = if grid_res == 0 { dx_min } else { dx_min + (dx_max - dx_min) * (i as f64 / grid_res as f64) };
            for j in 0..=grid_res {
                let dy = if grid_res == 0 { dy_min } else { dy_min + (dy_max - dy_min) * (j as f64 / grid_res as f64) };
                candidates.push((dx, dy, rot_idx));
            }
        }
    }
    
    candidates.sort_by(|a, b| {
        a.1.partial_cmp(&b.1).unwrap().then(a.0.partial_cmp(&b.0).unwrap())
    });
    
    for (dx, dy, rot_idx) in candidates {
        let rot = rotations[rot_idx];
        let rotated = &rotated_polys[rot_idx];
        
        let mut inside_stock = true;
        let mut translated = Vec::new();
        
        if !stock_is_rect {
            translated = translate_polygon(rotated, dx, dy);
            inside_stock = polygon_inside(&translated, stock_poly);
        }
        
        if inside_stock {
            if translated.is_empty() {
                translated = translate_polygon(rotated, dx, dy);
            }
            let bb_trans = bounding_box(&translated);
            let mut overlap = false;
            for pl in placed {
                if polygons_overlap_fast(&translated, bb_trans, pl) {
                    overlap = true;
                    break;
                }
            }
            if !overlap {
                return Some((dx, dy, rot, translated));
            }
        }
    }
    None
}

fn place_in_existing_stocks(
    piece: &[(f64, f64)],
    piece_area: f64,
    stock_area: f64,
    stock_poly: &[(f64, f64)],
    stocks: &mut [Stock],
    stock_is_rect: bool
) -> Option<(usize, f64, f64, f64, Vec<(f64, f64)>)> {
    let rotations = [0.0, std::f64::consts::PI / 2.0, std::f64::consts::PI, 3.0 * std::f64::consts::PI / 2.0];
    
    let res = if stocks.len() < 5 {
        60
    } else if stocks.len() < 20 {
        40
    } else if stocks.len() < 100 {
        20
    } else if stocks.len() < 500 {
        10
    } else {
        5
    };
    
    for (i, stock) in stocks.iter().enumerate() {
        if stock.used_area + piece_area > stock_area {
            continue;
        }
        if let Some((dx, dy, rot, translated)) = try_place(piece, stock_poly, &stock.placed, res, &rotations, stock_is_rect) {
            return Some((i, dx, dy, rot, translated));
        }
    }
    None
}

fn place_in_new_stock(
    piece: &[(f64, f64)],
    stock_poly: &[(f64, f64)],
    stock_is_rect: bool
) -> Option<(f64, f64, f64, Vec<(f64, f64)>)> {
    let rotations = [0.0, std::f64::consts::PI / 2.0, std::f64::consts::PI, 3.0 * std::f64::consts::PI / 2.0, 
                     std::f64::consts::PI / 4.0, 3.0 * std::f64::consts::PI / 4.0, 
                     5.0 * std::f64::consts::PI / 4.0, 7.0 * std::f64::consts::PI / 4.0];
    
    for res in [1, 10, 50, 100, 200] {
        if let Some(res) = try_place(piece, stock_poly, &[], res, &rotations, stock_is_rect) {
            return Some(res);
        }
    }
    
    let mut rng = Rng::new(12345);
    let (s_min_x, s_max_x, s_min_y, s_max_y) = bounding_box(stock_poly);
    for _ in 0..10000 {
        let rot = rng.next_f64() * 2.0 * std::f64::consts::PI;
        let rotated = rotate_polygon(piece, rot);
        let (p_min_x, p_max_x, p_min_y, p_max_y) = bounding_box(&rotated);
        let dx_min = s_min_x - p_min_x;
        let dx_max = s_max_x - p_max_x;
        let dy_min = s_min_y - p_min_y;
        let dy_max = s_max_y - p_max_y;
        if dx_min > dx_max || dy_min > dy_max { continue; }
        
        let dx = dx_min + rng.next_f64() * (dx_max - dx_min);
        let dy = dy_min + rng.next_f64() * (dy_max - dy_min);
        
        let mut inside_stock = true;
        let mut translated = Vec::new();
        
        if !stock_is_rect {
            translated = translate_polygon(&rotated, dx, dy);
            inside_stock = polygon_inside(&translated, stock_poly);
        }
        
        if inside_stock {
            if translated.is_empty() {
                translated = translate_polygon(&rotated, dx, dy);
            }
            return Some((dx, dy, rot, translated));
        }
    }
    
    Some((0.0, 0.0, 0.0, piece.to_vec()))
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().filter_map(|l| l.ok()).filter(|l| !l.trim().is_empty());

    let first_line = lines.next().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let sv = parts[0];
    let np = parts[1];

    let mut stock_poly = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap();
        let parts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        stock_poly.push((parts[0], parts[1]));
    }

    let mut pieces = Vec::new();
    for i in 0..np {
        let line = lines.next().unwrap();
        let pv: usize = line.trim().parse().unwrap();
        let mut piece_poly = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap();
            let parts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
            piece_poly.push((parts[0], parts[1]));
        }
        pieces.push((i, piece_poly));
    }
    
    pieces.sort_by(|a, b| {
        let area_a = polygon_area(&a.1);
        let area_b = polygon_area(&b.1);
        area_b.partial_cmp(&area_a).unwrap()
    });

    let stock_area = polygon_area(&stock_poly);
    let stock_is_rect = is_axis_aligned_rectangle(&stock_poly);
    
    let mut stocks: Vec<Stock> = Vec::new();
    let mut results = vec![(0, 0.0, 0.0, 0.0); np];
    
    let start_time = Instant::now();
    let max_time_secs = 290.0;
    
    for (idx, (piece_idx, piece_poly)) in pieces.into_iter().enumerate() {
        let elapsed = start_time.elapsed().as_secs_f64();
        let time_left = max_time_secs - elapsed;
        let skip_existing = time_left < 5.0;
        
        let piece_area = polygon_area(&piece_poly);
        let mut placed = false;
        
        if !skip_existing {
            if let Some((s_idx, dx, dy, rot, translated)) = place_in_existing_stocks(&piece_poly, piece_area, stock_area, &stock_poly, &mut stocks, stock_is_rect) {
                let bb = bounding_box(&translated);
                stocks[s_idx].placed.push(PlacedPiece { 
                    polygon: translated,
                    bbox: bb
                });
                stocks[s_idx].used_area += piece_area;
                results[piece_idx] = (s_idx, dx, dy, rot);
                placed = true;
            }
        }
        
        if !placed {
            let (dx, dy, rot, translated) = place_in_new_stock(&piece_poly, &stock_poly, stock_is_rect).unwrap();
            let bb = bounding_box(&translated);
            stocks.push(Stock {
                placed: vec![PlacedPiece { polygon: translated, bbox: bb }],
                used_area: piece_area,
            });
            results[piece_idx] = (stocks.len() - 1, dx, dy, rot);
        }
    }
    
    println!("{}", stocks.len());
    for (s_idx, dx, dy, rot) in results {
        println!("{} {:.6} {:.6} {:.6}", s_idx, dx, dy, rot);
    }
}
