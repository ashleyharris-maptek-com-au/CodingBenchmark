using System;
using System.IO;
using System.Runtime.CompilerServices;
using System.Text;

class Program
{
    static void Main(string[] args)
    {
        // Use extremely fast custom I/O for millions of elements
        using var stream = new BufferedStream(Console.OpenStandardInput(), 1024 * 1024);
        var reader = new CustomReader(stream);

        int N = reader.ReadInt();
        if (N <= 0)
        {
            Console.WriteLine(0);
            return;
        }
        int stockLength = reader.ReadInt();

        Cut[] cuts = new Cut[N];
        for (int i = 0; i < N; i++)
        {
            cuts[i] = new Cut { Size = reader.ReadInt(), OriginalIndex = i };
        }

        // FFD (First-Fit Decreasing) Heuristic: O(N log N)
        Array.Sort(cuts);

        // Segment tree to maintain the maximum remaining space in each bin subtree.
        // This allows us to find the first bin that can accommodate a cut in O(log N) time.
        // Maximum number of bins is N. Tree size 4*N is safely bounded.
        int[] maxSpace = new int[4 * N];
        Array.Fill(maxSpace, stockLength);

        // Linked list to store original indices of cuts for each bin efficiently.
        // This avoids allocating List<int> objects and reduces GC pressure to zero.
        int[] binHead = new int[N];
        Array.Fill(binHead, -1);
        int[] nextItem = new int[N];

        int numBins = 0;

        for (int i = 0; i < N; i++)
        {
            int binIndex = QueryAndUpdate(maxSpace, 1, 0, N - 1, cuts[i].Size);
            if (binIndex >= numBins)
            {
                numBins = binIndex + 1;
            }
            
            // Prepend cut index to the bin's linked list
            int cutOriginalIndex = cuts[i].OriginalIndex;
            nextItem[cutOriginalIndex] = binHead[binIndex];
            binHead[binIndex] = cutOriginalIndex;
        }

        // Fast output
        using var outStream = new BufferedStream(Console.OpenStandardOutput(), 1024 * 1024);
        using var writer = new StreamWriter(outStream, new UTF8Encoding(false), 1024 * 1024);
        
        writer.WriteLine(numBins);
        for (int i = 0; i < numBins; i++)
        {
            int curr = binHead[i];
            bool first = true;
            while (curr != -1)
            {
                if (!first) writer.Write(' ');
                writer.Write(curr);
                first = false;
                curr = nextItem[curr];
            }
            writer.WriteLine();
        }
    }

    struct Cut : IComparable<Cut>
    {
        public int Size;
        public int OriginalIndex;

        public int CompareTo(Cut other)
        {
            int cmp = other.Size.CompareTo(this.Size); // Descending order
            // Tie breaker for stable sorting
            if (cmp == 0) return this.OriginalIndex.CompareTo(other.OriginalIndex);
            return cmp;
        }
    }

    // Finds the smallest bin index that can accommodate 'val' space and updates the tree
    static int QueryAndUpdate(int[] tree, int node, int l, int r, int val)
    {
        if (l == r)
        {
            tree[node] -= val;
            return l;
        }
        int mid = l + (r - l) / 2;
        int res;
        
        // If left child has enough space, go left to maintain "First-Fit" property
        if (tree[2 * node] >= val)
        {
            res = QueryAndUpdate(tree, 2 * node, l, mid, val);
        }
        else
        {
            res = QueryAndUpdate(tree, 2 * node + 1, mid + 1, r, val);
        }
        
        // Update current node max space
        tree[node] = Math.Max(tree[2 * node], tree[2 * node + 1]);
        return res;
    }

    class CustomReader
    {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 1024];
        private int head = 0;
        private int tail = 0;

        public CustomReader(Stream stream)
        {
            this.stream = stream;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int ReadInt()
        {
            int c = Read();
            while (c != -1 && c <= 32)
            {
                c = Read();
            }
            if (c == -1) return -1;

            int res = 0;
            while (c > 32)
            {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return res;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (head >= tail)
            {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }
}
