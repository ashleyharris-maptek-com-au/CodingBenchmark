#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>
#include <climits>
#include <cstdlib>
#include <random>

using namespace std;

struct Task {
    int machine;
    int duration;
};

int J, M;
vector<vector<Task>> jobs;
vector<vector<int>> work_remaining;

void compute_work_remaining() {
    work_remaining.assign(J, vector<int>());
    for (int j = 0; j < J; ++j) {
        work_remaining[j].resize(jobs[j].size());
        int rem = 0;
        for (int t = (int)jobs[j].size() - 1; t >= 0; --t) {
            rem += jobs[j][t].duration;
            work_remaining[j][t] = rem;
        }
    }
}

void run_gt_heuristic(int heuristic_type, vector<vector<int>>& out_schedule, int& out_makespan, mt19937& rng) {
    vector<int> ready_time(J, 0);
    vector<int> m_free_time(M, 0);
    vector<int> next_task(J, 0);
    out_schedule.assign(J, vector<int>());
    for(int j=0; j<J; ++j) out_schedule[j].resize(jobs[j].size(), 0);
    
    int total_t = 0;
    vector<int> active_jobs;
    for(int j=0; j<J; ++j) {
        total_t += jobs[j].size();
        if (jobs[j].size() > 0) active_jobs.push_back(j);
    }
    
    int completed_tasks = 0;
    while(completed_tasks < total_t) {
        int min_comp = INT_MAX;
        int m_star = -1;
        int best_job = -1;
        
        for (int i = 0; i < (int)active_jobs.size(); ) {
            int j = active_jobs[i];
            if (next_task[j] == jobs[j].size()) {
                active_jobs[i] = active_jobs.back();
                active_jobs.pop_back();
                continue;
            }
            auto& task = jobs[j][next_task[j]];
            int start = max(ready_time[j], m_free_time[task.machine]);
            int comp = start + task.duration;
            if (comp < min_comp) {
                min_comp = comp;
                m_star = task.machine;
                best_job = j;
            }
            i++;
        }
        
        vector<int> S;
        for (int j : active_jobs) {
            auto& task = jobs[j][next_task[j]];
            if (task.machine == m_star && ready_time[j] < min_comp) {
                S.push_back(j);
            }
        }
        
        if (S.empty()) {
            S.push_back(best_job);
        }
        
        int chosen_j = S[0];
        if (S.size() > 1) {
            if (heuristic_type == 0) { // SPT
                int min_dur = INT_MAX;
                for (int j : S) {
                    if (jobs[j][next_task[j]].duration < min_dur) {
                        min_dur = jobs[j][next_task[j]].duration;
                        chosen_j = j;
                    }
                }
            } else if (heuristic_type == 1) { // MWKR
                int max_rem = -1;
                for (int j : S) {
                    if (work_remaining[j][next_task[j]] > max_rem) {
                        max_rem = work_remaining[j][next_task[j]];
                        chosen_j = j;
                    }
                }
            } else if (heuristic_type == 2) { // Random / GRASP
                uniform_int_distribution<int> dist(0, S.size() - 1);
                chosen_j = S[dist(rng)];
            } else if (heuristic_type == 3) { // GRASP with bias to MWKR
                sort(S.begin(), S.end(), [&](int a, int b) {
                    return work_remaining[a][next_task[a]] > work_remaining[b][next_task[b]];
                });
                int limit = min((int)S.size(), 3);
                uniform_int_distribution<int> dist(0, limit - 1);
                chosen_j = S[dist(rng)];
            }
        }
        
        auto& task = jobs[chosen_j][next_task[chosen_j]];
        int start = max(ready_time[chosen_j], m_free_time[m_star]);
        int comp = start + task.duration;
        
        ready_time[chosen_j] = comp;
        m_free_time[m_star] = comp;
        out_schedule[chosen_j][next_task[chosen_j]] = start;
        next_task[chosen_j]++;
        completed_tasks++;
    }
    
    out_makespan = 0;
    for (int m : m_free_time) out_makespan = max(out_makespan, m);
}

void dfs(vector<int>& ready_time, vector<int>& m_free_time, vector<int>& next_task, vector<vector<int>>& current_schedule, 
         int& dfs_nodes, int& best_makespan, vector<vector<int>>& best_schedule,
         chrono::time_point<chrono::high_resolution_clock> start_time, double time_limit_sec) {
    
    if (dfs_nodes % 10000 == 0) {
        chrono::duration<double> elapsed = chrono::high_resolution_clock::now() - start_time;
        if (elapsed.count() > time_limit_sec) return;
    }
    dfs_nodes++;
    
    int min_comp = INT_MAX;
    int m_star = -1;
    int best_job = -1;
    bool done = true;
    for (int j = 0; j < J; ++j) {
        if (next_task[j] < jobs[j].size()) {
            done = false;
            auto& task = jobs[j][next_task[j]];
            int start = max(ready_time[j], m_free_time[task.machine]);
            int comp = start + task.duration;
            if (comp < min_comp) {
                min_comp = comp;
                m_star = task.machine;
                best_job = j;
            }
        }
    }
    
    if (done) {
        int makespan = 0;
        for (int m : m_free_time) makespan = max(makespan, m);
        if (makespan < best_makespan) {
            best_makespan = makespan;
            best_schedule = current_schedule;
        }
        return;
    }
    
    vector<int> S;
    for (int j = 0; j < J; ++j) {
        if (next_task[j] < jobs[j].size()) {
            auto& task = jobs[j][next_task[j]];
            if (task.machine == m_star && ready_time[j] < min_comp) {
                S.push_back(j);
            }
        }
    }
    if (S.empty()) {
        S.push_back(best_job);
    }
    
    sort(S.begin(), S.end(), [&](int a, int b) {
        return work_remaining[a][next_task[a]] > work_remaining[b][next_task[b]];
    });
    
    for (int j : S) {
        auto& task = jobs[j][next_task[j]];
        int start = max(ready_time[j], m_free_time[m_star]);
        int comp = start + task.duration;
        
        if (comp >= best_makespan) continue;
        
        int old_ready = ready_time[j];
        int old_free = m_free_time[m_star];
        
        ready_time[j] = comp;
        m_free_time[m_star] = comp;
        current_schedule[j][next_task[j]] = start;
        next_task[j]++;
        
        dfs(ready_time, m_free_time, next_task, current_schedule, dfs_nodes, best_makespan, best_schedule, start_time, time_limit_sec);
        
        next_task[j]--;
        ready_time[j] = old_ready;
        m_free_time[m_star] = old_free;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> J >> M)) return 0;

    jobs.resize(J);
    int total_tasks = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        total_tasks += T;
        for (int t = 0; t < T; ++t) {
            int m, d;
            cin >> m >> d;
            jobs[j].push_back({m, d});
        }
    }

    if (total_tasks == 0) return 0;

    compute_work_remaining();

    vector<vector<int>> best_schedule;
    int best_makespan = INT_MAX;
    
    mt19937 rng(42);
    
    if (total_tasks <= 30) {
        run_gt_heuristic(1, best_schedule, best_makespan, rng);
        
        vector<int> ready_time(J, 0);
        vector<int> m_free_time(M, 0);
        vector<int> next_task(J, 0);
        vector<vector<int>> current_schedule(J);
        for(int j=0; j<J; ++j) current_schedule[j].resize(jobs[j].size(), 0);
        
        int dfs_nodes = 0;
        auto start_time = chrono::high_resolution_clock::now();
        dfs(ready_time, m_free_time, next_task, current_schedule, dfs_nodes, best_makespan, best_schedule, start_time, 2.0);
    } 
    else if (total_tasks <= 5000) {
        auto start_time = chrono::high_resolution_clock::now();
        vector<vector<int>> sched;
        int ms;
        
        run_gt_heuristic(0, sched, ms, rng);
        if (ms < best_makespan) { best_makespan = ms; best_schedule = sched; }
        
        run_gt_heuristic(1, sched, ms, rng);
        if (ms < best_makespan) { best_makespan = ms; best_schedule = sched; }
        
        while (true) {
            chrono::duration<double> elapsed = chrono::high_resolution_clock::now() - start_time;
            if (elapsed.count() > 2.0) break;
            
            run_gt_heuristic(3, sched, ms, rng);
            if (ms < best_makespan) { best_makespan = ms; best_schedule = sched; }
            
            run_gt_heuristic(2, sched, ms, rng);
            if (ms < best_makespan) { best_makespan = ms; best_schedule = sched; }
        }
    }
    else {
        auto start_time = chrono::high_resolution_clock::now();
        vector<vector<int>> sched;
        int ms;
        
        run_gt_heuristic(0, sched, ms, rng);
        if (ms < best_makespan) { best_makespan = ms; best_schedule = sched; }
        
        run_gt_heuristic(1, sched, ms, rng);
        if (ms < best_makespan) { best_makespan = ms; best_schedule = sched; }
        
        while (true) {
            chrono::duration<double> elapsed = chrono::high_resolution_clock::now() - start_time;
            if (elapsed.count() > 2.0) break;
            
            run_gt_heuristic(2, sched, ms, rng);
            if (ms < best_makespan) { best_makespan = ms; best_schedule = sched; }
        }
    }

    for (int j = 0; j < J; ++j) {
        for (int t = 0; t < jobs[j].size(); ++t) {
            cout << best_schedule[j][t] << " " << jobs[j][t].machine << (t == jobs[j].size() - 1 ? "" : " ");
        }
        cout << "\n";
    }

    return 0;
}
