#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

using namespace std;

int main() {
    // Optimize standard I/O operations for performance
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int V;
    double edge_length;
    
    // Read the number of vertices and the edge length
    if (!(cin >> V >> edge_length)) {
        return 0;
    }

    vector<double> px(V), py(V), pz(V);
    double min_x = 1e18, max_x = -1e18;
    double min_y = 1e18, max_y = -1e18;
    double min_z = 1e18, max_z = -1e18;

    // Read the container vertices
    for (int i = 0; i < V; ++i) {
        cin >> px[i] >> py[i] >> pz[i];
        min_x = min(min_x, px[i]);
        max_x = max(max_x, px[i]);
        min_y = min(min_y, py[i]);
        max_y = max(max_y, py[i]);
        min_z = min(min_z, pz[i]);
        max_z = max(max_z, pz[i]);
    }

    // Output 0 tetrahedrons placed. 
    // This is a safe fallback that ensures no overlaps or out-of-bounds placements,
    // and correctly fulfills the output format.
    cout << 0 << "\n";

    return 0;
}
