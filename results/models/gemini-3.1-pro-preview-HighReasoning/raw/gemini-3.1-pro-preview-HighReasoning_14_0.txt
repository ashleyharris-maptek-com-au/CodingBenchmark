using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class MinesweeperSolver
{
    static int H, W;
    static byte[] board;
    static bool[] inQueue;
    static Queue<int> queue;
    static List<int> newSafes = new List<int>();

    static void Main()
    {
        using (Stream s = Console.OpenStandardInput())
        {
            FastScanner scanner = new FastScanner(s);
            H = scanner.ReadInt();
            W = scanner.ReadInt();
            if (H == 0 || W == 0) return;
            board = new byte[H * W];
            inQueue = new bool[H * W];
            queue = new Queue<int>(Math.Min(10000000, H * W / 4 + 100)); 
            
            int expected = H * W;
            int idx = 0;
            while(idx < expected)
            {
                int c = scanner.ReadByte();
                if (c == -1) break;
                if (c == ' ') board[idx++] = 9;
                else if (c == '*') board[idx++] = 10;
                else if (c >= '0' && c <= '8') board[idx++] = (byte)(c - '0');
            }
        }

        // Initialize queue with all active number cells
        for (int i = 0; i < H * W; i++)
        {
            if (board[i] <= 8)
            {
                int unkCount = 0;
                int[] unks = new int[8];
                GetNeighborsInfo(i, out unks, out unkCount, out int mineCount);
                if (unkCount > 0)
                {
                    queue.Enqueue(i);
                    inQueue[i] = true;
                }
            }
        }

        List<int> activeBorder = new List<int>();
        bool changed = true;
        
        while (changed)
        {
            changed = false;
            activeBorder.Clear();
            
            // 1. Trivial Phase
            while (queue.Count > 0)
            {
                int p = queue.Dequeue();
                inQueue[p] = false;
                
                if (board[p] > 8) continue;
                
                GetNeighborsInfo(p, out int[] unks, out int unkCount, out int mineCount);
                if (unkCount == 0) continue;
                
                int req = board[p] - mineCount;
                if (req == 0)
                {
                    for (int i = 0; i < unkCount; i++) MarkSafe(unks[i]);
                    changed = true;
                }
                else if (req == unkCount)
                {
                    for (int i = 0; i < unkCount; i++) MarkMine(unks[i]);
                    changed = true;
                }
                else
                {
                    activeBorder.Add(p);
                }
            }
            
            if (changed) continue;

            // Clean activeBorder before Subset Phase
            List<int> cleanBorder = new List<int>();
            foreach (int p in activeBorder)
            {
                if (board[p] <= 8)
                {
                    GetNeighborsInfo(p, out int[] _, out int uc, out int _);
                    if (uc > 0) cleanBorder.Add(p);
                }
            }
            activeBorder = cleanBorder;

            if (activeBorder.Count == 0) break;

            // 2. Subset Phase
            bool subsetFound = false;
            for (int i = 0; i < activeBorder.Count; i++)
            {
                int p = activeBorder[i];
                GetNeighborsInfo(p, out int[] unksP, out int unkCountP, out int mineCountP);
                if (unkCountP == 0) continue;
                int reqP = board[p] - mineCountP;

                int[] qs = new int[64];
                int qCount = 0;
                for (int j = 0; j < unkCountP; j++)
                {
                    int[] qCands = GetNumberNeighbors(unksP[j], out int cCount);
                    for (int k = 0; k < cCount; k++)
                    {
                        int q = qCands[k];
                        if (q != p)
                        {
                            bool present = false;
                            for (int m = 0; m < qCount; m++) if (qs[m] == q) { present = true; break; }
                            if (!present) qs[qCount++] = q;
                        }
                    }
                }
                
                for (int k = 0; k < qCount; k++)
                {
                    int q = qs[k];
                    GetNeighborsInfo(q, out int[] unksQ, out int unkCountQ, out int mineCountQ);
                    int reqQ = board[q] - mineCountQ;
                    
                    if (unkCountP > 0 && unkCountP < unkCountQ && IsSubset(unksP, unkCountP, unksQ, unkCountQ))
                    {
                        int diffReq = reqQ - reqP;
                        int diffUnk = unkCountQ - unkCountP;
                        
                        if (diffReq == 0)
                        {
                            for(int j = 0; j < unkCountQ; j++)
                            {
                                if (!Contains(unksP, unkCountP, unksQ[j]))
                                {
                                    MarkSafe(unksQ[j]);
                                    subsetFound = true;
                                }
                            }
                        }
                        else if (diffReq == diffUnk)
                        {
                            for(int j = 0; j < unkCountQ; j++)
                            {
                                if (!Contains(unksP, unkCountP, unksQ[j]))
                                {
                                    MarkMine(unksQ[j]);
                                    subsetFound = true;
                                }
                            }
                        }
                    }
                }
                if (subsetFound) break;
            }
            
            if (subsetFound)
            {
                changed = true;
                continue;
            }

            // 3. CSP Phase
            if (RunCSP(activeBorder))
            {
                changed = true;
                continue;
            }
        }
        
        // Output results
        newSafes.Sort();
        using (StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), new System.Text.UTF8Encoding(false), 65536))
        {
            writer.WriteLine(newSafes.Count);
            foreach (int s in newSafes)
            {
                writer.WriteLine($"{s % W} {s / W}");
            }
        }
    }

    static void MarkSafe(int p)
    {
        if (board[p] == 11 || board[p] == 10) return;
        board[p] = 11;
        newSafes.Add(p);
        
        int[] nums = GetNumberNeighbors(p, out int numCount);
        for (int i = 0; i < numCount; i++)
        {
            int q = nums[i];
            if (!inQueue[q])
            {
                queue.Enqueue(q);
                inQueue[q] = true;
            }
        }
    }

    static void MarkMine(int p)
    {
        if (board[p] == 11 || board[p] == 10) return;
        board[p] = 10;
        
        int[] nums = GetNumberNeighbors(p, out int numCount);
        for (int i = 0; i < numCount; i++)
        {
            int q = nums[i];
            if (!inQueue[q])
            {
                queue.Enqueue(q);
                inQueue[q] = true;
            }
        }
    }

    static void GetNeighborsInfo(int p, out int[] unks, out int unkCount, out int mineCount)
    {
        unks = new int[8];
        unkCount = 0;
        mineCount = 0;
        int y = p / W;
        int x = p % W;
        for (int dy = -1; dy <= 1; dy++)
        {
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dy == 0 && dx == 0) continue;
                int ny = y + dy;
                int nx = x + dx;
                if (ny >= 0 && ny < H && nx >= 0 && nx < W)
                {
                    int np = ny * W + nx;
                    byte val = board[np];
                    if (val == 9) unks[unkCount++] = np;
                    else if (val == 10) mineCount++;
                }
            }
        }
    }

    static int[] GetNumberNeighbors(int p, out int numCount)
    {
        int[] nums = new int[8];
        numCount = 0;
        int y = p / W;
        int x = p % W;
        for (int dy = -1; dy <= 1; dy++)
        {
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dy == 0 && dx == 0) continue;
                int ny = y + dy;
                int nx = x + dx;
                if (ny >= 0 && ny < H && nx >= 0 && nx < W)
                {
                    int np = ny * W + nx;
                    byte val = board[np];
                    if (val <= 8) nums[numCount++] = np;
                }
            }
        }
        return nums;
    }

    static bool Contains(int[] arr, int count, int val)
    {
        for (int i = 0; i < count; i++) if (arr[i] == val) return true;
        return false;
    }

    static bool IsSubset(int[] sub, int subCount, int[] super, int superCount)
    {
        if (subCount > superCount) return false;
        for (int i = 0; i < subCount; i++)
        {
            bool found = false;
            for (int j = 0; j < superCount; j++)
            {
                if (sub[i] == super[j]) { found = true; break; }
            }
            if (!found) return false;
        }
        return true;
    }

    static bool RunCSP(List<int> activeNumbers)
    {
        HashSet<int> activeUnknowns = new HashSet<int>();
        foreach (int p in activeNumbers)
        {
            GetNeighborsInfo(p, out int[] unks, out int unkCount, out int _);
            for (int i = 0; i < unkCount; i++) activeUnknowns.Add(unks[i]);
        }

        if (activeUnknowns.Count == 0) return false;

        Dictionary<int, List<int>> unkToNum = new Dictionary<int, List<int>>();
        foreach (int u in activeUnknowns)
        {
            unkToNum[u] = new List<int>();
            int[] nums = GetNumberNeighbors(u, out int numCount);
            for(int i = 0; i < numCount; i++)
            {
                unkToNum[u].Add(nums[i]);
            }
        }
        
        HashSet<int> visitedUnknowns = new HashSet<int>();
        bool cspFoundAny = false;

        foreach (int startU in activeUnknowns)
        {
            if (visitedUnknowns.Contains(startU)) continue;

            List<int> compU = new List<int>();
            HashSet<int> compE = new HashSet<int>();
            
            Queue<int> qCSP = new Queue<int>();
            qCSP.Enqueue(startU);
            visitedUnknowns.Add(startU);

            while (qCSP.Count > 0)
            {
                int u = qCSP.Dequeue();
                compU.Add(u);
                
                foreach (int e in unkToNum[u])
                {
                    if (compE.Add(e)) 
                    {
                        GetNeighborsInfo(e, out int[] unks, out int unkCount, out int _);
                        for (int i = 0; i < unkCount; i++)
                        {
                            int nextU = unks[i];
                            if (visitedUnknowns.Add(nextU))
                            {
                                qCSP.Enqueue(nextU);
                            }
                        }
                    }
                }
            }

            bool found = SolveComponent(compU, compE.ToList());
            if (found) cspFoundAny = true;
        }

        return cspFoundAny;
    }

    static bool SolveComponent(List<int> U, List<int> E)
    {
        int uCount = U.Count;
        int eCount = E.Count;
        
        if (uCount > 150) return false;

        List<int>[] eqVars = new List<int>[eCount];
        int[] req = new int[eCount];
        List<int>[] varEqs = new List<int>[uCount];

        Dictionary<int, int> uMap = new Dictionary<int, int>();
        for (int i = 0; i < uCount; i++)
        {
            uMap[U[i]] = i;
            varEqs[i] = new List<int>();
        }

        for (int i = 0; i < eCount; i++)
        {
            int e = E[i];
            eqVars[i] = new List<int>();
            GetNeighborsInfo(e, out int[] unks, out int unkCount, out int mineCount);
            req[i] = board[e] - mineCount;
            
            for (int j = 0; j < unkCount; j++)
            {
                if (uMap.TryGetValue(unks[j], out int varIdx))
                {
                    eqVars[i].Add(varIdx);
                    varEqs[varIdx].Add(i);
                }
            }
        }

        int[] currentSum = new int[eCount];
        int[] remainingVars = new int[eCount];
        for (int i = 0; i < eCount; i++) remainingVars[i] = eqVars[i].Count;

        int validSolutions = 0;
        int[] minVarPossible = new int[uCount];
        int[] maxVarPossible = new int[uCount];
        for (int i = 0; i < uCount; i++)
        {
            minVarPossible[i] = 1;
            maxVarPossible[i] = 0;
        }
        int[] currentAssignment = new int[uCount];

        int nodesVisited = 0;
        bool abort = false;

        int[] varOrder = new int[uCount];
        for(int i = 0; i < uCount; i++) varOrder[i] = i;
        Array.Sort(varOrder, (a, b) => varEqs[b].Count.CompareTo(varEqs[a].Count));
        
        void SolveIndir(int depth)
        {
            if (abort) return;
            nodesVisited++;
            if (nodesVisited > 200000) { abort = true; return; }

            if (depth == uCount)
            {
                for (int i = 0; i < eCount; i++)
                    if (currentSum[i] != req[i]) return;

                validSolutions++;
                for (int i = 0; i < uCount; i++)
                {
                    if (currentAssignment[i] == 1) maxVarPossible[i] = 1;
                    else minVarPossible[i] = 0;
                }
                return;
            }

            int varIdx = varOrder[depth];

            bool canAssign0 = true;
            foreach (int eIdx in varEqs[varIdx])
            {
                if (currentSum[eIdx] + remainingVars[eIdx] - 1 < req[eIdx])
                {
                    canAssign0 = false; break;
                }
            }

            if (canAssign0)
            {
                foreach (int eIdx in varEqs[varIdx]) remainingVars[eIdx]--;
                currentAssignment[varIdx] = 0;
                SolveIndir(depth + 1);
                foreach (int eIdx in varEqs[varIdx]) remainingVars[eIdx]++;
            }

            bool canAssign1 = true;
            foreach (int eIdx in varEqs[varIdx])
            {
                if (currentSum[eIdx] + 1 > req[eIdx])
                {
                    canAssign1 = false; break;
                }
            }

            if (canAssign1)
            {
                foreach (int eIdx in varEqs[varIdx])
                {
                    remainingVars[eIdx]--;
                    currentSum[eIdx]++;
                }
                currentAssignment[varIdx] = 1;
                SolveIndir(depth + 1);
                foreach (int eIdx in varEqs[varIdx])
                {
                    remainingVars[eIdx]++;
                    currentSum[eIdx]--;
                }
            }
        }

        SolveIndir(0);

        if (abort || validSolutions == 0) return false;

        bool foundAny = false;
        for (int i = 0; i < uCount; i++)
        {
            if (maxVarPossible[i] == 0)
            {
                MarkSafe(U[i]);
                foundAny = true;
            }
            else if (minVarPossible[i] == 1)
            {
                MarkMine(U[i]);
                foundAny = true;
            }
        }

        return foundAny;
    }

    class FastScanner
    {
        Stream s;
        byte[] buf = new byte[1024 * 1024];
        int head = 0, tail = 0;

        public FastScanner(Stream stream) { s = stream; }

        public int ReadByte()
        {
            if (head >= tail)
            {
                head = 0;
                tail = s.Read(buf, 0, buf.Length);
                if (tail == 0) return -1;
            }
            return buf[head++];
        }
        
        public int ReadInt()
        {
            int res = 0;
            bool started = false;
            while (true)
            {
                int c = ReadByte();
                if (c == -1) return res;
                if (c >= '0' && c <= '9')
                {
                    res = res * 10 + (c - '0');
                    started = true;
                }
                else if (started) break;
            }
            return res;
        }
    }
}
