#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>
#include <cstdint>
#include <execution>

#ifdef _WIN32
#include <fcntl.h>
#include <io.h>
#endif

using namespace std;

const int BUF_SIZE = 1 << 26; // 67 MB
char buf[BUF_SIZE];
int buf_ptr = 0;
int buf_size = 0;

inline char next_char() {
    if (buf_ptr == buf_size) {
        buf_size = (int)fread(buf, 1, BUF_SIZE, stdin);
        buf_ptr = 0;
        if (buf_size == 0) return EOF;
    }
    return buf[buf_ptr++];
}

inline bool read_int(int &x) {
    char c = next_char();
    while (c != EOF && c <= ' ') c = next_char();
    if (c == EOF) return false;
    bool neg = false;
    if (c == '-') { neg = true; c = next_char(); }
    x = 0;
    while (c > ' ') {
        x = x * 10 + (c - '0');
        c = next_char();
    }
    if (neg) x = -x;
    return true;
}

inline bool read_float(float &x) {
    char c = next_char();
    while (c != EOF && c <= ' ') c = next_char();
    if (c == EOF) return false;
    bool neg = false;
    if (c == '-') { neg = true; c = next_char(); }
    double res = 0;
    while (c > ' ' && c != '.' && c != 'e' && c != 'E') {
        res = res * 10 + (c - '0');
        c = next_char();
    }
    if (c == '.') {
        c = next_char();
        double frac = 1;
        while (c > ' ' && c != 'e' && c != 'E') {
            frac *= 0.1;
            res += (c - '0') * frac;
            c = next_char();
        }
    }
    int exp = 0;
    if (c == 'e' || c == 'E') {
        c = next_char();
        bool exp_neg = false;
        if (c == '-') { exp_neg = true; c = next_char(); }
        else if (c == '+') { c = next_char(); }
        while (c > ' ') {
            exp = exp * 10 + (c - '0');
            c = next_char();
        }
        if (exp_neg) exp = -exp;
    }
    if (exp != 0) {
        res *= std::pow(10.0, exp);
    }
    x = neg ? (float)-res : (float)res;
    return true;
}

char out_buf[BUF_SIZE];
int out_ptr = 0;

inline void flush_out() {
    if (out_ptr > 0) {
        fwrite(out_buf, 1, out_ptr, stdout);
        out_ptr = 0;
    }
}

inline void write_int(int x) {
    if (x == 0) {
        if (out_ptr + 2 > BUF_SIZE) flush_out();
        out_buf[out_ptr++] = '0';
        out_buf[out_ptr++] = ' ';
        return;
    }
    char temp[12];
    int t_ptr = 0;
    while (x > 0) {
        temp[t_ptr++] = '0' + (x % 10);
        x /= 10;
    }
    if (out_ptr + t_ptr + 1 > BUF_SIZE) flush_out();
    while (t_ptr > 0) {
        out_buf[out_ptr++] = temp[--t_ptr];
    }
    out_buf[out_ptr++] = ' ';
}

inline uint64_t hilbert(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int s = 23; s >= 0; s--) {
        uint32_t rx = (x >> s) & 1;
        uint32_t ry = (y >> s) & 1;
        d += ((uint64_t)((3 * rx) ^ ry)) << (2 * s);
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x;
            x = y;
            y = t;
        }
    }
    return d;
}

struct Element {
    uint64_t h;
    int id;
    bool operator<(const Element& other) const {
        return h < other.h;
    }
};

int main() {
#ifdef _WIN32
    _setmode(_fileno(stdin), _O_BINARY);
    _setmode(_fileno(stdout), _O_BINARY);
#endif
    auto start_time = std::chrono::high_resolution_clock::now();
    auto elapsed = [&]() {
        auto now = std::chrono::high_resolution_clock::now();
        return std::chrono::duration_cast<std::chrono::duration<double>>(now - start_time).count();
    };

    int N;
    if (!read_int(N)) return 0;

    std::vector<float> X(N);
    std::vector<float> Y(N);
    float min_x = 1e30f, max_x = -1e30f;
    float min_y = 1e30f, max_y = -1e30f;

    for (int i = 0; i < N; ++i) {
        read_float(X[i]);
        read_float(Y[i]);
        if (X[i] < min_x) min_x = X[i];
        if (X[i] > max_x) max_x = X[i];
        if (Y[i] < min_y) min_y = Y[i];
        if (Y[i] > max_y) max_y = Y[i];
    }

    std::vector<int> ids(N);
    bool do_2opt = (N <= 2000000);

    if (N <= 2000) {
        std::vector<bool> visited(N, false);
        ids[0] = 0;
        visited[0] = true;
        int curr = 0;
        for (int i = 1; i < N; ++i) {
            int best_next = -1;
            float best_d = 1e30f;
            float cx = X[curr];
            float cy = Y[curr];
            for (int j = 0; j < N; ++j) {
                if (!visited[j]) {
                    float dx = X[j] - cx;
                    float dy = Y[j] - cy;
                    float d = dx*dx + dy*dy;
                    if (d < best_d) {
                        best_d = d;
                        best_next = j;
                    }
                }
            }
            ids[i] = best_next;
            visited[best_next] = true;
            curr = best_next;
        }
    } else {
        std::vector<Element> elements(N);
        float range_x = max_x - min_x;
        float range_y = max_y - min_y;
        float scale_x = range_x > 0 ? (16777215.0f / range_x) : 0;
        float scale_y = range_y > 0 ? (16777215.0f / range_y) : 0;

        for (int i = 0; i < N; ++i) {
            uint32_t ix = (uint32_t)((X[i] - min_x) * scale_x);
            uint32_t iy = (uint32_t)((Y[i] - min_y) * scale_y);
            if (ix > 16777215) ix = 16777215;
            if (iy > 16777215) iy = 16777215;
            elements[i].h = hilbert(ix, iy);
            elements[i].id = i;
        }

        if (!do_2opt) {
            std::vector<float>().swap(X);
            std::vector<float>().swap(Y);
        }

        std::sort(std::execution::par, elements.begin(), elements.end());

        for (int i = 0; i < N; ++i) {
            ids[i] = elements[i].id;
        }
        std::vector<Element>().swap(elements);
    }

    if (do_2opt) {
        int window_size = 50;
        if (N <= 2000) window_size = N;
        else if (N <= 10000) window_size = 600;
        else if (N <= 100000) window_size = 200;
        else if (N <= 500000) window_size = 80;
        else window_size = 40;

        auto dist = [&](int a, int b) {
            float dx = X[a] - X[b];
            float dy = Y[a] - Y[b];
            return std::sqrt(dx*dx + dy*dy);
        };

        bool improvement = true;
        int passes = 0;
        while (improvement && passes < 15 && elapsed() < 280.0) {
            improvement = false;
            passes++;
            for (int i = 0; i < N - 2; ++i) {
                int max_j = std::min(N - 1, i + window_size);
                for (int j = i + 2; j < max_j; ++j) {
                    float d1 = dist(ids[i], ids[j]);
                    float d2 = dist(ids[i+1], ids[j+1]);
                    float d3 = dist(ids[i], ids[i+1]);
                    float d4 = dist(ids[j], ids[j+1]);
                    if (d1 + d2 < d3 + d4 - 1e-5f) {
                        std::reverse(ids.begin() + i + 1, ids.begin() + j + 1);
                        improvement = true;
                    }
                }
            }
            if (passes % 2 == 1 && N > 2) {
                std::rotate(ids.begin(), ids.begin() + N / 2, ids.end());
                improvement = true;
            }
        }
        
        std::vector<float>().swap(X);
        std::vector<float>().swap(Y);
    }

    int start_idx = 0;
    for (int i = 0; i < N; ++i) {
        if (ids[i] == 0) {
            start_idx = i;
            break;
        }
    }

    for (int i = 0; i < N; ++i) {
        int idx = start_idx + i;
        if (idx >= N) idx -= N;
        write_int(ids[idx]);
    }

    flush_out();
    return 0;
}
