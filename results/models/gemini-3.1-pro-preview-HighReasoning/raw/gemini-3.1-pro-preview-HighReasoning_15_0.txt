use std::io::Read;

fn main() {
    let mut input = String::new();
    if std::io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut tokens = input.split_whitespace();

    let nv_str = tokens.next();
    if nv_str.is_none() { return; }
    let nv: usize = nv_str.unwrap().parse().unwrap();

    let mut vertices = Vec::new();
    for _ in 0..nv {
        let x: f64 = tokens.next().unwrap().parse().unwrap();
        let y: f64 = tokens.next().unwrap().parse().unwrap();
        vertices.push((x, y));
    }

    let sx: f64 = tokens.next().unwrap().parse().unwrap();
    let sy: f64 = tokens.next().unwrap().parse().unwrap();
    let sz: f64 = tokens.next().unwrap().parse().unwrap();

    if nv < 3 {
        println!("0");
        return;
    }

    let norm = (sx * sx + sy * sy + sz * sz).sqrt();
    let nx = -sx / norm;
    let ny = -sy / norm;
    let nz = -sz / norm;

    let mut qw = 1.0 + nz;
    let mut qx = -ny;
    let mut qy = nx;
    let mut qz = 0.0;

    let qnorm = (qw * qw + qx * qx + qy * qy + qz * qz).sqrt();
    if qnorm < 1e-8 {
        qw = 0.0;
        qx = 1.0;
        qy = 0.0;
        qz = 0.0;
    } else {
        qw /= qnorm;
        qx /= qnorm;
        qy /= qnorm;
        qz /= qnorm;
    }

    let x_loc_x = 1.0 - 2.0 * qy * qy - 2.0 * qz * qz;
    let x_loc_y = 2.0 * qx * qy + 2.0 * qz * qw;
    let x_loc_z = 2.0 * qx * qz - 2.0 * qy * qw;

    let y_loc_x = 2.0 * qx * qy - 2.0 * qz * qw;
    let y_loc_y = 1.0 - 2.0 * qx * qx - 2.0 * qz * qz;
    let y_loc_z = 2.0 * qy * qz + 2.0 * qx * qw;

    let s = 10.0;
    let u_x = s * (x_loc_x - x_loc_z * sx / sz);
    let u_y = s * (x_loc_y - x_loc_z * sy / sz);
    let v_x = s * (y_loc_x - y_loc_z * sx / sz);
    let v_y = s * (y_loc_y - y_loc_z * sy / sz);

    let t1_x = u_x + v_x;
    let t1_y = u_y + v_y;
    let t2_x = u_x - v_x;
    let t2_y = u_y - v_y;

    let d = t1_x * t2_y - t1_y * t2_x;

    let mut a_raw_vals = Vec::with_capacity(nv);
    let mut b_raw_vals = Vec::with_capacity(nv);
    for &(x, y) in &vertices {
        let a = (x * t2_y - y * t2_x) / d;
        let b = (-x * t1_y + y * t1_x) / d;
        a_raw_vals.push(a);
        b_raw_vals.push(b);
    }

    let min_a_raw = a_raw_vals.iter().copied().fold(f64::INFINITY, f64::min);
    let min_b_raw = b_raw_vals.iter().copied().fold(f64::INFINITY, f64::min);

    let a_offset = min_a_raw + 0.5 - 2e-7;
    let b_offset = min_b_raw + 0.5 - 2e-7;

    let mut p = Vec::with_capacity(nv);
    for i in 0..nv {
        p.push((a_raw_vals[i] - a_offset, b_raw_vals[i] - b_offset));
    }

    let min_b = p.iter().map(|v| v.1).fold(f64::INFINITY, f64::min);
    let max_b = p.iter().map(|v| v.1).fold(f64::NEG_INFINITY, f64::max);

    let j_min = (min_b - 0.5).floor() as i32;
    let j_max = (max_b + 0.5).ceil() as i32;

    let mut cells = Vec::new();

    for j in j_min..=j_max {
        let mut l_intervals = Vec::new();
        let mut intersections = Vec::new();

        for k in 0..nv {
            let p1 = p[k];
            let p2 = p[(k + 1) % nv];
            let min_y = p1.1.min(p2.1);
            let max_y = p1.1.max(p2.1);

            if min_y <= j as f64 && max_y > j as f64 {
                let t = (j as f64 - p1.1) / (p2.1 - p1.1);
                let a_int = p1.0 + t * (p2.0 - p1.0);
                intersections.push(a_int);
            }
        }

        intersections.sort_by(|x, y| x.partial_cmp(y).unwrap());
        for m in 0..intersections.len() / 2 {
            l_intervals.push((intersections[2 * m], intersections[2 * m + 1]));
        }

        for k in 0..nv {
            let p1 = p[k];
            let p2 = p[(k + 1) % nv];
            let min_y = p1.1.min(p2.1);
            let max_y = p1.1.max(p2.1);

            if max_y >= j as f64 - 0.5 && min_y <= j as f64 + 0.5 {
                if (p1.1 - p2.1).abs() < 1e-9 {
                    l_intervals.push((p1.0.min(p2.0), p1.0.max(p2.0)));
                } else {
                    let t_min = (j as f64 - 0.5 - p1.1) / (p2.1 - p1.1);
                    let t_max = (j as f64 + 0.5 - p1.1) / (p2.1 - p1.1);
                    let t_enter = t_min.min(t_max).clamp(0.0, 1.0);
                    let t_leave = t_min.max(t_max).clamp(0.0, 1.0);
                    if t_enter <= t_leave {
                        let c1 = p1.0 + t_enter * (p2.0 - p1.0);
                        let c2 = p1.0 + t_leave * (p2.0 - p1.0);
                        l_intervals.push((c1.min(c2), c1.max(c2)));
                    }
                }
            }
        }

        l_intervals.sort_by(|x, y| x.0.partial_cmp(&y.0).unwrap());
        let mut merged: Vec<(f64, f64)> = Vec::new();
        for interval in l_intervals {
            if let Some(last) = merged.last_mut() {
                if interval.0 <= last.1 + 1e-6 {
                    last.1 = last.1.max(interval.1);
                } else {
                    merged.push(interval);
                }
            } else {
                merged.push(interval);
            }
        }

        for interval in merged {
            let a_val = interval.0 - 1e-7;
            let b_val = interval.1 + 1e-7;
            let i_start = (a_val + 0.5).floor() as i32;
            let i_end = (b_val - 0.5).ceil() as i32;
            for i in i_start..=i_end {
                cells.push((i, j));
            }
        }
    }

    cells.sort_unstable();
    cells.dedup();

    println!("{}", cells.len());
    for (idx, &(i, j)) in cells.iter().enumerate() {
        let a_raw = i as f64 + a_offset;
        let b_raw = j as f64 + b_offset;

        let c_prime_x = a_raw * t1_x + b_raw * t2_x;
        let c_prime_y = a_raw * t1_y + b_raw * t2_y;

        let c_z = (idx as f64) * 20.0 + 20.0;
        let c_x = c_prime_x + c_z * sx / sz;
        let c_y = c_prime_y + c_z * sy / sz;

        println!("{} {} {} {:.6} {:.6} {:.6} {:.6} 10.0", c_x, c_y, c_z, qw, qx, qy, qz);
    }
}
