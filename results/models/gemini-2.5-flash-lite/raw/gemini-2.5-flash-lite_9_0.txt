The 1D Cutting Stock Problem has been implemented using the First Fit Decreasing (FFD) algorithm in `Solver.cs`. This approach sorts the required cuts by size in descending order and greedily assigns each cut to the first available stock piece that can accommodate it, providing a near-optimal solution efficiently.

### Implementation: `Solver.cs`

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class CuttingStockSolver
{
    static void Main(string[] args)
    {
        // Read input: N (number of cuts) and stock_length
        string[] line1 = Console.ReadLine()?.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        if (line1 == null || line1.Length < 2) return;
        
        int n = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);
        
        // Read cut lengths
        string[] line2 = Console.ReadLine()?.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        if (line2 == null) return;
        
        int[] cuts = new int[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(line2[i]);
        }

        // FFD algorithm:
        // 1. Sort cuts in descending order while keeping track of original indices
        var indexedCuts = cuts.Select((length, index) => new { length, index })
                              .OrderByDescending(c => c.length)
                              .ToList();

        // 2. Pack cuts into stocks
        List<List<int>> stocks = new List<List<int>>();
        List<int> stockRemaining = new List<int>();

        foreach (var cut in indexedCuts)
        {
            bool placed = false;
            // Try to fit the cut into an existing stock piece
            for (int i = 0; i < stocks.Count; i++)
            {
                if (stockRemaining[i] >= cut.length)
                {
                    stocks[i].Add(cut.index);
                    stockRemaining[i] -= cut.length;
                    placed = true;
                    break;
                }
            }

            // If it doesn't fit in any existing piece, start a new stock
            if (!placed)
            {
                stocks.Add(new List<int> { cut.index });
                stockRemaining.Add(stockLength - cut.length);
            }
        }

        // Output results
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
```
