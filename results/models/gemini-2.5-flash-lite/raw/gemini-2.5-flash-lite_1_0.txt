// Structure to represent a city with coordinates
struct City {
    int id;
    double x, y;
};

// Function to calculate Euclidean distance between two cities
double distance(const City& c1, const City& c2) {
    double dx = c1.x - c2.x;
    double dy = c1.y - c2.y;
    return std::sqrt(dx * dx + dy * dy);
}

// --- Heuristics ---

// Nearest Neighbor heuristic
std::vector<int> nearestNeighbor(const std::vector<City>& cities, int start_node = 0) {
    int n = cities.size();
    std::vector<int> tour(n);
    std::vector<bool> visited(n, false);
    int current_city_idx = start_node;

    tour[0] = cities[start_node].id;
    visited[start_node] = true;

    for (int i = 1; i < n; ++i) {
        double min_dist = -1.0;
        int nearest_city_idx = -1;

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                double d = distance(cities[current_city_idx], cities[j]);
                if (nearest_city_idx == -1 || d < min_dist) {
                    min_dist = d;
                    nearest_city_idx = j;
                }
            }
        }
        tour[i] = cities[nearest_city_idx].id;
        visited[nearest_city_idx] = true;
        current_city_idx = nearest_city_idx;
    }
    return tour;
}

// 2-opt local search for improving a tour
std::vector<int> twoOpt(const std::vector<int>& initial_tour, const std::vector<City>& cities) {
    int n = cities.size();
    std::vector<int> best_tour = initial_tour;
    double best_distance = 0.0;

    // Calculate initial tour distance
    for (int i = 0; i < n - 1; ++i) {
        best_distance += distance(cities[initial_tour[i]], cities[initial_tour[i + 1]]);
    }
    best_distance += distance(cities[initial_tour.back()], cities[initial_tour.front()]);

    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < n - 1; ++i) {
            for (int j = i + 1; j < n; ++j) {
                std::vector<int> current_tour = best_tour;
                // Reverse the segment between i and j
                std::reverse(current_tour.begin() + i, current_tour.begin() + j + 1);

                double current_distance = 0.0;
                for (int k = 0; k < n - 1; ++k) {
                    current_distance += distance(cities[current_tour[k]], cities[current_tour[k + 1]]);
                }
                current_distance += distance(cities[current_tour.back()], cities[current_tour.front()]);

                if (current_distance < best_distance) {
                    best_tour = current_tour;
                    best_distance = current_distance;
                    improved = true;
                }
            }
        }
    }
    return best_tour;
}

// Geometric Nearest Neighbor (more structured than simple NN)
std::vector<int> geometricNearestNeighbor(const std::vector<City>& cities, int start_node = 0) {
    int n = cities.size();
    std::vector<int> tour;
    tour.reserve(n);
    std::vector<bool> visited(n, false);
    int current_city_idx = start_node;

    tour.push_back(cities[start_node].id);
    visited[start_node] = true;

    for (int i = 1; i < n; ++i) {
        double min_dist = -1.0;
        int nearest_city_idx = -1;

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                double d = distance(cities[current_city_idx], cities[j]);
                if (nearest_city_idx == -1 || d < min_dist) {
                    min_dist = d;
                    nearest_city_idx = j;
                }
            }
        }
        tour.push_back(cities[nearest_city_idx].id);
        visited[nearest_city_idx] = true;
        current_city_idx = nearest_city_idx;
    }
    return tour;
}

// --- Adaptive Strategy ---

std::vector<int> solveTSP(const std::vector<City>& cities) {
    int n = cities.size();

    // Strategy selection based on N
    if (n <= 40) {
        // For very small N, we can consider brute force or a very good heuristic.
        // Since the problem statement emphasizes "ANY number of cities" and requires
        // a single algorithm, brute-force for such small N is not the primary focus
        // but we can use a strong heuristic.
        std::cout << std::fixed << std::setprecision(10); // For potential debugging

        // Nearest Neighbor followed by 2-opt for good quality
        std::vector<int> initial_tour = nearestNeighbor(cities, 0);
        return twoOpt(initial_tour, cities);

    } else if (n <= 1000) {
        // Medium scale: Nearest Neighbor + 2-opt is a good balance of speed and quality.
        std::vector<int> initial_tour = geometricNearestNeighbor(cities, 0);
        return twoOpt(initial_tour, cities);

    } else if (n <= 100000) {
        // Large scale: 2-opt might become too slow if applied exhaustively.
        // We'll use Nearest Neighbor and apply a limited number of 2-opt iterations or
        // a variant of it, or consider a more advanced heuristic.
        // For this scale, multiple runs of NN from different starting points and then
        // 2-opt can be beneficial.
        // Let's try NN + limited 2-opt or a simpler form.
        // A common approach for larger N is to use a fast construction heuristic
        // and then a limited local search.
        // For this implementation, we'll stick with NN + 2-opt but be mindful of its time complexity.
        // A pragmatic approach here would be to run 2-opt for a fixed number of iterations
        // or until no improvement is seen for a certain number of swaps.

        std::vector<int> initial_tour = geometricNearestNeighbor(cities, 0);
        // Limited 2-opt: Run 2-opt with a fixed number of iterations or a time limit.
        // Here, let's opt for a faster 2-opt that stops after a certain number of improvements
        // or a fixed iteration count if no improvement is found.

        int num_iterations = std::min(n * 5, 50000); // Limit iterations to avoid excessive runtime
        std::vector<int> best_tour = initial_tour;
        double best_distance = 0.0;

        for (int i = 0; i < n - 1; ++i) {
            best_distance += distance(cities[initial_tour[i]], cities[initial_tour[i + 1]]);
        }
        best_distance += distance(cities[initial_tour.back()], cities[initial_tour.front()]);

        for (int iter = 0; iter < num_iterations; ++iter) {
            bool improved = false;
            for (int i = 0; i < n - 1 && !improved; ++i) {
                for (int j = i + 1; j < n && !improved; ++j) {
                    std::vector<int> current_tour = best_tour;
                    std::reverse(current_tour.begin() + i, current_tour.begin() + j + 1);

                    double current_distance = 0.0;
                    for (int k = 0; k < n - 1; ++k) {
                        current_distance += distance(cities[current_tour[k]], cities[current_tour[k + 1]]);
                    }
                    current_distance += distance(cities[current_tour.back()], cities[current_tour.front()]);

                    if (current_distance < best_distance) {
                        best_tour = current_tour;
                        best_distance = current_distance;
                        improved = true;
                    }
                }
            }
            if (!improved) break; // Stop if no improvement in an iteration
        }
        return best_tour;

    } else if (n <= 5000000) {
        // Extreme scale: 2-opt is definitely too slow. We need very fast approximations.
        // Algorithms like Ant Colony Optimization (ACO), Genetic Algorithms (GA), or Simulated Annealing (SA)
        // can be used, but they are complex to implement for optimal performance and tuning.
        // For a single-pass solver that must work for "ANY number of cities," we need something very efficient.
        // A very fast heuristic like "Greedy Randomized Adaptive Search Procedure" (GRASP) or
        // a simplified version of nearest neighbor with some randomization, or a greedy approach on clusters.

        // Given the constraint of 300 seconds for 100,000,000 cities,
        // we must assume that the provided machine has significant parallel processing power.
        // However, the prompt also asks for a single algorithm that works across scales.
        // For this scale, a simple Nearest Neighbor followed by a very limited local search or
        // a multi-start Nearest Neighbor would be more appropriate than a full 2-opt.
        // Or, we can use a simpler greedy approach or a divide-and-conquer strategy with heuristics.

        // Let's try a multi-start Nearest Neighbor with a limited number of starts.
        // The number of starts should be adaptively chosen.
        std::mt19937 rng(std::chrono::steady_clock::now().time_since_epoch().count());
        int num_starts = std::min(n / 100, 50); // Number of random starts

        std::vector<int> best_tour_overall;
        double min_tour_dist = -1.0;

        for (int start_idx = 0; start_idx < num_starts; ++start_idx) {
            int current_start_node = (start_idx == 0) ? 0 : std::uniform_int_distribution<int>(0, n - 1)(rng);
            std::vector<int> current_tour = nearestNeighbor(cities, current_start_node);

            // Optional: A very light 2-opt or 3-opt on the current tour
            // For this scale, we might skip extensive local search to meet time limits.
            // Or a limited local search for a fixed number of swaps.

            double current_tour_dist = 0.0;
            for (int i = 0; i < n - 1; ++i) {
                current_tour_dist += distance(cities[current_tour[i]], cities[current_tour[i + 1]]);
            }
            current_tour_dist += distance(cities[current_tour.back()], cities[current_tour.front()]);

            if (min_tour_dist < 0 || current_tour_dist < min_tour_dist) {
                min_tour_dist = current_tour_dist;
                best_tour_overall = current_tour;
            }
        }

        // Ensure the tour starts with city 0. If not, rotate it.
        auto it = std::find(best_tour_overall.begin(), best_tour_overall.end(), 0);
        if (it != best_tour_overall.begin()) {
            std::rotate(best_tour_overall.begin(), it, best_tour_overall.end());
        }
        return best_tour_overall;

    } else { // N > 5000000 (Epic scale)
        // Extremely large N requires very fast approximations.
        // Nearest Neighbor is still a viable base. We can parallelize parts of it
        // or use a very fast clustering approach followed by TSP on clusters.
        // For a single thread, a highly optimized Nearest Neighbor variant is key.
        // The problem implies a single algorithm that *adapts*.
        // For such scale, pure NN is likely the best bet for speed.
        // The time complexity of NN is O(N^2) which is too slow.
        // We need a faster construction.

        // A faster approach for very large N could be a greedy insertion heuristic
        // or a k-d tree based nearest neighbor search to speed up finding nearest neighbors.
        // However, k-d tree construction itself takes time.

        // Given the extreme time constraints (300s for 100M cities implies O(N) or O(N log N) dominant part),
        // a pure O(N^2) algorithm like standard NN is not feasible.
        //
        // A possible O(N log N) or O(N) approach for construction:
        // 1. Sort cities by x-coordinate.
        // 2. Iterate through sorted cities, greedily inserting into tour.
        // This is similar to a greedy sweep line algorithm.
        //
        // Let's implement a simple greedy insertion into the current tour.
        // Start with city 0. Then, for each unvisited city, find the best insertion point.
        // This is still O(N^2) if done naively.

        // A pragmatic solution for extreme scale given single-threaded, limited time:
        // A randomized greedy approach or a highly optimized geometric NN.
        // Let's try a modified Nearest Neighbor that tries to be smarter.

        // For N = 10,000,000+, we must rely on extremely fast heuristics.
        // A simple Nearest Neighbor would be O(N^2).
        // To achieve O(N log N) or O(N), we need a different construction.

        // Option 1: Geometric Sweep - Sort cities by x, then greedily connect.
        // This will not guarantee a good tour but is fast.

        // Option 2: Fast Random Insertion - Pick a random city, add it to the tour
        // by inserting it between the two closest cities in the current tour.
        // This is O(N^2) if inserting into a list, but can be O(N log N) if using a data structure.

        // For this problem, the "single algorithm that works efficiently across ALL scales"
        // implies an adaptive strategy. For extreme scales, we need something that scales better
        // than O(N^2).

        // Let's use a randomized Nearest Neighbor with a limited number of candidates for 'nearest'.
        // This might not be strictly O(N), but could be faster in practice.
        // Or, focus on a geometric heuristic.

        // A fast strategy for extremely large N:
        // 1. Sort cities by x-coordinate.
        // 2. Iterate through cities. For each city, find its nearest neighbor among *already visited* cities.
        // This is still O(N^2) if done naively.

        // To beat O(N^2), we must avoid iterating through all possible neighbors every time.
        // Using a spatial data structure like a k-d tree for N up to 5M would be ideal,
        // but building it takes O(N log N) and querying O(log N). Total O(N log N).
        // However, the problem implies a single algorithm without heavy pre-processing for large N.

        // Given the constraints and the need for a single adaptable algorithm,
        // let's consider a highly optimized greedy approach.
        //
        // A common fast approximation is to divide the space into grids.
        // But that requires clustering.

        // Let's assume for N > 5M, the most critical factor is speed.
        // We'll use a randomized Nearest Neighbor. The key is to limit the search for the nearest neighbor.
        // Instead of checking all N-1 cities, check a random subset. This is heuristic.

        std::mt19937 rng(std::chrono::steady_clock::now().time_since_epoch().count());
        std::vector<int> tour;
        tour.reserve(n);
        std::vector<bool> visited(n, false);
        int current_city_idx = 0; // Start at city 0

        tour.push_back(cities[0].id);
        visited[0] = true;

        // Limit the number of cities to check for nearest neighbor
        // This is a trade-off: fewer checks = faster but potentially worse quality.
        // The number of checks should also scale with N but not too aggressively.
        int search_limit = std::max(100, n / 1000); // Heuristic search limit

        for (int i = 1; i < n; ++i) {
            double min_dist = -1.0;
            int nearest_city_idx = -1;

            // Collect a subset of unvisited cities to check
            std::vector<int> candidate_indices;
            for (int k = 0; k < n; ++k) {
                if (!visited[k]) {
                    candidate_indices.push_back(k);
                }
            }
            // Shuffle candidates to ensure randomness
            std::shuffle(candidate_indices.begin(), candidate_indices.end(), rng);

            int checks_done = 0;
            for (int cand_idx : candidate_indices) {
                if (checks_done >= search_limit && candidate_indices.size() > search_limit) break;

                double d = distance(cities[current_city_idx], cities[cand_idx]);
                if (nearest_city_idx == -1 || d < min_dist) {
                    min_dist = d;
                    nearest_city_idx = cand_idx;
                }
                checks_done++;
            }

            if (nearest_city_idx != -1) {
                tour.push_back(cities[nearest_city_idx].id);
                visited[nearest_city_idx] = true;
                current_city_idx = nearest_city_idx;
            } else {
                // Should not happen if n > 1 and there are unvisited cities,
                // but as a fallback, find any unvisited city.
                for (int k = 0; k < n; ++k) {
                    if (!visited[k]) {
                        tour.push_back(cities[k].id);
                        visited[k] = true;
                        current_city_idx = k;
                        break;
                    }
                }
            }
        }
        return tour;
    }
}
