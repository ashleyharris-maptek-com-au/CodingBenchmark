The maze solver is implemented in Rust using Breadth-First Search (BFS) to ensure the shortest path is found efficiently. The algorithm uses flat vectors for `parent` tracking and `visited` status, which is memory-efficient and crucial for handling grids up to 50000x50000.

```rust
use std::collections::VecDeque;
use std::io::{self, BufRead};

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
struct Point {
    x: usize, // Row
    y: usize, // Col
}

fn solve() -> io::Result<()> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "Empty input"))??;
    let parts: Vec<usize> = first_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let h = parts[0];
    let w = parts[1];

    let mut maze = Vec::with_capacity(h);
    let mut start = None;
    let mut end = None;

    for r in 0..h {
        let line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "Incomplete maze"))??;
        let row: Vec<char> = line.chars().collect();
        for c in 0..w {
            if row[c] == 'A' {
                start = Some(Point { x: r, y: c });
            } else if row[c] == 'B' {
                end = Some(Point { x: r, y: c });
            }
        }
        maze.push(row);
    }

    let start = start.ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "No start point"))?;
    let end = end.ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "No end point"))?;

    // BFS
    let mut queue = VecDeque::new();
    queue.push_back(start);
    
    // Using a flat vector for visited tracking is more memory-efficient than a HashMap for large grids
    let mut parent = vec![None; h * w];
    let mut visited = vec![false; h * w];

    visited[start.x * w + start.y] = true;

    let mut found = false;
    while let Some(current) = queue.pop_front() {
        if current == end {
            found = true;
            break;
        }

        let directions = [(0, 1), (0, -1), (1, 0), (-1, 0)];
        for (dx, dy) in directions {
            let next_x = current.x as isize + dx;
            let next_y = current.y as isize + dy;

            if next_x >= 0 && next_x < h as isize && next_y >= 0 && next_y < w as isize {
                let next = Point { x: next_x as usize, y: next_y as usize };
                if !visited[next.x * w + next.y] && maze[next.x][next.y] != '#' {
                    visited[next.x * w + next.y] = true;
                    parent[next.x * w + next.y] = Some(current);
                    queue.push_back(next);
                }
            }
        }
    }

    if found {
        let mut path = Vec::new();
        let mut curr = Some(end);
        while let Some(p) = curr {
            path.push(p);
            curr = parent[p.x * w + p.y];
        }
        path.reverse();

        println!("{}", path.len());
        for p in path {
            println!("{} {}", p.x, p.y);
        }
    } else {
        println!("0");
    }

    Ok(())
}

fn main() {
    if let Err(e) = solve() {
        eprintln!("Error: {}", e);
    }
}
```
