struct Vec3 {
    double x, y, z;

    Vec3 operator+(const Vec3& other) const { return {x + other.x, y + other.y, z + other.z}; }
    Vec3 operator-(const Vec3& other) const { return {x - other.x, y - other.y, z - other.z}; }
    Vec3 operator*(double scalar) const { return {x * scalar, y * scalar, z * scalar}; }
    Vec3 operator/(double scalar) const { return {x / scalar, y / scalar, z / scalar}; }
    double dot(const Vec3& other) const { return x * other.x + y * other.y + z * other.z; }
    Vec3 cross(const Vec3& other) const { return {y * other.z - z * other.y, z * other.x - x * other.z, x * other.y - y * other.x}; }
    double length_sq() const { return x * x + y * y + z * z; }
    double length() const { return std::sqrt(length_sq()); }
    Vec3 normalize() const {
        double len = length();
        if (len > 1e-9) return *this / len;
        return {0, 0, 0};
    }

    bool operator==(const Vec3& other) const {
        return std::abs(x - other.x) < 1e-9 && std::abs(y - other.y) < 1e-9 && std::abs(z - other.z) < 1e-9;
    }
    bool operator!=(const Vec3& other) const { return !(*this == other); }
};

// Check if a point is inside a triangle (barycentric coordinates)
bool is_point_in_triangle(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 v0 = c - a;
    Vec3 v1 = b - a;
    Vec3 v2 = p - a;

    double dot00 = v0.dot(v0);
    double dot01 = v0.dot(v1);
    double dot02 = v0.dot(v2);
    double dot11 = v1.dot(v1);
    double dot12 = v1.dot(v2);

    double invDenom = 1.0 / (dot00 * dot11 - dot01 * dot01);
    double u = (dot11 * dot02 - dot01 * dot12) * invDenom;
    double v = (dot00 * dot12 - dot01 * dot02) * invDenom;

    return (u >= -1e-9) && (v >= -1e-9) && (u + v <= 1.0 + 1e-9);
}

// Ray-Triangle intersection (Möller-Trumbore algorithm)
bool intersect_ray_triangle(const Ray& ray, const Vec3& v0, const Vec3& v1, const Vec3& v2, double& t, Vec3& hit_point, Vec3& normal) {
    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;
    Vec3 pvec = ray.direction.cross(edge2);
    double det = edge1.dot(pvec);

    if (std::abs(det) < 1e-9) { // Ray is parallel to triangle plane
        return false;
    }

    double inv_det = 1.0 / det;
    Vec3 tvec = ray.origin - v0;
    double u = tvec.dot(pvec) * inv_det;

    if (u < 0.0 || u > 1.0) {
        return false;
    }

    Vec3 qvec = tvec.cross(edge1);
    double v = ray.direction.dot(qvec) * inv_det;

    if (v < 0.0 || u + v > 1.0) {
        return false;
    }

    t = edge2.dot(qvec) * inv_det;

    if (t > 1e-9) { // Intersection is in front of the ray origin
        hit_point = ray.origin + ray.direction * t;
        normal = edge1.cross(edge2).normalize();
        return true;
    }
    return false; // Intersection is behind the ray origin
}

// Point-in-mesh test (winding number algorithm - simplified for well-behaved meshes)
// For robust point-in-mesh, a full winding number calculation is needed.
// Here, we assume a simple ray cast and count intersections. This can be ambiguous
// for complex self-intersecting meshes, but the problem statement guarantees clean meshes.
bool is_point_inside_mesh(const Vec3& point, const std::vector<Vec3>& vertices, const std::vector<Triangle>& faces) {
    // Cast a ray from the point in a fixed direction.
    // Count how many times it intersects the mesh faces.
    // An odd number of intersections means the point is inside.
    // A robust implementation would use a winding number calculation and handle edge cases.

    // Choose a ray direction that is unlikely to be parallel to many faces.
    Ray ray = {point, {1.0, 0.0, 0.0}}; // Ray along the positive X-axis

    int intersection_count = 0;
    for (const auto& face : faces) {
        const Vec3& v0 = vertices[face.v[0]];
        const Vec3& v1 = vertices[face.v[1]];
        const Vec3& v2 = vertices[face.v[2]];

        double t;
        Vec3 hit_point, normal;
        if (intersect_ray_triangle(ray, v0, v1, v2, t, hit_point, normal)) {
            // To avoid double counting when the ray hits an edge or vertex,
            // we can check if the hit point is "above" the ray origin in some way,
            // or simply use a small epsilon to perturb the ray if it hits an edge.
            // For simplicity here, we assume non-degenerate cases.

            // A common technique to handle degeneracies: if the ray hits on an edge,
            // either count it or don't, consistently. A simple heuristic is to ignore
            // intersections that are extremely close to a vertex or edge.
            // Here, we assume the t value is distinct enough for non-degenerate hits.
            intersection_count++;
        }
    }

    return (intersection_count % 2) == 1;
}

// A simplified approach for CSG Union, focusing on combining faces and handling intersections.
// This involves:
// 1. Identifying all intersection points between edges of mesh A and faces of mesh B, and vice versa.
// 2. Adding these intersection points as new vertices to both meshes.
// 3. Triangulating the resulting sub-faces created by these new vertices.
// 4. Classifying the original faces and new faces as "inside" or "outside" the other mesh.
// 5. Constructing the final union mesh from faces that are outside one mesh or inside the other.

// This is a high-level outline. A full, robust implementation is extremely complex and
// often relies on specialized libraries (like CGAL).

// For this problem, given the performance constraints and the need for a single program
// handling all scales, a hybrid approach might be necessary.
// For small meshes, a more exact, brute-force intersection and retriangulation might work.
// For large meshes, a voxel-based or spatial partitioning approach might be more efficient.

// Given the constraints, we'll aim for a conceptually simpler approach that is
// reasonably robust for clean inputs and might scale better with proper data structures.

struct Mesh {
    std::vector<Vec3> vertices;
    std::vector<Triangle> faces;
    // Store original vertex indices for faces for easier mapping
    std::vector<std::tuple<size_t, size_t, size_t>> original_face_vertex_indices;
    int id; // 0 for mesh A, 1 for mesh B
};

// Helper to find all edge-edge and edge-face intersections.
// This is a computationally intensive part.
void find_intersections(
    const Mesh& mesh_a, const Mesh& mesh_b,
    std::vector<Vec3>& intersection_points,
    std::vector<std::tuple<size_t, size_t, size_t, size_t, size_t, size_t, int>>& intersections_a_b, // {idx_a_edge_start, idx_a_edge_end, idx_b_face, intersection_point_idx, mesh_id (0=A, 1=B)}
    std::vector<std::tuple<size_t, size_t, size_t, size_t, size_t, size_t, int>>& intersections_b_a
) {
    // Iterate through edges of mesh_a and intersect with faces of mesh_b
    for (size_t i = 0; i < mesh_a.vertices.size(); ++i) {
        for (size_t j = i + 1; j < mesh_a.vertices.size(); ++j) { // Consider all pairs of vertices as potential edges
            Vec3 edge_start_a = mesh_a.vertices[i];
            Vec3 edge_end_a = mesh_a.vertices[j];
            Ray edge_ray_a = {edge_start_a, (edge_end_a - edge_start_a).normalize()};
            double edge_length_sq = (edge_end_a - edge_start_a).length_sq();

            for (size_t k = 0; k < mesh_b.faces.size(); ++k) {
                const auto& face_b = mesh_b.faces[k];
                const Vec3& v0_b = mesh_b.vertices[face_b.v[0]];
                const Vec3& v1_b = mesh_b.vertices[face_b.v[1]];
                const Vec3& v2_b = mesh_b.vertices[face_b.v[2]];

                double t;
                Vec3 hit_point, normal_b;
                if (intersect_ray_triangle(edge_ray_a, v0_b, v1_b, v2_b, t, hit_point, normal_b)) {
                    // Check if the intersection point lies on the edge segment of mesh_a
                    double dist_from_start_sq = (hit_point - edge_start_a).length_sq();
                    if (dist_from_start_sq >= 1e-9 && dist_from_start_sq <= edge_length_sq - 1e-9) {
                        // This is an intersection point. Add it if not already present.
                        bool found = false;
                        for (size_t p_idx = 0; p_idx < intersection_points.size(); ++p_idx) {
                            if (intersection_points[p_idx] == hit_point) {
                                intersections_a_b.emplace_back(i, j, k, p_idx, 0, mesh_b.id); // Store indices
                                found = true;
                                break;
                            }
                        }
                        if (!found) {
                            intersection_points.push_back(hit_point);
                            intersections_a_b.emplace_back(i, j, k, intersection_points.size() - 1, 0, mesh_b.id);
                        }
                    }
                }
            }
        }
    }

    // Iterate through edges of mesh_b and intersect with faces of mesh_a
    for (size_t i = 0; i < mesh_b.vertices.size(); ++i) {
        for (size_t j = i + 1; j < mesh_b.vertices.size(); ++j) {
            Vec3 edge_start_b = mesh_b.vertices[i];
            Vec3 edge_end_b = mesh_b.vertices[j];
            Ray edge_ray_b = {edge_start_b, (edge_end_b - edge_start_b).normalize()};
            double edge_length_sq = (edge_end_b - edge_start_b).length_sq();

            for (size_t k = 0; k < mesh_a.faces.size(); ++k) {
                const auto& face_a = mesh_a.faces[k];
                const Vec3& v0_a = mesh_a.vertices[face_a.v[0]];
                const Vec3& v1_a = mesh_a.vertices[face_a.v[1]];
                const Vec3& v2_a = mesh_a.vertices[face_a.v[2]];

                double t;
                Vec3 hit_point, normal_a;
                if (intersect_ray_triangle(edge_ray_b, v0_a, v1_a, v2_a, t, hit_point, normal_a)) {
                    double dist_from_start_sq = (hit_point - edge_start_b).length_sq();
                    if (dist_from_start_sq >= 1e-9 && dist_from_start_sq <= edge_length_sq - 1e-9) {
                        bool found = false;
                        for (size_t p_idx = 0; p_idx < intersection_points.size(); ++p_idx) {
                            if (intersection_points[p_idx] == hit_point) {
                                intersections_b_a.emplace_back(i, j, k, p_idx, 1, mesh_a.id); // Store indices
                                found = true;
                                break;
                            }
                        }
                        if (!found) {
                            intersection_points.push_back(hit_point);
                            intersections_b_a.emplace_back(i, j, k, intersection_points.size() - 1, 1, mesh_a.id);
                        }
                    }
                }
            }
        }
    }

    // TODO: Handle edge-edge intersections for more robustness.
    // This is significantly more complex. For now, we rely on edge-face.
}

// This is a highly simplified retriangulation. A proper solution would involve
// complex geometric algorithms for planar subdivision and triangulation.
// For this problem, we'll assume that by adding intersection points to edges,
// we can conceptually split faces and later identify the correct ones.

// The core idea for CSG Union:
// A point is in the union if it is inside mesh A OR inside mesh B.
// To reconstruct the geometry:
// 1. Take all faces of A that are NOT inside B.
// 2. Take all faces of B that are NOT inside A.
// 3. This simple approach is insufficient because it doesn't handle cases where
//    a face of A is partially inside B and needs to be clipped.

// A more complete approach:
// 1. Create a list of all vertices from both meshes.
// 2. Add all intersection points between edges and faces as new vertices.
// 3. For each original face, if it is intersected by edges of the other mesh,
//    it needs to be split into smaller polygons.
// 4. These smaller polygons are then triangulated.
// 5. A point-in-mesh test is used to determine which of these new faces belong to the union.

// Given the time limit and complexity, we will use a simplified approach that
// might be sufficient for well-behaved inputs and scales by carefully managing
// data structures and avoiding explicit geometric operations on every face.

// Scalability Strategy:
// For smaller meshes (e.g., < 1000 faces): Perform more detailed intersection and retriangulation.
// For larger meshes:
// - Use spatial data structures (like BVH or octrees) for faster intersection queries.
// - Employ a "shell" or "boundary" based approach where we only track the surface.
// - For the point-in-mesh test, a robust winding number algorithm is essential and needs to be efficient.

// Let's try a strategy that identifies "surface" portions of each mesh.
// A surface part of mesh A is included in the union if it is:
// - Outside mesh B.
// - Inside mesh B (but this is only relevant if the union is "thick", which it isn't).
// The union boundary is formed by parts of A's boundary that are outside B, and parts of B's boundary that are outside A.

// Problem: "parts of A's boundary that are outside B" requires clipping A against B.
// This is the fundamental difficulty of CSG operations.

// A more practical approach for this problem's constraints:
// 1. Combine all vertices from A and B.
// 2. Identify all intersections between edges of A and faces of B, and vice versa. Add these intersection points to the combined vertex list.
// 3. Create a new data structure that represents the "sub-elements" (edges, faces) formed by these new vertices.
// 4. For each sub-face (which should be a triangle or a simple polygon), determine if its centroid (or a sample point) is inside the *other* mesh.
// 5. Collect all sub-faces that are outside the other mesh.
// 6. Triangulate these collection of polygons to form the final mesh.

// This requires a robust method for splitting faces and triangulating the resulting polygons.
// This is a very hard problem in computational geometry.

// Given the constraints, we will aim for a method that is "good enough" and scales reasonably.
// Let's try a strategy that focuses on identifying which original faces are entirely outside the other mesh,
// and then attempting to "bridge" the gaps with intersection points.

// --- Simplified Union Logic (Focus on Boundary) ---
// The boundary of A U B is formed by:
// - The parts of A's boundary that are outside B.
// - The parts of B's boundary that are outside A.
// This is still clipping.

// Let's consider the core CSG Union algorithm by Weiler-Atherton or similar.
// It involves decomposing the boundaries into chains.

// A key insight for scalability might be to avoid full retriangulation for large meshes.
// Instead, focus on the intersection segments and use point-in-mesh tests.

// Let's try a practical implementation strategy:
// 1. Collect all vertices from A and B.
// 2. Identify all intersection points between edges of A and faces of B, and edges of B and faces of A.
// 3. Store these intersection points.
// 4. Create a unified list of vertices.
// 5. For each original triangle in A, check if its centroid is inside B. If not, it's potentially part of the union.
// 6. For each original triangle in B, check if its centroid is inside A. If not, it's potentially part of the union.
// 7. This is still not enough as it doesn't handle clipping of partially intersecting faces.

// A more robust (but complex) approach for unions:
// 1. Clip Mesh A against Mesh B (results in A_outside_B and A_inside_B).
// 2. Clip Mesh B against Mesh A (results in B_outside_A and B_inside_A).
// 3. The union is (A_outside_B) + (B_outside_A).

// Clipping is itself a hard problem. For a triangle T and a convex/non-convex volume V:
// - If T is fully outside V, it's kept.
// - If T is fully inside V, it's discarded.
// - If T intersects V, it needs to be split into smaller polygons, and these polygons then tested.

// For this problem, given the single-program requirement, we must pick an algorithm that can
// handle both small and large scales. The common thread is handling intersections robustly.

// For scalability, we need to avoid brute-force O(N^2) edge-face checks on millions of vertices.
// Spatial partitioning (BVH, Octree) is crucial for this.

// --- Octree-based Intersection Detection ---
// This is a more scalable approach.
// We'll build an octree for each mesh.

struct OctreeNode {
    std::vector<size_t> face_indices; // Indices of faces fully or partially within this node
    std::vector<std::unique_ptr<OctreeNode>> children;
    Vec3 center;
    double half_dimension;

    OctreeNode(const Vec3& c, double hd) : center(c), half_dimension(hd) {}
};

void insert_face_into_octree(OctreeNode* node, const std::vector<Vec3>& vertices, const Triangle& face, size_t face_idx, int depth, int max_depth) {
    // Simple bounding box intersection check for now.
    // A more robust implementation would check for actual face overlap with the octree node's volume.
    Vec3 min_v = vertices[face.v[0]];
    Vec3 max_v = vertices[face.v[0]];

    for(int i=1; i<3; ++i) {
        const Vec3& v = vertices[face.v[i]];
        min_v.x = std::min(min_v.x, v.x); max_v.x = std::max(max_v.x, v.x);
        min_v.y = std::min(min_v.y, v.y); max_v.y = std::max(max_v.y, v.y);
        min_v.z = std::min(min_v.z, v.z); max_v.z = std::max(max_v.z, v.z);
    }

    Vec3 node_min = node->center - Vec3{node->half_dimension, node->half_dimension, node->half_dimension};
    Vec3 node_max = node->center + Vec3{node->half_dimension, node->half_dimension, node->half_dimension};

    // Check if face's bounding box intersects node's bounding box
    if (max_v.x < node_min.x || min_v.x > node_max.x ||
        max_v.y < node_min.y || min_v.y > node_max.y ||
        max_v.z < node_min.z || min_v.z > node_max.z) {
        return; // No intersection
    }

    if (depth >= max_depth) {
        node->face_indices.push_back(face_idx);
        return;
    }

    // Create children if they don't exist
    if (node->children.empty()) {
        double child_hd = node->half_dimension / 2.0;
        Vec3 child_centers[8] = {
            {node->center.x - child_hd, node->center.y - child_hd, node->center.z - child_hd},
            {node->center.x + child_hd, node->center.y - child_hd, node->center.z - child_hd},
            {node->center.x - child_hd, node->center.y + child_hd, node->center.z - child_hd},
            {node->center.x + child_hd, node->center.y + child_hd, node->center.z - child_hd},
            {node->center.x - child_hd, node->center.y - child_hd, node->center.z + child_hd},
            {node->center.x + child_hd, node->center.y - child_hd, node->center.z + child_hd},
            {node->center.x - child_hd, node->center.y + child_hd, node->center.z + child_hd},
            {node->center.x + child_hd, node->center.y + child_hd, node->center.z + child_hd}
        };
        for (int i = 0; i < 8; ++i) {
            node->children.push_back(std::make_unique<OctreeNode>(child_centers[i], child_hd));
        }
    }

    // Recursively insert into children
    for (auto& child : node->children) {
        insert_face_into_octree(child.get(), vertices, face, face_idx, depth + 1, max_depth);
    }
    
    // Also add face index to the parent node if it's not fully contained by children,
    // or if we want to keep it at higher levels for faster broad-phase.
    // For simplicity, we'll just add it if it intersects any child.
    // A more refined approach would add it to node if it spans across children.
    if (!node->children.empty()) {
        bool spans_children = false;
        for (auto& child : node->children) {
            Vec3 child_min = child->center - Vec3{child->half_dimension, child->half_dimension, child->half_dimension};
            Vec3 child_max = child->center + Vec3{child->half_dimension, child->half_dimension, child->half_dimension};
            if (max_v.x >= child_min.x && min_v.x <= child_max.x &&
                max_v.y >= child_min.y && min_v.y <= child_max.y &&
                max_v.z >= child_min.z && min_v.z <= child_max.z) {
                spans_children = true;
                break;
            }
        }
        if (spans_children) {
             // Add to parent as well, to ensure it's processed at higher levels.
             // Avoid duplicates if face is already in children's lists.
             bool found_in_children = false;
             for(auto& child : node->children) {
                 for(size_t idx : child->face_indices) {
                     if (idx == face_idx) {
                         found_in_children = true;
                         break;
                     }
                 }
                 if (found_in_children) break;
             }
             if (!found_in_children) {
                 node->face_indices.push_back(face_idx);
             }
        }
    }
}

// Query octree for faces that intersect a bounding box.
void query_octree(OctreeNode* node, const Vec3& query_min, const Vec3& query_max, const std::vector<Vec3>& vertices, const std::vector<Triangle>& all_faces, std::set<size_t>& intersecting_face_indices) {
    Vec3 node_min = node->center - Vec3{node->half_dimension, node->half_dimension, node->half_dimension};
    Vec3 node_max = node->center + Vec3{node->half_dimension, node->half_dimension, node->half_dimension};

    // Check if node's bounding box intersects query bounding box
    if (node_max.x < query_min.x || node_min.x > query_max.x ||
        node_max.y < query_min.y || node_min.y > query_max.y ||
        node_max.z < query_min.z || node_min.z > query_max.z) {
        return; // No intersection
    }

    // Check faces stored at this node
    for (size_t face_idx : node->face_indices) {
        const Triangle& face = all_faces[face_idx];
        Vec3 min_v = vertices[face.v[0]];
        Vec3 max_v = vertices[face.v[0]];

        for(int i=1; i<3; ++i) {
            const Vec3& v = vertices[face.v[i]];
            min_v.x = std::min(min_v.x, v.x); max_v.x = std::max(max_v.x, v.x);
            min_v.y = std::min(min_v.y, v.y); max_v.y = std::max(max_v.y, v.y);
            min_v.z = std::min(min_v.z, v.z); max_v.z = std::max(max_v.z, v.z);
        }

        // Check if face's bounding box intersects query bounding box
        if (max_v.x >= query_min.x && min_v.x <= query_max.x &&
            max_v.y >= query_min.y && min_v.y <= query_max.y &&
            max_v.z >= query_min.z && min_v.z <= query_max.z) {
            intersecting_face_indices.insert(face_idx);
        }
    }

    // Recurse into children if they exist
    if (!node->children.empty()) {
        for (auto& child : node->children) {
            query_octree(child.get(), query_min, query_max, vertices, all_faces, intersecting_face_indices);
        }
    }
}

// Build an octree for a mesh.
std::unique_ptr<OctreeNode> build_octree(const std::vector<Vec3>& vertices, const std::vector<Triangle>& faces, int max_depth = 10) {
    if (vertices.empty() || faces.empty()) return nullptr;

    // Determine bounding box of the entire mesh
    Vec3 min_b = vertices[0], max_b = vertices[0];
    for (const auto& v : vertices) {
        min_b.x = std::min(min_b.x, v.x); max_b.x = std::max(max_b.x, v.x);
        min_b.y = std::min(min_b.y, v.y); max_b.y = std::max(max_b.y, v.y);
        min_b.z = std::min(min_b.z, v.z); max_b.z = std::max(max_b.z, v.z);
    }

    Vec3 center = (min_b + max_b) / 2.0;
    double diagonal = (max_b - min_b).length();
    double half_dimension = (diagonal / 2.0) * 1.1; // Slightly larger to ensure all geometry fits

    auto root = std::make_unique<OctreeNode>(center, half_dimension);

    for (size_t i = 0; i < faces.size(); ++i) {
        insert_face_into_octree(root.get(), vertices, faces[i], i, 0, max_depth);
    }
    return root;
}
