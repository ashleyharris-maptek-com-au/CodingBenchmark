using System;

public class Voxel
{
    public int X { get; set; }
    public int Y { get; set; }
    public int Z { get; set; }
    public int Mass { get; set; }
    public int ValuePerTonne { get; set; }
    public bool IsOre => ValuePerTonne > 0;
    public bool IsDirt => ValuePerTonne == 0;
    public long TotalValue => (long)Mass * ValuePerTonne * 1000; // Simplified for initial consideration

    public Voxel(int x, int y, int z, int mass, int valuePerTonne)
    {
        X = x;
        Y = y;
        Z = z;
        Mass = mass;
        ValuePerTonne = valuePerTonne;
    }
}

    }

    private void FindAndPlanMoveToDumbLocation(Voxel voxel)

                    {

                        if (dirtVoxel.Z > oreVoxel.Z && Math.Abs(dirtVoxel.X - oreVoxel.X) <= 1 && Math.Abs(dirtVoxel.Y - oreVoxel.Y) <= 1)

                            // Re-evaluate ore path after moving dirt

                                FindAndPlanMoveToPlant(oreVoxel);

    // Checks if a voxel is currently "exposed" and can be dug.
    // An exposed voxel is one that has air (or no voxel) above it OR a side voxel that is not occupied by solid ground.
    private bool IsDuggable(int x, int y, int z)
    {
        if (x < 0 || x >= XVoxelCount || y < 0 || y >= YVoxelCount || z < 0 || z >= ZVoxelCount)
        {
            return false; // Out of bounds
        }

        // If the voxel itself is empty, it's not diggable in the sense of "moving" it.
        if (voxelGrid[x, y, z] == null || voxelGrid[x, y, z].Mass == 0)
        {
            return false;
        }

        // Check if there's empty space above
        if (z + 1 < ZVoxelCount && voxelGrid[x, y, z + 1] == null && !occupiedGrid[x, y, z + 1])
        {
            return true;
        }

        // Check if any adjacent voxel (in X or Y) is empty (air).
        // This is a simplification. The "earth" connection is more nuanced.
        // A voxel is diggable if it's exposed from a side.
        // If a voxel is not at the bottom of a column, and it has an empty space next to it (in X or Y).

        int[] dx = { 0, 0, 1, -1 };
        int[] dy = { 1, -1, 0, 0 };

        for (int i = 0; i < 4; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];

            if (nx >= 0 && nx < XVoxelCount && ny >= 0 && ny < YVoxelCount)
            {
                // If the adjacent voxel is empty (air or not occupied by dumped dirt)
                if (voxelGrid[nx, ny, z] == null && !occupiedGrid[nx, ny, z])
                {
                    return true;
                }
            }
            else // If adjacent is out of bounds, consider it exposed.
            {
                return true;
            }
        }

        // Additional check for being connected to the "ground" if it's at the bottom layer.
        if (z == 0) return true; // Voxels at the very bottom are considered part of the stable ground.

        // If a voxel is not at the bottom and has no air above and no air to the sides,
        // but is *connected* to the ground below, it should be diggable.
        // This constraint is "If you disconnect a voxel(s), it will collapse."
        // This means we can dig under something IF it's still connected to the earth.
        // The prompt says "you can dig under any voxel without fear of it collapsing so long as it's connected to the earth."
        // This implies that if a voxel at (x,y,z) has a voxel below it at (x,y,z-1) that is either original ground or dumped dirt,
        // and that voxel at (x,y,z-1) is ultimately connected to the earth (z=0), then (x,y,z) can be dug from below.

        // Let's redefine diggability based on the constraints:
        // 1. Exposed from above (air above).
        // 2. Exposed from a side (air adjacent in X/Y).
        // 3. Can dig under if connected to earth (meaning there's a voxel directly below it).
        //    This implies we need to check if z > 0 and voxelGrid[x,y,z-1] or occupiedGrid[x,y,z-1] is true.

        if (z > 0) // If not at the bottom, check if there's something below it that can support it
        {
            bool isSupportedFromBelow = (voxelGrid[x, y, z - 1] != null && voxelGrid[x, y, z - 1].Mass > 0) || occupiedGrid[x, y, z - 1];
            if (isSupportedFromBelow)
            {
                // If it's supported from below, it's diggable, provided it doesn't cause collapse.
                // The collapse rule "If you disconnect a voxel(s), it will collapse" implies that a voxel *must* have
                // at least one neighbor in X, Y, or Z direction that is "solid" (original ground or dumped dirt).
                // This becomes a connectivity problem.
                // For simplicity here, we'll assume diggable if there's air above OR air to the sides.
                // The "dig under" rule is implicitly handled by the pathfinding, which will check for valid Z transitions.

                // Re-evaluating the prompt: "Can only dig dirt that is exposed from above or a side."
                // This simplifies it. The "dig under" rule might be more about how the digging process *happens* rather than *what* can be dug.
                // So, we stick to "exposed from above or a side."
            }
        }

        return false; // If none of the above conditions are met, it's not diggable.
    }

    // Finds the next available Z level for dumping at a given X, Y.
    private int GetNextDumpZ(int x, int y)
    {
        int z = 0;
        while (z < ZVoxelCount && (voxelGrid[x, y, z] != null || occupiedGrid[x, y, z]))
        {
            z++;
        }
        return z;
    }

    // Finds a suitable location to dump dirt.
    private (int x, int y) FindDumpingLocation(Voxel voxelToDump)
    {
        // Prioritize dumping further away from the processing plant and high-value ore.
        // Look for the highest available spot on the edges of the map, or in areas we've already cleared.

        // Iterate through potential dump locations, starting from the edges and moving inwards.
        // We need to find a spot where the Z is not excessively high and it's not blocking anything crucial.

        // Simple strategy: find the highest unoccupied spot within a reasonable radius.
        // Consider spots that are already "filled" with other dumped dirt.

        int bestDumpX = -1, bestDumpY = -1;
        int maxDumpZ = -1;

        // Try to dump on top of existing dumped dirt or original ground.
        for (int x = 0; x < XVoxelCount; x++)
        {
            for (int y = 0; y < YVoxelCount; y++)
            {
                // Avoid dumping directly at the processing plant unless absolutely necessary.
                if (x == PROCESSING_PLANT_X && y == PROCESSING_PLANT_Y) continue;

                int currentDumpZ = GetNextDumpZ(x, y);
                if (currentDumpZ < ZVoxelCount)
                {
                    // Heuristic: Prefer dumping higher up to get it out of the way, but not too high.
                    // And prefer dumping further from the plant.
                    double distanceToPlant = Math.Sqrt(Math.Pow(x - PROCESSING_PLANT_X, 2) + Math.Pow(y - PROCESSING_PLANT_Y, 2));

                    // Simple criteria: if this is the first valid spot, or if it's higher and further.
                    if (bestDumpX == -1 ||
                        (currentDumpZ > maxDumpZ) ||
                        (currentDumpZ == maxDumpZ && distanceToPlant > Math.Sqrt(Math.Pow(bestDumpX - PROCESSING_PLANT_X, 2) + Math.Pow(bestDumpY - PROCESSING_PLANT_Y, 2))))
                    {
                        bestDumpX = x;
                        bestDumpY = y;
                        maxDumpZ = currentDumpZ;
                    }
                }
            }
        }

        return (bestDumpX, bestDumpY);
    }

    // Checks if a voxel is directly reachable by a valid digging path to the processing plant.
    private bool CanReachProcessingPlant(Voxel voxel)
    {

        // For now, we check if the voxel is "exposed" and if the plant is "reachable" from it.

        // The plant is at (0,0). We need to be able to "walk" from voxel.X, voxel.Y to 0,0.

        // Let's try to get a direct path.
        var path = GetDigPathToTarget(voxel.X, voxel.Y, voxel.Z, PROCESSING_PLANT_X, PROCESSING_PLANT_Y);
        return path != null;
    }

    // Generates the path for digging a voxel from its current location to a target X, Y, Z.
    // Returns a list of (x, y, z) tuples representing the path.
    // The path is from the voxel's current location backwards to the target.
    private List<(int x, int y, int z)> GetDigPathToTarget(int startX, int startY, int startZ, int targetX, int targetY)
    {
        List<(int x, int y, int z)> path = new List<(int x, int y, int z)>();
        int currentX = startX;
        int currentY = startY;
        int currentZ = startZ;

        // Build path by moving towards target X and Y, adjusting Z according to slope.
        while (currentX != targetX || currentY != targetY)
        {
            // Decide which direction to move (X or Y). Prioritize the larger difference.
            int dx = Math.Sign(targetX - currentX);
            int dy = Math.Sign(targetY - currentY);

            // Calculate potential next Z based on slope constraints.
            // Max Z increase: 1 per X/Y step.
            // Max Z decrease: Unlimited (free).

            int potentialNextX = currentX + dx;
            int potentialNextY = currentY + dy;

            // Check if moving in X first
            if (dx != 0 && (Math.Abs(targetX - currentX) >= Math.Abs(targetY - currentY) || dy == 0))
            {

                // We want to reach a Z that is at most currentZ + 1 for an uphill step.
                // Or currentZ for flat, or currentZ - slope_factor for downhill.

                // Try to maintain or decrease Z, or go up by at most 1 if necessary.
                int targetZForStep = currentZ; // Default to staying level

                // For simplicity, let's assume the target Z for the *entire path* is the destination Z.
                // This is not strictly correct for intermediate steps.

                // Let's try to find a Z that allows us to reach the target Y with minimal Z change.
                // A better approach would be to know the Z of the target at the destination X,Y.
                // Since we are moving to the *processing plant*, the target Z is effectively 0.

                // We need to ensure that at *each step*, the Z difference relative to the X/Y difference is valid.
                // Z_difference / (X_step + Y_step) <= 1.

                // If moving from (currentX, currentY, currentZ) to (potentialNextX, currentY, nextZ)
                // The X/Y step is 1. The Z step is (nextZ - currentZ).
                // So, (nextZ - currentZ) / 1 <= 1  => nextZ <= currentZ + 1

                        // The constraint is on the difference in Z coordinate between the *current voxel* and the *target voxel*

                        // relative to the X/Y step we are about to take, towards the target.

                        // So, if currentZ > 0, we can go downhill.
                        // If currentZ <= 0, we can go uphill by at most 1.
                        // Since voxel Z is always >= 0, this means we can go up to currentZ + 1 if needed.

                            // For now, let's assume we can move to `bestNextZForX` if it's valid.
                            // The critical check is if `IsDuggable(currentX, currentY, currentZ)` was true.
                            // The path itself should consist of diggable voxels.

                            // We are constructing the path of the *voxel we are moving*.
                            // So, `currentX, currentY, currentZ` is the location *from*.
                            // We are moving *to* `potentialNextX, currentY, nextZ`.

                            // We need to ensure that `(potentialNextX, currentY, nextZ)` is a valid place to move to.
                            // It should be air or occupied by dumped dirt.

                            // Let's assume `bestNextZForX` is a valid Z.
                            nextZ = bestNextZForX;
                            path.Add((currentX, currentY, currentZ)); // Add current location to the path
                            currentX = potentialNextX;
                            currentZ = nextZ;
                        }
                        else if (dy != 0) // Move in Y
                        {
                            // Similar logic for Y
                            int bestNextZForY = currentZ;

                            int desiredZ = targetZForStep; // For plant, this is 0.

                            if (currentZ > desiredZ)
                            {
                                bestNextZForY = Math.Max(desiredZ, currentZ - 1);
                            }
                            else
                            {
                                bestNextZForY = Math.Min(currentZ + 1, desiredZ);
                            }
                            bestNextZForY = Math.Max(0, bestNextZForY);

                            nextZ = bestNextZForY;
                            path.Add((currentX, currentY, currentZ));
                            currentY = potentialNextY;
                            currentZ = nextZ;
                        }
                        else
                        {
                            // Should not happen if currentX != targetX || currentY != targetY
                            break;
                        }
                    }
                    else // If moving in X
                    {
                        // We are moving from (currentX, currentY, currentZ) to (potentialNextX, currentY, nextZ)
                        // The constraint is on the Z difference relative to the X/Y difference.
                        // (nextZ - currentZ) / (abs(potentialNextX - currentX) + abs(currentY - currentY)) <= 1
                        // (nextZ - currentZ) / 1 <= 1  => nextZ <= currentZ + 1

                        // If the destination is at a lower Z, we can go downhill freely.
                        // If the destination is at a higher Z, we can go uphill by at most 1.

                        // For the plant, the target Z is 0.
                        int targetZLevel = 0; // Target Z level for the plant.

                        int requiredNextZ = currentZ; // Assume level

                        if (targetZLevel > currentZ) // If target is uphill
                        {
                            // We can go up by at most 1 Z per X/Y step.
                            requiredNextZ = Math.Min(currentZ + 1, targetZLevel);
                        }
                        else // If target is downhill or level
                        {
                            // We can go downhill or stay level.
                            requiredNextZ = Math.Max(targetZLevel, currentZ - 1); // Descend but not below target Z
                        }
                        requiredNextZ = Math.Max(0, requiredNextZ); // Ensure Z is not negative.

                        nextZ = requiredNextZ;
                        path.Add((currentX, currentY, currentZ));
                        currentX = potentialNextX;
                        currentZ = nextZ;
                    }
                }
                else if (dy != 0) // If moving in Y
                {
                    // Similar logic for Y movement.
                    int targetZLevel = 0; // Target Z level for the plant.

                    int requiredNextZ = currentZ; // Assume level

                    if (targetZLevel > currentZ) // If target is uphill
                    {
                        // We can go up by at most 1 Z per X/Y step.
                        requiredNextZ = Math.Min(currentZ + 1, targetZLevel);
                    }
                    else // If target is downhill or level
                    {
                        // We can go downhill or stay level.
                        requiredNextZ = Math.Max(targetZLevel, currentZ - 1); // Descend but not below target Z
                    }
                    requiredNextZ = Math.Max(0, requiredNextZ); // Ensure Z is not negative.

                    nextZ = requiredNextZ;
                    path.Add((currentX, currentY, currentZ));
                    currentY = potentialNextY;
                    currentZ = nextZ;
                }
                else
                {
                    // Should not happen if currentX != targetX || currentY != targetY
                    break;
                }
            }
            else // If moving in Y
            {
                // Similar logic for Y movement.
                int targetZLevel = 0; // Target Z level for the plant.

                int requiredNextZ = currentZ; // Assume level

                if (targetZLevel > currentZ) // If target is uphill
                {
                    // We can go up by at most 1 Z per X/Y step.
                    requiredNextZ = Math.Min(currentZ + 1, targetZLevel);
                }
                else // If target is downhill or level
                {
                    // We can go downhill or stay level.
                    requiredNextZ = Math.Max(targetZLevel, currentZ - 1); // Descend but not below target Z
                }
                requiredNextZ = Math.Max(0, requiredNextZ); // Ensure Z is not negative.

                nextZ = requiredNextZ;
                path.Add((currentX, currentY, currentZ));
                currentY = potentialNextY;
                currentZ = nextZ;
            }
        }

        // Add the final target location as the last step in the path.
        path.Add((targetX, targetY, currentZ)); // The Z at the destination is the final Z.

        // The path is from start to target. We need to verify each step is diggable.
        // This requires a more robust pathfinding (e.g., A* on a grid with voxel presence).

        // For now, this path is conceptual. We need to ensure each step is valid.
        // A better `GetDigPathToTarget` would explore valid moves.

        // Let's refine the path generation with a BFS-like approach on a grid.
        return GenerateValidPath(startX, startY, startZ, targetX, targetY);
    }

    private List<(int x, int y, int z)> GenerateValidPath(int startX, int startY, int startZ, int targetX, int targetY)
    {
        var queue = new Queue<(int x, int y, int z, List<(int x, int y, int z)> path)>();
        var visited = new HashSet<(int x, int y, int z)>();

        var initialPath = new List<(int x, int y, int z)> { (startX, startY, startZ) };
        queue.Enqueue((startX, startY, startZ, initialPath));
        visited.Add((startX, startY, startZ));

        int[] dx = { 0, 0, 1, -1 };
        int[] dy = { 1, -1, 0, 0 };

        while (queue.Count > 0)
        {
            var (currX, currY, currZ, currentPath) = queue.Dequeue();

            if (currX == targetX && currY == targetY)
            {
                return currentPath; // Found a path to the target X/Y.
            }

            // Explore neighbors
            for (int i = 0; i < 4; i++)
            {
                int nextX = currX + dx[i];
                int nextY = currY + dy[i];

                // Check bounds for next X and Y
                if (nextX < 0 || nextX >= XVoxelCount || nextY < 0 || nextY >= YVoxelCount)
                {
                    continue;
                }

                // Determine the valid Z for the next step based on slope and terrain.
                // We want to find an available Z at (nextX, nextY) that respects slope.
                // The target Z is effectively 0 for the plant.

                int targetZLevel = 0; // For processing plant.

                // If the destination is uphill (target Z > current Z)
                if (targetZLevel > currZ)
                {
                    // We can go up at most 1 Z per X/Y step.
                    // The next Z should be at most currZ + 1.
                    // We also need to consider the actual terrain at (nextX, nextY).
                    // Find the highest available Z at (nextX, nextY) that is <= currZ + 1.
                    for (int nextZ = Math.Min(currZ + 1, ZVoxelCount - 1); nextZ >= 0; nextZ--)
                    {
                        // Check if the voxel at (nextX, nextY, nextZ) is air or dumped dirt.
                        bool isNextVoxelAir = voxelGrid[nextX, nextY, nextZ] == null && !occupiedGrid[nextX, nextY, nextZ];

                        // Check if there's something below to support it (if not at Z=0).
                        bool isSupported = nextZ == 0 || (voxelGrid[nextX, nextY, nextZ - 1] != null || occupiedGrid[nextX, nextY, nextZ - 1]);

                        if (isNextVoxelAir && isSupported && nextZ <= currZ + 1)
                        {
                            if (!visited.Contains((nextX, nextY, nextZ)))
                            {
                                var newPath = new List<(int x, int y, int z)>(currentPath);
                                newPath.Add((nextX, nextY, nextZ));
                                queue.Enqueue((nextX, nextY, nextZ, newPath));
                                visited.Add((nextX, nextY, nextZ));
                            }
                            break; // Found a valid Z, move to next X/Y.
                        }
                    }
                }
                else // If the destination is downhill or level
                {
                    // We can go downhill freely.
                    // Find the highest available Z at (nextX, nextY) that is <= currZ.
                    for (int nextZ = currZ; nextZ >= 0; nextZ--)
                    {
                        bool isNextVoxelAir = voxelGrid[nextX, nextY, nextZ] == null && !occupiedGrid[nextX, nextY, nextZ];
                        bool isSupported = nextZ == 0 || (voxelGrid[nextX, nextY, nextZ - 1] != null || occupiedGrid[nextX, nextY, nextZ - 1]);

                        if (isNextVoxelAir && isSupported)
                        {
                            if (!visited.Contains((nextX, nextY, nextZ)))
                            {
                                var newPath = new List<(int x, int y, int z)>(currentPath);
                                newPath.Add((nextX, nextY, nextZ));
                                queue.Enqueue((nextX, nextY, nextZ, newPath));
                                visited.Add((nextX, nextY, nextZ));
                            }
                            break; // Found a valid Z, move to next X/Y.
                        }
                    }
                }
            }
        }

        return null; // No path found.
    }


    private void CalculateOperationCostAndProfit(Operation operation)
    {
        long fuelCost = 0;
        long potentialProfit = 0;

        // Calculate fuel cost
        for (int i = 0; i < operation.DigPath.Count - 1; i++)
        {
            var currentStep = operation.DigPath[i];
            var nextStep = operation.DigPath[i + 1];

            int deltaX = Math.Abs(nextStep.x - currentStep.x);
            int deltaY = Math.Abs(nextStep.y - currentStep.y);
            int deltaZ = nextStep.z - currentStep.z; // Can be negative for downhill

            double xYSlopeDistance = deltaX + deltaY;

            // Calculate fuel cost for this step
            if (deltaZ > 0) // Uphill movement
            {
                fuelCost += (long)operation.VoxelToMove.Mass * FUEL_COST_PER_SLOPE_STEP * deltaZ;
            }
            else // Downhill or flat movement
            {
                fuelCost += (long)operation.VoxelToMove.Mass * FUEL_COST_PER_X_Y_STEP * xYSlopeDistance;
            }
        }
        operation.FuelCost = fuelCost;

        // Calculate profit/loss
        if (operation.Destination.x == PROCESSING_PLANT_X && operation.Destination.y == PROCESSING_PLANT_Y)
        {
            if (operation.VoxelToMove.IsOre)
            {
                potentialProfit = (long)operation.VoxelToMove.Mass * operation.VoxelToMove.ValuePerTonne * PROFIT_PER_ORE_TONNE_VALUE;
            }
            else // Dumping junk at plant
            {
                potentialProfit = -(long)operation.VoxelToMove.Mass * DAMAGE_PER_JUNK_TONNE;
            }
        }
        else // Dumping elsewhere
        {
            // No profit from dumping elsewhere, only fuel cost subtracted from total.
            // If it was ore dumped elsewhere, it's a loss of potential profit.
            if (operation.VoxelToMove.IsOre)
            {
                potentialProfit = -(long)operation.VoxelToMove.Mass * operation.VoxelToMove.ValuePerTonne * PROFIT_PER_ORE_TONNE_VALUE; // Lost opportunity cost
            }
            else
            {
                potentialProfit = 0; // Dirt dumped elsewhere has no inherent value or cost beyond fuel.
            }
        }
        operation.Profit = potentialProfit - fuelCost;
    }

    private void OutputResults()
    {
        Console.WriteLine(plannedOperations.Count);
        foreach (var op in plannedOperations)
        {
            var digPath = op.DigPath;
            var sb = new System.Text.StringBuilder();

            // Digging operations: list all X, Y, Z coordinates in the dig path.
            for (int i = 0; i < digPath.Count; i++)
            {
                sb.Append($"{digPath[i].x} {digPath[i].y} {digPath[i].z}");
                if (i < digPath.Count - 1)
                {
                    sb.Append(" ");
                }
            }

            // Destination
            sb.Append($" {op.Destination.x} {op.Destination.y}");
            Console.WriteLine(sb.ToString());
        }
    }
