struct Point {
    float x, y;
    int id;
};

float distSq(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// O(N^2) Nearest Neighbor
std::vector<int> solveNN(int N, const std::vector<Point>& cities) {
    std::vector<int> tour;
    tour.reserve(N);
    std::vector<bool> visited(N, false);
    int current = 0;
    tour.push_back(cities[0].id);
    visited[0] = true;
    for (int i = 1; i < N; ++i) {
        int nextCity = -1;
        float minDist = std::numeric_limits<float>::max();
        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                float d = distSq(cities[current], cities[j]);
                if (d < minDist) {
                    minDist = d;
                    nextCity = j;
                }
            }
        }
        current = nextCity;
        tour.push_back(cities[current].id);
        visited[current] = true;
    }
    return tour;
}

// O(N log N) approximation using sorting (Space-filling curve idea)
std::vector<int> solveFast(int N, std::vector<Point> cities) {
    // Sort by x, then y. This is a very crude approximation of a space-filling curve.
    std::sort(cities.begin(), cities.end(), [](const Point& a, const Point& b) {
        if (a.x != b.x) return a.x < b.x;
        return a.y < b.y;
    });

    std::vector<int> tour;
    tour.reserve(N);
    for (const auto& city : cities) {
        tour.push_back(city.id);
    }
    return tour;
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        edges.push((u, v));
    }

    let mut rng = rand::thread_rng();
    let mut nodes: Vec<Node> = (0..n)
        .map(|_| Node {
            x: rng.gen_range(-10.0..10.0),
            y: rng.gen_range(-10.0..10.0),
        })
        .collect();

    // Simple spring embedder: move connected nodes closer, all nodes apart
    for _ in 0..100 {
        let mut forces = vec![(0.0f32, 0.0f32); n];
        
        // Repulsion
        for i in 0..n {
            for j in 0..n {
                if i == j { continue; }
                let dx = nodes[i].x - nodes[j].x;
                let dy = nodes[i].y - nodes[j].y;
                let d2 = dx * dx + dy * dy + 0.01;
                let _d = d2.sqrt();
                forces[i].0 += dx / d2;
                forces[i].1 += dy / d2;
            }
        }
        
        // Attraction
        for &(u, v) in &edges {
            let dx = nodes[u].x - nodes[v].x;
            let dy = nodes[u].y - nodes[v].y;
            let d = (dx * dx + dy * dy).sqrt();
            let f = if d > 0.0 { (d - 1.0) / d } else { 0.0 };
            forces[u].0 -= dx * f;
            forces[u].1 -= dy * f;
            forces[v].0 += dx * f;
            forces[v].1 += dy * f;
        }
        
        // Update

            nodes[i].x += forces[i].0 * 0.1;
            nodes[i].y += forces[i].1 * 0.1;
        }
    }

fn find_path(cells: &mut Vec<Vec<Cell>>, path: &mut Vec<(usize, usize)>, target_count: usize, require_cycle: bool) -> bool {

struct Point {

impl Point {
    fn rotate(&self, angle: f64) -> Point {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Point {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }

#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|p| Point { x: p.x + dx, y: p.y + dy }).collect(),
        }
    }

    fn rotate(&self, angle: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|p| p.rotate(angle)).collect(),
        }
    }

    fn get_edges(&self) -> Vec<Point> {
        let mut edges = Vec::new();
        for i in 0..self.vertices.len() {
            let p1 = self.vertices[i];
            let p2 = self.vertices[(i + 1) % self.vertices.len()];
            edges.push(Point { x: p2.x - p1.x, y: p2.y - p1.y });
        }
        edges
    }

    fn get_normals(&self) -> Vec<Point> {
        self.get_edges().iter().map(|e| Point { x: -e.y, y: e.x }).collect()
    }

    fn project(&self, axis: &Point) -> (f64, f64) {
        let mut min = f64::INFINITY;
        let mut max = f64::NEG_INFINITY;
        for v in &self.vertices {
            let dot = v.x * axis.x + v.y * axis.y;
            min = min.min(dot);
            max = max.max(dot);
        }
        (min, max)
    }
}

fn polygons_collide(p1: &Polygon, p2: &Polygon) -> bool {
    let mut axes = p1.get_normals();
    axes.extend(p2.get_normals());

    for axis in axes {
        let (min1, max1) = p1.project(&axis);
        let (min2, max2) = p2.project(&axis);
        if max1 < min2 || max2 < min1 {
            return false;
        }
    }
    true
}

struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn rotate(&self, angle: f64) -> Point {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Point {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }
}

#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|p| Point { x: p.x + dx, y: p.y + dy }).collect(),
        }
    }

    fn rotate(&self, angle: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|p| p.rotate(angle)).collect(),
        }
    }

    fn get_edges(&self) -> Vec<Point> {
        let mut edges = Vec::new();
        for i in 0..self.vertices.len() {
            let p1 = self.vertices[i];
            let p2 = self.vertices[(i + 1) % self.vertices.len()];
            edges.push(Point { x: p2.x - p1.x, y: p2.y - p1.y });
        }
        edges
    }

    fn get_normals(&self) -> Vec<Point> {
        self.get_edges().iter().map(|e| Point { x: -e.y, y: e.x }).collect()
    }

    fn project(&self, axis: &Point) -> (f64, f64) {
        let mut min = f64::INFINITY;
        let mut max = f64::NEG_INFINITY;
        for v in &self.vertices {
            let dot = v.x * axis.x + v.y * axis.y;
            min = min.min(dot);
            max = max.max(dot);
        }
        (min, max)
    }
}

fn polygons_collide(p1: &Polygon, p2: &Polygon) -> bool {
    let mut axes = p1.get_normals();
    axes.extend(p2.get_normals());

    for axis in axes {
        let (min1, max1) = p1.project(&axis);
        let (min2, max2) = p2.project(&axis);
        if max1 < min2 || max2 < min1 {
            return false;
        }
    }
    true
}
