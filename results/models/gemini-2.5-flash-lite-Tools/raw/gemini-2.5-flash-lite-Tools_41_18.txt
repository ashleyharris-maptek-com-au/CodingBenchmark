; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; Annotations
               OpDecorate %outColor Location 0
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

; Types
       %void = OpTypeVoid
        %int = OpTypeInt 32 1
       %bool = OpTypeBool
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v2float = OpTypeVector %float 2
    %v4float = OpTypeVector %float 4
   %mat4float = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4 = OpTypePointer Uniform %v4float
   %ptr_Input_v3 = OpTypePointer Input %v3float
   %ptr_Input_v2 = OpTypePointer Input %v2float
  %ptr_Output_v4 = OpTypePointer Output %v4float
       %void_f = OpTypeFunction %void

; Constants
      %int_0 = OpConstant %int 0
      %int_4 = OpConstant %int 4
         %f0 = OpConstant %float 0.0
       %f0_5 = OpConstant %float 0.5
         %f1 = OpConstant %float 1.0
         %f2 = OpConstant %float 2.0

; Variables
        %ubo = OpVariable %ptr_Uniform_UBO Uniform
   %worldPos = OpVariable %ptr_Input_v3 Input
     %normal = OpVariable %ptr_Input_v3 Input
         %uv = OpVariable %ptr_Input_v2 Input
    %tangent = OpVariable %ptr_Input_v3 Input
      %color = OpVariable %ptr_Input_v3 Input
   %outColor = OpVariable %ptr_Output_v4 Output

; Function
       %main = OpFunction %void None %void_f
         %L1 = OpLabel

; Load normal and normalize
     %n_vec = OpLoad %v3float %normal
   %n_norm = OpExtInst %v3float %1 Normalize %n_vec

; Load worldPos and cameraPos to compute V
   %w_pos = OpLoad %v3float %worldPos
   %ptr_ubo_camPos = OpAccessChain %ptr_Uniform_v4 %ubo %int_4
   %camPos_v4 = OpLoad %v4float %ptr_ubo_camPos
   %camPos = OpVectorShuffle %v3float %camPos_v4 %camPos_v4 0 1 2

   %v_vec = OpFSub %v3float %camPos %w_pos
   %v_norm = OpExtInst %v3float %1 Normalize %v_vec

; NdotV = max(dot(N, V), 0)
   %dot_nv = OpDot %float %n_norm %v_norm
   %ndotv = OpExtInst %float %1 FMax %dot_nv %f0

   %cond = OpFLessThan %bool %ndotv %f0_5

; Branching
               OpSelectionMerge %merge None
               OpBranchConditional %cond %if_true %if_false

     %if_true = OpLabel
   %g_true = OpFMul %float %f2 %ndotv
   %color_t = OpCompositeConstruct %v4float %f1 %g_true %f0 %f1
               OpBranch %merge

     %if_false = OpLabel
   %diff = OpFSub %float %ndotv %f0_5
   %val = OpFMul %float %diff %f2
   %c_fg = OpFSub %float %f1 %val
   %color_f = OpCompositeConstruct %v4float %c_fg %c_fg %val %f1
               OpBranch %merge

     %merge = OpLabel
   %final_color = OpPhi %v4float %color_t %if_true %color_f %if_false
               OpStore %outColor %final_color
               OpReturn
               OpFunctionEnd
