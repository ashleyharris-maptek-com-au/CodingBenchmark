%float_2 = OpConstant %float 2.0

%ubo = OpVariable %ptr_ubo Uniform
    OpDecorate %ubo DescriptorSet 0
    OpDecorate %ubo Binding 0

; Input/Output variables
%worldPos_var = OpVariable %ptr_input_v3 Input
    OpDecorate %worldPos_var Location 0
%normal_var = OpVariable %ptr_input_v3 Input
    OpDecorate %normal_var Location 1
%uv_var = OpVariable %ptr_input_v3 Input ; Prompt lists vec2 but shows location 2 and implies vec3. Using vec3 as per prompt.
    OpDecorate %uv_var Location 2

    OpDecorate %tangent_var Location 3

; Entry point function
%main = OpFunction %void %func_void_main
    %label_entry = OpLabel

    ; Load UBO and camera position
    %ubo_val = OpLoad %ubo_struct %ubo
    %cameraPos = OpCompositeExtract %v4float %ubo_val 4 ; Offset 4 for cameraPos

    ; Load input variables that are actually used
    %worldPos_in = OpLoad %v3float %worldPos_var
    %normal_in = OpLoad %v3float %normal_var

    ; Calculate view direction V
    ; Cast worldPos_in (vec3) to vec4 with w=1.0 for subtraction with cameraPos (vec4)
    %worldPos_in_v4 = OpCompositeConstruct %v4float %worldPos_in %float_1
    %view_vec_4 = OpVectorSubtract %v4float %cameraPos %worldPos_in_v4
    ; Normalize the xyz components, discarding w. Result is vec3.
    %view_dir = OpNormalize %v3float %view_vec_4

    ; Calculate NdotV = max(dot(N, V), 0.0)
    %dot_nv = OpDot %float %normal_in %view_dir
    %ndotv = OpMax %float %dot_nv %float_0

    ; Conditional logic for color based on NdotV
    ; cond_lt_0_5 is true if NdotV < 0.5
    %cond_lt_0_5 = OpFOrdLessThan %float %ndotv %float_0_5

    ; Calculate components for the 'else' branch (NdotV >= 0.5)
    ; factor = (NdotV - 0.5) * 2.0
    %ndotv_minus_0_5 = OpFSub %float %ndotv %float_0_5
    %two_times_factor = OpFMul %float %ndotv_minus_0_5 %float_2
    ; 1.0 - factor
    %one_minus_two_times_factor = OpFSub %float %float_1 %two_times_factor

    ; Select components based on the condition
    ; r = OpSelect (cond ? 1.0 : 1.0 - factor)
    %r_val = OpSelect %float %cond_lt_0_5 %float_1 %one_minus_two_times_factor
    ; g = OpSelect (cond ? 2.0 * NdotV : 1.0 - factor)
    %factor_lt_0_5_g = OpFMul %float %ndotv %float_2 ; 2.0 * NdotV
    %g_val = OpSelect %float %cond_lt_0_5 %factor_lt_0_5_g %one_minus_two_times_factor
    ; b = OpSelect (cond ? 0.0 : factor)
    %b_val = OpSelect %float %cond_lt_0_5 %float_0 %two_times_factor

    ; Construct the final output color with alpha = 1.0
    %outColor_val = OpCompositeConstruct %v4float %r_val %g_val %b_val %float_1

    ; Store the computed color to the output variable
    OpStore %outColor_var %outColor_val

    OpReturn
OpFunctionEnd
