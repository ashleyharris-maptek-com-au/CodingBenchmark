#version 430

struct Boid {
    vec3 position;
    vec3 velocity;
};

layout(std430, binding = 0) buffer BoidBuffer {
    Boid boids[];
};

layout(std430, binding = 1) buffer NextBoidBuffer {
    Boid nextBoids[];
};

uniform float deltaTime;
uniform int numBoids;

// Simulation parameters
const float perceptionRadius = 1.0;
const float separationWeight = 1.5;
const float alignmentWeight = 1.0;
const float cohesionWeight = 1.0;
const float maxSpeed = 0.5;

layout(local_size_x = 256) in;

void main() {
    uint id = gl_GlobalInvocationID.x;
    if (id >= numBoids) return;

    vec3 pos = boids[id].position;
    vec3 vel = boids[id].velocity;

    vec3 separation = vec3(0.0);
    vec3 alignment = vec3(0.0);
    vec3 cohesion = vec3(0.0);
    int count = 0;

    for (int i = 0; i < numBoids; ++i) {
        if (i == id) continue;

        vec3 otherPos = boids[i].position;
        float dist = distance(pos, otherPos);

        if (dist < perceptionRadius && dist > 0.0) {
            // Separation
            separation += (pos - otherPos) / dist;
            
            // Alignment
            alignment += boids[i].velocity;
            
            // Cohesion
            cohesion += otherPos;
            
            count++;
        }
    }

    if (count > 0) {
        separation /= float(count);
        alignment = (alignment / float(count)) - vel;
        cohesion = (cohesion / float(count)) - pos;
    }

    // Apply rules
    vel += separation * separationWeight;
    vel += alignment * alignmentWeight;
    vel += cohesion * cohesionWeight;

    // Limit speed
    if (length(vel) > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    nextBoids[id].position = pos + vel * deltaTime;
    nextBoids[id].velocity = vel;
}
