#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 1. Calculate checkerboard color
    // UV checkerboard 8x8
    vec2 checker = floor(uv * 8.0);
    float parity = mod(checker.x + checker.y, 2.0);
    vec3 baseColor = mix(vec3(0.9), vec3(0.2), parity);

    // 2. Calculate Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    float diff = max(dot(N, L), 0.0);
    float diffuseFactor = 0.5 + 0.5 * diff;
    
    vec3 finalColor = baseColor * diffuseFactor;
    
    // 3. Output
    outColor = vec4(finalColor, 1.0);
}
