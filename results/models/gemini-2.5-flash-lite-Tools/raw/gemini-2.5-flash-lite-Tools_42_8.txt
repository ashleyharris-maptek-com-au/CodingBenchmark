; SPIR-V Geometry Shader: Pass-through triangles
; Capabilities
OpCapability Geometry
OpCapability Shader

; Memory Model
OpMemoryModel Logical GLSL450

; Entry Point
OpEntryPoint Geometry %main "main" %in_pos %out_pos

; Execution Modes
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

; Decorations
OpDecorate %in_pos BuiltIn Position
OpDecorate %out_pos BuiltIn Position

; Types and Constants
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec4 = OpTypeVector %float 4
%int = OpTypeInt 32 1
%bool = OpTypeBool
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_3 = OpConstant %int 3

%in_pos_array = OpTypeArray %vec4 %int_3
%ptr_in_pos_array = OpTypePointer Input %in_pos_array
%ptr_vec4_input = OpTypePointer Input %vec4
%ptr_vec4_output = OpTypePointer Output %vec4

%func = OpTypeFunction %void
%ptr_int = OpTypePointer Function %int

; Variables
%in_pos = OpVariable %ptr_in_pos_array Input
%out_pos = OpVariable %ptr_vec4_output Output

; Main Function
%main = OpFunction %void None %func
%label = OpLabel
%i = OpVariable %ptr_int Function
OpStore %i %int_0
OpBranch %loop_header

%loop_header = OpLabel
%i_val = OpLoad %int %i
%loop_cond = OpSLessThan %bool %i_val %int_3
OpLoopMerge %loop_break %loop_continue None
OpBranchConditional %loop_cond %loop_body %loop_break

%loop_body = OpLabel
%i_val_body = OpLoad %int %i
%ptr_in_v = OpAccessChain %ptr_vec4_input %in_pos %i_val_body
%in_v = OpLoad %vec4 %ptr_in_v
OpStore %out_pos %in_v
OpEmitVertex
OpBranch %loop_continue

%loop_continue = OpLabel
%i_val_cont = OpLoad %int %i
%next_i = OpIAdd %int %i_val_cont %int_1
OpStore %i %next_i
OpBranch %loop_header

%loop_break = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
