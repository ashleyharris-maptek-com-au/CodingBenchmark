; SPIR-V Assembly for Geometry Shader
; Task: Shrink each triangle toward its centroid.

; --- UBO (Uniform Buffer Object) ---
; Assume std140 layout for alignment
%mat4f = OpTypeMatrix %float 4 4
%vec3f = OpTypeVector %float 3
%vec4f = OpTypeVector %float 4
%float_params = OpTypeArray %float %uint_4 ; Assuming 4 float parameters

%UBO = OpTypeStruct %mat4f %mat4f %mat4f %vec3f %vec4f %float_params
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0            ; Model Matrix
OpMemberDecorate %UBO 1 Offset 64           ; View Matrix
OpMemberDecorate %UBO 2 Offset 128          ; Projection Matrix
OpMemberDecorate %UBO 3 Offset 192          ; Light Position (vec3)
OpMemberDecorate %UBO 4 Offset 208          ; Camera Position (vec4)
OpMemberDecorate %UBO 5 Offset 224          ; Parameters (4 floats)

%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_Uniform_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; --- Header and Capabilities ---
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450

; --- Entry Point and Execution Modes ---
; Input primitive: Triangles (3 vertices)
; Output primitive: TriangleStrip (3 vertices)
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor_out
OpExecutionMode %main Input
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; --- Type Declarations ---
%void = OpTypeVoid
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%v4i = OpTypeVector %int 4

; --- Constants ---
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0.0
%float_0_7 = OpConstant %float 0.7
%float_3 = OpConstant %float 3.0

; --- Input Structures ---
; Per-vertex input block (for gl_Position)
%gl_PerVertex_in = OpTypeStruct %v4f
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Array of per-vertex inputs (size 3 for Triangles)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%gl_in = OpVariable %arr_pv_in Input ; Array of gl_PerVertex_in

; User-defined inputs
%arr_v3f_in = OpTypeArray %v3f %uint_3
%worldPos_in = OpVariable %arr_v3f_in Input
%normal_in = OpVariable %arr_v3f_in Input
%color_in = OpVariable %arr_v3f_in Input

; --- Output Structures ---
; Per-vertex output block (for gl_Position)
%gl_PerVertex_out = OpTypeStruct %v4f
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Output variables (scalar/vector types for each emitted vertex)
%gl_out = OpVariable Output %gl_PerVertex_out
%outColor_out = OpVariable Output %v3f

; --- Function: main ---
%main = OpFunction %void None %entry_point_args
%entry_point_args = OpTypeFunction %void

    ; Get pointers to interface variables
    %ptr_gl_in = OpFunctionCall %void %get_gl_in
    %ptr_worldPos_in = OpFunctionCall %void %get_worldPos_in
    %ptr_normal_in = OpFunctionCall %void %get_normal_in
    %ptr_color_in = OpFunctionCall %void %get_color_in
    %ptr_gl_out = OpFunctionCall %void %get_gl_out
    %ptr_outColor_out = OpFunctionCall %void %get_outColor_out
    
    ; --- Access Input Data ---
    ; Access gl_in[i].gl_Position for i = 0, 1, 2
    %ptr_gl_pos_0 = OpAccessChain %v4f %ptr_gl_in %uint_0 %uint_0
    %ptr_gl_pos_1 = OpAccessChain %v4f %ptr_gl_in %uint_1 %uint_0
    %ptr_gl_pos_2 = OpAccessChain %v4f %ptr_gl_in %uint_2 %uint_0

    ; Access user-defined inputs for i = 0, 1, 2
    %ptr_worldPos_0 = OpAccessChain %v3f %ptr_worldPos_in %uint_0
    %ptr_worldPos_1 = OpAccessChain %v3f %ptr_worldPos_in %uint_1
    %ptr_worldPos_2 = OpAccessChain %v3f %ptr_worldPos_in %uint_2

    %ptr_color_0 = OpAccessChain %v3f %ptr_color_in %uint_0
    %ptr_color_1 = OpAccessChain %v3f %ptr_color_in %uint_1
    %ptr_color_2 = OpAccessChain %v3f %ptr_color_in %uint_2

    ; --- Access UBO ---
    %ptr_ubo = OpAccessChain %ptr_Uniform_UBO %ubo %uint_5 ; Pointer to params array (index 5)
    %params = OpLoad %float_params %ptr_ubo

    ; --- Compute Centroid ---
    %pos_0 = OpLoad %v4f %ptr_gl_pos_0
    %pos_1 = OpLoad %v4f %ptr_gl_pos_1
    %pos_2 = OpLoad %v4f %ptr_gl_pos_2

    ; Summing components
    %sum_x = OpFAdd %float (OpExtractComponent %float %pos_0 0) (OpFAdd %float (OpExtractComponent %float %pos_1 0) (OpExtractComponent %float %pos_2 0))
    %sum_y = OpFAdd %float (OpExtractComponent %float %pos_0 1) (OpFAdd %float (OpExtractComponent %float %pos_1 1) (OpExtractComponent %float %pos_2 1))
    %sum_z = OpFAdd %float (OpExtractComponent %float %pos_0 2) (OpFAdd %float (OpExtractComponent %float %pos_1 2) (OpExtractComponent %float %pos_2 2))
    %sum_w = OpFAdd %float (OpExtractComponent %float %pos_0 3) (OpFAdd %float (OpExtractComponent %float %pos_1 3) (OpExtractComponent %float %pos_2 3))

    %centroid_x = OpFDiv %float %sum_x %float_3
    %centroid_y = OpFDiv %float %sum_y %float_3
    %centroid_z = OpFDiv %float %sum_z %float_3
    %centroid_w = OpFDiv %float %sum_w %float_3 ; W component for centroid

    %centroid = OpCompositeConstruct %v4f %centroid_x %centroid_y %centroid_z %centroid_w

    ; --- Shrink Vertices ---
    %shrink_factor_val = OpExtractComponent %float %params 0 ; Use params[0] as the shrink factor
    %shrink_factor = %shrink_factor_val
    
    ; Vertex 0
    %vec_0 = OpFSub %v4f %pos_0 %centroid
    %shrunk_vec_0 = OpFMul %v4f %shrink_factor %vec_0
    %new_pos_0 = OpFAdd %v4f %centroid %shrunk_vec_0
    
    ; Vertex 1
    %vec_1 = OpFSub %v4f %pos_1 %centroid
    %shrunk_vec_1 = OpFMul %v4f %shrink_factor %vec_1
    %new_pos_1 = OpFAdd %v4f %centroid %shrunk_vec_1

    ; Vertex 2
    %vec_2 = OpFSub %v4f %pos_2 %centroid
    %shrunk_vec_2 = OpFMul %v4f %shrink_factor %vec_2
    %new_pos_2 = OpFAdd %v4f %centroid %shrunk_vec_2

    ; --- Emit Vertices ---
    ; Emit Vertex 0
    %ptr_out_pos_0 = OpAccessChain %v4f %ptr_gl_out %uint_0
    OpStore %ptr_out_pos_0 %new_pos_0
    %color_0 = OpLoad %v3f %ptr_color_0
    %ptr_outColor_0 = OpAccessChain %v3f %ptr_outColor_out %uint_0 ; Assuming direct access to output color array
    OpStore %ptr_outColor_0 %color_0
    OpEmitVertex
    
    ; Emit Vertex 1
    %ptr_out_pos_1 = OpAccessChain %v4f %ptr_gl_out %uint_1
    OpStore %ptr_out_pos_1 %new_pos_1
    %color_1 = OpLoad %v3f %ptr_color_1
    %ptr_outColor_1 = OpAccessChain %v3f %ptr_outColor_out %uint_1 ; Assuming direct access to output color array
    OpStore %ptr_outColor_1 %color_1
    OpEmitVertex

    ; Emit Vertex 2
    %ptr_out_pos_2 = OpAccessChain %v4f %ptr_gl_out %uint_2
    OpStore %ptr_out_pos_2 %new_pos_2
    %color_2 = OpLoad %v3f %ptr_color_2
    %ptr_outColor_2 = OpAccessChain %v3f %ptr_outColor_out %uint_2 ; Assuming direct access to output color array
    OpStore %ptr_outColor_2 %color_2
    OpEmitVertex

    OpEndPrimitive
    OpReturn
OpEndFunction

; Helper functions to get pointers to interface variables (simplification)
%get_gl_in = OpFunction %void None %void_args
%void_args = OpTypeFunction %void
    %ptr = OpVariable %arr_pv_in Input ; Must match the type of %gl_in
    OpReturnValue %ptr
OpEndFunction

%get_worldPos_in = OpFunction %void None %void_args
    %ptr = OpVariable %arr_v3f_in Input ; Must match the type of %worldPos_in
    OpReturnValue %ptr
OpEndFunction

%get_normal_in = OpFunction %void None %void_args
    %ptr = OpVariable %arr_v3f_in Input ; Must match the type of %normal_in
    OpReturnValue %ptr
OpEndFunction

%get_color_in = OpFunction %void None %void_args
    %ptr = OpVariable %arr_v3f_in Input ; Must match the type of %color_in
    OpReturnValue %ptr
OpEndFunction

%get_gl_out = OpFunction %void None %void_args
    %ptr = OpVariable %gl_PerVertex_out Output ; Must match the type of %gl_out
    OpReturnValue %ptr
OpEndFunction

%get_outColor_out = OpFunction %void None %void_args
    %ptr = OpVariable %v3f Output ; Must match the type of %outColor_out
    OpReturnValue %ptr
OpEndFunction
