#include <iostream>
#include <vector>
#include <algorithm> // For std::min, std::max, std::sort
#include <limits>    // For std::numeric_limits
#include <numeric>   // For std::accumulate (optional, for debugging/metrics)

// Structure to represent a rectangle
struct Rect {
    int x, y, width, height;
    bool rotated; // To indicate if the rectangle was rotated for placement
    int original_index; // To track the original index of the rectangle
    int board_index;    // To track which board it was placed on

    Rect(int x = 0, int y = 0, int w = 0, int h = 0, bool r = false, int orig_idx = -1, int b_idx = -1)
        : x(x), y(y), width(w), height(h), rotated(r), original_index(orig_idx), board_index(b_idx) {}

    // Convenience function to get the area of the rectangle
    int Area() const { return width * height; }
};

// MaxRectsBinPack class for packing rectangles into a single bin (board)
class MaxRectsBinPack {
public:
    int binWidth;
    int binHeight;

    std::vector<Rect> usedRectangles;
    std::vector<Rect> freeRectangles;

    MaxRectsBinPack(int width, int height) : binWidth(width), binHeight(height) {
        freeRectangles.clear();
        usedRectangles.clear();
        freeRectangles.push_back(Rect(0, 0, width, height));
    }

    /// Tries to place a rectangle of the given size into the bin.
    /// Returns the placed Rect structure if successful, otherwise an empty Rect (width=0).
    Rect PlaceRect(int rectWidth, int rectHeight, bool canRotate);

private:
    /// Chooses the best free rectangle to place the given rectangle using different heuristics.
    /// Returns the placed Rect structure (with x,y set) if a fit is found, otherwise an empty Rect.
    Rect ScoreBestFit(int rectWidth, int rectHeight, bool canRotate, int &bestFreeRectIndex);

    /// Various scoring heuristics for choosing the best fit.
    int ScoreBottomLeftRule(int rectWidth, int rectHeight, const Rect &freeRect) const;
    int ScoreBestShortSideFit(int rectWidth, int rectHeight, const Rect &freeRect) const;
    int ScoreBestLongSideFit(int rectWidth, int rectHeight, const Rect &freeRect) const;
    int ScoreBestAreaFit(int rectWidth, int rectHeight, const Rect &freeRect) const;

    // Helper function to check if rectangle 'a' is contained within rectangle 'b'.
    bool IsContainedIn(const Rect &a, const Rect &b) const;

    // Function to determine if two rectangles intersect.
    bool Intersects(const Rect& a, const Rect& b) const;

    // Pruning function, to remove free rectangles that are contained in other free rectangles.
    void PruneFreeList();

    // Function to perform the actual splitting and updating of free rectangles based on a used rectangle.
    void SplitFreeNode(Rect usedNode);
};

// Implementations for MaxRectsBinPack methods

/// Various scoring heuristics for choosing the best fit.
int MaxRectsBinPack::ScoreBottomLeftRule(int rectWidth, int rectHeight, const Rect &freeRect) const {
    if (freeRect.width >= rectWidth && freeRect.height >= rectHeight) {
        return freeRect.y * binWidth + freeRect.x;
    }
    return std::numeric_limits<int>::max(); // No fit
}

int MaxRectsBinPack::ScoreBestShortSideFit(int rectWidth, int rectHeight, const Rect &freeRect) const {
    if (freeRect.width >= rectWidth && freeRect.height >= rectHeight) {
        return std::min(freeRect.width - rectWidth, freeRect.height - rectHeight);
    }
    return std::numeric_limits<int>::max(); // No fit
}

int MaxRectsBinPack::ScoreBestLongSideFit(int rectWidth, int rectHeight, const Rect &freeRect) const {
    if (freeRect.width >= rectWidth && freeRect.height >= rectHeight) {
        return std::max(freeRect.width - rectWidth, freeRect.height - rectHeight);
    }
    return std::numeric_limits<int>::max(); // No fit
}

int MaxRectsBinPack::ScoreBestAreaFit(int rectWidth, int rectHeight, const Rect &freeRect) const {
    if (freeRect.width >= rectWidth && freeRect.height >= rectHeight) {
        return freeRect.Area() - (rectWidth * rectHeight);
    }
    return std::numeric_limits<int>::max(); // No fit
}

/// Chooses the best free rectangle to place the given rectangle using different heuristics.
/// Returns the placed Rect structure (with x,y set) if a fit is found, otherwise an empty Rect.
Rect MaxRectsBinPack::ScoreBestFit(int rectWidth, int rectHeight, bool canRotate, int &bestFreeRectIndex) {
    int bestScore = std::numeric_limits<int>::max();
    Rect bestRect;
    bestFreeRectIndex = -1;

    for (size_t i = 0; i < freeRectangles.size(); ++i) {
        const Rect &freeRect = freeRectangles[i];

        // Try placing the rectangle in its original orientation
        if (freeRect.width >= rectWidth && freeRect.height >= rectHeight) {
            int score = ScoreBottomLeftRule(rectWidth, rectHeight, freeRect); // Using BLL as primary heuristic
            if (score < bestScore) {
                bestScore = score;
                bestRect = Rect(freeRect.x, freeRect.y, rectWidth, rectHeight, false);
                bestFreeRectIndex = i;
            }
        }

        // Try placing the rectangle rotated (if allowed)
        if (canRotate && freeRect.width >= rectHeight && freeRect.height >= rectWidth) {
            int score = ScoreBottomLeftRule(rectHeight, rectWidth, freeRect); // Using BLL as primary heuristic
            if (score < bestScore) {
                bestScore = score;
                bestRect = Rect(freeRect.x, freeRect.y, rectHeight, rectWidth, true);
                bestFreeRectIndex = i;
                }
            }
        }

    if (bestFreeRectIndex != -1) {
        return bestRect;
    }
    return Rect(); // No fit found
}

// Helper function to check if rectangle 'a' is contained within rectangle 'b'.
bool MaxRectsBinPack::IsContainedIn(const Rect &a, const Rect &b) const {
    return a.x >= b.x && a.y >= b.y &&
           a.x + a.width <= b.x + b.width && a.y + a.height <= b.y + b.height;
}

// Function to determine if two rectangles intersect.
bool MaxRectsBinPack::Intersects(const Rect& a, const Rect& b) const {
    return !(a.x + a.width <= b.x || a.y + a.height <= b.y ||
             a.x >= b.x + b.width || a.y >= b.y + b.height);
}

// Pruning function, to remove free rectangles that are contained in other free rectangles.
void MaxRectsBinPack::PruneFreeList() {
    for (size_t i = 0; i < freeRectangles.size(); ++i) {
        for (size_t j = i + 1; j < freeRectangles.size(); ++j) {
            if (IsContainedIn(freeRectangles[i], freeRectangles[j])) {
                freeRectangles.erase(freeRectangles.begin() + i);
                --i;
                break;
            }
            if (IsContainedIn(freeRectangles[j], freeRectangles[i])) {
                freeRectangles.erase(freeRectangles.begin() + j);
                --j;
            }
        }
    }
}

// Function to perform the actual splitting and updating of free rectangles based on a used rectangle.
void MaxRectsBinPack::SplitFreeNode(Rect usedNode) {
    for (int i = freeRectangles.size() - 1; i >= 0; --i) {
        Rect freeRect = freeRectangles[i];

        if (!Intersects(freeRect, usedNode)) {
            continue;
        }

        if (IsContainedIn(freeRect, usedNode)) {
            freeRectangles.erase(freeRectangles.begin() + i);
            continue;
        }

        std::vector<Rect> newSplits;

        if (freeRect.y < usedNode.y && freeRect.y + freeRect.height > usedNode.y) {
            Rect newRect = freeRect;
            newRect.height = usedNode.y - newRect.y;
            newSplits.push_back(newRect);
        }

        if (freeRect.y + freeRect.height > usedNode.y + usedNode.height && freeRect.y < usedNode.y + usedNode.height) {
            Rect newRect = freeRect;
            newRect.y = usedNode.y + usedNode.height;
            newRect.height = freeRect.y + freeRect.height - (usedNode.y + usedNode.height);
            newSplits.push_back(newRect);
        }

        if (freeRect.x < usedNode.x && freeRect.x + freeRect.width > usedNode.x) {
            Rect newRect = freeRect;
            newRect.width = usedNode.x - newRect.x;
            newSplits.push_back(newRect);
        }

        if (freeRect.x + freeRect.width > usedNode.x + usedNode.width && freeRect.x < usedNode.x + usedNode.width) {
            Rect newRect = freeRect;
            newRect.x = usedNode.x + usedNode.width;
            newRect.width = freeRect.x + freeRect.width - (usedNode.x + usedNode.width);
            newSplits.push_back(newRect);
        }

        freeRectangles.erase(freeRectangles.begin() + i);
        for (const auto& splitRect : newSplits) {
            if (splitRect.width > 0 && splitRect.height > 0) {
                freeRectangles.push_back(splitRect);
            }
        }
    }
}


Rect MaxRectsBinPack::PlaceRect(int rectWidth, int rectHeight, bool canRotate) {
    int bestFreeRectIndex = -1;
    Rect newRect = ScoreBestFit(rectWidth, rectHeight, canRotate, bestFreeRectIndex);

    if (bestFreeRectIndex == -1) {
        return Rect(); // No place found
    }

    SplitFreeNode(newRect);
    PruneFreeList();

    usedRectangles.push_back(newRect);
    return newRect;
}


// Main function for the 2D Cutting Stock Problem
int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_rectangles;
    int board_width, board_height;

    // Read board dimensions and number of rectangles
    std::cin >> num_rectangles >> board_width >> board_height;

    std::vector<Rect> input_rects;
    for (int i = 0; i < num_rectangles; ++i) {
        int w, h;
        std::cin >> w >> h;
        input_rects.emplace_back(0, 0, w, h, false, i, -1); // x,y,w,h,rotated,original_index,board_index
    }

    // Sort rectangles by a heuristic (e.g., by area, or by longest side)
    // This often improves packing efficiency. Sorting by decreasing area is a common heuristic.
    std::sort(input_rects.begin(), input_rects.end(), [](const Rect &a, const Rect &b) {
        return a.Area() > b.Area(); // Sort in decreasing order of area
    });

    std::vector<MaxRectsBinPack> boards;
    std::vector<Rect> placed_rects(num_rectangles); // Store results by original index

    // Try to place each rectangle
    for (const auto& current_input_rect : input_rects) {
        bool placed = false;
        Rect result_rect;

        // Try placing in existing boards
        for (size_t i = 0; i < boards.size(); ++i) {
            result_rect = boards[i].PlaceRect(current_input_rect.width, current_input_rect.height, true); // true for canRotate
            if (result_rect.width > 0) { // Successfully placed
                result_rect.original_index = current_input_rect.original_index;
                result_rect.board_index = i;
                placed_rects[result_rect.original_index] = result_rect;
                placed = true;
                break;
            }
        }

        // If not placed, create a new board
        if (!placed) {
            boards.emplace_back(board_width, board_height);
            // Try placing on the newly created board
            result_rect = boards.back().PlaceRect(current_input_rect.width, current_input_rect.height, true);
            if (result_rect.width > 0) {
                result_rect.original_index = current_input_rect.original_index;
                result_rect.board_index = boards.size() - 1;
                placed_rects[result_rect.original_index] = result_rect;
                placed = true;
            } else {
                // This should ideally not happen if the rectangle fits on an empty board.
                // It means the rectangle is larger than the board itself.
                // The problem statement implies all rectangles can fit on a board.
                // For robustness, one might add error handling here.
            }
        }
    }

    // Output results
    std::cout << boards.size() << std::endl;
    for (const auto& r : placed_rects) {
        std::cout << r.board_index << " " << r.x << " " << r.y << " " << (r.rotated ? 1 : 0) << std::endl;
    }

    return 0;
}
