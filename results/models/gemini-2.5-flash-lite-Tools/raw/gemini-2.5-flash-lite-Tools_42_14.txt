; SPIR-V Geometry Shader
; Input: InputTrianglesAdjacency (6 vertices)
; Output: OutputTriangleStrip (3 vertices)
; Flat-shading triangles using face normal
; Lighting: Base * (0.2 + 0.8 * dot(N, L))

               OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos_in %color_in %outColor_out
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorate
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos_in Location 0
               OpDecorate %color_in Location 2
               OpDecorate %outColor_out Location 0

; Types
       %void = OpTypeVoid
      %bool  = OpTypeBool
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
      %uint_6 = OpConstant %uint 6
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
  %float_0_2 = OpConstant %float 0.2
  %float_0_8 = OpConstant %float 0.8
  %float_0_0 = OpConstant %float 0.0
  %float_1_0 = OpConstant %float 1.0
  %float_3_0 = OpConstant %float 3.0

; Structures
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output

%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor_out = OpVariable %ptr_out_v3 Output

%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%void_fn = OpTypeFunction %void
%ptr_int_fn = OpTypePointer Function %int
%arr_int_3 = OpTypeArray %int %int_3
%ptr_arr_int_3 = OpTypePointer Function %arr_int_3
%ptr_int_array_elem = OpTypePointer Function %int

; Main
       %main = OpFunction %void None %void_fn
      %entry = OpLabel

; 1. Get positions p0, p2, p4 (indices 0, 2, 4)
%ptr_p0 = OpAccessChain %ptr_in_v3 %worldPos_in %int_0
       %p0 = OpLoad %v3float %ptr_p0
%ptr_p2 = OpAccessChain %ptr_in_v3 %worldPos_in %int_2
       %p2 = OpLoad %v3float %ptr_p2
%ptr_p4 = OpAccessChain %ptr_in_v3 %worldPos_in %int_4
       %p4 = OpLoad %v3float %ptr_p4

; 2. Compute face normal: N = normalize(cross(p2-p0, p4-p0))
%v20 = OpFSub %v3float %p2 %p0
%v40 = OpFSub %v3float %p4 %p0
%cross = OpExtInst %v3float %1 Cross %v20 %v40
%N = OpExtInst %v3float %1 Normalize %cross

; 3. Light direction: L = normalize(vec3(1, 1, 1))
%light_dir_vec = OpConstantComposite %v3float %float_1_0 %float_1_0 %float_1_0
%L = OpExtInst %v3float %1 Normalize %light_dir_vec

; 4. Diffuse = max(dot(N, L), 0.0)
%dot = OpDot %float %N %L
%diffuse = OpExtInst %float %1 FMax %dot %float_0_0

; 5. Base color = average of color[0], color[2], color[4]
%ptr_c0 = OpAccessChain %ptr_in_v3 %color_in %int_0
       %c0 = OpLoad %v3float %ptr_c0
%ptr_c2 = OpAccessChain %ptr_in_v3 %color_in %int_2
       %c2 = OpLoad %v3float %ptr_c2
%ptr_c4 = OpAccessChain %ptr_in_v3 %color_in %int_4
       %c4 = OpLoad %v3float %ptr_c4
%sum_c = OpFAdd %v3float %c0 %c2
%sum_c2 = OpFAdd %v3float %sum_c %c4
%base_color = OpFDiv %v3float %sum_c2 %float_3_0

; 6. Final color = base * (0.2 + 0.8 * diffuse)
%mul_diff = OpFMul %float %float_0_8 %diffuse
%factor = OpFAdd %float %float_0_2 %mul_diff
%final_color = OpVectorTimesScalar %v3float %base_color %factor

; Loop to emit vertices 0, 2, 4
%emit_indices = OpConstantComposite %arr_int_3 %int_0 %int_2 %int_4
%i = OpVariable %ptr_int_fn Function
               OpStore %i %int_0
               OpBranch %loop

%loop = OpLabel
%loop_idx = OpLoad %int %i
%cond = OpSLessThan %bool %loop_idx %int_3
               OpBranchConditional %cond %body %end

%body = OpLabel
%ptr_idx = OpAccessChain %ptr_int_array_elem %emit_indices %loop_idx
%idx = OpLoad %int %ptr_idx
               OpStore %outColor_out %final_color
%ptr_pos_in = OpAccessChain %ptr_in_v4 %gl_in %idx %int_0
%pos_in = OpLoad %v4float %ptr_pos_in
%ptr_pos_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %ptr_pos_out %pos_in
               OpEmitVertex
%next_i = OpIAdd %int %loop_idx %int_1
               OpStore %i %next_i
               OpBranch %loop

%end = OpLabel
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
