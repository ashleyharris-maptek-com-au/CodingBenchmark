type Point = { x: number; y: number };
type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };
type Input = { grid: string[]; robots: Robot[]; jobs: Job[] };

function getShortestDistance(
    grid: string[],
    start: Point,
    end: Point
): number {
    const rows = grid.length;
    const cols = grid[0].length;
    const queue: [Point, number][] = [[start, 0]];
    const visited = new Set<string>();
    visited.add(`${start.x},${start.y}`);

    while (queue.length > 0) {
        const [{ x, y }, dist] = queue.shift()!;
        if (x === end.x && y === end.y) return dist;

        const neighbors = [
            { x: x + 1, y },
            { x: x - 1, y },
            { x, y: y + 1 },
            { x, y: y - 1 },
        ];

        for (const next of neighbors) {
            if (
                next.x >= 0 && next.x < cols &&
                next.y >= 0 && next.y < rows &&
                grid[next.y][next.x] !== '#' &&
                !visited.has(`${next.x},${next.y}`)
            ) {
                visited.add(`${next.x},${next.y}`);
                queue.push([next, dist + 1]);
            }
        }
    }
    return Infinity; // Should not happen based on problem constraints
}

function solve() {
    const fs = require('fs');
    const input: string = fs.readFileSync(0, 'utf8');
    const data: Input = JSON.parse(input);

    const { grid, robots, jobs } = data;
    const numRobots = robots.length;
    const assignments: number[][] = Array.from({ length: numRobots }, () => []);

    // Simple greedy assignment:
    // Calculate distances from all robots to all jobs
    // Repeatedly assign the closest job-robot pair
    
    let unassignedJobs = [...jobs];
    
    while (unassignedJobs.length > 0) {
        let bestDist = Infinity;
        let bestRobotIdx = -1;
        let bestJobIdx = -1;

        for (let r = 0; r < numRobots; r++) {
            const robot = robots[r];
            // Current position of robot is last assigned job, or initial pos
            const currentPos = assignments[r].length > 0
                ? jobs.find(j => j.id === assignments[r][assignments[r].length - 1])!
                : robot;

            for (let j = 0; j < unassignedJobs.length; j++) {
                const job = unassignedJobs[j];
                const dist = getShortestDistance(grid, { x: currentPos.x, y: currentPos.y }, { x: job.x, y: job.y });
                
                // Heuristic: divide distance by priority to prefer higher priority
                const score = dist / job.priority;

                if (score < bestDist) {
                    bestDist = score;
                    bestRobotIdx = r;
                    bestJobIdx = j;
                }
            }
        }

        const jobToAssign = unassignedJobs.splice(bestJobIdx, 1)[0];
        assignments[bestRobotIdx].push(jobToAssign.id);
    }

    process.stdout.write(JSON.stringify({ assignments }));
}

solve();
