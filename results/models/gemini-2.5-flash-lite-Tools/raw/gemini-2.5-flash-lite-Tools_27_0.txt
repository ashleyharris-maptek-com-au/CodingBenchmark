#include <iostream>
#include <vector>
#include <algorithm>
#include <set>

// DSATUR Coloring Algorithm
void solve() {
    int num_vertices, num_edges, num_colors;
    if (!(std::cin >> num_vertices >> num_edges >> num_colors)) return;

    std::vector<std::vector<int>> adj(num_vertices);
    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u >= 0 && u < num_vertices && v >= 0 && v < num_vertices) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    std::vector<int> colors(num_vertices, -1);
    std::vector<int> saturation(num_vertices, 0);
    std::vector<std::set<int>> neighbor_colors(num_vertices);
    std::vector<int> degree(num_vertices);
    
    for (int i = 0; i < num_vertices; ++i) {
        degree[i] = adj[i].size();
    }

    int colored_count = 0;
    while (colored_count < num_vertices) {
        // Select vertex with highest degree of saturation
        int best_v = -1;
        int max_sat = -1;
        int max_deg = -1;

        for (int i = 0; i < num_vertices; ++i) {
            if (colors[i] == -1) {
                if (saturation[i] > max_sat || (saturation[i] == max_sat && degree[i] > max_deg)) {
                    max_sat = saturation[i];
                    max_deg = degree[i];
                    best_v = i;
                }
            }
        }

        // Color best_v with lowest available color
        std::vector<bool> used(num_colors, false);
        for (int neighbor : adj[best_v]) {
            if (colors[neighbor] != -1) {
                used[colors[neighbor]] = true;
            }
        }

        int color = 0;
        while (color < num_colors && used[color]) {
            color++;
        }

        if (color < num_colors) {
            colors[best_v] = color;
            colored_count++;

            // Update neighbors' saturation
            for (int neighbor : adj[best_v]) {
                if (colors[neighbor] == -1) {
                    if (neighbor_colors[neighbor].find(color) == neighbor_colors[neighbor].end()) {
                        neighbor_colors[neighbor].insert(color);
                        saturation[neighbor]++;
                    }
                }
            }
        } else {
            // If no valid color, assign a default (0) and move on, though this implies failure.
            colors[best_v] = 0;
            colored_count++;
        }
    }

    for (int i = 0; i < num_vertices; ++i) {
        std::cout << colors[i] << (i == num_vertices - 1 ? "" : " ");
    }
    std::cout << std::endl;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    solve();
    return 0;
}
