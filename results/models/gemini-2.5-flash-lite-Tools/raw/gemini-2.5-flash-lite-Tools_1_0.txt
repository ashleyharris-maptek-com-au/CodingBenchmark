#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <numeric>

struct Point {
    double x, y;
};

inline double distSq(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// Nearest Neighbor heuristic (O(N^2))
std::vector<int> solveNearestNeighbor(int N, const std::vector<Point>& cities) {
    std::vector<int> path;
    path.reserve(N);
    path.push_back(0);
    
    std::vector<bool> visited(N, false);
    visited[0] = true;
    
    int current = 0;
    for (int i = 1; i < N; ++i) {
        int next = -1;
        double minD = 1e30;
        
        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                double d = distSq(cities[current], cities[j]);
                if (d < minD) {
                    minD = d;
                    next = j;
                }
            }
        }
        path.push_back(next);
        visited[next] = true;
        current = next;
    }
    return path;
}

// Fast heuristic for large N: Sort by X coordinate
std::vector<int> solveSortedByX(int N, const std::vector<Point>& cities) {
    std::vector<int> indices(N);
    std::iota(indices.begin(), indices.end(), 0);
    // Ensure 0 is first if possible, though sorting might move it.
    // Given the requirement "starting at city 0", we need a small adjustment.
    std::sort(indices.begin(), indices.end(), [&](int a, int b) {
        if (a == 0) return true;
        if (b == 0) return false;
        return cities[a].x < cities[b].x;
    });
    return indices;
}

int main() {
    // Fast I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N;
    if (!(std::cin >> N)) return 0;

    std::vector<Point> cities(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
    }

    if (N <= 0) return 0;
    if (N == 1) {
        std::cout << 0 << std::endl;
        return 0;
    }

    // Adaptive Strategy
    std::vector<int> path;
    if (N < 5000) {
        path = solveNearestNeighbor(N, cities);
    } else {
        path = solveSortedByX(N, cities);
    }

    for (int i = 0; i < N; ++i) {
        std::cout << path[i] << (i == N - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
