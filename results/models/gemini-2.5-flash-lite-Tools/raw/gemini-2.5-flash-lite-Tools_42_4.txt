OpCapability Geometry
OpCapability InputGeometry
OpCapability OutputGeometry
OpCapability StoredInput
OpCapability StoredOutput
OpCapability GLSL450

OpMemoryModel Logical GLSL450

; Types
%float_type = OpTypeFloat 32
%v2float = OpTypeVector %float_type 2
%v3float = OpTypeVector %float_type 3
%v4float = OpTypeVector %float_type 4
%uint_type = OpTypeInt 32 0
%int_type = OpTypeInt 32 1
%bool_type = OpTypeBool
%void_type = OpTypeVoid

; Constants
%uint_1 = OpConstant %uint_type 1
%uint_2 = OpConstant %uint_type 2
%uint_4 = OpConstant %uint_type 4

%int_0 = OpConstant %int_type 0
%int_1 = OpConstant %int_type 1
%int_2 = OpConstant %int_type 2

%float_0 = OpConstant %float_type 0.0
%float_1 = OpConstant %float_type 1.0
%float_0_04 = OpConstant %float_type 0.04
%float_0_005 = OpConstant %float_type 0.005

; Input structure for gl_PerVertex
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input array of gl_PerVertex (for InputLines)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User defined input variables (arrays of vec3)
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; Output structure for gl_PerVertex
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User defined output variable (vec3)
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Function types
%void_fn = OpTypeFunction %void_type

; Entry Point Function Type
%main_fn = OpTypeFunction %void_type

; Entry Point
%main = OpFunction %void_type %main_fn
    OpLabel

    ; --- Access input variables ---
    ; gl_in[i].gl_Position
    %ptr_gl_pos_0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
    %ptr_gl_pos_1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0

    ; worldPos[i]
    %ptr_worldPos_0 = OpAccessChain %ptr_in_v3 %worldPos %int_0
    %ptr_worldPos_1 = OpAccessChain %ptr_in_v3 %worldPos %int_1

    ; color[i]
    %ptr_color_0 = OpAccessChain %ptr_in_v3 %color %int_0
    %ptr_color_1 = OpAccessChain %ptr_in_v3 %color %int_1

    ; Load input values
    %worldPos_0 = OpLoad %v3float %ptr_worldPos_0
    %worldPos_1 = OpLoad %v3float %ptr_worldPos_1
    %color_0 = OpLoad %v3float %ptr_color_0
    %color_1 = OpLoad %v3float %ptr_color_1

    ; --- Geometry Calculation ---
    ; Vector from p0 to p1
    %v = OpSub %v3float %worldPos_1 %worldPos_0

    ; Extract components of v for perpendicular calculation
    %v_x = OpVectorExtractDynamic %float_type %v %int_0
    %v_y = OpVectorExtractDynamic %float_type %v %int_1

    ; Perpendicular vector in XY plane: (-v_y, v_x, 0.0)
    %perp_xy = OpCompositeConstruct %v3float (OpNegate %float_type %v_y), %v_x, %float_0

    ; Calculate length squared of perp_xy to normalize it.
    ; length^2 = (-v_y)^2 + v_x^2
    %perp_xy_len_sq = OpFMul %float_type %v_y (OpNegate %float_type %v_y) ; (-v_y)^2
    %perp_xy_len_sq = OpFAdd %float_type %perp_xy_len_sq (%OpFMul %float_type %v_x %v_x) ; + v_x^2

    ; Check if length squared is zero.
    %is_zero_len = OpExtInst %bool %GLSL450 FUnordEq %perp_xy_len_sq %float_0

    ; Default perpendicular if line is degenerate in XY (e.g., (1,0,0))
    %default_perp_xy = OpCompositeConstruct %v3float %float_1 %float_0 %float_0

    ; Calculate normalized perpendicular vector.
    ; If length is near zero, use default_perp_xy, otherwise use perp_xy / sqrt(length_sq).
    %sqrt_len_sq = OpSqrt %float_type %perp_xy_len_sq
    ; Use OpSelect to choose between normalized vector and default.
    ; If is_zero_len is true, use default_perp_xy. Otherwise, use perp_xy / sqrt_len_sq.
    %normalized_part = OpDiv %v3float %perp_xy %sqrt_len_sq
    %perp_normalized_xy = OpSelect %v3float %default_perp_xy %normalized_part %is_zero_len

    ; --- Emit Vertices (Triangle Strip) ---
    ; Vertex 0: p0 + perp * thick_start, color0
    %offset_p0_start = OpFMul %v3float %perp_normalized_xy %float_0_04
    %v0_pos = OpAdd %v3float %worldPos_0 %offset_p0_start
    %v0_final_pos = OpCompositeConstruct %v4float %v0_pos.x %v0_pos.y %v0_pos.z %float_1 ; w = 1.0
    OpStore %ptr_gl_out_pos_0 %v0_final_pos
    OpStore %outColor %color_0
    OpEmitVertex

    ; Vertex 1: p0 - perp * thick_start, color0
    %offset_p0_start_neg = OpFNegate %v3float %offset_p0_start
    %v1_pos = OpAdd %v3float %worldPos_0 %offset_p0_start_neg
    %v1_final_pos = OpCompositeConstruct %v4float %v1_pos.x %v1_pos.y %v1_pos.z %float_1 ; w = 1.0
    OpStore %ptr_gl_out_pos_0 %v1_final_pos
    OpStore %outColor %color_0
    OpEmitVertex

    ; Vertex 2: p1 + perp * thick_end, color1
    %offset_p1_end = OpFMul %v3float %perp_normalized_xy %float_0_005
    %v2_pos = OpAdd %v3float %worldPos_1 %offset_p1_end
    %v2_final_pos = OpCompositeConstruct %v4float %v2_pos.x %v2_pos.y %v2_pos.z %float_1 ; w = 1.0
    OpStore %ptr_gl_out_pos_0 %v2_final_pos
    OpStore %outColor %color_1
    OpEmitVertex

    ; Vertex 3: p1 - perp * thick_end, color1
    %offset_p1_end_neg = OpFNegate %v3float %offset_p1_end
    %v3_pos = OpAdd %v3float %worldPos_1 %offset_p1_end_neg
    %v3_final_pos = OpCompositeConstruct %v4float %v3_pos.x %v3_pos.y %v3_pos.z %float_1 ; w = 1.0
    OpStore %ptr_gl_out_pos_0 %v3_final_pos
    OpStore %outColor %color_1
    OpEmitVertex

    ; End primitive
    OpEndPrimitive

    OpReturn
OpEndFunction
