; SPIR-V Geometry Shader Assembly
; Minimal example: Pass-through geometry shader
; Input: Triangles
; Output: TriangleStrip
; Max Output Vertices: 3

OpCapability Shader
OpCapability Geometry

OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %in_pos %out_pos

; Execution Modes
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

; Decorate Inputs/Outputs
OpDecorate %in_pos BuiltIn Position
OpDecorate %out_pos BuiltIn Position

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_3 = OpConstant %int 3
%float_array_3 = OpTypeArray %v4float %int_3
%bool = OpTypeBool
%ptr_v4float = OpTypePointer Input %v4float
%void_fn = OpTypeFunction %void

; Variables
%in_pos = OpVariable %Input Input ; Input is array of 3 positions
%out_pos = OpVariable %Output Output ; Output is a single position

; Function
%main = OpFunction %void None %void_fn
%label = OpLabel

; Passthrough loop: 3 vertices
%i = OpVariable %Function %int
OpStore %i %int_0
OpBranch %loop

%loop = OpLabel
%i_val = OpLoad %int %i
%cond = OpSLessThan %bool %i_val %int_3
OpLoopMerge %exit %loop None
OpBranchConditional %cond %body %exit

%body = OpLabel
; Fetch Input[i]
%ptr = OpAccessChain %ptr_v4float %in_pos %i_val
%pos = OpLoad %v4float %ptr
OpStore %out_pos %pos
OpEmitVertex
OpBranch %continue

%continue = OpLabel
%new_i = OpIAdd %int %i_val %int_1
OpStore %i %new_i
OpBranch %loop

%exit = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
