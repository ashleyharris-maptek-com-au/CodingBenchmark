OpCapability Geometry
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               ; --- Decorations ---
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 0
               OpDecorate %outColor Location 0

               ; --- Types ---
%void        = OpTypeVoid
%float       = OpTypeFloat 32
%v4float     = OpTypeVector %float 4
%v3float     = OpTypeVector %float 3
%uint        = OpTypeInt 32 0
%int         = OpTypeInt 32 1
%uint_1      = OpConstant %uint 1
%uint_4      = OpConstant %uint 4
%int_0       = OpConstant %int 0
%float_0     = OpConstant %float 0.0
%float_1     = OpConstant %float 1.0
%float_h     = OpConstant %float 0.04
%float_neg_h = OpConstant %float -0.04

               ; --- Input/Output Structs ---
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in   = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in       = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
%gl_out      = OpVariable %ptr_out_pv Output

%arr_v3_in   = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%inColor     = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3  = OpTypePointer Output %v3float
%outColor    = OpVariable %ptr_out_v3 Output

%func        = OpTypeFunction %void

               ; --- Main Function ---
%main        = OpFunction %void None %func
%label       = OpLabel

               ; Load Input position and color
%ptr_in_pv0  = OpAccessChain %ptr_in_arr_pv %gl_in %int_0
%ptr_in_v4   = OpTypePointer Input %v4float
%ptr_pv0_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos         = OpLoad %v4float %ptr_pv0_pos
%x           = OpCompositeExtract %float %pos 0
%y           = OpCompositeExtract %float %pos 1
%z           = OpCompositeExtract %float %pos 2
%w           = OpCompositeExtract %float %pos 3

%ptr_in_c0   = OpAccessChain %ptr_in_arr_v3 %inColor %int_0
%color       = OpLoad %v3float %ptr_in_c0

               ; Helper to emit a vertex
%emit = OpLabel ; Just a label to mark start
               ; Vertex 1: (x-h, y-h)
%x1          = OpFAdd %float %x %float_neg_h
%y1          = OpFAdd %float %y %float_neg_h
%pos1        = OpCompositeConstruct %v4float %x1 %y1 %z %w
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %ptr_out_pos %pos1
               OpStore %outColor %color
               OpEmitVertex

               ; Vertex 2: (x+h, y-h)
%x2          = OpFAdd %float %x %float_h
%y2          = OpFAdd %float %y %float_neg_h
%pos2        = OpCompositeConstruct %v4float %x2 %y2 %z %w
               OpStore %ptr_out_pos %pos2
               OpStore %outColor %color
               OpEmitVertex

               ; Vertex 3: (x-h, y+h)
%x3          = OpFAdd %float %x %float_neg_h
%y3          = OpFAdd %float %y %float_h
%pos3        = OpCompositeConstruct %v4float %x3 %y3 %z %w
               OpStore %ptr_out_pos %pos3
               OpStore %outColor %color
               OpEmitVertex

               ; Vertex 4: (x+h, y+h)
%x4          = OpFAdd %float %x %float_h
%y4          = OpFAdd %float %y %float_h
%pos4        = OpCompositeConstruct %v4float %x4 %y4 %z %w
               OpStore %ptr_out_pos %pos4
               OpStore %outColor %color
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
