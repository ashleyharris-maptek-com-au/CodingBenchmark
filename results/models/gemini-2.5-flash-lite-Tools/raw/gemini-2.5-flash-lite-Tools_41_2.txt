; SPIR-V
; Version: 1.0
; Generator: Unknown

OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %outColor
OpExecutionMode %main OriginUpperLeft

; Input variables (from vertex shader)
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4

; Output variable
OpDecorate %outColor Location 0

; Uniform buffer
OpDecorate %UBOBlock BufferBlock
OpDecorate %ub 0 Binding 0
OpDecorate %ub DescriptorSet 0
OpDecorate %ub Offset 0
OpDecorate %ub ColMajor
OpDecorate %ub MatrixStride 16
OpDecorate %ub Stride 16

OpMemberDecorate %UBOBlock 0 Offset 0
OpMemberDecorate %UBOBlock 0 ColMajor
OpMemberDecorate %UBOBlock 0 MatrixStride 16
OpMemberDecorate %UBOBlock 1 Offset 64
OpMemberDecorate %UBOBlock 1 ColMajor
OpMemberDecorate %UBOBlock 1 MatrixStride 16
OpMemberDecorate %UBOBlock 2 Offset 128
OpMemberDecorate %UBOBlock 2 ColMajor
OpMemberDecorate %UBOBlock 2 MatrixStride 16
OpMemberDecorate %UBOBlock 3 Offset 192
OpMemberDecorate %UBOBlock 4 Offset 208
OpMemberDecorate %UBOBlock 5 Offset 224

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4float = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float
%UBOBlock = OpTypePointer Uniform %UBO
%ub = OpConstantNull %UBOBlock

%void = OpTypeVoid
%func = OpTypeFunction %void

%worldPos = OpTypeVector %float 3
%worldPos = OpTypePointer Input %worldPos
%worldPos = OpVariable %worldPos

%normal = OpTypeVector %float 3
%normal = OpTypePointer Input %normal
%normal = OpVariable %normal

%uv = OpTypeVector %float 2
%uv = OpTypePointer Input %uv
%uv = OpVariable %uv

%tangent = OpTypeVector %float 3
%tangent = OpTypePointer Input %tangent
%tangent = OpVariable %tangent

%color = OpTypeVector %float 3
%color = OpTypePointer Input %color
%color = OpVariable %color

%outColor = OpTypeVector %float 4
%outColor = OpTypePointer Output %outColor
%outColor = OpVariable %outColor

%main = OpFunction %void %func
%label1 = OpLabel
%uniform_buffer = OpLoad %UBO %ub
%uv_var = OpLoad %uv %uv
%uv_x = OpCompositeExtract %float %uv_var 0
%uv_y = OpCompositeExtract %float %uv_var 1
%zero_float = OpConstant %float 0.0
%one_float = OpConstant %float 1.0
%color_vec = OpCompositeConstruct %v4float %uv_x %uv_y %zero_float %one_float
OpStore %outColor %color_vec
OpReturn
OpFunctionEnd
