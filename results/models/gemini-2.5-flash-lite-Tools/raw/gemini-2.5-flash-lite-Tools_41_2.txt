; SPIR-V
; Version: 1.0
; Generator: LLM Bench

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %uv
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0
OpDecorate %uv Location 2

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%func = OpTypeFunction %void

; Inputs/Outputs
%uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

; Function
%main = OpFunction %void None %func
%label = OpLabel

; Read UV
%uv_val = OpLoad %vec2 %uv
%u = OpCompositeExtract %float %uv_val 0
%v = OpCompositeExtract %float %uv_val 1

; Construct Color: RGBA (u, v, 0, 1)
%color = OpCompositeConstruct %vec4 %u %v %float_0 %float_1
OpStore %outColor %color

OpReturn
OpFunctionEnd
