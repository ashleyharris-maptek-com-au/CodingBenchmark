import math

# Persistent state for the autopilot
autopilot_state_persistent = {}

class PIDController:
    def __init__(self, kp, ki, kd, output_limits=(-1.0, 1.0), integral_limits=(-1000.0, 1000.0)):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.output_limits = output_limits
        self.integral_limits = integral_limits
        self._integral = 0.0
        self._last_error = 0.0
        self._last_output = 0.0 # Store last output for rate limiting if needed

    def update(self, error, dt):
        # Proportional term
        p_term = self.kp * error

        # Integral term
        self._integral += error * dt
        self._integral = max(self.integral_limits[0], min(self._integral, self.integral_limits[1]))
        i_term = self.ki * self._integral

        # Derivative term
        d_term = self.kd * (error - self._last_error) / dt if dt > 0 else 0.0
        self._last_error = error

        output = p_term + i_term + d_term
        output = max(self.output_limits[0], min(output, self.output_limits[1]))
        
        self._last_output = output
        return output

    def reset(self):
        self._integral = 0.0
        self._last_error = 0.0
        self._last_output = 0.0

def _get_reliable_value(state: dict, sensor_prefix: str, threshold_factor: float = 0.1, default_value: float = 0.0) -> float:
    """
    Implements a 2-out-of-3 voting system for redundant sensors.
    If sensors are valid and consistent, returns the median.
    If one sensor is an outlier, returns the average of the other two.
    If all diverge, returns a default value or raises an alarm.
    """
    values = []
    for i in range(1, 4):
        valid_key = f"{sensor_prefix}_{i}_valid"
        value_key = f"{sensor_prefix}_{i}"
        
        if state.get(valid_key, False): # Assume invalid if key not present or explicitly false
            val = state.get(value_key)
            if val is not None:
                values.append(val)

    if not values:
        return default_value
    
    if len(values) == 1:
        return values[0]

    values.sort()

    if len(values) == 2:
        # If only two valid, return their average
        return sum(values) / 2.0
    
    # Three values - implement 2-out-of-3 voting
    if len(values) == 3:
        # Check consistency: median vs extremes
        median = values[1]
        diff1 = abs(values[0] - median)
        diff3 = abs(values[2] - median)
        
        # Determine a reasonable threshold based on the median value.
        # This threshold should be adjusted based on sensor type and expected variance.
        # For now, using a percentage of the median or a fixed small value.
        consistency_threshold = max(abs(median) * threshold_factor, 1.0) # e.g., 10% of median or 1.0 absolute

        if diff1 <= consistency_threshold and diff3 <= consistency_threshold:
            # All three are consistent, return median
            return median
        elif diff1 > consistency_threshold and diff3 <= consistency_threshold:
            # First value is an outlier, average the other two
            return (values[1] + values[2]) / 2.0
        elif diff1 <= consistency_threshold and diff3 > consistency_threshold:
            # Third value is an outlier, average the first two
            return (values[0] + values[1]) / 2.0
        else:
            # All three diverge or two are outliers - indicating a more serious issue.
            # For now, return median as a best guess, but this needs more robust handling
            # in a real system (e.g., triggering a failure mode, using historical data).
            # Could also log a warning here.
            return median
    
    return default_value # Should not reach here if logic is sound but as a safeguard


def _fuse_altitude(state: dict) -> float:
    # Prioritize Radar Altimeter below 2500ft AGL
    radar_alts = []
    for i in range(1, 4):
        if state.get(f"radar_alt_{i}_valid", False):
            val = state.get(f"radar_alt_{i}_ft")
            if val is not None:
                radar_alts.append(val)
    
    if radar_alts:
        avg_radar_alt = sum(radar_alts) / len(radar_alts)
        if avg_radar_alt < 2500: # Threshold for AGL
            return avg_radar_alt

    # Fallback to Barometric Altimeter
    baro_alt = _get_reliable_value(state, "baro_alt", threshold_factor=0.01) # Baro usually more precise
    if baro_alt != 0.0: # Check if it's not the default
        return baro_alt
    
    # Fallback to GPS Altitude
    gps_alts = []
    for i in range(1, 4):
        if state.get(f"gps_{i}_valid", False):
            val = state.get(f"gps_{i}_alt_ft")
            if val is not None:
                gps_alts.append(val)
    
    if gps_alts:
        return sum(gps_alts) / len(gps_alts)

    # If all fail, return 0 or last known good altitude (needs persistent state)
    # For now, return a default for simulation.
    return 0.0


def _fuse_airspeed(state: dict) -> float:
    # Prioritize IAS from ADCs
    ias = _get_reliable_value(state, "ias", threshold_factor=0.05)
    if ias != 0.0:
        return ias
    
    # Fallback to TAS if IAS is unavailable (requires temperature/pressure conversion for actual use, but here we'll just take TAS)
    tas = _get_reliable_value(state, "tas", threshold_factor=0.05)
    if tas != 0.0:
        return tas
    
    # Fallback to Ground Speed if air data is completely lost (very unreliable for control)
    gs = _get_reliable_value(state, "gps_gs", threshold_factor=0.05)
    if gs != 0.0:
        return gs
    
    return 0.0

def _fuse_mach(state: dict) -> float:
    mach = _get_reliable_value(state, "mach", threshold_factor=0.02)
    return mach

def _fuse_pitch_roll_heading(state: dict) -> tuple:
    # Prioritize IRS for attitude and heading
    irs_pitch = _get_reliable_value(state, "irs_pitch", threshold_factor=0.01)
    irs_roll = _get_reliable_value(state, "irs_roll", threshold_factor=0.01)
    irs_heading = _get_reliable_value(state, "irs_heading", threshold_factor=0.01)

    # TODO: Add logic to cross-check with gyros and accelerometers, especially if IRS fails.
    # For now, relying heavily on IRS.
    
    return irs_pitch, irs_roll, irs_heading

def _fuse_vertical_speed(state: dict) -> float:
    # Prioritize IRS vertical speed
    irs_vs = _get_reliable_value(state, "irs_vs", threshold_factor=0.05, default_value=state.get("vertical_speed_fpm", 0.0))
    # cross check with baro_alt change
    
    # TODO: Calculate VS from barometric altitude change if IRS is unreliable
    return irs_vs


def autopilot_step(state: dict, dt: float) -> dict:
    # Initialize persistent state on first call
    if 'initialized' not in autopilot_state_persistent:
        autopilot_state_persistent['initialized'] = True
        autopilot_state_persistent['altitude_pid'] = PIDController(kp=0.1, ki=0.01, kd=0.2, output_limits=(-1.0, 1.0)) # Elevator
        autopilot_state_persistent['airspeed_pid'] = PIDController(kp=0.01, ki=0.001, kd=0.05, output_limits=(0.0, 1.0)) # Throttle
        autopilot_state_persistent['heading_pid'] = PIDController(kp=0.05, ki=0.001, kd=0.1, output_limits=(-1.0, 1.0)) # Aileron
        autopilot_state_persistent['vs_pid'] = PIDController(kp=0.05, ki=0.01, kd=0.1, output_limits=(-1.0, 1.0)) # Elevator
        autopilot_state_persistent['elevator_trim_pid'] = PIDController(kp=0.001, ki=0.0001, kd=0.0005, output_limits=(-0.01, 0.01), integral_limits=(-1.0, 1.0)) # Elevator trim adjustment
        autopilot_state_persistent['current_elevator_trim'] = 0.0
        autopilot_state_persistent['last_fused_altitude'] = state.get('baro_alt_1_ft', 0.0) # Initial value

    # Control outputs dictionary
    control_outputs = {}

    # 1. Sensor Fusion
    fused_altitude = _fuse_altitude(state)
    fused_ias = _fuse_airspeed(state)
    fused_mach = _fuse_mach(state)
    fused_pitch, fused_roll, fused_heading = _fuse_pitch_roll_heading(state)
    fused_vs = _fuse_vertical_speed(state)
    fused_aoa = _get_reliable_value(state, "aoa", threshold_factor=0.1) # Assuming AOA has 3 sensors


    # 2. Flight Envelope Protection (Overrides if necessary)
    # Vne Protection
    VNE_IAS = 365 # KIAS
    VNE_MACH = 0.82
    VNE_BUFFER_IAS = 10 # KIAS
    VNE_BUFFER_MACH = 0.02

    if fused_ias > (VNE_IAS - VNE_BUFFER_IAS) or fused_mach > (VNE_MACH - VNE_BUFFER_MACH):
        control_outputs['throttle_1'] = 0.0 # Reduce thrust
        control_outputs['throttle_2'] = 0.0
        # Apply nose up to increase drag (small elevator change relative to trim)
        control_outputs['elevator'] = (autopilot_state_persistent['altitude_pid'].output_limits[1] * 0.2) # 20% of max nose up
        # We might want to disengage AP modes here. For now, just overriding controls.

    # Stall Protection (AoA)
    STALL_AOA_THRESHOLD = 12.0 # degrees, needs tuning
    AOA_BUFFER = 1.0

    if fused_aoa > (STALL_AOA_THRESHOLD - AOA_BUFFER):
        control_outputs['elevator'] = autopilot_state_persistent['altitude_pid'].output_limits[0] * 0.5 # 50% of max nose down
        control_outputs['throttle_1'] = 1.0 # Max thrust
        control_outputs['throttle_2'] = 1.0
        # Disengage AP modes.

    # If envelope protection engaged, skip normal control logic for affected controls
    if 'elevator' in control_outputs or 'throttle_1' in control_outputs:
        return control_outputs # Return early with emergency controls

    # 3. Autopilot Control Logic
    target_alt_ft = state.get('target_alt_ft', fused_altitude)
    target_ias_kt = state.get('target_ias_kt', fused_ias)
    target_hdg_deg = state.get('target_hdg_deg', fused_heading)
    target_vs_fpm = state.get('target_vs_fpm', fused_vs)

    ap_engaged = state.get('ap_engaged', False)
    at_engaged = state.get('at_engaged', False)

    # Altitude Hold
    if ap_engaged and target_alt_ft is not None:
        alt_error = target_alt_ft - fused_altitude
        # If we are far from target alt, prioritize vertical speed to reach it
        # Otherwise, maintain target altitude
        if abs(alt_error) > 100: # If error is significant, calculate a target VS
            target_vs_from_alt = alt_error * 10 # proportional to alt error
            target_vs_fpm = max(-3000, min(3000, target_vs_from_alt)) # Limit VS
            
            vs_error = target_vs_fpm - fused_vs
            elevator_cmd = autopilot_state_persistent['vs_pid'].update(vs_error, dt)
            control_outputs['elevator'] = elevator_cmd
        else: # Fine-tune altitude
            elevator_cmd = autopilot_state_persistent['altitude_pid'].update(alt_error, dt)
            control_outputs['elevator'] = elevator_cmd

    # Airspeed Hold (Autothrottle)
    if at_engaged and target_ias_kt is not None:
        ias_error = target_ias_kt - fused_ias
        throttle_cmd = autopilot_state_persistent['airspeed_pid'].update(ias_error, dt)
        control_outputs['throttle_1'] = throttle_cmd
        control_outputs['throttle_2'] = throttle_cmd

    # Heading Hold
    if ap_engaged and target_hdg_deg is not None:
        hdg_error = target_hdg_deg - fused_heading
        # Normalize heading error to -180 to +180 degrees
        hdg_error = (hdg_error + 180) % 360 - 180
        
        aileron_cmd = autopilot_state_persistent['heading_pid'].update(hdg_error, dt)
        control_outputs['aileron'] = aileron_cmd
        # Basic rudder for turn coordination (proportional to aileron)
        control_outputs['rudder'] = aileron_cmd * 0.5 # Needs proper tuning

    # Elevator Trim (adjusts based on elevator command)
    if 'elevator' in control_outputs:
        trim_adjustment = autopilot_state_persistent['elevator_trim_pid'].update(control_outputs['elevator'], dt)
        autopilot_state_persistent['current_elevator_trim'] += trim_adjustment
        autopilot_state_persistent['current_elevator_trim'] = max(-1.0, min(autopilot_state_persistent['current_elevator_trim'], 1.0))
        control_outputs['elevator_trim'] = autopilot_state_persistent['current_elevator_trim']
        # Recalculate elevator relative to new trim
        control_outputs['elevator'] -= autopilot_state_persistent['current_elevator_trim']


    # Auxiliary Controls (simple logic for now)
    # Gear
    if state.get('gear_handle_down', False) or (fused_altitude < 1500 and fused_ias < 200): # Auto-deploy below 1500ft and 200kts
        control_outputs['gear'] = 1
    else:
        control_outputs['gear'] = 0

    # Flaps (simple logic based on airspeed for now)
    # This would typically be more complex, tied to approach phase etc.
    if fused_ias < 180:
        control_outputs['flaps'] = 10 # Flaps 10
    if fused_ias < 150:
        control_outputs['flaps'] = 20 # Flaps 20
    if fused_ias < 130:
        control_outputs['flaps'] = 30 # Flaps 30 (Landing)
    else:
        control_outputs['flaps'] = 0

    # Speedbrake (deploy if overspeeding significantly or rapid descent needed)
    if fused_ias > (target_ias_kt + 20) and not at_engaged: # If overspeeding and AT not engaged
        control_outputs['speedbrake'] = 0.5 # Partial deployment
    elif fused_altitude > (target_alt_ft + 500) and not ap_engaged: # If too high and AP not engaged
        control_outputs['speedbrake'] = 0.5
    else:
        control_outputs['speedbrake'] = 0.0
    
    autopilot_state_persistent['last_fused_altitude'] = fused_altitude
    return control_outputs
