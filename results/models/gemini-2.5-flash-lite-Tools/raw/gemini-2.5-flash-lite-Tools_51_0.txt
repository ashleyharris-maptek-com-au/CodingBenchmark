class CStarController:
    """Simplified C* Law implementation."""
    def __init__(self, kp=1.0, ki=0.01, kd=0.5):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.integral = 0
        self.prev_error = 0

    def compute(self, target_c_star, current_nz, current_q, dt):
        # A320 G-Load Protection (example: limit to 2.5g)
        target_c_star = min(target_c_star, 2.5)
        
        # Simplified C* error: C* = nz + Vco * q
        current_c_star = current_nz + 210.0 * current_q
        error = target_c_star - current_c_star
        
        self.integral += error * dt
        # Anti-windup
        self.integral = max(-1.0, min(1.0, self.integral))
        
        derivative = (error - self.prev_error) / dt
        
        output = self.kp * error + self.ki * self.integral + self.kd * derivative
        self.prev_error = error
        
        # Actuator Saturation
        return max(-1.0, min(1.0, output))
