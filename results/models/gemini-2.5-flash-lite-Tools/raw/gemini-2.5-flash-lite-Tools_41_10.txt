; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

; --- Decorations ---
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %color Location 4
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; --- Types ---
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%func = OpTypeFunction %void
%bool = OpTypeBool

; --- Constants ---
%f0 = OpConstant %float 0.0
%f0_1 = OpConstant %float 0.1
%f0_4 = OpConstant %float 0.4
%f0_6 = OpConstant %float 0.6
%f0_9 = OpConstant %float 0.9
%f1 = OpConstant %float 1.0
%f3_14159 = OpConstant %float 3.14159
%f16 = OpConstant %float 16.0
%i0 = OpConstant %int 0
%i3 = OpConstant %int 3

; --- Globals ---
%ubo = OpVariable %ptr_uniform_UBO Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

; --- Main Function ---
%main = OpFunction %void None %func
%label = OpLabel

; Get UV.y
%uv_ptr = OpAccessChain %ptr_input_vec2 %uv
%uv_val = OpLoad %vec2 %uv
%uv_y = OpCompositeExtract %float %uv_val 1

; stripe_val = sin(uv.y * 3.14159 * 16.0)
%factor = OpFMul %float %f3_14159 %f16
%angle = OpFMul %float %uv_y %factor
%sine = OpSin %float %angle

; Compare: sine > 0
%gt_zero = OpFOrdGreaterThan %bool %sine %f0

; Select Color
%red = OpCompositeConstruct %vec3 %f0_9 %f0_1 %f0_1
%blue = OpCompositeConstruct %vec3 %f0_1 %f0_1 %f0_9
%base_color = OpSelect %vec3 %gt_zero %red %blue

; Lambertian Shading
; Access lightPos
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %i3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

; Light dir
%pos_val = OpLoad %vec3 %worldPos
%light_dir = OpFSub %vec3 %lightPos3 %pos_val
%light_dir_norm = OpExtInst %vec3 %1 Normalize %light_dir

; Normal
%norm_val = OpLoad %vec3 %normal
%norm_norm = OpExtInst %vec3 %1 Normalize %norm_val

; Diffuse = max(dot(N, L), 0)
%dot = OpDot %float %norm_norm %light_dir_norm
%diffuse = OpExtInst %float %1 FMax %dot %f0

; Shading factor = 0.4 + 0.6 * diffuse
%term = OpFMul %float %f0_6 %diffuse
%shade = OpFAdd %float %f0_4 %term

; Final color = base_color * shade
%final_rgb = OpVectorTimesScalar %vec3 %base_color %shade

; Set outColor
%final_color = OpCompositeExtract %float %final_rgb 0
%final_color1 = OpCompositeExtract %float %final_rgb 1
%final_color2 = OpCompositeExtract %float %final_rgb 2
%final_rgba = OpCompositeConstruct %vec4 %final_color %final_color1 %final_color2 %f1
OpStore %outColor %final_rgba

OpReturn
OpFunctionEnd
