; SPIR-V Geometry Shader
; Input: InputTrianglesAdjacency (6 vertices)
; Output: TriangleStrip, Max 12 vertices

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %inColor

; --- Decorations ---
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 1
OpDecorate %outColor Location 0

; --- Types ---
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_6 = OpConstant %uint 6
%uint_12 = OpConstant %uint 12
%int_0 = OpConstant %int 0

; Input/Output structs
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %gl_PerVertex_in
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_v3_6 = OpTypePointer Input %arr_v3_6
%inColor = OpVariable %ptr_in_v3_6 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; --- Main ---
%main = OpFunction %void None %func
%entry = OpLabel

; --- Logic Scaffolding ---
; 1. Define edges by index: E1=(0,2), E2=(2,4), E3=(4,0)
; 2. Loop through edges:
;    a. Load P0, P1 and Color0, Color1
;    b. Dir = normalize(P1 - P0)
;    c. Perpendicular = cross(Dir, (0,0,1)) * thickness
;    d. Emit 4 vertices as triangle strip:
;       V0 = P0 + Perpendicular
;       V1 = P0 - Perpendicular
;       V2 = P1 + Perpendicular
;       V3 = P1 - Perpendicular
;    e. OpEmitVertex for each, updating gl_out.Position and outColor
;    f. OpEndPrimitive

OpReturn
OpFunctionEnd

; --- Execution Modes ---
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1
