; SPIR-V
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %uv %tangent %color %Ubo
OpExecutionMode %main OriginUpperLeft

; Input variables
OpDecorate %worldPos Input
OpDecorate %worldPos Location 0
OpDecorate %normal Input
OpDecorate %normal Location 1
OpDecorate %uv Input
OpDecorate %uv Location 2
OpDecorate %tangent Input
OpDecorate %tangent Location 3
OpDecorate %color Input
OpDecorate %color Location 4

; Output variable
OpDecorate %outColor Output
OpDecorate %outColor Location 0

; Uniform buffer
OpDecorate %Ubo Binding 0
OpDecorate %Ubo DescriptorSet 0
OpDecorate %Ubo Block
OpMemberDecorate %type_Ubo 0 Offset 0
OpMemberDecorate %type_Ubo 1 Offset 64
OpMemberDecorate %type_Ubo 2 Offset 128
OpMemberDecorate %type_Ubo 3 Offset 192
OpMemberDecorate %type_Ubo 4 Offset 208
OpMemberDecorate %type_Ubo 5 Offset 224
OpDecorate %type_Ubo Block

%type_f32 = OpTypeFloat 32
%type_f32_vec3 = OpTypeVector %type_f32 3
%type_f32_vec4 = OpTypeVector %type_f32 4
%type_mat4 = OpTypeMatrix %type_f32 4

%type_Ubo = OpTypeStruct %mat4 %mat4 %mat4 %type_f32_vec4 %type_f32_vec4 %type_f32_vec4
%type_ptr_Ubo = OpTypePointer Uniform %type_Ubo
%Ubo = OpVariable %type_ptr_Ubo Uniform

; Function definitions
%main = OpFunction %type_void None %void
%label_begin = OpLabel
  %worldPos_load = OpLoad %type_f32_vec3 %worldPos
  %normal_load = OpLoad %type_f32_vec3 %normal
  %uv_load = OpLoad %type_f32_vec2 %uv
  %tangent_load = OpLoad %type_f32_vec3 %tangent
  %color_load = OpLoad %type_f32_vec3 %color

  %ubo_ptr_val = OpLoad %type_Ubo %Ubo
  %params = OpCompositeExtract %type_f32_vec4 %ubo_ptr_val 5

  %lightPos = OpCompositeExtract %type_f32_vec4 %ubo_ptr_val 3
  %cameraPos = OpCompositeExtract %type_f32_vec4 %ubo_ptr_val 4

  ; Base color (0.2, 0.3, 0.8)
  %baseColor = OpConstantComposite %type_f32_vec3 %type_f32 0.2 %type_f32 0.3 %type_f32 0.8
  
  ; Diffuse calculation (N dot L)
  %lightDir = OpLoad %type_f33_vec4 %lightPos 
  %lightDir_xyz = OpCompositeExtract %type_f32_vec3 %lightDir 0
  %normal_norm = OpCompositeNormalise %type_f32_vec3 %normal_load
  %lightDir_norm = OpCompositeNormalise %type_f32_vec3 %lightDir_xyz
  %diffuse = OpDot %type_f32 %normal_norm %lightDir_norm
  %diffuse_clamped = OpMax %type_f32 %diffuse %type_f32 0.0
  
  ; Modulate base color by (0.3 + 0.5*diffuse)
  %diffuse_factor = OpFAdd %type_f32 %type_f32 0.3 (%type_f32 0.5 * %diffuse_clamped)
  %diffuse_color = %baseColor * %diffuse_factor

  ; Rim lighting (Fresnel)
  ; N dot V: V is vector from fragment to camera. 
  ; Camera position is in UBO.worldPos.xyz
  %viewDir_raw = %cameraPos - %worldPos_load 
  %viewDir = OpCompositeNormalise %type_f32_vec3 %viewDir_raw
  %NdotV = OpDot %type_f32 %normal_norm %viewDir
  %rim_factor = OpPow %type_f32 (%type_f32 1.0 - OpMax %type_f32 %NdotV %type_f32 0.0) %type_f32 3.0
  
  ; Rim tint (1.0, 0.8, 0.5)
  %rimTint = OpConstantComposite %type_f32_vec3 %type_f32 1.0 %type_f32 0.8 %type_f32 0.5
  %rim_color = %rimTint * %rim_factor

  ; Final color: diffuse_color + rim_color
  %final_color_rgb = %diffuse_color + %rim_color
  %final_color_alpha = OpConstant %type_f32 1.0
  %outColor_val = OpCompositeComposite %type_f32_vec4 %final_color_rgb %final_color_alpha

  OpStore %outColor %outColor_val

  OpReturn
OpFunctionEnd

; Type definitions
%void = OpTypeVoid
%type_f32_vec2 = OpTypeVector %type_f32 2
%type_ptr_Output_vec4 = OpTypePointer Output %type_f32_vec4
%outColor = OpVariable %type_ptr_Output_vec4 Output
%type_ptr_Input_vec3 = OpTypePointer Input %type_f32_vec3
%worldPos = OpVariable %type_ptr_Input_vec3 Input
%normal = OpVariable %type_ptr_Input_vec3 Input
%type_ptr_Input_vec2 = OpTypePointer Input %type_f32_vec2
%uv = OpVariable %type_ptr_Input_vec2 Input
%tangent = OpVariable %type_ptr_Input_vec3 Input
%color = OpVariable %type_ptr_Input_vec3 Input
