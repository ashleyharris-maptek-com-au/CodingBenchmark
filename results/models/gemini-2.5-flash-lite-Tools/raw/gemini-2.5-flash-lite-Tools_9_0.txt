using System;
using System.Collections.Generic;
using System.Linq;

public class Cut
{
    public int OriginalIndex { get; set; }
    public int Length { get; set; }
}

public class Stock
{
    public int Id { get; set; }
    public int RemainingLength { get; set; }
    public List<int> CutIndices { get; set; }

    public Stock(int id, int stockLength)
    {
        Id = id;
        RemainingLength = stockLength;
        CutIndices = new List<int>();
    }

    public bool CanFit(int cutLength)
    {
        return RemainingLength >= cutLength;
    }

    public void AddCut(Cut cut)
    {
        if (CanFit(cut.Length))
        {
            RemainingLength -= cut.Length;
            CutIndices.Add(cut.OriginalIndex);
        }
        else
        {
            throw new InvalidOperationException("Cut does not fit in the stock.");
        }
    }
}

public class Program
{
    public static void Main(string[] args)
    {
        // Read input
        string[] firstLine = Console.ReadLine().Split(' ');
        int numCuts = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        string[] secondLine = Console.ReadLine().Split(' ');
        List<Cut> cuts = new List<Cut>();
        for (int i = 0; i < numCuts; i++)
        {
            cuts.Add(new Cut { OriginalIndex = i, Length = int.Parse(secondLine[i]) });
        }

        // Sort cuts by length in descending order (First Fit Decreasing)
        cuts = cuts.OrderByDescending(c => c.Length).ToList();

        // First Fit Algorithm
        List<Stock> stocks = new List<Stock>();
        int currentStockId = 0;

        foreach (var cut in cuts)
        {
            bool placed = false;
            // Try to fit into existing stocks
            for (int i = 0; i < stocks.Count; i++)
            {
                if (stocks[i].CanFit(cut.Length))
                {
                    stocks[i].AddCut(cut);
                    placed = true;
                    break; // First Fit
                }
            }

            // If not placed, create a new stock
            if (!placed)
            {
                currentStockId++;
                Stock newStock = new Stock(currentStockId, stockLength);
                newStock.AddCut(cut);
                stocks.Add(newStock);
            }
        }

        // Output results
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            // The problem asks for space-separated cut indices assigned to that stock
            // The order of indices within a stock doesn't seem to be specified,
            // but let's sort them for deterministic output.
            stock.CutIndices.Sort();
            Console.WriteLine(string.Join(" ", stock.CutIndices));
        }
    }
}
