using System;
using System.Collections.Generic;
using System.Linq;

class QAPSolver
{
    static void Main(string[] args)
    {
        // Read input
        string[] line1 = Console.ReadLine().Split(new[] { ' ', '	' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(line1[0]);

        int[,] flow = new int[N, N];
        for (int i = 0; i < N; i++)
        {
            string[] row = Console.ReadLine().Split(new[] { ' ', '	' }, StringSplitOptions.RemoveEmptyEntries);
            for (int j = 0; j < N; j++)
            {
                flow[i, j] = int.Parse(row[j]);
            }
        }

        int[,] dist = new int[N, N];
        for (int i = 0; i < N; i++)
        {
            string[] row = Console.ReadLine().Split(new[] { ' ', '	' }, StringSplitOptions.RemoveEmptyEntries);
            for (int j = 0; j < N; j++)
            {
                dist[i, j] = int.Parse(row[j]);
            }
        }

        // Initial Solution (identity permutation)
        int[] p = new int[N];
        for (int i = 0; i < N; i++) p[i] = i;

        // Simulated Annealing
        Random rand = new Random();
        long currentCost = CalculateFullCost(N, flow, dist, p);
        
        double temp = 1000.0;
        double coolingRate = 0.9995;

        for (int iter = 0; iter < 1000000; iter++)
        {
            int i = rand.Next(N);
            int j = rand.Next(N);
            if (i == j) continue;

            long delta = CalculateDelta(N, flow, dist, p, i, j);

            if (delta < 0 || Math.Exp(-delta / temp) > rand.NextDouble())
            {
                // Accept swap
                int temp_p = p[i];
                p[i] = p[j];
                p[j] = temp_p;
                currentCost += delta;
            }

            temp *= coolingRate;
            if (temp < 0.001) break;
        }

        // Output
        Console.WriteLine(currentCost);
        Console.WriteLine(string.Join(" ", p));
    }

    static long CalculateFullCost(int N, int[,] flow, int[,] dist, int[] p)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                cost += (long)flow[i, j] * dist[p[i], p[j]];
            }
        }
        return cost;
    }

    static long CalculateDelta(int N, int[,] flow, int[,] dist, int[] p, int r, int s)
    {
        // Calculate the change in cost if we swap locations of facilities r and s.
        // Location of facility r is p[r], location of facility s is p[s].
        // After swap, facility r is at p[s], facility s is at p[r].

        long delta = 0;

        // Contribution of r and s themselves
        delta += (long)flow[r, r] * (dist[p[s], p[s]] - dist[p[r], p[r]]);
        delta += (long)flow[r, s] * (dist[p[s], p[r]] - dist[p[r], p[s]]);
        delta += (long)flow[s, r] * (dist[p[r], p[s]] - dist[p[s], p[r]]);
        delta += (long)flow[s, s] * (dist[p[r], p[r]] - dist[p[s], p[s]]);

        for (int k = 0; k < N; k++)
        {
            if (k == r || k == s) continue;

            delta += (long)flow[k, r] * (dist[p[k], p[s]] - dist[p[k], p[r]]);
            delta += (long)flow[k, s] * (dist[p[k], p[r]] - dist[p[k], p[s]]);
            delta += (long)flow[r, k] * (dist[p[s], p[k]] - dist[p[r], p[k]]);
            delta += (long)flow[s, k] * (dist[p[r], p[k]] - dist[p[s], p[k]]);
        }

        return delta;
    }
}
