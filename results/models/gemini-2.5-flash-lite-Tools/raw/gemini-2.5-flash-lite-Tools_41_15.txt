; SPIR-V
OpCapability Shader
OpMemoryModel Logical GLSL450

; Entry point and execution mode
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %ubo
OpExecutionMode %main OriginUpperLeft

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ubo_ptr = OpTypePointer Uniform %ubo_struct
%float_ptr_in = OpTypePointer Input %float
%vec3_ptr_in = OpTypePointer Input %vec3
%vec4_ptr_in = OpTypePointer Input %vec4
%vec4_ptr_out = OpTypePointer Output %vec4

; Constants
%int = OpTypeInt 32
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_3 = OpConstant %int 3 ; index for lightPos in UBO
%int_5 = OpConstant %int 5 ; index for params in UBO

%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_1_0 = OpConstant %float 1.0

%vec3_coolColor = OpConstantComposite %vec3 %float_0_2 %float_0_2 %float_0_75
%vec3_warmColor = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_4
%vec3_one = OpConstantComposite %vec3 %float_1_0 %float_1_0 %float_1_0
%vec4_one_alpha = OpConstantComposite %vec4 %float_1_0 %float_1_0 %float_1_0 %float_1_0 ; For output alpha

; GLSL built-in extension
%glsl_module = OpExtInstImport "GLSL.std.450"

; Global variables (Inputs and Outputs)
%outColor = OpVariable Output %vec4_ptr_out
%worldPos = OpVariable Input %vec3_ptr_in
%normal = OpVariable Input %vec3_ptr_in
%ubo = OpVariable Uniform %ubo_ptr

; Main function
%main = OpFunction %void "main"
  %label_entry = OpLabel

  ; Load inputs
  %worldPos_val = OpLoad %vec3 %worldPos
  %normal_val = OpLoad %vec3 %normal

  ; Load UBO members
  %lightPos_ptr = OpAccessChain %vec4_ptr_in %ubo %int_3
  %lightPos_val = OpLoad %vec4 %lightPos_ptr
  %params_ptr = OpAccessChain %vec4_ptr_in %ubo %int_5
  %params_val = OpLoad %vec4 %params_ptr

  ; Normalize normal
  %normalized_normal = OpExtInst %vec3 %glsl_module OpNormalize %normal_val

  ; Calculate light direction (L)
  %lightPos_xyz = OpVectorShuffle %vec3 %lightPos_val %lightPos_val %int_0 %int_1 %int_2 ; Extract xyz from lightPos
  %lightDir = OpVector opSub %vec3 %lightPos_xyz %worldPos_val
  %normalized_lightDir = OpExtInst %vec3 %glsl_module OpNormalize %lightDir

  ; Calculate NdotL
  %NdotL = OpDot %float %normalized_normal %normalized_lightDir

  ; Calculate t = dot(N,L) * 0.5 + 0.5
  %t_term1 = OpFMul %float %NdotL %float_0_5
  %t = OpFAdd %float %t_term1 %float_0_5

  ; Calculate diffuse color: coolColor * (1.0 - t) + warmColor * t
  %one_minus_t = OpFSub %float %float_1_0 %t
  %cool_contrib = OpFMul %vec3 %vec3_coolColor %one_minus_t
  %warm_contrib = OpFMul %vec3 %vec3_warmColor %t
  %diffuse_color = OpFAdd %vec3 %cool_contrib %warm_contrib

  ; Calculate ambient term: ambientLight * params.y
  %ambientLight = %vec3_one ; Assuming white ambient light
  %Ka = OpCompositeExtract %float %params_val %int_1 ; Extract y component of params
  %ambient_term = OpFMul %vec3 %ambientLight %Ka

  ; Combine ambient and diffuse
  %final_rgb = OpFAdd %vec3 %ambient_term %diffuse_color

  ; Construct output color (RGBA) with alpha = 1.0
  %outColor_val = OpCompositeConstruct %vec4 %final_rgb %float_1_0
  OpStore %outColor %outColor_val

  OpReturn
  OpFunctionEnd
