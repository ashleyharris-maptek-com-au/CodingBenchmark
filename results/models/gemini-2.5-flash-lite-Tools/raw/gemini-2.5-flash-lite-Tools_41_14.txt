OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0

OpDecorate %normal Location 1

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_uniform_float = OpTypePointer Uniform %float

%func = OpTypeFunction %void

; Variables
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%tangent = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_UBO Uniform

; Constants
%int0 = OpConstant %int 0
%int1 = OpConstant %int 1
%int2 = OpConstant %int 2
%int3 = OpConstant %int 3
%int4 = OpConstant %int 4
%int5 = OpConstant %int 5
%float0 = OpConstant %float 0.0
%float1 = OpConstant %float 1.0
%float0_3 = OpConstant %float 0.3
%float0_7 = OpConstant %float 0.7
%float0_8 = OpConstant %float 0.8
%float0_2 = OpConstant %float 0.2
%float1_5 = OpConstant %float 1.5
%float3_0 = OpConstant %float 3.0

%main = OpFunction %void None %func
%label = OpLabel

; Load data
%val_worldPos = OpLoad %vec3 %worldPos
%val_normal = OpLoad %vec3 %normal
%val_cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int4
%val_cameraPos = OpLoad %vec4 %val_cameraPos_ptr
%val_lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int3
%val_lightPos = OpLoad %vec4 %val_lightPos_ptr

; Lighting
%norm_normal = OpExtInst %vec3 %1 Normalize %val_normal
%vec_to_light = OpVectorShuffle %vec3 %val_lightPos %val_lightPos 0 1 2
%dir_to_light = OpFSub %vec3 %vec_to_light %val_worldPos
%norm_dir_to_light = OpExtInst %vec3 %1 Normalize %dir_to_light
%diffuse = OpDot %float %norm_normal %norm_dir_to_light
%clamped_diffuse = OpExtInst %float %1 FClamp %diffuse %float0 %float1
%lambert = OpFMul %float %float0_7 %clamped_diffuse
%light_factor = OpFAdd %float %float0_3 %lambert

; Base Color
%base_color = OpCompositeConstruct %vec4 %float0_8 %float0_2 %float0_2 %float1
%lit_color = OpVectorTimesScalar %vec4 %base_color %light_factor

; Fog
%vec_to_cam = OpVectorShuffle %vec3 %val_cameraPos %val_cameraPos 0 1 2
%dist_vec = OpFSub %vec3 %vec_to_cam %val_worldPos
%dist = OpExtInst %float %1 Length %dist_vec
%fog_sub = OpFSub %float %dist %float1_5
%fog_factor_raw = OpFDiv %float %fog_sub %float3_0
%fog_factor = OpExtInst %float %1 FClamp %fog_factor_raw %float0 %float1

%fog_color = OpCompositeConstruct %vec4 %float0_7 %float0_7 %float0_8 %float1

%final_color = OpExtInst %vec4 %1 FMix %lit_color %fog_color %fog_factor

OpStore %outColor %final_color
OpReturn
OpFunctionEnd
