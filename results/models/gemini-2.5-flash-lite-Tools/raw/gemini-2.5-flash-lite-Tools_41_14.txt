Here is the complete SPIR-V assembly text for the fragment shader, implementing distance-based fog and Lambertian lighting:

```spirv
; SPIR-V
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; --- Types ---
OpTypeVoid
OpTypeBool
OpTypeFloat 32
OpTypeVector (OpTypeFloat 32) 2
OpTypeVector (OpTypeFloat 32) 3
OpTypeVector (OpTypeFloat 32) 4
OpTypeArray (OpTypeFloat 32) 4 ; vec4 array for mat4
OpTypeArray (OpTypeVector (OpTypeFloat 32) 4) 4 ; Correct type for mat4 as array of 4 vec4s
OpTypeStruct ; UBO struct type
    %mat4_type = OpTypeArray (OpTypeVector (OpTypeFloat 32) 4) 4 ; vec4[4] for mat4
    %ubo_type = OpTypeStruct
        %mat4_type ; model, offset 0
        %mat4_type ; view, offset 64
        %mat4_type ; projection, offset 128
        %v4float   ; lightPos, offset 192
        %v4float   ; cameraPos, offset 208
        %v4float   ; params, offset 224

; Pointers to types
%v4float_ptr = OpTypePointer Function (OpTypeVector (OpTypeFloat 32) 4)
%v3float_ptr = OpTypePointer Function (OpTypeVector (OpTypeFloat 32) 3)
%v2float_ptr = OpTypePointer Function (OpTypeVector (OpTypeFloat 32) 2)
%float_ptr = OpTypePointer Function (OpTypeFloat 32)
%bool_ptr = OpTypePointer Function (OpTypeBool)

; Input variables
%worldPos_ptr = OpTypePointer Input (OpTypeVector (OpTypeFloat 32) 3)
%normal_ptr = OpTypePointer Input (OpTypeVector (OpTypeFloat 32) 3)
%uv_ptr = OpTypePointer Input (OpTypeVector (OpTypeFloat 32) 2)
%tangent_ptr = OpTypePointer Input (OpTypeVector (OpTypeFloat 32) 3)
%color_ptr = OpTypePointer Input (OpTypeVector (OpTypeFloat 32) 3)

; Output variable
%outColor_ptr = OpTypePointer Output (OpTypeVector (OpTypeFloat 32) 4)

; Uniform buffer pointer
%ubo_ptr = OpTypePointer Uniform %ubo_type

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_1_5 = OpConstant %float 1.5
%float_3 = OpConstant %float 3.0
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0_7_fog = OpConstant %float 0.7
%float_0_8_fog = OpConstant %float 0.8
%v3_base_color_rgb = OpConstantComposite (OpTypeVector (OpTypeFloat 32) 3) %float_0_8 %float_0_2 %float_0_2
%v3_fog_color_rgb = OpConstantComposite (OpTypeVector (OpTypeFloat 32) 3) %float_0_7_fog %float_0_7_fog %float_0_8_fog
%v4_one = OpConstantComposite (OpTypeVector (OpTypeFloat 32) 4) %float_1 %float_1 %float_1 %float_1

; Variables for inputs and outputs
%worldPos = OpVariable Input %worldPos_ptr
%normal = OpVariable Input %normal_ptr
%uv = OpVariable Input %uv_ptr
%tangent = OpVariable Input %tangent_ptr
%color = OpVariable Input %color_ptr
%outColor = OpVariable Output %outColor_ptr

; Uniform buffer instantiation
%uniform_buffer = OpVariable Uniform %ubo_ptr

; --- Main function ---
%main = OpFunction %void None %void
%label_main_entry = OpLabel

; Load inputs
%loaded_worldPos = OpLoad %v3float %worldPos
%loaded_normal = OpLoad %v3float %normal
; %loaded_color = OpLoad %v3float %color ; Color input is not used in the final calculation per task description

; Load UBO
%ubo_data = OpLoad %ubo_type %uniform_buffer

; Extract cameraPos and lightPos from UBO
; lightPos is at struct index 3, cameraPos is at struct index 4
%lightPos_vec4 = OpCompositeExtract %v4float %ubo_data 3
%cameraPos_vec4 = OpCompositeExtract %v4float %ubo_data 4

; Extract XYZ from cameraPos and lightPos vec4
%lightPos = OpVectorExtractDynamic %v3float %lightPos_vec4 (OpConstant %float 0)
%cameraPos = OpVectorExtractDynamic %v3float %cameraPos_vec4 (OpConstant %float 0)

; --- Diffuse Calculation ---
; Calculate light direction vector
%light_dir_unnormalized = OpFSub %v3float %lightPos %loaded_worldPos

; Manual normalization of light direction
%light_dir_x = OpCompositeExtract %float %light_dir_unnormalized 0
%light_dir_y = OpCompositeExtract %float %light_dir_unnormalized 1
%light_dir_z = OpCompositeExtract %float %light_dir_unnormalized 2

%light_dir_x_sq = OpFMul %float %light_dir_x %light_dir_x
%light_dir_y_sq = OpFMul %float %light_dir_y %light_dir_y
%light_dir_z_sq = OpFMul %float %light_dir_z %light_dir_z

%light_dir_len_sq = OpFAdd %float %light_dir_x_sq (OpFAdd %float %light_dir_y_sq %light_dir_z_sq)
%light_dir_len = OpSqrt %float %light_dir_len_sq

; Safe normalization to avoid division by zero
%safe_inv_len = OpSelect %float %float_0 %float_1 (OpFOrdGreaterThan %bool %light_dir_len %float_0)
%inv_light_dir_len = OpFDiv %float %float_1 %safe_inv_len

%normalized_light_dir_x = OpFMul %float %light_dir_x %inv_light_dir_len
%normalized_light_dir_y = OpFMul %float %light_dir_y %inv_light_dir_len
%normalized_light_dir_z = OpFMul %float %light_dir_z %inv_light_dir_len
%normalized_light_dir = OpCompositeConstruct %v3float %normalized_light_dir_x %normalized_light_dir_y %normalized_light_dir_z

; Calculate diffuse term
%diffuse = OpDot %float %loaded_normal %normalized_light_dir

; Clamp diffuse to [0, 1]
%clamped_diffuse_val = OpFClamp %float %diffuse %float_0 %float_1

; Calculate Lambertian factor: 0.3 + 0.7 * clamped_diffuse
%lambertian_factor_term1 = OpFMul %float %clamped_diffuse_val %float_0_7
%lambertian_factor = OpFAdd %float %float_0_3 %lambertian_factor_term1

; Base RGB color calculation: lambertian_factor * (0.8, 0.2, 0.2)
%base_color_rgb_mul = OpFMul %v3float %lambertian_factor %v3_base_color_rgb

; --- Distance-based Fog ---
; Calculate distance from cameraPos to worldPos
%dist_vec = OpFSub %v3float %loaded_worldPos %cameraPos
%dist_sq = OpDot %float %dist_vec %dist_vec
%dist = OpSqrt %float %dist_sq

; Calculate fog factor: clamp((dist - 1.5) / 3.0)
%dist_minus_1_5 = OpFSub %float %dist %float_1_5
%fog_factor_unclamped = OpFDiv %float %dist_minus_1_5 %float_3
%fog_factor = OpFClamp %float %fog_factor_unclamped %float_0 %float_1

; --- Mix colors ---
; outColor.rgb = mix(base_color_rgb, fog_color, fog_factor)
%mixed_rgb = OpVectorMix %v3float %base_color_rgb_mul %v3_fog_color_rgb %fog_factor

; Set output color (alpha = 1.0)
%outColor_final = OpCompositeConstruct %v4float %mixed_rgb %float_1

; Store output
OpStore %outColor %outColor_final

OpReturn
OpEnd
```
