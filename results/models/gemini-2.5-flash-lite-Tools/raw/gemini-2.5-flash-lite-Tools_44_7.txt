#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(normal);

    // Light vector
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // View vector
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);

    // Diffuse component
    float diffuse = max(dot(N, L), 0.0);

    // Rim (Fresnel) lighting
    float rim = pow(1.0 - max(dot(N, V), 0.0), 3.0);

    // Base color and rim tint color
    vec3 base_color = vec3(0.2, 0.3, 0.8);
    vec3 rim_tint_color = vec3(1.0, 0.8, 0.5);

    // Final color calculation
    vec3 final_rgb = base_color * (0.3 + ubo.params.y * diffuse) + rim_tint_color * rim;

    outColor = vec4(final_rgb, 1.0);
}
