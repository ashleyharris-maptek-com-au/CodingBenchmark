struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Polka Dot Pattern
    float2 uv_scaled = input.uv * 6.0;
    float2 cell_center = floor(uv_scaled) + 0.5;
    float dist_to_center = distance(uv_scaled, cell_center);

    float3 dot_color = float3(1.0, 0.8, 0.0);       // Yellow
    float3 background_color = float3(0.1, 0.1, 0.3); // Dark blue

    float3 base_color = (dist_to_center < 0.3) ? dot_color : background_color;

    // Lambertian Shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);

    float diffuse = max(0.0, dot(N, L));
    float lambertian_factor = 0.3 + 0.7 * diffuse;

    float3 final_color = base_color * lambertian_factor;

    return float4(final_color, 1.0);
}
