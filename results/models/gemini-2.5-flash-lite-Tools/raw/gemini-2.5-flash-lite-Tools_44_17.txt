#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Lambertian shading
    float diff = max(dot(N, L), 0.0);
    float shading = 0.4 + 0.6 * diff;
    
    // Brick pattern
    vec2 scale = vec2(8.0, 16.0);
    vec2 st = uv * scale;
    
    // Staggered rows
    if (mod(floor(st.y), 2.0) == 1.0) {
        st.x += 0.5;
    }
    
    vec2 brick = floor(st);
    vec2 f = fract(st);
    
    // Mortar thickness
    vec2 mortar = vec2(0.05, 0.1);
    
    // Check if in mortar
    bool isMortar = (f.x < mortar.x) || (f.x > 1.0 - mortar.x) || 
                    (f.y < mortar.y) || (f.y > 1.0 - mortar.y);
    
    vec3 baseColor = isMortar ? vec3(0.7) : vec3(0.7, 0.2, 0.1);
    
    outColor = vec4(baseColor * shading, 1.0);
}
