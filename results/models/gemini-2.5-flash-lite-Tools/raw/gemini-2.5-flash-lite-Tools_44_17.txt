#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // ---- Brick Pattern Calculation ----
    vec2 scaledUV = uv * vec2(8.0, 16.0); // 8x16 UV scale

    // Apply half-brick offset to alternate rows
    float row = floor(scaledUV.y);
    if (mod(row, 2.0) < 0.5) { // Even rows
        scaledUV.x += 0.5;
    }

    vec2 brickUV = fract(scaledUV); // UV within a single brick unit

    // Mortar thickness
    vec2 mortarSize = vec2(0.05, 0.1); // 0.05 in X, 0.1 in Y

    // Check for mortar
    bool isMortarX = brickUV.x < mortarSize.x;
    bool isMortarY = brickUV.y < mortarSize.y;

    vec3 finalColor;
    if (isMortarX || isMortarY) {
        finalColor = vec3(0.7, 0.7, 0.7); // Mortar color
    } else {
        finalColor = vec3(0.7, 0.2, 0.1); // Brick color
    }

    // ---- Lambertian Shading ----
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    float diffuse = max(0.0, dot(N, L));
    vec3 shadedColor = finalColor * (0.4 + 0.6 * diffuse);

    outColor = vec4(shadedColor, 1.0);
}
