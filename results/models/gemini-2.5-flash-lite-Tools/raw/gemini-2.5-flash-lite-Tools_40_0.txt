#include <iostream>
#include <vector>
#include "ortools/sat/cp_model.h"

/**
 * This example solves a simple 0/1 Knapsack problem using Google OR-Tools CP-SAT.
 * 
 * Problem:
 * Maximize: 10*x0 + 20*x1 + 15*x2
 * Subject to:
 * 2*x0 + 3*x1 + 4*x2 <= 5
 * x0, x1, x2 in {0, 1}
 */

int main() {
    operations_research::sat::CpModelBuilder cp_model;

    // Variables
    const operations_research::sat::IntVar x0 = cp_model.NewBoolVar().WithName("x0");
    const operations_research::sat::IntVar x1 = cp_model.NewBoolVar().WithName("x1");
    const operations_research::sat::IntVar x2 = cp_model.NewBoolVar().WithName("x2");

    // Constraint: 2*x0 + 3*x1 + 4*x2 <= 5
    cp_model.AddLessOrEqual(2 * x0 + 3 * x1 + 4 * x2, 5);

    // Objective: Maximize 10*x0 + 20*x1 + 15*x2
    cp_model.Maximize(10 * x0 + 20 * x1 + 15 * x2);

    // Solve
    const operations_research::sat::CpSolverResponse response = operations_research::sat::Solve(cp_model.Build());

    if (response.status() == operations_research::sat::CpSolverStatus::OPTIMAL ||
        response.status() == operations_research::sat::CpSolverStatus::FEASIBLE) {
        std::cout << "Solution found:" << std::endl;
        std::cout << "x0 = " << operations_research::sat::SolutionIntegerValue(response, x0) << std::endl;
        std::cout << "x1 = " << operations_research::sat::SolutionIntegerValue(response, x1) << std::endl;
        std::cout << "x2 = " << operations_research::sat::SolutionIntegerValue(response, x2) << std::endl;
        std::cout << "Objective value = " << response.objective_value() << std::endl;
    } else {
        std::cout << "No solution found." << std::endl;
    }

    return 0;
}
