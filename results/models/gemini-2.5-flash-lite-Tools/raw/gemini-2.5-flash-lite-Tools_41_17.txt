; SPIR-V,1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

; --- Types ---
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4float = OpTypeMatrix %float 4 4
%bool = OpTypeBool

; --- UBO struct definition ---
; Members: mat4 model, mat4 view, mat4 projection, vec4 lightPos, vec4 cameraPos, vec4 params
%ubo_members = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float
%ubo_ptr = OpTypePointer Uniform %ubo_members

; --- Function types ---
%func_void = OpTypeFunction %void
%func_ubo_ptr = OpTypeFunction %ubo_ptr
%func_float = OpTypeFunction %float
%func_v3float_v4float_v4float_v3float = OpTypeFunction %float %v3float %v4float %v4float %v3float ; For calculate_diffuse
%func_v2float_float_float_float_float_v3float_v3float = OpTypeFunction %v3float %v2float %float %float %float %float %v3float %v3float ; For generate_brick_pattern

; --- Global Declarations ---

; Input variables from vertex shader
%worldPos_var = OpVariable %v3float Input
OpDecorate %worldPos_var Location 0

%normal_var = OpVariable %v3float Input
OpDecorate %normal_var Location 1

%uv_var = OpVariable %v2float Input
OpDecorate %uv_var Location 2

%tangent_var = OpVariable %v3float Input
OpDecorate %tangent_var Location 3

%color_var = OpVariable %v4float Input
OpDecorate %color_var Location 4

; Output variable
%outColor = OpVariable %v4float Output
OpDecorate %outColor Location 0

; Uniform buffer
%UBO = OpVariable %ubo_ptr Uniform
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0

; --- Constants ---
%const_8 = OpConstant %float 8.0
%const_16 = OpConstant %float 16.0
%const_0_05 = OpConstant %float 0.05
%const_0_1 = OpConstant %float 0.1
%const_0_7 = OpConstant %float 0.7
%const_0_2 = OpConstant %float 0.2
%const_0_4 = OpConstant %float 0.4
%const_0_6 = OpConstant %float 0.6
%const_0_5 = OpConstant %float 0.5
%const_1_0 = OpConstant %float 1.0
%const_0_0 = OpConstant %float 0.0
%const_2_0 = OpConstant %float 2.0

; --- Function to get UBO Data ---
%get_ubo_data = OpFunction %ubo_ptr None %func_ubo_ptr
    %ubo_load = OpLoad %ubo_ptr %UBO
    OpReturnValue %ubo_load
OpFunctionEnd

; --- Function to calculate diffuse lighting ---
%calculate_diffuse = OpFunction %float None %func_v3float_v4float_v4float_v3float
    %norm_param = OpFunctionParameter %v3float
    %lightPos_param = OpFunctionParameter %v4float
    %camPos_param = OpFunctionParameter %v4float ; Not used in this Lambertian calculation
    %worldPos_param = OpFunctionParameter %v3float

    ; Calculate light direction: lightPos - worldPos
    %lightDir_raw = OpVectorSubtract %v3float %lightPos_param %worldPos_param
    %lightDir_normalized = OpFNormalize %v3float %lightDir_raw

    ; Calculate dot product for diffuse
    %diffuse = OpDot %float %norm_param %lightDir_normalized

    ; Clamp diffuse factor to [0, 1]
    %clamped_diffuse = OpFClamp %float %diffuse %const_0_0 %const_1_0

    OpReturnValue %clamped_diffuse
OpFunctionEnd

; --- Function to generate brick pattern ---
%generate_brick_pattern = OpFunction %v3float None %func_v2float_float_float_float_float_v3float_v3float
    %uv_param = OpFunctionParameter %v2float
    %scale_x_param = OpFunctionParameter %float
    %scale_y_param = OpFunctionParameter %float
    %mortar_thick_x_param = OpFunctionParameter %float
    %mortar_thick_y_param = OpFunctionParameter %float
    %mortar_color_param = OpFunctionParameter %v3float
    %brick_color_param = OpFunctionParameter %v3float

    ; Scale UVs
    %scaled_uv_x = OpFMul %float %uv_param.x %scale_x_param
    %scaled_uv_y = OpFMul %float %uv_param.y %scale_y_param
    %scaled_uv = OpCompositeConstruct %v2float %scaled_uv_x %scaled_uv_y

    ; Row staggering calculation
    %row_index_y_float = OpFloor %float %scaled_uv.y
    %row_mod_2 = OpMod %float %row_index_y_float %const_2_0
    ; Check if row_mod_2 is close to 0.0 (for even rows)
    %is_even_row = OpQuantizeToScalar %bool %row_mod_2 %const_0_0 

    %offset_x = OpSelect %float %is_even_row %const_0_0 %const_0_5 ; Half-brick offset for odd rows

    ; Calculate coordinates within a single brick/mortar cell
    %cell_x = OpFract %float %scaled_uv.x
    %cell_y = OpFract %float %scaled_uv.y

    ; Apply offset to x coordinate
    %offset_cell_x = OpFAdd %float %cell_x %offset_x

    ; Check if inside mortar
    %in_mortar_x = OpFOrdLessThan %bool %offset_cell_x %mortar_thick_x_param
    %in_mortar_x2 = OpFOrdGreaterThan %bool %offset_cell_x (OpFSub %float %const_1_0 %mortar_thick_x_param)
    %in_mortar_y = OpFOrdLessThan %bool %cell_y %mortar_thick_y_param
    %in_mortar_y2 = OpFOrdGreaterThan %bool %cell_y (OpFSub %float %const_1_0 %mortar_thick_y_param)

    %is_in_mortar = OpLogicalOr %bool %in_mortar_x %in_mortar_x2
    %is_in_mortar = OpLogicalOr %bool %is_in_mortar %in_mortar_y
    %is_in_mortar = OpLogicalOr %bool %is_in_mortar %in_mortar_y2

    %result_color = OpSelect %v3float %is_in_mortar %mortar_color_param %brick_color_param

    OpReturnValue %result_color
OpFunctionEnd

; --- Main Entry Point ---
%main = OpFunction %void None %func_void
    ; Load inputs
    %worldPos_load = OpLoad %v3float %worldPos_var
    %normal_load = OpLoad %v3float %normal_var
    %uv_load = OpLoad %v2float %uv_var
    %color_load = OpLoad %v4float %color_var ; Not used in this shader but declared

    ; Get UBO data and required components
    %ubo_ptr_main = OpFunctionCall %ubo_ptr %get_ubo_data
    %lightPos_main_val = OpLoad %v4float (OpAccessChain %v4float %ubo_ptr_main 3 0) ; Load lightPos
    %camPos_main_val = OpLoad %v4float (OpAccessChain %v4float %ubo_ptr_main 4 0) ; Load cameraPos

    ; Calculate diffuse lighting
    %diffuse_factor = OpFunctionCall %float %calculate_diffuse %normal_load %lightPos_main_val %camPos_main_val %worldPos_load

    ; Calculate brick color using the pattern function
    %brick_color_result = OpFunctionCall %v3float %generate_brick_pattern %uv_load %const_8 %const_16 %const_0_05 %const_0_1 %const_0_7 %const_0_2

    ; Apply Lambertian shading: (0.4 + 0.6 * diffuse)
    %diffuse_term = OpFMul %float %diffuse_factor %const_0_6
    %shading_factor = OpFAdd %float %diffuse_term %const_0_4

    ; Multiply brick color by shading factor
    %final_color_rgb = OpVectorTimesScalar %v3float %brick_color_result %shading_factor

    ; Set output color (alpha = 1.0)
    %final_color_rgba = OpCompositeConstruct %v4float %final_color_rgb.x %final_color_rgb.y %final_color_rgb.z %const_1_0
    OpStore %outColor %final_color_rgba

    OpReturn
OpFunctionEnd

; --- Execution Modes ---
OpExecutionMode %main OriginUpperLeft
