; A simple SPIR-V fragment shader that outputs a constant red color.

; 1. Capabilities
OpCapability Shader

; 4. Memory Model
OpMemoryModel Logical GLSL450

; 5. Entry Points
; Fragment execution model, %main function, and the output interface variable (%outColor)
OpEntryPoint Fragment %main "main" %outColor

; 6. Execution Modes
OpExecutionMode %main OriginUpperLeft

; 8. Annotations (Layout)
OpDecorate %outColor Location 0

; 9. Types, Constants, and Global Variables
%void      = OpTypeVoid
%func      = OpTypeFunction %void
%float     = OpTypeFloat 32
%v4float   = OpTypeVector %float 4

; Pointers
%ptr_out_v4float = OpTypePointer Output %v4float

; Constants
%float_1 = OpConstant %float 1
%float_0 = OpConstant %float 0
%const_red = OpConstantComposite %v4float %float_1 %float_0 %float_0 %float_1

; Global Variables
%outColor = OpVariable %ptr_out_v4float Output

; 10. Function Definitions
%main = OpFunction %void None %func
%label = OpLabel

; Store red color to output
OpStore %outColor %const_red

OpReturn
OpFunctionEnd
