; SPIR-V
SPIRV-Assembly-Version 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%type.f32 = OpTypeFloat 32
%type.vec2.f32 = OpTypeVector %type.f32 2 ; For uv
%type.vec3.f32 = OpTypeVector %type.f32 3
%type.vec4.f32 = OpTypeVector %type.f32 4
%type.void = OpTypeVoid
%func_type_void = OpTypeFunction %type.void
%type.mat4.f32 = OpTypeMatrix %type.vec4.f32 4 ; GLSL mat4 is column-major, 4 columns of vec4
%type.ubo_ubo = OpTypeStruct %type.mat4.f32 %type.mat4.f32 %type.mat4.f32 %type.vec4.f32 %type.vec4.f32 %type.vec4.f32
%type.uniform_buffer.UBO = OpTypePointer Uniform %type.ubo_ubo

; Constants
%const_1_0 = OpConstant %type.f32 1.0
%const_0_5 = OpConstant %type.f32 0.5
%const_alpha_1_0 = OpConstant %type.f32 1.0

; Input Variables
%worldPos = OpVariable %type.vec3.f32 Input
    OpDecorate %worldPos Location 0
%normal = OpVariable %type.vec3.f32 Input
    OpDecorate %normal Location 1
%uv = OpVariable %type.vec2.f32 Input
    OpDecorate %uv Location 2
%tangent = OpVariable %type.vec3.f32 Input
    OpDecorate %tangent Location 3
%color = OpVariable %type.vec3.f32 Input
    OpDecorate %color Location 4

; Output Variable
%outColor = OpVariable %type.vec4.f32 Output
    OpDecorate %outColor Location 0

; Uniform Buffer
%ubo = OpVariable %type.uniform_buffer.UBO Uniform
    OpDecorate %ubo DescriptorSet 0
    OpDecorate %ubo Binding 0

; Entry Point
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Function Definition
%main = OpFunction %type.void %func_type_void
%main_label = OpLabel

    ; Load input normal
    %normal_loaded = OpLoad %type.vec3.f32 %normal

    ; Normalize the normal (in case it's not unit length after interpolation)
    %normalized_normal = OpNormalize %type.vec3.f32 %normal_loaded

    ; Map normal from [-1, 1] to [0, 1] for RGB components
    %normal_plus_1 = OpFAdd %type.vec3.f32 %normalized_normal %const_1_0
    %color_rgb = OpFMul %type.vec3.f32 %normal_plus_1 %const_0_5

    ; Construct the final RGBA color, setting Alpha to 1.0
    %out_color_final = OpCompositeConstruct %type.vec4.f32 %color_rgb %const_alpha_1_0

    ; Store the output color
    OpStore %outColor %out_color_final

    OpReturn
    OpFunctionEnd
