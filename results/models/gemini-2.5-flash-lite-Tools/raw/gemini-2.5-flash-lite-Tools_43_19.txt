// Vulkan-compatible HLSL Pixel Shader
// Requires DirectX Shader Compiler (DXC) with -spirv flag

// Push Constants
struct PushConstants {
    float4 colorModifier;
    float  time;
};
[[vk::push_constant]]
PushConstants push;

// Uniform Buffer (Set 0, Binding 0)
struct SceneData {
    float4x4 viewProj;
    float3   lightPos;
};
ConstantBuffer<SceneData> scene : register(b0, space0);

// Texture and Sampler (Set 1, Bindings 0 & 1)
[[vk::binding(0, 1)]] Texture2D    mainTex;
[[vk::binding(1, 1)]] SamplerState mainSampler;

// Input Structure
struct PSInput {
    float4 pos      : SV_POSITION; // Maps to gl_FragCoord
    [[vk::location(0)]] float2 uv    : TEXCOORD0;
    [[vk::location(1)]] float3 norm  : NORMAL;
};

// Entry Point
// SV_Target maps to the output frame buffer at location 0
float4 main(PSInput input) : SV_Target {
    // Sample texture
    float4 texColor = mainTex.Sample(mainSampler, input.uv);
    
    // Simple lighting and color modification
    float3 finalColor = texColor.rgb * push.colorModifier.rgb;
    
    // Simple diffuse lighting
    float3 normal = normalize(input.norm);
    float3 lightDir = normalize(scene.lightPos);
    float diff = max(dot(normal, lightDir), 0.2); // Add ambient
    
    finalColor *= diff;
    
    return float4(finalColor, texColor.a);
}
