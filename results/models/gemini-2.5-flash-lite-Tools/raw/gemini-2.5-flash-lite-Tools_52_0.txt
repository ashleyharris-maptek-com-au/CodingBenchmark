from autopilot.interfaces import Planner
from autopilot.models import Environment, VehicleState, Command

class BasicPlanner(Planner):
    def plan(self, state: VehicleState, env: Environment) -> Command:
        # Simple ACC logic
        target_speed = 30.0
        
        # Check for vehicle ahead in the same lane
        ahead_vehicle = None
        for v in env.surrounding_vehicles:
            if v.lane_id == state.lane_id and v.position > state.position:
                if ahead_vehicle is None or v.position < ahead_vehicle.position:
                    ahead_vehicle = v
        
        throttle = 0.0
        brake = 0.0
        
        if ahead_vehicle:
            distance = ahead_vehicle.position - state.position
            if distance < 50.0:
                # Slow down
                throttle = 0.0
                brake = 0.5
            else:
                # Cruise
                throttle = 0.2
        else:
            # Accelerate
            throttle = 0.5
            
        return Command(steering_angle=0.0, throttle=throttle, brake=brake)
