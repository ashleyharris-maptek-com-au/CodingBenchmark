// Example Vulkan Pixel Shader in HLSL
// Target: ps_6_0
// Compiled via DXC: dxc -T ps_6_0 -spirv -fspv-target-env=vulkan1.2 shader.hlsl -Fo shader.spv

// Input structure from vertex shader
struct PixelInput {
    float4 position : SV_Position;
    [[vk::location(0)]] float2 uv : TEXCOORD0;
    [[vk::location(1)]] float4 color : COLOR0;
};

// Constant Buffer for scene data
struct SceneData {
    float4x4 viewProj;
    float time;
};

// Resource Bindings
[[vk::binding(0, 0)]] ConstantBuffer<SceneData> sceneBuffer : register(b0);
[[vk::binding(1, 0)]] Texture2D mainTexture : register(t0);
[[vk::binding(2, 0)]] SamplerState mainSampler : register(s0);

// Main pixel shader entry point
float4 main(PixelInput input) : SV_Target {
    // Sample texture
    float4 texColor = mainTexture.Sample(mainSampler, input.uv);
    
    // Combine with input color and basic time animation
    float3 finalColor = texColor.rgb * input.color.rgb;
    finalColor *= (0.5 + 0.5 * sin(sceneBuffer.time));
    
    return float4(finalColor, texColor.a * input.color.a);
}
