Here is the complete C++ code for solving the Steiner Tree problem using the MST heuristic:

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <limits>
#include <algorithm>
#include <set>
#include <map> // For metric closure graph

// Structure to represent an edge in the original graph
struct Edge {
    int to;
    int weight;
};

// Structure to represent a node in the priority queue for Dijkstra's
struct State {
    int vertex;
    int cost;

    bool operator>(const State& other) const {
        return cost > other.cost;
    }
};

// Function to parse input and build the graph
void parse_input(int& num_vertices, int& num_edges, int& num_terminals,
                 std::vector<std::vector<Edge>>& adj, std::vector<int>& terminals) {
    std::cin >> num_vertices >> num_edges >> num_terminals;

    adj.resize(num_vertices);
    for (int i = 0; i < num_edges; ++i) {
        int u, v, weight;
        std::cin >> u >> v >> weight;
        adj[u].push_back({v, weight});
        adj[v].push_back({u, weight}); // Undirected graph
    }

    terminals.resize(num_terminals);
    for (int i = 0; i < num_terminals; ++i) {
        std::cin >> terminals[i];
    }
}

// Dijkstra's algorithm to find shortest paths from a source vertex
// Returns distances and predecessors
std::pair<std::vector<long long>, std::vector<int>> dijkstra(int start_node, int num_vertices, const std::vector<std::vector<Edge>>& adj) {
    std::vector<long long> dist(num_vertices, std::numeric_limits<long long>::max());
    std::vector<int> pred(num_vertices, -1);
    std::priority_queue<State, std::vector<State>, std::greater<State>> pq;

    dist[start_node] = 0;
    pq.push({start_node, 0});

    while (!pq.empty()) {
        State current = pq.top();
        pq.pop();

        int u = current.vertex;
        int d = current.cost;

        if (d > dist[u]) {
            continue;
        }

        for (const auto& edge : adj[u]) {
            int v = edge.to;
            int weight = edge.weight;
            if (dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                pred[v] = u;
                pq.push({v, (int)dist[v]});
            }
        }
    }
    return {dist, pred};
}

// Function to reconstruct the path from start to end using predecessors
// Stores edges in the `path_edges` set. Edges are stored as pairs of {min_vertex, max_vertex} to handle undirected nature.
void reconstruct_path(int start, int end, const std::vector<int>& pred,
                      const std::vector<std::vector<Edge>>& adj,
                      std::set<std::tuple<int, int, int>>& path_edges,
                      const std::vector<long long>& dist) {
    if (dist[end] == std::numeric_limits<long long>::max()) {
        // No path exists
        return;
    }

    int current = end;
    while (current != start && current != -1) {
        int p = pred[current];
        if (p == -1) break; // Should not happen if dist[end] is not max

        // Find the weight of the edge (p, current)
        int weight = -1;
        for (const auto& edge : adj[p]) {
            if (edge.to == current) {
                weight = edge.weight;
                break;
            }
        }
        // Ensure edge is added in a canonical form (smaller vertex first)
        path_edges.insert({std::min(p, current), std::max(p, current), weight});
        current = p;
    }
}

// Structure for MST edges
struct MSTEdge {
    int u, v, weight;
};

// Comparator for MST edges for sorting (used in Kruskal's or Prim's with vector)
bool compareMSTEdges(const MSTEdge& a, const MSTEdge& b) {
    return a.weight < b.weight;
}

// Prim's algorithm for Minimum Spanning Tree
// Takes an adjacency list representation of the graph and returns a vector of MST edges.
// `metric_closure_adj` maps terminal ID to its neighbors and weights.
std::vector<MSTEdge> prim_mst(int num_terminals, const std::map<int, std::vector<std::pair<int, int>>>& metric_closure_adj, const std::vector<int>& terminals_map_idx_to_id) {
    std::vector<MSTEdge> result_mst_edges;
    if (num_terminals == 0) return result_mst_edges;

    // Map to store minimum edge weight to connect a vertex to the MST
    std::vector<long long> min_edge_weight(num_terminals, std::numeric_limits<long long>::max());
    // Vector to store the parent of each vertex in the MST
    std::vector<int> parent(num_terminals, -1);
    // Boolean array to track vertices included in MST
    std::vector<bool> in_mst(num_terminals, false);

    // Start with the first terminal (index 0)
    min_edge_weight[0] = 0;

    // The MST will have num_terminals vertices
    for (int count = 0; count < num_terminals; ++count) {
        int u_idx = -1;
        long long min_val = std::numeric_limits<long long>::max();

        // Find the vertex with the minimum edge weight among those not yet included in MST
        for (int v_idx = 0; v_idx < num_terminals; ++v_idx) {
            if (!in_mst[v_idx] && min_edge_weight[v_idx] < min_val) {
                min_val = min_edge_weight[v_idx];
                u_idx = v_idx;
            }
        }

        // If no vertex can be added, break (e.g., disconnected graph, though metric closure should be connected)
        if (u_idx == -1) break;

        in_mst[u_idx] = true;

        // Add the edge to the result if it's not the first vertex
        if (parent[u_idx] != -1) {
            result_mst_edges.push_back({terminals_map_idx_to_id[parent[u_idx]], terminals_map_idx_to_id[u_idx], (int)min_val});
        }

        // Update edge weights of adjacent vertices
        int u_terminal_id = terminals_map_idx_to_id[u_idx];
        if (metric_closure_adj.count(u_terminal_id)) {
            for (const auto& edge_pair : metric_closure_adj.at(u_terminal_id)) {
                int v_terminal_id = edge_pair.first;
                int weight = edge_pair.second;

                // Find the index of v_terminal_id in terminals_map_idx_to_id
                int v_idx = -1;
                for(int i = 0; i < terminals_map_idx_to_id.size(); ++i) {
                    if (terminals_map_idx_to_id[i] == v_terminal_id) {
                        v_idx = i;
                        break;
                    }
                }

                if (v_idx != -1 && !in_mst[v_idx] && weight < min_edge_weight[v_idx]) {
                    parent[v_idx] = u_idx;
                    min_edge_weight[v_idx] = weight;
                }
            }
        }
    }
    return result_mst_edges;
}


int main() {
    // Fast I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_vertices, num_edges, num_terminals;
    std::vector<std::vector<Edge>> adj;
    std::vector<int> terminals;

    parse_input(num_vertices, num_edges, num_terminals, adj, terminals);

    if (num_terminals <= 1) {
        // If 0 or 1 terminal, the Steiner tree is trivial
        if (num_terminals == 1) {
            std::cout << "0 0\n"; // Weight 0, 0 edges
        } else {
            std::cout << "0 0\n";
        }
        return 0;
    }

    // Store shortest path distances and predecessors from each terminal
    std::vector<std::vector<long long>> terminal_dists(num_terminals, std::vector<long long>(num_vertices));
    std::vector<std::vector<int>> terminal_preds(num_terminals, std::vector<int>(num_vertices));
    
    // Map to create a compact index for terminals in Prim's
    // Key: terminal ID, Value: index (0 to num_terminals-1)
    std::map<int, int> terminal_id_to_idx;
    // Vector to map index back to terminal ID
    std::vector<int> terminals_map_idx_to_id(num_terminals);

    for (int i = 0; i < num_terminals; ++i) {
        terminal_id_to_idx[terminals[i]] = i;
        terminals_map_idx_to_id[i] = terminals[i];
        auto result = dijkstra(terminals[i], num_vertices, adj);
        terminal_dists[i] = result.first;
        terminal_preds[i] = result.second;
    }

    // Build the metric closure graph
    // Adjacency list for the metric closure graph: map terminal ID to list of {neighbor_terminal_ID, weight}
    std::map<int, std::vector<std::pair<int, int>>> metric_closure_adj;
    for (int i = 0; i < num_terminals; ++i) {
        for (int j = i + 1; j < num_terminals; ++j) {
            int u_term = terminals[i];
            int v_term = terminals[j];
            long long dist = terminal_dists[i][v_term]; // Distance from terminal i to terminal j

            if (dist != std::numeric_limits<long long>::max()) {
                metric_closure_adj[u_term].push_back({v_term, (int)dist});
                metric_closure_adj[v_term].push_back({u_term, (int)dist});
            }
        }
    }

    // Compute MST on the metric closure graph using Prim's algorithm
    std::vector<MSTEdge> metric_mst = prim_mst(num_terminals, metric_closure_adj, terminals_map_idx_to_id);

    // Collect all unique edges from the paths corresponding to the MST edges
    std::set<std::tuple<int, int, int>> steiner_tree_edges; // {u, v, weight}, sorted u, v
    
    for (const auto& mst_edge : metric_mst) {
        int u_term = mst_edge.u;
        int v_term = mst_edge.v;
        
        // Find which terminal index corresponds to u_term and v_term
        int u_term_idx = terminal_id_to_idx[u_term];
        // int v_term_idx = terminal_id_to_idx[v_term]; // Not directly used in reconstruct_path call

        // Reconstruct path using the correct predecessor array (based on which terminal was source)
        // The distance from u_term to v_term is given by terminal_dists[u_term_idx][v_term].
        // The predecessors for this path are stored in terminal_preds[u_term_idx].
        reconstruct_path(u_term, v_term, terminal_preds[u_term_idx], adj, steiner_tree_edges, terminal_dists[u_term_idx]);
    }

    // Union of edges might contain cycles. Compute a final MST on these collected edges.
    // First, collect all unique vertices involved in the steiner_tree_edges
    std::set<int> steiner_vertices_set;
    for (const auto& edge_tuple : steiner_tree_edges) {
        steiner_vertices_set.insert(std::get<0>(edge_tuple));
        steiner_vertices_set.insert(std::get<1>(edge_tuple));
    }
    
    // Convert set of vertices to a vector and create a map for new indices for MST
    std::vector<int> steiner_vertices(steiner_vertices_set.begin(), steiner_vertices_set.end());
    std::map<int, int> steiner_vertex_id_to_idx;
    for(int i = 0; i < steiner_vertices.size(); ++i) {
        steiner_vertex_id_to_idx[steiner_vertices[i]] = i;
    }

    // Build adjacency list for the subgraph formed by steiner_tree_edges
    // This is not directly used in Kruskal's but can be useful for debugging or other MST algos.
    // std::vector<std::vector<Edge>> steiner_subgraph_adj(steiner_vertices.size()); // Not needed for Kruskal's implementation below
    std::vector<MSTEdge> final_mst_candidate_edges;
    for (const auto& edge_tuple : steiner_tree_edges) {
        int u = std::get<0>(edge_tuple);
        int v = std::get<1>(edge_tuple);
        int weight = std::get<2>(edge_tuple);
        
        // int u_idx = steiner_vertex_id_to_idx[u]; // Not needed for Kruskal's candidate edges
        // int v_idx = steiner_vertex_id_to_idx[v]; // Not needed for Kruskal's candidate edges
        
        // steiner_subgraph_adj[u_idx].push_back({v_idx, weight}); // Not needed
        // steiner_subgraph_adj[v_idx].push_back({u_idx, weight}); // Not needed
        final_mst_candidate_edges.push_back({u, v, weight});
    }

    // Sort edges for Kruskal's algorithm
    std::sort(final_mst_candidate_edges.begin(), final_mst_candidate_edges.end(), compareMSTEdges);

    // Kruskal's algorithm for the final MST
    // Disjoint Set Union (DSU) structure
    std::vector<int> dsu_parent(steiner_vertices.size());
    std::vector<int> dsu_rank(steiner_vertices.size(), 0);
    for(int i = 0; i < steiner_vertices.size(); ++i) {
        dsu_parent[i] = i;
    }

    auto find_set = [&](auto self, int v_idx) -> int {
        if (v_idx == dsu_parent[v_idx])
            return v_idx;
        return dsu_parent[v_idx] = self(self, dsu_parent[v_idx]);
    };

    auto union_sets = [&](int a_idx, int b_idx) -> bool {
        a_idx = find_set(find_set, a_idx);
        b_idx = find_set(find_set, b_idx);
        if (a_idx != b_idx) {
            if (dsu_rank[a_idx] < dsu_rank[b_idx]) std::swap(a_idx, b_idx);
            dsu_parent[b_idx] = a_idx;
            if (dsu_rank[a_idx] == dsu_rank[b_idx]) dsu_rank[a_idx]++;
            return true;
        }
        return false;
    };

    std::vector<MSTEdge> final_steiner_tree;
    long long total_weight = 0;
    int edges_count = 0;

    for (const auto& edge : final_mst_candidate_edges) {
        int u_idx = steiner_vertex_id_to_idx[edge.u];
        int v_idx = steiner_vertex_id_to_idx[edge.v];

        if (union_sets(u_idx, v_idx)) {
            final_steiner_tree.push_back(edge);
            total_weight += edge.weight;
            edges_count++;
        }
    }

    // The problem statement guarantees a connected graph for terminals.
    // The resulting graph should connect all terminals.

    // Output
    std::cout << total_weight << " " << edges_count << "\n";
    for (const auto& edge : final_steiner_tree) {
        std::cout << edge.u << " " << edge.v << "\n";
    }

    return 0;
}
```
