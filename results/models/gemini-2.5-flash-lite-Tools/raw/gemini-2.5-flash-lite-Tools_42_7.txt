; SPIR-V
; Version: 1.0
; Generator: Google Shaderc over GLSL; 14
; Capabilities: Geometry
; Extensions: SPV_EXT_shader_viewport_index_layer
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpExtInstImport %glsl = "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; All types, structures, and constants
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%m4v4float = OpTypeMatrix %v4float 4

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
; %uint_4 = OpConstant %uint 4 ; Not strictly needed but kept for consistency with previous thought process
; %uint_7 = OpConstant %uint 7 ; Not strictly needed here, was for shuffle which I replaced with CompositeConstruct

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
; %int_3 = OpConstant %int 3 ; Not strictly needed here

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_08 = OpConstant %float 0.08
%float_3 = OpConstant %float 3
; %float_0_33333334 = OpConstant %float 0.33333334 ; Not strictly needed, using direct division

; Composite constants
%v4float_0_0_0_0 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0
%v3float_3 = OpConstantComposite %v3float %float_3 %float_3 %float_3

; Pointer Types
%ptr_input_v4float = OpTypePointer Input %v4float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v3float = OpTypePointer Output %v3float

; gl_PerVertex (Input)
%gl_PerVertex_in_type = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_in_type 0 BuiltIn Position
OpDecorate %gl_PerVertex_in_type Block
%arr_gl_PerVertex_in_type = OpTypeArray %gl_PerVertex_in_type %uint_3
%ptr_input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in_type

; gl_PerVertex (Output)
%gl_PerVertex_out_type = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_out_type 0 BuiltIn Position
OpDecorate %gl_PerVertex_out_type Block
%ptr_output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out_type

; User defined inputs (arrays of 3 vec3)
%arr_v3float_in_type = OpTypeArray %v3float %uint_3
%ptr_input_arr_v3float = OpTypePointer Input %arr_v3float_in_type

; UBO Types (from test 41 description)
%type_UBO = OpTypeStruct %m4v4float %m4v4float %m4v4float %v4float %v4float %v4float
OpMemberDecorate %type_UBO 0 Offset 0
OpMemberDecorate %type_UBO 1 Offset 64
OpMemberDecorate %type_UBO 2 Offset 128
OpMemberDecorate %type_UBO 3 Offset 192
OpMemberDecorate %type_UBO 4 Offset 208
OpMemberDecorate %type_UBO 5 Offset 224
OpDecorate %type_UBO Block
%ptr_uniform_type_UBO = OpTypePointer Uniform %type_UBO

; Global variables
%gl_in = OpVariable %ptr_input_arr_gl_PerVertex_in Input
%gl_out = OpVariable %ptr_output_gl_PerVertex_out Output
%worldPos = OpVariable %ptr_input_arr_v3float Input
%normal = OpVariable %ptr_input_arr_v3float Input
%color = OpVariable %ptr_input_arr_v3float Input
%outColor = OpVariable %ptr_output_v3float Output
%ubo = OpVariable %ptr_uniform_type_UBO Uniform

; Annotations (Decorations for locations)
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void_fn = OpTypeFunction %void
%main = OpFunction %void None %void_fn
%label = OpLabel

; Read input positions (gl_Position)
%gl_in_0_pos_ptr = OpAccessChain %ptr_input_v4float %gl_in %uint_0 %uint_0
%load_p0_v4 = OpLoad %v4float %gl_in_0_pos_ptr
%gl_in_1_pos_ptr = OpAccessChain %ptr_input_v4float %gl_in %uint_1 %uint_0
%load_p1_v4 = OpLoad %v4float %gl_in_1_pos_ptr
%gl_in_2_pos_ptr = OpAccessChain %ptr_input_v4float %gl_in %uint_2 %uint_0
%load_p2_v4 = OpLoad %v4float %gl_in_2_pos_ptr

; Convert p0, p1, p2 (v4float) to v3float for face normal calculations
; Extract x, y, z components
%p0_v3_calc = OpVectorShuffle %v3float %load_p0_v4 %v4float_0_0_0_0 %uint_0 %uint_1 %uint_2
%p1_v3_calc = OpVectorShuffle %v3float %load_p1_v4 %v4float_0_0_0_0 %uint_0 %uint_1 %uint_2
%p2_v3_calc = OpVectorShuffle %v3float %load_p2_v4 %v4float_0_0_0_0 %uint_0 %uint_1 %uint_2

; Compute face center: center = (p0 + p1 + p2) / 3.
%p0_plus_p1 = OpFAdd %v3float %p0_v3_calc %p1_v3_calc
%sum_p = OpFAdd %v3float %p0_plus_p1 %p2_v3_calc
%center_v3 = OpFDiv %v3float %sum_p %v3float_3

; Compute face normal: e1 = p1-p0, e2 = p2-p0, N = normalize(cross(e1, e2)).
%e1 = OpFSub %v3float %p1_v3_calc %p0_v3_calc
%e2 = OpFSub %v3float %p2_v3_calc %p0_v3_calc

%cross_e1_e2 = OpExtInst %v3float %glsl Cross %e1 %e2
%N = OpExtInst %v3float %glsl Normalize %cross_e1_e2

; Offset = N * 0.08
%offset_v3 = OpVectorTimesScalar %v3float %N %float_0_08

; Prepare offset_v4 (vec4(offset_v3, 0)) for adding to gl_Position
%offset_v4_x = OpCompositeExtract %float %offset_v3 %uint_0
%offset_v4_y = OpCompositeExtract %float %offset_v3 %uint_1
%offset_v4_z = OpCompositeExtract %float %offset_v3 %uint_2
%offset_v4 = OpCompositeConstruct %v4float %offset_v4_x %offset_v4_y %offset_v4_z %float_0

; Emit 3 vertices, each shifted by the offset and colored with original color
; Vertex 0
%new_pos_0 = OpFAdd %v4float %load_p0_v4 %offset_v4
OpStore %gl_out %new_pos_0

%color_0_ptr = OpAccessChain %ptr_input_v3float %color %uint_0
%load_c0_v3 = OpLoad %v3float %color_0_ptr
OpStore %outColor %load_c0_v3
OpEmitVertex

; Vertex 1
%new_pos_1 = OpFAdd %v4float %load_p1_v4 %offset_v4
OpStore %gl_out %new_pos_1

%color_1_ptr = OpAccessChain %ptr_input_v3float %color %uint_1
%load_c1_v3 = OpLoad %v3float %color_1_ptr
OpStore %outColor %load_c1_v3
OpEmitVertex

; Vertex 2
%new_pos_2 = OpFAdd %v4float %load_p2_v4 %offset_v4
OpStore %gl_out %new_pos_2

%color_2_ptr = OpAccessChain %ptr_input_v3float %color %uint_2
%load_c2_v3 = OpLoad %v3float %color_2_ptr
OpStore %outColor %load_c2_v3
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
