; SPIR-V Geometry Shader: Explode Triangles
               OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorate Interface
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 0
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
      %v3float = OpTypeVector %float 3
      %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
      %uint_3 = OpConstant %uint 3
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
     %float_0_08 = OpConstant %float 0.08
     %float_0_0 = OpConstant %float 0.0

; Pointer Types
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; PerVertex Blocks
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Attributes
%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
%inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Main Function
       %main = OpFunction %void None %func
      %label = OpLabel

; 1. Read positions
      %ptr0 = OpAccessChain %ptr_in_arr_pv %gl_in %int_0
      %pv0 = OpLoad %gl_PerVertex_in %ptr0
      %p0 = OpCompositeExtract %v4float %pv0 0
      
      %ptr1 = OpAccessChain %ptr_in_arr_pv %gl_in %int_1
      %pv1 = OpLoad %gl_PerVertex_in %ptr1
      %p1 = OpCompositeExtract %v4float %pv1 0
      
      %ptr2 = OpAccessChain %ptr_in_arr_pv %gl_in %int_2
      %pv2 = OpLoad %gl_PerVertex_in %ptr2
      %p2 = OpCompositeExtract %v4float %pv2 0

; 2. Compute face normal
      %p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
      %p1_v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
      %p2_v3 = OpVectorShuffle %v3float %p2 %p2 0 1 2
      
      %e1 = OpFSub %v3float %p1_v3 %p0_v3
      %e2 = OpFSub %v3float %p2_v3 %p0_v3
      %cross = OpExtInst %v3float %1 Cross %e1 %e2
      %normal = OpExtInst %v3float %1 Normalize %cross
      
; 3. Offset = N * 0.08
      %offset = OpVectorTimesScalar %v3float %normal %float_0_08
      %offset4 = OpCompositeConstruct %v4float %offset %float_0.0
      
; Helper to emit a vertex
; Input: vertex position (v4), color index (int)
; Uses global p0, p1, p2, offset4, inColor, outColor, gl_out
; Since assembly needs to repeat this for 3 vertices,
; I will emit them directly.

; Emit 0
      %p0_new = OpFAdd %v4float %p0 %offset4
      %ptr_out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %ptr_out_pos0 %p0_new
      %ptr_in_color0 = OpAccessChain %ptr_in_v3 %inColor %int_0
      %c0 = OpLoad %v3float %ptr_in_color0
      OpStore %outColor %c0
      OpEmitVertex

; Emit 1
      %p1_new = OpFAdd %v4float %p1 %offset4
      %ptr_out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %ptr_out_pos1 %p1_new
      %ptr_in_color1 = OpAccessChain %ptr_in_v3 %inColor %int_1
      %c1 = OpLoad %v3float %ptr_in_color1
      OpStore %outColor %c1
      OpEmitVertex

; Emit 2
      %p2_new = OpFAdd %v4float %p2 %offset4
      %ptr_out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
      OpStore %ptr_out_pos2 %p2_new
      %ptr_in_color2 = OpAccessChain %ptr_in_v3 %inColor %int_2
      %c2 = OpLoad %v3float %ptr_in_color2
      OpStore %outColor %c2
      OpEmitVertex
      
      OpEndPrimitive
      OpReturn
      OpFunctionEnd
