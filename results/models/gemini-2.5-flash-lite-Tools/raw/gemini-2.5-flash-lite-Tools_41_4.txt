; --- 1. Capabilities ---
OpCapability Shader

; --- 3. Extended Instruction Sets ---
%1 = OpExtInstImport "GLSL.std.450"

; --- 4. Memory Model ---
OpMemoryModel Logical GLSL450

; --- 5. Entry Points ---
; %main is the function, "main" is the string name, %outColor is an interface variable
OpEntryPoint Fragment %main "main" %outColor

; --- 6. Execution Modes ---
OpExecutionMode %main OriginUpperLeft

; --- 7. Debug Info ---
OpName %main "main"
OpName %outColor "outColor"

; --- 8. Annotations ---
OpDecorate %outColor Location 0

; --- 9. Types, Constants, and Globals ---
%void      = OpTypeVoid
%func_type = OpTypeFunction %void
%float     = OpTypeFloat 32
%v4float   = OpTypeVector %float 4

; Constants for the color (1.0, 0.0, 0.0, 1.0)
%f1        = OpConstant %float 1.0
%f0        = OpConstant %float 0.0
%red_color = OpConstantComposite %v4float %f1 %f0 %f0 %f1

; Output variable declaration
%ptr_out   = OpTypePointer Output %v4float
%outColor  = OpVariable %ptr_out Output

; --- 10. Functions ---
%main = OpFunction %void None %func_type
%label = OpLabel
         ; Store the constant color into the output variable
         OpStore %outColor %red_color
         OpReturn
OpFunctionEnd
