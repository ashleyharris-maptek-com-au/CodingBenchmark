struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal
    float3 normalizedNormal = normalize(input.normal);

    // Calculate light vector
    float3 lightVec = lightPos.xyz - input.worldPos;
    float3 normalizedLightVec = normalize(lightVec);

    // Calculate diffuse factor (Lambertian)
    float diffuse = max(0.0, dot(normalizedNormal, normalizedLightVec));
    float lambertianFactor = 0.3 + 0.7 * diffuse;

    // Base color
    float3 baseColor = float3(0.8, 0.2, 0.2) * lambertianFactor;

    // Calculate distance from camera to worldPos
    float dist = length(cameraPos.xyz - input.worldPos);

    // Calculate fog factor
    float fogFactor = clamp((dist - 1.5) / 3.0, 0.0, 1.0);

    // Fog color
    float3 fogColor = float3(0.7, 0.7, 0.8);

    // Lerp between base color and fog color
    float3 finalColor = lerp(baseColor, fogColor, fogFactor);

    return float4(finalColor, 1.0);
}
