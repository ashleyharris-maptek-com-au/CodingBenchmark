// HLSL pixel shader for Vulkan
// Complied using DXC to SPIR-V.

struct PSInput {
    float4 position : SV_POSITION;
    float4 color : COLOR;
};

struct PSOutput {
    float4 color : SV_TARGET;
};

// Map to Vulkan descriptor set 0, binding 0
[[vk::binding(0, 0)]]
Texture2D mainTexture : register(t0);

// Map to Vulkan sampler
[[vk::binding(0, 1)]]
SamplerState mainSampler : register(s0);

PSOutput main(PSInput input) {
    PSOutput output;
    
    // Sample texture
    float4 texColor = mainTexture.Sample(mainSampler, input.color.xy);
    
    // Output color
    output.color = texColor * input.color;
    
    return output;
}
