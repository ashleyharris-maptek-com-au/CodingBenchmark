    OpLabel

    %uv_val = OpLoad %v2float %uv
    %tangent_val = OpLoad %v3float %tangent
    %color_val = OpLoad %v3float %color

    ; --- Normalize normal ---

    ; --- Hemisphere Ambient Lighting ---
    ; Blend factor: n.y * 0.5 + 0.5
    %dot_n_y = OpDot %float %n_normalized %v3_010 ; Dot product with (0,1,0) gives n.y
    %blend_step1 = OpFMul %float %dot_n_y %float_0_5
    %blend = OpFAdd %float %blend_step1 %float_0_5

    ; Mix sky and ground colors using GLSL.std.450.Mix
    %ambient_rgb = OpExtInst %v3float %glsl_std_450 Mix %v4_groundcolor %v4_skycolor %blend

    ; --- Small Diffuse Term ---
    ; Load UBO params and extract params.y
    %ubo_params_ptr = OpAccessChain %_ptr Uniform %ubo_var %int_5 ; Index 5 for params
    %ubo_params_val = OpLoad %v4float %ubo_params_ptr
    %params_y = OpCompositeExtract %float %ubo_params_val 1 ; Extract y component

    ; Calculate diffuse term: 0.3 * params.y
    %diffuse_term_scalar = OpFMul %float %float_0_3 %params_y

    ; Broadcast diffuse term scalar to vec3 for addition
    %diffuse_term_vec = OpCompositeConstruct %v3float %diffuse_term_scalar %diffuse_term_scalar %diffuse_term_scalar

    ; --- Combine Ambient and Diffuse ---
    %final_rgb = OpFAdd %v3float %ambient_rgb %diffuse_term_vec

    ; --- Set Alpha ---
    %final_rgba = OpCompositeConstruct %v4float %final_rgb %float_1_0 ; Set alpha to 1.0

    ; --- Store Output ---
    OpStore %outColor %final_rgba

    OpReturn
OpFunctionEnd
