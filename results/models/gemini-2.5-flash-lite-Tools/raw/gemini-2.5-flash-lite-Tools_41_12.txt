OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft

; Decorations (Required for Vulkan/OpenGL interface)
               OpDecorate %outColor Location 0

; Types and Variables
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
%ptr_v4float = OpTypePointer Output %v4float
   %outColor = OpVariable %ptr_v4float Output

; Constants (Red: 1.0, 0.0, 0.0, 1.0)
    %float_1 = OpConstant %float 1
    %float_0 = OpConstant %float 0
      %color = OpConstantComposite %v4float %float_1 %float_0 %float_0 %float_1

; Main Function
       %main = OpFunction %void None %3
      %label = OpLabel
               OpStore %outColor %color
               OpReturn
               OpFunctionEnd
