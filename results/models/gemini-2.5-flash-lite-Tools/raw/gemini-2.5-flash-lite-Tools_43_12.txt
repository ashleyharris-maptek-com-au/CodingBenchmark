// Simple HLSL Pixel Shader for Vulkan
// Compiled with: dxc -spirv -T ps_6_0 -fspv-target-env=vulkan1.2 -E main shader.hlsl -Fo shader.spv

struct PSInput {
    float4 pos : SV_POSITION;
    [[vk::location(0)]] float2 uv : TEXCOORD0;
};

// Define resources for Descriptor Set 0
[[vk::binding(0, 0)]] Texture2D mainTex : register(t0);
[[vk::binding(1, 0)]] SamplerState mainSampler : register(s0);

// Simple pixel shader entry point
float4 main(PSInput input) : SV_Target {
    // Sample texture and output color
    float4 color = mainTex.Sample(mainSampler, input.uv);
    return color;
}
