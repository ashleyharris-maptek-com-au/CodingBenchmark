struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float3 normalizedNormal = normalize(input.normal);

    // Hemisphere ambient lighting
    float3 skyColor = float3(0.4, 0.6, 1.0);
    float3 groundColor = float3(0.3, 0.15, 0.05);

    float blendFactor = normalizedNormal.y * 0.5 + 0.5;
    float3 ambientColor = lerp(groundColor, skyColor, blendFactor);

    // Small diffuse term
    float3 diffuseTerm = 0.3 * input.color;

    float3 finalColor = ambientColor + diffuseTerm;

    return float4(finalColor, 1.0);
}
