OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_Position_in %worldPos_in %normal_in %color_in %gl_Position_out %outColor_out
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Debug info (optional, but good practice for readability)
OpSource GLSL 450
OpSourceExtension "GL_ARB_separate_shader_objects"
OpName %main "main"
OpName %gl_Position_in "gl_Position"
OpName %worldPos_in "worldPos"
OpName %normal_in "normal"
OpName %color_in "color"
OpName %gl_Position_out "gl_Position"
OpName %outColor_out "outColor"

; Decorations for BuiltIns and Locations
OpDecorate %gl_Position_in BuiltIn Position
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %gl_Position_out BuiltIn Position
OpDecorate %outColor_out Location 0

; Types
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%bool = OpTypeBool
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; gl_PerVertex_in block
%gl_PerVertex_in_type = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_in_type 0 BuiltIn Position
OpDecorate %gl_PerVertex_in_type Block

; gl_PerVertex_out block
%gl_PerVertex_out_type = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_out_type 0 BuiltIn Position
OpDecorate %gl_PerVertex_out_type Block

; Pointers
%ptr_input_v4float = OpTypePointer Input %v4float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v3float = OpTypePointer Output %v3float
%ptr_function_int = OpTypePointer Function %int
%ptr_function_float = OpTypePointer Function %float
%ptr_function_v3float = OpTypePointer Function %v3float
%ptr_uniform_mat4 = OpTypePointer Uniform %mat4
%ptr_uniform_v4float = OpTypePointer Uniform %v4float

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%uint_10 = OpConstant %uint 10
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0_0 = OpConstant %float 0.0
%float_1_0 = OpConstant %float 1.0
%float_0_02 = OpConstant %float 0.02
%float_0_25 = OpConstant %float 0.25
%float_neg_1_0 = OpConstant %float -1.0
%v3float_0_0_0_const = OpConstantComposite %v3float %float_0_0 %float_0_0 %float_0_0
%v4float_0_0_0_1_const = OpConstantComposite %v4float %float_0_0 %float_0_0 %float_0_0 %float_1_0

; Input arrays for gl_Position
%arr_gl_PerVertex_in_4 = OpTypeArray %gl_PerVertex_in_type %uint_4
%ptr_input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in_4
%gl_Position_in = OpVariable %ptr_input_arr_gl_PerVertex_in Input

; Input arrays for user-defined attributes
%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_input_arr_v3float_4 = OpTypePointer Input %arr_v3float_4
%worldPos_in = OpVariable %ptr_input_arr_v3float_4 Input
%normal_in = OpVariable %ptr_input_arr_v3float_4 Input
%color_in = OpVariable %ptr_input_arr_v3float_4 Input

; Output variables
%gl_Position_out = OpVariable %ptr_output_v4float Output
%outColor_out = OpVariable %ptr_output_v3float Output

; UBO (set=0, binding=0) - same layout as test 41
%type_ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
OpMemberDecorate %type_ubo_struct 0 ColMajor
OpMemberDecorate %type_ubo_struct 0 MatrixStride 16
OpMemberDecorate %type_ubo_struct 0 Offset 0
OpMemberDecorate %type_ubo_struct 1 ColMajor
OpMemberDecorate %type_ubo_struct 1 MatrixStride 16
OpMemberDecorate %type_ubo_struct 1 Offset 64
OpMemberDecorate %type_ubo_struct 2 ColMajor
OpMemberDecorate %type_ubo_struct 2 MatrixStride 16
OpMemberDecorate %type_ubo_struct 2 Offset 128
OpMemberDecorate %type_ubo_struct 3 Offset 192
OpMemberDecorate %type_ubo_struct 4 Offset 208
OpMemberDecorate %type_ubo_struct 5 Offset 224
OpDecorate %type_ubo_struct Block

%ptr_ubo_type_ubo_struct = OpTypePointer Uniform %type_ubo_struct
%ubo = OpVariable %ptr_ubo_type_ubo_struct Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; UBO member pointers (for consistency with test 41, though not all are used)
%ubo_model_ac = OpAccessChain %ptr_uniform_mat4 %ubo %uint_0
%ubo_view_ac = OpAccessChain %ptr_uniform_mat4 %ubo %uint_1
%ubo_proj_ac = OpAccessChain %ptr_uniform_mat4 %ubo %uint_2
%ubo_lightPos_ac = OpAccessChain %ptr_uniform_v4float %ubo %uint_3
%ubo_cameraPos_ac = OpAccessChain %ptr_uniform_v4float %ubo %uint_4
%ubo_params_ac = OpAccessChain %ptr_uniform_v4float %ubo %uint_5

; Function main
%main = OpFunction %void None %void_func
%entry = OpLabel

; Load input worldPos (P0, P1, P2, P3) - used for geometric calculations (despite "clip space" label)
%worldPos0_ptr = OpAccessChain %ptr_input_v3float %worldPos_in %uint_0
%worldPos1_ptr = OpAccessChain %ptr_input_v3float %worldPos_in %uint_1
%worldPos2_ptr = OpAccessChain %ptr_input_v3float %worldPos_in %uint_2
%worldPos3_ptr = OpAccessChain %ptr_input_v3float %worldPos_in %uint_3

%worldPos0_v3 = OpLoad %v3float %worldPos0_ptr
%worldPos1_v3 = OpLoad %v3float %worldPos1_ptr
%worldPos2_v3 = OpLoad %v3float %worldPos2_ptr
%worldPos3_v3 = OpLoad %v3float %worldPos3_ptr

; Load input colors for interpolation (Color1, Color2)
%color1_ptr = OpAccessChain %ptr_input_v3float %color_in %uint_1
%color2_ptr = OpAccessChain %ptr_input_v3float %color_in %uint_2

%color1_v3 = OpLoad %v3float %color1_ptr
%color2_v3 = OpLoad %v3float %color2_ptr

; Tangents for smooth perpendiculars
; T_start = P2 - P0
; T_end = P3 - P1
%T_start_v3 = OpFSub %v3float %worldPos2_v3 %worldPos0_v3
%T_end_v3 = OpFSub %v3float %worldPos3_v3 %worldPos1_v3

; Thickness constant
%thickness_val = OpCopyObject %float %float_0_02

; Loop variable i (0 to 4)
%loop_var_i = OpVariable %ptr_function_int Function
OpStore %loop_var_i %int_0

%loop_header = OpLabel
%i_val = OpLoad %int %loop_var_i
%loop_continue_cond = OpSLessThan %bool %i_val %int_5
OpLoopMerge %loop_end %loop_body None
OpBranchConditional %loop_continue_cond %loop_body %loop_end

%loop_body = OpLabel
; Calculate t (0.0, 0.25, 0.5, 0.75, 1.0)
%i_float = OpConvertSToF %float %i_val
%t_val = OpFMul %float %i_float %float_0_25

; Interpolate Position (linear between P1 and P2)
%one_minus_t = OpFSub %float %float_1_0 %t_val
%interp_P1_term = OpVectorTimesScalar %v3float %worldPos1_v3 %one_minus_t
%interp_P2_term = OpVectorTimesScalar %v3float %worldPos2_v3 %t_val
%interp_pos_v3 = OpFAdd %v3float %interp_P1_term %interp_P2_term

; Interpolate Tangent (linear between T_start and T_end for perpendicular direction)
%interp_T_start_term = OpVectorTimesScalar %v3float %T_start_v3 %one_minus_t
%interp_T_end_term = OpVectorTimesScalar %v3float %T_end_v3 %t_val
%interp_tangent_v3 = OpFAdd %v3float %interp_T_start_term %interp_T_end_term
%norm_interp_tangent_v3 = OpNormalize %v3float %interp_tangent_v3

; Calculate Perpendicular in XY plane (for a 2D-like thickness effect)
%tangent_x = OpCompositeExtract %float %norm_interp_tangent_v3 %int_0
%tangent_y = OpCompositeExtract %float %norm_interp_tangent_v3 %int_1
%neg_tangent_y = OpFNegate %float %tangent_y
%perp_xy_v3 = OpCompositeConstruct %v3float %neg_tangent_y %tangent_x %float_0_0

; Normalize and scale by thickness
%norm_perp_xy_v3 = OpNormalize %v3float %perp_xy_v3
%scaled_perp_v3 = OpVectorTimesScalar %v3float %norm_perp_xy_v3 %thickness_val

; Calculate pos + perp and pos - perp (still in vec3 clip space)
%pos_plus_perp_v3 = OpFAdd %v3float %interp_pos_v3 %scaled_perp_v3
%pos_minus_perp_v3 = OpFSub %v3float %interp_pos_v3 %scaled_perp_v3

; Convert final vec3 positions to vec4 (gl_Position expects vec4)
%final_pos_plus_perp_v4 = OpCompositeConstruct %v4float %pos_plus_perp_v3 %float_1_0
%final_pos_minus_perp_v4 = OpCompositeConstruct %v4float %pos_minus_perp_v3 %float_1_0

; Interpolate Color (linear between color[1] and color[2])
%interp_color1_term = OpVectorTimesScalar %v3float %color1_v3 %one_minus_t
%interp_color2_term = OpVectorTimesScalar %v3float %color2_v3 %t_val
%interp_color_v3 = OpFAdd %v3float %interp_color1_term %interp_color2_term

; Emit pos + perp vertex
OpStore %gl_Position_out %final_pos_plus_perp_v4
OpStore %outColor_out %interp_color_v3
OpEmitVertex

; Emit pos - perp vertex
OpStore %gl_Position_out %final_pos_minus_perp_v4
OpStore %outColor_out %interp_color_v3
OpEmitVertex

; Increment loop variable
%i_next = OpIAdd %int %i_val %int_1
OpStore %loop_var_i %i_next
OpBranch %loop_header

%loop_end = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
