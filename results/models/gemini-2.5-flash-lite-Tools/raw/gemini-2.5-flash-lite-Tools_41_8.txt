; SPIR-V
; Version: 1.0
; Generator: Gemini CLI;
; Target: Vulkan 1.0

OpCapability Shader
OpExtension "SPV_GOOGLE_hlsl_functionality1"
OpMemoryModel Logical GLSL450

; --- Constants ---
%float_0_0 = Constant %3 0.0
%float_0_2 = Constant %3 0.2
%float_0_25 = Constant %3 0.25
%float_0_4 = Constant %3 0.4
%float_0_5 = Constant %3 0.5
%float_0_7 = Constant %3 0.7
%float_0_75 = Constant %3 0.75
%float_1_0 = Constant %3 1.0

; --- Types ---
%void = TypeVoid
%bool = TypeBool
%float = TypeFloat 32
%int = TypeInt 32 0
%v2f = TypeVector %float 2
%v3f = TypeVector %float 3
%v4f = TypeVector %float 4
%mat4 = TypeMatrix %v4f 4

; UBO Struct Definition
%UboStruct = TypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ubo_ptr_type = TypePointer Uniform %UboStruct ; Pointer to UBO struct

; --- Variables ---
; Input Variables
%worldPos_in = Variable Input %v3f
%normal_in = Variable Input %v3f
%uv_in = Variable Input %v2f
%tangent_in = Variable Input %v3f
%color_in = Variable Input %v4f

; Output Variable
%outColor_out = Variable Output %v4f

; Uniform Buffer Variable
%ubo_buffer = Variable Uniform %UboStruct

; --- Entry Point Function ---
%main = Function %void None %{
  %label_entry = Label

  ; Load Input Variables
  %worldPos = Load %v3f %worldPos_in
  %normal = Load %v3f %normal_in
  %uv = Load %v2f %uv_in
  %tangent = Load %v3f %tangent_in
  %vertexColor = Load %v4f %color_in

  ; Load Uniform Buffer
  %ubo_access_chain = AccessChain %ubo_ptr_type %ubo_buffer
  %ubo = Load %UboStruct %ubo_access_chain
  %lightPos = CompositeExtract %v4f %ubo 3 ; offset 192

  ; --- Calculations ---
  ; Light Direction: L = normalize(worldPos - lightPos.xyz)
  %lightPos_xyz = CompositeExtract %v3f %lightPos 0
  %worldPos_minus_lightPos = Composite tristeza %v3f %worldPos %lightPos_xyz
  %lightDir = Normalize %v3f %worldPos_minus_lightPos

  ; Normalize Normal Vector
  %normal_norm = Normalize %v3f %normal

  ; Diffuse Term: d = max(dot(N, L), 0.0)
  %dot_nl = Dot %float %normal_norm %lightDir
  %diffuse = FMax %float %dot_nl %float_0_0

  ; Toon Shading Quantization
  ; Thresholds: 0.75, 0.5, 0.25
  ; Values: 1.0, 0.7, 0.4, 0.2

  ; Logic:
  ; if diffuse > 0.75: 1.0
  ; else if diffuse > 0.5: 0.7
  ; else if diffuse > 0.25: 0.4
  ; else: 0.2

  ; Use cascaded Select statements
  ; Start with the lowest threshold condition
  %cond_gt_025 = FOrdGreaterThan %bool %diffuse %float_0_25
  %banded_diffuse = Select %float %float_0_2 %float_0_4 %cond_gt_025 ; If diffuse > 0.25, use 0.4, else use 0.2

  %cond_gt_05 = FOrdGreaterThan %bool %diffuse %float_0_5
  %banded_diffuse = Select %float %banded_diffuse %float_0_7 %cond_gt_05 ; If diffuse > 0.5, use 0.7, else use current value (0.2 or 0.4)

  %cond_gt_075 = FOrdGreaterThan %bool %diffuse %float_0_75
  %banded_diffuse = Select %float %banded_diffuse %float_1_0 %cond_gt_075 ; If diffuse > 0.75, use 1.0, else use current value

  ; Multiply vertex color by diffuse band
  %final_color_rgb = FMul %v3f %banded_diffuse %vertexColor.xyz
  %final_color_rgba = CompositeConstruct %v4f %final_color_rgb %float_1_0 ; Output alpha is 1.0

  ; Store Output
  Store %v4f %final_color_rgba %outColor_out

  Return
%}

; --- Decorations ---

OpMemberDecorate %UboStruct %v4f offset 208
