; SPIR-V Geometry Shader Example
; This shader takes Triangles as input and outputs TriangleStrips.

               OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %in_pos %out_pos
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorate interface variables (e.g., location 0)
               OpDecorate %in_pos Location 0
               OpDecorate %out_pos Location 0

; Types
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
      %vec4  = OpTypeVector %float 4
       %int  = OpTypeInt 32 0
       %bool = OpTypeBool
; Input: Array of 3 vectors (for a triangle)
   %vec4_arr = OpTypeArray %vec4 %3
       %3    = OpConstant %int 3
; Input/Output variable pointers
  %ptr_in    = OpTypePointer Input %vec4_arr
  %ptr_vec4  = OpTypePointer Input %vec4
  %ptr_out   = OpTypePointer Output %vec4

; Globals
     %in_pos = OpVariable %ptr_in Input
    %out_pos = OpVariable %ptr_out Output

; Constants for loop
   %zero = OpConstant %int 0
     %one = OpConstant %int 1
     %two = OpConstant %int 2

; Main function
       %main = OpFunction %void None %func
 %label_main = OpLabel
               OpBranch %loop_header

%loop_header = OpLabel
         %i  = OpPhi %int %zero %label_main %i_next %loop_body
         %cond = OpULessThan %bool %i %3
               OpBranchConditional %cond %loop_body %loop_exit

%loop_body   = OpLabel
; Get pointer to current input vertex
   %in_ptr  = OpAccessChain %ptr_vec4 %in_pos %i
   %val     = OpLoad %vec4 %in_ptr
; Set output vertex position
               OpStore %out_pos %val
; Emit vertex
               OpEmitVertex
; Increment index
   %i_next  = OpIAdd %int %i %one
               OpBranch %loop_header

%loop_exit   = OpLabel
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
