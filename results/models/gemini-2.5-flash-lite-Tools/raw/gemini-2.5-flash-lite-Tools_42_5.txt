; Geometry Shader SPIR-V Assembly

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; --- Types ---
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%float_0 = OpConstant %float 0.0
%float_0_015 = OpConstant %float 0.015
%float_0_75 = OpConstant %float 0.75
%float_0_05 = OpConstant %float 0.05
%float_1 = OpConstant %float 1.0

; --- Inputs ---
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

; --- Outputs ---
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

OpDecorate %outColor Location 0
%outColor = OpVariable %ptr_out_v3 Output
%ptr_out_v3 = OpTypePointer Output %v3float

; --- Main Function ---
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%label = OpLabel

; Load positions and colors (simplified)
%ptr0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr0
%ptr1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr1

%p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%dir = OpFSub %v3float %p1_v3 %p0_v3
%perp = OpVectorNormalize %v3float (OpCompositeConstruct %v3float (OpFNegate %float (OpCompositeExtract %float %dir 1)) (OpCompositeExtract %float %dir 0) %float_0)

%thickness = OpVectorTimesScalar %v3float %perp %float_0_015
%mid = OpFAdd %v3float %p0_v3 (OpVectorTimesScalar %v3float %dir %float_0_75)
%arrow_perp = OpVectorTimesScalar %v3float %perp %float_0_05

; --- Emit Shaft ---
; Vertex 1: p0 - thickness
%pos1 = OpCompositeConstruct %v4float (OpFSub %v3float %p0_v3 %thickness) %float_1
%ptr_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_pos1 %pos1
OpEmitVertex

; Vertex 2: p0 + thickness
%pos2 = OpCompositeConstruct %v4float (OpFAdd %v3float %p0_v3 %thickness) %float_1
OpStore %ptr_pos1 %pos2
OpEmitVertex

; Vertex 3: mid - thickness
%pos3 = OpCompositeConstruct %v4float (OpFSub %v3float %mid %thickness) %float_1
OpStore %ptr_pos1 %pos3
OpEmitVertex

; Vertex 4: mid + thickness
%pos4 = OpCompositeConstruct %v4float (OpFAdd %v3float %mid %thickness) %float_1
OpStore %ptr_pos1 %pos4
OpEmitVertex

OpEndPrimitive

; --- Emit Arrowhead ---
; Vertex 5: mid + arrow_perp
%pos5 = OpCompositeConstruct %v4float (OpFAdd %v3float %mid %arrow_perp) %float_1
OpStore %ptr_pos1 %pos5
OpEmitVertex

; Vertex 6: mid - arrow_perp
%pos6 = OpCompositeConstruct %v4float (OpFSub %v3float %mid %arrow_perp) %float_1
OpStore %ptr_pos1 %pos6
OpEmitVertex

; Vertex 7: p1
%pos7 = OpCompositeConstruct %v4float %p1_v3 %float_1
OpStore %ptr_pos1 %pos7
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
