// scheduler.js
const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);

  const machineCount = data.machineCount;
  const tasks = data.tasks;

  // Build adjacency list and in-degrees for topological sort
  const graph = new Map(); // id -> [successors]
  const inDegrees = new Map(); // id -> count
  const taskMap = new Map(); // id -> task object

  tasks.forEach(task => {
    graph.set(task.id, []);
    inDegrees.set(task.id, 0);
    taskMap.set(task.id, task);
  });

  tasks.forEach(task => {
    task.deps.forEach(depId => {
      graph.get(depId).push(task.id);
      inDegrees.set(task.id, inDegrees.get(task.id) + 1);
    });
  });

  // Priority queue for tasks ready to be scheduled
  // Prioritize tasks:
  // 1. Earlier deadline
  // 2. Higher weight (tie-breaker)
  const readyQueue = []; // This will act as a min-priority queue for deadlines, max-priority queue for weights

  tasks.forEach(task => {
    if (inDegrees.get(task.id) === 0) {
      readyQueue.push(task.id);
    }
  });

  // Custom sort function for the ready queue
  // Sort by deadline (ascending), then by weight (descending)
  readyQueue.sort((aId, bId) => {
    const a = taskMap.get(aId);
    const b = taskMap.get(bId);
    if (a.deadline !== b.deadline) {
      return a.deadline - b.deadline;
    }
    return b.weight - a.weight;
  });

  const order = [];

  while (readyQueue.length > 0) {
    // Pop the highest priority task (last element after sorting)
    const taskId = readyQueue.shift();
    order.push(taskId);

    graph.get(taskId).forEach(successorId => {
      inDegrees.set(successorId, inDegrees.get(successorId) - 1);
      if (inDegrees.get(successorId) === 0) {
        const successor = taskMap.get(successorId);
        const insertIndex = readyQueue.findIndex(currentInQueueId => {
          const currentInQueue = taskMap.get(currentInQueueId);
          if (successor.deadline !== currentInQueue.deadline) {
            return successor.deadline < currentInQueue.deadline;
          }
          return successor.weight > currentInQueue.weight;
        });
        if (insertIndex === -1) {
          readyQueue.push(successorId);
        } else {
          readyQueue.splice(insertIndex, 0, successorId);
        }
      }
    });
  }

  // Check for cycles (if order.length is not equal to tasks.length, there's a cycle)
  if (order.length !== tasks.length) {
    // This case should ideally not happen with valid input, but good for robustness
    // For now, if a cycle is detected, we might just output the partial order or throw an error.
    // The problem implies valid DAGs.
  }

  process.stdout.write(JSON.stringify({ order: order }));
}

solve();
