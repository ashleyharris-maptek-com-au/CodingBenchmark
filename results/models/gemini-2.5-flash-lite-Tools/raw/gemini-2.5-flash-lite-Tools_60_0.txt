const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  if (!input) return;
  const data = JSON.parse(input);
  
  const { machineCount, tasks } = data;
  
  // Calculate dependencies
  const taskMap = new Map();
  tasks.forEach(task => taskMap.set(task.id, task));
  
  // Topological sort / Priority heuristic
  // Simple heuristic: Latest finish time / Critical path
  // For now, let's do a simple dependency-aware ordering based on weights and deadlines.
  
  const inDegree = new Map();
  const adj = new Map();
  
  tasks.forEach(task => {
    inDegree.set(task.id, 0);
    adj.set(task.id, []);
  });
  
  tasks.forEach(task => {
    task.deps.forEach(depId => {
      adj.get(depId).push(task.id);
      inDegree.set(task.id, inDegree.get(task.id) + 1);
    });
  });
  
  const queue = [];
  tasks.forEach(task => {
    if (inDegree.get(task.id) === 0) {
      queue.push(task);
    }
  });
  
  // Priority: prioritize based on deadline and weight (EDF/Weighted)
  const order = [];
  while (queue.length > 0) {
    queue.sort((a, b) => {
      // Heuristic: prioritize sooner deadline, then higher weight
      if (a.deadline !== b.deadline) return a.deadline - b.deadline;
      return b.weight - a.weight;
    });
    
    const task = queue.shift();
    order.push(task.id);
    
    adj.get(task.id).forEach(nextId => {
      inDegree.set(nextId, inDegree.get(nextId) - 1);
      if (inDegree.get(nextId) === 0) {
        queue.push(taskMap.get(nextId));
      }
    });
  }
  
  console.log(JSON.stringify({ order }));
}

solve();
