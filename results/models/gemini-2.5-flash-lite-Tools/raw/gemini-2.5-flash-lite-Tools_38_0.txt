import sys

def exact_cover(matrix):
    """
    Solves the Exact Cover problem using Algorithm X.
    
    matrix: A dictionary where keys are column identifiers,
            and values are sets of row identifiers that cover that column.
            
    Returns a list of selected row identifiers, or None if no solution exists.
    """
    if not matrix:
        return []
    
    # Choose column with fewest rows (heuristic)
    column = min(matrix, key=lambda c: len(matrix[c]))
    
    if not matrix[column]:
        return None  # Dead end
    
    for row in list(matrix[column]):
        # Select this row
        solution = [row]
        
        # Create a copy of the matrix to modify
        new_matrix = {c: set(rows) for c, rows in matrix.items()}
        
        # Identify columns covered by this row
        # (We need to look up which columns this row covers)
        # To do this efficiently, we might need a better matrix representation
        # but for now, we scan.
        cols_to_remove = []
        for c, rows in new_matrix.items():
            if row in rows:
                cols_to_remove.append(c)
        
        # Remove columns and rows that conflict
        rows_to_remove = set()
        for c in cols_to_remove:
            # All rows that cover this column must be removed
            for r in new_matrix[c]:
                rows_to_remove.add(r)
            
            # The column itself is removed
            del new_matrix[c]
        
        # Remove the conflicting rows from all remaining columns
        for c in new_matrix:
            new_matrix[c] -= rows_to_remove
        
        # Recursively search
        sub_solution = exact_cover(new_matrix)
        
        if sub_solution is not None:
            return solution + sub_solution
            
    return None
