// Pixel Shader for Vulkan
// Compiled with: dxc -spirv -T ps_6_0 -E main pixel_shader.hlsl -Fo pixel_shader.spv

// --- Descriptor Set 0: Global Frame Data ---
struct SceneData {
    float4x4 viewProj;
    float3 cameraPos;
    float pad;
};
[[vk::binding(0, 0)]] ConstantBuffer<SceneData> scene : register(b0);

// --- Descriptor Set 1: Material/Object Data ---
struct MaterialData {
    float4 baseColor;
    float roughness;
    float metallic;
};
[[vk::binding(0, 1)]] ConstantBuffer<MaterialData> material : register(b1);

[[vk::binding(1, 1)]] Texture2D baseColorTexture : register(t0);
[[vk::binding(2, 1)]] SamplerState samplerState : register(s0);

// --- Input/Output Structs ---
struct PSInput {
    [[vk::location(0)]] float2 uv : TEXCOORD0;
    [[vk::location(1)]] float3 normal : NORMAL;
    [[vk::location(2)]] float3 worldPos : WORLDPOS;
    float4 pos : SV_Position;
};

struct PSOutput {
    [[vk::location(0)]] float4 color : SV_Target0;
};

// --- Entry Point ---
[shader("pixel")]
PSOutput main(PSInput input) {
    PSOutput output;

    // Simple Lighting (Placeholder for PBR)
    float3 normal = normalize(input.normal);
    float3 lightDir = normalize(float3(0.5, 1.0, 0.5));
    float diff = max(dot(normal, lightDir), 0.0);

    float4 texColor = baseColorTexture.Sample(samplerState, input.uv);
    float3 lighting = diff * float3(1.0, 1.0, 1.0); // Simple directional light

    output.color = float4(texColor.rgb * material.baseColor.rgb * lighting, texColor.a * material.baseColor.a);

    return output;
}
