// Pixel shader targeting Vulkan via SPIR-V (DXC)

struct PSInput {
    // SV_Position in PS input maps to gl_FragCoord in GLSL
    float4 position : SV_Position;
    
    // Explicit location matching the Vertex Shader output
    [[vk::location(0)]] float3 worldPos : POSITION;
    [[vk::location(1)]] float3 normal   : NORMAL;
    [[vk::location(2)]] float2 uv       : TEXCOORD0;
};

// Scene data constant buffer
struct SceneData {
    float4x4 viewProj;
    float3 cameraPos;
};
[[vk::binding(0, 0)]] ConstantBuffer<SceneData> sceneUbo : register(b0);

// Base color map and sampler
[[vk::binding(1, 0)]] Texture2D baseColorMap : register(t1);
[[vk::binding(2, 0)]] SamplerState linearSampler : register(s2);

// Push constants for tint
struct PushConstants {
    float4 tint;
};
[[vk::push_constant]] PushConstants pc;

// Entry Point
// SV_Target0 maps to the first color attachment (Location 0)
float4 main(PSInput input) : SV_Target {
    // Sample texture
    float3 color = baseColorMap.Sample(linearSampler, input.uv).rgb;
    
    // Simple lighting calculation
    float3 lightDir = normalize(float3(1.0, 1.0, 1.0));
    float diffuse = max(dot(normalize(input.normal), lightDir), 0.0);
    
    // Combine lighting and tint
    float3 finalColor = color * diffuse * pc.tint.rgb;
    
    return float4(finalColor, 1.0);
}
