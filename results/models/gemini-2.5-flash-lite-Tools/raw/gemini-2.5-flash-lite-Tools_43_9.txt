struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);

    // Calculate checkerboard pattern
    float2 uv_scaled = input.uv * 8.0;
    float2 uv_floor = floor(uv_scaled);
    float checker_pattern = fmod(uv_floor.x + uv_floor.y, 2.0); // 0.0 or 1.0

    float base_color_value;
    if (checker_pattern < 0.5) { // Effectively checker_pattern == 0.0
        base_color_value = 0.9;
    } else { // Effectively checker_pattern == 1.0
        base_color_value = 0.2;
    }
    float3 base_color = float3(base_color_value, base_color_value, base_color_value);

    // Lambertian shading
    float3 L_dir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L_dir));
    float lambertian_factor = 0.5 + 0.5 * diffuse;

    // Combine checkerboard color and Lambertian shading
    float3 final_color_rgb = base_color * lambertian_factor;

    return float4(final_color_rgb, 1.0);
}
