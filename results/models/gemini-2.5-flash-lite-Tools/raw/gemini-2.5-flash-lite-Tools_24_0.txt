#include <iostream>
#include <thread>
#include <string>
#include <vector>
#include "terrain.hpp"
#include "controller.hpp"

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(NULL);

    int width, depth, height;
    double gravity, max_thrust, fuel;
    std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel;

    double start_x, start_y, start_z, target_x, target_y, target_z;
    std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;

    std::string terrain_path;
    std::getline(std::cin >> std::ws, terrain_path);

    Terrain terrain;
    std::thread terrainThread([&]() {
        terrain.load(terrain_path, width, depth);
    });
    terrainThread.detach();

    Controller controller;

    while (true) {
        LanderState state;
        if (!(std::cin >> state.x >> state.y >> state.z >> state.vx >> state.vy >> state.vz >> 
                     state.pitch >> state.yaw >> state.roll >> 
                     state.pitch_rate >> state.yaw_rate >> state.roll_rate >> state.fuel)) {
            break;
        }
        controller.updateState(state);

        Command cmd = controller.calculateNextCommand();
        std::cout << cmd.thrust << " " << cmd.pitch_cmd << " " << cmd.yaw_cmd << " " << cmd.roll_cmd << std::endl;
    }

    return 0;
}
