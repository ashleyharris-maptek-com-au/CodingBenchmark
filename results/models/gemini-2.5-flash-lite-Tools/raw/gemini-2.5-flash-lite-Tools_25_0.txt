    public struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator *(Vector3 a, double s) => new Vector3(a.X * s, a.Y * s, a.Z * s);
        public double LengthSquared => X * X + Y * Y + Z * Z;
        public double Length => Math.Sqrt(LengthSquared);
    }

    public class Physics
    {
        private const double G = 6.67430e-11;
        public static void CalculateAccelerations(List<Body> bodies)
        {
            foreach (var b in bodies) b.Acceleration = new Vector3(0, 0, 0);
            for (int i = 0; i < bodies.Count; i++)
            {
                for (int j = i + 1; j < bodies.Count; j++)
                {
                    var b1 = bodies[i];
                    var b2 = bodies[j];
                    var diff = b2.Position - b1.Position;
                    var dist = diff.Length;
                    var force = (G * b1.Mass * b2.Mass) / (dist * dist * dist);
                    var accel = diff * (force / b1.Mass); // Actually this is for b1, needs adjustment
                    b1.Acceleration += diff * (G * b2.Mass / (dist * dist * dist));
                    b2.Acceleration -= diff * (G * b1.Mass / (dist * dist * dist));
                }
            }
        }
    }
