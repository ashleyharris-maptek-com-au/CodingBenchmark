; SPIR-V Fragment Shader
; Output: Solid Red

               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; --- Layout ---
               OpDecorate %outColor Location 0

; --- Types ---
       %void = OpTypeVoid
    %float32 = OpTypeFloat 32
    %vec4f32 = OpTypeVector %float32 4
       %func = OpTypeFunction %void

; --- Constants ---
    %float_1 = OpConstant %float32 1.0
    %float_0 = OpConstant %float32 0.0
    %red     = OpConstantComposite %vec4f32 %float_1 %float_0 %float_0 %float_1

; --- Variables ---
   %outColor = OpVariable %_ptr_Output_vec4f32 Output
%_ptr_Output_vec4f32 = OpTypePointer Output %vec4f32

; --- Functions ---
       %main = OpFunction %void None %func
      %label = OpLabel
               OpStore %outColor %red
               OpReturn
               OpFunctionEnd
