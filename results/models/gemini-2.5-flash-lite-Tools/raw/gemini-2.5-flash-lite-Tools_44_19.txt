#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize interpolated normal
    vec3 N = normalize(normal);

    // Calculate light direction
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);

    // Calculate view direction
    vec3 viewDir = normalize(ubo.cameraPos.xyz - worldPos);

    // Ambient component
    vec3 ambient = vec3(0.1, 0.05, 0.05);

    // Diffuse component
    float diff = max(dot(N, lightDir), 0.0);
    vec3 diffuse = color.rgb * diff;

    // Specular component
    vec3 reflectDir = reflect(-lightDir, N);
    float spec = pow(max(dot(reflectDir, viewDir), 0.0), 32.0);
    vec3 specular = vec3(1.0, 1.0, 1.0) * spec;

    // Combine all components and clamp
    vec3 finalColor = ambient + diffuse + specular;
    outColor = vec4(clamp(finalColor, 0.0, 1.0), 1.0);
}
