; --- 1. Capabilities & Imports ---
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

; --- 2. Entry Point & Execution Mode ---
; %main is the function ID, %out_color is an interface variable
OpEntryPoint Fragment %main "main" %out_color
OpExecutionMode %main OriginUpperLeft

; --- 3. Debug & Names ---
OpName %main "main"
OpName %out_color "outColor"

; --- 4. Annotations (Decorations) ---
; Assigns the output variable to Location 0
OpDecorate %out_color Location 0

; --- 5. Types, Constants, & Variables ---
%void      = OpTypeVoid
%func_type = OpTypeFunction %void
%float     = OpTypeFloat 32
%v4float   = OpTypeVector %float 4
%ptr_out   = OpTypePointer Output %v4float

; Global output variable
%out_color = OpVariable %ptr_out Output

; Constant color (Red: 1.0, 0.0, 0.0, 1.0)
%f1        = OpConstant %float 1.0
%f0        = OpConstant %float 0.0
%red       = OpConstantComposite %v4float %f1 %f0 %f0 %f1

; --- 6. Function Definition ---
%main = OpFunction %void None %func_type
%label = OpLabel
         OpStore %out_color %red
         OpReturn
         OpFunctionEnd
