; SPIR-V
; Version: 1.0
; Generator: Unknown
; Target: Vulkan 1.0

OpCapability Shader
OpMemoryModel Logical GLSL450

; Entrypoint
OpEntryPoint Fragment %main "main" %outColor
OpExecutionMode %main OriginUpperLeft

; Input variables
OpDecorate %worldPos Input
OpDecorate %normal Input
OpDecorate %uv Input
OpDecorate %tangent Input
OpDecorate %color Input
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4

; Output variable
OpDecorate %outColor Output
OpDecorate %outColor Location 0

; Uniform buffer
OpDecorate %_uniform_buffer Uniform
OpDecorate %_uniform_buffer Binding 0
OpDecorate %_uniform_buffer DescriptorSet 0
OpMemberDecorate %_uniform_buffer UniformBuffer
OpMemberDecorate %_uniform_buffer BufferStride 16
OpMemberDecorate %_uniform_buffer Offset 0
OpMemberDecorate %_uniform_buffer MatrixStride 16
OpMemberDecorate %_uniform_ubo Ubo
OpMemberDecorate %_uniform_ubo Offset 0

%_runtime_array_ubo = OpTypeRuntimeArray %_uniform_ubo
%_uniform_buffer_type = OpTypeStruct %_runtime_array_ubo
%_uniform_buffer = OpVariable %_uniform_buffer_type Uniform

%_uniform_ubo = OpTypeStruct ; UBO struct
%_uniform_ubo_model = OpTypeMatrix %_f32 4 4
%_uniform_ubo_view = OpTypeMatrix %_f32 4 4
%_uniform_ubo_projection = OpTypeMatrix %_f32 4 4
%_uniform_ubo_lightPos = OpTypeVector %_f32 4
%_uniform_ubo_cameraPos = OpTypeVector %_f32 4
%_uniform_ubo_params = OpTypeVector %_f32 4

%_uniform_ubo = OpTypeStruct %_uniform_ubo_model %_uniform_ubo_view %_uniform_ubo_projection %_uniform_ubo_lightPos %_uniform_ubo_cameraPos %_uniform_ubo_params

; Basic types
%_void = OpTypeVoid
%_f32 = OpTypeFloat 32
%_vec3 = OpTypeVector %_f32 3
%_vec4 = OpTypeVector %_f32 4

; Input variable types
%_in_worldPos = OpTypeVector %_f32 3
%_in_normal = OpTypeVector %_f32 3
%_in_uv = OpTypeVector %_f32 2
%_in_tangent = OpTypeVector %_f32 3
%_in_color = OpTypeVector %_f32 4

; Output variable type
%_outColor = OpTypeVector %_f32 4

; Main function
%main = OpFunction %_void None %_
  %label = OpLabel
    ; Load inputs
    %worldPos = OpLoad %_vec3 %worldPos
    %normal = OpLoad %_vec3 %normal
    %uv = OpLoad %_vec2 %uv
    %tangent = OpLoad %_vec3 %tangent
    %color = OpLoad %_vec4 %color

    ; Construct the output color: per-vertex color with alpha 1.0
    %const_alpha = OpConstant %_f32 1.0
    %outColor = OpCompositeConstruct %_vec4 %color %const_alpha

    ; Store output
    OpStore %outColor %outColor

    OpReturn
  OpFunctionEnd
