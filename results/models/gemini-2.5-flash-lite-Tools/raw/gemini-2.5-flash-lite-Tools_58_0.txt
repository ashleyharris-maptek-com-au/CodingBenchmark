const fs = require('fs');

/**
 * Calculates the shortest path distance between two points in a grid using BFS.
 * @param {string[]} grid 
 * @param {number} startX 
 * @param {number} startY 
 * @param {number} targetX 
 * @param {number} targetY 
 * @returns {number} Distance, or Infinity if unreachable.
 */
function getDistance(grid, startX, startY, targetX, targetY) {
    if (startX === targetX && startY === targetY) return 0;
    const rows = grid.length;
    const cols = grid[0].length;
    const queue = [[startX, startY, 0]];
    const visited = new Set();
    visited.add(`${startX},${startY}`);

    while (queue.length > 0) {
        const [x, y, dist] = queue.shift();

        const neighbors = [
            [x + 1, y], [x - 1, y], [x, y + 1], [x, y - 1]
        ];

        for (const [nx, ny] of neighbors) {
            if (nx >= 0 && nx < rows && ny >= 0 && ny < cols &&
                grid[nx][ny] === '.' && !visited.has(`${nx},${ny}`)) {
                
                if (nx === targetX && ny === targetY) return dist + 1;
                
                visited.add(`${nx},${ny}`);
                queue.push([nx, ny, dist + 1]);
            }
        }
    }
    return Infinity;
}

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const { grid, robots, jobs } = JSON.parse(input);

    const numRobots = robots.length;
    const assignments = Array.from({ length: numRobots }, () => []);
    
    // Create a working list of jobs
    let remainingJobs = jobs.map(job => ({...job}));

    // Simple greedy approach:
    // While jobs remain, find the robot that can reach a job with the shortest distance,
    // prioritizing higher priority jobs if possible.
    
    while (remainingJobs.length > 0) {
        let bestPair = null;
        let minDistance = Infinity;

        for (let i = 0; i < numRobots; i++) {
            const robot = robots[i];
            for (let j = 0; j < remainingJobs.length; j++) {
                const job = remainingJobs[j];
                const dist = getDistance(grid, robot.x, robot.y, job.x, job.y);
                
                // Heuristic: Prefer shorter distance, then higher priority
                if (dist < minDistance || (dist === minDistance && bestPair && job.priority > remainingJobs[bestPair.jobIdx].priority)) {
                    minDistance = dist;
                    bestPair = { robotIdx: i, jobIdx: j };
                }
            }
        }

        if (bestPair) {
            assignments[bestPair.robotIdx].push(remainingJobs[bestPair.jobIdx].id);
            remainingJobs.splice(bestPair.jobIdx, 1);
        } else {
            // Should not happen if all jobs are reachable
            break;
        }
    }

    console.log(JSON.stringify({ assignments }));
}

solve();
