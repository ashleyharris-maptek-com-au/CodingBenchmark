const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const grid = data.grid;
    const robots = data.robots;
    const jobs = data.jobs;

    const numRows = grid.length;
    const numCols = grid[0].length;

    // Helper to check if a cell is valid and not blocked
    function isValid(r, c) {
        return r >= 0 && r < numRows && c >= 0 && c < numCols && grid[r][c] === '.';
    }

    // BFS to find the shortest path distance between two points
    function bfs(startX, startY, endX, endY) {
        if (!isValid(startY, startX) || !isValid(endY, endX)) {
            return Infinity; // Cannot reach or start from an invalid cell
        }
        if (startX === endX && startY === endY) {
            return 0; // Same start and end point
        }

        const queue = [[startY, startX, 0]]; // [row, col, distance]
        const visited = new Set();
        visited.add(`${startY},${startX}`);

        const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]]; // Right, Left, Down, Up

        while (queue.length > 0) {
            const [r, c, dist] = queue.shift();

            for (const [dr, dc] of directions) {
                const newR = r + dr;
                const newC = c + dc;

                if (isValid(newR, newC) && !visited.has(`${newR},${newC}`)) {
                    if (newR === endY && newC === endX) {
                        return dist + 1;
                    }
                    visited.add(`${newR},${newC}`);
                    queue.push([newR, newC, dist + 1]);
                }
            }
        }
        return Infinity; // No path found
    }

    // --- Main Assignment Logic ---
    const robotAssignments = Array(robots.length).fill(null).map(() => []);
    const robotCurrentPositions = robots.map(robot => ({ x: robot.x, y: robot.y }));
    const unassignedJobs = new Set(jobs.map(job => job.id));

    // Map job IDs to their actual job objects for easy lookup
    const jobMap = new Map(jobs.map(job => [job.id, job]));

    while (unassignedJobs.size > 0) {
        let bestAssignment = null;
        let minDistance = Infinity;

        // Iterate through all unassigned jobs and all robots to find the best next assignment
        for (const jobId of unassignedJobs) {
            const job = jobMap.get(jobId);
            for (let i = 0; i < robots.length; i++) {
                const robotPos = robotCurrentPositions[i];
                const distance = bfs(robotPos.x, robotPos.y, job.x, job.y);

                if (distance < minDistance) {
                    minDistance = distance;
                    bestAssignment = { robotIndex: i, jobId: jobId, jobX: job.x, jobY: job.y };
                }
            }
        }

        if (bestAssignment) {
            robotAssignments[bestAssignment.robotIndex].push(bestAssignment.jobId);
            robotCurrentPositions[bestAssignment.robotIndex] = { x: bestAssignment.jobX, y: bestAssignment.jobY };
            unassignedJobs.delete(bestAssignment.jobId);
        } else {
            // This case should ideally not happen if all jobs are reachable
            // For robustness, if no assignment can be made, break to prevent infinite loop
            break;
        }
    }

    const output = {
        assignments: robotAssignments
    };

    console.log(JSON.stringify(output));
}

solve();
