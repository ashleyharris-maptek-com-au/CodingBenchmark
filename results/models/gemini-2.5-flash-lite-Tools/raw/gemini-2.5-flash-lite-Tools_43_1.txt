struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

struct UBO {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

[[vk::binding(0, 0)]]
cbuffer UBO_binding : register(b0) {
    UBO ubo;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the normal vector as it may be interpolated
    float3 normalizedNormal = normalize(input.normal);

    // Map the normal from [-1, 1] to [0, 1] for color visualization
    float3 color = normalizedNormal * 0.5 + 0.5;

    // Output the color with alpha 1.0
    return float4(color, 1.0);
}
