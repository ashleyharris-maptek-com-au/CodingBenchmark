; A simple SPIR-V assembly fragment shader that outputs a solid red color.

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft

               ; Debug information
               OpName %main "main"
               OpName %outColor "outColor"

               ; Decorations
               OpDecorate %outColor Location 0

               ; Types, Constants, and Variables
      %void  = OpTypeVoid
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %ptr_out = OpTypePointer Output %v4float
   %outColor = OpVariable %ptr_out Output
    %float_1 = OpConstant %float 1
    %float_0 = OpConstant %float 0
      %color = OpConstantComposite %v4float %float_1 %float_0 %float_0 %float_1

               ; Function Body
      %main  = OpFunction %void None %void
     %entry  = OpLabel
               OpStore %outColor %color
               OpReturn
               OpFunctionEnd
