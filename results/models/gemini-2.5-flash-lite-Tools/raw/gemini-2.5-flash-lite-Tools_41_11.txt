; SPIR-V
OpCapability Shader
OpMemoryModel Logical GLSL450

; --- Types ---
%void_type = OpTypeVoid
%bool_type = OpTypeBool
%int_type = OpTypeInt 32 0
%float_type = OpTypeFloat 32
%vec2_type = OpTypeVector %float_type 2
%vec3_type = OpTypeVector %float_type 3
%vec4_type = OpTypeVector %float_type 4
%mat4_type = OpTypeMatrix %float_type 4

; --- Constants ---
%const_0_0 = OpConstant %float_type 0.0
%const_1_0 = OpConstant %float_type 1.0
%const_0_1 = OpConstant %float_type 0.1
%const_0_3 = OpConstant %float_type 0.3
%const_0_5 = OpConstant %float_type 0.5
%const_0_7 = OpConstant %float_type 0.7
%const_0_8 = OpConstant %float_type 0.8
%const_6_float = OpConstant %float_type 6.0
%const_0_int = OpConstant %int_type 0
%const_1_int = OpConstant %int_type 1
%const_2_int = OpConstant %int_type 2
%const_3_int = OpConstant %int_type 3
%const_4_int = OpConstant %int_type 4

; --- Input Variables ---
%worldPos_var = OpVariable %Input %vec3_type
OpDecorate %worldPos_var Location 0
%normal_var = OpVariable %Input %vec3_type
OpDecorate %normal_var Location 1
%uv_var = OpVariable %Input %vec2_type
OpDecorate %uv_var Location 2
%tangent_var = OpVariable %Input %vec3_type
OpDecorate %tangent_var Location 3
%color_var = OpVariable %Input %vec3_type
OpDecorate %color_var Location 4

; --- Uniform Buffer ---
%ubo_struct_members = OpTypeStruct %mat4_type %mat4_type %mat4_type %vec4_type %vec4_type %vec4_type
%ubo_var = OpVariable Uniform %ubo_struct_members
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; --- Output Variable ---
%outColor_var = OpVariable %Output %vec4_type
OpDecorate %outColor_var Location 0

; --- Main Function ---
%main_func = OpFunction %void_type
    ; Load Input variables
    %worldPos_val = OpLoad %vec3_type %worldPos_var
    %normal_val = OpLoad %vec3_type %normal_var
    %uv_val = OpLoad %vec2_type %uv_var
    %tangent_val = OpLoad %vec3_type %tangent_var
    %color_val = OpLoad %vec3_type %color_var

    ; Access UBO members
    %lightPos_val = OpLoad %vec4_type OpAccessChain %ubo_var %const_3_int ; lightPos (offset 192)
    %cameraPos_val = OpLoad %vec4_type OpAccessChain %ubo_var %const_4_int ; cameraPos (offset 208)

    ; --- Polka Dot Logic ---
    ; uv_times_6 = uv * 6.0
    %uv_times_6 = OpFMul %vec2_type %uv_val (%const_6_float, %const_6_float)

    ; cell_idx_float = floor(uv * 6.0)
    %cell_idx_float = OpFloor %vec2_type %uv_times_6

    ; cell_uv_relative = (uv * 6.0) - floor(uv * 6.0)
    %cell_uv_relative = OpFSub %vec2_type %uv_times_6 %cell_idx_float

    ; Distance from cell center (0.5, 0.5)
    %cell_center = OpCompositeConstruct %vec2_type %const_0_5 %const_0_5
    %diff_from_center = OpFSub %vec2_type %cell_uv_relative %cell_center
    %dist_sq = OpDot %float_type %diff_from_center %diff_from_center
    %dist = OpSqrt %float_type %dist_sq
    %is_dot = OpFOrdLessThan %bool_type %dist %const_0_3

    ; Base Colors (Yellow and Dark Blue)
    %dot_color = OpCompositeConstruct %vec4_type %const_1_0 %const_0_8 %const_0_0 %const_1_0 ; Yellow RGBA
    %bg_color = OpCompositeConstruct %vec4_type %const_0_1 %const_0_1 %const_0_3 %const_1_0 ; Dark Blue RGBA

    ; Select base color based on whether it's a dot
    %base_color = OpSelect %vec4_type %dot_color %bg_color %is_dot

    ; --- Lambertian Shading ---
    ; Calculate light direction vector: lightPos.xyz - cameraPos.xyz
    %lightPos_xyz = OpCompositeExtract %vec3_type %lightPos_val 0
    %cameraPos_xyz = OpCompositeExtract %vec3_type %cameraPos_val 0
    %light_dir_vec = OpFSub %vec3_type %lightPos_xyz %cameraPos_xyz
    ; Normalize the light direction
    %light_dir_norm = OpNormalize %vec3_type %light_dir_vec

    ; Normalize the surface normal
    %normal_norm = OpNormalize %vec3_type %normal_val
    ; Calculate diffuse term: dot(normal_norm, light_dir_norm)
    %diffuse = OpDot %float_type %normal_norm %light_dir_norm

    ; Clamp diffuse to the range [0, 1]
    %diffuse_pos = OpMax %float_type %diffuse %const_0_0
    %diffuse_clamped = OpMin %float_type %diffuse_pos %const_1_0

    ; Calculate shading factor: 0.3 + 0.7 * diffuse_clamped
    %shading_factor_term1 = OpFMul %float_type %const_0_7 %diffuse_clamped
    %shading_factor = OpFAdd %float_type %const_0_3 %shading_factor_term1

    ; Apply shading to the base color
    ; Extend shading_factor to a vec4 for element-wise multiplication
    %shading_factor_vec4 = OpCompositeConstruct %vec4_type %shading_factor %shading_factor %shading_factor %shading_factor
    %final_color = OpFMul %vec4_type %base_color %shading_factor_vec4

    ; Store the final computed color into the output variable
    OpStore %outColor_var %final_color

    OpReturn
OpEndFunction

; --- Entry Point Declaration ---
; Declares the 'main' function as the entry point for the fragment shader,
; linking the global input and output variables.
OpEntryPoint Fragment %main_func "main" %worldPos_var %normal_var %uv_var %tangent_var %color_var %outColor_var
; Specifies the origin for fragment processing (OriginUpperLeft is common for Vulkan).
OpExecutionMode %main_func OriginUpperLeft
