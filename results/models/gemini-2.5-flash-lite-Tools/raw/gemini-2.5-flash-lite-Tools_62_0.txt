import * as fs from 'fs';

interface Town {
  id: string;
  x: number;
  y: number;
  value: number;
}

interface InputData {
  width: number;
  height: number;
  grid: string[];
  ignitions: [number, number][];
  towns: Town[];
  cutBudget: number;
}

function solve() {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: InputData = JSON.parse(input);

  const { width, height, grid, ignitions, towns, cutBudget } = data;

  // Simple heuristic: 
  // 1. Run BFS/Dijkstra to determine which forest cells ('.') 
  //    are on the path from ignitions to towns.
  // 2. Prioritize cutting cells that are on paths to the most valuable towns.
  
  // This is a simplified approach, but should be a good starting point.
  
  const firebreaks: [number, number][] = [];
  let budgetUsed = 0;

  // Simple heuristic:
  // For each town, find the shortest path from *any* ignition point using BFS.
  // The cells on this path are candidates for cutting.
  // We prioritize towns with high value and reachable paths.

  // 1. Map grid for easy access
  const gridCells = grid.map(row => row.split(''));

  // 2. Simple BFS to get distance from ignitions to all cells
  const dist = Array.from({ length: height }, () => Array(width).fill(Infinity));
  const queue: [number, number, number][] = []; // [x, y, d]
  for (const [ix, iy] of ignitions) {
    if (gridCells[iy][ix] === '.') {
      dist[iy][ix] = 0;
      queue.push([ix, iy, 0]);
    }
  }

  const dx = [0, 0, 1, -1];
  const dy = [1, -1, 0, 0];

  while (queue.length > 0) {
    const [x, y, d] = queue.shift()!;
    for (let i = 0; i < 4; i++) {
      const nx = x + dx[i];
      const ny = y + dy[i];
      if (nx >= 0 && nx < width && ny >= 0 && ny < height && gridCells[ny][nx] === '.' && dist[ny][nx] === Infinity) {
        dist[ny][nx] = d + 1;
        queue.push([nx, ny, d + 1]);
      }
    }
  }

  // 3. For each town, find path from closest ignited cell and add cells to candidates
  const candidates: {x: number, y: number, townValue: number, dist: number}[] = [];
  for (const town of towns) {
      // Find path from town back to ignition
      let currX = town.x;
      let currY = town.y;
      
      while (dist[currY][currX] > 0) {
          let bestD = dist[currY][currX];
          let nextX = currX;
          let nextY = currY;
          
          for (let i = 0; i < 4; i++) {
              const nx = currX + dx[i];
              const ny = currY + dy[i];
              if (nx >= 0 && nx < width && ny >= 0 && ny < height && dist[ny][nx] < bestD) {
                  bestD = dist[ny][nx];
                  nextX = nx;
                  nextY = ny;
              }
          }
          
          if (nextX === currX && nextY === currY) break; // Path blocked or no path
          
          currX = nextX;
          currY = nextY;
          
          if (gridCells[currY][currX] === '.') {
              candidates.push({x: currX, y: currY, townValue: town.value, dist: dist[currY][currX]});
          }
      }
  }
  
  // 4. Sort candidates: prioritize high value/dist, cut
  candidates.sort((a, b) => (b.townValue / (b.dist + 1)) - (a.townValue / (a.dist + 1)));

  for (const cand of candidates) {
      if (budgetUsed < cutBudget) {
          firebreaks.push([cand.x, cand.y]);
          budgetUsed++;
      } else {
          break;
      }
  }
  console.log(JSON.stringify({ firebreaks }));
}

solve();
