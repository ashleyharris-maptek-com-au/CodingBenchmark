import numpy as np

class AircraftDynamics:
    def __init__(self):
        self.x = 10000.0  # Distance from runway (m)
        self.y = 0.0      # Lateral deviation (m)
        self.z = 1000.0   # Altitude (m)
        self.vx = -75.0   # Ground speed (m/s)
        self.vy = 0.0
        self.vz = 0.0
        self.dt = 0.1

    def update(self, pitch_cmd, roll_cmd):
        # Simplified dynamics
        self.z += self.vz * self.dt
        self.x += self.vx * self.dt
        self.y += self.vy * self.dt
        
        # Basic control response
        self.vz += (pitch_cmd - self.vz) * 0.1
        self.vy += (roll_cmd - self.vy) * 0.1

class Autopilot:
    def __init__(self):
        self.kp_pitch = 0.1
        self.kp_roll = 0.1
        self.mode = "APPROACH"

    def compute_commands(self, aircraft):
        if self.mode == "APPROACH":
            # ILS logic: target glideslope (3 deg) and localizer (center)
            target_z = (aircraft.x * np.tan(np.radians(3.0)))
            pitch_cmd = (target_z - aircraft.z) * self.kp_pitch
            roll_cmd = (-aircraft.y) * self.kp_roll
            
            if aircraft.x < 50.0:  # Start flare
                self.mode = "FLARE"
            return pitch_cmd, roll_cmd
            
        elif self.mode == "FLARE":
            # Flare logic: reduce sink rate
            pitch_cmd = -0.5 # Flare pitch command
            roll_cmd = (-aircraft.y) * self.kp_roll
            if aircraft.z < 0.5: # Touchdown
                self.mode = "ROLLOUT"
            return pitch_cmd, roll_cmd
            
        elif self.mode == "ROLLOUT":
            # Rollout logic: maintain centerline
            pitch_cmd = 0.0
            roll_cmd = (-aircraft.y) * self.kp_roll
            return pitch_cmd, roll_cmd

# Simulation
ac = AircraftDynamics()
ap = Autopilot()

for _ in range(2000):
    pitch, roll = ap.compute_commands(ac)
    ac.update(pitch, roll)
    if _ % 100 == 0:
        print(f"Mode: {ap.mode}, Alt: {ac.z:.2f}, Lat: {ac.y:.2f}, Dist: {ac.x:.2f}")
