; SPIR-V Fragment Shader
; Simple fragment shader that outputs a red color.

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; --- Declarations ---
; Output variable: location 0, vec4
               OpDecorate %outColor Location 0

; --- Types ---
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
      %vec4  = OpTypeVector %float 4
%ptr_out_vec4 = OpTypePointer Output %vec4

; --- Variables ---
   %outColor = OpVariable %ptr_out_vec4 Output

; --- Constants ---
      %f1_0  = OpConstant %float 1.0
      %f0_0  = OpConstant %float 0.0
      %red   = OpConstantComposite %vec4 %f1_0 %f0_0 %f0_0 %f1_0

; --- Main Function ---
       %main = OpFunction %void None %func
         %lb = OpLabel
               OpStore %outColor %red
               OpReturn
               OpFunctionEnd
