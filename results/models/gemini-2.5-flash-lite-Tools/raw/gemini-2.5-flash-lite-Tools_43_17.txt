struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;          // offset 0   (column-major)
    float4x4 view;           // offset 64
    float4x4 projection;     // offset 128
    float4 lightPos;         // value: (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;        // value: (0.0, 0.0, 3.0, 1.0)
    float4 params;           // value: (0.0, 0.5, 0.0, 0.0)
};

float4 main(PSInput input) : SV_Target0 {
    // 1. Brick pattern parameters
    float2 brickScale = float2(8.0, 16.0);
    float2 mortarThickness = float2(0.05, 0.1); // x and y thickness
    float4 mortarColor = float4(0.7, 0.7, 0.7, 1.0);
    float4 brickColor = float4(0.7, 0.2, 0.1, 1.0);

    // 2. Scaled UV coordinates
    float2 scaledUV = input.uv * brickScale;

    // 3. Staggered row effect
    float row = floor(scaledUV.y);
    if (fmod(row, 2.0) < 1.0) { // Even rows
        // No offset
    } else { // Odd rows
        scaledUV.x += 0.5; // Half-brick offset
    }

    // 4. Calculate brick and mortar regions
    float2 brickUV = fmod(scaledUV, 1.0); // Fractional part within each brick unit

    // Adjust mortar thickness to be symmetrical around the edges
    // If brickUV.x < mortarThickness.x OR brickUV.x > (1.0 - mortarThickness.x), then it's mortar.
    // The same for Y.
    bool isMortarX = (brickUV.x < mortarThickness.x || brickUV.x > (1.0 - mortarThickness.x));
    bool isMortarY = (brickUV.y < mortarThickness.y || brickUV.y > (1.0 - mortarThickness.y));
    
    bool isMortar = (isMortarX || isMortarY);

    float4 finalColor = isMortar ? mortarColor : brickColor;

    // 5. Lambertian shading
    float3 worldNormal = normalize(input.normal);
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);

    float diffuse = max(0.0, dot(worldNormal, lightDir));
    float lambertianFactor = 0.4 + 0.6 * diffuse;

    finalColor.rgb *= lambertianFactor;
    finalColor.a = 1.0; // Ensure alpha is 1.0

    return finalColor;
}
