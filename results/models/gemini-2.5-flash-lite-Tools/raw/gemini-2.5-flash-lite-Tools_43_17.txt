// Vulkan HLSL pixel shader for procedural brick pattern
// Compiled with glslangValidator -D

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Brick settings
    const float2 brickScale = float2(8.0, 16.0);
    const float2 mortarSize = float2(0.05, 0.1);
    const float3 mortarColor = float3(0.7, 0.7, 0.7);
    const float3 brickColor = float3(0.7, 0.2, 0.1);

    // UV manipulation for staggered bricks
    float2 uv = input.uv * brickScale;

    // Staggered rows: shift every other row
    if (frac(uv.y * 0.5) > 0.5) {
        uv.x += 0.5;
    }

    float2 brickUV = frac(uv);
    float2 brickID = floor(uv);

    // Check if in mortar area
    float2 mortarMask = step(mortarSize, brickUV) * step(brickUV, 1.0 - mortarSize);
    float isBrick = mortarMask.x * mortarMask.y;

    float3 finalColor = lerp(mortarColor, brickColor, isBrick);

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shading = 0.4 + 0.6 * diffuse;

    return float4(finalColor * shading, 1.0);
}
