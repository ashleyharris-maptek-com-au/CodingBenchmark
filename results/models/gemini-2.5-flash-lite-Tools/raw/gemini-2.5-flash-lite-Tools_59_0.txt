const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const { probability, drones, maxStepsPerDrone } = data;
    const height = probability.length;
    const width = probability[0].length;

    const paths = [];

    // Basic greedy algorithm: each drone moves to the highest value neighbor 
    // that hasn't been visited by another drone in this step, or stays if no better move.
    // For simplicity, we just have each drone try to maximize its own coverage.

    for (const drone of drones) {
        const path = [[drone.x, drone.y]];
        let currX = drone.x;
        let currY = drone.y;

        for (let step = 0; step < maxStepsPerDrone; step++) {
            let bestVal = -1;
            let bestMove = [currX, currY];

            // Possible moves: stay, up, down, left, right
            const moves = [
                [0, 0], [-1, 0], [1, 0], [0, -1], [0, 1]
            ];

            for (const [dx, dy] of moves) {
                const nextX = currX + dx;
                const nextY = currY + dy;

                if (nextX >= 0 && nextX < height && nextY >= 0 && nextY < width) {
                    const val = probability[nextX][nextY];
                    if (val >= 0 && val > bestVal) {
                        bestVal = val;
                        bestMove = [nextX, nextY];
                    }
                }
            }

            currX = bestMove[0];
            currY = bestMove[1];
            path.push([currX, currY]);
        }
        paths.push(path);
    }

    process.stdout.write(JSON.stringify({ paths }));
}

solve();
