OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1
               OpSource GLSL 450
               OpName %main "main"
               OpName %thickness "thickness"
               OpName %_arr_gl_PerVertex_Input "gl_PerVertex"
               OpMemberName %_arr_gl_PerVertex_Input 0 "gl_Position"
               OpName %gl_in "gl_in"
               OpName %worldPos "worldPos"
               OpName %normal "normal"
               OpName %color "color"
               OpName %_gl_PerVertex_Output "gl_PerVertex"
               OpMemberName %_gl_PerVertex_Output 0 "gl_Position"
               OpName %gl_out "gl_out"
               OpName %outColor "outColor"

       %void = OpTypeVoid
         %32 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
      %v2float = OpTypeVector %float 2
      %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0
     %uint_0 = OpConstant %uint 0
     %uint_1 = OpConstant %uint 1
     %uint_2 = OpConstant %uint 2
     %uint_3 = OpConstant %uint 3
     %float_0_03 = OpConstant %float 0.03
     %float_0 = OpConstant %float 0
     %float_1 = OpConstant %float 1
     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1

%_arr_gl_PerVertex_Input = OpTypeStruct %v4float
               OpMemberDecorate %_arr_gl_PerVertex_Input 0 BuiltIn Position
               OpDecorate %_arr_gl_PerVertex_Input Block
%_arr_gl_PerVertex_Type = OpTypeArray %_arr_gl_PerVertex_Input %uint_2
%_ptr_Input_arr_gl_PerVertex = OpTypePointer Input %_arr_gl_PerVertex_Type
%_ptr_Input_v3float = OpTypePointer Input %v3float ; Pointer to single v3float (for access chain elements)
%_arr_v3float_Type = OpTypeArray %v3float %uint_2 ; Type for array of 2 v3floats
%_ptr_Input_arr_v3float = OpTypePointer Input %_arr_v3float_Type ; Pointer to array of 2 v3floats

       %gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex Input
     %worldPos = OpVariable %_ptr_Input_arr_v3float Input
      %normal = OpVariable %_ptr_Input_arr_v3float Input
       %color = OpVariable %_ptr_Input_arr_v3float Input

%_gl_PerVertex_Output = OpTypeStruct %v4float
               OpMemberDecorate %_gl_PerVertex_Output 0 BuiltIn Position
               OpDecorate %_gl_PerVertex_Output Block
%_ptr_Output_gl_PerVertex = OpTypePointer Output %_gl_PerVertex_Output
       %gl_out = OpVariable %_ptr_Output_gl_PerVertex Output
%_ptr_Output_v3float = OpTypePointer Output %v3float
     %outColor = OpVariable %_ptr_Output_v3float Output

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Output_v4float = OpTypePointer Output %v4float

       %main = OpFunction %void None %32
         %5 = OpLabel
      %vec4_0 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0
      %vec4_1 = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1

          %p0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %uint_0 %uint_0
          %p0 = OpLoad %v4float %p0_ptr
          %p1_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %uint_1 %uint_0
          %p1 = OpLoad %v4float %p1_ptr

          %c0_ptr = OpAccessChain %_ptr_Input_v3float %color %uint_0
          %c0 = OpLoad %v3float %c0_ptr
          %c1_ptr = OpAccessChain %_ptr_Input_v3float %color %uint_1
          %c1 = OpLoad %v3float %c1_ptr

          %dir = OpFSub %v4float %p1 %p0
          %dir_xy = OpVectorShuffle %v2float %dir %dir %uint_0 %uint_1
          %neg_dir_y = OpFNegate %float (OpCompositeExtract %float %dir %uint_1)
          %dir_x = OpCompositeExtract %float %dir %uint_0
          %perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x

          %len_sq = OpDot %float %perp_raw %perp_raw
          %len = OpExtInst %float %1 Sqrt %len_sq
          %len_inv = OpFDiv %float %float_1 %len
          %perp_norm = OpVectorTimesScalar %v2float %perp_raw %len_inv
          %perp_scaled = OpVectorTimesScalar %v2float %perp_norm %float_0_03

          %perp_v4 = OpCompositeConstruct %v4float %perp_scaled %float_0 %float_0 %float_0

          %p0_plus_perp = OpFAdd %v4float %p0 %perp_v4
          %p0_minus_perp = OpFSub %v4float %p0 %perp_v4
          %p1_plus_perp = OpFAdd %v4float %p1 %perp_v4
          %p1_minus_perp = OpFSub %v4float %p1 %perp_v4

          %out_pos_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %uint_0
          %out_color_ptr = OpAccessChain %_ptr_Output_v3float %outColor


               OpStore %out_color_ptr %c0
               OpStore %out_pos_ptr %p0_plus_perp
               OpEmitVertex

               OpStore %out_color_ptr %c0
               OpStore %out_pos_ptr %p0_minus_perp
               OpEmitVertex

               OpStore %out_color_ptr %c1
               OpStore %out_pos_ptr %p1_plus_perp
               OpEmitVertex

               OpStore %out_color_ptr %c1
               OpStore %out_pos_ptr %p1_minus_perp
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
