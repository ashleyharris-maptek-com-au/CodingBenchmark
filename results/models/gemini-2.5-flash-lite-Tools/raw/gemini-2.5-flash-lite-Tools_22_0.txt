using System;
using System.Collections.Generic;
using System.Linq;

namespace TetrisBot
{
    public class TetrisEngine
    {
        public int Width { get; }
        public int Height { get; }
        public int[,] Board { get; private set; }

        public static Dictionary<string, List<int[,]>> PieceDefinitions = new Dictionary<string, List<int[,]>>
        {
            { "I", new List<int[,]> { new int[,] { { 1, 1, 1, 1 } }, new int[,] { { 1 }, { 1 }, { 1 }, { 1 } } } },
            { "O", new List<int[,]> { new int[,] { { 1, 1 }, { 1, 1 } } } },
            { "T", new List<int[,]> { new int[,] { { 0, 1, 0 }, { 1, 1, 1 } }, new int[,] { { 1, 0 }, { 1, 1 }, { 1, 0 } }, new int[,] { { 1, 1, 1 }, { 0, 1, 0 } }, new int[,] { { 0, 1 }, { 1, 1 }, { 0, 1 } } } },
            { "S", new List<int[,]> { new int[,] { { 0, 1, 1 }, { 1, 1, 0 } }, new int[,] { { 1, 0 }, { 1, 1 }, { 0, 1 } } } },
            { "Z", new List<int[,]> { new int[,] { { 1, 1, 0 }, { 0, 1, 1 } }, new int[,] { { 0, 1 }, { 1, 1 }, { 1, 0 } } } },
            { "J", new List<int[,]> { new int[,] { { 1, 0, 0 }, { 1, 1, 1 } }, new int[,] { { 1, 1 }, { 1, 0 }, { 1, 0 } }, new int[,] { { 1, 1, 1 }, { 0, 0, 1 } }, new int[,] { { 0, 1 }, { 0, 1 }, { 1, 1 } } } },
            { "L", new List<int[,]> { new int[,] { { 0, 0, 1 }, { 1, 1, 1 } }, new int[,] { { 1, 0 }, { 1, 0 }, { 1, 1 } }, new int[,] { { 1, 1, 1 }, { 1, 0, 0 } }, new int[,] { { 1, 1 }, { 0, 1 }, { 0, 1 } } } }
        };

        public TetrisEngine(int width, int height)
        {
            Width = width;
            Height = height;
            Board = new int[height, width];
        }

        public (int rotation, int column) FindBestMove(string pieceType)
        {
            double bestScore = double.MinValue;
            (int rotation, int column) bestMove = (0, 0);

            var rotations = PieceDefinitions[pieceType];
            for (int r = 0; r < rotations.Count; r++)
            {
                var piece = rotations[r];
                for (int c = 0; c <= Width - piece.GetLength(1); c++)
                {
                    int[,] nextBoard = SimulateDrop(piece, c);
                    if (nextBoard == null) continue; // Collision

                    double score = EvaluateBoard(nextBoard);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestMove = (r, c);
                    }
                }
            }
            return bestMove;
        }

        private int[,] SimulateDrop(int[,] piece, int col)
        {
            int pieceHeight = piece.GetLength(0);
            int pieceWidth = piece.GetLength(1);
            int[,] nextBoard = (int[,])Board.Clone();

            // Find lowest position
            for (int r = 0; r <= Height - pieceHeight; r++)
            {
                if (CheckCollision(piece, r, col))
                {
                    if (r == 0) return null; // Instant gameover
                    PlacePiece(nextBoard, piece, r - 1, col);
                    return nextBoard;
                }
            }
            PlacePiece(nextBoard, piece, Height - pieceHeight, col);
            return nextBoard;
        }

        private bool CheckCollision(int[,] piece, int row, int col)
        {
            for (int r = 0; r < piece.GetLength(0); r++)
                for (int c = 0; c < piece.GetLength(1); c++)
                    if (piece[r, c] != 0 && (row + r >= Height || Board[row + r, col + c] != 0))
                        return true;
            return false;
        }

        private void PlacePiece(int[,] board, int[,] piece, int row, int col)
        {
            for (int r = 0; r < piece.GetLength(0); r++)
                for (int c = 0; c < piece.GetLength(1); c++)
                    if (piece[r, c] != 0) board[row + r, col + c] = 1;
            
            // Clear lines
            for (int r = Height - 1; r >= 0; r--)
            {
                bool full = true;
                for (int c = 0; c < Width; c++) if (board[r, c] == 0) full = false;
                if (full)
                {
                    for (int i = r; i > 0; i--)
                        for (int c = 0; c < Width; c++) board[i, c] = board[i - 1, c];
                    for (int c = 0; c < Width; c++) board[0, c] = 0;
                    r++;
                }
            }
        }

        private double EvaluateBoard(int[,] board)
        {
            int height = 0, holes = 0;
            for (int c = 0; c < Width; c++)
            {
                bool foundTop = false;
                for (int r = 0; r < Height; r++)
                {
                    if (board[r, c] != 0)
                    {
                        if (!foundTop) { height += (Height - r); foundTop = true; }
                    }
                    else if (foundTop) holes++;
                }
            }
            return -(height * 0.5) - (holes * 2.0);
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            string[] setup = Console.ReadLine()?.Split(' ', StringSplitOptions.RemoveEmptyEntries);
            if (setup == null || setup.Length < 3) return;

            int width = int.Parse(setup[0]);
            int height = int.Parse(setup[1]);
            int numPieces = int.Parse(setup[2]);

            TetrisEngine engine = new TetrisEngine(width, height);

            for (int i = 0; i < numPieces; i++)
            {
                string pieceType = Console.ReadLine()?.Trim();
                if (pieceType == null) break;

                var bestMove = engine.FindBestMove(pieceType);
                Console.WriteLine($"{bestMove.rotation} {bestMove.column}");
                Console.Out.Flush();

                string result = Console.ReadLine()?.Trim();
                // Update board based on result if necessary
                if (result == "gameover") break;
            }
        }
    }
}
