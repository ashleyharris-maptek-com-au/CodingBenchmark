using System;
using System.Collections.Generic;
using System.IO;

public class VoxelGrid
{
    public int XCount, YCount, ZCount;
    public double XSize, YSize, ZSize;
    private const int ChunkSize = 32;
    private byte[][][][] chunks; 

    public VoxelGrid(int x, int y, int z, double xs, double ys, double zs)
    {
        XCount = x; YCount = y; ZCount = z;
        XSize = xs; YSize = ys; ZSize = zs;
        
        int cx = (x + ChunkSize - 1) / ChunkSize;
        int cy = (y + ChunkSize - 1) / ChunkSize;
        int cz = (z + ChunkSize - 1) / ChunkSize;
        chunks = new byte[cx][][][];
        for(int i=0; i<cx; i++) {
            chunks[i] = new byte[cy][][];
            for(int j=0; j<cy; j++) {
                chunks[i][j] = new byte[cz][];
            }
        }
    }

    public void SetVoxel(int x, int y, int z, byte val)
    {
        int cx = x / ChunkSize, cy = y / ChunkSize, cz = z / ChunkSize;
        if(chunks[cx][cy][cz] == null) chunks[cx][cy][cz] = new byte[ChunkSize * ChunkSize * ChunkSize];
        int lx = x % ChunkSize, ly = y % ChunkSize, lz = z % ChunkSize;
        chunks[cx][cy][cz][lx * ChunkSize * ChunkSize + ly * ChunkSize + lz] = val;
    }

    public byte GetVoxel(int x, int y, int z)
    {
        if(x < 0 || x >= XCount || y < 0 || y >= YCount || z < 0 || z >= ZCount) return 0;
        int cx = x / ChunkSize, cy = y / ChunkSize, cz = z / ChunkSize;
        if(chunks[cx][cy][cz] == null) return 0;
        int lx = x % ChunkSize, ly = y % ChunkSize, lz = z % ChunkSize;
        return chunks[cx][cy][cz][lx * ChunkSize * ChunkSize + ly * ChunkSize + lz];
    }
}
