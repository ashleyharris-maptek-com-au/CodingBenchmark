#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Guard against gl_GlobalInvocationID.x >= N
    if (i >= N) {
        return;
    }

    // Read current body's data
    vec4 body_i_pos_mass = inData[i * 2];
    vec3 pos_i = body_i_pos_mass.xyz;
    float mass_i = body_i_pos_mass.w;

    vec4 body_i_vel_dummy = inData[i * 2 + 1];
    vec3 vel_i = body_i_vel_dummy.xyz;

    vec3 acceleration = vec3(0.0);

    // Calculate acceleration from all other bodies
    for (uint j = 0; j < N; ++j) {
        // Skip self-interaction
        if (i == j) {
            continue;
        }

        vec4 body_j_pos_mass = inData[j * 2];
        vec3 pos_j = body_j_pos_mass.xyz;
        float mass_j = body_j_pos_mass.w;

        vec3 r_ij = pos_j - pos_i;
        float dist_sq = dot(r_ij, r_ij) + softening * softening;
        // Using `inversesqrt` to optimize `1.0 / sqrt(dist_sq)^3` which is `1.0 / dist_sq^(3/2)`
        // `rsqrt` gives `1/sqrt(x)`. So `1/sqrt(dist_sq)^3` is `(rsqrt(dist_sq))^3`.
        float inv_dist_cubed = inversesqrt(dist_sq) * inversesqrt(dist_sq) * inversesqrt(dist_sq);

        acceleration += G * mass_j * r_ij * inv_dist_cubed;
    }

    // Update velocity using Euler/leapfrog (simplified to Euler for this step)
    // In a true leapfrog, velocity is half-stepped, then position full-stepped, then velocity half-stepped.
    // The prompt specifies "leapfrog/Euler integration method", and "Velocity is advanced by acc * dt and position is advanced by vel * dt".
    // This implies a straightforward Euler step for both, which is a common simplification for N-body in this context.
    vec3 new_vel_i = vel_i + acceleration * dt;

    // Update position
    vec3 new_pos_i = pos_i + new_vel_i * dt; // Using the newly calculated velocity for position update

    // Write updated values to output
    outData[i * 2] = vec4(new_pos_i, mass_i);
    outData[i * 2 + 1] = vec4(new_vel_i, 0.0);
}
