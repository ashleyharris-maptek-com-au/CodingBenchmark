#version 430 core

layout(local_size_x = 256) in;

struct Particle {
    vec4 position; // xyz: position, w: mass
    vec4 velocity; // xyz: velocity, w: padding/other
};

layout(std430, binding = 0) buffer ParticleBuffer {
    Particle particles[];
};

uniform float dt;
uniform float softening;

void main() {
    uint index = gl_GlobalInvocationID.x;
    
    // Total number of particles is assumed to be available or manageable via buffer length
    // For this example, we assume it's fixed or passed as a uniform.
    // Assuming particles.length() works in GLSL 4.3+
    uint numParticles = particles.length();

    vec3 pos = particles[index].position.xyz;
    vec3 vel = particles[index].velocity.xyz;
    float mass = particles[index].position.w;

    vec3 acc = vec3(0.0);

    for (uint i = 0; i < numParticles; ++i) {
        if (i == index) continue;

        vec3 otherPos = particles[i].position.xyz;
        float otherMass = particles[i].position.w;

        vec3 diff = otherPos - pos;
        float distSq = dot(diff, diff) + softening * softening;
        float invDist = 1.0 / sqrt(distSq);
        float invDist3 = invDist * invDist * invDist;

        acc += diff * otherMass * invDist3;
    }

    vel += acc * dt;
    pos += vel * dt;

    particles[index].velocity.xyz = vel;
    particles[index].position.xyz = pos;
}
