def _get_reliable_range(state: dict) -> float | None:
    """
    Fuses range finders and docking camera range.
    Prioritizes docking camera if active, then range finders.
    Returns the closest valid range.
    """
    valid_ranges = []

    # Docking camera range
    if state.get("dock_camera_active") and state.get("dock_camera_range_m") is not None:
        valid_ranges.append(state["dock_camera_range_m"])

    # Range finders
    for i in range(1, 4):
        rng, valid = _get_reliable_data(state, f"range_{i}", "_m", "_valid")
        if valid and rng is not None:
            valid_ranges.append(rng)
    
    if valid_ranges:
        return min(valid_ranges)
    return None

def autopilot_step(state: dict, dt: float) -> dict:

           _integral_pos_error, _integral_vel_error, _integral_att_error, 
           _prev_pos_error, _prev_vel_error, _prev_att_error

        # Reset PID integral terms on major sensor loss
        _integral_pos_error.fill(0.0)
        _integral_vel_error.fill(0.0)
        _integral_att_error.fill(0.0)
        _prev_pos_error.fill(0.0)
        _prev_vel_error.fill(0.0)
        _prev_att_error.fill(0.0)
        # Fallback: try to maintain last known state or minimal thrusts
        return {
            "thrust_x": _last_thrust[0],
            "thrust_y": _last_thrust[1],
            "thrust_z": _last_thrust[2],
            "att_rate_roll": _last_att_rate[1],
            "att_rate_pitch": _last_att_rate[0],
            "att_rate_yaw": _last_att_rate[2],

            _current_phase = "MEDIUM_RANGE"
        elif 10.0 < current_range <= 50.0:
            _current_phase = "CLOSE_RANGE"
        else: # current_range <= 10.0
            _current_phase = "FINAL_APPROACH"
    
    # Docking success check
    if current_range < 2.0 and np.linalg.norm(rel_vel) < 0.3:
        # Stop all thrusts and attitude changes
        _integral_pos_error.fill(0.0)
        _integral_vel_error.fill(0.0)
        _integral_att_error.fill(0.0)
        _prev_pos_error.fill(0.0)
        _prev_vel_error.fill(0.0)
        _prev_att_error.fill(0.0)

            "thrust_x": 0.0,
            "thrust_y": 0.0,
            "thrust_z": 0.0,
            "att_rate_roll": 0.0,
            "att_rate_pitch": 0.0,
            "att_rate_yaw": 0.0,
        }

    # Initialize thrust and attitude rates
    thrust = np.array([0.0, 0.0, 0.0])
    att_rate = np.array([0.0, 0.0, 0.0]) # pitch, roll, yaw

    # 3. Control Logic for each phase
    # Initialize error terms for current step
    pos_error = -rel_pos # Target is 0,0,0 relative position
    vel_error = np.array([0.0, 0.0, 0.0]) # Will be set based on target_rel_vel

    if _current_phase == "FAR_RANGE":
        # Target: reduce along-track separation (y) and achieve small closing velocity
        # Critical insight: Retrograde thrust to lower orbit, increasing speed to catch up.
        # If behind (negative y), we want to increase speed (more negative rel_vel_y)
        _target_rel_vel = np.array([0.0, -0.1, 0.0]) # Aim for 0.1 m/s closing speed

        vel_error = _target_rel_vel - rel_vel

        # Dampen radial (x) and cross-track (z) corrections in FAR_RANGE, prioritize along-track
        # We still want to eventually align on x and z, but Y is most critical here
        p_factor_pos_xz = 0.05
        i_factor_pos_xz = 0.00005
        d_factor_pos_xz = 0.005
        
    elif _current_phase == "MEDIUM_RANGE":
        # V-bar approach: maintain constant radial separation, slowly reduce along-track

        i_factor_pos_xz = 1.0

        i_factor_pos_xz = 0.1
        d_factor_pos_xz = 0.1

    p_thrust_vel = KP_VEL * vel_error

    _integral_vel_error += vel_error * dt
    i_thrust_pos = KI_POS * _integral_pos_error
    i_thrust_vel = KI_VEL * _integral_vel_error

    # Derivative terms

    d_thrust_vel = KD_VEL * (vel_error - _prev_vel_error) / dt if dt > 0 else np.array([0.0, 0.0, 0.0])
    
    # Combine all terms for thrust, applying phase-specific dampening for position P, I, D
    thrust = np.array([0.0, 0.0, 0.0])

    thrust[0] = (p_thrust_pos[0] * p_factor_pos_xz) + (i_thrust_pos[0] * i_factor_pos_xz) + (d_thrust_pos[0] * d_factor_pos_xz) + p_thrust_vel[0] + i_thrust_vel[0] + d_thrust_vel[0]
    thrust[2] = (p_thrust_pos[2] * p_factor_pos_xz) + (i_thrust_pos[2] * i_factor_pos_xz) + (d_thrust_pos[2] * d_factor_pos_xz) + p_thrust_vel[2] + i_thrust_vel[2] + d_thrust_vel[2]
    
    # Y (along-track)
    thrust[1] = p_thrust_pos[1] + i_thrust_pos[1] + d_thrust_pos[1] + p_thrust_vel[1] + i_thrust_vel[1] + d_thrust_vel[1]


    # 4. Attitude Control
    target_attitude = np.array([0.0, 0.0, 0.0]) # Desired pitch, roll, yaw
    target_ang_vel = np.array([0.0, 0.0, 0.0]) # Desired angular velocity

    if current_attitude is not None and current_ang_vel is not None:
        att_error = target_attitude - current_attitude
        att_rate_error = target_ang_vel - current_ang_vel

        p_att_rate = KP_ATT * att_error

        # I term
        _integral_att_error += att_error * dt
        i_att_rate = KI_ATT * _integral_att_error

        # D term
        d_att_rate = KD_ATT * (att_error - _prev_att_error) / dt if dt > 0 else np.array([0.0, 0.0, 0.0])
        
        att_rate = p_att_rate + i_att_rate + d_att_rate
        att_rate = np.clip(att_rate, -0.035, 0.035) # Limit attitude rates
    
    # If docking camera is active and valid, incorporate bearing for fine alignment
    if state.get("dock_camera_active") and 
       state.get("dock_camera_bearing_x_valid") and 
       state.get("dock_camera_bearing_y_valid") and 
       state.get("dock_camera_bearing_z_valid"):
        
        bearing_x = state["dock_camera_bearing_x"] * np.pi / 180.0
        bearing_y = state["dock_camera_bearing_y"] * np.pi / 180.0
        bearing_z = state["dock_camera_bearing_z"] * np.pi / 180.0 # Assuming bearing_z provides roll correction
        
        # Adjust attitude rates based on bearing error
        att_rate[0] += KP_ATT * bearing_x * 0.1 # Pitch adjustment
        att_rate[2] += KP_ATT * bearing_y * 0.1 # Yaw adjustment
        att_rate[1] += KP_ATT * bearing_z * 0.1 # Roll adjustment
        
        att_rate = np.clip(att_rate, -0.035, 0.035)

    # 5. Apply Thrust Limits (-1.0 to +1.0)
    thrust = np.clip(thrust, -1.0, 1.0)

    # Update last thrust and attitude for fallback
    _last_thrust = thrust
    _last_att_rate = att_rate

    # Store current errors for next derivative calculation
    _prev_pos_error = pos_error
    _prev_vel_error = vel_error
    if current_attitude is not None:
        _prev_att_error = att_error

    return {
        "thrust_x": thrust[0],
        "thrust_y": thrust[1],
        "thrust_z": thrust[2],
        "att_rate_roll": att_rate[1],  # map to output
        "att_rate_pitch": att_rate[0], # map to output
        "att_rate_yaw": att_rate[2],   # map to output
    }
