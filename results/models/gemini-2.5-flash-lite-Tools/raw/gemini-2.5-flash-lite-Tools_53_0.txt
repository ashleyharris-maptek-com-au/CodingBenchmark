import numpy as np

class Spacecraft:
    def __init__(self, pos, vel):
        self.pos = np.array(pos, dtype=float)
        self.vel = np.array(vel, dtype=float)
        self.mass = 1000.0

    def apply_thrust(self, thrust, dt):
        # F = ma -> a = F/m
        accel = np.array(thrust) / self.mass
        self.vel += accel * dt
        self.pos += self.vel * dt

class PIDController:
    def __init__(self, kp, ki, kd):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.prev_error = np.array([0.0, 0.0])
        self.integral = np.array([0.0, 0.0])

    def compute(self, error, dt):
        self.integral += error * dt
        derivative = (error - self.prev_error) / dt
        output = self.kp * error + self.ki * self.integral + self.kd * derivative
        self.prev_error = error
        return output

def simulate_docking():
    craft = Spacecraft(pos=[100.0, 50.0], vel=[-1.0, -0.5])
    pid = PIDController(kp=10.0, ki=0.1, kd=5.0)
    dt = 0.1
    
    print(f"Initial State: Pos={craft.pos}, Vel={craft.vel}")
    
    for i in range(100):
        error = -craft.pos  # Target is at [0, 0]
        thrust = pid.compute(error, dt)
        
        # Limit thrust
        thrust = np.clip(thrust, -500, 500)
        
        craft.apply_thrust(thrust, dt)
        
        if i % 10 == 0:
            print(f"Step {i}: Pos={craft.pos}, Vel={craft.vel}")
            
    print(f"Final State: Pos={craft.pos}, Vel={craft.vel}")

if __name__ == "__main__":
    simulate_docking()
