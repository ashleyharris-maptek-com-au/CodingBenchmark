const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const { width, height, grid, ignitions, towns, cutBudget } = JSON.parse(input);

    // Represent grid as a 2D array for easier access
    const forest = grid.map(row => row.split(''));

    // BFS to find the distance of each cell from the nearest ignition
    const distances = Array.from({ length: height }, () => Array(width).fill(Infinity));
    const queue = [];

    for (const [x, y] of ignitions) {
        if (forest[y][x] === '.') {
            distances[y][x] = 0;
            queue.push([x, y]);
        }
    }

    let head = 0;
    while (head < queue.length) {
        const [x, y] = queue[head++];
        const d = distances[y][x];

        const neighbors = [
            [x + 1, y], [x - 1, y], [x, y + 1], [x, y - 1]
        ];

        for (const [nx, ny] of neighbors) {
            if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
                forest[ny][nx] === '.' && distances[ny][nx] === Infinity) {
                distances[ny][nx] = d + 1;
                queue.push([nx, ny]);
            }
        }
    }

    // Identify candidate cut cells: burnable cells that are on the shortest path
    // between fire and towns. Simple heuristic: prioritize cells with low distance
    // from fire that are close to high-value towns.
    const candidates = [];
    for (let y = 0; y < height; y++) {
        for (let x = 0; x < width; x++) {
            if (forest[y][x] === '.' && distances[y][x] !== Infinity) {
                let townPotential = 0;
                for (const town of towns) {
                    const distToTown = Math.abs(town.x - x) + Math.abs(town.y - y);
                    // Heuristic: cells closer to fire AND closer to towns are high priority
                    // The idea is to cut them off early.
                    townPotential += town.value / (distToTown + 1);
                }
                candidates.push({ x, y, score: townPotential / (distances[y][x] + 1) });
            }
        }
    }

    // Sort by score descending and take the top cutBudget cells
    candidates.sort((a, b) => b.score - a.score);
    const firebreaks = candidates.slice(0, cutBudget).map(c => [c.x, c.y]);

    process.stdout.write(JSON.stringify({ firebreaks }));
}

solve();
