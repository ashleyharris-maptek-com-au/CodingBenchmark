const fs = require('fs');

// Input interfaces
interface InputTask {
    id: string;
    duration: number;
    deps: string[];
    deadline: number;
    weight: number;
}

interface InputData {
    machineCount: number;
    tasks: InputTask[];
}

// Internal Task representation (with added fields for processing)
interface Task extends InputTask {
    // earlyStart?: number; // Earliest possible start time due to dependencies - not strictly needed as it's calculated on the fly
    // finishTime?: number; // Actual finish time after scheduling - tracked in taskFinishTimes map
}

// Output interfaces
interface ScheduleEntry {
    id: string;
    start: number;
    machine: number;
}

interface OutputData {
    schedule?: ScheduleEntry[];
    order?: string[]; // Fallback/simpler output if preferred, but primary is schedule
}

function solve() {
    // 1. Read Input
    const input: string = fs.readFileSync(0, 'utf8');
    const data: InputData = JSON.parse(input);

    // 2. Preprocessing & Graph Construction
    const tasksMap = new Map<string, Task>();
    const dependents = new Map<string, string[]>(); // key: task ID, value: list of task IDs that depend on it
    const inDegree = new Map<string, number>(); // key: task ID, value: count of unmet dependencies

    // Initialize inDegree for all tasks to 0 and populate tasksMap
    data.tasks.forEach(task => {
        tasksMap.set(task.id, task);
        inDegree.set(task.id, 0);
        dependents.set(task.id, []); // Initialize dependents list for all tasks
    });

    // Build dependency graph and calculate initial in-degrees
    data.tasks.forEach(task => {
        task.deps.forEach(depId => {
            const currentInDegree = inDegree.get(task.id) || 0;
            inDegree.set(task.id, currentInDegree + 1);

            const currentDependents = dependents.get(depId);
            if (currentDependents) {
                currentDependents.push(task.id);
            } else {
                // This case should ideally not happen if dependents map is pre-initialized for all tasks
                // but adding defensive code just in case.
                dependents.set(depId, [task.id]);
            }
        });
    });

    // 3. Topological Sort (Kahn's Algorithm)
    const readyQueue: string[] = [];
    const topologicalOrder: string[] = [];

    // Initialize readyQueue with all tasks that have an in-degree of 0
    inDegree.forEach((degree, taskId) => {
        if (degree === 0) {
            readyQueue.push(taskId);
        }
    });

    // Process tasks in topological order
    while (readyQueue.length > 0) {
        const currentTaskId = readyQueue.shift()!;
        topologicalOrder.push(currentTaskId);

        const currentDependents = dependents.get(currentTaskId) || [];
        currentDependents.forEach(dependentId => {
            const currentInDegree = inDegree.get(dependentId)!;
            inDegree.set(dependentId, currentInDegree - 1);
            if (inDegree.get(dependentId) === 0) {
                readyQueue.push(dependentId);
            }
        });
    }

    // Error Check: If topologicalOrder.length !== data.tasks.length, there's a cycle.
    // For this benchmark, assuming valid DAG.

    // 4. Greedy Schedule Generation
    const machineFinishTimes: number[] = new Array(data.machineCount).fill(0); // tracks when each machine is free
    const schedule: ScheduleEntry[] = [];
    const taskFinishTimes = new Map<string, number>(); // Store actual finish times for dependency checking

    // Iterate through the tasks in topological order
    for (const taskId of topologicalOrder) {
        const task = tasksMap.get(taskId)!;

        // Calculate earliest possible start time based on dependencies
        let depsFinishTime = 0;
        for (const depId of task.deps) {
            depsFinishTime = Math.max(depsFinishTime, taskFinishTimes.get(depId) || 0);
        }

        // Find the best machine (earliest available)
        let bestMachineIndex = -1;
        let earliestAvailableTime = Infinity;

        for (let m = 0; m < data.machineCount; m++) {
            const availableTimeOnMachine = Math.max(machineFinishTimes[m], depsFinishTime);
            if (availableTimeOnMachine < earliestAvailableTime) {
                earliestAvailableTime = availableTimeOnMachine;
                bestMachineIndex = m;
            }
        }

        // Assign the task
        const actualStartTime = earliestAvailableTime;
        const actualFinishTime = actualStartTime + task.duration;

        machineFinishTimes[bestMachineIndex] = actualFinishTime;
        taskFinishTimes.set(taskId, actualFinishTime); // Store for dependents

        schedule.push({ id: taskId, start: actualStartTime, machine: bestMachineIndex });
    }

    // 5. Output Generation
    const output: OutputData = { schedule: schedule };
    process.stdout.write(JSON.stringify(output, null, 2));
}

solve();
