interface Task {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
}

interface Input {
  machineCount: number;
  tasks: Task[];
}

interface ScheduleEntry {
  id: string;
  start: number;
  machine: number;
}

function solve() {
  const fs = require('fs');
  const input: string = fs.readFileSync(0, 'utf8');
  const data: Input = JSON.parse(input);

  const { machineCount, tasks } = data;

  // Build graph
  const adj = new Map<string, string[]>();
  const inDegree = new Map<string, number>();
  const taskMap = new Map<string, Task>();
  
  for (const task of tasks) {
    taskMap.set(task.id, task);
    if (!adj.has(task.id)) adj.set(task.id, []);
    if (!inDegree.has(task.id)) inDegree.set(task.id, 0);
  }

  for (const task of tasks) {
    for (const dep of task.deps) {
      if (!adj.has(dep)) adj.set(dep, []);
      adj.get(dep)!.push(task.id);
      inDegree.set(task.id, (inDegree.get(task.id) || 0) + 1);
    }
  }

  // Priority Queue for scheduling based on deadline and weight
  // Simple heuristic: earlier deadline is more urgent
  const readyTasks: Task[] = [];
  for (const [id, degree] of inDegree.entries()) {
    if (degree === 0) {
      readyTasks.push(taskMap.get(id)!);
    }
  }

  const schedule: ScheduleEntry[] = [];
  const machineAvailableTime = new Array(machineCount).fill(0);
  const taskFinishTime = new Map<string, number>();
  const completedTasks = new Set<string>();

  while (readyTasks.length > 0) {
    // Sort ready tasks: earliest deadline first
    readyTasks.sort((a, b) => a.deadline - b.deadline);
    
    const task = readyTasks.shift()!;
    
    // Find earliest start time based on dependencies
    let startTime = 0;
    for (const dep of task.deps) {
      startTime = Math.max(startTime, taskFinishTime.get(dep) || 0);
    }

    // Find earliest available machine
    let machineIdx = 0;
    let minAvailableTime = Infinity;
    
    // Actually, we want to find the machine that becomes free soonest,
    // but the task cannot start before dependencies finish.
    // So the task can start at max(machineAvailableTime[i], dependencyReadyTime).
    
    let bestMachine = 0;
    let bestStartTime = Infinity;

    for(let i=0; i<machineCount; i++) {
        const potentialStartTime = Math.max(startTime, machineAvailableTime[i]);
        if(potentialStartTime < bestStartTime) {
            bestStartTime = potentialStartTime;
            bestMachine = i;
        }
    }

    const finishTime = bestStartTime + task.duration;
    
    schedule.push({
      id: task.id,
      start: bestStartTime,
      machine: bestMachine
    });

    machineAvailableTime[bestMachine] = finishTime;
    taskFinishTime.set(task.id, finishTime);
    completedTasks.add(task.id);

    // Update ready tasks
    for (const neighbor of (adj.get(task.id) || [])) {
      inDegree.set(neighbor, inDegree.get(neighbor)! - 1);
      if (inDegree.get(neighbor) === 0) {
        readyTasks.push(taskMap.get(neighbor)!);
      }
    }
  }

  process.stdout.write(JSON.stringify({ schedule }));
}

solve();
