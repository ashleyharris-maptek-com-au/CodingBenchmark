OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft

               ; Decorations
               OpDecorate %outColor Location 0

               ; Types
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
%ptr_Output_v4float = OpTypePointer Output %v4float
   %voidfunc = OpTypeFunction %void

               ; Constants (Red: 1.0, 0.0, 0.0, 1.0)
    %float_1 = OpConstant %float 1
    %float_0 = OpConstant %float 0
      %red = OpConstantComposite %v4float %float_1 %float_0 %float_0 %float_1

               ; Variables
   %outColor = OpVariable %ptr_Output_v4float Output

               ; Main Function
       %main = OpFunction %void None %voidfunc
      %label = OpLabel
               OpStore %outColor %red
               OpReturn
               OpFunctionEnd
