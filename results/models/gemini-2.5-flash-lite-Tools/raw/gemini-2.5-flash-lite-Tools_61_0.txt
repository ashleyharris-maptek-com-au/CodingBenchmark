declare const require: any;
const fs = require('fs');

interface Point {
    x: number;
    y: number;
}

interface Package {
    id: number;
    x: number;
    y: number;
}

interface InputData {
    depot: [number, number];
    truckCount: number;
    packages: Package[];
}

function distance(p1: Point, p2: Point): number {
    return Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
}

function solve(): void {
    const input: string = fs.readFileSync(0, 'utf8');
    const data: InputData = JSON.parse(input);

    const depot: Point = { x: data.depot[0], y: data.depot[1] };
    const packages = [...data.packages];
    const routes: number[][] = Array.from({ length: data.truckCount }, () => []);

    // Greedy approach: assign each package to the truck that makes the most sense.
    // To simplify: distribute packages cyclically or by proximity.
    // Let's use a simple approach: assign to the truck that adds the least distance.

    const currentPositions: Point[] = Array.from({ length: data.truckCount }, () => ({ ...depot }));

    while (packages.length > 0) {
        for (let t = 0; t < data.truckCount; t++) {
            if (packages.length === 0) break;

            let bestPackageIndex = -1;
            let minDistance = Infinity;

            for (let i = 0; i < packages.length; i++) {
                const dist = distance(currentPositions[t], packages[i]);
                if (dist < minDistance) {
                    minDistance = dist;
                    bestPackageIndex = i;
                }
            }

            const pkg = packages.splice(bestPackageIndex, 1)[0];
            routes[t].push(pkg.id);
            currentPositions[t] = { x: pkg.x, y: pkg.y };
        }
    }

    process.stdout.write(JSON.stringify({ routes }));
}

solve();
