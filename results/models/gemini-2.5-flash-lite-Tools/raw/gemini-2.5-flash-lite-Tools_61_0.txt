interface Point {
    x: number;
    y: number;
}

interface Package extends Point {
    id: string;
}

interface InputData {
    depot: [number, number];
    truckCount: number;
    packages: Package[];
}

interface OutputData {
    routes: string[][];
}

// Function to calculate Euclidean distance between two points
function calculateDistance(p1: Point, p2: Point): number {
    return Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
}

function solveVRP() {
    // Read input from stdin
    const fs = require('fs');
    const input: string = fs.readFileSync(0, 'utf8');
    const data: InputData = JSON.parse(input);

    const depot: Point = { x: data.depot[0], y: data.depot[1] };
    const packages: Package[] = data.packages;
    const truckCount: number = data.truckCount;

    // Initialize routes
    // Each route will be an array of Package objects, making distance calculation easier
    const routes: Package[][] = [];
    for (let i = 0; i < truckCount; i++) {
        routes.push([]);
    }

    // Keep track of unassigned packages
    const unassignedPackages: Set<string> = new Set(packages.map(p => p.id));
    const packageMap: Map<string, Package> = new Map(packages.map(p => [p.id, p]));

    // Simple greedy approach: assign packages one by one to the route/position
    // that minimizes the immediate increase in total route distance.

    // Sort packages by distance from depot to prioritize closer packages?
    // Not strictly necessary for this greedy approach, but can sometimes help.
    // For now, process in original order.

    for (const pkg of packages) {
        let bestRouteIndex: number = -1;
        let bestInsertIndex: number = -1;
        let minIncrease: number = Infinity;

        for (let rIdx = 0; rIdx < routes.length; rIdx++) {
            const currentRoute = routes[rIdx]!;

            // Consider inserting at the beginning (after depot)
            // depot -> pkg -> first_in_route ...
            let currentRouteDistance = 0;
            if (currentRoute.length > 0) {
                currentRouteDistance = calculateDistance(depot, currentRoute[0]!);
                for (let i = 0; i < currentRoute.length - 1; i++) {
                    currentRouteDistance += calculateDistance(currentRoute[i]!, currentRoute[i + 1]!);
                }
                currentRouteDistance += calculateDistance(currentRoute[currentRoute.length - 1]!, depot);
            } else {
                // Empty route, just depot -> pkg -> depot
                currentRouteDistance = calculateDistance(depot, depot); // Should be 0
            }


            // Option 1: Insert at the beginning (after depot)
            // depot -> pkg -> first_package -> ... -> depot
            if (currentRoute.length === 0) {
                // If route is empty, inserting pkg means depot -> pkg -> depot
                const increase = calculateDistance(depot, pkg) + calculateDistance(pkg, depot);
                if (increase < minIncrease) {
                    minIncrease = increase;
                    bestRouteIndex = rIdx;
                    bestInsertIndex = 0;
                }
            } else {
                // If route has packages, inserting at beginning means depot -> pkg -> currentRoute[0]
                const originalSegment = calculateDistance(depot, currentRoute[0]!);
                const newSegment = calculateDistance(depot, pkg) + calculateDistance(pkg, currentRoute[0]!);
                const increase = newSegment - originalSegment;
                if (increase < minIncrease) {
                    minIncrease = increase;
                    bestRouteIndex = rIdx;
                    bestInsertIndex = 0;
                }
            }


            // Consider inserting between existing packages
            // ... -> currentRoute[i] -> pkg -> currentRoute[i+1] -> ...
            for (let i = 0; i < currentRoute.length - 1; i++) {
                const p1 = currentRoute[i]!;
                const p2 = currentRoute[i + 1]!;
                const originalSegment = calculateDistance(p1, p2);
                const newSegment = calculateDistance(p1, pkg) + calculateDistance(pkg, p2);
                const increase = newSegment - originalSegment;

                if (increase < minIncrease) {
                    minIncrease = increase;
                    bestRouteIndex = rIdx;
                    bestInsertIndex = i + 1;
                }
            }

            // Consider inserting at the end (before depot)
            // ... -> last_package -> pkg -> depot
            if (currentRoute.length > 0) {
                const lastPackage = currentRoute[currentRoute.length - 1]!;
                const originalSegment = calculateDistance(lastPackage, depot);
                const newSegment = calculateDistance(lastPackage, pkg) + calculateDistance(pkg, depot);
                const increase = newSegment - originalSegment;

                if (increase < minIncrease) {
                    minIncrease = increase;
                    bestRouteIndex = rIdx;
                    bestInsertIndex = currentRoute.length;
                }
            }
        }
        
        // If a valid insertion point was found, perform the insertion
        if (bestRouteIndex !== -1) {
            routes[bestRouteIndex]!.splice(bestInsertIndex, 0, pkg);
            unassignedPackages.delete(pkg.id);
        } else {
            // This case should ideally not happen if truckCount is sufficient
            // or if an empty route is always an option.
            // For now, let's assume it means a package couldn't be placed.
            // This might happen if truckCount is 0, but problem implies truckCount >= 1.
            // If the code reaches here and there are unassigned packages, it's an issue.
            // Log for debugging if necessary, but final output must be clean.
            // console.error(`Could not assign package ${pkg.id}`);
        }
    }


    // Convert routes of Package objects to routes of package IDs
    const outputRoutes: string[][] = routes
        .filter(route => route.length > 0) // Remove empty routes if any
        .map(route => route.map(pkg => pkg.id));

    const output: OutputData = {
        routes: outputRoutes,
    };

    // Print output to stdout
    console.log(JSON.stringify(output));
}

solveVRP();
