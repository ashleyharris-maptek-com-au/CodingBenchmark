struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);

    // Calculate the light direction vector
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Calculate the Lambertian diffuse intensity
    // Max with 0 to ensure no negative lighting
    float diffuse_intensity = max(dot(N, L), 0.0);

    // Output grayscale intensity with alpha 1.0
    return float4(diffuse_intensity, diffuse_intensity, diffuse_intensity, 1.0);
}
