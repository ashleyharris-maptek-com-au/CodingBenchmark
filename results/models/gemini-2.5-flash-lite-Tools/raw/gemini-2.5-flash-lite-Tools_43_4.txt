// Pixel Shader for Vulkan
// This shader samples a texture and multiplies it by a vertex color.
// It is designed to be compiled to SPIR-V using DXC (DirectX Shader Compiler).

// 1. Input structure from Vertex Shader
struct PixelInput {
    float4 position : SV_POSITION; // Screen-space position
    float2 uv       : TEXCOORD0;   // Texture coordinates
    float4 color    : COLOR0;      // Interpolated vertex color
};

// 2. Resources with explicit Vulkan binding layouts
// Binding 0, Set 0 for the Texture
[[vk::binding(0, 0)]] Texture2D    MainTexture;
// Binding 1, Set 0 for the Sampler
[[vk::binding(1, 0)]] SamplerState MainSampler;

// 3. Entry point function
// Output to the first Render Target (SV_TARGET0)
float4 main(PixelInput input) : SV_TARGET0 {
    // Sample the texture at the given UV coordinates
    float4 texColor = MainTexture.Sample(MainSampler, input.uv);
    
    // Multiply texture color by the vertex color (tinting)
    float4 finalColor = texColor * input.color;
    
    return finalColor;
}
