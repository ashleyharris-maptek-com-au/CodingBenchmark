import * as fs from 'fs';

interface Input {
    probability: number[][];
    drones: { id: number; x: number; y: number }[];
    maxStepsPerDrone: number;
}

interface Output {
    paths: [number, number][][];
}

function solve() {
    const inputString = fs.readFileSync(0, 'utf8');
    const data: Input = JSON.parse(inputString);

    const { probability, drones, maxStepsPerDrone } = data;
    const rows = probability.length;
    const cols = probability[0].length;

    // Use a set to track visited high-value cells to avoid redundant coverage
    const visited = new Set<string>();

    const paths: [number, number][][] = drones.map(drone => {
        let { x, y } = drone;
        const path: [number, number][] = [[x, y]];
        visited.add(`${x},${y}`);

        for (let step = 0; step < maxStepsPerDrone; step++) {
            // Greedy strategy: find best next cell
            let bestNext: [number, number] = [x, y];
            let bestValue = -1;

            const moves = [
                [0, 1], [0, -1], [1, 0], [-1, 0], [0, 0] // include stay
            ];

            for (const [dx, dy] of moves) {
                const nx = x + dx;
                const ny = y + dy;

                if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && probability[ny][nx] !== -1) {
                    const key = `${nx},${ny}`;
                    let value = probability[ny][nx];
                    
                    // If already visited, value is 0 (don't want to re-cover if possible)
                    if (visited.has(key)) {
                        value = 0;
                    }

                    if (value > bestValue) {
                        bestValue = value;
                        bestNext = [nx, ny];
                    }
                }
            }

            x = bestNext[0];
            y = bestNext[1];
            path.push([x, y]);
            visited.add(`${x},${y}`);
        }
        return path;
    });

    const output: Output = { paths };
    process.stdout.write(JSON.stringify(output));
}

solve();
