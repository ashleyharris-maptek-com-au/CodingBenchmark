; SPIR-V
; Version: 1.0
; Generator: Unknown
; Target: Vulkan 1.0

OpCapability Shader
OpMemoryModel Logical GLSL450

; Entry point
OpEntryPoint Fragment %main "main" %outColor
OpExecutionMode %main OriginUpperLeft

; Input/Output variables
OpDecorate %worldPos Input
OpDecorate %worldPos Location 0
OpDecorate %normal Input
OpDecorate %normal Location 1
OpDecorate %uv Input
OpDecorate %uv Location 2
OpDecorate %tangent Input
OpDecorate %tangent Location 3
OpDecorate %color Input
OpDecorate %color Location 4
OpDecorate %outColor Output
OpDecorate %outColor Location 0

; Uniform buffer
OpDecorate %ubo Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %uboUniform Buffer
OpDecorate %uboUniform Offset 0
OpDecorate %uboUniform MatrixStride 16
OpDecorate %uboUniform ColMajor

OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%f32 = OpConstant %float 1.0
%f0_5 = OpConstant %float 0.5
%f0_1 = OpConstant %float 0.1
%f0_4 = OpConstant %float 0.4
%f1_0 = OpConstant %float 1.0

%mat4x4float = OpTypeMatrix %v4float 4

%ubo = OpTypeStruct %mat4x4float %mat4x4float %mat4x4float %v4float %v4float %v4float
%uniformConstant = OpTypeQualifier %ubo Uniform
%uniformConstantSampler = OpTypeImage %ubo %void %void %void %void %void %void %void %void %void

%ptrUniform = OpTypePointer Uniform %ubo
%ptrOutput = OpTypePointer Output %v4float

; Variables
%ubo = OpVariable %ptrUniform Uniform

%worldPos = OpVariable Input %v3float
%normal = OpVariable Input %v3float
%uv = OpVariable Input %v2float
%tangent = OpVariable Input %v3float
%color = OpVariable Input %v4float
%outColor = OpVariable %ptrOutput

; Entry point function
%main = OpFunction %void None %easingFunction

; Function body
    %temp_worldPos = OpLoad %v3float %worldPos
    %worldPosY = OpCompositeExtract %float %temp_worldPos 1

    ; t = clamp(worldPos.y * 0.5 + 0.5)
    %mult_y = OpFMul %float %worldPosY %f0_5
    %add_y = OpFAdd %float %mult_y %f0_5
    %t = OpClamp %float %add_y %f0_0 %f1_0

    ; bottom color (0.1, 0.4, 0.1)
    %bottomColor = OpConstantComposite %v3float %f0_1 %f0_4 %f0_1

    ; top color (1.0, 1.0, 1.0)
    %topColor = OpConstantComposite %v3float %f1_0 %f1_0 %f1_0

    ; mix bottom to top based on t
    %mixedColor = OpBlendVBlendV %v3float %t %bottomColor %topColor

    ; output alpha 1.0
    %finalColor = OpCompositeConstruct %v4float %mixedColor %f1_0

    ; Store output
    OpStore %outColor %finalColor

    OpReturn
%easingFunction = OpTypeFunction %void
%f0_0 = OpConstant %float 0.0
%f1_0 = OpConstant %float 1.0
