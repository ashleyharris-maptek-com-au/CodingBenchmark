; SPIR-V Fragment Shader
; Output: Solid Red Color

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450
               OpDecorate %outColor Location 0

; Types, Constants
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
%float_1_0 = OpConstant %float 1.0
%float_0_0 = OpConstant %float 0.0
   %v4red = OpConstantComposite %v4float %float_1_0 %float_0_0 %float_0_0 %float_1_0

; Function
    %void_f = OpTypeFunction %void
   %outColor = OpVariable %_ptr_Output_v4float Output
%_ptr_Output_v4float = OpTypePointer Output %v4float

       %main = OpFunction %void None %void_f
     %label = OpLabel
               OpStore %outColor %v4red
               OpReturn
               OpFunctionEnd
