; SPIR-V Assembly for a simple Geometry Shader
; Input: Triangles
; Output: TriangleStrip
; Passthrough geometry shader

               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %in_pos %out_pos
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorations (Layout)
               OpDecorate %in_pos BuiltIn Position
               OpDecorate %out_pos BuiltIn Position

; Types
%void       = OpTypeVoid
%float      = OpTypeFloat 32
%int        = OpTypeInt 32 0
%vec4       = OpTypeVector %float 4
%int_0      = OpConstant %int 0
%int_1      = OpConstant %int 1
%int_2      = OpConstant %int 2
%float_arr  = OpTypeArray %float 3 ; Input array size
%vec4_arr   = OpTypeArray %vec4 %float_arr
%ptr_in     = OpTypePointer Input %vec4_arr
%ptr_out    = OpTypePointer Output %vec4

; Inputs/Outputs
%in_pos     = OpVariable %ptr_in Input
%out_pos    = OpVariable %ptr_out Output

; Functions
%void_func  = OpTypeFunction %void
%main       = OpFunction %void None %void_func
%entry      = OpLabel

; Assuming fixed 3 vertices for passthrough
; Vertex 0
%pos0_ptr   = OpAccessChain %ptr_in %in_pos %int_0
%pos0       = OpLoad %vec4 %pos0_ptr
               OpStore %out_pos %pos0
               OpEmitVertex

; Vertex 1
%pos1_ptr   = OpAccessChain %ptr_in %in_pos %int_1
%pos1       = OpLoad %vec4 %pos1_ptr
               OpStore %out_pos %pos1
               OpEmitVertex

; Vertex 2
%pos2_ptr   = OpAccessChain %ptr_in %in_pos %int_2
%pos2       = OpLoad %vec4 %pos2_ptr
               OpStore %out_pos %pos2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
