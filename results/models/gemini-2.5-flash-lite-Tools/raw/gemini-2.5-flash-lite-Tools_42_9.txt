; SPIR-V Geometry Shader
; Input: InputLinesAdjacency (4 vertices per invocation)
; Output: OutputTriangleStrip (4 vertices)

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
%glsl_std_450 = OpExtInstImport "GLSL.std.450"

; --- Decorations ---
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; --- Types ---
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v4 = OpTypePointer Input %v4float

%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_1 = OpConstant %float 1.0
%float_0_025 = OpConstant %float 0.025

; --- Inputs ---
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; --- Outputs ---
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; --- Main Function ---
%func_main = OpTypeFunction %void
%main = OpFunction %void None %func_main
%label_entry = OpLabel

; Execution Modes
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Load positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_v3 = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p3_v3 = OpVectorShuffle %v3float %p3 %p3 0 1 2

%diff0 = OpFSub %v3float %p2_v3 %p0_v3
%t0 = OpExtInst %v3float %glsl_std_450 25 %diff0

%diff1 = OpFSub %v3float %p3_v3 %p1_v3
%t1 = OpExtInst %v3float %glsl_std_450 25 %diff1

%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%perp0_x = OpFNegate %float %t0_y
%perp0_y = %t0_x

%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%perp1_x = OpFNegate %float %t1_y
%perp1_y = %t1_x

%thick = OpConstant %float 0.025

; Vertex 1: start + perp0 * thickness
%p1_x = OpCompositeExtract %float %p1_v3 0
%p1_y = OpCompositeExtract %float %p1_v3 1
%p1_z = OpCompositeExtract %float %p1_v3 2
%mul0x = OpFMul %float %perp0_x %thick
%v1_x = OpFAdd %float %p1_x %mul0x
%mul0y = OpFMul %float %perp0_y %thick
%v1_y = OpFAdd %float %p1_y %mul0y
%v1_v4 = OpCompositeConstruct %v4float %v1_x %v1_y %p1_z %float_1
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %v1_v4
%color1_ptr = OpAccessChain %ptr_in_arr_v3 %color %int_1
%color1 = OpLoad %v3float %color1_ptr
OpStore %outColor %color1
OpEmitVertex

; Vertex 2: start - perp0 * thickness
%v2_x = OpFSub %float %p1_x %mul0x
%v2_y = OpFSub %float %p1_y %mul0y
%v2_v4 = OpCompositeConstruct %v4float %v2_x %v2_y %p1_z %float_1
OpStore %out_pos_ptr %v2_v4
OpStore %outColor %color1
OpEmitVertex

; Vertex 3: end + perp1 * thickness
%p2_x = OpCompositeExtract %float %p2_v3 0
%p2_y = OpCompositeExtract %float %p2_v3 1
%p2_z = OpCompositeExtract %float %p2_v3 2
%mul1x = OpFMul %float %perp1_x %thick
%v3_x = OpFAdd %float %p2_x %mul1x
%mul1y = OpFMul %float %perp1_y %thick
%v3_y = OpFAdd %float %p2_y %mul1y
%v3_v4 = OpCompositeConstruct %v4float %v3_x %v3_y %p2_z %float_1
OpStore %out_pos_ptr %v3_v4
%color2_ptr = OpAccessChain %ptr_in_arr_v3 %color %int_2
%color2 = OpLoad %v3float %color2_ptr
OpStore %outColor %color2
OpEmitVertex

; Vertex 4: end - perp1 * thickness
%v4_x = OpFSub %float %p2_x %mul1x
%v4_y = OpFSub %float %p2_y %mul1y
%v4_v4 = OpCompositeConstruct %v4float %v4_x %v4_y %p2_z %float_1
OpStore %out_pos_ptr %v4_v4
OpStore %outColor %color2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
