const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  const { depot, truckCount, packages } = JSON.parse(input);

  const numTrucks = truckCount;
  const numPackages = packages.length;

  if (numPackages === 0) {
    console.log(JSON.stringify({ routes: Array(numTrucks).fill([]) }));
    return;
  }

  // Calculate distances for quick reference
  const getDist = (p1, p2) => Math.hypot(p1[0] - p2[0], p1[1] - p2[1]);

  // Packages with depot as a reference
  const unvisited = new Set(packages);
  const routes = Array.from({ length: numTrucks }, () => []);
  
  // Current position of each truck (starts at depot)
  const truckPositions = Array.from({ length: numTrucks }, () => depot);

  while (unvisited.size > 0) {
    for (let i = 0; i < numTrucks; i++) {
      if (unvisited.size === 0) break;

      let closestPackage = null;
      let minDistance = Infinity;

      for (const pkg of unvisited) {
        const dist = getDist(truckPositions[i], [pkg.x, pkg.y]);
        if (dist < minDistance) {
          minDistance = dist;
          closestPackage = pkg;
        }
      }

      routes[i].push(closestPackage.id);
      truckPositions[i] = [closestPackage.x, closestPackage.y];
      unvisited.delete(closestPackage);
    }
  }

  console.log(JSON.stringify({ routes }));
}

solve();
