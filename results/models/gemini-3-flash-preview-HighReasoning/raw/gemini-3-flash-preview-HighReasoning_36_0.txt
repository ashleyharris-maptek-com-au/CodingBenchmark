The Maximum Independent Set (MIS) problem is NP-hard, requiring an exact algorithm for the strict evaluator. The implemented solution uses a **Branch-and-Bound** approach optimized with **Bitsets** and **Graph Reductions**.

### Key Features:
1.  **Bitset Representation:** Adjacency and candidate sets are represented using bitsets (`std::vector<uint64_t>`), allowing for $O(N/64)$ set operations (intersection, subtraction, counting). This is highly efficient for the branching process.
2.  **Reductions:** The solver applies **Degree 0** and **Degree 1** reduction rules at each step. These rules are essential for handling large sparse graphs, as they can significantly prune the search space.
3.  **Branching Heuristic:** The algorithm branches on the vertex with the **maximum degree** in the current subgraph. This heuristic helps in finding large independent sets early and reduces the problem size quickly.
4.  **Greedy Initialization:** A "Minimum Degree First" greedy algorithm provides a strong initial lower bound, enabling more effective pruning from the start.
5.  **Memory Efficiency:** The bitset adjacency matrix for $N=100,000$ vertices uses approximately 1.25 GB, well within the 32 GB limit.

### C++ Implementation:

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>
#include <stdint.h>
#include <string.h>
#include <bit>

using namespace std;

typedef uint64_t uint64;

int N_global, N64_global;

// Bitset wrapper for efficient set operations on the graph
struct Bitset {
    uint64* bits;
    Bitset(uint64* p) : bits(p) {}
    inline void set(int i) { bits[i >> 6] |= (1ULL << (i & 63)); }
    inline void reset(int i) { bits[i >> 6] &= ~(1ULL << (i & 63)); }
    inline bool test(int i) const { return (bits[i >> 6] >> (i & 63)) & 1; }
    inline int count() const {
        int cnt = 0;
        for (int i = 0; i < N64_global; ++i) cnt += (int)std::popcount(bits[i]);
        return cnt;
    }
    inline int find_first() const {
        for (int i = 0; i < N64_global; ++i) {
            if (bits[i]) return (i << 6) + std::countr_zero(bits[i]);
        }
        return -1;
    }
    inline int find_next(int prev) const {
        int i = prev >> 6;
        uint64 b;
        if ((prev & 63) == 63) b = 0;
        else b = bits[i] & ~((1ULL << ((prev & 63) + 1)) - 1);
        if (b) return (i << 6) + std::countr_zero(b);
        for (i++; i < N64_global; ++i) {
            if (bits[i]) return (i << 6) + std::countr_zero(bits[i]);
        }
        return -1;
    }
};

vector<uint64> adj_pool;
vector<uint64*> adj_ptrs;
vector<int> best_is;
int max_is_size = 0;
vector<int> current_is;

// Recursive Branch-and-Bound solver
void solve(uint64* p_bits) {
    Bitset P(p_bits);
    int p_cnt = P.count();
    
    // Pruning: if current set + remaining candidates cannot beat max_is_size
    if ((int)current_is.size() + p_cnt <= max_is_size) return;
    
    if (p_cnt == 0) {
        if ((int)current_is.size() > max_is_size) {
            max_is_size = (int)current_is.size();
            best_is = current_is;
        }
        return;
    }

    int v_max = -1;
    int max_d = -1;
    
    // Iterate over candidates to find max degree and apply reductions
    for (int i = P.find_first(); i != -1; i = P.find_next(i)) {
        int d = 0;
        uint64* a = adj_ptrs[i];
        for (int k = 0; k < N64_global; ++k) d += (int)std::popcount(a[k] & p_bits[k]);
        
        // Degree 0 Reduction: vertex must be in MIS
        if (d == 0) {
            current_is.push_back(i); P.reset(i);
            solve(p_bits);
            P.set(i); current_is.pop_back();
            return;
        }
        // Degree 1 Reduction: vertex is better or equal to its neighbor
        if (d == 1) {
            int neighbor = -1;
            for (int k = 0; k < N64_global; ++k) {
                uint64 common = adj_ptrs[i][k] & p_bits[k];
                if (common) { neighbor = (k << 6) + std::countr_zero(common); break; }
            }
            current_is.push_back(i); P.reset(i);
            bool n_removed = false;
            if (neighbor != -1 && P.test(neighbor)) { P.reset(neighbor); n_removed = true; }
            solve(p_bits);
            if (n_removed) P.set(neighbor);
            P.set(i); current_is.pop_back();
            return;
        }
        if (d > max_d) { max_d = d; v_max = i; }
    }

    if (v_max == -1) return;

    int v = v_max;
    // Branch 1: v is in the Maximum Independent Set
    current_is.push_back(v);
    vector<uint64> next_p(N64_global);
    for (int k = 0; k < N64_global; ++k) next_p[k] = p_bits[k] & ~adj_ptrs[v][k];
    next_p[v >> 6] &= ~(1ULL << (v & 63));
    solve(next_p.data());
    current_is.pop_back();
    
    // Branch 2: v is NOT in the Maximum Independent Set
    P.reset(v);
    solve(p_bits);
    P.set(v);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    long long m;
    if (!(cin >> n >> m)) return 0;
    if (n <= 0) {
        cout << "0\n" << endl;
        return 0;
    }
    
    N_global = n;
    N64_global = (n + 63) / 64;
    
    // Allocate memory for adjacency bitsets
    try {
        adj_pool.assign((size_t)N_global * N64_global, 0);
    } catch (...) {
        return 1;
    }
    adj_ptrs.resize(N_global);
    for (int i = 0; i < N_global; ++i) adj_ptrs[i] = &adj_pool[(size_t)i * N64_global];
    
    for (long long i = 0; i < m; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u == v) continue;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            adj_ptrs[u][v >> 6] |= (1ULL << (v & 63));
            adj_ptrs[v][u >> 6] |= (1ULL << (u & 63));
        }
    }

    // Initial Greedy solution (Min-Degree First)
    {
        vector<int> p(n);
        for(int i=0; i<n; ++i) p[i] = i;
        vector<int> deg(n);
        for(int i=0; i<n; ++i) {
            int d = 0;
            for(int k=0; k<N64_global; ++k) d += (int)std::popcount(adj_ptrs[i][k]);
            deg[i] = d;
        }
        sort(p.begin(), p.end(), [&](int a, int b){ return deg[a] < deg[b]; });
        vector<uint64> rem_bits(N64_global, 0);
        Bitset rem_bs(rem_bits.data());
        vector<int> g_is;
        for(int u : p) {
            if(!rem_bs.test(u)) {
                g_is.push_back(u);
                rem_bs.set(u);
                for(int k=0; k<N64_global; ++k) rem_bs.bits[k] |= adj_ptrs[u][k];
            }
        }
        max_is_size = (int)g_is.size();
        best_is = g_is;
    }

    // Initial candidate set P contains all vertices
    vector<uint64> p_bits(N64_global, ~0ULL);
    if (n % 64 != 0) p_bits[N64_global - 1] = (1ULL << (n % 64)) - 1;
    
    solve(p_bits.data());

    // Output results
    cout << max_is_size << "\n";
    for (int i = 0; i < (int)best_is.size(); ++i) {
        cout << best_is[i] << (i == (int)best_is.size() - 1 ? "" : " ");
    }
    cout << endl;
    
    return 0;
}
```
