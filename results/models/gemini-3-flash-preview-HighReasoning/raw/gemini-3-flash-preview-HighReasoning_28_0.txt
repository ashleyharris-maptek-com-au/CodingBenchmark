use std::io::{self, Read, Write, BufWriter};

struct FastReader<R: Read> {
    reader: R,
    buffer: [u8; 1 << 16],
    pos: usize,
    limit: usize,
}

impl<R: Read> FastReader<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buffer: [0; 1 << 16],
            pos: 0,
            limit: 0,
        }
    }

    fn next_u32(&mut self) -> Option<u32> {
        let mut res = 0u32;
        let mut found = false;
        loop {
            if self.pos >= self.limit {
                match self.reader.read(&mut self.buffer) {
                    Ok(n) if n > 0 => {
                        self.limit = n;
                        self.pos = 0;
                    }
                    _ => return if found { Some(res) } else { None },
                }
            }
            let c = self.buffer[self.pos];
            self.pos += 1;
            if c >= b'0' && c <= b'9' {
                res = res * 10 + (c - b'0') as u32;
                found = true;
            } else if found {
                return Some(res);
            }
        }
    }
}

fn main() {
    let stdin = io::stdin();
    let mut reader = FastReader::new(stdin.lock());

    let n = match reader.next_u32() {
        Some(v) => v as usize,
        None => return,
    };
    let m = match reader.next_u32() {
        Some(v) => v as usize,
        None => return,
    };

    let mut head = vec![!0u32; n];
    let mut next = vec![0u32; 2 * m];
    let mut to = vec![0u32; 2 * m];
    let mut degrees = vec![0u32; n];

    for i in 0..m {
        let u = match reader.next_u32() {
            Some(v) => v as usize,
            None => break,
        };
        let v = match reader.next_u32() {
            Some(v) => v as usize,
            None => break,
        };
        
        if u >= n || v >= n {
            continue;
        }

        degrees[u] += 1;
        degrees[v] += 1;
        
        to[2 * i] = v as u32;
        next[2 * i] = head[u];
        head[u] = (2 * i) as u32;

        to[2 * i + 1] = u as u32;
        next[2 * i + 1] = head[v];
        head[v] = (2 * i + 1) as u32;
    }

    let mut current_degree = degrees.clone();
    let mut in_vc = vec![false; n];
    let mut bucket_pos = vec![!0u32; n];
    let mut max_d = 0usize;
    for &d in &degrees {
        if d as usize > max_d {
            max_d = d as usize;
        }
    }

    let mut buckets = vec![Vec::new(); max_d + 1];
    for i in 0..n {
        let d = degrees[i] as usize;
        bucket_pos[i] = buckets[d].len() as u32;
        buckets[d].push(i as u32);
    }

    let mut d_ptr = max_d;
    while d_ptr > 0 {
        if buckets[d_ptr].is_empty() {
            d_ptr -= 1;
            continue;
        }
        
        let v = buckets[d_ptr].pop().unwrap();
        bucket_pos[v as usize] = !0;
        in_vc[v as usize] = true;

        let mut e = head[v as usize];
        while e != !0 {
            let u = to[e as usize] as usize;
            if !in_vc[u] && bucket_pos[u] != !0 {
                let ud = current_degree[u] as usize;
                if ud > 0 {
                    // Move u from buckets[ud] to buckets[ud-1]
                    let pos = bucket_pos[u] as usize;
                    let last_val = *buckets[ud].last().unwrap();
                    buckets[ud][pos] = last_val;
                    bucket_pos[last_val as usize] = pos as u32;
                    buckets[ud].pop();

                    current_degree[u] -= 1;
                    let new_ud = ud - 1;
                    bucket_pos[u] = buckets[new_ud].len() as u32;
                    buckets[new_ud].push(u as u32);
                }
            }
            e = next[e as usize];
        }
    }

    // Post-processing: remove redundant vertices
    for v in 0..n {
        if in_vc[v] {
            let mut redundant = true;
            let mut e = head[v];
            while e != !0 {
                let u = to[e as usize] as usize;
                if !in_vc[u] {
                    redundant = false;
                    break;
                }
                e = next[e as usize];
            }
            if redundant {
                in_vc[v] = false;
            }
        }
    }

    let mut vc_count = 0usize;
    for &v in &in_vc {
        if v {
            vc_count += 1;
        }
    }

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    let _ = writeln!(writer, "{}", vc_count);
    let mut first = true;
    for i in 0..n {
        if in_vc[i] {
            if !first {
                let _ = write!(writer, " ");
            }
            let _ = write!(writer, "{}", i);
            first = false;
        }
    }
    let _ = writeln!(writer);
}
