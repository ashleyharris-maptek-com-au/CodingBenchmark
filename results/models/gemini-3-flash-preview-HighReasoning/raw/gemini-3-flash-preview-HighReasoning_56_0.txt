```javascript
"use strict";

const fs = require('fs');

/**
 * Solves the Vehicle Routing Problem (VRP) using the Sweep Algorithm (Angle Sort).
 * For each radial slice (route), it further optimizes the delivery order using 
 * the Nearest Neighbor heuristic for small segments, or a Distance Sort for larger ones.
 * This approach scales efficiently to a large number of packages and trucks.
 */
function solve() {
    let input;
    try {
        // Read the entire input from stdin synchronously
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;
    
    let data;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }

    const depot = data.depot;
    const truckCount = data.truckCount;
    const packages = data.packages;

    // Handle empty package list
    if (!packages || packages.length === 0) {
        process.stdout.write(JSON.stringify({ routes: [] }));
        return;
    }

    const n = packages.length;
    const depotX = depot[0];
    const depotY = depot[1];

    // Annotate packages with polar angle relative to the depot for the Sweep algorithm.
    // Also include distance squared for tie-breaking and large segment optimization.
    const annotated = packages.map(p => {
        const dx = p.x - depotX;
        const dy = p.y - depotY;
        return {
            id: p.id,
            x: p.x,
            y: p.y,
            angle: Math.atan2(dy, dx),
            distSq: dx * dx + dy * dy
        };
    });

    // Sort all packages by polar angle to create radial segments.
    annotated.sort((a, b) => a.angle - b.angle);

    // Determine the number of routes to use (up to truckCount).
    const actualTruckCount = Math.max(1, Math.min(truckCount, n));
    
    const routes = [];
    const segmentSize = Math.floor(n / actualTruckCount);
    let extra = n % actualTruckCount;

    let currentIndex = 0;
    for (let i = 0; i < actualTruckCount; i++) {
        // Distribute packages as evenly as possible among routes
        const currentSegmentSize = segmentSize + (extra > 0 ? 1 : 0);
        extra--;

        if (currentSegmentSize <= 0) continue;

        const segment = annotated.slice(currentIndex, currentIndex + currentSegmentSize);
        currentIndex += currentSegmentSize;

        let routeIds;
        /**
         * For each segment, we optimize the visit order.
         * Nearest Neighbor is $O(M^2)$ where M is segment size.
         * We use it for segments up to 1500 packages to keep runtime manageable.
         * For larger segments, a simple Distance Sort provides a decent out-and-back approximation.
         */
        if (currentSegmentSize <= 1500) {
            routeIds = getNNRoute(segment, depotX, depotY);
        } else {
            // Within a large radial slice, sorting by distance minimizes cross-trips.
            segment.sort((a, b) => a.distSq - b.distSq);
            routeIds = segment.map(p => p.id);
        }
        routes.push(routeIds);
    }

    // Output the resulting routes as a single JSON document.
    process.stdout.write(JSON.stringify({ routes }));
}

/**
 * Implements the Nearest Neighbor heuristic for a given set of points starting from a fixed location.
 * @param {Array} points - Array of package objects.
 * @param {number} startX - X coordinate of the starting point (depot).
 * @param {number} startY - Y coordinate of the starting point (depot).
 * @returns {Array} - Ordered array of package IDs.
 */
function getNNRoute(points, startX, startY) {
    let currentX = startX;
    let currentY = startY;
    const n = points.length;
    const result = new Array(n);
    const visited = new Uint8Array(n);
    
    for (let i = 0; i < n; i++) {
        let bestIdx = -1;
        let bestDistSq = Infinity;
        
        for (let j = 0; j < n; j++) {
            if (visited[j]) continue;
            
            const dx = points[j].x - currentX;
            const dy = points[j].y - currentY;
            const dSq = dx * dx + dy * dy;
            
            if (dSq < bestDistSq) {
                bestDistSq = dSq;
                bestIdx = j;
            }
        }
        
        visited[bestIdx] = 1;
        const p = points[bestIdx];
        result[i] = p.id;
        currentX = p.x;
        currentY = p.y;
    }
    
    return result;
}

// Execute the solver
solve();
```
