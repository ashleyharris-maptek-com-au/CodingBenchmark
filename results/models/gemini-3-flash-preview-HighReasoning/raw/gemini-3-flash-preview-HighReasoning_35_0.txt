use std::io::{self, BufRead, Write};

fn solve() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    
    let mut input = String::new();
    if reader.read_to_string(&mut input).is_err() {
        return Ok(());
    }
    let mut tokens = input.split_whitespace();

    let num_vertices: usize = match tokens.next() {
        Some(t) => t.parse().unwrap_or(0),
        None => return Ok(()),
    };
    let num_edges: usize = match tokens.next() {
        Some(t) => t.parse().unwrap_or(0),
        None => return Ok(()),
    };

    let mut adj = vec![Vec::new(); num_vertices];
    for _ in 0..num_edges {
        let u: usize = match tokens.next() {
            Some(t) => t.parse().unwrap_or(num_vertices),
            None => break,
        };
        let v: usize = match tokens.next() {
            Some(t) => t.parse().unwrap_or(num_vertices),
            None => break,
        };
        if u < num_vertices && v < num_vertices {
            if u != v {
                adj[u].push(v);
                adj[v].push(u);
            }
        }
    }

    // De-duplicate neighbors
    for i in 0..num_vertices {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    let mut covered = vec![false; num_vertices];
    let mut coverage_count = vec![0; num_vertices];
    let mut max_coverage = 0;
    for i in 0..num_vertices {
        coverage_count[i] = adj[i].len() + 1;
        if coverage_count[i] > max_coverage {
            max_coverage = coverage_count[i];
        }
    }

    let mut buckets = vec![Vec::new(); max_coverage + 1];
    let mut pos_in_bucket = vec![0; num_vertices];
    for i in 0..num_vertices {
        pos_in_bucket[i] = buckets[coverage_count[i]].len();
        buckets[coverage_count[i]].push(i);
    }

    let mut dominating_set = Vec::new();
    let mut num_covered = 0;
    let mut current_max = max_coverage;

    while num_covered < num_vertices {
        while current_max > 0 && buckets[current_max].is_empty() {
            current_max -= 1;
        }
        if current_max == 0 {
            for i in 0..num_vertices {
                if !covered[i] {
                    dominating_set.push(i);
                    covered[i] = true;
                    num_covered += 1;
                }
            }
            break;
        }

        let best_v = buckets[current_max].pop().unwrap();
        if coverage_count[best_v] != current_max {
            continue;
        }
        
        coverage_count[best_v] = 0;
        dominating_set.push(best_v);

        let mut to_mark = Vec::new();
        if !covered[best_v] {
            covered[best_v] = true;
            num_covered += 1;
            to_mark.push(best_v);
        }
        for i in 0..adj[best_v].len() {
            let u = adj[best_v][i];
            if !covered[u] {
                covered[u] = true;
                num_covered += 1;
                to_mark.push(u);
            }
        }

        for u in to_mark {
            // Case w = u
            {
                let w = u;
                let old_count = coverage_count[w];
                if old_count > 0 {
                    let pos = pos_in_bucket[w];
                    if pos < buckets[old_count].len() && buckets[old_count][pos] == w {
                        let last_v = *buckets[old_count].last().unwrap();
                        buckets[old_count][pos] = last_v;
                        pos_in_bucket[last_v] = pos;
                        buckets[old_count].pop();
                        let new_count = old_count - 1;
                        coverage_count[w] = new_count;
                        if new_count > 0 {
                            pos_in_bucket[w] = buckets[new_count].len();
                            buckets[new_count].push(w);
                        }
                    }
                }
            }
            // Case w in adj[u]
            for i in 0..adj[u].len() {
                let w = adj[u][i];
                let old_count = coverage_count[w];
                if old_count > 0 {
                    let pos = pos_in_bucket[w];
                    if pos < buckets[old_count].len() && buckets[old_count][pos] == w {
                        let last_v = *buckets[old_count].last().unwrap();
                        buckets[old_count][pos] = last_v;
                        pos_in_bucket[last_v] = pos;
                        buckets[old_count].pop();
                        let new_count = old_count - 1;
                        coverage_count[w] = new_count;
                        if new_count > 0 {
                            pos_in_bucket[w] = buckets[new_count].len();
                            buckets[new_count].push(w);
                        }
                    }
                }
            }
        }
    }

    // Redundant removal
    let mut cover_count = vec![0u32; num_vertices];
    for &v in &dominating_set {
        cover_count[v] += 1;
        for &u in &adj[v] {
            cover_count[u] += 1;
        }
    }

    let mut final_set = Vec::new();
    // Try removing in reverse order of addition
    for &v in dominating_set.iter().rev() {
        let mut can_remove = true;
        if cover_count[v] <= 1 {
            can_remove = false;
        } else {
            for &u in &adj[v] {
                if cover_count[u] <= 1 {
                    can_remove = false;
                    break;
                }
            }
        }

        if can_remove {
            cover_count[v] -= 1;
            for &u in &adj[v] {
                cover_count[u] -= 1;
            }
        } else {
            final_set.push(v);
        }
    }

    let stdout = io::stdout();
    let mut writer = io::BufWriter::new(stdout.lock());
    writeln!(writer, "{}", final_set.len())?;
    for (i, &v) in final_set.iter().enumerate() {
        if i > 0 {
            write!(writer, " ")?;
        }
        write!(writer, "{}", v)?;
    }
    writeln!(writer)?;
    Ok(())
}

fn main() {
    let _ = solve();
}
