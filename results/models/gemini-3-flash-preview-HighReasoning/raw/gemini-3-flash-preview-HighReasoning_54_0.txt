import math

# Persistence across calls within a scenario or go-around attempt
_ap_state = {
    'initialized': False,
    'integrated_loc': 0.0,
    'integrated_gs': 0.0,
    'integrated_ias': 0.0,
    'field_alt': None,
    'go_around': False,
    'trim': 0.0,
    'last_gs': 0.0,
    'last_loc': 0.0,
}

def autopilot_step(state: dict, dt: float) -> dict:
    global _ap_state
    
    # Initialize state on the first call
    if not _ap_state.get('initialized'):
        _ap_state['initialized'] = True
        _ap_state['integrated_loc'] = 0.0
        _ap_state['integrated_gs'] = 0.0
        _ap_state['integrated_ias'] = 0.0
        _ap_state['field_alt'] = None
        _ap_state['go_around'] = False
        _ap_state['trim'] = 0.0
        _ap_state['last_gs'] = 0.0
        _ap_state['last_loc'] = 0.0

    def get_median(keys, valid_keys=None):
        vals = []
        for i, k in enumerate(keys):
            if valid_keys is None or state.get(valid_keys[i], True):
                v = state.get(k)
                if v is not None:
                    vals.append(v)
        if not vals: return None
        vals.sort()
        return vals[len(vals)//2]

    # --- Sensor Fusion ---
    pitch = get_median(['irs_1_pitch_deg', 'irs_2_pitch_deg', 'irs_3_pitch_deg'], 
                       ['irs_1_valid', 'irs_2_valid', 'irs_3_valid']) or 0.0
    roll = get_median(['irs_1_roll_deg', 'irs_2_roll_deg', 'irs_3_roll_deg'], 
                      ['irs_1_valid', 'irs_2_valid', 'irs_3_valid']) or 0.0
    vs = get_median(['irs_1_vs_fpm', 'irs_2_vs_fpm', 'irs_3_vs_fpm'], 
                    ['irs_1_valid', 'irs_2_valid', 'irs_3_valid']) or 0.0
    heading = get_median(['irs_1_heading_deg', 'irs_2_heading_deg', 'irs_3_heading_deg'], 
                         ['irs_1_valid', 'irs_2_valid', 'irs_3_valid']) or 0.0
    
    gs_dots = get_median(['glideslope_1_dots', 'glideslope_2_dots', 'glideslope_3_dots'], 
                         ['glideslope_1_valid', 'glideslope_2_valid', 'glideslope_3_valid'])
    loc_dots = get_median(['localizer_1_dots', 'localizer_2_dots', 'localizer_3_dots'], 
                          ['localizer_1_valid', 'localizer_2_valid', 'localizer_3_valid'])
    
    # Hold last valid ILS signal to handle flickering
    if gs_dots is not None: _ap_state['last_gs'] = gs_dots
    else: gs_dots = _ap_state['last_gs']
    if loc_dots is not None: _ap_state['last_loc'] = loc_dots
    else: loc_dots = _ap_state['last_loc']
    
    ias = get_median(['ias_1_kt', 'ias_2_kt', 'ias_3_kt'], 
                     ['ias_1_valid', 'ias_2_valid', 'ias_3_valid']) or 140.0
    
    radio_alt = state.get('radio_alt_ft')
    if isinstance(radio_alt, list): radio_alt = sum(radio_alt)/len(radio_alt)
    
    baro_alt = get_median(['baro_alt_1_ft', 'baro_alt_2_ft', 'baro_alt_3_ft'], 
                          ['baro_alt_1_valid', 'baro_alt_2_valid', 'baro_alt_3_valid']) or 2000.0
    
    # Airfield elevation calibration
    if _ap_state['field_alt'] is None and radio_alt is not None and radio_alt > 100:
        _ap_state['field_alt'] = baro_alt - radio_alt
    
    # Radio altimeter failure fallback
    if radio_alt is None or radio_alt < -50:
        if _ap_state['field_alt'] is not None: radio_alt = baro_alt - _ap_state['field_alt']
        else: radio_alt = 1000.0

    target_ias = state.get('target_ias_kt') or state.get('target_vref_kt') or 140.0
    target_vref = state.get('target_vref_kt') or 137.0
    target_hdg = state.get('target_hdg_deg') or heading
    
    on_ground = state.get('on_ground', False) or state.get('weight_on_wheels', False) or (radio_alt < 5 and vs > -250)
    
    # --- Phase Management ---
    if _ap_state['go_around']:
        phase = 'goaround'
    elif on_ground:
        phase = 'rollout'
    elif radio_alt < 50:
        phase = 'flare'
    else:
        phase = 'approach'
        
    # --- Stability Check ---
    if phase == 'approach' and radio_alt < 1000:
        unstable = False
        if abs(gs_dots) > 2.2: unstable = True
        if abs(loc_dots) > 2.2: unstable = True
        if vs < -1600: unstable = True
        if ias < target_vref - 10: unstable = True
        if state.get('missed_approach_required'): unstable = True
        if unstable:
            _ap_state['go_around'] = True
            phase = 'goaround'

    res = {}
    def clamp(v, low, high): return max(low, min(v, high))

    if phase == 'goaround':
        # TOGA procedure
        res.update({'throttle_1': 1.0, 'throttle_2': 1.0, 'gear': 0.0, 'flaps': 15.0, 'speedbrake': 0.0, 'spoiler': 0.0})
        # Rotate to 15 degrees
        target_pitch = 15.0
        res['elevator'] = clamp((target_pitch - pitch) * 0.1, -1.0, 1.0)
        res['aileron'] = clamp(-roll * 0.05, -0.5, 0.5)
        
    elif phase == 'rollout':
        # Slow down and stay on centerline
        res.update({'throttle_1': 0.0, 'throttle_2': 0.0, 'reverse_1': 1.0, 'reverse_2': 1.0, 
                    'spoiler': 1.0, 'autobrake': 1.0, 'speedbrake': 1.0, 'gear': 1.0})
        hdg_err = (target_hdg - heading + 180) % 360 - 180
        res['rudder'] = clamp(hdg_err * 0.1, -1.0, 1.0)
        res['aileron'] = clamp(-roll * 0.1, -0.5, 0.5)
        res['elevator'] = -0.1
        
    elif phase == 'flare':
        # Transition to soft touchdown
        res.update({'throttle_1': 0.0, 'throttle_2': 0.0, 'gear': 1.0, 'flaps': 40.0})
        # Gradually reduce sink rate
        target_vs_flare = clamp(-radio_alt * 2.0 - 50, -500, -100)
        res['elevator'] = clamp((target_vs_flare - vs) * 0.008, -1.0, 1.0)
        res['aileron'] = clamp(-roll * 0.1, -0.8, 0.8)
        hdg_err = (target_hdg - heading + 180) % 360 - 180
        res['rudder'] = clamp(hdg_err * 0.15, -1.0, 1.0)
        
    else: # Approach
        # Pitch / Glideslope
        target_vs_gs = clamp(-gs_dots * 400 - 700, -1500, -300)
        res['elevator'] = clamp((target_vs_gs - vs) * 0.004, -1.0, 1.0)
        
        # Roll / Localizer
        _ap_state['integrated_loc'] += loc_dots * dt
        _ap_state['integrated_loc'] = clamp(_ap_state['integrated_loc'], -5, 5)
        target_bank = clamp(-loc_dots * 18 - _ap_state['integrated_loc'] * 1.5, -25, 25)
        res['aileron'] = clamp((target_bank - roll) * 0.08, -1.0, 1.0)
        
        # Power / Speed
        e1, e2 = state.get('engine_1_running', True), state.get('engine_2_running', True)
        if e1 or e2:
            speed_err = target_ias - ias
            _ap_state['integrated_ias'] += speed_err * dt
            _ap_state['integrated_ias'] = clamp(_ap_state['integrated_ias'], -10, 20)
            thrust = speed_err * 0.2 + _ap_state['integrated_ias'] * 0.1 + 0.45
            res['throttle_1'] = clamp(thrust, 0, 1) if e1 else 0.0
            res['throttle_2'] = clamp(thrust, 0, 1) if e2 else 0.0
            if not e1: res['rudder'] = clamp(thrust * 0.4, -1, 1)
            if not e2: res['rudder'] = clamp(-thrust * 0.4, -1, 1)
        else: # Glide
            res['elevator'] = clamp((target_vref - ias) * 0.15, -1.0, 1.0)
            res['speedbrake'] = 1.0 if gs_dots > 0.5 else 0.0
            
        res.update({'gear': 1.0, 'flaps': 15.0 if radio_alt > 1500 else (30.0 if radio_alt > 1000 else 40.0)})

    if 'elevator' in res:
        _ap_state['trim'] += res['elevator'] * 0.3 * dt
        _ap_state['trim'] = clamp(_ap_state['trim'], -1.0, 1.0)
        res['elevator_trim'] = _ap_state['trim']
        
    return res
