The Quadratic Assignment Problem (QAP) is solved using a heuristic approach consisting of a greedy initialization followed by a 2-opt local search (hill climbing). This approach is well-suited for the problem's NP-hard nature and the constraint of handling $N$ up to 2000 within a 5-minute time limit.

### Key Features of the Implementation:
1.  **Fast Input Parsing:** Uses a custom `Tokenizer` with a buffered `Stream` to efficiently read large input matrices from `stdin`.
2.  **Greedy Initialization:** Generates a starting permutation by assigning facilities with high total flow to locations with low total distance, which often provides a strong baseline.
3.  **Optimized Local Search:** Implements a 2-opt swap heuristic. The cost delta of a swap is calculated in $O(N)$ time, making a full sweep across all pairs $O(N^3)$.
4.  **Performance Optimizations:**
    *   **Transposed Matrices:** Maintains transposed copies of the flow and distance matrices to ensure row-major access patterns, avoiding slow column accesses in C#.
    *   **Efficient Delta Calculation:** Uses a simplified delta formula for general (non-symmetric) matrices.
5.  **Robustness:** Uses `long` for cost calculations to prevent overflows on large instances and includes a `Stopwatch` to respect the 5-minute execution limit.
6.  **Metaheuristic Exploration:** For smaller $N$, the solver performs multiple random restarts to escape local optima and improve the final solution.

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Text;

public class QAPSolver
{
    private static int N;
    private static int[][] flow;
    private static int[][] dist;
    private static int[][] flowT;
    private static int[][] distT;

    public static void Main(string[] args)
    {
        var stopwatch = Stopwatch.StartNew();
        if (!ReadInput()) return;

        int[] bestP = null;
        long bestCost = long.MaxValue;

        bool firstRun = true;
        while (stopwatch.ElapsedMilliseconds < 280000) // 4 min 40 sec limit
        {
            int[] p;
            if (firstRun)
            {
                p = InitialSolution();
                firstRun = false;
            }
            else
            {
                p = Enumerable.Range(0, N).ToArray();
                Shuffle(p);
            }

            long currentCost = CalculateCost(p);
            
            int[] indices = Enumerable.Range(0, N).ToArray();
            bool improved = true;
            while (improved && stopwatch.ElapsedMilliseconds < 280000)
            {
                improved = false;
                Shuffle(indices);

                for (int i = 0; i < N && stopwatch.ElapsedMilliseconds < 280000; i++)
                {
                    int u = indices[i];
                    int[] rowFu = flow[u];
                    int[] rowFuT = flowT[u];

                    for (int j = i + 1; j < N && stopwatch.ElapsedMilliseconds < 280000; j++)
                    {
                        int v = indices[j];
                        long delta = CalculateDelta(p, u, v, rowFu, flow[v], rowFuT, flowT[v]);
                        if (delta < 0)
                        {
                            currentCost += delta;
                            int tmp = p[u];
                            p[u] = p[v];
                            p[v] = tmp;
                            improved = true;
                        }
                    }
                }
            }

            if (currentCost < bestCost)
            {
                bestCost = currentCost;
                bestP = (int[])p.Clone();
            }
            
            if (N > 500) break; // For large N, one thorough run is preferred
        }

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestP));
    }

    private static bool ReadInput()
    {
        var reader = new Tokenizer(Console.OpenStandardInput());
        string nStr = reader.Next();
        if (nStr == null) return false;
        if (!int.TryParse(nStr, out N)) return false;

        flow = new int[N][];
        flowT = new int[N][];
        for (int i = 0; i < N; i++) { flow[i] = new int[N]; flowT[i] = new int[N]; }
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                flow[i][j] = int.Parse(reader.Next());
            }
        }
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                flowT[j][i] = flow[i][j];

        dist = new int[N][];
        distT = new int[N][];
        for (int i = 0; i < N; i++) { dist[i] = new int[N]; distT[i] = new int[N]; }
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                dist[i][j] = int.Parse(reader.Next());
            }
        }
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                distT[j][i] = dist[i][j];

        return true;
    }

    private static int[] InitialSolution()
    {
        long[] flowSum = new long[N];
        long[] distSum = new long[N];

        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                flowSum[i] += flow[i][j] + flow[j][i];
                distSum[i] += dist[i][j] + dist[j][i];
            }
        }

        var facilities = Enumerable.Range(0, N).OrderByDescending(i => flowSum[i]).ToArray();
        var locations = Enumerable.Range(0, N).OrderBy(i => distSum[i]).ToArray();

        int[] p = new int[N];
        for (int i = 0; i < N; i++)
        {
            p[facilities[i]] = locations[i];
        }

        return p;
    }

    private static long CalculateCost(int[] p)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            int pi = p[i];
            int[] rowFi = flow[i];
            for (int j = 0; j < N; j++)
            {
                cost += (long)rowFi[j] * dist[pi][p[j]];
            }
        }
        return cost;
    }

    private static long CalculateDelta(int[] p, int i, int j, int[] rowFi, int[] rowFj, int[] rowFiT, int[] rowFjT)
    {
        int pi = p[i];
        int pj = p[j];
        int[] rowDpi = dist[pi];
        int[] rowDpj = dist[pj];
        int[] rowDpiT = distT[pi];
        int[] rowDpjT = distT[pj];
        
        long delta = 0;
        for (int k = 0; k < N; k++)
        {
            if (k == i || k == j) continue;
            int pk = p[k];
            // Delta contribution for asymmetric flow/distance matrices
            delta += (long)(rowFi[k] - rowFj[k]) * (rowDpj[pk] - rowDpi[pk]);
            delta += (long)(rowFiT[k] - rowFjT[k]) * (rowDpjT[pk] - rowDpiT[pk]);
        }
        
        delta += (long)flow[i][j] * (dist[pj][pi] - dist[pi][pj]);
        delta += (long)flow[j][i] * (dist[pi][pj] - dist[pj][pi]);
        delta += (long)flow[i][i] * (dist[pj][pj] - dist[pi][pi]);
        delta += (long)flow[j][j] * (dist[pi][pi] - dist[pj][pj]);

        return delta;
    }

    private static Random rand = new Random();
    private static void Shuffle(int[] array)
    {
        for (int i = array.Length - 1; i > 0; i--)
        {
            int j = rand.Next(i + 1);
            int tmp = array[i];
            array[i] = array[j];
            array[j] = tmp;
        }
    }

    private class Tokenizer
    {
        private Stream stream;
        private byte[] buffer = new byte[65536];
        private int pos, limit;

        public Tokenizer(Stream stream) { this.stream = stream; }

        public string Next()
        {
            var sb = new StringBuilder();
            int b;
            while ((b = ReadByte()) != -1 && char.IsWhiteSpace((char)b)) ;
            if (b == -1) return null;
            sb.Append((char)b);
            while ((b = ReadByte()) != -1 && !char.IsWhiteSpace((char)b))
                sb.Append((char)b);
            return sb.ToString();
        }

        private int ReadByte()
        {
            if (pos >= limit)
            {
                pos = 0;
                limit = stream.Read(buffer, 0, buffer.Length);
                if (limit <= 0) return -1;
            }
            return buffer[pos++];
        }
    }
}
```
