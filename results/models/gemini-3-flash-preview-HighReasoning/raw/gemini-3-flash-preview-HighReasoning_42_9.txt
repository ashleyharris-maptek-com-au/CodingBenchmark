OpCapability Shader
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
       %uint = OpTypeInt 32 0
     %uint_4 = OpConstant %uint 4
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

  %arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %color_in = OpVariable %ptr_in_arr_v3 Input

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3

%float_0_025 = OpConstant %float 0.025

  %ptr_in_v4 = OpTypePointer Input %v4float
  %ptr_in_v3 = OpTypePointer Input %v3float
 %ptr_out_v4 = OpTypePointer Output %v4float

; Main
       %main = OpFunction %void None %func
      %label = OpLabel

; 1. Read all 4 positions
    %p_ptr0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %P0 = OpLoad %v4float %p_ptr0
    %p_ptr1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %P1 = OpLoad %v4float %p_ptr1
    %p_ptr2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %P2 = OpLoad %v4float %p_ptr2
    %p_ptr3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
        %P3 = OpLoad %v4float %p_ptr3

; 2. Compute tangent at start: t0 = normalize(gl_in[2].pos - gl_in[0].pos)
     %diff0 = OpFSub %v4float %P2 %P0
        %t0 = OpExtInst %v4float %1 Normalize %diff0

; 3. Compute tangent at end: t1 = normalize(gl_in[3].pos - gl_in[1].pos)
     %diff1 = OpFSub %v4float %P3 %P1
        %t1 = OpExtInst %v4float %1 Normalize %diff1

; 4. Compute perpendiculars: perp0 = vec2(-t0.y, t0.x), perp1 = vec2(-t1.y, t1.x)
      %t0_y = OpCompositeExtract %float %t0 1
      %t0_x = OpCompositeExtract %float %t0 0
  %neg_t0_y = OpFNegate %float %t0_y
     %perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

      %t1_y = OpCompositeExtract %float %t1 1
      %t1_x = OpCompositeExtract %float %t1 0
  %neg_t1_y = OpFNegate %float %t1_y
     %perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; 5. Thickness = 0.025
   %offset0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
   %offset1 = OpVectorTimesScalar %v2float %perp1 %float_0_025

; Output gl_Position pointer
   %o_pos_p = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Load colors from color[1] and color[2]
    %c_ptr1 = OpAccessChain %ptr_in_v3 %color_in %int_1
        %C1 = OpLoad %v3float %c_ptr1
    %c_ptr2 = OpAccessChain %ptr_in_v3 %color_in %int_2
        %C2 = OpLoad %v3float %c_ptr2

; 6. Emit 4 vertices as triangle strip
; Vertex 1: start + perp0*thickness, color[1]
     %P1_xy = OpVectorShuffle %v2float %P1 %P1 0 1
    %P1a_xy = OpFAdd %v2float %P1_xy %offset0
       %V1a = OpVectorShuffle %v4float %P1 %P1a_xy 4 5 2 3
               OpStore %outColor %C1
               OpStore %o_pos_p %V1a
               OpEmitVertex

; Vertex 2: start - perp0*thickness, color[1]
    %P1b_xy = OpFSub %v2float %P1_xy %offset0
       %V1b = OpVectorShuffle %v4float %P1 %P1b_xy 4 5 2 3
               OpStore %outColor %C1
               OpStore %o_pos_p %V1b
               OpEmitVertex

; Vertex 3: end + perp1*thickness, color[2]
     %P2_xy = OpVectorShuffle %v2float %P2 %P2 0 1
    %P2a_xy = OpFAdd %v2float %P2_xy %offset1
       %V2a = OpVectorShuffle %v4float %P2 %P2a_xy 4 5 2 3
               OpStore %outColor %C2
               OpStore %o_pos_p %V2a
               OpEmitVertex

; Vertex 4: end - perp1*thickness, color[2]
    %P2b_xy = OpFSub %v2float %P2_xy %offset1
       %V2b = OpVectorShuffle %v4float %P2 %P2b_xy 4 5 2 3
               OpStore %outColor %C2
               OpStore %o_pos_p %V2b
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
