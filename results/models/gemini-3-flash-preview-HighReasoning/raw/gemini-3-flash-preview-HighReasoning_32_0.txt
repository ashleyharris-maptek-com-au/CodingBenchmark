To solve the Steiner Tree problem, I have implemented the **Dreyfus-Wagner Dynamic Programming** algorithm. This approach provides an exact solution for the minimum-weight tree connecting a set of terminals.

### Algorithm Overview
1.  **State Definition**: The core DP state is `dp[mask][v]`, which represents the minimum weight to connect a subset of terminals (represented by the bitmask `mask`) with vertex `v` as the root or a connecting node.
2.  **Transitions**:
    *   **Split Step**: For a fixed vertex `v`, we can split the terminal set `mask` into two disjoint subsets `sub` and `mask \ sub`. The cost is `dp[sub][v] + dp[mask \ sub][v]`. This is done for all subsets of the current mask.
    *   **Dijkstra Step**: For a fixed terminal set `mask`, we can improve the connection cost to each vertex `v` by moving from an adjacent vertex `u`. This is equivalent to running a single-source shortest path (Dijkstra) algorithm for each terminal subset.
3.  **Reconstruction**: To output the actual edges, the program stores backtracking information (`back[mask][v]`) during the DP phase. A recursive function then traverses these pointers from the optimal full-mask state to collect all participating edges.
4.  **Complexity**: The time complexity is $O(3^k \cdot n + 2^k \cdot (m \log n))$, where $k$ is the number of terminals, $n$ is the number of vertices, and $m$ is the number of edges. This is highly efficient for small terminal sets (typically up to $k \approx 20$ given the memory and time limits).

### C++ Code Implementation

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <bit>

using namespace std;

/**
 * Steiner Tree Problem Solver (NP-Hard)
 * Algorithm: Dreyfus-Wagner Dynamic Programming
 * Complexity: O(3^k * n + 2^k * (m log n))
 */

typedef long long ll;
const ll INF = 1e16; // Large value to avoid overflow during additions

struct Edge {
    int to;
    ll weight;
};

// Backtracking information to reconstruct the tree edges
struct Back {
    int mask;
    int v;
    bool is_split;
};

void solve() {
    int n, m, k;
    if (!(cin >> n >> m >> k)) return;

    // Graph represented as an adjacency list
    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        ll w;
        if (!(cin >> u >> v >> w)) break;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            adj[u].push_back({v, w});
            adj[v].push_back({u, w});
        }
    }

    // Terminals that need to be connected
    vector<int> raw_terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> raw_terminals[i];
    }
    
    // De-duplicate and validate terminals
    sort(raw_terminals.begin(), raw_terminals.end());
    raw_terminals.erase(unique(raw_terminals.begin(), raw_terminals.end()), raw_terminals.end());
    
    vector<int> terminals;
    for (int t : raw_terminals) {
        if (t >= 0 && t < n) {
            terminals.push_back(t);
        }
    }
    k = (int)terminals.size();

    // Base cases for k
    if (k == 0) {
        cout << "0 0" << endl;
        return;
    }
    if (k == 1) {
        cout << "0 0" << endl;
        return;
    }
    
    // Safety check for mask size (k=20 is roughly the limit for 32GB RAM)
    if (k >= 25) return;

    int num_masks = 1 << k;
    // dp[mask][v] stores min weight to connect 'mask' terminals with 'v' as root
    vector<vector<ll>> dp(num_masks, vector<ll>(n, INF));
    // back[mask][v] stores backtracking info for reconstruction
    vector<vector<Back>> back(num_masks, vector<Back>(n, {0, 0, false}));

    // Initialization: each terminal is a Steiner tree of weight 0
    for (int i = 0; i < k; ++i) {
        dp[1 << i][terminals[i]] = 0;
    }

    // Main Dreyfus-Wagner DP loop
    for (int mask = 1; mask < num_masks; ++mask) {
        unsigned int umask = (unsigned int)mask;
        
        // Transition 1: Vertex-based split
        if (std::popcount(umask) >= 2) {
            for (int v = 0; v < n; ++v) {
                for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
                    if (sub < (mask ^ sub)) {
                        ll combined = dp[sub][v] + dp[mask ^ sub][v];
                        if (combined < dp[mask][v]) {
                            dp[mask][v] = combined;
                            back[mask][v] = {sub, v, true};
                        }
                    }
                }
            }
        }

        // Transition 2: Edge-based extension (Dijkstra step)
        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] < INF) {
                pq.push({dp[mask][v], v});
            }
        }

        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();

            if (d > dp[mask][u]) continue;

            for (const auto& edge : adj[u]) {
                if (dp[mask][u] + edge.weight < dp[mask][edge.to]) {
                    dp[mask][edge.to] = dp[mask][u] + edge.weight;
                    back[mask][edge.to] = {mask, u, false};
                    pq.push({dp[mask][edge.to], edge.to});
                }
            }
        }
    }

    // Find the vertex yielding the minimum weight Steiner Tree for the full mask
    ll min_weight = INF;
    int best_v = -1;
    for (int v = 0; v < n; ++v) {
        if (dp[num_masks - 1][v] < min_weight) {
            min_weight = dp[num_masks - 1][v];
            best_v = v;
        }
    }

    if (best_v == -1 || min_weight >= INF) return;

    // Reconstruction of the optimal tree edges
    vector<pair<int, int>> tree_edges;
    vector<vector<bool>> visited_state(num_masks, vector<bool>(n, false));

    auto reconstruct = [&](auto self, int msk, int v) -> void {
        if (std::popcount((unsigned int)msk) == 1 && dp[msk][v] == 0) return;
        if (visited_state[msk][v]) return;
        visited_state[msk][v] = true;

        Back b = back[msk][v];
        if (b.is_split) {
            self(self, b.mask, v);
            self(self, msk ^ b.mask, v);
        } else {
            if (msk != 0 && b.mask == msk) {
                tree_edges.push_back({b.v, v});
                self(self, b.mask, b.v);
            }
        }
    };

    reconstruct(reconstruct, num_masks - 1, best_v);

    // Normalize and remove duplicate edges
    for (auto& e : tree_edges) {
        if (e.first > e.second) swap(e.first, e.second);
    }
    sort(tree_edges.begin(), tree_edges.end());
    tree_edges.erase(unique(tree_edges.begin(), tree_edges.end()), tree_edges.end());

    // Print result: weight, edge count, and edges
    cout << min_weight << " " << (int)tree_edges.size() << endl;
    for (const auto& e : tree_edges) {
        cout << e.first << " " << e.second << endl;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
```
