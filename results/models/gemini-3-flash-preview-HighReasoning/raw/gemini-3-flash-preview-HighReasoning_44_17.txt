#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 8x16 UV scale
    vec2 st = uv * vec2(8.0, 16.0);
    
    // Staggered rows: half-brick offset every other row
    if (mod(floor(st.y), 2.0) > 0.5) {
        st.x += 0.5;
    }
    
    // Local coordinates within the brick cell
    vec2 local = fract(st);
    
    // Mortar thickness 0.05 in X and 0.1 in Y
    // brickMask is 1.0 for brick, 0.0 for mortar
    float brickMask = step(0.05, local.x) * step(0.1, local.y);
    
    // Specified colors
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 baseColor = mix(mortarColor, brickColor, brickMask);
    
    // Lambertian shading: factor (0.4 + 0.6 * diffuse)
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float factor = 0.4 + 0.6 * diffuse;
    
    outColor = vec4(baseColor * factor, 1.0);
}
