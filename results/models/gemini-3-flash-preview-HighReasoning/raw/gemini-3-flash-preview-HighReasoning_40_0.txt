The following C++17 code solves the 0/1 Integer Linear Program (specifically, the Maximum Weight Independent Set problem) using a backtracking algorithm with pruning. It utilizes a greedy vertex ordering heuristic (weight descending) and suffix-sum-based pruning to efficiently find the true optimum.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

using namespace std;

/**
 * Problem: 0/1 Integer Linear Program (Maximum Weight Independent Set)
 * Maximize sum(c_i * x_i) subject to x_u + x_v <= 1 for each constraint pair (u, v).
 * 
 * Approach: Exact backtracking with pruning.
 * We order vertices by weight (descending) to find high-objective solutions early.
 * Pruning: If (current_weight + suffix_sum_of_remaining_positive_weights <= max_found), prune.
 */

struct Vertex {
    int original_id;
    long long weight;
    vector<int> future_neighbors; // Indices in the sorted order, only those > current idx
};

int N, M;
vector<Vertex> vertices;
vector<long long> suffix_max_weight;
vector<int> best_x_original;
long long max_objective = -1;

void solve(int idx, long long current_weight, vector<int>& current_x, vector<int>& neighbor_included_count) {
    if (idx == N) {
        if (current_weight > max_objective) {
            max_objective = current_weight;
            for (int i = 0; i < N; ++i) {
                best_x_original[vertices[i].original_id] = current_x[i];
            }
        }
        return;
    }

    // Pruning: if the best we can possibly do from here doesn't beat our current best, stop.
    if (current_weight + suffix_max_weight[idx] <= max_objective) {
        return;
    }

    // If weight is non-positive, including it can only decrease or maintain the objective
    // while making constraints harder to satisfy. We can safely skip inclusion.
    if (vertices[idx].weight <= 0) {
        current_x[idx] = 0;
        solve(idx + 1, current_weight, current_x, neighbor_included_count);
        return;
    }

    // Vertex idx MUST be 0 if any of its neighbors are already in the independent set.
    if (neighbor_included_count[idx] > 0) {
        current_x[idx] = 0;
        solve(idx + 1, current_weight, current_x, neighbor_included_count);
        return;
    }

    // Branch 1: Include vertex idx in the independent set.
    // Try inclusion first to find a large objective value quickly.
    current_x[idx] = 1;
    for (int nb : vertices[idx].future_neighbors) {
        neighbor_included_count[nb]++;
    }
    solve(idx + 1, current_weight + vertices[idx].weight, current_x, neighbor_included_count);
    
    // Backtrack neighbor counts.
    for (int nb : vertices[idx].future_neighbors) {
        neighbor_included_count[nb]--;
    }

    // Branch 2: Exclude vertex idx from the independent set.
    current_x[idx] = 0;
    solve(idx + 1, current_weight, current_x, neighbor_included_count);
}

int main() {
    // Standard I/O optimization
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> N >> M)) return 0;

    vector<long long> c(N);
    for (int i = 0; i < N; ++i) cin >> c[i];

    vector<vector<int>> adj_orig(N);
    for (int i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v;
        if (u >= 0 && u < N && v >= 0 && v < N) {
            adj_orig[u].push_back(v);
            adj_orig[v].push_back(u);
        }
    }

    // Heuristic: Process vertices with higher weights first.
    vector<int> p(N);
    iota(p.begin(), p.end(), 0);
    sort(p.begin(), p.end(), [&](int a, int b) {
        if (c[a] != c[b]) return c[a] > c[b];
        return a < b;
    });

    vector<int> old_to_new(N);
    vertices.resize(N);
    for (int i = 0; i < N; ++i) {
        old_to_new[p[i]] = i;
        vertices[i].original_id = p[i];
        vertices[i].weight = c[p[i]];
    }

    for (int i = 0; i < N; ++i) {
        for (int neighbor_orig : adj_orig[vertices[i].original_id]) {
            int nb_new = old_to_new[neighbor_orig];
            if (nb_new > i) {
                vertices[i].future_neighbors.push_back(nb_new);
            }
        }
    }

    // Precalculate suffix sums of positive weights for pruning.
    suffix_max_weight.assign(N + 1, 0);
    for (int i = N - 1; i >= 0; --i) {
        suffix_max_weight[i] = suffix_max_weight[i+1] + max(0LL, vertices[i].weight);
    }

    vector<int> current_x(N, 0);
    vector<int> neighbor_included_count(N, 0);
    best_x_original.assign(N, 0);

    solve(0, 0, current_x, neighbor_included_count);

    if (max_objective == -1) max_objective = 0;

    // Output formatted results
    cout << max_objective << "\n";
    for (int i = 0; i < N; ++i) {
        cout << best_x_original[i] << (i == N - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
```
