OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorations
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
    %fn_type = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
     %uint_6 = OpConstant %uint 6
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_4 = OpConstant %int 4

%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

 %arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

 %ptr_in_v4 = OpTypePointer Input %v4float
 %ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_in_v3 = OpTypePointer Input %v3float

; Constants
    %float_0 = OpConstant %float 0
  %float_0_2 = OpConstant %float 0.2
  %float_0_8 = OpConstant %float 0.8
    %float_1 = OpConstant %float 1
    %float_3 = OpConstant %float 3
 %vec3_1_1_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Function
       %main = OpFunction %void None %fn_type
      %entry = OpLabel

; Load p0, p2, p4 from worldPos
     %ptr_p0 = OpAccessChain %ptr_in_v3 %worldPos %int_0
         %p0 = OpLoad %v3float %ptr_p0
     %ptr_p2 = OpAccessChain %ptr_in_v3 %worldPos %int_2
         %p2 = OpLoad %v3float %ptr_p2
     %ptr_p4 = OpAccessChain %ptr_in_v3 %worldPos %int_4
         %p4 = OpLoad %v3float %ptr_p4

; v1 = p2 - p0
         %v1 = OpFSub %v3float %p2 %p0
; v2 = p4 - p0
         %v2 = OpFSub %v3float %p4 %p0

; cross product (a.y*b.z-a.z*b.y, a.z*b.x-a.x*b.z, a.x*b.y-a.y*b.x)
        %v1x = OpCompositeExtract %float %v1 0
        %v1y = OpCompositeExtract %float %v1 1
        %v1z = OpCompositeExtract %float %v1 2
        %v2x = OpCompositeExtract %float %v2 0
        %v2y = OpCompositeExtract %float %v2 1
        %v2z = OpCompositeExtract %float %v2 2

       %t0_0 = OpFMul %float %v1y %v2z
       %t0_1 = OpFMul %float %v1z %v2y
         %nx = OpFSub %float %t0_0 %t0_1

       %t1_0 = OpFMul %float %v1z %v2x
       %t1_1 = OpFMul %float %v1x %v2z
         %ny = OpFSub %float %t1_0 %t1_1

       %t2_0 = OpFMul %float %v1x %v2y
       %t2_1 = OpFMul %float %v1y %v2x
         %nz = OpFSub %float %t2_0 %t2_1

     %N_raw = OpCompositeConstruct %v3float %nx %ny %nz
          %N = OpExtInst %v3float %1 Normalize %N_raw

; L = normalize(vec3(1, 1, 1))
          %L = OpExtInst %v3float %1 Normalize %vec3_1_1_1

; diffuse = max(dot(N, L), 0.0)
     %dot_NL = OpDot %float %N %L
    %diffuse = OpExtInst %float %1 FMax %dot_NL %float_0

; Base color = average of color[0], color[2], color[4]
     %ptr_c0 = OpAccessChain %ptr_in_v3 %color %int_0
         %c0 = OpLoad %v3float %ptr_c0
     %ptr_c2 = OpAccessChain %ptr_in_v3 %color %int_2
         %c2 = OpLoad %v3float %ptr_c2
     %ptr_c4 = OpAccessChain %ptr_in_v3 %color %int_4
         %c4 = OpLoad %v3float %ptr_c4

     %sum_c1 = OpFAdd %v3float %c0 %c2
      %sum_c = OpFAdd %v3float %sum_c1 %c4
      %inv_3 = OpFDiv %float %float_1 %float_3
 %base_color = OpVectorTimesScalar %v3float %sum_c %inv_3

; Final color = base * (0.2 + 0.8 * diffuse)
     %t_diff = OpFMul %float %float_0_8 %diffuse
     %factor = OpFAdd %float %float_0_2 %t_diff
%final_color = OpVectorTimesScalar %v3float %base_color %factor

; Output gl_out.gl_Position and outColor, then emit
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Vertex 0
   %ptr_pos0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
       %pos0 = OpLoad %v4float %ptr_pos0
               OpStore %ptr_out_pos %pos0
               OpStore %outColor %final_color
               OpEmitVertex

; Vertex 2
   %ptr_pos2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
       %pos2 = OpLoad %v4float %ptr_pos2
               OpStore %ptr_out_pos %pos2
               OpStore %outColor %final_color
               OpEmitVertex

; Vertex 4
   %ptr_pos4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
       %pos4 = OpLoad %v4float %ptr_pos4
               OpStore %ptr_out_pos %pos4
               OpStore %outColor %final_color
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
