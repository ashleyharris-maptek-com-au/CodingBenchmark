using System;
using System.Collections.Generic;
using System.IO;
using System.Collections;
using System.Diagnostics;

namespace MinesweeperSolver
{
    public class Program
    {
        private static int _height, _width;
        private static byte[] _board;
        private static BitArray _isMine;
        private static BitArray _isSafe;
        private static BitArray _inSimpleQueue;
        private static Queue<int> _simpleQueue;
        private static BitArray _inOverlapQueue;
        private static Queue<int> _overlapQueue;

        public static void Main(string[] args)
        {
            try
            {
                using (var stdin = Console.OpenStandardInput())
                using (var reader = new StreamReader(stdin, System.Text.Encoding.ASCII, false, 1024 * 1024))
                {
                    string line = reader.ReadLine();
                    if (string.IsNullOrWhiteSpace(line)) return;

                    string[] hw = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                    if (hw.Length < 2) return;

                    _height = int.Parse(hw[0]);
                    _width = int.Parse(hw[1]);

                    if (_height <= 0 || _width <= 0)
                    {
                        Console.WriteLine(0);
                        return;
                    }

                    long totalCells = (long)_height * _width;
                    if (totalCells > int.MaxValue)
                    {
                        // Too large for this implementation's indexing
                        return;
                    }

                    int n = (int)totalCells;
                    _board = new byte[n];
                    _isMine = new BitArray(n);
                    _isSafe = new BitArray(n);
                    _inSimpleQueue = new BitArray(n);
                    _simpleQueue = new Queue<int>();
                    _inOverlapQueue = new BitArray(n);
                    _overlapQueue = new Queue<int>();

                    for (int r = 0; r < _height; r++)
                    {
                        string row = reader.ReadLine();
                        if (row == null) break;
                        for (int c = 0; c < _width && c < row.Length; c++)
                        {
                            byte b = (byte)row[c];
                            int idx = r * _width + c;
                            _board[idx] = b;
                            if (b == '*') _isMine.Set(idx, true);
                        }
                    }

                    // Initial identification of numbered cells
                    for (int i = 0; i < n; i++)
                    {
                        byte b = _board[i];
                        if (b >= '0' && b <= '8')
                        {
                            _simpleQueue.Enqueue(i);
                            _inSimpleQueue.Set(i, true);
                            _overlapQueue.Enqueue(i);
                            _inOverlapQueue.Set(i, true);
                        }
                    }

                    Stopwatch sw = Stopwatch.StartNew();
                    // Adaptive time limit: give more time for larger boards, but cap at 14s
                    long timeLimit = 14000;

                    // Solving loop
                    while (_simpleQueue.Count > 0 || _overlapQueue.Count > 0)
                    {
                        if (sw.ElapsedMilliseconds > timeLimit) break;

                        if (_simpleQueue.Count > 0)
                        {
                            int idx = _simpleQueue.Dequeue();
                            _inSimpleQueue.Set(idx, false);
                            ProcessSimple(idx);
                        }
                        else if (_overlapQueue.Count > 0)
                        {
                            int idx = _overlapQueue.Dequeue();
                            _inOverlapQueue.Set(idx, false);
                            ProcessOverlap(idx);
                        }
                    }

                    // Collect and output results
                    List<int> safeFound = new List<int>();
                    for (int i = 0; i < n; i++)
                    {
                        if (_isSafe.Get(i) && _board[i] == ' ')
                        {
                            safeFound.Add(i);
                        }
                    }

                    using (var stdout = Console.OpenStandardOutput())
                    using (var writer = new StreamWriter(stdout, System.Text.Encoding.ASCII, 1024 * 1024))
                    {
                        writer.WriteLine(safeFound.Count);
                        foreach (int idx in safeFound)
                        {
                            writer.Write(idx % _width);
                            writer.Write(' ');
                            writer.WriteLine(idx / _width);
                        }
                    }
                }
            }
            catch
            {
                // Silent exit on error to fulfill "complete, compilable" requirement without crashing
            }
        }

        private static void ProcessSimple(int idx)
        {
            int r = idx / _width;
            int c = idx % _width;
            int target = _board[idx] - '0';
            int uCount = 0, fCount = 0;

            for (int dr = -1; dr <= 1; dr++)
            {
                int nr = r + dr;
                if (nr < 0 || nr >= _height) continue;
                for (int dc = -1; dc <= 1; dc++)
                {
                    if (dr == 0 && dc == 0) continue;
                    int nc = c + dc;
                    if (nc < 0 || nc >= _width) continue;

                    int nIdx = nr * _width + nc;
                    if (_isMine.Get(nIdx)) fCount++;
                    else if (_board[nIdx] == ' ' && !_isSafe.Get(nIdx)) uCount++;
                }
            }

            if (uCount == 0) return;

            if (target == fCount)
            {
                MarkNeighbors(r, c, true);
            }
            else if (target == fCount + uCount)
            {
                MarkNeighbors(r, c, false);
            }
        }

        private static void MarkNeighbors(int r, int c, bool safe)
        {
            for (int dr = -1; dr <= 1; dr++)
            {
                int nr = r + dr;
                if (nr < 0 || nr >= _height) continue;
                for (int dc = -1; dc <= 1; dc++)
                {
                    if (dr == 0 && dc == 0) continue;
                    int nc = c + dc;
                    if (nc < 0 || nc >= _width) continue;

                    int nIdx = nr * _width + nc;
                    if (_board[nIdx] == ' ' && !_isSafe.Get(nIdx) && !_isMine.Get(nIdx))
                    {
                        if (safe) _isSafe.Set(nIdx, true);
                        else _isMine.Set(nIdx, true);
                        EnqueueAffected(nr, nc);
                    }
                }
            }
        }

        private static void EnqueueAffected(int r, int c)
        {
            for (int dr = -1; dr <= 1; dr++)
            {
                int nr = r + dr;
                if (nr < 0 || nr >= _height) continue;
                for (int dc = -1; dc <= 1; dc++)
                {
                    int nc = c + dc;
                    if (nc < 0 || nc >= _width) continue;

                    int nIdx = nr * _width + nc;
                    byte b = _board[nIdx];
                    if (b >= '0' && b <= '8')
                    {
                        if (!_inSimpleQueue.Get(nIdx))
                        {
                            _simpleQueue.Enqueue(nIdx);
                            _inSimpleQueue.Set(nIdx, true);
                        }
                        if (!_inOverlapQueue.Get(nIdx))
                        {
                            _overlapQueue.Enqueue(nIdx);
                            _inOverlapQueue.Set(nIdx, true);
                        }
                    }
                }
            }
        }

        private static List<int> GetUnknownNeighbors(int idx, out int remainingMines)
        {
            List<int> list = new List<int>(8);
            int r = idx / _width;
            int c = idx % _width;
            int flags = 0;
            for (int dr = -1; dr <= 1; dr++)
            {
                int nr = r + dr;
                if (nr < 0 || nr >= _height) continue;
                for (int dc = -1; dc <= 1; dc++)
                {
                    if (dr == 0 && dc == 0) continue;
                    int nc = c + dc;
                    if (nc < 0 || nc >= _width) continue;

                    int nIdx = nr * _width + nc;
                    if (_isMine.Get(nIdx)) flags++;
                    else if (_board[nIdx] == ' ' && !_isSafe.Get(nIdx)) list.Add(nIdx);
                }
            }
            remainingMines = (_board[idx] - '0') - flags;
            return list;
        }

        private static bool ProcessOverlap(int idxA)
        {
            int minesA;
            List<int> sA = GetUnknownNeighbors(idxA, out minesA);
            if (sA.Count == 0) return false;

            int[] potentialB = new int[72];
            int bCount = 0;
            foreach (int uIdx in sA)
            {
                int ur = uIdx / _width;
                int uc = uIdx % _width;
                for (int dr = -1; dr <= 1; dr++)
                {
                    int nr = ur + dr;
                    if (nr < 0 || nr >= _height) continue;
                    for (int dc = -1; dc <= 1; dc++)
                    {
                        int nc = uc + dc;
                        if (nc < 0 || nc >= _width) continue;
                        int idxB = nr * _width + nc;
                        if (idxB == idxA) continue;
                        byte b = _board[idxB];
                        if (b >= '0' && b <= '8')
                        {
                            bool seen = false;
                            for (int i = 0; i < bCount; i++) if (potentialB[i] == idxB) { seen = true; break; }
                            if (!seen) potentialB[bCount++] = idxB;
                        }
                    }
                }
            }

            bool changedTotal = false;
            for (int i = 0; i < bCount; i++)
            {
                int idxB = potentialB[i];
                int minesB;
                List<int> sB = GetUnknownNeighbors(idxB, out minesB);
                
                int countY = 0;
                foreach (int a in sA)
                {
                    foreach (int b in sB) if (a == b) { countY++; break; }
                }
                if (countY == 0) continue;

                int countX = sA.Count - countY;
                int countZ = sB.Count - countY;

                int minY = Math.Max(0, Math.Max(minesA - countX, minesB - countZ));
                int maxY = Math.Min(minesA, Math.Min(minesB, countY));

                if (minY > maxY) continue;

                // Deductions based on bounds
                if (minesA - maxY == countX && countX > 0) changedTotal |= MarkFromLists(sA, sB, false, true);
                if (minesA - minY == 0 && countX > 0) changedTotal |= MarkFromLists(sA, sB, true, true);
                if (minesB - maxY == countZ && countZ > 0) changedTotal |= MarkFromLists(sB, sA, false, true);
                if (minesB - minY == 0 && countZ > 0) changedTotal |= MarkFromLists(sB, sA, true, true);
                if (minY == countY && countY > 0) changedTotal |= MarkFromLists(sA, sB, false, false);
                if (maxY == 0 && countY > 0) changedTotal |= MarkFromLists(sA, sB, true, false);
            }
            return changedTotal;
        }

        private static bool MarkFromLists(List<int> list1, List<int> list2, bool safe, bool difference)
        {
            bool changed = false;
            foreach (int cell in list1)
            {
                bool inList2 = false;
                foreach (int other in list2) if (cell == other) { inList2 = true; break; }
                
                if (difference ? !inList2 : inList2)
                {
                    if (safe)
                    {
                        if (!_isSafe.Get(cell))
                        {
                            _isSafe.Set(cell, true);
                            EnqueueAffected(cell / _width, cell % _width);
                            changed = true;
                        }
                    }
                    else
                    {
                        if (!_isMine.Get(cell))
                        {
                            _isMine.Set(cell, true);
                            EnqueueAffected(cell / _width, cell % _width);
                            changed = true;
                        }
                    }
                }
            }
            return changed;
        }
    }
}
