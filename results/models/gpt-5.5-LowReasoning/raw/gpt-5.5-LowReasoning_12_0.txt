using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    const double EPS = 1e-9;

    struct Vec
    {
        public double X, Y, Z;
        public Vec(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    struct Orientation
    {
        public int[] M;              // row-major 3x3 signed permutation matrix
        public double Qw, Qx, Qy, Qz;
        public double MinX, MinY, MinZ, SizeX, SizeY, SizeZ;
        public long Count;
        public double WasteScore;
    }

    sealed class FastScanner
    {
        readonly Stream _stream = Console.OpenStandardInput();
        readonly byte[] _buffer = new byte[1 << 16];
        int _len, _ptr;

        int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string Next()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);
            if (c < 0) return "";
            var chars = new char[64];
            int n = 0;
            while (c > 32)
            {
                if (n == chars.Length) Array.Resize(ref chars, chars.Length * 2);
                chars[n++] = (char)c;
                c = Read();
            }
            return new string(chars, 0, n);
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);
    }

    public static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        var fs = new FastScanner();

        string first = fs.Next();
        if (string.IsNullOrEmpty(first)) return;
        int vCount = int.Parse(first, CultureInfo.InvariantCulture);
        int fCount = fs.NextInt();

        var vertices = new Vec[vCount];
        for (int i = 0; i < vCount; i++)
            vertices[i] = new Vec(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());

        // Faces are not needed for the conservative non-overlapping AABB packing
        // strategy, but they are consumed to honor the input format.
        for (int i = 0; i < fCount; i++)
        {
            fs.NextInt(); fs.NextInt(); fs.NextInt();
        }

        double W = fs.NextDouble(), H = fs.NextDouble(), D = fs.NextDouble();
        if (vCount == 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        List<Orientation> orientations = BuildOrientations(vertices, W, H, D);
        if (orientations.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        Orientation best = ChooseBest(orientations, W, H, D, vCount, fCount);
        if (best.Count <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        long nx = FitCount(W, best.SizeX);
        long ny = FitCount(H, best.SizeY);
        long nz = FitCount(D, best.SizeZ);

        using var output = new StreamWriter(Console.OpenStandardOutput(), new System.Text.UTF8Encoding(false), 1 << 20);
        output.WriteLine(best.Count.ToString(CultureInfo.InvariantCulture));

        // Place each rotated polyhedron at the low corner of its AABB cell, shifted
        // by -min(rotated vertices) so every vertex is inside the container.
        double baseX = -best.MinX, baseY = -best.MinY, baseZ = -best.MinZ;

        // Adaptive loop order: for very large outputs, keep X as the innermost
        // loop to make coordinate generation and text output cache-friendly.
        for (long iz = 0; iz < nz; iz++)
        {
            double tz = baseZ + iz * best.SizeZ;
            for (long iy = 0; iy < ny; iy++)
            {
                double ty = baseY + iy * best.SizeY;
                for (long ix = 0; ix < nx; ix++)
                {
                    double tx = baseX + ix * best.SizeX;
                    WritePlacement(output, tx, ty, tz, best);
                }
            }
        }
    }

    static List<Orientation> BuildOrientations(Vec[] vertices, double W, double H, double D)
    {
        var result = new List<Orientation>(24);
        int[][] perms =
        {
            new[] {0,1,2}, new[] {0,2,1}, new[] {1,0,2},
            new[] {1,2,0}, new[] {2,0,1}, new[] {2,1,0}
        };

        foreach (var p in perms)
        {
            for (int sx = -1; sx <= 1; sx += 2)
            for (int sy = -1; sy <= 1; sy += 2)
            for (int sz = -1; sz <= 1; sz += 2)
            {
                int[] m = new int[9];
                m[0 * 3 + p[0]] = sx;
                m[1 * 3 + p[1]] = sy;
                m[2 * 3 + p[2]] = sz;
                if (Det(m) != 1) continue; // proper rotations only

                var o = new Orientation { M = m };
                MatrixToQuaternion(m, out o.Qw, out o.Qx, out o.Qy, out o.Qz);
                ComputeBounds(vertices, ref o);

                if (o.SizeX <= EPS || o.SizeY <= EPS || o.SizeZ <= EPS) continue;
                long nx = FitCount(W, o.SizeX);
                long ny = FitCount(H, o.SizeY);
                long nz = FitCount(D, o.SizeZ);
                if (nx == 0 || ny == 0 || nz == 0) continue;

                o.Count = SaturatingMul(SaturatingMul(nx, ny), nz);
                o.WasteScore = (W - nx * o.SizeX) / Math.Max(W, EPS)
                             + (H - ny * o.SizeY) / Math.Max(H, EPS)
                             + (D - nz * o.SizeZ) / Math.Max(D, EPS);
                result.Add(o);
            }
        }
        return result;
    }

    static Orientation ChooseBest(List<Orientation> orientations, double W, double H, double D, int vCount, int fCount)
    {
        double volume = W * H * D;
        double complexity = vCount + fCount;

        Orientation best = orientations[0];
        foreach (var o in orientations)
        {
            bool better;
            if (volume <= 8000 && complexity <= 200)
                better = o.Count > best.Count || (o.Count == best.Count && o.WasteScore < best.WasteScore);
            else if (volume <= 1_000_000 && complexity <= 5000)
                better = o.Count > best.Count || (o.Count == best.Count && CellVolume(o) < CellVolume(best));
            else
                better = o.Count > best.Count || (o.Count == best.Count && LongestSide(o) < LongestSide(best));

            if (better) best = o;
        }
        return best;
    }

    static void ComputeBounds(Vec[] vertices, ref Orientation o)
    {
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;

        foreach (var v in vertices)
        {
            double x = o.M[0] * v.X + o.M[1] * v.Y + o.M[2] * v.Z;
            double y = o.M[3] * v.X + o.M[4] * v.Y + o.M[5] * v.Z;
            double z = o.M[6] * v.X + o.M[7] * v.Y + o.M[8] * v.Z;
            if (x < minX) minX = x; if (x > maxX) maxX = x;
            if (y < minY) minY = y; if (y > maxY) maxY = y;
            if (z < minZ) minZ = z; if (z > maxZ) maxZ = z;
        }

        o.MinX = minX; o.MinY = minY; o.MinZ = minZ;
        o.SizeX = Math.Max(maxX - minX, EPS);
        o.SizeY = Math.Max(maxY - minY, EPS);
        o.SizeZ = Math.Max(maxZ - minZ, EPS);
    }

    static long FitCount(double container, double item)
    {
        if (item <= EPS) return 0;
        double raw = Math.Floor((container + 1e-8) / item);
        if (raw <= 0) return 0;
        if (raw > long.MaxValue) return long.MaxValue;
        return (long)raw;
    }

    static long SaturatingMul(long a, long b)
    {
        if (a == 0 || b == 0) return 0;
        if (a > long.MaxValue / b) return long.MaxValue;
        return a * b;
    }

    static double CellVolume(Orientation o) => o.SizeX * o.SizeY * o.SizeZ;
    static double LongestSide(Orientation o) => Math.Max(o.SizeX, Math.Max(o.SizeY, o.SizeZ));

    static int Det(int[] m)
    {
        return m[0] * (m[4] * m[8] - m[5] * m[7])
             - m[1] * (m[3] * m[8] - m[5] * m[6])
             + m[2] * (m[3] * m[7] - m[4] * m[6]);
    }

    static void MatrixToQuaternion(int[] m, out double qw, out double qx, out double qy, out double qz)
    {
        double r00 = m[0], r01 = m[1], r02 = m[2];
        double r10 = m[3], r11 = m[4], r12 = m[5];
        double r20 = m[6], r21 = m[7], r22 = m[8];
        double trace = r00 + r11 + r22;

        if (trace > 0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0;
            qw = 0.25 * s;
            qx = (r21 - r12) / s;
            qy = (r02 - r20) / s;
            qz = (r10 - r01) / s;
        }
        else if (r00 > r11 && r00 > r22)
        {
            double s = Math.Sqrt(1.0 + r00 - r11 - r22) * 2.0;
            qw = (r21 - r12) / s;
            qx = 0.25 * s;
            qy = (r01 + r10) / s;
            qz = (r02 + r20) / s;
        }
        else if (r11 > r22)
        {
            double s = Math.Sqrt(1.0 + r11 - r00 - r22) * 2.0;
            qw = (r02 - r20) / s;
            qx = (r01 + r10) / s;
            qy = 0.25 * s;
            qz = (r12 + r21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + r22 - r00 - r11) * 2.0;
            qw = (r10 - r01) / s;
            qx = (r02 + r20) / s;
            qy = (r12 + r21) / s;
            qz = 0.25 * s;
        }

        double invLen = 1.0 / Math.Sqrt(qw * qw + qx * qx + qy * qy + qz * qz);
        qw *= invLen; qx *= invLen; qy *= invLen; qz *= invLen;
        if (qw < 0) { qw = -qw; qx = -qx; qy = -qy; qz = -qz; }
    }

    static void WritePlacement(StreamWriter w, double tx, double ty, double tz, Orientation o)
    {
        w.Write(Format(tx)); w.Write(' ');
        w.Write(Format(ty)); w.Write(' ');
        w.Write(Format(tz)); w.Write(' ');
        w.Write(Format(o.Qw)); w.Write(' ');
        w.Write(Format(o.Qx)); w.Write(' ');
        w.Write(Format(o.Qy)); w.Write(' ');
        w.WriteLine(Format(o.Qz));
    }

    static string Format(double x)
    {
        if (Math.Abs(x) < 5e-12) x = 0;
        return x.ToString("G17", CultureInfo.InvariantCulture);
    }
}
