using System;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _len, _ptr;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }
            return value * sign;
        }
    }

    private sealed class FastOutput
    {
        private readonly Stream _stream = Console.OpenStandardOutput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _ptr;

        public void WriteInt(int value)
        {
            if (_ptr > _buffer.Length - 16) Flush();

            if (value == 0)
            {
                _buffer[_ptr++] = (byte)'0';
                return;
            }

            if (value < 0)
            {
                _buffer[_ptr++] = (byte)'-';
                value = -value;
            }

            int start = _ptr;
            while (value > 0)
            {
                _buffer[_ptr++] = (byte)('0' + value % 10);
                value /= 10;
            }

            int end = _ptr - 1;
            while (start < end)
            {
                byte t = _buffer[start];
                _buffer[start++] = _buffer[end];
                _buffer[end--] = t;
            }
        }

        public void WriteSpace()
        {
            if (_ptr == _buffer.Length) Flush();
            _buffer[_ptr++] = (byte)' ';
        }

        public void Flush()
        {
            if (_ptr == 0) return;
            _stream.Write(_buffer, 0, _ptr);
            _ptr = 0;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        int arcCount = m * 2;
        int[] head = new int[n];
        int[] current = new int[n];
        int[] to = new int[arcCount];
        int[] next = new int[arcCount];
        Array.Fill(head, -1);

        int arc = 0;
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            fs.NextInt(); // Weight is irrelevant for this scalable covering route.

            to[arc] = v;
            next[arc] = head[u];
            head[u] = arc++;

            to[arc] = u;
            next[arc] = head[v];
            head[v] = arc++;
        }

        Array.Copy(head, current, n);

        bool[] used = new bool[arcCount];
        int maxRoute = arcCount + 1;
        int[] stack = new int[maxRoute];
        int[] route = new int[maxRoute];
        int sp = 0, rp = 0;
        stack[sp++] = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = current[v];

            while (e != -1 && used[e])
                e = next[e];

            current[v] = e;

            if (e == -1)
            {
                route[rp++] = v;
                sp--;
            }
            else
            {
                used[e] = true;
                current[v] = next[e];
                stack[sp++] = to[e];
            }
        }

        var output = new FastOutput();
        for (int i = rp - 1; i >= 0; i--)
        {
            output.WriteInt(route[i]);
            if (i != 0) output.WriteSpace();
        }
        output.Flush();
    }
}
