// Adaptive TSP heuristic solver, C++20 / MSVC compatible.
//
// The program always outputs a valid permutation beginning with city 0.
// Strategy:
//   * N <= 12: Held-Karp exact dynamic programming.
//   * N <= 2500: nearest-neighbor construction followed by bounded 2-opt.
//   * N <= 200000: Morton/Z-order construction followed by bounded local 2-opt.
//   * Larger: streaming-friendly Morton/Z-order construction only.
//
// For very large inputs the dominant cost is necessarily reading coordinates,
// sorting compact keys, and writing N integers. This keeps memory use practical
// for 10M+ cities on the stated machine and avoids non-standard headers.

#include <algorithm>
#include <array>
#include <bit>
#include <cstdint>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <limits>
#include <numeric>
#include <string>
#include <vector>

struct City {
    double x;
    double y;
};

struct KeyIndex {
    std::uint64_t key;
    std::uint32_t index;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = a.x - b.x;
    const double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static std::uint64_t part1by1(std::uint32_t x) {
    std::uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2))  & 0x3333333333333333ULL;
    v = (v | (v << 1))  & 0x5555555555555555ULL;
    return v;
}

static std::uint64_t morton2d(std::uint32_t x, std::uint32_t y) {
    return part1by1(x) | (part1by1(y) << 1);
}

static std::vector<int> exact_held_karp(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    if (n <= 1) return {0};

    const int m = n - 1;
    const int states = 1 << m;
    const double inf = std::numeric_limits<double>::infinity();
    std::vector<double> dp(static_cast<std::size_t>(states) * m, inf);
    std::vector<unsigned char> parent(static_cast<std::size_t>(states) * m, 255);

    for (int j = 0; j < m; ++j) {
        dp[static_cast<std::size_t>(1 << j) * m + j] = std::sqrt(dist2(c[0], c[j + 1]));
    }

    for (int mask = 1; mask < states; ++mask) {
        for (int last = 0; last < m; ++last) {
            const double cur = dp[static_cast<std::size_t>(mask) * m + last];
            if (!std::isfinite(cur)) continue;
            const int remaining = ((states - 1) ^ mask);
            for (int bits = remaining; bits; bits &= bits - 1) {
                const int nb = std::countr_zero(static_cast<unsigned>(bits));
                const int nmask = mask | (1 << nb);
                const double cand = cur + std::sqrt(dist2(c[last + 1], c[nb + 1]));
                double& slot = dp[static_cast<std::size_t>(nmask) * m + nb];
                if (cand < slot) {
                    slot = cand;
                    parent[static_cast<std::size_t>(nmask) * m + nb] = static_cast<unsigned char>(last);
                }
            }
        }
    }

    const int full = states - 1;
    double best = inf;
    int last = 0;
    for (int j = 0; j < m; ++j) {
        const double cand = dp[static_cast<std::size_t>(full) * m + j] + std::sqrt(dist2(c[j + 1], c[0]));
        if (cand < best) {
            best = cand;
            last = j;
        }
    }

    std::vector<int> rev;
    int mask = full;
    while (mask) {
        rev.push_back(last + 1);
        const int prev = parent[static_cast<std::size_t>(mask) * m + last];
        mask ^= (1 << last);
        if (!mask) break;
        last = prev;
    }
    std::reverse(rev.begin(), rev.end());

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    route.insert(route.end(), rev.begin(), rev.end());
    return route;
}

static std::vector<int> nearest_neighbor(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    std::vector<int> route;
    route.reserve(n);
    std::vector<unsigned char> used(n, 0);
    int cur = 0;
    route.push_back(0);
    used[0] = 1;

    for (int step = 1; step < n; ++step) {
        double best = std::numeric_limits<double>::infinity();
        int bi = -1;
        for (int i = 0; i < n; ++i) {
            if (!used[i]) {
                const double d = dist2(c[cur], c[i]);
                if (d < best) {
                    best = d;
                    bi = i;
                }
            }
        }
        cur = bi;
        used[cur] = 1;
        route.push_back(cur);
    }
    return route;
}

static bool two_opt_pass(std::vector<int>& r, const std::vector<City>& c, int window) {
    const int n = static_cast<int>(r.size());
    bool changed = false;
    for (int i = 1; i + 2 < n; ++i) {
        const int j_end = std::min(n - 1, i + window);
        for (int j = i + 1; j <= j_end; ++j) {
            const int a = r[i - 1], b = r[i], cc = r[j], d = r[(j + 1) % n];
            const double old_len = std::sqrt(dist2(c[a], c[b])) + std::sqrt(dist2(c[cc], c[d]));
            const double new_len = std::sqrt(dist2(c[a], c[cc])) + std::sqrt(dist2(c[b], c[d]));
            if (new_len + 1e-12 < old_len) {
                std::reverse(r.begin() + i, r.begin() + j + 1);
                changed = true;
            }
        }
    }
    return changed;
}

static void bounded_two_opt(std::vector<int>& r, const std::vector<City>& c, int passes, int window) {
    for (int p = 0; p < passes; ++p) {
        if (!two_opt_pass(r, c, window)) break;
    }
}

static std::vector<int> morton_route(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    double minx = c[0].x, maxx = c[0].x, miny = c[0].y, maxy = c[0].y;
    for (const City& p : c) {
        minx = std::min(minx, p.x);
        maxx = std::max(maxx, p.x);
        miny = std::min(miny, p.y);
        maxy = std::max(maxy, p.y);
    }

    const double sx = (maxx > minx) ? (4294967295.0 / (maxx - minx)) : 0.0;
    const double sy = (maxy > miny) ? (4294967295.0 / (maxy - miny)) : 0.0;

    std::vector<KeyIndex> keys;
    keys.reserve(n > 0 ? n - 1 : 0);
    for (int i = 1; i < n; ++i) {
        const std::uint32_t qx = static_cast<std::uint32_t>(std::clamp((c[i].x - minx) * sx, 0.0, 4294967295.0));
        const std::uint32_t qy = static_cast<std::uint32_t>(std::clamp((c[i].y - miny) * sy, 0.0, 4294967295.0));
        keys.push_back({morton2d(qx, qy), static_cast<std::uint32_t>(i)});
    }

    std::sort(keys.begin(), keys.end(), [](const KeyIndex& a, const KeyIndex& b) {
        if (a.key != b.key) return a.key < b.key;
        return a.index < b.index;
    });

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    for (const KeyIndex& ki : keys) {
        route.push_back(static_cast<int>(ki.index));
    }
    return route;
}

static void write_route(const std::vector<int>& route) {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::string out;
    out.reserve(std::min<std::size_t>(route.size() * 12ULL, 1ULL << 22));
    for (std::size_t i = 0; i < route.size(); ++i) {
        if (i) out.push_back(' ');
        out += std::to_string(route[i]);
        if (out.size() > (1ULL << 20)) {
            std::cout << out;
            out.clear();
        }
    }
    out.push_back('\n');
    std::cout << out;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::uint64_t n64 = 0;
    if (!(std::cin >> n64)) return 0;
    if (n64 == 0) {
        std::cout << '\n';
        return 0;
    }

    const std::uint64_t max_supported = static_cast<std::uint64_t>(std::numeric_limits<int>::max());
    const int n = static_cast<int>(std::min(n64, max_supported));

    std::vector<City> cities;
    cities.reserve(static_cast<std::size_t>(n));
    for (int i = 0; i < n; ++i) {
        City p{};
        std::cin >> p.x >> p.y;
        cities.push_back(p);
    }

    std::vector<int> route;
    if (n <= 12) {
        route = exact_held_karp(cities);
    } else if (n <= 2500) {
        route = nearest_neighbor(cities);
        bounded_two_opt(route, cities, 2, n <= 500 ? 80 : 35);
    } else {
        route = morton_route(cities);
        if (n <= 200000) {
            bounded_two_opt(route, cities, 1, n <= 20000 ? 24 : 10);
        }
    }

    // Ensure the output starts at city 0 even if a future strategy changes route orientation.
    const auto it = std::find(route.begin(), route.end(), 0);
    if (it != route.begin() && it != route.end()) {
        std::rotate(route.begin(), it, route.end());
    }

    write_route(route);
    return 0;
}
