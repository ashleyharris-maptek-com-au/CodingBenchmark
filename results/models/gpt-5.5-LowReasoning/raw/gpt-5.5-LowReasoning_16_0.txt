#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <string>
#include <utility>
#include <vector>

using namespace std;

struct Task {
    int machine = 0;
    long long duration = 0;
};

struct Problem {
    int J = 0, M = 0;
    long long totalTasks = 0;
    vector<vector<Task>> jobs;
    vector<vector<long long>> remainingJobWork;
};

struct Schedule {
    vector<vector<long long>> start;
    long long makespan = numeric_limits<long long>::max();
};

static Problem readProblem() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Problem p;
    if (!(cin >> p.J >> p.M)) return p;
    p.jobs.resize(p.J);
    for (int j = 0; j < p.J; ++j) {
        int T;
        cin >> T;
        p.jobs[j].resize(T);
        p.totalTasks += T;
        for (int i = 0; i < T; ++i) {
            cin >> p.jobs[j][i].machine >> p.jobs[j][i].duration;
        }
    }

    p.remainingJobWork.resize(p.J);
    for (int j = 0; j < p.J; ++j) {
        int T = (int)p.jobs[j].size();
        p.remainingJobWork[j].assign(T + 1, 0);
        for (int i = T - 1; i >= 0; --i) {
            p.remainingJobWork[j][i] = p.remainingJobWork[j][i + 1] + p.jobs[j][i].duration;
        }
    }
    return p;
}

enum class Rule {
    EarliestStart,
    ShortestProcessingTime,
    LongestProcessingTime,
    MostWorkRemaining,
    LeastWorkRemaining,
    MostOperationsRemaining,
    LeastOperationsRemaining
};

struct QueueItem {
    long long earliest = 0;
    long long secondary = 0;
    int job = 0;
    int version = 0;
};

struct ItemGreater {
    bool operator()(const QueueItem& a, const QueueItem& b) const {
        if (a.earliest != b.earliest) return a.earliest > b.earliest;
        if (a.secondary != b.secondary) return a.secondary > b.secondary;
        return a.job > b.job;
    }
};

static long long secondaryKey(const Problem& p, Rule rule, int job, int op) {
    const Task& t = p.jobs[job][op];
    int remainingOps = (int)p.jobs[job].size() - op;
    switch (rule) {
        case Rule::ShortestProcessingTime: return t.duration;
        case Rule::LongestProcessingTime: return -t.duration;
        case Rule::MostWorkRemaining: return -p.remainingJobWork[job][op];
        case Rule::LeastWorkRemaining: return p.remainingJobWork[job][op];
        case Rule::MostOperationsRemaining: return -remainingOps;
        case Rule::LeastOperationsRemaining: return remainingOps;
        case Rule::EarliestStart:
        default: return job;
    }
}

static Schedule dispatchSchedule(const Problem& p, Rule rule) {
    Schedule s;
    s.start.resize(p.J);
    vector<int> nextOp(p.J, 0), version(p.J, 0);
    vector<long long> jobReady(p.J, 0), machineReady(max(1, p.M), 0);
    priority_queue<QueueItem, vector<QueueItem>, ItemGreater> pq;

    for (int j = 0; j < p.J; ++j) {
        s.start[j].assign(p.jobs[j].size(), 0);
        if (!p.jobs[j].empty()) {
            const Task& t = p.jobs[j][0];
            long long earliest = max(jobReady[j], machineReady[t.machine]);
            pq.push({earliest, secondaryKey(p, rule, j, 0), j, version[j]});
        }
    }

    long long done = 0;
    while (done < p.totalTasks && !pq.empty()) {
        QueueItem cur = pq.top();
        pq.pop();
        int j = cur.job;
        if (cur.version != version[j]) continue;
        int op = nextOp[j];
        if (op >= (int)p.jobs[j].size()) continue;

        const Task& t = p.jobs[j][op];
        long long trueEarliest = max(jobReady[j], machineReady[t.machine]);
        long long trueSecondary = secondaryKey(p, rule, j, op);
        if (cur.earliest != trueEarliest || cur.secondary != trueSecondary) {
            pq.push({trueEarliest, trueSecondary, j, version[j]});
            continue;
        }

        long long st = trueEarliest;
        long long en = st + t.duration;
        s.start[j][op] = st;
        jobReady[j] = en;
        machineReady[t.machine] = en;
        s.makespan = max(s.makespan == numeric_limits<long long>::max() ? 0 : s.makespan, en);
        ++nextOp[j];
        ++version[j];
        ++done;

        if (nextOp[j] < (int)p.jobs[j].size()) {
            const Task& nt = p.jobs[j][nextOp[j]];
            long long earliest = max(jobReady[j], machineReady[nt.machine]);
            pq.push({earliest, secondaryKey(p, rule, j, nextOp[j]), j, version[j]});
        }
    }

    if (p.totalTasks == 0) s.makespan = 0;
    return s;
}

struct ExactSearch {
    const Problem& p;
    long long deadlineMs;
    chrono::steady_clock::time_point begin;
    vector<int> nextOp;
    vector<long long> jobReady, machineReady;
    Schedule cur, best;
    long long nodes = 0;
    long long nodeLimit;

    ExactSearch(const Problem& problem, Schedule incumbent, long long ms, long long limit)
        : p(problem), deadlineMs(ms), begin(chrono::steady_clock::now()),
          nextOp(problem.J, 0), jobReady(problem.J, 0), machineReady(max(1, problem.M), 0),
          cur(), best(std::move(incumbent)), nodeLimit(limit) {
        cur.start.resize(p.J);
        for (int j = 0; j < p.J; ++j) cur.start[j].assign(p.jobs[j].size(), 0);
    }

    bool timedOut() const {
        auto elapsed = chrono::duration_cast<chrono::milliseconds>(chrono::steady_clock::now() - begin).count();
        return elapsed >= deadlineMs;
    }

    long long lowerBound(long long currentMax) const {
        long long lb = currentMax;
        for (int j = 0; j < p.J; ++j) {
            lb = max(lb, jobReady[j] + p.remainingJobWork[j][nextOp[j]]);
        }
        return lb;
    }

    void dfs(long long placed, long long currentMax) {
        if (++nodes > nodeLimit || timedOut()) return;
        if (lowerBound(currentMax) >= best.makespan) return;
        if (placed == p.totalTasks) {
            if (currentMax < best.makespan) {
                best = cur;
                best.makespan = currentMax;
            }
            return;
        }

        vector<int> cand;
        cand.reserve(p.J);
        for (int j = 0; j < p.J; ++j) {
            if (nextOp[j] < (int)p.jobs[j].size()) cand.push_back(j);
        }
        sort(cand.begin(), cand.end(), [&](int a, int b) {
            const Task& ta = p.jobs[a][nextOp[a]];
            const Task& tb = p.jobs[b][nextOp[b]];
            long long ea = max(jobReady[a], machineReady[ta.machine]);
            long long eb = max(jobReady[b], machineReady[tb.machine]);
            if (ea != eb) return ea < eb;
            if (p.remainingJobWork[a][nextOp[a]] != p.remainingJobWork[b][nextOp[b]]) {
                return p.remainingJobWork[a][nextOp[a]] > p.remainingJobWork[b][nextOp[b]];
            }
            return a < b;
        });

        for (int j : cand) {
            int op = nextOp[j];
            const Task& t = p.jobs[j][op];
            long long oldJob = jobReady[j], oldMachine = machineReady[t.machine];
            long long st = max(oldJob, oldMachine);
            long long en = st + t.duration;

            cur.start[j][op] = st;
            jobReady[j] = en;
            machineReady[t.machine] = en;
            ++nextOp[j];
            dfs(placed + 1, max(currentMax, en));
            --nextOp[j];
            jobReady[j] = oldJob;
            machineReady[t.machine] = oldMachine;
        }
    }
};

static Schedule solve(const Problem& p) {
    vector<Rule> rules;
    if (p.totalTasks <= 2000) {
        rules = {Rule::EarliestStart, Rule::MostWorkRemaining, Rule::ShortestProcessingTime,
                 Rule::LongestProcessingTime, Rule::LeastWorkRemaining,
                 Rule::MostOperationsRemaining, Rule::LeastOperationsRemaining};
    } else if (p.totalTasks <= 20000) {
        rules = {Rule::EarliestStart, Rule::MostWorkRemaining, Rule::ShortestProcessingTime,
                 Rule::MostOperationsRemaining};
    } else {
        rules = {Rule::EarliestStart, Rule::MostWorkRemaining};
    }

    Schedule best;
    for (Rule rule : rules) {
        Schedule candidate = dispatchSchedule(p, rule);
        if (candidate.makespan < best.makespan) best = std::move(candidate);
    }

    if (p.totalTasks <= 14 && p.J <= 8) {
        ExactSearch ex(p, best, 750, 2000000);
        ex.dfs(0, 0);
        best = std::move(ex.best);
    }
    return best;
}

int main() {
    Problem p = readProblem();
    Schedule s = solve(p);

    for (int j = 0; j < p.J; ++j) {
        for (int i = 0; i < (int)p.jobs[j].size(); ++i) {
            if (i) cout << ' ';
            cout << s.start[j][i] << ' ' << p.jobs[j][i].machine;
        }
        cout << '\n';
    }
    return 0;
}
