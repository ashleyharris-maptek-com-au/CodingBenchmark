struct Vec3 {
    double x = 0, y = 0, z = 0;
};

struct Tri {
    int a = 0, b = 0, c = 0;
};

struct Aabb {
    Vec3 mn, mx;
};

static Vec3 operator+(const Vec3& a, const Vec3& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
static Vec3 operator-(const Vec3& a, const Vec3& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
static Vec3 operator*(const Vec3& a, double s) { return {a.x * s, a.y * s, a.z * s}; }
static double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}

struct Mesh {
    vector<Vec3> v;
    vector<Tri> f;
};

static Aabb empty_box() {
    double inf = std::numeric_limits<double>::infinity();
    return {{inf, inf, inf}, {-inf, -inf, -inf}};
}

static void extend(Aabb& b, const Vec3& p) {
    b.mn.x = std::min(b.mn.x, p.x); b.mn.y = std::min(b.mn.y, p.y); b.mn.z = std::min(b.mn.z, p.z);
    b.mx.x = std::max(b.mx.x, p.x); b.mx.y = std::max(b.mx.y, p.y); b.mx.z = std::max(b.mx.z, p.z);
}

static Aabb mesh_box(const Mesh& m) {
    Aabb b = empty_box();
    for (const auto& p : m.v) extend(b, p);
    return b;
}

static bool disjoint(const Aabb& a, const Aabb& b) {
    const double eps = 1e-10;
    return a.mx.x < b.mn.x - eps || b.mx.x < a.mn.x - eps ||
           a.mx.y < b.mn.y - eps || b.mx.y < a.mn.y - eps ||
           a.mx.z < b.mn.z - eps || b.mx.z < a.mn.z - eps;
}

static bool point_in_box(const Vec3& p, const Aabb& b) {
    const double eps = 1e-10;
    return p.x >= b.mn.x - eps && p.x <= b.mx.x + eps &&
           p.y >= b.mn.y - eps && p.y <= b.mx.y + eps &&
           p.z >= b.mn.z - eps && p.z <= b.mx.z + eps;
}

struct Bvh {
    struct Node {
        Aabb box;
        int left = -1, right = -1;
        int begin = 0, end = 0;
    };

    const Mesh* mesh = nullptr;
    vector<int> ids;
    vector<Node> nodes;

    explicit Bvh(const Mesh& m) : mesh(&m), ids(m.f.size()) {
        std::iota(ids.begin(), ids.end(), 0);
        nodes.reserve(m.f.empty() ? 0 : m.f.size() * 2);
        if (!ids.empty()) build(0, static_cast<int>(ids.size()));
    }

    Aabb tri_box(int id) const {
        const Tri& t = mesh->f[id];
        Aabb b = empty_box();
        extend(b, mesh->v[t.a]);
        extend(b, mesh->v[t.b]);
        extend(b, mesh->v[t.c]);
        return b;
    }

    Vec3 centroid(int id) const {
        const Tri& t = mesh->f[id];
        return (mesh->v[t.a] + mesh->v[t.b] + mesh->v[t.c]) * (1.0 / 3.0);
    }

    int build(int begin, int end) {
        int idx = static_cast<int>(nodes.size());
        nodes.push_back({});
        Node& n = nodes[idx];
        n.begin = begin;
        n.end = end;
        n.box = empty_box();
        Aabb cb = empty_box();
        for (int i = begin; i < end; ++i) {
            Aabb tb = tri_box(ids[i]);
            extend(n.box, tb.mn);
            extend(n.box, tb.mx);
            extend(cb, centroid(ids[i]));
        }
        int count = end - begin;
        if (count <= 8) return idx;
        Vec3 ext = cb.mx - cb.mn;
        int axis = 0;
        if (ext.y > ext.x && ext.y >= ext.z) axis = 1;
        else if (ext.z > ext.x && ext.z >= ext.y) axis = 2;
        int mid = begin + count / 2;
        std::nth_element(ids.begin() + begin, ids.begin() + mid, ids.begin() + end,
            [&](int lhs, int rhs) {
                Vec3 a = centroid(lhs), b = centroid(rhs);
                return axis == 0 ? a.x < b.x : (axis == 1 ? a.y < b.y : a.z < b.z);
            });
        int left = build(begin, mid);
        int right = build(mid, end);
        nodes[idx].left = left;
        nodes[idx].right = right;
        return idx;
    }

    static bool ray_box(const Vec3& p, const Aabb& b) {
        const double eps = 1e-12;
        if (p.y < b.mn.y - eps || p.y > b.mx.y + eps || p.z < b.mn.z - eps || p.z > b.mx.z + eps) return false;
        return b.mx.x >= p.x - eps;
    }

    bool ray_tri_x(const Vec3& p, int id, double& tx) const {
        const Tri& tr = mesh->f[id];
        Vec3 v0 = mesh->v[tr.a], v1 = mesh->v[tr.b], v2 = mesh->v[tr.c];
        Vec3 dir{1.0, 0.0, 0.0};
        Vec3 e1 = v1 - v0, e2 = v2 - v0;
        Vec3 h = cross(dir, e2);
        double det = dot(e1, h);
        if (std::fabs(det) < 1e-14) return false;
        double inv = 1.0 / det;
        Vec3 s = p - v0;
        double u = inv * dot(s, h);
        if (u < -1e-12 || u > 1.0 + 1e-12) return false;
        Vec3 q = cross(s, e1);
        double v = inv * dot(dir, q);
        if (v < -1e-12 || u + v > 1.0 + 1e-12) return false;
        tx = inv * dot(e2, q);
        return tx > 1e-10;
    }

    void ray_hits(int node, const Vec3& p, vector<double>& xs) const {
        const Node& n = nodes[node];
        if (!ray_box(p, n.box)) return;
        if (n.left < 0) {
            for (int i = n.begin; i < n.end; ++i) {
                double t = 0.0;
                if (ray_tri_x(p, ids[i], t)) xs.push_back(t);
            }
            return;
        }
        ray_hits(n.left, p, xs);
        ray_hits(n.right, p, xs);
    }

    bool contains(const Vec3& p) const {
        if (nodes.empty() || !point_in_box(p, nodes[0].box)) return false;
        vector<double> xs;
        xs.reserve(32);
        ray_hits(0, p, xs);
        if (xs.empty()) return false;
        std::sort(xs.begin(), xs.end());
        int unique_hits = 0;
        double last = -1.0;
        for (double x : xs) {
            if (unique_hits == 0 || std::fabs(x - last) > 1e-8) {
                ++unique_hits;
                last = x;
            }
        }
        return (unique_hits & 1) != 0;
    }
};

static Mesh read_mesh(int vc, int fc) {
    Mesh m;
    m.v.resize(vc);
    m.f.resize(fc);
    for (auto& p : m.v) cin >> p.x >> p.y >> p.z;
    for (auto& t : m.f) cin >> t.a >> t.b >> t.c;
    return m;
}

static void append_mesh(Mesh& out, const Mesh& m) {
    int base = static_cast<int>(out.v.size());
    out.v.insert(out.v.end(), m.v.begin(), m.v.end());
    for (Tri t : m.f) out.f.push_back({t.a + base, t.b + base, t.c + base});
}

static Mesh compact_from_faces(const Mesh& a, const vector<char>& keep_a, const Mesh& b, const vector<char>& keep_b) {
    Mesh out;
    vector<int> map_a(a.v.size(), -1), map_b(b.v.size(), -1);
    auto add_vertex = [&](const Vec3& p) {
        out.v.push_back(p);
        return static_cast<int>(out.v.size() - 1);
    };
    for (size_t i = 0; i < a.f.size(); ++i) if (keep_a[i]) {
        Tri t = a.f[i];
        if (map_a[t.a] < 0) map_a[t.a] = add_vertex(a.v[t.a]);
        if (map_a[t.b] < 0) map_a[t.b] = add_vertex(a.v[t.b]);
        if (map_a[t.c] < 0) map_a[t.c] = add_vertex(a.v[t.c]);
        out.f.push_back({map_a[t.a], map_a[t.b], map_a[t.c]});
    }
    for (size_t i = 0; i < b.f.size(); ++i) if (keep_b[i]) {
        Tri t = b.f[i];
        if (map_b[t.a] < 0) map_b[t.a] = add_vertex(b.v[t.a]);
        if (map_b[t.b] < 0) map_b[t.b] = add_vertex(b.v[t.b]);
        if (map_b[t.c] < 0) map_b[t.c] = add_vertex(b.v[t.c]);
        out.f.push_back({map_b[t.a], map_b[t.b], map_b[t.c]});
    }
    return out;
}

static bool all_vertices_inside(const Mesh& inner, const Bvh& outer) {
    if (inner.v.empty()) return true;
    for (const auto& p : inner.v) if (!outer.contains(p)) return false;
    return true;
}

static Vec3 face_center(const Mesh& m, const Tri& t) {
    return (m.v[t.a] + m.v[t.b] + m.v[t.c]) * (1.0 / 3.0);
}

static void write_mesh(const Mesh& m) {
    cout << m.v.size() << ' ' << m.f.size() << '\n';
    cout << std::setprecision(17);
    for (const auto& p : m.v) cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const auto& t : m.f) cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int va, fa, vb, fb;
    if (!(cin >> va >> fa >> vb >> fb)) return 0;
    Mesh A = read_mesh(va, fa);
    Mesh B = read_mesh(vb, fb);

    Aabb ba = mesh_box(A), bb = mesh_box(B);
    Mesh out;

    if (disjoint(ba, bb)) {
        append_mesh(out, A);
        append_mesh(out, B);
        write_mesh(out);
        return 0;
    }

    Bvh bvhA(A), bvhB(B);

    if (all_vertices_inside(B, bvhA)) {
        write_mesh(A);
        return 0;
    }
    if (all_vertices_inside(A, bvhB)) {
        write_mesh(B);
        return 0;
    }

    vector<char> keepA(A.f.size(), 1), keepB(B.f.size(), 1);
    for (size_t i = 0; i < A.f.size(); ++i) keepA[i] = !bvhB.contains(face_center(A, A.f[i]));
    for (size_t i = 0; i < B.f.size(); ++i) keepB[i] = !bvhA.contains(face_center(B, B.f[i]));

    out = compact_from_faces(A, keepA, B, keepB);
    if (out.f.empty()) {
        append_mesh(out, A);
        append_mesh(out, B);
    }
    write_mesh(out);
    return 0;
}
