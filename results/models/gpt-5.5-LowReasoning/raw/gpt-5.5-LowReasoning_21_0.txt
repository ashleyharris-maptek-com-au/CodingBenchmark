// N-dimensional Snake solver, C++20 / standard library only.
//
// The input format supplies the snake head and length, but not the initial body
// coordinates.  This solver therefore starts with a conservative degenerate
// body at the head, then simulates every emitted move.  Each turn it first tries
// a bounded breadth-first search to a remaining food item; if the searchable
// area is too large or blocked, it falls back to a safe greedy move that reduces
// distance to food while preferring moves with more immediate escape options.

#include <algorithm>
#include <cstdlib>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <queue>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using Coord = std::vector<int>;

struct Move {
    int axis = 0;
    int dir = 1;
    bool valid = false;
};

struct SearchNode {
    Coord pos;
    int parent = -1;
    Move move;
};

static int D = 0;
static long long maxTurns = 0;
static Coord bounds;
static std::unordered_set<std::string> obstacles;
static std::unordered_set<std::string> foods;
static std::deque<Coord> snake;
static std::unordered_map<std::string, int> occupied;

static std::string keyOf(const Coord& c) {
    std::string s;
    s.reserve(c.size() * 4);
    for (int v : c) {
        s += std::to_string(v);
        s.push_back(',');
    }
    return s;
}

static bool inside(const Coord& c) {
    for (int i = 0; i < D; ++i) {
        if (c[i] < 0 || c[i] >= bounds[i]) return false;
    }
    return true;
}

static long long manhattan(const Coord& a, const Coord& b) {
    long long dist = 0;
    for (int i = 0; i < D; ++i) {
        dist += std::llabs(static_cast<long long>(a[i]) - b[i]);
    }
    return dist;
}

static long long nearestFoodDistance(const Coord& p) {
    if (foods.empty()) return 0;
    long long best = std::numeric_limits<long long>::max() / 4;

    Coord f;
    for (const std::string& fk : foods) {
        f.clear();
        int value = 0;
        bool inNumber = false;
        for (char ch : fk) {
            if (ch == ',') {
                if (inNumber) f.push_back(value);
                value = 0;
                inNumber = false;
            } else {
                value = value * 10 + (ch - '0');
                inNumber = true;
            }
        }
        if (static_cast<int>(f.size()) == D) {
            best = std::min(best, manhattan(p, f));
        }
    }
    return best;
}

static bool cellBlockedForMove(const Coord& next, bool willGrow) {
    if (!inside(next)) return true;
    const std::string k = keyOf(next);
    if (obstacles.find(k) != obstacles.end()) return true;

    auto it = occupied.find(k);
    if (it == occupied.end() || it->second <= 0) return false;

    // Moving into the current tail is legal only if the snake does not grow.
    if (!willGrow && !snake.empty() && k == keyOf(snake.back()) && it->second == 1) {
        return false;
    }
    return true;
}

static Coord applyMove(const Coord& p, const Move& m) {
    Coord q = p;
    q[m.axis] += m.dir;
    return q;
}

static int localFreedom(const Coord& p) {
    int freeCount = 0;
    for (int axis = 0; axis < D; ++axis) {
        for (int dir : {-1, 1}) {
            Coord q = p;
            q[axis] += dir;
            bool grows = foods.find(keyOf(q)) != foods.end();
            if (!cellBlockedForMove(q, grows)) ++freeCount;
        }
    }
    return freeCount;
}

static Move bfsMoveToFood(const Coord& start) {
    if (foods.empty()) return Move{};

    const int nodeLimit = 250000;
    std::vector<SearchNode> nodes;
    nodes.reserve(4096);
    std::queue<int> q;
    std::unordered_map<std::string, int> seen;
    seen.reserve(8192);

    nodes.push_back(SearchNode{start, -1, Move{}});
    seen.emplace(keyOf(start), 0);
    q.push(0);

    int found = -1;
    while (!q.empty() && static_cast<int>(nodes.size()) < nodeLimit) {
        int idx = q.front();
        q.pop();

        const Coord& p = nodes[idx].pos;
        if (idx != 0 && foods.find(keyOf(p)) != foods.end()) {
            found = idx;
            break;
        }

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Move mv{axis, dir, true};
                Coord np = applyMove(p, mv);
                if (!inside(np)) continue;

                const std::string nk = keyOf(np);
                if (seen.find(nk) != seen.end()) continue;
                if (obstacles.find(nk) != obstacles.end()) continue;

                // For planning, keep current body cells blocked except the
                // present tail.  Replanning every turn corrects this conservatism.
                auto occ = occupied.find(nk);
                if (occ != occupied.end() && occ->second > 0) {
                    if (snake.empty() || nk != keyOf(snake.back()) || occ->second != 1) {
                        continue;
                    }
                }

                int nextIdx = static_cast<int>(nodes.size());
                nodes.push_back(SearchNode{np, idx, mv});
                seen.emplace(nk, nextIdx);
                q.push(nextIdx);
            }
        }
    }

    if (found < 0) return Move{};
    int cur = found;
    while (nodes[cur].parent != 0 && nodes[cur].parent >= 0) {
        cur = nodes[cur].parent;
    }
    Move ans = nodes[cur].move;
    ans.valid = true;
    return ans;
}

static Move greedyMove(const Coord& head) {
    Move best;
    long long bestScore = std::numeric_limits<long long>::min();

    for (int axis = 0; axis < D; ++axis) {
        for (int dir : {-1, 1}) {
            Move mv{axis, dir, true};
            Coord np = applyMove(head, mv);
            bool grows = foods.find(keyOf(np)) != foods.end();
            if (cellBlockedForMove(np, grows)) continue;

            long long dist = nearestFoodDistance(np);
            long long centerBias = -std::llabs(2LL * np[axis] + 1 - bounds[axis]);
            long long score = -1000LL * dist + 25LL * localFreedom(np) + centerBias;
            if (!best.valid || score > bestScore) {
                best = mv;
                bestScore = score;
            }
        }
    }
    return best;
}

static bool advanceSnake(const Move& mv) {
    Coord next = applyMove(snake.front(), mv);
    const std::string nk = keyOf(next);
    bool grows = foods.find(nk) != foods.end();
    if (cellBlockedForMove(next, grows)) return false;

    snake.push_front(next);
    ++occupied[nk];

    if (grows) {
        foods.erase(nk);
    } else {
        const std::string tk = keyOf(snake.back());
        snake.pop_back();
        auto it = occupied.find(tk);
        if (it != occupied.end()) {
            --it->second;
            if (it->second <= 0) occupied.erase(it);
        }
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    if (!(std::cin >> D >> maxTurns)) return 0;
    bounds.assign(D, 0);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];

    int initialLength = 1;
    Coord start(D, 0);
    std::cin >> initialLength;
    for (int i = 0; i < D; ++i) std::cin >> start[i];

    int F = 0;
    std::cin >> F;
    for (int i = 0; i < F; ++i) {
        Coord f(D, 0);
        for (int j = 0; j < D; ++j) std::cin >> f[j];
        if (inside(f)) foods.insert(keyOf(f));
    }

    int O = 0;
    std::cin >> O;
    for (int i = 0; i < O; ++i) {
        Coord o(D, 0);
        for (int j = 0; j < D; ++j) std::cin >> o[j];
        if (inside(o)) obstacles.insert(keyOf(o));
    }

    initialLength = std::max(1, initialLength);
    for (int i = 0; i < initialLength; ++i) snake.push_back(start);
    occupied[keyOf(start)] = initialLength;

    for (long long turn = 0; turn < maxTurns && !foods.empty(); ++turn) {
        Move mv = bfsMoveToFood(snake.front());
        if (!mv.valid) mv = greedyMove(snake.front());
        if (!mv.valid) break;

        std::cout << mv.axis << ' ' << mv.dir << '\n';
        if (!advanceSnake(mv)) break;
    }

    return 0;
}
