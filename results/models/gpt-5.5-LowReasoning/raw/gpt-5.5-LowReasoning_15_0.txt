use std::cmp::Ordering;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy)]
struct Footprint {
    offset: Point,
    radius: f64,
    quat: Quat,
}

fn cross(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn dist_to_line(p: Point, a: Point, b: Point) -> f64 {
    let dx = b.x - a.x;
    let dy = b.y - a.y;
    let len = (dx * dx + dy * dy).sqrt();
    if len <= 1e-12 {
        0.0
    } else {
        ((p.x - a.x) * dy - (p.y - a.y) * dx).abs() / len
    }
}

fn convex_hull(mut pts: Vec<Point>) -> Vec<Point> {
    pts.sort_by(|a, b| {
        a.x.partial_cmp(&b.x)
            .unwrap_or(Ordering::Equal)
            .then(a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal))
    });
    let mut lower: Vec<Point> = Vec::new();
    for p in &pts {
        while lower.len() >= 2
            && cross(lower[lower.len() - 2], lower[lower.len() - 1], *p) <= 1e-12
        {
            lower.pop();
        }
        lower.push(*p);
    }
    let mut upper: Vec<Point> = Vec::new();
    for p in pts.iter().rev() {
        while upper.len() >= 2
            && cross(upper[upper.len() - 2], upper[upper.len() - 1], *p) <= 1e-12
        {
            upper.pop();
        }
        upper.push(*p);
    }
    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn polygon_centroid(poly: &[Point]) -> Point {
    let mut area2 = 0.0;
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let cr = a.x * b.y - b.x * a.y;
        area2 += cr;
        cx += (a.x + b.x) * cr;
        cy += (a.y + b.y) * cr;
    }
    if area2.abs() <= 1e-12 {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for p in poly {
            sx += p.x;
            sy += p.y;
        }
        let n = poly.len() as f64;
        Point { x: sx / n, y: sy / n }
    } else {
        Point {
            x: cx / (3.0 * area2),
            y: cy / (3.0 * area2),
        }
    }
}

fn point_in_poly(p: Point, poly: &[Point]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[j];
        if cross(a, b, p).abs() < 1e-9
            && p.x >= a.x.min(b.x) - 1e-9
            && p.x <= a.x.max(b.x) + 1e-9
            && p.y >= a.y.min(b.y) - 1e-9
            && p.y <= a.y.max(b.y) + 1e-9
        {
            return true;
        }
        let hit = (a.y > p.y) != (b.y > p.y)
            && p.x < (b.x - a.x) * (p.y - a.y) / ((b.y - a.y).abs().max(1e-300).copysign(b.y - a.y)) + a.x;
        if hit {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn orient(a: Point, b: Point, c: Point) -> f64 {
    cross(a, b, c)
}

fn segments_intersect(a: Point, b: Point, c: Point, d: Point) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    if o1.abs() < 1e-9 && c.x >= a.x.min(b.x) - 1e-9 && c.x <= a.x.max(b.x) + 1e-9 && c.y >= a.y.min(b.y) - 1e-9 && c.y <= a.y.max(b.y) + 1e-9 {
        return true;
    }
    if o2.abs() < 1e-9 && d.x >= a.x.min(b.x) - 1e-9 && d.x <= a.x.max(b.x) + 1e-9 && d.y >= a.y.min(b.y) - 1e-9 && d.y <= a.y.max(b.y) + 1e-9 {
        return true;
    }
    if o3.abs() < 1e-9 && a.x >= c.x.min(d.x) - 1e-9 && a.x <= c.x.max(d.x) + 1e-9 && a.y >= c.y.min(d.y) - 1e-9 && a.y <= c.y.max(d.y) + 1e-9 {
        return true;
    }
    if o4.abs() < 1e-9 && b.x >= c.x.min(d.x) - 1e-9 && b.x <= c.x.max(d.x) + 1e-9 && b.y >= c.y.min(d.y) - 1e-9 && b.y <= c.y.max(d.y) + 1e-9 {
        return true;
    }
    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn rect_touches_poly(cx: f64, cy: f64, half: f64, poly: &[Point]) -> bool {
    let minx = cx - half;
    let maxx = cx + half;
    let miny = cy - half;
    let maxy = cy + half;
    let corners = [
        Point { x: minx, y: miny },
        Point { x: maxx, y: miny },
        Point { x: maxx, y: maxy },
        Point { x: minx, y: maxy },
    ];
    for c in corners {
        if point_in_poly(c, poly) {
            return true;
        }
    }
    if point_in_poly(Point { x: cx, y: cy }, poly) {
        return true;
    }
    for p in poly {
        if p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy {
            return true;
        }
    }
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        for e in 0..4 {
            if segments_intersect(a, b, corners[e], corners[(e + 1) % 4]) {
                return true;
            }
        }
    }
    false
}

fn footprint_for(angle: f64, sx: f64, sy: f64, sz: f64, scale: f64) -> Footprint {
    let base = [
        (1.0, 0.0, -0.707),
        (-1.0, 0.0, -0.707),
        (0.0, 1.0, 0.707),
        (0.0, -1.0, 0.707),
    ];
    let ca = angle.cos();
    let sa = angle.sin();
    let ax = sx / sz;
    let ay = sy / sz;
    let mut proj = Vec::with_capacity(4);
    for (x, y, z) in base {
        let rx = ca * x - sa * y;
        let ry = sa * x + ca * y;
        proj.push(Point {
            x: (rx - ax * z) * scale,
            y: (ry - ay * z) * scale,
        });
    }
    let hull = convex_hull(proj);
    let offset = polygon_centroid(&hull);
    let mut radius = f64::INFINITY;
    for i in 0..hull.len() {
        radius = radius.min(dist_to_line(offset, hull[i], hull[(i + 1) % hull.len()]));
    }
    let half = angle * 0.5;
    Footprint {
        offset,
        radius: radius.max(0.05),
        quat: Quat {
            w: half.cos(),
            x: 0.0,
            y: 0.0,
            z: half.sin(),
        },
    }
}

fn best_footprint(sx: f64, sy: f64, sz: f64, scale: f64) -> Footprint {
    let mut best = footprint_for(0.0, sx, sy, sz, scale);
    let samples = 48;
    for i in 1..samples {
        let angle = (i as f64) * std::f64::consts::TAU / (samples as f64);
        let fp = footprint_for(angle, sx, sy, sz, scale);
        if fp.radius > best.radius {
            best = fp;
        }
    }
    best
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("stdin");
    let mut it = input.split_whitespace();
    let nv = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) if v > 0 => v,
        _ => {
            println!("0");
            return;
        }
    };
    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let y = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        poly.push(Point { x, y });
    }
    let sx = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let sy = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let mut sz = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(-1.0);
    if sz.abs() < 1e-6 {
        sz = if sz.is_sign_negative() { -1e-6 } else { 1e-6 };
    }

    let scale = 10.0;
    let fp = best_footprint(sx, sy, sz, scale);
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in &poly {
        minx = minx.min(p.x);
        maxx = maxx.max(p.x);
        miny = miny.min(p.y);
        maxy = maxy.max(p.y);
    }

    let center = polygon_centroid(&poly);
    let single_ok = poly.iter().all(|p| {
        let dx = p.x - center.x;
        let dy = p.y - center.y;
        (dx * dx + dy * dy).sqrt() <= fp.radius * 0.92
    });
    let mut out: Vec<(f64, f64, f64)> = Vec::new();
    if single_ok {
        out.push((center.x, center.y, 20.0));
    } else {
        let side = (fp.radius * std::f64::consts::SQRT_2 * 0.92).max(0.1);
        let half = side * 0.5;
        let nx = (((maxx - minx) / side).ceil() as i64 + 1).max(1);
        let ny = (((maxy - miny) / side).ceil() as i64 + 1).max(1);
        let dense = nv <= 20_000 && nx.saturating_mul(ny) <= 2_000_000;
        for iy in 0..ny {
            let cy = miny + (iy as f64 + 0.5) * side;
            for ix in 0..nx {
                let cx = minx + (ix as f64 + 0.5) * side;
                let touches = if dense {
                    rect_touches_poly(cx, cy, half, &poly)
                } else {
                    cx >= minx - half && cx <= maxx + half && cy >= miny - half && cy <= maxy + half
                };
                if touches {
                    let idx = out.len() as f64;
                    out.push((cx, cy, 20.0 + idx * 35.0));
                }
            }
        }
    }

    println!("{}", out.len());
    let ax = sx / sz;
    let ay = sy / sz;
    for (shadow_x, shadow_y, pz) in out {
        let px = shadow_x + ax * pz - fp.offset.x;
        let py = shadow_y + ay * pz - fp.offset.y;
        println!(
            "{:.9} {:.9} {:.9} {:.12} {:.12} {:.12} {:.12} {:.9}",
            px, py, pz, fp.quat.w, fp.quat.x, fp.quat.y, fp.quat.z, scale
        );
    }
}
