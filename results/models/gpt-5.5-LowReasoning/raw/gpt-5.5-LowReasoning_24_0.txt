#include <algorithm>
#include <atomic>
#include <cmath>
#include <cstdint>
#include <fstream>
#include <iomanip>
#include <iostream>
#include <limits>
#include <mutex>
#include <string>
#include <thread>

using std::clamp;

static constexpr double PI = 3.141592653589793238462643383279502884;

static double wrapPi(double a) {
    while (a > PI) a -= 2.0 * PI;
    while (a < -PI) a += 2.0 * PI;
    return a;
}

static double sat(double v, double lo, double hi) {
    return std::max(lo, std::min(hi, v));
}

struct TerrainInfo {
    std::atomic<bool> ready{false};
    std::atomic<double> targetGround{0.0};
    std::atomic<double> targetCeiling{std::numeric_limits<double>::infinity()};
};

static uint16_t readLe16(const unsigned char* p) {
    return static_cast<uint16_t>(p[0] | (static_cast<uint16_t>(p[1]) << 8));
}

static void readTargetColumnAsync(
    std::string path, long long width, long long depth, double targetX, double targetY, TerrainInfo* info) {
    const long long tx = clamp(static_cast<long long>(std::llround(targetX)), 0LL, std::max(0LL, width - 1));
    const long long ty = clamp(static_cast<long long>(std::llround(targetY)), 0LL, std::max(0LL, depth - 1));
    const std::uint64_t offset =
        (static_cast<std::uint64_t>(tx) * static_cast<std::uint64_t>(depth) + static_cast<std::uint64_t>(ty)) * 6ULL;

    std::ifstream in(path, std::ios::binary);
    if (!in) return;
    in.seekg(static_cast<std::streamoff>(offset), std::ios::beg);
    unsigned char buf[6] = {};
    in.read(reinterpret_cast<char*>(buf), 6);
    if (in.gcount() != 6) return;

    const uint16_t a = readLe16(buf);
    const uint16_t b = readLe16(buf + 2);
    const uint16_t c = readLe16(buf + 4);
    info->targetGround.store(static_cast<double>(a), std::memory_order_relaxed);
    info->targetCeiling.store((b == c) ? std::numeric_limits<double>::infinity() : static_cast<double>(b),
                              std::memory_order_relaxed);
    info->ready.store(true, std::memory_order_release);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(6);

    double widthD, depthD, heightD, gravity, maxThrust, startFuel;
    double startX, startY, startZ, targetX, targetY, targetZ;
    if (!(std::cin >> widthD >> depthD >> heightD >> gravity >> maxThrust >> startFuel)) return 0;
    if (!(std::cin >> startX >> startY >> startZ >> targetX >> targetY >> targetZ)) return 0;

    std::string terrainPath;
    std::getline(std::cin, terrainPath);
    std::getline(std::cin, terrainPath);

    std::string stateToken;
    std::cin >> stateToken;

    const long long width = static_cast<long long>(std::llround(widthD));
    const long long depth = static_cast<long long>(std::llround(depthD));
    TerrainInfo terrain;
    std::thread terrainThread(readTargetColumnAsync, terrainPath, width, depth, targetX, targetY, &terrain);
    terrainThread.detach();

    const double landRadius = std::max(5.0, std::min(widthD, depthD) * 0.02);
    const double fullAccel = std::max(0.1, maxThrust);
    const double maxTilt = 1.30;
    const double maxHorizAccel = std::max(0.05, std::sin(maxTilt) * fullAccel * 0.92);

    double x, y, z, vx, vy, vz, pitch, yaw, roll, pitchRate, yawRate, rollRate, fuel;
    while (std::cin >> x >> y >> z >> vx >> vy >> vz >> pitch >> yaw >> roll >> pitchRate >> yawRate >> rollRate >> fuel) {
        const double dx = targetX - x;
        const double dy = targetY - y;
        const double dz = targetZ - z;
        const double distH = std::hypot(dx, dy);
        const double speedH = std::hypot(vx, vy);
        const double alt = std::max(0.0, z - targetZ);

        double ground = targetZ;
        if (terrain.ready.load(std::memory_order_acquire)) {
            ground = std::max(ground, terrain.targetGround.load(std::memory_order_relaxed));
        }

        const double worldScale = std::max({widthD, depthD, heightD, 100.0});
        const double cruiseClearance = sat(std::max(40.0, 0.16 * distH), 20.0, std::max(60.0, 0.25 * worldScale));
        const double desiredZHigh = std::min(heightD - 5.0, std::max(targetZ + 12.0, ground + cruiseClearance));

        double vDesX = 0.0;
        double vDesY = 0.0;
        if (distH > 1e-6) {
            const double stoppingTime = std::sqrt(std::max(0.0, 2.0 * distH / maxHorizAccel));
            const double tGo = sat(stoppingTime + 1.5, 2.5, 90.0);
            double maxVH = sat(0.55 * distH + 4.0, 4.0, std::max(12.0, 0.012 * worldScale));
            if (distH < 4.0 * landRadius) maxVH = sat(0.45 * distH + 0.8, 0.8, 8.0);
            const double vh = std::min(maxVH, distH / tGo);
            vDesX = dx / distH * vh;
            vDesY = dy / distH * vh;
        }

        double ax = sat((vDesX - vx) * 0.85 + dx * 0.018, -maxHorizAccel, maxHorizAccel);
        double ay = sat((vDesY - vy) * 0.85 + dy * 0.018, -maxHorizAccel, maxHorizAccel);
        const double aH = std::hypot(ax, ay);
        if (aH > maxHorizAccel) {
            ax *= maxHorizAccel / aH;
            ay *= maxHorizAccel / aH;
        }

        double vDesZ;
        if (distH > 2.5 * landRadius) {
            const double climbError = desiredZHigh - z;
            vDesZ = sat(0.10 * climbError, -10.0, 12.0);
            if (z < targetZ + 25.0) vDesZ = std::max(vDesZ, 3.0);
        } else {
            vDesZ = -sat(std::sqrt(std::max(0.0, 2.0 * gravity * std::max(0.0, alt))) * 0.55, 0.45, 18.0);
            if (alt < 20.0) vDesZ = -sat(0.16 * alt + 0.25, 0.25, 2.2);
            if (alt < 6.0) vDesZ = -0.35;
        }

        double az = gravity + sat((vDesZ - vz) * 0.95 + dz * 0.015, -0.65 * fullAccel, 0.90 * fullAccel);
        az = sat(az, 0.05, fullAccel * 0.98);

        double desiredPitch = 0.0;
        double desiredYaw = yaw;
        const double desiredH = std::hypot(ax, ay);
        if (desiredH > 1e-5) {
            desiredPitch = sat(std::atan2(desiredH, std::max(0.1, az)), 0.0, maxTilt);
            desiredYaw = std::atan2(ay, ax) + PI;
        }

        const double thrustAccel = std::sqrt(ax * ax + ay * ay + az * az);
        double thrust = sat(thrustAccel / fullAccel, 0.0, 1.0);
        if (fuel <= 0.01) thrust = 0.0;

        const double pitchErr = wrapPi(desiredPitch - pitch);
        const double yawErr = wrapPi(desiredYaw - yaw);
        const double rollErr = wrapPi(0.0 - roll);

        const double pitchCmd = sat((5.0 * pitchErr - 2.0 * pitchRate) / 2.0, -1.0, 1.0);
        const double yawCmd = sat((4.0 * yawErr - 1.8 * yawRate) / 2.0, -1.0, 1.0);
        const double rollCmd = sat((3.0 * rollErr - 1.5 * rollRate) / 2.0, -1.0, 1.0);

        std::cout << thrust << ' ' << pitchCmd << ' ' << yawCmd << ' ' << rollCmd << '\n';
        std::cout.flush();
    }

    return 0;
}
