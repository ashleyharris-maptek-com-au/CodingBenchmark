use std::cmp::min;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read, Write};

#[derive(Clone, Copy)]
struct Pt {
    x: f64,
    y: f64,
}

struct Dsu {
    p: Vec<usize>,
    sz: Vec<usize>,
}

impl Dsu {
    fn new(n: usize) -> Self {
        let mut p = Vec::with_capacity(n);
        for i in 0..n {
            p.push(i);
        }
        Self { p, sz: vec![1; n] }
    }
    fn find(&mut self, x: usize) -> usize {
        let mut r = x;
        while self.p[r] != r {
            r = self.p[r];
        }
        let mut v = x;
        while self.p[v] != v {
            let nxt = self.p[v];
            self.p[v] = r;
            v = nxt;
        }
        r
    }
    fn union(&mut self, a: usize, b: usize) {
        let mut ra = self.find(a);
        let mut rb = self.find(b);
        if ra == rb {
            return;
        }
        if self.sz[ra] < self.sz[rb] {
            std::mem::swap(&mut ra, &mut rb);
        }
        self.p[rb] = ra;
        self.sz[ra] += self.sz[rb];
    }
}

fn hash01(mut x: u64) -> f64 {
    x ^= x >> 33;
    x = x.wrapping_mul(0xff51afd7ed558ccd);
    x ^= x >> 33;
    x = x.wrapping_mul(0xc4ceb9fe1a85ec53);
    x ^= x >> 33;
    ((x >> 11) as f64) * (1.0 / ((1u64 << 53) as f64))
}

fn bfs_order(nodes: &[usize], adj: &[Vec<usize>]) -> (Vec<usize>, Vec<usize>) {
    if nodes.is_empty() {
        return (Vec::new(), Vec::new());
    }
    let n_all = adj.len();
    let mut mark = vec![false; n_all];
    for &v in nodes {
        mark[v] = true;
    }
    let start = *nodes.iter().max_by_key(|&&v| adj[v].len()).unwrap();
    let mut seen = vec![false; n_all];
    let mut depth = vec![0usize; n_all];
    let mut order = Vec::with_capacity(nodes.len());
    let mut q = VecDeque::new();
    seen[start] = true;
    q.push_back(start);
    while let Some(v) = q.pop_front() {
        order.push(v);
        let mut ns: Vec<usize> = adj[v].iter().copied().filter(|&u| mark[u] && !seen[u]).collect();
        ns.sort_unstable_by_key(|&u| std::cmp::Reverse(adj[u].len()));
        for u in ns {
            seen[u] = true;
            depth[u] = depth[v] + 1;
            q.push_back(u);
        }
    }
    for &v in nodes {
        if !seen[v] {
            seen[v] = true;
            order.push(v);
        }
    }
    (order, depth)
}

fn layout_small_component(nodes: &[usize], adj: &[Vec<usize>], pos: &mut [Pt]) {
    let k = nodes.len();
    if k == 1 {
        pos[nodes[0]] = Pt { x: 0.0, y: 0.0 };
        return;
    }
    let (order, depth) = bfs_order(nodes, adj);
    let max_d = order.iter().map(|&v| depth[v]).max().unwrap_or(1).max(1) as f64;
    for (i, &v) in order.iter().enumerate() {
        let a = 2.0 * PI * (i as f64) / (k as f64);
        let r = 25.0 + 18.0 * (depth[v] as f64) / max_d + 8.0 * hash01(v as u64 + 17);
        pos[v] = Pt { x: r * a.cos(), y: r * a.sin() };
    }

    let iterations = if k <= 20 { 260 } else if k <= 100 { 160 } else { 80 };
    let area = (k as f64).sqrt() * 55.0;
    let ideal = (area * area / k as f64).sqrt();
    let node_mark: std::collections::HashSet<usize> = nodes.iter().copied().collect();
    let mut edges = Vec::new();
    for &v in nodes {
        for &u in &adj[v] {
            if v < u && node_mark.contains(&u) {
                edges.push((v, u));
            }
        }
    }
    for it in 0..iterations {
        let temp = ideal * (1.0 - it as f64 / iterations as f64).max(0.05);
        let mut disp = vec![Pt { x: 0.0, y: 0.0 }; k];
        for i in 0..k {
            for j in i + 1..k {
                let a = nodes[i];
                let b = nodes[j];
                let mut dx = pos[a].x - pos[b].x;
                let mut dy = pos[a].y - pos[b].y;
                let d2 = dx * dx + dy * dy + 0.01;
                let d = d2.sqrt();
                dx /= d;
                dy /= d;
                let f = ideal * ideal / d;
                disp[i].x += dx * f;
                disp[i].y += dy * f;
                disp[j].x -= dx * f;
                disp[j].y -= dy * f;
            }
        }
        for &(a, b) in &edges {
            let ia = nodes.iter().position(|&v| v == a).unwrap();
            let ib = nodes.iter().position(|&v| v == b).unwrap();
            let mut dx = pos[a].x - pos[b].x;
            let mut dy = pos[a].y - pos[b].y;
            let d = (dx * dx + dy * dy + 0.01).sqrt();
            dx /= d;
            dy /= d;
            let f = d * d / ideal;
            disp[ia].x -= dx * f;
            disp[ia].y -= dy * f;
            disp[ib].x += dx * f;
            disp[ib].y += dy * f;
        }
        for (i, &v) in nodes.iter().enumerate() {
            let d = (disp[i].x * disp[i].x + disp[i].y * disp[i].y).sqrt();
            if d > 1e-9 {
                let step = temp.min(d) / d;
                pos[v].x += disp[i].x * step;
                pos[v].y += disp[i].y * step;
            }
        }
    }
}

fn layout_large_component(nodes: &[usize], adj: &[Vec<usize>], pos: &mut [Pt]) {
    let k = nodes.len();
    let (order, depth) = bfs_order(nodes, adj);
    let max_d = order.iter().map(|&v| depth[v]).max().unwrap_or(1).max(1) as f64;
    let golden = PI * (3.0 - 5.0_f64.sqrt());
    for (i, &v) in order.iter().enumerate() {
        let t = i as f64 + 0.5;
        let a = t * golden;
        let shell = depth[v] as f64 / max_d;
        let r = 8.0 * (t.sqrt()) + 60.0 * shell + 6.0 * hash01((v as u64).wrapping_mul(1315423911));
        pos[v] = Pt { x: r * a.cos(), y: r * a.sin() };
    }

    if k <= 5000 {
        let rounds = if k <= 1200 { 24 } else { 10 };
        let alpha0 = 0.18;
        let mut idx = vec![usize::MAX; adj.len()];
        for (i, &v) in nodes.iter().enumerate() {
            idx[v] = i;
        }
        let mut next = vec![Pt { x: 0.0, y: 0.0 }; k];
        for r in 0..rounds {
            let alpha = alpha0 * (1.0 - r as f64 / rounds as f64);
            for (i, &v) in nodes.iter().enumerate() {
                let mut sx = pos[v].x;
                let mut sy = pos[v].y;
                let mut cnt = 1.0;
                for &u in &adj[v] {
                    if idx[u] != usize::MAX {
                        sx += pos[u].x;
                        sy += pos[u].y;
                        cnt += 1.0;
                    }
                }
                let cx = sx / cnt;
                let cy = sy / cnt;
                next[i] = Pt {
                    x: pos[v].x * (1.0 - alpha) + cx * alpha,
                    y: pos[v].y * (1.0 - alpha) + cy * alpha,
                };
            }
            for (i, &v) in nodes.iter().enumerate() {
                pos[v] = next[i];
            }
        }
    }
}

fn pack_components(components: &[Vec<usize>], pos: &mut [Pt]) {
    let mut comps: Vec<&Vec<usize>> = components.iter().collect();
    comps.sort_by_key(|c| std::cmp::Reverse(c.len()));
    let total = pos.len().max(1) as f64;
    let gap = 80.0;
    let mut x_cursor = 0.0;
    let mut y_cursor = 0.0;
    let mut row_h = 0.0;
    let target_w = 240.0 * total.sqrt();
    for comp in comps {
        let mut minx = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for &v in comp {
            minx = minx.min(pos[v].x);
            maxx = maxx.max(pos[v].x);
            miny = miny.min(pos[v].y);
            maxy = maxy.max(pos[v].y);
        }
        let w = (maxx - minx).max(20.0);
        let h = (maxy - miny).max(20.0);
        if x_cursor > 0.0 && x_cursor + w > target_w {
            x_cursor = 0.0;
            y_cursor += row_h + gap;
            row_h = 0.0;
        }
        let dx = x_cursor - minx;
        let dy = y_cursor - miny;
        for &v in comp {
            pos[v].x += dx;
            pos[v].y += dy;
        }
        x_cursor += w + gap;
        row_h = row_h.max(h);
    }
}

fn main() {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();
    let n: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let m: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let mut dsu = Dsu::new(n);
    let mut edges = Vec::with_capacity(min(m, 4_000_000));
    let mut deg = vec![0usize; n];
    for _ in 0..m {
        let Some(a) = it.next() else { break };
        let Some(b) = it.next() else { break };
        let u: usize = a.parse().unwrap_or(n);
        let v: usize = b.parse().unwrap_or(n);
        if u >= n || v >= n || u == v {
            continue;
        }
        dsu.union(u, v);
        if edges.len() < 4_000_000 {
            edges.push((u, v));
            deg[u] += 1;
            deg[v] += 1;
        }
    }

    let mut adj = vec![Vec::<usize>::new(); n];
    for &(u, v) in &edges {
        adj[u].push(v);
        adj[v].push(u);
    }
    for v in 0..n {
        if adj[v].len() > 512 && n > 50_000 {
            adj[v].sort_unstable_by_key(|&u| std::cmp::Reverse(deg[u]));
            adj[v].truncate(512);
        }
    }

    let mut by_root: Vec<Vec<usize>> = Vec::new();
    let mut root_to_id = vec![usize::MAX; n];
    for v in 0..n {
        let r = dsu.find(v);
        if root_to_id[r] == usize::MAX {
            root_to_id[r] = by_root.len();
            by_root.push(Vec::new());
        }
        by_root[root_to_id[r]].push(v);
    }

    let mut pos = vec![Pt { x: 0.0, y: 0.0 }; n];
    for comp in &by_root {
        if comp.len() <= 180 {
            layout_small_component(comp, &adj, &mut pos);
        } else {
            layout_large_component(comp, &adj, &mut pos);
        }
    }
    pack_components(&by_root, &mut pos);

    let mut out = String::with_capacity(n * 28);
    for p in pos {
        out.push_str(&format!("{:.6} {:.6}\n", p.x, p.y));
    }
    let mut w = io::BufWriter::new(io::stdout());
    w.write_all(out.as_bytes()).unwrap();
}
