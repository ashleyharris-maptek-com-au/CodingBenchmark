// Graph coloring solver, C++17, standard library only.
//
// Reads:
//   n m k
//   u v
// Writes one color per vertex in [0, k).
//
// The program is deliberately adaptive:
//   * bipartite BFS for easy 2-colorable cases,
//   * degeneracy ordering for large/sparse graphs,
//   * DSATUR-style constructive coloring for smaller or denser graphs,
//   * small-instance exact backtracking when the heuristic cannot prove success,
//   * bounded min-conflicts repair as a last resort.
//
// No polynomial algorithm can guarantee success for all inputs unless P=NP, but
// this mix handles the common benchmark spectrum well, including very large
// sparse graphs with dozens of colors.

#include <algorithm>
#include <chrono>
#include <cstdint>
#include <functional>
#include <iostream>
#include <numeric>
#include <queue>
#include <random>
#include <set>
#include <utility>
#include <vector>

using namespace std;

struct Timer {
    chrono::steady_clock::time_point start = chrono::steady_clock::now();
    double seconds() const {
        return chrono::duration<double>(chrono::steady_clock::now() - start).count();
    }
};

struct XorShift64 {
    uint64_t x;
    explicit XorShift64(uint64_t seed = 88172645463393265ull) : x(seed ? seed : 1) {}
    uint64_t next() {
        x ^= x << 7;
        x ^= x >> 9;
        return x;
    }
    int next_int(int n) { return int(next() % uint64_t(n)); }
};

static bool verify_coloring(const vector<int>& color,
                            const vector<pair<int, int>>& edges,
                            int k) {
    for (int c : color) {
        if (c < 0 || c >= k) return false;
    }
    for (const auto& e : edges) {
        if (color[e.first] == color[e.second]) return false;
    }
    return true;
}

static void print_coloring(const vector<int>& color) {
    for (size_t i = 0; i < color.size(); ++i) {
        if (i) cout << ' ';
        cout << color[i];
    }
    cout << '\n';
}

static bool bipartite_color(const vector<vector<int>>& adj, int k, vector<int>& color) {
    if (k < 2) return adj.empty() || all_of(adj.begin(), adj.end(), [](const vector<int>& v) { return v.empty(); });
    int n = (int)adj.size();
    color.assign(n, -1);
    queue<int> q;
    for (int s = 0; s < n; ++s) {
        if (color[s] != -1) continue;
        color[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (int u : adj[v]) {
                if (color[u] < 0) {
                    color[u] = color[v] ^ 1;
                    q.push(u);
                } else if (color[u] == color[v]) {
                    color.clear();
                    return false;
                }
            }
        }
    }
    return true;
}

static vector<int> degeneracy_order(const vector<vector<int>>& adj) {
    int n = (int)adj.size();
    vector<int> deg(n), order;
    order.reserve(n);
    int maxd = 0;
    for (int i = 0; i < n; ++i) {
        deg[i] = (int)adj[i].size();
        maxd = max(maxd, deg[i]);
    }
    vector<vector<int>> buckets(maxd + 1);
    for (int i = 0; i < n; ++i) buckets[deg[i]].push_back(i);
    vector<char> removed(n, 0);
    int cur = 0;
    for (int taken = 0; taken < n; ++taken) {
        while (cur <= maxd) {
            while (!buckets[cur].empty() && (removed[buckets[cur].back()] || deg[buckets[cur].back()] != cur)) {
                buckets[cur].pop_back();
            }
            if (!buckets[cur].empty()) break;
            ++cur;
        }
        if (cur > maxd) break;
        int v = buckets[cur].back();
        buckets[cur].pop_back();
        if (removed[v]) {
            --taken;
            continue;
        }
        removed[v] = 1;
        order.push_back(v);
        for (int u : adj[v]) {
            if (!removed[u]) {
                int nd = --deg[u];
                buckets[nd].push_back(u);
                if (nd < cur) cur = nd;
            }
        }
    }
    return order;
}

static bool greedy_in_order(const vector<vector<int>>& adj,
                            const vector<int>& order,
                            int k,
                            vector<int>& color) {
    int n = (int)adj.size();
    color.assign(n, -1);
    vector<int> mark(k, -1);
    int stamp = 0;
    for (int v : order) {
        ++stamp;
        for (int u : adj[v]) {
            int c = color[u];
            if (c >= 0) mark[c] = stamp;
        }
        int best = -1;
        for (int c = 0; c < k; ++c) {
            if (mark[c] != stamp) {
                best = c;
                break;
            }
        }
        if (best < 0) return false;
        color[v] = best;
    }
    return true;
}

static bool dsatur_greedy(const vector<vector<int>>& adj, int k, vector<int>& color, double time_limit, const Timer& timer) {
    int n = (int)adj.size();
    color.assign(n, -1);
    vector<int> sat(n, 0), deg(n), forbidden_stamp(k, -1), colored_neighbors(n, 0);
    vector<vector<unsigned char>> seen;
    bool keep_seen = (1LL * n * k <= 12000000LL);
    if (keep_seen) seen.assign(n, vector<unsigned char>(k, 0));
    for (int i = 0; i < n; ++i) deg[i] = (int)adj[i].size();

    for (int step = 0; step < n; ++step) {
        if ((step & 4095) == 0 && timer.seconds() > time_limit) return false;
        int v = -1;
        for (int i = 0; i < n; ++i) {
            if (color[i] >= 0) continue;
            if (v < 0 || sat[i] > sat[v] || (sat[i] == sat[v] && deg[i] > deg[v])) v = i;
        }
        if (v < 0) return true;

        int stamp = step + 1;
        for (int u : adj[v]) if (color[u] >= 0) forbidden_stamp[color[u]] = stamp;

        int best_c = -1, best_score = -1;
        for (int c = 0; c < k; ++c) {
            if (forbidden_stamp[c] == stamp) continue;
            int score = 0;
            for (int u : adj[v]) {
                if (color[u] < 0) {
                    if (keep_seen) score += !seen[u][c];
                    else score += 1;
                }
            }
            if (score > best_score) {
                best_score = score;
                best_c = c;
            }
        }
        if (best_c < 0) return false;
        color[v] = best_c;

        for (int u : adj[v]) {
            if (color[u] < 0) {
                ++colored_neighbors[u];
                if (keep_seen && !seen[u][best_c]) {
                    seen[u][best_c] = 1;
                    ++sat[u];
                } else if (!keep_seen) {
                    int cnt = 0;
                    vector<char> local(k, 0);
                    for (int w : adj[u]) if (color[w] >= 0 && !local[color[w]]) {
                        local[color[w]] = 1;
                        ++cnt;
                    }
                    sat[u] = cnt;
                }
            }
        }
    }
    return true;
}

static bool exact_backtrack(const vector<vector<int>>& adj, int k, vector<int>& color, double time_limit, const Timer& timer) {
    int n = (int)adj.size();
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) { return adj[a].size() > adj[b].size(); });
    color.assign(n, -1);
    vector<int> mark(k, -1);

    function<bool(int)> dfs = [&](int depth) -> bool {
        if ((depth & 255) == 0 && timer.seconds() > time_limit) return false;
        if (depth == n) return true;
        int best_pos = -1, best_avail = k + 1, best_sat = -1;
        for (int pos = 0; pos < n; ++pos) {
            int v = order[pos];
            if (color[v] >= 0) continue;
            int stamp = depth * n + pos + 1;
            int sat = 0, avail = 0;
            for (int u : adj[v]) if (color[u] >= 0 && mark[color[u]] != stamp) {
                mark[color[u]] = stamp;
                ++sat;
            }
            avail = k - sat;
            if (avail < best_avail || (avail == best_avail && sat > best_sat)) {
                best_avail = avail;
                best_sat = sat;
                best_pos = pos;
                if (avail == 0) break;
            }
        }
        if (best_pos < 0) return true;
        if (best_avail <= 0) return false;
        int v = order[best_pos];
        int stamp = depth + 1000003;
        for (int u : adj[v]) if (color[u] >= 0) mark[color[u]] = stamp;
        for (int c = 0; c < k; ++c) {
            if (mark[c] == stamp) continue;
            color[v] = c;
            if (dfs(depth + 1)) return true;
            color[v] = -1;
        }
        return false;
    };
    return dfs(0);
}

static bool min_conflicts(const vector<vector<int>>& adj,
                          const vector<pair<int, int>>& edges,
                          int k,
                          vector<int>& color,
                          double time_limit,
                          const Timer& timer) {
    int n = (int)adj.size();
    if (n == 0) return true;
    XorShift64 rng(1469598103934665603ull ^ (uint64_t)n ^ ((uint64_t)edges.size() << 32));
    if ((int)color.size() != n) {
        color.resize(n);
        for (int i = 0; i < n; ++i) color[i] = rng.next_int(k);
    }
    for (int& c : color) if (c < 0 || c >= k) c = rng.next_int(k);

    vector<int> bad_vertices;
    vector<char> is_bad(n, 0);
    auto rebuild_bad = [&]() {
        bad_vertices.clear();
        fill(is_bad.begin(), is_bad.end(), 0);
        for (const auto& e : edges) {
            if (color[e.first] == color[e.second]) {
                if (!is_bad[e.first]) {
                    is_bad[e.first] = 1;
                    bad_vertices.push_back(e.first);
                }
                if (!is_bad[e.second]) {
                    is_bad[e.second] = 1;
                    bad_vertices.push_back(e.second);
                }
            }
        }
    };

    vector<int> cnt(k, 0);
    rebuild_bad();
    int iter = 0;
    while (!bad_vertices.empty() && timer.seconds() < time_limit) {
        if ((iter++ & 4095) == 0) rebuild_bad();
        if (bad_vertices.empty()) break;
        int v = bad_vertices[rng.next_int((int)bad_vertices.size())];
        if (!is_bad[v]) continue;

        fill(cnt.begin(), cnt.end(), 0);
        for (int u : adj[v]) ++cnt[color[u]];
        int best = color[v], best_conf = cnt[best];
        int start = rng.next_int(k);
        for (int t = 0; t < k; ++t) {
            int c = (start + t) % k;
            if (cnt[c] < best_conf) {
                best_conf = cnt[c];
                best = c;
                if (best_conf == 0) break;
            }
        }
        if (best != color[v] || (rng.next() & 31ull) == 0) {
            color[v] = best;
            for (int u : adj[v]) {
                bool bad = false;
                for (int w : adj[u]) {
                    if (color[w] == color[u]) {
                        bad = true;
                        break;
                    }
                }
                is_bad[u] = bad;
            }
            bool bad = false;
            for (int u : adj[v]) {
                if (color[u] == color[v]) {
                    bad = true;
                    break;
                }
            }
            is_bad[v] = bad;
        }
    }
    return verify_coloring(color, edges, k);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    long long m_input;
    if (!(cin >> n >> m_input >> k)) return 0;
    if (n < 0 || k <= 0 || m_input < 0) return 0;

    vector<pair<int, int>> edges;
    edges.reserve((size_t)min<long long>(m_input, 200000000LL));
    vector<int> degree(n, 0);
    bool impossible = false;
    for (long long i = 0; i < m_input; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            impossible = true;
            continue;
        }
        edges.push_back({u, v});
        ++degree[u];
        ++degree[v];
    }

    vector<vector<int>> adj(n);
    for (int i = 0; i < n; ++i) adj[i].reserve(degree[i]);
    for (const auto& e : edges) {
        adj[e.first].push_back(e.second);
        adj[e.second].push_back(e.first);
    }

    Timer timer;
    const double total_limit = 285.0;
    vector<int> color;

    if (n == 0) {
        cout << '\n';
        return 0;
    }

    if (!impossible && edges.empty()) {
        color.assign(n, 0);
        print_coloring(color);
        return 0;
    }

    if (!impossible && k >= 2 && bipartite_color(adj, k, color) && verify_coloring(color, edges, k)) {
        print_coloring(color);
        return 0;
    }

    vector<int> order = degeneracy_order(adj);
    reverse(order.begin(), order.end());
    if (!impossible && greedy_in_order(adj, order, k, color) && verify_coloring(color, edges, k)) {
        print_coloring(color);
        return 0;
    }

    if (!impossible && n <= 200000 && dsatur_greedy(adj, k, color, min(120.0, total_limit), timer) &&
        verify_coloring(color, edges, k)) {
        print_coloring(color);
        return 0;
    }

    if (!impossible && n <= 70 && exact_backtrack(adj, k, color, min(240.0, total_limit), timer) &&
        verify_coloring(color, edges, k)) {
        print_coloring(color);
        return 0;
    }

    if (!impossible && min_conflicts(adj, edges, k, color, total_limit, timer)) {
        print_coloring(color);
        return 0;
    }

    // Last ditch: output a complete assignment in the required range. For truly
    // impossible or time-exhausted inputs this may be rejected, but it preserves
    // the output contract instead of hanging or emitting partial data.
    if ((int)color.size() != n) color.assign(n, 0);
    for (int i = 0; i < n; ++i) {
        if (color[i] < 0 || color[i] >= k) color[i] = i % k;
    }
    print_coloring(color);
    return 0;
}
