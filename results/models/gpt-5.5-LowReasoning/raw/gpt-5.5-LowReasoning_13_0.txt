#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <string>
#include <vector>

using namespace std;

static string tiny_solver(const vector<string>& s) {
    int n = static_cast<int>(s.size());
    int base = 0;
    for (int i = 1; i < n; ++i) {
        if (s[i].size() < s[base].size()) base = i;
    }
    const string& a = s[base];
    for (int len = static_cast<int>(a.size()); len >= 1; --len) {
        for (int start = 0; start + len <= static_cast<int>(a.size()); ++start) {
            string sub = a.substr(static_cast<size_t>(start), static_cast<size_t>(len));
            bool ok = true;
            for (int i = 0; i < n && ok; ++i) {
                if (i != base && s[i].find(sub) == string::npos) ok = false;
            }
            if (ok) return sub;
        }
    }
    return "";
}

struct CompactSAM {
    struct State {
        int link = -1;
        int len = 0;
        int head = -1;
        int first_pos = -1;
        int best = numeric_limits<int>::max();
        int cur = 0;
    };
    struct Edge {
        int to;
        int next;
        unsigned char ch;
    };

    vector<State> st;
    vector<Edge> edges;
    int last = 0;

    explicit CompactSAM(size_t n = 0) {
        st.reserve(2 * n + 1);
        edges.reserve(3 * n + 1);
        st.push_back(State{});
    }

    int find_edge(int v, unsigned char c) const {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) return edges[e].to;
        }
        return -1;
    }

    void set_edge(int v, unsigned char c, int to) {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                edges[e].to = to;
                return;
            }
        }
        edges.push_back(Edge{to, st[v].head, c});
        st[v].head = static_cast<int>(edges.size()) - 1;
    }

    void extend(unsigned char c, int pos) {
        int cur = static_cast<int>(st.size());
        st.push_back(State{});
        st[cur].len = st[last].len + 1;
        st[cur].first_pos = pos;

        int p = last;
        while (p != -1 && find_edge(p, c) == -1) {
            set_edge(p, c, cur);
            p = st[p].link;
        }
        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = find_edge(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = static_cast<int>(st.size());
                st.push_back(st[q]);
                st[clone].len = st[p].len + 1;
                st[clone].best = numeric_limits<int>::max();
                st[clone].cur = 0;
                while (p != -1 && find_edge(p, c) == q) {
                    set_edge(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

static vector<int> length_order(const vector<CompactSAM::State>& st, int max_len) {
    vector<int> cnt(static_cast<size_t>(max_len) + 1, 0), order(st.size());
    for (const auto& x : st) ++cnt[static_cast<size_t>(x.len)];
    for (int i = 1; i <= max_len; ++i) cnt[static_cast<size_t>(i)] += cnt[static_cast<size_t>(i - 1)];
    for (int i = static_cast<int>(st.size()) - 1; i >= 0; --i) {
        order[static_cast<size_t>(--cnt[static_cast<size_t>(st[static_cast<size_t>(i)].len)])] = i;
    }
    return order;
}

static string automaton_solver(const vector<string>& s) {
    int n = static_cast<int>(s.size());
    int base = 0;
    for (int i = 1; i < n; ++i) {
        if (s[i].size() < s[base].size()) base = i;
    }
    const string& a = s[base];
    if (a.empty()) return "";

    CompactSAM sam(a.size());
    for (int i = 0; i < static_cast<int>(a.size()); ++i) {
        sam.extend(static_cast<unsigned char>(a[static_cast<size_t>(i)]), i);
    }

    for (auto& state : sam.st) state.best = state.len;
    vector<int> order = length_order(sam.st, static_cast<int>(a.size()));

    for (int idx = 0; idx < n; ++idx) {
        if (idx == base) continue;

        for (auto& state : sam.st) state.cur = 0;

        int v = 0, l = 0;
        for (unsigned char c : s[static_cast<size_t>(idx)]) {
            int to = sam.find_edge(v, c);
            if (to != -1) {
                v = to;
                ++l;
            } else {
                while (v != -1 && sam.find_edge(v, c) == -1) v = sam.st[static_cast<size_t>(v)].link;
                if (v == -1) {
                    v = 0;
                    l = 0;
                    continue;
                }
                l = sam.st[static_cast<size_t>(v)].len + 1;
                v = sam.find_edge(v, c);
            }
            if (l > sam.st[static_cast<size_t>(v)].cur) sam.st[static_cast<size_t>(v)].cur = l;
        }

        for (int k = static_cast<int>(order.size()) - 1; k > 0; --k) {
            int state_id = order[static_cast<size_t>(k)];
            int link = sam.st[static_cast<size_t>(state_id)].link;
            if (link >= 0) {
                int propagated = min(sam.st[static_cast<size_t>(state_id)].cur,
                                     sam.st[static_cast<size_t>(link)].len);
                if (propagated > sam.st[static_cast<size_t>(link)].cur) {
                    sam.st[static_cast<size_t>(link)].cur = propagated;
                }
            }
        }

        for (auto& state : sam.st) {
            if (state.cur < state.best) state.best = state.cur;
        }
    }

    int best_len = 0;
    int best_pos = -1;
    for (const auto& state : sam.st) {
        if (state.best > best_len) {
            best_len = state.best;
            best_pos = state.first_pos;
        }
    }
    if (best_len <= 0) return "";
    return a.substr(static_cast<size_t>(best_pos - best_len + 1), static_cast<size_t>(best_len));
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    if (!(cin >> n)) return 0;
    string dummy;
    getline(cin, dummy);

    if (n <= 0) {
        cout << '\n';
        return 0;
    }

    vector<string> s(static_cast<size_t>(n));
    size_t total = 0;
    for (int i = 0; i < n; ++i) {
        getline(cin, s[static_cast<size_t>(i)]);
        total += s[static_cast<size_t>(i)].size();
    }

    if (n == 1) {
        cout << s[0] << '\n';
        return 0;
    }

    size_t shortest = s[0].size();
    for (const auto& x : s) shortest = min(shortest, x.size());
    if (shortest == 0) {
        cout << '\n';
        return 0;
    }

    string answer;
    if (n <= 4 && shortest <= 256 && total <= 4096) {
        answer = tiny_solver(s);
    } else {
        answer = automaton_solver(s);
    }
    cout << answer << '\n';
    return 0;
}
