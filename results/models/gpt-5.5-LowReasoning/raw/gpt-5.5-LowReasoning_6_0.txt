using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    struct Vec
    {
        public double X, Y, Z;
        public Vec(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec operator +(Vec a, Vec b) => new Vec(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec operator -(Vec a, Vec b) => new Vec(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec operator *(Vec a, double s) => new Vec(a.X * s, a.Y * s, a.Z * s);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public static double Dot(Vec a, Vec b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec Cross(Vec a, Vec b) =>
            new Vec(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    sealed class Orbit
    {
        public Vec R, V, H, E;
        public double Rn, Vn, Hn, Energy, A, Ecc, Inc, SpeedCirc, Period, RadiusRank;
    }

    static int N;
    static double Mu;
    static Orbit Start;
    static Orbit[] Stations;
    static double[,] Cost;
    static readonly Random Rng = new Random(1234567);

    public static void Main()
    {
        var tokens = Console.In.ReadToEnd()
            .Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (tokens.Length < 8) return;
        int p = 0;
        N = int.Parse(tokens[p++], CultureInfo.InvariantCulture);
        Mu = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        Start = ReadOrbit(tokens, ref p);
        Stations = new Orbit[N];
        for (int i = 0; i < N; i++) Stations[i] = ReadOrbit(tokens, ref p);

        if (N <= 0) { Console.WriteLine(); return; }
        BuildCosts();

        int[] order;
        if (N <= 11) order = SolveExact();
        else order = SolveHeuristic();

        Console.WriteLine(string.Join(" ", order));
    }

    static Orbit ReadOrbit(string[] t, ref int p)
    {
        var r = new Vec(Parse(t[p++]), Parse(t[p++]), Parse(t[p++]));
        var v = new Vec(Parse(t[p++]), Parse(t[p++]), Parse(t[p++]));
        var o = new Orbit { R = r, V = v };
        o.Rn = r.Norm();
        o.Vn = v.Norm();
        o.H = Vec.Cross(r, v);
        o.Hn = o.H.Norm();
        o.Energy = 0.5 * o.Vn * o.Vn - Mu / Math.Max(o.Rn, 1e-9);
        o.A = o.Energy < -1e-9 ? -Mu / (2.0 * o.Energy) : o.Rn;
        o.E = (Vec.Cross(v, o.H) * (1.0 / Mu)) - (r * (1.0 / Math.Max(o.Rn, 1e-9)));
        o.Ecc = Math.Min(2.0, o.E.Norm());
        o.Inc = Math.Acos(Clamp(o.H.Z / Math.Max(o.Hn, 1e-9), -1.0, 1.0));
        o.SpeedCirc = Math.Sqrt(Mu / Math.Max(o.Rn, 1e-9));
        o.Period = o.A > 0 ? 2.0 * Math.PI * Math.Sqrt(o.A * o.A * o.A / Mu) : 1e9;
        o.RadiusRank = 0.65 * o.A + 0.35 * o.Rn;
        return o;
    }

    static double Parse(string s) => double.Parse(s, CultureInfo.InvariantCulture);
    static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);

    static void BuildCosts()
    {
        Cost = new double[N + 1, N];
        for (int j = 0; j < N; j++) Cost[0, j] = TransferCost(Start, Stations[j]);
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                Cost[i + 1, j] = i == j ? 1e100 : TransferCost(Stations[i], Stations[j]);
    }

    static double TransferCost(Orbit a, Orbit b)
    {
        double r1 = Math.Max(a.Rn, 1.0), r2 = Math.Max(b.Rn, 1.0);
        double sma1 = Math.Max(a.A, 1.0), sma2 = Math.Max(b.A, 1.0);
        double vc1 = Math.Sqrt(Mu / r1), vc2 = Math.Sqrt(Mu / r2);
        double at = Math.Max((r1 + r2) * 0.5, 1.0);
        double dv1 = Math.Abs(Math.Sqrt(Mu * (2.0 / r1 - 1.0 / at)) - vc1);
        double dv2 = Math.Abs(vc2 - Math.Sqrt(Mu * (2.0 / r2 - 1.0 / at)));
        double hohmann = dv1 + dv2;

        double cosPlane = Clamp(Vec.Dot(a.H, b.H) / Math.Max(a.Hn * b.Hn, 1e-9), -1.0, 1.0);
        double planeAngle = Math.Acos(cosPlane);
        double plane = 2.0 * Math.Min(a.Vn, b.Vn) * Math.Sin(planeAngle * 0.5);

        double vel = (a.V - b.V).Norm();
        double posChord = (a.R - b.R).Norm();
        double radial = Math.Abs(sma1 - sma2) / Math.Max((sma1 + sma2) * 0.5, 1.0);
        double ecc = Math.Abs(a.Ecc - b.Ecc);
        double phase = Math.Min(posChord / Math.Max((r1 + r2) * 0.5, 1.0), 2.0);
        double period = Math.Abs(a.Period - b.Period) / Math.Max(Math.Max(a.Period, b.Period), 1.0);

        return 2.6 * hohmann + 1.2 * plane + 0.18 * vel + 0.10 * phase + 0.20 * radial + 0.12 * ecc + 0.05 * period;
    }

    static int[] SolveExact()
    {
        int masks = 1 << N;
        double[,] dp = new double[masks, N];
        short[,] parent = new short[masks, N];
        for (int m = 0; m < masks; m++)
            for (int i = 0; i < N; i++) { dp[m, i] = 1e100; parent[m, i] = -1; }
        for (int i = 0; i < N; i++) dp[1 << i, i] = Cost[0, i];

        for (int mask = 1; mask < masks; mask++)
        {
            for (int last = 0; last < N; last++)
            {
                double cur = dp[mask, last];
                if (cur >= 1e90) continue;
                int rem = ((masks - 1) ^ mask);
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int nxt = BitIndex(bit);
                    int nm = mask | bit;
                    double cand = cur + Cost[last + 1, nxt];
                    if (cand < dp[nm, nxt])
                    {
                        dp[nm, nxt] = cand;
                        parent[nm, nxt] = (short)last;
                    }
                    rem -= bit;
                }
            }
        }

        int full = masks - 1, best = 0;
        for (int i = 1; i < N; i++) if (dp[full, i] < dp[full, best]) best = i;
        int[] ans = new int[N];
        int msk = full, at = best;
        for (int k = N - 1; k >= 0; k--)
        {
            ans[k] = at;
            int prev = parent[msk, at];
            msk ^= 1 << at;
            at = prev;
        }
        return ans;
    }

    static int BitIndex(int bit)
    {
        int idx = 0;
        while ((bit >>= 1) != 0) idx++;
        return idx;
    }

    static int[] SolveHeuristic()
    {
        var starts = CandidateStarts();
        int[] best = Array.Empty<int>();
        double bestCost = 1e100;
        int attempts = N <= 20 ? Math.Min(20, starts.Count) : Math.Min(35, starts.Count);

        for (int a = 0; a < attempts; a++)
        {
            int[] route = BuildNearest(starts[a], a);
            Improve(route, N <= 30 ? 9 : 5);
            double c = RouteCost(route);
            if (c < bestCost) { bestCost = c; best = (int[])route.Clone(); }
        }

        Improve(best, N <= 30 ? 14 : 8);
        return best;
    }

    static List<int> CandidateStarts()
    {
        var list = Enumerable.Range(0, N).OrderBy(i => Cost[0, i]).Take(Math.Min(N, 18)).ToList();
        foreach (int i in Enumerable.Range(0, N).OrderBy(i => Stations[i].RadiusRank).Take(Math.Min(N, 6)))
            if (!list.Contains(i)) list.Add(i);
        foreach (int i in Enumerable.Range(0, N).OrderByDescending(i => Stations[i].RadiusRank).Take(Math.Min(N, 6)))
            if (!list.Contains(i)) list.Add(i);
        for (int i = 0; i < N && list.Count < Math.Min(N, 40); i++)
        {
            int x = Rng.Next(N);
            if (!list.Contains(x)) list.Add(x);
        }
        return list;
    }

    static int[] BuildNearest(int first, int variant)
    {
        bool[] used = new bool[N];
        int[] route = new int[N];
        route[0] = first;
        used[first] = true;
        for (int k = 1; k < N; k++)
        {
            int last = route[k - 1], best = -1;
            double bv = 1e100;
            for (int j = 0; j < N; j++)
            {
                if (used[j]) continue;
                double c = Cost[last + 1, j];
                if ((variant & 1) != 0) c += 0.04 * Cost[0, j];
                if ((variant & 2) != 0) c += 0.000003 * Math.Abs(Stations[last].RadiusRank - Stations[j].RadiusRank);
                if (c < bv) { bv = c; best = j; }
            }
            route[k] = best;
            used[best] = true;
        }
        return route;
    }

    static void Improve(int[] route, int passes)
    {
        if (route.Length < 3) return;
        for (int pass = 0; pass < passes; pass++)
        {
            bool changed = TwoOpt(route);
            changed |= Relocate(route);
            if (!changed) break;
        }
    }

    static bool TwoOpt(int[] r)
    {
        bool changed = false;
        int n = r.Length;
        for (int i = 0; i < n - 2; i++)
        {
            int beforeA = i == 0 ? -1 : r[i - 1];
            for (int k = i + 1; k < n - 1; k++)
            {
                double old = SegmentCost(r, beforeA, i, k, false) + Edge(r[k], r[k + 1]);
                double neu = SegmentCost(r, beforeA, i, k, true) + Edge(r[i], r[k + 1]);
                if (neu + 1e-10 < old)
                {
                    Array.Reverse(r, i, k - i + 1);
                    changed = true;
                }
            }
        }
        return changed;
    }

    static double SegmentCost(int[] r, int before, int lo, int hi, bool reversed)
    {
        double s = Edge(before, reversed ? r[hi] : r[lo]);
        if (!reversed)
            for (int i = lo + 1; i <= hi; i++) s += Edge(r[i - 1], r[i]);
        else
            for (int i = hi - 1; i >= lo; i--) s += Edge(r[i + 1], r[i]);
        return s;
    }

    static bool Relocate(int[] r)
    {
        int n = r.Length;
        bool changed = false;
        for (int i = 0; i < n; i++)
        {
            int node = r[i];
            int prev = i == 0 ? -1 : r[i - 1];
            int next = i == n - 1 ? -2 : r[i + 1];
            double remove = Edge(prev, node) + (next == -2 ? 0.0 : Edge(node, next))
                          - (next == -2 ? 0.0 : Edge(prev, next));

            int bestPos = -1;
            double bestGain = 1e-10;
            for (int pos = 0; pos <= n; pos++)
            {
                if (pos == i || pos == i + 1) continue;
                int left = pos == 0 ? -1 : r[pos - 1];
                int right = pos == n ? -2 : r[pos];
                if (left == node || right == node) continue;
                double add = Edge(left, node) + (right == -2 ? 0.0 : Edge(node, right))
                           - (right == -2 ? 0.0 : Edge(left, right));
                double gain = remove - add;
                if (gain > bestGain) { bestGain = gain; bestPos = pos; }
            }

            if (bestPos >= 0)
            {
                var list = new List<int>(r);
                list.RemoveAt(i);
                if (bestPos > i) bestPos--;
                list.Insert(bestPos, node);
                for (int q = 0; q < n; q++) r[q] = list[q];
                changed = true;
            }
        }
        return changed;
    }

    static double Edge(int from, int to) => from < 0 ? Cost[0, to] : Cost[from + 1, to];

    static double RouteCost(int[] r)
    {
        double s = Edge(-1, r[0]);
        for (int i = 1; i < r.Length; i++) s += Edge(r[i - 1], r[i]);
        return s;
    }
}
