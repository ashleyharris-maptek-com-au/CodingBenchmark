#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

using namespace std;

struct Item {
    int id;
    int w;
    int h;
    long long area;
};

struct Rect {
    int x;
    int y;
    int w;
    int h;
};

struct Placement {
    int board = 0;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

static bool containsRect(const Rect& a, const Rect& b) {
    return a.x <= b.x && a.y <= b.y &&
           a.x + a.w >= b.x + b.w &&
           a.y + a.h >= b.y + b.h;
}

struct Board {
    vector<Rect> freeRects;
    long long usedArea = 0;
    int newest = 0;
};

struct Candidate {
    bool ok = false;
    int freeIndex = -1;
    int w = 0;
    int h = 0;
    int rotated = 0;
    long long score1 = numeric_limits<long long>::max();
    long long score2 = numeric_limits<long long>::max();
};

static Candidate findBestOnBoard(const Board& board, const Item& item) {
    Candidate best;
    for (int i = 0; i < (int)board.freeRects.size(); ++i) {
        const Rect& f = board.freeRects[i];
        for (int r = 0; r < 2; ++r) {
            int w = r ? item.h : item.w;
            int h = r ? item.w : item.h;
            if (r == 1 && item.w == item.h) continue;
            if (w > f.w || h > f.h) continue;

            long long leftoverW = f.w - w;
            long long leftoverH = f.h - h;
            long long shortSide = min(leftoverW, leftoverH);
            long long longSide = max(leftoverW, leftoverH);
            long long waste = 1LL * f.w * f.h - 1LL * w * h;
            long long score1 = shortSide * 4096 + longSide;
            long long score2 = waste;

            if (!best.ok || score1 < best.score1 ||
                (score1 == best.score1 && score2 < best.score2)) {
                best.ok = true;
                best.freeIndex = i;
                best.w = w;
                best.h = h;
                best.rotated = r;
                best.score1 = score1;
                best.score2 = score2;
            }
        }
    }
    return best;
}

static void pruneAndCap(vector<Rect>& freeRects, int cap) {
    vector<char> removed(freeRects.size(), 0);
    for (int i = 0; i < (int)freeRects.size(); ++i) {
        if (removed[i]) continue;
        if (freeRects[i].w <= 0 || freeRects[i].h <= 0) {
            removed[i] = 1;
            continue;
        }
        for (int j = i + 1; j < (int)freeRects.size(); ++j) {
            if (removed[j]) continue;
            if (containsRect(freeRects[i], freeRects[j])) {
                removed[j] = 1;
            } else if (containsRect(freeRects[j], freeRects[i])) {
                removed[i] = 1;
                break;
            }
        }
    }

    vector<Rect> kept;
    kept.reserve(freeRects.size());
    for (int i = 0; i < (int)freeRects.size(); ++i) {
        if (!removed[i]) kept.push_back(freeRects[i]);
    }

    if ((int)kept.size() > cap) {
        nth_element(kept.begin(), kept.begin() + cap, kept.end(),
            [](const Rect& a, const Rect& b) {
                long long aa = 1LL * a.w * a.h;
                long long bb = 1LL * b.w * b.h;
                if (aa != bb) return aa > bb;
                return max(a.w, a.h) > max(b.w, b.h);
            });
        kept.resize(cap);
    }
    freeRects.swap(kept);
}

static void placeOnBoard(Board& board, const Candidate& c, int boardW, int boardH) {
    Rect used{board.freeRects[c.freeIndex].x, board.freeRects[c.freeIndex].y, c.w, c.h};
    vector<Rect> next;
    next.reserve(board.freeRects.size() + 4);

    for (const Rect& f : board.freeRects) {
        bool disjoint = used.x >= f.x + f.w || used.x + used.w <= f.x ||
                        used.y >= f.y + f.h || used.y + used.h <= f.y;
        if (disjoint) {
            next.push_back(f);
            continue;
        }

        if (used.x > f.x) {
            next.push_back(Rect{f.x, f.y, used.x - f.x, f.h});
        }
        if (used.x + used.w < f.x + f.w) {
            next.push_back(Rect{used.x + used.w, f.y,
                                f.x + f.w - (used.x + used.w), f.h});
        }
        if (used.y > f.y) {
            next.push_back(Rect{f.x, f.y, f.w, used.y - f.y});
        }
        if (used.y + used.h < f.y + f.h) {
            next.push_back(Rect{f.x, used.y + used.h, f.w,
                                f.y + f.h - (used.y + used.h)});
        }
    }

    board.freeRects.swap(next);
    int cap = (boardW * 1LL * boardH <= 4000000LL) ? 192 : 256;
    pruneAndCap(board.freeRects, cap);
    board.usedArea += 1LL * used.w * used.h;
    ++board.newest;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, boardW, boardH;
    if (!(cin >> n >> boardW >> boardH)) return 0;

    vector<Item> items;
    items.reserve(n);
    vector<Placement> ans(n);

    for (int i = 0; i < n; ++i) {
        int w, h;
        cin >> w >> h;
        items.push_back(Item{i, w, h, 1LL * w * h});
    }

    sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        int aLong = max(a.w, a.h), bLong = max(b.w, b.h);
        int aShort = min(a.w, a.h), bShort = min(b.w, b.h);
        if (a.area != b.area) return a.area > b.area;
        if (aLong != bLong) return aLong > bLong;
        return aShort > bShort;
    });

    vector<Board> boards;
    boards.reserve(max(1, n / 8));
    const long long boardArea = 1LL * boardW * boardH;

    for (const Item& item : items) {
        int naturalFits = (item.w <= boardW && item.h <= boardH);
        int rotatedFits = (item.h <= boardW && item.w <= boardH);
        if (!naturalFits && !rotatedFits) {
            ans[item.id] = Placement{0, 0, 0, 0};
            continue;
        }

        int bestBoard = -1;
        Candidate bestCand;

        int boardCount = (int)boards.size();
        int scanLimit = boardCount;
        if (n > 20000) scanLimit = min(boardCount, 80);
        else if (n > 5000) scanLimit = min(boardCount, 160);
        else if (n > 1000) scanLimit = min(boardCount, 320);

        int passes = (n <= 5000) ? 2 : 1;
        for (int pass = 0; pass < passes; ++pass) {
            int scanned = 0;
            for (int bi = boardCount - 1; bi >= 0; --bi) {
                Board& b = boards[bi];
                if (pass == 0 && scanned++ >= scanLimit) break;
                if (pass == 1 && boardArea - b.usedArea < item.area) continue;

                Candidate c = findBestOnBoard(b, item);
                if (!c.ok) continue;

                long long fillBonus = boardArea - b.usedArea - item.area;
                long long score1 = c.score1;
                long long score2 = c.score2 + fillBonus / 1024;
                if (bestBoard < 0 || score1 < bestCand.score1 ||
                    (score1 == bestCand.score1 && score2 < bestCand.score2)) {
                    bestBoard = bi;
                    bestCand = c;
                    bestCand.score2 = score2;
                }
            }
            if (bestBoard >= 0 || pass == passes - 1) break;
            if (scanLimit == boardCount) break;
        }

        if (bestBoard < 0) {
            Board b;
            b.freeRects.push_back(Rect{0, 0, boardW, boardH});
            boards.push_back(std::move(b));
            bestBoard = (int)boards.size() - 1;
            bestCand = findBestOnBoard(boards.back(), item);
        }

        Rect spot = boards[bestBoard].freeRects[bestCand.freeIndex];
        ans[item.id] = Placement{bestBoard, spot.x, spot.y, bestCand.rotated};
        placeOnBoard(boards[bestBoard], bestCand, boardW, boardH);
    }

    cout << boards.size() << '\n';
    for (int i = 0; i < n; ++i) {
        cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rotated << '\n';
    }
    return 0;
}
