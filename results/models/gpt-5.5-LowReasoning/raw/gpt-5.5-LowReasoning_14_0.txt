using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    static readonly int[] DX = { -1, 0, 1, -1, 1, -1, 0, 1 };
    static readonly int[] DY = { -1, -1, -1, 0, 0, 1, 1, 1 };

    struct Constraint
    {
        public List<int> Vars;
        public int Need;
        public Constraint(List<int> vars, int need) { Vars = vars; Need = need; }
    }

    sealed class FastScanner
    {
        readonly Stream input = Console.OpenStandardInput();
        readonly byte[] buffer = new byte[1 << 16];
        int len, ptr;

        int ReadByte()
        {
            if (ptr >= len)
            {
                len = input.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len == 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByte(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int v = 0;
            while (c > 32)
            {
                v = v * 10 + c - '0';
                c = ReadByte();
            }
            return v * sign;
        }

        public char[] NextBoardRow(int w)
        {
            int c;
            do { c = ReadByte(); } while (c == '\r' || c == '\n');
            char[] row = new char[w];
            int i = 0;
            while (i < w && c >= 0 && c != '\r' && c != '\n')
            {
                row[i++] = (char)c;
                c = ReadByte();
            }
            while (i < w) row[i++] = ' ';
            if (c == '\r')
            {
                int n = ReadByte();
                if (n != '\n' && n >= 0) ptr--;
            }
            return row;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int h = fs.NextInt(), w = fs.NextInt();
        char[][] board = new char[h][];
        for (int y = 0; y < h; y++) board[y] = fs.NextBoardRow(w);

        var varIdByCell = new Dictionary<long, int>();
        var xs = new List<int>();
        var ys = new List<int>();
        var constraints = new List<Constraint>();

        for (int y = 0; y < h; y++)
        {
            char[] row = board[y];
            for (int x = 0; x < w; x++)
            {
                char c = row[x];
                if (c < '0' || c > '8') continue;

                int need = c - '0';
                var vars = new List<int>(8);
                for (int k = 0; k < 8; k++)
                {
                    int nx = x + DX[k], ny = y + DY[k];
                    if ((uint)nx >= (uint)w || (uint)ny >= (uint)h) continue;
                    char nc = board[ny][nx];
                    if (nc == '*') need--;
                    else if (nc == ' ')
                    {
                        long key = (((long)ny) << 32) | (uint)nx;
                        if (!varIdByCell.TryGetValue(key, out int id))
                        {
                            id = xs.Count;
                            varIdByCell[key] = id;
                            xs.Add(nx);
                            ys.Add(ny);
                        }
                        vars.Add(id);
                    }
                }
                if (vars.Count == 0) continue;
                if (need < 0) need = 0;
                if (need > vars.Count) need = vars.Count;
                constraints.Add(new Constraint(vars, need));
            }
        }

        int nVars = xs.Count;
        bool[] safe = new bool[nVars];
        bool[] mine = new bool[nVars];

        bool changed = true;
        int passes = 0;
        while (changed && passes++ < 64)
        {
            changed = false;
            foreach (var con in constraints)
            {
                int need = con.Need, unknown = 0;
                foreach (int v in con.Vars)
                {
                    if (mine[v]) need--;
                    else if (!safe[v]) unknown++;
                }
                if (unknown == 0) continue;
                if (need == 0)
                {
                    foreach (int v in con.Vars)
                        if (!mine[v] && !safe[v]) { safe[v] = true; changed = true; }
                }
                else if (need == unknown)
                {
                    foreach (int v in con.Vars)
                        if (!mine[v] && !safe[v]) { mine[v] = true; changed = true; }
                }
            }
        }

        int activeVars = 0;
        for (int i = 0; i < nVars; i++) if (!safe[i] && !mine[i]) activeVars++;
        if (activeVars <= 250000) SubsetReasoning(constraints, safe, mine);

        activeVars = 0;
        for (int i = 0; i < nVars; i++) if (!safe[i] && !mine[i]) activeVars++;
        if (activeVars <= 200000) ExactSmallComponents(constraints, safe, mine, nVars);

        var output = new StringBuilder();
        int count = 0;
        for (int i = 0; i < nVars; i++) if (safe[i]) count++;
        output.Append(count).Append('\n');
        for (int i = 0; i < nVars; i++)
            if (safe[i]) output.Append(xs[i]).Append(' ').Append(ys[i]).Append('\n');
        Console.Write(output.ToString());
    }

    static void SubsetReasoning(List<Constraint> constraints, bool[] safe, bool[] mine)
    {
        var reduced = new List<Constraint>();
        foreach (var c in constraints)
        {
            int need = c.Need;
            var vars = new List<int>(c.Vars.Count);
            foreach (int v in c.Vars)
            {
                if (mine[v]) need--;
                else if (!safe[v]) vars.Add(v);
            }
            if (vars.Count > 0 && need >= 0 && need <= vars.Count)
            {
                vars.Sort();
                reduced.Add(new Constraint(vars, need));
            }
        }

        int limit = Math.Min(reduced.Count, 120000);
        var byFirst = new Dictionary<int, List<int>>();
        for (int i = 0; i < limit; i++)
        {
            foreach (int v in reduced[i].Vars)
            {
                if (!byFirst.TryGetValue(v, out var list)) byFirst[v] = list = new List<int>();
                list.Add(i);
            }
        }

        for (int i = 0; i < limit; i++)
        {
            var a = reduced[i];
            if (a.Vars.Count == 0) continue;
            var candidates = new HashSet<int>();
            foreach (int v in a.Vars)
                if (byFirst.TryGetValue(v, out var list))
                    foreach (int j in list) candidates.Add(j);

            foreach (int j in candidates)
            {
                if (i == j) continue;
                var b = reduced[j];
                if (a.Vars.Count >= b.Vars.Count) continue;
                if (!IsSubset(a.Vars, b.Vars)) continue;
                int diffNeed = b.Need - a.Need;
                int diffCount = b.Vars.Count - a.Vars.Count;
                if (diffNeed == 0)
                    MarkDifference(b.Vars, a.Vars, safe, false);
                else if (diffNeed == diffCount)
                    MarkDifference(b.Vars, a.Vars, mine, false);
            }
        }
    }

    static bool IsSubset(List<int> a, List<int> b)
    {
        int i = 0, j = 0;
        while (i < a.Count && j < b.Count)
        {
            if (a[i] == b[j]) { i++; j++; }
            else if (a[i] > b[j]) j++;
            else return false;
        }
        return i == a.Count;
    }

    static void MarkDifference(List<int> sup, List<int> sub, bool[] mark, bool value)
    {
        int i = 0, j = 0;
        while (i < sup.Count)
        {
            if (j < sub.Count && sup[i] == sub[j]) { i++; j++; }
            else mark[sup[i++]] = true;
        }
    }

    static void ExactSmallComponents(List<Constraint> constraints, bool[] safe, bool[] mine, int nVars)
    {
        long assignmentBudget = 50000000;
        var varCons = new List<int>[nVars];
        for (int i = 0; i < nVars; i++) varCons[i] = new List<int>();
        var cons = new List<Constraint>();
        foreach (var c in constraints)
        {
            int need = c.Need;
            var vars = new List<int>();
            foreach (int v in c.Vars)
            {
                if (mine[v]) need--;
                else if (!safe[v]) vars.Add(v);
            }
            if (vars.Count > 0 && need >= 0 && need <= vars.Count)
            {
                int ci = cons.Count;
                cons.Add(new Constraint(vars, need));
                foreach (int v in vars) varCons[v].Add(ci);
            }
        }

        bool[] seen = new bool[nVars];
        for (int start = 0; start < nVars; start++)
        {
            if (seen[start] || safe[start] || mine[start] || varCons[start].Count == 0) continue;
            var compVars = new List<int>();
            var compConsSet = new HashSet<int>();
            var q = new Queue<int>();
            q.Enqueue(start); seen[start] = true;
            while (q.Count > 0 && compVars.Count <= 32)
            {
                int v = q.Dequeue();
                compVars.Add(v);
                foreach (int ci in varCons[v])
                {
                    compConsSet.Add(ci);
                    foreach (int u in cons[ci].Vars)
                        if (!seen[u]) { seen[u] = true; q.Enqueue(u); }
                }
            }
            if (compVars.Count > 24) continue;
            long cost = 1L << compVars.Count;
            if (cost > assignmentBudget) continue;
            assignmentBudget -= cost;
            ExactEnumerate(compVars, compConsSet, cons, safe);
            if (assignmentBudget <= 0) return;
        }
    }

    static void ExactEnumerate(List<int> compVars, HashSet<int> compConsSet, List<Constraint> cons, bool[] safe)
    {
        int n = compVars.Count;
        var local = new Dictionary<int, int>();
        for (int i = 0; i < n; i++) local[compVars[i]] = i;

        var masks = new List<int>();
        var needs = new List<int>();
        foreach (int ci in compConsSet)
        {
            int mask = 0;
            foreach (int v in cons[ci].Vars) mask |= 1 << local[v];
            masks.Add(mask);
            needs.Add(cons[ci].Need);
        }

        long valid = 0;
        int alwaysMine = (1 << n) - 1;
        int everMine = 0;
        int total = 1 << n;
        for (int assn = 0; assn < total; assn++)
        {
            bool ok = true;
            for (int c = 0; c < masks.Count; c++)
            {
                if (PopCount((uint)(assn & masks[c])) != needs[c]) { ok = false; break; }
            }
            if (!ok) continue;
            valid++;
            alwaysMine &= assn;
            everMine |= assn;
        }
        if (valid == 0) return;
        int alwaysSafeMask = ((1 << n) - 1) ^ everMine;
        for (int i = 0; i < n; i++)
            if (((alwaysSafeMask >> i) & 1) != 0) safe[compVars[i]] = true;
    }

    static int PopCount(uint x)
    {
        x -= (x >> 1) & 0x55555555u;
        x = (x & 0x33333333u) + ((x >> 2) & 0x33333333u);
        return (int)((((x + (x >> 4)) & 0x0F0F0F0Fu) * 0x01010101u) >> 24);
    }
}
