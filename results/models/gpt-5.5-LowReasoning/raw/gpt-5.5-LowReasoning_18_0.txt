struct FastScanner {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t pos = 0, len = 0;

    inline char readChar() {
        if (pos >= len) {
            len = fread(buf, 1, BUFSIZE, stdin);
            pos = 0;
            if (len == 0) return 0;
        }
        return buf[pos++];
    }

    template <class T>
    bool readUInt(T& out) {
        char c;
        do {
            c = readChar();
            if (!c) return false;
        } while (c <= ' ');
        T x = 0;
        while (c > ' ') {
            x = x * 10 + static_cast<T>(c - '0');
            c = readChar();
        }
        out = x;
        return true;
    }
};

struct Edge {
    uint32_t u, v;
};

static void emitCut(const vector<Edge>& edges, const vector<unsigned char>& side) {
    vector<Edge> cut;
    cut.reserve(1024);
    for (const Edge& e : edges) {
        if (side[e.u] != side[e.v]) cut.push_back(e);
    }
    cout << cut.size() << '\n';
    for (const Edge& e : cut) cout << e.u << ' ' << e.v << '\n';
}

static bool findBridgeCut(uint32_t n, const vector<Edge>& edges, vector<unsigned char>& side) {
    vector<uint32_t> deg(n, 0);
    for (const Edge& e : edges) {
        if (e.u != e.v) {
            ++deg[e.u];
            ++deg[e.v];
        }
    }

    vector<uint64_t> start(static_cast<size_t>(n) + 1, 0);
    for (uint32_t i = 0; i < n; ++i) start[i + 1] = start[i] + deg[i];

    vector<uint32_t> to(start[n]), eid(start[n]);
    vector<uint64_t> cur = start;
    for (uint32_t i = 0; i < edges.size(); ++i) {
        uint32_t u = edges[i].u, v = edges[i].v;
        if (u == v) continue;
        uint64_t a = cur[u]++;
        to[a] = v;
        eid[a] = i;
        uint64_t b = cur[v]++;
        to[b] = u;
        eid[b] = i;
    }

    vector<uint32_t> tin(n, 0), low(n, 0), parent(n, numeric_limits<uint32_t>::max());
    vector<uint32_t> parentEdge(n, numeric_limits<uint32_t>::max());
    vector<uint64_t> it(n, 0);
    uint32_t timer = 0, bridgeA = numeric_limits<uint32_t>::max(), bridgeB = 0;

    for (uint32_t root = 0; root < n && bridgeA == numeric_limits<uint32_t>::max(); ++root) {
        if (tin[root]) continue;
        vector<uint32_t> st;
        st.push_back(root);
        parent[root] = root;
        while (!st.empty() && bridgeA == numeric_limits<uint32_t>::max()) {
            uint32_t v = st.back();
            if (!tin[v]) {
                tin[v] = low[v] = ++timer;
                it[v] = start[v];
            }
            if (it[v] < start[v + 1]) {
                uint32_t p = it[v]++;
                uint32_t w = to[p], id = eid[p];
                if (id == parentEdge[v]) continue;
                if (!tin[w]) {
                    parent[w] = v;
                    parentEdge[w] = id;
                    st.push_back(w);
                } else {
                    low[v] = min(low[v], tin[w]);
                }
            } else {
                st.pop_back();
                if (parent[v] != v) {
                    uint32_t p = parent[v];
                    low[p] = min(low[p], low[v]);
                    if (low[v] > tin[p]) {
                        bridgeA = p;
                        bridgeB = v;
                    }
                }
            }
        }
    }

    if (bridgeA == numeric_limits<uint32_t>::max()) return false;

    side.assign(n, 0);
    vector<uint32_t> st{bridgeB};
    side[bridgeB] = 1;
    while (!st.empty()) {
        uint32_t v = st.back();
        st.pop_back();
        for (uint64_t p = start[v]; p < start[v + 1]; ++p) {
            if (eid[p] == parentEdge[bridgeB]) continue;
            uint32_t w = to[p];
            if (!side[w]) {
                side[w] = 1;
                st.push_back(w);
            }
        }
    }
    return true;
}

static pair<uint64_t, vector<unsigned char>> stoerWagner(uint32_t n, const vector<Edge>& edges) {
    vector<unordered_map<uint32_t, uint64_t>> g(n);
    for (const Edge& e : edges) {
        if (e.u == e.v) continue;
        ++g[e.u][e.v];
        ++g[e.v][e.u];
    }

    vector<uint32_t> active(n);
    iota(active.begin(), active.end(), 0);
    vector<vector<uint32_t>> members(n);
    for (uint32_t i = 0; i < n; ++i) members[i].push_back(i);

    uint64_t best = numeric_limits<uint64_t>::max();
    vector<unsigned char> bestSide(n, 0), used(n, 0);
    vector<uint64_t> weight(n, 0);

    while (active.size() > 1) {
        fill(used.begin(), used.end(), 0);
        fill(weight.begin(), weight.end(), 0);
        uint32_t prev = active[0], sel = active[0];

        for (size_t i = 0; i < active.size(); ++i) {
            sel = active[0];
            for (uint32_t v : active) {
                if (!used[v] && (used[sel] || weight[v] > weight[sel])) sel = v;
            }
            if (i + 1 == active.size()) {
                uint64_t cutValue = weight[sel];
                if (cutValue < best) {
                    best = cutValue;
                    fill(bestSide.begin(), bestSide.end(), 0);
                    for (uint32_t v : members[sel]) bestSide[v] = 1;
                }

                if (prev != sel) {
                    for (const auto& kv : g[sel]) {
                        uint32_t v = kv.first;
                        uint64_t w = kv.second;
                        if (v == prev) continue;
                        g[prev][v] += w;
                        g[v][prev] += w;
                        g[v].erase(sel);
                    }
                    g[prev].erase(sel);
                    g[sel].clear();
                    members[prev].insert(members[prev].end(), members[sel].begin(), members[sel].end());
                    members[sel].clear();
                    active.erase(find(active.begin(), active.end(), sel));
                }
                break;
            }

            used[sel] = 1;
            prev = sel;
            for (const auto& kv : g[sel]) {
                if (!used[kv.first]) weight[kv.first] += kv.second;
            }
        }
    }
    return {best, bestSide};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    FastScanner fs;
    uint64_t n64 = 0, m64 = 0;
    if (!fs.readUInt(n64) || !fs.readUInt(m64)) return 0;
    if (n64 <= 1) {
        cout << "0\n";
        return 0;
    }
    if (n64 > numeric_limits<uint32_t>::max() || m64 > numeric_limits<uint32_t>::max()) {
        return 1;
    }

    uint32_t n = static_cast<uint32_t>(n64);
    uint32_t m = static_cast<uint32_t>(m64);
    vector<Edge> edges;
    edges.reserve(static_cast<size_t>(m));
    vector<uint64_t> degree(n, 0);

    for (uint32_t i = 0; i < m; ++i) {
        uint64_t a = 0, b = 0;
        fs.readUInt(a);
        fs.readUInt(b);
        uint32_t u = static_cast<uint32_t>(a), v = static_cast<uint32_t>(b);
        edges.push_back({u, v});
        if (u != v) {
            ++degree[u];
            ++degree[v];
        }
    }

    vector<unsigned char> side(n, 0);

    uint32_t minDegVertex = 0;
    for (uint32_t i = 1; i < n; ++i) {
        if (degree[i] < degree[minDegVertex]) minDegVertex = i;
    }
    if (degree[minDegVertex] == 0) {
        side[minDegVertex] = 1;
        emitCut(edges, side);
        return 0;
    }

    if (findBridgeCut(n, edges, side)) {
        emitCut(edges, side);
        return 0;
    }

    const uint64_t exactWorkLimit = 25000000ULL;
    if (static_cast<uint64_t>(n) * static_cast<uint64_t>(max<uint32_t>(1, n)) <= exactWorkLimit ||
        (n <= 6000 && m <= 2000000)) {
        auto result = stoerWagner(n, edges);
        emitCut(edges, result.second);
        return 0;
    }

    side.assign(n, 0);
    side[minDegVertex] = 1;
    emitCut(edges, side);
    return 0;
}
