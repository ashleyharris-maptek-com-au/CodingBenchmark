using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    struct Box
    {
        public int Id;
        public long W, H, D, V;
    }

    struct Cuboid
    {
        public long X, Y, Z, W, H, D;
        public long X2 { get { return X + W; } }
        public long Y2 { get { return Y + H; } }
        public long Z2 { get { return Z + D; } }
    }

    struct Placement
    {
        public int Id;
        public long X, Y, Z;
    }

    sealed class FastScanner
    {
        private readonly Stream input = Console.OpenStandardInput();
        private readonly byte[] data = new byte[1 << 16];
        private int len, ptr;

        private int ReadByte()
        {
            if (ptr >= len)
            {
                len = input.Read(data, 0, data.Length);
                ptr = 0;
                if (len == 0) return -1;
            }
            return data[ptr++];
        }

        public long NextLong()
        {
            int c;
            do c = ReadByte(); while (c <= 32 && c >= 0);
            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            long v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = ReadByte();
            }
            return v * sign;
        }
    }

    static bool Intersects(Cuboid a, Cuboid b)
    {
        return a.X < b.X2 && a.X2 > b.X &&
               a.Y < b.Y2 && a.Y2 > b.Y &&
               a.Z < b.Z2 && a.Z2 > b.Z;
    }

    static bool Contains(Cuboid a, Cuboid b)
    {
        return a.X <= b.X && a.Y <= b.Y && a.Z <= b.Z &&
               a.X2 >= b.X2 && a.Y2 >= b.Y2 && a.Z2 >= b.Z2;
    }

    static void AddIfValid(List<Cuboid> list, Cuboid c)
    {
        if (c.W > 0 && c.H > 0 && c.D > 0) list.Add(c);
    }

    static void SplitFreeCuboid(Cuboid f, Cuboid used, List<Cuboid> output)
    {
        if (!Intersects(f, used))
        {
            output.Add(f);
            return;
        }

        if (used.X > f.X)
            AddIfValid(output, new Cuboid { X = f.X, Y = f.Y, Z = f.Z, W = used.X - f.X, H = f.H, D = f.D });
        if (used.X2 < f.X2)
            AddIfValid(output, new Cuboid { X = used.X2, Y = f.Y, Z = f.Z, W = f.X2 - used.X2, H = f.H, D = f.D });

        long ix1 = Math.Max(f.X, used.X);
        long ix2 = Math.Min(f.X2, used.X2);

        if (used.Y > f.Y)
            AddIfValid(output, new Cuboid { X = ix1, Y = f.Y, Z = f.Z, W = ix2 - ix1, H = used.Y - f.Y, D = f.D });
        if (used.Y2 < f.Y2)
            AddIfValid(output, new Cuboid { X = ix1, Y = used.Y2, Z = f.Z, W = ix2 - ix1, H = f.Y2 - used.Y2, D = f.D });

        long iy1 = Math.Max(f.Y, used.Y);
        long iy2 = Math.Min(f.Y2, used.Y2);

        if (used.Z > f.Z)
            AddIfValid(output, new Cuboid { X = ix1, Y = iy1, Z = f.Z, W = ix2 - ix1, H = iy2 - iy1, D = used.Z - f.Z });
        if (used.Z2 < f.Z2)
            AddIfValid(output, new Cuboid { X = ix1, Y = iy1, Z = used.Z2, W = ix2 - ix1, H = iy2 - iy1, D = f.Z2 - used.Z2 });
    }

    static void PruneFreeList(List<Cuboid> free)
    {
        for (int i = 0; i < free.Count; i++)
        {
            for (int j = i + 1; j < free.Count; j++)
            {
                if (Contains(free[i], free[j]))
                {
                    free.RemoveAt(j--);
                }
                else if (Contains(free[j], free[i]))
                {
                    free.RemoveAt(i--);
                    break;
                }
            }
        }
    }

    static void Place(Cuboid used, List<Cuboid> free)
    {
        var next = new List<Cuboid>(free.Count + 8);
        for (int i = 0; i < free.Count; i++) SplitFreeCuboid(free[i], used, next);
        free.Clear();
        free.AddRange(next);
        PruneFreeList(free);
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = (int)fs.NextLong();
        long W = fs.NextLong(), H = fs.NextLong(), D = fs.NextLong();

        var boxes = new List<Box>(n);
        for (int i = 0; i < n; i++)
        {
            long w = fs.NextLong(), h = fs.NextLong(), d = fs.NextLong();
            if (w <= W && h <= H && d <= D)
                boxes.Add(new Box { Id = i, W = w, H = h, D = d, V = w * h * d });
        }

        boxes.Sort((a, b) =>
        {
            int c = a.V.CompareTo(b.V);
            if (c != 0) return c;
            c = Math.Max(a.W, Math.Max(a.H, a.D)).CompareTo(Math.Max(b.W, Math.Max(b.H, b.D)));
            return c != 0 ? c : a.Id.CompareTo(b.Id);
        });

        var free = new List<Cuboid> { new Cuboid { X = 0, Y = 0, Z = 0, W = W, H = H, D = D } };
        var placed = new List<Placement>();

        foreach (var box in boxes)
        {
            int best = -1;
            long bestScore1 = long.MaxValue, bestScore2 = long.MaxValue, bestScore3 = long.MaxValue;

            for (int i = 0; i < free.Count; i++)
            {
                var f = free[i];
                if (box.W > f.W || box.H > f.H || box.D > f.D) continue;

                long lw = f.W - box.W, lh = f.H - box.H, ld = f.D - box.D;
                long shortSide = Math.Min(lw, Math.Min(lh, ld));
                long waste = f.W * f.H * f.D - box.V;
                long coord = f.X + f.Y + f.Z;

                if (shortSide < bestScore1 ||
                    (shortSide == bestScore1 && waste < bestScore2) ||
                    (shortSide == bestScore1 && waste == bestScore2 && coord < bestScore3))
                {
                    best = i;
                    bestScore1 = shortSide;
                    bestScore2 = waste;
                    bestScore3 = coord;
                }
            }

            if (best < 0) continue;

            var spot = free[best];
            var used = new Cuboid { X = spot.X, Y = spot.Y, Z = spot.Z, W = box.W, H = box.H, D = box.D };
            placed.Add(new Placement { Id = box.Id, X = used.X, Y = used.Y, Z = used.Z });
            Place(used, free);
        }

        var sb = new StringBuilder();
        sb.AppendLine(placed.Count.ToString());
        for (int i = 0; i < placed.Count; i++)
        {
            var p = placed[i];
            sb.Append(p.Id).Append(' ').Append(p.X).Append(' ').Append(p.Y).Append(' ').Append(p.Z).AppendLine();
        }
        Console.Write(sb.ToString());
    }
}
