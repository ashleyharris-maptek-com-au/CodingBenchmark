using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    struct Op
    {
        public int X, Y, Z, N;
        public byte Value, Mass;
    }

    sealed class FastInput
    {
        readonly Stream _s = Console.OpenStandardInput();

        public string NextToken()
        {
            var bytes = new List<byte>(32);
            int c;
            do
            {
                c = _s.ReadByte();
            } while (c >= 0 && c <= 32);

            while (c > 32)
            {
                bytes.Add((byte)c);
                c = _s.ReadByte();
            }
            return Encoding.ASCII.GetString(bytes.ToArray());
        }

        public int ReadByteRaw()
        {
            return _s.ReadByte();
        }
    }

    public static void Main()
    {
        var input = new FastInput();

        int xCount = int.Parse(input.NextToken(), CultureInfo.InvariantCulture);
        int yCount = int.Parse(input.NextToken(), CultureInfo.InvariantCulture);
        int zCount = int.Parse(input.NextToken(), CultureInfo.InvariantCulture);
        double xBlock = double.Parse(input.NextToken(), CultureInfo.InvariantCulture);
        double yBlock = double.Parse(input.NextToken(), CultureInfo.InvariantCulture);
        double zBlock = double.Parse(input.NextToken(), CultureInfo.InvariantCulture);

        long columns = (long)xCount * yCount;
        long total = columns * (long)zCount;

        // Adaptive depth: the top layer is always pathable on the original flat surface.
        // One more layer remains pathable because the neighboring ground is only 1 z-voxel higher.
        // For tiny cases, keep a third layer only at/near the plant where route risk is minimal.
        int layersToKeep;
        if (columns <= 1_000_000L) layersToKeep = Math.Min(zCount, 3);
        else if (columns <= 25_000_000L) layersToKeep = Math.Min(zCount, 2);
        else layersToKeep = Math.Min(zCount, 1);

        long retained = columns * layersToKeep;
        if (retained > 80_000_000L)
        {
            layersToKeep = Math.Max(1, (int)(80_000_000L / Math.Max(1, columns)));
        }

        // For pathological benchmark dimensions, consuming a multi-terabyte stream cannot be useful.
        // Emit a valid no-op answer instead of exhausting time or memory.
        if (total > 2_000_000_000L || layersToKeep <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        byte[] top = new byte[columns * layersToKeep];

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                long col = ((long)x * yCount + y) * layersToKeep;
                for (int z = 0; z < zCount; z++)
                {
                    int b = input.ReadByteRaw();
                    if (b < 0) b = 0;
                    int fromTop = zCount - 1 - z;
                    if (fromTop >= 0 && fromTop < layersToKeep)
                        top[col + fromTop] = (byte)b;
                }
            }
        }

        var ops = new List<Op>();
        long maxOps = columns <= 200_000L ? 200_000L : 60_000L;
        long maxOutputSteps = 3_000_000L;
        long outputSteps = 0;

        for (int layer = 0; layer < layersToKeep && ops.Count < maxOps; layer++)
        {
            int z = zCount - 1 - layer;
            for (int x = 0; x < xCount && ops.Count < maxOps; x++)
            {
                for (int y = 0; y < yCount && ops.Count < maxOps; y++)
                {
                    long idx = ((long)x * yCount + y) * layersToKeep + layer;
                    byte d = top[idx];
                    byte mass = (byte)(d & 15);
                    byte value = (byte)(d >> 4);
                    if (mass == 0 || value == 0) continue;

                    int dist = x + y;
                    if (dist == 0) continue; // already at the plant; no XY route to describe.

                    // Revenue dwarfs normal haul cost, but keep a profit check and output cap.
                    double fuel = EstimateFuelCost(x, y, mass, xBlock, yBlock, zBlock);
                    double revenue = 1000.0 * mass * value;
                    if (revenue <= fuel) continue;
                    if (outputSteps + dist > maxOutputSteps) continue;

                    // Layer 2 is only safe next to the plant on a flat untouched route; otherwise
                    // the first step could require climbing more than one z voxel.
                    if (layer >= 2 && dist > 1) continue;

                    ops.Add(new Op { X = x, Y = y, Z = z, N = dist, Value = value, Mass = mass });
                    outputSteps += dist;
                }
            }
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        writer.WriteLine(ops.Count);
        var line = new StringBuilder(256);
        foreach (var op in ops)
        {
            line.Clear();
            line.Append(op.N).Append(' ').Append(op.X).Append(' ').Append(op.Y).Append(' ').Append(op.Z);

            int cx = op.X;
            int cy = op.Y;
            while (cx > 0)
            {
                cx--;
                line.Append(' ').Append(cx).Append(' ').Append(cy);
            }
            while (cy > 0)
            {
                cy--;
                line.Append(' ').Append(cx).Append(' ').Append(cy);
            }

            line.Append(" 0 0");
            writer.WriteLine(line.ToString());
        }
    }

    static double EstimateFuelCost(int x, int y, int mass, double xBlock, double yBlock, double zBlock)
    {
        // The default route is flat on the initial surface. Keep block sizes in the estimate so
        // unusual anisotropic test cases still prefer shorter physical hauls.
        double xyCost = x * Math.Max(1.0, xBlock) + y * Math.Max(1.0, yBlock);
        double verticalPenalty = Math.Max(1.0, zBlock) <= 0 ? 1.0 : 1.0;
        return mass * xyCost * verticalPenalty;
    }
}
