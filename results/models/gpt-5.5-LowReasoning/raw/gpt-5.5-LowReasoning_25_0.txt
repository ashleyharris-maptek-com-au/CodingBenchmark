using System;
using System.Collections.Generic;
using System.Globalization;

public class Program
{
    const double G = 6.67430e-11;
    const double Day = 86400.0;
    const double AU = 149597870700.0;

    struct V
    {
        public double X, Y, Z;
        public V(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static V operator +(V a, V b) => new V(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static V operator -(V a, V b) => new V(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static V operator -(V a) => new V(-a.X, -a.Y, -a.Z);
        public static V operator *(V a, double s) => new V(a.X * s, a.Y * s, a.Z * s);
        public static V operator *(double s, V a) => a * s;
        public static V operator /(V a, double s) => new V(a.X / s, a.Y / s, a.Z / s);
        public double N2 => X * X + Y * Y + Z * Z;
        public double N => Math.Sqrt(N2);
        public static double Dot(V a, V b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static V Cross(V a, V b) => new V(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
        public V Unit() { double n = N; return n > 0 ? this / n : new V(); }
    }

    class Body
    {
        public string Name = "";
        public double M;
        public V R, U;
    }

    class Node
    {
        public double T;
        public V Er, Ev, Ar, Av;
    }

    static void Main()
    {
        var toks = Console.In.ReadToEnd().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (toks.Length < 7) return;
        int p = 0;
        int n = int.Parse(toks[p++], CultureInfo.InvariantCulture);
        double warning = D(toks[p++]);
        double impactorMass = D(toks[p++]);
        double budget = D(toks[p++]);
        double asteroidMass = D(toks[p++]);
        p++; // t0_unix_utc; retained only for ephemeris-aware solvers.

        var bodies = new List<Body>();
        Body earth = null, ast = null, sun = null;
        for (int i = 0; i < n && p + 7 < toks.Length; i++)
        {
            var b = new Body();
            b.Name = toks[p++];
            b.M = D(toks[p++]);
            b.R = new V(D(toks[p++]), D(toks[p++]), D(toks[p++]));
            b.U = new V(D(toks[p++]), D(toks[p++]), D(toks[p++]));
            bodies.Add(b);
            string lower = b.Name.ToLowerInvariant();
            if (lower == "earth") earth = b;
            else if (lower == "asteroid") ast = b;
            else if (lower == "sun") sun = b;
        }

        if (earth == null || ast == null)
        {
            Console.WriteLine("1\n0 0 0 0");
            return;
        }
        if (sun == null)
        {
            sun = bodies.Count > 0 ? bodies[0] : new Body { M = 1.98847e30, R = new V(), U = new V() };
        }

        double mu = G * Math.Max(sun.M, 1.98847e30 * 0.1);
        double horizon = Math.Max(1.0, warning);
        var eph = BuildEphemeris(bodies, earth, ast, horizon);
        double impactDay = FindClosestDay(eph, horizon);

        double bestScore = double.PositiveInfinity;
        double bestLaunch = 0, bestImpact = Math.Max(0.1, impactDay * 0.55);
        V bestDv0 = new V(), bestDvc = new V(), bestDvi = new V();
        bool bestHasCorrection = false;

        double[] launchFractions = { 0, 0.02, 0.05, 0.1, 0.18, 0.28, 0.4 };
        double[] interceptFractions = { 0.18, 0.25, 0.33, 0.42, 0.52, 0.62, 0.72, 0.82, 0.9 };

        foreach (double lf in launchFractions)
        {
            double tl = Math.Min(warning, Math.Max(0, impactDay * lf));
            var L = Sample(eph, tl);
            foreach (double q in interceptFractions)
            {
                double ti = Math.Min(warning, Math.Max(tl + 0.02, impactDay * q));
                if (ti <= tl) continue;
                var I = Sample(eph, ti);
                V r1 = L.Er - sun.R;
                V r2 = I.Ar - sun.R;
                if (!Lambert(r1, r2, (ti - tl) * Day, mu, out V v1, out V v2)) continue;
                V dv0 = v1 - (L.Ev - sun.U);
                double dv = dv0.N;
                V relAtImpact = (I.Av - sun.U) - v2;
                double momentumScore = asteroidMass > 0 ? (impactorMass * relAtImpact.N / asteroidMass) : relAtImpact.N;
                double timing = Math.Max(1.0, impactDay - ti);
                double score = Penalty(dv, budget) + 1.0 / (1e-9 + momentumScore * timing) + 1e-7 * ti;
                if (score < bestScore)
                {
                    bestScore = score; bestLaunch = tl; bestImpact = ti; bestDv0 = dv0; bestDvi = new V(); bestHasCorrection = false;
                }

                double tc = (tl + ti) * 0.5;
                V craftMidR = PropagateKepler(r1, v1, (tc - tl) * Day, mu, out V craftMidV) + sun.R;
                var C = Sample(eph, tc);
                if (Lambert(craftMidR - sun.R, r2, (ti - tc) * Day, mu, out V vc1, out V vc2))
                {
                    V corr = vc1 - craftMidV;
                    double dvt = dv0.N + corr.N;
                    V rel2 = (I.Av - sun.U) - vc2;
                    double mom2 = asteroidMass > 0 ? (impactorMass * rel2.N / asteroidMass) : rel2.N;
                    double score2 = Penalty(dvt, budget) + 1.0 / (1e-9 + mom2 * timing) + 1e-7 * ti + 1e-12 * (craftMidR - C.Ar).N;
                    if (score2 < bestScore)
                    {
                        bestScore = score2; bestLaunch = tl; bestImpact = ti; bestDv0 = dv0; bestDvc = corr; bestDvi = new V(); bestHasCorrection = true;
                    }
                }
            }
        }

        double used = bestDv0.N + (bestHasCorrection ? bestDvc.N : 0.0);
        if (used > budget && used > 0)
        {
            double s = Math.Max(0.0, budget * 0.995 / used);
            bestDv0 *= s;
            bestDvc *= s;
        }

        if (bestHasCorrection)
        {
            Console.WriteLine("3");
            Print(bestLaunch, bestDv0);
            Print((bestLaunch + bestImpact) * 0.5, bestDvc);
            Print(bestImpact, bestDvi);
        }
        else
        {
            Console.WriteLine("2");
            Print(bestLaunch, bestDv0);
            Print(bestImpact, bestDvi);
        }
    }

    static double D(string s) => double.Parse(s, NumberStyles.Float, CultureInfo.InvariantCulture);
    static double Penalty(double dv, double budget) => dv <= budget ? dv / Math.Max(1.0, budget) : 1e6 + (dv - budget) * (dv - budget);
    static void Print(double t, V v) => Console.WriteLine(string.Format(CultureInfo.InvariantCulture, "{0:R} {1:R} {2:R} {3:R}", t, v.X, v.Y, v.Z));

    static List<Node> BuildEphemeris(List<Body> bodies, Body earth, Body ast, double days)
    {
        var work = new List<Body>();
        foreach (var b in bodies) work.Add(new Body { Name = b.Name, M = b.M, R = b.R, U = b.U });
        int ei = bodies.IndexOf(earth), ai = bodies.IndexOf(ast);
        var res = new List<Node>();
        double dtDay = days > 4000 ? 2.0 : (days > 1200 ? 1.0 : 0.25);
        double t = 0;
        while (t <= days + 1e-9)
        {
            res.Add(new Node { T = t, Er = work[ei].R, Ev = work[ei].U, Ar = work[ai].R, Av = work[ai].U });
            double step = Math.Min(dtDay, days - t);
            if (step <= 1e-12) break;
            StepRK4(work, step * Day);
            t += step;
        }
        return res;
    }

    static void StepRK4(List<Body> b, double h)
    {
        int n = b.Count;
        V[] r0 = new V[n], v0 = new V[n], a0, a1, a2, a3, rt = new V[n], vt = new V[n];
        for (int i = 0; i < n; i++) { r0[i] = b[i].R; v0[i] = b[i].U; }
        a0 = Accel(b, r0);
        for (int i = 0; i < n; i++) { rt[i] = r0[i] + v0[i] * (h * 0.5); vt[i] = v0[i] + a0[i] * (h * 0.5); }
        a1 = Accel(b, rt);
        for (int i = 0; i < n; i++) { rt[i] = r0[i] + vt[i] * (h * 0.5); vt[i] = v0[i] + a1[i] * (h * 0.5); }
        a2 = Accel(b, rt);
        for (int i = 0; i < n; i++) { rt[i] = r0[i] + vt[i] * h; vt[i] = v0[i] + a2[i] * h; }
        a3 = Accel(b, rt);
        for (int i = 0; i < n; i++)
        {
            b[i].R = r0[i] + (v0[i] + 2.0 * (v0[i] + a0[i] * (h * 0.5)) + 2.0 * (v0[i] + a1[i] * (h * 0.5)) + (v0[i] + a2[i] * h)) * (h / 6.0);
            b[i].U = v0[i] + (a0[i] + 2.0 * a1[i] + 2.0 * a2[i] + a3[i]) * (h / 6.0);
        }
    }

    static V[] Accel(List<Body> b, V[] r)
    {
        int n = b.Count; var a = new V[n];
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++) if (i != j && b[j].M > 0)
            {
                V d = r[j] - r[i];
                double r2 = d.N2 + 1e6;
                a[i] += d * (G * b[j].M / (r2 * Math.Sqrt(r2)));
            }
        return a;
    }

    static Node Sample(List<Node> e, double t)
    {
        if (t <= e[0].T) return e[0];
        if (t >= e[e.Count - 1].T) return e[e.Count - 1];
        int lo = 0, hi = e.Count - 1;
        while (hi - lo > 1) { int m = (lo + hi) >> 1; if (e[m].T < t) lo = m; else hi = m; }
        var a = e[lo]; var b = e[hi]; double f = (t - a.T) / (b.T - a.T);
        return new Node { T = t, Er = a.Er + (b.Er - a.Er) * f, Ev = a.Ev + (b.Ev - a.Ev) * f, Ar = a.Ar + (b.Ar - a.Ar) * f, Av = a.Av + (b.Av - a.Av) * f };
    }

    static double FindClosestDay(List<Node> e, double horizon)
    {
        double bt = horizon, bd = double.PositiveInfinity;
        foreach (var x in e)
        {
            double d = (x.Ar - x.Er).N2;
            if (d < bd) { bd = d; bt = x.T; }
        }
        return Math.Max(0.1, Math.Min(horizon, bt));
    }

    static bool Lambert(V r1, V r2, double tof, double mu, out V v1, out V v2)
    {
        v1 = new V(); v2 = new V();
        double R1 = r1.N, R2 = r2.N;
        if (R1 <= 0 || R2 <= 0 || tof <= 0) return false;
        double cos = Math.Max(-1, Math.Min(1, V.Dot(r1, r2) / (R1 * R2)));
        double sin = V.Cross(r1, r2).Z >= 0 ? Math.Sqrt(Math.Max(0, 1 - cos * cos)) : -Math.Sqrt(Math.Max(0, 1 - cos * cos));
        if (Math.Abs(1 - cos) < 1e-12) cos = 1 - 1e-12;
        double A = sin * Math.Sqrt(R1 * R2 / (1 - cos));
        if (Math.Abs(A) < 1e-9) return false;
        double z = 0, lo = -4 * Math.PI * Math.PI, hi = 4 * Math.PI * Math.PI;
        for (int it = 0; it < 80; it++)
        {
            double C = StumpC(z), S = StumpS(z);
            double y = R1 + R2 + A * (z * S - 1) / Math.Sqrt(C);
            if (A > 0 && y < 0) { lo = z; z = (z + hi) * 0.5; continue; }
            double x = Math.Sqrt(Math.Max(0, y / C));
            double t = (x * x * x * S + A * Math.Sqrt(Math.Max(0, y))) / Math.Sqrt(mu);
            if (double.IsNaN(t) || double.IsInfinity(t)) return false;
            if (t <= tof) lo = z; else hi = z;
            z = (lo + hi) * 0.5;
        }
        double Cf = StumpC(z), Sf = StumpS(z);
        double yf = R1 + R2 + A * (z * Sf - 1) / Math.Sqrt(Cf);
        if (yf <= 0) return false;
        double f = 1 - yf / R1;
        double g = A * Math.Sqrt(yf / mu);
        double gd = 1 - yf / R2;
        if (Math.Abs(g) < 1e-9) return false;
        v1 = (r2 - r1 * f) / g;
        v2 = (r2 * gd - r1) / g;
        return v1.N < 1e6 && v2.N < 1e6;
    }

    static double StumpC(double z)
    {
        if (z > 1e-8) return (1 - Math.Cos(Math.Sqrt(z))) / z;
        if (z < -1e-8) return (Math.Cosh(Math.Sqrt(-z)) - 1) / (-z);
        return 0.5 - z / 24.0 + z * z / 720.0;
    }

    static double StumpS(double z)
    {
        if (z > 1e-8) { double s = Math.Sqrt(z); return (s - Math.Sin(s)) / (s * s * s); }
        if (z < -1e-8) { double s = Math.Sqrt(-z); return (Math.Sinh(s) - s) / (s * s * s); }
        return 1.0 / 6.0 - z / 120.0 + z * z / 5040.0;
    }

    static V PropagateKepler(V r, V v, double dt, double mu, out V vv)
    {
        int steps = Math.Max(1, Math.Min(2000, (int)Math.Ceiling(Math.Abs(dt) / (2 * Day))));
        double h = dt / steps;
        V rr = r, uu = v;
        for (int i = 0; i < steps; i++)
        {
            V a1 = TwoAcc(rr, mu);
            V k1r = uu, k1v = a1;
            V k2r = uu + k1v * (h * 0.5), k2v = TwoAcc(rr + k1r * (h * 0.5), mu);
            V k3r = uu + k2v * (h * 0.5), k3v = TwoAcc(rr + k2r * (h * 0.5), mu);
            V k4r = uu + k3v * h, k4v = TwoAcc(rr + k3r * h, mu);
            rr += (k1r + 2 * k2r + 2 * k3r + k4r) * (h / 6.0);
            uu += (k1v + 2 * k2v + 2 * k3v + k4v) * (h / 6.0);
        }
        vv = uu;
        return rr;
    }

    static V TwoAcc(V r, double mu)
    {
        double n2 = r.N2 + 1e6;
        return r * (-mu / (n2 * Math.Sqrt(n2)));
    }
}
