use std::io::{self, BufReader, Read, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy)]
struct Config {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
    zone_left: f64,
    zone_right: f64,
}

#[derive(Clone, Copy)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    seen: bool,
}

struct Scanner<R: Read> {
    reader: BufReader<R>,
    buf: Vec<u8>,
}

impl<R: Read> Scanner<R> {
    fn new(reader: R) -> Self {
        Self {
            reader: BufReader::with_capacity(1 << 20, reader),
            buf: Vec::with_capacity(64),
        }
    }

    fn next(&mut self) -> Option<String> {
        self.buf.clear();
        let mut byte = [0u8; 1];
        loop {
            match self.reader.read(&mut byte) {
                Ok(0) => return None,
                Ok(_) if !byte[0].is_ascii_whitespace() => {
                    self.buf.push(byte[0]);
                    break;
                }
                Ok(_) => {}
                Err(_) => return None,
            }
        }
        loop {
            match self.reader.read(&mut byte) {
                Ok(0) => break,
                Ok(_) if byte[0].is_ascii_whitespace() => break,
                Ok(_) => self.buf.push(byte[0]),
                Err(_) => return None,
            }
        }
        String::from_utf8(self.buf.clone()).ok()
    }

    fn f64(&mut self) -> Option<f64> {
        self.next()?.parse().ok()
    }

    fn usize(&mut self) -> Option<usize> {
        self.next()?.parse().ok()
    }
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

fn wrap_pi(mut a: f64) -> f64 {
    while a > std::f64::consts::PI {
        a -= 2.0 * std::f64::consts::PI;
    }
    while a < -std::f64::consts::PI {
        a += 2.0 * std::f64::consts::PI;
    }
    a
}

fn command(cfg: Config, s: State) -> (f64, f64) {
    if !s.seen || cfg.max_thrust <= 0.0 {
        return (0.0, 0.0);
    }

    let pad_x = clamp(cfg.target_x, cfg.zone_left + 0.4, cfg.zone_right - 0.4);
    let dx = pad_x - s.x;
    let altitude = (cfg.target_y - s.y).max(0.0);

    let mut desired_vx = clamp(dx * 0.22, -10.0, 10.0);
    if altitude < 18.0 {
        desired_vx = clamp(dx * 0.35, -3.0, 3.0);
    }
    if altitude < 6.0 {
        desired_vx = clamp(dx * 0.45, -1.2, 1.2);
    }

    let desired_ax = clamp((desired_vx - s.vx) * 0.55, -cfg.max_thrust * 0.45, cfg.max_thrust * 0.45);
    let mut desired_angle = (desired_ax / cfg.max_thrust).asin();
    let angle_limit = if altitude < 12.0 { 0.28 } else { 0.72 };
    desired_angle = clamp(desired_angle, -angle_limit, angle_limit);

    let angle_err = wrap_pi(desired_angle - s.angle);
    let turn = clamp(angle_err * 3.8 - s.angular_vel * 1.4, -1.0, 1.0);

    let target_descent = if altitude > 80.0 {
        clamp(altitude * 0.055 + 2.0, 4.0, 18.0)
    } else if altitude > 25.0 {
        clamp(altitude * 0.075 + 1.0, 2.2, 8.0)
    } else if altitude > 8.0 {
        clamp(altitude * 0.13 + 0.4, 1.1, 3.0)
    } else {
        clamp(altitude * 0.16 + 0.25, 0.35, 1.35)
    };

    let vertical_err = target_descent - s.vy;
    let desired_ay = clamp(vertical_err * 0.85, -cfg.max_thrust * 0.9, cfg.max_thrust * 0.45);
    let c = s.angle.cos().max(0.18);
    let mut thrust = (cfg.gravity - desired_ay) / (cfg.max_thrust * c);

    let speed = (s.vx * s.vx + s.vy * s.vy).sqrt();
    if altitude < 15.0 && (s.vy > 2.0 || speed > 3.0) {
        thrust = thrust.max(0.82);
    }
    if altitude < 5.0 && (s.vy > 1.35 || speed > 2.0) {
        thrust = 1.0;
    }
    if s.fuel < 0.2 {
        thrust = 0.0;
    }

    (clamp(thrust, 0.0, 1.0), turn)
}

fn main() {
    let stdin = io::stdin();
    let mut sc = Scanner::new(stdin.lock());

    let width = match sc.usize() {
        Some(v) => v,
        None => return,
    };
    let height = match sc.usize() {
        Some(v) => v,
        None => return,
    };
    let gravity = sc.f64().unwrap_or(1.62);
    let max_thrust = sc.f64().unwrap_or(4.0);
    let _initial_fuel = sc.f64().unwrap_or(0.0);

    let _start_x = sc.f64().unwrap_or(0.0);
    let _start_y = sc.f64().unwrap_or(0.0);
    let target_x = sc.f64().unwrap_or(0.0);
    let target_y_input = sc.f64().unwrap_or(0.0);

    let mut zone_left = target_x - 5.0;
    let mut zone_right = target_x + 5.0;
    let mut zone_y = target_y_input;
    let mut found_zone = false;

    for y in 0..height {
        let row = match sc.next() {
            Some(r) => r,
            None => break,
        };
        let bytes = row.as_bytes();
        let scan_len = bytes.len().min(width);
        for x in 0..scan_len {
            let b = bytes[x];
            if b == b'L' || b == b'T' {
                if !found_zone {
                    zone_left = x as f64;
                    zone_right = x as f64;
                    zone_y = y as f64;
                    found_zone = true;
                } else {
                    zone_left = zone_left.min(x as f64);
                    zone_right = zone_right.max(x as f64);
                    zone_y = zone_y.max(y as f64);
                }
            }
        }
    }

    let cfg = Config {
        gravity,
        max_thrust,
        target_x,
        target_y: zone_y,
        zone_left,
        zone_right,
    };
    let state = Arc::new(Mutex::new(State {
        x: target_x,
        y: target_y_input,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: 0.0,
        seen: false,
    }));

    let out_state = Arc::clone(&state);
    thread::spawn(move || {
        let stdout = io::stdout();
        let mut out = io::BufWriter::new(stdout.lock());
        let tick = Duration::from_millis(40);
        loop {
            let now = Instant::now();
            let s = *out_state.lock().unwrap();
            let (thrust, turn) = command(cfg, s);
            if writeln!(out, "{:.6} {:.6}", thrust, turn).is_err() {
                break;
            }
            let _ = out.flush();
            let elapsed = now.elapsed();
            if elapsed < tick {
                thread::sleep(tick - elapsed);
            }
        }
    });

    while let Some(x) = sc.f64() {
        let y = match sc.f64() { Some(v) => v, None => break };
        let vx = match sc.f64() { Some(v) => v, None => break };
        let vy = match sc.f64() { Some(v) => v, None => break };
        let angle = match sc.f64() { Some(v) => v, None => break };
        let angular_vel = match sc.f64() { Some(v) => v, None => break };
        let fuel = match sc.f64() { Some(v) => v, None => break };

        let mut s = state.lock().unwrap();
        *s = State {
            x,
            y,
            vx,
            vy,
            angle,
            angular_vel,
            fuel,
            seen: true,
        };
    }
}
