using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private struct Move
    {
        public int Rotation;
        public int Column;
        public double Score;
        public int Lines;
        public bool Valid;
    }

    private sealed class Shape
    {
        public readonly int Width;
        public readonly int Height;
        public readonly int[] X;
        public readonly int[] Y;
        public readonly int[] Bottom;
        public readonly int[] Top;

        public Shape(params (int x, int y)[] cells)
        {
            X = new int[4];
            Y = new int[4];
            int maxX = 0, maxY = 0;
            for (int i = 0; i < 4; i++)
            {
                X[i] = cells[i].x;
                Y[i] = cells[i].y;
                if (X[i] > maxX) maxX = X[i];
                if (Y[i] > maxY) maxY = Y[i];
            }
            Width = maxX + 1;
            Height = maxY + 1;
            Bottom = new int[Width];
            Top = new int[Width];
            for (int c = 0; c < Width; c++)
            {
                Bottom[c] = int.MaxValue;
                Top[c] = int.MinValue;
            }
            for (int i = 0; i < 4; i++)
            {
                if (Y[i] < Bottom[X[i]]) Bottom[X[i]] = Y[i];
                if (Y[i] > Top[X[i]]) Top[X[i]] = Y[i];
            }
        }
    }

    private static int W, H;
    private static bool[,] board = null!;
    private static int[] heights = null!;
    private static int piecesSeen;
    private static int recentClears;
    private static readonly Dictionary<char, Shape[]> Shapes = BuildShapes();

    public static void Main()
    {
        var fs = new FastScanner();
        string? sw = fs.Next();
        if (sw == null) return;
        W = int.Parse(sw, CultureInfo.InvariantCulture);
        H = int.Parse(fs.Next() ?? "20", CultureInfo.InvariantCulture);
        int n = int.Parse(fs.Next() ?? "0", CultureInfo.InvariantCulture);

        board = new bool[H + 8, W];
        heights = new int[W];

        for (int i = 0; i < n; i++)
        {
            string? token = fs.Next();
            if (token == null) break;
            char piece = char.ToUpperInvariant(token[0]);

            Move best = ChooseMove(piece);
            if (!best.Valid)
                best = EmergencyMove(piece);

            Console.WriteLine(best.Rotation.ToString(CultureInfo.InvariantCulture) + " " + best.Column.ToString(CultureInfo.InvariantCulture));
            Console.Out.Flush();

            int predictedLines;
            bool placed = ApplyMove(board, heights, piece, best.Rotation, best.Column, out predictedLines);
            string? result = fs.Next();
            if (result == null) break;
            if (result == "gameover") break;

            int reportedLines = 0;
            if (result.StartsWith("cleared_", StringComparison.Ordinal))
                int.TryParse(result.Substring(8), NumberStyles.Integer, CultureInfo.InvariantCulture, out reportedLines);
            if (!placed || reportedLines != predictedLines)
            {
                if (placed && reportedLines > predictedLines)
                    recentClears += reportedLines - predictedLines;
                RecomputeHeights();
            }

            piecesSeen++;
            recentClears = recentClears * 7 / 8 + reportedLines;
        }
    }

    private static Move ChooseMove(char piece)
    {
        bool marathon = piecesSeen > 250 || H >= 24 || W >= 14;
        bool danger = MaxHeight(heights) > H * 7 / 10;
        bool useLookahead = W <= 12 && !danger && piecesSeen < 120;

        double[] weights = marathon
            ? new[] { -0.510066, 0.760666, -0.35663, -0.184483, -0.22, -0.42, -0.12 }
            : danger
                ? new[] { -0.82, 0.62, -0.46, -0.31, -0.60, -0.72, -0.20 }
                : new[] { -0.55, 0.72, -0.37, -0.20, -0.28, -0.48, -0.12 };

        Move best = new Move { Score = double.NegativeInfinity };
        Shape[] rots = Shapes[piece];
        for (int r = 0; r < rots.Length; r++)
        {
            Shape s = rots[r];
            for (int col = 0; col <= W - s.Width; col++)
            {
                bool[,] b2 = CloneBoard(board);
                int[] h2 = (int[])heights.Clone();
                int lines;
                if (!ApplyMove(b2, h2, piece, r, col, out lines)) continue;
                double score = Evaluate(b2, h2, lines, weights);
                if (useLookahead)
                    score += 0.18 * AverageNextPieceValue(b2, h2, weights);
                if (score > best.Score)
                    best = new Move { Rotation = r, Column = col, Score = score, Lines = lines, Valid = true };
            }
        }
        return best;
    }

    private static double AverageNextPieceValue(bool[,] b, int[] h, double[] weights)
    {
        char[] bag = { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };
        double sum = 0;
        foreach (char p in bag)
        {
            double best = -1.0e100;
            Shape[] rots = Shapes[p];
            for (int r = 0; r < rots.Length; r++)
            {
                Shape s = rots[r];
                for (int col = 0; col <= W - s.Width; col++)
                {
                    bool[,] b2 = CloneBoard(b);
                    int[] h2 = (int[])h.Clone();
                    int lines;
                    if (ApplyMove(b2, h2, p, r, col, out lines))
                    {
                        double v = Evaluate(b2, h2, lines, weights);
                        if (v > best) best = v;
                    }
                }
            }
            sum += best < -1.0e90 ? -5000 : best;
        }
        return sum / bag.Length;
    }

    private static Move EmergencyMove(char piece)
    {
        Shape[] rots = Shapes[piece];
        int bestR = 0, bestC = 0, bestLanding = int.MaxValue;
        for (int r = 0; r < rots.Length; r++)
        {
            Shape s = rots[r];
            for (int col = 0; col <= W - s.Width; col++)
            {
                int y = LandingY(heights, s, col);
                if (y + s.Height <= H + 4 && y < bestLanding)
                {
                    bestLanding = y;
                    bestR = r;
                    bestC = col;
                }
            }
        }
        return new Move { Rotation = bestR, Column = bestC, Valid = true };
    }

    private static bool ApplyMove(bool[,] b, int[] h, char piece, int rot, int col, out int lines)
    {
        lines = 0;
        Shape s = Shapes[piece][rot % Shapes[piece].Length];
        if (col < 0 || col + s.Width > W) return false;
        int y = LandingY(h, s, col);
        if (y + s.Height > H + 7) return false;
        for (int i = 0; i < 4; i++)
        {
            int x = col + s.X[i];
            int yy = y + s.Y[i];
            if (yy < 0 || yy >= H + 8 || b[yy, x]) return false;
            b[yy, x] = true;
        }

        for (int row = y; row < Math.Min(H + 8, y + s.Height); row++)
        {
            bool full = true;
            for (int x = 0; x < W; x++)
            {
                if (!b[row, x]) { full = false; break; }
            }
            if (!full) continue;
            lines++;
            for (int yy = row; yy < H + 7; yy++)
                for (int x = 0; x < W; x++)
                    b[yy, x] = b[yy + 1, x];
            for (int x = 0; x < W; x++) b[H + 7, x] = false;
            row--;
        }
        RecomputeHeights(b, h);
        return MaxHeight(h) <= H + 4;
    }

    private static int LandingY(int[] h, Shape s, int col)
    {
        int y = 0;
        for (int c = 0; c < s.Width; c++)
        {
            int needed = h[col + c] - s.Bottom[c];
            if (needed > y) y = needed;
        }
        return y;
    }

    private static double Evaluate(bool[,] b, int[] h, int lines, double[] w)
    {
        int aggregate = 0, bump = 0, holes = 0, wells = 0, blockade = 0;
        int max = 0, min = int.MaxValue;
        for (int x = 0; x < W; x++)
        {
            int hx = h[x];
            aggregate += hx;
            if (hx > max) max = hx;
            if (hx < min) min = hx;
            bool seenBlock = false;
            for (int y = H + 7; y >= 0; y--)
            {
                if (b[y, x]) seenBlock = true;
                else if (seenBlock)
                {
                    holes++;
                    int above = 0;
                    for (int yy = y + 1; yy < H + 8 && b[yy, x]; yy++) above++;
                    blockade += above;
                }
            }
        }
        for (int x = 0; x < W - 1; x++)
            bump += Math.Abs(h[x] - h[x + 1]);
        for (int x = 0; x < W; x++)
        {
            int left = x == 0 ? H : h[x - 1];
            int right = x == W - 1 ? H : h[x + 1];
            int depth = Math.Min(left, right) - h[x];
            if (depth > 1) wells += depth * (depth + 1) / 2;
        }

        int danger = Math.Max(0, max - (H * 3 / 4));
        int relief = (max - min);
        return w[0] * aggregate
             + w[1] * lines * lines
             + w[2] * holes
             + w[3] * bump
             + w[4] * wells
             + w[5] * blockade
             + w[6] * relief
             - danger * danger * 2.0;
    }

    private static void RecomputeHeights()
    {
        RecomputeHeights(board, heights);
    }

    private static void RecomputeHeights(bool[,] b, int[] h)
    {
        for (int x = 0; x < W; x++)
        {
            int y = H + 7;
            while (y >= 0 && !b[y, x]) y--;
            h[x] = y + 1;
        }
    }

    private static int MaxHeight(int[] h)
    {
        int m = 0;
        for (int i = 0; i < h.Length; i++) if (h[i] > m) m = h[i];
        return m;
    }

    private static bool[,] CloneBoard(bool[,] src)
    {
        var dst = new bool[H + 8, W];
        Array.Copy(src, dst, src.Length);
        return dst;
    }

    private static Dictionary<char, Shape[]> BuildShapes()
    {
        return new Dictionary<char, Shape[]>
        {
            ['I'] = Unique(new Shape( (0,0), (1,0), (2,0), (3,0) ),
                          new Shape( (0,0), (0,1), (0,2), (0,3) )),
            ['O'] = Unique(new Shape( (0,0), (1,0), (0,1), (1,1) )),
            ['T'] = Unique(new Shape( (0,0), (1,0), (2,0), (1,1) ),
                          new Shape( (0,0), (0,1), (0,2), (1,1) ),
                          new Shape( (0,1), (1,1), (2,1), (1,0) ),
                          new Shape( (1,0), (1,1), (1,2), (0,1) )),
            ['S'] = Unique(new Shape( (1,0), (2,0), (0,1), (1,1) ),
                          new Shape( (0,0), (0,1), (1,1), (1,2) )),
            ['Z'] = Unique(new Shape( (0,0), (1,0), (1,1), (2,1) ),
                          new Shape( (1,0), (0,1), (1,1), (0,2) )),
            ['J'] = Unique(new Shape( (0,0), (1,0), (2,0), (0,1) ),
                          new Shape( (0,0), (0,1), (0,2), (1,0) ),
                          new Shape( (0,0), (1,0), (2,0), (2,1) ),
                          new Shape( (1,0), (1,1), (1,2), (0,2) )),
            ['L'] = Unique(new Shape( (0,0), (1,0), (2,0), (2,1) ),
                          new Shape( (0,0), (1,0), (0,1), (0,2) ),
                          new Shape( (0,0), (0,1), (1,1), (2,1) ),
                          new Shape( (1,0), (1,1), (1,2), (0,0) ))
        };
    }

    private static Shape[] Unique(params Shape[] shapes)
    {
        return shapes;
    }

    private sealed class FastScanner
    {
        private readonly System.IO.TextReader input = Console.In;
        private readonly char[] buffer = new char[1 << 12];
        private int len, ptr;

        public string? Next()
        {
            var chars = new List<char>(16);
            int c;
            do
            {
                c = Read();
                if (c < 0) return null;
            } while (char.IsWhiteSpace((char)c));

            while (c >= 0 && !char.IsWhiteSpace((char)c))
            {
                chars.Add((char)c);
                c = Read();
            }
            return new string(chars.ToArray());
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = input.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }
    }
}
