using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    struct Cut
    {
        public int Length;
        public int Index;
    }

    sealed class CutComparer : IComparer<Cut>
    {
        public int Compare(Cut a, Cut b)
        {
            int c = b.Length.CompareTo(a.Length);
            return c != 0 ? c : a.Index.CompareTo(b.Index);
        }
    }

    sealed class FastScanner
    {
        readonly Stream input = Console.OpenStandardInput();
        readonly byte[] buffer = new byte[1 << 20];
        int len, ptr;

        int ReadByteFast()
        {
            if (ptr >= len)
            {
                len = input.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByteFast(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByteFast();
            }
            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = ReadByteFast();
            }
            return v * sign;
        }
    }

    sealed class Treap
    {
        sealed class Node
        {
            public int Key, Priority;
            public Node? Left, Right;
            public readonly List<int> Bins = new List<int>(1);

            public Node(int key, int priority, int bin)
            {
                Key = key;
                Priority = priority;
                Bins.Add(bin);
            }
        }

        Node? root;
        uint rng = 2463534242u;

        int NextPriority()
        {
            uint x = rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            rng = x;
            return unchecked((int)x);
        }

        static Node RotateRight(Node y)
        {
            Node x = y.Left!;
            y.Left = x.Right;
            x.Right = y;
            return x;
        }

        static Node RotateLeft(Node x)
        {
            Node y = x.Right!;
            x.Right = y.Left;
            y.Left = x;
            return y;
        }

        public void Insert(int key, int bin)
        {
            root = Insert(root, key, bin);
        }

        Node Insert(Node? n, int key, int bin)
        {
            if (n == null) return new Node(key, NextPriority(), bin);
            if (key == n.Key)
            {
                n.Bins.Add(bin);
            }
            else if (key < n.Key)
            {
                n.Left = Insert(n.Left, key, bin);
                if (n.Left!.Priority < n.Priority) n = RotateRight(n);
            }
            else
            {
                n.Right = Insert(n.Right, key, bin);
                if (n.Right!.Priority < n.Priority) n = RotateLeft(n);
            }
            return n;
        }

        public int PopLowerBound(int need, out int key)
        {
            Node? n = root;
            Node? best = null;
            while (n != null)
            {
                if (n.Key >= need)
                {
                    best = n;
                    n = n.Left;
                }
                else
                {
                    n = n.Right;
                }
            }

            if (best == null)
            {
                key = 0;
                return -1;
            }

            key = best.Key;
            int last = best.Bins.Count - 1;
            int bin = best.Bins[last];
            best.Bins.RemoveAt(last);
            if (best.Bins.Count == 0) root = RemoveNode(root, key);
            return bin;
        }

        static Node? RemoveNode(Node? n, int key)
        {
            if (n == null) return null;
            if (key < n.Key)
            {
                n.Left = RemoveNode(n.Left, key);
            }
            else if (key > n.Key)
            {
                n.Right = RemoveNode(n.Right, key);
            }
            else
            {
                if (n.Left == null) return n.Right;
                if (n.Right == null) return n.Left;
                if (n.Left.Priority < n.Right.Priority)
                {
                    n = RotateRight(n);
                    n.Right = RemoveNode(n.Right, key);
                }
                else
                {
                    n = RotateLeft(n);
                    n.Left = RemoveNode(n.Left, key);
                }
            }
            return n;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int stockLength = fs.NextInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i].Length = fs.NextInt();
            cuts[i].Index = i;
        }

        Array.Sort(cuts, new CutComparer());

        int[] next = new int[n];
        int[] heads = new int[Math.Max(4, Math.Min(n, 1024))];
        int[] tails = new int[heads.Length];
        int[] remaining = new int[heads.Length];
        Array.Fill(next, -1);

        int stockCount = 0;
        var freeByRemaining = new Treap();

        for (int i = 0; i < n; i++)
        {
            int length = cuts[i].Length;
            int originalIndex = cuts[i].Index;

            int oldRemaining;
            int stock = freeByRemaining.PopLowerBound(length, out oldRemaining);
            if (stock < 0)
            {
                stock = stockCount++;
                if (stock == heads.Length)
                {
                    int newSize = heads.Length * 2;
                    Array.Resize(ref heads, newSize);
                    Array.Resize(ref tails, newSize);
                    Array.Resize(ref remaining, newSize);
                }
                heads[stock] = tails[stock] = originalIndex;
                remaining[stock] = stockLength - length;
            }
            else
            {
                next[tails[stock]] = originalIndex;
                tails[stock] = originalIndex;
                remaining[stock] = oldRemaining - length;
            }

            if (remaining[stock] > 0) freeByRemaining.Insert(remaining[stock], stock);
        }

        using var output = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 1 << 20);
        output.WriteLine(stockCount);
        for (int s = 0; s < stockCount; s++)
        {
            bool first = true;
            for (int idx = heads[s]; idx >= 0; idx = next[idx])
            {
                if (!first) output.Write(' ');
                first = false;
                output.Write(idx);
            }
            output.WriteLine();
        }
    }
}
