using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class MinesweeperSolver
{
    public static void Main(string[] args)
    {
        // Increase buffer for massive input parsing
        using (var stdin = Console.OpenStandardInput())
        using (var reader = new StreamReader(stdin, Encoding.UTF8, false, 1024 * 1024))
        {
            string firstLine = reader.ReadLine();
            if (string.IsNullOrEmpty(firstLine)) return;
            var hw = firstLine.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            if (hw.Length < 2) return;
            int H = int.Parse(hw[0]);
            int W = int.Parse(hw[1]);

            // Board mapping: 0-8: numbers, 9: unknown (' '), 10: mine (flagged '*' or found), 11: found safe
            byte[] board = new byte[H * W];
            for (int r = 0; r < H; r++)
            {
                string line = reader.ReadLine();
                if (line == null) break;
                for (int c = 0; c < W && c < line.Length; c++)
                {
                    char ch = line[c];
                    if (ch == ' ') board[r * W + c] = 9;
                    else if (ch == '*') board[r * W + c] = 10;
                    else if (ch >= '0' && ch <= '8') board[r * W + c] = (byte)(ch - '0');
                    else board[r * W + c] = 9;
                }
            }

            int[] minesRemaining = new int[H * W];
            int[] unknownsCount = new int[H * W];
            Queue<int> changedCells = new Queue<int>();

            // Phase 1: Initial Scan and Basic Rule Propagation
            for (int r = 0; r < H; r++)
            {
                for (int c = 0; c < W; c++)
                {
                    int idx = r * W + c;
                    if (board[idx] <= 8)
                    {
                        int rem = board[idx];
                        int unknowns = 0;
                        for (int dr = -1; dr <= 1; dr++)
                        {
                            for (int dc = -1; dc <= 1; dc++)
                            {
                                if (dr == 0 && dc == 0) continue;
                                int nr = r + dr, nc = c + dc;
                                if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                                {
                                    int nidx = nr * W + nc;
                                    if (board[nidx] == 10) rem--;
                                    else if (board[nidx] == 9) unknowns++;
                                }
                            }
                        }
                        minesRemaining[idx] = rem;
                        unknownsCount[idx] = unknowns;

                        if (unknowns > 0 && (rem == 0 || rem == unknowns))
                        {
                            byte target = (rem == 0) ? (byte)11 : (byte)10;
                            for (int dr = -1; dr <= 1; dr++)
                            {
                                for (int dc = -1; dc <= 1; dc++)
                                {
                                    if (dr == 0 && dc == 0) continue;
                                    int nr = r + dr, nc = c + dc;
                                    if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                                    {
                                        int nidx = nr * W + nc;
                                        if (board[nidx] == 9)
                                        {
                                            board[nidx] = target;
                                            changedCells.Enqueue(nidx);
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }

            // Phase 2: Iterative Propagation of simple deductions
            while (changedCells.Count > 0)
            {
                int uidx = changedCells.Dequeue();
                bool isMine = (board[uidx] == 10);
                int r = uidx / W, c = uidx % W;

                for (int dr = -1; dr <= 1; dr++)
                {
                    for (int dc = -1; dc <= 1; dc++)
                    {
                        if (dr == 0 && dc == 0) continue;
                        int nr = r + dr, nc = c + dc;
                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                        {
                            int nidx = nr * W + nc;
                            if (board[nidx] <= 8)
                            {
                                if (isMine) minesRemaining[nidx]--;
                                unknownsCount[nidx]--;

                                int rem = minesRemaining[nidx];
                                int unknowns = unknownsCount[nidx];
                                if (unknowns > 0 && (rem == 0 || rem == unknowns))
                                {
                                    byte target = (rem == 0) ? (byte)11 : (byte)10;
                                    for (int ddr = -1; ddr <= 1; ddr++)
                                    {
                                        for (int ddc = -1; ddc <= 1; ddc++)
                                        {
                                            if (ddr == 0 && ddc == 0) continue;
                                            int nnr = nr + ddr, nnc = nc + ddc;
                                            if (nnr >= 0 && nnr < H && nnc >= 0 && nnc < W)
                                            {
                                                int nnidx = nnr * W + nnc;
                                                if (board[nnidx] == 9)
                                                {
                                                    board[nnidx] = target;
                                                    changedCells.Enqueue(nnidx);
                                                }
                                            }
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }

            // Phase 3: Advanced Coupled Rules for Complex Patterns
            // Identify active numbered cells that still have unknown neighbors
            List<int> activeIndices = new List<int>();
            for (int i = 0; i < H * W; i++)
            {
                if (board[i] <= 8 && unknownsCount[i] > 0) activeIndices.Add(i);
            }

            if (activeIndices.Count > 0)
            {
                // Build efficient spatial lookup for coupled rules
                int totalUnknownRefs = 0;
                foreach (int idx in activeIndices) totalUnknownRefs += unknownsCount[idx];
                int[] unFlat = new int[totalUnknownRefs];
                int[] activeToUnOffset = new int[H * W];
                int currentUn = 0;
                foreach (int idx in activeIndices)
                {
                    activeToUnOffset[idx] = currentUn;
                    int r = idx / W, c = idx % W;
                    for (int dr = -1; dr <= 1; dr++)
                    {
                        for (int dc = -1; dc <= 1; dc++)
                        {
                            if (dr == 0 && dc == 0) continue;
                            int nr = r + dr, nc = c + dc;
                            if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                            {
                                int nidx = nr * W + nc;
                                if (board[nidx] == 9) unFlat[currentUn++] = nidx;
                            }
                        }
                    }
                }

                int[] uCounts = new int[H * W];
                for (int i = 0; i < currentUn; i++) uCounts[unFlat[i]]++;
                int[] uOffsets = new int[H * W + 1];
                for (int i = 0; i < H * W; i++) uOffsets[i + 1] = uOffsets[i] + uCounts[i];
                int[] uFlat = new int[uOffsets[H * W]];
                int[] uCurrent = (int[])uOffsets.Clone();
                foreach (int idx in activeIndices)
                {
                    int start = activeToUnOffset[idx];
                    int end = start + unknownsCount[idx];
                    for (int i = start; i < end; i++) uFlat[uCurrent[unFlat[i]]++] = idx;
                }

                int[] lastSeenB = new int[H * W];
                bool changed = true;
                int pass = 0;
                // Adaptive logic: more passes for small boards, fewer for ludicrous scale
                int maxPasses = (H * W < 1000000) ? 10 : 3;

                while (changed && pass < maxPasses)
                {
                    changed = false;
                    pass++;
                    Array.Fill(lastSeenB, -1);

                    foreach (int A in activeIndices)
                    {
                        if (unknownsCount[A] == 0) continue;
                        int startA = activeToUnOffset[A];
                        int endA = startA + unknownsCount[A];

                        for (int k = startA; k < endA; k++)
                        {
                            int u = unFlat[k];
                            if (board[u] != 9) continue;
                            int startU = uOffsets[u], endU = uOffsets[u + 1];
                            for (int m = startU; m < endU; m++)
                            {
                                int B = uFlat[m];
                                if (B <= A || lastSeenB[B] == A) continue;
                                lastSeenB[B] = A;
                                if (unknownsCount[B] == 0) continue;

                                int startB = activeToUnOffset[B];
                                int endB = startB + unknownsCount[B];

                                int intersection = 0;
                                int currentUA = 0;
                                for (int i = startA; i < endA; i++)
                                {
                                    int uidxA = unFlat[i];
                                    if (board[uidxA] == 9)
                                    {
                                        currentUA++;
                                        for (int j = startB; j < endB; j++)
                                            if (uidxA == unFlat[j]) { intersection++; break; }
                                    }
                                }
                                if (intersection == 0) continue;

                                int currentUB = 0;
                                for (int j = startB; j < endB; j++)
                                    if (board[unFlat[j]] == 9) currentUB++;

                                int mA = minesRemaining[A];
                                int mB = minesRemaining[B];
                                int sizeI = intersection;
                                int sizeRA = currentUA - sizeI;
                                int sizeRB = currentUB - sizeI;

                                // Coupled Rule Deductions
                                int LI = Math.Max(0, Math.Max(mA - sizeRA, mB - sizeRB));
                                int UI = Math.Min(mA, Math.Min(mB, sizeI));

                                if (mA - LI == 0 || mA - UI == sizeRA || mB - LI == 0 || mB - UI == sizeRB || LI == UI)
                                {
                                    if (mA - LI == 0) // R_A is safe
                                        for (int i = startA; i < endA; i++) {
                                            int uidx = unFlat[i];
                                            if (board[uidx] == 9) {
                                                bool inB = false;
                                                for (int j = startB; j < endB; j++) if (uidx == unFlat[j]) { inB = true; break; }
                                                if (!inB) { board[uidx] = 11; changed = true; }
                                            }
                                        }
                                    if (mA - UI == sizeRA) // R_A is mines
                                        for (int i = startA; i < endA; i++) {
                                            int uidx = unFlat[i];
                                            if (board[uidx] == 9) {
                                                bool inB = false;
                                                for (int j = startB; j < endB; j++) if (uidx == unFlat[j]) { inB = true; break; }
                                                if (!inB) { board[uidx] = 10; changed = true; }
                                            }
                                        }
                                    if (mB - LI == 0) // R_B is safe
                                        for (int j = startB; j < endB; j++) {
                                            int uidx = unFlat[j];
                                            if (board[uidx] == 9) {
                                                bool inA = false;
                                                for (int i = startA; i < endA; i++) if (uidx == unFlat[i]) { inA = true; break; }
                                                if (!inA) { board[uidx] = 11; changed = true; }
                                            }
                                        }
                                    if (mB - UI == sizeRB) // R_B is mines
                                        for (int j = startB; j < endB; j++) {
                                            int uidx = unFlat[j];
                                            if (board[uidx] == 9) {
                                                bool inA = false;
                                                for (int i = startA; i < endA; i++) if (uidx == unFlat[i]) { inA = true; break; }
                                                if (!inA) { board[uidx] = 10; changed = true; }
                                            }
                                        }
                                    if (LI == UI) { // Intersection mines fixed
                                        byte target = (LI == 0) ? (byte)11 : (byte)10;
                                        if (LI == 0 || LI == sizeI) {
                                            for (int i = startA; i < endA; i++) {
                                                int uidx = unFlat[i];
                                                if (board[uidx] == 9) {
                                                    for (int j = startB; j < endB; j++)
                                                        if (uidx == unFlat[j]) { board[uidx] = target; changed = true; break; }
                                                }
                                            }
                                        }
                                    }
                                }
                            }
                        }
                    }

                    if (changed) // Recalculate remaining mine counts after deductions
                    {
                        foreach (int idx in activeIndices)
                        {
                            int r = idx / W, c = idx % W;
                            int rem = board[idx];
                            int unk = 0;
                            for (int dr = -1; dr <= 1; dr++)
                                for (int dc = -1; dc <= 1; dc++) {
                                    if (dr == 0 && dc == 0) continue;
                                    int nr = r + dr, nc = c + dc;
                                    if (nr >= 0 && nr < H && nc >= 0 && nc < W) {
                                        int nidx = nr * W + nc;
                                        if (board[nidx] == 10) rem--;
                                        else if (board[nidx] == 9) unk++;
                                    }
                                }
                            minesRemaining[idx] = rem;
                            unknownsCount[idx] = unk;
                        }
                    }
                }
            }

            // Output all provably safe cells
            int finalSafeCount = 0;
            for (int i = 0; i < H * W; i++) if (board[i] == 11) finalSafeCount++;

            using (var stdout = Console.OpenStandardOutput())
            using (var writer = new StreamWriter(stdout, Encoding.UTF8, 1024 * 1024))
            {
                writer.WriteLine(finalSafeCount);
                for (int i = 0; i < H * W; i++)
                {
                    if (board[i] == 11)
                    {
                        writer.WriteLine($"{i % W} {i / W}");
                    }
                }
            }
        }
    }
}
