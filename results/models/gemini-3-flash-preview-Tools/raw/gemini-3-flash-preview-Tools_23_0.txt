Below is the complete, compilable Rust code for the lunar lander controller.

### Implementation Details:
- **Multithreaded Stdin Processing:** Uses a dedicated background thread to continuously read state updates from `stdin`. This ensures the input buffer never fills up, preventing lag or crashes as requested.
- **20Hz Control Loop:** The main loop runs at 20Hz (50ms ticks) to meet the frequency requirement, using `Instant` for precise timing and `std::thread::sleep` for rate limiting.
- **Memory-Efficient Obstacle Avoidance:** Instead of storing the full voxel map (which could be multi-GB), it builds a compact 1D heightmap (`min_y`) of the highest obstacles in each column. This uses only 4 bytes per meter of map width, making it extremely efficient.
- **PD Control Strategy:** Employs Proportional-Derivative (PD) controllers for both horizontal transit and vertical descent. It calculates the necessary thrust vector to achieve target velocities and then maps that to `thrust` and `turn` commands based on the lander's physics.
- **Safety Margin:** Implements a cruise altitude logic that flies the lander safely above any obstacles detected between the current position and the target.

```rust
use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

/// Represents the physical state of the lunar lander.
#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    last_update: Instant,
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut handle = stdin.lock();
    let mut buffer = String::new();

    // 1. Read Map Header
    // Header: width height gravity max_thrust fuel
    if handle.read_line(&mut buffer)? == 0 { return Ok(()); }
    let header_parts: Vec<f64> = buffer.split_whitespace().filter_map(|s| s.parse().ok()).collect();
    if header_parts.len() < 5 { return Ok(()); }
    let width = header_parts[0] as usize;
    let height = header_parts[1] as usize;
    let gravity = header_parts[2];
    let max_thrust = header_parts[3];
    let initial_fuel = header_parts[4];
    buffer.clear();

    // Start/Target: start_x start_y target_x target_y
    if handle.read_line(&mut buffer)? == 0 { return Ok(()); }
    let pos_parts: Vec<f64> = buffer.split_whitespace().filter_map(|s| s.parse().ok()).collect();
    if pos_parts.len() < 4 { return Ok(()); }
    let start_x = pos_parts[0];
    let start_y = pos_parts[1];
    let target_x = pos_parts[2];
    let target_y = pos_parts[3];
    buffer.clear();

    // 2. Process Map Data
    // We use a memory-efficient heightmap (min_y) to store the highest obstacle in each column.
    // This allows us to handle multi-GB maps while fitting in RAM.
    let mut min_y = vec![height as f32; width];
    
    for y in 0..height {
        buffer.clear();
        if handle.read_line(&mut buffer)? == 0 { break; }
        let bytes = buffer.trim_end().as_bytes();
        for (x, &b) in bytes.iter().enumerate() {
            // Treat '#', 'L', and 'T' as ground/obstacles to be cleared during flight
            if x < width && (b == b'#' || b == b'L' || b == b'T') {
                if (y as f32) < min_y[x] {
                    min_y[x] = y as f32;
                }
            }
        }
    }

    // 3. Setup Shared State and Reader Thread
    let shared_state = Arc::new(Mutex::new(State {
        x: start_x,
        y: start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: initial_fuel,
        last_update: Instant::now(),
    }));

    let state_clone = Arc::clone(&shared_state);
    thread::spawn(move || {
        let inner_stdin = io::stdin();
        let mut inner_handle = inner_stdin.lock();
        let mut inner_buffer = String::new();
        loop {
            inner_buffer.clear();
            match inner_handle.read_line(&mut inner_buffer) {
                Ok(0) | Err(_) => break, // Stdin closed
                Ok(_) => {
                    let parts: Vec<f64> = inner_buffer.split_whitespace().filter_map(|s| s.parse().ok()).collect();
                    if parts.len() >= 7 {
                        let mut s = state_clone.lock().unwrap();
                        s.x = parts[0];
                        s.y = parts[1];
                        s.vx = parts[2];
                        s.vy = parts[3];
                        s.angle = parts[4];
                        s.angular_vel = parts[5];
                        s.fuel = parts[6];
                        s.last_update = Instant::now();
                    }
                }
            }
        }
    });

    // 4. Main Control Loop (20Hz)
    let mut stdout = io::stdout();
    let tick_rate = Duration::from_millis(50);
    
    loop {
        let start_tick = Instant::now();
        
        // Get latest state snapshot
        let mut s = {
            let s_guard = shared_state.lock().unwrap();
            *s_guard
        };

        // Simple extrapolation for smooth control between updates
        let dt_extrap = s.last_update.elapsed().as_secs_f64();
        if dt_extrap > 0.0 && dt_extrap < 0.2 {
            s.x += s.vx * dt_extrap;
            s.y += s.vy * dt_extrap;
        }

        // --- Path Planning & Obstacle Avoidance ---
        // Find the maximum obstacle height in the window between current and target x.
        let x_current_idx = (s.x.round() as isize).clamp(0, (width as isize) - 1) as usize;
        let x_target_idx = (target_x.round() as isize).clamp(0, (width as isize) - 1) as usize;
        let x_min = x_current_idx.min(x_target_idx);
        let x_max = x_current_idx.max(x_target_idx);
        
        // Safety buffer window
        let window_min = if x_min > 5 { x_min - 5 } else { 0 };
        let window_max = if x_max + 5 < width { x_max + 5 } else { width - 1 };
        
        let mut obstacle_top = height as f32;
        for x in window_min..=window_max {
            if min_y[x] < obstacle_top {
                obstacle_top = min_y[x];
            }
        }
        
        // Safety margin (meters above ground/obstacles)
        let safety_margin = 12.0;
        let cruise_y = (obstacle_top as f64 - safety_margin).min(target_y - safety_margin);
        
        // Fly at cruise altitude until we are horizontally close to the target
        let dist_x = (s.x - target_x).abs();
        let ty = if dist_x > 5.0 { cruise_y } else { target_y };

        // --- PD Control Logic ---
        
        // 1. Target Velocities
        let target_vx = ((target_x - s.x) * 0.5).clamp(-15.0, 15.0);
        let target_vy = if dist_x > 5.0 {
            // Horizontal transit phase
            ((ty - s.y) * 0.5).clamp(-10.0, 10.0)
        } else {
            // Final vertical landing phase
            let dy = target_y - s.y;
            if dy > 15.0 {
                6.0
            } else if dy > 5.0 {
                2.0
            } else {
                (dy * 0.5).clamp(0.1, 1.2) // Controlled final descent
            }
        };

        // 2. Desired Accelerations
        let target_ax = (target_vx - s.vx) * 2.0;
        let target_ay = (target_vy - s.vy) * 2.0;

        // 3. Thrust Vector Calculation
        // Physics: 
        // ax = thrust * max_thrust * sin(angle)
        // ay = -thrust * max_thrust * cos(angle) + gravity
        let force_x = target_ax;
        let force_y = gravity - target_ay;
        
        let mut target_angle = force_x.atan2(force_y);
        let required_thrust_force = (force_x * force_x + force_y * force_y).sqrt();
        
        // Clamp tilt angle to stay mostly upright
        let max_tilt = 1.25; // ~72 degrees
        target_angle = target_angle.clamp(-max_tilt, max_tilt);
        
        let thrust = (required_thrust_force / max_thrust).clamp(0.0, 1.0);
        
        // 4. Angular Control (PD)
        let mut angle_diff = target_angle - s.angle;
        // Normalize angle difference to [-PI, PI]
        while angle_diff > std::f64::consts::PI { angle_diff -= 2.0 * std::f64::consts::PI; }
        while angle_diff < -std::f64::consts::PI { angle_diff += 2.0 * std::f64::consts::PI; }
        
        let kp_angle = 6.0;
        let kd_angle = 1.8;
        let turn = (angle_diff * kp_angle - s.angular_vel * kd_angle).clamp(-1.0, 1.0);

        // Send command to engine
        if writeln!(stdout, "{:.4} {:.4}", thrust, turn).is_err() {
            break; // Likely stdout closed
        }
        let _ = stdout.flush();

        // Tick timing
        let elapsed = start_tick.elapsed();
        if elapsed < tick_rate {
            thread::sleep(tick_rate - elapsed);
        }
    }
    Ok(())
}
```
