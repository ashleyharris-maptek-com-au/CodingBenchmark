OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1

     %uint_6 = OpConstant %uint 6
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4

    %float_0 = OpConstant %float 0.0
  %float_0_2 = OpConstant %float 0.2
  %float_0_8 = OpConstant %float 0.8
    %float_1 = OpConstant %float 1.0
    %float_3 = OpConstant %float 3.0

     %L_vec = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

   %arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

 %ptr_out_v3 = OpTypePointer Output %v3float
 %ptr_in_v3 = OpTypePointer Input %v3float
 %ptr_in_v4 = OpTypePointer Input %v4float
 %ptr_out_v4 = OpTypePointer Output %v4float

      %gl_in = OpVariable %ptr_in_arr_pv Input
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input
     %gl_out = OpVariable %ptr_out_pv Output
   %outColor = OpVariable %ptr_out_v3 Output

  %main_type = OpTypeFunction %void
       %main = OpFunction %void None %main_type
      %entry = OpLabel

     %p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
         %p0 = OpLoad %v3float %p0_ptr
     %p2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
         %p2 = OpLoad %v3float %p2_ptr
     %p4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
         %p4 = OpLoad %v3float %p4_ptr

          %a = OpFSub %v3float %p2 %p0
          %b = OpFSub %v3float %p4 %p0

         %ax = OpCompositeExtract %float %a 0
         %ay = OpCompositeExtract %float %a 1
         %az = OpCompositeExtract %float %a 2
         %bx = OpCompositeExtract %float %b 0
         %by = OpCompositeExtract %float %b 1
         %bz = OpCompositeExtract %float %b 2

       %aybz = OpFMul %float %ay %bz
       %azby = OpFMul %float %az %by
         %cx = OpFSub %float %aybz %azby

       %azbx = OpFMul %float %az %bx
       %axbz = OpFMul %float %ax %bz
         %cy = OpFSub %float %azbx %axbz

       %axby = OpFMul %float %ax %by
       %aybx = OpFMul %float %ay %bx
         %cz = OpFSub %float %axby %aybx

      %cross = OpCompositeConstruct %v3float %cx %cy %cz
          %N = OpExtInst %v3float %1 Normalize %cross

          %L = OpExtInst %v3float %1 Normalize %L_vec
      %dotNL = OpDot %float %N %L
    %diffuse = OpExtInst %float %1 FMax %dotNL %float_0

     %c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
         %c0 = OpLoad %v3float %c0_ptr
     %c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
         %c2 = OpLoad %v3float %c2_ptr
     %c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
         %c4 = OpLoad %v3float %c4_ptr

      %sum02 = OpFAdd %v3float %c0 %c2
     %sum024 = OpFAdd %v3float %sum02 %c4
       %inv3 = OpFDiv %float %float_1 %float_3
   %avgColor = OpVectorTimesScalar %v3float %sum024 %inv3

      %term2 = OpFMul %float %float_0_8 %diffuse
     %factor = OpFAdd %float %float_0_2 %term2
 %finalColor = OpVectorTimesScalar %v3float %avgColor %factor

  %p0_in_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %p0_in = OpLoad %v4float %p0_in_ptr
 %p0_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %p0_out_ptr %p0_in
               OpStore %outColor %finalColor
               OpEmitVertex

  %p2_in_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
      %p2_in = OpLoad %v4float %p2_in_ptr
 %p2_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %p2_out_ptr %p2_in
               OpStore %outColor %finalColor
               OpEmitVertex

  %p4_in_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
      %p4_in = OpLoad %v4float %p4_in_ptr
 %p4_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %p4_out_ptr %p4_in
               OpStore %outColor %finalColor
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
