#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 6x6 UV polka-dot pattern
    vec2 cellUV = uv * 6.0;
    vec2 localUV = fract(cellUV);
    float dist = distance(localUV, vec2(0.5));
    
    // Dot radius 0.3 (in cell space)
    // Yellow (1.0, 0.8, 0.0) on dark blue (0.1, 0.1, 0.3)
    vec3 baseColor;
    if (dist < 0.3) {
        baseColor = vec3(1.0, 0.8, 0.0);
    } else {
        baseColor = vec3(0.1, 0.1, 0.3);
    }

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shading = 0.3 + 0.7 * diffuse;

    outColor = vec4(baseColor * shading, 1.0);
}
