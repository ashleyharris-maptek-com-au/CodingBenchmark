The TSP solver is implemented as a C++20 program designed to handle city counts from 10 to over 1,000,000,000. It employs an adaptive strategy:
1.  **Small to Medium Scales (N < 5000):** It uses a **Nearest Neighbor** heuristic starting from city 0 to build an initial tour, followed by an iterative **2-opt** refinement to optimize the route. This ensures high-quality results for smaller problem sizes within a few seconds.
2.  **Large to Epic Scales (N >= 5000):** For massive city counts, it uses a **Morton (Z-order) curve** sorting approach. This O(N log N) algorithm maps 2D coordinates to a 1D space-filling curve by interleaving bits of the normalized coordinates. Sorting cities along this curve provides a computationally efficient approximation of the TSP that scales linearly with memory and log-linearly with time.

**Key Optimizations:**
- **Memory Efficiency:** Coordinates are stored in a `vector<float>` to save space (8 bytes/city). For the Morton sort, city data is packed into a single `uint64_t` (8 bytes/city), and the coordinate vector is released before sorting. This keeps the total memory usage around 16 bytes per city, fitting 1 billion cities within 32 GB.
- **I/O Performance:** The program uses a custom fast I/O implementation with `fread` and `fwrite` to minimize the bottleneck of reading and writing billions of values.
- **Robustness:** The algorithm handles diverse coordinate distributions by normalizing the bounding box and supports scientific notation in input. It guarantees the output starts at city 0 and visits each city exactly once.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <cstdio>
#include <chrono>

using namespace std;

// Fast I/O
static char in_buf[1 << 20];
static int in_pos = 0;
static int in_len = 0;

inline int getChar() {
    if (in_pos == in_len) {
        in_pos = 0;
        in_len = (int)fread(in_buf, 1, 1 << 20, stdin);
        if (in_len == 0) return EOF;
    }
    return in_buf[in_pos++];
}

inline bool readInt(int& x) {
    int c = getChar();
    while (c != EOF && (c < '0' || c > '9')) c = getChar();
    if (c == EOF) return false;
    x = 0;
    while (c >= '0' && c <= '9') {
        x = x * 10 + (c - '0');
        c = getChar();
    }
    return true;
}

inline bool readFloat(float& x) {
    int c = getChar();
    while (c != EOF && c != '-' && (c < '0' || c > '9') && c != '.') c = getChar();
    if (c == EOF) return false;
    bool neg = false;
    if (c == '-') {
        neg = true;
        c = getChar();
    }
    double res = 0;
    while (c >= '0' && c <= '9') {
        res = res * 10 + (c - '0');
        c = getChar();
    }
    if (c == '.') {
        double div = 10;
        c = getChar();
        while (c >= '0' && c <= '9') {
            res += (c - '0') / div;
            div *= 10;
            c = getChar();
        }
    }
    if (c == 'e' || c == 'E') {
        c = getChar();
        bool eneg = false;
        if (c == '-') { eneg = true; c = getChar(); }
        else if (c == '+') { c = getChar(); }
        int expVal = 0;
        while (c >= '0' && c <= '9') {
            expVal = expVal * 10 + (c - '0');
            c = getChar();
        }
        if (eneg) res /= pow(10.0, (double)expVal);
        else res *= pow(10.0, (double)expVal);
    }
    x = (float)(neg ? -res : res);
    return true;
}

static char out_buf[1 << 20];
static int out_pos = 0;

inline void flushOut() {
    if (out_pos > 0) {
        fwrite(out_buf, 1, out_pos, stdout);
        out_pos = 0;
    }
}

inline void writeChar(char c) {
    if (out_pos == (1 << 20)) flushOut();
    out_buf[out_pos++] = c;
}

inline void writeInt(int x) {
    if (x == 0) {
        writeChar('0');
        return;
    }
    char s[12];
    int n = 0;
    while (x) {
        s[n++] = (char)(x % 10 + '0');
        x /= 10;
    }
    while (n--) writeChar(s[n]);
}

// Morton Helper: Expands a 32-bit integer by inserting zeros between bits
uint64_t expandBit32(uint32_t x) {
    uint64_t a = x;
    a = (a | (a << 16)) & 0x0000FFFF0000FFFFULL;
    a = (a | (a << 8)) & 0x00FF00FF00FF00FFULL;
    a = (a | (a << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    a = (a | (a << 2)) & 0x3333333333333333ULL;
    a = (a | (a << 1)) & 0x5555555555555555ULL;
    return a;
}

int main() {
    auto startGlobalTime = chrono::steady_clock::now();
    int N;
    if (!readInt(N)) return 0;
    if (N <= 0) return 0;

    if (N == 1) {
        writeInt(0);
        writeChar('\n');
        flushOut();
        return 0;
    }

    vector<float> coords(2ULL * N);
    float minX = 1e30f, maxX = -1e30f, minY = 1e30f, maxY = -1e30f;
    for (int i = 0; i < N; ++i) {
        readFloat(coords[2ULL * i]);
        readFloat(coords[2ULL * i + 1]);
        if (coords[2ULL * i] < minX) minX = coords[2ULL * i];
        if (coords[2ULL * i] > maxX) maxX = coords[2ULL * i];
        if (coords[2ULL * i + 1] < minY) minY = coords[2ULL * i + 1];
        if (coords[2ULL * i + 1] > maxY) maxY = coords[2ULL * i + 1];
    }

    if (N < 5000) {
        // Strategy: Nearest Neighbor followed by 2-opt refinement
        vector<int> tour(N);
        vector<bool> used(N, false);
        tour[0] = 0;
        used[0] = true;
        for (int i = 1; i < N; ++i) {
            int last = tour[i - 1];
            int best = -1;
            float bestDistSq = 1e30f;
            for (int j = 0; j < N; ++j) {
                if (!used[j]) {
                    float dx = coords[2ULL * last] - coords[2ULL * j];
                    float dy = coords[2ULL * last + 1] - coords[2ULL * j + 1];
                    float d2 = dx * dx + dy * dy;
                    if (d2 < bestDistSq) {
                        bestDistSq = d2;
                        best = j;
                    }
                }
            }
            tour[i] = best;
            used[best] = true;
        }

        auto dist = [&](int i, int j) {
            float dx = coords[2ULL * i] - coords[2ULL * j];
            float dy = coords[2ULL * i + 1] - coords[2ULL * j + 1];
            return sqrtf(dx * dx + dy * dy);
        };

        bool improved = true;
        int iter = 0;
        while (improved && iter < 100) {
            improved = false;
            auto now = chrono::steady_clock::now();
            if (chrono::duration_cast<chrono::seconds>(now - startGlobalTime).count() > 280) break;
            
            for (int i = 0; i < N - 2; ++i) {
                for (int j = i + 2; j < N; ++j) {
                    int a = tour[i];
                    int b = tour[i + 1];
                    int c = tour[j];
                    int d = tour[(j + 1) % N];
                    if (dist(a, b) + dist(c, d) > dist(a, c) + dist(b, d) + 1e-6f) {
                        reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        improved = true;
                    }
                }
            }
            iter++;
        }

        int s = 0;
        for (int i = 0; i < N; ++i) if (tour[i] == 0) { s = i; break; }
        for (int i = 0; i < N; ++i) {
            writeInt(tour[(s + i) % N]);
            writeChar(i == N - 1 ? '\n' : ' ');
        }
    } else {
        // Strategy: Morton (Z-order) sorting for massive scale
        float rangeX = maxX - minX;
        float rangeY = maxY - minY;
        if (rangeX <= 0) rangeX = 1.0f;
        if (rangeY <= 0) rangeY = 1.0f;

        vector<uint64_t> packed(N);
        for (int i = 0; i < N; ++i) {
            uint32_t ix = (uint32_t)((coords[2ULL * i] - minX) / rangeX * 65535.0f);
            uint32_t iy = (uint32_t)((coords[2ULL * i + 1] - minY) / rangeY * 65535.0f);
            uint64_t code32 = (expandBit32(iy) << 1) | expandBit32(ix);
            packed[i] = (code32 << 31) | (uint64_t)i;
        }
        
        vector<float>().swap(coords); // Free memory
        sort(packed.begin(), packed.end());

        int startIdx = 0;
        for (int i = 0; i < N; ++i) {
            if ((packed[i] & 0x7FFFFFFF) == 0) {
                startIdx = i;
                break;
            }
        }

        for (int i = 0; i < N; ++i) {
            writeInt((int)(packed[(startIdx + i) % N] & 0x7FFFFFFF));
            writeChar(i == N - 1 ? '\n' : ' ');
        }
    }

    flushOut();
    return 0;
}
```
