The Quadratic Assignment Problem (QAP) is an NP-hard optimization problem. To solve it effectively for $N$ up to 2000 within a 5-minute time limit, this implementation employs a greedy constructive heuristic followed by an iterative local search (2-opt).

### Approach:
1.  **Greedy Initialization:** High-flow facilities are assigned to low-distance locations to provide a strong starting point.
2.  **Local Search (2-opt):** The algorithm iteratively explores swaps of facility locations. A fast $O(N)$ incremental cost update is used to evaluate swaps efficiently, allowing for thousands of evaluations per second even for $N=2000$.
3.  **Perturbation (Iterated Local Search):** When a local optimum is reached, the solution is slightly perturbed by random swaps to escape and explore new regions of the search space.
4.  **Efficiency:** The implementation uses jagged arrays for faster access and pre-calculated transposed matrices to ensure row-major memory access patterns in critical inner loops.

```csharp
using System;
using System.Diagnostics;
using System.Linq;
using System.Collections.Generic;

class QAP
{
    static int N;
    static int[][] flow = null!;
    static int[][] dist = null!;
    static int[][] flowT = null!;
    static int[][] distT = null!;
    static int[] p = null!;
    static int[] bestP = null!;
    static long currentCost;
    static long bestCost;

    static void Main()
    {
        if (!ReadInput()) return;
        if (N <= 0) return;

        Stopwatch sw = Stopwatch.StartNew();

        // 1. Generate initial greedy solution
        InitialSolution();
        currentCost = CalculateTotalCost(p);
        bestP = (int[])p.Clone();
        bestCost = currentCost;

        // 2. Local Search with restarts/perturbations
        if (N > 1)
        {
            LocalSearch(sw);
        }

        // 3. Output best result found
        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestP));
    }

    static bool ReadInput()
    {
        string? nStr = FastReader.Next();
        if (nStr == null) return false;
        if (!int.TryParse(nStr, out N)) return false;
        if (N <= 0) return true;

        flow = new int[N][];
        flowT = new int[N][];
        for (int i = 0; i < N; i++) { flow[i] = new int[N]; flowT[i] = new int[N]; }
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                int val = FastReader.NextInt();
                flow[i][j] = val;
                flowT[j][i] = val;
            }
        }

        dist = new int[N][];
        distT = new int[N][];
        for (int i = 0; i < N; i++) { dist[i] = new int[N]; distT[i] = new int[N]; }
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                int val = FastReader.NextInt();
                dist[i][j] = val;
                distT[j][i] = val;
            }
        }
        return true;
    }

    static void InitialSolution()
    {
        long[] flowSum = new long[N];
        long[] distSum = new long[N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                flowSum[i] += (long)flow[i][j] + flow[j][i];
                distSum[i] += (long)dist[i][j] + dist[j][i];
            }
        }

        var facilities = Enumerable.Range(0, N).OrderByDescending(i => flowSum[i]).ToArray();
        var locations = Enumerable.Range(0, N).OrderBy(i => distSum[i]).ToArray();

        p = new int[N];
        for (int i = 0; i < N; i++)
        {
            p[facilities[i]] = locations[i];
        }
    }

    static long CalculateTotalCost(int[] perm)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            int[] fi = flow[i];
            int pi = perm[i];
            int[] dpi = dist[pi];
            for (int j = 0; j < N; j++)
            {
                cost += (long)fi[j] * dpi[perm[j]];
            }
        }
        return cost;
    }

    static void LocalSearch(Stopwatch sw)
    {
        Random rnd = new Random();
        long timeLimitMs = 285000; // 4 minutes 45 seconds

        int[] indices = Enumerable.Range(0, N).ToArray();

        while (sw.ElapsedMilliseconds < timeLimitMs)
        {
            bool improved = false;
            
            // Shuffle indices for diverse search paths
            for (int i = 0; i < N; i++)
            {
                int rIndex = i + rnd.Next(N - i);
                int tmp = indices[i];
                indices[i] = indices[rIndex];
                indices[rIndex] = tmp;
            }

            for (int i = 0; i < N; i++)
            {
                if (sw.ElapsedMilliseconds >= timeLimitMs) break;
                int r = indices[i];
                for (int j = i + 1; j < N; j++)
                {
                    if (sw.ElapsedMilliseconds >= timeLimitMs) break;
                    int s = indices[j];
                    
                    long delta = CalculateDelta(r, s);
                    if (delta < 0)
                    {
                        int tmp = p[r];
                        p[r] = p[s];
                        p[s] = tmp;
                        currentCost += delta;
                        improved = true;
                    }
                }
            }
            
            if (currentCost < bestCost)
            {
                bestCost = currentCost;
                Array.Copy(p, bestP, N);
            }

            if (!improved)
            {
                if (sw.ElapsedMilliseconds < timeLimitMs)
                {
                    Perturb(rnd);
                }
                else break;
            }
        }
    }

    static void Perturb(Random rnd)
    {
        int numSwaps = Math.Max(2, N / 50);
        for (int k = 0; k < numSwaps; k++)
        {
            int i = rnd.Next(N);
            int j = rnd.Next(N);
            if (i == j) continue;
            int tmp = p[i];
            p[i] = p[j];
            p[j] = tmp;
        }
        currentCost = CalculateTotalCost(p);
        if (currentCost < bestCost)
        {
            bestCost = currentCost;
            Array.Copy(p, bestP, N);
        }
    }

    static long CalculateDelta(int r, int s)
    {
        int pr = p[r];
        int ps = p[s];
        long delta = 0;
        
        int[] fr = flow[r], fs = flow[s];
        int[] fTr = flowT[r], fTs = flowT[s];
        int[] dpr = dist[pr], dps = dist[ps];
        int[] dTpr = distT[pr], dTps = distT[ps];

        for (int j = 0; j < N; j++)
        {
            if (j == r || j == s) continue;
            int pj = p[j];
            // Compute incremental change for terms involving r and s
            delta += (long)(fr[j] - fs[j]) * (dps[pj] - dpr[pj])
                   + (long)(fTr[j] - fTs[j]) * (dTps[pj] - dTpr[pj]);
        }

        // Mutual and self-interaction terms
        delta += (long)flow[r][s] * (dist[ps][pr] - dist[pr][ps]);
        delta += (long)flow[s][r] * (dist[pr][ps] - dist[ps][pr]);
        delta += (long)flow[r][r] * (dist[ps][ps] - dist[pr][pr]);
        delta += (long)flow[s][s] * (dist[pr][pr] - dist[ps][ps]);

        return delta;
    }

    static class FastReader
    {
        static string[]? tokens;
        static int currentToken = 0;

        public static string? Next()
        {
            while (tokens == null || currentToken >= tokens.Length)
            {
                string? line = Console.ReadLine();
                if (line == null) return null;
                tokens = line.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
                currentToken = 0;
            }
            return tokens[currentToken++];
        }

        public static int NextInt()
        {
            string? s = Next();
            if (s == null) return 0;
            int res;
            int.TryParse(s, out res);
            return res;
        }
    }
}
```
