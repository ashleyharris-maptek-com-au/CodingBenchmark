The Capacitated Vehicle Routing Problem (CVRP) is solved using the **Clarke-Wright Savings algorithm**, a powerful and widely-used heuristic for this NP-hard problem. This approach starts by assigning each customer to their own route (Depot → Customer → Depot) and then iteratively merges routes that yield the highest "savings" in total distance, while respecting vehicle capacity and route structure constraints.

The implementation includes:
1.  **Robust Input Parsing:** Uses a generator to handle large inputs efficiently.
2.  **Distance Matrix:** Precalculates distances between all pairs of nodes (Depot and $N$ customers).
3.  **Savings Heuristic:** Calculates $S_{ij} = d(Depot, i) + d(Depot, j) - d(i, j)$ for all pairs and sorts them to prioritize the most beneficial merges.
4.  **Route Management:** Maintains routes as ordered lists of customer indices and ensures merges only happen at route endpoints to maintain valid paths.
5.  **Capacity Constraint:** Strictly enforces that the sum of demands in any merged route does not exceed the specified `CAPACITY`.

### Solution Code

```python
import sys
import math

def solve():
    # Use generator to read words one by one for efficiency
    def get_input():
        for line in sys.stdin:
            for word in line.split():
                yield word
    
    input_gen = get_input()
    
    try:
        line1_n = next(input_gen)
        N = int(line1_n)
        K = int(next(input_gen))
        CAPACITY = int(next(input_gen))
        
        depot_x = float(next(input_gen))
        depot_y = float(next(input_gen))
        
        customers = []
        for i in range(N):
            x = float(next(input_gen))
            y = float(next(input_gen))
            demand = int(next(input_gen))
            customers.append({'x': x, 'y': y, 'demand': demand})
    except (StopIteration, ValueError):
        return

    if N == 0:
        print(0.0)
        print(0)
        return
        
    def dist(p1_x, p1_y, p2_x, p2_y):
        return math.sqrt((p1_x - p2_x)**2 + (p1_y - p2_y)**2)
    
    # distance from depot to each customer
    d0 = [dist(depot_x, depot_y, c['x'], c['y']) for c in customers]
    
    # distance matrix between customers
    d = [[0.0] * N for _ in range(N)]
    for i in range(N):
        for j in range(i + 1, N):
            val = dist(customers[i]['x'], customers[i]['y'], customers[j]['x'], customers[j]['y'])
            d[i][j] = d[j][i] = val
            
    # Calculate savings for merging any two customers i and j
    savings = []
    for i in range(N):
        for j in range(i + 1, N):
            s = d0[i] + d0[j] - d[i][j]
            savings.append((s, i, j))
            
    # Sort savings in descending order to prioritize best merges
    savings.sort(key=lambda x: x[0], reverse=True)
    
    # Initialize routes: each customer starts in its own route [Depot -> i -> Depot]
    routes = [[i] for i in range(N)]
    route_demands = [c['demand'] for c in customers]
    
    # Map customer index to its current route index for O(1) lookup
    customer_to_route = list(range(N))
    
    for s, i, j in savings:
        ri = customer_to_route[i]
        rj = customer_to_route[j]
        
        # Skip if customers are already in the same route
        if ri == rj:
            continue
            
        # Check if i and j are both "exterior" (connected to the depot) in their respective routes
        if (routes[ri][0] == i or routes[ri][-1] == i) and \
           (routes[rj][0] == j or routes[rj][-1] == j):
            
            # Check capacity constraint
            if route_demands[ri] + route_demands[rj] <= CAPACITY:
                # Merge routes: Normalize order so i is at the end of ri and j is at start of rj
                if routes[ri][0] == i:
                    routes[ri].reverse()
                if routes[rj][-1] == j:
                    routes[rj].reverse()
                
                # Update mapping for all customers in the route being merged
                for cust_id in routes[rj]:
                    customer_to_route[cust_id] = ri
                
                # Append rj to ri and update metadata
                routes[ri].extend(routes[rj])
                route_demands[ri] += route_demands[rj]
                
                # Mark rj as empty
                routes[rj] = []
                route_demands[rj] = 0

    # Filter out empty routes and finalize results
    final_routes = [r for r in routes if r]
    
    # Calculate total travel distance
    total_dist = 0
    for r in final_routes:
        total_dist += d0[r[0]]  # Depot to first customer
        for k in range(len(r) - 1):
            total_dist += d[r[k]][r[k+1]]  # Between customers
        total_dist += d0[r[-1]]  # Last customer to Depot
        
    # Output total distance, number of routes, and each route's sequence
    print(total_dist)
    print(len(final_routes))
    for r in final_routes:
        print(" ".join(map(str, r)))

if __name__ == "__main__":
    solve()
```
