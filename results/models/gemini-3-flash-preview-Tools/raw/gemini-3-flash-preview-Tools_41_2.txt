; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types, Constants, and Variables
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%ptrInV2 = OpTypePointer Input %v2float
%ptrInV3 = OpTypePointer Input %v3float
%ptrOutV4 = OpTypePointer Output %v4float
%ptrUbo = OpTypePointer Uniform %UBO

%voidFunc = OpTypeFunction %void

%float0 = OpConstant %float 0
%float1 = OpConstant %float 1

%inWorldPos = OpVariable %ptrInV3 Input
%inNormal = OpVariable %ptrInV3 Input
%inUV = OpVariable %ptrInV2 Input
%inTangent = OpVariable %ptrInV3 Input
%inColor = OpVariable %ptrInV3 Input
%outColor = OpVariable %ptrOutV4 Output
%ubo = OpVariable %ptrUbo Uniform

; Main Function
%main = OpFunction %void None %voidFunc
%entry = OpLabel
%uvVal = OpLoad %v2float %inUV
%res = OpCompositeConstruct %v4float %uvVal %float0 %float1
OpStore %outColor %res
OpReturn
OpFunctionEnd
