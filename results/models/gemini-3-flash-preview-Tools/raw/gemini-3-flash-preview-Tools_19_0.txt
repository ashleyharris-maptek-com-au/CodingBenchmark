using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Text;

public class MiningOptimizer
{
    public static void Main(string[] args)
    {
        Stream stdin = Console.OpenStandardInput(1024 * 1024);
        
        // Read header line from stream manually to avoid Console.ReadLine buffer issues
        StringBuilder sbHeader = new StringBuilder();
        int b;
        while ((b = stdin.ReadByte()) != -1)
        {
            if (b == '\n') break;
            if (b == '\r') continue;
            sbHeader.Append((char)b);
        }
        
        string headerLine = sbHeader.ToString();
        if (string.IsNullOrEmpty(headerLine)) return;

        string[] headerParts = headerLine.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        if (headerParts.Length < 6) return;

        int XV = int.Parse(headerParts[0]);
        int YV = int.Parse(headerParts[1]);
        int ZV = int.Parse(headerParts[2]);
        // Block sizes are in the header but not directly used for profit calculation in voxel units
        double XB = double.Parse(headerParts[3]);
        double YB = double.Parse(headerParts[4]);
        double ZB = double.Parse(headerParts[5]);

        Stopwatch sw = Stopwatch.StartNew();
        
        // heightMap[x, y] stores the current top-most voxel Z coordinate
        ushort[,] heightMap = new ushort[XV, YV];
        
        // Buffer for reading one column (ZV voxels)
        byte[] zBuffer = new byte[ZV];
        
        List<int[]> opPaths = new List<int[]>();
        List<int[]> opDigs = new List<int[]>(); // Stores [x, y, z]
        
        long totalStepsStored = 0;
        const long MaxTotalSteps = 50000000;
        bool resourceLimitReached = false;

        for (int x = 0; x < XV; x++)
        {
            if (resourceLimitReached) break;
            
            for (int y = 0; y < YV; y++)
            {
                // Read a full column of Z voxels
                int totalRead = 0;
                while (totalRead < ZV)
                {
                    int r = stdin.Read(zBuffer, totalRead, ZV - totalRead);
                    if (r <= 0) break;
                    totalRead += r;
                }

                // Find initial height of the column
                int h = -1;
                for (int z = ZV - 1; z >= 0; z--)
                {
                    if (zBuffer[z] > 0)
                    {
                        h = z;
                        break;
                    }
                }
                
                if (h < 0)
                {
                    heightMap[x, y] = 0; // Empty column
                    continue;
                }
                
                heightMap[x, y] = (ushort)h;

                // Processing plant is at (0,0). We don't dig it.
                if (x == 0 && y == 0) continue;

                long currentColumnProfit = 0;
                int bestDepth = -1;
                long maxCumulativeProfit = 0;

                // Calculate cumulative profit to decide how deep to dig
                for (int z = h; z >= 0; z--)
                {
                    byte d = zBuffer[z];
                    if (d == 0) continue;

                    int mass = d % 16;
                    int val = d / 16;
                    if (mass == 0) continue;

                    // Check path feasibility and fuel cost
                    long fuelCost = GetPathFuel(x, y, z, heightMap, mass);

                    if (fuelCost < 0) break; // Path blocked

                    long gain = (long)val * mass * 1000;
                    long cost = (val == 0) ? (1000L * mass + fuelCost) : fuelCost;
                    
                    currentColumnProfit += (gain - cost);
                    
                    if (currentColumnProfit > maxCumulativeProfit)
                    {
                        maxCumulativeProfit = currentColumnProfit;
                        bestDepth = h - z;
                    }
                    
                    // Stop if profit is dropping significantly below peak
                    if (currentColumnProfit < maxCumulativeProfit - 1000000) break; 
                }

                // If digging the column is profitable, store operations
                if (maxCumulativeProfit > 0 && bestDepth >= 0)
                {
                    for (int z = h; z >= h - bestDepth; z--)
                    {
                        if (totalStepsStored >= MaxTotalSteps) { resourceLimitReached = true; break; }

                        byte d = zBuffer[z];
                        int mass = d % 16;
                        if (mass == 0) continue;

                        List<int> path = new List<int>();
                        long fuel = GetPathAndBuild(x, y, z, heightMap, mass, path);
                        
                        if (fuel >= 0)
                        {
                            opDigs.Add(new int[] { x, y, z });
                            opPaths.Add(path.ToArray());
                            totalStepsStored += (path.Count / 2);
                            
                            // Update height as we dig
                            heightMap[x, y] = (ushort)(z - 1 >= 0 ? z - 1 : 0);
                        }
                        else break;
                    }
                }
                
                // Adaptive time management
                if (sw.ElapsedMilliseconds > 260000) 
                {
                    resourceLimitReached = true;
                    break;
                }
            }
        }

        // Output results in specified format
        using (StreamWriter swOut = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 128 * 1024))
        {
            swOut.WriteLine(opDigs.Count);
            for (int i = 0; i < opDigs.Count; i++)
            {
                int[] dig = opDigs[i];
                int[] path = opPaths[i];
                int N = path.Length / 2;
                swOut.Write(N);
                swOut.Write(" ");
                swOut.Write(dig[0]);
                swOut.Write(" ");
                swOut.Write(dig[1]);
                swOut.Write(" ");
                swOut.Write(dig[2]);
                for (int j = 0; j < path.Length; j++)
                {
                    swOut.Write(" ");
                    swOut.Write(path[j]);
                }
                swOut.WriteLine();
            }
        }
    }

    private static long GetPathFuel(int sx, int sy, int sz, ushort[,] heightMap, int mass)
    {
        int cx = sx, cy = sy, cz = sz;
        long fuel = 0;
        while (cx > 0 || cy > 0)
        {
            bool canX = false, canY = false;
            int hz_x = -1, hz_y = -1;
            if (cx > 0) { hz_x = heightMap[cx - 1, cy]; if (Math.Abs(hz_x - cz) <= 1) canX = true; }
            if (cy > 0) { hz_y = heightMap[cx, cy - 1]; if (Math.Abs(hz_y - cz) <= 1) canY = true; }

            if (canX && canY)
            {
                // Prefer the direction that is flatter or downhill
                if (hz_x <= hz_y) { fuel += CalculateStepFuel(cz, hz_x, mass); cx--; cz = hz_x; }
                else { fuel += CalculateStepFuel(cz, hz_y, mass); cy--; cz = hz_y; }
            }
            else if (canX) { fuel += CalculateStepFuel(cz, hz_x, mass); cx--; cz = hz_x; }
            else if (canY) { fuel += CalculateStepFuel(cz, hz_y, mass); cy--; cz = hz_y; }
            else return -1;
        }
        return fuel;
    }

    private static long GetPathAndBuild(int sx, int sy, int sz, ushort[,] heightMap, int mass, List<int> path)
    {
        int cx = sx, cy = sy, cz = sz;
        long fuel = 0;
        while (cx > 0 || cy > 0)
        {
            bool canX = false, canY = false;
            int hz_x = -1, hz_y = -1;
            if (cx > 0) { hz_x = heightMap[cx - 1, cy]; if (Math.Abs(hz_x - cz) <= 1) canX = true; }
            if (cy > 0) { hz_y = heightMap[cx, cy - 1]; if (Math.Abs(hz_y - cz) <= 1) canY = true; }

            if (canX && canY)
            {
                if (hz_x <= hz_y) { fuel += CalculateStepFuel(cz, hz_x, mass); cx--; cz = hz_x; }
                else { fuel += CalculateStepFuel(cz, hz_y, mass); cy--; cz = hz_y; }
            }
            else if (canX) { fuel += CalculateStepFuel(cz, hz_x, mass); cx--; cz = hz_x; }
            else if (canY) { fuel += CalculateStepFuel(cz, hz_y, mass); cy--; cz = hz_y; }
            else return -1;
            path.Add(cx); path.Add(cy);
        }
        return fuel;
    }

    private static long CalculateStepFuel(int cz, int nz, int mass)
    {
        if (nz > cz) return 5L * mass;
        if (nz == cz) return 1L * mass;
        return 0; // Downhill is free
    }
}
