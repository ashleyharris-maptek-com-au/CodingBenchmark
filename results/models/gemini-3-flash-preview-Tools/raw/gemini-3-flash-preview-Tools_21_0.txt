#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <queue>
#include <algorithm>
#include <cmath>

using namespace std;

// Position in N-D space
typedef vector<int> Pos;

// Hash function for Pos (vector<int>)
struct PosHash {
    size_t operator()(const Pos& v) const {
        size_t seed = v.size();
        for (int x : v) {
            // Standard hash combiner (boost style)
            seed ^= hash<int>{}(x) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
        }
        return seed;
    }
};

int D;
long long max_turns;
Pos bounds;
int initial_length;
Pos start_head;
int F_count;
vector<Pos> foods;
int O_count;
unordered_set<Pos, PosHash> obstacles;

deque<Pos> snake;
// Tracks number of snake segments at each position to handle overlapping segments
unordered_map<Pos, int, PosHash> snake_body_count;

// Check if a point is within the hypercube boundaries
bool is_in_bounds(const Pos& p) {
    for (int i = 0; i < D; ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

// Check if a position is safe (no obstacle and no snake body)
bool is_safe(const Pos& p) {
    if (!is_in_bounds(p)) return false;
    if (obstacles.count(p)) return false;
    auto it = snake_body_count.find(p);
    if (it != snake_body_count.end() && it->second > 0) return false;
    return true;
}

// Manhattan distance in N-D space
inline int manhattan_dist(const Pos& a, const Pos& b) {
    int d = 0;
    int n = (int)a.size();
    for (int i = 0; i < n; ++i) {
        int diff = a[i] - b[i];
        d += (diff < 0) ? -diff : diff;
    }
    return d;
}

// State for A* search
struct State {
    Pos p;
    int first_axis;
    int first_dir;
    int g; // current cost
    int h; // heuristic cost

    bool operator>(const State& other) const {
        if ((g + h) != (other.g + other.h))
            return (g + h) > (other.g + other.h);
        return h > other.h;
    }
};

// Determines the move for the current turn using A* search with fallbacks
pair<int, int> solve_turn() {
    if (foods.empty()) return {-1, 0};

    Pos head = snake.front();
    
    // Find nearest food to target
    int best_f_idx = -1;
    int min_f_dist = 2000000000;
    for (int i = 0; i < (int)foods.size(); ++i) {
        int d = manhattan_dist(head, foods[i]);
        if (d < min_f_dist) {
            min_f_dist = d;
            best_f_idx = i;
        }
    }
    
    if (best_f_idx == -1) return {-1, 0};
    const Pos& target_food = foods[best_f_idx];

    // Priority queue for A*
    priority_queue<State, vector<State>, greater<State>> pq;
    unordered_map<Pos, int, PosHash> visited;

    // Explore neighbors of the head
    for (int i = 0; i < D; ++i) {
        for (int d : {-1, 1}) {
            Pos next_p = head;
            next_p[i] += d;
            if (is_safe(next_p)) {
                int h = manhattan_dist(next_p, target_food);
                if (h == 0) return {i, d};
                pq.push({next_p, i, d, 1, h});
                visited[next_p] = 1;
            }
        }
    }

    // Node limit for search to prevent long turns
    int nodes_limit = 10000;
    int nodes = 0;
    while (!pq.empty() && nodes < nodes_limit) {
        State curr = pq.top();
        pq.pop();
        nodes++;

        if (curr.h == 0) return {curr.first_axis, curr.first_dir};

        for (int i = 0; i < D; ++i) {
            for (int d : {-1, 1}) {
                Pos next_p = curr.p;
                next_p[i] += d;
                if (is_safe(next_p)) {
                    int next_g = curr.g + 1;
                    auto it = visited.find(next_p);
                    if (it == visited.end() || it->second > next_g) {
                        visited[next_p] = next_g;
                        pq.push({next_p, curr.first_axis, curr.first_dir, next_g, manhattan_dist(next_p, target_food)});
                    }
                }
            }
        }
    }

    // Fallback 1: Greedy move towards the nearest food
    int best_greedy_axis = -1, best_greedy_dir = 0;
    int best_greedy_dist = 2000000000;
    for (int i = 0; i < D; ++i) {
        for (int d : {-1, 1}) {
            Pos next_p = head;
            next_p[i] += d;
            if (is_safe(next_p)) {
                int dist = manhattan_dist(next_p, target_food);
                if (dist < best_greedy_dist) {
                    best_greedy_dist = dist;
                    best_greedy_axis = i;
                    best_greedy_dir = d;
                }
            }
        }
    }
    if (best_greedy_axis != -1) return {best_greedy_axis, best_greedy_dir};

    // Fallback 2: Any safe move
    for (int i = 0; i < D; ++i) {
        for (int d : {-1, 1}) {
            Pos next_p = head;
            next_p[i] += d;
            if (is_safe(next_p)) return {i, d};
        }
    }

    return {-1, 0}; // No move found
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    // Read problem dimensions and turn limit
    if (!(cin >> D >> max_turns)) return 0;
    
    // Read bounds
    bounds.resize(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];
    
    // Read initial snake length and head position
    cin >> initial_length;
    start_head.resize(D);
    for (int i = 0; i < D; ++i) cin >> start_head[i];
    
    // Read food items
    cin >> F_count;
    foods.resize(F_count);
    for (int i = 0; i < F_count; ++i) {
        foods[i].resize(D);
        for (int j = 0; j < D; ++j) cin >> foods[i][j];
    }
    
    // Read obstacles
    cin >> O_count;
    for (int i = 0; i < O_count; ++i) {
        Pos o(D);
        for (int j = 0; j < D; ++j) cin >> o[j];
        obstacles.insert(o);
    }

    // Initialize snake body locally
    for (int i = 0; i < initial_length; ++i) {
        snake.push_back(start_head);
        snake_body_count[start_head]++;
    }

    // Main game loop
    for (long long turn = 0; turn < max_turns; ++turn) {
        pair<int, int> move = solve_turn();
        
        // End simulation if no move is found
        if (move.first == -1) break;

        // Output move: axis direction
        cout << move.first << " " << move.second << endl;

        // Update state for next turn simulation
        Pos next_head = snake.front();
        next_head[move.first] += move.second;

        // Check if food was eaten
        bool ate = false;
        for (int i = 0; i < (int)foods.size(); ++i) {
            if (foods[i] == next_head) {
                foods.erase(foods.begin() + i);
                ate = true;
                break;
            }
        }

        // Add new head to snake
        snake.push_front(next_head);
        snake_body_count[next_head]++;

        // If no food eaten, remove the tail
        if (!ate) {
            Pos tail = snake.back();
            snake.pop_back();
            if (--snake_body_count[tail] == 0) {
                snake_body_count.erase(tail);
            }
        }

        // Check if all food is consumed
        if (foods.empty()) break;
    }

    return 0;
}
