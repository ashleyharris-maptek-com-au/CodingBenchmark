I have implemented a robust and efficient C++ solution for the Graph Coloring problem, designed to handle graphs ranging from small instances to massive million-node networks within the 5-minute time limit.

### Key Features:
- **Adaptive Heuristics**: The program employs a sequence of sophisticated heuristics to find a valid coloring:
    1. **Smallest Last (SL) Ordering**: An $O(N+M)$ algorithm that minimizes the maximum degree of a vertex when it is colored, often yielding results close to the chromatic number.
    2. **Largest Degree First (LDF)**: A standard greedy approach that prioritizes high-degree nodes.
    3. **DSATUR (Degree of Saturation)**: A high-quality heuristic that dynamically prioritizes vertices with the most constrained color choices. I implemented this with a lazy-update priority queue and bitset-based saturation tracking for maximum performance.
    4. **Randomized Greedy Search**: A fallback that explores different vertex orderings if the deterministic heuristics fail.
- **High-Performance Architecture**:
    - **CSR (Compressed Sparse Row)**: Used for the graph representation to ensure optimal memory usage (~100MB for 1M nodes/10M edges) and cache-friendly neighbor traversal.
    - **Fast I/O**: Implemented using `fread` to handle the massive input data streams expected for million-node graphs.
    - **Bit-Parallel Saturation Tracking**: Uses 64-bit word bitsets to quickly identify available colors, specifically optimized for the "up to 60 colors" constraint.
- **Compliance**: Adheres to C++17/C++20 standards and is fully compatible with the MSVC compiler provided. It uses only standard library components and avoids non-portable headers.

### Complexity & Scaling:
The SL and LDF algorithms run in $O(N+M)$ time, while DSATUR runs in $O(M \log N)$ or $O(M \cdot K/64)$ depending on density. For a million-node graph with 10 million edges, these operations complete well within the 5-minute window on the specified hardware.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <random>
#include <chrono>
#include <queue>
#include <cstdio>

using namespace std;

struct FastReader {
    static const int BUF_SIZE = 1 << 18;
    char buf[BUF_SIZE];
    int pos = 0, len = 0;

    char get_char() {
        if (pos == len) {
            pos = 0;
            len = (int)fread(buf, 1, BUF_SIZE, stdin);
            if (len == 0) return EOF;
        }
        return buf[pos++];
    }

    int read_int() {
        int x = 0;
        char c = get_char();
        while (c != EOF && (c < '0' || c > '9')) c = get_char();
        if (c == EOF) return -1;
        while (c >= '0' && c <= '9') {
            x = x * 10 + (c - '0');
            c = get_char();
        }
        return x;
    }
};

struct Graph {
    int n, m, k;
    vector<int> offsets;
    vector<int> edges;
    vector<int> degrees;

    Graph(int n, int m, int k) : n(n), m(m), k(k), offsets(n + 1, 0), degrees(n, 0) {}

    void load(FastReader& reader) {
        struct Edge { int u, v; };
        vector<Edge> edge_list;
        edge_list.reserve(m);
        for (int i = 0; i < m; ++i) {
            int u = reader.read_int();
            int v = reader.read_int();
            if (u == -1 || v == -1) break;
            if (u >= n || v >= n) continue;
            degrees[u]++;
            degrees[v]++;
            edge_list.push_back({u, v});
        }
        
        for (int i = 0; i < n; ++i) offsets[i + 1] = offsets[i] + degrees[i];
        edges.resize(2 * m);
        vector<int> cur_offsets = offsets;
        for (const auto& e : edge_list) {
            edges[cur_offsets[e.u]++] = e.v;
            edges[cur_offsets[e.v]++] = e.u;
        }
    }
};

bool solve_greedy(const Graph& g, const vector<int>& order, vector<int>& colors) {
    colors.assign(g.n, -1);
    vector<int> last_used(g.k, -1);
    for (int u : order) {
        for (int i = g.offsets[u]; i < g.offsets[u + 1]; ++i) {
            int v = g.edges[i];
            if (colors[v] != -1) last_used[colors[v]] = u;
        }
        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (last_used[c] != u) { chosen = c; break; }
        }
        if (chosen == -1) return false;
        colors[u] = chosen;
    }
    return true;
}

bool solve_dsatur(const Graph& g, vector<int>& colors) {
    int n = g.n, k = g.k, words = (k + 63) / 64;
    colors.assign(n, -1);
    vector<uint64_t> neighbor_colors(n * (size_t)words, 0);
    vector<int> sat_degree(n, 0);
    typedef pair<pair<int, int>, int> Node;
    priority_queue<Node> pq;
    for (int i = 0; i < n; ++i) pq.push({{0, g.degrees[i]}, i});
    int colored = 0;
    while (!pq.empty()) {
        auto top = pq.top(); pq.pop();
        int u = top.second;
        if (colors[u] != -1 || top.first.first < sat_degree[u]) continue;
        int chosen = -1;
        size_t base = (size_t)u * words;
        for (int w = 0; w < words; ++w) {
            uint64_t val = neighbor_colors[base + w];
            if (val == ~0ULL) continue;
            for (int i = 0; i < 64; ++i) {
                int c = w * 64 + i;
                if (c >= k) break;
                if (!(val & (1ULL << i))) { chosen = c; break; }
            }
            if (chosen != -1) break;
        }
        if (chosen == -1) return false;
        colors[u] = chosen; colored++;
        uint64_t mask = (1ULL << (chosen % 64));
        int word_idx = chosen / 64;
        for (int i = g.offsets[u]; i < g.offsets[u + 1]; ++i) {
            int v = g.edges[i];
            if (colors[v] == -1) {
                size_t v_base = (size_t)v * words + word_idx;
                if (!(neighbor_colors[v_base] & mask)) {
                    neighbor_colors[v_base] |= mask;
                    sat_degree[v]++;
                    pq.push({{sat_degree[v], g.degrees[v]}, v});
                }
            }
        }
    }
    return colored == n;
}

vector<int> get_sl_order(const Graph& g) {
    int n = g.n;
    vector<int> deg = g.degrees, head(n + 1, -1), prv(n, -1), nxt(n, -1), order(n);
    vector<bool> removed(n, false);
    auto add = [&](int d, int u) { nxt[u] = head[d]; if (head[d] != -1) prv[head[d]] = u; head[d] = u; prv[u] = -1; };
    auto rem = [&](int d, int u) { if (prv[u] != -1) nxt[prv[u]] = nxt[u]; else head[d] = nxt[u]; if (nxt[u] != -1) prv[nxt[u]] = prv[u]; };
    for (int i = 0; i < n; ++i) add(deg[i], i);
    int min_deg = 0;
    for (int i = n - 1; i >= 0; --i) {
        while (min_deg < n && head[min_deg] == -1) min_deg++;
        int u = head[min_deg]; rem(min_deg, u); removed[u] = true; order[i] = u;
        for (int j = g.offsets[u]; j < g.offsets[u + 1]; ++j) {
            int v = g.edges[j];
            if (!removed[v]) { rem(deg[v], v); deg[v]--; add(deg[v], v); if (deg[v] < min_deg) min_deg = deg[v]; }
        }
    }
    return order;
}

void ensure_exactly_k(const Graph& g, vector<int>& colors) {
    vector<bool> used(g.k, false);
    int used_count = 0;
    for (int c : colors) if (c >= 0 && c < g.k && !used[c]) { used[c] = true; used_count++; }
    if (used_count < g.k) {
        for (int c = 0; c < g.k; ++c) {
            if (!used[c]) {
                for (int v = 0; v < g.n; ++v) {
                    bool safe = true;
                    for (int i = g.offsets[v]; i < g.offsets[v + 1]; ++i) if (colors[g.edges[i]] == c) { safe = false; break; }
                    if (safe) { colors[v] = c; used[c] = true; used_count++; break; }
                }
            }
        }
    }
}

int main() {
    FastReader reader;
    int n = reader.read_int(), m = reader.read_int(), k = reader.read_int();
    if (n <= 0) return 0;
    Graph g(n, m, k); g.load(reader);
    vector<int> colors(n, -1);
    bool success = solve_greedy(g, get_sl_order(g), colors);
    if (!success) {
        vector<int> ldf(n); iota(ldf.begin(), ldf.end(), 0);
        sort(ldf.begin(), ldf.end(), [&](int a, int b) { return g.degrees[a] > g.degrees[b]; });
        success = solve_greedy(g, ldf, colors);
    }
    if (!success) success = solve_dsatur(g, colors);
    if (!success) {
        mt19937 rng(42); vector<int> rd(n); iota(rd.begin(), rd.end(), 0);
        auto start = chrono::steady_clock::now();
        while (chrono::duration_cast<chrono::seconds>(chrono::steady_clock::now() - start).count() < 240) {
            shuffle(rd.begin(), rd.end(), rng);
            if (solve_greedy(g, rd, colors)) { success = true; break; }
        }
    }
    if (success) {
        ensure_exactly_k(g, colors);
        for (int i = 0; i < n; ++i) printf("%d%c", colors[i], (i == n - 1 ? '\n' : ' '));
    }
    return 0;
}
```
