; SPIR-V
; Version: 1.0
; Generator: Manual
; Bound: 100
; Schema: 0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %u DescriptorSet 0
OpDecorate %u Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%m4f = OpTypeMatrix %v4f 4

%UBO = OpTypeStruct %m4f %m4f %m4f %v4f %v4f %v4f
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%u = OpVariable %ptr_Uniform_UBO Uniform

%ptr_Input_v2f = OpTypePointer Input %v2f
%ptr_Input_v3f = OpTypePointer Input %v3f
%worldPos = OpVariable %ptr_Input_v3f Input
%normal = OpVariable %ptr_Input_v3f Input
%uv = OpVariable %ptr_Input_v2f Input
%tangent = OpVariable %ptr_Input_v3f Input
%color = OpVariable %ptr_Input_v3f Input

%ptr_Output_v4f = OpTypePointer Output %v4f
%outColor = OpVariable %ptr_Output_v4f Output

%void_fn = OpTypeFunction %void

; Constants
%f0 = OpConstant %float 0.0
%f0_05 = OpConstant %float 0.05
%f0_1 = OpConstant %float 0.1
%f0_2 = OpConstant %float 0.2
%f0_4 = OpConstant %float 0.4
%f0_5 = OpConstant %float 0.5
%f0_6 = OpConstant %float 0.6
%f0_7 = OpConstant %float 0.7
%f1 = OpConstant %float 1.0
%f2 = OpConstant %float 2.0
%f8 = OpConstant %float 8.0
%f16 = OpConstant %float 16.0

%v2f_8_16 = OpConstantComposite %v2f %f8 %f16
%v3f_mortar = OpConstantComposite %v3f %f0_7 %f0_7 %f0_7
%v3f_brick = OpConstantComposite %v3f %f0_7 %f0_2 %f0_1

%int = OpTypeInt 32 1
%i3 = OpConstant %int 3
%ptr_Uniform_v4f = OpTypePointer Uniform %v4f

; Main
%main = OpFunction %void None %void_fn
%label = OpLabel

; Normalize normal
%n_raw = OpLoad %v3f %normal
%n = OpExtInst %v3f %1 Normalize %n_raw

; Shading calculation
%w_pos = OpLoad %v3f %worldPos
%l_ptr = OpAccessChain %ptr_Uniform_v4f %u %i3
%l_pos4 = OpLoad %v4f %l_ptr
%l_pos3 = OpVectorShuffle %v3f %l_pos4 %l_pos4 0 1 2
%l_vec = OpFSub %v3f %l_pos3 %w_pos
%l = OpExtInst %v3f %1 Normalize %l_vec

%dot_nl = OpDot %float %n %l
%diff = OpExtInst %float %1 FMax %dot_nl %f0
%sh_mul = OpFMul %float %f0_6 %diff
%shading = OpFAdd %float %f0_4 %sh_mul

; Brick pattern
%uv_raw = OpLoad %v2f %uv
%uv_s = OpFMul %v2f %uv_raw %v2f_8_16
%uv_x = OpCompositeExtract %float %uv_s 0
%uv_y = OpCompositeExtract %float %uv_s 1

%y_f = OpExtInst %float %1 Floor %uv_y
; Compute y_mod2 = y_f - 2.0 * floor(y_f / 2.0)
%div2 = OpFDiv %float %y_f %f2
%fdiv2 = OpExtInst %float %1 Floor %div2
%m2 = OpFMul %float %fdiv2 %f2
%y_mod2 = OpFSub %float %y_f %m2

%off = OpFMul %float %y_mod2 %f0_5
%x_st = OpFAdd %float %uv_x %off
%x_f = OpExtInst %float %1 Floor %x_st

%px = OpFSub %float %x_st %x_f
%py = OpFSub %float %uv_y %y_f

%bx = OpFOrdGreaterThan %bool %px %f0_05
%by = OpFOrdGreaterThan %bool %py %f0_1
%is_brick = OpLogicalAnd %bool %bx %by

%b_color = OpSelect %v3f %is_brick %v3f_brick %v3f_mortar

; Apply shading
%f_rgb = OpVectorTimesScalar %v3f %b_color %shading

; Build final color
%fr = OpCompositeExtract %float %f_rgb 0
%fg = OpCompositeExtract %float %f_rgb 1
%fb = OpCompositeExtract %float %f_rgb 2
%res = OpCompositeConstruct %v4f %fr %fg %fb %f1

OpStore %outColor %res
OpReturn
OpFunctionEnd
