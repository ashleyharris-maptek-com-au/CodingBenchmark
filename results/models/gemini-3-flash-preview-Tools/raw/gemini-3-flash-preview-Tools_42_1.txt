OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

; Debug info
               OpName %main "main"
               OpName %gl_PerVertex_in "gl_PerVertex"
               OpMemberName %gl_PerVertex_in 0 "gl_Position"
               OpName %gl_in "gl_in"
               OpName %worldPos "worldPos"
               OpName %normal "normal"
               OpName %color "color"
               OpName %gl_PerVertex_out "gl_PerVertex"
               OpMemberName %gl_PerVertex_out 0 "gl_Position"
               OpName %gl_out "gl_out"
               OpName %outColor "outColor"

; Decorations
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %outColor Location 0

; Types and constants
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %uint = OpTypeInt 32 0
     %uint_1 = OpConstant %uint 1
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0

%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4float = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3float Output

%ptr_in_v3float = OpTypePointer Input %v3float

    %float_h = OpConstant %float 0.05

; Function main
       %main = OpFunction %void None %3
          %5 = OpLabel

; Load input color from color[0]
%ptr_color_0 = OpAccessChain %ptr_in_v3float %color %int_0
      %c_val = OpLoad %v3float %ptr_color_0
               OpStore %outColor %c_val

; Load input position from gl_in[0].gl_Position
%ptr_pos_in = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
      %p_val = OpLoad %v4float %ptr_pos_in
        %p_x = OpCompositeExtract %float %p_val 0
        %p_y = OpCompositeExtract %float %p_val 1
        %p_z = OpCompositeExtract %float %p_val 2
        %p_w = OpCompositeExtract %float %p_val 3

%ptr_pos_out = OpAccessChain %ptr_out_v4float %gl_out %int_0

; Emit 4 vertices in triangle strip order: top, right, left, bottom

; 1. Top: (x, y+h)
      %y_top = OpFAdd %float %p_y %float_h
      %v_top = OpCompositeConstruct %v4float %p_x %y_top %p_z %p_w
               OpStore %ptr_pos_out %v_top
               OpEmitVertex

; 2. Right: (x+h, y)
    %x_right = OpFAdd %float %p_x %float_h
    %v_right = OpCompositeConstruct %v4float %x_right %p_y %p_z %p_w
               OpStore %ptr_pos_out %v_right
               OpEmitVertex

; 3. Left: (x-h, y)
     %x_left = OpFSub %float %p_x %float_h
     %v_left = OpCompositeConstruct %v4float %x_left %p_y %p_z %p_w
               OpStore %ptr_pos_out %v_left
               OpEmitVertex

; 4. Bottom: (x, y-h)
     %y_bot = OpFSub %float %p_y %float_h
     %v_bot = OpCompositeConstruct %v4float %p_x %y_bot %p_z %p_w
               OpStore %ptr_pos_out %v_bot
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
