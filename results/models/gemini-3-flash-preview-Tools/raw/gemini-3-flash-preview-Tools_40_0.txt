The task was to solve a 0/1 Integer Linear Program, specifically the Maximum Weight Independent Set problem. I have implemented a robust C++17 solution using a branch-and-bound backtracking algorithm. 

The solver includes several optimizations to ensure it can find the true optimum efficiently:
1.  **Branch-and-Bound Pruning**: The search space is pruned if the current weight plus the sum of all remaining potential weights cannot exceed the best weight found so far.
2.  **Heuristic Vertex Ordering**: Vertices are sorted by the heuristic `weight / (degree + 1)`. Prioritizing high-weight, low-degree vertices for inclusion helps find a high-quality solution early, which significantly improves pruning effectiveness.
3.  **Mandatory Inclusion**: If a candidate vertex has no neighbors within the current set of candidates, it is automatically included in the independent set for that branch, as its inclusion is always optimal.
4.  **Memory Efficiency**: The candidate sets are managed using reference passing and recursive restoration to minimize memory allocations and stack usage.

The complete C++17 code is provided below:

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long ll;

/**
 * Maximum Weight Independent Set (MWIS) Solver.
 * This problem is NP-hard. We use a branch-and-bound backtracking algorithm.
 * 
 * Pruning:
 * 1. Simple bound: If (current_weight + sum of weights of all candidate vertices) <= max_weight_found, prune.
 * 2. Mandatory inclusion: If a candidate vertex has no neighbors among other candidates, it's always optimal to include it.
 * 3. Heuristic ordering: We sort vertices by a heuristic (weight / (degree + 1)) to find good solutions early.
 */

int N, M;
ll W[2005];
bool adj[2005][2005];
vector<int> adj_list[2005];
ll max_w = 0;
vector<int> best_is;

void solve(vector<int>& P, ll sum_p, ll cur_w, vector<int>& cur_is) {
    if (P.empty()) {
        if (cur_w > max_w) {
            max_w = cur_w;
            best_is = cur_is;
        }
        return;
    }

    // Pruning: bound based on sum of available weights
    if (cur_w + sum_p <= max_w) return;

    // Pick a vertex to branch on. We take the one at the back (highest heuristic value).
    int v = P.back();
    P.pop_back();
    ll weight_v = W[v];

    // Check if v is isolated in current P
    bool has_neighbor = false;
    for (int u : P) {
        if (adj[v][u]) {
            has_neighbor = true;
            break;
        }
    }

    if (!has_neighbor) {
        // v is isolated in the current subgraph. Inclusion is always better than exclusion (for W[v] > 0).
        cur_is.push_back(v);
        solve(P, sum_p - weight_v, cur_w + weight_v, cur_is);
        cur_is.pop_back();
    } else {
        // Branch 1: v IS in the Independent Set
        vector<int> P_in;
        ll sum_in = 0;
        P_in.reserve(P.size());
        for (int u : P) {
            if (!adj[v][u]) {
                P_in.push_back(u);
                sum_in += W[u];
            }
        }
        cur_is.push_back(v);
        solve(P_in, sum_in, cur_w + weight_v, cur_is);
        cur_is.pop_back();

        // Branch 2: v is NOT in the Independent Set
        if (cur_w + sum_p - weight_v > max_w) {
            solve(P, sum_p - weight_v, cur_w, cur_is);
        }
    }

    // Restore P for the caller
    P.push_back(v);
}

int main() {
    // Optimize I/O
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> N >> M)) return 0;

    for (int i = 0; i < N; ++i) {
        cin >> W[i];
    }

    vector<bool> forced_zero(N, false);
    for (int i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v;
        if (u >= 0 && u < N && v >= 0 && v < N) {
            if (u == v) {
                forced_zero[u] = true;
            } else {
                if (!adj[u][v]) {
                    adj[u][v] = adj[v][u] = true;
                    adj_list[u].push_back(v);
                    adj_list[v].push_back(u);
                }
            }
        }
    }

    // Prepare candidate vertices
    vector<int> P;
    ll initial_sum = 0;
    for (int i = 0; i < N; ++i) {
        // We only consider vertices with positive weight and not forbidden by self-loops
        if (!forced_zero[i] && W[i] > 0) {
            P.push_back(i);
            initial_sum += W[i];
        }
    }

    // Heuristic sorting: weight / (degree + 1)
    sort(P.begin(), P.end(), [](int a, int b) {
        double ha = (double)W[a] / (adj_list[a].size() + 1);
        double hb = (double)W[b] / (adj_list[b].size() + 1);
        if (ha != hb) return ha < hb;
        return a < b;
    });

    vector<int> cur_is;
    solve(P, initial_sum, 0, cur_is);

    // Final Output
    cout << max_w << "\n";
    vector<int> final_x(N, 0);
    for (int v : best_is) final_x[v] = 1;
    for (int i = 0; i < N; ++i) {
        cout << final_x[i] << (i == N - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
```
