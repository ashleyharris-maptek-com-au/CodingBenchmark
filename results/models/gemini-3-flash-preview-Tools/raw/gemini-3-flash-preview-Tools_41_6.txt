; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %out_outColor
               OpExecutionMode %main OriginUpperLeft

               ; Decorations
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %out_outColor Location 0
               
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo_var DescriptorSet 0
               OpDecorate %ubo_var Binding 0

               ; Types and Constants
       %void = OpTypeVoid
    %void_fn = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
    %m4float = OpTypeMatrix %v4float 4
    
       %UBO = OpTypeStruct %m4float %m4float %m4float %v4float %v4float %v4float
    
    %ptr_Input_v3float = OpTypePointer Input %v3float
    %ptr_Output_v4float = OpTypePointer Output %v4float
    %ptr_Uniform_UBO = OpTypePointer Uniform %UBO
    %ptr_Uniform_v4float = OpTypePointer Uniform %v4float

    %float_0 = OpConstant %float 0.0
    %float_0_1 = OpConstant %float 0.1
    %float_0_6 = OpConstant %float 0.6
    %float_0_8 = OpConstant %float 0.8
    %float_1 = OpConstant %float 1.0
    %float_64 = OpConstant %float 64.0
    
        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4

               ; Variables
%in_worldPos = OpVariable %ptr_Input_v3float Input
  %in_normal = OpVariable %ptr_Input_v3float Input
%out_outColor = OpVariable %ptr_Output_v4float Output
    %ubo_var = OpVariable %ptr_Uniform_UBO Uniform

               ; Main Function
       %main = OpFunction %void None %void_fn
      %label = OpLabel
      
               ; Load worldPos and normal
   %worldPos = OpLoad %v3float %in_worldPos
     %normal = OpLoad %v3float %in_normal
     
               ; Normalize normal (N)
          %N = OpExtInst %v3float %1 Normalize %normal
          
               ; Calculate Light Vector (L)
   %lp_ptr = OpAccessChain %ptr_Uniform_v4float %ubo_var %int_3
        %lp4 = OpLoad %v4float %lp_ptr
        %lp3 = OpVectorShuffle %v3float %lp4 %lp4 0 1 2
      %L_raw = OpFSub %v3float %lp3 %worldPos
          %L = OpExtInst %v3float %1 Normalize %L_raw
          
               ; Calculate View Vector (V)
   %cp_ptr = OpAccessChain %ptr_Uniform_v4float %ubo_var %int_4
        %cp4 = OpLoad %v4float %cp_ptr
        %cp3 = OpVectorShuffle %v3float %cp4 %cp4 0 1 2
      %V_raw = OpFSub %v3float %cp3 %worldPos
          %V = OpExtInst %v3float %1 Normalize %V_raw
          
               ; Calculate Half-Vector (H)
      %H_raw = OpFAdd %v3float %L %V
          %H = OpExtInst %v3float %1 Normalize %H_raw
          
               ; Diffuse = max(dot(N, L), 0.0)
      %dotNL = OpDot %float %N %L
    %diffuse = OpExtInst %float %1 FMax %dotNL %float_0
    
               ; Specular = pow(max(dot(N, H), 0.0), 64.0)
      %dotNH = OpDot %float %N %H
    %maxNH0 = OpExtInst %float %1 FMax %dotNH %float_0
   %specular = OpExtInst %float %1 Pow %maxNH0 %float_64

               ; intensity = 0.1 + 0.6 * diffuse + 0.8 * specular
     %term1 = OpFMul %float %float_0_6 %diffuse
     %term2 = OpFMul %float %float_0_8 %specular
     %sum1 = OpFAdd %float %float_0_1 %term1
  %intensity = OpFAdd %float %sum1 %term2

               ; Final Color
   %out_res = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1
               OpStore %out_outColor %out_res
               
               OpReturn
               OpFunctionEnd
