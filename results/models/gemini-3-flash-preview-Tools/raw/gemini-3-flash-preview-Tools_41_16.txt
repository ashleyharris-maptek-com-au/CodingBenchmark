; Bound: 150

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inUV %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Pointers
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO

; Variables
%inUV = OpVariable %ptr_Input_v2float Input
%outColor = OpVariable %ptr_Output_v4float Output
%uboVar = OpVariable %ptr_Uniform_UBO Uniform

; Constants
%f0_0 = OpConstant %float 0.0
%f0_5 = OpConstant %float 0.5
%f1_0 = OpConstant %float 1.0
%f2_0 = OpConstant %float 2.0
%f3_0 = OpConstant %float 3.0
%f4_0 = OpConstant %float 4.0
%f20_0 = OpConstant %float 20.0

%i0 = OpConstant %int 0
%i1 = OpConstant %int 1
%i20 = OpConstant %int 20

%v2f0_0 = OpConstantComposite %v2float %f0_0 %f0_0

; Main Function
%main = OpFunction %void None %void
%label_entry = OpLabel

; Load UV
%uv = OpLoad %v2float %inUV

; Calculate c = (uv.x*3-2, uv.y*2-1)
%uv_x = OpCompositeExtract %float %uv 0
%c_x_tmp = OpFMul %float %uv_x %f3_0
%c_x = OpFSub %float %c_x_tmp %f2_0

%uv_y = OpCompositeExtract %float %uv 1
%c_y_tmp = OpFMul %float %uv_y %f2_0
%c_y = OpFSub %float %c_y_tmp %f1_0

%c = OpCompositeConstruct %v2float %c_x %c_y

OpBranch %label_header

%label_header = OpLabel
%z = OpPhi %v2float %v2f0_0 %label_entry %z_new %label_body
%iter = OpPhi %int %i0 %label_entry %iter_new %label_body

; Loop Condition: iter < 20
%cond_iter = OpSLessThan %bool %iter %i20

; Escape Condition: |z|^2 <= 4
%zx = OpCompositeExtract %float %z 0
%zy = OpCompositeExtract %float %z 1
%zx2 = OpFMul %float %zx %zx
%zy2 = OpFMul %float %zy %zy
%z2 = OpFAdd %float %zx2 %zy2
%cond_z = OpFOrdLessThanEqual %bool %z2 %f4_0

%cond = OpLogicalAnd %bool %cond_iter %cond_z

OpLoopMerge %label_merge %label_body None
OpBranchConditional %cond %label_body %label_merge

%label_body = OpLabel
; x_new = z.x*z.x - z.y*z.y + c.x
%x_new_tmp = OpFSub %float %zx2 %zy2
%x_new = OpFAdd %float %x_new_tmp %c_x

; y_new = 2.0*z.x*z.y + c.y
%y_new_tmp1 = OpFMul %float %f2_0 %zx
%y_new_tmp2 = OpFMul %float %y_new_tmp1 %zy
%y_new = OpFAdd %float %y_new_tmp2 %c_y

%z_new = OpCompositeConstruct %v2float %x_new %y_new
%iter_new = OpIAdd %int %iter %i1
OpBranch %label_header

%label_merge = OpLabel
; t = iterations / 20
%iter_f = OpConvertSToF %float %iter
%t = OpFDiv %float %iter_f %f20_0

; Color (t, t*0.5, 1.0-t, 1.0)
%t_05 = OpFMul %float %t %f0_5
%one_minus_t = OpFSub %float %f1_0 %t
%final_color = OpCompositeConstruct %v4float %t %t_05 %one_minus_t %f1_0

OpStore %outColor %final_color
OpReturn
OpFunctionEnd
