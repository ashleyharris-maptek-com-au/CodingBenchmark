               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std450"

               OpEntryPoint Fragment %main "main" %worldPos %outColor
               OpExecutionMode %main OriginUpperLeft

               ; Decorations
               OpDecorate %worldPos Location 0
               OpDecorate %outColor Location 0
               
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 2 Offset 128

      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
    %mat4v4  = OpTypeMatrix %v4float 4
       %UBO  = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO

               ; Constants
    %float_0 = OpConstant %float 0.0
    %float_1 = OpConstant %float 1.0
  %float_0_5 = OpConstant %float 0.5
  %float_0_1 = OpConstant %float 0.1
  %float_0_4 = OpConstant %float 0.4

   %botColor = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
   %topColor = OpConstantComposite %v3float %float_1 %float_1 %float_1

               ; Variables

   %outColor = OpVariable %ptr_Output_v4float Output
   %ubo_var  = OpVariable %ptr_Uniform_UBO Uniform

               ; Main
       %main = OpFunction %void None %3
          %5 = OpLabel
    %pos_val = OpLoad %v3float %worldPos
      %pos_y = OpCompositeExtract %float %pos_val 1
               
               ; t = clamp(worldPos.y * 0.5 + 0.5, 0.0, 1.0)
         %14 = OpFMul %float %pos_y %float_0_5
         %15 = OpFAdd %float %14 %float_0_5
          %t = OpExtInst %float %1 FClamp %15 %float_0 %float_1

               ; color = mix(bot, top, t)
        %mix = OpExtInst %v3float %1 FMix %botColor %topColor %t

               ; outColor = vec4(color, 1.0)
   %res_vec4 = OpCompositeConstruct %v4float %mix %float_1

               OpStore %outColor %res_vec4
               OpReturn
