use std::io::{self, Read, Write};
use std::time::Instant;

struct Point {
    x: f32,
    y: f32,
}

struct Edge {
    u: usize,
    v: usize,
}

fn main() {
    let start_time = Instant::now();
    
    // Fast I/O: Read all at once since we have plenty of memory (32GB)
    let mut input = String::new();
    if let Err(_) = io::stdin().read_to_string(&mut input) {
        return;
    }
    let mut tokens = input.split_ascii_whitespace();
    
    let n: usize = match tokens.next() {
        Some(t) => t.parse().unwrap_or(0),
        None => return,
    };
    let m: usize = match tokens.next() {
        Some(t) => t.parse().unwrap_or(0),
        None => 0,
    };

    if n == 0 { return; }

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u = tokens.next().and_then(|t| t.parse::<usize>().ok());
        let v = tokens.next().and_then(|t| t.parse::<usize>().ok());
        if let (Some(u_val), Some(v_val)) = (u, v) {
            if u_val < n && v_val < n && u_val != v_val {
                edges.push(Edge { u: u_val, v: v_val });
            }
        }
    }

    // Layout initialization
    let mut nodes = Vec::with_capacity(n);
    let mut rng_state = 42u64;
    let mut next_rng = || {
        rng_state = rng_state.wrapping_mul(6364136223846793005).wrapping_add(1);
        ((rng_state >> 32) as u32) as f32 / u32::MAX as f32
    };

    let area = n as f32;
    let side = area.sqrt();
    for _ in 0..n {
        nodes.push(Point {
            x: next_rng() * side,
            y: next_rng() * side,
        });
    }

    let k = 1.0;
    // Adapt iterations to graph size
    let iterations = if n < 500 {
        300
    } else if n < 5000 {
        200
    } else if n < 50000 {
        100
    } else {
        60
    };

    let mut t = side / 5.0; // Slightly higher initial temperature
    let initial_t = t;
    
    let mut dx = vec![0.0f32; n];
    let mut dy = vec![0.0f32; n];

    for _iter in 0..iterations {
        // Safety timeout
        if start_time.elapsed().as_secs_f32() > 27.0 {
            break;
        }

        for i in 0..n {
            dx[i] = 0.0;
            dy[i] = 0.0;
        }

        // Repulsive forces
        if n <= 1000 {
            // O(N^2) for small graphs
            for i in 0..n {
                for j in i + 1..n {
                    let d_x = nodes[i].x - nodes[j].x;
                    let d_y = nodes[i].y - nodes[j].y;
                    let dist_sq = d_x * d_x + d_y * d_y + 1e-6;
                    let dist = dist_sq.sqrt();
                    // Global repulsion for small graphs
                    let force = k * k / dist;
                    let fx = (d_x / dist) * force;
                    let fy = (d_y / dist) * force;
                    dx[i] += fx;
                    dy[i] += fy;
                    dx[j] -= fx;
                    dy[j] -= fy;
                }
            }
        } else {
            // Grid-based repulsion for large graphs
            let mut min_x = nodes[0].x;
            let mut max_x = nodes[0].x;
            let mut min_y = nodes[0].y;
            let mut max_y = nodes[0].y;
            for p in &nodes {
                if p.x < min_x { min_x = p.x; }
                if p.x > max_x { max_x = p.x; }
                if p.y < min_y { min_y = p.y; }
                if p.y > max_y { max_y = p.y; }
            }

            let grid_dim = (n as f32).sqrt().ceil() as usize;
            let grid_dim = grid_dim.min(400); // Balance between memory and precision
            let cell_w = (max_x - min_x + 1e-6) / grid_dim as f32;
            let cell_h = (max_y - min_y + 1e-6) / grid_dim as f32;

            let mut head = vec![None; grid_dim * grid_dim];
            let mut next = vec![None; n];

            for i in 0..n {
                let gx = (((nodes[i].x - min_x) / cell_w) as usize).min(grid_dim - 1);
                let gy = (((nodes[i].y - min_y) / cell_h) as usize).min(grid_dim - 1);
                let g_idx = gy * grid_dim + gx;
                next[i] = head[g_idx];
                head[g_idx] = Some(i);
            }

            // Radius for repulsion (in cell units)
            // Since we want to cover ~2k distance, and cell_w is roughly (side/sqrt(N)) = 1k,
            // 2-3 cells radius is sufficient.
            let search_r = 2;

            for i in 0..n {
                let gx = (((nodes[i].x - min_x) / cell_w) as usize).min(grid_dim - 1);
                let gy = (((nodes[i].y - min_y) / cell_h) as usize).min(grid_dim - 1);

                for dy_grid in -search_r..=search_r {
                    let ny = gy as i32 + dy_grid;
                    if ny < 0 || ny >= grid_dim as i32 { continue; }
                    for dx_grid in -search_r..=search_r {
                        let nx = gx as i32 + dx_grid;
                        if nx < 0 || nx >= grid_dim as i32 { continue; }
                        
                        let g_idx = (ny as usize) * grid_dim + (nx as usize);
                        let mut curr = head[g_idx];
                        while let Some(j) = curr {
                            if i < j {
                                let d_x = nodes[i].x - nodes[j].x;
                                let d_y = nodes[i].y - nodes[j].y;
                                let dist_sq = d_x * d_x + d_y * d_y + 1e-6;
                                let dist = dist_sq.sqrt();
                                // Limit range for large graphs to maintain O(N)
                                if dist < 3.0 * k {
                                    let force = k * k / dist;
                                    let fx = (d_x / dist) * force;
                                    let fy = (d_y / dist) * force;
                                    dx[i] += fx;
                                    dy[i] += fy;
                                    dx[j] -= fx;
                                    dy[j] -= fy;
                                }
                            }
                            curr = next[j];
                        }
                    }
                }
            }
        }

        // Attractive forces (Edges)
        for edge in &edges {
            let u = edge.u;
            let v = edge.v;
            let d_x = nodes[u].x - nodes[v].x;
            let d_y = nodes[u].y - nodes[v].y;
            let dist_sq = d_x * d_x + d_y * d_y + 1e-6;
            let dist = dist_sq.sqrt();
            let force = dist_sq / k;
            let fx = (d_x / dist) * force;
            let fy = (d_y / dist) * force;
            dx[u] -= fx;
            dy[u] -= fy;
            dx[v] += fx;
            dy[v] += fy;
        }

        // Global gravity to keep components together
        let mut sum_x = 0.0;
        let mut sum_y = 0.0;
        for p in &nodes {
            sum_x += p.x;
            sum_y += p.y;
        }
        let cx = sum_x / n as f32;
        let cy = sum_y / n as f32;
        let gravity = 0.02;
        for i in 0..n {
            dx[i] -= (nodes[i].x - cx) * gravity;
            dy[i] -= (nodes[i].y - cy) * gravity;
        }

        // Update positions within temperature bounds
        for i in 0..n {
            let dist_sq = dx[i] * dx[i] + dy[i] * dy[i] + 1e-6;
            let dist = dist_sq.sqrt();
            let limited_dist = dist.min(t);
            nodes[i].x += (dx[i] / dist) * limited_dist;
            nodes[i].y += (dy[i] / dist) * limited_dist;
        }

        // Cool down
        t -= initial_t / iterations as f32;
    }

    // Output results
    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    let _ = writeln!(out, "{}", n);
    for p in nodes {
        let _ = writeln!(out, "{:.6} {:.6}", p.x, p.y);
    }
}
