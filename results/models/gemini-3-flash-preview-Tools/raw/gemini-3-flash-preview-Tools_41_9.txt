; SPIR-V
; Version: 1.0
; Generator: Google Games; 0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %outColor
               OpExecutionMode %main OriginUpperLeft

; Decorations
               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %inUV Location 2
               OpDecorate %outColor Location 0

               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubovar DescriptorSet 0
               OpDecorate %ubovar Binding 0

; Types
       %void = OpTypeVoid
    %void_fn = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
    %mat4v4  = OpTypeMatrix %v4float 4

        %UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float

%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_input_v2float = OpTypePointer Input %v2float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_v4float = OpTypePointer Uniform %v4float

; Inputs/Outputs/Uniforms
%inWorldPos = OpVariable %ptr_input_v3float Input
%inNormal   = OpVariable %ptr_input_v3float Input
%inUV       = OpVariable %ptr_input_v2float Input
%outColor   = OpVariable %ptr_output_v4float Output
%ubovar     = OpVariable %ptr_uniform_UBO Uniform

; Constants
     %f0_0 = OpConstant %float 0.0
     %f0_2 = OpConstant %float 0.2
     %f0_5 = OpConstant %float 0.5
     %f0_9 = OpConstant %float 0.9
     %f1_0 = OpConstant %float 1.0
     %f2_0 = OpConstant %float 2.0
     %f8_0 = OpConstant %float 8.0

      %int = OpTypeInt 32 1
    %int_3 = OpConstant %int 3
     %bool = OpTypeBool

; Main
       %main = OpFunction %void None %void_fn
      %label = OpLabel

; Lambertian Shading
     %normal = OpLoad %v3float %inNormal
    %n_norm = OpExtInst %v3float %1 Normalize %normal

   %worldPos = OpLoad %v3float %inWorldPos
 %lightPos_ptr = OpAccessChain %ptr_uniform_v4float %ubovar %int_3
  %lightPos4 = OpLoad %v4float %lightPos_ptr
  %lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

      %L_vec = OpFSub %v3float %lightPos %worldPos
     %L_norm = OpExtInst %v3float %1 Normalize %L_vec

       %dot = OpDot %float %n_norm %L_norm
   %diffuse = OpExtInst %float %1 FMax %dot %f0_0
    %diffuse_05 = OpFMul %float %diffuse %f0_5
    %factor = OpFAdd %float %f0_5 %diffuse_05

; Checkerboard
         %uv = OpLoad %v2float %inUV
       %uv8 = OpVectorTimesScalar %v2float %uv %f8_0
     %uv8_f = OpExtInst %v2float %1 Floor %uv8
        %u8 = OpCompositeExtract %float %uv8_f 0
        %v8 = OpCompositeExtract %float %uv8_f 1
      %sum8 = OpFAdd %float %u8 %v8
    %parity = OpFMod %float %sum8 %f2_0
    
 %is_zero = OpFOrdLessThan %bool %parity %f1_0
 %checker = OpSelect %float %is_zero %f0_9 %f0_2

; Final Color
  %final_val = OpFMul %float %checker %factor
     %res_v4 = OpCompositeConstruct %v4float %final_val %final_val %final_val %f1_0
               OpStore %outColor %res_v4
               OpReturn
               OpFunctionEnd
