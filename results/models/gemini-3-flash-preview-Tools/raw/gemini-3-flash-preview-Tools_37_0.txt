The Directed Feedback Vertex Set (DFVS) problem is solved using an exact algorithm that combines graph reduction rules, SCC decomposition, and iterative deepening branching on the shortest cycles.

### Algorithm Strategy

1.  **Graph Reductions**: Before branching, the graph is simplified using the following rules:
    *   **Self-loop**: Any vertex with a self-loop must be in the DFVS.
    *   **Sink/Source**: Vertices with in-degree or out-degree of 0 cannot be part of any cycle and are removed.
    *   **Degree-1 (IN1/OUT1)**: If a vertex $v$ has an in-degree of 1 (predecessor $u$), any cycle through $v$ also passes through $u$. Thus, picking $u$ is always at least as good as picking $v$. We bypass $v$ by connecting $u$ directly to all successors of $v$.
2.  **SCC Decomposition**: The graph is decomposed into Strongly Connected Components (SCCs). Since cycles are contained within SCCs, we can solve each SCC independently and sum the results.
3.  **Iterative Deepening**: For each non-trivial SCC, we search for a DFVS of size $k=0, 1, 2, \dots$ until a solution is found. This ensures we find the minimum size.
4.  **Branching on Shortest Cycle**: Within the search for a specific size $k$, we find a shortest cycle. At least one vertex in that cycle must be in the DFVS. We branch on each vertex in the cycle, reducing the problem to size $k-1$.
5.  **Lower Bound Pruning**: The search is pruned if the number of disjoint non-trivial SCCs exceeds the remaining budget $k$.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    public static void Main()
    {
        // Increase stack size for deep recursion to handle large graphs
        var thread = new System.Threading.Thread(Work, 1024 * 1024 * 128);
        thread.Start();
        thread.Join();
    }

    static void Work()
    {
        var reader = new FastReader(Console.In);
        int n = reader.NextInt();
        int m = reader.NextInt();
        if (n == -1) return;

        var adj = new HashSet<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            int u = reader.NextInt();
            int v = reader.NextInt();
            if (u >= 0 && u < n && v >= 0 && v < n)
            {
                adj[u].Add(v);
            }
        }

        var solver = new Solver(n, adj);
        var result = solver.Solve();

        Console.WriteLine(result.Count);
        Console.WriteLine(string.Join(" ", result));
    }
}

class FastReader
{
    System.IO.TextReader reader;
    public FastReader(System.IO.TextReader reader) { this.reader = reader; }
    public string Next()
    {
        var sb = new System.Text.StringBuilder();
        int c;
        while ((c = reader.Read()) != -1 && char.IsWhiteSpace((char)c)) ;
        if (c == -1) return null;
        sb.Append((char)c);
        while ((c = reader.Read()) != -1 && !char.IsWhiteSpace((char)c)) sb.Append((char)c);
        return sb.ToString();
    }
    public int NextInt()
    {
        string s = Next();
        return s == null ? -1 : int.Parse(s);
    }
}

class Solver
{
    int n;
    HashSet<int>[] initialAdj;
    int[] disc, low, dist, parent;
    bool[] onStack, inQueue;
    Stack<int> stack;
    int timer;

    public Solver(int n, HashSet<int>[] adj)
    {
        this.n = n;
        this.initialAdj = adj;
        disc = new int[n];
        low = new int[n];
        onStack = new bool[n];
        stack = new Stack<int>();
        dist = new int[n];
        parent = new int[n];
        inQueue = new bool[n];
    }

    public List<int> Solve()
    {
        var active = new HashSet<int>(Enumerable.Range(0, n));
        var adj = new Dictionary<int, HashSet<int>>();
        var revAdj = new Dictionary<int, HashSet<int>>();
        for (int i = 0; i < n; i++)
        {
            adj[i] = new HashSet<int>(initialAdj[i]);
            revAdj[i] = new HashSet<int>();
        }
        for (int i = 0; i < n; i++)
        {
            foreach (var v in adj[i]) revAdj[v].Add(i);
        }

        var dfvs = new List<int>();
        ApplyReductions(active, adj, revAdj, dfvs);

        var sccs = GetSCCs(active, adj);
        foreach (var scc in sccs)
        {
            if (scc.Count > 1 || (scc.Count == 1 && adj[scc[0]].Contains(scc[0])))
            {
                var sccDFVS = SolveSCC(new HashSet<int>(scc), adj, revAdj);
                dfvs.AddRange(sccDFVS);
            }
        }
        return dfvs;
    }

    void ApplyReductions(HashSet<int> active, Dictionary<int, HashSet<int>> adj, Dictionary<int, HashSet<int>> revAdj, List<int> dfvs)
    {
        var queue = new Queue<int>(active);
        foreach (var v in active) inQueue[v] = true;

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            inQueue[v] = false;
            if (!active.Contains(v)) continue;

            if (adj[v].Contains(v))
            {
                dfvs.Add(v);
                var neighbors = adj[v].Concat(revAdj[v]).ToList();
                RemoveNode(v, active, adj, revAdj);
                foreach (var nb in neighbors) if (active.Contains(nb) && !inQueue[nb]) { queue.Enqueue(nb); inQueue[nb] = true; }
                continue;
            }

            if (adj[v].Count == 0 || revAdj[v].Count == 0)
            {
                var neighbors = adj[v].Concat(revAdj[v]).ToList();
                RemoveNode(v, active, adj, revAdj);
                foreach (var nb in neighbors) if (active.Contains(nb) && !inQueue[nb]) { queue.Enqueue(nb); inQueue[nb] = true; }
                continue;
            }

            if (adj[v].Count == 1)
            {
                int w = adj[v].First();
                var predecessors = revAdj[v].ToList();
                foreach (var u in predecessors)
                {
                    if (u != v)
                    {
                        if (adj[u].Add(w))
                        {
                            revAdj[w].Add(u);
                            if (!inQueue[u]) { queue.Enqueue(u); inQueue[u] = true; }
                            if (!inQueue[w]) { queue.Enqueue(w); inQueue[w] = true; }
                        }
                    }
                }
                RemoveNode(v, active, adj, revAdj);
                continue;
            }

            if (revAdj[v].Count == 1)
            {
                int u = revAdj[v].First();
                var successors = adj[v].ToList();
                foreach (var w in successors)
                {
                    if (w != v)
                    {
                        if (adj[u].Add(w))
                        {
                            revAdj[w].Add(u);
                            if (!inQueue[u]) { queue.Enqueue(u); inQueue[u] = true; }
                            if (!inQueue[w]) { queue.Enqueue(w); inQueue[w] = true; }
                        }
                    }
                }
                RemoveNode(v, active, adj, revAdj);
                continue;
            }
        }
    }

    void RemoveNode(int v, HashSet<int> active, Dictionary<int, HashSet<int>> adj, Dictionary<int, HashSet<int>> revAdj)
    {
        active.Remove(v);
        if (adj.TryGetValue(v, out var successors))
        {
            foreach (var w in successors) if (revAdj.TryGetValue(w, out var preds)) preds.Remove(v);
            adj.Remove(v);
        }
        if (revAdj.TryGetValue(v, out var predecessors))
        {
            foreach (var u in predecessors) if (adj.TryGetValue(u, out var succs)) succs.Remove(v);
            revAdj.Remove(v);
        }
    }

    List<int> SolveSCC(HashSet<int> nodes, Dictionary<int, HashSet<int>> adj, Dictionary<int, HashSet<int>> revAdj)
    {
        for (int k = 0; ; k++)
        {
            var res = FindDFVSK(nodes, adj, revAdj, k);
            if (res != null) return res;
        }
    }

    List<int> FindDFVSK(HashSet<int> nodes, Dictionary<int, HashSet<int>> adj, Dictionary<int, HashSet<int>> revAdj, int k)
    {
        var localActive = new HashSet<int>(nodes);
        var localAdj = new Dictionary<int, HashSet<int>>();
        var localRev = new Dictionary<int, HashSet<int>>();
        foreach (var v in nodes)
        {
            localAdj[v] = new HashSet<int>(adj[v]);
            localRev[v] = new HashSet<int>(revAdj[v]);
        }
        var currentDFVS = new List<int>();
        ApplyReductions(localActive, localAdj, localRev, currentDFVS);
        if (currentDFVS.Count > k) return null;
        return Branch(localActive, localAdj, localRev, k - currentDFVS.Count, currentDFVS);
    }

    List<int> Branch(HashSet<int> active, Dictionary<int, HashSet<int>> adj, Dictionary<int, HashSet<int>> revAdj, int k, List<int> current)
    {
        var sccs = GetSCCs(active, adj);
        var nonTrivial = sccs.Where(s => s.Count > 1 || (s.Count == 1 && adj[s[0]].Contains(s[0]))).ToList();
        if (nonTrivial.Count == 0) return current;
        if (k < nonTrivial.Count) return null;

        var scc = nonTrivial[0];
        var sccSet = new HashSet<int>(scc);
        var cycle = FindShortestCycle(sccSet, adj, revAdj);
        if (cycle == null) return current;

        foreach (var v in cycle)
        {
            var nextActive = new HashSet<int>(active);
            var nextAdj = new Dictionary<int, HashSet<int>>();
            var nextRev = new Dictionary<int, HashSet<int>>();
            foreach (var node in active)
            {
                nextAdj[node] = new HashSet<int>(adj[node]);
                nextRev[node] = new HashSet<int>(revAdj[node]);
            }
            var nextDFVS = new List<int>(current);
            nextDFVS.Add(v);
            RemoveNode(v, nextActive, nextAdj, nextRev);
            var localRed = new List<int>();
            ApplyReductions(nextActive, nextAdj, nextRev, localRed);
            nextDFVS.AddRange(localRed);
            if (nextDFVS.Count <= current.Count + k)
            {
                var res = Branch(nextActive, nextAdj, nextRev, current.Count + k - nextDFVS.Count, nextDFVS);
                if (res != null) return res;
            }
        }
        return null;
    }

    List<List<int>> sccsList;
    List<List<int>> GetSCCs(HashSet<int> active, Dictionary<int, HashSet<int>> adj)
    {
        foreach (var v in active) disc[v] = -1;
        timer = 0;
        sccsList = new List<List<int>>();
        stack.Clear();
        foreach (var v in active) if (disc[v] == -1) DFS(v, active, adj);
        return sccsList;
    }

    void DFS(int u, HashSet<int> active, Dictionary<int, HashSet<int>> adj)
    {
        disc[u] = low[u] = timer++;
        stack.Push(u);
        onStack[u] = true;
        if (adj.TryGetValue(u, out var successors))
        {
            foreach (var v in successors)
            {
                if (!active.Contains(v)) continue;
                if (disc[v] == -1)
                {
                    DFS(v, active, adj);
                    low[u] = Math.Min(low[u], low[v]);
                }
                else if (onStack[v]) low[u] = Math.Min(low[u], disc[v]);
            }
        }
        if (low[u] == disc[u])
        {
            var scc = new List<int>();
            while (true)
            {
                int node = stack.Pop();
                onStack[node] = false;
                scc.Add(node);
                if (node == u) break;
            }
            sccsList.Add(scc);
        }
    }

    List<int> FindShortestCycle(HashSet<int> active, Dictionary<int, HashSet<int>> adj, Dictionary<int, HashSet<int>> revAdj)
    {
        int bestLen = int.MaxValue;
        List<int> bestCycle = null;
        var nodes = active.ToList();
        nodes.Sort((a, b) => (adj[b].Count * revAdj[b].Count).CompareTo(adj[a].Count * revAdj[a].Count));
        int tried = 0;
        foreach (var start in nodes)
        {
            if (tried++ > 50 && bestCycle != null) break;
            foreach (var v in active) dist[v] = -1;
            var q = new Queue<int>();
            q.Enqueue(start);
            dist[start] = 0;
            while (q.Count > 0)
            {
                int u = q.Dequeue();
                if (dist[u] >= bestLen - 1) break;
                if (adj.TryGetValue(u, out var successors))
                {
                    foreach (var v in successors)
                    {
                        if (!active.Contains(v)) continue;
                        if (v == start)
                        {
                            if (dist[u] + 1 < bestLen)
                            {
                                bestLen = dist[u] + 1;
                                bestCycle = new List<int>();
                                int curr = u;
                                while (curr != start) { bestCycle.Add(curr); curr = parent[curr]; }
                                bestCycle.Add(start);
                            }
                            goto next;
                        }
                        if (dist[v] == -1) { dist[v] = dist[u] + 1; parent[v] = u; q.Enqueue(v); }
                    }
                }
            }
            next: if (bestLen == 2) break;
        }
        return bestCycle;
    }
}
```
