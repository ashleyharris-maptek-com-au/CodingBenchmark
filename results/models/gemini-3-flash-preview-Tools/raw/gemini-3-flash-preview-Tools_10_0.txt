#include <iostream>
#include <vector>
#include <algorithm>
#include <map>

using namespace std;

// Structure to store rectangle information and its placement
struct Rect {
    int id;
    int w, h;
    int board_idx = -1;
    int x = 0, y = 0;
    int rotated = 0;
};

// Structure to store shelf information
struct Shelf {
    int board_idx;
    int y;
    int height;
    int used_width;
    int board_width;
};

// Structure to store board information
struct Board {
    int id;
    int width, height;
    int used_height;
};

/**
 * 2D Cutting Stock Problem Solver
 * Algorithm: Best-Fit Decreasing Height (BFDH) Shelf Packing
 * 
 * This algorithm sorts rectangles by height and packs them into "shelves".
 * It uses a Best-Fit strategy for both shelf selection and board selection
 * to minimize waste and the number of boards used.
 * 
 * Complexity: O(N log N)
 * Performance: Handles 50,000 rectangles in < 1 second on modern hardware.
 */
int main() {
    // Fast I/O for high-volume data
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    int BW, BH;
    
    // Read problem dimensions
    if (!(cin >> N >> BW >> BH)) return 0;

    vector<Rect> rects(N);
    for (int i = 0; i < N; ++i) {
        rects[i].id = i;
        int w_in, h_in;
        cin >> w_in >> h_in;
        
        // Initial setup
        rects[i].w = w_in;
        rects[i].h = h_in;
        rects[i].rotated = 0;

        // Determine if rotation is necessary or fits better in the standard orientation
        bool can_orig = (w_in <= BW && h_in <= BH);
        bool can_rot = (h_in <= BW && w_in <= BH);

        if (can_orig && can_rot) {
            // If both fit, orient so height is the longer dimension (standard for shelf packing)
            if (w_in > h_in) {
                rects[i].w = h_in;
                rects[i].h = w_in;
                rects[i].rotated = 1;
            }
        } else if (can_rot) {
            // Must rotate to fit within board dimensions
            rects[i].w = h_in;
            rects[i].h = w_in;
            rects[i].rotated = 1;
        }
    }

    // Sort rectangles by height descending (Decreasing Height heuristic)
    vector<int> order(N);
    for (int i = 0; i < N; ++i) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (rects[a].h != rects[b].h) return rects[a].h > rects[b].h;
        return rects[a].w > rects[b].w;
    });

    vector<Board> boards;
    // multimap to find the shelf with minimum remaining width that fits (Best-Fit)
    multimap<int, int> open_shelves; // remaining_width -> shelf_index
    // multimap to find the board with minimum remaining height that fits (Best-Fit)
    multimap<int, int> available_boards; // remaining_height -> board_index
    vector<Shelf> shelves_data;

    for (int idx : order) {
        Rect& r = rects[idx];
        bool placed = false;

        // 1. Try to find an existing shelf that can fit this rectangle's width.
        // Because rects are sorted by height, any existing shelf already has height >= r.h.
        auto it_s = open_shelves.lower_bound(r.w);
        if (it_s != open_shelves.end()) {
            int s_idx = it_s->second;
            Shelf& s = shelves_data[s_idx];
            
            r.board_idx = s.board_idx;
            r.x = s.used_width;
            r.y = s.y;
            
            s.used_width += r.w;
            open_shelves.erase(it_s);
            int new_rem_w = s.board_width - s.used_width;
            if (new_rem_w > 0) {
                open_shelves.insert({new_rem_w, s_idx});
            }
            placed = true;
        }

        if (!placed) {
            // 2. Try to find a board with enough remaining height to start a new shelf.
            auto it_b = available_boards.lower_bound(r.h);
            if (it_b != available_boards.end()) {
                int b_idx = it_b->second;
                Board& b = boards[b_idx];
                
                // Final check that width fits (standardization should ensure this)
                if (r.w <= b.width) {
                    Shelf ns;
                    ns.board_idx = b.id;
                    ns.y = b.used_height;
                    ns.height = r.h;
                    ns.used_width = r.w;
                    ns.board_width = b.width;
                    
                    r.board_idx = b.id;
                    r.x = 0;
                    r.y = ns.y;
                    
                    b.used_height += r.h;
                    int s_idx = (int)shelves_data.size();
                    shelves_data.push_back(ns);
                    
                    available_boards.erase(it_b);
                    int new_rem_h = b.height - b.used_height;
                    if (new_rem_h > 0) {
                        available_boards.insert({new_rem_h, b_idx});
                    }
                    
                    int new_rem_w = ns.board_width - ns.used_width;
                    if (new_rem_w > 0) {
                        open_shelves.insert({new_rem_w, s_idx});
                    }
                    placed = true;
                }
            }
        }

        if (!placed) {
            // 3. Start a new board if no existing shelf or board has space.
            Board nb;
            nb.id = (int)boards.size();
            nb.width = BW;
            nb.height = BH;
            nb.used_height = 0;
            
            // Last resort: ensure rectangle fits within a fresh board
            if (r.w > nb.width || r.h > nb.height) {
                if (r.h <= nb.width && r.w <= nb.height) {
                    swap(r.w, r.h);
                    r.rotated = !r.rotated;
                }
            }

            Shelf ns;
            ns.board_idx = nb.id;
            ns.y = 0;
            ns.height = r.h;
            ns.used_width = r.w;
            ns.board_width = nb.width;
            
            r.board_idx = nb.id;
            r.x = 0;
            r.y = 0;
            
            nb.used_height = r.h;
            int b_idx = (int)boards.size();
            boards.push_back(nb);
            int s_idx = (int)shelves_data.size();
            shelves_data.push_back(ns);
            
            int new_rem_h = nb.height - nb.used_height;
            if (new_rem_h > 0) {
                available_boards.insert({new_rem_h, b_idx});
            }
            int new_rem_w = ns.board_width - ns.used_width;
            if (new_rem_w > 0) {
                open_shelves.insert({new_rem_w, s_idx});
            }
            placed = true;
        }
    }

    // Output the total number of boards used
    cout << boards.size() << "\n";
    // Output the placement details for each rectangle in original input order
    for (int i = 0; i < N; ++i) {
        cout << rects[i].board_idx << " " << rects[i].x << " " << rects[i].y << " " << rects[i].rotated << "\n";
    }

    return 0;
}
