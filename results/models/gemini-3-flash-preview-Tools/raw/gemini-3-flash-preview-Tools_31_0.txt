use std::io::{self, Read, Write};

// Simple Xorshift RNG
struct MyRng {
    state: u64,
}

impl MyRng {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }
    fn next_u64(&mut self) -> u64 {
        self.state ^= self.state << 13;
        self.state ^= self.state >> 7;
        self.state ^= self.state << 17;
        self.state
    }
}

struct Item {
    value: u64,
    indices: Vec<usize>,
}

fn main() {
    let mut input = Vec::new();
    if let Err(_) = io::stdin().read_to_end(&mut input) {
        return;
    }
    
    let mut offset = 0;
    let n_raw = read_u128(&input, &mut offset);
    if offset >= input.len() && n_raw == 0 { return; }
    let n = n_raw as usize;
    let target = read_u128(&input, &mut offset);
    
    let mut nums = Vec::with_capacity(n.min(100_000_000));
    let mut total_sum = 0u128;
    for i in 0..n {
        let v = read_u128(&input, &mut offset);
        nums.push((v, i));
        total_sum = total_sum.saturating_add(v);
    }
    
    if target == 0 {
        println!("YES");
        println!("[]");
        return;
    }
    
    if total_sum < target {
        println!("NO");
        return;
    }
    
    if total_sum == target {
        let indices: Vec<usize> = (0..n).collect();
        print_solution(&indices);
        return;
    }

    let solve_complement = target > total_sum / 2;
    let effective_target = if solve_complement { total_sum - target } else { target };

    if n <= 36 && effective_target < (1u128 << 60) {
        let mitm_nums: Vec<(u64, usize)> = nums.iter()
            .filter(|&&(v, _)| v <= effective_target)
            .map(|&(v, i)| (v as u64, i))
            .collect();
        if let Some(mut res) = meet_in_the_middle(&mitm_nums, effective_target as u64) {
            if solve_complement {
                res = get_complement(n, &res);
            }
            print_solution(&res);
            return;
        } else if n <= 36 {
            // For small n, if MITM fails, it's definitely NO
            println!("NO");
            return;
        }
    }

    // Bitset DP
    if effective_target <= 2_500_000_000 {
        let bitset_nums: Vec<(u64, usize)> = nums.iter()
            .filter(|&&(v, _)| v <= effective_target)
            .map(|&(v, i)| (v as u64, i))
            .collect();
        if let Some(mut res) = solve_bitset(&bitset_nums, effective_target as u64) {
            if solve_complement {
                res = get_complement(n, &res);
            }
            print_solution(&res);
            return;
        } else if effective_target < 500_000_000 {
            // For smaller targets, bitset is exact.
            println!("NO");
            return;
        }
    }

    // Randomized Search
    let result = std::sync::Arc::new(std::sync::Mutex::new(None));
    let mut threads = vec![];
    let num_threads = 12;
    let arc_nums = std::sync::Arc::new(nums);
    
    for i in 0..num_threads {
        let res_clone = result.clone();
        let nums_clone = arc_nums.clone();
        threads.push(std::thread::spawn(move || {
            if let Some(res) = randomized_search(&nums_clone, effective_target, 0xDEADBEEF + i as u64) {
                let mut lock = res_clone.lock().unwrap();
                if lock.is_none() {
                    *lock = Some(res);
                }
            }
        }));
    }
    
    for t in threads {
        let _ = t.join();
    }
    
    let final_res = result.lock().unwrap();
    if let Some(mut res) = final_res.clone() {
        if solve_complement {
            res = get_complement(n, &res);
        }
        print_solution(&res);
    } else {
        println!("NO");
    }
}

fn read_u128(input: &[u8], offset: &mut usize) -> u128 {
    while *offset < input.len() && !input[*offset].is_ascii_digit() {
        *offset += 1;
    }
    let mut res = 0u128;
    while *offset < input.len() && input[*offset].is_ascii_digit() {
        res = res.wrapping_mul(10).wrapping_add((input[*offset] - b'0') as u128);
        *offset += 1;
    }
    res
}

fn print_solution(indices: &[usize]) {
    let stdout = io::stdout();
    let mut writer = io::BufWriter::new(stdout.lock());
    let _ = writeln!(writer, "YES");
    let _ = write!(writer, "[");
    let mut sorted_indices = indices.to_vec();
    sorted_indices.sort_unstable();
    for (i, &idx) in sorted_indices.iter().enumerate() {
        if i > 0 { let _ = write!(writer, " "); }
        let _ = write!(writer, "{}", idx);
    }
    let _ = writeln!(writer, "]");
}

fn get_complement(n: usize, indices: &[usize]) -> Vec<usize> {
    let mut used = vec![false; n];
    for &idx in indices {
        if idx < n { used[idx] = true; }
    }
    let mut res = Vec::new();
    for i in 0..n {
        if !used[i] {
            res.push(i);
        }
    }
    res
}

fn meet_in_the_middle(nums: &[(u64, usize)], target: u64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;
    let left = &nums[..mid];
    let right = &nums[mid..];
    
    fn get_sums(items: &[(u64, usize)]) -> std::collections::HashMap<u64, Vec<usize>> {
        let mut sums = std::collections::HashMap::new();
        sums.insert(0, vec![]);
        for &(v, idx) in items {
            let mut next_sums = Vec::new();
            for (&s, indices) in &sums {
                let s: u64 = s;
                if let Some(new_s) = s.checked_add(v) {
                    let mut new_indices = indices.clone();
                    new_indices.push(idx);
                    next_sums.push((new_s, new_indices));
                }
            }
            for (s, indices) in next_sums {
                sums.entry(s).or_insert(indices);
            }
        }
        sums
    }
    
    let left_sums = get_sums(left);
    let right_sums = get_sums(right);
    
    for (s_left, indices_left) in left_sums {
        if s_left <= target {
            if let Some(indices_right) = right_sums.get(&(target - s_left)) {
                let mut res = indices_left;
                res.extend(indices_right.clone());
                return Some(res);
            }
        }
    }
    
    None
}

fn solve_bitset(nums: &[(u64, usize)], target: u64) -> Option<Vec<usize>> {
    let mut value_map = std::collections::BTreeMap::new();
    for &(v, idx) in nums {
        if v <= target {
            value_map.entry(v).or_insert_with(Vec::new).push(idx);
        }
    }
    
    let mut grouped_items = Vec::new();
    for (v, indices) in value_map {
        let mut count = indices.len();
        let mut k = 1;
        let mut offset = 0;
        while count >= k {
            if v.saturating_mul(k as u64) <= target {
                grouped_items.push(Item {
                    value: v * k as u64,
                    indices: indices[offset..offset + k].to_vec(),
                });
            }
            count -= k;
            offset += k;
            k *= 2;
        }
        if count > 0 {
            if v.saturating_mul(count as u64) <= target {
                grouped_items.push(Item {
                    value: v * count as u64,
                    indices: indices[offset..offset + count].to_vec(),
                });
            }
        }
    }
    
    reconstruct_bitset(&grouped_items, target as usize)
}

fn reconstruct_bitset(items: &[Item], target: usize) -> Option<Vec<usize>> {
    let mut bits = vec![0u64; (target >> 6) + 1];
    bits[0] = 1;
    
    let block_size = 80;
    let mut checkpoints = Vec::new();
    
    for (i, item) in items.iter().enumerate() {
        if i % block_size == 0 {
            checkpoints.push(bits.clone());
        }
        shift_or(&mut bits, item.value as usize);
    }
    
    if !get_bit(&bits, target) {
        return None;
    }
    
    let mut current_target = target;
    let mut result_indices = Vec::new();
    
    for (block_idx, block) in items.chunks(block_size).enumerate().rev() {
        let mut block_history = Vec::with_capacity(block.len());
        let mut temp_bits = checkpoints[block_idx].clone();
        for item in block {
            block_history.push(temp_bits.clone());
            shift_or(&mut temp_bits, item.value as usize);
        }
        
        for (i, item) in block.iter().enumerate().rev() {
            if current_target >= item.value as usize && !get_bit(&block_history[i], current_target) {
                current_target -= item.value as usize;
                result_indices.extend(&item.indices);
            }
        }
    }
    
    Some(result_indices)
}

fn shift_or(bits: &mut [u64], shift: usize) {
    let q = shift >> 6;
    let r = shift & 63;
    let len = bits.len();
    if q >= len { return; }
    if r == 0 {
        for i in (q..len).rev() {
            bits[i] |= bits[i - q];
        }
    } else {
        for i in (q + 1..len).rev() {
            bits[i] |= (bits[i - q] << r) | (bits[i - q - 1] >> (64 - r));
        }
        bits[q] |= bits[0] << r;
    }
}

fn get_bit(bits: &[u64], i: usize) -> bool {
    if i >= bits.len() * 64 { return false; }
    (bits[i >> 6] & (1 << (i & 63))) != 0
}

fn randomized_search(nums: &[(u128, usize)], target: u128, seed: u64) -> Option<Vec<usize>> {
    let mut rng = MyRng::new(seed);
    let n = nums.len();
    
    // For large N, the problem is usually easy to find a solution for
    // because of the high density.
    let search_nums = if n > 100_000 {
        // Take a spread of numbers to improve chances
        let mut spread = Vec::with_capacity(100_000);
        for i in 0..100_000 {
            spread.push(nums[(i as u64 * n as u64 / 100_000 as u64) as usize]);
        }
        spread
    } else {
        nums.to_vec()
    };
    
    let n_search = search_nums.len();
    let mut in_subset = vec![false; n_search];

    for _restart in 0..5 {
        let mut current_sum = 0u128;
        in_subset.fill(false);
        for i in 0..n_search {
            if current_sum + search_nums[i].0 <= target {
                if rng.next_u64() % 100 < 50 {
                    current_sum += search_nums[i].0;
                    in_subset[i] = true;
                }
            }
        }
        
        if current_sum == target {
            return Some(in_subset.iter().enumerate().filter(|&(_, &b)| b).map(|(i, _)| search_nums[i].1).collect());
        }

        for _iter in 0..1_000_000 {
            let i = (rng.next_u64() as usize) % n_search;
            if in_subset[i] {
                current_sum -= search_nums[i].0;
                in_subset[i] = false;
            } else {
                if current_sum + search_nums[i].0 <= target {
                    current_sum += search_nums[i].0;
                    in_subset[i] = true;
                } else {
                    let j = (rng.next_u64() as usize) % n_search;
                    if in_subset[j] {
                        let next_sum = current_sum - search_nums[j].0 + search_nums[i].0;
                        if next_sum <= target && (next_sum > current_sum || rng.next_u64() % 100 < 5) {
                            current_sum = next_sum;
                            in_subset[j] = false;
                            in_subset[i] = true;
                        }
                    }
                }
            }
            if current_sum == target {
                return Some(in_subset.iter().enumerate().filter(|&(_, &b)| b).map(|(i, _)| search_nums[i].1).collect());
            }
        }
    }
    None
}
