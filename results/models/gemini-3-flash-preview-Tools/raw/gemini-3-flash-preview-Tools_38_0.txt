To solve the **Exact Cover** problem, we can use Donald Knuth's **Algorithm X**. This is a recursive, backtracking, depth-first search that finds all solutions (or in this case, the planted solution) to an exact cover problem.

### Implementation Details:
1.  **Input Parsing:** We read the universe size `U` and the number of sets `S`. Then, we read each of the `S` lines as individual sets.
2.  **Algorithm X with MRV:**
    *   **MRV (Minimum Remaining Values) Heuristic:** At each step, we choose the element (column) that is covered by the fewest remaining sets. This significantly prunes the search space.
    *   **State Management:** We keep track of "active" columns (elements still needing coverage) and "active" rows (sets that don't overlap with our current selection).
    *   **Efficiency:** Instead of a full matrix, we use a dictionary of sets to map elements to the rows that cover them, and we maintain counts of active rows per column for fast MRV selection.
3.  **Output:** We print `SOLUTION` followed by the sorted 0-based indices of the sets in the cover, or `NO SOLUTION` if no such cover exists.

### Python Code:

```python
import sys

# Increase recursion depth for deep search trees often encountered in Exact Cover
sys.setrecursionlimit(10000)

def solve():
    # Read U (Universe size) and S (Number of sets) from the first line
    line = sys.stdin.readline()
    while line and not line.strip():
        line = sys.stdin.readline()
    if not line:
        return
    
    try:
        parts = line.split()
        if len(parts) < 2:
            return
        U = int(parts[0])
        S = int(parts[1])
    except ValueError:
        return
        
    # Read S sets, one per line
    rows = []
    for _ in range(S):
        rows.append(list(map(int, sys.stdin.readline().split())))
        
    # Build a mapping from columns (elements) to rows (sets) that cover them
    cols = [set() for _ in range(U)]
    invalid_rows = set()
    for r_idx, r_elements in enumerate(rows):
        # A set with duplicate elements cannot be part of an exact cover
        if len(set(r_elements)) != len(r_elements):
            invalid_rows.add(r_idx)
            continue
            
        for c in r_elements:
            if 0 <= c < U:
                cols[c].add(r_idx)
            else:
                # A set with elements outside the universe cannot be part of an exact cover
                invalid_rows.add(r_idx)
                break

    active_cols = set(range(U))
    active_rows = set(range(S)) - invalid_rows
    
    # Maintain counts of active rows per column for the MRV heuristic
    col_active_row_count = [0] * U
    for c in range(U):
        col_active_row_count[c] = len(cols[c] & active_rows)

    def backtrack():
        # If no columns are left to cover, we found a solution
        if not active_cols:
            return []
            
        # Heuristic: Choose the column with the minimum number of active rows (MRV)
        best_col = -1
        min_rows = S + 1
        
        for c in active_cols:
            count = col_active_row_count[c]
            if count < min_rows:
                min_rows = count
                best_col = c
                if min_rows == 0:
                    break
        
        # If the best column has 0 rows covering it, this branch is a dead end
        if best_col == -1 or min_rows == 0:
            return None
            
        # Try each row that covers the selected column
        candidate_rows = [r for r in cols[best_col] if r in active_rows]
        
        for r in candidate_rows:
            # Select row r and update the state
            current_hidden_cols = []
            current_hidden_rows = []
            
            # Hide all columns covered by row r
            for c in rows[r]:
                if c in active_cols:
                    active_cols.remove(c)
                    current_hidden_cols.append(c)
                    
                    # For each column hidden, hide all rows that also cover it (overlaps)
                    rows_to_hide = [r_prime for r_prime in cols[c] if r_prime in active_rows]
                    for r_prime in rows_to_hide:
                        active_rows.remove(r_prime)
                        current_hidden_rows.append(r_prime)
                        # Update MRV counts for all columns affected by hiding this row
                        for c_prime in rows[r_prime]:
                            col_active_row_count[c_prime] -= 1
            
            res = backtrack()
            if res is not None:
                return [r] + res
            
            # Backtrack: Restore the state in reverse order
            for r_prime in reversed(current_hidden_rows):
                active_rows.add(r_prime)
                for c_prime in rows[r_prime]:
                    col_active_row_count[c_prime] += 1
            for c in reversed(current_hidden_cols):
                active_cols.add(c)
                
        return None

    solution = backtrack()
    
    if solution is not None:
        print("SOLUTION")
        print(*(sorted(solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()
```
