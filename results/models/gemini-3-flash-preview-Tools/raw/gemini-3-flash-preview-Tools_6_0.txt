The provided C# solution addresses the Orbital Travelling Salesman Problem (OTSP) by combining orbital mechanics approximations with scalable optimization algorithms.

### Key Features:
- **Orbital Mechanics Cost Estimation**: The solver estimates the delta-V required to transfer between orbits using a bi-impulsive approximation. It accounts for:
  - **Plane Change Cost**: Calculated as $2v_{circ} \sin(\theta/2)$, where $\theta$ is the angle between the angular momentum vectors of the two orbits.
  - **Shape Change Cost**: Approximated by the difference in orbital speeds, providing a reasonable estimate for energy adjustment.
- **Scalable TSP Solver**:
  - **Small Scales ($N \le 11$)**: Employs a brute-force permutation search to find the absolute optimal visitation order.
  - **Large to Extreme Scales ($100$ stations)**: Uses a multi-start heuristic combining **Nearest Neighbor** initialization with **2-opt** local search. It also includes random restarts to escape local optima within the 5-minute time limit.
- **Performance**: Optimized for speed, the solver can handle 100 stations comfortably, providing a high-quality visitation order well within the allocated time.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

namespace OrbitalTSP
{
    public struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator *(Vector3 a, double b) => new Vector3(a.X * b, a.Y * b, a.Z * b);
        public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X
        );
        public double Magnitude() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double LengthSquared() => X * X + Y * Y + Z * Z;
    }

    public class Orbit
    {
        public Vector3 R, V;
        public double A; // semi-major axis
        public Vector3 H; // angular momentum
        public double Mu;

        public Orbit(Vector3 r, Vector3 v, double mu)
        {
            R = r;
            V = v;
            Mu = mu;
            H = Vector3.Cross(r, v);
            double rMag = r.Magnitude();
            double vSq = v.LengthSquared();
            double energy = vSq / 2.0 - mu / rMag;
            if (Math.Abs(energy) < 1e-12) A = 1e12;
            else A = -mu / (2.0 * energy);
        }
    }

    public class Program
    {
        static double mu;

        static double EstimateCost(Orbit a, Orbit b)
        {
            double h1m = a.H.Magnitude();
            double h2m = b.H.Magnitude();
            if (h1m < 1e-9 || h2m < 1e-9) return 1e9;
            double cosTheta = Vector3.Dot(a.H, b.H) / (h1m * h2m);
            cosTheta = Math.Clamp(cosTheta, -1.0, 1.0);
            double theta = Math.Acos(cosTheta);
            double a1 = a.A;
            double a2 = b.A;
            double a_avg = (Math.Abs(a1) + Math.Abs(a2)) / 2.0;
            double v_circ = Math.Sqrt(mu / a_avg);
            double dV_plane = 2.0 * v_circ * Math.Sin(theta / 2.0);
            double v1 = Math.Sqrt(mu / Math.Abs(a1));
            double v2 = Math.Sqrt(mu / Math.Abs(a2));
            double dV_shape = Math.Abs(v1 - v2);
            return dV_plane + dV_shape;
        }

        public static void Main(string[] args)
        {
            string? line1 = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line1)) return;
            string[] parts1 = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts1.Length < 2) return;
            int N = int.Parse(parts1[0]);
            mu = double.Parse(parts1[1]);

            Orbit? startOrbit = ReadOrbit();
            if (startOrbit == null) return;

            Orbit[] stations = new Orbit[N];
            for (int i = 0; i < N; i++)
            {
                stations[i] = ReadOrbit()!;
            }

            double[,] matrix = new double[N, N];
            for (int i = 0; i < N; i++)
            {
                for (int j = 0; j < N; j++)
                {
                    if (i == j) matrix[i, j] = 0;
                    else matrix[i, j] = EstimateCost(stations[i], stations[j]); 
                }
            }

            double[] startCosts = new double[N];
            for (int i = 0; i < N; i++)
            {
                startCosts[i] = EstimateCost(startOrbit, stations[i]);
            }

            int[] bestPath = SolveTSP(N, matrix, startCosts);
            if (bestPath != null)
            {
                Console.WriteLine(string.Join(" ", bestPath));
            }
        }

        static Orbit? ReadOrbit()
        {
            string? line = "";
            while (string.IsNullOrWhiteSpace(line))
            {
                line = Console.ReadLine();
                if (line == null) return null;
            }
            string[] p = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (p.Length < 6) return null;
            return new Orbit(
                new Vector3(double.Parse(p[0]), double.Parse(p[1]), double.Parse(p[2])),
                new Vector3(double.Parse(p[3]), double.Parse(p[4]), double.Parse(p[5])),
                mu
            );
        }

        static int[] SolveTSP(int N, double[,] matrix, double[] startCosts)     
        {
            if (N <= 11) return BruteForce(N, matrix, startCosts);
            else return HeuristicTSP(N, matrix, startCosts);
        }

        static int[] BruteForce(int N, double[,] matrix, double[] startCosts)   
        {
            int[] p = Enumerable.Range(0, N).ToArray();
            int[] best = (int[])p.Clone();
            double minCost = double.MaxValue;
            void Permute(int k)
            {
                if (k == N)
                {
                    double currentCost = startCosts[p[0]];
                    for (int i = 0; i < N - 1; i++) currentCost += matrix[p[i], p[i + 1]];
                    if (currentCost < minCost)
                    {
                        minCost = currentCost;
                        Array.Copy(p, best, N);
                    }
                    return;
                }
                for (int i = k; i < N; i++)
                {
                    int temp = p[k]; p[k] = p[i]; p[i] = temp;
                    Permute(k + 1);
                    temp = p[k]; p[k] = p[i]; p[i] = temp;
                }
            }
            Permute(0);
            return best;
        }

        static int[] HeuristicTSP(int N, double[,] matrix, double[] startCosts) 
        {
            int[]? bestOverallPath = null;
            double minOverallCost = double.MaxValue;
            for (int startNode = 0; startNode < N; startNode++)
            {
                int[] path = RunNNAnd2Opt(N, matrix, startCosts, startNode, out double cost);
                if (cost < minOverallCost)
                {
                    minOverallCost = cost;
                    bestOverallPath = (int[])path.Clone();
                }
            }
            Random rng = new Random(42);
            DateTime endTime = DateTime.Now.AddSeconds(15);
            while (DateTime.Now < endTime)
            {
                int[] path = Enumerable.Range(0, N).ToArray();
                for (int i = 0; i < N; i++)
                {
                    int j = rng.Next(i, N);
                    int temp = path[i]; path[i] = path[j]; path[j] = temp;      
                }
                path = Run2Opt(N, matrix, startCosts, path, out double cost);   
                if (cost < minOverallCost)
                {
                    minOverallCost = cost;
                    bestOverallPath = (int[])path.Clone();
                }
            }
            return bestOverallPath!;
        }

        static int[] RunNNAnd2Opt(int N, double[,] matrix, double[] startCosts, int startNode, out double finalCost)
        {
            int[] path = new int[N];
            bool[] visited = new bool[N];
            path[0] = startNode;
            visited[startNode] = true;
            for (int i = 1; i < N; i++)
            {
                int last = path[i - 1];
                int bestNext = -1;
                double bestNextCost = double.MaxValue;
                for (int j = 0; j < N; j++)
                {
                    if (!visited[j] && matrix[last, j] < bestNextCost)
                    {
                        bestNextCost = matrix[last, j];
                        bestNext = j;
                    }
                }
                path[i] = bestNext;
                visited[bestNext] = true;
            }
            return Run2Opt(N, matrix, startCosts, path, out finalCost);
        }

        static int[] Run2Opt(int N, double[,] matrix, double[] startCosts, int[] path, out double finalCost)
        {
            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < N - 1; i++)
                {
                    for (int j = i + 1; j < N; j++)
                    {
                        double oldEdge1 = (i == 0) ? startCosts[path[0]] : matrix[path[i - 1], path[i]];
                        double oldEdge2 = (j < N - 1) ? matrix[path[j], path[j + 1]] : 0;
                        double newEdge1 = (i == 0) ? startCosts[path[j]] : matrix[path[i - 1], path[j]];
                        double newEdge2 = (j < N - 1) ? matrix[path[i], path[j + 1]] : 0;
                        double delta = (newEdge1 + newEdge2) - (oldEdge1 + oldEdge2);
                        if (delta < -1e-9)
                        {
                            Array.Reverse(path, i, j - i + 1);
                            improved = true;
                        }
                    }
                }
            }
            double total = startCosts[path[0]];
            for (int i = 0; i < N - 1; i++) total += matrix[path[i], path[i + 1]];
            finalCost = total;
            return path;
        }
    }
}
```
