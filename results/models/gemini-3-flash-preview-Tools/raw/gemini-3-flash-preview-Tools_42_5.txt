; SPIR-V
; Version: 1.0
; Generator: Khronos; 0
; Bound: 100
; Schema: 0
               OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputVertices 7

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_2 = OpConstant %uint 2
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
    %float_0 = OpConstant %float 0.0
%float_0_015 = OpConstant %float 0.015
 %float_0_05 = OpConstant %float 0.05
 %float_0_75 = OpConstant %float 0.75

; Input variables
%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

 %arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input

; Output variables
%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

; Pointers for access
  %ptr_in_v4 = OpTypePointer Input %v4float
  %ptr_in_v3 = OpTypePointer Input %v3float
 %ptr_out_v4 = OpTypePointer Output %v4float

; Main Function
       %main = OpFunction %void None %3
         %20 = OpLabel

; Load p0, p1
     %ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %ptr_p0
     %ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %ptr_p1

; Load colors c0, c1
     %ptr_c0 = OpAccessChain %ptr_in_v3 %color %int_0
         %c0 = OpLoad %v3float %ptr_c0
     %ptr_c1 = OpAccessChain %ptr_in_v3 %color %int_1
         %c1 = OpLoad %v3float %ptr_c1

; dir.xy = p1.xy - p0.xy
      %p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
      %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
        %dir = OpFSub %v2float %p1_xy %p0_xy

; perp = normalize(-dir.y, dir.x)
      %dir_x = OpCompositeExtract %float %dir 0
      %dir_y = OpCompositeExtract %float %dir 1
  %neg_dir_y = OpFNegate %float %dir_y
   %raw_perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x
       %perp = OpExtInst %v2float %1 Normalize %raw_perp

; offsets
  %off_shaft = OpVectorTimesScalar %v2float %perp %float_0_015
  %off_arrow = OpVectorTimesScalar %v2float %perp %float_0_05

       %ox_s = OpCompositeExtract %float %off_shaft 0
       %oy_s = OpCompositeExtract %float %off_shaft 1
%v_off_shaft = OpCompositeConstruct %v4float %ox_s %oy_s %float_0 %float_0

       %ox_a = OpCompositeExtract %float %off_arrow 0
       %oy_a = OpCompositeExtract %float %off_arrow 1
%v_off_arrow = OpCompositeConstruct %v4float %ox_a %oy_a %float_0 %float_0

; mid = p0 + 0.75 * (p1 - p0)
%p1_minus_p0 = OpFSub %v4float %p1 %p0
%scaled_diff = OpVectorTimesScalar %v4float %p1_minus_p0 %float_0_75
        %mid = OpFAdd %v4float %p0 %scaled_diff

; Output position pointer
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; --- Emit Shaft (4 vertices) ---
               OpStore %outColor %c0

      %v0_pos = OpFAdd %v4float %p0 %v_off_shaft
               OpStore %ptr_out_pos %v0_pos
               OpEmitVertex

      %v1_pos = OpFSub %v4float %p0 %v_off_shaft
               OpStore %ptr_out_pos %v1_pos
               OpEmitVertex

      %v2_pos = OpFAdd %v4float %mid %v_off_shaft
               OpStore %ptr_out_pos %v2_pos
               OpEmitVertex

      %v3_pos = OpFSub %v4float %mid %v_off_shaft
               OpStore %ptr_out_pos %v3_pos
               OpEmitVertex

               OpEndPrimitive

; --- Emit Arrowhead (3 vertices) ---
               OpStore %outColor %c1

      %v4_pos = OpFAdd %v4float %mid %v_off_arrow
               OpStore %ptr_out_pos %v4_pos
               OpEmitVertex

      %v5_pos = OpFSub %v4float %mid %v_off_arrow
               OpStore %ptr_out_pos %v5_pos
               OpEmitVertex

               OpStore %ptr_out_pos %p1
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
