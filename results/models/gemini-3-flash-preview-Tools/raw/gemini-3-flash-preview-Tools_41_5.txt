; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
               OpExecutionMode %main OriginUpperLeft

; Decorations
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %outColor Location 0
               OpDecorate %struct_UBO Block
               OpMemberDecorate %struct_UBO 0 Offset 0
               OpMemberDecorate %struct_UBO 0 ColMajor
               OpMemberDecorate %struct_UBO 0 MatrixStride 16
               OpMemberDecorate %struct_UBO 1 Offset 64
               OpMemberDecorate %struct_UBO 1 ColMajor
               OpMemberDecorate %struct_UBO 1 MatrixStride 16
               OpMemberDecorate %struct_UBO 2 Offset 128
               OpMemberDecorate %struct_UBO 2 ColMajor
               OpMemberDecorate %struct_UBO 2 MatrixStride 16
               OpMemberDecorate %struct_UBO 3 Offset 192
               OpMemberDecorate %struct_UBO 4 Offset 208
               OpMemberDecorate %struct_UBO 5 Offset 224
               OpDecorate %ubo_var DescriptorSet 0
               OpDecorate %ubo_var Binding 0

; Types and Constants
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
 %struct_UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_struct_UBO = OpTypePointer Uniform %struct_UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

       %void = OpTypeVoid
   %void_fnt = OpTypeFunction %void

    %float_0 = OpConstant %float 0.0
    %float_1 = OpConstant %float 1.0
  %float_0_2 = OpConstant %float 0.2
  %float_0_5 = OpConstant %float 0.5
  %float_0_8 = OpConstant %float 0.8
   %float_32 = OpConstant %float 32.0

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4

; Global Variables
%in_worldPos = OpVariable %ptr_Input_v3float Input
  %in_normal = OpVariable %ptr_Input_v3float Input
   %outColor = OpVariable %ptr_Output_v4float Output
    %ubo_var = OpVariable %ptr_Uniform_struct_UBO Uniform

; Function main
       %main = OpFunction %void None %void_fnt
 %main_label = OpLabel

; Load Input data
   %worldPos = OpLoad %v3float %in_worldPos
  %normal_in = OpLoad %v3float %in_normal

; Load lightPos and cameraPos from UBO
 %ptr_lightPos = OpAccessChain %ptr_Uniform_v4float %ubo_var %int_3
 %lightPos_v4  = OpLoad %v4float %ptr_lightPos
 %lightPos     = OpVectorShuffle %v3float %lightPos_v4 %lightPos_v4 0 1 2

 %ptr_cameraPos = OpAccessChain %ptr_Uniform_v4float %ubo_var %int_4
 %cameraPos_v4  = OpLoad %v4float %ptr_cameraPos
 %cameraPos     = OpVectorShuffle %v3float %cameraPos_v4 %cameraPos_v4 0 1 2

; Normalize interpolated normal
          %N = OpExtInst %v3float %1 Normalize %normal_in

; L = normalize(lightPos - worldPos)
      %L_un  = OpFSub %v3float %lightPos %worldPos
          %L = OpExtInst %v3float %1 Normalize %L_un

; V = normalize(cameraPos - worldPos)
      %V_un  = OpFSub %v3float %cameraPos %worldPos
          %V = OpExtInst %v3float %1 Normalize %V_un

; R = reflect(-L, N)
    %neg_L   = OpFNegate %v3float %L
          %R = OpExtInst %v3float %1 Reflect %neg_L %N

; Diffuse: max(dot(N, L), 0.0)
      %dotNL = OpDot %float %N %L
    %diffuse = OpExtInst %float %1 FMax %dotNL %float_0

; Specular: pow(max(dot(R, V), 0.0), 32.0)
      %dotRV = OpDot %float %R %V
    %maxRV   = OpExtInst %float %1 FMax %dotRV %float_0
   %specular = OpExtInst %float %1 Pow %maxRV %float_32

; intensity = 0.2 + 0.5 * diffuse + 0.8 * specular
    %term1   = OpFMul %float %float_0_5 %diffuse
    %term2   = OpFMul %float %float_0_8 %specular
    %sum1    = OpFAdd %float %float_0_2 %term1
  %intensity = OpFAdd %float %sum1 %term2

; Final color: vec4(intensity, intensity, intensity, 1.0)
   %color_v4 = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1
               OpStore %outColor %color_v4

               OpReturn
               OpFunctionEnd
