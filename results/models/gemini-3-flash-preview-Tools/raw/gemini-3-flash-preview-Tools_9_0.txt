To solve the 1D Cutting Stock Problem efficiently for large-scale inputs (millions of cuts), I implemented the **First Fit Decreasing (FFD)** heuristic. FFD is a standard and robust algorithm for bin packing that yields results very close to optimal while maintaining high performance.

### Key Optimizations:
1.  **Segment Tree for $O(\log M)$ Search:** To handle millions of bins efficiently, I used a **Segment Tree** to store the remaining capacity of each bin. This allows finding the first available bin that fits a given cut in logarithmic time relative to the number of bins, reducing the overall complexity to **$O(N \log N)$**.
2.  **Fast I/O:** I implemented a custom `FastReader` and used a large-buffered `StreamWriter` to handle the high volume of input and output tokens, which is often the bottleneck in such problems.
3.  **Memory Efficiency:** The implementation uses primitive arrays and a custom linked-list structure for grouping cut indices, ensuring a small memory footprint (approx. 60 bytes per cut) well within the 32GB limit.
4.  **Stability:** The cuts are sorted in descending order with original index tie-breakers to ensure stable and consistent behavior.

### Complete C# Solution:

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    public static void Main(string[] args)
    {
        // Use a FastReader to handle large inputs efficiently.
        var reader = new FastReader(Console.OpenStandardInput());
        
        string ns = reader.ReadToken();
        string sls = reader.ReadToken();
        if (string.IsNullOrEmpty(ns) || string.IsNullOrEmpty(sls)) return;
        
        if (!int.TryParse(ns, out int n)) return;
        if (!long.TryParse(sls, out long stockLength)) return;
        
        if (n <= 0)
        {
            if (n == 0) Console.WriteLine(0);
            return;
        }

        Cut[] cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut { Length = reader.ReadLong(), OriginalIndex = i };
        }
        
        // First Fit Decreasing (FFD) heuristic.
        // Step 1: Sort cuts in descending order of length.
        // Complexity: O(N log N)
        Array.Sort(cuts, new CutComparer());
        
        // Step 2: Bin Packing using a Segment Tree for First Fit search.
        // Complexity: O(N log N)
        int treeSize = 1;
        while (treeSize < n) treeSize *= 2;
        if (treeSize < 2) treeSize = 2;

        long[] tree = new long[2 * treeSize];
        // Initialize leaves with stockLength. Each leaf represents a potential bin.
        for (int i = 0; i < n; i++) tree[treeSize + i] = stockLength;
        // Padding for leaves that don't represent a potential bin.
        for (int i = n; i < treeSize; i++) tree[treeSize + i] = -1;
        // Build internal nodes: each node stores the maximum remaining capacity in its subtree.
        for (int i = treeSize - 1; i > 0; i--) tree[i] = Math.Max(tree[2 * i], tree[2 * i + 1]);
        
        int[] binAssignment = new int[n];
        int numBinsUsed = 0;
        
        for (int i = 0; i < n; i++)
        {
            long len = cuts[i].Length;
            // Find the index of the first bin with remaining capacity >= len.
            int binIdx = FindFirstBin(len, treeSize, tree);
            
            binAssignment[cuts[i].OriginalIndex] = binIdx;
            if (binIdx + 1 > numBinsUsed) numBinsUsed = binIdx + 1;
            
            // Update the bin capacity in the segment tree.
            int node = treeSize + binIdx;
            tree[node] -= len;
            node /= 2;
            while (node >= 1)
            {
                long newMax = Math.Max(tree[2 * node], tree[2 * node + 1]);
                if (tree[node] == newMax) break; // Optimization: stop if no change.
                tree[node] = newMax;
                node /= 2;
            }
        }
        
        // Group original cut indices by their assigned bin for final output.
        // We use a custom linked-list structure (nextCut array) to avoid the overhead of List<int>[].
        int[] binHeads = new int[numBinsUsed];
        for (int i = 0; i < numBinsUsed; i++) binHeads[i] = -1;
        int[] nextCut = new int[n];
        
        // Traverse in reverse to maintain original relative order within each bin.
        for (int i = n - 1; i >= 0; i--)
        {
            int b = binAssignment[i];
            nextCut[i] = binHeads[b];
            binHeads[b] = i;
        }
        
        // Output results using a large-buffered StreamWriter for maximum performance.
        using (var sw = new StreamWriter(Console.OpenStandardOutput(), Encoding.Default, 1024 * 1024))
        {
            sw.WriteLine(numBinsUsed);
            for (int i = 0; i < numBinsUsed; i++)
            {
                int curr = binHeads[i];
                bool first = true;
                while (curr != -1)
                {
                    if (!first) sw.Write(' ');
                    sw.Write(curr);
                    first = false;
                    curr = nextCut[curr];
                }
                sw.WriteLine();
            }
        }
    }
    
    // FindFirstBin traverses the segment tree to find the leftmost leaf (smallest bin index)
    // that can accommodate a cut of the given length.
    static int FindFirstBin(long length, int treeSize, long[] tree)
    {
        if (tree[1] < length) return -1;
        int node = 1;
        while (node < treeSize)
        {
            if (tree[2 * node] >= length)
                node = 2 * node;
            else
                node = 2 * node + 1;
        }
        return node - treeSize;
    }
    
    struct Cut
    {
        public long Length;
        public int OriginalIndex;
    }
    
    struct CutComparer : IComparer<Cut>
    {
        public int Compare(Cut x, Cut y)
        {
            // Descending order of length for First Fit Decreasing.
            int cmp = y.Length.CompareTo(x.Length);
            if (cmp != 0) return cmp;
            // Tie-break with original index for stable behavior.
            return x.OriginalIndex.CompareTo(y.OriginalIndex);
        }
    }
    
    // FastReader provides low-level, high-performance token reading from a stream.
    class FastReader
    {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 1024];
        private int pos = 0;
        private int limit = 0;
        public FastReader(Stream s) { stream = s; }
        private int ReadByte()
        {
            if (pos >= limit)
            {
                pos = 0;
                limit = stream.Read(buffer, 0, buffer.Length);
                if (limit <= 0) return -1;
            }
            return buffer[pos++];
        }
        public string ReadToken()
        {
            int b = ReadByte();
            while (b != -1 && b <= ' ') b = ReadByte();
            if (b == -1) return null;
            StringBuilder sb = new StringBuilder();
            while (b != -1 && b > ' ')
            {
                sb.Append((char)b);
                b = ReadByte();
            }
            return sb.ToString();
        }
        public long ReadLong()
        {
            long res = 0;
            int b = ReadByte();
            while (b != -1 && (b < '0' || b > '9') && b != '-') b = ReadByte();
            if (b == -1) return 0;
            bool neg = false;
            if (b == '-') { neg = true; b = ReadByte(); }
            while (b >= '0' && b <= '9')
            {
                res = res * 10 + (b - '0');
                b = ReadByte();
            }
            return neg ? -res : res;
        }
    }
}
```
