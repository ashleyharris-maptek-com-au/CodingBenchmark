OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_333333333 = OpConstant %float 0.333333333
%float_0_666666667 = OpConstant %float 0.666666667
%float_0_01 = OpConstant %float 0.01

%v2_0_333 = OpCompositeConstruct %v2float %float_0_333333333 %float_0_333333333
%v3_0_333 = OpCompositeConstruct %v3float %float_0_333333333 %float_0_333333333 %float_0_333333333
%v4_0_333 = OpCompositeConstruct %v4float %float_0_333333333 %float_0_333333333 %float_0_333333333 %float_0_333333333

%v2_0_666 = OpCompositeConstruct %v2float %float_0_666666667 %float_0_666666667
%v3_0_666 = OpCompositeConstruct %v3float %float_0_666666667 %float_0_666666667 %float_0_666666667
%v4_0_666 = OpCompositeConstruct %v4float %float_0_666666667 %float_0_666666667 %float_0_666666667 %float_0_666666667

; Input block
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Other inputs
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; Output block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Other outputs
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Function type
%main_fn = OpTypeFunction %void

; Pointers for access
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; Main function
%main = OpFunction %void None %main_fn
%entry = OpLabel

; Load positions
%ptr_P0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%ptr_P1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%ptr_P2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%ptr_P3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%P0 = OpLoad %v4float %ptr_P0
%P1 = OpLoad %v4float %ptr_P1
%P2 = OpLoad %v4float %ptr_P2
%P3 = OpLoad %v4float %ptr_P3

; Load colors
%ptr_C1 = OpAccessChain %ptr_in_v3 %color %int_1
%ptr_C2 = OpAccessChain %ptr_in_v3 %color %int_2
%C1 = OpLoad %v3float %ptr_C1
%C2 = OpLoad %v3float %ptr_C2

; Compute N1 (adjacency tangent at P1: P2 - P0)
%D1 = OpFSub %v4float %P2 %P0
%D1_xy = OpVectorShuffle %v2float %D1 %D1 0 1
%T1 = OpExtInst %v2float %glsl450 Normalize %D1_xy
%T1_x = OpCompositeExtract %float %T1 0
%T1_y = OpCompositeExtract %float %T1 1
%neg_T1_y = OpFNegate %float %T1_y
%N1_raw = OpCompositeConstruct %v2float %neg_T1_y %T1_x
%N1 = OpVectorTimesScalar %v2float %N1_raw %float_0_01

; Compute N2 (adjacency tangent at P2: P3 - P1)
%D2 = OpFSub %v4float %P3 %P1
%D2_xy = OpVectorShuffle %v2float %D2 %D2 0 1
%T2 = OpExtInst %v2float %glsl450 Normalize %D2_xy
%T2_x = OpCompositeExtract %float %T2 0
%T2_y = OpCompositeExtract %float %T2 1
%neg_T2_y = OpFNegate %float %T2_y
%N2_raw = OpCompositeConstruct %v2float %neg_T2_y %T2_x
%N2 = OpVectorTimesScalar %v2float %N2_raw %float_0_01

%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; --- Quad 1 (t=0 to t=1/3) ---

; t = 0
%P1_xy = OpVectorShuffle %v2float %P1 %P1 0 1
%P1_zw = OpVectorShuffle %v2float %P1 %P1 2 3

; Vertex 1: P1 + N1
%V1_xy = OpFAdd %v2float %P1_xy %N1
%V1_pos = OpVectorShuffle %v4float %V1_xy %P1_zw 0 1 2 3
OpStore %outColor %C1
OpStore %ptr_out_pos %V1_pos
OpEmitVertex

; Vertex 2: P1 - N1
%V2_xy = OpFSub %v2float %P1_xy %N1
%V2_pos = OpVectorShuffle %v4float %V2_xy %P1_zw 0 1 2 3
OpStore %outColor %C1
OpStore %ptr_out_pos %V2_pos
OpEmitVertex

; t = 1/3
%Pa = OpExtInst %v4float %glsl450 FMix %P1 %P2 %v4_0_333
%Na = OpExtInst %v2float %glsl450 FMix %N1 %N2 %v2_0_333
%Ca = OpExtInst %v3float %glsl450 FMix %C1 %C2 %v3_0_333

%Pa_xy = OpVectorShuffle %v2float %Pa %Pa 0 1
%Pa_zw = OpVectorShuffle %v2float %Pa %Pa 2 3

; Vertex 3: Pa + Na
%V3_xy = OpFAdd %v2float %Pa_xy %Na
%V3_pos = OpVectorShuffle %v4float %V3_xy %Pa_zw 0 1 2 3
OpStore %outColor %Ca
OpStore %ptr_out_pos %V3_pos
OpEmitVertex

; Vertex 4: Pa - Na
%V4_xy = OpFSub %v2float %Pa_xy %Na
%V4_pos = OpVectorShuffle %v4float %V4_xy %Pa_zw 0 1 2 3
OpStore %outColor %Ca
OpStore %ptr_out_pos %V4_pos
OpEmitVertex

OpEndPrimitive

; --- Quad 2 (t=2/3 to t=1) ---

; t = 2/3
%Pb = OpExtInst %v4float %glsl450 FMix %P1 %P2 %v4_0_666
%Nb = OpExtInst %v2float %glsl450 FMix %N1 %N2 %v2_0_666
%Cb = OpExtInst %v3float %glsl450 FMix %C1 %C2 %v3_0_666

%Pb_xy = OpVectorShuffle %v2float %Pb %Pb 0 1
%Pb_zw = OpVectorShuffle %v2float %Pb %Pb 2 3

; Vertex 5: Pb + Nb
%V5_xy = OpFAdd %v2float %Pb_xy %Nb
%V5_pos = OpVectorShuffle %v4float %V5_xy %Pb_zw 0 1 2 3
OpStore %outColor %Cb
OpStore %ptr_out_pos %V5_pos
OpEmitVertex

; Vertex 6: Pb - Nb
%V6_xy = OpFSub %v2float %Pb_xy %Nb
%V6_pos = OpVectorShuffle %v4float %V6_xy %Pb_zw 0 1 2 3
OpStore %outColor %Cb
OpStore %ptr_out_pos %V6_pos
OpEmitVertex

; t = 1
%P2_xy = OpVectorShuffle %v2float %P2 %P2 0 1
%P2_zw = OpVectorShuffle %v2float %P2 %P2 2 3

; Vertex 7: P2 + N2
%V7_xy = OpFAdd %v2float %P2_xy %N2
%V7_pos = OpVectorShuffle %v4float %V7_xy %P2_zw 0 1 2 3
OpStore %outColor %C2
OpStore %ptr_out_pos %V7_pos
OpEmitVertex

; Vertex 8: P2 - N2
%V8_xy = OpFSub %v2float %P2_xy %N2
%V8_pos = OpVectorShuffle %v4float %V8_xy %P2_zw 0 1 2 3
OpStore %outColor %C2
OpStore %ptr_out_pos %V8_pos
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
