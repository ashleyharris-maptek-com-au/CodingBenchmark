use std::io::{self, Read, Write, BufReader, BufWriter, BufRead};
use std::sync::Arc;
use std::thread;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    #[inline(always)]
    fn dist_sq(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

struct KdNode {
    point: Point,
    idx: usize,
    left: Option<Box<KdNode>>,
    right: Option<Box<KdNode>>,
    axis: u8,
}

fn build_kdtree(points: &mut [(Point, usize)], depth: usize) -> Option<Box<KdNode>> {
    if points.is_empty() {
        return None;
    }
    let axis = (depth % 3) as u8;
    points.sort_by(|a, b| {
        let va = if axis == 0 { a.0.x } else if axis == 1 { a.0.y } else { a.0.z };
        let vb = if axis == 0 { b.0.x } else if axis == 1 { b.0.y } else { b.0.z };
        va.partial_cmp(&vb).unwrap_or(std::cmp::Ordering::Equal)
    });
    let mid = points.len() / 2;
    let (left_pts, mid_and_right) = points.split_at_mut(mid);
    let (mid_pt, right_pts) = mid_and_right.split_at_mut(1);
    
    Some(Box::new(KdNode {
        point: mid_pt[0].0,
        idx: mid_pt[0].1,
        left: build_kdtree(left_pts, depth + 1),
        right: build_kdtree(right_pts, depth + 1),
        axis,
    }))
}

fn find_nn_ref(node: &KdNode, p: &Point, best_idx: &mut usize, best_dist_sq: &mut f32) {
    let d2 = p.dist_sq(&node.point);
    if d2 < *best_dist_sq {
        *best_dist_sq = d2;
        *best_idx = node.idx;
    }
    
    let val_p = if node.axis == 0 { p.x } else if node.axis == 1 { p.y } else { p.z };
    let val_n = if node.axis == 0 { node.point.x } else if node.axis == 1 { node.point.y } else { node.point.z };
    
    let (near, far) = if val_p < val_n { (&node.left, &node.right) } else { (&node.right, &node.left) };
    
    if let Some(n) = near {
        find_nn_ref(n, p, best_idx, best_dist_sq);
    }
    
    let diff = val_p - val_n;
    if diff * diff < *best_dist_sq {
        if let Some(f) = far {
            find_nn_ref(f, p, best_idx, best_dist_sq);
        }
    }
}

struct Solver {
    centroids: Vec<Point>,
    norms_sq: Vec<f32>,
    kdtree: Option<Box<KdNode>>,
}

impl Solver {
    fn new(centroids: Vec<Point>) -> Self {
        let norms_sq = centroids.iter().map(|c| c.x * c.x + c.y * c.y + c.z * c.z).collect();
        let mut kdtree = None;
        if centroids.len() > 128 {
            let mut pts_with_idx: Vec<(Point, usize)> = centroids.iter().cloned().enumerate().map(|(i, p)| (p, i)).collect();
            kdtree = build_kdtree(&mut pts_with_idx, 0);
        }
        Solver { centroids, norms_sq, kdtree }
    }

    #[inline(always)]
    fn find_nearest(&self, p: &Point) -> usize {
        if let Some(ref root) = self.kdtree {
            let mut best_idx = 0;
            let mut best_dist_sq = f32::MAX;
            find_nn_ref(root, p, &mut best_idx, &mut best_dist_sq);
            best_idx
        } else {
            let mut min_score = f32::MAX;
            let mut best_idx = 0;
            for (i, (c, &n2)) in self.centroids.iter().zip(&self.norms_sq).enumerate() {
                let score = n2 - 2.0 * (p.x * c.x + p.y * c.y + p.z * c.z);
                if score < min_score {
                    min_score = score;
                    best_idx = i;
                }
            }
            best_idx
        }
    }
}

fn run_kmeans(points: &[Point], k: usize) -> Vec<Point> {
    if k >= points.len() {
        let mut res = points.to_vec();
        while res.len() < k {
            res.push(Point { x: 0.0, y: 0.0, z: 0.0 });
        }
        return res;
    }

    let mut centroids = Vec::with_capacity(k);
    let mut rng_state = 42u64;
    let mut next_u64 = || {
        rng_state = rng_state.wrapping_mul(6364136223846793005).wrapping_add(1);
        rng_state
    };

    centroids.push(points[next_u64() as usize % points.len()]);
    let mut min_dists_sq = vec![f32::MAX; points.len()];
    
    // K-means++ initialization
    for _ in 1..k {
        let last_c = centroids.last().unwrap();
        let mut sum_dist_sq = 0.0f64;
        for (i, p) in points.iter().enumerate() {
            let d2 = p.dist_sq(last_c);
            if d2 < min_dists_sq[i] {
                min_dists_sq[i] = d2;
            }
            sum_dist_sq += min_dists_sq[i] as f64;
        }

        let r = (next_u64() as f64 / u64::MAX as f64) * sum_dist_sq;
        let mut acc = 0.0f64;
        let mut chosen_idx = points.len() - 1;
        for (i, &d2) in min_dists_sq.iter().enumerate() {
            acc += d2 as f64;
            if acc >= r {
                chosen_idx = i;
                break;
            }
        }
        centroids.push(points[chosen_idx]);
    }

    let iterations = if k > 500 { 10 } else { 20 };
    for _ in 0..iterations {
        let mut sums_x = vec![0.0f64; k];
        let mut sums_y = vec![0.0f64; k];
        let mut sums_z = vec![0.0f64; k];
        let mut counts = vec![0usize; k];

        let solver = Solver::new(centroids.clone());
        
        for p in points {
            let best_idx = solver.find_nearest(p);
            sums_x[best_idx] += p.x as f64;
            sums_y[best_idx] += p.y as f64;
            sums_z[best_idx] += p.z as f64;
            counts[best_idx] += 1;
        }

        for i in 0..k {
            if counts[i] > 0 {
                centroids[i] = Point {
                    x: (sums_x[i] / counts[i] as f64) as f32,
                    y: (sums_y[i] / counts[i] as f64) as f32,
                    z: (sums_z[i] / counts[i] as f64) as f32,
                };
            } else {
                centroids[i] = points[next_u64() as usize % points.len()];
            }
        }
    }

    centroids
}

fn compute_assignments_parallel(points: &[Point], solver: Arc<Solver>) -> Vec<u32> {
    let num_points = points.len();
    if num_points == 0 {
        return vec![];
    }
    
    let num_threads = thread::available_parallelism().map(|n| n.get()).unwrap_or(1);
    if num_threads <= 1 || num_points < 2000 {
        return points.iter().map(|p| solver.find_nearest(p) as u32).collect();
    }

    let chunk_size = (num_points + num_threads - 1) / num_threads;
    let points_vec = Arc::new(points.to_vec());
    let mut handles = Vec::new();

    for t in 0..num_threads {
        let solver = Arc::clone(&solver);
        let points = Arc::clone(&points_vec);
        let start = t * chunk_size;
        let end = (start + chunk_size).min(num_points);
        if start >= num_points { break; }

        handles.push(thread::spawn(move || {
            let mut local = Vec::with_capacity(end - start);
            for i in start..end {
                local.push(solver.find_nearest(&points[i]) as u32);
            }
            local
        }));
    }

    let mut all_assignments = Vec::with_capacity(num_points);
    for h in handles {
        all_assignments.extend(h.join().unwrap());
    }
    all_assignments
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());

    let mut header = String::new();
    while header.split_whitespace().count() < 2 {
        if reader.read_line(&mut header)? == 0 {
            break;
        }
    }
    
    let parts: Vec<&str> = header.split_whitespace().collect();
    if parts.len() < 2 {
        return Ok(());
    }
    
    let num_points: u64 = parts[0].parse().unwrap_or(0);
    let num_clusters: usize = parts[1].parse().unwrap_or(0);
    
    if num_points == 0 || num_clusters == 0 {
        return Ok(());
    }

    let sample_limit = 2_000_000_u64.min(num_points);
    let mut sample = Vec::with_capacity(sample_limit as usize);
    let mut buf = [0u8; 12];
    for _ in 0..sample_limit {
        reader.read_exact(&mut buf)?;
        sample.push(Point {
            x: f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]),
            y: f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]),
            z: f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]),
        });
    }

    let centroids = run_kmeans(&sample, num_clusters);
    let solver = Arc::new(Solver::new(centroids));

    let stdout = io::stdout();
    let mut writer = BufWriter::with_capacity(1 << 20, stdout.lock());

    let assignments = compute_assignments_parallel(&sample, solver.clone());
    for idx in assignments {
        writer.write_all(&idx.to_le_bytes())?;
    }

    if num_points > sample_limit {
        let remaining = num_points - sample_limit;
        let batch_size = 200_000;
        let mut batch = Vec::with_capacity(batch_size);
        let mut processed = 0;
        while processed < remaining {
            let to_read = (remaining - processed).min(batch_size as u64) as usize;
            batch.clear();
            for _ in 0..to_read {
                reader.read_exact(&mut buf)?;
                batch.push(Point {
                    x: f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]),
                    y: f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]),
                    z: f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]),
                });
            }
            let assignments = compute_assignments_parallel(&batch, solver.clone());
            for idx in assignments {
                writer.write_all(&idx.to_le_bytes())?;
            }
            processed += to_read as u64;
        }
    }
    
    writer.flush()?;
    Ok(())
}
