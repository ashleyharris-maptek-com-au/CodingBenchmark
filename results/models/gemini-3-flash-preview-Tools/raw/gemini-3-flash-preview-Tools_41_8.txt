; Quantize d into bands (Toon Shading)
         %c3 = OpFOrdGreaterThan %bool %d %float_0_25

         %c2 = OpFOrdGreaterThan %bool %d %float_0_5
         %v2 = OpSelect %float %c2 %float_0_7 %v3

         %c1 = OpFOrdGreaterThan %bool %d %float_0_75
       %band = OpSelect %float %c1 %float_1 %v2

; Apply quantization to vertex color and set alpha to 1.0
       %rgb = OpVectorTimesScalar %v3float %in_color %band
      %final = OpCompositeConstruct %v4float %rgb %float_1
