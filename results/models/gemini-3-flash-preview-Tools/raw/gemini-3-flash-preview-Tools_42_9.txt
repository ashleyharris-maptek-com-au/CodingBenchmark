OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor %worldPos %normal
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
  %main_type = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

       %uint = OpTypeInt 32 0
     %uint_4 = OpConstant %uint 4

        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3

; Input gl_PerVertex
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

; User inputs
%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input

; User output
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

; Pointers for access
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Constants
%float_0_025 = OpConstant %float 0.025

; Main
       %main = OpFunction %void None %main_type
      %label = OpLabel

; Load positions
    %ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
    %ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
    %ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
    %ptr_p3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0

         %p0 = OpLoad %v4float %ptr_p0
         %p1 = OpLoad %v4float %ptr_p1
         %p2 = OpLoad %v4float %ptr_p2
         %p3 = OpLoad %v4float %ptr_p3

; XYZ components for math
       %p0_3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
       %p1_3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
       %p2_3 = OpVectorShuffle %v3float %p2 %p2 0 1 2
       %p3_3 = OpVectorShuffle %v3float %p3 %p3 0 1 2

; Tangents
       %v0_3 = OpFSub %v3float %p2_3 %p0_3
       %v1_3 = OpFSub %v3float %p3_3 %p1_3
         %t0 = OpExtInst %v3float %1 Normalize %v0_3
         %t1 = OpExtInst %v3float %1 Normalize %v1_3

; Perps (2D)
      %t0_x = OpCompositeExtract %float %t0 0
      %t0_y = OpCompositeExtract %float %t0 1
      %t1_x = OpCompositeExtract %float %t1 0
      %t1_y = OpCompositeExtract %float %t1 1

     %neg_t0_y = OpFNegate %float %t0_y
     %neg_t1_y = OpFNegate %float %t1_y

      %perp0_2 = OpCompositeConstruct %v2float %neg_t0_y %t0_x
      %perp1_2 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Thickness offset
     %offset0_2 = OpVectorTimesScalar %v2float %perp0_2 %float_0_025
     %offset1_2 = OpVectorTimesScalar %v2float %perp1_2 %float_0_025

; Load colors
   %ptr_col1 = OpAccessChain %ptr_in_v3 %color %int_1
   %ptr_col2 = OpAccessChain %ptr_in_v3 %color %int_2
       %col1 = OpLoad %v3float %ptr_col1
       %col2 = OpLoad %v3float %ptr_col2

    %ptr_out_p = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit vertex 1: p1 + offset0
     %off0_x = OpCompositeExtract %float %offset0_2 0
     %off0_y = OpCompositeExtract %float %offset0_2 1
       %p1_x = OpCompositeExtract %float %p1 0
       %p1_y = OpCompositeExtract %float %p1 1
       %p1_z = OpCompositeExtract %float %p1 2
       %p1_w = OpCompositeExtract %float %p1 3
      %v1_x_pos = OpFAdd %float %p1_x %off0_x
      %v1_y_pos = OpFAdd %float %p1_y %off0_y
     %pos1_v4 = OpCompositeConstruct %v4float %v1_x_pos %v1_y_pos %p1_z %p1_w
               OpStore %outColor %col1
               OpStore %ptr_out_p %pos1_v4
               OpEmitVertex

; Emit vertex 2: p1 - offset0
      %v1_x_neg = OpFSub %float %p1_x %off0_x
      %v1_y_neg = OpFSub %float %p1_y %off0_y
     %pos2_v4 = OpCompositeConstruct %v4float %v1_x_neg %v1_y_neg %p1_z %p1_w
               OpStore %outColor %col1
               OpStore %ptr_out_p %pos2_v4
               OpEmitVertex

; Emit vertex 3: p2 + offset1
     %off1_x = OpCompositeExtract %float %offset1_2 0
     %off1_y = OpCompositeExtract %float %offset1_2 1
       %p2_x = OpCompositeExtract %float %p2 0
       %p2_y = OpCompositeExtract %float %p2 1
       %p2_z = OpCompositeExtract %float %p2 2
       %p2_w = OpCompositeExtract %float %p2 3
      %v2_x_pos = OpFAdd %float %p2_x %off1_x
      %v2_y_pos = OpFAdd %float %p2_y %off1_y
     %pos3_v4 = OpCompositeConstruct %v4float %v2_x_pos %v2_y_pos %p2_z %p2_w
               OpStore %outColor %col2
               OpStore %ptr_out_p %pos3_v4
               OpEmitVertex

; Emit vertex 4: p2 - offset1
      %v2_x_neg = OpFSub %float %p2_x %off1_x
      %v2_y_neg = OpFSub %float %p2_y %off1_y
     %pos4_v4 = OpCompositeConstruct %v4float %v2_x_neg %v2_y_neg %p2_z %p2_w
               OpStore %outColor %col2
               OpStore %ptr_out_p %pos4_v4
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
