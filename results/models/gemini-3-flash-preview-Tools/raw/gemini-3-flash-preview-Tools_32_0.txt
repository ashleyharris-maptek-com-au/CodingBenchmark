This C++ solution addresses the NP-Hard **Steiner Tree problem** by employing a hybrid approach that balances optimality and efficiency.

### Algorithm Design
1.  **Exact Solution (Dreyfus-Wagner DP):** For small terminal sets (where $k \le 13$ and the state space is manageable), the code implements the **Dreyfus-Wagner dynamic programming** algorithm. This approach computes the minimum weight tree connecting subsets of terminals using the recurrence:
    $$dp[S][v] = \min(\min_{T \subset S} (dp[T][v] + dp[S \setminus T][v]), \min_{u} (dp[S][u] + dist(u, v)))$$
    The implementation includes an optimized split step and uses Dijkstra's algorithm for shortest-path updates within the state space.
2.  **Heuristic Solution (Distance Network):** For larger terminal sets, the code falls back to the **Distance Network Heuristic** (also known as Mehlhorn's approach). It constructs a complete graph of terminals with edges representing original shortest paths, finds the Minimum Spanning Tree (MST) of this network, maps it back to original edges, and prunes unnecessary leaves. This provides a guaranteed 2-approximation in polynomial time.
3.  **Edge Reconstruction:** Both methods include logic to reconstruct and output the specific edges forming the tree. The DP version uses back-pointers to trace the optimal solution, while the heuristic version prunes a subgraph MST to ensure the final result is a tree including all terminals.

### Implementation Details
- **Memory Efficiency:** State tables are managed using `std::vector` and accessed sequentially to improve cache locality.
- **Robustness:** A final MST step ensures that the output is always acyclic and minimally weighted for the discovered connectivity.
- **Language Standard:** The code adheres to the **C++20** standard, utilizing `<bit>` for efficient bit manipulation and standard library containers for safety.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>
#include <set>
#include <bit>

using namespace std;

typedef long long ll;
const ll INF = LLONG_MAX / 4;

struct Edge {
    int to;
    int weight;
};

struct State {
    int v;
    ll dist;
    bool operator>(const State& other) const {
        return dist > other.dist;
    }
};

struct BackLink {
    int submask;
    int v_prev;
};

struct FullEdge {
    int u, v, w;
};

struct DSU {
    vector<int> parent;
    DSU(int n) {
        parent.resize(n);
        for (int i = 0; i < n; ++i) parent[i] = i;
    }
    int find(int i) {
        if (parent[i] == i) return i;
        return parent[i] = find(parent[i]);
    }
    bool unite(int i, int j) {
        int root_i = find(i);
        int root_j = find(j);
        if (root_i != root_j) {
            parent[root_i] = root_j;
            return true;
        }
        return false;
    }
};

void solve_dp(int n, int m, int k, const vector<vector<Edge>>& adj, const vector<int>& terminals) {
    vector<vector<ll>> dp(1 << k, vector<ll>(n, INF));
    vector<vector<BackLink>> parent(1 << k, vector<BackLink>(n, {0, -1}));

    for (int i = 0; i < k; ++i) {
        if (terminals[i] >= 0 && terminals[i] < n) {
            dp[1 << i][terminals[i]] = 0;
        }
    }

    for (int mask = 1; mask < (1 << k); ++mask) {
        if (std::popcount((unsigned int)mask) > 1) {
            for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
                if (sub < (mask ^ sub)) continue;
                ll* dps = dp[sub].data();
                ll* dps2 = dp[mask ^ sub].data();
                ll* dpm = dp[mask].data();
                for (int v = 0; v < n; ++v) {
                    if (dps[v] < INF && dps2[v] < INF) {
                        ll val = dps[v] + dps2[v];
                        if (val < dpm[v]) {
                            dpm[v] = val;
                            parent[mask][v] = {sub, -1};
                        }
                    }
                }
            }
        }

        priority_queue<State, vector<State>, greater<State>> pq;
        ll* dpm = dp[mask].data();
        for (int v = 0; v < n; ++v) {
            if (dpm[v] < INF) {
                pq.push({v, dpm[v]});
            }
        }

        while (!pq.empty()) {
            State top = pq.top();
            pq.pop();
            if (top.dist > dpm[top.v]) continue;
            for (auto& e : adj[top.v]) {
                if (dpm[e.to] > top.dist + e.weight) {
                    dpm[e.to] = top.dist + e.weight;
                    parent[mask][e.to] = {0, top.v};
                    pq.push({e.to, dpm[e.to]});
                }
            }
        }
    }

    ll min_weight = INF;
    int root = -1;
    for (int v = 0; v < n; ++v) {
        if (dp[(1 << k) - 1][v] < min_weight) {
            min_weight = dp[(1 << k) - 1][v];
            root = v;
        }
    }

    if (root == -1) return;

    set<pair<int, int>> tree_edges;
    vector<vector<bool>> used(1 << k, vector<bool>(n, false));

    auto reconstruct = [&](auto self, int mask, int v) -> void {
        if (mask <= 0 || used[mask][v]) return;
        used[mask][v] = true;
        if (std::popcount((unsigned int)mask) == 1) {
            int t_idx = std::countr_zero((unsigned int)mask);
            if (terminals[t_idx] == v) return;
        }
        if (parent[mask][v].submask > 0) {
            self(self, parent[mask][v].submask, v);
            self(self, mask ^ parent[mask][v].submask, v);
        } else if (parent[mask][v].v_prev != -1) {
            int pv = parent[mask][v].v_prev;
            int u = pv, vv = v;
            if (u > vv) swap(u, vv);
            tree_edges.insert({u, vv});
            self(self, mask, pv);
        }
    };

    reconstruct(reconstruct, (1 << k) - 1, root);

    vector<FullEdge> all_edges;
    for (auto& e : tree_edges) {
        for (auto& ae : adj[e.first]) {
            if (ae.to == e.second) {
                all_edges.push_back({e.first, e.second, ae.weight});
                break;
            }
        }
    }
    sort(all_edges.begin(), all_edges.end(), [](const FullEdge& a, const FullEdge& b){
        if (a.w != b.w) return a.w < b.w;
        if (a.u != b.u) return a.u < b.u;
        return a.v < b.v;
    });
    
    DSU dsu_final(n);
    vector<pair<int, int>> final_tree;
    ll final_weight = 0;
    for (auto& e : all_edges) {
        if (dsu_final.unite(e.u, e.v)) {
            final_tree.push_back({e.u, e.v});
            final_weight += e.w;
        }
    }
    
    cout << final_weight << " " << final_tree.size() << "\n";
    for (auto& e : final_tree) cout << e.first << " " << e.second << "\n";
}

void solve_heuristic(int n, int m, int k, const vector<vector<Edge>>& adj, const vector<int>& terminals) {
    vector<vector<ll>> dists(k, vector<ll>(n, INF));
    vector<vector<int>> prev_v(k, vector<int>(n, -1));

    for (int i = 0; i < k; ++i) {
        priority_queue<State, vector<State>, greater<State>> pq;
        dists[i][terminals[i]] = 0;
        pq.push({terminals[i], 0});
        while (!pq.empty()) {
            State top = pq.top();
            pq.pop();
            if (top.dist > dists[i][top.v]) continue;
            for (auto& e : adj[top.v]) {
                if (dists[i][e.to] > top.dist + e.weight) {
                    dists[i][e.to] = top.dist + e.weight;
                    prev_v[i][e.to] = top.v;
                    pq.push({e.to, dists[i][e.to]});
                }
            }
        }
    }

    struct KEdge { int u, v; ll w; bool operator<(const KEdge& o) const { return w < o.w; } };
    vector<KEdge> k_edges;
    for (int i = 0; i < k; ++i)
        for (int j = i + 1; j < k; ++j)
            if (dists[i][terminals[j]] < INF) k_edges.push_back({i, j, dists[i][terminals[j]]});
    sort(k_edges.begin(), k_edges.end());

    DSU dsu_k(k);
    set<pair<int, int>> sub_edges;
    for (auto& ke : k_edges) {
        if (dsu_k.unite(ke.u, ke.v)) {
            int curr = terminals[ke.v];
            while (curr != terminals[ke.u]) {
                int p = prev_v[ke.u][curr];
                if (p == -1) break; 
                int a = curr, b = p;
                if (a > b) swap(a, b);
                sub_edges.insert({a, b});
                curr = p;
            }
        }
    }

    vector<FullEdge> fe;
    for (auto& se : sub_edges)
        for (auto& ae : adj[se.first])
            if (ae.to == se.second) { fe.push_back({se.first, se.second, ae.weight}); break; }
    sort(fe.begin(), fe.end(), [](const FullEdge& a, const FullEdge& b) {
        if (a.w != b.w) return a.w < b.w;
        return a.u < b.u;
    });

    DSU dsu_n(n);
    vector<pair<int, int>> mst_edges;
    for (auto& e : fe) if (dsu_n.unite(e.u, e.v)) mst_edges.push_back({e.u, e.v});

    vector<int> degree(n, 0);
    vector<vector<int>> mst_adj(n);
    for (auto& e : mst_edges) {
        degree[e.first]++; degree[e.second]++;
        mst_adj[e.first].push_back(e.second); mst_adj[e.second].push_back(e.first);
    }

    set<int> terminal_set(terminals.begin(), terminals.end());
    queue<int> leaves;
    for (int i = 0; i < n; ++i) if (degree[i] == 1 && terminal_set.find(i) == terminal_set.end()) leaves.push(i);

    vector<bool> removed(n, false);
    while (!leaves.empty()) {
        int u = leaves.front(); leaves.pop();
        removed[u] = true;
        for (int v : mst_adj[u]) if (!removed[v]) {
            degree[v]--;
            if (degree[v] == 1 && terminal_set.find(v) == terminal_set.end()) leaves.push(v);
        }
    }

    vector<pair<int, int>> final_edges;
    ll final_w = 0;
    for (auto& e : mst_edges) if (!removed[e.first] && !removed[e.second]) {
        final_edges.push_back(e);
        for (auto& ae : adj[e.first]) if (ae.to == e.second) { final_w += ae.weight; break; }
    }

    cout << final_w << " " << final_edges.size() << "\n";
    for (auto& e : final_edges) cout << e.first << " " << e.second << "\n";
}

int main() {
    ios::sync_with_stdio(false); cin.tie(nullptr);
    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;
    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v, w; cin >> u >> v >> w;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        adj[u].push_back({v, w}); adj[v].push_back({u, w});
    }
    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];
    if (k <= 13 && (ll)(1 << k) * n <= 20000000) solve_dp(n, m, k, adj, terminals);
    else solve_heuristic(n, m, k, adj, terminals);
    return 0;
}
```
