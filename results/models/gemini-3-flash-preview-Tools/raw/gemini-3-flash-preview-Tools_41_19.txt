; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %color %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_v4 = OpTypePointer Uniform %v4float

; Constants
%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_05 = OpConstant %float 0.05
%float_32 = OpConstant %float 32.0

%v3_ambient = OpConstantComposite %v3float %float_0_1 %float_0_05 %float_0_05
%v3_specular_base = OpConstantComposite %v3float %float_1 %float_1 %float_1
%v3_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0
%v3_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Variables
%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input
%color = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%ubo_var = OpVariable %ptr_uniform_UBO Uniform

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; Load input data
%wPos = OpLoad %v3float %worldPos
%nRaw = OpLoad %v3float %normal
%vColor = OpLoad %v3float %color

; Normalize the normal vector
%N = OpExtInst %v3float %1 Normalize %nRaw

; Extract light position and camera position from UBO
%lightPosPtr = OpAccessChain %ptr_uniform_v4 %ubo_var %int_3
%lightPos4 = OpLoad %v4float %lightPosPtr
%lPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

%cameraPosPtr = OpAccessChain %ptr_uniform_v4 %ubo_var %int_4
%cameraPos4 = OpLoad %v4float %cameraPosPtr
%cPos = OpVectorShuffle %v3float %cameraPos4 %cameraPos4 0 1 2

; Vector from fragment to light (L)
%lDir = OpFSub %v3float %lPos %wPos
%L = OpExtInst %v3float %1 Normalize %lDir

; Vector from fragment to camera (V)
%vDir = OpFSub %v3float %cPos %wPos
%V = OpExtInst %v3float %1 Normalize %vDir

; Diffuse lighting
%dotNL = OpDot %float %N %L
%diffuseFactor = OpExtInst %float %1 FMax %dotNL %float_0
%diffuse = OpVectorTimesScalar %v3float %vColor %diffuseFactor

; Specular lighting (Phong)
%negL = OpFNegate %v3float %L
%R = OpExtInst %v3float %1 Reflect %negL %N
%dotRV = OpDot %float %R %V
%dotRV_clamped = OpExtInst %float %1 FMax %dotRV %float_0
%specularFactor = OpExtInst %float %1 Pow %dotRV_clamped %float_32
%specular = OpVectorTimesScalar %v3float %v3_specular_base %specularFactor

; Combine ambient, diffuse, and specular components
%sum1 = OpFAdd %v3float %v3_ambient %diffuse
%totalColor = OpFAdd %v3float %sum1 %specular

; Clamp and construct final RGBA color
%clampedColor = OpExtInst %v3float %1 FClamp %totalColor %v3_0 %v3_1
%x = OpCompositeExtract %float %clampedColor 0
%y = OpCompositeExtract %float %clampedColor 1
%z = OpCompositeExtract %float %clampedColor 2
%resColor = OpCompositeConstruct %v4float %x %y %z %float_1

OpStore %outColor %resColor
OpReturn
OpFunctionEnd
