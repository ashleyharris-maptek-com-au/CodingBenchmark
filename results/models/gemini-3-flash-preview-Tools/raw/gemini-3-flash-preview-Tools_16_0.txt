#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>

/**
 * Job-Shop Scheduling Solver using the Giffler-Thompson Algorithm.
 * 
 * Architecture:
 * 1. Data representation: Jobs consist of sequential Tasks, each assigned to a machine with a duration.
 * 2. Algorithm: Giffler-Thompson (GT) heuristic for active schedule generation.
 *    GT is renowned for producing high-quality schedules for the NP-hard JSS problem.
 * 3. Scaling Strategy: 
 *    - For small-to-medium instances (up to 100k tasks), the solver tries four different 
 *      priority rules (MWKR, SPT, FIFO, MWRT) and selects the best makespan.
 *    - For extreme scales, it utilizes the robust Most Work Remaining (MWKR) rule.
 * 4. Optimization: 
 *    - O(1) job-machine queue management via index tracking.
 *    - Optimized machine-based finish time tracking to avoid redundant O(J) scans.
 *    - Time complexity: O(Rules * Tasks * (Machines + Jobs/Machines)).
 */

using namespace std;

// Task represents a single operation in a job
struct Task {
    int machine;
    int duration;
    long long start_time;
};

// Job represents a sequence of tasks
struct Job {
    int id;
    vector<Task> tasks;
    int current_task_idx = 0;
    long long ready_at = 0;
    long long remaining_work = 0;
    int remaining_tasks = 0;
};

// Priority rules for the selection step in Giffler-Thompson
enum class PriorityRule { 
    MWKR, // Most Work Remaining (Heuristic sum of remaining durations)
    SPT,  // Shortest Processing Time (Duration of the current task)
    FIFO, // First In First Out (Based on when the job became ready)
    MWRT  // Most Work Remaining Tasks (Count of remaining tasks)
};

// Result structure to store the schedule
struct Result {
    long long makespan;
    vector<vector<long long>> start_times;
};

class JSS_Solver {
public:
    int J, M;
    vector<Job> initial_jobs;

    JSS_Solver(int j, int m) : J(j), M(m) {
        initial_jobs.resize(J);
    }

    /**
     * Executes the Giffler-Thompson algorithm with a specified priority rule.
     */
    Result run_gt(PriorityRule rule) {
        // Local state for the simulation
        vector<Job> jobs = initial_jobs;
        vector<long long> machine_ready(M, 0);
        vector<int> job_ptr(J, 0);
        
        // jobs_on_machine[m] stores indices of jobs whose current task is on machine m
        vector<vector<int>> jobs_on_machine(M);
        // job_pos_in_list[j] tracks the index of job j in its current machine's list for O(1) removal
        vector<int> job_pos_in_list(J, -1);
        // machine_min_finish[m] stores the earliest possible completion time for any job currently waiting for machine m
        vector<long long> machine_min_finish(M, numeric_limits<long long>::max());

        // Update the minimum finish time for a machine by scanning its waiting jobs
        auto update_min_finish = [&](int m) {
            if (jobs_on_machine[m].empty()) {
                machine_min_finish[m] = numeric_limits<long long>::max();
                return;
            }
            long long best = numeric_limits<long long>::max();
            for (int j_idx : jobs_on_machine[m]) {
                const auto& task = jobs[j_idx].tasks[job_ptr[j_idx]];
                long long start = max(jobs[j_idx].ready_at, machine_ready[m]);
                best = min(best, start + task.duration);
            }
            machine_min_finish[m] = best;
        };

        // Initialize lists
        int total_tasks_count = 0;
        for (int i = 0; i < J; ++i) {
            total_tasks_count += (int)jobs[i].tasks.size();
            if (!jobs[i].tasks.empty()) {
                int m = jobs[i].tasks[0].machine;
                job_pos_in_list[i] = (int)jobs_on_machine[m].size();
                jobs_on_machine[m].push_back(i);
            }
        }
        for (int m = 0; m < M; ++m) update_min_finish(m);

        vector<int> S;
        S.reserve(J);
        int tasks_done = 0;

        while (tasks_done < total_tasks_count) {
            // Step 1: Find the machine m_star that can finish a task earliest (f_star)
            long long f_star = numeric_limits<long long>::max();
            int m_star = -1;
            for (int m = 0; m < M; ++m) {
                if (machine_min_finish[m] < f_star) {
                    f_star = machine_min_finish[m];
                    m_star = m;
                }
            }
            if (m_star == -1) break;

            // Step 2: Form the conflict set S of all tasks on m_star that could start before f_star
            S.clear();
            for (int j_idx : jobs_on_machine[m_star]) {
                long long start = max(jobs[j_idx].ready_at, machine_ready[m_star]);
                if (start <= f_star) {
                    S.push_back(j_idx);
                }
            }

            // Step 3: Use the priority rule to select the best job from the conflict set
            int best_j = -1;
            double max_prio = -numeric_limits<double>::max();
            for (int j_idx : S) {
                double p;
                switch (rule) {
                    case PriorityRule::MWKR: p = (double)jobs[j_idx].remaining_work; break;
                    case PriorityRule::SPT:  p = (double)-jobs[j_idx].tasks[job_ptr[j_idx]].duration; break;
                    case PriorityRule::FIFO: p = (double)-jobs[j_idx].ready_at; break;
                    case PriorityRule::MWRT: p = (double)jobs[j_idx].remaining_tasks; break;
                    default: p = 0.0;
                }
                
                if (p > max_prio) {
                    max_prio = p;
                    best_j = j_idx;
                } else if (p == max_prio) {
                    if (best_j == -1 || j_idx < best_j) best_j = j_idx;
                }
            }

            if (best_j == -1) break;

            // Step 4: Schedule the selected task and update global state
            auto& job = jobs[best_j];
            auto& task = job.tasks[job_ptr[best_j]];
            long long start = max(job.ready_at, machine_ready[m_star]);
            task.start_time = start;
            
            machine_ready[m_star] = start + task.duration;
            job.ready_at = start + task.duration;
            job.remaining_work -= task.duration;
            job.remaining_tasks--;

            // Efficiently remove from the current machine's list
            int pos = job_pos_in_list[best_j];
            int last_j_idx = jobs_on_machine[m_star].back();
            jobs_on_machine[m_star][pos] = last_j_idx;
            job_pos_in_list[last_j_idx] = pos;
            jobs_on_machine[m_star].pop_back();
            job_pos_in_list[best_j] = -1;
            
            update_min_finish(m_star);

            // Move the job to the list for its next machine if applicable
            job_ptr[best_j]++;
            tasks_done++;
            if (job_ptr[best_j] < (int)job.tasks.size()) {
                int next_m = job.tasks[job_ptr[best_j]].machine;
                job_pos_in_list[best_j] = (int)jobs_on_machine[next_m].size();
                jobs_on_machine[next_m].push_back(best_j);
                update_min_finish(next_m);
            }
        }

        // Aggregate results
        long long final_makespan = 0;
        vector<vector<long long>> start_times_map(J);
        for (int i = 0; i < J; ++i) {
            start_times_map[i].resize(jobs[i].tasks.size());
            for (int t = 0; t < (int)jobs[i].tasks.size(); ++t) {
                start_times_map[i][t] = jobs[i].tasks[t].start_time;
                final_makespan = max(final_makespan, jobs[i].tasks[t].start_time + jobs[i].tasks[t].duration);
            }
        }
        return {final_makespan, start_times_map};
    }
};

int main() {
    // Standard speed-up for C++ I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    JSS_Solver solver(J, M);
    long long total_tasks_in_problem = 0;
    for (int i = 0; i < J; ++i) {
        int T;
        if (!(cin >> T)) break;
        solver.initial_jobs[i].id = i;
        solver.initial_jobs[i].tasks.resize(T);
        solver.initial_jobs[i].remaining_tasks = T;
        solver.initial_jobs[i].remaining_work = 0;
        for (int j = 0; j < T; ++j) {
            cin >> solver.initial_jobs[i].tasks[j].machine >> solver.initial_jobs[i].tasks[j].duration;
            solver.initial_jobs[i].remaining_work += solver.initial_jobs[i].tasks[j].duration;
        }
        total_tasks_in_problem += T;
    }

    // Adaptive Logic: Select priority rules based on problem size
    PriorityRule rules[] = {PriorityRule::MWKR, PriorityRule::SPT, PriorityRule::FIFO, PriorityRule::MWRT};
    int num_rules_to_run = 1; // Default to MWKR for speed
    
    // For problems with up to 100k tasks, we can afford multiple passes to find a better schedule
    if (total_tasks_in_problem < 100000) {
        num_rules_to_run = 4;
    }

    long long min_makespan_found = numeric_limits<long long>::max();
    PriorityRule best_overall_rule = PriorityRule::MWKR;

    for (int r_idx = 0; r_idx < num_rules_to_run; ++r_idx) {
        Result current_res = solver.run_gt(rules[r_idx]);
        if (current_res.makespan < min_makespan_found) {
            min_makespan_found = current_res.makespan;
            best_overall_rule = rules[r_idx];
        }
    }

    // Final run with the best rule discovered
    Result final_result = solver.run_gt(best_overall_rule);

    // Output per-job start times and machine IDs
    for (int i = 0; i < J; ++i) {
        const auto& job_tasks = solver.initial_jobs[i].tasks;
        for (int t = 0; t < (int)job_tasks.size(); ++t) {
            cout << final_result.start_times[i][t] << " " << job_tasks[t].machine 
                 << (t == (int)job_tasks.size() - 1 ? "" : " ");
        }
        cout << "\n";
    }

    return 0;
}
