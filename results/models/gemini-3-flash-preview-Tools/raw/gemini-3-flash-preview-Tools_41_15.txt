; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
               OpExecutionMode %main OriginUpperLeft

; Decorations
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %outColor Location 0
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

; Types and Constants
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

        %UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_ptr_Uniform_UBO Uniform

%_ptr_Input_v3float = OpTypePointer Input %v3float
   %worldPos = OpVariable %_ptr_Input_v3float Input
     %normal = OpVariable %_ptr_Input_v3float Input

%_ptr_Output_v4float = OpTypePointer Output %v4float
   %outColor = OpVariable %_ptr_Output_v4float Output

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
    %float_0 = OpConstant %float 0
  %float_0_5 = OpConstant %float 0.5
  %float_1_0 = OpConstant %float 1.0
  %float_0_2 = OpConstant %float 0.2
 %float_0_75 = OpConstant %float 0.75
  %float_0_7 = OpConstant %float 0.7
  %float_0_4 = OpConstant %float 0.4

      %cool = OpConstantComposite %v3float %float_0_2 %float_0_2 %float_0_75
      %warm = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_4

%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

; Entry point
       %main = OpFunction %void None %3
          %5 = OpLabel

; Load light position and extract xyz
     %lp_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
      %lp_v4 = OpLoad %v4float %lp_ptr
      %lp_v3 = OpVectorShuffle %v3float %lp_v4 %lp_v4 0 1 2

; Load world position
         %wp = OpLoad %v3float %worldPos

; Calculate L = normalize(lightPos.xyz - worldPos)
      %L_sub = OpFSub %v3float %lp_v3 %wp
          %L = OpExtInst %v3float %1 Normalize %L_sub

; Calculate N = normalize(normal)
      %n_in  = OpLoad %v3float %normal
          %N = OpExtInst %v3float %1 Normalize %n_in

; Calculate t = dot(N, L) * 0.5 + 0.5
      %dotNL = OpDot %float %N %L
      %t_mul = OpFMul %float %dotNL %float_0_5
          %t = OpFAdd %float %t_mul %float_0_5

; Calculate Gooch shading: mix(cool, warm, t)
     %result = OpExtInst %v3float %1 FMix %cool %warm %t

; Output final color with alpha 1.0
     %final = OpCompositeConstruct %v4float %result %float_1_0
               OpStore %outColor %final
               OpReturn
               OpFunctionEnd
