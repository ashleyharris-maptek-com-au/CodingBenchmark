use std::collections::{BinaryHeap, VecDeque};
use std::io::{self, Read, Write, BufWriter};

struct FastReader {
    buffer: Vec<u8>,
    pos: usize,
}

impl FastReader {
    fn new() -> Self {
        let mut buffer = Vec::new();
        let _ = io::stdin().read_to_end(&mut buffer);
        Self { buffer, pos: 0 }
    }

    fn next_u32(&mut self) -> Option<u32> {
        while self.pos < self.buffer.len() && !self.buffer[self.pos].is_ascii_digit() {
            self.pos += 1;
        }
        if self.pos == self.buffer.len() {
            return None;
        }
        let mut n = 0u32;
        while self.pos < self.buffer.len() && self.buffer[self.pos].is_ascii_digit() {
            let digit = (self.buffer[self.pos] - b'0') as u32;
            n = n.wrapping_mul(10).wrapping_add(digit);
            self.pos += 1;
        }
        Some(n)
    }
}

fn main() {
    let mut reader = FastReader::new();
    
    let n = match reader.next_u32() {
        Some(val) => val as usize,
        None => return,
    };
    let m = match reader.next_u32() {
        Some(val) => val as usize,
        None => 0,
    };
    
    if n == 0 {
        let stdout = io::stdout();
        let mut writer = BufWriter::new(stdout.lock());
        let _ = writeln!(writer, "0");
        let _ = writeln!(writer);
        return;
    }

    let mut edges = Vec::with_capacity(m);
    let mut initial_degrees = vec![0u32; n];
    let mut has_self_loop = vec![false; n];
    
    for _ in 0..m {
        let u_opt = reader.next_u32();
        let v_opt = reader.next_u32();
        if let (Some(u_u32), Some(v_u32)) = (u_opt, v_opt) {
            let u = u_u32 as usize;
            let v = v_u32 as usize;
            if u < n && v < n {
                if u == v {
                    has_self_loop[u] = true;
                } else {
                    initial_degrees[u] += 1;
                    initial_degrees[v] += 1;
                    edges.push((u_u32, v_u32));
                }
            }
        }
    }
    
    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + initial_degrees[i] as usize;
    }
    
    let mut neighbors = vec![0u32; offsets[n]];
    let mut current_pos = offsets.clone();
    for (u, v) in edges {
        neighbors[current_pos[u as usize]] = v;
        current_pos[u as usize] += 1;
        neighbors[current_pos[v as usize]] = u;
        current_pos[v as usize] += 1;
    }
    
    let mut current_degrees = initial_degrees;
    let mut removed = vec![false; n];
    let mut in_vc = vec![false; n];
    let mut q1 = VecDeque::new();
    
    // Handle self-loops: a vertex with a self-loop must be in the vertex cover
    for i in 0..n {
        if has_self_loop[i] {
            in_vc[i] = true;
            removed[i] = true;
        }
    }
    
    // Decrement degrees of neighbors for vertices already in VC due to self-loops
    for i in 0..n {
        if in_vc[i] {
            for j in offsets[i]..offsets[i+1] {
                let v = neighbors[j] as usize;
                if !removed[v] {
                    current_degrees[v] -= 1;
                }
            }
        }
    }
    
    // Identify vertices with degree 0 or 1
    for i in 0..n {
        if !removed[i] {
            if current_degrees[i] == 0 {
                removed[i] = true;
            } else if current_degrees[i] == 1 {
                q1.push_back(i);
            }
        }
    }

    let mut pq = BinaryHeap::with_capacity(n);
    for i in 0..n {
        if !removed[i] && current_degrees[i] > 1 {
            pq.push((current_degrees[i], i as u32));
        }
    }
    
    loop {
        // Step 1: Process degree-1 vertices (Reduction rule)
        while let Some(v) = q1.pop_front() {
            if removed[v] || current_degrees[v] != 1 {
                continue;
            }
            
            // Find the unique neighbor u of v that is not removed
            let mut neighbor_u = None;
            for j in offsets[v]..offsets[v+1] {
                let u_idx = neighbors[j] as usize;
                if !removed[u_idx] {
                    neighbor_u = Some(u_idx);
                    break;
                }
            }
            
            if let Some(u) = neighbor_u {
                // For any vertex with degree 1, its only neighbor must be in a minimum vertex cover
                in_vc[u] = true;
                removed[u] = true;
                for j in offsets[u]..offsets[u+1] {
                    let w = neighbors[j] as usize;
                    if !removed[w] {
                        current_degrees[w] -= 1;
                        if current_degrees[w] == 1 {
                            q1.push_back(w);
                        } else if current_degrees[w] == 0 {
                            removed[w] = true;
                        }
                    }
                }
            } else {
                // This case should only occur if the graph is inconsistent
                removed[v] = true;
            }
        }
        
        // Step 2: Greedy choice - pick vertex with maximum degree
        let mut picked_greedy = false;
        while let Some((deg, u_idx)) = pq.pop() {
            let u = u_idx as usize;
            if removed[u] || deg < current_degrees[u] {
                continue;
            }
            
            in_vc[u] = true;
            removed[u] = true;
            for j in offsets[u]..offsets[u+1] {
                let w = neighbors[j] as usize;
                if !removed[w] {
                    current_degrees[w] -= 1;
                    if current_degrees[w] == 1 {
                        q1.push_back(w);
                    } else if current_degrees[w] == 0 {
                        removed[w] = true;
                    } else {
                        pq.push((current_degrees[w], w as u32));
                    }
                }
            }
            picked_greedy = true;
            break; // Return to check if more degree-1 vertices were created
        }
        
        if !picked_greedy && q1.is_empty() {
            break;
        }
    }
    
    let mut vc_list = Vec::new();
    for i in 0..n {
        if in_vc[i] {
            vc_list.push(i);
        }
    }
    
    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    let _ = writeln!(writer, "{}", vc_list.len());
    for (i, v) in vc_list.iter().enumerate() {
        if i > 0 {
            let _ = write!(writer, " ");
        }
        let _ = write!(writer, "{}", v);
    }
    let _ = writeln!(writer);
}
