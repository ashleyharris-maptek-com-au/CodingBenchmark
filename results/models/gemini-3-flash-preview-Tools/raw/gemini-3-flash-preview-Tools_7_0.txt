#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <limits>
#include <stdint.h>
#include <random>

using namespace std;

const double EPS = 1e-8;

struct Vec3 {
    double x, y, z;
    Vec3 operator+(const Vec3& v) const { return {x + v.x, y + v.y, z + v.z}; }
    Vec3 operator-(const Vec3& v) const { return {x - v.x, y - v.y, z - v.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return {y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x};
    }
    double lenSq() const { return x * x + y * y + z * z; }
    double len() const { return sqrt(lengthSq()); }
    double lengthSq() const { return x * x + y * y + z * z; }
    Vec3 normalize() const {
        double l = len();
        return (l > 1e-12) ? (*this * (1.0 / l)) : Vec3{0, 0, 0};
    }
};

struct AABB {
    Vec3 min, max;
    AABB() {
        min = {1e18, 1e18, 1e18};
        max = {-1e18, -1e18, -1e18};
    }
    void expand(Vec3 v) {
        min.x = std::min(min.x, v.x); min.y = std::min(min.y, v.y); min.z = std::min(min.z, v.z);
        max.x = std::max(max.x, v.x); max.y = std::max(max.y, v.y); max.z = std::max(max.z, v.z);
    }
    bool intersects(const AABB& o) const {
        return (min.x <= o.max.x + EPS && max.x >= o.min.x - EPS) &&
               (min.y <= o.max.y + EPS && max.y >= o.min.y - EPS) &&
               (min.z <= o.max.z + EPS && max.z >= o.min.z - EPS);
    }
};

struct Tri {
    int v[3];
};

struct BVHNode {
    AABB box;
    int left, right; // if left < 0, right is tri index
};

vector<BVHNode> build_bvh(const vector<Vec3>& verts, const vector<Tri>& tris) {
    int n = tris.size();
    if (n == 0) return {};
    vector<int> indices(n);
    for (int i = 0; i < n; ++i) indices[i] = i;
    vector<BVHNode> nodes;
    nodes.reserve(2 * n);
    auto build = [&](auto self, int start, int end) -> int {
        int nodeIdx = (int)nodes.size();
        nodes.push_back({});
        AABB box;
        for (int i = start; i < end; ++i) {
            int t = indices[i];
            box.expand(verts[tris[t].v[0]]);
            box.expand(verts[tris[t].v[1]]);
            box.expand(verts[tris[t].v[2]]);
        }
        nodes[nodeIdx].box = box;
        if (end - start <= 1) {
            nodes[nodeIdx].left = -1;
            nodes[nodeIdx].right = indices[start];
            return nodeIdx;
        }
        int axis = 0;
        double dx = box.max.x - box.min.x, dy = box.max.y - box.min.y, dz = box.max.z - box.min.z;
        if (dy > dx && dy > dz) axis = 1; else if (dz > dx && dz > dy) axis = 2;
        int mid = (start + end) / 2;
        nth_element(indices.begin() + start, indices.begin() + mid, indices.begin() + end,
            [&](int a, int b) {
                double ca = (verts[tris[a].v[0]].dot({axis==0?1.0:0.0, axis==1?1.0:0.0, axis==2?1.0:0.0}) +
                             verts[tris[a].v[1]].dot({axis==0?1.0:0.0, axis==1?1.0:0.0, axis==2?1.0:0.0}) +
                             verts[tris[a].v[2]].dot({axis==0?1.0:0.0, axis==1?1.0:0.0, axis==2?1.0:0.0}));
                double cb = (verts[tris[b].v[0]].dot({axis==0?1.0:0.0, axis==1?1.0:0.0, axis==2?1.0:0.0}) +
                             verts[tris[b].v[1]].dot({axis==0?1.0:0.0, axis==1?1.0:0.0, axis==2?1.0:0.0}) +
                             verts[tris[b].v[2]].dot({axis==0?1.0:0.0, axis==1?1.0:0.0, axis==2?1.0:0.0}));
                return ca < cb;
            });
        int l = self(self, start, mid);
        int r = self(self, mid, end);
        nodes[nodeIdx].left = l;
        nodes[nodeIdx].right = r;
        return nodeIdx;
    };
    build(build, 0, n);
    return nodes;
}

bool intersect_ray_tri(const Vec3& orig, const Vec3& dir, const Vec3& v0, const Vec3& v1, const Vec3& v2, double& t) {
    Vec3 e1 = v1 - v0, e2 = v2 - v0;
    Vec3 h = dir.cross(e2);
    double a = e1.dot(h);
    if (abs(a) < 1e-12) return false;
    double f = 1.0 / a;
    Vec3 s = orig - v0;
    double u = f * s.dot(h);
    if (u < 0.0 || u > 1.0) return false;
    Vec3 q = s.cross(e1);
    double v = f * dir.dot(q);
    if (v < 0.0 || u + v > 1.0) return false;
    t = f * e2.dot(q);
    return t > 1e-9;
}

bool ray_cast(const Vec3& P, const Vec3& dir, const vector<Vec3>& verts, const vector<Tri>& tris, const vector<BVHNode>& nodes) {
    if (nodes.empty()) return false;
    int count = 0;
    vector<int> stack = {0};
    while (!stack.empty()) {
        int idx = stack.back(); stack.pop_back();
        const auto& node = nodes[idx];
        double tmin = -1e18, tmax = 1e18;
        for (int i = 0; i < 3; ++i) {
            double d = (i == 0 ? dir.x : (i == 1 ? dir.y : dir.z));
            double o = (i == 0 ? P.x : (i == 1 ? P.y : P.z));
            double mn = (i == 0 ? node.box.min.x : (i == 1 ? node.box.min.y : node.box.min.z));
            double mx = (i == 0 ? node.box.max.x : (i == 1 ? node.box.max.y : node.box.max.z));
            if (abs(d) < 1e-12) { if (o < mn - EPS || o > mx + EPS) { tmin = 1; tmax = -1; break; } }
            else { double t1 = (mn - o) / d, t2 = (mx - o) / d; if (t1 > t2) swap(t1, t2); tmin = max(tmin, t1); tmax = min(tmax, t2); }
        }
        if (tmin > tmax || tmax < 1e-9) continue;
        if (node.left < 0) {
            double t;
            if (intersect_ray_tri(P, dir, verts[tris[node.right].v[0]], verts[tris[node.right].v[1]], verts[tris[node.right].v[2]], t)) count++;
        } else { stack.push_back(node.left); stack.push_back(node.right); }
    }
    return (count % 2) != 0;
}

bool is_inside(const Vec3& P, const vector<Vec3>& verts, const vector<Tri>& tris, const vector<BVHNode>& nodes) {
    int inside = 0;
    Vec3 dirs[3] = {{1.0, 0.013, 0.017}, {-0.011, 1.0, 0.023}, {0.019, -0.021, 1.0}};
    for (int i = 0; i < 3; ++i) if (ray_cast(P, dirs[i].normalize(), verts, tris, nodes)) inside++;
    return inside >= 2;
}

void split_tri_by_plane(const vector<Vec3>& tri, Vec3 N, double D, vector<vector<Vec3>>& res) {
    double dists[3];
    bool all_f = true, all_b = true;
    for (int i = 0; i < 3; ++i) {
        dists[i] = tri[i].dot(N) + D;
        if (dists[i] < -1e-9) all_f = false;
        if (dists[i] > 1e-9) all_b = false;
    }
    if (all_f || all_b) { res.push_back(tri); return; }
    vector<Vec3> f_pts, b_pts;
    for (int i = 0; i < 3; ++i) {
        int j = (i + 1) % 3;
        if (dists[i] > -1e-11) f_pts.push_back(tri[i]);
        if (dists[i] < 1e-11) b_pts.push_back(tri[i]);
        if ((dists[i] > 1e-10 && dists[j] < -1e-10) || (dists[i] < -1e-10 && dists[j] > 1e-10)) {
            double t = dists[i] / (dists[i] - dists[j]);
            Vec3 p = tri[i] + (tri[j] - tri[i]) * t;
            f_pts.push_back(p); b_pts.push_back(p);
        }
    }
    auto triangulate = [&](const vector<Vec3>& p) {
        if (p.size() < 3) return;
        vector<Vec3> up;
        for (auto& v : p) {
            bool found = false;
            for (auto& uv : up) if ((v-uv).lenSq() < 1e-18) { found = true; break; }
            if (!found) up.push_back(v);
        }
        if (up.size() < 3) return;
        for (int i = 1; i < (int)up.size() - 1; ++i) res.push_back({up[0], up[i], up[i+1]});
    };
    triangulate(f_pts); triangulate(b_pts);
}

int main() {
    ios::sync_with_stdio(false); cin.tie(0);
    int va, fa, vb, fb;
    if (!(cin >> va >> fa >> vb >> fb)) return 0;
    vector<Vec3> vertsA(va), vertsB(vb);
    vector<Tri> trisA(fa), trisB(fb);
    for (int i = 0; i < va; ++i) cin >> vertsA[i].x >> vertsA[i].y >> vertsA[i].z;
    for (int i = 0; i < fa; ++i) cin >> trisA[i].v[0] >> trisA[i].v[1] >> trisA[i].v[2];
    for (int i = 0; i < vb; ++i) cin >> vertsB[i].x >> vertsB[i].y >> vertsB[i].z;
    for (int i = 0; i < fb; ++i) cin >> trisB[i].v[0] >> trisB[i].v[1] >> trisB[i].v[2];

    auto bvhA = build_bvh(vertsA, trisA);
    auto bvhB = build_bvh(vertsB, trisB);
    vector<vector<Vec3>> final_tris;

    auto process = [&](const vector<Vec3>& vA, const vector<Tri>& tA, const vector<BVHNode>& nA,
                       const vector<Vec3>& vB, const vector<Tri>& tB, const vector<BVHNode>& nB, bool isA) {
        for (int i = 0; i < (int)tA.size(); ++i) {
            AABB box;
            box.expand(vA[tA[i].v[0]]); box.expand(vA[tA[i].v[1]]); box.expand(vA[tA[i].v[2]]);
            vector<int> intersecting;
            vector<int> stack = {0};
            if (!nB.empty()) {
                while (!stack.empty()) {
                    int idx = stack.back(); stack.pop_back();
                    if (nB[idx].box.intersects(box)) {
                        if (nB[idx].left < 0) intersecting.push_back(nB[idx].right);
                        else { stack.push_back(nB[idx].left); stack.push_back(nB[idx].right); }
                    }
                }
            }
            vector<vector<Vec3>> current = {{vA[tA[i].v[0]], vA[tA[i].v[1]], vA[tA[i].v[2]]}};
            for (int j : intersecting) {
                vector<vector<Vec3>> next;
                Vec3 v0 = vB[tB[j].v[0]], v1 = vB[tB[j].v[1]], v2 = vB[tB[j].v[2]];
                Vec3 N = (v1 - v0).cross(v2 - v0);
                double ln = N.len();
                if (ln < 1e-12) continue;
                N = N * (1.0 / ln);
                double D = -v0.dot(N);
                for (auto& tri : current) split_tri_by_plane(tri, N, D, next);
                current = next;
            }
            for (auto& tri : current) {
                Vec3 centroid = (tri[0] + tri[1] + tri[2]) * (1.0 / 3.0);
                Vec3 N = (tri[1] - tri[0]).cross(tri[2] - tri[0]);
                double ln = N.len();
                if (ln < 1e-15) continue;
                N = N * (1.0 / ln);
                if (isA) {
                    if (!is_inside(centroid + N * 1e-7, vB, tB, nB)) final_tris.push_back(tri);
                } else {
                    if (!is_inside(centroid + N * 1e-7, vA, tA, nA) && !is_inside(centroid - N * 1e-7, vA, tA, nA))
                        final_tris.push_back(tri);
                }
            }
        }
    };

    process(vertsA, trisA, bvhA, vertsB, trisB, bvhB, true);
    process(vertsB, trisB, bvhB, vertsA, trisA, bvhA, false);

    vector<Vec3> all_v;
    for (auto& t : final_tris) for (int i = 0; i < 3; ++i) all_v.push_back(t[i]);
    auto cmp = [](const Vec3& a, const Vec3& b) {
        if (abs(a.x - b.x) > 1e-9) return a.x < b.x;
        if (abs(a.y - b.y) > 1e-9) return a.y < b.y;
        return a.z < b.z - 1e-9;
    };
    sort(all_v.begin(), all_v.end(), cmp);
    all_v.erase(unique(all_v.begin(), all_v.end(), [](const Vec3& a, const Vec3& b) {
        return (a - b).lenSq() < 1e-17;
    }), all_v.end());

    cout << all_v.size() << " " << final_tris.size() << "\n";
    for (auto& v : all_v) cout << v.x << " " << v.y << " " << v.z << "\n";
    for (auto& t : final_tris) {
        for (int i = 0; i < 3; ++i) {
            auto it = lower_bound(all_v.begin(), all_v.end(), t[i], cmp);
            cout << (int)(it - all_v.begin()) << (i == 2 ? "" : " ");
        }
        cout << "\n";
    }
    return 0;
}
