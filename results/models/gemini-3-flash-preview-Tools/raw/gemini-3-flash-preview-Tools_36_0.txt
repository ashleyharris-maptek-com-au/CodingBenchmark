The following C++ implementation solves the **Maximum Independent Set** problem exactly using a branching algorithm with several key optimizations for both speed and scalability:

1.  **Backtracking with Pruning:** The core of the solver is a backtracking algorithm that explores whether to include or exclude vertices. It uses the simple but effective pruning rule: if the current set plus the remaining candidates cannot exceed the best size found so far, the branch is pruned.
2.  **Degree-based Reductions:**
    *   **Degree-0 Rule:** Vertices with no neighbors in the current subgraph are always included in the Maximum Independent Set.
    *   **Degree-1 Rule:** For a vertex $v$ with exactly one neighbor $u$, it is always optimal to include $v$ and exclude $u$. This significantly reduces the search space for sparse graphs.
3.  **Component Separation:** The graph is divided into its connected components, and the problem is solved independently for each. This is crucial for handling "very large" sparse graphs where components are typically small or have simpler structures.
4.  **Bitset-Accelerated Operations:** A custom bitset implementation (optimized with MSVC intrinsics like `_BitScanForward64` and `__popcnt64`) is used to maintain the candidate set and perform fast vertex iteration and neighbor checks.
5.  **Efficient Degree Maintenance:** The algorithm maintains the degrees of all vertices within the current candidate set dynamically, allowing for $O(1)$ or $O(\text{degree})$ updates and fast selection of high-degree vertices for branching.
6.  **Heuristic Branching:** The solver branches on the vertex with the maximum degree in the current subgraph, which tends to eliminate more candidates in the "include" branch, leading to faster discovery of large independent sets and more effective pruning.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>
#include <intrin.h>

#pragma comment(linker, "/STACK:268435456")

using namespace std;

typedef uint64_t u64;

// Simple bitset for fast vertex set operations
struct Bitset {
    vector<u64> bits;
    Bitset(int n = 0) { if (n > 0) bits.assign((n + 63) / 64, 0); }
    inline void set(int i) { bits[i >> 6] |= (1ULL << (i & 63)); }
    inline void reset(int i) { bits[i >> 6] &= ~(1ULL << (i & 63)); }
    inline bool test(int i) const { return (bits[i >> 6] >> (i & 63)) & 1; }
};

int N, M;
vector<vector<int>> adj;
vector<int> component_is;
int component_max_size;
vector<int> deg;
chrono::steady_clock::time_point start_time;

// Helper to remove a vertex from the candidate set and update neighbor degrees
void remove_from_candidates(int x, Bitset& candidates, int& cand_cnt) {
    candidates.reset(x);
    cand_cnt--;
    for (int n : adj[x]) if (candidates.test(n)) deg[n]--;
}

// Helper to restore a vertex to the candidate set and update neighbor degrees
void restore_to_candidates(int x, Bitset& candidates, int& cand_cnt) {
    for (int n : adj[x]) if (candidates.test(n)) deg[n]++;
    candidates.set(x);
    cand_cnt++;
}

void solve_recursive(Bitset& candidates, int cand_cnt, vector<int>& current, const vector<int>& active_words) {
    // Time limit check to ensure we return a result
    if (chrono::steady_clock::now() - start_time > chrono::milliseconds(9800)) return;
    
    // Base case: no more candidates
    if (cand_cnt == 0) {
        if ((int)current.size() > component_max_size) {
            component_max_size = (int)current.size();
            component_is = current;
        }
        return;
    }
    
    // Pruning: bound check
    if ((int)current.size() + cand_cnt <= component_max_size) return;

    int best_v = -1;
    int max_d = -1;
    int v_deg0 = -1;
    int v_deg1 = -1;

    // Find special cases (deg 0, 1) and identify max degree vertex for branching
    for (int i : active_words) {
        u64 b = candidates.bits[i];
        while (b) {
            unsigned long idx;
            if (_BitScanForward64(&idx, b)) {
                int v = (i << 6) | (int)idx;
                b &= ~(1ULL << idx);

                int d = deg[v];
                if (d == 0) { v_deg0 = v; goto found; }
                if (d == 1) { v_deg1 = v; }
                if (d > max_d) { max_d = d; best_v = v; }
            } else break;
        }
    }

found:
    // Degree-0 Rule: Always include in MIS
    if (v_deg0 != -1) {
        int v = v_deg0;
        remove_from_candidates(v, candidates, cand_cnt);
        current.push_back(v);
        solve_recursive(candidates, cand_cnt, current, active_words);
        current.pop_back();
        restore_to_candidates(v, candidates, cand_cnt);
        return;
    }
    
    // Degree-1 Rule: Always include v and exclude its only neighbor
    if (v_deg1 != -1) {
        int v = v_deg1;
        int u = -1;
        for (int n : adj[v]) if (candidates.test(n)) { u = n; break; }
        remove_from_candidates(v, candidates, cand_cnt);
        bool u_in = (u != -1 && candidates.test(u));
        if (u_in) remove_from_candidates(u, candidates, cand_cnt);
        current.push_back(v);
        solve_recursive(candidates, cand_cnt, current, active_words);
        current.pop_back();
        if (u_in) restore_to_candidates(u, candidates, cand_cnt);
        restore_to_candidates(v, candidates, cand_cnt);
        return;
    }

    if (best_v == -1) return;

    // Branching: Option 1 - Include best_v in MIS
    {
        int v = best_v;
        vector<int> removed;
        removed.push_back(v);
        for (int n : adj[v]) if (candidates.test(n)) removed.push_back(n);
        for (int x : removed) remove_from_candidates(x, candidates, cand_cnt);
        current.push_back(v);
        solve_recursive(candidates, cand_cnt, current, active_words);
        current.pop_back();
        for (int i = (int)removed.size() - 1; i >= 0; --i) restore_to_candidates(removed[i], candidates, cand_cnt);
    }
    
    // Branching: Option 2 - Exclude best_v from MIS
    {
        int v = best_v;
        remove_from_candidates(v, candidates, cand_cnt);
        solve_recursive(candidates, cand_cnt, current, active_words);
        restore_to_candidates(v, candidates, cand_cnt);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> N >> M)) return 0;
    adj.resize(N);
    for (int i = 0; i < M; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u == v || u < 0 || u >= N || v < 0 || v >= N) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    // Deduplicate edges
    for (int i = 0; i < N; ++i) {
        sort(adj[i].begin(), adj[i].end());
        adj[i].erase(unique(adj[i].begin(), adj[i].end()), adj[i].end());
    }

    start_time = chrono::steady_clock::now();
    vector<int> final_is;
    vector<bool> global_visited(N, false);
    deg.resize(N);
    for (int i = 0; i < N; ++i) deg[i] = (int)adj[i].size();

    Bitset candidates(N);
    vector<bool> word_used((N + 63) / 64, false);

    // Separating the graph into connected components
    for (int i = 0; i < N; ++i) {
        if (!global_visited[i]) {
            vector<int> comp;
            vector<int> q;
            q.push_back(i);
            global_visited[i] = true;
            int head = 0;
            while (head < (int)q.size()) {
                int u = q[head++];
                comp.push_back(u);
                for (int v : adj[u]) {
                    if (!global_visited[v]) {
                        global_visited[v] = true;
                        q.push_back(v);
                    }
                }
            }

            if (comp.size() == 1) {
                final_is.push_back(comp[0]);
                continue;
            }

            // Optimize by tracking which bitset words are actually used by this component
            vector<int> active_words;
            for (int v : comp) {
                candidates.set(v);
                int w = v >> 6;
                if (!word_used[w]) {
                    word_used[w] = true;
                    active_words.push_back(w);
                }
            }
            sort(active_words.begin(), active_words.end());
            for (int w : active_words) word_used[w] = false;

            component_max_size = 0;
            component_is.clear();
            vector<int> current;
            solve_recursive(candidates, (int)comp.size(), current, active_words);

            for (int v : component_is) final_is.push_back(v);
            
            // Clean up for the next component
            for (int v : comp) candidates.reset(v);
        }
    }

    // Output results
    cout << final_is.size() << "\n";
    for (int i = 0; i < (int)final_is.size(); ++i) {
        cout << final_is[i] << (i == (int)final_is.size() - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
```
