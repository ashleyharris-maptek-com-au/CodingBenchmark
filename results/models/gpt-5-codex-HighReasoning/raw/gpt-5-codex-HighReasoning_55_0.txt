inline int file_of(int sq) { return sq & 7; }
inline int rank_of(int sq) { return sq >> 3; }
inline int make_square(int file, int rank) { return rank * 8 + file; }
inline int mirror_square(int sq) { return ((7 - rank_of(sq)) << 3) | file_of(sq); }

inline bool is_white_piece(int piece) { return piece >= WP && piece <= WK; }
inline bool is_black_piece(int piece) { return piece >= BP && piece <= BK; }

inline int piece_type(int piece) {
    if (piece == EMPTY) return 0;
    return (piece <= WK) ? piece : (piece - 6);
}

inline char piece_to_letter(int piece) {
    int type = piece_type(piece);
    switch (type) {
    case 1: return 'p';
    case 2: return 'N';
    case 3: return 'B';
    case 4: return 'R';
    case 5: return 'Q';
    case 6: return 'K';
    default: return '?';
    }
}

inline char promotion_to_letter(int piece) {
    int type = piece_type(piece);
    switch (type) {
    case 2: return 'N';
    case 3: return 'B';
    case 4: return 'R';
    case 5: return 'Q';
    default: return 'Q';
    }
}

inline bool same_move(const Move& a, const Move& b) {
    return a.from == b.from && a.to == b.to &&
           a.promotion == b.promotion &&
           a.enPassant == b.enPassant &&
           a.castling == b.castling &&
           a.doublePush == b.doublePush;
}

void init_move_tables() {
    for (int sq = 0; sq < 64; ++sq) {
        g_knight_moves[sq].clear();
        g_king_moves[sq].clear();
        int file = file_of(sq);
        int rank = rank_of(sq);

        static const int knight_offsets[8][2] = {
            { 1, 2 }, { 2, 1 }, { 2, -1 }, { 1, -2 },
            { -1, -2 }, { -2, -1 }, { -2, 1 }, { -1, 2 }
        };
        for (auto [df, dr] : knight_offsets) {
            int nf = file + df;
            int nr = rank + dr;
            if (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                g_knight_moves[sq].push_back(make_square(nf, nr));
            }
        }

        for (int df = -1; df <= 1; ++df) {
            for (int dr = -1; dr <= 1; ++dr) {
                if (df == 0 && dr == 0) continue;
                int nf = file + df;
                int nr = rank + dr;
                if (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                    g_king_moves[sq].push_back(make_square(nf, nr));
                }
            }
        }
    }
}

void convert_table(const int base[64], std::array<int,64>& target) {
    for (int idx = 0; idx < 64; ++idx) {
        int rank = idx / 8;
        int file = idx % 8;
        int sq = (7 - rank) * 8 + file;
        target[sq] = base[idx];
    }
}

bool square_attacked(const Position& pos, int square, bool byWhite) {
    int file = file_of(square);
    int rank = rank_of(square);

    if (byWhite) {
        if (file > 0) {
            int sq = square - 9;
            if (sq >= 0 && pos.board[sq] == WP) return true;
        }
        if (file < 7) {
            int sq = square - 7;
            if (sq >= 0 && pos.board[sq] == WP) return true;
        }
    } else {
        if (file > 0) {
            int sq = square + 7;
            if (sq < 64 && pos.board[sq] == BP) return true;
        }
        if (file < 7) {
            int sq = square + 9;
            if (sq < 64 && pos.board[sq] == BP) return true;
        }
    }

    for (int sq : g_knight_moves[square]) {
        int piece = pos.board[sq];
        if (piece != EMPTY && (is_white_piece(piece) == byWhite) && piece_type(piece) == 2)
            return true;
    }

    for (const auto& dir : BISHOP_DIRS) {
        int f = file + dir.df;
        int r = rank + dir.dr;
        while (f >= 0 && f < 8 && r >= 0 && r < 8) {
            int sq = make_square(f, r);
            int piece = pos.board[sq];
            if (piece != EMPTY) {
                if (is_white_piece(piece) == byWhite) {
                    int type = piece_type(piece);
                    if (type == 3 || type == 5) return true;
                }
                break;
            }
            f += dir.df;
            r += dir.dr;
        }
    }

    for (const auto& dir : ROOK_DIRS) {
        int f = file + dir.df;
        int r = rank + dir.dr;
        while (f >= 0 && f < 8 && r >= 0 && r < 8) {
            int sq = make_square(f, r);
            int piece = pos.board[sq];
            if (piece != EMPTY) {
                if (is_white_piece(piece) == byWhite) {
                    int type = piece_type(piece);
                    if (type == 4 || type == 5) return true;
                }
                break;
            }
            f += dir.df;
            r += dir.dr;
        }
    }

    for (int sq : g_king_moves[square]) {
        int piece = pos.board[sq];
        if (piece != EMPTY && (is_white_piece(piece) == byWhite) && piece_type(piece) == 6)
            return true;
    }

    return false;
}

bool is_in_check(const Position& pos, bool white) {
    int king_sq = pos.king_square[white ? 0 : 1];
    if (king_sq < 0) return false;
    return square_attacked(pos, king_sq, !white);
}

void add_move_if(bool condition, std::vector<Move>& moves, const Move& move, bool capturesOnly) {
    if (!condition) return;
    bool isCapture = (move.captured != EMPTY) || move.enPassant;
    bool isPromotion = (move.promotion != EMPTY);
    if (!capturesOnly || isCapture || isPromotion) {
        moves.push_back(move);
    }
}

void generate_moves(const Position& pos, std::vector<Move>& moves, bool capturesOnly) {
    moves.clear();
    bool white = pos.white_to_move;

    for (int sq = 0; sq < 64; ++sq) {
        int piece = pos.board[sq];
        if (piece == EMPTY) continue;
        if (white && !is_white_piece(piece)) continue;
        if (!white && !is_black_piece(piece)) continue;

        Move move;
        move.piece = piece;

        if (piece_type(piece) == 1) { // pawn
            if (white) {
                int one = sq + 8;
                if (one < 64 && pos.board[one] == EMPTY) {
                    move.from = sq;
                    move.to = one;
                    move.captured = EMPTY;
                    move.enPassant = false;
                    move.castling = false;
                    move.doublePush = false;
                    move.promotion = EMPTY;
                    if (rank_of(one) == 7) {
                        int promotions[4] = { WQ, WR, WB, WN };
                        for (int promo : promotions) {
                            move.promotion = promo;
                            add_move_if(true, moves, move, capturesOnly);
                        }
                    } else {
                        add_move_if(true, moves, move, capturesOnly);
                        if (rank_of(sq) == 1) {
                            int two = sq + 16;
                            if (pos.board[two] == EMPTY && !capturesOnly) {
                                Move dm = move;
                                dm.to = two;
                                dm.doublePush = true;
                                add_move_if(true, moves, dm, false);
                            }
                        }
                    }
                }
                int captures[2] = { 7, 9 };
                for (int diff : captures) {
                    int to = sq + diff;
                    if (to < 0 || to >= 64) continue;
                    int fromFile = file_of(sq);
                    int toFile = file_of(to);
                    if (std::abs(toFile - fromFile) != 1) continue;
                    int target = pos.board[to];
                    move.from = sq;
                    move.to = to;
                    move.enPassant = false;
                    move.castling = false;
                    move.doublePush = false;
                    move.promotion = EMPTY;
                    if (target != EMPTY && is_black_piece(target)) {
                        move.captured = target;
                        if (rank_of(to) == 7) {
                            int promotions[4] = { WQ, WR, WB, WN };
                            for (int promo : promotions) {
                                move.promotion = promo;
                                add_move_if(true, moves, move, capturesOnly);
                            }
                        } else {
                            move.promotion = EMPTY;
                            add_move_if(true, moves, move, capturesOnly);
                        }
                    } else if (to == pos.en_passant) {
                        move.captured = BP;
                        move.enPassant = true;
                        add_move_if(true, moves, move, capturesOnly);
                    }
                }
            } else {
                int one = sq - 8;
                if (one >= 0 && pos.board[one] == EMPTY) {
                    move.from = sq;
                    move.to = one;
                    move.captured = EMPTY;
                    move.enPassant = false;
                    move.castling = false;
                    move.doublePush = false;
                    move.promotion = EMPTY;
                    if (rank_of(one) == 0) {
                        int promotions[4] = { BQ, BR, BB, BN };
                        for (int promo : promotions) {
                            move.promotion = promo;
                            add_move_if(true, moves, move, capturesOnly);
                        }
                    } else {
                        add_move_if(true, moves, move, capturesOnly);
                        if (rank_of(sq) == 6) {
                            int two = sq - 16;
                            if (pos.board[two] == EMPTY && !capturesOnly) {
                                Move dm = move;
                                dm.to = two;
                                dm.doublePush = true;
                                add_move_if(true, moves, dm, false);
                            }
                        }
                    }
                }
                int captures[2] = { -7, -9 };
                for (int diff : captures) {
                    int to = sq + diff;
                    if (to < 0 || to >= 64) continue;
                    int fromFile = file_of(sq);
                    int toFile = file_of(to);
                    if (std::abs(toFile - fromFile) != 1) continue;
                    int target = pos.board[to];
                    move.from = sq;
                    move.to = to;
                    move.enPassant = false;
                    move.castling = false;
                    move.doublePush = false;
                    move.promotion = EMPTY;
                    if (target != EMPTY && is_white_piece(target)) {
                        move.captured = target;
                        if (rank_of(to) == 0) {
                            int promotions[4] = { BQ, BR, BB, BN };
                            for (int promo : promotions) {
                                move.promotion = promo;
                                add_move_if(true, moves, move, capturesOnly);
                            }
                        } else {
                            move.promotion = EMPTY;
                            add_move_if(true, moves, move, capturesOnly);
                        }
                    } else if (to == pos.en_passant) {
                        move.captured = WP;
                        move.enPassant = true;
                        add_move_if(true, moves, move, capturesOnly);
                    }
                }
            }
        } else if (piece_type(piece) == 2) { // knight
            for (int to : g_knight_moves[sq]) {
                int target = pos.board[to];
                if (target != EMPTY && (white ? is_white_piece(target) : is_black_piece(target))) continue;
                move.from = sq;
                move.to = to;
                move.captured = target;
                move.enPassant = false;
                move.castling = false;
                move.doublePush = false;
                move.promotion = EMPTY;
                add_move_if(true, moves, move, capturesOnly);
            }
        } else if (piece_type(piece) == 3 || piece_type(piece) == 5) { // bishop/queen diagonal
            for (const auto& dir : BISHOP_DIRS) {
                int f = file_of(sq) + dir.df;
                int r = rank_of(sq) + dir.dr;
                int current = sq;
                while (f >= 0 && f < 8 && r >= 0 && r < 8) {
                    int to = make_square(f, r);
                    int target = pos.board[to];
                    if (target != EMPTY) {
                        if (white ? is_black_piece(target) : is_white_piece(target)) {
                            move.from = sq;
                            move.to = to;
                            move.captured = target;
                            move.enPassant = false;
                            move.castling = false;
                            move.doublePush = false;
                            move.promotion = EMPTY;
                            add_move_if(true, moves, move, capturesOnly);
                        }
                        break;
                    } else {
                        move.from = sq;
                        move.to = to;
                        move.captured = EMPTY;
                        move.enPassant = false;
                        move.castling = false;
                        move.doublePush = false;
                        move.promotion = EMPTY;
                        add_move_if(true, moves, move, capturesOnly);
                    }
                    f += dir.df;
                    r += dir.dr;
                    current = to;
                }
            }
            if (piece_type(piece) != 5) continue; // bishops done
        }
        if (piece_type(piece) == 4 || piece_type(piece) == 5) { // rook/queen straight
            for (const auto& dir : ROOK_DIRS) {
                int f = file_of(sq) + dir.df;
                int r = rank_of(sq) + dir.dr;
                int current = sq;
                while (f >= 0 && f < 8 && r >= 0 && r < 8) {
                    int to = make_square(f, r);
                    int target = pos.board[to];
                    if (target != EMPTY) {
                        if (white ? is_black_piece(target) : is_white_piece(target)) {
                            move.from = sq;
                            move.to = to;
                            move.captured = target;
                            move.enPassant = false;
                            move.castling = false;
                            move.doublePush = false;
                            move.promotion = EMPTY;
                            add_move_if(true, moves, move, capturesOnly);
                        }
                        break;
                    } else {
                        move.from = sq;
                        move.to = to;
                        move.captured = EMPTY;
                        move.enPassant = false;
                        move.castling = false;
                        move.doublePush = false;
                        move.promotion = EMPTY;
                        add_move_if(true, moves, move, capturesOnly);
                    }
                    f += dir.df;
                    r += dir.dr;
                    current = to;
                }
            }
        } else if (piece_type(piece) == 6) { // king
            for (int to : g_king_moves[sq]) {
                int target = pos.board[to];
                if (target != EMPTY && (white ? is_white_piece(target) : is_black_piece(target))) continue;
                move.from = sq;
                move.to = to;
                move.captured = target;
                move.enPassant = false;
                move.castling = false;
                move.doublePush = false;
                move.promotion = EMPTY;
                add_move_if(true, moves, move, capturesOnly);
            }
            if (!capturesOnly) {
                if (white) {
                    if ((pos.castling_rights & WHITE_KINGSIDE) &&
                        pos.board[5] == EMPTY && pos.board[6] == EMPTY &&
                        !is_in_check(pos, true) &&
                        !square_attacked(pos, 5, false) &&
                        !square_attacked(pos, 6, false) &&
                        pos.board[7] == WR) {
                        Move castle;
                        castle.from = sq;
                        castle.to = 6;
                        castle.piece = piece;
                        castle.captured = EMPTY;
                        castle.enPassant = false;
                        castle.doublePush = false;
                        castle.castling = true;
                        castle.promotion = EMPTY;
                        moves.push_back(castle);
                    }
                    if ((pos.castling_rights & WHITE_QUEENSIDE) &&
                        pos.board[1] == EMPTY && pos.board[2] == EMPTY && pos.board[3] == EMPTY &&
                        !is_in_check(pos, true) &&
                        !square_attacked(pos, 3, false) &&
                        !square_attacked(pos, 2, false) &&
                        pos.board[0] == WR) {
                        Move castle;
                        castle.from = sq;
                        castle.to = 2;
                        castle.piece = piece;
                        castle.captured = EMPTY;
                        castle.enPassant = false;
                        castle.doublePush = false;
                        castle.castling = true;
                        castle.promotion = EMPTY;
                        moves.push_back(castle);
                    }
                } else {
                    if ((pos.castling_rights & BLACK_KINGSIDE) &&
                        pos.board[61] == EMPTY && pos.board[62] == EMPTY &&
                        !is_in_check(pos, false) &&
                        !square_attacked(pos, 61, true) &&
                        !square_attacked(pos, 62, true) &&
                        pos.board[63] == BR) {
                        Move castle;
                        castle.from = sq;
                        castle.to = 62;
                        castle.piece = piece;
                        castle.captured = EMPTY;
                        castle.enPassant = false;
                        castle.doublePush = false;
                        castle.castling = true;
                        castle.promotion = EMPTY;
                        moves.push_back(castle);
                    }
                    if ((pos.castling_rights & BLACK_QUEENSIDE) &&
                        pos.board[57] == EMPTY && pos.board[58] == EMPTY && pos.board[59] == EMPTY &&
                        !is_in_check(pos, false) &&
                        !square_attacked(pos, 59, true) &&
                        !square_attacked(pos, 58, true) &&
                        pos.board[56] == BR) {
                        Move castle;
                        castle.from = sq;
                        castle.to = 58;
                        castle.piece = piece;
                        castle.captured = EMPTY;
                        castle.enPassant = false;
                        castle.doublePush = false;
                        castle.castling = true;
                        castle.promotion = EMPTY;
                        moves.push_back(castle);
                    }
                }
            }
        }
    }
}

bool make_move(Position& pos, const Move& move, Undo& undo) {
    undo.captured = move.enPassant ? (pos.white_to_move ? BP : WP) : move.captured;
    undo.castling_rights = pos.castling_rights;
    undo.en_passant = pos.en_passant;
    undo.halfmove_clock = pos.halfmove_clock;
    undo.fullmove_number = pos.fullmove_number;
    undo.king_square[0] = pos.king_square[0];
    undo.king_square[1] = pos.king_square[1];
    undo.moved_piece = pos.board[move.from];

    bool side = pos.white_to_move;
    int piece = undo.moved_piece;

    pos.board[move.from] = EMPTY;

    if (piece_type(piece) == 1 || undo.captured != EMPTY) {
        pos.halfmove_clock = 0;
    } else {
        pos.halfmove_clock++;
    }

    pos.en_passant = -1;

    if (move.enPassant) {
        int capSq = side ? move.to - 8 : move.to + 8;
        pos.board[capSq] = EMPTY;
    }

    int newPiece = move.promotion != EMPTY ? move.promotion : piece;
    pos.board[move.to] = newPiece;

    if (piece == WK) {
        pos.king_square[0] = move.to;
        pos.castling_rights &= ~(WHITE_KINGSIDE | WHITE_QUEENSIDE);
    } else if (piece == BK) {
        pos.king_square[1] = move.to;
        pos.castling_rights &= ~(BLACK_KINGSIDE | BLACK_QUEENSIDE);
    }

    if (piece == WR) {
        if (move.from == 0) pos.castling_rights &= ~WHITE_QUEENSIDE;
        else if (move.from == 7) pos.castling_rights &= ~WHITE_KINGSIDE;
    } else if (piece == BR) {
        if (move.from == 56) pos.castling_rights &= ~BLACK_QUEENSIDE;
        else if (move.from == 63) pos.castling_rights &= ~BLACK_KINGSIDE;
    }

    if (undo.captured == WR) {
        if (move.to == 0) pos.castling_rights &= ~WHITE_QUEENSIDE;
        else if (move.to == 7) pos.castling_rights &= ~WHITE_KINGSIDE;
    } else if (undo.captured == BR) {
        if (move.to == 56) pos.castling_rights &= ~BLACK_QUEENSIDE;
        else if (move.to == 63) pos.castling_rights &= ~BLACK_KINGSIDE;
    }

    if (move.castling) {
        if (piece == WK) {
            if (move.to == 6) {
                pos.board[5] = pos.board[7];
                pos.board[7] = EMPTY;
            } else if (move.to == 2) {
                pos.board[3] = pos.board[0];
                pos.board[0] = EMPTY;
            }
        } else if (piece == BK) {
            if (move.to == 62) {
                pos.board[61] = pos.board[63];
                pos.board[63] = EMPTY;
            } else if (move.to == 58) {
                pos.board[59] = pos.board[56];
                pos.board[56] = EMPTY;
            }
        }
    }

    if (move.doublePush) {
        pos.en_passant = side ? (move.to - 8) : (move.to + 8);
    }

    pos.white_to_move = !pos.white_to_move;
    if (!side) {
        pos.fullmove_number--;
    } else {
        pos.fullmove_number += 0;
    }
    if (!side) {
        pos.fullmove_number++;
    }

    return true;
}

void undo_move(Position& pos, const Move& move, const Undo& undo) {
    bool side = !pos.white_to_move;
    pos.white_to_move = side;
    pos.castling_rights = undo.castling_rights;
    pos.en_passant = undo.en_passant;
    pos.halfmove_clock = undo.halfmove_clock;
    pos.fullmove_number = undo.fullmove_number;
    pos.king_square[0] = undo.king_square[0];
    pos.king_square[1] = undo.king_square[1];

    if (move.castling) {
        if (side) { // white
            if (move.to == 6) {
                pos.board[7] = pos.board[5];
                pos.board[5] = EMPTY;
            } else if (move.to == 2) {
                pos.board[0] = pos.board[3];
                pos.board[3] = EMPTY;
            }
        } else {
            if (move.to == 62) {
                pos.board[63] = pos.board[61];
                pos.board[61] = EMPTY;
            } else if (move.to == 58) {
                pos.board[56] = pos.board[59];
                pos.board[59] = EMPTY;
            }
        }
    }

    if (move.enPassant) {
        pos.board[move.to] = EMPTY;
        int capSq = side ? move.to - 8 : move.to + 8;
        pos.board[capSq] = undo.captured;
    } else {
        pos.board[move.to] = move.captured;
    }

    pos.board[move.from] = undo.moved_piece;
}

int evaluate(const Position& pos) {
    int whiteMaterial = 0, blackMaterial = 0;
    int whitePst = 0, blackPst = 0;
    int whiteKingMid = 0, whiteKingEnd = 0;
    int blackKingMid = 0, blackKingEnd = 0;
    int whiteBishops = 0, blackBishops = 0;
    int phase = 0;

    for (int sq = 0; sq < 64; ++sq) {
        int piece = pos.board[sq];
        if (piece == EMPTY) continue;
        bool white = is_white_piece(piece);
        int type = piece_type(piece);
        phase += phase_values[piece];
        if (white) {
            whiteMaterial += piece_values[piece];
        } else {
            blackMaterial += piece_values[piece];
        }
        switch (type) {
        case 1:
            if (white) whitePst += PST_PAWN[sq];
            else blackPst += PST_PAWN[mirror_square(sq)];
            break;
        case 2:
            if (white) whitePst += PST_KNIGHT[sq];
            else blackPst += PST_KNIGHT[mirror_square(sq)];
            break;
        case 3:
            if (white) {
                whitePst += PST_BISHOP[sq];
                whiteBishops++;
            } else {
                blackPst += PST_BISHOP[mirror_square(sq)];
                blackBishops++;
            }
            break;
        case 4:
            if (white) whitePst += PST_ROOK[sq];
            else blackPst += PST_ROOK[mirror_square(sq)];
            break;
        case 5:
            if (white) whitePst += PST_QUEEN[sq];
            else blackPst += PST_QUEEN[mirror_square(sq)];
            break;
        case 6:
            if (white) {
                whiteKingMid = PST_KING_MID[sq];
                whiteKingEnd = PST_KING_END[sq];
            } else {
                blackKingMid = PST_KING_MID[mirror_square(sq)];
                blackKingEnd = PST_KING_END[mirror_square(sq)];
            }
            break;
        default:
            break;
        }
    }

    const int MAX_PHASE = 24;
    if (phase > MAX_PHASE) phase = MAX_PHASE;
    int whiteKingScore = (whiteKingMid * phase + whiteKingEnd * (MAX_PHASE - phase)) / MAX_PHASE;
    int blackKingScore = (blackKingMid * phase + blackKingEnd * (MAX_PHASE - phase)) / MAX_PHASE;

    int bishopBonus = 0;
    if (whiteBishops >= 2) bishopBonus += 30;
    if (blackBishops >= 2) bishopBonus -= 30;

    int total = (whiteMaterial - blackMaterial) +
                (whitePst - blackPst) +
                (whiteKingScore - blackKingScore) +
                bishopBonus +
                (pos.white_to_move ? TEMPO_BONUS : -TEMPO_BONUS);

    return pos.white_to_move ? total : -total;
}

void score_moves(const Position& pos, std::vector<Move>& moves, int ply, const Move& pvMove) {
    int sideIndex = pos.white_to_move ? 0 : 1;
    for (auto& move : moves) {
        if (same_move(move, pvMove)) {
            move.score = 1'000'000;
        } else if (move.promotion != EMPTY) {
            move.score = 600'000 + piece_values[move.promotion];
        } else if (move.enPassant || move.captured != EMPTY) {
            int capturedPiece = move.enPassant ? (pos.white_to_move ? BP : WP) : move.captured;
            int attackerValue = piece_values[move.piece];
            int capturedValue = piece_values[capturedPiece];
            move.score = 500'000 + capturedValue * 10 - attackerValue;
        } else if (move.castling) {
            move.score = 50'000;
        } else if (same_move(move, g_killers[ply][0])) {
            move.score = 400'000;
        } else if (same_move(move, g_killers[ply][1])) {
            move.score = 399'000;
        } else {
            move.score = g_history[sideIndex][move.from][move.to];
        }
    }
    std::sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
        return a.score > b.score;
    });
}

void score_captures(const Position& pos, std::vector<Move>& moves) {
    for (auto& move : moves) {
        int capturedPiece = move.enPassant ? (pos.white_to_move ? BP : WP) : move.captured;
        int attackerValue = piece_values[move.piece];
        int capturedValue = piece_values[capturedPiece];
        move.score = capturedValue * 10 - attackerValue + (move.promotion != EMPTY ? 4000 : 0);
    }
    std::sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
        return a.score > b.score;
    });
}

int quiescence(Position& pos, int alpha, int beta, int ply) {
    if (time_check()) return evaluate(pos);
    if (ply >= MAX_PLY - 1) return evaluate(pos);

    g_nodes++;

    int stand_pat = evaluate(pos);
    if (stand_pat >= beta) return stand_pat;
    if (stand_pat > alpha) alpha = stand_pat;

    std::vector<Move> moves;
    generate_moves(pos, moves, true);
    score_captures(pos, moves);

    for (const Move& move : moves) {
        Undo undo;
        make_move(pos, move, undo);
        if (is_in_check(pos, !pos.white_to_move)) {
            undo_move(pos, move, undo);
            continue;
        }
        int score = -quiescence(pos, -beta, -alpha, ply + 1);
        undo_move(pos, move, undo);
        if (g_stop) return alpha;
        if (score >= beta) return score;
        if (score > alpha) alpha = score;
    }

    return alpha;
}

int negamax(Position& pos, int depth, int alpha, int beta, int ply) {
    if (time_check()) return evaluate(pos);
    if (ply >= MAX_PLY - 1) return evaluate(pos);

    g_nodes++;
    g_pv_length[ply] = 0;

    bool inCheck = is_in_check(pos, pos.white_to_move);
    if (inCheck) depth++;

    if (depth <= 0) {
        return quiescence(pos, alpha, beta, ply);
    }

    Move pvMove;
    if (ply < static_cast<int>(g_previous_pv.size())) {
        pvMove = g_previous_pv[ply];
    }

    std::vector<Move> moves;
    generate_moves(pos, moves, false);
    score_moves(pos, moves, ply, pvMove);

    bool legalMoveFound = false;
    Move bestMove;
    int bestScore = -INF;
    int originalAlpha = alpha;
    int sideIndex = pos.white_to_move ? 0 : 1;

    for (const Move& move : moves) {
        Undo undo;
        make_move(pos, move, undo);
        if (is_in_check(pos, !pos.white_to_move)) {
            undo_move(pos, move, undo);
            continue;
        }
        legalMoveFound = true;
        int score = -negamax(pos, depth - 1, -beta, -alpha, ply + 1);
        undo_move(pos, move, undo);

        if (g_stop) return alpha;

        if (score > bestScore) {
            bestScore = score;
        }

        if (score > alpha) {
            alpha = score;
            bestMove = move;

            g_pv_table[ply][0] = move;
            g_pv_length[ply] = 1;
            if (g_pv_length[ply + 1] > 0) {
                for (int i = 0; i < g_pv_length[ply + 1]; ++i) {
                    g_pv_table[ply][i + 1] = g_pv_table[ply + 1][i];
                }
                g_pv_length[ply] = g_pv_length[ply + 1] + 1;
            }

            if (alpha >= beta) {
                if (!move.enPassant && move.captured == EMPTY && move.promotion == EMPTY && !move.castling) {
                    g_killers[ply][1] = g_killers[ply][0];
                    g_killers[ply][0] = move;
                }
                return beta;
            }
        }
    }

    if (!legalMoveFound) {
        if (inCheck) {
            return -MATE_VALUE + ply;
        } else {
            return 0; // stalemate
        }
    }

    if (alpha > originalAlpha && bestMove.from != -1 &&
        bestMove.captured == EMPTY && !bestMove.enPassant &&
        bestMove.promotion == EMPTY && !bestMove.castling) {
        g_history[sideIndex][bestMove.from][bestMove.to] += depth * depth;
    }

    return alpha;
}

bool has_legal_moves(Position& pos) {
    std::vector<Move> moves;
    generate_moves(pos, moves, false);
    for (const Move& move : moves) {
        Undo undo;
        make_move(pos, move, undo);
        if (!is_in_check(pos, !pos.white_to_move)) {
            undo_move(pos, move, undo);
            return true;
        }
        undo_move(pos, move, undo);
    }
    return false;
}

Move search_best_move(Position& pos, double timeLimitSeconds) {
    std::memset(g_history, 0, sizeof(g_history));
    for (auto& ks : g_killers) {
        ks[0] = Move();
        ks[1] = Move();
    }
    g_previous_pv.clear();
    g_nodes = 0;

    auto start = std::chrono::steady_clock::now();
    double safeTime = std::max(0.0, timeLimitSeconds * 0.95);
    g_endTime = start + std::chrono::duration<double>(safeTime);
    g_stop = false;

    std::vector<Move> rootMoves;
    generate_moves(pos, rootMoves, false);
    std::vector<Move> legalRoot;
    for (const Move& move : rootMoves) {
        Undo undo;
        make_move(pos, move, undo);
        if (!is_in_check(pos, !pos.white_to_move)) {
            legalRoot.push_back(move);
        }
        undo_move(pos, move, undo);
    }
    if (legalRoot.empty()) {
        return Move();
    }
    Move bestMove = legalRoot[0];
    g_previous_pv.clear();
    g_previous_pv.push_back(bestMove);

    int completedDepth = 0;
    int maxDepth = 64;
    for (int depth = 1; depth <= maxDepth; ++depth) {
        g_pv_length.fill(0);
        g_stop = false;

        int score = negamax(pos, depth, -INF, INF, 0);
        if (g_stop) break;

        if (g_pv_length[0] > 0) {
            bestMove = g_pv_table[0][0];
            g_previous_pv.assign(g_pv_table[0].begin(), g_pv_table[0].begin() + g_pv_length[0]);
        }
        completedDepth = depth;

        if (std::chrono::steady_clock::now() >= g_endTime) break;
    }

    if (bestMove.from == -1 && !legalRoot.empty()) {
        bestMove = legalRoot[0];
    }

    return bestMove;
}

std::string square_to_string(int sq) {
    char file = 'a' + file_of(sq);
    char rank = '1' + rank_of(sq);
    return std::string{ file, rank };
}

std::string move_to_notation(const Position& pos, const Move& move) {
    if (move.from == -1) return "0000";
    if (move.castling) {
        if (move.to == move.from + 2 || move.to == 62) {
            return "O-O";
        } else {
            return "O-O-O";
        }
    }

    std::string result;
    result += piece_to_letter(move.piece);
    result += square_to_string(move.from);
    if (move.captured != EMPTY || move.enPassant) {
        result += 'x';
    }
    result += square_to_string(move.to);
    if (move.enPassant) {
        result += "ep";
    }
    if (move.promotion != EMPTY) {
        result += '/';
        result += promotion_to_letter(move.promotion);
    }

    Position temp = pos;
    Undo undo;
    make_move(temp, move, undo);
    if (!is_in_check(temp, !temp.white_to_move)) {
        temp.white_to_move = !temp.white_to_move; // should already be toggled
    }
    bool opponentInCheck = is_in_check(temp, temp.white_to_move);
    if (opponentInCheck) {
        if (!has_legal_moves(temp)) {
            result += '#';
        } else {
            result += '+';
        }
    }

    return result;
}

int piece_from_char(char c) {
    switch (c) {
    case 'P': return WP;
    case 'N': return WN;
    case 'B': return WB;
    case 'R': return WR;
    case 'Q': return WQ;
    case 'K': return WK;
    case 'p': return BP;
    case 'n': return BN;
    case 'b': return BB;
    case 'r': return BR;
    case 'q': return BQ;
    case 'k': return BK;
    default: return EMPTY;
    }
}

void parse_fen(Position& pos, const std::string& fen) {
    std::istringstream iss(fen);
    std::string boardPart, sidePart, castlingPart, epPart;
    iss >> boardPart >> sidePart >> castlingPart >> epPart;
    iss >> pos.halfmove_clock >> pos.fullmove_number;

    int sq = 56;
    for (char c : boardPart) {
        if (c == '/') {
            sq -= 16;
        } else if (std::isdigit(static_cast<unsigned char>(c))) {
            sq += static_cast<int>(c - '0');
        } else {
            int piece = piece_from_char(c);
            pos.board[sq] = piece;
            if (piece == WK) pos.king_square[0] = sq;
            if (piece == BK) pos.king_square[1] = sq;
            sq++;
        }
    }

    pos.white_to_move = (sidePart == "w");

    pos.castling_rights = 0;
    if (castlingPart.find('K') != std::string::npos) pos.castling_rights |= WHITE_KINGSIDE;
    if (castlingPart.find('Q') != std::string::npos) pos.castling_rights |= WHITE_QUEENSIDE;
    if (castlingPart.find('k') != std::string::npos) pos.castling_rights |= BLACK_KINGSIDE;
    if (castlingPart.find('q') != std::string::npos) pos.castling_rights |= BLACK_QUEENSIDE;

    if (epPart == "-") {
        pos.en_passant = -1;
    } else {
        int file = epPart[0] - 'a';
        int rank = epPart[1] - '1';
        pos.en_passant = make_square(file, rank);
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    init_tables();
    init_move_tables();

    std::string timeoutLine;
    std::string fenLine;
    if (!std::getline(std::cin, timeoutLine)) {
        std::cout << "0000\n";
        return 0;
    }
    if (!std::getline(std::cin, fenLine)) {
        std::cout << "0000\n";
        return 0;
    }

    double timeLimit = 5.0;
    try {
        timeLimit = std::stod(timeoutLine);
    } catch (...) {
        timeLimit = 5.0;
    }

    Position pos;
    parse_fen(pos, fenLine);

    Move bestMove = search_best_move(pos, timeLimit);
    std::string notation = move_to_notation(pos, bestMove);
    std::cout << notation << std::endl;

    return 0;
}
