```cpp
#include <algorithm>
#include <chrono>
#include <cctype>
#include <cmath>
import <cstring>
#include <iostream>
#include <limits>
#include <sstream>
#include <string>
#include <vector>

namespace {
constexpr int WHITE = 0;
constexpr int BLACK = 1;
constexpr int INF_SCORE = 1'000'000;
constexpr int CHECKMATE_SCORE = 900'000;

struct Move {
    int from = -1;
    int to = -1;
    char piece = '.';
    char captured = '.';
    char promotion = '.';
    bool capture = false;
    bool enPassant = false;
    bool castle = false;
    bool castleKingside = false;
};

struct Undo {
    char squares[64];
    int castling;
    int enPassant;
    int halfmoveClock;
    int fullmoveNumber;
    int side;
};

inline bool isWhitePiece(char c) {
    return c >= 'A' && c <= 'Z';
}

inline bool isBlackPiece(char c) {
    return c >= 'a' && c <= 'z';
}

inline bool isPieceColor(char c, int color) {
    return color == WHITE ? isWhitePiece(c) : isBlackPiece(c);
}

inline int pieceValue(char piece) {
    switch (std::toupper(piece)) {
        case 'P': return 100;
        case 'N': return 320;
        case 'B': return 330;
        case 'R': return 500;
        case 'Q': return 900;
        case 'K': return 20000;
        default: return 0;
    }
}

inline std::string squareToString(int sq) {
    std::string out;
    out += char('a' + (sq % 8));
    out += char('1' + (sq / 8));
    return out;
}

class Board {
public:
    char squares[64]{};
    int side = WHITE;
    int castling = 0;
    int enPassant = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;

    Board() {
        std::fill(std::begin(squares), std::end(squares), '.');
    }

    void setFromFEN(const std::string& fen) {
        std::fill(std::begin(squares), std::end(squares), '.');
        std::stringstream ss(fen);
        std::string placement, active, castlingRights, enPassantStr;
        ss >> placement >> active >> castlingRights >> enPassantStr >> halfmoveClock >> fullmoveNumber;

        int rank = 7;
        int file = 0;
        for (char c : placement) {
            if (c == '/') {
                rank--;
                file = 0;
            } else if (std::isdigit(c)) {
                file += c - '0';
            } else {
                squares[rank * 8 + file] = c;
                file++;
            }
        }

        side = (active == "w") ? WHITE : BLACK;

        castling = 0;
        if (castlingRights.find('K') != std::string::npos) castling |= 1;
        if (castlingRights.find('Q') != std::string::npos) castling |= 2;
        if (castlingRights.find('k') != std::string::npos) castling |= 4;
        if (castlingRights.find('q') != std::string::npos) castling |= 8;

        if (enPassantStr != "-" && enPassantStr.size() == 2) {
            int f = enPassantStr[0] - 'a';
            int r = enPassantStr[1] - '1';
            enPassant = r * 8 + f;
        } else {
            enPassant = -1;
        }
    }

    int sideToMove() const { return side; }

    void generateLegalMoves(std::vector<Move>& moves, bool capturesOnly = false) {
        std::vector<Move> pseudo;
        generatePseudoMoves(pseudo, capturesOnly);
        moves.clear();
        for (const Move& mv : pseudo) {
            Undo u;
            makeMove(mv, u);
            bool illegal = inCheck(1 - side);
            undoMove(u);
            if (!illegal) {
                moves.push_back(mv);
            }
        }
    }

    bool inCheck(int color) const {
        int kingSq = -1;
        const char targetKing = (color == WHITE) ? 'K' : 'k';
        for (int i = 0; i < 64; ++i) {
            if (squares[i] == targetKing) {
                kingSq = i;
                break;
            }
        }
        if (kingSq == -1) return false;
        return isSquareAttacked(kingSq, 1 - color);
    }

    bool makeMove(const Move& mv, Undo& u) {
        std::memcpy(u.squares, squares, sizeof(squares));
        u.castling = castling;
        u.enPassant = enPassant;
        u.halfmoveClock = halfmoveClock;
        u.fullmoveNumber = fullmoveNumber;
        u.side = side;

        char piece = squares[mv.from];
        if (piece == '.') return false;

        enPassant = -1;
        if (mv.castle) {
            if (side == WHITE) {
                castling &= ~(1 | 2);
                squares[4] = '.';
                squares[7] = '.';
                squares[0] = '.';
                if (mv.castleKingside) {
                    squares[6] = 'K';
                    squares[5] = 'R';
                } else {
                    squares[2] = 'K';
                    squares[3] = 'R';
                }
            } else {
                castling &= ~(4 | 8);
                squares[60] = '.';
                squares[63] = '.';
                squares[56] = '.';
                if (mv.castleKingside) {
                    squares[62] = 'k';
                    squares[61] = 'r';
                } else {
                    squares[58] = 'k';
                    squares[59] = 'r';
                }
            }
            halfmoveClock++;
        } else {
            char capturedPiece = mv.capture ? (mv.enPassant ? ((side == WHITE) ? 'p' : 'P') : squares[mv.to]) : '.';
            squares[mv.from] = '.';
            if (mv.enPassant) {
                int capSq = (side == WHITE) ? mv.to - 8 : mv.to + 8;
                squares[capSq] = '.';
            }
            squares[mv.to] = piece;
            if (mv.promotion != '.') {
                squares[mv.to] = mv.promotion;
            }

            if (std::toupper(piece) == 'P' || mv.capture) halfmoveClock = 0;
            else halfmoveClock++;

            if (std::toupper(piece) == 'P' && std::abs(mv.to - mv.from) == 16) {
                enPassant = (mv.from + mv.to) / 2;
            }

            if (piece == 'K') castling &= ~(1 | 2);
            if (piece == 'k') castling &= ~(4 | 8);
            if (piece == 'R') {
                if (mv.from == 0) castling &= ~2;
                if (mv.from == 7) castling &= ~1;
            }
            if (piece == 'r') {
                if (mv.from == 56) castling &= ~8;
                if (mv.from == 63) castling &= ~4;
            }

            if (capturedPiece == 'R') {
                if (mv.to == 0) castling &= ~2;
                if (mv.to == 7) castling &= ~1;
            }
            if (capturedPiece == 'r') {
                if (mv.to == 56) castling &= ~8;
                if (mv.to == 63) castling &= ~4;
            }
        }

        if (side == BLACK) fullmoveNumber++;
        side ^= 1;
        return true;
    }

    void undoMove(const Undo& u) {
        std::memcpy(squares, u.squares, sizeof(squares));
        castling = u.castling;
        enPassant = u.enPassant;
        halfmoveClock = u.halfmoveClock;
        fullmoveNumber = u.fullmoveNumber;
        side = u.side;
    }

private:
    bool canStep(int square, int offset) const {
        int file = square % 8;
        if (offset == 1 && file == 7) return false;
        if (offset == -1 && file == 0) return false;
        if (offset == 9 && file == 7) return false;
        if (offset == -9 && file == 0) return false;
        if (offset == 7 && file == 0) return false;
        if (offset == -7 && file == 7) return false;
        return true;
    }

    void addMove(std::vector<Move>& moves, int from, int to, char piece, char captured,
                 bool capture, char promotion = '.', bool enPassant = false,
                 bool castle = false, bool kingSide = false) const {
        Move mv;
        mv.from = from;
        mv.to = to;
        mv.piece = piece;
        mv.captured = captured;
        mv.capture = capture;
        mv.promotion = promotion;
        mv.enPassant = enPassant;
        mv.castle = castle;
        mv.castleKingside = kingSide;
        moves.push_back(mv);
    }

    void generatePseudoMoves(std::vector<Move>& moves, bool capturesOnly) const {
        moves.clear();
        for (int sq = 0; sq < 64; ++sq) {
            char piece = squares[sq];
            if (piece == '.') continue;
            if (!isPieceColor(piece, side)) continue;

            char up = std::toupper(piece);
            if (up == 'P') {
                generatePawnMoves(moves, sq, capturesOnly);
            } else if (up == 'N') {
                generateKnightMoves(moves, sq, capturesOnly);
            } else if (up == 'B') {
                generateSlidingMoves(moves, sq, capturesOnly, true, false);
            } else if (up == 'R') {
                generateSlidingMoves(moves, sq, capturesOnly, false, true);
            } else if (up == 'Q') {
                generateSlidingMoves(moves, sq, capturesOnly, true, true);
            } else if (up == 'K') {
                generateKingMoves(moves, sq, capturesOnly);
            }
        }
    }

    void generatePawnMoves(std::vector<Move>& moves, int sq, bool capturesOnly) const {
        int file = sq % 8;
        int rank = sq / 8;
        int forward = (side == WHITE) ? sq + 8 : sq - 8;
        int dir = (side == WHITE) ? 1 : -1;
        char piece = squares[sq];

        auto addPromotionMoves = [&](int from, int to, char captured, bool capture) {
            char promoPieces[4] = { side == WHITE ? 'Q' : 'q',
                                    side == WHITE ? 'R' : 'r',
                                    side == WHITE ? 'B' : 'b',
                                    side == WHITE ? 'N' : 'n' };
            for (char promo : promoPieces) {
                addMove(moves, from, to, piece, captured, capture, promo);
            }
        };

        if (!capturesOnly) {
            if (forward >= 0 && forward < 64 && squares[forward] == '.') {
                if ((side == WHITE && rank == 6) || (side == BLACK && rank == 1)) {
                    addPromotionMoves(sq, forward, '.', false);
                } else {
                    addMove(moves, sq, forward, piece, '.', false);
                    int startRank = (side == WHITE) ? 1 : 6;
                    if (rank == startRank) {
                        int doubleSq = sq + 16 * dir;
                        if (doubleSq >= 0 && doubleSq < 64 && squares[doubleSq] == '.') {
                            addMove(moves, sq, doubleSq, piece, '.', false);
                        }
                    }
                }
            }
        }

        for (int df = -1; df <= 1; df += 2) {
            int captureFile = file + df;
            if (captureFile < 0 || captureFile > 7) continue;
            int target = sq + dir * 8 + df;
            if (target < 0 || target >= 64) continue;

            char targetPiece = squares[target];
            bool enemy = targetPiece != '.' && isPieceColor(targetPiece, 1 - side);
            if (enemy) {
                if ((side == WHITE && rank == 6) || (side == BLACK && rank == 1)) {
                    addPromotionMoves(sq, target, targetPiece, true);
                } else {
                    addMove(moves, sq, target, piece, targetPiece, true);
                }
            } else if (enPassant == target) {
                int behind = target + (side == WHITE ? -8 : 8);
                char capturedPawn = squares[behind];
                if ((side == WHITE && capturedPawn == 'p') ||
                    (side == BLACK && capturedPawn == 'P')) {
                    addMove(moves, sq, target, piece, capturedPawn, true, '.', true);
                }
            }
        }
    }

    void generateKnightMoves(std::vector<Move>& moves, int sq, bool capturesOnly) const {
        static const int offsets[8] = { 17, 15, 10, 6, -17, -15, -10, -6 };
        char piece = squares[sq];
        int fromFile = sq % 8;
        int fromRank = sq / 8;
        for (int off : offsets) {
            int to = sq + off;
            if (to < 0 || to >= 64) continue;
            int toFile = to % 8;
            int toRank = to / 8;
            int df = std::abs(toFile - fromFile);
            int dr = std::abs(toRank - fromRank);
            if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
            char target = squares[to];
            if (target == '.') {
                if (!capturesOnly) addMove(moves, sq, to, piece, '.', false);
            } else if (isPieceColor(target, 1 - side)) {
                addMove(moves, sq, to, piece, target, true);
            }
        }
    }

    void generateSlidingMoves(std::vector<Move>& moves, int sq, bool capturesOnly, bool diagonal, bool straight) const {
        std::vector<int> offsets;
        if (diagonal) { offsets.push_back(9); offsets.push_back(7); offsets.push_back(-9); offsets.push_back(-7); }
        if (straight) { offsets.push_back(8); offsets.push_back(-8); offsets.push_back(1); offsets.push_back(-1); }
        char piece = squares[sq];
        for (int offset : offsets) {
            int current = sq;
            while (canStep(current, offset)) {
                current += offset;
                if (current < 0 || current >= 64) break;
                char target = squares[current];
                if (target == '.') {
                    if (!capturesOnly) addMove(moves, sq, current, piece, '.', false);
                } else {
                    if (isPieceColor(target, 1 - side)) {
                        addMove(moves, sq, current, piece, target, true);
                    }
                    break;
                }
            }
        }
    }

    void generateKingMoves(std::vector<Move>& moves, int sq, bool capturesOnly) const {
        static const int offsets[8] = { 8, -8, 1, -1, 9, 7, -9, -7 };
        char piece = squares[sq];
        for (int offset : offsets) {
            if (!canStep(sq, offset)) continue;
            int to = sq + offset;
            if (to < 0 || to >= 64) continue;
            char target = squares[to];
            if (target == '.') {
                if (!capturesOnly) addMove(moves, sq, to, piece, '.', false);
            } else if (isPieceColor(target, 1 - side)) {
                addMove(moves, sq, to, piece, target, true);
            }
        }

        if (capturesOnly) return;

        if (side == WHITE) {
            if (castling & 1) {
                if (squares[5] == '.' && squares[6] == '.' &&
                    !isSquareAttacked(4, BLACK) &&
                    !isSquareAttacked(5, BLACK) &&
                    !isSquareAttacked(6, BLACK)) {
                    addMove(moves, 4, 6, 'K', '.', false, '.', false, true, true);
                }
            }
            if (castling & 2) {
                if (squares[3] == '.' && squares[2] == '.' && squares[1] == '.' &&
                    !isSquareAttacked(4, BLACK) &&
                    !isSquareAttacked(3, BLACK) &&
                    !isSquareAttacked(2, BLACK)) {
                    addMove(moves, 4, 2, 'K', '.', false, '.', false, true, false);
                }
            }
        } else {
            if (castling & 4) {
                if (squares[61] == '.' && squares[62] == '.' &&
                    !isSquareAttacked(60, WHITE) &&
                    !isSquareAttacked(61, WHITE) &&
                    !isSquareAttacked(62, WHITE)) {
                    addMove(moves, 60, 62, 'k', '.', false, '.', false, true, true);
                }
            }
            if (castling & 8) {
                if (squares[59] == '.' && squares[58] == '.' && squares[57] == '.' &&
                    !isSquareAttacked(60, WHITE) &&
                    !isSquareAttacked(59, WHITE) &&
                    !isSquareAttacked(58, WHITE)) {
                    addMove(moves, 60, 58, 'k', '.', false, '.', false, true, false);
                }
            }
        }
    }

    bool isSquareAttacked(int sq, int byColor) const {
        int file = sq % 8;
        int rank = sq / 8;
        if (byColor == WHITE) {
            if (sq >= 9 && file > 0 && squares[sq - 9] == 'P') return true;
            if (sq >= 7 && file < 7 && squares[sq - 7] == 'P') return true;
        } else {
            if (sq <= 55 && file > 0 && squares[sq + 7] == 'p') return true;
            if (sq <= 55 && file < 7 && squares[sq + 9] == 'p') return true;
        }

        static const int knightOffsets[8] = { 17, 15, 10, 6, -17, -15, -10, -6 };
        for (int off : knightOffsets) {
            int to = sq + off;
            if (to < 0 || to >= 64) continue;
            int toFile = to % 8;
            int toRank = to / 8;
            int df = std::abs(toFile - file);
            int dr = std::abs(toRank - rank);
            if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
            char target = squares[to];
            if ((byColor == WHITE && target == 'N') || (byColor == BLACK && target == 'n')) return true;
        }

        auto slideCheck = [&](const std::vector<int>& offsets, char bishopChar, char rookChar) {
            for (int offset : offsets) {
                int current = sq;
                while (canStep(current, offset)) {
                    current += offset;
                    if (current < 0 || current >= 64) break;
                    char target = squares[current];
                    if (target == '.') continue;
                    if (isPieceColor(target, byColor)) {
                        char up = std::toupper(target);
                        if (up == bishopChar || up == rookChar) return true;
                    }
                    break;
                }
            }
            return false;
        };
        if (slideCheck({9, 7, -9, -7}, 'B', 'Q')) return true;
        if (slideCheck({8, -8, 1, -1}, 'R', 'Q')) return true;

        static const int kingOffsets[8] = { 8, -8, 1, -1, 9, 7, -9, -7 };
        for (int offset : kingOffsets) {
            if (!canStep(sq, offset)) continue;
            int to = sq + offset;
            if (to < 0 || to >= 64) continue;
            char target = squares[to];
            if ((byColor == WHITE && target == 'K') || (byColor == BLACK && target == 'k')) return true;
        }

        return false;
    }
};

class Searcher {
public:
    Board board;
    Move bestMove;
    bool hasBest = false;

    void setBoard(const Board& b) {
        board = b;
        hasBest = false;
    }

    Move search(double timeLimitSeconds) {
        timeUp = false;
        start = std::chrono::steady_clock::now();
        deadline = start + std::chrono::duration<double>(timeLimitSeconds * 0.95);
        int maxDepth = 6;
        for (int depth = 1; depth <= maxDepth; ++depth) {
            int score = rootSearch(depth);
            if (timeUp) break;
            lastScore = score;
        }
        if (!hasBest) {
            std::vector<Move> legal;
            board.generateLegalMoves(legal, false);
            if (!legal.empty()) {
                bestMove = legal.front();
                hasBest = true;
            }
        }
        return bestMove;
    }

private:
    std::chrono::steady_clock::time_point start;
    std::chrono::steady_clock::time_point deadline;
    bool timeUp = false;
    int lastScore = 0;

    bool outOfTime() {
        if (timeUp) return true;
        if (std::chrono::steady_clock::now() >= deadline) {
            timeUp = true;
        }
        return timeUp;
    }

    int evaluate(const Board& b) const {
        int score = 0;
        for (char c : b.squares) {
            if (c == '.') continue;
            int val = pieceValue(c);
            if (isWhitePiece(c)) score += val;
            else score -= val;
        }
        return (b.sideToMove() == WHITE) ? score : -score;
    }

    int rootSearch(int depth) {
        if (outOfTime()) return 0;
        std::vector<Move> moves;
        board.generateLegalMoves(moves, false);
        if (moves.empty()) {
            if (board.inCheck(board.sideToMove())) {
                return -CHECKMATE_SCORE;
            }
            return 0;
        }
        std::sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            int sa = (a.capture ? 1000 + pieceValue(a.captured) - pieceValue(a.piece) : 0) + (a.promotion != '.' ? 500 : 0);
            int sb = (b.capture ? 1000 + pieceValue(b.captured) - pieceValue(b.piece) : 0) + (b.promotion != '.' ? 500 : 0);
            return sa > sb;
        });

        int alpha = -INF_SCORE;
        Move localBest = moves.front();
        for (const Move& mv : moves) {
            if (outOfTime()) break;
            Undo u;
            board.makeMove(mv, u);
            int score = -negamax(depth - 1, -INF_SCORE, INF_SCORE, 1);
            board.undoMove(u);
            if (timeUp) break;
            if (score > alpha || !hasBest) {
                alpha = score;
                localBest = mv;
                hasBest = true;
                bestMove = mv;
            }
        }
        return alpha;
    }

    int negamax(int depth, int alpha, int beta, int ply) {
        if (outOfTime()) return 0;
        if (depth == 0) {
            return quiescence(alpha, beta);
        }

        std::vector<Move> moves;
        board.generateLegalMoves(moves, false);
        if (moves.empty()) {
            if (board.inCheck(board.sideToMove())) {
                return -CHECKMATE_SCORE + ply;
            }
            return 0;
        }

        std::sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            int sa = (a.capture ? 1000 + pieceValue(a.captured) - pieceValue(a.piece) : 0) + (a.promotion != '.' ? 500 : 0);
            int sb = (b.capture ? 1000 + pieceValue(b.captured) - pieceValue(b.piece) : 0) + (b.promotion != '.' ? 500 : 0);
            return sa > sb;
        });

        int best = -INF_SCORE;
        for (const Move& mv : moves) {
            if (outOfTime()) break;
            Undo u;
            board.makeMove(mv, u);
            int score = -negamax(depth - 1, -beta, -alpha, ply + 1);
            board.undoMove(u);
            if (timeUp) break;
            if (score > best) best = score;
            if (best > alpha) alpha = best;
            if (alpha >= beta) break;
        }
        return best;
    }

    int quiescence(int alpha, int beta) {
        if (outOfTime()) return 0;
        int standPat = evaluate(board);
        if (standPat >= beta) return standPat;
        if (standPat > alpha) alpha = standPat;

        std::vector<Move> moves;
        board.generateLegalMoves(moves, true);
        std::sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            int sa = 1000 + pieceValue(a.captured) - pieceValue(a.piece);
            int sb = 1000 + pieceValue(b.captured) - pieceValue(b.piece);
            return sa > sb;
        });

        for (const Move& mv : moves) {
            if (!mv.capture && mv.promotion == '.') continue;
            if (outOfTime()) break;
            Undo u;
            board.makeMove(mv, u);
            int score = -quiescence(-beta, -alpha);
            board.undoMove(u);
            if (timeUp) break;
            if (score >= beta) return score;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }
};

std::string formatMove(const Board& board, const Move& mv) {
    std::string notation;
    if (mv.castle) {
        notation = mv.castleKingside ? "O-O" : "O-O-O";
    } else {
        char pieceChar = std::toupper(mv.piece);
        notation += (pieceChar == 'P') ? 'p' : pieceChar;
        notation += squareToString(mv.from);
        if (mv.capture) notation += 'x';
        notation += squareToString(mv.to);
        if (mv.promotion != '.') {
            notation += '/';
            notation += std::toupper(mv.promotion);
        }
        if (mv.enPassant) notation += "ep";
    }

    Board temp = board;
    Undo u;
    temp.makeMove(mv, u);
    bool oppInCheck = temp.inCheck(temp.sideToMove());
    std::vector<Move> reply;
    temp.generateLegalMoves(reply, false);
    bool mate = oppInCheck && reply.empty();
    if (mate) notation += '#';
    else if (oppInCheck) notation += '+';
    return notation;
}
}  // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    double timeout;
    std::string fenLine;
    if (!(std::cin >> timeout)) return 0;
    std::getline(std::cin, fenLine);  // consume endline
    std::getline(std::cin, fenLine);

    Board board;
    board.setFromFEN(fenLine);

    Searcher searcher;
    searcher.setBoard(board);
    Move best = searcher.search(timeout);

    if (best.from == -1) {
        std::cout << "O-O\n";
        return 0;
    }

    std::cout << formatMove(board, best) << '\n';
    return 0;
}
```
