inline bool is_white(char p) {
    return p >= 'A' && p <= 'Z';
}

inline bool is_black(char p) {
    return p >= 'a' && p <= 'z';
}

int mirror_square(int sq) {
    int row = sq / 8;
    int col = sq % 8;
    int mirroredRow = 7 - row;
    return mirroredRow * 8 + col;
}

void init_timer(double limitSeconds) {
    using namespace std::chrono;
    g_startTime = steady_clock::now();
    double budget = std::max(0.1, limitSeconds * 0.95);
    g_deadline = g_startTime + duration<double>(budget);
    g_timeExpired = false;
}

bool check_time() {
    if (g_timeExpired) return true;
    if (std::chrono::steady_clock::now() >= g_deadline) {
        g_timeExpired = true;
    }
    return g_timeExpired;
}

int piece_value(char p) {
    switch (std::toupper(static_cast<unsigned char>(p))) {
        case 'P': return 100;
        case 'N': return 320;
        case 'B': return 330;
        case 'R': return 500;
        case 'Q': return 900;
        case 'K': return 20000;
        default: return 0;
    }
}

int evaluate(const Board& board) {
    int score = 0;
    int whiteBishops = 0;
    int blackBishops = 0;

    for (int sq = 0; sq < 64; ++sq) {
        char piece = board.squares[sq];
        if (piece == '.') continue;

        bool white = is_white(piece);
        char upper = std::toupper(static_cast<unsigned char>(piece));
        int base = piece_value(piece);
        int pst = 0;
        int mirrored = mirror_square(sq);

        switch (upper) {
            case 'P':
                pst = white ? PAWN_PST[sq] : PAWN_PST[mirrored];
                break;
            case 'N':
                pst = white ? KNIGHT_PST[sq] : KNIGHT_PST[mirrored];
                break;
            case 'B':
                pst = white ? BISHOP_PST[sq] : BISHOP_PST[mirrored];
                break;
            case 'R':
                pst = white ? ROOK_PST[sq] : ROOK_PST[mirrored];
                break;
            case 'Q':
                pst = white ? QUEEN_PST[sq] : QUEEN_PST[mirrored];
                break;
            case 'K':
                pst = white ? KING_PST[sq] : KING_PST[mirrored];
                break;
            default:
                break;
        }

        if (white) {
            score += base + pst;
            if (upper == 'B') ++whiteBishops;
        } else {
            score -= base + pst;
            if (upper == 'B') ++blackBishops;
        }
    }

    if (whiteBishops >= 2) score += 30;
    if (blackBishops >= 2) score -= 30;

    score += board.whiteToMove ? 10 : -10;

    return score;
}

bool parse_fen(const std::string& fen, Board& board) {
    std::istringstream iss(fen);
    std::string boardPart, sidePart, castlingPart, epPart;
    if (!(iss >> boardPart >> sidePart >> castlingPart >> epPart)) {
        return false;
    }

    board.squares.fill('.');
    int row = 7;
    int col = 0;
    for (char ch : boardPart) {
        if (ch == '/') {
            --row;
            col = 0;
            continue;
        }
        if (std::isdigit(static_cast<unsigned char>(ch))) {
            col += ch - '0';
        } else {
            if (row < 0 || col >= 8) return false;
            board.squares[row * 8 + col] = ch;
            ++col;
        }
    }

    board.whiteToMove = (sidePart == "w");

    board.castleWK = board.castleWQ = board.castleBK = board.castleBQ = false;
    if (castlingPart != "-") {
        for (char c : castlingPart) {
            switch (c) {
                case 'K': board.castleWK = true; break;
                case 'Q': board.castleWQ = true; break;
                case 'k': board.castleBK = true; break;
                case 'q': board.castleBQ = true; break;
                default: break;
            }
        }
    }

    if (epPart != "-") {
        int file = epPart[0] - 'a';
        int rank = epPart[1] - '1';
        if (file >= 0 && file < 8 && rank >= 0 && rank < 8) {
            board.enPassant = rank * 8 + file;
        } else {
            board.enPassant = -1;
        }
    } else {
        board.enPassant = -1;
    }

    if (!(iss >> board.halfmoveClock)) board.halfmoveClock = 0;
    if (!(iss >> board.fullmoveNumber)) board.fullmoveNumber = 1;

    return true;
}

bool in_check(const Board& board, bool whiteKing) {
    char kingChar = whiteKing ? 'K' : 'k';
    int kingSq = -1;
    for (int i = 0; i < 64; ++i) {
        if (board.squares[i] == kingChar) {
            kingSq = i;
            break;
        }
    }
    if (kingSq == -1) return false;
    return square_attacked(board, kingSq, !whiteKing);
}

void apply_move(Board& board, const Move& move) {
    char piece = board.squares[move.from];
    bool movingWhite = is_white(piece);

    board.enPassant = -1;
    if (std::toupper(static_cast<unsigned char>(piece)) == 'P' || move.isCapture) {
        board.halfmoveClock = 0;
    } else {
        board.halfmoveClock++;
    }

    board.squares[move.from] = '.';
    if (move.isEnPassant) {
        int capSq = movingWhite ? move.to - 8 : move.to + 8;
        board.squares[capSq] = '.';
    }

    char placed = (move.promotion != 0) ? move.promotion : piece;
    board.squares[move.to] = placed;

    if (std::toupper(static_cast<unsigned char>(piece)) == 'P' &&
        std::abs(move.to - move.from) == 16) {
        board.enPassant = movingWhite ? move.from + 8 : move.from - 8;
    }

    if (piece == 'K') {
        board.castleWK = board.castleWQ = false;
    } else if (piece == 'k') {
        board.castleBK = board.castleBQ = false;
    }

    if (piece == 'R') {
        if (move.from == 0) board.castleWQ = false;
        else if (move.from == 7) board.castleWK = false;
    } else if (piece == 'r') {
        if (move.from == 56) board.castleBQ = false;
        else if (move.from == 63) board.castleBK = false;
    }

    if (move.isCapture && !move.isEnPassant) {
        if (move.captured == 'R') {
            if (move.to == 0) board.castleWQ = false;
            else if (move.to == 7) board.castleWK = false;
        } else if (move.captured == 'r') {
            if (move.to == 56) board.castleBQ = false;
            else if (move.to == 63) board.castleBK = false;
        }
    }

    if (move.isCastle) {
        if (piece == 'K') {
            if (move.to == 6) {
                board.squares[5] = 'R';
                board.squares[7] = '.';
            } else if (move.to == 2) {
                board.squares[3] = 'R';
                board.squares[0] = '.';
            }
        } else if (piece == 'k') {
            if (move.to == 62) {
                board.squares[61] = 'r';
                board.squares[63] = '.';
            } else if (move.to == 58) {
                board.squares[59] = 'r';
                board.squares[56] = '.';
            }
        }
    }

    if (!movingWhite) {
        board.fullmoveNumber++;
    }

    board.whiteToMove = !movingWhite;
}

void push_move(std::vector<Move>& moves, int from, int to, char piece,
               char captured, char promotion = 0, bool enPassant = false,
               bool castle = false) {
    Move m;
    m.from = from;
    m.to = to;
    m.piece = piece;
    m.captured = captured;
    m.promotion = promotion;
    m.isEnPassant = enPassant;
    m.isCastle = castle;
    m.isCapture = (captured != '.' || enPassant);
    moves.push_back(m);
}

void generate_pseudo_moves(const Board& board, std::vector<Move>& moves) {
    static const int knightOffsets[8][2] = {
        {1, 2}, {2, 1}, {2, -1}, {1, -2},
        {-1, -2}, {-2, -1}, {-2, 1}, {-1, 2}
    };
    static const int bishopDirs[4][2] = {{1,1},{1,-1},{-1,1},{-1,-1}};
    static const int rookDirs[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};

    bool white = board.whiteToMove;
    for (int sq = 0; sq < 64; ++sq) {
        char piece = board.squares[sq];
        if (piece == '.') continue;
        if (is_white(piece) != white) continue;

        int row = sq / 8;
        int col = sq % 8;
        char upper = std::toupper(static_cast<unsigned char>(piece));

        if (upper == 'P') {
            if (white) {
                int forwardRow = row + 1;
                if (forwardRow < 8) {
                    int forwardSq = forwardRow * 8 + col;
                    if (board.squares[forwardSq] == '.') {
                        if (forwardRow == 7) {
                            for (char promo : {'Q','R','B','N'}) {
                                push_move(moves, sq, forwardSq, piece, '.', promo);
                            }
                        } else {
                            push_move(moves, sq, forwardSq, piece, '.');
                            if (row == 1) {
                                int doubleSq = (row + 2) * 8 + col;
                                if (board.squares[doubleSq] == '.') {
                                    push_move(moves, sq, doubleSq, piece, '.');
                                }
                            }
                        }
                    }
                }
                if (row + 1 < 8 && col > 0) {
                    int to = (row + 1) * 8 + (col - 1);
                    char target = board.squares[to];
                    if (is_black(target)) {
                        if (row + 1 == 7) {
                            for (char promo : {'Q','R','B','N'}) {
                                push_move(moves, sq, to, piece, target, promo);
                            }
                        } else {
                            push_move(moves, sq, to, piece, target);
                        }
                    }
                }
                if (row + 1 < 8 && col < 7) {
                    int to = (row + 1) * 8 + (col + 1);
                    char target = board.squares[to];
                    if (is_black(target)) {
                        if (row + 1 == 7) {
                            for (char promo : {'Q','R','B','N'}) {
                                push_move(moves, sq, to, piece, target, promo);
                            }
                        } else {
                            push_move(moves, sq, to, piece, target);
                        }
                    }
                }
                if (board.enPassant != -1) {
                    int targetRow = row + 1;
                    if (targetRow < 8) {
                        if (col > 0 && board.enPassant == targetRow * 8 + (col - 1)) {
                            push_move(moves, sq, board.enPassant, piece, 'p', 0, true);
                        }
                        if (col < 7 && board.enPassant == targetRow * 8 + (col + 1)) {
                            push_move(moves, sq, board.enPassant, piece, 'p', 0, true);
                        }
                    }
                }
            } else {
                int forwardRow = row - 1;
                if (forwardRow >= 0) {
                    int forwardSq = forwardRow * 8 + col;
                    if (board.squares[forwardSq] == '.') {
                        if (forwardRow == 0) {
                            for (char promo : {'q','r','b','n'}) {
                                push_move(moves, sq, forwardSq, piece, '.', promo);
                            }
                        } else {
                            push_move(moves, sq, forwardSq, piece, '.');
                            if (row == 6) {
                                int doubleSq = (row - 2) * 8 + col;
                                if (board.squares[doubleSq] == '.') {
                                    push_move(moves, sq, doubleSq, piece, '.');
                                }
                            }
                        }
                    }
                }
                if (row - 1 >= 0 && col > 0) {
                    int to = (row - 1) * 8 + (col - 1);
                    char target = board.squares[to];
                    if (is_white(target)) {
                        if (row - 1 == 0) {
                            for (char promo : {'q','r','b','n'}) {
                                push_move(moves, sq, to, piece, target, promo);
                            }
                        } else {
                            push_move(moves, sq, to, piece, target);
                        }
                    }
                }
                if (row - 1 >= 0 && col < 7) {
                    int to = (row - 1) * 8 + (col + 1);
                    char target = board.squares[to];
                    if (is_white(target)) {
                        if (row - 1 == 0) {
                            for (char promo : {'q','r','b','n'}) {
                                push_move(moves, sq, to, piece, target, promo);
                            }
                        } else {
                            push_move(moves, sq, to, piece, target);
                        }
                    }
                }
                if (board.enPassant != -1) {
                    int targetRow = row - 1;
                    if (targetRow >= 0) {
                        if (col > 0 && board.enPassant == targetRow * 8 + (col - 1)) {
                            push_move(moves, sq, board.enPassant, piece, 'P', 0, true);
                        }
                        if (col < 7 && board.enPassant == targetRow * 8 + (col + 1)) {
                            push_move(moves, sq, board.enPassant, piece, 'P', 0, true);
                        }
                    }
                }
            }
        } else if (upper == 'N') {
            for (const auto& off : knightOffsets) {
                int r = row + off[0];
                int c = col + off[1];
                if (r < 0 || r >= 8 || c < 0 || c >= 8) continue;
                int to = r * 8 + c;
                char target = board.squares[to];
                if (target == '.' || is_white(target) != white) {
                    push_move(moves, sq, to, piece, target);
                }
            }
        } else if (upper == 'B' || upper == 'Q') {
            for (const auto& dir : bishopDirs) {
                int r = row + dir[0];
                int c = col + dir[1];
                while (r >= 0 && r < 8 && c >= 0 && c < 8) {
                    int to = r * 8 + c;
                    char target = board.squares[to];
                    if (target == '.') {
                        push_move(moves, sq, to, piece, '.');
                    } else {
                        if (is_white(target) != white) {
                            push_move(moves, sq, to, piece, target);
                        }
                        break;
                    }
                    r += dir[0];
                    c += dir[1];
                }
            }
        }

        if (upper == 'R' || upper == 'Q') {
            for (const auto& dir : rookDirs) {
                int r = row + dir[0];
                int c = col + dir[1];
                while (r >= 0 && r < 8 && c >= 0 && c < 8) {
                    int to = r * 8 + c;
                    char target = board.squares[to];
                    if (target == '.') {
                        push_move(moves, sq, to, piece, '.');
                    } else {
                        if (is_white(target) != white) {
                            push_move(moves, sq, to, piece, target);
                        }
                        break;
                    }
                    r += dir[0];
                    c += dir[1];
                }
            }
        }

        if (upper == 'K') {
            for (int dr = -1; dr <= 1; ++dr) {
                for (int dc = -1; dc <= 1; ++dc) {
                    if (dr == 0 && dc == 0) continue;
                    int r = row + dr;
                    int c = col + dc;
                    if (r < 0 || r >= 8 || c < 0 || c >= 8) continue;
                    int to = r * 8 + c;
                    char target = board.squares[to];
                    if (target == '.' || is_white(target) != white) {
                        push_move(moves, sq, to, piece, target);
                    }
                }
            }

            if (white) {
                if (board.castleWK) {
                    bool clear = board.squares[5] == '.' && board.squares[6] == '.';
                    if (clear && !square_attacked(board, 4, false) &&
                        !square_attacked(board, 5, false) &&
                        !square_attacked(board, 6, false)) {
                        push_move(moves, sq, 6, piece, '.', 0, false, true);
                    }
                }
                if (board.castleWQ) {
                    bool clear = board.squares[3] == '.' && board.squares[2] == '.' && board.squares[1] == '.';
                    if (clear && !square_attacked(board, 4, false) &&
                        !square_attacked(board, 3, false) &&
                        !square_attacked(board, 2, false)) {
                        push_move(moves, sq, 2, piece, '.', 0, false, true);
                    }
                }
            } else {
                if (board.castleBK) {
                    bool clear = board.squares[61] == '.' && board.squares[62] == '.';
                    if (clear && !square_attacked(board, 60, true) &&
                        !square_attacked(board, 61, true) &&
                        !square_attacked(board, 62, true)) {
                        push_move(moves, sq, 62, piece, '.', 0, false, true);
                    }
                }
                if (board.castleBQ) {
                    bool clear = board.squares[59] == '.' && board.squares[58] == '.' && board.squares[57] == '.';
                    if (clear && !square_attacked(board, 60, true) &&
                        !square_attacked(board, 59, true) &&
                        !square_attacked(board, 58, true)) {
                        push_move(moves, sq, 58, piece, '.', 0, false, true);
                    }
                }
            }
        }
    }
}

std::vector<Move> generate_legal_moves(const Board& board) {
    std::vector<Move> pseudo;
    pseudo.reserve(64);
    generate_pseudo_moves(board, pseudo);
    std::vector<Move> legal;
    legal.reserve(pseudo.size());
    bool movingWhite = board.whiteToMove;

    for (const Move& move : pseudo) {
        Board next = board;
        apply_move(next, move);
        if (!in_check(next, movingWhite)) {
            legal.push_back(move);
        }
    }
    return legal;
}

bool square_attacked(const Board& board, int square, bool byWhite) {
    int row = square / 8;
    int col = square % 8;
    if (byWhite) {
        if (row > 0 && col > 0 && board.squares[(row - 1) * 8 + (col - 1)] == 'P') return true;
        if (row > 0 && col < 7 && board.squares[(row - 1) * 8 + (col + 1)] == 'P') return true;
    } else {
        if (row < 7 && col > 0 && board.squares[(row + 1) * 8 + (col - 1)] == 'p') return true;
        if (row < 7 && col < 7 && board.squares[(row + 1) * 8 + (col + 1)] == 'p') return true;
    }

    static const int knightOffsets[8][2] = {
        {1, 2}, {2, 1}, {2, -1}, {1, -2},
        {-1, -2}, {-2, -1}, {-2, 1}, {-1, 2}
    };
    for (const auto& off : knightOffsets) {
        int r = row + off[0];
        int c = col + off[1];
        if (r < 0 || r >= 8 || c < 0 || c >= 8) continue;
        char piece = board.squares[r * 8 + c];
        if (piece != '.' && is_white(piece) == byWhite &&
            std::toupper(static_cast<unsigned char>(piece)) == 'N') {
            return true;
        }
    }

    static const int bishopDirs[4][2] = {{1,1},{1,-1},{-1,1},{-1,-1}};
    for (const auto& dir : bishopDirs) {
        int r = row + dir[0];
        int c = col + dir[1];
        while (r >= 0 && r < 8 && c >= 0 && c < 8) {
            char piece = board.squares[r * 8 + c];
            if (piece != '.') {
                if (is_white(piece) == byWhite) {
                    char up = std::toupper(static_cast<unsigned char>(piece));
                    if (up == 'B' || up == 'Q') return true;
                }
                break;
            }
            r += dir[0];
            c += dir[1];
        }
    }

    static const int rookDirs[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for (const auto& dir : rookDirs) {
        int r = row + dir[0];
        int c = col + dir[1];
        while (r >= 0 && r < 8 && c >= 0 && c < 8) {
            char piece = board.squares[r * 8 + c];
            if (piece != '.') {
                if (is_white(piece) == byWhite) {
                    char up = std::toupper(static_cast<unsigned char>(piece));
                    if (up == 'R' || up == 'Q') return true;
                }
                break;
            }
            r += dir[0];
            c += dir[1];
        }
    }

    for (int dr = -1; dr <= 1; ++dr) {
        for (int dc = -1; dc <= 1; ++dc) {
            if (dr == 0 && dc == 0) continue;
            int r = row + dr;
            int c = col + dc;
            if (r < 0 || r >= 8 || c < 0 || c >= 8) continue;
            char piece = board.squares[r * 8 + c];
            if (piece != '.' && is_white(piece) == byWhite &&
                std::toupper(static_cast<unsigned char>(piece)) == 'K') {
                return true;
            }
        }
    }

    return false;
}

void order_moves(std::vector<Move>& moves) {
    for (auto& move : moves) {
        int score = 0;
        if (move.isCapture) {
            score = 100000 + piece_value(move.captured) * 10 - piece_value(move.piece);
        }
        if (move.promotion) {
            score += 90000 + piece_value(move.promotion);
        }
        if (move.isCastle) score += 500;
        move.score = score;
    }
    std::stable_sort(moves.begin(), moves.end(),
                     [](const Move& a, const Move& b) { return a.score > b.score; });
}

int search_root(Board& board, int depth, Move& bestMove, bool& completed) {
    auto moves = generate_legal_moves(board);
    if (moves.empty()) {
        completed = true;
        if (in_check(board, board.whiteToMove)) {
            return -MATE_VALUE + depth;
        }
        return 0;
    }
    order_moves(moves);
    int alpha = -INF_SCORE;
    int beta = INF_SCORE;
    int bestScore = -INF_SCORE;
    Move localBest = moves[0];
    completed = true;

    for (const auto& move : moves) {
        if (check_time()) {
            completed = false;
            break;
        }
        Board next = board;
        apply_move(next, move);
        int score = -negamax(next, depth - 1, -beta, -alpha);
        if (g_timeExpired) {
            completed = false;
            break;
        }
        if (score > bestScore) {
            bestScore = score;
            localBest = move;
        }
        if (score > alpha) alpha = score;
    }

    if (completed) {
        bestMove = localBest;
    }
    return bestScore;
}

int negamax(Board& board, int depth, int alpha, int beta) {
    if (check_time()) return 0;
    if (depth == 0) {
        return evaluate(board);
    }

    auto moves = generate_legal_moves(board);
    if (moves.empty()) {
        if (in_check(board, board.whiteToMove)) {
            return -MATE_VALUE + depth;
        }
        return 0;
    }

    order_moves(moves);
    int best = -INF_SCORE;

    for (const auto& move : moves) {
        if (g_timeExpired) break;
        Board next = board;
        apply_move(next, move);
        int score = -negamax(next, depth - 1, -beta, -alpha);
        if (g_timeExpired) return 0;
        if (score > best) best = score;
        if (best > alpha) alpha = best;
        if (alpha >= beta) break;
    }
    return best;
}

std::string square_to_string(int sq) {
    std::string s;
    s += static_cast<char>('a' + (sq % 8));
    s += static_cast<char>('1' + (sq / 8));
    return s;
}

std::string move_to_string(const Board& board, const Move& move) {
    if (move.isCastle) {
        if (move.to % 8 == 6) return "O-O";
        return "O-O-O";
    }
    std::string res;
    char pieceLetter = std::toupper(static_cast<unsigned char>(move.piece));
    if (pieceLetter == 'P') res += 'p';
    else res += pieceLetter;

    res += square_to_string(move.from);
    if (move.isCapture) res += 'x';
    res += square_to_string(move.to);

    if (move.promotion) {
        res += '/';
        res += static_cast<char>(std::toupper(static_cast<unsigned char>(move.promotion)));
    }
    if (move.isEnPassant) {
        res += "ep";
    }

    Board next = board;
    apply_move(next, move);
    if (in_check(next, next.whiteToMove)) {
        auto reply = generate_legal_moves(next);
        if (reply.empty()) res += '#';
        else res += '+';
    }

    return res;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::string timeLine;
    if (!std::getline(std::cin, timeLine)) {
        return 0;
    }
    while (!timeLine.empty() && (timeLine.back() == '\r' || timeLine.back() == '\n')) {
        timeLine.pop_back();
    }
    while (timeLine.empty() && std::getline(std::cin, timeLine)) {
        while (!timeLine.empty() && (timeLine.back() == '\r' || timeLine.back() == '\n')) {
            timeLine.pop_back();
        }
    }

    std::string fenLine;
    if (!std::getline(std::cin, fenLine)) {
        return 0;
    }
    while (!fenLine.empty() && (fenLine.back() == '\r' || fenLine.back() == '\n')) {
        fenLine.pop_back();
    }
    while (fenLine.empty() && std::getline(std::cin, fenLine)) {
        while (!fenLine.empty() && (fenLine.back() == '\r' || fenLine.back() == '\n')) {
            fenLine.pop_back();
        }
    }

    double timeoutSeconds = 5.0;
    {
        std::stringstream ss(timeLine);
        ss >> timeoutSeconds;
        if (timeoutSeconds <= 0) timeoutSeconds = 5.0;
    }

    Board board;
    if (!parse_fen(fenLine, board)) {
        std::cout << "O-O\n";
        return 0;
    }

    init_timer(timeoutSeconds);

    auto legalMoves = generate_legal_moves(board);
    if (legalMoves.empty()) {
        std::cout << "O-O\n";
        return 0;
    }

    Move bestMove = legalMoves[0];

    int maxDepth = 5;
    if (timeoutSeconds > 8) maxDepth = 6;
    if (timeoutSeconds > 15) maxDepth = 7;
    if (timeoutSeconds > 30) maxDepth = 8;

    for (int depth = 1; depth <= maxDepth; ++depth) {
        if (g_timeExpired) break;
        Move candidate;
        bool completed = false;
        search_root(board, depth, candidate, completed);
        if (g_timeExpired || !completed) break;
        bestMove = candidate;
    }

    std::cout << move_to_string(board, bestMove) << "\n";
    return 0;
}
