inline uint32_t encode_move(int from, int to, int piece, int captured = -1, int promo = -1, int flags = 0) {
    return (from) | (to << 6) | (piece << 12) |
           ((captured + 1) << 16) | ((promo + 1) << 20) | ((flags & 0x3F) << 24);
}

inline int move_from(Move m) { return m.data & 63; }

inline int move_capture(Move m) {
    int v = (m.data >> 16) & 15;

}
inline int move_promo(Move m) {
    int v = (m.data >> 20) & 15;
    return v ? v - 1 : -1;
}
inline int move_flags(Move m) { return (m.data >> 24) & 0x3F; }

uint64_t rook_attacks(int sq, uint64_t occ) {
    uint64_t attacks = 0;
    int rank = sq / 8;
    int file = sq % 8;
    for (int r = rank + 1; r < 8; ++r) {
        int s = r * 8 + file;
        attacks |= bit(s);
        if (occ & bit(s)) break;
    }
    for (int r = rank - 1; r >= 0; --r) {
        int s = r * 8 + file;
        attacks |= bit(s);
        if (occ & bit(s)) break;
    }
    for (int f = file + 1; f < 8; ++f) {
        int s = rank * 8 + f;
        attacks |= bit(s);
        if (occ & bit(s)) break;
    }
    for (int f = file - 1; f >= 0; --f) {
        int s = rank * 8 + f;
        attacks |= bit(s);
        if (occ & bit(s)) break;
    }
    return attacks;
}

uint64_t bishop_attacks(int sq, uint64_t occ) {
    uint64_t attacks = 0;
    int rank = sq / 8;
    int file = sq % 8;
    for (int r = rank + 1, f = file + 1; r < 8 && f < 8; ++r, ++f) {
        int s = r * 8 + f;
        attacks |= bit(s);
        if (occ & bit(s)) break;
    }
    for (int r = rank + 1, f = file - 1; r < 8 && f >= 0; ++r, --f) {
        int s = r * 8 + f;
        attacks |= bit(s);
        if (occ & bit(s)) break;
    }
    for (int r = rank - 1, f = file + 1; r >= 0 && f < 8; --r, ++f) {
        int s = r * 8 + f;
        attacks |= bit(s);
        if (occ & bit(s)) break;
    }
    for (int r = rank - 1, f = file - 1; r >= 0 && f >= 0; --r, --f) {
        int s = r * 8 + f;
        attacks |= bit(s);
        if (occ & bit(s)) break;
    }
    return attacks;
}

void init_attack_tables() {
    for (int sq = 0; sq < 64; ++sq) {
        knightAttacks[sq] = 0;
        kingAttacks[sq] = 0;
        int rank = sq / 8;
        int file = sq % 8;
        constexpr int knightMoves[8][2] = {
            {1, 2}, {2, 1}, {2, -1}, {1, -2},
            {-1, -2}, {-2, -1}, {-2, 1}, {-1, 2}
        };
        for (auto [df, dr] : knightMoves) {
            int nf = file + df;
            int nr = rank + dr;
            if (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                knightAttacks[sq] |= bit(nr * 8 + nf);
            }
        }
        for (int dr = -1; dr <= 1; ++dr) {
            for (int df = -1; df <= 1; ++df) {
                if (dr == 0 && df == 0) continue;
                int nf = file + df;
                int nr = rank + dr;
                if (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                    kingAttacks[sq] |= bit(nr * 8 + nf);
                }
            }
        }
    }
}

    for (int i = 0; i < 8; ++i) zobristEnPassant[i] = dist(rng);
    zobristSide = dist(rng);
}

class Board {
public:
    array<uint64_t, 12> pieces{};
    array<int, 64> boardSquares{};
    uint64_t occWhite = 0, occBlack = 0, occAll = 0;
    int sideToMove = WHITE;
    int castlingRights = 0;
    int enPassantSquare = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
    int kingSquare[2] = {-1, -1};
    uint64_t zobrist = 0;
    vector<uint64_t> history;

    Board() {
        boardSquares.fill(-1);
        history.reserve(1024);
    }

    void clear() {
        pieces.fill(0);
        boardSquares.fill(-1);
        occWhite = occBlack = occAll = 0;
        sideToMove = WHITE;
        castlingRights = 0;
        enPassantSquare = -1;
        halfmoveClock = 0;
        fullmoveNumber = 1;
        kingSquare[WHITE] = kingSquare[BLACK] = -1;
        zobrist = 0;
        history.clear();
    }

    void update_occupancies() {
        occWhite = pieces[WP] | pieces[WN] | pieces[WB] | pieces[WR] | pieces[WQ] | pieces[WK];
        occBlack = pieces[BP] | pieces[BN] | pieces[BB] | pieces[BR] | pieces[BQ] | pieces[BK];
        occAll = occWhite | occBlack;
    }

    void load_fen(const string& fen) {
        clear();
        string boardPart, sidePart, castlingPart, epPart;
        int halfmove = 0, fullmove = 1;
        stringstream ss(fen);
        ss >> boardPart >> sidePart >> castlingPart >> epPart >> halfmove >> fullmove;
        int rank = 7, file = 0;
        for (char c : boardPart) {
            if (c == '/') {
                rank--;
                file = 0;
                continue;
            }
            if (isdigit(c)) {
                file += c - '0';
                continue;
            }
            int piece = -1;
            switch (c) {
                case 'P': piece = WP; break;
                case 'N': piece = WN; break;
                case 'B': piece = WB; break;
                case 'R': piece = WR; break;
                case 'Q': piece = WQ; break;
                case 'K': piece = WK; break;
                case 'p': piece = BP; break;
                case 'n': piece = BN; break;
                case 'b': piece = BB; break;
                case 'r': piece = BR; break;
                case 'q': piece = BQ; break;
                case 'k': piece = BK; break;
            }
            if (piece != -1) {
                int sq = rank * 8 + file;
                pieces[piece] |= bit(sq);
                boardSquares[sq] = piece;
                if (piece == WK) kingSquare[WHITE] = sq;
                if (piece == BK) kingSquare[BLACK] = sq;
            }
            file++;
        }
        sideToMove = (sidePart == "w") ? WHITE : BLACK;
        castlingRights = 0;
        if (castlingPart.find('K') != string::npos) castlingRights |= WHITE_OO;
        if (castlingPart.find('Q') != string::npos) castlingRights |= WHITE_OOO;
        if (castlingPart.find('k') != string::npos) castlingRights |= BLACK_OO;
        if (castlingPart.find('q') != string::npos) castlingRights |= BLACK_OOO;
        enPassantSquare = -1;
        if (epPart != "-") {
            int fileEp = epPart[0] - 'a';
            int rankEp = epPart[1] - '1';
            enPassantSquare = rankEp * 8 + fileEp;
        }
        halfmoveClock = halfmove;
        fullmoveNumber = fullmove;
        update_occupancies();
        zobrist = 0;
        for (int p = 0; p < 12; ++p) {
            uint64_t bb = pieces[p];
            while (bb) {
                int sq = pop_lsb(bb);
                zobrist ^= zobristPiece[p][sq];
            }
        }
        if (sideToMove == BLACK) zobrist ^= zobristSide;
        zobrist ^= zobristCastling[castlingRights];
        if (enPassantSquare != -1) zobrist ^= zobristEnPassant[enPassantSquare % 8];
        history.push_back(zobrist);
    }

    bool make_move(Move move, State& st) {
        apply_move(move, st);
        int mover = sideToMove;
        int opponent = mover ^ 1;
        if (is_square_attacked(kingSquare[mover], opponent)) {
            undo_move_effects(move, st);
            return false;
        }
        sideToMove = opponent;
        zobrist ^= zobristSide;
        history.push_back(zobrist);
        return true;
    }

    void undo_move(Move move, const State& st) {
        if (!history.empty()) history.pop_back();
        sideToMove ^= 1;
        undo_move_effects(move, st);
    }

    void generate_moves(vector<Move>& moves, bool capturesOnly) const {
        moves.clear();
        int color = sideToMove;
        uint64_t ownOcc = (color == WHITE) ? occWhite : occBlack;
        uint64_t oppOcc = (color == WHITE) ? occBlack : occWhite;
        uint64_t occ = occAll;
        const int pawnIndex = (color == WHITE) ? WP : BP;
        const int knightIndex = (color == WHITE) ? WN : BN;
        const int bishopIndex = (color == WHITE) ? WB : BB;
        const int rookIndex = (color == WHITE) ? WR : BR;
        const int queenIndex = (color == WHITE) ? WQ : BQ;
        const int kingIndex = (color == WHITE) ? WK : BK;
        uint64_t pawns = pieces[pawnIndex];
        const int promoPiecesWhite[4] = {WQ, WR, WB, WN};
        const int promoPiecesBlack[4] = {BQ, BR, BB, BN};
        while (pawns) {
            int from = pop_lsb(pawns);
            int rank = from / 8;
            int file = from % 8;
            if (color == WHITE) {
                int one = from + 8;
                if (!capturesOnly && one < 64 && !(occ & bit(one))) {
                    bool promotion = one >= 56;
                    if (promotion) {
                        for (int promo : promoPiecesWhite) {
                            moves.emplace_back(encode_move(from, one, pawnIndex, -1, promo, FLAG_PROMOTION));
                        }
                    } else {
                        moves.emplace_back(encode_move(from, one, pawnIndex, -1, -1, 0));
                        int two = from + 16;
                        if (rank == 1 && !(occ & bit(two))) {
                            moves.emplace_back(encode_move(from, two, pawnIndex, -1, -1, FLAG_DOUBLE));
                        }
                    }
                }
                for (int df : {-1, 1}) {
                    int toFile = file + df;
                    if (toFile < 0 || toFile >= 8) continue;
                    int to = from + (df == -1 ? 7 : 9);
                    bool isPromotion = to >= 56;
                    bool isCapture = (oppOcc & bit(to)) != 0;
                    bool isEP = (to == enPassantSquare);
                    if (!isCapture && !isEP) continue;
                    int flags = FLAG_CAPTURE;
                    if (isEP) flags |= FLAG_ENPASSANT;
                    int captured = isEP ? BP : boardSquares[to];
                    if (capturesOnly && !isPromotion && !isCapture && !isEP) continue;
                    if (isPromotion) {
                        for (int promo : promoPiecesWhite) {
                            moves.emplace_back(encode_move(from, to, pawnIndex, captured, promo, flags | FLAG_PROMOTION));
                        }
                    } else {
                        moves.emplace_back(encode_move(from, to, pawnIndex, captured, -1, flags));
                    }
                }
            } else {
                int one = from - 8;
                if (!capturesOnly && one >= 0 && !(occ & bit(one))) {
                    bool promotion = one <= 7;
                    if (promotion) {
                        for (int promo : promoPiecesBlack) {
                            moves.emplace_back(encode_move(from, one, pawnIndex, -1, promo, FLAG_PROMOTION));
                        }
                    } else {
                        moves.emplace_back(encode_move(from, one, pawnIndex, -1, -1, 0));
                        int two = from - 16;
                        if (rank == 6 && !(occ & bit(two))) {
                            moves.emplace_back(encode_move(from, two, pawnIndex, -1, -1, FLAG_DOUBLE));
                        }
                    }
                }
                for (int df : {-1, 1}) {
                    int toFile = file + df;
                    if (toFile < 0 || toFile >= 8) continue;
                    int to = from - (df == -1 ? 9 : 7);
                    bool isPromotion = to <= 7;
                    bool isCapture = (oppOcc & bit(to)) != 0;
                    bool isEP = (to == enPassantSquare);
                    if (!isCapture && !isEP) continue;
                    int flags = FLAG_CAPTURE;
                    if (isEP) flags |= FLAG_ENPASSANT;
                    int captured = isEP ? WP : boardSquares[to];
                    if (capturesOnly && !isPromotion && !isCapture && !isEP) continue;
                    if (isPromotion) {
                        for (int promo : promoPiecesBlack) {
                            moves.emplace_back(encode_move(from, to, pawnIndex, captured, promo, flags | FLAG_PROMOTION));
                        }
                    } else {
                        moves.emplace_back(encode_move(from, to, pawnIndex, captured, -1, flags));
                    }
                }
            }
        }
        auto add_piece_moves = [&](uint64_t bb, const uint64_t* attackTable = nullptr) {
            while (bb) {
                int from = pop_lsb(bb);
                uint64_t targets;
                if (attackTable) targets = attackTable[from] & ~ownOcc;
                else targets = 0;
                if (!attackTable) targets = 0;
            }
        };
        uint64_t knights = pieces[knightIndex];
        while (knights) {
            int from = pop_lsb(knights);
            uint64_t movesMask = knightAttacks[from] & ~ownOcc;
            while (movesMask) {
                int to = pop_lsb(movesMask);
                bool capture = (oppOcc & bit(to)) != 0;
                if (capturesOnly && !capture) continue;
                int flags = capture ? FLAG_CAPTURE : 0;
                int captured = capture ? boardSquares[to] : -1;
                moves.emplace_back(encode_move(from, to, knightIndex, captured, -1, flags));
            }
        }
        uint64_t bishops = pieces[bishopIndex];
        while (bishops) {
            int from = pop_lsb(bishops);
            uint64_t movesMask = bishop_attacks(from, occ) & ~ownOcc;
            while (movesMask) {
                int to = pop_lsb(movesMask);
                bool capture = (oppOcc & bit(to)) != 0;
                if (capturesOnly && !capture) continue;
                int flags = capture ? FLAG_CAPTURE : 0;
                int captured = capture ? boardSquares[to] : -1;
                moves.emplace_back(encode_move(from, to, bishopIndex, captured, -1, flags));
            }
        }
        uint64_t rooks = pieces[rookIndex];
        while (rooks) {
            int from = pop_lsb(rooks);
            uint64_t movesMask = rook_attacks(from, occ) & ~ownOcc;
            while (movesMask) {
                int to = pop_lsb(movesMask);
                bool capture = (oppOcc & bit(to)) != 0;
                if (capturesOnly && !capture) continue;
                int flags = capture ? FLAG_CAPTURE : 0;
                int captured = capture ? boardSquares[to] : -1;
                moves.emplace_back(encode_move(from, to, rookIndex, captured, -1, flags));
            }
        }
        uint64_t queens = pieces[queenIndex];
        while (queens) {
            int from = pop_lsb(queens);
            uint64_t movesMask = (rook_attacks(from, occ) | bishop_attacks(from, occ)) & ~ownOcc;
            while (movesMask) {
                int to = pop_lsb(movesMask);
                bool capture = (oppOcc & bit(to)) != 0;
                if (capturesOnly && !capture) continue;
                int flags = capture ? FLAG_CAPTURE : 0;
                int captured = capture ? boardSquares[to] : -1;
                moves.emplace_back(encode_move(from, to, queenIndex, captured, -1, flags));
            }
        }
        uint64_t king = pieces[kingIndex];
        if (king) {
            int from = std::countr_zero(king);
            uint64_t movesMask = kingAttacks[from] & ~ownOcc;
            while (movesMask) {
                int to = pop_lsb(movesMask);
                bool capture = (oppOcc & bit(to)) != 0;
                if (capturesOnly && !capture) continue;
                int flags = capture ? FLAG_CAPTURE : 0;
                int captured = capture ? boardSquares[to] : -1;
                moves.emplace_back(encode_move(from, to, kingIndex, captured, -1, flags));
            }
            if (!capturesOnly) {
                if (color == WHITE) {
                    if ((castlingRights & WHITE_OO) &&
                        boardSquares[5] == -1 && boardSquares[6] == -1 &&
                        !is_square_attacked(4, BLACK) &&
                        !is_square_attacked(5, BLACK) &&
                        !is_square_attacked(6, BLACK)) {
                        moves.emplace_back(encode_move(4, 6, WK, -1, -1, FLAG_KING_CASTLE));
                    }
                    if ((castlingRights & WHITE_OOO) &&
                        boardSquares[1] == -1 && boardSquares[2] == -1 && boardSquares[3] == -1 &&
                        !is_square_attacked(4, BLACK) &&
                        !is_square_attacked(3, BLACK) &&
                        !is_square_attacked(2, BLACK)) {
                        moves.emplace_back(encode_move(4, 2, WK, -1, -1, FLAG_QUEEN_CASTLE));
                    }
                } else {
                    if ((castlingRights & BLACK_OO) &&
                        boardSquares[61] == -1 && boardSquares[62] == -1 &&
                        !is_square_attacked(60, WHITE) &&
                        !is_square_attacked(61, WHITE) &&
                        !is_square_attacked(62, WHITE)) {
                        moves.emplace_back(encode_move(60, 62, BK, -1, -1, FLAG_KING_CASTLE));
                    }
                    if ((castlingRights & BLACK_OOO) &&
                        boardSquares[57] == -1 && boardSquares[58] == -1 && boardSquares[59] == -1 &&
                        !is_square_attacked(60, WHITE) &&
                        !is_square_attacked(59, WHITE) &&
                        !is_square_attacked(58, WHITE)) {
                        moves.emplace_back(encode_move(60, 58, BK, -1, -1, FLAG_QUEEN_CASTLE));
                    }
                }
            }
        }
    }

    bool is_square_attacked(int sq, int byColor) const {
        if (sq < 0) return false;
        uint64_t mask = bit(sq);
        if (byColor == WHITE) {
            uint64_t pawns = pieces[WP];
            uint64_t attacks = ((pawns & NOT_FILE_H) << 7) | ((pawns & NOT_FILE_A) << 9);
            if (attacks & mask) return true;
            if (knightAttacks[sq] & pieces[WN]) return true;
            if (kingAttacks[sq] & pieces[WK]) return true;
            uint64_t bishopsQueens = pieces[WB] | pieces[WQ];
            if (bishop_attacks(sq, occAll) & bishopsQueens) return true;
            uint64_t rooksQueens = pieces[WR] | pieces[WQ];
            if (rook_attacks(sq, occAll) & rooksQueens) return true;
        } else {
            uint64_t pawns = pieces[BP];
            uint64_t attacks = ((pawns & NOT_FILE_A) >> 7) | ((pawns & NOT_FILE_H) >> 9);
            if (attacks & mask) return true;
            if (knightAttacks[sq] & pieces[BN]) return true;
            if (kingAttacks[sq] & pieces[BK]) return true;
            uint64_t bishopsQueens = pieces[BB] | pieces[BQ];
            if (bishop_attacks(sq, occAll) & bishopsQueens) return true;
            uint64_t rooksQueens = pieces[BR] | pieces[BQ];
            if (rook_attacks(sq, occAll) & rooksQueens) return true;
        }
        return false;
    }

    bool in_check(int color) const {
        return is_square_attacked(kingSquare[color], color ^ 1);
    }

    bool is_repetition() const {
        if (history.size() < 3) return false;
        uint64_t current = history.back();
        int repeats = 0;
        for (int i = int(history.size()) - 1; i >= 0 && (history.size() - 1 - i) <= halfmoveClock; i -= 2) {
            if (history[i] == current) {
                repeats++;
                if (repeats >= 3) return true;
            }
        }
        return false;
    }

    int evaluate() const {
        int score = 0;
        for (int piece = 0; piece < 12; ++piece) {
            uint64_t bb = pieces[piece];
            if (!bb) continue;
            int color = piece_color(piece);
            int type = piece_type(piece);
            uint64_t temp = bb;
            while (temp) {
                int sq = pop_lsb(temp);
                if (color == WHITE) {
                    score += evalPieceValues[type] + pieceSquareTable[type][sq];
                } else {
                    score -= evalPieceValues[type] + pieceSquareTable[type][mirror_square(sq)];
                }
            }
        }
        if (std::popcount(pieces[WB]) >= 2) score += 30;
        if (std::popcount(pieces[BB]) >= 2) score -= 30;
        score += (sideToMove == WHITE) ? 10 : -10;
        return score;
    }

private:
    void apply_move(Move move, State& st) {
        st.castlingRights = castlingRights;
        st.enPassantSquare = enPassantSquare;
        st.halfmoveClock = halfmoveClock;
        st.fullmoveNumber = fullmoveNumber;
        st.zobrist = zobrist;
        st.capturedPiece = -1;
        int from = move_from(move);
        int to = move_to(move);
        int piece = move_piece(move);
        int promo = move_promo(move);
        int flags = move_flags(move);
        int color = piece_color(piece);

        if (enPassantSquare != -1) {
            zobrist ^= zobristEnPassant[enPassantSquare % 8];
        }
        enPassantSquare = -1;

        pieces[piece] &= ~bit(from);
        zobrist ^= zobristPiece[piece][from];
        boardSquares[from] = -1;

        halfmoveClock++;
        if (piece == WP || piece == BP) halfmoveClock = 0;

        if (flags & FLAG_ENPASSANT) {
            int capSq = (color == WHITE) ? to - 8 : to + 8;
            int capPiece = (color == WHITE) ? BP : WP;
            pieces[capPiece] &= ~bit(capSq);
            zobrist ^= zobristPiece[capPiece][capSq];
            boardSquares[capSq] = -1;
            st.capturedPiece = capPiece;
            halfmoveClock = 0;
        } else if (move_capture(move) != -1) {
            int capPiece = move_capture(move);
            pieces[capPiece] &= ~bit(to);
            zobrist ^= zobristPiece[capPiece][to];
            boardSquares[to] = -1;
            st.capturedPiece = capPiece;
            halfmoveClock = 0;
        }

        int placed = piece;
        if (flags & FLAG_PROMOTION) {
            placed = promo;
        }
        pieces[placed] |= bit(to);
        zobrist ^= zobristPiece[placed][to];
        boardSquares[to] = placed;

        if (piece == WK) kingSquare[WHITE] = to;
        else if (piece == BK) kingSquare[BLACK] = to;

        if (flags & FLAG_KING_CASTLE) {
            if (color == WHITE) {
                pieces[WR] &= ~bit(7);
                pieces[WR] |= bit(5);
                zobrist ^= zobristPiece[WR][7];
                zobrist ^= zobristPiece[WR][5];
                boardSquares[7] = -1;
                boardSquares[5] = WR;
            } else {
                pieces[BR] &= ~bit(63);
                pieces[BR] |= bit(61);
                zobrist ^= zobristPiece[BR][63];
                zobrist ^= zobristPiece[BR][61];
                boardSquares[63] = -1;
                boardSquares[61] = BR;
            }
        } else if (flags & FLAG_QUEEN_CASTLE) {
            if (color == WHITE) {
                pieces[WR] &= ~bit(0);
                pieces[WR] |= bit(3);
                zobrist ^= zobristPiece[WR][0];
                zobrist ^= zobristPiece[WR][3];
                boardSquares[0] = -1;
                boardSquares[3] = WR;
            } else {
                pieces[BR] &= ~bit(56);
                pieces[BR] |= bit(59);
                zobrist ^= zobristPiece[BR][56];
                zobrist ^= zobristPiece[BR][59];
                boardSquares[56] = -1;
                boardSquares[59] = BR;
            }
        }

        int newCastling = castlingRights;
        if (piece == WK) newCastling &= ~(WHITE_OO | WHITE_OOO);
        if (piece == BK) newCastling &= ~(BLACK_OO | BLACK_OOO);
        if (piece == WR) {
            if (from == 7) newCastling &= ~WHITE_OO;
            else if (from == 0) newCastling &= ~WHITE_OOO;
        }
        if (piece == BR) {
            if (from == 63) newCastling &= ~BLACK_OO;
            else if (from == 56) newCastling &= ~BLACK_OOO;
        }
        if (st.capturedPiece == WR) {
            if (to == 7) newCastling &= ~WHITE_OO;
            else if (to == 0) newCastling &= ~WHITE_OOO;
        }
        if (st.capturedPiece == BR) {
            if (to == 63) newCastling &= ~BLACK_OO;
            else if (to == 56) newCastling &= ~BLACK_OOO;
        }
        if (newCastling != castlingRights) {
            zobrist ^= zobristCastling[castlingRights];
            zobrist ^= zobristCastling[newCastling];
            castlingRights = newCastling;
        }

        if (flags & FLAG_DOUBLE) {
            enPassantSquare = (color == WHITE) ? (to - 8) : (to + 8);
            zobrist ^= zobristEnPassant[enPassantSquare % 8];
        }

        if (color == BLACK) ++fullmoveNumber;

        update_occupancies();
    }

    void undo_move_effects(Move move, const State& st) {
        int from = move_from(move);
        int to = move_to(move);
        int piece = move_piece(move);
        int promo = move_promo(move);
        int flags = move_flags(move);
        int color = sideToMove;

        if (flags & FLAG_PROMOTION) {
            int promoPiece = promo;
            pieces[promoPiece] &= ~bit(to);
            boardSquares[to] = -1;
            pieces[piece] |= bit(from);
            boardSquares[from] = piece;
        } else {
            pieces[piece] &= ~bit(to);
            pieces[piece] |= bit(from);
            boardSquares[to] = -1;
            boardSquares[from] = piece;
        }

        if (piece == WK) kingSquare[WHITE] = from;
        else if (piece == BK) kingSquare[BLACK] = from;

        if (flags & FLAG_KING_CASTLE) {
            if (color == WHITE) {
                pieces[WR] &= ~bit(5);
                pieces[WR] |= bit(7);
                boardSquares[5] = -1;
                boardSquares[7] = WR;
            } else {
                pieces[BR] &= ~bit(61);
                pieces[BR] |= bit(63);
                boardSquares[61] = -1;
                boardSquares[63] = BR;
            }
        } else if (flags & FLAG_QUEEN_CASTLE) {
            if (color == WHITE) {
                pieces[WR] &= ~bit(3);
                pieces[WR] |= bit(0);
                boardSquares[3] = -1;
                boardSquares[0] = WR;
            } else {
                pieces[BR] &= ~bit(59);
                pieces[BR] |= bit(56);
                boardSquares[59] = -1;
                boardSquares[56] = BR;
            }
        }

        if (st.capturedPiece != -1) {
            int capSq = (flags & FLAG_ENPASSANT)
                            ? ((color == WHITE) ? to - 8 : to + 8)
                            : to;
            pieces[st.capturedPiece] |= bit(capSq);
            boardSquares[capSq] = st.capturedPiece;
        }

        castlingRights = st.castlingRights;
        enPassantSquare = st.enPassantSquare;
        halfmoveClock = st.halfmoveClock;
        fullmoveNumber = st.fullmoveNumber;
        zobrist = st.zobrist;
        update_occupancies();
    }
};

struct TTEntry {
    uint64_t key = 0;
    int16_t score = 0;
    int8_t depth = -1;
    uint8_t flag = 0;
    uint32_t move = 0;
};

enum TTFlag { TT_EMPTY, TT_EXACT, TT_LOWER, TT_UPPER };

class Search {
public:
    explicit Search(Board& b) : board(b), tt(TT_SIZE) {
        for (auto& plane : history)
            for (auto& row : plane)
                std::fill(row.begin(), row.end(), 0);
    }

    Move search_best_move(double timeLimitSeconds) {
        nodes = 0;
        stop = false;
        auto now = chrono::steady_clock::now();
        auto limit = chrono::duration_cast<chrono::steady_clock::duration>(chrono::duration<double>(timeLimitSeconds));
        endTime = now + limit;
        Move best;
        for (int depth = 1; depth <= 64; ++depth) {
            rootBestMove = Move();
            int score = alphabeta(depth, -INF_SCORE, INF_SCORE, 0, true);
            if (stop) break;
            if (rootBestMove.data != 0) {
                best = rootBestMove;
            }
        }
        if (best.data == 0) {
            vector<Move> moves;
            board.generate_moves(moves, false);
            for (const Move& m : moves) {
                State st;
                if (board.make_move(m, st)) {
                    best = m;
                    board.undo_move(m, st);
                    break;
                }
            }
        }
        return best;
    }

private:
    static constexpr size_t TT_BITS = 20;
    static constexpr size_t TT_SIZE = 1ull << TT_BITS;
    static constexpr uint64_t TT_MASK = TT_SIZE - 1;

    Board& board;
    vector<TTEntry> tt;
    Move rootBestMove;
    uint64_t nodes = 0;
    bool stop = false;
    chrono::steady_clock::time_point endTime;
    array<array<array<int, 64>, 64>, 2> history{};

    bool time_exceeded() {
        if (chrono::steady_clock::now() >= endTime) {
            stop = true;
            return true;
        }
        return false;
    }

    int alphabeta(int depth, int alpha, int beta, int ply, bool root) {
        if (stop) return 0;
        if ((nodes & 0x3FF) == 0 && time_exceeded()) return 0;
        nodes++;
        if (ply >= MAX_PLY - 1) return board.evaluate();
        bool inCheck = board.in_check(board.sideToMove);
        if (board.is_repetition() || board.halfmoveClock >= 100) return 0;
        if (inCheck) depth++;
        if (depth <= 0) return quiescence(alpha, beta, ply);

        uint64_t key = board.zobrist;
        TTEntry& entry = tt[key & TT_MASK];
        Move ttMove;
        if (entry.key == key) {
            int ttScore = entry.score;
            if (ttScore > MATE_THRESHOLD) ttScore -= ply;
            else if (ttScore < -MATE_THRESHOLD) ttScore += ply;
            ttMove = Move(entry.move);
            if (entry.depth >= depth) {
                if (entry.flag == TT_EXACT) return ttScore;
                if (entry.flag == TT_LOWER && ttScore >= beta) return ttScore;
                if (entry.flag == TT_UPPER && ttScore <= alpha) return ttScore;
            }
        }

        vector<Move> pseudo;
        board.generate_moves(pseudo, false);
        if (pseudo.empty()) {
            return inCheck ? (-MATE_VALUE + ply) : 0;
        }

        struct MoveScore { Move move; int score; };
        vector<MoveScore> ordered;
        ordered.reserve(pseudo.size());
        int color = board.sideToMove;
        for (const Move& m : pseudo) {
            int score = 0;
            if (ttMove.data != 0 && m.data == ttMove.data) {
                score = 1'000'000;
            } else {
                int flags = move_flags(m);
                if (flags & FLAG_CAPTURE) {
                    int captured = move_capture(m);
                    score = 500'000 + mvvLva[captured][move_piece(m)];
                } else if (flags & FLAG_PROMOTION) {
                    score = 400'000 + pieceValues[piece_type(move_promo(m))];
                } else {
                    score = history[color][move_from(m)][move_to(m)];
                }
            }
            ordered.push_back({m, score});
        }
        sort(ordered.begin(), ordered.end(), [](const MoveScore& a, const MoveScore& b) {
            return a.score > b.score;
        });

        bool hasLegal = false;
        Move bestLocal;
        int bestScore = -INF_SCORE;
        int alphaOrig = alpha;

        for (const auto& ms : ordered) {
            Move move = ms.move;
            State st;
            if (!board.make_move(move, st)) continue;
            hasLegal = true;
            int score = -alphabeta(depth - 1, -beta, -alpha, ply + 1, false);
            board.undo_move(move, st);
            if (stop) return 0;
            if (score > bestScore) {
                bestScore = score;
                bestLocal = move;
            }
            if (score >= beta) {
                if (!(move_flags(move) & (FLAG_CAPTURE | FLAG_PROMOTION | FLAG_KING_CASTLE | FLAG_QUEEN_CASTLE))) {
                    int& hist = history[color][move_from(move)][move_to(move)];
                    hist += depth * depth;
                    hist = min(hist, 2000000);
                }
                store_tt(key, depth, beta, TT_LOWER, move, ply);
                return beta;
            }
            if (score > alpha) {
                alpha = score;
                if (!(move_flags(move) & (FLAG_CAPTURE | FLAG_PROMOTION | FLAG_KING_CASTLE | FLAG_QUEEN_CASTLE))) {
                    int& hist = history[color][move_from(move)][move_to(move)];
                    hist += depth * depth;
                    hist = min(hist, 2000000);
                }
                if (root) rootBestMove = move;
            }
        }

        if (!hasLegal) {
            return inCheck ? (-MATE_VALUE + ply) : 0;
        }

        if (!stop) {
            int flag = (alpha <= alphaOrig) ? TT_UPPER : TT_EXACT;
            store_tt(key, depth, alpha, flag, bestLocal, ply);
        }
        return alpha;
    }

    int quiescence(int alpha, int beta, int ply) {
        if (stop) return 0;
        if ((nodes & 0x3FF) == 0 && time_exceeded()) return 0;
        nodes++;
        if (board.is_repetition() || board.halfmoveClock >= 100) return 0;
        int stand = board.evaluate();
        if (stand >= beta) return stand;
        if (stand > alpha) alpha = stand;

        vector<Move> moves;
        board.generate_moves(moves, true);
        struct MoveScore { Move move; int score; };
        vector<MoveScore> ordered;
        ordered.reserve(moves.size());
        for (const Move& m : moves) {
            int score = 0;
            if (move_flags(m) & FLAG_CAPTURE) {
                int captured = move_capture(m);
                score = mvvLva[captured][move_piece(m)];
            } else if (move_flags(m) & FLAG_PROMOTION) {
                score = 10000 + pieceValues[piece_type(move_promo(m))];
            }
            ordered.push_back({m, score});
        }
        sort(ordered.begin(), ordered.end(), [](const MoveScore& a, const MoveScore& b) {
            return a.score > b.score;
        });

        for (const auto& ms : ordered) {
            Move move = ms.move;
            State st;
            if (!board.make_move(move, st)) continue;
            int score = -quiescence(-beta, -alpha, ply + 1);
            board.undo_move(move, st);
            if (stop) return 0;
            if (score >= beta) return score;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }

    void store_tt(uint64_t key, int depth, int score, int flag, Move move, int ply) {
        TTEntry& entry = tt[key & TT_MASK];
        if (entry.depth > depth && entry.key == key) return;
        int storeScore = score;
        if (storeScore > MATE_THRESHOLD) storeScore += ply;
        else if (storeScore < -MATE_THRESHOLD) storeScore -= ply;
        entry.key = key;
        entry.depth = depth;
        entry.flag = flag;
        entry.score = static_cast<int16_t>(storeScore);
        entry.move = move.data;
    }
};

string format_move(Board& board, Move move) {
    State st;
    if (!board.make_move(move, st)) return "O-O";
    bool givesCheck = board.in_check(board.sideToMove);
    bool mate = false;
    vector<Move> replies;
    board.generate_moves(replies, false);
    bool hasReply = false;
    for (const Move& reply : replies) {
        State st2;
        if (board.make_move(reply, st2)) {
            hasReply = true;
            board.undo_move(reply, st2);
            break;
        }
    }
    if (!hasReply && givesCheck) mate = true;
    string notation;
    int flags = move_flags(move);
    if (flags & FLAG_KING_CASTLE) {
        notation = "O-O";
    } else if (flags & FLAG_QUEEN_CASTLE) {
        notation = "O-O-O";
    } else {
        char pieceChar = piece_letter(move_piece(move));
        notation.push_back(pieceChar);
        notation += square_to_string(move_from(move));
        bool capture = (flags & FLAG_CAPTURE) != 0;
        if (capture) notation.push_back('x');
        notation += square_to_string(move_to(move));
        if (flags & FLAG_ENPASSANT) notation += "ep";
        if (flags & FLAG_PROMOTION) {
            notation.push_back('/');
            char promoChar = piece_letter(move_promo(move));
            notation.push_back(promoChar == 'p' ? 'Q' : promoChar);
        }
    }
    if (mate) notation.push_back('#');
    else if (givesCheck) notation.push_back('+');
    board.undo_move(move, st);
    return notation;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    init_attack_tables();
    init_zobrist();
    init_piece_square_tables();
    init_mvv();

    string timeoutLine;
    while (getline(cin, timeoutLine)) {
        if (!timeoutLine.empty() && timeoutLine != "\r") break;
    }
    if (timeoutLine.empty()) return 0;
    if (!timeoutLine.empty() && timeoutLine.back() == '\r') timeoutLine.pop_back();

    string fen;
    while (getline(cin, fen)) {
        if (!fen.empty() && fen != "\r") break;
    }
    if (!fen.empty() && fen.back() == '\r') fen.pop_back();

    double timeoutSeconds = 5.0;
    {
        stringstream ss(timeoutLine);
        ss >> timeoutSeconds;
    }
    double safeLimit = max(0.05, timeoutSeconds * 0.95);

    Board board;
    board.load_fen(fen);

    Search search(board);
    Move best = search.search_best_move(safeLimit);
    if (best.data == 0) {
        cout << "O-O" << '\n';
        return 0;
    }

    string notation = format_move(board, best);
    cout << notation << '\n';
    return 0;
}
