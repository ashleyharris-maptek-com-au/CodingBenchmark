const EPS: f64 = 1e-9;
const BORDER_MARGIN: f64 = 1e-3;
const PACKING_PADDING: f64 = 1e-3;
const MIN_STEP: f64 = 1e-2;
const BASE_ROTATION_SAMPLES: usize = 12;
const MAX_ROTATIONS_PER_PIECE: usize = 20;

#[derive(Clone, Copy, Debug)]

    x: f64,
    y: f64,
}

    max_x: f64,

impl StockState {

            polygon: vertices,
            bbox,
            usable_bbox,

        }
    }
}

impl SmallRng {
    fn new(seed: u64) -> Self {
        let init = if seed == 0 { 0x1234_5678_9ABC_DEF0 } else { seed };
        SmallRng { state: init }
    }
    fn next_u64(&mut self) -> u64 {
        self.state = self.state.wrapping_mul(6364136223846793005).wrapping_add(1);
        self.state
    }
    fn next_f64(&mut self) -> f64 {
        let bits = self.next_u64() >> 11;
        (bits as f64) / ((1u64 << 53) as f64)
    }
    fn next_usize(&mut self, upper: usize) -> usize {
        if upper <= 1 {
            0
        } else {
            ((self.next_f64() * upper as f64) as usize).min(upper - 1)
        }
    }
}

fn next_value<T: FromStr>(iter: &mut SplitWhitespace<'_>) -> T {
    iter.next()
        .and_then(|s| s.parse::<T>().ok())
        .expect("invalid input")
}

fn build_piece(idx: usize, vertices: Vec<Point>) -> Piece {
    let area = polygon_area(&vertices);
    let mut angles = generate_rotation_angles(&vertices);
    let mut rotations = Vec::with_capacity(angles.len());
    for angle in angles.drain(..) {
        rotations.push(build_rotated_variant(&vertices, angle));
    }
    rotations.sort_by(|a, b| a.bbox_area.total_cmp(&b.bbox_area));
    Piece {
        original_index: idx,
        original_vertices: vertices,
        rotations,
        approx_area: area,
    }
}

fn compute_bbox(points: &[Point]) -> BBox {
    if points.is_empty() {
        return BBox {
            min_x: 0.0,
            max_x: 0.0,
            min_y: 0.0,
            max_y: 0.0,
        };
    }
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in points {
        if p.x < min_x {
            min_x = p.x;
        }
        if p.x > max_x {
            max_x = p.x;
        }
        if p.y < min_y {
            min_y = p.y;
        }
        if p.y > max_y {
            max_y = p.y;
        }
    }
    BBox {
        min_x,
        max_x,
        min_y,
        max_y,
    }
}

fn shrink_bbox(bbox: &BBox, margin: f64) -> BBox {
    let width = bbox.width();
    let height = bbox.height();
    let (min_x, max_x) = if width > 2.0 * margin {
        (bbox.min_x + margin, bbox.max_x - margin)
    } else {
        (bbox.min_x, bbox.max_x)
    };
    let (min_y, max_y) = if height > 2.0 * margin {
        (bbox.min_y + margin, bbox.max_y - margin)
    } else {
        (bbox.min_y, bbox.max_y)
    };
    BBox {
        min_x,
        max_x,
        min_y,
        max_y,
    }
}

fn polygon_area(vertices: &[Point]) -> f64 {
    if vertices.len() < 3 {
        return 0.0;
    }
    let mut area = 0.0;
    for i in 0..vertices.len() {
        let a = vertices[i];
        let b = vertices[(i + 1) % vertices.len()];
        area += a.x * b.y - b.x * a.y;
       }
    area.abs() * 0.5
}

fn generate_rotation_angles(vertices: &[Point]) -> Vec<f64> {
    let mut angles = Vec::new();
    let base = BASE_ROTATION_SAMPLES.max(1);
    for i in 0..base {
        angles.push(TAU * (i as f64) / (base as f64));
    }
    let n = vertices.len();
    for i in 0..n {
        let a = vertices[i];
        let b = vertices[(i + 1) % n];
        let dx = b.x - a.x;
        let dy = b.y - a.y;
        if dx.abs() < EPS && dy.abs() < EPS {
            continue;
        }
        let mut ang = dy.atan2(dx);
        while ang < 0.0 {
            ang += TAU;
        }
        while ang >= TAU {
            ang -= TAU;
        }
        angles.push(ang);
        let mut ang2 = ang + PI / 2.0;
        if ang2 >= TAU {
            ang2 -= TAU;
        }
        angles.push(ang2);
    }
    angles.push(0.0);
    angles.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    angles.dedup_by(|a, b| (a - b).abs() < 1e-3);
    if angles.len() > MAX_ROTATIONS_PER_PIECE {
        let mut reduced = Vec::with_capacity(MAX_ROTATIONS_PER_PIECE);
        if MAX_ROTATIONS_PER_PIECE <= 1 {
            reduced.push(angles[0]);
        } else {
            for i in 0..MAX_ROTATIONS_PER_PIECE {
                let pos = i as f64 * (angles.len() as f64 - 1.0)
                    / (MAX_ROTATIONS_PER_PIECE as f64 - 1.0);
                let idx = pos.round() as usize;
                reduced.push(angles[idx.min(angles.len() - 1)]);
            }
            reduced.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
            reduced.dedup_by(|a, b| (a - b).abs() < 1e-3);
        }
        if reduced.is_empty() {
            reduced.push(angles[0]);
        }
        angles = reduced;
    }
    angles
}

fn build_rotated_variant(vertices: &[Point], angle: f64) -> RotatedVariant {
    let cos_a = angle.cos();
    let sin_a = angle.sin();
    let mut rotated = Vec::with_capacity(vertices.len());
    let mut bbox = BBox {
        min_x: f64::INFINITY,
        max_x: f64::NEG_INFINITY,
        min_y: f64::INFINITY,
        max_y: f64::NEG_INFINITY,
    };
    for v in vertices {
        let rx = v.x * cos_a - v.y * sin_a;
        let ry = v.x * sin_a + v.y * cos_a;
        rotated.push(Point { x: rx, y: ry });
        if rx < bbox.min_x {
            bbox.min_x = rx;
        }
        if rx > bbox.max_x {
            bbox.max_x = rx;
        }
        if ry < bbox.min_y {
            bbox.min_y = ry;
        }
        if ry > bbox.max_y {
            bbox.max_y = ry;
        }
    }
    if rotated.is_empty() {
        bbox = BBox {
            min_x: 0.0,
            max_x: 0.0,
            min_y: 0.0,
            max_y: 0.0,
        };
    }
    let area = bbox.width().max(0.0) * bbox.height().max(0.0);
    RotatedVariant {
        angle,
        vertices: rotated,
        bbox,
        bbox_area: area,
    }
}

fn build_edges(vertices: &[Point]) -> Vec<Edge> {
    let n = vertices.len();
    let mut edges = Vec::with_capacity(n);
    if n == 0 {
        return edges;
    }
    for i in 0..n {
        let a = vertices[i];
        let b = vertices[(i + 1) % n];
        edges.push(Edge {
            a,
            b,
            min_x: a.x.min(b.x),
            max_x: a.x.max(b.x),
            min_y: a.y.min(b.y),
            max_y: a.y.max(b.y),
        });
    }
    edges
}

fn choose_tuning(np: usize) -> Tuning {
    if np <= 40 {
        Tuning {
            horizontal_samples: 140,
            fallback_grid: 60,
            fallback_random: 2500,
        }
    } else if np <= 400 {
        Tuning {
            horizontal_samples: 90,
            fallback_grid: 40,
            fallback_random: 1500,
        }
    } else {
        Tuning {
            horizontal_samples: 60,
            fallback_grid: 30,
            fallback_random: 800,
        }
    }
}

fn place_single_piece(
    piece: &Piece,
    stock: &StockGeometry,
    states: &mut Vec<StockState>,
    tuning: &Tuning,
) -> Placement {
    loop {
        for idx in 0..states.len() {
            if !states[idx].available {
                continue;
            }
            if let Some((dx, dy, angle, new_state)) =
                attempt_on_stock(piece, stock, &states[idx], tuning.horizontal_samples)
            {
                states[idx] = new_state;
                return Placement {
                    stock_idx: idx,
                    dx,
                    dy,
                    angle,
                };
            }
        }

        let stock_idx = states.len();
        states.push(StockState::new(&stock.usable_bbox));
        if let Some((dx, dy, angle, new_state)) =
            attempt_on_stock(piece, stock, &states[stock_idx], tuning.horizontal_samples)
        {
            states[stock_idx] = new_state;
            return Placement {
                stock_idx,
                dx,
                dy,
                angle,
            };
        }

        let seed =
            piece.original_index as u64 ^ ((stock_idx as u64 + 1) * RNG_MAGIC);
        if let Some((dx, dy, angle)) = fallback_place(piece, stock, tuning, seed) {
            let mut closed = StockState::new(&stock.usable_bbox);
            closed.available = false;
            closed.cursor_x = stock.usable_bbox.max_x + 10.0;
            closed.cursor_y = stock.usable_bbox.max_y + 10.0;
            states[stock_idx] = closed;
            return Placement {
                stock_idx,
                dx,
                dy,
                angle,
            };
        } else {
            panic!("Unable to place piece {}", piece.original_index);
        }
    }
}

fn attempt_on_stock(
    piece: &Piece,
    stock: &StockGeometry,
    state: &StockState,
    horizontal_limit: usize,
) -> Option<(f64, f64, f64, StockState)> {
    if !state.available {
        return None;
    }
    let usable = &stock.usable_bbox;
    for variant in &piece.rotations {
        let width = variant.bbox.width();
        let height = variant.bbox.height();
        if width > usable.width() + EPS || height > usable.height() + EPS {
            continue;
        }
        let mut temp_state = state.clone();
        temp_state.cursor_x = temp_state.cursor_x.max(usable.min_x);
        temp_state.cursor_y = temp_state.cursor_y.max(usable.min_y);
        loop {
            if temp_state.cursor_y + height > usable.max_y + EPS {
                break;
            }
            let start_x = temp_state.cursor_x.max(usable.min_x);
            let max_start = usable.max_x - width;
            if max_start < usable.min_x - EPS {
                break;
            }
            if start_x <= max_start + EPS {
                let available = (max_start - start_x).max(0.0);
                let mut step = (width.max(height)) * 0.5 + PACKING_PADDING;
                if step < MIN_STEP {
                    step = MIN_STEP;
                }
                if horizontal_limit > 0 && available > 0.0 {
                    let required = (available / step).ceil() as usize;
                    if required > horizontal_limit {
                        step = available / horizontal_limit as f64;
                    }
                }
                let mut candidate_x = start_x;
                let mut attempts = 0usize;
                while candidate_x <= max_start + EPS {
                    if let Some((dx, dy)) =
                        check_position(variant, candidate_x, temp_state.cursor_y, stock)
                    {
                        let mut new_state = temp_state.clone();
                        new_state.cursor_x =
                            (candidate_x + width + PACKING_PADDING).min(usable.max_x);
                        new_state.shelf_height =
                            new_state.shelf_height.max(height + PACKING_PADDING);
                        return Some((dx, dy, variant.angle, new_state));
                    }
                    attempts += 1;
                    if horizontal_limit > 0 && attempts > horizontal_limit {
                        break;
                    }
                    candidate_x += step;
                }
                if candidate_x - step < max_start - EPS {
                    if let Some((dx, dy)) =
                        check_position(variant, max_start, temp_state.cursor_y, stock)
                    {
                        let mut new_state = temp_state.clone();
                        new_state.cursor_x =
                            (max_start + width + PACKING_PADDING).min(usable.max_x);
                        new_state.shelf_height =
                            new_state.shelf_height.max(height + PACKING_PADDING);
                        return Some((dx, dy, variant.angle, new_state));
                    }
                }
            }
            let advance = if temp_state.shelf_height > 0.0 {
                temp_state.shelf_height
            } else {
                height + PACKING_PADDING
            };
            temp_state.cursor_y += advance;
            temp_state.cursor_x = usable.min_x;
            temp_state.shelf_height = 0.0;
        }
    }
    None
}

fn check_position(
    variant: &RotatedVariant,
    target_min_x: f64,
    target_min_y: f64,
    stock: &StockGeometry,
) -> Option<(f64, f64)> {
    let dx = target_min_x - variant.bbox.min_x;
    let dy = target_min_y - variant.bbox.min_y;
    if polygon_inside_stock(&variant.vertices, dx, dy, stock) {
        Some((dx, dy))
    } else {
        None
    }
}

fn polygon_inside_stock(
    vertices: &[Point],
    dx: f64,
    dy: f64,
    stock: &StockGeometry,
) -> bool {
    for v in vertices {
        let pt = Point {
            x: v.x + dx,
            y: v.y + dy,
        };
        if !point_in_polygon(pt, &stock.polygon, &stock.bbox) {
            return false;
        }
    }
    let stock_edges = &stock.edges;
    for i in 0..vertices.len() {
        let p1 = Point {
            x: vertices[i].x + dx,
            y: vertices[i].y + dy,
        };
        let p2 = Point {
            x: vertices[(i + 1) % vertices.len()].x + dx,
            y: vertices[(i + 1) % vertices.len()].y + dy,
        };
        let e_min_x = p1.x.min(p2.x) - EPS;
        let e_max_x = p1.x.max(p2.x) + EPS;
        let e_min_y = p1.y.min(p2.y) - EPS;
        let e_max_y = p1.y.max(p2.y) + EPS;
        for edge in stock_edges {
            if e_max_x < edge.min_x - EPS
                || e_min_x > edge.max_x + EPS
                || e_max_y < edge.min_y - EPS
                || e_min_y > edge.max_y + EPS
            {
                continue;
            }
            if segments_strict_intersect(p1, p2, edge.a, edge.b) {
                return false;
            }
        }
    }
    true
}

fn point_in_polygon(pt: Point, polygon: &[Point], bbox: &BBox) -> bool {
    if polygon.is_empty() {
        return false;
    }
    if pt.x < bbox.min_x - EPS
        || pt.x > bbox.max_x + EPS
        || pt.y < bbox.min_y - EPS
        || pt.y > bbox.max_y + EPS
    {
        return false;
    }
    let mut inside = false;
    for i in 0..polygon.len() {
        let a = polygon[i];
        let b = polygon[(i + 1) % polygon.len()];
        if point_on_segment(pt, a, b) {
            return true;
        }
        if (a.y > pt.y) != (b.y > pt.y) {
            let denom = b.y - a.y;
            if denom.abs() < EPS {
                continue;
            }
            let t = (pt.y - a.y) / denom;
            if t >= 0.0 && t <= 1.0 {
                let x_cross = a.x + t * (b.x - a.x);
                if x_cross > pt.x {
                    inside = !inside;
                }
            }
        }
    }
    inside
}

fn point_on_segment(p: Point, a: Point, b: Point) -> bool {
    let cross = orientation(a, b, p).abs();
    if cross > 1e-8 {
        return false;
    }
    let min_x = a.x.min(b.x) - 1e-8;
    let max_x = a.x.max(b.x) + 1e-8;
    let min_y = a.y.min(b.y) - 1e-8;
    let max_y = a.y.max(b.y) + 1e-8;
    p.x >= min_x && p.x <= max_x && p.y >= min_y && p.y <= max_y
}

fn orientation(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn segments_strict_intersect(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let o1 = orientation(a1, a2, b1);
    let o2 = orientation(a1, a2, b2);
    let o3 = orientation(b1, b2, a1);
    let o4 = orientation(b1, b2, a2);
    (o1 * o2 < -EPS) && (o3 * o4 < -EPS)
}

fn fallback_place(
    piece: &Piece,
    stock: &StockGeometry,
    tuning: &Tuning,
    seed: u64,
) -> Option<(f64, f64, f64)> {
    let mut valid: Vec<&RotatedVariant> = piece
        .rotations
        .iter()
        .filter(|v| {
            v.bbox.width() <= stock.usable_bbox.width() + EPS
                && v.bbox.height() <= stock.usable_bbox.height() + EPS
        })
        .collect();

    for pass in 0..2 {
        let grid_steps = (tuning.fallback_grid * (pass + 1)).min(100).max(10);
        if let Some(res) = fallback_grid_search(&valid, stock, grid_steps) {
            return Some(res);
        }
        let random_attempts = tuning.fallback_random * (pass + 1);
        if random_attempts > 0 {
            if let Some(res) = fallback_random_search(
                &valid,
                stock,
                random_attempts,
                seed ^ ((pass as u64 + 1) * RNG_MAGIC),
            ) {
                return Some(res);
            }
        }
    }

    if let Some(res) = fallback_dense_rotation(piece, stock, seed ^ 0xA5A5_A5A5_A5A5_A5A5) {
        return Some(res);
    }
    fallback_super_random(piece, stock, seed ^ 0xDEAD_BEEF_CAFE_BABE)
}

fn fallback_grid_search(
    variants: &[&RotatedVariant],
    stock: &StockGeometry,
    grid_steps: usize,
) -> Option<(f64, f64, f64)> {
    for variant in variants {
        if let Some((dx, dy)) = grid_search_variant(variant, stock, grid_steps) {
            return Some((dx, dy, variant.angle));
        }
    }
    None
}

fn fallback_random_search(
    variants: &[&RotatedVariant],
    stock: &StockGeometry,
    attempts: usize,
    seed: u64,
) -> Option<(f64, f64, f64)> {
    if variants.is_empty() {
        return None;
    }
    let mut rng = SmallRng::new(seed);
    let usable = &stock.usable_bbox;
    for _ in 0..attempts {
        let idx = rng.next_usize(variants.len());
        let variant = variants[idx];
        let width = variant.bbox.width();
        let height = variant.bbox.height();
        if width > usable.width() + EPS || height > usable.height() + EPS {
            continue;
        }
        let min_x = usable.min_x;
        let max_x = usable.max_x - width;
        let min_y = usable.min_y;
        let max_y = usable.max_y - height;
        if max_x < min_x - EPS || max_y < min_y - EPS {
            continue;
        }
        let candidate_x = min_x + rng.next_f64() * (max_x - min_x);
        let candidate_y = min_y + rng.next_f64() * (max_y - min_y);
        let dx = candidate_x - variant.bbox.min_x;
        let dy = candidate_y - variant.bbox.min_y;
        if polygon_inside_stock(&variant.vertices, dx, dy, stock) {
            return Some((dx, dy, variant.angle));
        }
    }
    None
}

fn grid_search_variant(
    variant: &RotatedVariant,
    stock: &StockGeometry,
    grid_steps: usize,
) -> Option<(f64, f64)> {
    let usable = &stock.usable_bbox;
    let width = variant.bbox.width();
    let height = variant.bbox.height();
    if width > usable.width() + EPS || height > usable.height() + EPS {
        return None;
    }
    let min_x = usable.min_x;
    let max_x = usable.max_x - width;
    let min_y = usable.min_y;
    let max_y = usable.max_y - height;
    if max_x < min_x - EPS || max_y < min_y - EPS {
        return None;
    }
    let steps_x = grid_steps.max(1);
    let steps_y = grid_steps.max(1);
    for ix in 0..=steps_x {
        let ratio_x = if steps_x == 0 {
            0.0
        } else {
            ix as f64 / steps_x as f64
        };
        let candidate_x = min_x + (max_x - min_x) * ratio_x;
        for iy in 0..=steps_y {
            let ratio_y = if steps_y == 0 {
                0.0
            } else {
                iy as f64 / steps_y as f64
            };
            let candidate_y = min_y + (max_y - min_y) * ratio_y;
            let dx = candidate_x - variant.bbox.min_x;
            let dy = candidate_y - variant.bbox.min_y;
            if polygon_inside_stock(&variant.vertices, dx, dy, stock) {
                return Some((dx, dy));
            }
        }
    }
    None
}

fn random_for_variant(
    variant: &RotatedVariant,
    stock: &StockGeometry,
    attempts: usize,
    rng: &mut SmallRng,
) -> Option<(f64, f64)> {
    let usable = &stock.usable_bbox;
    let width = variant.bbox.width();
    let height = variant.bbox.height();
    if width > usable.width() + EPS || height > usable.height() + EPS {
        return None;
    }
    let min_x = usable.min_x;
    let max_x = usable.max_x - width;
    let min_y = usable.min_y;
    let max_y = usable.max_y - height;
    if max_x < min_x - EPS || max_y < min_y - EPS {
        return None;
    }
    for _ in 0..attempts {
        let candidate_x = min_x + rng.next_f64() * (max_x - min_x);
        let candidate_y = min_y + rng.next_f64() * (max_y - min_y);
        let dx = candidate_x - variant.bbox.min_x;
        let dy = candidate_y - variant.bbox.min_y;
        if polygon_inside_stock(&variant.vertices, dx, dy, stock) {
            return Some((dx, dy));
        }
    }
    None
}

fn fallback_dense_rotation(
    piece: &Piece,
    stock: &StockGeometry,
    seed: u64,
) -> Option<(f64, f64, f64)> {
    let mut rng = SmallRng::new(seed);
    for i in 0..DENSE_ROTATION_STEPS {
        let angle = TAU * (i as f64) / (DENSE_ROTATION_STEPS as f64);
        let variant = build_rotated_variant(&piece.original_vertices, angle);
        if variant.bbox.width() > stock.usable_bbox.width() + EPS
            || variant.bbox.height() > stock.usable_bbox.height() + EPS
        {
            continue;
        }
        if let Some((dx, dy)) = grid_search_variant(&variant, stock, 50) {
            return Some((dx, dy, angle));
        }
        if let Some((dx, dy)) =
            random_for_variant(&variant, stock, DENSE_RANDOM_TRANSLATIONS_PER_ANGLE, &mut rng)
        {
            return Some((dx, dy, angle));
        }
    }
    for _ in 0..DENSE_RANDOM_ANGLES {
        let angle = rng.next_f64() * TAU;
        let variant = build_rotated_variant(&piece.original_vertices, angle);
        if variant.bbox.width() > stock.usable_bbox.width() + EPS
            || variant.bbox.height() > stock.usable_bbox.height() + EPS
        {
            continue;
        }
        if let Some((dx, dy)) = random_for_variant(
            &variant,
            stock,
            DENSE_RANDOM_TRANSLATIONS_PER_ANGLE * 2,
            &mut rng,
        ) {
            return Some((dx, dy, angle));
        }
    }
    None
}

fn fallback_super_random(
    piece: &Piece,
    stock: &StockGeometry,
    seed: u64,
) -> Option<(f64, f64, f64)> {
    let mut rng = SmallRng::new(seed);
    for _ in 0..SUPER_RANDOM_ATTEMPTS {
        let angle = rng.next_f64() * TAU;
        let variant = build_rotated_variant(&piece.original_vertices, angle);
        if variant.bbox.width() > stock.usable_bbox.width() + EPS
            || variant.bbox.height() > stock.usable_bbox.height() + EPS
        {
            continue;
        }
        let min_x = stock.usable_bbox.min_x;
        let max_x = stock.usable_bbox.max_x - variant.bbox.width();
        let min_y = stock.usable_bbox.min_y;
        let max_y = stock.usable_bbox.max_y - variant.bbox.height();
        if max_x < min_x - EPS || max_y < min_y - EPS {
            continue;
        }
        let candidate_x = min_x + rng.next_f64() * (max_x - min_x);
        let candidate_y = min_y + rng.next_f64() * (max_y - min_y);
        let dx = candidate_x - variant.bbox.min_x;
        let dy = candidate_y - variant.bbox.min_y;
        if polygon_inside_stock(&variant.vertices, dx, dy, stock) {
            return Some((dx, dy, angle));
        }
    }
    None
}
