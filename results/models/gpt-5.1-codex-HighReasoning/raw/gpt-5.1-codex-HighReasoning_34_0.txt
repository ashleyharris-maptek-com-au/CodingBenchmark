```python
import sys
import math


def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return

    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    capacity = float(next(it))
    depot_x = float(next(it))
    depot_y = float(next(it))

    positions = [(depot_x, depot_y)]
    demands = []
    for _ in range(N):
        x = float(next(it))
        y = float(next(it))
        demand = float(next(it))
        positions.append((x, y))
        demands.append(demand)

    size = N + 1
    dist = [[0.0] * size for _ in range(size)]
    for i in range(size):
        xi, yi = positions[i]
        for j in range(i + 1, size):
            xj, yj = positions[j]
            d = math.hypot(xi - xj, yi - yj)
            dist[i][j] = dist[j][i] = d

    def route_distance(seq):
        if not seq:
            return 0.0
        total = dist[0][seq[0] + 1]
        for idx in range(len(seq) - 1):
            total += dist[seq[idx] + 1][seq[idx + 1] + 1]
        total += dist[seq[-1] + 1][0]
        return total

    def best_join(seq_a, seq_b):
        cost_a = route_distance(seq_a)
        cost_b = route_distance(seq_b)
        base_cost = cost_a + cost_b
        best_diff = float('inf')
        best_seq = list(seq_a + seq_b)
        for rev_a in (False, True):
            oriented_a = seq_a[::-1] if rev_a else seq_a
            for rev_b in (False, True):
                oriented_b = seq_b[::-1] if rev_b else seq_b
                seq_ab = oriented_a + oriented_b
                diff = route_distance(seq_ab) - base_cost
                if diff < best_diff:
                    best_diff = diff
                    best_seq = list(seq_ab)
                seq_ba = oriented_b + oriented_a
                diff = route_distance(seq_ba) - base_cost
                if diff < best_diff:
                    best_diff = diff
                    best_seq = list(seq_ba)
        return best_diff, best_seq

    def two_opt(seq):
        n = len(seq)
        if n < 3:
            return seq

        def node_dist(a, b):
            idx_a = 0 if a == -1 else a + 1
            idx_b = 0 if b == -1 else b + 1
            return dist[idx_a][idx_b]

        improved = True
        while improved:
            improved = False
            for i in range(n - 1):
                for j in range(i + 1, n):
                    prev_node = -1 if i == 0 else seq[i - 1]
                    next_node = -1 if j == n - 1 else seq[j + 1]
                    current = node_dist(prev_node, seq[i]) + node_dist(seq[j], next_node)
                    proposed = node_dist(prev_node, seq[j]) + node_dist(seq[i], next_node)
                    if proposed + 1e-9 < current:
                        seq[i:j + 1] = reversed(seq[i:j + 1])
                        improved = True
                        break
                if improved:
                    break
        return seq

    routes = []
    cust_to_route = [-1] * N
    for idx in range(N):
        routes.append({'customers': [idx], 'demand': demands[idx]})
        cust_to_route[idx] = idx

    savings = []
    for i in range(N):
        for j in range(i + 1, N):
            s = dist[0][i + 1] + dist[0][j + 1] - dist[i + 1][j + 1]
            savings.append((s, i, j))
    savings.sort(reverse=True, key=lambda x: x[0])

    EPS = 1e-9

    for s, i, j in savings:
        if s <= EPS:
            continue
        idx_i = cust_to_route[i]
        idx_j = cust_to_route[j]
        if idx_i == idx_j:
            continue
        route_i = routes[idx_i]
        route_j = routes[idx_j]
        if route_i is None or route_j is None:
            continue
        if route_i['demand'] + route_j['demand'] > capacity + EPS:
            continue
        seq_i = route_i['customers']
        seq_j = route_j['customers']
        if seq_i[0] == i:
            oriented_i = list(reversed(seq_i))
        elif seq_i[-1] == i:
            oriented_i = list(seq_i)
        else:
            continue
        if seq_j[0] == j:
            oriented_j = list(seq_j)
        elif seq_j[-1] == j:
            oriented_j = list(reversed(seq_j))
        else:
            continue

        new_seq = oriented_i + oriented_j
        combined_demand = route_i['demand'] + route_j['demand']
        route_i['customers'] = new_seq
        route_i['demand'] = combined_demand
        routes[idx_j] = None
        for customer in new_seq:
            cust_to_route[customer] = idx_i

    active = [idx for idx, route in enumerate(routes) if route is not None]
    while len(active) > K:
        best_pair = None
        best_diff = float('inf')
        best_seq = None
        for pos_a in range(len(active)):
            idx_a = active[pos_a]
            route_a = routes[idx_a]
            for pos_b in range(pos_a + 1, len(active)):
                idx_b = active[pos_b]
                route_b = routes[idx_b]
                if route_a['demand'] + route_b['demand'] > capacity + EPS:
                    continue
                diff, seq = best_join(route_a['customers'], route_b['customers'])
                if diff < best_diff:
                    best_diff = diff
                    best_pair = (idx_a, idx_b)
                    best_seq = seq
        if best_pair is None:
            break
        idx_a, idx_b = best_pair
        route_a = routes[idx_a]
        route_b = routes[idx_b]
        route_a['customers'] = best_seq
        route_a['demand'] = route_a['demand'] + route_b['demand']
        routes[idx_b] = None
        for customer in best_seq:
            cust_to_route[customer] = idx_a
        active = [idx for idx, route in enumerate(routes) if route is not None]

    final_routes = [route for route in routes if route is not None]

    for route in final_routes:
        route['customers'] = two_opt(route['customers'])

    total_distance = sum(route_distance(route['customers']) for route in final_routes)

    output_lines = [f"{total_distance:.6f}", str(len(final_routes))]
    for route in final_routes:
        if route['customers']:
            output_lines.append(" ".join(str(idx) for idx in route['customers']))
        else:
            output_lines.append("")
    sys.stdout.write("\n".join(output_lines))


if __name__ == "__main__":
    main()
```
