use std::cmp::Ordering;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, idx: 0 }
    }
    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.buf[self.idx] == b'-' {
            sign = -1;
            self.idx += 1;
        }
        let mut val = 0i64;
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_digit() {
            val = val * 10 + (self.buf[self.idx] - b'0') as i64;
            self.idx += 1;
        }
        let signed = val * sign;
        Some(if signed < 0 { 0 } else { signed as usize })
    }
}

struct SimpleRng {
    state: u64,
}
impl SimpleRng {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x >> 12;
        x ^= x << 25;
        x ^= x >> 27;
        self.state = x;
        x.wrapping_mul(0x2545F4914F6CDD1D)
    }
    fn next_f64(&mut self) -> f64 {
        (self.next_u64() >> 11) as f64 * (1.0 / (1u64 << 53) as f64)
    }
    fn rand_range(&mut self, min: f64, max: f64) -> f64 {
        if !min.is_finite() || !max.is_finite() {
            return 0.0;
        }
        if max <= min {
            return min;
        }
        min + (max - min) * self.next_f64()
    }
}

#[derive(Clone)]
struct ComponentBounds {
    comp_index: usize,
    min_x: f64,
    min_y: f64,
    width: f64,
    height: f64,
    size: usize,
}

fn layout_force_directed(
    nodes: &[usize],
    adj: &[Vec<usize>],
    node_to_local: &[usize],
    rng: &mut SimpleRng,
) -> Vec<(f64, f64)> {
    let n = nodes.len();
    let mut coords = vec![(0.0, 0.0); n];
    if n <= 1 {
        return coords;
    }
    for p in &mut coords {
        *p = (rng.rand_range(-1.0, 1.0), rng.rand_range(-1.0, 1.0));
    }
    let mut disp = vec![(0.0, 0.0); n];
    let mut edges = Vec::new();
    for &u in nodes {
        let iu = node_to_local[u];
        for &v in &adj[u] {
            if u == v {
                continue;
            }
            let iv = node_to_local[v];
            if iv == usize::MAX || iu >= iv {
                continue;
            }
            edges.push((iu, iv));
        }
    }
    let area = n as f64;
    let k = area.sqrt().max(0.5);
    let iterations = if n <= 30 {
        200
    } else if n <= 80 {
        160
    } else if n <= 200 {
        120
    } else {
        90
    };
    let initial_temp = (k * 0.9).max(0.1) + (n as f64).sqrt() * 0.2;
    for iter in 0..iterations {
        for d in &mut disp {
            *d = (0.0, 0.0);
        }
        for i in 0..n {
            for j in (i + 1)..n {
                let dx = coords[i].0 - coords[j].0;
                let dy = coords[i].1 - coords[j].1;
                let dist = (dx * dx + dy * dy).sqrt().max(1e-6);
                let force = (k * k) / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }
        for &(u, v) in &edges {
            let dx = coords[u].0 - coords[v].0;
            let dy = coords[u].1 - coords[v].1;
            let dist = (dx * dx + dy * dy).sqrt().max(1e-6);
            let force = (dist * dist) / k;
            let fx = dx / dist * force;
            let fy = dy / dist * force;
            disp[u].0 -= fx;
            disp[u].1 -= fy;
            disp[v].0 += fx;
            disp[v].1 += fy;
        }
        let t = initial_temp * (1.0 - iter as f64 / iterations as f64);
        for i in 0..n {
            let dx = disp[i].0;
            let dy = disp[i].1;
            let len = (dx * dx + dy * dy).sqrt();
            if len > 0.0 {
                let limited = len.min(t);
                coords[i].0 += dx / len * limited;
                coords[i].1 += dy / len * limited;
            }
            coords[i].0 = coords[i].0.clamp(-1e6, 1e6);
            coords[i].1 = coords[i].1.clamp(-1e6, 1e6);
        }
    }
    let (sum_x, sum_y) =
        coords
            .iter()
            .fold((0.0, 0.0), |acc, &(x, y)| (acc.0 + x, acc.1 + y));
    let n_f = n as f64;
    for p in &mut coords {
        p.0 -= sum_x / n_f;
        p.1 -= sum_y / n_f;
    }
    coords
}

fn layout_spiral(
    nodes: &[usize],
    adj: &[Vec<usize>],
    node_to_local: &[usize],
    rng: &mut SimpleRng,
) -> Vec<(f64, f64)> {
    let n = nodes.len();
    let mut coords = vec![(0.0, 0.0); n];
    if n <= 1 {
        return coords;
    }
    let mut visited = vec![false; n];
    let mut order = Vec::with_capacity(n);
    let mut queue = VecDeque::new();
    let root = *nodes
        .iter()
        .max_by_key(|&&u| adj[u].len())
        .unwrap_or(&nodes[0]);
    let root_idx = node_to_local[root];
    visited[root_idx] = true;
    queue.push_back(root);
    while let Some(u) = queue.pop_front() {
        order.push(u);
        for &v in &adj[u] {
            let lv = node_to_local[v];
            if lv == usize::MAX || visited[lv] {
                continue;
            }
            visited[lv] = true;
            queue.push_back(v);
        }
    }
    for &start in nodes {
        let idx = node_to_local[start];
        if visited[idx] {
            continue;
        }
        visited[idx] = true;
        queue.push_back(start);
        while let Some(u) = queue.pop_front() {
            order.push(u);
            for &v in &adj[u] {
                let lv = node_to_local[v];
                if lv == usize::MAX || visited[lv] {
                    continue;
                }
                visited[lv] = true;
                queue.push_back(v);
            }
        }
    }
    let golden_angle = PI * (3.0 - 5.0_f64.sqrt());
    let base_spacing = 1.5 + (n as f64).ln().max(1.0) * 0.3;
    let jitter = base_spacing * 0.05;
    for (idx, &node) in order.iter().enumerate() {
        let angle = idx as f64 * golden_angle;
        let radius = base_spacing * (idx as f64 + 1.0).sqrt();
        let mut x = radius * angle.cos();
        let mut y = radius * angle.sin();
        if jitter > 0.0 {
            x += rng.rand_range(-jitter, jitter);
            y += rng.rand_range(-jitter, jitter);
        }
        let local_idx = node_to_local[node];
        coords[local_idx] = (x, y);
    }
    coords
}

fn layout_radial(
    nodes: &[usize],
    adj: &[Vec<usize>],
    node_to_local: &[usize],
    rng: &mut SimpleRng,
) -> Vec<(f64, f64)> {
    let n = nodes.len();
    let mut coords = vec![(0.0, 0.0); n];
    if n <= 1 {
        return coords;
    }
    let root = *nodes
        .iter()
        .max_by_key(|&&u| adj[u].len())
        .unwrap_or(&nodes[0]);
    let mut dist = vec![usize::MAX; n];
    let mut queue = VecDeque::new();
    let root_idx = node_to_local[root];
    dist[root_idx] = 0;
    queue.push_back(root);
    while let Some(u) = queue.pop_front() {
        let idx_u = node_to_local[u];
        let d = dist[idx_u];
        for &v in &adj[u] {
            let idx_v = node_to_local[v];
            if idx_v == usize::MAX {
                continue;
            }
            if dist[idx_v] == usize::MAX {
                dist[idx_v] = d + 1;
                queue.push_back(v);
            }
        }
    }
    let mut layers: Vec<Vec<usize>> = Vec::new();
    for &node in nodes {
        let idx = node_to_local[node];
        let layer = dist[idx];
        let layer = if layer == usize::MAX { 0 } else { layer };
        if layers.len() <= layer {
            layers.resize(layer + 1, Vec::new());
        }
        layers[layer].push(node);
    }
    let layer_gap = 2.0 + (n as f64).ln().max(1.0);
    for (layer_idx, layer_nodes) in layers.iter().enumerate() {
        if layer_nodes.is_empty() {
            continue;
        }
        if layer_idx == 0 {
            for &node in layer_nodes {
                let idx = node_to_local[node];
                coords[idx] = (0.0, 0.0);
            }
            continue;
        }
        let radius = layer_gap * layer_idx as f64;
        let count = layer_nodes.len();
        let offset = rng.rand_range(0.0, 2.0 * PI);
        for (i, &node) in layer_nodes.iter().enumerate() {
            let angle = offset + 2.0 * PI * (i as f64) / (count as f64);
            let mut r = radius + rng.rand_range(-layer_gap * 0.1, layer_gap * 0.1);
            if r < layer_gap * 0.2 {
                r = layer_gap * 0.2;
            }
            let x = r * angle.cos();
            let y = r * angle.sin();
            let idx = node_to_local[node];
            coords[idx] = (x, y);
        }
    }
    coords
}

fn main() {
    let mut scanner = Scanner::new();
    let n = match scanner.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = scanner.next_usize().unwrap_or(0);
    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u = match scanner.next_usize() {
            Some(v) => v,
            None => continue,
        };
        let v = match scanner.next_usize() {
            Some(v) => v,
            None => continue,
        };
        if u >= n || v >= n {
            continue;
        }
        adj[u].push(v);
        if u != v {
            adj[v].push(u);
        }
    }
    if n == 0 {
        return;
    }
    let mut visited = vec![false; n];
    let mut components: Vec<Vec<usize>> = Vec::new();
    for node in 0..n {
        if visited[node] {
            continue;
        }
        let mut stack = Vec::new();
        stack.push(node);
        visited[node] = true;
        let mut comp_nodes = Vec::new();
        while let Some(u) = stack.pop() {
            comp_nodes.push(u);
            for &v in &adj[u] {
                if v < n && !visited[v] {
                    visited[v] = true;
                    stack.push(v);
                }
            }
        }
        components.push(comp_nodes);
    }
    let mut positions = vec![(0.0_f64, 0.0_f64); n];
    let mut node_to_local = vec![usize::MAX; n];
    let mut rng = SimpleRng::new(
        0xdecafbad12345678u64 ^ (n as u64).wrapping_mul(0x9e3779b97f4a7c15) ^ (m as u64),
    );
    let mut bounds: Vec<ComponentBounds> = Vec::new();
    for (comp_idx, comp_nodes) in components.iter().enumerate() {
        for (local_idx, &node) in comp_nodes.iter().enumerate() {
            node_to_local[node] = local_idx;
        }
        let coords = if comp_nodes.len() <= 400 {
            layout_force_directed(comp_nodes, &adj, &node_to_local, &mut rng)
        } else if comp_nodes.len() <= 5000 {
            layout_spiral(comp_nodes, &adj, &node_to_local, &mut rng)
        } else {
            layout_radial(comp_nodes, &adj, &node_to_local, &mut rng)
        };
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for (local_idx, &node) in comp_nodes.iter().enumerate() {
            let (x, y) = coords[local_idx];
            positions[node] = (x, y);
            if x.is_finite() {
                min_x = min_x.min(x);
                max_x = max_x.max(x);
            }
            if y.is_finite() {
                min_y = min_y.min(y);
                max_y = max_y.max(y);
            }
        }
        if !min_x.is_finite() || !max_x.is_finite() {
            min_x = 0.0;
            max_x = 1.0;
        }
        if !min_y.is_finite() || !max_y.is_finite() {
            min_y = 0.0;
            max_y = 1.0;
        }
        if (max_x - min_x).abs() < 1e-6 {
            max_x = min_x + 1.0;
        }
        if (max_y - min_y).abs() < 1e-6 {
            max_y = min_y + 1.0;
        }
        bounds.push(ComponentBounds {
            comp_index: comp_idx,
            min_x,
            min_y,
            width: max_x - min_x,
            height: max_y - min_y,
            size: comp_nodes.len(),
        });
        for &node in comp_nodes {
            node_to_local[node] = usize::MAX;
        }
    }
    bounds.sort_by(|a, b| {
        b.size
            .cmp(&a.size)
            .then_with(|| b.height.partial_cmp(&a.height).unwrap_or(Ordering::Equal))
    });
    let total_area: f64 = bounds.iter().map(|b| b.width * b.height).sum();
    let max_comp_width = bounds.iter().fold(0.0, |acc, b| acc.max(b.width));
    let mut max_row_width = total_area.sqrt().max(max_comp_width);
    if !max_row_width.is_finite() || max_row_width < 1.0 {
        max_row_width = max_comp_width.max(50.0);
    }
    max_row_width = max_row_width * 1.2 + 10.0;
    let spacing = 10.0;
    let mut current_x = 0.0;
    let mut current_y = 0.0;
    let mut row_height = 0.0;
    for bound in bounds {
        let width_with_spacing = bound.width + spacing;
        if current_x > 0.0 && current_x + width_with_spacing > max_row_width {
            current_x = 0.0;
            current_y += row_height + spacing;
            row_height = 0.0;
        }
        let offset_x = current_x - bound.min_x;
        let offset_y = current_y - bound.min_y;
        for &node in &components[bound.comp_index] {
            positions[node].0 += offset_x;
            positions[node].1 += offset_y;
        }
        current_x += width_with_spacing;
        row_height = row_height.max(bound.height);
    }
    let mut output = String::with_capacity(n * 20);
    for &(x, y) in &positions {
        let xx = if x.is_finite() { x } else { 0.0 };
        let yy = if y.is_finite() { y } else { 0.0 };
        output.push_str(&format!("{:.6} {:.6}\n", xx, yy));
    }
    print!("{}", output);
}
