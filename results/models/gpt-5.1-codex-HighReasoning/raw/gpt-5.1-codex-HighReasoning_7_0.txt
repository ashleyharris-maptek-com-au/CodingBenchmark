namespace {
constexpr double SPLIT_EPS = 1e-7;
constexpr double MERGE_EPS = 1e-8;
constexpr double TRI_AREA_EPS = 1e-16;
constexpr double QUANT_SCALE = 1e7;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double xx, double yy, double zz) : x(xx), y(yy), z(zz) {}
    Vec3 operator+(const Vec3& o) const { return Vec3{x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return Vec3{x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return Vec3{x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return Vec3{x / s, y / s, z / s}; }
    Vec3 operator-() const { return Vec3{-x, -y, -z}; }
    Vec3& operator+=(const Vec3& o) { x += o.x; y += o.y; z += o.z; return *this; }
    Vec3& operator-=(const Vec3& o) { x -= o.x; y -= o.y; z -= o.z; return *this; }
    Vec3& operator*=(double s) { x *= s; y *= s; z *= s; return *this; }
    double lengthSquared() const { return x * x + y * y + z * z; }
};

inline Vec3 operator*(double s, const Vec3& v) { return v * s; }

inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return Vec3{
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

inline double squaredDistance(const Vec3& a, const Vec3& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    double dz = a.z - b.z;
    return dx * dx + dy * dy + dz * dz;
}

struct Plane {
    Vec3 normal;
    double w;
    Plane() : normal(), w(0.0) {}
    Plane(const Vec3& n, double ww) : normal(n), w(ww) {}
    static Plane fromPoints(const Vec3& a, const Vec3& b, const Vec3& c) {
        Vec3 ab = b - a;
        Vec3 ac = c - a;
        Vec3 n = cross(ab, ac);
        double len2 = n.lengthSquared();
        if (len2 < 1e-16) {
            return Plane();
        }
        double len = std::sqrt(len2);
        Vec3 nn = n / len;
        return Plane(nn, dot(nn, a));
    }
    bool valid() const {
        return normal.lengthSquared() > 1e-16;
    }
    void flip() {
        normal = normal * -1.0;
        w *= -1.0;
    }
};

struct Vertex {
    Vec3 pos;
};

struct Polygon {
    std::vector<Vertex> vertices;
    Plane plane;
    void flip() {
        std::reverse(vertices.begin(), vertices.end());
        plane.flip();
    }
};

void cleanupVertices(std::vector<Vertex>& verts) {
    if (verts.empty()) return;
    const double tol2 = MERGE_EPS * MERGE_EPS;
    std::vector<Vertex> cleaned;
    cleaned.reserve(verts.size());
    for (const auto& v : verts) {
        if (cleaned.empty() || squaredDistance(v.pos, cleaned.back().pos) > tol2) {
            cleaned.push_back(v);
        }
    }
    if (cleaned.size() >= 2 &&
        squaredDistance(cleaned.front().pos, cleaned.back().pos) <= tol2) {
        cleaned.pop_back();
    }
    verts.swap(cleaned);
}

enum PositionType {
    VT_COPLANAR = 0,
    VT_FRONT = 1,
    VT_BACK = 2
};

void splitPolygon(const Plane& plane,
                  Polygon polygon,
                  std::vector<Polygon>& coplanarFront,
                  std::vector<Polygon>& coplanarBack,
                  std::vector<Polygon>& front,
                  std::vector<Polygon>& back) {
    const size_t count = polygon.vertices.size();
    if (count == 0) return;

    std::vector<int> types(count, VT_COPLANAR);
    std::vector<double> distances(count, 0.0);
    int polygonType = 0;

    for (size_t i = 0; i < count; ++i) {
        double dist = dot(plane.normal, polygon.vertices[i].pos) - plane.w;
        int type = VT_COPLANAR;
        if (dist > SPLIT_EPS) type = VT_FRONT;
        else if (dist < -SPLIT_EPS) type = VT_BACK;
        types[i] = type;
        distances[i] = dist;
        polygonType |= type;
    }

    if (polygonType == VT_COPLANAR) {
        double orientation = dot(plane.normal, polygon.plane.normal);
        if (orientation >= 0) {
            coplanarFront.push_back(std::move(polygon));
        } else {
            coplanarBack.push_back(std::move(polygon));
        }
    } else if (polygonType == VT_FRONT) {
        front.push_back(std::move(polygon));
    } else if (polygonType == VT_BACK) {
        back.push_back(std::move(polygon));
    } else {
        std::vector<Vertex> frontVerts;
        std::vector<Vertex> backVerts;
        frontVerts.reserve(count + 1);
        backVerts.reserve(count + 1);

        for (size_t i = 0; i < count; ++i) {
            size_t j = (i + 1) % count;
            const Vertex& vi = polygon.vertices[i];
            const Vertex& vj = polygon.vertices[j];
            int ti = types[i];
            int tj = types[j];

            if (ti != VT_BACK) {
                frontVerts.push_back(vi);
            }
            if (ti != VT_FRONT) {
                backVerts.push_back(vi);
            }

            if ((ti == VT_FRONT && tj == VT_BACK) ||
                (ti == VT_BACK && tj == VT_FRONT)) {
                double denom = distances[i] - distances[j];
                if (std::abs(denom) < 1e-12) {
                    continue;
                }
                double t = distances[i] / denom;
                t = std::clamp(t, 0.0, 1.0);
                Vec3 pos = vi.pos + (vj.pos - vi.pos) * t;
                Vertex v{pos};
                frontVerts.push_back(v);
                backVerts.push_back(v);
            }
        }

        cleanupVertices(frontVerts);
        cleanupVertices(backVerts);

        if (frontVerts.size() >= 3) {
            Polygon poly;
            poly.vertices = std::move(frontVerts);
            poly.plane = polygon.plane;
            front.push_back(std::move(poly));
        }
        if (backVerts.size() >= 3) {
            Polygon poly;
            poly.vertices = std::move(backVerts);
            poly.plane = polygon.plane;
            back.push_back(std::move(poly));
        }
    }
}

class BSPNode {
public:
    BSPNode() = default;

    void build(std::vector<Polygon> list) {
        if (list.empty()) return;
        if (!plane.has_value()) {
            plane = list.front().plane;
        }
        std::vector<Polygon> frontList;
        std::vector<Polygon> backList;
        frontList.reserve(list.size());
        backList.reserve(list.size());

        for (auto& poly : list) {
            splitPolygon(*plane, std::move(poly), polygons, polygons, frontList, backList);
        }

        if (!frontList.empty()) {
            if (!front) front = std::make_unique<BSPNode>();
            front->build(std::move(frontList));
        }
        if (!backList.empty()) {
            if (!back) back = std::make_unique<BSPNode>();
            back->build(std::move(backList));
        }
    }

    std::vector<Polygon> clipPolygons(std::vector<Polygon> list) const {
        if (!plane.has_value() || list.empty()) {
            return list;
        }
        std::vector<Polygon> frontList;
        std::vector<Polygon> backList;
        frontList.reserve(list.size());
        backList.reserve(list.size());

        for (auto& poly : list) {
            splitPolygon(*plane, std::move(poly), frontList, backList, frontList, backList);
        }

        if (front) {
            frontList = front->clipPolygons(std::move(frontList));
        }
        if (back) {
            backList = back->clipPolygons(std::move(backList));
        } else {
            backList.clear();
        }

        frontList.insert(frontList.end(),
                         std::make_move_iterator(backList.begin()),
                         std::make_move_iterator(backList.end()));
        return frontList;
    }

    void clipTo(const BSPNode& other) {
        polygons = other.clipPolygons(std::move(polygons));
        if (front) front->clipTo(other);
        if (back) back->clipTo(other);
    }

    void invert() {
        for (auto& poly : polygons) {
            poly.flip();
        }
        if (plane) {
            plane->flip();
        }
        if (front) front->invert();
        if (back) back->invert();
        std::swap(front, back);
    }

    std::vector<Polygon> allPolygons() const {
        std::vector<Polygon> result = polygons;
        if (front) {
            auto more = front->allPolygons();
            result.insert(result.end(),
                          std::make_move_iterator(more.begin()),
                          std::make_move_iterator(more.end()));
        }
        if (back) {
            auto more = back->allPolygons();
            result.insert(result.end(),
                          std::make_move_iterator(more.begin()),
                          std::make_move_iterator(more.end()));
        }
        return result;
    }

private:
    std::vector<Polygon> polygons;
    std::unique_ptr<BSPNode> front;
    std::unique_ptr<BSPNode> back;
    std::optional<Plane> plane;
};

struct BoundingBox {
    Vec3 min;
    Vec3 max;
    bool initialized = false;

    void expand(const Vec3& p) {
        if (!initialized) {
            min = max = p;
            initialized = true;
        } else {
            min.x = std::min(min.x, p.x);
            min.y = std::min(min.y, p.y);
            min.z = std::min(min.z, p.z);
            max.x = std::max(max.x, p.x);
            max.y = std::max(max.y, p.y);
            max.z = std::max(max.z, p.z);
        }
    }
};

bool boxesOverlap(const BoundingBox& a, const BoundingBox& b, double tol) {
    if (!a.initialized || !b.initialized) return false;
    if (a.max.x + tol < b.min.x || b.max.x + tol < a.min.x) return false;
    if (a.max.y + tol < b.min.y || b.max.y + tol < a.min.y) return false;
    if (a.max.z + tol < b.min.z || b.max.z + tol < a.min.z) return false;
    return true;
}

std::vector<Polygon> buildPolygons(const std::vector<Vec3>& vertices,
                                   const std::vector<std::array<int, 3>>& faces) {
    std::vector<Polygon> polys;
    polys.reserve(faces.size());
    const int n = static_cast<int>(vertices.size());
    for (const auto& f : faces) {
        bool valid = true;
        Polygon poly;
        poly.vertices.reserve(3);
        for (int idx : f) {
            if (idx < 0 || idx >= n) {
                valid = false;
                break;
            }
            poly.vertices.push_back(Vertex{vertices[idx]});
        }
        if (!valid) continue;
        poly.plane = Plane::fromPoints(poly.vertices[0].pos,
                                       poly.vertices[1].pos,
                                       poly.vertices[2].pos);
        if (!poly.plane.valid()) continue;
        polys.push_back(std::move(poly));
    }
    return polys;
}

std::vector<Polygon> computeUnion(std::vector<Polygon> a, std::vector<Polygon> b) {
    if (a.empty()) return b;
    if (b.empty()) return a;

    BSPNode nodeA;
    nodeA.build(std::move(a));
    BSPNode nodeB;
    nodeB.build(std::move(b));

    nodeA.clipTo(nodeB);
    nodeB.clipTo(nodeA);
    nodeB.invert();
    nodeB.clipTo(nodeA);
    nodeB.invert();
    nodeA.build(nodeB.allPolygons());
    return nodeA.allPolygons();
}

struct QuantizedKey {
    long long x, y, z;
    bool operator==(const QuantizedKey& other) const {
        return x == other.x && y == other.y && z == other.z;
    }
};

struct KeyHash {
    std::size_t operator()(const QuantizedKey& k) const noexcept {
        std::size_t h = std::hash<long long>{}(k.x);
        h = h * 31 + std::hash<long long>{}(k.y);
        h = h * 31 + std::hash<long long>{}(k.z);
        return h;
    }
};

int vertexIndexFor(const Vec3& pos,
                   std::vector<Vec3>& vertices,
                   std::unordered_map<QuantizedKey, int, KeyHash>& map) {
    QuantizedKey key{
        static_cast<long long>(std::llround(pos.x * QUANT_SCALE)),
        static_cast<long long>(std::llround(pos.y * QUANT_SCALE)),
        static_cast<long long>(std::llround(pos.z * QUANT_SCALE))
    };
    auto it = map.find(key);
    if (it != map.end()) return it->second;
    int idx = static_cast<int>(vertices.size());
    vertices.push_back(pos);
    map.emplace(key, idx);
    return idx;
}

std::vector<Vec3> cleanPositions(const std::vector<Vertex>& verts) {
    std::vector<Vec3> positions;
    if (verts.empty()) return positions;
    const double tol2 = MERGE_EPS * MERGE_EPS;
    positions.reserve(verts.size());
    for (const auto& v : verts) {
        if (positions.empty() ||
            squaredDistance(v.pos, positions.back()) > tol2) {
            positions.push_back(v.pos);
        }
    }
    if (positions.size() >= 2 &&
        squaredDistance(positions.front(), positions.back()) <= tol2) {
        positions.pop_back();
    }
    return positions;
}

void polygonsToMesh(const std::vector<Polygon>& polys,
                    std::vector<Vec3>& outVertices,
                    std::vector<std::array<int, 3>>& outFaces) {
    std::unordered_map<QuantizedKey, int, KeyHash> vertexMap;
    vertexMap.reserve(polys.size() * 3 + 1);

    for (const auto& poly : polys) {
        auto positions = cleanPositions(poly.vertices);
        if (positions.size() < 3) continue;
        std::vector<int> indices(positions.size());
        for (std::size_t i = 0; i < positions.size(); ++i) {
            indices[i] = vertexIndexFor(positions[i], outVertices, vertexMap);
        }
        const Vec3& p0 = positions[0];
        for (std::size_t i = 1; i + 1 < positions.size(); ++i) {
            const Vec3& p1 = positions[i];
            const Vec3& p2 = positions[i + 1];
            Vec3 n = cross(p1 - p0, p2 - p0);
            if (n.lengthSquared() <= TRI_AREA_EPS) continue;
            int id0 = indices[0];
            int id1 = indices[i];
            int id2 = indices[i + 1];
            if (id0 == id1 || id0 == id2 || id1 == id2) continue;
            outFaces.push_back({id0, id1, id2});
        }
    }
}
} // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::size_t VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) {
        return 0;
    }

    std::vector<Vec3> vertsA(VA);
    BoundingBox boxA;
    for (std::size_t i = 0; i < VA; ++i) {
        double x, y, z;
        std::cin >> x >> y >> z;
        vertsA[i] = Vec3{x, y, z};
        boxA.expand(vertsA[i]);
    }

    std::vector<std::array<int, 3>> facesA(FA);
    for (std::size_t i = 0; i < FA; ++i) {
        int a, b, c;
        std::cin >> a >> b >> c;
        facesA[i] = {a, b, c};
    }

    std::vector<Vec3> vertsB(VB);
    BoundingBox boxB;
    for (std::size_t i = 0; i < VB; ++i) {
        double x, y, z;
        std::cin >> x >> y >> z;
        vertsB[i] = Vec3{x, y, z};
        boxB.expand(vertsB[i]);
    }

    std::vector<std::array<int, 3>> facesB(FB);
    for (std::size_t i = 0; i < FB; ++i) {
        int a, b, c;
        std::cin >> a >> b >> c;
        facesB[i] = {a, b, c};
    }

    auto polysA = buildPolygons(vertsA, facesA);
    auto polysB = buildPolygons(vertsB, facesB);

    std::vector<Polygon> finalPolys;
    if (polysA.empty() && polysB.empty()) {
        finalPolys.clear();
    } else if (polysA.empty()) {
        finalPolys = std::move(polysB);
    } else if (polysB.empty()) {
        finalPolys = std::move(polysA);
    } else if (!boxesOverlap(boxA, boxB, 1e-9)) {
        finalPolys = std::move(polysA);
        finalPolys.insert(finalPolys.end(),
                          std::make_move_iterator(polysB.begin()),
                          std::make_move_iterator(polysB.end()));
    } else {
        finalPolys = computeUnion(std::move(polysA), std::move(polysB));
    }

    std::vector<Vec3> outVertices;
    std::vector<std::array<int, 3>> outFaces;
    outVertices.reserve(finalPolys.size() * 3);
    outFaces.reserve(finalPolys.size());
    polygonsToMesh(finalPolys, outVertices, outFaces);

    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(10);
    std::cout << outVertices.size() << " " << outFaces.size() << "\n";
    for (const auto& v : outVertices) {
        std::cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& f : outFaces) {
        std::cout << f[0] << " " << f[1] << " " << f[2] << "\n";
    }
    return 0;
}
