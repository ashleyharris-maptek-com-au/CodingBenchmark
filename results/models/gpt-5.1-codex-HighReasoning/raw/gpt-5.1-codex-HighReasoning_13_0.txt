#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <random>
#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

class HashTable {
public:
    void init(std::size_t expected) {
        if (expected == 0) {
            expected = 1;
        }
        std::size_t need = expected + expected / 2 + 1;
        if (need < 8) {
            need = 8;
        }
        std::size_t desired = 1;
        while (desired < need) {
            desired <<= 1;
        }
        if (desired > capacity) {
            capacity = desired;
            key1.assign(capacity, 0);
            key2.assign(capacity, 0);
            pos.assign(capacity, 0);
            matched.assign(capacity, 0);
            stamp.assign(capacity, 0);
        }
        mask = capacity - 1;
        ++epoch;
        if (epoch == 0) {
            std::fill(stamp.begin(), stamp.end(), 0);
            epoch = 1;
        }
        slots.clear();
        slots.reserve(expected);
    }

    void addIfAbsent(uint64_t h1, uint64_t h2, std::size_t position) {
        std::size_t idx = hashIndex(h1, h2);
        while (stamp[idx] == epoch) {
            if (key1[idx] == h1 && key2[idx] == h2) {
                return;
            }
            idx = (idx + 1) & mask;
        }
        stamp[idx] = epoch;
        key1[idx] = h1;
        key2[idx] = h2;
        pos[idx] = position;
        matched[idx] = 1;
        slots.push_back(idx);
    }

    bool tryAdvance(uint64_t h1, uint64_t h2, uint32_t expected) {
        std::size_t idx = hashIndex(h1, h2);
        while (true) {
            if (stamp[idx] != epoch) {
                return false;
            }
            if (key1[idx] == h1 && key2[idx] == h2) {
                if (matched[idx] == expected) {
                    matched[idx] = expected + 1;
                    return true;
                }
                return false;
            }
            idx = (idx + 1) & mask;
        }
    }

    const std::vector<std::size_t>& getUsedSlots() const {
        return slots;
    }

    uint32_t getMatched(std::size_t slot) const {
        return matched[slot];
    }

    std::size_t getPos(std::size_t slot) const {
        return pos[slot];
    }

private:
    std::size_t hashIndex(uint64_t h1, uint64_t h2) const {
        uint64_t mixed = h1 ^ (h2 * 0x9e3779b97f4a7c15ULL);
        return static_cast<std::size_t>(mixed & mask);
    }

    std::size_t capacity = 0;
    std::size_t mask = 0;
    uint32_t epoch = 1;
    std::vector<uint64_t> key1;
    std::vector<uint64_t> key2;
    std::vector<std::size_t> pos;
    std::vector<uint32_t> matched;
    std::vector<uint32_t> stamp;
    std::vector<std::size_t> slots;
};

class AdaptiveLCS {
public:
    AdaptiveLCS(const std::vector<std::string>& data, int pivotIdx, std::vector<int> orderList)
        : strings(data), pivotIndex(pivotIdx), order(std::move(orderList)) {
        std::random_device rd;
        std::mt19937_64 rng(rd());
        base1 = pickBase(rng);
        do {
            base2 = pickBase(rng);
        } while (base2 == base1);
    }

    std::string solve(std::size_t minLen) {
        std::string best;
        if (minLen == 0) {
            return best;
        }
        std::size_t lo = 1;
        std::size_t hi = minLen;
        while (lo <= hi) {
            std::size_t mid = lo + (hi - lo) / 2;
            std::string candidate;
            if (check(mid, candidate)) {
                best = std::move(candidate);
                lo = mid + 1;
            } else {
                if (mid == 0) {
                    break;
                }
                hi = mid - 1;
            }
        }
        return best;
    }

private:
    const std::vector<std::string>& strings;
    int pivotIndex;
    std::vector<int> order;
    HashTable table;
    uint64_t base1 = 0;
    uint64_t base2 = 0;

    static uint64_t pickBase(std::mt19937_64& rng) {
        std::uniform_int_distribution<uint64_t> dist(1315423911ULL, (1ULL << 61) - 1);
        return dist(rng) | 1ULL;
    }

    static uint64_t encodeChar(unsigned char ch) {
        return static_cast<uint64_t>(ch) + 1ULL;
    }

    static uint64_t fastPow(uint64_t base, std::size_t exp) {
        uint64_t result = 1;
        while (exp) {
            if (exp & 1) {
                result *= base;
            }
            base *= base;
            exp >>= 1;
        }
        return result;
    }

    bool processString(const std::string& s, std::size_t len, uint32_t processed,
                       uint64_t pow1, uint64_t pow2) {
        std::size_t n = s.size();
        if (n < len) {
            return false;
        }
        uint64_t h1 = 0;
        uint64_t h2 = 0;
        bool progressed = false;

        for (std::size_t i = 0; i < len; ++i) {
            uint64_t val = encodeChar(static_cast<unsigned char>(s[i]));
            h1 = h1 * base1 + val;
            h2 = h2 * base2 + val;
        }
        if (table.tryAdvance(h1, h2, processed)) {
            progressed = true;
        }
        for (std::size_t i = len; i < n; ++i) {
            uint64_t outVal = encodeChar(static_cast<unsigned char>(s[i - len]));
            h1 -= outVal * pow1;
            h2 -= outVal * pow2;
            uint64_t inVal = encodeChar(static_cast<unsigned char>(s[i]));
            h1 = h1 * base1 + inVal;
            h2 = h2 * base2 + inVal;
            if (table.tryAdvance(h1, h2, processed)) {
                progressed = true;
            }
        }
        return progressed;
    }

    bool check(std::size_t len, std::string& out) {
        if (len == 0) {
            out.clear();
            return true;
        }

        const std::string& pivot = strings[pivotIndex];
        if (pivot.size() < len) {
            return false;
        }
        for (int idx : order) {
            if (strings[idx].size() < len) {
                return false;
            }
        }

        std::size_t possible = pivot.size() - len + 1;
        table.init(possible);

        uint64_t pow1 = fastPow(base1, len - 1);
        uint64_t pow2 = fastPow(base2, len - 1);

        uint64_t h1 = 0;
        uint64_t h2 = 0;
        for (std::size_t i = 0; i < len; ++i) {
            uint64_t val = encodeChar(static_cast<unsigned char>(pivot[i]));
            h1 = h1 * base1 + val;
            h2 = h2 * base2 + val;
        }
        table.addIfAbsent(h1, h2, 0);

        for (std::size_t i = len; i < pivot.size(); ++i) {
            uint64_t outVal = encodeChar(static_cast<unsigned char>(pivot[i - len]));
            h1 -= outVal * pow1;
            h2 -= outVal * pow2;
            uint64_t inVal = encodeChar(static_cast<unsigned char>(pivot[i]));
            h1 = h1 * base1 + inVal;
            h2 = h2 * base2 + inVal;
            table.addIfAbsent(h1, h2, i - len + 1);
        }

        uint32_t processed = 1;
        for (int idx : order) {
            if (!processString(strings[idx], len, processed, pow1, pow2)) {
                return false;
            }
            ++processed;
        }

        uint32_t target = static_cast<uint32_t>(strings.size());
        for (std::size_t slot : table.getUsedSlots()) {
            if (table.getMatched(slot) == target) {
                std::size_t start = table.getPos(slot);
                out = pivot.substr(start, len);
                return true;
            }
        }
        return false;
    }
};

std::string bruteForceLCS(const std::vector<std::string>& strings,
                          int pivotIndex,
                          const std::vector<int>& order) {
    const std::string& pivot = strings[pivotIndex];
    int n = static_cast<int>(pivot.size());
    if (n == 0) {
        return "";
    }
    for (int len = n; len >= 1; --len) {
        std::unordered_set<std::string> seen;
        int variants = n - len + 1;
        seen.reserve(std::max(1, variants * 2));
        for (int start = 0; start + len <= n; ++start) {
            std::string sub = pivot.substr(static_cast<std::size_t>(start),
                                           static_cast<std::size_t>(len));
            if (!seen.insert(sub).second) {
                continue;
            }
            bool ok = true;
            for (int idx : order) {
                if (strings[idx].find(sub) == std::string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                return sub;
            }
        }
    }
    return "";
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    if (!(std::cin >> N)) {
        return 0;
    }
    std::string dummy;
    std::getline(std::cin, dummy);  // consume endline after N

    if (N <= 0) {
        std::cout << "\n";
        return 0;
    }

    std::vector<std::string> strings(N);
    for (int i = 0; i < N; ++i) {
        std::getline(std::cin, strings[i]);
    }

    std::size_t minLen = std::numeric_limits<std::size_t>::max();
    unsigned long long totalLen = 0;
    int pivotIndex = 0;
    for (int i = 0; i < N; ++i) {
        totalLen += strings[i].size();
        if (strings[i].size() < minLen) {
            minLen = strings[i].size();
            pivotIndex = i;
        }
    }

    if (minLen == std::numeric_limits<std::size_t>::max()) {
        std::cout << "\n";
        return 0;
    }

    if (minLen == 0) {
        std::cout << "\n";
        return 0;
    }

    std::vector<int> order;
    order.reserve(N - 1);
    for (int i = 0; i < N; ++i) {
        if (i != pivotIndex) {
            order.push_back(i);
        }
    }
    std::sort(order.begin(), order.end(),
              [&](int a, int b) { return strings[a].size() < strings[b].size(); });

    std::string answer;
    constexpr unsigned long long SMALL_INPUT_THRESHOLD = 5000ULL;
    if (totalLen <= SMALL_INPUT_THRESHOLD) {
        answer = bruteForceLCS(strings, pivotIndex, order);
    } else {
        AdaptiveLCS solver(strings, pivotIndex, order);
        answer = solver.solve(minLen);
    }

    std::cout << answer << '\n';
    return 0;
}
