OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1
%glsl450 = OpExtInstImport "GLSL.std.450"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_4 = OpConstant %uint 4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_input_v4float = OpTypePointer Input %v4float
%arr_v3float = OpTypeArray %v3float %uint_4
%ptr_input_arr_v3float = OpTypePointer Input %arr_v3float
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_output_v3float = OpTypePointer Output %v3float
%ptr_private_v4float = OpTypePointer Private %v4float
%ptr_private_v3float = OpTypePointer Private %v3float
%ptr_private_v2float = OpTypePointer Private %v2float
%gl_in = OpVariable %ptr_input_arr_pv Input
%worldPos = OpVariable %ptr_input_arr_v3float Input
%normal = OpVariable %ptr_input_arr_v3float Input
%color = OpVariable %ptr_input_arr_v3float Input
%gl_out = OpVariable %ptr_output_pv Output
%outColor = OpVariable %ptr_output_v3float Output
%g_startClip = OpVariable %ptr_private_v4float Private
%g_diffClip = OpVariable %ptr_private_v4float Private
%g_normStart = OpVariable %ptr_private_v2float Private
%g_normDelta = OpVariable %ptr_private_v2float Private
%g_colorStart = OpVariable %ptr_private_v3float Private
%g_colorDiff = OpVariable %ptr_private_v3float Private
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_neg1 = OpConstant %float -1
%float_thickness = OpConstant %float 0.02
%float_oneThird = OpConstant %float 0.33333334
%float_twoThird = OpConstant %float 0.66666669
%float_eps = OpConstant %float 1e-06
%fn_main = OpTypeFunction %void
%fn_emit = OpTypeFunction %void %float %float

%emitVertex = OpFunction %void None %fn_emit
%param_t = OpFunctionParameter %float
%param_sign = OpFunctionParameter %float
%label_emit = OpLabel
%start = OpLoad %v4float %g_startClip
%diff = OpLoad %v4float %g_diffClip
%scaledDiff = OpVectorTimesScalar %v4float %diff %param_t
%base = OpFAdd %v4float %start %scaledDiff
%base_w = OpCompositeExtract %float %base 3
%normStartVal = OpLoad %v2float %g_normStart
%normDeltaVal = OpLoad %v2float %g_normDelta
%scaledNormDelta = OpVectorTimesScalar %v2float %normDeltaVal %param_t
%normMixRaw = OpFAdd %v2float %normStartVal %scaledNormDelta
%dotNorm = OpDot %float %normMixRaw %normMixRaw
%lenNorm = OpExtInst %float %glsl450 Sqrt %dotNorm
%lenNormClamped = OpExtInst %float %glsl450 FMax %lenNorm %float_eps
%invLenNorm = OpFDiv %float %float_1 %lenNormClamped
%normMix = OpVectorTimesScalar %v2float %normMixRaw %invLenNorm
%signThickness = OpFMul %float %param_sign %float_thickness
%offsetNDC = OpVectorTimesScalar %v2float %normMix %signThickness
%offsetClipXY = OpVectorTimesScalar %v2float %offsetNDC %base_w
%offsetX = OpCompositeExtract %float %offsetClipXY 0
%offsetY = OpCompositeExtract %float %offsetClipXY 1
%offset4 = OpCompositeConstruct %v4float %offsetX %offsetY %float_0 %float_0
%newClip = OpFAdd %v4float %base %offset4
%out_pos_ptr = OpAccessChain %ptr_output_v4float %gl_out %int_0
OpStore %out_pos_ptr %newClip
%colorStartVal = OpLoad %v3float %g_colorStart
%colorDiffVal = OpLoad %v3float %g_colorDiff
%scaledColor = OpVectorTimesScalar %v3float %colorDiffVal %param_t
%colorVal = OpFAdd %v3float %colorStartVal %scaledColor
OpStore %outColor %colorVal
OpEmitVertex
OpReturn
OpFunctionEnd

%main = OpFunction %void None %fn_main
%label_main = OpLabel
%pos0_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%pos2_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos3_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %pos3_ptr
%pos0_xyz = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
%pos0_w = OpCompositeExtract %float %pos0 3
%invw0 = OpFDiv %float %float_1 %pos0_w
%ndc0 = OpVectorTimesScalar %v3float %pos0_xyz %invw0
%ndc0_xy = OpVectorShuffle %v2float %ndc0 %ndc0 0 1
%pos1_xyz = OpVectorShuffle %v3float %pos1 %pos1 0 1 2
%pos1_w = OpCompositeExtract %float %pos1 3
%invw1 = OpFDiv %float %float_1 %pos1_w
%ndc1 = OpVectorTimesScalar %v3float %pos1_xyz %invw1
%ndc1_xy = OpVectorShuffle %v2float %ndc1 %ndc1 0 1
%pos2_xyz = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
%pos2_w = OpCompositeExtract %float %pos2 3
%invw2 = OpFDiv %float %float_1 %pos2_w
%ndc2 = OpVectorTimesScalar %v3float %pos2_xyz %invw2
%ndc2_xy = OpVectorShuffle %v2float %ndc2 %ndc2 0 1
%pos3_xyz = OpVectorShuffle %v3float %pos3 %pos3 0 1 2
%pos3_w = OpCompositeExtract %float %pos3 3
%invw3 = OpFDiv %float %float_1 %pos3_w
%ndc3 = OpVectorTimesScalar %v3float %pos3_xyz %invw3
%ndc3_xy = OpVectorShuffle %v2float %ndc3 %ndc3 0 1
%dirStart = OpFSub %v2float %ndc2_xy %ndc0_xy
%dotStart = OpDot %float %dirStart %dirStart
%sqrtStart = OpExtInst %float %glsl450 Sqrt %dotStart
%lenStart = OpExtInst %float %glsl450 FMax %sqrtStart %float_eps
%invLenStart = OpFDiv %float %float_1 %lenStart
%tanStart = OpVectorTimesScalar %v2float %dirStart %invLenStart
%dirEnd = OpFSub %v2float %ndc3_xy %ndc1_xy
%dotEnd = OpDot %float %dirEnd %dirEnd
%sqrtEnd = OpExtInst %float %glsl450 Sqrt %dotEnd
%lenEnd = OpExtInst %float %glsl450 FMax %sqrtEnd %float_eps
%invLenEnd = OpFDiv %float %float_1 %lenEnd
%tanEnd = OpVectorTimesScalar %v2float %dirEnd %invLenEnd
%tanStart_x = OpCompositeExtract %float %tanStart 0
%tanStart_y = OpCompositeExtract %float %tanStart 1
%neg_tanStart_y = OpFNegate %float %tanStart_y
%normStart = OpCompositeConstruct %v2float %neg_tanStart_y %tanStart_x
%tanEnd_x = OpCompositeExtract %float %tanEnd 0
%tanEnd_y = OpCompositeExtract %float %tanEnd 1
%neg_tanEnd_y = OpFNegate %float %tanEnd_y
%normEnd = OpCompositeConstruct %v2float %neg_tanEnd_y %tanEnd_x
%normDelta = OpFSub %v2float %normEnd %normStart
%color1_ptr = OpAccessChain %ptr_input_v3float %color %int_1
%color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_input_v3float %color %int_2
%color2 = OpLoad %v3float %color2_ptr
%colorDiff = OpFSub %v3float %color2 %color1
%diffClip = OpFSub %v4float %pos2 %pos1
OpStore %g_startClip %pos1
OpStore %g_diffClip %diffClip
OpStore %g_normStart %normStart
OpStore %g_normDelta %normDelta
OpStore %g_colorStart %color1
OpStore %g_colorDiff %colorDiff
OpFunctionCall %void %emitVertex %float_0 %float_1
OpFunctionCall %void %emitVertex %float_0 %float_neg1
OpFunctionCall %void %emitVertex %float_oneThird %float_1
OpFunctionCall %void %emitVertex %float_oneThird %float_neg1
OpEndPrimitive
OpFunctionCall %void %emitVertex %float_twoThird %float_1
OpFunctionCall %void %emitVertex %float_twoThird %float_neg1
OpFunctionCall %void %emitVertex %float_1 %float_1
OpFunctionCall %void %emitVertex %float_1 %float_neg1
OpEndPrimitive
OpReturn
OpFunctionEnd
