```
OpCapability Geometry
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void        = OpTypeVoid
%functype    = OpTypeFunction %void
%float       = OpTypeFloat 32
%v2float     = OpTypeVector %float 2
%v3float     = OpTypeVector %float 3
%v4float     = OpTypeVector %float 4
%int         = OpTypeInt 32 1
%int_0       = OpConstant %int 0
%int_1       = OpConstant %int 1
%uint        = OpTypeInt 32 0
%uint_2      = OpConstant %uint 2
%float_0     = OpConstant %float 0
%float_1     = OpConstant %float 1
%float_n1    = OpConstant %float -1
%float_0_015 = OpConstant %float 0.015
%float_0_05  = OpConstant %float 0.05
%float_0_75  = OpConstant %float 0.75
%vec2_negpos = OpConstantComposite %v2float %float_n1 %float_1

%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in            = OpVariable %ptr_Input_arr_pv Input
%ptr_Input_v4     = OpTypePointer Input %v4float

%arr_v3_2        = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3= OpTypePointer Input %arr_v3_2
%color_in        = OpVariable %ptr_Input_arr_v3 Input
%ptr_Input_v3    = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_pv    = OpTypePointer Output %gl_PerVertex_out
%gl_out           = OpVariable %ptr_Output_pv Output
%ptr_Output_v4    = OpTypePointer Output %v4float
%ptr_Output_v3    = OpTypePointer Output %v3float
%outColor         = OpVariable %ptr_Output_v3 Output

%main = OpFunction %void None %functype
%entry = OpLabel
%p0_ptr   = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0       = OpLoad %v4float %p0_ptr
%p1_ptr   = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1       = OpLoad %v4float %p1_ptr
%col0_ptr = OpAccessChain %ptr_Input_v3 %color_in %int_0
%col0     = OpLoad %v3float %col0_ptr
%col1_ptr = OpAccessChain %ptr_Input_v3 %color_in %int_1
%col1     = OpLoad %v3float %col1_ptr

%p0_xy    = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy    = OpVectorShuffle %v2float %p1 %p1 0 1
%dir_xy   = OpFSub %v2float %p1_xy %p0_xy
%dir_yx   = OpVectorShuffle %v2float %dir_xy %dir_xy 1 0
%perp_raw = OpFMul %v2float %dir_yx %vec2_negpos
%len2     = OpDot %float %perp_raw %perp_raw
%invLen   = OpExtInst %float %glsl InverseSqrt %len2
%perp_norm  = OpVectorTimesScalar %v2float %perp_raw %invLen
%perp_thick = OpVectorTimesScalar %v2float %perp_norm %float_0_015
%perp_arrow = OpVectorTimesScalar %v2float %perp_norm %float_0_05

%dir4        = OpFSub %v4float %p1 %p0
%dir4_scaled = OpVectorTimesScalar %v4float %dir4 %float_0_75
%mid         = OpFAdd %v4float %p0 %dir4_scaled

%perp_thick_x = OpCompositeExtract %float %perp_thick 0
%perp_thick_y = OpCompositeExtract %float %perp_thick 1
%perp_thick_v4= OpCompositeConstruct %v4float %perp_thick_x %perp_thick_y %float_0 %float_0

%perp_arrow_x = OpCompositeExtract %float %perp_arrow 0
%perp_arrow_y = OpCompositeExtract %float %perp_arrow 1
%perp_arrow_v4= OpCompositeConstruct %v4float %perp_arrow_x %perp_arrow_y %float_0 %float_0

%ptr_gl_pos = OpAccessChain %ptr_Output_v4 %gl_out %int_0

%shaft0 = OpFAdd %v4float %p0 %perp_thick_v4
OpStore %ptr_gl_pos %shaft0
OpStore %outColor %col0
OpEmitVertex

%shaft1 = OpFSub %v4float %p0 %perp_thick_v4
OpStore %ptr_gl_pos %shaft1
OpStore %outColor %col0
OpEmitVertex

%shaft2 = OpFAdd %v4float %mid %perp_thick_v4
OpStore %ptr_gl_pos %shaft2
OpStore %outColor %col0
OpEmitVertex

%shaft3 = OpFSub %v4float %mid %perp_thick_v4
OpStore %ptr_gl_pos %shaft3
OpStore %outColor %col0
OpEmitVertex
OpEndPrimitive

%arrow0 = OpFAdd %v4float %mid %perp_arrow_v4
OpStore %ptr_gl_pos %arrow0
OpStore %outColor %col1
OpEmitVertex

%arrow1 = OpFSub %v4float %mid %perp_arrow_v4
OpStore %ptr_gl_pos %arrow1
OpStore %outColor %col1
OpEmitVertex

OpStore %ptr_gl_pos %p1
OpStore %outColor %col1
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
