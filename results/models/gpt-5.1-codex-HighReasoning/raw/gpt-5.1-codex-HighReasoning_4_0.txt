#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <random>
#include <unordered_map>
#include <unordered_set>
#include <vector>

struct Vec3 {
    double x, y, z;
    Vec3() : x(0.0), y(0.0), z(0.0) {}
    Vec3(double xi, double yi, double zi) : x(xi), y(yi), z(zi) {}
    Vec3 operator+(const Vec3 &o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3 &o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }
    Vec3 &operator+=(const Vec3 &o) {
        x += o.x;
        y += o.y;
        z += o.z;
        return *this;
    }
    double length() const { return std::sqrt(x * x + y * y + z * z); }
};

inline Vec3 operator*(double s, const Vec3 &v) { return Vec3(v.x * s, v.y * s, v.z * s); }

inline double dot(const Vec3 &a, const Vec3 &b) { return a.x * b.x + a.y * b.y + a.z * b.z; }

inline Vec3 cross(const Vec3 &a, const Vec3 &b) {
    return Vec3(a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x);
}

struct Face {
    int a = 0, b = 0, c = 0;
    Vec3 normal;
    double d = 0.0;
    bool deleted = false;
};

Face createFace(int a, int b, int c, const std::vector<Vec3> &pts, const Vec3 &interior, double eps) {
    Face f;
    f.a = a;
    f.b = b;
    f.c = c;
    f.deleted = false;

    auto build = [&](Face &face) -> bool {
        Vec3 ab = pts[face.b] - pts[face.a];
        Vec3 ac = pts[face.c] - pts[face.a];
        Vec3 n = cross(ab, ac);
        double len = n.length();
        if (len < eps) return false;
        n = n / len;
        double dVal = -dot(n, pts[face.a]);
        face.normal = n;
        face.d = dVal;
        return true;
    };

    if (!build(f)) {
        f.deleted = true;
        return f;
    }
    double side = dot(f.normal, interior) + f.d;
    if (side > 0.0) {
        std::swap(f.b, f.c);
        if (!build(f)) {
            f.deleted = true;
            return f;
        }
    }
    return f;
}

bool buildConvexHull(const std::vector<Vec3> &pts, double eps, std::vector<Face> &facesOut, Vec3 &interiorOut) {
    const int n = static_cast<int>(pts.size());
    if (n < 4) return false;
    int p0 = 0;
    int p1 = -1;
    for (int i = 1; i < n; ++i) {
        if ((pts[i] - pts[p0]).length() > eps) {
            p1 = i;
            break;
        }
    }
    if (p1 == -1) return false;

    int p2 = -1;
    Vec3 base = pts[p1] - pts[p0];
    for (int i = 0; i < n; ++i) {
        if (i == p0 || i == p1) continue;
        Vec3 v = pts[i] - pts[p0];
        Vec3 cr = cross(base, v);
        if (cr.length() > eps) {
            p2 = i;
            break;
        }
    }
    if (p2 == -1) return false;

    int p3 = -1;
    Vec3 normal = cross(pts[p1] - pts[p0], pts[p2] - pts[p0]);
    double planeTol = std::max(eps * 10.0, 1e-9);
    for (int i = 0; i < n; ++i) {
        if (i == p0 || i == p1 || i == p2) continue;
        double vol = std::abs(dot(pts[i] - pts[p0], normal));
        if (vol > planeTol) {
            p3 = i;
            break;
        }
    }
    if (p3 == -1) return false;

    interiorOut = (pts[p0] + pts[p1] + pts[p2] + pts[p3]) * 0.25;

    std::vector<Face> faces;
    faces.reserve(n * 2);

    auto addInitial = [&](int a, int b, int c) {
        Face f = createFace(a, b, c, pts, interiorOut, eps);
        if (!f.deleted) faces.push_back(f);
    };

    addInitial(p0, p1, p2);
    addInitial(p0, p3, p1);
    addInitial(p0, p2, p3);
    addInitial(p1, p3, p2);

    std::vector<bool> used(n, false);
    used[p0] = used[p1] = used[p2] = used[p3] = true;

    for (int idx = 0; idx < n; ++idx) {
        if (used[idx]) continue;
        std::vector<int> visible;
        visible.reserve(faces.size());
        for (int fi = 0; fi < static_cast<int>(faces.size()); ++fi) {
            Face &f = faces[fi];
            if (f.deleted) continue;
            double dist = dot(f.normal, pts[idx]) + f.d;
            if (dist > planeTol) {
                f.deleted = true;
                visible.push_back(fi);
            }
        }
        if (visible.empty()) continue;

        std::unordered_set<std::uint64_t> horizon;
        horizon.reserve(visible.size() * 3);

        auto addEdge = [&](int a, int b) {
            std::uint64_t key = (static_cast<std::uint64_t>(static_cast<std::uint32_t>(a)) << 32) |
                                static_cast<std::uint32_t>(b);
            std::uint64_t rev = (static_cast<std::uint64_t>(static_cast<std::uint32_t>(b)) << 32) |
                                static_cast<std::uint32_t>(a);
            auto it = horizon.find(rev);
            if (it != horizon.end())
                horizon.erase(it);
            else
                horizon.insert(key);
        };

        for (int fi : visible) {
            Face &f = faces[fi];
            addEdge(f.a, f.b);
            addEdge(f.b, f.c);
            addEdge(f.c, f.a);
        }

        for (std::uint64_t key : horizon) {
            int a = static_cast<int>(key >> 32);
            int b = static_cast<int>(key & 0xffffffffu);
            Face nf = createFace(a, b, idx, pts, interiorOut, eps);
            if (!nf.deleted) faces.push_back(nf);
        }
    }

    facesOut.clear();
    for (const Face &f : faces) {
        if (!f.deleted) facesOut.push_back(f);
    }
    return facesOut.size() >= 4;
}

struct Container {
    bool hasHull = false;
    std::vector<Face> faces;
    Vec3 minCorner;
    Vec3 maxCorner;
    double tol = 1e-6;

    bool contains(const Vec3 &p) const {
        const double eps = tol;
        if (p.x < minCorner.x - eps || p.x > maxCorner.x + eps) return false;
        if (p.y < minCorner.y - eps || p.y > maxCorner.y + eps) return false;
        if (p.z < minCorner.z - eps || p.z > maxCorner.z + eps) return false;
        if (hasHull) {
            for (const Face &f : faces) {
                double val = dot(f.normal, p) + f.d;
                if (val > eps) return false;
            }
        }
        return true;
    }
};

struct GridInfo {
    Vec3 start;
    Vec3 end;
    int sx = 1;
    int sy = 1;
    int sz = 1;
    long double estimated = 1.0L;
};

GridInfo buildGridInfo(double step, const Vec3 &minCorner, const Vec3 &maxCorner, double margin) {
    GridInfo info;
    Vec3 marginVec(margin, margin, margin);
    info.start = minCorner - marginVec;
    info.end = maxCorner + marginVec;
    auto computeSteps = [&](double len) -> int {
        if (len <= 0.0) return 3;
        int steps = static_cast<int>(std::floor(len / step)) + 3;
        if (steps < 3) steps = 3;
        return steps;
    };
    double lenx = std::max(0.0, info.end.x - info.start.x);
    double leny = std::max(0.0, info.end.y - info.start.y);
    double lenz = std::max(0.0, info.end.z - info.start.z);
    info.sx = computeSteps(lenx);
    info.sy = computeSteps(leny);
    info.sz = computeSteps(lenz);
    info.estimated = static_cast<long double>(info.sx) * info.sy * info.sz;
    return info;
}

struct CellKey {
    long long x;
    long long y;
    long long z;
    bool operator==(const CellKey &o) const noexcept { return x == o.x && y == o.y && z == o.z; }
};

struct CellHash {
    std::size_t operator()(const CellKey &key) const noexcept {
        std::size_t hx = std::hash<long long>()(key.x);
        std::size_t hy = std::hash<long long>()(key.y);
        std::size_t hz = std::hash<long long>()(key.z);
        return hx ^ (hy << 1) ^ (hz << 2);
    }
};

class SpatialHash {
public:
    SpatialHash(double cellSize, double minSpacing, std::vector<Vec3> &ref)
        : cell(std::max(cellSize, 1e-6)), data(ref) {
        double slack = std::max(1e-7, minSpacing * 1e-4);
        double safe = std::max(0.0, minSpacing - slack);
        if (safe <= 0.0) safe = minSpacing;
        minDist2 = safe * safe;
        buckets.reserve(1024);
    }

    bool canPlace(const Vec3 &p) {
        long long ix = static_cast<long long>(std::floor(p.x / cell));
        long long iy = static_cast<long long>(std::floor(p.y / cell));
        long long iz = static_cast<long long>(std::floor(p.z / cell));
        for (long long dx = -1; dx <= 1; ++dx) {
            for (long long dy = -1; dy <= 1; ++dy) {
                for (long long dz = -1; dz <= 1; ++dz) {
                    CellKey key{ix + dx, iy + dy, iz + dz};
                    auto it = buckets.find(key);
                    if (it == buckets.end()) continue;
                    for (int idx : it->second) {
                        Vec3 diff = data[idx] - p;
                        double dist2 = dot(diff, diff);
                        if (dist2 < minDist2) return false;
                    }
                }
            }
        }
        return true;
    }

    void insert(const Vec3 &p, int idx) {
        long long ix = static_cast<long long>(std::floor(p.x / cell));
        long long iy = static_cast<long long>(std::floor(p.y / cell));
        long long iz = static_cast<long long>(std::floor(p.z / cell));
        CellKey key{ix, iy, iz};
        buckets[key].push_back(idx);
    }

private:
    double cell;
    double minDist2;
    std::vector<Vec3> &data;
    std::unordered_map<CellKey, std::vector<int>, CellHash> buckets;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V;
    double edgeLength;
    if (!(std::cin >> V >> edgeLength)) {
        return 0;
    }
    if (V <= 0 || edgeLength <= 0.0) {
        std::cout << 0 << "\n";
        return 0;
    }
    std::vector<Vec3> vertices(V);
    for (int i = 0; i < V; ++i) {
        std::cin >> vertices[i].x >> vertices[i].y >> vertices[i].z;
    }

    Vec3 minCorner = vertices[0];
    Vec3 maxCorner = vertices[0];
    for (const Vec3 &p : vertices) {
        minCorner.x = std::min(minCorner.x, p.x);
        minCorner.y = std::min(minCorner.y, p.y);
        minCorner.z = std::min(minCorner.z, p.z);
        maxCorner.x = std::max(maxCorner.x, p.x);
        maxCorner.y = std::max(maxCorner.y, p.y);
        maxCorner.z = std::max(maxCorner.z, p.z);
    }

    Vec3 diagVec = maxCorner - minCorner;
    double diag = diagVec.length();
    double hullEps = std::max(1e-9, diag * 1e-9);

    std::vector<Face> hullFaces;
    Vec3 interiorPoint;
    bool hullOk = buildConvexHull(vertices, hullEps, hullFaces, interiorPoint);

    Container container;
    container.hasHull = hullOk;
    container.faces = hullFaces;
    container.minCorner = minCorner;
    container.maxCorner = maxCorner;
    container.tol = std::max(1e-6, diag * 1e-9 + 1e-7);

    const double baseScale = edgeLength / std::sqrt(8.0);
    std::array<Vec3, 4> localVerts = {
        Vec3(baseScale, baseScale, baseScale),
        Vec3(baseScale, -baseScale, -baseScale),
        Vec3(-baseScale, baseScale, -baseScale),
        Vec3(-baseScale, -baseScale, baseScale),
    };

    const double radius = std::sqrt(3.0 / 8.0) * edgeLength;
    double minSpacing = std::max(1e-6, 2.0 * radius);
    double candidateStep = minSpacing * 0.5;

    const long double MAX_CANDIDATES = 250000.0L;
    GridInfo grid = buildGridInfo(candidateStep, minCorner, maxCorner, minSpacing);
    for (int iter = 0; iter < 6 && grid.estimated > MAX_CANDIDATES; ++iter) {
        double factor = std::cbrt(static_cast<double>(grid.estimated / MAX_CANDIDATES));
        factor = std::max(factor, 1.01);
        candidateStep *= factor;
        grid = buildGridInfo(candidateStep, minCorner, maxCorner, minSpacing);
    }
    if (grid.estimated > MAX_CANDIDATES) {
        double factor = std::cbrt(static_cast<double>(grid.estimated / MAX_CANDIDATES) + 1e-6);
        candidateStep *= factor;
        grid = buildGridInfo(candidateStep, minCorner, maxCorner, minSpacing);
    }

    std::vector<double> xOffsets = {0.0};
    std::vector<double> yOffsets = {0.0};
    std::vector<double> zOffsets = {0.0};
    if (grid.estimated * 8.0L <= MAX_CANDIDATES) {
        double shift = candidateStep * 0.5;
        xOffsets.push_back(shift);
        yOffsets.push_back(shift);
        zOffsets.push_back(shift);
    }

    long double effectiveBudget =
        grid.estimated * xOffsets.size() * yOffsets.size() * zOffsets.size();
    if (effectiveBudget > MAX_CANDIDATES) {
        xOffsets.resize(1);
        yOffsets.resize(1);
        zOffsets.resize(1);
        effectiveBudget = grid.estimated;
    }

    const std::size_t placementCap = static_cast<std::size_t>(MAX_CANDIDATES);
    std::vector<Vec3> centers;
    centers.reserve(static_cast<std::size_t>(std::min<long double>(effectiveBudget, MAX_CANDIDATES)));

    SpatialHash hash(minSpacing, minSpacing, centers);

    auto tryPlace = [&](const Vec3 &center) -> bool {
        if (!container.contains(center)) return false;
        for (const Vec3 &v : localVerts) {
            if (!container.contains(center + v)) return false;
        }
        if (!hash.canPlace(center)) return false;
        centers.push_back(center);
        hash.insert(center, static_cast<int>(centers.size() - 1));
        return true;
    };

    bool stop = false;
    const double limitX = grid.end.x + candidateStep;
    const double limitY = grid.end.y + candidateStep;
    const double limitZ = grid.end.z + candidateStep;

    for (double ox : xOffsets) {
        for (double oy : yOffsets) {
            for (double oz : zOffsets) {
                if (stop) break;
                for (int ix = 0; ix < grid.sx && !stop; ++ix) {
                    double x = grid.start.x + ox + ix * candidateStep;
                    if (x > limitX) break;
                    for (int iy = 0; iy < grid.sy && !stop; ++iy) {
                        double y = grid.start.y + oy + iy * candidateStep;
                        if (y > limitY) break;
                        for (int iz = 0; iz < grid.sz; ++iz) {
                            double z = grid.start.z + oz + iz * candidateStep;
                            if (z > limitZ) break;
                            if (tryPlace(Vec3(x, y, z)) && centers.size() >= placementCap) {
                                stop = true;
                                break;
                            }
                        }
                        if (centers.size() >= placementCap) {
                            stop = true;
                            break;
                        }
                    }
                }
            }
        }
    }

    if (centers.size() < placementCap) {
        std::random_device rd;
        std::mt19937_64 rng(rd());
        std::uniform_real_distribution<double> distX(grid.start.x, grid.end.x);
        std::uniform_real_distribution<double> distY(grid.start.y, grid.end.y);
        std::uniform_real_distribution<double> distZ(grid.start.z, grid.end.z);
        std::size_t remaining = placementCap - centers.size();
        int randomAttempts = 0;
        if (remaining > 0) {
            randomAttempts = static_cast<int>(std::min<std::size_t>(remaining * 20, static_cast<std::size_t>(20000)));
            randomAttempts = std::max(randomAttempts, 2000);
        }
        for (int attempt = 0;
             attempt < randomAttempts && centers.size() < placementCap;
             ++attempt) {
            Vec3 candidate(distX(rng), distY(rng), distZ(rng));
            tryPlace(candidate);
        }
    }

    std::cout << centers.size() << "\n";
    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(9);
    for (const Vec3 &c : centers) {
        std::cout << c.x << ' ' << c.y << ' ' << c.z << " 0 0 0\n";
    }
    return 0;
}
