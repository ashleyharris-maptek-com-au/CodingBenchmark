use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(token) => token.parse().unwrap_or(0),
        None => {
            println!("0");
            println!();
            return;
        }
    };
    let m: usize = match it.next() {
        Some(token) => token.parse().unwrap_or(0),
        None => {
            println!("0");
            println!();
            return;
        }
    };

    let mut adj_list = vec![Vec::<usize>::new(); n];
    let mut adj_matrix = vec![vec![0usize; n]; n];
    for _ in 0..m {
        let u_token = match it.next() {
            Some(v) => v,
            None => break,
        };
        let v_token = match it.next() {
            Some(v) => v,
            None => break,
        };
        let u: usize = match u_token.parse() {
            Ok(val) => val,
            Err(_) => continue,
        };
        let v: usize = match v_token.parse() {
            Ok(val) => val,
            Err(_) => continue,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj_list[u].push(v);
        adj_list[v].push(u);
        adj_matrix[u][v] += 1;
        adj_matrix[v][u] += 1;
    }

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let target = n / 2;
    let mut heuristic_part = build_initial_partition(&adj_list, &adj_matrix);
    if heuristic_part.iter().filter(|&&p| p == 0).count() != target {
        heuristic_part = (0..n).map(|i| if i < target { 0 } else { 1 }).collect();
    }
    let heuristic_cut = compute_cut(&heuristic_part, &adj_list);

    let degrees: Vec<usize> = adj_list.iter().map(|nbrs| nbrs.len()).collect();
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&a, &b| {
        degrees[b]
            .cmp(&degrees[a])
            .then_with(|| a.cmp(&b))
    });

    let mut state = SearchState {
        n,
        target,
        order,
        adj: &adj_list,
        neighbor_counts: vec![[0usize; 2]; n],
        part: vec![2u8; n],
        best_cut: heuristic_cut as i64,
        best_assignment: heuristic_part.clone(),
    };

    if heuristic_cut > 0 {
        if !state.order.is_empty() {
            let root = state.order[0];
            state.part[root] = 0;
            let adj_ref = state.adj;
            for &u in &adj_ref[root] {
                state.neighbor_counts[u][0] += 1;
            }
            state.dfs(1, 1, 0, 0);
        }
    }

    let final_cut = state.best_cut as usize;
    let mut part_a: Vec<String> = Vec::with_capacity(target);
    for (v, &p) in state.best_assignment.iter().enumerate() {
        if p == 0 {
            part_a.push(v.to_string());
        }
    }
    println!("{}", final_cut);
    if part_a.is_empty() {
        println!();
    } else {
        println!("{}", part_a.join(" "));
    }
}

fn compute_cut(part: &[u8], adj_list: &[Vec<usize>]) -> usize {
    let mut cut = 0usize;
    for (u, neighbors) in adj_list.iter().enumerate() {
        for &v in neighbors {
            if v > u && part[u] != part[v] {
                cut += 1;
            }
        }
    }
    cut
}

fn greedy_initial(adj_list: &[Vec<usize>]) -> Vec<u8> {
    let n = adj_list.len();
    if n == 0 {
        return Vec::new();
    }
    let target = n / 2;
    let mut part = vec![2u8; n];
    let mut counts = [0usize; 2];
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&a, &b| {
        adj_list[b]
            .len()
            .cmp(&adj_list[a].len())
            .then_with(|| a.cmp(&b))
    });

    for &v in &order {
        let mut best_part = None;
        let mut best_cost = usize::MAX;
        for p in 0..2 {
            if counts[p] >= target {
                continue;
            }
            let mut cost = 0usize;
            for &u in &adj_list[v] {
                if part[u] == (1 - p as u8) {
                    cost += 1;
                }
            }
            if best_part.is_none() || cost < best_cost {
                best_cost = cost;
                best_part = Some(p as u8);
            }
        }
        if let Some(chosen) = best_part {
            part[v] = chosen;
            counts[chosen as usize] += 1;
        }
    }

    for v in 0..n {
        if part[v] == 2 {
            let assign = if counts[0] < target { 0 } else { 1 };
            part[v] = assign;
            counts[assign as usize] += 1;
        }
    }
    part
}

fn build_initial_partition(adj_list: &[Vec<usize>], adj_matrix: &[Vec<usize>]) -> Vec<u8> {
    let mut part = greedy_initial(adj_list);
    kernighan_lin(&mut part, adj_list, adj_matrix);
    part
}

fn compute_d(part: &[u8], adj_list: &[Vec<usize>]) -> Vec<i64> {
    let n = part.len();
    let mut d = vec![0i64; n];
    for v in 0..n {
        let mut external = 0i64;
        let mut internal = 0i64;
        for &u in &adj_list[v] {
            if part[u] == part[v] {
                internal += 1;
            } else {
                external += 1;
            }
        }
        d[v] = external - internal;
    }
    d
}

fn kernighan_lin(part: &mut Vec<u8>, adj_list: &[Vec<usize>], adj_matrix: &[Vec<usize>]) {
    let n = part.len();
    if n == 0 {
        return;
    }
    let k = n / 2;
    if k == 0 {
        return;
    }
    let max_passes = 6;
    for _ in 0..max_passes {
        let mut work_part = part.clone();
        let mut d = compute_d(&work_part, adj_list);
        let mut locked = vec![false; n];
        let mut seq: Vec<(usize, usize, i64)> = Vec::new();
        let mut cum_gain = 0i64;
        let mut best_gain = i64::MIN;
        let mut best_idx: isize = -1;

        for step in 0..k {
            let mut max_gain = i64::MIN;
            let mut best_pair = None;
            for a in 0..n {
                if work_part[a] != 0 || locked[a] {
                    continue;
                }
                for b in 0..n {
                    if work_part[b] != 1 || locked[b] {
                        continue;
                    }
                    let mut gain = d[a] + d[b];
                    if adj_matrix[a][b] > 0 {
                        gain -= 2 * adj_matrix[a][b] as i64;
                    }
                    if gain > max_gain {
                        max_gain = gain;
                        best_pair = Some((a, b));
                    }
                }
            }
            let (a, b) = match best_pair {
                Some(pair) => pair,
                None => break,
            };
            locked[a] = true;
            locked[b] = true;
            seq.push((a, b, max_gain));
            cum_gain += max_gain;
            if cum_gain > best_gain {
                best_gain = cum_gain;
                best_idx = step as isize;
            }

            work_part[a] = 1;
            work_part[b] = 0;
            for &v in &adj_list[a] {
                if locked[v] {
                    continue;
                }
                if work_part[v] == 0 {
                    d[v] += 2;
                } else {
                    d[v] -= 2;
                }
            }
            for &v in &adj_list[b] {
                if locked[v] {
                    continue;
                }
                if work_part[v] == 1 {
                    d[v] += 2;
                } else {
                    d[v] -= 2;
                }
            }
        }

        if best_gain <= 0 || best_idx < 0 {
            break;
        }
        for i in 0..=best_idx as usize {
            let (a, b, _) = seq[i];
            part[a] = 1;
            part[b] = 0;
        }
    }
}

struct SearchState<'a> {
    n: usize,
    target: usize,
    order: Vec<usize>,
    adj: &'a [Vec<usize>],
    neighbor_counts: Vec<[usize; 2]>,
    part: Vec<u8>,
    best_cut: i64,
    best_assignment: Vec<u8>,
}

impl<'a> SearchState<'a> {
    fn dfs(&mut self, idx: usize, size_a: usize, size_b: usize, current_cut: usize) {
        if (current_cut as i64) >= self.best_cut {
            return;
        }
        if idx == self.n {
            if size_a == self.target {
                self.best_cut = current_cut as i64;
                self.best_assignment = self.part.clone();
            }
            return;
        }
        if size_a > self.target || size_b > self.target {
            return;
        }

        let remaining = self.n - idx;
        let slots_a = self.target - size_a;
        let slots_b = self.target - size_b;
        if slots_a > remaining || slots_b > remaining {
            return;
        }

        let lower = self.lower_bound(idx, slots_a, current_cut);
        if lower >= self.best_cut {
            return;
        }

        let v = self.order[idx];
        debug_assert!(self.part[v] == 2);

        let mut options = [0u8; 2];
        let mut opt_len = 0usize;
        if slots_a == remaining {
            options[0] = 0;
            opt_len = 1;
        } else if slots_b == remaining {
            options[0] = 1;
            opt_len = 1;
        } else {
            if slots_a > 0 {
                options[opt_len] = 0;
                opt_len += 1;
            }
            if slots_b > 0 {
                options[opt_len] = 1;
                opt_len += 1;
            }
        }
        if opt_len == 0 {
            return;
        }

        if opt_len == 2 {
            let delta0 = self.neighbor_counts[v][1 - options[0] as usize];
            let delta1 = self.neighbor_counts[v][1 - options[1] as usize];
            if delta0 > delta1 {
                options.swap(0, 1);
            }
        }

        let adj_ref = self.adj;
        for i in 0..opt_len {
            let part_value = options[i];
            let part_idx = part_value as usize;
            let delta = self.neighbor_counts[v][1 - part_idx];

            self.part[v] = part_value;
            for &u in &adj_ref[v] {
                self.neighbor_counts[u][part_idx] += 1;
            }

            let (next_a, next_b) = if part_value == 0 {
                (size_a + 1, size_b)
            } else {
                (size_a, size_b + 1)
            };
            self.dfs(idx + 1, next_a, next_b, current_cut + delta);

            for &u in &adj_ref[v] {
                self.neighbor_counts[u][part_idx] -= 1;
            }
            self.part[v] = 2;
        }
    }

    fn lower_bound(&self, idx: usize, slots_a: usize, current_cut: usize) -> i64 {
        let remaining = self.n - idx;
        if remaining == 0 {
            return current_cut as i64;
        }
        let mut base = 0i64;
        let mut diffs: Vec<i64> = Vec::with_capacity(remaining);
        for t in idx..self.n {
            let v = self.order[t];
            let cost_b = self.neighbor_counts[v][0] as i64;
            let cost_a = self.neighbor_counts[v][1] as i64;
            base += cost_b;
            diffs.push(cost_a - cost_b);
        }
        if slots_a > diffs.len() {
            return i64::MAX / 4;
        }
        let mut extra = base;
        if slots_a > 0 {
            if slots_a == diffs.len() {
                extra += diffs.iter().sum::<i64>();
            } else {
                let k_index = slots_a - 1;
                diffs.select_nth_unstable(k_index);
                extra += diffs[..slots_a].iter().sum::<i64>();
            }
        }
        current_cut as i64 + extra
    }
}
