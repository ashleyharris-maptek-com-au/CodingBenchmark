%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%inWorldPos = OpVariable %_ptr_Input_v3float Input
%inNormal = OpVariable %_ptr_Input_v3float Input
%inUV = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0

%float_0_7 = OpConstant %float 0.699999988

%float_0_8 = OpConstant %float 0.800000012
%float_0_1 = OpConstant %float 0.100000001
%float_0_3_c = OpConstant %float 0.300000012
%v2const_half = OpConstantComposite %v2float %float_0_5 %float_0_5
%v3dotColor = OpConstantComposite %v3float %float_1 %float_0_8 %float_0
%v3bgColor = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_3_c
%main = OpFunction %void None %3
%5 = OpLabel
%worldPosVal = OpLoad %v3float %inWorldPos
%normalVal = OpLoad %v3float %inNormal
%uvVal = OpLoad %v2float %inUV
%norm = OpExtInst %v3float %glsl Normalize %normalVal
%lightPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightVec4 = OpLoad %v4float %lightPtr
%lightVec3 = OpVectorShuffle %v3float %lightVec4 %lightVec4 0 1 2
%lightDirVec = OpFSub %v3float %lightVec3 %worldPosVal
%lightDir = OpExtInst %v3float %glsl Normalize %lightDirVec
%dotNL = OpDot %float %norm %lightDir
%diffClamp = OpExtInst %float %glsl FMax %dotNL %float_0
%diffScaled = OpFMul %float %diffClamp %float_0_7
%shade = OpFAdd %float %diffScaled %float_0_3
%uvScaled = OpVectorTimesScalar %v2float %uvVal %float_6
%cell = OpExtInst %v2float %glsl Fract %uvScaled
%offset = OpFSub %v2float %cell %v2const_half
%dist = OpExtInst %float %glsl Length %offset
%inDot = OpFOrdLessThan %bool %dist %float_0_3
%mask = OpSelect %float %inDot %float_1 %float_0
%colorDiff = OpFSub %v3float %v3dotColor %v3bgColor
%scaledDiff = OpVectorTimesScalar %v3float %colorDiff %mask
%baseColor = OpFAdd %v3float %v3bgColor %scaledDiff
%shadedColor = OpVectorTimesScalar %v3float %baseColor %shade
%finalColor = OpCompositeConstruct %v4float %shadedColor %float_1
OpStore %outColor %finalColor
