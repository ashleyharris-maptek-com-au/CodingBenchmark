```rust
use std::cmp::Ordering;
use std::io::{self, Read};

fn main() {
    // --- Read entire stdin ---
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            println!("0\n");
            return;
        }
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u >= n || v >= n {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    if n == 0 {
        println!("0\n");
        return;
    }

    // --- Greedy construction (degree ordering) ---
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&a, &b| {
        let deg_a = adj[a].len();
        let deg_b = adj[b].len();
        match deg_b.cmp(&deg_a) {
            Ordering::Equal => a.cmp(&b),
            other => other,
        }
    });

    let mut dominated = vec![false; n];
    let mut selected = vec![false; n];

    for &v in &order {
        if dominated[v] {
            continue;
        }
        select_vertex(v, &adj, &mut dominated, &mut selected);
    }

    // Safety pass: ensure all dominated (handles isolated vertices)
    for v in 0..n {
        if !dominated[v] {
            select_vertex(v, &adj, &mut dominated, &mut selected);
        }
    }

    // --- Redundancy elimination ---
    let mut coverage = vec![0usize; n];
    for v in 0..n {
        if selected[v] {
            coverage[v] += 1;
            for &nbr in &adj[v] {
                coverage[nbr] += 1;
            }
        }
    }

    let snapshot: Vec<usize> = selected
        .iter()
        .enumerate()
        .filter_map(|(idx, &sel)| if sel { Some(idx) } else { None })
        .collect();

    for &v in &snapshot {
        if !selected[v] {
            continue;
        }
        if coverage[v] <= 1 {
            continue;
        }
        let mut removable = true;
        for &nbr in &adj[v] {
            if coverage[nbr] <= 1 {
                removable = false;
                break;
            }
        }
        if removable {
            selected[v] = false;
            coverage[v] -= 1;
            for &nbr in &adj[v] {
                coverage[nbr] -= 1;
            }
        }
    }

    // --- Collect result and output ---
    let result: Vec<usize> = (0..n).filter(|&v| selected[v]).collect();

    let mut out = String::new();
    out.push_str(&format!("{}\n", result.len()));
    if !result.is_empty() {
        for (i, v) in result.iter().enumerate() {
            if i > 0 {
                out.push(' ');
            }
            out.push_str(&v.to_string());
        }
        out.push('\n');
    } else {
        out.push('\n');
    }
    print!("{out}");
}

fn select_vertex(
    v: usize,
    adj: &[Vec<usize>],
    dominated: &mut [bool],
    selected: &mut [bool],
) {
    if selected[v] {
        return;
    }
    selected[v] = true;
    if !dominated[v] {
        dominated[v] = true;
    }
    for &nbr in &adj[v] {
        if !dominated[nbr] {
            dominated[nbr] = true;
        }
    }
}
```
