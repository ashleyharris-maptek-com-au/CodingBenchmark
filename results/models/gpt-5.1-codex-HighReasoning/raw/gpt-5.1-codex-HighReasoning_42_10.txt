OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %arr_v3_in ArrayStride 16
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void          = OpTypeVoid
%func          = OpTypeFunction %void
%bool          = OpTypeBool
%int           = OpTypeInt 32 1
%uint          = OpTypeInt 32 0
%float         = OpTypeFloat 32
%v2float       = OpTypeVector %float 2
%v3float       = OpTypeVector %float 3
%v4float       = OpTypeVector %float 4
%gl_PerVertex_in  = OpTypeStruct %v4float
%uint_4        = OpConstant %uint 4
%arr_pv_in     = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_input_v4  = OpTypePointer Input %v4float
%gl_in         = OpVariable %ptr_input_arr_pv Input
%arr_v3_in     = OpTypeArray %v3float %uint_4
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_input_v3  = OpTypePointer Input %v3float
%in_color      = OpVariable %ptr_input_arr_v3 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_glpv = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4 = OpTypePointer Output %v4float
%gl_out        = OpVariable %ptr_output_glpv Output
%ptr_output_v3 = OpTypePointer Output %v3float
%outColor      = OpVariable %ptr_output_v3 Output
%ptr_function_int = OpTypePointer Function %int

%int_0         = OpConstant %int 0
%int_1         = OpConstant %int 1
%int_2         = OpConstant %int 2
%int_3         = OpConstant %int 3
%int_5         = OpConstant %int 5
%float_0       = OpConstant %float 0
%float_0_25    = OpConstant %float 0.25
%float_0_02    = OpConstant %float 0.02
%float_eps     = OpConstant %float 0.000001

%main = OpFunction %void None %func
%entry = OpLabel
%outPosPtr = OpAccessChain %ptr_output_v4 %gl_out %int_0
%ptrP0 = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%P0 = OpLoad %v4float %ptrP0
%ptrP1 = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
%P1 = OpLoad %v4float %ptrP1
%ptrP2 = OpAccessChain %ptr_input_v4 %gl_in %int_2 %int_0
%P2 = OpLoad %v4float %ptrP2
%ptrP3 = OpAccessChain %ptr_input_v4 %gl_in %int_3 %int_0
%P3 = OpLoad %v4float %ptrP3
%P0xy = OpVectorShuffle %v2float %P0 %P0 0 1
%P1xy = OpVectorShuffle %v2float %P1 %P1 0 1
%P2xy = OpVectorShuffle %v2float %P2 %P2 0 1
%P3xy = OpVectorShuffle %v2float %P3 %P3 0 1
%startTan = OpFSub %v2float %P2xy %P0xy
%endTan = OpFSub %v2float %P3xy %P1xy
%tanDiff = OpFSub %v2float %endTan %startTan
%posDiff = OpFSub %v4float %P2 %P1
%ptrColor1 = OpAccessChain %ptr_input_v3 %in_color %int_1
%color1 = OpLoad %v3float %ptrColor1
%ptrColor2 = OpAccessChain %ptr_input_v3 %in_color %int_2
%color2 = OpLoad %v3float %ptrColor2
%colorDiff = OpFSub %v3float %color2 %color1
%iVar = OpVariable %ptr_function_int Function
OpStore %iVar %int_0
OpBranch %loop

%loop = OpLabel
OpLoopMerge %merge %continue None
OpBranch %check

%check = OpLabel
%i_check = OpLoad %int %iVar
%cond = OpSLessThan %bool %i_check %int_5
OpBranchConditional %cond %body %merge

%body = OpLabel
%i_val = OpLoad %int %iVar
%fi = OpConvertSToF %float %i_val
%t = OpFMul %float %fi %float_0_25
%tmpPos = OpVectorTimesScalar %v4float %posDiff %t
%base = OpFAdd %v4float %P1 %tmpPos
%tmpTan = OpVectorTimesScalar %v2float %tanDiff %t
%tan = OpFAdd %v2float %startTan %tmpTan
%lenSq = OpDot %float %tan %tan
%lenClamp = OpFMax %float %lenSq %float_eps
%invLen = OpInverseSqrt %float %lenClamp
%tanNorm = OpVectorTimesScalar %v2float %tan %invLen
%tx = OpCompositeExtract %float %tanNorm 0
%ty = OpCompositeExtract %float %tanNorm 1
%negTy = OpFNegate %float %ty
%perpVec = OpCompositeConstruct %v2float %negTy %tx
%offset2 = OpVectorTimesScalar %v2float %perpVec %float_0_02
%offX = OpCompositeExtract %float %offset2 0
%offY = OpCompositeExtract %float %offset2 1
%offset4 = OpCompositeConstruct %v4float %offX %offY %float_0 %float_0
%posPlus = OpFAdd %v4float %base %offset4
%posMinus = OpFSub %v4float %base %offset4
%tmpColor = OpVectorTimesScalar %v3float %colorDiff %t
%colorCur = OpFAdd %v3float %color1 %tmpColor
OpStore %outPosPtr %posPlus
OpStore %outColor %colorCur
OpEmitVertex
OpStore %outPosPtr %posMinus
OpStore %outColor %colorCur
OpEmitVertex
%i_next = OpIAdd %int %i_val %int_1
OpStore %iVar %i_next
OpBranch %continue

%continue = OpLabel
OpBranch %loop

%merge = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
