use std::io::{self, BufRead, Write};

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();

    let (h, w) = match read_dimensions(&mut reader) {
        Some(dim) => dim,
        None => {
            println!("0");
            return;
        }
    };

    let total = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut passable = vec![0u8; (total + 7) / 8];
    let mut start_idx: Option<usize> = None;
    let mut goal_idx: Option<usize> = None;

    let mut row_buf = Vec::with_capacity(w.max(1));
    for y in 0..h {
        if !read_row(&mut reader, &mut row_buf, w) || row_buf.len() != w {
            println!("0");
            return;
        }
        let base = y * w;
        for (x, &cell) in row_buf.iter().enumerate() {
            let idx = base + x;
            match cell {
                b'#' => {}
                b'A' => {
                    set_bit(&mut passable, idx);
                    if start_idx.is_none() {
                        start_idx = Some(idx);
                    }
                }
                b'B' => {
                    set_bit(&mut passable, idx);
                    if goal_idx.is_none() {
                        goal_idx = Some(idx);
                    }
                }
                _ => {
                    // Any non-wall tile is traversable (spaces, start, goal, etc.)
                    set_bit(&mut passable, idx);
                }
            }
        }
    }

    let start = match start_idx {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let goal = match goal_idx {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut visited = vec![0u8; (total + 7) / 8];
    let mut parent_dir = vec![0u8; (total + 3) / 4];
    let mut queue: Vec<usize> = Vec::with_capacity(1024);

    set_bit(&mut visited, start);
    queue.push(start);

    let mut head = 0usize;
    let mut found = false;

    'bfs: while head < queue.len() {
        let cur = queue[head];
        head += 1;

        if cur == goal {
            found = true;
            break;
        }

        let x = cur % w;
        let y = cur / w;

        if y > 0 && try_enqueue(cur - w, 2, goal, &passable, &mut visited, &mut parent_dir, &mut queue) {
            found = true;
            break 'bfs;
        }
        if y + 1 < h && try_enqueue(cur + w, 0, goal, &passable, &mut visited, &mut parent_dir, &mut queue) {
            found = true;
            break 'bfs;
        }
        if x > 0 && try_enqueue(cur - 1, 1, goal, &passable, &mut visited, &mut parent_dir, &mut queue) {
            found = true;
            break 'bfs;
        }
        if x + 1 < w && try_enqueue(cur + 1, 3, goal, &passable, &mut visited, &mut parent_dir, &mut queue) {
            found = true;
            break 'bfs;
        }
    }

    if !found {
        println!("0");
        return;
    }

    let mut path = Vec::new();
    let mut cur = goal;
    path.push(cur);
    while cur != start {
        let dir = get_parent(&parent_dir, cur);
        cur = match dir {
            0 => cur - w, // move up
            1 => cur + 1, // move right
            2 => cur + w, // move down
            3 => cur - 1, // move left
            _ => unreachable!(),
        };
        path.push(cur);
    }

    let stdout = io::stdout();
    let mut writer = io::BufWriter::new(stdout.lock());
    writeln!(writer, "{}", path.len()).unwrap();
    for &idx in path.iter().rev() {
        writeln!(writer, "{} {}", idx % w, idx / w).unwrap();
    }
}

fn read_dimensions<R: BufRead>(reader: &mut R) -> Option<(usize, usize)> {
    let mut line = String::new();
    loop {
        line.clear();
        let bytes = reader.read_line(&mut line).ok()?;
        if bytes == 0 {
            return None;
        }
        if line.trim().is_empty() {
            continue;
        }
        let mut parts = line.split_whitespace();
        let h_token = parts.next()?.trim_start_matches('\u{feff}');
        let w_token = parts.next()?;
        let h = h_token.parse().ok()?;
        let w = w_token.parse().ok()?;
        return Some((h, w));
    }
}

fn read_row<R: BufRead>(reader: &mut R, buffer: &mut Vec<u8>, width: usize) -> bool {
    loop {
        buffer.clear();
        match reader.read_until(b'\n', buffer) {
            Ok(0) => {
                if buffer.is_empty() {
                    return false;
                }
            }
            Ok(_) => {}
            Err(_) => return false,
        }
        if let Some(&b'\n') = buffer.last() {
            buffer.pop();
            if let Some(&b'\r') = buffer.last() {
                buffer.pop();
            }
        }
        if width > 0 && buffer.is_empty() {
            continue;
        }
        return true;
    }
}

#[inline]
fn set_bit(bits: &mut [u8], idx: usize) {
    let byte = idx >> 3;
    let shift = (idx & 7) as u32;
    bits[byte] |= 1u8 << shift;
}

#[inline]
fn bit_is_set(bits: &[u8], idx: usize) -> bool {
    let byte = idx >> 3;
    let shift = (idx & 7) as u32;
    (bits[byte] >> shift) & 1u8 != 0
}

#[inline]
fn set_parent(parent: &mut [u8], idx: usize, dir: u8) {
    let byte = idx >> 2;
    let shift = ((idx & 3) * 2) as u32;
    let mask = 0b11u8 << shift;
    parent[byte] = (parent[byte] & !mask) | ((dir & 0b11) << shift);
}

#[inline]
fn get_parent(parent: &[u8], idx: usize) -> u8 {
    let byte = idx >> 2;
    let shift = ((idx & 3) * 2) as u32;
    ((parent[byte] >> shift) & 0b11) as u8
}

#[inline]
fn try_enqueue(
    idx: usize,
    dir_code: u8,
    goal: usize,
    passable: &[u8],
    visited: &mut [u8],
    parent: &mut [u8],
    queue: &mut Vec<usize>,
) -> bool {
    if !bit_is_set(passable, idx) || bit_is_set(visited, idx) {
        return false;
    }
    set_bit(visited, idx);
    set_parent(parent, idx, dir_code);
    if idx == goal {
        true
    } else {
        queue.push(idx);
        false
    }
}
