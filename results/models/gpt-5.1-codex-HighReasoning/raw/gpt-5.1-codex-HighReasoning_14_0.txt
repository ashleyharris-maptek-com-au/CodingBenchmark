using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string? first = ReadNonEmptyLine();
        if (first == null)
        {
            Console.WriteLine("0");
            return;
        }

        var parts = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2)
        {
            Console.WriteLine("0");
            return;
        }

        int height = int.Parse(parts[0]);
        int width = int.Parse(parts[1]);
        long totalCells = (long)height * width;

        char[][] board = new char[height][];
        bool[][] flagged = new bool[height][];
        bool[][] safe = new bool[height][];
        bool cacheNumbers = totalCells <= 4_000_000;
        List<(int row, int col)>? numberCache = cacheNumbers ? new List<(int, int)>() : null;

        for (int r = 0; r < height; r++)
        {
            string? line = Console.ReadLine() ?? string.Empty;
            if (line.Length < width)
            {
                line = line.PadRight(width, ' ');
            }
            else if (line.Length > width)
            {
                line = line.Substring(0, width);
            }

            char[] rowChars = line.ToCharArray();
            board[r] = rowChars;
            flagged[r] = new bool[width];
            safe[r] = new bool[width];

            for (int c = 0; c < width; c++)
            {
                char ch = rowChars[c];
                if (ch == '*')
                {
                    flagged[r][c] = true;
                }
                if (ch >= '0' && ch <= '8')
                {
                    numberCache?.Add((r, c));
                }
            }
        }

        var solver = new MinesweeperSolver(board, flagged, safe, numberCache, totalCells);
        solver.Solve();

        var safeCells = solver.GetSafeCells();
        Console.WriteLine(safeCells.Count);
        foreach (var (x, y) in safeCells)
        {
            Console.WriteLine($"{x} {y}");
        }
    }

    private static string? ReadNonEmptyLine()
    {
        string? line;
        while ((line = Console.ReadLine()) != null)
        {
            if (!string.IsNullOrWhiteSpace(line))
            {
                return line;
            }
        }
        return null;
    }
}

class MinesweeperSolver
{
    private readonly char[][] _board;
    private readonly bool[][] _flagged;
    private readonly bool[][] _safe;
    private readonly int _height;
    private readonly int _width;
    private readonly List<(int row, int col)>? _numberCache;
    private readonly long _totalCells;

    private static readonly int[] NeighborRows = { -1, -1, -1, 0, 0, 1, 1, 1 };
    private static readonly int[] NeighborCols = { -1, 0, 1, -1, 1, -1, 0, 1 };

    public MinesweeperSolver(char[][] board,
                             bool[][] flagged,
                             bool[][] safe,
                             List<(int row, int col)>? numberCache,
                             long totalCells)
    {
        _board = board;
        _flagged = flagged;
        _safe = safe;
        _height = board.Length;
        _width = board.Length == 0 ? 0 : board[0].Length;
        _numberCache = numberCache;
        _totalCells = totalCells;
    }

    public void Solve()
    {
        int guard = 0;
        const int maxIterations = 500;

        while (guard++ < maxIterations)
        {
            if (ApplyBasicRules())
            {
                continue;
            }

            if (ApplyAdvancedStrategies())
            {
                continue;
            }

            break;
        }
    }

    public List<(int x, int y)> GetSafeCells()
    {
        var result = new List<(int x, int y)>();
        for (int r = 0; r < _height; r++)
        {
            for (int c = 0; c < _width; c++)
            {
                if (_safe[r][c] && _board[r][c] == ' ')
                {
                    result.Add((c, r));
                }
            }
        }

        result.Sort((a, b) =>
        {
            int cmp = a.y.CompareTo(b.y);
            if (cmp != 0) return cmp;
            return a.x.CompareTo(b.x);
        });

        return result;
    }

    private bool ApplyBasicRules()
    {
        bool changed = false;
        List<(int r, int c)> unknowns = new List<(int, int)>(8);

        foreach (var (row, col) in EnumerateNumbers())
        {
            char ch = _board[row][col];
            if (ch < '0' || ch > '8')
                continue;

            int number = ch - '0';
            int flaggedCount = 0;
            unknowns.Clear();

            for (int k = 0; k < 8; k++)
            {
                int nr = row + NeighborRows[k];
                int nc = col + NeighborCols[k];
                if (nr < 0 || nr >= _height || nc < 0 || nc >= _width)
                    continue;

                if (_flagged[nr][nc])
                {
                    flaggedCount++;
                }
                else if (IsUnknown(nr, nc))
                {
                    unknowns.Add((nr, nc));
                }
            }

            if (unknowns.Count == 0)
                continue;

            int remaining = number - flaggedCount;
            if (remaining == 0)
            {
                foreach (var (r, c) in unknowns)
                {
                    if (MarkSafe(r, c))
                        changed = true;
                }
            }
            else if (remaining == unknowns.Count)
            {
                foreach (var (r, c) in unknowns)
                {
                    if (MarkFlag(r, c))
                        changed = true;
                }
            }
        }

        return changed;
    }

    private bool ApplyAdvancedStrategies()
    {
        if (!BuildConstraints(out var constraints,
                              out var cells,
                              out var cellToConstraints))
        {
            return false;
        }

        if (constraints.Count == 0 || cells.Count == 0)
            return false;

        if (ApplySubsetReasoning(constraints, cells))
            return true;

        if (ApplyEnumeration(constraints, cells, cellToConstraints))
            return true;

        return false;
    }

    private bool BuildConstraints(out List<Constraint> constraints,
                                  out List<(int row, int col)> cellPositions,
                                  out List<int>[] cellToConstraints)
    {
        constraints = new List<Constraint>();
        cellPositions = new List<(int, int)>();
        var cellConstraintLists = new List<List<int>>();
        var cellMap = new Dictionary<long, int>();

        int GetOrAddCellId(int r, int c)
        {
            long key = (((long)r) << 32) ^ (uint)c;
            if (!cellMap.TryGetValue(key, out int id))
            {
                id = cellPositions.Count;
                cellMap[key] = id;
                cellPositions.Add((r, c));
                cellConstraintLists.Add(new List<int>());
            }
            return id;
        }

        List<int> temp = new List<int>(8);

        foreach (var (row, col) in EnumerateNumbers())
        {
            char ch = _board[row][col];
            if (ch < '0' || ch > '8')
                continue;

            int number = ch - '0';
            int flaggedCount = 0;
            temp.Clear();

            for (int k = 0; k < 8; k++)
            {
                int nr = row + NeighborRows[k];
                int nc = col + NeighborCols[k];
                if (nr < 0 || nr >= _height || nc < 0 || nc >= _width)
                    continue;

                if (_flagged[nr][nc])
                {
                    flaggedCount++;
                }
                else if (IsUnknown(nr, nc))
                {
                    int id = GetOrAddCellId(nr, nc);
                    temp.Add(id);
                }
            }

            if (temp.Count == 0)
                continue;

            int remaining = number - flaggedCount;
            if (remaining < 0 || remaining > temp.Count)
                continue;

            temp.Sort();
            var cellsCopy = new List<int>(temp);
            var constraint = new Constraint(cellsCopy, remaining);

            if (constraint.Required < 0 || constraint.Required > constraint.Cells.Count)
                continue;

            int index = constraints.Count;
            constraints.Add(constraint);

            foreach (int id in constraint.Cells)
            {
                cellConstraintLists[id].Add(index);
            }
        }

        if (constraints.Count == 0 || cellPositions.Count == 0)
        {
            cellToConstraints = Array.Empty<List<int>>();
            return false;
        }

        cellToConstraints = new List<int>[cellConstraintLists.Count];
        for (int i = 0; i < cellConstraintLists.Count; i++)
        {
            cellToConstraints[i] = cellConstraintLists[i];
        }

        return true;
    }

    private bool ApplySubsetReasoning(List<Constraint> constraints,
                                      List<(int row, int col)> cellPositions)
    {
        bool changed = false;

        for (int i = 0; i < constraints.Count; i++)
        {
            var a = constraints[i];
            if (a.Cells.Count == 0)
                continue;

            for (int j = i + 1; j < constraints.Count; j++)
            {
                var b = constraints[j];
                if (b.Cells.Count == 0)
                    continue;

                if (ProcessSubset(a, b, cellPositions))
                    changed = true;
                if (ProcessSubset(b, a, cellPositions))
                    changed = true;

                if (changed)
                    return true;
            }
        }

        return changed;
    }

    private bool ProcessSubset(Constraint smaller,
                               Constraint larger,
                               List<(int row, int col)> cellPositions)
    {
        if (smaller.Cells.Count == 0 || smaller.Cells.Count > larger.Cells.Count)
            return false;
        if (smaller.Required > larger.Required)
            return false;
        if (!IsSubset(smaller.Cells, larger.Cells))
            return false;

        var diff = Difference(larger.Cells, smaller.Cells);
        if (diff.Count == 0)
            return false;

        int diffReq = larger.Required - smaller.Required;
        if (diffReq < 0 || diffReq > diff.Count)
            return false;

        if (diffReq == 0)
        {
            bool changed = false;
            foreach (int id in diff)
            {
                var (r, c) = cellPositions[id];
                if (MarkSafe(r, c))
                    changed = true;
            }
            return changed;
        }

        if (diffReq == diff.Count)
        {
            bool changed = false;
            foreach (int id in diff)
            {
                var (r, c) = cellPositions[id];
                if (MarkFlag(r, c))
                    changed = true;
            }
            return changed;
        }

        return false;
    }

    private bool ApplyEnumeration(List<Constraint> constraints,
                                  List<(int row, int col)> cellPositions,
                                  List<int>[] cellToConstraints)
    {
        int frontier = cellPositions.Count;
        int enumerationLimit = DetermineEnumerationLimit(frontier);
        if (enumerationLimit <= 0)
            return false;

        bool changed = false;
        int cellCount = cellPositions.Count;
        var visitedCells = new bool[cellCount];
        var constraintTagged = new bool[constraints.Count];

        for (int start = 0; start < cellCount; start++)
        {
            if (visitedCells[start])
                continue;

            var queue = new Queue<int>();
            var componentCells = new List<int>();
            var componentConstraints = new List<int>();
            var taggedThisComponent = new List<int>();

            queue.Enqueue(start);
            visitedCells[start] = true;

            while (queue.Count > 0)
            {
                int cellId = queue.Dequeue();
                componentCells.Add(cellId);

                foreach (int constraintIndex in cellToConstraints[cellId])
                {
                    if (constraintTagged[constraintIndex])
                        continue;

                    constraintTagged[constraintIndex] = true;
                    taggedThisComponent.Add(constraintIndex);
                    componentConstraints.Add(constraintIndex);

                    foreach (int neighborCell in constraints[constraintIndex].Cells)
                    {
                        if (!visitedCells[neighborCell])
                        {
                            visitedCells[neighborCell] = true;
                            queue.Enqueue(neighborCell);
                        }
                    }
                }
            }

            foreach (int idx in taggedThisComponent)
            {
                constraintTagged[idx] = false;
            }

            if (componentCells.Count == 0 || componentConstraints.Count == 0)
                continue;
            if (componentCells.Count > enumerationLimit)
                continue;

            if (AnalyzeComponent(componentCells, componentConstraints, constraints, cellPositions))
            {
                changed = true;
                break;
            }
        }

        return changed;
    }

    private bool AnalyzeComponent(List<int> componentCells,
                                  List<int> componentConstraints,
                                  List<Constraint> constraints,
                                  List<(int row, int col)> cellPositions)
    {
        int varCount = componentCells.Count;
        var indexMap = new Dictionary<int, int>(varCount);
        for (int i = 0; i < varCount; i++)
        {
            indexMap[componentCells[i]] = i;
        }

        var localConstraints = new List<LocalConstraint>();
        foreach (int constraintIndex in componentConstraints)
        {
            var constraint = constraints[constraintIndex];
            var vars = new int[constraint.Cells.Count];
            bool valid = true;
            for (int i = 0; i < constraint.Cells.Count; i++)
            {
                int cellId = constraint.Cells[i];
                if (!indexMap.TryGetValue(cellId, out int localId))
                {
                    valid = false;
                    break;
                }
                vars[i] = localId;
            }

            if (!valid || vars.Length == 0)
                continue;

            localConstraints.Add(new LocalConstraint(vars, constraint.Required));
        }

        if (localConstraints.Count == 0)
            return false;

        var solver = new ComponentSolver(varCount, localConstraints);
        ComponentResult? result = solver.Solve();
        if (result == null)
            return false;

        bool changed = false;
        var res = result.Value;

        for (int i = 0; i < varCount; i++)
        {
            var (row, col) = cellPositions[componentCells[i]];
            if (res.Safe[i])
            {
                if (MarkSafe(row, col))
                    changed = true;
            }
            else if (res.Mines[i])
            {
                if (MarkFlag(row, col))
                    changed = true;
            }
        }

        return changed;
    }

    private int DetermineEnumerationLimit(int frontier)
    {
        if (_totalCells > 80_000_000)
        {
            if (frontier > 2000)
                return 0;
            return 10;
        }

        if (frontier <= 150) return 20;
        if (frontier <= 600) return 16;
        if (frontier <= 2000) return 14;
        if (frontier <= 6000) return 12;
        return 0;
    }

    private bool MarkSafe(int r, int c)
    {
        if (r < 0 || r >= _height || c < 0 || c >= _width)
            return false;
        if (_safe[r][c])
            return false;
        if (_flagged[r][c])
            return false;
        if (_board[r][c] != ' ')
            return false;

        _safe[r][c] = true;
        return true;
    }

    private bool MarkFlag(int r, int c)
    {
        if (r < 0 || r >= _height || c < 0 || c >= _width)
            return false;
        if (_flagged[r][c])
            return false;
        if (_board[r][c] != ' ' && _board[r][c] != '*')
            return false;

        _flagged[r][c] = true;
        _board[r][c] = '*';
        return true;
    }

    private bool IsUnknown(int r, int c)
    {
        return !_flagged[r][c] && !_safe[r][c] && _board[r][c] == ' ';
    }

    private IEnumerable<(int row, int col)> EnumerateNumbers()
    {
        if (_numberCache != null)
        {
            foreach (var entry in _numberCache)
                yield return entry;
        }
        else
        {
            for (int r = 0; r < _height; r++)
            {
                for (int c = 0; c < _width; c++)
                {
                    if (_board[r][c] >= '0' && _board[r][c] <= '8')
                        yield return (r, c);
                }
            }
        }
    }

    private static bool IsSubset(List<int> smaller, List<int> larger)
    {
        int i = 0, j = 0;
        while (i < smaller.Count && j < larger.Count)
        {
            if (smaller[i] == larger[j])
            {
                i++;
                j++;
            }
            else if (smaller[i] > larger[j])
            {
                j++;
            }
            else
            {
                return false;
            }
        }
        return i == smaller.Count;
    }

    private static List<int> Difference(List<int> sup, List<int> sub)
    {
        var diff = new List<int>();
        int i = 0, j = 0;
        while (j < sup.Count)
        {
            if (i < sub.Count && sub[i] == sup[j])
            {
                i++;
                j++;
            }
            else if (i < sub.Count && sub[i] < sup[j])
            {
                i++;
            }
            else
            {
                diff.Add(sup[j]);
                j++;
            }
        }
        return diff;
    }

    private readonly struct Constraint
    {
        public Constraint(List<int> cells, int required)
        {
            Cells = cells;
            Required = required;
        }

        public List<int> Cells { get; }
        public int Required { get; }
    }

    private readonly struct LocalConstraint
    {
        public LocalConstraint(int[] variables, int required)
        {
            Variables = variables;
            Required = required;
        }

        public int[] Variables { get; }
        public int Required { get; }
    }

    private readonly struct ComponentResult
    {
        public ComponentResult(bool[] safe, bool[] mines)
        {
            Safe = safe;
            Mines = mines;
        }

        public bool[] Safe { get; }
        public bool[] Mines { get; }
    }

    private sealed class ComponentSolver
    {
        private readonly List<LocalConstraint> _constraints;
        private readonly int _varCount;
        private readonly List<int>[] _varToConstraints;
        private readonly int[] _remainingMines;
        private readonly int[] _remainingUnknowns;
        private readonly int[] _assignment;
        private readonly bool[] _seenZero;
        private readonly bool[] _seenOne;
        private readonly int[] _order;

        private int _solutions;
        private bool _stopSearch;
        private int _varsNeedingProof;

        public ComponentSolver(int varCount, List<LocalConstraint> constraints)
        {
            _varCount = varCount;
            _constraints = constraints;

            _varToConstraints = new List<int>[varCount];
            for (int i = 0; i < varCount; i++)
            {
                _varToConstraints[i] = new List<int>();
            }

            _remainingMines = new int[constraints.Count];
            _remainingUnknowns = new int[constraints.Count];

            for (int i = 0; i < constraints.Count; i++)
            {
                _remainingMines[i] = constraints[i].Required;
                _remainingUnknowns[i] = constraints[i].Variables.Length;
                foreach (int var in constraints[i].Variables)
                {
                    _varToConstraints[var].Add(i);
                }
            }

            _assignment = Enumerable.Repeat(-1, varCount).ToArray();
            _seenZero = new bool[varCount];
            _seenOne = new bool[varCount];
            _varsNeedingProof = varCount;

            _order = Enumerable.Range(0, varCount)
                                .OrderByDescending(i => _varToConstraints[i].Count)
                                .ToArray();
        }

        public ComponentResult? Solve()
        {
            if (_constraints.Count == 0)
                return null;

            for (int i = 0; i < _constraints.Count; i++)
            {
                if (_remainingMines[i] < 0 || _remainingMines[i] > _remainingUnknowns[i])
                    return null;
            }

            Search(0);

            if (_solutions == 0)
                return null;

            var safe = new bool[_varCount];
            var mines = new bool[_varCount];

            for (int i = 0; i < _varCount; i++)
            {
                if (_seenZero[i] && !_seenOne[i])
                {
                    safe[i] = true;
                }
                else if (_seenOne[i] && !_seenZero[i])
                {
                    mines[i] = true;
                }
            }

            bool any = safe.Any(s => s) || mines.Any(m => m);
            if (!any)
                return new ComponentResult(safe, mines);

            return new ComponentResult(safe, mines);
        }

        private void Search(int depth)
        {
            if (_stopSearch)
                return;

            if (depth == _order.Length)
            {
                RecordSolution();
                return;
            }

            int variable = _order[depth];
            if (_assignment[variable] != -1)
            {
                Search(depth + 1);
                return;
            }

            var touched = new List<int>();

            if (TryAssign(variable, 0, touched))
            {
                _assignment[variable] = 0;
                Search(depth + 1);
                _assignment[variable] = -1;
                UndoAssign(0, touched);
                if (_stopSearch)
                    return;
            }

            if (TryAssign(variable, 1, touched))
            {
                _assignment[variable] = 1;
                Search(depth + 1);
                _assignment[variable] = -1;
                UndoAssign(1, touched);
            }
        }

        private bool TryAssign(int variable, int value, List<int> touched)
        {
            touched.Clear();

            foreach (int constraintIndex in _varToConstraints[variable])
            {
                touched.Add(constraintIndex);
                _remainingUnknowns[constraintIndex]--;
                if (value == 1)
                    _remainingMines[constraintIndex]--;

                if (_remainingMines[constraintIndex] < 0 ||
                    _remainingMines[constraintIndex] > _remainingUnknowns[constraintIndex])
                {
                    for (int i = touched.Count - 1; i >= 0; i--)
                    {
                        int idx = touched[i];
                        if (value == 1)
                            _remainingMines[idx]++;
                        _remainingUnknowns[idx]++;
                    }
                    touched.Clear();
                    return false;
                }
            }

            return true;
        }

        private void UndoAssign(int value, List<int> touched)
        {
            for (int i = touched.Count - 1; i >= 0; i--)
            {
                int idx = touched[i];
                if (value == 1)
                    _remainingMines[idx]++;
                _remainingUnknowns[idx]++;
            }
            touched.Clear();
        }

        private void RecordSolution()
        {
            _solutions++;

            for (int i = 0; i < _varCount; i++)
            {
                int val = _assignment[i];
                if (val == 0)
                {
                    if (!_seenZero[i])
                    {
                        _seenZero[i] = true;
                        if (_seenOne[i])
                            _varsNeedingProof--;
                    }
                }
                else if (val == 1)
                {
                    if (!_seenOne[i])
                    {
                        _seenOne[i] = true;
                        if (_seenZero[i])
                            _varsNeedingProof--;
                    }
                }
            }

            if (_varsNeedingProof == 0)
            {
                _stopSearch = true;
            }
        }
    }
}
