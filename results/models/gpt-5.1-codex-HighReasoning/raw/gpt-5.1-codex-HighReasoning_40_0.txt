```cpp
#include <bits/stdc++.h>
using namespace std;

using ll = long long;

vector<vector<uint64_t>> adjMatrix;   // compatibility graph (complement of conflict graph)
vector<ll> weights;
vector<int> colorOfVertex;
vector<int> currentClique, bestClique;
ll bestWeight = 0;

inline bool isAdjacent(int u, int v) {
    return (adjMatrix[u][v >> 6] >> (v & 63)) & 1ULL;
}

void colorOrder(const vector<int>& P, vector<int>& order, vector<ll>& bound) {
    int sz = (int)P.size();
    order.clear();
    order.reserve(sz);
    bound.assign(sz, 0);
    if (sz == 0) return;

    for (int v : P) colorOfVertex[v] = 0;

    vector<int> remaining = P;
    int color = 0;
    vector<ll> maxWeightPerColor(1, 0);

    while (!remaining.empty()) {
        ++color;
        vector<int> Q;
        Q.swap(remaining);
        ll colorMax = 0;

        while (!Q.empty()) {
            int v = Q.back();
            Q.pop_back();
            colorOfVertex[v] = color;
            order.push_back(v);
            colorMax = max(colorMax, weights[v]);

            vector<int> next;
            next.reserve(Q.size());
            for (int u : Q) {
                if (!isAdjacent(v, u))
                    next.push_back(u);   // can share this color
                else
                    remaining.push_back(u); // postpone to later colors
            }
            Q.swap(next);
        }
        maxWeightPerColor.push_back(colorMax);
    }

    vector<ll> prefix(color + 1, 0);
    for (int c = 1; c <= color; ++c)
        prefix[c] = prefix[c - 1] + maxWeightPerColor[c];

    for (size_t i = 0; i < order.size(); ++i)
        bound[i] = prefix[colorOfVertex[order[i]]];
}

void expand(const vector<int>& P, ll curWeight) {
    if (P.empty()) return;

    vector<int> order;
    vector<ll> bound;
    order.reserve(P.size());
    bound.reserve(P.size());

    colorOrder(P, order, bound);
    int sz = (int)order.size();

    for (int idx = sz - 1; idx >= 0; --idx) {
        if (curWeight + bound[idx] <= bestWeight) return;

        int v = order[idx];
        currentClique.push_back(v);
        ll newWeight = curWeight + weights[v];
        if (newWeight > bestWeight) {
            bestWeight = newWeight;
            bestClique = currentClique;
        }

        vector<int> nextSet;
        nextSet.reserve(idx);
        for (int j = 0; j < idx; ++j) {
            int u = order[j];
            if (isAdjacent(v, u))
                nextSet.push_back(u);
        }
        if (!nextSet.empty())
            expand(nextSet, newWeight);

        currentClique.pop_back();
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<ll> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<pair<int,int>> edges;
    edges.reserve(m);
    vector<char> forbidden(n, false);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            forbidden[u] = true;
            continue;
        }
        edges.emplace_back(u, v);
    }

    vector<int> mapOldToNew(n, -1);
    vector<int> newToOld;
    vector<ll> filteredWeights;
    filteredWeights.reserve(n);

    for (int i = 0; i < n; ++i) {
        if (!forbidden[i] && c[i] > 0) {
            mapOldToNew[i] = (int)newToOld.size();
            newToOld.push_back(i);
            filteredWeights.push_back(c[i]);
        }
    }

    int N = (int)newToOld.size();
    vector<int> assignment(n, 0);

    if (N == 0) {
        cout << 0 << "\n";
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << 0;
        }
        cout << "\n";
        return 0;
    }

    weights = filteredWeights;
    int words = (N + 63) / 64;
    vector<uint64_t> fullMask(words, ~0ULL);
    if (N % 64 != 0)
        fullMask.back() = (1ULL << (N % 64)) - 1;

    adjMatrix.assign(N, vector<uint64_t>(words, 0));
    for (int i = 0; i < N; ++i) {
        adjMatrix[i] = fullMask;
        adjMatrix[i][i >> 6] &= ~(1ULL << (i & 63));
    }

    for (auto [u0, v0] : edges) {
        int u = mapOldToNew[u0];
        int v = mapOldToNew[v0];
        if (u == -1 || v == -1 || u == v) continue;
        adjMatrix[u][v >> 6] &= ~(1ULL << (v & 63));
        adjMatrix[v][u >> 6] &= ~(1ULL << (u & 63));
    }

    colorOfVertex.assign(N, 0);
    currentClique.clear();
    currentClique.reserve(N);
    bestClique.clear();
    bestClique.reserve(N);
    bestWeight = 0;

    vector<int> initial(N);
    iota(initial.begin(), initial.end(), 0);
    sort(initial.begin(), initial.end(), [&](int a, int b) {
        if (weights[a] != weights[b]) return weights[a] > weights[b];
        return a < b;
    });

    expand(initial, 0);

    if (bestClique.empty()) {
        int idx = int(max_element(weights.begin(), weights.end()) - weights.begin());
        bestClique = {idx};
        bestWeight = weights[idx];
    }

    for (int v : bestClique)
        assignment[newToOld[v]] = 1;

    cout << bestWeight << "\n";
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << assignment[i];
    }
    cout << "\n";
    return 0;
}
```
