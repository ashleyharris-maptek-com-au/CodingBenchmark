class TokenReader
{
    private readonly Queue<string> _buffer = new Queue<string>();
    private readonly TextReader _reader;
    private static readonly char[] Separators = { ' ', '\t', '\r', '\n' };

    public TokenReader(IEnumerable<string> initialTokens)
    {
        _reader = Console.In;
        if (initialTokens != null)
        {
            foreach (var token in initialTokens)
            {
                if (!string.IsNullOrWhiteSpace(token))
                {
                    _buffer.Enqueue(token);
                }
            }
        }
    }

    public string Next()
    {
        while (_buffer.Count == 0)
        {
            string line = _reader.ReadLine();
            if (line == null)
            {
                return null;
            }

            var tokens = line.Split(Separators, StringSplitOptions.RemoveEmptyEntries);
            foreach (var token in tokens)
            {
                _buffer.Enqueue(token);
            }
        }

        return _buffer.Dequeue();
    }

    public double NextDouble()
    {
        string token = Next();
        if (token == null)
        {
            throw new InvalidOperationException("Unexpected end of input");
        }

        return double.Parse(token, CultureInfo.InvariantCulture);
    }
}

static class MathUtil
{
    public static double Norm(double[] v)
    {
        return Math.Sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2]);
    }

    public static double Distance(double[] a, double[] b)
    {
        double dx = a[0] - b[0];
        double dy = a[1] - b[1];
        double dz = a[2] - b[2];
        return Math.Sqrt(dx * dx + dy * dy + dz * dz);
    }

    public static double[] Cross(double[] a, double[] b)
    {
        return new double[]
        {
            a[1] * b[2] - a[2] * b[1],
            a[2] * b[0] - a[0] * b[2],
            a[0] * b[1] - a[1] * b[0]
        };
    }

    public static double Dot(double[] a, double[] b)
    {
        return a[0] * b[0] + a[1] * b[1] + a[2] * b[2];
    }
}

class OrbitalState
{
    public double[] Position { get; }
    public double[] Velocity { get; }
    public double Radius { get; }
    public double Speed { get; }
    public double[] AngularMomentumUnit { get; }
    public double[] EccentricityVector { get; }
    public double Eccentricity { get; }
    public double SemiMajorAxis { get; }

    public OrbitalState(double[] position, double[] velocity, double mu)
    {
        Position = (double[])position.Clone();
        Velocity = (double[])velocity.Clone();
        Radius = MathUtil.Norm(Position);
        Speed = MathUtil.Norm(Velocity);

        var h = MathUtil.Cross(Position, Velocity);
        double hNorm = MathUtil.Norm(h);
        AngularMomentumUnit = hNorm < 1e-9
            ? new double[] { 0.0, 0.0, 1.0 }
            : new double[] { h[0] / hNorm, h[1] / hNorm, h[2] / hNorm };

        var rHat = Radius > 1e-9
            ? new double[] { Position[0] / Radius, Position[1] / Radius, Position[2] / Radius }
            : new double[] { 0.0, 0.0, 0.0 };

        var vCrossH = MathUtil.Cross(Velocity, h);
        var ecc = new double[3];
        if (mu > 0 && Radius > 1e-9)
        {
            ecc[0] = vCrossH[0] / mu - rHat[0];
            ecc[1] = vCrossH[1] / mu - rHat[1];
            ecc[2] = vCrossH[2] / mu - rHat[2];
        }
        else
        {
            ecc[0] = ecc[1] = ecc[2] = 0.0;
        }

        EccentricityVector = ecc;
        Eccentricity = MathUtil.Norm(ecc);

        if (Radius > 0)
        {
            double specificEnergy = 0.5 * Speed * Speed - mu / Radius;
            if (Math.Abs(specificEnergy) < 1e-9)
            {
                SemiMajorAxis = double.PositiveInfinity;
            }
            else
            {
                SemiMajorAxis = -mu / (2.0 * specificEnergy);
            }
        }
        else
        {
            SemiMajorAxis = double.PositiveInfinity;
        }
    }
}

    private void TwoOpt(int[] route)
    {
        int n = route.Length;
        if (n < 3)
        {
            return;
        }

        bool improved = true;
        int maxPasses = Math.Max(4, n / 2);
        int pass = 0;
        while (improved && pass < maxPasses)

            improved = false;

                    if (delta < -1e-9)

    private double TwoOptDelta(int[] route, int i, int j)
    {
        int prev = i == 0 ? -1 : route[i - 1];
        int next = j == route.Length - 1 ? -1 : route[j + 1];
        int first = route[i];
        int last = route[j];

        double before = prev == -1 ? _startCost[first] : _cost[prev, first];
        if (next != -1)
        {
            before += _cost[last, next];
        }

        double after = prev == -1 ? _startCost[last] : _cost[prev, last];
        if (next != -1)
        {
            after += _cost[first, next];
        }

        return after - before;
    }

    private void ReverseSegment(int[] route, int start, int end)
    {
        while (start < end)
        {
            int tmp = route[start];
            route[start] = route[end];
            route[end] = tmp;
            start++;
            end--;
        }
    }

    private double EvaluateRoute(int[] route)

        double total = _startCost[route[0]];
        for (int i = 1; i < route.Length; i++)

            total += _cost[route[i - 1], route[i]];

        return total;

    private double EstimateDeltaV(OrbitalState a, OrbitalState b)
    {
        double hohmann = HohmannDeltaV(a.Radius, b.Radius);
        double avgSpeed = 0.5 * (a.Speed + b.Speed);

        double planeAngle = AngleBetween(a.AngularMomentumUnit, b.AngularMomentumUnit);
        double planeCost = 2.0 * avgSpeed * Math.Sin(planeAngle * 0.5);

        double speedAlign = Math.Abs(a.Speed - b.Speed) * 0.4;
        double eccMagnitudeCost = Math.Abs(a.Eccentricity - b.Eccentricity) * avgSpeed * 0.4;

        double eccOrientationCost = 0.0;
        if (a.Eccentricity > 1e-3 && b.Eccentricity > 1e-3)
        {
            double eccAngle = AngleBetween(a.EccentricityVector, b.EccentricityVector);
            eccOrientationCost = avgSpeed * Math.Sin(eccAngle * 0.5) * 0.5;
        }

        double posDiff = MathUtil.Distance(a.Position, b.Position);
        double velDiff = MathUtil.Distance(a.Velocity, b.Velocity);

        double translational = posDiff * 0.0007;
        double velVectorCost = velDiff * 0.3;

        double total = hohmann + planeCost + speedAlign + eccMagnitudeCost + eccOrientationCost + translational + velVectorCost;

        if (double.IsNaN(total) || total < 0 || double.IsInfinity(total))
        {
            total = Math.Abs(hohmann) + Math.Abs(planeCost) + Math.Abs(speedAlign) + Math.Abs(translational) + Math.Abs(velVectorCost);
        }

        return total;
    }

    private double HohmannDeltaV(double r1, double r2)
    {
        const double minRadius = 100.0;
        double rr1 = Math.Max(minRadius, r1);
        double rr2 = Math.Max(minRadius, r2);
        if (Math.Abs(rr1 - rr2) < 1e-6)
        {
            return 0.0;
        }

        double sqrtMuR1 = Math.Sqrt(_mu / rr1);
        double sqrtMuR2 = Math.Sqrt(_mu / rr2);
        double term1 = Math.Sqrt(2.0 * rr2 / (rr1 + rr2));
        double term2 = Math.Sqrt(2.0 * rr1 / (rr1 + rr2));
        double dv1 = Math.Abs(sqrtMuR1 * (term1 - 1.0));
        double dv2 = Math.Abs(sqrtMuR2 * (1.0 - term2));
        return dv1 + dv2;
    }

    private static double AngleBetween(double[] a, double[] b)
    {
        double normA = MathUtil.Norm(a);
        double normB = MathUtil.Norm(b);
        if (normA < 1e-9 || normB < 1e-9)
        {
            return 0.0;
        }

        double dot = MathUtil.Dot(a, b) / (normA * normB);
        if (dot > 1.0)
        {
            dot = 1.0;
        }
        else if (dot < -1.0)
        {
            dot = -1.0;
        }
        return Math.Acos(dot);
    }
