declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : { depot: [0, 0], truckCount: 0, packages: [] };

interface Package {
    id: number;
    x: number;
    y: number;
}

interface Point {
    x: number;
    y: number;
}

const depot: Point = { x: data.depot?.[0] ?? 0, y: data.depot?.[1] ?? 0 };
const truckCount: number = Math.max(0, data.truckCount || 0);
const packages: Package[] = Array.isArray(data.packages)
    ? data.packages.map((p: any) => ({ id: p.id, x: p.x, y: p.y }))
    : [];

function distanceXY(x1: number, y1: number, x2: number, y2: number): number {
    const dx = x1 - x2;
    const dy = y1 - y2;
    return Math.hypot(dx, dy);
}

function distancePoint(a: Point, b: Point): number {
    return distanceXY(a.x, a.y, b.x, b.y);
}

function kMeansCluster(items: Package[], k: number): Package[][] {
    if (k === 0 || items.length === 0) {
        return [];
    }

    // Initialization using a deterministic k-means++ like strategy
    const centers: Point[] = [];
    const chosen = new Set<number>();
    let firstIndex = 0;
    let minId = Infinity;
    for (let i = 0; i < items.length; i++) {
        if (items[i].id < minId) {
            minId = items[i].id;
            firstIndex = i;
        }
    }
    centers.push({ x: items[firstIndex].x, y: items[firstIndex].y });
    chosen.add(firstIndex);

    while (centers.length < k) {
        let bestIdx = -1;
        let bestDist = -1;
        for (let i = 0; i < items.length; i++) {
            if (chosen.has(i)) continue;
            let minDist = Infinity;
            for (const center of centers) {
                const d = distanceXY(items[i].x, items[i].y, center.x, center.y);
                if (d < minDist) minDist = d;
            }
            if (minDist > bestDist) {
                bestDist = minDist;
                bestIdx = i;
            }
        }
        if (bestIdx === -1) break;
        centers.push({ x: items[bestIdx].x, y: items[bestIdx].y });
        chosen.add(bestIdx);
    }

    while (centers.length < k) {
        centers.push({ x: items[0].x, y: items[0].y });
    }

    const iterations = Math.min(20, items.length ? 20 : 0);
    let assignments: Package[][] = Array.from({ length: k }, () => []);

    for (let iter = 0; iter < iterations; iter++) {
        assignments = Array.from({ length: k }, () => []);
        for (const pkg of items) {
            let bestIdx = 0;
            let bestDist = Infinity;
            for (let c = 0; c < k; c++) {
                const d = distanceXY(pkg.x, pkg.y, centers[c].x, centers[c].y);
                if (d < bestDist) {
                    bestDist = d;
                    bestIdx = c;
                }
            }
            assignments[bestIdx].push(pkg);
        }

        // Fix empty clusters by moving farthest points from larger clusters
        let emptyIndex = assignments.findIndex(group => group.length === 0);
        while (emptyIndex !== -1) {
            let donor = -1;
            let donorSize = 0;
            for (let c = 0; c < k; c++) {
                if (assignments[c].length > donorSize) {
                    donorSize = assignments[c].length;
                    donor = c;
                }
            }
            if (donor === -1 || donorSize <= 1) break;
            const center = centers[donor];
            let farIdx = 0;
            let farDist = -1;
            for (let i = 0; i < assignments[donor].length; i++) {
                const pkg = assignments[donor][i];
                const d = distanceXY(pkg.x, pkg.y, center.x, center.y);
                if (d > farDist) {
                    farDist = d;
                    farIdx = i;
                }
            }
            const moved = assignments[donor].splice(farIdx, 1)[0];
            assignments[emptyIndex].push(moved);
            emptyIndex = assignments.findIndex(group => group.length === 0);
        }

        let movedCenter = false;
        for (let c = 0; c < k; c++) {
            if (assignments[c].length === 0) continue;
            let sumX = 0;
            let sumY = 0;
            for (const pkg of assignments[c]) {
                sumX += pkg.x;
                sumY += pkg.y;
            }
            const newX = sumX / assignments[c].length;
            const newY = sumY / assignments[c].length;
            if (newX !== centers[c].x || newY !== centers[c].y) {
                centers[c] = { x: newX, y: newY };
                movedCenter = true;
            }
        }
        if (!movedCenter) break;
    }

    return assignments;
}

function nearestNeighborOrder(cluster: Package[], origin: Point): Package[] {
    const remaining = cluster.slice();
    const order: Package[] = [];
    let current = { x: origin.x, y: origin.y };
    while (remaining.length > 0) {
        let bestIdx = 0;
        let bestDist = Infinity;
        let bestId = Infinity;
        for (let i = 0; i < remaining.length; i++) {
            const pkg = remaining[i];
            const d = distancePoint(current, { x: pkg.x, y: pkg.y });
            if (d + 1e-9 < bestDist || (Math.abs(d - bestDist) <= 1e-9 && pkg.id < bestId)) {
                bestDist = d;
                bestIdx = i;
                bestId = pkg.id;
            }
        }
        const next = remaining.splice(bestIdx, 1)[0];
        order.push(next);
        current = { x: next.x, y: next.y };
    }
    return order;
}

function twoOpt(order: Package[], origin: Point): void {
    const n = order.length;
    if (n < 3) return;
    const maxIterations = Math.min(2000, n * n * 2);
    let iteration = 0;
    while (iteration < maxIterations) {
        let improved = false;
        for (let i = 0; i < n - 1; i++) {
            const prev: Point = i === 0 ? origin : order[i - 1];
            for (let j = i + 1; j < n; j++) {
                const next: Point = j === n - 1 ? origin : order[j + 1];
                const a = order[i];
                const b = order[j];
                const currentDist = distancePoint(prev, a) + distancePoint(b, next);
                const newDist = distancePoint(prev, b) + distancePoint(a, next);
                if (newDist + 1e-9 < currentDist) {
                    let left = i;
                    let right = j;
                    while (left < right) {
                        const temp = order[left];
                        order[left] = order[right];
                        order[right] = temp;
                        left++;
                        right--;
                    }
                    improved = true;
                    break;
                }
            }
            if (improved) break;
        }
        if (!improved) break;
        iteration++;
    }
}

function planRoute(cluster: Package[], origin: Point): number[] {
    if (cluster.length === 0) return [];
    const order = nearestNeighborOrder(cluster, origin);
    twoOpt(order, origin);
    return order.map(pkg => pkg.id);
}

const resultRoutes: number[][] = [];

if (truckCount === 0) {
    // No trucks, return empty routes respecting requirement (no packages deliverable)
    console.log(JSON.stringify({ routes: [] }));
} else if (packages.length === 0) {
    for (let i = 0; i < truckCount; i++) resultRoutes.push([]);
    console.log(JSON.stringify({ routes: resultRoutes }));
} else {
    const activeRoutes = Math.min(truckCount, packages.length);
    const clusters = kMeansCluster(packages, activeRoutes);
    for (let i = 0; i < activeRoutes; i++) {
        const route = planRoute(clusters[i], depot);
        resultRoutes.push(route);
    }
    for (let i = activeRoutes; i < truckCount; i++) {
        resultRoutes.push([]);
    }
    console.log(JSON.stringify({ routes: resultRoutes }));
}
