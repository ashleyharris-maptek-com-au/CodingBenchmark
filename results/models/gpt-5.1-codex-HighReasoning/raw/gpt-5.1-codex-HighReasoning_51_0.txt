```python
import math
from statistics import median

# Persistent controller state (kept across autopilot_step calls)
_ap_state = None


def _init_state():
    return {
        "pitch_int": 0.0,
        "spd_int": 0.0,
        "trim_bias": 0.0,
        "throttle_cmd": None,
        "last_altitude_ft": None,
        "vs_filt": 0.0,
        "last_ias": None,
        "last_pitch": 0.0,
        "last_roll": 0.0,
        "last_heading": 0.0,
        "last_target_ias": 250.0,
    }


def _valid_number(value):
    return isinstance(value, (int, float)) and math.isfinite(value)


def _collect(values, vmin=None, vmax=None):
    out = []
    for v in values:
        if _valid_number(v):
            if (vmin is None or v >= vmin) and (vmax is None or v <= vmax):
                out.append(float(v))
    return out


def _median(values):
    usable = _collect(values)
    if not usable:
        return None
    usable.sort()
    n = len(usable)
    mid = n // 2
    if n % 2:
        return usable[mid]
    return 0.5 * (usable[mid - 1] + usable[mid])


def _fuse_angle(values):
    s = 0.0
    c = 0.0
    count = 0
    for v in values:
        if _valid_number(v):
            rad = math.radians(v % 360.0)
            s += math.sin(rad)
            c += math.cos(rad)
            count += 1
    if count == 0:
        return None
    ang = math.degrees(math.atan2(s / count, c / count))
    if ang < 0:
        ang += 360.0
    return ang


def _wrap180(angle):
    if angle is None:
        return 0.0
    a = (angle + 180.0) % 360.0 - 180.0
    return a


def _clamp(value, min_value, max_value):
    return max(min_value, min(max_value, value))


def autopilot_step(state: dict, dt: float) -> dict:
    global _ap_state
    if _ap_state is None:
        _ap_state = _init_state()

    sim_dt = state.get("dt", dt) or dt or 0.05
    dt = max(sim_dt, 1e-3)

    commands = {}

    # --- Sensor fusion ------------------------------------------------------
    vs_vals = _collect(
        [
            state.get("irs_1_vs_fpm"),
            state.get("irs_2_vs_fpm"),
            state.get("irs_3_vs_fpm"),
            state.get("vertical_speed_fpm"),
        ],
        -8000,
        8000,
    )
    vs_raw = median(vs_vals) if vs_vals else None
    if vs_raw is None:
        vs_est = _ap_state["vs_filt"]
    else:
        _ap_state["vs_filt"] = 0.7 * _ap_state["vs_filt"] + 0.3 * vs_raw
        vs_est = _ap_state["vs_filt"]

    pressure_alt_vals = _collect(
        [
            state.get("baro_alt_1_ft"),
            state.get("baro_alt_2_ft"),
            state.get("baro_alt_3_ft"),
            state.get("adc_1_pressure_alt_ft"),
            state.get("adc_2_pressure_alt_ft"),
            state.get("adc_3_pressure_alt_ft"),
        ],
        -2000,
        60000,
    )
    baro_alt = median(pressure_alt_vals) if pressure_alt_vals else None

    gps_alt_vals = _collect(
        [
            state.get("gps_1_alt_ft"),
            state.get("gps_2_alt_ft"),
            state.get("gps_3_alt_ft"),
        ],
        -2000,
        60000,
    )
    gps_alt = median(gps_alt_vals) if gps_alt_vals else None

    alt_ft = baro_alt
    if baro_alt is None and gps_alt is not None:
        alt_ft = gps_alt
    elif baro_alt is not None and gps_alt is not None:
        if abs(baro_alt - gps_alt) < 1200:
            alt_ft = 0.7 * baro_alt + 0.3 * gps_alt
        elif _ap_state["last_altitude_ft"] is not None and vs_est is not None:
            predicted = _ap_state["last_altitude_ft"] + vs_est * (dt / 60.0)
            if abs(gps_alt - predicted) < abs(baro_alt - predicted):
                alt_ft = gps_alt
    if alt_ft is None:
        alt_ft = _ap_state["last_altitude_ft"] if _ap_state["last_altitude_ft"] is not None else 0.0
    _ap_state["last_altitude_ft"] = alt_ft

    radar_vals = _collect(
        [
            state.get("radar_alt_1_ft"),
            state.get("radar_alt_2_ft"),
            state.get("radar_alt_3_ft"),
        ],
        0,
        5000,
    )
    radar_agl = median(radar_vals) if radar_vals else None

    ias_vals = _collect(
        [
            state.get("ias_1_kt"),
            state.get("ias_2_kt"),
            state.get("ias_3_kt"),
        ],
        40,
        450,
    )
    ias = median(ias_vals) if ias_vals else _ap_state["last_ias"]
    if ias is None:
        ias = 200.0
    _ap_state["last_ias"] = ias

    tas_vals = _collect(
        [
            state.get("tas_1_kt"),
            state.get("tas_2_kt"),
            state.get("tas_3_kt"),
        ],
        40,
        600,
    )
    tas = median(tas_vals) if tas_vals else ias

    mach_vals = _collect(
        [
            state.get("mach_1"),
            state.get("mach_2"),
            state.get("mach_3"),
        ],
        0,
        1.0,
    )
    mach = median(mach_vals) if mach_vals else (tas / 661.0 if tas else 0.6)

    pitch_vals = _collect(
        [
            state.get("irs_1_pitch_deg"),
            state.get("irs_2_pitch_deg"),
            state.get("irs_3_pitch_deg"),
        ],
        -25,
        20,
    )
    pitch_deg = median(pitch_vals) if pitch_vals else _ap_state["last_pitch"]
    _ap_state["last_pitch"] = pitch_deg

    roll_vals = _collect(
        [
            state.get("irs_1_roll_deg"),
            state.get("irs_2_roll_deg"),
            state.get("irs_3_roll_deg"),
        ],
        -60,
        60,
    )
    roll_deg = median(roll_vals) if roll_vals else _ap_state["last_roll"]
    _ap_state["last_roll"] = roll_deg

    heading_sources = [
        state.get("true_heading_deg"),
        state.get("irs_1_heading_deg"),
        state.get("irs_2_heading_deg"),
        state.get("irs_3_heading_deg"),
    ]
    heading_deg = _fuse_angle(heading_sources)
    if heading_deg is None:
        magnetic = state.get("magnetic_heading_deg")
        variation = state.get("magnetic_variation_deg")
        if _valid_number(magnetic) and _valid_number(variation):
            heading_deg = (magnetic + variation) % 360
        else:
            heading_deg = _ap_state["last_heading"]
    _ap_state["last_heading"] = heading_deg

    track_sources = [
        state.get("track_deg"),
        state.get("gps_1_track_deg"),
        state.get("gps_2_track_deg"),
        state.get("gps_3_track_deg"),
        state.get("irs_1_track_deg"),
        state.get("irs_2_track_deg"),
        state.get("irs_3_track_deg"),
    ]
    track_deg = _fuse_angle(track_sources)
    if track_deg is None:
        track_deg = heading_deg

    drift = state.get("drift_angle_deg")
    if not _valid_number(drift):
        drift = _wrap180(track_deg - heading_deg)

    roll_rate_vals = _collect(
        [
            state.get("gyro_1_roll_rate_dps"),
            state.get("gyro_2_roll_rate_dps"),
            state.get("gyro_3_roll_rate_dps"),
        ],
        -90,
        90,
    )
    roll_rate = median(roll_rate_vals) if roll_rate_vals else 0.0

    pitch_rate_vals = _collect(
        [
            state.get("gyro_1_pitch_rate_dps"),
            state.get("gyro_2_pitch_rate_dps"),
            state.get("gyro_3_pitch_rate_dps"),
        ],
        -30,
        30,
    )
    pitch_rate = median(pitch_rate_vals) if pitch_rate_vals else 0.0

    yaw_rate_vals = _collect(
        [
            state.get("gyro_1_yaw_rate_dps"),
            state.get("gyro_2_yaw_rate_dps"),
            state.get("gyro_3_yaw_rate_dps"),
        ],
        -30,
        30,
    )
    yaw_rate = median(yaw_rate_vals) if yaw_rate_vals else 0.0

    gs_vals = _collect(
        [
            state.get("ground_speed_kt"),
            state.get("gps_1_gs_kt"),
            state.get("gps_2_gs_kt"),
            state.get("gps_3_gs_kt"),
            state.get("irs_1_gs_kt"),
            state.get("irs_2_gs_kt"),
            state.get("irs_3_gs_kt"),
        ],
        0,
        700,
    )
    ground_speed = median(gs_vals) if gs_vals else tas

    flap_left = state.get("flap_pos_left_deg", 0.0)
    flap_right = state.get("flap_pos_right_deg", 0.0)
    flap_avg = 0.5 * (flap_left + flap_right)

    gross_weight = state.get("gross_weight_kg", 65000.0)

    # --- Flight envelope protection thresholds ------------------------------
    if flap_avg < 5:
        min_speed = 170.0
    elif flap_avg < 15:
        min_speed = 155.0
    elif flap_avg < 25:
        min_speed = 145.0
    else:
        min_speed = 135.0
    weight_adjust = _clamp((gross_weight - 65000.0) / 5000.0, -10.0, 15.0)
    min_speed = max(130.0, min_speed + weight_adjust)

    overspeed = (ias is not None and (ias > 350.0)) or (mach is not None and mach > 0.80)

    # --- Autothrottle -------------------------------------------------------
    target_ias = state.get("target_ias_kt")
    if not _valid_number(target_ias):
        target_ias = _ap_state["last_target_ias"]
    target_ias = _clamp(target_ias, 140.0, 330.0)
    _ap_state["last_target_ias"] = target_ias

    at_engaged = bool(state.get("at_engaged"))
    if at_engaged:
        if _ap_state["throttle_cmd"] is None:
            current_avg_throttle = _clamp(
                0.5 * (state.get("throttle_1_pos", 0.5) + state.get("throttle_2_pos", 0.5)),
                0.0,
                1.0,
            )
            _ap_state["throttle_cmd"] = current_avg_throttle

        speed_error = _clamp(target_ias - (ias or target_ias), -100.0, 100.0)
        _ap_state["spd_int"] = _clamp(_ap_state["spd_int"] + speed_error * dt * 0.02, -50.0, 50.0)

        throttle_delta = 0.0025 * speed_error + 0.0005 * _ap_state["spd_int"]
        throttle_cmd = _clamp(_ap_state["throttle_cmd"] + throttle_delta, 0.0, 1.0)

        low_speed_margin = max(0.0, min_speed - (ias or min_speed))
        if low_speed_margin > 0:
            throttle_cmd = max(throttle_cmd, 0.90)

        if overspeed:
            throttle_cmd = min(throttle_cmd, 0.10)

        _ap_state["throttle_cmd"] = throttle_cmd

        throttle_1 = throttle_cmd
        throttle_2 = throttle_cmd
        if state.get("engine_1_fire"):
            throttle_1 = 0.0
        if state.get("engine_2_fire"):
            throttle_2 = 0.0

        commands["throttle_1"] = throttle_1
        commands["throttle_2"] = throttle_2
    else:
        _ap_state["throttle_cmd"] = None
        _ap_state["spd_int"] = 0.0

    # --- Autopilot (surface controls) ---------------------------------------
    ap_engaged = bool(state.get("ap_engaged"))
    if not ap_engaged:
        commands["elevator"] = 0.0
        commands["aileron"] = 0.0
        commands["rudder"] = 0.0
        commands["elevator_trim"] = _ap_state["trim_bias"]
    else:
        target_alt = state.get("target_alt_ft")
        target_vs = state.get("target_vs_fpm")

        if not _valid_number(target_alt) and not _valid_number(target_vs):
            target_alt = alt_ft

        vs_sp = 0.0
        if _valid_number(target_alt):
            alt_error = target_alt - alt_ft
            vs_from_alt = _clamp(alt_error * 0.8, -2500.0, 2500.0)
            capture_band = 120.0
            if abs(alt_error) < capture_band:
                ratio = abs(alt_error) / capture_band
                vs_from_alt *= ratio
                if abs(alt_error) < 20.0:
                    vs_from_alt = 0.0
            if _valid_number(target_vs) and abs(alt_error) > 400.0:
                commanded_vs = _clamp(target_vs, -2500.0, 2500.0)
                vs_sp = _clamp(0.7 * vs_from_alt + 0.3 * commanded_vs, -2500.0, 2500.0)
            else:
                vs_sp = vs_from_alt
        elif _valid_number(target_vs):
            vs_sp = _clamp(target_vs, -2500.0, 2500.0)

        low_speed_margin = max(0.0, min_speed - (ias or min_speed))
        if low_speed_margin > 0 and vs_sp > 0:
            vs_sp = min(vs_sp, 500.0)
        if overspeed and vs_sp < 0:
            vs_sp = max(vs_sp, -500.0)
        if radar_agl is not None and radar_agl < 400.0 and vs_sp < -700.0:
            vs_sp = -700.0

        pitch_target = _clamp(vs_sp / 600.0, -10.0, 10.0)
        pitch_error = pitch_target - pitch_deg
        _ap_state["pitch_int"] = _clamp(_ap_state["pitch_int"] + pitch_error * dt * 0.1, -2.0, 2.0)

        vs_error = vs_sp - vs_est
        elevator_cmd = (
            0.075 * pitch_error
            + 0.02 * _ap_state["pitch_int"]
            + 0.0004 * vs_error
            - 0.02 * pitch_rate
        )

        if low_speed_margin > 0:
            elevator_cmd -= _clamp(low_speed_margin / 40.0, 0.0, 0.35)
        if overspeed:
            elevator_cmd += _clamp(((ias or 350.0) - 340.0) / 80.0, 0.0, 0.3)

        elevator_cmd = _clamp(elevator_cmd, -1.0, 1.0)

        # -- Lateral/Directional control
        target_hdg = state.get("target_hdg_deg")
        if not _valid_number(target_hdg):
            target_hdg = heading_deg
        heading_error = _wrap180(target_hdg - heading_deg)

        turbulence = state.get("turbulence_intensity", 0.0) or 0.0
        max_bank = 25.0 - 10.0 * _clamp(turbulence, 0.0, 1.0)
        bank_target = _clamp(heading_error * 0.5, -max_bank, max_bank)
        if abs(heading_error) < 3.0:
            bank_target *= 0.5

        roll_error = bank_target - roll_deg
        aileron_cmd = 0.06 * roll_error - 0.02 * roll_rate + 0.01 * heading_error
        aileron_cmd = _clamp(aileron_cmd, -1.0, 1.0)

        thrust_diff = (
            (state.get("engine_1_thrust_pct", 0.0) or 0.0)
            - (state.get("engine_2_thrust_pct", 0.0) or 0.0)
        ) / 100.0
        rudder_cmd = -0.07 * drift - 0.03 * yaw_rate + 0.25 * thrust_diff
        rudder_cmd = _clamp(rudder_cmd, -0.5, 0.5)

        # Trim: slowly drive toward neutral elevator
        _ap_state["trim_bias"] = _clamp(
            _ap_state["trim_bias"] + elevator_cmd * dt * 0.08, -0.6, 0.6
        )

        commands["elevator"] = elevator_cmd
        commands["aileron"] = aileron_cmd
        commands["rudder"] = rudder_cmd
        commands["elevator_trim"] = _ap_state["trim_bias"]

    # --- Drag devices / landing gear ----------------------------------------
    speedbrake_cmd = 0.0
    if overspeed and not (state.get("gear_handle_down") and radar_agl and radar_agl < 100.0):
        speedbrake_cmd = _clamp(((ias or 350.0) - 340.0) / 80.0, 0.0, 0.4)
    commands["speedbrake"] = speedbrake_cmd

    gear_cmd = 1 if state.get("gear_handle_down") else 0
    commands["gear"] = gear_cmd

    return commands
```
