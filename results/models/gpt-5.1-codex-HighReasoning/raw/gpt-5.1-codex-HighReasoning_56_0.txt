const fs = require('fs');

function distance(a, b) {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.sqrt(dx * dx + dy * dy);
}

function reverseSegment(arr, start, end) {
  while (start < end) {
    const tmp = arr[start];
    arr[start] = arr[end];
    arr[end] = tmp;
    start++;
    end--;
  }
}

function determineRouteCount(total, maxTrucks) {
  if (total <= 0) return 0;
  const maxVehicles = Math.max(1, maxTrucks);
  const approxSqrt = Math.max(1, Math.round(Math.sqrt(total)));
  const approxSize = Math.max(1, Math.ceil(total / 25));
  let desired = Math.max(approxSqrt, approxSize);
  desired = Math.min(desired, total);
  return Math.min(maxVehicles, desired);
}

function orderByAngle(packages, depotPoint) {
  const TWO_PI = Math.PI * 2;
  const list = packages.map(p => {
    const dx = p.x - depotPoint.x;
    const dy = p.y - depotPoint.y;
    let angle = Math.atan2(dy, dx);
    if (angle < 0) angle += TWO_PI;
    const dist2 = dx * dx + dy * dy;
    return { id: p.id, x: p.x, y: p.y, angle, dist2 };
  });

  list.sort((a, b) => {
    let diff = a.angle - b.angle;
    if (diff !== 0) return diff;
    diff = a.dist2 - b.dist2;
    if (diff !== 0) return diff;
    const aNum = typeof a.id === 'number';
    const bNum = typeof b.id === 'number';
    if (aNum && bNum) return a.id - b.id;
    return String(a.id).localeCompare(String(b.id));
  });

  if (list.length <= 1) return list;

  let maxGap = -1;
  let startIndex = 0;
  for (let i = 0; i < list.length; i++) {
    const current = list[i].angle;
    const nextAngle = (i === list.length - 1) ? list[0].angle + TWO_PI : list[i + 1].angle;
    const gap = nextAngle - current;
    if (gap > maxGap) {
      maxGap = gap;
      startIndex = (i + 1) % list.length;
    }
  }

  if (startIndex === 0) return list;
  return list.slice(startIndex).concat(list.slice(0, startIndex));
}

function splitSequentially(list, count) {
  const result = [];
  if (count <= 0) return result;
  let start = 0;
  const base = Math.floor(list.length / count);
  let remainder = list.length % count;
  for (let i = 0; i < count; i++) {
    const size = base + (i < remainder ? 1 : 0);
    result.push(list.slice(start, start + size));
    start += size;
  }
  return result;
}

function nearestCentroidIndex(point, centroids) {
  let bestIndex = 0;
  let bestDist = Infinity;
  for (let i = 0; i < centroids.length; i++) {
    const dx = point.x - centroids[i].x;
    const dy = point.y - centroids[i].y;
    const dist = dx * dx + dy * dy;
    if (dist < bestDist) {
      bestDist = dist;
      bestIndex = i;
    }
  }
  return bestIndex;
}

function clusterUsingKMeans(points, k) {
  const len = points.length;
  if (k <= 1 || len <= 1) {
    return [points.slice()];
  }
  const seeds = [];
  const step = len / k;
  for (let i = 0; i < k; i++) {
    const idx = Math.min(len - 1, Math.floor(i * step));
    seeds.push(idx);
  }
  const centroids = seeds.map(idx => ({ x: points[idx].x, y: points[idx].y }));
  const assignments = new Array(len).fill(-1);
  const locked = new Array(len).fill(false);

  for (let clusterIndex = 0; clusterIndex < k; clusterIndex++) {
    const pointIndex = seeds[clusterIndex];
    assignments[pointIndex] = clusterIndex;
    locked[pointIndex] = true;
  }

  for (let i = 0; i < len; i++) {
    if (assignments[i] !== -1) continue;
    assignments[i] = nearestCentroidIndex(points[i], centroids);
  }

  const iterationLimit = Math.min(8, Math.max(2, Math.floor(4000 / Math.max(1, k))));
  for (let iter = 0; iter < iterationLimit; iter++) {
    const counts = new Array(k).fill(0);
    const sumX = new Array(k).fill(0);
    const sumY = new Array(k).fill(0);
    for (let i = 0; i < len; i++) {
      const cluster = assignments[i];
      counts[cluster]++;
      sumX[cluster] += points[i].x;
      sumY[cluster] += points[i].y;
    }
    for (let c = 0; c < k; c++) {
      if (counts[c] > 0) {
        centroids[c].x = sumX[c] / counts[c];
        centroids[c].y = sumY[c] / counts[c];
      }
    }
    let anyChange = false;
    for (let i = 0; i < len; i++) {
      if (locked[i]) continue;
      const point = points[i];
      let bestCluster = assignments[i];
      let bestDist = Infinity;
      for (let c = 0; c < k; c++) {
        const dx = point.x - centroids[c].x;
        const dy = point.y - centroids[c].y;
        const dist = dx * dx + dy * dy;
        if (dist < bestDist) {
          bestDist = dist;
          bestCluster = c;
        }
      }
      if (bestCluster !== assignments[i]) {
        assignments[i] = bestCluster;
        anyChange = true;
      }
    }
    if (!anyChange) break;
  }

  const clusters = Array.from({ length: k }, () => []);
  for (let i = 0; i < len; i++) {
    const clusterIndex = assignments[i];
    if (clusterIndex >= 0) {
      clusters[clusterIndex].push(points[i]);
    }
  }
  return clusters;
}

function planCluster(cluster, depotPoint) {
  if (!cluster || cluster.length === 0) return [];
  if (cluster.length === 1) return [cluster[0].id];

  const n = cluster.length;
  const visited = new Array(n).fill(false);
  const order = [];
  let currentX = depotPoint.x;
  let currentY = depotPoint.y;

  for (let step = 0; step < n; step++) {
    let bestIdx = -1;
    let bestDist = Infinity;
    for (let i = 0; i < n; i++) {
      if (visited[i]) continue;
      const dx = cluster[i].x - currentX;
      const dy = cluster[i].y - currentY;
      const dist = dx * dx + dy * dy;
      if (dist < bestDist) {
        bestDist = dist;
        bestIdx = i;
      }
    }
    visited[bestIdx] = true;
    const nextPoint = cluster[bestIdx];
    order.push(nextPoint);
    currentX = nextPoint.x;
    currentY = nextPoint.y;
  }

  twoOpt(order, depotPoint);
  return order.map(p => p.id);
}

function twoOpt(order, depotPoint) {
  const n = order.length;
  if (n < 4) return;

  let window;
  if (n > 2000) window = 18;
  else if (n > 1000) window = 25;
  else if (n > 600) window = 35;
  else if (n > 300) window = 55;
  else window = n;

  const iterationLimit = n > 2000 ? 2 : n > 1000 ? 3 : n > 400 ? 5 : 8;
  const lastIndex = n - 1;

  for (let iter = 0; iter < iterationLimit; iter++) {
    let improved = false;
    for (let i = 0; i < n - 1 && !improved; i++) {
      const A = (i === 0) ? depotPoint : order[i - 1];
      const B = order[i];
      for (let k = i + 2; k <= lastIndex; k++) {
        if (k - i > window && k < lastIndex - 1) continue;
        if (i === 0 && k === lastIndex) continue;
        const C = order[k];
        const D = (k === lastIndex) ? depotPoint : order[k + 1];
        const current = distance(A, B) + distance(C, D);
        const proposed = distance(A, C) + distance(B, D);
        if (proposed + 1e-9 < current) {
          reverseSegment(order, i, k);
          improved = true;
          break;
        }
      }
    }
    if (!improved) break;
  }
}

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  process.stdout.write('{"routes":[]}');
  process.exit(0);
}
const data = JSON.parse(input);

const depotArray = Array.isArray(data.depot) ? data.depot : [0, 0];
const depotPoint = {
  x: Number(depotArray[0]) || 0,
  y: Number(depotArray[1]) || 0
};

const rawPackages = Array.isArray(data.packages) ? data.packages : [];
const packages = rawPackages.map(pkg => ({
  id: pkg.id,
  x: Number(pkg.x) || 0,
  y: Number(pkg.y) || 0
}));

const totalPackages = packages.length;
if (totalPackages === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

let maxTrucks = Number(data.truckCount);
if (!Number.isFinite(maxTrucks)) {
  maxTrucks = Math.max(1, totalPackages);
} else {
  maxTrucks = Math.floor(maxTrucks);
  if (maxTrucks < 1) maxTrucks = 1;
}

const orderedList = orderByAngle(packages, depotPoint);
const routeCount = determineRouteCount(totalPackages, maxTrucks) || 1;

let clusters;
const useKMeans = routeCount > 1 && (totalPackages * routeCount <= 5000000);
if (useKMeans) {
  clusters = clusterUsingKMeans(orderedList, routeCount);
  let nonEmpty = 0;
  for (let i = 0; i < clusters.length; i++) {
    if (clusters[i].length > 0) nonEmpty++;
  }
  if (nonEmpty < routeCount) {
    clusters = splitSequentially(orderedList, routeCount);
  }
} else {
  clusters = splitSequentially(orderedList, routeCount);
}

const routes = [];
for (const cluster of clusters) {
  if (!cluster || cluster.length === 0) continue;
  routes.push(planCluster(cluster, depotPoint));
}

let assignedCount = 0;
for (const route of routes) {
  assignedCount += route.length;
}
if (assignedCount !== totalPackages) {
  routes.length = 0;
  routes.push(orderedList.map(p => p.id));
}

process.stdout.write(JSON.stringify({ routes }));
