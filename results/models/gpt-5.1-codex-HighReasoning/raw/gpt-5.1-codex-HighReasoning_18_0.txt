#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <unordered_map>
#include <utility>
#include <vector>

struct AdjEdge {
    int to;
    int id;
};

struct MergeNode {
    int left;
    int right;
};

struct CutResult {
    long long value;
    std::vector<char> side;
};

std::vector<char> buildMembership(int n, const std::vector<MergeNode>& tree,
                                  const std::vector<int>& roots) {
    std::vector<char> mark(n, 0);
    if (roots.empty()) {
        return mark;
    }
    std::vector<int> stack;
    stack.reserve(roots.size() * 2 + n);
    for (int root : roots) {
        if (root < 0 || root >= static_cast<int>(tree.size())) {
            continue;
        }
        stack.push_back(root);
        while (!stack.empty()) {
            int node = stack.back();
            stack.pop_back();
            if (node < 0 || node >= static_cast<int>(tree.size())) {
                continue;
            }
            const MergeNode& cur = tree[node];
            if (cur.left == -1 && cur.right == -1) {
                if (node < n) {
                    mark[node] = 1;
                }
            } else {
                if (cur.left != -1) stack.push_back(cur.left);
                if (cur.right != -1) stack.push_back(cur.right);
            }
        }
    }
    return mark;
}

void ensureNonTrivialCut(std::vector<char>& side, const std::vector<uint64_t>& degree) {
    int n = static_cast<int>(side.size());
    int cnt = 0;
    for (char v : side) cnt += (v != 0);
    if (cnt == 0 || cnt == n) {
        int bestVertex = -1;
        uint64_t bestDeg = std::numeric_limits<uint64_t>::max();
        for (int i = 0; i < n; ++i) {
            if (degree[i] > 0 && degree[i] < bestDeg) {
                bestDeg = degree[i];
                bestVertex = i;
            }
        }
        if (bestVertex == -1) {
            if (n == 0) return;
            bestVertex = 0;
        }
        std::fill(side.begin(), side.end(), 0);
        side[bestVertex] = 1;
    }
}

int findBridge(const std::vector<std::vector<AdjEdge>>& g) {
    int n = static_cast<int>(g.size());
    std::vector<int> tin(n, -1), low(n, -1), parent(n, -1), parentEdge(n, -1);
    int timer = 0;
    int bridge = -1;
    std::vector<std::pair<int, size_t>> stack;
    stack.reserve(n);
    for (int start = 0; start < n && bridge == -1; ++start) {
        if (tin[start] != -1) continue;
        stack.clear();
        stack.push_back({start, 0});
        parent[start] = -1;
        parentEdge[start] = -1;
        while (!stack.empty() && bridge == -1) {
            auto& frame = stack.back();
            int v = frame.first;
            if (frame.second == 0 && tin[v] == -1) {
                tin[v] = low[v] = timer++;
            }
            if (frame.second < g[v].size()) {
                const auto& edge = g[v][frame.second++];
                int to = edge.to;
                if (edge.id == parentEdge[v]) continue;
                if (tin[to] == -1) {
                    parent[to] = v;
                    parentEdge[to] = edge.id;
                    stack.push_back({to, 0});
                } else {
                    low[v] = std::min(low[v], tin[to]);
                }
            } else {
                stack.pop_back();
                if (parent[v] != -1) {
                    low[parent[v]] = std::min(low[parent[v]], low[v]);
                    if (low[v] > tin[parent[v]]) {
                        bridge = parentEdge[v];
                        break;
                    }
                }
            }
        }
    }
    return bridge;
}

CutResult stoerW_dense(int n, const std::vector<std::pair<int, int>>& edges) {
    std::vector<std::vector<long long>> cap(n, std::vector<long long>(n, 0));
    for (const auto& e : edges) {
        int u = e.first;
        int v = e.second;
        cap[u][v] += 1;
        cap[v][u] += 1;
    }

    std::vector<MergeNode> mergeTree;
    mergeTree.reserve(static_cast<size_t>(2) * n);
    for (int i = 0; i < n; ++i) mergeTree.push_back({-1, -1});
    std::vector<int> treeId(n);
    std::iota(treeId.begin(), treeId.end(), 0);

    std::vector<int> verts(n);
    std::iota(verts.begin(), verts.end(), 0);
    int activeCount = n;

    std::vector<long long> weight(n, 0);
    std::vector<char> added(n, 0);
    std::vector<int> order;
    order.reserve(n);

    long long bestCut = std::numeric_limits<long long>::max();
    std::vector<int> bestRoots;

    for (int phase = 0; phase < n - 1 && activeCount > 1; ++phase) {
        for (int idx = 0; idx < activeCount; ++idx) {
            int v = verts[idx];
            weight[v] = 0;
            added[v] = 0;
        }
        order.clear();
        int prev = -1;
        int last = -1;

        for (int iter = 0; iter < activeCount; ++iter) {
            int sel = -1;
            for (int idx = 0; idx < activeCount; ++idx) {
                int v = verts[idx];
                if (added[v]) continue;
                if (sel == -1 || weight[v] > weight[sel]) sel = v;
            }
            if (sel == -1) break;
            added[sel] = 1;
            order.push_back(sel);
            prev = last;
            last = sel;
            for (int idx = 0; idx < activeCount; ++idx) {
                int v = verts[idx];
                if (!added[v]) weight[v] += cap[sel][v];
            }
        }

        if (last == -1) break;
        long long cutValue = weight[last];
        if (!order.empty() && cutValue < bestCut) {
            bestCut = cutValue;
            bestRoots.clear();
            for (size_t idx = 0; idx + 1 < order.size(); ++idx) {
                bestRoots.push_back(treeId[order[idx]]);
            }
        }
        if (prev == -1) break;

        int a = prev;
        int b = last;
        for (int idx = 0; idx < activeCount; ++idx) {
            int v = verts[idx];
            if (v == a || v == b) continue;
            cap[a][v] += cap[b][v];
            cap[v][a] = cap[a][v];
        }

        auto it = std::find(verts.begin(), verts.begin() + activeCount, b);
        if (it != verts.begin() + activeCount) {
            verts.erase(it);
        }
        --activeCount;

        mergeTree.push_back({treeId[a], treeId[b]});
        int newNode = static_cast<int>(mergeTree.size()) - 1;
        treeId[a] = newNode;
        treeId[b] = newNode;
    }

    if (bestRoots.empty() && !verts.empty()) {
        bestRoots.push_back(treeId[verts[0]]);
        bestCut = 0;
    }

    std::vector<char> side = buildMembership(n, mergeTree, bestRoots);
    return {bestCut, std::move(side)};
}

CutResult stoerW_sparse(int n, const std::vector<std::pair<int, int>>& edges,
                        const std::vector<uint64_t>& degree) {
    std::vector<std::unordered_map<int, long long>> adj(n);
    for (int i = 0; i < n; ++i) {
        size_t reserveSize = static_cast<size_t>(std::max<uint64_t>(1, degree[i] * 2 + 1));
        adj[i].reserve(reserveSize);
        adj[i].max_load_factor(0.7f);
    }
    for (const auto& e : edges) {
        int u = e.first;
        int v = e.second;
        adj[u][v] += 1;
        adj[v][u] += 1;
    }

    std::vector<MergeNode> mergeTree;
    mergeTree.reserve(static_cast<size_t>(2) * n);
    for (int i = 0; i < n; ++i) mergeTree.push_back({-1, -1});
    std::vector<int> treeId(n);
    std::iota(treeId.begin(), treeId.end(), 0);

    std::vector<int> activeOrder(n);
    std::iota(activeOrder.begin(), activeOrder.end(), 0);
    std::vector<int> position(n);
    for (int i = 0; i < n; ++i) position[i] = i;
    std::vector<char> active(n, 1);
    int activeCount = n;

    std::vector<long long> weight(n, 0);
    std::vector<char> added(n, 0);
    std::vector<int> order;
    order.reserve(n);
    std::priority_queue<std::pair<long long, int>> pq;

    long long bestCut = std::numeric_limits<long long>::max();
    std::vector<int> bestRoots;

    while (activeCount > 1) {
        for (int idx = 0; idx < activeCount; ++idx) {
            int v = activeOrder[idx];
            weight[v] = 0;
            added[v] = 0;
        }
        order.clear();
        while (!pq.empty()) pq.pop();

        int prev = -1;
        int last = -1;

        for (int iter = 0; iter < activeCount; ++iter) {
            int sel = -1;
            while (!pq.empty()) {
                auto [w, v] = pq.top();
                pq.pop();
                if (!active[v] || added[v]) continue;
                if (w != weight[v]) continue;
                sel = v;
                break;
            }
            if (sel == -1) {
                for (int idx = 0; idx < activeCount; ++idx) {
                    int v = activeOrder[idx];
                    if (!added[v]) {
                        sel = v;
                        break;
                    }
                }
            }
            if (sel == -1) break;

            added[sel] = 1;
            order.push_back(sel);
            prev = last;
            last = sel;

            for (const auto& kv : adj[sel]) {
                int to = kv.first;
                if (!active[to] || added[to]) continue;
                weight[to] += kv.second;
                pq.emplace(weight[to], to);
            }
        }

        if (last == -1) break;
        long long cutValue = weight[last];
        if (!order.empty() && cutValue < bestCut) {
            bestCut = cutValue;
            bestRoots.clear();
            for (size_t idx = 0; idx + 1 < order.size(); ++idx) {
                bestRoots.push_back(treeId[order[idx]]);
            }
        }
        if (prev == -1) break;

        int a = prev;
        int b = last;

        int newNode = static_cast<int>(mergeTree.size());
        mergeTree.push_back({treeId[a], treeId[b]});
        treeId[a] = newNode;
        treeId[b] = newNode;

        adj[a].reserve(adj[a].size() + adj[b].size() + 1);
        for (const auto& kv : adj[b]) {
            int neighbor = kv.first;
            long long w = kv.second;
            if (neighbor == a) continue;
            if (!active[neighbor]) continue;
            adj[a][neighbor] += w;
            adj[neighbor][a] += w;
            adj[neighbor].erase(b);
        }
        adj[a].erase(b);
        adj[a].erase(a);
        active[b] = 0;
        adj[b].clear();

        int pos = position[b];
        int lastIdx = activeCount - 1;
        int swapV = activeOrder[lastIdx];
        std::swap(activeOrder[pos], activeOrder[lastIdx]);
        position[swapV] = pos;
        --activeCount;
        position[b] = -1;
    }

    if (bestRoots.empty() && activeCount > 0) {
        bestRoots.push_back(treeId[activeOrder[0]]);
        bestCut = 0;
    }

    std::vector<char> side = buildMembership(n, mergeTree, bestRoots);
    return {bestCut, std::move(side)};
}

std::vector<std::pair<int, int>> gatherCutEdges(const std::vector<std::pair<int, int>>& edges,
                                                const std::vector<char>& side,
                                                long long expected) {
    size_t reserveCount = 0;
    if (expected > 0) {
        long long limited = std::min<long long>(expected, static_cast<long long>(edges.size()));
        reserveCount = static_cast<size_t>(limited);
    }
    std::vector<std::pair<int, int>> cut;
    if (reserveCount > 0) cut.reserve(reserveCount);
    for (const auto& e : edges) {
        if (side[e.first] != side[e.second]) {
            cut.push_back(e);
        }
    }
    return cut;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    long long mInput;
    if (!(std::cin >> n >> mInput)) {
        return 0;
    }

    std::vector<std::pair<int, int>> edges;
    if (mInput > 0) {
        edges.reserve(static_cast<size_t>(mInput));
    }
    std::vector<uint64_t> degree;
    if (n > 0) degree.assign(n, 0);

    for (long long i = 0; i < mInput; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (n == 0) continue;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        edges.emplace_back(u, v);
        degree[u]++;
        degree[v]++;
    }

    if (n == 0) {
        std::cout << 0 << "\n";
        return 0;
    }
    if (n == 1) {
        std::cout << 0 << "\n";
        return 0;
    }
    if (edges.empty()) {
        std::cout << 0 << "\n";
        return 0;
    }

    for (int i = 0; i < n; ++i) {
        if (degree[i] == 0) {
            std::cout << 0 << "\n";
            return 0;
        }
    }

    std::vector<std::vector<AdjEdge>> g(n);
    for (int i = 0; i < n; ++i) {
        g[i].reserve(static_cast<size_t>(degree[i]));
    }
    for (int id = 0; id < static_cast<int>(edges.size()); ++id) {
        auto [u, v] = edges[id];
        g[u].push_back({v, id});
        g[v].push_back({u, id});
    }

    for (int i = 0; i < n; ++i) {
        if (degree[i] == 1 && !g[i].empty()) {
            auto e = edges[g[i][0].id];
            std::cout << 1 << "\n" << e.first << " " << e.second << "\n";
            return 0;
        }
    }

    int bridgeEdge = findBridge(g);
    if (bridgeEdge != -1) {
        auto e = edges[bridgeEdge];
        std::cout << 1 << "\n" << e.first << " " << e.second << "\n";
        return 0;
    }

    std::vector<std::vector<AdjEdge>>().swap(g);

    constexpr int DENSE_THRESHOLD = 1200;
    CutResult result;
    if (n <= DENSE_THRESHOLD) {
        result = stoerW_dense(n, edges);
    } else {
        result = stoerW_sparse(n, edges, degree);
    }

    ensureNonTrivialCut(result.side, degree);

    auto cutEdges = gatherCutEdges(edges, result.side, result.value);
    if (cutEdges.empty()) {
        int bestVertex = -1;
        uint64_t bestDeg = std::numeric_limits<uint64_t>::max();
        for (int i = 0; i < n; ++i) {
            if (degree[i] > 0 && degree[i] < bestDeg) {
                bestDeg = degree[i];
                bestVertex = i;
            }
        }
        if (bestVertex == -1) bestVertex = 0;
        for (const auto& e : edges) {
            if (e.first == bestVertex || e.second == bestVertex) {
                cutEdges.push_back(e);
            }
        }
    }

    std::cout << cutEdges.size() << "\n";
    for (const auto& e : cutEdges) {
        std::cout << e.first << " " << e.second << "\n";
    }
    return 0;
}
