const fs = require('fs');

class BinaryHeap {
    constructor(compare) {
        this.data = [];
        this.compare = compare;
    }
    size() {
        return this.data.length;
    }
    peek() {
        return this.data.length > 0 ? this.data[0] : null;
       }
    push(value) {
        this.data.push(value);
        this._siftUp(this.data.length - 1);
    }
    pop() {
        if (this.data.length === 0) return null;
        const root = this.data[0];
        const last = this.data.pop();
        if (this.data.length > 0) {
            this.data[0] = last;
            this._siftDown(0);
        }
        return root;
    }
    _siftUp(index) {
        const data = this.data;
        const compare = this.compare;
        while (index > 0) {
            const parent = (index - 1) >> 1;
            if (compare(data[index], data[parent]) < 0) {
                [data[index], data[parent]] = [data[parent], data[index]];
                index = parent;
            } else {
                break;
            }
        }
    }
    _siftDown(index) {
        const data = this.data;
        const compare = this.compare;
        const length = data.length;
        while (true) {
            let left = index * 2 + 1;
            if (left >= length) break;
            let right = left + 1;
            let smallest = left;
            if (right < length && compare(data[right], data[left]) < 0) {
                smallest = right;
            }
            if (compare(data[smallest], data[index]) < 0) {
                [data[smallest], data[index]] = [data[index], data[smallest]];
                index = smallest;
            } else {
                break;
            }
        }
    }
}

const keyFor = (value) => `${typeof value}:${String(value)}`;
const MIN_WEIGHT = 1e-6;

function main() {
    let input = '';
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (err) {
        input = '';
    }
    if (!input.trim()) {
        process.stdout.write('{"schedule":[]}');
        return;
    }
    let data;
    try {
        data = JSON.parse(input);
    } catch (err) {
        process.stdout.write('{"schedule":[]}');
        return;
    }
    const tasks = Array.isArray(data.tasks) ? data.tasks : [];
    let machineCount = Number(data.machineCount);
    if (!Number.isFinite(machineCount)) machineCount = 1;
    machineCount = Math.max(1, Math.floor(machineCount));
    const n = tasks.length;
    if (n === 0) {
        process.stdout.write('{"schedule":[]}');
        return;
    }
    if (machineCount > n) machineCount = n;

    const ids = new Array(n);
    const idToIndexExact = new Map();
    const idToIndexLoose = new Map();

    for (let i = 0; i < n; i++) {
        let idVal = tasks[i].id;
        if (idVal === undefined || idVal === null) {
            idVal = `task_${i}`;
        }
        ids[i] = idVal;
        const exactKey = keyFor(idVal);
        idToIndexExact.set(exactKey, i);
        const looseKey = String(idVal);
        if (!idToIndexLoose.has(looseKey)) {
            idToIndexLoose.set(looseKey, i);
        }
    }

    const durations = new Array(n);
    const deadlines = new Array(n);
    const weightsEff = new Array(n);
    const ratios = new Array(n);
    const dependents = Array.from({ length: n }, () => []);
    const indegreeOriginal = new Array(n);

    for (let i = 0; i < n; i++) {
        const task = tasks[i];
        let dur = Number(task.duration);
        if (!Number.isFinite(dur) || dur < 0) dur = 0;
        durations[i] = dur;

        let dl = Number(task.deadline);
        deadlines[i] = Number.isFinite(dl) ? dl : Infinity;

        let w = Number(task.weight);
        if (!Number.isFinite(w)) w = 1;
        if (w <= 0) w = MIN_WEIGHT;
        weightsEff[i] = w;
        ratios[i] = dur / w;

        const depsInput = Array.isArray(task.deps) ? task.deps : [];
        const seen = new Set();
        for (const dep of depsInput) {
            let depIdx = idToIndexExact.get(keyFor(dep));
            if (depIdx === undefined) {
                depIdx = idToIndexLoose.get(String(dep));
            }
            if (depIdx === undefined) continue;
            if (depIdx === i) continue;
            if (seen.has(depIdx)) continue;
            seen.add(depIdx);
            dependents[depIdx].push(i);
        }
        indegreeOriginal[i] = seen.size;
    }

    const compareDeadlineOnly = (a, b) => {
        const da = deadlines[a];
        const db = deadlines[b];
        const finiteA = Number.isFinite(da);
        const finiteB = Number.isFinite(db);
        if (finiteA && !finiteB) return -1;
        if (!finiteA && finiteB) return 1;
        if (da === db) return 0;
        if (!finiteA && !finiteB) return 0;
        return da - db;
    };

    const fallbackOrder = (() => {
        const indeg = indegreeOriginal.slice();
        const heap = new BinaryHeap((a, b) => {
            const cmp = compareDeadlineOnly(a, b);
            if (cmp !== 0) return cmp;
            const ratioDiff = ratios[a] - ratios[b];
            if (ratioDiff !== 0) return ratioDiff;
            const durDiff = durations[a] - durations[b];
            if (durDiff !== 0) return durDiff;
            const weightDiff = weightsEff[b] - weightsEff[a];
            if (weightDiff !== 0) return weightDiff;
            const idA = String(ids[a]);
            const idB = String(ids[b]);
            if (idA < idB) return -1;
            if (idA > idB) return 1;
            return a - b;
        });
        for (let i = 0; i < n; i++) {
            if (indeg[i] === 0) heap.push(i);
        }
        const orderIdx = [];
        while (heap.size() > 0) {
            const idx = heap.pop();
            orderIdx.push(idx);
            for (const dep of dependents[idx]) {
                indeg[dep]--;
                if (indeg[dep] === 0) heap.push(dep);
            }
        }
        if (orderIdx.length < n) {
            const seen = new Array(n).fill(false);
            for (const idx of orderIdx) seen[idx] = true;
            for (let i = 0; i < n; i++) {
                if (!seen[i]) orderIdx.push(i);
            }
        }
        return orderIdx.map((idx) => ids[idx]);
    })();

    const schedule = (() => {
        const readyTime = new Array(n).fill(0);
        const indegree = indegreeOriginal.slice();

        const readyComparator = (a, b) => {
            let cmp = compareDeadlineOnly(a, b);
            if (cmp !== 0) return cmp;
            const da = deadlines[a];
            const db = deadlines[b];
            const finiteA = Number.isFinite(da);
            const finiteB = Number.isFinite(db);
            if (finiteA || finiteB) {
                const slackA = finiteA ? da - readyTime[a] - durations[a] : Infinity;
                const slackB = finiteB ? db - readyTime[b] - durations[b] : Infinity;
                if (slackA !== slackB) {
                    const finiteSlackA = Number.isFinite(slackA);
                    const finiteSlackB = Number.isFinite(slackB);
                    if (finiteSlackA && !finiteSlackB) return -1;
                    if (!finiteSlackA && finiteSlackB) return 1;
                    const diffSlack = slackA - slackB;
                    if (diffSlack !== 0) return diffSlack;
                }
            }
            const ratioDiff = ratios[a] - ratios[b];
            if (ratioDiff !== 0) return ratioDiff;
            const durDiff = durations[a] - durations[b];
            if (durDiff !== 0) return durDiff;
            const weightDiff = weightsEff[b] - weightsEff[a];
            if (weightDiff !== 0) return weightDiff;
            const readyDiff = readyTime[a] - readyTime[b];
            if (readyDiff !== 0) return readyDiff;
            const idA = String(ids[a]);
            const idB = String(ids[b]);
            if (idA < idB) return -1;
            if (idA > idB) return 1;
            return a - b;
        };

        const readyHeap = new BinaryHeap(readyComparator);
        const notReadyHeap = new BinaryHeap((a, b) => {
            const diff = readyTime[a] - readyTime[b];
            if (diff !== 0) return diff;
            return a - b;
        });
        const machineHeap = new BinaryHeap((a, b) => {
            if (a.time !== b.time) return a.time - b.time;
            return a.machine - b.machine;
        });

        for (let i = 0; i < machineCount; i++) {
            machineHeap.push({ time: 0, machine: i });
        }
        for (let i = 0; i < n; i++) {
            if (indegree[i] === 0) {
                notReadyHeap.push(i);
            }
        }

        const result = [];
        let scheduledCount = 0;
        let feasible = true;

        while (scheduledCount < n) {
            if (machineHeap.size() === 0) {
                feasible = false;
                break;
            }
            const machine = machineHeap.pop();
            let currentTime = machine.time;
            let foundTask = true;

            while (true) {
                while (notReadyHeap.size() > 0) {
                    const candidate = notReadyHeap.peek();
                    if (candidate === null) break;
                    if (readyTime[candidate] <= currentTime) {
                        const item = notReadyHeap.pop();
                        if (item === null) break;
                        readyHeap.push(item);
                    } else {
                        break;
                    }
                }
                if (readyHeap.size() > 0) {
                    break;
                }
                if (notReadyHeap.size() === 0) {
                    foundTask = false;
                    break;
                }
                const nextIdx = notReadyHeap.peek();
                if (nextIdx === null) {
                    foundTask = false;
                    break;
                }
                const nextTime = readyTime[nextIdx];
                if (!Number.isFinite(nextTime)) {
                    foundTask = false;
                    break;
                }
                currentTime = Math.max(currentTime, nextTime);
            }

            if (!foundTask || readyHeap.size() === 0) {
                feasible = false;
                break;
            }

            const taskIdx = readyHeap.pop();
            const startTime = Math.max(currentTime, readyTime[taskIdx]);
            const finishTime = startTime + durations[taskIdx];
            result.push({ id: ids[taskIdx], start: startTime, machine: machine.machine });
            machineHeap.push({ time: finishTime, machine: machine.machine });
            scheduledCount++;

            for (const dep of dependents[taskIdx]) {
                if (readyTime[dep] < finishTime) {
                    readyTime[dep] = finishTime;
                }
                indegree[dep]--;
                if (indegree[dep] === 0) {
                    notReadyHeap.push(dep);
                }
            }
        }

        if (!feasible || result.length !== n) {
            return null;
        }
        return result;
    })();

    if (schedule) {
        schedule.sort((a, b) => {
            if (a.start !== b.start) return a.start - b.start;
            if (a.machine !== b.machine) return a.machine - b.machine;
            const idA = String(a.id);
            const idB = String(b.id);
            if (idA < idB) return -1;
            if (idA > idB) return 1;
            return 0;
        });
        process.stdout.write(JSON.stringify({ schedule }));
    } else {
        process.stdout.write(JSON.stringify({ order: fallbackOrder }));
    }
}

main();
