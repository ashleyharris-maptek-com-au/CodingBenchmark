OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeInt 32 0

%gl_PerVertex_out = OpTypeStruct %v4float

%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out

%arr_v3float_uint_2 = OpTypeArray %v3float %uint_2

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v3float = OpTypePointer Output %v3float

%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_neg1 = OpConstant %float -1
%float_0_03 = OpConstant %float 0.03
%vec_sign = OpConstantComposite %v2float %float_neg1 %float_1
%gl_in = OpVariable %ptr_Input_arr_pv Input
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_pv Output
%outColor = OpVariable %ptr_Output_v3float Output
%main = OpFunction %void None %fn
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%c1 = OpLoad %v3float %c1_ptr
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy
%dir_yx = OpVectorShuffle %v2float %dir %dir 1 0
%perp_temp = OpFMul %v2float %dir_yx %vec_sign
%dot = OpDot %float %perp_temp %perp_temp
%len = OpExtInst %float %glsl Sqrt %dot
%scale = OpFDiv %float %float_0_03 %len
%scale_vec = OpCompositeConstruct %v2float %scale %scale
%perp = OpFMul %v2float %perp_temp %scale_vec
%perp_x = OpCompositeExtract %float %perp 0
%perp_y = OpCompositeExtract %float %perp 1
%perp4 = OpCompositeConstruct %v4float %perp_x %perp_y %float_0 %float_0
%neg_perp4 = OpFNegate %v4float %perp4
%p0_plus = OpFAdd %v4float %p0 %perp4
%p0_minus = OpFAdd %v4float %p0 %neg_perp4
%p1_plus = OpFAdd %v4float %p1 %perp4
%p1_minus = OpFAdd %v4float %p1 %neg_perp4
%pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %pos_ptr %p0_plus
OpStore %outColor %c0
OpEmitVertex
OpStore %pos_ptr %p0_minus
OpStore %outColor %c0
OpEmitVertex
OpStore %pos_ptr %p1_plus
OpStore %outColor %c1
OpEmitVertex

OpStore %outColor %c1
