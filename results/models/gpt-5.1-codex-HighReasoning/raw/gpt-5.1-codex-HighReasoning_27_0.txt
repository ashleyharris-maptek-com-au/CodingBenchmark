using namespace std;

struct FastScanner {
    static const int BUFSIZE = 1 << 20;
    int idx = 0, size = 0;
    char buf[BUFSIZE];

    inline char read() {
        if (idx >= size) {
            size = static_cast<int>(fread(buf, 1, BUFSIZE, stdin));
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template<typename T>
    bool nextInt(T& out) {
        char c = read();
        while (c && isspace(static_cast<unsigned char>(c))) c = read();
        if (!c) return false;
        T sign = 1;
        if (c == '-') {
            sign = -1;
            c = read();
        }
        T val = 0;
        while (c && c >= '0' && c <= '9') {
            val = val * 10 + (c - '0');
            c = read();
        }
        out = val * sign;
        return true;
    }
};

static vector<int> buildDegreeOrder(const vector<int>& degree) {
    int n = static_cast<int>(degree.size());
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (degree[a] != degree[b]) return degree[a] > degree[b];
        return a < b;
    });
    return order;
}

static vector<int> computeDegeneracyOrder(const vector<vector<int>>& adj) {
    int n = static_cast<int>(adj.size());
    vector<int> degree(n);
    for (int i = 0; i < n; ++i) degree[i] = static_cast<int>(adj[i].size());
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    for (int i = 0; i < n; ++i) pq.emplace(degree[i], i);
    vector<char> removed(n, false);
    vector<int> order;
    order.reserve(n);
    while (static_cast<int>(order.size()) < n) {
        auto cur = pq.top(); pq.pop();
        int deg = cur.first;
        int v = cur.second;
        if (removed[v]) continue;
        if (deg != degree[v]) continue;
        removed[v] = true;
        order.push_back(v);
        for (int u : adj[v]) {
            if (!removed[u]) {
                --degree[u];
                pq.emplace(degree[u], u);
            }
        }
    }
    reverse(order.begin(), order.end());
    return order;
}

static bool simpleGreedy(const vector<vector<int>>& adj, int k, const vector<int>& order, vector<int>& color) {
    if (k <= 0) return adj.empty();
    int n = static_cast<int>(adj.size());
    color.assign(n, -1);
    vector<int> used(k, -1);
    int token = 1;
    for (int v : order) {
        for (int u : adj[v]) {
            int c = color[u];
            if (c != -1) used[c] = token;
        }
        int chosen = -1;
        for (int c = 0; c < k; ++c) {
            if (used[c] != token) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) return false;
        color[v] = chosen;
        ++token;
        if (token >= INT_MAX / 4) {
            fill(used.begin(), used.end(), -1);
            token = 1;
        }
    }
    return true;
}

static int chooseDepthLimit(int k) {
    if (k <= 4) return 5;
    if (k <= 8) return 4;
    if (k <= 16) return 3;
    if (k <= 64) return 2;
    return 1;
}

class GreedyRecolor {
public:
    GreedyRecolor(const vector<vector<int>>& graph, int kColors)
        : adj(graph), k(kColors),
          used(max(kColors, 1), -1), usedToken(1),
          seen(graph.size(), 0), seenToken(1),
          colorPtr(nullptr) {}

    void attachColor(vector<int>* colorVec) {
        colorPtr = colorVec;
    }

    bool colorVertex(int v, int depthLimit) {
        if (!colorPtr) return false;
        if ((*colorPtr)[v] != -1) return true;
        ++seenToken;
        if (seenToken >= INT_MAX / 4) {
            fill(seen.begin(), seen.end(), 0);
            seenToken = 1;
        }
        return recolorDFS(v, depthLimit, -1, seenToken);
    }

private:
    const vector<vector<int>>& adj;
    int k;
    vector<int> used;
    int usedToken;
    vector<int> seen;
    int seenToken;
    vector<int>* colorPtr;

    int nextToken() {
        ++usedToken;
        if (usedToken >= INT_MAX / 4) {
            fill(used.begin(), used.end(), -1);
            usedToken = 1;
        }
        return usedToken;
    }

    bool recolorDFS(int v, int depthLimit, int forbidColor, int attemptId) {
        auto& color = *colorPtr;
        if (seen[v] == attemptId) return false;
        seen[v] = attemptId;
        int token = nextToken();
        for (int u : adj[v]) {
            int cu = color[u];
            if (cu != -1) used[cu] = token;
        }
        for (int c = 0; c < k; ++c) {
            if (c == forbidColor) continue;
            if (used[c] != token) {
                color[v] = c;
                return true;
            }
        }
        if (depthLimit == 0) return false;
        for (int u : adj[v]) {
            int cu = color[u];
            if (cu == -1) continue;
            if (seen[u] == attemptId) continue;
            int saved = color[u];
            color[u] = -1;
            if (recolorDFS(u, depthLimit - 1, saved, attemptId)) {
                color[v] = saved;
                return true;
            }
            color[u] = saved;
        }
        return false;
    }
};

static bool greedyWithRecolor(const vector<vector<int>>& adj, int k, const vector<int>& order, vector<int>& color, int depthLimit) {
    if (k <= 0) return adj.empty();
    int n = static_cast<int>(adj.size());
    color.assign(n, -1);
    GreedyRecolor helper(adj, k);
    helper.attachColor(&color);
    for (int v : order) {
        if (color[v] == -1) {
            if (!helper.colorVertex(v, depthLimit)) {
                return false;
            }
        }
    }
    return true;
}

static bool dsaturColor(const vector<vector<int>>& adj, int k, vector<int>& color) {
    if (k <= 0) return adj.empty();
    int n = static_cast<int>(adj.size());
    color.assign(n, -1);
    vector<int> degree(n);
    for (int i = 0; i < n; ++i) degree[i] = static_cast<int>(adj[i].size());
    vector<int> saturation(n, 0);
    vector<vector<uint8_t>> colorSeen(n, vector<uint8_t>(k, 0));
    struct Node {
        int sat;
        int deg;
        int id;
    };
    struct Compare {
        bool operator()(const Node& a, const Node& b) const {
            if (a.sat != b.sat) return a.sat < b.sat;
            if (a.deg != b.deg) return a.deg < b.deg;
            return a.id > b.id;
        }
    };
    priority_queue<Node, vector<Node>, Compare> pq;
    for (int i = 0; i < n; ++i) pq.push(Node{0, degree[i], i});
    vector<int> used(k, -1);
    int token = 1;
    auto nextToken = [&]() {
        ++token;
        if (token >= INT_MAX / 4) {
            fill(used.begin(), used.end(), -1);
            token = 1;
        }
        return token;
    };
    int colored = 0;
    while (colored < n && !pq.empty()) {
        Node cur = pq.top(); pq.pop();
        int v = cur.id;
        if (color[v] != -1) continue;
        if (cur.sat != saturation[v]) continue;
        int mark = nextToken();
        for (int u : adj[v]) {
            int c = color[u];
            if (c != -1) used[c] = mark;
        }
        int chosen = -1;
        for (int c = 0; c < k; ++c) {
            if (used[c] != mark) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) return false;
        color[v] = chosen;
        ++colored;
        for (int u : adj[v]) {
            if (color[u] != -1) continue;
            if (!colorSeen[u][chosen]) {
                colorSeen[u][chosen] = 1;
                ++saturation[u];
                pq.push(Node{saturation[u], degree[u], u});
            }
        }
    }
    return colored == n;
}

static bool minConflictsColoring(const vector<vector<int>>& adj, int k, vector<int>& color, uint64_t seed, long long maxSteps) {
    if (k <= 0) return adj.empty();
    int n = static_cast<int>(adj.size());
    if (n == 0) {
        color.clear();
        return true;
    }
    mt19937_64 rng(seed);
    if (maxSteps <= 0) maxSteps = 1;
    color.assign(n, 0);
    uniform_int_distribution<int> dist(0, k - 1);
    for (int i = 0; i < n; ++i) color[i] = dist(rng);
    vector<int> conflictCount(n, 0);
    vector<int> pos(n, -1);
    vector<int> conflicts;
    conflicts.reserve(n / 2 + 1);
    auto addConflict = [&](int v) {
        if (pos[v] != -1) return;
        pos[v] = static_cast<int>(conflicts.size());
        conflicts.push_back(v);
    };
    auto removeConflict = [&](int v) {
        if (pos[v] == -1) return;
        int idx = pos[v];
        int last = conflicts.back();
        conflicts[idx] = last;
        pos[last] = idx;
        conflicts.pop_back();
        pos[v] = -1;
    };
    for (int v = 0; v < n; ++v) {
        int cnt = 0;
        for (int u : adj[v]) {
            if (color[u] == color[v]) ++cnt;
        }
        conflictCount[v] = cnt;
        if (cnt > 0) addConflict(v);
    }
    if (conflicts.empty()) return true;
    vector<int> freq(k, 0);
    vector<int> freqStamp(k, 0);
    int stamp = 1;
    auto nextStamp = [&]() {
        ++stamp;
        if (stamp >= INT_MAX / 4) {
            fill(freqStamp.begin(), freqStamp.end(), 0);
            stamp = 1;
        }
        return stamp;
    };
    for (long long step = 0; step < maxSteps; ++step) {
        if (conflicts.empty()) return true;
        size_t idx = rng() % conflicts.size();
        int v = conflicts[idx];
        int stampVal = nextStamp();
        for (int u : adj[v]) {
            int cu = color[u];
            if (freqStamp[cu] != stampVal) {
                freqStamp[cu] = stampVal;
                freq[cu] = 0;
            }
            ++freq[cu];
        }
        int bestColor = color[v];
        int bestScore = INT_MAX;
        for (int c = 0; c < k; ++c) {
            int occ = (freqStamp[c] == stampVal) ? freq[c] : 0;
            if (occ < bestScore) {
                bestScore = occ;
                bestColor = c;
            } else if (occ == bestScore) {
                if ((rng() & 1ULL) == 0ULL) bestColor = c;
            }
        }
        if (bestColor == color[v]) {
            if (bestScore == 0) {
                conflictCount[v] = 0;
                removeConflict(v);
                continue;
            } else if (k > 1) {
                bestColor = (color[v] + 1 + static_cast<int>(rng() % (k - 1))) % k;
            } else {
                continue;
            }
        }
        int oldColor = color[v];
        if (oldColor == bestColor) continue;
        int newConflicts = (freqStamp[bestColor] == stampVal) ? freq[bestColor] : 0;
        for (int u : adj[v]) {
            if (color[u] == oldColor) {
                --conflictCount[u];
                if (conflictCount[u] <= 0) {
                    conflictCount[u] = 0;
                    removeConflict(u);
                }
            }
        }
        color[v] = bestColor;
        for (int u : adj[v]) {
            if (color[u] == bestColor) {
                ++conflictCount[u];
                addConflict(u);
            }
        }
        conflictCount[v] = newConflicts;
        if (newConflicts > 0) addConflict(v);
        else removeConflict(v);
    }
    return conflicts.empty();
}

static bool validateColoring(const vector<vector<int>>& adj, int k, const vector<int>& color) {
    int n = static_cast<int>(adj.size());
    if (static_cast<int>(color.size()) != n) return false;
    if (k <= 0) return n == 0;
    for (int v = 0; v < n; ++v) {
        int cv = color[v];
        if (cv < 0 || cv >= k) return false;
        for (int u : adj[v]) {
            if (color[u] == cv) return false;
        }
    }
    return true;
}

static long long computeMaxSteps(long long n, long long m) {
    long long steps = 5LL * m + 200LL * n + 10000LL;
    if (steps < 10000LL) steps = 10000LL;
    long long limit = 60'000'000LL;
    if (steps > limit) steps = limit;
    return steps;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    FastScanner fs;
    int nInput, kInput;
    long long mInput;
    if (!fs.nextInt(nInput)) return 0;
    if (!fs.nextInt(mInput)) return 0;
    if (!fs.nextInt(kInput)) return 0;

    int n = max(0, nInput);
    long long m_ll = max(0LL, mInput);
    int k = kInput;

    size_t m = static_cast<size_t>(m_ll);
    vector<int> degree(n, 0);
    vector<int> edge_u(m), edge_v(m);
    bool hasSelfLoop = false;

    for (size_t i = 0; i < m; ++i) {
        int u = 0, v = 0;
        if (!fs.nextInt(u)) u = 0;
        if (!fs.nextInt(v)) v = 0;
        edge_u[i] = u;
        edge_v[i] = v;
        if (u == v) hasSelfLoop = true;
        if (0 <= u && u < n) degree[u]++;
        if (0 <= v && v < n && v != u) degree[v]++;
    }

    vector<vector<int>> adj(n);
    for (int i = 0; i < n; ++i) adj[i].reserve(degree[i]);
    for (size_t i = 0; i < m; ++i) {
        int u = edge_u[i];
        int v = edge_v[i];
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        adj[u].push_back(v);
        if (u != v) adj[v].push_back(u);
    }

    if (n == 0) {
        cout << '\n';
        return 0;
    }

    if (k <= 0) {
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << 0;
        }
        cout << '\n';
        return 0;
    }

    vector<int> color(n, 0);
    bool success = false;

    vector<int> degreeOrder;
    bool degreeOrderReady = false;
    vector<int> degeneracyOrder;
    bool degeneracyReady = false;

    auto getDegreeOrder = [&]() -> const vector<int>& {
        if (!degreeOrderReady) {
            degreeOrder = buildDegreeOrder(degree);
            degreeOrderReady = true;
        }
        return degreeOrder;
    };

    auto getDegeneracyOrder = [&]() -> const vector<int>& {
        if (!degeneracyReady) {
            degeneracyOrder = computeDegeneracyOrder(adj);
            degeneracyReady = true;
        }
        return degeneracyOrder;
    };

    bool useDSATUR = n <= 8000 && (long long)n * (long long)k <= 6'000'000LL;

    if (!hasSelfLoop) {
        if (useDSATUR) {
            success = dsaturColor(adj, k, color);
        }
        if (!success) {
            const auto& order = getDegreeOrder();
            success = simpleGreedy(adj, k, order, color);
        }
        if (!success) {
            const auto& order = getDegeneracyOrder();
            success = simpleGreedy(adj, k, order, color);
        }
        if (!success) {
            const auto& order = getDegeneracyOrder();
            int depthLimit = chooseDepthLimit(k);
            success = greedyWithRecolor(adj, k, order, color, depthLimit);
        }
        if (!success) {
            const auto& baseOrder = getDegeneracyOrder();
            vector<int> randomOrder = baseOrder;
            mt19937_64 rng(chrono::steady_clock::now().time_since_epoch().count() ^ 0x123456789ULL);
            int attempts = (n > 200000 ? 2 : 5);
            int depthLimit = chooseDepthLimit(k);
            for (int attempt = 0; attempt < attempts && !success; ++attempt) {
                shuffle(randomOrder.begin(), randomOrder.end(), rng);
                success = greedyWithRecolor(adj, k, randomOrder, color, depthLimit);
            }
        }
        bool minConflictsTried = false;
        if (!success) {
            long long maxSteps = computeMaxSteps(n, m_ll);
            uint64_t seed = chrono::steady_clock::now().time_since_epoch().count() ^ 0x9e3779b97f4a7c15ULL;
            success = minConflictsColoring(adj, k, color, seed, maxSteps);
            minConflictsTried = true;
        }
        if (success) {
            if (!validateColoring(adj, k, color)) {
                success = false;
            }
        }
        if (!success && !hasSelfLoop) {
            long long maxSteps = computeMaxSteps(n, m_ll);
            uint64_t seed = chrono::steady_clock::now().time_since_epoch().count() ^ 0xBADC0FFEEULL;
            if (!minConflictsColoring(adj, k, color, seed, maxSteps))
                success = false;
            else
                success = validateColoring(adj, k, color);
        }
    }

    if (!success) {
        int safeK = max(1, k);
        color.resize(n);
        for (int i = 0; i < n; ++i) color[i] = i % safeK;
    }

    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        int c = color[i];
        if (c < 0) c = 0;
        if (c >= k) c %= k;
        cout << c;
    }
    cout << '\n';
    return 0;
}
