```glsl
#version 450

layout(local_size_x = 256) in;

// inData : [pos.xyz, 0], [vel.xyz, 0]
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = number of boids (N)
    vec4 radii;      // x = dt, y = separation radius, z = alignment radius, w = cohesion radius
    vec4 weights;    // x = separation weight, y = alignment weight, z = cohesion weight, w = max speed
    vec4 bounds;     // x = bound size, y = bound force (z,w unused)
};

void main()
{
    uint idx = gl_GlobalInvocationID.x;
    uint boidCount = counts.x;
    if (idx >= boidCount)
        return;

    uint baseIdx = idx * 2u;
    vec3 position = inData[baseIdx].xyz;
    vec3 velocity = inData[baseIdx + 1u].xyz;

    vec3 sepAccum = vec3(0.0);
    vec3 alignAccum = vec3(0.0);
    vec3 cohAccum = vec3(0.0);
    float sepCount = 0.0;
    float alignCount = 0.0;
    float cohCount = 0.0;

    float dt          = radii.x;
    float sepRadius   = radii.y;
    float alignRadius = radii.z;
    float cohRadius   = radii.w;

    float sepWeight   = weights.x;
    float alignWeight = weights.y;
    float cohWeight   = weights.z;
    float maxSpeed    = weights.w;

    float boundSize   = bounds.x;
    float boundForce  = bounds.y;
    float halfSize    = boundSize * 0.5;
    float triggerSize = boundSize * 0.4;
    float denomBound  = max(halfSize - triggerSize, 1e-5);

    // Accumulate neighbor influences
    for (uint j = 0u; j < boidCount; ++j)
    {
        if (j == idx) continue;

        vec3 otherPos = inData[j * 2u].xyz;
        vec3 toOther = otherPos - position;
        float dist = length(toOther);
        if (dist <= 0.0) continue;

        // Separation
        if (dist < sepRadius) {
            sepAccum -= toOther / dist;  // push away
            sepCount += 1.0;
        }

        // Alignment
        if (dist < alignRadius) {
            vec3 otherVel = inData[j * 2u + 1u].xyz;
            alignAccum += otherVel;
            alignCount += 1.0;
        }

        // Cohesion
        if (dist < cohRadius) {
            cohAccum += otherPos;
            cohCount += 1.0;
        }
    }

    vec3 acceleration = vec3(0.0);

    if (sepCount > 0.0) {
        vec3 sepDir = sepAccum / sepCount;
        acceleration += sepDir * sepWeight;
    }

    if (alignCount > 0.0) {
        vec3 avgVel = alignAccum / alignCount;
        vec3 alignForce = avgVel - velocity;
        acceleration += alignForce * alignWeight;
    }

    if (cohCount > 0.0) {
        vec3 avgPos = cohAccum / cohCount;
        vec3 cohesionForce = avgPos - position;
        acceleration += cohesionForce * cohWeight;
    }

    // Boundary force per axis
    for (int axis = 0; axis < 3; ++axis) {
        float coord = position[axis];
        float absCoord = abs(coord);
        if (absCoord > triggerSize) {
            float normalized = clamp((absCoord - triggerSize) / denomBound, 0.0, 1.0);
            acceleration[axis] += -sign(coord) * boundForce * normalized;
        }
    }

    // Integrate velocity and clamp speed
    velocity += acceleration * dt;
    float speed = length(velocity);
    if (speed > maxSpeed) {
        velocity = (velocity / speed) * maxSpeed;
    }

    // Integrate position
    position += velocity * dt;

    outData[baseIdx]     = vec4(position, 0.0);
    outData[baseIdx + 1] = vec4(velocity, 0.0);
}
```
