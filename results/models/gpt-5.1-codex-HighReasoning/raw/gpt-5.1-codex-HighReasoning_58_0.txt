const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  console.log(JSON.stringify({ assignments: [] }));
  return;
}

const data = JSON.parse(input);
const grid = Array.isArray(data.grid) ? data.grid : [];
const robots = Array.isArray(data.robots) ? data.robots : [];
const jobs = Array.isArray(data.jobs) ? data.jobs : [];

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;

const bestDist = Array.from({ length: height }, () => new Array(width).fill(Infinity));
const bestOwner = Array.from({ length: height }, () => new Array(width).fill(-1));

class MinHeap {
  constructor() {
    this.heap = [];
  }
  push(node) {
    const heap = this.heap;
    heap.push(node);
    let idx = heap.length - 1;
    while (idx > 0) {
      const parent = (idx - 1) >> 1;
      if (this._less(heap[idx], heap[parent])) {
        [heap[idx], heap[parent]] = [heap[parent], heap[idx]];
        idx = parent;
      } else break;
    }
  }
  pop() {
    const heap = this.heap;
    if (!heap.length) return null;
    const top = heap[0];
    const end = heap.pop();
    if (heap.length) {
      heap[0] = end;
      let idx = 0;
      while (true) {
        const left = idx * 2 + 1;
        const right = left + 1;
        let smallest = idx;
        if (left < heap.length && this._less(heap[left], heap[smallest])) smallest = left;
        if (right < heap.length && this._less(heap[right], heap[smallest])) smallest = right;
        if (smallest === idx) break;
        [heap[idx], heap[smallest]] = [heap[smallest], heap[idx]];
        idx = smallest;
      }
    }
    return top;
  }
  size() {
    return this.heap.length;
  }
  _less(a, b) {
    if (a.dist !== b.dist) return a.dist < b.dist;
    return a.id < b.id;
  }
}

if (robots.length && width && height) {
  const heap = new MinHeap();
  for (const robot of robots) {
    if (!robot || typeof robot.id !== 'number') continue;
    const rid = robot.id | 0;
    const x = robot.x | 0;
    const y = robot.y | 0;
    if (y < 0 || y >= height || x < 0 || x >= width) continue;
    const currentDist = bestDist[y][x];
    const currentOwner = bestOwner[y][x];
    if (
      currentDist === Infinity ||
      currentDist > 0 ||
      (currentDist === 0 && (currentOwner === -1 || rid < currentOwner))
    ) {
      bestDist[y][x] = 0;
      bestOwner[y][x] = rid;
      heap.push({ dist: 0, id: rid, x, y });
    }
  }

  const dirs = [
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];

  while (heap.size()) {
    const node = heap.pop();
    if (!node) break;
    const { dist, id, x, y } = node;
    if (dist > bestDist[y][x]) continue;
    if (dist === bestDist[y][x] && id !== bestOwner[y][x]) continue;

    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (ny < 0 || ny >= height || nx < 0 || nx >= width) continue;
      if (grid[ny][nx] === '#') continue;
      const nd = dist + 1;
      const currDist = bestDist[ny][nx];
      const currOwner = bestOwner[ny][nx];
      if (
        nd < currDist ||
        (nd === currDist && (currOwner === -1 || id < currOwner))
      ) {
        bestDist[ny][nx] = nd;
        bestOwner[ny][nx] = id;
        heap.push({ dist: nd, id, x: nx, y: ny });
      }
    }
  }
}

const maxRobotId = robots.reduce(
  (max, r) => (typeof r.id === 'number' ? Math.max(max, r.id | 0) : max),
  -1
);
const assignments =
  maxRobotId >= 0
    ? Array.from({ length: maxRobotId + 1 }, () => [])
    : [[]];

const minRobotId =
  robots.length > 0
    ? robots.reduce(
        (min, r) =>
          typeof r.id === 'number' ? Math.min(min, r.id | 0) : min,
        robots[0].id | 0
      )
    : 0;

const sortedJobs = [...jobs];
sortedJobs.sort((a, b) => {
  const pa = typeof a.priority === 'number' ? a.priority : 0;
  const pb = typeof b.priority === 'number' ? b.priority : 0;
  if (pb !== pa) return pb - pa;
  return (a.id | 0) - (b.id | 0);
});

for (const job of sortedJobs) {
  if (!job || typeof job.id !== 'number') continue;
  const jid = job.id | 0;
  const x = job.x | 0;
  const y = job.y | 0;
  let owner = -1;
  if (y >= 0 && y < height && x >= 0 && x < width) {
    owner = bestOwner[y][x];
  }
  if (owner === -1) owner = minRobotId;
  if (!assignments[owner]) {
    const targetIndex = owner >= 0 ? owner : 0;
    while (assignments.length <= targetIndex) assignments.push([]);
  }
  assignments[owner].push(jid);
}

console.log(JSON.stringify({ assignments }));
