using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;

class Program
{
    static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        if (n < 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }
        int m = scanner.NextInt();

        var adj = new List<int>[n];
        var radj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            radj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if (u < 0 || v < 0 || u >= n || v >= n)
                continue;
            adj[u].Add(v);
            radj[v].Add(u);
        }

        bool[] removed = new bool[n];
        int[] indeg = new int[n];
        int[] outdeg = new int[n];
        for (int u = 0; u < n; u++)
        {
            foreach (var v in adj[u])
            {
                outdeg[u]++;
                indeg[v]++;
            }
        }

        Queue<int> queue = new Queue<int>();
        for (int i = 0; i < n; i++)
        {
            if (indeg[i] == 0 || outdeg[i] == 0)
                queue.Enqueue(i);
        }

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (removed[v]) continue;
            removed[v] = true;
            foreach (var to in adj[v])
            {
                indeg[to]--;
                if (!removed[to] && indeg[to] == 0)
                    queue.Enqueue(to);
            }
            foreach (var from in radj[v])
            {
                outdeg[from]--;
                if (!removed[from] && outdeg[from] == 0)
                    queue.Enqueue(from);
            }
        }

        bool[] inCycle = new bool[n];
        for (int i = 0; i < n; i++)
            inCycle[i] = !removed[i];

        var order = new List<int>(n);
        bool[] visited = new bool[n];
        for (int i = 0; i < n; i++)
        {
            if (!inCycle[i] || visited[i]) continue;
            DFSOrder(i, adj, inCycle, visited, order);
        }

        bool[] assigned = new bool[n];
        var components = new List<List<int>>();
        for (int i = order.Count - 1; i >= 0; i--)
        {
            int v = order[i];
            if (!inCycle[v] || assigned[v]) continue;
            var stack = new Stack<int>();
            var comp = new List<int>();
            stack.Push(v);
            assigned[v] = true;
            while (stack.Count > 0)
            {
                int node = stack.Pop();
                comp.Add(node);
                foreach (var to in radj[node])
                {
                    if (inCycle[to] && !assigned[to])
                    {
                        assigned[to] = true;
                        stack.Push(to);
                    }
                }
            }

            if (comp.Count == 1)
            {
                int node = comp[0];
                bool hasSelf = false;
                foreach (var to in adj[node])
                {
                    if (to == node)
                    {
                        hasSelf = true;
                        break;
                    }
                }
                if (!hasSelf)
                    continue;
            }
            components.Add(comp);
        }

        var finalSolution = new List<int>();
        foreach (var comp in components)
        {
            var solver = BuildSolver(comp, adj, radj);
            var part = solver.SolveAll();
            finalSolution.AddRange(part);
        }

        finalSolution.Sort();
        Console.WriteLine(finalSolution.Count);
        if (finalSolution.Count > 0)
            Console.WriteLine(string.Join(" ", finalSolution));
        else
            Console.WriteLine();
    }

    static void DFSOrder(int start, List<int>[] adj, bool[] valid, bool[] visited, List<int> order)
    {
        var stack = new Stack<(int node, int idx)>();
        stack.Push((start, 0));
        while (stack.Count > 0)
        {
            var (node, idx) = stack.Pop();
            if (idx == 0)
            {
                if (visited[node]) continue;
                visited[node] = true;
            }

            var neighbors = adj[node];
            if (idx < neighbors.Count)
            {
                stack.Push((node, idx + 1));
                int to = neighbors[idx];
                if (valid[to] && !visited[to])
                    stack.Push((to, 0));
            }
            else
            {
                order.Add(node);
            }
        }
    }

    static ComponentSolver BuildSolver(List<int> component, List<int>[] adj, List<int>[] radj)
    {
        int size = component.Count;
        var nodes = component.ToArray();
        var map = new Dictionary<int, int>(size);
        for (int i = 0; i < size; i++)
            map[nodes[i]] = i;

        var localAdj = new List<int>[size];
        var localRadj = new List<int>[size];
        for (int i = 0; i < size; i++)
        {
            localAdj[i] = new List<int>();
            foreach (var to in adj[nodes[i]])
            {
                if (map.TryGetValue(to, out int idx))
                    localAdj[i].Add(idx);
            }
            localRadj[i] = new List<int>();
            foreach (var from in radj[nodes[i]])
            {
                if (map.TryGetValue(from, out int idx))
                    localRadj[i].Add(idx);
            }
        }

        return new ComponentSolver(nodes, localAdj, localRadj);
    }
}

class ComponentSolver
{
    private readonly int size;
    private readonly int[] globalIds;
    private readonly List<int>[] adj;
    private readonly List<int>[] radj;
    private readonly Dictionary<BigInteger, int[]> memo;
    private readonly BitSet fullState;

    public ComponentSolver(int[] globalIds, List<int>[] adj, List<int>[] radj)
    {
        this.size = globalIds.Length;
        this.globalIds = globalIds;
        this.adj = adj;
        this.radj = radj;
        this.memo = new Dictionary<BigInteger, int[]>();
        fullState = new BitSet(size);
        for (int i = 0; i < size; i++)
            fullState.Set(i);
    }

    public int[] SolveAll()
    {
        return Solve(fullState);
    }

    private int[] Solve(BitSet state)
    {
        if (state.IsEmpty)
            return Array.Empty<int>();

        BigInteger key = state.ToBigInteger();
        if (memo.TryGetValue(key, out var cached))
            return cached;

        bool changed;
        var reduced = Reduce(state, out changed);
        if (reduced.IsEmpty)
        {
            memo[key] = Array.Empty<int>();
            return memo[key];
        }
        if (changed)
        {
            var res = Solve(reduced);
            memo[key] = res;
            return res;
        }

        var sccs = GetSccs(state);
        if (sccs.Count > 1)
        {
            var combined = new List<int>();
            foreach (var comp in sccs)
            {
                var part = Solve(comp);
                combined.AddRange(part);
            }
            var arr = combined.ToArray();
            memo[key] = arr;
            return arr;
        }

        var result = SolveStrong(state);
        memo[key] = result;
        return result;
    }

    private BitSet Reduce(BitSet state, out bool changed)
    {
        int[] indeg = new int[size];
        int[] outdeg = new int[size];
        foreach (var v in state.GetSetBits())
        {
            foreach (var to in adj[v])
            {
                if (state.Get(to))
                {
                    indeg[to]++;
                    outdeg[v]++;
                }
            }
        }

        Queue<int> queue = new Queue<int>();
        foreach (var v in state.GetSetBits())
        {
            if (indeg[v] == 0 || outdeg[v] == 0)
                queue.Enqueue(v);
        }

        if (queue.Count == 0)
        {
            changed = false;
            return state;
        }

        var result = state.Clone();
        changed = true;

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (!result.Get(v)) continue;
            result.Clear(v);
            foreach (var to in adj[v])
            {
                if (result.Get(to))
                {
                    indeg[to]--;
                    if (indeg[to] == 0)
                        queue.Enqueue(to);
                }
            }
            foreach (var from in radj[v])
            {
                if (result.Get(from))
                {
                    outdeg[from]--;
                    if (outdeg[from] == 0)
                        queue.Enqueue(from);
                }
            }
        }

        return result;
    }

    private List<BitSet> GetSccs(BitSet state)
    {
        var comps = new List<BitSet>();
        bool[] visited = new bool[size];
        var order = new List<int>();
        var stack = new Stack<(int node, int idx)>();

        foreach (var start in state.GetSetBits())
        {
            if (visited[start]) continue;
            stack.Push((start, 0));
            while (stack.Count > 0)
            {
                var (node, idx) = stack.Pop();
                if (idx == 0)
                {
                    if (visited[node]) continue;
                    visited[node] = true;
                }
                var neighbors = adj[node];
                if (idx < neighbors.Count)
                {
                    stack.Push((node, idx + 1));
                    int to = neighbors[idx];
                    if (state.Get(to) && !visited[to])
                        stack.Push((to, 0));
                }
                else
                {
                    order.Add(node);
                }
            }
        }

        bool[] assigned = new bool[size];
        for (int i = order.Count - 1; i >= 0; i--)
        {
            int node = order[i];
            if (!state.Get(node) || assigned[node]) continue;
            var compBits = new BitSet(size);
            var st = new Stack<int>();
            st.Push(node);
            assigned[node] = true;
            compBits.Set(node);
            while (st.Count > 0)
            {
                int v = st.Pop();
                foreach (var to in radj[v])
                {
                    if (!state.Get(to) || assigned[to]) continue;
                    assigned[to] = true;
                    compBits.Set(to);
                    st.Push(to);
                }
            }
            comps.Add(compBits);
        }

        return comps;
    }

    private int[] SolveStrong(BitSet state)
    {
        var cycle = FindCycle(state);
        if (cycle.Count == 0)
            return Array.Empty<int>();

        var greedy = Greedy(state);
        int bestSize = greedy.Length;
        int[] bestSet = greedy;

        cycle.Sort((a, b) =>
        {
            int db = DegreeWithin(b, state);
            int da = DegreeWithin(a, state);
            int diff = db.CompareTo(da);
            if (diff != 0) return diff;
            return globalIds[a].CompareTo(globalIds[b]);
        });

        foreach (var vertex in cycle)
        {
            var nextState = state.Clone();
            nextState.Clear(vertex);
            var rest = Solve(nextState);
            int candidateSize = 1 + rest.Length;
            if (candidateSize < bestSize)
            {
                bestSize = candidateSize;
                var candidate = new int[candidateSize];
                candidate[0] = globalIds[vertex];
                Array.Copy(rest, 0, candidate, 1, rest.Length);
                bestSet = candidate;
            }
        }

        return bestSet;
    }

    private int[] Greedy(BitSet state)
    {
        var current = state.Clone();
        var removed = new List<int>();
        while (true)
        {
            bool changed;
            var reduced = Reduce(current, out changed);
            if (reduced.IsEmpty)
                break;
            if (changed)
                current = reduced;

            int bestVertex = -1;
            int bestDeg = -1;
            foreach (var v in current.GetSetBits())
            {
                int deg = DegreeWithin(v, current);
                if (deg > bestDeg)
                {
                    bestDeg = deg;
                    bestVertex = v;
                }
            }
            if (bestVertex == -1)
                break;

            removed.Add(globalIds[bestVertex]);
            current.Clear(bestVertex);
        }
        return removed.ToArray();
    }

    private int DegreeWithin(int v, BitSet state)
    {
        int deg = 0;
        foreach (var to in adj[v])
            if (state.Get(to))
                deg++;
        foreach (var from in radj[v])
            if (state.Get(from))
                deg++;
        return deg;
    }

    private List<int> FindCycle(BitSet state)
    {
        var visited = new bool[size];
        var onStack = new bool[size];
        var stack = new List<int>();
        List<int> cycle = null;

        foreach (var start in state.GetSetBits())
        {
            if (visited[start]) continue;
            if (Dfs(start))
                break;
        }

        if (cycle == null)
        {
            var fallback = new List<int>();
            foreach (var v in state.GetSetBits())
            {
                fallback.Add(v);
                break;
            }
            return fallback ?? new List<int>();
        }
        return cycle;

        bool Dfs(int v)
        {
            visited[v] = true;
            onStack[v] = true;
            stack.Add(v);

            foreach (var to in adj[v])
            {
                if (!state.Get(to)) continue;
                if (!visited[to])
                {
                    if (Dfs(to)) return true;
                }
                else if (onStack[to])
                {
                    cycle = new List<int>();
                    for (int i = stack.Count - 1; i >= 0; i--)
                    {
                        int node = stack[i];
                        cycle.Add(node);
                        if (node == to) break;
                    }
                    cycle.Reverse();
                    return true;
                }
            }

            stack.RemoveAt(stack.Count - 1);
            onStack[v] = false;
            return false;
        }
    }
}

class BitSet
{
    private readonly ulong[] data;
    public int Length { get; }

    public BitSet(int length)
    {
        Length = length;
        data = new ulong[(length + 63) >> 6];
    }

    private BitSet(int length, ulong[] backing)
    {
        Length = length;
        data = backing;
    }

    public void Set(int index)
    {
        data[index >> 6] |= 1UL << (index & 63);
    }

    public void Clear(int index)
    {
        data[index >> 6] &= ~(1UL << (index & 63));
    }

    public bool Get(int index)
    {
        return ((data[index >> 6] >> (index & 63)) & 1UL) != 0;
    }

    public BitSet Clone()
    {
        return new BitSet(Length, (ulong[])data.Clone());
    }

    public bool IsEmpty
    {
        get
        {
            foreach (var word in data)
                if (word != 0) return false;
            return true;
        }
    }

    public IEnumerable<int> GetSetBits()
    {
        for (int wordIndex = 0; wordIndex < data.Length; wordIndex++)
        {
            ulong word = data[wordIndex];
            while (word != 0)
            {
                int bit = BitOperations.TrailingZeroCount(word);
                int index = (wordIndex << 6) + bit;
                if (index >= Length)
                    break;
                yield return index;
                word &= word - 1;
            }
        }
    }

    public BigInteger ToBigInteger()
    {
        BigInteger result = BigInteger.Zero;
        for (int i = data.Length - 1; i >= 0; i--)
        {
            result <<= 64;
            result |= data[i];
        }
        return result;
    }
}

class FastScanner
{
    private readonly Stream stream = Console.OpenStandardInput();
    private readonly byte[] buffer = new byte[1 << 16];
    private int ptr = 0;
    private int len = 0;

    private byte Read()
    {
        if (ptr >= len)
        {
            len = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (len <= 0) return 0;
        }
        return buffer[ptr++];
    }

    public int NextInt()
    {
        byte c;
        while ((c = Read()) <= ' ')
        {
            if (c == 0) return -1;
        }
        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }
        int value = c - '0';
        while ((c = Read()) > ' ')
        {
            value = value * 10 + (c - '0');
        }
        return value * sign;
    }
}
