#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <utility>

struct Task {
    int machine;
    long long duration;
};

struct ScheduleResult {
    long long makespan = 0;
    std::vector<std::vector<long long>> start_times;
};

struct Candidate {
    long long est;       // earliest feasible start time (may be updated)
    long long secondary; // heuristic tie-breaker
    int job;
    int seq;
};

struct CandidateCompare {
    bool operator()(const Candidate& a, const Candidate& b) const {
        if (a.est != b.est) return a.est > b.est;
        if (a.secondary != b.secondary) return a.secondary > b.secondary;
        if (a.job != b.job) return a.job > b.job;
        return a.seq > b.seq;
    }
};

ScheduleResult runSchedule(
        int mode,
        const std::vector<std::vector<Task>>& jobs,
        const std::vector<std::vector<long long>>& suffix,
        const std::vector<long long>& machine_load,
        long long total_tasks,
        int machine_count) {

    int job_count = static_cast<int>(jobs.size());
    ScheduleResult result;
    result.start_times.assign(job_count, {});
    for (int j = 0; j < job_count; ++j) {
        result.start_times[j].assign(jobs[j].size(), 0);
    }

    std::vector<long long> job_ready(job_count, 0);
    int machine_ready_size = std::max(1, machine_count);
    std::vector<long long> machine_ready(machine_ready_size, 0);
    std::vector<int> next_idx(job_count, 0);

    auto getSecondary = [&](int job, int seq, const Task& task) -> long long {
        long long duration = task.duration;
        long long remaining = suffix[job][seq];
        switch (mode) {
            case 0: // earliest first / FIFO per job
                return job;
            case 1: // prefer long tasks (LPT-like)
                return -duration;
            case 2: // most remaining work first
                return -remaining;
            case 3: default: { // large scale: bias by machine load + task length
                long long load = (task.machine >= 0 && task.machine < (int)machine_load.size())
                                 ? machine_load[task.machine] : 0LL;
                return -(duration * 10 + load);
            }
        }
    };

    std::priority_queue<Candidate, std::vector<Candidate>, CandidateCompare> pq;

    auto enqueue = [&](int job) {
        int seq = next_idx[job];
        if (seq >= static_cast<int>(jobs[job].size())) return;
        const Task& task = jobs[job][seq];
        long long machine_ready_time = machine_ready[task.machine];
        long long est = std::max(job_ready[job], machine_ready_time);
        long long secondary = getSecondary(job, seq, task);
        pq.push(Candidate{est, secondary, job, seq});
    };

    for (int j = 0; j < job_count; ++j) {
        enqueue(j);
    }

    long long scheduled = 0;
    result.makespan = 0;

    while (scheduled < total_tasks) {
        if (pq.empty()) {
            // Should never happen; safeguard by attempting to enqueue remaining jobs.
            for (int j = 0; j < job_count; ++j) enqueue(j);
            if (pq.empty()) break;
        }
        Candidate cur = pq.top();
        pq.pop();

        if (next_idx[cur.job] != cur.seq) continue; // stale entry

        const Task& task = jobs[cur.job][cur.seq];
        long long actual_start = std::max(job_ready[cur.job], machine_ready[task.machine]);
        if (actual_start > cur.est) {
            pq.push(Candidate{actual_start, cur.secondary, cur.job, cur.seq});
            continue;
        }

        result.start_times[cur.job][cur.seq] = actual_start;
        long long finish = actual_start + task.duration;
        job_ready[cur.job] = finish;
        machine_ready[task.machine] = finish;
        ++next_idx[cur.job];
        ++scheduled;
        result.makespan = std::max(result.makespan, finish);

        enqueue(cur.job);
    }

    return result;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int J, M;
    if (!(std::cin >> J >> M)) {
        return 0;
    }

    std::vector<std::vector<Task>> jobs(J);
    long long total_tasks = 0;
    std::vector<long long> machine_load(std::max(M, 0), 0);

    for (int j = 0; j < J; ++j) {
        int T;
        std::cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int machine;
            long long duration;
            std::cin >> machine >> duration;
            jobs[j][t] = Task{machine, duration};
            if (machine >= 0 && machine < M) {
                machine_load[machine] += duration;
            }
            ++total_tasks;
        }
    }

    std::vector<std::vector<long long>> suffix(J);
    for (int j = 0; j < J; ++j) {
        int L = static_cast<int>(jobs[j].size());
        suffix[j].resize(L);
        long long acc = 0;
        for (int i = L - 1; i >= 0; --i) {
            acc += jobs[j][i].duration;
            suffix[j][i] = acc;
        }
    }

    // Adaptive strategy selection based on scale.
    std::vector<int> mode_plan;
    if (total_tasks == 0) {
        mode_plan = {0};
    } else if (total_tasks <= 500) {
        mode_plan = {0, 1, 2}; // try multiple heuristics when small
    } else if (total_tasks <= 5000) {
        mode_plan = {0, 2};    // moderate size: two heuristics
    } else {
        mode_plan = {3};       // large scale: single pass tuned for throughput
    }

    ScheduleResult best;
    long long best_makespan = std::numeric_limits<long long>::max();
    bool has_best = false;

    for (int mode : mode_plan) {
        ScheduleResult current = runSchedule(mode, jobs, suffix, machine_load, total_tasks, M);
        if (!has_best || current.makespan < best_makespan) {
            best_makespan = current.makespan;
            best = std::move(current);
            has_best = true;
        }
    }

    for (int j = 0; j < J; ++j) {
        bool first = true;
        int T = static_cast<int>(jobs[j].size());
        for (int t = 0; t < T; ++t) {
            if (!first) std::cout << ' ';
            std::cout << best.start_times[j][t] << ' ' << jobs[j][t].machine;
            first = false;
        }
        std::cout << '\n';
    }

    return 0;
}
