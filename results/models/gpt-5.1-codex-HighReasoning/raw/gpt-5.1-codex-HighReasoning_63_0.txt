declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data: any = input.trim().length > 0 ? JSON.parse(input) : {};

interface RobotData {
    id: number;
    x: number;
    y: number;
}

interface JobData {
    id: number;
    x: number;
    y: number;
    priority: number;
}

interface JobInfo extends JobData {
    pointIdx: number;
}

interface RobotState {
    id: number;
    pointIdx: number;
    jobIndices: number[];
    totalDistance: number;
    startPointIdx: number;
}

class DistanceOracle {
    private width: number;
    private height: number;
    private walkable: Uint8Array;
    private pointIndicesByCell: (number[] | undefined)[];
    private pointCells: Int32Array;
    private totalPoints: number;
    private distCache: Map<number, number>[];
    private visitedStamp: Int32Array;
    private distToCell: Int32Array;
    private queue: Int32Array;
    private stampCounter: number;
    private computed: Uint8Array;

    constructor(
        width: number,
        height: number,
        walkable: Uint8Array,
        pointIndicesByCell: (number[] | undefined)[],
        pointCells: Int32Array,
    ) {
        this.width = width;
        this.height = height;
        this.walkable = walkable;
        this.pointIndicesByCell = pointIndicesByCell;
        this.pointCells = pointCells;
        this.totalPoints = pointCells.length;
        const size = width * height;
        this.distCache = Array.from({ length: this.totalPoints }, () => new Map<number, number>());
        this.visitedStamp = new Int32Array(size);
        this.distToCell = new Int32Array(size);
        this.queue = new Int32Array(size);
        this.stampCounter = 1;
        this.computed = new Uint8Array(this.totalPoints);
    }

    public getDistance(sourceIdx: number, targetIdx: number): number {
        if (sourceIdx === targetIdx) {
            return 0;
        }
        if (!this.computed[sourceIdx]) {
            this.computeFrom(sourceIdx);
            this.computed[sourceIdx] = 1;
        }
        const cache = this.distCache[sourceIdx];
        const stored = cache.get(targetIdx);
        return stored !== undefined ? stored : Number.POSITIVE_INFINITY;
    }

    private computeFrom(sourceIdx: number): void {
        const totalCells = this.width * this.height;
        if (totalCells === 0 || sourceIdx < 0 || sourceIdx >= this.totalPoints) {
            return;
        }
        const cache = this.distCache[sourceIdx];
        let head = 0;
        let tail = 0;
        const queue = this.queue;
        const dist = this.distToCell;
        const visited = this.visitedStamp;
        let stamp = this.stampCounter++;
        if (this.stampCounter === 0) {
            visited.fill(0);
            this.stampCounter = 1;
            stamp = this.stampCounter++;
        }
        const startCell = this.pointCells[sourceIdx];
        if (startCell < 0 || startCell >= totalCells) {
            return;
        }
        queue[tail++] = startCell;
        visited[startCell] = stamp;
        dist[startCell] = 0;
        let remaining = this.totalPoints - 1;
        const pointIndicesByCell = this.pointIndicesByCell;
        const walkable = this.walkable;
        const width = this.width;
        const height = this.height;

        while (head < tail && remaining > 0) {
            const cell = queue[head++];
            const cellDist = dist[cell];
            const cellPoints = pointIndicesByCell[cell];
            if (cellPoints) {
                for (let i = 0; i < cellPoints.length; i++) {
                    const pointIdx = cellPoints[i];
                    if (pointIdx === sourceIdx) {
                        continue;
                    }
                    if (!cache.has(pointIdx)) {
                        cache.set(pointIdx, cellDist);
                        remaining--;
                    }
                }
                if (remaining <= 0) {
                    break;
                }
            }
            const y = (cell / width) | 0;
            const x = cell - y * width;
            const nextDist = cellDist + 1;

            if (x + 1 < width) {
                const n = cell + 1;
                if (walkable[n] && visited[n] !== stamp) {
                    visited[n] = stamp;
                    dist[n] = nextDist;
                    queue[tail++] = n;
                }
            }
            if (x - 1 >= 0) {
                const n = cell - 1;
                if (walkable[n] && visited[n] !== stamp) {
                    visited[n] = stamp;
                    dist[n] = nextDist;
                    queue[tail++] = n;
                }
            }
            if (y + 1 < height) {
                const n = cell + width;
                if (walkable[n] && visited[n] !== stamp) {
                    visited[n] = stamp;
                    dist[n] = nextDist;
                    queue[tail++] = n;
                }
            }
            if (y - 1 >= 0) {
                const n = cell - width;
                if (walkable[n] && visited[n] !== stamp) {
                    visited[n] = stamp;
                    dist[n] = nextDist;
                    queue[tail++] = n;
                }
            }
        }
    }
}

function main(): void {
    const gridRows: string[] = Array.isArray(data.grid)
        ? data.grid.map((row: any) => (typeof row === 'string' ? row : String(row ?? '')))
        : [];

    const robotRaw: any[] = Array.isArray(data.robots) ? data.robots : [];
    const jobRaw: any[] = Array.isArray(data.jobs) ? data.jobs : [];

    const height = gridRows.length;
    let width = 0;
    for (const row of gridRows) {
        if (row.length > width) {
            width = row.length;
        }
    }

    const robots: RobotData[] = sanitizeRobots(robotRaw, width, height);
    const jobs: JobData[] = sanitizeJobs(jobRaw, width, height);

    if (robots.length === 0) {
        const assignments: number[][] = [];
        process.stdout.write(JSON.stringify({ assignments }));
        return;
    }

    const size = width * height;
    if (size === 0) {
        const assignments = simpleAssignment(robots, jobs);
        process.stdout.write(JSON.stringify({ assignments }));
        return;
    }

    const walkable = new Uint8Array(size);
    for (let y = 0; y < height; y++) {
        const row = gridRows[y] ?? '';
        for (let x = 0; x < width; x++) {
            const ch = row.charAt(x);
            walkable[y * width + x] = ch === '.' ? 1 : 0;
        }
    }

    const totalPoints = robots.length + jobs.length;
    const pointCells = new Int32Array(totalPoints);
    const pointIndicesByCell: (number[] | undefined)[] = new Array(size);

    for (let i = 0; i < robots.length; i++) {
        const robot = robots[i];
        const cell = robot.y * width + robot.x;
        pointCells[i] = cell;
        walkable[cell] = 1;
        const bucket = pointIndicesByCell[cell];
        if (bucket) {
            bucket.push(i);
        } else {
            pointIndicesByCell[cell] = [i];
        }
    }

    const jobInfos: JobInfo[] = new Array(jobs.length);
    for (let j = 0; j < jobs.length; j++) {
        const job = jobs[j];
        const idx = robots.length + j;
        const cell = job.y * width + job.x;
        pointCells[idx] = cell;
        walkable[cell] = 1;
        const bucket = pointIndicesByCell[cell];
        if (bucket) {
            bucket.push(idx);
        } else {
            pointIndicesByCell[cell] = [idx];
        }
        jobInfos[j] = {
            id: job.id,
            x: job.x,
            y: job.y,
            priority: job.priority,
            pointIdx: idx,
        };
    }

    const oracle = new DistanceOracle(width, height, walkable, pointIndicesByCell, pointCells);

    const robotStates: RobotState[] = robots.map((robot, index) => ({
        id: robot.id,
        pointIdx: index,
        jobIndices: [],
        totalDistance: 0,
        startPointIdx: index,
    }));
    const robotOrder = robotStates.slice().sort((a, b) => a.id - b.id);

    const jobOrder: number[] = jobInfos.map((_, idx) => idx);
    jobOrder.sort((a, b) => {
        const jobA = jobInfos[a];
        const jobB = jobInfos[b];
        if (jobB.priority !== jobA.priority) {
            return jobB.priority - jobA.priority;
        }
        if (jobA.id !== jobB.id) {
            return jobA.id - jobB.id;
        }
        return a - b;
    });

    const largeDistance = Math.max(1e6, width * height * 8 + 10);

    for (const jobIdx of jobOrder) {
        const job = jobInfos[jobIdx];
        const jobPointIdx = job.pointIdx;
        let bestRobot: RobotState | null = null;
        let bestDist = Number.POSITIVE_INFINITY;
        let bestLoad = Number.POSITIVE_INFINITY;
        let bestId = Number.POSITIVE_INFINITY;

        for (const robot of robotOrder) {
            const distRaw = oracle.getDistance(robot.pointIdx, jobPointIdx);
            const distVal = Number.isFinite(distRaw) ? distRaw : largeDistance;
            if (
                distVal < bestDist ||
                (distVal === bestDist &&
                    (robot.totalDistance < bestLoad ||
                        (robot.totalDistance === bestLoad && robot.id < bestId)))
            ) {
                bestRobot = robot;
                bestDist = distVal;
                bestLoad = robot.totalDistance;
                bestId = robot.id;
            }
        }

        const target = bestRobot ?? robotOrder[0];
        target.jobIndices.push(jobIdx);
        target.pointIdx = jobPointIdx;
        target.totalDistance += Number.isFinite(bestDist) ? bestDist : largeDistance;
    }

    let maxRobotId = robots[0].id;
    for (const robot of robots) {
        if (robot.id > maxRobotId) {
            maxRobotId = robot.id;
        }
    }
    if (maxRobotId < 0) {
        maxRobotId = 0;
    }

    const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

    for (const state of robotOrder) {
        assignments[state.id] = buildRoute(state, jobInfos, oracle, largeDistance);
    }

    process.stdout.write(JSON.stringify({ assignments }));
}

main();

function sanitizeRobots(raw: any[], width: number, height: number): RobotData[] {
    const result: RobotData[] = [];
    for (const entry of raw) {
        const obj = typeof entry === 'object' && entry !== null ? entry : {};
        const id = parseId((obj as any).id);
        const x = clampCoord((obj as any).x, width);
        const y = clampCoord((obj as any).y, height);
        result.push({ id, x, y });
    }
    return result;
}

function sanitizeJobs(raw: any[], width: number, height: number): JobData[] {
    const result: JobData[] = [];
    for (const entry of raw) {
        const obj = typeof entry === 'object' && entry !== null ? entry : {};
        const id = parseId((obj as any).id);
        const x = clampCoord((obj as any).x, width);
        const y = clampCoord((obj as any).y, height);
        const priority = parsePriority((obj as any).priority);
        result.push({ id, x, y, priority });
    }
    return result;
}

function simpleAssignment(robots: RobotData[], jobs: JobData[]): number[][] {
    let maxRobotId = -1;
    for (const robot of robots) {
        if (robot.id > maxRobotId) {
            maxRobotId = robot.id;
        }
    }
    if (maxRobotId < 0) {
        return [];
    }
    const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);
    if (jobs.length === 0 || robots.length === 0) {
        return assignments;
    }
    const robotIds = robots.map((r) => r.id).sort((a, b) => a - b);
    if (robotIds.length === 0) {
        return assignments;
    }
    const jobOrder = jobs.map((_, idx) => idx);
    jobOrder.sort((a, b) => {
        const jobA = jobs[a];
        const jobB = jobs[b];
        if (jobB.priority !== jobA.priority) {
            return jobB.priority - jobA.priority;
        }
        if (jobA.id !== jobB.id) {
            return jobA.id - jobB.id;
        }
        return a - b;
    });
    let pointer = 0;
    for (const jobIdx of jobOrder) {
        const robotId = robotIds[pointer % robotIds.length];
        assignments[robotId].push(jobs[jobIdx].id);
        pointer++;
    }
    return assignments;
}

function buildRoute(
    state: RobotState,
    jobInfos: JobInfo[],
    oracle: DistanceOracle,
    largeDistance: number,
): number[] {
    if (state.jobIndices.length === 0) {
        return [];
    }
    const remaining = state.jobIndices.slice();
    const orderedJobIds: number[] = [];
    let currentPoint = state.startPointIdx;

    while (remaining.length > 0) {
        let bestIdx = 0;
        let bestDist = Number.POSITIVE_INFINITY;
        let bestPriority = -Number.POSITIVE_INFINITY;
        let bestJobId = Number.POSITIVE_INFINITY;
        let bestOriginal = Number.POSITIVE_INFINITY;

        for (let i = 0; i < remaining.length; i++) {
            const jobIdx = remaining[i];
            const job = jobInfos[jobIdx];
            const distRaw = oracle.getDistance(currentPoint, job.pointIdx);
            const distVal = Number.isFinite(distRaw) ? distRaw : largeDistance;
            if (
                distVal < bestDist ||
                (distVal === bestDist &&
                    (job.priority > bestPriority ||
                        (job.priority === bestPriority &&
                            (job.id < bestJobId ||
                                (job.id === bestJobId && jobIdx < bestOriginal)))))
            ) {
                bestDist = distVal;
                bestPriority = job.priority;
                bestJobId = job.id;
                bestOriginal = jobIdx;
                bestIdx = i;
            }
        }

        const chosen = remaining.splice(bestIdx, 1)[0];
        orderedJobIds.push(jobInfos[chosen].id);
        currentPoint = jobInfos[chosen].pointIdx;
    }

    return orderedJobIds;
}

function toFiniteNumber(value: any, fallback: number): number {
    if (typeof value === 'number' && Number.isFinite(value)) {
        return value;
    }
    const parsed = Number(value);
    return Number.isFinite(parsed) ? parsed : fallback;
}

function parseId(value: any): number {
    let num = Math.trunc(toFiniteNumber(value, 0));
    if (num < 0) {
        num = 0;
    }
    return num;
}

function clampCoord(value: any, limit: number): number {
    if (limit <= 0) {
        return 0;
    }
    let num = Math.trunc(toFiniteNumber(value, 0));
    if (num < 0) {
        num = 0;
    }
    if (num >= limit) {
        num = limit - 1;
    }
    return num;
}

function parsePriority(value: any): number {
    return toFiniteNumber(value, 0);
}
