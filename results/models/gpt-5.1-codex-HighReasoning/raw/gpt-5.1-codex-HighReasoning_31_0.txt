use std::collections::{HashMap, HashSet};
use std::fmt::Write;
use std::io::{self, Read};
use std::time::{SystemTime, UNIX_EPOCH};

#[derive(Clone, Copy)]
struct Number {
    value: i64,
    index: usize,
}

struct Reader {
    buf: Vec<u8>,
    pos: usize,
}

impl Reader {
    fn new() -> Self {
        let mut input = Vec::new();
        let _ = io::stdin().read_to_end(&mut input);
        Self { buf: input, pos: 0 }
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.pos < self.buf.len()
            && (self.buf[self.pos] == b' ' || self.buf[self.pos] == b'\n' || self.buf[self.pos] == b'\t' || self.buf[self.pos] == b'\r')
        {
            self.pos += 1;
        }
        if self.pos >= self.buf.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.buf[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        } else if self.buf[self.pos] == b'+' {
            self.pos += 1;
        }
        let mut num = 0i64;
        while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_digit() {
            num = num * 10 + (self.buf[self.pos] - b'0') as i64;
            self.pos += 1;
        }
        Some(sign * num)
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_i64().map(|v| v as usize)
    }
}

struct SubsetSumSolver {
    numbers: Vec<Number>,
    target: i64,
}

impl SubsetSumSolver {
    fn new(numbers: Vec<Number>, target: i64) -> Self {
        Self { numbers, target }
    }

    fn solve(&self) -> Option<Vec<usize>> {
        if self.target == 0 {
            return Some(Vec::new());
        }
        if self.numbers.is_empty() {
            return None;
        }
        for num in &self.numbers {
            if num.value == self.target {
                return Some(vec![num.index]);
            }
        }

        const MITM_LIMIT: usize = 40;
        if self.numbers.len() <= MITM_LIMIT {
            if let Some(ans) = meet_in_middle_slice(&self.numbers, self.target) {
                return Some(ans);
            }
        }

        if let Some(ans) = dense_dp(&self.numbers, self.target) {
            return Some(ans);
        }

        if let Some(ans) = hash_dp(&self.numbers, self.target) {
            return Some(ans);
        }

        if self.numbers.len() > MITM_LIMIT {
            let prefix = self.numbers.len().min(32);
            if prefix >= 2 {
                if let Some(ans) = meet_in_middle_slice(&self.numbers[..prefix], self.target) {
                    return Some(ans);
                }
            }
        }

        self.randomized_search()
    }

    fn randomized_search(&self) -> Option<Vec<usize>> {
        const SAMPLE_LIMIT: usize = 32;
        if self.numbers.len() <= 1 {
            return None;
        }
        let sample_size = self.numbers.len().min(SAMPLE_LIMIT);
        if sample_size == 0 {
            return None;
        }
        let iterations = 512usize;
        let seed = SystemTime::now()
            .duration_since(UNIX_EPOCH)
            .unwrap_or_default()
            .as_nanos() as u64
            ^ (self.numbers.len() as u64)
            ^ (self.target as u64);
        let mut rng = SplitMix64::new(seed);

        for _ in 0..iterations {
            let mut chosen = Vec::with_capacity(sample_size);
            while chosen.len() < sample_size {
                let idx = (rng.next_u64() % (self.numbers.len() as u64)) as usize;
                if !chosen.contains(&idx) {
                    chosen.push(idx);
                }
            }
            let mut subset = Vec::with_capacity(sample_size);
            for &idx in &chosen {
                subset.push(self.numbers[idx]);
            }
            if let Some(ans) = meet_in_middle_slice(&subset, self.target) {
                return Some(ans);
            }
        }
        None
    }
}

fn meet_in_middle_slice(slice: &[Number], target: i64) -> Option<Vec<usize>> {
    let n = slice.len();
    if n == 0 {
        return None;
    }
    let mid = n / 2;
    let left = &slice[..mid];
    let right = &slice[mid..];

    let left_total = 1usize << left.len();
    let mut left_sums = vec![0i64; left_total];
    for mask in 1..left_total {
        let lsb = mask & (!mask + 1);
        let idx = lsb.trailing_zeros() as usize;
        left_sums[mask] = left_sums[mask ^ lsb] + left[idx].value;
    }
    let mut left_map: HashMap<i64, u64> = HashMap::with_capacity(left_total);
    for mask in 0..left_total {
        let sum = left_sums[mask];
        if sum == target {
            let mut result = Vec::new();
            collect_indices(left, mask as u64, &mut result);
            return Some(result);
        }
        left_map.entry(sum).or_insert(mask as u64);
    }

    let right_total = 1usize << right.len();
    let mut right_sums = vec![0i64; right_total];
    for mask in 1..right_total {
        let lsb = mask & (!mask + 1);
        let idx = lsb.trailing_zeros() as usize;
        right_sums[mask] = right_sums[mask ^ lsb] + right[idx].value;
    }

    for mask in 0..right_total {
        let sum = right_sums[mask];
        if sum == target {
            let mut result = Vec::new();
            collect_indices(right, mask as u64, &mut result);
            return Some(result);
        }
        let need = target - sum;
        if let Some(&left_mask) = left_map.get(&need) {
            let mut result = Vec::new();
            collect_indices(left, left_mask, &mut result);
            collect_indices(right, mask as u64, &mut result);
            return Some(result);
        }
    }
    None
}

fn collect_indices(slice: &[Number], mask: u64, out: &mut Vec<usize>) {
    for (i, num) in slice.iter().enumerate() {
        if ((mask >> i) & 1) == 1 {
            out.push(num.index);
        }
    }
}

fn dense_dp(numbers: &[Number], target: i64) -> Option<Vec<usize>> {
    if numbers.is_empty() {
        return None;
    }
    const RANGE_LIMIT: i64 = 20_000_000;
    const COMPLEXITY_LIMIT: i128 = 800_000_000;

    let mut min_sum: i128 = 0;
    let mut max_sum: i128 = 0;
    for num in numbers {
        if num.value >= 0 {
            max_sum += num.value as i128;
        } else {
            min_sum += num.value as i128;
        }
    }
    let range = max_sum - min_sum;
    if range < 0 || range > RANGE_LIMIT as i128 {
        return None;
    }
    if (numbers.len() as i128) * range > COMPLEXITY_LIMIT {
        return None;
    }
    if min_sum < i64::MIN as i128 || max_sum > i64::MAX as i128 {
        return None;
    }

    let min_sum_i64 = min_sum as i64;
    let target_offset = (target as i128) - min_sum;
    if target_offset < 0 || target_offset > range {
        return None;
    }
    let len = (range as usize) + 1;
    let offset = (-min_sum) as usize;
    if offset >= len {
        return None;
    }
    let target_idx = target_offset as usize;

    let mut reachable = vec![false; len];
    let mut parent_idx = vec![-1i32; len];
    let mut parent_pos = vec![-1i32; len];
    reachable[offset] = true;
    parent_pos[offset] = offset as i32;

    for (idx, num) in numbers.iter().enumerate() {
        let val = num.value;
        if val == 0 {
            continue;
        }
        if val > 0 {
            let shift = val as usize;
            if shift >= len {
                continue;
            }
            for pos in (0..len - shift).rev() {
                if reachable[pos] && !reachable[pos + shift] {
                    reachable[pos + shift] = true;
                    parent_idx[pos + shift] = idx as i32;
                    parent_pos[pos + shift] = pos as i32;
                }
            }
        } else {
            let shift = (-val) as usize;
            if shift >= len {
                continue;
            }
            for pos in shift..len {
                if reachable[pos] && !reachable[pos - shift] {
                    reachable[pos - shift] = true;
                    parent_idx[pos - shift] = idx as i32;
                    parent_pos[pos - shift] = pos as i32;
                }
            }
        }
        if reachable[target_idx] {
            break;
        }
    }

    if !reachable[target_idx] {
        return None;
    }
    let mut result = Vec::new();
    let mut pos = target_idx;
    while parent_idx[pos] >= 0 {
        let idx = parent_idx[pos] as usize;
        result.push(numbers[idx].index);
        pos = parent_pos[pos] as usize;
    }
    Some(result)
}

fn hash_dp(numbers: &[Number], target: i64) -> Option<Vec<usize>> {
    const HASH_DP_N_LIMIT: usize = 400;
    const MAX_STATES: usize = 250_000;
    if numbers.len() > HASH_DP_N_LIMIT {
        return None;
    }
    let mut map: HashMap<i64, (i32, i64)> = HashMap::new();
    map.insert(0, (-1, 0));
    let mut temp_set: HashSet<i64> = HashSet::new();

    for (idx, num) in numbers.iter().enumerate() {
        let val = num.value;
        let keys: Vec<i64> = map.keys().copied().collect();
        let mut pending: Vec<(i64, (i32, i64))> = Vec::with_capacity(keys.len());
        temp_set.clear();
        for sum in keys {
            let new_sum = sum.wrapping_add(val);
            if map.contains_key(&new_sum) || temp_set.contains(&new_sum) {
                continue;
            }
            temp_set.insert(new_sum);
            pending.push((new_sum, (idx as i32, sum)));
        }
        for (s, info) in pending {
            map.insert(s, info);
            if s == target {
                return backtrack_hash(&map, target, numbers);
            }
        }
        if map.len() > MAX_STATES {
            prune_states(&mut map, target, MAX_STATES);
        }
        if map.contains_key(&target) {
            return backtrack_hash(&map, target, numbers);
        }
    }
    None
}

fn backtrack_hash(map: &HashMap<i64, (i32, i64)>, target: i64, numbers: &[Number]) -> Option<Vec<usize>> {
    let mut current = target;
    let mut result = Vec::new();
    while current != 0 {
        let &(idx, prev) = map.get(&current)?;
        if idx < 0 {
            break;
        }
        result.push(numbers[idx as usize].index);
        current = prev;
    }
    if current == 0 {
        Some(result)
    } else {
        None
    }
}

fn prune_states(map: &mut HashMap<i64, (i32, i64)>, target: i64, limit: usize) {
    if map.len() <= limit {
        return;
    }
    let zero_info = map.get(&0).copied().unwrap_or((-1, 0));
    let mut entries: Vec<(i64, (i32, i64))> = map.iter().map(|(&sum, &info)| (sum, info)).collect();
    entries.sort_unstable_by(|a, b| {
        let da = a.0.abs_diff(target);
        let db = b.0.abs_diff(target);
        da.cmp(&db)
    });
    if entries.len() > limit {
        entries.truncate(limit);
    }
    if entries.iter().all(|(sum, _)| *sum != 0) {
        if entries.len() == limit {
            entries.pop();
        }
        entries.push((0, zero_info));
    }
    map.clear();
    for (sum, info) in entries {
        map.insert(sum, info);
    }
}

struct SplitMix64 {
    state: u64,
}

impl SplitMix64 {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }

    fn next_u64(&mut self) -> u64 {
        self.state = self.state.wrapping_add(0x9E3779B97F4A7C15);
        let mut z = self.state;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
        z ^ (z >> 31)
    }
}

fn main() {
    let mut reader = Reader::new();
    let n = match reader.next_usize() {
        Some(v) => v,
        None => return,
    };
    let target = match reader.next_i64() {
        Some(v) => v,
        None => return,
    };
    let mut numbers = Vec::with_capacity(n);
    for idx in 0..n {
        if let Some(value) = reader.next_i64() {
            numbers.push(Number { value, index: idx });
        } else {
            break;
        }
    }

    let solver = SubsetSumSolver::new(numbers, target);
    if let Some(mut indices) = solver.solve() {
        println!("YES");
        indices.sort_unstable();
        let mut line = String::new();
        line.push('[');
        for (i, idx) in indices.iter().enumerate() {
            if i > 0 {
                line.push(' ');
            }
            let _ = write!(line, "{idx}");
        }
        line.push(']');
        println!("{line}");
    } else {
        println!("NO");
    }
}
