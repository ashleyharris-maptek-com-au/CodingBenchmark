declare const require: any;

const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : { machineCount: 1, tasks: [] };

type TaskInput = {
  id: string | number;
  duration: number;
  deps?: Array<string | number>;
  deadline?: number;
  weight?: number;
};

interface TaskInfo {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
}

class BinaryHeap<T> {
  private data: T[] = [];
  private cmp: (a: T, b: T) => number;

  constructor(cmp: (a: T, b: T) => number) {
    this.cmp = cmp;
  }

  size(): number {
    return this.data.length;
  }

  push(value: T): void {
    this.data.push(value);
    this.bubbleUp(this.data.length - 1);
  }

  pop(): T | undefined {
    if (this.data.length === 0) return undefined;
    const top = this.data[0];
    const end = this.data.pop()!;
    if (this.data.length > 0) {
      this.data[0] = end;
      this.bubbleDown(0);
    }
    return top;
  }

  private bubbleUp(index: number): void {
    while (index > 0) {
      const parent = Math.floor((index - 1) / 2);
      if (this.cmp(this.data[index], this.data[parent]) < 0) {
        [this.data[index], this.data[parent]] = [this.data[parent], this.data[index]];
        index = parent;
      } else {
        break;
      }
    }
  }

  private bubbleDown(index: number): void {
    const length = this.data.length;
    while (true) {
      const left = index * 2 + 1;
      const right = left + 1;
      let smallest = index;

      if (left < length && this.cmp(this.data[left], this.data[smallest]) < 0) {
        smallest = left;
      }

      if (right < length && this.cmp(this.data[right], this.data[smallest]) < 0) {
        smallest = right;
      }

      if (smallest !== index) {
        [this.data[index], this.data[smallest]] = [this.data[smallest], this.data[index]];
        index = smallest;
      } else {
        break;
      }
    }
  }
}

function normalizeNumber(value: any, fallback: number): number {
  const num = Number(value);
  return Number.isFinite(num) ? num : fallback;
}

const rawTasks: TaskInput[] = Array.isArray(data.tasks) ? data.tasks : [];
const machineCount: number = Math.max(1, Math.floor(normalizeNumber(data.machineCount, 1)));

if (rawTasks.length === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const tasksById = new Map<string, TaskInfo>();
const dependents = new Map<string, string[]>();
const baseOrder = new Map<string, number>();

for (let idx = 0; idx < rawTasks.length; idx++) {
  const raw = rawTasks[idx];
  if (raw == null || typeof raw.id === 'undefined') continue;
  const id = String(raw.id);
  const duration = Math.max(0, normalizeNumber(raw.duration, 0));
  const depsRaw = Array.isArray(raw.deps) ? raw.deps : [];
  const deps: string[] = [];
  for (const dep of depsRaw) {
    if (typeof dep === 'undefined' || dep === null) continue;
    const depId = String(dep);
    if (depId === id) continue;
    deps.push(depId);
  }
  const deadline = Number.isFinite(raw.deadline) ? Number(raw.deadline) : Number.POSITIVE_INFINITY;
  const weight = Number.isFinite(raw.weight) ? Number(raw.weight) : 1;

  tasksById.set(id, { id, duration, deps, deadline, weight });
  baseOrder.set(id, idx);
}

if (tasksById.size === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

tasksById.forEach((task) => {
  dependents.set(task.id, []);
});

tasksById.forEach((task) => {
  const validDeps: string[] = [];
  for (const depId of task.deps) {
    if (!tasksById.has(depId)) continue;
    validDeps.push(depId);
    const list = dependents.get(depId);
    if (list) list.push(task.id);
  }
  task.deps = validDeps;
});

const indegree = new Map<string, number>();
const readyTime = new Map<string, number>();

tasksById.forEach((task, id) => {
  indegree.set(id, task.deps.length);
  readyTime.set(id, 0);
});

const compareAsc = (a: number, b: number): number => {
  if (a === b) return 0;
  return a < b ? -1 : 1;
};

const compareDesc = (a: number, b: number): number => {
  if (a === b) return 0;
  return a > b ? -1 : 1;
};

const readyHeap = new BinaryHeap<string>((a, b) => {
  const taskA = tasksById.get(a)!;
  const taskB = tasksById.get(b)!;
  const readyA = readyTime.get(a) ?? 0;
  const readyB = readyTime.get(b) ?? 0;

  const slackA = Number.isFinite(taskA.deadline)
    ? taskA.deadline - readyA - taskA.duration
    : Number.POSITIVE_INFINITY;
  const slackB = Number.isFinite(taskB.deadline)
    ? taskB.deadline - readyB - taskB.duration
    : Number.POSITIVE_INFINITY;

  const slackCmp = compareAsc(slackA, slackB);
  if (slackCmp !== 0) return slackCmp;

  const ratioA = taskA.weight / (taskA.duration > 0 ? taskA.duration : 1e-9);
  const ratioB = taskB.weight / (taskB.duration > 0 ? taskB.duration : 1e-9);
  const ratioCmp = compareDesc(ratioA, ratioB);
  if (ratioCmp !== 0) return ratioCmp;

  const deadlineCmp = compareAsc(taskA.deadline, taskB.deadline);
  if (deadlineCmp !== 0) return deadlineCmp;

  const readyCmp = compareAsc(readyA, readyB);
  if (readyCmp !== 0) return readyCmp;

  const baseA = baseOrder.get(a) ?? 0;
  const baseB = baseOrder.get(b) ?? 0;
  const baseCmp = compareAsc(baseA, baseB);
  if (baseCmp !== 0) return baseCmp;

  return a < b ? -1 : a > b ? 1 : 0;
});

indegree.forEach((deg, id) => {
  if (deg === 0) {
    readyHeap.push(id);
  }
});

const machineAvailability = new Array<number>(machineCount).fill(0);
const completionTime = new Map<string, number>();
const scheduleRecords: { id: string; start: number; machine: number }[] = [];
const epsilon = 1e-9;
let scheduledCount = 0;

while (readyHeap.size() > 0) {
  const taskId = readyHeap.pop()!;
  const task = tasksById.get(taskId)!;
  const release = readyTime.get(taskId) ?? 0;

  let bestMachine = 0;
  let bestStart = Number.POSITIVE_INFINITY;
  let bestFinish = Number.POSITIVE_INFINITY;

  for (let m = 0; m < machineCount; m++) {
    const start = Math.max(machineAvailability[m], release);
    const finish = start + task.duration;
    let better = false;

    if (start + epsilon < bestStart) {
      better = true;
    } else if (Math.abs(start - bestStart) <= epsilon) {
      if (finish + epsilon < bestFinish) {
        better = true;
      } else if (Math.abs(finish - bestFinish) <= epsilon && m < bestMachine) {
        better = true;
      }
    }

    if (better) {
      bestMachine = m;
      bestStart = start;
      bestFinish = finish;
    }
  }

  machineAvailability[bestMachine] = bestFinish;
  completionTime.set(taskId, bestFinish);
  scheduleRecords.push({ id: taskId, start: bestStart, machine: bestMachine });
  scheduledCount++;

  const outList = dependents.get(taskId);
  if (outList) {
    for (const nextId of outList) {
      const prevReady = readyTime.get(nextId) ?? 0;
      if (bestFinish > prevReady) {
        readyTime.set(nextId, bestFinish);
      }
      const newIn = (indegree.get(nextId) ?? 0) - 1;
      indegree.set(nextId, newIn);
      if (newIn === 0) readyHeap.push(nextId);
    }
  }
}

const totalTasks = tasksById.size;

if (scheduledCount === totalTasks) {
  process.stdout.write(JSON.stringify({ schedule: scheduleRecords }));
  process.exit(0);
}

const buildTopologicalOrder = (): string[] => {
  const localIndegree = new Map<string, number>();
  tasksById.forEach((task, id) => {
    localIndegree.set(id, task.deps.length);
  });

  const orderHeap = new BinaryHeap<string>((a, b) => {
    const baseA = baseOrder.get(a) ?? 0;
    const baseB = baseOrder.get(b) ?? 0;
    if (baseA !== baseB) return baseA - baseB;
    return a < b ? -1 : a > b ? 1 : 0;
  });

  localIndegree.forEach((deg, id) => {
    if (deg === 0) orderHeap.push(id);
  });

  const order: string[] = [];
  while (orderHeap.size() > 0) {
    const id = orderHeap.pop()!;
    order.push(id);
    const outs = dependents.get(id);
    if (outs) {
      for (const next of outs) {
        const newDeg = (localIndegree.get(next) ?? 0) - 1;
        localIndegree.set(next, newDeg);
        if (newDeg === 0) orderHeap.push(next);
      }
    }
  }

  if (order.length < totalTasks) {
    const visited = new Set(order);
    const remaining: string[] = [];
    tasksById.forEach((_, id) => {
      if (!visited.has(id)) remaining.push(id);
    });
    remaining.sort((a, b) => {
      const baseA = baseOrder.get(a) ?? 0;
      const baseB = baseOrder.get(b) ?? 0;
      if (baseA !== baseB) return baseA - baseB;
      return a < b ? -1 : a > b ? 1 : 0;
    });
    order.push(...remaining);
  }

  return order;
};

const fallbackOrder = buildTopologicalOrder();
process.stdout.write(JSON.stringify({ order: fallbackOrder }));
