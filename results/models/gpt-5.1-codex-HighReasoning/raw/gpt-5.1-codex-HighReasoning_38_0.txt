import sys

def parse_input():
    lines = sys.stdin.read().splitlines()
    idx = 0
    n = len(lines)
    while idx < n and not lines[idx].strip():
        idx += 1
    if idx == n:
        return None
    header = lines[idx].split()
    if len(header) < 2:
        return None
    try:
        universe_size = int(header[0])
        set_count = int(header[1])
    except ValueError:
        return None
    idx += 1
    sets = []
    for _ in range(set_count):
        line = ""
        if idx < n:
            line = lines[idx].strip()
            idx += 1
        if not line:
            sets.append([])
            continue
        seen = set()
        subset = []
        for token in line.split():
            try:
                val = int(token)
            except ValueError:
                continue
            if 0 <= val < universe_size and val not in seen:
                seen.add(val)
                subset.append(val)
        sets.append(subset)
    return universe_size, set_count, sets

def exact_cover(universe_size, sets):
    element_to_sets = [[] for _ in range(universe_size)]
    for idx, subset in enumerate(sets):
        for element in subset:
            element_to_sets[element].append(idx)

    covered = [False] * universe_size
    remaining = universe_size
    solution = []

    def is_compatible(set_idx):
        for element in sets[set_idx]:
            if covered[element]:
                return False
        return True

    def backtrack():
        nonlocal remaining
        if remaining == 0:
            return True

        best_options = None
        min_count = None
        for element in range(universe_size):
            if covered[element]:
                continue
            options = []
            for set_idx in element_to_sets[element]:
                if is_compatible(set_idx):
                    options.append(set_idx)
            if not options:
                return False
            if min_count is None or len(options) < min_count:
                min_count = len(options)
                best_options = options
                if min_count == 1:
                    break

        if best_options is None:
            return False

        for set_idx in best_options:
            added = []
            conflict = False
            for element in sets[set_idx]:
                if covered[element]:
                    conflict = True
                    break
                covered[element] = True
                added.append(element)
            if conflict:
                for element in added:
                    covered[element] = False
                continue

            remaining -= len(added)
            solution.append(set_idx)
            if backtrack():
                return True
            solution.pop()
            remaining += len(added)
            for element in added:
                covered[element] = False

        return False

    if backtrack():
        return True, solution[:]
    return False, []

def main():
    sys.setrecursionlimit(10000)
    parsed = parse_input()
    if parsed is None:
        print("NO SOLUTION")
        return
    universe_size, _, sets = parsed
    success, solution = exact_cover(universe_size, sets)
    if success:
        print("SOLUTION")
        print(" ".join(map(str, solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
