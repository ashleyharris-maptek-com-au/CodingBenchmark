```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <limits>

struct Item {
    int id;
    int w;
    int h;
};

struct Node {
    int x;
    int y;
    int w;
    int h;
};

struct Candidate {
    bool valid = false;
    long long areaFit = 0;
    int shortSideFit = 0;
    int longSideFit = 0;
    Node rect{0, 0, 0, 0};
    bool rotated = false;
};

inline bool betterCandidate(const Candidate &a, const Candidate &b) {
    if (!a.valid) return false;
    if (!b.valid) return true;
    if (a.areaFit != b.areaFit) return a.areaFit < b.areaFit;
    if (a.shortSideFit != b.shortSideFit) return a.shortSideFit < b.shortSideFit;
    if (a.longSideFit != b.longSideFit) return a.longSideFit < b.longSideFit;
    if (a.rect.y != b.rect.y) return a.rect.y < b.rect.y;
    if (a.rect.x != b.rect.x) return a.rect.x < b.rect.x;
    return (!a.rotated && b.rotated);
}

struct Placement {
    int x = 0;
    int y = 0;
    bool rotated = false;
};

class Board {
public:
    Board() : width(0), height(0) {}
    Board(int w, int h) { init(w, h); }

    Candidate findPosition(int w, int h) const {
        Candidate best;
        for (const Node &freeRect : freeRects) {
            if (freeRect.w >= w && freeRect.h >= h) {
                evaluatePosition(freeRect, w, h, false, best);
            }
            if (w != h && freeRect.w >= h && freeRect.h >= w) {
                evaluatePosition(freeRect, h, w, true, best);
            }
        }
        return best;
    }

    bool insert(int w, int h, Placement &out) {
        Candidate best = findPosition(w, h);
        if (!best.valid) return false;

        Node placed = best.rect;
        out.x = placed.x;
        out.y = placed.y;
        out.rotated = best.rotated;

        size_t i = 0;
        while (i < freeRects.size()) {
            if (splitFreeNode(freeRects[i], placed)) {
                freeRects[i] = freeRects.back();
                freeRects.pop_back();
            } else {
                ++i;
            }
        }
        pruneFreeList();
        return true;
    }

private:
    int width, height;
    std::vector<Node> freeRects;

    void init(int w, int h) {
        width = w;
        height = h;
        freeRects.clear();
        freeRects.reserve(64);
        freeRects.push_back({0, 0, w, h});
    }

    void evaluatePosition(const Node &freeRect, int rectW, int rectH, bool rotated, Candidate &best) const {
        Candidate cur;
        cur.valid = true;
        cur.rotated = rotated;
        cur.rect = {freeRect.x, freeRect.y, rectW, rectH};
        cur.areaFit = 1LL * freeRect.w * freeRect.h - 1LL * rectW * rectH;
        int leftoverW = freeRect.w - rectW;
        int leftoverH = freeRect.h - rectH;
        cur.shortSideFit = std::min(leftoverW, leftoverH);
        cur.longSideFit = std::max(leftoverW, leftoverH);
        if (betterCandidate(cur, best)) {
            best = cur;
        }
    }

    bool splitFreeNode(const Node &freeRect, const Node &usedNode) {
        if (usedNode.x >= freeRect.x + freeRect.w || usedNode.x + usedNode.w <= freeRect.x ||
            usedNode.y >= freeRect.y + freeRect.h || usedNode.y + usedNode.h <= freeRect.y) {
            return false;
        }

        if (usedNode.x < freeRect.x + freeRect.w && usedNode.x + usedNode.w > freeRect.x) {
            if (usedNode.y > freeRect.y && usedNode.y < freeRect.y + freeRect.h) {
                Node newNode = freeRect;
                newNode.h = usedNode.y - newNode.y;
                if (newNode.w > 0 && newNode.h > 0) freeRects.push_back(newNode);
            }
            if (usedNode.y + usedNode.h < freeRect.y + freeRect.h) {
                Node newNode = freeRect;
                newNode.y = usedNode.y + usedNode.h;
                newNode.h = freeRect.y + freeRect.h - (usedNode.y + usedNode.h);
                if (newNode.w > 0 && newNode.h > 0) freeRects.push_back(newNode);
            }
        }

        if (usedNode.y < freeRect.y + freeRect.h && usedNode.y + usedNode.h > freeRect.y) {
            if (usedNode.x > freeRect.x && usedNode.x < freeRect.x + freeRect.w) {
                Node newNode = freeRect;
                newNode.w = usedNode.x - newNode.x;
                if (newNode.w > 0 && newNode.h > 0) freeRects.push_back(newNode);
            }
            if (usedNode.x + usedNode.w < freeRect.x + freeRect.w) {
                Node newNode = freeRect;
                newNode.x = usedNode.x + usedNode.w;
                newNode.w = freeRect.x + freeRect.w - (usedNode.x + usedNode.w);
                if (newNode.w > 0 && newNode.h > 0) freeRects.push_back(newNode);
            }
        }
        return true;
    }

    static bool contains(const Node &a, const Node &b) {
        return a.x >= b.x && a.y >= b.y &&
               a.x + a.w <= b.x + b.w &&
               a.y + a.h <= b.y + b.h;
    }

    void pruneFreeList() {
        size_t i = 0;
        while (i < freeRects.size()) {
            size_t j = i + 1;
            bool removedI = false;
            while (j < freeRects.size()) {
                if (contains(freeRects[i], freeRects[j])) {
                    freeRects[i] = freeRects.back();
                    freeRects.pop_back();
                    removedI = true;
                    break;
                }
                if (contains(freeRects[j], freeRects[i])) {
                    freeRects[j] = freeRects.back();
                    freeRects.pop_back();
                } else {
                    ++j;
                }
            }
            if (!removedI) {
                ++i;
            }
        }
    }
};

struct OutputPlacement {
    int board = 0;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, boardW, boardH;
    if (!(std::cin >> N >> boardW >> boardH)) {
        return 0;
    }

    std::vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        items[i] = {i, w, h};
    }

    std::vector<int> order(N);
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int lhs, int rhs) {
        const auto &a = items[lhs];
        const auto &b = items[rhs];
        int maxA = std::max(a.w, a.h);
        int maxB = std::max(b.w, b.h);
        if (maxA != maxB) return maxA > maxB;
        long long areaA = 1LL * a.w * a.h;
        long long areaB = 1LL * b.w * b.h;
        if (areaA != areaB) return areaA > areaB;
        int minA = std::min(a.w, a.h);
        int minB = std::min(b.w, b.h);
        if (minA != minB) return minA > minB;
        return lhs < rhs;
    });

    std::vector<Board> boards;
    boards.reserve(std::max(1, N / 8));
    std::vector<OutputPlacement> answer(N);

    for (int idx : order) {
        const Item &rect = items[idx];

        bool fits = ((rect.w <= boardW && rect.h <= boardH) ||
                     (rect.h <= boardW && rect.w <= boardH));
        if (!fits) {
            // Impossible piece, still create placeholder board entry
            boards.emplace_back(boardW, boardH);
            answer[rect.id] = {static_cast<int>(boards.size() - 1), 0, 0, 0};
            continue;
        }

        Candidate best;
        int bestBoard = -1;
        for (int b = 0; b < static_cast<int>(boards.size()); ++b) {
            Candidate cand = boards[b].findPosition(rect.w, rect.h);
            if (!cand.valid) continue;
            if (!best.valid || betterCandidate(cand, best)) {
                best = cand;
                bestBoard = b;
                if (best.areaFit == 0 && best.shortSideFit == 0) break;
            }
        }

        if (bestBoard == -1) {
            boards.emplace_back(boardW, boardH);
            bestBoard = static_cast<int>(boards.size() - 1);
        }

        Placement placed;
        if (!boards[bestBoard].insert(rect.w, rect.h, placed)) {
            boards.emplace_back(boardW, boardH);
            bestBoard = static_cast<int>(boards.size() - 1);
            boards[bestBoard].insert(rect.w, rect.h, placed);
        }

        answer[rect.id] = {bestBoard, placed.x, placed.y, placed.rotated ? 1 : 0};
    }

    std::cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        const auto &res = answer[i];
        std::cout << res.board << ' ' << res.x << ' ' << res.y << ' ' << res.rotated << '\n';
    }
    return 0;
}
```

This solver implements a scalable MaxRects-based heuristic:

- Rectangles are ordered by dominating dimension & area to maximize early packing efficiency.
- Each board maintains a dynamic list of free rectangles, using best-area-fit placement with optional rotation.
- Guillotine splits and dominance pruning keep the free-space list small and efficient, scaling from a handful to tens of thousands of rectangles within the required time.
