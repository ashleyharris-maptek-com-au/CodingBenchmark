using System;
using System.Collections.Generic;
using System.IO;

public class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        if (!scanner.TryReadInt(out int n))
        {
            Console.WriteLine(0);
            return;
        }

        int stockLength = scanner.NextInt();
        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int len = scanner.NextInt();
            cuts[i] = new Cut { Length = len, Index = i };
        }

        Array.Sort(cuts, (a, b) => b.Length.CompareTo(a.Length));

        var bins = new List<List<int>>();
        var tree = new Treap();

        foreach (var cut in cuts)
        {
            if (cut.Length > stockLength)
            {
                int newId = bins.Count;
                bins.Add(new List<int> { cut.Index });
                continue;
            }

            if (tree.TryExtractBestFit(cut.Length, out int remaining, out int binId))
            {
                bins[binId].Add(cut.Index);
                int newRemain = remaining - cut.Length;
                if (newRemain > 0)
                {
                    tree.Add(newRemain, binId);
                }
            }
            else
            {
                int newId = bins.Count;
                bins.Add(new List<int> { cut.Index });
                int remain = stockLength - cut.Length;
                if (remain > 0)
                {
                    tree.Add(remain, newId);
                }
            }
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput());
        writer.WriteLine(bins.Count);
        foreach (var list in bins)
        {
            if (list.Count == 0)
            {
                writer.WriteLine();
                continue;
            }

            writer.Write(list[0]);
            for (int i = 1; i < list.Count; i++)
            {
                writer.Write(' ');
                writer.Write(list[i]);
            }
            writer.WriteLine();
        }
    }

    struct Cut
    {
        public int Length;
        public int Index;
    }

    sealed class Treap
    {
        private TreapNode _root;
        private int _seed = 1234567;

        public void Add(int remaining, int id)
        {
            if (remaining <= 0) return;
            var node = new TreapNode(remaining, id, NextPriority());
            Insert(ref _root, node);
        }

        public bool TryExtractBestFit(int length, out int remaining, out int id)
        {
            TreapNode current = _root;
            TreapNode candidate = null;

            while (current != null)
            {
                if (current.Remaining >= length)
                {
                    candidate = current;
                    current = current.Left;
                }
                else
                {
                    current = current.Right;
                }
            }

            if (candidate == null)
            {
                remaining = 0;
                id = -1;
                return false;
            }

            remaining = candidate.Remaining;
            id = candidate.Id;
            Remove(ref _root, remaining, id);
            return true;
        }

        private int NextPriority()
        {
            _seed = unchecked(_seed * 1103515245 + 12345);
            return _seed;
        }

        private static void Insert(ref TreapNode root, TreapNode node)
        {
            if (root == null)
            {
                root = node;
                return;
            }

            if (Compare(node, root) < 0)
            {
                Insert(ref root.Left, node);
                if (root.Left.Priority < root.Priority)
                {
                    RotateRight(ref root);
                }
            }
            else
            {
                Insert(ref root.Right, node);
                if (root.Right.Priority < root.Priority)
                {
                    RotateLeft(ref root);
                }
            }
        }

        private static void Remove(ref TreapNode root, int remaining, int id)
        {
            if (root == null) return;

            int cmp = Compare(remaining, id, root.Remaining, root.Id);
            if (cmp < 0)
            {
                Remove(ref root.Left, remaining, id);
            }
            else if (cmp > 0)
            {
                Remove(ref root.Right, remaining, id);
            }
            else
            {
                if (root.Left == null)
                {
                    root = root.Right;
                }
                else if (root.Right == null)
                {
                    root = root.Left;
                }
                else
                {
                    if (root.Left.Priority < root.Right.Priority)
                    {
                        RotateRight(ref root);
                        Remove(ref root.Right, remaining, id);
                    }
                    else
                    {
                        RotateLeft(ref root);
                        Remove(ref root.Left, remaining, id);
                    }
                }
            }
        }

        private static int Compare(TreapNode a, TreapNode b) => Compare(a.Remaining, a.Id, b.Remaining, b.Id);

        private static int Compare(int remA, int idA, int remB, int idB)
        {
            if (remA != remB) return remA < remB ? -1 : 1;
            if (idA == idB) return 0;
            return idA < idB ? -1 : 1;
        }

        private static void RotateLeft(ref TreapNode root)
        {
            var r = root.Right;
            root.Right = r.Left;
            r.Left = root;
            root = r;
        }

        private static void RotateRight(ref TreapNode root)
        {
            var l = root.Left;
            root.Left = l.Right;
            l.Right = root;
            root = l;
        }

        private sealed class TreapNode
        {
            public int Remaining;
            public int Id;
            public int Priority;
            public TreapNode Left;
            public TreapNode Right;

            public TreapNode(int remaining, int id, int priority)
            {
                Remaining = remaining;
                Id = id;
                Priority = priority;
            }
        }
    }

    sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _position;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private byte Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0) return 0;
            }
            return _buffer[_position++];
        }

        public bool TryReadInt(out int value)
        {
            byte b;
            do
            {
                b = Read();
                if (b == 0)
                {
                    value = 0;
                    return false;
                }
            } while (b <= 32);

            int sign = 1;
            if (b == '-')
            {
                sign = -1;
                b = Read();
            }

            int result = 0;
            while (b > 32)
            {
                result = result * 10 + (b - 48);
                b = Read();
            }

            value = result * sign;
            return true;
        }

        public int NextInt()
        {
            if (!TryReadInt(out int v))
                throw new InvalidOperationException("Unexpected end of input");
            return v;
        }
    }
}
