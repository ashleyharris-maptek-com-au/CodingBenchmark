OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
%glsl = OpExtInstImport "GLSL.std.450"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %arr_v3float ArrayStride 16
OpDecorate %outColor Location 0

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_Input_arr_pv Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_Output_pv Output
%arr_v3float = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3float
%ptr_Input_v3 = OpTypePointer Input %v3float
%color_in = OpVariable %ptr_Input_arr_v3 Input
%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005
%float_eps = OpConstant %float 0.000001

%main = OpFunction %void None %functype
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%col0_ptr = OpAccessChain %ptr_Input_v3 %color_in %int_0
%col1_ptr = OpAccessChain %ptr_Input_v3 %color_in %int_1
%col0 = OpLoad %v3float %col0_ptr
%col1 = OpLoad %v3float %col1_ptr
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%diff2 = OpFSub %v2float %p1_xy %p0_xy
%lenSq = OpDot %float %diff2 %diff2
%len = OpExtInst %float %glsl Sqrt %lenSq
%safeLen = OpExtInst %float %glsl FMax %len %float_eps
%invLen = OpFDiv %float %float_1 %safeLen
%diff_x = OpCompositeExtract %float %diff2 0
%diff_y = OpCompositeExtract %float %diff2 1
%neg_diff_y = OpFNegate %float %diff_y
%perp2 = OpCompositeConstruct %v2float %neg_diff_y %diff_x
%invLenVec = OpCompositeConstruct %v2float %invLen %invLen
%perpNorm2 = OpFMul %v2float %perp2 %invLenVec
%thickStartVec = OpCompositeConstruct %v2float %float_0_04 %float_0_04
%thickEndVec = OpCompositeConstruct %v2float %float_0_005 %float_0_005
%offsetStart2 = OpFMul %v2float %perpNorm2 %thickStartVec
%offsetEnd2 = OpFMul %v2float %perpNorm2 %thickEndVec
%offStartX = OpCompositeExtract %float %offsetStart2 0
%offStartY = OpCompositeExtract %float %offsetStart2 1
%offsetStart4 = OpCompositeConstruct %v4float %offStartX %offStartY %float_0 %float_0
%offEndX = OpCompositeExtract %float %offsetEnd2 0
%offEndY = OpCompositeExtract %float %offsetEnd2 1
%offsetEnd4 = OpCompositeConstruct %v4float %offEndX %offEndY %float_0 %float_0
%plusStart = OpFAdd %v4float %p0 %offsetStart4
%minusStart = OpFSub %v4float %p0 %offsetStart4
%plusEnd = OpFAdd %v4float %p1 %offsetEnd4
%minusEnd = OpFSub %v4float %p1 %offsetEnd4
%outPosPtr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %outPosPtr %plusStart
OpStore %outColor %col0
OpEmitVertex
OpStore %outPosPtr %minusStart
OpStore %outColor %col0
OpEmitVertex
OpStore %outPosPtr %plusEnd
OpStore %outColor %col1
OpEmitVertex
OpStore %outPosPtr %minusEnd
OpStore %outColor %col1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
