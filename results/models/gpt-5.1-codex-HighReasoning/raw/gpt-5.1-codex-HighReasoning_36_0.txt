#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>
#include <bit>
#include <utility>

using namespace std;

int nVertices = 0;
int blockCount = 0;
vector<vector<uint64_t>> compAdj;
vector<int> compDegree;
vector<int> bestClique;
int bestSize = 0;

inline bool isCompAdjacent(int u, int v) {
    if (blockCount == 0) return false;
    return (compAdj[u][v >> 6] >> (v & 63)) & 1ULL;
}

void bitsetToVector(const vector<uint64_t>& bits, vector<int>& out) {
    out.clear();
    for (int block = 0; block < blockCount; ++block) {
        uint64_t val = bits[block];
        while (val) {
            int offset = std::countr_zero(val);
            int idx = (block << 6) + offset;
            if (idx >= nVertices) break;
            out.push_back(idx);
            val &= val - 1;
        }
    }
}

void colorSort(vector<int> Pvec, vector<int>& order, vector<int>& colors) {
    vector<int> classMembers;
    classMembers.reserve(Pvec.size());
    order.clear();
    colors.clear();
    int color = 0;
    while (!Pvec.empty()) {
        ++color;
        classMembers.clear();
        size_t nextSize = 0;
        for (size_t i = 0; i < Pvec.size(); ++i) {
            int v = Pvec[i];
            bool conflict = false;
            for (int u : classMembers) {
                if (isCompAdjacent(v, u)) {
                    conflict = true;
                    break;
                }
            }
            if (!conflict) {
                classMembers.push_back(v);
                order.push_back(v);
                colors.push_back(color);
            } else {
                Pvec[nextSize++] = v;
            }
        }
        Pvec.resize(nextSize);
    }
}

void expand(vector<uint64_t> Pbits, vector<int>& current) {
    vector<int> Pvec;
    bitsetToVector(Pbits, Pvec);
    if (Pvec.empty()) {
        if (static_cast<int>(current.size()) > bestSize) {
            bestSize = static_cast<int>(current.size());
            bestClique = current;
        }
        return;
    }

    int currSize = static_cast<int>(current.size());
    if (currSize + static_cast<int>(Pvec.size()) <= bestSize) {
        return;
    }

    sort(Pvec.begin(), Pvec.end(), [&](int a, int b) {
        if (compDegree[a] != compDegree[b]) return compDegree[a] > compDegree[b];
        return a < b;
    });

    int candidateCount = static_cast<int>(Pvec.size());
    vector<int> order;
    vector<int> colors;
    order.reserve(candidateCount);
    colors.reserve(candidateCount);
    colorSort(std::move(Pvec), order, colors);

    for (int idx = static_cast<int>(order.size()) - 1; idx >= 0; --idx) {
        if (currSize + colors[idx] <= bestSize) {
            return;
        }
        int v = order[idx];
        uint64_t mask = 1ULL << (v & 63);
        if ((Pbits[v >> 6] & mask) == 0) continue;

        current.push_back(v);
        vector<uint64_t> newP(blockCount);
        bool empty = true;
        for (int b = 0; b < blockCount; ++b) {
            newP[b] = Pbits[b] & compAdj[v][b];
            if (newP[b]) empty = false;
        }
        if (empty) {
            if (static_cast<int>(current.size()) > bestSize) {
                bestSize = static_cast<int>(current.size());
                bestClique = current;
            }
        } else {
            expand(std::move(newP), current);
        }
        current.pop_back();
        Pbits[v >> 6] &= ~mask;
    }
}

bool hasEdgesWithinAllowed(const vector<uint64_t>& allowedBits) {
    if (blockCount == 0) return false;
    for (int v = 0; v < nVertices; ++v) {
        if ((allowedBits[v >> 6] & (1ULL << (v & 63))) == 0) continue;
        for (int b = 0; b < blockCount; ++b) {
            uint64_t mask = allowedBits[b];
            if (!mask) continue;
            uint64_t missing = (~compAdj[v][b]) & mask;
            if (b == (v >> 6)) {
                missing &= ~(1ULL << (v & 63));
            }
            if (missing) return true;
        }
    }
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    long long m;
    if (!(cin >> n >> m)) {
        return 0;
    }

    if (n == 0) {
        cout << 0 << "\n\n";
        return 0;
    }
    if (m == 0) {
        cout << n << "\n";
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << i;
        }
        cout << "\n";
        return 0;
    }

    nVertices = n;
    blockCount = (n + 63) / 64;
    vector<uint64_t> validMask(blockCount, ~0ULL);
    int remainder = n % 64;
    if (blockCount > 0 && remainder != 0) {
        validMask.back() = (1ULL << remainder) - 1;
    }

    compAdj.assign(n, vector<uint64_t>(blockCount, 0));
    for (int i = 0; i < n; ++i) {
        for (int b = 0; b < blockCount; ++b) {
            compAdj[i][b] = validMask[b];
        }
        compAdj[i][i >> 6] &= ~(1ULL << (i & 63));
    }

    vector<char> forbidden(n, 0);
    vector<int> degree(n, 0);

    for (long long e = 0; e < m; ++e) {
        int u, v;
        if (!(cin >> u >> v)) return 0;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            forbidden[u] = 1;
            continue;
        }
        if (forbidden[u] || forbidden[v]) {
            continue;
        }
        uint64_t maskV = 1ULL << (v & 63);
        uint64_t maskU = 1ULL << (u & 63);
        int blockUV = v >> 6;
        int blockVU = u >> 6;
        if (compAdj[u][blockUV] & maskV) {
            compAdj[u][blockUV] &= ~maskV;
            compAdj[v][blockVU] &= ~maskU;
            if (degree[u] < n - 1) ++degree[u];
            if (degree[v] < n - 1) ++degree[v];
        }
    }

    vector<uint64_t> initial(blockCount, 0);
    for (int b = 0; b < blockCount; ++b) {
        initial[b] = validMask[b];
    }
    for (int v = 0; v < n; ++v) {
        if (forbidden[v]) {
            initial[v >> 6] &= ~(1ULL << (v & 63));
        }
    }

    bool hasAllowed = false;
    for (int b = 0; b < blockCount; ++b) {
        if (initial[b]) {
            hasAllowed = true;
            break;
        }
    }
    if (!hasAllowed) {
        cout << 0 << "\n\n";
        return 0;
    }

    if (!hasEdgesWithinAllowed(initial)) {
        vector<int> answer;
        bitsetToVector(initial, answer);
        cout << answer.size() << "\n";
        for (size_t i = 0; i < answer.size(); ++i) {
            if (i) cout << ' ';
            cout << answer[i];
        }
        cout << "\n";
        return 0;
    }

    compDegree.assign(n, 0);
    for (int i = 0; i < n; ++i) {
        if (forbidden[i]) {
            compDegree[i] = -1;
            for (int b = 0; b < blockCount; ++b) {
                compAdj[i][b] = 0;
            }
        } else {
            int deg = degree[i];
            if (deg > n - 1) deg = n - 1;
            compDegree[i] = (n - 1) - deg;
        }
    }

    bestClique.clear();
    bestClique.reserve(n);
    bestSize = 0;
    vector<int> current;
    current.reserve(n);
    expand(std::move(initial), current);

    sort(bestClique.begin(), bestClique.end());
    cout << bestSize << "\n";
    if (bestSize > 0) {
        for (int i = 0; i < bestSize; ++i) {
            if (i) cout << ' ';
            cout << bestClique[i];
        }
    }
    cout << "\n";
    return 0;
}
