```typescript
declare const require: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');

(() => {
  if (!input || !input.trim()) {
    console.log(JSON.stringify({ firebreaks: [] }));
    return;
  }

  const data = JSON.parse(input);

  const toIntOrNull = (value: any): number | null => {
    const num = Number(value);
    if (!Number.isFinite(num)) {
      return null;
    }
    return num < 0 ? Math.ceil(num) : Math.floor(num);
  };

  const widthValue = toIntOrNull(data.width);
  const heightValue = toIntOrNull(data.height);
  const width = widthValue !== null ? Math.max(0, widthValue) : 0;
  const height = heightValue !== null ? Math.max(0, heightValue) : 0;
  const size = width * height;

  const cutBudgetValue = Number(data.cutBudget);
  const cutBudget = Number.isFinite(cutBudgetValue) ? Math.max(0, Math.floor(cutBudgetValue)) : 0;

  const firebreaks: number[][] = [];

  if (width <= 0 || height <= 0 || size <= 0 || cutBudget === 0) {
    console.log(JSON.stringify({ firebreaks }));
    return;
  }

  const gridInput: string[] = Array.isArray(data.grid) ? data.grid : [];
  const burnable = new Uint8Array(size);
  for (let y = 0; y < height; y++) {
    const row = typeof gridInput[y] === 'string' ? gridInput[y] : '';
    const base = y * width;
    for (let x = 0; x < width; x++) {
      const code = row.charCodeAt(x);
      burnable[base + x] = code === 46 ? 1 : 0; // '.'
    }
  }

  const nodeTownValue = new Float64Array(size);
  const townsInput = Array.isArray(data.towns) ? data.towns : [];
  let totalTownValue = 0;
  for (const town of townsInput) {
    if (!town) continue;
    const value = Number(town.value);
    if (!Number.isFinite(value) || value <= 0) continue;
    const xCoord = toIntOrNull(town.x);
    const yCoord = toIntOrNull(town.y);
    if (xCoord === null || yCoord === null) continue;
    const x = xCoord;
    const y = yCoord;
    if (x < 0 || x >= width || y < 0 || y >= height) continue;
    const idx = y * width + x;
    nodeTownValue[idx] += value;
    totalTownValue += value;
  }

  if (totalTownValue <= 0) {
    console.log(JSON.stringify({ firebreaks }));
    return;
  }

  const parent = new Int32Array(size);
  parent.fill(-1);
  const depth = new Int32Array(size);
  depth.fill(-1);
  const firstChild = new Int32Array(size);
  firstChild.fill(-1);
  const nextSibling = new Int32Array(size);
  nextSibling.fill(-1);

  const queue = new Int32Array(size);
  let qh = 0;
  let qt = 0;
  const order: number[] = [];
  const roots: number[] = [];
  const isIgnitionCell = new Uint8Array(size);

  const ignitionsInput = Array.isArray(data.ignitions) ? data.ignitions : [];

  const enqueueRoot = (idx: number): void => {
    depth[idx] = 0;
    parent[idx] = -1;
    queue[qt++] = idx;
    order.push(idx);
    roots.push(idx);
  };

  for (const ig of ignitionsInput) {
    if (ig == null) continue;
    let xCoord: number | null = null;
    let yCoord: number | null = null;
    if (Array.isArray(ig)) {
      xCoord = toIntOrNull(ig[0]);
      yCoord = toIntOrNull(ig[1]);
    } else if (typeof ig === 'object') {
      xCoord = toIntOrNull((ig as any).x);
      yCoord = toIntOrNull((ig as any).y);
    }
    if (xCoord === null || yCoord === null) continue;
    const x = xCoord;
    const y = yCoord;
    if (x < 0 || x >= width || y < 0 || y >= height) continue;
    const idx = y * width + x;
    isIgnitionCell[idx] = 1;
    if (burnable[idx] === 0 || depth[idx] !== -1) continue;
    enqueueRoot(idx);
  }

  if (qt === 0) {
    console.log(JSON.stringify({ firebreaks }));
    return;
  }

  while (qh < qt) {
    const idx = queue[qh++];
    const x = idx % width;
    const y = (idx / width) | 0;
    const nextDepth = depth[idx] + 1;

    if (x > 0) {
      const nIdx = idx - 1;
      if (burnable[nIdx] === 1 && depth[nIdx] === -1) {
        depth[nIdx] = nextDepth;
        parent[nIdx] = idx;
        nextSibling[nIdx] = firstChild[idx];
        firstChild[idx] = nIdx;
        queue[qt++] = nIdx;
        order.push(nIdx);
      }
    }
    if (x + 1 < width) {
      const nIdx = idx + 1;
      if (burnable[nIdx] === 1 && depth[nIdx] === -1) {
        depth[nIdx] = nextDepth;
        parent[nIdx] = idx;
        nextSibling[nIdx] = firstChild[idx];
        firstChild[idx] = nIdx;
        queue[qt++] = nIdx;
        order.push(nIdx);
      }
    }
    if (y > 0) {
      const nIdx = idx - width;
      if (burnable[nIdx] === 1 && depth[nIdx] === -1) {
        depth[nIdx] = nextDepth;
        parent[nIdx] = idx;
        nextSibling[nIdx] = firstChild[idx];
        firstChild[idx] = nIdx;
        queue[qt++] = nIdx;
        order.push(nIdx);
      }
    }
    if (y + 1 < height) {
      const nIdx = idx + width;
      if (burnable[nIdx] === 1 && depth[nIdx] === -1) {
        depth[nIdx] = nextDepth;
        parent[nIdx] = idx;
        nextSibling[nIdx] = firstChild[idx];
        firstChild[idx] = nIdx;
        queue[qt++] = nIdx;
        order.push(nIdx);
      }
    }
  }

  if (order.length === 0) {
    console.log(JSON.stringify({ firebreaks }));
    return;
  }

  const visitedCount = order.length;
  const entry = new Int32Array(size);
  const exit = new Int32Array(size);
  const entryOrder = new Int32Array(visitedCount);
  entryOrder.fill(-1);

  let time = 0;
  const stackNodes: number[] = [];
  const stackNext: number[] = [];

  for (const root of roots) {
    stackNodes.push(root);
    stackNext.push(firstChild[root]);
    entry[root] = time;
    entryOrder[time] = root;
    time++;
    while (stackNodes.length > 0) {
      const current = stackNodes[stackNodes.length - 1];
      let child = stackNext[stackNext.length - 1];
      if (child !== -1) {
        stackNext[stackNext.length - 1] = nextSibling[child];
        entry[child] = time;
        entryOrder[time] = child;
        time++;
        stackNodes.push(child);
        stackNext.push(firstChild[child]);
      } else {
        exit[current] = time - 1;
        stackNodes.pop();
        stackNext.pop();
      }
    }
  }

  const subtree = new Float64Array(size);
  for (let i = order.length - 1; i >= 0; i--) {
    const idx = order[i];
    let total = nodeTownValue[idx];
    let child = firstChild[idx];
    while (child !== -1) {
      total += subtree[child];
      child = nextSibling[child];
    }
    subtree[idx] = total;
  }

  const currentValue = new Float64Array(subtree);

  const EPS = 1e-6;

  let remainingValue = 0;
  for (const root of roots) {
    remainingValue += subtree[root];
  }

  if (remainingValue <= EPS) {
    console.log(JSON.stringify({ firebreaks }));
    return;
  }

  const leafValues = new Float64Array(visitedCount);
  for (const idx of order) {
    const pos = entry[idx];
    if (pos >= 0 && !isIgnitionCell[idx]) {
      const val = subtree[idx];
      if (val > 0) {
        leafValues[pos] = val;
      }
    }
  }

  class SegmentTree {
    private readonly size: number;
    private readonly n: number;
    private readonly maxVal: Float64Array;
    private readonly maxIdx: Int32Array;
    private readonly lazyZero: Uint8Array;

    constructor(values: Float64Array) {
      this.n = values.length;
      let sz = 1;
      while (sz < this.n) {
        sz <<= 1;
      }
      this.size = sz;
      this.maxVal = new Float64Array(this.size << 1);
      this.maxIdx = new Int32Array(this.size << 1);
      this.lazyZero = new Uint8Array(this.size << 1);

      for (let i = 0; i < this.size; i++) {
        const val = i < this.n ? values[i] : 0;
        const node = this.size + i;
        this.maxVal[node] = val;
        this.maxIdx[node] = val > EPS && i < this.n ? i : -1;
      }
      for (let i = this.size - 1; i >= 1; i--) {
        this.pull(i);
      }
    }

    private pull(node: number): void {
      const left = node << 1;
      const right = left | 1;
      if (this.maxVal[left] >= this.maxVal[right]) {
        this.maxVal[node] = this.maxVal[left];
        this.maxIdx[node] = this.maxIdx[left];
      } else {
        this.maxVal[node] = this.maxVal[right];
        this.maxIdx[node] = this.maxIdx[right];
      }
    }

    private applyZero(node: number): void {
      this.maxVal[node] = 0;
      this.maxIdx[node] = -1;
      this.lazyZero[node] = 1;
    }

    private push(node: number): void {
      if (this.lazyZero[node] && node < this.size) {
        const left = node << 1;
        const right = left | 1;
        this.applyZero(left);
        this.applyZero(right);
        this.lazyZero[node] = 0;
      }
    }

    rangeZero(l: number, r: number): void {
      if (this.n === 0) return;
      if (l < 0) l = 0;
      if (r >= this.n) r = this.n - 1;
      if (l > r) return;
      this.rangeZeroRec(1, 0, this.size - 1, l, r);
    }

    private rangeZeroRec(node: number, left: number, right: number, l: number, r: number): void {
      if (l > right || r < left) return;
      if (l <= left && right <= r) {
        this.applyZero(node);
        return;
      }
      this.push(node);
      const mid = (left + right) >> 1;
      if (l <= mid) this.rangeZeroRec(node << 1, left, mid, l, r);
      if (r > mid) this.rangeZeroRec((node << 1) | 1, mid + 1, right, l, r);
      this.pull(node);
    }

    pointUpdate(pos: number, value: number): void {
      if (pos < 0 || pos >= this.n) return;
      let node = this.size + pos;
      this.maxVal[node] = value;
      this.maxIdx[node] = value > EPS ? pos : -1;
      node >>= 1;
      while (node >= 1) {
        this.pull(node);
        node >>= 1;
      }
    }

    getMax(): { value: number; index: number } {
      return { value: this.maxVal[1], index: this.maxIdx[1] };
    }
  }

  const segTree = new SegmentTree(leafValues);
  let remainingCuts = cutBudget;

  while (remainingCuts > 0 && remainingValue > EPS) {
    const best = segTree.getMax();
    if (best.index < 0 || best.value <= EPS) {
      break;
    }
    const nodeIdx = entryOrder[best.index];
    if (nodeIdx == null || nodeIdx < 0) {
      segTree.pointUpdate(best.index, 0);
      continue;
    }
    const gain = currentValue[nodeIdx];
    if (!(gain > EPS)) {
      segTree.pointUpdate(best.index, 0);
      continue;
    }
    const diff = Math.abs(gain - best.value);
    if (diff > EPS * Math.max(1, Math.abs(best.value))) {
      segTree.pointUpdate(best.index, gain);
      continue;
    }

    const x = nodeIdx % width;
    const y = (nodeIdx / width) | 0;
    firebreaks.push([x, y]);
    remainingCuts--;
    currentValue[nodeIdx] = 0;
    segTree.rangeZero(entry[nodeIdx], exit[nodeIdx]);
    remainingValue -= gain;
    if (remainingValue < 0) remainingValue = 0;

    let ancestor = parent[nodeIdx];
    while (ancestor !== -1) {
      const oldValue = currentValue[ancestor];
      let newValue = oldValue - gain;
      if (newValue < 0) newValue = 0;
      if (Math.abs(newValue - oldValue) > EPS) {
        currentValue[ancestor] = newValue;
        if (!isIgnitionCell[ancestor]) {
          segTree.pointUpdate(entry[ancestor], newValue);
        }
      }
      ancestor = parent[ancestor];
    }
  }

  console.log(JSON.stringify({ firebreaks }));
})();
```
