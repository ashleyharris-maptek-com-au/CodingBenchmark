const fs = require('fs');

let input = '';
try {
  input = fs.readFileSync(0, 'utf8');
} catch (_) {}

if (!input || !input.trim()) {
  process.stdout.write('{"firebreaks":[]}');
  process.exit(0);
}

let data;
try {
  data = JSON.parse(input);
} catch (_) {
  process.stdout.write('{"firebreaks":[]}');
  process.exit(0);
}

const providedGrid = Array.isArray(data.grid) ? data.grid : [];
let width = Number.isFinite(Number(data.width)) ? Math.max(0, Math.floor(Number(data.width))) : 0;
let height = Number.isFinite(Number(data.height)) ? Math.max(0, Math.floor(Number(data.height))) : 0;

if (providedGrid.length > 0) {
  height = Math.max(height, providedGrid.length);
  for (const row of providedGrid) {
    if (typeof row === 'string') {
      width = Math.max(width, row.length);
    }
  }
}

const gridStrings = [];
for (let y = 0; y < height; y++) {
  const row = providedGrid[y];
  gridStrings.push(typeof row === 'string' ? row : '');
}

const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const rawTowns = Array.isArray(data.towns) ? data.towns : [];
const cutBudget = Math.max(0, Math.floor(Number.isFinite(Number(data.cutBudget)) ? Number(data.cutBudget) : 0));

const directions = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function rawCharAt(x, y) {
  if (!inBounds(x, y)) return '~';
  const row = gridStrings[y];
  if (typeof row !== 'string' || x < 0 || x >= row.length) return '~';
  const ch = row.charAt(x);
  return ch === '' ? '~' : ch;
}

function isBurnable(x, y) {
  return inBounds(x, y) && rawCharAt(x, y) === '.';
}

const idx = (x, y) => y * width + x;
const size = width * height;
const dist = new Array(size).fill(Infinity);
const queueX = [];
const queueY = [];
let qhead = 0;
let qtail = 0;

function enqueue(x, y) {
  queueX[qtail] = x;
  queueY[qtail] = y;
  qtail++;
}

if (width > 0 && height > 0) {
  for (const pt of ignitions) {
    if (!Array.isArray(pt) || pt.length < 2) continue;
    const xNum = Number(pt[0]);
    const yNum = Number(pt[1]);
    if (!Number.isFinite(xNum) || !Number.isFinite(yNum)) continue;
    const x = Math.round(xNum);
    const y = Math.round(yNum);
    if (!isBurnable(x, y)) continue;
    const id = idx(x, y);
    if (dist[id] !== Infinity) continue;
    dist[id] = 0;
    enqueue(x, y);
  }

  while (qhead < qtail) {
    const x = queueX[qhead];
    const y = queueY[qhead];
    const baseIdx = idx(x, y);
    const baseDist = dist[baseIdx];
    qhead++;

    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;
      if (!isBurnable(nx, ny)) continue;
      const nIdx = idx(nx, ny);
      if (dist[nIdx] !== Infinity) continue;
      dist[nIdx] = baseDist + 1;
      enqueue(nx, ny);
    }
  }
}

const townCells = new Set();
const townsNormalized = [];
for (const t of rawTowns) {
  if (!t) continue;
  const xNum = Number(t.x);
  const yNum = Number(t.y);
  if (!Number.isFinite(xNum) || !Number.isFinite(yNum)) continue;
  const x = Math.round(xNum);
  const y = Math.round(yNum);
  if (!inBounds(x, y)) continue;
  const valueNum = Number(t.value);
  const value = Number.isFinite(valueNum) ? valueNum : 0;
  townsNormalized.push({ id: t.id, x, y, value });
  townCells.add(`${x},${y}`);
}

const townsWithInfo = townsNormalized.map((t) => {
  let d = Infinity;
  if (isBurnable(t.x, t.y)) {
    d = dist[idx(t.x, t.y)];
  }
  const priority = Number.isFinite(d) && d >= 0 ? (t.value || 0) / (1 + d) : 0;
  return { ...t, dist: d, priority };
});

townsWithInfo.sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  if ((b.value || 0) !== (a.value || 0)) return (b.value || 0) - (a.value || 0);
  const aId = a.id !== undefined ? String(a.id) : '';
  const bId = b.id !== undefined ? String(b.id) : '';
  return aId.localeCompare(bId);
});

let remaining = cutBudget;
const firebreaks = [];
const cutSet = new Set();

function tryCut(x, y) {
  if (remaining <= 0) return false;
  if (!isBurnable(x, y)) return false;
  const key = `${x},${y}`;
  if (townCells.has(key) || cutSet.has(key)) return false;
  cutSet.add(key);
  firebreaks.push([x, y]);
  remaining--;
  return true;
}

function computePathCells(town, limit) {
  const path = [];
  if (!Number.isFinite(town.dist) || town.dist <= 0) return path;
  let cx = town.x;
  let cy = town.y;
  let currentDist = town.dist;
  let steps = 0;
  while (currentDist > 0 && steps < limit) {
    let best = null;
    for (const [dx, dy] of directions) {
      const nx = cx + dx;
      const ny = cy + dy;
      if (!isBurnable(nx, ny)) continue;
      const nIdx = idx(nx, ny);
      const nd = dist[nIdx];
      if (!Number.isFinite(nd)) continue;
      if (nd < currentDist && (best === null || nd < best.dist)) {
        best = { x: nx, y: ny, dist: nd };
        if (nd === currentDist - 1) break;
      }
    }
    if (!best) break;
    path.push([best.x, best.y]);
    cx = best.x;
    cy = best.y;
    currentDist = best.dist;
    steps++;
  }
  return path;
}

// Stage 1: cut immediate neighbors of high-priority towns
for (const town of townsWithInfo) {
  if (remaining <= 0) break;
  if (!Number.isFinite(town.dist)) continue;
  const neighbors = [];
  for (const [dx, dy] of directions) {
    const nx = town.x + dx;
    const ny = town.y + dy;
    if (!isBurnable(nx, ny)) continue;
    const nIdx = idx(nx, ny);
    const nd = dist[nIdx];
    if (!Number.isFinite(nd)) continue;
    neighbors.push({ x: nx, y: ny, dist: nd });
  }
  neighbors.sort((a, b) => a.dist - b.dist);
  for (const cell of neighbors) {
    if (remaining <= 0) break;
    tryCut(cell.x, cell.y);
  }
}

// Stage 2: cut along shortest paths toward ignitions
const pathLimit = 5;
for (const town of townsWithInfo) {
  if (remaining <= 0) break;
  if (!Number.isFinite(town.dist) || town.dist <= 0) continue;
  const pathCells = computePathCells(town, pathLimit);
  for (const [px, py] of pathCells) {
    if (remaining <= 0) break;
    tryCut(px, py);
  }
}

// Stage 3: cut additional early-burning cells if budget remains
if (remaining > 0) {
  for (let i = 0; i < qtail && remaining > 0; i++) {
    const x = queueX[i];
    const y = queueY[i];
    tryCut(x, y);
  }
}

process.stdout.write(JSON.stringify({ firebreaks }));
