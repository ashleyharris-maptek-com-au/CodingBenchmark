declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = input.trim().length ? JSON.parse(input) : {};

type DroneInput = { id?: any; x?: number; y?: number };

const grid: any[] = Array.isArray(data.probability) ? data.probability : [];
const drones: DroneInput[] = Array.isArray(data.drones) ? data.drones : [];
const rawSteps = Number(data.maxStepsPerDrone);
const maxSteps: number = Number.isFinite(rawSteps) && rawSteps > 0 ? Math.floor(rawSteps) : 0;

const height: number = grid.length;
const width: number = grid.reduce(
    (max: number, row: any) => (Array.isArray(row) ? Math.max(max, row.length) : max),
    0
);

if (width === 0 || height === 0) {
    const fallbackPaths: number[][][] = drones.map(() => {
        const path: number[][] = [];
        const len = maxSteps + 1;
        for (let i = 0; i < len; i++) {
            path.push([0, 0]);
        }
        return path;
    });
    console.log(JSON.stringify({ paths: fallbackPaths }));
    process.exit(0);
}

const size: number = width * height;
const probability: Float64Array = new Float64Array(size);
const passable: Uint8Array = new Uint8Array(size);

for (let y = 0; y < height; y++) {
    const row = Array.isArray(grid[y]) ? grid[y] : [];
    for (let x = 0; x < width; x++) {
        const idx = y * width + x;
        const val = Number(row[x]);
        if (Number.isFinite(val) && val >= 0) {
            probability[idx] = val;
            passable[idx] = 1;
        } else {
            probability[idx] = 0;
            passable[idx] = 0;
        }
    }
}

const localValues: Float64Array = new Float64Array(size);
const orthoWeight = 0.35;
const diagWeight = 0.15;

for (let idx = 0; idx < size; idx++) {
    if (!passable[idx]) {
        localValues[idx] = 0;
        continue;
    }
    let sum = probability[idx];
    const x = idx % width;
    const y = (idx / width) | 0;

    if (x > 0) sum += probability[idx - 1] * orthoWeight;
    if (x + 1 < width) sum += probability[idx + 1] * orthoWeight;
    if (y > 0) sum += probability[idx - width] * orthoWeight;
    if (y + 1 < height) sum += probability[idx + width] * orthoWeight;

    if (x > 0 && y > 0) sum += probability[idx - width - 1] * diagWeight;
    if (x > 0 && y + 1 < height) sum += probability[idx + width - 1] * diagWeight;
    if (x + 1 < width && y > 0) sum += probability[idx - width + 1] * diagWeight;
    if (x + 1 < width && y + 1 < height) sum += probability[idx + width + 1] * diagWeight;

    localValues[idx] = sum;
}

const candidateCells: number[] = [];
for (let idx = 0; idx < size; idx++) {
    if (!passable[idx]) continue;
    if (localValues[idx] > 1e-6 || probability[idx] > 0) {
        candidateCells.push(idx);
    }
}

interface StartInfo {
    x: number;
    y: number;
    idx: number;
    valid: boolean;
}

const nearestVisited: Int32Array = new Int32Array(size);
let nearestToken = 1;
const nearestQueue: Int32Array = new Int32Array(size);

function clampCoordinate(value: any, limit: number): number {
    if (limit <= 0) return 0;
    const num = Number(value);
    if (!Number.isFinite(num)) return 0;
    let coord = Math.round(num);
    if (coord < 0) coord = 0;
    if (coord >= limit) coord = limit - 1;
    return coord;
}

function locateStart(rawX: any, rawY: any): StartInfo {
    if (size === 0) {
        return { x: 0, y: 0, idx: 0, valid: false };
    }
    let x = clampCoordinate(rawX, width);
    let y = clampCoordinate(rawY, height);
    let idx = y * width + x;
    if (passable[idx]) {
        return { x, y, idx, valid: true };
    }

    nearestToken++;
    if (nearestToken >= 0x7fffffff) {
        nearestVisited.fill(0);
        nearestToken = 1;
    }

    let head = 0;
    let tail = 0;
    nearestVisited[idx] = nearestToken;
    nearestQueue[tail++] = idx;

    while (head < tail) {
        const current = nearestQueue[head++];
        const cx = current % width;
        const cy = (current / width) | 0;

        if (cx > 0) {
            const left = current - 1;
            if (nearestVisited[left] !== nearestToken) {
                nearestVisited[left] = nearestToken;
                if (passable[left]) {
                    return { x: cx - 1, y: cy, idx: left, valid: true };
                }
                nearestQueue[tail++] = left;
            }
        }
        if (cx + 1 < width) {
            const right = current + 1;
            if (nearestVisited[right] !== nearestToken) {
                nearestVisited[right] = nearestToken;
                if (passable[right]) {
                    return { x: cx + 1, y: cy, idx: right, valid: true };
                }
                nearestQueue[tail++] = right;
            }
        }
        if (cy > 0) {
            const up = current - width;
            if (nearestVisited[up] !== nearestToken) {
                nearestVisited[up] = nearestToken;
                if (passable[up]) {
                    return { x: cx, y: cy - 1, idx: up, valid: true };
                }
                nearestQueue[tail++] = up;
            }
        }
        if (cy + 1 < height) {
            const down = current + width;
            if (nearestVisited[down] !== nearestToken) {
                nearestVisited[down] = nearestToken;
                if (passable[down]) {
                    return { x: cx, y: cy + 1, idx: down, valid: true };
                }
                nearestQueue[tail++] = down;
            }
        }
    }
    return { x, y, idx, valid: false };
}

const startInfos: StartInfo[] = drones.map((drone) => {
    const rawX = typeof drone?.x !== 'undefined' ? drone.x : 0;
    const rawY = typeof drone?.y !== 'undefined' ? drone.y : 0;
    return locateStart(rawX, rawY);
});

const covered: Uint8Array = new Uint8Array(size);
for (const info of startInfos) {
    if (info && info.valid) {
        covered[info.idx] = 1;
    }
}

const ownerOpsLimit = 5e7;
let cellOwner: Int32Array | null = null;
const usableStarts = startInfos.filter((info) => info && info.valid);
if (usableStarts.length > 0 && size > 0 && drones.length * size <= ownerOpsLimit) {
    cellOwner = new Int32Array(size);
    cellOwner.fill(-1);
    for (let idx = 0; idx < size; idx++) {
        const x = idx % width;
        const y = (idx / width) | 0;
        let best = -1;
        let bestDist = Number.POSITIVE_INFINITY;
        for (let dIdx = 0; dIdx < startInfos.length; dIdx++) {
            const info = startInfos[dIdx];
            if (!info || !info.valid) continue;
            const dist = Math.abs(x - info.x) + Math.abs(y - info.y);
            if (dist < bestDist) {
                bestDist = dist;
                best = dIdx;
            }
        }
        cellOwner[idx] = best;
    }
}

interface TargetSelection {
    idx: number;
}

function runBfs(
    startIdx: number,
    limit: number,
    dist: Int32Array,
    parent: Int32Array,
    queue: Int32Array
): void {
    dist.fill(-1);
    parent.fill(-1);
    let head = 0;
    let tail = 0;
    queue[tail++] = startIdx;
    dist[startIdx] = 0;

    while (head < tail) {
        const idx = queue[head++];
        const currentDist = dist[idx];
        if (currentDist >= limit) {
            continue;
        }
        const x = idx % width;
        const y = (idx / width) | 0;

        if (x > 0) {
            const left = idx - 1;
            if (passable[left] && dist[left] === -1) {
                dist[left] = currentDist + 1;
                parent[left] = idx;
                queue[tail++] = left;
            }
        }
        if (x + 1 < width) {
            const right = idx + 1;
            if (passable[right] && dist[right] === -1) {
                dist[right] = currentDist + 1;
                parent[right] = idx;
                queue[tail++] = right;
            }
        }
        if (y > 0) {
            const up = idx - width;
            if (passable[up] && dist[up] === -1) {
                dist[up] = currentDist + 1;
                parent[up] = idx;
                queue[tail++] = up;
            }
        }
        if (y + 1 < height) {
            const down = idx + width;
            if (passable[down] && dist[down] === -1) {
                dist[down] = currentDist + 1;
                parent[down] = idx;
                queue[tail++] = down;
            }
        }
    }
}

function chooseNextTarget(
    droneIndex: number,
    currentIdx: number,
    remaining: number,
    dist: Int32Array,
    parent: Int32Array,
    queue: Int32Array
): TargetSelection | null {
    if (remaining <= 0) {
        return null;
    }
    runBfs(currentIdx, remaining, dist, parent, queue);

    let bestIdx = -1;
    let bestScore = -Infinity;

    for (let i = 0; i < candidateCells.length; i++) {
        const idx = candidateCells[i];
        const d = dist[idx];
        if (d < 0 || d > remaining || d === 0) {
            continue;
        }

        const localValue = localValues[idx];
        const baseValue = probability[idx];
        if (localValue <= 0 && baseValue <= 0) {
            continue;
        }

        const isCovered = covered[idx] === 1;
        const coverageFactor = isCovered ? 0.35 : 1.0;

        let combinedValue = localValue;
        let neighborBonus = 0;
        const x = idx % width;
        const y = (idx / width) | 0;

        if (x > 0) {
            const left = idx - 1;
            if (passable[left] && covered[left] === 0) {
                neighborBonus += probability[left] * 0.1;
            }
        }
        if (x + 1 < width) {
            const right = idx + 1;
            if (passable[right] && covered[right] === 0) {
                neighborBonus += probability[right] * 0.1;
            }
        }
        if (y > 0) {
            const up = idx - width;
            if (passable[up] && covered[up] === 0) {
                neighborBonus += probability[up] * 0.1;
            }
        }
        if (y + 1 < height) {
            const down = idx + width;
            if (passable[down] && covered[down] === 0) {
                neighborBonus += probability[down] * 0.1;
            }
        }

        combinedValue += neighborBonus;

        if (combinedValue <= 0 && coverageFactor < 1) {
            continue;
        }

        let score = combinedValue * coverageFactor;

        if (cellOwner) {
            const owner = cellOwner[idx];
            if (owner === droneIndex) {
                score *= 1.15;
            } else if (owner !== -1) {
                score *= 0.93;
            }
        }

        const distancePenalty = 1 + d * 0.35;
        score /= distancePenalty;

        if (!isCovered && baseValue > 0) {
            score += baseValue * 0.06;
        } else if (baseValue > 0) {
            score += baseValue * 0.02;
        }

        if (neighborBonus > 0) {
            score += (neighborBonus / (1 + d)) * 0.25;
        }

        const spareSteps = remaining - d;
        if (spareSteps > 0) {
            const spareRatio = Math.min(spareSteps / Math.max(remaining, 1), 1);
            score *= 1 + spareRatio * 0.2;
        }

        score += ((idx % 17) * 1e-6);

        if (score > bestScore) {
            bestScore = score;
            bestIdx = idx;
        }
    }

    if (bestIdx === -1 || bestScore <= 1e-9) {
        return null;
    }
    return { idx: bestIdx };
}

function buildPathForDrone(
    droneIndex: number,
    startInfo: StartInfo,
    dist: Int32Array,
    parent: Int32Array,
    queue: Int32Array
): number[][] {
    const path: number[][] = [];
    const startX = startInfo ? startInfo.x : 0;
    const startY = startInfo ? startInfo.y : 0;
    const startIdx = startInfo ? startInfo.idx : 0;
    const validStart = Boolean(startInfo && startInfo.valid && passable[startIdx]);

    path.push([startX, startY]);

    if (!validStart || maxSteps === 0) {
        for (let s = 0; s < maxSteps; s++) {
            path.push([startX, startY]);
        }
        if (validStart && !covered[startIdx]) {
            covered[startIdx] = 1;
        }
        return path;
    }

    if (!covered[startIdx]) {
        covered[startIdx] = 1;
    }

    let stepsUsed = 0;
    let currentIdx = startIdx;
    let currentX = startX;
    let currentY = startY;

    while (stepsUsed < maxSteps) {
        const remaining = maxSteps - stepsUsed;
        if (remaining <= 0) {
            break;
        }
        const target = chooseNextTarget(droneIndex, currentIdx, remaining, dist, parent, queue);
        if (!target) {
            break;
        }
        const targetIdx = target.idx;
        if (targetIdx === currentIdx) {
            break;
        }

        const pathIndices: number[] = [];
        let walker = targetIdx;
        while (walker !== currentIdx && walker !== -1) {
            pathIndices.push(walker);
            walker = parent[walker];
        }
        if (walker !== currentIdx || pathIndices.length === 0) {
            break;
        }
        pathIndices.reverse();

        for (const idxMove of pathIndices) {
            if (stepsUsed >= maxSteps) {
                break;
            }
            const px = idxMove % width;
            const py = (idxMove / width) | 0;
            path.push([px, py]);
            stepsUsed++;
            currentIdx = idxMove;
            currentX = px;
            currentY = py;
            if (!covered[idxMove]) {
                covered[idxMove] = 1;
            }
        }
    }

    while (stepsUsed < maxSteps) {
        path.push([currentX, currentY]);
        stepsUsed++;
    }

    return path;
}

const sharedDist = new Int32Array(size);
const sharedParent = new Int32Array(size);
const sharedQueue = new Int32Array(size);

const paths: number[][][] = [];
for (let i = 0; i < drones.length; i++) {
    const startInfo = startInfos[i] ?? { x: 0, y: 0, idx: 0, valid: false };
    const path = buildPathForDrone(i, startInfo, sharedDist, sharedParent, sharedQueue);
    paths.push(path);
}
