use std::{
    f64::consts::PI,
    fmt::Debug,
    io::{self, BufRead, BufReader, Write},
    sync::{mpsc, Arc, Mutex},
    thread,
    time::{Duration, Instant},
};

struct Scanner<R: BufRead> {
    reader: R,
    buffer: Vec<u8>,
}

impl<R: BufRead> Scanner<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buffer: Vec::with_capacity(256),
        }
    }

    fn fill_token(&mut self) -> io::Result<Option<()>> {
        self.buffer.clear();
        loop {
            let available = self.reader.fill_buf()?;
            if available.is_empty() {
                return if self.buffer.is_empty() {
                    Ok(None)
                } else {
                    Ok(Some(()))
                };
            }

            let mut consumed = 0;
            while consumed < available.len() {
                let byte = available[consumed];
                if byte.is_ascii_whitespace() {
                    consumed += 1;
                    if !self.buffer.is_empty() {
                        self.reader.consume(consumed);
                        return Ok(Some(()));
                    }
                } else {
                    self.buffer.push(byte);
                    consumed += 1;
                }
            }
            self.reader.consume(consumed);
        }
    }

    fn read_required<T>(&mut self, label: &str) -> T
    where
        T: std::str::FromStr,
        <T as std::str::FromStr>::Err: Debug,
    {
        self.fill_token()
            .expect("Failed to read token")
            .unwrap_or_else(|| panic!("Unexpected EOF while reading {}", label));
        let slice = std::str::from_utf8(&self.buffer).expect("Invalid UTF-8 in token");
        slice
            .parse::<T>()
            .unwrap_or_else(|e| panic!("Failed to parse {} ({}): {:?}", label, slice, e))
    }

    fn read_optional<T>(&mut self) -> io::Result<Option<T>>
    where
        T: std::str::FromStr,
        <T as std::str::FromStr>::Err: Debug,
    {
        match self.fill_token()? {
            Some(()) => {
                let slice = std::str::from_utf8(&self.buffer).expect("Invalid UTF-8 in token");
                let value = slice
                    .parse::<T>()
                    .unwrap_or_else(|e| panic!("Failed to parse value {}: {:?}", slice, e));
                Ok(Some(value))
            }
            None => Ok(None),
        }
    }

    fn skip_token(&mut self) -> io::Result<bool> {
        Ok(self.fill_token()?.is_some())
    }
}

#[derive(Clone, Copy, Debug)]
struct Config {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
}

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    updated_at: Instant,
}

fn main() {
    let shared_state = Arc::new(Mutex::new(None));
    let (config_tx, config_rx) = mpsc::channel();

    {
        let shared_state = Arc::clone(&shared_state);
        thread::spawn(move || {
            run_input(shared_state, config_tx);
        });
    }

    let config = match config_rx.recv() {
        Ok(cfg) => cfg,
        Err(_) => return,
    };

    run_control_loop(config, shared_state);
}

fn run_input(shared_state: Arc<Mutex<Option<State>>>, config_tx: mpsc::Sender<Config>) {
    let stdin = io::stdin();
    let reader = BufReader::new(stdin.lock());
    let mut scanner = Scanner::new(reader);

    let _width: usize = scanner.read_required("width");
    let height: usize = scanner.read_required("height");
    let gravity: f64 = scanner.read_required("gravity");
    let max_thrust: f64 = scanner.read_required("max_thrust");
    let _fuel: f64 = scanner.read_required("fuel");

    let _start_x: f64 = scanner.read_required("start_x");
    let _start_y: f64 = scanner.read_required("start_y");
    let target_x: f64 = scanner.read_required("target_x");
    let target_y: f64 = scanner.read_required("target_y");

    let _ = config_tx.send(Config {
        gravity,
        max_thrust,
        target_x,
        target_y,
    });

    for _ in 0..height {
        if !scanner.skip_token().unwrap_or(false) {
            return;
        }
    }

    loop {
        let x = match scanner.read_optional::<f64>() {
            Ok(Some(v)) => v,
            _ => break,
        };
        let y = match scanner.read_optional::<f64>() {
            Ok(Some(v)) => v,
            _ => break,
        };
        let vx = match scanner.read_optional::<f64>() {
            Ok(Some(v)) => v,
            _ => break,
        };
        let vy = match scanner.read_optional::<f64>() {
            Ok(Some(v)) => v,
            _ => break,
        };
        let angle = match scanner.read_optional::<f64>() {
            Ok(Some(v)) => v,
            _ => break,
        };
        let angular_vel = match scanner.read_optional::<f64>() {
            Ok(Some(v)) => v,
            _ => break,
        };
        let fuel_remaining = match scanner.read_optional::<f64>() {
            Ok(Some(v)) => v,
            _ => break,
        };

        let state = State {
            x,
            y,
            vx,
            vy,
            angle,
            angular_vel,
            fuel: fuel_remaining,
            updated_at: Instant::now(),
        };

        store_state(&shared_state, state);
    }
}

fn store_state(shared: &Arc<Mutex<Option<State>>>, state: State) {
    match shared.lock() {
        Ok(mut guard) => {
            *guard = Some(state);
        }
        Err(poisoned) => {
            *poisoned.into_inner() = Some(state);
        }
    }
}

fn run_control_loop(config: Config, shared_state: Arc<Mutex<Option<State>>>) {
    let control_period = Duration::from_millis(30);
    let mut stdout = io::BufWriter::new(io::stdout());

    loop {
        let loop_start = Instant::now();
        let maybe_state = {
            let guard = shared_state.lock().unwrap_or_else(|e| e.into_inner());
            *guard
        };

        let (thrust, turn) = match maybe_state {
            Some(state) => {
                if state.updated_at.elapsed() > Duration::from_millis(400) {
                    (0.0, 0.0)
                } else {
                    guidance(&state, &config)
                }
            }
            None => (0.0, 0.0),
        };

        if writeln!(stdout, "{:.4} {:.4}", thrust, turn).is_err() {
            break;
        }
        if stdout.flush().is_err() {
            break;
        }

        let elapsed = loop_start.elapsed();
        if elapsed < control_period {
            thread::sleep(control_period - elapsed);
        }
    }
}

fn guidance(state: &State, cfg: &Config) -> (f64, f64) {
    let dx = cfg.target_x - state.x;
    let dy = cfg.target_y - state.y;

    let capped_dx = dx.clamp(-400.0, 400.0);
    let mut desired_angle = (capped_dx * 0.002 - state.vx * 0.08).clamp(-0.8, 0.8);

    let altitude_above = dy.max(0.0);
    let tilt_scale = (altitude_above / 60.0).clamp(0.0, 1.0);
    desired_angle *= 0.2 + 0.8 * tilt_scale;

    if altitude_above < 6.0 && state.vy.abs() < 3.0 {
        desired_angle *= 0.2;
    }

    let angle_error = wrap_angle(desired_angle - state.angle);
    let mut turn = angle_error * 3.0 - state.angular_vel * 0.7;
    turn = turn.clamp(-1.0, 1.0);

    let mut desired_vy = dy * 0.18;
    desired_vy = desired_vy.clamp(-3.0, 7.0);
    if altitude_above < 15.0 {
        desired_vy = desired_vy.clamp(-2.0, 3.0);
    }
    if altitude_above < 6.0 {
        desired_vy = desired_vy.clamp(-1.0, 2.0);
    }

    let vy_error = desired_vy - state.vy;
    let mut desired_ay = vy_error * 1.3 + dy * 0.03;
    if altitude_above < 5.0 {
        desired_ay -= state.vy * 0.5;
    }
    desired_ay = desired_ay.clamp(-cfg.max_thrust, cfg.gravity * 1.8);

    let cos_component = state.angle.cos().abs().max(0.1);
    let vertical_acc_cap = (cfg.max_thrust * cos_component).max(1e-3);

    let mut thrust = (cfg.gravity - desired_ay) / vertical_acc_cap;
    thrust = thrust.clamp(0.0, 1.0);

    if altitude_above < 0.5 && state.vy.abs() < 0.5 && state.y >= cfg.target_y - 0.5 {
        thrust = 0.0;
    }

    if state.fuel <= 0.0 {
        thrust = 0.0;
    }

    (thrust, turn)
}

fn wrap_angle(angle: f64) -> f64 {
    let two_pi = 2.0 * PI;
    ((angle + PI).rem_euclid(two_pi)) - PI
}
