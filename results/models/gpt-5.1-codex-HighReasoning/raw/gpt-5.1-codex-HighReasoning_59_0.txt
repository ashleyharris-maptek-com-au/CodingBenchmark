```javascript
const fs = require('fs');

const rawInput = fs.readFileSync(0, 'utf8');
const sanitizedInput = rawInput.replace(/^\uFEFF/, '');
const trimmedInput = sanitizedInput.trim();
const data = trimmedInput ? JSON.parse(trimmedInput) : {};

function parseNumber(value, fallback = 0) {
  const num = Number(value);
  return Number.isFinite(num) ? num : fallback;
}

function roundCoord(value) {
  return Math.round(parseNumber(value, 0));
}

function clamp(value, min, max) {
  if (max < min) return min;
  if (value < min) return min;
  if (value > max) return max;
  return value;
}

const rawProbability = Array.isArray(data.probability) ? data.probability : [];
const rows = rawProbability.length;
let cols = 0;
for (const row of rawProbability) {
  if (Array.isArray(row) && row.length > cols) {
    cols = row.length;
  }
}

const grid = [];
for (let r = 0; r < rows; r++) {
  const sourceRow = Array.isArray(rawProbability[r]) ? rawProbability[r] : [];
  const newRow = new Array(cols);
  for (let c = 0; c < cols; c++) {
    newRow[c] = parseNumber(sourceRow[c], 0);
  }
  grid[r] = newRow;
}

const drones = Array.isArray(data.drones) ? data.drones : [];
const stepsLimit = Math.max(0, Math.floor(parseNumber(data.maxStepsPerDrone, 0)));

if (rows === 0 || cols === 0) {
  const fallbackPaths = drones.map((drone) => {
    const x = roundCoord(drone && drone.x);
    const y = roundCoord(drone && drone.y);
    return [[x, y]];
  });
  console.log(JSON.stringify({ paths: fallbackPaths }));
} else {
  const gridSize = rows * cols;
  const candidateByIdx = new Array(gridSize);
  const candidateList = [];
  for (let r = 0; r < rows; r++) {
    for (let c = 0; c < cols; c++) {
      const value = grid[r][c];
      if (value > 0) {
        const idx = r * cols + c;
        const cand = { row: r, col: c, value, idx, taken: false };
        candidateByIdx[idx] = cand;
        candidateList.push(cand);
      }
    }
  }
  candidateList.sort((a, b) => b.value - a.value);
  let remainingValuable = candidateList.length;

  const dirs = [
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];

  function isValidCell(r, c) {
    return r >= 0 && r < rows && c >= 0 && c < cols && grid[r][c] !== -1;
  }

  const nearestVisited = new Uint32Array(gridSize);
  let nearestVisitToken = 1;
  const nearestQueueR = new Int32Array(gridSize);
  const nearestQueueC = new Int32Array(gridSize);

  function findNearestValidCell(row, col) {
    if (grid[row][col] !== -1) {
      return { row, col };
    }
    const token = nearestVisitToken++;
    if (nearestVisitToken === 0xffffffff) {
      nearestVisited.fill(0);
      nearestVisitToken = 1;
    }
    let head = 0;
    let tail = 0;
    const startIdx = row * cols + col;
    nearestVisited[startIdx] = token;
    nearestQueueR[tail] = row;
    nearestQueueC[tail] = col;
    tail++;
    while (head < tail) {
      const r = nearestQueueR[head];
      const c = nearestQueueC[head];
      head++;
      if (grid[r][c] !== -1) {
        return { row: r, col: c };
      }
      for (const [dr, dc] of dirs) {
        const nr = r + dr;
        const nc = c + dc;
        if (nr < 0 || nr >= rows || nc < 0 || nc >= cols) continue;
        const idx = nr * cols + nc;
        if (nearestVisited[idx] === token) continue;
        nearestVisited[idx] = token;
        nearestQueueR[tail] = nr;
        nearestQueueC[tail] = nc;
        tail++;
      }
    }
    return null;
  }

  const bfsDist = new Int32Array(gridSize);
  const bfsParent = new Int32Array(gridSize);
  const bfsQueueR = new Int32Array(gridSize);
  const bfsQueueC = new Int32Array(gridSize);

  function bfsAll(startRow, startCol) {
    bfsDist.fill(-1);
    bfsParent.fill(-1);
    let head = 0;
    let tail = 0;
    const startIdx = startRow * cols + startCol;
    bfsDist[startIdx] = 0;
    bfsQueueR[tail] = startRow;
    bfsQueueC[tail] = startCol;
    tail++;
    while (head < tail) {
      const r = bfsQueueR[head];
      const c = bfsQueueC[head];
      head++;
      const baseIdx = r * cols + c;
      const nextDist = bfsDist[baseIdx] + 1;
      for (const [dr, dc] of dirs) {
        const nr = r + dr;
        const nc = c + dc;
        if (nr < 0 || nr >= rows || nc < 0 || nc >= cols) continue;
        if (grid[nr][nc] === -1) continue;
        const idx = nr * cols + nc;
        if (bfsDist[idx] !== -1) continue;
        bfsDist[idx] = nextDist;
        bfsParent[idx] = baseIdx;
        bfsQueueR[tail] = nr;
        bfsQueueC[tail] = nc;
        tail++;
      }
    }
    return { dist: bfsDist, parent: bfsParent };
  }

  function reconstructPath(parent, startIdx, targetIdx) {
    const sequence = [];
    let current = targetIdx;
    while (current !== startIdx) {
      sequence.push(current);
      current = parent[current];
      if (current === -1) {
        return null;
      }
    }
    sequence.push(startIdx);
    sequence.reverse();
    return sequence;
  }

  function markVisitedPosition(r, c) {
    const idx = r * cols + c;
    const cand = candidateByIdx[idx];
    if (cand && !cand.taken) {
      cand.taken = true;
      remainingValuable--;
    }
  }

  const states = drones.map((drone) => {
    let col = clamp(roundCoord(drone && drone.x), 0, cols - 1);
    let row = clamp(roundCoord(drone && drone.y), 0, rows - 1);
    if (!isValidCell(row, col)) {
      const alternative = findNearestValidCell(row, col);
      if (alternative) {
        row = alternative.row;
        col = alternative.col;
      }
    }
    const path = [[col, row]];
    markVisitedPosition(row, col);
    return {
      row,
      col,
      stepsLeft: stepsLimit,
      path,
      canMove: stepsLimit > 0 && isValidCell(row, col),
    };
  });

  function planNextTarget(state) {
    if (state.stepsLeft <= 0 || !isValidCell(state.row, state.col)) {
      return false;
    }
    const { dist, parent } = bfsAll(state.row, state.col);
    let bestTarget = null;
    let bestScore = -Infinity;
    let bestDist = Infinity;
    for (const cand of candidateList) {
      if (cand.taken) continue;
      const d = dist[cand.idx];
      if (d === -1 || d > state.stepsLeft) continue;
      const score = cand.value / (d + 1);
      if (
        !bestTarget ||
        score > bestScore ||
        (score === bestScore &&
          (cand.value > bestTarget.value || (cand.value === bestTarget.value && d < bestDist)))
      ) {
        bestTarget = cand;
        bestScore = score;
        bestDist = d;
      }
    }
    if (!bestTarget) {
      return false;
    }
    const startIdx = state.row * cols + state.col;
    const sequence = reconstructPath(parent, startIdx, bestTarget.idx);
    if (!sequence) {
      return false;
    }
    if (sequence.length <= 1) {
      markVisitedPosition(state.row, state.col);
      return true;
    }
    for (let i = 1; i < sequence.length && state.stepsLeft > 0; i++) {
      const idx = sequence[i];
      const r = Math.floor(idx / cols);
      const c = idx % cols;
      state.path.push([c, r]);
      state.stepsLeft--;
      state.row = r;
      state.col = c;
      markVisitedPosition(r, c);
      if (remainingValuable === 0) {
        break;
      }
    }
    return true;
  }

  if (remainingValuable > 0) {
    let progress = true;
    while (progress && remainingValuable > 0) {
      progress = false;
      for (const state of states) {
        if (!state.canMove) continue;
        if (state.stepsLeft <= 0) {
          state.canMove = false;
          continue;
        }
        const moved = planNextTarget(state);
        if (moved) {
          progress = true;
          if (state.stepsLeft <= 0) {
            state.canMove = false;
          }
        } else {
          state.canMove = false;
        }
        if (remainingValuable === 0) {
          break;
        }
      }
    }
  }

  const outputPaths = states.map((state) => state.path);
  console.log(JSON.stringify({ paths: outputPaths }));
}
```
