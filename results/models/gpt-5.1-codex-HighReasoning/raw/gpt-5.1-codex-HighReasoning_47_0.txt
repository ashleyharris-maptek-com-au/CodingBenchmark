@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;

fn rotl32(x : u32, shift : u32) -> u32 {
    return (x << shift) | (x >> (32u - shift));
}

fn ctz32(x : u32) -> u32 {
    if (x == 0u) {
        return 32u;
    }
    var v : u32 = x;
    var n : u32 = 0u;
    if ((v & 0x0000FFFFu) == 0u) {
        n = n + 16u;
        v = v >> 16u;
    }
    if ((v & 0x000000FFu) == 0u) {
        n = n + 8u;
        v = v >> 8u;
    }
    if ((v & 0x0000000Fu) == 0u) {
        n = n + 4u;
        v = v >> 4u;
    }
    if ((v & 0x00000003u) == 0u) {
        n = n + 2u;
        v = v >> 2u;
    }
    if ((v & 0x00000001u) == 0u) {
        n = n + 1u;
    }
    return n;
}

fn trailingZeros64(h0 : u32, h1 : u32) -> u32 {
    if (h0 != 0u) {
        return ctz32(h0);
    }
    if (h1 != 0u) {
        return 32u + ctz32(h1);
    }
    return 64u;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let idx : u32 = gid.x;
    if (idx >= params.z) {
        return;
    }

    let nonce : u32 = params.y + idx;

    var state : array<u32, 16>;
    for (var i : u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    for (var round : u32 = 0u; round < 8u; round = round + 1u) {
        // Group 0: state[0..3]
        state[0] = state[0] + state[1];
        state[3] = rotl32(state[3] ^ state[0], 16u);
        state[2] = state[2] + state[3];
        state[1] = rotl32(state[1] ^ state[2], 12u);
        state[0] = state[0] + state[1];
        state[3] = rotl32(state[3] ^ state[0], 8u);
        state[2] = state[2] + state[3];
        state[1] = rotl32(state[1] ^ state[2], 7u);

        // Group 1: state[4..7]
        state[4] = state[4] + state[5];
        state[7] = rotl32(state[7] ^ state[4], 16u);
        state[6] = state[6] + state[7];
        state[5] = rotl32(state[5] ^ state[6], 12u);
        state[4] = state[4] + state[5];
        state[7] = rotl32(state[7] ^ state[4], 8u);
        state[6] = state[6] + state[7];
        state[5] = rotl32(state[5] ^ state[6], 7u);

        // Group 2: state[8..11]
        state[8] = state[8] + state[9];
        state[11] = rotl32(state[11] ^ state[8], 16u);
        state[10] = state[10] + state[11];
        state[9] = rotl32(state[9] ^ state[10], 12u);
        state[8] = state[8] + state[9];
        state[11] = rotl32(state[11] ^ state[8], 8u);
        state[10] = state[10] + state[11];
        state[9] = rotl32(state[9] ^ state[10], 7u);

        // Group 3: state[12..15]
        state[12] = state[12] + state[13];
        state[15] = rotl32(state[15] ^ state[12], 16u);
        state[14] = state[14] + state[15];
        state[13] = rotl32(state[13] ^ state[14], 12u);
        state[12] = state[12] + state[13];
        state[15] = rotl32(state[15] ^ state[12], 8u);
        state[14] = state[14] + state[15];
        state[13] = rotl32(state[13] ^ state[14], 7u);
    }

    var h0 : u32 = 0u;
    var h1 : u32 = 0u;
    for (var j : u32 = 0u; j < 16u; j = j + 2u) {
        h0 = h0 ^ state[j];
        h1 = h1 ^ state[j + 1u];
    }

    let tz : u32 = trailingZeros64(h0, h1);
    if (tz >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
