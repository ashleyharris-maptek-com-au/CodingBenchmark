Here's a complete Python implementation of a solver for 3-SAT using a combination of DPLL (for small instances) and WalkSAT (for larger ones). It reads from standard input and outputs a satisfying assignment guaranteed to satisfy all clauses.

```python
import sys
import random
import time
from array import array

USE_LIST_THRESHOLD = 200000
DPLL_MAX_VARS = 32
DPLL_MAX_CLAUSES = 180
GLOBAL_TIME_LIMIT = 295.0
MAX_WALKSAT_FLIPS = 20_000_000
WALKSAT_NOISE = 0.45
TIME_CHECK_INTERVAL = 4096


def read_ints():
    for line in sys.stdin.buffer:
        for num in line.split():
            yield int(num)


def read_instance():
    ints = read_ints()
    try:
        n = next(ints)
        m = next(ints)
    except StopIteration:
        return 0, 0, [], [[]], True
    clauses = []
    max_var = 0
    for _ in range(m):
        try:
            a = next(ints)
            b = next(ints)
            c = next(ints)
        except StopIteration:
            break
        clause = (a, b, c)
        clauses.append(clause)
        max_var = max(max_var, abs(a), abs(b), abs(c))
    m = len(clauses)
    if max_var > n:
        n = max_var
    occ_is_list = n <= USE_LIST_THRESHOLD
    if occ_is_list:
        occurrences = [[] for _ in range(n + 1)]
        for idx, clause in enumerate(clauses):
            for lit in clause:
                occurrences[abs(lit)].append(idx)
    else:
        occurrences = {}
        for idx, clause in enumerate(clauses):
            for lit in clause:
                var = abs(lit)
                occurrences.setdefault(var, []).append(idx)
    return n, m, clauses, occurrences, occ_is_list


def clause_value(clause, assignment):
    for lit in clause:
        var = abs(lit)
        val = assignment[var]
        if lit > 0:
            if val:
                return True
        else:
            if not val:
                return True
    return False


def clause_value_after_flip(clause, var, assignment):
    flipped = assignment[var] ^ 1
    for lit in clause:
        v = abs(lit)
        val = flipped if v == var else assignment[v]
        if lit > 0:
            if val:
                return True
        else:
            if not val:
                return True
    return False


def randomize_assignment(assignment, rng):
    n = len(assignment) - 1
    i = 1
    while i <= n:
        bits = rng.getrandbits(64)
        for _ in range(64):
            if i > n:
                break
            assignment[i] = bits & 1
            bits >>= 1
            i += 1


def initialize_state(assignment, clauses, clause_sat, unsat_list, unsat_pos):
    unsat_list.clear()
    for idx, clause in enumerate(clauses):
        sat = clause_value(clause, assignment)
        clause_sat[idx] = 1 if sat else 0
        if sat:
            unsat_pos[idx] = -1
        else:
            unsat_pos[idx] = len(unsat_list)
            unsat_list.append(idx)


def add_unsat(idx, unsat_list, unsat_pos):
    if unsat_pos[idx] != -1:
        return
    unsat_pos[idx] = len(unsat_list)
    unsat_list.append(idx)


def remove_unsat(idx, unsat_list, unsat_pos):
    pos = unsat_pos[idx]
    if pos == -1:
        return
    last = unsat_list.pop()
    unsat_pos[idx] = -1
    if last != idx:
        unsat_list[pos] = last
        unsat_pos[last] = pos


def flip_variable(var, assignment, clauses, occurrences, occ_is_list,
                  clause_sat, unsat_list, unsat_pos):
    assignment[var] ^= 1
    if occ_is_list:
        related = occurrences[var]
    else:
        related = occurrences.get(var)
        if not related:
            return
    for ci in related:
        prev = clause_sat[ci]
        sat = clause_value(clauses[ci], assignment)
        clause_sat[ci] = 1 if sat else 0
        if sat:
            if prev == 0:
                remove_unsat(ci, unsat_list, unsat_pos)
        else:
            if prev == 1:
                add_unsat(ci, unsat_list, unsat_pos)


def compute_break_count(var, occurrences, occ_is_list,
                        clause_sat, clauses, assignment):
    if occ_is_list:
        related = occurrences[var]
    else:
        related = occurrences.get(var)
        if not related:
            return 0
    count = 0
    for ci in related:
        if clause_sat[ci] and not clause_value_after_flip(clauses[ci], var, assignment):
            count += 1
    return count


def choose_best_variable(clause, occurrences, occ_is_list,
                         clause_sat, clauses, assignment, rng):
    best_vars = []
    best_break = None
    seen = []
    for lit in clause:
        var = abs(lit)
        if var in seen:
            continue
        seen.append(var)
        b = compute_break_count(var, occurrences, occ_is_list, clause_sat, clauses, assignment)
        if best_break is None or b < best_break:
            best_break = b
            best_vars = [var]
        elif b == best_break:
            best_vars.append(var)
    if not best_vars:
        return abs(clause[0])
    return rng.choice(best_vars)


def walksat(n, clauses, occurrences, occ_is_list, time_budget):
    m = len(clauses)
    if m == 0:
        return bytearray(n + 1)
    rng = random.Random()
    assignment = bytearray(n + 1)
    assignment[0] = 0
    clause_sat = bytearray(m)
    unsat_list = []
    unsat_pos = array('i', (-1 for _ in range(m)))
    base_flips = max(2000, 4 * m, 10 * n)
    if base_flips > MAX_WALKSAT_FLIPS:
        base_flips = MAX_WALKSAT_FLIPS
    start = time.time()
    restarts = 0
    while True:
        if time.time() - start >= time_budget:
            break
        restarts += 1
        randomize_assignment(assignment, rng)
        initialize_state(assignment, clauses, clause_sat, unsat_list, unsat_pos)
        if not unsat_list:
            return assignment
        growth = 1 + restarts // 4
        flips_this = base_flips * growth
        if flips_this > MAX_WALKSAT_FLIPS:
            flips_this = MAX_WALKSAT_FLIPS
        checks = 0
        for _ in range(flips_this):
            if not unsat_list:
                return assignment
            checks += 1
            if checks >= TIME_CHECK_INTERVAL:
                checks = 0
                if time.time() - start >= time_budget:
                    break
            clause_idx = unsat_list[rng.randrange(len(unsat_list))]
            clause = clauses[clause_idx]
            if rng.random() < WALKSAT_NOISE:
                var_to_flip = abs(clause[rng.randrange(len(clause))])
            else:
                var_to_flip = choose_best_variable(
                    clause, occurrences, occ_is_list, clause_sat, clauses, assignment, rng
                )
            flip_variable(var_to_flip, assignment, clauses, occurrences,
                          occ_is_list, clause_sat, unsat_list, unsat_pos)
    return None


def simplify_clauses(clauses, var, value):
    new_clauses = []
    for clause in clauses:
        satisfied = False
        new_clause = []
        for lit in clause:
            v = abs(lit)
            if v == var:
                if (lit > 0) == value:
                    satisfied = True
                    break
                else:
                    continue
            else:
                new_clause.append(lit)
        if satisfied:
            continue
        if not new_clause:
            return None
        new_clauses.append(tuple(new_clause))
    return new_clauses


def assign_and_simplify(clauses, assignment, unassigned, var, value):
    current = assignment[var]
    if current is not None:
        if current != value:
            return None
        return clauses, assignment, unassigned
    new_clauses = simplify_clauses(clauses, var, value)
    if new_clauses is None:
        return None
    assignment[var] = value
    unassigned.discard(var)
    return new_clauses, assignment, unassigned


def propagate(clauses, assignment, unassigned, deadline):
    while True:
        if time.time() > deadline:
            return None
        unit_literal = None
        for clause in clauses:
            if len(clause) == 0:
                return None
            if len(clause) == 1:
                unit_literal = clause[0]
                break
        if unit_literal is not None:
            var = abs(unit_literal)
            value = unit_literal > 0
            res = assign_and_simplify(clauses, assignment, unassigned, var, value)
            if res is None:
                return None
            clauses, assignment, unassigned = res
            continue
        literal_state = {}
        for clause in clauses:
            for lit in clause:
                var = abs(lit)
                if assignment[var] is not None:
                    continue
                sign = 1 if lit > 0 else -1
                prev = literal_state.get(var)
                if prev is None:
                    literal_state[var] = sign
                elif prev != sign:
                    literal_state[var] = 0
        applied = False
        for var, state in literal_state.items():
            if state == 0 or assignment[var] is not None:
                continue
            value = state == 1
            res = assign_and_simplify(clauses, assignment, unassigned, var, value)
            if res is None:
                return None
            clauses, assignment, unassigned = res
            applied = True
            break
        if not applied:
            break
    return clauses, assignment, unassigned


def select_branch_var(clauses, assignment):
    freq = {}
    balance = {}
    for clause in clauses:
        for lit in clause:
            var = abs(lit)
            if assignment[var] is not None:
                continue
            freq[var] = freq.get(var, 0) + 1
            balance[var] = balance.get(var, 0) + (1 if lit > 0 else -1)
    if not freq:
        return None, True
    best_var = max(freq.items(), key=lambda x: x[1])[0]
    prefer_true = balance.get(best_var, 0) >= 0
    return best_var, prefer_true


def dpll_recursive(clauses, assignment, unassigned, deadline):
    if time.time() > deadline:
        return None
    propagated = propagate(clauses, assignment, unassigned, deadline)
    if propagated is None:
        return None
    clauses, assignment, unassigned = propagated
    if not clauses:
        return assignment
    var, prefer_true = select_branch_var(clauses, assignment)
    if var is None:
        return assignment
    order = (prefer_true, not prefer_true)
    for value in order:
        new_assignment = assignment[:]
        new_unassigned = unassigned.copy()
        new_unassigned.discard(var)
        new_assignment[var] = value
        simplified = simplify_clauses(clauses, var, value)
        if simplified is None:
            continue
        result = dpll_recursive(simplified, new_assignment, new_unassigned, deadline)
        if result is not None:
            return result
    return None


def solve_with_dpll(n, clauses, time_limit):
    clause_lists = [tuple(c) for c in clauses]
    assignment = [None] * (n + 1)
    unassigned = set(range(1, n + 1))
    deadline = time.time() + max(0.2, time_limit)
    result = dpll_recursive(clause_lists, assignment, unassigned, deadline)
    if result is None:
        return None
    arr = bytearray(n + 1)
    for i in range(1, n + 1):
        arr[i] = 1 if result[i] else 0
    return arr


def verify_assignment(assignment, clauses):
    for clause in clauses:
        if not clause_value(clause, assignment):
            return False
    return True


def print_assignment(assignment):
    out = sys.stdout
    out.write("SAT\n")
    n = len(assignment) - 1
    if n <= 0:
        out.write("\n")
        return
    out.write('1' if assignment[1] else '0')
    for i in range(2, n + 1):
        out.write(' 1' if assignment[i] else ' 0')
    out.write('\n')


def main():
    sys.setrecursionlimit(1_000_000)
    start_time = time.time()
    n, m, clauses, occurrences, occ_is_list = read_instance()
    if n == 0:
        print("SAT")
        print()
        return
    if m == 0:
        assignment = bytearray(n + 1)
        print_assignment(assignment)
        return
    solution = None
    if (n <= DPLL_MAX_VARS) or (m <= DPLL_MAX_CLAUSES):
        dpll_time = min(5.0, GLOBAL_TIME_LIMIT * 0.15)
        dpll_time = max(0.5, dpll_time)
        candidate = solve_with_dpll(n, clauses, dpll_time)
        if candidate is not None and verify_assignment(candidate, clauses):
            solution = candidate
    if solution is None:
        elapsed = time.time() - start_time
        remaining = max(1.0, GLOBAL_TIME_LIMIT - elapsed)
        candidate = walksat(n, clauses, occurrences, occ_is_list, remaining)
        if candidate is not None and verify_assignment(candidate, clauses):
            solution = candidate
    if solution is None:
        elapsed = time.time() - start_time
        remaining = GLOBAL_TIME_LIMIT - elapsed
        if remaining > 0.5:
            candidate = solve_with_dpll(n, clauses, min(remaining, 5.0))
            if candidate is not None and verify_assignment(candidate, clauses):
                solution = candidate
    if solution is None:
        raise SystemExit("Failed to find a satisfying assignment in time")
    print_assignment(solution)


if __name__ == "__main__":
    main()
```

This code reads input, builds clauses and variable occurrences, tries DPLL for small instances, and WalkSAT for larger cases. It outputs "SAT" followed by a satisfying assignment as 1s and 0s.
