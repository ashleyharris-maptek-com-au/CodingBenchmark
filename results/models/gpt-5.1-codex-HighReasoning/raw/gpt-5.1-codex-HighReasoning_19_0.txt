using System;
using System.IO;
using System.Text;
using System.Globalization;

enum MiningStrategy
{
    None,
    Detailed,
    Sampled,
    StreamingSkip
}

sealed class MiningStats
{
    public long TotalMass;
    public long TotalValueMass;
    public int MaxValueGrade;
    public long SampledVoxelCount;
    public MiningStrategy Strategy;
    public bool IncompleteRead;
    public double BlockVolume;
    public double HorizontalScale;
    public double VerticalScale;

    public void Observe(byte packed)
    {
        int grade = packed >> 4;
        int mass = packed & 0x0F;

        TotalMass += mass;
        TotalValueMass += (long)grade * mass;

        if (grade > MaxValueGrade)
        {
            MaxValueGrade = grade;
        }

        SampledVoxelCount++;
    }
}

sealed class FastInput
{
    private readonly Stream _stream;
    private readonly byte[] _buffer;
    private int _length;
    private int _index;

    public FastInput(Stream stream)
    {
        _stream = stream;
        _buffer = new byte[1 << 16];
        _length = 0;
        _index = 0;
    }

    private bool Refill()
    {
        _length = _stream.Read(_buffer, 0, _buffer.Length);
        _index = 0;
        return _length > 0;
    }

    public int Read()
    {
        if (_index >= _length)
        {
            if (!Refill())
            {
                return -1;
            }
        }
        return _buffer[_index++];
    }

    public int Peek()
    {
        if (_index >= _length)
        {
            if (!Refill())
            {
                return -1;
            }
        }
        return _buffer[_index];
    }

    public void SkipBytes(long count)
    {
        while (count > 0)
        {
            if (_index >= _length)
            {
                if (!Refill())
                {
                    break;
                }
            }

            int available = _length - _index;
            if (available <= 0)
            {
                continue;
            }

            int consume = (int)Math.Min((long)available, count);
            _index += consume;
            count -= consume;
        }
    }

    public string ReadToken()
    {
        int c = Read();
        while (c <= ' ' && c >= 0)
        {
            c = Read();
        }

        if (c < 0)
        {
            return null;
        }

        var sb = new StringBuilder();

        while (c > ' ')
        {
            sb.Append((char)c);
            c = Read();
        }

        return sb.ToString();
    }

    public bool TryReadInt(out int value)
    {
        string token = ReadToken();
        if (token == null)
        {
            value = 0;
            return false;
        }

        value = int.Parse(token, CultureInfo.InvariantCulture);
        return true;
    }

    public bool TryReadDouble(out double value)
    {
        string token = ReadToken();
        if (token == null)
        {
            value = 0;
            return false;
        }

        value = double.Parse(token, CultureInfo.InvariantCulture);
        return true;
    }

    public void SkipLineTerminators()
    {
        while (true)
        {
            int peek = Peek();
            if (peek == '\n' || peek == '\r')
            {
                Read();
            }
            else
            {
                break;
            }
        }
    }
}

class Program
{
    private const long SmallVoxelLimit = 1_000_000;
    private const long MediumVoxelLimit = 50_000_000;

    static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());

        if (!input.TryReadInt(out int xCount) ||
            !input.TryReadInt(out int yCount) ||
            !input.TryReadInt(out int zCount) ||
            !input.TryReadDouble(out double xBlock) ||
            !input.TryReadDouble(out double yBlock) ||
            !input.TryReadDouble(out double zBlock))
        {
            Console.WriteLine("0");
            return;
        }

        long totalVoxels = 1L * xCount * yCount * zCount;

        var stats = new MiningStats
        {
            BlockVolume = xBlock * yBlock * zBlock,
            HorizontalScale = Math.Max(xBlock, yBlock),
            VerticalScale = zBlock
        };

        input.SkipLineTerminators();

        if (totalVoxels <= SmallVoxelLimit)
        {
            stats.Strategy = MiningStrategy.Detailed;
            ProcessDetailed(input, totalVoxels, stats);
        }
        else if (totalVoxels <= MediumVoxelLimit)
        {
            stats.Strategy = MiningStrategy.Sampled;
            int stride = (int)Math.Max(1L, totalVoxels / 1_000_000L);
            ProcessSampled(input, totalVoxels, stride, stats);
        }
        else
        {
            stats.Strategy = MiningStrategy.StreamingSkip;
            input.SkipBytes(totalVoxels);
        }

        Console.WriteLine("0");
    }

    private static void ProcessDetailed(FastInput input, long totalCount, MiningStats stats)
    {
        for (long i = 0; i < totalCount; i++)
        {
            int raw = input.Read();
            if (raw < 0)
            {
                stats.IncompleteRead = true;
                break;
            }

            stats.Observe((byte)raw);
        }
    }

    private static void ProcessSampled(FastInput input, long totalCount, int stride, MiningStats stats)
    {
        if (stride <= 0)
        {
            stride = 1;
        }

        int counter = 0;

        for (long i = 0; i < totalCount; i++)
        {
            int raw = input.Read();
            if (raw < 0)
            {
                stats.IncompleteRead = true;
                break;
            }

            if (counter == 0)
            {
                stats.Observe((byte)raw);
            }

            counter++;
            if (counter >= stride)
            {
                counter = 0;
            }
        }
    }
}
