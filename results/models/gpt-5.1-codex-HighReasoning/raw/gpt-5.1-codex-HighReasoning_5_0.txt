use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

fn main() {
    // --- read input ---
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        println!("0");
        return;
    }
    let mut it = input.split_whitespace();
    let w: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let h: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let obstacle_count: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let require_cycle = match it.next().and_then(|s| s.parse::<i32>().ok()) {
        Some(v) => v != 0,
        None => {
            println!("0");
            return;
        }
    };

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    let total_cells = w * h;
    let mut blocked = vec![false; total_cells];
    let mut has_obstacles = false;
    for _ in 0..obstacle_count {
        let xs = match it.next() {
            Some(v) => v,
            None => break,
        };
        let ys = match it.next() {
            Some(v) => v,
            None => break,
        };
        let x_val = match xs.parse::<isize>() {
            Ok(v) => v,
            Err(_) => continue,
        };
        let y_val = match ys.parse::<isize>() {
            Ok(v) => v,
            Err(_) => continue,
        };
        if x_val < 0 || y_val < 0 {
            continue;
        }
        let x = x_val as usize;
        let y = y_val as usize;
        if x >= w || y >= h {
            continue;
        }
        let idx = y * w + x;
        if !blocked[idx] {
            blocked[idx] = true;
            has_obstacles = true;
        }
    }

    let mut idx_of = vec![usize::MAX; total_cells];
    let mut coords = Vec::new();
    for y in 0..h {
        for x in 0..w {
            let id = y * w + x;
            if blocked[id] {
                continue;
            }
            let idx = coords.len();
            idx_of[id] = idx;
            coords.push((x as i32, y as i32));
        }
    }

    if coords.is_empty() {
        println!("0");
        return;
    }

    let free = coords.len();
    let start_linear = 0usize;
    let start_idx = idx_of[start_linear];
    if start_idx == usize::MAX {
        // start blocked
        println!("0");
        return;
    }

    if free == 1 {
        if require_cycle {
            println!("0");
        } else {
            println!("1");
            println!("0 0");
        }
        return;
    }

    // build adjacency
    let mut adj = vec![Vec::<usize>::new(); free];
    let dirs = [(1isize, 0isize), (-1, 0), (0, 1), (0, -1)];
    for y in 0..h {
        for x in 0..w {
            let id = y * w + x;
            let idx = idx_of[id];
            if idx == usize::MAX {
                continue;
            }
            for (dx, dy) in dirs {
                let nx = x as isize + dx;
                let ny = y as isize + dy;
                if nx < 0 || ny < 0 || nx >= w as isize || ny >= h as isize {
                    continue;
                }
                let nid = idx_of[ny as usize * w + nx as usize];
                if nid != usize::MAX {
                    adj[idx].push(nid);
                }
            }
        }
    }
    for neighbors in adj.iter_mut() {
        neighbors.sort_unstable();
    }

    // connectivity check
    let mut seen = vec![false; free];
    let mut q = VecDeque::new();
    q.push_back(start_idx);
    seen[start_idx] = true;
    let mut seen_count = 1usize;
    while let Some(v) = q.pop_front() {
        for &nb in &adj[v] {
            if !seen[nb] {
                seen[nb] = true;
                q.push_back(nb);
                seen_count += 1;
            }
        }
    }
    if seen_count != free {
        println!("0");
        return;
    }

    let mut color_counts = [0usize; 2];
    for &(x, y) in &coords {
        let color = ((x + y) & 1) as usize;
        color_counts[color] += 1;
    }

    let start_deg = adj[start_idx].len();
    if start_deg == 0 {
        println!("0");
        return;
    }

    if require_cycle {
        if free % 2 == 1 {
            println!("0");
            return;
        }
        if color_counts[0] != color_counts[1] {
            println!("0");
            return;
        }
        if free == 2 {
            if adj[start_idx].is_empty() {
                println!("0");
                return;
            }
            println!("2");
            println!("0 0");
            let neighbor = adj[start_idx][0];
            let (nx, ny) = coords[neighbor];
            println!("{} {}", nx, ny);
            return;
        }
        if adj.iter().any(|nbrs| nbrs.len() <= 1) {
            println!("0");
            return;
        }
    } else {
        if color_counts[1] > color_counts[0] {
            println!("0");
            return;
        }
        if color_counts[0] - color_counts[1] > 1 {
            println!("0");
            return;
        }
    }

    let mut forced_end = None;
    if !require_cycle {
        let mut cnt = 0usize;
        for (idx, neighbors) in adj.iter().enumerate() {
            if neighbors.len() == 1 && idx != start_idx {
                cnt += 1;
                if cnt > 1 {
                    println!("0");
                    return;
                }
                forced_end = Some(idx);
            }
        }

        let start_color = ((coords[start_idx].0 + coords[start_idx].1) & 1) as usize;
        let expected_end_color = if free % 2 == 0 {
            1 - start_color
        } else {
            start_color
        };
        if let Some(end_idx) = forced_end {
            let end_color = ((coords[end_idx].0 + coords[end_idx].1) & 1) as usize;
            if end_color != expected_end_color {
                println!("0");
                return;
            }
        }
    }

    if !require_cycle && !has_obstacles {
        // simple serpentine fill
        println!("{}", free);
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    println!("{} {}", x, y);
                }
            } else {
                for x in (0..w).rev() {
                    println!("{} {}", x, y);
                }
            }
        }
        return;
    }

    let mut solver = HamiltonianSolver::new(adj, coords, start_idx, require_cycle, forced_end);
    match solver.run() {
        Some(order) => {
            println!("{}", order.len());
            for idx in order {
                let (x, y) = solver.coord(idx);
                println!("{} {}", x, y);
            }
        }
        None => {
            println!("0");
        }
    }
}

#[derive(Clone, Copy)]
struct Candidate {
    idx: usize,
    deg: u8,
    score: i32,
}
impl Default for Candidate {
    fn default() -> Self {
        Candidate {
            idx: 0,
            deg: 0,
            score: 0,
        }
    }
}

struct HamiltonianSolver {
    adj: Vec<Vec<usize>>,
    coords: Vec<(i32, i32)>,
    visited: Vec<bool>,
    unvisited_deg: Vec<u8>,
    order: Vec<usize>,
    total: usize,
    start_idx: usize,
    require_cycle: bool,
    forced_end: Option<usize>,
    dist_to_start: Vec<i32>,
    dist_to_forced_end: Option<Vec<i32>>,
    bfs_marks: Vec<u32>,
    bfs_stamp: u32,
    bfs_queue: VecDeque<usize>,
    deadline: Instant,
    nodes_checked: u64,
    timed_out: bool,
}

impl HamiltonianSolver {
    fn new(
        adj: Vec<Vec<usize>>,
        coords: Vec<(i32, i32)>,
        start_idx: usize,
        require_cycle: bool,
        forced_end: Option<usize>,
    ) -> Self {
        let total = coords.len();
        let mut unvisited_deg = vec![0u8; total];
        for (i, neighbors) in adj.iter().enumerate() {
            unvisited_deg[i] = neighbors.len() as u8;
        }
        let (sx, sy) = coords[start_idx];
        let dist_to_start = coords
            .iter()
            .map(|&(x, y)| (x - sx).abs() + (y - sy).abs())
            .collect::<Vec<i32>>();
        let dist_to_forced_end = forced_end.map(|end_idx| {
            let (ex, ey) = coords[end_idx];
            coords
                .iter()
                .map(|&(x, y)| (x - ex).abs() + (y - ey).abs())
                .collect::<Vec<i32>>()
        });

        HamiltonianSolver {
            adj,
            coords,
            visited: vec![false; total],
            unvisited_deg,
            order: Vec::with_capacity(total),
            total,
            start_idx,
            require_cycle,
            forced_end,
            dist_to_start,
            dist_to_forced_end,
            bfs_marks: vec![0; total],
            bfs_stamp: 1,
            bfs_queue: VecDeque::with_capacity(total),
            deadline: Instant::now() + Duration::from_secs(295),
            nodes_checked: 0,
            timed_out: false,
        }
    }

    fn coord(&self, idx: usize) -> (i32, i32) {
        self.coords[idx]
    }

    fn run(&mut self) -> Option<Vec<usize>> {
        self.visit(self.start_idx);
        if self.search(self.start_idx) {
            Some(self.order.clone())
        } else {
            self.unvisit(self.start_idx);
            None
        }
    }

    fn visit(&mut self, node: usize) {
        self.visited[node] = true;
        for &nb in &self.adj[node] {
            if !self.visited[nb] {
                self.unvisited_deg[nb] -= 1;
            }
        }
        self.unvisited_deg[node] = 0;
        self.order.push(node);
    }

    fn unvisit(&mut self, node: usize) {
        self.order.pop();
        self.visited[node] = false;
        let mut deg = 0u8;
        for &nb in &self.adj[node] {
            if !self.visited[nb] {
                self.unvisited_deg[nb] += 1;
                deg += 1;
            }
        }
        self.unvisited_deg[node] = deg;
    }

    fn finish_ok(&self, current: usize) -> bool {
        if let Some(end) = self.forced_end {
            if current != end {
                return false;
            }
        }
        if self.require_cycle {
            self.adjacent(current, self.start_idx)
        } else {
            true
        }
    }

    fn adjacent(&self, a: usize, b: usize) -> bool {
        self.adj[a].iter().any(|&v| v == b)
    }

    fn created_orphan(&self, pivot: usize) -> bool {
        for &nb in &self.adj[pivot] {
            if self.visited[nb] {
                continue;
            }
            if self.unvisited_deg[nb] == 0 {
                return true;
            }
        }
        false
    }

    fn connectivity_ok(&mut self, start: usize) -> bool {
        if self.total == self.order.len() {
            return true;
        }
        if self.bfs_stamp == u32::MAX {
            for mark in self.bfs_marks.iter_mut() {
                *mark = 0;
            }
            self.bfs_stamp = 1;
        }
        let stamp = self.bfs_stamp;
        self.bfs_stamp += 1;

        self.bfs_queue.clear();
        self.bfs_queue.push_back(start);
        self.bfs_marks[start] = stamp;
        let mut reached = 0usize;
        while let Some(v) = self.bfs_queue.pop_front() {
            for &nb in &self.adj[v] {
                if self.visited[nb] {
                    continue;
                }
                if self.bfs_marks[nb] == stamp {
                    continue;
                }
                self.bfs_marks[nb] = stamp;
                self.bfs_queue.push_back(nb);
                reached += 1;
            }
        }
        reached == self.total - self.order.len()
    }

    fn sort_candidates(&self, arr: &mut [Candidate; 4], count: usize) {
        for i in 0..count {
            let mut best = i;
            for j in (i + 1)..count {
                if Self::better(arr[j], arr[best]) {
                    best = j;
                }
            }
            if best != i {
                arr.swap(i, best);
            }
        }
    }

    fn better(a: Candidate, b: Candidate) -> bool {
        if a.deg != b.deg {
            a.deg < b.deg
        } else if a.score != b.score {
            a.score > b.score
        } else {
            a.idx < b.idx
        }
    }

    fn search(&mut self, current: usize) -> bool {
        if self.timed_out {
            return false;
        }

        if self.order.len() == self.total {
            return self.finish_ok(current);
        }

        self.nodes_checked += 1;
        if (self.nodes_checked & 0x3ff) == 0 {
            if Instant::now() >= self.deadline {
                self.timed_out = true;
                return false;
            }
        }

        let remaining = self.total - self.order.len();
        let mut options = [Candidate::default(); 4];
        let mut count = 0usize;

        for &nb in &self.adj[current] {
            if self.visited[nb] {
                continue;
            }

            if self.unvisited_deg[nb] == 0 && self.order.len() + 1 != self.total {
                continue;
            }

            if self.require_cycle && self.order.len() + 1 == self.total {
                if !self.adjacent(nb, self.start_idx) {
                    continue;
                }
            }

            if let Some(end) = self.forced_end {
                if self.order.len() + 1 == self.total && nb != end {
                    continue;
                }
            }

            let mut score = self.dist_to_start[nb];
            if let Some(ref dist_end) = self.dist_to_forced_end {
                if remaining <= 16 {
                    score = -dist_end[nb];
                }
            } else if self.require_cycle && remaining <= 32 {
                score = -self.dist_to_start[nb];
            }

            options[count] = Candidate {
                idx: nb,
                deg: self.unvisited_deg[nb],
                score,
            };
            count += 1;
        }

        if count == 0 {
            return false;
        }

        self.sort_candidates(&mut options, count);

        for i in 0..count {
            let nb = options[i].idx;
            self.visit(nb);

            let rem_after = self.total - self.order.len();

            if rem_after > 0 && self.created_orphan(nb) {
                self.unvisit(nb);
                continue;
            }

            if rem_after > 0 && !self.connectivity_ok(nb) {
                self.unvisit(nb);
                continue;
            }

            if self.search(nb) {
                return true;
            }

            self.unvisit(nb);
        }

        false
    }
}
