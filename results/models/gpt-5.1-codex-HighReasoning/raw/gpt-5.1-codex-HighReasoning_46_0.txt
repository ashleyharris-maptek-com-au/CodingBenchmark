```glsl
#version 450

layout(local_size_x = 256) in;

// Each body occupies two consecutive vec4 entries:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;   // counts.x = number of bodies (N)
    vec4 physics;   // physics.x = dt, physics.y = softening, physics.z = G
};

void main()
{
    uint N   = counts.x;
    uint gid = gl_GlobalInvocationID.x;

    if (gid >= N)
        return;

    float dt        = physics.x;
    float softening = physics.y;
    float G         = physics.z;
    float soft2     = softening * softening;

    uint baseIndex  = gid * 2u;
    vec4 posMass    = inData[baseIndex];
    vec4 velData    = inData[baseIndex + 1u];

    vec3 pos  = posMass.xyz;
    float m   = posMass.w;
    vec3 vel  = velData.xyz;

    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == gid)
            continue;

        vec4 otherPosMass = inData[j * 2u];
        vec3 dir = otherPosMass.xyz - pos;

        float distSqr   = dot(dir, dir) + soft2;
        float invDist   = inversesqrt(distSqr);
        float invDist3  = invDist * invDist * invDist;
        float influence = G * otherPosMass.w * invDist3;

        acc += dir * influence;
    }

    vec3 newVel = vel + acc * dt;
    vec3 newPos = pos + newVel * dt;  // Semi-implicit Euler / leapfrog-style step

    outData[baseIndex]     = vec4(newPos, m);
    outData[baseIndex + 1] = vec4(newVel, 0.0);
}
```
