```csharp
using System;
using System.Text;
using System.Diagnostics;

public class Program
{
    const int FullLocalSearchThreshold = 600;
    const int RandomAttemptsPerN = 40;
    const int RandomSearchTimeMs = 4000;

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n == int.MinValue)
            return;
        if (n <= 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        int[][] flow = new int[n][];
        int[][] dist = new int[n][];
        long[] flowRow = new long[n];
        long[] flowCol = new long[n];

        for (int i = 0; i < n; i++)
        {
            flow[i] = new int[n];
            for (int j = 0; j < n; j++)
            {
                int val = scanner.NextInt();
                flow[i][j] = val;
                flowRow[i] += val;
                flowCol[j] += val;
            }
        }
        long[] facilityScore = new long[n];
        for (int i = 0; i < n; i++)
            facilityScore[i] = flowRow[i] + flowCol[i];

        long[] distRow = new long[n];
        long[] distCol = new long[n];
        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            for (int j = 0; j < n; j++)
            {
                int val = scanner.NextInt();
                dist[i][j] = val;
                distRow[i] += val;
                distCol[j] += val;
            }
        }
        long[] locationScore = new long[n];
        for (int i = 0; i < n; i++)
            locationScore[i] = distRow[i] + distCol[i];

        int[] facilityOrder = new int[n];
        int[] locationOrder = new int[n];
        for (int i = 0; i < n; i++)
        {
            facilityOrder[i] = i;
            locationOrder[i] = i;
        }
        Array.Sort(facilityOrder, (a, b) => facilityScore[b].CompareTo(facilityScore[a]));
        Array.Sort(locationOrder, (a, b) => locationScore[a].CompareTo(locationScore[b]));

        int[] perm = new int[n];
        for (int i = 0; i < n; i++)
            perm[facilityOrder[i]] = locationOrder[i];

        long totalCost = ComputeTotalCost(perm, flow, dist);

        if (n <= FullLocalSearchThreshold)
        {
            FullLocalSearch(perm, flow, dist, ref totalCost);
        }
        else
        {
            int attempts = Math.Max(n * RandomAttemptsPerN, 20000);
            RandomLocalSearch(perm, flow, dist, ref totalCost, attempts, RandomSearchTimeMs);
        }

        var sb = new StringBuilder();
        sb.Append(totalCost).Append('\n');
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.Append('\n');
        Console.Write(sb.ToString());
    }

    static long ComputeTotalCost(int[] perm, int[][] flow, int[][] dist)
    {
        int n = perm.Length;
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            int[] flowRow = flow[i];
            int[] distRowPi = dist[pi];
            for (int j = 0; j < n; j++)
            {
                int pj = perm[j];
                total += (long)flowRow[j] * distRowPi[pj];
            }
        }
        return total;
    }

    static void FullLocalSearch(int[] perm, int[][] flow, int[][] dist, ref long totalCost)
    {
        int n = perm.Length;
        while (true)
        {
            bool improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    long delta = ComputeSwapDelta(i, j, perm, flow, dist);
                    if (delta < 0)
                    {
                        int tmp = perm[i];
                        perm[i] = perm[j];
                        perm[j] = tmp;
                        totalCost += delta;
                        improved = true;
                        goto EndIteration;
                    }
                }
            }
        EndIteration:
            if (!improved) break;
        }
    }

    static void RandomLocalSearch(int[] perm, int[][] flow, int[][] dist, ref long totalCost, int maxAttempts, int timeLimitMs)
    {
        int n = perm.Length;
        var rnd = new Random(1234567);
        var sw = Stopwatch.StartNew();
        for (int attempt = 0; attempt < maxAttempts; attempt++)
        {
            if (timeLimitMs > 0 && sw.ElapsedMilliseconds > timeLimitMs) break;
            int i = rnd.Next(n);
            int j = rnd.Next(n);
            if (i == j) continue;
            if (i > j)
            {
                int t = i;
                i = j;
                j = t;
            }
            long delta = ComputeSwapDelta(i, j, perm, flow, dist);
            if (delta < 0)
            {
                int tmp = perm[i];
                perm[i] = perm[j];
                perm[j] = tmp;
                totalCost += delta;
            }
        }
    }

    static long ComputeSwapDelta(int i, int j, int[] perm, int[][] flow, int[][] dist)
    {
        if (i == j) return 0;
        int pi = perm[i];
        int pj = perm[j];
        if (pi == pj) return 0;

        int[] flowRowI = flow[i];
        int[] flowRowJ = flow[j];
        int[] distRowPi = dist[pi];
        int[] distRowPj = dist[pj];
        long delta = 0;

        int n = perm.Length;
        for (int k = 0; k < n; k++)
        {
            if (k == i || k == j) continue;
            int pk = perm[k];
            int[] distRowPk = dist[pk];
            int[] flowRowK = flow[k];

            int dpipk = distRowPi[pk];
            int dpjpk = distRowPj[pk];
            int dpkpi = distRowPk[pi];
            int dpkpj = distRowPk[pj];

            delta += (long)flowRowI[k] * (dpjpk - dpipk);
            delta += (long)flowRowK[i] * (dpkpj - dpkpi);
            delta += (long)flowRowJ[k] * (dpipk - dpjpk);
            delta += (long)flowRowK[j] * (dpkpi - dpkpj);
        }

        delta += (long)flowRowI[i] * (distRowPj[pj] - distRowPi[pi]);
        delta += (long)flowRowJ[j] * (distRowPi[pi] - distRowPj[pj]);
        delta += (long)flowRowI[j] * (distRowPj[pi] - distRowPi[pj]);
        delta += (long)flowRowJ[i] * (distRowPi[pj] - distRowPj[pi]);

        return delta;
    }

    sealed class FastScanner
    {
        private readonly System.IO.Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr;
        private int len;

        public FastScanner(System.IO.Stream stream)
        {
            this.stream = stream;
        }

        private byte Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return 0;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            byte c = Read();
            while (c <= ' ')
            {
                if (c == 0) return int.MinValue;
                c = Read();
            }
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
```
