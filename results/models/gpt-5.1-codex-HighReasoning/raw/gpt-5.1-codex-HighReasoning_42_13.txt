%posB_xy = OpVectorShuffle %v2float %emit_posB %emit_posB 0 1

%delta_y = OpCompositeExtract %float %delta 1

%delta_y2 = OpFMul %float %delta_y %delta_y

%len = OpExtInst %float %glsl Sqrt %lenSqMax

%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

%offset_ndc = OpVectorTimesScalar %v2float %perp %float_0_008

%offset_clipA_xy = OpVectorTimesScalar %v2float %offset_ndc %wA
%offset_clipA_x = OpCompositeExtract %float %offset_clipA_xy 0
%offset_clipA_y = OpCompositeExtract %float %offset_clipA_xy 1
%offset_clipA_vec4 = OpCompositeConstruct %v4float %offset_clipA_x %offset_clipA_y %float_0 %float_0

%posA_minus = OpFSub %v4float %emit_posA %offset_clipA_vec4
%wB = OpCompositeExtract %float %emit_posB 3
%offset_clipB_xy = OpVectorTimesScalar %v2float %offset_ndc %wB
%offset_clipB_x = OpCompositeExtract %float %offset_clipB_xy 0
%offset_clipB_y = OpCompositeExtract %float %offset_clipB_xy 1

%posB_plus = OpFAdd %v4float %emit_posB %offset_clipB_vec4
%posB_minus = OpFSub %v4float %emit_posB %offset_clipB_vec4
OpStore %ptr_pos_out %posA_plus
OpStore %outColor %avgColor
