// order must be a power of two; returns Hilbert index of (x,y) on [0, order).
static uint64_t hilbertD(uint32_t order, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = order >> 1; s > 0; s >>= 1) {
        uint32_t rx = (x & s) ? 1u : 0u;
        uint32_t ry = (y & s) ? 1u : 0u;
        d += (uint64_t)s * (uint64_t)s * ((3u * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = s - 1 - x;
                y = s - 1 - y;
            }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

    static int ExactMatch(List<(int to, int eid, long w)>[] adj, List<int> oddVertices, List<long> edgeW, int nextEid, int N) {
        int K = oddVertices.Count;
        var dist = new long[K][];
        var parent = new int[K][];
        var parentEdge = new int[K][];
        for (int i = 0; i < K; i++) {
            (dist[i], parent[i], parentEdge[i]) = Dijkstra(adj, oddVertices[i], N);
        }

        int full = 1 << K;
        var dp = new long[full];
        var dpChoice = new int[full];
        for (int i = 0; i < full; i++) dp[i] = long.MaxValue;
        dp[0] = 0;

        for (int mask = 0; mask < full; mask++) {
            if (dp[mask] == long.MaxValue) continue;
            int i = 0;
            while (i < K && (mask & (1 << i)) != 0) i++;
            if (i == K) continue;
            long baseCost = dp[mask];
            var distI = dist[i];
            for (int j = i + 1; j < K; j++) {
                if ((mask & (1 << j)) != 0) continue;
                long d = distI[oddVertices[j]];
                if (d == long.MaxValue) continue;
                int newMask = mask | (1 << i) | (1 << j);
                long newCost = baseCost + d;
                if (newCost < dp[newMask]) {
                    dp[newMask] = newCost;
                    dpChoice[newMask] = (i << 16) | j;
                }
            }
        }

        int curMask = full - 1;
        while (curMask != 0) {
            int choice = dpChoice[curMask];
            int i = choice >> 16;
            int j = choice & 0xFFFF;
            int prevMask = curMask ^ ((1 << i) | (1 << j));
            int src = oddVertices[i];
            int dst = oddVertices[j];
            int cur = dst;
            int safety = 0;
            while (cur != src && cur != -1 && safety++ < N + 1) {
                int p = parent[i][cur];
                if (p == -1) break;
                int eid = parentEdge[i][cur];
                long w = edgeW[eid];
                int newEid = nextEid++;
                edgeW.Add(w);
                adj[p].Add((cur, newEid, w));
                adj[cur].Add((p, newEid, w));
                cur = p;
            }
            curMask = prevMask;
        }
        return nextEid;
    }

    static (long[], int[], int[]) Dijkstra(List<(int to, int eid, long w)>[] adj, int src, int N) {
        var dist = new long[N];
        var parent = new int[N];
        var parentEdge = new int[N];
        for (int i = 0; i < N; i++) {
            dist[i] = long.MaxValue;
            parent[i] = -1;
            parentEdge[i] = -1;
        }
        dist[src] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);
        while (pq.Count > 0) {
            int u;
            long d;
            pq.TryDequeue(out u, out d);
            if (d > dist[u]) continue;
            var au = adj[u];
            int auc = au.Count;
            for (int k = 0; k < auc; k++) {
                var edge = au[k];
                long nd = d + edge.w;
                if (nd < dist[edge.to]) {
                    dist[edge.to] = nd;
                    parent[edge.to] = u;
                    parentEdge[edge.to] = edge.eid;
                    pq.Enqueue(edge.to, nd);
                }
            }
        }
        return (dist, parent, parentEdge);
    }

                        parent[edge.to] = u;
                        parentEdge[edge.to] = edge.eid;
                        bfsQueue[qTail++] = edge.to;
                        order[orderLen++] = edge.to;
                    }
                }
            }
        }

        var oddCount = new int[N];
        for (int i = 0; i < N; i++) if (isOdd[i]) oddCount[i] = 1;
        for (int i = orderLen - 1; i >= 0; i--) {
            int v = order[i];
            if (parent[v] != -1) {

                adj[v].Add((u, newEid, w));
            }
        }
        return nextEid;
    }

struct BHTree {
    cx: Vec<f64>,
    cy: Vec<f64>,
    half: Vec<f64>,
    com_x: Vec<f64>,
    com_y: Vec<f64>,
    mass: Vec<f64>,

    point: Vec<i32>,
}

impl BHTree {
    fn new() -> Self {
        BHTree {
            cx: Vec::new(),
            cy: Vec::new(),
            half: Vec::new(),
            com_x: Vec::new(),
            com_y: Vec::new(),
            mass: Vec::new(),
            children: Vec::new(),

    fn clear(&mut self) {
        self.cx.clear();
        self.cy.clear();
        self.half.clear();
        self.com_x.clear();
        self.com_y.clear();
        self.mass.clear();
        self.children.clear();
        self.point.clear();

    #[inline]
    fn add_node(&mut self, cx: f64, cy: f64, half: f64) -> usize {
        let idx = self.cx.len();
        self.cx.push(cx);
        self.cy.push(cy);
        self.half.push(half);
        self.com_x.push(0.0);
        self.com_y.push(0.0);
        self.mass.push(0.0);
        self.children.push([-1i32; 4]);
        self.point.push(-1);
        idx
    }

    #[inline]
    fn quadrant(&self, node: usize, x: f64, y: f64) -> usize {
        let r = (x >= self.cx[node]) as usize;
        let t = (y >= self.cy[node]) as usize;
        r | (t << 1)
    }

    #[inline]
    fn quad_center(&self, node: usize, q: usize) -> (f64, f64, f64) {
        let h = self.half[node] * 0.5;
        let dx = if q & 1 != 0 { h } else { -h };
        let dy = if q & 2 != 0 { h } else { -h };
        (self.cx[node] + dx, self.cy[node] + dy, h)
    }

    fn build(&mut self, positions: &[(f64, f64)]) {
        self.clear();
        if positions.is_empty() {
            return;
        }

        let mut minx = positions[0].0;
        let mut maxx = positions[0].0;
        let mut miny = positions[0].1;
        let mut maxy = positions[0].1;
        for &(x, y) in positions.iter().skip(1) {
            if x < minx { minx = x; }
            if x > maxx { maxx = x; }
            if y < miny { miny = y; }
            if y > maxy { maxy = y; }
        }
        let cx = (minx + maxx) * 0.5;
        let cy = (miny + maxy) * 0.5;
        let half = ((maxx - minx).max(maxy - miny) * 0.5 + 1.0).max(1.0);

        self.add_node(cx, cy, half);
        for i in 0..positions.len() {
            self.insert_point(positions, i as i32);
        }
    }

    fn insert_point(&mut self, positions: &[(f64, f64)], point_idx: i32) {
        let (px, py) = positions[point_idx as usize];
        let mut node = 0usize;
        loop {
            // Empty node: store point and stop.
            if self.mass[node] == 0.0 {
                self.point[node] = point_idx;
                self.mass[node] = 1.0;
                self.com_x[node] = px;
                self.com_y[node] = py;
                return;
            }

            // Reached depth/precision limit: accumulate mass at this node.
            if self.half[node] < 1e-7 {
                let nm = self.mass[node] + 1.0;
                self.com_x[node] = (self.com_x[node] * self.mass[node] + px) / nm;
                self.com_y[node] = (self.com_y[node] * self.mass[node] + py) / nm;
                self.mass[node] = nm;
                return;
            }

            // Single-point leaf: subdivide.
            if self.point[node] != -1 {
                let existing = self.point[node];
                let (ex, ey) = positions[existing as usize];
                self.point[node] = -1;

                let q_ex = self.quadrant(node, ex, ey);
                let (ccx, ccy, ch) = self.quad_center(node, q_ex);
                let child = self.add_node(ccx, ccy, ch);
                self.children[node][q_ex] = child as i32;
                self.point[child] = existing;
                self.mass[child] = 1.0;
                self.com_x[child] = ex;
                self.com_y[child] = ey;
                // Fallthrough: node is now internal, insert px,py below.
            }

            // Internal node: update COM and descend.
            let nm = self.mass[node] + 1.0;
            self.com_x[node] = (self.com_x[node] * self.mass[node] + px) / nm;
            self.com_y[node] = (self.com_y[node] * self.mass[node] + py) / nm;
            self.mass[node] = nm;

            let q = self.quadrant(node, px, py);
            if self.children[node][q] == -1 {
                let (ccx, ccy, ch) = self.quad_center(node, q);
                let new_child = self.add_node(ccx, ccy, ch);
                self.children[node][q] = new_child as i32;
                self.point[new_child] = point_idx;
                self.mass[new_child] = 1.0;
                self.com_x[new_child] = px;
                self.com_y[new_child] = py;
                return;
            }
            node = self.children[node][q] as usize;
        }
    }

    fn force(&self, x: f64, y: f64, theta2: f64, k2: f64, stack: &mut Vec<i32>) -> (f64, f64) {
        let mut fx = 0.0;
        let mut fy = 0.0;
        stack.clear();
        stack.push(0);
        while let Some(node_i) = stack.pop() {
            let node = node_i as usize;
            let m = self.mass[node];
            if m == 0.0 {
                continue;
            }
            let dx = x - self.com_x[node];
            let dy = y - self.com_y[node];
            let d2 = dx * dx + dy * dy + 1e-9;
            let s = self.half[node] * 2.0;

            if self.point[node] != -1 || s * s < theta2 * d2 {
                // Repulsive force k^2 * m / d as a vector: dx * k^2 * m / d^2.
                let factor = k2 * m / d2;
                fx += dx * factor;
                fy += dy * factor;
            } else {
                let c = self.children[node];
                if c[0] != -1 { stack.push(c[0]); }
                if c[1] != -1 { stack.push(c[1]); }
                if c[2] != -1 { stack.push(c[2]); }
                if c[3] != -1 { stack.push(c[3]); }
            }
        }
        (fx, fy)
    }

    } else if n <= 500 {
        300
    } else if n <= 2000 {
        200
    } else if n <= 10000 {
        100
    } else if n <= 50000 {
        60
    } else {
        40

    let area = target_radius * target_radius;
    let k = (area / n as f64).sqrt();

    let final_t = k * 0.01;

            *d = (0.0, 0.0);

            for i in 0..n {
                let (fx, fy) = tree.force(positions[i].0, positions[i].1, theta2, k2, &mut stack);
                disp[i].0 += fx;
                disp[i].1 += fy;

            let v = vv as usize;

            let factor = d / k;
            let fx = dx * factor;
            let fy = dy * factor;
            disp[u].0 -= fx;
            disp[u].1 -= fy;
            disp[v].0 += fx;
            disp[v].1 += fy;

        // drifting to infinity.
        let gravity = 0.02 * k;

            let dx = positions[i].0;
            let dy = positions[i].1;
            let dd = (dx * dx + dy * dy).sqrt() + 1e-9;
            disp[i].0 -= dx / dd * gravity;
            disp[i].1 -= dy / dd * gravity;

        for i in 0..n {

            let dy = disp[i].1;
            let d = (dx * dx + dy * dy).sqrt();

                let nx = positions[i].0 + dx / d * step;
                let ny = positions[i].1 + dy / d * step;
                positions[i].0 = nx.max(-bound).min(bound);
                positions[i].1 = ny.max(-bound).min(bound);
            }
        }
    }

    // Guard against any non-finite coordinates.
    for i in 0..n {
        if !positions[i].0.is_finite() {
            positions[i].0 = 0.0;
        }
        if !positions[i].1.is_finite() {
            positions[i].1 = 0.0;
        }
    }

    for &(x, y) in &positions {
        writeln!(out, "{:.6} {:.6}", x, y).unwrap();
    }

    while (c >= '0' && c <= '9') { x = x * 10 + (c - '0'); c = gc(); }
    return sign * x;
}

static double readDouble() {
    int c, sign = 1;
    while ((c = gc()) != -1 && c <= ' ') ;
    if (c == -1) return 0;
    if (c == '-') { sign = -1; c = gc(); }
    else if (c == '+') c = gc();
    double x = 0;
    while (c >= '0' && c <= '9') { x = x * 10 + (c - '0'); c = gc(); }
    if (c == '.') {
        c = gc();
        double f = 0.1;
        while (c >= '0' && c <= '9') { x += (c - '0') * f; f *= 0.1; c = gc(); }
    }
    if (c == 'e' || c == 'E') {
        c = gc();
        int eSign = 1;
        if (c == '-') { eSign = -1; c = gc(); }
        else if (c == '+') c = gc();
        int e = 0;
        while (c >= '0' && c <= '9') { e = e * 10 + (c - '0'); c = gc(); }
        x *= std::pow(10.0, eSign * e);
    }
    return sign * x;
}

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double a, double b, double c) : x(a), y(b), z(c) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    double dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x};
    }
    double length() const { return std::sqrt(x*x + y*y + z*z); }
    Vec3 normalize() const {
        double l = length();
        return l > 0 ? *this * (1.0/l) : Vec3();
    }
};

static constexpr double EPS = 1e-6;

struct Plane {
    Vec3 normal;
    double w;
    Plane() : normal(), w(0) {}
    Plane(const Vec3& n, double w_) : normal(n), w(w_) {}
    static Plane fromPoints(const Vec3& a, const Vec3& b, const Vec3& c) {
        Vec3 n = (b - a).cross(c - a).normalize();
        return {n, n.dot(a)};
    }
    void flip() { normal = normal * -1; w = -w; }
};

struct Polygon {
    std::vector<Vec3> vertices;
    Plane plane;
    Polygon() = default;
    Polygon(std::vector<Vec3> verts, const Plane& p)
        : vertices(std::move(verts)), plane(p) {}
    Polygon(std::vector<Vec3> verts) : vertices(std::move(verts)) {
        if (vertices.size() >= 3)
            plane = Plane::fromPoints(vertices[0], vertices[1], vertices[2]);
    }
    void flip() {
        std::reverse(vertices.begin(), vertices.end());
        plane.flip();
    }
};

static void splitPolygon(const Plane& plane, const Polygon& poly,
                         std::vector<Polygon>& coplanarFront,
                         std::vector<Polygon>& coplanarBack,
                         std::vector<Polygon>& front,
                         std::vector<Polygon>& back) {
    const int COPLANAR = 0, FRONT_T = 1, BACK_T = 2, SPANNING = 3;
    int polyType = 0;
    size_t n = poly.vertices.size();
    std::vector<int> types(n);
    for (size_t i = 0; i < n; i++) {
        double t = plane.normal.dot(poly.vertices[i]) - plane.w;
        int type = (t < -EPS) ? BACK_T : (t > EPS) ? FRONT_T : COPLANAR;
        polyType |= type;
        types[i] = type;
    }
    switch (polyType) {
        case COPLANAR:
            if (plane.normal.dot(poly.plane.normal) > 0)
                coplanarFront.push_back(poly);
            else
                coplanarBack.push_back(poly);
            break;
        case FRONT_T:
            front.push_back(poly);
            break;
        case BACK_T:
            back.push_back(poly);
            break;
        case SPANNING: {
            std::vector<Vec3> f, b;
            f.reserve(n + 1);
            b.reserve(n + 1);
            for (size_t i = 0; i < n; i++) {
                size_t j = (i + 1) % n;
                int ti = types[i], tj = types[j];
                const Vec3& vi = poly.vertices[i];
                const Vec3& vj = poly.vertices[j];
                if (ti != BACK_T) f.push_back(vi);
                if (ti != FRONT_T) b.push_back(vi);
                if ((ti | tj) == SPANNING) {
                    double denom = plane.normal.dot(vj - vi);
                    if (std::abs(denom) > 1e-15) {
                        double t = (plane.w - plane.normal.dot(vi)) / denom;
                        Vec3 v = vi + (vj - vi) * t;
                        f.push_back(v);
                        b.push_back(v);
                    }
                }
            }
            if (f.size() >= 3) front.emplace_back(std::move(f), poly.plane);
            if (b.size() >= 3) back.emplace_back(std::move(b), poly.plane);
            break;
        }
    }
}

struct Node {
    Plane plane;
    std::unique_ptr<Node> front, back;
    std::vector<Polygon> polygons;
    bool hasPlane = false;

    void build(std::vector<Polygon> polys) {
        struct Task { Node* node; std::vector<Polygon> polys; };
        std::vector<Task> stack;
        stack.push_back({this, std::move(polys)});
        while (!stack.empty()) {
            Node* n = stack.back().node;
            std::vector<Polygon> ps = std::move(stack.back().polys);
            stack.pop_back();
            if (ps.empty()) continue;
            if (!n->hasPlane) {
                n->plane = ps[0].plane;
                n->hasPlane = true;
            }
            std::vector<Polygon> frontPolys, backPolys;
            for (const auto& p : ps) {
                splitPolygon(n->plane, p, n->polygons, n->polygons,
                             frontPolys, backPolys);
            }
            if (!frontPolys.empty()) {
                if (!n->front) n->front = std::make_unique<Node>();
                stack.push_back({n->front.get(), std::move(frontPolys)});
            }
            if (!backPolys.empty()) {
                if (!n->back) n->back = std::make_unique<Node>();
                stack.push_back({n->back.get(), std::move(backPolys)});
            }
        }
    }

    void invert() {
        std::vector<Node*> stack = {this};
        while (!stack.empty()) {
            Node* n = stack.back(); stack.pop_back();
            for (auto& p : n->polygons) p.flip();
            if (n->hasPlane) n->plane.flip();
            std::swap(n->front, n->back);
            if (n->front) stack.push_back(n->front.get());
            if (n->back) stack.push_back(n->back.get());
        }
    }

    std::vector<Polygon> clipPolygons(const std::vector<Polygon>& polys) const {
        std::vector<Polygon> result;
        struct Task { const Node* node; std::vector<Polygon> polys; };
        std::vector<Task> stack;
        stack.push_back({this, polys});
        while (!stack.empty()) {
            const Node* n = stack.back().node;
            std::vector<Polygon> ps = std::move(stack.back().polys);
            stack.pop_back();
            if (!n->hasPlane) {
                result.insert(result.end(),
                              std::make_move_iterator(ps.begin()),
                              std::make_move_iterator(ps.end()));
                continue;
            }
            std::vector<Polygon> frontPolys, backPolys;
            for (const auto& p : ps) {
                splitPolygon(n->plane, p, frontPolys, backPolys,
                             frontPolys, backPolys);
            }
            if (n->front) {
                stack.push_back({n->front.get(), std::move(frontPolys)});
            } else {
                result.insert(result.end(),
                              std::make_move_iterator(frontPolys.begin()),
                              std::make_move_iterator(frontPolys.end()));
            }
            if (n->back) {
                stack.push_back({n->back.get(), std::move(backPolys)});
            }
            // else drop backPolys (inside this BSP)
        }
        return result;
    }

    void clipTo(const Node& other) {
        std::vector<Node*> stack = {this};
        while (!stack.empty()) {
            Node* n = stack.back(); stack.pop_back();
            n->polygons = other.clipPolygons(n->polygons);
            if (n->front) stack.push_back(n->front.get());
            if (n->back) stack.push_back(n->back.get());
        }
    }

    std::vector<Polygon> allPolygons() const {
        std::vector<Polygon> result;
        std::vector<const Node*> stack = {this};
        while (!stack.empty()) {
            const Node* n = stack.back(); stack.pop_back();
            result.insert(result.end(), n->polygons.begin(), n->polygons.end());
            if (n->front) stack.push_back(n->front.get());
            if (n->back) stack.push_back(n->back.get());
        }
        return result;
    }
};

struct VKey {
    int64_t x, y, z;
    bool operator==(const VKey& o) const {
        return x == o.x && y == o.y && z == o.z;
    }
};
struct VKeyHash {
    size_t operator()(const VKey& k) const noexcept {
        size_t h = std::hash<int64_t>()(k.x);
        h ^= std::hash<int64_t>()(k.y) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        h ^= std::hash<int64_t>()(k.z) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        return h;
    }
};

    int VA = readInt(), FA = readInt(), VB = readInt(), FB = readInt();
    std::vector<Vec3> vertsA(VA), vertsB(VB);
    for (int i = 0; i < VA; i++) {
        vertsA[i].x = readDouble();

        facesA[i][2] = readInt();
    }

        vertsB[i].y = readDouble();
        vertsB[i].z = readDouble();

        facesB[i][0] = readInt();
        facesB[i][1] = readInt();
        facesB[i][2] = readInt();
    }

    // AABBs - quick disjoint test
    Vec3 minA(1e300, 1e300, 1e300), maxA(-1e300, -1e300, -1e300);
    Vec3 minB(1e300, 1e300, 1e300), maxB(-1e300, -1e300, -1e300);
    for (const auto& v : vertsA) {
        if (v.x < minA.x) minA.x = v.x; if (v.y < minA.y) minA.y = v.y; if (v.z < minA.z) minA.z = v.z;
        if (v.x > maxA.x) maxA.x = v.x; if (v.y > maxA.y) maxA.y = v.y; if (v.z > maxA.z) maxA.z = v.z;
    }
    for (const auto& v : vertsB) {
        if (v.x < minB.x) minB.x = v.x; if (v.y < minB.y) minB.y = v.y; if (v.z < minB.z) minB.z = v.z;
        if (v.x > maxB.x) maxB.x = v.x; if (v.y > maxB.y) maxB.y = v.y; if (v.z > maxB.z) maxB.z = v.z;
    }

    bool disjoint = (maxA.x < minB.x - EPS) || (maxB.x < minA.x - EPS) ||
                    (maxA.y < minB.y - EPS) || (maxB.y < minA.y - EPS) ||
                    (maxA.z < minB.z - EPS) || (maxB.z < minA.z - EPS);

    std::vector<Vec3> outVerts;
    std::vector<std::array<int,3>> outFaces;

    if (disjoint) {
        // Disjoint AABBs -> meshes don't overlap. Union is concatenation;
        // each component remains manifold/watertight per problem guarantee.
        outVerts = vertsA;
        outVerts.insert(outVerts.end(), vertsB.begin(), vertsB.end());
        outFaces.reserve(FA + FB);
        for (const auto& f : facesA)
            outFaces.push_back({f[0], f[1], f[2]});
        for (const auto& f : facesB)

        std::vector<Polygon> polysA, polysB;
        polysA.reserve(FA);
        polysB.reserve(FB);
        for (const auto& f : facesA) {
            std::vector<Vec3> v{vertsA[f[0]], vertsA[f[1]], vertsA[f[2]]};
            Vec3 n = (v[1] - v[0]).cross(v[2] - v[0]);
            if (n.length() < 1e-20) continue;
            polysA.emplace_back(std::move(v));
        }
        for (const auto& f : facesB) {
            std::vector<Vec3> v{vertsB[f[0]], vertsB[f[1]], vertsB[f[2]]};
            Vec3 n = (v[1] - v[0]).cross(v[2] - v[0]);
            if (n.length() < 1e-20) continue;
            polysB.emplace_back(std::move(v));
        }

        // Classic BSP CSG Union (csg.js / Naylor-Thibault style).
        Node a, b;
        a.build(std::move(polysA));
        b.build(std::move(polysB));
        a.clipTo(b);
        b.clipTo(a);
        b.invert();
        b.clipTo(a);
        b.invert();
        a.build(b.allPolygons());

        auto result = a.allPolygons();

        // Deduplicate vertices via spatial quantization.
        std::unordered_map<VKey, int, VKeyHash> vmap;
        vmap.reserve(result.size() * 3);
        auto getIdx = [&](const Vec3& v) -> int {
            VKey k{(int64_t)std::llround(v.x * 1e7),
                   (int64_t)std::llround(v.y * 1e7),
                   (int64_t)std::llround(v.z * 1e7)};
            auto it = vmap.find(k);
            if (it != vmap.end()) return it->second;
            int idx = (int)outVerts.size();
            outVerts.push_back(v);
            vmap.emplace(k, idx);
            return idx;
        };

        for (const auto& p : result) {
            if (p.vertices.size() < 3) continue;
            // Polygons stay convex after BSP splits -> fan triangulation OK.
            int i0 = getIdx(p.vertices[0]);
            for (size_t i = 1; i + 1 < p.vertices.size(); i++) {
                int i1 = getIdx(p.vertices[i]);
                int i2 = getIdx(p.vertices[i + 1]);
                if (i0 != i1 && i1 != i2 && i0 != i2) {
                    outFaces.push_back({i0, i1, i2});
                }
            }
        }
    }

    // ---------- Output ----------
    static char outBuf[1 << 16];
    setvbuf(stdout, outBuf, _IOFBF, sizeof(outBuf));

           (long long)outVerts.size(),
           (long long)outFaces.size());
    for (const auto& v : outVerts) {
        printf("%.17g %.17g %.17g\n", v.x, v.y, v.z);
    }
    for (const auto& f : outFaces) {
        printf("%d %d %d\n", f[0], f[1], f[2]);
    }

    return 0;
}
