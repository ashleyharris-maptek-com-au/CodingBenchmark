using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

static class Program
{
    // State codes:
    //   0 = unknown (originally ' ')
    //   1 = mine (originally '*' or deduced as mine)
    //   2 = deduced safe (originally ' ', proven safe)
    //   3 = revealed number ('0'-'8')
    static int H, W, N;
    static byte[] board;
    static byte[] st;

    static Queue<int> dirty;
    static bool[] inDirty;

    static void Main()
    {
        ReadInput();
        Solve();
        WriteOutput();
    }

    // ---------------- Input ----------------
    static void ReadInput()
    {
        string input;
        using (var sr = new StreamReader(Console.OpenStandardInput(), Encoding.ASCII, false, 1 << 16))
            input = sr.ReadToEnd();

        int pos = 0;
        H = ParseInt(input, ref pos);
        W = ParseInt(input, ref pos);
        // Skip to next line
        while (pos < input.Length && input[pos] != '\n') pos++;
        if (pos < input.Length) pos++;

        if (H < 0) H = 0;
        if (W < 0) W = 0;

        N = H * W;
        board = new byte[N];
        st = new byte[N];

        for (int r = 0; r < H; r++)
        {
            int baseIdx = r * W;
            int c = 0;
            while (c < W && pos < input.Length && input[pos] != '\n')
            {
                char ch = input[pos++];
                if (ch == '\r') continue;
                board[baseIdx + c] = (byte)ch;
                if (ch == ' ') st[baseIdx + c] = 0;
                else if (ch == '*') st[baseIdx + c] = 1;
                else if (ch >= '0' && ch <= '8') st[baseIdx + c] = 3;
                else
                {
                    // Treat unexpected chars as unknown
                    board[baseIdx + c] = (byte)' ';
                    st[baseIdx + c] = 0;
                }
                c++;
            }
            // Pad short rows with unknowns
            while (c < W) { board[baseIdx + c] = (byte)' '; st[baseIdx + c] = 0; c++; }
            // Consume rest of line
            while (pos < input.Length && input[pos] != '\n') pos++;
            if (pos < input.Length) pos++;
        }
    }

    static int ParseInt(string s, ref int pos)
    {
        while (pos < s.Length && !(s[pos] >= '0' && s[pos] <= '9')) pos++;
        int v = 0;
        while (pos < s.Length && s[pos] >= '0' && s[pos] <= '9')
        {
            v = v * 10 + (s[pos] - '0');
            pos++;
        }
        return v;
    }

    // ---------------- Solver ----------------
    static void Solve()
    {
        if (N == 0) return;

        dirty = new Queue<int>(Math.Min(N, 1 << 20));
        inDirty = new bool[N];

        // Seed with all numbered cells
        for (int i = 0; i < N; i++)
        {
            if (st[i] == 3) { dirty.Enqueue(i); inDirty[i] = true; }
        }

        // Pass 1: simple single-cell deductions (work-list driven, very scalable)
        ProcessSimple();

        // Adaptive: small/medium boards get more subset-rule iterations.
        // For huge boards (>1M cells) we cap iterations to keep things bounded.
        int maxIter;
        if (N <= 10_000) maxIter = 30;
        else if (N <= 1_000_000) maxIter = 10;
        else if (N <= 25_000_000) maxIter = 5;
        else maxIter = 3;

        for (int iter = 0; iter < maxIter; iter++)
        {
            bool prog = SubsetPass();
            if (!prog) break;
            ProcessSimple();
        }
    }

    static void EnqueueConstraintsAround(int idx)
    {
        int r = idx / W, c = idx % W;
        int rmin = r > 0 ? r - 1 : 0;
        int rmax = r < H - 1 ? r + 1 : H - 1;
        int cmin = c > 0 ? c - 1 : 0;
        int cmax = c < W - 1 ? c + 1 : W - 1;
        for (int nr = rmin; nr <= rmax; nr++)
        {
            int nrBase = nr * W;
            for (int nc = cmin; nc <= cmax; nc++)
            {
                int nidx = nrBase + nc;
                if (st[nidx] == 3 && !inDirty[nidx])
                {
                    dirty.Enqueue(nidx);
                    inDirty[nidx] = true;
                }
            }
        }
    }

    // Single-cell rule:
    //   For a numbered cell with value n, let f = flagged neighbors,
    //   u = unknown neighbors, remaining = n - f.
    //   If remaining == 0  -> all unknown neighbors are safe.
    //   If remaining == u  -> all unknown neighbors are mines.
    static void ProcessSimple()
    {
        while (dirty.Count > 0)
        {
            int idx = dirty.Dequeue();
            inDirty[idx] = false;
            if (st[idx] != 3) continue;

            int r = idx / W, c = idx % W;
            int n = board[idx] - '0';
            int flagged = 0, unknown = 0;
            int rmin = r > 0 ? r - 1 : 0;
            int rmax = r < H - 1 ? r + 1 : H - 1;
            int cmin = c > 0 ? c - 1 : 0;
            int cmax = c < W - 1 ? c + 1 : W - 1;

            for (int nr = rmin; nr <= rmax; nr++)
            {
                int nrBase = nr * W;
                for (int nc = cmin; nc <= cmax; nc++)
                {
                    if (nr == r && nc == c) continue;
                    byte s = st[nrBase + nc];
                    if (s == 1) flagged++;
                    else if (s == 0) unknown++;
                }
            }

            if (unknown == 0) continue;
            int remaining = n - flagged;
            if (remaining < 0 || remaining > unknown) continue; // inconsistent input, skip

            if (remaining == 0)
            {
                // All unknown neighbors are safe
                for (int nr = rmin; nr <= rmax; nr++)
                {
                    int nrBase = nr * W;
                    for (int nc = cmin; nc <= cmax; nc++)
                    {
                        if (nr == r && nc == c) continue;
                        int nidx = nrBase + nc;
                        if (st[nidx] == 0)
                        {
                            st[nidx] = 2;
                            EnqueueConstraintsAround(nidx);
                        }
                    }
                }
            }
            else if (remaining == unknown)
            {
                // All unknown neighbors are mines
                for (int nr = rmin; nr <= rmax; nr++)
                {
                    int nrBase = nr * W;
                    for (int nc = cmin; nc <= cmax; nc++)
                    {
                        if (nr == r && nc == c) continue;
                        int nidx = nrBase + nc;
                        if (st[nidx] == 0)
                        {
                            st[nidx] = 1;
                            EnqueueConstraintsAround(nidx);
                        }
                    }
                }
            }
        }
    }

    // Subset rule:
    //   For numbered cells A,B with unknown sets U_a,U_b and remaining
    //   counts r_a,r_b:  if U_a is a strict subset of U_b then:
    //     - mines in (U_b \ U_a) = r_b - r_a
    //     - if r_b - r_a == 0          -> every cell in U_b\U_a is safe
    //     - if r_b - r_a == |U_b\U_a|  -> every cell in U_b\U_a is a mine
    //   Only consider pairs (A,B) within Chebyshev distance 2 (otherwise
    //   their 3x3 neighbourhoods don't overlap and no subset relation can hold).
    static bool SubsetPass()
    {
        bool progress = false;
        Span<int> ua = stackalloc int[8];
        Span<int> ub = stackalloc int[8];

        for (int ar = 0; ar < H; ar++)
        {
            int arBase = ar * W;
            for (int ac = 0; ac < W; ac++)
            {
                int aidx = arBase + ac;
                if (st[aidx] != 3) continue;

                int n_a = board[aidx] - '0';
                int flagged_a = 0;
                int uaLen = 0;
                int armin = ar > 0 ? ar - 1 : 0;
                int armax = ar < H - 1 ? ar + 1 : H - 1;
                int acmin = ac > 0 ? ac - 1 : 0;
                int acmax = ac < W - 1 ? ac + 1 : W - 1;

                for (int nr = armin; nr <= armax; nr++)
                {
                    int nrBase = nr * W;
                    for (int nc = acmin; nc <= acmax; nc++)
                    {
                        if (nr == ar && nc == ac) continue;
                        byte s = st[nrBase + nc];
                        if (s == 1) flagged_a++;
                        else if (s == 0) ua[uaLen++] = nrBase + nc;
                    }
                }
                if (uaLen == 0) continue;
                int r_a = n_a - flagged_a;
                if (r_a < 0 || r_a > uaLen) continue;

                int brrmin = ar - 2; if (brrmin < 0) brrmin = 0;
                int brrmax = ar + 2; if (brrmax > H - 1) brrmax = H - 1;
                int bccmin = ac - 2; if (bccmin < 0) bccmin = 0;
                int bccmax = ac + 2; if (bccmax > W - 1) bccmax = W - 1;

                for (int br = brrmin; br <= brrmax; br++)
                {
                    int brBase = br * W;
                    for (int bc = bccmin; bc <= bccmax; bc++)
                    {
                        int bidx = brBase + bc;
                        if (bidx == aidx) continue;
                        if (st[bidx] != 3) continue;

                        int n_b = board[bidx] - '0';
                        int flagged_b = 0;
                        int ubLen = 0;
                        int brmin = br > 0 ? br - 1 : 0;
                        int brmax = br < H - 1 ? br + 1 : H - 1;
                        int bcmin = bc > 0 ? bc - 1 : 0;
                        int bcmax = bc < W - 1 ? bc + 1 : W - 1;

                        for (int nr = brmin; nr <= brmax; nr++)
                        {
                            int nrBase = nr * W;
                            for (int nc = bcmin; nc <= bcmax; nc++)
                            {
                                if (nr == br && nc == bc) continue;
                                byte s = st[nrBase + nc];
                                if (s == 1) flagged_b++;
                                else if (s == 0) ub[ubLen++] = nrBase + nc;
                            }
                        }
                        if (ubLen <= uaLen) continue; // need strict superset for new info
                        int r_b = n_b - flagged_b;
                        if (r_b < 0 || r_b > ubLen) continue;

                        // Check U_a is subset of U_b
                        bool subset = true;
                        for (int i = 0; i < uaLen; i++)
                        {
                            int va = ua[i];
                            bool found = false;
                            for (int j = 0; j < ubLen; j++)
                            {
                                if (ub[j] == va) { found = true; break; }
                            }
                            if (!found) { subset = false; break; }
                        }
                        if (!subset) continue;

                        int diff = r_b - r_a;
                        int extras = ubLen - uaLen;
                        if (diff < 0 || diff > extras) continue; // inconsistent

                        if (diff == 0)
                        {
                            // U_b \ U_a all safe
                            for (int j = 0; j < ubLen; j++)
                            {
                                int target = ub[j];
                                bool inA = false;
                                for (int i = 0; i < uaLen; i++)
                                {
                                    if (ua[i] == target) { inA = true; break; }
                                }
                                if (!inA && st[target] == 0)
                                {
                                    st[target] = 2;
                                    EnqueueConstraintsAround(target);
                                    progress = true;
                                }
                            }
                        }
                        else if (diff == extras)
                        {
                            // U_b \ U_a all mines
                            for (int j = 0; j < ubLen; j++)
                            {
                                int target = ub[j];
                                bool inA = false;
                                for (int i = 0; i < uaLen; i++)
                                {
                                    if (ua[i] == target) { inA = true; break; }
                                }
                                if (!inA && st[target] == 0)
                                {
                                    st[target] = 1;
                                    EnqueueConstraintsAround(target);
                                    progress = true;
                                }
                            }
                        }
                    }
                }
            }
        }
        return progress;
    }

    // ---------------- Output ----------------
    static void WriteOutput()
    {
        int count = 0;
        for (int i = 0; i < N; i++) if (st[i] == 2) count++;

        var sw = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1 << 16);
        sw.Write(count);
        sw.Write('\n');
        if (count > 0)
        {
            for (int r = 0; r < H; r++)
            {
                int rowBase = r * W;
                for (int c = 0; c < W; c++)
                {
                    if (st[rowBase + c] == 2)
                    {
                        sw.Write(c);
                        sw.Write(' ');
                        sw.Write(r);
                        sw.Write('\n');
                    }
                }
            }
        }
        sw.Flush();
    }
}
