; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %normal %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %normal Location 1
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
   %funcType = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
     %normal = OpVariable %_ptr_Input_v3float Input
   %outColor = OpVariable %_ptr_Output_v4float Output
  %float_0_5 = OpConstant %float 0.5
    %float_1 = OpConstant %float 1
   %half_vec = OpConstantComposite %v3float %float_0_5 %float_0_5 %float_0_5

       %main = OpFunction %void None %funcType
      %entry = OpLabel
          %n = OpLoad %v3float %normal
         %nn = OpExtInst %v3float %1 Normalize %n
     %scaled = OpVectorTimesScalar %v3float %nn %float_0_5
     %mapped = OpFAdd %v3float %scaled %half_vec
          %r = OpCompositeExtract %float %mapped 0
          %g = OpCompositeExtract %float %mapped 1
          %b = OpCompositeExtract %float %mapped 2
        %col = OpCompositeConstruct %v4float %r %g %b %float_1
               OpStore %outColor %col
               OpReturn
               OpFunctionEnd
