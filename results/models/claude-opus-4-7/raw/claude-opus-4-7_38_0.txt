import sys

def main():
    raw = sys.stdin.read().splitlines()
    if not raw:
        print("NO SOLUTION")
        return
    header = raw[0].split()
    U = int(header[0])
    S = int(header[1])

    sets = []
    for i in range(1, 1 + S):
        if i < len(raw):
            parts = raw[i].split()
            elems = frozenset(int(x) for x in parts)
        else:
            elems = frozenset()
        sets.append(elems)

    elem_to_sets = {e: [] for e in range(U)}
    for si, s in enumerate(sets):
        for e in s:
            if 0 <= e < U:
                elem_to_sets[e].append(si)

    for e in range(U):
        if not elem_to_sets[e]:
            print("NO SOLUTION")
            return

    uncovered = set(range(U))
    available = set(range(S))
    chosen = []
    solution = [None]

    def search():
        if solution[0] is not None:
            return
        if not uncovered:
            solution[0] = list(chosen)
            return
        best_e = None
        best_options = None
        best_count = None
        for e in uncovered:
            opts = [si for si in elem_to_sets[e] if si in available]
            if not opts:
                return
            if best_count is None or len(opts) < best_count:
                best_count = len(opts)
                best_e = e
                best_options = opts
                if best_count == 1:
                    break

        for si in best_options:
            s = sets[si]
            if not s.issubset(uncovered):
                continue
            chosen.append(si)
            removed_elems = []
            for e in s:
                uncovered.discard(e)
                removed_elems.append(e)
            removed_sets = []
            to_remove = []
            for asi in available:
                if asi == si:
                    to_remove.append(asi)
                    continue
                if not sets[asi].isdisjoint(s):
                    to_remove.append(asi)
            for asi in to_remove:
                available.discard(asi)
                removed_sets.append(asi)

            search()

            for asi in removed_sets:
                available.add(asi)
            for e in removed_elems:
                uncovered.add(e)
            chosen.pop()

            if solution[0] is not None:
                return

    search()

    if solution[0] is None:
        print("NO SOLUTION")
    else:
        sol = sorted(solution[0])
        print("SOLUTION")
        print(" ".join(str(x) for x in sol))

main()
