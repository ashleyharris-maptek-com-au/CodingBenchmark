use std::io::{self, Read, Write, BufWriter};

struct Solver {
    n: usize,
    adj: Vec<Vec<usize>>,
    target_a: usize,
    target_b: usize,
    assignment: Vec<u8>,
    count_a: usize,
    count_b: usize,
    cut: usize,
    count_to_a: Vec<usize>,
    count_to_b: Vec<usize>,
    best_cut: usize,
    best_assignment: Vec<u8>,
}

impl Solver {
    fn recurse(&mut self, v: usize) {
        if v == self.n {
            if self.count_a == self.target_a && self.cut < self.best_cut {
                self.best_cut = self.cut;
                self.best_assignment.copy_from_slice(&self.assignment);
            }
            return;
        }

        if self.cut >= self.best_cut {
            return;
        }

        let remaining = self.n - v;
        let need_a = self.target_a - self.count_a;
        let need_b = self.target_b - self.count_b;
        if need_a > remaining || need_b > remaining {
            return;
        }

        // Tight lower bound: optimal assignment of unplaced vertices ignoring
        // edges among unplaced vertices but respecting size constraints.
        // For each unplaced u: cost to put in A = count_to_b[u], cost in B = count_to_a[u].
        // Total = sum count_to_a + sum_{chosen for A}(count_to_b - count_to_a)
        // Pick the need_a smallest (count_to_b - count_to_a) for A.
        let mut total_a_costs: i64 = 0;
        let mut diffs: Vec<i64> = Vec::with_capacity(remaining);
        for u in v..self.n {
            total_a_costs += self.count_to_a[u] as i64;
            diffs.push(self.count_to_b[u] as i64 - self.count_to_a[u] as i64);
        }
        diffs.sort_unstable();
        let mut additional = total_a_costs;
        for i in 0..need_a {
            additional += diffs[i];
        }
        if (self.cut as i64) + additional >= (self.best_cut as i64) {
            return;
        }

        let cost_a = self.count_to_b[v] as i64;
        let cost_b = self.count_to_a[v] as i64;
        let a_first = cost_a <= cost_b;

        if a_first {
            self.try_a(v);
            self.try_b(v);
        } else {
            self.try_b(v);
            self.try_a(v);
        }
    }

    fn try_a(&mut self, v: usize) {
        if self.count_a >= self.target_a {
            return;
        }
        let added = self.count_to_b[v];
        if self.cut + added >= self.best_cut {
            return;
        }
        self.assignment[v] = 0;
        self.count_a += 1;
        self.cut += added;
        let nbrs_len = self.adj[v].len();
        for i in 0..nbrs_len {
            let u = self.adj[v][i];
            self.count_to_a[u] += 1;
        }
        self.recurse(v + 1);
        for i in 0..nbrs_len {
            let u = self.adj[v][i];
            self.count_to_a[u] -= 1;
        }
        self.cut -= added;
        self.count_a -= 1;
    }

    fn try_b(&mut self, v: usize) {
        if self.count_b >= self.target_b {
            return;
        }
        let added = self.count_to_a[v];
        if self.cut + added >= self.best_cut {
            return;
        }
        self.assignment[v] = 1;
        self.count_b += 1;
        self.cut += added;
        let nbrs_len = self.adj[v].len();
        for i in 0..nbrs_len {
            let u = self.adj[v][i];
            self.count_to_b[u] += 1;
        }
        self.recurse(v + 1);
        for i in 0..nbrs_len {
            let u = self.adj[v][i];
            self.count_to_b[u] -= 1;
        }
        self.cut -= added;
        self.count_b -= 1;
    }
}

fn compute_order(n: usize, adj: &[Vec<usize>]) -> Vec<usize> {
    if n == 0 {
        return vec![];
    }
    let mut order = Vec::with_capacity(n);
    let mut placed = vec![false; n];
    let mut connectivity = vec![0i64; n];

    // Start from highest-degree vertex.
    let start = (0..n).max_by_key(|&v| adj[v].len()).unwrap();
    order.push(start);
    placed[start] = true;
    for &u in &adj[start] {
        if !placed[u] {
            connectivity[u] += 1;
        }
    }

    while order.len() < n {
        let mut best: Option<usize> = None;
        let mut best_score = (i64::MIN, i64::MIN);
        for v in 0..n {
            if !placed[v] {
                let score = (connectivity[v], adj[v].len() as i64);
                if best.is_none() || score > best_score {
                    best_score = score;
                    best = Some(v);
                }
            }
        }
        let v = best.unwrap();
        order.push(v);
        placed[v] = true;
        for &u in &adj[v] {
            if !placed[u] {
                connectivity[u] += 1;
            }
        }
    }

    order
}

fn compute_cut(n: usize, adj: &[Vec<usize>], assignment: &[u8]) -> usize {
    let mut c = 0usize;
    for u in 0..n {
        for &v in &adj[u] {
            if u < v && assignment[u] != assignment[v] {
                c += 1;
            }
        }
    }
    c
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut iter = input.split_ascii_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    let mut adj_orig: Vec<Vec<usize>> = vec![vec![]; n];
    let mut edge_set: std::collections::HashSet<(usize, usize)> = std::collections::HashSet::new();
    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        if u != v {
            let key = if u < v { (u, v) } else { (v, u) };
            if edge_set.insert(key) {
                adj_orig[u].push(v);
                adj_orig[v].push(u);
            }
        }
    }

    if n == 0 {
        writeln!(out, "0").unwrap();
        writeln!(out).unwrap();
        return;
    }

    let order = compute_order(n, &adj_orig);
    let mut new_idx = vec![0usize; n];
    for (i, &v) in order.iter().enumerate() {
        new_idx[v] = i;
    }
    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    for orig_u in 0..n {
        for &orig_v in &adj_orig[orig_u] {
            adj[new_idx[orig_u]].push(new_idx[orig_v]);
        }
    }

    let target_a = n / 2;
    let target_b = n - target_a;

    // Initial upper bound: split by BFS-like order (first half in A).
    let mut init_assign = vec![1u8; n];
    for i in 0..target_a {
        init_assign[i] = 0;
    }
    let init_cut = compute_cut(n, &adj, &init_assign);

    let mut solver = Solver {
        n,
        adj,
        target_a,
        target_b,
        assignment: vec![2u8; n],
        count_a: 0,
        count_b: 0,
        cut: 0,
        count_to_a: vec![0; n],
        count_to_b: vec![0; n],
        best_cut: init_cut,
        best_assignment: init_assign,
    };

    // Symmetry breaking: vertex 0 (highest priority in our order) goes in A.
    solver.assignment[0] = 0;
    solver.count_a = 1;
    let v0_nbrs: Vec<usize> = solver.adj[0].clone();
    for u in v0_nbrs {
        solver.count_to_a[u] += 1;
    }

    solver.recurse(1);

    let partition_a: Vec<usize> = (0..n)
        .filter(|&i| solver.best_assignment[i] == 0)
        .map(|i| order[i])
        .collect();

    writeln!(out, "{}", solver.best_cut).unwrap();
    let strs: Vec<String> = partition_a.iter().map(|x| x.to_string()).collect();
    writeln!(out, "{}", strs.join(" ")).unwrap();
}
