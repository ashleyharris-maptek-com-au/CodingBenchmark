; SPIR-V
; Version: 1.0
; Geometry shader: wireframe edges of triangle-adjacency primitive
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 12
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

               ; Basic types
       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

     %uint_6 = OpConstant %uint 6
      %int_0 = OpConstant %int 0
      %int_2 = OpConstant %int 2
      %int_4 = OpConstant %int 4
    %float_0 = OpConstant %float 0
   %float_05 = OpConstant %float 0.5
%float_thick = OpConstant %float 0.004

               ; Input gl_PerVertex array
%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

               ; Output gl_PerVertex
%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float

               ; User input arrays (size 6)
  %arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

               ; Output color
 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

               ; Load main-triangle positions (indices 0, 2, 4)
     %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
     %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2_ptr
     %p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
         %p4 = OpLoad %v4float %p4_ptr

               ; Load main-triangle colors
     %c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
         %c0 = OpLoad %v3float %c0_ptr
     %c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
         %c2 = OpLoad %v3float %c2_ptr
     %c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
         %c4 = OpLoad %v3float %c4_ptr

               ; ====== Edge 0 -> 2 ======
    %e0_csum = OpFAdd %v3float %c0 %c2
    %e0_cavg = OpVectorTimesScalar %v3float %e0_csum %float_05
    %e0_p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
    %e0_p1xy = OpVectorShuffle %v2float %p2 %p2 0 1
     %e0_dir = OpFSub %v2float %e0_p1xy %e0_p0xy
    %e0_dirn = OpExtInst %v2float %1 Normalize %e0_dir
      %e0_dx = OpCompositeExtract %float %e0_dirn 0
      %e0_dy = OpCompositeExtract %float %e0_dirn 1
     %e0_ndx = OpFNegate %float %e0_dy
   %e0_perp2 = OpCompositeConstruct %v2float %e0_ndx %e0_dx
    %e0_perp = OpVectorTimesScalar %v2float %e0_perp2 %float_thick
      %e0_px = OpCompositeExtract %float %e0_perp 0
      %e0_py = OpCompositeExtract %float %e0_perp 1
  %e0_offset = OpCompositeConstruct %v4float %e0_px %e0_py %float_0 %float_0
 %e0_negoffs = OpFNegate %v4float %e0_offset

      %e0_v0 = OpFAdd %v4float %p0 %e0_offset
               OpStore %pos_out_ptr %e0_v0
               OpStore %outColor %e0_cavg
               OpEmitVertex
      %e0_v1 = OpFAdd %v4float %p0 %e0_negoffs
               OpStore %pos_out_ptr %e0_v1
               OpStore %outColor %e0_cavg
               OpEmitVertex
      %e0_v2 = OpFAdd %v4float %p2 %e0_offset
               OpStore %pos_out_ptr %e0_v2
               OpStore %outColor %e0_cavg
               OpEmitVertex
      %e0_v3 = OpFAdd %v4float %p2 %e0_negoffs
               OpStore %pos_out_ptr %e0_v3
               OpStore %outColor %e0_cavg
               OpEmitVertex
               OpEndPrimitive

               ; ====== Edge 2 -> 4 ======
    %e1_csum = OpFAdd %v3float %c2 %c4
    %e1_cavg = OpVectorTimesScalar %v3float %e1_csum %float_05
    %e1_p0xy = OpVectorShuffle %v2float %p2 %p2 0 1
    %e1_p1xy = OpVectorShuffle %v2float %p4 %p4 0 1
     %e1_dir = OpFSub %v2float %e1_p1xy %e1_p0xy
    %e1_dirn = OpExtInst %v2float %1 Normalize %e1_dir
      %e1_dx = OpCompositeExtract %float %e1_dirn 0
      %e1_dy = OpCompositeExtract %float %e1_dirn 1
     %e1_ndx = OpFNegate %float %e1_dy
   %e1_perp2 = OpCompositeConstruct %v2float %e1_ndx %e1_dx
    %e1_perp = OpVectorTimesScalar %v2float %e1_perp2 %float_thick
      %e1_px = OpCompositeExtract %float %e1_perp 0
      %e1_py = OpCompositeExtract %float %e1_perp 1
  %e1_offset = OpCompositeConstruct %v4float %e1_px %e1_py %float_0 %float_0
 %e1_negoffs = OpFNegate %v4float %e1_offset

      %e1_v0 = OpFAdd %v4float %p2 %e1_offset
               OpStore %pos_out_ptr %e1_v0
               OpStore %outColor %e1_cavg
               OpEmitVertex
      %e1_v1 = OpFAdd %v4float %p2 %e1_negoffs
               OpStore %pos_out_ptr %e1_v1
               OpStore %outColor %e1_cavg
               OpEmitVertex
      %e1_v2 = OpFAdd %v4float %p4 %e1_offset
               OpStore %pos_out_ptr %e1_v2
               OpStore %outColor %e1_cavg
               OpEmitVertex
      %e1_v3 = OpFAdd %v4float %p4 %e1_negoffs
               OpStore %pos_out_ptr %e1_v3
               OpStore %outColor %e1_cavg
               OpEmitVertex
               OpEndPrimitive

               ; ====== Edge 4 -> 0 ======
    %e2_csum = OpFAdd %v3float %c4 %c0
    %e2_cavg = OpVectorTimesScalar %v3float %e2_csum %float_05
    %e2_p0xy = OpVectorShuffle %v2float %p4 %p4 0 1
    %e2_p1xy = OpVectorShuffle %v2float %p0 %p0 0 1
     %e2_dir = OpFSub %v2float %e2_p1xy %e2_p0xy
    %e2_dirn = OpExtInst %v2float %1 Normalize %e2_dir
      %e2_dx = OpCompositeExtract %float %e2_dirn 0
      %e2_dy = OpCompositeExtract %float %e2_dirn 1
     %e2_ndx = OpFNegate %float %e2_dy
   %e2_perp2 = OpCompositeConstruct %v2float %e2_ndx %e2_dx
    %e2_perp = OpVectorTimesScalar %v2float %e2_perp2 %float_thick
      %e2_px = OpCompositeExtract %float %e2_perp 0
      %e2_py = OpCompositeExtract %float %e2_perp 1
  %e2_offset = OpCompositeConstruct %v4float %e2_px %e2_py %float_0 %float_0
 %e2_negoffs = OpFNegate %v4float %e2_offset

      %e2_v0 = OpFAdd %v4float %p4 %e2_offset
               OpStore %pos_out_ptr %e2_v0
               OpStore %outColor %e2_cavg
               OpEmitVertex
      %e2_v1 = OpFAdd %v4float %p4 %e2_negoffs
               OpStore %pos_out_ptr %e2_v1
               OpStore %outColor %e2_cavg
               OpEmitVertex
      %e2_v2 = OpFAdd %v4float %p0 %e2_offset
               OpStore %pos_out_ptr %e2_v2
               OpStore %outColor %e2_cavg
               OpEmitVertex
      %e2_v3 = OpFAdd %v4float %p0 %e2_negoffs
               OpStore %pos_out_ptr %e2_v3
               OpStore %outColor %e2_cavg
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
