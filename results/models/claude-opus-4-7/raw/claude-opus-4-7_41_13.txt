; SPIR-V
; Version: 1.0
               OpCapability Shader
        %ext = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4

   %_ptr_Input_v3f   = OpTypePointer Input %v3f
   %_ptr_Input_float = OpTypePointer Input %float
   %_ptr_Output_v4f  = OpTypePointer Output %v4f

   %worldPos = OpVariable %_ptr_Input_v3f Input
   %outColor = OpVariable %_ptr_Output_v4f Output

   %uint_1    = OpConstant %uint 1
   %float_0   = OpConstant %float 0.0
   %float_1   = OpConstant %float 1.0
   %float_0_5 = OpConstant %float 0.5
   %float_0_1 = OpConstant %float 0.1
   %float_0_4 = OpConstant %float 0.4
   %bottom    = OpConstantComposite %v3f %float_0_1 %float_0_4 %float_0_1
   %top       = OpConstantComposite %v3f %float_1 %float_1 %float_1

   %main  = OpFunction %void None %fn_void
   %entry = OpLabel
       %py  = OpAccessChain %_ptr_Input_float %worldPos %uint_1
       %y   = OpLoad %float %py
       %yh  = OpFMul %float %y %float_0_5
       %ya  = OpFAdd %float %yh %float_0_5
       %t   = OpExtInst %float %ext FClamp %ya %float_0 %float_1
       %tv  = OpCompositeConstruct %v3f %t %t %t
       %rgb = OpExtInst %v3f %ext FMix %bottom %top %tv
       %r   = OpCompositeExtract %float %rgb 0
       %g   = OpCompositeExtract %float %rgb 1
       %b   = OpCompositeExtract %float %rgb 2
       %out = OpCompositeConstruct %v4f %r %g %b %float_1
              OpStore %outColor %out
              OpReturn
              OpFunctionEnd
