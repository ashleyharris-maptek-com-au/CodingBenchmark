using System;
using System.IO;

class Program
{
    static long[] segMax;
    static int segSize;

    static void SegInit(int n)
    {
        segSize = 1;
        while (segSize < n) segSize <<= 1;
        if (segSize < 1) segSize = 1;
        segMax = new long[segSize * 2];
        for (int i = 0; i < segMax.Length; i++) segMax[i] = -1;
    }

    static void SegUpdate(int idx, long val)
    {
        idx += segSize;
        segMax[idx] = val;
        for (idx >>= 1; idx > 0; idx >>= 1)
        {
            long a = segMax[idx * 2];
            long b = segMax[idx * 2 + 1];
            long m = a > b ? a : b;
            if (segMax[idx] == m) break;
            segMax[idx] = m;
        }
    }

    // Find leftmost index whose stored remaining capacity >= val, or -1
    static int SegFindFirst(long val)
    {
        if (segMax[1] < val) return -1;
        int node = 1;
        while (node < segSize)
        {
            if (segMax[node * 2] >= val) node = node * 2;
            else node = node * 2 + 1;
        }
        return node - segSize;
    }

    static void Main()
    {
        // Slurp all stdin into a byte buffer for fast parsing
        var sin = Console.OpenStandardInput();
        var ms = new MemoryStream();
        sin.CopyTo(ms);
        byte[] data = ms.ToArray();
        int dpos = 0;
        int dlen = data.Length;

        long ReadLong()
        {
            while (dpos < dlen && (data[dpos] < (byte)'0' || data[dpos] > (byte)'9') && data[dpos] != (byte)'-') dpos++;
            bool neg = false;
            if (dpos < dlen && data[dpos] == (byte)'-') { neg = true; dpos++; }
            long n = 0;
            while (dpos < dlen && data[dpos] >= (byte)'0' && data[dpos] <= (byte)'9')
            {
                n = n * 10 + (data[dpos] - (byte)'0');
                dpos++;
            }
            return neg ? -n : n;
        }

        int N = (int)ReadLong();
        long stockLength = ReadLong();

        var cuts = new long[N];
        for (int i = 0; i < N; i++) cuts[i] = ReadLong();

        // Build sorted (descending by cut length) order using negated keys.
        // Array.Sort(keys, items) uses fast non-delegate comparison.
        var keys = new long[N];
        var indices = new int[N];
        for (int i = 0; i < N; i++) { keys[i] = -cuts[i]; indices[i] = i; }
        Array.Sort(keys, indices);

        SegInit(Math.Max(N, 1));

        var assignment = new int[N];
        int numBins = 0;

        for (int k = 0; k < N; k++)
        {
            int idx = indices[k];
            long len = cuts[idx];
            int bin = SegFindFirst(len);
            if (bin == -1)
            {
                bin = numBins++;
                SegUpdate(bin, stockLength - len);
            }
            else
            {
                long remaining = segMax[bin + segSize] - len;
                SegUpdate(bin, remaining);
            }
            assignment[idx] = bin;
        }

        // Group cuts by bin, preserving original index order within each bin
        var binCounts = new int[numBins];
        for (int i = 0; i < N; i++) binCounts[assignment[i]]++;
        var binStarts = new int[numBins + 1];
        for (int i = 0; i < numBins; i++) binStarts[i + 1] = binStarts[i] + binCounts[i];
        var binCursors = new int[numBins];
        var flat = new int[N];
        for (int i = 0; i < N; i++)
        {
            int b = assignment[i];
            flat[binStarts[b] + binCursors[b]++] = i;
        }

        // Fast buffered output
        var stdout = Console.OpenStandardOutput();
        byte[] obuf = new byte[1 << 20];
        int opos = 0;

        void Flush()
        {
            if (opos > 0) { stdout.Write(obuf, 0, opos); opos = 0; }
        }
        void WriteIntFast(int v)
        {
            if (opos > obuf.Length - 16) Flush();
            if (v == 0) { obuf[opos++] = (byte)'0'; return; }
            if (v < 0) { obuf[opos++] = (byte)'-'; v = -v; }
            int start = opos;
            while (v > 0) { obuf[opos++] = (byte)('0' + v % 10); v /= 10; }
            int end = opos - 1;
            while (start < end)
            {
                byte t = obuf[start]; obuf[start] = obuf[end]; obuf[end] = t;
                start++; end--;
            }
        }
        void WriteByteFast(byte b)
        {
            if (opos >= obuf.Length) Flush();
            obuf[opos++] = b;
        }

        WriteIntFast(numBins);
        WriteByteFast((byte)'\n');

        for (int b = 0; b < numBins; b++)
        {
            int s = binStarts[b], e = binStarts[b + 1];
            for (int j = s; j < e; j++)
            {
                if (j > s) WriteByteFast((byte)' ');
                WriteIntFast(flat[j]);
            }
            WriteByteFast((byte)'\n');
        }
        Flush();
    }
}
