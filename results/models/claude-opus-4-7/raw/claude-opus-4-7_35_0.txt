use std::io::{self, Read, Write, BufWriter};
use std::collections::BinaryHeap;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut iter = input.split_ascii_whitespace();
    let n: usize = match iter.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => 0,
    };
    let m: usize = match iter.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => 0,
    };

    let mut adj: Vec<Vec<u32>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = match iter.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => break,
        };
        let v: usize = match iter.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => break,
        };
        if u != v && u < n && v < n {
            adj[u].push(v as u32);
            adj[v].push(u as u32);
        }
    }

    for a in adj.iter_mut() {
        a.sort_unstable();
        a.dedup();
    }

    if n == 0 {
        writeln!(out, "0").unwrap();
        writeln!(out, "").unwrap();
        return;
    }

    // Greedy with max-heap (lazy deletion)
    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];
    let mut coverage: Vec<i32> = (0..n).map(|v| 1 + adj[v].len() as i32).collect();

    let mut remaining = n;
    let mut heap: BinaryHeap<(i32, u32)> = BinaryHeap::with_capacity(n * 2);
    for v in 0..n {
        heap.push((coverage[v], v as u32));
    }

    let mut set: Vec<u32> = Vec::new();

    while remaining > 0 {
        let mut picked: Option<usize> = None;
        loop {
            match heap.pop() {
                Some((c, v)) => {
                    let vu = v as usize;
                    if !in_set[vu] && c == coverage[vu] {
                        picked = Some(vu);
                        break;
                    }
                }
                None => break,
            }
        }
        let v = match picked {
            Some(x) => x,
            None => break,
        };
        if coverage[v] <= 0 {
            break;
        }

        in_set[v] = true;
        set.push(v as u32);

        // Collect newly dominated vertices
        let mut newly: Vec<usize> = Vec::new();
        if !dominated[v] {
            newly.push(v);
        }
        for &u in &adj[v] {
            let u = u as usize;
            if !dominated[u] {
                newly.push(u);
            }
        }

        for &u in &newly {
            dominated[u] = true;
            remaining -= 1;
            coverage[u] -= 1;
            if !in_set[u] {
                heap.push((coverage[u], u as u32));
            }
            for &w in &adj[u] {
                let w = w as usize;
                coverage[w] -= 1;
                if !in_set[w] {
                    heap.push((coverage[w], w as u32));
                }
            }
        }
    }

    // Redundancy removal pass
    let mut cover_count = vec![0i32; n];
    for &v in &set {
        let v = v as usize;
        cover_count[v] += 1;
        for &u in &adj[v] {
            cover_count[u as usize] += 1;
        }
    }

    let mut improved = true;
    while improved {
        improved = false;
        let mut order: Vec<usize> = (0..set.len()).collect();
        order.sort_by_key(|&i| adj[set[i] as usize].len());

        let mut to_remove = vec![false; set.len()];
        for &i in &order {
            if to_remove[i] {
                continue;
            }
            let v = set[i] as usize;
            let mut can_remove = cover_count[v] >= 2;
            if can_remove {
                for &u in &adj[v] {
                    if cover_count[u as usize] < 2 {
                        can_remove = false;
                        break;
                    }
                }
            }
            if can_remove {
                to_remove[i] = true;
                cover_count[v] -= 1;
                for &u in &adj[v] {
                    cover_count[u as usize] -= 1;
                }
                in_set[v] = false;
                improved = true;
            }
        }

        if improved {
            let mut new_set: Vec<u32> = Vec::with_capacity(set.len());
            for (i, &v) in set.iter().enumerate() {
                if !to_remove[i] {
                    new_set.push(v);
                }
            }
            set = new_set;
        }
    }

    // Local search: 2-for-1 swaps for small/medium graphs
    if n <= 3000 {
        let mut improved = true;
        let mut iters = 0;
        while improved && iters < 5 {
            improved = false;
            iters += 1;
            for w in 0..n {
                if in_set[w] {
                    continue;
                }
                in_set[w] = true;
                cover_count[w] += 1;
                for &u in &adj[w] {
                    cover_count[u as usize] += 1;
                }

                let snapshot: Vec<u32> = set.clone();
                let mut removed_count = 0;
                let mut rollback: Vec<usize> = Vec::new();
                for &v in &snapshot {
                    let v = v as usize;
                    if !in_set[v] {
                        continue;
                    }
                    let mut can = cover_count[v] >= 2;
                    if can {
                        for &u in &adj[v] {
                            if cover_count[u as usize] < 2 {
                                can = false;
                                break;
                            }
                        }
                    }
                    if can {
                        cover_count[v] -= 1;
                        for &u in &adj[v] {
                            cover_count[u as usize] -= 1;
                        }
                        in_set[v] = false;
                        rollback.push(v);
                        removed_count += 1;
                    }
                }

                if removed_count >= 2 {
                    set.retain(|&x| in_set[x as usize]);
                    set.push(w as u32);
                    improved = true;
                } else {
                    for &v in &rollback {
                        cover_count[v] += 1;
                        for &u in &adj[v] {
                            cover_count[u as usize] += 1;
                        }
                        in_set[v] = true;
                    }
                    cover_count[w] -= 1;
                    for &u in &adj[w] {
                        cover_count[u as usize] -= 1;
                    }
                    in_set[w] = false;
                }
            }
        }
    }

    writeln!(out, "{}", set.len()).unwrap();
    let strs: Vec<String> = set.iter().map(|v| v.to_string()).collect();
    writeln!(out, "{}", strs.join(" ")).unwrap();
}
