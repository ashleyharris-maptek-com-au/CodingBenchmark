using System;
using System.Collections.Generic;

class Tetris
{
    static int W, H;
    static int[] board;       // row-major H*W, 0=empty, 1=filled
    static int[] heights;     // heights[c] = number of filled cells from bottom in column c
    static Dictionary<char, int[][][]> shapes;
    static Queue<string> tokenQueue = new Queue<string>();

    static string NextToken()
    {
        while (tokenQueue.Count == 0)
        {
            string line = Console.ReadLine();
            if (line == null) return null;
            foreach (var t in line.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries))
                tokenQueue.Enqueue(t);
        }
        return tokenQueue.Dequeue();
    }

    static int NextInt() { return int.Parse(NextToken()); }

    static void InitShapes()
    {
        // Shapes are normalized so min row offset = 0 and min col offset = 0.
        // Each rotation: list of (dr, dc) cells.  Rotation index = clockwise quarter turns.
        shapes = new Dictionary<char, int[][][]>();

        // I: 4x1 line piece
        shapes['I'] = new int[][][]
        {
            new[] { new[] {0,0}, new[] {0,1}, new[] {0,2}, new[] {0,3} },           // horizontal
            new[] { new[] {0,0}, new[] {1,0}, new[] {2,0}, new[] {3,0} },           // vertical
        };

        // O: 2x2 square
        shapes['O'] = new int[][][]
        {
            new[] { new[] {0,0}, new[] {0,1}, new[] {1,0}, new[] {1,1} },
        };

        // T piece
        shapes['T'] = new int[][][]
        {
            new[] { new[] {0,0}, new[] {0,1}, new[] {0,2}, new[] {1,1} },           // T pointing down
            new[] { new[] {0,0}, new[] {1,0}, new[] {1,1}, new[] {2,0} },           // T pointing right
            new[] { new[] {0,1}, new[] {1,0}, new[] {1,1}, new[] {1,2} },           // T pointing up
            new[] { new[] {0,1}, new[] {1,0}, new[] {1,1}, new[] {2,1} },           // T pointing left
        };

        // S (right-leaning)
        shapes['S'] = new int[][][]
        {
            new[] { new[] {0,1}, new[] {0,2}, new[] {1,0}, new[] {1,1} },           // .##/##.
            new[] { new[] {0,0}, new[] {1,0}, new[] {1,1}, new[] {2,1} },           // #./##/.#
        };

        // Z (left-leaning)
        shapes['Z'] = new int[][][]
        {
            new[] { new[] {0,0}, new[] {0,1}, new[] {1,1}, new[] {1,2} },           // ##./.##
            new[] { new[] {0,1}, new[] {1,0}, new[] {1,1}, new[] {2,0} },           // .#/##/#.
        };

        // J (left-leaning L)
        shapes['J'] = new int[][][]
        {
            new[] { new[] {0,0}, new[] {1,0}, new[] {1,1}, new[] {1,2} },           // #../###
            new[] { new[] {0,0}, new[] {0,1}, new[] {1,0}, new[] {2,0} },           // ##/#./#.
            new[] { new[] {0,0}, new[] {0,1}, new[] {0,2}, new[] {1,2} },           // ###/..#
            new[] { new[] {0,1}, new[] {1,1}, new[] {2,0}, new[] {2,1} },           // .#/.#/##
        };

        // L (right-leaning)
        shapes['L'] = new int[][][]
        {
            new[] { new[] {0,2}, new[] {1,0}, new[] {1,1}, new[] {1,2} },           // ..#/###
            new[] { new[] {0,0}, new[] {1,0}, new[] {2,0}, new[] {2,1} },           // #./#./##
            new[] { new[] {0,0}, new[] {0,1}, new[] {0,2}, new[] {1,0} },           // ###/#..
            new[] { new[] {0,0}, new[] {0,1}, new[] {1,1}, new[] {2,1} },           // ##/.#/.#
        };
    }

    // Compute resting r0 (top row of bounding box) for piece at given rotation/column.
    // Returns int.MinValue if column out of bounds.  r0 may be negative (sticks out top).
    static int FindRestRow(int[][] shape, int col, int[] hts)
    {
        int maxDc = 0;
        foreach (var cell in shape)
            if (cell[1] > maxDc) maxDc = cell[1];
        int pieceWidth = maxDc + 1;
        if (col < 0 || col + pieceWidth > W) return int.MinValue;

        int[] maxDrForCol = new int[pieceWidth];
        for (int i = 0; i < pieceWidth; i++) maxDrForCol[i] = -1;
        foreach (var cell in shape)
            if (cell[0] > maxDrForCol[cell[1]]) maxDrForCol[cell[1]] = cell[0];

        int r0 = int.MaxValue;
        for (int i = 0; i < pieceWidth; i++)
        {
            if (maxDrForCol[i] < 0) continue;
            // Resting condition: piece bottom in this local col lands on row (H - heights[col+i] - 1).
            int targetR0 = H - hts[col + i] - 1 - maxDrForCol[i];
            if (targetR0 < r0) r0 = targetR0;
        }
        return r0;
    }

    static double Evaluate(char piece, int rot, int col)
    {
        var shape = shapes[piece][rot];
        int r0 = FindRestRow(shape, col, heights);
        if (r0 == int.MinValue) return double.NegativeInfinity;

        int minDr = int.MaxValue;
        foreach (var cell in shape)
            if (cell[0] < minDr) minDr = cell[0];
        if (r0 + minDr < 0) return double.NegativeInfinity; // sticks out top => game over

        int[] nb = (int[])board.Clone();
        foreach (var cell in shape)
        {
            int r = r0 + cell[0];
            int c = col + cell[1];
            nb[r * W + c] = 1;
        }

        // Clear filled lines using a two-pointer compact approach.
        int linesCleared = 0;
        int writeRow = H - 1;
        for (int r = H - 1; r >= 0; r--)
        {
            bool full = true;
            int rowBase = r * W;
            for (int c = 0; c < W; c++)
            {
                if (nb[rowBase + c] == 0) { full = false; break; }
            }
            if (!full)
            {
                if (writeRow != r)
                {
                    int dst = writeRow * W;
                    for (int c = 0; c < W; c++)
                        nb[dst + c] = nb[rowBase + c];
                }
                writeRow--;
            }
            else linesCleared++;
        }
        for (int r = writeRow; r >= 0; r--)
        {
            int rb = r * W;
            for (int c = 0; c < W; c++) nb[rb + c] = 0;
        }

        int aggregateHeight = 0;
        int holes = 0;
        int maxH = 0;
        int[] nh = new int[W];
        int rowTransitions = 0;
        int colTransitions = 0;
        int wells = 0;

        for (int c = 0; c < W; c++)
        {
            bool foundTop = false;
            int h = 0;
            int prevCell = 1; // floor treated as filled for column transition counting
            for (int r = 0; r < H; r++)
            {
                int v = nb[r * W + c];
                if (v == 1)
                {
                    if (!foundTop) { foundTop = true; h = H - r; }
                }
                else if (foundTop)
                {
                    holes++;
                }
                if (v != prevCell) colTransitions++;
                prevCell = v;
            }
            if (prevCell == 0) colTransitions++;
            nh[c] = h;
            aggregateHeight += h;
            if (h > maxH) maxH = h;
        }

        for (int r = 0; r < H; r++)
        {
            int prev = 1; // walls treated as filled
            int rb = r * W;
            for (int c = 0; c < W; c++)
            {
                int v = nb[rb + c];
                if (v != prev) rowTransitions++;
                prev = v;
            }
            if (prev != 1) rowTransitions++;
        }

        int bumpiness = 0;
        for (int c = 0; c < W; c++)
        {
            int leftH = (c == 0) ? H : nh[c - 1];
            int rightH = (c == W - 1) ? H : nh[c + 1];
            int wellDepth = Math.Min(leftH, rightH) - nh[c];
            if (wellDepth > 0) wells += wellDepth * (wellDepth + 1) / 2;
            if (c < W - 1) bumpiness += Math.Abs(nh[c] - nh[c + 1]);
        }

        // Adaptive weighting: pick a strategy based on stack danger level.
        double dangerLevel = (double)maxH / H;
        double wHeight, wLines, wHoles, wBump, wMax, wRowT, wColT, wWell;

        if (dangerLevel > 0.75)
        {
            // EMERGENCY: prioritize line clears, accept bumpiness to survive.
            wHeight = -0.55; wLines = 2.2; wHoles = -1.2; wBump = -0.20;
            wMax = -0.6; wRowT = -0.10; wColT = -0.10; wWell = -0.20;
        }
        else if (dangerLevel > 0.45)
        {
            // MID-GAME: balanced Dellacherie-like weights.
            wHeight = -0.510066; wLines = 1.10; wHoles = -0.45; wBump = -0.184483;
            wMax = -0.20; wRowT = -0.05; wColT = -0.05; wWell = -0.10;
        }
        else
        {
            // EARLY/LOW: keep the board clean, allow shallow wells for future I-pieces.
            wHeight = -0.40; wLines = 0.60; wHoles = -0.60; wBump = -0.18;
            wMax = -0.05; wRowT = -0.03; wColT = -0.03; wWell = -0.05;
        }

        // Reward multi-line clears nonlinearly (Tetris > single).
        double linesScore = linesCleared switch
        {
            0 => 0.0,
            1 => 1.0,
            2 => 3.0,
            3 => 6.0,
            4 => 12.0,
            _ => linesCleared * 3.0
        };

        double score = wHeight * aggregateHeight
                     + wLines * linesScore
                     + wHoles * holes
                     + wBump * bumpiness
                     + wMax * maxH
                     + wRowT * rowTransitions
                     + wColT * colTransitions
                     + wWell * wells;
        return score;
    }

    static (int, int) ChooseBest(char piece)
    {
        double bestScore = double.NegativeInfinity;
        int bestRot = 0, bestCol = 0;
        int numRot = shapes[piece].Length;

        for (int rot = 0; rot < numRot; rot++)
        {
            var shape = shapes[piece][rot];
            int maxDc = 0;
            foreach (var cell in shape) if (cell[1] > maxDc) maxDc = cell[1];
            int pieceWidth = maxDc + 1;
            int maxCol = W - pieceWidth;
            for (int col = 0; col <= maxCol; col++)
            {
                double s = Evaluate(piece, rot, col);
                if (s > bestScore)
                {
                    bestScore = s;
                    bestRot = rot;
                    bestCol = col;
                }
            }
        }
        return (bestRot, bestCol);
    }

    static void ApplyPlacement(char piece, int rot, int col)
    {
        var shape = shapes[piece][rot];
        int r0 = FindRestRow(shape, col, heights);
        if (r0 == int.MinValue) return;

        int minDr = int.MaxValue;
        foreach (var cell in shape)
            if (cell[0] < minDr) minDr = cell[0];
        if (r0 + minDr < 0) return;

        foreach (var cell in shape)
        {
            int r = r0 + cell[0];
            int c = col + cell[1];
            if (r >= 0 && r < H && c >= 0 && c < W)
                board[r * W + c] = 1;
        }

        int writeRow = H - 1;
        for (int r = H - 1; r >= 0; r--)
        {
            bool full = true;
            int rb = r * W;
            for (int c = 0; c < W; c++)
            {
                if (board[rb + c] == 0) { full = false; break; }
            }
            if (!full)
            {
                if (writeRow != r)
                {
                    int dst = writeRow * W;
                    for (int c = 0; c < W; c++) board[dst + c] = board[rb + c];
                }
                writeRow--;
            }
        }
        for (int r = writeRow; r >= 0; r--)
        {
            int rb = r * W;
            for (int c = 0; c < W; c++) board[rb + c] = 0;
        }

        for (int c = 0; c < W; c++)
        {
            heights[c] = 0;
            for (int r = 0; r < H; r++)
            {
                if (board[r * W + c] == 1) { heights[c] = H - r; break; }
            }
        }
    }

    static void Main()
    {
        InitShapes();

        W = NextInt();
        H = NextInt();
        int N = NextInt();

        board = new int[H * W];
        heights = new int[W];

        var stdout = new System.IO.StreamWriter(Console.OpenStandardOutput());
        stdout.AutoFlush = false;

        for (int p = 0; p < N; p++)
        {
            string pieceTok = NextToken();
            if (pieceTok == null) break;
            char piece = pieceTok[0];

            if (!shapes.ContainsKey(piece))
            {
                stdout.Write("0 0\n");
                stdout.Flush();
                string r2 = NextToken();
                if (r2 == null || r2.StartsWith("gameover")) break;
                continue;
            }

            var best = ChooseBest(piece);
            int bestRot = best.Item1;
            int bestCol = best.Item2;

            stdout.Write(bestRot);
            stdout.Write(' ');
            stdout.Write(bestCol);
            stdout.Write('\n');
            stdout.Flush();

            ApplyPlacement(piece, bestRot, bestCol);

            string result = NextToken();
            if (result == null) break;
            if (result.StartsWith("gameover")) break;
            // "ok" and "cleared_N" both indicate continue; our local board already reflects clears.
        }
    }
}
