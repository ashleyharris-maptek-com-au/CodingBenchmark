// 2D Cutting Stock Problem (Rectangle Bin Packing) Solver
// Strategy:
//   - Skyline Bottom-Left with waste minimization per board
//   - Try multiple sort heuristics; pick best result subject to time budget
//   - Adaptive: more attempts for smaller inputs, fewer for huge inputs
//
// Compiles cleanly with MSVC /std:c++20.

#include <cstdio>
#include <cstdlib>
#include <cstdint>
#include <cstring>
#include <cmath>
#include <vector>
#include <array>
#include <algorithm>
#include <chrono>
#include <limits>
#include <string>

using std::vector;
using std::int32_t;
using std::int64_t;
using std::size_t;

struct Rect {
    int w, h;
    int idx; // original index
};

struct Placement {
    int board;
    int x, y;
    int rotated; // 0 or 1
};

struct SkylineNode {
    int x;     // x coordinate of left edge
    int y;     // y coordinate of top of skyline at this segment
    int width; // width of segment
};

struct Skyline {
    int W, H;
    vector<SkylineNode> nodes;

    void init(int width, int height) {
        W = width;
        H = height;
        nodes.clear();
        nodes.push_back({0, 0, W});
    }

    // Computes y if rectangle of (w,h) is placed at index i.
    // Returns -1 if can't fit. Also returns the wasted area below the placement.
    int fitAt(int i, int w, int h, int& wastedArea) const {
        int x = nodes[i].x;
        if (x + w > W) return -1;
        int widthLeft = w;
        int j = i;
        int y = nodes[i].y;
        while (widthLeft > 0) {
            if (j >= (int)nodes.size()) return -1;
            if (nodes[j].y > y) y = nodes[j].y;
            if (y + h > H) return -1;
            widthLeft -= nodes[j].width;
            ++j;
        }
        // Compute waste below: sum over covered segments of (y - node.y) * node.width
        wastedArea = 0;
        int remain = w;
        int k = i;
        while (remain > 0) {
            int segW = nodes[k].width;
            int useW = segW < remain ? segW : remain;
            wastedArea += (y - nodes[k].y) * useW;
            remain -= useW;
            ++k;
        }
        return y;
    }

    void place(int i, int x, int y, int w, int h) {
        SkylineNode newNode{x, y + h, w};
        int endX = x + w;

        int removeStart = i;
        int removeEnd = i;
        while (removeEnd < (int)nodes.size() && nodes[removeEnd].x + nodes[removeEnd].width <= endX) {
            ++removeEnd;
        }
        bool trimExisting = false;
        SkylineNode trimmed{};
        if (removeEnd < (int)nodes.size() && nodes[removeEnd].x < endX) {
            int overlap = endX - nodes[removeEnd].x;
            trimmed = nodes[removeEnd];
            trimmed.x += overlap;
            trimmed.width -= overlap;
            trimExisting = true;
            ++removeEnd;
        }

        nodes.erase(nodes.begin() + removeStart, nodes.begin() + removeEnd);

        if (trimExisting) {
            nodes.insert(nodes.begin() + removeStart, trimmed);
        }
        nodes.insert(nodes.begin() + removeStart, newNode);

        if (removeStart > 0 && nodes[removeStart - 1].y == nodes[removeStart].y) {
            nodes[removeStart - 1].width += nodes[removeStart].width;
            nodes.erase(nodes.begin() + removeStart);
            --removeStart;
        }
        if (removeStart + 1 < (int)nodes.size() && nodes[removeStart].y == nodes[removeStart + 1].y) {
            nodes[removeStart].width += nodes[removeStart + 1].width;
            nodes.erase(nodes.begin() + removeStart + 1);
        }
    }

    bool findBestFit(int w, int h, int& outX, int& outY, int& outIdx, int& outWaste) const {
        int bestY = std::numeric_limits<int>::max();
        int bestWaste = std::numeric_limits<int>::max();
        int bestIdx = -1;
        int bestX = -1;
        for (int i = 0; i < (int)nodes.size(); ++i) {
            int waste = 0;
            int y = fitAt(i, w, h, waste);
            if (y < 0) continue;
            int topY = y + h;
            if (topY < bestY || (topY == bestY && waste < bestWaste)) {
                bestY = topY;
                bestWaste = waste;
                bestIdx = i;
                bestX = nodes[i].x;
            }
        }
        if (bestIdx < 0) return false;
        outX = bestX;
        outY = bestY - h;
        outIdx = bestIdx;
        outWaste = bestWaste;
        return true;
    }
};

struct Solution {
    int numBoards = 0;
    vector<Placement> places;
};

static int BW, BH;

static Solution packSkyline(const vector<Rect>& rects, bool allowRotate, int N) {
    Solution sol;
    sol.places.assign(N, {0,0,0,0});

    vector<Skyline> boards;
    boards.reserve(8);

    for (const auto& r : rects) {
        int placedBoard = -1;
        int placedX = 0, placedY = 0;
        int placedRot = 0;
        int bestBoardWaste = std::numeric_limits<int>::max();
        int bestBoardTopY = std::numeric_limits<int>::max();

        for (int b = 0; b < (int)boards.size(); ++b) {
            if (r.w <= boards[b].W && r.h <= boards[b].H) {
                int x, y, idx, waste;
                if (boards[b].findBestFit(r.w, r.h, x, y, idx, waste)) {
                    int topY = y + r.h;
                    if (topY < bestBoardTopY || (topY == bestBoardTopY && waste < bestBoardWaste)) {
                        bestBoardTopY = topY;
                        bestBoardWaste = waste;
                        placedBoard = b;
                        placedX = x;
                        placedY = y;
                        placedRot = 0;
                    }
                }
            }
            if (allowRotate && r.w != r.h && r.h <= boards[b].W && r.w <= boards[b].H) {
                int x, y, idx, waste;
                if (boards[b].findBestFit(r.h, r.w, x, y, idx, waste)) {
                    int topY = y + r.w;
                    if (topY < bestBoardTopY || (topY == bestBoardTopY && waste < bestBoardWaste)) {
                        bestBoardTopY = topY;
                        bestBoardWaste = waste;
                        placedBoard = b;
                        placedX = x;
                        placedY = y;
                        placedRot = 1;
                    }
                }
            }
        }

        if (placedBoard < 0) {
            Skyline sk;
            sk.init(BW, BH);
            int w = r.w, h = r.h, rot = 0;
            if (w > BW || h > BH) {
                if (allowRotate && r.h <= BW && r.w <= BH) {
                    w = r.h; h = r.w; rot = 1;
                }
            } else if (allowRotate && r.w != r.h && r.h <= BW && r.w <= BH) {
                if (r.h > r.w) {
                    w = r.h; h = r.w; rot = 1;
                }
            }
            int idx = 0, waste = 0;
            int x = 0, y = 0;
            sk.findBestFit(w, h, x, y, idx, waste);
            sk.place(idx, x, y, w, h);
            boards.push_back(std::move(sk));

            sol.places[r.idx] = {(int)boards.size() - 1, x, y, rot};
        } else {
            int w = (placedRot ? r.h : r.w);
            int h = (placedRot ? r.w : r.h);
            int idx = -1;
            for (int i = 0; i < (int)boards[placedBoard].nodes.size(); ++i) {
                if (boards[placedBoard].nodes[i].x == placedX) {
                    idx = i;
                    break;
                }
            }
            if (idx < 0) {
                int x2, y2, waste2;
                boards[placedBoard].findBestFit(w, h, x2, y2, idx, waste2);
            }
            boards[placedBoard].place(idx, placedX, placedY, w, h);
            sol.places[r.idx] = {placedBoard, placedX, placedY, placedRot};
        }
    }

    sol.numBoards = (int)boards.size();
    return sol;
}

int main() {
    std::ios::sync_with_stdio(false);

    int N;
    if (scanf("%d %d %d", &N, &BW, &BH) != 3) return 0;

    vector<Rect> rects(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        scanf("%d %d", &w, &h);
        rects[i].w = w;
        rects[i].h = h;
        rects[i].idx = i;
    }

    auto t0 = std::chrono::steady_clock::now();
    auto timeLeft = [&]() {
        auto now = std::chrono::steady_clock::now();
        double sec = std::chrono::duration<double>(now - t0).count();
        return 27.0 - sec;
    };

    auto trySort = [&](int mode) -> vector<Rect> {
        vector<Rect> v = rects;
        switch (mode) {
            case 0: // longest side desc, shorter desc
                std::sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                    int ah = std::max(a.w, a.h), bh = std::max(b.w, b.h);
                    if (ah != bh) return ah > bh;
                    int aw = std::min(a.w, a.h), bw = std::min(b.w, b.h);
                    return aw > bw;
                });
                break;
            case 1: // area desc
                std::sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                    int64_t aa = (int64_t)a.w * a.h, bb = (int64_t)b.w * b.h;
                    if (aa != bb) return aa > bb;
                    return std::max(a.w, a.h) > std::max(b.w, b.h);
                });
                break;
            case 2: // longest side desc, shorter asc
                std::sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                    int aw = std::max(a.w, a.h), bw = std::max(b.w, b.h);
                    if (aw != bw) return aw > bw;
                    int ah = std::min(a.w, a.h), bh = std::min(b.w, b.h);
                    return ah < bh;
                });
                break;
            case 3: // perimeter desc
                std::sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                    int ap = a.w + a.h, bp = b.w + b.h;
                    if (ap != bp) return ap > bp;
                    return (int64_t)a.w * a.h > (int64_t)b.w * b.h;
                });
                break;
            case 4: // short side desc
                std::sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                    int as = std::min(a.w, a.h), bs = std::min(b.w, b.h);
                    if (as != bs) return as > bs;
                    return std::max(a.w, a.h) > std::max(b.w, b.h);
                });
                break;
        }
        return v;
    };

    int maxAttempts;
    if (N <= 200) maxAttempts = 5;
    else if (N <= 2000) maxAttempts = 4;
    else if (N <= 10000) maxAttempts = 3;
    else if (N <= 25000) maxAttempts = 2;
    else maxAttempts = 1;

    Solution best;
    best.numBoards = std::numeric_limits<int>::max();

    int sortOrders[5] = {0, 1, 3, 2, 4};

    for (int attempt = 0; attempt < maxAttempts; ++attempt) {
        if (timeLeft() < 1.0 && attempt > 0) break;
        vector<Rect> sorted = trySort(sortOrders[attempt]);
        Solution s = packSkyline(sorted, true, N);
        if (s.numBoards < best.numBoards) {
            best = std::move(s);
        }
        if (best.numBoards <= 1) break;
    }

    std::string out;
    out.reserve((size_t)N * 24 + 32);
    char buf[64];
    int len = snprintf(buf, sizeof(buf), "%d\n", best.numBoards);
    out.append(buf, len);
    for (int i = 0; i < N; ++i) {
        const auto& p = best.places[i];
        len = snprintf(buf, sizeof(buf), "%d %d %d %d\n", p.board, p.x, p.y, p.rotated);
        out.append(buf, len);
    }
    fwrite(out.data(), 1, out.size(), stdout);
    return 0;
}
