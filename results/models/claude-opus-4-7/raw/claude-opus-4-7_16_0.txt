#include <iostream>
#include <vector>
#include <algorithm>
#include <random>
#include <chrono>
#include <climits>
#include <cstdint>
#include <cstdio>
#include <string>

using namespace std;

static int J, M;

struct Op {
    int machine;
    int duration;
};

static vector<vector<Op>> jobOps;
static vector<long long> totalWork;

// Giffler-Thompson active schedule generation with a configurable priority rule.
// Rules:
//   0 = MWR    (Most Work Remaining)
//   1 = SPT    (Shortest Processing Time)
//   2 = LPT    (Longest Processing Time)
//   3 = LWR    (Least Work Remaining)
//   4 = FIFO   (lowest job index)
//   5 = LIFO   (highest job index)
//   6 = Random
//   7 = MOPNR  (Most Operations Remaining)
//   8 = LOPNR  (Least Operations Remaining)
static long long runDispatch(int rule, uint32_t seed,
                             vector<vector<int>>& starts)
{
    starts.assign(J, vector<int>());
    for (int j = 0; j < J; j++) starts[j].assign(jobOps[j].size(), 0);

    vector<int> nextOp(J, 0);
    vector<long long> jobReady(J, 0);
    vector<long long> machineReady(M, 0);
    vector<long long> workRemaining(totalWork);
    vector<int> opsRemaining(J);
    int remaining = 0;
    for (int j = 0; j < J; j++) {
        opsRemaining[j] = (int)jobOps[j].size();
        remaining += opsRemaining[j];
    }

    mt19937 rng(seed ? seed : 1u);
    uniform_real_distribution<double> dist(0.0, 1.0);
    vector<double> jitter(J);
    for (int j = 0; j < J; j++) jitter[j] = dist(rng);

    while (remaining > 0) {
        // Step 1: find minimum completion time among the next op of every job,
        //         and the machine on which that minimum completes.
        long long minComp = LLONG_MAX;
        int conflictMachine = -1;
        for (int j = 0; j < J; j++) {
            int k = nextOp[j];
            if (k >= (int)jobOps[j].size()) continue;
            const Op& op = jobOps[j][k];
            long long s = jobReady[j];
            long long mr = machineReady[op.machine];
            if (mr > s) s = mr;
            long long c = s + op.duration;
            if (c < minComp) {
                minComp = c;
                conflictMachine = op.machine;
            }
        }

        // Step 2: among all next ops on conflictMachine whose earliest start
        //         is strictly less than minComp, pick by priority rule.
        int bestJob = -1;
        double bestPrio = -1e300;
        for (int j = 0; j < J; j++) {
            int k = nextOp[j];
            if (k >= (int)jobOps[j].size()) continue;
            const Op& op = jobOps[j][k];
            if (op.machine != conflictMachine) continue;
            long long s = jobReady[j];
            long long mr = machineReady[op.machine];
            if (mr > s) s = mr;
            if (s >= minComp) continue;

            double p;
            switch (rule) {
                case 0: p =  (double)workRemaining[j]; break;
                case 1: p = -(double)op.duration; break;
                case 2: p =  (double)op.duration; break;
                case 3: p = -(double)workRemaining[j]; break;
                case 4: p = -(double)j; break;
                case 5: p =  (double)j; break;
                case 6: p =  jitter[j]; break;
                case 7: p =  (double)opsRemaining[j]; break;
                case 8: p = -(double)opsRemaining[j]; break;
                default: p = (double)workRemaining[j];
            }
            p += jitter[j] * 1e-9; // deterministic tiebreak per seed
            if (p > bestPrio) {
                bestPrio = p;
                bestJob = j;
            }
        }

        if (bestJob < 0) {
            // Guard against pathological input (should not normally trigger).
            for (int j = 0; j < J; j++) {
                if (nextOp[j] < (int)jobOps[j].size()) { bestJob = j; break; }
            }
            if (bestJob < 0) break;
        }

        int j = bestJob;
        int k = nextOp[j];
        const Op& op = jobOps[j][k];
        long long s = jobReady[j];
        long long mr = machineReady[op.machine];
        if (mr > s) s = mr;
        starts[j][k] = (int)s;
        long long e = s + op.duration;
        jobReady[j] = e;
        machineReady[op.machine] = e;
        workRemaining[j] -= op.duration;
        opsRemaining[j]--;
        nextOp[j]++;
        remaining--;
    }

    long long makespan = 0;
    for (int j = 0; j < J; j++)
        if (jobReady[j] > makespan) makespan = jobReady[j];
    return makespan;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> J >> M)) return 0;
    if (J <= 0) return 0;
    if (M <= 0) M = 1;
    jobOps.assign(J, {});
    totalWork.assign(J, 0);

    long long totalOps = 0;
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        if (T < 0) T = 0;
        jobOps[j].assign(T, Op{});
        for (int i = 0; i < T; i++) {
            cin >> jobOps[j][i].machine >> jobOps[j][i].duration;
            if (jobOps[j][i].duration < 0) jobOps[j][i].duration = 0;
            totalWork[j] += jobOps[j][i].duration;
        }
        totalOps += T;
    }

    auto t0 = chrono::high_resolution_clock::now();
    // Adaptive time budget — bigger problems get more time.
    double timeLimit;
    if (totalOps <= 100)         timeLimit = 1.5;
    else if (totalOps <= 1000)   timeLimit = 4.0;
    else if (totalOps <= 10000)  timeLimit = 10.0;
    else if (totalOps <= 50000)  timeLimit = 18.0;
    else                          timeLimit = 25.0;

    // Number of restarts to try.
    int maxRuns;
    if (totalOps <= 50)          maxRuns = 200;
    else if (totalOps <= 500)    maxRuns = 120;
    else if (totalOps <= 5000)   maxRuns = 40;
    else if (totalOps <= 20000)  maxRuns = 15;
    else if (totalOps <= 50000)  maxRuns = 6;
    else                          maxRuns = 3;

    vector<vector<int>> bestStarts;
    long long bestMakespan = LLONG_MAX;

    // Order chosen to put strong, complementary rules first.
    const int baseRules[] = {0, 7, 2, 3, 1, 4, 5, 8, 6};
    const int numBaseRules = (int)(sizeof(baseRules) / sizeof(baseRules[0]));

    for (int run = 0; run < maxRuns; run++) {
        auto now = chrono::high_resolution_clock::now();
        double elapsed = chrono::duration<double>(now - t0).count();
        if (elapsed > timeLimit && bestMakespan != LLONG_MAX) break;

        int rule;
        uint32_t seed;
        if (run < numBaseRules) {
            rule = baseRules[run];
            seed = (uint32_t)(0x9E3779B1u ^ (uint32_t)(run + 1));
        } else {
            // Random-priority restarts with fresh jitter.
            rule = 6;
            seed = (uint32_t)(0xDEADBEEFu * (uint32_t)(run + 1) + 12345u);
        }

        vector<vector<int>> starts;
        long long ms = runDispatch(rule, seed, starts);

        if (ms < bestMakespan) {
            bestMakespan = ms;
            bestStarts = std::move(starts);
        }
    }

    // Safety fallback (should not be needed in normal use).
    if ((int)bestStarts.size() != J) {
        bestStarts.assign(J, vector<int>());
        for (int j = 0; j < J; j++) bestStarts[j].assign(jobOps[j].size(), 0);
    }

    // Output: J lines, "start machine start machine ..." per job.
    string out;
    out.reserve((size_t)(totalOps * 12 + (size_t)J * 2));
    char buf[64];
    for (int j = 0; j < J; j++) {
        int T = (int)jobOps[j].size();
        for (int i = 0; i < T; i++) {
            if (i > 0) out.push_back(' ');
            int n = snprintf(buf, sizeof(buf), "%d %d",
                             bestStarts[j][i], jobOps[j][i].machine);
            out.append(buf, n);
        }
        out.push_back('\n');
    }
    fwrite(out.data(), 1, out.size(), stdout);
    return 0;
}
