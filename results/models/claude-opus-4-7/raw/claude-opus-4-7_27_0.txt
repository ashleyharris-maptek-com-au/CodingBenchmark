#include <iostream>
#include <vector>
#include <algorithm>
#include <random>
#include <chrono>
#include <cstring>
#include <cstdio>
#include <numeric>
#include <climits>
#include <cstdint>

using namespace std;
using namespace std::chrono;

static int N, M, K;
static vector<vector<int>> adj;
static vector<int> color;
static high_resolution_clock::time_point start_time;

static double elapsed() {
    return duration<double>(high_resolution_clock::now() - start_time).count();
}

// Largest-degree-first greedy
static bool greedy_LDF() {
    color.assign(N, -1);
    vector<int> order(N);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [](int a, int b) {
        if (adj[a].size() != adj[b].size()) return adj[a].size() > adj[b].size();
        return a < b;
    });
    vector<int> used(K, -1);
    for (int v : order) {
        for (int u : adj[v]) {
            if (color[u] != -1) used[color[u]] = v;
        }
        int c = -1;
        for (int i = 0; i < K; i++) {
            if (used[i] != v) { c = i; break; }
        }
        if (c == -1) return false;
        color[v] = c;
    }
    return true;
}

// Smallest-last (degeneracy) ordering greedy
static bool greedy_smallest_last() {
    color.assign(N, -1);
    vector<int> deg(N);
    int maxd = 0;
    for (int i = 0; i < N; i++) { deg[i] = (int)adj[i].size(); if (deg[i] > maxd) maxd = deg[i]; }
    vector<vector<int>> buckets(maxd + 1);
    for (int i = 0; i < N; i++) buckets[deg[i]].push_back(i);
    vector<int> order;
    order.reserve(N);
    vector<char> removed(N, 0);
    int cur = 0;
    while ((int)order.size() < N) {
        while (cur <= maxd && buckets[cur].empty()) cur++;
        if (cur > maxd) break;
        int v = buckets[cur].back();
        buckets[cur].pop_back();
        if (removed[v]) continue;
        removed[v] = 1;
        order.push_back(v);
        for (int u : adj[v]) {
            if (removed[u]) continue;
            int d = deg[u];
            deg[u] = d - 1;
            if (deg[u] >= 0) buckets[deg[u]].push_back(u);
            if (deg[u] >= 0 && deg[u] < cur) cur = deg[u];
        }
    }
    // Color in reverse order (smallest-last)
    vector<int> used(K, -1);
    for (int i = (int)order.size() - 1; i >= 0; i--) {
        int v = order[i];
        for (int u : adj[v]) {
            if (color[u] != -1) used[color[u]] = v;
        }
        int c = -1;
        for (int j = 0; j < K; j++) {
            if (used[j] != v) { c = j; break; }
        }
        if (c == -1) return false;
        color[v] = c;
    }
    return true;
}

// DSATUR for small graphs (O(N^2))
static bool dsatur() {
    color.assign(N, -1);
    vector<int> sat(N, 0);
    vector<int> deg(N);
    vector<vector<uint8_t>> nbcolor(N, vector<uint8_t>(K, 0));
    for (int i = 0; i < N; i++) deg[i] = (int)adj[i].size();

    for (int step = 0; step < N; step++) {
        int best = -1, best_sat = -1, best_deg = -1;
        for (int v = 0; v < N; v++) {
            if (color[v] != -1) continue;
            if (sat[v] > best_sat || (sat[v] == best_sat && deg[v] > best_deg)) {
                best_sat = sat[v];
                best_deg = deg[v];
                best = v;
            }
        }
        if (best == -1) break;
        int c = -1;
        for (int i = 0; i < K; i++) {
            if (nbcolor[best][i] == 0) { c = i; break; }
        }
        if (c == -1) return false;
        color[best] = c;
        for (int u : adj[best]) {
            if (color[u] == -1) {
                if (nbcolor[u][c] == 0) sat[u]++;
                if (nbcolor[u][c] < 255) nbcolor[u][c]++;
            }
        }
    }
    return true;
}

// Tabu search (TabuCol) to remove conflicts
static bool tabu_search(double max_time) {
    mt19937 rng(0xC0FFEEu);
    if ((int)color.size() != N) color.assign(N, 0);
    for (int v = 0; v < N; v++) {
        if (color[v] < 0 || color[v] >= K) color[v] = (int)(rng() % (uint32_t)K);
    }

    // adj_color[v][c] = number of neighbors of v with color c
    vector<vector<int>> adj_color(N, vector<int>(K, 0));
    for (int v = 0; v < N; v++) {
        for (int u : adj[v]) {
            adj_color[v][color[u]]++;
        }
    }

    long long total = 0;
    for (int v = 0; v < N; v++) total += adj_color[v][color[v]];
    total /= 2;

    if (total == 0) return true;

    vector<vector<int>> tabu(N, vector<int>(K, 0));

    vector<int> conflict_list;
    vector<int> conflict_pos(N, -1);
    auto add_conflict = [&](int v) {
        if (conflict_pos[v] == -1) {
            conflict_pos[v] = (int)conflict_list.size();
            conflict_list.push_back(v);
        }
    };
    auto rem_conflict = [&](int v) {
        int p = conflict_pos[v];
        if (p != -1) {
            int last = conflict_list.back();
            conflict_list[p] = last;
            conflict_pos[last] = p;
            conflict_list.pop_back();
            conflict_pos[v] = -1;
        }
    };
    for (int v = 0; v < N; v++) {
        if (adj_color[v][color[v]] > 0) add_conflict(v);
    }

    long long best_total = total;
    int iter = 0;
    int stagnation = 0;

    while (total > 0 && elapsed() < max_time) {
        iter++;
        int best_v = -1, best_c = -1, best_delta = INT_MAX;
        int found_count = 0;

        for (size_t i = 0; i < conflict_list.size(); i++) {
            int v = conflict_list[i];
            int cur = adj_color[v][color[v]];
            for (int c = 0; c < K; c++) {
                if (c == color[v]) continue;
                int delta = adj_color[v][c] - cur;
                bool is_tabu = tabu[v][c] > iter;
                bool aspire = ((long long)total + delta) < best_total;
                if (is_tabu && !aspire) continue;
                if (delta < best_delta) {
                    best_delta = delta;
                    best_v = v;
                    best_c = c;
                    found_count = 1;
                } else if (delta == best_delta) {
                    found_count++;
                    if ((int)(rng() % (uint32_t)found_count) == 0) {
                        best_v = v;
                        best_c = c;
                    }
                }
            }
        }

        if (best_v == -1) {
            if (conflict_list.empty()) break;
            int v = conflict_list[rng() % conflict_list.size()];
            int c = (int)(rng() % (uint32_t)K);
            if (c == color[v]) c = (c + 1) % K;
            best_v = v; best_c = c;
            best_delta = adj_color[v][c] - adj_color[v][color[v]];
        }

        int old_c = color[best_v];
        color[best_v] = best_c;
        for (int u : adj[best_v]) {
            adj_color[u][old_c]--;
            adj_color[u][best_c]++;
            if (adj_color[u][color[u]] > 0) add_conflict(u);
            else rem_conflict(u);
        }
        if (adj_color[best_v][best_c] > 0) add_conflict(best_v);
        else rem_conflict(best_v);

        total += best_delta;
        int tenure = (int)(0.6 * (double)total) + 1 + (int)(rng() % 10);
        tabu[best_v][old_c] = iter + tenure;

        if (total < best_total) {
            best_total = total;
            stagnation = 0;
        } else {
            stagnation++;
        }

        if (stagnation > 50000) {
            stagnation = 0;
            int kick = max(1, (int)conflict_list.size() / 4);
            for (int j = 0; j < kick && !conflict_list.empty(); j++) {
                int v = conflict_list[rng() % conflict_list.size()];
                int new_c = (int)(rng() % (uint32_t)K);
                if (new_c == color[v]) continue;
                int oc = color[v];
                color[v] = new_c;
                for (int u : adj[v]) {
                    adj_color[u][oc]--;
                    adj_color[u][new_c]++;
                    if (adj_color[u][color[u]] > 0) add_conflict(u);
                    else rem_conflict(u);
                }
                if (adj_color[v][new_c] > 0) add_conflict(v);
                else rem_conflict(v);
            }
            long long t2 = 0;
            for (int v = 0; v < N; v++) t2 += adj_color[v][color[v]];
            total = t2 / 2;
            if (total < best_total) best_total = total;
        }
    }
    return total == 0;
}

static long long count_conflicts() {
    long long c = 0;
    for (int u = 0; u < N; u++) {
        for (int v : adj[u]) {
            if (u < v && color[u] == color[v]) c++;
        }
    }
    return c;
}

int main() {
    start_time = high_resolution_clock::now();
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> N >> M >> K)) return 1;
    if (N <= 0) { cout << '\n'; return 0; }
    if (K <= 0) K = 1;

    adj.assign(N, {});
    for (int i = 0; i < M; i++) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u < 0 || v < 0 || u >= N || v >= N) continue;
        if (u == v) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    for (int v = 0; v < N; v++) {
        sort(adj[v].begin(), adj[v].end());
        adj[v].erase(unique(adj[v].begin(), adj[v].end()), adj[v].end());
    }

    bool solved = false;

    // Strategy 1: For small graphs, DSATUR (good quality, O(N^2))
    if (!solved && N <= 1500 && elapsed() < 30.0) {
        if (dsatur() && count_conflicts() == 0) solved = true;
    }

    // Strategy 2: Smallest-last (degeneracy) ordering
    if (!solved && elapsed() < 90.0) {
        if (greedy_smallest_last() && count_conflicts() == 0) solved = true;
    }

    // Strategy 3: Largest-degree-first as another shot
    if (!solved && elapsed() < 120.0) {
        if (greedy_LDF() && count_conflicts() == 0) solved = true;
    }

    // Strategy 4: Tabu search on best initial coloring
    if (!solved) {
        double budget = 270.0;
        solved = tabu_search(budget);
    }

    if ((int)color.size() != N) color.assign(N, 0);
    for (int i = 0; i < N; i++) {
        if (color[i] < 0 || color[i] >= K) color[i] = 0;
    }

    string out;
    out.reserve((size_t)N * 4);
    char buf[16];
    for (int i = 0; i < N; i++) {
        if (i) out.push_back(' ');
        int len = snprintf(buf, sizeof(buf), "%d", color[i]);
        out.append(buf, len);
    }
    out.push_back('\n');
    fwrite(out.data(), 1, out.size(), stdout);
    return 0;
}
