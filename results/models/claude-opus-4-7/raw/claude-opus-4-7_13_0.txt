using namespace std;

// Two 31-bit primes combined into a 62-bit hash key for low collision probability.
constexpr uint64_t MOD_A = 2147483647ULL;   // 2^31 - 1 (Mersenne prime)
constexpr uint64_t MOD_B = 2147483629ULL;   // another large 31-bit prime
constexpr uint64_t BASE_A = 131;
constexpr uint64_t BASE_B = 137;

// When pairs has fewer than this many items, switch to hash-set-based intersection
// (avoids sorting 10M+ hashes of new strings when candidate set is already small).
constexpr size_t LARGE_THRESHOLD = 500000;

static uint64_t modpow(uint64_t base, uint64_t exp, uint64_t mod) {
    uint64_t r = 1 % mod;
    base %= mod;
    while (exp) {
        if (exp & 1) r = r * base % mod;
        base = base * base % mod;
        exp >>= 1;
    }
    return r;
}

// Compute combined 62-bit hash for every length-L substring of s.
static void compute_hashes(const string& s, int L, vector<uint64_t>& out) {
    int n = (int)s.size();
    out.clear();
    if (n < L) return;
    int m = n - L + 1;
    out.resize(m);

    uint64_t pow_a = modpow(BASE_A, (uint64_t)(L - 1), MOD_A);
    uint64_t pow_b = modpow(BASE_B, (uint64_t)(L - 1), MOD_B);

    const uint8_t* d = (const uint8_t*)s.data();

    uint64_t h_a = 0, h_b = 0;
    for (int i = 0; i < L; i++) {
        h_a = (h_a * BASE_A + d[i]) % MOD_A;
        h_b = (h_b * BASE_B + d[i]) % MOD_B;
    }
    out[0] = (h_a << 32) | h_b;

    for (int i = 1; i < m; i++) {
        h_a = (h_a + MOD_A - (uint64_t)d[i-1] * pow_a % MOD_A) % MOD_A;
        h_a = (h_a * BASE_A + d[i+L-1]) % MOD_A;
        h_b = (h_b + MOD_B - (uint64_t)d[i-1] * pow_b % MOD_B) % MOD_B;
        h_b = (h_b * BASE_B + d[i+L-1]) % MOD_B;
        out[i] = (h_a << 32) | h_b;
    }
}

// Returns a starting position in strings[base_idx] of a length-L substring
// that occurs in every string, or -1 if no such substring exists.
static int check(const vector<string>& strings, int base_idx, int L) {
    int N = (int)strings.size();
    if ((int)strings[base_idx].size() < L) return -1;

    vector<uint64_t> base_h;
    compute_hashes(strings[base_idx], L, base_h);
    int m = (int)base_h.size();
    if (m == 0) return -1;

    // (hash, position) pairs, sorted by hash with duplicates removed (smallest pos kept).
    vector<pair<uint64_t, int>> pairs(m);
    for (int i = 0; i < m; i++) pairs[i] = {base_h[i], i};
    base_h.clear();
    base_h.shrink_to_fit();

    sort(pairs.begin(), pairs.end());
    {
        size_t w = 0;
        for (size_t i = 0; i < pairs.size(); i++) {
            if (w == 0 || pairs[w-1].first != pairs[i].first) {
                pairs[w++] = pairs[i];
            }
        }
        pairs.resize(w);
    }

    vector<uint64_t> other_h;
    uint64_t pow_a = modpow(BASE_A, (uint64_t)(L - 1), MOD_A);
    uint64_t pow_b = modpow(BASE_B, (uint64_t)(L - 1), MOD_B);

    for (int idx = 0; idx < N; idx++) {
        if (idx == base_idx) continue;
        if ((int)strings[idx].size() < L) return -1;
        if (pairs.empty()) return -1;

        if (pairs.size() > LARGE_THRESHOLD) {
            // Sort-merge intersection: best when candidate set is still large.
            compute_hashes(strings[idx], L, other_h);
            sort(other_h.begin(), other_h.end());
            other_h.erase(unique(other_h.begin(), other_h.end()), other_h.end());

            size_t a = 0, b = 0, w = 0;
            while (a < pairs.size() && b < other_h.size()) {
                if (pairs[a].first < other_h[b]) a++;
                else if (pairs[a].first > other_h[b]) b++;
                else {
                    pairs[w++] = pairs[a];
                    a++; b++;
                }
            }
            pairs.resize(w);
        } else {
            // Hash-set lookup: avoids sorting the new string's 10M+ hashes.
            unordered_set<uint64_t> cand_set;
            cand_set.reserve(pairs.size() * 2);
            for (auto& p : pairs) cand_set.insert(p.first);

            unordered_set<uint64_t> found;
            found.reserve(pairs.size());

            const uint8_t* d = (const uint8_t*)strings[idx].data();
            int len = (int)strings[idx].size();
            uint64_t h_a = 0, h_b = 0;
            for (int i = 0; i < L; i++) {
                h_a = (h_a * BASE_A + d[i]) % MOD_A;
                h_b = (h_b * BASE_B + d[i]) % MOD_B;
            }
            uint64_t h = (h_a << 32) | h_b;
            if (cand_set.count(h)) found.insert(h);

            int last = len - L;
            for (int i = 1; i <= last; i++) {
                h_a = (h_a + MOD_A - (uint64_t)d[i-1] * pow_a % MOD_A) % MOD_A;
                h_a = (h_a * BASE_A + d[i+L-1]) % MOD_A;
                h_b = (h_b + MOD_B - (uint64_t)d[i-1] * pow_b % MOD_B) % MOD_B;
                h_b = (h_b * BASE_B + d[i+L-1]) % MOD_B;
                h = (h_a << 32) | h_b;
                if (cand_set.count(h)) found.insert(h);
            }

            size_t w = 0;
            for (size_t i = 0; i < pairs.size(); i++) {
                if (found.find(pairs[i].first) != found.end()) {
                    pairs[w++] = pairs[i];
                }
            }
            pairs.resize(w);
        }
    }

    if (pairs.empty()) return -1;
    int best_pos = pairs[0].second;
    for (size_t i = 1; i < pairs.size(); i++)
        if (pairs[i].second < best_pos) best_pos = pairs[i].second;
    return best_pos;
}

// Verify candidate substring really appears in every string (guards against the
// vanishingly small chance of a double-hash collision).
static bool verify_substring(const vector<string>& strings, int base_idx,
                             int pos, int L) {
    if (L <= 0) return true;
    const char* pat = strings[base_idx].data() + pos;
    for (int i = 0; i < (int)strings.size(); i++) {
        if (i == base_idx) continue;
        const string& s = strings[i];
        if ((int)s.size() < L) return false;
        auto it = std::search(s.begin(), s.end(), pat, pat + L);
        if (it == s.end()) return false;
    }
    return true;
}

int main() {
#ifdef _WIN32
    _setmode(_fileno(stdin), _O_BINARY);
    _setmode(_fileno(stdout), _O_BINARY);
#endif

    // Fast bulk read of stdin.
    vector<char> buf;
    buf.reserve(1 << 20);
    char tmp[1 << 16];
    size_t r;
    while ((r = fread(tmp, 1, sizeof(tmp), stdin)) > 0) {
        buf.insert(buf.end(), tmp, tmp + r);
    }

    size_t pos = 0;
    auto read_line = [&](string& out) {
        size_t start = pos;
        while (pos < buf.size() && buf[pos] != '\n') pos++;
        size_t end = pos;
        if (end > start && buf[end-1] == '\r') end--;
        out.assign(buf.data() + start, end - start);
        if (pos < buf.size()) pos++;
    };

    string line;
    read_line(line);
    int N = 0;
    try {
        N = stoi(line);
    } catch (...) {
        fputc('\n', stdout);
        return 0;
    }

    if (N <= 0) { fputc('\n', stdout); return 0; }

    vector<string> strings(N);
    for (int i = 0; i < N; i++) read_line(strings[i]);

    if (N == 1) {
        fwrite(strings[0].data(), 1, strings[0].size(), stdout);
        fputc('\n', stdout);
        return 0;
    }

    // Use the shortest string as the base — it has the fewest substrings to track.
    int base_idx = 0;
    int min_len = (int)strings[0].size();
    for (int i = 1; i < N; i++) {
        if ((int)strings[i].size() < min_len) {
            min_len = (int)strings[i].size();
            base_idx = i;
        }
    }

    if (min_len == 0) { fputc('\n', stdout); return 0; }

    int last_good_L = 0, last_good_pos = -1;
    int first_bad_L = min_len + 1;

    // Exponential probe: cheap when the answer is small (random/extreme case).
    int L = 1;
    while (L <= min_len) {
        int p = check(strings, base_idx, L);
        if (p >= 0) {
            last_good_L = L;
            last_good_pos = p;
            if (L == min_len) break;
            int nxt = (L > min_len / 2) ? min_len : L * 2;
            if (nxt <= L) nxt = L + 1;
            L = nxt;
        } else {
            first_bad_L = L;
            break;
        }
    }

    // Binary search inside the (last_good_L, first_bad_L) gap.
    int lo = last_good_L + 1, hi = first_bad_L - 1;
    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;
        int p = check(strings, base_idx, mid);
        if (p >= 0) {
            last_good_L = mid;
            last_good_pos = p;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    // Defensive verification — re-shrink only if a collision somehow fooled us.
    while (last_good_L > 0 && last_good_pos >= 0 &&
           !verify_substring(strings, base_idx, last_good_pos, last_good_L)) {
        last_good_L--;
        if (last_good_L <= 0) { last_good_pos = -1; break; }
        int p = check(strings, base_idx, last_good_L);
        if (p < 0) { last_good_L = 0; last_good_pos = -1; break; }
        last_good_pos = p;
    }

    if (last_good_L > 0 && last_good_pos >= 0) {
        fwrite(strings[base_idx].data() + last_good_pos, 1, last_good_L, stdout);
        fputc('\n', stdout);
    } else {
        fputc('\n', stdout);
    }

    return 0;
}
