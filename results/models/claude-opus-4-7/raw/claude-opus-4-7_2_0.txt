using System;
using System.Collections.Generic;
using System.IO;

class Solution {
    static byte[] _inbuf = new byte[1 << 16];
    static int _inLen = 0, _inPos = 0;
    static Stream _inStream;

    static int ReadByte() {
        if (_inPos >= _inLen) {
            _inLen = _inStream.Read(_inbuf, 0, _inbuf.Length);
            _inPos = 0;
            if (_inLen == 0) return -1;
        }
        return _inbuf[_inPos++];
    }

    static long ReadLong() {
        int c;
        do { c = ReadByte(); } while (c != -1 && (c < '0' || c > '9') && c != '-');
        if (c == -1) return 0;
        bool neg = false;
        if (c == '-') { neg = true; c = ReadByte(); }
        long v = 0;
        while (c >= '0' && c <= '9') {
            v = v * 10 + (c - '0');
            c = ReadByte();
        }
        return neg ? -v : v;
    }

    static int ReadInt() { return (int)ReadLong(); }

    static void Main() {
        _inStream = Console.OpenStandardInput();

        int N = ReadInt();
        int M = ReadInt();

        if (N <= 0) N = 1;

        var adj = new List<(int to, int eid, long w)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int to, int eid, long w)>();

        var edgeW = new List<long>(Math.Max(1, M * 2));

        for (int i = 0; i < M; i++) {
            int u = ReadInt();
            int v = ReadInt();
            long w = ReadLong();
            if (u < 0 || u >= N || v < 0 || v >= N) {
                edgeW.Add(0);
                continue;
            }
            edgeW.Add(w);
            adj[u].Add((v, i, w));
            adj[v].Add((u, i, w));
        }

        var oddVertices = new List<int>();
        for (int i = 0; i < N; i++) {
            if ((adj[i].Count & 1) == 1) oddVertices.Add(i);
        }

        int K = oddVertices.Count;
        int nextEid = edgeW.Count;

        if (K > 0) {
            if (K <= 18) {
                nextEid = ExactMatch(adj, oddVertices, edgeW, nextEid, N);
            } else {
                nextEid = SpanningTreeTJoin(adj, oddVertices, edgeW, nextEid, N);
            }
        }

        int totalEdges = nextEid;
        var used = new bool[totalEdges];
        var idx = new int[N];
        var stackArr = new int[totalEdges + 16];
        int stackTop = 0;
        stackArr[stackTop++] = 0;
        var route = new int[totalEdges + 2];
        int routeLen = 0;

        while (stackTop > 0) {
            int v = stackArr[stackTop - 1];
            var avs = adj[v];
            int avc = avs.Count;
            int iv = idx[v];
            while (iv < avc && used[avs[iv].eid]) iv++;
            if (iv < avc) {
                var edge = avs[iv];
                used[edge.eid] = true;
                idx[v] = iv + 1;
                stackArr[stackTop++] = edge.to;
            } else {
                idx[v] = iv;
                route[routeLen++] = stackArr[--stackTop];
            }
        }

        var outStream = Console.OpenStandardOutput();
        int outBufSize = Math.Max(64, (routeLen + 4) * 12);
        var outBuf = new byte[outBufSize];
        int outLen = 0;

        outLen += WriteUIntToBuf(routeLen, outBuf, outLen);
        outBuf[outLen++] = (byte)'\n';

        for (int i = routeLen - 1; i >= 0; i--) {
            if (i < routeLen - 1) outBuf[outLen++] = (byte)' ';
            outLen += WriteUIntToBuf(route[i], outBuf, outLen);
        }
        outBuf[outLen++] = (byte)'\n';

        outStream.Write(outBuf, 0, outLen);
        outStream.Flush();
    }

    static int WriteUIntToBuf(int x, byte[] buf, int pos) {
        if (x == 0) {
            buf[pos] = (byte)'0';
            return 1;
        }
        int startPos = pos;
        while (x > 0) {
            buf[pos++] = (byte)('0' + x % 10);
            x /= 10;
        }
        int written = pos - startPos;
        int half = written / 2;
        for (int i = 0; i < half; i++) {
            byte tmp = buf[startPos + i];
            buf[startPos + i] = buf[pos - 1 - i];
            buf[pos - 1 - i] = tmp;
        }
        return written;
    }

    static int ExactMatch(List<(int to, int eid, long w)>[] adj, List<int> oddVertices, List<long> edgeW, int nextEid, int N) {
        int K = oddVertices.Count;
        var dist = new long[K][];
        var parent = new int[K][];
        var parentEdge = new int[K][];
        for (int i = 0; i < K; i++) {
            (dist[i], parent[i], parentEdge[i]) = Dijkstra(adj, oddVertices[i], N);
        }

        int full = 1 << K;
        var dp = new long[full];
        var dpChoice = new int[full];
        for (int i = 0; i < full; i++) dp[i] = long.MaxValue;
        dp[0] = 0;

        for (int mask = 0; mask < full; mask++) {
            if (dp[mask] == long.MaxValue) continue;
            int i = 0;
            while (i < K && (mask & (1 << i)) != 0) i++;
            if (i == K) continue;
            long baseCost = dp[mask];
            var distI = dist[i];
            for (int j = i + 1; j < K; j++) {
                if ((mask & (1 << j)) != 0) continue;
                long d = distI[oddVertices[j]];
                if (d == long.MaxValue) continue;
                int newMask = mask | (1 << i) | (1 << j);
                long newCost = baseCost + d;
                if (newCost < dp[newMask]) {
                    dp[newMask] = newCost;
                    dpChoice[newMask] = (i << 16) | j;
                }
            }
        }

        int curMask = full - 1;
        while (curMask != 0) {
            int choice = dpChoice[curMask];
            int i = choice >> 16;
            int j = choice & 0xFFFF;
            int prevMask = curMask ^ ((1 << i) | (1 << j));
            int src = oddVertices[i];
            int dst = oddVertices[j];
            int cur = dst;
            int safety = 0;
            while (cur != src && cur != -1 && safety++ < N + 1) {
                int p = parent[i][cur];
                if (p == -1) break;
                int eid = parentEdge[i][cur];
                long w = edgeW[eid];
                int newEid = nextEid++;
                edgeW.Add(w);
                adj[p].Add((cur, newEid, w));
                adj[cur].Add((p, newEid, w));
                cur = p;
            }
            curMask = prevMask;
        }
        return nextEid;
    }

    static (long[], int[], int[]) Dijkstra(List<(int to, int eid, long w)>[] adj, int src, int N) {
        var dist = new long[N];
        var parent = new int[N];
        var parentEdge = new int[N];
        for (int i = 0; i < N; i++) {
            dist[i] = long.MaxValue;
            parent[i] = -1;
            parentEdge[i] = -1;
        }
        dist[src] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);
        while (pq.Count > 0) {
            int u;
            long d;
            pq.TryDequeue(out u, out d);
            if (d > dist[u]) continue;
            var au = adj[u];
            int auc = au.Count;
            for (int k = 0; k < auc; k++) {
                var edge = au[k];
                long nd = d + edge.w;
                if (nd < dist[edge.to]) {
                    dist[edge.to] = nd;
                    parent[edge.to] = u;
                    parentEdge[edge.to] = edge.eid;
                    pq.Enqueue(edge.to, nd);
                }
            }
        }
        return (dist, parent, parentEdge);
    }

    static int SpanningTreeTJoin(List<(int to, int eid, long w)>[] adj, List<int> oddVertices, List<long> edgeW, int nextEid, int N) {
        var isOdd = new bool[N];
        foreach (var v in oddVertices) isOdd[v] = true;

        var parent = new int[N];
        var parentEdge = new int[N];
        for (int i = 0; i < N; i++) {
            parent[i] = -1;
            parentEdge[i] = -1;
        }
        var visited = new bool[N];
        var order = new int[N];
        int orderLen = 0;
        var bfsQueue = new int[N];
        int qHead = 0, qTail = 0;

        visited[0] = true;
        bfsQueue[qTail++] = 0;
        order[orderLen++] = 0;
        while (qHead < qTail) {
            int u = bfsQueue[qHead++];
            var au = adj[u];
            int auc = au.Count;
            for (int k = 0; k < auc; k++) {
                var edge = au[k];
                if (!visited[edge.to]) {
                    visited[edge.to] = true;
                    parent[edge.to] = u;
                    parentEdge[edge.to] = edge.eid;
                    bfsQueue[qTail++] = edge.to;
                    order[orderLen++] = edge.to;
                }
            }
        }

        // Handle any unvisited nodes (disconnected components) as separate trees
        for (int s = 0; s < N; s++) {
            if (visited[s]) continue;
            visited[s] = true;
            bfsQueue[qTail++] = s;
            order[orderLen++] = s;
            while (qHead < qTail) {
                int u = bfsQueue[qHead++];
                var au = adj[u];
                int auc = au.Count;
                for (int k = 0; k < auc; k++) {
                    var edge = au[k];
                    if (!visited[edge.to]) {
                        visited[edge.to] = true;
                        parent[edge.to] = u;
                        parentEdge[edge.to] = edge.eid;
                        bfsQueue[qTail++] = edge.to;
                        order[orderLen++] = edge.to;
                    }
                }
            }
        }

        var oddCount = new int[N];
        for (int i = 0; i < N; i++) if (isOdd[i]) oddCount[i] = 1;
        for (int i = orderLen - 1; i >= 0; i--) {
            int v = order[i];
            if (parent[v] != -1) {
                oddCount[parent[v]] += oddCount[v];
            }
        }

        for (int v = 0; v < N; v++) {
            if (parent[v] == -1) continue;
            if ((oddCount[v] & 1) == 1) {
                int eid = parentEdge[v];
                int u = parent[v];
                long w = edgeW[eid];
                int newEid = nextEid++;
                edgeW.Add(w);
                adj[u].Add((v, newEid, w));
                adj[v].Add((u, newEid, w));
            }
        }
        return nextEid;
    }
}
