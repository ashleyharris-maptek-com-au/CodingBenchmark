; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %outColor Location 0

               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
     %fnVoid = OpTypeFunction %void
      %float = OpTypeFloat 32
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4f 4
        %int = OpTypeInt 32 1
        %iv2 = OpTypeVector %int 2
       %bool = OpTypeBool
        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4f = OpTypePointer Uniform %v4f
%_ptr_Input_v3f = OpTypePointer Input %v3f
%_ptr_Input_v2f = OpTypePointer Input %v2f
%_ptr_Output_v4f = OpTypePointer Output %v4f
        %ubo = OpVariable %_ptr_Uniform_UBO Uniform
   %worldPos = OpVariable %_ptr_Input_v3f Input
     %normal = OpVariable %_ptr_Input_v3f Input
         %uv = OpVariable %_ptr_Input_v2f Input
   %outColor = OpVariable %_ptr_Output_v4f Output

       %f_0  = OpConstant %float 0.0
      %f_05  = OpConstant %float 0.5
       %f_1  = OpConstant %float 1.0
       %f_8  = OpConstant %float 8.0
      %f_09  = OpConstant %float 0.9
      %f_02  = OpConstant %float 0.2
       %i_0  = OpConstant %int 0
       %i_2  = OpConstant %int 2
       %i_3  = OpConstant %int 3

       %main = OpFunction %void None %fnVoid
      %entry = OpLabel

     %uv_val = OpLoad %v2f %uv
     %scaled = OpVectorTimesScalar %v2f %uv_val %f_8
    %floored = OpExtInst %v2f %1 Floor %scaled
       %ivec = OpConvertFToS %iv2 %floored
         %xi = OpCompositeExtract %int %ivec 0
         %yi = OpCompositeExtract %int %ivec 1
        %sum = OpIAdd %int %xi %yi
     %parity = OpSMod %int %sum %i_2
     %iseven = OpIEqual %bool %parity %i_0
    %checker = OpSelect %float %iseven %f_09 %f_02

       %nval = OpLoad %v3f %normal
      %nnorm = OpExtInst %v3f %1 Normalize %nval
       %wpos = OpLoad %v3f %worldPos
   %lpos_ptr = OpAccessChain %_ptr_Uniform_v4f %ubo %i_3
       %lpos = OpLoad %v4f %lpos_ptr
       %lxyz = OpVectorShuffle %v3f %lpos %lpos 0 1 2
       %ldir = OpFSub %v3f %lxyz %wpos
      %lnorm = OpExtInst %v3f %1 Normalize %ldir
          %d = OpDot %float %nnorm %lnorm
     %dclamp = OpExtInst %float %1 FMax %d %f_0
      %halfd = OpFMul %float %f_05 %dclamp
     %factor = OpFAdd %float %f_05 %halfd
        %lit = OpFMul %float %checker %factor
      %v4out = OpCompositeConstruct %v4f %lit %lit %lit %f_1

               OpStore %outColor %v4out
               OpReturn
               OpFunctionEnd
