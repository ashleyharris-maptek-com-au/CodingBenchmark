use std::io::{self, Read, Write, BufWriter};

// Fast xorshift PRNG
#[inline]
fn xor_next(state: &mut u64) -> u64 {
    let mut x = *state;
    x ^= x << 13;
    x ^= x >> 7;
    x ^= x << 17;
    *state = x;
    x
}

#[inline]
fn rand_f64(state: &mut u64) -> f64 {
    (xor_next(state) >> 11) as f64 / 9007199254740992.0
}

// Barnes-Hut quadtree stored in SoA layout
struct BHTree {
    cx: Vec<f64>,
    cy: Vec<f64>,
    half: Vec<f64>,
    com_x: Vec<f64>,
    com_y: Vec<f64>,
    mass: Vec<f64>,
    children: Vec<[i32; 4]>,
    point: Vec<i32>,
}

impl BHTree {
    fn new() -> Self {
        BHTree {
            cx: Vec::new(),
            cy: Vec::new(),
            half: Vec::new(),
            com_x: Vec::new(),
            com_y: Vec::new(),
            mass: Vec::new(),
            children: Vec::new(),
            point: Vec::new(),
        }
    }

    fn clear(&mut self) {
        self.cx.clear();
        self.cy.clear();
        self.half.clear();
        self.com_x.clear();
        self.com_y.clear();
        self.mass.clear();
        self.children.clear();
        self.point.clear();
    }

    #[inline]
    fn add_node(&mut self, cx: f64, cy: f64, half: f64) -> usize {
        let idx = self.cx.len();
        self.cx.push(cx);
        self.cy.push(cy);
        self.half.push(half);
        self.com_x.push(0.0);
        self.com_y.push(0.0);
        self.mass.push(0.0);
        self.children.push([-1i32; 4]);
        self.point.push(-1);
        idx
    }

    #[inline]
    fn quadrant(&self, node: usize, x: f64, y: f64) -> usize {
        let r = (x >= self.cx[node]) as usize;
        let t = (y >= self.cy[node]) as usize;
        r | (t << 1)
    }

    #[inline]
    fn quad_center(&self, node: usize, q: usize) -> (f64, f64, f64) {
        let h = self.half[node] * 0.5;
        let dx = if q & 1 != 0 { h } else { -h };
        let dy = if q & 2 != 0 { h } else { -h };
        (self.cx[node] + dx, self.cy[node] + dy, h)
    }

    fn build(&mut self, positions: &[(f64, f64)]) {
        self.clear();
        if positions.is_empty() {
            return;
        }

        let mut minx = positions[0].0;
        let mut maxx = positions[0].0;
        let mut miny = positions[0].1;
        let mut maxy = positions[0].1;
        for &(x, y) in positions.iter().skip(1) {
            if x < minx { minx = x; }
            if x > maxx { maxx = x; }
            if y < miny { miny = y; }
            if y > maxy { maxy = y; }
        }
        let cx = (minx + maxx) * 0.5;
        let cy = (miny + maxy) * 0.5;
        let half = ((maxx - minx).max(maxy - miny) * 0.5 + 1.0).max(1.0);

        self.add_node(cx, cy, half);
        for i in 0..positions.len() {
            self.insert_point(positions, i as i32);
        }
    }

    fn insert_point(&mut self, positions: &[(f64, f64)], point_idx: i32) {
        let (px, py) = positions[point_idx as usize];
        let mut node = 0usize;
        loop {
            // Empty node: store point and stop.
            if self.mass[node] == 0.0 {
                self.point[node] = point_idx;
                self.mass[node] = 1.0;
                self.com_x[node] = px;
                self.com_y[node] = py;
                return;
            }

            // Reached depth/precision limit: accumulate mass at this node.
            if self.half[node] < 1e-7 {
                let nm = self.mass[node] + 1.0;
                self.com_x[node] = (self.com_x[node] * self.mass[node] + px) / nm;
                self.com_y[node] = (self.com_y[node] * self.mass[node] + py) / nm;
                self.mass[node] = nm;
                return;
            }

            // Single-point leaf: subdivide.
            if self.point[node] != -1 {
                let existing = self.point[node];
                let (ex, ey) = positions[existing as usize];
                self.point[node] = -1;

                let q_ex = self.quadrant(node, ex, ey);
                let (ccx, ccy, ch) = self.quad_center(node, q_ex);
                let child = self.add_node(ccx, ccy, ch);
                self.children[node][q_ex] = child as i32;
                self.point[child] = existing;
                self.mass[child] = 1.0;
                self.com_x[child] = ex;
                self.com_y[child] = ey;
                // Fallthrough: node is now internal, insert px,py below.
            }

            // Internal node: update COM and descend.
            let nm = self.mass[node] + 1.0;
            self.com_x[node] = (self.com_x[node] * self.mass[node] + px) / nm;
            self.com_y[node] = (self.com_y[node] * self.mass[node] + py) / nm;
            self.mass[node] = nm;

            let q = self.quadrant(node, px, py);
            if self.children[node][q] == -1 {
                let (ccx, ccy, ch) = self.quad_center(node, q);
                let new_child = self.add_node(ccx, ccy, ch);
                self.children[node][q] = new_child as i32;
                self.point[new_child] = point_idx;
                self.mass[new_child] = 1.0;
                self.com_x[new_child] = px;
                self.com_y[new_child] = py;
                return;
            }
            node = self.children[node][q] as usize;
        }
    }

    fn force(&self, x: f64, y: f64, theta2: f64, k2: f64, stack: &mut Vec<i32>) -> (f64, f64) {
        let mut fx = 0.0;
        let mut fy = 0.0;
        stack.clear();
        stack.push(0);
        while let Some(node_i) = stack.pop() {
            let node = node_i as usize;
            let m = self.mass[node];
            if m == 0.0 {
                continue;
            }
            let dx = x - self.com_x[node];
            let dy = y - self.com_y[node];
            let d2 = dx * dx + dy * dy + 1e-9;
            let s = self.half[node] * 2.0;

            if self.point[node] != -1 || s * s < theta2 * d2 {
                // Repulsive force k^2 * m / d as a vector: dx * k^2 * m / d^2.
                let factor = k2 * m / d2;
                fx += dx * factor;
                fy += dy * factor;
            } else {
                let c = self.children[node];
                if c[0] != -1 { stack.push(c[0]); }
                if c[1] != -1 { stack.push(c[1]); }
                if c[2] != -1 { stack.push(c[2]); }
                if c[3] != -1 { stack.push(c[3]); }
            }
        }
        (fx, fy)
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut tokens = input.split_ascii_whitespace();
    let n: usize = tokens.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    let m: usize = tokens.next().and_then(|s| s.parse().ok()).unwrap_or(0);

    if n == 0 {
        return;
    }

    // Read edges. Skip self-loops and out-of-range indices. Parallel edges are
    // kept since their cumulative attraction is standard FR behavior.
    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = match tokens.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        let v: usize = match tokens.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        edges.push((u as u32, v as u32));
    }

    // Trivial case.
    if n == 1 {
        writeln!(out, "{}", n).unwrap();
        writeln!(out, "0.000000 0.000000").unwrap();
        return;
    }

    // Random initial placement on a disk of radius ~ sqrt(N) * 5.
    let mut positions: Vec<(f64, f64)> = vec![(0.0, 0.0); n];
    let mut rng: u64 = 0xCAFEBABE12345678u64
        .wrapping_add(n as u64)
        .wrapping_add(m as u64);
    if rng == 0 {
        rng = 0xDEADBEEF;
    }
    let target_radius = (n as f64).sqrt() * 5.0;
    for i in 0..n {
        let a = rand_f64(&mut rng) * 2.0 * std::f64::consts::PI;
        let r = rand_f64(&mut rng).sqrt() * target_radius;
        positions[i] = (r * a.cos(), r * a.sin());
    }

    // Adaptive iteration count: small graphs converge fully, huge graphs stay
    // well under the 30 s budget.
    let iter_count: usize = if n <= 30 {
        800
    } else if n <= 100 {
        500
    } else if n <= 500 {
        300
    } else if n <= 2000 {
        200
    } else if n <= 10000 {
        100
    } else if n <= 50000 {
        60
    } else {
        40
    };

    // Switch to Barnes-Hut once exact O(N^2) repulsion is too expensive.
    let use_bh = n > 400;
    let theta: f64 = 1.2;
    let theta2 = theta * theta;

    // Standard Fruchterman-Reingold ideal edge length.
    let area = target_radius * target_radius;
    let k = (area / n as f64).sqrt();
    let k2 = k * k;

    let initial_t = k * 5.0;
    let final_t = k * 0.01;

    let mut tree = BHTree::new();
    let mut stack: Vec<i32> = Vec::with_capacity(256);
    let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

    let bound = target_radius * 3.0;

    for it in 0..iter_count {
        for d in disp.iter_mut() {
            *d = (0.0, 0.0);
        }

        // Linear cooling.
        let frac = it as f64 / iter_count as f64;
        let t = initial_t * (1.0 - frac) + final_t * frac;

        // Repulsive forces.
        if use_bh {
            tree.build(&positions);
            for i in 0..n {
                let (fx, fy) = tree.force(positions[i].0, positions[i].1, theta2, k2, &mut stack);
                disp[i].0 += fx;
                disp[i].1 += fy;
            }
        } else {
            for i in 0..n {
                let (xi, yi) = positions[i];
                for j in (i + 1)..n {
                    let dx = xi - positions[j].0;
                    let dy = yi - positions[j].1;
                    let d2 = dx * dx + dy * dy + 1e-9;
                    let factor = k2 / d2;
                    let fx = dx * factor;
                    let fy = dy * factor;
                    disp[i].0 += fx;
                    disp[i].1 += fy;
                    disp[j].0 -= fx;
                    disp[j].1 -= fy;
                }
            }
        }

        // Attractive forces along edges. Magnitude d^2/k toward partner;
        // vector form is dx * d / k.
        for &(uu, vv) in &edges {
            let u = uu as usize;
            let v = vv as usize;
            let dx = positions[u].0 - positions[v].0;
            let dy = positions[u].1 - positions[v].1;
            let d2 = dx * dx + dy * dy + 1e-9;
            let d = d2.sqrt();
            let factor = d / k;
            let fx = dx * factor;
            let fy = dy * factor;
            disp[u].0 -= fx;
            disp[u].1 -= fy;
            disp[v].0 += fx;
            disp[v].1 += fy;
        }

        // Gentle gravity toward the origin keeps disconnected components from
        // drifting to infinity.
        let gravity = 0.02 * k;
        for i in 0..n {
            let dx = positions[i].0;
            let dy = positions[i].1;
            let dd = (dx * dx + dy * dy).sqrt() + 1e-9;
            disp[i].0 -= dx / dd * gravity;
            disp[i].1 -= dy / dd * gravity;
        }

        // Apply displacement with per-step temperature cap and bounding box.
        for i in 0..n {
            let dx = disp[i].0;
            let dy = disp[i].1;
            let d = (dx * dx + dy * dy).sqrt();
            if d > 1e-12 {
                let step = d.min(t);
                let nx = positions[i].0 + dx / d * step;
                let ny = positions[i].1 + dy / d * step;
                positions[i].0 = nx.max(-bound).min(bound);
                positions[i].1 = ny.max(-bound).min(bound);
            }
        }
    }

    // Guard against any non-finite coordinates.
    for i in 0..n {
        if !positions[i].0.is_finite() {
            positions[i].0 = 0.0;
        }
        if !positions[i].1.is_finite() {
            positions[i].1 = 0.0;
        }
    }

    writeln!(out, "{}", n).unwrap();
    for &(x, y) in &positions {
        writeln!(out, "{:.6} {:.6}", x, y).unwrap();
    }
}
