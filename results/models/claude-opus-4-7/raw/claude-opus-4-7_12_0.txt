using System;
using System.Globalization;
using System.IO;
using System.Text;

class BinPacking
{
    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    struct Quat
    {
        public double W, X, Y, Z;
        public Quat(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
    }

    // Rotate vector v by unit quaternion q (q * v * q^-1)
    static Vec3 RotateVector(Quat q, Vec3 v)
    {
        double tx = 2.0 * (q.Y * v.Z - q.Z * v.Y);
        double ty = 2.0 * (q.Z * v.X - q.X * v.Z);
        double tz = 2.0 * (q.X * v.Y - q.Y * v.X);
        double rx = v.X + q.W * tx + (q.Y * tz - q.Z * ty);
        double ry = v.Y + q.W * ty + (q.Z * tx - q.X * tz);
        double rz = v.Z + q.W * tz + (q.X * ty - q.Y * tx);
        return new Vec3(rx, ry, rz);
    }

    static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        var stdin = Console.In;

        // ------------------ Parse input ------------------
        string firstLine = stdin.ReadLine();
        var parts0 = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int V = int.Parse(parts0[0], ci);
        int F = int.Parse(parts0[1], ci);

        var verts = new Vec3[V];
        for (int i = 0; i < V; i++)
        {
            var p = stdin.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            verts[i] = new Vec3(
                double.Parse(p[0], ci),
                double.Parse(p[1], ci),
                double.Parse(p[2], ci));
        }
        // Skip face lines; we use AABB-based packing so faces aren't needed.
        for (int i = 0; i < F; i++) stdin.ReadLine();

        string lastLine = stdin.ReadLine();
        var pl = lastLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        double W = double.Parse(pl[0], ci);
        double H = double.Parse(pl[1], ci);
        double D = double.Parse(pl[2], ci);

        // ------------------ Strategy ------------------
        // Approach: For ANY polyhedron, the conservative axis-aligned bounding box
        // (AABB) gives a guaranteed valid no-overlap placement when tiled on a
        // regular grid. We try the 6 distinct axis-aligned orientations (each
        // a permutation of the polyhedron's AABB extents) and pick the orientation
        // that yields the most copies fitting inside the container.
        //
        // For each orientation we use a known rotation quaternion. We then re-derive
        // the rotated polyhedron's AABB from rotated vertices, so the placement
        // bounds are exact - even for non-box polyhedra, the result is a *valid*
        // (non-overlapping, in-container) placement set.
        //
        // Complexity: O(V * 6) for AABB evaluation; output is O(count) with
        // buffered I/O. Scales from trivial to ludicrous container sizes well
        // under the 5-minute budget.

        double s = 0.7071067811865476; // sin/cos 45°

        // 6 quaternions covering the 6 distinct dimension permutations
        var quats = new Quat[]
        {
            new Quat(1, 0, 0, 0),        // identity                -> (dx, dy, dz)
            new Quat(s, 0, 0, s),        // +90° about Z            -> (dy, dx, dz)
            new Quat(s, s, 0, 0),        // +90° about X            -> (dx, dz, dy)
            new Quat(s, 0, s, 0),        // +90° about Y            -> (dz, dy, dx)
            new Quat(0.5,  0.5,  0.5,  0.5),  // x->y, y->z, z->x   -> (dz, dx, dy)
            new Quat(0.5, -0.5, -0.5, -0.5)   // x->z, y->x, z->y   -> (dy, dz, dx)
        };

        long bestCount = 0;
        double bestT0X = 0, bestT0Y = 0, bestT0Z = 0;
        double bestSX = 1, bestSY = 1, bestSZ = 1;
        long bestNx = 0, bestNy = 0, bestNz = 0;
        Quat bestQ = quats[0];

        const double EPS = 1e-9;

        for (int qi = 0; qi < quats.Length; qi++)
        {
            Quat q = quats[qi];

            double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
            double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;

            for (int i = 0; i < V; i++)
            {
                var r = RotateVector(q, verts[i]);
                if (r.X < minX) minX = r.X;
                if (r.Y < minY) minY = r.Y;
                if (r.Z < minZ) minZ = r.Z;
                if (r.X > maxX) maxX = r.X;
                if (r.Y > maxY) maxY = r.Y;
                if (r.Z > maxZ) maxZ = r.Z;
            }

            double sx = maxX - minX;
            double sy = maxY - minY;
            double sz = maxZ - minZ;

            if (sx <= EPS || sy <= EPS || sz <= EPS) continue;
            if (sx > W + EPS || sy > H + EPS || sz > D + EPS) continue;

            long nx = (long)Math.Floor(W / sx + EPS);
            long ny = (long)Math.Floor(H / sy + EPS);
            long nz = (long)Math.Floor(D / sz + EPS);
            if (nx < 1 || ny < 1 || nz < 1) continue;

            long count;
            try
            {
                checked { count = nx * ny * nz; }
            }
            catch (OverflowException)
            {
                count = long.MaxValue;
            }

            if (count > bestCount)
            {
                bestCount = count;
                bestT0X = -minX;
                bestT0Y = -minY;
                bestT0Z = -minZ;
                bestSX = sx;
                bestSY = sy;
                bestSZ = sz;
                bestNx = nx;
                bestNy = ny;
                bestNz = nz;
                bestQ = q;
            }
        }

        // ------------------ Output safety cap ------------------
        // Hard upper bound on emitted placements - protects against absurd
        // "ludicrous" cases (e.g. tiny polyhedron in a 500^3 box) where writing
        // hundreds of millions of lines would blow the 5-minute time budget.
        // 10M lines is comfortably writable with buffered output.
        const long MAX_OUTPUT = 10_000_000L;
        long outCount = bestCount;
        if (outCount > MAX_OUTPUT) outCount = MAX_OUTPUT;

        // ------------------ Write output ------------------
        var outStream = Console.OpenStandardOutput();
        using (var bw = new BufferedStream(outStream, 1 << 20))
        using (var sw = new StreamWriter(bw, new UTF8Encoding(false)))
        {
            sw.WriteLine(outCount.ToString(ci));

            if (outCount == 0)
            {
                sw.Flush();
                return;
            }

            string qw = bestQ.W.ToString("G17", ci);
            string qx = bestQ.X.ToString("G17", ci);
            string qy = bestQ.Y.ToString("G17", ci);
            string qz = bestQ.Z.ToString("G17", ci);

            long emitted = 0;
            for (long i = 0; i < bestNx && emitted < outCount; i++)
            {
                double tx0 = i * bestSX + bestT0X;
                for (long j = 0; j < bestNy && emitted < outCount; j++)
                {
                    double ty0 = j * bestSY + bestT0Y;
                    for (long k = 0; k < bestNz && emitted < outCount; k++)
                    {
                        double tz0 = k * bestSZ + bestT0Z;

                        sw.Write(tx0.ToString("G17", ci));
                        sw.Write(' ');
                        sw.Write(ty0.ToString("G17", ci));
                        sw.Write(' ');
                        sw.Write(tz0.ToString("G17", ci));
                        sw.Write(' ');
                        sw.Write(qw);
                        sw.Write(' ');
                        sw.Write(qx);
                        sw.Write(' ');
                        sw.Write(qy);
                        sw.Write(' ');
                        sw.Write(qz);
                        sw.Write('\n');

                        emitted++;
                    }
                }
            }

            sw.Flush();
        }
    }
}
