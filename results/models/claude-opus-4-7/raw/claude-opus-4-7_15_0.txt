use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut iter = input.split_ascii_whitespace();
    let nv: usize = iter.next().unwrap().parse().unwrap();
    let mut poly: Vec<(f64, f64)> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = iter.next().unwrap().parse().unwrap();
        let y: f64 = iter.next().unwrap().parse().unwrap();
        poly.push((x, y));
    }
    let sx: f64 = iter.next().unwrap().parse().unwrap();
    let sy: f64 = iter.next().unwrap().parse().unwrap();
    let sz: f64 = iter.next().unwrap().parse().unwrap();

    // Compute polygon bounding box
    let mut xmin = f64::INFINITY;
    let mut xmax = f64::NEG_INFINITY;
    let mut ymin = f64::INFINITY;
    let mut ymax = f64::NEG_INFINITY;
    for &(x, y) in &poly {
        if x < xmin { xmin = x; }
        if x > xmax { xmax = x; }
        if y < ymin { ymin = y; }
        if y > ymax { ymax = y; }
    }

    let w = (xmax - xmin).max(0.0);
    let h = (ymax - ymin).max(0.0);
    let cx_bbox = (xmin + xmax) * 0.5;
    let cy_bbox = (ymin + ymax) * 0.5;

    // For projection along sun direction onto z=0:
    // A point (vx,vy,vz) projects to (vx - vz*sx/sz, vy - vz*sy/sz)
    let ax = if sz.abs() > 1e-12 { sx / sz } else { 0.0 };
    let ay = if sz.abs() > 1e-12 { sy / sz } else { 0.0 };

    // For an identity-rotation tetrahedron at center (px,py,pz) with scale s,
    // shadow centroid lies at (px - pz*ax, py - pz*ay), so we can place the
    // tetrahedron at any pz and compensate (px,py) to put its shadow at (X,Y).
    // Shadow vertices (relative to centroid) for identity rotation form a
    // diamond-like kite of "radius" s; distorted by tilt of the sun direction.

    let tilt = (ax * ax + ay * ay).sqrt();
    let safety = 1.0 + 0.7 * tilt;

    // Single diamond of radius s covers bbox WxH when s >= (W+H)/2.
    let single_s = ((w + h) * 0.5) * safety + 0.25;

    if w + h < 1e-9 {
        // Degenerate polygon (point)
        writeln!(out, "1").unwrap();
        writeln!(out, "{} {} 0 1 0 0 0 1.0", cx_bbox, cy_bbox).unwrap();
        return;
    }

    // Adaptive strategy: single tetrahedron if it fits within scale 10, else tile.
    if single_s <= 10.0 {
        let s = single_s.min(10.0).max(0.25);
        let pz = 0.0_f64;
        let px = cx_bbox + pz * ax;
        let py = cy_bbox + pz * ay;
        writeln!(out, "1").unwrap();
        writeln!(out, "{} {} {} 1 0 0 0 {}", px, py, pz, s).unwrap();
    } else {
        // Tile with a grid of tetrahedrons. Each tet uses a unique z-level
        // (z-separation > 1.414*s) so they cannot intersect in 3D regardless
        // of (x,y). For each tet, (px,py) is offset by pz*(ax,ay) so its
        // shadow lands on the desired grid cell.
        let s = 9.0_f64;
        // Each diamond of radius s covers axis-aligned square of side s.
        let tile = (s / safety).max(0.5);

        let nx = ((w / tile).ceil() as usize).max(1);
        let ny = ((h / tile).ceil() as usize).max(1);

        let step_x = w / (nx as f64);
        let step_y = h / (ny as f64);

        let total = nx * ny;
        writeln!(out, "{}", total).unwrap();

        let z_step = 1.5 * s;
        let mut idx: usize = 0;
        for j in 0..ny {
            for i in 0..nx {
                let shadow_x = xmin + (i as f64 + 0.5) * step_x;
                let shadow_y = ymin + (j as f64 + 0.5) * step_y;
                let pz = (idx as f64 + 1.0) * z_step;
                let px = shadow_x + pz * ax;
                let py = shadow_y + pz * ay;
                writeln!(out, "{} {} {} 1 0 0 0 {}", px, py, pz, s).unwrap();
                idx += 1;
            }
        }
    }
}
