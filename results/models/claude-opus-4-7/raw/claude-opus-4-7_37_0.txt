using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

class Program {
    static int n;
    static int[][] succArr, predArr;
    static bool[] removed;
    static int[] outDeg, inDeg;
    static List<int> currentSol;
    static List<int> bestSol;

    sealed class FastReader {
        Stream stream;
        byte[] buffer = new byte[1 << 16];
        int bufPos, bufLen;

        public FastReader() { stream = Console.OpenStandardInput(); }

        int Read() {
            if (bufPos >= bufLen) {
                bufLen = stream.Read(buffer, 0, buffer.Length);
                bufPos = 0;
                if (bufLen == 0) return -1;
            }
            return buffer[bufPos++];
        }

        public int ReadInt() {
            int c;
            do { c = Read(); } while (c != -1 && (c < '0' || c > '9') && c != '-');
            if (c == -1) return 0;
            bool neg = false;
            if (c == '-') { neg = true; c = Read(); }
            int val = 0;
            while (c >= '0' && c <= '9') {
                val = val * 10 + c - '0';
                c = Read();
            }
            return neg ? -val : val;
        }
    }

    static void Main() {
        var reader = new FastReader();
        n = reader.ReadInt();
        int m = reader.ReadInt();

        var succSet = new HashSet<int>[n];
        var predSet = new HashSet<int>[n];
        for (int i = 0; i < n; i++) {
            succSet[i] = new HashSet<int>();
            predSet[i] = new HashSet<int>();
        }
        removed = new bool[n];
        var solution = new List<int>();

        for (int i = 0; i < m; i++) {
            int u = reader.ReadInt();
            int v = reader.ReadInt();
            if (u == v) {
                if (!removed[u]) {
                    removed[u] = true;
                    solution.Add(u);
                }
                continue;
            }
            succSet[u].Add(v);
            predSet[v].Add(u);
        }

        succArr = new int[n][];
        predArr = new int[n][];
        for (int i = 0; i < n; i++) {
            succArr[i] = succSet[i].ToArray();
            predArr[i] = predSet[i].ToArray();
        }
        succSet = null;
        predSet = null;

        outDeg = new int[n];
        inDeg = new int[n];
        for (int v = 0; v < n; v++) {
            if (removed[v]) continue;
            int o = 0, ii = 0;
            foreach (int w in succArr[v]) if (!removed[w]) o++;
            foreach (int w in predArr[v]) if (!removed[w]) ii++;
            outDeg[v] = o;
            inDeg[v] = ii;
        }

        ReduceGlobal();

        var sccs = FindAllSCCs();
        foreach (var scc in sccs) {
            if (scc.Count < 2) continue;
            var sccSet = new HashSet<int>(scc);
            var globalSavedRemoved = (bool[])removed.Clone();
            var globalSavedOutDeg = (int[])outDeg.Clone();
            var globalSavedInDeg = (int[])inDeg.Clone();
            for (int v = 0; v < n; v++) {
                if (!removed[v] && !sccSet.Contains(v)) removed[v] = true;
            }
            foreach (int v in scc) {
                int o = 0, ii = 0;
                foreach (int w in succArr[v]) if (!removed[w]) o++;
                foreach (int w in predArr[v]) if (!removed[w]) ii++;
                outDeg[v] = o;
                inDeg[v] = ii;
            }

            currentSol = new List<int>();
            bestSol = GreedyFVS(scc);
            Solve();
            solution.AddRange(bestSol);

            removed = globalSavedRemoved;
            outDeg = globalSavedOutDeg;
            inDeg = globalSavedInDeg;
        }

        var sb = new StringBuilder();
        sb.Append(solution.Count).Append('\n');
        sb.Append(string.Join(" ", solution)).Append('\n');
        Console.Out.Write(sb.ToString());
    }

    static void ReduceGlobal() {
        bool changed = true;
        while (changed) {
            changed = false;
            for (int v = 0; v < n; v++) {
                if (removed[v]) continue;
                if (outDeg[v] == 0 || inDeg[v] == 0) {
                    RemoveVertexUpdateDeg(v);
                    changed = true;
                }
            }
        }
    }

    static void RemoveVertexUpdateDeg(int v) {
        removed[v] = true;
        foreach (int w in succArr[v]) if (!removed[w]) inDeg[w]--;
        foreach (int w in predArr[v]) if (!removed[w]) outDeg[w]--;
    }

    static void RestoreVertexUpdateDeg(int v) {
        removed[v] = false;
        int newOut = 0, newIn = 0;
        foreach (int w in succArr[v]) if (!removed[w]) { newOut++; inDeg[w]++; }
        foreach (int w in predArr[v]) if (!removed[w]) { newIn++; outDeg[w]++; }
        outDeg[v] = newOut;
        inDeg[v] = newIn;
    }

    static List<List<int>> FindAllSCCs() {
        var index = new int[n];
        var lowlink = new int[n];
        var onStack = new bool[n];
        var assigned = new bool[n];
        var stack = new List<int>();
        int currentIdx = 1;
        var sccs = new List<List<int>>();

        var dfsV = new List<int>();
        var dfsIdx = new List<int>();

        for (int start = 0; start < n; start++) {
            if (removed[start] || assigned[start]) continue;
            dfsV.Clear();
            dfsIdx.Clear();
            index[start] = currentIdx;
            lowlink[start] = currentIdx;
            currentIdx++;
            assigned[start] = true;
            stack.Add(start);
            onStack[start] = true;
            dfsV.Add(start);
            dfsIdx.Add(0);

            while (dfsV.Count > 0) {
                int top = dfsV.Count - 1;
                int v = dfsV[top];
                int idx = dfsIdx[top];
                bool descended = false;
                int[] neighbors = succArr[v];
                while (idx < neighbors.Length) {
                    int w = neighbors[idx];
                    idx++;
                    if (removed[w]) continue;
                    if (!assigned[w]) {
                        dfsIdx[top] = idx;
                        index[w] = currentIdx;
                        lowlink[w] = currentIdx;
                        currentIdx++;
                        assigned[w] = true;
                        stack.Add(w);
                        onStack[w] = true;
                        dfsV.Add(w);
                        dfsIdx.Add(0);
                        descended = true;
                        break;
                    } else if (onStack[w]) {
                        if (index[w] < lowlink[v]) lowlink[v] = index[w];
                    }
                }
                if (descended) continue;

                dfsV.RemoveAt(top);
                dfsIdx.RemoveAt(top);
                if (lowlink[v] == index[v]) {
                    var scc = new List<int>();
                    while (true) {
                        int last = stack.Count - 1;
                        int w = stack[last];
                        stack.RemoveAt(last);
                        onStack[w] = false;
                        scc.Add(w);
                        if (w == v) break;
                    }
                    sccs.Add(scc);
                }
                if (dfsV.Count > 0) {
                    int parent = dfsV[dfsV.Count - 1];
                    if (lowlink[v] < lowlink[parent]) lowlink[parent] = lowlink[v];
                }
            }
        }
        return sccs;
    }

    static int[] FindCycleInActive() {
        var color = new int[n];
        var parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = -1;

        var dfsV = new List<int>();
        var dfsIdx = new List<int>();

        for (int start = 0; start < n; start++) {
            if (removed[start] || color[start] != 0) continue;
            color[start] = 1;
            dfsV.Clear();
            dfsIdx.Clear();
            dfsV.Add(start);
            dfsIdx.Add(0);

            while (dfsV.Count > 0) {
                int top = dfsV.Count - 1;
                int v = dfsV[top];
                int idx = dfsIdx[top];
                bool advance = false;
                int[] neighbors = succArr[v];
                while (idx < neighbors.Length) {
                    int w = neighbors[idx];
                    idx++;
                    if (removed[w]) continue;
                    if (color[w] == 0) {
                        parent[w] = v;
                        color[w] = 1;
                        dfsIdx[top] = idx;
                        dfsV.Add(w);
                        dfsIdx.Add(0);
                        advance = true;
                        break;
                    } else if (color[w] == 1) {
                        var cycle = new List<int>();
                        cycle.Add(v);
                        int cur = v;
                        while (cur != w) {
                            cur = parent[cur];
                            cycle.Add(cur);
                        }
                        return cycle.ToArray();
                    }
                }
                if (advance) continue;
                dfsV.RemoveAt(top);
                dfsIdx.RemoveAt(top);
                color[v] = 2;
            }
        }
        return null;
    }

    static void Solve() {
        var changesRemoved = new Stack<int>();
        bool any = true;
        while (any) {
            any = false;
            for (int v = 0; v < n; v++) {
                if (removed[v]) continue;
                if (outDeg[v] == 0 || inDeg[v] == 0) {
                    RemoveVertexUpdateDeg(v);
                    changesRemoved.Push(v);
                    any = true;
                }
            }
        }

        if (currentSol.Count < bestSol.Count) {
            var cycle = FindCycleInActive();
            if (cycle == null) {
                bestSol = new List<int>(currentSol);
            } else if (currentSol.Count + 1 < bestSol.Count) {
                Array.Sort(cycle, (a, b) => ((long)inDeg[b] * outDeg[b]).CompareTo((long)inDeg[a] * outDeg[a]));
                foreach (int v in cycle) {
                    if (currentSol.Count + 1 >= bestSol.Count) break;
                    currentSol.Add(v);
                    RemoveVertexUpdateDeg(v);
                    Solve();
                    RestoreVertexUpdateDeg(v);
                    currentSol.RemoveAt(currentSol.Count - 1);
                }
            }
        }

        while (changesRemoved.Count > 0) {
            int v = changesRemoved.Pop();
            RestoreVertexUpdateDeg(v);
        }
    }

    static List<int> GreedyFVS(List<int> scc) {
        var savedRemoved = (bool[])removed.Clone();
        var savedOutDeg = (int[])outDeg.Clone();
        var savedInDeg = (int[])inDeg.Clone();

        var solution = new List<int>();

        while (true) {
            bool any = true;
            while (any) {
                any = false;
                foreach (int v in scc) {
                    if (removed[v]) continue;
                    if (outDeg[v] == 0 || inDeg[v] == 0) {
                        RemoveVertexUpdateDeg(v);
                        any = true;
                    }
                }
            }
            if (FindCycleInActive() == null) break;
            int bestV = -1;
            long bestScore = -1;
            foreach (int v in scc) {
                if (removed[v]) continue;
                long score = (long)inDeg[v] * outDeg[v];
                if (score > bestScore) { bestScore = score; bestV = v; }
            }
            if (bestV == -1) break;
            solution.Add(bestV);
            RemoveVertexUpdateDeg(bestV);
        }

        removed = savedRemoved;
        outDeg = savedOutDeg;
        inDeg = savedInDeg;
        return solution;
    }
}
