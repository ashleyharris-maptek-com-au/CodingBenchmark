#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>
#include <chrono>
#include <cstdio>

struct Vec3 {
    double x, y, z;
};

static inline Vec3 vsub(const Vec3& a, const Vec3& b) { return {a.x-b.x, a.y-b.y, a.z-b.z}; }
static inline Vec3 vadd(const Vec3& a, const Vec3& b) { return {a.x+b.x, a.y+b.y, a.z+b.z}; }
static inline Vec3 vscale(const Vec3& a, double s) { return {a.x*s, a.y*s, a.z*s}; }
static inline double vdot(const Vec3& a, const Vec3& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
static inline Vec3 vcross(const Vec3& a, const Vec3& b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}

struct Plane {
    Vec3 n;   // outward unit normal
    double d; // plane eq: n.p + d = 0; inside means n.p + d <= 0
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    auto start = std::chrono::steady_clock::now();

    int V;
    double e;
    if (!(std::cin >> V >> e)) {
        std::cout << 0 << "\n";
        return 0;
    }
    if (V <= 0 || !(e > 0.0)) {
        std::cout << 0 << "\n";
        return 0;
    }

    std::vector<Vec3> verts(V);
    Vec3 lo = { 1e300, 1e300, 1e300 };
    Vec3 hi = { -1e300, -1e300, -1e300 };
    Vec3 sumv = { 0, 0, 0 };

    for (int i = 0; i < V; i++) {
        std::cin >> verts[i].x >> verts[i].y >> verts[i].z;
        lo.x = std::min(lo.x, verts[i].x);
        lo.y = std::min(lo.y, verts[i].y);
        lo.z = std::min(lo.z, verts[i].z);
        hi.x = std::max(hi.x, verts[i].x);
        hi.y = std::max(hi.y, verts[i].y);
        hi.z = std::max(hi.z, verts[i].z);
        sumv = vadd(sumv, verts[i]);
    }
    Vec3 centroid = vscale(sumv, 1.0 / V);

    // Tetrahedron metrics (edge e)
    const double R = e * std::sqrt(3.0 / 8.0); // circumradius

    // Container diagonal extent for tolerance scaling
    double diag = std::sqrt(
        (hi.x - lo.x) * (hi.x - lo.x) +
        (hi.y - lo.y) * (hi.y - lo.y) +
        (hi.z - lo.z) * (hi.z - lo.z));
    double eps_abs = std::max(1e-9, 1e-7 * std::max(1.0, diag));

    // ---- Compute convex hull face planes via O(V^4) brute force (small V) ----
    std::vector<Plane> planes;
    if (V <= 80) {
        planes.reserve(64);
        for (int i = 0; i < V; i++) {
            for (int j = i + 1; j < V; j++) {
                for (int k = j + 1; k < V; k++) {
                    Vec3 a = vsub(verts[j], verts[i]);
                    Vec3 b = vsub(verts[k], verts[i]);
                    Vec3 n = vcross(a, b);
                    double nl = std::sqrt(vdot(n, n));
                    if (nl < 1e-12) continue;
                    n = vscale(n, 1.0 / nl);

                    double maxs = -1e300, mins = 1e300;
                    for (int l = 0; l < V; l++) {
                        if (l == i || l == j || l == k) continue;
                        double s = vdot(n, vsub(verts[l], verts[i]));
                        if (s > maxs) maxs = s;
                        if (s < mins) mins = s;
                    }

                    if (maxs <= eps_abs) {
                        double dval = -vdot(n, verts[i]);
                        if (vdot(n, centroid) + dval <= eps_abs) {
                            planes.push_back({ n, dval });
                        }
                    } else if (mins >= -eps_abs) {
                        Vec3 nn = vscale(n, -1.0);
                        double dval = -vdot(nn, verts[i]);
                        if (vdot(nn, centroid) + dval <= eps_abs) {
                            planes.push_back({ nn, dval });
                        }
                    }
                }
            }
        }
    }

    bool use_planes = !planes.empty();

    // Sphere-of-radius-R inside test (safe regardless of tetra orientation).
    auto sphere_inside = [&](const Vec3& c) -> bool {
        const double tol = 1e-9;
        if (use_planes) {
            for (const auto& pl : planes) {
                double sd = vdot(pl.n, c) + pl.d; // signed distance (n is unit)
                if (sd > -R + tol) return false;
            }
            return true;
        } else {
            return c.x >= lo.x + R - tol && c.x <= hi.x - R + tol
                && c.y >= lo.y + R - tol && c.y <= hi.y - R + tol
                && c.z >= lo.z + R - tol && c.z <= hi.z - R + tol;
        }
    };

    // 2R between centers -> circumspheres tangent at most -> guaranteed non-overlap.
    double spacing = 2.0 * R * 1.005;
    double dx = spacing, dy = spacing, dz = spacing;

    // Adaptive cap for ludicrous-scale containers.
    const std::size_t MAX_TETRAS = 400000;
    double bx = std::max(0.0, hi.x - lo.x - 2.0 * R);
    double by = std::max(0.0, hi.y - lo.y - 2.0 * R);
    double bz = std::max(0.0, hi.z - lo.z - 2.0 * R);
    double box_vol = bx * by * bz;
    double cell_vol = dx * dy * dz;

    if (box_vol > 0 && cell_vol > 0) {
        double est = box_vol / cell_vol;
        if (est > (double)MAX_TETRAS * 1.5) {
            double f = std::cbrt(est / (double)MAX_TETRAS);
            dx *= f; dy *= f; dz *= f;
        }
    }

    std::vector<Vec3> placed;
    placed.reserve(std::min<std::size_t>(MAX_TETRAS, 200000));

    double x_start = lo.x + R;
    double y_start = lo.y + R;
    double z_start = lo.z + R;
    double x_end = hi.x - R;
    double y_end = hi.y - R;
    double z_end = hi.z - R;

    long long counter = 0;
    bool stop = false;

    if (x_start <= x_end + 1e-9 && y_start <= y_end + 1e-9 && z_start <= z_end + 1e-9) {
        for (double cx = x_start; cx <= x_end + 1e-9; cx += dx) {
            if (stop) break;
            for (double cy = y_start; cy <= y_end + 1e-9; cy += dy) {
                if (stop) break;
                for (double cz = z_start; cz <= z_end + 1e-9; cz += dz) {
                    Vec3 c = { cx, cy, cz };
                    if (sphere_inside(c)) {
                        placed.push_back(c);
                        if (placed.size() >= MAX_TETRAS) { stop = true; break; }
                    }
                    counter++;
                    if ((counter & 131071LL) == 0) {
                        auto now = std::chrono::steady_clock::now();
                        long long ms = std::chrono::duration_cast<std::chrono::milliseconds>(now - start).count();
                        if (ms > 25000) { stop = true; break; }
                    }
                }
            }
        }
    }

    std::printf("%zu\n", placed.size());
    for (const auto& c : placed) {
        std::printf("%.10f %.10f %.10f 0 0 0\n", c.x, c.y, c.z);
    }

    return 0;
}
