use std::io::{self, Read, Write, BufWriter};

struct Piece {
    idx: usize,
    minx: f64,
    miny: f64,
    w: f64,
    h: f64,
}

fn main() {
    // Read all input at once for speed
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_ascii_whitespace();

    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    // Read stock polygon, compute bounding box
    let mut smin_x = f64::INFINITY;
    let mut smax_x = f64::NEG_INFINITY;
    let mut smin_y = f64::INFINITY;
    let mut smax_y = f64::NEG_INFINITY;
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        if x < smin_x { smin_x = x; }
        if x > smax_x { smax_x = x; }
        if y < smin_y { smin_y = y; }
        if y > smax_y { smax_y = y; }
    }
    let sw = smax_x - smin_x;
    let sh = smax_y - smin_y;

    // Read pieces, compute axis-aligned bounding boxes
    let mut pieces: Vec<Piece> = Vec::with_capacity(np);
    for i in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut minx = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            if x < minx { minx = x; }
            if x > maxx { maxx = x; }
            if y < miny { miny = y; }
            if y > maxy { maxy = y; }
        }
        pieces.push(Piece { idx: i, minx, miny, w: maxx - minx, h: maxy - miny });
    }

    // First Fit Decreasing Height (shelf packing)
    // Sort piece indices by height descending; tiebreak by width descending
    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        let ha = pieces[a].h;
        let hb = pieces[b].h;
        match hb.partial_cmp(&ha).unwrap_or(std::cmp::Ordering::Equal) {
            std::cmp::Ordering::Equal => pieces[b].w.partial_cmp(&pieces[a].w).unwrap_or(std::cmp::Ordering::Equal),
            other => other,
        }
    });

    // placements[i] = (stock_idx, tx, ty, rotation)
    let mut placements: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); np];

    // Multi-shelf state for current stock.
    // Each shelf: (y_bottom, shelf_height, current_x_used)
    let mut stock_idx: usize = 0;
    let mut shelves: Vec<(f64, f64, f64)> = Vec::new();
    let mut next_y: f64 = 0.0; // y position for next new shelf

    let eps = 1e-9_f64;

    for &oi in &order {
        let p = &pieces[oi];
        let mut w = p.w;
        let mut h = p.h;
        let mut rot = 0.0_f64;

        let fits_no_rot = w <= sw + eps && h <= sh + eps;
        let fits_rot = h <= sw + eps && w <= sh + eps;

        if !fits_no_rot && !fits_rot {
            // Too big for any stock - best effort placement on its own stock
            if !shelves.is_empty() || next_y > 0.0 {
                stock_idx += 1;
                shelves.clear();
                next_y = 0.0;
            }
            placements[p.idx] = (stock_idx, smin_x - p.minx, smin_y - p.miny, 0.0);
            stock_idx += 1;
            shelves.clear();
            next_y = 0.0;
            continue;
        }

        // Prefer non-rotated if it fits; else rotate 90 degrees
        if !fits_no_rot {
            std::mem::swap(&mut w, &mut h);
            rot = std::f64::consts::FRAC_PI_2;
        }

        // Try to place on an existing shelf (First Fit)
        let mut placed = false;
        let mut chosen_x = 0.0;
        let mut chosen_y = 0.0;

        for s in shelves.iter_mut() {
            let sy = s.0;
            let sh_shelf = s.1;
            let sxu = s.2;
            if h <= sh_shelf + eps && sxu + w <= sw + eps {
                chosen_x = sxu;
                chosen_y = sy;
                s.2 = sxu + w;
                placed = true;
                break;
            }
        }

        if !placed {
            // Need a new shelf on current stock
            if next_y + h <= sh + eps {
                chosen_x = 0.0;
                chosen_y = next_y;
                shelves.push((next_y, h, w));
                next_y += h;
            } else {
                // Move to a new stock
                stock_idx += 1;
                shelves.clear();
                next_y = 0.0;

                // Re-evaluate orientation on the fresh stock (prefer no-rotation)
                w = p.w;
                h = p.h;
                rot = 0.0;
                if w > sw + eps || h > sh + eps {
                    std::mem::swap(&mut w, &mut h);
                    rot = std::f64::consts::FRAC_PI_2;
                }
                chosen_x = 0.0;
                chosen_y = 0.0;
                shelves.push((0.0, h, w));
                next_y = h;
            }
        }

        // Compute final translation given rotation
        let (tx, ty) = if rot == 0.0 {
            (smin_x + chosen_x - p.minx, smin_y + chosen_y - p.miny)
        } else {
            // 90 deg CCW: vertex (x,y) -> (-y, x), then translate by (tx,ty)
            // Rotated piece bbox-min = (-(p.miny + p.h), p.minx)
            // We want bbox-min at (smin_x + chosen_x, smin_y + chosen_y)
            (
                smin_x + chosen_x + p.miny + p.h,
                smin_y + chosen_y - p.minx,
            )
        };

        placements[p.idx] = (stock_idx, tx, ty, rot);
    }

    let num_stocks = if np == 0 { 0 } else { stock_idx + 1 };

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", num_stocks).unwrap();
    for (s, tx, ty, rot) in placements {
        writeln!(out, "{} {:.10} {:.10} {:.10}", s, tx, ty, rot).unwrap();
    }
}
