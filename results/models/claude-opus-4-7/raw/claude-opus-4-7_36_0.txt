#include <bits/stdc++.h>
using namespace std;

// ============================================================
// Maximum Independent Set
//
// Strategy:
//   1. Parse graph; record self-loops (those vertices can never be in
//      an IS) and de-duplicate adjacency implicitly via bitsets.
//   2. Decompose into connected components.
//   3. For each component:
//        - If bipartite -> Hopcroft-Karp + Koenig's theorem (poly-time).
//        - Otherwise   -> Bitset branch-and-bound with degree-0 / 1
//                          reductions and an "upper bound = remaining
//                          vertices" prune.
//   4. Concatenate per-component IS to form the global answer.
// ============================================================

static int N, M;
static vector<vector<int>> adjGlobal;
static vector<bool> selfLoop;
static vector<int> finalIS;

// ============================================================
// Bitset utilities (dynamic-width)
// ============================================================

static int Wlocal;
static vector<vector<uint64_t>> Amat;   // adjacency matrix (per-component)
static vector<uint64_t> activeBS;       // active set
static int activeCount;
static int bestSizeLocal;
static vector<int> bestSetLocal;
static vector<int> currentSetLocal;

static inline void bs_set(vector<uint64_t>& b, int i) {
    b[i >> 6] |= (1ULL << (i & 63));
}
static inline void bs_reset(vector<uint64_t>& b, int i) {
    b[i >> 6] &= ~(1ULL << (i & 63));
}
static inline bool bs_test(const vector<uint64_t>& b, int i) {
    return (b[i >> 6] >> (i & 63)) & 1ULL;
}

static inline int bs_neighbor_count(int v) {
    int c = 0;
    const uint64_t* __restrict A = Amat[v].data();
    const uint64_t* __restrict S = activeBS.data();
    for (int i = 0; i < Wlocal; i++)
        c += __builtin_popcountll(A[i] & S[i]);
    return c;
}

// ============================================================
// Branch-and-bound for general MIS
// ============================================================

static void bbSearch() {
    // Reduction stack: kind 0 = vertex added to IS (in currentSet),
    //                  kind 1 = vertex removed but not in IS.
    vector<pair<int,int>> revertStack;
    revertStack.reserve(16);

    // Repeatedly apply degree-0 and degree-1 reductions.
    while (true) {
        bool changed = false;
        for (int wi = 0; wi < Wlocal && !changed; wi++) {
            uint64_t w = activeBS[wi];
            while (w) {
                int j = __builtin_ctzll(w);
                w &= w - 1;
                int v = wi * 64 + j;
                int d = bs_neighbor_count(v);
                if (d == 0) {
                    bs_reset(activeBS, v);
                    activeCount--;
                    currentSetLocal.push_back(v);
                    revertStack.push_back({0, v});
                    changed = true;
                    break;
                } else if (d == 1) {
                    int u = -1;
                    for (int k = 0; k < Wlocal; k++) {
                        uint64_t nbm = Amat[v][k] & activeBS[k];
                        if (nbm) { u = k * 64 + __builtin_ctzll(nbm); break; }
                    }
                    bs_reset(activeBS, v);
                    bs_reset(activeBS, u);
                    activeCount -= 2;
                    currentSetLocal.push_back(v);
                    revertStack.push_back({0, v});
                    revertStack.push_back({1, u});
                    changed = true;
                    break;
                }
            }
        }
        if (!changed) break;
    }

    auto undo = [&]() {
        for (auto it = revertStack.rbegin(); it != revertStack.rend(); ++it) {
            bs_set(activeBS, it->second);
            activeCount++;
            if (it->first == 0) currentSetLocal.pop_back();
        }
    };

    // Upper-bound prune (every remaining vertex could in theory be added).
    if ((int)currentSetLocal.size() + activeCount <= bestSizeLocal) {
        undo();
        return;
    }

    if (activeCount == 0) {
        if ((int)currentSetLocal.size() > bestSizeLocal) {
            bestSizeLocal = (int)currentSetLocal.size();
            bestSetLocal = currentSetLocal;
        }
        undo();
        return;
    }

    // Branch on highest-degree active vertex.
    int maxDeg = -1, maxV = -1;
    for (int wi = 0; wi < Wlocal; wi++) {
        uint64_t w = activeBS[wi];
        while (w) {
            int j = __builtin_ctzll(w);
            w &= w - 1;
            int v = wi * 64 + j;
            int d = bs_neighbor_count(v);
            if (d > maxDeg) { maxDeg = d; maxV = v; }
        }
    }

    // Branch 1: include maxV (remove maxV and N(maxV)).
    vector<int> removed;
    removed.reserve((size_t)maxDeg);
    for (int k = 0; k < Wlocal; k++) {
        uint64_t nbm = Amat[maxV][k] & activeBS[k];
        while (nbm) {
            int j = __builtin_ctzll(nbm);
            nbm &= nbm - 1;
            int u = k * 64 + j;
            removed.push_back(u);
            bs_reset(activeBS, u);
            activeCount--;
        }
    }
    bs_reset(activeBS, maxV);
    activeCount--;
    currentSetLocal.push_back(maxV);
    bbSearch();
    currentSetLocal.pop_back();
    bs_set(activeBS, maxV);
    activeCount++;
    for (int u : removed) { bs_set(activeBS, u); activeCount++; }

    // Branch 2: exclude maxV.
    bs_reset(activeBS, maxV);
    activeCount--;
    bbSearch();
    bs_set(activeBS, maxV);
    activeCount++;

    undo();
}

static vector<int> solveGeneralIS(const vector<int>& nodes) {
    int n = (int)nodes.size();
    vector<int> nodeMap(N, -1);
    for (int i = 0; i < n; i++) nodeMap[nodes[i]] = i;

    Wlocal = max(1, (n + 63) / 64);
    Amat.assign(n, vector<uint64_t>(Wlocal, 0));
    for (int i = 0; i < n; i++) {
        for (int v : adjGlobal[nodes[i]]) {
            int j = nodeMap[v];
            if (j != -1 && j != i) bs_set(Amat[i], j);
        }
    }

    activeBS.assign(Wlocal, 0);
    for (int i = 0; i < n; i++) bs_set(activeBS, i);
    activeCount = n;

    bestSizeLocal = 0;
    bestSetLocal.clear();
    currentSetLocal.clear();

    bbSearch();

    vector<int> result;
    result.reserve(bestSetLocal.size());
    for (int v : bestSetLocal) result.push_back(nodes[v]);
    return result;
}

// ============================================================
// Bipartite check & 2-coloring (BFS)
// ============================================================

static bool checkBipartiteAndSplit(const vector<int>& nodes,
                                   vector<int>& leftSide,
                                   vector<int>& rightSide) {
    static vector<int> color;
    color.assign(N, -1);
    queue<int> q;
    int start = nodes[0];
    color[start] = 0;
    q.push(start);
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int v : adjGlobal[u]) {
            if (color[v] == -1) {
                color[v] = 1 - color[u];
                q.push(v);
            } else if (color[v] == color[u]) {
                return false;
            }
        }
    }
    for (int u : nodes) {
        if (color[u] == 0) leftSide.push_back(u);
        else rightSide.push_back(u);
    }
    return true;
}

// ============================================================
// Hopcroft-Karp bipartite matching + Koenig's IS reconstruction
// ============================================================

static vector<int> solveBipartiteIS(const vector<int>& leftSide,
                                    const vector<int>& rightSide) {
    int nL = (int)leftSide.size(), nR = (int)rightSide.size();
    vector<int> Rid(N, -1);
    for (int i = 0; i < nR; i++) Rid[rightSide[i]] = i;

    vector<vector<int>> hkAdj(nL);
    for (int i = 0; i < nL; i++) {
        for (int v : adjGlobal[leftSide[i]]) {
            int j = Rid[v];
            if (j != -1) hkAdj[i].push_back(j);
        }
        sort(hkAdj[i].begin(), hkAdj[i].end());
        hkAdj[i].erase(unique(hkAdj[i].begin(), hkAdj[i].end()), hkAdj[i].end());
    }

    vector<int> matchL(nL, -1), matchR(nR, -1);
    vector<int> dist(nL);
    const int INF = INT_MAX;

    auto bfs = [&]() -> bool {
        queue<int> q;
        for (int u = 0; u < nL; u++) {
            if (matchL[u] == -1) { dist[u] = 0; q.push(u); }
            else dist[u] = INF;
        }
        bool found = false;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int v : hkAdj[u]) {
                int nxt = matchR[v];
                if (nxt == -1) found = true;
                else if (dist[nxt] == INF) {
                    dist[nxt] = dist[u] + 1;
                    q.push(nxt);
                }
            }
        }
        return found;
    };

    function<bool(int)> dfsM = [&](int u) -> bool {
        for (int v : hkAdj[u]) {
            int nxt = matchR[v];
            if (nxt == -1 || (dist[nxt] == dist[u] + 1 && dfsM(nxt))) {
                matchL[u] = v;
                matchR[v] = u;
                return true;
            }
        }
        dist[u] = INF;
        return false;
    };

    while (bfs()) {
        for (int u = 0; u < nL; u++) {
            if (matchL[u] == -1) dfsM(u);
        }
    }

    // Koenig: BFS alternating paths from unmatched L.
    vector<char> visL(nL, 0), visR(nR, 0);
    queue<int> q;
    for (int u = 0; u < nL; u++) {
        if (matchL[u] == -1) { visL[u] = 1; q.push(u); }
    }
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int v : hkAdj[u]) {
            if (matchL[u] == v) continue;       // skip matching edge
            if (!visR[v]) {
                visR[v] = 1;
                int u2 = matchR[v];
                if (u2 != -1 && !visL[u2]) {
                    visL[u2] = 1;
                    q.push(u2);
                }
            }
        }
    }

    // MIS = (L cap Z) cup (R minus Z)
    vector<int> result;
    result.reserve(nL + nR);
    for (int u = 0; u < nL; u++) if (visL[u]) result.push_back(leftSide[u]);
    for (int v = 0; v < nR; v++) if (!visR[v]) result.push_back(rightSide[v]);
    return result;
}

// ============================================================
// Component dispatcher
// ============================================================

static void solveComponent(const vector<int>& nodes) {
    if (nodes.empty()) return;
    if (nodes.size() == 1) { finalIS.push_back(nodes[0]); return; }

    vector<int> leftSide, rightSide;
    if (checkBipartiteAndSplit(nodes, leftSide, rightSide)) {
        vector<int> result = solveBipartiteIS(leftSide, rightSide);
        for (int v : result) finalIS.push_back(v);
        return;
    }

    vector<int> result = solveGeneralIS(nodes);
    for (int v : result) finalIS.push_back(v);
}

// ============================================================
// Main
// ============================================================

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> N >> M)) { cout << 0 << "\n\n"; return 0; }
    adjGlobal.assign(N, {});
    selfLoop.assign(N, false);

    for (int i = 0; i < M; i++) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u == v) {
            if (u >= 0 && u < N) selfLoop[u] = true;
        } else {
            if (u >= 0 && u < N && v >= 0 && v < N) {
                adjGlobal[u].push_back(v);
                adjGlobal[v].push_back(u);
            }
        }
    }

    // Connected components (skip self-looped vertices: never in any IS).
    vector<int> comp(N, -1);
    int numComp = 0;
    for (int start = 0; start < N; start++) {
        if (comp[start] != -1) continue;
        if (selfLoop[start]) continue;

        vector<int> nodes;
        queue<int> q;
        q.push(start);
        comp[start] = numComp;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            nodes.push_back(u);
            for (int v : adjGlobal[u]) {
                if (comp[v] == -1 && !selfLoop[v]) {
                    comp[v] = numComp;
                    q.push(v);
                }
            }
        }
        numComp++;
        solveComponent(nodes);
    }

    cout << finalIS.size() << "\n";
    for (size_t i = 0; i < finalIS.size(); i++) {
        if (i > 0) cout << ' ';
        cout << finalIS[i];
    }
    cout << "\n";
    return 0;
}
