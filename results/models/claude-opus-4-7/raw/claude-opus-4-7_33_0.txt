using System;
using System.IO;
using System.Text;
using System.Diagnostics;

class Program
{
    static int N;
    static int[] F;   // flow,  row-major
    static int[] D;   // dist,  row-major
    static Stopwatch sw = new Stopwatch();
    static long timeLimitMs = 270_000; // 4 min 30s, leaving margin from 5 min cap

    static void Main()
    {
        sw.Start();
        ReadInput();
        Solve();
    }

    static void ReadInput()
    {
        using var stream = Console.OpenStandardInput();
        using var reader = new StreamReader(stream, Encoding.ASCII, false, 1 << 16);
        string all = reader.ReadToEnd();
        int pos = 0;
        N = ParseInt(all, ref pos);
        F = new int[N * N];
        for (int i = 0; i < N * N; i++) F[i] = ParseInt(all, ref pos);
        D = new int[N * N];
        for (int i = 0; i < N * N; i++) D[i] = ParseInt(all, ref pos);
    }

    static int ParseInt(string s, ref int pos)
    {
        int len = s.Length;
        while (pos < len)
        {
            char c = s[pos];
            if (c == ' ' || c == '\n' || c == '\r' || c == '\t') pos++;
            else break;
        }
        bool neg = false;
        if (pos < len && s[pos] == '-') { neg = true; pos++; }
        int v = 0;
        while (pos < len)
        {
            char c = s[pos];
            if (c >= '0' && c <= '9')
            {
                v = v * 10 + (c - '0');
                pos++;
            }
            else break;
        }
        return neg ? -v : v;
    }

    static long ComputeCost(int[] p)
    {
        long c = 0;
        int n = N;
        for (int i = 0; i < n; i++)
        {
            int rowBase = i * n;
            int piN = p[i] * n;
            for (int j = 0; j < n; j++)
            {
                c += (long)F[rowBase + j] * D[piN + p[j]];
            }
        }
        return c;
    }

    // Change in cost when swapping p[r] and p[s].
    static long SwapDelta(int[] p, int r, int s)
    {
        int n = N;
        int a = p[r], b = p[s];
        int rN = r * n, sN = s * n, aN = a * n, bN = b * n;

        long d = (long)(F[rN + r] - F[sN + s]) * (D[bN + b] - D[aN + a])
               + (long)(F[rN + s] - F[sN + r]) * (D[bN + a] - D[aN + b]);

        for (int k = 0; k < n; k++)
        {
            if (k == r || k == s) continue;
            int pk = p[k];
            int pkN = pk * n;
            int kN = k * n;
            d += (long)(F[rN + k] - F[sN + k]) * (D[bN + pk] - D[aN + pk])
               + (long)(F[kN + r] - F[kN + s]) * (D[pkN + b] - D[pkN + a]);
        }
        return d;
    }

    static void Solve()
    {
        if (N <= 1)
        {
            int[] p1 = new int[N];
            for (int i = 0; i < N; i++) p1[i] = i;
            WriteOutput(p1, ComputeCost(p1));
            return;
        }

        Random rng = new Random(20260516);

        int[] best = new int[N];
        for (int i = 0; i < N; i++) best[i] = i;
        Shuffle(best, rng);
        long bestCost = ComputeCost(best);

        // Try also identity permutation as starter
        {
            int[] id = new int[N];
            for (int i = 0; i < N; i++) id[i] = i;
            long idCost = ComputeCost(id);
            if (idCost < bestCost) { bestCost = idCost; Array.Copy(id, best, N); }
        }

        // Initial local search from best
        int[] cur = (int[])best.Clone();
        long curCost = bestCost;
        LocalSearch(cur, ref curCost);
        if (curCost < bestCost) { bestCost = curCost; Array.Copy(cur, best, N); }

        // Iterated Local Search with growing perturbation
        int basePert = Math.Max(2, N / 20);
        int strength = basePert;
        int sinceImprove = 0;
        int maxStrength = Math.Max(basePert + 1, N / 2);

        while (sw.ElapsedMilliseconds < timeLimitMs)
        {
            int[] trial = (int[])best.Clone();
            int swaps = strength;
            for (int k = 0; k < swaps; k++)
            {
                int i = rng.Next(N);
                int j = rng.Next(N);
                if (i != j)
                {
                    int t = trial[i]; trial[i] = trial[j]; trial[j] = t;
                }
            }
            long trialCost = ComputeCost(trial);
            LocalSearch(trial, ref trialCost);

            if (trialCost < bestCost)
            {
                bestCost = trialCost;
                Array.Copy(trial, best, N);
                strength = basePert;
                sinceImprove = 0;
            }
            else
            {
                sinceImprove++;
                strength++;
                if (strength > maxStrength) strength = basePert;

                if (sinceImprove > 50 + N / 5)
                {
                    int[] restart = new int[N];
                    for (int i = 0; i < N; i++) restart[i] = i;
                    Shuffle(restart, rng);
                    long rc = ComputeCost(restart);
                    LocalSearch(restart, ref rc);
                    if (rc < bestCost)
                    {
                        bestCost = rc;
                        Array.Copy(restart, best, N);
                    }
                    sinceImprove = 0;
                    strength = basePert;
                }
            }
        }

        WriteOutput(best, bestCost);
    }

    static void LocalSearch(int[] p, ref long cost)
    {
        bool improved = true;
        while (improved)
        {
            if (sw.ElapsedMilliseconds >= timeLimitMs) return;
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                if (sw.ElapsedMilliseconds >= timeLimitMs) return;
                for (int j = i + 1; j < N; j++)
                {
                    long d = SwapDelta(p, i, j);
                    if (d < 0)
                    {
                        int t = p[i]; p[i] = p[j]; p[j] = t;
                        cost += d;
                        improved = true;
                    }
                }
            }
        }
    }

    static void Shuffle(int[] a, Random rng)
    {
        for (int i = a.Length - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            int t = a[i]; a[i] = a[j]; a[j] = t;
        }
    }

    static void WriteOutput(int[] p, long cost)
    {
        var sb = new StringBuilder();
        sb.Append(cost);
        sb.Append('\n');
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(p[i]);
        }
        sb.Append('\n');
        var outStream = Console.OpenStandardOutput();
        var bytes = Encoding.ASCII.GetBytes(sb.ToString());
        outStream.Write(bytes, 0, bytes.Length);
        outStream.Flush();
    }
}
