#include <iostream>
#include <vector>
#include <unordered_set>
#include <unordered_map>
#include <queue>
#include <deque>
#include <algorithm>
#include <cstdint>
#include <climits>
#include <cmath>

using namespace std;

static int D;
static int max_turns;
static vector<int> g_bounds;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        // FNV-1a-ish hash mixed with rotation
        size_t h = 1469598103934665603ULL;
        for (int x : v) {
            uint64_t u = (uint64_t)(uint32_t)x;
            h ^= u;
            h *= 1099511628211ULL;
            h ^= (h >> 27);
        }
        return h;
    }
};

using PosSet = unordered_set<vector<int>, VecHash>;

static inline bool inBounds(const vector<int>& p) {
    for (int i = 0; i < D; i++)
        if (p[i] < 0 || p[i] >= g_bounds[i]) return false;
    return true;
}

static inline int manhattan(const vector<int>& a, const vector<int>& b) {
    int d = 0;
    for (int i = 0; i < D; i++) d += abs(a[i] - b[i]);
    return d;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> D >> max_turns)) return 0;
    if (D <= 0) return 0;
    g_bounds.assign(D, 1);
    for (int i = 0; i < D; i++) cin >> g_bounds[i];

    int snake_length = 1;
    cin >> snake_length;
    vector<int> head(D, 0);
    for (int i = 0; i < D; i++) cin >> head[i];

    int F = 0;
    cin >> F;
    vector<vector<int>> foods(F, vector<int>(D));
    PosSet foodSet;
    for (int i = 0; i < F; i++) {
        for (int j = 0; j < D; j++) cin >> foods[i][j];
        foodSet.insert(foods[i]);
    }

    int O = 0;
    cin >> O;
    PosSet obstacles;
    obstacles.reserve((size_t)O * 2 + 16);
    for (int i = 0; i < O; i++) {
        vector<int> obs(D);
        for (int j = 0; j < D; j++) cin >> obs[j];
        obstacles.insert(std::move(obs));
    }

    // Snake body. We only know the head position; body shape is unspecified.
    // Start as a single point and let it grow naturally toward snake_length.
    deque<vector<int>> body;
    body.push_back(head);
    PosSet bodySet;
    bodySet.insert(head);
    int curLength = max(1, snake_length);

    vector<vector<int>> remainingFoods = foods;

    auto cellBlockedStatic = [&](const vector<int>& p) -> bool {
        if (!inBounds(p)) return true;
        if (obstacles.count(p)) return true;
        return false;
    };

    // Check if moving to p is safe given current body and whether we'll grow.
    auto safeMove = [&](const vector<int>& p, bool willGrow) -> bool {
        if (cellBlockedStatic(p)) return false;
        if (bodySet.count(p)) {
            // The tail vacates next turn iff we don't grow and body has >1 segments.
            if (!willGrow && body.size() > 1 && p == body.back()) return true;
            return false;
        }
        return true;
    };

    auto findNearestFoodIdx = [&](const vector<int>& from) -> int {
        int best = -1, bd = INT_MAX;
        for (int i = 0; i < (int)remainingFoods.size(); i++) {
            int d = manhattan(from, remainingFoods[i]);
            if (d < bd) { bd = d; best = i; if (bd == 0) break; }
        }
        return best;
    };

    // BFS pathfinding using hash sets. Caps node expansion for high-D safety.
    // Treats current body cells as blocked (except the tail, which will move).
    auto findPath = [&](const vector<int>& start, const vector<int>& goal, int nodeLimit) -> vector<pair<int,int>> {
        vector<pair<int,int>> result;
        if (start == goal) return result;
        if (cellBlockedStatic(goal)) return result;

        unordered_map<vector<int>, pair<vector<int>, pair<int,int>>, VecHash> parent;
        parent.reserve(64);
        queue<vector<int>> q;
        q.push(start);
        parent.emplace(start, make_pair(vector<int>(), make_pair(-1, 0)));

        int nodes = 0;
        bool found = false;
        const vector<int>* tailPtr = body.empty() ? nullptr : &body.back();

        while (!q.empty() && nodes < nodeLimit) {
            vector<int> cur = q.front(); q.pop();
            nodes++;

            // Prefer axes that reduce remaining distance to goal first.
            vector<int> axOrder(D);
            for (int i = 0; i < D; i++) axOrder[i] = i;
            sort(axOrder.begin(), axOrder.end(), [&](int x, int y){
                return abs(goal[x] - cur[x]) > abs(goal[y] - cur[y]);
            });

            for (int idx = 0; idx < D && !found; idx++) {
                int a = axOrder[idx];
                int prefDir = (goal[a] >= cur[a]) ? 1 : -1;
                for (int k = 0; k < 2 && !found; k++) {
                    int d = (k == 0) ? prefDir : -prefDir;
                    vector<int> np = cur;
                    np[a] += d;
                    if (!inBounds(np)) continue;
                    if (obstacles.count(np)) continue;
                    if (bodySet.count(np)) {
                        if (!(tailPtr && np == *tailPtr && body.size() > 1)) continue;
                    }
                    if (parent.count(np)) continue;
                    parent.emplace(np, make_pair(cur, make_pair(a, d)));
                    if (np == goal) { found = true; break; }
                    q.push(std::move(np));
                }
            }
            if (found) break;
        }

        if (!found) return result;

        vector<int> cur = goal;
        while (cur != start) {
            auto it = parent.find(cur);
            if (it == parent.end()) { result.clear(); return result; }
            result.push_back(it->second.second);
            cur = it->second.first;
        }
        reverse(result.begin(), result.end());
        return result;
    };

    // Greedy single-step move toward target.
    auto greedyStep = [&](const vector<int>& target) -> pair<int,int> {
        vector<pair<int,int>> moves;
        moves.reserve(D);
        for (int a = 0; a < D; a++) {
            if (target[a] > head[a]) moves.push_back({a, +1});
            else if (target[a] < head[a]) moves.push_back({a, -1});
        }
        sort(moves.begin(), moves.end(), [&](const pair<int,int>& x, const pair<int,int>& y){
            return abs(target[x.first] - head[x.first]) > abs(target[y.first] - head[y.first]);
        });
        for (auto& mv : moves) {
            vector<int> np = head;
            np[mv.first] += mv.second;
            bool grow = foodSet.count(np) > 0;
            if (safeMove(np, grow)) return mv;
        }
        // Any safe move (including perpendicular / reverse).
        for (int a = 0; a < D; a++) {
            for (int d = -1; d <= 1; d += 2) {
                vector<int> np = head;
                np[a] += d;
                bool grow = foodSet.count(np) > 0;
                if (safeMove(np, grow)) return {a, d};
            }
        }
        return {-1, 0};
    };

    // Decide pathfinding budget based on dimensionality / volume.
    long long approxCells = 1;
    bool huge = false;
    for (int i = 0; i < D; i++) {
        if (g_bounds[i] <= 0) { huge = true; break; }
        if (approxCells > (long long)2e8 / g_bounds[i]) { huge = true; break; }
        approxCells *= g_bounds[i];
    }
    int nodeLimit;
    if (huge || D >= 10) nodeLimit = 4000;
    else if (D >= 6)     nodeLimit = 30000;
    else                 nodeLimit = 200000;

    vector<pair<int,int>> path;
    int pathIdx = 0;
    vector<int> pathTarget;

    auto applyMove = [&](int axis, int dir) {
        vector<int> newHead = head;
        newHead[axis] += dir;
        bool grew = foodSet.count(newHead) > 0;
        if (grew) {
            curLength++;
            foodSet.erase(newHead);
            for (size_t i = 0; i < remainingFoods.size(); i++) {
                if (remainingFoods[i] == newHead) {
                    remainingFoods[i] = remainingFoods.back();
                    remainingFoods.pop_back();
                    break;
                }
            }
            path.clear();
            pathIdx = 0;
        }
        body.push_front(newHead);
        bodySet.insert(newHead);
        while ((int)body.size() > curLength) {
            bodySet.erase(body.back());
            body.pop_back();
        }
        head = newHead;
    };

    for (int turn = 0; turn < max_turns; turn++) {
        if (remainingFoods.empty()) break;

        int tIdx = findNearestFoodIdx(head);
        if (tIdx < 0) break;
        vector<int> target = remainingFoods[tIdx];

        if (path.empty() || pathIdx >= (int)path.size() || pathTarget != target) {
            path = findPath(head, target, nodeLimit);
            pathIdx = 0;
            pathTarget = target;
        }

        int axis = -1, dir = 0;

        if (!path.empty() && pathIdx < (int)path.size()) {
            auto mv = path[pathIdx];
            vector<int> np = head;
            np[mv.first] += mv.second;
            bool grow = foodSet.count(np) > 0;
            if (safeMove(np, grow)) {
                axis = mv.first; dir = mv.second;
                pathIdx++;
            } else {
                path = findPath(head, target, nodeLimit);
                pathIdx = 0;
                if (!path.empty()) {
                    auto mv2 = path[0];
                    vector<int> np2 = head;
                    np2[mv2.first] += mv2.second;
                    bool grow2 = foodSet.count(np2) > 0;
                    if (safeMove(np2, grow2)) {
                        axis = mv2.first; dir = mv2.second;
                        pathIdx = 1;
                    }
                }
            }
        }

        if (axis < 0) {
            auto mv = greedyStep(target);
            if (mv.first < 0) {
                // No safe move; emit any in-bounds move and stop.
                int fallA = 0, fallD = 1;
                bool found = false;
                for (int a = 0; a < D && !found; a++) {
                    for (int dd = -1; dd <= 1 && !found; dd += 2) {
                        vector<int> np = head;
                        np[a] += dd;
                        if (inBounds(np)) { fallA = a; fallD = dd; found = true; }
                    }
                }
                cout << fallA << ' ' << fallD << '\n';
                break;
            }
            axis = mv.first; dir = mv.second;
            path.clear();
            pathIdx = 0;
        }

        cout << axis << ' ' << dir << '\n';
        applyMove(axis, dir);
    }

    cout.flush();
    return 0;
}
