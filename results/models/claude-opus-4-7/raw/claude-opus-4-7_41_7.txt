; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
   %funcType = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %ptr_uniform_UBO Uniform
%ptr_uniform_v4 = OpTypePointer Uniform %v4float
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float

   %worldPos = OpVariable %ptr_input_v3 Input
     %normal = OpVariable %ptr_input_v3 Input
   %outColor = OpVariable %ptr_output_v4 Output

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
    %float_0 = OpConstant %float 0.0
    %float_1 = OpConstant %float 1.0
    %float_3 = OpConstant %float 3.0
  %float_0_2 = OpConstant %float 0.2
  %float_0_3 = OpConstant %float 0.3
  %float_0_5 = OpConstant %float 0.5
  %float_0_8 = OpConstant %float 0.8
  %baseColor = OpConstantComposite %v3float %float_0_2 %float_0_3 %float_0_8
    %rimTint = OpConstantComposite %v3float %float_1 %float_0_8 %float_0_5

       %main = OpFunction %void None %funcType
      %entry = OpLabel
         %wp = OpLoad %v3float %worldPos
       %nraw = OpLoad %v3float %normal
          %N = OpExtInst %v3float %1 Normalize %nraw
   %lightPtr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
    %lightP4 = OpLoad %v4float %lightPtr
    %lightP3 = OpVectorShuffle %v3float %lightP4 %lightP4 0 1 2
     %camPtr = OpAccessChain %ptr_uniform_v4 %ubo %int_4
      %camP4 = OpLoad %v4float %camPtr
      %camP3 = OpVectorShuffle %v3float %camP4 %camP4 0 1 2
         %Ld = OpFSub %v3float %lightP3 %wp
          %L = OpExtInst %v3float %1 Normalize %Ld
         %Vd = OpFSub %v3float %camP3 %wp
          %V = OpExtInst %v3float %1 Normalize %Vd
      %NdotL = OpDot %float %N %L
       %diff = OpExtInst %float %1 FMax %NdotL %float_0
      %NdotV = OpDot %float %N %V
     %NdotVc = OpExtInst %float %1 FMax %NdotV %float_0
     %oneMin = OpFSub %float %float_1 %NdotVc
        %rim = OpExtInst %float %1 Pow %oneMin %float_3
     %diffSc = OpFMul %float %diff %float_0_5
      %modul = OpFAdd %float %float_0_3 %diffSc
    %litBase = OpVectorTimesScalar %v3float %baseColor %modul
   %rimColor = OpVectorTimesScalar %v3float %rimTint %rim
   %finalRGB = OpFAdd %v3float %litBase %rimColor
     %outVal = OpCompositeConstruct %v4float %finalRGB %float_1
               OpStore %outColor %outVal
               OpReturn
               OpFunctionEnd
