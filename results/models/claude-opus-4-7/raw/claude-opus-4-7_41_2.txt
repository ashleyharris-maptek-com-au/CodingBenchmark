; SPIR-V
; Version: 1.0
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %uv %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %uv Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
     %fnType = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v4float = OpTypeVector %float 4
  %ptr_in_v2 = OpTypePointer Input %v2float
 %ptr_out_v4 = OpTypePointer Output %v4float
         %uv = OpVariable %ptr_in_v2 Input
   %outColor = OpVariable %ptr_out_v4 Output
       %zero = OpConstant %float 0
        %one = OpConstant %float 1

       %main = OpFunction %void None %fnType
      %label = OpLabel
     %uv_val = OpLoad %v2float %uv
       %uv_x = OpCompositeExtract %float %uv_val 0
       %uv_y = OpCompositeExtract %float %uv_val 1
      %color = OpCompositeConstruct %v4float %uv_x %uv_y %zero %one
               OpStore %outColor %color
               OpReturn
               OpFunctionEnd
