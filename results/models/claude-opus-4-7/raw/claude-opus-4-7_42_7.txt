OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void

      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
     %uint_3 = OpConstant %uint 3
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2

    %float_0 = OpConstant %float 0.0
 %float_0_08 = OpConstant %float 0.08

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

   %arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
    %inColor = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

     %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %p0_v4 = OpLoad %v4float %p0_ptr
         %p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2

     %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
      %p1_v4 = OpLoad %v4float %p1_ptr
         %p1 = OpVectorShuffle %v3float %p1_v4 %p1_v4 0 1 2

     %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
      %p2_v4 = OpLoad %v4float %p2_ptr
         %p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2

         %e1 = OpFSub %v3float %p1 %p0
         %e2 = OpFSub %v3float %p2 %p0

      %cross = OpExtInst %v3float %1 Cross %e1 %e2
          %N = OpExtInst %v3float %1 Normalize %cross

     %offset = OpVectorTimesScalar %v3float %N %float_0_08

      %off_x = OpCompositeExtract %float %offset 0
      %off_y = OpCompositeExtract %float %offset 1
      %off_z = OpCompositeExtract %float %offset 2
    %offset4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0

     %new_p0 = OpFAdd %v4float %p0_v4 %offset4
   %out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos0 %new_p0
     %c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
         %c0 = OpLoad %v3float %c0_ptr
               OpStore %outColor %c0
               OpEmitVertex

     %new_p1 = OpFAdd %v4float %p1_v4 %offset4
   %out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos1 %new_p1
     %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
         %c1 = OpLoad %v3float %c1_ptr
               OpStore %outColor %c1
               OpEmitVertex

     %new_p2 = OpFAdd %v4float %p2_v4 %offset4
   %out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos2 %new_p2
     %c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
         %c2 = OpLoad %v3float %c2_ptr
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
