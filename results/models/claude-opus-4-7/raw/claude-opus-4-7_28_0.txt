use std::io::{self, Read, Write, BufWriter};

fn parse_u32(bytes: &[u8], pos: &mut usize) -> u32 {
    let len = bytes.len();
    let mut p = *pos;
    while p < len {
        let b = bytes[p];
        if b >= b'0' && b <= b'9' { break; }
        p += 1;
    }
    let mut val: u32 = 0;
    while p < len {
        let b = bytes[p];
        if b < b'0' || b > b'9' { break; }
        val = val * 10 + (b - b'0') as u32;
        p += 1;
    }
    *pos = p;
    val
}

fn main() {
    let mut buf = Vec::new();
    io::stdin().read_to_end(&mut buf).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut pos = 0usize;
    let n = parse_u32(&buf, &mut pos) as usize;
    let m = parse_u32(&buf, &mut pos) as usize;

    let mut adj: Vec<Vec<u32>> = vec![Vec::new(); n];

    for _ in 0..m {
        let u = parse_u32(&buf, &mut pos) as usize;
        let v = parse_u32(&buf, &mut pos) as usize;
        if u != v && u < n && v < n {
            adj[u].push(v as u32);
            adj[v].push(u as u32);
        }
    }
    drop(buf);

    // Deduplicate parallel edges.
    for nbrs in adj.iter_mut() {
        if nbrs.len() > 1 {
            nbrs.sort_unstable();
            nbrs.dedup();
        }
    }

    let mut in_cover = vec![false; n];
    let mut deg: Vec<i32> = adj.iter().map(|v| v.len() as i32).collect();

    let max_deg_init = deg.iter().copied().max().unwrap_or(0).max(0) as usize;

    // Bucket priority queue: doubly-linked lists keyed by degree.
    let bucket_size = max_deg_init + 2;
    let mut bucket_head: Vec<i32> = vec![-1; bucket_size];
    let mut prev_link: Vec<i32> = vec![-1; n];
    let mut next_link: Vec<i32> = vec![-1; n];
    let mut cur_bucket: Vec<i32> = vec![-1; n];

    for v in 0..n {
        let d = deg[v];
        if d <= 0 {
            continue;
        }
        let d = d as usize;
        let head = bucket_head[d];
        next_link[v] = head;
        prev_link[v] = -1;
        if head >= 0 {
            prev_link[head as usize] = v as i32;
        }
        bucket_head[d] = v as i32;
        cur_bucket[v] = d as i32;
    }

    let mut max_d = max_deg_init;

    loop {
        while max_d > 0 && bucket_head[max_d] < 0 {
            max_d -= 1;
        }
        if max_d == 0 {
            break;
        }

        // Take the max-degree vertex into the cover.
        let v = bucket_head[max_d] as usize;
        let nx = next_link[v];
        bucket_head[max_d] = nx;
        if nx >= 0 {
            prev_link[nx as usize] = -1;
        }
        cur_bucket[v] = -1;
        prev_link[v] = -1;
        next_link[v] = -1;

        in_cover[v] = true;

        let nbrs = std::mem::take(&mut adj[v]);
        for &uu in &nbrs {
            let u = uu as usize;
            if in_cover[u] {
                continue;
            }
            let b = cur_bucket[u];
            if b < 0 {
                continue;
            }
            let b = b as usize;
            let pu = prev_link[u];
            let nu = next_link[u];
            if pu >= 0 {
                next_link[pu as usize] = nu;
            } else {
                bucket_head[b] = nu;
            }
            if nu >= 0 {
                prev_link[nu as usize] = pu;
            }

            deg[u] -= 1;
            let new_d = deg[u];
            if new_d > 0 {
                let nd = new_d as usize;
                let head = bucket_head[nd];
                next_link[u] = head;
                prev_link[u] = -1;
                if head >= 0 {
                    prev_link[head as usize] = u as i32;
                }
                bucket_head[nd] = u as i32;
                cur_bucket[u] = nd as i32;
            } else {
                cur_bucket[u] = -1;
                prev_link[u] = -1;
                next_link[u] = -1;
            }
        }
    }

    // Count and emit cover.
    let mut count: usize = 0;
    for v in 0..n {
        if in_cover[v] {
            count += 1;
        }
    }
    writeln!(out, "{}", count).unwrap();

    let mut s = String::with_capacity(count.saturating_mul(8));
    let mut first = true;
    for v in 0..n {
        if in_cover[v] {
            if !first {
                s.push(' ');
            }
            first = false;
            let mut x = v as u32;
            if x == 0 {
                s.push('0');
            } else {
                let mut tmp = [0u8; 10];
                let mut i = 0;
                while x > 0 {
                    tmp[i] = b'0' + (x % 10) as u8;
                    x /= 10;
                    i += 1;
                }
                while i > 0 {
                    i -= 1;
                    s.push(tmp[i] as char);
                }
            }
        }
    }
    writeln!(out, "{}", s).unwrap();
}
