OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %inColor %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18

               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %worldPos Location 0
               OpDecorate %normal   Location 1
               OpDecorate %inColor  Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

      %int_0 = OpConstant %int 0
     %uint_1 = OpConstant %uint 1

        %f_0 = OpConstant %float 0
        %f_r = OpConstant %float 0.04
       %f_rh = OpConstant %float 0.02
     %f_rs3h = OpConstant %float 0.0346410162
       %f_nr = OpConstant %float -0.04
      %f_nrh = OpConstant %float -0.02
    %f_nrs3h = OpConstant %float -0.0346410162
     %f_dark = OpConstant %float 0.7

%gl_PerVertex_in = OpTypeStruct %v4float
     %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
 %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
         %gl_in = OpVariable %ptr_in_arr_pv Input
     %ptr_in_v4 = OpTypePointer Input %v4float

      %arr_v3_1 = OpTypeArray %v3float %uint_1
 %ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1
      %worldPos = OpVariable %ptr_in_arr_v3 Input
        %normal = OpVariable %ptr_in_arr_v3 Input
       %inColor = OpVariable %ptr_in_arr_v3 Input
     %ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
    %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
        %gl_out = OpVariable %ptr_out_pv Output
    %ptr_out_v4 = OpTypePointer Output %v4float

    %ptr_out_v3 = OpTypePointer Output %v3float
      %outColor = OpVariable %ptr_out_v3 Output

          %main = OpFunction %void None %fn_void
         %entry = OpLabel

    %center_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %center = OpLoad %v4float %center_ptr

       %col_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
    %center_col = OpLoad %v3float %col_ptr

      %edge_col = OpVectorTimesScalar %v3float %center_col %f_dark

        %e0_off = OpCompositeConstruct %v4float %f_r    %f_0      %f_0 %f_0
        %e0_pos = OpFAdd %v4float %center %e0_off
        %e1_off = OpCompositeConstruct %v4float %f_rh   %f_rs3h   %f_0 %f_0
        %e1_pos = OpFAdd %v4float %center %e1_off
        %e2_off = OpCompositeConstruct %v4float %f_nrh  %f_rs3h   %f_0 %f_0
        %e2_pos = OpFAdd %v4float %center %e2_off
        %e3_off = OpCompositeConstruct %v4float %f_nr   %f_0      %f_0 %f_0
        %e3_pos = OpFAdd %v4float %center %e3_off
        %e4_off = OpCompositeConstruct %v4float %f_nrh  %f_nrs3h  %f_0 %f_0
        %e4_pos = OpFAdd %v4float %center %e4_off
        %e5_off = OpCompositeConstruct %v4float %f_rh   %f_nrs3h  %f_0 %f_0
        %e5_pos = OpFAdd %v4float %center %e5_off

   %out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

                 OpStore %out_pos_ptr %center
                 OpStore %outColor %center_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e0_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e1_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpEndPrimitive

                 OpStore %out_pos_ptr %center
                 OpStore %outColor %center_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e1_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e2_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpEndPrimitive

                 OpStore %out_pos_ptr %center
                 OpStore %outColor %center_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e2_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e3_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpEndPrimitive

                 OpStore %out_pos_ptr %center
                 OpStore %outColor %center_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e3_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e4_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpEndPrimitive

                 OpStore %out_pos_ptr %center
                 OpStore %outColor %center_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e4_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e5_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpEndPrimitive

                 OpStore %out_pos_ptr %center
                 OpStore %outColor %center_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e5_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpStore %out_pos_ptr %e0_pos
                 OpStore %outColor %edge_col
                 OpEmitVertex
                 OpEndPrimitive

                 OpReturn
                 OpFunctionEnd
