OpCapability Geometry
        %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 8

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
     %uint_4 = OpConstant %uint 4
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
    %f_third = OpConstant %float 0.33333334
 %f_twothirds = OpConstant %float 0.6666667
 %f_halfthick = OpConstant %float 0.01

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float

   %arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
    %inColor = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
       %lbl0 = OpLabel

       ; Load gl_in[0..3].gl_Position
     %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
     %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr
     %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2_ptr
     %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
         %p3 = OpLoad %v4float %p3_ptr

       ; Load color[1], color[2]
     %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
         %c1 = OpLoad %v3float %c1_ptr
     %c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
         %c2 = OpLoad %v3float %c2_ptr

       ; Extract xy components
      %p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
      %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
      %p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
      %p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1

       ; Adjacency tangents: tangent_start = normalize(p2-p0), tangent_end = normalize(p3-p1)
     %ts_unn = OpFSub %v2float %p2_xy %p0_xy
         %ts = OpExtInst %v2float %glsl Normalize %ts_unn
     %te_unn = OpFSub %v2float %p3_xy %p1_xy
         %te = OpExtInst %v2float %glsl Normalize %te_unn

       ; Perpendicular at start = (-ts.y, ts.x) * halfThick
       %ts_x = OpCompositeExtract %float %ts 0
       %ts_y = OpCompositeExtract %float %ts 1
      %nts_y = OpFNegate %float %ts_y
 %perp_s_dir = OpCompositeConstruct %v2float %nts_y %ts_x
      %ht_v2 = OpCompositeConstruct %v2float %f_halfthick %f_halfthick
     %perp_s = OpFMul %v2float %perp_s_dir %ht_v2

       ; Perpendicular at end = (-te.y, te.x) * halfThick
       %te_x = OpCompositeExtract %float %te 0
       %te_y = OpCompositeExtract %float %te 1
      %nte_y = OpFNegate %float %te_y
 %perp_e_dir = OpCompositeConstruct %v2float %nte_y %te_x
     %perp_e = OpFMul %v2float %perp_e_dir %ht_v2

       ; Intermediate xy positions (t=1/3, t=2/3)
     %dir_xy = OpFSub %v2float %p2_xy %p1_xy
     %t13_v2 = OpCompositeConstruct %v2float %f_third %f_third
     %t23_v2 = OpCompositeConstruct %v2float %f_twothirds %f_twothirds
     %off_13 = OpFMul %v2float %dir_xy %t13_v2
    %pt13_xy = OpFAdd %v2float %p1_xy %off_13
     %off_23 = OpFMul %v2float %dir_xy %t23_v2
    %pt23_xy = OpFAdd %v2float %p1_xy %off_23

       ; Smoothly interpolated perpendiculars
  %perp_diff = OpFSub %v2float %perp_e %perp_s
%perp_off_13 = OpFMul %v2float %perp_diff %t13_v2
   %perp_t13 = OpFAdd %v2float %perp_s %perp_off_13
%perp_off_23 = OpFMul %v2float %perp_diff %t23_v2
   %perp_t23 = OpFAdd %v2float %perp_s %perp_off_23

       ; Interpolated colors
     %t13_v3 = OpCompositeConstruct %v3float %f_third %f_third %f_third
     %t23_v3 = OpCompositeConstruct %v3float %f_twothirds %f_twothirds %f_twothirds
     %c_diff = OpFSub %v3float %c2 %c1
   %c_off_13 = OpFMul %v3float %c_diff %t13_v3
      %c_t13 = OpFAdd %v3float %c1 %c_off_13
   %c_off_23 = OpFMul %v3float %c_diff %t23_v3
      %c_t23 = OpFAdd %v3float %c1 %c_off_23

       ; Interpolated z, w
       %p1_z = OpCompositeExtract %float %p1 2
       %p1_w = OpCompositeExtract %float %p1 3
       %p2_z = OpCompositeExtract %float %p2 2
       %p2_w = OpCompositeExtract %float %p2 3
         %dz = OpFSub %float %p2_z %p1_z
         %dw = OpFSub %float %p2_w %p1_w
       %dz13 = OpFMul %float %dz %f_third
       %zt13 = OpFAdd %float %p1_z %dz13
       %dw13 = OpFMul %float %dw %f_third
       %wt13 = OpFAdd %float %p1_w %dw13
       %dz23 = OpFMul %float %dz %f_twothirds
       %zt23 = OpFAdd %float %p1_z %dz23
       %dw23 = OpFMul %float %dw %f_twothirds
       %wt23 = OpFAdd %float %p1_w %dw23

       ; Output position pointer (reused)
  %gl_pos_pp = OpAccessChain %ptr_out_v4 %gl_out %int_0

       ; ===== Quad 1: from t=0 (start) to t=1/3 =====
       ; v0: start + perp_s
      %v0_xy = OpFAdd %v2float %p1_xy %perp_s
       %v0_x = OpCompositeExtract %float %v0_xy 0
       %v0_y = OpCompositeExtract %float %v0_xy 1
     %v0_pos = OpCompositeConstruct %v4float %v0_x %v0_y %p1_z %p1_w
               OpStore %gl_pos_pp %v0_pos
               OpStore %outColor %c1
               OpEmitVertex

       ; v1: start - perp_s
      %v1_xy = OpFSub %v2float %p1_xy %perp_s
       %v1_x = OpCompositeExtract %float %v1_xy 0
       %v1_y = OpCompositeExtract %float %v1_xy 1
     %v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %p1_z %p1_w
               OpStore %gl_pos_pp %v1_pos
               OpStore %outColor %c1
               OpEmitVertex

       ; v2: pt13 + perp_t13
      %v2_xy = OpFAdd %v2float %pt13_xy %perp_t13
       %v2_x = OpCompositeExtract %float %v2_xy 0
       %v2_y = OpCompositeExtract %float %v2_xy 1
     %v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %zt13 %wt13
               OpStore %gl_pos_pp %v2_pos
               OpStore %outColor %c_t13
               OpEmitVertex

       ; v3: pt13 - perp_t13
      %v3_xy = OpFSub %v2float %pt13_xy %perp_t13
       %v3_x = OpCompositeExtract %float %v3_xy 0
       %v3_y = OpCompositeExtract %float %v3_xy 1
     %v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %zt13 %wt13
               OpStore %gl_pos_pp %v3_pos
               OpStore %outColor %c_t13
               OpEmitVertex

               OpEndPrimitive

       ; ===== Quad 2: from t=2/3 to t=1 (end) =====
       ; v4: pt23 + perp_t23
      %v4_xy = OpFAdd %v2float %pt23_xy %perp_t23
       %v4_x = OpCompositeExtract %float %v4_xy 0
       %v4_y = OpCompositeExtract %float %v4_xy 1
     %v4_pos = OpCompositeConstruct %v4float %v4_x %v4_y %zt23 %wt23
               OpStore %gl_pos_pp %v4_pos
               OpStore %outColor %c_t23
               OpEmitVertex

       ; v5: pt23 - perp_t23
      %v5_xy = OpFSub %v2float %pt23_xy %perp_t23
       %v5_x = OpCompositeExtract %float %v5_xy 0
       %v5_y = OpCompositeExtract %float %v5_xy 1
     %v5_pos = OpCompositeConstruct %v4float %v5_x %v5_y %zt23 %wt23
               OpStore %gl_pos_pp %v5_pos
               OpStore %outColor %c_t23
               OpEmitVertex

       ; v6: end + perp_e
      %v6_xy = OpFAdd %v2float %p2_xy %perp_e
       %v6_x = OpCompositeExtract %float %v6_xy 0
       %v6_y = OpCompositeExtract %float %v6_xy 1
     %v6_pos = OpCompositeConstruct %v4float %v6_x %v6_y %p2_z %p2_w
               OpStore %gl_pos_pp %v6_pos
               OpStore %outColor %c2
               OpEmitVertex

       ; v7: end - perp_e
      %v7_xy = OpFSub %v2float %p2_xy %perp_e
       %v7_x = OpCompositeExtract %float %v7_xy 0
       %v7_y = OpCompositeExtract %float %v7_xy 1
     %v7_pos = OpCompositeConstruct %v4float %v7_x %v7_y %p2_z %p2_w
               OpStore %gl_pos_pp %v7_pos
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
