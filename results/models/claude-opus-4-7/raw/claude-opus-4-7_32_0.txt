#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <map>
#include <algorithm>
#include <climits>
#include <cstdint>

#include <bit>
using namespace std;

typedef long long ll;
typedef pair<ll, int> pli;

const ll INF = (ll)1e18;

int N, M, K;
vector<vector<pair<int,int>>> adj;
vector<int> terms;
vector<int> isTerm;
map<pair<int,int>, int> edge_weight;

int get_weight(int u, int v) {
    if (u > v) swap(u, v);
    auto it = edge_weight.find({u, v});
    return (it == edge_weight.end()) ? 0 : it->second;
}

void output_result(const vector<pair<int,int>>& edges, ll total) {
    cout << total << " " << edges.size() << "\n";
    for (auto& e : edges) {
        cout << e.first << " " << e.second << "\n";
    }
}

void solve_heuristic();

void solve_dreyfus_wagner() {
    int FULL = (1 << K) - 1;
    vector<vector<ll>> dp(FULL + 1, vector<ll>(N, INF));
    vector<vector<int>> par_split(FULL + 1, vector<int>(N, -1));
    vector<vector<int>> par_pred(FULL + 1, vector<int>(N, -1));

    for (int i = 0; i < K; i++) {
        dp[1 << i][terms[i]] = 0;
    }

    vector<vector<int>> masks_by_pc(K + 2);
    for (int m = 1; m <= FULL; m++) {
        masks_by_pc[std::popcount((unsigned)m)].push_back(m);
    }

    for (int pc = 1; pc <= K; pc++) {
        for (int S : masks_by_pc[pc]) {
            if (pc > 1) {
                for (int T = (S - 1) & S; T > 0; T = (T - 1) & S) {
                    int T2 = S ^ T;
                    if (T > T2) continue;
                    for (int v = 0; v < N; v++) {
                        if (dp[T][v] >= INF || dp[T2][v] >= INF) continue;
                        ll nv = dp[T][v] + dp[T2][v];
                        if (nv < dp[S][v]) {
                            dp[S][v] = nv;
                            par_split[S][v] = T;
                            par_pred[S][v] = -1;
                        }
                    }
                }
            }

            priority_queue<pli, vector<pli>, greater<pli>> pq;
            for (int v = 0; v < N; v++) {
                if (dp[S][v] < INF) pq.push({dp[S][v], v});
            }
            while (!pq.empty()) {
                auto top = pq.top(); pq.pop();
                ll d = top.first; int u = top.second;
                if (d > dp[S][u]) continue;
                for (auto& nb : adj[u]) {
                    int v = nb.first; int w = nb.second;
                    if (d + (ll)w < dp[S][v]) {
                        dp[S][v] = d + (ll)w;
                        par_pred[S][v] = u;
                        par_split[S][v] = -1;
                        pq.push({dp[S][v], v});
                    }
                }
            }
        }
    }

    ll best = INF;
    int best_v = -1;
    for (int v = 0; v < N; v++) {
        if (dp[FULL][v] < best) {
            best = dp[FULL][v];
            best_v = v;
        }
    }

    if (best == INF || best_v == -1) {
        solve_heuristic();
        return;
    }

    set<pair<int,int>> result_edges_set;
    function<void(int, int)> reconstruct = [&](int S, int v) {
        if (par_pred[S][v] != -1) {
            int prev = par_pred[S][v];
            int a = min(v, prev), b = max(v, prev);
            result_edges_set.insert({a, b});
            reconstruct(S, prev);
        } else if (par_split[S][v] != -1) {
            int T = par_split[S][v];
            reconstruct(T, v);
            reconstruct(S ^ T, v);
        }
    };

    reconstruct(FULL, best_v);

    vector<pair<int,int>> res(result_edges_set.begin(), result_edges_set.end());
    output_result(res, best);
}

void solve_heuristic() {
    vector<bool> in_tree(N, false);
    set<pair<int,int>> tree_edges;
    vector<bool> term_added(K, false);

    in_tree[terms[0]] = true;
    term_added[0] = true;
    int remaining = K - 1;

    while (remaining > 0) {
        vector<ll> dist(N, INF);
        vector<int> prev_node(N, -1);
        priority_queue<pli, vector<pli>, greater<pli>> pq;
        for (int v = 0; v < N; v++) {
            if (in_tree[v]) {
                dist[v] = 0;
                pq.push({0, v});
            }
        }
        while (!pq.empty()) {
            auto top = pq.top(); pq.pop();
            ll d = top.first; int u = top.second;
            if (d > dist[u]) continue;
            for (auto& nb : adj[u]) {
                int v = nb.first; int w = nb.second;
                if (d + (ll)w < dist[v]) {
                    dist[v] = d + (ll)w;
                    prev_node[v] = u;
                    pq.push({dist[v], v});
                }
            }
        }

        int best_term_idx = -1;
        ll best_dist = INF;
        for (int i = 0; i < K; i++) {
            if (!term_added[i] && dist[terms[i]] < best_dist) {
                best_dist = dist[terms[i]];
                best_term_idx = i;
            }
        }
        if (best_term_idx == -1) break;

        int cur = terms[best_term_idx];
        while (prev_node[cur] != -1) {
            int p = prev_node[cur];
            int a = min(cur, p), b = max(cur, p);
            tree_edges.insert({a, b});
            in_tree[cur] = true;
            cur = p;
        }
        in_tree[cur] = true;
        term_added[best_term_idx] = true;
        remaining--;
    }

    vector<tuple<int,int,int>> sg_edges;
    for (auto& e : tree_edges) {
        sg_edges.push_back(make_tuple(get_weight(e.first, e.second), e.first, e.second));
    }
    sort(sg_edges.begin(), sg_edges.end());

    vector<int> par(N);
    iota(par.begin(), par.end(), 0);
    function<int(int)> findp = [&](int x) {
        while (par[x] != x) { par[x] = par[par[x]]; x = par[x]; }
        return x;
    };

    set<pair<int,int>> mst_edges;
    ll total = 0;
    for (auto& e : sg_edges) {
        int w = get<0>(e), u = get<1>(e), v = get<2>(e);
        int pu = findp(u), pv = findp(v);
        if (pu != pv) {
            par[pu] = pv;
            mst_edges.insert({u, v});
            total += w;
        }
    }

    bool changed = true;
    while (changed) {
        changed = false;
        map<int, int> degree;
        for (auto& e : mst_edges) { degree[e.first]++; degree[e.second]++; }
        set<pair<int,int>> new_edges;
        for (auto& e : mst_edges) {
            int u = e.first, v = e.second;
            bool remove = false;
            if (degree[u] == 1 && isTerm[u] == -1) remove = true;
            else if (degree[v] == 1 && isTerm[v] == -1) remove = true;
            if (remove) {
                total -= get_weight(u, v);
                changed = true;
            } else {
                new_edges.insert({u, v});
            }
        }
        mst_edges = new_edges;
    }

    vector<pair<int,int>> result(mst_edges.begin(), mst_edges.end());
    output_result(result, total);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> N >> M >> K;
    adj.assign(N, {});
    for (int i = 0; i < M; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        if (u < 0 || v < 0 || u >= N || v >= N) continue;
        if (u == v) continue;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
        int a = min(u, v), b = max(u, v);
        auto it = edge_weight.find({a, b});
        if (it == edge_weight.end() || it->second > w) {
            edge_weight[{a, b}] = w;
        }
    }
    terms.resize(K);
    isTerm.assign(N, -1);
    for (int i = 0; i < K; i++) {
        cin >> terms[i];
        isTerm[terms[i]] = i;
    }

    if (K <= 1) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    bool use_dw = false;
    if (K <= 16) {
        ll combine_ops = 1;
        for (int i = 0; i < K; i++) combine_ops *= 3;
        combine_ops *= (ll)N;
        ll mem = (1LL << K) * (ll)N * 16;
        if (combine_ops <= (ll)1e9 && mem <= (ll)5e8) {
            use_dw = true;
        }
    }

    if (use_dw) solve_dreyfus_wagner();
    else solve_heuristic();

    return 0;
}
