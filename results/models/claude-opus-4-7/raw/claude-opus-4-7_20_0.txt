use std::io::{self, Read, Write, BufWriter};
use std::collections::HashMap;

struct Suspect {
    hole_idx: u32,
    sample_idx: u32,
    prop_idx: u16,
    confidence: f32,
}

#[inline]
fn cell_key(cx: i64, cy: i64, cz: i64) -> i64 {
    let ux = (cx as i64).wrapping_add(1 << 20) as u64 & 0x1FFFFF;
    let uy = (cy as i64).wrapping_add(1 << 20) as u64 & 0x1FFFFF;
    let uz = (cz as i64).wrapping_add(1 << 20) as u64 & 0x1FFFFF;
    ((ux << 42) | (uy << 21) | uz) as i64
}

fn median_inplace(v: &mut [f64]) -> f64 {
    v.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
    let n = v.len();
    if n == 0 { return 0.0; }
    if n % 2 == 1 { v[n / 2] } else { 0.5 * (v[n / 2 - 1] + v[n / 2]) }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    macro_rules! next_tok { () => { it.next().unwrap_or("") } }
    macro_rules! next_f64 { () => {{
        let s = next_tok!();
        s.parse::<f64>().unwrap_or(f64::NAN)
    }} }
    macro_rules! next_usize { () => {{
        let s = next_tok!();
        s.parse::<usize>().unwrap_or(0)
    }} }

    let num_holes: usize = next_usize!();
    let num_props: usize = next_usize!();

    let mut prop_names: Vec<String> = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(next_tok!().to_string());
    }

    let est = (input.len() / 40).max(num_holes * 4);
    let mut hole_ids: Vec<String> = Vec::with_capacity(num_holes);
    let mut sample_x: Vec<f32> = Vec::with_capacity(est);
    let mut sample_y: Vec<f32> = Vec::with_capacity(est);
    let mut sample_z: Vec<f32> = Vec::with_capacity(est);
    let mut sample_hole: Vec<u32> = Vec::with_capacity(est);
    let mut sample_local_idx: Vec<u32> = Vec::with_capacity(est);
    let mut values: Vec<Vec<f32>> = (0..num_props).map(|_| Vec::with_capacity(est)).collect();

    for h in 0..num_holes {
        let hid = next_tok!().to_string();
        let sx = next_f64!();
        let sy = next_f64!();
        let sz = next_f64!();
        let dx = next_f64!();
        let dy = next_f64!();
        let dz = next_f64!();
        let _length = next_f64!();
        let ns = next_usize!();

        let dlen2 = dx * dx + dy * dy + dz * dz;
        let (ndx, ndy, ndz) = if dlen2 > 1e-24 {
            let dl = dlen2.sqrt();
            (dx / dl, dy / dl, dz / dl)
        } else {
            (0.0, 0.0, -1.0)
        };

        hole_ids.push(hid);

        for s in 0..ns {
            let depth = next_f64!();
            let x = sx + ndx * depth;
            let y = sy + ndy * depth;
            let z = sz + ndz * depth;
            sample_x.push(x as f32);
            sample_y.push(y as f32);
            sample_z.push(z as f32);
            sample_hole.push(h as u32);
            sample_local_idx.push(s as u32);
            for p in 0..num_props {
                let v = next_f64!();
                values[p].push(if v.is_finite() { v as f32 } else { f32::NAN });
            }
        }
    }

    let n = sample_x.len();
    if n == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut global_med = vec![0.0f64; num_props];
    let mut global_mad_scaled = vec![1.0f64; num_props];
    for p in 0..num_props {
        let mut buf: Vec<f64> = values[p].iter().filter(|x| x.is_finite()).map(|x| *x as f64).collect();
        if buf.is_empty() { continue; }
        let med = median_inplace(&mut buf);
        global_med[p] = med;
        let mut dev: Vec<f64> = buf.iter().map(|x| (x - med).abs()).collect();
        let mad = median_inplace(&mut dev);
        let scaled = mad * 1.4826;
        if scaled > 1e-12 {
            global_mad_scaled[p] = scaled;
        } else {
            let mut sumsq = 0.0;
            let mut cnt = 0.0;
            for &v in &buf { sumsq += (v - med) * (v - med); cnt += 1.0; }
            let sd = if cnt > 1.0 { (sumsq / (cnt - 1.0)).sqrt() } else { 0.0 };
            global_mad_scaled[p] = if sd > 1e-12 { sd } else { 1.0 };
        }
    }

    let (mut minx, mut maxx) = (f32::INFINITY, f32::NEG_INFINITY);
    let (mut miny, mut maxy) = (f32::INFINITY, f32::NEG_INFINITY);
    let (mut minz, mut maxz) = (f32::INFINITY, f32::NEG_INFINITY);
    for i in 0..n {
        let (x, y, z) = (sample_x[i], sample_y[i], sample_z[i]);
        if x < minx { minx = x; } if x > maxx { maxx = x; }
        if y < miny { miny = y; } if y > maxy { maxy = y; }
        if z < minz { minz = z; } if z > maxz { maxz = z; }
    }
    let span_x = (maxx - minx).max(1e-3) as f64;
    let span_y = (maxy - miny).max(1e-3) as f64;
    let span_z = (maxz - minz).max(1e-3) as f64;
    let vol = span_x * span_y * span_z;

    let (k_target, z_threshold, max_report_per_prop): (usize, f64, usize) = if n < 1_000 {
        (16, 3.5, n)
    } else if n < 20_000 {
        (14, 4.0, n / 4)
    } else if n < 200_000 {
        (12, 4.5, 5000)
    } else {
        (10, 5.0, 5000)
    };

    let target_per_cell = (k_target as f64 * 1.5).max(8.0);
    let cell_size = ((vol * target_per_cell / n as f64).cbrt()).max(1e-3);

    let mut grid: HashMap<i64, Vec<u32>> = HashMap::with_capacity((n / 8).max(16));
    let inv_cs = 1.0 / cell_size;
    let mx = minx as f64;
    let my = miny as f64;
    let mz = minz as f64;
    for i in 0..n {
        let cx = ((sample_x[i] as f64 - mx) * inv_cs).floor() as i64;
        let cy = ((sample_y[i] as f64 - my) * inv_cs).floor() as i64;
        let cz = ((sample_z[i] as f64 - mz) * inv_cs).floor() as i64;
        grid.entry(cell_key(cx, cy, cz)).or_insert_with(Vec::new).push(i as u32);
    }

    let mut suspects: Vec<Suspect> = Vec::new();
    let mut neighbor_buf: Vec<(f32, u32)> = Vec::with_capacity(256);
    let mut nbr_vals: Vec<f64> = Vec::with_capacity(64);
    let mut nbr_dev: Vec<f64> = Vec::with_capacity(64);

    let mut total_cells = grid.len();
    if total_cells == 0 { total_cells = 1; }
    let avg_per_cell = n as f64 / total_cells as f64;
    let search_radius: i64 = if avg_per_cell >= target_per_cell { 1 }
                              else if avg_per_cell >= target_per_cell * 0.3 { 2 }
                              else { 3 };

    for i in 0..n {
        let xi = sample_x[i] as f64;
        let yi = sample_y[i] as f64;
        let zi = sample_z[i] as f64;
        let cx = ((xi - mx) * inv_cs).floor() as i64;
        let cy = ((yi - my) * inv_cs).floor() as i64;
        let cz = ((zi - mz) * inv_cs).floor() as i64;

        neighbor_buf.clear();
        let mut r = 1i64;
        let max_r = search_radius;
        loop {
            for dx in -r..=r {
                for dy in -r..=r {
                    for dz in -r..=r {
                        if r > 1 && dx.abs() != r && dy.abs() != r && dz.abs() != r { continue; }
                        let key = cell_key(cx + dx, cy + dy, cz + dz);
                        if let Some(cell) = grid.get(&key) {
                            for &j in cell {
                                if j as usize == i { continue; }
                                let jdx = j as usize;
                                let ddx = sample_x[jdx] as f64 - xi;
                                let ddy = sample_y[jdx] as f64 - yi;
                                let ddz = sample_z[jdx] as f64 - zi;
                                let d2 = (ddx * ddx + ddy * ddy + ddz * ddz) as f32;
                                neighbor_buf.push((d2, j));
                            }
                        }
                    }
                }
            }
            if neighbor_buf.len() >= k_target || r >= max_r { break; }
            r += 1;
        }

        if neighbor_buf.is_empty() {
            for p in 0..num_props {
                let v = values[p][i];
                if !v.is_finite() { continue; }
                let z = ((v as f64 - global_med[p]).abs()) / global_mad_scaled[p].max(1e-12);
                if z > z_threshold {
                    suspects.push(Suspect {
                        hole_idx: sample_hole[i],
                        sample_idx: sample_local_idx[i],
                        prop_idx: p as u16,
                        confidence: z as f32,
                    });
                }
            }
            continue;
        }

        if neighbor_buf.len() > k_target {
            neighbor_buf.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));
            neighbor_buf.truncate(k_target);
        }
        let nn = neighbor_buf.len();

        for p in 0..num_props {
            let v = values[p][i];
            if !v.is_finite() { continue; }

            nbr_vals.clear();
            for k in 0..nn {
                let j = neighbor_buf[k].1 as usize;
                let nv = values[p][j];
                if nv.is_finite() {
                    nbr_vals.push(nv as f64);
                }
            }
            if nbr_vals.len() < 3 {
                let z = ((v as f64 - global_med[p]).abs()) / global_mad_scaled[p].max(1e-12);
                if z > z_threshold {
                    suspects.push(Suspect {
                        hole_idx: sample_hole[i],
                        sample_idx: sample_local_idx[i],
                        prop_idx: p as u16,
                        confidence: z as f32,
                    });
                }
                continue;
            }

            let med = {
                let mut tmp = nbr_vals.clone();
                median_inplace(&mut tmp)
            };
            nbr_dev.clear();
            for &x in &nbr_vals { nbr_dev.push((x - med).abs()); }
            let mad = median_inplace(&mut nbr_dev);
            let scale = if mad > 1e-12 { mad * 1.4826 } else { global_mad_scaled[p].max(1e-12) };
            let z = ((v as f64 - med).abs()) / scale;

            if z > z_threshold {
                suspects.push(Suspect {
                    hole_idx: sample_hole[i],
                    sample_idx: sample_local_idx[i],
                    prop_idx: p as u16,
                    confidence: z as f32,
                });
            }
        }
    }

    suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));

    if n >= 20_000 && num_props > 0 && suspects.len() > num_props * max_report_per_prop {
        let mut counts = vec![0usize; num_props];
        let mut kept: Vec<Suspect> = Vec::with_capacity(num_props * max_report_per_prop);
        for s in suspects.drain(..) {
            let pi = s.prop_idx as usize;
            if counts[pi] < max_report_per_prop {
                counts[pi] += 1;
                kept.push(s);
            }
        }
        suspects = kept;
    }

    writeln!(out, "{}", suspects.len()).unwrap();
    for s in &suspects {
        writeln!(
            out,
            "{} {} {} {:.4}",
            hole_ids[s.hole_idx as usize],
            s.sample_idx,
            prop_names[s.prop_idx as usize],
            s.confidence
        ).unwrap();
    }
}
