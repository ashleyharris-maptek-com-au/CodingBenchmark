use std::io::{self, Read, Write, BufWriter};

fn main() {
    // Read all of stdin as bytes (mazes can be up to ~2.5 GB of text).
    let mut input: Vec<u8> = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();

    let stdout = io::stdout();
    let mut out = BufWriter::with_capacity(1 << 20, stdout.lock());

    // ---- Parse header: H W ----
    let mut i: usize = 0;
    let len = input.len();

    while i < len && (input[i] == b' ' || input[i] == b'\t' || input[i] == b'\r' || input[i] == b'\n') {
        i += 1;
    }
    let mut h: usize = 0;
    while i < len && input[i] >= b'0' && input[i] <= b'9' {
        h = h * 10 + (input[i] - b'0') as usize;
        i += 1;
    }
    while i < len && (input[i] == b' ' || input[i] == b'\t') {
        i += 1;
    }
    let mut w: usize = 0;
    while i < len && input[i] >= b'0' && input[i] <= b'9' {
        w = w * 10 + (input[i] - b'0') as usize;
        i += 1;
    }
    // Skip to end of header line
    while i < len && input[i] != b'\n' {
        i += 1;
    }
    if i < len {
        i += 1;
    }

    if h == 0 || w == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let n = h.checked_mul(w).expect("maze too large");

    // ---- Load grid (row-major, one byte per cell) ----
    let mut grid: Vec<u8> = vec![b' '; n];
    let mut start: usize = usize::MAX;
    let mut goal: usize = usize::MAX;

    for y in 0..h {
        let row_off = y * w;
        let mut x = 0usize;
        while x < w && i < len {
            let c = input[i];
            if c == b'\n' {
                break;
            }
            if c == b'\r' {
                i += 1;
                continue;
            }
            grid[row_off + x] = c;
            if c == b'A' {
                start = row_off + x;
            } else if c == b'B' {
                goal = row_off + x;
            }
            x += 1;
            i += 1;
        }
        // Skip rest of input line
        while i < len && input[i] != b'\n' {
            i += 1;
        }
        if i < len {
            i += 1;
        }
    }

    drop(input);

    if start == usize::MAX || goal == usize::MAX {
        writeln!(out, "0").unwrap();
        return;
    }

    if start == goal {
        let y = start / w;
        let x = start - y * w;
        writeln!(out, "1").unwrap();
        writeln!(out, "{} {}", x, y).unwrap();
        return;
    }

    // ---- BFS ----
    // `from[cell]`:
    //   0 = unvisited
    //   1 = parent is above  (parent = cell - w)
    //   2 = parent is below  (parent = cell + w)
    //   3 = parent is left   (parent = cell - 1)
    //   4 = parent is right  (parent = cell + 1)
    //   5 = start
    let mut from: Vec<u8> = vec![0u8; n];
    from[start] = 5;

    // u32 indices: 50000*50000 = 2.5e9 < u32::MAX (~4.3e9).
    assert!(n <= u32::MAX as usize, "maze exceeds u32 index range");

    let mut queue: Vec<u32> = Vec::with_capacity(1024);
    queue.push(start as u32);
    let mut head: usize = 0;
    let mut found = false;

    while head < queue.len() {
        let cur = queue[head] as usize;
        head += 1;

        if cur == goal {
            found = true;
            break;
        }

        let y = cur / w;
        let x = cur - y * w;

        // Up
        if y > 0 {
            let nb = cur - w;
            unsafe {
                if *from.get_unchecked(nb) == 0 && *grid.get_unchecked(nb) != b'#' {
                    *from.get_unchecked_mut(nb) = 2;
                    queue.push(nb as u32);
                }
            }
        }
        // Down
        if y + 1 < h {
            let nb = cur + w;
            unsafe {
                if *from.get_unchecked(nb) == 0 && *grid.get_unchecked(nb) != b'#' {
                    *from.get_unchecked_mut(nb) = 1;
                    queue.push(nb as u32);
                }
            }
        }
        // Left
        if x > 0 {
            let nb = cur - 1;
            unsafe {
                if *from.get_unchecked(nb) == 0 && *grid.get_unchecked(nb) != b'#' {
                    *from.get_unchecked_mut(nb) = 4;
                    queue.push(nb as u32);
                }
            }
        }
        // Right
        if x + 1 < w {
            let nb = cur + 1;
            unsafe {
                if *from.get_unchecked(nb) == 0 && *grid.get_unchecked(nb) != b'#' {
                    *from.get_unchecked_mut(nb) = 3;
                    queue.push(nb as u32);
                }
            }
        }
    }

    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    // Free large buffers before allocating the path output.
    drop(queue);
    drop(grid);

    // ---- Reconstruct path from goal back to start ----
    let mut path: Vec<u32> = Vec::new();
    let mut cur = goal;
    loop {
        path.push(cur as u32);
        if cur == start {
            break;
        }
        let dir = from[cur];
        cur = match dir {
            1 => cur - w,
            2 => cur + w,
            3 => cur - 1,
            4 => cur + 1,
            _ => break,
        };
    }
    drop(from);
    path.reverse();

    // ---- Output ----
    writeln!(out, "{}", path.len()).unwrap();

    let mut buf = [0u8; 24];
    for &cell in &path {
        let cell = cell as usize;
        let y = cell / w;
        let x = cell - y * w;

        write_usize(&mut out, x, &mut buf);
        out.write_all(b" ").unwrap();
        write_usize(&mut out, y, &mut buf);
        out.write_all(b"\n").unwrap();
    }
}

#[inline]
fn write_usize<W: Write>(out: &mut W, mut v: usize, buf: &mut [u8; 24]) {
    if v == 0 {
        out.write_all(b"0").unwrap();
        return;
    }
    let mut i = buf.len();
    while v > 0 {
        i -= 1;
        buf[i] = b'0' + (v % 10) as u8;
        v /= 10;
    }
    out.write_all(&buf[i..]).unwrap();
}
