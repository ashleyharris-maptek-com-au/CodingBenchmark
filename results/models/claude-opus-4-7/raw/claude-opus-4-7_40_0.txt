#include <bits/stdc++.h>
using namespace std;

int N;
vector<long long> W;
vector<vector<int>> ADJ;

long long bestVal;
vector<int> bestSet;

void solve(vector<char> alive, vector<int> taken, long long curVal) {
    // Apply reductions until fixed point
    bool changed = true;
    while (changed) {
        changed = false;
        for (int v = 0; v < N; v++) {
            if (!alive[v]) continue;
            int d = 0;
            int sole = -1;
            for (int u : ADJ[v]) {
                if (alive[u]) {
                    d++;
                    sole = u;
                    if (d > 1) break;
                }
            }
            if (d == 0) {
                // Isolated: include iff weight > 0
                if (W[v] > 0) {
                    taken.push_back(v);
                    curVal += W[v];
                }
                alive[v] = 0;
                changed = true;
            } else if (W[v] <= 0) {
                // Never include a non-positive weight vertex
                alive[v] = 0;
                changed = true;
            } else if (d == 1 && W[v] >= W[sole]) {
                // Degree-1 with weight >= neighbor: safely include
                taken.push_back(v);
                curVal += W[v];
                alive[v] = 0;
                alive[sole] = 0;
                changed = true;
            }
        }
    }

    // Compute simple upper bound and pick branching vertex (max degree)
    long long ub = curVal;
    int branchV = -1;
    int maxDeg = -1;
    for (int v = 0; v < N; v++) {
        if (!alive[v]) continue;
        if (W[v] > 0) ub += W[v];
        int d = 0;
        for (int u : ADJ[v]) if (alive[u]) d++;
        if (d > maxDeg) { maxDeg = d; branchV = v; }
    }

    if (ub <= bestVal) return;

    if (branchV == -1) {
        if (curVal > bestVal) {
            bestVal = curVal;
            bestSet = taken;
        }
        return;
    }

    // Branch 1: include branchV
    {
        vector<char> alive2 = alive;
        vector<int> taken2 = taken;
        taken2.push_back(branchV);
        alive2[branchV] = 0;
        for (int u : ADJ[branchV]) alive2[u] = 0;
        solve(move(alive2), move(taken2), curVal + W[branchV]);
    }

    // Branch 2: exclude branchV
    alive[branchV] = 0;
    solve(move(alive), move(taken), curVal);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int M;
    if (!(cin >> N >> M)) return 0;
    W.assign(N, 0);
    ADJ.assign(N, {});
    for (int i = 0; i < N; i++) cin >> W[i];

    vector<char> forced_zero(N, 0);
    set<pair<int,int>> edges;
    for (int i = 0; i < M; i++) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= N || v >= N) continue;
        if (u == v) { forced_zero[u] = 1; continue; }
        int a = u, b = v;
        if (a > b) swap(a, b);
        if (edges.count({a, b})) continue;
        edges.insert({a, b});
        ADJ[a].push_back(b);
        ADJ[b].push_back(a);
    }

    vector<char> alive(N, 1);
    for (int v = 0; v < N; v++) if (forced_zero[v]) alive[v] = 0;

    // Greedy initial lower bound to seed pruning
    vector<int> order(N);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        double ra = (double)W[a] / (double)(ADJ[a].size() + 1);
        double rb = (double)W[b] / (double)(ADJ[b].size() + 1);
        return ra > rb;
    });
    {
        vector<char> ga = alive;
        long long gv = 0;
        vector<int> gs;
        for (int v : order) {
            if (!ga[v]) continue;
            if (W[v] <= 0) continue;
            gs.push_back(v);
            gv += W[v];
            ga[v] = 0;
            for (int u : ADJ[v]) ga[u] = 0;
        }
        bestVal = gv;
        bestSet = gs;
    }
    if (bestVal < 0) { bestVal = 0; bestSet.clear(); }

    vector<int> initTaken;
    solve(alive, initTaken, 0);

    cout << bestVal << "\n";
    vector<int> x(N, 0);
    for (int v : bestSet) x[v] = 1;
    for (int i = 0; i < N; i++) {
        cout << x[i];
        if (i + 1 < N) cout << " ";
    }
    cout << "\n";
    return 0;
}
