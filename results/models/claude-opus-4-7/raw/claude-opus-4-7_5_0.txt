use std::io::{self, Read, Write, BufWriter};
use std::time::{Instant, Duration};

const DX: [i32; 4] = [1, -1, 0, 0];
const DY: [i32; 4] = [0, 0, 1, -1];

struct Solver {
    w: i32,
    h: i32,
    blocked: Vec<bool>,
    visited: Vec<bool>,
    path: Vec<usize>,
    target: usize,
    require_cycle: bool,
    start_t: Instant,
    deadline: Duration,
    aborted: bool,
    node_counter: u64,
}

impl Solver {
    #[inline]
    fn neighbors(&self, p: usize) -> [i32; 4] {
        let x = (p as i32) % self.w;
        let y = (p as i32) / self.w;
        let mut out = [-1i32; 4];
        for d in 0..4 {
            let nx = x + DX[d];
            let ny = y + DY[d];
            if nx >= 0 && nx < self.w && ny >= 0 && ny < self.h {
                let np = (ny * self.w + nx) as usize;
                if !self.blocked[np] {
                    out[d] = np as i32;
                }
            }
        }
        out
    }

    #[inline]
    fn unvisited_neighbor_count(&self, p: usize) -> i32 {
        let nbrs = self.neighbors(p);
        let mut c = 0;
        for d in 0..4 {
            if nbrs[d] >= 0 && !self.visited[nbrs[d] as usize] {
                c += 1;
            }
        }
        c
    }

    fn check_time(&mut self) -> bool {
        if self.aborted {
            return false;
        }
        self.node_counter += 1;
        if self.node_counter & 0x3FFF == 0 {
            if self.start_t.elapsed() > self.deadline {
                self.aborted = true;
                return false;
            }
        }
        true
    }

    fn dfs(&mut self) -> bool {
        if !self.check_time() {
            return false;
        }

        if self.path.len() == self.target {
            if self.require_cycle {
                let last = *self.path.last().unwrap();
                let nbrs = self.neighbors(last);
                for d in 0..4 {
                    if nbrs[d] == 0 {
                        return true;
                    }
                }
                return false;
            }
            return true;
        }

        let current = *self.path.last().unwrap();
        let nbrs = self.neighbors(current);

        let mut options: [(i32, i32); 4] = [(0, -1); 4];
        let mut nopt = 0;
        for d in 0..4 {
            if nbrs[d] >= 0 && !self.visited[nbrs[d] as usize] {
                let c = self.unvisited_neighbor_count(nbrs[d] as usize);
                options[nopt] = (c, nbrs[d]);
                nopt += 1;
            }
        }

        if nopt == 0 {
            return false;
        }

        for i in 1..nopt {
            let mut j = i;
            while j > 0 && options[j - 1].0 > options[j].0 {
                options.swap(j - 1, j);
                j -= 1;
            }
        }

        let cells_left = self.target - self.path.len();

        let mut zero_count = 0;
        for i in 0..nopt {
            if options[i].0 == 0 {
                zero_count += 1;
            }
        }

        if zero_count > 1 {
            return false;
        }

        if zero_count == 1 && cells_left > 1 {
            return false;
        }

        for i in 0..nopt {
            let np = options[i].1 as usize;

            if self.require_cycle && cells_left == 1 {
                let nbrs2 = self.neighbors(np);
                let mut adj_to_start = false;
                for d in 0..4 {
                    if nbrs2[d] == 0 {
                        adj_to_start = true;
                        break;
                    }
                }
                if !adj_to_start {
                    continue;
                }
            }

            self.visited[np] = true;
            self.path.push(np);

            if self.dfs() {
                return true;
            }

            self.path.pop();
            self.visited[np] = false;

            if self.aborted {
                return false;
            }
        }

        false
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let w: i32 = it.next().unwrap().parse().unwrap();
    let h: i32 = it.next().unwrap().parse().unwrap();
    let obs_count: usize = it.next().unwrap().parse().unwrap();
    let rc: i32 = it.next().unwrap().parse().unwrap();
    let require_cycle = rc != 0;

    if w <= 0 || h <= 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let n = (w * h) as usize;
    let mut blocked = vec![false; n];
    for _ in 0..obs_count {
        let x: i32 = it.next().unwrap().parse().unwrap();
        let y: i32 = it.next().unwrap().parse().unwrap();
        if x >= 0 && x < w && y >= 0 && y < h {
            blocked[(y * w + x) as usize] = true;
        }
    }

    if blocked[0] {
        writeln!(out, "0").unwrap();
        return;
    }

    let target = n - blocked.iter().filter(|&&b| b).count();

    if target == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    if target == 1 {
        if require_cycle {
            writeln!(out, "0").unwrap();
        } else {
            writeln!(out, "1").unwrap();
            writeln!(out, "0 0").unwrap();
        }
        return;
    }

    let mut seen = vec![false; n];
    let mut stack: Vec<usize> = Vec::with_capacity(n);
    stack.push(0);
    seen[0] = true;
    let mut cnt = 0usize;
    while let Some(p) = stack.pop() {
        cnt += 1;
        let x = (p as i32) % w;
        let y = (p as i32) / w;
        for d in 0..4 {
            let nx = x + DX[d];
            let ny = y + DY[d];
            if nx >= 0 && nx < w && ny >= 0 && ny < h {
                let np = (ny * w + nx) as usize;
                if !blocked[np] && !seen[np] {
                    seen[np] = true;
                    stack.push(np);
                }
            }
        }
    }
    if cnt != target {
        writeln!(out, "0").unwrap();
        return;
    }

    if require_cycle {
        let mut b = 0i64;
        let mut ww = 0i64;
        for y in 0..h {
            for x in 0..w {
                if !blocked[(y * w + x) as usize] {
                    if (x + y) % 2 == 0 {
                        b += 1;
                    } else {
                        ww += 1;
                    }
                }
            }
        }
        if b != ww {
            writeln!(out, "0").unwrap();
            return;
        }
        let mut deg = 0;
        for d in 0..4 {
            let nx = DX[d];
            let ny = DY[d];
            if nx >= 0 && nx < w && ny >= 0 && ny < h {
                if !blocked[(ny * w + nx) as usize] {
                    deg += 1;
                }
            }
        }
        if deg < 2 {
            writeln!(out, "0").unwrap();
            return;
        }
    }

    let mut solver = Solver {
        w,
        h,
        blocked,
        visited: vec![false; n],
        path: Vec::with_capacity(target),
        target,
        require_cycle,
        start_t: Instant::now(),
        deadline: Duration::from_secs(270),
        aborted: false,
        node_counter: 0,
    };

    solver.visited[0] = true;
    solver.path.push(0);

    let found = solver.dfs();

    if found && solver.path.len() == solver.target {
        writeln!(out, "{}", solver.path.len()).unwrap();
        for &p in &solver.path {
            let x = (p as i32) % w;
            let y = (p as i32) / w;
            writeln!(out, "{} {}", x, y).unwrap();
        }
    } else {
        writeln!(out, "0").unwrap();
    }
}
