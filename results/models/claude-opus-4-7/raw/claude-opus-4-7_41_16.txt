; SPIR-V
; Version: 1.0
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %uv %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %uv Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
       %fn_v = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %bool = OpTypeBool
        %v2f = OpTypeVector %float 2
        %v4f = OpTypeVector %float 4

        %f_0 = OpConstant %float 0.0
        %f_1 = OpConstant %float 1.0
        %f_2 = OpConstant %float 2.0
        %f_3 = OpConstant %float 3.0
        %f_4 = OpConstant %float 4.0
       %f_05 = OpConstant %float 0.5
       %f_n1 = OpConstant %float -1.0
       %f_n2 = OpConstant %float -2.0
       %f_20 = OpConstant %float 20.0

        %i_0 = OpConstant %int 0
        %i_1 = OpConstant %int 1
       %i_20 = OpConstant %int 20

       %v2_0 = OpConstantComposite %v2f %f_0 %f_0

     %pi_v2f = OpTypePointer Input %v2f
     %po_v4f = OpTypePointer Output %v4f
     %pf_v2f = OpTypePointer Function %v2f
     %pf_int = OpTypePointer Function %int

         %uv = OpVariable %pi_v2f Input
   %outColor = OpVariable %po_v4f Output

       %main = OpFunction %void None %fn_v
      %entry = OpLabel
      %z_var = OpVariable %pf_v2f Function
      %i_var = OpVariable %pf_int Function
               OpStore %z_var %v2_0
               OpStore %i_var %i_0

       %uv_v = OpLoad %v2f %uv
        %uvx = OpCompositeExtract %float %uv_v 0
        %uvy = OpCompositeExtract %float %uv_v 1
        %cxt = OpFMul %float %uvx %f_3
         %cx = OpFAdd %float %cxt %f_n2
        %cyt = OpFMul %float %uvy %f_2
         %cy = OpFAdd %float %cyt %f_n1

               OpBranch %loop_h

     %loop_h = OpLabel
               OpLoopMerge %loop_m %loop_c None
               OpBranch %loop_cond

  %loop_cond = OpLabel
      %i_cur = OpLoad %int %i_var
        %lt  = OpSLessThan %bool %i_cur %i_20
               OpBranchConditional %lt %loop_b %loop_m

     %loop_b = OpLabel
        %z_v = OpLoad %v2f %z_var
         %zx = OpCompositeExtract %float %z_v 0
         %zy = OpCompositeExtract %float %z_v 1
        %zx2 = OpFMul %float %zx %zx
        %zy2 = OpFMul %float %zy %zy
       %mag2 = OpFAdd %float %zx2 %zy2
        %esc = OpFOrdGreaterThan %bool %mag2 %f_4
               OpSelectionMerge %sel_m None
               OpBranchConditional %esc %sel_b %sel_a

      %sel_b = OpLabel
               OpBranch %loop_m

      %sel_a = OpLabel
       %nzxt = OpFSub %float %zx2 %zy2
        %nzx = OpFAdd %float %nzxt %cx
       %zxzy = OpFMul %float %zx %zy
       %nzyt = OpFMul %float %zxzy %f_2
        %nzy = OpFAdd %float %nzyt %cy
      %new_z = OpCompositeConstruct %v2f %nzx %nzy
               OpStore %z_var %new_z
               OpBranch %sel_m

      %sel_m = OpLabel
               OpBranch %loop_c

     %loop_c = OpLabel
         %ic = OpLoad %int %i_var
        %inn = OpIAdd %int %ic %i_1
               OpStore %i_var %inn
               OpBranch %loop_h

     %loop_m = OpLabel
         %fi = OpLoad %int %i_var
        %fif = OpConvertSToF %float %fi
          %t = OpFDiv %float %fif %f_20
         %th = OpFMul %float %t %f_05
        %omt = OpFSub %float %f_1 %t
        %col = OpCompositeConstruct %v4f %t %th %omt %f_1
               OpStore %outColor %col
               OpReturn
               OpFunctionEnd
