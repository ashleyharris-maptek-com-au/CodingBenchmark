import sys
import math


def read_input():
    data = sys.stdin.read().split()
    idx = 0
    N = int(data[idx]); idx += 1
    K = int(data[idx]); idx += 1
    CAP = float(data[idx]); idx += 1
    dx = float(data[idx]); idx += 1
    dy = float(data[idx]); idx += 1
    customers = []
    for i in range(N):
        x = float(data[idx]); idx += 1
        y = float(data[idx]); idx += 1
        d = float(data[idx]); idx += 1
        customers.append((x, y, d))
    return N, K, CAP, (dx, dy), customers


def euclid(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def build_distance_matrix(depot, customers):
    N = len(customers)
    pts = [depot] + [(c[0], c[1]) for c in customers]
    M = N + 1
    D = [[0.0] * M for _ in range(M)]
    for i in range(M):
        for j in range(i + 1, M):
            d = euclid(pts[i], pts[j])
            D[i][j] = d
            D[j][i] = d
    return D


def route_distance(route, D):
    # route is list of 0-based customer indices; depot is node 0 in D, customers are 1..N
    if not route:
        return 0.0
    total = D[0][route[0] + 1]
    for i in range(len(route) - 1):
        total += D[route[i] + 1][route[i + 1] + 1]
    total += D[route[-1] + 1][0]
    return total


def clarke_wright(N, CAP, customers, D):
    routes = {i: [i] for i in range(N)}
    route_of = list(range(N))
    load = [customers[i][2] for i in range(N)]

    savings = []
    for i in range(N):
        for j in range(i + 1, N):
            s = D[0][i + 1] + D[0][j + 1] - D[i + 1][j + 1]
            if s > 0:
                savings.append((s, i, j))
    savings.sort(reverse=True)

    for s, i, j in savings:
        ri = route_of[i]
        rj = route_of[j]
        if ri == rj:
            continue
        route_i = routes[ri]
        route_j = routes[rj]
        i_at_start = route_i[0] == i
        i_at_end = route_i[-1] == i
        j_at_start = route_j[0] == j
        j_at_end = route_j[-1] == j
        if not (i_at_start or i_at_end):
            continue
        if not (j_at_start or j_at_end):
            continue
        if load[ri] + load[rj] > CAP + 1e-9:
            continue
        if i_at_end and j_at_start:
            new_route = route_i + route_j
        elif i_at_end and j_at_end:
            new_route = route_i + route_j[::-1]
        elif i_at_start and j_at_end:
            new_route = route_j + route_i
        elif i_at_start and j_at_start:
            new_route = route_i[::-1] + route_j
        else:
            continue
        new_load = load[ri] + load[rj]
        new_id = ri
        for c in new_route:
            route_of[c] = new_id
        routes[new_id] = new_route
        load[new_id] = new_load
        if rj in routes and rj != new_id:
            del routes[rj]

    return list(routes.values())


def two_opt(route, D, max_iter=50):
    if len(route) < 4:
        return route
    best = route[:]
    improved = True
    it = 0
    while improved and it < max_iter:
        improved = False
        it += 1
        nodes = [0] + [c + 1 for c in best] + [0]
        n = len(best)
        for i in range(1, n):
            for k in range(i + 1, n + 1):
                a, b = nodes[i - 1], nodes[i]
                c, d = nodes[k], nodes[k + 1]
                if a == c or b == d:
                    continue
                delta = (D[a][c] + D[b][d]) - (D[a][b] + D[c][d])
                if delta < -1e-12:
                    best = best[:i - 1] + best[i - 1:k][::-1] + best[k:]
                    improved = True
                    nodes = [0] + [cc + 1 for cc in best] + [0]
    return best


def merge_to_k(routes, K, customers, CAP, D):
    routes = [r[:] for r in routes if r]
    while len(routes) > K:
        best = None
        m = len(routes)
        loads = [sum(customers[c][2] for c in r) for r in routes]
        dists = [route_distance(r, D) for r in routes]
        for i in range(m):
            for j in range(m):
                if i == j:
                    continue
                if loads[i] + loads[j] > CAP + 1e-9:
                    continue
                merged = routes[i] + routes[j]
                new = route_distance(merged, D)
                delta = new - (dists[i] + dists[j])
                if best is None or delta < best[0]:
                    best = (delta, i, j, merged)
        if best is None:
            break
        _, i, j, merged = best
        if i > j:
            routes.pop(i); routes.pop(j)
        else:
            routes.pop(j); routes.pop(i)
        routes.append(merged)
    return routes


def solve():
    N, K, CAP, depot, customers = read_input()
    D = build_distance_matrix(depot, customers)

    if N == 0:
        print("0.000000")
        print(0)
        return

    routes = clarke_wright(N, CAP, customers, D)

    if len(routes) > K:
        routes = merge_to_k(routes, K, customers, CAP, D)

    routes = [two_opt(r, D) for r in routes if r]

    total = sum(route_distance(r, D) for r in routes)

    out_lines = [f"{total:.6f}", str(len(routes))]
    for r in routes:
        out_lines.append(" ".join(str(c) for c in r))
    sys.stdout.write("\n".join(out_lines) + "\n")


if __name__ == "__main__":
    solve()
