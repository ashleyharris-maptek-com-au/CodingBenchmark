using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Text;

public class Program
{
    static int CW, CH, CD;

    // Placed box storage (struct-of-arrays for cache friendliness).
    static int[] PX, PY, PZ, PX2, PY2, PZ2;
    static int placedCount = 0;

    // Sparse spatial hash grid for collision detection.
    static Dictionary<long, List<int>> grid;
    static int cellSize;
    static long gridStride;

    static long CellKey(int x, int y, int z)
    {
        return ((long)x * gridStride + y) * gridStride + z;
    }

    public static void Main()
    {
        var sw = Stopwatch.StartNew();
        long timeLimit = 25000; // leave headroom under the 30s wall

        string input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int ti = 0;
        int N = int.Parse(tokens[ti++]);
        CW = int.Parse(tokens[ti++]);
        CH = int.Parse(tokens[ti++]);
        CD = int.Parse(tokens[ti++]);

        if (N <= 0)
        {
            Console.Write("0\n");
            return;
        }

        int[] bw = new int[N], bh = new int[N], bd = new int[N];
        long[] vol = new long[N];
        int[] order = new int[N];
        long totalVol = 0;
        int validBoxCount = 0;

        for (int i = 0; i < N; i++)
        {
            bw[i] = int.Parse(tokens[ti++]);
            bh[i] = int.Parse(tokens[ti++]);
            bd[i] = int.Parse(tokens[ti++]);
            vol[i] = (long)bw[i] * bh[i] * bd[i];
            order[i] = i;
            if (CanFitContainer(bw[i], bh[i], bd[i]))
            {
                totalVol += vol[i];
                validBoxCount++;
            }
        }

        // Largest-first ordering.
        Array.Sort(order, (a, b) => vol[b].CompareTo(vol[a]));

        PX = new int[N]; PY = new int[N]; PZ = new int[N];
        PX2 = new int[N]; PY2 = new int[N]; PZ2 = new int[N];

        // Adaptive cell size: aim for cells holding O(1) boxes.
        long avgVol = validBoxCount > 0 ? totalVol / Math.Max(1, validBoxCount) : 1;
        long cbrt = (long)Math.Max(1.0, Math.Cbrt((double)avgVol));
        cellSize = (int)Math.Max(1L, cbrt);
        int maxCD = Math.Max(CW, Math.Max(CH, CD));
        long gs = (long)(maxCD / cellSize) + 2L;
        if (gs < 4) gs = 4;
        gridStride = gs;
        grid = new Dictionary<long, List<int>>(Math.Min(1 << 20, Math.Max(1024, N * 4)));

        var results = new List<(int idx, int x, int y, int z)>(N);

        int[] epX = new int[64];
        int[] epY = new int[64];
        int[] epZ = new int[64];
        int epCount = 1;
        epX[0] = 0; epY[0] = 0; epZ[0] = 0;

        // For very large N, switch to first-fit.
        bool firstFit = N > 800;

        int[] rotW = new int[6];
        int[] rotH = new int[6];
        int[] rotD = new int[6];

        for (int oi = 0; oi < N; oi++)
        {
            if ((oi & 15) == 0 && sw.ElapsedMilliseconds > timeLimit) break;

            int bi = order[oi];
            int w = bw[bi], h = bh[bi], d = bd[bi];

            // Generate unique permutations of (w,h,d).
            int rotCount = 0;
            AddRot(rotW, rotH, rotD, ref rotCount, w, h, d);
            AddRot(rotW, rotH, rotD, ref rotCount, w, d, h);
            AddRot(rotW, rotH, rotD, ref rotCount, h, w, d);
            AddRot(rotW, rotH, rotD, ref rotCount, h, d, w);
            AddRot(rotW, rotH, rotD, ref rotCount, d, w, h);
            AddRot(rotW, rotH, rotD, ref rotCount, d, h, w);

            int bestEp = -1;
            int bestX = 0, bestY = 0, bestZ = 0;
            int bestW = 0, bestH = 0, bestD = 0;
            long bestScore = long.MaxValue;

            bool done = false;
            for (int e = 0; e < epCount && !done; e++)
            {
                int ex = epX[e], ey = epY[e], ez = epZ[e];
                for (int r = 0; r < rotCount; r++)
                {
                    int rw = rotW[r], rh = rotH[r], rd = rotD[r];
                    if (ex + rw > CW || ey + rh > CH || ez + rd > CD) continue;
                    if (OverlapsGrid(ex, ey, ez, rw, rh, rd)) continue;

                    // Score: bottom (y) first, then back (z), then left (x).
                    long score = (long)ey * 1_000_000_000_000L
                               + (long)ez * 1_000_000L
                               + ex;
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestEp = e;
                        bestX = ex; bestY = ey; bestZ = ez;
                        bestW = rw; bestH = rh; bestD = rd;
                        if (firstFit) { done = true; break; }
                    }
                }
            }

            if (bestEp >= 0)
            {
                int pi = placedCount++;
                PX[pi] = bestX; PY[pi] = bestY; PZ[pi] = bestZ;
                PX2[pi] = bestX + bestW;
                PY2[pi] = bestY + bestH;
                PZ2[pi] = bestZ + bestD;
                results.Add((bi, bestX, bestY, bestZ));
                AddToGrid(pi);

                // Swap-remove the used EP.
                epX[bestEp] = epX[epCount - 1];
                epY[bestEp] = epY[epCount - 1];
                epZ[bestEp] = epZ[epCount - 1];
                epCount--;

                // Add 3 new extreme points at the placed box's far corners.
                AddEp(ref epX, ref epY, ref epZ, ref epCount, bestX + bestW, bestY, bestZ);
                AddEp(ref epX, ref epY, ref epZ, ref epCount, bestX, bestY + bestH, bestZ);
                AddEp(ref epX, ref epY, ref epZ, ref epCount, bestX, bestY, bestZ + bestD);
            }
        }

        var sb = new StringBuilder(results.Count * 24 + 16);
        sb.Append(results.Count).Append('\n');
        for (int i = 0; i < results.Count; i++)
        {
            var r = results[i];
            sb.Append(r.idx).Append(' ')
              .Append(r.x).Append(' ')
              .Append(r.y).Append(' ')
              .Append(r.z).Append('\n');
        }
        Console.Write(sb.ToString());
    }

    static bool CanFitContainer(int w, int h, int d)
    {
        if (w <= CW && h <= CH && d <= CD) return true;
        if (w <= CW && d <= CH && h <= CD) return true;
        if (h <= CW && w <= CH && d <= CD) return true;
        if (h <= CW && d <= CH && w <= CD) return true;
        if (d <= CW && w <= CH && h <= CD) return true;
        if (d <= CW && h <= CH && w <= CD) return true;
        return false;
    }

    static void AddRot(int[] rW, int[] rH, int[] rD, ref int count, int w, int h, int d)
    {
        for (int i = 0; i < count; i++)
        {
            if (rW[i] == w && rH[i] == h && rD[i] == d) return;
        }
        rW[count] = w; rH[count] = h; rD[count] = d;
        count++;
    }

    static void AddToGrid(int pi)
    {
        int gx1 = PX[pi] / cellSize;
        int gx2 = (PX2[pi] - 1) / cellSize;
        int gy1 = PY[pi] / cellSize;
        int gy2 = (PY2[pi] - 1) / cellSize;
        int gz1 = PZ[pi] / cellSize;
        int gz2 = (PZ2[pi] - 1) / cellSize;
        for (int gx = gx1; gx <= gx2; gx++)
        {
            for (int gy = gy1; gy <= gy2; gy++)
            {
                for (int gz = gz1; gz <= gz2; gz++)
                {
                    long k = CellKey(gx, gy, gz);
                    if (!grid.TryGetValue(k, out var list))
                    {
                        list = new List<int>(2);
                        grid[k] = list;
                    }
                    list.Add(pi);
                }
            }
        }
    }

    static HashSet<int> _seen = new HashSet<int>(64);

    static bool OverlapsGrid(int x, int y, int z, int w, int h, int d)
    {
        int x2 = x + w, y2 = y + h, z2 = z + d;
        int gx1 = x / cellSize, gx2 = (x2 - 1) / cellSize;
        int gy1 = y / cellSize, gy2 = (y2 - 1) / cellSize;
        int gz1 = z / cellSize, gz2 = (z2 - 1) / cellSize;
        _seen.Clear();
        for (int gx = gx1; gx <= gx2; gx++)
        {
            for (int gy = gy1; gy <= gy2; gy++)
            {
                for (int gz = gz1; gz <= gz2; gz++)
                {
                    long k = CellKey(gx, gy, gz);
                    if (!grid.TryGetValue(k, out var list)) continue;
                    int cnt = list.Count;
                    for (int li = 0; li < cnt; li++)
                    {
                        int pi = list[li];
                        if (!_seen.Add(pi)) continue;
                        if (x < PX2[pi] && x2 > PX[pi] &&
                            y < PY2[pi] && y2 > PY[pi] &&
                            z < PZ2[pi] && z2 > PZ[pi])
                            return true;
                    }
                }
            }
        }
        return false;
    }

    static void AddEp(ref int[] epX, ref int[] epY, ref int[] epZ, ref int epCount, int x, int y, int z)
    {
        if (x >= CW || y >= CH || z >= CD) return;
        if (x < 0 || y < 0 || z < 0) return;
        for (int i = 0; i < epCount; i++)
        {
            if (epX[i] == x && epY[i] == y && epZ[i] == z) return;
        }
        if (epCount >= epX.Length)
        {
            int nlen = epX.Length * 2;
            Array.Resize(ref epX, nlen);
            Array.Resize(ref epY, nlen);
            Array.Resize(ref epZ, nlen);
        }
        epX[epCount] = x;
        epY[epCount] = y;
        epZ[epCount] = z;
        epCount++;
    }
}
