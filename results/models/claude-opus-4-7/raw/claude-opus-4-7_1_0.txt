// Adaptive TSP solver
// - N <= 8000   : Nearest-neighbor construction + time-bounded 2-opt
// - N >  8000   : Hilbert space-filling curve ordering,
//                 with sliding-window 2-opt refinement for moderate N
// Output starts at city 0 (rotation applied at the end).

#include <cstdio>
#include <cstdint>
#include <vector>
#include <algorithm>
#include <cmath>
#include <chrono>
#include <utility>

using namespace std;
using namespace std::chrono;

struct Pt { float x, y; };

// ---------------- Fast input ----------------
static char inbuf[1 << 16];
static int  inpos = 0, inend = 0;

static inline int gc() {
    if (inpos >= inend) {
        inend = (int)fread(inbuf, 1, sizeof(inbuf), stdin);
        inpos = 0;
        if (inend <= 0) return -1;
    }
    return (unsigned char)inbuf[inpos++];
}
static inline int rInt() {
    int c, x = 0, neg = 0;
    do { c = gc(); } while (c != -1 && c <= ' ');
    if (c == -1) return 0;
    if (c == '-') { neg = 1; c = gc(); }
    while (c >= '0' && c <= '9') { x = x * 10 + (c - '0'); c = gc(); }
    return neg ? -x : x;
}
static inline float rFlt() {
    int c;
    do { c = gc(); } while (c != -1 && c <= ' ');
    if (c == -1) return 0.f;
    int neg = 0;
    if (c == '-') { neg = 1; c = gc(); }
    else if (c == '+') { c = gc(); }
    double x = 0.0;
    while (c >= '0' && c <= '9') { x = x * 10.0 + (c - '0'); c = gc(); }
    if (c == '.') {
        c = gc();
        double frac = 0.1;
        while (c >= '0' && c <= '9') {
            x   += (c - '0') * frac;
            frac *= 0.1;
            c    = gc();
        }
    }
    if (c == 'e' || c == 'E') {
        c = gc();
        int eneg = 0;
        if (c == '-') { eneg = 1; c = gc(); }
        else if (c == '+') { c = gc(); }
        int e = 0;
        while (c >= '0' && c <= '9') { e = e * 10 + (c - '0'); c = gc(); }
        double m = 1.0;
        for (int i = 0; i < e; i++) m *= 10.0;
        if (eneg) x /= m; else x *= m;
    }
    return (float)(neg ? -x : x);
}

// ---------------- Geometry ----------------
static inline float distEu(const Pt& a, const Pt& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return sqrtf(dx * dx + dy * dy);
}

// ---------------- Hilbert curve (xy -> d) ----------------
// order must be a power of two; returns Hilbert index of (x,y) on [0, order).
static uint64_t hilbertD(uint32_t order, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = order >> 1; s > 0; s >>= 1) {
        uint32_t rx = (x & s) ? 1u : 0u;
        uint32_t ry = (y & s) ? 1u : 0u;
        d += (uint64_t)s * (uint64_t)s * ((3u * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = s - 1 - x;
                y = s - 1 - y;
            }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

int main() {
    auto t0 = steady_clock::now();
    auto elapsed = [&]() {
        return duration_cast<duration<double>>(steady_clock::now() - t0).count();
    };
    const double TIME_LIMIT = 270.0;          // hard wall, well below 300s
    const double NN_BUDGET  = 0.55 * TIME_LIMIT;
    const double OPT_BUDGET = 0.88 * TIME_LIMIT;

    int N = rInt();
    if (N <= 0) return 0;

    vector<Pt> pts(N);
    for (int i = 0; i < N; i++) {
        pts[i].x = rFlt();
        pts[i].y = rFlt();
    }

    vector<int> tour(N);

    if (N == 1) {
        tour[0] = 0;
    }
    else if (N <= 8000) {
        // --- Nearest neighbor construction starting from 0 ---
        vector<char> used(N, 0);
        tour[0] = 0;
        used[0] = 1;
        for (int k = 1; k < N; k++) {
            const Pt& pp = pts[tour[k - 1]];
            int   best  = -1;
            float bestD = INFINITY;
            for (int j = 0; j < N; j++) {
                if (used[j]) continue;
                float dx = pts[j].x - pp.x;
                float dy = pts[j].y - pp.y;
                float d  = dx * dx + dy * dy;
                if (d < bestD) { bestD = d; best = j; }
            }
            tour[k]    = best;
            used[best] = 1;
            if (elapsed() > NN_BUDGET) {
                // Safety fallback: append remaining indices in natural order
                int idx = k + 1;
                for (int j = 0; j < N && idx < N; j++) {
                    if (!used[j]) { tour[idx++] = j; used[j] = 1; }
                }
                break;
            }
        }

        // --- 2-opt refinement (full neighborhood, time-bounded) ---
        bool improved = true;
        while (improved && elapsed() < OPT_BUDGET) {
            improved = false;
            for (int i = 0; i + 1 < N; i++) {
                if ((i & 127) == 0 && elapsed() > OPT_BUDGET) break;
                int   a   = tour[i];
                int   b   = tour[i + 1];
                float dab = distEu(pts[a], pts[b]);
                for (int j = i + 2; j < N; j++) {
                    int c = tour[j];
                    int d = tour[(j + 1) % N];
                    if (d == a) continue;            // shared closing edge
                    float dcd = distEu(pts[c], pts[d]);
                    float dac = distEu(pts[a], pts[c]);
                    float dbd = distEu(pts[b], pts[d]);
                    if (dac + dbd + 1e-6f < dab + dcd) {
                        reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        improved = true;
                        b   = tour[i + 1];
                        dab = distEu(pts[a], pts[b]);
                    }
                }
            }
        }
    }
    else {
        // --- Hilbert space-filling curve ordering ---
        float minX = pts[0].x, maxX = pts[0].x;
        float minY = pts[0].y, maxY = pts[0].y;
        for (int i = 1; i < N; i++) {
            float x = pts[i].x, y = pts[i].y;
            if (x < minX) minX = x; else if (x > maxX) maxX = x;
            if (y < minY) minY = y; else if (y > maxY) maxY = y;
        }
        float range = (maxX - minX > maxY - minY) ? (maxX - minX)
                                                  : (maxY - minY);
        if (!(range > 0.f)) range = 1.f;

        const uint32_t HORDER = 1u << 16;             // 65536 x 65536 grid
        const float    scale  = (float)(HORDER - 1) / range;

        vector<pair<uint64_t, int>> hd(N);
        for (int i = 0; i < N; i++) {
            float fx = (pts[i].x - minX) * scale;
            float fy = (pts[i].y - minY) * scale;
            uint32_t gx = (fx <= 0.f)                    ? 0u
                        : (fx >= (float)(HORDER - 1))    ? HORDER - 1
                                                         : (uint32_t)fx;
            uint32_t gy = (fy <= 0.f)                    ? 0u
                        : (fy >= (float)(HORDER - 1))    ? HORDER - 1
                                                         : (uint32_t)fy;
            hd[i] = { hilbertD(HORDER, gx, gy), i };
        }
        sort(hd.begin(), hd.end(),
             [](const pair<uint64_t, int>& a, const pair<uint64_t, int>& b) {
                 return a.first < b.first;
             });
        for (int i = 0; i < N; i++) tour[i] = hd[i].second;
        vector<pair<uint64_t, int>>().swap(hd);       // release scratch

        // --- Sliding-window 2-opt refinement for moderate sizes ---
        // Per-pass cost ~ O(N * W^2). Skip entirely for huge N.
        int W = 0;
        if      (N <=     50000) W = 40;
        else if (N <=    200000) W = 20;
        else if (N <=   1000000) W = 10;
        else                      W = 0;

        if (W > 0) {
            bool stop = false;
            for (int pass = 0; pass < 6 && !stop && elapsed() < OPT_BUDGET; pass++) {
                bool anyImp = false;
                for (int start = 0; start + W < N; start++) {
                    if ((start & 4095) == 0 && elapsed() > OPT_BUDGET) {
                        stop = true; break;
                    }
                    int end = start + W;
                    for (int i = start; i < end - 1; i++) {
                        int   a   = tour[i];
                        int   b   = tour[i + 1];
                        float dab = distEu(pts[a], pts[b]);
                        for (int j = i + 2; j <= end; j++) {
                            int c = tour[j];
                            int d = (j + 1 < N) ? tour[j + 1] : tour[0];
                            if (d == a) continue;
                            float dcd = distEu(pts[c], pts[d]);
                            float dac = distEu(pts[a], pts[c]);
                            float dbd = distEu(pts[b], pts[d]);
                            if (dac + dbd + 1e-6f < dab + dcd) {
                                reverse(tour.begin() + i + 1,
                                        tour.begin() + j + 1);
                                anyImp = true;
                                b   = tour[i + 1];
                                dab = distEu(pts[a], pts[b]);
                            }
                        }
                    }
                }
                if (!anyImp) break;
            }
        }
    }

    // --- Rotate so city 0 is first ---
    if (!tour.empty() && tour[0] != 0) {
        int pos0 = 0;
        for (int i = 0; i < N; i++) {
            if (tour[i] == 0) { pos0 = i; break; }
        }
        rotate(tour.begin(), tour.begin() + pos0, tour.end());
    }

    // --- Fast output ---
    constexpr int BUFSZ = 1 << 20;
    static char outbuf[BUFSZ];
    int op = 0;
    for (int i = 0; i < N; i++) {
        if (op > BUFSZ - 16) {
            fwrite(outbuf, 1, op, stdout);
            op = 0;
        }
        int v = tour[i];
        if (v == 0) {
            outbuf[op++] = '0';
        } else {
            char tmp[12];
            int  k = 0;
            while (v > 0) { tmp[k++] = (char)('0' + (v % 10)); v /= 10; }
            while (k > 0)  outbuf[op++] = tmp[--k];
        }
        outbuf[op++] = (i + 1 == N) ? '\n' : ' ';
    }
    if (op > 0) fwrite(outbuf, 1, op, stdout);

    return 0;
}
