; SPIR-V
; Version: 1.0
               OpCapability Shader
       %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

               OpDecorate %worldPos Location 0
               OpDecorate %normal   Location 1
               OpDecorate %uv       Location 2
               OpDecorate %tangent  Location 3
               OpDecorate %color    Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
    %void_fn = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4float 4
       %bool = OpTypeBool
        %int = OpTypeInt 32 1
      %int_4 = OpConstant %int 4

   %float_0   = OpConstant %float 0.0
   %float_0_5 = OpConstant %float 0.5
   %float_1   = OpConstant %float 1.0
   %float_2   = OpConstant %float 2.0

        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO     = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
        %ubo = OpVariable %_ptr_Uniform_UBO Uniform

%_ptr_Input_v3float  = OpTypePointer Input %v3float
%_ptr_Input_v2float  = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float

   %worldPos = OpVariable %_ptr_Input_v3float Input
     %normal = OpVariable %_ptr_Input_v3float Input
         %uv = OpVariable %_ptr_Input_v2float Input
    %tangent = OpVariable %_ptr_Input_v3float Input
      %color = OpVariable %_ptr_Input_v3float Input
   %outColor = OpVariable %_ptr_Output_v4float Output

       %main = OpFunction %void None %void_fn
      %entry = OpLabel
          %n = OpLoad %v3float %normal
          %N = OpExtInst %v3float %glsl Normalize %n
    %cam_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
        %cam = OpLoad %v4float %cam_ptr
    %cam_xyz = OpVectorShuffle %v3float %cam %cam 0 1 2
         %wp = OpLoad %v3float %worldPos
       %V_un = OpFSub %v3float %cam_xyz %wp
          %V = OpExtInst %v3float %glsl Normalize %V_un
    %NdotV_r = OpDot %float %N %V
      %NdotV = OpExtInst %float %glsl FMax %NdotV_r %float_0
       %cond = OpFOrdLessThan %bool %NdotV %float_0_5

        %g_a = OpFMul %float %float_2 %NdotV

        %tmp = OpFSub %float %NdotV %float_0_5
          %t = OpFMul %float %tmp %float_2
       %rg_b = OpFSub %float %float_1 %t

          %r = OpSelect %float %cond %float_1 %rg_b
          %g = OpSelect %float %cond %g_a     %rg_b
          %b = OpSelect %float %cond %float_0 %t

  %color_out = OpCompositeConstruct %v4float %r %g %b %float_1
               OpStore %outColor %color_out
               OpReturn
               OpFunctionEnd
