; SPIR-V Geometry Shader - Silhouette/Boundary Edge Rendering
; InputTrianglesAdjacency -> OutputTriangleStrip (15 verts max)
; Emits a filled triangle (dimmed colors) plus a thick quad for each of the 3 edges.

               OpCapability Geometry
   %glsl_ext = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 15

               ; --- Decorations ---
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

               ; --- Basic types ---
       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

               ; --- Constants ---
     %uint_6 = OpConstant %uint 6
      %int_0 = OpConstant %int 0
      %int_2 = OpConstant %int 2
      %int_4 = OpConstant %int 4
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
   %float_03 = OpConstant %float 0.3
 %float_0015 = OpConstant %float 0.015
      %white = OpConstantComposite %v3float %float_1 %float_1 %float_1

               ; --- Input: gl_in[6].gl_Position ---
  %gl_PerVertex_in = OpTypeStruct %v4float
        %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
    %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
            %gl_in = OpVariable %ptr_in_arr_pv Input
        %ptr_in_v4 = OpTypePointer Input %v4float

               ; --- Output: gl_out.gl_Position ---
 %gl_PerVertex_out = OpTypeStruct %v4float
       %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
           %gl_out = OpVariable %ptr_out_pv Output
       %ptr_out_v4 = OpTypePointer Output %v4float

               ; --- Input: color[6] ---
         %arr_v3_6 = OpTypeArray %v3float %uint_6
    %ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
            %color = OpVariable %ptr_in_arr_v3 Input
        %ptr_in_v3 = OpTypePointer Input %v3float

               ; --- Output: outColor ---
       %ptr_out_v3 = OpTypePointer Output %v3float
         %outColor = OpVariable %ptr_out_v3 Output

               ; --- Function ---
       %main = OpFunction %void None %fn_void
      %entry = OpLabel

               ; Load the three triangle vertex positions (indices 0,2,4)
       %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
           %p0 = OpLoad %v4float %p0_ptr
       %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
           %p2 = OpLoad %v4float %p2_ptr
       %p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
           %p4 = OpLoad %v4float %p4_ptr

               ; Load the three triangle vertex colors
       %c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
           %c0 = OpLoad %v3float %c0_ptr
       %c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
           %c2 = OpLoad %v3float %c2_ptr
       %c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
           %c4 = OpLoad %v3float %c4_ptr

               ; Dimmed (color * 0.3) for the filled triangle
          %c0d = OpVectorTimesScalar %v3float %c0 %float_03
          %c2d = OpVectorTimesScalar %v3float %c2 %float_03
          %c4d = OpVectorTimesScalar %v3float %c4 %float_03

               ; Output position access chain (reused on each store)
       %out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

               ; ============================================
               ; Emit the filled triangle (3 verts) with dimmed colors
               ; ============================================
               OpStore %out_pos %p0
               OpStore %outColor %c0d
               OpEmitVertex
               OpStore %out_pos %p2
               OpStore %outColor %c2d
               OpEmitVertex
               OpStore %out_pos %p4
               OpStore %outColor %c4d
               OpEmitVertex
               OpEndPrimitive

               ; ============================================
               ; Edge 0: p0 -> p2 (thick quad, 4 verts, white)
               ; ============================================
       %p0xy_0 = OpVectorShuffle %v2float %p0 %p0 0 1
       %p2xy_0 = OpVectorShuffle %v2float %p2 %p2 0 1
        %dir_0 = OpFSub %v2float %p2xy_0 %p0xy_0
       %ndir_0 = OpExtInst %v2float %glsl_ext Normalize %dir_0
         %dx_0 = OpCompositeExtract %float %ndir_0 0
         %dy_0 = OpCompositeExtract %float %ndir_0 1
        %ndy_0 = OpFNegate %float %dy_0
       %perp_0 = OpCompositeConstruct %v2float %ndy_0 %dx_0
       %ofs2_0 = OpVectorTimesScalar %v2float %perp_0 %float_0015
        %ofx_0 = OpCompositeExtract %float %ofs2_0 0
        %ofy_0 = OpCompositeExtract %float %ofs2_0 1
       %ofs4_0 = OpCompositeConstruct %v4float %ofx_0 %ofy_0 %float_0 %float_0
      %nofs4_0 = OpFNegate %v4float %ofs4_0

        %p0m_0 = OpFAdd %v4float %p0 %nofs4_0
        %p0p_0 = OpFAdd %v4float %p0 %ofs4_0
        %p2m_0 = OpFAdd %v4float %p2 %nofs4_0
        %p2p_0 = OpFAdd %v4float %p2 %ofs4_0

               OpStore %out_pos %p0m_0
               OpStore %outColor %white
               OpEmitVertex
               OpStore %out_pos %p0p_0
               OpStore %outColor %white
               OpEmitVertex
               OpStore %out_pos %p2m_0
               OpStore %outColor %white
               OpEmitVertex
               OpStore %out_pos %p2p_0
               OpStore %outColor %white
               OpEmitVertex
               OpEndPrimitive

               ; ============================================
               ; Edge 1: p2 -> p4 (thick quad, 4 verts, white)
               ; ============================================
       %p2xy_1 = OpVectorShuffle %v2float %p2 %p2 0 1
       %p4xy_1 = OpVectorShuffle %v2float %p4 %p4 0 1
        %dir_1 = OpFSub %v2float %p4xy_1 %p2xy_1
       %ndir_1 = OpExtInst %v2float %glsl_ext Normalize %dir_1
         %dx_1 = OpCompositeExtract %float %ndir_1 0
         %dy_1 = OpCompositeExtract %float %ndir_1 1
        %ndy_1 = OpFNegate %float %dy_1
       %perp_1 = OpCompositeConstruct %v2float %ndy_1 %dx_1
       %ofs2_1 = OpVectorTimesScalar %v2float %perp_1 %float_0015
        %ofx_1 = OpCompositeExtract %float %ofs2_1 0
        %ofy_1 = OpCompositeExtract %float %ofs2_1 1
       %ofs4_1 = OpCompositeConstruct %v4float %ofx_1 %ofy_1 %float_0 %float_0
      %nofs4_1 = OpFNegate %v4float %ofs4_1

        %p2m_1 = OpFAdd %v4float %p2 %nofs4_1
        %p2p_1 = OpFAdd %v4float %p2 %ofs4_1
        %p4m_1 = OpFAdd %v4float %p4 %nofs4_1
        %p4p_1 = OpFAdd %v4float %p4 %ofs4_1

               OpStore %out_pos %p2m_1
               OpStore %outColor %white
               OpEmitVertex
               OpStore %out_pos %p2p_1
               OpStore %outColor %white
               OpEmitVertex
               OpStore %out_pos %p4m_1
               OpStore %outColor %white
               OpEmitVertex
               OpStore %out_pos %p4p_1
               OpStore %outColor %white
               OpEmitVertex
               OpEndPrimitive

               ; ============================================
               ; Edge 2: p4 -> p0 (thick quad, 4 verts, white)
               ; ============================================
       %p4xy_2 = OpVectorShuffle %v2float %p4 %p4 0 1
       %p0xy_2 = OpVectorShuffle %v2float %p0 %p0 0 1
        %dir_2 = OpFSub %v2float %p0xy_2 %p4xy_2
       %ndir_2 = OpExtInst %v2float %glsl_ext Normalize %dir_2
         %dx_2 = OpCompositeExtract %float %ndir_2 0
         %dy_2 = OpCompositeExtract %float %ndir_2 1
        %ndy_2 = OpFNegate %float %dy_2
       %perp_2 = OpCompositeConstruct %v2float %ndy_2 %dx_2
       %ofs2_2 = OpVectorTimesScalar %v2float %perp_2 %float_0015
        %ofx_2 = OpCompositeExtract %float %ofs2_2 0
        %ofy_2 = OpCompositeExtract %float %ofs2_2 1
       %ofs4_2 = OpCompositeConstruct %v4float %ofx_2 %ofy_2 %float_0 %float_0
      %nofs4_2 = OpFNegate %v4float %ofs4_2

        %p4m_2 = OpFAdd %v4float %p4 %nofs4_2
        %p4p_2 = OpFAdd %v4float %p4 %ofs4_2
        %p0m_2 = OpFAdd %v4float %p0 %nofs4_2
        %p0p_2 = OpFAdd %v4float %p0 %ofs4_2

               OpStore %out_pos %p4m_2
               OpStore %outColor %white
               OpEmitVertex
               OpStore %out_pos %p4p_2
               OpStore %outColor %white
               OpEmitVertex
               OpStore %out_pos %p0m_2
               OpStore %outColor %white
               OpEmitVertex
               OpStore %out_pos %p0p_2
               OpStore %outColor %white
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
