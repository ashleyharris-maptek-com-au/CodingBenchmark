using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.IO;
using System.Linq;

class Program
{
    static double mu;
    const double EarthRadius = 6371.0;

    static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        var input = Console.In.ReadToEnd().Split(new[] { '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);

        var firstLine = input[0].Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(firstLine[0], ci);
        mu = double.Parse(firstLine[1], ci);

        var startParts = input[1].Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        double[] start = new double[6];
        for (int i = 0; i < 6; i++) start[i] = double.Parse(startParts[i], ci);

        double[][] stations = new double[N][];
        for (int i = 0; i < N; i++)
        {
            var parts = input[2 + i].Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            stations[i] = new double[6];
            for (int j = 0; j < 6; j++) stations[i][j] = double.Parse(parts[j], ci);
        }

        if (N == 0)
        {
            Console.WriteLine("");
            return;
        }
        if (N == 1)
        {
            Console.WriteLine("0");
            return;
        }

        double[] startElem = ComputeElements(start);
        double[][] elem = new double[N][];
        for (int i = 0; i < N; i++) elem[i] = ComputeElements(stations[i]);

        double[] fromStart = new double[N];
        for (int i = 0; i < N; i++) fromStart[i] = TransferCost(start, stations[i], startElem, elem[i]);

        double[,] cost = new double[N, N];
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                if (i != j) cost[i, j] = TransferCost(stations[i], stations[j], elem[i], elem[j]);

        var sw = Stopwatch.StartNew();
        long timeLimitMs = 4L * 60L * 1000L;

        int[] order;

        if (N <= 15)
        {
            order = HeldKarp(N, fromStart, cost);
        }
        else
        {
            order = MultiStartNearestNeighbor(N, fromStart, cost, sw, timeLimitMs / 4);
            TwoOpt(order, fromStart, cost, sw, timeLimitMs * 3 / 4);
            OrOpt(order, fromStart, cost, sw, timeLimitMs);
            TwoOpt(order, fromStart, cost, sw, timeLimitMs);
        }

        Console.WriteLine(string.Join(" ", order));
    }

    static double[] ComputeElements(double[] sv)
    {
        double rx = sv[0], ry = sv[1], rz = sv[2];
        double vx = sv[3], vy = sv[4], vz = sv[5];
        double r = Math.Sqrt(rx * rx + ry * ry + rz * rz);
        double v2 = vx * vx + vy * vy + vz * vz;
        double energy = v2 / 2.0 - mu / r;
        double a = -mu / (2.0 * energy);

        double hx = ry * vz - rz * vy;
        double hy = rz * vx - rx * vz;
        double hz = rx * vy - ry * vx;
        double h = Math.Sqrt(hx * hx + hy * hy + hz * hz);

        double rv = rx * vx + ry * vy + rz * vz;
        double ex = (v2 - mu / r) * rx / mu - rv * vx / mu;
        double ey = (v2 - mu / r) * ry / mu - rv * vy / mu;
        double ez = (v2 - mu / r) * rz / mu - rv * vz / mu;
        double e = Math.Sqrt(ex * ex + ey * ey + ez * ez);

        return new double[] { a, e, hx, hy, hz, h };
    }

    static double TransferCost(double[] a, double[] b, double[] ea, double[] eb)
    {
        double dvx = a[3] - b[3], dvy = a[4] - b[4], dvz = a[5] - b[5];
        double velDiff = Math.Sqrt(dvx * dvx + dvy * dvy + dvz * dvz);

        double daDiff = Math.Abs(ea[0] - eb[0]);

        double hDot = ea[2] * eb[2] + ea[3] * eb[3] + ea[4] * eb[4];
        double hMag = ea[5] * eb[5];
        double planeAngle = 0;
        if (hMag > 1e-9)
        {
            double cosA = hDot / hMag;
            if (cosA > 1.0) cosA = 1.0;
            if (cosA < -1.0) cosA = -1.0;
            planeAngle = Math.Acos(cosA);
        }

        double aAvg = 0.5 * (Math.Abs(ea[0]) + Math.Abs(eb[0]));
        double vMean = (aAvg > 1.0) ? Math.Sqrt(mu / aAvg) : Math.Sqrt(mu / EarthRadius);
        double planeDv = 2.0 * vMean * Math.Sin(planeAngle / 2.0);

        double energyDv = 0;
        if (aAvg > 0 && daDiff > 0)
        {
            energyDv = vMean * (daDiff / (2.0 * aAvg));
        }

        double dx = a[0] - b[0], dy = a[1] - b[1], dz = a[2] - b[2];
        double posDist = Math.Sqrt(dx * dx + dy * dy + dz * dz);

        double eccDiff = Math.Abs(ea[1] - eb[1]);

        return planeDv + energyDv + 0.3 * velDiff + 0.0001 * posDist + vMean * eccDiff * 0.5;
    }

    static int[] NearestNeighborFrom(int N, int startIdx, double[] fromStart, double[,] cost)
    {
        int[] order = new int[N];
        bool[] visited = new bool[N];
        int current;
        int step;

        if (startIdx < 0)
        {
            current = -1;
            step = 0;
        }
        else
        {
            order[0] = startIdx;
            visited[startIdx] = true;
            current = startIdx;
            step = 1;
        }

        for (; step < N; step++)
        {
            int best = -1;
            double bestCost = double.MaxValue;
            for (int i = 0; i < N; i++)
            {
                if (visited[i]) continue;
                double c = (current == -1) ? fromStart[i] : cost[current, i];
                if (c < bestCost)
                {
                    bestCost = c;
                    best = i;
                }
            }
            visited[best] = true;
            order[step] = best;
            current = best;
        }
        return order;
    }

    static int[] MultiStartNearestNeighbor(int N, double[] fromStart, double[,] cost, Stopwatch sw, long timeLimitMs)
    {
        int[] bestOrder = NearestNeighborFrom(N, -1, fromStart, cost);
        double bestCost = TotalCost(bestOrder, fromStart, cost);

        var sortedStarts = Enumerable.Range(0, N).OrderBy(i => fromStart[i]).Take(Math.Min(N, 8)).ToArray();
        foreach (int s in sortedStarts)
        {
            if (sw.ElapsedMilliseconds > timeLimitMs) break;
            int[] candidate = NearestNeighborFrom(N, s, fromStart, cost);
            double c = TotalCost(candidate, fromStart, cost);
            if (c < bestCost)
            {
                bestCost = c;
                bestOrder = candidate;
            }
        }
        return bestOrder;
    }

    static double TotalCost(int[] order, double[] fromStart, double[,] cost)
    {
        double total = fromStart[order[0]];
        for (int i = 1; i < order.Length; i++) total += cost[order[i - 1], order[i]];
        return total;
    }

    static void TwoOpt(int[] order, double[] fromStart, double[,] cost, Stopwatch sw, long timeLimitMs)
    {
        int N = order.Length;
        bool improved = true;
        while (improved && sw.ElapsedMilliseconds < timeLimitMs)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                if (sw.ElapsedMilliseconds > timeLimitMs) break;
                for (int j = i + 1; j < N; j++)
                {
                    double oldEdge1 = (i == 0) ? fromStart[order[i]] : cost[order[i - 1], order[i]];
                    double oldEdge2 = (j == N - 1) ? 0 : cost[order[j], order[j + 1]];
                    double newEdge1 = (i == 0) ? fromStart[order[j]] : cost[order[i - 1], order[j]];
                    double newEdge2 = (j == N - 1) ? 0 : cost[order[i], order[j + 1]];

                    if (newEdge1 + newEdge2 < oldEdge1 + oldEdge2 - 1e-9)
                    {
                        Array.Reverse(order, i, j - i + 1);
                        improved = true;
                    }
                }
            }
        }
    }

    static void OrOpt(int[] order, double[] fromStart, double[,] cost, Stopwatch sw, long timeLimitMs)
    {
        int N = order.Length;
        bool improved = true;
        int passes = 0;
        while (improved && sw.ElapsedMilliseconds < timeLimitMs && passes < 50)
        {
            improved = false;
            passes++;
            for (int segLen = 1; segLen <= 3 && segLen < N; segLen++)
            {
                for (int i = 0; i + segLen <= N; i++)
                {
                    if (sw.ElapsedMilliseconds > timeLimitMs) break;
                    int segStart = i;
                    int segEnd = i + segLen - 1;

                    double edgeIn = (segStart == 0) ? fromStart[order[segStart]] : cost[order[segStart - 1], order[segStart]];
                    double edgeOut = (segEnd == N - 1) ? 0 : cost[order[segEnd], order[segEnd + 1]];
                    double bridgeAfterRemove;
                    if (segStart == 0 && segEnd == N - 1) bridgeAfterRemove = 0;
                    else if (segStart == 0) bridgeAfterRemove = fromStart[order[segEnd + 1]];
                    else if (segEnd == N - 1) bridgeAfterRemove = 0;
                    else bridgeAfterRemove = cost[order[segStart - 1], order[segEnd + 1]];

                    double removeBenefit = edgeIn + edgeOut - bridgeAfterRemove;

                    int bestK = -1;
                    double bestDelta = -1e-9;

                    for (int k = 0; k <= N; k++)
                    {
                        if (k >= segStart && k <= segEnd + 1) continue;

                        int kPrev = k - 1;
                        int kNext = k;

                        if (kPrev >= segStart && kPrev <= segEnd) continue;
                        if (kNext >= segStart && kNext <= segEnd) continue;

                        double insEdgePrev = (kPrev < 0) ? fromStart[order[segStart]] : cost[order[kPrev], order[segStart]];
                        double insEdgeNext = (kNext >= N) ? 0 : cost[order[segEnd], order[kNext]];
                        double oldBridgeAtK;
                        if (kPrev < 0 && kNext >= N) oldBridgeAtK = 0;
                        else if (kPrev < 0) oldBridgeAtK = fromStart[order[kNext]];
                        else if (kNext >= N) oldBridgeAtK = 0;
                        else oldBridgeAtK = cost[order[kPrev], order[kNext]];

                        double insertCost = insEdgePrev + insEdgeNext - oldBridgeAtK;
                        double delta = insertCost - removeBenefit;

                        if (delta < bestDelta)
                        {
                            bestDelta = delta;
                            bestK = k;
                        }
                    }

                    if (bestK >= 0)
                    {
                        int[] segment = new int[segLen];
                        for (int s = 0; s < segLen; s++) segment[s] = order[segStart + s];
                        var list = new List<int>(order);
                        list.RemoveRange(segStart, segLen);
                        int insertAt = (bestK > segEnd) ? bestK - segLen : bestK;
                        list.InsertRange(insertAt, segment);
                        for (int s = 0; s < N; s++) order[s] = list[s];
                        improved = true;
                        break;
                    }
                }
                if (improved) break;
            }
        }
    }

    static int[] HeldKarp(int N, double[] fromStart, double[,] cost)
    {
        int states = 1 << N;
        double[,] dp = new double[states, N];
        int[,] parent = new int[states, N];
        for (int i = 0; i < states; i++)
            for (int j = 0; j < N; j++)
            {
                dp[i, j] = double.MaxValue;
                parent[i, j] = -1;
            }

        for (int i = 0; i < N; i++) dp[1 << i, i] = fromStart[i];

        for (int mask = 1; mask < states; mask++)
        {
            for (int i = 0; i < N; i++)
            {
                if ((mask & (1 << i)) == 0) continue;
                double cur = dp[mask, i];
                if (cur >= double.MaxValue) continue;
                for (int j = 0; j < N; j++)
                {
                    if ((mask & (1 << j)) != 0) continue;
                    int newMask = mask | (1 << j);
                    double newCost = cur + cost[i, j];
                    if (newCost < dp[newMask, j])
                    {
                        dp[newMask, j] = newCost;
                        parent[newMask, j] = i;
                    }
                }
            }
        }

        int fullMask = states - 1;
        int bestEnd = 0;
        double bestVal = double.MaxValue;
        for (int i = 0; i < N; i++)
        {
            if (dp[fullMask, i] < bestVal)
            {
                bestVal = dp[fullMask, i];
                bestEnd = i;
            }
        }

        int[] order = new int[N];
        int cur2 = bestEnd;
        int curMask = fullMask;
        for (int step = N - 1; step >= 0; step--)
        {
            order[step] = cur2;
            int p = parent[curMask, cur2];
            curMask ^= (1 << cur2);
            if (p < 0) break;
            cur2 = p;
        }
        return order;
    }
}
