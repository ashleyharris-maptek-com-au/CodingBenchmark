; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 0
; Schema: 0
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %colorIn %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 7
               OpExecutionMode %main Invocations 1

       %glsl = OpExtInstImport "GLSL.std.450"

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %colorIn Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
       %fnty = OpTypeFunction %void
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
     %uint_2 = OpConstant %uint 2
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
%float_0_015 = OpConstant %float 0.015
 %float_0_05 = OpConstant %float 0.05
 %float_0_75 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%colorIn = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fnty
%entry = OpLabel

; Load p0, p1 (vec4 clip-space positions from gl_in)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load c0, c1
%c0_ptr = OpAccessChain %ptr_in_v3 %colorIn %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %colorIn %int_1
%c1 = OpLoad %v3float %c1_ptr

; dir = p1 - p0
%dir = OpFSub %v4float %p1 %p0
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %glsl Normalize %perp_raw

; mid = p0 + 0.75 * dir
%dir75 = OpVectorTimesScalar %v4float %dir %float_0_75
%mid = OpFAdd %v4float %p0 %dir75

; shaft_off = (perp*0.015, 0, 0)
%shaft_perp = OpVectorTimesScalar %v2float %perp %float_0_015
%sp_x = OpCompositeExtract %float %shaft_perp 0
%sp_y = OpCompositeExtract %float %shaft_perp 1
%shaft_off = OpCompositeConstruct %v4float %sp_x %sp_y %float_0 %float_0

; arrow_off = (perp*0.05, 0, 0)
%arrow_perp = OpVectorTimesScalar %v2float %perp %float_0_05
%ap_x = OpCompositeExtract %float %arrow_perp 0
%ap_y = OpCompositeExtract %float %arrow_perp 1
%arrow_off = OpCompositeConstruct %v4float %ap_x %ap_y %float_0 %float_0

; Shaft vertices (4)
%vert1 = OpFAdd %v4float %p0 %shaft_off
%vert2 = OpFSub %v4float %p0 %shaft_off
%vert3 = OpFAdd %v4float %mid %shaft_off
%vert4 = OpFSub %v4float %mid %shaft_off

; Arrow vertices (vert5, vert6; tip = p1)
%vert5 = OpFAdd %v4float %mid %arrow_off
%vert6 = OpFSub %v4float %mid %arrow_off

; ===== Emit shaft triangle strip (color = c0) =====
%pos_ptr_1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_ptr_1 %vert1
OpStore %outColor %c0
OpEmitVertex

%pos_ptr_2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_ptr_2 %vert2
OpStore %outColor %c0
OpEmitVertex

%pos_ptr_3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_ptr_3 %vert3
OpStore %outColor %c0
OpEmitVertex

%pos_ptr_4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_ptr_4 %vert4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; ===== Emit arrowhead triangle strip (color = c1) =====
%pos_ptr_5 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_ptr_5 %vert5
OpStore %outColor %c1
OpEmitVertex

%pos_ptr_6 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_ptr_6 %vert6
OpStore %outColor %c1
OpEmitVertex

%pos_ptr_7 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_ptr_7 %p1
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
