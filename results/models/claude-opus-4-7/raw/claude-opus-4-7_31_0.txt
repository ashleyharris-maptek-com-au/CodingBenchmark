use std::io::{self, Read, Write, BufWriter};

fn write_yes<W: Write>(out: &mut W, indices: &[usize]) {
    writeln!(out, "YES").unwrap();
    let mut s = String::from("[");
    for (i, &idx) in indices.iter().enumerate() {
        if i > 0 { s.push(' '); }
        s.push_str(&idx.to_string());
    }
    s.push(']');
    writeln!(out, "{}", s).unwrap();
}

fn write_no<W: Write>(out: &mut W) {
    writeln!(out, "NO").unwrap();
}

// Meet-in-the-middle: exact, works for any signed integers, up to n ~= 44.
fn solve_mim(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let half = n / 2;
    let right_n = n - half;

    let mut left: Vec<(i64, u64)> = Vec::with_capacity(1usize << half);
    for mask in 0u64..(1u64 << half) {
        let mut s: i64 = 0;
        let mut m = mask;
        while m != 0 {
            let b = m.trailing_zeros() as usize;
            s = s.wrapping_add(nums[b]);
            m &= m - 1;
        }
        left.push((s, mask));
    }
    left.sort_by_key(|&(v, _)| v);

    for mask in 0u64..(1u64 << right_n) {
        let mut s: i64 = 0;
        let mut m = mask;
        while m != 0 {
            let b = m.trailing_zeros() as usize;
            s = s.wrapping_add(nums[half + b]);
            m &= m - 1;
        }
        let need = target.wrapping_sub(s);
        if let Ok(idx) = left.binary_search_by_key(&need, |&(v, _)| v) {
            let left_mask = left[idx].1;
            let mut indices = Vec::new();
            let mut m = left_mask;
            while m != 0 {
                let b = m.trailing_zeros() as usize;
                indices.push(b);
                m &= m - 1;
            }
            let mut m2 = mask;
            while m2 != 0 {
                let b = m2.trailing_zeros() as usize;
                indices.push(half + b);
                m2 &= m2 - 1;
            }
            indices.sort();
            return Some(indices);
        }
    }
    None
}

// Bitset DP: exact for non-negative numbers with bounded positive target.
// Tracks the first iteration index that achieved each reachable sum
// so we can reconstruct a witness subset.
fn solve_bitset_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let max_t = target as usize;
    let words = max_t / 64 + 1;
    let mut dp: Vec<u64> = vec![0u64; words];
    dp[0] = 1;

    let mut first_set: Vec<i32> = vec![-1; max_t + 1];

    let t_word = max_t / 64;
    let t_bit_mask: u64 = 1u64 << (max_t % 64);

    for (k, &v_signed) in nums.iter().enumerate() {
        if v_signed <= 0 { continue; }
        let v = v_signed as usize;
        if v > max_t { continue; }

        let word_shift = v / 64;
        let bit_shift = v % 64;

        let snap = dp.clone();

        if bit_shift == 0 {
            for i in (word_shift..words).rev() {
                let shifted = snap[i - word_shift];
                if shifted == 0 { continue; }
                let new_bits = shifted & !dp[i];
                if new_bits != 0 {
                    let mut bits = new_bits;
                    while bits != 0 {
                        let b = bits.trailing_zeros() as usize;
                        let s = i * 64 + b;
                        if s <= max_t {
                            first_set[s] = k as i32;
                        }
                        bits &= bits - 1;
                    }
                    dp[i] |= shifted;
                }
            }
        } else {
            for i in (0..words).rev() {
                let mut shifted: u64 = 0;
                if i >= word_shift {
                    shifted |= snap[i - word_shift] << bit_shift;
                    if i > word_shift {
                        shifted |= snap[i - word_shift - 1] >> (64 - bit_shift);
                    }
                }
                if shifted == 0 { continue; }
                let new_bits = shifted & !dp[i];
                if new_bits != 0 {
                    let mut bits = new_bits;
                    while bits != 0 {
                        let b = bits.trailing_zeros() as usize;
                        let s = i * 64 + b;
                        if s <= max_t {
                            first_set[s] = k as i32;
                        }
                        bits &= bits - 1;
                    }
                    dp[i] |= shifted;
                }
            }
        }

        if dp[t_word] & t_bit_mask != 0 {
            break;
        }
    }

    if dp[t_word] & t_bit_mask == 0 {
        return None;
    }

    let mut indices = Vec::new();
    let mut s = max_t;
    while s > 0 {
        let k = first_set[s];
        if k < 0 { return None; }
        indices.push(k as usize);
        let v = nums[k as usize] as usize;
        if v > s { return None; }
        s -= v;
    }
    indices.sort();
    Some(indices)
}

fn try_greedy_perm(nums: &[i64], target: i64, perm: &[usize]) -> Option<Vec<usize>> {
    let mut current: i64 = 0;
    let mut used: Vec<usize> = Vec::new();
    for &i in perm {
        let v = nums[i];
        let new_sum = current.saturating_add(v);
        if new_sum == target {
            used.push(i);
            used.sort();
            return Some(used);
        }
        let dist_now = current.saturating_sub(target).saturating_abs();
        let dist_new = new_sum.saturating_sub(target).saturating_abs();
        if dist_new < dist_now {
            current = new_sum;
            used.push(i);
        }
    }
    None
}

// Approximation: try several greedy orderings plus randomized restarts.
fn solve_greedy(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mut perm: Vec<usize> = (0..n).collect();

    perm.sort_by(|&a, &b| nums[b].abs().cmp(&nums[a].abs()));
    if let Some(r) = try_greedy_perm(nums, target, &perm) { return Some(r); }

    perm.sort_by(|&a, &b| nums[a].abs().cmp(&nums[b].abs()));
    if let Some(r) = try_greedy_perm(nums, target, &perm) { return Some(r); }

    perm.sort_by(|&a, &b| nums[a].cmp(&nums[b]));
    if let Some(r) = try_greedy_perm(nums, target, &perm) { return Some(r); }

    perm.sort_by(|&a, &b| nums[b].cmp(&nums[a]));
    if let Some(r) = try_greedy_perm(nums, target, &perm) { return Some(r); }

    let attempts = if n > 1_000_000 { 5 }
        else if n > 100_000 { 20 }
        else if n > 1_000 { 100 }
        else { 500 };

    let mut rng: u64 = 0x9E3779B97F4A7C15u64
        .wrapping_add(n as u64)
        .wrapping_mul((target.wrapping_abs() as u64) | 1);
    if rng == 0 { rng = 0xDEADBEEFCAFEBABE; }

    for _ in 0..attempts {
        for i in (1..n).rev() {
            rng ^= rng << 13;
            rng ^= rng >> 7;
            rng ^= rng << 17;
            let j = (rng as usize) % (i + 1);
            perm.swap(i, j);
        }
        if let Some(r) = try_greedy_perm(nums, target, &perm) { return Some(r); }
    }

    None
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut iter = input.split_ascii_whitespace();
    let n: usize = iter.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    let target: i64 = iter.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    let mut nums: Vec<i64> = Vec::with_capacity(n);
    for _ in 0..n {
        match iter.next() {
            Some(s) => {
                if let Ok(v) = s.parse::<i64>() {
                    nums.push(v);
                }
            }
            None => break,
        }
    }

    if target == 0 {
        write_yes(&mut out, &[]);
        return;
    }

    if nums.is_empty() {
        write_no(&mut out);
        return;
    }

    for (i, &v) in nums.iter().enumerate() {
        if v == target {
            write_yes(&mut out, &[i]);
            return;
        }
    }

    let total: i64 = nums.iter().fold(0i64, |a, &b| a.saturating_add(b));
    if total == target {
        let indices: Vec<usize> = (0..nums.len()).collect();
        write_yes(&mut out, &indices);
        return;
    }

    let n_actual = nums.len();
    let can_bitset = target > 0
        && nums.iter().all(|&v| v >= 0)
        && (target as u64) <= 200_000_000;

    let result = if n_actual <= 44 {
        solve_mim(&nums, target)
    } else if can_bitset {
        solve_bitset_dp(&nums, target)
    } else {
        solve_greedy(&nums, target)
    };

    match result {
        Some(indices) => write_yes(&mut out, &indices),
        None => write_no(&mut out),
    }
}
