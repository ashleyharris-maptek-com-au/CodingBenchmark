using System;
using System.IO;
using System.Collections.Generic;
using System.Text;
using System.Diagnostics;

class Program
{
    static int xCount, yCount, zCount;
    static int[] topZ;
    static byte[] topByte;

    static int Idx(int x, int y) => x * yCount + y;

    static void Main()
    {
        var sw = Stopwatch.StartNew();
        var stdin = Console.OpenStandardInput();

        // Read header line (text, space-delimited, newline terminated)
        var hb = new StringBuilder();
        int b;
        while ((b = stdin.ReadByte()) != -1 && b != '\n')
        {
            if (b != '\r') hb.Append((char)b);
        }
        var parts = hb.ToString().Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 3)
        {
            Console.WriteLine(0);
            return;
        }
        xCount = int.Parse(parts[0]);
        yCount = int.Parse(parts[1]);
        zCount = int.Parse(parts[2]);
        // BlockSize values (parts[3..5]) are accepted but not needed for the scoring logic.

        if (xCount <= 0 || yCount <= 0 || zCount <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        long cellCount = (long)xCount * yCount;
        // Memory guard for huge worlds: bail with 0 ops if surface arrays would be too big.
        if (cellCount > 200_000_000L)
        {
            Console.WriteLine(0);
            return;
        }

        topZ = new int[(int)cellCount];
        topByte = new byte[(int)cellCount];
        for (int i = 0; i < cellCount; i++) topZ[i] = -1;

        var bs = new BufferedStream(stdin, 1 << 20);
        byte[] buf = new byte[zCount];

        // Stream column-by-column to extract surface info without storing the full grid.
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int got = 0;
                while (got < zCount)
                {
                    int n = bs.Read(buf, got, zCount - got);
                    if (n <= 0) break;
                    got += n;
                }
                if (got < zCount) break;

                int tZ = -1;
                byte tB = 0;
                // Scan from top down for first non-empty voxel
                for (int z = zCount - 1; z >= 0; z--)
                {
                    if ((buf[z] & 15) > 0)
                    {
                        tZ = z;
                        tB = buf[z];
                        break;
                    }
                }
                int idx = Idx(x, y);
                topZ[idx] = tZ;
                topByte[idx] = tB;
            }
        }

        // Adaptive strategy:
        // - Small/medium worlds: full scan of all surface cells.
        // - Large worlds: iterate by Manhattan distance from origin (closer/more profitable first),
        //   bounded by a time budget.
        var candidates = new List<(int x, int y, long profit)>();
        TimeSpan searchBudget = TimeSpan.FromSeconds(180);
        bool largeScale = cellCount > 4_000_000L;

        if (!largeScale)
        {
            for (int x = 0; x < xCount; x++)
            {
                if (sw.Elapsed > searchBudget) break;
                for (int y = 0; y < yCount; y++)
                {
                    EvaluateCandidate(x, y, candidates);
                }
            }
        }
        else
        {
            int maxDist = (xCount - 1) + (yCount - 1);
            for (int dist = 0; dist <= maxDist; dist++)
            {
                if (sw.Elapsed > searchBudget) break;
                int xStart = Math.Max(0, dist - (yCount - 1));
                int xEnd = Math.Min(dist, xCount - 1);
                for (int x = xStart; x <= xEnd; x++)
                {
                    int y = dist - x;
                    if (y < 0 || y >= yCount) continue;
                    EvaluateCandidate(x, y, candidates);
                }
            }
        }

        // Highest-profit operations first
        candidates.Sort((a, b) => b.profit.CompareTo(a.profit));

        var ops = new List<string>();
        var sbOut = new StringBuilder();
        TimeSpan execBudget = TimeSpan.FromSeconds(260);

        foreach (var cand in candidates)
        {
            if (sw.Elapsed > execBudget) break;

            int cx = cand.x, cy = cand.y;
            int i = Idx(cx, cy);
            if (topZ[i] < 0) continue;
            byte d = topByte[i];
            int mass = d & 15;
            int value = d >> 4;
            if (value == 0 || mass == 0) continue;
            int z = topZ[i];

            var pr = TryPathFull(cx, cy);
            long cost = pr.cost;
            var steps = pr.steps;
            if (cost < 0) continue;

            long revenue = (long)mass * value * 1000L;
            long totalCost = cost * mass;
            if (revenue <= totalCost) continue;

            sbOut.Clear();
            sbOut.Append(steps.Count);
            sbOut.Append(' ').Append(cx).Append(' ').Append(cy).Append(' ').Append(z);
            for (int s = 0; s < steps.Count; s++)
            {
                sbOut.Append(' ').Append(steps[s].Item1).Append(' ').Append(steps[s].Item2);
            }
            sbOut.Append(" 0 0");
            ops.Add(sbOut.ToString());

            // After digging the top voxel, mark this column as unknown.
            // Subsequent paths through this column become invalid (safer than guessing).
            topZ[i] = -1;
            topByte[i] = 0;
        }

        var outStream = Console.OpenStandardOutput();
        using (var w = new StreamWriter(outStream))
        {
            w.WriteLine(ops.Count);
            for (int i = 0; i < ops.Count; i++)
            {
                w.WriteLine(ops[i]);
            }
        }
    }

    static void EvaluateCandidate(int x, int y, List<(int x, int y, long profit)> candidates)
    {
        int i = Idx(x, y);
        if (topZ[i] < 0) return;
        byte d = topByte[i];
        int mass = d & 15;
        int value = d >> 4;
        if (value == 0 || mass == 0) return;

        long revenue = (long)mass * value * 1000L;
        if (revenue <= 0) return;

        long cost = TryPathCost(x, y);
        if (cost < 0) return;
        long totalCost = cost * mass;
        long profit = revenue - totalCost;
        if (profit > 0)
        {
            candidates.Add((x, y, profit));
        }
    }

    // L-path: traverse X to 0 first, then Y to 0. Returns -1 if any slope is invalid.
    // Cost is per unit mass (caller multiplies by tonnage).
    static long TryPathCost(int sx, int sy)
    {
        long cost = 0;
        int cx = sx, cy = sy;
        int ch = topZ[Idx(cx, cy)];
        if (ch < 0) return -1;
        while (cx != 0)
        {
            int nx = cx > 0 ? cx - 1 : cx + 1;
            int nh = topZ[Idx(nx, cy)];
            if (nh < 0) return -1;
            int diff = nh - ch;
            if (diff > 1 || diff < -1) return -1;
            if (diff > 0) cost += 5;          // uphill (10% slope)
            else if (diff == 0) cost += 1;    // flat
            // downhill: free
            cx = nx;
            ch = nh;
        }
        while (cy != 0)
        {
            int ny = cy > 0 ? cy - 1 : cy + 1;
            int nh = topZ[Idx(cx, ny)];
            if (nh < 0) return -1;
            int diff = nh - ch;
            if (diff > 1 || diff < -1) return -1;
            if (diff > 0) cost += 5;
            else if (diff == 0) cost += 1;
            cy = ny;
            ch = nh;
        }
        return cost;
    }

    static (long cost, List<(int, int)> steps) TryPathFull(int sx, int sy)
    {
        var steps = new List<(int, int)>();
        long cost = 0;
        int cx = sx, cy = sy;
        int ch = topZ[Idx(cx, cy)];
        if (ch < 0) return (-1, steps);
        while (cx != 0)
        {
            int nx = cx > 0 ? cx - 1 : cx + 1;
            int nh = topZ[Idx(nx, cy)];
            if (nh < 0) return (-1, steps);
            int diff = nh - ch;
            if (diff > 1 || diff < -1) return (-1, steps);
            if (diff > 0) cost += 5;
            else if (diff == 0) cost += 1;
            cx = nx;
            ch = nh;
            if (cx != 0 || cy != 0) steps.Add((cx, cy));
        }
        while (cy != 0)
        {
            int ny = cy > 0 ? cy - 1 : cy + 1;
            int nh = topZ[Idx(cx, ny)];
            if (nh < 0) return (-1, steps);
            int diff = nh - ch;
            if (diff > 1 || diff < -1) return (-1, steps);
            if (diff > 0) cost += 5;
            else if (diff == 0) cost += 1;
            cy = ny;
            ch = nh;
            if (cx != 0 || cy != 0) steps.Add((cx, cy));
        }
        return (cost, steps);
    }
}
