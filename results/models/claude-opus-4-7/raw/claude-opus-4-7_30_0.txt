import sys
import time
import numpy as np
import warnings
warnings.filterwarnings('ignore')


def main():
    start_time = time.time()
    TIME_LIMIT = 270.0  # leave ~30s buffer for I/O and output

    # ---- Fast input ----
    raw = sys.stdin.buffer.read()
    try:
        arr = np.fromstring(raw, sep=' ', dtype=np.int32)
    except Exception:
        arr = np.array(raw.split(), dtype=np.int32)

    if arr.size < 2:
        sys.stdout.buffer.write(b"SAT\n\n")
        return

    N = int(arr[0])
    M = int(arr[1])

    # ---- Trivial cases ----
    if N == 0:
        sys.stdout.buffer.write(b"SAT\n\n")
        return

    if M == 0:
        sys.stdout.buffer.write(b"SAT\n")
        out = np.empty(2 * N - 1 if N > 0 else 0, dtype=np.uint8)
        if N >= 1:
            out_arr = np.empty(2 * N - 1, dtype=np.uint8)
            out_arr[0::2] = ord('1')
            if N > 1:
                out_arr[1::2] = ord(' ')
            sys.stdout.buffer.write(out_arr.tobytes())
        sys.stdout.buffer.write(b"\n")
        return

    # ---- Parse clauses ----
    # lits shape (M, 3); each entry is signed (positive = var must be True,
    # negative = var must be False).
    lits = arr[2:2 + 3 * M].reshape(M, 3)
    del arr

    abs_lits = np.abs(lits).astype(np.int32)         # variable indices (1..N)
    pos = (lits > 0).astype(np.int8)                  # 1 for positive literal, 0 for negative

    # ---- Polarity-based initial assignment ----
    # For each variable, sum +1 for each positive occurrence, -1 for each negative.
    # Initialize to True if the sum is >= 0, otherwise False.
    flat = lits.ravel()
    flat_abs = np.abs(flat).astype(np.int64)
    sign_w = np.where(flat > 0, np.int64(1), np.int64(-1))
    score = np.bincount(flat_abs, weights=sign_w, minlength=N + 1)
    del flat, flat_abs, sign_w, lits

    assignment = (score >= 0).astype(np.int8)
    assignment[0] = 0  # index 0 unused (1-indexed variables)
    del score

    rng = np.random.default_rng(42)

    best_unsat = M + 1
    best_asgn = assignment.copy()
    stall = 0

    # ---- Parallel WalkSAT loop ----
    # Each iteration:
    #   1. Evaluate all clauses.
    #   2. For each unsatisfied clause, pick a random variable in it and mark
    #      for flipping. Variables marked an odd number of times get flipped.
    # This is a heavily-vectorized variant of Schoening's random walk.
    iter_count = 0
    MAX_ITERS = 5000
    while iter_count < MAX_ITERS and (time.time() - start_time) < TIME_LIMIT:
        iter_count += 1

        vals = assignment[abs_lits]               # (M, 3) values (0/1)
        lit_sat = (vals == pos)                   # (M, 3) literal-satisfied
        clause_sat = lit_sat.any(axis=1)          # (M,)
        sat_count = int(clause_sat.sum())
        n_unsat = M - sat_count

        if n_unsat < best_unsat:
            best_unsat = n_unsat
            best_asgn = assignment.copy()
            stall = 0
        else:
            stall += 1

        if n_unsat == 0:
            break

        # If we're stuck, restart from the best assignment with a random perturbation.
        if stall > 30:
            assignment = best_asgn.copy()
            mask = rng.random(N + 1) < 0.10
            mask[0] = False
            assignment ^= mask.astype(np.int8)
            stall = 0
            continue

        # Indices of unsatisfied clauses
        unsat_idx = np.where(~clause_sat)[0]
        nu = unsat_idx.shape[0]

        # For each unsatisfied clause, pick a random one of its 3 variables.
        col = rng.integers(0, 3, size=nu, dtype=np.int64)
        flip_vars = abs_lits[unsat_idx, col]

        # If a variable is selected an odd number of times, flip it.
        cnt = np.bincount(flip_vars, minlength=N + 1)
        assignment ^= (cnt & 1).astype(np.int8)

    # ---- Pick whichever assignment is better ----
    vals = assignment[abs_lits]
    clause_sat = (vals == pos).any(axis=1)
    cur_unsat = M - int(clause_sat.sum())

    if cur_unsat > 0:
        vals_b = best_asgn[abs_lits]
        cs_b = (vals_b == pos).any(axis=1)
        best_check = M - int(cs_b.sum())
        if best_check < cur_unsat:
            assignment = best_asgn
            cur_unsat = best_check

    # ---- Fallback: if there are still unsatisfied clauses, do a final greedy
    # serial pass over remaining unsatisfied clauses. ----
    if cur_unsat > 0 and (time.time() - start_time) < TIME_LIMIT - 5:
        vals = assignment[abs_lits]
        clause_sat = (vals == pos).any(axis=1)
        unsat_idx = np.where(~clause_sat)[0]
        for repeat in range(5):
            if (time.time() - start_time) >= TIME_LIMIT - 2:
                break
            if unsat_idx.size == 0:
                break
            # For each unsat clause set its first literal to its required polarity.
            first_var = abs_lits[unsat_idx, 0]
            first_pos = pos[unsat_idx, 0]
            assignment[first_var] = first_pos
            assignment[0] = 0
            vals = assignment[abs_lits]
            clause_sat = (vals == pos).any(axis=1)
            unsat_idx = np.where(~clause_sat)[0]

    # ---- Fast output ----
    sys.stdout.buffer.write(b"SAT\n")
    vals_arr = assignment[1:].astype(np.uint8) + np.uint8(ord('0'))
    if N == 1:
        sys.stdout.buffer.write(vals_arr.tobytes())
    else:
        out = np.empty(2 * N - 1, dtype=np.uint8)
        out[0::2] = vals_arr
        out[1::2] = ord(' ')
        sys.stdout.buffer.write(out.tobytes())
    sys.stdout.buffer.write(b"\n")


main()
