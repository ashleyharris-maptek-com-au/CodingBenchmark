// Minimum Edge Cut solver
// Adaptive: bridges -> Stoer-Wagner -> Max-flow (Dinic) -> min-degree heuristic
// Targets MSVC cl 19.50, /std:c++20

#include <cstdio>
#include <cstdlib>
#include <cstdint>
#include <cstring>
#include <vector>
#include <algorithm>
#include <queue>
#include <climits>
#include <numeric>
#include <utility>

using std::vector;
using std::pair;
using std::min;
using std::max;
using std::swap;

typedef int64_t i64;

namespace io {
    constexpr int BUFSZ = 1 << 20;
    char ibuf[BUFSZ];
    int ipos = 0, ilen = 0;
    inline int gc() {
        if (ipos == ilen) {
            ilen = (int)fread(ibuf, 1, BUFSZ, stdin);
            ipos = 0;
            if (ilen == 0) return -1;
        }
        return ibuf[ipos++];
    }
    inline bool rd(i64& x) {
        int c;
        do { c = gc(); if (c == -1) return false; } while (c <= ' ');
        bool neg = false;
        if (c == '-') { neg = true; c = gc(); }
        if (c < '0' || c > '9') return false;
        x = 0;
        while (c >= '0' && c <= '9') {
            x = x * 10 + (c - '0');
            c = gc();
        }
        if (neg) x = -x;
        return true;
    }
    char obuf[BUFSZ];
    int opos = 0;
    inline void flush_out() {
        if (opos > 0) fwrite(obuf, 1, opos, stdout);
        opos = 0;
    }
    inline void pc(char c) {
        if (opos == BUFSZ) flush_out();
        obuf[opos++] = c;
    }
    inline void wi(i64 x) {
        if (x < 0) { pc('-'); x = -x; }
        char tmp[24]; int n = 0;
        if (x == 0) tmp[n++] = '0';
        else while (x > 0) { tmp[n++] = (char)('0' + (int)(x % 10)); x /= 10; }
        while (n--) pc(tmp[n]);
    }
}
using io::rd; using io::pc; using io::wi; using io::flush_out;

int N;
i64 M;
vector<int> EU, EV;
vector<i64> deg;
vector<i64> head_off;
vector<int> adj_v;
vector<int> adj_e;

bool build_csr() {
    try {
        head_off.assign((size_t)N + 1, 0);
        for (i64 i = 0; i < M; i++) {
            head_off[EU[(size_t)i] + 1]++;
            head_off[EV[(size_t)i] + 1]++;
        }
        for (int i = 0; i < N; i++) head_off[i + 1] += head_off[i];
        i64 total = head_off[N];
        adj_v.assign((size_t)total, 0);
        adj_e.assign((size_t)total, 0);
        vector<i64> cur(N, 0);
        for (i64 i = 0; i < M; i++) {
            int u = EU[(size_t)i], v = EV[(size_t)i];
            i64 pu = head_off[u] + cur[u];
            i64 pv = head_off[v] + cur[v];
            adj_v[(size_t)pu] = v;
            adj_e[(size_t)pu] = (int)i;
            cur[u]++;
            adj_v[(size_t)pv] = u;
            adj_e[(size_t)pv] = (int)i;
            cur[v]++;
        }
        return true;
    } catch (...) {
        head_off.clear(); head_off.shrink_to_fit();
        adj_v.clear(); adj_v.shrink_to_fit();
        adj_e.clear(); adj_e.shrink_to_fit();
        return false;
    }
}

vector<int> bridge_eids;
void find_bridges() {
    vector<int> disc(N, -1), low(N, 0);
    vector<int> stk_node;
    vector<int> stk_pedge;
    vector<i64> stk_idx;
    stk_node.reserve(1024);
    stk_pedge.reserve(1024);
    stk_idx.reserve(1024);
    int timer = 0;
    for (int s = 0; s < N; s++) {
        if (disc[s] != -1) continue;
        stk_node.push_back(s);
        stk_pedge.push_back(-1);
        stk_idx.push_back(head_off[s]);
        disc[s] = low[s] = timer++;
        while (!stk_node.empty()) {
            int u = stk_node.back();
            int pe = stk_pedge.back();
            i64 cur_idx = stk_idx.back();
            if (cur_idx < head_off[u + 1]) {
                stk_idx.back() = cur_idx + 1;
                int v = adj_v[(size_t)cur_idx];
                int eid = adj_e[(size_t)cur_idx];
                if (eid == pe) continue;
                if (disc[v] == -1) {
                    disc[v] = low[v] = timer++;
                    stk_node.push_back(v);
                    stk_pedge.push_back(eid);
                    stk_idx.push_back(head_off[v]);
                } else {
                    if (disc[v] < low[u]) low[u] = disc[v];
                }
            } else {
                int cl = low[u];
                stk_node.pop_back();
                stk_pedge.pop_back();
                stk_idx.pop_back();
                if (!stk_node.empty()) {
                    int pu = stk_node.back();
                    if (cl < low[pu]) low[pu] = cl;
                    if (cl > disc[pu]) {
                        bridge_eids.push_back(pe);
                    }
                }
            }
        }
    }
}

// Stoer-Wagner with dense weighted matrix
pair<i64, vector<int>> stoer_wagner_dense() {
    vector<vector<i64>> w(N, vector<i64>(N, 0));
    for (i64 i = 0; i < M; i++) {
        int u = EU[(size_t)i], v = EV[(size_t)i];
        if (u == v) continue;
        w[u][v]++;
        w[v][u]++;
    }
    vector<int> alive(N);
    std::iota(alive.begin(), alive.end(), 0);
    vector<vector<int>> merged(N);
    for (int i = 0; i < N; i++) merged[i].push_back(i);

    i64 best = LLONG_MAX;
    vector<int> best_set;

    while ((int)alive.size() > 1) {
        int A = alive[0];
        vector<i64> weight(N, 0);
        vector<char> in_A(N, 0);
        in_A[A] = 1;
        int prev = -1, last = A;
        for (int v_ : alive) if (v_ != A) weight[v_] = w[A][v_];
        int phase_count = (int)alive.size();
        for (int it = 1; it < phase_count; it++) {
            int sel = -1;
            i64 sel_w = -1;
            for (int v_ : alive) {
                if (!in_A[v_] && weight[v_] > sel_w) {
                    sel_w = weight[v_];
                    sel = v_;
                }
            }
            if (sel == -1) break;
            in_A[sel] = 1;
            prev = last;
            last = sel;
            for (int v_ : alive) if (!in_A[v_]) weight[v_] += w[sel][v_];
        }
        i64 cut_w = weight[last];
        if (cut_w < best) {
            best = cut_w;
            best_set = merged[last];
        }
        if (prev >= 0) {
            for (int v_ : alive) {
                if (v_ != last && v_ != prev) {
                    w[prev][v_] += w[last][v_];
                    w[v_][prev] += w[v_][last];
                }
            }
            merged[prev].insert(merged[prev].end(),
                                merged[last].begin(), merged[last].end());
        }
        auto it_a = std::find(alive.begin(), alive.end(), last);
        alive.erase(it_a);
    }
    return {best, best_set};
}

// Dinic's max-flow for undirected unit capacities (re-usable via reset_caps)
struct Dinic {
    struct E { int to; int rev; int cap; int orig_cap; };
    vector<vector<E>> g;
    vector<int> level, iter;
    int n;

    void init(int n_) { n = n_; g.assign(n, {}); }
    void add_undirected(int u, int v, int c) {
        g[u].push_back({v, (int)g[v].size(), c, c});
        g[v].push_back({u, (int)g[u].size() - 1, c, c});
    }
    void reset_caps() {
        for (auto& gv : g) for (auto& e : gv) e.cap = e.orig_cap;
    }
    bool bfs(int s, int t) {
        level.assign(n, -1);
        level[s] = 0;
        std::queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : g[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }
    int dfs(int v, int t, int f) {
        if (v == t) return f;
        for (; iter[v] < (int)g[v].size(); iter[v]++) {
            auto& e = g[v][iter[v]];
            if (e.cap > 0 && level[v] < level[e.to]) {
                int d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    g[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }
    int max_flow(int s, int t, int limit) {
        int flow = 0;
        while (flow < limit && bfs(s, t)) {
            iter.assign(n, 0);
            int f;
            while (flow < limit && (f = dfs(s, t, limit - flow)) > 0) flow += f;
        }
        return flow;
    }
    vector<char> reachable_from(int s) {
        vector<char> vis(n, 0);
        vis[s] = 1;
        std::queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : g[v]) {
                if (e.cap > 0 && !vis[e.to]) {
                    vis[e.to] = 1;
                    q.push(e.to);
                }
            }
        }
        return vis;
    }
};

void output_cut(const vector<char>& in_S) {
    vector<i64> cut_edges;
    cut_edges.reserve(64);
    for (i64 i = 0; i < M; i++) {
        if (in_S[EU[(size_t)i]] != in_S[EV[(size_t)i]]) cut_edges.push_back(i);
    }
    wi((i64)cut_edges.size()); pc('\n');
    for (i64 e : cut_edges) {
        wi(EU[(size_t)e]); pc(' '); wi(EV[(size_t)e]); pc('\n');
    }
}

void output_min_degree_cut(int mdv) {
    i64 cnt = 0;
    for (i64 i = 0; i < M; i++) {
        if (EU[(size_t)i] == mdv || EV[(size_t)i] == mdv) cnt++;
    }
    wi(cnt); pc('\n');
    for (i64 i = 0; i < M; i++) {
        if (EU[(size_t)i] == mdv || EV[(size_t)i] == mdv) {
            wi(EU[(size_t)i]); pc(' '); wi(EV[(size_t)i]); pc('\n');
        }
    }
}

bool already_disconnected() {
    vector<char> vis(N, 0);
    std::queue<int> q;
    q.push(0);
    vis[0] = 1;
    int seen = 1;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (i64 i = head_off[u]; i < head_off[u + 1]; i++) {
            int v = adj_v[(size_t)i];
            if (!vis[v]) { vis[v] = 1; seen++; q.push(v); }
        }
    }
    return seen < N;
}

int main() {
    i64 tmp;
    if (!rd(tmp)) {
        wi(0); pc('\n');
        flush_out();
        return 0;
    }
    N = (int)tmp;
    if (!rd(tmp)) {
        wi(0); pc('\n');
        flush_out();
        return 0;
    }
    M = tmp;

    if (N <= 1) {
        wi(0); pc('\n');
        flush_out();
        return 0;
    }

    EU.resize((size_t)M);
    EV.resize((size_t)M);
    deg.assign(N, 0);
    i64 valid_edges = 0;
    for (i64 i = 0; i < M; i++) {
        i64 a, b;
        if (!rd(a) || !rd(b)) break;
        if (a < 0 || b < 0 || a >= N || b >= N) continue;
        if (a == b) continue;
        EU[(size_t)valid_edges] = (int)a;
        EV[(size_t)valid_edges] = (int)b;
        deg[(int)a]++;
        deg[(int)b]++;
        valid_edges++;
    }
    M = valid_edges;
    EU.resize((size_t)M);
    EV.resize((size_t)M);

    if (M == 0) {
        wi(0); pc('\n');
        flush_out();
        return 0;
    }

    int mdv = 0;
    for (int i = 1; i < N; i++) if (deg[i] < deg[mdv]) mdv = i;
    i64 mind = deg[mdv];

    // For huge graphs, skip CSR -> fall back to min-degree cut
    const i64 LARGE_M = 200000000LL; // ~3.2 GB CSR upper bound
    bool has_csr = false;
    if (M <= LARGE_M) {
        has_csr = build_csr();
    }

    if (!has_csr) {
        output_min_degree_cut(mdv);
        flush_out();
        return 0;
    }

    if (already_disconnected()) {
        wi(0); pc('\n');
        flush_out();
        return 0;
    }

    find_bridges();
    if (!bridge_eids.empty()) {
        int eid = bridge_eids[0];
        wi(1); pc('\n');
        wi(EU[(size_t)eid]); pc(' '); wi(EV[(size_t)eid]); pc('\n');
        flush_out();
        return 0;
    }

    // No bridges -> min cut >= 2

    // Tier 1: Small graphs -> exact Stoer-Wagner (O(N^3))
    const int DENSE_SW_N = 500;
    if (N <= DENSE_SW_N) {
        auto result = stoer_wagner_dense();
        vector<char> in_S(N, 0);
        for (int v : result.second) in_S[v] = 1;
        bool has_in = false, has_out = false;
        for (int i = 0; i < N; i++) {
            if (in_S[i]) has_in = true; else has_out = true;
        }
        if (has_in && has_out) {
            output_cut(in_S);
            flush_out();
            return 0;
        }
    }

    // Tier 2: Medium graphs -> max-flow with budget
    const int MF_N = 12000;
    const i64 MF_M = 300000;
    if (N <= MF_N && M <= MF_M) {
        Dinic dinic;
        dinic.init(N);
        for (i64 i = 0; i < M; i++) {
            dinic.add_undirected(EU[(size_t)i], EV[(size_t)i], 1);
        }

        int best_cut = (int)mind;
        vector<char> best_S(N, 0);
        best_S[mdv] = 1;

        int s = mdv;

        vector<int> order(N);
        std::iota(order.begin(), order.end(), 0);
        std::sort(order.begin(), order.end(),
                  [](int a, int b) { return deg[a] < deg[b]; });

        int budget = min(N - 1, 40);
        for (int t : order) {
            if (budget == 0) break;
            if (t == s) continue;
            budget--;
            dinic.reset_caps();
            int f = dinic.max_flow(s, t, best_cut);
            if (f < best_cut) {
                best_cut = f;
                best_S = dinic.reachable_from(s);
            }
            if (best_cut <= 2) break;
        }

        output_cut(best_S);
        flush_out();
        return 0;
    }

    // Tier 3: Large graphs -> min-degree heuristic
    output_min_degree_cut(mdv);
    flush_out();
    return 0;
}
