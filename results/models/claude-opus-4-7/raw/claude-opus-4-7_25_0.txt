using System;
using System.Collections.Generic;
using System.Globalization;
using System.Text;

class Body
{
    public string Name;
    public double Mass;
    public double[] Pos = new double[3];
    public double[] Vel = new double[3];
}

class Program
{
    const double G = 6.67430e-11;
    const double DAY = 86400.0;
    static int N;
    static double[] GM;

    static void Main()
    {
        var sw = System.Diagnostics.Stopwatch.StartNew();
        string all = Console.In.ReadToEnd();
        var tokens = all.Split(new[] { ' ', '\t', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int numBodies = int.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        double warningDays = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        double dvBudget = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        double t0Unix = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);

        var bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = tokens[idx++];
            b.Mass = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
            for (int j = 0; j < 3; j++) b.Pos[j] = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
            for (int j = 0; j < 3; j++) b.Vel[j] = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
            bodies.Add(b);
        }
        N = bodies.Count;

        int earthIdx = bodies.FindIndex(b => b.Name == "Earth");
        int astIdx = bodies.FindIndex(b => b.Name == "Asteroid");
        int sunIdx = bodies.FindIndex(b => b.Name == "Sun");
        if (sunIdx < 0)
        {
            sunIdx = 0;
            for (int i = 1; i < N; i++) if (bodies[i].Mass > bodies[sunIdx].Mass) sunIdx = i;
        }
        if (earthIdx < 0) earthIdx = 0;
        if (astIdx < 0) astIdx = N - 1;

        GM = new double[N];
        for (int i = 0; i < N; i++) GM[i] = G * bodies[i].Mass;
        double GMsun = GM[sunIdx];

        double[][] pos = new double[N][];
        double[][] vel = new double[N][];
        for (int i = 0; i < N; i++)
        {
            pos[i] = (double[])bodies[i].Pos.Clone();
            vel[i] = (double[])bodies[i].Vel.Clone();
        }

        // Step size scales with mission length
        double dt;
        if (warningDays < 3 * 365) dt = 3600.0;
        else if (warningDays < 10 * 365) dt = 7200.0;
        else if (warningDays < 25 * 365) dt = 14400.0;
        else dt = 21600.0;

        int simDays = (int)Math.Ceiling(warningDays * 1.5) + 30;
        int stepsPerDay = Math.Max(1, (int)Math.Round(DAY / dt));
        double actualDt = DAY / stepsPerDay;

        double[][][] posHist = new double[simDays + 1][][];
        double[][][] velHist = new double[simDays + 1][][];
        for (int d = 0; d <= simDays; d++)
        {
            posHist[d] = new double[N][];
            velHist[d] = new double[N][];
            for (int i = 0; i < N; i++)
            {
                posHist[d][i] = new double[3];
                velHist[d][i] = new double[3];
            }
        }
        for (int i = 0; i < N; i++)
        {
            Array.Copy(pos[i], posHist[0][i], 3);
            Array.Copy(vel[i], velHist[0][i], 3);
        }

        double[][] acc = new double[N][];
        for (int i = 0; i < N; i++) acc[i] = new double[3];

        int actualSimDays = simDays;
        for (int d = 0; d < simDays; d++)
        {
            for (int s = 0; s < stepsPerDay; s++)
            {
                KdkStep(pos, vel, acc, actualDt);
            }
            for (int i = 0; i < N; i++)
            {
                Array.Copy(pos[i], posHist[d + 1][i], 3);
                Array.Copy(vel[i], velHist[d + 1][i], 3);
            }
            if (sw.Elapsed.TotalSeconds > 120)
            {
                actualSimDays = d + 1;
                break;
            }
        }

        // Find approximate Earth-asteroid closest approach day
        double minDist = double.MaxValue;
        int impactDay = actualSimDays;
        for (int d = 1; d <= actualSimDays; d++)
        {
            double dx = posHist[d][astIdx][0] - posHist[d][earthIdx][0];
            double dy = posHist[d][astIdx][1] - posHist[d][earthIdx][1];
            double dz = posHist[d][astIdx][2] - posHist[d][earthIdx][2];
            double dist = Math.Sqrt(dx * dx + dy * dy + dz * dz);
            if (dist < minDist)
            {
                minDist = dist;
                impactDay = d;
            }
        }

        // Search for best Lambert intercept
        double bestCost = double.MaxValue;
        int bestL = -1, bestI = -1;
        double[] bestDvL = null;

        int maxLaunch = Math.Min((int)warningDays, actualSimDays - 2);
        int maxIntercept = Math.Min(impactDay - 1, actualSimDays - 1);
        if (maxIntercept < 2) maxIntercept = Math.Min(actualSimDays - 1, (int)warningDays);
        if (maxLaunch < 1) maxLaunch = 1;

        int minTransfer = Math.Max(5, maxLaunch / 200);

        int searchPoints = 80;
        int launchStep = Math.Max(1, maxLaunch / searchPoints);
        int interceptStep = Math.Max(1, maxIntercept / searchPoints);

        for (int tL = 0; tL <= maxLaunch; tL += launchStep)
        {
            for (int tI = tL + minTransfer; tI <= maxIntercept; tI += interceptStep)
            {
                double[] earthP = posHist[tL][earthIdx];
                double[] earthV = velHist[tL][earthIdx];
                double[] astP = posHist[tI][astIdx];
                double[] sunPL = posHist[tL][sunIdx];
                double[] sunVL = velHist[tL][sunIdx];
                double[] sunPI = posHist[tI][sunIdx];

                double[] r1 = new double[3];
                double[] r2 = new double[3];
                for (int j = 0; j < 3; j++)
                {
                    r1[j] = earthP[j] - sunPL[j];
                    r2[j] = astP[j] - sunPI[j];
                }

                double tof = (tI - tL) * DAY;

                for (int dir = 0; dir < 2; dir++)
                {
                    double[] v1S, v2S;
                    if (LambertSolve(r1, r2, tof, GMsun, dir == 0, out v1S, out v2S))
                    {
                        double dvLmag = 0;
                        double[] dvL = new double[3];
                        for (int j = 0; j < 3; j++)
                        {
                            dvL[j] = v1S[j] + sunVL[j] - earthV[j];
                            dvLmag += dvL[j] * dvL[j];
                        }
                        dvLmag = Math.Sqrt(dvLmag);

                        if (dvLmag > dvBudget) continue;
                        if (double.IsNaN(dvLmag) || double.IsInfinity(dvLmag)) continue;

                        double timeBonus = (double)tI / Math.Max(1, maxIntercept);
                        double cost = dvLmag + 1.0 * timeBonus;

                        if (cost < bestCost)
                        {
                            bestCost = cost;
                            bestL = tL;
                            bestI = tI;
                            bestDvL = dvL;
                        }
                    }
                }
            }
        }

        // Refined search around the best candidate
        if (bestL >= 0 && (launchStep > 1 || interceptStep > 1))
        {
            int rL0 = Math.Max(0, bestL - launchStep);
            int rL1 = Math.Min(maxLaunch, bestL + launchStep);
            int rI0 = Math.Max(bestL + minTransfer, bestI - interceptStep);
            int rI1 = Math.Min(maxIntercept, bestI + interceptStep);

            for (int tL = rL0; tL <= rL1; tL += 1)
            {
                for (int tI = Math.Max(tL + minTransfer, rI0); tI <= rI1; tI += 1)
                {
                    double[] earthP = posHist[tL][earthIdx];
                    double[] earthV = velHist[tL][earthIdx];
                    double[] astP = posHist[tI][astIdx];
                    double[] sunPL = posHist[tL][sunIdx];
                    double[] sunVL = velHist[tL][sunIdx];
                    double[] sunPI = posHist[tI][sunIdx];

                    double[] r1 = new double[3];
                    double[] r2 = new double[3];
                    for (int j = 0; j < 3; j++)
                    {
                        r1[j] = earthP[j] - sunPL[j];
                        r2[j] = astP[j] - sunPI[j];
                    }
                    double tof = (tI - tL) * DAY;
                    for (int dir = 0; dir < 2; dir++)
                    {
                        double[] v1S, v2S;
                        if (LambertSolve(r1, r2, tof, GMsun, dir == 0, out v1S, out v2S))
                        {
                            double dvLmag = 0;
                            double[] dvL = new double[3];
                            for (int j = 0; j < 3; j++)
                            {
                                dvL[j] = v1S[j] + sunVL[j] - earthV[j];
                                dvLmag += dvL[j] * dvL[j];
                            }
                            dvLmag = Math.Sqrt(dvLmag);
                            if (dvLmag > dvBudget) continue;
                            if (double.IsNaN(dvLmag) || double.IsInfinity(dvLmag)) continue;

                            double timeBonus = (double)tI / Math.Max(1, maxIntercept);
                            double cost = dvLmag + 1.0 * timeBonus;
                            if (cost < bestCost)
                            {
                                bestCost = cost;
                                bestL = tL;
                                bestI = tI;
                                bestDvL = dvL;
                            }
                        }
                    }
                }
            }
        }

        if (bestL == -1)
        {
            // Fallback: full-budget direct chase
            int tL = 0;
            int tI = Math.Min(maxIntercept, Math.Max(60, maxLaunch / 2));
            if (tI <= tL) tI = tL + 1;
            double[] earthP = posHist[tL][earthIdx];
            double[] astP = posHist[tI][astIdx];
            double[] dirvec = new double[3];
            double dmag = 0;
            for (int j = 0; j < 3; j++)
            {
                dirvec[j] = astP[j] - earthP[j];
                dmag += dirvec[j] * dirvec[j];
            }
            dmag = Math.Sqrt(dmag);
            if (dmag < 1) dmag = 1;
            bestDvL = new double[3];
            double scale = Math.Max(0.0, dvBudget * 0.99);
            for (int j = 0; j < 3; j++) bestDvL[j] = dirvec[j] / dmag * scale;
            bestL = tL;
            bestI = tI;
        }

        var ci = CultureInfo.InvariantCulture;
        var sb = new StringBuilder();
        sb.AppendLine("2");
        sb.Append(bestL.ToString(ci)); sb.Append(' ');
        sb.Append(bestDvL[0].ToString("R", ci)); sb.Append(' ');
        sb.Append(bestDvL[1].ToString("R", ci)); sb.Append(' ');
        sb.AppendLine(bestDvL[2].ToString("R", ci));
        sb.Append(bestI.ToString(ci)); sb.Append(" 0 0 0");
        sb.AppendLine();
        Console.Write(sb.ToString());
    }

    static void ComputeAccel(double[][] pos, double[][] acc)
    {
        for (int i = 0; i < N; i++)
        {
            acc[i][0] = 0; acc[i][1] = 0; acc[i][2] = 0;
        }
        for (int i = 0; i < N; i++)
        {
            double[] pi = pos[i];
            double[] ai = acc[i];
            double gmi = GM[i];
            for (int k = i + 1; k < N; k++)
            {
                double[] pk = pos[k];
                double dx = pk[0] - pi[0];
                double dy = pk[1] - pi[1];
                double dz = pk[2] - pi[2];
                double r2 = dx * dx + dy * dy + dz * dz;
                if (r2 < 1.0) continue;
                double r = Math.Sqrt(r2);
                double inv_r3 = 1.0 / (r * r2);
                double gmk = GM[k];
                ai[0] += gmk * dx * inv_r3;
                ai[1] += gmk * dy * inv_r3;
                ai[2] += gmk * dz * inv_r3;
                double[] ak = acc[k];
                ak[0] -= gmi * dx * inv_r3;
                ak[1] -= gmi * dy * inv_r3;
                ak[2] -= gmi * dz * inv_r3;
            }
        }
    }

    static void KdkStep(double[][] pos, double[][] vel, double[][] acc, double dt)
    {
        ComputeAccel(pos, acc);
        double h = 0.5 * dt;
        for (int i = 0; i < N; i++)
        {
            vel[i][0] += h * acc[i][0];
            vel[i][1] += h * acc[i][1];
            vel[i][2] += h * acc[i][2];
        }
        for (int i = 0; i < N; i++)
        {
            pos[i][0] += dt * vel[i][0];
            pos[i][1] += dt * vel[i][1];
            pos[i][2] += dt * vel[i][2];
        }
        ComputeAccel(pos, acc);
        for (int i = 0; i < N; i++)
        {
            vel[i][0] += h * acc[i][0];
            vel[i][1] += h * acc[i][1];
            vel[i][2] += h * acc[i][2];
        }
    }

    // Universal variables Lambert solver (single revolution)
    static bool LambertSolve(double[] r1, double[] r2, double tof, double mu, bool prograde,
                              out double[] v1, out double[] v2)
    {
        v1 = new double[3];
        v2 = new double[3];

        double r1m = Math.Sqrt(r1[0] * r1[0] + r1[1] * r1[1] + r1[2] * r1[2]);
        double r2m = Math.Sqrt(r2[0] * r2[0] + r2[1] * r2[1] + r2[2] * r2[2]);
        if (r1m < 1 || r2m < 1) return false;

        double cos_dnu = (r1[0] * r2[0] + r1[1] * r2[1] + r1[2] * r2[2]) / (r1m * r2m);
        if (cos_dnu > 1) cos_dnu = 1;
        if (cos_dnu < -1) cos_dnu = -1;

        double crossZ = r1[0] * r2[1] - r1[1] * r2[0];
        int dm = prograde ? (crossZ >= 0 ? 1 : -1) : (crossZ >= 0 ? -1 : 1);

        double A = dm * Math.Sqrt(r1m * r2m * (1 + cos_dnu));
        if (Math.Abs(A) < 1e-9) return false;

        double zMin = -50.0;
        double zMax = 4.0 * Math.PI * Math.PI - 0.001;
        double z = 0.0;

        double y = 0, t = 0;
        bool converged = false;

        for (int iter = 0; iter < 200; iter++)
        {
            double Cv = Cz(z);
            double Sv = Sz(z);

            if (Cv <= 1e-15)
            {
                zMin = z;
                z = 0.5 * (zMin + zMax);
                continue;
            }

            y = r1m + r2m + A * (z * Sv - 1.0) / Math.Sqrt(Cv);

            if (y < 0)
            {
                zMin = z;
                z = 0.5 * (zMin + zMax);
                if (Math.Abs(zMax - zMin) < 1e-12) return false;
                continue;
            }

            double chi = Math.Sqrt(y / Cv);
            t = (chi * chi * chi * Sv + A * Math.Sqrt(y)) / Math.Sqrt(mu);

            if (double.IsNaN(t) || double.IsInfinity(t)) return false;

            if (Math.Abs(t - tof) < 1e-5 * Math.Max(tof, 1.0))
            {
                converged = true;
                break;
            }

            if (t < tof) zMin = z;
            else zMax = z;

            double zNew = 0.5 * (zMin + zMax);
            if (Math.Abs(zNew - z) < 1e-13) { converged = true; break; }
            z = zNew;
        }

        if (!converged && Math.Abs(t - tof) > 1e-3 * Math.Max(tof, 1.0)) return false;
        if (y < 0 || double.IsNaN(y)) return false;

        double f = 1.0 - y / r1m;
        double g = A * Math.Sqrt(y / mu);
        double gdot = 1.0 - y / r2m;

        if (Math.Abs(g) < 1e-9) return false;

        for (int j = 0; j < 3; j++)
        {
            v1[j] = (r2[j] - f * r1[j]) / g;
            v2[j] = (gdot * r2[j] - r1[j]) / g;
        }

        for (int j = 0; j < 3; j++)
        {
            if (double.IsNaN(v1[j]) || double.IsInfinity(v1[j])) return false;
            if (double.IsNaN(v2[j]) || double.IsInfinity(v2[j])) return false;
        }

        return true;
    }

    static double Cz(double z)
    {
        if (z > 1e-6)
        {
            double sz = Math.Sqrt(z);
            return (1.0 - Math.Cos(sz)) / z;
        }
        else if (z < -1e-6)
        {
            double sz = Math.Sqrt(-z);
            return (Math.Cosh(sz) - 1.0) / (-z);
        }
        else
        {
            return 0.5 - z / 24.0 + z * z / 720.0;
        }
    }

    static double Sz(double z)
    {
        if (z > 1e-6)
        {
            double sz = Math.Sqrt(z);
            return (sz - Math.Sin(sz)) / (sz * sz * sz);
        }
        else if (z < -1e-6)
        {
            double sz = Math.Sqrt(-z);
            return (Math.Sinh(sz) - sz) / (sz * sz * sz);
        }
        else
        {
            return 1.0 / 6.0 - z / 120.0 + z * z / 5040.0;
        }
    }
}
