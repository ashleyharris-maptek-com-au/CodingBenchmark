using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;
using System.Diagnostics;
using System.Numerics;
using System.Text;

class Program
{
    static int n;
    static int[][] adj;
    static volatile int bestSize = 0;
    static int[] bestClique = new int[0];
    static readonly object bestLock = new object();
    static Stopwatch sw = new Stopwatch();
    static long timeLimitMs = 270_000; // 4 min 30 sec, leave headroom

    [ThreadStatic]
    static int[] _indexOfBuf;

    static void Main()
    {
        sw.Start();

        try
        {
            ReadInput();
        }
        catch (Exception)
        {
            WriteOutput();
            return;
        }

        if (n <= 0)
        {
            WriteOutput();
            return;
        }

        // Initial best = a single vertex
        bestClique = new[] { 0 };
        bestSize = 1;

        try { GreedyClique(); } catch { }

        try
        {
            int[] order = DegeneracyOrder();
            int[] orderPos = new int[n];
            for (int i = 0; i < n; i++) orderPos[order[i]] = i;

            var laterCounts = new int[n];
            for (int v = 0; v < n; v++)
            {
                int c = 0;
                int op = orderPos[v];
                var av = adj[v];
                for (int j = 0; j < av.Length; j++)
                    if (orderPos[av[j]] > op) c++;
                laterCounts[v] = c;
            }

            var toProcessList = new List<int>(n);
            for (int v = 0; v < n; v++)
            {
                if (laterCounts[v] + 1 > bestSize) toProcessList.Add(v);
            }
            // Process larger neighborhoods first - usually contain bigger cliques
            toProcessList.Sort((a, b) => laterCounts[b].CompareTo(laterCounts[a]));
            var toProcess = toProcessList.ToArray();

            int parallelism = Math.Max(1, Environment.ProcessorCount);
            var opts = new ParallelOptions { MaxDegreeOfParallelism = parallelism };

            Parallel.ForEach(toProcess, opts, v =>
            {
                if (sw.ElapsedMilliseconds > timeLimitMs) return;
                int lc = laterCounts[v];
                if (lc + 1 <= bestSize) return;

                var laterNbrs = new int[lc];
                int idx = 0;
                int op = orderPos[v];
                var av = adj[v];
                for (int j = 0; j < av.Length; j++)
                    if (orderPos[av[j]] > op) laterNbrs[idx++] = av[j];

                try { ProcessVertex(v, laterNbrs); } catch { }
            });
        }
        catch { }

        WriteOutput();
    }

    static void WriteOutput()
    {
        try
        {
            var stdout = Console.OpenStandardOutput();
            var output = new StringBuilder();
            output.Append(bestSize);
            output.Append('\n');
            if (bestClique != null && bestClique.Length > 0)
            {
                var sorted = (int[])bestClique.Clone();
                Array.Sort(sorted);
                for (int i = 0; i < sorted.Length; i++)
                {
                    if (i > 0) output.Append(' ');
                    output.Append(sorted[i]);
                }
            }
            output.Append('\n');
            var bytes = Encoding.ASCII.GetBytes(output.ToString());
            stdout.Write(bytes, 0, bytes.Length);
            stdout.Flush();
        }
        catch { }
    }

    static void ReadInput()
    {
        var stdin = Console.OpenStandardInput();
        var buf = new byte[1 << 20];
        int bufLen = 0, bufPos = 0;
        bool eof = false;

        int ReadByte()
        {
            if (bufPos >= bufLen)
            {
                if (eof) return -1;
                bufLen = stdin.Read(buf, 0, buf.Length);
                bufPos = 0;
                if (bufLen == 0) { eof = true; return -1; }
            }
            return buf[bufPos++];
        }

        int ReadInt()
        {
            int c;
            do { c = ReadByte(); if (c < 0) return -1; } while (!(c >= '0' && c <= '9') && c != '-');
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
                if (c < 0) break;
            }
            return val * sign;
        }

        n = ReadInt();
        int m = ReadInt();

        if (n <= 0) { n = 0; adj = new int[0][]; return; }
        if (m < 0) m = 0;

        int[] edgeU = null, edgeV = null;
        try
        {
            edgeU = new int[m];
            edgeV = new int[m];
        }
        catch (OutOfMemoryException)
        {
            var listU = new List<int>();
            var listV = new List<int>();
            for (int e = 0; e < m; e++)
            {
                int u = ReadInt();
                int v = ReadInt();
                if (u < 0 || u >= n || v < 0 || v >= n || u == v) continue;
                listU.Add(u); listV.Add(v);
            }
            edgeU = listU.ToArray();
            edgeV = listV.ToArray();
            int ml = edgeU.Length;
            BuildAdjacency(edgeU, edgeV, ml);
            return;
        }

        var deg = new int[n];
        int actualEdges = 0;

        for (int e = 0; e < m; e++)
        {
            int u = ReadInt();
            int v = ReadInt();
            if (u < 0 || u >= n || v < 0 || v >= n || u == v) continue;
            edgeU[actualEdges] = u;
            edgeV[actualEdges] = v;
            deg[u]++;
            deg[v]++;
            actualEdges++;
        }

        adj = new int[n][];
        var emptyArr = new int[0];
        for (int i = 0; i < n; i++)
        {
            adj[i] = deg[i] > 0 ? new int[deg[i]] : emptyArr;
        }
        var cur = new int[n];
        for (int e = 0; e < actualEdges; e++)
        {
            int u = edgeU[e], v = edgeV[e];
            adj[u][cur[u]++] = v;
            adj[v][cur[v]++] = u;
        }

        edgeU = null;
        edgeV = null;
        cur = null;
        deg = null;
        GC.Collect();

        for (int i = 0; i < n; i++)
        {
            var row = adj[i];
            if (row.Length <= 1) continue;
            Array.Sort(row);
            int w = 0;
            for (int j = 0; j < row.Length; j++)
            {
                if (j == 0 || row[j] != row[j - 1]) row[w++] = row[j];
            }
            if (w < row.Length)
            {
                var newRow = new int[w];
                Array.Copy(row, 0, newRow, 0, w);
                adj[i] = newRow;
            }
        }
    }

    static void BuildAdjacency(int[] edgeU, int[] edgeV, int m)
    {
        var deg = new int[n];
        for (int e = 0; e < m; e++) { deg[edgeU[e]]++; deg[edgeV[e]]++; }
        adj = new int[n][];
        var emptyArr = new int[0];
        for (int i = 0; i < n; i++) adj[i] = deg[i] > 0 ? new int[deg[i]] : emptyArr;
        var cur = new int[n];
        for (int e = 0; e < m; e++)
        {
            int u = edgeU[e], v = edgeV[e];
            adj[u][cur[u]++] = v;
            adj[v][cur[v]++] = u;
        }
        for (int i = 0; i < n; i++)
        {
            var row = adj[i];
            if (row.Length <= 1) continue;
            Array.Sort(row);
            int w = 0;
            for (int j = 0; j < row.Length; j++)
                if (j == 0 || row[j] != row[j - 1]) row[w++] = row[j];
            if (w < row.Length)
            {
                var newRow = new int[w];
                Array.Copy(row, 0, newRow, 0, w);
                adj[i] = newRow;
            }
        }
    }

    static int[] DegeneracyOrder()
    {
        var deg = new int[n];
        int maxDeg = 0;
        for (int i = 0; i < n; i++)
        {
            deg[i] = adj[i].Length;
            if (deg[i] > maxDeg) maxDeg = deg[i];
        }

        var bucketHead = new int[maxDeg + 2];
        for (int i = 0; i <= maxDeg + 1; i++) bucketHead[i] = -1;
        var prev = new int[n];
        var next = new int[n];

        for (int v = 0; v < n; v++)
        {
            int d = deg[v];
            next[v] = bucketHead[d];
            prev[v] = -1;
            if (bucketHead[d] != -1) prev[bucketHead[d]] = v;
            bucketHead[d] = v;
        }

        var order = new int[n];
        var removed = new bool[n];
        int idx = 0;
        int curMin = 0;

        while (idx < n)
        {
            while (curMin <= maxDeg && bucketHead[curMin] == -1) curMin++;
            if (curMin > maxDeg) break;

            int v = bucketHead[curMin];
            bucketHead[curMin] = next[v];
            if (next[v] != -1) prev[next[v]] = -1;

            order[idx++] = v;
            removed[v] = true;

            var av = adj[v];
            for (int k = 0; k < av.Length; k++)
            {
                int u = av[k];
                if (removed[u]) continue;
                int d = deg[u];
                int pu = prev[u], nu = next[u];
                if (pu == -1) bucketHead[d] = nu;
                else next[pu] = nu;
                if (nu != -1) prev[nu] = pu;

                int nd = d - 1;
                deg[u] = nd;
                next[u] = bucketHead[nd];
                prev[u] = -1;
                if (bucketHead[nd] != -1) prev[bucketHead[nd]] = u;
                bucketHead[nd] = u;

                if (nd < curMin) curMin = nd;
            }
        }

        return order;
    }

    static void GreedyClique()
    {
        int trials = Math.Min(n, 50);
        var topVertices = Enumerable.Range(0, n)
            .OrderByDescending(v => adj[v].Length)
            .Take(trials)
            .ToArray();

        long greedyDeadline = sw.ElapsedMilliseconds + 30_000;

        Parallel.ForEach(topVertices, start =>
        {
            if (sw.ElapsedMilliseconds > greedyDeadline) return;

            var clique = new List<int> { start };
            var candSet = new HashSet<int>(adj[start]);

            while (candSet.Count > 0)
            {
                if (sw.ElapsedMilliseconds > greedyDeadline) break;
                int bestV = -1, bestDeg = -1;
                foreach (int c in candSet)
                {
                    int d = 0;
                    var ac = adj[c];
                    for (int j = 0; j < ac.Length; j++)
                        if (candSet.Contains(ac[j])) d++;
                    if (d > bestDeg) { bestDeg = d; bestV = c; }
                }
                if (bestV == -1) break;
                clique.Add(bestV);
                candSet.IntersectWith(adj[bestV]);
            }

            UpdateBest(clique.ToArray());
        });
    }

    static void ProcessVertex(int rootV, int[] candidates)
    {
        int k = candidates.Length;
        if (k == 0) return;
        if (1 + k <= bestSize) return;
        if (k > 200_000) return;

        if (_indexOfBuf == null || _indexOfBuf.Length < n)
        {
            _indexOfBuf = new int[n];
            for (int i = 0; i < _indexOfBuf.Length; i++) _indexOfBuf[i] = -1;
        }
        var indexOf = _indexOfBuf;
        for (int i = 0; i < k; i++) indexOf[candidates[i]] = i;

        int words = (k + 63) >> 6;
        ulong[][] localAdj;
        try
        {
            localAdj = new ulong[k][];
            for (int i = 0; i < k; i++)
            {
                var row = new ulong[words];
                localAdj[i] = row;
                var ai = adj[candidates[i]];
                for (int j = 0; j < ai.Length; j++)
                {
                    int u = ai[j];
                    int idx = indexOf[u];
                    if (idx >= 0)
                    {
                        row[idx >> 6] |= 1UL << (idx & 63);
                    }
                }
            }
        }
        finally
        {
            for (int i = 0; i < k; i++) indexOf[candidates[i]] = -1;
        }

        var P = new ulong[words];
        for (int i = 0; i < k; i++) P[i >> 6] |= 1UL << (i & 63);

        var clique = new List<int>(64);
        clique.Add(-1); // placeholder for rootV
        MCQ(localAdj, P, words, clique, candidates, rootV);
    }

    static void MCQ(ulong[][] localAdj, ulong[] P, int words, List<int> clique, int[] candidates, int rootV)
    {
        if (sw.ElapsedMilliseconds > timeLimitMs) return;

        int pCount = 0;
        for (int i = 0; i < words; i++) pCount += BitOperations.PopCount(P[i]);

        if (pCount == 0)
        {
            if (clique.Count > bestSize)
            {
                var result = new int[clique.Count];
                result[0] = rootV;
                for (int i = 1; i < clique.Count; i++) result[i] = candidates[clique[i]];
                UpdateBest(result);
            }
            return;
        }

        if (clique.Count + pCount <= bestSize) return;

        var order = new int[pCount];
        var colors = new int[pCount];
        GreedyColor(localAdj, P, words, order, colors, pCount);

        for (int i = pCount - 1; i >= 0; i--)
        {
            if (clique.Count + colors[i] <= bestSize) return;

            int vi = order[i];
            clique.Add(vi);

            var newP = new ulong[words];
            var adjVi = localAdj[vi];
            for (int w = 0; w < words; w++)
                newP[w] = P[w] & adjVi[w];

            MCQ(localAdj, newP, words, clique, candidates, rootV);

            clique.RemoveAt(clique.Count - 1);

            P[vi >> 6] &= ~(1UL << (vi & 63));
        }
    }

    static void GreedyColor(ulong[][] adjLocal, ulong[] Pin, int words, int[] order, int[] colors, int pCount)
    {
        int numColors = 0;
        int filled = 0;
        var remaining = (ulong[])Pin.Clone();
        var available = new ulong[words];

        while (filled < pCount)
        {
            numColors++;
            Array.Copy(remaining, available, words);

            while (true)
            {
                int v = -1;
                for (int w = 0; w < words; w++)
                {
                    if (available[w] != 0)
                    {
                        v = (w << 6) + BitOperations.TrailingZeroCount(available[w]);
                        break;
                    }
                }
                if (v == -1) break;

                order[filled] = v;
                colors[filled] = numColors;
                filled++;

                ulong mask = 1UL << (v & 63);
                remaining[v >> 6] &= ~mask;
                available[v >> 6] &= ~mask;

                var adjV = adjLocal[v];
                for (int w = 0; w < words; w++)
                {
                    available[w] &= ~adjV[w];
                }
            }
        }
    }

    static void UpdateBest(int[] result)
    {
        if (result.Length <= bestSize) return;
        lock (bestLock)
        {
            if (result.Length > bestSize)
            {
                bestSize = result.Length;
                bestClique = result;
            }
        }
    }
}
