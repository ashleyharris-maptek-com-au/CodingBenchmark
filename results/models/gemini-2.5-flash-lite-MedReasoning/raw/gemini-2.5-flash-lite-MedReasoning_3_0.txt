use layout::backends::svg::SVGWriter;
use layout::core::base::Orientation;
use layout::core::geometry::Point;
use layout::core::style::{LineStyle, StyleAttr};
use layout::core::utils::save_to_file;
use layout::std_shapes::shapes::{Arrow, Element, Rect};
use layout::topo::layout::VisualGraph;
use layout::topo::placer::Placer;

fn main() {
    let mut vg = VisualGraph::new(Orientation::TopToBottom);

    let node_a = Element::create(
        Rect::new().with_label("A"),
        Point::new(0.0, 0.0),
        StyleAttr::default(),
    );
    let node_b = Element::create(
        Rect::new().with_label("B"),
        Point::new(0.0, 0.0),
        StyleAttr::default(),
    );
    let node_c = Element::create(
        Rect::new().with_label("C"),
        Point::new(0.0, 0.0),
        StyleAttr::default(),
    );
    let node_d = Element::create(
        Rect::new().with_label("D"),
        Point::new(0.0, 0.0),
        StyleAttr::default(),
    );

    let handle_a = vg.add_node(node_a);
    let handle_b = vg.add_node(node_b);
    let handle_c = vg.add_node(node_c);
    let handle_d = vg.add_node(node_d);

    let arrow = Arrow::new(LineStyle::Solid);
    vg.add_edge(arrow.clone(), handle_a, handle_c);
    vg.add_edge(arrow.clone(), handle_b, handle_d);
    vg.add_edge(arrow.clone(), handle_a, handle_d);
    vg.add_edge(arrow.clone(), handle_b, handle_c);

    let mut placer = Placer::new();
    vg.do_it(false, &mut placer, &mut false);

    let mut svg = SVGWriter::new();
    vg.render(&mut svg);
    
    save_to_file("output.svg", &svg.finalize()).expect("Failed to save SVG");
    println!("Graph layout saved to output.svg");
}
