#ifndef M_PI_2
#define M_PI_2 1.57079632679489661923 // pi/2
#endif

// Define a small epsilon for floating point comparisons
const double EPSILON = 1e-9;

// Structure to represent a 3D point
struct Point {
    double x, y, z;

    Point(double _x = 0.0, double _y = 0.0, double _z = 0.0) : x(_x), y(_y), z(_z) {}

    // Vector operations
    Point operator+(const Point& other) const { return Point(x + other.x, y + other.y, z + other.z); }
    Point operator-(const Point& other) const { return Point(x - other.x, y - other.y, z - other.y); }
    Point operator*(double scalar) const { return Point(x * scalar, y * scalar, z * scalar); }
    Point operator/(double scalar) const { return Point(x / scalar, y / scalar, z / scalar); }

    double dot(const Point& other) const { return x * other.x + y * other.y + z * other.z; }
    Point cross(const Point& other) const {
        return Point(y * other.z - z * other.y,
                     z * other.x - x * other.z,
                     x * other.y - y * other.x);
    }
    double length() const { return std::sqrt(x * x + y * y + z * z); }
    Point normalize() const {
        double len = length();
        return (len > EPSILON) ? Point(x / len, y / len, z / len) : Point();
    }
};

// 3x3 Matrix for rotations
struct Matrix3x3 {
    double m[3][3];

    Matrix3x3() {
        for (int i = 0; i < 3; ++i) {
            for (int j = 0; j < 3; ++j) {
                m[i][j] = 0.0;
            }
        }
    }

    static Matrix3x3 identity() {
        Matrix3x3 mat;
        mat.m[0][0] = mat.m[1][1] = mat.m[2][2] = 1.0;
        return mat;
    }

    // Matrix multiplication
    Matrix3x3 operator*(const Matrix3x3& other) const {
        Matrix3x3 result;
        for (int i = 0; i < 3; ++i) {
            for (int j = 0; j < 3; ++j) {
                for (int k = 0; k < 3; ++k) {
                    result.m[i][j] += m[i][k] * other.m[k][j];
                }
            }
        }
        return result;
    }

    // Apply matrix to a point (vector)
    Point operator*(const Point& p) const {
        return Point(m[0][0] * p.x + m[0][1] * p.y + m[0][2] * p.z,
                     m[1][0] * p.x + m[1][1] * p.y + m[1][2] * p.z,
                     m[2][0] * p.x + m[2][1] * p.y + m[2][2] * p.z);
    }

    // Factory methods for rotation matrices
    static Matrix3x3 rotation_x(double angle) {
        Matrix3x3 mat = identity();
        double c = std::cos(angle);
        double s = std::sin(angle);
        mat.m[1][1] = c; mat.m[1][2] = -s;
        mat.m[2][1] = s; mat.m[2][2] = c;
        return mat;
    }

    static Matrix3x3 rotation_y(double angle) {
        Matrix3x3 mat = identity();
        double c = std::cos(angle);
        double s = std::sin(angle);
        mat.m[0][0] = c; mat.m[0][2] = s;
        mat.m[2][0] = -s; mat.m[2][2] = c;
        return mat;
    }

    static Matrix3x3 rotation_z(double angle) {
        Matrix3x3 mat = identity();
        double c = std::cos(angle);
        double s = std::sin(angle);
        mat.m[0][0] = c; mat.m[0][1] = -s;
        mat.m[1][0] = s; mat.m[1][1] = c;
        return mat;
    }
};

// Represents a regular tetrahedron
struct Tetrahedron {
    Point center;
    // Euler angles (radians) for rotation around X, Y, Z axes
    double rx, ry, rz;
    double edge_length;

    Tetrahedron(Point c, double _rx, double _ry, double _rz, double e_len)
        : center(c), rx(_rx), ry(_ry), rz(_rz), edge_length(e_len) {}

    std::vector<Point> get_vertices() const {
        std::vector<Point> vertices;
        vertices.reserve(4);

        // Canonical vertices for a regular tetrahedron with edge length 1, centered at origin
        // Vertex coordinates are +/- (1 / (2*sqrt(2))) for each axis
        double a = 1.0 / (2.0 * std::sqrt(2.0));
        std::vector<Point> canonical_vertices = {
            {a, a, a},
            {a, -a, -a},
            {-a, a, -a},
            {-a, -a, a}
        };

        // Combine rotations: Z then Y then X (standard Euler angles)
        Matrix3x3 rotation_matrix = Matrix3x3::rotation_z(rz) * Matrix3x3::rotation_y(ry) * Matrix3x3::rotation_x(rx);

        for (const auto& canon_v : canonical_vertices) {
            // Apply rotation, then scale, then translate
            Point rotated_v = rotation_matrix * canon_v;
            Point scaled_v = rotated_v * edge_length; // Scale by actual edge length
            vertices.push_back(center + scaled_v);
        }
        return vertices;
    }

    // Get the normalized face normals of the tetrahedron
    std::vector<Point> get_face_normals() const {
        std::vector<Point> normals;
        std::vector<Point> verts = get_vertices();
        if (verts.size() != 4) return normals;

        // Face 0: V0, V1, V2 (normal pointing "out" if CCW)
        // Ensure consistent winding for normals
        Point normal0 = (verts[1] - verts[0]).cross(verts[2] - verts[0]).normalize();
        normals.push_back(normal0);
        
        Point normal1 = (verts[0] - verts[1]).cross(verts[3] - verts[1]).normalize();
        normals.push_back(normal1);

        Point normal2 = (verts[0] - verts[2]).cross(verts[1] - verts[2]).normalize();
        normals.push_back(normal2);

        Point normal3 = (verts[1] - verts[3]).cross(verts[2] - verts[3]).normalize();
        normals.push_back(normal3);

        return normals;
    }

    // Get the edge vectors of the tetrahedron
    std::vector<Point> get_edge_vectors() const {
        std::vector<Point> edges;
        std::vector<Point> verts = get_vertices();
        if (verts.size() != 4) return edges;

        edges.push_back(verts[1] - verts[0]);
        edges.push_back(verts[2] - verts[0]);
        edges.push_back(verts[3] - verts[0]);
        edges.push_back(verts[2] - verts[1]);
        edges.push_back(verts[3] - verts[1]);
        edges.push_back(verts[3] - verts[2]);

        return edges;
    }
};

// Helper function to project a tetrahedron onto an axis
void project_tetrahedron_on_axis(const Tetrahedron& tetra, const Point& axis, double& min_proj, double& max_proj) {
    std::vector<Point> verts = tetra.get_vertices();
    if (verts.empty()) return;

    min_proj = verts[0].dot(axis);
    max_proj = min_proj;

    for (size_t i = 1; i < verts.size(); ++i) {
        double projection = verts[i].dot(axis);
        if (projection < min_proj) min_proj = projection;
        if (projection > max_proj) max_proj = projection;
    }
}

// Structure to represent a plane
struct Plane {
    Point normal;
    double d; // ax + by + cz + d = 0

    Plane(const Point& n, double _d) : normal(n.normalize()), d(_d) {}

    // Calculate signed distance from a point to the plane
    double signed_distance(const Point& p) const {
        return normal.dot(p) + d;
    }
};

// Structure to represent a face of a polyhedron
struct Face {
    std::vector<int> vertex_indices; // Indices into the polyhedron's vertex list
    Plane plane; // The plane containing this face

    Face(const std::vector<int>& indices, const Plane& p) : vertex_indices(indices), plane(p) {}
};

// Structure to represent a polyhedron (the container)
struct Polyhedron {
    std::vector<Point> vertices;
    std::vector<Face> faces;

    // A placeholder function to compute faces from vertices.
    // In a real application, this would involve a convex hull algorithm
    // if only vertices are given, or directly loaded if face data is available.
    void compute_faces() {
        // This is a complex geometric problem (e.g., convex hull algorithm)
        // If the container is assumed to be convex, we need to find its faces.
        // For simplicity, if we don't have a convex hull library, one might
        // approximate with an AABB or manually define faces for known shapes.
        // Given just a list of vertices, we cannot reliably determine faces
        // without a convex hull algorithm.

        // Example: If we assume it's a simple bounding box based on min/max coordinates
        // This is NOT a general solution for arbitrary polyhedra but can work for simple cases.
        if (vertices.empty()) return;

        double min_x = vertices[0].x, max_x = vertices[0].x;
        double min_y = vertices[0].y, max_y = vertices[0].y;
        double min_z = vertices[0].z, max_z = vertices[0].z;

        for (const auto& v : vertices) {
            min_x = std::min(min_x, v.x);
            max_x = std::max(max_x, v.x);
            min_y = std::min(min_y, v.y);
            max_y = std::max(max_y, v.y);
            min_z = std::min(min_z, v.z);
            max_z = std::max(max_z, v.z);
        }

        // Create 6 faces for an AABB
        // Face 1: -X (normal (-1,0,0))
        faces.emplace_back(std::vector<int>{}, Plane({-1, 0, 0}, -min_x));
        // Face 2: +X (normal (1,0,0))
        faces.emplace_back(std::vector<int>{}, Plane({1, 0, 0}, -max_x));
        // Face 3: -Y (normal (0,-1,0))
        faces.emplace_back(std::vector<int>{}, Plane({0, -1, 0}, -min_y));
        // Face 4: +Y (normal (0,1,0))
        faces.emplace_back(std::vector<int>{}, Plane({0, 1, 0}, -max_y));
        // Face 5: -Z (normal (0,0,-1))
        faces.emplace_back(std::vector<int>{}, Plane({0, 0, -1}, -min_z));
        // Face 6: +Z (normal (0,0,1))
        faces.emplace_back(std::vector<int>{}, Plane({0, 0, 1}, -max_z));
    }

    // Check if a point is inside the polyhedron (assuming convex)
    // A point is inside a convex polyhedron if it's on the "negative" side
    // of all its faces' planes (i.e., signed distance <= 0).
    bool is_point_inside(const Point& p) const {
        if (faces.empty()) {
            // If no faces defined, cannot determine if point is inside.
            std::cerr << "Warning: Polyhedron has no faces defined. Cannot check if point is inside." << std::endl;
            return false;
        }
        for (const auto& face : faces) {
            if (face.plane.signed_distance(p) > EPSILON) {
                return false; // Point is outside this face's plane
            }
        }
        return true; // Point is inside or on all faces
    }
};

// Check if a tetrahedron is entirely contained within the polyhedron
bool tetrahedron_fits_in_polyhedron(const Tetrahedron& tetra, const Polyhedron& container) {
    if (container.faces.empty()) {
        // If the container has no defined faces (e.g., compute_faces wasn't called or failed),
        // we cannot reliably check for containment.
        std::cerr << "Warning: Container has no faces, cannot check if tetrahedron fits." << std::endl;
        return false;
    }

    std::vector<Point> tetra_vertices = tetra.get_vertices();
    for (const auto& v : tetra_vertices) {
        if (!container.is_point_inside(v)) {
            return false; // At least one vertex is outside
        }
    }
    return true; // All vertices are inside or on the boundary
}

// Helper function to check for overlap between two 1D intervals
bool overlaps_1d(double min1, double max1, double min2, double max2) {
    return (min1 <= max2 - EPSILON) && (min2 <= max1 - EPSILON);
}

// Check for overlap between two tetrahedrons using Separating Axis Theorem (SAT)
bool tetrahedron_overlaps_tetrahedron(const Tetrahedron& t1, const Tetrahedron& t2) {
    std::vector<Point> axes;

    // Axes from face normals of t1
    std::vector<Point> t1_normals = t1.get_face_normals();
    axes.insert(axes.end(), t1_normals.begin(), t1_normals.end());

    // Axes from face normals of t2
    std::vector<Point> t2_normals = t2.get_face_normals();
    axes.insert(axes.end(), t2_normals.begin(), t2_normals.end());

    // Axes from cross products of edges from t1 and t2
    std::vector<Point> t1_edges = t1.get_edge_vectors();
    std::vector<Point> t2_edges = t2.get_edge_vectors();

    for (const auto& edge1 : t1_edges) {
        for (const auto& edge2 : t2_edges) {
            Point axis = edge1.cross(edge2);
            if (axis.length() > EPSILON) { // Only add non-zero axes
                axes.push_back(axis.normalize());
            }
        }
    }

    // Check each axis for separation
    for (const auto& axis : axes) {
        double t1_min, t1_max;
        double t2_min, t2_max;

        project_tetrahedron_on_axis(t1, axis, t1_min, t1_max);
        project_tetrahedron_on_axis(t2, axis, t2_min, t2_max);

        // If the projections don't overlap, there's a separating axis
        if (!overlaps_1d(t1_min, t1_max, t2_min, t2_max)) {
            return false; // Found a separating axis, no overlap
        }
    }

    return true; // No separating axis found, tetrahedrons overlap
}


int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int V; // Number of vertices in the container
    double edge_length; // Edge length of the regular tetrahedrons

    std::cin >> V >> edge_length;

    Polyhedron container;
    container.vertices.reserve(V);

    for (int i = 0; i < V; ++i) {
        double x, y, z;
        std::cin >> x >> y >> z;
        container.vertices.emplace_back(x, y, z);
    }

    // IMPORTANT: In a real-world scenario, if the input polyhedron is complex and
    // only vertices are provided, a robust convex hull algorithm would be needed
    // to correctly compute its faces. The current `compute_faces()` method
    // approximates the container with an Axis-Aligned Bounding Box (AABB),
    // which is a simplification.
    container.compute_faces();

    // For now, just print the parsed input to verify
    std::cerr << "Parsed Container with " << V << " vertices and tetrahedron edge length " << edge_length << std::endl;
    std::cerr << "Container vertices:" << std::endl;
    for (const auto& v : container.vertices) {
        std::cerr << std::fixed << std::setprecision(6) << v.x << " " << v.y << " " << v.z << std::endl;
    }

    // Placeholder for packing logic
    std::vector<Tetrahedron> placed_tetrahedrons;

    // --- Packing Algorithm (Greedy Approach) ---
    // This is a very basic greedy approach. For "ludicrous scale" and optimal packing,
    // a more sophisticated algorithm (e.g., spatial partitioning, more advanced heuristics,
    // randomized search, or global optimization techniques) would be required.
    // The current approach might be slow and not find the optimal packing for large inputs.

    // Define a range of orientations to test (e.g., 90-degree rotations around each axis)
    std::vector<Point> test_orientations;
    // Add 16 orientations for a basic exploration: 0 and 90 degrees around each axis
    test_orientations.emplace_back(0.0, 0.0, 0.0);
    test_orientations.emplace_back(M_PI_2, 0.0, 0.0);
    test_orientations.emplace_back(0.0, M_PI_2, 0.0);
    test_orientations.emplace_back(0.0, 0.0, M_PI_2);
    test_orientations.emplace_back(M_PI_2, M_PI_2, 0.0);
    test_orientations.emplace_back(M_PI_2, 0.0, M_PI_2);
    test_orientations.emplace_back(0.0, M_PI_2, M_PI_2);
    test_orientations.emplace_back(M_PI_2, M_PI_2, M_PI_2);
    // Adding 180 degree rotations for more coverage
    test_orientations.emplace_back(M_PI, 0.0, 0.0);
    test_orientations.emplace_back(0.0, M_PI, 0.0);
    test_orientations.emplace_back(0.0, 0.0, M_PI);
    test_orientations.emplace_back(M_PI, M_PI_2, 0.0);
    test_orientations.emplace_back(M_PI, 0.0, M_PI_2);
    test_orientations.emplace_back(M_PI_2, M_PI, 0.0);
    test_orientations.emplace_back(0.0, M_PI, M_PI_2);
    test_orientations.emplace_back(M_PI, M_PI, 0.0);
    test_orientations.emplace_back(M_PI, 0.0, M_PI);
    test_orientations.emplace_back(0.0, M_PI, M_PI);


    // Determine the bounding box of the container to define a search space for centers
    double min_x = container.vertices[0].x, max_x = container.vertices[0].x;
    double min_y = container.vertices[0].y, max_y = container.vertices[0].y;
    double min_z = container.vertices[0].z, max_z = container.vertices[0].z;

    for (const auto& v : container.vertices) {
        min_x = std::min(min_x, v.x);
        max_x = std::max(max_x, v.x);
        min_y = std::min(min_y, v.y);
        max_y = std::max(max_y, v.y);
        min_z = std::min(min_z, v.z);
        max_z = std::max(max_z, v.z);
    }

    // Define a step size for iterating through potential centers
    // A reasonable step size is related to the tetrahedron's circumradius
    double circumradius = edge_length * std::sqrt(3.0 / 8.0);
    double step_size = circumradius * 0.5; // Half of circumradius for density

    // Loop to try and place tetrahedrons
    bool placed_new_tetra = true;
    while (placed_new_tetra) {
        placed_new_tetra = false;
        // Iterate through potential center positions within the container's bounding box
        for (double x = min_x + circumradius; x <= max_x - circumradius; x += step_size) {
            for (double y = min_y + circumradius; y <= max_y - circumradius; y += step_size) {
                for (double z = min_z + circumradius; z <= max_z - circumradius; z += step_size) {
                    Point current_center(x, y, z);

                    for (const auto& orientation : test_orientations) {
                        Tetrahedron candidate_tetra(current_center, orientation.x, orientation.y, orientation.z, edge_length);

                        // 1. Check if the candidate tetrahedron fits within the container
                        if (!tetrahedron_fits_in_polyhedron(candidate_tetra, container)) {
                            continue; // Doesn't fit
                        }

                        // 2. Check for overlap with already placed tetrahedrons
                        bool overlaps_existing = false;
                        for (const auto& existing_tetra : placed_tetrahedrons) {
                            if (tetrahedron_overlaps_tetrahedron(candidate_tetra, existing_tetra)) {
                                overlaps_existing = true;
                                break;
                            }
                        }

                        if (!overlaps_existing) {
                            // Valid placement found!
                            placed_tetrahedrons.push_back(candidate_tetra);
                            placed_new_tetra = true;
                            // Found a placement for this iteration, break to search for next one
                            goto next_tetra_search;
                        }
                    }
                }
            }
        }
        next_tetra_search:; // Label for goto
    }


    // Output results
    std::cout << placed_tetrahedrons.size() << std::endl;
    for (const auto& tetra : placed_tetrahedrons) {
        std::cout << std::fixed << std::setprecision(6)
                  << tetra.center.x << " " << tetra.center.y << " " << tetra.center.z << " "
                  << tetra.rx << " " << tetra.ry << " " << tetra.rz << std::endl;
    }

    return 0;
}
