using System;
using System.Collections.Generic;
using System.Linq;

namespace ChinesePostman
{
    public class ChinesePostmanSolver
    {
        private Graph _graph;

        public ChinesePostmanSolver(Graph graph)
        {
            _graph = graph;
        }

        // Generates all possible pairings of odd-degree vertices
        public List<List<(Vertex, Vertex)>> GeneratePairings(List<Vertex> oddVertices)
        {
            List<List<(Vertex, Vertex)>> allPairings = new List<List<(Vertex, Vertex)>>();
            if (oddVertices.Count == 0)
            {
                allPairings.Add(new List<(Vertex, Vertex)>());
                return allPairings;
            }

            // Recursive helper function
            void FindPairings(List<Vertex> currentRemainingVertices, List<(Vertex, Vertex)> currentPairing)
            {
                if (currentRemainingVertices.Count == 0)
                {
                    allPairings.Add(new List<(Vertex, Vertex)>(currentPairing));
                    return;
                }

                Vertex u = currentRemainingVertices[0];
                // The list of vertices to pair 'u' with
                List<Vertex> partners = currentRemainingVertices.Skip(1).ToList();

                for (int i = 0; i < partners.Count; i++)
                {
                    Vertex v = partners[i];

                    // Create the list for the next recursive call:
                    // Remove 'u' (which is currentRemainingVertices[0]) and 'v'
                    List<Vertex> nextRemaining = currentRemainingVertices
                                                .Where(x => x != u && x != v)
                                                .ToList();

                    currentPairing.Add((u, v));
                    FindPairings(nextRemaining, currentPairing);
                    currentPairing.RemoveAt(currentPairing.Count - 1); // Backtrack
                }
            }

            FindPairings(oddVertices, new List<(Vertex, Vertex)>());
            return allPairings;
        }


        public List<string> Solve()
        {
            List<Vertex> oddVertices = _graph.GetOddDegreeVertices();

            if (oddVertices.Count == 0)
            {
                // Graph already has an Eulerian circuit, just find it.
                return FindEulerianCircuit(_graph);
            }

            List<List<(Vertex, Vertex)>> allPairings = GeneratePairings(oddVertices);

            List<(Vertex u, Vertex v, int distance, List<Vertex> path)> bestPathAugmentations = null;
            int minTotalDistance = int.MaxValue;

            foreach (var pairing in allPairings)
            {
                int currentTotalDistance = 0;
                // Store path as a list of edges for augmentation
                List<List<Edge>> currentPathAugmentations = new List<List<Edge>>(); 

                foreach (var (u, v) in pairing)
                {
                    var (distance, path) = _graph.Dijkstra(u, v);
                    if (distance == int.MaxValue)
                    {
                        currentTotalDistance = int.MaxValue;
                        break;
                    }
                    currentTotalDistance += distance;

                    // Extract edges from the shortest path for augmentation
                    List<Edge> pathEdges = new List<Edge>();
                    for (int i = 0; i < path.Count - 1; i++)
                    {
                        Vertex p_source = path[i];
                        Vertex p_destination = path[i + 1];
                        // Find the original edge in _graph that connects p_source and p_destination
                        Edge originalEdge = _graph.Edges.FirstOrDefault(e =>
                            (e.Source == p_source && e.Destination == p_destination) ||
                            (e.Source == p_destination && e.Destination == p_source));
                        
                        if (originalEdge != null)
                        {
                            pathEdges.Add(new Edge(p_source, p_destination, originalEdge.Weight));
                        }
                        else
                        {
                            // This should ideally not happen if the graph is connected and Dijkstra works correctly
                            Console.WriteLine($"Error: Missing original edge between {p_source.Name} and {p_destination.Name} for augmentation.");
                        }
                    }
                    currentPathAugmentations.Add(pathEdges);
                }

                if (currentTotalDistance < minTotalDistance)
                {
                    minTotalDistance = currentTotalDistance;
                    // Store the list of lists of edges for the best augmentation
                    bestPathAugmentations = currentPathAugmentations;
                }
            }

            // Create a copy of the graph to augment
            Graph augmentedGraph = new Graph();
            foreach (var vertex in _graph.Vertices)
            {
                augmentedGraph.AddVertex(vertex.Name);
            }
            // Add original edges to the augmented graph
            foreach (var edge in _graph.Edges)
            {
                // Use AddEdge to ensure both directions are added to adjacency lists for degree calculation
                augmentedGraph.AddEdge(edge.Source.Name, edge.Destination.Name, edge.Weight);
            }

            // Augment the graph with the best path augmentations
            foreach (var pathEdges in bestPathAugmentations) // pathEdges is List<Edge> for one shortest path
            {
                foreach (var edgeToDuplicate in pathEdges)
                {
                    // Call AddEdge with existing source/destination names and weight
                    // This will add another "logical" edge (two Edge objects to vertex.Edges lists, one to Graph.Edges)
                    augmentedGraph.AddEdge(edgeToDuplicate.Source.Name, edgeToDuplicate.Destination.Name, edgeToDuplicate.Weight);
                }
            }
            
            // Find Eulerian circuit in the augmented graph
            return FindEulerianCircuit(augmentedGraph);
        }

        // Hierholzer's Algorithm for finding Eulerian Circuit
        private List<string> FindEulerianCircuit(Graph graph)
        {
            List<string> circuit = new List<string>();

            // Check if all vertices have even degree
            if (graph.Vertices.Any(v => v.Degree() % 2 != 0))
            {
                throw new InvalidOperationException("Graph is not Eulerian.");
            }

            if (!graph.Vertices.Any())
            {
                return circuit;
            }

            Stack<Vertex> currentPathStack = new Stack<Vertex>();
            Vertex currentVertex = graph.Vertices.First(); // Start from an arbitrary vertex
            currentPathStack.Push(currentVertex);

            // Create a mutable adjacency list for Hierholzer's algorithm
            // Need to make a deep copy of edges or use a way to track visited
            Dictionary<Vertex, List<Edge>> adjacency = new Dictionary<Vertex, List<Edge>>();
            foreach (var v in graph.Vertices)
            {
                // Create new Edge objects to allow removal without affecting original graph's Edge lists
                adjacency[v] = v.Edges.Select(e => new Edge(e.Source, e.Destination, e.Weight)).ToList();
            }

            while (currentPathStack.Any())
            {
                if (adjacency[currentVertex].Any())
                {
                    // If current vertex has untraversed edges
                    currentPathStack.Push(currentVertex);
                    Edge nextEdge = adjacency[currentVertex].First();
                    adjacency[currentVertex].Remove(nextEdge);
                    
                    // Find and remove the reverse edge from the destination's adjacency list
                    // It's important to match the specific edge, not just any edge between the two vertices
                    Edge reverseEdgeToRemove = adjacency[nextEdge.Destination].FirstOrDefault(e =>
                        e.Source == nextEdge.Destination &&
                        e.Destination == currentVertex &&
                        e.Weight == nextEdge.Weight); // Match by weight for precision

                    if (reverseEdgeToRemove != null)
                    {
                        adjacency[nextEdge.Destination].Remove(reverseEdgeToRemove);
                    }
                    else
                    {
                        // This indicates a problem in graph construction or an assumption mismatch.
                        // For a correctly formed undirected multigraph, a reverse edge should always exist.
                        Console.WriteLine($"Warning: Corresponding reverse edge not found for {nextEdge.Source.Name} -> {nextEdge.Destination.Name} ({nextEdge.Weight}).");
                    }

                    currentVertex = nextEdge.Destination;
                }
                else
                {
                    // No more untraversed edges from current vertex, add to circuit
                    circuit.Add(currentVertex.Name);
                    currentVertex = currentPathStack.Pop();
                }
            }
            circuit.Reverse(); // The algorithm builds the circuit in reverse
            return circuit;
