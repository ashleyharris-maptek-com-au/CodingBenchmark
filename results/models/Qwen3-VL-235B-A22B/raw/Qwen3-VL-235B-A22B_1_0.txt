// Define types for clarity
using Coord = std::pair<float, float>;
using Path = std::vector<int>;
using DistanceMatrix = std::vector<std::vector<float>>;

// Simple distance calculation
inline float distance(const Coord& a, const Coord& b) {
    float dx = a.first - b.first;
    float dy = a.second - b.second;
    return std::sqrt(dx * dx + dy * dy);
}

// For very large N, we'll use a simple greedy approach
Path greedy_tsp(const std::vector<Coord>& cities) {
    int n = cities.size();
    if (n <= 1) return {0};
    
    Path path;
    path.reserve(n);
    path.push_back(0); // Start at city 0
    
    std::vector<bool> visited(n, false);
    visited[0] = true;
    
    int current = 0;
    
    for (int i = 1; i < n; ++i) {
        int next = -1;
        float min_dist = std::numeric_limits<float>::max();
        
        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                float dist = distance(cities[current], cities[j]);
                if (dist < min_dist) {
                    min_dist = dist;
                    next = j;
                }
            }
        }
        
        if (next == -1) break; // Shouldn't happen
        
        path.push_back(next);
        visited[next] = true;
        current = next;
    }
    
    return path;
}

// For medium sizes, use 2-opt improvement on greedy solution
Path two_opt_improve(const std::vector<Coord>& cities, Path path) {
    int n = cities.size();
    if (n <= 2) return path;
    
    bool improved = true;
    int iterations = 0;
    const int max_iterations = 100000;
    
    while (improved && iterations < max_iterations) {
        improved = false;
        iterations++;
        
        for (int i = 0; i < n - 2; ++i) {
            for (int j = i + 2; j < n; ++j) {
                // Calculate current distance for segments (i, i+1) and (j, j+1)
                int i1 = path[i];
                int i2 = path[(i + 1) % n];
                int j1 = path[j];
                int j2 = path[(j + 1) % n];
                
                float d1 = distance(cities[i1], cities[i2]) + distance(cities[j1], cities[j2]);
                float d2 = distance(cities[i1], cities[j1]) + distance(cities[i2], cities[j2]);
                
                if (d2 < d1 - 1e-9) { // Improve if significantly better
                    // Reverse path from i+1 to j
                    std::reverse(path.begin() + i + 1, path.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
    
    return path;
}

// For small to medium sizes, use a more sophisticated approach
Path nearest_neighbor_with_2opt(const std::vector<Coord>& cities) {
    int n = cities.size();
    if (n <= 1) return {0};
    
    // Start with greedy (nearest neighbor)
    Path path = greedy_tsp(cities);
    
    // Apply 2-opt improvement
    path = two_opt_improve(cities, path);
    
    return path;
}

// For very large datasets, use a hierarchical approach
Path hierarchical_tsp(const std::vector<Coord>& cities) {
    int n = cities.size();
    if (n <= 1) return {0};
    
    // For very large N, use a simplified approach
    if (n > 1000000) {
        // Use a simple space-filling curve approach (Hilbert curve approximation)
        // For simplicity, we'll use a greedy approach with random sampling for large N
        return greedy_tsp(cities);
    }
    
    // For 100000-1000000 cities, use a sampling approach
    if (n > 100000) {
        // Sample 10000 points for initial solution, then refine
        int sample_size = std::min(10000, n);
        std::vector<int> sample_indices;
        for (int i = 0; i < n; ++i) {
            sample_indices.push_back(i);
        }
        
        // Randomly sample (but keep city 0)
        std::random_device rd;
        std::mt19937 gen(rd());
        std::shuffle(sample_indices.begin(), sample_indices.end(), gen);
        
        // Ensure city 0 is included
        auto it = std::find(sample_indices.begin(), sample_indices.end(), 0);
        if (it != sample_indices.end()) {
            std::swap(sample_indices[0], *it);
        } else {
            sample_indices[0] = 0;
        }
        
        sample_indices.resize(sample_size);
        
        // Create sampled cities
        std::vector<Coord> sampled_cities;
        for (int idx : sample_indices) {
            sampled_cities.push_back(cities[idx]);
        }
        
        // Solve for sample
        Path sample_path = nearest_neighbor_with_2opt(sampled_cities);
        
        // Map back to original indices
        Path result;
        for (int idx : sample_path) {
            result.push_back(sample_indices[idx]);
        }
        
        // Add remaining cities using greedy insertion
        std::vector<bool> added(n, false);
        for (int idx : result) {
            added[idx] = true;
        }
        
        for (int i = 0; i < n; ++i) {
            if (!added[i]) {
                // Find best position to insert
                float best_cost = std::numeric_limits<float>::max();
                int best_pos = 0;
                
                for (int j = 0; j < result.size(); ++j) {
                    int prev = result[j];
                    int next = result[(j + 1) % result.size()];
                    
                    float old_cost = distance(cities[prev], cities[next]);
                    float new_cost = distance(cities[prev], cities[i]) + distance(cities[i], cities[next]);
                    float cost_diff = new_cost - old_cost;
                    
                    if (cost_diff < best_cost) {
                        best_cost = cost_diff;
                        best_pos = j + 1;
                    }
                }
                
                if (best_pos == result.size()) {
                    result.push_back(i);
                } else {
                    result.insert(result.begin() + best_pos, i);
                }
            }
        }
        
        return result;
    }
    
    // For 10000-100000 cities, use standard approach
    return nearest_neighbor_with_2opt(cities);
}

// For small N (<= 40), use brute force with optimization
Path brute_force_tsp(const std::vector<Coord>& cities) {
    int n = cities.size();
    if (n <= 1) return {0};
    if (n == 2) return {0, 1};
    
    // Use next_permutation for small N
    std::vector<int> perm;
    for (int i = 1; i < n; ++i) {
        perm.push_back(i);
    }
    
    float min_dist = std::numeric_limits<float>::max();
    Path best_path = {0};
    
    do {
        Path current_path = {0};
        current_path.insert(current_path.end(), perm.begin(), perm.end());
        
        // Calculate total distance
        float total_dist = 0.0f;
        for (int i = 0; i < n - 1; ++i) {
            total_dist += distance(cities[current_path[i]], cities[current_path[i + 1]]);
        }
        // Don't add return to start (as per requirements)
        
        if (total_dist < min_dist) {
            min_dist = total_dist;
            best_path = current_path;
        }
        
    } while (std::next_permutation(perm.begin(), perm.end()));
    
    return best_path;
}

// Main solver that adapts based on problem size
Path solve_tsp(const std::vector<Coord>& cities) {
    int n = cities.size();
    
    if (n <= 1) {
        return {0};
    }
    
    // Choose strategy based on problem size
    if (n <= 40) {
        // Brute force for small instances
        return brute_force_tsp(cities);
    } else if (n <= 1000) {
        // Nearest neighbor with 2-opt for medium instances
        return nearest_neighbor_with_2opt(cities);
    } else if (n <= 100000) {
        // Hierarchical approach for large instances
        return hierarchical_tsp(cities);
    } else if (n <= 10000000) {
        // For very large instances, use greedy with some optimization
        Path path = greedy_tsp(cities);
        
        // Apply limited 2-opt for improvement
        if (n <= 1000000) {
            path = two_opt_improve(cities, path);
        }
        
        return path;
    } else {
        // For epic scale, use simple greedy
        return greedy_tsp(cities);
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int n;
    std::cin >> n;
    
    std::vector<Coord> cities;
    cities.reserve(n);
    
    for (int i = 0; i < n; ++i) {
        float x, y;
        std::cin >> x >> y;
        cities.push_back({x, y});
    }
    
    // Solve TSP
    Path result = solve_tsp(cities);
    
    // Output the result (0-indexed, starting with 0, no return to start)
    for (int i = 0; i < result.size(); ++i) {
        if (i > 0) std::cout << " ";
        std::cout << result[i];
    }
    std::cout << std::endl;
    
    return 0;
}
