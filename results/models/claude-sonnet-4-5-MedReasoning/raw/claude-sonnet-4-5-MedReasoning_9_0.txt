using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        // Read input
        var firstLine = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        var cutLengths = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries)
            .Select(int.Parse).ToArray();

        // Create array of (length, originalIndex) and sort by length descending (FFD heuristic)
        var cuts = new (int length, int index)[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = (cutLengths[i], i);
        }
        Array.Sort(cuts, (a, b) => b.length.CompareTo(a.length));

        // First Fit Decreasing algorithm
        var stocks = new List<Stock>();

        foreach (var (length, index) in cuts)
        {
            // Find first stock that can fit this cut
            bool placed = false;
            for (int i = 0; i < stocks.Count; i++)
            {
                if (stocks[i].Remaining >= length)
                {
                    stocks[i].AddCut(index, length);
                    placed = true;
                    break;
                }
            }

            // If no stock can fit, create a new one
            if (!placed)
            {
                var newStock = new Stock(stockLength);
                newStock.AddCut(index, length);
                stocks.Add(newStock);
            }
        }

        // Output
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock.Cuts));
        }
    }

    class Stock
    {
        public int Remaining { get; private set; }
        public List<int> Cuts { get; }

        public Stock(int capacity)
        {
            Remaining = capacity;
            Cuts = new List<int>();
        }

        public void AddCut(int cutIndex, int length)
        {
            Cuts.Add(cutIndex);
            Remaining -= length;
        }
    }
}
