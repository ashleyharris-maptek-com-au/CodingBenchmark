use std::io::{self, BufRead};
use std::collections::VecDeque;

#[derive(Clone)]
struct Grid {
    width: usize,
    height: usize,
    obstacles: Vec<Vec<bool>>,
    require_cycle: bool,
    target_cells: usize,
}

impl Grid {
    fn new(width: usize, height: usize, require_cycle: bool) -> Self {
        Grid {
            width,
            height,
            obstacles: vec![vec![false; width]; height],
            require_cycle,
            target_cells: 0,
        }
    }

    fn add_obstacle(&mut self, x: usize, y: usize) {
        self.obstacles[y][x] = true;
    }

    fn finalize(&mut self) {
        self.target_cells = self.width * self.height - self.count_obstacles();
    }

    fn count_obstacles(&self) -> usize {
        self.obstacles.iter().flatten().filter(|&&b| b).count()
    }

    fn is_valid(&self, x: i32, y: i32) -> bool {
        x >= 0 && y >= 0 &&
        (x as usize) < self.width &&
        (y as usize) < self.height &&
        !self.obstacles[y as usize][x as usize]
    }

    fn get_neighbors(&self, x: i32, y: i32) -> Vec<(i32, i32)> {
        let dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)];
        dirs.iter()
            .map(|(dx, dy)| (x + dx, y + dy))
            .filter(|&(nx, ny)| self.is_valid(nx, ny))
            .collect()
    }
}

struct Solver {
    grid: Grid,
    visited: Vec<Vec<bool>>,
    path: Vec<(i32, i32)>,
    best_path: Option<Vec<(i32, i32)>>,
}

impl Solver {
    fn new(grid: Grid) -> Self {
        let visited = vec![vec![false; grid.width]; grid.height];
        Solver {
            grid,
            visited,
            path: Vec::new(),
            best_path: None,
        }
    }

    fn solve(&mut self) -> Option<Vec<(i32, i32)>> {
        // Check if start is valid
        if !self.grid.is_valid(0, 0) {
            return None;
        }

        // Quick connectivity check
        if !self.is_connected() {
            return None;
        }

        // Start DFS from (0, 0)
        self.visited[0][0] = true;
        self.path.push((0, 0));
        self.dfs(0, 0);
        self.best_path.clone()
    }

    fn dfs(&mut self, x: i32, y: i32) -> bool {
        // Check if we've visited all cells
        if self.path.len() == self.grid.target_cells {
            if self.grid.require_cycle {
                // Check if current cell is adjacent to (0, 0)
                let neighbors = self.grid.get_neighbors(x, y);
                if neighbors.contains(&(0, 0)) {
                    self.best_path = Some(self.path.clone());
                    return true;
                }
            } else {
                self.best_path = Some(self.path.clone());
                return true;
            }
        }

        // Pruning: check if remaining cells can be reached
        if !self.can_complete(x, y) {
            return false;
        }

        // Try all neighbors
        let neighbors = self.grid.get_neighbors(x, y);

        // Sort neighbors by heuristic (fewest unvisited neighbors first - Warnsdorff's rule)
        let mut scored_neighbors: Vec<_> = neighbors.iter()
            .filter(|&&(nx, ny)| !self.visited[ny as usize][nx as usize])
            .map(|&(nx, ny)| {
                let unvisited_count = self.grid.get_neighbors(nx, ny)
                    .iter()
                    .filter(|&&(nnx, nny)| !self.visited[nny as usize][nnx as usize])
                    .count();
                (nx, ny, unvisited_count)
            })
            .collect();

        scored_neighbors.sort_by_key(|&(_, _, count)| count);

        for (nx, ny, _) in scored_neighbors {
            self.visited[ny as usize][nx as usize] = true;
            self.path.push((nx, ny));

            if self.dfs(nx, ny) {
                return true;
            }

            self.path.pop();
            self.visited[ny as usize][nx as usize] = false;
        }

        false
    }

    fn can_complete(&self, x: i32, y: i32) -> bool {
        // Check if all unvisited cells are still reachable
        let unvisited_count = self.grid.target_cells - self.path.len();

        if unvisited_count == 0 {
            return true;
        }

        // BFS to check connectivity of unvisited cells
        let mut queue = VecDeque::new();
        let mut temp_visited = self.visited.clone();
        let mut reachable = 0;

        // Start from current position
        queue.push_back((x, y));

        while let Some((cx, cy)) = queue.pop_front() {
            for (nx, ny) in self.grid.get_neighbors(cx, cy) {
                if !temp_visited[ny as usize][nx as usize] {
                    temp_visited[ny as usize][nx as usize] = true;
                    reachable += 1;
                    queue.push_back((nx, ny));
                }
            }
        }

        reachable >= unvisited_count
    }

    fn is_connected(&self) -> bool {
        // Check if all non-obstacle cells are connected
        let mut visited = vec![vec![false; self.grid.width]; self.grid.height];
        let mut queue = VecDeque::new();
        let mut count = 0;

        // Find starting position
        if !self.grid.is_valid(0, 0) {
            return false;
        }

        queue.push_back((0, 0));
        visited[0][0] = true;
        count += 1;

        while let Some((x, y)) = queue.pop_front() {
            for (nx, ny) in self.grid.get_neighbors(x, y) {
                if !visited[ny as usize][nx as usize] {
                    visited[ny as usize][nx as usize] = true;
                    count += 1;
                    queue.push_back((nx, ny));
                }
            }
        }

        count == self.grid.target_cells
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2];
    let require_cycle = parts[3] == 1;

    let mut grid = Grid::new(width, height, require_cycle);

    // Read obstacles
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        grid.add_obstacle(coords[0], coords[1]);
    }

    grid.finalize();

    // Solve
    let mut solver = Solver::new(grid);

    match solver.solve() {
        Some(path) => {
            println!("{}", path.len());
            for (x, y) in path {
                println!("{} {}", x, y);
            }
        }
        None => {
            println!("0");
        }
    }
}
