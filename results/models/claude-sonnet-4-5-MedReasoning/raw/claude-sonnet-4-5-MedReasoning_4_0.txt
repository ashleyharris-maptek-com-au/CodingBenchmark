#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <array>
#include <set>

using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }
    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalized() const {
        double l = length();
        return l > 1e-10 ? *this * (1.0 / l) : Vec3(0, 0, 0);
    }
};

struct Tetrahedron {
    Vec3 center;
    double rx, ry, rz;

    void rotate(Vec3& v, double rx, double ry, double rz) const {
        // Rotate around X
        double y1 = v.y * cos(rx) - v.z * sin(rx);
        double z1 = v.y * sin(rx) + v.z * cos(rx);
        v.y = y1; v.z = z1;

        // Rotate around Y
        double x2 = v.x * cos(ry) + v.z * sin(ry);
        double z2 = -v.x * sin(ry) + v.z * cos(ry);
        v.x = x2; v.z = z2;

        // Rotate around Z
        double x3 = v.x * cos(rz) - v.y * sin(rz);
        double y3 = v.x * sin(rz) + v.y * cos(rz);
        v.x = x3; v.y = y3;
    }

    array<Vec3, 4> getVertices(double edge_length) const {
        double h = edge_length * sqrt(2.0 / 3.0);
        double a = edge_length / sqrt(3.0);

        array<Vec3, 4> verts;
        verts[0] = Vec3(a, 0, 0);
        verts[1] = Vec3(-a/2, a*sqrt(3)/2, 0);
        verts[2] = Vec3(-a/2, -a*sqrt(3)/2, 0);
        verts[3] = Vec3(0, 0, h);

        // Center at origin
        Vec3 centroid = (verts[0] + verts[1] + verts[2] + verts[3]) * 0.25;
        for (int i = 0; i < 4; i++) {
            verts[i] = verts[i] - centroid;
            rotate(verts[i], rx, ry, rz);
            verts[i] = verts[i] + center;
        }

        return verts;
    }
};

class TetrahedronPacker {
private:
    vector<Vec3> container_vertices;
    double edge_length;
    Vec3 min_bound, max_bound;
    vector<Tetrahedron> packed;

    void computeBounds() {
        min_bound = Vec3(1e9, 1e9, 1e9);
        max_bound = Vec3(-1e9, -1e9, -1e9);
        for (const auto& v : container_vertices) {
            min_bound.x = min(min_bound.x, v.x);
            min_bound.y = min(min_bound.y, v.y);
            min_bound.z = min(min_bound.z, v.z);
            max_bound.x = max(max_bound.x, v.x);
            max_bound.y = max(max_bound.y, v.y);
            max_bound.z = max(max_bound.z, v.z);
        }
    }

    bool pointInConvexHull(const Vec3& p) const {
        // For simplicity, check bounding box first
        if (p.x < min_bound.x || p.x > max_bound.x ||
            p.y < min_bound.y || p.y > max_bound.y ||
            p.z < min_bound.z || p.z > max_bound.z) {
            return false;
        }
        return true;
    }

    bool isTetrahedronValid(const Tetrahedron& tet) const {
        auto verts = tet.getVertices(edge_length);

        // Check all vertices are inside container
        for (const auto& v : verts) {
            if (!pointInConvexHull(v)) return false;
        }

        // Check overlap with existing tetrahedrons
        double circumradius = edge_length * sqrt(3.0 / 8.0);
        double min_dist = 2.0 * circumradius * 0.85; // Safety margin

        for (const auto& existing : packed) {
            double dist = (tet.center - existing.center).length();
            if (dist < min_dist) {
                return false;
            }
        }

        return true;
    }

public:
    void setContainer(const vector<Vec3>& vertices, double edge) {
        container_vertices = vertices;
        edge_length = edge;
        computeBounds();
    }

    void pack() {
        packed.clear();

        double h = edge_length * sqrt(2.0 / 3.0);
        double r = edge_length * sqrt(3.0 / 8.0);

        // Calculate container volume for adaptive spacing
        double dx = max_bound.x - min_bound.x;
        double dy = max_bound.y - min_bound.y;
        double dz = max_bound.z - min_bound.z;
        double volume = dx * dy * dz;

        // Adaptive grid spacing based on volume
        double spacing = edge_length * 0.7;
        if (volume > 10000) {
            spacing = edge_length * 1.5;
        } else if (volume > 1000) {
            spacing = edge_length * 1.0;
        }

        double margin = r * 2;

        // Try multiple orientations for better packing
        vector<array<double, 3>> orientations = {
            {0, 0, 0},
            {0, 0, M_PI/3},
            {0, M_PI/6, 0},
            {M_PI/6, 0, 0}
        };

        // Grid-based packing with multiple orientations
        for (double x = min_bound.x + margin; x <= max_bound.x - margin; x += spacing) {
            for (double y = min_bound.y + margin; y <= max_bound.y - margin; y += spacing) {
                for (double z = min_bound.z + margin; z <= max_bound.z - margin; z += spacing) {
                    for (const auto& orient : orientations) {
                        Tetrahedron tet;
                        tet.center = Vec3(x, y, z);
                        tet.rx = orient[0];
                        tet.ry = orient[1];
                        tet.rz = orient[2];

                        if (isTetrahedronValid(tet)) {
                            packed.push_back(tet);
                            break; // Found valid orientation, move to next position
                        }
                    }
                }
            }
        }
    }

    const vector<Tetrahedron>& getPacked() const {
        return packed;
    }
};

int main() {
    int V;
    double edge_length;
    cin >> V >> edge_length;

    vector<Vec3> vertices(V);
    for (int i = 0; i < V; i++) {
        cin >> vertices[i].x >> vertices[i].y >> vertices[i].z;
    }

    TetrahedronPacker packer;
    packer.setContainer(vertices, edge_length);
    packer.pack();

    const auto& packed = packer.getPacked();
    cout << packed.size() << endl;
    for (const auto& tet : packed) {
        cout << tet.center.x << " " << tet.center.y << " " << tet.center.z << " "
             << tet.rx << " " << tet.ry << " " << tet.rz << endl;
    }

    return 0;
}
