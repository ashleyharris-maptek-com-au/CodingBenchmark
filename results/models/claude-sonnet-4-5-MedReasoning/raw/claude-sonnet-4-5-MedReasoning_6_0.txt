using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        // Parse input
        var firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        double mu = double.Parse(firstLine[1]);

        var startLine = Console.ReadLine().Split();
        var startOrbit = new OrbitalState(
            double.Parse(startLine[0]), double.Parse(startLine[1]), double.Parse(startLine[2]),
            double.Parse(startLine[3]), double.Parse(startLine[4]), double.Parse(startLine[5])
        );

        var stations = new OrbitalState[N];
        for (int i = 0; i < N; i++)
        {
            var line = Console.ReadLine().Split();
            stations[i] = new OrbitalState(
                double.Parse(line[0]), double.Parse(line[1]), double.Parse(line[2]),
                double.Parse(line[3]), double.Parse(line[4]), double.Parse(line[5])
            );
        }

        // Solve TSP with scale-adaptive strategy
        var solver = new OrbitalTSPSolver(startOrbit, stations, mu);
        var order = solver.Solve();

        // Output result
        Console.WriteLine(string.Join(" ", order));
    }
}

struct Vec3
{
    public double X, Y, Z;

    public Vec3(double x, double y, double z)
    {
        X = x; Y = y; Z = z;
    }

    public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vec3 operator *(Vec3 v, double s) => new Vec3(v.X * s, v.Y * s, v.Z * s);
    public static Vec3 operator *(double s, Vec3 v) => v * s;
    public static Vec3 operator /(Vec3 v, double s) => new Vec3(v.X / s, v.Y / s, v.Z / s);

    public double Magnitude => Math.Sqrt(X * X + Y * Y + Z * Z);
    public Vec3 Normalized => this / Magnitude;
    public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
    public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(
        a.Y * b.Z - a.Z * b.Y,
        a.Z * b.X - a.X * b.Z,
        a.X * b.Y - a.Y * b.X
    );
}

class OrbitalState
{
    public Vec3 Position;
    public Vec3 Velocity;

    public OrbitalState(double x, double y, double z, double vx, double vy, double vz)
    {
        Position = new Vec3(x, y, z);
        Velocity = new Vec3(vx, vy, vz);
    }

    public OrbitalState(Vec3 pos, Vec3 vel)
    {
        Position = pos;
        Velocity = vel;
    }
}

class OrbitalTSPSolver
{
    private OrbitalState start;
    private OrbitalState[] stations;
    private double mu;
    private double[,] deltaVMatrix;

    public OrbitalTSPSolver(OrbitalState start, OrbitalState[] stations, double mu)
    {
        this.start = start;
        this.stations = stations;
        this.mu = mu;
        this.deltaVMatrix = new double[stations.Length + 1, stations.Length + 1];
        PrecomputeDeltaV();
    }

    private void PrecomputeDeltaV()
    {
        // Index 0 represents start position
        for (int i = 0; i < stations.Length; i++)
        {
            deltaVMatrix[0, i + 1] = EstimateDeltaV(start, stations[i]);
        }

        for (int i = 0; i < stations.Length; i++)
        {
            for (int j = 0; j < stations.Length; j++)
            {
                if (i != j)
                {
                    deltaVMatrix[i + 1, j + 1] = EstimateDeltaV(stations[i], stations[j]);
                }
            }
        }
    }

    private double EstimateDeltaV(OrbitalState from, OrbitalState to)
    {
        // Simplified delta-V estimation using Hohmann-like transfer approximation
        // and rendezvous delta-V

        double r1 = from.Position.Magnitude;
        double r2 = to.Position.Magnitude;

        // Orbital velocities
        double v1 = from.Velocity.Magnitude;
        double v2 = to.Velocity.Magnitude;

        // Simple Hohmann transfer approximation
        double vCirc1 = Math.Sqrt(mu / r1);
        double vCirc2 = Math.Sqrt(mu / r2);

        // Transfer orbit
        double a_transfer = (r1 + r2) / 2.0;
        double v_transfer_1 = Math.Sqrt(mu * (2.0 / r1 - 1.0 / a_transfer));
        double v_transfer_2 = Math.Sqrt(mu * (2.0 / r2 - 1.0 / a_transfer));

        // Delta-V for departure and arrival
        double dv1 = Math.Abs(v_transfer_1 - vCirc1);
        double dv2 = Math.Abs(vCirc2 - v_transfer_2);

        // Add position and velocity difference penalty
        double positionDiff = (to.Position - from.Position).Magnitude;
        double velocityDiff = (to.Velocity - from.Velocity).Magnitude;

        // Combined estimate with phasing cost
        double phasingCost = Math.Min(positionDiff / 10000.0, 2.0); // Normalize
        double velocityPenalty = velocityDiff * 0.5;

        return dv1 + dv2 + phasingCost + velocityPenalty;
    }

    public int[] Solve()
    {
        int n = stations.Length;

        if (n <= 8)
        {
            // For small N, try all permutations (limited to avoid timeout)
            return SolveBruteForce();
        }
        else
        {
            // For larger N, use nearest neighbor + 2-opt
            return SolveHeuristic();
        }
    }

    private int[] SolveBruteForce()
    {
        int n = stations.Length;
        var indices = Enumerable.Range(0, n).ToArray();
        var bestOrder = new int[n];
        double bestCost = double.MaxValue;

        // Limit permutations for performance
        int maxPermutations = n <= 8 ? int.MaxValue : 50000;
        int permCount = 0;

        foreach (var perm in Permutations(indices))
        {
            if (++permCount > maxPermutations) break;

            double cost = CalculateTourCost(perm);
            if (cost < bestCost)
            {
                bestCost = cost;
                Array.Copy(perm, bestOrder, n);
            }
        }

        return bestOrder;
    }

    private int[] SolveHeuristic()
    {
        // Nearest neighbor construction
        var tour = NearestNeighborTour();

        // 2-opt improvement
        tour = TwoOptImprove(tour);

        return tour;
    }

    private int[] NearestNeighborTour()
    {
        int n = stations.Length;
        var tour = new List<int>();
        var visited = new bool[n];
        int current = -1; // Start from launch position (index 0 in matrix)

        for (int i = 0; i < n; i++)
        {
            int next = -1;
            double minDeltaV = double.MaxValue;

            for (int j = 0; j < n; j++)
            {
                if (!visited[j])
                {
                    double dv = deltaVMatrix[current + 1, j + 1];
                    if (dv < minDeltaV)
                    {
                        minDeltaV = dv;
                        next = j;
                    }
                }
            }

            visited[next] = true;
            tour.Add(next);
            current = next;
        }

        return tour.ToArray();
    }

    private int[] TwoOptImprove(int[] tour)
    {
        int n = tour.Length;
        bool improved = true;
        var currentTour = (int[])tour.Clone();

        while (improved)
        {
            improved = false;

            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    double currentCost = EdgeCost(currentTour, i, i + 1) + EdgeCost(currentTour, j, (j + 1) % n);

                    // Reverse segment [i+1, j]
                    Array.Reverse(currentTour, i + 1, j - i);
                    double newCost = EdgeCost(currentTour, i, i + 1) + EdgeCost(currentTour, j, (j + 1) % n);

                    if (newCost < currentCost)
                    {
                        improved = true;
                        break;
                    }
                    else
                    {
                        // Reverse back
                        Array.Reverse(currentTour, i + 1, j - i);
                    }
                }
                if (improved) break;
            }
        }

        return currentTour;
    }

    private double EdgeCost(int[] tour, int idx1, int idx2)
    {
        int n = tour.Length;
        int from = idx1 < 0 ? -1 : (idx1 >= n ? -1 : tour[idx1]);
        int to = idx2 >= n ? -1 : tour[idx2 % n];

        if (to == -1) return 0;

        return deltaVMatrix[from + 1, to + 1];
    }

    private double CalculateTourCost(int[] tour)
    {
        double cost = deltaVMatrix[0, tour[0] + 1]; // Start to first station

        for (int i = 0; i < tour.Length - 1; i++)
        {
            cost += deltaVMatrix[tour[i] + 1, tour[i + 1] + 1];
        }

        return cost;
    }

    private IEnumerable<int[]> Permutations(int[] array)
    {
        int n = array.Length;
        var indices = new int[n];
        yield return (int[])array.Clone();

        int i = 0;
        while (i < n)
        {
            if (indices[i] < i)
            {
                int swapIdx = (i % 2 == 0) ? 0 : indices[i];
                int temp = array[swapIdx];
                array[swapIdx] = array[i];
                array[i] = temp;

                yield return (int[])array.Clone();

                indices[i]++;
                i = 0;
            }
            else
            {
                indices[i] = 0;
                i++;
            }
        }
    }
}
