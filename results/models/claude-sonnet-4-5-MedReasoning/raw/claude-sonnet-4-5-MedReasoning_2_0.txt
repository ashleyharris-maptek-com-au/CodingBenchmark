using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostman
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int n = int.Parse(input[0]);
        int m = int.Parse(input[1]);

        var graph = new List<(int to, int weight, int edgeId)>[n];
        for (int i = 0; i < n; i++)
            graph[i] = new List<(int, int, int)>();

        for (int i = 0; i < m; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            int w = int.Parse(edge[2]);

            graph[u].Add((v, w, i));
            graph[v].Add((u, w, i));
        }

        // Find odd degree vertices
        var oddVertices = new List<int>();
        for (int i = 0; i < n; i++)
            if (graph[i].Count % 2 == 1)
                oddVertices.Add(i);

        // Build augmented graph
        var augmentedGraph = new List<(int to, int edgeId)>[n];
        for (int i = 0; i < n; i++)
        {
            augmentedGraph[i] = new List<(int, int)>();
            foreach (var e in graph[i])
                augmentedGraph[i].Add((e.to, e.edgeId));
        }

        // If there are odd vertices, add minimum matching edges
        int extraEdgeId = m;
        if (oddVertices.Count > 0)
        {
            var pathData = ComputeAllShortestPaths(n, graph, oddVertices);
            List<(int, int)> matching;

            if (oddVertices.Count <= 20)
            {
                matching = MinimumWeightMatchingDP(oddVertices, pathData.distances);
            }
            else
            {
                matching = GreedyMatching(oddVertices, pathData.distances);
            }

            foreach (var (u, v) in matching)
            {
                var path = pathData.GetPath(u, v);
                for (int i = 0; i < path.Count - 1; i++)
                {
                    augmentedGraph[path[i]].Add((path[i + 1], extraEdgeId));
                    augmentedGraph[path[i + 1]].Add((path[i], extraEdgeId));
                    extraEdgeId++;
                }
            }
        }

        // Find Eulerian circuit
        var circuit = FindEulerianCircuit(n, augmentedGraph, 0);

        // Output
        Console.WriteLine(string.Join(" ", circuit));
    }

    class PathData
    {
        public Dictionary<int, int[]> distances = new Dictionary<int, int[]>();
        public Dictionary<int, int[]> parent = new Dictionary<int, int[]>();

        public List<int> GetPath(int from, int to)
        {
            if (!parent.ContainsKey(from))
                return new List<int> { from, to };

            var path = new List<int>();
            int current = to;
            while (current != from)
            {
                path.Add(current);
                current = parent[from][current];
            }
            path.Add(from);
            path.Reverse();
            return path;
        }
    }

    static PathData ComputeAllShortestPaths(int n, List<(int to, int weight, int edgeId)>[] graph, List<int> vertices)
    {
        var pathData = new PathData();

        foreach (int start in vertices)
        {
            var (dist, parent) = Dijkstra(n, graph, start);
            pathData.distances[start] = dist;
            pathData.parent[start] = parent;
        }

        return pathData;
    }

    static (int[], int[]) Dijkstra(int n, List<(int to, int weight, int edgeId)>[] graph, int start)
    {
        var dist = new int[n];
        var parent = new int[n];
        Array.Fill(dist, int.MaxValue);
        Array.Fill(parent, -1);
        dist[start] = 0;

        var pq = new PriorityQueue<int, int>();
        pq.Enqueue(start, 0);
        var visited = new bool[n];

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();
            if (visited[u]) continue;
            visited[u] = true;

            foreach (var (v, w, _) in graph[u])
            {
                if (dist[u] != int.MaxValue && dist[u] + w < dist[v])
                {
                    dist[v] = dist[u] + w;
                    parent[v] = u;
                    pq.Enqueue(v, dist[v]);
                }
            }
        }

        return (dist, parent);
    }

    static List<(int, int)> MinimumWeightMatchingDP(List<int> vertices, Dictionary<int, int[]> distances)
    {
        int k = vertices.Count;
        var dp = new int[1 << k];
        var parent = new int[1 << k];
        Array.Fill(dp, int.MaxValue);
        Array.Fill(parent, -1);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << k); mask++)
        {
            if (dp[mask] == int.MaxValue) continue;

            int first = -1;
            for (int i = 0; i < k; i++)
            {
                if ((mask & (1 << i)) == 0)
                {
                    first = i;
                    break;
                }
            }

            if (first == -1) continue;

            for (int second = first + 1; second < k; second++)
            {
                if ((mask & (1 << second)) == 0)
                {
                    int newMask = mask | (1 << first) | (1 << second);
                    int dist = distances[vertices[first]][vertices[second]];
                    if (dist == int.MaxValue) continue;

                    int newDist = dp[mask] + dist;
                    if (newDist < dp[newMask])
                    {
                        dp[newMask] = newDist;
                        parent[newMask] = mask | (first << 16) | (second << 24);
                    }
                }
            }
        }

        var matching = new List<(int, int)>();
        int currentMask = (1 << k) - 1;
        while (currentMask > 0 && parent[currentMask] != -1)
        {
            int p = parent[currentMask];
            int first = (p >> 16) & 0xFF;
            int second = (p >> 24) & 0xFF;
            matching.Add((vertices[first], vertices[second]));
            currentMask = p & 0xFFFF;
        }

        return matching;
    }

    static List<(int, int)> GreedyMatching(List<int> vertices, Dictionary<int, int[]> distances)
    {
        var matching = new List<(int, int)>();
        var used = new HashSet<int>();

        var edges = new List<(int dist, int u, int v)>();
        for (int i = 0; i < vertices.Count; i++)
        {
            for (int j = i + 1; j < vertices.Count; j++)
            {
                int dist = distances[vertices[i]][vertices[j]];
                if (dist != int.MaxValue)
                    edges.Add((dist, vertices[i], vertices[j]));
            }
        }

        edges.Sort();

        foreach (var (dist, u, v) in edges)
        {
            if (!used.Contains(u) && !used.Contains(v))
            {
                matching.Add((u, v));
                used.Add(u);
                used.Add(v);
                if (matching.Count * 2 == vertices.Count)
                    break;
            }
        }

        return matching;
    }

    static List<int> FindEulerianCircuit(int n, List<(int to, int edgeId)>[] graph, int start)
    {
        var g = new List<(int to, int edgeId)>[n];
        for (int i = 0; i < n; i++)
            g[i] = new List<(int, int)>(graph[i]);

        var stack = new Stack<int>();
        var circuit = new List<int>();
        var usedEdges = new HashSet<int>();

        stack.Push(start);

        while (stack.Count > 0)
        {
            int v = stack.Peek();
            bool found = false;

            while (g[v].Count > 0)
            {
                var (to, edgeId) = g[v][g[v].Count - 1];
                g[v].RemoveAt(g[v].Count - 1);

                if (!usedEdges.Contains(edgeId))
                {
                    usedEdges.Add(edgeId);
                    stack.Push(to);
                    found = true;
                    break;
                }
            }

            if (!found)
            {
                circuit.Add(stack.Pop());
            }
        }

        circuit.Reverse();
        return circuit;
    }
}
