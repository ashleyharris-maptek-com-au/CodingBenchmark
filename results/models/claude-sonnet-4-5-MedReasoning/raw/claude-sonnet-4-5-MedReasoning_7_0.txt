using namespace std;

const double EPSILON = 1e-9;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double t) const { return Vec3(x * t, y * t, z * t); }
    Vec3 operator/(double t) const { return Vec3(x / t, y / t, z / t); }

    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }

    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const { double l = length(); return l > EPSILON ? (*this) / l : Vec3(0, 0, 0); }
};

struct Triangle {
    int v[3];
    int meshId; // 0 for mesh A, 1 for mesh B
};

struct Mesh {
    vector<Vec3> vertices;
    vector<Triangle> faces;
};

// Hash function for Vec3
struct Vec3Hash {
    size_t operator()(const Vec3& v) const {
        size_t h1 = hash<double>()(v.x);
        size_t h2 = hash<double>()(v.y);
        size_t h3 = hash<double>()(v.z);
        return h1 ^ (h2 << 1) ^ (h3 << 2);
    }
};

struct Vec3Equal {
    bool operator()(const Vec3& a, const Vec3& b) const {
        return abs(a.x - b.x) < EPSILON && abs(a.y - b.y) < EPSILON && abs(a.z - b.z) < EPSILON;
    }
};

// Check if point is inside a mesh using ray casting
bool isPointInside(const Vec3& point, const Mesh& mesh) {
    // Ray direction (arbitrary)
    Vec3 rayDir(1.0, 0.0, 0.0);

    int intersections = 0;

    for (const auto& tri : mesh.faces) {
        const Vec3& v0 = mesh.vertices[tri.v[0]];
        const Vec3& v1 = mesh.vertices[tri.v[1]];
        const Vec3& v2 = mesh.vertices[tri.v[2]];

        Vec3 e1 = v1 - v0;
        Vec3 e2 = v2 - v0;
        Vec3 h = rayDir.cross(e2);
        double a = e1.dot(h);

        if (abs(a) < EPSILON) continue;

        double f = 1.0 / a;
        Vec3 s = point - v0;
        double u = f * s.dot(h);

        if (u < 0.0 || u > 1.0) continue;

        Vec3 q = s.cross(e1);
        double v = f * rayDir.dot(q);

        if (v < 0.0 || u + v > 1.0) continue;

        double t = f * e2.dot(q);

        if (t > EPSILON) {
            intersections++;
        }
    }

    return (intersections % 2) == 1;
}

// Compute triangle centroid
Vec3 triangleCentroid(const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    return (v0 + v1 + v2) / 3.0;
}

// Simple CSG Union: keep faces that are outside the other mesh
Mesh computeUnion(const Mesh& meshA, const Mesh& meshB) {
    Mesh result;
    unordered_map<Vec3, int, Vec3Hash, Vec3Equal> vertexMap;

    auto addVertex = [&](const Vec3& v) -> int {
        auto it = vertexMap.find(v);
        if (it != vertexMap.end()) {
            return it->second;
        }
        int idx = result.vertices.size();
        result.vertices.push_back(v);
        vertexMap[v] = idx;
        return idx;
    };

    // Check each face of mesh A
    for (const auto& tri : meshA.faces) {
        const Vec3& v0 = meshA.vertices[tri.v[0]];
        const Vec3& v1 = meshA.vertices[tri.v[1]];
        const Vec3& v2 = meshA.vertices[tri.v[2]];

        Vec3 centroid = triangleCentroid(v0, v1, v2);

        // Keep face if it's outside mesh B
        if (!isPointInside(centroid, meshB)) {
            Triangle newTri;
            newTri.v[0] = addVertex(v0);
            newTri.v[1] = addVertex(v1);
            newTri.v[2] = addVertex(v2);
            newTri.meshId = 0;
            result.faces.push_back(newTri);
        }
    }

    // Check each face of mesh B
    for (const auto& tri : meshB.faces) {
        const Vec3& v0 = meshB.vertices[tri.v[0]];
        const Vec3& v1 = meshB.vertices[tri.v[1]];
        const Vec3& v2 = meshB.vertices[tri.v[2]];

        Vec3 centroid = triangleCentroid(v0, v1, v2);

        // Keep face if it's outside mesh A
        if (!isPointInside(centroid, meshA)) {
            Triangle newTri;
            newTri.v[0] = addVertex(v0);
            newTri.v[1] = addVertex(v1);
            newTri.v[2] = addVertex(v2);
            newTri.meshId = 1;
            result.faces.push_back(newTri);
        }
    }

    return result;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh meshA, meshB;

    // Read mesh A vertices
    meshA.vertices.resize(VA);
    for (int i = 0; i < VA; i++) {
        cin >> meshA.vertices[i].x >> meshA.vertices[i].y >> meshA.vertices[i].z;
    }

    // Read mesh A faces
    meshA.faces.resize(FA);
    for (int i = 0; i < FA; i++) {
        cin >> meshA.faces[i].v[0] >> meshA.faces[i].v[1] >> meshA.faces[i].v[2];
        meshA.faces[i].meshId = 0;
    }

    // Read mesh B vertices
    meshB.vertices.resize(VB);
    for (int i = 0; i < VB; i++) {
        cin >> meshB.vertices[i].x >> meshB.vertices[i].y >> meshB.vertices[i].z;
    }

    // Read mesh B faces
    meshB.faces.resize(FB);
    for (int i = 0; i < FB; i++) {
        cin >> meshB.faces[i].v[0] >> meshB.faces[i].v[1] >> meshB.faces[i].v[2];
        meshB.faces[i].meshId = 1;
    }

    // Compute union
    Mesh result = computeUnion(meshA, meshB);

    // Output result
    cout << result.vertices.size() << " " << result.faces.size() << "\n";

    cout.precision(17);
    for (const auto& v : result.vertices) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    for (const auto& f : result.faces) {
        cout << f.v[0] << " " << f.v[1] << " " << f.v[2] << "\n";
    }

    return 0;
}
