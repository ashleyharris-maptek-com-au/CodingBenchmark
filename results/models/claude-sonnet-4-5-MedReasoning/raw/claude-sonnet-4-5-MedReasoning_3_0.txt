use std::collections::{HashMap, HashSet};
use std::io::{self, BufRead};
use std::f64::consts::PI;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let n = parts[0];
    let m = parts[1];

    let mut edges = Vec::with_capacity(m);
    let mut adj: HashMap<usize, HashSet<usize>> = HashMap::new();

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let edge: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();

        let u = edge[0];
        let v = edge[1];

        if u != v {
            edges.push((u, v));
            adj.entry(u).or_insert_with(HashSet::new).insert(v);
            adj.entry(v).or_insert_with(HashSet::new).insert(u);
        }
    }

    let positions = compute_layout(n, &edges, &adj);

    println!("{}", n);
    for i in 0..n {
        let (x, y) = positions[i];
        println!("{} {}", x, y);
    }
}

fn compute_layout(n: usize, edges: &[(usize, usize)], adj: &HashMap<usize, HashSet<usize>>) -> Vec<(f64, f64)> {
    if n <= 500 {
        force_directed_layout(n, edges, adj)
    } else if n <= 5000 {
        fast_force_directed_layout(n, edges, adj)
    } else if n <= 50000 {
        grid_layout(n)
    } else {
        simple_grid_layout(n)
    }
}

fn force_directed_layout(n: usize, edges: &[(usize, usize)], adj: &HashMap<usize, HashSet<usize>>) -> Vec<(f64, f64)> {
    let mut positions = initialize_positions(n);

    let iterations = if n <= 50 { 1000 } else if n <= 200 { 500 } else { 200 };
    let k = (1000.0 / (n as f64).sqrt()).max(10.0);
    let area = 1000.0 * 1000.0;
    let cooling_factor = 0.95;

    let mut temperature = 100.0;

    for iter in 0..iterations {
        let mut forces = vec![(0.0, 0.0); n];

        // Repulsive forces between all pairs
        for i in 0..n {
            for j in (i + 1)..n {
                let dx = positions[j].0 - positions[i].0;
                let dy = positions[j].1 - positions[i].1;
                let dist = (dx * dx + dy * dy).sqrt().max(0.01);

                let repulsion = k * k / dist;
                let fx = (dx / dist) * repulsion;
                let fy = (dy / dist) * repulsion;

                forces[i].0 -= fx;
                forces[i].1 -= fy;
                forces[j].0 += fx;
                forces[j].1 += fy;
            }
        }

        // Attractive forces for edges
        for &(u, v) in edges {
            let dx = positions[v].0 - positions[u].0;
            let dy = positions[v].1 - positions[u].1;
            let dist = (dx * dx + dy * dy).sqrt().max(0.01);

            let attraction = dist * dist / k;
            let fx = (dx / dist) * attraction;
            let fy = (dy / dist) * attraction;

            forces[u].0 += fx;
            forces[u].1 += fy;
            forces[v].0 -= fx;
            forces[v].1 -= fy;
        }

        // Apply forces
        for i in 0..n {
            let force_mag = (forces[i].0 * forces[i].0 + forces[i].1 * forces[i].1).sqrt();
            if force_mag > 0.01 {
                let displacement = force_mag.min(temperature);
                positions[i].0 += (forces[i].0 / force_mag) * displacement;
                positions[i].1 += (forces[i].1 / force_mag) * displacement;
            }
        }

        temperature *= cooling_factor;

        if iter % 50 == 0 && temperature < 0.1 {
            break;
        }
    }

    normalize_positions(&mut positions);
    positions
}

fn fast_force_directed_layout(n: usize, edges: &[(usize, usize)], adj: &HashMap<usize, HashSet<usize>>) -> Vec<(f64, f64)> {
    let mut positions = initialize_positions(n);

    let iterations = 100;
    let k = (1000.0 / (n as f64).sqrt()).max(5.0);
    let mut temperature = 50.0;
    let cooling_factor = 0.95;

    for _iter in 0..iterations {
        let mut forces = vec![(0.0, 0.0); n];

        // Simplified repulsion: only sample some pairs
        let sample_size = (n / 10).max(20).min(100);
        for i in 0..n {
            let step = (n / sample_size).max(1);
            for j in (0..n).step_by(step) {
                if i == j { continue; }

                let dx = positions[j].0 - positions[i].0;
                let dy = positions[j].1 - positions[i].1;
                let dist = (dx * dx + dy * dy).sqrt().max(0.01);

                let repulsion = k * k / dist;
                let fx = (dx / dist) * repulsion;
                let fy = (dy / dist) * repulsion;

                forces[i].0 -= fx;
                forces[i].1 -= fy;
            }
        }

        // Attractive forces for edges
        for &(u, v) in edges {
            let dx = positions[v].0 - positions[u].0;
            let dy = positions[v].1 - positions[u].1;
            let dist = (dx * dx + dy * dy).sqrt().max(0.01);

            let attraction = dist / k;
            let fx = dx * attraction;
            let fy = dy * attraction;

            forces[u].0 += fx;
            forces[u].1 += fy;
            forces[v].0 -= fx;
            forces[v].1 -= fy;
        }

        // Apply forces
        for i in 0..n {
            let force_mag = (forces[i].0 * forces[i].0 + forces[i].1 * forces[i].1).sqrt();
            if force_mag > 0.01 {
                let displacement = force_mag.min(temperature);
                positions[i].0 += (forces[i].0 / force_mag) * displacement;
                positions[i].1 += (forces[i].1 / force_mag) * displacement;
            }
        }

        temperature *= cooling_factor;
    }

    normalize_positions(&mut positions);
    positions
}

fn grid_layout(n: usize) -> Vec<(f64, f64)> {
    let mut positions = Vec::with_capacity(n);
    let grid_size = (n as f64).sqrt().ceil() as usize;

    let mut rng_state = 12345u64;
    let spacing = 1000.0 / grid_size as f64;

    for i in 0..n {
        let row = i / grid_size;
        let col = i % grid_size;

        // Add small random jitter
        rng_state = rng_state.wrapping_mul(1103515245).wrapping_add(12345);
        let jitter_x = ((rng_state >> 16) as f64 / 65536.0 - 0.5) * spacing * 0.3;
        rng_state = rng_state.wrapping_mul(1103515245).wrapping_add(12345);
        let jitter_y = ((rng_state >> 16) as f64 / 65536.0 - 0.5) * spacing * 0.3;

        positions.push((
            col as f64 * spacing + jitter_x,
            row as f64 * spacing + jitter_y,
        ));
    }

    normalize_positions(&mut positions);
    positions
}

fn simple_grid_layout(n: usize) -> Vec<(f64, f64)> {
    let mut positions = Vec::with_capacity(n);
    let grid_size = (n as f64).sqrt().ceil() as usize;
    let spacing = 1000.0 / grid_size as f64;

    for i in 0..n {
        let row = i / grid_size;
        let col = i % grid_size;
        positions.push((col as f64 * spacing, row as f64 * spacing));
    }

    normalize_positions(&mut positions);
    positions
}

fn initialize_positions(n: usize) -> Vec<(f64, f64)> {
    let mut positions = Vec::with_capacity(n);
    let mut rng_state = 12345u64;

    if n <= 100 {
        // Circular layout for small graphs
        for i in 0..n {
            let angle = 2.0 * PI * (i as f64) / (n as f64);
            let radius = 400.0;
            positions.push((
                500.0 + radius * angle.cos(),
                500.0 + radius * angle.sin(),
            ));
        }
    } else {
        // Random positions for larger graphs
        for _ in 0..n {
            rng_state = rng_state.wrapping_mul(1103515245).wrapping_add(12345);
            let x = ((rng_state >> 16) as f64 / 65536.0) * 1000.0;
            rng_state = rng_state.wrapping_mul(1103515245).wrapping_add(12345);
            let y = ((rng_state >> 16) as f64 / 65536.0) * 1000.0;
            positions.push((x, y));
        }
    }

    positions
}

fn normalize_positions(positions: &mut [(f64, f64)]) {
    if positions.is_empty() {
        return;
    }

    let mut min_x = positions[0].0;
    let mut max_x = positions[0].0;
    let mut min_y = positions[0].1;
    let mut max_y = positions[0].1;

    for &(x, y) in positions.iter() {
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }

    let width = max_x - min_x;
    let height = max_y - min_y;
    let scale = if width.max(height) > 0.0 {
        900.0 / width.max(height)
    } else {
        1.0
    };

    for pos in positions.iter_mut() {
        pos.0 = (pos.0 - min_x) * scale + 50.0;
        pos.1 = (pos.1 - min_y) * scale + 50.0;
    }
}
