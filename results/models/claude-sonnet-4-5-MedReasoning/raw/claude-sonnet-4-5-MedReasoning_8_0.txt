use std::collections::{VecDeque, HashMap};
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read dimensions
    let first_line = lines.next().unwrap().unwrap();
    let dims: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let (h, w) = (dims[0], dims[1]);

    // Read maze and find start/goal positions
    let mut maze = Vec::with_capacity(h);
    let mut start = (0, 0);
    let mut goal = (0, 0);

    for y in 0..h {
        let row = lines.next().unwrap().unwrap();
        let bytes = row.into_bytes();
        for x in 0..bytes.len().min(w) {
            if bytes[x] == b'A' {
                start = (y, x);
            } else if bytes[x] == b'B' {
                goal = (y, x);
            }
        }
        maze.push(bytes);
    }

    // BFS pathfinding
    if let Some(path) = bfs(&maze, h, w, start, goal) {
        println!("{}", path.len());
        for (y, x) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn bfs(maze: &[Vec<u8>], h: usize, w: usize, start: (usize, usize), goal: (usize, usize)) -> Option<Vec<(usize, usize)>> {
    let mut queue = VecDeque::new();
    let mut visited = vec![false; h * w];
    let mut parent: HashMap<(usize, usize), (usize, usize)> = HashMap::new();

    queue.push_back(start);
    visited[start.0 * w + start.1] = true;

    while let Some((y, x)) = queue.pop_front() {
        if (y, x) == goal {
            // Reconstruct path from goal to start
            let mut path = Vec::new();
            let mut current = goal;

            loop {
                path.push(current);
                if current == start {
                    break;
                }
                current = *parent.get(&current).unwrap();
            }
            path.reverse();

            return Some(path);
        }

        // Explore 4 directions: right, down, left, up
        let directions = [(0i32, 1i32), (1, 0), (0, -1), (-1, 0)];

        for (dy, dx) in directions {
            let ny = y as i32 + dy;
            let nx = x as i32 + dx;

            if ny >= 0 && (ny as usize) < h && nx >= 0 && (nx as usize) < w {
                let ny = ny as usize;
                let nx = nx as usize;
                let idx = ny * w + nx;

                if !visited[idx] && maze[ny][nx] != b'#' {
                    visited[idx] = true;
                    parent.insert((ny, nx), (y, x));
                    queue.push_back((ny, nx));
                }
            }
        }
    }

    None
}
