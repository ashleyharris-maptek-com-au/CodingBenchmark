    static (int[], int[]) Dijkstra(int n, List<(int to, int weight, int edgeId)>[] graph, int start)
    {
        var dist = new int[n];
        var parent = new int[n];
        Array.Fill(dist, int.MaxValue);
        Array.Fill(parent, -1);
        dist[start] = 0;

        var pq = new PriorityQueue<int, int>();
        pq.Enqueue(start, 0);
        var visited = new bool[n];

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();
            if (visited[u]) continue;
            visited[u] = true;

            foreach (var (v, w, _) in graph[u])
            {
                if (dist[u] != int.MaxValue && dist[u] + w < dist[v])
                {
                    dist[v] = dist[u] + w;
                    parent[v] = u;
                    pq.Enqueue(v, dist[v]);
                }
            }
        }

        self.obstacles[y][x] = true;
    }

    fn finalize(&mut self) {
        self.target_cells = self.width * self.height - self.count_obstacles();
    }

    fn count_obstacles(&self) -> usize {
        self.obstacles.iter().flatten().filter(|&&b| b).count()
    }

    fn is_valid(&self, x: i32, y: i32) -> bool {
        x >= 0 && y >= 0 &&
        (x as usize) < self.width &&
        (y as usize) < self.height &&
        !self.obstacles[y as usize][x as usize]
    }

    fn get_neighbors(&self, x: i32, y: i32) -> Vec<(i32, i32)> {
        let dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)];
        dirs.iter()
            .map(|(dx, dy)| (x + dx, y + dy))
            .filter(|&(nx, ny)| self.is_valid(nx, ny))
            .collect()
    }
}

struct Solver {
    grid: Grid,
    visited: Vec<Vec<bool>>,
    path: Vec<(i32, i32)>,
    best_path: Option<Vec<(i32, i32)>>,
}

impl Solver {
    fn new(grid: Grid) -> Self {
        let visited = vec![vec![false; grid.width]; grid.height];
        Solver {
            grid,
            visited,
            path: Vec::new(),
            best_path: None,
        }
    }

    fn solve(&mut self) -> Option<Vec<(i32, i32)>> {
        // Check if start is valid
        if !self.grid.is_valid(0, 0) {
            return None;
        }

        // Quick connectivity check
        if !self.is_connected() {
            return None;
        }

        // Start DFS from (0, 0)
        self.visited[0][0] = true;
        self.path.push((0, 0));
        self.dfs(0, 0);
        self.best_path.clone()
    }

    fn dfs(&mut self, x: i32, y: i32) -> bool {
        // Check if we've visited all cells
        if self.path.len() == self.grid.target_cells {
            if self.grid.require_cycle {
                // Check if current cell is adjacent to (0, 0)
                let neighbors = self.grid.get_neighbors(x, y);
                if neighbors.contains(&(0, 0)) {
                    self.best_path = Some(self.path.clone());
                    return true;
                }
            } else {
                self.best_path = Some(self.path.clone());
                return true;
            }
        }

        // Pruning: check if remaining cells can be reached
        if !self.can_complete(x, y) {
            return false;
        }

        // Try all neighbors
        let neighbors = self.grid.get_neighbors(x, y);

        // Sort neighbors by heuristic (fewest unvisited neighbors first - Warnsdorff's rule)
        let mut scored_neighbors: Vec<_> = neighbors.iter()
            .filter(|&&(nx, ny)| !self.visited[ny as usize][nx as usize])
            .map(|&(nx, ny)| {
                let unvisited_count = self.grid.get_neighbors(nx, ny)
                    .iter()
                    .filter(|&&(nnx, nny)| !self.visited[nny as usize][nnx as usize])
                    .count();
                (nx, ny, unvisited_count)
            })
            .collect();

        scored_neighbors.sort_by_key(|&(_, _, count)| count);

        for (nx, ny, _) in scored_neighbors {
            self.visited[ny as usize][nx as usize] = true;
            self.path.push((nx, ny));

            if self.dfs(nx, ny) {
                return true;
            }

            self.path.pop();
            self.visited[ny as usize][nx as usize] = false;
        }

        false
    }

    fn can_complete(&self, x: i32, y: i32) -> bool {
        // Check if all unvisited cells are still reachable
        let unvisited_count = self.grid.target_cells - self.path.len();

        if unvisited_count == 0 {
            return true;
        }

        // BFS to check connectivity of unvisited cells
        let mut queue = VecDeque::new();
        let mut temp_visited = self.visited.clone();
        let mut reachable = 0;

        // Start from current position
        queue.push_back((x, y));

        while let Some((cx, cy)) = queue.pop_front() {
            for (nx, ny) in self.grid.get_neighbors(cx, cy) {
                if !temp_visited[ny as usize][nx as usize] {
                    temp_visited[ny as usize][nx as usize] = true;
                    reachable += 1;
                    queue.push_back((nx, ny));
                }
            }
        }

        reachable >= unvisited_count
    }

    fn is_connected(&self) -> bool {
        // Check if all non-obstacle cells are connected
        let mut visited = vec![vec![false; self.grid.width]; self.grid.height];
        let mut queue = VecDeque::new();
        let mut count = 0;

        // Find starting position
        if !self.grid.is_valid(0, 0) {
            return false;
        }

        queue.push_back((0, 0));
        visited[0][0] = true;
        count += 1;

        while let Some((x, y)) = queue.pop_front() {
            for (nx, ny) in self.grid.get_neighbors(x, y) {
                if !visited[ny as usize][nx as usize] {
                    visited[ny as usize][nx as usize] = true;
                    count += 1;
                    queue.push_back((nx, ny));
                }
            }
        }

        count == self.grid.target_cells
    }
}

class OrbitalTSPSolver
{
    private OrbitalState start;
    private OrbitalState[] stations;
    private double mu;
    private double[,] deltaVMatrix;

    public OrbitalTSPSolver(OrbitalState start, OrbitalState[] stations, double mu)
    {
        this.start = start;
        this.stations = stations;
        this.mu = mu;
        this.deltaVMatrix = new double[stations.Length + 1, stations.Length + 1];
        PrecomputeDeltaV();
    }

    private void PrecomputeDeltaV()
    {
        // Index 0 represents start position
        for (int i = 0; i < stations.Length; i++)
        {
            deltaVMatrix[0, i + 1] = EstimateDeltaV(start, stations[i]);
        }

        for (int i = 0; i < stations.Length; i++)
        {
            for (int j = 0; j < stations.Length; j++)
            {
                if (i != j)
                {
                    deltaVMatrix[i + 1, j + 1] = EstimateDeltaV(stations[i], stations[j]);
                }
            }
        }
    }

    private double EstimateDeltaV(OrbitalState from, OrbitalState to)
    {
        // Simplified delta-V estimation using Hohmann-like transfer approximation
        // and rendezvous delta-V

        double r1 = from.Position.Magnitude;
        double r2 = to.Position.Magnitude;

        // Orbital velocities
        double v1 = from.Velocity.Magnitude;
        double v2 = to.Velocity.Magnitude;

        // Simple Hohmann transfer approximation
        double vCirc1 = Math.Sqrt(mu / r1);
        double vCirc2 = Math.Sqrt(mu / r2);

        // Transfer orbit
        double a_transfer = (r1 + r2) / 2.0;
        double v_transfer_1 = Math.Sqrt(mu * (2.0 / r1 - 1.0 / a_transfer));
        double v_transfer_2 = Math.Sqrt(mu * (2.0 / r2 - 1.0 / a_transfer));

        // Delta-V for departure and arrival
        double dv1 = Math.Abs(v_transfer_1 - vCirc1);
        double dv2 = Math.Abs(vCirc2 - v_transfer_2);

        // Add position and velocity difference penalty
        double positionDiff = (to.Position - from.Position).Magnitude;
        double velocityDiff = (to.Velocity - from.Velocity).Magnitude;

        // Combined estimate with phasing cost
        double phasingCost = Math.Min(positionDiff / 10000.0, 2.0); // Normalize
        double velocityPenalty = velocityDiff * 0.5;

        return dv1 + dv2 + phasingCost + velocityPenalty;
    }

        int n = stations.Length;

        if (n <= 8)

            return SolveBruteForce();
        }
        else

            // For larger N, use nearest neighbor + 2-opt
            return SolveHeuristic();

    private int[] SolveBruteForce()
    {
        int n = stations.Length;
        var indices = Enumerable.Range(0, n).ToArray();
        var bestOrder = new int[n];

                bestCost = cost;
                Array.Copy(perm, bestOrder, n);

        }

        return bestOrder;

    private int[] NearestNeighborTour()
    {
        int n = stations.Length;
        var tour = new List<int>();
        var visited = new bool[n];
        int current = -1; // Start from launch position (index 0 in matrix)

        for (int i = 0; i < n; i++)
        {
            int next = -1;
            double minDeltaV = double.MaxValue;

            for (int j = 0; j < n; j++)
            {
                if (!visited[j])
                {
                    double dv = deltaVMatrix[current + 1, j + 1];
                    if (dv < minDeltaV)
                    {
                        minDeltaV = dv;
                        next = j;
                    }
                }
            }

            visited[next] = true;
            tour.Add(next);
            current = next;
        }

        return tour.ToArray();
    }

    private int[] TwoOptImprove(int[] tour)
    {
        int n = tour.Length;
        bool improved = true;
        var currentTour = (int[])tour.Clone();

        while (improved)
        {
            improved = false;

            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    double currentCost = EdgeCost(currentTour, i, i + 1) + EdgeCost(currentTour, j, (j + 1) % n);

                    // Reverse segment [i+1, j]
                    Array.Reverse(currentTour, i + 1, j - i);
                    double newCost = EdgeCost(currentTour, i, i + 1) + EdgeCost(currentTour, j, (j + 1) % n);

                    if (newCost < currentCost)
                    {
                        improved = true;
                        break;
                    }
                    else
                    {
                        // Reverse back
                        Array.Reverse(currentTour, i + 1, j - i);
                    }
                }
                if (improved) break;
            }
        }

        return currentTour;
    }

    private double EdgeCost(int[] tour, int idx1, int idx2)
    {
        int n = tour.Length;
        int from = idx1 < 0 ? -1 : (idx1 >= n ? -1 : tour[idx1]);
        int to = idx2 >= n ? -1 : tour[idx2 % n];

        if (to == -1) return 0;

        return deltaVMatrix[from + 1, to + 1];
    }

    private double CalculateTourCost(int[] tour)
    {
        double cost = deltaVMatrix[0, tour[0] + 1]; // Start to first station

        for (int i = 0; i < tour.Length - 1; i++)
        {
            cost += deltaVMatrix[tour[i] + 1, tour[i + 1] + 1];
        }

        return cost;
    }

    private IEnumerable<int[]> Permutations(int[] array)
    {
        int n = array.Length;
        var indices = new int[n];
        yield return (int[])array.Clone();

        int i = 0;
        while (i < n)
        {
            if (indices[i] < i)
            {
                int swapIdx = (i % 2 == 0) ? 0 : indices[i];
                int temp = array[swapIdx];
                array[swapIdx] = array[i];
                array[i] = temp;

                yield return (int[])array.Clone();

                indices[i]++;
                i = 0;
            }
            else
            {
                indices[i] = 0;
                i++;
            }
        }
    }

bool isPointInside(const Vec3& point, const Mesh& mesh) {
    // Ray direction (arbitrary)
    Vec3 rayDir(1.0, 0.0, 0.0);

    int intersections = 0;

    for (const auto& tri : mesh.faces) {
        const Vec3& v0 = mesh.vertices[tri.v[0]];
        const Vec3& v1 = mesh.vertices[tri.v[1]];
        const Vec3& v2 = mesh.vertices[tri.v[2]];

        Vec3 e1 = v1 - v0;
        Vec3 e2 = v2 - v0;
        Vec3 h = rayDir.cross(e2);
        double a = e1.dot(h);

        if (abs(a) < EPSILON) continue;

        double f = 1.0 / a;
        Vec3 s = point - v0;
        double u = f * s.dot(h);

        if (u < 0.0 || u > 1.0) continue;

        Vec3 q = s.cross(e1);
        double v = f * rayDir.dot(q);

        if (v < 0.0 || u + v > 1.0) continue;

        double t = f * e2.dot(q);

        if (t > EPSILON) {
            intersections++;
        }
    }

    return (intersections % 2) == 1;
}
