// =================== Time Control ===================

static double timeLimitSeconds = 5.0;
static chrono::steady_clock::time_point startTime;
static atomic<bool> timeUp(false);

inline bool isTimeUp() {
    if (timeUp.load(memory_order_relaxed)) return true;
    auto now = chrono::steady_clock::now();
    double elapsed = chrono::duration<double>(now - startTime).count();
    if (elapsed >= timeLimitSeconds * 0.99) {
        timeUp.store(true, memory_order_relaxed);
        return true;
    }
    return false;
}

// =================== Board Representation ===================

enum Piece {
    EMPTY = 0,
    WP = 1, WN, WB, WR, WQ, WK,
    BP = -1, BN = -2, BB = -3, BR = -4, BQ = -5, BK = -6
};

inline bool isWhite(int p) { return p > 0; }
inline bool isBlack(int p) { return p < 0; }
inline int pieceType(int p) { return p > 0 ? p : -p; }

struct Move {
    int from;
    int to;
    int piece;      // moving piece
    int captured;   // captured piece (if any)
    int promo;      // promotion piece (WP..WK / BP..BK or 0)
    bool isCapture;
    bool isEP;
    bool isCastle;
};

struct Board {
    int squares[64];
    int sideToMove; // +1 white, -1 black
    bool wCastleK, wCastleQ, bCastleK, bCastleQ;
    int epSquare;   // -1 if none, else 0..63
    int kingSqWhite;
    int kingSqBlack;

    Board() {
        for (int i = 0; i < 64; ++i) squares[i] = EMPTY;
        sideToMove = 1;
        wCastleK = wCastleQ = bCastleK = bCastleQ = false;
        epSquare = -1;
        kingSqWhite = kingSqBlack = -1;
    }
};

// Precomputed knight and king moves (offsets)
const int knightOffsets[8] = { 17, 15, 10, 6, -17, -15, -10, -6 };
const int kingOffsets[8]   = { 8, -8, 1, -1, 9, 7, -9, -7 };

// =================== Utility ===================

inline int fileOf(int sq) { return sq & 7; }
inline int rankOf(int sq) { return sq >> 3; }

inline bool onBoard(int sq) { return sq >= 0 && sq < 64; }

inline bool sameFile(int a, int b) { return fileOf(a) == fileOf(b); }
inline bool sameRank(int a, int b) { return rankOf(a) == rankOf(b); }

inline int makeSquare(int file, int rank) { return rank * 8 + file; }

// =================== FEN Parsing ===================

bool parseFEN(const string& fen, Board& b) {
    stringstream ss(fen);
    string boardPart, sidePart, castlePart, epPart;
    int halfmove, fullmove;
    if (!(ss >> boardPart >> sidePart >> castlePart >> epPart >> halfmove >> fullmove)) {
        return false;
    }

    int sq = 56; // start at a8
    for (char c : boardPart) {
        if (c == '/') {
            sq -= 16; // go to next rank
        } else if (isdigit(static_cast<unsigned char>(c))) {
            int empty = c - '0';
            for (int i = 0; i < empty; ++i) {
                b.squares[sq++] = EMPTY;
            }
        } else {
            int p = EMPTY;
            switch (c) {
                case 'P': p = WP; break;
                case 'N': p = WN; break;
                case 'B': p = WB; break;
                case 'R': p = WR; break;
                case 'Q': p = WQ; break;
                case 'K': p = WK; break;
                case 'p': p = BP; break;
                case 'n': p = BN; break;
                case 'b': p = BB; break;
                case 'r': p = BR; break;
                case 'q': p = BQ; break;
                case 'k': p = BK; break;
                default: return false;
            }
            b.squares[sq] = p;
            if (p == WK) b.kingSqWhite = sq;
            else if (p == BK) b.kingSqBlack = sq;
            sq++;
        }
    }

    b.sideToMove = (sidePart == "w") ? 1 : -1;

    b.wCastleK = b.wCastleQ = b.bCastleK = b.bCastleQ = false;
    if (castlePart != "-") {
        for (char c : castlePart) {
            if (c == 'K') b.wCastleK = true;
            else if (c == 'Q') b.wCastleQ = true;
            else if (c == 'k') b.bCastleK = true;
            else if (c == 'q') b.bCastleQ = true;
        }
    }

    if (epPart == "-") b.epSquare = -1;
    else {
        if (epPart.size() != 2) return false;
        int file = epPart[0] - 'a';
        int rank = epPart[1] - '1';
        if (file < 0 || file > 7 || rank < 0 || rank > 7) return false;
        b.epSquare = makeSquare(file, rank);
    }

    return true;
}

// =================== Attack Detection ===================

bool isSquareAttacked(const Board& b, int sq, int bySide) {
    // Pawns
    if (bySide == 1) { // white attacks
        int r = rankOf(sq), f = fileOf(sq);
        if (r > 0) {
            if (f > 0) {
                int s = sq - 9;
                if (onBoard(s) && b.squares[s] == WP) return true;
            }
            if (f < 7) {
                int s = sq - 7;
                if (onBoard(s) && b.squares[s] == WP) return true;
            }
        }
    } else { // black attacks
        int r = rankOf(sq), f = fileOf(sq);
        if (r < 7) {
            if (f > 0) {
                int s = sq + 7;
                if (onBoard(s) && b.squares[s] == BP) return true;
            }
            if (f < 7) {
                int s = sq + 9;
                if (onBoard(s) && b.squares[s] == BP) return true;
            }
        }
    }

    // Knights
    for (int off : knightOffsets) {
        int to = sq + off;
        if (!onBoard(to)) continue;
        int df = abs(fileOf(to) - fileOf(sq));
        int dr = abs(rankOf(to) - rankOf(sq));
        if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
        int p = b.squares[to];
        if (p == EMPTY) continue;
        if (bySide == 1 && p == WN) return true;
        if (bySide == -1 && p == BN) return true;
    }

    // Bishops / Queens (diagonals)
    const int bishopDirs[4] = { 9, 7, -9, -7 };
    for (int d : bishopDirs) {
        int cur = sq + d;
        while (onBoard(cur) && abs(fileOf(cur) - fileOf(cur - d)) == 1) {
            int p = b.squares[cur];
            if (p != EMPTY) {
                if (bySide == 1 && (p == WB || p == WQ)) return true;
                if (bySide == -1 && (p == BB || p == BQ)) return true;
                break;
            }
            cur += d;
        }
    }

    // Rooks / Queens (orthogonals)
    const int rookDirs[4] = { 8, -8, 1, -1 };
    for (int d : rookDirs) {
        int cur = sq + d;
        while (onBoard(cur) && (
            (d == 1 || d == -1) ? sameRank(cur, cur - d) : true)) {
            int p = b.squares[cur];
            if (p != EMPTY) {
                if (bySide == 1 && (p == WR || p == WQ)) return true;
                if (bySide == -1 && (p == BR || p == BQ)) return true;
                break;
            }
            cur += d;
        }
    }

    // King
    for (int off : kingOffsets) {
        int to = sq + off;
        if (!onBoard(to)) continue;
        if (abs(fileOf(to) - fileOf(sq)) > 1 ||
            abs(rankOf(to) - rankOf(sq)) > 1) continue;
        int p = b.squares[to];
        if (bySide == 1 && p == WK) return true;
        if (bySide == -1 && p == BK) return true;
    }

    return false;
}

bool inCheck(const Board& b, int side) {
    int kingSq = (side == 1) ? b.kingSqWhite : b.kingSqBlack;
    if (kingSq < 0) return false; // should not happen
    return isSquareAttacked(b, kingSq, -side);
}

// =================== Move Generation ===================

void addMove(vector<Move>& moves, int from, int to, int piece,
             int captured = EMPTY, int promo = 0,
             bool isEP = false, bool isCastle = false) {
    Move m;
    m.from = from;
    m.to = to;
    m.piece = piece;
    m.captured = captured;
    m.promo = promo;
    m.isCapture = (captured != EMPTY);
    m.isEP = isEP;
    m.isCastle = isCastle;
    moves.push_back(m);
}

void generatePseudoLegalMoves(const Board& b, vector<Move>& moves) {
    moves.clear();
    int side = b.sideToMove;
    bool white = (side == 1);

    for (int sq = 0; sq < 64; ++sq) {
        int p = b.squares[sq];
        if (p == EMPTY) continue;
        if (white && !isWhite(p)) continue;
        if (!white && !isBlack(p)) continue;

        int pt = pieceType(p);

        if (pt == 1) { // Pawn
            int dir = white ? 8 : -8;
            int startRank = white ? 1 : 6;
            int promoRank = white ? 6 : 1;
            int rank = rankOf(sq);
            int file = fileOf(sq);

            int oneForward = sq + dir;
            if (onBoard(oneForward) && b.squares[oneForward] == EMPTY) {
                if (rank == promoRank) {
                    // promotions
                    int promoPieces[4] = { white ? WQ : BQ, white ? WR : BR,
                                           white ? WB : BB, white ? WN : BN };
                    for (int pr : promoPieces) {
                        addMove(moves, sq, oneForward, p, EMPTY, pr);
                    }
                } else {
                    addMove(moves, sq, oneForward, p);
                    // double push
                    if (rank == startRank) {
                        int twoForward = oneForward + dir;
                        if (onBoard(twoForward) && b.squares[twoForward] == EMPTY) {
                            addMove(moves, sq, twoForward, p);
                        }
                    }
                }
            }

            // Captures
            int capDirs[2] = { dir + 1, dir - 1 };
            for (int d : capDirs) {
                int to = sq + d;
                if (!onBoard(to)) continue;
                if (abs(fileOf(to) - file) != 1) continue;
                int target = b.squares[to];
                if (target != EMPTY && ((white && isBlack(target)) || (!white && isWhite(target)))) {
                    if (rank == promoRank) {
                        int promoPieces[4] = { white ? WQ : BQ, white ? WR : BR,
                                               white ? WB : BB, white ? WN : BN };
                        for (int pr : promoPieces) {
                            addMove(moves, sq, to, p, target, pr);
                        }
                    } else {
                        addMove(moves, sq, to, p, target);
                    }
                }
                // En passant
                if (to == b.epSquare) {
                    int capSq = white ? to - 8 : to + 8;
                    int captured = b.squares[capSq];
                    if ((white && captured == BP) || (!white && captured == WP)) {
                        addMove(moves, sq, to, p, captured, 0, true);
                    }
                }
            }
        } else if (pt == 2) { // Knight
            for (int off : knightOffsets) {
                int to = sq + off;
                if (!onBoard(to)) continue;
                int df = abs(fileOf(to) - fileOf(sq));
                int dr = abs(rankOf(to) - rankOf(sq));
                if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
                int target = b.squares[to];
                if (target == EMPTY) {
                    addMove(moves, sq, to, p);
                } else if ((white && isBlack(target)) || (!white && isWhite(target))) {
                    addMove(moves, sq, to, p, target);
                }
            }
        } else if (pt == 3 || pt == 4 || pt == 5) { // B, R, Q
            const int bishopDirs[4] = { 9, 7, -9, -7 };
            const int rookDirs[4] = { 8, -8, 1, -1 };
            const int* dirs = nullptr;
            int dirCount = 0;
            if (pt == 3) { dirs = bishopDirs; dirCount = 4; }
            else if (pt == 4) { dirs = rookDirs; dirCount = 4; }
            else { static int queenDirs[8] = {8,-8,1,-1,9,7,-9,-7}; dirs = queenDirs; dirCount = 8; }

            for (int di = 0; di < dirCount; ++di) {
                int d = dirs[di];
                int cur = sq + d;
                while (onBoard(cur) && (
                    (d == 1 || d == -1) ? sameRank(cur, cur - d) : true)) {
                    int target = b.squares[cur];
                    if (target == EMPTY) {
                        addMove(moves, sq, cur, p);
                    } else {
                        if ((white && isBlack(target)) || (!white && isWhite(target))) {
                            addMove(moves, sq, cur, p, target);
                        }
                        break;
                    }
                    cur += d;
                }
            }
        } else if (pt == 6) { // King
            for (int off : kingOffsets) {
                int to = sq + off;
                if (!onBoard(to)) continue;
                if (abs(fileOf(to) - fileOf(sq)) > 1 ||
                    abs(rankOf(to) - rankOf(sq)) > 1) continue;
                int target = b.squares[to];
                if (target == EMPTY) {
                    addMove(moves, sq, to, p);
                } else if ((white && isBlack(target)) || (!white && isWhite(target))) {
                    addMove(moves, sq, to, p, target);
                }
            }

            // Castling
            if (white) {
                if (b.wCastleK) {
                    // squares f1 (5), g1 (6) must be empty and not attacked, and king not in check
                    if (b.squares[makeSquare(5,0)] == EMPTY &&
                        b.squares[makeSquare(6,0)] == EMPTY) {
                        if (!inCheck(b, 1) &&
                            !isSquareAttacked(b, makeSquare(5,0), -1) &&
                            !isSquareAttacked(b, makeSquare(6,0), -1)) {
                            addMove(moves, sq, makeSquare(6,0), p, EMPTY, 0, false, true);
                        }
                    }
                }
                if (b.wCastleQ) {
                    // squares b1(1), c1(2), d1(3) empty, c1 and d1 not attacked, and king not in check
                    if (b.squares[makeSquare(1,0)] == EMPTY &&
                        b.squares[makeSquare(2,0)] == EMPTY &&
                        b.squares[makeSquare(3,0)] == EMPTY) {
                        if (!inCheck(b, 1) &&
                            !isSquareAttacked(b, makeSquare(3,0), -1) &&
                            !isSquareAttacked(b, makeSquare(2,0), -1)) {
                            addMove(moves, sq, makeSquare(2,0), p, EMPTY, 0, false, true);
                        }
                    }
                }
            } else {
                if (b.bCastleK) {
                    // f8(61), g8(62)
                    if (b.squares[makeSquare(5,7)] == EMPTY &&
                        b.squares[makeSquare(6,7)] == EMPTY) {
                        if (!inCheck(b, -1) &&
                            !isSquareAttacked(b, makeSquare(5,7), 1) &&
                            !isSquareAttacked(b, makeSquare(6,7), 1)) {
                            addMove(moves, sq, makeSquare(6,7), p, EMPTY, 0, false, true);
                        }
                    }
                }
                if (b.bCastleQ) {
                    // b8(57), c8(58), d8(59)
                    if (b.squares[makeSquare(1,7)] == EMPTY &&
                        b.squares[makeSquare(2,7)] == EMPTY &&
                        b.squares[makeSquare(3,7)] == EMPTY) {
                            if (!inCheck(b, -1) &&
                                !isSquareAttacked(b, makeSquare(3,7), 1) &&
                                !isSquareAttacked(b, makeSquare(2,7), 1)) {
                                addMove(moves, sq, makeSquare(2,7), p, EMPTY, 0, false, true);
                            }
                    }
                }
            }
        }
    }
}

void makeMove(Board& b, const Move& m, Board& nb) {
    nb = b; // copy
    int from = m.from;
    int to = m.to;
    int p = nb.squares[from];
    int side = nb.sideToMove;
    bool white = (side == 1);

    nb.epSquare = -1;

    // Move piece
    nb.squares[from] = EMPTY;

    // Handle en passant capture
    if (m.isEP) {
        int capSq = white ? to - 8 : to + 8;
        nb.squares[capSq] = EMPTY;
    }

    // Castling: move rook as well
    if (m.isCastle && pieceType(p) == 6) {
        if (white) {
            if (to == makeSquare(6,0)) { // O-O
                nb.squares[to] = p;
                nb.squares[makeSquare(5,0)] = WR;
                nb.squares[makeSquare(7,0)] = EMPTY;
            } else if (to == makeSquare(2,0)) { // O-O-O
                nb.squares[to] = p;
                nb.squares[makeSquare(3,0)] = WR;
                nb.squares[makeSquare(0,0)] = EMPTY;
            }
        } else {
            if (to == makeSquare(6,7)) { // O-O
                nb.squares[to] = p;
                nb.squares[makeSquare(5,7)] = BR;
                nb.squares[makeSquare(7,7)] = EMPTY;
            } else if (to == makeSquare(2,7)) { // O-O-O
                nb.squares[to] = p;
                nb.squares[makeSquare(3,7)] = BR;
                nb.squares[makeSquare(0,7)] = EMPTY;
            }
        }
    } else {
        nb.squares[to] = p;
    }

    // Promotion
    if (m.promo != 0) {
        nb.squares[to] = m.promo;
    }

    // Update king locations
    if (p == WK) nb.kingSqWhite = to;
    else if (p == BK) nb.kingSqBlack = to;

    // Update castling rights if king or rook moves or rook captured
    if (b.wCastleK || b.wCastleQ) {
        if (p == WK) {
            nb.wCastleK = nb.wCastleQ = false;
        } else if (p == WR) {
            if (from == makeSquare(7,0)) nb.wCastleK = false;
            else if (from == makeSquare(0,0)) nb.wCastleQ = false;
        }
        if (m.captured != EMPTY && isBlack(m.captured)) {
            if (to == makeSquare(7,7)) nb.bCastleK = false;
            else if (to == makeSquare(0,7)) nb.bCastleQ = false;
        }
    }
    if (b.bCastleK || b.bCastleQ) {
        if (p == BK) {
            nb.bCastleK = nb.bCastleQ = false;
        } else if (p == BR) {
            if (from == makeSquare(7,7)) nb.bCastleK = false;
            else if (from == makeSquare(0,7)) nb.bCastleQ = false;
        }
        if (m.captured != EMPTY && isWhite(m.captured)) {
            if (to == makeSquare(7,0)) nb.wCastleK = false;
            else if (to == makeSquare(0,0)) nb.wCastleQ = false;
        }
    }

    // Set en passant square on double pawn push
    if (pieceType(p) == 1) {
        int diff = to - from;
        if (diff == 16 || diff == -16) {
            nb.epSquare = white ? from + 8 : from - 8;
        }
    }

    // Switch side
    nb.sideToMove = -side;
}

// Generate legal moves (filter pseudo-legal by king safety)
void generateLegalMoves(const Board& b, vector<Move>& legal) {
    vector<Move> pseudo;
    generatePseudoLegalMoves(b, pseudo);
    legal.clear();
    Board nb;
    for (const Move& m : pseudo) {
        makeMove(b, m, nb);
        if (!inCheck(nb, b.sideToMove)) {
            legal.push_back(m);
        }
    }
}

int evaluate(const Board& b) {
    int score = 0;
    for (int sq = 0; sq < 64; ++sq) {
        int p = b.squares[sq];
        if (p == EMPTY) continue;
        int pt = pieceType(p);
        int val = pieceValues[pt];
        int psq = 0;
        int idx = sq;
        if (isWhite(p)) {
            // use table as is
            switch (pt) {
                case 1: psq = pawnTable[idx]; break;
                case 2: psq = knightTable[idx]; break;
                case 3: psq = bishopTable[idx]; break;
                case 4: psq = rookTable[idx]; break;
                case 5: psq = queenTable[idx]; break;
                case 6: psq = kingTable[idx]; break;
            }
            score += val + psq;
        } else {
            // mirror for black
            int mir = (7 - rankOf(sq)) * 8 + fileOf(sq);
            switch (pt) {
                case 1: psq = pawnTable[mir]; break;
                case 2: psq = knightTable[mir]; break;
                case 3: psq = bishopTable[mir]; break;
                case 4: psq = rookTable[mir]; break;
                case 5: psq = queenTable[mir]; break;
                case 6: psq = kingTable[mir]; break;
            }
            score -= val + psq;
        }
    }

    // side to move bonus
    score += (b.sideToMove == 1 ? 10 : -10);
    return score;
}

// =================== Search ===================

static const int INF = 1000000000;
static uint64_t nodes = 0;

int negamax(Board& b, int depth, int alpha, int beta) {
    if (isTimeUp()) return evaluate(b);

    if (depth == 0) {
        return evaluate(b);
    }

    nodes++;

    vector<Move> moves;
    generateLegalMoves(b, moves);

    if (moves.empty()) {
        // checkmate or stalemate
        if (inCheck(b, b.sideToMove)) {
            return -INF + 100; // mate
        } else {
            return 0; // draw
        }
    }

    // Simple move ordering: captures first
    sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
        if (a.isCapture != b.isCapture) return a.isCapture > b.isCapture;
        if (a.promo != 0 || b.promo != 0) return a.promo != 0;
        return false;
    });

    int best = -INF;

    for (const Move& m : moves) {
        Board nb;
        makeMove(b, m, nb);
        int score = -negamax(nb, depth - 1, -beta, -alpha);
        if (isTimeUp()) return evaluate(b);
        if (score > best) best = score;
        if (score > alpha) {
            alpha = score;
            if (alpha >= beta) break; // beta cutoff
        }
    }

    return best;
}

Move iterativeDeepening(Board& b, int maxDepth) {
    vector<Move> rootMoves;
    generateLegalMoves(b, rootMoves);
    if (rootMoves.empty()) {
        // dummy move; position is terminal
        Move m{};
        if (b.sideToMove == 1) {
            m.from = makeSquare(4,0);
            m.to = makeSquare(6,0);
            m.piece = WK;
            m.isCastle = true;
        } else {
            m.from = makeSquare(4,7);
            m.to = makeSquare(6,7);
            m.piece = BK;
            m.isCastle = true;
        }
        return m;
    }

    Move bestMove = rootMoves[0];
    int bestScore = -INF;

    for (int depth = 1; depth <= maxDepth; ++depth) {
        if (isTimeUp()) break;
        int alpha = -INF;
        int beta = INF;
        Move currentBest = bestMove;
        int currentBestScore = -INF;

        // order root moves each iteration (captures first)
        sort(rootMoves.begin(), rootMoves.end(), [](const Move& a, const Move& b) {
            if (a.isCapture != b.isCapture) return a.isCapture > b.isCapture;
            if (a.promo != 0 || b.promo != 0) return a.promo != 0;
            return false;
        });

        for (const Move& m : rootMoves) {
            if (isTimeUp()) break;
            Board nb;
            makeMove(b, m, nb);
            int score = -negamax(nb, depth - 1, -beta, -alpha);
            if (isTimeUp()) break;
            if (score > currentBestScore) {
                currentBestScore = score;
                currentBest = m;
            }
            if (score > alpha) {
                alpha = score;
            }
        }

        if (!isTimeUp()) {
            bestMove = currentBest;
            bestScore = currentBestScore;
        } else {
            break;
        }
    }

    (void)bestScore; // unused but kept for potential logging
    return bestMove;
}

// =================== Notation Conversion ===================

string squareToString(int sq) {
    string s;
    s += char('a' + fileOf(sq));
    s += char('1' + rankOf(sq));
    return s;
}

char pieceLetterForNotation(int piece) {
    int pt = pieceType(piece);
    if (pt == 1) return 'p';
    if (pt == 2) return 'N';
    if (pt == 3) return 'B';
    if (pt == 4) return 'R';
    if (pt == 5) return 'Q';
    if (pt == 6) return 'K';
    return 'p';
}

char promotionLetter(int promoPiece) {
    int pt = pieceType(promoPiece);
    switch (pt) {
        case 2: return 'N';
        case 3: return 'B';
        case 4: return 'R';
        case 5: return 'Q';
        default: return 'Q';
    }
}

bool isCheckAfterMove(const Board& b, const Move& m) {
    Board nb;
    makeMove(b, m, nb);
    return inCheck(nb, nb.sideToMove);
}

bool isMateAfterMove(const Board& b, const Move& m) {
    Board nb;
    makeMove(b, m, nb);
    if (!inCheck(nb, nb.sideToMove)) return false;
    vector<Move> moves;
    generateLegalMoves(nb, moves);
    return moves.empty();
}

string moveToNotation(const Board& b, const Move& m) {
    // Castling
    if (m.isCastle && pieceType(m.piece) == 6) {
        bool white = isWhite(m.piece);
        bool kingside = (fileOf(m.to) == 6);
        string s = kingside ? "O-O" : "O-O-O";
        if (isMateAfterMove(b, m)) s += "#";
        else if (isCheckAfterMove(b, m)) s += "+";
        return s;
    }

    string res;
    res += pieceLetterForNotation(m.piece);
    res += squareToString(m.from);
    if (m.isCapture) res += 'x';
    res += squareToString(m.to);
    if (m.isEP) res += "ep";
    if (m.promo != 0) {
        res += '/';
        res += promotionLetter(m.promo);
    }
    if (isMateAfterMove(b, m)) res += '#';
    else if (isCheckAfterMove(b, m)) res += '+';
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSec;
    string fenLine;
    if (!(cin >> timeoutSec)) {
        return 0;
    }
    // Read rest of line and the FEN (which may contain spaces)
    getline(cin, fenLine); // consume endline
    if (!getline(cin, fenLine)) {
        return 0;
    }

    timeLimitSeconds = timeoutSec;
    Board b;
    if (!parseFEN(fenLine, b)) {
        // invalid FEN; output dummy move
        cout << "O-O\n";
        return 0;
    }

    startTime = chrono::steady_clock::now();
    timeUp.store(false, memory_order_relaxed);
    nodes = 0;

    // Reasonable depth cap; time control will stop earlier if needed
    Move best = iterativeDeepening(b, 8);

    string notation = moveToNotation(b, best);
    cout << notation << "\n";

    return 0;
}
