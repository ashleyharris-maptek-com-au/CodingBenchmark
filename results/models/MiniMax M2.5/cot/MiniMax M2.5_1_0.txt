vector<int> solveExact(const vector<Point>& pts) {
    int N = pts.size();
    const int MAXMASK = 1 << N;
    const double INF = 1e100;
    vector<double> dp(MAXMASK * N, INF);
    vector<int> pred(MAXMASK * N, -1);
    auto idx = [&](int mask, int i) { return mask * N + i; };
    dp[idx(1<<0, 0)] = 0.0;
    for(int mask = 1; mask < MAXMASK; ++mask) {
        for(int i = 0; i < N; ++i) {
            if(!(mask & (1<<i))) continue;
            double cur = dp[idx(mask,i)];
            if(cur >= INF/2) continue;
            for(int j = 0; j < N; ++j) {
                if(mask & (1<<j)) continue;
                int nmask = mask | (1<<j);
                double nd = cur + dist(pts[i], pts[j]);
                int id = idx(nmask,j);
                if(nd < dp[id]) {
                    dp[id] = nd;
                    pred[id] = i;
                }
            }
        }
    }
    int fullMask = (1<<N) - 1;
    double best = INF;
    int last = -1;
    for(int i = 1; i < N; ++i) {
        double total = dp[idx(fullMask,i)] + dist(pts[i], pts[0]);
        if(total < best) {
            best = total;
            last = i;
        }
    }
    // reconstruct
    vector<int> path;
    int mask = fullMask;
    int cur = last;
    while(true) {
        path.push_back(cur);
        if(cur == 0) break;
        int pmask = mask ^ (1<<cur);
        cur = pred[idx(mask,cur)];
        mask = pmask;
    }
    reverse(path.begin(), path.end());
    // path starts with 0
    return path;
}

vector<int> solveHeuristic(const vector<Point>& pts) {
    int N = pts.size();
    vector<int> tour;
    tour.reserve(N);
    vector<char> visited(N, 0);
    tour.push_back(0);
    visited[0] = 1;
    // nearest neighbor
    for(int step = 1; step < N; ++step) {
        int current = tour.back();
        double bestDist = 1e100;
        int bestIdx = -1;
        for(int j = 0; j < N; ++j) {
            if(!visited[j]) {
                double d = dist(pts[current], pts[j]);
                if(d < bestDist) {
                    bestDist = d;
                    bestIdx = j;
                }
            }
        }
        // bestIdx must be found
        tour.push_back(bestIdx);
        visited[bestIdx] = 1;
    }

    // 2-opt improvement
    const int MAX_ITERS = 2;
    for(int iter = 0; iter < MAX_ITERS; ++iter) {
        bool improved = false;
        for(int i = 0; i < N - 2; ++i) {
            int a = tour[i];
            int b = tour[i+1];
            for(int j = i+2; j < N - 1; ++j) {
                int c = tour[j];
                int d = tour[j+1];
                double before = dist(pts[a], b) + dist(pts[c], d);
                double after = dist(pts[a], c) + dist(pts[b], d);
                if(after < before - 1e-12) {
                    // reverse segment i+1..j
                    reverse(tour.begin()+i+1, tour.begin()+j+1);
                    improved = true;
                }
            }
        }
        if(!improved) break;
    }
    return tour;
}

vector<int> solveLarge(const vector<Point>& pts) {
    int N = pts.size();
    // compute centroid
    double sumx = 0, sumy = 0;
    for(const auto& p: pts) {
        sumx += p.x;
        sumy += p.y;
    }
    double cx = sumx / N;
    double cy = sumy / N;

    struct Entry {
        double angle;
        int idx;
    };
    vector<Entry> arr(N);
    for(int i = 0; i < N; ++i) {
        arr[i].angle = atan2(pts[i].y - cy, pts[i].x - cx);
        arr[i].idx = i;
    }
    sort(arr.begin(), arr.end(), [](const Entry& a, const Entry& b){
        return a.angle < b.angle;
    });
    // find city 0
    int pos = -1;
    for(int i = 0; i < N; ++i) {
        if(arr[i].idx == 0) {
            pos = i;
            break;
        }
    }
    // rotate
    vector<int> tour;
    tour.reserve(N);
    for(int i = 0; i < N; ++i) {
        tour.push_back(arr[(pos + i) % N].idx);
    }
    return tour;
}
