#include "board.h"
#include <cstring>

Board::Board() {
    memset(pieces, 0, sizeof(pieces));
    memset(occupancy, 0, sizeof(occupancy));
    sideToMove = WHITE;
    castlingRights = 0;
    enPassant = SQ_NONE;
}

void Board::updateOccupancy() {
    occupancy[WHITE] = 0;
    occupancy[BLACK] = 0;
    for (int pt = 0; pt < 6; ++pt) {
        occupancy[WHITE] |= pieces[WHITE][pt];
        occupancy[BLACK] |= pieces[BLACK][pt];
    }
    occupancy[BOTH] = occupancy[WHITE] | occupancy[BLACK];
}

void Board::setFEN(const std::string& fen) {
    memset(pieces, 0, sizeof(pieces));
    std::stringstream ss(fen);
    std::string pos, side, castling, ep;
    ss >> pos >> side >> castling >> ep;

    int r = 7, c = 0;
    for (char ch : pos) {
        if (ch == '/') { r--; c = 0; }
        else if (isdigit(ch)) { c += ch - '0'; }
        else {
            Color color = isupper(ch) ? WHITE : BLACK;
            PieceType pt;
            char lch = tolower(ch);
            if (lch == 'p') pt = PAWN;
            else if (lch == 'n') pt = KNIGHT;
            else if (lch == 'b') pt = BISHOP;
            else if (lch == 'r') pt = ROOK;
            else if (lch == 'q') pt = QUEEN;
            else if (lch == 'k') pt = KING;
            pieces[color][pt] |= (1ULL << (r * 8 + c));
            c++;
        }
    }

    sideToMove = (side == "w") ? WHITE : BLACK;
    castlingRights = 0;
    if (castling.find('K') != std::string::npos) castlingRights |= 1;
    if (castling.find('Q') != std::string::npos) castlingRights |= 2;
    if (castling.find('k') != std::string::npos) castlingRights |= 4;
    if (castling.find('q') != std::string::npos) castlingRights |= 8;

    if (ep == "-") enPassant = SQ_NONE;
    else enPassant = (Square)((ep[1] - '1') * 8 + (ep[0] - 'a'));

    updateOccupancy();
}

Bitboard knightAttacks(Square sq) {
    Bitboard b = 1ULL << sq;
    Bitboard res = 0;
    if (sq % 8 > 1 && sq / 8 < 7) res |= b << 6;
    if (sq % 8 > 0 && sq / 8 < 6) res |= b << 15;
    if (sq % 8 < 7 && sq / 8 < 6) res |= b << 17;
    if (sq % 8 < 6 && sq / 8 < 7) res |= b << 10;
    if (sq % 8 < 6 && sq / 8 > 0) res |= b >> 6;
    if (sq % 8 < 7 && sq / 8 > 1) res |= b >> 15;
    if (sq % 8 > 0 && sq / 8 > 1) res |= b >> 17;
    if (sq % 8 > 1 && sq / 8 > 0) res |= b >> 10;
    return res;
}

Bitboard kingAttacks(Square sq) {
    Bitboard b = 1ULL << sq;
    Bitboard res = 0;
    if (sq % 8 > 0) res |= b >> 1;
    if (sq % 8 < 7) res |= b << 1;
    if (sq / 8 > 0) res |= b >> 8;
    if (sq / 8 < 7) res |= b << 8;
    if (sq % 8 > 0 && sq / 8 > 0) res |= b >> 9;
    if (sq % 8 < 7 && sq / 8 < 7) res |= b << 9;
    if (sq % 8 > 0 && sq / 8 < 7) res |= b << 7;
    if (sq % 8 < 7 && sq / 8 > 0) res |= b >> 7;
    return res;
}

void Board::addPieceMoves(std::vector<Move>& moves, Color c, PieceType pt) const {
    Bitboard b = pieces[c][pt];
    while (b) {
        int from = get_lsb(b);
        b &= b - 1;
        Bitboard attacks = 0;
        if (pt == KNIGHT) attacks = knightAttacks((Square)from);
        else if (pt == KING) attacks = kingAttacks((Square)from);
        else if (pt == BISHOP || pt == QUEEN) {
            int dr[] = {1, 1, -1, -1};
            int dc[] = {1, -1, 1, -1};
            for (int i = 0; i < 4; i++) {
                for (int d = 1; d < 8; d++) {
                    int nr = (from / 8) + dr[i] * d;
                    int nc = (from % 8) + dc[i] * d;
                    if (nr < 0 || nr > 7 || nc < 0 || nc > 7) break;
                    int to = nr * 8 + nc;
                    attacks |= (1ULL << to);
                    if (occupancy[BOTH] & (1ULL << to)) break;
                }
            }
        }
        if (pt == ROOK || pt == QUEEN) {
            int dr[] = {1, -1, 0, 0};
            int dc[] = {0, 0, 1, -1};
            for (int i = 0; i < 4; i++) {
                for (int d = 1; d < 8; d++) {
                    int nr = (from / 8) + dr[i] * d;
                    int nc = (from % 8) + dc[i] * d;
                    if (nr < 0 || nr > 7 || nc < 0 || nc > 7) break;
                    int to = nr * 8 + nc;
                    attacks |= (1ULL << to);
                    if (occupancy[BOTH] & (1ULL << to)) break;
                }
            }
        }
        attacks &= ~occupancy[c];
        while (attacks) {
            int to = get_lsb(attacks);
            attacks &= attacks - 1;
            Move m;
            m.from = (Square)from;
            m.to = (Square)to;
            m.piece = pt;
            m.isCapture = (occupancy[1 - c] & (1ULL << to)) != 0;
            moves.push_back(m);
        }
    }
}

void Board::addPawnMoves(std::vector<Move>& moves, Color c) const {
    Bitboard b = pieces[c][PAWN];
    int dir = (c == WHITE) ? 1 : -1;
    int startRank = (c == WHITE) ? 1 : 6;
    int promoRank = (c == WHITE) ? 7 : 0;

    while (b) {
        int from = get_lsb(b);
        b &= b - 1;
        int to = from + dir * 8;
        if (!(occupancy[BOTH] & (1ULL << to))) {
            if (to / 8 == promoRank) {
                for (PieceType p : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                    Move m; m.from = (Square)from; m.to = (Square)to; m.piece = PAWN; m.promotion = p;
                    moves.push_back(m);
                }
            } else {
                Move m; m.from = (Square)from; m.to = (Square)to; m.piece = PAWN;
                moves.push_back(m);
                if (from / 8 == startRank && !(occupancy[BOTH] & (1ULL << (from + 2 * dir * 8)))) {
                    Move m2; m2.from = (Square)from; m2.to = (Square)(from + 2 * dir * 8); m2.piece = PAWN;
                    moves.push_back(m2);
                }
            }
        }
        for (int side : {-1, 1}) {
            int nc = (from % 8) + side;
            if (nc >= 0 && nc <= 7) {
                int to_cap = from + dir * 8 + side;
                if (occupancy[1 - c] & (1ULL << to_cap)) {
                    if (to_cap / 8 == promoRank) {
                        for (PieceType p : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                            Move m; m.from = (Square)from; m.to = (Square)to_cap; m.piece = PAWN; m.isCapture = true; m.promotion = p;
                            moves.push_back(m);
                        }
                    } else {
                        Move m; m.from = (Square)from; m.to = (Square)to_cap; m.piece = PAWN; m.isCapture = true;
                        moves.push_back(m);
                    }
                } else if (to_cap == enPassant) {
                    Move m; m.from = (Square)from; m.to = (Square)to_cap; m.piece = PAWN; m.isCapture = true; m.isEnPassant = true;
                    moves.push_back(m);
                }
            }
        }
    }
}

void Board::addCastlingMoves(std::vector<Move>& moves, Color c) const {
    if (inCheck()) return;
    if (c == WHITE) {
        if ((castlingRights & 1) && !(occupancy[BOTH] & ((1ULL << F1) | (1ULL << G1)))) {
            if (!isSquareAttacked(F1, BLACK) && !isSquareAttacked(G1, BLACK)) {
                Move m; m.from = E1; m.to = G1; m.piece = KING; m.isCastleK = true;
                moves.push_back(m);
            }
        }
        if ((castlingRights & 2) && !(occupancy[BOTH] & ((1ULL << D1) | (1ULL << C1) | (1ULL << B1)))) {
            if (!isSquareAttacked(D1, BLACK) && !isSquareAttacked(C1, BLACK)) {
                Move m; m.from = E1; m.to = C1; m.piece = KING; m.isCastleQ = true;
                moves.push_back(m);
            }
        }
    } else {
        if ((castlingRights & 4) && !(occupancy[BOTH] & ((1ULL << F8) | (1ULL << G8)))) {
            if (!isSquareAttacked(F8, WHITE) && !isSquareAttacked(G8, WHITE)) {
                Move m; m.from = E8; m.to = G8; m.piece = KING; m.isCastleK = true;
                moves.push_back(m);
            }
        }
        if ((castlingRights & 8) && !(occupancy[BOTH] & ((1ULL << D8) | (1ULL << C8) | (1ULL << B8)))) {
            if (!isSquareAttacked(D8, WHITE) && !isSquareAttacked(C8, WHITE)) {
                Move m; m.from = E8; m.to = C8; m.piece = KING; m.isCastleQ = true;
                moves.push_back(m);
            }
        }
    }
}

std::vector<Move> Board::generateMoves() const {
    std::vector<Move> moves;
    addPawnMoves(moves, sideToMove);
    addPieceMoves(moves, sideToMove, KNIGHT);
    addPieceMoves(moves, sideToMove, BISHOP);
    addPieceMoves(moves, sideToMove, ROOK);
    addPieceMoves(moves, sideToMove, QUEEN);
    addPieceMoves(moves, sideToMove, KING);
    addCastlingMoves(moves, sideToMove);
    
    std::vector<Move> legalMoves;
    for (auto m : moves) {
        Board next = *this;
        next.makeMove(m);
        if (!next.isSquareAttacked((Square)get_lsb(next.pieces[sideToMove][KING]), (Color)(1 - sideToMove))) {
            legalMoves.push_back(m);
        }
    }
    return legalMoves;
}

bool Board::isSquareAttacked(Square sq, Color attacker) const {
    int dir = (attacker == WHITE) ? -1 : 1;
    for (int side : {-1, 1}) {
        int nc = (sq % 8) + side;
        if (nc >= 0 && nc <= 7) {
            int from = sq + dir * 8 + side;
            if (from >= 0 && from < 64 && (pieces[attacker][PAWN] & (1ULL << from))) return true;
        }
    }
    if (knightAttacks(sq) & pieces[attacker][KNIGHT]) return true;
    if (kingAttacks(sq) & pieces[attacker][KING]) return true;
    int dr[] = {1, 1, -1, -1, 1, -1, 0, 0};
    int dc[] = {1, -1, 1, -1, 0, 0, 1, -1};
    for (int i = 0; i < 8; i++) {
        for (int d = 1; d < 8; d++) {
            int nr = (sq / 8) + dr[i] * d;
            int nc = (sq % 8) + dc[i] * d;
            if (nr < 0 || nr > 7 || nc < 0 || nc > 7) break;
            int to = nr * 8 + nc;
            Bitboard b = (1ULL << to);
            if (occupancy[BOTH] & b) {
                if (i < 4) { if (pieces[attacker][BISHOP] & b || pieces[attacker][QUEEN] & b) return true; }
                else { if (pieces[attacker][ROOK] & b || pieces[attacker][QUEEN] & b) return true; }
                break;
            }
        }
    }
    return false;
}

void Board::makeMove(const Move& move) {
    PieceType pt = move.piece;
    pieces[sideToMove][pt] &= ~(1ULL << move.from);
    for (int i = 0; i < 6; i++) pieces[1 - sideToMove][i] &= ~(1ULL << move.to);
    if (pt == PAWN && move.to == enPassant) {
        int capSq = move.to + (sideToMove == WHITE ? -8 : 8);
        pieces[1 - sideToMove][PAWN] &= ~(1ULL << capSq);
    }
    if (move.isCastleK) {
        int r = (sideToMove == WHITE ? 0 : 7);
        pieces[sideToMove][ROOK] &= ~(1ULL << (r * 8 + 7));
        pieces[sideToMove][ROOK] |= (1ULL << (r * 8 + 5));
    } else if (move.isCastleQ) {
        int r = (sideToMove == WHITE ? 0 : 7);
        pieces[sideToMove][ROOK] &= ~(1ULL << (r * 8 + 0));
        pieces[sideToMove][ROOK] |= (1ULL << (r * 8 + 3));
    }
    if (move.promotion != NONE) pieces[sideToMove][move.promotion] |= (1ULL << move.to);
    else pieces[sideToMove][pt] |= (1ULL << move.to);
    if (pt == KING) {
        if (sideToMove == WHITE) castlingRights &= ~3;
        else castlingRights &= ~12;
    }
    if (move.from == A1 || move.to == A1) castlingRights &= ~2;
    if (move.from == H1 || move.to == H1) castlingRights &= ~1;
    if (move.from == A8 || move.to == A8) castlingRights &= ~8;
    if (move.from == H8 || move.to == H8) castlingRights &= ~4;
    enPassant = SQ_NONE;
    if (pt == PAWN && abs(move.to - move.from) == 16) enPassant = (Square)((move.from + move.to) / 2);
    sideToMove = (Color)(1 - sideToMove);
    updateOccupancy();
}
