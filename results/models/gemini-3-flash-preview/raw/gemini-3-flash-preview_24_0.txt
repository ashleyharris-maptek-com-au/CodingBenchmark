#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <thread>
#include <mutex>
#include <cmath>
#include <fstream>
#include <atomic>
#include <iomanip>
#include <condition_variable>
#include <algorithm>

#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

using namespace std;

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

struct Header {
    double width, depth, height, gravity, max_thrust, fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    string terrain_path;
};

atomic<bool> running{true};
State current_state;
mutex state_mutex;
condition_variable state_cv;
bool state_updated = false;

void read_stdin() {
    string line;
    while (getline(cin, line)) {
        if (line.empty()) continue;
        if (line.find("STATE") != string::npos) continue;
        
        stringstream ss(line);
        State s_new;
        if (ss >> s_new.x >> s_new.y >> s_new.z >> s_new.vx >> s_new.vy >> s_new.vz 
               >> s_new.pitch >> s_new.yaw >> s_new.roll 
               >> s_new.pitch_rate >> s_new.yaw_rate >> s_new.roll_rate >> s_new.fuel) {
            lock_guard<mutex> lock(state_mutex);
            current_state = s_new;
            state_updated = true;
            state_cv.notify_one();
        }
    }
    running = false;
    state_cv.notify_one();
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    Header h;
    if (!(cin >> h.width >> h.depth >> h.height >> h.gravity >> h.max_thrust >> h.fuel)) return 0;
    if (!(cin >> h.start_x >> h.start_y >> h.start_z >> h.target_x >> h.target_y >> h.target_z)) return 0;
    
    string dummy;
    getline(cin, dummy); // consume newline
    if (!getline(cin, h.terrain_path)) return 0;
    
    // Consume "STATE" line
    getline(cin, dummy);

    thread input_thread(read_stdin);

    State s;
    bool first_state = true;

    while (running) {
        {
            unique_lock<mutex> lock(state_mutex);
            state_cv.wait(lock, []{ return state_updated || !running; });
            if (!running) break;
            s = current_state;
            state_updated = false;
        }

        // PD control parameters
        double Kp_pos = 0.5;
        double Kv_pos = 1.2;
        double Kp_angle = 8.0;
        double Kv_angle = 1.5;

        double dx = h.target_x - s.x;
        double dy = h.target_y - s.y;
        double dz = h.target_z - s.z;
        double dist_horiz = sqrt(dx*dx + dy*dy);

        // Horizontal velocity target
        double target_vx = dx * Kp_pos;
        double target_vy = dy * Kp_pos;
        
        // Limit horizontal velocity based on distance to avoid overshooting
        double max_v_horiz = min(150.0, 3.0 + dist_horiz * 0.5);
        double curr_target_v_horiz = sqrt(target_vx*target_vx + target_vy*target_vy);
        if (curr_target_v_horiz > max_v_horiz) {
            target_vx *= max_v_horiz / curr_target_v_horiz;
            target_vy *= max_v_horiz / curr_target_v_horiz;
        }

        // Vertical velocity target
        double target_vz;
        // Navigation: stay high until horizontally close to target
        double safe_altitude = max(h.start_z, h.target_z + 100.0);
        safe_altitude = min(safe_altitude, h.height - 10.0);

        if (dist_horiz > 30.0) {
            target_vz = (safe_altitude - s.z) * Kp_pos;
        } else {
            // Near target, descend
            target_vz = (h.target_z - s.z) * Kp_pos;
        }
        
        // Limit descent speed for a soft landing (< 3m/s)
        if (target_vz < -2.0) target_vz = -2.0; 
        if (target_vz > 20.0) target_vz = 20.0;

        // Desired acceleration (to correct velocity error)
        double ax_des = (target_vx - s.vx) * Kv_pos;
        double ay_des = (target_vy - s.vy) * Kv_pos;
        double az_des = (target_vz - s.vz) * Kv_pos + h.gravity;

        double thrust_total = sqrt(ax_des*ax_des + ay_des*ay_des + az_des*az_des);
        
        // Normalize thrust command
        double thrust_cmd = thrust_total / h.max_thrust;
        if (thrust_cmd > 1.0) {
            // Priority: scale down horizontal components if thrust is saturated
            // but keep az_des to maintain vertical control
            thrust_cmd = 1.0;
            thrust_total = h.max_thrust;
        }
        if (thrust_cmd < 0.0) thrust_cmd = 0.0;

        // Calculate desired orientation (thrust vector direction)
        double nx = ax_des / thrust_total;
        double ny = ay_des / thrust_total;
        double nz = az_des / thrust_total;

        // p_des in [0, pi/2] as we need nz >= 0
        double p_des = acos(max(0.0, min(1.0, nz)));
        double y_des = s.yaw;
        if (abs(nx) > 1e-6 || abs(ny) > 1e-6) {
            y_des = atan2(-ny, -nx);
        }

        // Angular error calculation with wrap-around for yaw/roll
        auto angle_diff = [](double target, double current) {
            double diff = target - current;
            while (diff > M_PI) diff -= 2 * M_PI;
            while (diff < -M_PI) diff += 2 * M_PI;
            return diff;
        };

        double p_err = p_des - s.pitch;
        double y_err = angle_diff(y_des, s.yaw);
        double r_err = angle_diff(0.0, s.roll);

        // PD control for angular rates
        // Command applies 2 rad/s^2 acceleration. Damping is 0.95 per 0.1s tick.
        double dt = 0.1;
        double p_rate_target = p_err * Kp_angle;
        double y_rate_target = y_err * Kp_angle;
        double r_rate_target = r_err * Kp_angle;

        double pitch_cmd = (p_rate_target - s.pitch_rate * 0.95) / (2.0 * dt);
        double yaw_cmd = (y_rate_target - s.yaw_rate * 0.95) / (2.0 * dt);
        double roll_cmd = (r_rate_target - s.roll_rate * 0.95) / (2.0 * dt);

        // Clamp command outputs to [-1, 1]
        auto clamp_cmd = [](double v) { return max(-1.0, min(1.0, v)); };
        pitch_cmd = clamp_cmd(pitch_cmd);
        yaw_cmd = clamp_cmd(yaw_cmd);
        roll_cmd = clamp_cmd(roll_cmd);

        cout << fixed << setprecision(6) 
             << thrust_cmd << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        cout.flush();
    }

    if (input_thread.joinable()) input_thread.join();
    return 0;
}
