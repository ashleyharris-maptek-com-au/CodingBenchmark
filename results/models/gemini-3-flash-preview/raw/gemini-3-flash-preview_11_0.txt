use std::io::{self, BufRead};

#[derive(Clone, Copy, Debug)]
struct Point { x: f64, y: f64 }

struct Polygon {
    vertices: Vec<Point>,
    min_x: f64, min_y: f64, max_x: f64, max_y: f64,
    area: f64,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        let (mut min_x, mut min_y, mut max_x, mut max_y) = (f64::INFINITY, f64::INFINITY, f64::NEG_INFINITY, f64::NEG_INFINITY);
        for p in &vertices {
            min_x = min_x.min(p.x); min_y = min_y.min(p.y);
            max_x = max_x.max(p.x); max_y = max_y.max(p.y);
        }
        let mut area = 0.0;
        for i in 0..vertices.len() {
            let (p1, p2) = (vertices[i], vertices[(i + 1) % vertices.len()]);
            area += p1.x * p2.y - p2.x * p1.y;
        }
        Polygon { vertices, min_x, min_y, max_x, max_y, area: area.abs() * 0.5 }
    }
    fn rotated(&self, angle: f64) -> Self {
        if angle == 0.0 { return Polygon { vertices: self.vertices.clone(), ..*self }; }
        let (s, c) = angle.sin_cos();
        Polygon::new(self.vertices.iter().map(|p| Point { x: p.x * c - p.y * s, y: p.x * s + p.y * c }).collect())
    }
}

const GRID: usize = 1024;
const WORDS: usize = GRID / 64;
type BitRow = [u64; WORDS];

fn set_range(row: &mut BitRow, x1: usize, x2: usize) {
    let (x1, x2) = (x1.min(GRID - 1), x2.min(GRID - 1));
    if x1 > x2 { return; }
    let (s, e) = (x1 / 64, x2 / 64);
    if s == e {
        let mask = if x2 - x1 == 63 { !0u64 } else { ((1u64 << (x2 - x1 + 1)) - 1) << (x1 % 64) };
        row[s] |= mask;
    } else {
        row[s] |= !0u64 << (x1 % 64);
        for i in s + 1..e { row[i] = !0u64; }
        row[e] |= !0u64 >> (63 - (x2 % 64));
    }
}

fn is_range_set(row: &BitRow, x1: usize, x2: usize) -> bool {
    let (x1, x2) = (x1.min(GRID - 1), x2.min(GRID - 1));
    if x1 > x2 { return true; }
    let (s, e) = (x1 / 64, x2 / 64);
    if s == e {
        let mask = if x2 - x1 == 63 { !0u64 } else { ((1u64 << (x2 - x1 + 1)) - 1) << (x1 % 64) };
        return (row[s] & mask) == mask;
    }
    if (row[s] & (!0u64 << (x1 % 64))) != (!0u64 << (x1 % 64)) { return false; }
    for i in s + 1..e { if row[i] != !0u64 { return false; } }
    let last_mask = !0u64 >> (63 - (x2 % 64));
    (row[e] & last_mask) == last_mask
}

fn is_range_clear(row: &BitRow, x1: usize, x2: usize) -> bool {
    let (x1, x2) = (x1.min(GRID - 1), x2.min(GRID - 1));
    if x1 > x2 { return true; }
    let (s, e) = (x1 / 64, x2 / 64);
    if s == e {
        let mask = if x2 - x1 == 63 { !0u64 } else { ((1u64 << (x2 - x1 + 1)) - 1) << (x1 % 64) };
        return (row[s] & mask) == 0;
    }
    if (row[s] & (!0u64 << (x1 % 64))) != 0 { return false; }
    for i in s + 1..e { if row[i] != 0 { return false; } }
    let last_mask = !0u64 >> (63 - (x2 % 64));
    (row[e] & last_mask) == 0
}

fn rasterize(poly: &Polygon, scale: f64, ox: f64, oy: f64) -> Vec<(i32, Vec<(i32, i32)>)> {
    let mut rows = Vec::new();
    let min_y = (poly.min_y * scale + oy).floor() as i32;
    let max_y = (poly.max_y * scale + oy).ceil() as i32;
    for y in min_y..=max_y {
        let fy = (y as f64 + 0.5 - oy) / scale;
        let mut intersects = Vec::new();
        for i in 0..poly.vertices.len() {
            let (p1, p2) = (poly.vertices[i], poly.vertices[(i + 1) % poly.vertices.len()]);
            if (p1.y <= fy && p2.y > fy) || (p2.y <= fy && p1.y > fy) {
                intersects.push(p1.x + (fy - p1.y) * (p2.x - p1.x) / (p2.y - p1.y));
            }
        }
        if intersects.is_empty() { continue; }
        intersects.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
        let mut intervals = Vec::new();
        for i in (0..intersects.len()).step_by(2) {
            if i + 1 < intersects.len() {
                let (x1, x2) = ((intersects[i] * scale + ox).round() as i32, (intersects[i+1] * scale + ox).round() as i32);
                if x1 <= x2 { intervals.push((x1, x2)); }
            }
        }
        if !intervals.is_empty() { rows.push((y, intervals)); }
    }
    rows
}

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let mut line = String::new();
    if reader.read_line(&mut line).is_err() || line.trim().is_empty() { return; }
    let mut parts = line.split_whitespace();
    let sv = parts.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    let np = parts.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    if sv == 0 || np == 0 { return; }
    
    let mut stock_verts = Vec::new();
    for _ in 0..sv {
        line.clear(); reader.read_line(&mut line).unwrap();
        let mut pts = line.split_whitespace();
        let x = pts.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let y = pts.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        stock_verts.push(Point { x, y });
    }
    let stock_poly = Polygon::new(stock_verts);
    
    let mut pieces = Vec::new();
    for i in 0..np {
        line.clear(); reader.read_line(&mut line).unwrap();
        let pv = line.trim().parse::<usize>().unwrap_or(0);
        let mut verts = Vec::new();
        for _ in 0..pv {
            line.clear(); reader.read_line(&mut line).unwrap();
            let mut pts = line.split_whitespace();
            let x = pts.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
            let y = pts.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
            verts.push(Point { x, y });
        }
        pieces.push((i, Polygon::new(verts)));
    }
    pieces.sort_by(|a, b| b.1.area.partial_cmp(&a.1.area).unwrap_or(std::cmp::Ordering::Equal));

    let scale = (GRID as f64 - 10.0) / (stock_poly.max_x - stock_poly.min_x).max(stock_poly.max_y - stock_poly.min_y).max(0.001);
    let (ox, oy) = (-stock_poly.min_x * scale + 5.0, -stock_poly.min_y * scale + 5.0);
    let mut stock_mask = [[0u64; WORDS]; GRID];
    for (y, ints) in rasterize(&stock_poly, scale, ox, oy) {
        if y >= 0 && y < GRID as i32 {
            for (x1, x2) in ints {
                set_range(&mut stock_mask[y as usize], x1.max(0) as usize, x2.min(GRID as i32 - 1) as usize);
            }
        }
    }

    let mut stocks = Vec::new();
    let mut results = vec![(0, 0.0, 0.0, 0.0); np];
    let rots = [0.0, 1.5707963, 3.1415926, 4.7123889];

    for (orig_idx, piece_poly) in pieces {
        let mut piece_variants = Vec::new();
        for &r in &rots {
            let rp = piece_poly.rotated(r);
            let ints = rasterize(&rp, scale, 0.0, 0.0);
            if !ints.is_empty() { piece_variants.push((r, rp, ints)); }
        }

        let mut placed = false;
        'st: for (s_idx, (mask, skyline)) in stocks.iter_mut().enumerate() {
            for (r, _rp, ints) in &piece_variants {
                let (mut min_px, mut max_px) = (i32::MAX, i32::MIN);
                for (_, ri) in ints { for (x1, x2) in ri { min_px = min_px.min(*x1); max_px = max_px.max(*x2); } }
                let (py_min, py_max) = (ints[0].0, ints.last().unwrap().0);

                let x_step = if np > 1000 { 4 } else { 1 };
                for x_off in (0..GRID as i32 - (max_px - min_px)).step_by(x_step) {
                    let mut y_start = 0;
                    for dx in 0..=(max_px - min_px) {
                        y_start = y_start.max(skyline[(x_off + dx) as usize] - py_min);
                    }
                    
                    for y_search in y_start..=(GRID as i32 - (py_max - py_min)) {
                        let mut ok = true;
                        for (dy, ri) in ints {
                            let sy = y_search + dy;
                            if sy < 0 || sy >= GRID as i32 { ok = false; break; }
                            for (x1, x2) in ri {
                                let (sx1, sx2) = (x_off + x1 - min_px, x_off + x2 - min_px);
                                if sx1 < 0 || sx2 >= GRID as i32 || !is_range_set(&stock_mask[sy as usize], sx1 as usize, sx2 as usize) || !is_range_clear(&mask[sy as usize], sx1 as usize, sx2 as usize) {
                                    ok = false; break;
                                }
                            }
                            if !ok { break; }
                        }
                        if ok {
                            results[orig_idx] = (s_idx, (x_off as f64 - min_px as f64 - ox) / scale, (y_search as f64 - py_min as f64 - oy) / scale, *r);
                            for (dy, ri) in ints {
                                let sy = y_search + dy;
                                for (x1, x2) in ri {
                                    let (sx1, sx2) = (x_off + x1 - min_px, x_off + x2 - min_px);
                                    set_range(&mut mask[sy as usize], sx1 as usize, sx2 as usize);
                                    for x in sx1..=sx2 { skyline[x as usize] = skyline[x as usize].max(sy + 1); }
                                }
                            }
                            placed = true; break 'st;
                        }
                        if y_search > y_start + 5 { break; }
                    }
                }
            }
        }

        if !placed {
            let s_idx = stocks.len();
            stocks.push(([[0u64; WORDS]; GRID], vec![0; GRID]));
            let (mask, skyline) = stocks.last_mut().unwrap();
            
            let mut found = false;
            for (r, _rp, ints) in &piece_variants {
                let (mut min_px, mut max_px) = (i32::MAX, i32::MIN);
                for (_, ri) in ints { for (x1, x2) in ri { min_px = min_px.min(*x1); max_px = max_px.max(*x2); } }
                let (py_min, py_max) = (ints[0].0, ints.last().unwrap().0);

                for y_off in 0..=(GRID as i32 - (py_max - py_min)) {
                    for x_off in 0..=(GRID as i32 - (max_px - min_px)) {
                        let mut ok = true;
                        for (dy, ri) in ints {
                            let sy = y_off + dy;
                            if sy < 0 || sy >= GRID as i32 { ok = false; break; }
                            for (x1, x2) in ri {
                                let (sx1, sx2) = (x_off + x1 - min_px, x_off + x2 - min_px);
                                if sx1 < 0 || sx2 >= GRID as i32 || !is_range_set(&stock_mask[sy as usize], sx1 as usize, sx2 as usize) {
                                    ok = false; break;
                                }
                            }
                            if !ok { break; }
                        }
                        if ok {
                            results[orig_idx] = (s_idx, (x_off as f64 - min_px as f64 - ox) / scale, (y_off as f64 - py_min as f64 - oy) / scale, *r);
                            for (dy, ri) in ints {
                                let sy = y_off + dy;
                                for (x1, x2) in ri {
                                    let (sx1, sx2) = (x_off + x1 - min_px, x_off + x2 - min_px);
                                    set_range(&mut mask[sy as usize], sx1 as usize, sx2 as usize);
                                    for x in sx1..=sx2 { skyline[x as usize] = skyline[x as usize].max(sy + 1); }
                                }
                            }
                            found = true; break;
                        }
                    }
                    if found { break; }
                }
                if found { break; }
            }
        }
    }

    println!("{}", stocks.len());
    for (s, x, y, r) in results { println!("{} {:.8} {:.8} {:.8}", s, x, y, r); }
}
