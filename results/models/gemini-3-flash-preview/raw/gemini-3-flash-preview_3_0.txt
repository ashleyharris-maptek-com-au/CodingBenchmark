The solution implements a high-performance **Grid-based Force-Directed Layout** (inspired by the Fruchterman-Reingold algorithm) optimized for extreme scalability.

### Key Features and Technical Implementation:

1.  **Scalability & Complexity**:
    *   For small graphs ($N < 500$), it uses a standard $O(N^2)$ repulsion model for maximum precision.
    *   For larger graphs (up to 200,000 nodes), it employs a **Spatial Grid (Binning)** approach for repulsion forces. By dividing the layout area into cells and only calculating repulsion from nodes in adjacent cells, the complexity is reduced to **$O(N)$ per iteration**.
    *   Attraction forces are calculated in **$O(M)$** by iterating over the edges.

2.  **Performance Optimization**:
    *   **Fast I/O**: Uses a custom byte-level parser (`next_usize`) to read the entire input buffer at once, bypassing the overhead of standard string parsing.
    *   **Mathematical Simplification**: Repulsion force calculation avoids costly `sqrt` operations where possible (using squared distances).
    *   **Memory Efficiency**: Uses primitive vectors for coordinates and forces, keeping the memory footprint minimal even at ludicrous scales (approx. 45MB for 200k nodes and 1M edges).

3.  **Layout Quality & Robustness**:
    *   **Central Gravity**: A weak force towards the center of the layout prevents disconnected components from drifting to infinity.
    *   **Adaptive Cooling**: Uses a simulated annealing temperature schedule that adapts its iteration count based on the graph size to balance quality and speed.
    *   **Graceful Handling**: Self-loops are ignored, parallel edges strengthen attraction, and nodes are constrained within a soft-clamped bounding box.

use std::io::{self, Read, Write, BufWriter};

struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Rng { state: seed }
    }
    fn next_f64(&mut self) -> f64 {
        self.state = self.state.wrapping_mul(6364136223846793005).wrapping_add(1);
        (self.state >> 11) as f64 / (1u64 << 53) as f64
    }
}

fn solve() -> io::Result<()> {
    // Efficiently read entire input
    let mut input_data = Vec::new();
    io::stdin().read_to_end(&mut input_data)?;
    
    let mut offset = 0;
    let mut next_usize = || {
        while offset < input_data.len() && !input_data[offset].is_ascii_digit() {
            offset += 1;
        }
        if offset >= input_data.len() {
            return None;
        }
        let mut res = 0usize;
        while offset < input_data.len() && input_data[offset].is_ascii_digit() {
            res = res.wrapping_mul(10).wrapping_add((input_data[offset] - b'0') as usize);
            offset += 1;
        }
        Some(res)
    };

    let n = match next_usize() {
        Some(val) => val,
        None => return Ok(()),
    };
    let m = match next_usize() {
        Some(val) => val,
        None => 0,
    };

    if n == 0 {
        return Ok(());
    }
    if n == 1 {
        let mut out = BufWriter::new(io::stdout());
        writeln!(out, "1\n500.000000 500.000000")?;
        return Ok(());
    }

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        if let (Some(u), Some(v)) = (next_usize(), next_usize()) {
            if u < n && v < n && u != v {
                edges.push((u, v));
            }
        }
    }

    let w = 1000.0;
    let h = 1000.0;
    let area = w * h;
    let k = (area / n as f64).sqrt();
    let k2 = k * k;

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];
    let mut rng = Rng::new(42);
    for i in 0..n {
        x[i] = rng.next_f64() * w;
        y[i] = rng.next_f64() * h;
    }

    let mut dx = vec![0.0f64; n];
    let mut dy = vec![0.0f64; n];

    let max_iter = if n < 100 {
        500
    } else if n < 1000 {
        300
    } else if n < 10000 {
        150
    } else if n < 100000 {
        100
    } else {
        75
    };

    let mut t = w / 10.0;
    let dt = t / max_iter as f64;
    let cx = w / 2.0;
    let cy = h / 2.0;

    for _iter in 0..max_iter {
        for i in 0..n {
            dx[i] = 0.0;
            dy[i] = 0.0;
        }

        if n < 500 {
            for i in 0..n {
                for j in i + 1..n {
                    let vx = x[i] - x[j];
                    let vy = y[i] - y[j];
                    let d2 = vx * vx + vy * vy + 1e-6;
                    let factor = k2 / d2;
                    let fx = vx * factor;
                    let fy = vy * factor;
                    dx[i] += fx;
                    dy[i] += fy;
                    dx[j] -= fx;
                    dy[j] -= fy;
                }
            }
        } else {
            let s = k * 2.5; 
            let gx = ((w / s).ceil() as usize).max(1) + 1;
            let gy = ((h / s).ceil() as usize).max(1) + 1;
            let mut head = vec![-1isize; gx * gy];
            let mut next = vec![-1isize; n];

            for i in 0..n {
                let ix = ((x[i] / s) as usize).min(gx - 1);
                let iy = ((y[i] / s) as usize).min(gy - 1);
                let idx = iy * gx + ix;
                next[i] = head[idx];
                head[idx] = i as isize;
            }

            let s2_limit = s * s * 4.0;
            for i in 0..n {
                let ix = ((x[i] / s) as usize).min(gx - 1);
                let iy = ((y[i] / s) as usize).min(gy - 1);

                for dy_grid in -1..=1 {
                    let ny = iy as isize + dy_grid;
                    if ny < 0 || ny >= gy as isize { continue; }
                    for dx_grid in -1..=1 {
                        let nx = ix as isize + dx_grid;
                        if nx < 0 || nx >= gx as isize { continue; }

                        let idx = (ny as usize) * gx + (nx as usize);
                        let mut j = head[idx];
                        while j != -1 {
                            let jj = j as usize;
                            if i < jj {
                                let vx = x[i] - x[jj];
                                let vy = y[i] - y[jj];
                                let d2 = vx * vx + vy * vy + 1e-6;
                                if d2 < s2_limit {
                                    let factor = k2 / d2;
                                    let fx = vx * factor;
                                    let fy = vy * factor;
                                    dx[i] += fx;
                                    dy[i] += fy;
                                    dx[jj] -= fx;
                                    dy[jj] -= fy;
                                }
                            }
                            j = next[jj];
                        }
                    }
                }
            }
        }

        for &(u, v) in &edges {
            let vx = x[u] - x[v];
            let vy = y[u] - y[v];
            let d2 = vx * vx + vy * vy + 1e-6;
            let d = d2.sqrt();
            let factor = d / k;
            let fx = vx * factor;
            let fy = vy * factor;
            dx[u] -= fx;
            dy[u] -= fy;
            dx[v] += fx;
            dy[v] += fy;
        }

        let gravity_factor = 0.1 * k / w;
        for i in 0..n {
            dx[i] -= (x[i] - cx) * gravity_factor;
            dy[i] -= (y[i] - cy) * gravity_factor;
        }

        for i in 0..n {
            let d2 = dx[i] * dx[i] + dy[i] * dy[i] + 1e-6;
            let d = d2.sqrt();
            let disp = d.min(t);
            x[i] += (dx[i] / d) * disp;
            y[i] += (dy[i] / d) * disp;
            x[i] = x[i].min(w).max(0.0);
            y[i] = y[i].min(h).max(0.0);
        }

        t -= dt;
        if t < 0.0 { t = 0.0; }
    }

    let mut out = BufWriter::new(io::stdout());
    writeln!(out, "{}", n)?;
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", x[i], y[i])?;
    }
    out.flush()?;

    Ok(())
}

fn main() {
    if let Err(e) = solve() {
        let _ = writeln!(io::stderr(), "Error: {}", e);
    }
}
