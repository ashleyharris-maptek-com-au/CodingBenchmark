using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;
using System.Globalization;
using System.Diagnostics;
using System.Text;

public class Program
{
    public class VoxelizedRotation
    {
        public int W, H, D;
        public ulong[] Bits;
        public int StrideZ;
        public Quaternion Rotation;
        public Vector3 MinOffset;
        public int FirstVoxelX, FirstVoxelY, FirstVoxelZ;
        public int VoxelCount;
    }

    public class Result
    {
        public Vector3 Translation;
        public Quaternion Rotation;
    }

    public static string ReadNonEmptyLine()
    {
        string line;
        while ((line = Console.ReadLine()) != null)
        {
            if (!string.IsNullOrWhiteSpace(line)) return line;
        }
        return null;
    }

    public static void Main()
    {
        System.Threading.Thread.CurrentThread.CurrentCulture = CultureInfo.InvariantCulture;

        string line = ReadNonEmptyLine();
        if (line == null) return;
        string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2) return;
        int V = int.Parse(parts[0]);
        int F = int.Parse(parts[1]);

        Vector3[] vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            line = ReadNonEmptyLine();
            if (line == null) break;
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            vertices[i] = new Vector3(float.Parse(parts[0]), float.Parse(parts[1]), float.Parse(parts[2]));
        }

        int[][] faces = new int[F][];
        for (int i = 0; i < F; i++)
        {
            line = ReadNonEmptyLine();
            if (line == null) break;
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            faces[i] = new int[] { int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]) };
        }

        line = ReadNonEmptyLine();
        if (line == null) return;
        parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        float CW = float.Parse(parts[0]);
        float CH = float.Parse(parts[1]);
        float CD = float.Parse(parts[2]);

        // Adaptive scaling for small containers to maintain precision
        float scale = 1.0f;
        if (CW < 30 || CH < 30 || CD < 30)
        {
            scale = 10.0f;
            if (CW < 3 || CH < 3 || CD < 3) scale = 100.0f;
        }

        if (scale != 1.0f)
        {
            CW *= scale; CH *= scale; CD *= scale;
            for (int i = 0; i < V; i++) vertices[i] *= scale;
        }

        var rotations = Get24Rotations();
        var voxelized = new List<VoxelizedRotation>();
        foreach (var q in rotations)
        {
            var vr = Voxelize(vertices, faces, q, CW, CH, CD);
            if (vr != null) voxelized.Add(vr);
        }

        if (voxelized.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Sort rotations: prefer shorter rotations in Z for better layering
        voxelized = voxelized.OrderBy(v => v.D).ThenBy(v => v.VoxelCount).ToList();

        int iCW = (int)Math.Floor(CW);
        int iCH = (int)Math.Floor(CH);
        int iCD = (int)Math.Floor(CD);
        Container container = new Container(iCW, iCH, iCD);
        List<Result> results = new List<Result>();

        Stopwatch sw = Stopwatch.StartNew();

        // Optimized First-Fit packing
        for (int z = 0; z < iCD; z++)
        {
            if (sw.ElapsedMilliseconds > 285000) break; // Time limit safety
            for (int y = 0; y < iCH; y++)
            {
                for (int x = 0; x < iCW; x++)
                {
                    if (container.Get(x, y, z)) continue;

                    foreach (var rot in voxelized)
                    {
                        int tx = x - rot.FirstVoxelX;
                        int ty = y - rot.FirstVoxelY;
                        int tz = z - rot.FirstVoxelZ;

                        if (container.CanPlace(tx, ty, tz, rot))
                        {
                            container.Place(tx, ty, tz, rot);
                            results.Add(new Result
                            {
                                Translation = (new Vector3(tx, ty, tz) - rot.MinOffset) / scale,
                                Rotation = rot.Rotation
                            });
                            break;
                        }
                    }
                }
            }
        }

        Console.WriteLine(results.Count);
        StringBuilder outSb = new StringBuilder();
        foreach (var res in results)
        {
            outSb.AppendLine($"{res.Translation.X:F4} {res.Translation.Y:F4} {res.Translation.Z:F4} {res.Rotation.W:F4} {res.Rotation.X:F4} {res.Rotation.Y:F4} {res.Rotation.Z:F4}");
            if (outSb.Length > 16384) { Console.Write(outSb.ToString()); outSb.Clear(); }
        }
        Console.Write(outSb.ToString());
    }

    public static Quaternion[] Get24Rotations()
    {
        var result = new List<Quaternion>();
        Vector3[] norms = { Vector3.UnitX, -Vector3.UnitX, Vector3.UnitY, -Vector3.UnitY, Vector3.UnitZ, -Vector3.UnitZ };
        foreach (var fwd in norms)
        {
            foreach (var up in norms)
            {
                if (Math.Abs(Vector3.Dot(fwd, up)) < 0.01f)
                {
                    Vector3 rgt = Vector3.Cross(up, fwd);
                    Matrix4x4 m = new Matrix4x4(rgt.X, rgt.Y, rgt.Z, 0, up.X, up.Y, up.Z, 0, fwd.X, fwd.Y, fwd.Z, 0, 0, 0, 0, 1);
                    Quaternion q = Quaternion.CreateFromRotationMatrix(m);
                    if (!result.Any(existing => Math.Abs(Quaternion.Dot(existing, q)) > 0.999f)) result.Add(q);
                }
            }
        }
        return result.ToArray();
    }

    public static VoxelizedRotation Voxelize(Vector3[] vertices, int[][] faces, Quaternion q, float gW, float gH, float gD)
    {
        int V = vertices.Length;
        Vector3[] rotV = new Vector3[V];
        Vector3 min = new Vector3(float.MaxValue), max = new Vector3(float.MinValue);
        for (int i = 0; i < V; i++)
        {
            rotV[i] = Vector3.Transform(vertices[i], q);
            min = Vector3.Min(min, rotV[i]);
            max = Vector3.Max(max, rotV[i]);
        }
        for (int i = 0; i < V; i++) rotV[i] -= min;
        max -= min;
        int W = Math.Max(1, (int)Math.Ceiling(max.X));
        int H = Math.Max(1, (int)Math.Ceiling(max.Y));
        int D = Math.Max(1, (int)Math.Ceiling(max.Z));

        if (W > gW || H > gH || D > gD) return null;

        int strideZ = (D + 63) / 64;
        ulong[] bits = new ulong[W * H * strideZ];
        List<float>[] rays = new List<float>[W * H];

        foreach (var f in faces)
        {
            Vector3 v0 = rotV[f[0]], v1 = rotV[f[1]], v2 = rotV[f[2]];
            float minX = Math.Min(v0.X, Math.Min(v1.X, v2.X)), maxX = Math.Max(v0.X, Math.Max(v1.X, v2.X));
            float minY = Math.Min(v0.Y, Math.Min(v1.Y, v2.Y)), maxY = Math.Max(v0.Y, Math.Max(v1.Y, v2.Y));
            int iX0 = Math.Max(0, (int)Math.Floor(minX)), iX1 = Math.Min(W - 1, (int)Math.Floor(maxX));
            int iY0 = Math.Max(0, (int)Math.Floor(minY)), iY1 = Math.Min(H - 1, (int)Math.Floor(maxY));
            for (int x = iX0; x <= iX1; x++) for (int y = iY0; y <= iY1; y++)
            {
                if (RayTriangleIntersect(new Vector3(x + 0.501f, y + 0.502f, -1.0f), Vector3.UnitZ, v0, v1, v2, out float t))
                {
                    if (rays[x * H + y] == null) rays[x * H + y] = new List<float>();
                    rays[x * H + y].Add(t - 1.0f);
                }
            }
        }

        int vCount = 0;
        for (int x = 0; x < W; x++)
        {
            for (int y = 0; y < H; y++)
            {
                var inters = rays[x * H + y]; if (inters == null) continue;
                inters.Sort();
                for (int z = 0; z < D; z++)
                {
                    float zC = z + 0.5f; int c = 0;
                    foreach (var t in inters) { if (t < zC) c++; else break; }
                    if (c % 2 == 1)
                    {
                        bits[(x * H + y) * strideZ + (z >> 6)] |= (1UL << (z & 63));
                        vCount++;
                    }
                }
            }
        }
        if (vCount == 0) return null;

        int fx = -1, fy = -1, fz = -1;
        for (int z = 0; z < D && fz == -1; z++)
            for (int y = 0; y < H && fz == -1; y++)
                for (int x = 0; x < W && fz == -1; x++)
                    if ((bits[(x * H + y) * strideZ + (z >> 6)] & (1UL << (z & 63))) != 0) { fx = x; fy = y; fz = z; }

        return new VoxelizedRotation { W = W, H = H, D = D, Bits = bits, StrideZ = strideZ, Rotation = q, MinOffset = min, FirstVoxelX = fx, FirstVoxelY = fy, FirstVoxelZ = fz, VoxelCount = vCount };
    }

    public static bool RayTriangleIntersect(Vector3 orig, Vector3 dir, Vector3 v0, Vector3 v1, Vector3 v2, out float t)
    {
        t = 0; Vector3 e1 = v1 - v0, e2 = v2 - v0, h = Vector3.Cross(dir, e2); float a = Vector3.Dot(e1, h);
        if (a > -1e-7f && a < 1e-7f) return false;
        float f = 1.0f / a; Vector3 s = orig - v0; float u = f * Vector3.Dot(s, h);
        if (u < 0.0f || u > 1.0f) return false;
        Vector3 q = Vector3.Cross(s, e1); float v = f * Vector3.Dot(dir, q);
        if (v < 0.0f || u + v > 1.0f) return false;
        t = f * Vector3.Dot(e2, q); return t > 0.00001f;
    }

    public class Container
    {
        public int W, H, D, StrideZ; public ulong[] Bits;
        public Container(int w, int h, int d) { W = w; H = h; D = d; StrideZ = (D + 63) / 64; Bits = new ulong[W * H * StrideZ]; }
        public bool Get(int x, int y, int z) { return (Bits[(x * H + y) * StrideZ + (z >> 6)] & (1UL << (z & 63))) != 0; }
        public bool CanPlace(int tx, int ty, int tz, VoxelizedRotation rot)
        {
            if (tx < 0 || ty < 0 || tz < 0 || tx + rot.W > W || ty + rot.H > H || tz + rot.D > D) return false;
            int bO = tz >> 6, bitO = tz & 63;
            for (int ix = 0; ix < rot.W; ix++)
            {
                int gx = tx + ix;
                for (int iy = 0; iy < rot.H; iy++)
                {
                    int gy = ty + iy;
                    int rB = (ix * rot.H + iy) * rot.StrideZ, gB = (gx * H + gy) * StrideZ;
                    for (int ib = 0; ib < rot.StrideZ; ib++)
                    {
                        ulong iV = rot.Bits[rB + ib]; if (iV == 0) continue;
                        if ((Bits[gB + bO + ib] & (iV << bitO)) != 0) return false;
                        if (bitO > 0 && bO + ib + 1 < StrideZ && (Bits[gB + bO + ib + 1] & (iV >> (64 - bitO))) != 0) return false;
                    }
                }
            }
            return true;
        }
        public void Place(int tx, int ty, int tz, VoxelizedRotation rot)
        {
            int bO = tz >> 6, bitO = tz & 63;
            for (int ix = 0; ix < rot.W; ix++)
            {
                int gx = tx + ix;
                for (int iy = 0; iy < rot.H; iy++)
                {
                    int gy = ty + iy;
                    int rB = (ix * rot.H + iy) * rot.StrideZ, gB = (gx * H + gy) * StrideZ;
                    for (int ib = 0; ib < rot.StrideZ; ib++)
                    {
                        ulong iV = rot.Bits[rB + ib]; if (iV == 0) continue;
                        Bits[gB + bO + ib] |= (iV << bitO);
                        if (bitO > 0 && bO + ib + 1 < StrideZ) Bits[gB + bO + ib + 1] |= (iV >> (64 - bitO));
                    }
                }
            }
        }
    }
}
