use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    #[allow(dead_code)]
    fuel_remaining: f64,
}

fn main() {
    // Set up stdin and read initial configuration
    let (width, height, gravity, max_thrust, target_x, target_y, skyline, skyline_bins, bin_size) = {
        let stdin = io::stdin();
        let mut handle = stdin.lock();
        
        let mut line = String::new();
        if handle.read_line(&mut line).is_err() || line.trim().is_empty() {
            return;
        }
        let parts: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
        if parts.len() < 4 { return; }
        let width = parts[0] as usize;
        let height = parts[1] as usize;
        let gravity = parts[2];
        let max_thrust = parts[3];

        line.clear();
        if handle.read_line(&mut line).is_err() { return; }
        let parts: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
        if parts.len() < 4 { return; }
        let target_x = parts[2];
        let target_y = parts[3];

        // Read map and find skyline (highest obstacle at each X)
        let mut skyline = vec![height as f32; width];
        for y in 0..height {
            line.clear();
            if handle.read_line(&mut line).is_err() || line.is_empty() { break; }
            let bytes = line.as_bytes();
            let limit = width.min(bytes.len());
            for x in 0..limit {
                let c = bytes[x];
                // Check if it's an obstacle, target, or landing zone
                if c == b'#' || c == b'L' || c == b'T' {
                    if (y as f32) < skyline[x] {
                        skyline[x] = y as f32;
                    }
                }
            }
        }

        // Pre-calculate coarse skyline for faster searching across long distances
        let bin_size = 1024;
        let mut skyline_bins = vec![height as f32; (width + bin_size - 1) / bin_size];
        for (i, &y) in skyline.iter().enumerate() {
            let bin = i / bin_size;
            if y < skyline_bins[bin] {
                skyline_bins[bin] = y;
            }
        }
        (width, height, gravity, max_thrust, target_x, target_y, skyline, skyline_bins, bin_size)
    };

    // Shared state for communication between input thread and main control loop
    let state_arc = Arc::new(Mutex::new(None));
    let state_clone = state_arc.clone();

    // Background thread to read continuous state updates from stdin
    thread::spawn(move || {
        let stdin = io::stdin();
        let mut handle = stdin.lock();
        let mut line = String::new();
        loop {
            line.clear();
            if handle.read_line(&mut line).is_err() || line.is_empty() {
                break;
            }
            let parts: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
            if parts.len() >= 7 {
                let mut s = state_clone.lock().unwrap();
                *s = Some(State {
                    x: parts[0],
                    y: parts[1],
                    vx: parts[2],
                    vy: parts[3],
                    angle: parts[4],
                    angular_vel: parts[5],
                    fuel_remaining: parts[6],
                });
            }
        }
    });

    // Wait for the first state update to begin control
    let mut current_state: Option<State> = None;
    while current_state.is_none() {
        {
            let s = state_arc.lock().unwrap();
            current_state = *s;
        }
        thread::sleep(Duration::from_millis(5));
    }

    let mut next_tick = Instant::now();
    let interval = Duration::from_millis(50); // Target 20Hz output
    let mut stdout = io::stdout().lock();

    loop {
        // Poll for the latest state update
        {
            let s = state_arc.lock().unwrap();
            if let Some(new_s) = *s {
                current_state = Some(new_s);
            }
        }

        if let Some(s) = current_state {
            let (thrust, turn) = compute_control(
                &s, target_x, target_y, gravity, max_thrust, &skyline, &skyline_bins, bin_size
            );
            
            if writeln!(stdout, "{:.4} {:.4}", thrust, turn).is_err() {
                break;
            }
            let _ = stdout.flush();
        }

        // Timing control for 20Hz frequency
        next_tick += interval;
        let now = Instant::now();
        if next_tick > now {
            thread::sleep(next_tick - now);
        } else {
            // If we are lagging, reset the next tick to now to avoid a burst of updates
            next_tick = now;
        }
    }
}

/// PD-based control logic to guide the lander to the target safely.
fn compute_control(
    s: &State, tx: f64, ty: f64, g: f64, max_t: f64, 
    skyline: &[f32], skyline_bins: &[f32], bin_size: usize
) -> (f64, f64) {
    let dx = tx - s.x;
    
    // Find the highest obstacle in the path to avoid collisions
    let start_x = (s.x.min(tx) - 2.0).clamp(0.0, (skyline.len() - 1) as f64).round() as isize;
    let end_x = (s.x.max(tx) + 2.0).clamp(0.0, (skyline.len() - 1) as f64).round() as isize;
    
    let mut highest_ground_y = ty as f32;
    let mut curr = start_x;
    while curr <= end_x {
        // Optimization: Use pre-calculated bins to skip large empty areas
        if curr % bin_size as isize == 0 && curr + bin_size as isize - 1 <= end_x {
            let bin = curr as usize / bin_size;
            if skyline_bins[bin] < highest_ground_y {
                highest_ground_y = skyline_bins[bin];
            }
            curr += bin_size as isize;
        } else {
            if curr >= 0 && curr < skyline.len() as isize {
                if skyline[curr as usize] < highest_ground_y {
                    highest_ground_y = skyline[curr as usize];
                }
            }
            curr += 1;
        }
    }
    
    // 1. Horizontal Control: Aim for a target velocity proportional to distance
    let mut desired_vx = dx * 0.7;
    desired_vx = desired_vx.clamp(-20.0, 20.0);
    let ax_req = (desired_vx - s.vx) * 1.5;
    
    // 2. Vertical Control: Stay safely above ground until close to target X
    let safe_y = if dx.abs() > 3.0 {
        // 25m safety margin above any obstacle between here and target
        (highest_ground_y as f64 - 25.0).max(0.0) 
    } else {
        ty
    };
    
    let dy = safe_y - s.y;
    let mut desired_vy = dy * 0.8;
    
    // Safety: limit descent rate near the target pad for soft landing
    if dy.abs() < 5.0 && dx.abs() < 2.5 {
        desired_vy = desired_vy.clamp(-1.5, 1.0); // Slow downward, allow some upward
    } else {
        desired_vy = desired_vy.clamp(-15.0, 15.0);
    }
    
    let ay_req = (desired_vy - s.vy) * 1.5;
    
    // 3. Acceleration Decomposition: Determine thrust and angle
    // ay_total = g - thrust_y => thrust_y = g - ay_req
    let ay_thrust_req = (g - ay_req).max(0.1); // Ensure we don't try to thrust downwards
    
    // Use atan2 to find the angle that provides the required horizontal/vertical acceleration ratio
    let mut desired_angle = ax_req.atan2(ay_thrust_req);
    // Limit tilt to prevent flipping and maintain vertical lift
    desired_angle = desired_angle.clamp(-1.2, 1.2);
    
    // Calculate total thrust needed to achieve the required acceleration vector
    let thrust_req = (ax_req.powi(2) + ay_thrust_req.powi(2)).sqrt() / max_t;
    let thrust = thrust_req.clamp(0.0, 1.0);
    
    // 4. Rotational Control: PD controller for turning rate
    // Target turning rate to achieve desired angle, with damping on angular velocity
    let mut turn = (desired_angle - s.angle) * 12.0 - s.angular_vel * 4.5;
    turn = turn.clamp(-1.0, 1.0);
    
    (thrust, turn)
}
