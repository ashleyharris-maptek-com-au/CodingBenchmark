use std::io::{self, Read, Write, BufWriter};

struct Scanner<'a> {
    data: &'a [u8],
    pos: usize,
}

impl<'a> Scanner<'a> {
    fn new(data: &'a [u8]) -> Self {
        Self { data, pos: 0 }
    }

    fn next_u32(&mut self) -> Option<u32> {
        while self.pos < self.data.len() && !self.data[self.pos].is_ascii_digit() {
            self.pos += 1;
        }
        if self.pos >= self.data.len() {
            return None;
        }
        let mut res = 0u32;
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_digit() {
            res = res.wrapping_mul(10).wrapping_add((self.data[self.pos] - b'0') as u32);
            self.pos += 1;
        }
        Some(res)
    }
}

fn insert_into_bucket(u: i32, d: u32, head: &mut [i32], next: &mut [i32], prev: &mut [i32]) {
    let d = d as usize;
    let u_idx = u as usize;
    next[u_idx] = head[d];
    if head[d] != -1 {
        prev[head[d] as usize] = u;
    }
    head[d] = u;
    prev[u_idx] = -1;
}

fn remove_from_bucket(u: i32, d: u32, head: &mut [i32], next: &mut [i32], prev: &mut [i32]) {
    let d = d as usize;
    let u_idx = u as usize;
    let p = prev[u_idx];
    let n = next[u_idx];
    if p != -1 {
        next[p as usize] = n;
    } else {
        head[d] = n;
    }
    if n != -1 {
        prev[n as usize] = p;
    }
    next[u_idx] = -1;
    prev[u_idx] = -1;
}

fn main() -> io::Result<()> {
    let mut stdin_data = Vec::new();
    io::stdin().read_to_end(&mut stdin_data)?;
    let mut sc = Scanner::new(&stdin_data);

    let n = match sc.next_u32() {
        Some(val) => val as usize,
        None => return Ok(()),
    };
    let m = match sc.next_u32() {
        Some(val) => val as usize,
        None => 0,
    };

    if n == 0 {
        let stdout = io::stdout();
        let mut out = BufWriter::new(stdout.lock());
        writeln!(out, "0")?;
        writeln!(out)?;
        return Ok(());
    }

    let mut adj_offsets = vec![0u32; n + 1];
    let mut edge_data = vec![0u32; 2 * m];
    for i in 0..m {
        let u = sc.next_u32().unwrap_or(0);
        let v = sc.next_u32().unwrap_or(0);
        edge_data[2 * i] = u;
        edge_data[2 * i + 1] = v;
        if (u as usize) < n && (v as usize) < n {
            adj_offsets[u as usize + 1] += 1;
            adj_offsets[v as usize + 1] += 1;
        }
    }

    for i in 0..n {
        adj_offsets[i + 1] += adj_offsets[i];
    }

    let mut adj_data = vec![0u32; adj_offsets[n] as usize];
    let mut current_offsets = adj_offsets.clone();
    for i in 0..m {
        let u = edge_data[2 * i] as usize;
        let v = edge_data[2 * i + 1] as usize;
        if u < n && v < n {
            adj_data[current_offsets[u] as usize] = v as u32;
            current_offsets[u] += 1;
            adj_data[current_offsets[v] as usize] = u as u32;
            current_offsets[v] += 1;
        }
    }

    let mut degree = vec![0u32; n];
    let mut max_deg = 0u32;
    for i in 0..n {
        degree[i] = adj_offsets[i + 1] - adj_offsets[i];
        if degree[i] > max_deg {
            max_deg = degree[i];
        }
    }

    let mut head = vec![-1i32; max_deg as usize + 1];
    let mut next = vec![-1i32; n];
    let mut prev = vec![-1i32; n];
    let mut removed = vec![false; n];
    let mut in_cover = vec![false; n];

    for i in 0..n {
        if degree[i] > 0 {
            insert_into_bucket(i as i32, degree[i], &mut head, &mut next, &mut prev);
        } else {
            removed[i] = true;
        }
    }

    let mut current_max_deg = max_deg;
    while current_max_deg > 0 {
        if head[current_max_deg as usize] == -1 {
            current_max_deg -= 1;
            continue;
        }

        let u_to_add: usize;
        if head[1] != -1 {
            let deg1_u = head[1] as usize;
            let mut neighbor = -1i32;
            let start = adj_offsets[deg1_u] as usize;
            let end = adj_offsets[deg1_u + 1] as usize;
            for j in start..end {
                let v = adj_data[j] as usize;
                if !removed[v] {
                    neighbor = v as i32;
                    break;
                }
            }
            if neighbor == -1 {
                remove_from_bucket(deg1_u as i32, 1, &mut head, &mut next, &mut prev);
                removed[deg1_u] = true;
                continue;
            }
            u_to_add = neighbor as usize;
        } else {
            u_to_add = head[current_max_deg as usize] as usize;
        }

        remove_from_bucket(u_to_add as i32, degree[u_to_add], &mut head, &mut next, &mut prev);
        in_cover[u_to_add] = true;
        removed[u_to_add] = true;

        let start = adj_offsets[u_to_add] as usize;
        let end = adj_offsets[u_to_add + 1] as usize;
        for j in start..end {
            let v = adj_data[j] as usize;
            if !removed[v] {
                let v_deg = degree[v];
                remove_from_bucket(v as i32, v_deg, &mut head, &mut next, &mut prev);
                degree[v] -= 1;
                if degree[v] > 0 {
                    insert_into_bucket(v as i32, degree[v], &mut head, &mut next, &mut prev);
                } else {
                    removed[v] = true;
                }
            }
        }
    }

    let mut count = 0;
    for i in 0..n {
        if in_cover[i] {
            count += 1;
        }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", count)?;
    let mut first = true;
    for i in 0..n {
        if in_cover[i] {
            if !first {
                write!(out, " ")?;
            }
            write!(out, "{}", i)?;
            first = false;
        }
    }
    writeln!(out)?;

    Ok(())
}
