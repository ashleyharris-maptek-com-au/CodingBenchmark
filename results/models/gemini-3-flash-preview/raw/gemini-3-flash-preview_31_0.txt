use std::io::{self, Read, Write};
use std::collections::HashMap;
use std::thread;
use std::sync::mpsc;

struct Scanner {
    buffer: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buffer = Vec::new();
        let _ = io::stdin().read_to_end(&mut buffer);
        Scanner {
            buffer,
            pos: 0,
        }
    }

    fn next_u64(&mut self) -> Option<u64> {
        while self.pos < self.buffer.len() && self.buffer[self.pos] <= b' ' {
            self.pos += 1;
        }
        if self.pos == self.buffer.len() {
            return None;
        }
        let mut res: u64 = 0;
        while self.pos < self.buffer.len() && self.buffer[self.pos] > b' ' {
            res = res.wrapping_mul(10).wrapping_add((self.buffer[self.pos] - b'0') as u64);
            self.pos += 1;
        }
        Some(res)
    }
}

struct Bitset {
    data: Vec<u64>,
    len: usize,
}

impl Bitset {
    fn new(len: usize) -> Self {
        let size = (len + 63) / 64;
        Bitset {
            data: vec![0; size],
            len,
        }
    }

    fn set(&mut self, i: usize) {
        if i < self.len {
            self.data[i >> 6] |= 1 << (i & 63);
        }
    }

    fn get(&self, i: usize) -> bool {
        if i >= self.len {
            return false;
        }
        (self.data[i >> 6] >> (i & 63)) & 1 != 0
    }

    fn shift_or(&mut self, shift: usize) {
        if shift == 0 || shift >= self.len {
            return;
        }
        let s_u64 = shift >> 6;
        let s_bit = shift & 63;
        if s_bit == 0 {
            for i in (s_u64..self.data.len()).rev() {
                self.data[i] |= self.data[i - s_u64];
            }
        } else {
            let s_bit_inv = 64 - s_bit;
            for i in (s_u64 + 1..self.data.len()).rev() {
                let prev = self.data[i - s_u64];
                let prev_prev = self.data[i - s_u64 - 1];
                self.data[i] |= (prev << s_bit) | (prev_prev >> s_bit_inv);
            }
            self.data[s_u64] |= self.data[0] << s_bit;
        }
    }
}

fn generate_sums(nums: &[(u64, usize)]) -> Vec<(u64, u64)> {
    let mut sums = Vec::with_capacity(1 << nums.len());
    sums.push((0, 0));
    for (i, &(v, _)) in nums.iter().enumerate() {
        let len = sums.len();
        for j in 0..len {
            let (s, m) = sums[j];
            sums.push((s + v, m | (1 << i)));
        }
    }
    sums
}

fn solve_mitm(nums: &[(u64, usize)], target: u64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;
    let left = &nums[..mid];
    let right = &nums[mid..];

    let left_sums = generate_sums(left);
    let mut right_sums = generate_sums(right);
    right_sums.sort_unstable_by_key(|x| x.0);

    let num_threads = 12;
    let chunk_size = (left_sums.len() + num_threads - 1) / num_threads;
    let (tx, rx) = mpsc::channel();

    thread::scope(|s| {
        for chunk in left_sums.chunks(chunk_size) {
            let tx = tx.clone();
            let right_sums_ref = &right_sums;
            s.spawn(move || {
                for &(s1, m1) in chunk {
                    if s1 <= target {
                        let rem = target - s1;
                        if let Ok(idx) = right_sums_ref.binary_search_by_key(&rem, |x| x.0) {
                            let (_, m2) = right_sums_ref[idx];
                            let _ = tx.send(Some((m1, m2)));
                            return;
                        }
                    }
                }
            });
        }
    });

    while let Ok(res) = rx.try_recv() {
        if let Some((m1, m2)) = res {
            let mut result = Vec::new();
            for (i, &(_, idx)) in left.iter().enumerate() {
                if (m1 >> i) & 1 != 0 {
                    result.push(idx);
                }
            }
            for (i, &(_, idx)) in right.iter().enumerate() {
                if (m2 >> i) & 1 != 0 {
                    result.push(idx);
                }
            }
            return Some(result);
        }
    }
    None
}

fn compute_bits(items: &[(u64, Vec<usize>)], target: usize) -> Bitset {
    let mut bits = Bitset::new(target + 1);
    bits.set(0);
    for &(v, _) in items {
        if v <= target as u64 {
            bits.shift_or(v as usize);
        }
    }
    bits
}

fn reconstruct_dc(items: &[(u64, Vec<usize>)], target: usize) -> Vec<usize> {
    if target == 0 {
        return vec![];
    }
    if items.len() == 1 {
        if items[0].0 == target as u64 {
            return items[0].1.clone();
        } else {
            unreachable!();
        }
    }

    let mid = items.len() / 2;
    let left = &items[..mid];
    let right = &items[mid..];

    let bits_l = compute_bits(left, target);
    let bits_r = compute_bits(right, target);

    for s in (0..=target).rev() {
        if bits_l.get(s) && bits_r.get(target - s) {
            let mut res = reconstruct_dc(left, s);
            res.extend(reconstruct_dc(right, target - s));
            return res;
        }
    }
    unreachable!()
}

fn solve_bitset(nums: &[(u64, usize)], target: u64) -> Option<Vec<usize>> {
    let target_usize = target as usize;
    let mut counts: HashMap<u64, Vec<usize>> = HashMap::new();
    for &(v, idx) in nums {
        counts.entry(v).or_default().push(idx);
    }

    let mut items = Vec::new();
    let mut sorted_values: Vec<u64> = counts.keys().cloned().collect();
    sorted_values.sort_unstable();

    for v in sorted_values {
        let indices = counts.get(&v).unwrap();
        let mut k = indices.len();
        let mut p = 1;
        let mut start = 0;
        while p <= k {
            let mut group = Vec::new();
            for _ in 0..p {
                group.push(indices[start]);
                start += 1;
            }
            if v * p as u64 <= target {
                items.push((v * p as u64, group));
            }
            k -= p;
            p *= 2;
        }
        if k > 0 {
            let mut group = Vec::new();
            for _ in 0..k {
                group.push(indices[start]);
                start += 1;
            }
            if v * k as u64 <= target {
                items.push((v * k as u64, group));
            }
        }
    }

    let bits = compute_bits(&items, target_usize);
    if !bits.get(target_usize) {
        return None;
    }

    Some(reconstruct_dc(&items, target_usize))
}

struct SimpleRng(u64);
impl SimpleRng {
    fn new(seed: u64) -> Self {
        SimpleRng(seed)
    }
    fn next(&mut self) -> u64 {
        self.0 = self.0.wrapping_mul(6364136223846793005).wrapping_add(1);
        self.0
    }
    fn gen_range(&mut self, end: usize) -> usize {
        if end == 0 { return 0; }
        (self.next() as usize) % end
    }
}

fn solve_randomized(nums: &[(u64, usize)], target: u64) -> Option<Vec<usize>> {
    let mut rng = SimpleRng::new(42);
    let n = nums.len();
    
    let mut current_sum: u64 = 0;
    let mut in_subset = vec![false; n];
    let mut sorted_indices: Vec<usize> = (0..n).collect();
    sorted_indices.sort_unstable_by(|&i, &j| nums[j].0.cmp(&nums[i].0));

    for &i in &sorted_indices {
        if current_sum + nums[i].0 <= target {
            current_sum += nums[i].0;
            in_subset[i] = true;
        }
    }

    if current_sum == target {
        let mut res = Vec::new();
        for i in 0..n {
            if in_subset[i] { res.push(nums[i].1); }
        }
        return Some(res);
    }

    let mut subset_indices = Vec::new();
    let mut not_subset_indices = Vec::new();
    for i in 0..n {
        if in_subset[i] { subset_indices.push(i); }
        else { not_subset_indices.push(i); }
    }

    for _ in 0..20_000_000 {
        let move_type = rng.next() % 3;
        if move_type == 0 && !subset_indices.is_empty() && !not_subset_indices.is_empty() {
            let i_idx = rng.gen_range(subset_indices.len());
            let j_idx = rng.gen_range(not_subset_indices.len());
            let i = subset_indices[i_idx];
            let j = not_subset_indices[j_idx];
            
            let next_sum = current_sum.wrapping_sub(nums[i].0).wrapping_add(nums[j].0);
            if next_sum == target {
                in_subset[i] = false;
                in_subset[j] = true;
                let mut res = Vec::new();
                for k in 0..n { if in_subset[k] { res.push(nums[k].1); } }
                return Some(res);
            }
            if next_sum <= target && (target - next_sum < target - current_sum || (rng.next() % 100 == 0)) {
                current_sum = next_sum;
                in_subset[i] = false;
                in_subset[j] = true;
                subset_indices[i_idx] = j;
                not_subset_indices[j_idx] = i;
            }
        } else if move_type == 1 && !not_subset_indices.is_empty() {
            let j_idx = rng.gen_range(not_subset_indices.len());
            let j = not_subset_indices[j_idx];
            let next_sum = current_sum + nums[j].0;
            if next_sum == target {
                in_subset[j] = true;
                let mut res = Vec::new();
                for k in 0..n { if in_subset[k] { res.push(nums[k].1); } }
                return Some(res);
            }
            if next_sum <= target {
                current_sum = next_sum;
                in_subset[j] = true;
                subset_indices.push(j);
                not_subset_indices.swap_remove(j_idx);
            }
        } else if move_type == 2 && !subset_indices.is_empty() {
            let i_idx = rng.gen_range(subset_indices.len());
            let i = subset_indices[i_idx];
            let next_sum = current_sum - nums[i].0;
            if rng.next() % 100 == 0 {
                current_sum = next_sum;
                in_subset[i] = false;
                not_subset_indices.push(i);
                subset_indices.swap_remove(i_idx);
            }
        }
    }

    None
}

fn print_indices(indices: &[usize]) {
    let mut stdout = io::BufWriter::new(io::stdout());
    let _ = write!(stdout, "[");
    for (i, &idx) in indices.iter().enumerate() {
        if i > 0 {
            let _ = write!(stdout, " ");
        }
        let _ = write!(stdout, "{}", idx);
    }
    let _ = writeln!(stdout, "]");
}

fn main() {
    let mut scanner = Scanner::new();
    let n = match scanner.next_u64() {
        Some(v) => v as usize,
        None => return,
    };
    let target = match scanner.next_u64() {
        Some(v) => v,
        None => return,
    };

    if target == 0 {
        println!("YES");
        println!("[]");
        return;
    }

    let mut nums = Vec::with_capacity(n);
    let mut total_sum: u128 = 0;
    for i in 0..n {
        match scanner.next_u64() {
            Some(v) => {
                if v <= target {
                    nums.push((v, i));
                    total_sum += v as u128;
                }
            }
            None => break,
        }
    }

    if total_sum < target as u128 {
        println!("NO");
        return;
    }

    if total_sum == target as u128 {
        println!("YES");
        let mut res: Vec<usize> = nums.iter().map(|&(_, i)| i).collect();
        res.sort();
        print_indices(&res);
        return;
    }

    if nums.len() <= 40 {
        if let Some(mut res) = solve_mitm(&nums, target) {
            println!("YES");
            res.sort();
            print_indices(&res);
            return;
        } else {
            println!("NO");
            return;
        }
    }

    if target <= 50_000_000 {
        if let Some(mut res) = solve_bitset(&nums, target) {
            println!("YES");
            res.sort();
            print_indices(&res);
            return;
        } else {
            println!("NO");
            return;
        }
    }

    if let Some(mut res) = solve_randomized(&nums, target) {
        println!("YES");
        res.sort();
        print_indices(&res);
        return;
    }

    println!("NO");
}
