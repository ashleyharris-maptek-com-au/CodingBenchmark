#include <iostream>

using namespace std;

// Basic Vector Math
struct Vec3 {
    double x, y, z;
    Vec3 operator+(const Vec3& v) const { return {x + v.x, y + v.y, z + v.z}; }
    Vec3 operator-(const Vec3& v) const { return {x - v.x, y - v.y, z - v.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

struct Triangle {
    int v[3];
};

struct Mesh {
    vector<Vec3> verts;
    vector<Triangle> faces;
};

// Marching Cubes Lookup Tables (Standard)
extern int edgeTable[256];
extern int triTable[256][16];

// Voxelization Grid
struct Voxelizer {
    int RES;
    Vec3 bmin, bmax, bsize;
    vector<uint8_t> grid;

    Voxelizer(int res, Vec3 mi, Vec3 ma) : RES(res), bmin(mi), bmax(ma) {
        bsize = bmax - bmin;
        grid.resize((size_t)RES * RES * RES, 0);
    }

    void voxelize(const Mesh& mesh) {
        vector<vector<float>> ray_events((size_t)RES * RES);
        vector<mutex> row_mutexes(RES);

        int num_threads = thread::hardware_concurrency();
        if (num_threads == 0) num_threads = 8;
        vector<thread> workers;

        for (int t = 0; t < num_threads; ++t) {
            workers.emplace_back([&, t]() {
                for (int i = t; i < (int)mesh.faces.size(); i += num_threads) {
                    const auto& f = mesh.faces[i];
                    Vec3 v0 = mesh.verts[f.v[0]], v1 = mesh.verts[f.v[1]], v2 = mesh.verts[f.v[2]];
                    
                    double xmin = min({v0.x, v1.x, v2.x}), xmax = max({v0.x, v1.x, v2.x});
                    double ymin = min({v0.y, v1.y, v2.y}), ymax = max({v0.y, v1.y, v2.y});
                    
                    int ix_min = max(0, (int)((xmin - bmin.x) / bsize.x * (RES - 1)));
                    int ix_max = min(RES - 1, (int)((xmax - bmin.x) / bsize.x * (RES - 1)));
                    int iy_min = max(0, (int)((ymin - bmin.y) / bsize.y * (RES - 1)));
                    int iy_max = min(RES - 1, (int)((ymax - bmin.y) / bsize.y * (RES - 1)));

                    double det = (v1.y - v2.y) * (v0.x - v2.x) + (v2.x - v1.x) * (v0.y - v2.y);
                    if (abs(det) < 1e-12) continue;

                    for (int ix = ix_min; ix <= ix_max; ++ix) {
                        double px = bmin.x + ix * bsize.x / (RES - 1);
                        for (int iy = iy_min; iy <= iy_max; ++iy) {
                            double py = bmin.y + iy * bsize.y / (RES - 1);
                            
                            // 2D Barycentric coordinates for Point-in-Triangle test
                            double w0 = ((v1.y - v2.y) * (px - v2.x) + (v2.x - v1.x) * (py - v2.y)) / det;
                            double w1 = ((v2.y - v0.y) * (px - v2.x) + (v0.x - v2.x) * (py - v2.y)) / det;
                            double w2 = 1.0 - w0 - w1;

                            if (w0 >= 0 && w1 >= 0 && w2 >= 0) {
                                float z = (float)(w0 * v0.z + w1 * v1.z + w2 * v2.z);
                                lock_guard<mutex> lock(row_mutexes[iy]);
                                ray_events[iy * RES + ix].push_back(z);
                            }
                        }
                    }
                }
            });
        }
        for (auto& w : workers) w.join();

        // Fill grid using even-odd rule
        vector<thread> fillers;
        for (int t = 0; t < num_threads; ++t) {
            fillers.emplace_back([&, t]() {
                for (int i = t; i < RES * RES; i += num_threads) {
                    auto& evs = ray_events[i];
                    if (evs.empty()) continue;
                    sort(evs.begin(), evs.end());
                    int ix = i % RES;
                    int iy = i / RES;
                    for (size_t j = 0; j + 1 < evs.size(); j += 2) {
                        int z_start = max(0, (int)((evs[j] - bmin.z) / bsize.z * (RES - 1)));
                        int z_end = min(RES - 1, (int)((evs[j + 1] - bmin.z) / bsize.z * (RES - 1)));
                        for (int iz = z_start; iz <= z_end; ++iz) {
                            grid[(size_t)iz * RES * RES + iy * RES + ix] = 1;
                        }
                    }
                }
            });
        }
        for (auto& w : fillers) w.join();
    }
};

// Marching Cubes Implementation
void marchingCubes(const vector<uint8_t>& grid, int RES, Vec3 bmin, Vec3 bsize, Mesh& out) {
    auto get_idx = [&](int x, int y, int z) { return (size_t)z * RES * RES + y * RES + x; };
    vector<int> edge_to_vert((size_t)3 * RES * RES * RES, -1);
    mutex out_mutex;

    int num_threads = thread::hardware_concurrency();
    vector<thread> workers;
    for (int t = 0; t < num_threads; ++t) {
        workers.emplace_back([&, t]() {
            vector<Vec3> local_verts;
            vector<Triangle> local_faces;
            for (int iz = t; iz < RES - 1; iz += num_threads) {
                for (int iy = 0; iy < RES - 1; ++iy) {
                    for (int ix = 0; ix < RES - 1; ++ix) {
                        int cube_idx = 0;
                        if (grid[get_idx(ix, iy, iz)]) cube_idx |= 1;
                        if (grid[get_idx(ix + 1, iy, iz)]) cube_idx |= 2;
                        if (grid[get_idx(ix + 1, iy + 1, iz)]) cube_idx |= 4;
                        if (grid[get_idx(ix, iy + 1, iz)]) cube_idx |= 8;
                        if (grid[get_idx(ix, iy, iz + 1)]) cube_idx |= 16;
                        if (grid[get_idx(ix + 1, iy, iz + 1)]) cube_idx |= 32;
                        if (grid[get_idx(ix + 1, iy + 1, iz + 1)]) cube_idx |= 64;
                        if (grid[get_idx(ix, iy + 1, iz + 1)]) cube_idx |= 128;

                        if (edgeTable[cube_idx] == 0) continue;

                        int edge_indices[12];
                        auto get_edge_v = [&](int x, int y, int z, int type) -> int {
                            size_t eid = (size_t)3 * ((size_t)z * RES * RES + y * RES + x) + type;
                            // For simplicity in a single file, we'll recreate vertices. 
                            // To make it manifold, we'd need a global edge-map, but 
                            // creating unique vertices per cell is faster and still valid for STL.
                            // However, let's do a simple vertex calculation:
                            Vec3 p;
                            if (type == 0) p = bmin + Vec3{ (x + 0.5) * bsize.x / (RES - 1), (double)y * bsize.y / (RES - 1), (double)z * bsize.z / (RES - 1) };
                            else if (type == 1) p = bmin + Vec3{ (double)x * bsize.x / (RES - 1), (y + 0.5) * bsize.y / (RES - 1), (double)z * bsize.z / (RES - 1) };
                            else p = bmin + Vec3{ (double)x * bsize.x / (RES - 1), (double)y * bsize.y / (RES - 1), (z + 0.5) * bsize.z / (RES - 1) };
                            int idx = (int)local_verts.size();
                            local_verts.push_back(p);
                            return idx;
                        };

                        if (edgeTable[cube_idx] & 1) edge_indices[0] = get_edge_v(ix, iy, iz, 0);
                        if (edgeTable[cube_idx] & 2) edge_indices[1] = get_edge_v(ix + 1, iy, iz, 1);
                        if (edgeTable[cube_idx] & 4) edge_indices[2] = get_edge_v(ix, iy + 1, iz, 0);
                        if (edgeTable[cube_idx] & 8) edge_indices[3] = get_edge_v(ix, iy, iz, 1);
                        if (edgeTable[cube_idx] & 16) edge_indices[4] = get_edge_v(ix, iy, iz + 1, 0);
                        if (edgeTable[cube_idx] & 32) edge_indices[5] = get_edge_v(ix + 1, iy, iz + 1, 1);
                        if (edgeTable[cube_idx] & 64) edge_indices[6] = get_edge_v(ix, iy + 1, iz + 1, 0);
                        if (edgeTable[cube_idx] & 128) edge_indices[7] = get_edge_v(ix, iy, iz + 1, 1);
                        if (edgeTable[cube_idx] & 256) edge_indices[8] = get_edge_v(ix, iy, iz, 2);
                        if (edgeTable[cube_idx] & 512) edge_indices[9] = get_edge_v(ix + 1, iy, iz, 2);
                        if (edgeTable[cube_idx] & 1024) edge_indices[10] = get_edge_v(ix + 1, iy + 1, iz, 2);
                        if (edgeTable[cube_idx] & 2048) edge_indices[11] = get_edge_v(ix, iy + 1, iz, 2);

                        for (int i = 0; triTable[cube_idx][i] != -1; i += 3) {
                            local_faces.push_back({ {edge_indices[triTable[cube_idx][i]], 
                                                     edge_indices[triTable[cube_idx][i+1]], 
                                                     edge_indices[triTable[cube_idx][i+2]]} });
                        }
                    }
                }
            }
            lock_guard<mutex> lock(out_mutex);
            int offset = (int)out.verts.size();
            for (auto& f : local_faces) { f.v[0] += offset; f.v[1] += offset; f.v[2] += offset; }
            out.verts.insert(out.verts.end(), local_verts.begin(), local_verts.end());
            out.faces.insert(out.faces.end(), local_faces.begin(), local_faces.end());
        });
    }
    for (auto& w : workers) w.join();
}
