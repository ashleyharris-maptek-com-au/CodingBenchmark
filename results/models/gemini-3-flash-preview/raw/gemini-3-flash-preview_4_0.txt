#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <cstdio>
#include <chrono>

using namespace std;

// Point structure to represent 3D coordinates
struct Point {
    double x, y, z;
};

// Plane structure to represent a face of the convex polyhedron (ax + by + cz + d <= 0)
struct Plane {
    double a, b, c, d;
};

// Function to check if a point is inside the polyhedron defined by face planes
bool is_inside(const Point& p, const vector<Plane>& planes) {
    for (const auto& plane : planes) {
        if (plane.a * p.x + plane.b * p.y + plane.c * p.z + plane.d > 1e-8) {
            return false;
        }
    }
    return true;
}

int main() {
    int V;
    double e_len;
    // Read number of vertices and tetrahedron edge length
    if (scanf("%d %lf", &V, &e_len) != 2) return 0;

    auto start_time = chrono::steady_clock::now();

    vector<Point> container_verts(V);
    double min_x = 1e18, max_x = -1e18;
    double min_y = 1e18, max_y = -1e18;
    double min_z = 1e18, max_z = -1e18;

    // Read container vertices and calculate bounding box
    for (int i = 0; i < V; ++i) {
        if (scanf("%lf %lf %lf", &container_verts[i].x, &container_verts[i].y, &container_verts[i].z) != 3) break;
        min_x = min(min_x, container_verts[i].x);
        max_x = max(max_x, container_verts[i].x);
        min_y = min(min_y, container_verts[i].y);
        max_y = max(max_y, container_verts[i].y);
        min_z = min(min_z, container_verts[i].z);
        max_z = max(max_z, container_verts[i].z);
    }

    vector<Plane> planes;
    // Find face planes of the convex hull.
    // O(V^4) complexity is acceptable for small V (up to ~200) within the time limit.
    if (V <= 200) {
        for (int i = 0; i < V; ++i) {
            for (int j = i + 1; j < V; ++j) {
                for (int k = j + 1; k < V; ++k) {
                    // Compute plane normal for vertices i, j, k
                    double ax = container_verts[j].x - container_verts[i].x;
                    double ay = container_verts[j].y - container_verts[i].y;
                    double az = container_verts[j].z - container_verts[i].z;
                    double bx = container_verts[k].x - container_verts[i].x;
                    double by = container_verts[k].y - container_verts[i].y;
                    double bz = container_verts[k].z - container_verts[i].z;

                    double nx = ay * bz - az * by;
                    double ny = az * bx - ax * bz;
                    double nz = ax * by - ay * bx;

                    double norm_sq = nx * nx + ny * ny + nz * nz;
                    if (norm_sq < 1e-12) continue;
                    double norm = sqrt(norm_sq);
                    nx /= norm; ny /= norm; nz /= norm;
                    double d = -(nx * container_verts[i].x + ny * container_verts[i].y + nz * container_verts[i].z);

                    // Check if all other points lie on one side of this plane
                    int side = 0;
                    bool possible = true;
                    for (int m = 0; m < V; ++m) {
                        double val = nx * container_verts[m].x + ny * container_verts[m].y + nz * container_verts[m].z + d;
                        if (abs(val) < 1e-7) continue;
                        if (side == 0) side = (val > 0 ? 1 : -1);
                        else if ((val > 0 && side == -1) || (val < 0 && side == 1)) {
                            possible = false;
                            break;
                        }
                    }

                    if (possible && side != 0) {
                        if (side == 1) { nx = -nx; ny = -ny; nz = -nz; d = -d; }
                        // Filter duplicate planes
                        bool duplicate = false;
                        for (const auto& p : planes) {
                            if (abs(p.a - nx) < 1e-7 && abs(p.b - ny) < 1e-7 && abs(p.c - nz) < 1e-7 && abs(p.d - d) < 1e-7) {
                                duplicate = true; break;
                            }
                        }
                        if (!duplicate) planes.push_back({nx, ny, nz, d});
                    }
                }
            }
        }
    } else {
        // Fallback for massive V: use the container's axis-aligned bounding box as a simple container
        planes.push_back({1, 0, 0, -max_x});
        planes.push_back({-1, 0, 0, min_x});
        planes.push_back({0, 1, 0, -max_y});
        planes.push_back({0, -1, 0, min_y});
        planes.push_back({0, 0, 1, -max_z});
        planes.push_back({0, 0, -1, min_z});
    }

    // A regular tetrahedron can be inscribed in a cube of side a = e / sqrt(2).
    // The density of this packing is 1/3.
    double s = e_len / (2.0 * sqrt(2.0));
    Point tet_local[4] = {
        {s, s, s}, {s, -s, -s}, {-s, s, -s}, {-s, -s, s}
    };

    double step = e_len / sqrt(2.0);
    vector<Point> placements;
    placements.reserve(1000000);

    // Grid-based packing within the container's bounding box
    for (double x = min_x + s; x <= max_x - s; x += step) {
        for (double y = min_y + s; y <= max_y - s; y += step) {
            for (double z = min_z + s; z <= max_z - s; z += step) {
                bool all_inside = true;
                // Check if all four vertices of the tetrahedron are inside the container
                for (int i = 0; i < 4; ++i) {
                    if (!is_inside({x + tet_local[i].x, y + tet_local[i].y, z + tet_local[i].z}, planes)) {
                        all_inside = false;
                        break;
                    }
                }
                if (all_inside) {
                    placements.push_back({x, y, z});
                    // Cap the number of tetrahedrons to ensure timely output
                    if (placements.size() >= 4000000) goto finish_search;
                }
            }
        }
        // Time check to ensure completion within 30 seconds
        auto now = chrono::steady_clock::now();
        if (chrono::duration_cast<chrono::seconds>(now - start_time).count() > 27) {
            goto finish_search;
        }
    }

finish_search:

    // Output the number of tetrahedrons placed
    printf("%zu\n", placements.size());
    // Output center coordinates and Euler angles (0, 0, 0 as we use a fixed orientation)
    for (const auto& p : placements) {
        printf("%.8f %.8f %.8f 0.000000 0.000000 0.000000\n", p.x, p.y, p.z);
    }

    return 0;
}
