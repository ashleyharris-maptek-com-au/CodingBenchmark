#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <set>
#include <cstdio>
#include <ctime>

/**
 * 3D Packing Problem: Regular Tetrahedrons into Polyhedra
 * 
 * Algorithm:
 * 1. Compute the convex hull planes of the container vertices using a simplified O(V^2 to V^3) 
 *    face-finding algorithm. For massive vertex counts, sample extremal points to build a 
 *    valid interior hull.
 * 2. Utilize a grid-based packing strategy with a grid cell size of e/sqrt(2).
 *    A regular tetrahedron fits perfectly inside a cube of side e/sqrt(2), ensuring 
 *    that disjoint grid cells result in non-overlapping tetrahedrons.
 * 3. Use OpenMP to parallelize the grid search across all 64 cores.
 * 4. Verify each tetrahedron's vertices against the container planes to ensure 
 *    it is completely inside the polyhedron.
 * 5. Scale the grid iterations based on container volume and edge length to complete within 30s.
 */

#if defined(_OPENMP)
#include <omp.h>
#else
inline int omp_get_max_threads() { return 1; }
inline int omp_get_thread_num() { return 0; }
#endif

using namespace std;

struct Point {
    double x, y, z;
    Point operator-(const Point& other) const { return {x - other.x, y - other.y, z - other.z}; }
    double dot(const Point& other) const { return x * other.x + y * other.y + z * other.z; }
    Point cross(const Point& other) const {
        return {y * other.z - z * other.y, z * other.x - x * other.z, x * other.y - y * other.x};
    }
    double norm() const { return sqrt(x * x + y * y + z * z + 1e-18); }
};

struct Plane {
    Point n;
    double d;
    // Checks if a point satisfies the plane equation ax + by + cz <= d
    inline bool isInside(const Point& p) const {
        return n.x * p.x + n.y * p.y + n.z * p.z <= d + 1e-7;
    }
};

// Finds all faces of the convex hull defined by vertices pts
vector<Plane> getHullPlanes(const vector<Point>& pts) {
    int n = (int)pts.size();
    vector<Plane> planes;
    if (n < 4) return planes;

    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Point normal = (pts[j] - pts[i]).cross(pts[k] - pts[i]);
                double mag = normal.norm();
                if (mag < 1e-9) continue;
                normal.x /= mag; normal.y /= mag; normal.z /= mag;
                double d = normal.dot(pts[i]);

                bool all_below = true, all_above = true;
                for (int l = 0; l < n; ++l) {
                    double dt = normal.dot(pts[l]);
                    if (dt > d + 1e-7) all_below = false;
                    if (dt < d - 1e-7) all_above = false;
                    if (!all_below && !all_above) break;
                }

                if (all_below || all_above) {
                    if (all_above) {
                        normal.x = -normal.x; normal.y = -normal.y; normal.z = -normal.z;
                        d = -d;
                    }
                    // Filter duplicate planes
                    bool duplicate = false;
                    for (const auto& plane : planes) {
                        if (abs(plane.n.dot(normal) - 1.0) < 1e-8 && abs(plane.d - d) < 1e-8) {
                            duplicate = true; break;
                        }
                    }
                    if (!duplicate) planes.push_back({normal, d});
                }
            }
        }
    }
    return planes;
}

int main() {
    int V_count;
    double edge_len;
    if (scanf("%d %lf", &V_count, &edge_len) != 2) return 0;

    vector<Point> container_pts(V_count);
    double minX = 1e18, maxX = -1e18, minY = 1e18, maxY = -1e18, minZ = 1e18, maxZ = -1e18;
    for (int i = 0; i < V_count; ++i) {
        scanf("%lf %lf %lf", &container_pts[i].x, &container_pts[i].y, &container_pts[i].z);
        minX = min(minX, container_pts[i].x); maxX = max(maxX, container_pts[i].x);
        minY = min(minY, container_pts[i].y); maxY = max(maxY, container_pts[i].y);
        minZ = min(minZ, container_pts[i].z); maxZ = max(maxZ, container_pts[i].z);
    }

    // Simplify the input for convex hull calculation if vertex count is huge
    srand(42);
    vector<Point> hull_input = container_pts;
    if (V_count > 64) {
        set<int> extreme_indices;
        for (int i = 0; i < 256; ++i) {
            double theta = (double)rand() / RAND_MAX * 2.0 * 3.1415926535;
            double phi = (double)rand() / RAND_MAX * 3.1415926535;
            Point dir = { cos(theta)*sin(phi), sin(theta)*sin(phi), cos(phi) };
            double max_dot = -1e18; int best_idx = -1;
            for (int j = 0; j < V_count; ++j) {
                double dt = container_pts[j].x * dir.x + container_pts[j].y * dir.y + container_pts[j].z * dir.z;
                if (dt > max_dot) { max_dot = dt; best_idx = j; }
            }
            if (best_idx != -1) extreme_indices.insert(best_idx);
        }
        hull_input.clear();
        for (int idx : extreme_indices) hull_input.push_back(container_pts[idx]);
    }

    vector<Plane> planes = getHullPlanes(hull_input);

    // Geometry: s is half the side of the bounding cube of the tetrahedron
    double s = edge_len / sqrt(8.0);
    double S = edge_len / sqrt(2.0); // Side of the cube container for the tet

    int num_threads = omp_get_max_threads();
    if (num_threads > 64) num_threads = 64;
    vector<Point> local_results[64];

    long long nx = (long long)((maxX - minX) / S) + 2;
    long long ny = (long long)((maxY - minY) / S) + 2;
    long long nz = (long long)((maxZ - minZ) / S) + 2;
    
    // Safety cap to prevent memory overflow or excessive compute on ludicrous scales
    long long total_iter = nx * ny * nz;
    if (total_iter > 300000000) { 
        double ratio = pow(300000000.0 / total_iter, 1.0/3.0);
        nx = (long long)(nx * ratio); 
        ny = (long long)(ny * ratio); 
        nz = (long long)(nz * ratio);
    }
    if (nx < 1) nx = 1; if (ny < 1) ny = 1; if (nz < 1) nz = 1;

    // Parallel grid processing
    #pragma omp parallel for schedule(dynamic)
    for (long long i = 0; i < nx; ++i) {
        int tid = omp_get_thread_num();
        if (local_results[tid].size() > 2500000) continue; // Output memory cap per thread
        double cx = minX + s + i * S;
        for (long long j = 0; j < ny; ++j) {
            double cy = minY + s + j * S;
            for (long long k = 0; k < nz; ++k) {
                double cz = minZ + s + k * S;
                // Vertices for a regular tetrahedron centered at (cx, cy, cz)
                Point v[4] = { 
                    {cx + s, cy + s, cz + s}, 
                    {cx + s, cy - s, cz - s}, 
                    {cx - s, cy + s, cz - s}, 
                    {cx - s, cy - s, cz + s} 
                };
                
                bool inside = true;
                for (int m = 0; m < 4; ++m) {
                    for (const auto& plane : planes) {
                        if (!plane.isInside(v[m])) { inside = false; break; }
                    }
                    if (!inside) break;
                }
                if (inside) {
                    local_results[tid].push_back({cx, cy, cz});
                }
            }
        }
    }

    // Result collection
    long long final_count = 0;
    for (int i = 0; i < num_threads; ++i) final_count += local_results[i].size();

    // Standard output
    printf("%lld\n", final_count);
    for (int i = 0; i < num_threads; ++i) {
        for (const auto& p : local_results[i]) {
            // center + Euler angles (0 0 0 corresponds to the base orientation)
            printf("%.6f %.6f %.6f 0.0 0.0 0.0\n", p.x, p.y, p.z);
        }
    }

    return 0;
}
