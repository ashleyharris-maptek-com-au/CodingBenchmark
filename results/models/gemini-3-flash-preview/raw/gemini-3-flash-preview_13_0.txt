#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

/**
 * Senior Engineer Note:
 * For the Longest Common Substring (LCS) of N strings, a Generalized Suffix Automaton 
 * or building a SAM on the shortest string and matching others against it is highly efficient.
 * Time complexity: O(\sum |S_i|)
 * Space complexity: O(|S_min|) for the SAM structure, plus the memory for the strings themselves.
 * 
 * To handle "Extreme" scale (1GB+ total text) within 32GB RAM:
 * 1. Build SAM on the SHORTEST string to minimize automaton states.
 * 2. Use a memory-compact transition storage (sorted vectors instead of maps/arrays).
 * 3. Clear/shrink strings as they are processed to stay within memory limits.
 * 4. Use Fast I/O for large inputs.
 */

struct State {
    int len, link, first_pos, min_match;
};

struct RawTrans {
    int to;
    int next;
    unsigned char c;
};

struct CompactTrans {
    int to;
    unsigned char c;
};

vector<State> st;
vector<RawTrans> raw_trans;
vector<int> head;
int last;

void sam_init(size_t n) {
    st.clear();
    // For n chars, at most 2n-1 states. reserve slightly more for safety.
    st.reserve(2 * n + 5);
    raw_trans.clear();
    raw_trans.reserve(3 * n + 5);
    head.clear();
    head.assign(2 * n + 5, -1);
    st.push_back({0, -1, -1, 0});
    last = 0;
}

void add_raw_trans(int u, unsigned char c, int v) {
    raw_trans.push_back({v, head[u], c});
    head[u] = (int)raw_trans.size() - 1;
}

int get_raw_trans(int u, unsigned char c) {
    for (int i = head[u]; i != -1; i = raw_trans[i].next) {
        if (raw_trans[i].c == c) return raw_trans[i].to;
    }
    return -1;
}

void update_raw_trans(int u, unsigned char c, int new_v) {
    for (int i = head[u]; i != -1; i = raw_trans[i].next) {
        if (raw_trans[i].c == c) {
            raw_trans[i].to = new_v;
            return;
        }
    }
}

void sam_extend(unsigned char c, int pos) {
    int cur = (int)st.size();
    st.push_back({st[last].len + 1, 0, pos, st[last].len + 1});
    if (head.size() <= (size_t)cur) head.push_back(-1);
    int p = last;
    while (p != -1 && get_raw_trans(p, c) == -1) {
        add_raw_trans(p, c, cur);
        p = st[p].link;
    }
    if (p == -1) {
        st[cur].link = 0;
    } else {
        int q = get_raw_trans(p, c);
        if (st[p].len + 1 == st[q].len) {
            st[cur].link = q;
        } else {
            int clone = (int)st.size();
            st.push_back({st[p].len + 1, st[q].link, st[q].first_pos, st[p].len + 1});
            if (head.size() <= (size_t)clone) head.push_back(-1);
            for (int i = head[q]; i != -1; i = raw_trans[i].next) {
                add_raw_trans(clone, raw_trans[i].c, raw_trans[i].to);
            }
            while (p != -1 && get_raw_trans(p, c) == q) {
                update_raw_trans(p, c, clone);
                p = st[p].link;
            }
            st[q].link = st[cur].link = clone;
        }
    }
    last = cur;
}

vector<CompactTrans> compact_trans;
vector<int> trans_offset;

// Compact the transition storage to save memory and speed up matching (binary search)
void sam_compact() {
    trans_offset.assign(st.size() + 1, 0);
    compact_trans.clear();
    compact_trans.reserve(raw_trans.size());
    for (int i = 0; i < (int)st.size(); ++i) {
        trans_offset[i] = (int)compact_trans.size();
        for (int j = head[i]; j != -1; j = raw_trans[j].next) {
            compact_trans.push_back({raw_trans[j].to, raw_trans[j].c});
        }
        sort(compact_trans.begin() + trans_offset[i], compact_trans.end(), [](CompactTrans a, CompactTrans b) {
            return a.c < b.c;
        });
    }
    trans_offset[st.size()] = (int)compact_trans.size();
    // Free raw transition pool memory
    raw_trans.clear();
    raw_trans.shrink_to_fit();
    head.clear();
    head.shrink_to_fit();
}

inline int get_trans(int u, unsigned char c) {
    int l = trans_offset[u], r = trans_offset[u + 1] - 1;
    while (l <= r) {
        int m = l + (r - l) / 2;
        if (compact_trans[m].c == c) return compact_trans[m].to;
        if (compact_trans[m].c < c) l = m + 1;
        else r = m - 1;
    }
    return -1;
}

void solve() {
    int N;
    if (!(cin >> N)) return;
    string dummy;
    getline(cin, dummy); // Consume leftover newline

    if (N <= 0) {
        cout << endl;
        return;
    }

    vector<string> s(N);
    int min_len = -1;
    int min_idx = 0;
    for (int i = 0; i < N; ++i) {
        if (!getline(cin, s[i])) break;
        if (!s[i].empty() && s[i].back() == '\r') s[i].pop_back(); // Handle CRLF
        if (min_len == -1 || (int)s[i].length() < min_len) {
            min_len = (int)s[i].length();
            min_idx = i;
        }
    }

    if (min_len == 0) {
        cout << endl;
        return;
    }

    if (N == 1) {
        cout << s[0] << endl;
        return;
    }

    // Always build SAM on the shortest string for efficiency
    swap(s[0], s[min_idx]);

    sam_init(min_len);
    for (int i = 0; i < min_len; ++i) {
        sam_extend((unsigned char)s[0][i], i);
    }

    sam_compact();

    // Prepare topological sort by length (to propagate match info from longer to shorter substrings)
    vector<int> states_by_len(st.size());
    {
        int max_l = 0;
        for (auto& state : st) if (state.len > max_l) max_l = state.len;
        vector<int> count(max_l + 1, 0);
        for (auto& state : st) count[state.len]++;
        for (int i = 1; i <= max_l; ++i) count[i] += count[i - 1];
        for (int i = (int)st.size() - 1; i >= 0; --i) {
            states_by_len[--count[st[i].len]] = i;
        }
        reverse(states_by_len.begin(), states_by_len.end()); // Large to small for propagation
    }

    vector<int> temp_match(st.size());
    for (int i = 1; i < N; ++i) {
        fill(temp_match.begin(), temp_match.end(), 0);
        int v = 0, l = 0;
        for (unsigned char c : s[i]) {
            while (v != 0 && get_trans(v, c) == -1) {
                v = st[v].link;
                l = st[v].len;
            }
            int next_v = get_trans(v, c);
            if (next_v != -1) {
                v = next_v;
                l++;
            } else {
                v = 0;
                l = 0;
            }
            if (l > temp_match[v]) temp_match[v] = l;
        }

        // Propagate matches up the suffix link tree
        for (int v : states_by_len) {
            if (v == 0) continue;
            int link = st[v].link;
            if (link != -1) {
                int m = min(st[link].len, temp_match[v]);
                if (m > temp_match[link]) temp_match[link] = m;
            }
            if (temp_match[v] < st[v].min_match) st[v].min_match = temp_match[v];
        }
        // Free memory as we go
        s[i].clear();
        s[i].shrink_to_fit();
    }

    int best_len = 0;
    int best_state = 0;
    for (int i = 0; i < (int)st.size(); ++i) {
        if (st[i].min_match > best_len) {
            best_len = st[i].min_match;
            best_state = i;
        }
    }

    if (best_len == 0) {
        cout << endl;
    } else {
        cout.write(&s[0][st[best_state].first_pos - best_len + 1], (streamsize)best_len);
        cout << endl;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
