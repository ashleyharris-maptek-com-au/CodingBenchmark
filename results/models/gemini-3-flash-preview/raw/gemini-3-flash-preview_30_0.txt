import sys
import array
import re
import random

def solve():
    # Set seed for reproducibility
    random.seed(42)
    
    # Use fast I/O
    stdin_buffer = sys.stdin.buffer
    
    # Read n_vars and n_clauses
    header = []
    while len(header) < 2:
        line = stdin_buffer.readline()
        if not line:
            break
        # Using re.finditer for robust parsing of the header
        for m in re.finditer(rb"-?\d+", line):
            header.append(int(m.group()))
            if len(header) == 2:
                break
    
    if len(header) < 2:
        return
        
    n_vars = header[0]
    n_clauses = header[1]
    
    # Read the rest of the literals
    # We read all at once to speed up, but use an iterator for parsing to save memory.
    # For 85M clauses, this could be around 1GB of text.
    data = stdin_buffer.read()
    
    # Pre-allocate array for clauses (3 literals per clause)
    # Using 'i' for signed 32-bit integers (4 bytes per element)
    # 255M integers * 4 bytes = 1.02 GB.
    clauses = array.array('i', [0] * (3 * n_clauses))
    
    # Parse literals
    # re.finditer is more memory-efficient than re.findall or data.split() 
    # as it avoids creating a massive list of all matches at once.
    idx = 0
    for m in re.finditer(rb"-?\d+", data):
        if idx >= 3 * n_clauses:
            break
        clauses[idx] = int(m.group())
        idx += 1
    
    # Free raw data memory as soon as possible
    del data
    
    # Actual number of literals found
    num_literals = idx
    
    # Initialize assignment
    # Use 'b' for signed 8-bit integers (1 byte per variable)
    # 20M variables * 1 byte = 20 MB.
    # Initialize all to 1 (True) as a starting point.
    assignment = array.array('b', b'\x01' * (n_vars + 1))
    
    # Local references for speed in the inner loop
    a = assignment
    c = clauses
    
    # Local search - simple greedy repair (Schöning's-like heuristic)
    # With 85M clauses, we can only afford a few passes in Python.
    # Since instances are guaranteed to be satisfiable, a few passes are often enough.
    for pass_idx in range(4):
        any_unsatisfied = False
        for i in range(0, num_literals, 3):
            # i is 0, 3, 6, ...
            # We assume exactly 3 literals per clause as per 3-SAT definition.
            l1 = c[i]
            l2 = c[i+1]
            l3 = c[i+2]
            
            # Check if clause is satisfied (at least one literal is True)
            # A literal l is True if (l > 0 and a[l] == 1) or (l < 0 and a[-l] == 0)
            
            satisfied = False
            # Check Literal 1
            if l1 > 0:
                if a[l1]: satisfied = True
            else:
                if not a[-l1]: satisfied = True
            
            if not satisfied:
                # Check Literal 2
                if l2 > 0:
                    if a[l2]: satisfied = True
                else:
                    if not a[-l2]: satisfied = True
            
            if not satisfied:
                # Check Literal 3
                if l3 > 0:
                    if a[l3]: satisfied = True
                else:
                    if not a[-l3]: satisfied = True
            
            if not satisfied:
                any_unsatisfied = True
                # Pick one variable to flip to satisfy this clause.
                # Use a rotating choice that varies by clause and pass.
                choice = (i // 3 + pass_idx) % 3
                if choice == 0: target = l1
                elif choice == 1: target = l2
                else: target = l3
                
                # Flip the variable of the chosen literal
                if target > 0:
                    a[target] = 1
                else:
                    a[-target] = 0
                    
        if not any_unsatisfied:
            break
            
    # Final output
    sys.stdout.write("SAT\n")
    # For large n_vars, creating a single string for join is better than many print calls.
    # Using a generator for str(a[i]) avoids creating a large intermediate list.
    output = " ".join(str(a[i]) for i in range(1, n_vars + 1))
    sys.stdout.write(output + "\n")

if __name__ == "__main__":
    solve()
