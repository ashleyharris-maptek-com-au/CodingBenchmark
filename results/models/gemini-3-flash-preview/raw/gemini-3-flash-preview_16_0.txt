#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>
#include <cstdint>

using namespace std;

struct Task {
    int machine;
    int duration;
};

struct Job {
    int id;
    vector<Task> tasks;
    int next_task_idx = 0;
    long long ready_time = 0;
    long long work_remaining = 0;
};

struct ScheduleResult {
    vector<vector<long long>> start_times;
    long long makespan;
};

ScheduleResult solve_with_priority(int J, int M, const vector<vector<Task>>& job_data, int rule) {
    vector<Job> jobs(J);
    int total_tasks = 0;
    for (int i = 0; i < J; ++i) {
        jobs[i].id = i;
        jobs[i].tasks = job_data[i];
        jobs[i].work_remaining = 0;
        for (const auto& t : jobs[i].tasks) {
            jobs[i].work_remaining += t.duration;
        }
        total_tasks += (int)jobs[i].tasks.size();
    }

    vector<long long> machine_free_time(M, 0);
    vector<vector<long long>> start_times(J);
    for (int i = 0; i < J; ++i) {
        start_times[i].resize(job_data[i].size());
    }

    vector<int> ready_jobs;
    ready_jobs.reserve(J);
    for (int i = 0; i < J; ++i) {
        if (!jobs[i].tasks.empty()) {
            ready_jobs.push_back(i);
        }
    }

    int completed_tasks = 0;
    while (completed_tasks < total_tasks) {
        int best_ready_idx = -1;
        long long best_priority = numeric_limits<long long>::min();

        // Tight loop for priority selection
        for (int i = 0; i < (int)ready_jobs.size(); ++i) {
            int j_id = ready_jobs[i];
            const Job& job = jobs[j_id];
            const Task& task = job.tasks[job.next_task_idx];
            
            long long priority;
            switch (rule) {
                case 0: priority = job.work_remaining; break; // MWR
                case 1: priority = -(long long)task.duration; break; // SPT
                case 2: priority = -(long long)j_id; break; // FIFO
                case 3: 
                default:
                    priority = -max(job.ready_time, machine_free_time[task.machine]); break; // EST
            }

            if (best_ready_idx == -1 || priority > best_priority) {
                best_priority = priority;
                best_ready_idx = i;
            }
        }

        int j_id = ready_jobs[best_ready_idx];
        Job& job = jobs[j_id];
        const Task& task = job.tasks[job.next_task_idx];

        long long start_t = max(job.ready_time, machine_free_time[task.machine]);
        start_times[j_id][job.next_task_idx] = start_t;
        
        long long finish_t = start_t + task.duration;
        machine_free_time[task.machine] = finish_t;
        job.ready_time = finish_t;
        job.work_remaining -= task.duration;
        job.next_task_idx++;

        if (job.next_task_idx == (int)job.tasks.size()) {
            ready_jobs[best_ready_idx] = ready_jobs.back();
            ready_jobs.pop_back();
        }
        completed_tasks++;
    }

    long long makespan = 0;
    for (int i = 0; i < M; ++i) {
        if (machine_free_time[i] > makespan) makespan = machine_free_time[i];
    }

    return {move(start_times), makespan};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> job_data(J);
    long long total_tasks = 0;
    for (int i = 0; i < J; ++i) {
        int T;
        if (!(cin >> T)) break;
        job_data[i].reserve(T);
        for (int j = 0; j < T; ++j) {
            int m, d;
            cin >> m >> d;
            job_data[i].push_back({m, d});
        }
        total_tasks += T;
    }

    ScheduleResult best_res;
    long long min_makespan = numeric_limits<long long>::max();

    int rules_to_try = 1;
    if (total_tasks < 2000) rules_to_try = 4;
    else if (total_tasks < 20000) rules_to_try = 2;

    for (int r = 0; r < rules_to_try; ++r) {
        ScheduleResult res = solve_with_priority(J, M, job_data, r);
        if (r == 0 || res.makespan < min_makespan) {
            min_makespan = res.makespan;
            best_res = move(res);
        }
    }

    for (int i = 0; i < J; ++i) {
        for (int j = 0; j < (int)best_res.start_times[i].size(); ++j) {
            cout << best_res.start_times[i][j] << " " << job_data[i][j].machine << (j == (int)best_res.start_times[i].size() - 1 ? "" : " ");
        }
        cout << "\n";
    }

    return 0;
}
