use std::io::{self, BufRead, Write};

struct Edge {
    x1: f64,
    y1: f64,
    x2: f64,
    y2: f64,
}

impl Edge {
    fn x_at(&self, y: f64) -> f64 {
        let dy = self.y2 - self.y1;
        if dy.abs() < 1e-12 {
            self.x1
        } else {
            self.x1 + (self.x2 - self.x1) * (y - self.y1) / dy
        }
    }
}

fn main() {
    let stdin = io::stdin();
    let mut stdin_lock = stdin.lock();
    let mut buf = String::new();
    
    if stdin_lock.read_line(&mut buf).is_err() { return; }
    let nv: usize = buf.trim().parse().unwrap_or(0);
    buf.clear();

    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        if stdin_lock.read_line(&mut buf).is_err() { break; }
        let mut parts = buf.split_whitespace();
        let x: f64 = parts.next().unwrap_or("0").parse().unwrap_or(0.0);
        let y: f64 = parts.next().unwrap_or("0").parse().unwrap_or(0.0);
        poly.push((x, y));
        buf.clear();
    }

    if stdin_lock.read_line(&mut buf).is_err() { }
    let mut parts = buf.split_whitespace();
    let sx: f64 = parts.next().unwrap_or("0").parse().unwrap_or(0.0);
    let sy: f64 = parts.next().unwrap_or("0").parse().unwrap_or(0.0);
    let mut sz: f64 = parts.next().unwrap_or("-1").parse().unwrap_or(-1.0);

    if sz.abs() < 1e-9 {
        sz = if sz >= 0.0 { 1e-9 } else { -1e-9 };
    }

    if nv < 3 {
        if nv > 0 {
            let mut stdout = io::BufWriter::new(io::stdout());
            writeln!(stdout, "0").unwrap();
        }
        return;
    }

    let u = -sx / sz;
    let v = -sy / sz;

    let d = 8.0;
    let scale = 10.0;

    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for &(x, y) in &poly {
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }

    let mut edges = Vec::with_capacity(nv);
    for i in 0..nv {
        let p1 = poly[i];
        let p2 = poly[(i + 1) % nv];
        edges.push(Edge {
            x1: p1.0,
            y1: p1.1,
            x2: p2.0,
            y2: p2.1,
        });
    }

    let j_start = (min_y / d).floor() as i64;
    let j_end = (max_y / d).floor() as i64;

    let mut selected_cells = Vec::new();

    for j in j_start..=j_end {
        let y_low = j as f64 * d;
        let y_high = (j + 1) as f64 * d;
        let y_mid = y_low + 0.5 * d;

        let mut row_cells = Vec::new();
        let mut even_odd_points = Vec::new();

        for edge in &edges {
            let e_min_y = edge.y1.min(edge.y2);
            let e_max_y = edge.y1.max(edge.y2);

            if e_min_y <= y_high && e_max_y >= y_low {
                let clip_y_low = y_low.max(e_min_y);
                let clip_y_high = y_high.min(e_max_y);
                
                if (edge.y2 - edge.y1).abs() < 1e-12 {
                    let i_start = (edge.x1.min(edge.x2) / d).floor() as i64;
                    let i_end = (edge.x1.max(edge.x2) / d).floor() as i64;
                    for i in i_start..=i_end {
                        row_cells.push(i);
                    }
                } else {
                    let x_at_low = edge.x_at(clip_y_low);
                    let x_at_high = edge.x_at(clip_y_high);
                    let i_start = (x_at_low.min(x_at_high) / d).floor() as i64;
                    let i_end = (x_at_low.max(x_at_high) / d).floor() as i64;
                    for i in i_start..=i_end {
                        row_cells.push(i);
                    }
                }
            }

            if e_min_y <= y_mid && e_max_y > y_mid {
                even_odd_points.push(edge.x_at(y_mid));
            }
        }

        even_odd_points.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
        for k in (0..even_odd_points.len()).step_by(2) {
            if k + 1 < even_odd_points.len() {
                let x_start = even_odd_points[k];
                let x_end = even_odd_points[k + 1];
                let i_start = (x_start / d).floor() as i64;
                let i_end = (x_end / d).floor() as i64;
                for i_cell in i_start..=i_end {
                    row_cells.push(i_cell);
                }
            }
        }

        row_cells.sort();
        row_cells.dedup();
        for i in row_cells {
            selected_cells.push((i, j));
        }
    }

    let mut stdout = io::BufWriter::new(io::stdout());
    writeln!(stdout, "{}", selected_cells.len()).unwrap();
    
    let theta = if u == 0.0 && v == 0.0 {
        std::f64::consts::FRAC_PI_4
    } else {
        (-u).atan2(v)
    };
    
    let qw = (theta / 2.0).cos();
    let qx = 0.0;
    let qy = 0.0;
    let qz = (theta / 2.0).sin();

    for (idx, &(i, j)) in selected_cells.iter().enumerate() {
        let cx = (i as f64 + 0.5) * d;
        let cy = (j as f64 + 0.5) * d;
        let pz = idx as f64 * 20.0;
        let px = cx - pz * u;
        let py = cy - pz * v;
        
        writeln!(stdout, "{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}", 
                 px, py, pz, qw, qx, qy, qz, scale).unwrap();
    }
}
