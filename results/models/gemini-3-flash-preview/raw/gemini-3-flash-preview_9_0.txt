using System;
using System.IO;
using System.Runtime.CompilerServices;
using System.Text;

public class Program
{
    // Cut struct for sorting and tracking original indices
    struct Cut : IComparable<Cut>
    {
        public long Length;
        public int OriginalIndex;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int CompareTo(Cut other)
        {
            // Sort by Length descending for First Fit Decreasing (FFD)
            int cmp = other.Length.CompareTo(this.Length);
            if (cmp != 0) return cmp;
            // Stable sort by original index to maintain consistency
            return this.OriginalIndex.CompareTo(other.OriginalIndex);
        }
    }

    // Assignment struct to store which cut goes into which bin
    struct Assignment : IComparable<Assignment>
    {
        public int BinIdx;
        public int CutIdx;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int CompareTo(Assignment other)
        {
            // Sort by BinIdx ascending to group cuts by stock
            int cmp = this.BinIdx.CompareTo(other.BinIdx);
            if (cmp != 0) return cmp;
            // Sort by CutIdx within the same bin
            return this.CutIdx.CompareTo(other.CutIdx);
        }
    }

    public static void Main()
    {
        // Use custom fast reader for performance with millions of inputs
        FastReader reader = new FastReader(Console.OpenStandardInput());
        
        int n = reader.NextInt();
        if (n < 0) return;
        
        long stockLength = reader.NextLong();

        // Handle edge case of no cuts
        if (n == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Read all cuts
        Cut[] cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut { Length = reader.NextLong(), OriginalIndex = i };
        }

        // 1. Sort cuts descending (The 'Decreasing' part of FFD)
        // This heuristic significantly improves the packing density.
        Array.Sort(cuts);

        // 2. Assign cuts to bins using a Segment Tree for O(log N) lookup
        // Each leaf in the segment tree represents a bin's remaining capacity.
        SegmentTree st = new SegmentTree(n, stockLength);
        Assignment[] assignments = new Assignment[n];

        int maxBinUsed = -1;
        for (int i = 0; i < n; i++)
        {
            // Find the first bin that can fit this cut length
            int binIdx = st.FindAndSubtract(cuts[i].Length);
            assignments[i] = new Assignment { BinIdx = binIdx, CutIdx = cuts[i].OriginalIndex };
            if (binIdx > maxBinUsed) maxBinUsed = binIdx;
        }

        // 3. Output results
        int numStocks = maxBinUsed + 1;
        
        // Use a large buffer for output to handle millions of lines efficiently
        using (StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1024 * 1024))
        {
            writer.WriteLine(numStocks);

            // Sort assignments to group them by BinIdx for sequential output
            Array.Sort(assignments);

            int current = 0;
            for (int b = 0; b <= maxBinUsed; b++)
            {
                bool first = true;
                while (current < n && assignments[current].BinIdx == b)
                {
                    if (!first) writer.Write(' ');
                    writer.Write(assignments[current].CutIdx);
                    first = false;
                    current++;
                }
                writer.WriteLine();
            }
        }
    }
}

// Fast I/O for reading integers and longs directly from the stream buffer
class FastReader
{
    private readonly Stream stream;
    private readonly byte[] buffer = new byte[1024 * 1024];
    private int pos = 0;
    private int limit = 0;

    public FastReader(Stream s) { stream = s; }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private int ReadByte()
    {
        if (pos >= limit)
        {
            pos = 0;
            limit = stream.Read(buffer, 0, buffer.Length);
            if (limit <= 0) return -1;
        }
        return buffer[pos++];
    }

    public int NextInt()
    {
        int c = ReadByte();
        while (c != -1 && (c < '0' || c > '9')) c = ReadByte();
        if (c == -1) return -1;
        int res = 0;
        while (c >= '0' && c <= '9')
        {
            res = res * 10 + (c - '0');
            c = ReadByte();
        }
        return res;
    }

    public long NextLong()
    {
        int c = ReadByte();
        while (c != -1 && (c < '0' || c > '9')) c = ReadByte();
        if (c == -1) return -1;
        long res = 0;
        while (c >= '0' && c <= '9')
        {
            res = res * 10 + (c - '0');
            c = ReadByte();
        }
        return res;
    }
}

// Segment Tree to find the first bin with enough capacity in O(log N) time.
// This allows the FFD algorithm to run in O(N log N) total time.
class SegmentTree
{
    private readonly long[] tree;
    private readonly int size;

    public SegmentTree(int n, long stockLength)
    {
        // size must be a power of 2 for a balanced segment tree
        size = 1;
        while (size < n) size <<= 1;
        tree = new long[2 * size];
        
        // Initialize the first n bins with the full stockLength capacity
        for (int i = 0; i < n; i++) tree[size + i] = stockLength;
        // Pad the rest of the leaves with -1 to indicate they aren't available
        for (int i = n; i < size; i++) tree[size + i] = -1;
        // Build the tree bottom-up: each internal node stores the max capacity in its subtree
        for (int i = size - 1; i > 0; i--) tree[i] = Math.Max(tree[i << 1], tree[(i << 1) | 1]);
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public int FindAndSubtract(long cutLength)
    {
        int node = 1;
        // Root tree[1] stores the global maximum capacity across all bins.
        // If it's less than cutLength, we technically ran out of bins (not possible here as we start with n bins).
        
        // Walk down the tree to find the first (leftmost) bin that can fit the cut
        while (node < size)
        {
            if (tree[node << 1] >= cutLength) 
                node <<= 1;
            else 
                node = (node << 1) | 1;
        }
        
        int binIdx = node - size;
        tree[node] -= cutLength;
        
        // Update the maximums along the path back to the root
        while (node > 1)
        {
            node >>= 1;
            long newVal = Math.Max(tree[node << 1], tree[(node << 1) | 1]);
            // Optimization: if the node value doesn't change, its ancestors won't either
            if (tree[node] == newVal) break;
            tree[node] = newVal;
        }
        return binIdx;
    }
}
