using System;
using System.Collections.Generic;
using System.Linq;

namespace BinPacking
{
    class Program
    {
        struct Box
        {
            public int Index;
            public int W, H, D;
            public long Volume;
        }

        struct Cuboid
        {
            public int X, Y, Z;
            public int W, H, D;
            public Cuboid(int x, int y, int z, int w, int h, int d) { X = x; Y = y; Z = z; W = w; H = h; D = d; }
        }

        struct PackedBox
        {
            public int Index;
            public int X, Y, Z;
        }

        static void Main(string[] args)
        {
            // Use faster I/O
            var inputRaw = Console.In.ReadToEnd();
            if (string.IsNullOrWhiteSpace(inputRaw)) return;
            var input = inputRaw.Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);

            if (input.Length < 4) return;
            int ptr = 0;
            int N = int.Parse(input[ptr++]);
            int CW = int.Parse(input[ptr++]);
            int CH = int.Parse(input[ptr++]);
            int CD = int.Parse(input[ptr++]);

            List<Box> boxes = new List<Box>(N);
            for (int i = 0; i < N; i++)
            {
                if (ptr + 2 >= input.Length) break;
                int w = int.Parse(input[ptr++]);
                int h = int.Parse(input[ptr++]);
                int d = int.Parse(input[ptr++]);
                boxes.Add(new Box { Index = i, W = w, H = h, D = d, Volume = (long)w * h * d });
            }

            List<PackedBox> bestPacked = new List<PackedBox>();

            // Try different sorts to maximize count
            // 1. Volume Ascending (Priority for count)
            // 2. Volume Descending (Standard for volume)
            // 3. Difficulty (W+H+D) Descending
            // 4. Max Dimension Descending
            
            var sort1 = boxes.OrderBy(b => b.Volume).ToList();
            var sort2 = boxes.OrderByDescending(b => b.Volume).ToList();
            var sort3 = boxes.OrderByDescending(b => b.W + b.H + b.D).ToList();
            var sort4 = boxes.OrderByDescending(b => Math.Max(b.W, Math.Max(b.H, b.D))).ToList();

            List<List<Box>> sorts = new List<List<Box>> { sort1, sort2, sort3, sort4 };

            foreach (var sortedBoxes in sorts)
            {
                var currentPacked = PackGuillotine(sortedBoxes, CW, CH, CD);
                if (currentPacked.Count > bestPacked.Count)
                {
                    bestPacked = currentPacked;
                }
                if (bestPacked.Count == N) break;
            }

            Console.WriteLine(bestPacked.Count);
            foreach (var pb in bestPacked)
            {
                Console.WriteLine($"{pb.Index} {pb.X} {pb.Y} {pb.Z}");
            }
        }

        static List<PackedBox> PackGuillotine(List<Box> boxes, int CW, int CH, int CD)
        {
            List<Cuboid> freeSpaces = new List<Cuboid> { new Cuboid(0, 0, 0, CW, CH, CD) };
            List<PackedBox> packed = new List<PackedBox>();

            foreach (var box in boxes)
            {
                int bestFSIndex = -1;
                int bestRot = -1;
                long bestVol = long.MaxValue;
                int bestZ = int.MaxValue, bestY = int.MaxValue, bestX = int.MaxValue;

                // Try all free spaces
                for (int i = 0; i < freeSpaces.Count; i++)
                {
                    var fs = freeSpaces[i];
                    if (box.W > fs.W && box.W > fs.H && box.W > fs.D &&
                        box.H > fs.W && box.H > fs.H && box.H > fs.D &&
                        box.D > fs.W && box.D > fs.H && box.D > fs.D) continue;

                    // Try unique rotations
                    int w = box.W, h = box.H, d = box.D;
                    (int, int, int)[] rots = {
                        (w, h, d), (w, d, h),
                        (h, w, d), (h, d, w),
                        (d, w, h), (d, h, w)
                    };

                    for (int r = 0; r < 6; r++)
                    {
                        bool duplicate = false;
                        for (int k = 0; k < r; k++)
                        {
                            if (rots[k].Item1 == rots[r].Item1 && rots[k].Item2 == rots[r].Item2 && rots[k].Item3 == rots[r].Item3)
                            {
                                duplicate = true; break;
                            }
                        }
                        if (duplicate) continue;

                        int bw = rots[r].Item1;
                        int bh = rots[r].Item2;
                        int bd = rots[r].Item3;

                        if (bw <= fs.W && bh <= fs.H && bd <= fs.D)
                        {
                            long vol = (long)fs.W * fs.H * fs.D;
                            if (vol < bestVol)
                            {
                                bestVol = vol;
                                bestZ = fs.Z; bestY = fs.Y; bestX = fs.X;
                                bestFSIndex = i; bestRot = r;
                            }
                            else if (vol == bestVol)
                            {
                                if (fs.Z < bestZ || (fs.Z == bestZ && (fs.Y < bestY || (fs.Y == bestY && fs.X < bestX))))
                                {
                                    bestZ = fs.Z; bestY = fs.Y; bestX = fs.X;
                                    bestFSIndex = i; bestRot = r;
                                }
                            }
                        }
                    }
                }

                if (bestFSIndex != -1)
                {
                    var fs = freeSpaces[bestFSIndex];
                    freeSpaces.RemoveAt(bestFSIndex);
                    
                    int bw, bh, bd;
                    int w = box.W, h = box.H, d = box.D;
                    switch (bestRot)
                    {
                        case 0: bw = w; bh = h; bd = d; break;
                        case 1: bw = w; bh = d; bd = h; break;
                        case 2: bw = h; bh = w; bd = d; break;
                        case 3: bw = h; bh = d; bd = w; break;
                        case 4: bw = d; bh = w; bd = h; break;
                        default: bw = d; bh = h; bd = w; break;
                    }

                    packed.Add(new PackedBox { Index = box.Index, X = fs.X, Y = fs.Y, Z = fs.Z });

                    // Split remaining space into 3 cuboids using Longer Axis Split (LAS)
                    int dw = fs.W - bw;
                    int dh = fs.H - bh;
                    int dd = fs.D - bd;

                    if (dw >= dh && dw >= dd)
                    {
                        if (dw > 0) freeSpaces.Add(new Cuboid(fs.X + bw, fs.Y, fs.Z, dw, fs.H, fs.D));
                        if (dh >= dd)
                        {
                            if (dh > 0) freeSpaces.Add(new Cuboid(fs.X, fs.Y + bh, fs.Z, bw, dh, fs.D));
                            if (dd > 0) freeSpaces.Add(new Cuboid(fs.X, fs.Y, fs.Z + bd, bw, bh, dd));
                        }
                        else
                        {
                            if (dd > 0) freeSpaces.Add(new Cuboid(fs.X, fs.Y, fs.Z + bd, bw, fs.H, dd));
                            if (dh > 0) freeSpaces.Add(new Cuboid(fs.X, fs.Y + bh, fs.Z, bw, dh, bd));
                        }
                    }
                    else if (dh >= dw && dh >= dd)
                    {
                        if (dh > 0) freeSpaces.Add(new Cuboid(fs.X, fs.Y + bh, fs.Z, fs.W, dh, fs.D));
                        if (dw >= dd)
                        {
                            if (dw > 0) freeSpaces.Add(new Cuboid(fs.X + bw, fs.Y, fs.Z, dw, bh, fs.D));
                            if (dd > 0) freeSpaces.Add(new Cuboid(fs.X, fs.Y, fs.Z + bd, bw, bh, dd));
                        }
                        else
                        {
                            if (dd > 0) freeSpaces.Add(new Cuboid(fs.X, fs.Y, fs.Z + bd, fs.W, bh, dd));
                            if (dw > 0) freeSpaces.Add(new Cuboid(fs.X + bw, fs.Y, fs.Z, dw, bh, bd));
                        }
                    }
                    else
                    {
                        if (dd > 0) freeSpaces.Add(new Cuboid(fs.X, fs.Y, fs.Z + bd, fs.W, fs.H, dd));
                        if (dw >= dh)
                        {
                            if (dw > 0) freeSpaces.Add(new Cuboid(fs.X + bw, fs.Y, fs.Z, dw, fs.H, bd));
                            if (dh > 0) freeSpaces.Add(new Cuboid(fs.X, fs.Y + bh, fs.Z, bw, dh, bd));
                        }
                        else
                        {
                            if (dh > 0) freeSpaces.Add(new Cuboid(fs.X, fs.Y + bh, fs.Z, fs.W, dh, bd));
                            if (dw > 0) freeSpaces.Add(new Cuboid(fs.X + bw, fs.Y, fs.Z, dw, bh, bd));
                        }
                    }
                }
            }
            return packed;
        }
    }
}
