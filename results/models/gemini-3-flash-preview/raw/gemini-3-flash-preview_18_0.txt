// Fast I/O using a simpler approach for high compatibility
struct FastReader {
    static inline int read_int() {
        int x = 0;
        int c = _getchar_nolock();
        while (c != EOF && (c < '0' || c > '9')) c = _getchar_nolock();
        if (c == EOF) return -1;
        while (c >= '0' && c <= '9') {
            x = x * 10 + (c - '0');
            c = _getchar_nolock();
        }
        return x;
    }

    static inline long long read_ll() {
        long long x = 0;
        int c = _getchar_nolock();
        while (c != EOF && (c < '0' || c > '9')) c = _getchar_nolock();
        if (c == EOF) return -1;
        while (c >= '0' && c <= '9') {
            x = x * 10 + (c - '0');
            c = _getchar_nolock();
        }
        return x;
    }
};

#ifdef _WIN32
#define GETCHAR _getchar_nolock
#else
#define GETCHAR getchar_unlocked
#endif

inline long long fast_read() {
    long long x = 0;
    int c = GETCHAR();
    while (c != EOF && (c < '0' || c > '9')) c = GETCHAR();
    if (c == EOF) return -1;
    while (c >= '0' && c <= '9') {
        x = x * 10 + (c - '0');
        c = GETCHAR();
    }
    return x;
}

struct Edge {
    uint32_t to;
    uint32_t next;
};

struct EdgeBits {
    unique_ptr<uint8_t[]> bits;
    size_t n_edges;
    EdgeBits(size_t n) : n_edges(n) {
        size_t sz = (n + 7) / 8;
        bits = make_unique<uint8_t[]>(sz);
        fill(bits.get(), bits.get() + sz, 0xFF);
    }
    inline bool get(size_t i) const { return (bits[i >> 3] >> (i & 7)) & 1; }
    inline void set(size_t i, bool val) {
        if (val) bits[i >> 3] |= (1 << (i & 7));
        else bits[i >> 3] &= ~(1 << (i & 7));
    }
};

struct Dinic {
    int n;
    const uint32_t* head;
    const Edge* edges;
    EdgeBits available;
    vector<int> level;
    vector<uint32_t> ptr;

    Dinic(int _n, const uint32_t* _head, const Edge* _edges, size_t m2)
        : n(_n), head(_head), edges(_edges), available(m2), level(_n), ptr(_n) {}

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        level[s] = 0;
        queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (uint32_t i = head[u]; i != 0xFFFFFFFF; i = edges[i].next) {
                if (available.get(i)) {
                    int v = edges[i].to;
                    if (level[v] == -1) {
                        level[v] = level[u] + 1;
                        q.push(v);
                        if (v == t) return true;
                    }
                }
            }
        }
        return level[t] != -1;
    }

    int dfs_iterative(int s, int t) {
        vector<int> path_nodes;
        vector<uint32_t> path_edges;
        path_nodes.push_back(s);
        while(!path_nodes.empty()) {
            int u = path_nodes.back();
            if (u == t) {
                for (uint32_t i : path_edges) {
                    available.set(i, false);
                    available.set(i ^ 1, true);
                }
                return 1;
            }
            bool found = false;
            for (uint32_t& i = ptr[u]; i != 0xFFFFFFFF; i = edges[i].next) {
                int v = edges[i].to;
                if (level[v] == level[u] + 1 && available.get(i)) {
                    path_nodes.push_back(v);
                    path_edges.push_back(i);
                    found = true;
                    break;
                }
            }
            if (!found) {
                path_nodes.pop_back();
                if (!path_edges.empty()) path_edges.pop_back();
                level[u] = -1;
            }
        }
        return 0;
    }

    int max_flow(int s, int t, int limit) {
        int flow = 0;
        while (flow < limit && bfs(s, t)) {
            for (int i = 0; i < n; ++i) ptr[i] = head[i];
            while (flow < limit) {
                int pushed = dfs_iterative(s, t);
                if (pushed == 0) break;
                flow += pushed;
            }
        }
        return flow;
    }
    
    void get_reachable(int s, vector<bool>& reachable) {
        fill(reachable.begin(), reachable.end(), false);
        reachable[s] = true;
        queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (uint32_t i = head[u]; i != 0xFFFFFFFF; i = edges[i].next) {
                if (available.get(i)) {
                    int v = edges[i].to;
                    if (!reachable[v]) {
                        reachable[v] = true;
                        q.push(v);
                    }
                }
            }
        }
    }
};

int main() {
    int n = (int)fast_read();
    long long m_ll = fast_read();
    if (n < 2) { if (n > 0) printf("0\n"); return 0; }
    size_t m = (size_t)m_ll;

    struct RawEdge { int u, v; };
    unique_ptr<RawEdge[]> raw_edges = make_unique<RawEdge[]>(m);
    vector<int> degree(n, 0);

    for (size_t i = 0; i < m; ++i) {
        int u = (int)fast_read();
        int v = (int)fast_read();
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        raw_edges[i] = {u, v};
        degree[u]++;
        degree[v]++;
    }

    int min_deg = n + 1;
    int s_node = -1;
    for (int i = 0; i < n; ++i) {
        if (degree[i] < min_deg) {
            min_deg = degree[i];
            s_node = i;
        }
    }

    if (min_deg == 0) { printf("0\n"); return 0; }

    unique_ptr<uint32_t[]> head = make_unique<uint32_t[]>(n);
    fill(head.get(), head.get() + n, 0xFFFFFFFF);
    unique_ptr<Edge[]> edges = make_unique<Edge[]>(2 * m);
    for (size_t i = 0; i < m; ++i) {
        int u = raw_edges[i].u;
        int v = raw_edges[i].v;
        edges[2 * i] = {(uint32_t)v, head[u]};
        head[u] = (uint32_t)(2 * i);
        edges[2 * i + 1] = {(uint32_t)u, head[v]};
        head[v] = (uint32_t)(2 * i + 1);
    }
    raw_edges.reset();

    // Bridge finding (iterative Tarjan)
    {
        unique_ptr<int[]> disc = make_unique<int[]>(n);
        unique_ptr<int[]> low = make_unique<int[]>(n);
        fill(disc.get(), disc.get() + n, 0);
        int timer = 0;
        struct DFSFrame {
            int u;
            uint32_t p_edge;
            uint32_t edge_ptr;
        };
        vector<DFSFrame> st;
        bool found_bridge = false;
        int bridge_u = -1, bridge_v = -1;

        for (int i = 0; i < n && !found_bridge; ++i) {
            if (!disc[i]) {
                st.push_back({i, 0xFFFFFFFF, head[i]});
                disc[i] = low[i] = ++timer;
                while (!st.empty()) {
                    DFSFrame& frame = st.back();
                    int u = frame.u;
                    bool found_child = false;
                    while (frame.edge_ptr != 0xFFFFFFFF) {
                        uint32_t e_idx = frame.edge_ptr;
                        uint32_t v = edges[e_idx].to;
                        frame.edge_ptr = edges[e_idx].next;
                        if (e_idx == (frame.p_edge ^ 1)) continue;
                        if (disc[v]) {
                            low[u] = min(low[u], disc[v]);
                        } else {
                            disc[v] = low[v] = ++timer;
                            st.push_back({(int)v, e_idx, head[v]});
                            found_child = true;
                            break;
                        }
                    }
                    if (!found_child) {
                        if (frame.p_edge != 0xFFFFFFFF) {
                            int p = edges[frame.p_edge ^ 1].to;
                            low[p] = min(low[p], low[u]);
                            if (low[u] > disc[p]) {
                                found_bridge = true;
                                bridge_u = p;
                                bridge_v = u;
                                break;
                            }
                        }
                        st.pop_back();
                    }
                }
            }
        }
        if (found_bridge) {
            printf("1\n%d %d\n", bridge_u, bridge_v);
            return 0;
        }
    }

    atomic<int> global_best_cut(min_deg);
    vector<bool> best_partition(n, false);
    best_partition[s_node] = true;
    mutex mtx;

    vector<int> candidates;
    for (int i = 0; i < n; ++i) if (i != s_node) candidates.push_back(i);
    shuffle(candidates.begin(), candidates.end(), mt19937(42));
    
    // Sort nodes by degree and prioritize small degree ones
    vector<int> small_deg_nodes(n);
    iota(small_deg_nodes.begin(), small_deg_nodes.end(), 0);
    sort(small_deg_nodes.begin(), small_deg_nodes.end(), [&](int a, int b) {
        return degree[a] < degree[b];
    });
    
    vector<int> prioritized;
    for (int i = 0; i < min(n, 20); ++i) if (small_deg_nodes[i] != s_node) prioritized.push_back(small_deg_nodes[i]);
    for (int t : candidates) {
        if (prioritized.size() >= 40) break;
        prioritized.push_back(t);
    }
    sort(prioritized.begin(), prioritized.end());
    prioritized.erase(unique(prioritized.begin(), prioritized.end()), prioritized.end());

    auto worker = [&](int t) {
        Dinic dinic(n, head.get(), edges.get(), 2 * m);
        int flow = dinic.max_flow(s_node, t, global_best_cut.load());
        if (flow < global_best_cut) {
            lock_guard<mutex> lock(mtx);
            if (flow < global_best_cut) {
                global_best_cut = flow;
                vector<bool> reachable(n);
                dinic.get_reachable(s_node, reachable);
                best_partition = reachable;
            }
        }
    };

    vector<thread> threads;
    for (int t : prioritized) {
        if (global_best_cut <= 1) break;
        if (threads.size() >= 12) {
            for (auto& th : threads) th.join();
            threads.clear();
        }
        threads.push_back(thread(worker, t));
    }
    for (auto& th : threads) th.join();

    printf("%d\n", global_best_cut.load());
    for (int u = 0; u < n; ++u) {
        for (uint32_t i = head[u]; i != 0xFFFFFFFF; i = edges[i].next) {
            if (i % 2 == 0) {
                int v = edges[i].to;
                if (best_partition[u] != best_partition[v]) {
                    printf("%d %d\n", u, v);
                }
            }
        }
    }

    return 0;
}
