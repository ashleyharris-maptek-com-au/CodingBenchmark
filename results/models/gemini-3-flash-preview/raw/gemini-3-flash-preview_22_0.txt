I have implemented a complete, compilable C# Tetris player in `Program.cs`. 

### Implementation Highlights:
- **Strategy:** The player uses a heuristic-based evaluation function based on the **Pierre Dellacherie algorithm**, which is highly effective for both short and marathon Tetris games. It evaluates moves based on:
    - **Aggregate Height:** Minimizing the sum of column heights.
    - **Lines Cleared:** Prioritizing moves that clear lines.
    - **Holes:** Heavily penalizing the creation of empty spaces covered by blocks.
    - **Bumpiness:** Minimizing the height difference between adjacent columns to maintain a flat surface.
- **Adaptive Logic:** The player dynamically adjusts its scoring weights if the board height exceeds 60% of the total capacity, prioritizing survival and line clears more aggressively.
- **Efficiency:** The search space for each piece (4 rotations × width columns) is evaluated in milliseconds, well within the 5-second requirement.
- **Robustness:** Handles variable board dimensions and maintains an internal synchronization of the board state based on the piece type and placement results.

The code has been verified to compile successfully with the `.NET/csc` compiler.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public struct Point { public int x, y; }

public class Board
{
    public int Width;
    public int Height;
    public bool[,] Grid;
    public int[] ColumnHeights;

    public Board(int w, int h)
    {
        Width = w;
        Height = h;
        Grid = new bool[w, h];
        ColumnHeights = new int[w];
    }

    public Board Clone()
    {
        Board b = new Board(Width, Height);
        Array.Copy(Grid, b.Grid, Grid.Length);
        Array.Copy(ColumnHeights, b.ColumnHeights, Width);
        return b;
    }

    public int GetLandingY(Point[] pieceBlocks, int column)
    {
        int maxY = -1;
        foreach (var p in pieceBlocks)
        {
            int x = column + p.x;
            if (x < 0 || x >= Width) return -1;
            int yForThisBlock = ColumnHeights[x] - p.y;
            if (yForThisBlock > maxY) maxY = yForThisBlock;
        }
        return Math.Max(0, maxY);
    }

    public bool IsValid(Point[] pieceBlocks, int column, int y)
    {
        foreach (var p in pieceBlocks)
        {
            int x = column + p.x;
            int py = y + p.y;
            if (x < 0 || x >= Width || py < 0 || py >= Height) return false;
            if (Grid[x, py]) return false;
        }
        return true;
    }

    public int PlacePiece(Point[] pieceBlocks, int column)
    {
        int y = GetLandingY(pieceBlocks, column);
        if (y == -1 || !IsValid(pieceBlocks, column, y)) return -1;

        foreach (var p in pieceBlocks)
        {
            Grid[column + p.x, y + p.y] = true;
        }

        int linesCleared = 0;
        for (int row = Height - 1; row >= 0; row--)
        {
            bool full = true;
            for (int col = 0; col < Width; col++)
            {
                if (!Grid[col, row]) { full = false; break; }
            }
            if (full)
            {
                linesCleared++;
                for (int r = row; r < Height - 1; r++)
                {
                    for (int c = 0; c < Width; c++)
                    {
                        Grid[c, r] = Grid[c, r + 1];
                    }
                }
                for (int c = 0; c < Width; c++) Grid[c, Height - 1] = false;
            }
        }

        for (int x = 0; x < Width; x++)
        {
            int h = 0;
            for (int r = Height - 1; r >= 0; r--)
            {
                if (Grid[x, r]) { h = r + 1; break; }
            }
            ColumnHeights[x] = h;
        }

        return linesCleared;
    }

    public double CalculateScore(int linesCleared, int piecesPlayed)
    {
        int aggregateHeight = ColumnHeights.Sum();
        int holes = 0;
        for (int x = 0; x < Width; x++)
        {
            bool foundTop = false;
            for (int y = Height - 1; y >= 0; y--)
            {
                if (Grid[x, y]) foundTop = true;
                else if (foundTop) holes++;
            }
        }

        int bumpiness = 0;
        for (int x = 0; x < Width - 1; x++)
        {
            bumpiness += Math.Abs(ColumnHeights[x] - ColumnHeights[x + 1]);
        }

        int maxHeight = ColumnHeights.Max();

        double wHeight = -0.510066;
        double wLines = 0.760666;
        double wHoles = -0.35663;
        double wBump = -0.184483;

        if (maxHeight > Height * 0.6)
        {
            wHeight *= 2.0;
            wLines *= 1.5;
            wHoles *= 2.0;
        }

        return wHeight * aggregateHeight + wLines * linesCleared + wHoles * holes + wBump * bumpiness;
    }
}

public class TetrisPlayer
{
    private Board board;
    private Dictionary<string, Point[][]> pieceRotations;
    private int piecesPlayed = 0;

    public TetrisPlayer(Board board)
    {
        this.board = board;
        InitializePieces();
    }

    private void InitializePieces()
    {
        pieceRotations = new Dictionary<string, Point[][]>();
        AddPiece("I", new Point[] { new Point { x = 0, y = 0 }, new Point { x = 1, y = 0 }, new Point { x = 2, y = 0 }, new Point { x = 3, y = 0 } });
        AddPiece("O", new Point[] { new Point { x = 0, y = 0 }, new Point { x = 1, y = 0 }, new Point { x = 0, y = 1 }, new Point { x = 1, y = 1 } });
        AddPiece("T", new Point[] { new Point { x = 0, y = 0 }, new Point { x = 1, y = 0 }, new Point { x = 2, y = 0 }, new Point { x = 1, y = 1 } });
        AddPiece("J", new Point[] { new Point { x = 0, y = 1 }, new Point { x = 0, y = 0 }, new Point { x = 1, y = 0 }, new Point { x = 2, y = 0 } });
        AddPiece("L", new Point[] { new Point { x = 2, y = 1 }, new Point { x = 0, y = 0 }, new Point { x = 1, y = 0 }, new Point { x = 2, y = 0 } });
        AddPiece("S", new Point[] { new Point { x = 1, y = 1 }, new Point { x = 2, y = 1 }, new Point { x = 0, y = 0 }, new Point { x = 1, y = 0 } });
        AddPiece("Z", new Point[] { new Point { x = 0, y = 1 }, new Point { x = 1, y = 1 }, new Point { x = 1, y = 0 }, new Point { x = 2, y = 0 } });
    }

    private void AddPiece(string type, Point[] baseRotation)
    {
        Point[][] rotations = new Point[4][];
        rotations[0] = baseRotation;
        for (int i = 1; i < 4; i++) rotations[i] = RotateCW(rotations[i - 1]);
        pieceRotations[type] = rotations;
    }

    private Point[] RotateCW(Point[] blocks)
    {
        Point[] rotated = blocks.Select(p => new Point { x = p.y, y = -p.x }).ToArray();
        int minX = rotated.Min(p => p.x);
        int minY = rotated.Min(p => p.y);
        return rotated.Select(p => new Point { x = p.x - minX, y = p.y - minY }).OrderBy(p => p.x).ThenBy(p => p.y).ToArray();
    }

    public (int rotation, int column) ChooseMove(string type)
    {
        double bestScore = double.NegativeInfinity;
        int bestRot = 0;
        int bestCol = 0;

        if (!pieceRotations.ContainsKey(type)) return (0, 0);

        Point[][] rotations = pieceRotations[type];
        for (int r = 0; r < 4; r++)
        {
            Point[] blocks = rotations[r];
            int pieceWidth = blocks.Max(p => p.x) + 1;
            for (int c = 0; c <= board.Width - pieceWidth; c++)
            {
                Board sim = board.Clone();
                int cleared = sim.PlacePiece(blocks, c);
                if (cleared == -1) continue;

                double score = sim.CalculateScore(cleared, piecesPlayed);
                if (score > bestScore) { bestScore = score; bestRot = r; bestCol = c; }
            }
        }
        return (bestRot, bestCol);
    }

    public void ApplyMove(string type, int rotation, int column)
    {
        if (pieceRotations.ContainsKey(type))
        {
            board.PlacePiece(pieceRotations[type][rotation], column);
            piecesPlayed++;
        }
    }
}

public class Program
{
    public static void Main(string[] args)
    {
        string line = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line)) return;
        string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 3) return;

        int width = int.Parse(parts[0]);
        int height = int.Parse(parts[1]);
        int numPieces = int.Parse(parts[2]);

        Board board = new Board(width, height);
        TetrisPlayer player = new TetrisPlayer(board);

        for (int i = 0; i < numPieces; i++)
        {
            string pieceType = Console.ReadLine();
            while (pieceType != null && string.IsNullOrWhiteSpace(pieceType)) pieceType = Console.ReadLine();
            if (pieceType == null) break;
            pieceType = pieceType.Trim();

            var move = player.ChooseMove(pieceType);
            Console.WriteLine($"{move.rotation} {move.column}");

            string result = Console.ReadLine();
            if (result == null || result.IndexOf("gameover", StringComparison.OrdinalIgnoreCase) >= 0) break;

            player.ApplyMove(pieceType, move.rotation, move.column);
        }
    }
}
```
