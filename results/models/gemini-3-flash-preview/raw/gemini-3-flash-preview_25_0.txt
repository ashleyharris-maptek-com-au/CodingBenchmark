using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Threading.Tasks;

namespace AsteroidSolver {
    public struct Vector3 {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator *(Vector3 a, double b) => new Vector3(a.X * b, a.Y * b, a.Z * b);
        public static Vector3 operator *(double a, Vector3 b) => new Vector3(a * b.X, a * b.Y, a * b.Z);
        public static Vector3 operator /(Vector3 a, double b) => new Vector3(a.X / b, a.Y / b, a.Z / b);
        public double LengthSq() => X * X + Y * Y + Z * Z;
        public double Length() => Math.Sqrt(LengthSq());
        public Vector3 Normalize() { double l = Length(); return l > 1e-20 ? this / l : new Vector3(); }
        public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    class Body {
        public string Name;
        public double Mass;
        public Vector3 Position;
        public Vector3 Velocity;
        public Vector3 InitialPosition;
        public Vector3 InitialVelocity;
    }

    class Program {
        static double G = 6.67430e-11;
        static List<Body> bodies = new List<Body>();
        static double warningDays, impactorMass, deltaVBudget, asteroidMass, t0_unix;
        static string bspPath;

        static void Main(string[] args) {
            System.Threading.Thread.CurrentThread.CurrentCulture = System.Globalization.CultureInfo.InvariantCulture;
            string firstLine = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(firstLine)) return;
            string[] parts = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 6) return;
            int numBodies = int.Parse(parts[0]);
            warningDays = double.Parse(parts[1]);
            impactorMass = double.Parse(parts[2]);
            deltaVBudget = double.Parse(parts[3]);
            asteroidMass = double.Parse(parts[4]);
            t0_unix = double.Parse(parts[5]);

            for (int i = 0; i < numBodies; i++) {
                string line = Console.ReadLine();
                if (string.IsNullOrWhiteSpace(line)) continue;
                string[] bParts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                bodies.Add(new Body {
                    Name = bParts[0],
                    Mass = double.Parse(bParts[1]),
                    Position = new Vector3(double.Parse(bParts[2]), double.Parse(bParts[3]), double.Parse(bParts[4])),
                    Velocity = new Vector3(double.Parse(bParts[5]), double.Parse(bParts[6]), double.Parse(bParts[7])),
                    InitialPosition = new Vector3(double.Parse(bParts[2]), double.Parse(bParts[3]), double.Parse(bParts[4])),
                    InitialVelocity = new Vector3(double.Parse(bParts[5]), double.Parse(bParts[6]), double.Parse(bParts[7]))
                });
            }
            bspPath = Console.ReadLine()?.Trim();

            double tColl = FindCollisionTime();
            int totalSteps = (int)(tColl + 20);
            Vector3[][] positions = new Vector3[bodies.Count][];
            Vector3[][] velocities = new Vector3[bodies.Count][];
            for (int i = 0; i < bodies.Count; i++) {
                positions[i] = new Vector3[totalSteps + 1];
                velocities[i] = new Vector3[totalSteps + 1];
            }
            
            ResetBodies();
            double dtSearch = 0.1; 
            for (int step = 0; step <= totalSteps * 10; step++) {
                if (step % 10 == 0) {
                    int day = step / 10;
                    if (day < totalSteps + 1) {
                        for (int i = 0; i < bodies.Count; i++) {
                            positions[i][day] = bodies[i].Position;
                            velocities[i][day] = bodies[i].Velocity;
                        }
                    }
                }
                StepBodies(dtSearch);
            }

            double bestDeflection = -1;
            double bestTLaunch = 0, bestTImpact = 0;
            Vector3 bestBurn1Initial = new Vector3();

            int earthIdx = bodies.FindIndex(b => b.Name == "Earth");
            int astIdx = bodies.FindIndex(b => b.Name == "Asteroid");
            int sunIdx = bodies.FindIndex(b => b.Name == "Sun");
            double mu = (sunIdx >= 0) ? bodies[sunIdx].Mass * G : 1.32712440018e20;

            int launchSteps = 80;
            int impactSteps = 80;
            object lockObj = new object();

            Parallel.For(0, launchSteps, i => {
                double tL = (warningDays * 0.95 / launchSteps) * i;
                for (int j = 0; j < impactSteps; j++) {
                    double tI = tL + (tColl - tL) * (0.02 + 0.96 * (j + 0.5) / impactSteps);
                    if (tI >= tColl) continue;

                    Vector3 p1 = GetPos(earthIdx, tL, positions);
                    Vector3 p2 = GetPos(astIdx, tI, positions);
                    Vector3 vE1 = GetVel(earthIdx, tL, velocities);
                    Vector3 vA2 = GetVel(astIdx, tI, velocities);
                    
                    Vector3 vL = Shoot2Body(p1, p2, (tI - tL) * 86400, mu);
                    Vector3 dv1 = vL - vE1;
                    double dv1Mag = dv1.Length();
                    
                    if (dv1Mag < deltaVBudget) {
                        Vector3 vI = Propagate2Body(p1, vL, (tI - tL) * 86400, mu);
                        Vector3 vRel = vI - vA2;
                        double remDV = deltaVBudget - dv1Mag;
                        Vector3 vRelNew = vRel + vRel.Normalize() * remDV;
                        Vector3 dvAst = (impactorMass / (asteroidMass + impactorMass)) * vRelNew;
                        double deflection = dvAst.Length() * (tColl - tI) * 86400;
                        
                        lock (lockObj) {
                            if (deflection > bestDeflection) {
                                bestDeflection = deflection;
                                bestTLaunch = tL;
                                bestTImpact = tI;
                                bestBurn1Initial = dv1;
                            }
                        }
                    }
                }
            });

            if (bestDeflection > 0) {
                Vector3 p1 = GetPos(earthIdx, bestTLaunch, positions);
                Vector3 p2 = GetPos(astIdx, bestTImpact, positions);
                Vector3 vE1 = GetVel(earthIdx, bestTLaunch, velocities);
                
                Vector3 vL = bestBurn1Initial + vE1;
                vL = RefineShooting(p1, vL, bestTLaunch, bestTImpact, p2, positions);
                Vector3 bestBurn1 = vL - vE1;
                if (bestBurn1.Length() > deltaVBudget) bestBurn1 = bestBurn1.Normalize() * deltaVBudget;
                
                Vector3 finalVSC, finalPSC;
                PropagateSCFull(p1, vE1 + bestBurn1, bestTLaunch, bestTImpact, positions, out finalPSC, out finalVSC);
                Vector3 vA2 = GetVel(astIdx, bestTImpact, velocities);
                Vector3 vRel = finalVSC - vA2;
                double remDV = Math.Max(0, deltaVBudget - bestBurn1.Length());
                Vector3 bestBurn2 = vRel.Normalize() * remDV;
                
                Console.WriteLine(2);
                Console.WriteLine($"{bestTLaunch:F10} {bestBurn1.X:E16} {bestBurn1.Y:E16} {bestBurn1.Z:E16}");
                Console.WriteLine($"{bestTImpact:F10} {bestBurn2.X:E16} {bestBurn2.Y:E16} {bestBurn2.Z:E16}");
            } else {
                Console.WriteLine(1);
                Console.WriteLine("0 0 0 0");
            }
        }

        static double FindCollisionTime() {
            ResetBodies();
            double dt = 1.0;
            int earthIdx = bodies.FindIndex(b => b.Name == "Earth");
            int astIdx = bodies.FindIndex(b => b.Name == "Asteroid");
            double minDist = double.MaxValue;
            double tMinDist = 0;
            for (double t = 0; t <= warningDays + 10; t += dt) {
                double d = (bodies[earthIdx].Position - bodies[astIdx].Position).Length();
                if (d < minDist) {
                    minDist = d;
                    tMinDist = t;
                }
                StepBodies(dt);
            }
            return tMinDist;
        }

        static void ResetBodies() {
            foreach (var b in bodies) {
                b.Position = b.InitialPosition;
                b.Velocity = b.InitialVelocity;
            }
        }

        static void StepBodies(double dtDays) {
            double dt = dtDays * 86400;
            Vector3[] accs = new Vector3[bodies.Count];
            for (int i = 0; i < bodies.Count; i++) accs[i] = GetAcc(bodies[i].Position, i);
            for (int i = 0; i < bodies.Count; i++) bodies[i].Velocity += accs[i] * (dt * 0.5);
            for (int i = 0; i < bodies.Count; i++) bodies[i].Position += bodies[i].Velocity * dt;
            for (int i = 0; i < bodies.Count; i++) accs[i] = GetAcc(bodies[i].Position, i);
            for (int i = 0; i < bodies.Count; i++) bodies[i].Velocity += accs[i] * (dt * 0.5);
        }

        static Vector3 GetAcc(Vector3 pos, int skipIdx) {
            Vector3 acc = new Vector3();
            for (int i = 0; i < bodies.Count; i++) {
                if (i == skipIdx) continue;
                Vector3 diff = bodies[i].Position - pos;
                double r2 = diff.LengthSq();
                if (r2 < 1e6) r2 = 1e6;
                acc += (bodies[i].Mass * G / (r2 * Math.Sqrt(r2))) * diff;
            }
            return acc;
        }

        static Vector3 GetPos(int idx, double t, Vector3[][] posData) {
            int d = (int)t; if (d < 0) d = 0; if (d >= posData[idx].Length - 1) d = posData[idx].Length - 2;
            return posData[idx][d] + (posData[idx][d+1] - posData[idx][d]) * (t - d);
        }
        
        static Vector3 GetVel(int idx, double t, Vector3[][] velData) {
            int d = (int)t; if (d < 0) d = 0; if (d >= velData[idx].Length - 1) d = velData[idx].Length - 2;
            return velData[idx][d] + (velData[idx][d+1] - velData[idx][d]) * (t - d);
        }

        static Vector3 Shoot2Body(Vector3 r1, Vector3 r2, double dt, double mu) {
            Vector3 r_mid = (r1 + r2) * 0.5;
            double rm2 = r_mid.LengthSq();
            Vector3 a_mid = (-mu / Math.Max(1e-10, rm2 * Math.Sqrt(rm2))) * r_mid;
            Vector3 v = (r2 - r1) / dt - a_mid * (dt * 0.5);
            for (int iter = 0; iter < 10; iter++) {
                Vector3 rf = Propagate2Body(r1, v, dt, mu);
                Vector3 err = rf - r2;
                if (err.Length() < 500) break;
                double eps = Math.Max(v.Length() * 1e-7, 0.1);
                Vector3 rx = (Propagate2Body(r1, v + new Vector3(eps, 0, 0), dt, mu) - rf) / eps;
                Vector3 ry = (Propagate2Body(r1, v + new Vector3(0, eps, 0), dt, mu) - rf) / eps;
                Vector3 rz = (Propagate2Body(r1, v + new Vector3(0, 0, eps), dt, mu) - rf) / eps;
                double det = rx.X * (ry.Y * rz.Z - ry.Z * rz.Y) - rx.Y * (ry.X * rz.Z - ry.Z * rz.X) + rx.Z * (ry.X * rz.Y - ry.Y * rz.X);
                if (Math.Abs(det) < 1e-35) break;
                v = v + new Vector3(
                    ((ry.Y * rz.Z - ry.Z * rz.Y) * (-err.X) - (rx.Y * rz.Z - rx.Z * rz.Y) * (-err.Y) + (rx.Y * ry.Z - rx.Z * ry.Y) * (-err.Z)) / det,
                    (-(ry.X * rz.Z - ry.Z * rz.X) * (-err.X) + (rx.X * rz.Z - rx.Z * rz.X) * (-err.Y) - (rx.X * ry.Z - rx.Z * ry.X) * (-err.Z)) / det,
                    ((ry.X * rz.Y - ry.Y * rz.X) * (-err.X) - (rx.X * rz.Y - rx.Y * rz.X) * (-err.Y) + (rx.X * ry.Y - rx.Y * ry.X) * (-err.Z)) / det
                );
            }
            return v;
        }

        static Vector3 Propagate2Body(Vector3 r, Vector3 v, double dt, double mu) {
            int steps = Math.Max(120, (int)(dt / 86400));
            double subdt = dt / steps;
            Vector3 cr = r, cv = v;
            for (int i = 0; i < steps; i++) {
                double r2 = cr.LengthSq();
                Vector3 a1 = (-mu / Math.Max(1e-10, r2 * Math.Sqrt(r2))) * cr;
                cv += a1 * (subdt * 0.5);
                cr += cv * subdt;
                r2 = cr.LengthSq();
                Vector3 a2 = (-mu / Math.Max(1e-10, r2 * Math.Sqrt(r2))) * cr;
                cv += a2 * (subdt * 0.5);
            }
            return cr;
        }

        static Vector3 RefineShooting(Vector3 p1, Vector3 vL, double tL, double tI, Vector3 pT, Vector3[][] bp) {
            Vector3 v = vL;
            for (int iter = 0; iter < 10; iter++) {
                Vector3 rf, vf;
                PropagateSCFull(p1, v, tL, tI, bp, out rf, out vf);
                Vector3 err = rf - pT;
                if (err.Length() < 500) break;
                double eps = Math.Max(v.Length() * 1e-7, 0.1);
                Vector3 rxf, ryf, rzf, vxf, vyf, vzf;
                PropagateSCFull(p1, v + new Vector3(eps, 0, 0), tL, tI, bp, out rxf, out vxf);
                PropagateSCFull(p1, v + new Vector3(0, eps, 0), tL, tI, bp, out ryf, out vyf);
                PropagateSCFull(p1, v + new Vector3(0, 0, eps), tL, tI, bp, out rzf, out vzf);
                Vector3 rx = (rxf - rf) / eps, ry = (ryf - rf) / eps, rz = (rzf - rf) / eps;
                double det = rx.X * (ry.Y * rz.Z - ry.Z * rz.Y) - rx.Y * (ry.X * rz.Z - ry.Z * rz.X) + rx.Z * (ry.X * rz.Y - ry.Y * rz.X);
                if (Math.Abs(det) < 1e-35) break;
                v = v + new Vector3(
                    ((ry.Y * rz.Z - ry.Z * rz.Y) * (-err.X) - (rx.Y * rz.Z - rx.Z * rz.Y) * (-err.Y) + (rx.Y * ry.Z - rx.Z * ry.Y) * (-err.Z)) / det,
                    (-(ry.X * rz.Z - ry.Z * rz.X) * (-err.X) + (rx.X * rz.Z - rx.Z * rz.X) * (-err.Y) - (rx.X * ry.Z - rx.Z * ry.X) * (-err.Z)) / det,
                    ((ry.X * rz.Y - ry.Y * rz.X) * (-err.X) - (rx.X * rz.Y - rx.Y * rz.X) * (-err.Y) + (rx.X * ry.Y - rx.Y * ry.X) * (-err.Z)) / det
                );
            }
            return v;
        }

        static void PropagateSCFull(Vector3 p, Vector3 v, double tL, double tI, Vector3[][] bp, out Vector3 rf, out Vector3 vf) {
            double t = tL; rf = p; vf = v;
            double dtDays = 0.02;
            while (t < tI) {
                double step = Math.Min(dtDays, tI - t);
                double dt = step * 86400;
                Vector3 a1 = GetSCAcc(rf, t, bp);
                vf += a1 * (dt * 0.5);
                rf += vf * dt;
                t += step;
                Vector3 a2 = GetSCAcc(rf, t, bp);
                vf += a2 * (dt * 0.5);
            }
        }

        static Vector3 GetSCAcc(Vector3 pos, double t, Vector3[][] bp) {
            Vector3 acc = new Vector3();
            for (int i = 0; i < bodies.Count; i++) {
                Vector3 bP = GetPos(i, t, bp);
                Vector3 diff = bP - pos;
                double r2 = diff.LengthSq();
                if (r2 < 1e6) r2 = 1e6;
                acc += (bodies[i].Mass * G / (r2 * Math.Sqrt(r2))) * diff;
            }
            return acc;
        }
    }
}
