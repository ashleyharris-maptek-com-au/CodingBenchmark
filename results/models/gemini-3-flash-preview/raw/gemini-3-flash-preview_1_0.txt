#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <stdint.h>
#include <cstdio>
#include <cstdlib>

using namespace std;

// Pack structs to minimize memory usage for massive city counts
#pragma pack(push, 1)
struct RawCity {
    float x, y;
};

struct MortonCity {
    uint64_t code;
    int id;
};
#pragma pack(pop)

// Fast I/O class for rapid reading and writing of millions/billions of values
class FastIO {
    static const int BUF_SIZE = 1 << 20;
    char inBuf[BUF_SIZE];
    int inPos = 0, inLen = 0;
    char outBuf[BUF_SIZE];
    int outPos = 0;

    char readChar() {
        if (inPos == inLen) {
            inPos = 0;
            inLen = (int)fread(inBuf, 1, BUF_SIZE, stdin);
            if (inLen == 0) return EOF;
        }
        return inBuf[inPos++];
    }

public:
    bool readInt(int& n) {
        char c = readChar();
        while (c != EOF && (c < '0' || c > '9') && c != '-') c = readChar();
        if (c == EOF) return false;
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = readChar();
        }
        n = 0;
        while (c >= '0' && c <= '9') {
            n = n * 10 + (c - '0');
            c = readChar();
        }
        if (neg) n = -n;
        return true;
    }

    bool readFloat(float& f) {
        char c = readChar();
        while (c != EOF && (c < '0' || c > '9') && c != '-' && c != '.') c = readChar();
        if (c == EOF) return false;
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = readChar();
        }
        double res = 0;
        while (c >= '0' && c <= '9') {
            res = res * 10 + (c - '0');
            c = readChar();
        }
        if (c == '.') {
            c = readChar();
            double fract = 0.1;
            while (c >= '0' && c <= '9') {
                res += (c - '0') * fract;
                fract *= 0.1;
                c = readChar();
            }
        }
        // Basic scientific notation support
        if (c == 'e' || c == 'E') {
            c = readChar();
            bool eneg = false;
            if (c == '-') { eneg = true; c = readChar(); }
            else if (c == '+') { c = readChar(); }
            int exp = 0;
            while (c >= '0' && c <= '9') {
                exp = exp * 10 + (c - '0');
                c = readChar();
            }
            if (eneg) res /= pow(10.0, exp);
            else res *= pow(10.0, exp);
        }
        if (neg) res = -res;
        f = (float)res;
        return true;
    }

    void putChar(char c) {
        if (outPos == BUF_SIZE) {
            fwrite(outBuf, 1, outPos, stdout);
            outPos = 0;
        }
        outBuf[outPos++] = c;
    }

    void putInt(int n) {
        if (n == 0) {
            putChar('0');
            return;
        }
        static const char digits[201] = 
            "0001020304050607080910111213141516171819"
            "2021222324252627282930313233343536373839"
            "4041424344454647484950515253545556575859"
            "6061626364656667686970717273747576777879"
            "8081828384858687888990919293949596979899";
        char buf[12];
        int p = 12;
        while (n >= 100) {
            int r = n % 100;
            n /= 100;
            buf[--p] = digits[r*2 + 1];
            buf[--p] = digits[r*2];
        }
        if (n >= 10) {
            buf[--p] = digits[n*2 + 1];
            buf[--p] = digits[n*2];
        } else {
            buf[--p] = n + '0';
        }
        while (p < 12) putChar(buf[p++]);
    }

    void flush() {
        if (outPos > 0) {
            fwrite(outBuf, 1, outPos, stdout);
            outPos = 0;
        }
    }
} fio;

// Z-order (Morton) code helper functions
uint64_t split(uint32_t x) {
    uint64_t z = x;
    z = (z | (z << 16)) & 0x0000FFFF0000FFFF;
    z = (z | (z << 8)) & 0x00FF00FF00FF00FF;
    z = (z | (z << 4)) & 0x0F0F0F0F0F0F0F0F;
    z = (z | (z << 2)) & 0x3333333333333333;
    z = (z | (z << 1)) & 0x5555555555555555;
    return z;
}

uint64_t getMorton(uint32_t x, uint32_t y) {
    return split(x) | (split(y) << 1);
}

int main() {
    int N;
    if (!fio.readInt(N)) return 0;
    if (N <= 0) return 0;

    // First pass: read all cities and find the bounding box
    vector<RawCity> raw(N);
    float minX = 1e30f, maxX = -1e30f;
    float minY = 1e30f, maxY = -1e30f;

    for (int i = 0; i < N; i++) {
        fio.readFloat(raw[i].x);
        fio.readFloat(raw[i].y);
        if (raw[i].x < minX) minX = raw[i].x;
        if (raw[i].x > maxX) maxX = raw[i].x;
        if (raw[i].y < minY) minY = raw[i].y;
        if (raw[i].y > maxY) maxY = raw[i].y;
    }

    // Adaptive Strategy
    if (N <= 2500) {
        // For small N, use Nearest Neighbor followed by 2-opt refinement
        vector<int> tour(N);
        vector<bool> visited(N, false);
        tour[0] = 0;
        visited[0] = true;
        for (int i = 1; i < N; i++) {
            int last = tour[i - 1];
            int best = -1;
            float bestD = 1e30f;
            for (int j = 0; j < N; j++) {
                if (!visited[j]) {
                    float dx = raw[last].x - raw[j].x;
                    float dy = raw[last].y - raw[j].y;
                    float d = dx * dx + dy * dy;
                    if (d < bestD) {
                        bestD = d;
                        best = j;
                    }
                }
            }
            tour[i] = best;
            visited[best] = true;
        }

        // 2-opt improvement loop
        bool improved = true;
        for (int iter = 0; iter < 200 && improved; iter++) {
            improved = false;
            for (int i = 1; i < N - 1; i++) {
                for (int j = i + 1; j < N; j++) {
                    int a = tour[i - 1], b = tour[i], c = tour[j], d = (j == N - 1) ? tour[0] : tour[j + 1];
                    // Standard 2-opt edge swap condition: d(a,b) + d(c,d) > d(a,c) + d(b,d)
                    double d1 = sqrt((double)(raw[a].x - raw[b].x) * (raw[a].x - raw[b].x) + (double)(raw[a].y - raw[b].y) * (raw[a].y - raw[b].y)) +
                               sqrt((double)(raw[c].x - raw[d].x) * (raw[c].x - raw[d].x) + (double)(raw[c].y - raw[d].y) * (raw[c].y - raw[d].y));
                    double d2 = sqrt((double)(raw[a].x - raw[c].x) * (raw[a].x - raw[c].x) + (double)(raw[a].y - raw[c].y) * (raw[a].y - raw[c].y)) +
                               sqrt((double)(raw[b].x - raw[d].x) * (raw[b].x - raw[d].x) + (double)(raw[b].y - raw[d].y) * (raw[b].y - raw[d].y));
                    if (d2 < d1 * 0.999999) {
                        reverse(tour.begin() + i, tour.begin() + j + 1);
                        improved = true;
                    }
                }
            }
        }
        for (int i = 0; i < N; i++) {
            fio.putInt(tour[i]);
            fio.putChar(i == N - 1 ? '\n' : ' ');
        }
        fio.flush();
    } else {
        // For large to epic scales, use space-filling curve (Morton order)
        // This is O(N log N) and provides a good balance between speed and quality
        float rangeX = maxX - minX;
        float rangeY = maxY - minY;
        if (rangeX == 0) rangeX = 1;
        if (rangeY == 0) rangeY = 1;

        vector<MortonCity> mc(N);
        for (int i = 0; i < N; i++) {
            // Map coordinates to 32-bit integer space for Morton interleave
            uint32_t ix = (uint32_t)(((double)raw[i].x - minX) / rangeX * 4294967295.0);
            uint32_t iy = (uint32_t)(((double)raw[i].y - minY) / rangeY * 4294967295.0);
            mc[i] = {getMorton(ix, iy), i};
        }
        
        // Free raw coordinate memory to ensure we fit in 32GB for 10^9 cities
        raw.clear();
        raw.shrink_to_fit();

        // Sort by Morton code
        sort(mc.begin(), mc.end(), [](const MortonCity& a, const MortonCity& b) {
            return a.code < b.code;
        });

        // Rotate the tour so it starts at city index 0
        int startIdx = 0;
        for (int i = 0; i < N; i++) {
            if (mc[i].id == 0) {
                startIdx = i;
                break;
            }
        }

        // Print the result using fast I/O
        for (int i = 0; i < N; i++) {
            fio.putInt(mc[(startIdx + i) % N].id);
            fio.putChar(i == N - 1 ? '\n' : ' ');
        }
        fio.flush();
    }

    return 0;
}
