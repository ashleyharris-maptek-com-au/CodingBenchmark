#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <cstdio>

/**
 * TSP Solver using Hilbert Curve Sorting and Windowed 2-Opt Refinement.
 * Scalable for any number of cities from Trivial to Epic (10^7+).
 * Complexity: O(N log N) for Hilbert sorting and O(N * Window) for 2-opt.
 * Memory: ~24 bytes per city (fits in 128GB for 1,000,000,000 cities).
 */

using namespace std;

// Coordinate storage - Minimal size to save memory
struct Point {
    float x, y;
};

// Index storage with Hilbert value for sorting
struct HilbertPoint {
    uint64_t h;
    int id;
};

// Euclidean distance squared (faster than distance)
inline float distSq(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

/**
 * Converts 2D coordinates to a 1D Hilbert curve index.
 * Maps coordinates inside a square of side n (power of 2) into [0, n^2 - 1].
 */
uint64_t xy2d(uint32_t n, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = n / 2; s > 0; s /= 2) {
        uint32_t rx = (x & s) > 0;
        uint32_t ry = (y & s) > 0;
        d += (uint64_t)s * s * ((3ULL * (uint64_t)rx) ^ (uint64_t)ry);
        // Transform the coordinate system for the next iteration
        if (ry == 0) {
            if (rx == 1) {
                x = s - 1 - x;
                y = s - 1 - y;
            }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

int main() {
    int N;
    // Faster I/O using scanf for large data
    if (scanf("%d", &N) != 1) return 0;
    if (N <= 0) return 0;

    vector<Point> points(N);
    double minX = 1e30, maxX = -1e30, minY = 1e30, maxY = -1e30;
    
    // Read input and find bounding box for normalization
    for (int i = 0; i < N; ++i) {
        if (scanf("%f %f", &points[i].x, &points[i].y) != 2) break;
        if (points[i].x < minX) minX = points[i].x;
        if (points[i].x > maxX) maxX = points[i].x;
        if (points[i].y < minY) minY = points[i].y;
        if (points[i].y > maxY) maxY = points[i].y;
    }

    vector<HilbertPoint> hPoints(N);

    // Strategy Selection based on N
    if (N < 3000) {
        // Trivial/Small: Nearest Neighbor search starting at 0
        vector<bool> visited(N, false);
        int curr = 0;
        visited[0] = true;
        hPoints[0] = {0, 0};
        for (int i = 1; i < N; ++i) {
            int best = -1;
            float bestD = 1e30f;
            for (int next = 0; next < N; ++next) {
                if (!visited[next]) {
                    float d = distSq(points[curr], points[next]);
                    if (d < bestD) {
                        bestD = d;
                        best = next;
                    }
                }
            }
            curr = (best == -1 ? 0 : best);
            visited[curr] = true;
            hPoints[i] = {0, curr};
        }
    } else {
        // Medium/Large/Epic: Hilbert Curve Sorting
        uint32_t n = 1U << 31; // Range power of 2
        double rangeX = maxX - minX;
        double rangeY = maxY - minY;
        if (rangeX <= 0) rangeX = 1.0;
        if (rangeY <= 0) rangeY = 1.0;

        for (int i = 0; i < N; ++i) {
            // Map coordinates to integers in range [0, 2^31 - 1]
            uint32_t x_int = (uint32_t)(((double)points[i].x - minX) / rangeX * 2147483647.0);
            uint32_t y_int = (uint32_t)(((double)points[i].y - minY) / rangeY * 2147483647.0);
            hPoints[i] = {xy2d(n, x_int, y_int), i};
        }
        
        // Sort based on Hilbert index for locality
        sort(hPoints.begin(), hPoints.end(), [](const HilbertPoint& a, const HilbertPoint& b) {
            return a.h < b.h;
        });

        // Ensure the path starts at city 0 as per requirements
        int startIdx = 0;
        for (int i = 0; i < N; ++i) {
            if (hPoints[i].id == 0) {
                startIdx = i;
                break;
            }
        }
        if (startIdx != 0) {
            rotate(hPoints.begin(), hPoints.begin() + startIdx, hPoints.end());
        }
    }

    // Windowed 2-Opt local improvement.
    // Window size and pass count adapt to N to ensure performance within 300s.
    int window = 0;
    int passes = 0;
    if (N <= 500) { window = N; passes = 100; }
    else if (N <= 3000) { window = N; passes = 30; }
    else if (N <= 50000) { window = 300; passes = 10; }
    else if (N <= 500000) { window = 100; passes = 5; }
    else if (N <= 2000000) { window = 40; passes = 2; }
    else if (N <= 10000000) { window = 10; passes = 1; }

    for (int p = 0; p < passes; ++p) {
        bool improved = false;
        // Check local swaps in windowed range
        for (int i = 0; i < N - 2; ++i) {
            int limit = min(i + window, N - 1);
            for (int j = i + 2; j < limit; ++j) {
                // Current path: ...i -> i+1...j -> j+1...
                // Swapped path: ...i -> j...i+1 -> j+1...
                float d_curr = distSq(points[hPoints[i].id], points[hPoints[i + 1].id]) +
                               distSq(points[hPoints[j].id], points[hPoints[j + 1].id]);
                float d_next = distSq(points[hPoints[i].id], points[hPoints[j].id]) +
                               distSq(points[hPoints[i + 1].id], points[hPoints[j + 1].id]);
                
                if (d_next < d_curr) {
                    // Reverse internal segment [i+1, j]
                    reverse(hPoints.begin() + i + 1, hPoints.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }

    // Output visit order
    for (int i = 0; i < N; ++i) {
        printf("%d%c", hPoints[i].id, (i == N - 1 ? '\n' : ' '));
    }

    return 0;
}
