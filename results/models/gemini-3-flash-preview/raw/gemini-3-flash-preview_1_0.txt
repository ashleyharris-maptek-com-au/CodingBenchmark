#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <cstdio>
#include <execution>
#include <cstddef>

using namespace std;

/**
 * TSP Solver with Adaptive Logic
 * -----------------------------
 * For N < 5000: Nearest Neighbor (NN) heuristic with 2-opt refinement (for N < 1000).
 * For N >= 5000: Morton Order (Z-order curve) approximation.
 * 
 * Scalability:
 * - Time: O(N log N) for Morton, O(N^2) for NN.
 * - Memory: ~24 bytes per city for Morton (Point + MortonPoint structs).
 *   At N = 1,000,000,000, this requires ~24GB, fitting within 32GB RAM.
 * 
 * Performance:
 * - Uses std::execution::par for parallel sorting on large scales.
 * - Fast I/O for efficient data handling.
 * - MSVC cl /std:c++20 compatible.
 */

// Fast I/O for stdin
static char in_buf[1 << 20];
static int in_buf_ptr = 0;
static int in_buf_size = 0;

inline char get_char() {
    if (in_buf_ptr == in_buf_size) {
        in_buf_size = (int)fread(in_buf, 1, 1 << 20, stdin);
        in_buf_ptr = 0;
        if (in_buf_size == 0) return EOF;
    }
    return in_buf[in_buf_ptr++];
}

inline bool read_long(long long &x) {
    x = 0;
    char ch = get_char();
    while (ch != EOF && (ch < '0' || ch > '9')) ch = get_char();
    if (ch == EOF) return false;
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + (ch - '0');
        ch = get_char();
    }
    return true;
}

inline bool read_float(float &x) {
    double res = 0.0;
    int f = 1;
    char ch = get_char();
    while (ch != EOF && ch != '-' && (ch < '0' || ch > '9') && ch != '.') ch = get_char();
    if (ch == EOF) return false;
    if (ch == '-') {
        f = -1;
        ch = get_char();
    }
    bool has_digit = false;
    while (ch >= '0' && ch <= '9') {
        res = res * 10.0 + (double)(ch - '0');
        ch = get_char();
        has_digit = true;
    }
    if (ch == '.') {
        double tmp = 0.1;
        ch = get_char();
        while (ch >= '0' && ch <= '9') {
            res += (double)(ch - '0') * tmp;
            tmp /= 10.0;
            ch = get_char();
            has_digit = true;
        }
    }
    x = (float)(res * f);
    return has_digit;
}

// Fast I/O for stdout
static char out_buf[1 << 20];
static int out_ptr = 0;

inline void put_char(char ch) {
    if (out_ptr == (1 << 20)) {
        fwrite(out_buf, 1, 1 << 20, stdout);
        out_ptr = 0;
    }
    out_buf[out_ptr++] = ch;
}

inline void write_int(int x) {
    if (x == 0) {
        put_char('0');
        return;
    }
    if (x < 0) {
        put_char('-');
        x = -x;
    }
    static char s[12];
    int len = 0;
    while (x) {
        s[len++] = x % 10 + '0';
        x /= 10;
    }
    while (len--) put_char(s[len]);
}

inline void flush_out() {
    fwrite(out_buf, 1, out_ptr, stdout);
    out_ptr = 0;
}

struct Point {
    float x, y;
};

struct MortonPoint {
    uint64_t code;
    int id;
    bool operator<(const MortonPoint& other) const {
        return code < other.code;
    }
};

// Morton curve: Interleave bits of two 32-bit integers
uint64_t split_bits(uint32_t x) {
    uint64_t x64 = x;
    x64 = (x64 | (x64 << 16)) & 0x0000FFFF0000FFFF;
    x64 = (x64 | (x64 << 8)) & 0x00FF00FF00FF00FF;
    x64 = (x64 | (x64 << 4)) & 0x0F0F0F0F0F0F0F0F;
    x64 = (x64 | (x64 << 2)) & 0x3333333333333333;
    x64 = (x64 | (x64 << 1)) & 0x5555555555555555;
    return x64;
}

uint64_t get_morton(float x, float y, float minX, float minY, float scaleX, float scaleY) {
    uint32_t ux = (uint32_t)((x - minX) * scaleX);
    uint32_t uy = (uint32_t)((y - minY) * scaleY);
    return (split_bits(ux) << 1) | split_bits(uy);
}

int main() {
    long long N;
    if (!read_long(N)) return 0;
    if (N <= 0) return 0;

    if (N < 5000) {
        // Nearest Neighbor + 2-opt for small to medium N
        vector<Point> points((size_t)N);
        for (long long i = 0; i < N; ++i) {
            read_float(points[(size_t)i].x);
            read_float(points[(size_t)i].y);
        }

        vector<int> tour;
        tour.reserve((size_t)N);
        vector<bool> visited((size_t)N, false);
        
        tour.push_back(0);
        visited[0] = true;
        int curr = 0;
        
        for (long long i = 1; i < N; ++i) {
            int best = -1;
            float bestDistSq = 1e30f;
            for (int j = 0; j < (int)N; ++j) {
                if (!visited[(size_t)j]) {
                    float dx = points[(size_t)curr].x - points[(size_t)j].x;
                    float dy = points[(size_t)curr].y - points[(size_t)j].y;
                    float d2 = dx * dx + dy * dy;
                    if (d2 < bestDistSq) {
                        bestDistSq = d2;
                        best = j;
                    }
                }
            }
            tour.push_back(best);
            visited[(size_t)best] = true;
            curr = best;
        }

        // 2-opt for small N
        if (N < 1000) {
            bool improved = true;
            int max_iters = 100;
            while (improved && max_iters--) {
                improved = false;
                for (int i = 0; i < (int)N - 2; ++i) {
                    for (int j = i + 2; j < (int)N; ++j) {
                        int a = tour[(size_t)i];
                        int b = tour[(size_t)i + 1];
                        int c = tour[(size_t)j];
                        int d = tour[((size_t)j + 1) % (size_t)N];

                        float d_ab = sqrtf((points[(size_t)a].x - points[(size_t)b].x) * (points[(size_t)a].x - points[(size_t)b].x) + (points[(size_t)a].y - points[(size_t)b].y) * (points[(size_t)a].y - points[(size_t)b].y));
                        float d_cd = sqrtf((points[(size_t)c].x - points[(size_t)d].x) * (points[(size_t)c].x - points[(size_t)d].x) + (points[(size_t)c].y - points[(size_t)d].y) * (points[(size_t)c].y - points[(size_t)d].y));
                        float d_ac = sqrtf((points[(size_t)a].x - points[(size_t)c].x) * (points[(size_t)a].x - points[(size_t)c].x) + (points[(size_t)a].y - points[(size_t)c].y) * (points[(size_t)a].y - points[(size_t)c].y));
                        float d_bd = sqrtf((points[(size_t)b].x - points[(size_t)d].x) * (points[(size_t)b].x - points[(size_t)d].x) + (points[(size_t)b].y - points[(size_t)d].y) * (points[(size_t)b].y - points[(size_t)d].y));

                        if (d_ac + d_bd < d_ab + d_cd - 1e-6f) {
                            reverse(tour.begin() + (size_t)i + 1, tour.begin() + (size_t)j + 1);
                            improved = true;
                        }
                    }
                }
            }
        }

        for (long long i = 0; i < N; ++i) {
            write_int(tour[(size_t)i]);
            put_char(i == N - 1 ? '\n' : ' ');
        }
    } else {
        // Morton Sorting for large to ludicrous scale
        vector<Point> points((size_t)N);
        float minX = 1e30f, maxX = -1e30f;
        float minY = 1e30f, maxY = -1e30f;
        for (long long i = 0; i < N; ++i) {
            read_float(points[(size_t)i].x);
            read_float(points[(size_t)i].y);
            if (points[(size_t)i].x < minX) minX = points[(size_t)i].x;
            if (points[(size_t)i].x > maxX) maxX = points[(size_t)i].x;
            if (points[(size_t)i].y < minY) minY = points[(size_t)i].y;
            if (points[(size_t)i].y > maxY) maxY = points[(size_t)i].y;
        }

        float rangeX = maxX - minX;
        float rangeY = maxY - minY;
        // Map to 31-bit integers for Morton curve
        float scaleX = (rangeX > 0) ? (float)((1ULL << 31) - 1) / rangeX : 0;
        float scaleY = (rangeY > 0) ? (float)((1ULL << 31) - 1) / rangeY : 0;

        vector<MortonPoint> m_points((size_t)N);
        for (long long i = 0; i < N; ++i) {
            m_points[(size_t)i].code = get_morton(points[(size_t)i].x, points[(size_t)i].y, minX, minY, scaleX, scaleY);
            m_points[(size_t)i].id = (int)i;
        }
        
        // Free original points before sorting to manage memory (peak: ~24 bytes * N)
        points.clear();
        points.shrink_to_fit();

        // Use parallel sort for massive N (standard since C++17, supported by MSVC)
        if (N > 100000) {
            std::sort(std::execution::par, m_points.begin(), m_points.end());
        } else {
            std::sort(m_points.begin(), m_points.end());
        }

        // Rotate the Morton sequence to start from city 0
        long long startIdx = 0;
        for (long long i = 0; i < N; ++i) {
            if (m_points[(size_t)i].id == 0) {
                startIdx = i;
                break;
            }
        }

        for (long long i = 0; i < N; ++i) {
            write_int(m_points[(size_t)((startIdx + i) % N)].id);
            put_char(i == N - 1 ? '\n' : ' ');
        }
    }

    flush_out();
    return 0;
}
