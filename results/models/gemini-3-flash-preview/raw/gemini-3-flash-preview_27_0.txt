#include <vector>
#include <algorithm>
#include <numeric>

#ifdef _MSC_VER
#include <intrin.h>
#else
#include <x86intrin.h>
#endif

using namespace std;

/**
 * Fast popcount and count trailing zeros (ctz) using compiler intrinsics.
 * These are essential for the bitmask-based color selection in the DSATUR algorithm.
 */
#ifdef _MSC_VER
inline int count_trailing_zeros(unsigned __int64 mask) {
    unsigned long index;
    if (_BitScanForward64(&index, mask)) return (int)index;
    return 64;
}
inline int popcount64(unsigned __int64 mask) {
    return (int)__popcnt64(mask);
}
#else
inline int count_trailing_zeros(unsigned long long mask) {
    if (mask == 0) return 64;
    return __builtin_ctzll(mask);
}
inline int popcount64(unsigned long long mask) {
    return __builtin_popcountll(mask);
}
#endif

struct Edge {
    int to;
    int next;
};

/**
 * Memory-efficient graph representation using a static adjacency list (edges and head vectors).
 * This structure avoids the overhead of std::vector<std::vector<int>> and is faster for large graphs.
 */
struct Graph {
    int n;
    vector<int> head;
    vector<Edge> edges;
    vector<int> degree;

    Graph(int n, int m) : n(n), head(n, -1), degree(n, 0) {
        if (m > 0) edges.reserve(2 * (size_t)m);
    }

    void add_edge(int u, int v) {
        edges.push_back({v, head[u]});
        head[u] = (int)edges.size() - 1;
        degree[u]++;
        edges.push_back({u, head[v]});
        head[v] = (int)edges.size() - 1;
        degree[v]++;
    }
};

/**
 * Node for the priority queue in the DSATUR algorithm.
 * DSATUR prioritizes vertices with the highest saturation degree (number of distinct colors used by neighbors).
 * Ties are broken by the static degree of the vertex in the original graph.
 */
struct Node {
    int v;
    int sat;
    int deg;
    int rand_val;
    bool operator<(const Node& other) const {
        if (sat != other.sat) return sat < other.sat;
        if (deg != other.deg) return deg < other.deg;
        return rand_val < other.rand_val;
    }
};

/**
 * DSATUR algorithm implementation.
 * It tracks the colors used by neighbors using bitmasks for speed and memory efficiency.
 * If the number of colors exceeds the bitmask width, it falls back to a slower but correct search.
 */
vector<int> solve_dsatur(const Graph& g, int k_limit, unsigned int seed) {
    int n = g.n;
    if (n == 0) return {};
    vector<int> colors(n, -1);
    vector<int> saturation(n, 0);
    
    // Determine how many colors to track in the bitmasks. 
    // We prioritize the given k_limit but cap it to prevent excessive memory usage on massive graphs.
    int track_colors = k_limit;
    if (track_colors < 128) track_colors = 128;
    // Limit bitmask memory to roughly 1GB. For 1M nodes, this allows tracking ~8192 colors.
    if ((long long)n * ((track_colors + 63) / 64) * 8 > 1024 * 1024 * 1024) {
        track_colors = 64; 
    }
    int width = (track_colors + 63) / 64;
    vector<uint64_t> masks((size_t)n * width, 0);

    mt19937 rng(seed);
    vector<int> rand_vals(n);
    for (int i = 0; i < n; ++i) rand_vals[i] = (int)rng();

    priority_queue<Node> pq;
    for (int i = 0; i < n; ++i) {
        pq.push({i, 0, g.degree[i], rand_vals[i]});
    }

    vector<int> used_at;
    int timer = 0;

    int colored_count = 0;
    while (!pq.empty()) {
        Node top = pq.top();
        pq.pop();

        int u = top.v;
        if (colors[u] != -1) continue;
        if (top.sat != saturation[u]) continue;

        // Find the smallest available color (Mex - Minimum Excluded value)
        int color = -1;
        for (int i = 0; i < width; ++i) {
            uint64_t m = masks[(size_t)u * width + i];
            if (m != ~0ULL) {
                int bit = count_trailing_zeros(~m);
                int res = i * 64 + bit;
                if (res < track_colors) {
                    color = res;
                    break;
                }
            }
        }

        // Fallback for colors beyond the tracked bitmask width
        if (color == -1) {
            if (used_at.empty()) used_at.assign(max(n + 1, k_limit + 1), 0);
            timer++;
            for (int i = g.head[u]; i != -1; i = g.edges[i].next) {
                int v = g.edges[i].to;
                if (colors[v] != -1) {
                    if (colors[v] < (int)used_at.size()) used_at[colors[v]] = timer;
                }
            }
            for (int c = 0; ; ++c) {
                if (c >= (int)used_at.size() || used_at[c] != timer) {
                    color = c;
                    break;
                }
            }
        }

        colors[u] = color;
        colored_count++;

        // Update neighbors' saturation
        for (int i = g.head[u]; i != -1; i = g.edges[i].next) {
            int v = g.edges[i].to;
            if (colors[v] == -1) {
                if (color < track_colors) {
                    size_t idx = (size_t)v * width + (color / 64);
                    uint64_t bit = 1ULL << (color % 64);
                    if (!(masks[idx] & bit)) {
                        masks[idx] |= bit;
                        saturation[v]++;
                        pq.push({v, saturation[v], g.degree[v], rand_vals[v]});
                    }
                }
            }
        }
    }

    return colors;
}

int main() {
    // Fast I/O for handling massive graphs
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    Graph g(n, m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u >= 0 && u < n && v >= 0 && v < n && u != v) {
            g.add_edge(u, v);
        }
    }

    auto start_time = chrono::steady_clock::now();
    
    int attempts = 0;
    vector<int> best_colors;
    int min_colors = n + 1;
    
    // Adaptive logic: try multiple times for small/medium graphs, once for massive ones.
    while (attempts < 1) {
        vector<int> current_colors = solve_dsatur(g, k, 42 + attempts);
        int current_max = -1;
        for (int c : current_colors) if (c > current_max) current_max = c;
        int current_num_colors = current_max + 1;
        
        if (current_num_colors < min_colors) {
            min_colors = current_num_colors;
            best_colors = current_colors;
        }
        attempts++;
        
        auto now = chrono::steady_clock::now();
        auto elapsed = chrono::duration_cast<chrono::seconds>(now - start_time).count();
        if (elapsed > 240) break; // Leave buffer for final processing
        if (min_colors <= k) break; // Goal achieved

        // Allow more attempts for smaller instances within time limits
        if (n < 10000 && attempts < 5) continue;
        if (n < 100000 && attempts < 2) continue;
        break;
    }

    // Output the color assignment for each vertex
    for (int i = 0; i < n; ++i) {
        cout << (best_colors.empty() ? 0 : best_colors[i]) << (i == n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
