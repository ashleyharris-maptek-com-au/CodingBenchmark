use std::io::{self, BufRead, BufReader, BufWriter, Write};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    let mut line = String::new();
    if reader.read_line(&mut line).is_err() || line.trim().is_empty() {
        return;
    }
    let mut parts = line.split_whitespace();
    let h: usize = match parts.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return,
    };
    let w: usize = match parts.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return,
    };

    if h == 0 || w == 0 {
        return;
    }

    let mut grid = vec![0u8; h * w];
    let mut start_idx = usize::MAX;
    let mut end_idx = usize::MAX;

    let mut buf = Vec::with_capacity(w + 16);
    for r in 0..h {
        buf.clear();
        if reader.read_until(b'\n', &mut buf).is_err() {
            break;
        }
        for c in 0..w {
            if c >= buf.len() { break; }
            let b = buf[c];
            match b {
                b'#' => grid[r * w + c] = 1,
                b'A' => {
                    start_idx = r * w + c;
                    grid[r * w + c] = 2;
                }
                b'B' => {
                    end_idx = r * w + c;
                    grid[r * w + c] = 3;
                }
                b' ' => grid[r * w + c] = 0,
                _ => grid[r * w + c] = 0,
            }
        }
    }

    if start_idx == usize::MAX || end_idx == usize::MAX {
        return;
    }

    let mut queue = VecDeque::with_capacity((w + h).min(h * w));
    queue.push_back(start_idx as u32);
    
    // Internal representation:
    // 0: Unvisited Path
    // 1: Wall
    // 2: Start
    // 3: Goal
    // 4: Visited from Left (parent is idx - 1)
    // 5: Visited from Right (parent is idx + 1)
    // 6: Visited from Up (parent is idx - w)
    // 7: Visited from Down (parent is idx + w)
    // 8: Start marked as visited
    grid[start_idx] = 8;

    let mut found = false;
    if start_idx == end_idx {
        found = true;
    } else {
        while let Some(curr_u32) = queue.pop_front() {
            let curr = curr_u32 as usize;
            let r = curr / w;
            let c = curr % w;

            // Neighbors: Up, Down, Left, Right
            
            // Up
            if r > 0 {
                let nidx = curr - w;
                let val = grid[nidx];
                if val == 0 || val == 3 {
                    grid[nidx] = 7;
                    if val == 3 { found = true; break; }
                    queue.push_back(nidx as u32);
                }
            }
            // Down
            if r + 1 < h {
                let nidx = curr + w;
                let val = grid[nidx];
                if val == 0 || val == 3 {
                    grid[nidx] = 6;
                    if val == 3 { found = true; break; }
                    queue.push_back(nidx as u32);
                }
            }
            // Left
            if c > 0 {
                let nidx = curr - 1;
                let val = grid[nidx];
                if val == 0 || val == 3 {
                    grid[nidx] = 5;
                    if val == 3 { found = true; break; }
                    queue.push_back(nidx as u32);
                }
            }
            // Right
            if c + 1 < w {
                let nidx = curr + 1;
                let val = grid[nidx];
                if val == 0 || val == 3 {
                    grid[nidx] = 4;
                    if val == 3 { found = true; break; }
                    queue.push_back(nidx as u32);
                }
            }
        }
    }

    if found {
        let mut path = Vec::new();
        let mut curr = end_idx;
        path.push(curr as u32);
        while curr != start_idx {
            let val = grid[curr];
            match val {
                4 => curr -= 1,
                5 => curr += 1,
                6 => curr -= w,
                7 => curr += w,
                _ => break,
            }
            path.push(curr as u32);
        }
        
        let stdout = io::stdout();
        let mut writer = BufWriter::new(stdout.lock());
        let _ = writeln!(writer, "{}", path.len());
        for &idx_u32 in path.iter().rev() {
            let idx = idx_u32 as usize;
            let _ = writeln!(writer, "{} {}", idx % w, idx / w);
        }
        let _ = writer.flush();
    } else {
        println!("0");
    }
}
