use std::io::{self, Read, Write, BufReader, BufWriter, BufRead};

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    #[inline(always)]
    fn dist_sq(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

struct KdNode {
    point: Point,
    idx: u32,
    left: Option<Box<KdNode>>,
    right: Option<Box<KdNode>>,
    axis: usize,
}

struct KdTree {
    root: Option<Box<KdNode>>,
}

impl KdTree {
    fn build(mut items: Vec<(Point, u32)>) -> Self {
        fn build_rec(items: &mut [(Point, u32)], depth: usize) -> Option<Box<KdNode>> {
            if items.is_empty() {
                return None;
            }
            let axis = depth % 3;
            items.sort_by(|a, b| {
                let va = match axis {
                    0 => a.0.x,
                    1 => a.0.y,
                    _ => a.0.z,
                };
                let vb = match axis {
                    0 => b.0.x,
                    1 => b.0.y,
                    _ => b.0.z,
                };
                va.partial_cmp(&vb).unwrap_or(std::cmp::Ordering::Equal)
            });
            let mid = items.len() / 2;
            let (point, idx) = items[mid];
            Some(Box::new(KdNode {
                point,
                idx,
                left: build_rec(&mut items[..mid], depth + 1),
                right: build_rec(&mut items[mid + 1..], depth + 1),
                axis,
            }))
        }
        KdTree {
            root: build_rec(&mut items, 0),
        }
    }

    fn find_nearest(&self, p: &Point, best_idx: &mut u32, best_dist: &mut f32) {
        if let Some(ref root) = self.root {
            self.find_rec(root, p, best_idx, best_dist);
        }
    }

    fn find_rec(&self, node: &KdNode, p: &Point, best_idx: &mut u32, best_dist: &mut f32) {
        let d = p.dist_sq(&node.point);
        if d < *best_dist {
            *best_dist = d;
            *best_idx = node.idx;
        }

        let diff = match node.axis {
            0 => p.x - node.point.x,
            1 => p.y - node.point.y,
            _ => p.z - node.point.z,
        };

        let (near, far) = if diff < 0.0 {
            (&node.left, &node.right)
        } else {
            (&node.right, &node.left)
        };

        if let Some(ref n) = near {
            self.find_rec(n, p, best_idx, best_dist);
        }

        if diff * diff < *best_dist {
            if let Some(ref f) = far {
                self.find_rec(f, p, best_idx, best_dist);
            }
        }
    }
}

fn rand_simple(seed: &mut u64) -> f64 {
    *seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1);
    ((*seed >> 32) as u32) as f64 / 4294967296.0
}

#[inline(always)]
fn parse_point(buf: &[u8]) -> Point {
    Point {
        x: f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]),
        y: f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]),
        z: f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]),
    }
}

fn main() -> io::Result<()> {
    let stdin_raw = io::stdin();
    let mut reader = BufReader::with_capacity(1024 * 1024, stdin_raw);
    
    let mut header = String::new();
    reader.read_line(&mut header)?;
    let parts: Vec<&str> = header.split_whitespace().collect();
    if parts.len() < 2 {
        return Ok(());
    }
    let num_points: u64 = parts[0].parse().unwrap_or(0);
    let num_clusters: usize = parts[1].parse().unwrap_or(0);

    if num_clusters == 0 || num_points == 0 {
        for _ in 0..num_points {
            io::stdout().write_all(&0u32.to_le_bytes())?;
        }
        return Ok(());
    }

    // 1. Sample the data
    let sample_limit = 1_000_000;
    let sample_size = if num_points < sample_limit as u64 { num_points as usize } else { sample_limit };
    let mut sample = Vec::with_capacity(sample_size);
    let mut buf = [0u8; 12];
    for _ in 0..sample_size {
        reader.read_exact(&mut buf)?;
        sample.push(parse_point(&buf));
    }

    // 2. Initialize Centroids (K-Means++)
    let mut centroids = vec![Point { x: 0.0, y: 0.0, z: 0.0 }; num_clusters];
    let mut seed = 42u64;
    centroids[0] = sample[0];
    let mut min_dists = vec![f32::INFINITY; sample.len()];
    
    for i in 1..num_clusters {
        let prev = centroids[i - 1];
        let mut sum_dist = 0.0f64;
        for (j, p) in sample.iter().enumerate() {
            let d = p.dist_sq(&prev);
            if d < min_dists[j] {
                min_dists[j] = d;
            }
            sum_dist += min_dists[j] as f64;
        }
        
        if sum_dist > 0.0 {
            let mut target = rand_simple(&mut seed) * sum_dist;
            let mut found = false;
            for (j, p) in sample.iter().enumerate() {
                target -= min_dists[j] as f64;
                if target <= 0.0 {
                    centroids[i] = *p;
                    found = true;
                    break;
                }
            }
            if !found {
                centroids[i] = sample[sample.len() - 1];
            }
        } else {
            // All points are identical, just fill the rest
            for k in i..num_clusters {
                centroids[k] = sample[0];
            }
            break;
        }
    }

    // 3. Refine Centroids on Sample
    let iterations = if num_points <= 10_000_000 { 15 } else { 5 };
    for _ in 0..iterations {
        let mut sums = vec![(0.0f64, 0.0f64, 0.0f64, 0u64); num_clusters];
        if num_clusters <= 128 {
            for p in &sample {
                let mut min_d = f32::INFINITY;
                let mut min_i = 0;
                for (i, c) in centroids.iter().enumerate() {
                    let d = p.dist_sq(c);
                    if d < min_d {
                        min_d = d;
                        min_i = i;
                    }
                }
                let s = &mut sums[min_i];
                s.0 += p.x as f64; s.1 += p.y as f64; s.2 += p.z as f64; s.3 += 1;
            }
        } else {
            let tree = KdTree::build(centroids.iter().enumerate().map(|(i, &c)| (c, i as u32)).collect());
            for p in &sample {
                let mut min_i = 0;
                let mut min_d = f32::INFINITY;
                tree.find_nearest(p, &mut min_i, &mut min_d);
                let s = &mut sums[min_i as usize];
                s.0 += p.x as f64; s.1 += p.y as f64; s.2 += p.z as f64; s.3 += 1;
            }
        }
        for i in 0..num_clusters {
            let s = sums[i];
            if s.3 > 0 {
                centroids[i] = Point {
                    x: (s.0 / s.3 as f64) as f32,
                    y: (s.1 / s.3 as f64) as f32,
                    z: (s.2 / s.3 as f64) as f32,
                };
            }
        }
    }

    // 4. Output Phase
    let mut stdout = BufWriter::with_capacity(1024 * 1024, io::stdout());
    if num_clusters <= 128 {
        let mut counts = vec![1u64; num_clusters];
        let use_online = num_points > 100_000_000;
        
        // Output for sample points
        for p in &sample {
            let mut min_d = f32::INFINITY;
            let mut min_i = 0u32;
            for (i, c) in centroids.iter().enumerate() {
                let d = p.dist_sq(c);
                if d < min_d {
                    min_d = d;
                    min_i = i as u32;
                }
            }
            stdout.write_all(&min_i.to_le_bytes())?;
            if use_online {
                let c = &mut centroids[min_i as usize];
                let alpha = 1.0 / (counts[min_i as usize] + 1) as f32;
                c.x += (p.x - c.x) * alpha;
                c.y += (p.y - c.y) * alpha;
                c.z += (p.z - c.z) * alpha;
                counts[min_i as usize] += 1;
            }
        }
        
        // Output for remaining points
        let mut b = [0u8; 12];
        for _ in (sample_size as u64)..num_points {
            reader.read_exact(&mut b)?;
            let p = parse_point(&b);
            let mut min_d = f32::INFINITY;
            let mut min_i = 0u32;
            for (i, c) in centroids.iter().enumerate() {
                let d = p.dist_sq(c);
                if d < min_d {
                    min_d = d;
                    min_i = i as u32;
                }
            }
            stdout.write_all(&min_i.to_le_bytes())?;
            if use_online {
                let c = &mut centroids[min_i as usize];
                let alpha = 1.0 / (counts[min_i as usize] + 1) as f32;
                c.x += (p.x - c.x) * alpha;
                c.y += (p.y - c.y) * alpha;
                c.z += (p.z - c.z) * alpha;
                counts[min_i as usize] += 1;
            }
        }
    } else {
        // KD-Tree for large N
        let tree = KdTree::build(centroids.iter().enumerate().map(|(i, &c)| (c, i as u32)).collect());
        let mut out_buf = vec![0u8; 4096 * 4];
        let mut batch_buf = vec![0u8; 4096 * 12];
        
        // Output for sample points
        for chunk in sample.chunks(4096) {
            for (i, p) in chunk.iter().enumerate() {
                let mut best_idx = 0;
                let mut best_dist = f32::INFINITY;
                tree.find_nearest(p, &mut best_idx, &mut best_dist);
                let bytes = best_idx.to_le_bytes();
                out_buf[i * 4..i * 4 + 4].copy_from_slice(&bytes);
            }
            stdout.write_all(&out_buf[..chunk.len() * 4])?;
        }
        
        // Output for remaining points
        let mut points_processed = sample_size as u64;
        while points_processed < num_points {
            let to_read = (num_points - points_processed).min(4096);
            reader.read_exact(&mut batch_buf[..to_read as usize * 12])?;
            for i in 0..to_read as usize {
                let p = parse_point(&batch_buf[i * 12..i * 12 + 12]);
                let mut best_idx = 0;
                let mut best_dist = f32::INFINITY;
                tree.find_nearest(&p, &mut best_idx, &mut best_dist);
                let bytes = best_idx.to_le_bytes();
                out_buf[i * 4..i * 4 + 4].copy_from_slice(&bytes);
            }
            stdout.write_all(&out_buf[..to_read as usize * 4])?;
            points_processed += to_read;
        }
    }
    stdout.flush()?;
    Ok(())
}
