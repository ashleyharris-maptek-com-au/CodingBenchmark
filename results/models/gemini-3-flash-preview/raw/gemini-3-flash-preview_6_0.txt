using System;
using System.Collections.Generic;
using System.Linq;
using System.Globalization;

namespace OrbitalTSP
{
    public struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator *(Vector3 a, double b) => new Vector3(a.X * b, a.Y * b, a.Z * b);
        public static Vector3 operator *(double a, Vector3 b) => new Vector3(a * b.X, a * b.Y, a * b.Z);
        public static Vector3 operator /(Vector3 a, double b) => new Vector3(a.X / b, a.Y / b, a.Z / b);
        public double Length => Math.Sqrt(X * X + Y * Y + Z * Z);
        public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    public class Station
    {
        public Vector3 Pos, Vel, H, E;
        public double A;
    }

    class Program
    {
        static double MU;

        static void Main(string[] args)
        {
            CultureInfo.DefaultThreadCurrentCulture = CultureInfo.InvariantCulture;
            string line1 = Console.ReadLine();
            if (string.IsNullOrEmpty(line1)) return;
            string[] parts1 = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int N = int.Parse(parts1[0]);
            MU = double.Parse(parts1[1]);

            Station start = ParseStation(Console.ReadLine());
            Station[] stations = new Station[N];
            for (int i = 0; i < N; i++) stations[i] = ParseStation(Console.ReadLine());

            double[,] dist = new double[N + 1, N + 1];
            Station[] all = new Station[N + 1];
            all[0] = start;
            for (int i = 0; i < N; i++) all[i + 1] = stations[i];

            for (int i = 0; i <= N; i++)
                for (int j = 0; j <= N; j++)
                    if (i != j) dist[i, j] = CalcCost(all[i], all[j]);

            int[] path = new int[N + 1];
            bool[] used = new bool[N + 1];
            path[0] = 0; used[0] = true;
            for (int i = 1; i <= N; i++)
            {
                int best = -1;
                for (int j = 1; j <= N; j++)
                    if (!used[j] && (best == -1 || dist[path[i - 1], j] < dist[path[i - 1], best])) best = j;
                path[i] = best; used[best] = true;
            }

            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 1; i < N; i++)
                {
                    for (int j = i + 1; j <= N; j++)
                    {
                        double oldD = dist[path[i - 1], path[i]] + (j < N ? dist[path[j], path[j + 1]] : 0);
                        double newD = dist[path[i - 1], path[j]] + (j < N ? dist[path[i], path[j + 1]] : 0);
                        if (newD < oldD - 1e-9)
                        {
                            Array.Reverse(path, i, j - i + 1);
                            improved = true;
                        }
                    }
                }
            }
            Console.WriteLine(string.Join(" ", path.Skip(1).Select(x => x - 1)));
        }

        static Station ParseStation(string line)
        {
            double[] v = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries).Select(double.Parse).ToArray();
            Vector3 p = new Vector3(v[0], v[1], v[2]), vel = new Vector3(v[3], v[4], v[5]);
            Vector3 h = Vector3.Cross(p, vel), e = (Vector3.Cross(vel, h) / MU) - (p / p.Length);
            return new Station { Pos = p, Vel = vel, H = h, E = e, A = -MU / (2.0 * (vel.Length * vel.Length / 2.0 - MU / p.Length)) };
        }

        static double CalcCost(Station s1, Station s2)
        {
            double v = (s1.Vel.Length + s2.Vel.Length) / 2, cos = Math.Clamp(Vector3.Dot(s1.H, s2.H) / (s1.H.Length * s2.H.Length), -1, 1);
            return Math.Sqrt(Math.Pow(2 * v * Math.Sin(Math.Acos(cos) / 2), 2) + Math.Pow(0.5 * v * Math.Sqrt(Math.Pow((Math.Abs(s1.A) - Math.Abs(s2.A)) / ((Math.Abs(s1.A) + Math.Abs(s2.A)) / 2), 2) + (s1.E - s2.E).LengthSquared()), 2)) + (s1.Pos - s2.Pos).Length / 10000;
        }
    }

    public static class Ext { public static double LengthSquared(this Vector3 v) => v.X * v.X + v.Y * v.Y + v.Z * v.Z; }
}
