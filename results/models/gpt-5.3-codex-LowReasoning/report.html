#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

struct City {
    float x;
    float y;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static std::vector<int> nearest_neighbor(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    std::vector<int> tour;
    tour.reserve(n);

    std::vector<unsigned char> used(n, 0);
    int cur = 0;
    used[0] = 1;
    tour.push_back(0);

    for (int step = 1; step < n; ++step) {
        int best = -1;
        double bestd = std::numeric_limits<double>::infinity();
        for (int j = 1; j < n; ++j) {
            if (!used[j]) {
                const double d = dist2(c[cur], c[j]);
                if (d < bestd) {
                    bestd = d;
                    best = j;
                }
            }
        }
        if (best < 0) {
            for (int j = 1; j < n; ++j) {
                if (!used[j]) {
                    best = j;
                    break;
                }
            }
        }
        used[best] = 1;
        tour.push_back(best);
        cur = best;
    }
    return tour;
}

static void two_opt_limited(const std::vector<City>& c, std::vector<int>& tour, int max_passes, int window) {
    const int n = static_cast<int>(tour.size());
    if (n < 4) return;

    for (int pass = 0; pass < max_passes; ++pass) {
        bool improved = false;
        for (int i = 0; i + 2 < n; ++i) {
            const int a = tour[i];
            const int b = tour[(i + 1) % n];
            const int j_end = std::min(n - 1, i + window);

            for (int j = i + 2; j <= j_end; ++j) {
                if (i == 0 && j == n - 1) continue;
                const int c1 = tour[j];
                const int d = tour[(j + 1) % n];

                const double old_cost = dist2(c[a], c[b]) + dist2(c[c1], c[d]);
                const double new_cost = dist2(c[a], c[c1]) + dist2(c[b], c[d]);

                if (new_cost + 1e-12 < old_cost) {
                    std::reverse(tour.begin() + (i + 1), tour.begin() + (j + 1));
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

struct HilbertKey {
    uint32_t key;
    int idx;
};

static inline uint32_t part1by1(uint16_t x) {
    uint32_t v = x;
    v = (v | (v << 8)) & 0x00FF00FFu;
    v = (v | (v << 4)) & 0x0F0F0F0Fu;
    v = (v | (v << 2)) & 0x33333333u;
    v = (v | (v << 1)) & 0x55555555u;
    return v;
}

static inline uint32_t morton2D(uint16_t x, uint16_t y) {
    return (part1by1(y) << 1) | part1by1(x);
}

static std::vector<int> space_filling_order(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    std::vector<int> order;
    order.reserve(n);

    if (n == 0) return order;
    if (n == 1) return {0};

    float minx = c[0].x, maxx = c[0].x;
    float miny = c[0].y, maxy = c[0].y;
    for (const auto& p : c) {
        if (p.x < minx) minx = p.x;
        if (p.x > maxx) maxx = p.x;
        if (p.y < miny) miny = p.y;
        if (p.y > maxy) maxy = p.y;
    }

    const double rangex = std::max(1e-20, static_cast<double>(maxx) - minx);
    const double rangey = std::max(1e-20, static_cast<double>(maxy) - miny);

    std::vector<HilbertKey> keys;
    keys.reserve(n - 1);

    for (int i = 1; i < n; ++i) {
        const double nx = (static_cast<double>(c[i].x) - minx) / rangex;
        const double ny = (static_cast<double>(c[i].y) - miny) / rangey;
        const uint16_t qx = static_cast<uint16_t>(std::min(65535.0, std::max(0.0, nx * 65535.0)));
        const uint16_t qy = static_cast<uint16_t>(std::min(65535.0, std::max(0.0, ny * 65535.0)));
        keys.push_back({morton2D(qx, qy), i});
    }

    std::sort(keys.begin(), keys.end(), [](const HilbertKey& a, const HilbertKey& b) {
        if (a.key != b.key) return a.key < b.key;
        return a.idx < b.idx;
    });

    order.push_back(0);
    for (const auto& k : keys) order.push_back(k.idx);
    return order;
}

static std::vector<int> brute_force_small(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    if (n <= 1) return {0};

    std::vector<int> perm(n - 1);
    std::iota(perm.begin(), perm.end(), 1);

    std::vector<int> best = perm;
    double best_cost = std::numeric_limits<double>::infinity();

    do {
        double cost = 0.0;
        int prev = 0;
        for (int v : perm) {
            cost += dist2(c[prev], c[v]);
            prev = v;
        }
        cost += dist2(c[prev], c[0]);

        if (cost < best_cost) {
            best_cost = cost;
            best = perm;
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    std::vector<int> tour;
    tour.reserve(n);
    tour.push_back(0);
    for (int v : best) tour.push_back(v);
    return tour;
}

            if (!seen[i]) fixed.push_back(i);
        }

    }

        if (i) std::cout << ' ';
        std::cout << tour[i];
    }
    std::cout << '\n';
    return 0;
}
