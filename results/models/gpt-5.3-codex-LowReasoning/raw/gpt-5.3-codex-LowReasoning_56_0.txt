const fs = require('fs');

function readInput() {
  const input = fs.readFileSync(0, 'utf8');
  return JSON.parse(input);
}

function dist(a, b) {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
}

function routeLength(route, points, depot) {
  if (route.length === 0) return 0;
  let total = dist(depot, points[route[0]]);
  for (let i = 1; i < route.length; i++) {
    total += dist(points[route[i - 1]], points[route[i]]);
  }
  total += dist(points[route[route.length - 1]], depot);
  return total;
}

function nearestNeighborOrder(indices, points, depot) {
  const n = indices.length;
  if (n <= 1) return indices.slice();

  let startPos = 0;
  let bestStart = Infinity;
  for (let i = 0; i < n; i++) {
    const d = dist(depot, points[indices[i]]);
    if (d < bestStart) {
      bestStart = d;
      startPos = i;
    }
  }

  const used = new Array(n).fill(false);
  const order = [];
  let currPos = startPos;
  used[currPos] = true;
  order.push(indices[currPos]);

  for (let step = 1; step < n; step++) {
    const currIdx = indices[currPos];
    let nextPos = -1;
    let best = Infinity;
    for (let j = 0; j < n; j++) {
      if (used[j]) continue;
      const cand = dist(points[currIdx], points[indices[j]]);
      if (cand < best) {
        best = cand;
        nextPos = j;
      }
    }
    used[nextPos] = true;
    order.push(indices[nextPos]);
    currPos = nextPos;
  }

  return order;
}

function twoOpt(route, points, depot) {
  const n = route.length;
  if (n < 4) return route;

  let improved = true;
  let guard = 0;
  const maxOuter = 20;

  while (improved && guard < maxOuter) {
    improved = false;
    guard++;

    for (let i = 0; i < n - 2; i++) {
      const a = (i === 0) ? depot : points[route[i - 1]];
      const b = points[route[i]];

      for (let k = i + 1; k < n - 1; k++) {
        const c = points[route[k]];
        const d = points[route[k + 1]];

        const before = dist(a, b) + dist(c, d);
        const after = dist(a, c) + dist(b, d);

        if (after + 1e-9 < before) {
          let l = i;
          let r = k;
          while (l < r) {
            const tmp = route[l];
            route[l] = route[r];
            route[r] = tmp;
            l++;
            r--;
          }
          improved = true;
        }
      }

      const c2 = points[route[n - 1]];
      const d2 = depot;
      const before2 = dist(a, b) + dist(c2, d2);
      const after2 = dist(a, c2) + dist(b, d2);
      if (after2 + 1e-9 < before2) {
        let l = i;
        let r = n - 1;
        while (l < r) {
          const tmp = route[l];
          route[l] = route[r];
          route[r] = tmp;
          l++;
          r--;
        }
        improved = true;
      }
    }
  }

  return route;
}

function splitByAngle(points, depot, truckCount) {
  const n = points.length;
  const t = Math.max(1, Math.min(truckCount || 1, n));

  const items = new Array(n);
  for (let i = 0; i < n; i++) {
    const p = points[i];
    const angle = Math.atan2(p.y - depot.y, p.x - depot.x);
    const r = dist(p, depot);
    items[i] = { idx: i, angle, r };
  }

  items.sort((a, b) => a.angle - b.angle || a.r - b.r);

  const routes = Array.from({ length: t }, () => []);
  const base = Math.floor(n / t);
  let extra = n % t;
  let pos = 0;
  for (let i = 0; i < t; i++) {
    const size = base + (extra > 0 ? 1 : 0);
    if (extra > 0) extra--;
    for (let j = 0; j < size; j++) {
      routes[i].push(items[pos++].idx);
    }
  }

  return routes;
}

function rebalance(routes, points, depot, iterations) {
  const nonEmpty = () => routes.filter(r => r.length > 0);

  for (let iter = 0; iter < iterations; iter++) {
    let bestDelta = 0;
    let bestMove = null;

    for (let a = 0; a < routes.length; a++) {
      const ra = routes[a];
      if (ra.length <= 1) continue;
      const lenA = routeLength(ra, points, depot);

      for (let ia = 0; ia < ra.length; ia++) {
        const moved = ra[ia];

        for (let b = 0; b < routes.length; b++) {
          if (a === b) continue;
          const rb = routes[b];

          const ra2 = ra.slice(0, ia).concat(ra.slice(ia + 1));

          let bestLocalLen = Infinity;
          let bestPos = 0;
          for (let pos = 0; pos <= rb.length; pos++) {
            const rb2 = rb.slice(0, pos).concat([moved], rb.slice(pos));
            const candidateLen = routeLength(rb2, points, depot);
            if (candidateLen < bestLocalLen) {
              bestLocalLen = candidateLen;
              bestPos = pos;
            }
          }

          const delta = (routeLength(ra2, points, depot) + bestLocalLen) - (lenA + routeLength(rb, points, depot));

          if (delta < bestDelta) {
            bestDelta = delta;
            bestMove = { a, ia, b, insertPos: bestPos };
          }
        }
      }
    }

    if (!bestMove) break;

    const from = routes[bestMove.a];
    const [node] = from.splice(bestMove.ia, 1);
    const to = routes[bestMove.b];
    to.splice(bestMove.insertPos, 0, node);

    routes = nonEmpty();
  }

  return routes;
}

function solve(data) {
  const depot = { x: data.depot[0], y: data.depot[1] };
  const packages = data.packages || [];

  if (packages.length === 0) {
    return { routes: [] };
  }

  const points = packages.map(p => ({ x: p.x, y: p.y, id: p.id }));
  const truckCount = Math.max(1, Math.floor(data.truckCount || 1));

  let clusters = splitByAngle(points, depot, truckCount);

  for (let i = 0; i < clusters.length; i++) {
    clusters[i] = nearestNeighborOrder(clusters[i], points, depot);
    clusters[i] = twoOpt(clusters[i], points, depot);
  }

  clusters = rebalance(clusters, points, depot, 3);

  for (let i = 0; i < clusters.length; i++) {
    clusters[i] = nearestNeighborOrder(clusters[i], points, depot);
    clusters[i] = twoOpt(clusters[i], points, depot);
  }

  const routes = clusters.map(route => route.map(idx => points[idx].id));
  return { routes };
}

const data = readInput();
const result = solve(data);
process.stdout.write(JSON.stringify(result));
