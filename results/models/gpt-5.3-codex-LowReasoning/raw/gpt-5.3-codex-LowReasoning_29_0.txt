using System;
using System.Collections.Generic;
using System.Linq;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly System.IO.Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sign;
        }
    }

    private static List<int>[] _adj = Array.Empty<List<int>>();
    private static bool[][] _mat = Array.Empty<bool[]>();

    private static List<int> _best = new List<int>();

    public static void Main()
    {
        var fs = new FastScanner();
        int n;
        try { n = fs.NextInt(); }
        catch { return; }
        int m = fs.NextInt();

        _adj = new List<int>[n];
        for (int i = 0; i < n; i++) _adj[i] = new List<int>();

        _mat = new bool[n][];
        for (int i = 0; i < n; i++) _mat[i] = new bool[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u == v) continue;
            if (!_mat[u][v])
            {
                _mat[u][v] = _mat[v][u] = true;
                _adj[u].Add(v);
                _adj[v].Add(u);
            }
        }

        // For very large graphs, exact MC is infeasible; run high-quality heuristic.
        // For smaller graphs, use branch-and-bound exact search with greedy coloring bound.
        if (n <= 220)
        {
            SolveExact(n);
        }
        else
        {
            SolveHeuristic(n);
        }

        _best.Sort();
        Console.WriteLine(_best.Count);
        if (_best.Count > 0)
            Console.WriteLine(string.Join(" ", _best));
        else
            Console.WriteLine();
    }

    private static void SolveExact(int n)
    {
        var order = Enumerable.Range(0, n)
            .OrderByDescending(v => _adj[v].Count)
            .ToList();

        var current = new List<int>(n);
        Expand(order, current);

        // Safety fallback if exact failed to improve from empty for any reason.
        if (_best.Count == 0 && n > 0)
        {
            int v = 0;
            for (int i = 1; i < n; i++) if (_adj[i].Count > _adj[v].Count) v = i;
            _best.Add(v);
        }
    }

    private static void Expand(List<int> candidates, List<int> current)
    {
        if (candidates.Count == 0)
        {
            if (current.Count > _best.Count) _best = new List<int>(current);
            return;
        }

        GreedyColorBound(candidates, out var order, out var colors);

        for (int i = order.Count - 1; i >= 0; i--)
        {
            if (current.Count + colors[i] <= _best.Count) return;

            int v = order[i];
            current.Add(v);

            var next = new List<int>();
            for (int j = 0; j < i; j++)
            {
                int u = order[j];
                if (_mat[v][u]) next.Add(u);
            }

            if (next.Count == 0)
            {
                if (current.Count > _best.Count) _best = new List<int>(current);
            }
            else
            {
                Expand(next, current);
            }

            current.RemoveAt(current.Count - 1);
        }
    }

    private static void GreedyColorBound(List<int> candidates, out List<int> order, out List<int> colors)
    {
        order = new List<int>(candidates.Count);
        colors = new List<int>(candidates.Count);

        // Process high-degree first for tighter coloring.
        var work = candidates.OrderByDescending(v => _adj[v].Count).ToList();
        var bins = new List<List<int>>();

        foreach (int v in work)
        {
            int c = 0;
            for (; c < bins.Count; c++)
            {
                bool ok = true;
                var bin = bins[c];
                for (int i = 0; i < bin.Count; i++)
                {
                    if (_mat[v][bin[i]]) { ok = false; break; }
                }
                if (ok)
                {
                    bin.Add(v);
                    break;
                }
            }
            if (c == bins.Count)
            {
                bins.Add(new List<int> { v });
            }
        }

        for (int c = 0; c < bins.Count; c++)
        {
            foreach (int v in bins[c])
            {
                order.Add(v);
                colors.Add(c + 1);
            }
        }
    }

    private static void SolveHeuristic(int n)
    {
        // Multi-start greedy + local improvement; deterministic seed for reproducibility.
        var rnd = new Random(123456789);

        // Degree-based ordering base.
        var byDeg = Enumerable.Range(0, n).OrderByDescending(v => _adj[v].Count).ToArray();

        // Start count scaled by n, capped for runtime safety.
        int starts = Math.Min(64, Math.Max(12, 2000000 / Math.Max(1, n)));

        List<int> bestLocal = new List<int>();

        for (int s = 0; s < starts; s++)
        {
            int mode = s % 4;
            List<int> cand;

            if (mode == 0)
            {
                cand = byDeg.ToList();
            }
            else if (mode == 1)
            {
                cand = byDeg.Take(Math.Max(1, n / 2)).Concat(byDeg.Skip(Math.Max(1, n / 2)).OrderBy(_ => rnd.Next())).ToList();
            }
            else if (mode == 2)
            {
                cand = byDeg.OrderBy(_ => rnd.Next()).ToList();
            }
            else
            {
                int pivot = byDeg[rnd.Next(Math.Min(n, Math.Max(1, n / 8)))];
                cand = new List<int> { pivot };
                cand.AddRange(_adj[pivot].OrderByDescending(v => _adj[v].Count));
                cand.AddRange(byDeg.Where(v => v != pivot && !_adj[pivot].Contains(v)));
            }

            var clique = GreedyCliqueFromOrder(cand);
            ImproveClique(clique);

            if (clique.Count > bestLocal.Count)
                bestLocal = clique;
        }

        if (bestLocal.Count == 0 && n > 0) bestLocal.Add(byDeg[0]);
        _best = bestLocal;
    }

    private static List<int> GreedyCliqueFromOrder(List<int> order)
    {
        var clique = new List<int>();
        foreach (int v in order)
        {
            bool ok = true;
            for (int i = 0; i < clique.Count; i++)
            {
                if (!_mat[v][clique[i]]) { ok = false; break; }
            }
            if (ok) clique.Add(v);
        }
        return clique;
    }

    private static void ImproveClique(List<int> clique)
    {
        // 1-exchange local search: add any vertex adjacent to all in clique.
        int n = _adj.Length;
        var inClique = new bool[n];
        foreach (int v in clique) inClique[v] = true;

        bool improved;
        do
        {
            improved = false;
            for (int v = 0; v < n; v++)
            {
                if (inClique[v]) continue;
                bool ok = true;
                for (int i = 0; i < clique.Count; i++)
                {
                    if (!_mat[v][clique[i]]) { ok = false; break; }
                }
                if (ok)
                {
                    clique.Add(v);
                    inClique[v] = true;
                    improved = true;
                }
            }

            if (improved) continue;

            // Simple 1-for-2 style attempt: remove one, try to add two.
            for (int remIdx = 0; remIdx < clique.Count && !improved; remIdx++)
            {
                int rem = clique[remIdx];

                var baseClique = new List<int>(clique.Count - 1);
                for (int i = 0; i < clique.Count; i++) if (i != remIdx) baseClique.Add(clique[i]);

                var addable = new List<int>();
                for (int v = 0; v < n; v++)
                {
                    if (inClique[v] && v != rem) continue;
                    bool ok = true;
                    for (int i = 0; i < baseClique.Count; i++)
                    {
                        if (!_mat[v][baseClique[i]]) { ok = false; break; }
                    }
                    if (ok) addable.Add(v);
                }

                for (int i = 0; i < addable.Count && !improved; i++)
                {
                    for (int j = i + 1; j < addable.Count && !improved; j++)
                    {
                        int a = addable[i], b = addable[j];
                        if (a == b || !_mat[a][b]) continue;

                        inClique[rem] = false;
                        clique.RemoveAt(remIdx);

                        if (!inClique[a]) { clique.Add(a); inClique[a] = true; }
                        if (!inClique[b]) { clique.Add(b); inClique[b] = true; }

                        improved = true;
                    }
                }
            }

        } while (improved);
    }
}
