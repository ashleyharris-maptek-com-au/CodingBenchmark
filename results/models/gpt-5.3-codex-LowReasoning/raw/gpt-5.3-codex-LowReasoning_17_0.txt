using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    private struct Box
    {
        public int Id;
        public int W;
        public int H;
        public int D;
        public int Volume;
        public int MinSide;
        public int MaxSide;
    }

    private struct Placement
    {
        public int Id;
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
    }

    private struct Space
    {
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;

        public Space(int x, int y, int z, int w, int h, int d)
        {
            X = x; Y = y; Z = z; W = w; H = h; D = d;
        }

        public int Volume => W * H * D;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = Read();
            }
            return v * sign;
        }
    }

    private static readonly (int a, int b, int c)[] Orientations =
    {
        (0,1,2), (0,2,1), (1,0,2), (1,2,0), (2,0,1), (2,1,0)
    };

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n;
        try { n = fs.NextInt(); }
        catch { return; }

        int containerW = fs.NextInt();
        int containerH = fs.NextInt();
        int containerD = fs.NextInt();

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();
            int minSide = Math.Min(w, Math.Min(h, d));
            int maxSide = Math.Max(w, Math.Max(h, d));
            boxes[i] = new Box
            {
                Id = i,
                W = w,
                H = h,
                D = d,
                Volume = w * h * d,
                MinSide = minSide,
                MaxSide = maxSide
            };
        }

        // Adaptive limits to keep runtime bounded on huge instances.
        int spaceLimit = n <= 2000 ? 12000 : (n <= 10000 ? 8000 : 5000);
        int candidateBoxesPerSpace = n <= 500 ? n : (n <= 4000 ? 200 : 96);

        var order = Enumerable.Range(0, n).ToArray();
        Array.Sort(order, (i, j) => CompareBoxes(boxes[i], boxes[j]));

        var unpacked = new bool[n];
        Array.Fill(unpacked, true);

        var placements = new List<Placement>(Math.Min(n, 8192));
        var spaces = new List<Space>(spaceLimit) { new Space(0, 0, 0, containerW, containerH, containerD) };

        int packedCount = 0;

        while (spaces.Count > 0)
        {
            int si = SelectBestSpaceIndex(spaces);
            Space s = spaces[si];
            spaces[si] = spaces[spaces.Count - 1];
            spaces.RemoveAt(spaces.Count - 1);

            if (s.W <= 0 || s.H <= 0 || s.D <= 0) continue;

            int bestBoxIndex = -1;
            int bestW = 0, bestH = 0, bestD = 0;
            long bestScore = long.MinValue;

            int scanned = 0;
            for (int p = 0; p < order.Length; p++)
            {
                int bi = order[p];
                if (!unpacked[bi]) continue;
                ref Box b = ref boxes[bi];

                if (b.MinSide > s.W && b.MinSide > s.H && b.MinSide > s.D) continue;

                int[] dims = { b.W, b.H, b.D };
                int lastA = -1, lastB = -1, lastC = -1;

                for (int oi = 0; oi < 6; oi++)
                {
                    var o = Orientations[oi];
                    int w = dims[o.a], h = dims[o.b], d = dims[o.c];

                    if (w == lastA && h == lastB && d == lastC) continue;
                    lastA = w; lastB = h; lastC = d;

                    if (w <= s.W && h <= s.H && d <= s.D)
                    {
                        long fill = (long)w * h * d;
                        int remW = s.W - w;
                        int remH = s.H - h;
                        int remD = s.D - d;

                        // Bias toward high fill and tighter fit.
                        long score = fill * 1000000L
                                   - (long)(remW + remH + remD) * 5000L
                                   - (long)Math.Abs(remW - remH)
                                   - (long)Math.Abs(remH - remD)
                                   - (long)Math.Abs(remW - remD);

                        if (score > bestScore)
                        {
                            bestScore = score;
                            bestBoxIndex = bi;
                            bestW = w; bestH = h; bestD = d;
                        }
                    }
                }

                scanned++;
                if (scanned >= candidateBoxesPerSpace && bestBoxIndex >= 0) break;
            }

            if (bestBoxIndex < 0) continue;

            unpacked[bestBoxIndex] = false;
            packedCount++;
            placements.Add(new Placement
            {
                Id = boxes[bestBoxIndex].Id,
                X = s.X,
                Y = s.Y,
                Z = s.Z,
                W = bestW,
                H = bestH,
                D = bestD
            });

            SplitSpace(s, bestW, bestH, bestD, spaces, spaceLimit);
            PruneSpaces(spaces, spaceLimit);
        }

        using var sw = new StreamWriter(Console.OpenStandardOutput()) { NewLine = "\n" };
        sw.WriteLine(packedCount.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < placements.Count; i++)
        {
            var pl = placements[i];
            sw.Write(pl.Id); sw.Write(' ');
            sw.Write(pl.X); sw.Write(' ');
            sw.Write(pl.Y); sw.Write(' ');
            sw.Write(pl.Z); sw.Write('\n');
        }
    }

    private static int CompareBoxes(Box a, Box b)
    {
        int cmp = b.Volume.CompareTo(a.Volume);
        if (cmp != 0) return cmp;
        cmp = b.MaxSide.CompareTo(a.MaxSide);
        if (cmp != 0) return cmp;
        return b.MinSide.CompareTo(a.MinSide);
    }

    private static int SelectBestSpaceIndex(List<Space> spaces)
    {
        int best = 0;
        long bestScore = long.MinValue;
        for (int i = 0; i < spaces.Count; i++)
        {
            var s = spaces[i];
            // Prefer lower y then z then x to build stable layers.
            long score = -((long)s.Y * 1_000_000_000L + (long)s.Z * 1_000_000L + s.X);
            score += (long)s.Volume;
            if (score > bestScore)
            {
                bestScore = score;
                best = i;
            }
        }
        return best;
    }

    private static void SplitSpace(Space s, int w, int h, int d, List<Space> spaces, int spaceLimit)
    {
        // Place at (s.X,s.Y,s.Z), then generate guillotine-like residual spaces.
        AddSpace(spaces, new Space(s.X + w, s.Y, s.Z, s.W - w, h, d), spaceLimit);
        AddSpace(spaces, new Space(s.X, s.Y + h, s.Z, s.W, s.H - h, d), spaceLimit);
        AddSpace(spaces, new Space(s.X, s.Y, s.Z + d, s.W, s.H, s.D - d), spaceLimit);

        AddSpace(spaces, new Space(s.X + w, s.Y + h, s.Z, s.W - w, s.H - h, d), spaceLimit);
        AddSpace(spaces, new Space(s.X + w, s.Y, s.Z + d, s.W - w, h, s.D - d), spaceLimit);
        AddSpace(spaces, new Space(s.X, s.Y + h, s.Z + d, s.W, s.H - h, s.D - d), spaceLimit);
    }

    private static void AddSpace(List<Space> spaces, Space s, int spaceLimit)
    {
        if (s.W <= 0 || s.H <= 0 || s.D <= 0) return;
        spaces.Add(s);
        if (spaces.Count > spaceLimit)
        {
            // Remove one low-value space quickly.
            int worst = 0;
            long worstScore = long.MaxValue;
            for (int i = 0; i < spaces.Count; i++)
            {
                var t = spaces[i];
                long score = (long)t.Volume - ((long)t.Y * 10_000L + t.Z * 100L + t.X);
                if (score < worstScore)
                {
                    worstScore = score;
                    worst = i;
                }
            }
            spaces[worst] = spaces[spaces.Count - 1];
            spaces.RemoveAt(spaces.Count - 1);
        }
    }

    private static void PruneSpaces(List<Space> spaces, int spaceLimit)
    {
        // Lightweight dominance pruning: remove spaces contained by another.
        int m = spaces.Count;
        if (m <= 1) return;

        bool[] removed = new bool[m];
        for (int i = 0; i < m; i++)
        {
            if (removed[i]) continue;
            var a = spaces[i];
            for (int j = i + 1; j < m; j++)
            {
                if (removed[j]) continue;
                var b = spaces[j];

                if (Contains(a, b)) removed[j] = true;
                else if (Contains(b, a)) { removed[i] = true; break; }
            }
        }

        if (Array.IndexOf(removed, true) >= 0)
        {
            int write = 0;
            for (int i = 0; i < m; i++)
            {
                if (!removed[i])
                {
                    spaces[write++] = spaces[i];
                }
            }
            if (write < m) spaces.RemoveRange(write, m - write);
        }

        if (spaces.Count > spaceLimit)
        {
            spaces.Sort((a, b) => b.Volume.CompareTo(a.Volume));
            spaces.RemoveRange(spaceLimit, spaces.Count - spaceLimit);
        }
    }

    private static bool Contains(Space a, Space b)
    {
        return a.X <= b.X && a.Y <= b.Y && a.Z <= b.Z
            && a.X + a.W >= b.X + b.W
            && a.Y + a.H >= b.Y + b.H
            && a.Z + a.D >= b.Z + b.D;
    }
}
