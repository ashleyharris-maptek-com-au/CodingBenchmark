declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Cell = [number, number];
type DroneInput = { id: string; x: number; y: number };

type ParsedInput = {
  probability: number[][];
  drones: DroneInput[];
  maxStepsPerDrone: number;
};

type GridInfo = {
  h: number;
  w: number;
  passable: boolean[];
  value: number[];
  valuableCells: number[];
  neighbors: number[][];
};

function idx(x: number, y: number, w: number): number {
  return x * w + y;
}

function inBounds(x: number, y: number, h: number, w: number): boolean {
  return x >= 0 && x < h && y >= 0 && y < w;
}

function buildGrid(probability: number[][]): GridInfo {
  const h = probability.length;
  const w = h > 0 ? probability[0].length : 0;
  const n = h * w;

  const passable = new Array<boolean>(n).fill(false);
  const value = new Array<number>(n).fill(0);
  const valuableCells: number[] = [];

  for (let x = 0; x < h; x++) {
    for (let y = 0; y < w; y++) {
      const v = probability[x][y];
      const id = idx(x, y, w);
      if (v >= 0) {
        passable[id] = true;
        value[id] = v;
        if (v > 0) valuableCells.push(id);
      }
    }
  }

  const neighbors: number[][] = new Array(n);
  const dirs = [
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
    [0, 0],
  ];

  for (let x = 0; x < h; x++) {
    for (let y = 0; y < w; y++) {
      const id = idx(x, y, w);
      if (!passable[id]) {
        neighbors[id] = [];
        continue;
      }
      const list: number[] = [];
      for (const [dx, dy] of dirs) {
        const nx = x + dx;
        const ny = y + dy;
        if (!inBounds(nx, ny, h, w)) continue;
        const nid = idx(nx, ny, w);
        if (passable[nid]) list.push(nid);
      }
      neighbors[id] = list;
    }
  }

  return { h, w, passable, value, valuableCells, neighbors };
}

function bfsDistances(start: number, grid: GridInfo): Int32Array {
  const n = grid.h * grid.w;
  const dist = new Int32Array(n);
  dist.fill(-1);
  if (!grid.passable[start]) return dist;

  const q = new Int32Array(n);
  let head = 0;
  let tail = 0;

  dist[start] = 0;
  q[tail++] = start;

  while (head < tail) {
    const cur = q[head++];
    const d = dist[cur] + 1;
    const nbrs = grid.neighbors[cur];
    for (let i = 0; i < nbrs.length; i++) {
      const nxt = nbrs[i];
      if (dist[nxt] !== -1) continue;
      dist[nxt] = d;
      q[tail++] = nxt;
    }
  }

  return dist;
}

function reconstructStepToward(
  from: number,
  to: number,
  distFromTarget: Int32Array,
  neighbors: number[]
): number {
  if (from === to) return from;
  const d = distFromTarget[from];
  if (d <= 0) return from;

  let best = from;
  let bestDist = d;
  for (let i = 0; i < neighbors.length; i++) {
    const nb = neighbors[i];
    const nd = distFromTarget[nb];
    if (nd >= 0 && nd < bestDist) {
      bestDist = nd;
      best = nb;
    }
  }
  return best;
}

function solve(inputData: ParsedInput): { paths: Cell[][] } {
  const probability = Array.isArray(inputData.probability) ? inputData.probability : [];
  const drones = Array.isArray(inputData.drones) ? inputData.drones : [];
  const maxSteps = Math.max(0, inputData.maxStepsPerDrone | 0);

  const grid = buildGrid(probability);
  const n = grid.h * grid.w;

  const numDrones = drones.length;
  const positions = new Array<number>(numDrones).fill(0);
  const paths: Cell[][] = new Array(numDrones);

  for (let i = 0; i < numDrones; i++) {
    const d = drones[i];
    const x = d?.x | 0;
    const y = d?.y | 0;
    let p = 0;
    if (inBounds(x, y, grid.h, grid.w)) {
      const id = idx(x, y, grid.w);
      if (grid.passable[id]) p = id;
      else {
        let found = -1;
        const dirs = [
          [0, 0],
          [1, 0],
          [-1, 0],
          [0, 1],
          [0, -1],
        ];
        for (const [dx, dy] of dirs) {
          const nx = x + dx;
          const ny = y + dy;
          if (!inBounds(nx, ny, grid.h, grid.w)) continue;
          const nid = idx(nx, ny, grid.w);
          if (grid.passable[nid]) {
            found = nid;
            break;
          }
        }
        p = found >= 0 ? found : 0;
      }
    }
    if (!grid.passable[p]) {
      for (let j = 0; j < n; j++) {
        if (grid.passable[j]) {
          p = j;
          break;
        }
      }
    }
    positions[i] = p;
    const px = Math.floor(p / grid.w);
    const py = p % grid.w;
    paths[i] = [[px, py]];
  }

  const covered = new Uint8Array(n);
  for (let i = 0; i < numDrones; i++) {
    const p = positions[i];
    if (grid.value[p] > 0) covered[p] = 1;
  }

  const distCache = new Map<number, Int32Array>();
  function getDistFromTarget(target: number): Int32Array {
    const cached = distCache.get(target);
    if (cached) return cached;
    const d = bfsDistances(target, grid);
    distCache.set(target, d);
    return d;
  }

  const maxCandidateTargets = 48;

  for (let step = 0; step < maxSteps; step++) {
    const plannedNext = new Array<number>(numDrones);
    const occupiedNext = new Set<number>();

    const droneOrder = Array.from({ length: numDrones }, (_, i) => i);
    droneOrder.sort((a, b) => {
      const va = grid.value[positions[a]];
      const vb = grid.value[positions[b]];
      return vb - va;
    });

    for (let orderIdx = 0; orderIdx < droneOrder.length; orderIdx++) {
      const i = droneOrder[orderIdx];
      const cur = positions[i];

      let bestTarget = -1;
      let bestScore = -1;

      let candidatesChecked = 0;
      for (let k = 0; k < grid.valuableCells.length; k++) {
        const target = grid.valuableCells[k];
        if (covered[target]) continue;

        const distFromTarget = getDistFromTarget(target);
        const d = distFromTarget[cur];
        if (d < 0) continue;

        const v = grid.value[target];
        const score = v / (d + 1);
        if (score > bestScore) {
          bestScore = score;
          bestTarget = target;
        }

        candidatesChecked++;
        if (candidatesChecked >= maxCandidateTargets && bestTarget >= 0) break;
      }

      let next = cur;
      if (bestTarget >= 0) {
        const distFromTarget = getDistFromTarget(bestTarget);
        next = reconstructStepToward(cur, bestTarget, distFromTarget, grid.neighbors[cur]);
      } else {
        let localBest = cur;
        let localGain = covered[cur] ? 0 : grid.value[cur];
        const nbrs = grid.neighbors[cur];
        for (let ni = 0; ni < nbrs.length; ni++) {
          const nb = nbrs[ni];
          const gain = covered[nb] ? 0 : grid.value[nb];
          if (gain > localGain) {
            localGain = gain;
            localBest = nb;
          }
        }
        next = localBest;
      }

      if (occupiedNext.has(next)) {
        const nbrs = grid.neighbors[cur];
        let fallback = cur;
        let fallbackGain = -1;
        for (let ni = 0; ni < nbrs.length; ni++) {
          const nb = nbrs[ni];
          if (occupiedNext.has(nb)) continue;
          const gain = covered[nb] ? 0 : grid.value[nb];
          if (gain > fallbackGain) {
            fallbackGain = gain;
            fallback = nb;
          }
        }
        next = fallback;
      }

      plannedNext[i] = next;
      occupiedNext.add(next);
    }

    for (let i = 0; i < numDrones; i++) {
      positions[i] = plannedNext[i];
      if (grid.value[positions[i]] > 0) covered[positions[i]] = 1;
      const x = Math.floor(positions[i] / grid.w);
      const y = positions[i] % grid.w;
      paths[i].push([x, y]);
    }
  }

  return { paths };
}

const result = solve(data as ParsedInput);
process.stdout.write(JSON.stringify(result));
