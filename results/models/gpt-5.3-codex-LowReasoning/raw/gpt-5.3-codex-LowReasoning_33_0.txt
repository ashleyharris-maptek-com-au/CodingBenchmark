using System;
using System.IO;
using System.Text;
using System.Linq;
using System.Collections.Generic;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n;
        try { n = fs.NextInt(); }
        catch { return; }

        var flow = new int[n, n];
        var dist = new int[n, n];

        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                flow[i, j] = fs.NextInt();

        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                dist[i, j] = fs.NextInt();

        int[] perm;

        if (n <= 10)
        {
            perm = SolveExactBruteforce(flow, dist, n);
        }
        else
        {
            perm = BuildGreedy(flow, dist, n);
            ImproveByLocalSearch(flow, dist, perm, n);
        }

        long cost = ComputeCost(flow, dist, perm, n);

        var sb = new StringBuilder();
        sb.AppendLine(cost.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.AppendLine();

        Console.Write(sb.ToString());
    }

    private static long ComputeCost(int[,] flow, int[,] dist, int[] p, int n)
    {
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = p[i];
            for (int j = 0; j < n; j++)
                total += (long)flow[i, j] * dist[pi, p[j]];
        }
        return total;
    }

    private static int[] SolveExactBruteforce(int[,] flow, int[,] dist, int n)
    {
        var used = new bool[n];
        var cur = new int[n];
        var best = new int[n];
        long bestCost = long.MaxValue;

        // Precompute a crude lower bound ingredient: minimal outgoing distance from each location.
        var minDistRow = new int[n];
        for (int a = 0; a < n; a++)
        {
            int mn = int.MaxValue;
            for (int b = 0; b < n; b++) mn = Math.Min(mn, dist[a, b]);
            minDistRow[a] = mn;
        }

        // Precompute flow row sums (absolute scale for bound).
        var flowRowSum = new long[n];
        for (int i = 0; i < n; i++)
        {
            long s = 0;
            for (int j = 0; j < n; j++) s += Math.Abs((long)flow[i, j]);
            flowRowSum[i] = s;
        }

        void Dfs(int idx, long partialCost)
        {
            if (idx == n)
            {
                if (partialCost < bestCost)
                {
                    bestCost = partialCost;
                    Array.Copy(cur, best, n);
                }
                return;
            }

            // Simple optimistic bound for remaining facilities.
            long optimistic = partialCost;
            for (int i = idx; i < n; i++)
            {
                int md = int.MaxValue;
                for (int loc = 0; loc < n; loc++)
                    if (!used[loc]) md = Math.Min(md, minDistRow[loc]);
                optimistic += flowRowSum[i] * md;
                if (optimistic >= bestCost) return;
            }

            for (int loc = 0; loc < n; loc++)
            {
                if (used[loc]) continue;

                long add = 0;
                for (int j = 0; j < idx; j++)
                {
                    int pj = cur[j];
                    add += (long)flow[idx, j] * dist[loc, pj];
                    add += (long)flow[j, idx] * dist[pj, loc];
                }
                add += (long)flow[idx, idx] * dist[loc, loc];

                long next = partialCost + add;
                if (next >= bestCost) continue;

                used[loc] = true;
                cur[idx] = loc;
                Dfs(idx + 1, next);
                used[loc] = false;
            }
        }

        Dfs(0, 0);
        return best;
    }

    private static int[] BuildGreedy(int[,] flow, int[,] dist, int n)
    {
        var facilityScore = new long[n];
        var locationScore = new long[n];

        for (int i = 0; i < n; i++)
        {
            long s = 0;
            for (int j = 0; j < n; j++) s += Math.Abs((long)flow[i, j]) + Math.Abs((long)flow[j, i]);
            facilityScore[i] = s;
        }

        for (int a = 0; a < n; a++)
        {
            long s = 0;
            for (int b = 0; b < n; b++) s += Math.Abs((long)dist[a, b]) + Math.Abs((long)dist[b, a]);
            locationScore[a] = s;
        }

        int[] facilities = Enumerable.Range(0, n).ToArray();
        int[] locations = Enumerable.Range(0, n).ToArray();

        // High-flow facilities to "central" (small score) locations.
        Array.Sort(facilities, (x, y) => facilityScore[y].CompareTo(facilityScore[x]));
        Array.Sort(locations, (x, y) => locationScore[x].CompareTo(locationScore[y]));

        var p = new int[n];
        for (int k = 0; k < n; k++) p[facilities[k]] = locations[k];
        return p;
    }

    private static void ImproveByLocalSearch(int[,] flow, int[,] dist, int[] p, int n)
    {
        // For large n, sample-based pair swaps with exact delta computation over all k (O(n) each).
        // Iteration budget scales roughly O(n^2) for small/medium and capped for large instances.
        int maxOuter = n <= 200 ? 12 : (n <= 800 ? 8 : 5);
        int candidatesPerI = n <= 200 ? Math.Min(n - 1, 80) : (n <= 800 ? 40 : 20);

        int[] inv = new int[n];
        for (int i = 0; i < n; i++) inv[p[i]] = i;

        var rng = new Random(1234567);
        var allIdx = Enumerable.Range(0, n).ToArray();

        bool improvedAny = true;
        int pass = 0;

        while (improvedAny && pass < maxOuter)
        {
            improvedAny = false;
            pass++;

            // Shuffle facility order each pass.
            for (int i = n - 1; i > 0; i--)
            {
                int j = rng.Next(i + 1);
                int t = allIdx[i]; allIdx[i] = allIdx[j]; allIdx[j] = t;
            }

            foreach (int i in allIdx)
            {
                long bestDelta = 0;
                int bestJ = -1;

                for (int t = 0; t < candidatesPerI; t++)
                {
                    int j = rng.Next(n);
                    if (j == i) continue;

                    long delta = SwapDelta(flow, dist, p, i, j, n);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestJ = j;
                    }
                }

                if (bestJ >= 0)
                {
                    int li = p[i], lj = p[bestJ];
                    p[i] = lj;
                    p[bestJ] = li;
                    inv[li] = bestJ;
                    inv[lj] = i;
                    improvedAny = true;
                }
            }
        }
    }

    private static long SwapDelta(int[,] flow, int[,] dist, int[] p, int r, int s, int n)
    {
        if (r == s) return 0;

        int pr = p[r], ps = p[s];
        long delta = 0;

        // Terms involving r and s with all k.
        for (int k = 0; k < n; k++)
        {
            if (k == r || k == s) continue;
            int pk = p[k];

            delta += (long)flow[r, k] * (dist[ps, pk] - dist[pr, pk]);
            delta += (long)flow[s, k] * (dist[pr, pk] - dist[ps, pk]);
            delta += (long)flow[k, r] * (dist[pk, ps] - dist[pk, pr]);
            delta += (long)flow[k, s] * (dist[pk, pr] - dist[pk, ps]);
        }

        // Self and pair cross terms.
        delta += (long)flow[r, r] * (dist[ps, ps] - dist[pr, pr]);
        delta += (long)flow[s, s] * (dist[pr, pr] - dist[ps, ps]);
        delta += (long)flow[r, s] * (dist[ps, pr] - dist[pr, ps]);
        delta += (long)flow[s, r] * (dist[pr, ps] - dist[ps, pr]);

        return delta;
    }
}
