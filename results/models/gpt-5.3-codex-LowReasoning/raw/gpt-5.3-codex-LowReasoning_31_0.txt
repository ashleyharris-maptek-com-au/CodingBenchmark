use std::cmp::Ordering;
use std::collections::{HashMap, HashSet};
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Item {
    idx: usize,
    val: i64,
}

fn parse_input() -> Option<(usize, i64, Vec<i64>)> {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return None;
    }
    let mut it = input.split_whitespace();
    let n = it.next()?.parse::<usize>().ok()?;
    let target = it.next()?.parse::<i64>().ok()?;

    let mut nums = Vec::with_capacity(n);
    for _ in 0..n {
        let x = it.next()?.parse::<i64>().ok()?;
        nums.push(x);
    }
    Some((n, target, nums))
}

fn bounded_exact_subset(items: &[Item], target: i64, max_states: usize) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }

    let mut states: HashMap<i64, Option<(i64, usize)>> = HashMap::with_capacity(1024);
    states.insert(0, None);

    for item in items {
        let snapshot: Vec<(i64, Option<(i64, usize)>)> = states
            .iter()
            .map(|(s, p)| (*s, *p))
            .collect();

        for (sum, _) in &snapshot {
            let ns = *sum + item.val;
            if let std::collections::hash_map::Entry::Vacant(e) = states.entry(ns) {
                e.insert(Some((*sum, item.idx)));
                if ns == target {
                    let mut path = Vec::new();
                    let mut cur = target;
                    while cur != 0 {
                        if let Some(Some((prev, used_idx))) = states.get(&cur) {
                            path.push(*used_idx);
                            cur = *prev;
                        } else {
                            break;
                        }
                    }
                    path.sort_unstable();
                    return Some(path);
                }
            }
        }

        if states.len() > max_states {
            return None;
        }
    }

    if !states.contains_key(&target) {
        return None;
    }

    let mut path = Vec::new();
    let mut cur = target;
    while cur != 0 {
        if let Some(Some((prev, used_idx))) = states.get(&cur) {
            path.push(*used_idx);
            cur = *prev;
        } else {
            break;
        }
    }
    path.sort_unstable();
    Some(path)
}

fn bitset_subset_nonneg(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let t = usize::try_from(target).ok()?;

    let mut parent: Vec<Option<(usize, usize)>> = vec![None; t + 1];
    let mut seen = vec![false; t + 1];
    seen[0] = true;

    for (idx, &x) in nums.iter().enumerate() {
        if x < 0 {
            return None;
        }
        let v = usize::try_from(x).ok()?;
        if v > t {
            continue;
        }
        for s in (v..=t).rev() {
            if !seen[s] && seen[s - v] {
                seen[s] = true;
                parent[s] = Some((s - v, idx));
            }
        }
    }

    if !seen[t] {
        return None;
    }

    let mut ans = Vec::new();
    let mut cur = t;
    while cur != 0 {
        if let Some((prev, idx)) = parent[cur] {
            ans.push(idx);
            cur = prev;
        } else {
            break;
        }
    }
    ans.sort_unstable();
    Some(ans)
}

fn greedy_approx(items: &[Item], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }

    let mut positives: Vec<Item> = items.iter().copied().filter(|x| x.val > 0).collect();
    let mut negatives: Vec<Item> = items.iter().copied().filter(|x| x.val < 0).collect();

    if target > 0 {
        positives.sort_by(|a, b| b.val.cmp(&a.val));
        let mut sum = 0i64;
        let mut chosen = Vec::new();
        for it in positives {
            if sum + it.val <= target {
                sum += it.val;
                chosen.push(it.idx);
                if sum == target {
                    chosen.sort_unstable();
                    return Some(chosen);
                }
            }
        }
    } else {
        negatives.sort_by(|a, b| a.val.cmp(&b.val));
        let mut sum = 0i64;
        let mut chosen = Vec::new();
        for it in negatives {
            if sum + it.val >= target {
                sum += it.val;
                chosen.push(it.idx);
                if sum == target {
                    chosen.sort_unstable();
                    return Some(chosen);
                }
            }
        }
    }

    None
}

fn solve(nums: Vec<i64>, target: i64) -> Option<Vec<usize>> {
    let n = nums.len();

    if target == 0 {
        return Some(Vec::new());
    }

    if let Some((idx, _)) = nums.iter().enumerate().find(|(_, v)| **v == target) {
        return Some(vec![idx]);
    }

    let items: Vec<Item> = nums
        .iter()
        .enumerate()
        .map(|(i, &v)| Item { idx: i, val: v })
        .collect();

    let all_nonneg = nums.iter().all(|&x| x >= 0);

    if all_nonneg {
        if let Ok(tu) = usize::try_from(target) {
            if tu <= 5_000_000 {
                if let Some(ans) = bitset_subset_nonneg(&nums, target) {
                    return Some(ans);
                }
            }
        }
    }

    let max_states = if n <= 2_000 {
        2_000_000
    } else if n <= 50_000 {
        1_000_000
    } else {
        200_000
    };

    let mut by_abs = items.clone();
    by_abs.sort_by(|a, b| {
        let aa = a.val.abs();
        let bb = b.val.abs();
        bb.cmp(&aa)
    });

    let head_len = by_abs.len().min(50_000);
    if let Some(ans) = bounded_exact_subset(&by_abs[..head_len], target, max_states) {
        return Some(ans);
    }

    let mut by_sign_proximity = items.clone();
    by_sign_proximity.sort_by(|a, b| {
        let da = (target - a.val).abs();
        let db = (target - b.val).abs();
        match da.cmp(&db) {
            Ordering::Equal => b.val.abs().cmp(&a.val.abs()),
            other => other,
        }
    });

    let second_len = by_sign_proximity.len().min(30_000);
    if let Some(ans) = bounded_exact_subset(&by_sign_proximity[..second_len], target, max_states / 2) {
        return Some(ans);
    }

    greedy_approx(&items, target)
}

fn main() {
    let parsed = parse_input();
    let (n, target, nums) = match parsed {
        Some(v) => v,
        None => {
            println!("NO");
            return;
        }
    };

    if n != nums.len() {
        println!("NO");
        return;
    }

    match solve(nums, target) {
        Some(indices) => {
            let mut uniq = HashSet::with_capacity(indices.len());
            for &i in &indices {
                if i >= n || !uniq.insert(i) {
                    println!("NO");
                    return;
                }
            }
            println!("YES");
            if indices.is_empty() {
                println!();
            } else {
                let line = indices
                    .iter()
                    .map(|x| x.to_string())
                    .collect::<Vec<String>>()
                    .join(" ");
                println!("{}", line);
            }
        }
        None => {
            println!("NO");
        }
    }
}
