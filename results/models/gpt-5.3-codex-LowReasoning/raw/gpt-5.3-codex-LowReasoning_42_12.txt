OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_6 = OpConstant %uint 6
%arr_gl_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_gl_in = OpTypePointer Input %arr_gl_in
%gl_in = OpVariable %ptr_in_arr_gl_in Input
%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_out_gl_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_out Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_float = OpTypePointer Input %float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015

%v3_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%main = OpFunction %void None %func
%entry = OpLabel

; Emit filled main triangle (indices 0,2,4), dimmed color
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%c0_dim = OpVectorTimesScalar %v3float %c0 %float_0_3
%go0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %go0 %p0
OpStore %outColor %c0_dim
OpEmitVertex

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2 = OpLoad %v3float %c2_ptr
%c2_dim = OpVectorTimesScalar %v3float %c2 %float_0_3
OpStore %go0 %p2
OpStore %outColor %c2_dim
OpEmitVertex

%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr
%c4_ptr = OpAccessChain %ptr_in_v3 %inColor %int_4
%c4 = OpLoad %v3float %c4_ptr
%c4_dim = OpVectorTimesScalar %v3float %c4 %float_0_3
OpStore %go0 %p4
OpStore %outColor %c4_dim
OpEmitVertex
OpEndPrimitive

; Edge 0->2 thick quad
%x0_ptr = OpAccessChain %ptr_in_float %gl_in %int_0 %int_0 %int_0
%y0_ptr = OpAccessChain %ptr_in_float %gl_in %int_0 %int_0 %int_1
%x2_ptr = OpAccessChain %ptr_in_float %gl_in %int_2 %int_0 %int_0
%y2_ptr = OpAccessChain %ptr_in_float %gl_in %int_2 %int_0 %int_1
%x0 = OpLoad %float %x0_ptr
%y0 = OpLoad %float %y0_ptr
%x2 = OpLoad %float %x2_ptr
%y2 = OpLoad %float %y2_ptr
%dx02 = OpFSub %float %x2 %x0
%dy02 = OpFSub %float %y2 %y0
%perp02 = OpCompositeConstruct %v2float %dy02 %dx02
%len02 = OpExtInst %float %1 Length %perp02
%invLen02 = OpFDiv %float %float_1 %len02
%nx02 = OpFMul %float %dy02 %invLen02
%ny02a = OpFMul %float %dx02 %invLen02
%ny02 = OpFNegate %float %ny02a
%ox02 = OpFMul %float %nx02 %float_0_015
%oy02 = OpFMul %float %ny02 %float_0_015

%x0p = OpFAdd %float %x0 %ox02
%y0p = OpFAdd %float %y0 %oy02
%x0m = OpFSub %float %x0 %ox02
%y0m = OpFSub %float %y0 %oy02
%x2p = OpFAdd %float %x2 %ox02
%y2p = OpFAdd %float %y2 %oy02
%x2m = OpFSub %float %x2 %ox02
%y2m = OpFSub %float %y2 %oy02

%z0_ptr = OpAccessChain %ptr_in_float %gl_in %int_0 %int_0 %int_2
%w0_ptr = OpAccessChain %ptr_in_float %gl_in %int_0 %int_0 %int_3
%z2_ptr = OpAccessChain %ptr_in_float %gl_in %int_2 %int_0 %int_2
%w2_ptr = OpAccessChain %ptr_in_float %gl_in %int_2 %int_0 %int_3
%z0 = OpLoad %float %z0_ptr
%w0 = OpLoad %float %w0_ptr
%z2 = OpLoad %float %z2_ptr
%w2 = OpLoad %float %w2_ptr

%v0p = OpCompositeConstruct %v4float %x0p %y0p %z0 %w0
%v0m = OpCompositeConstruct %v4float %x0m %y0m %z0 %w0
%v2p = OpCompositeConstruct %v4float %x2p %y2p %z2 %w2
%v2m = OpCompositeConstruct %v4float %x2m %y2m %z2 %w2

OpStore %go0 %v0p
OpStore %outColor %v3_white
OpEmitVertex
OpStore %go0 %v0m
OpStore %outColor %v3_white
OpEmitVertex
OpStore %go0 %v2p
OpStore %outColor %v3_white
OpEmitVertex
OpStore %go0 %v2m
OpStore %outColor %v3_white
OpEmitVertex
OpEndPrimitive

; Edge 2->4 thick quad
%x4_ptr = OpAccessChain %ptr_in_float %gl_in %int_4 %int_0 %int_0
%y4_ptr = OpAccessChain %ptr_in_float %gl_in %int_4 %int_0 %int_1
%x4 = OpLoad %float %x4_ptr
%y4 = OpLoad %float %y4_ptr
%dx24 = OpFSub %float %x4 %x2
%dy24 = OpFSub %float %y4 %y2
%perp24 = OpCompositeConstruct %v2float %dy24 %dx24
%len24 = OpExtInst %float %1 Length %perp24
%invLen24 = OpFDiv %float %float_1 %len24
%nx24 = OpFMul %float %dy24 %invLen24
%ny24a = OpFMul %float %dx24 %invLen24
%ny24 = OpFNegate %float %ny24a
%ox24 = OpFMul %float %nx24 %float_0_015
%oy24 = OpFMul %float %ny24 %float_0_015

%x2p = OpFAdd %float %x2 %ox24
%y2p = OpFAdd %float %y2 %oy24
%x2m = OpFSub %float %x2 %ox24
%y2m = OpFSub %float %y2 %oy24
%x4p = OpFAdd %float %x4 %ox24
%y4p = OpFAdd %float %y4 %oy24
%x4m = OpFSub %float %x4 %ox24
%y4m = OpFSub %float %y4 %oy24

%z4_ptr = OpAccessChain %ptr_in_float %gl_in %int_4 %int_0 %int_2
%w4_ptr = OpAccessChain %ptr_in_float %gl_in %int_4 %int_0 %int_3
%z4 = OpLoad %float %z4_ptr
%w4 = OpLoad %float %w4_ptr

%v2p_b = OpCompositeConstruct %v4float %x2p %y2p %z2 %w2
%v2m_b = OpCompositeConstruct %v4float %x2m %y2m %z2 %w2
%v4p = OpCompositeConstruct %v4float %x4p %y4p %z4 %w4
%v4m = OpCompositeConstruct %v4float %x4m %y4m %z4 %w4

OpStore %go0 %v2p_b
OpStore %outColor %v3_white
OpEmitVertex
OpStore %go0 %v2m_b
OpStore %outColor %v3_white
OpEmitVertex
OpStore %go0 %v4p
OpStore %outColor %v3_white
OpEmitVertex
OpStore %go0 %v4m
OpStore %outColor %v3_white
OpEmitVertex
OpEndPrimitive

; Edge 4->0 thick quad
%dx40 = OpFSub %float %x0 %x4
%dy40 = OpFSub %float %y0 %y4
%perp40 = OpCompositeConstruct %v2float %dy40 %dx40
%len40 = OpExtInst %float %1 Length %perp40
%invLen40 = OpFDiv %float %float_1 %len40
%nx40 = OpFMul %float %dy40 %invLen40
%ny40a = OpFMul %float %dx40 %invLen40
%ny40 = OpFNegate %float %ny40a
%ox40 = OpFMul %float %nx40 %float_0_015
%oy40 = OpFMul %float %ny40 %float_0_015

%x4p2 = OpFAdd %float %x4 %ox40
%y4p2 = OpFAdd %float %y4 %oy40
%x4m2 = OpFSub %float %x4 %ox40
%y4m2 = OpFSub %float %y4 %oy40
%x0p2 = OpFAdd %float %x0 %ox40
%y0p2 = OpFAdd %float %y0 %oy40
%x0m2 = OpFSub %float %x0 %ox40
%y0m2 = OpFSub %float %y0 %oy40

%v4p2 = OpCompositeConstruct %v4float %x4p2 %y4p2 %z4 %w4
%v4m2 = OpCompositeConstruct %v4float %x4m2 %y4m2 %z4 %w4
%v0p2 = OpCompositeConstruct %v4float %x0p2 %y0p2 %z0 %w0
%v0m2 = OpCompositeConstruct %v4float %x0m2 %y0m2 %z0 %w0

OpStore %go0 %v4p2
OpStore %outColor %v3_white
OpEmitVertex
OpStore %go0 %v4m2
OpStore %outColor %v3_white
OpEmitVertex
OpStore %go0 %v0p2
OpStore %outColor %v3_white
OpEmitVertex
OpStore %go0 %v0m2
OpStore %outColor %v3_white
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
