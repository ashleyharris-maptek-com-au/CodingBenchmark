#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <unordered_map>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int u;
    int v;
};

struct DenseStoerWagnerResult {
    long long cutValue;
    vector<char> side;
};

static DenseStoerWagnerResult stoerWagnerDense(int n, const vector<Edge>& edges) {
    vector<vector<long long>> w(n, vector<long long>(n, 0));
    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        w[e.u][e.v] += 1;
        w[e.v][e.u] += 1;
    }

    vector<int> verts(n);
    iota(verts.begin(), verts.end(), 0);

    vector<vector<int>> members(n);
    for (int i = 0; i < n; ++i) members[i].push_back(i);

    long long best = numeric_limits<long long>::max();
    vector<char> bestSide(n, 0);

    while (verts.size() > 1) {
        int k = static_cast<int>(verts.size());
        vector<long long> dist(n, 0);
        vector<char> inA(n, 0);

        int prev = -1;
        int last = -1;

        for (int i = 0; i < k; ++i) {
            int sel = -1;
            long long mx = -1;
            for (int v : verts) {
                if (!inA[v] && dist[v] > mx) {
                    mx = dist[v];
                    sel = v;
                }
            }
            inA[sel] = 1;
            prev = last;
            last = sel;

            if (i == k - 1) {
                long long cut = dist[last];
                if (cut < best) {
                    best = cut;
                    fill(bestSide.begin(), bestSide.end(), 0);
                    for (int orig : members[last]) bestSide[orig] = 1;
                }

                if (prev != -1) {
                    for (int v : verts) {
                        if (v == prev || v == last) continue;
                        w[prev][v] += w[last][v];
                        w[v][prev] = w[prev][v];
                    }
                    auto& dst = members[prev];
                    auto& src = members[last];
                    dst.insert(dst.end(), src.begin(), src.end());
                    src.clear();

                    verts.erase(remove(verts.begin(), verts.end(), last), verts.end());
                }
            } else {
                for (int v : verts) {
                    if (!inA[v]) dist[v] += w[sel][v];
                }
            }
        }
    }

    return {best, bestSide};
}

struct SparseStoerWagnerResult {
    long long cutValue;
    vector<char> side;
};

static SparseStoerWagnerResult stoerWagnerSparse(int n, const vector<Edge>& edges) {
    vector<unordered_map<int, long long>> g(n);
    g.reserve(n);

    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        ++g[e.u][e.v];
        ++g[e.v][e.u];
    }

    vector<int> verts(n);
    iota(verts.begin(), verts.end(), 0);

    vector<vector<int>> members(n);
    for (int i = 0; i < n; ++i) members[i].push_back(i);

    long long best = numeric_limits<long long>::max();
    vector<char> bestSide(n, 0);

    vector<char> active(n, 1);

    while (verts.size() > 1) {
        vector<long long> key(n, 0);
        vector<char> inA(n, 0);

        priority_queue<pair<long long, int>> pq;
        int start = verts[0];
        pq.push({0, start});

        int prev = -1;
        int last = -1;

        for (size_t it = 0; it < verts.size(); ++it) {
            int v = -1;
            while (!pq.empty()) {
                auto [k, x] = pq.top();
                pq.pop();
                if (active[x] && !inA[x] && k == key[x]) {
                    v = x;
                    break;
                }
            }
            if (v == -1) {
                for (int x : verts) {
                    if (!inA[x]) {
                        v = x;
                        break;
                    }
                }
            }

            inA[v] = 1;
            prev = last;
            last = v;

            if (it + 1 == verts.size()) {
                long long cut = key[last];
                if (cut < best) {
                    best = cut;
                    fill(bestSide.begin(), bestSide.end(), 0);
                    for (int orig : members[last]) bestSide[orig] = 1;
                }

                if (prev != -1) {
                    auto& gp = g[prev];
                    auto& gl = g[last];

                    for (const auto& kv : gl) {
                        int nei = kv.first;
                        long long w = kv.second;
                        if (nei == prev || !active[nei]) continue;
                        gp[nei] += w;
                        g[nei][prev] += w;
                        g[nei].erase(last);
                    }

                    gp.erase(last);
                    gl.clear();

                    auto& dst = members[prev];
                    auto& src = members[last];
                    dst.insert(dst.end(), src.begin(), src.end());
                    src.clear();

                    active[last] = 0;
                    verts.erase(remove(verts.begin(), verts.end(), last), verts.end());
                }
            } else {
                for (const auto& kv : g[v]) {
                    int nei = kv.first;
                    long long w = kv.second;
                    if (!active[nei] || inA[nei]) continue;
                    key[nei] += w;
                    pq.push({key[nei], nei});
                }
            }
        }
    }

    return {best, bestSide};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    long long mll;
    if (!(cin >> n >> mll)) return 0;

    if (n <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    if (mll < 0) return 0;
    size_t m = static_cast<size_t>(mll);

    vector<Edge> edges;
    edges.reserve(m);

    for (size_t i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        edges.push_back({u, v});
    }

    const long long nn = static_cast<long long>(n);
    const long long thresholdDense = 14000000LL; // tune for ~memory/time balance

    vector<char> side;
    if (nn * nn <= thresholdDense && edges.size() > static_cast<size_t>(nn)) {
        auto res = stoerWagnerDense(n, edges);
        side = move(res.side);
    } else {
        auto res = stoerWagnerSparse(n, edges);
        side = move(res.side);
    }

    vector<Edge> cut;
    cut.reserve(edges.size() / 8 + 1);
    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        if (side[e.u] != side[e.v]) cut.push_back(e);
    }

    cout << cut.size() << '\n';
    for (const auto& e : cut) {
        cout << e.u << ' ' << e.v << '\n';
    }

    return 0;
}
