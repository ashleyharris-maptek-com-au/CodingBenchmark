struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float2 uvScaled = input.uv * float2(8.0, 16.0);

    float row = floor(uvScaled.y);
    float rowOffset = fmod(row, 2.0) * 0.5;

    float2 brickUV;
    brickUV.x = frac(uvScaled.x + rowOffset);
    brickUV.y = frac(uvScaled.y);

    float mortarX = step(brickUV.x, 0.05) + step(0.95, brickUV.x);
    float mortarY = step(brickUV.y, 0.1) + step(0.9, brickUV.y);
    float mortarMask = saturate(mortarX + mortarY);

    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 baseColor = lerp(brickColor, mortarColor, mortarMask);

    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lighting = 0.4 + 0.6 * diffuse;

    float3 finalColor = baseColor * lighting;
    return float4(finalColor, 1.0);
}
