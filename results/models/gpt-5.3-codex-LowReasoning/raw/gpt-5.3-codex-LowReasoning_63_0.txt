const fs = require('fs');

interface Robot {
  id: number;
  x: number;
  y: number;
}

interface Job {
  id: number;
  x: number;
  y: number;
  priority: number;
}

interface InputData {
  grid: string[];
  robots: Robot[];
  jobs: Job[];
}

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const grid = data.grid;
const robots = data.robots.slice();
const jobs = data.jobs.slice();

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;
const nCells = h * w;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < w && y >= 0 && y < h;
}

function cellId(x: number, y: number): number {
  return y * w + x;
}

const blocked = new Uint8Array(nCells);
for (let y = 0; y < h; y++) {
  const row = grid[y];
  for (let x = 0; x < w; x++) {
    if (row[x] === '#') blocked[cellId(x, y)] = 1;
  }
}

const dx = [1, -1, 0, 0];
const dy = [0, 0, 1, -1];

const distCache = new Map<number, Int32Array>();

function bfsFrom(srcCell: number): Int32Array {
  const cached = distCache.get(srcCell);
  if (cached) return cached;

  const dist = new Int32Array(nCells);
  dist.fill(-1);

  if (srcCell < 0 || srcCell >= nCells || blocked[srcCell]) {
    distCache.set(srcCell, dist);
    return dist;
  }

  const q = new Int32Array(nCells);
  let head = 0;
  let tail = 0;

  dist[srcCell] = 0;
  q[tail++] = srcCell;

  while (head < tail) {
    const cur = q[head++];
    const cx = cur % w;
    const cy = (cur / w) | 0;
    const nd = dist[cur] + 1;

    for (let k = 0; k < 4; k++) {
      const nx = cx + dx[k];
      const ny = cy + dy[k];
      if (!inBounds(nx, ny)) continue;
      const nid = cellId(nx, ny);
      if (blocked[nid] || dist[nid] !== -1) continue;
      dist[nid] = nd;
      q[tail++] = nid;
    }
  }

  distCache.set(srcCell, dist);
  return dist;
}

const maxRobotId = robots.reduce((m, r) => Math.max(m, r.id), -1);
const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

// Robot mutable state for greedy routing.
const rState = robots.map((r) => ({
  id: r.id,
  x: r.x,
  y: r.y,
  cell: cellId(r.x, r.y),
}));

// Deterministic order: higher priority first, then lower job id.
jobs.sort((a, b) => {
  if (a.priority !== b.priority) return b.priority - a.priority;
  return a.id - b.id;
});

for (const job of jobs) {
  const jCell = cellId(job.x, job.y);

  let bestRobotIdx = -1;
  let bestDist = Number.POSITIVE_INFINITY;

  for (let i = 0; i < rState.length; i++) {
    const r = rState[i];
    const dArr = bfsFrom(r.cell);
    const d = dArr[jCell];
    const val = d >= 0 ? d : Number.POSITIVE_INFINITY;

    if (
      val < bestDist ||
      (val === bestDist && bestRobotIdx >= 0 && r.id < rState[bestRobotIdx].id)
    ) {
      bestDist = val;
      bestRobotIdx = i;
    }
  }

  if (bestRobotIdx < 0) {
    // Fallback (should not happen with at least one robot): choose smallest robot id.
    let minIdx = 0;
    for (let i = 1; i < rState.length; i++) {
      if (rState[i].id < rState[minIdx].id) minIdx = i;
    }
    bestRobotIdx = minIdx;
  }

  const chosen = rState[bestRobotIdx];
  assignments[chosen.id].push(job.id);
  chosen.x = job.x;
  chosen.y = job.y;
  chosen.cell = jCell;
}

const output = { assignments };
process.stdout.write(JSON.stringify(output));
