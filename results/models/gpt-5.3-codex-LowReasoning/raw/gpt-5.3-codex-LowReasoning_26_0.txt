use std::io::{self, Read, Write, BufWriter};

#[derive(Clone, Copy, Debug, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

#[derive(Clone, Copy, Debug, Default)]
struct Centroid {
    x: f32,
    y: f32,
    z: f32,
}

impl Centroid {
    #[inline]
    fn dist2(&self, p: &Point) -> f32 {
        let dx = self.x - p.x;
        let dy = self.y - p.y;
        let dz = self.z - p.z;
        dx * dx + dy * dy + dz * dz
    }

    #[inline]
    fn from_point(p: &Point) -> Self {
        Self { x: p.x, y: p.y, z: p.z }
    }
}

#[derive(Clone, Copy)]
enum Strategy {
    InMemory,
    MiniBatch,
    Streaming,
}

fn read_header() -> io::Result<(u64, usize)> {
    let mut stdin = io::stdin().lock();
    let mut header_bytes = Vec::with_capacity(64);
    let mut buf = [0u8; 1];

    while stdin.read(&mut buf)? == 1 {
        let b = buf[0];
        if b.is_ascii_whitespace() {
            if !header_bytes.is_empty() {
                break;
            }
        } else {
            header_bytes.push(b);
        }
    }

    if header_bytes.is_empty() {
        return Err(io::Error::new(io::ErrorKind::InvalidData, "missing num_points"));
    }
    let num_points: u64 = std::str::from_utf8(&header_bytes)
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid UTF-8 in num_points"))?
        .parse()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid num_points"))?;

    header_bytes.clear();
    while stdin.read(&mut buf)? == 1 {
        let b = buf[0];
        if b.is_ascii_whitespace() {
            if !header_bytes.is_empty() {
                break;
            }
        } else {
            header_bytes.push(b);
        }
    }

    if header_bytes.is_empty() {
        return Err(io::Error::new(io::ErrorKind::InvalidData, "missing num_clusters"));
    }
    let num_clusters: usize = std::str::from_utf8(&header_bytes)
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid UTF-8 in num_clusters"))?
        .parse()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid num_clusters"))?;

    if num_clusters == 0 {
        return Err(io::Error::new(io::ErrorKind::InvalidInput, "num_clusters must be > 0"));
    }

    Ok((num_points, num_clusters))
}

fn choose_strategy(n: u64, k: usize) -> Strategy {
    if n <= 10_000_000 {
        Strategy::InMemory
    } else if n <= 300_000_000 && k <= 10_000 {
        Strategy::MiniBatch
    } else {
        Strategy::Streaming
    }
}

fn next_u64(state: &mut u64) -> u64 {
    let mut x = *state;
    x ^= x >> 12;
    x ^= x << 25;
    x ^= x >> 27;
    *state = x;
    x.wrapping_mul(0x2545F4914F6CDD1D)
}

fn nearest_cluster(centroids: &[Centroid], p: &Point) -> usize {
    let mut best = 0usize;
    let mut best_d = centroids[0].dist2(p);
    for i in 1..centroids.len() {
        let d = centroids[i].dist2(p);
        if d < best_d {
            best_d = d;
            best = i;
        }
    }
    best
}

fn read_all_points(n: u64) -> io::Result<Vec<Point>> {
    let n_usize = usize::try_from(n)
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "num_points too large for this platform"))?;

    let mut stdin = io::stdin().lock();
    let mut raw = vec![0u8; n_usize.saturating_mul(12)];
    stdin.read_exact(&mut raw)?;

    let mut pts = Vec::with_capacity(n_usize);
    for c in raw.chunks_exact(12) {
        let x = f32::from_le_bytes([c[0], c[1], c[2], c[3]]);
        let y = f32::from_le_bytes([c[4], c[5], c[6], c[7]]);
        let z = f32::from_le_bytes([c[8], c[9], c[10], c[11]]);
        pts.push(Point { x, y, z });
    }
    Ok(pts)
}

fn init_centroids_from_points(points: &[Point], k: usize) -> Vec<Centroid> {
    let n = points.len();
    let mut centroids = Vec::with_capacity(k);
    if n == 0 {
        centroids.resize(k, Centroid::default());
        return centroids;
    }

    let step = (n / k.max(1)).max(1);
    let mut idx = 0usize;
    for _ in 0..k {
        centroids.push(Centroid::from_point(&points[idx.min(n - 1)]));
        idx = idx.saturating_add(step);
    }
    centroids
}

fn write_assignments(assignments: &[u32]) -> io::Result<()> {
    let mut out = BufWriter::new(io::stdout().lock());
    for &a in assignments {
        out.write_all(&a.to_le_bytes())?;
    }
    out.flush()
}

fn run_in_memory(n: u64, k: usize) -> io::Result<()> {
    let points = read_all_points(n)?;
    let n_usize = points.len();

    if n_usize == 0 {
        return Ok(());
    }

    let mut centroids = init_centroids_from_points(&points, k);
    let mut assignments = vec![0u32; n_usize];
    let mut prev_changed = n_usize;

    let max_iters = 20usize;
    for _ in 0..max_iters {
        let mut sums = vec![Centroid::default(); k];
        let mut counts = vec![0u32; k];
        let mut changed = 0usize;

        for (i, p) in points.iter().enumerate() {
            let c = nearest_cluster(&centroids, p);
            if assignments[i] != c as u32 {
                changed += 1;
                assignments[i] = c as u32;
            }
            sums[c].x += p.x;
            sums[c].y += p.y;
            sums[c].z += p.z;
            counts[c] += 1;
        }

        for i in 0..k {
            if counts[i] > 0 {
                let inv = 1.0f32 / counts[i] as f32;
                centroids[i].x = sums[i].x * inv;
                centroids[i].y = sums[i].y * inv;
                centroids[i].z = sums[i].z * inv;
            } else {
                let repl = points[(i * 9973) % n_usize];
                centroids[i] = Centroid::from_point(&repl);
            }
        }

        if changed == 0 || (changed >= prev_changed && changed * 1000 / n_usize < 1) {
            break;
        }
        prev_changed = changed;
    }

    write_assignments(&assignments)
}

fn read_points_chunk(stdin: &mut dyn Read, buf: &mut [u8], max_points: usize) -> io::Result<Vec<Point>> {
    let want = max_points.saturating_mul(12).min(buf.len());
    let mut read_total = 0usize;
    while read_total < want {
        let n = stdin.read(&mut buf[read_total..want])?;
        if n == 0 {
            break;
        }
        read_total += n;
    }

    let aligned = read_total - (read_total % 12);
    let mut pts = Vec::with_capacity(aligned / 12);
    for c in buf[..aligned].chunks_exact(12) {
        let x = f32::from_le_bytes([c[0], c[1], c[2], c[3]]);
        let y = f32::from_le_bytes([c[4], c[5], c[6], c[7]]);
        let z = f32::from_le_bytes([c[8], c[9], c[10], c[11]]);
        pts.push(Point { x, y, z });
    }
    Ok(pts)
}

fn run_streaming_like(n: u64, k: usize, mini_batch: bool) -> io::Result<()> {
    let mut stdin = io::stdin().lock();

    let mut centroids = vec![Centroid::default(); k];
    let mut init_done = 0usize;
    let mut rng = 0x9E3779B97F4A7C15u64;

    let mut counts = vec![1u32; k];

    let train_points = if mini_batch { (n.min(50_000_000)) } else { n.min(10_000_000) };
    let passes = if mini_batch { 2 } else { 1 };

    let chunk_points = if mini_batch { 1_000_000usize } else { 250_000usize };
    let mut buf = vec![0u8; chunk_points * 12];

    for _pass in 0..passes {
        let mut seen: u64 = 0;
        loop {
            let pts = read_points_chunk(&mut stdin, &mut buf, chunk_points)?;
            if pts.is_empty() {
                break;
            }

            for p in &pts {
                if init_done < k {
                    centroids[init_done] = Centroid::from_point(p);
                    init_done += 1;
                    seen += 1;
                    continue;
                }

                let c = nearest_cluster(&centroids, p);
                counts[c] = counts[c].saturating_add(1);
                let eta = if mini_batch {
                    0.2f32 / (1.0 + (counts[c] as f32).ln() * 0.1)
                } else {
                    1.0f32 / (counts[c] as f32).sqrt()
                };
                centroids[c].x += eta * (p.x - centroids[c].x);
                centroids[c].y += eta * (p.y - centroids[c].y);
                centroids[c].z += eta * (p.z - centroids[c].z);

                if !mini_batch {
                    let j = (next_u64(&mut rng) as usize) % k;
                    if counts[j] > counts[c].saturating_mul(20) {
                        centroids[j] = *p;
                        counts[j] = 1;
                    }
                }

                seen += 1;
                if seen >= train_points {
                    break;
                }
            }

            if seen >= train_points {
                break;
            }
        }

        if init_done == 0 {
            break;
        }

        if init_done < k {
            for i in init_done..k {
                centroids[i] = centroids[i % init_done];
            }
            init_done = k;
        }
    }

    if init_done == 0 {
        return Ok(());
    }

    let mut out = BufWriter::new(io::stdout().lock());
    let fallback_cluster = 0u32;

    loop {
        let pts = read_points_chunk(&mut stdin, &mut buf, chunk_points)?;
        if pts.is_empty() {
            break;
        }
        for p in &pts {
            let c = if k > 0 {
                nearest_cluster(&centroids, p) as u32
            } else {
                fallback_cluster
            };
            out.write_all(&c.to_le_bytes())?;
        }
    }
    out.flush()
}

fn main() -> io::Result<()> {
    let (n, k) = read_header()?;

    if n == 0 {
        return Ok(());
    }

    match choose_strategy(n, k) {
        Strategy::InMemory => run_in_memory(n, k),
        Strategy::MiniBatch => run_streaming_like(n, k, true),
        Strategy::Streaming => run_streaming_like(n, k, false),
    }
}
