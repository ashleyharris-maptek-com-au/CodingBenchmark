import sys


def parse_input():
    data = sys.stdin.read().splitlines()
    if not data:
        return None

    first = data[0].strip().split()
    if len(first) != 2:
        return None

    try:
        U = int(first[0])
        S = int(first[1])
    except ValueError:
        return None

    sets = []
    for i in range(S):
        line_idx = 1 + i
        if line_idx >= len(data):
            elems = []
        else:
            line = data[line_idx].strip()
            elems = [] if line == "" else [int(x) for x in line.split()]
        sets.append(elems)

    return U, S, sets


def solve_exact_cover(U, S, sets):
    # Reject invalid set contents early.
    for st in sets:
        seen = set()
        for e in st:
            if e < 0 or e >= U or e in seen:
                return None
            seen.add(e)

    element_to_sets = [[] for _ in range(U)]
    for i, st in enumerate(sets):
        for e in st:
            element_to_sets[e].append(i)

    # If any universe element is in no set, impossible.
    for e in range(U):
        if not element_to_sets[e]:
            return None

    used_elements = [False] * U
    chosen = []

    # Deterministic order helps stable output.
    for lst in element_to_sets:
        lst.sort()

    def backtrack(covered_count):
        if covered_count == U:
            return True

        # Choose an uncovered element with minimum available compatible sets.
        best_e = -1
        best_candidates = None

        for e in range(U):
            if used_elements[e]:
                continue

            candidates = []
            for si in element_to_sets[e]:
                st = sets[si]
                ok = True
                for x in st:
                    if used_elements[x]:
                        ok = False
                        break
                if ok:
                    candidates.append(si)

            if not candidates:
                return False

            if best_candidates is None or len(candidates) < len(best_candidates):
                best_e = e
                best_candidates = candidates
                if len(best_candidates) == 1:
                    break

        # Try candidates in ascending index for deterministic solution.
        for si in best_candidates:
            st = sets[si]
            for x in st:
                used_elements[x] = True
            chosen.append(si)

            if backtrack(covered_count + len(st)):
                return True

            chosen.pop()
            for x in st:
                used_elements[x] = False

        return False

    if backtrack(0):
        chosen.sort()
        return chosen
    return None


def main():
    parsed = parse_input()
    if parsed is None:
        print("NO SOLUTION")
        return

    U, S, sets = parsed
    ans = solve_exact_cover(U, S, sets)

    if ans is None:
        print("NO SOLUTION")
    else:
        print("SOLUTION")
        print(" ".join(map(str, ans)))


if __name__ == "__main__":
    main()
