using namespace std;

struct U128Hash {
    size_t operator()(const unsigned __int128& x) const noexcept {
        uint64_t lo = static_cast<uint64_t>(x);
        uint64_t hi = static_cast<uint64_t>(x >> 64);
        uint64_t h = lo ^ (hi + 0x9e3779b97f4a7c15ULL + (lo << 6) + (lo >> 2));
        return static_cast<size_t>(h);
    }
};

struct RH {
    static constexpr uint64_t MOD1 = 1000000007ULL;
    static constexpr uint64_t MOD2 = 1000000009ULL;
    static constexpr uint64_t BASE1 = 911382323ULL;
    static constexpr uint64_t BASE2 = 972663749ULL;

    vector<uint64_t> p1, p2, h1, h2;

    RH() = default;
    explicit RH(const string& s) { build(s); }

    void build(const string& s) {
        int n = static_cast<int>(s.size());
        p1.assign(n + 1, 1);
        p2.assign(n + 1, 1);
        h1.assign(n + 1, 0);
        h2.assign(n + 1, 0);

        for (int i = 0; i < n; ++i) {
            p1[i + 1] = (p1[i] * BASE1) % MOD1;
            p2[i + 1] = (p2[i] * BASE2) % MOD2;
            uint64_t c = static_cast<unsigned char>(s[i]) + 1ULL;
            h1[i + 1] = (h1[i] * BASE1 + c) % MOD1;
            h2[i + 1] = (h2[i] * BASE2 + c) % MOD2;
        }
    }

    inline pair<uint64_t, uint64_t> get(int l, int r) const {
        uint64_t x1 = (h1[r] + MOD1 - (h1[l] * p1[r - l]) % MOD1) % MOD1;
        uint64_t x2 = (h2[r] + MOD2 - (h2[l] * p2[r - l]) % MOD2) % MOD2;
        return {x1, x2};
    }
};

static inline unsigned __int128 packHash(uint64_t a, uint64_t b) {
    return (static_cast<unsigned __int128>(a) << 64) | static_cast<unsigned __int128>(b);
}

string bruteForceLCS(const vector<string>& s) {
    if (s.empty()) return "";

    int minIdx = 0;
    for (int i = 1; i < static_cast<int>(s.size()); ++i) {
        if (s[i].size() < s[minIdx].size()) minIdx = i;
    }

    vector<string> t;
    t.reserve(s.size());
    t.push_back(s[minIdx]);
    for (int i = 0; i < static_cast<int>(s.size()); ++i) {
        if (i != minIdx) t.push_back(s[i]);
    }

    const string& base = t[0];
    int n = static_cast<int>(base.size());

    for (int len = n; len >= 1; --len) {
        for (int i = 0; i + len <= n; ++i) {
            string sub = base.substr(i, len);
            bool ok = true;
            for (int k = 1; k < static_cast<int>(t.size()); ++k) {
                if (t[k].find(sub) == string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) return sub;
        }
    }

    return "";
}

struct AdaptiveLCS {
    vector<string> s;
    vector<RH> hs;
    vector<int> order;

    explicit AdaptiveLCS(vector<string> in) : s(move(in)) {}

    void sortByLength() {
        int n = static_cast<int>(s.size());
        order.resize(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b) {
            return s[a].size() < s[b].size();
        });

        vector<string> ns;
        ns.reserve(n);
        for (int id : order) ns.push_back(move(s[id]));
        s.swap(ns);
    }

    bool checkLen(int L, string* witness) {
        if (L == 0) {
            if (witness) *witness = "";
            return true;
        }

        unordered_set<unsigned __int128, U128Hash> common;
        common.reserve(s[0].size());

        for (int i = 0; i + L <= static_cast<int>(s[0].size()); ++i) {
            auto [a, b] = hs[0].get(i, i + L);
            common.insert(packHash(a, b));
        }
        if (common.empty()) return false;

        for (int k = 1; k < static_cast<int>(s.size()); ++k) {
            unordered_set<unsigned __int128, U128Hash> cur;
            cur.reserve(common.size());
            for (int i = 0; i + L <= static_cast<int>(s[k].size()); ++i) {
                auto [a, b] = hs[k].get(i, i + L);
                unsigned __int128 h = packHash(a, b);
                if (common.find(h) != common.end()) cur.insert(h);
            }
            common.swap(cur);
            if (common.empty()) return false;
        }

        if (!witness) return true;

        for (int i = 0; i + L <= static_cast<int>(s[0].size()); ++i) {
            auto [a, b] = hs[0].get(i, i + L);
            unsigned __int128 h = packHash(a, b);
            if (common.find(h) != common.end()) {
                *witness = s[0].substr(i, L);
                return true;
            }
        }

        return false;
    }

    string solveHashBinarySearch() {
        sortByLength();
        hs.clear();
        hs.reserve(s.size());
        for (const auto& x : s) hs.emplace_back(x);

        int lo = 0;
        int hi = static_cast<int>(s[0].size());
        string best;

        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if (checkLen(mid, nullptr)) {
                lo = mid + 1;
            } else {
                hi = mid - 1;
            }
        }

        checkLen(hi, &best);
        return best;
    }

    string solve() {
        const int n = static_cast<int>(s.size());
        size_t minLen = numeric_limits<size_t>::max();
        size_t total = 0;
        for (const auto& x : s) {
            minLen = min(minLen, x.size());
            total += x.size();
        }

        // Adaptive strategy:
        // - Tiny inputs: exact brute force over shortest string substrings.
        // - Everything else: double rolling-hash + binary search on length.
        // This keeps constants low for tiny cases while scaling to very large text.
        if (n <= 3 && minLen <= 80 && total <= 300) {
            return bruteForceLCS(s);
        }
        return solveHashBinarySearch();
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) {
        return 0;
    }
    string dummy;
    getline(cin, dummy); // consume endline after N

    vector<string> arr;
    arr.reserve(max(0, N));

    for (int i = 0; i < N; ++i) {
        string line;
        getline(cin, line);
        arr.push_back(move(line));
    }

    if (arr.empty()) {
        cout << '\n';
        return 0;
    }

    AdaptiveLCS solver(move(arr));
    string ans = solver.solve();
    cout << ans << '\n';
    return 0;
}
