#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <random>
#include <unordered_set>
#include <vector>

using namespace std;

struct Graph {
    int n = 0;
    int m = 0;
    int k = 0;
    vector<vector<int>> adj;
    vector<int> degree;
};

static inline uint64_t make_key(int a, int b) {
    if (a > b) swap(a, b);
    return (static_cast<uint64_t>(static_cast<uint32_t>(a)) << 32) |
           static_cast<uint32_t>(b);
}

bool read_input(Graph& g, vector<pair<int, int>>& edges) {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> g.n >> g.m >> g.k)) return false;
    if (g.n < 0 || g.m < 0 || g.k <= 0) return false;

    edges.clear();
    edges.reserve(static_cast<size_t>(g.m));

    g.adj.assign(g.n, {});
    g.degree.assign(g.n, 0);

    unordered_set<uint64_t> seen;
    seen.reserve(static_cast<size_t>(g.m) * 2 + 1);

    for (int i = 0; i < g.m; ++i) {
        int u, v;
        cin >> u >> v;
        if (!cin) return false;
        if (u < 0 || u >= g.n || v < 0 || v >= g.n || u == v) continue;
        uint64_t key = make_key(u, v);
        if (seen.insert(key).second) {
            edges.emplace_back(u, v);
            g.adj[u].push_back(v);
            g.adj[v].push_back(u);
            g.degree[u]++;
            g.degree[v]++;
        }
    }

    g.m = static_cast<int>(edges.size());
    return true;
}

vector<int> greedy_dsatur(const Graph& g, mt19937_64& rng) {
    const int n = g.n;
    const int k = g.k;

    vector<int> color(n, -1);
    vector<int> sat(n, 0);
    vector<int> rem_deg = g.degree;

    vector<uint64_t> forbidden(static_cast<size_t>(n), 0ULL);

    auto can_forbid_bitset = [&]() { return k <= 63; };

    int colored = 0;
    while (colored < n) {
        int best = -1;
        for (int v = 0; v < n; ++v) {
            if (color[v] != -1) continue;
            if (best == -1 || sat[v] > sat[best] ||
                (sat[v] == sat[best] && rem_deg[v] > rem_deg[best])) {
                best = v;
            }
        }
        if (best == -1) break;

        int chosen = -1;
        if (can_forbid_bitset()) {
            uint64_t mask = forbidden[best];
            for (int c = 0; c < k; ++c) {
                if (((mask >> c) & 1ULL) == 0ULL) {
                    chosen = c;
                    break;
                }
            }
        } else {
            vector<char> used(k, 0);
            for (int nb : g.adj[best]) {
                if (color[nb] != -1) used[color[nb]] = 1;
            }
            for (int c = 0; c < k; ++c) {
                if (!used[c]) {
                    chosen = c;
                    break;
                }
            }
        }

        if (chosen == -1) {
            vector<int> order(k);
            iota(order.begin(), order.end(), 0);
            shuffle(order.begin(), order.end(), rng);
            chosen = order[0];
        }

        color[best] = chosen;
        colored++;

        for (int nb : g.adj[best]) {
            if (color[nb] != -1) continue;
            if (k <= 63) {
                uint64_t old = forbidden[nb];
                uint64_t nw = old | (1ULL << chosen);
                if (nw != old) sat[nb]++;
                forbidden[nb] = nw;
            }
            rem_deg[nb]--;
        }
    }

    return color;
}

long long count_conflicts(const Graph& g, const vector<pair<int, int>>& edges, const vector<int>& color,
                          vector<int>* per_v = nullptr) {
    if (per_v) per_v->assign(g.n, 0);
    long long bad = 0;
    for (const auto& e : edges) {
        int u = e.first, v = e.second;
        if (color[u] == color[v]) {
            bad++;
            if (per_v) {
                (*per_v)[u]++;
                (*per_v)[v]++;
            }
        }
    }
    return bad;
}

bool min_conflicts_repair(const Graph& g, const vector<pair<int, int>>& edges, vector<int>& color,
                          mt19937_64& rng, chrono::steady_clock::time_point deadline) {
    const int n = g.n;
    const int k = g.k;

    if (n == 0) return true;

    vector<int> conflicts_per_v;
    long long conflicts = count_conflicts(g, edges, color, &conflicts_per_v);
    if (conflicts == 0) return true;

    vector<int> neigh_color_count;
    if (k <= 4096) neigh_color_count.resize(k);

    uniform_int_distribution<int> pick_v(0, n - 1);

    long long steps = 0;
    long long plateau = 0;
    const long long restart_after = max<long long>(50000, 10LL * n);

    while (chrono::steady_clock::now() < deadline) {
        if (conflicts == 0) return true;

        int v = -1;
        for (int tries = 0; tries < 24; ++tries) {
            int t = pick_v(rng);
            if (conflicts_per_v[t] > 0) {
                v = t;
                break;
            }
        }
        if (v == -1) {
            int best = -1;
            for (int i = 0; i < n; ++i) {
                if (conflicts_per_v[i] > 0 && (best == -1 || conflicts_per_v[i] > conflicts_per_v[best])) best = i;
            }
            if (best == -1) return true;
            v = best;
        }

        int oldc = color[v];
        int bestc = oldc;
        int bestScore = INT32_MAX;

        if (k <= 4096) {
            fill(neigh_color_count.begin(), neigh_color_count.end(), 0);
            for (int nb : g.adj[v]) neigh_color_count[color[nb]]++;

            for (int c = 0; c < k; ++c) {
                int sc = neigh_color_count[c];
                if (sc < bestScore) {
                    bestScore = sc;
                    bestc = c;
                }
            }
        } else {
            for (int c = 0; c < k; ++c) {
                int sc = 0;
                for (int nb : g.adj[v]) sc += (color[nb] == c);
                if (sc < bestScore) {
                    bestScore = sc;
                    bestc = c;
                }
            }
        }

        if (bestc != oldc) {
            color[v] = bestc;
            conflicts = count_conflicts(g, edges, color, &conflicts_per_v);
            plateau = 0;
        } else {
            plateau++;
        }

        steps++;

        if (plateau > restart_after) {
            for (int i = 0; i < n; ++i) {
                if (conflicts_per_v[i] > 0) {
                    color[i] = static_cast<int>(rng() % static_cast<uint64_t>(k));
                }
            }
            conflicts = count_conflicts(g, edges, color, &conflicts_per_v);
            plateau = 0;
        }
    }

    return count_conflicts(g, edges, color) == 0;
}

vector<int> largest_first_order(const Graph& g) {
    vector<int> ord(g.n);
    iota(ord.begin(), ord.end(), 0);
    sort(ord.begin(), ord.end(), [&](int a, int b) {
        if (g.degree[a] != g.degree[b]) return g.degree[a] > g.degree[b];
        return a < b;
    });
    return ord;
}

vector<int> greedy_order_coloring(const Graph& g, const vector<int>& order) {
    vector<int> color(g.n, -1);
    vector<char> used(static_cast<size_t>(g.k), 0);

    for (int v : order) {
        fill(used.begin(), used.end(), 0);
        for (int nb : g.adj[v]) {
            if (color[nb] != -1) used[color[nb]] = 1;
        }
        int pick = -1;
        for (int c = 0; c < g.k; ++c) {
            if (!used[c]) {
                pick = c;
                break;
            }
        }
        if (pick == -1) pick = 0;
        color[v] = pick;
    }
    return color;
}

int main() {
    Graph g;
    vector<pair<int, int>> edges;
    if (!read_input(g, edges)) return 0;

    auto start = chrono::steady_clock::now();
    auto deadline = start + chrono::minutes(4) + chrono::seconds(50);

    mt19937_64 rng((uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count() ^ 0x9E3779B97F4A7C15ULL);

    vector<int> best(g.n, 0);
    long long best_conf = (1LL << 62);

    // Strategy selection by size/density.
    const double density = (g.n <= 1) ? 0.0 : (2.0 * g.m) / (1.0 * g.n * (g.n - 1));

    // Strategy 1: largest-first greedy for sparse huge graphs.
    if (g.n >= 200000 || density < 0.0002) {
        auto ord = largest_first_order(g);
        auto c = greedy_order_coloring(g, ord);
        long long cf = count_conflicts(g, edges, c);
        if (cf < best_conf) {
            best_conf = cf;
            best = move(c);
        }
    }

    // Strategy 2: repeated DSATUR-style construction + min-conflicts repair.
    int rounds = 0;
    while (chrono::steady_clock::now() < deadline) {
        vector<int> cand = greedy_dsatur(g, rng);
        for (int i = 0; i < g.n; ++i) {
            if (cand[i] < 0 || cand[i] >= g.k) cand[i] = static_cast<int>(rng() % static_cast<uint64_t>(g.k));
        }

        auto repair_deadline = min(deadline, chrono::steady_clock::now() + chrono::milliseconds(g.n <= 20000 ? 300 : 120));
        min_conflicts_repair(g, edges, cand, rng, repair_deadline);

        long long cf = count_conflicts(g, edges, cand);
        if (cf < best_conf) {
            best_conf = cf;
            best = cand;
            if (best_conf == 0) break;
        }

        rounds++;
        if (g.n > 300000 && rounds >= 2) break;
        if (g.n > 100000 && rounds >= 4) break;
    }

    // Final safety repair attempt to maximize chance of valid complete coloring.
    if (best_conf > 0) {
        auto final_deadline = min(deadline, chrono::steady_clock::now() + chrono::milliseconds(800));
        min_conflicts_repair(g, edges, best, rng, final_deadline);
    }

    for (int i = 0; i < g.n; ++i) {
        int c = best[i];
        if (c < 0 || c >= g.k) c = ((c % g.k) + g.k) % g.k;
        if (i) cout << ' ';
        cout << c;
    }
    cout << '\n';

    return 0;
}
