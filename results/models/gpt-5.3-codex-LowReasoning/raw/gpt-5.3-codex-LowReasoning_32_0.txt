#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int u, v;
    long long w;
};

struct DSU {
    vector<int> p, r;
    DSU() = default;
    explicit DSU(int n) : p(n), r(n, 0) { iota(p.begin(), p.end(), 0); }
    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

static const long long INF = numeric_limits<long long>::max() / 4;

struct ParentInfo {
    // type: 0 = none/base, 1 = combine (split), 2 = dijkstra relax from prev_v by edge_id
    uint8_t type = 0;
    int split = -1;
    int prev_v = -1;
    int edge_id = -1;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<Edge> edges(m);
    vector<vector<pair<int, int>>> adj(n); // (to, edge_id)
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges[i] = {u, v, w};
        adj[u].push_back({v, i});
        adj[v].push_back({u, i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    auto build_tree_from_edge_ids = [&](const vector<int>& chosen_edges) {
        vector<int> dedup = chosen_edges;
        sort(dedup.begin(), dedup.end());
        dedup.erase(unique(dedup.begin(), dedup.end()), dedup.end());

        vector<int> used_vertices;
        used_vertices.reserve(dedup.size() * 2);
        for (int id : dedup) {
            used_vertices.push_back(edges[id].u);
            used_vertices.push_back(edges[id].v);
        }
        sort(used_vertices.begin(), used_vertices.end());
        used_vertices.erase(unique(used_vertices.begin(), used_vertices.end()), used_vertices.end());

        vector<int> id_of(n, -1);
        for (int i = 0; i < (int)used_vertices.size(); ++i) id_of[used_vertices[i]] = i;

        vector<int> mst_ids;
        mst_ids.reserve(dedup.size());
        vector<int> sorted_edges = dedup;
        sort(sorted_edges.begin(), sorted_edges.end(), [&](int a, int b) {
            if (edges[a].w != edges[b].w) return edges[a].w < edges[b].w;
            return a < b;
        });

        DSU dsu((int)used_vertices.size());
        for (int id : sorted_edges) {
            int a = id_of[edges[id].u], b = id_of[edges[id].v];
            if (a >= 0 && b >= 0 && dsu.unite(a, b)) mst_ids.push_back(id);
        }

        // Prune non-terminal leaves repeatedly.
        vector<char> is_terminal(n, 0);
        for (int t : terminals) is_terminal[t] = 1;

        vector<vector<pair<int, int>>> t_adj(n);
        vector<char> alive(m, 0);
        for (int id : mst_ids) {
            alive[id] = 1;
            int u = edges[id].u, v = edges[id].v;
            t_adj[u].push_back({v, id});
            t_adj[v].push_back({u, id});
        }

        vector<int> deg(n, 0);
        for (int v = 0; v < n; ++v) {
            for (auto &p : t_adj[v]) if (alive[p.second]) deg[v]++;
        }

        queue<int> q;
        for (int v = 0; v < n; ++v) {
            if (!is_terminal[v] && deg[v] == 1) q.push(v);
        }

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            if (is_terminal[v] || deg[v] != 1) continue;
            int eid = -1, to = -1;
            for (auto &pr : t_adj[v]) {
                if (alive[pr.second]) {
                    eid = pr.second;
                    to = pr.first;
                    break;
                }
            }
            if (eid == -1) continue;
            alive[eid] = 0;
            deg[v]--;
            deg[to]--;
            if (!is_terminal[to] && deg[to] == 1) q.push(to);
        }

        vector<int> out;
        for (int id : mst_ids) if (alive[id]) out.push_back(id);
        return out;
    };

    vector<int> final_edge_ids;

    if (k == 0) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }
    if (k == 1) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    // Exact Dreyfus-Wagner for small k.
    if (k <= 15) {
        int S = 1 << k;
        vector<vector<long long>> dp(S, vector<long long>(n, INF));
        vector<vector<ParentInfo>> par(S, vector<ParentInfo>(n));

        for (int i = 0; i < k; ++i) {
            dp[1 << i][terminals[i]] = 0;
            par[1 << i][terminals[i]].type = 0;
        }

        using Node = pair<long long, int>;

        for (int mask = 1; mask < S; ++mask) {
            // subset merge
            for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                int other = mask ^ sub;
                if (other == 0) continue;
                if (sub > other) continue;
                for (int v = 0; v < n; ++v) {
                    long long a = dp[sub][v], b = dp[other][v];
                    if (a == INF || b == INF) continue;
                    long long cand = a + b;
                    if (cand < dp[mask][v]) {
                        dp[mask][v] = cand;
                        par[mask][v].type = 1;
                        par[mask][v].split = sub;
                    }
                }
            }

            // multi-source dijkstra to propagate through graph
            priority_queue<Node, vector<Node>, greater<Node>> pq;
            vector<char> vis(n, 0);
            for (int v = 0; v < n; ++v) {
                if (dp[mask][v] < INF) pq.push({dp[mask][v], v});
            }

            while (!pq.empty()) {
                auto [d, v] = pq.top();
                pq.pop();
                if (vis[v]) continue;
                vis[v] = 1;
                if (d != dp[mask][v]) continue;
                for (auto [to, eid] : adj[v]) {
                    long long nd = d + edges[eid].w;
                    if (nd < dp[mask][to]) {
                        dp[mask][to] = nd;
                        par[mask][to].type = 2;
                        par[mask][to].prev_v = v;
                        par[mask][to].edge_id = eid;
                        pq.push({nd, to});
                    }
                }
            }
        }

        int full = S - 1;
        int best_v = -1;
        long long best = INF;
        for (int v = 0; v < n; ++v) {
            if (dp[full][v] < best) {
                best = dp[full][v];
                best_v = v;
            }
        }

        vector<int> collected;
        unordered_set<long long> seen;

        function<void(int, int)> rec = [&](int mask, int v) {
            long long key = (static_cast<long long>(mask) << 32) ^ static_cast<unsigned>(v);
            if (seen.find(key) != seen.end()) return;
            seen.insert(key);

            ParentInfo pi = par[mask][v];
            if (pi.type == 0) return;
            if (pi.type == 1) {
                int a = pi.split;
                int b = mask ^ a;
                rec(a, v);
                rec(b, v);
            } else if (pi.type == 2) {
                collected.push_back(pi.edge_id);
                rec(mask, pi.prev_v);
            }
        };

        rec(full, best_v);
        final_edge_ids = build_tree_from_edge_ids(collected);
    } else {
        // Heuristic: terminal metric closure MST, expand shortest paths, then prune.
        vector<vector<long long>> distT(k, vector<long long>(n, INF));
        vector<vector<int>> parent_edge(k, vector<int>(n, -1));

        using Node = pair<long long, int>;
        for (int i = 0; i < k; ++i) {
            priority_queue<Node, vector<Node>, greater<Node>> pq;
            distT[i][terminals[i]] = 0;
            pq.push({0, terminals[i]});
            vector<char> vis(n, 0);
            while (!pq.empty()) {
                auto [d, v] = pq.top();
                pq.pop();
                if (vis[v]) continue;
                vis[v] = 1;
                if (d != distT[i][v]) continue;
                for (auto [to, eid] : adj[v]) {
                    long long nd = d + edges[eid].w;
                    if (nd < distT[i][to]) {
                        distT[i][to] = nd;
                        parent_edge[i][to] = eid;
                        pq.push({nd, to});
                    }
                }
            }
        }

        struct TEdge { int a, b; long long w; };
        vector<TEdge> tEdges;
        tEdges.reserve((long long)k * (k - 1) / 2);
        for (int i = 0; i < k; ++i) {
            for (int j = i + 1; j < k; ++j) {
                tEdges.push_back({i, j, distT[i][terminals[j]]});
            }
        }
        sort(tEdges.begin(), tEdges.end(), [](const TEdge& x, const TEdge& y) {
            if (x.w != y.w) return x.w < y.w;
            if (x.a != y.a) return x.a < y.a;
            return x.b < y.b;
        });

        DSU dsuK(k);
        vector<pair<int, int>> chosenPairs;
        for (auto &e : tEdges) {
            if (dsuK.unite(e.a, e.b)) {
                chosenPairs.push_back({e.a, e.b});
                if ((int)chosenPairs.size() == k - 1) break;
            }
        }

        vector<int> collected;
        for (auto [a, b] : chosenPairs) {
            int srcIdx = a;
            int cur = terminals[b];
            while (cur != terminals[srcIdx]) {
                int eid = parent_edge[srcIdx][cur];
                if (eid < 0) break; // disconnected guard
                collected.push_back(eid);
                int u = edges[eid].u, v = edges[eid].v;
                cur = (cur == u ? v : u);
            }
        }

        final_edge_ids = build_tree_from_edge_ids(collected);
    }

    long long total = 0;
    for (int id : final_edge_ids) total += edges[id].w;

    cout << total << " " << final_edge_ids.size() << "\n";
    for (int id : final_edge_ids) {
        cout << edges[id].u << " " << edges[id].v << "\n";
    }

    return 0;
}
