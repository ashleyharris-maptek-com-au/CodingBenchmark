using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    static void Main()
    {
        var scanner = new FastScanner();
        int width = scanner.NextInt();
        int height = scanner.NextInt();
        int numPieces = scanner.NextInt();

        var ai = new TetrisAI(width, height, numPieces);

        for (int i = 0; i < numPieces; i++)
        {
            string piece = scanner.NextToken();
            if (string.IsNullOrEmpty(piece)) break;

            var move = ai.ChooseMove(piece[0]);
            Console.WriteLine(move.rotation.ToString(CultureInfo.InvariantCulture) + " " + move.column.ToString(CultureInfo.InvariantCulture));
            Console.Out.Flush();

            string result = scanner.NextToken();
            if (string.IsNullOrEmpty(result)) break;
            ai.ApplyFeedback(piece[0], move.rotation, move.column, result);
            if (result == "gameover") break;
        }
    }
}

public sealed class TetrisAI
{
    private readonly int width;
    private readonly int height;
    private readonly int[] board;
    private readonly int totalPieces;
    private int turn;

    private readonly Dictionary<char, PieceDef> defs;

    public TetrisAI(int width, int height, int totalPieces)
    {
        this.width = width;
        this.height = height;
        this.totalPieces = totalPieces;
        board = new int[width];
        turn = 0;
        defs = PieceLibrary.Build();
    }

    public (int rotation, int column) ChooseMove(char pieceType)
    {
        var def = defs[pieceType];
        var candidates = GenerateMoves(def);

        bool complexBoard = IsComplexBoard();
        bool marathonMode = totalPieces > 800 || turn > 200;

        Weights w;
        if (complexBoard)
        {
            w = marathonMode
                ? new Weights(-0.65, 0.85, 0.35, 0.30, 0.55, -0.04)
                : new Weights(-0.55, 0.95, 0.45, 0.40, 0.70, -0.03);
        }
        else
        {
            w = marathonMode
                ? new Weights(-0.78, 0.70, 0.22, 0.20, 0.35, -0.02)
                : new Weights(-0.88, 0.62, 0.18, 0.18, 0.28, -0.015);
        }

        double bestScore = double.NegativeInfinity;
        int bestRot = 0;
        int bestCol = 0;

        for (int i = 0; i < candidates.Count; i++)
        {
            var c = candidates[i];
            double s = Score(c.features, w);
            if (s > bestScore)
            {
                bestScore = s;
                bestRot = c.rotation;
                bestCol = c.column;
            }
        }

        return (bestRot, bestCol);
    }

    public void ApplyFeedback(char pieceType, int rotation, int column, string result)
    {
        if (result == "gameover")
        {
            turn++;
            return;
        }

        var def = defs[pieceType];
        Shape s = def.Rotations[rotation % def.Rotations.Count];

        if (!TryPlace(board, s, column, out int[] newBoard, out int dropY))
        {
            turn++;
            return;
        }

        int cleared = ClearFullRows(newBoard, width, height);
        Array.Copy(newBoard, board, width);
        turn++;
    }

    private bool IsComplexBoard()
    {
        int maxH = 0;
        int minH = int.MaxValue;
        int bumps = 0;
        int holes = CountHoles(board);

        for (int i = 0; i < width; i++)
        {
            int h = board[i];
            if (h > maxH) maxH = h;
            if (h < minH) minH = h;
            if (i > 0) bumps += Math.Abs(board[i] - board[i - 1]);
        }

        int relief = maxH - minH;
        return holes >= Math.Max(2, width / 4) || bumps > width * 2 || relief > Math.Max(5, height / 3);
    }

    private List<Candidate> GenerateMoves(PieceDef def)
    {
        var list = new List<Candidate>(64);

        for (int r = 0; r < def.Rotations.Count; r++)
        {
            Shape s = def.Rotations[r];
            int maxCol = width - s.Width;
            for (int col = 0; col <= maxCol; col++)
            {
                if (!TryPlace(board, s, col, out int[] nextBoard, out int dropY)) continue;

                int cleared = ClearFullRows(nextBoard, width, height);
                Features f = ExtractFeatures(nextBoard, cleared);
                list.Add(new Candidate(r, col, f));
            }
        }

        if (list.Count == 0)
        {
            return new List<Candidate> { new Candidate(0, 0, default) };
        }

        return list;
    }

    private static bool TryPlace(int[] baseBoard, Shape s, int col, out int[] nextBoard, out int dropY)
    {
        int width = baseBoard.Length;
        nextBoard = null;
        dropY = -1;

        int landing = int.MinValue;
        for (int lx = 0; lx < s.Width; lx++)
        {
            int x = col + lx;
            if (x < 0 || x >= width) return false;
            int y = baseBoard[x] - s.Bottom[lx];
            if (y > landing) landing = y;
        }

        if (landing < 0) landing = 0;

        int[] nb = new int[width];
        Array.Copy(baseBoard, nb, width);

        for (int lx = 0; lx < s.Width; lx++)
        {
            int x = col + lx;
            int colTop = landing + s.Top[lx];
            if (colTop > nb[x]) nb[x] = colTop;
        }

        if (landing + s.Height > 2000000000)
        {
            return false;
        }

        nextBoard = nb;
        dropY = landing;
        return true;
    }

    private static int ClearFullRows(int[] heights, int width, int logicalHeight)
    {
        int minTop = int.MaxValue;
        int maxTop = 0;
        for (int i = 0; i < width; i++)
        {
            if (heights[i] < minTop) minTop = heights[i];
            if (heights[i] > maxTop) maxTop = heights[i];
        }

        if (minTop <= 0 || minTop == int.MaxValue) return 0;

        int clear = minTop;
        for (int i = 0; i < width; i++)
        {
            heights[i] -= clear;
            if (heights[i] < 0) heights[i] = 0;
        }

        return clear;
    }

    private Features ExtractFeatures(int[] h, int linesCleared)
    {
        int aggregate = 0;
        int maxH = 0;
        int bump = 0;

        for (int i = 0; i < width; i++)
        {
            aggregate += h[i];
            if (h[i] > maxH) maxH = h[i];
            if (i > 0) bump += Math.Abs(h[i] - h[i - 1]);
        }

        int holes = CountHoles(h);
        int wells = CountWells(h);

        return new Features
        {
            aggregateHeight = aggregate,
            maxHeight = maxH,
            bumpiness = bump,
            holes = holes,
            wells = wells,
            linesCleared = linesCleared
        };
    }

    private int CountHoles(int[] h)
    {
        int maxH = 0;
        for (int i = 0; i < width; i++) if (h[i] > maxH) maxH = h[i];

        int holes = 0;
        for (int y = 0; y < maxH; y++)
        {
            int blocks = 0;
            for (int x = 0; x < width; x++)
            {
                if (h[x] > y) blocks++;
            }
            holes += width - blocks;
        }

        return holes / 4;
    }

    private int CountWells(int[] h)
    {
        int wells = 0;
        for (int x = 0; x < width; x++)
        {
            int left = x == 0 ? h[x] + 1 : h[x - 1];
            int right = x == width - 1 ? h[x] + 1 : h[x + 1];
            int minNeighbor = left < right ? left : right;
            if (minNeighbor > h[x]) wells += (minNeighbor - h[x]);
        }
        return wells;
    }

    private static double Score(Features f, Weights w)
    {
        return f.linesCleared * w.lines
             + f.aggregateHeight * w.aggregateHeight
             + f.holes * w.holes
             + f.bumpiness * w.bumpiness
             + f.wells * w.wells
             + f.maxHeight * w.maxHeight;
    }

    private readonly struct Candidate
    {
        public readonly int rotation;
        public readonly int column;
        public readonly Features features;

        public Candidate(int rotation, int column, Features features)
        {
            this.rotation = rotation;
            this.column = column;
            this.features = features;
        }
    }

    private struct Features
    {
        public int aggregateHeight;
        public int maxHeight;
        public int holes;
        public int bumpiness;
        public int wells;
        public int linesCleared;
    }

    private readonly struct Weights
    {
        public readonly double aggregateHeight;
        public readonly double holes;
        public readonly double bumpiness;
        public readonly double wells;
        public readonly double maxHeight;
        public readonly double lines;

        public Weights(double aggregateHeight, double holes, double bumpiness, double wells, double maxHeight, double lines)
        {
            this.aggregateHeight = aggregateHeight;
            this.holes = holes;
            this.bumpiness = bumpiness;
            this.wells = wells;
            this.maxHeight = maxHeight;
            this.lines = lines;
        }
    }
}

public sealed class PieceDef
{
    public readonly List<Shape> Rotations;

    public PieceDef(List<Shape> rotations)
    {
        Rotations = rotations;
    }
}

public readonly struct Shape
{
    public readonly int Width;
    public readonly int Height;
    public readonly int[] Bottom;
    public readonly int[] Top;

    public Shape(int width, int height, int[] bottom, int[] top)
    {
        Width = width;
        Height = height;
        Bottom = bottom;
        Top = top;
    }
}

public static class PieceLibrary
{
    public static Dictionary<char, PieceDef> Build()
    {
        var d = new Dictionary<char, PieceDef>();

        d['I'] = new PieceDef(new List<Shape>
        {
            new Shape(4, 1, new[] { 0, 0, 0, 0 }, new[] { 1, 1, 1, 1 }),
            new Shape(1, 4, new[] { 0 }, new[] { 4 })
        });

        d['O'] = new PieceDef(new List<Shape>
        {
            new Shape(2, 2, new[] { 0, 0 }, new[] { 2, 2 })
        });

        d['T'] = new PieceDef(new List<Shape>
        {
            new Shape(3, 2, new[] { 0, 0, 0 }, new[] { 1, 2, 1 }),
            new Shape(2, 3, new[] { 0, 1 }, new[] { 3, 2 }),
            new Shape(3, 2, new[] { 1, 0, 1 }, new[] { 2, 2, 2 }),
            new Shape(2, 3, new[] { 1, 0 }, new[] { 2, 3 })
        });

        d['S'] = new PieceDef(new List<Shape>
        {
            new Shape(3, 2, new[] { 0, 0, 1 }, new[] { 1, 2, 2 }),
            new Shape(2, 3, new[] { 1, 0 }, new[] { 3, 2 })
        });

        d['Z'] = new PieceDef(new List<Shape>
        {
            new Shape(3, 2, new[] { 1, 0, 0 }, new[] { 2, 2, 1 }),
            new Shape(2, 3, new[] { 0, 1 }, new[] { 2, 3 })
        });

        d['J'] = new PieceDef(new List<Shape>
        {
            new Shape(3, 2, new[] { 0, 0, 0 }, new[] { 2, 1, 1 }),
            new Shape(2, 3, new[] { 0, 0 }, new[] { 3, 1 }),
            new Shape(3, 2, new[] { 0, 1, 1 }, new[] { 1, 2, 2 }),
            new Shape(2, 3, new[] { 2, 0 }, new[] { 3, 3 })
        });

        d['L'] = new PieceDef(new List<Shape>
        {
            new Shape(3, 2, new[] { 0, 0, 0 }, new[] { 1, 1, 2 }),
            new Shape(2, 3, new[] { 0, 2 }, new[] { 3, 3 }),
            new Shape(3, 2, new[] { 1, 1, 0 }, new[] { 2, 2, 1 }),
            new Shape(2, 3, new[] { 0, 0 }, new[] { 1, 3 })
        });

        return d;
    }
}

public sealed class FastScanner
{
    private readonly Queue<string> tokens = new Queue<string>();

    public string NextToken()
    {
        while (tokens.Count == 0)
        {
            string line = Console.ReadLine();
            if (line == null) return null;
            var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            for (int i = 0; i < parts.Length; i++) tokens.Enqueue(parts[i]);
        }
        return tokens.Dequeue();
    }

    public int NextInt()
    {
        string t = NextToken();
        return t == null ? 0 : int.Parse(t, CultureInfo.InvariantCulture);
    }
}
