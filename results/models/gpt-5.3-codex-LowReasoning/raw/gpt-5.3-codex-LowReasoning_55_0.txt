#include <algorithm>
#include <array>
#include <chrono>

#include <cmath>
#include <iostream>
#include <limits>
#include <optional>

#include <vector>

using namespace std;

struct Move {
    int from = -1;
    int to = -1;
    char promo = 0; // q,r,b,n
    bool isCapture = false;
    bool isEnPassant = false;
    bool isCastleK = false;
    bool isCastleQ = false;
};

struct State {
    array<char, 64> b{}; // '.' or piece chars PNBRQK / pnbrqk
    bool whiteToMove = true;
    bool wk = false, wq = false, bk = false, bq = false;
    int ep = -1; // square index
    int halfmove = 0;
    int fullmove = 1;
};

static inline int fileOf(int s) { return s & 7; }
static inline int rankOf(int s) { return s >> 3; }
static inline bool inBoard(int f, int r) { return f >= 0 && f < 8 && r >= 0 && r < 8; }
static inline int sq(int f, int r) { return r * 8 + f; }
static inline bool isWhite(char p) { return p >= 'A' && p <= 'Z'; }
static inline bool isBlack(char p) { return p >= 'a' && p <= 'z'; }
static inline char lowerPiece(char p) { return (char)tolower((unsigned char)p); }

string sqToStr(int s) {
    string out;
    out += char('a' + fileOf(s));
    out += char('1' + rankOf(s));
    return out;
}

int strToSq(const string& s) {
    if (s.size() != 2) return -1;
    int f = s[0] - 'a';
    int r = s[1] - '1';
    if (!inBoard(f, r)) return -1;
    return sq(f, r);
}

bool parseFEN(const string& fen, State& st) {
    fill(st.b.begin(), st.b.end(), '.');
    st.ep = -1; st.wk = st.wq = st.bk = st.bq = false;

    istringstream iss(fen);
    string board, side, castling, ep;
    if (!(iss >> board >> side >> castling >> ep >> st.halfmove >> st.fullmove)) return false;

    int r = 7, f = 0;
    for (char c : board) {
        if (c == '/') { r--; f = 0; continue; }
        if (isdigit((unsigned char)c)) {
            f += c - '0';
        } else {
            if (!inBoard(f, r)) return false;
            st.b[sq(f, r)] = c;
            f++;
        }
    }

    st.whiteToMove = (side == "w");
    if (castling.find('K') != string::npos) st.wk = true;
    if (castling.find('Q') != string::npos) st.wq = true;
    if (castling.find('k') != string::npos) st.bk = true;
    if (castling.find('q') != string::npos) st.bq = true;
    if (ep != "-") st.ep = strToSq(ep);

    return true;
}

bool isSquareAttacked(const State& st, int s, bool byWhite) {
    int f = fileOf(s), r = rankOf(s);

    // Pawns
    if (byWhite) {
        int rr = r - 1;
        if (rr >= 0) {
            if (f - 1 >= 0 && st.b[sq(f - 1, rr)] == 'P') return true;
            if (f + 1 < 8 && st.b[sq(f + 1, rr)] == 'P') return true;
        }
    } else {
        int rr = r + 1;
        if (rr < 8) {
            if (f - 1 >= 0 && st.b[sq(f - 1, rr)] == 'p') return true;
            if (f + 1 < 8 && st.b[sq(f + 1, rr)] == 'p') return true;
        }
    }

    // Knights
    static const int kf[8] = {1,2,2,1,-1,-2,-2,-1};
    static const int kr[8] = {2,1,-1,-2,-2,-1,1,2};
    for (int i = 0; i < 8; i++) {
        int nf = f + kf[i], nr = r + kr[i];
        if (!inBoard(nf, nr)) continue;
        char p = st.b[sq(nf, nr)];
        if (byWhite && p == 'N') return true;
        if (!byWhite && p == 'n') return true;
    }

    // Bishops/Queens diagonals
    static const int dfD[4] = {1,1,-1,-1};
    static const int drD[4] = {1,-1,1,-1};
    for (int d = 0; d < 4; d++) {
        int nf = f + dfD[d], nr = r + drD[d];
        while (inBoard(nf, nr)) {
            char p = st.b[sq(nf, nr)];
            if (p != '.') {
                if (byWhite && (p == 'B' || p == 'Q')) return true;
                if (!byWhite && (p == 'b' || p == 'q')) return true;
                break;
            }
            nf += dfD[d]; nr += drD[d];
        }
    }

    // Rooks/Queens straight
    static const int dfS[4] = {1,-1,0,0};
    static const int drS[4] = {0,0,1,-1};
    for (int d = 0; d < 4; d++) {
        int nf = f + dfS[d], nr = r + drS[d];
        while (inBoard(nf, nr)) {
            char p = st.b[sq(nf, nr)];
            if (p != '.') {
                if (byWhite && (p == 'R' || p == 'Q')) return true;
                if (!byWhite && (p == 'r' || p == 'q')) return true;
                break;
            }
            nf += dfS[d]; nr += drS[d];
        }
    }

    // King
    for (int df = -1; df <= 1; df++) for (int dr = -1; dr <= 1; dr++) {
        if (!df && !dr) continue;
        int nf = f + df, nr = r + dr;
        if (!inBoard(nf, nr)) continue;
        char p = st.b[sq(nf, nr)];
        if (byWhite && p == 'K') return true;
        if (!byWhite && p == 'k') return true;
    }

    return false;
}

int findKing(const State& st, bool white) {
    char k = white ? 'K' : 'k';
    for (int i = 0; i < 64; i++) if (st.b[i] == k) return i;
    return -1;
}

bool inCheck(const State& st, bool white) {
    int ksq = findKing(st, white);
    if (ksq < 0) return false;
    return isSquareAttacked(st, ksq, !white);
}

State applyMove(const State& st, const Move& m) {
    State n = st;
    char p = n.b[m.from];
    char target = n.b[m.to];
    n.b[m.from] = '.';

    if (m.isCastleK || m.isCastleQ) {
        if (p == 'K') {
            n.b[m.to] = 'K';
            if (m.isCastleK) { n.b[sq(5,0)] = 'R'; n.b[sq(7,0)] = '.'; }
            else { n.b[sq(3,0)] = 'R'; n.b[sq(0,0)] = '.'; }
        } else {
            n.b[m.to] = 'k';
            if (m.isCastleK) { n.b[sq(5,7)] = 'r'; n.b[sq(7,7)] = '.'; }
            else { n.b[sq(3,7)] = 'r'; n.b[sq(0,7)] = '.'; }
        }
    } else if (m.isEnPassant) {
        n.b[m.to] = p;
        int capSq = st.whiteToMove ? (m.to - 8) : (m.to + 8);
        n.b[capSq] = '.';
    } else {
        if (m.promo) {
            char pp = m.promo;
            if (isWhite(p)) pp = (char)toupper((unsigned char)pp);
            else pp = (char)tolower((unsigned char)pp);
            n.b[m.to] = pp;
        } else n.b[m.to] = p;
    }

    // Update castling rights
    if (p == 'K') { n.wk = n.wq = false; }
    if (p == 'k') { n.bk = n.bq = false; }
    if (m.from == sq(0,0) || m.to == sq(0,0)) n.wq = false;
    if (m.from == sq(7,0) || m.to == sq(7,0)) n.wk = false;
    if (m.from == sq(0,7) || m.to == sq(0,7)) n.bq = false;
    if (m.from == sq(7,7) || m.to == sq(7,7)) n.bk = false;

    n.ep = -1;
    if (lowerPiece(p) == 'p' && abs(m.to - m.from) == 16) {
        n.ep = (m.from + m.to) / 2;
    }

    if (lowerPiece(p) == 'p' || m.isCapture || target != '.') n.halfmove = 0;
    else n.halfmove++;

    if (!st.whiteToMove) n.fullmove++;
    n.whiteToMove = !st.whiteToMove;
    return n;
}

void addMove(vector<Move>& mv, int from, int to, bool cap=false, bool ep=false, char promo=0) {
    Move m; m.from = from; m.to = to; m.isCapture = cap; m.isEnPassant = ep; m.promo = promo; mv.push_back(m);
}

vector<Move> pseudoMoves(const State& st) {
    vector<Move> mv;
    bool white = st.whiteToMove;

    for (int s = 0; s < 64; s++) {
        char p = st.b[s];
        if (p == '.') continue;
        if (white && !isWhite(p)) continue;
        if (!white && !isBlack(p)) continue;

        int f = fileOf(s), r = rankOf(s);
        char lp = lowerPiece(p);

        if (lp == 'p') {
            int dir = white ? 1 : -1;
            int startRank = white ? 1 : 6;
            int promoRank = white ? 6 : 1;
            int nr = r + dir;
            if (inBoard(f, nr) && st.b[sq(f, nr)] == '.') {
                if (r == promoRank) {
                    for (char pr : {'q','r','b','n'}) addMove(mv, s, sq(f, nr), false, false, pr);
                } else addMove(mv, s, sq(f, nr));
                if (r == startRank) {
                    int nnr = r + 2 * dir;
                    if (inBoard(f, nnr) && st.b[sq(f, nnr)] == '.') addMove(mv, s, sq(f, nnr));
                }
            }
            for (int df : {-1, 1}) {
                int nf = f + df;
                if (!inBoard(nf, nr)) continue;
                int t = sq(nf, nr);
                if (t == st.ep) {
                    addMove(mv, s, t, true, true);
                } else {
                    char tp = st.b[t];
                    if (tp != '.' && ((white && isBlack(tp)) || (!white && isWhite(tp)))) {
                        if (r == promoRank) {
                            for (char pr : {'q','r','b','n'}) addMove(mv, s, t, true, false, pr);
                        } else addMove(mv, s, t, true, false);
                    }
                }
            }
        } else if (lp == 'n') {
            static const int kf[8] = {1,2,2,1,-1,-2,-2,-1};
            static const int kr[8] = {2,1,-1,-2,-2,-1,1,2};
            for (int i = 0; i < 8; i++) {
                int nf = f + kf[i], nr = r + kr[i];
                if (!inBoard(nf, nr)) continue;
                int t = sq(nf, nr);
                char tp = st.b[t];
                if (tp == '.') addMove(mv, s, t);
                else if ((white && isBlack(tp)) || (!white && isWhite(tp))) addMove(mv, s, t, true);
            }
        } else if (lp == 'b' || lp == 'r' || lp == 'q') {
            vector<pair<int,int>> dirs;
            if (lp == 'b' || lp == 'q') dirs.insert(dirs.end(), {{1,1},{1,-1},{-1,1},{-1,-1}});
            if (lp == 'r' || lp == 'q') dirs.insert(dirs.end(), {{1,0},{-1,0},{0,1},{0,-1}});
            for (auto [df, dr] : dirs) {
                int nf = f + df, nr = r + dr;
                while (inBoard(nf, nr)) {
                    int t = sq(nf, nr);
                    char tp = st.b[t];
                    if (tp == '.') addMove(mv, s, t);
                    else {
                        if ((white && isBlack(tp)) || (!white && isWhite(tp))) addMove(mv, s, t, true);
                        break;
                    }
                    nf += df; nr += dr;
                }
            }
        } else if (lp == 'k') {
            for (int df = -1; df <= 1; df++) for (int dr = -1; dr <= 1; dr++) {
                if (!df && !dr) continue;
                int nf = f + df, nr = r + dr;
                if (!inBoard(nf, nr)) continue;
                int t = sq(nf, nr);
                char tp = st.b[t];
                if (tp == '.') addMove(mv, s, t);
                else if ((white && isBlack(tp)) || (!white && isWhite(tp))) addMove(mv, s, t, true);
            }
            if (white) {
                if (st.wk && st.b[sq(5,0)] == '.' && st.b[sq(6,0)] == '.' && !inCheck(st, true) && !isSquareAttacked(st, sq(5,0), false) && !isSquareAttacked(st, sq(6,0), false)) {
                    Move m; m.from = s; m.to = sq(6,0); m.isCastleK = true; mv.push_back(m);
                }
                if (st.wq && st.b[sq(3,0)] == '.' && st.b[sq(2,0)] == '.' && st.b[sq(1,0)] == '.' && !inCheck(st, true) && !isSquareAttacked(st, sq(3,0), false) && !isSquareAttacked(st, sq(2,0), false)) {
                    Move m; m.from = s; m.to = sq(2,0); m.isCastleQ = true; mv.push_back(m);
                }
            } else {
                if (st.bk && st.b[sq(5,7)] == '.' && st.b[sq(6,7)] == '.' && !inCheck(st, false) && !isSquareAttacked(st, sq(5,7), true) && !isSquareAttacked(st, sq(6,7), true)) {
                    Move m; m.from = s; m.to = sq(6,7); m.isCastleK = true; mv.push_back(m);
                }
                if (st.bq && st.b[sq(3,7)] == '.' && st.b[sq(2,7)] == '.' && st.b[sq(1,7)] == '.' && !inCheck(st, false) && !isSquareAttacked(st, sq(3,7), true) && !isSquareAttacked(st, sq(2,7), true)) {
                    Move m; m.from = s; m.to = sq(2,7); m.isCastleQ = true; mv.push_back(m);
                }
            }
        }
    }
    return mv;
}

vector<Move> legalMoves(const State& st) {
    vector<Move> out;
    auto pm = pseudoMoves(st);
    bool white = st.whiteToMove;
    out.reserve(pm.size());
    for (const auto& m : pm) {
        State n = applyMove(st, m);
        if (!inCheck(n, white)) out.push_back(m);
    }
    return out;
}

int eval(const State& st) {
    static const int val[128] = {
        ['p']=100,['n']=320,['b']=330,['r']=500,['q']=900,['k']=0
    };
    int score = 0;
    for (char p : st.b) {
        if (p == '.') continue;
        int v = val[(int)lowerPiece(p)];
        score += isWhite(p) ? v : -v;
    }
    return st.whiteToMove ? score : -score;
}

struct SearchCtx {
    chrono::steady_clock::time_point deadline;
    bool timedOut = false;
};

int negamax(const State& st, int depth, int alpha, int beta, SearchCtx& ctx) {
    if (chrono::steady_clock::now() >= ctx.deadline) { ctx.timedOut = true; return 0; }

    auto lm = legalMoves(st);
    if (depth == 0) return eval(st);
    if (lm.empty()) {
        if (inCheck(st, st.whiteToMove)) return -100000 + (6 - depth);
        return 0;
    }

    // Simple move ordering: captures first
    stable_sort(lm.begin(), lm.end(), [](const Move& a, const Move& b){ return a.isCapture > b.isCapture; });

    int best = numeric_limits<int>::min() / 2;
    for (const auto& m : lm) {
        State n = applyMove(st, m);
        int sc = -negamax(n, depth - 1, -beta, -alpha, ctx);
        if (ctx.timedOut) return 0;
        if (sc > best) best = sc;
        alpha = max(alpha, sc);
        if (alpha >= beta) break;
    }
    return best;
}

string moveToNotation(const State& st, const Move& m) {
    if (m.isCastleK || m.isCastleQ) {
        State n = applyMove(st, m);
        auto oppMoves = legalMoves(n);
        bool chk = inCheck(n, n.whiteToMove);
        bool mate = chk && oppMoves.empty();
        string s = m.isCastleK ? "O-O" : "O-O-O";
        if (mate) s += "#";
        else if (chk) s += "+";
        return s;
    }

    char p = st.b[m.from];
    string s;
    if (lowerPiece(p) == 'p') s += 'p';
    else s += (char)toupper((unsigned char)lowerPiece(p));
    s += sqToStr(m.from);
    s += m.isCapture ? "x" : "";
    s += sqToStr(m.to);

    if (m.isEnPassant) s += "ep";
    if (m.promo) {
        s += "/";
        s += (char)toupper((unsigned char)m.promo);
    }

    State n = applyMove(st, m);
    auto oppMoves = legalMoves(n);
    bool chk = inCheck(n, n.whiteToMove);
    bool mate = chk && oppMoves.empty();
    if (mate) s += "#";
    else if (chk) s += "+";
    return s;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSec;
    if (!(cin >> timeoutSec)) return 0;
    string fen;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    getline(cin, fen);

    State st;
    if (!parseFEN(fen, st)) {
        cout << "" << '\n';
        return 0;
    }

    auto lm = legalMoves(st);
    if (lm.empty()) {
        cout << "" << '\n';
        return 0;
    }

    SearchCtx ctx;
    auto start = chrono::steady_clock::now();
    double budget = max(0.1, timeoutSec - 0.05);
    ctx.deadline = start + chrono::milliseconds((int)(budget * 1000.0));

    Move bestMove = lm[0];
    int bestScore = numeric_limits<int>::min() / 2;

    for (int depth = 1; depth <= 64; depth++) {
        if (chrono::steady_clock::now() >= ctx.deadline) break;
        ctx.timedOut = false;

        auto rootMoves = legalMoves(st);
        stable_sort(rootMoves.begin(), rootMoves.end(), [](const Move& a, const Move& b){ return a.isCapture > b.isCapture; });

        int localBest = numeric_limits<int>::min() / 2;
        Move localMove = rootMoves[0];

        for (const auto& m : rootMoves) {
            State n = applyMove(st, m);
            int sc = -negamax(n, depth - 1, -1000000, 1000000, ctx);
            if (ctx.timedOut) break;
            if (sc > localBest) {
                localBest = sc;
                localMove = m;
            }
        }

        if (!ctx.timedOut) {
            bestScore = localBest;
            bestMove = localMove;
            (void)bestScore;
        } else {
            break;
        }
    }

    cout << moveToNotation(st, bestMove) << '\n';
    return 0;
}
