use std::cmp::Ordering;
use std::collections::{HashMap, HashSet};
use std::io::{self, Read};

#[derive(Clone)]
struct Sample {
    hole_idx: usize,
    hole_id: String,
    sample_idx: usize,
    point: [f64; 3],
    props: Vec<Option<f64>>,
}

#[derive(Clone, Copy)]
struct Stats {
    median: f64,
    mad: f64,
}

#[derive(Clone)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    prop_name: String,
    confidence: f64,
}

fn dot(a: [f64; 3], b: [f64; 3]) -> f64 {
    a[0] * b[0] + a[1] * b[1] + a[2] * b[2]
}

fn norm(a: [f64; 3]) -> f64 {
    dot(a, a).sqrt()
}

fn normalize(v: [f64; 3]) -> [f64; 3] {
    let n = norm(v);
    if n <= 1e-12 {
        [0.0, 0.0, -1.0]
    } else {
        [v[0] / n, v[1] / n, v[2] / n]
    }
}

fn clamp(x: f64, lo: f64, hi: f64) -> f64 {
    if x < lo {
        lo
    } else if x > hi {
        hi
    } else {
        x
    }
}

fn parse_f64(tok: Option<&&str>) -> Option<f64> {
    let t = *tok?;
    if t.eq_ignore_ascii_case("nan")
        || t.eq_ignore_ascii_case("na")
        || t.eq_ignore_ascii_case("null")
        || t == "-"
    {
        return None;
    }
    t.parse::<f64>().ok().filter(|v| v.is_finite())
}

fn median(vals: &mut [f64]) -> Option<f64> {
    if vals.is_empty() {
        return None;
    }
    vals.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let n = vals.len();
    if n % 2 == 1 {
        Some(vals[n / 2])
    } else {
        Some((vals[n / 2 - 1] + vals[n / 2]) * 0.5)
    }
}

fn robust_stats(values: &[f64]) -> Option<Stats> {
    if values.is_empty() {
        return None;
    }
    let mut v = values.to_vec();
    let med = median(&mut v)?;
    let mut dev: Vec<f64> = values.iter().map(|x| (x - med).abs()).collect();
    let mad = median(&mut dev).unwrap_or(0.0);
    Some(Stats {
        median: med,
        mad: mad.max(1e-12),
    })
}

fn dist2(a: [f64; 3], b: [f64; 3]) -> f64 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

fn axis_bin(v: f64, min_v: f64, cell: f64) -> i32 {
    ((v - min_v) / cell).floor() as i32
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        println!("0");
        return;
    }

    let lines: Vec<&str> = input.lines().collect();
    if lines.is_empty() {
        println!("0");
        return;
    }

    let mut idx = 0usize;
    while idx < lines.len() && lines[idx].trim().is_empty() {
        idx += 1;
    }
    if idx >= lines.len() {
        println!("0");
        return;
    }

    let hdr: Vec<&str> = lines[idx].split_whitespace().collect();
    idx += 1;
    if hdr.len() < 2 {
        println!("0");
        return;
    }

    let num_holes = hdr[0].parse::<usize>().unwrap_or(0);
    let num_props = hdr[1].parse::<usize>().unwrap_or(0);
    if num_holes == 0 || num_props == 0 {
        println!("0");
        return;
    }

    while idx < lines.len() && lines[idx].trim().is_empty() {
        idx += 1;
    }
    if idx >= lines.len() {
        println!("0");
        return;
    }

    let mut prop_names: Vec<String> = lines[idx]
        .split_whitespace()
        .take(num_props)
        .map(|s| s.to_string())
        .collect();
    idx += 1;
    while prop_names.len() < num_props {
        prop_names.push(format!("prop_{}", prop_names.len()));
    }

    let mut samples: Vec<Sample> = Vec::new();
    let mut seen_keys: HashSet<String> = HashSet::new();

    for hole_i in 0..num_holes {
        while idx < lines.len() && lines[idx].trim().is_empty() {
            idx += 1;
        }
        if idx >= lines.len() {
            break;
        }

        let h: Vec<&str> = lines[idx].split_whitespace().collect();
        idx += 1;
        if h.len() < 9 {
            continue;
        }

        let hole_id = h[0].to_string();
        let sx = h[1].parse::<f64>().unwrap_or(0.0);
        let sy = h[2].parse::<f64>().unwrap_or(0.0);
        let sz = h[3].parse::<f64>().unwrap_or(0.0);
        let dx = h[4].parse::<f64>().unwrap_or(0.0);
        let dy = h[5].parse::<f64>().unwrap_or(0.0);
        let dz = h[6].parse::<f64>().unwrap_or(-1.0);
        let length = h[7].parse::<f64>().unwrap_or(0.0).max(0.0);
        let n_samples = h[8].parse::<usize>().unwrap_or(0);

        let dir = normalize([dx, dy, dz]);

        for s_idx in 0..n_samples {
            if idx >= lines.len() {
                break;
            }
            let row: Vec<&str> = lines[idx].split_whitespace().collect();
            idx += 1;
            if row.is_empty() {
                continue;
            }

            let d_raw = row[0].parse::<f64>().unwrap_or(0.0);
            let d = clamp(d_raw, 0.0, length);
            let p = [sx + dir[0] * d, sy + dir[1] * d, sz + dir[2] * d];

            let mut props = vec![None; num_props];
            for j in 0..num_props {
                props[j] = parse_f64(row.get(j + 1));
            }

            let key = format!("{}|{}", hole_id, s_idx);
            if !seen_keys.insert(key) {
                continue;
            }

            samples.push(Sample {
                hole_idx: hole_i,
                hole_id: hole_id.clone(),
                sample_idx: s_idx,
                point: p,
                props,
            });
        }
    }

    let n = samples.len();
    if n == 0 {
        println!("0");
        return;
    }

    let mut global_stats: Vec<Option<Stats>> = vec![None; num_props];
    for p in 0..num_props {
        let vals: Vec<f64> = samples.iter().filter_map(|s| s.props[p]).collect();
        global_stats[p] = robust_stats(&vals);
    }

    let (k_neighbors, radius_mult, score_thresh, strategy_grid) = if n <= 5000 {
        (24usize, 3.0f64, 3.5f64, false)
    } else if n <= 50000 {
        (20usize, 2.5f64, 4.0f64, true)
    } else {
        (16usize, 2.0f64, 4.5f64, true)
    };

    let mut local_scale = 1.0;
    {
        let probe = n.min(1200);
        let step = (n / probe.max(1)).max(1);
        let mut dists = Vec::new();
        let mut i = 0usize;
        while i < n {
            let mut best = f64::INFINITY;
            for j in 0..n {
                if i == j {
                    continue;
                }
                let d2 = dist2(samples[i].point, samples[j].point);
                if d2 < best {
                    best = d2;
                }
            }
            if best.is_finite() {
                dists.push(best.sqrt());
            }
            i += step;
            if dists.len() >= probe {
                break;
            }
        }
        if let Some(s) = robust_stats(&dists) {
            local_scale = (s.median + s.mad).max(1e-6);
        }
    }
    let radius = local_scale * radius_mult;
    let radius2 = radius * radius;

    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
    let mut mins = [f64::INFINITY; 3];
    if strategy_grid {
        for s in &samples {
            mins[0] = mins[0].min(s.point[0]);
            mins[1] = mins[1].min(s.point[1]);
            mins[2] = mins[2].min(s.point[2]);
        }
        let cell = radius.max(1e-6);
        for (i, s) in samples.iter().enumerate() {
            let key = (
                axis_bin(s.point[0], mins[0], cell),
                axis_bin(s.point[1], mins[1], cell),
                axis_bin(s.point[2], mins[2], cell),
            );
            grid.entry(key).or_default().push(i);
        }
    }

    let mut suspects: Vec<Suspect> = Vec::new();

    for i in 0..n {
        let s = &samples[i];

        let neighbor_indices: Vec<usize> = if strategy_grid {
            let cell = radius.max(1e-6);
            let bx = axis_bin(s.point[0], mins[0], cell);
            let by = axis_bin(s.point[1], mins[1], cell);
            let bz = axis_bin(s.point[2], mins[2], cell);
            let mut cand: Vec<(f64, usize)> = Vec::new();
            for ox in -1..=1 {
                for oy in -1..=1 {
                    for oz in -1..=1 {
                        if let Some(list) = grid.get(&(bx + ox, by + oy, bz + oz)) {
                            for &j in list {
                                if i == j {
                                    continue;
                                }
                                let d2 = dist2(s.point, samples[j].point);
                                if d2 <= radius2 {
                                    cand.push((d2, j));
                                }
                            }
                        }
                    }
                }
            }
            cand.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
            cand.into_iter().take(k_neighbors).map(|x| x.1).collect()
        } else {
            let mut cand: Vec<(f64, usize)> = Vec::new();
            for j in 0..n {
                if i == j {
                    continue;
                }
                let d2 = dist2(s.point, samples[j].point);
                if d2 <= radius2 {
                    cand.push((d2, j));
                }
            }
            cand.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
            cand.into_iter().take(k_neighbors).map(|x| x.1).collect()
        };

        for p in 0..num_props {
            let v = match s.props[p] {
                Some(x) => x,
                None => continue,
            };

            let mut neigh_vals = Vec::new();
            for &j in &neighbor_indices {
                if samples[j].hole_idx == s.hole_idx && (samples[j].sample_idx as isize - s.sample_idx as isize).abs() <= 1 {
                    continue;
                }
                if let Some(nv) = samples[j].props[p] {
                    neigh_vals.push(nv);
                }
            }

            if neigh_vals.len() < 5 {
                continue;
            }

            let local = match robust_stats(&neigh_vals) {
                Some(st) => st,
                None => continue,
            };
            let z_local = (v - local.median).abs() / (1.4826 * local.mad + 1e-9);

            let z_global = if let Some(gs) = global_stats[p] {
                (v - gs.median).abs() / (1.4826 * gs.mad + 1e-9)
            } else {
                0.0
            };

            let score = z_local * 0.75 + z_global * 0.25;
            if score >= score_thresh {
                let confidence = (1.0 - (-0.35 * (score - score_thresh + 1.0)).exp()) * 100.0;
                suspects.push(Suspect {
                    hole_id: s.hole_id.clone(),
                    sample_idx: s.sample_idx,
                    prop_name: prop_names[p].clone(),
                    confidence,
                });
            }
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.prop_name.cmp(&b.prop_name))
    });

    println!("{}", suspects.len());
    for s in suspects {
        println!("{} {} {} {:.3}", s.hole_id, s.sample_idx, s.prop_name, s.confidence);
    }
}
