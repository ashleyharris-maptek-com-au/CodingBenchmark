OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %inColor "inColor"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0_7 = OpConstant %float 0.699999988
%float_3 = OpConstant %float 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_gl_PerVertex_in_3 = OpTypePointer Input %arr_gl_PerVertex_in_3
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_3 Input

%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3float_3 = OpTypePointer Input %arr_v3float_3
%inColor = OpVariable %ptr_Input_arr_v3float_3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output

%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output

%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%c = OpVectorTimesScalar %v4float %sum012 %float_0_333333343

%d0 = OpFSub %v4float %p0 %c
%sd0 = OpVectorTimesScalar %v4float %d0 %float_0_7
%np0 = OpFAdd %v4float %c %sd0

%d1 = OpFSub %v4float %p1 %c
%sd1 = OpVectorTimesScalar %v4float %d1 %float_0_7
%np1 = OpFAdd %v4float %c %sd1

%d2 = OpFSub %v4float %p2 %c
%sd2 = OpVectorTimesScalar %v4float %d2 %float_0_7
%np2 = OpFAdd %v4float %c %sd2

%c0_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%out_pos_ptr0 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr0 %np0
OpStore %outColor %c0
OpEmitVertex

%c1_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr
%out_pos_ptr1 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr1 %np1
OpStore %outColor %c1
OpEmitVertex

%c2_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_2
%c2 = OpLoad %v3float %c2_ptr
%out_pos_ptr2 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr2 %np2
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
