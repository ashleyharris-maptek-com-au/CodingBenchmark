use std::collections::VecDeque;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }

    let mut lines = input.lines();
    let first = match lines.next() {
        Some(s) => s,
        None => {
            println!("0");
            return;
        }
    };

    let mut hw = first.split_whitespace();
    let h: usize = match hw.next().and_then(|x| x.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let w: usize = match hw.next().and_then(|x| x.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    let n = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut grid = vec![b'#'; n];
    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    for r in 0..h {
        let row = lines.next().unwrap_or("");
        let bytes = row.as_bytes();
        for c in 0..w {
            let ch = if c < bytes.len() { bytes[c] } else { b'#' };
            let idx = r * w + c;
            grid[idx] = ch;
            if ch == b'A' {
                start = Some(idx);
            } else if ch == b'B' {
                goal = Some(idx);
            }
        }
    }

    let s = match start {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let t = match goal {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if s == t {
        let x = s / w;
        let y = s % w;
        println!("1");
        println!("{} {}", x, y);
        return;
    }

    let mut parent = vec![usize::MAX; n];
    let mut q = VecDeque::new();

    parent[s] = s;
    q.push_back(s);

    while let Some(cur) = q.pop_front() {
        if cur == t {
            break;
        }

        let r = cur / w;
        let c = cur % w;

        if r > 0 {
            let nxt = cur - w;
            if parent[nxt] == usize::MAX && grid[nxt] != b'#' {
                parent[nxt] = cur;
                q.push_back(nxt);
            }
        }
        if r + 1 < h {
            let nxt = cur + w;
            if parent[nxt] == usize::MAX && grid[nxt] != b'#' {
                parent[nxt] = cur;
                q.push_back(nxt);
            }
        }
        if c > 0 {
            let nxt = cur - 1;
            if parent[nxt] == usize::MAX && grid[nxt] != b'#' {
                parent[nxt] = cur;
                q.push_back(nxt);
            }
        }
        if c + 1 < w {
            let nxt = cur + 1;
            if parent[nxt] == usize::MAX && grid[nxt] != b'#' {
                parent[nxt] = cur;
                q.push_back(nxt);
            }
        }
    }

    if parent[t] == usize::MAX {
        println!("0");
        return;
    }

    let mut path = Vec::new();
    let mut cur = t;
    while cur != s {
        path.push(cur);
        cur = parent[cur];
    }
    path.push(s);
    path.reverse();

    let mut out = String::new();
    out.push_str(&path.len().to_string());
    out.push('\n');
    for idx in path {
        let x = idx / w;
        let y = idx % w;
        out.push_str(&x.to_string());
        out.push(' ');
        out.push_str(&y.to_string());
        out.push('\n');
    }

    print!("{}", out);
}
