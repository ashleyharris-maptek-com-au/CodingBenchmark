OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_0_5 = OpConstant %float 0.5
%float_0_02 = OpConstant %float 0.02
%float_0_25 = OpConstant %float 0.25

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_func_v4 = OpTypePointer Function %v4float
%ptr_func_v3 = OpTypePointer Function %v3float
%ptr_func_v2 = OpTypePointer Function %v2float
%ptr_func_f = OpTypePointer Function %float
%ptr_func_i = OpTypePointer Function %int

%main = OpFunction %void None %func
%entry = OpLabel
%p0 = OpVariable %ptr_func_v4 Function
%p1 = OpVariable %ptr_func_v4 Function
%p2 = OpVariable %ptr_func_v4 Function
%p3 = OpVariable %ptr_func_v4 Function
%c1 = OpVariable %ptr_func_v3 Function
%c2 = OpVariable %ptr_func_v3 Function
%t0 = OpVariable %ptr_func_v2 Function
%t1 = OpVariable %ptr_func_v2 Function
%i = OpVariable %ptr_func_i Function

%ac_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%ld_p0 = OpLoad %v4float %ac_p0
OpStore %p0 %ld_p0
%ac_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%ld_p1 = OpLoad %v4float %ac_p1
OpStore %p1 %ld_p1
%ac_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%ld_p2 = OpLoad %v4float %ac_p2
OpStore %p2 %ld_p2
%ac_p3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%ld_p3 = OpLoad %v4float %ac_p3
OpStore %p3 %ld_p3

%ac_c1 = OpAccessChain %ptr_in_v3 %color %int_1
%ld_c1 = OpLoad %v3float %ac_c1
OpStore %c1 %ld_c1
%ac_c2 = OpAccessChain %ptr_in_v3 %color %int_2
%ld_c2 = OpLoad %v3float %ac_c2
OpStore %c2 %ld_c2

%p2v = OpLoad %v4float %p2
%p0v = OpLoad %v4float %p0
%d0 = OpFSub %v4float %p2v %p0v
%d0xy = OpVectorShuffle %v2float %d0 %d0 0 1
%p3v = OpLoad %v4float %p3
%p1v = OpLoad %v4float %p1
%d1 = OpFSub %v4float %p3v %p1v
%d1xy = OpVectorShuffle %v2float %d1 %d1 0 1
OpStore %t0 %d0xy
OpStore %t1 %d1xy

OpStore %i %int_0
OpBranch %loop

%loop = OpLabel
%iv = OpLoad %int %i
%cond = OpSLessThanEqual %bool %iv %int_4
OpBranchConditional %cond %body %done

%body = OpLabel
%if = OpConvertSToF %float %iv
%tf = OpFMul %float %if %float_0_25
%one_minus_t = OpFSub %float %float_1 %tf

%p1c = OpLoad %v4float %p1
%p2c = OpLoad %v4float %p2
%p1xy = OpVectorShuffle %v2float %p1c %p1c 0 1
%p2xy = OpVectorShuffle %v2float %p2c %p2c 0 1
%a = OpVectorTimesScalar %v2float %p1xy %one_minus_t
%b = OpVectorTimesScalar %v2float %p2xy %tf
%base = OpFAdd %v2float %a %b

%t0v = OpLoad %v2float %t0
%t1v = OpLoad %v2float %t1
%ta = OpVectorTimesScalar %v2float %t0v %one_minus_t
%tb = OpVectorTimesScalar %v2float %t1v %tf
%tan = OpFAdd %v2float %ta %tb
%tx = OpCompositeExtract %float %tan 0
%ty = OpCompositeExtract %float %tan 1
%nx0 = OpFNegate %float %ty
%ny0 = OpCopyObject %float %tx
%nn = OpCompositeConstruct %v2float %nx0 %ny0
%len2 = OpDot %float %nn %nn
%len = OpExtInst %float %1 Sqrt %len2
%nrm = OpFDiv %v2float %nn %len
%off = OpVectorTimesScalar %v2float %nrm %float_0_02

%px = OpCompositeExtract %float %base 0
%py = OpCompositeExtract %float %base 1
%ox = OpCompositeExtract %float %off 0
%oy = OpCompositeExtract %float %off 1
%plusx = OpFAdd %float %px %ox
%plusy = OpFAdd %float %py %oy
%minusx = OpFSub %float %px %ox
%minusy = OpFSub %float %py %oy

%p1z = OpCompositeExtract %float %p1c 2
%p2z = OpCompositeExtract %float %p2c 2
%p1w = OpCompositeExtract %float %p1c 3
%p2w = OpCompositeExtract %float %p2c 3
%z1 = OpFMul %float %p1z %one_minus_t
%z2 = OpFMul %float %p2z %tf
%zw = OpFAdd %float %z1 %z2
%w1 = OpFMul %float %p1w %one_minus_t
%w2 = OpFMul %float %p2w %tf
%ww = OpFAdd %float %w1 %w2

%c1v = OpLoad %v3float %c1
%c2v = OpLoad %v3float %c2
%cc1 = OpVectorTimesScalar %v3float %c1v %one_minus_t
%cc2 = OpVectorTimesScalar %v3float %c2v %tf
%cc = OpFAdd %v3float %cc1 %cc2
OpStore %outColor %cc

%vplus = OpCompositeConstruct %v4float %plusx %plusy %zw %ww
%ac_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ac_out %vplus
OpEmitVertex

OpStore %outColor %cc
%vminus = OpCompositeConstruct %v4float %minusx %minusy %zw %ww
%ac_out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ac_out2 %vminus
OpEmitVertex

%inext = OpIAdd %int %iv %int_1
OpStore %i %inext
OpBranch %loop

%done = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
