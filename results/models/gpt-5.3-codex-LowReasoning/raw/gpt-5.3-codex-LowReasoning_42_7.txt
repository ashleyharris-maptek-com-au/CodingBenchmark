OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

%1 = OpExtInstImport "GLSL.std.450"
OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %ubo_t "ubo_t"
OpName %ubo "ubo"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo_t Block
OpMemberDecorate %ubo_t 0 ColMajor
OpMemberDecorate %ubo_t 0 Offset 0
OpMemberDecorate %ubo_t 0 MatrixStride 16
OpMemberDecorate %ubo_t 1 ColMajor
OpMemberDecorate %ubo_t 1 Offset 64
OpMemberDecorate %ubo_t 1 MatrixStride 16
OpMemberDecorate %ubo_t 2 ColMajor
OpMemberDecorate %ubo_t 2 Offset 128
OpMemberDecorate %ubo_t 2 MatrixStride 16
OpMemberDecorate %ubo_t 3 Offset 192
OpMemberDecorate %ubo_t 4 Offset 208
OpMemberDecorate %ubo_t 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0
%float_3 = OpConstant %float 3
%float_0_08 = OpConstant %float 0.08

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%mat4 = OpTypeMatrix %v4float 4
%ubo_t = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %ubo_t
%ubo = OpVariable %ptr_ubo Uniform

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%main = OpFunction %void None %3
%5 = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

%sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
%sum012 = OpFAdd %v3float %sum01 %p2_xyz
%center = OpVectorTimesScalar %v3float %sum012 %float_3
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%cross = OpExtInst %v3float %1 Cross %e1 %e2
%N = OpExtInst %v3float %1 Normalize %cross
%offset = OpVectorTimesScalar %v3float %N %float_0_08
%ox = OpCompositeExtract %float %offset 0
%oy = OpCompositeExtract %float %offset 1
%oz = OpCompositeExtract %float %offset 2
%offset4 = OpCompositeConstruct %v4float %ox %oy %oz %float_0

%p0_shift = OpFAdd %v4float %p0 %offset4
%out_pos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr0 %p0_shift
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
OpStore %outColor %c0
OpEmitVertex

%p1_shift = OpFAdd %v4float %p1 %offset4
%out_pos_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr1 %p1_shift
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr
OpStore %outColor %c1
OpEmitVertex

%p2_shift = OpFAdd %v4float %p2 %offset4
%out_pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr2 %p2_shift
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
