const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const probability = data.probability || [];
const drones = data.drones || [];
const maxStepsPerDrone = Math.max(0, data.maxStepsPerDrone | 0);

const h = probability.length;
const w = h > 0 ? probability[0].length : 0;

function inBounds(x, y) {
  return x >= 0 && x < w && y >= 0 && y < h;
}

function isFree(x, y) {
  return inBounds(x, y) && probability[y][x] !== -1;
}

const moves = [
  [0, 0],
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

// Shared coverage set (unique cells across all drones)
const covered = new Uint8Array(w * h);

function idx(x, y) {
  return y * w + x;
}

function cellValue(x, y) {
  const v = probability[y][x];
  return v > 0 ? v : 0;
}

function planForDrone(startX, startY, budget) {
  const path = [];
  if (!isFree(startX, startY)) return path;

  let x = startX;
  let y = startY;
  path.push([x, y]);

  const startI = idx(x, y);
  if (!covered[startI]) covered[startI] = 1;

  // Radius-limited local lookahead for scalable performance.
  // Tries to move toward nearby high-value uncovered cells while allowing transit.
  const LOOK_R = 6;
  const DECAY = 0.86;
  const VISIT_PENALTY = 0.35;
  const LOCAL_W = 0.2;

  // Per-drone visit frequency to reduce oscillation.
  const seen = new Uint16Array(w * h);
  seen[startI] = 1;

  for (let step = 0; step < budget; step++) {
    let bestScore = -Infinity;
    let bestNx = x;
    let bestNy = y;

    // Evaluate each immediate move by running a bounded BFS around candidate position.
    for (let m = 0; m < moves.length; m++) {
      const nx = x + moves[m][0];
      const ny = y + moves[m][1];
      if (!isFree(nx, ny)) continue;

      const baseI = idx(nx, ny);
      let score = 0;

      // Immediate gain if this move lands on uncovered value.
      if (!covered[baseI]) score += cellValue(nx, ny) * 2.0;

      // Small local preference, even if already covered, for high-probability terrain.
      score += cellValue(nx, ny) * LOCAL_W;

      // Penalty for revisiting same cells too often by this drone.
      score -= seen[baseI] * VISIT_PENALTY;

      // Bounded BFS lookahead around (nx, ny)
      const qx = new Int32Array((2 * LOOK_R + 1) * (2 * LOOK_R + 1) + 8);
      const qy = new Int32Array(qx.length);
      const qd = new Int16Array(qx.length);
      let head = 0;
      let tail = 0;

      // Stamp-based visited to avoid clearing full arrays each time.
      const mark = new Uint8Array(qx.length * 2 + 16);
      // For simplicity and speed under small radius, use local hash set via objectless key array.
      // Keys are packed offsets in a local square.
      const side = 2 * LOOK_R + 1;
      const localVisited = new Uint8Array(side * side);

      qx[tail] = nx;
      qy[tail] = ny;
      qd[tail] = 0;
      tail++;
      localVisited[LOOK_R * side + LOOK_R] = 1;

      while (head < tail) {
        const cx = qx[head];
        const cy = qy[head];
        const cd = qd[head];
        head++;

        if (cd >= LOOK_R) continue;

        for (let k = 1; k < 5; k++) {
          const tx = cx + moves[k][0];
          const ty = cy + moves[k][1];
          if (!isFree(tx, ty)) continue;

          const ox = tx - nx + LOOK_R;
          const oy = ty - ny + LOOK_R;
          if (ox < 0 || oy < 0 || ox >= side || oy >= side) continue;
          const lidx = oy * side + ox;
          if (localVisited[lidx]) continue;
          localVisited[lidx] = 1;

          const nd = cd + 1;
          qx[tail] = tx;
          qy[tail] = ty;
          qd[tail] = nd;
          tail++;

          const ti = idx(tx, ty);
          const v = cellValue(tx, ty);
          if (v > 0) {
            const novelty = covered[ti] ? 0.2 : 1.0;
            score += v * novelty * Math.pow(DECAY, nd);
          }
        }
      }

      if (score > bestScore) {
        bestScore = score;
        bestNx = nx;
        bestNy = ny;
      }
    }

    x = bestNx;
    y = bestNy;
    path.push([x, y]);

    const ii = idx(x, y);
    if (!covered[ii]) covered[ii] = 1;
    if (seen[ii] < 65535) seen[ii]++;
  }

  return path;
}

const paths = [];
for (let i = 0; i < drones.length; i++) {
  const d = drones[i] || {};
  const sx = d.x | 0;
  const sy = d.y | 0;
  paths.push(planForDrone(sx, sy, maxStepsPerDrone));
}

process.stdout.write(JSON.stringify({ paths }));
