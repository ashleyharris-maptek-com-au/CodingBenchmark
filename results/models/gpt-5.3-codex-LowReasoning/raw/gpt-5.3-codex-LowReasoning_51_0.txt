from math import atan2, cos, sin

# Persistent AP state
_AP = {
    "alt_i": 0.0,
    "spd_i": 0.0,
    "hdg_i": 0.0,
    "yaw_i": 0.0,
    "trim": 0.0,
    "last_hdg_err": 0.0,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _wrap180(deg):
    x = (deg + 180.0) % 360.0 - 180.0
    return x


def _median(vals):
    vals = sorted(vals)
    n = len(vals)
    if n == 0:
        return None
    if n % 2:
        return vals[n // 2]
    return 0.5 * (vals[n // 2 - 1] + vals[n // 2])


def _robust_triplet(state, base, unit="", plausible=None, max_pair_diff=None, fallback=None):
    cands = []
    for i in (1, 2, 3):
        k = f"{base}_{i}{unit}"
        v = state.get(k)
        if isinstance(v, (int, float)):
            if plausible is None or (plausible[0] <= v <= plausible[1]):
                cands.append(v)
    if not cands:
        return fallback
    if len(cands) == 1:
        return cands[0]
    if max_pair_diff is not None:
        best = None
        best_score = 1e30
        for v in cands:
            score = sum(abs(v - u) for u in cands)
            if score < best_score:
                best_score = score
                best = v
        if len(cands) == 2 and abs(cands[0] - cands[1]) > max_pair_diff:
            return fallback if fallback is not None else best
        return best
    return _median(cands)


def autopilot_step(state: dict, dt: float) -> dict:
    global _AP

    dt = float(state.get("dt", dt) or dt or 0.05)
    dt = _clamp(dt, 0.01, 0.2)

    # Fused sensors
    alt_baro = _robust_triplet(state, "baro_alt", "_ft", plausible=(-2000, 60000), max_pair_diff=1200)
    alt_radar = _robust_triplet(state, "radar_alt", "_ft", plausible=(-100, 5000), max_pair_diff=300)
    alt_gps = _robust_triplet(state, "gps", "_alt_ft", plausible=(-2000, 60000), max_pair_diff=1500)

    ias = _robust_triplet(state, "ias", "_kt", plausible=(30, 450), max_pair_diff=40)
    mach = _median([state.get("mach_1"), state.get("mach_2"), state.get("mach_3")])
    aoa = _robust_triplet(state, "aoa", "_deg", plausible=(-10, 35), max_pair_diff=6, fallback=5.0)

    hdg = _robust_triplet(state, "irs", "_heading_deg", plausible=(0, 360), max_pair_diff=25)
    track = _robust_triplet(state, "irs", "_track_deg", plausible=(0, 360), max_pair_diff=25)
    roll = _robust_triplet(state, "irs", "_roll_deg", plausible=(-80, 80), max_pair_diff=10, fallback=0.0)
    pitch = _robust_triplet(state, "irs", "_pitch_deg", plausible=(-25, 25), max_pair_diff=8, fallback=0.0)
    vs = _robust_triplet(state, "irs", "_vs_fpm", plausible=(-8000, 8000), max_pair_diff=1500)

    yaw_rate = _robust_triplet(state, "gyro", "_yaw_rate_dps", plausible=(-30, 30), max_pair_diff=6, fallback=0.0)
    roll_rate = _robust_triplet(state, "gyro", "_roll_rate_dps", plausible=(-60, 60), max_pair_diff=10, fallback=0.0)
    pitch_rate = _robust_triplet(state, "gyro", "_pitch_rate_dps", plausible=(-30, 30), max_pair_diff=6, fallback=0.0)

    # Secondary fallbacks
    if vs is None:
        vs = float(state.get("vertical_speed_fpm", 0.0) or 0.0)
    if hdg is None:
        hdg = float(state.get("magnetic_heading_deg", state.get("true_heading_deg", 0.0)) or 0.0)
    if track is None:
        track = float(state.get("track_deg", hdg) or hdg)
    if ias is None:
        ias = float(state.get("target_ias_kt", 250.0) or 250.0)
    if alt_baro is None and alt_gps is not None:
        alt_baro = alt_gps
    if alt_baro is None:
        alt_baro = float(state.get("target_alt_ft", 0.0) or 0.0)

    # Smart altitude pick near ground (prefer radar), otherwise baro/GPS consensus
    if alt_radar is not None and alt_radar < 2500:
        alt = alt_radar
    elif alt_baro is not None and alt_gps is not None and abs(alt_baro - alt_gps) < 1000:
        alt = 0.7 * alt_baro + 0.3 * alt_gps
    else:
        alt = alt_baro if alt_baro is not None else (alt_gps if alt_gps is not None else 0.0)

    # Targets
    target_alt = float(state.get("target_alt_ft", alt))
    target_hdg = float(state.get("target_hdg_deg", hdg)) % 360.0
    target_ias = float(state.get("target_ias_kt", ias))

    # Envelope/protection
    stall_aoa = aoa > 13.5
    low_speed = ias < 135
    high_speed = (ias > 345) or (isinstance(mach, (int, float)) and mach > 0.80)

    # Lateral: heading + track blend, bank-command to aileron
    hdg_ref = 0.6 * hdg + 0.4 * track
    hdg_err = _wrap180(target_hdg - hdg_ref)
    _AP["hdg_i"] = _clamp(_AP["hdg_i"] + hdg_err * dt, -60, 60)
    bank_cmd = _clamp(0.18 * hdg_err + 0.010 * _AP["hdg_i"], -25, 25)
    if low_speed:
        bank_cmd = _clamp(bank_cmd, -15, 15)

    roll_err = bank_cmd - roll
    aileron = _clamp(0.035 * roll_err - 0.010 * roll_rate, -1.0, 1.0)

    # Rudder yaw damper + turn coordination
    rudder = _clamp(-0.055 * yaw_rate + 0.0025 * roll, -0.45, 0.45)

    # Vertical: altitude -> VS command -> pitch/elevator
    alt_err = _clamp(target_alt - alt, -6000, 6000)
    vs_cmd = _clamp(0.09 * alt_err, -2200, 2200)
    if low_speed:
        vs_cmd = min(vs_cmd, 800)
    if high_speed:
        vs_cmd = max(vs_cmd, -1200)

    vs_err = vs_cmd - vs
    pitch_cmd = _clamp(0.0048 * vs_err, -10, 12)
    if stall_aoa:
        pitch_cmd = min(pitch_cmd, -2.0)

    pitch_err = pitch_cmd - pitch
    elevator = _clamp(0.085 * pitch_err - 0.022 * pitch_rate, -1.0, 1.0)

    # Autothrottle / throttle split for asymmetry damping
    spd_err = _clamp(target_ias - ias, -80, 80)
    _AP["spd_i"] = _clamp(_AP["spd_i"] + spd_err * dt, -100, 100)

    base_thr = 0.50 + 0.0065 * spd_err + 0.0009 * _AP["spd_i"] + 0.00006 * max(0.0, vs_cmd)
    if high_speed:
        base_thr -= 0.20
    if low_speed or stall_aoa:
        base_thr += 0.20

    eng1_run = bool(state.get("engine_1_running", True)) and not bool(state.get("engine_1_fire", False))
    eng2_run = bool(state.get("engine_2_running", True)) and not bool(state.get("engine_2_fire", False))

    thr1 = base_thr
    thr2 = base_thr

    if eng1_run and not eng2_run:
        thr2 = 0.0
        rudder += 0.18
    elif eng2_run and not eng1_run:
        thr1 = 0.0
        rudder -= 0.18

    # Trim: slowly remove steady elevator bias
    _AP["trim"] = _clamp(_AP["trim"] + 0.16 * elevator * dt, -1.0, 1.0)
    elevator_trim = _AP["trim"]

    # Config automation
    wow = bool(state.get("weight_on_wheels_left", False) or state.get("weight_on_wheels_right", False) or state.get("weight_on_wheels_nose", False))
    gear = 1 if (wow or alt < 1200 or target_alt < 1500) else 0

    if ias < 150:
        flaps = 15
    elif ias < 190:
        flaps = 5
    else:
        flaps = 0
    if high_speed:
        flaps = 0

    speedbrake = 0.0
    if high_speed and alt_err < -300:
        speedbrake = 0.35

    # Final clamp
    thr1 = _clamp(thr1, 0.0, 1.0)
    thr2 = _clamp(thr2, 0.0, 1.0)
    rudder = _clamp(rudder, -1.0, 1.0)

    return {
        "elevator": elevator,
        "aileron": aileron,
        "rudder": rudder,
        "throttle_1": thr1,
        "throttle_2": thr2,
        "elevator_trim": elevator_trim,
        "flaps": int(flaps),
        "speedbrake": speedbrake,
        "gear": int(gear),
    }
