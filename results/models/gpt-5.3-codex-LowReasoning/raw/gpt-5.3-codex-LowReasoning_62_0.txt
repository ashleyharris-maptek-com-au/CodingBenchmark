const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid;
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = Math.max(0, data.cutBudget | 0);

const n = width * height;
const idx = (x, y) => y * width + x;
const xOf = (i) => i % width;
const yOf = (i) => (i / width) | 0;

const burnable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) burnable[idx(x, y)] = 1; // '.'
  }
}

const isIgnition = new Uint8Array(n);
for (const p of ignitions) {
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (x >= 0 && x < width && y >= 0 && y < height) {
    isIgnition[idx(x, y)] = 1;
  }
}

const isTown = new Uint8Array(n);
const townValueAt = new Float64Array(n);
for (const t of towns) {
  const x = t.x | 0;
  const y = t.y | 0;
  if (x >= 0 && x < width && y >= 0 && y < height) {
    const i = idx(x, y);
    isTown[i] = 1;
    townValueAt[i] += Number(t.value) || 0;
  }
}

// Multi-source BFS for shortest fire-arrival distances and one predecessor tree.
const dist = new Int32Array(n);
for (let i = 0; i < n; i++) dist[i] = -1;
const parent = new Int32Array(n);
for (let i = 0; i < n; i++) parent[i] = -1;

const q = new Int32Array(n);
let qh = 0;
let qt = 0;

for (const p of ignitions) {
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const s = idx(x, y);
  if (!burnable[s]) continue;
  if (dist[s] !== -1) continue;
  dist[s] = 0;
  parent[s] = -1;
  q[qt++] = s;
}

while (qh < qt) {
  const u = q[qh++];
  const ux = xOf(u);
  const uy = yOf(u);
  const nd = dist[u] + 1;

  let v;
  if (ux > 0) {
    v = u - 1;
    if (burnable[v] && dist[v] === -1) {
      dist[v] = nd;
      parent[v] = u;
      q[qt++] = v;
    }
  }
  if (ux + 1 < width) {
    v = u + 1;
    if (burnable[v] && dist[v] === -1) {
      dist[v] = nd;
      parent[v] = u;
      q[qt++] = v;
    }
  }
  if (uy > 0) {
    v = u - width;
    if (burnable[v] && dist[v] === -1) {
      dist[v] = nd;
      parent[v] = u;
      q[qt++] = v;
    }
  }
  if (uy + 1 < height) {
    v = u + width;
    if (burnable[v] && dist[v] === -1) {
      dist[v] = nd;
      parent[v] = u;
      q[qt++] = v;
    }
  }
}

// Build child adjacency for BFS predecessor forest.
const childCount = new Int32Array(n);
for (let i = 0; i < n; i++) {
  const p = parent[i];
  if (p >= 0) childCount[p]++;
}

const offsets = new Int32Array(n + 1);
for (let i = 0; i < n; i++) offsets[i + 1] = offsets[i] + childCount[i];
const fill = new Int32Array(offsets);
const children = new Int32Array(offsets[n]);
for (let i = 0; i < n; i++) {
  const p = parent[i];
  if (p >= 0) {
    const pos = fill[p]++;
    children[pos] = i;
  }
}

// Postorder over predecessor forest to compute subtree town values.
const order = new Int32Array(qt);
for (let i = 0; i < qt; i++) order[i] = q[i];

const subtree = new Float64Array(n);
for (let i = 0; i < qt; i++) {
  const u = order[qt - 1 - i];
  let total = townValueAt[u];
  for (let p = offsets[u]; p < offsets[u + 1]; p++) total += subtree[children[p]];
  subtree[u] = total;
}

// Euler tour indices to quickly skip descendants of chosen cuts.
const tin = new Int32Array(n);
const tout = new Int32Array(n);
for (let i = 0; i < n; i++) {
  tin[i] = -1;
  tout[i] = -1;
}
let timer = 0;
const stackNode = new Int32Array(qt * 2 + 1);
const stackState = new Int8Array(qt * 2 + 1);
let sp = 0;
for (let i = 0; i < qt; i++) {
  const r = order[i];
  if (parent[r] !== -1 || tin[r] !== -1) continue;
  stackNode[sp] = r;
  stackState[sp] = 0;
  sp++;
  while (sp > 0) {
    sp--;
    const u = stackNode[sp];
    const st = stackState[sp];
    if (st === 0) {
      tin[u] = timer++;
      stackNode[sp] = u;
      stackState[sp] = 1;
      sp++;
      for (let p = offsets[u + 1] - 1; p >= offsets[u]; p--) {
        const v = children[p];
        stackNode[sp] = v;
        stackState[sp] = 0;
        sp++;
      }
    } else {
      tout[u] = timer;
    }
  }
}

const candidates = [];
for (let i = 0; i < n; i++) {
  if (!burnable[i]) continue;
  if (dist[i] < 0) continue;
  if (isIgnition[i]) continue;
  if (isTown[i]) continue;
  const gain = subtree[i];
  if (gain > 0) {
    candidates.push({ i, gain, d: dist[i], tin: tin[i], tout: tout[i] });
  }
}

candidates.sort((a, b) => {
  if (b.gain !== a.gain) return b.gain - a.gain;
  return b.d - a.d; // prefer later cut points for locality/flexibility
});

const selected = [];
const selectedIntervals = [];

for (let c = 0; c < candidates.length && selected.length < cutBudget; c++) {
  const cand = candidates[c];
  let covered = false;
  for (let j = 0; j < selectedIntervals.length; j++) {
    const it = selectedIntervals[j];
    if (cand.tin >= it.tin && cand.tin < it.tout) {
      covered = true;
      break;
    }
  }
  if (covered) continue;
  selected.push([xOf(cand.i), yOf(cand.i)]);
  selectedIntervals.push({ tin: cand.tin, tout: cand.tout });
}

process.stdout.write(JSON.stringify({ firebreaks: selected }));
