use std::cmp::Reverse;
use std::collections::BinaryHeap;
use std::io::{self, Read};

fn parse_input() -> Result<(usize, Vec<(usize, usize)>), String> {
    let mut input = String::new();
    io::stdin()
        .read_to_string(&mut input)
        .map_err(|e| format!("failed to read stdin: {e}"))?;

    let mut it = input.split_whitespace();
    let n: usize = it
        .next()
        .ok_or_else(|| "missing num_vertices".to_string())?
        .parse()
        .map_err(|e| format!("invalid num_vertices: {e}"))?;
    let m: usize = it
        .next()
        .ok_or_else(|| "missing num_edges".to_string())?
        .parse()
        .map_err(|e| format!("invalid num_edges: {e}"))?;

    let mut edges = Vec::with_capacity(m);
    for i in 0..m {
        let u: usize = it
            .next()
            .ok_or_else(|| format!("missing u for edge {i}"))?
            .parse()
            .map_err(|e| format!("invalid u for edge {i}: {e}"))?;
        let v: usize = it
            .next()
            .ok_or_else(|| format!("missing v for edge {i}"))?
            .parse()
            .map_err(|e| format!("invalid v for edge {i}: {e}"))?;
        if u >= n || v >= n {
            return Err(format!("edge {i} has vertex out of range: ({u}, {v}) with n={n}"));
        }
        if u != v {
            edges.push((u, v));
        }
    }

    Ok((n, edges))
}

fn add_vertex_to_cover(
    v: usize,
    in_cover: &mut [bool],
    alive_edge: &mut [bool],
    degree: &mut [usize],
    adj: &[Vec<usize>],
    edges: &[(usize, usize)],
    uncovered_count: &mut usize,
) {
    if in_cover[v] {
        return;
    }
    in_cover[v] = true;

    for &eid in &adj[v] {
        if !alive_edge[eid] {
            continue;
        }
        alive_edge[eid] = false;
        *uncovered_count -= 1;

        let (a, b) = edges[eid];
        if degree[a] > 0 {
            degree[a] -= 1;
        }
        if degree[b] > 0 {
            degree[b] -= 1;
        }
    }
}

fn approx_vertex_cover(n: usize, edges: &[(usize, usize)]) -> Vec<usize> {
    let m = edges.len();
    if m == 0 {
        return Vec::new();
    }

    let mut adj = vec![Vec::<usize>::new(); n];
    for (eid, &(u, v)) in edges.iter().enumerate() {
        adj[u].push(eid);
        adj[v].push(eid);
    }

    let mut in_cover = vec![false; n];
    let mut alive_edge = vec![true; m];
    let mut degree = vec![0usize; n];
    for &(u, v) in edges {
        degree[u] += 1;
        degree[v] += 1;
    }

    let mut uncovered_count = m;

    let mut heap: BinaryHeap<(usize, usize)> = BinaryHeap::with_capacity(n);
    for (v, &d) in degree.iter().enumerate() {
        if d > 0 {
            heap.push((d, v));
        }
    }

    while uncovered_count > 0 {
        let mut picked: Option<usize> = None;

        while let Some((d, v)) = heap.pop() {
            if in_cover[v] {
                continue;
            }
            if degree[v] != d {
                continue;
            }
            if d == 0 {
                continue;
            }
            picked = Some(v);
            break;
        }

        if let Some(v) = picked {
            add_vertex_to_cover(
                v,
                &mut in_cover,
                &mut alive_edge,
                &mut degree,
                &adj,
                edges,
                &mut uncovered_count,
            );

            for &eid in &adj[v] {
                let (a, b) = edges[eid];
                if !in_cover[a] && degree[a] > 0 {
                    heap.push((degree[a], a));
                }
                if !in_cover[b] && degree[b] > 0 {
                    heap.push((degree[b], b));
                }
            }
        } else {
            let mut found = None;
            for (eid, &alive) in alive_edge.iter().enumerate() {
                if alive {
                    found = Some(eid);
                    break;
                }
            }

            if let Some(eid) = found {
                let (u, v) = edges[eid];
                let pick = if degree[u] >= degree[v] { u } else { v };
                add_vertex_to_cover(
                    pick,
                    &mut in_cover,
                    &mut alive_edge,
                    &mut degree,
                    &adj,
                    edges,
                    &mut uncovered_count,
                );
            } else {
                break;
            }
        }
    }

    let mut cover = Vec::new();
    for (v, &chosen) in in_cover.iter().enumerate() {
        if chosen {
            cover.push(v);
        }
    }

    // Local improvement: remove redundant vertices while preserving edge coverage.
    // Track how many cover-endpoints each edge currently has.
    let mut cover_endpoints = vec![0u8; m];
    for (eid, &(u, v)) in edges.iter().enumerate() {
        let mut c = 0u8;
        if in_cover[u] {
            c += 1;
        }
        if in_cover[v] {
            c += 1;
        }
        cover_endpoints[eid] = c;
    }

    let mut order: Vec<usize> = cover.clone();
    order.sort_by_key(|&v| Reverse(adj[v].len()));

    for v in order {
        if !in_cover[v] {
            continue;
        }

        let mut removable = true;
        for &eid in &adj[v] {
            if cover_endpoints[eid] <= 1 {
                removable = false;
                break;
            }
        }

        if removable {
            in_cover[v] = false;
            for &eid in &adj[v] {
                cover_endpoints[eid] -= 1;
            }
        }
    }

    cover.clear();
    for (v, &chosen) in in_cover.iter().enumerate() {
        if chosen {
            cover.push(v);
        }
    }

    cover
}

fn main() {
    let (n, edges) = match parse_input() {
        Ok(x) => x,
        Err(_) => {
            println!("0");
            println!();
            return;
        }
    };

    let cover = approx_vertex_cover(n, &edges);
    println!("{}", cover.len());
    if cover.is_empty() {
        println!();
    } else {
        for (i, v) in cover.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    }
}
