import math
import sys


def dist(a, b):
    dx = a[0] - b[0]
    dy = a[1] - b[1]
    return math.hypot(dx, dy)


def route_cost(route, coords, depot):
    if not route:
        return 0.0
    total = dist(depot, coords[route[0]])
    for i in range(len(route) - 1):
        total += dist(coords[route[i]], coords[route[i + 1]])
    total += dist(coords[route[-1]], depot)
    return total


def total_cost(routes, coords, depot):
    return sum(route_cost(r, coords, depot) for r in routes)


def nearest_neighbor_seed(customers, demands, coords, depot, cap):
    unassigned = set(customers)
    routes = []

    while unassigned:
        route = []
        load = 0
        current = None  # None means depot

        while True:
            best = None
            best_d = float('inf')
            curr_pt = depot if current is None else coords[current]

            for c in unassigned:
                if load + demands[c] <= cap:
                    d = dist(curr_pt, coords[c])
                    if d < best_d:
                        best_d = d
                        best = c

            if best is None:
                break

            route.append(best)
            unassigned.remove(best)
            load += demands[best]
            current = best

        if not route:
            c = next(iter(unassigned))
            route = [c]
            unassigned.remove(c)

        routes.append(route)

    return routes


def two_opt(route, coords, depot):
    n = len(route)
    if n < 4:
        return route

    improved = True
    while improved:
        improved = False
        for i in range(n - 2):
            for j in range(i + 2, n):
                a = depot if i == 0 else coords[route[i - 1]]
                b = coords[route[i]]
                c = coords[route[j - 1]]
                d = depot if j == n else coords[route[j]]

                old = dist(a, b) + dist(c, d)
                new = dist(a, c) + dist(b, d)
                if new + 1e-12 < old:
                    route[i:j] = reversed(route[i:j])
                    improved = True
        
    return route


def improve_relocate(routes, loads, demands, coords, depot, cap):
    improved = True
    while improved:
        improved = False
        base_cost = total_cost(routes, coords, depot)

        for r1 in range(len(routes)):
            for i in range(len(routes[r1])):
                node = routes[r1][i]
                dem = demands[node]

                for r2 in range(len(routes)):
                    if r1 != r2 and loads[r2] + dem > cap:
                        continue

                    pos_range = range(len(routes[r2]) + 1)
                    if r1 == r2:
                        pos_range = range(len(routes[r2]) + 1)

                    for pos in pos_range:
                        if r1 == r2 and (pos == i or pos == i + 1):
                            continue

                        new_routes = [r[:] for r in routes]
                        moved = new_routes[r1].pop(i)

                        insert_pos = pos
                        if r1 == r2 and pos > i:
                            insert_pos -= 1
                        new_routes[r2].insert(insert_pos, moved)

                        new_loads = loads[:]
                        if r1 != r2:
                            new_loads[r1] -= dem
                            new_loads[r2] += dem

                        new_cost = total_cost(new_routes, coords, depot)
                        if new_cost + 1e-9 < base_cost:
                            routes[:] = new_routes
                            loads[:] = new_loads
                            improved = True
                            base_cost = new_cost
                            break
                    if improved:
                        break
                if improved:
                    break
            if improved:
                break


def merge_to_k(routes, loads, demands, coords, depot, cap, k):
    while len(routes) > k:
        best_pair = None
        best_delta = float('inf')

        for i in range(len(routes)):
            for j in range(i + 1, len(routes)):
                if loads[i] + loads[j] > cap:
                    continue

                r1, r2 = routes[i], routes[j]
                candidates = [
                    r1 + r2,
                    r1 + list(reversed(r2)),
                    list(reversed(r1)) + r2,
                    list(reversed(r1)) + list(reversed(r2)),
                ]

                old = route_cost(r1, coords, depot) + route_cost(r2, coords, depot)
                best_new = min(route_cost(c, coords, depot) for c in candidates)
                delta = best_new - old

                if delta < best_delta:
                    best_delta = delta
                    best_pair = (i, j, candidates)

        if best_pair is None:
            break

        i, j, candidates = best_pair
        best_route = min(candidates, key=lambda r: route_cost(r, coords, depot))

        routes[i] = best_route
        loads[i] = sum(demands[c] for c in best_route)

        del routes[j]
        del loads[j]


def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return

    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = int(next(it))

    depot = (float(next(it)), float(next(it)))

    coords = []
    demands = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        coords.append((x, y))
        demands.append(d)

    customers = list(range(n))

    routes = nearest_neighbor_seed(customers, demands, coords, depot, cap)
    loads = [sum(demands[c] for c in r) for r in routes]

    for i in range(len(routes)):
        routes[i] = two_opt(routes[i], coords, depot)

    improve_relocate(routes, loads, demands, coords, depot, cap)

    if len(routes) > k:
        merge_to_k(routes, loads, demands, coords, depot, cap, k)
        for i in range(len(routes)):
            routes[i] = two_opt(routes[i], coords, depot)

    visited = [0] * n
    for r in routes:
        load = sum(demands[c] for c in r)
        if load > cap:
            raise RuntimeError("Capacity violation in constructed solution")
        for c in r:
            visited[c] += 1

    if any(v != 1 for v in visited):
        raise RuntimeError("Customer coverage violation")

    ans_cost = total_cost(routes, coords, depot)

    out = []
    out.append(f"{ans_cost:.6f}")
    out.append(str(len(routes)))
    for r in routes:
        out.append(" ".join(map(str, r)))

    sys.stdout.write("\n".join(out))


if __name__ == "__main__":
    main()
