using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private struct AdjEdge
    {
        public int To;
        public int W;
        public int Id;
        public AdjEdge(int to, int w, int id) { To = to; W = w; Id = id; }
    }

    private struct PairState : IComparable<PairState>
    {
        public long Dist;
        public int V;
        public int Mask;
        public PairState(long dist, int v, int mask) { Dist = dist; V = v; Mask = mask; }
        public int CompareTo(PairState other)
        {
            int c = Dist.CompareTo(other.Dist);
            if (c != 0) return c;
            c = V.CompareTo(other.V);
            if (c != 0) return c;
            return Mask.CompareTo(other.Mask);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;
        public FastScanner(Stream stream) { _stream = stream; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    private sealed class MinHeap
    {
        private (long d, int v)[] _a;
        public int Count { get; private set; }
        public MinHeap(int cap = 16) { _a = new (long, int)[cap]; }

        public void Push(long d, int v)
        {
            if (Count == _a.Length) Array.Resize(ref _a, _a.Length << 1);
            int i = Count++;
            _a[i] = (d, v);
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_a[p].d <= _a[i].d) break;
                (_a[p], _a[i]) = (_a[i], _a[p]);
                i = p;
            }
        }

        public (long d, int v) Pop()
        {
            var root = _a[0];
            var x = _a[--Count];
            if (Count > 0)
            {
                _a[0] = x;
                int i = 0;
                while (true)
                {
                    int l = (i << 1) + 1;
                    if (l >= Count) break;
                    int r = l + 1;
                    int m = (r < Count && _a[r].d < _a[l].d) ? r : l;
                    if (_a[i].d <= _a[m].d) break;
                    (_a[i], _a[m]) = (_a[m], _a[i]);
                    i = m;
                }
            }
            return root;
        }
    }

    private static long[] Dijkstra(int n, List<AdjEdge>[] g, int src, int[] parent)
    {
        const long INF = long.MaxValue / 4;
        var dist = new long[n];
        Array.Fill(dist, INF);
        if (parent != null) Array.Fill(parent, -1);
        dist[src] = 0;
        var pq = new MinHeap();
        pq.Push(0, src);

        while (pq.Count > 0)
        {
            var (cd, u) = pq.Pop();
            if (cd != dist[u]) continue;
            var lst = g[u];
            for (int i = 0; i < lst.Count; i++)
            {
                var e = lst[i];
                long nd = cd + e.W;
                if (nd < dist[e.To])
                {
                    dist[e.To] = nd;
                    if (parent != null) parent[e.To] = u;
                    pq.Push(nd, e.To);
                }
            }
        }
        return dist;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        var g = new List<AdjEdge>[n];
        for (int i = 0; i < n; i++) g[i] = new List<AdjEdge>();
        var deg = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();
            g[u].Add(new AdjEdge(v, w, i));
            g[v].Add(new AdjEdge(u, w, i));
            deg[u]++;
            deg[v]++;
        }

        // Build multiplicity map for augmented multigraph.
        var mult = new Dictionary<long, int>(m * 2 + 16);
        static long Key(int a, int b) => ((long)a << 32) | (uint)b;
        void AddArc(int a, int b)
        {
            long k = Key(a, b);
            mult.TryGetValue(k, out int c);
            mult[k] = c + 1;
        }

        for (int u = 0; u < n; u++)
        {
            var lst = g[u];
            for (int i = 0; i < lst.Count; i++)
            {
                var e = lst[i];
                if (u < e.To)
                {
                    AddArc(u, e.To);
                    AddArc(e.To, u);
                }
            }
        }

        var odd = new List<int>();
        for (int i = 0; i < n; i++) if ((deg[i] & 1) != 0) odd.Add(i);

        if (odd.Count > 0)
        {
            if (odd.Count <= 20)
            {
                int k = odd.Count;
                var distMat = new long[k, k];
                var allDist = new long[k][];
                for (int i = 0; i < k; i++) allDist[i] = Dijkstra(n, g, odd[i], null);
                for (int i = 0; i < k; i++)
                    for (int j = 0; j < k; j++)
                        distMat[i, j] = allDist[i][odd[j]];

                int maxMask = 1 << k;
                var dp = new long[maxMask];
                var choiceA = new int[maxMask];
                var choiceB = new int[maxMask];
                const long INF = long.MaxValue / 4;
                for (int i = 0; i < maxMask; i++) dp[i] = INF;
                dp[0] = 0;

                for (int mask = 1; mask < maxMask; mask++)
                {
                    if ((BitOperations.PopCount((uint)mask) & 1) == 1) continue;
                    int a = 0;
                    while (((mask >> a) & 1) == 0) a++;
                    int maskWithoutA = mask ^ (1 << a);
                    long best = INF;
                    int bestB = -1;
                    for (int b = a + 1; b < k; b++)
                    {
                        if (((maskWithoutA >> b) & 1) == 0) continue;
                        int prev = maskWithoutA ^ (1 << b);
                        long cand = dp[prev] + distMat[a, b];
                        if (cand < best)
                        {
                            best = cand;
                            bestB = b;
                        }
                    }
                    dp[mask] = best;
                    choiceA[mask] = a;
                    choiceB[mask] = bestB;
                }

                int full = maxMask - 1;
                int curMask = full;
                while (curMask != 0)
                {
                    int a = choiceA[curMask];
                    int b = choiceB[curMask];
                    int s = odd[a], t = odd[b];
                    var parent = new int[n];
                    Dijkstra(n, g, s, parent);
                    int x = t;
                    while (x != s)
                    {
                        int p = parent[x];
                        AddArc(p, x);
                        AddArc(x, p);
                        x = p;
                    }
                    curMask ^= (1 << a);
                    curMask ^= (1 << b);
                }
            }
            else
            {
                // Scalable heuristic: nearest-neighbor pairing with on-demand Dijkstra.
                int k = odd.Count;
                var alive = new bool[k];
                Array.Fill(alive, true);
                int remaining = k;

                while (remaining > 0)
                {
                    int ai = -1;
                    for (int i = 0; i < k; i++) if (alive[i]) { ai = i; break; }
                    int a = odd[ai];

                    var dist = Dijkstra(n, g, a, null);
                    long best = long.MaxValue / 4;
                    int bj = -1;
                    for (int j = 0; j < k; j++)
                    {
                        if (!alive[j] || j == ai) continue;
                        long d = dist[odd[j]];
                        if (d < best)
                        {
                            best = d;
                            bj = j;
                        }
                    }

                    int b = odd[bj];
                    var parent = new int[n];
                    Dijkstra(n, g, a, parent);
                    int x = b;
                    while (x != a)
                    {
                        int p = parent[x];
                        AddArc(p, x);
                        AddArc(x, p);
                        x = p;
                    }

                    alive[ai] = false;
                    alive[bj] = false;
                    remaining -= 2;
                }
            }
        }

        // Hierholzer on directed arc multiplicities (each undirected edge contributes two arcs).
        var next = new Dictionary<int, List<int>>(n);
        foreach (var kv in mult)
        {
            if (kv.Value <= 0) continue;
            int u = (int)(kv.Key >> 32);
            int v = (int)(uint)kv.Key;
            if (!next.TryGetValue(u, out var lst))
            {
                lst = new List<int>();
                next[u] = lst;
            }
            for (int c = 0; c < kv.Value; c++) lst.Add(v);
        }

        var st = new Stack<int>();
        var circuit = new List<int>(m + 16);
        st.Push(0);

        while (st.Count > 0)
        {
            int u = st.Peek();
            if (next.TryGetValue(u, out var lst) && lst.Count > 0)
            {
                int v = lst[lst.Count - 1];
                lst.RemoveAt(lst.Count - 1);
                st.Push(v);
            }
            else
            {
                circuit.Add(u);
                st.Pop();
            }
        }

        circuit.Reverse();

        var sb = new StringBuilder();
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(circuit[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}

internal static class BitOperations
{
    public static int PopCount(uint x)
    {
        x = x - ((x >> 1) & 0x55555555u);
        x = (x & 0x33333333u) + ((x >> 2) & 0x33333333u);
        return (int)((((x + (x >> 4)) & 0x0F0F0F0Fu) * 0x01010101u) >> 24);
    }
}
