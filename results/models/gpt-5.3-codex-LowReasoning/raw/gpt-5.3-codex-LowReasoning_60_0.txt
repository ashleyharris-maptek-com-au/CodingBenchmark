const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const machineCount = Math.max(1, (data.machineCount | 0) || 1);
const tasksInput = Array.isArray(data.tasks) ? data.tasks : [];

class MaxHeap {
  constructor(compare) {
    this.a = [];
    this.compare = compare;
  }
  size() { return this.a.length; }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.compare(a[i], a[p]) <= 0) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }
  pop() {
    const a = this.a;
    if (!a.length) return null;
    const top = a[0];
    const last = a.pop();
    if (a.length) {
      a[0] = last;
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        const r = l + 1;
        let best = i;
        if (l < a.length && this.compare(a[l], a[best]) > 0) best = l;
        if (r < a.length && this.compare(a[r], a[best]) > 0) best = r;
        if (best === i) break;
        [a[i], a[best]] = [a[best], a[i]];
        i = best;
      }
    }
    return top;
  }
}

// Normalize tasks and build id -> index map.
const n = tasksInput.length;
const idToIdx = new Map();
const tasks = new Array(n);
for (let i = 0; i < n; i++) {
  const t = tasksInput[i] || {};
  const id = t.id;
  idToIdx.set(id, i);
  tasks[i] = {
    id,
    duration: Math.max(0, (t.duration | 0) || 0),
    depsRaw: Array.isArray(t.deps) ? t.deps : [],
    deadline: Number.isFinite(t.deadline) ? t.deadline : Number.POSITIVE_INFINITY,
    weight: Number.isFinite(t.weight) ? t.weight : 1,
    succ: [],
    indeg: 0,
    depMaxFinish: 0,
    bottom: 0,
  };
}

// Build graph with only valid dependencies.
for (let i = 0; i < n; i++) {
  const t = tasks[i];
  for (const depId of t.depsRaw) {
    const p = idToIdx.get(depId);
    if (p === undefined || p === i) continue;
    tasks[p].succ.push(i);
    t.indeg++;
  }
}

// Critical-path-like bottom levels (duration + max successor bottom).
// Kahn order then reverse DP; if cycles exist, unresolved nodes keep own duration.
const indegTmp = tasks.map(t => t.indeg);
const q = [];
for (let i = 0; i < n; i++) if (indegTmp[i] === 0) q.push(i);
const topo = [];
for (let h = 0; h < q.length; h++) {
  const u = q[h];
  topo.push(u);
  for (const v of tasks[u].succ) {
    if (--indegTmp[v] === 0) q.push(v);
  }
}
for (let i = n - 1; i >= 0; i--) {
  const u = topo[i];
  let bestSucc = 0;
  for (const v of tasks[u].succ) {
    if (tasks[v].bottom > bestSucc) bestSucc = tasks[v].bottom;
  }
  tasks[u].bottom = tasks[u].duration + bestSucc;
}
for (let i = 0; i < n; i++) {
  if (tasks[i].bottom === 0) tasks[i].bottom = tasks[i].duration;
}

// Ready queue priority: urgent by projected lateness, then higher criticality/weight.
function priority(t, now) {
  const finish = now + t.duration;
  const late = Number.isFinite(t.deadline) ? Math.max(0, finish - t.deadline) : 0;
  const urgency = late * (t.weight > 0 ? t.weight : 1);
  const slack = Number.isFinite(t.deadline) ? (t.deadline - finish) : Number.POSITIVE_INFINITY;
  return { urgency, slack };
}

const ready = new MaxHeap((a, b) => {
  // Higher urgency first.
  if (a.urgency !== b.urgency) return a.urgency - b.urgency;
  // Smaller slack first.
  if (a.slack !== b.slack) return b.slack - a.slack;
  // Larger bottom first.
  if (a.bottom !== b.bottom) return a.bottom - b.bottom;
  // Larger weight first.
  if (a.weight !== b.weight) return a.weight - b.weight;
  // Longer duration first.
  if (a.duration !== b.duration) return a.duration - b.duration;
  // Stable by index.
  return b.idx - a.idx;
});

for (let i = 0; i < n; i++) {
  if (tasks[i].indeg === 0) {
    const p = priority(tasks[i], 0);
    ready.push({ idx: i, ...tasks[i], ...p });
  }
}

// Machine availability heap (min by time) simulated with sorted scan (m is small).
const machineFree = new Array(machineCount).fill(0);

const remainingIndeg = tasks.map(t => t.indeg);
const schedule = [];
let scheduled = 0;

while (scheduled < n) {
  if (ready.size() === 0) {
    // Cycle or invalid graph segment: emit deterministic fallback order for unscheduled tasks.
    for (let i = 0; i < n; i++) {
      if (remainingIndeg[i] >= 0) {
        // mark as forced once
        remainingIndeg[i] = -1;
      }
    }
    break;
  }

  // Find machine with earliest free time.
  let bestMachine = 0;
  for (let m = 1; m < machineCount; m++) {
    if (machineFree[m] < machineFree[bestMachine]) bestMachine = m;
  }

  const pick = ready.pop();
  const t = tasks[pick.idx];
  const start = Math.max(machineFree[bestMachine], t.depMaxFinish);
  const finish = start + t.duration;

  schedule.push({ id: t.id, start, machine: bestMachine });
  machineFree[bestMachine] = finish;
  scheduled++;

  // Mark as scheduled.
  remainingIndeg[pick.idx] = -1;

  for (const v of t.succ) {
    if (remainingIndeg[v] <= 0) continue;
    if (finish > tasks[v].depMaxFinish) tasks[v].depMaxFinish = finish;
    remainingIndeg[v]--;
    if (remainingIndeg[v] === 0) {
      const now = tasks[v].depMaxFinish;
      const p = priority(tasks[v], now);
      ready.push({ idx: v, ...tasks[v], ...p });
    }
  }
}

if (scheduled < n) {
  // Fallback: provide an order for robustness in cyclic/invalid inputs.
  const inSchedule = new Set(schedule.map(x => x.id));
  const order = schedule.map(x => x.id);
  for (const t of tasks) if (!inSchedule.has(t.id)) order.push(t.id);
  process.stdout.write(JSON.stringify({ order }));
} else {
  process.stdout.write(JSON.stringify({ schedule }));
}
