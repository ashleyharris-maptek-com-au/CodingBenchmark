#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    int duration;
};

struct Instance {
    int J = 0, M = 0;
    vector<vector<Task>> jobs;
    vector<int> T;
    int totalOps = 0;
};

struct Schedule {
    vector<vector<long long>> start; // start[j][k]
    long long makespan = (1LL << 62);
};

static inline long long llmax(long long a, long long b) { return a > b ? a : b; }

static long long evaluateMakespan(const Instance& ins, const vector<vector<long long>>& start) {
    long long ms = 0;
    for (int j = 0; j < ins.J; ++j) {
        for (int k = 0; k < ins.T[j]; ++k) {
            ms = max(ms, start[j][k] + (long long)ins.jobs[j][k].duration);
        }
    }
    return ms;
}

struct HeuristicParams {
    double wSlack = 0.0;
    double wRemain = 0.0;
    double wDur = 0.0;
    double wMachLoad = 0.0;
    bool preferEarlyStart = true;
    uint64_t seed = 1;
};

static vector<long long> computeMachineLoads(const Instance& ins) {
    vector<long long> load(ins.M, 0);
    for (int j = 0; j < ins.J; ++j) {
        for (auto &t : ins.jobs[j]) load[t.machine] += t.duration;
    }
    return load;
}

static Schedule runHeuristic(const Instance& ins, const HeuristicParams& p) {
    vector<vector<long long>> st(ins.J);
    for (int j = 0; j < ins.J; ++j) st[j].assign(ins.T[j], -1);

    vector<long long> jobReady(ins.J, 0), machReady(ins.M, 0);
    vector<int> nextOp(ins.J, 0);
    vector<long long> remain(ins.J, 0);
    for (int j = 0; j < ins.J; ++j) {
        long long s = 0;
        for (auto &t : ins.jobs[j]) s += t.duration;
        remain[j] = s;
    }
    vector<long long> machLoad = computeMachineLoads(ins);

    vector<int> active;
    active.reserve(ins.J);
    for (int j = 0; j < ins.J; ++j) if (ins.T[j] > 0) active.push_back(j);

    std::mt19937_64 rng(p.seed);
    std::uniform_real_distribution<double> noise(-1e-6, 1e-6);

    int done = 0;
    while (done < ins.totalOps) {
        int bestJ = -1;
        long long bestEST = 0;
        double bestScore = -1e300;

        for (int idx = 0; idx < (int)active.size(); ++idx) {
            int j = active[idx];
            int k = nextOp[j];
            if (k >= ins.T[j]) continue;
            const auto& t = ins.jobs[j][k];
            long long est = max(jobReady[j], machReady[t.machine]);

            double score = 0.0;
            if (p.preferEarlyStart) score -= (double)est * 1e-3;
            score += p.wRemain * (double)remain[j];
            score += p.wDur * (double)t.duration;
            score += p.wMachLoad * (double)machLoad[t.machine];
            double slack = (double)(est - jobReady[j]);
            score -= p.wSlack * slack;
            score += noise(rng);

            if (bestJ == -1 || score > bestScore || (fabs(score - bestScore) < 1e-12 && est < bestEST)) {
                bestJ = j;
                bestEST = est;
                bestScore = score;
            }
        }

        if (bestJ == -1) break;

        int k = nextOp[bestJ];
        const auto& t = ins.jobs[bestJ][k];
        long long s = max(jobReady[bestJ], machReady[t.machine]);

        st[bestJ][k] = s;
        long long end = s + t.duration;
        jobReady[bestJ] = end;
        machReady[t.machine] = end;
        remain[bestJ] -= t.duration;
        machLoad[t.machine] -= t.duration;
        nextOp[bestJ]++;
        done++;

        if (nextOp[bestJ] >= ins.T[bestJ]) {
            // remove finished job from active in O(1)
            for (int i = 0; i < (int)active.size(); ++i) {
                if (active[i] == bestJ) {
                    active[i] = active.back();
                    active.pop_back();
                    break;
                }
            }
        }
    }

    Schedule out;
    out.start = std::move(st);
    out.makespan = evaluateMakespan(ins, out.start);
    return out;
}

struct ExactSolver {
    const Instance* ins;
    vector<vector<long long>> curStart, bestStart;
    vector<long long> jobReady, machReady;
    vector<int> nextOp;
    long long bestMs;

    void dfs(int scheduled, long long curMax) {
        if (scheduled == ins->totalOps) {
            if (curMax < bestMs) {
                bestMs = curMax;
                bestStart = curStart;
            }
            return;
        }

        // Admissible lower bound
        long long lb = curMax;
        for (int j = 0; j < ins->J; ++j) {
            long long rem = 0;
            for (int k = nextOp[j]; k < ins->T[j]; ++k) rem += ins->jobs[j][k].duration;
            lb = max(lb, jobReady[j] + rem);
        }
        if (lb >= bestMs) return;

        vector<int> cand;
        cand.reserve(ins->J);
        for (int j = 0; j < ins->J; ++j) if (nextOp[j] < ins->T[j]) cand.push_back(j);

        sort(cand.begin(), cand.end(), [&](int a, int b) {
            int ka = nextOp[a], kb = nextOp[b];
            const auto& ta = ins->jobs[a][ka];
            const auto& tb = ins->jobs[b][kb];
            long long esta = max(jobReady[a], machReady[ta.machine]);
            long long estb = max(jobReady[b], machReady[tb.machine]);
            if (esta != estb) return esta < estb;
            return ta.duration > tb.duration;
        });

        for (int j : cand) {
            int k = nextOp[j];
            const auto& t = ins->jobs[j][k];
            long long s = max(jobReady[j], machReady[t.machine]);
            long long e = s + t.duration;

            auto oldJR = jobReady[j];
            auto oldMR = machReady[t.machine];
            nextOp[j]++;
            jobReady[j] = e;
            machReady[t.machine] = e;
            curStart[j][k] = s;

            dfs(scheduled + 1, max(curMax, e));

            curStart[j][k] = -1;
            machReady[t.machine] = oldMR;
            jobReady[j] = oldJR;
            nextOp[j]--;
        }
    }

    Schedule solve(const Instance& in) {
        ins = &in;
        curStart.assign(ins->J, {});
        bestStart.assign(ins->J, {});
        for (int j = 0; j < ins->J; ++j) {
            curStart[j].assign(ins->T[j], -1);
            bestStart[j].assign(ins->T[j], -1);
        }
        jobReady.assign(ins->J, 0);
        machReady.assign(ins->M, 0);
        nextOp.assign(ins->J, 0);

        bestMs = (1LL << 62);
        dfs(0, 0);

        Schedule s;
        s.start = std::move(bestStart);
        s.makespan = bestMs;
        return s;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Instance ins;
    if (!(cin >> ins.J >> ins.M)) return 0;
    ins.jobs.assign(ins.J, {});
    ins.T.assign(ins.J, 0);

    for (int j = 0; j < ins.J; ++j) {
        int Tj;
        cin >> Tj;
        ins.T[j] = Tj;
        ins.jobs[j].resize(Tj);
        for (int k = 0; k < Tj; ++k) {
            int m, d;
            cin >> m >> d;
            if (m < 0) m = 0;
            if (m >= ins.M) m = ins.M - 1;
            if (d < 0) d = 0;
            ins.jobs[j][k] = {m, d};
            ins.totalOps++;
        }
    }

    if (ins.J == 0) return 0;

    Schedule best;
    bool solved = false;

    // Adaptive strategy selection by scale
    if (ins.totalOps <= 18 && ins.J <= 8 && ins.M <= 8) {
        ExactSolver ex;
        best = ex.solve(ins);
        solved = true;
    }

    if (!solved) {
        vector<HeuristicParams> configs;

        if (ins.totalOps <= 4000) {
            configs = {
                {0.7, 1.0, 0.2, 0.1, true, 1},
                {0.3, 1.2, 0.0, 0.2, true, 2},
                {1.0, 0.8, 0.3, 0.0, true, 3},
                {0.0, 1.4, 0.1, 0.3, true, 4},
                {0.6, 0.9, 0.7, 0.0, true, 5},
                {0.8, 0.6, 0.2, 0.6, true, 6},
                {0.4, 1.1, 0.4, 0.1, true, 7},
                {0.5, 1.0, 0.5, 0.5, true, 8}
            };
        } else if (ins.totalOps <= 20000) {
            configs = {
                {0.6, 1.0, 0.2, 0.2, true, 11},
                {0.3, 1.2, 0.1, 0.3, true, 12},
                {0.9, 0.8, 0.3, 0.1, true, 13},
                {0.2, 1.1, 0.4, 0.3, true, 14}
            };
        } else {
            configs = {
                {0.7, 1.0, 0.1, 0.2, true, 21},
                {0.4, 1.1, 0.2, 0.2, true, 22}
            };
        }

        best.makespan = (1LL << 62);
        for (const auto& cfg : configs) {
            Schedule cand = runHeuristic(ins, cfg);
            if (cand.makespan < best.makespan) best = std::move(cand);
        }
        solved = true;
    }

    for (int j = 0; j < ins.J; ++j) {
        for (int k = 0; k < ins.T[j]; ++k) {
            if (k) cout << ' ';
            cout << best.start[j][k] << ' ' << ins.jobs[j][k].machine;
        }
        cout << '\n';
    }

    return 0;
}
