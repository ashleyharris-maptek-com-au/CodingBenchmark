use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Default)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    #[inline]
    fn add(self, o: Pt) -> Pt {
        Pt { x: self.x + o.x, y: self.y + o.y }
    }
    #[inline]
    fn sub(self, o: Pt) -> Pt {
        Pt { x: self.x - o.x, y: self.y - o.y }
    }
    #[inline]
    fn mul(self, k: f64) -> Pt {
        Pt { x: self.x * k, y: self.y * k }
    }
    #[inline]
    fn norm2(self) -> f64 {
        self.x * self.x + self.y * self.y
    }
}

#[derive(Clone)]
struct XorShift64 {
    s: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { s: if seed == 0 { 0x9E3779B97F4A7C15 } else { seed } }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.s;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.s = x;
        x
    }
    #[inline]
    fn next_f64(&mut self) -> f64 {
        (self.next_u64() >> 11) as f64 * (1.0 / ((1u64 << 53) as f64))
    }
}

fn component_decompose(adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let n = adj.len();
    let mut vis = vec![false; n];
    let mut comps = Vec::new();
    let mut q = VecDeque::new();
    for s in 0..n {
        if vis[s] { continue; }
        vis[s] = true;
        q.push_back(s);
        let mut comp = Vec::new();
        while let Some(u) = q.pop_front() {
            comp.push(u);
            for &v in &adj[u] {
                if !vis[v] {
                    vis[v] = true;
                    q.push_back(v);
                }
            }
        }
        comps.push(comp);
    }
    comps
}

fn two_sweep_farthest(start: usize, comp_nodes: &[usize], adj: &[Vec<usize>], mark: &[bool]) -> usize {
    let n = adj.len();
    let mut dist = vec![-1i32; n];
    let mut q = VecDeque::new();
    dist[start] = 0;
    q.push_back(start);
    let mut best = start;
    while let Some(u) = q.pop_front() {
        if dist[u] > dist[best] { best = u; }
        for &v in &adj[u] {
            if mark[v] && dist[v] < 0 {
                dist[v] = dist[u] + 1;
                q.push_back(v);
            }
        }
    }
    if dist[best] < 0 {
        *comp_nodes.first().unwrap_or(&start)
    } else {
        best
    }
}

fn bfs_dist(src: usize, adj: &[Vec<usize>], mark: &[bool]) -> Vec<i32> {
    let n = adj.len();
    let mut dist = vec![-1i32; n];
    let mut q = VecDeque::new();
    dist[src] = 0;
    q.push_back(src);
    while let Some(u) = q.pop_front() {
        for &v in &adj[u] {
            if mark[v] && dist[v] < 0 {
                dist[v] = dist[u] + 1;
                q.push_back(v);
            }
        }
    }
    dist
}

fn layout_small_force(comp: &[usize], adj: &[Vec<usize>], pos: &mut [Pt], seed: u64) {
    let n = comp.len();
    if n == 1 {
        pos[comp[0]] = Pt { x: 0.0, y: 0.0 };
        return;
    }
    let mut rng = XorShift64::new(seed ^ (n as u64).wrapping_mul(0xD6E8FEB86659FD93));

    let mut idx = vec![usize::MAX; pos.len()];
    for (i, &u) in comp.iter().enumerate() { idx[u] = i; }

    let mut local = vec![Pt::default(); n];
    let r0 = (n as f64).sqrt();
    for i in 0..n {
        let a = 2.0 * std::f64::consts::PI * (i as f64) / (n as f64);
        let rr = r0 * (0.7 + 0.6 * rng.next_f64());
        local[i] = Pt { x: rr * a.cos(), y: rr * a.sin() };
    }

    let iters = if n <= 100 { 250 } else if n <= 400 { 180 } else { 120 };
    let k = (1.0 + n as f64).sqrt();

    for t in 0..iters {
        let mut disp = vec![Pt::default(); n];

        let samples = 24usize.min(n.saturating_sub(1).max(1));
        for i in 0..n {
            for _ in 0..samples {
                let mut j = (rng.next_u64() as usize) % n;
                if j == i { j = (j + 1) % n; }
                let d = local[i].sub(local[j]);
                let d2 = d.norm2() + 1e-6;
                let f = (k * k) / d2;
                disp[i] = disp[i].add(d.mul(f));
            }
        }

        for (li, &u) in comp.iter().enumerate() {
            for &v in &adj[u] {
                let lj = idx[v];
                if lj == usize::MAX { continue; }
                if li < lj {
                    let d = local[li].sub(local[lj]);
                    let dist = (d.norm2() + 1e-9).sqrt();
                    let f = dist / k;
                    let dir = d.mul(1.0 / dist.max(1e-9));
                    disp[li] = disp[li].sub(dir.mul(f));
                    disp[lj] = disp[lj].add(dir.mul(f));
                }
            }
        }

        let temp = (1.0 - (t as f64) / (iters as f64)).max(0.02) * (0.8 * k + 2.0);
        for i in 0..n {
            let d2 = disp[i].norm2();
            if d2 > 0.0 {
                let step = temp / d2.sqrt();
                local[i] = local[i].add(disp[i].mul(step.min(1.0)));
            }
        }
    }

    for (i, &u) in comp.iter().enumerate() {
        pos[u] = local[i];
    }
}

fn layout_large_graph(comp: &[usize], adj: &[Vec<usize>], pos: &mut [Pt], seed: u64) {
    let mut rng = XorShift64::new(seed ^ 0xA0761D6478BD642F);
    let n_total = pos.len();
    let mut in_comp = vec![false; n_total];
    for &u in comp { in_comp[u] = true; }

    let root = comp[0];
    let a = two_sweep_farthest(root, comp, adj, &in_comp);
    let b = two_sweep_farthest(a, comp, adj, &in_comp);
    let da = bfs_dist(a, adj, &in_comp);
    let db = bfs_dist(b, adj, &in_comp);

    let mut layer_buckets: Vec<Vec<usize>> = Vec::new();
    let mut keys = Vec::with_capacity(comp.len());
    let mut max_layer = 0usize;

    for &u in comp {
        let x = if da[u] >= 0 && db[u] >= 0 { da[u] - db[u] } else { 0 };
        let y = if da[u] >= 0 && db[u] >= 0 { da[u] + db[u] } else { 0 };
        keys.push((u, x, y));
        max_layer = max_layer.max((x + 200000) as usize);
    }
    layer_buckets.resize(max_layer + 1, Vec::new());

    for (u, x, _) in &keys {
        layer_buckets[(*x + 200000) as usize].push(*u);
    }

    for (u, x, y) in keys {
        let j = rng.next_f64() - 0.5;
        pos[u] = Pt {
            x: x as f64 * 1.0 + j * 0.2,
            y: y as f64 * 0.35 + j,
        };
    }

    let smooth_rounds = if comp.len() > 50_000 { 2 } else { 5 };
    let alpha0 = if comp.len() > 50_000 { 0.45 } else { 0.6 };

    for r in 0..smooth_rounds {
        let alpha = alpha0 * (1.0 - (r as f64) / ((smooth_rounds + 1) as f64));
        let mut newp: Vec<(usize, Pt)> = Vec::with_capacity(comp.len());
        for &u in comp {
            if adj[u].is_empty() {
                newp.push((u, pos[u]));
                continue;
            }
            let mut sum = Pt::default();
            let mut deg = 0usize;
            for &v in &adj[u] {
                if in_comp[v] {
                    sum = sum.add(pos[v]);
                    deg += 1;
                }
            }
            if deg == 0 {
                newp.push((u, pos[u]));
            } else {
                let avg = sum.mul(1.0 / deg as f64);
                let p = pos[u].mul(1.0 - alpha).add(avg.mul(alpha));
                newp.push((u, p));
            }
        }
        for (u, p) in newp { pos[u] = p; }
    }
}

fn normalize_component(comp: &[usize], pos: &mut [Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &u in comp {
        minx = minx.min(pos[u].x);
        maxx = maxx.max(pos[u].x);
        miny = miny.min(pos[u].y);
        maxy = maxy.max(pos[u].y);
    }
    let w = (maxx - minx).max(1e-6);
    let h = (maxy - miny).max(1e-6);
    let scale = 100.0 / w.max(h);
    for &u in comp {
        pos[u].x = (pos[u].x - minx) * scale;
        pos[u].y = (pos[u].y - miny) * scale;
    }
    (0.0, 100.0 * w / w.max(h), 0.0, 100.0 * h / w.max(h))
}

fn pack_components(comps: &[Vec<usize>], pos: &mut [Pt]) {
    if comps.is_empty() { return; }
    let mut order: Vec<usize> = (0..comps.len()).collect();
    order.sort_by(|&i, &j| comps[j].len().cmp(&comps[i].len()));

    let mut dims = vec![(0.0, 0.0); comps.len()];
    for (ci, comp) in comps.iter().enumerate() {
        let (_, w, _, h) = normalize_component(comp, pos);
        dims[ci] = (w.max(3.0), h.max(3.0));
    }

    let total_area: f64 = order.iter().map(|&i| dims[i].0 * dims[i].1).sum();
    let row_target = total_area.sqrt() * 1.3;

    let mut x = 0.0;
    let mut y = 0.0;
    let mut row_h: f64 = 0.0;
    let gap = 20.0;

    for &ci in &order {
        let (w, h) = dims[ci];
        if x > 0.0 && x + w > row_target {
            x = 0.0;
            y += row_h + gap;
            row_h = 0.0;
        }
        for &u in &comps[ci] {
            pos[u].x += x;
            pos[u].y += y;
        }
        x += w + gap;
        row_h = row_h.max(h);
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next().and_then(|x| x.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let m: usize = match it.next().and_then(|x| x.parse().ok()) {
        Some(v) => v,
        None => {
            for _ in 0..n { println!("0 0"); }
            return;
        }
    };

    let mut adj = vec![Vec::<usize>::new(); n];
    let mut edges_read = 0usize;
    while edges_read < m {
        let uo = it.next();
        let vo = it.next();
        if uo.is_none() || vo.is_none() { break; }
        let u: usize = match uo.unwrap().parse() { Ok(v) => v, Err(_) => { edges_read += 1; continue; } };
        let v: usize = match vo.unwrap().parse() { Ok(v) => v, Err(_) => { edges_read += 1; continue; } };
        edges_read += 1;
        if u >= n || v >= n || u == v { continue; }
        adj[u].push(v);
        adj[v].push(u);
    }

    for nbrs in &mut adj {
        nbrs.sort_unstable();
        nbrs.dedup();
    }

    let comps = component_decompose(&adj);
    let mut pos = vec![Pt::default(); n];

    for (cid, comp) in comps.iter().enumerate() {
        if comp.len() <= 1500 {
            layout_small_force(comp, &adj, &mut pos, 0x1234_5678_9ABC_DEF0u64 ^ cid as u64);
        } else {
            layout_large_graph(comp, &adj, &mut pos, 0xF00D_BAAD_CAFE_BEEFu64 ^ cid as u64);
        }
    }

    pack_components(&comps, &mut pos);

    for p in pos {
        let x = if p.x.is_finite() { p.x } else { 0.0 };
        let y = if p.y.is_finite() { p.y } else { 0.0 };
        println!("{:.6} {:.6}", x, y);
    }
}
