#include <bits/stdc++.h>
using namespace std;

struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    int idx = 0, size = 0;
    char buf[BUFSIZE];

    inline char read() {
        if (idx >= size) {
            size = (int)fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    bool nextInt(int &out) {
        char c;
        do {
            c = read();
            if (!c) return false;
        } while (c <= ' ');

        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = read();
        }

        int x = 0;
        while (c > ' ') {
            x = x * 10 + (c - '0');
            c = read();
        }
        out = x * sign;
        return true;
    }
};

struct MIS {
    int n;
    vector<vector<int>> adj;
    vector<unsigned long long> alive;
    vector<int> degree;

    vector<int> current;
    vector<int> best;

    MIS(int n_) : n(n_), adj(n_) {}

    inline bool bitGet(const vector<unsigned long long> &b, int v) const {
        return (b[v >> 6] >> (v & 63)) & 1ULL;
    }

    inline void bitSet(vector<unsigned long long> &b, int v) const {
        b[v >> 6] |= (1ULL << (v & 63));
    }

    inline void bitReset(vector<unsigned long long> &b, int v) const {
        b[v >> 6] &= ~(1ULL << (v & 63));
    }

    int popcountBits(const vector<unsigned long long> &b) const {
        int s = 0;
        for (auto x : b) s += __popcnt64(x);
        return s;
    }

    void buildDegrees(const vector<unsigned long long> &A, vector<int> &deg) const {
        deg.assign(n, 0);
        for (int v = 0; v < n; ++v) {
            if (!bitGet(A, v)) continue;
            int d = 0;
            for (int u : adj[v]) if (bitGet(A, u)) ++d;
            deg[v] = d;
        }
    }

    vector<int> aliveVertices(const vector<unsigned long long> &A) const {
        vector<int> v;
        v.reserve(popcountBits(A));
        for (int i = 0; i < n; ++i) if (bitGet(A, i)) v.push_back(i);
        return v;
    }

    int greedyColorBound(const vector<unsigned long long> &A, vector<int> &order, vector<int> &color) const {
        order.clear();
        color.clear();

        vector<int> verts = aliveVertices(A);
        sort(verts.begin(), verts.end(), [&](int x, int y) {
            return adj[x].size() > adj[y].size();
        });

        vector<vector<int>> colorClasses;
        colorClasses.reserve(64);

        for (int v : verts) {
            int c = 0;
            for (; c < (int)colorClasses.size(); ++c) {
                bool ok = true;
                for (int u : colorClasses[c]) {
                    // edge in original graph => cannot share color
                    // For MIS upper bound we color original graph.
                    bool isNbr = false;
                    for (int w : adj[v]) {
                        if (w == u) {
                            isNbr = true;
                            break;
                        }
                    }
                    if (isNbr) {
                        ok = false;
                        break;
                    }
                }
                if (ok) break;
            }
            if (c == (int)colorClasses.size()) colorClasses.push_back({});
            colorClasses[c].push_back(v);
        }

        int maxColor = (int)colorClasses.size();
        order.reserve(verts.size());
        color.reserve(verts.size());

        for (int c = 0; c < maxColor; ++c) {
            for (int v : colorClasses[c]) {
                order.push_back(v);
                color.push_back(c + 1);
            }
        }

        return maxColor;
    }

    void includeVertex(vector<unsigned long long> &A, int v) {
        if (!bitGet(A, v)) return;
        bitReset(A, v);
        for (int u : adj[v]) if (bitGet(A, u)) bitReset(A, u);
    }

    bool reduce(vector<unsigned long long> &A, vector<int> &chosen) {
        bool changed = true;
        vector<int> deg;

        while (changed) {
            changed = false;
            buildDegrees(A, deg);

            for (int v = 0; v < n; ++v) {
                if (!bitGet(A, v)) continue;
                if (deg[v] == 0) {
                    chosen.push_back(v);
                    includeVertex(A, v);
                    changed = true;
                }
            }

            if (changed) continue;

            buildDegrees(A, deg);
            for (int v = 0; v < n; ++v) {
                if (!bitGet(A, v)) continue;
                if (deg[v] == 1) {
                    int u = -1;
                    for (int x : adj[v]) if (bitGet(A, x)) { u = x; break; }
                    if (u == -1) continue;

                    int dv = deg[v], du = deg[u];
                    if (dv >= du) {
                        chosen.push_back(v);
                        includeVertex(A, v);
                    } else {
                        chosen.push_back(u);
                        includeVertex(A, u);
                    }
                    changed = true;
                    break;
                }
            }
        }
        return true;
    }

    int chooseBranchVertex(const vector<unsigned long long> &A) const {
        int bestV = -1;
        int bestDeg = -1;
        for (int v = 0; v < n; ++v) {
            if (!bitGet(A, v)) continue;
            int d = 0;
            for (int u : adj[v]) if (bitGet(A, u)) ++d;
            if (d > bestDeg) {
                bestDeg = d;
                bestV = v;
            }
        }
        return bestV;
    }

    void dfs(vector<unsigned long long> A, vector<int> chosen) {
        reduce(A, chosen);

        int rem = popcountBits(A);
        if (rem == 0) {
            if (chosen.size() > best.size()) best = move(chosen);
            return;
        }

        // Fast optimistic bound: chosen + remaining vertices
        if ((int)chosen.size() + rem <= (int)best.size()) return;

        // Tighter bound via greedy coloring on original graph
        vector<int> order, color;
        int ub = greedyColorBound(A, order, color);
        if ((int)chosen.size() + ub <= (int)best.size()) return;

        int v = chooseBranchVertex(A);
        if (v == -1) {
            if (chosen.size() > best.size()) best = move(chosen);
            return;
        }

        // Branch 1: include v
        {
            auto A1 = A;
            auto c1 = chosen;
            c1.push_back(v);
            includeVertex(A1, v);
            dfs(move(A1), move(c1));
        }

        // Branch 2: exclude v
        bitReset(A, v);
        dfs(move(A), move(chosen));
    }

    vector<int> solve() {
        alive.assign((n + 63) >> 6, ~0ULL);
        if (n % 64) alive.back() = (1ULL << (n % 64)) - 1ULL;

        vector<int> chosen;
        chosen.reserve(n);
        best.clear();
        best.reserve(n);

        dfs(alive, chosen);
        sort(best.begin(), best.end());
        return best;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    FastInput in;
    int n, m;
    if (!in.nextInt(n)) return 0;
    in.nextInt(m);

    MIS solver(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        in.nextInt(u);
        in.nextInt(v);
        if (u < 0 || v < 0 || u >= n || v >= n || u == v) continue;
        solver.adj[u].push_back(v);
        solver.adj[v].push_back(u);
    }

    for (int i = 0; i < n; ++i) {
        auto &a = solver.adj[i];
        sort(a.begin(), a.end());
        a.erase(unique(a.begin(), a.end()), a.end());
    }

    vector<int> ans = solver.solve();

    cout << ans.size() << '\n';
    for (int i = 0; i < (int)ans.size(); ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
