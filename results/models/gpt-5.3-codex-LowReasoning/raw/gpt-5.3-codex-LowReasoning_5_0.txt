use std::collections::{HashSet, VecDeque};
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
struct Pt {
    x: i32,
    y: i32,
}

fn idx(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn neighbors4(i: usize, w: usize, h: usize, blocked: &[bool]) -> Vec<usize> {
    let x = i % w;
    let y = i / w;
    let mut v = Vec::with_capacity(4);
    if x > 0 {
        let j = i - 1;
        if !blocked[j] {
            v.push(j);
        }
    }
    if x + 1 < w {
        let j = i + 1;
        if !blocked[j] {
            v.push(j);
        }
    }
    if y > 0 {
        let j = i - w;
        if !blocked[j] {
            v.push(j);
        }
    }
    if y + 1 < h {
        let j = i + w;
        if !blocked[j] {
            v.push(j);
        }
    }
    v
}

fn is_adj(a: usize, b: usize, w: usize) -> bool {
    let ax = a % w;
    let ay = a / w;
    let bx = b % w;
    let by = b / w;
    (ax == bx && ay.abs_diff(by) == 1) || (ay == by && ax.abs_diff(bx) == 1)
}

fn component_count(unused: &[bool], blocked: &[bool], w: usize, h: usize, must_include: Option<usize>) -> usize {
    let n = w * h;
    let mut seen = vec![false; n];
    let mut comps = 0usize;

    for i in 0..n {
        if blocked[i] || !unused[i] || seen[i] {
            continue;
        }
        comps += 1;
        let mut q = VecDeque::new();
        q.push_back(i);
        seen[i] = true;
        while let Some(u) = q.pop_front() {
            for v in neighbors4(u, w, h, blocked) {
                if unused[v] && !seen[v] {
                    seen[v] = true;
                    q.push_back(v);
                }
            }
        }
    }

    if let Some(m) = must_include {
        if !blocked[m] && unused[m] {
            let mut ok = false;
            for i in 0..n {
                if blocked[i] || !unused[i] {
                    continue;
                }
                if i == m {
                    ok = true;
                    break;
                }
            }
            if !ok {
                return usize::MAX;
            }
        }
    }

    comps
}

fn bipartite_counts(blocked: &[bool], w: usize, h: usize) -> (usize, usize) {
    let mut black = 0usize;
    let mut white = 0usize;
    for y in 0..h {
        for x in 0..w {
            let i = idx(x, y, w);
            if blocked[i] {
                continue;
            }
            if (x + y) % 2 == 0 {
                black += 1;
            } else {
                white += 1;
            }
        }
    }
    (black, white)
}

fn quick_impossible_checks(start: usize, blocked: &[bool], w: usize, h: usize, require_cycle: bool) -> bool {
    if blocked[start] {
        return true;
    }

    let nfree = blocked.iter().filter(|&&b| !b).count();
    if nfree == 0 {
        return true;
    }

    let mut unused = vec![false; w * h];
    for i in 0..w * h {
        if !blocked[i] {
            unused[i] = true;
        }
    }
    if component_count(&unused, blocked, w, h, Some(start)) != 1 {
        return true;
    }

    let (black, white) = bipartite_counts(blocked, w, h);
    let diff = black.abs_diff(white);
    if require_cycle {
        if black != white {
            return true;
        }
    } else if diff > 1 {
        return true;
    }

    false
}

fn snake_obstacle_free(w: usize, h: usize, require_cycle: bool) -> Option<Vec<usize>> {
    if w == 0 || h == 0 {
        return None;
    }
    if w == 1 && h == 1 {
        if require_cycle {
            return None;
        }
        return Some(vec![0]);
    }

    // Start fixed at (0,0). We can construct for many parity/layout cases.
    // Prefer row-wise snake when h even, else column-wise when w even.
    let mut path = Vec::with_capacity(w * h);

    if h % 2 == 0 {
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    path.push(idx(x, y, w));
                }
            } else {
                for x in (0..w).rev() {
                    path.push(idx(x, y, w));
                }
            }
        }
    } else if w % 2 == 0 {
        for x in 0..w {
            if x % 2 == 0 {
                for y in 0..h {
                    path.push(idx(x, y, w));
                }
            } else {
                for y in (0..h).rev() {
                    path.push(idx(x, y, w));
                }
            }
        }
    } else {
        // both odd: Hamiltonian path exists but cycle impossible. Construct path ending near corner.
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    path.push(idx(x, y, w));
                }
            } else {
                for x in (1..w).rev() {
                    path.push(idx(x, y, w));
                }
            }
        }
        for y in (1..h).rev() {
            path.push(idx(0, y, w));
        }
        if path.len() != w * h || path[0] != 0 {
            return None;
        }
    }

    if path.first().copied() != Some(0) || path.len() != w * h {
        return None;
    }
    if require_cycle {
        let last = *path.last().unwrap();
        if !is_adj(last, 0, w) {
            return None;
        }
    }

    Some(path)
}

struct Solver {
    w: usize,
    h: usize,
    blocked: Vec<bool>,
    require_cycle: bool,
    start: usize,
    deadline: Instant,

    path: Vec<usize>,
    used: Vec<bool>,
    free_count: usize,
    nbrs: Vec<Vec<usize>>,
}

impl Solver {
    fn new(w: usize, h: usize, blocked: Vec<bool>, require_cycle: bool, time_budget: Duration) -> Self {
        let n = w * h;
        let mut nbrs = vec![Vec::new(); n];
        for i in 0..n {
            if !blocked[i] {
                nbrs[i] = neighbors4(i, w, h, &blocked);
            }
        }
        let free_count = blocked.iter().filter(|&&b| !b).count();
        Self {
            w,
            h,
            blocked,
            require_cycle,
            start: 0,
            deadline: Instant::now() + time_budget,
            path: Vec::with_capacity(free_count),
            used: vec![false; n],
            free_count,
            nbrs,
        }
    }

    fn remaining_degree(&self, v: usize) -> usize {
        self.nbrs[v]
            .iter()
            .filter(|&&u| !self.used[u])
            .count()
    }

    fn forced_next(&self, cur: usize) -> Option<usize> {
        let mut forced = None;
        for &n in &self.nbrs[cur] {
            if self.used[n] {
                continue;
            }
            if self.remaining_degree(n) <= 1 {
                if forced.is_some() {
                    return None;
                }
                forced = Some(n);
            }
        }
        forced
    }

    fn connectivity_ok_after_mark(&self) -> bool {
        let n = self.w * self.h;
        let mut rem = vec![false; n];
        let mut rem_count = 0usize;
        let mut seed = None;
        for i in 0..n {
            if !self.blocked[i] && !self.used[i] {
                rem[i] = true;
                rem_count += 1;
                if seed.is_none() {
                    seed = Some(i);
                }
            }
        }
        if rem_count <= 1 {
            return true;
        }
        let s = seed.unwrap();
        let mut seen = vec![false; n];
        let mut q = VecDeque::new();
        q.push_back(s);
        seen[s] = true;
        let mut seen_count = 1usize;
        while let Some(u) = q.pop_front() {
            for &v in &self.nbrs[u] {
                if rem[v] && !seen[v] {
                    seen[v] = true;
                    seen_count += 1;
                    q.push_back(v);
                }
            }
        }
        seen_count == rem_count
    }

    fn candidate_order(&self, cur: usize) -> Vec<usize> {
        let mut cands: Vec<usize> = self.nbrs[cur]
            .iter()
            .copied()
            .filter(|&v| !self.used[v])
            .collect();

        cands.sort_by_key(|&v| {
            let deg = self.remaining_degree(v);
            let manhattan_to_start = {
                let x = v % self.w;
                let y = v / self.w;
                x + y
            };
            (deg, manhattan_to_start)
        });
        cands
    }

    fn dfs(&mut self, cur: usize) -> bool {
        if Instant::now() >= self.deadline {
            return false;
        }

        if self.path.len() == self.free_count {
            if self.require_cycle {
                return is_adj(cur, self.start, self.w);
            }
            return true;
        }

        if let Some(f) = self.forced_next(cur) {
            self.used[f] = true;
            self.path.push(f);
            let ok = self.connectivity_ok_after_mark() && self.dfs(f);
            if ok {
                return true;
            }
            self.path.pop();
            self.used[f] = false;
            return false;
        }

        let cands = self.candidate_order(cur);
        for v in cands {
            self.used[v] = true;
            self.path.push(v);

            let rem = self.free_count - self.path.len();
            let plausible = if self.require_cycle {
                // Need enough steps to eventually return adjacent to start.
                let vx = v % self.w;
                let vy = v / self.w;
                let d0 = vx + vy;
                rem + 1 >= d0
            } else {
                true
            };

            let ok = plausible && self.connectivity_ok_after_mark() && self.dfs(v);
            if ok {
                return true;
            }

            self.path.pop();
            self.used[v] = false;
        }
        false
    }

    fn solve_exact(mut self) -> Option<Vec<usize>> {
        if self.blocked[self.start] {
            return None;
        }
        self.used[self.start] = true;
        self.path.push(self.start);
        if self.dfs(self.start) {
            Some(self.path)
        } else {
            None
        }
    }
}

fn validate_path(path: &[usize], w: usize, h: usize, blocked: &[bool], require_cycle: bool) -> bool {
    let nfree = blocked.iter().filter(|&&b| !b).count();
    if path.len() != nfree || path.is_empty() || path[0] != 0 {
        return false;
    }
    let n = w * h;
    let mut seen = vec![false; n];
    for (k, &v) in path.iter().enumerate() {
        if v >= n || blocked[v] || seen[v] {
            return false;
        }
        seen[v] = true;
        if k > 0 && !is_adj(path[k - 1], v, w) {
            return false;
        }
    }
    if require_cycle && !is_adj(*path.last().unwrap(), 0, w) {
        return false;
    }
    true
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let h: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let k: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let require_cycle = match it.next().and_then(|s| s.parse::<u8>().ok()) {
        Some(v) => v != 0,
        None => {
            println!("0");
            return;
        }
    };

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    let n = w * h;
    let mut blocked = vec![false; n];
    let mut obs = HashSet::new();

    for _ in 0..k {
        let x: i32 = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        let y: i32 = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        if x < 0 || y < 0 || x as usize >= w || y as usize >= h {
            continue;
        }
        obs.insert((x as usize, y as usize));
    }

    for (x, y) in obs {
        blocked[idx(x, y, w)] = true;
    }

    if quick_impossible_checks(0, &blocked, w, h, require_cycle) {
        println!("0");
        return;
    }

    // Fast path: obstacle-free constructive solution.
    if blocked.iter().all(|&b| !b) {
        if let Some(path) = snake_obstacle_free(w, h, require_cycle) {
            if validate_path(&path, w, h, &blocked, require_cycle) {
                println!("{}", path.len());
                for &v in &path {
                    let x = v % w;
                    let y = v / w;
                    println!("{} {}", x, y);
                }
                return;
            }
        }
    }

    // Adaptive exact search budget.
    let free_count = blocked.iter().filter(|&&b| !b).count();
    let budget = if free_count <= 120 {
        Duration::from_secs(295)
    } else if free_count <= 220 {
        Duration::from_secs(120)
    } else {
        Duration::from_secs(25)
    };

    let solver = Solver::new(w, h, blocked.clone(), require_cycle, budget);
    if let Some(path) = solver.solve_exact() {
        if validate_path(&path, w, h, &blocked, require_cycle) {
            println!("{}", path.len());
            for &v in &path {
                let x = v % w;
                let y = v / w;
                println!("{} {}", x, y);
            }
            return;
        }
    }

    println!("0");
}
