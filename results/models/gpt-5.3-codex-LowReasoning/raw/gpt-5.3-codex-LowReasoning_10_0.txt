#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct Rect {
    int w;
    int h;
    int id;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rot = 0;
};

struct FreeRect {
    int x;
    int y;
    int w;
    int h;
};

struct Board {
    std::vector<FreeRect> freeRects;
};

static inline bool canFit(const FreeRect& f, int w, int h) {
    return w <= f.w && h <= f.h;
}

static inline std::int64_t scoreWaste(const FreeRect& f, int w, int h) {
    return static_cast<std::int64_t>(f.w) * f.h - static_cast<std::int64_t>(w) * h;
}

static void pruneFreeRects(std::vector<FreeRect>& fr) {
    for (size_t i = 0; i < fr.size(); ++i) {
        if (fr[i].w <= 0 || fr[i].h <= 0) {
            std::swap(fr[i], fr.back());
            fr.pop_back();
            --i;
        }
    }

    for (size_t i = 0; i < fr.size(); ++i) {
        bool removeI = false;
        for (size_t j = 0; j < fr.size(); ++j) {
            if (i == j) continue;
            const auto& a = fr[i];
            const auto& b = fr[j];
            if (a.x >= b.x && a.y >= b.y && a.x + a.w <= b.x + b.w && a.y + a.h <= b.y + b.h) {
                removeI = true;
                break;
            }
        }
        if (removeI) {
            std::swap(fr[i], fr.back());
            fr.pop_back();
            --i;
        }
    }
}

static void splitGuillotine(Board& b, size_t idx, int rw, int rh) {
    FreeRect f = b.freeRects[idx];
    b.freeRects[idx] = b.freeRects.back();
    b.freeRects.pop_back();

    int rightW = f.w - rw;
    int topH = f.h - rh;

    // Split along the larger leftover side to reduce fragmentation.
    if (rightW > topH) {
        if (rightW > 0) b.freeRects.push_back({f.x + rw, f.y, rightW, rh});
        if (topH > 0) b.freeRects.push_back({f.x, f.y + rh, f.w, topH});
    } else {
        if (topH > 0) b.freeRects.push_back({f.x, f.y + rh, rw, topH});
        if (rightW > 0) b.freeRects.push_back({f.x + rw, f.y, rightW, f.h});
    }

    pruneFreeRects(b.freeRects);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, BW, BH;
    if (!(std::cin >> N >> BW >> BH)) {
        return 0;
    }

    std::vector<Rect> rects;
    rects.reserve(N);
    std::vector<Placement> ans(N);

    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        rects.push_back({w, h, i});
    }

    // Pre-validate and normalize impossible cases.
    for (const auto& r : rects) {
        bool ok = (r.w <= BW && r.h <= BH) || (r.h <= BW && r.w <= BH);
        if (!ok) {
            // No valid packing exists; emit zero boards and sentinel placements.
            std::cout << 0 << '\n';
            for (int i = 0; i < N; ++i) {
                std::cout << -1 << " 0 0 0\n";
            }
            return 0;
        }
    }

    // Sort by descending max side, then area, then min side.
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        int aMax = std::max(a.w, a.h), bMax = std::max(b.w, b.h);
        if (aMax != bMax) return aMax > bMax;
        std::int64_t aArea = static_cast<std::int64_t>(a.w) * a.h;
        std::int64_t bArea = static_cast<std::int64_t>(b.w) * b.h;
        if (aArea != bArea) return aArea > bArea;
        int aMin = std::min(a.w, a.h), bMin = std::min(b.w, b.h);
        return aMin > bMin;
    });

    std::vector<Board> boards;
    boards.reserve(std::max(1, N / 8));

    const bool huge = N >= 10000;
    const int boardScanCap = huge ? 48 : 256;
    const int freeScanCap = huge ? 96 : 512;

    for (const auto& r : rects) {
        int bestBoard = -1;
        size_t bestFree = 0;
        int bestW = 0, bestH = 0, bestRot = 0;
        std::int64_t bestScore = std::numeric_limits<std::int64_t>::max();

        int startBoard = std::max(0, static_cast<int>(boards.size()) - boardScanCap);
        for (int bi = static_cast<int>(boards.size()) - 1; bi >= startBoard; --bi) {
            auto& fr = boards[bi].freeRects;
            int fStart = std::max(0, static_cast<int>(fr.size()) - freeScanCap);

            for (int fi = static_cast<int>(fr.size()) - 1; fi >= fStart; --fi) {
                const auto& f = fr[fi];

                if (canFit(f, r.w, r.h)) {
                    auto s = scoreWaste(f, r.w, r.h);
                    if (s < bestScore) {
                        bestScore = s;
                        bestBoard = bi;
                        bestFree = static_cast<size_t>(fi);
                        bestW = r.w;
                        bestH = r.h;
                        bestRot = 0;
                        if (s == 0) break;
                    }
                }
                if (r.w != r.h && canFit(f, r.h, r.w)) {
                    auto s = scoreWaste(f, r.h, r.w);
                    if (s < bestScore) {
                        bestScore = s;
                        bestBoard = bi;
                        bestFree = static_cast<size_t>(fi);
                        bestW = r.h;
                        bestH = r.w;
                        bestRot = 1;
                        if (s == 0) break;
                    }
                }
            }
            if (bestScore == 0) break;
        }

        if (bestBoard == -1) {
            Board nb;
            nb.freeRects.push_back({0, 0, BW, BH});
            boards.push_back(std::move(nb));
            bestBoard = static_cast<int>(boards.size()) - 1;

            auto& f = boards[bestBoard].freeRects[0];
            if (canFit(f, r.w, r.h) && canFit(f, r.h, r.w)) {
                // Choose orientation that leaves less max leftover edge.
                int remA = std::max(f.w - r.w, f.h - r.h);
                int remB = std::max(f.w - r.h, f.h - r.w);
                if (remB < remA) {
                    bestW = r.h;
                    bestH = r.w;
                    bestRot = 1;
                } else {
                    bestW = r.w;
                    bestH = r.h;
                    bestRot = 0;
                }
            } else if (canFit(f, r.w, r.h)) {
                bestW = r.w;
                bestH = r.h;
                bestRot = 0;
            } else {
                bestW = r.h;
                bestH = r.w;
                bestRot = 1;
            }
            bestFree = 0;
        }

        const FreeRect placed = boards[bestBoard].freeRects[bestFree];
        ans[r.id] = {bestBoard, placed.x, placed.y, bestRot};
        splitGuillotine(boards[bestBoard], bestFree, bestW, bestH);
    }

    std::cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        std::cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rot << '\n';
    }

    return 0;
}
