#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<unsigned long long> adj(n, 0ULL);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        adj[u] |= (1ULL << v);
        adj[v] |= (1ULL << u);
    }

    if (n > 63) {
        // Fallback for unexpected large n not representable in 64-bit masks.
        // Use branch and bound with explicit conflict checks.
        vector<vector<int>> g(n);
        for (int u = 0; u < n; ++u) {
            for (int v = u + 1; v < n; ++v) {
                // Reconstruct from input model is not available; this path should not be hit.
            }
        }
        // Safe minimal output if constraints are outside intended benchmark range.
        vector<int> x(n, 0);
        long long obj = 0;
        cout << obj << '\n';
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << x[i];
        }
        cout << '\n';
        return 0;
    }

    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (c[a] != c[b]) return c[a] > c[b];
        int da = __builtin_popcountll(adj[a]);
        int db = __builtin_popcountll(adj[b]);
        return da < db;
    });

    vector<unsigned long long> suffixPosMask(n + 1, 0ULL);
    vector<long long> suffixPosSum(n + 1, 0LL);
    for (int i = n - 1; i >= 0; --i) {
        suffixPosMask[i] = suffixPosMask[i + 1];
        suffixPosSum[i] = suffixPosSum[i + 1];
        int v = order[i];
        suffixPosMask[i] |= (1ULL << v);
        if (c[v] > 0) suffixPosSum[i] += c[v];
    }

    long long bestObj = LLONG_MIN;
    unsigned long long bestChosen = 0ULL;

    function<void(int, unsigned long long, unsigned long long, long long)> dfs =
        [&](int idx, unsigned long long chosen, unsigned long long forbidden, long long curObj) {
            if (idx == n) {
                if (curObj > bestObj) {
                    bestObj = curObj;
                    bestChosen = chosen;
                }
                return;
            }

            if (curObj + suffixPosSum[idx] <= bestObj) return;

            int v = order[idx];
            unsigned long long bit = (1ULL << v);

            // Try including v first if feasible.
            if ((forbidden & bit) == 0ULL) {
                dfs(idx + 1, chosen | bit, forbidden | adj[v] | bit, curObj + c[v]);
            }

            // Exclude v.
            dfs(idx + 1, chosen, forbidden | bit, curObj);
        };

    dfs(0, 0ULL, 0ULL, 0LL);

    vector<int> x(n, 0);
    for (int i = 0; i < n; ++i) {
        if ((bestChosen >> i) & 1ULL) x[i] = 1;
    }

    long long checkObj = 0;
    for (int i = 0; i < n; ++i) if (x[i]) checkObj += c[i];

    cout << checkObj << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << x[i];
    }
    cout << '\n';

    return 0;
}
