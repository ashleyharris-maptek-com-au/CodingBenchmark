use std::cmp::Ordering;
use std::collections::HashMap;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    #[inline]
    fn add(self, o: Pt) -> Pt { Pt { x: self.x + o.x, y: self.y + o.y } }
    #[inline]
    fn sub(self, o: Pt) -> Pt { Pt { x: self.x - o.x, y: self.y - o.y } }
    #[inline]
    fn cross(self, o: Pt) -> f64 { self.x * o.y - self.y * o.x }
}

#[derive(Clone)]
struct Poly {
    v: Vec<Pt>,
    area: f64,
    bbox: BBox,
    centroid: Pt,
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
}

impl BBox {
    fn from_pts(pts: &[Pt]) -> Self {
        let mut minx = f64::INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for &p in pts {
            minx = minx.min(p.x);
            miny = miny.min(p.y);
            maxx = maxx.max(p.x);
            maxy = maxy.max(p.y);
        }
        BBox { minx, miny, maxx, maxy }
    }
    #[inline]
    fn overlaps(&self, o: &BBox) -> bool {
        !(self.maxx < o.minx + EPS || o.maxx < self.minx + EPS || self.maxy < o.miny + EPS || o.maxy < self.miny + EPS)
    }
    #[inline]
    fn width(&self) -> f64 { self.maxx - self.minx }
    #[inline]
    fn height(&self) -> f64 { self.maxy - self.miny }
}

#[derive(Clone)]
struct Piece {
    idx: usize,
    orig: Poly,
    rotations: Vec<(f64, Poly)>,
}

#[derive(Clone)]
struct Placement {
    piece_idx: usize,
    poly: Poly,
    rot: f64,
    tx: f64,
    ty: f64,
}

struct StockBin {
    placed: Vec<Placement>,
    grid: HashMap<(i32, i32), Vec<usize>>,
    cell: f64,
}

fn signed_area(pts: &[Pt]) -> f64 {
    let n = pts.len();
    let mut a = 0.0;
    for i in 0..n {
        let p = pts[i];
        let q = pts[(i + 1) % n];
        a += p.cross(q);
    }
    0.5 * a
}

fn centroid_poly(pts: &[Pt]) -> Pt {
    let n = pts.len();
    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut a2 = 0.0;
    for i in 0..n {
        let p = pts[i];
        let q = pts[(i + 1) % n];
        let c = p.cross(q);
        a2 += c;
        cx += (p.x + q.x) * c;
        cy += (p.y + q.y) * c;
    }
    if a2.abs() < EPS {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for p in pts { sx += p.x; sy += p.y; }
        return Pt { x: sx / pts.len() as f64, y: sy / pts.len() as f64 };
    }
    Pt { x: cx / (3.0 * a2), y: cy / (3.0 * a2) }
}

fn make_poly(mut v: Vec<Pt>) -> Poly {
    if signed_area(&v) < 0.0 {
        v.reverse();
    }
    let area = signed_area(&v).abs();
    let bbox = BBox::from_pts(&v);
    let centroid = centroid_poly(&v);
    Poly { v, area, bbox, centroid }
}

fn rotate_about_centroid(poly: &Poly, ang: f64) -> Poly {
    let c = poly.centroid;
    let (s, co) = ang.sin_cos();
    let mut out = Vec::with_capacity(poly.v.len());
    for &p in &poly.v {
        let x = p.x - c.x;
        let y = p.y - c.y;
        out.push(Pt { x: c.x + x * co - y * s, y: c.y + x * s + y * co });
    }
    make_poly(out)
}

fn translate_poly(poly: &Poly, tx: f64, ty: f64) -> Poly {
    let mut out = Vec::with_capacity(poly.v.len());
    for &p in &poly.v {
        out.push(Pt { x: p.x + tx, y: p.y + ty });
    }
    make_poly(out)
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    b.sub(a).cross(c.sub(a))
}

fn on_seg(a: Pt, b: Pt, p: Pt) -> bool {
    orient(a, b, p).abs() <= 1e-8 &&
    p.x >= a.x.min(b.x) - 1e-8 && p.x <= a.x.max(b.x) + 1e-8 &&
    p.y >= a.y.min(b.y) - 1e-8 && p.y <= a.y.max(b.y) + 1e-8
}

fn seg_inter(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    if o1.abs() <= 1e-8 && on_seg(a, b, c) { return true; }
    if o2.abs() <= 1e-8 && on_seg(a, b, d) { return true; }
    if o3.abs() <= 1e-8 && on_seg(c, d, a) { return true; }
    if o4.abs() <= 1e-8 && on_seg(c, d, b) { return true; }
    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn point_in_poly(p: Pt, poly: &Poly) -> i32 {
    let n = poly.v.len();
    let mut inside = false;
    for i in 0..n {
        let a = poly.v[i];
        let b = poly.v[(i + 1) % n];
        if on_seg(a, b, p) { return 0; }
        let yi = a.y > p.y;
        let yj = b.y > p.y;
        if yi != yj {
            let xint = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if xint > p.x { inside = !inside; }
        }
    }
    if inside { 1 } else { -1 }
}

fn poly_inside(stock: &Poly, p: &Poly) -> bool {
    for &q in &p.v {
        if point_in_poly(q, stock) < 0 { return false; }
    }
    let ns = stock.v.len();
    let np = p.v.len();
    for i in 0..np {
        let a = p.v[i];
        let b = p.v[(i + 1) % np];
        for j in 0..ns {
            let c = stock.v[j];
            let d = stock.v[(j + 1) % ns];
            if seg_inter(a, b, c, d) {
                let a_on = on_seg(c, d, a);
                let b_on = on_seg(c, d, b);
                if !(a_on && b_on) {
                    let c_on = on_seg(a, b, c);
                    let d_on = on_seg(a, b, d);
                    if !(c_on && d_on) {
                        return false;
                    }
                }
            }
        }
    }
    true
}

fn polys_overlap(a: &Poly, b: &Poly) -> bool {
    if !a.bbox.overlaps(&b.bbox) { return false; }
    let na = a.v.len();
    let nb = b.v.len();
    for i in 0..na {
        let p1 = a.v[i];
        let p2 = a.v[(i + 1) % na];
        for j in 0..nb {
            let q1 = b.v[j];
            let q2 = b.v[(j + 1) % nb];
            if seg_inter(p1, p2, q1, q2) {
                if on_seg(q1, q2, p1) || on_seg(q1, q2, p2) || on_seg(p1, p2, q1) || on_seg(p1, p2, q2) {
                    continue;
                }
                return true;
            }
        }
    }
    if point_in_poly(a.v[0], b) > 0 { return true; }
    if point_in_poly(b.v[0], a) > 0 { return true; }
    false
}

fn grid_cells(bb: &BBox, cell: f64) -> (i32, i32, i32, i32) {
    let x0 = (bb.minx / cell).floor() as i32;
    let y0 = (bb.miny / cell).floor() as i32;
    let x1 = (bb.maxx / cell).floor() as i32;
    let y1 = (bb.maxy / cell).floor() as i32;
    (x0, y0, x1, y1)
}

fn main() {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();

    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock_pts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_pts.push(Pt { x, y });
    }
    let stock = make_poly(stock_pts);

    let mut pieces = Vec::with_capacity(np);
    for idx in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut pts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            pts.push(Pt { x, y });
        }
        let poly = make_poly(pts);
        pieces.push(Piece { idx, orig: poly, rotations: Vec::new() });
    }

    let level = if np <= 60 { 0 } else if np <= 400 { 1 } else { 2 };
    let rot_count = match level { 0 => 24, 1 => 12, _ => 6 };

    for pc in &mut pieces {
        let mut rots = Vec::with_capacity(rot_count + 2);
        rots.push((0.0, pc.orig.clone()));
        for k in 1..rot_count {
            let ang = 2.0 * PI * (k as f64) / (rot_count as f64);
            rots.push((ang, rotate_about_centroid(&pc.orig, ang)));
        }
        pc.rotations = rots;
    }

    pieces.sort_by(|a, b| {
        b.orig.area.partial_cmp(&a.orig.area).unwrap_or(Ordering::Equal)
            .then_with(|| b.orig.v.len().cmp(&a.orig.v.len()))
    });

    let scale = ((stock.bbox.width() * stock.bbox.height()) / (np.max(1) as f64)).sqrt().max(1.0);
    let cell = scale * if level == 2 { 1.3 } else { 0.9 };

    let mut bins: Vec<StockBin> = Vec::new();
    let mut out: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); np];

    let stock_vertices = stock.v.clone();

    for piece in pieces {
        let mut placed_ok = false;

        for bi in 0..=bins.len() {
            if bi == bins.len() {
                bins.push(StockBin { placed: Vec::new(), grid: HashMap::new(), cell });
            }

            let mut candidates = Vec::<Pt>::new();
            for &svt in &stock_vertices {
                candidates.push(svt);
            }

            for pl in &bins[bi].placed {
                candidates.push(Pt { x: pl.poly.bbox.minx, y: pl.poly.bbox.maxy });
                candidates.push(Pt { x: pl.poly.bbox.maxx, y: pl.poly.bbox.miny });
                candidates.push(Pt { x: pl.poly.bbox.maxx, y: pl.poly.bbox.maxy });
            }

            let max_try = if level == 0 { 1200 } else if level == 1 { 320 } else { 100 };
            let mut trials = 0usize;

            'rotloop: for (ang, rot_poly) in &piece.rotations {
                for &c in &candidates {
                    if trials >= max_try { break 'rotloop; }
                    trials += 1;

                    let tx = c.x - rot_poly.bbox.minx;
                    let ty = c.y - rot_poly.bbox.miny;
                    let moved = translate_poly(rot_poly, tx, ty);

                    if !poly_inside(&stock, &moved) { continue; }

                    let (x0, y0, x1, y1) = grid_cells(&moved.bbox, bins[bi].cell);
                    let mut seen = HashMap::<usize, bool>::new();
                    let mut collide = false;
                    for gx in x0..=x1 {
                        for gy in y0..=y1 {
                            if let Some(lst) = bins[bi].grid.get(&(gx, gy)) {
                                for &pi in lst {
                                    if seen.insert(pi, true).is_none() {
                                        let other = &bins[bi].placed[pi].poly;
                                        if polys_overlap(&moved, other) {
                                            collide = true;
                                            break;
                                        }
                                    }
                                }
                            }
                            if collide { break; }
                        }
                        if collide { break; }
                    }
                    if collide { continue; }

                    let new_idx = bins[bi].placed.len();
                    bins[bi].placed.push(Placement {
                        piece_idx: piece.idx,
                        poly: moved.clone(),
                        rot: *ang,
                        tx,
                        ty,
                    });

                    let (ix0, iy0, ix1, iy1) = grid_cells(&moved.bbox, bins[bi].cell);
                    for gx in ix0..=ix1 {
                        for gy in iy0..=iy1 {
                            bins[bi].grid.entry((gx, gy)).or_default().push(new_idx);
                        }
                    }

                    out[piece.idx] = (bi, tx, ty, *ang);
                    placed_ok = true;
                    break 'rotloop;
                }
            }

            if placed_ok { break; }
        }

        if !placed_ok {
            // Fallback: force onto a new stock at stock bbox min with zero rotation.
            let bi = bins.len();
            bins.push(StockBin { placed: Vec::new(), grid: HashMap::new(), cell });
            let tx = stock.bbox.minx - piece.orig.bbox.minx;
            let ty = stock.bbox.miny - piece.orig.bbox.miny;
            out[piece.idx] = (bi, tx, ty, 0.0);
            let moved = translate_poly(&piece.orig, tx, ty);
            bins[bi].placed.push(Placement { piece_idx: piece.idx, poly: moved.clone(), rot: 0.0, tx, ty });
            let (x0, y0, x1, y1) = grid_cells(&moved.bbox, cell);
            for gx in x0..=x1 {
                for gy in y0..=y1 {
                    bins[bi].grid.entry((gx, gy)).or_default().push(0);
                }
            }
        }
    }

    println!("{}", bins.len());
    for i in 0..np {
        let (bi, x, y, r) = out[i];
        println!("{} {:.9} {:.9} {:.9}", bi, x, y, r);
    }
}
