using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public static class Program
{
    private struct Op
    {
        public int DigX, DigY, DigZ;
        public int DestX, DestY;
        public int[] Steps; // x0,y0,x1,y1,...
    }

    public static void Main()
    {
        using var input = Console.OpenStandardInput();
        using var output = Console.OpenStandardOutput();
        using var writer = new StreamWriter(output, new UTF8Encoding(false)) { AutoFlush = false };

        if (!TryReadHeader(input, out int xCount, out int yCount, out int zCount, out _))
        {
            writer.WriteLine(0);
            writer.Flush();
            return;
        }

        long columnCount = (long)xCount * yCount;
        long voxelCount = columnCount * zCount;

        bool huge = voxelCount > 120_000_000L || columnCount > 25_000_000L;
        int maxDepthPerColumn = huge ? 1 : (voxelCount > 10_000_000L ? 2 : 4);
        int maxOps = huge ? 25_000 : (voxelCount > 50_000_000L ? 80_000 : 250_000);

        var ops = new List<Op>(Math.Min(maxOps, 200_000));

        byte[] one = new byte[1];
        int ground00 = 0;
        int dumpHeight = 0;
        int dumpX = xCount > 1 ? xCount - 1 : 0;
        int dumpY = yCount > 1 ? yCount - 1 : 0;

        // Stream through columns. Input layout: x then y then z ascending.
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int topZ = zCount - 1;
                int captured = 0;
                int[] topBytes = maxDepthPerColumn > 0 ? new int[maxDepthPerColumn] : Array.Empty<int>();

                for (int z = 0; z < zCount; z++)
                {
                    if (!ReadExact(input, one, 1))
                    {
                        // Truncated input; output best effort.
                        writer.WriteLine(ops.Count);
                        WriteOps(writer, ops);
                        writer.Flush();
                        return;
                    }

                    if (x == 0 && y == 0)
                        ground00 = z + 1;

                    // Keep only top K voxels in this column.
                    int posFromTop = topZ - z;
                    if (posFromTop >= 0 && posFromTop < maxDepthPerColumn)
                    {
                        topBytes[posFromTop] = one[0];
                        captured++;
                    }
                }

                if (captured == 0) continue;

                // Evaluate from exposed top downward a little.
                for (int d = 0; d < captured; d++)
                {
                    if (ops.Count >= maxOps) break;

                    int z = zCount - 1 - d;
                    int data = topBytes[d];
                    int mass = data & 15;
                    int value = (data >> 4) & 15;
                    if (mass <= 0) continue;

                    // Travel estimate: move loaded from dig site to destination.
                    int distToPlant = Math.Abs(x) + Math.Abs(y);
                    int zAtDig = z;
                    int climbToPlant = Math.Max(0, ground00 - zAtDig);
                    int climbCostPlant = 5 * climbToPlant;
                    int moveCostPlant = mass * (distToPlant + climbCostPlant);
                    int revenue = value > 0 ? (1000 * mass * value) : (-1000 * mass);
                    int netToPlant = revenue - moveCostPlant;

                    // For waste, dump at far corner if not worth processing.
                    int distToDump = Math.Abs(x - dumpX) + Math.Abs(y - dumpY);
                    int dumpTop = dumpHeight;
                    int climbToDump = Math.Max(0, dumpTop - zAtDig);
                    int moveCostDump = mass * (distToDump + 5 * climbToDump);
                    int netDump = -moveCostDump;

                    bool sendToPlant = netToPlant > netDump && value > 0;
                    if (!sendToPlant && value > 0 && netToPlant > 0) sendToPlant = true;

                    int destX = sendToPlant ? 0 : dumpX;
                    int destY = sendToPlant ? 0 : dumpY;

                    // Skip very bad moves in large worlds.
                    if (!sendToPlant && huge && distToDump > 2000) continue;
                    if (sendToPlant && netToPlant <= 0) continue;

                    int[] steps = BuildManhattanPath(x, y, destX, destY);
                    ops.Add(new Op { DigX = x, DigY = y, DigZ = z, DestX = destX, DestY = destY, Steps = steps });

                    if (!sendToPlant)
                        dumpHeight++;
                }
            }
        }

        writer.WriteLine(ops.Count);
        WriteOps(writer, ops);
        writer.Flush();
    }

    private static void WriteOps(StreamWriter writer, List<Op> ops)
    {
        foreach (var op in ops)
        {
            int n = op.Steps.Length / 2;
            writer.Write(n);
            writer.Write(' ');
            writer.Write(op.DigX);
            writer.Write(' ');
            writer.Write(op.DigY);
            writer.Write(' ');
            writer.Write(op.DigZ);

            for (int i = 0; i < op.Steps.Length; i += 2)
            {
                writer.Write(' ');
                writer.Write(op.Steps[i]);
                writer.Write(' ');
                writer.Write(op.Steps[i + 1]);
            }

            writer.Write(' ');
            writer.Write(op.DestX);
            writer.Write(' ');
            writer.Write(op.DestY);
            writer.WriteLine();
        }
    }

    private static int[] BuildManhattanPath(int sx, int sy, int dx, int dy)
    {
        int len = Math.Abs(dx - sx) + Math.Abs(dy - sy);
        if (len <= 0) return Array.Empty<int>();

        int[] path = new int[len * 2];
        int idx = 0;
        int x = sx;
        int y = sy;

        while (x != dx)
        {
            x += x < dx ? 1 : -1;
            path[idx++] = x;
            path[idx++] = y;
        }
        while (y != dy)
        {
            y += y < dy ? 1 : -1;
            path[idx++] = x;
            path[idx++] = y;
        }

        return path;
    }

    private static bool TryReadHeader(Stream input, out int x, out int y, out int z, out double[] blockSizes)
    {
        x = y = z = 0;
        blockSizes = new double[3];

        string line = ReadLineAscii(input);
        if (string.IsNullOrWhiteSpace(line)) return false;
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 6) return false;

        if (!int.TryParse(parts[0], out x)) return false;
        if (!int.TryParse(parts[1], out y)) return false;
        if (!int.TryParse(parts[2], out z)) return false;

        double.TryParse(parts[3], out blockSizes[0]);
        double.TryParse(parts[4], out blockSizes[1]);
        double.TryParse(parts[5], out blockSizes[2]);
        return x > 0 && y > 0 && z > 0;
    }

    private static string ReadLineAscii(Stream input)
    {
        var sb = new StringBuilder(128);
        byte[] b = new byte[1];
        while (true)
        {
            int n = input.Read(b, 0, 1);
            if (n == 0) break;
            if (b[0] == (byte)'\n') break;
            if (b[0] != (byte)'\r') sb.Append((char)b[0]);
        }
        return sb.ToString();
    }

    private static bool ReadExact(Stream s, byte[] buf, int count)
    {
        int off = 0;
        while (off < count)
        {
            int n = s.Read(buf, off, count - off);
            if (n <= 0) return false;
            off += n;
        }
        return true;
    }
}
