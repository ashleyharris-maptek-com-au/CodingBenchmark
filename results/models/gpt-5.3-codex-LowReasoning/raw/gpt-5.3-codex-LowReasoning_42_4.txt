OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%thick_start = OpConstant %float 0.0399999991
%thick_end = OpConstant %float 0.00499999989

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v3 = OpTypePointer Input %v3float

%ext = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func
%entry = OpLabel

%wp0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%wp1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p0 = OpLoad %v3float %wp0_ptr
%p1 = OpLoad %v3float %wp1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy
%dir_n = OpExtInst %v2float %ext Normalize %dir
%dx = OpCompositeExtract %float %dir_n 0
%dy = OpCompositeExtract %float %dir_n 1
%ndx = OpFNegate %float %dy
%perp = OpCompositeConstruct %v2float %ndx %dx

%off0 = OpVectorTimesScalar %v2float %perp %thick_start
%off1 = OpVectorTimesScalar %v2float %perp %thick_end

%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1z = OpCompositeExtract %float %p1 2

%off0x = OpCompositeExtract %float %off0 0
%off0y = OpCompositeExtract %float %off0 1
%off1x = OpCompositeExtract %float %off1 0
%off1y = OpCompositeExtract %float %off1 1

%v0x = OpFAdd %float %p0x %off0x
%v0y = OpFAdd %float %p0y %off0y
%v1x = OpFSub %float %p0x %off0x
%v1y = OpFSub %float %p0y %off0y
%v2x = OpFAdd %float %p1x %off1x
%v2y = OpFAdd %float %p1y %off1y
%v3x = OpFSub %float %p1x %off1x
%v3y = OpFSub %float %p1y %off1y

%pos0 = OpCompositeConstruct %v4float %v0x %v0y %p0z %float_1
%pos1 = OpCompositeConstruct %v4float %v1x %v1y %p0z %float_1
%pos2 = OpCompositeConstruct %v4float %v2x %v2y %p1z %float_1
%pos3 = OpCompositeConstruct %v4float %v3x %v3y %p1z %float_1

%glout_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %glout_pos %pos0
OpStore %outColor %c0
OpEmitVertex

OpStore %glout_pos %pos1
OpStore %outColor %c0
OpEmitVertex

OpStore %glout_pos %pos2
OpStore %outColor %c1
OpEmitVertex

OpStore %glout_pos %pos3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
