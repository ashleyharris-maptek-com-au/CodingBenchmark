using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = Read();
            }
            return v * sign;
        }
    }

    private sealed class State
    {
        public readonly int N;
        public readonly List<int>[] Out;
        public readonly List<int>[] In;
        public readonly bool[] Alive;
        public readonly int[] InDeg;
        public readonly int[] OutDeg;
        public int AliveCount;

        public State(int n, List<int>[] outAdj, List<int>[] inAdj)
        {
            N = n;
            Out = outAdj;
            In = inAdj;
            Alive = new bool[n];
            InDeg = new int[n];
            OutDeg = new int[n];
            for (int i = 0; i < n; i++)
            {
                Alive[i] = true;
                InDeg[i] = inAdj[i].Count;
                OutDeg[i] = outAdj[i].Count;
            }
            AliveCount = n;
        }

        public State Clone()
        {
            var s = new State(N, Out, In);
            Array.Copy(Alive, s.Alive, N);
            Array.Copy(InDeg, s.InDeg, N);
            Array.Copy(OutDeg, s.OutDeg, N);
            s.AliveCount = AliveCount;
            return s;
        }

        public void RemoveVertex(int v)
        {
            if (!Alive[v]) return;
            Alive[v] = false;
            AliveCount--;

            var outL = Out[v];
            for (int i = 0; i < outL.Count; i++)
            {
                int to = outL[i];
                if (Alive[to]) InDeg[to]--;
            }

            var inL = In[v];
            for (int i = 0; i < inL.Count; i++)
            {
                int from = inL[i];
                if (Alive[from]) OutDeg[from]--;
            }

            InDeg[v] = 0;
            OutDeg[v] = 0;
        }
    }

    private static int _bestSize;
    private static List<int> _bestSet = new List<int>();

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        var outAdj = new List<int>[n];
        var inAdj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            outAdj[i] = new List<int>();
            inAdj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            outAdj[u].Add(v);
            inAdj[v].Add(u);
        }

        var initial = new State(n, outAdj, inAdj);
        var current = new List<int>();

        _bestSize = n + 1;

        Dfs(initial, current);

        var sb = new StringBuilder();
        sb.AppendLine(_bestSet.Count.ToString());
        if (_bestSet.Count > 0)
        {
            for (int i = 0; i < _bestSet.Count; i++)
            {
                if (i > 0) sb.Append(' ');
                sb.Append(_bestSet[i]);
            }
            sb.AppendLine();
        }
        else
        {
            sb.AppendLine();
        }
        Console.Write(sb.ToString());
    }

    private static void Dfs(State s, List<int> chosen)
    {
        if (chosen.Count >= _bestSize) return;

        ReduceAcyclicBoundary(s);

        if (!HasCycle(s))
        {
            if (chosen.Count < _bestSize)
            {
                _bestSize = chosen.Count;
                _bestSet = new List<int>(chosen);
            }
            return;
        }

        int lb = LowerBoundByPackingCycles(s, _bestSize - chosen.Count - 1);
        if (chosen.Count + lb >= _bestSize) return;

        int branchV = PickBranchVertex(s);

        var s1 = s.Clone();
        var chosen1 = new List<int>(chosen);
        s1.RemoveVertex(branchV);
        chosen1.Add(branchV);
        Dfs(s1, chosen1);

        if (chosen.Count + 1 >= _bestSize) return;

        var forced = ForceKeepAndCollectCycleVertices(s, branchV);
        if (forced == null) return;

        var cycle = forced;
        for (int i = 0; i < cycle.Count; i++)
        {
            int v = cycle[i];
            if (v == branchV) continue;
            var s2 = s.Clone();
            var chosen2 = new List<int>(chosen);
            s2.RemoveVertex(v);
            chosen2.Add(v);
            Dfs(s2, chosen2);
        }
    }

    private static void ReduceAcyclicBoundary(State s)
    {
        var q = new Queue<int>();
        for (int v = 0; v < s.N; v++)
        {
            if (s.Alive[v] && (s.InDeg[v] == 0 || s.OutDeg[v] == 0))
                q.Enqueue(v);
        }

        while (q.Count > 0)
        {
            int v = q.Dequeue();
            if (!s.Alive[v]) continue;
            if (s.InDeg[v] > 0 && s.OutDeg[v] > 0) continue;

            var outL = s.Out[v];
            var inL = s.In[v];

            s.Alive[v] = false;
            s.AliveCount--;

            for (int i = 0; i < outL.Count; i++)
            {
                int to = outL[i];
                if (!s.Alive[to]) continue;
                s.InDeg[to]--;
                if (s.InDeg[to] == 0 || s.OutDeg[to] == 0) q.Enqueue(to);
            }

            for (int i = 0; i < inL.Count; i++)
            {
                int from = inL[i];
                if (!s.Alive[from]) continue;
                s.OutDeg[from]--;
                if (s.InDeg[from] == 0 || s.OutDeg[from] == 0) q.Enqueue(from);
            }

            s.InDeg[v] = 0;
            s.OutDeg[v] = 0;
        }
    }

    private static bool HasCycle(State s)
    {
        int n = s.N;
        var indeg = new int[n];
        int alive = 0;
        for (int i = 0; i < n; i++)
        {
            if (!s.Alive[i]) continue;
            indeg[i] = s.InDeg[i];
            alive++;
        }

        var q = new Queue<int>();
        for (int i = 0; i < n; i++)
            if (s.Alive[i] && indeg[i] == 0) q.Enqueue(i);

        int removed = 0;
        while (q.Count > 0)
        {
            int v = q.Dequeue();
            removed++;
            var outL = s.Out[v];
            for (int i = 0; i < outL.Count; i++)
            {
                int to = outL[i];
                if (!s.Alive[to]) continue;
                if (--indeg[to] == 0) q.Enqueue(to);
            }
        }

        return removed != alive;
    }

    private static int PickBranchVertex(State s)
    {
        int n = s.N;
        int best = -1;
        int bestScore = -1;
        for (int v = 0; v < n; v++)
        {
            if (!s.Alive[v]) continue;
            int score = s.InDeg[v] * s.OutDeg[v] + s.InDeg[v] + s.OutDeg[v];
            if (score > bestScore)
            {
                bestScore = score;
                best = v;
            }
        }
        return best;
    }

    private static int LowerBoundByPackingCycles(State s, int cap)
    {
        int n = s.N;
        var used = new bool[n];
        int packed = 0;

        for (int start = 0; start < n && packed <= cap; start++)
        {
            if (!s.Alive[start] || used[start]) continue;

            var cycle = FindCycleAvoidingUsed(s, used, start);
            if (cycle == null) continue;

            for (int i = 0; i < cycle.Count; i++) used[cycle[i]] = true;
            packed++;
        }

        return packed;
    }

    private static List<int> FindCycleAvoidingUsed(State s, bool[] banned, int start)
    {
        int n = s.N;
        var color = new byte[n];
        var parent = new int[n];
        Array.Fill(parent, -1);

        bool DfsCycle(int v)
        {
            color[v] = 1;
            var outL = s.Out[v];
            for (int i = 0; i < outL.Count; i++)
            {
                int to = outL[i];
                if (!s.Alive[to] || banned[to]) continue;
                if (color[to] == 0)
                {
                    parent[to] = v;
                    if (DfsCycle(to)) return true;
                }
                else if (color[to] == 1)
                {
                    _tmpCycle = new List<int>();
                    _tmpCycle.Add(to);
                    int cur = v;
                    while (cur != to)
                    {
                        _tmpCycle.Add(cur);
                        cur = parent[cur];
                    }
                    return true;
                }
            }
            color[v] = 2;
            return false;
        }

        _tmpCycle = null;
        if (!banned[start] && s.Alive[start]) DfsCycle(start);
        return _tmpCycle;
    }

    private static List<int> _tmpCycle;

    private static List<int> ForceKeepAndCollectCycleVertices(State s, int keepV)
    {
        int n = s.N;
        if (!s.Alive[keepV]) return null;

        var aliveWithout = new bool[n];
        Array.Copy(s.Alive, aliveWithout, n);
        aliveWithout[keepV] = false;

        int[] comp = ComputeScc(s, aliveWithout);
        int compV = comp[keepV];
        if (compV < 0) return null;

        var members = new List<int>();
        for (int i = 0; i < n; i++)
            if (s.Alive[i] && comp[i] == compV) members.Add(i);

        if (members.Count <= 1) return null;
        return members;
    }

    private static int[] ComputeScc(State s, bool[] aliveMask)
    {
        int n = s.N;
        var order = new List<int>(n);
        var vis = new bool[n];

        for (int i = 0; i < n; i++)
        {
            if (!aliveMask[i] || vis[i]) continue;
            IterativeDfs1(s, aliveMask, i, vis, order);
        }

        var comp = new int[n];
        Array.Fill(comp, -1);
        int cid = 0;
        for (int i = order.Count - 1; i >= 0; i--)
        {
            int v = order[i];
            if (!aliveMask[v] || comp[v] != -1) continue;
            IterativeDfs2(s, aliveMask, v, cid, comp);
            cid++;
        }

        return comp;
    }

    private static void IterativeDfs1(State s, bool[] aliveMask, int root, bool[] vis, List<int> order)
    {
        var stackV = new Stack<int>();
        var stackI = new Stack<int>();
        stackV.Push(root);
        stackI.Push(0);
        vis[root] = true;

        while (stackV.Count > 0)
        {
            int v = stackV.Peek();
            int idx = stackI.Pop();
            var outL = s.Out[v];

            while (idx < outL.Count)
            {
                int to = outL[idx++];
                if (!aliveMask[to] || vis[to]) continue;

                stackI.Push(idx);
                stackV.Push(to);
                stackI.Push(0);
                vis[to] = true;
                goto Next;
            }

            stackV.Pop();
            order.Add(v);

            Next: ;
        }
    }

    private static void IterativeDfs2(State s, bool[] aliveMask, int root, int cid, int[] comp)
    {
        var st = new Stack<int>();
        st.Push(root);
        comp[root] = cid;
        while (st.Count > 0)
        {
            int v = st.Pop();
            var inL = s.In[v];
            for (int i = 0; i < inL.Count; i++)
            {
                int to = inL[i];
                if (!aliveMask[to] || comp[to] != -1) continue;
                comp[to] = cid;
                st.Push(to);
            }
        }
    }
}
