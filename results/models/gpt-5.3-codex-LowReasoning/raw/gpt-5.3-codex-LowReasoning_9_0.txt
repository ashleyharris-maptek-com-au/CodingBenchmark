using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public static class Program
{
    private struct Item
    {
        public int Length;
        public int Index;
        public Item(int length, int index)
        {
            Length = length;
            Index = index;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int stockLength = fs.NextInt();

        if (n <= 0 || stockLength <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var items = new Item[n];
        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();
            if (len > stockLength)
            {
                // Invalid instance per constraints; emit no feasible packing.
                Console.WriteLine(0);
                return;
            }
            items[i] = new Item(len, i);
        }

        Array.Sort(items, (a, b) => b.Length.CompareTo(a.Length)); // Descending (FFD)

        var binsRemaining = new List<int>(Math.Min(n, 1024));
        var binsCuts = new List<List<int>>(Math.Min(n, 1024));

        for (int i = 0; i < n; i++)
        {
            int len = items[i].Length;
            int originalIndex = items[i].Index;

            int bin = FindFirstFit(binsRemaining, len);
            if (bin < 0)
            {
                bin = binsRemaining.Count;
                binsRemaining.Add(stockLength - len);
                var cuts = new List<int>(4);
                cuts.Add(originalIndex);
                binsCuts.Add(cuts);
            }
            else
            {
                binsRemaining[bin] -= len;
                binsCuts[bin].Add(originalIndex);
            }
        }

        var sb = new StringBuilder(16 + n * 4);
        sb.Append(binsCuts.Count).Append('\n');
        for (int b = 0; b < binsCuts.Count; b++)
        {
            var cuts = binsCuts[b];
            if (cuts.Count > 0)
            {
                sb.Append(cuts[0]);
                for (int j = 1; j < cuts.Count; j++)
                {
                    sb.Append(' ').Append(cuts[j]);
                }
            }
            sb.Append('\n');
        }

        Console.Write(sb.ToString());
    }

    // First-fit over residual capacities.
    // For large instances this remains very fast in practice with FFD ordering.
    private static int FindFirstFit(List<int> remaining, int need)
    {
        for (int i = 0; i < remaining.Count; i++)
        {
            if (remaining[i] >= need) return i;
        }
        return -1;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _ptr;
        private int _len;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }
}
