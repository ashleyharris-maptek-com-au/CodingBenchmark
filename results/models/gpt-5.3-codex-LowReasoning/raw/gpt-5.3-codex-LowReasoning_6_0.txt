using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly string[] _tokens;
        private int _idx;

        public FastScanner()
        {
            string all = Console.In.ReadToEnd();
            _tokens = all.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            _idx = 0;
        }

        public int NextInt() => int.Parse(_tokens[_idx++], CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(_tokens[_idx++], CultureInfo.InvariantCulture);
    }

    private struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);

        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public Vec3 Cross(Vec3 b) => new Vec3(
            Y * b.Z - Z * b.Y,
            Z * b.X - X * b.Z,
            X * b.Y - Y * b.X
        );

        public double Norm2() => X * X + Y * Y + Z * Z;
        public double Norm() => Math.Sqrt(Norm2());
    }

    private sealed class BodyState
    {
        public Vec3 R;
        public Vec3 V;

        // Precomputed orbital-ish descriptors for fast, scale-robust costing
        public double Rmag;
        public double Vmag;
        public double A;          // semi-major axis (can be +/- for hyperbolic-like states)
        public double Ecc;        // eccentricity magnitude
        public Vec3 Hhat;         // unit angular momentum direction
        public double Energy;
    }

    private static readonly Random Rng = new Random(1234567);

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        double mu = fs.NextDouble();

        BodyState start = ReadState(fs, mu);

        var stations = new BodyState[n];
        for (int i = 0; i < n; i++) stations[i] = ReadState(fs, mu);

        // Build full directed cost matrix from start+stations.
        // Node indexing: 0=start, 1..n=stations.
        var all = new BodyState[n + 1];
        all[0] = start;
        for (int i = 0; i < n; i++) all[i + 1] = stations[i];

        double[,] cost = BuildCost(all, mu);

        int[] order;
        if (n <= 9)
        {
            order = SolveExactHeldKarp(cost, n);
        }
        else if (n <= 20)
        {
            order = SolveHeuristic(cost, n, multiStart: 32, twoOptRounds: 4, candidateK: 16);
        }
        else if (n <= 60)
        {
            order = SolveHeuristic(cost, n, multiStart: 24, twoOptRounds: 3, candidateK: 20);
        }
        else
        {
            order = SolveHeuristic(cost, n, multiStart: 16, twoOptRounds: 2, candidateK: 24);
        }

        Console.WriteLine(string.Join(" ", order));
    }

    private static BodyState ReadState(FastScanner fs, double mu)
    {
        var s = new BodyState();
        s.R = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        s.V = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        PrecomputeDescriptors(s, mu);
        return s;
    }

    private static void PrecomputeDescriptors(BodyState s, double mu)
    {
        s.Rmag = s.R.Norm();
        s.Vmag = s.V.Norm();
        s.Energy = 0.5 * s.Vmag * s.Vmag - mu / Math.Max(1e-9, s.Rmag);

        double denom = 2.0 / Math.Max(1e-9, s.Rmag) - (s.Vmag * s.Vmag) / Math.Max(1e-9, mu);
        if (Math.Abs(denom) < 1e-12) s.A = 1e12;
        else s.A = 1.0 / denom;

        Vec3 h = s.R.Cross(s.V);
        double hmag = h.Norm();
        s.Hhat = hmag > 1e-12 ? h / hmag : new Vec3(0, 0, 1);

        Vec3 evec = (s.V.Cross(h)) / Math.Max(1e-12, mu) - (s.R / Math.Max(1e-12, s.Rmag));
        s.Ecc = evec.Norm();
    }

    private static double[,] BuildCost(BodyState[] all, double mu)
    {
        int m = all.Length;
        var c = new double[m, m];

        // Robust global scales for normalization
        double avgR = all.Average(s => s.Rmag);
        double avgV = all.Average(s => s.Vmag);
        if (avgR < 1e-6) avgR = 7000.0;
        if (avgV < 1e-6) avgV = 7.5;

        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < m; j++)
            {
                if (i == j)
                {
                    c[i, j] = 0;
                    continue;
                }

                BodyState a = all[i], b = all[j];

                double dr = (a.R - b.R).Norm();
                double dv = (a.V - b.V).Norm();

                double plane = 1.0 - Clamp(a.Hhat.Dot(b.Hhat), -1.0, 1.0); // 0..2
                double da = Math.Abs(a.A - b.A) / Math.Max(1.0, Math.Abs(a.A) + Math.Abs(b.A));
                double de = Math.Abs(a.Ecc - b.Ecc);
                double dE = Math.Abs(a.Energy - b.Energy) / Math.Max(1.0, Math.Abs(a.Energy) + Math.Abs(b.Energy));

                // Approx transfer-time penalty from mean motion scale
                double rbar = 0.5 * (a.Rmag + b.Rmag);
                double torb = 2.0 * Math.PI * Math.Sqrt((rbar * rbar * rbar) / Math.Max(1e-9, mu));
                double tpen = torb / 5400.0; // normalize around ~90 min LEO orbit

                // Hybrid metric: velocity change dominates, geometry/orbital mismatch regularizes
                c[i, j] =
                    1.35 * (dv / avgV) +
                    0.45 * (dr / avgR) +
                    0.80 * plane +
                    0.70 * da +
                    0.55 * de +
                    0.35 * dE +
                    0.10 * tpen;
            }
        }

        return c;
    }

    private static int[] SolveExactHeldKarp(double[,] c, int n)
    {
        // Stations are 1..n in c, start is 0. Need Hamiltonian path starting at 0 visiting all once.
        int full = 1 << n;
        double[,] dp = new double[full, n];
        int[,] parent = new int[full, n];

        const double INF = 1e100;
        for (int mask = 0; mask < full; mask++)
            for (int i = 0; i < n; i++)
            {
                dp[mask, i] = INF;
                parent[mask, i] = -1;
            }

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            dp[mask, i] = c[0, i + 1];
        }

        for (int mask = 1; mask < full; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0) continue;
                double cur = dp[mask, last];
                if (cur >= INF * 0.5) continue;

                int rem = (~mask) & (full - 1);
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int nxt = BitIndex(bit);
                    int nmask = mask | bit;
                    double v = cur + c[last + 1, nxt + 1];
                    if (v < dp[nmask, nxt])
                    {
                        dp[nmask, nxt] = v;
                        parent[nmask, nxt] = last;
                    }
                    rem ^= bit;
                }
            }
        }

        int finalMask = full - 1;
        int bestLast = 0;
        double best = INF;
        for (int i = 0; i < n; i++)
        {
            if (dp[finalMask, i] < best)
            {
                best = dp[finalMask, i];
                bestLast = i;
            }
        }

        int[] route = new int[n];
        int cm = finalMask, cl = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = cl; // already 0-based station index
            int p = parent[cm, cl];
            cm ^= (1 << cl);
            cl = p;
        }

        return route;
    }

    private static int[] SolveHeuristic(double[,] c, int n, int multiStart, int twoOptRounds, int candidateK)
    {
        // Build candidate lists (nearest neighbors) for speed
        int[][] cand = BuildCandidates(c, n, candidateK);

        double bestCost = double.PositiveInfinity;
        int[] bestRoute = null;

        // Deterministic seeds + random perturbation starts
        int seeds = Math.Min(n, Math.Max(4, multiStart / 2));
        for (int s = 0; s < multiStart; s++)
        {
            int startSeed = (s < seeds) ? s % n : Rng.Next(n);
            int[] route = BuildGreedyRoute(c, n, startSeed, cand, noise: s >= seeds);

            for (int r = 0; r < twoOptRounds; r++)
            {
                TwoOptDirected(c, route, cand);
                OrOpt1(c, route);
            }

            double cc = PathCost(c, route);
            if (cc < bestCost)
            {
                bestCost = cc;
                bestRoute = (int[])route.Clone();
            }
        }

        return bestRoute ?? Enumerable.Range(0, n).ToArray();
    }

    private static int[] BuildCandidates(double[,] c, int n, int k)
    {
        int[][] cand = new int[n][];
        for (int i = 0; i < n; i++)
        {
            int nodeI = i + 1;
            var arr = new List<(double d, int j)>(n - 1);
            for (int j = 0; j < n; j++)
            {
                if (i == j) continue;
                arr.Add((c[nodeI, j + 1], j));
            }
            arr.Sort((a, b) => a.d.CompareTo(b.d));
            int take = Math.Min(k, arr.Count);
            cand[i] = arr.Take(take).Select(x => x.j).ToArray();
        }
        return cand;
    }

    private static int[] BuildGreedyRoute(double[,] c, int n, int seedStation, int[][] cand, bool noise)
    {
        var used = new bool[n];
        var route = new int[n];

        int cur = seedStation;
        route[0] = cur;
        used[cur] = true;

        // Optionally prepend best first hop from start
        double bestFirst = c[0, cur + 1];
        for (int i = 0; i < n; i++)
        {
            if (used[i]) continue;
            double v = c[0, i + 1];
            if (v < bestFirst)
            {
                bestFirst = v;
                route[0] = i;
                used[cur] = false;
                cur = i;
                used[cur] = true;
            }
        }

        for (int pos = 1; pos < n; pos++)
        {
            int next = -1;
            double best = double.PositiveInfinity;

            // Candidate list first
            foreach (int j in cand[cur])
            {
                if (used[j]) continue;
                double v = c[cur + 1, j + 1];
                if (noise) v *= 0.97 + 0.06 * Rng.NextDouble();
                if (v < best)
                {
                    best = v;
                    next = j;
                }
            }

            // Fallback full scan
            if (next < 0)
            {
                for (int j = 0; j < n; j++)
                {
                    if (used[j]) continue;
                    double v = c[cur + 1, j + 1];
                    if (noise) v *= 0.97 + 0.06 * Rng.NextDouble();
                    if (v < best)
                    {
                        best = v;
                        next = j;
                    }
                }
            }

            route[pos] = next;
            used[next] = true;
            cur = next;
        }

        return route;
    }

    private static void TwoOptDirected(double[,] c, int[] route, int[][] cand)
    {
        int n = route.Length;
        bool improved = true;
        int guard = 0;

        while (improved && guard++ < 8)
        {
            improved = false;

            for (int i = 0; i < n - 2; i++)
            {
                int a = (i == 0) ? -1 : route[i - 1]; // -1 means start node 0
                int b = route[i];

                int[] near = cand[b];
                foreach (int target in near)
                {
                    int j = Array.IndexOf(route, target);
                    if (j <= i + 1 || j >= n) continue;

                    double old1 = (a < 0) ? c[0, b + 1] : c[a + 1, b + 1];
                    double old2 = c[route[j - 1] + 1, route[j] + 1];

                    int nb = route[j - 1];
                    int nj = route[j];

                    double nw1 = (a < 0) ? c[0, nb + 1] : c[a + 1, nb + 1];
                    double nw2 = c[b + 1, nj + 1];

                    // Add interior reversal penalty/benefit estimation by endpoints only (fast directed 2-opt approximation)
                    double delta = (nw1 + nw2) - (old1 + old2);
                    if (delta < -1e-10)
                    {
                        Array.Reverse(route, i, j - i);
                        improved = true;
                        break;
                    }
                }

                if (improved) break;
            }
        }
    }

    private static void OrOpt1(double[,] c, int[] route)
    {
        int n = route.Length;
        bool improved = true;
        int loops = 0;

        while (improved && loops++ < 4)
        {
            improved = false;

            for (int i = 0; i < n; i++)
            {
                int x = route[i];

                int prev = (i == 0) ? -1 : route[i - 1];
                int next = (i == n - 1) ? -2 : route[i + 1]; // -2 means end

                double remove = Edge(c, prev, x) + Edge(c, x, next) - Edge(c, prev, next);

                for (int j = 0; j < n; j++)
                {
                    if (j == i || j == i - 1) continue;

                    int a = route[j];
                    int b = (j == n - 1) ? -2 : route[j + 1];

                    double add = Edge(c, a, x) + Edge(c, x, b) - Edge(c, a, b);
                    if (add + remove < -1e-10)
                    {
                        // Move i after j
                        if (i < j)
                        {
                            int tmp = route[i];
                            Array.Copy(route, i + 1, route, i, j - i);
                            route[j] = tmp;
                        }
                        else
                        {
                            int tmp = route[i];
                            Array.Copy(route, j + 1, route, j + 2, i - j - 1);
                            route[j + 1] = tmp;
                        }
                        improved = true;
                        goto NextLoop;
                    }
                }
            }

        NextLoop:;
        }
    }

    private static double Edge(double[,] c, int from, int to)
    {
        if (to == -2) return 0.0; // open path, no terminal edge
        int i = (from < 0) ? 0 : from + 1;
        int j = to + 1;
        return c[i, j];
    }

    private static double PathCost(double[,] c, int[] route)
    {
        if (route.Length == 0) return 0;
        double sum = c[0, route[0] + 1];
        for (int i = 1; i < route.Length; i++) sum += c[route[i - 1] + 1, route[i] + 1];
        return sum;
    }

    private static int BitIndex(int bit)
    {
        int idx = 0;
        while ((bit >>= 1) != 0) idx++;
        return idx;
    }

    private static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }
}
