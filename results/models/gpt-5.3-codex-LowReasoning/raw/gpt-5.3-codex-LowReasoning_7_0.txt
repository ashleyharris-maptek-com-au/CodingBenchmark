#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>

#include <utility>
#include <vector>

struct Vec3 {
    double x{}, y{}, z{};
    Vec3() = default;
    Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return Vec3(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    );
}

struct Tri { int a{}, b{}, c{}; };

struct Mesh {
    std::vector<Vec3> v;
    std::vector<Tri> f;
};

struct AABB {
    Vec3 mn{ std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity() };
    Vec3 mx{ -std::numeric_limits<double>::infinity(), -std::numeric_limits<double>::infinity(), -std::numeric_limits<double>::infinity() };
    void expand(const Vec3& p) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }
};

static AABB bounds(const Mesh& m) {
    AABB b;
    for (const auto& p : m.v) b.expand(p);
    return b;
}

static bool aabbIntersect(const AABB& a, const AABB& b) {
    return !(a.mx.x < b.mn.x || b.mx.x < a.mn.x ||
             a.mx.y < b.mn.y || b.mx.y < a.mn.y ||
             a.mx.z < b.mn.z || b.mx.z < a.mn.z);
}

static bool rayTriIntersect(const Vec3& ro, const Vec3& rd, const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    const double EPS = 1e-12;
    Vec3 e1 = v1 - v0;
    Vec3 e2 = v2 - v0;
    Vec3 p = cross(rd, e2);
    double det = dot(e1, p);
    if (std::abs(det) < EPS) return false;
    double invDet = 1.0 / det;
    Vec3 t = ro - v0;
    double u = dot(t, p) * invDet;
    if (u < 0.0 || u > 1.0) return false;
    Vec3 q = cross(t, e1);
    double v = dot(rd, q) * invDet;
    if (v < 0.0 || u + v > 1.0) return false;
    double tt = dot(e2, q) * invDet;
    return tt > EPS;
}

static bool pointInsideMesh(const Mesh& m, const Vec3& p) {
    // Odd-even ray test in +X direction.
    Vec3 rd(1.0, 0.1234567, 0.7654321);
    int hits = 0;
    for (const auto& t : m.f) {
        if (rayTriIntersect(p, rd, m.v[t.a], m.v[t.b], m.v[t.c])) ++hits;
    }
    return (hits & 1) != 0;
}

static Vec3 triCentroid(const Mesh& m, const Tri& t) {
    const Vec3& A = m.v[t.a];
    const Vec3& B = m.v[t.b];
    const Vec3& C = m.v[t.c];
    return Vec3((A.x + B.x + C.x) / 3.0, (A.y + B.y + C.y) / 3.0, (A.z + B.z + C.z) / 3.0);
}

static Mesh concatMeshes(const Mesh& a, const Mesh& b) {
    Mesh out;
    out.v = a.v;
    out.v.insert(out.v.end(), b.v.begin(), b.v.end());
    out.f = a.f;
    int off = static_cast<int>(a.v.size());
    for (const auto& t : b.f) out.f.push_back({t.a + off, t.b + off, t.c + off});
    return out;
}

static Mesh keepExteriorFaces(const Mesh& src, const Mesh& other) {
    Mesh out;
    out.v = src.v;
    out.f.reserve(src.f.size());
    for (const auto& t : src.f) {
        Vec3 c = triCentroid(src, t);
        if (!pointInsideMesh(other, c)) out.f.push_back(t);
    }
    return out;
}

static Mesh unionApprox(const Mesh& a, const Mesh& b) {
    // Fast path for separated meshes: exact by concatenation.
    AABB ba = bounds(a), bb = bounds(b);
    if (!aabbIntersect(ba, bb)) return concatMeshes(a, b);

    // Containment fast paths.
    Vec3 ca = triCentroid(a, a.f.empty() ? Tri{0,0,0} : a.f[0]);
    Vec3 cb = triCentroid(b, b.f.empty() ? Tri{0,0,0} : b.f[0]);
    bool aInB = !a.f.empty() && pointInsideMesh(b, ca);
    bool bInA = !b.f.empty() && pointInsideMesh(a, cb);
    if (aInB && !bInA) return b;
    if (bInA && !aInB) return a;

    // Surface-filter union approximation: keep triangles not inside the opposite mesh.
    Mesh aExt = keepExteriorFaces(a, b);
    Mesh bExt = keepExteriorFaces(b, a);

    Mesh out;
    out.v = aExt.v;
    out.v.insert(out.v.end(), bExt.v.begin(), bExt.v.end());
    out.f = aExt.f;
    int off = static_cast<int>(aExt.v.size());
    for (const auto& t : bExt.f) out.f.push_back({t.a + off, t.b + off, t.c + off});

    // Degenerate fallback.
    if (out.f.empty()) return concatMeshes(a, b);
    return out;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.v.resize(VA);
    A.f.resize(FA);
    B.v.resize(VB);
    B.f.resize(FB);

    for (int i = 0; i < VA; ++i) std::cin >> A.v[i].x >> A.v[i].y >> A.v[i].z;
    for (int i = 0; i < FA; ++i) std::cin >> A.f[i].a >> A.f[i].b >> A.f[i].c;
    for (int i = 0; i < VB; ++i) std::cin >> B.v[i].x >> B.v[i].y >> B.v[i].z;
    for (int i = 0; i < FB; ++i) std::cin >> B.f[i].a >> B.f[i].b >> B.f[i].c;

    Mesh U = unionApprox(A, B);

    std::cout << U.v.size() << ' ' << U.f.size() << '\n';
    std::cout << std::setprecision(17);
    for (const auto& p : U.v) {
        std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    }
    for (const auto& t : U.f) {
        std::cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
    }

    return 0;
}
