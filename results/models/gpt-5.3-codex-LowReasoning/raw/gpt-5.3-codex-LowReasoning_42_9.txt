OpCapability Geometry
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%uint_4 = OpConstant %uint 4
%arr_gl_PerVertex_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr_gl_PerVertex_in_4 = OpTypePointer Input %arr_gl_PerVertex_in_4
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_4 Input
%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3float_4 = OpTypePointer Input %arr_v3float_4
%inWorldPos = OpVariable %ptr_Input_arr_v3float_4 Input
%inNormal = OpVariable %ptr_Input_arr_v3float_4 Input
%inColor = OpVariable %ptr_Input_arr_v3float_4 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_0_025 = OpConstant %float 0.025
%float_n1 = OpConstant %float -1

%main = OpFunction %void None %func
%label = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr

%d0 = OpFSub %v4float %p2 %p0
%d1 = OpFSub %v4float %p3 %p1
%t0_xy = OpVectorShuffle %v2float %d0 %d0 0 1
%t1_xy = OpVectorShuffle %v2float %d1 %d1 0 1
%t0 = OpExtInst %v2float %1 Normalize %t0_xy
%t1 = OpExtInst %v2float %1 Normalize %t1_xy

%t0x = OpCompositeExtract %float %t0 0
%t0y = OpCompositeExtract %float %t0 1
%t1x = OpCompositeExtract %float %t1 0
%t1y = OpCompositeExtract %float %t1 1
%nt0y = OpFMul %float %t0y %float_n1
%nt1y = OpFMul %float %t1y %float_n1
%perp0 = OpCompositeConstruct %v2float %nt0y %t0x
%perp1 = OpCompositeConstruct %v2float %nt1y %t1x

%off0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
%off1 = OpVectorTimesScalar %v2float %perp1 %float_0_025

%sxy = OpVectorShuffle %v2float %p1 %p1 0 1
%exy = OpVectorShuffle %v2float %p2 %p2 0 1
%s_plus = OpFAdd %v2float %sxy %off0
%s_minus = OpFSub %v2float %sxy %off0
%e_plus = OpFAdd %v2float %exy %off1
%e_minus = OpFSub %v2float %exy %off1

%p1z = OpCompositeExtract %float %p1 2
%p1w = OpCompositeExtract %float %p1 3
%p2z = OpCompositeExtract %float %p2 2
%p2w = OpCompositeExtract %float %p2 3

%spx = OpCompositeExtract %float %s_plus 0
%spy = OpCompositeExtract %float %s_plus 1
%smx = OpCompositeExtract %float %s_minus 0
%smy = OpCompositeExtract %float %s_minus 1
%epx = OpCompositeExtract %float %e_plus 0
%epy = OpCompositeExtract %float %e_plus 1
%emx = OpCompositeExtract %float %e_minus 0
%emy = OpCompositeExtract %float %e_minus 1

%v0 = OpCompositeConstruct %v4float %spx %spy %p1z %p1w
%v1 = OpCompositeConstruct %v4float %smx %smy %p1z %p1w
%v2 = OpCompositeConstruct %v4float %epx %epy %p2z %p2w
%v3 = OpCompositeConstruct %v4float %emx %emy %p2z %p2w

%c1_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_1
%c2_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

OpStore %out_pos_ptr %v0
OpStore %outColor %c1
OpEmitVertex

OpStore %out_pos_ptr %v1
OpStore %outColor %c1
OpEmitVertex

OpStore %out_pos_ptr %v2
OpStore %outColor %c2
OpEmitVertex

OpStore %out_pos_ptr %v3
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
