using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(double s, Vec3 v) => new Vec3(s * v.X, s * v.Y, s * v.Z);
        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X
        );
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
    }

    struct Quat
    {
        public double W, X, Y, Z;
        public Quat(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
        public Quat Normalized()
        {
            double n = Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
            if (n <= 1e-18) return new Quat(1, 0, 0, 0);
            return new Quat(W / n, X / n, Y / n, Z / n);
        }
        public Vec3 Rotate(Vec3 v)
        {
            Vec3 qv = new Vec3(X, Y, Z);
            Vec3 t = 2.0 * Vec3.Cross(qv, v);
            return v + (W * t) + Vec3.Cross(qv, t);
        }
    }

    class Orientation
    {
        public Quat Q;
        public double Ex, Ey, Ez; // extents after rotation
        public long GridCount;
    }

    static readonly CultureInfo CI = CultureInfo.InvariantCulture;

    public static void Main()
    {
        var data = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(data))
        {
            Console.WriteLine(0);
            return;
        }

        var toks = data.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;

        int V = int.Parse(toks[p++], CI);
        int F = int.Parse(toks[p++], CI);

        var verts = new Vec3[V];
        for (int i = 0; i < V; i++)
        {
            double x = double.Parse(toks[p++], CI);
            double y = double.Parse(toks[p++], CI);
            double z = double.Parse(toks[p++], CI);
            verts[i] = new Vec3(x, y, z);
        }

        var faces = new (int a, int b, int c)[F];
        for (int i = 0; i < F; i++)
        {
            int a = int.Parse(toks[p++], CI);
            int b = int.Parse(toks[p++], CI);
            int c = int.Parse(toks[p++], CI);
            faces[i] = (a, b, c);
        }

        double W = double.Parse(toks[p++], CI);
        double H = double.Parse(toks[p++], CI);
        double D = double.Parse(toks[p++], CI);

        if (W <= 0 || H <= 0 || D <= 0 || V == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Center mesh at origin to make rotation handling robust.
        Vec3 centroid = new Vec3(0, 0, 0);
        for (int i = 0; i < V; i++) centroid += verts[i];
        centroid = (1.0 / V) * centroid;
        for (int i = 0; i < V; i++) verts[i] = verts[i] - centroid;

        double volume = Math.Abs(ComputeVolume(verts, faces));
        if (volume < 1e-12) volume = 1e-12;

        var baseAabb = ComputeAabb(verts);
        double dx0 = Math.Max(1e-9, baseAabb.max.X - baseAabb.min.X);
        double dy0 = Math.Max(1e-9, baseAabb.max.Y - baseAabb.min.Y);
        double dz0 = Math.Max(1e-9, baseAabb.max.Z - baseAabb.min.Z);

        double containerVol = W * H * D;
        double complexity = (double)V + 2.0 * F;
        double scaleScore = Math.Max(W, Math.Max(H, D));

        // Adaptive strategy tiers.
        int orientationBudget;
        long maxOutput;
        if (scaleScore <= 20 && complexity <= 300)
        {
            orientationBudget = 220; // Trivial/small: more orientation exploration.
            maxOutput = 2_000_000;
        }
        else if (scaleScore <= 100 && complexity <= 5000)
        {
            orientationBudget = 120; // Medium/Large.
            maxOutput = 1_200_000;
        }
        else
        {
            orientationBudget = 40;  // Extreme scale: prioritize speed.
            maxOutput = 600_000;
        }

        var candidates = BuildCandidateOrientations(orientationBudget);
        var orientations = new List<Orientation>(candidates.Count);

        foreach (var q0 in candidates)
        {
            var q = q0.Normalized();
            var aabb = ComputeRotatedAabb(verts, q);
            double ex = Math.Max(1e-9, aabb.max.X - aabb.min.X);
            double ey = Math.Max(1e-9, aabb.max.Y - aabb.min.Y);
            double ez = Math.Max(1e-9, aabb.max.Z - aabb.min.Z);

            long nx = (long)Math.Floor(W / ex + 1e-12);
            long ny = (long)Math.Floor(H / ey + 1e-12);
            long nz = (long)Math.Floor(D / ez + 1e-12);
            if (nx <= 0 || ny <= 0 || nz <= 0) continue;

            long c = SafeMul(nx, SafeMul(ny, nz));
            orientations.Add(new Orientation { Q = q, Ex = ex, Ey = ey, Ez = ez, GridCount = c });
        }

        if (orientations.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Pick best orientation by primary count and secondary fill ratio.
        orientations.Sort((a, b) =>
        {
            int cmp = b.GridCount.CompareTo(a.GridCount);
            if (cmp != 0) return cmp;
            double fa = (Math.Floor(W / a.Ex) * a.Ex) * (Math.Floor(H / a.Ey) * a.Ey) * (Math.Floor(D / a.Ez) * a.Ez);
            double fb = (Math.Floor(W / b.Ex) * b.Ex) * (Math.Floor(H / b.Ey) * b.Ey) * (Math.Floor(D / b.Ez) * b.Ez);
            return fb.CompareTo(fa);
        });

        var best = orientations[0];
        long gx = (long)Math.Floor(W / best.Ex + 1e-12);
        long gy = (long)Math.Floor(H / best.Ey + 1e-12);
        long gz = (long)Math.Floor(D / best.Ez + 1e-12);
        long count = SafeMul(gx, SafeMul(gy, gz));

        // Safety cap to avoid pathological stdout size/timeouts.
        if (count > maxOutput) count = maxOutput;

        Console.WriteLine(count.ToString(CI));
        if (count == 0) return;

        // Translate on a regular non-overlapping grid. Conservative and always valid.
        // Coordinates are the placement anchor; we use lower corner of oriented AABB cell.
        long emitted = 0;
        for (long iz = 0; iz < gz && emitted < count; iz++)
        {
            double tz = iz * best.Ez;
            for (long iy = 0; iy < gy && emitted < count; iy++)
            {
                double ty = iy * best.Ey;
                for (long ix = 0; ix < gx && emitted < count; ix++)
                {
                    double tx = ix * best.Ex;
                    Console.WriteLine(string.Create(CI, $"{tx:0.######} {ty:0.######} {tz:0.######} {best.Q.W:0.######} {best.Q.X:0.######} {best.Q.Y:0.######} {best.Q.Z:0.######}"));
                    emitted++;
                }
            }
        }
    }

    static (Vec3 min, Vec3 max) ComputeAabb(Vec3[] verts)
    {
        double minx = double.PositiveInfinity, miny = double.PositiveInfinity, minz = double.PositiveInfinity;
        double maxx = double.NegativeInfinity, maxy = double.NegativeInfinity, maxz = double.NegativeInfinity;
        for (int i = 0; i < verts.Length; i++)
        {
            var v = verts[i];
            if (v.X < minx) minx = v.X; if (v.X > maxx) maxx = v.X;
            if (v.Y < miny) miny = v.Y; if (v.Y > maxy) maxy = v.Y;
            if (v.Z < minz) minz = v.Z; if (v.Z > maxz) maxz = v.Z;
        }
        return (new Vec3(minx, miny, minz), new Vec3(maxx, maxy, maxz));
    }

    static (Vec3 min, Vec3 max) ComputeRotatedAabb(Vec3[] verts, Quat q)
    {
        double minx = double.PositiveInfinity, miny = double.PositiveInfinity, minz = double.PositiveInfinity;
        double maxx = double.NegativeInfinity, maxy = double.NegativeInfinity, maxz = double.NegativeInfinity;
        for (int i = 0; i < verts.Length; i++)
        {
            var v = q.Rotate(verts[i]);
            if (v.X < minx) minx = v.X; if (v.X > maxx) maxx = v.X;
            if (v.Y < miny) miny = v.Y; if (v.Y > maxy) maxy = v.Y;
            if (v.Z < minz) minz = v.Z; if (v.Z > maxz) maxz = v.Z;
        }
        return (new Vec3(minx, miny, minz), new Vec3(maxx, maxy, maxz));
    }

    static double ComputeVolume(Vec3[] verts, (int a, int b, int c)[] faces)
    {
        double vol6 = 0.0;
        for (int i = 0; i < faces.Length; i++)
        {
            var f = faces[i];
            if ((uint)f.a >= (uint)verts.Length || (uint)f.b >= (uint)verts.Length || (uint)f.c >= (uint)verts.Length) continue;
            Vec3 A = verts[f.a], B = verts[f.b], C = verts[f.c];
            vol6 += Vec3.Dot(A, Vec3.Cross(B, C));
        }
        return vol6 / 6.0;
    }

    static long SafeMul(long a, long b)
    {
        if (a <= 0 || b <= 0) return 0;
        if (a > long.MaxValue / b) return long.MaxValue;
        return a * b;
    }

    static List<Quat> BuildCandidateOrientations(int budget)
    {
        var list = new List<Quat>(budget);

        // Deterministic seed orientations: identity + principal quarter-turn rotations.
        list.Add(new Quat(1, 0, 0, 0));
        double s2 = Math.Sqrt(0.5);
        list.Add(new Quat(s2, s2, 0, 0));
        list.Add(new Quat(s2, -s2, 0, 0));
        list.Add(new Quat(s2, 0, s2, 0));
        list.Add(new Quat(s2, 0, -s2, 0));
        list.Add(new Quat(s2, 0, 0, s2));
        list.Add(new Quat(s2, 0, 0, -s2));

        // Fibonacci-like deterministic orientation sampling over S^3.
        for (int i = 0; list.Count < budget; i++)
        {
            double u1 = Frac(i * 0.6180339887498949);
            double u2 = Frac(i * 0.7548776662466927 + 0.1234567);
            double u3 = Frac(i * 0.5698402909980532 + 0.7654321);

            double r1 = Math.Sqrt(1.0 - u1);
            double r2 = Math.Sqrt(u1);
            double t1 = 2.0 * Math.PI * u2;
            double t2 = 2.0 * Math.PI * u3;

            double x = r1 * Math.Sin(t1);
            double y = r1 * Math.Cos(t1);
            double z = r2 * Math.Sin(t2);
            double w = r2 * Math.Cos(t2);
            list.Add(new Quat(w, x, y, z));
        }

        return list;
    }

    static double Frac(double x)
    {
        return x - Math.Floor(x);
    }
}
