using System;
using System.Collections.Generic;
using System.Text;

public static class Program
{
    struct Pos
    {
        public int X;
        public int Y;
        public Pos(int x, int y) { X = x; Y = y; }
    }

    sealed class NumberCell
    {
        public int X;
        public int Y;
        public int Need; // mines still needed around this number
        public int UnknownCount;
        public int[] UnknownNeighbors = Array.Empty<int>();
    }

    static readonly int[] DX = { -1, 0, 1, -1, 1, -1, 0, 1 };
    static readonly int[] DY = { -1, -1, -1, 0, 0, 1, 1, 1 };

    public static void Main()
    {
        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.WriteLine(0);
            return;
        }

        string[] hw = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int h = int.Parse(hw[0]);
        int w = int.Parse(hw[1]);

        char[][] board = new char[h][];
        for (int y = 0; y < h; y++)
        {
            string? line = Console.ReadLine() ?? string.Empty;
            if (line.Length < w) line = line.PadRight(w, ' ');
            else if (line.Length > w) line = line.Substring(0, w);
            board[y] = line.ToCharArray();
        }

        bool[] safe = new bool[h * w];
        bool[] mine = new bool[h * w];

        // Phase 1: ultra-fast deterministic local rules (single-number constraints).
        ApplyLocalRules(board, h, w, safe, mine);

        // Determine complexity and decide whether to run heavier boundary CSP.
        int frontierUnknowns = CountFrontierUnknowns(board, h, w, safe, mine);
        bool runCsp = ShouldRunCsp(h, w, frontierUnknowns);

        if (runCsp)
        {
            ApplyBoundaryCsp(board, h, w, safe, mine);
        }

        // Output provably safe unknown cells only.
        var outList = new List<Pos>();
        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                int id = Id(x, y, w);
                if (board[y][x] == ' ' && safe[id] && !mine[id])
                {
                    outList.Add(new Pos(x, y));
                }
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(outList.Count.ToString());
        for (int i = 0; i < outList.Count; i++)
        {
            sb.Append(outList[i].X).Append(' ').Append(outList[i].Y).Append('\n');
        }
        Console.Write(sb.ToString());
    }

    static int Id(int x, int y, int w) => y * w + x;

    static bool InBounds(int x, int y, int w, int h)
        => (uint)x < (uint)w && (uint)y < (uint)h;

    static bool IsNumber(char c) => c >= '0' && c <= '8';

    static void ApplyLocalRules(char[][] board, int h, int w, bool[] safe, bool[] mine)
    {
        // Iterate to fixed point because a newly deduced mine/safe can enable other cells.
        bool changed;
        do
        {
            changed = false;
            for (int y = 0; y < h; y++)
            {
                for (int x = 0; x < w; x++)
                {
                    char c = board[y][x];
                    if (!IsNumber(c)) continue;

                    int number = c - '0';
                    int flagged = 0;
                    int knownMines = 0;
                    int unknownCount = 0;

                    Span<int> unknownIds = stackalloc int[8];

                    for (int k = 0; k < 8; k++)
                    {
                        int nx = x + DX[k], ny = y + DY[k];
                        if (!InBounds(nx, ny, w, h)) continue;

                        char nc = board[ny][nx];
                        int nid = Id(nx, ny, w);

                        if (nc == '*') flagged++;
                        else if (mine[nid]) knownMines++;

                        if (nc == ' ' && !mine[nid] && !safe[nid])
                        {
                            unknownIds[unknownCount++] = nid;
                        }
                    }

                    int need = number - flagged - knownMines;
                    if (need < 0) continue; // inconsistent input; ignore.

                    if (unknownCount == 0) continue;

                    if (need == 0)
                    {
                        for (int i = 0; i < unknownCount; i++)
                        {
                            int id = unknownIds[i];
                            if (!safe[id])
                            {
                                safe[id] = true;
                                changed = true;
                            }
                        }
                    }
                    else if (need == unknownCount)
                    {
                        for (int i = 0; i < unknownCount; i++)
                        {
                            int id = unknownIds[i];
                            if (!mine[id])
                            {
                                mine[id] = true;
                                changed = true;
                            }
                        }
                    }
                }
            }
        } while (changed);
    }

    static int CountFrontierUnknowns(char[][] board, int h, int w, bool[] safe, bool[] mine)
    {
        int count = 0;
        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                if (board[y][x] != ' ') continue;
                int id = Id(x, y, w);
                if (safe[id] || mine[id]) continue;
                if (TouchesNumber(board, h, w, x, y)) count++;
            }
        }
        return count;
    }

    static bool TouchesNumber(char[][] board, int h, int w, int x, int y)
    {
        for (int k = 0; k < 8; k++)
        {
            int nx = x + DX[k], ny = y + DY[k];
            if (InBounds(nx, ny, w, h) && IsNumber(board[ny][nx])) return true;
        }
        return false;
    }

    static bool ShouldRunCsp(int h, int w, int frontierUnknowns)
    {
        long cells = (long)h * w;

        // Adaptive policy:
        // - Always allow CSP on small/medium boards.
        // - On very large boards, only run when boundary remains manageable.
        if (cells <= 2_000_000) return frontierUnknowns <= 12_000;
        if (cells <= 20_000_000) return frontierUnknowns <= 7_000;
        return frontierUnknowns <= 3_500;
    }

    static void ApplyBoundaryCsp(char[][] board, int h, int w, bool[] safe, bool[] mine)
    {
        // Build boundary unknown variable indices.
        var varIds = new List<int>(1024);
        var varIndex = new Dictionary<int, int>(2048);

        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                if (board[y][x] != ' ') continue;
                int id = Id(x, y, w);
                if (safe[id] || mine[id]) continue;
                if (!TouchesNumber(board, h, w, x, y)) continue;

                int idx = varIds.Count;
                varIds.Add(id);
                varIndex[id] = idx;
            }
        }

        if (varIds.Count == 0) return;

        // Build number constraints over boundary vars.
        var constraints = new List<NumberCell>(1024);

        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                char c = board[y][x];
                if (!IsNumber(c)) continue;

                int number = c - '0';
                int flagged = 0;
                int knownMines = 0;

                Span<int> temp = stackalloc int[8];
                int tc = 0;

                for (int k = 0; k < 8; k++)
                {
                    int nx = x + DX[k], ny = y + DY[k];
                    if (!InBounds(nx, ny, w, h)) continue;
                    char nc = board[ny][nx];
                    int nid = Id(nx, ny, w);

                    if (nc == '*') flagged++;
                    else if (mine[nid]) knownMines++;

                    if (nc == ' ' && !safe[nid] && !mine[nid] && varIndex.TryGetValue(nid, out int vidx))
                    {
                        temp[tc++] = vidx;
                    }
                }

                int need = number - flagged - knownMines;
                if (need < 0) continue;
                if (tc == 0) continue;

                int[] vars = new int[tc];
                for (int i = 0; i < tc; i++) vars[i] = temp[i];

                constraints.Add(new NumberCell { X = x, Y = y, Need = need, UnknownCount = tc, UnknownNeighbors = vars });
            }
        }

        if (constraints.Count == 0) return;

        // Build variable->constraint adjacency and componentize.
        var varToCons = new List<int>[varIds.Count];
        for (int i = 0; i < varIds.Count; i++) varToCons[i] = new List<int>(4);
        for (int ci = 0; ci < constraints.Count; ci++)
        {
            int[] vars = constraints[ci].UnknownNeighbors;
            for (int j = 0; j < vars.Length; j++) varToCons[vars[j]].Add(ci);
        }

        bool[] seenVar = new bool[varIds.Count];

        // Global budget to keep worst-case bounded on huge inputs.
        long globalNodeBudget = 3_000_000;

        for (int start = 0; start < varIds.Count; start++)
        {
            if (seenVar[start]) continue;

            // BFS component.
            var compVars = new List<int>();
            var compConsSet = new HashSet<int>();
            var q = new Queue<int>();
            q.Enqueue(start);
            seenVar[start] = true;

            while (q.Count > 0)
            {
                int v = q.Dequeue();
                compVars.Add(v);
                var cl = varToCons[v];
                for (int i = 0; i < cl.Count; i++)
                {
                    int ci = cl[i];
                    if (!compConsSet.Add(ci)) continue;
                    int[] nvars = constraints[ci].UnknownNeighbors;
                    for (int k = 0; k < nvars.Length; k++)
                    {
                        int nv = nvars[k];
                        if (!seenVar[nv])
                        {
                            seenVar[nv] = true;
                            q.Enqueue(nv);
                        }
                    }
                }
            }

            // Skip components too large for exact enumeration under budget.
            if (compVars.Count > 28) continue;

            var compCons = new List<int>(compConsSet);
            if (compCons.Count == 0) continue;

            // Reindex vars to local 0..n-1
            var localPos = new Dictionary<int, int>(compVars.Count);
            for (int i = 0; i < compVars.Count; i++) localPos[compVars[i]] = i;

            int ccount = compCons.Count;
            int[][] cVars = new int[ccount][];
            int[] need = new int[ccount];
            int[] deg = new int[ccount];
            int[] rem = new int[ccount];

            for (int i = 0; i < ccount; i++)
            {
                NumberCell nc = constraints[compCons[i]];
                need[i] = nc.Need;
                int[] src = nc.UnknownNeighbors;
                int[] dst = new int[src.Length];
                for (int j = 0; j < src.Length; j++) dst[j] = localPos[src[j]];
                cVars[i] = dst;
                deg[i] = dst.Length;
                rem[i] = dst.Length;
            }

            // var->local constraints
            var vToC = new List<int>[compVars.Count];
            for (int i = 0; i < compVars.Count; i++) vToC[i] = new List<int>(4);
            for (int ci = 0; ci < ccount; ci++)
            {
                int[] arr = cVars[ci];
                for (int j = 0; j < arr.Length; j++) vToC[arr[j]].Add(ci);
            }

            int n = compVars.Count;
            int[] assignment = new int[n]; // -1 unset, 0 safe, 1 mine
            for (int i = 0; i < n; i++) assignment[i] = -1;

            long solutions = 0;
            long[] mineCount = new long[n];
            long nodes = 0;

            // Heuristic order: highest degree first.
            int[] order = new int[n];
            for (int i = 0; i < n; i++) order[i] = i;
            Array.Sort(order, (a, b) => vToC[b].Count.CompareTo(vToC[a].Count));

            bool aborted = false;

            void Dfs(int p)
            {
                if (aborted) return;
                nodes++;
                if (nodes > 600_000 || globalNodeBudget <= 0)
                {
                    aborted = true;
                    return;
                }
                globalNodeBudget--;

                if (p == n)
                {
                    solutions++;
                    for (int i = 0; i < n; i++) if (assignment[i] == 1) mineCount[i]++;
                    return;
                }

                int v = order[p];

                // Try safe then mine (ordering does not affect correctness).
                for (int val = 0; val <= 1; val++)
                {
                    assignment[v] = val;

                    bool ok = true;
                    var related = vToC[v];
                    for (int i = 0; i < related.Count; i++)
                    {
                        int ci = related[i];
                        if (val == 1) need[ci]--;
                        rem[ci]--;

                        // Bounds: 0 <= need <= rem
                        if (need[ci] < 0 || need[ci] > rem[ci]) ok = false;
                    }

                    if (ok) Dfs(p + 1);

                    for (int i = 0; i < related.Count; i++)
                    {
                        int ci = related[i];
                        rem[ci]++;
                        if (val == 1) need[ci]++;
                    }

                    assignment[v] = -1;
                    if (aborted) return;
                }
            }

            Dfs(0);

            if (aborted || solutions == 0) continue;

            // If variable is mine in 0 solutions => always safe.
            for (int lv = 0; lv < n; lv++)
            {
                if (mineCount[lv] == 0)
                {
                    int globalVar = compVars[lv];
                    int id = varIds[globalVar];
                    safe[id] = true;
                }
            }
        }
    }
}
