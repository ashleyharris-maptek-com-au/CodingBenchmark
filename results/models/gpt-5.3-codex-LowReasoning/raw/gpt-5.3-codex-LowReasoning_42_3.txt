OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450
OpName %main "main"
OpName %gl_in "gl_in"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%uint_2 = OpConstant %uint 2
%arr_pv_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_2
%gl_in = OpVariable %ptr_in_arr_pv Input
%arr_v3_in_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in_2
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_thickness = OpConstant %float 0.0299999993

%ext = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %fn
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_n = OpExtInst %v2float %ext Normalize %perp_raw
%perp = OpVectorTimesScalar %v2float %perp_n %float_thickness

%p0p = OpFAdd %v2float %p0_xy %perp
%p0m = OpFSub %v2float %p0_xy %perp
%p1p = OpFAdd %v2float %p1_xy %perp
%p1m = OpFSub %v2float %p1_xy %perp

%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

%p0p_x = OpCompositeExtract %float %p0p 0
%p0p_y = OpCompositeExtract %float %p0p 1
%p0m_x = OpCompositeExtract %float %p0m 0
%p0m_y = OpCompositeExtract %float %p0m 1
%p1p_x = OpCompositeExtract %float %p1p 0
%p1p_y = OpCompositeExtract %float %p1p 1
%p1m_x = OpCompositeExtract %float %p1m 0
%p1m_y = OpCompositeExtract %float %p1m 1

%vtx0 = OpCompositeConstruct %v4float %p0p_x %p0p_y %p0_z %p0_w
%vtx1 = OpCompositeConstruct %v4float %p0m_x %p0m_y %p0_z %p0_w
%vtx2 = OpCompositeConstruct %v4float %p1p_x %p1p_y %p1_z %p1_w
%vtx3 = OpCompositeConstruct %v4float %p1m_x %p1m_y %p1_z %p1_w

%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %pos_out_ptr %vtx0
OpStore %outColor %c0
OpEmitVertex

OpStore %pos_out_ptr %vtx1
OpStore %outColor %c0
OpEmitVertex

OpStore %pos_out_ptr %vtx2
OpStore %outColor %c1
OpEmitVertex

OpStore %pos_out_ptr %vtx3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
