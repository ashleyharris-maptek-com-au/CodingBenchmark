// Mission-control scheduler for parallel workstations.
declare const require: any;
const fs = require('fs');

interface TaskIn {
  id: number;
  duration: number;
  deps: number[];
  deadline: number;
  weight: number;
}

interface InputData {
  machineCount: number;
  tasks: TaskIn[];
}

interface SchedItem {
  id: number;
  start: number;
  machine: number;
}

class Heap<T> {
  private a: T[] = [];
  constructor(private readonly less: (x: T, y: T) => boolean) {}
  size(): number { return this.a.length; }
  peek(): T | undefined { return this.a[0]; }
  push(v: T): void {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      const t = a[i]; a[i] = a[p]; a[p] = t;
      i = p;
    }
  }
  pop(): T | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const out = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      for (;;) {
        const l = i * 2 + 1;
        const r = l + 1;
        let b = i;
        if (l < a.length && this.less(a[l], a[b])) b = l;
        if (r < a.length && this.less(a[r], a[b])) b = r;
        if (b === i) break;
        const t = a[i]; a[i] = a[b]; a[b] = t;
        i = b;
      }
    }
    return out;
  }
}

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const tasks = data.tasks;
const n = tasks.length;
const m = Math.max(1, data.machineCount | 0);

const idToIdx = new Map<number, number>();
for (let i = 0; i < n; i++) idToIdx.set(tasks[i].id, i);

const indeg = new Array<number>(n).fill(0);
const succ: number[][] = Array.from({ length: n }, () => []);
const depIdxs: number[][] = Array.from({ length: n }, () => []);

for (let i = 0; i < n; i++) {
  const deps = tasks[i].deps || [];
  for (let k = 0; k < deps.length; k++) {
    const d = idToIdx.get(deps[k]);
    if (d === undefined) continue;
    indeg[i]++;
    succ[d].push(i);
    depIdxs[i].push(d);
  }
}

// Longest-path urgency estimate from each node to sink.
const outdeg = succ.map(s => s.length);
const lp = new Array<number>(n).fill(0);
const q: number[] = [];
for (let i = 0; i < n; i++) if (outdeg[i] === 0) {
  lp[i] = Math.max(0, tasks[i].duration);
  q.push(i);
}
for (let qi = 0; qi < q.length; qi++) {
  const v = q[qi];
  const deps = depIdxs[v];
  for (let j = 0; j < deps.length; j++) {
    const p = deps[j];
    const cand = Math.max(0, tasks[p].duration) + lp[v];
    if (cand > lp[p]) lp[p] = cand;
    outdeg[p]--;
    if (outdeg[p] === 0) q.push(p);
  }
}

const finish = new Array<number>(n).fill(0);
const depReady = new Array<number>(n).fill(0);

const ready = new Heap<number>((ia, ib) => {
  const a = tasks[ia], b = tasks[ib];
  const sa = (a.weight * 1000) / Math.max(1, a.duration) + lp[ia] * 0.5 - a.deadline * 0.01;
  const sb = (b.weight * 1000) / Math.max(1, b.duration) + lp[ib] * 0.5 - b.deadline * 0.01;
  if (sa !== sb) return sa > sb;
  if (a.deadline !== b.deadline) return a.deadline < b.deadline;
  if (a.weight !== b.weight) return a.weight > b.weight;
  if (a.duration !== b.duration) return a.duration > b.duration;
  return a.id < b.id;
});

for (let i = 0; i < n; i++) if (indeg[i] === 0) ready.push(i);

const machines = new Heap<{ t: number; machine: number }>((x, y) => {
  if (x.t !== y.t) return x.t < y.t;
  return x.machine < y.machine;
});
for (let machine = 0; machine < m; machine++) machines.push({ t: 0, machine });

const schedule: SchedItem[] = [];
let done = 0;

while (done < n && ready.size() > 0) {
  const taskIdx = ready.pop()!;
  const task = tasks[taskIdx];
  const slot = machines.pop()!;
  const start = Math.max(slot.t, depReady[taskIdx]);
  const end = start + Math.max(0, task.duration);

  schedule.push({ id: task.id, start, machine: slot.machine });
  finish[taskIdx] = end;
  machines.push({ t: end, machine: slot.machine });

  const nexts = succ[taskIdx];
  for (let i = 0; i < nexts.length; i++) {
    const v = nexts[i];
    if (end > depReady[v]) depReady[v] = end;
    indeg[v]--;
    if (indeg[v] === 0) ready.push(v);
  }
  done++;
}

if (done < n) {
  // Cycle or malformed deps fallback: output deterministic order by id.
  const order = tasks.map(t => t.id).sort((a, b) => a - b);
  process.stdout.write(JSON.stringify({ order }));
} else {
  process.stdout.write(JSON.stringify({ schedule }));
}
