use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

fn point_in_polygon(p: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    let n = poly.len();
    if n < 3 {
        return false;
    }
    let mut j = n - 1;
    for i in 0..n {
        let pi = poly[i];
        let pj = poly[j];
        let intersects = ((pi.y > p.y) != (pj.y > p.y))
            && (p.x
                < (pj.x - pi.x) * (p.y - pi.y) / ((pj.y - pi.y).abs().max(1e-18) * (if pj.y >= pi.y { 1.0 } else { -1.0 }))
                    + pi.x);
        if intersects {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn dist_point_segment(p: Pt, a: Pt, b: Pt) -> f64 {
    let vx = b.x - a.x;
    let vy = b.y - a.y;
    let wx = p.x - a.x;
    let wy = p.y - a.y;
    let c1 = vx * wx + vy * wy;
    if c1 <= 0.0 {
        return ((p.x - a.x).powi(2) + (p.y - a.y).powi(2)).sqrt();
    }
    let c2 = vx * vx + vy * vy;
    if c2 <= 1e-18 {
        return ((p.x - a.x).powi(2) + (p.y - a.y).powi(2)).sqrt();
    }
    let t = (c1 / c2).clamp(0.0, 1.0);
    let px = a.x + t * vx;
    let py = a.y + t * vy;
    ((p.x - px).powi(2) + (p.y - py).powi(2)).sqrt()
}

fn near_boundary(p: Pt, poly: &[Pt], eps: f64) -> bool {
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if dist_point_segment(p, a, b) <= eps {
            return true;
        }
    }
    false
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("read stdin");
    let mut it = input.split_whitespace();

    let nv: usize = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let y = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        poly.push(Pt { x, y });
    }

    let sx = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let sy = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let mut sz = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(-1.0);

    if sz.abs() < 1e-9 {
        sz = -1.0;
    }

    if poly.len() < 3 {
        println!("0");
        return;
    }

    let (mut minx, mut maxx) = (f64::INFINITY, f64::NEG_INFINITY);
    let (mut miny, mut maxy) = (f64::INFINITY, f64::NEG_INFINITY);
    for p in &poly {
        minx = minx.min(p.x);
        maxx = maxx.max(p.x);
        miny = miny.min(p.y);
        maxy = maxy.max(p.y);
    }

    let width = (maxx - minx).max(1e-6);
    let height = (maxy - miny).max(1e-6);
    let diag = (width * width + height * height).sqrt();

    // Standard tetrahedron vertices.
    let verts = [
        (1.0_f64, 0.0_f64, -0.707_f64),
        (-1.0_f64, 0.0_f64, -0.707_f64),
        (0.0_f64, 1.0_f64, 0.707_f64),
        (0.0_f64, -1.0_f64, 0.707_f64),
    ];

    let tx: Vec<f64> = verts.iter().map(|(x, _, z)| x - sx / sz * z).collect();
    let ty: Vec<f64> = verts.iter().map(|(_, y, z)| y - sy / sz * z).collect();

    let base_w = (tx.iter().copied().fold(f64::NEG_INFINITY, f64::max)
        - tx.iter().copied().fold(f64::INFINITY, f64::min))
        .max(1e-6);
    let base_h = (ty.iter().copied().fold(f64::NEG_INFINITY, f64::max)
        - ty.iter().copied().fold(f64::INFINITY, f64::min))
        .max(1e-6);

    // Adaptive strategy: smaller polygons use finer placement, larger polygons use larger scale/step.
    let density = if nv <= 32 {
        28.0
    } else if nv <= 256 {
        20.0
    } else {
        14.0
    };

    let target_step = (diag / density).max(0.25);
    let mut scale = (target_step / base_w.min(base_h)).clamp(0.15, 10.0);
    if nv > 1000 {
        scale = (scale * 1.25).clamp(0.15, 10.0);
    }

    let step_x = (base_w * scale * 0.78).max(1e-4);
    let step_y = (base_h * scale * 0.78).max(1e-4);
    let boundary_eps = 0.9 * step_x.min(step_y);

    let norm = (sx * sx + sy * sy + sz * sz).sqrt().max(1e-9);
    let dx = sx / norm;
    let dy = sy / norm;
    let dz = sz / norm;

    // Move each tetrahedron along sun direction to keep the same shadow while separating in 3D.
    let separation = (2.4 * scale + 0.5).max(1.0);

    let mut centers: Vec<(f64, f64)> = Vec::new();

    let mut y = miny - step_y;
    while y <= maxy + step_y {
        let row_shift = if ((y / step_y).abs() as i64) % 2 == 0 { 0.0 } else { 0.5 * step_x };
        let mut x = minx - step_x + row_shift;
        while x <= maxx + step_x {
            let p = Pt { x, y };
            if point_in_polygon(p, &poly) || near_boundary(p, &poly, boundary_eps) {
                centers.push((x, y));
            }
            x += step_x;
        }
        y += step_y;
    }

    if centers.is_empty() {
        let cx = (minx + maxx) * 0.5;
        let cy = (miny + maxy) * 0.5;
        centers.push((cx, cy));
        scale = scale.max(0.5);
    }

    println!("{}", centers.len());
    for (i, (ax, ay)) in centers.into_iter().enumerate() {
        let t = separation * (i as f64 + 1.0);
        let px = ax + dx * t;
        let py = ay + dy * t;
        let pz = dz * t;

        // Identity rotation quaternion.
        let qw = 1.0;
        let qx = 0.0;
        let qy = 0.0;
        let qz = 0.0;

        println!("{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}", px, py, pz, qw, qx, qy, qz, scale);
    }
}
