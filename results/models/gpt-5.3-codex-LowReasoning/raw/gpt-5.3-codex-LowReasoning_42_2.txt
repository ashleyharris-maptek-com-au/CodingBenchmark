OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %inColor "inColor"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_r = OpConstant %float 0.04
%float_dark = OpConstant %float 0.8
%float_c0 = OpConstant %float 1.0
%float_s0 = OpConstant %float 0.0
%float_c60 = OpConstant %float 0.5
%float_s60 = OpConstant %float 0.8660254
%float_c120 = OpConstant %float -0.5
%float_s120 = OpConstant %float 0.8660254
%float_c180 = OpConstant %float -1.0
%float_s180 = OpConstant %float 0.0
%float_c240 = OpConstant %float -0.5
%float_s240 = OpConstant %float -0.8660254
%float_c300 = OpConstant %float 0.5
%float_s300 = OpConstant %float -0.8660254

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%gl_in = OpVariable %ptr_in_arr_pv Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%center_pos = OpLoad %v4float %in_pos_ptr
%in_col_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%base_col = OpLoad %v3float %in_col_ptr
%edge_col = OpVectorTimesScalar %v3float %base_col %float_dark

%c0x = OpFMul %float %float_r %float_c0
%c0y = OpFMul %float %float_r %float_s0
%off0 = OpCompositeConstruct %v4float %c0x %c0y %float_0 %float_0
%e0 = OpFAdd %v4float %center_pos %off0

%c1x = OpFMul %float %float_r %float_c60
%c1y = OpFMul %float %float_r %float_s60
%off1 = OpCompositeConstruct %v4float %c1x %c1y %float_0 %float_0
%e1 = OpFAdd %v4float %center_pos %off1

%c2x = OpFMul %float %float_r %float_c120
%c2y = OpFMul %float %float_r %float_s120
%off2 = OpCompositeConstruct %v4float %c2x %c2y %float_0 %float_0
%e2 = OpFAdd %v4float %center_pos %off2

%c3x = OpFMul %float %float_r %float_c180
%c3y = OpFMul %float %float_r %float_s180
%off3 = OpCompositeConstruct %v4float %c3x %c3y %float_0 %float_0
%e3 = OpFAdd %v4float %center_pos %off3

%c4x = OpFMul %float %float_r %float_c240
%c4y = OpFMul %float %float_r %float_s240
%off4 = OpCompositeConstruct %v4float %c4x %c4y %float_0 %float_0
%e4 = OpFAdd %v4float %center_pos %off4

%c5x = OpFMul %float %float_r %float_c300
%c5y = OpFMul %float %float_r %float_s300
%off5 = OpCompositeConstruct %v4float %c5x %c5y %float_0 %float_0
%e5 = OpFAdd %v4float %center_pos %off5

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %outColor %base_col
OpStore %out_pos_ptr %center_pos
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e0
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e1
OpEmitVertex
OpEndPrimitive

OpStore %outColor %base_col
OpStore %out_pos_ptr %center_pos
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e1
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e2
OpEmitVertex
OpEndPrimitive

OpStore %outColor %base_col
OpStore %out_pos_ptr %center_pos
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e2
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e3
OpEmitVertex
OpEndPrimitive

OpStore %outColor %base_col

OpStore %out_pos_ptr %e3
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e4
OpEmitVertex
OpEndPrimitive

OpStore %outColor %base_col
OpStore %out_pos_ptr %center_pos
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e4
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e5
OpEmitVertex
OpEndPrimitive

OpStore %outColor %base_col
OpStore %out_pos_ptr %center_pos
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e5
OpEmitVertex
OpStore %outColor %edge_col
OpStore %out_pos_ptr %e0
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
