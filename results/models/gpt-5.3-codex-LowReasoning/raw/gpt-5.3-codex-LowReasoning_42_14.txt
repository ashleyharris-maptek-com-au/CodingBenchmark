OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_3 = OpConstant %float 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%v3_111 = OpConstantComposite %v3float %float_1 %float_1 %float_1

%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%p2 = OpLoad %v3float %p2_ptr
%p4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p4 = OpLoad %v3float %p4_ptr

%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0

%ax = OpCompositeExtract %float %a 0
%ay = OpCompositeExtract %float %a 1
%az = OpCompositeExtract %float %a 2
%bx = OpCompositeExtract %float %b 0
%by = OpCompositeExtract %float %b 1
%bz = OpCompositeExtract %float %b 2

%nx_t1 = OpFMul %float %ay %bz
%nx_t2 = OpFMul %float %az %by
%nx = OpFSub %float %nx_t1 %nx_t2
%ny_t1 = OpFMul %float %az %bx
%ny_t2 = OpFMul %float %ax %bz
%ny = OpFSub %float %ny_t1 %ny_t2
%nz_t1 = OpFMul %float %ax %by
%nz_t2 = OpFMul %float %ay %bx
%nz = OpFSub %float %nz_t1 %nz_t2

%n_raw = OpCompositeConstruct %v3float %nx %ny %nz
%N = OpExtInst %v3float %glsl Normalize %n_raw
%L = OpExtInst %v3float %glsl Normalize %v3_111
%ndotl = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %ndotl %float_0

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %c4_ptr
%csum1 = OpFAdd %v3float %c0 %c2
%csum = OpFAdd %v3float %csum1 %c4
%inv3 = OpFDiv %float %float_1 %float_3
%avg = OpVectorTimesScalar %v3float %csum %inv3

%scale_t = OpFMul %float %float_0_8 %diffuse
%scale = OpFAdd %float %float_0_2 %scale_t
%finalColor = OpVectorTimesScalar %v3float %avg %scale

%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%out_pos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr0 %pos0
OpStore %outColor %finalColor
OpEmitVertex

%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%out_pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr2 %pos2
OpStore %outColor %finalColor
OpEmitVertex

%pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr
%out_pos_ptr4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr4 %pos4
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
