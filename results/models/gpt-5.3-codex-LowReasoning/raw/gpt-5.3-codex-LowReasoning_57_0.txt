const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid;
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const towns = Array.isArray(data.towns) ? data.towns : [];
const cutBudget = Math.max(0, data.cutBudget | 0);

const n = width * height;
const INF = 1e9;

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}
function idx(x, y) {
  return y * width + x;
}
function xy(i) {
  return [i % width, (i / width) | 0];
}

const burnable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) burnable[idx(x, y)] = 1; // '.'
  }
}

const isIgnition = new Uint8Array(n);
const ignitionList = [];
for (const p of ignitions) {
  if (!Array.isArray(p) || p.length < 2) continue;
  const x = p[0] | 0, y = p[1] | 0;
  if (!inBounds(x, y)) continue;
  const i = idx(x, y);
  if (!burnable[i] || isIgnition[i]) continue;
  isIgnition[i] = 1;
  ignitionList.push(i);
}

if (cutBudget === 0 || ignitionList.length === 0) {
  process.stdout.write('{"firebreaks":[]}');
  process.exit(0);
}

function bfsFromSources(sources) {
  const dist = new Int32Array(n);
  for (let i = 0; i < n; i++) dist[i] = INF;
  const q = new Int32Array(n);
  let qh = 0, qt = 0;

  for (const s of sources) {
    dist[s] = 0;
    q[qt++] = s;
  }

  while (qh < qt) {
    const v = q[qh++];
    const d = dist[v] + 1;
    const x = v % width;
    const y = (v / width) | 0;

    let u;
    if (x > 0) {
      u = v - 1;
      if (burnable[u] && dist[u] === INF) { dist[u] = d; q[qt++] = u; }
    }
    if (x + 1 < width) {
      u = v + 1;
      if (burnable[u] && dist[u] === INF) { dist[u] = d; q[qt++] = u; }
    }
    if (y > 0) {
      u = v - width;
      if (burnable[u] && dist[u] === INF) { dist[u] = d; q[qt++] = u; }
    }
    if (y + 1 < height) {
      u = v + width;
      if (burnable[u] && dist[u] === INF) { dist[u] = d; q[qt++] = u; }
    }
  }
  return dist;
}

const fireDist = bfsFromSources(ignitionList);

const burnedTowns = [];
for (const t of towns) {
  const x = t.x | 0, y = t.y | 0;
  if (!inBounds(x, y)) continue;
  const i = idx(x, y);
  if (!burnable[i]) continue;
  if (fireDist[i] !== INF) {
    burnedTowns.push({ x, y, i, value: +t.value || 0, fireTime: fireDist[i] });
  }
}

if (burnedTowns.length === 0) {
  process.stdout.write('{"firebreaks":[]}');
  process.exit(0);
}

burnedTowns.sort((a, b) => b.value - a.value);
const focusTowns = burnedTowns.slice(0, Math.min(64, burnedTowns.length));

const townSet = new Uint8Array(n);
for (const t of towns) {
  const x = t.x | 0, y = t.y | 0;
  if (inBounds(x, y)) townSet[idx(x, y)] = 1;
}

const candidates = [];
const baseScore = new Float64Array(n);

for (let v = 0; v < n; v++) {
  if (!burnable[v]) continue;
  if (isIgnition[v]) continue;
  if (townSet[v]) continue;
  const tf = fireDist[v];
  if (tf === INF) continue;

  const x = v % width;
  const y = (v / width) | 0;

  let blockedAdj = 0;
  if (x === 0 || !burnable[v - 1]) blockedAdj++;
  if (x + 1 === width || !burnable[v + 1]) blockedAdj++;
  if (y === 0 || !burnable[v - width]) blockedAdj++;
  if (y + 1 === height || !burnable[v + width]) blockedAdj++;

  let s = 0;
  for (let k = 0; k < focusTowns.length; k++) {
    const t = focusTowns[k];
    const md = Math.abs(x - t.x) + Math.abs(y - t.y);
    const slack = Math.abs((tf + md) - t.fireTime);
    const pathLikelihood = 1.0 / (1.0 + slack);
    const proximity = 1.0 / ((md + 1) * (md + 1));
    s += t.value * pathLikelihood * proximity;
  }

  s *= (1 + 0.35 * blockedAdj);
  s *= (1 + Math.min(tf, 30) * 0.02);

  if (s > 0) {
    baseScore[v] = s;
    candidates.push(v);
  }
}

if (candidates.length === 0) {
  process.stdout.write('{"firebreaks":[]}');
  process.exit(0);
}

candidates.sort((a, b) => baseScore[b] - baseScore[a]);
const poolSize = Math.min(candidates.length, Math.max(cutBudget * 25, 1000));
const pool = candidates.slice(0, poolSize);

const selected = [];
const chosen = new Uint8Array(n);

function selectedNeighbors(v) {
  let c = 0;
  const x = v % width;
  const y = (v / width) | 0;
  if (x > 0 && chosen[v - 1]) c++;
  if (x + 1 < width && chosen[v + 1]) c++;
  if (y > 0 && chosen[v - width]) c++;
  if (y + 1 < height && chosen[v + width]) c++;
  return c;
}

const toPick = Math.min(cutBudget, pool.length);
for (let step = 0; step < toPick; step++) {
  let bestV = -1;
  let best = -1;

  for (let j = 0; j < pool.length; j++) {
    const v = pool[j];
    if (chosen[v]) continue;
    const conn = selectedNeighbors(v);
    const score = baseScore[v] + conn * (baseScore[v] * 0.6 + 1.0);
    if (score > best) {
      best = score;
      bestV = v;
    }
  }

  if (bestV < 0) break;
  chosen[bestV] = 1;
  selected.push(bestV);
}

const firebreaks = selected.map(xy);
process.stdout.write(JSON.stringify({ firebreaks }));
