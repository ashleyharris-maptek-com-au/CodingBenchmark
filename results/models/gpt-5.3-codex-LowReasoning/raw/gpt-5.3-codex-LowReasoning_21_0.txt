#include <algorithm>
#include <array>
#include <cstdint>
#include <deque>
#include <functional>
#include <iostream>
#include <limits>
#include <queue>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

struct VecHash {
    std::size_t operator()(const std::vector<int>& v) const noexcept {
        // 64-bit FNV-1a style mix
        std::uint64_t h = 1469598103934665603ULL;
        for (int x : v) {
            std::uint64_t y = static_cast<std::uint64_t>(static_cast<std::int64_t>(x));
            h ^= y + 0x9e3779b97f4a7c15ULL + (h << 6U) + (h >> 2U);
            h *= 1099511628211ULL;
        }
        return static_cast<std::size_t>(h);
    }
};

struct Move {
    int axis = 0;
    int dir = 1;
};

static inline bool in_bounds(const std::vector<int>& p, const std::vector<int>& bounds) {
    for (std::size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int D = 0;
    int max_turns = 0;
    if (!(std::cin >> D >> max_turns)) return 0;

    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];

    int snake_length = 1;
    std::vector<int> start(D);
    std::cin >> snake_length;
    for (int i = 0; i < D; ++i) std::cin >> start[i];

    int F = 0;
    std::cin >> F;
    std::vector<std::vector<int>> foods;
    foods.reserve(std::max(0, F));
    for (int i = 0; i < F; ++i) {
        std::vector<int> p(D);
        for (int j = 0; j < D; ++j) std::cin >> p[j];
        foods.push_back(std::move(p));
    }

    int O = 0;
    std::cin >> O;
    std::unordered_set<std::vector<int>, VecHash> obstacles;
    obstacles.reserve(static_cast<std::size_t>(O) * 2 + 1U);
    for (int i = 0; i < O; ++i) {
        std::vector<int> p(D);
        for (int j = 0; j < D; ++j) std::cin >> p[j];
        obstacles.insert(std::move(p));
    }

    std::unordered_set<std::vector<int>, VecHash> food_set;
    food_set.reserve(foods.size() * 2 + 1U);
    for (const auto& f : foods) food_set.insert(f);

    std::deque<std::vector<int>> body;
    body.push_front(start);

    // If initial length > 1 but body coordinates are not provided, defer tail movement
    // so effective occupied length grows to the declared initial length.
    int deferred_tail_steps = std::max(0, snake_length - 1);

    auto make_next = [&](const std::vector<int>& p, int axis, int dir) {
        std::vector<int> q = p;
        q[axis] += dir;
        return q;
    };

    auto choose_target = [&](const std::vector<int>& head) -> std::vector<int> {
        int best_dist = std::numeric_limits<int>::max();
        std::vector<int> best;
        bool found = false;
        for (const auto& f : food_set) {
            long long md = 0;
            for (int i = 0; i < D; ++i) md += std::llabs(static_cast<long long>(head[i]) - f[i]);
            if (!found || md < best_dist) {
                found = true;
                best_dist = static_cast<int>(md);
                best = f;
            }
        }
        return best;
    };

    auto build_occupancy = [&]() {
        std::unordered_set<std::vector<int>, VecHash> occ;
        occ.reserve(body.size() * 2 + 1U);
        for (std::size_t i = 0; i < body.size(); ++i) {
            // tail may move away this turn unless we are in deferred tail phase
            if (i + 1 == body.size() && deferred_tail_steps == 0) continue;
            occ.insert(body[i]);
        }
        return occ;
    };

    auto bfs_first_move = [&](const std::vector<int>& src, const std::vector<int>& dst) -> std::pair<bool, Move> {
        if (src == dst) return {false, Move{0, 1}};

        auto occ = build_occupancy();

        struct Parent {
            std::vector<int> prev;
            Move mv;
        };

        std::queue<std::vector<int>> q;
        std::unordered_set<std::vector<int>, VecHash> vis;
        std::unordered_map<std::vector<int>, Parent, VecHash> parent;

        q.push(src);
        vis.insert(src);

        bool reached = false;
        while (!q.empty() && !reached) {
            std::vector<int> cur = q.front();
            q.pop();

            for (int ax = 0; ax < D; ++ax) {
                for (int di = -1; di <= 1; di += 2) {
                    std::vector<int> nxt = make_next(cur, ax, di);
                    if (!in_bounds(nxt, bounds)) continue;
                    if (obstacles.find(nxt) != obstacles.end()) continue;
                    if (occ.find(nxt) != occ.end()) continue;
                    if (vis.find(nxt) != vis.end()) continue;

                    vis.insert(nxt);
                    parent.emplace(nxt, Parent{cur, Move{ax, di}});
                    if (nxt == dst) {
                        reached = true;
                        break;
                    }
                    q.push(std::move(nxt));
                }
                if (reached) break;
            }
        }

        if (!reached) return {false, Move{0, 1}};

        std::vector<int> cur = dst;
        while (true) {
            auto it = parent.find(cur);
            if (it == parent.end()) break;
            if (it->second.prev == src) return {true, it->second.mv};
            cur = it->second.prev;
        }
        return {false, Move{0, 1}};
    };

    auto fallback_safe_move = [&](const std::vector<int>& head) -> Move {
        auto occ = build_occupancy();

        int best_score = -1;
        Move best{0, 1};
        bool found = false;

        for (int ax = 0; ax < D; ++ax) {
            for (int di = -1; di <= 1; di += 2) {
                std::vector<int> nxt = make_next(head, ax, di);
                if (!in_bounds(nxt, bounds)) continue;
                if (obstacles.find(nxt) != obstacles.end()) continue;
                if (occ.find(nxt) != occ.end()) continue;

                int deg = 0;
                for (int ax2 = 0; ax2 < D; ++ax2) {
                    for (int di2 = -1; di2 <= 1; di2 += 2) {
                        std::vector<int> n2 = make_next(nxt, ax2, di2);
                        if (!in_bounds(n2, bounds)) continue;
                        if (obstacles.find(n2) != obstacles.end()) continue;
                        if (occ.find(n2) != occ.end() && n2 != body.back()) continue;
                        ++deg;
                    }
                }

                if (!found || deg > best_score) {
                    found = true;
                    best_score = deg;
                    best = Move{ax, di};
                }
            }
        }

        if (found) return best;

        // No safe move found; output a deterministic move.
        return Move{0, 1};
    };

    for (int turn = 0; turn < max_turns; ++turn) {
        if (food_set.empty()) break;

        const std::vector<int>& head = body.front();
        std::vector<int> target = choose_target(head);

        Move mv;
        auto [ok, m] = bfs_first_move(head, target);
        if (ok) mv = m;
        else mv = fallback_safe_move(head);

        std::cout << mv.axis << ' ' << mv.dir << '\n';

        std::vector<int> new_head = make_next(head, mv.axis, mv.dir);
        if (!in_bounds(new_head, bounds)) break;
        if (obstacles.find(new_head) != obstacles.end()) break;

        // Self-collision check for simulation consistency.
        bool grow = (food_set.find(new_head) != food_set.end());
        std::size_t check_upto = body.size();
        if (!grow && deferred_tail_steps == 0 && check_upto > 0) --check_upto;
        for (std::size_t i = 0; i < check_upto; ++i) {
            if (body[i] == new_head) {
                turn = max_turns; // terminate outer loop
                break;
            }
        }
        if (turn >= max_turns) break;

        body.push_front(new_head);

        if (grow) {
            food_set.erase(new_head);
        } else if (deferred_tail_steps > 0) {
            --deferred_tail_steps;
        } else {
            body.pop_back();
        }
    }

    return 0;
}
