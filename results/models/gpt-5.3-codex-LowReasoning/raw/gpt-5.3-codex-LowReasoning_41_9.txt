; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor %ubo
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450
               OpName %main "main"
               OpName %worldPos "worldPos"
               OpName %normal "normal"
               OpName %uv "uv"
               OpName %outColor "outColor"
               OpName %UBO "UBO"
               OpName %ubo "ubo"
               OpMemberName %UBO 0 "model"
               OpMemberName %UBO 1 "view"
               OpMemberName %UBO 2 "projection"
               OpMemberName %UBO 3 "lightPos"
               OpMemberName %UBO 4 "cameraPos"
               OpMemberName %UBO 5 "params"
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %outColor Location 0
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224

      %void = OpTypeVoid
         %3 = OpTypeFunction %void
     %float = OpTypeFloat 32
       %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0
      %bool = OpTypeBool
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4float 4
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
       %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

  %worldPos = OpVariable %_ptr_Input_v3float Input
    %normal = OpVariable %_ptr_Input_v3float Input
        %uv = OpVariable %_ptr_Input_v2float Input
  %outColor = OpVariable %_ptr_Output_v4float Output
       %ubo = OpVariable %_ptr_Uniform_UBO Uniform

    %int_3 = OpConstant %int 3
    %int_1 = OpConstant %int 1
  %float_8 = OpConstant %float 8
%float_0_9 = OpConstant %float 0.899999976
%float_0_2 = OpConstant %float 0.200000003
%float_0_5 = OpConstant %float 0.5
%float_0_0 = OpConstant %float 0
    %int_0 = OpConstant %int 0
%float_1_0 = OpConstant %float 1
      %v3_1 = OpConstantComposite %v3float %float_1_0 %float_1_0 %float_1_0

      %main = OpFunction %void None %3
         %5 = OpLabel
        %n0 = OpLoad %v3float %normal
        %p0 = OpLoad %v3float %worldPos
       %uv0 = OpLoad %v2float %uv

     %nrmN = OpExtInst %v3float %1 Normalize %n0
        %19 = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
   %light4 = OpLoad %v4float %19
   %light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
     %Lraw = OpFSub %v3float %light3 %p0
     %nrmL = OpExtInst %v3float %1 Normalize %Lraw
      %dot = OpDot %float %nrmN %nrmL
     %diff = OpExtInst %float %1 FMax %dot %float_0_0
      %halfDiff = OpFMul %float %float_0_5 %diff
   %lambert2 = OpFAdd %float %float_0_5 %halfDiff

       %uv8 = OpFMul %v2float %uv0 %float_8
   %uvFloor = OpExtInst %v2float %1 Floor %uv8
        %ux = OpCompositeExtract %float %uvFloor 0
        %uy = OpCompositeExtract %float %uvFloor 1
        %ix = OpConvertFToS %int %ux
        %iy = OpConvertFToS %int %uy
      %sum = OpIAdd %int %ix %iy
   %parity = OpBitwiseAnd %int %sum %int_1
   %isEven = OpIEqual %bool %parity %int_0

 %checkerF = OpSelect %float %isEven %float_0_9 %float_0_2
   %shadeF = OpFMul %float %checkerF %lambert2
      %rgb = OpVectorTimesScalar %v3float %v3_1 %shadeF
            %rx = OpCompositeExtract %float %rgb 0
      %ry = OpCompositeExtract %float %rgb 1
      %rz = OpCompositeExtract %float %rgb 2
      %out = OpCompositeConstruct %v4float %rx %ry %rz %float_1_0
               OpStore %outColor %out
               OpReturn
               OpFunctionEnd
