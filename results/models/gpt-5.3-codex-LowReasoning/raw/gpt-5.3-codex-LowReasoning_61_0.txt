const fs = require('fs');

interface PackagePoint {
  id: string | number;
  x: number;
  y: number;
}

interface InputData {
  depot: [number, number];
  truckCount: number;
  packages: PackagePoint[];
}

interface Point {
  id: string | number;
  x: number;
  y: number;
  angle: number;
  radius: number;
}

function dist(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx;
  const dy = ay - by;
  return Math.hypot(dx, dy);
}

function routeDistance(route: Point[], dx: number, dy: number): number {
  if (route.length === 0) return 0;
  let total = dist(dx, dy, route[0].x, route[0].y);
  for (let i = 1; i < route.length; i++) {
    total += dist(route[i - 1].x, route[i - 1].y, route[i].x, route[i].y);
  }
  total += dist(route[route.length - 1].x, route[route.length - 1].y, dx, dy);
  return total;
}

function twoOpt(route: Point[], dx: number, dy: number): Point[] {
  const n = route.length;
  if (n < 4) return route;

  let improved = true;
  let pass = 0;
  const maxPasses = 8;

  while (improved && pass < maxPasses) {
    improved = false;
    pass++;

    for (let i = 0; i < n - 2; i++) {
      const aX = i === 0 ? dx : route[i - 1].x;
      const aY = i === 0 ? dy : route[i - 1].y;
      const bX = route[i].x;
      const bY = route[i].y;

      for (let k = i + 1; k < n - 1; k++) {
        const cX = route[k].x;
        const cY = route[k].y;
        const dX = route[k + 1].x;
        const dY = route[k + 1].y;

        const oldLen = dist(aX, aY, bX, bY) + dist(cX, cY, dX, dY);
        const newLen = dist(aX, aY, cX, cY) + dist(bX, bY, dX, dY);

        if (newLen + 1e-9 < oldLen) {
          let l = i;
          let r = k;
          while (l < r) {
            const tmp = route[l];
            route[l] = route[r];
            route[r] = tmp;
            l++;
            r--;
          }
          improved = true;
        }
      }

      const cX = route[n - 1].x;
      const cY = route[n - 1].y;
      const dX = dx;
      const dY = dy;
      const oldLenEnd = dist(aX, aY, bX, bY) + dist(cX, cY, dX, dY);
      const newLenEnd = dist(aX, aY, cX, cY) + dist(bX, bY, dX, dY);
      if (newLenEnd + 1e-9 < oldLenEnd) {
        let l = i;
        let r = n - 1;
        while (l < r) {
          const tmp = route[l];
          route[l] = route[r];
          route[r] = tmp;
          l++;
          r--;
        }
        improved = true;
      }
    }
  }

  return route;
}

function splitBySweep(points: Point[], truckCount: number): Point[][] {
  const n = points.length;
  if (n === 0) return [];

  const k = Math.max(1, Math.min(truckCount, n));
  const sorted = points.slice().sort((a, b) => {
    if (a.angle !== b.angle) return a.angle - b.angle;
    return a.radius - b.radius;
  });

  const routes: Point[][] = Array.from({ length: k }, () => []);

  const base = Math.floor(n / k);
  let extra = n % k;
  let idx = 0;

  for (let r = 0; r < k; r++) {
    const size = base + (extra > 0 ? 1 : 0);
    if (extra > 0) extra--;
    for (let j = 0; j < size; j++) {
      routes[r].push(sorted[idx++]);
    }
  }

  return routes;
}

function buildNearestNeighborRoute(points: Point[], dx: number, dy: number): Point[] {
  const n = points.length;
  if (n <= 1) return points.slice();

  const unused = points.slice();
  let bestStart = 0;
  let far = -1;
  for (let i = 0; i < unused.length; i++) {
    const d = dist(dx, dy, unused[i].x, unused[i].y);
    if (d > far) {
      far = d;
      bestStart = i;
    }
  }

  const route: Point[] = [];
  route.push(unused[bestStart]);
  unused.splice(bestStart, 1);

  while (unused.length > 0) {
    const last = route[route.length - 1];
    let bi = 0;
    let bd = Infinity;
    for (let i = 0; i < unused.length; i++) {
      const d = dist(last.x, last.y, unused[i].x, unused[i].y);
      if (d < bd) {
        bd = d;
        bi = i;
      }
    }
    route.push(unused[bi]);
    unused.splice(bi, 1);
  }

  return route;
}

function main(): void {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: InputData = JSON.parse(input);

  const depotX = data.depot[0];
  const depotY = data.depot[1];
  const packages = data.packages || [];
  const n = packages.length;

  if (n === 0) {
    process.stdout.write(JSON.stringify({ routes: [] }));
    return;
  }

  const points: Point[] = new Array(n);
  for (let i = 0; i < n; i++) {
    const p = packages[i];
    const rx = p.x - depotX;
    const ry = p.y - depotY;
    points[i] = {
      id: p.id,
      x: p.x,
      y: p.y,
      angle: Math.atan2(ry, rx),
      radius: rx * rx + ry * ry,
    };
  }

  const initialRoutes = splitBySweep(points, data.truckCount);

  const optimized: Point[][] = [];
  for (let i = 0; i < initialRoutes.length; i++) {
    const cluster = initialRoutes[i];
    if (cluster.length === 0) continue;
    let route = buildNearestNeighborRoute(cluster, depotX, depotY);
    route = twoOpt(route, depotX, depotY);
    optimized.push(route);
  }

  optimized.sort((a, b) => routeDistance(b, depotX, depotY) - routeDistance(a, depotX, depotY));

  const out = {
    routes: optimized.map((r) => r.map((p) => p.id)),
  };

  process.stdout.write(JSON.stringify(out));
}

main();
