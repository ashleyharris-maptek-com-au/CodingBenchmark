use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    let half = n / 2;

    // side[v] = -1 (unassigned), 0 (A), 1 (B)
    let mut side = vec![-1_i8; n];
    let mut best_cut = i32::MAX;
    let mut best_a: Vec<usize> = Vec::new();

    // Symmetry break: force vertex 0 into A.
    side[0] = 0;

    fn dfs(
        idx: usize,
        n: usize,
        half: usize,
        adj: &Vec<Vec<usize>>,
        side: &mut Vec<i8>,
        count_a: usize,
        count_b: usize,
        cur_cut: i32,
        best_cut: &mut i32,
        best_a: &mut Vec<usize>,
    ) {
        if cur_cut >= *best_cut {
            return;
        }

        if idx == n {
            if count_a == half && count_b == half {
                *best_cut = cur_cut;
                best_a.clear();
                for v in 0..n {
                    if side[v] == 0 {
                        best_a.push(v);
                    }
                }
            }
            return;
        }

        // Remaining vertices including idx.
        let rem = n - idx;

        // Feasibility pruning for exact partition sizes.
        if count_a > half || count_b > half {
            return;
        }
        if count_a + rem < half || count_b + rem < half {
            return;
        }

        // Try put idx in A.
        if count_a < half {
            let mut add = 0_i32;
            for &u in &adj[idx] {
                if u < idx && side[u] == 1 {
                    add += 1;
                }
            }
            side[idx] = 0;
            dfs(
                idx + 1,
                n,
                half,
                adj,
                side,
                count_a + 1,
                count_b,
                cur_cut + add,
                best_cut,
                best_a,
            );
            side[idx] = -1;
        }

        // Try put idx in B.
        if count_b < half {
            let mut add = 0_i32;
            for &u in &adj[idx] {
                if u < idx && side[u] == 0 {
                    add += 1;
                }
            }
            side[idx] = 1;
            dfs(
                idx + 1,
                n,
                half,
                adj,
                side,
                count_a,
                count_b + 1,
                cur_cut + add,
                best_cut,
                best_a,
            );
            side[idx] = -1;
        }
    }

    dfs(
        1,
        n,
        half,
        &adj,
        &mut side,
        1,
        0,
        0,
        &mut best_cut,
        &mut best_a,
    );

    // Safety check (should always pass if input has even n).
    if best_cut == i32::MAX || best_a.len() != half {
        println!("0");
        println!();
        return;
    }

    println!("{}", best_cut);
    for i in 0..best_a.len() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", best_a[i]);
    }
    println!();
}
