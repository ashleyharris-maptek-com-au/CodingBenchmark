using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private const double DaySec = 86400.0;

    private struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);

        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);

        public Vec3 WithMagnitude(double mag)
        {
            var n = Norm();
            if (n < 1e-12) return new Vec3(0, 0, 0);
            return this * (mag / n);
        }
    }

    private sealed class Body
    {
        public string Name = "";
        public double Mass;
        public Vec3 R0;
        public Vec3 V0;

        public Vec3 PosAtDays(double tDays) => R0 + V0 * (tDays * DaySec);
        public Vec3 VelAtDays(double tDays) => V0;
    }

    private struct Burn
    {
        public double TDays;
        public Vec3 Dv;
    }

    public static void Main()
    {
        var tokens = ReadAllTokens();
        int idx = 0;

        int numBodies = ParseInt(tokens, ref idx);
        double warningDays = ParseDouble(tokens, ref idx);
        _ = ParseDouble(tokens, ref idx); // impactor_mass (not required by this guidance solver)
        double dvBudget = ParseDouble(tokens, ref idx);
        _ = ParseDouble(tokens, ref idx); // asteroid_mass
        _ = ParseDouble(tokens, ref idx); // t0_unix_utc

        var bodies = new List<Body>(numBodies);
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = tokens[idx++];
            b.Mass = ParseDouble(tokens, ref idx);
            b.R0 = new Vec3(ParseDouble(tokens, ref idx), ParseDouble(tokens, ref idx), ParseDouble(tokens, ref idx));
            b.V0 = new Vec3(ParseDouble(tokens, ref idx), ParseDouble(tokens, ref idx), ParseDouble(tokens, ref idx));
            bodies.Add(b);
        }

        // Consume ephemeris path token if present.
        if (idx < tokens.Count) idx++;

        var earth = bodies.FirstOrDefault(b => string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase));
        var asteroid = bodies.FirstOrDefault(b => string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase));

        if (earth == null || asteroid == null)
        {
            // Fallback valid output.
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        // Estimate nominal Earth-asteroid closest approach time in [0, warningDays].
        double tImpactEst = EstimateClosestApproachTimeDays(earth.R0, earth.V0, asteroid.R0, asteroid.V0, 0.0, warningDays);

        // Choose intercept lead based on available warning.
        double leadDays = Math.Max(3.0, Math.Min(120.0, 0.2 * warningDays));
        double tIntercept = Clamp(tImpactEst - leadDays, 0.2, warningDays);

        // Choose launch time before intercept; keep within window.
        double nominalFlightDays = Math.Max(1.0, Math.Min(120.0, 0.35 * warningDays));
        double tLaunch = Clamp(tIntercept - nominalFlightDays, 0.0, warningDays);
        if (tLaunch >= tIntercept)
        {
            tLaunch = Math.Max(0.0, tIntercept - 0.5);
        }

        double used = 0.0;
        var burns = new List<Burn>();

        // Burn 1: direct intercept targeting under linear propagation.
        var rL = earth.PosAtDays(tLaunch);
        var vL = earth.VelAtDays(tLaunch);
        var rI = asteroid.PosAtDays(tIntercept);
        double dt1 = Math.Max(1e-6, (tIntercept - tLaunch) * DaySec);

        Vec3 vReq1 = (rI - rL) / dt1;
        Vec3 dv1 = vReq1 - vL;

        double dv1Cap = dvBudget * 0.8;
        if (dv1.Norm() > dv1Cap) dv1 = dv1.WithMagnitude(dv1Cap);
        used += dv1.Norm();
        burns.Add(new Burn { TDays = tLaunch, Dv = dv1 });

        // Burn 2: mid-course correction if budget remains.
        double remaining = Math.Max(0.0, dvBudget - used);
        double tMid = tLaunch + 0.5 * (tIntercept - tLaunch);
        Vec3 rMidSc = rL + (vL + dv1) * ((tMid - tLaunch) * DaySec);
        Vec3 vMidSc = vL + dv1;
        Vec3 rTarget = asteroid.PosAtDays(tIntercept);
        double dt2 = Math.Max(1e-6, (tIntercept - tMid) * DaySec);

        Vec3 vReq2 = (rTarget - rMidSc) / dt2;
        Vec3 dv2 = vReq2 - vMidSc;

        if (remaining > 0.05)
        {
            if (dv2.Norm() > remaining) dv2 = dv2.WithMagnitude(remaining);
            used += dv2.Norm();
            burns.Add(new Burn { TDays = tMid, Dv = dv2 });
        }

        // Final burn at impact epoch (set to zero if budget is exhausted / not needed).
        burns.Add(new Burn { TDays = tIntercept, Dv = new Vec3(0, 0, 0) });

        Console.WriteLine(burns.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var b in burns.OrderBy(x => x.TDays))
        {
            Console.WriteLine(string.Format(
                CultureInfo.InvariantCulture,
                "{0:R} {1:R} {2:R} {3:R}",
                b.TDays, b.Dv.X, b.Dv.Y, b.Dv.Z));
        }
    }

    private static double EstimateClosestApproachTimeDays(Vec3 rE0, Vec3 vE, Vec3 rA0, Vec3 vA, double tMinDays, double tMaxDays)
    {
        Vec3 dr0 = rA0 - rE0;
        Vec3 dv = vA - vE;
        double dv2 = dv.Dot(dv);

        if (dv2 < 1e-18) return tMinDays;

        double tStarSec = -dr0.Dot(dv) / dv2;
        double tStarDays = tStarSec / DaySec;
        return Clamp(tStarDays, tMinDays, tMaxDays);
    }

    private static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }

    private static List<string> ReadAllTokens()
    {
        string text = Console.In.ReadToEnd();
        var parts = text.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        return new List<string>(parts);
    }

    private static int ParseInt(List<string> t, ref int i)
        => int.Parse(t[i++], NumberStyles.Integer, CultureInfo.InvariantCulture);

    private static double ParseDouble(List<string> t, ref int i)
        => double.Parse(t[i++], NumberStyles.Float | NumberStyles.AllowThousands, CultureInfo.InvariantCulture);
}
