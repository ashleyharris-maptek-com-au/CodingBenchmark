OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %in_color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.0399999991

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_glpvin_1 = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_glpvin_1 = OpTypePointer Input %arr_glpvin_1
%gl_in = OpVariable %ptr_Input_arr_glpvin_1 Input
%ptr_Input_v4float = OpTypePointer Input %v4float

%arr_v3float_1 = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3float_1 = OpTypePointer Input %arr_v3float_1
%in_color = OpVariable %ptr_Input_arr_v3float_1 Input
%ptr_Input_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_glpvout = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_glpvout Output
%ptr_Output_v4float = OpTypePointer Output %v4float

%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %func
%entry = OpLabel

%inpos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%inpos = OpLoad %v4float %inpos_ptr
%base_x = OpCompositeExtract %float %inpos 0
%base_y = OpCompositeExtract %float %inpos 1
%color_ptr = OpAccessChain %ptr_Input_v3float %in_color %int_0
%c = OpLoad %v3float %color_ptr

%x0 = OpFSub %float %base_x %float_h
%y0 = OpFSub %float %base_y %float_h
%p0 = OpCompositeConstruct %v4float %x0 %y0 %float_0 %float_1
%outpos_ptr0 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %outpos_ptr0 %p0
OpStore %outColor %c
OpEmitVertex

%x1 = OpFAdd %float %base_x %float_h
%y1 = OpFSub %float %base_y %float_h
%p1 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
%outpos_ptr1 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %outpos_ptr1 %p1
OpStore %outColor %c
OpEmitVertex

%x2 = OpFSub %float %base_x %float_h
%y2 = OpFAdd %float %base_y %float_h
%p2 = OpCompositeConstruct %v4float %x2 %y2 %float_0 %float_1
%outpos_ptr2 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %outpos_ptr2 %p2
OpStore %outColor %c
OpEmitVertex

%x3 = OpFAdd %float %base_x %float_h
%y3 = OpFAdd %float %base_y %float_h
%p3 = OpCompositeConstruct %v4float %x3 %y3 %float_0 %float_1
%outpos_ptr3 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %outpos_ptr3 %p3
OpStore %outColor %c
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
