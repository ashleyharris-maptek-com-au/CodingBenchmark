; SPIR-V
; Version: 1.0
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
               OpExecutionMode %main OriginUpperLeft

               OpName %main "main"
               OpName %UBO "UBO"
               OpName %ubo "ubo"
               OpName %inWorldPos "worldPos"
               OpName %inNormal "normal"
               OpName %outColor "outColor"

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
  %float_0_1 = OpConstant %float 0.1
  %float_0_6 = OpConstant %float 0.6
  %float_0_8 = OpConstant %float 0.8
   %float_64 = OpConstant %float 64
     %v3float = OpTypeVector %float 3
     %v4float = OpTypeVector %float 4
    %mat4v4f = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4v4f %mat4v4f %mat4v4f %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_ptr_Uniform_UBO Uniform
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
 %inWorldPos = OpVariable %_ptr_Input_v3float Input
   %inNormal = OpVariable %_ptr_Input_v3float Input
%_ptr_Output_v4float = OpTypePointer Output %v4float
   %outColor = OpVariable %_ptr_Output_v4float Output

       %main = OpFunction %void None %3
          %5 = OpLabel
         %N0 = OpLoad %v3float %inNormal
          %N = OpExtInst %v3float %1 Normalize %N0

    %lp_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
        %lp4 = OpLoad %v4float %lp_ptr
         %Lp = OpVectorShuffle %v3float %lp4 %lp4 0 1 2

    %wp_pos = OpLoad %v3float %inWorldPos
       %Lun = OpFSub %v3float %Lp %wp_pos
          %L = OpExtInst %v3float %1 Normalize %Lun

    %cp_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
        %cp4 = OpLoad %v4float %cp_ptr
         %Cp = OpVectorShuffle %v3float %cp4 %cp4 0 1 2
       %Vun = OpFSub %v3float %Cp %wp_pos
          %V = OpExtInst %v3float %1 Normalize %Vun

       %LaddV = OpFAdd %v3float %L %V
          %H = OpExtInst %v3float %1 Normalize %LaddV

      %dotNL = OpDot %float %N %L
   %diffuse = OpExtInst %float %1 FMax %dotNL %float_0

      %dotNH = OpDot %float %N %H
    %specIn = OpExtInst %float %1 FMax %dotNH %float_0
  %specular = OpExtInst %float %1 Pow %specIn %float_64

   %termDif = OpFMul %float %float_0_6 %diffuse
  %termSpec = OpFMul %float %float_0_8 %specular
      %sum1 = OpFAdd %float %float_0_1 %termDif
 %intensity = OpFAdd %float %sum1 %termSpec

   %rgb = OpCompositeConstruct %v3float %intensity %intensity %intensity
 %finalColor = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1
               OpStore %outColor %finalColor
               OpReturn
               OpFunctionEnd

%1 = OpExtInstImport "GLSL.std.450"
