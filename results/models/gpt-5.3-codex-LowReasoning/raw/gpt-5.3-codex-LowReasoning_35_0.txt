use std::collections::BinaryHeap;
use std::io::{self, Read};

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap_or(0),
        None => {
            // Empty input => empty dominating set
            println!("0\n");
            return;
        }
    };
    let m: usize = it.next().and_then(|x| x.parse().ok()).unwrap_or(0);

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let u = match it.next() {
            Some(x) => x.parse::<usize>().unwrap_or(0),
            None => break,
        };
        let v = match it.next() {
            Some(x) => x.parse::<usize>().unwrap_or(0),
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            // Self-loop doesn't change domination neighborhood in this model
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Sort+dedup adjacency lists
    for nbrs in &mut adj {
        nbrs.sort_unstable();
        nbrs.dedup();
    }

    if n == 0 {
        println!("0\n");
        return;
    }

    // closed[v] = {v} U N(v)
    let mut closed: Vec<Vec<usize>> = vec![Vec::new(); n];
    for v in 0..n {
        let mut c = Vec::with_capacity(adj[v].len() + 1);
        c.push(v);
        c.extend_from_slice(&adj[v]);
        c.sort_unstable();
        c.dedup();
        closed[v] = c;
    }

    // dom_count[u] = number of selected vertices whose closed neighborhood contains u
    let mut dom_count = vec![0u32; n];
    let mut selected = vec![false; n];
    let mut solution: Vec<usize> = Vec::new();

    // Helper to add a vertex to solution
    let mut add_vertex = |v: usize,
                          selected: &mut [bool],
                          dom_count: &mut [u32],
                          solution: &mut Vec<usize>,
                          closed: &Vec<Vec<usize>>| {
        if selected[v] {
            return;
        }
        selected[v] = true;
        solution.push(v);
        for &u in &closed[v] {
            dom_count[u] += 1;
        }
    };

    // 1) Mandatory picks for isolated vertices
    for v in 0..n {
        if adj[v].is_empty() && !selected[v] {
            add_vertex(v, &mut selected, &mut dom_count, &mut solution, &closed);
        }
    }

    // 2) Greedy construction with lazy heap updates
    // score(v) = number of currently undominated vertices in closed[v]
    let mut heap: BinaryHeap<(usize, usize)> = BinaryHeap::new();
    for v in 0..n {
        if !selected[v] {
            heap.push((closed[v].len(), v));
        }
    }

    let mut undominated = dom_count.iter().filter(|&&x| x == 0).count();

    while undominated > 0 {
        let mut chosen: Option<usize> = None;

        while let Some((key, v)) = heap.pop() {
            if selected[v] {
                continue;
            }
            let mut real = 0usize;
            for &u in &closed[v] {
                if dom_count[u] == 0 {
                    real += 1;
                }
            }
            if real == 0 {
                continue;
            }
            if real == key {
                chosen = Some(v);
                break;
            }
            heap.push((real, v));
        }

        if let Some(v) = chosen {
            add_vertex(v, &mut selected, &mut dom_count, &mut solution, &closed);
        } else {
            // Fallback: pick one undominated vertex and add its best local candidate
            let mut u0 = None;
            for u in 0..n {
                if dom_count[u] == 0 {
                    u0 = Some(u);
                    break;
                }
            }
            let u = u0.expect("undominated should exist");
            let mut best_v = u;
            let mut best_gain = 0usize;

            for &cand in &closed[u] {
                if selected[cand] {
                    continue;
                }
                let mut gain = 0usize;
                for &x in &closed[cand] {
                    if dom_count[x] == 0 {
                        gain += 1;
                    }
                }
                if gain > best_gain {
                    best_gain = gain;
                    best_v = cand;
                }
            }
            add_vertex(best_v, &mut selected, &mut dom_count, &mut solution, &closed);
        }

        undominated = dom_count.iter().filter(|&&x| x == 0).count();
    }

    // 3) Redundancy elimination pass
    // Remove any selected vertex whose deletion keeps all vertices dominated.
    let mut idx = 0usize;
    while idx < solution.len() {
        let v = solution[idx];
        let mut removable = true;
        for &u in &closed[v] {
            if dom_count[u] <= 1 {
                removable = false;
                break;
            }
        }

        if removable {
            for &u in &closed[v] {
                dom_count[u] -= 1;
            }
            selected[v] = false;
            solution.swap_remove(idx);
        } else {
            idx += 1;
        }
    }

    // 4) Cheap 1->1 local improvement (attempt replacement that may allow later removals)
    // For each selected v, try replacing it with a non-selected w that covers every "critical" vertex of v.
    // Critical vertices of v are those with dom_count == 1 and dominated only by v.
    let mut improved = true;
    let mut passes = 0;
    while improved && passes < 2 {
        improved = false;
        passes += 1;

        let mut i = 0usize;
        while i < solution.len() {
            let v = solution[i];

            let mut critical: Vec<usize> = Vec::new();
            for &u in &closed[v] {
                if dom_count[u] == 1 {
                    critical.push(u);
                }
            }

            // If no critical vertices, v is removable (should have been removed), do it now.
            if critical.is_empty() {
                for &u in &closed[v] {
                    dom_count[u] -= 1;
                }
                selected[v] = false;
                solution.swap_remove(i);
                improved = true;
                continue;
            }

            // Candidate w must dominate all critical vertices.
            // Use union of closed neighborhoods of critical[0] as small-ish pool.
            let mut candidate_pool: Vec<usize> = Vec::new();
            {
                let c0 = critical[0];
                candidate_pool.push(c0);
                candidate_pool.extend_from_slice(&adj[c0]);
                candidate_pool.sort_unstable();
                candidate_pool.dedup();
            }

            let mut best_w: Option<usize> = None;
            let mut best_extra_cover = 0usize;

            'cand_loop: for &w in &candidate_pool {
                if w == v || selected[w] {
                    continue;
                }
                // Check all critical vertices are covered by w
                for &u in &critical {
                    if closed[w].binary_search(&u).is_err() {
                        continue 'cand_loop;
                    }
                }

                // Tie-break: maximize extra currently singly-dominated vertices covered by w,
                // which can help future removals.
                let mut extra = 0usize;
                for &u in &closed[w] {
                    if dom_count[u] == 1 {
                        extra += 1;
                    }
                }

                if extra > best_extra_cover {
                    best_extra_cover = extra;
                    best_w = Some(w);
                }
            }

            if let Some(w) = best_w {
                // Perform swap v -> w
                for &u in &closed[v] {
                    dom_count[u] -= 1;
                }
                selected[v] = false;

                selected[w] = true;
                for &u in &closed[w] {
                    dom_count[u] += 1;
                }

                solution[i] = w;
                improved = true;
            }

            i += 1;
        }

        // After replacements, run one more redundancy cleanup
        let mut j = 0usize;
        while j < solution.len() {
            let v = solution[j];
            let mut removable = true;
            for &u in &closed[v] {
                if dom_count[u] <= 1 {
                    removable = false;
                    break;
                }
            }
            if removable {
                for &u in &closed[v] {
                    dom_count[u] -= 1;
                }
                selected[v] = false;
                solution.swap_remove(j);
                improved = true;
            } else {
                j += 1;
            }
        }
    }

    // Safety check (can be disabled for performance; O(n))
    debug_assert!(dom_count.iter().all(|&x| x > 0));

    solution.sort_unstable();
    println!("{}", solution.len());
    if solution.is_empty() {
        println!();
    } else {
        for (i, v) in solution.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    }
}
