const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const grid = data.grid || [];
const robotsInput = data.robots || [];
const jobsInput = data.jobs || [];

const h = grid.length;
const w = h ? grid[0].length : 0;
const nCells = h * w;

function inBounds(x, y) {
  return x >= 0 && x < w && y >= 0 && y < h;
}

function cellId(x, y) {
  return y * w + x;
}

function parseNumericId(v) {
  if (typeof v === 'number' && Number.isFinite(v)) return v;
  if (typeof v === 'string' && /^-?\d+$/.test(v)) return Number(v);
  return null;
}

function idCompare(a, b) {
  const an = parseNumericId(a.id);
  const bn = parseNumericId(b.id);
  if (an !== null && bn !== null && an !== bn) return an - bn;
  const as = String(a.id);
  const bs = String(b.id);
  if (as < bs) return -1;
  if (as > bs) return 1;
  return 0;
}

// Validate / normalize robot list ordered by robot id (array index in output is robot id).
const robots = robotsInput.slice().sort(idCompare).map((r) => ({
  id: r.id,
  x: r.x | 0,
  y: r.y | 0,
  pos: cellId(r.x | 0, r.y | 0),
  total: 0,
}));

const assignments = Array.from({ length: robots.length }, () => []);

// Jobs sorted by priority (desc), then id (asc) for deterministic behavior.
const jobs = jobsInput.slice().sort((a, b) => {
  const pa = Number.isFinite(a.priority) ? a.priority : 0;
  const pb = Number.isFinite(b.priority) ? b.priority : 0;
  if (pa !== pb) return pb - pa;
  return idCompare(a, b);
}).map((j) => ({
  id: j.id,
  x: j.x | 0,
  y: j.y | 0,
  pos: cellId(j.x | 0, j.y | 0),
}));

// Pre-mark walkability.
const walkable = new Uint8Array(nCells);
for (let y = 0; y < h; y++) {
  const row = grid[y];
  for (let x = 0; x < w; x++) {
    if (row[x] === '.') walkable[cellId(x, y)] = 1;
  }
}

const dirs = [1, 0, -1, 0, 1];

// Cache BFS distance vectors from source cell id.
const distCache = new Map();

function bfsFrom(source) {
  if (distCache.has(source)) return distCache.get(source);

  const dist = new Int32Array(nCells);
  dist.fill(-1);

  if (source < 0 || source >= nCells || !walkable[source]) {
    distCache.set(source, dist);
    return dist;
  }

  const q = new Int32Array(nCells);
  let head = 0;
  let tail = 0;

  dist[source] = 0;
  q[tail++] = source;

  while (head < tail) {
    const cur = q[head++];
    const cx = cur % w;
    const cy = (cur / w) | 0;
    const nd = dist[cur] + 1;

    for (let k = 0; k < 4; k++) {
      const nx = cx + dirs[k];
      const ny = cy + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const nid = cellId(nx, ny);
      if (!walkable[nid] || dist[nid] !== -1) continue;
      dist[nid] = nd;
      q[tail++] = nid;
    }
  }

  distCache.set(source, dist);
  return dist;
}

function chooseRobotForJob(jobPos) {
  let bestRobot = -1;
  let bestDist = Infinity;
  let bestTotal = Infinity;

  for (let i = 0; i < robots.length; i++) {
    const r = robots[i];
    const d = bfsFrom(r.pos)[jobPos];
    const distVal = d >= 0 ? d : 1e15;

    if (
      distVal < bestDist ||
      (distVal === bestDist && r.total < bestTotal) ||
      (distVal === bestDist && r.total === bestTotal && i < bestRobot)
    ) {
      bestRobot = i;
      bestDist = distVal;
      bestTotal = r.total;
    }
  }

  return { robotIndex: bestRobot, distance: bestDist };
}

for (const job of jobs) {
  const pick = chooseRobotForJob(job.pos);
  const ri = pick.robotIndex;
  assignments[ri].push(job.id);

  if (pick.distance < 1e15) {
    robots[ri].total += pick.distance;
  }
  robots[ri].pos = job.pos;
}

process.stdout.write(JSON.stringify({ assignments }));
