OpCapability Geometry

OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"

OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"

%1 = OpExtInstImport "GLSL.std.450"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_0_02 = OpConstant %float 0.02
%float_0_33333334 = OpConstant %float 0.33333334
%float_0_6666667 = OpConstant %float 0.6666667
%float_n1 = OpConstant %float -1

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%uint_4 = OpConstant %uint 4

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_glpvin_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr_glpvin_4 = OpTypePointer Input %arr_glpvin_4
%gl_in = OpVariable %ptr_Input_arr_glpvin_4 Input

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3_4 = OpTypePointer Input %arr_v3_4
%inWorldPos = OpVariable %ptr_Input_arr_v3_4 Input
%inNormal = OpVariable %ptr_Input_arr_v3_4 Input
%inColor = OpVariable %ptr_Input_arr_v3_4 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_glpvout = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_glpvout Output

%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output

%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%label = OpLabel

%p0ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p1ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p2ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p3ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0ptr
%p1 = OpLoad %v4float %p1ptr
%p2 = OpLoad %v4float %p2ptr
%p3 = OpLoad %v4float %p3ptr

%c1ptr = OpAccessChain %ptr_Input_v3 %inColor %int_1
%c2ptr = OpAccessChain %ptr_Input_v3 %inColor %int_2
%c1 = OpLoad %v3float %c1ptr
%c2 = OpLoad %v3float %c2ptr

%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3xy = OpVectorShuffle %v2float %p3 %p3 0 1

%d10 = OpFSub %v2float %p1xy %p0xy
%d21 = OpFSub %v2float %p2xy %p1xy
%d32 = OpFSub %v2float %p3xy %p2xy

%t1s = OpFAdd %v2float %d10 %d21
%t2s = OpFAdd %v2float %d21 %d32

%t1x = OpCompositeExtract %float %t1s 0
%t1y = OpCompositeExtract %float %t1s 1
%t2x = OpCompositeExtract %float %t2s 0
%t2y = OpCompositeExtract %float %t2s 1

%t1len2x = OpFMul %float %t1x %t1x
%t1len2y = OpFMul %float %t1y %t1y
%t1len2 = OpFAdd %float %t1len2x %t1len2y
%t1inv = OpExtInst %float %1 InverseSqrt %t1len2
%t1 = OpVectorTimesScalar %v2float %t1s %t1inv

%t2len2x = OpFMul %float %t2x %t2x
%t2len2y = OpFMul %float %t2y %t2y
%t2len2 = OpFAdd %float %t2len2x %t2len2y
%t2inv = OpExtInst %float %1 InverseSqrt %t2len2
%t2 = OpVectorTimesScalar %v2float %t2s %t2inv

%t1nx = OpCompositeExtract %float %t1 1
%t1ny0 = OpCompositeExtract %float %t1 0
%t1ny = OpFNegate %float %t1ny0
%n1 = OpCompositeConstruct %v2float %t1nx %t1ny

%t2nx = OpCompositeExtract %float %t2 1
%t2ny0 = OpCompositeExtract %float %t2 0
%t2ny = OpFNegate %float %t2ny0
%n2 = OpCompositeConstruct %v2float %t2nx %t2ny

%hn1 = OpVectorTimesScalar %v2float %n1 %float_0_02
%hn2 = OpVectorTimesScalar %v2float %n2 %float_0_02

%seg = OpFSub %v4float %p2 %p1
%one3 = OpVectorTimesScalar %v4float %seg %float_0_33333334
%two3 = OpVectorTimesScalar %v4float %seg %float_0_6666667
%a0 = %p1
%a1 = OpFAdd %v4float %p1 %one3
%b0 = OpFAdd %v4float %p1 %two3
%b1 = %p2

%ca = OpFSub %v3float %c2 %c1
%ca1 = OpVectorTimesScalar %v3float %ca %float_0_33333334
%ca2 = OpVectorTimesScalar %v3float %ca %float_0_6666667
%col_a0 = %c1
%col_a1 = OpFAdd %v3float %c1 %ca1
%col_b0 = OpFAdd %v3float %c1 %ca2
%col_b1 = %c2

%a0xy = OpVectorShuffle %v2float %a0 %a0 0 1
%a1xy = OpVectorShuffle %v2float %a1 %a1 0 1
%b0xy = OpVectorShuffle %v2float %b0 %b0 0 1
%b1xy = OpVectorShuffle %v2float %b1 %b1 0 1

%a0lxy = OpFSub %v2float %a0xy %hn1
%a0rxy = OpFAdd %v2float %a0xy %hn1
%a1lxy = OpFSub %v2float %a1xy %hn2
%a1rxy = OpFAdd %v2float %a1xy %hn2

%a0z = OpCompositeExtract %float %a0 2
%a0w = OpCompositeExtract %float %a0 3
%a1z = OpCompositeExtract %float %a1 2
%a1w = OpCompositeExtract %float %a1 3

%a0lx = OpCompositeExtract %float %a0lxy 0
%a0ly = OpCompositeExtract %float %a0lxy 1
%a0rx = OpCompositeExtract %float %a0rxy 0
%a0ry = OpCompositeExtract %float %a0rxy 1
%a1lx = OpCompositeExtract %float %a1lxy 0
%a1ly = OpCompositeExtract %float %a1lxy 1
%a1rx = OpCompositeExtract %float %a1rxy 0
%a1ry = OpCompositeExtract %float %a1rxy 1

%a0l = OpCompositeConstruct %v4float %a0lx %a0ly %a0z %a0w
%a0r = OpCompositeConstruct %v4float %a0rx %a0ry %a0z %a0w
%a1l = OpCompositeConstruct %v4float %a1lx %a1ly %a1z %a1w
%a1r = OpCompositeConstruct %v4float %a1rx %a1ry %a1z %a1w

%outPosPtr = OpAccessChain %ptr_Output_v4 %gl_out %int_0

OpStore %outColor %col_a0
OpStore %outPosPtr %a0l
OpEmitVertex
OpStore %outColor %col_a0
OpStore %outPosPtr %a0r
OpEmitVertex
OpStore %outColor %col_a1
OpStore %outPosPtr %a1l
OpEmitVertex
OpStore %outColor %col_a1
OpStore %outPosPtr %a1r
OpEmitVertex
OpEndPrimitive

%b0lxy = OpFSub %v2float %b0xy %hn1
%b0rxy = OpFAdd %v2float %b0xy %hn1
%b1lxy = OpFSub %v2float %b1xy %hn2
%b1rxy = OpFAdd %v2float %b1xy %hn2

%b0z = OpCompositeExtract %float %b0 2
%b0w = OpCompositeExtract %float %b0 3
%b1z = OpCompositeExtract %float %b1 2
%b1w = OpCompositeExtract %float %b1 3

%b0lx = OpCompositeExtract %float %b0lxy 0
%b0ly = OpCompositeExtract %float %b0lxy 1
%b0rx = OpCompositeExtract %float %b0rxy 0
%b0ry = OpCompositeExtract %float %b0rxy 1
%b1lx = OpCompositeExtract %float %b1lxy 0
%b1ly = OpCompositeExtract %float %b1lxy 1
%b1rx = OpCompositeExtract %float %b1rxy 0
%b1ry = OpCompositeExtract %float %b1rxy 1

%b0l = OpCompositeConstruct %v4float %b0lx %b0ly %b0z %b0w
%b0r = OpCompositeConstruct %v4float %b0rx %b0ry %b0z %b0w
%b1l = OpCompositeConstruct %v4float %b1lx %b1ly %b1z %b1w
%b1r = OpCompositeConstruct %v4float %b1rx %b1ry %b1z %b1w

OpStore %outColor %col_b0
OpStore %outPosPtr %b0l
OpEmitVertex
OpStore %outColor %col_b0
OpStore %outPosPtr %b0r
OpEmitVertex
OpStore %outColor %col_b1
OpStore %outPosPtr %b1l
OpEmitVertex
OpStore %outColor %col_b1
OpStore %outPosPtr %b1r
OpEmitVertex
OpEndPrimitive
