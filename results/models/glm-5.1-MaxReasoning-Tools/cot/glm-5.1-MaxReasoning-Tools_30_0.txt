Let me implement WalkSAT in C:

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

static inline unsigned int xorshift32(unsigned int *state) {
    unsigned int x = *state;
    x ^= x << 13;
    x ^= x >> 17;
    x ^= x << 5;
    *state = x;
    return x;
}

int main() {
    // Read input
    int n, m;
    scanf("%d %d", &n, &m);
    
    int *clauses = malloc(3 * m * sizeof(int));
    for (int i = 0; i < 3 * m; i++) {
        scanf("%d", &clauses[i]);
    }
    
    // Build adjacency: for each variable, list of clause indices
    // First, count
    int *var_count = calloc(n + 1, sizeof(int));
    for (int i = 0; i < 3 * m; i++) {
        var_count[abs(clauses[i])]++;
    }
    
    // Build offsets and lists
    int *var_offset = malloc((n + 2) * sizeof(int));
    var_offset[0] = 0;
    for (int i = 1; i <= n + 1; i++) {
        var_offset[i] = var_offset[i - 1] + var_count[i - 1];
    }
    
    int *var_clauses = malloc(3 * m * sizeof(int));  // clause index for each appearance
    int *var_lit_idx = malloc(3 * m * sizeof(int));  // which literal position (0,1,2)
    int *var_pos = calloc(n + 1, sizeof(int));  // position counter
    
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < 3; j++) {
            int v = abs(clauses[3 * i + j]);
            int pos = var_offset[v] + var_pos[v];
            var_clauses[pos] = i;
            var_lit_idx[pos] = j;
            var_pos[v]++;
        }
    }
    
    free(var_pos);
    free(var_count);
    
    // Random assignment
    char *assign = malloc(n + 1);
    unsigned int rng = 12345;
    for (int i = 1; i <= n; i++) {
        assign[i] = xorshift32(&rng) & 1;
    }
    
    // Compute clause satisfaction count
    int *clause_sat_count = calloc(m, sizeof(int));  // number of satisfied literals
    
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < 3; j++) {
            int lit = clauses[3 * i + j];
            int v = abs(lit);
            char val = assign[v];
            char sat = (lit > 0) ? val : (1 - val);
            clause_sat_count[i] += sat;
        }
    }
    
    // Build unsatisfied clause list
    int *unsat_clauses = malloc(m * sizeof(int));
    int num_unsat = 0;
    for (int i = 0; i < m; i++) {
        if (clause_sat_count[i] == 0) {
            unsat_clauses[num_unsat++] = i;
        }
    }
    
    // WalkSAT
    long long max_iter = 100000000LL;  // 100M iterations
    double p = 0.5;  // probability of random flip
    
    for (long long iter = 0; iter < max_iter && num_unsat > 0; iter++) {
        // Pick random unsatisfied clause
        int idx = xorshift32(&rng) % num_unsat;
        int ci = unsat_clauses[idx];
        
        // Choose variable to flip
        int flip_var;
        if ((xorshift32(&rng) & 1)) {
            // Random flip: pick random literal
            int lit_idx = xorshift32(&rng) % 3;
```
