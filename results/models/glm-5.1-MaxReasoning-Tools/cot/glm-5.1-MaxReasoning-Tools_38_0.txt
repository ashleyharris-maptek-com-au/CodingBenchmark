```python
    # Elements are 0..U-1
    # Sets are indexed 0..S-1

    element_to_sets = [[] for _ in range(U)]
    for i, s in enumerate(sets):
        for e in s:
            if 0 <= e < U:
                element_to_sets[e].append(i)
    
    # Algorithm X with backtracking
    # uncovered: set of elements still to cover
    # used: set of set indices already chosen
    # solution: list of chosen set indices
    
    uncovered = set(range(U))
    used = set()
    solution = []
    
    def search():
        if not uncovered:
            return True
        
        # Choose the element with the fewest covering sets (MRV heuristic)
        min_count = float('inf')
        chosen_element = -1
        for e in uncovered:
            count = sum(1 for s in element_to_sets[e] if s not in used)
            if count < min_count:
                min_count = count
                chosen_element = e

            return False

        for s_idx in element_to_sets[chosen_element]:
            if s_idx in used:
                continue
            
            s = sets[s_idx]
            # Check if this set can be added (no overlap with already covered elements)
            can_add = True
            for e in s:
                if e not in uncovered:
                    can_add = False
                    break

            used.remove(s_idx)
```

```python
class DLX:
    def __init__(self, num_cols):
        # Create header nodes
        self.header = {'L': None, 'R': None, 'U': None, 'D': None, 'C': None, 'S': 0, 'N': 'header'}
        self.cols = []
        prev = self.header
        for i in range(num_cols):
            col = {'L': prev, 'R': None, 'U': None, 'D': None, 'C': None, 'S': 0, 'N': i}
            col['U'] = col
            col['D'] = col
            col['C'] = col
            prev['R'] = col
            self.cols.append(col)
            prev = col
        prev['R'] = self.header
        self.header['L'] = prev
        self.header['R'] = self.cols[0] if self.cols else self.header
        
    def add_row(self, row_idx, cols):
        if not cols:
            return
        first = None
        prev = None
        for col_idx in cols:
            col = self.cols[col_idx]
            node = {'L': None, 'R': None, 'U': col['U'], 'D': col, 'C': col, 'row': row_idx}
            col['U']['D'] = node
            col['U'] = node
            col['S'] += 1
            if first is None:
                first = node
                prev = node
            else:
                node['L'] = prev
                prev['R'] = node
                prev = node
        first['L'] = prev
        prev['R'] = first
    
    def cover(self, col):
        col['R']['L'] = col['L']
        col['L']['R'] = col['R']
        i = col['D']
        while i is not col:
            j = i['R']
            while j is not i:
                j['D']['U'] = j['U']
                j['U']['D'] = j['D']
                j['C']['S'] -= 1
                j = j['R']
            i = i['D']
    
    def uncover(self, col):
        i = col['U']
        while i is not col:
            j = i['L']
            while j is not i:
                j['C']['S'] += 1
                j['D']['U'] = j
                j['U']['D'] = j
                j = j['L']
            i = i['U']
        col['R']['L'] = col
        col['L']['R'] = col
    
    def search(self, solution):
        if self.header['R'] is self.header:
            return solution[:]
        
        # Choose column with minimum size (MRV)
        min_size = float('inf')
        col = None
        c = self.header['R']
        while c is not self.header:
            if c['S'] < min_size:
                min_size = c['S']
                col = c
            c = c['R']
        
        if min_size == 0:
            return None
        
        self.cover(col)
        
        r = col['D']
        while r is not col:
            solution.append(r['row'])
            j = r['R']
            while j is not r:
                self.cover(j['C'])
                j = j['R']
            
            result = self.search(solution)
            if result is not None:
                return result
            
            solution.pop()
            j = r['L']
            while j is not r:
                self.uncover(j['C'])
                j = j['L']
            
            r = r['D']
        
        self.uncover(col)
        return None
```
