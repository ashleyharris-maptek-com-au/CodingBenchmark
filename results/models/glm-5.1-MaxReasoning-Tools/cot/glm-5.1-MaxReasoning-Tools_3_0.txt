```rust
fn layout(adj: &[Vec<usize>], n: usize) -> (Vec<f64>, Vec<f64>) {
    let area = n as f64;
    let k = area.sqrt(); // optimal distance
    
    // Initialize positions
    let mut xs = Vec::with_capacity(n);
    let mut ys = Vec::with_capacity(n);
    for i in 0..n {
        let angle = 2.0 * std::f64::consts::PI * i as f64 / n as f64;
        xs.push(k * 5.0 * angle.cos());
        ys.push(k * 5.0 * angle.sin());
    }
    
    // Force-directed iterations
    let iterations = if n <= 100 { 500 } else if n <= 1000 { 200 } else { 100 };
    let mut t = k * 10.0; // initial temperature
    
    for _ in 0..iterations {
        let mut dx = vec![0.0f64; n];
        let mut dy = vec![0.0f64; n];
        
        // Repulsive forces (all pairs)
        for i in 0..n {
            for j in (i+1)..n {
                let ddx = xs[i] - xs[j];
                let ddy = ys[i] - ys[j];
                let dist = (ddx * ddx + ddy * ddy).sqrt().max(0.01);
                let force = k * k / dist;
                let fx = ddx / dist * force;
                let fy = ddy / dist * force;
                dx[i] += fx;
                dy[i] += fy;
                dx[j] -= fx;
                dy[j] -= fy;
            }
        }
        
        // Attractive forces (edges)
        for u in 0..n {
            for &v in &adj[u] {
                if u < v {
                    let ddx = xs[u] - xs[v];
                    let ddy = ys[u] - ys[v];
                    let dist = (ddx * ddx + ddy * ddy).sqrt().max(0.01);
                    let force = dist * dist / k;
                    let fx = ddx / dist * force;
                    let fy = ddy / dist * force;
                    dx[u] -= fx;
                    dy[u] -= fy;
                    dx[v] += fx;
                    dy[v] += fy;
                }
            }
        }
        
        // Apply forces with temperature limiting
        for i in 0..n {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt().max(0.01);
            let scale = t / disp;
            xs[i] += dx[i] * scale.min(1.0);
            ys[i] += dy[i] * scale.min(1.0);
        }
        
        t *= 0.95; // cool down
    }
```

```rust
struct QuadTree {
    // Bounding box
    x_min: f64,
    y_min: f64,
    size: f64,
    // Center of mass
    mass: f64,
    cx: f64,
    cy: f64,
    // If this is a leaf with a single body
    body: Option<usize>,
    // Children
    children: [Option<Box<QuadTree>>; 4],
}

impl QuadTree {
    fn new(x_min: f64, y_min: f64, size: f64) -> Self {
        QuadTree {
            x_min, y_min, size,
            mass: 0.0, cx: 0.0, cy: 0.0,
            body: None,
            children: [None, None, None, None],
        }
    }
    
    fn insert(&mut self, idx: usize, x: f64, y: f64, mass: f64) {
        if self.mass == 0.0 {
            // Empty leaf
            self.body = Some(idx);
            self.mass = mass;
            self.cx = x;
            self.cy = y;
            return;
        }
        
        // If this is a leaf with one body, subdivide
        if self.body.is_some() {
            let existing_idx = self.body.unwrap();
            let existing_x = self.cx;
            let existing_y = self.cy;
            let existing_mass = self.mass;
            self.body = None;
            self.mass = 0.0;
            self.cx = 0.0;
            self.cy = 0.0;
            
            self.insert_into_child(existing_idx, existing_x, existing_y, existing_mass);
            self.insert_into_child(idx, x, y, mass);
        } else {
            // Internal node
            self.insert_into_child(idx, x, y, mass);
        }
        
        // Update center of mass
        let new_mass = self.mass + mass;
        self.cx = (self.cx * self.mass + x * mass) / new_mass;
        self.cy = (self.cy * self.mass + y * mass) / new_mass;
        self.mass = new_mass;
    }
    
    fn insert_into_child(&mut self, idx: usize, x: f64, y: f64, mass: f64) {
        let half = self.size / 2.0;
        let mid_x = self.x_min + half;
        let mid_y = self.y_min + half;
        
        let quadrant = if x < mid_x {
```

```rust
        while let Some(node) = stack.pop() {
            if visited[node] { continue; }
            visited[node] = true;
            component.push(node);
            for &neighbor in &adj[node] {
                if !visited[neighbor] {
                    stack.push(neighbor);
                }
            }
        }
        components.push(component);
    }
    
    components
}
```

```rust
fn layout_direct(adj: &[Vec<usize>], n: usize) -> (Vec<f64>, Vec<f64>) {
    let area = (n as f64) * 100.0;
    let k = (area / n as f64).sqrt();
    
    // Initialize positions - circular
    let mut xs: Vec<f64> = (0..n).map(|i| k * 5.0 * (2.0 * PI * i as f64 / n as f64).cos()).collect();
    let mut ys: Vec<f64> = (0..n).map(|i| k * 5.0 * (2.0 * PI * i as f64 / n as f64).sin()).collect();
    
    let iterations = if n <= 50 { 800 } else if n <= 200 { 500 } else { 300 };
    let mut temperature = k * 10.0;
    let cooling = 0.95;
    
    for _ in 0..iterations {
        let mut dx = vec![0.0f64; n];
        let mut dy = vec![0.0f64; n];
        
        // Repulsive forces
        for i in 0..n {
            for j in (i+1)..n {
                let ddx = xs[i] - xs[j];
                let ddy = ys[i] - ys[j];
                let dist_sq = ddx * ddx + ddy * ddy;
                let dist = dist_sq.sqrt().max(0.01);
                let force = k * k / dist;
                let fx = (ddx / dist) * force;
                let fy = (ddy / dist) * force;
                dx[i] += fx; dy[i] += fy;
                dx[j] -= fx; dy[j] -= fy;
            }
        }
        
        // Attractive forces
        for u in 0..n {
            for &v in &adj[u] {
                if u < v {
                    let ddx = xs[u] - xs[v];
                    let ddy = ys[u] - ys[v];
                    let dist = (ddx * ddx + ddy * ddy).sqrt().max(0.01);
                    let force = dist / k;
                    let fx = (ddx / dist) * force;
                    let fy = (ddy / dist) * force;
                    dx[u] -= fx; dy[u] -= fy;
                    dx[v] += fx; dy[v] += fy;
                }
            }
        }
        
        // Apply with temperature
        for i in 0..n {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt();
            if disp > 0.0 {
                let scale = temperature.min(disp) / disp;
                xs[i] += dx[i] * scale;
                ys[i] += dy[i] * scale;
            }
        }
        
        temperature *= cooling;
    }
    
    (xs, ys)
}
```
