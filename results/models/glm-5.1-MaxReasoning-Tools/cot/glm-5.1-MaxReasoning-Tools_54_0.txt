    # Localizer
    loc_vals = []
    for i in range(1, 4):
        if state.get(f'localizer_{i}_valid', False):
            loc_vals.append(state.get(f'localizer_{i}_dots', 0))
    raw_loc = sorted(loc_vals)[len(loc_vals)//2] if loc_vals else None
    
    # Smooth ILS signals (for flickering)
    alpha = 0.3
    if raw_gs is not None:
        S['filtered_gs'] = alpha * raw_gs + (1 - alpha) * S['filtered_gs']
        gs_dots = S['filtered_gs']
    else:
        gs_dots = S['filtered_gs']  # Use last known
    
    if raw_loc is not None:
        S['filtered_loc'] = alpha * raw_loc + (1 - alpha) * S['filtered_loc']
        loc_dots = S['filtered_loc']
    else:
        loc_dots = S['filtered_loc']
    
    # Radio altitude
    radio_alt = state.get('radio_alt_ft', None)
    if radio_alt is not None and radio_alt < -100:
        radio_alt = None
    
    # Baro altitude
    baro_vals = []
    for i in range(1, 4):
        if state.get(f'baro_alt_{i}_valid', False):
            baro_vals.append(state.get(f'baro_alt_{i}_ft', 0))
    baro_alt = sorted(baro_vals)[len(baro_vals)//2] if baro_vals else None
    
    # IAS
    ias_vals = []
    for i in range(1, 4):
        if state.get(f'ias_{i}_valid', False):
            ias_vals.append(state.get(f'ias_{i}_kt', 0))
    ias = sorted(ias_vals)[len(ias_vals)//2] if ias_vals else None
    
    # IRS
    irs_pitch_vals = []
    irs_roll_vals = []
    irs_heading_vals = []
    irs_vs_vals = []
    irs_gs_vals = []
    for i in range(1, 4):
        if state.get(f'irs_{i}_valid', False):
            irs_pitch_vals.append(state.get(f'irs_{i}_pitch_deg', 0))
            irs_roll_vals.append(state.get(f'irs_{i}_roll_deg', 0))
            irs_heading_vals.append(state.get(f'irs_{i}_heading_deg', 0))
            irs_vs_vals.append(state.get(f'irs_{i}_vs_fpm', 0))
            irs_gs_vals.append(state.get(f'irs_{i}_gs_kt', 0))
    
    pitch = sorted(irs_pitch_vals)[len(irs_pitch_vals)//2] if irs_pitch_vals else 0
    roll = sorted(irs_roll_vals)[len(irs_roll_vals)//2] if irs_roll_vals else 0
    heading = sorted(irs_heading_vals)[len(irs_heading_vals)//2] if irs_heading_vals else 0
    vs_fpm = sorted(irs_vs_vals)[len(irs_vs_vals)//2] if irs_vs_vals else -700
    groundspeed = sorted(irs_gs_vals)[len(irs_gs_vals)//2] if irs_gs_vals else 140
    
    # GPS
    gps_gs_vals = []
    gps_alt_vals = []
    for i in range(1, 4):
        if state.get(f'gps_{i}_valid', False):
            gps_gs_vals.append(state.get(f'gps_{i}_gs_kt', 0))
            gps_alt_vals.append(state.get(f'gps_{i}_alt_ft', 0))
    gps_gs = sorted(gps_gs_vals)[len(gps_gs_vals)//2] if gps_gs_vals else groundspeed
    gps_alt = sorted(gps_alt_vals)[len(gps_alt_vals)//2] if gps_alt_vals else None
    
    # Target parameters
    target_ias = state.get('target_ias_kt', 137)
    vref = state.get('target_vref_kt', 137)
    target_hdg = state.get('target_hdg_deg', heading)
    target_vs = state.get('target_vs_fpm', -700)
    target_alt = state.get('target_alt_ft', 0)
    
    # DME
    dme_vals = [state.get(f'dme_{i}_nm', 0) for i in range(1, 3)]
    dme_vals = [v for v in dme_vals if v is not None and v > 0]
    dme = sum(dme_vals) / len(dme_vals) if dme_vals else None
    
    # Engine status
    eng1_running = state.get('engine_1_running', True) and not state.get('engine_1_fire', False)
    eng2_running = state.get('engine_2_running', True) and not state.get('engine_2_fire', False)
    eng1_n1 = state.get('engine_1_n1_pct', 0)
    eng2_n1 = state.get('engine_2_n2_pct', 0)
    
    dual_engine_fail = not eng1_running and not eng2_running
    single_engine_fail = (not eng1_running or not eng2_running) and not dual_engine_fail
    
    # Ground
    on_ground = state.get('on_ground', False)
    wow = state.get('weight_on_wheels', False)
    groundspeed_kt = state.get('groundspeed_kt', groundspeed)
    
    # AoA
    aoa_vals = []
    for i in range(1, 4):
        if state.get(f'aoa_{i}_valid', False):
            aoa_vals.append(state.get(f'aoa_{i}_deg', 0))
    aoa = sorted(aoa_vals)[len(aoa_vals)//2] if aoa_vals else None
    
    # Below decision height
    below_dh = state.get('below_decision_height', False)
    
    # ============ ESTIMATE FIELD ELEVATION ============
    if radio_alt is not None and baro_alt is not None and radio_alt > 100:
        S['field_elev'] = baro_alt - radio_alt
    
    # If radar alt failed, estimate from baro
    est_radio_alt = radio_alt
    if radio_alt is None or radio_alt < -50:
        if S.get('field_elev') is not None and baro_alt is not None:
            est_radio_alt = baro_alt - S['field_elev']
        elif S.get('field_elev') is not None and gps_alt is not None:
            est_radio_alt = gps_alt - S['field_elev']
        else:
            # Use DME to estimate
            if dme is not None:
                est_radio_alt = dme * math.tan(math.radians(3)) * 6076
            else:
                est_radio_alt = 1000  # Default
    
    # ============ PHASE DETERMINATION ============
    
    if S['go_around_active']:
        phase = 'go_around'
    elif on_ground or wow:
        phase = 'rollout'
    elif est_radio_alt < 50:
        phase = 'flare'
    elif est_radio_alt < 200:
        phase = 'short_final'
    elif est_radio_alt < 1000:
        phase = 'approach'
    else:
        phase = 'intercept'
    
    # ============ GO-AROUND LOGIC ============
    
    should_go_around = False
    
    # Don't go around with dual engine failure
    if not dual_engine_fail and not S['go_around_active']:
        # Check stability
        if est_radio_alt is not None and est_radio_alt < 500 and est_radio_alt > 50:
            # GS deviation too large
            if gs_dots is not None and abs(gs_dots) > 2.0:
                should_go_around = True
            # LOC deviation too large
            if loc_dots is not None and abs(loc_dots) > 2.5:
                should_go_around = True
            # Speed too far off
            if ias is not None and (ias > vref + 25 or ias < vref - 15):
                should_go_around = True
        
        # Windshear detection
        if ias is not None and S.get('prev_ias') is not None and dt > 0:
            ias_rate = (ias - S['prev_ias']) / dt
            if ias_rate < -10:  # Losing more than 10kt/sec
                should_go_around = True
    
    if should_go_around and S['go_around_count'] < 9:
        S['go_around_active'] = True
        S['go_around_count'] += 1
        phase = 'go_around'
    
    # ============ CONTROL LAWS ============
    
    elevator = 0.0
    aileron = 0.0
    rudder = 0.0
    throttle_1 = 0.0
    throttle_2 = 0.0
    elevator_trim = 0.0
    flaps = 0.0
    gear = 0.0
    speedbrake = 0.0
    spoiler = 0.0
    autobrake = 0.0
    reverse_1 = 0.0
    reverse_2 = 0.0
    
    if phase == 'go_around':
        # TOGA thrust
        if eng1_running: throttle_1 = 1.0
        if eng2_running: throttle_2 = 1.0
        
        # Pitch up to 15°
        target_pitch_ga = 15.0
        elevator = 0.05 * (target_pitch_ga - pitch)
        elevator = max(-1, min(1, elevator))
        
        # Level wings
        aileron = -0.05 * roll
        aileron = max(-1, min(1, aileron))
        
        # Gear up
        gear = 0.0
        
        # Flaps - reduce
        flaps = 15.0
        
        # No speedbrake
        speedbrake = 0.0
        
        # Check if reached 1500ft AGL
        if est_radio_alt > 1500:
            S['go_around_active'] = False
    
    elif phase in ['intercept', 'approach', 'short_final']:
        
        # ---- GLIDESLOPE PITCH CONTROL ----
        if gs_dots is not None:
            # Compute target VS from GS deviation
            nominal_vs = -groundspeed * math.sin(math.radians(3)) * 101.27
            gs_correction = -150 * gs_dots  # fpm per dot
            target_vs_gs = nominal_vs + gs_correction
        else:
            target_vs_gs = target_vs
        
        vs_error = target_vs_gs - vs_fpm
        S['gs_int'] += vs_error * dt
        S['gs_int'] = max(-100, min(100, S['gs_int']))
        
        elevator = 0.0008 * vs_error + 0.00005 * S['gs_int']
        
        # Pitch damping
        elevator -= 0.005 * pitch
        
        # ---- LOCALIZER ROLL CONTROL ----
        if loc_dots is not None:
            target_roll_loc = -4.0 * loc_dots  # degrees per dot
            target_roll_loc = max(-30, min(30, target_roll_loc))
