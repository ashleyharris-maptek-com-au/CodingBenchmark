- More precisely, it's a tree that connects all terminals in S, and v is a vertex in the tree.

- dp[{t}][v] = dist(t, v) for each terminal t and vertex v

// ===== Dijkstra from single source =====
pair<vector<ll>, vector<int>> dijkstra_single(int src) {
    vector<ll> dist(n, INF);
    vector<int> pred(n, -1);
    dist[src] = 0;
    priority_queue<pli, vector<pli>, greater<pli>> pq;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto& e : adj[u]) {
            if (dist[u] + e.w < dist[e.to]) {
                dist[e.to] = dist[u] + e.w;
                pred[e.to] = u;
                pq.push({dist[e.to], e.to});
            }
        }
    }
    return {dist, pred};
}

// ===== Dreyfus-Wagner DP =====
pair<ll, set<pair<int,int>>> steiner_dw() {
    int full = (1 << k) - 1;
    
    // dp[mask][v]
    vector<vector<ll>> dp(full + 1, vector<ll>(n, INF));
    
    // Precompute shortest paths from each terminal
    vector<vector<ll>> term_dist(k);
    vector<vector<int>> term_pred(k);
    for (int i = 0; i < k; i++) {
        auto [d, p] = dijkstra_single(terminals[i]);
        term_dist[i] = d;
        term_pred[i] = p;
        int mask = 1 << i;
        for (int v = 0; v < n; v++) {
            dp[mask][v] = d[v];
        }
    }
    
    // DP
    for (int mask = 1; mask <= full; mask++) {
        if (__builtin_popcount(mask) == 1) continue;
        
        // Combine step
        for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
            int comp = mask ^ sub;
            if (sub > comp) continue;
            for (int v = 0; v < n; v++) {
                if (dp[sub][v] < INF && dp[comp][v] < INF) {
                    ll val = dp[sub][v] + dp[comp][v];
                    if (val < dp[mask][v]) {
                        dp[mask][v] = val;
                    }
                }
            }
        }
        
        // Propagation: multi-source Dijkstra
        vector<ll>& cur = dp[mask];
        priority_queue<pli, vector<pli>, greater<pli>> pq;
        for (int v = 0; v < n; v++) {
            if (cur[v] < INF) pq.push({cur[v], v});
        }
        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d > cur[u]) continue;
            for (auto& e : adj[u]) {
                if (cur[u] + e.w < cur[e.to]) {
                    cur[e.to] = cur[u] + e.w;
                    pq.push({cur[e.to], e.to});
                }
            }
        }
    }
    
    // Find best vertex
    ll best = INF;
    int best_v = -1;
    for (int v = 0; v < n; v++) {
        if (dp[full][v] < best) {
            best = dp[full][v];
            best_v = v;
        }
    }
    
    // Reconstruct
    set<pair<int,int>> result;
    
    function<void(int, int)> recon = [&](int mask, int v) {
        if (__builtin_popcount(mask) == 1) {
            int ti = __builtin_ctz(mask);
            int t = terminals[ti];
            int cur = v;
            while (cur != t) {
                int p = term_pred[ti][cur];
                int a = min(cur, p), b = max(cur, p);
                result.insert({a, b});
                cur = p;
            }
            return;
        }
        
        // Try combine
        for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
            int comp = mask ^ sub;
            if (sub > comp) continue;
            if (dp[sub][v] < INF && dp[comp][v] < INF && dp[sub][v] + dp[comp][v] == dp[mask][v]) {
                recon(sub, v);
                recon(comp, v);
                return;
            }
        }
        
        // Must be from propagation
        for (auto& e : adj[v]) {
            if (dp[mask][e.to] < INF && dp[mask][e.to] + e.w == dp[mask][v]) {
                int a = min(v, e.to), b = max(v, e.to);
                result.insert({a, b});
                recon(mask, e.to);
                return;
            }
        }
    };
    
    recon(full, best_v);
    return {best, result};
}

// ===== Shortest Path Heuristic =====
pair<ll, set<pair<int,int>>> sph() {
    set<int> in_tree;
    set<int> connected; // terminal indices
    set<pair<int,int>> result;
    
    in_tree.insert(terminals[0]);
    connected.insert(0);
    
    while ((int)connected.size() < k) {
        // Dijkstra from all tree vertices
        vector<ll> dist(n, INF);
        vector<int> pred(n, -1);
        priority_queue<pli, vector<pli>, greater<pli>> pq;
        for (int v : in_tree) {
            dist[v] = 0;
            pq.push({0, v});
        }
        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d > dist[u]) continue;
            for (auto& e : adj[u]) {
                if (dist[u] + e.w < dist[e.to]) {
                    dist[e.to] = dist[u] + e.w;
                    pred[e.to] = u;
                    pq.push({dist[e.to], e.to});
                }
            }
        }
        
        // Find closest unconnected terminal
        ll best_dist = INF;
        int best_idx = -1;
        for (int i = 0; i < k; i++) {
            if (connected.find(i) == connected.end() && dist[terminals[i]] < best_dist) {
                best_dist = dist[terminals[i]];
                best_idx = i;
            }
        }
        
        // Add path
        int cur = terminals[best_idx];
        while (in_tree.find(cur) == in_tree.end()) {
            int p = pred[cur];
            int a = min(cur, p), b = max(cur, p);
            result.insert({a, b});
            in_tree.insert(cur);
            cur = p;
        }
        connected.insert(best_idx);
    }
    
    return {compute_weight(result), result};
}
