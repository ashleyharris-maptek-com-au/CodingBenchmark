def string_words(s):
    """Encode string as SPIR-V words (null-terminated, padded)."""
    s_bytes = s.encode('ascii') + b'\x00'
    # Pad to 4-byte boundary
    while len(s_bytes) % 4 != 0:
        s_bytes += b'\x00'
    words = []
    for i in range(0, len(s_bytes), 4):
        word = struct.unpack('<I', s_bytes[i:i+4])[0]
        words.append(word)
    return words
