Now, for the strategy. The key aspects of a good Tetris player are:

1. **Minimize holes** - gaps created below placed blocks that can't be filled easily
2. **Keep the board flat** - minimize the height differences between columns
3. **Clear lines** - maximize line clears, especially multiple lines at once
4. **Don't stack too high** - keep the maximum height low
5. **Prepare for I-pieces** - keep a column open for Tetris clears

A common evaluation function considers:
- Aggregate height (sum of column heights) - negative weight
- Complete lines - positive weight
- Holes - negative weight
- Bumpiness (sum of |height[i] - height[i+1]|) - negative weight

For the adaptive strategy:
- Short games: focus on survival, keep board flat
- Long games: focus on line clearing efficiency, prepare I-piece columns
- When board is getting high: prioritize line clears and keeping things low

    static (int rot, int col) FindBestPlacement(char pieceType)
    {
        var rotations = pieceDefs[pieceType];
        double bestScore = double.NegativeInfinity;
        int bestRot = 0, bestCol = 0;
        
        for (int rot = 0; rot < 4; rot++)
        {
            var cells = rotations[rot];
            int pieceWidth = cells.Max(c => c.c) + 1;
            
            for (int col = 0; col <= boardWidth - pieceWidth; col++)
            {
                // Find the drop row
                int dropRow = FindDropRow(cells, col);
                
                if (dropRow < 0) continue; // Can't place here
                
                // Check if any cell is above the board
                bool aboveBoard = false;
                foreach (var (r, c) in cells)
                {
                    if (dropRow + r < 0)
                    {
                        aboveBoard = true;
                        break;
                    }
                }
                if (aboveBoard) continue;
                
                // Evaluate this placement
                double score = EvaluatePlacement(cells, dropRow, col);
                
                if (score > bestScore)
                {
                    bestScore = score;
                    bestRot = rot;
                    bestCol = col;
                }
            }
        }
        
        return (bestRot, bestCol);
    }
    
    static int FindDropRow(List<(int r, int c)> cells, int col)
    {
        // Find the highest row where the piece can be dropped (gravity)
        // The piece falls from above. We need to find the row such that:
        // - All cells of the piece at (row + offset_r, col + offset_c) are within bounds
        // - All cells are on empty board positions
        // - The piece can't go any lower
        
        // Start from top and find where it lands
        // Actually, let's compute the landing position by dropping from the top
        
        int minRow = -4; // piece can start above the board
        int maxRow = boardHeight - 1;
        
        int landingRow = minRow;
        
        for (int row = minRow; row <= maxRow; row++)
        {
            bool canPlace = true;
            foreach (var (r, c) in cells)
            {
                int boardR = row + r;
                int boardC = col + c;
                
                if (boardR < 0) continue; // above board is fine
                if (boardR >= boardHeight || boardC < 0 || boardC >= boardWidth)
                {
                    canPlace = false;
                    break;
                }
                if (board[boardR, boardC])
                {
                    canPlace = false;
                    break;
                }
            }
            
            if (canPlace)
            {
                landingRow = row;
            }
            else
            {
                break;
            }
        }
        
        return landingRow;
    }
    
    static double EvaluatePlacement(List<(int r, int c)> cells, int dropRow, int col)
    {
        // Simulate placing the piece
        bool[,] simBoard = (bool[,])board.Clone();
        
        foreach (var (r, c) in cells)
        {
            int boardR = dropRow + r;
            int boardC = col + c;
            if (boardR >= 0 && boardR < boardHeight && boardC >= 0 && boardC < boardWidth)
            {
                simBoard[boardR, boardC] = true;
            }
        }
        
        // Count complete lines
        int completeLines = 0;
        for (int r = 0; r < boardHeight; r++)
        {
            bool complete = true;
            for (int c = 0; c < boardWidth; c++)
            {
                if (!simBoard[r, c])
                {
                    complete = false;
                    break;
                }
            }
            if (complete) completeLines++;
        }
        
        // Calculate heights
        int[] heights = new int[boardWidth];
        for (int c = 0; c < boardWidth; c++)
        {
            heights[c] = 0;
            for (int r = 0; r < boardHeight; r++)
            {
                if (simBoard[r, c])
                {
                    heights[c] = boardHeight - r;
                    break;
                }
            }
        }
        
        int aggregateHeight = heights.Sum();
        int maxHeight = heights.Max();
        
        // Count holes
        int holes = 0;
        for (int c = 0; c < boardWidth; c++)
        {
            bool foundBlock = false;
            for (int r = 0; r < boardHeight; r++)
            {
                if (simBoard[r, c])
                {
                    foundBlock = true;
                }
                else if (foundBlock)
                {
                    holes++;
                }
            }
        }
        
        // Calculate bumpiness
        int bumpiness = 0;
        for (int c = 0; c < boardWidth - 1; c++)
        {
            bumpiness += Math.Abs(heights[c] - heights[c + 1]);
        }
        
        // Adaptive weights based on game state
        double heightWeight, linesWeight, holesWeight, bumpinessWeight, maxHoleDepth;
        
        if (maxHeight > boardHeight * 0.75)
        {
            // Danger zone - prioritize line clears and low height
            heightWeight = -0.7;
            linesWeight = 1.0;
            holesWeight = -0.8;
            bumpinessWeight = -0.2;
        }
        else if (maxHeight < boardHeight * 0.3)
        {
            // Safe zone - can afford to build
            heightWeight = -0.3;
            linesWeight = 0.6;
            holesWeight = -0.7;
            bumpinessWeight = -0.3;
        }
        else
        {
            // Medium
            heightWeight = -0.5;
            linesWeight = 0.76;
            holesWeight = -0.75;
            bumpinessWeight = -0.25;
        }
        
        double score = heightWeight * aggregateHeight 
                      + linesWeight * completeLines 
                      + holesWeight * holes 
                      + bumpinessWeight * bumpiness;
        
        // Bonus for wells (for I-piece clears)
        // ... could add well depth bonus
        
        return score;
    }
    
    static void ApplyPlacement(char pieceType, int rot, int col)
    {
        var cells = pieceDefs[pieceType][rot];
        int dropRow = FindDropRow(cells, col);
        
        foreach (var (r, c) in cells)
        {
            int boardR = dropRow + r;
            int boardC = col + c;
            if (boardR >= 0 && boardR < boardHeight && boardC >= 0 && boardC < boardWidth)
            {
                board[boardR, boardC] = true;
            }
        }
    }
    
    static void ClearLines(int numCleared)
    {
        // Remove completed lines from the board
        List<int> clearedRows = new List<int>();
        for (int r = 0; r < boardHeight; r++)
        {
            bool complete = true;
            for (int c = 0; c < boardWidth; c++)
            {
                if (!board[r, c])
                {
                    complete = false;
                    break;
                }
            }
            if (complete) clearedRows.Add(r);
        }
        
        // Sort descending so we remove from bottom up
        clearedRows.Sort((a, b) => b.CompareTo(a));
        
        foreach (int r in clearedRows)
        {
            // Shift everything above down
            for (int row = r; row > 0; row--)
            {
                for (int c = 0; c < boardWidth; c++)
                {
                    board[row, c] = board[row - 1, c];
                }
            }
            // Top row is empty
            for (int c = 0; c < boardWidth; c++)
            {
                board[0, c] = false;
            }
        }
    }
}

            if (result.StartsWith("cleared_"))
            {
                int numCleared = int.Parse(result.Substring(8));
                ClearLines(numCleared);
            }
        }
    }
    
    static int PieceIndex(char c)
    {
        return c switch { 'I' => 0, 'O' => 1, 'T' => 2, 'S' => 3, 'Z' => 4, 'J' => 5, 'L' => 6, _ => 0 };
    }

    static int FindDropRow(PieceShape shape, int col)
    {
        // Find the row where the piece lands (highest row where it fits, considering gravity)
        // Start from the top and go down until we can't go further
        
        int result = -shape.Rows; // can start above the board
        for (int row = result; row < H; row++)
        {
            bool canPlace = true;
            foreach (var (dr, dc) in shape.Cells)
            {
                int br = row + dr;
                int bc = col + dc;
                if (bc < 0 || bc >= W) { canPlace = false; break; }
                if (br >= H) { canPlace = false; break; }
                if (br >= 0 && board[br, bc]) { canPlace = false; break; }
            }
            if (canPlace)
            {
                result = row;
            }
            else if (row > result + shape.Rows + 2)
            {
                // We've gone past where we could place, and then some
                // Actually, just break when we first can't place after we could
                break;
            }
        }
        
        return result;
    }
    
    static bool CanPlace(PieceShape shape, int row, int col)
    {
        foreach (var (dr, dc) in shape.Cells)
        {
            int br = row + dr;
            int bc = col + dc;
            if (bc < 0 || bc >= W) return false;
            if (br >= H) return false;
            if (br >= 0 && board[br, bc]) return false;
        }
        return true;
    }
