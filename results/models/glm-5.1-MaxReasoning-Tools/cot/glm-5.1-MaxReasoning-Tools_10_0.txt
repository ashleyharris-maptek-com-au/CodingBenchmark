struct MaxRectsBoard {

    void init(int w, int h) {
        W = w; H = h;
        freeRects.clear();
        freeRects.push_back({0, 0, w, h});
    }
    
    // Find best placement using BSSF
    // Returns true if found, sets bx, by, brotated
    bool findBest(int rw, int rh, int& bx, int& by, int& brotated, int& bscore) {
        bool found = false;
        bscore = INT_MAX;
        
        // Try without rotation
        for (auto& f : freeRects) {
            if (rw <= f.w && rh <= f.h) {
                int leftover = min(f.w - rw, f.h - rh);
                if (leftover < bscore) {
                    bscore = leftover;
                    bx = f.x;
                    by = f.y;
                    brotated = 0;
                    found = true;
                }
            }
        }
        
        // Try with rotation (if different)
        if (rw != rh) {
            for (auto& f : freeRects) {
                if (rh <= f.w && rw <= f.h) {
                    int leftover = min(f.w - rh, f.h - rw);
                    if (leftover < bscore) {
                        bscore = leftover;
                        bx = f.x;
                        by = f.y;
                        brotated = 1;
                        found = true;
                    }
                }
            }
        }
        
        return found;
    }

    void place(int px, int py, int pw, int ph) {
        vector<Rect> newFreeRects;
        
        for (auto& f : freeRects) {
            // Check if f overlaps with placed rect
            if (px >= f.x + f.w || px + pw <= f.x || py >= f.y + f.h || py + ph <= f.y) {
                // No overlap, keep f
                newFreeRects.push_back(f);
                continue;

            // Left part
            if (px > f.x) {

            if (px + pw < f.x + f.w) {

            }

                newFreeRects.push_back({f.x, py + ph, f.w, f.y + f.h - py - ph});

            }
        }
        
        freeRects = filtered;
    }
