```cpp
    uint64_t d = 0;
```

```cpp
struct City {

double dist(const City& a, const City& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy; // squared distance for comparison
}

double real_dist(const City& a, const City& b) {
    return sqrt(dist(a, b));
}

// Brute force for small N
void brute_force(const vector<City>& cities, vector<int>& tour) {
    int n = cities.size();
    vector<int> perm(n - 1);
    iota(perm.begin(), perm.end(), 1);
    
    double best = 1e18;
    vector<int> best_perm = perm;
    
    do {
        double d = real_dist(cities[0], cities[perm[0]]);
        for (int i = 1; i < n - 1; i++) {
            d += real_dist(cities[perm[i-1]], cities[perm[i]]);
        }
        d += real_dist(cities[perm[n-2]], cities[0]);

            best = d;
            best_perm = perm;
        }
    } while (next_permutation(perm.begin(), perm.end()));
    
    tour[0] = 0;
    for (int i = 0; i < n - 1; i++) {
        tour[i + 1] = best_perm[i];
    }
}

// Hilbert curve encoding
uint64_t hilbert_xy2d(int n, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (uint64_t s = (1ULL << (n - 1)); s > 0; s >>= 1) {
        uint64_t rx = (x & s) ? 1 : 0;
        uint64_t ry = (y & s) ? 1 : 0;
        d += s * s * ((3 * rx) ^ ry);
        // rotate
        if (ry == 0) {
            if (rx == 1) {
                x = s - 1 - x;
                y = s - 1 - y;
            }
            swap(x, y);
        }
    }
    return d;
}

void hilbert_sort(vector<City>& cities, vector<int>& tour) {
    int n = cities.size();
    
    // Find bounding box
    double minx = 1e18, miny = 1e18, maxx = -1e18, maxy = -1e18;
    for (int i = 0; i < n; i++) {
        minx = min(minx, cities[i].x);
        miny = min(miny, cities[i].y);
        maxx = max(maxx, cities[i].x);
        maxy = max(maxy, cities[i].y);
    }
    
    double range = max(maxx - minx, maxy - miny);
    if (range < 1e-10) range = 1.0;
    
    int order = 0;
    while ((1ULL << order) < (uint64_t)n || order < 16) order++;
    if (order > 31) order = 31; // limit to avoid overflow
    
    vector<pair<uint64_t, int>> hilbert_idx(n);
    for (int i = 0; i < n; i++) {
        uint64_t ix = (uint64_t)((cities[i].x - minx) / range * ((1ULL << order) - 1));
        uint64_t iy = (uint64_t)((cities[i].y - miny) / range * ((1ULL << order) - 1));
        hilbert_idx[i] = {hilbert_xy2d(order, ix, iy), i};
    }
    
    sort(hilbert_idx.begin(), hilbert_idx.end());
    
    // Find position of city 0 and rotate tour to start there
    int start_pos = 0;
    for (int i = 0; i < n; i++) {
        if (hilbert_idx[i].second == 0) {
            start_pos = i;
            break;
        }
    }
    
    for (int i = 0; i < n; i++) {
        tour[i] = hilbert_idx[(start_pos + i) % n].second;
    }
}

                    }
                }
            }

        visited[best_next] = true;

    }
```

```cpp
struct City {

inline double dist2(int a, int b) {
    double dx = cities[a].x - cities[b].x;
    double dy = cities[a].y - cities[b].y;
    return dx*dx + dy*dy;
}

inline double dist(int a, int b) {
    return sqrt(dist2(a, b));
}

// ---- Brute Force for small N ----
void brute_force(vector<int>& tour) {
    vector<int> perm(N - 1);
    iota(perm.begin(), perm.end(), 1);
    
    double best = 1e18;
    vector<int> best_perm = perm;
    
    do {
        double d = dist(0, perm[0]);
        for (int i = 1; i < N - 1; i++) d += dist(perm[i-1], perm[i]);
        d += dist(perm[N-2], 0);
        if (d < best) { best = d; best_perm = perm; }
    } while (next_permutation(perm.begin(), perm.end()));
    
    tour[0] = 0;
    for (int i = 0; i < N - 1; i++) tour[i+1] = best_perm[i];
}

// ---- Hilbert Curve Sort ----
uint64_t hilbert_xy2d(int order, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (uint64_t s = (1ULL << (order - 1)); s > 0; s >>= 1) {
        uint64_t rx = (x & s) ? 1 : 0;
        uint64_t ry = (y & s) ? 1 : 0;
        d += s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            swap(x, y);
        }
    }
    return d;
}

void hilbert_tour(vector<int>& tour) {
    double minx = 1e18, miny = 1e18, maxx = -1e18, maxy = -1e18;
    for (int i = 0; i < N; i++) {
        minx = min(minx, cities[i].x);
        miny = min(miny, cities[i].y);
        maxx = max(maxx, cities[i].x);
        maxy = max(maxy, cities[i].y);
    }
    
    double range = max(maxx - minx, maxy - miny);
    if (range < 1e-10) range = 1.0;
    
    int order = 16;
    while ((1ULL << order) < (uint64_t)N * 2) order++;
    if (order > 31) order = 31;
    
    struct HI { uint64_t h; int idx; };
    vector<HI> hidx(N);
    
    double scale = ((1ULL << order) - 1) / range;
    for (int i = 0; i < N; i++) {
        uint64_t ix = (uint64_t)((cities[i].x - minx) * scale);
        uint64_t iy = (uint64_t)((cities[i].y - miny) * scale);
        if (ix >= (1ULL << order)) ix = (1ULL << order) - 1;
        if (iy >= (1ULL << order)) iy = (1ULL << order) - 1;
        hidx[i] = {hilbert_xy2d(order, ix, iy), i};
    }
    
    sort(hidx.begin(), hidx.end(), [](const HI& a, const HI& b) { return a.h < b.h; });
    
    int start_pos = 0;
    for (int i = 0; i < N; i++) {
        if (hidx[i].idx == 0) { start_pos = i; break; }
    }
    
    for (int i = 0; i < N; i++) {
        tour[i] = hidx[(start_pos + i) % N].idx;
    }
}

// ---- Nearest Neighbor Construction ----
void nn_tour(vector<int>& tour) {
    vector<bool> visited(N, false);
    tour[0] = 0;
    visited[0] = true;
    
    int cur = 0;
    for (int step = 1; step < N; step++) {
        double best_d = 1e18;
        int best_next = -1;
        for (int i = 0; i < N; i++) {
            if (!visited[i]) {
                double d = dist2(cur, i);
                if (d < best_d) { best_d = d; best_next = i; }
            }
        }
        tour[step] = best_next;
        visited[best_next] = true;
        cur = best_next;
    }
}

// ---- 2-opt Local Search ----
void two_opt(vector<int>& tour) {
    vector<int> pos(N);
    for (int i = 0; i < N; i++) pos[tour[i]] = i;
    
    vector<bool> dont_look(N, false);
    bool improved = true;
    
    while (improved) {
        improved = false;
        for (int i = 0; i < N; i++) {
            if (dont_look[tour[i]]) continue;
            bool local_improved = false;
            int pi = tour[i];
            int pi1 = tour[(i + 1) % N];
            
            for (int j = i + 2; j < N; j++) {
                if (i == 0 && j == N - 1) continue; // skip if they're adjacent
                int pj = tour[j];
                int pj1 = tour[(j + 1) % N];
                
                double d_old = dist2(pi, pi1) + dist2(pj, pj1);
                double d_new = dist2(pi, pj) + dist2(pi1, pj1);
                
                if (d_new < d_old - 1e-10) {
                    // Reverse segment from i+1 to j
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    // Update positions
                    for (int k = i + 1; k <= j; k++) pos[tour[k]] = k;
                    // Unmark affected cities
                    dont_look[pi] = false;
                    dont_look[pi1] = false;
                    dont_look[pj] = false;
                    dont_look[pj1] = false;
                    
                    pi1 = tour[(i + 1) % N]; // updated
                    improved = true;
                    local_improved = true;
                }
            }
            if (!local_improved) dont_look[tour[i]] = true;
        }
    }
}
```

```cpp
// Brute force for N <= 11
void brute_force(vector<int>& tour) {
    vector<int> perm(N - 1);
    iota(perm.begin(), perm.end(), 1);
    double best = 1e18;
    vector<int> best_perm = perm;
    do {
        double d = 0;
        int prev = 0;
        for (int i = 0; i < N-1; i++) {
            d += sqrt(dist2(prev, perm[i]));
            prev = perm[i];
        }
        d += sqrt(dist2(prev, 0));

    } while (next_permutation(perm.begin(), perm.end()));

// Hilbert curve sort
uint64_t hilbert_xy2d(int order, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (uint64_t s = (1ULL << (order - 1)); s > 0; s >>= 1) {
        uint64_t rx = (x & s) ? 1 : 0;
        uint64_t ry = (y & s) ? 1 : 0;
        d += s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            swap(x, y);
        }
    }
    return d;
}

void hilbert_tour(vector<int>& tour) {
    double minx = 1e18, miny = 1e18, maxx = -1e18, maxy = -1e18;
    for (int i = 0; i < N; i++) {
        minx = min(minx, cities[i].x);
        miny = min(miny, cities[i].y);
        maxx = max(maxx, cities[i].x);
        maxy = max(maxy, cities[i].y);
    }
    double range = max(maxx - minx, maxy - miny);
    if (range < 1e-10) range = 1.0;
    
    int order = 16;
    while ((1ULL << order) < (uint64_t)N * 2) order++;
    if (order > 31) order = 31;
    
    double scale = (double)((1ULL << order) - 1) / range;
    uint64_t max_coord = (1ULL << order) - 1;
    
    struct HI { uint64_t h; int idx; };
    vector<HI> hidx(N);
    for (int i = 0; i < N; i++) {
        uint64_t ix = (uint64_t)((cities[i].x - minx) * scale);
        uint64_t iy = (uint64_t)((cities[i].y - miny) * scale);
        if (ix > max_coord) ix = max_coord;
        if (iy > max_coord) iy = max_coord;
        hidx[i] = {hilbert_xy2d(order, ix, iy), i};
    }
    sort(hidx.begin(), hidx.end(), [](const HI& a, const HI& b) { return a.h < b.h; });
    
    int start_pos = 0;
    for (int i = 0; i < N; i++) {
        if (hidx[i].idx == 0) { start_pos = i; break; }
    }
    for (int i = 0; i < N; i++) {
        tour[i] = hidx[(start_pos + i) % N].idx;
    }
}

void nn_tour(vector<int>& tour) {

    tour[0] = 0;
    visited[0] = true;
    int cur = 0;
    for (int step = 1; step < N; step++) {
        double best_d = 1e18;
        int best_next = -1;
        for (int i = 0; i < N; i++) {

        cur = best_next;
    }
}

// Full 2-opt
void two_opt(vector<int>& tour) {
    vector<bool> dont_look(N, false);
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N; i++) {
            if (dont_look[tour[i]]) continue;
            bool local_improved = false;
            int ci = tour[i];
            for (int j = i + 2; j < N; j++) {
                if (i == 0 && j == N - 1) continue;

                double d_old = dist2(ci, ci1) + dist2(cj, cj1);

        }

}
```

2. Search expanding Chebyshev distance rings
3. For each cell in the ring, check all cities and update the best candidate
4. Stop when the minimum distance to the next ring exceeds the best distance found

- max(0, x1 - px, px - x2)² + max(0, y1 - py, py - y2)² (squared)

```cpp
        miny = min(miny, cities[i].y);
        maxx = max(maxx, cities[i].x);
        maxy = max(maxy, cities[i].y);
    }
    
    double rangex = maxx - minx;
    double rangey = maxy - miny;

    int G = max(1, (int)sqrt(N / 4.0));

    double cell_h = rangey / G;
    if (cell_w < 1e-10) cell_w = 1.0;
    if (cell_h < 1e-10) cell_h = 1.0;
    
    vector<vector<int>> grid(G * G);

        int cy = (int)((y - miny) / cell_h);

        cy = max(0, min(cy, G-1));
```

```cpp
inline double dist2(int a, int b) {
    double dx = cities[a].x - cities[b].x;
    double dy = cities[a].y - cities[b].y;
    return dx*dx + dy*dy;

// ---- Brute Force for small N ----
void brute_force(vector<int>& tour) {
    vector<int> perm(N - 1);
    iota(perm.begin(), perm.end(), 1);
    double best = 1e18;
    vector<int> best_perm = perm;
    do {
        double d = 0;
        int prev = 0;
        for (int i = 0; i < N-1; i++) {
            d += sqrt(dist2(prev, perm[i]));
            prev = perm[i];
        }
        d += sqrt(dist2(prev, 0));
        if (d < best) { best = d; best_perm = perm; }
    } while (next_permutation(perm.begin(), perm.end()));
    tour[0] = 0;
    for (int i = 0; i < N-1; i++) tour[i+1] = best_perm[i];
}

// ---- Hilbert Curve Sort ----
static uint64_t hilbert_xy2d(int order, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (uint64_t s = (1ULL << (order - 1)); s > 0; s >>= 1) {
        uint64_t rx = (x & s) ? 1 : 0;
        uint64_t ry = (y & s) ? 1 : 0;
        d += s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            swap(x, y);
        }
    }
    return d;
}

void hilbert_tour(vector<int>& tour) {
    double minx = 1e18, miny = 1e18, maxx = -1e18, maxy = -1e18;
    for (int i = 0; i < N; i++) {
        minx = min(minx, cities[i].x);
        miny = min(miny, cities[i].y);
        maxx = max(maxx, cities[i].x);
        maxy = max(maxy, cities[i].y);
    }
    double range = max(maxx - minx, maxy - miny);
    if (range < 1e-10) range = 1.0;
    
    int order = 1;
    while ((1ULL << order) < (uint64_t)N * 2) order++;
    if (order > 31) order = 31;
    
    double scale = (double)((1ULL << order) - 1) / range;
    uint64_t maxc = (1ULL << order) - 1;
    
    struct HI { uint64_t h; int idx; };
    vector<HI> hidx(N);
    for (int i = 0; i < N; i++) {
        uint64_t ix = (uint64_t)((cities[i].x - minx) * scale);
        uint64_t iy = (uint64_t)((cities[i].y - miny) * scale);
        if (ix > maxc) ix = maxc;
        if (iy > maxc) iy = maxc;
        hidx[i] = {hilbert_xy2d(order, ix, iy), i};
    }
    sort(hidx.begin(), hidx.end(), [](const HI& a, const HI& b) { return a.h < b.h; });
    
    int start_pos = 0;
    for (int i = 0; i < N; i++) {
        if (hidx[i].idx == 0) { start_pos = i; break; }
    }
    
    // Try both directions, pick shorter
    double fwd = 0, bwd = 0;
    for (int i = 0; i < N; i++) {
        int a = hidx[(start_pos + i) % N].idx;
        int b = hidx[(start_pos + i + 1) % N].idx;
        fwd += dist2(a, b);
    }
    for (int i = 0; i < N; i++) {
        int a = hidx[(start_pos - i + N) % N].idx;
        int b = hidx[(start_pos - i - 1 + N) % N].idx;
        bwd += dist2(a, b);
    }
    
    if (fwd <= bwd) {
        for (int i = 0; i < N; i++) tour[i] = hidx[(start_pos + i) % N].idx;
    } else {
        for (int i = 0; i < N; i++) tour[i] = hidx[(start_pos - i + N) % N].idx;
    }
}

// ---- Nearest Neighbor Construction ----
void nn_tour(vector<int>& tour) {
    vector<char> visited(N, 0);
    tour[0] = 0;
    visited[0] = 1;
    int cur = 0;
    for (int step = 1; step < N; step++) {
        double best_d = 1e18;
        int best_next = -1;
        for (int i = 0; i < N; i++) {
            if (!visited[i]) {
                double d = dist2(cur, i);
                if (d < best_d) { best_d = d; best_next = i; }
            }
        }
        tour[step] = best_next;
        visited[best_next] = 1;
        cur = best_next;
    }
}

// ---- Full 2-opt ----
void two_opt(vector<int>& tour) {
    vector<char> dont_look(N, 0);
    bool improved = true;
    int iter = 0;
    while (improved && iter < 100) {
        improved = false;
        iter++;
        for (int i = 0; i < N; i++) {
            if (dont_look[tour[i]]) continue;
            bool local_improved = false;
            int ci = tour[i];
            int ci1 = tour[(i + 1) % N];
            for (int j = i + 2; j < N; j++) {
                if (i == 0 && j == N - 1) continue;
                int cj = tour[j];
                int cj1 = tour[(j + 1) % N];
                double d_old = dist2(ci, ci1) + dist2(cj, cj1);
                double d_new = dist2(ci, cj) + dist2(ci1, cj1);
                if (d_new < d_old - 1e-10) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    dont_look[ci] = 0;
                    dont_look[cj] = 0;
                    dont_look[tour[(i+1)%N]] = 0;
                    dont_look[cj1] = 0;
                    ci1 = tour[(i+1)%N];
                    improved = true;
                    local_improved = true;
                    break;
                }
            }
            if (!local_improved) dont_look[tour[i]] = 1;
        }
    }
}

void limited_two_opt(vector<int>& tour, int K) {

                        improved = true;
                        local_improved = true;

                }

    }
}
```

```cpp
void or_opt(vector<int>& tour, int K) {
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N; i++) {
            int ci = tour[i];
            int prev = tour[(i - 1 + N) % N];
            int next = tour[(i + 1) % N];
            double d_removed = dist2(prev, next); // new edge after removing ci
            double d_old = dist2(prev, ci) + dist2(ci, next); // old edges
            
            // Try inserting ci after position j (between tour[j] and tour[j+1])
            for (int dj = -K; dj <= K; dj++) {
                int j = (i + dj + N) % N;
                if (j == i || j == (i - 1 + N) % N) continue;
                int tj = tour[j];
                int tj1 = tour[(j + 1) % N];
                double d_new_edges = dist2(tj, ci) + dist2(ci, tj1);
                double d_old_edges = dist2(tj, tj1);
                
                if (d_removed + d_new_edges + d_old < d_old_edges + d_old + 1e-10) {
                    // Improvement found
                    // Remove ci from position i, insert after position j
                    if (j > i) {
                        // Shift left from i+1 to j
                        for (int k = i; k < j; k++) tour[k] = tour[k+1];
```

```cpp
void brute_force(vector<int>& tour) {

// ============ Hilbert Curve Sort ============
static uint64_t hilbert_xy2d(int order, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (uint64_t s = (1ULL << (order - 1)); s > 0; s >>= 1) {
        uint64_t rx = (x & s) ? 1 : 0;
        uint64_t ry = (y & s) ? 1 : 0;
        d += s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            uint64_t t = x; x = y; y = t;
        }
    }
    return d;
}

void hilbert_tour(vector<int>& tour) {
    double minx = 1e18, miny = 1e18, maxx = -1e18, maxy = -1e18;
    for (int i = 0; i < N; i++) {
        if (cities[i].x < minx) minx = cities[i].x;
        if (cities[i].y < miny) miny = cities[i].y;
        if (cities[i].x > maxx) maxx = cities[i].x;
        if (cities[i].y > maxy) maxy = cities[i].y;
    }
    double range = maxx - minx;
    if (maxy - miny > range) range = maxy - miny;
    if (range < 1e-10) range = 1.0;
    
    int order = 1;
    while ((1ULL << order) < (uint64_t)N * 2) order++;
    if (order > 31) order = 31;
    
    double scale = (double)((1ULL << order) - 1) / range;
    uint64_t maxc = (1ULL << order) - 1;
    
    struct HI { uint64_t h; int idx; };
    vector<HI> hidx(N);
    for (int i = 0; i < N; i++) {
        uint64_t ix = (uint64_t)((cities[i].x - minx) * scale);
        uint64_t iy = (uint64_t)((cities[i].y - miny) * scale);
        if (ix > maxc) ix = maxc;
        if (iy > maxc) iy = maxc;
        hidx[i].h = hilbert_xy2d(order, ix, iy);
        hidx[i].idx = i;
    }
    sort(hidx.begin(), hidx.end(), [](const HI& a, const HI& b) { return a.h < b.h; });
    
    int start_pos = 0;
    for (int i = 0; i < N; i++) {
        if (hidx[i].idx == 0) { start_pos = i; break; }
    }
    
    // Try forward and backward, pick shorter
    double fwd_len = 0, bwd_len = 0;
    for (int i = 0; i < N; i++) {
        fwd_len += dist2(hidx[(start_pos + i) % N].idx, hidx[(start_pos + i + 1) % N].idx);
        bwd_len += dist2(hidx[(start_pos - i + N) % N].idx, hidx[(start_pos - i - 1 + N) % N].idx);
    }
    
    if (fwd_len <= bwd_len) {
        for (int i = 0; i < N; i++) tour[i] = hidx[(start_pos + i) % N].idx;
    } else {
        for (int i = 0; i < N; i++) tour[i] = hidx[(start_pos - i + N) % N].idx;
    }
}

// ============ Nearest Neighbor ============
void nn_tour(vector<int>& tour) {
    vector<char> visited(N, 0);
    tour[0] = 0;
    visited[0] = 1;
    int cur = 0;
    for (int step = 1; step < N; step++) {
        double best_d = 1e18;
        int best_next = -1;
        for (int i = 0; i < N; i++) {
            if (!visited[i]) {
                double d = dist2(cur, i);
                if (d < best_d) { best_d = d; best_next = i; }
            }
        }
        tour[step] = best_next;
        visited[best_next] = 1;
        cur = best_next;
    }
}

// ============ Full 2-opt ============
void two_opt(vector<int>& tour) {
    vector<char> dont_look(N, 0);
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N; i++) {
            int ci = tour[i];
            if (dont_look[ci]) continue;
            bool local_improved = false;
            int ci1 = tour[(i + 1) % N];
            for (int j = i + 2; j < N; j++) {
                if (i == 0 && j == N - 1) continue;
                int cj = tour[j];
                int cj1 = tour[(j + 1) % N];
                double d_old = dist2(ci, ci1) + dist2(cj, cj1);
                double d_new = dist2(ci, cj) + dist2(ci1, cj1);
                if (d_new < d_old - 1e-10) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    dont_look[ci] = 0;
                    dont_look[cj] = 0;
                    dont_look[tour[(i + 1) % N]] = 0;
                    dont_look[cj1] = 0;
                    improved = true;
                    local_improved = true;
                    break;
                }
            }
            if (!local_improved) dont_look[ci] = 1;
        }
    }
}

                        improved = true;
                        local_imp = true;
                        break;
                    }
                }
            }
        }
    }
```
```cpp
}

// Brute Force
void brute_force(vector<int>& tour) {
    vector<int> perm(N - 1);
    iota(perm.begin(), perm.end(), 1);
    double best = 1e18;
    vector<int> best_perm = perm;
    do {
        double d = sqrt(dist2(0, perm[0]));
        for (int i = 1; i < N - 1; i++)
            d += sqrt(dist2(perm[i - 1], perm[i]));
        d += sqrt(dist2(perm[N - 2], 0));
        if (d < best) { best = d; best_perm = perm; }
    } while (next_permutation(perm.begin(), perm.end()));
    tour[0] = 0;
    for (int i = 0; i < N - 1; i++) tour[i + 1] = best_perm[i];
}

// Hilbert Curve
static uint64_t hilbert_xy2d(int order, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (uint64_t s = (1ULL << (order - 1)); s > 0; s >>= 1) {
        uint64_t rx = (x & s) ? 1 : 0;
        uint64_t ry = (y & s) ? 1 : 0;
        d += s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            uint64_t t = x; x = y; y = t;
        }
    }
    return d;
}

void hilbert_tour(vector<int>& tour) {
    double minx = 1e18, miny = 1e18, maxx = -1e18, maxy = -1e18;
    for (int i = 0; i < N; i++) {
        if (cities[i].x < minx) minx = cities[i].x;
        if (cities[i].y < miny) miny = cities[i].y;
        if (cities[i].x > maxx) maxx = cities[i].x;
        if (cities[i].y > maxy) maxy = cities[i].y;
    }
    double range = maxx - minx;
    if (maxy - miny > range) range = maxy - miny;
    if (range < 1e-10) range = 1.0;
    
    int order = 1;
    while ((1ULL << order) < (uint64_t)N * 2) order++;
    if (order > 31) order = 31;
    
    double scale = (double)((1ULL << order) - 1) / range;
    uint64_t maxc = (1ULL << order) - 1;
    
    struct HI { uint64_t h; int idx; };
    vector<HI> hidx(N);
    for (int i = 0; i < N; i++) {
        uint64_t ix = (uint64_t)((cities[i].x - minx) * scale);
        uint64_t iy = (uint64_t)((cities[i].y - miny) * scale);
        if (ix > maxc) ix = maxc;
        if (iy > maxc) iy = maxc;
        hidx[i].h = hilbert_xy2d(order, ix, iy);
        hidx[i].idx = i;
    }
    sort(hidx.begin(), hidx.end(), [](const HI& a, const HI& b) {
        return a.h < b.h || (a.h == b.h && a.idx < b.idx);
    });
    
    int start_pos = 0;
    for (int i = 0; i < N; i++) {
        if (hidx[i].idx == 0) { start_pos = i; break; }
    }
    
    // Try both directions
    double fwd = 0, bwd = 0;
    for (int i = 0; i < N; i++) {
        fwd += dist2(hidx[(start_pos + i) % N].idx, hidx[(start_pos + i + 1) % N].idx);
        bwd += dist2(hidx[(start_pos - i + N) % N].idx, hidx[(start_pos - i - 1 + N) % N].idx);
    }
    
    if (fwd <= bwd) {
        for (int i = 0; i < N; i++) tour[i] = hidx[(start_pos + i) % N].idx;
    } else {
        for (int i = 0; i < N; i++) tour[i] = hidx[(start_pos - i + N) % N].idx;
    }
}

// Nearest Neighbor
void nn_tour(vector<int>& tour) {
    vector<char> visited(N, 0);
    tour[0] = 0;
    visited[0] = 1;
    int cur = 0;
    for (int step = 1; step < N; step++) {
        double best_d = 1e18;
        int best_next = -1;
        for (int i = 0; i < N; i++) {
            if (!visited[i]) {
                double d = dist2(cur, i);
                if (d < best_d) { best_d = d; best_next = i; }
            }
        }
        tour[step] = best_next;
        visited[best_next] = 1;
        cur = best_next;
    }
}

// Full 2-opt
void two_opt(vector<int>& tour) {
    vector<char> dont_look(N, 0);
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N; i++) {
            int ci = tour[i];
            if (dont_look[ci]) continue;
            bool local_improved = false;
            int ci1 = tour[(i + 1) % N];
            for (int j = i + 2; j < N; j++) {
                if (i == 0 && j == N - 1) continue;
                int cj = tour[j];
                int cj1 = tour[(j + 1) % N];
                double d_old = dist2(ci, ci1) + dist2(cj, cj1);
                double d_new = dist2(ci, cj) + dist2(ci1, cj1);
                if (d_new < d_old - 1e-10) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    dont_look[ci] = 0;
                    dont_look[cj] = 0;
                    dont_look[tour[(i + 1) % N]] = 0;
                    dont_look[cj1] = 0;
                    improved = true;
                    local_improved = true;
                    break;
                }
            }
            if (!local_improved) dont_look[ci] = 1;
        }
    }
}

// Limited 2-opt
void limited_two_opt(vector<int>& tour, int K) {
    bool improved = true;
    int passes = 0;
    while (improved && passes < 20) {
        improved = false;
        passes++;
        for (int i = 0; i < N; i++) {
            bool local_imp = true;
            int tries = 0;
            while (local_imp && tries < 5) {
                local_imp = false;
                tries++;
                int ci = tour[i], ci1 = tour[(i + 1) % N];
                for (int dj = 2; dj <= K && i + dj < N; dj++) {
                    int j = i + dj;
                    if (i == 0 && j == N - 1) continue;
                    int cj = tour[j], cj1 = tour[(j + 1) % N];
                    double d_old = dist2(ci, ci1) + dist2(cj, cj1);

                        reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        improved = true;
                        local_imp = true;
                        break;
                    }
                }
            }
        }
    }
}
```

```cpp
void brute_force(vector<int>& tour) {

// ---- Hilbert Curve Sort ----
static uint64_t hilbert_xy2d(int order, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (uint64_t s = (1ULL << (order - 1)); s > 0; s >>= 1) {
        uint64_t rx = (x & s) ? 1 : 0;
        uint64_t ry = (y & s) ? 1 : 0;
        d += s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            uint64_t t = x; x = y; y = t;
        }
    }
    return d;
}

void hilbert_tour(vector<int>& tour) {
    double minx = 1e18, miny = 1e18, maxx = -1e18, maxy = -1e18;
    for (int i = 0; i < N; i++) {
        if (cities[i].x < minx) minx = cities[i].x;
        if (cities[i].y < miny) miny = cities[i].y;
        if (cities[i].x > maxx) maxx = cities[i].x;
        if (cities[i].y > maxy) maxy = cities[i].y;
    }
    double range = maxx - minx;
    if (maxy - miny > range) range = maxy - miny;
    if (range < 1e-10) range = 1.0;
    
    int order = 1;
    while ((1ULL << order) < (uint64_t)N * 2) order++;
    if (order > 31) order = 31;
    
    double scale = (double)((1ULL << order) - 1) / range;
    uint64_t maxc = (1ULL << order) - 1;
    
    struct HI { uint64_t h; int idx; };
    vector<HI> hidx(N);
    for (int i = 0; i < N; i++) {
        uint64_t ix = (uint64_t)((cities[i].x - minx) * scale);
        uint64_t iy = (uint64_t)((cities[i].y - miny) * scale);
        if (ix > maxc) ix = maxc;
        if (iy > maxc) iy = maxc;
        hidx[i].h = hilbert_xy2d(order, ix, iy);
        hidx[i].idx = i;
    }
    sort(hidx.begin(), hidx.end(), [](const HI& a, const HI& b) {
        return a.h < b.h || (a.h == b.h && a.idx < b.idx);
    });
    
    int start_pos = 0;
    for (int i = 0; i < N; i++) {
        if (hidx[i].idx == 0) { start_pos = i; break; }
    }
    
    // Try both directions, pick shorter total squared distance
    double fwd = 0, bwd = 0;
    for (int i = 0; i < N; i++) {
        fwd += dist2(hidx[(start_pos + i) % N].idx, hidx[(start_pos + i + 1) % N].idx);
        bwd += dist2(hidx[(start_pos - i + N) % N].idx, hidx[(start_pos - i - 1 + N) % N].idx);
    }
    
    if (fwd <= bwd) {
        for (int i = 0; i < N; i++) tour[i] = hidx[(start_pos + i) % N].idx;
    } else {
        for (int i = 0; i < N; i++) tour[i] = hidx[(start_pos - i + N) % N].idx;
    }
}

// ---- Nearest Neighbor ----
void nn_tour(vector<int>& tour) {
    vector<char> visited(N, 0);
    tour[0] = 0;
    visited[0] = 1;
    int cur = 0;
    for (int step = 1; step < N; step++) {
        double best_d = 1e18;
        int best_next = -1;
        for (int i = 0; i < N; i++) {
            if (!visited[i]) {
                double d = dist2(cur, i);
                if (d < best_d) { best_d = d; best_next = i; }
            }
        }
        tour[step] = best_next;
        visited[best_next] = 1;
        cur = best_next;
    }
}

// ---- Full 2-opt ----
void two_opt(vector<int>& tour) {
    vector<char> dont_look(N, 0);
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N; i++) {
            int ci = tour[i];
            if (dont_look[ci]) continue;
            bool local_improved = false;
            int ci1 = tour[(i + 1) % N];
            for (int j = i + 2; j < N; j++) {
                if (i == 0 && j == N - 1) continue;
                int cj = tour[j];
                int cj1 = tour[(j + 1) % N];
                double d_old = dist2(ci, ci1) + dist2(cj, cj1);
                double d_new = dist2(ci, cj) + dist2(ci1, cj1);
                if (d_new < d_old - 1e-10) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    dont_look[ci] = 0;
                    dont_look[cj] = 0;
                    dont_look[tour[(i + 1) % N]] = 0;
                    dont_look[cj1] = 0;
                    improved = true;
                    local_improved = true;
                    break;
                }
            }
            if (!local_improved) dont_look[ci] = 1;
        }
    }
}

// ---- Limited 2-opt ----
void limited_two_opt(vector<int>& tour, int K, int max_passes) {
    bool improved = true;
    int passes = 0;
    while (improved && passes < max_passes) {
        improved = false;
        passes++;
        for (int i = 0; i < N; i++) {
            bool local_imp = true;
            int tries = 0;
            while (local_imp && tries < 3) {
                local_imp = false;
                tries++;
                int ci = tour[i], ci1 = tour[(i + 1) % N];
                for (int dj = 2; dj <= K && i + dj < N; dj++) {
                    int j = i + dj;
                    if (i == 0 && j == N - 1) continue;
                    int cj = tour[j], cj1 = tour[(j + 1) % N];
                    double d_old = dist2(ci, ci1) + dist2(cj, cj1);

                        reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        improved = true;
                        local_imp = true;
                        break;
                    }
                }
            }
        }
    }
}
```
