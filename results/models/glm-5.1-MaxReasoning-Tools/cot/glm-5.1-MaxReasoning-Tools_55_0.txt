```cpp
Bitboard bishop_attacks(int sq, Bitboard occ) {
    Bitboard attacks = 0;
    int r = rank_of(sq), f = file_of(sq);
    int dirs[4][2] = {{1,1},{1,-1},{-1,1},{-1,-1}};
    for (auto& d : dirs) {
        int nr = r + d[0], nf = f + d[1];
        while (nr >= 0 && nr < 8 && nf >= 0 && nf < 8) {
            int s = sq(nr, nf);
            attacks |= bit(s);
            if (occ & bit(s)) break;
            nr += d[0]; nf += d[1];
        }
    }
    return attacks;
}

Bitboard rook_attacks(int sq, Bitboard occ) {
    Bitboard attacks = 0;
    int r = rank_of(sq), f = file_of(sq);
    int dirs[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for (auto& d : dirs) {
        int nr = r + d[0], nf = f + d[1];
        while (nr >= 0 && nr < 8 && nf >= 0 && nf < 8) {
            int s = sq(nr, nf);
            attacks |= bit(s);
            if (occ & bit(s)) break;
            nr += d[0]; nf += d[1];
        }
    }
    return attacks;
}

Bitboard queen_attacks(int sq, Bitboard occ) {
```

```cpp
        return state * 2685821657736338717ULL;
    };
    
    for (int c = 0; c < 2; c++)
        for (int p = 0; p < 6; p++)
            for (int s = 0; s < 64; s++)
                zobrist_pieces[c][p][s] = rng();

        zobrist_castle[i] = rng();
    
    for (int s = 0; s < 64; s++)
        zobrist_ep[s] = rng();
    
    zobrist_side = rng();
}
```

```cpp
void parse_fen(Position& pos, const string& fen) {
    pos.clear();
    istringstream ss(fen);
    
    // Board part
    string board_str;
    ss >> board_str;
    
    int sq = 56; // a8
    for (char c : board_str) {
        if (c == '/') {
            sq -= 16; // go to start of next rank down
        } else if (c >= '1' && c <= '8') {
            sq += (c - '0');
        } else {
            int color = (c >= 'a' && c <= 'z') ? BLACK : WHITE;
            char lower_c = (c >= 'A' && c <= 'Z') ? (c + 32) : c;
            int piece;
            switch (lower_c) {
                case 'p': piece = PAWN; break;
                case 'n': piece = KNIGHT; break;
                case 'b': piece = BISHOP; break;
                case 'r': piece = ROOK; break;
                case 'q': piece = QUEEN; break;
                case 'k': piece = KING; break;
                default: piece = NO_PIECE; break;
            }
            if (piece != NO_PIECE)
                pos.place_piece(sq, piece, color);
            sq++;
        }
    }
    
    // Side to move
    string side_str;
    ss >> side_str;
    pos.side = (side_str == "w") ? WHITE : BLACK;
    
    // Castling
    string castle_str;
    ss >> castle_str;
    pos.castle = 0;
    for (char c : castle_str) {
        switch (c) {
            case 'K': pos.castle |= WK; break;
            case 'Q': pos.castle |= WQ; break;
            case 'k': pos.castle |= BK; break;
            case 'q': pos.castle |= BQ; break;
        }
    }
    
    // En passant
    string ep_str;
    ss >> ep_str;
    if (ep_str != "-") {
        int f = ep_str[0] - 'a';
        int r = ep_str[1] - '1';
        pos.ep = make_sq(r, f);
    } else {
        pos.ep = -1;
    }
    
    // Halfmove and fullmove
    ss >> pos.halfmove >> pos.fullmove;
    
    // Compute hash
    pos.hash = 0;
    for (int s = 0; s < 64; s++) {
        if (pos.board[s] != NO_PIECE)
            pos.hash ^= zobrist_pieces[pos.board_color[s]][pos.board[s]][s];
    }
    pos.hash ^= zobrist_castle[pos.castle];
    if (pos.ep >= 0) pos.hash ^= zobrist_ep[pos.ep];
    if (pos.side == BLACK) pos.hash ^= zobrist_side;
}
```

```cpp
// Generate all pseudo-legal moves for the current position
// A move is pseudo-legal if it follows the basic movement rules,
// but might leave the king in check.

void gen_pseudo_legal_moves(const Position& pos, vector<Move32>& moves) {
    int side = pos.side;
    int opp = 1 - side;
    U64 own = pos.color_occ[side];
    U64 enemy = pos.color_occ[opp];
    U64 occ = pos.all_occ;
    
    // Pawns
    gen_pawn_moves(pos, side, own, enemy, occ, moves);
    
    // Knights
    U64 knights = pos.pieces[side][KNIGHT];
    while (knights) {
        int from = lsb(knights);
        knights &= knights - 1;
        U64 atk = knight_attacks[from] & ~own;
        while (atk) {
            int to = lsb(atk);
            atk &= atk - 1;
            moves.push_back(pack_move(from, to, 0, 0));
        }
    }
    
    // Bishops
    U64 bishops = pos.pieces[side][BISHOP];
    while (bishops) {
        int from = lsb(bishops);
        bishops &= bishops - 1;
        U64 atk = bishop_attacks(from, occ) & ~own;
        while (atk) {
            int to = lsb(atk);
            atk &= atk - 1;
            moves.push_back(pack_move(from, to, 0, 0));
        }
    }
    
    // Rooks
    U64 rooks = pos.pieces[side][ROOK];
    while (rooks) {
        int from = lsb(rooks);
        rooks &= rooks - 1;
        U64 atk = rook_attacks(from, occ) & ~own;
        while (atk) {
            int to = lsb(atk);
            atk &= atk - 1;
            moves.push_back(pack_move(from, to, 0, 0));
        }
    }
    
    // Queens
    U64 queens = pos.pieces[side][QUEEN];
    while (queens) {
        int from = lsb(queens);
        queens &= queens - 1;
        U64 atk = queen_attacks(from, occ) & ~own;
        while (atk) {
            int to = lsb(atk);
            atk &= atk - 1;
            moves.push_back(pack_move(from, to, 0, 0));
        }
    }
    
    // King
    int king_sq = lsb(pos.pieces[side][KING]);
    U64 atk = king_attacks[king_sq] & ~own;
    while (atk) {
        int to = lsb(atk);
        atk &= atk - 1;
        moves.push_back(pack_move(king_sq, to, 0, 0));
    }
    
    // Castling
    gen_castling_moves(pos, side, occ, enemy, moves);
}
```

```cpp
void gen_pawn_moves(const Position& pos, int side, U64 own, U64 enemy, U64 occ, vector<Move32>& moves) {
    U64 pawns = pos.pieces[side][PAWN];
    int push_dir = (side == WHITE) ? 8 : -8;
    int start_rank = (side == WHITE) ? 1 : 6;
    int promo_rank = (side == WHITE) ? 7 : 0;
    
    while (pawns) {
        int from = lsb(pawns);
        pawns &= pawns - 1;
        int r = rank_of(from);
        
        // Single push
        int to = from + push_dir;
        if (to >= 0 && to < 64 && !(occ & bit(to))) {
            if (rank_of(to) == promo_rank) {
                // Promotion
                for (int p = 1; p <= 4; p++) // N, B, R, Q
                    moves.push_back(pack_move(from, to, p, 3));
            } else {
                moves.push_back(pack_move(from, to, 0, 0));
            }
            
            // Double push
            if (r == start_rank) {
                int to2 = from + 2 * push_dir;
                if (!(occ & bit(to2))) {
                    moves.push_back(pack_move(from, to2, 0, 0));
                }
            }
        }
        
        // Captures
        U64 captures = pawn_attacks[side][from] & enemy;
        while (captures) {
            int to = lsb(captures);
            captures &= captures - 1;
            if (rank_of(to) == promo_rank) {
                for (int p = 1; p <= 4; p++)
                    moves.push_back(pack_move(from, to, p, 3));
            } else {
                moves.push_back(pack_move(from, to, 0, 0));
            }
        }
        
        // En passant
        if (pos.ep >= 0 && (pawn_attacks[side][from] & bit(pos.ep))) {
            moves.push_back(pack_move(from, pos.ep, 0, 1));
        }
    }
}
```

```cpp
void gen_castling_moves(const Position& pos, int side, U64 occ, U64 enemy, vector<Move32>& moves) {
    // Check if king is in check - can't castle out of check
    int king_sq = lsb(pos.pieces[side][KING]);
    if (is_attacked(pos, king_sq, 1 - side)) return;
    
    if (side == WHITE) {
        // Kingside
        if ((pos.castle & WK) && !(occ & (bit(5) | bit(6)))) {
            if (!is_attacked(pos, 5, BLACK) && !is_attacked(pos, 6, BLACK)) {
                moves.push_back(pack_move(4, 6, 0, 2));
            }
        }
        // Queenside
        if ((pos.castle & WQ) && !(occ & (bit(1) | bit(2) | bit(3)))) {
            if (!is_attacked(pos, 2, BLACK) && !is_attacked(pos, 3, BLACK)) {
                moves.push_back(pack_move(4, 2, 0, 2));
            }
        }
    } else {
        // Kingside
        if ((pos.castle & BK) && !(occ & (bit(61) | bit(62)))) {
            if (!is_attacked(pos, 61, WHITE) && !is_attacked(pos, 62, WHITE)) {
                moves.push_back(pack_move(60, 62, 0, 2));

            }
        }
    }
}
```

```cpp
bool is_attacked(const Position& pos, int sq, int by_color) {
    U64 occ = pos.all_occ;
    
    // Pawn attacks
    if (pawn_attacks[1 - by_color][sq] & pos.pieces[by_color][PAWN]) return true;
    
    // Knight attacks
    if (knight_attacks[sq] & pos.pieces[by_color][KNIGHT]) return true;
    
    // Bishop/Queen diagonal attacks
    U64 diag = bishop_attacks(sq, occ);
    if (diag & (pos.pieces[by_color][BISHOP] | pos.pieces[by_color][QUEEN])) return true;
    
    // Rook/Queen orthogonal attacks
    U64 orth = rook_attacks(sq, occ);
    if (orth & (pos.pieces[by_color][ROOK] | pos.pieces[by_color][QUEEN])) return true;
    
    // King attacks
    if (king_attacks[sq] & pos.pieces[by_color][KING]) return true;
    
    return false;
}
```
