```rust
fn greedy(adj: &[Vec<usize>], n: usize, rng: &mut impl Rng) -> Vec<bool> {
    let mut in_set = vec![false; n];
    let mut dom_count = vec![0usize; n];
    
    let mut uncovered = n;
    
    while uncovered > 0 {
        // Find vertex with maximum gain
        let mut best_v = 0;
        let mut best_gain = 0;
        
        for v in 0..n {
            if in_set[v] { continue; }
            let mut gain = 0;
            if dom_count[v] == 0 { gain += 1; }
            for &u in &adj[v] {
                if dom_count[u] == 0 { gain += 1; }
            }
            if gain > best_gain || (gain == best_gain && /* random tie-break */) {
                best_gain = gain;
                best_v = v;
            }
        }
        
        if best_gain == 0 { break; }
        
        // Add best_v to set
        in_set[best_v] = true;
        if dom_count[best_v] == 0 { uncovered -= 1; }
        dom_count[best_v] += 1;
        for &u in &adj[best_v] {
            if dom_count[u] == 0 { uncovered -= 1; }
            dom_count[u] += 1;
        }
    }
    
    in_set
}
```

```rust
// Simple xorshift64 random number generator
struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Rng { state: if seed == 0 { 1 } else { seed } }
    }
    
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    
    fn next_usize(&mut self, n: usize) -> usize {
        (self.next_u64() % n as u64) as usize
    }
    
    fn next_f64(&mut self) -> f64 {
        (self.next_u64() as f64) / (u64::MAX as f64)
```

```rust
fn grasp_construct(adj: &[Vec<usize>], n: usize, rng: &mut Rng, alpha: f64) -> (Vec<bool>, Vec<usize>) {

    let mut uncovered = n;

        let mut max_gain = 0;
        let mut min_gain = usize::MAX;
        let mut gains: Vec<(usize, usize)> = Vec::new(); // (gain, vertex)
        
        for v in 0..n {
            if in_set[v] { continue; }
            let mut gain = 0;
            if dom_count[v] == 0 { gain += 1; }
            for &u in &adj[v] {
                if dom_count[u] == 0 { gain += 1; }
            }
            if gain > 0 {
                gains.push((gain, v));
                if gain > max_gain { max_gain = gain; }
                if gain < min_gain { min_gain = gain; }
            }
        }
        
        if max_gain == 0 { break; }
        
        // Build restricted candidate list
        let threshold = if max_gain == min_gain {
            max_gain
        } else {
            (max_gain as f64 - alpha * (max_gain - min_gain) as f64).ceil() as usize
        };
        
        let rcl: Vec<usize> = gains.iter()
            .filter(|&&(g, _)| g >= threshold)
            .map(|&(_, v)| v)
            .collect();
        
        // Pick randomly from RCL
        let v = rcl[rng.next_usize(rcl.len())];
        
        // Find the gain for this vertex
        let gain = gains.iter().find(|&&(_, vertex)| vertex == v).unwrap().0;
        
        in_set[v] = true;
        if dom_count[v] == 0 { uncovered -= 1; }
        dom_count[v] += 1;
        for &u in &adj[v] {
            if dom_count[u] == 0 { uncovered -= 1; }
            dom_count[u] += 1;
        }
    }
    
    (in_set, dom_count)
}
```

```rust
        let mut min_gain = usize::MAX;

            for &u in &adj[v] {
                if dom_count[u] == 0 { gain += 1; }

                min_gain = min_gain.min(gain);

    for &u in &adj[v] { dom_count[u] -= 1; }
}

fn add_vertex(adj: &[Vec<usize>], v: usize, in_set: &mut Vec<bool>, dom_count: &mut Vec<usize>) {

    for &u in &adj[v] { dom_count[u] += 1; }

fn greedy_fill(adj: &[Vec<usize>], n: usize, in_set: &mut Vec<bool>, dom_count: &mut Vec<usize>, rng: &mut Rng, alpha: f64) {
    let mut uncovered = (0..n).filter(|&v| dom_count[v] == 0).count();
    
    while uncovered > 0 {
        let mut best_v = 0;
        let mut best_gain = 0;
        let mut candidates: Vec<usize> = vec![];
        
        for v in 0..n {
            if in_set[v] { continue; }
            let mut gain = 0;
            if dom_count[v] == 0 { gain += 1; }
            for &u in &adj[v] {
                if dom_count[u] == 0 { gain += 1; }
            }
            if gain > best_gain {
                best_gain = gain;
                candidates.clear();
                candidates.push(v);
            } else if gain == best_gain && gain > 0 {
                candidates.push(v);
            }
        }
        
        if best_gain == 0 { break; }
        
        // Apply GRASP-style selection
        if alpha > 0.0 && candidates.len() > 1 {
            // Also consider near-best candidates
            let threshold = ((best_gain as f64) * (1.0 - alpha)).ceil() as usize;
            let mut rcl: Vec<usize> = vec![];
            for v in 0..n {
                if in_set[v] { continue; }
                let mut gain = 0;
                if dom_count[v] == 0 { gain += 1; }
                for &u in &adj[v] {
                    if dom_count[u] == 0 { gain += 1; }
                }
                if gain >= threshold && gain > 0 {
                    rcl.push(v);
                }
            }
            if !rcl.is_empty() {
                let v = rcl[rng.usize(rcl.len())];
                add_vertex(adj, v, in_set, dom_count);
                uncovered -= best_gain; // approximate
                continue;
            }
        }
        
        let v = candidates[rng.usize(candidates.len())];
        add_vertex(adj, v, in_set, dom_count);
        // Recount uncovered properly
        uncovered = (0..n).filter(|&v| dom_count[v] == 0).count();
    }
}

fn swap_improve(adj: &[Vec<usize>], n: usize, in_set: &mut Vec<bool>, dom_count: &mut Vec<usize>, best_size: &mut usize, deadline: Instant) {
```
