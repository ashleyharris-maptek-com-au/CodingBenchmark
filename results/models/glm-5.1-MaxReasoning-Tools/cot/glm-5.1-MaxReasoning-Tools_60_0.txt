```javascript
// Simple max-heap
class MaxHeap {
    constructor(compare) {
        this.heap = [];
        this.compare = compare;
    }
    
    push(val) {
        this.heap.push(val);
        this._bubbleUp(this.heap.length - 1);
    }
    
    pop() {
        if (this.heap.length === 0) return undefined;
        const top = this.heap[0];
        const last = this.heap.pop();
        if (this.heap.length > 0) {
            this.heap[0] = last;
            this._sinkDown(0);
        }
        return top;
    }
    
    get size() { return this.heap.length; }
    
    _bubbleUp(i) {
        while (i > 0) {
            const parent = (i - 1) >> 1;
            if (this.compare(this.heap[i], this.heap[parent]) > 0) {
                [this.heap[i], this.heap[parent]] = [this.heap[parent], this.heap[i]];
                i = parent;
            } else break;
        }
    }
    
    _sinkDown(i) {
        const n = this.heap.length;
        while (true) {
            let largest = i;
            const left = 2 * i + 1;
            const right = 2 * i + 2;
            if (left < n && this.compare(this.heap[left], this.heap[largest]) > 0) largest = left;
            if (right < n && this.compare(this.heap[right], this.heap[largest]) > 0) largest = right;
            if (largest !== i) {
                [this.heap[i], this.heap[largest]] = [this.heap[largest], this.heap[i]];
                i = largest;
            } else break;
        }
    }
}
```

```javascript
function findEarliestGap(intervals, earliestStart, duration) {
    // intervals is sorted by start time, non-overlapping
    // Find the earliest time >= earliestStart where we can fit a task of given duration
    
    let t = earliestStart;
    for (const iv of intervals) {
        if (t + duration <= iv.start) {
            return t; // Fits before this interval
        }
        if (iv.end > t) {
            t = iv.end; // Move past this interval
        }
    }
    return t; // Fits after all intervals
}

function insertInterval(intervals, start, end) {
    // Insert and merge overlapping intervals
    intervals.push({start, end});
    intervals.sort((a, b) => a.start - b.start);
    // Merge overlapping
    const merged = [];
    for (const iv of intervals) {
        if (merged.length > 0 && iv.start <= merged[merged.length-1].end) {
            merged[merged.length-1].end = Math.max(merged[merged.length-1].end, iv.end);
        } else {
            merged.push({...iv});
        }
    }
    intervals.length = 0;
    intervals.push(...merged);
}
```

Actually, a better approach: precompute all priorities (they don't change during scheduling), and use a max-heap based on priority. When a task becomes ready, push it onto the heap. When we need the next task, pop from the heap.

```javascript
function listSchedule(prioMap) {
    const remDeps = new Map();
    for (const t of tasks) remDeps.set(t.id, (t.deps || []).length);
    
    // Max-heap: tasks with higher priority come first
    // Heap elements: [priority, id]
    const heap = [];
    
    function heapPush(val) {
        heap.push(val);
        let i = heap.length - 1;
        while (i > 0) {
            const parent = (i - 1) >> 1;
            if (heap[i][0] > heap[parent][0]) {
                [heap[i], heap[parent]] = [heap[parent], heap[i]];
                i = parent;
            } else break;
        }
    }
    
    function heapPop() {
        const top = heap[0];
        const last = heap.pop();
        if (heap.length > 0) {
            heap[0] = last;
            let i = 0;
            while (true) {
                let largest = i;
                const l = 2*i+1, r = 2*i+2;
                if (l < heap.length && heap[l][0] > heap[largest][0]) largest = l;
                if (r < heap.length && heap[r][0] > heap[largest][0]) largest = r;
                if (largest !== i) {
                    [heap[i], heap[largest]] = [heap[largest], heap[i]];
                    i = largest;
                } else break;
            }
        }
        return top;
    }
    
    for (const t of tasks) {
        if ((t.deps || []).length === 0) {
            heapPush([prioMap.get(t.id), t.id]);
        }
    }
    
    const fin = new Map();
    const machFree = new Array(machineCount).fill(0);
    const result = [];
    
    while (heap.length > 0) {
        const [p, id] = heapPop();
        const t = taskMap.get(id);
        
        let depEnd = 0;
        for (const d of (t.deps || [])) depEnd = Math.max(depEnd, fin.get(d));
        
        let bestM = 0, bestS = Infinity;
        for (let m = 0; m < machineCount; m++) {
            const s = Math.max(depEnd, machFree[m]);
            if (s < bestS || (s === bestS && m < bestM)) {
                bestS = s;
                bestM = m;
            }
        }
        
        machFree[bestM] = bestS + t.duration;
        fin.set(id, bestS + t.duration);
        result.push({ id, start: bestS, machine: bestM });
        
        for (const c of (succ.get(id) || [])) {
            remDeps.set(c, remDeps.get(c) - 1);
            if (remDeps.get(c) === 0) {
                heapPush([prioMap.get(c), c]);
            }
        }
    }
    
    return result;
}
```
