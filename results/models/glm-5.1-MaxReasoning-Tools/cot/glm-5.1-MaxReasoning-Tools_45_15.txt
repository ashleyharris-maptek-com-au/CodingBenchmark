```python
import struct

class SpirvBuilder:
    def __init__(self):
        self.words = []
        self.next_id = 1
        self.bound_idx = 0
        
    def alloc_id(self):
        id = self.next_id
        self.next_id += 1
        return id
    
    def emit(self, *args):
        for a in args:
            self.words.append(a)
    
    def emit_header(self):
        self.words.append(0x07230203)  # Magic
        self.words.append(0x00010000)  # Version 1.0
        self.words.append(0x00000000)  # Generator
        self.bound_idx = len(self.words)
        self.words.append(0)  # Placeholder for bound
        self.words.append(0x00000000)  # Reserved
    
    def inst(self, word_count, opcode, *operands):
        self.emit((word_count << 16) | opcode, *operands)
    
    def emit_string_words(self, s):
        encoded = s.encode('utf-8') + b'\0'
        while len(encoded) % 4 != 0:
            encoded += b'\0'
        for i in range(0, len(encoded), 4):
            word = struct.unpack('<I', encoded[i:i+4])[0]
            self.words.append(word)
    
    def finalize(self):
        self.words[self.bound_idx] = self.next_id
        # Convert to hex
        result = b''
        for w in self.words:
            result += struct.pack('<I', w)
        return result.hex()
```
