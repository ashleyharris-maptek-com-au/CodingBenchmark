%ptr_in_arr_v3f = OpTypePointer Input %arr_v3f_4
%ptr_in_v3f = OpTypePointer Input %v3f

%gl_PerVertex_out = OpTypeStruct %v4f
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4f = OpTypePointer Output %v4f
%ptr_out_v3f = OpTypePointer Output %v3f

; UBO type
%UBO = OpTypeStruct %mat4v4f %mat4v4f %mat4v4f %v3f %v3f %v4f
%ptr_ubo = OpTypePointer Uniform %UBO

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%inWorldPos = OpVariable %ptr_in_arr_v3f Input
%inNormal = OpVariable %ptr_in_arr_v3f Input
%inColor = OpVariable %ptr_in_arr_v3f Input
%outColor = OpVariable %ptr_out_v3f Output
%uboVar = OpVariable %ptr_ubo Uniform

; Main function
%main = OpFunction %void None %fn_void
%lbl = OpLabel

; ---- Load positions ----
%p0ptr = OpAccessChain %ptr_in_v4f %gl_in %int_0 %int_0
%p1ptr = OpAccessChain %ptr_in_v4f %gl_in %int_1 %int_0
%p2ptr = OpAccessChain %ptr_in_v4f %gl_in %int_2 %int_0
%p3ptr = OpAccessChain %ptr_in_v4f %gl_in %int_3 %int_0
%p0 = OpLoad %v4f %p0ptr
%p1 = OpLoad %v4f %p1ptr
%p2 = OpLoad %v4f %p2ptr
%p3 = OpLoad %v4f %p3ptr

; ---- Load colors ----
%c1ptr = OpAccessChain %ptr_in_v3f %inColor %int_1
%c2ptr = OpAccessChain %ptr_in_v3f %inColor %int_2
%c1 = OpLoad %v3f %c1ptr
%c2 = OpLoad %v3f %c2ptr

; ---- Compute 2D adjacency tangents ----
%p0xy = OpVectorShuffle %v2f %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2f %p1 %p1 0 1
%p2xy = OpVectorShuffle %v2f %p2 %p2 0 1
%p3xy = OpVectorShuffle %v2f %p3 %p3 0 1

; tangent1 = normalize(p2xy - p0xy)
%dt1 = OpFSub %v2f %p2xy %p0xy
%dt1sq = OpDot %float %dt1 %dt1
%dt1len = OpExtInst %float %1 Sqrt %dt1sq
%dt1inv = OpFDiv %float %float_1 %dt1len
%tan1 = OpVectorTimesScalar %v2f %dt1 %dt1inv

; tangent2 = normalize(p3xy - p1xy)
%dt2 = OpFSub %v2f %p3xy %p1xy
%dt2sq = OpDot %float %dt2 %dt2
%dt2len = OpExtInst %float %1 Sqrt %dt2sq
%dt2inv = OpFDiv %float %float_1 %dt2len
%tan2 = OpVectorTimesScalar %v2f %dt2 %dt2inv

; ---- Compute perpendiculars at start and end ----
%tan1x = OpCompositeExtract %float %tan1 0
%tan1y = OpCompositeExtract %float %tan1 1
%nTan1y = OpFNegate %float %tan1y
%perp1 = OpCompositeConstruct %v2f %nTan1y %tan1x
%perp1s = OpVectorTimesScalar %v2f %perp1 %float_0_01

%tan2x = OpCompositeExtract %float %tan2 0
%tan2y = OpCompositeExtract %float %tan2 1
%nTan2y = OpFNegate %float %tan2y
%perp2 = OpCompositeConstruct %v2f %nTan2y %tan2x
%perp2s = OpVectorTimesScalar %v2f %perp2 %float_0_01

; ---- Compute interpolated tangents ----
; tangent13 = normalize(tan1 * (2/3) + tan2 * (1/3))
%tan1s23 = OpVectorTimesScalar %v2f %tan1 %float_2_3
%tan2s13 = OpVectorTimesScalar %v2f %tan2 %float_1_3
%mt13 = OpFAdd %v2f %tan1s23 %tan2s13
%mt13sq = OpDot %float %mt13 %mt13
%mt13len = OpExtInst %float %1 Sqrt %mt13sq
%mt13inv = OpFDiv %float %float_1 %mt13len
%tan13 = OpVectorTimesScalar %v2f %mt13 %mt13inv

; tangent23 = normalize(tan1 * (1/3) + tan2 * (2/3))
%tan1s13 = OpVectorTimesScalar %v2f %tan1 %float_1_3
%tan2s23 = OpVectorTimesScalar %v2f %tan2 %float_2_3
%mt23 = OpFAdd %v2f %tan1s13 %tan2s23
%mt23sq = OpDot %float %mt23 %mt23
%mt23len = OpExtInst %float %1 Sqrt %mt23sq
%mt23inv = OpFDiv %float %float_1 %mt23len
%tan23 = OpVectorTimesScalar %v2f %mt23 %mt23inv

; ---- Compute perpendiculars at 1/3 and 2/3 ----
%tan13x = OpCompositeExtract %float %tan13 0
%tan13y = OpCompositeExtract %float %tan13 1
%nTan13y = OpFNegate %float %tan13y
%perp13 = OpCompositeConstruct %v2f %nTan13y %tan13x
%perp13s = OpVectorTimesScalar %v2f %perp13 %float_0_01

%tan23x = OpCompositeExtract %float %tan23 0
%tan23y = OpCompositeExtract %float %tan23 1
%nTan23y = OpFNegate %float %tan23y
%perp23 = OpCompositeConstruct %v2f %nTan23y %tan23x
%perp23s = OpVectorTimesScalar %v2f %perp23 %float_0_01

; ---- Compute sub-segment positions ----
%seg = OpFSub %v4f %p2 %p1
%seg13 = OpVectorTimesScalar %v4f %seg %float_1_3
%seg23 = OpVectorTimesScalar %v4f %seg %float_2_3
%pos13 = OpFAdd %v4f %p1 %seg13
%pos23 = OpFAdd %v4f %p1 %seg23

; ---- Compute interpolated colors ----
%c1s23 = OpVectorTimesScalar %v3f %c1 %float_2_3
%c2s13 = OpVectorTimesScalar %v3f %c2 %float_1_3
%col13 = OpFAdd %v3f %c1s23 %c2s13

%c1s13 = OpVectorTimesScalar %v3f %c1 %float_1_3
%c2s23 = OpVectorTimesScalar %v3f %c2 %float_2_3
%col23 = OpFAdd %v3f %c1s13 %c2s23

; ---- Output position pointer ----
%outPosPtr = OpAccessChain %ptr_out_v4f %gl_out %int_0
