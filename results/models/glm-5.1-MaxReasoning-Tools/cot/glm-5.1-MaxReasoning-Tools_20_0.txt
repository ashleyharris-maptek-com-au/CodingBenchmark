    let mut cell_medians: HashMap<(i64, i64, i64), Vec<f64>> = HashMap::new();
    let mut cell_mads: HashMap<(i64, i64, i64), Vec<f64>> = HashMap::new();
    let mut cell_counts: HashMap<(i64, i64, i64), usize> = HashMap::new();
    
    // Global stats
    let mut global_medians = vec![0.0f64; num_props];
    let mut global_mads = vec![0.0f64; num_props];
    
    for p in 0..num_props {
        let mut vals: Vec<f64> = props[p].iter().copied().filter(|v| !v.is_nan()).collect();
        if vals.is_empty() { continue; }
        vals.sort_by(|a, b| a.partial_cmp(b).unwrap());
        let med = if vals.len() % 2 == 0 {
            (vals[vals.len()/2 - 1] + vals[vals.len()/2]) / 2.0
        } else {
            vals[vals.len()/2]
        };
        global_medians[p] = med;
        
        let mut devs: Vec<f64> = vals.iter().map(|v| (v - med).abs()).collect();
        devs.sort_by(|a, b| a.partial_cmp(b).unwrap());
        let mad = if devs.len() % 2 == 0 {
            (devs[devs.len()/2 - 1] + devs[devs.len()/2]) / 2.0
        } else {
            devs[devs.len()/2]
        };
        global_mads[p] = mad;
    }
    
    // Compute cell stats
    for (key, indices) in &grid {
        if indices.len() < 3 { continue; }
        let mut medians = vec![0.0f64; num_props];
        let mut mads = vec![0.0f64; num_props];
        
        for p in 0..num_props {
            let mut vals: Vec<f64> = indices.iter()
                .map(|&i| props[p][i])
                .filter(|v| !v.is_nan())
                .collect();
            if vals.len() < 3 { continue; }
            vals.sort_by(|a, b| a.partial_cmp(b).unwrap());
            let med = if vals.len() % 2 == 0 {
                (vals[vals.len()/2 - 1] + vals[vals.len()/2]) / 2.0
            } else {
                vals[vals.len()/2]
            };
            medians[p] = med;
            
            let mut devs: Vec<f64> = vals.iter().map(|v| (v - med).abs()).collect();
            devs.sort_by(|a, b| a.partial_cmp(b).unwrap());
            let mad = if devs.len() % 2 == 0 {
                (devs[devs.len()/2 - 1] + devs[devs.len()/2]) / 2.0
            } else {
                devs[devs.len()/2]
            };
            mads[p] = if mad > 0.0 { mad } else { global_mads[p].max(1e-10) };
        }
        
        cell_medians.insert(*key, medians);
        cell_mads.insert(*key, mads);
        cell_counts.insert(*key, indices.len());
    }
    
    // Compute within-hole statistics
    // For each hole, compute median and MAD of each property
    let mut hole_medians: Vec<Vec<f64>> = Vec::new(); // per hole, per property
    let mut hole_mads: Vec<Vec<f64>> = Vec::new();
    
    for (hole_idx, &(start, end)) in hole_ranges.iter().enumerate() {
        let mut medians = vec![0.0f64; num_props];
        let mut mads = vec![0.0f64; num_props];
        
        for p in 0..num_props {
            let mut vals: Vec<f64> = (start..end)
                .map(|i| props[p][i])
                .filter(|v| !v.is_nan())
                .collect();
            if vals.len() < 3 {
                mads[p] = global_mads[p].max(1e-10);
                continue;
            }
            vals.sort_by(|a, b| a.partial_cmp(b).unwrap());
            let med = if vals.len() % 2 == 0 {
                (vals[vals.len()/2 - 1] + vals[vals.len()/2]) / 2.0
            } else {
                vals[vals.len()/2]
            };
            medians[p] = med;
            
            let mut devs: Vec<f64> = vals.iter().map(|v| (v - med).abs()).collect();
            devs.sort_by(|a, b| a.partial_cmp(b).unwrap());
            let mad = if devs.len() % 2 == 0 {
                (devs[devs.len()/2 - 1] + devs[devs.len()/2]) / 2.0
            } else {
                devs[devs.len()/2]
            };
            mads[p] = if mad > 0.0 { mad } else { global_mads[p].max(1e-10) };
        }
        
        hole_medians.push(medians);
        hole_mads.push(mads);
    }
    
    // Detect outliers
    let mut suspects: Vec<(u32, u32, String, f64)> = Vec::new();
    
    let cr = 1i64; // Check 3x3x3 neighborhood
    
    for i in 0..n {
        let cx = (xs[i] * inv_cell).floor() as i64;
        let cy = (ys[i] * inv_cell).floor() as i64;
        let cz = (zs[i] * inv_cell).floor() as i64;
        
        for p in 0..num_props {
            let val = props[p][i];
            if val.is_nan() { continue; }
            
            // Spatial z-score
            let mut neighbor_medians = Vec::new();
            let mut neighbor_mads = Vec::new();
            
            for dx in -cr..=cr {
                for dy in -cr..=cr {
                    for dz_cell in -cr..=cr {
                        let key = (cx + dx, cy + dy, cz + dz_cell);
                        if let Some(med) = cell_medians.get(&key) {
                            if let Some(mad) = cell_mads.get(&key) {
                                if cell_counts[&key] >= 3 {
                                    neighbor_medians.push(med[p]);
                                    neighbor_mads.push(mad[p]);
                                }
                            }
                        }
                    }
                }
            }
            
            let spatial_z = if neighbor_medians.len() >= 3 {
                neighbor_medians.sort_by(|a, b| a.partial_cmp(b).unwrap());
                let expected = if neighbor_medians.len() % 2 == 0 {
                    (neighbor_medians[neighbor_medians.len()/2 - 1] + neighbor_medians[neighbor_medians.len()/2]) / 2.0
                } else {
                    neighbor_medians[neighbor_medians.len()/2]
                };
                
                neighbor_mads.sort_by(|a, b| a.partial_cmp(b).unwrap());
                let variability = if neighbor_mads.len() % 2 == 0 {
                    (neighbor_mads[neighbor_mads.len()/2 - 1] + neighbor_mads[neighbor_mads.len()/2]) / 2.0
                } else {
                    neighbor_mads[neighbor_mads.len()/2]
                };
                
                let variability = variability.max(global_mads[p] * 0.01).max(1e-10);
                0.6745 * (val - expected).abs() / variability
            } else {
                // Fall back to global stats
                let variability = global_mads[p].max(1e-10);
                0.6745 * (val - global_medians[p]).abs() / variability
            };
            
            // Within-hole z-score
            // Find which hole this sample belongs to
            let hole_idx = hole_ranges.iter().position(|&(s, e)| i >= s && i < e).unwrap_or(0);
            let hole_z = {
                let h_med = hole_medians[hole_idx][p];
                let h_mad = hole_mads[hole_idx][p].max(global_mads[p] * 0.01).max(1e-10);
                0.6745 * (val - h_med).abs() / h_mad
            };
            
            // Adjacent within-hole z-score (detect sudden jumps)
            let (start, end) = hole_ranges[hole_idx];
            let local_idx = i - start;
            let adj_z = if end - start >= 3 && local_idx > 0 && local_idx < end - start - 1 {
                let prev_val = props[p][i - 1];
                let next_val = props[p][i + 1];
                if !prev_val.is_nan() && !next_val.is_nan() {
                    let expected = (prev_val + next_val) / 2.0;
                    let local_scale = ((next_val - prev_val).abs() / 2.0).max(global_mads[p] * 0.01).max(1e-10);
                    0.6745 * (val - expected).abs() / local_scale
                } else {
                    0.0
                }
            } else {
                0.0
            };
            
            let confidence = spatial_z.max(hole_z).max(adj_z);
            
            if confidence > 3.5 {
                suspects.push((hole_ids[i], sample_indices[i], prop_names[p].clone(), confidence));
            }
        }
    }
    
    // Sort by confidence descending
    suspects.sort_by(|a, b| b.3.partial_cmp(&a.3).unwrap());
    
    // Output
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", suspects.len()).unwrap();
    for (hid, sidx, pname, conf) in &suspects {
        writeln!(out, "{} {} {} {:.4}", hid, sidx, pname, conf).unwrap();
    }
}

Wait, I have a performance issue. Finding which hole a sample belongs to uses `iter().position()` which is O(num_holes) per sample. For 20M samples, that's O(20M * 200K) = way too slow.

I need to precompute a mapping from sample index to hole index. Since samples are stored contiguously by hole, I can use binary search on hole_ranges.

Or even better, I can store the hole index for each sample during parsing.

This is O(1) per sample lookup.

Let me also think about the performance of the main loop. For each of the N samples, for each of the P properties, I do:
1. Look up 27 cells in the HashMap: 27 * O(1) = O(27)
2. Collect medians and MADs: O(27)
3. Sort 27 values: O(27 * log(27)) ≈ O(135)
4. Compute z-score: O(1)

Total per sample per property: ~O(200)
Total: N * P * 200

For N = 20M, P = 5: 20B operations. At ~1 GFLOPS, that's 20 seconds. Should be OK for 5 minutes.
