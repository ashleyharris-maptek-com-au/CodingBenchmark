void load_terrain(const std::string& path) {
    std::ifstream file(path, std::ios::binary);
    if (!file.is_open()) {
        // Handle error
        return;
    }
    
    int w = (int)config.width;
    int d = (int)config.depth;
    size_t total = (size_t)w * d;
    
    terrain.width = w;
    terrain.depth = d;
    terrain.ground_top.resize(total);
    terrain.float_bottom.resize(total);
    terrain.float_top.resize(total);
    
    // Read all data at once for speed
    std::vector<uint16_t> buffer(total * 3);
    file.read(reinterpret_cast<char*>(buffer.data()), total * 3 * sizeof(uint16_t));
    
    for (size_t i = 0; i < total; i++) {
        terrain.ground_top[i] = buffer[i * 3];
        terrain.float_bottom[i] = buffer[i * 3 + 1];
        terrain.float_top[i] = buffer[i * 3 + 2];
    }
    
    std::lock_guard<std::mutex> lock(terrain.mtx);
    terrain.loaded = true;
}

bool is_solid(double x, double y, double z) {
    std::lock_guard<std::mutex> lock(terrain.mtx);
    if (!terrain.loaded) return false;
    
    int ix = (int)std::floor(x);
    int iy = (int)std::floor(y);
    
    if (ix < 0 || ix >= terrain.width || iy < 0 || iy >= terrain.depth) return true;
    
    size_t idx = (size_t)ix * terrain.depth + iy;
    uint16_t a = terrain.ground_top[idx];
    uint16_t b = terrain.float_bottom[idx];
    uint16_t c = terrain.float_top[idx];
    
    int iz = (int)std::floor(z);
    if (iz < 0) return true;
    if (iz < a) return true;
    if (iz >= b && iz < c) return true;
    
    return false;
}

    std::lock_guard<std::mutex> lock(terrain.mtx);
    if (!terrain.loaded) return 0;
    
    int ix = (int)std::floor(x);
    int iy = (int)std::floor(y);
    
    if (ix < 0 || ix >= terrain.width || iy < 0 || iy >= terrain.depth) return 0;
    
    size_t idx = (size_t)ix * terrain.depth + iy;
    return (double)terrain.ground_top[idx];
}

double get_ceiling_height(double x, double y, double z) {
    // Returns the height of the lowest floating rock above z at (x,y)
    // Returns config.height if no floating rock
    std::lock_guard<std::mutex> lock(terrain.mtx);
    if (!terrain.loaded) return config.height;

    int iy = (int)std::floor(y);
    
    if (ix < 0 || ix >= terrain.width || iy < 0 || iy >= terrain.depth) return config.height;
    
    size_t idx = (size_t)ix * terrain.depth + iy;
    uint16_t b = terrain.float_bottom[idx];
    uint16_t c = terrain.float_top[idx];
    
    if (b == c) return config.height; // No floating rock
    if (b > z) return (double)b;
    return config.height;
