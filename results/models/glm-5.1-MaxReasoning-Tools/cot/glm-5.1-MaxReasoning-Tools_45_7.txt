import struct

def make_word(word_count, opcode):
    return (word_count << 16) | opcode

def float_to_word(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]

def string_to_words(s):
    """Convert a string to SPIR-V words (null-terminated, padded to 4 bytes)"""
    s_bytes = s.encode('ascii') + b'\0'
    # Pad to 4-byte boundary
    while len(s_bytes) % 4 != 0:
        s_bytes += b'\0'
    words = []
    for i in range(0, len(s_bytes), 4):
        word = struct.unpack('<I', s_bytes[i:i+4])[0]
        words.append(word)
    return words
