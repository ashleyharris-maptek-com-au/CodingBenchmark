const W = grid.length > 0 ? grid[0].length : 0;

robots.sort((a, b) => a.id - b.id);

// Pre-convert grid to flat array for fast access
const gridFlat = new Uint8Array(H * W);
for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
        gridFlat[y * W + x] = grid[y].charCodeAt(x) === 46 ? 1 : 0; // '.' = 46
    }

const posX = new Int32Array(N);
const posY = new Int32Array(N);
for (let r = 0; r < R; r++) {
    posX[r] = robots[r].x;
    posY[r] = robots[r].y;
}
for (let j = 0; j < J; j++) {
    posX[R + j] = jobs[j].x;
    posY[R + j] = jobs[j].y;
}

// Distance matrix: dist[i * N + j] = BFS distance from position i to position j
const dist = new Int32Array(N * N).fill(-1);

// BFS from position srcIdx
function computeBFS(srcIdx: number): void {
    const sx = posX[srcIdx];
    const sy = posY[srcIdx];
    const d = new Int32Array(H * W).fill(-1);
    const start = sy * W + sx;
    d[start] = 0;
    const q = new Int32Array(H * W);
    q[0] = start;
    let head = 0, tail = 1;
    while (head < tail) {
        const ck = q[head++];
        const cd = d[ck];
        const nd = cd + 1;
        // Right
        const rk = ck + 1;
        if ((ck % W) + 1 < W && gridFlat[rk] && d[rk] === -1) {
            d[rk] = nd;
            q[tail++] = rk;
        }
        // Left
        const lk = ck - 1;
        if (ck % W > 0 && gridFlat[lk] && d[lk] === -1) {
            d[lk] = nd;
            q[tail++] = lk;
        }
        // Down
        const dk = ck + W;
        if (dk < H * W && gridFlat[dk] && d[dk] === -1) {
            d[dk] = nd;
            q[tail++] = dk;
        }
        // Up
        const uk = ck - W;
        if (uk >= 0 && gridFlat[uk] && d[uk] === -1) {
            d[uk] = nd;
            q[tail++] = uk;
        }
    }
    // Extract distances to all positions
    const base = srcIdx * N;
    for (let i = 0; i < N; i++) {
        dist[base + i] = d[posY[i] * W + posX[i]];
    }
}

    computeBFS(i);
}

        const route = routes[r];
        const d_rj = dist[r * N + jPosIdx]; // distance from robot r to job j
        
        if (d_rj === -1) continue; // unreachable from this robot
        
        if (route.length === 0) {
            // Insert as only job
            const costIncrease = d_rj;
            if (costIncrease < bestCostIncrease || (costIncrease === bestCostIncrease && r < bestRobot)) {
                bestCostIncrease = costIncrease;
                bestRobot = r;
                bestPos = 0;
            }
        } else {
            // Try inserting at the beginning
            const firstJ = route[0];
            const firstJPosIdx = R + firstJ;
            const d_j_first = dist[jPosIdx * N + firstJPosIdx]; // distance from j to first job
            const d_r_first = dist[r * N + firstJPosIdx]; // distance from robot r to first job
            
            let costIncrease = d_rj + d_j_first - d_r_first;
            if (costIncrease < bestCostIncrease || (costIncrease === bestCostIncrease && r < bestRobot)) {
                bestCostIncrease = costIncrease;
                bestRobot = r;
                bestPos = 0;
            }
            
            // Try inserting between consecutive jobs
            for (let p = 0; p < route.length - 1; p++) {
                const prevJ = route[p];
                const nextJ = route[p + 1];
                const prevJPosIdx = R + prevJ;
                const nextJPosIdx = R + nextJ;
                const d_prev_j = dist[prevJPosIdx * N + jPosIdx];
                const d_j_next = dist[jPosIdx * N + nextJPosIdx];
                const d_prev_next = dist[prevJPosIdx * N + nextJPosIdx];
                
                costIncrease = d_prev_j + d_j_next - d_prev_next;
                if (costIncrease < bestCostIncrease || (costIncrease === bestCostIncrease && r < bestRobot)) {
                    bestCostIncrease = costIncrease;
                    bestRobot = r;
                    bestPos = p + 1;
                }
            }
            
            // Try inserting at the end
            const lastJ = route[route.length - 1];
            const lastJPosIdx = R + lastJ;
            const d_last_j = dist[lastJPosIdx * N + jPosIdx];
            
            costIncrease = d_last_j;
            if (costIncrease < bestCostIncrease || (costIncrease === bestCostIncrease && r < bestRobot)) {
                bestCostIncrease = costIncrease;
                bestRobot = r;
                bestPos = route.length;
            }
        }
    }
    
    if (bestRobot === -1) {
        // Job is unreachable from all robots - assign to robot 0 as fallback
        bestRobot = 0;
        bestPos = routes[0].length;
    }
    
    routes[bestRobot].splice(bestPos, 0, j);
}

// Try to improve by relocating jobs between routes

function computeRouteCost(r: number): number {
    const route = routes[r];
    if (route.length === 0) return 0;
    let cost = dist[r * N + (R + route[0])]; // robot to first job
    for (let i = 0; i < route.length - 1; i++) {
        cost += dist[(R + route[i]) * N + (R + route[i + 1])];
    }
    return cost;
}

function totalCost(): number {
    let cost = 0;
    for (let r = 0; r < R; r++) {
        cost += computeRouteCost(r);
    }
    return cost;
}
