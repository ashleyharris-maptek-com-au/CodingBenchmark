use std::io::{self, Read, Write, BufReader, BufWriter, Seek, SeekFrom};

// Simple xorshift64 RNG
struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Rng { state: if seed == 0 { 1 } else { seed } }
    }
    
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    
    fn next_usize(&mut self, n: usize) -> usize {
        (self.next_u64() % n as u64) as usize
    }
    
    fn next_f64(&mut self) -> f64 {
        (self.next_u64() >> 11) as f64 / (1u64 << 53) as f64
    }
}

// ---- Simple RNG ----
struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Rng { state: if seed == 0 { 1 } else { seed } }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    fn next_usize(&mut self, n: usize) -> usize {
        (self.next_u64() % n as u64) as usize
    }
    fn next_f64(&mut self) -> f64 {
        (self.next_u64() >> 11) as f64 / (1u64 << 53) as f64
    }
}

fn dist_sq(p: &Point, c: &Centroid) -> f64 {
    let dx = p.x as f64 - c.x;
    let dy = p.y as f64 - c.y;
    let dz = p.z as f64 - c.z;
    dx * dx + dy * dy + dz * dz
}

fn nearest(p: &Point, centroids: &[Centroid]) -> usize {
    let mut best = 0;
    let mut best_d = dist_sq(p, &centroids[0]);
    for i in 1..centroids.len() {
        let d = dist_sq(p, &centroids[i]);
        if d < best_d {
            best_d = d;
            best = i;
        }
    }
    best
}

        }
        if changed == 0 { break; }
        
        // Update
        for c in centroids.iter_mut() {
            c.x = 0.0; c.y = 0.0; c.z = 0.0; c.count = 0;

            let c = &mut centroids[assignments[i] as usize];
            c.x += points[i].x as f64;
            c.y += points[i].y as f64;
            c.z += points[i].z as f64;
            c.count += 1;
        }
        for c in centroids.iter_mut() {
            if c.count > 0 {
                c.x /= c.count as f64;
                c.y /= c.count as f64;
                c.z /= c.count as f64;
            }
        }
        handle_empty_clusters(&mut centroids, &mut rng);
    }
    
    // Final assignment
    for i in 0..n {
        assignments[i] = nearest(&points[i], &centroids) as u32;
    }
    
    // Output
    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    for &a in &assignments {
        writer.write_all(&a.to_le_bytes()).unwrap();
    }
    writer.flush().unwrap();
}

// Simple xorshift64 RNG
struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Self { state: if seed == 0 { 1 } else { seed } }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    fn next_usize(&mut self, n: usize) -> usize {
        (self.next_u64() % n as u64) as usize
    }
    fn next_f64(&mut self) -> f64 {
        (self.next_u64() >> 11) as f64 / (1u64 << 53) as f64
    }
}

#[derive(Clone, Copy)]
struct Point { x: f32, y: f32, z: f32 }

#[derive(Clone)]
struct Centroid { x: f64, y: f64, z: f64 }

impl Centroid {
    fn from_point(p: &Point) -> Self {
        Centroid { x: p.x as f64, y: p.y as f64, z: p.z as f64 }
    }
    fn zero() -> Self {
        Centroid { x: 0.0, y: 0.0, z: 0.0 }
    }
    fn dist_sq(&self, p: &Point) -> f64 {
        let dx = self.x - p.x as f64;
        let dy = self.y - p.y as f64;
        let dz = self.z - p.z as f64;
        dx * dx + dy * dy + dz * dz
    }
}

    Point {
        x: f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]),
        y: f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]),
        z: f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]),
    }
}
