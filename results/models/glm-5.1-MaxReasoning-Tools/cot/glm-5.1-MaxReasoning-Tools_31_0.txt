fn bitset_dp(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let t = target as usize;
    let num_words = (t + 64) / 64;
    let mut bits = vec![0u64; num_words];
    let mut prev = vec![0u32; t + 1]; // prev[s] = index of number added to achieve sum s
    let mut prev_sum = vec![0u32; t + 1]; // prev_sum[s] = sum before adding
    
    bits[0] = 1; // bit 0 is set
    prev[0] = u32::MAX; // sentinel for base case
    
    for (j, &num) in numbers.iter().enumerate() {
        if num <= 0 || num as usize > t {
            continue;
        }
        let shift = num as usize;
        let word_shift = shift / 64;
        let bit_shift = shift % 64;
        
        // Compute bits << shift
        let mut shifted = vec![0u64; num_words];
        for i in word_shift..num_words {
            shifted[i] = bits[i - word_shift] << bit_shift;
            if bit_shift > 0 && i > word_shift {
                shifted[i] |= bits[i - word_shift - 1] >> (64 - bit_shift);
            }
        }
        
        // Find newly set bits and update prev
        for i in 0..num_words {
            let new_bits = shifted[i] & !bits[i]; // newly set bits in this word
            if new_bits != 0 {
                let base = i * 64;
                let mut remaining = new_bits;
                while remaining != 0 {
                    let bit_pos = remaining.trailing_zeros() as usize;
                    let s = base + bit_pos;
                    if s <= t {
                        prev[s] = j as u32;
                        prev_sum[s] = (s - shift) as u32;
                    }
                    remaining &= remaining - 1; // clear lowest set bit
                }
            }
        }
        
        // Update bits
        for i in 0..num_words {
            bits[i] |= shifted[i];
        }
    }
    
    if (bits[t / 64] >> (t % 64)) & 1 == 0 {
        return None;
    }
    
    // Backtrack
    let mut indices = Vec::new();
    let mut s = t;
    while s > 0 {
        let idx = prev[s] as usize;
        indices.push(idx);
        s = prev_sum[s] as usize;
    }
    
    Some(indices)
}

fn meet_in_the_middle(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let mid = n / 2;
    let left = &numbers[..mid];
    let right = &numbers[mid..];
    
    // Enumerate all subsets of left half
    let mut left_sums: HashMap<i64, Vec<usize>> = HashMap::new();
    for mask in 0..(1 << left.len()) {
        let mut sum = 0i64;
        let mut indices = Vec::new();
        for i in 0..left.len() {
            if mask & (1 << i) != 0 {
                sum += left[i];
                indices.push(i);
            }
        }
        left_sums.insert(sum, indices);
    }
    
    // Enumerate all subsets of right half
    for mask in 0..(1 << right.len()) {
        let mut sum = 0i64;
        let mut indices = Vec::new();
        for i in 0..right.len() {
            if mask & (1 << i) != 0 {
                sum += right[i];
                indices.push(mid + i);
            }
        }
        let needed = target - sum;
        if let Some(left_indices) = left_sums.get(&needed) {
            let mut result = left_indices.clone();
            result.extend(indices);
            return Some(result);
        }
    }
    
    None
}

    let mut states: HashMap<i64, Vec<usize>> = HashMap::new();
    states.insert(0, vec![]);

            }

        for (sum, indices) in new_states {
            states.entry(sum).or_insert(indices);
        }
        
        // Trim: keep only max_states closest to target
        if states.len() > max_states {
            let mut entries: Vec<(i64, Vec<usize>)> = states.into_iter().collect();
            entries.sort_by_key(|(s, _)| (target - s).abs());
            entries.truncate(max_states);
            states = entries.into_iter().collect();
        }
    }
    
    states.get(&target).cloned()
}

fn hashmap_dp(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut parent: HashMap<i64, (i64, usize)> = HashMap::new();
    parent.insert(0, (0, usize::MAX)); // sentinel
    
    let mut achievable: HashSet<i64> = HashSet::new();
    achievable.insert(0);
    
    for (j, &num) in numbers.iter().enumerate() {
        if num == 0 {
            continue;
        }
        
        let current_sums: Vec<i64> = achievable.iter().cloned().collect();
        for sum in current_sums {
            let new_sum = sum + num;
            if !achievable.contains(&new_sum) {
                achievable.insert(new_sum);
                parent.insert(new_sum, (sum, j));
            }
        }
        
        if achievable.contains(&target) {
            // Backtrack
            let mut indices = Vec::new();
            let mut s = target;
            while s != 0 {
                let (prev_s, idx) = parent[&s];
                indices.push(idx);
                s = prev_s;
            }
            return Some(indices);
        }
        
        if achievable.len() > 10_000_000 {
            return None;
        }
    }
    
    if achievable.contains(&target) {
        let mut indices = Vec::new();
        let mut s = target;
        while s != 0 {
            let (prev_s, idx) = parent[&s];
            indices.push(idx);
            s = prev_s;
        }
        Some(indices)
    } else {
        None
    }
}
