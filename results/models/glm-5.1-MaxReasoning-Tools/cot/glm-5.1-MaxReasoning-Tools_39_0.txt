    }

fn lower_bound(adj: &Vec<Vec<usize>>, partition: &Vec<i8>, count_a: usize, current_cut: usize, target: usize, n: usize) -> usize {

    let remaining_b = target - count_b;

        return current_cut;
    }
    
    // For each unassigned vertex, compute a_v - b_v
    let mut diffs: Vec<i32> = Vec::new();
    let mut total_a: i32 = 0;
    
    for v in 0..n {
        if partition[v] != -1 { continue; }
        let mut a_v: i32 = 0;
        let mut b_v: i32 = 0;
        for &u in &adj[v] {
            if partition[u] == 0 { a_v += 1; }
            else if partition[u] == 1 { b_v += 1; }
        }
        total_a += a_v;
        diffs.push(a_v - b_v);
    }
    
    if diffs.is_empty() {
        return current_cut;
    }
    
    // Sort descending and take top remaining_a
    diffs.sort_by(|a, b| b.cmp(a));
    let top_sum: i32 = diffs.iter().take(remaining_a).sum();
    
    let min_boundary = (total_a - top_sum).max(0) as usize;
    
    current_cut + min_boundary
}
