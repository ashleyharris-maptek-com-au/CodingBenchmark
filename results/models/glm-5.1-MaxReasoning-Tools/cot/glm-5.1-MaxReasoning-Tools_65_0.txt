function findEarliestSlot(slots: Array<{start: number, finish: number}>, earliest: number, duration: number): number {
    // Find the earliest time >= earliest where a slot of 'duration' fits
    if (slots.length === 0) return earliest;
    
    // Try before the first slot
    if (slots[0].start >= earliest + duration) return earliest;
    
    // Try between slots
    for (let i = 0; i < slots.length - 1; i++) {
        const gapStart = Math.max(earliest, slots[i].finish);
        const gapEnd = slots[i + 1].start;
        if (gapStart + duration <= gapEnd) return gapStart;
    }
    
    // After the last slot
    return Math.max(earliest, slots[slots.length - 1].finish);
}

function insertIntoSchedule(slots: Array<{start: number, finish: number}>, start: number, finish: number): void {
    let pos = 0;
    while (pos < slots.length && slots[pos].start < start) pos++;
    slots.splice(pos, 0, {start, finish});
}
