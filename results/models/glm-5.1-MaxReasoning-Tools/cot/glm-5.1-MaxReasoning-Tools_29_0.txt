class MinHeap
{
    private List<(int priority, int value)> heap = new List<(int, int)>();
    
    public int Count => heap.Count;
    
    public void Enqueue(int priority, int value)
    {
        heap.Add((priority, value));
        int i = heap.Count - 1;
        while (i > 0)
        {
            int parent = (i - 1) / 2;
            if (heap[parent].priority <= heap[i].priority) break;
            (heap[parent], heap[i]) = (heap[i], heap[parent]);
            i = parent;
        }
    }
    
    public (int priority, int value) Dequeue()
    {
        var result = heap[0];
        heap[0] = heap[heap.Count - 1];
        heap.RemoveAt(heap.Count - 1);
        int i = 0;
        while (true)
        {
            int left = 2 * i + 1, right = 2 * i + 2, smallest = i;
            if (left < heap.Count && heap[left].priority < heap[smallest].priority) smallest = left;
            if (right < heap.Count && heap[right].priority < heap[smallest].priority) smallest = right;
            if (smallest == i) break;
            (heap[i], heap[smallest]) = (heap[smallest], heap[i]);
            i = smallest;
        }
        return result;
    }
}

Now, for the sequential coloring, I need to color the vertices in P in a way that gives a good bound. The standard approach:

1. Sort P by degree (descending) within the subgraph induced by P
2. Color greedily: for each vertex in order, assign the smallest color not used by its already-colored neighbors

But computing degrees within P is expensive. An alternative is to just use the global degree ordering.

Tomita's approach: color vertices in the order they appear in P (which is typically sorted by some heuristic), and assign colors greedily.

Let me implement the coloring:

```csharp
// Color P[0..pSize-1] and sort by color (ascending)
// Returns the number of colors used
// After this function, P is reordered and colorArray[level][i] = color of P[i]
static int SequentialColorAndSort(int[] P, int pSize, int level)
{
    int[] colors = colorArray[level];
    int[] tempP = tempPArray[level];
    
    // Copy P to tempP
    Array.Copy(P, tempP, pSize);
    
    int numColors = 0;
    int[] colorOf = colorArray[level]; // reuse
    
    // Color vertices greedily
    for (int i = 0; i < pSize; i++)
    {
        // Find smallest color not used by already-colored neighbors
        bool[] used = usedColorArray[level];
        for (int c = 1; c <= numColors; c++)
            used[c] = false;
        
        int v = tempP[i];
        for (int j = 0; j < i; j++)
        {
            if (IsAdj(v, tempP[j]))
                used[colorOf[j]] = true;
        }
        
        colorOf[i] = 1;
        while (used[colorOf[i]])
            colorOf[i]++;
        
        if (colorOf[i] > numColors)
            numColors = colorOf[i];
    }
    
    // Now sort tempP by color (ascending) and copy back to P
    // Also need to track the color assignment
    var indexed = new (int vertex, int color)[pSize];
    for (int i = 0; i < pSize; i++)
        indexed[i] = (tempP[i], colorOf[i]);
    
    Array.Sort(indexed, (a, b) => a.color.CompareTo(b.color));
    
    for (int i = 0; i < pSize; i++)
    {
        P[i] = indexed[i].vertex;
        colors[i] = indexed[i].color;
    }
    
    return numColors;
}
```
