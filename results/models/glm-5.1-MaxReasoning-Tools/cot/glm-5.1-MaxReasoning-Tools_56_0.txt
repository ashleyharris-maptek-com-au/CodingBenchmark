            newRoute = routeJ.concat(routeI);

function clarkeWright() {
    // Each package in its own route
    let routes = [];
    for (let i = 1; i <= n; i++) routes.push([i]);
    
    const routeOf = new Int32Array(n + 1);
    for (let i = 1; i <= n; i++) routeOf[i] = i - 1;
    
    // Compute savings
    const savings = [];
    for (let i = 1; i <= n; i++) {
        for (let j = i + 1; j <= n; j++) {
            const s = dist(0, i) + dist(0, j) - dist(i, j);
            if (s > 0) savings.push([s, i, j]);
        }
    }
    savings.sort((a, b) => b[0] - a[0]);
    
    const active = new Uint8Array(n);

        const routeI = routes[ri], routeJ = routes[rj];
        const iAtStart = routeI[0] === i;
        const iAtEnd = routeI[routeI.length - 1] === i;
        const jAtStart = routeJ[0] === j;
        const jAtEnd = routeJ[routeJ.length - 1] === j;
        
        if (!iAtStart && !iAtEnd) continue;
        if (!jAtStart && !jAtEnd) continue;

        } else if (iAtStart && jAtEnd) {

            newRoute = [...routeI].reverse().concat(routeJ);

        routes[ri] = newRoute;
        routes[rj] = [];
        active[rj] = 0;
        activeCount--;
        
        for (const p of newRoute) routeOf[p] = ri;
    }
    
    let result = [];

        if (active[i] && routes[i].length > 0) result.push(routes[i]);

    while (result.length > truckCount) {

                    rb.concat(ra),

    let improved = true;
