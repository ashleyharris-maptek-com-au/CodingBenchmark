use std::io::{self, BufRead, Write, BufWriter, BufReader};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone)]
struct Config {
    width: f64,
    height: f64,
    gravity: f64,
    max_thrust: f64,
    initial_fuel: f64,
    target_x: f64,
    target_y: f64,
}

fn main() {
    let state = Arc::new(Mutex::new(State {
        x: 0.0, y: 0.0, vx: 0.0, vy: 0.0,
        angle: 0.0, angular_vel: 0.0, fuel: 0.0,
    }));
    let config = Arc::new(Mutex::new(None::<Config>));
    let state_available = Arc::new(Mutex::new(false));
    
    // Reader thread
    let state_r = Arc::clone(&state);
    let config_r = Arc::clone(&config);
    let state_available_r = Arc::clone(&state_available);
    
    let reader_handle = thread::spawn(move || {
        let stdin = io::stdin();
        let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());
        let mut line = String::with_capacity(1 << 16);
        
        // Read first line
        line.clear();
        reader.read_line(&mut line).unwrap();
        let parts: Vec<f64> = line.trim().split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let width = parts[0];
        let height = parts[1];
        let gravity = parts[2];
        let max_thrust = parts[3];
        let initial_fuel = parts[4];
        
        // Read second line
        line.clear();
        reader.read_line(&mut line).unwrap();
        let parts: Vec<f64> = line.trim().split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let start_x = parts[0];
        let start_y = parts[1];
        let target_x = parts[2];
        let target_y = parts[3];
        
        // Store config
        {
            let mut c = config_r.lock().unwrap();
            *c = Some(Config {
                width, height, gravity, max_thrust, initial_fuel,
                target_x, target_y,
            });
        }
        
        // Set initial state
        {
            let mut s = state_r.lock().unwrap();
            s.x = start_x;
            s.y = start_y;
            s.fuel = initial_fuel;
        }
        *state_available_r.lock().unwrap() = true;
        
        // Read map rows - skip them
        let height_int = height as usize;
        for _ in 0..height_int {
            line.clear();
            if reader.read_line(&mut line).unwrap() == 0 {
                break;
            }
        }

        loop {
            line.clear();
            match reader.read_line(&mut line) {
