```python
def _fuse_median(values_with_validity):
    """Fuse multiple sensor readings using median with outlier rejection.
    values_with_validity: list of (value, valid) tuples
    Returns: fused value or None
    """
    # First pass: only use values marked valid
    valid_values = [v for v, valid in values_with_validity if valid]
    
    if len(valid_values) >= 3:
        # With 3+ values, use median with outlier rejection
        median = sorted(valid_values)[len(valid_values)//2]
        # Reject values more than threshold from median
        threshold = max(abs(median) * 0.5, 5.0)  # reasonable threshold
        filtered = [v for v in valid_values if abs(v - median) < threshold]
        if filtered:
            return sorted(filtered)[len(filtered)//2]
        return median
    elif len(valid_values) == 2:
        return sum(valid_values) / 2.0
    elif len(valid_values) == 1:
        return valid_values[0]
    else:
        # No valid readings - try using all readings
        all_values = [v for v, valid in values_with_validity]
        if all_values:
            median = sorted(all_values)[len(all_values)//2]
            return median
        return None
```

```python
def _robust_median(values, default=0.0):
    """Compute median with outlier rejection."""
    if not values:
        return default
    if len(values) == 1:
        return values[0]
    sorted_vals = sorted(values)
    n = len(sorted_vals)
    median = sorted_vals[n // 2] if n % 2 == 1 else (sorted_vals[n//2-1] + sorted_vals[n//2]) / 2.0
    
    if n >= 3:
        # Reject outliers more than 3 MAD from median
        abs_devs = sorted([abs(v - median) for v in sorted_vals])
        mad = abs_devs[n // 2] * 1.4826  # MAD to std conversion
        threshold = max(3.0 * mad, abs(median) * 0.5 + 1.0)
        filtered = [v for v in sorted_vals if abs(v - median) <= threshold]
        if filtered:
            nf = len(filtered)
            return filtered[nf // 2] if nf % 2 == 1 else (filtered[nf//2-1] + filtered[nf//2]) / 2.0
    
    return median


def _rate_limit(target, current, max_delta):
    """Limit the rate of change."""
    delta = target - current
    if abs(delta) <= max_delta:
        return target
    return current + max_delta * (1 if delta > 0 else -1)
```
