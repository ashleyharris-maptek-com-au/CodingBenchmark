bool backtrack(int idx, const vector<int>& order, vector<uint64_t>& domains) {
    if (idx == n) return true;
    
    int v = order[idx];
    uint64_t avail = domains[v] & ((1ULL << k) - 1);
    
    while (avail) {
        int c = __builtin_ctzll(avail);
        avail &= avail - 1;
        
        color[v] = c;
        
        // Forward checking: remove c from neighbors' domains
        vector<int> propagated;
        bool ok = true;
        for (int u : adj[v]) {
            if (color[u] == -1 && (domains[u] & (1ULL << c))) {
                domains[u] &= ~(1ULL << c);
                propagated.push_back(u);
                if ((domains[u] & ((1ULL << k) - 1)) == 0) {
                    ok = false;
                    break;
                }
            }
        }
        
        if (ok && backtrack(idx + 1, order, domains)) return true;
        
        // Undo
        for (int u : propagated) {
            domains[u] |= (1ULL << c);
        }
        color[v] = -1;
    }
    
    return false;
}
