def word(val):
    return struct.pack('<I', val & 0xFFFFFFFF)

def fword(val):
    return struct.pack('<f', val)

def encode_string(s):
    """Encode a null-terminated string padded to 4-byte boundary"""
    b = s.encode('ascii') + b'\x00'
    b += b'\x00' * ((4 - len(b) % 4) % 4)
    return b

def instr(word_count, opcode, *operands):
    """Encode a SPIR-V instruction"""
    result = word((word_count << 16) | opcode)
    for op in operands:
        if isinstance(op, bytes):
            result += op
        else:
            result += word(op)
    return result

def instr(word_count, opcode, *operands):

    s_bytes = s.encode('ascii') + b'\0'
