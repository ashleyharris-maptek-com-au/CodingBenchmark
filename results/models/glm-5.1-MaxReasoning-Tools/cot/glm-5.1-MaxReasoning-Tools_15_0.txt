use std::io::{self, BufRead, Write, BufWriter};
use std::collections::HashSet;

type Point = (f64, f64);

fn cross2d(o: Point, a: Point, b: Point) -> f64 {
    (a.0 - o.0) * (b.1 - o.1) - (a.1 - o.1) * (b.0 - o.0)
}

fn point_in_triangle(p: Point, a: Point, b: Point, c: Point) -> bool {
    let d1 = cross2d(p, a, b);
    let d2 = cross2d(p, b, c);
    let d3 = cross2d(p, c, a);
    let has_neg = d1 < 0.0 || d2 < 0.0 || d3 < 0.0;
    let has_pos = d1 > 0.0 || d2 > 0.0 || d3 > 0.0;
    !(has_neg && has_pos)
}

fn triangulate(poly: &[Point]) -> Vec<(Point, Point, Point)> {
    let n = poly.len();
    if n < 3 {
        return vec![];
    }
    if n == 3 {
        return vec![(poly[0], poly[1], poly[2])];
    }
    
    // Ensure CCW
    let mut vertices: Vec<usize> = (0..n).collect();
    let area: f64 = (0..n).map(|i| {
        let j = (i + 1) % n;
        poly[i].0 * poly[j].1 - poly[j].0 * poly[i].1
    }).sum();
    if area < 0.0 {
        vertices.reverse();
    }
    
    let mut triangles = Vec::new();
    let mut remaining = vertices.clone();
    
    while remaining.len() > 3 {
        let len = remaining.len();
        let mut ear_found = false;
        
        for i in 0..len {
            let prev = remaining[(i + len - 1) % len];
            let curr = remaining[i];
            let next = remaining[(i + 1) % len];
            
            let a = poly[prev];
            let b = poly[curr];
            let c = poly[next];
            
            // Check if ear (convex)
            if cross2d(a, b, c) <= 0.0 {
                continue;
            }
            
            // Check no other vertex inside
            let mut is_ear = true;
            for j in 0..len {
                if j == (i + len - 1) % len || j == i || j == (i + 1) % len {
                    continue;
                }
                if point_in_triangle(poly[remaining[j]], a, b, c) {
                    is_ear = false;
                    break;
                }
            }
            
            if is_ear {
                triangles.push((a, b, c));
                remaining.remove(i);
                ear_found = true;
                break;
            }
        }
        
        if !ear_found {
            break; // shouldn't happen for simple polygons
        }
    }
    
    if remaining.len() == 3 {
        triangles.push((poly[remaining[0]], poly[remaining[1]], poly[remaining[2]]));
    }
    
    triangles
}
