# SPIR-V builder
class SpirvBuilder:
    def __init__(self):
        self.words = []
        self.next_id = 1
        self.instructions = []  # list of list of words
        self.header_words = None
        
    def alloc_id(self):
        id = self.next_id
        self.next_id += 1
        return id
    
    def emit(self, *words):
        self.words.extend(words)
    
    def encode_instruction(self, word_count, opcode, *operands):
        first_word = (word_count << 16) | opcode
        result = [first_word] + list(operands)
        return result
    
    def finalize(self):
        bound = self.next_id
        header = [
            0x07230203,  # Magic
            0x00010000,  # Version 1.0
            0x00000000,  # Generator
            bound,
            0x00000000   # Reserved
        ]
        all_words = header + self.words
        # Convert to bytes (little-endian)
        result = b''
        for w in all_words:
            result += struct.pack('<I', w)
        return result

    def __init__(self):
        self.words = []
        self.next_id = 1
    
    def alloc_id(self):
        i = self.next_id
        self.next_id += 1
        return i
    
    def emit(self, *args):
        for a in args:
            self.words.append(a)
    
    def instr(self, word_count, opcode, *operands):
