using System;
using System.Diagnostics;

class Program
{
    static int N;
    static long[] flow, dist;
    static Random rng = new Random(42);

    static long ComputeCost(int[] p)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            int pi = p[i], iN = i * N, piN = pi * N;
            for (int j = 0; j < N; j++)
                cost += flow[iN + j] * dist[piN + p[j]];
        }
        return cost;
    }

    static long SwapDelta(int[] p, int a, int b)
    {
        long delta = 0;
        int pa = p[a], pb = p[b];
        int aN = a * N, bN = b * N, paN = pa * N, pbN = pb * N;
        for (int k = 0; k < N; k++)
        {
            if (k == a || k == b) continue;
            int pk = p[k], kN = k * N;
            delta += (flow[aN + k] - flow[bN + k]) * (dist[pbN + pk] - dist[paN + pk]);
            delta += (flow[kN + a] - flow[kN + b]) * (dist[pk * N + pb] - dist[pk * N + pa]);
        }
        delta += flow[aN + b] * (dist[pbN + pa] - dist[paN + pb]);
        delta += flow[bN + a] * (dist[paN + pb] - dist[pbN + pa]);
        delta += flow[aN + a] * (dist[pbN + pb] - dist[paN + pa]);
        delta += flow[bN + b] * (dist[paN + pa] - dist[pbN + pb]);
        return delta;
    }

    static void Main()
    {
        var sw = Stopwatch.StartNew();

        string allInput = Console.In.ReadToEnd();
        string[] tokens = allInput.Split(new char[] { ' ', '\t', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        N = int.Parse(tokens[idx++]);
        flow = new long[N * N];
        dist = new long[N * N];
        for (int i = 0; i < N * N; i++) flow[i] = long.Parse(tokens[idx++]);
        for (int i = 0; i < N * N; i++) dist[i] = long.Parse(tokens[idx++]);

        if (N == 1)
        {
            Console.WriteLine(flow[0] * dist[0]);
            Console.WriteLine(0);
            return;
        }

        int[] bestP = new int[N];
        long bestCost = long.MaxValue;

        // Exact search for small N
        if (N <= 10)
        {
            int[] perm = new int[N];
            bool[] used = new bool[N];
            long currentPartialCost = 0; // Not used for pruning in this simple version
            
            void Search(int pos)
            {
                if (pos == N)
                {
                    long c = ComputeCost(perm);
                    if (c < bestCost)
                    {
                        bestCost = c;
                        Array.Copy(perm, bestP, N);
                    }
                    return;
                }
                for (int i = 0; i < N; i++)
                {
                    if (used[i]) continue;
                    used[i] = true;
                    perm[pos] = i;
                    Search(pos + 1);
                    used[i] = false;
                }
            }
            
            Search(0);
            Console.WriteLine(bestCost);
            Console.WriteLine(string.Join(" ", bestP));
            return;
        }

        // Heuristic approach for N > 10
        
        // Greedy construction
        long[] fSum = new long[N], dSum = new long[N];
        for (int i = 0; i < N; i++)
        {
            int iN = i * N;
            for (int j = 0; j < N; j++)
            {
                fSum[i] += flow[iN + j];
                dSum[j] += dist[iN + j];
            }
        }
        int[] fOrd = new int[N], dOrd = new int[N];
        for (int i = 0; i < N; i++) { fOrd[i] = i; dOrd[i] = i; }
        Array.Sort(fOrd, (x, y) => fSum[y].CompareTo(fSum[x]));
        Array.Sort(dOrd, (x, y) => dSum[x].CompareTo(dSum[y]));
        int[] p = new int[N];
        for (int i = 0; i < N; i++) p[fOrd[i]] = dOrd[i];
        long cost = ComputeCost(p);
        
        if (cost < bestCost) { bestCost = cost; bestP = (int[])p.Clone(); }

        // Full 2-opt local search (for N <= 256)
        if (N <= 256)
        {
            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < N - 1; i++)
                    for (int j = i + 1; j < N; j++)
                    {
                        long d = SwapDelta(p, i, j);
                        if (d < 0)
                        {
                            int tmp = p[i]; p[i] = p[j]; p[j] = tmp;
                            cost += d;
                            improved = true;
                        }
                    }
            }
            if (cost < bestCost) { bestCost = cost; bestP = (int[])p.Clone(); }
        }

        // Simulated annealing
        // Calibrate temperature
        long sumAbs = 0;
        int nSamp = Math.Min(5000, N * (N - 1) / 2);
        for (int s = 0; s < nSamp; s++)
        {
            int a = rng.Next(N), b = rng.Next(N - 1);
            if (b >= a) b++;
            if (a > b) { int t = a; a = b; b = t; }
            sumAbs += Math.Abs(SwapDelta(bestP, a, b));
        }
        double avgAbs = (double)sumAbs / nSamp;
        double T0 = Math.Max(1.0, avgAbs * 2.0);
        double T = T0;

        p = (int[])bestP.Clone();
        cost = bestCost;

        double timeLimitSec = 270.0;
        double coolingRate;
        
        // Estimate total SA iterations
        long estItersPerSec = Math.Max(1, 200000000L / Math.Max(1, 8 * N));
        long totalEstIters = (long)(estItersPerSec * (timeLimitSec - sw.Elapsed.TotalSeconds) * 0.85);
        totalEstIters = Math.Max(10000, totalEstIters);
        
        double finalT = 0.01;
        coolingRate = Math.Pow(finalT / T0, 1.0 / totalEstIters);

        long saIter = 0;
        long noImproveCount = 0;
        long noImproveThreshold = totalEstIters / 4;
        
        while (sw.Elapsed.TotalSeconds < timeLimitSec)
        {
            int a = rng.Next(N), b = rng.Next(N - 1);
            if (b >= a) b++;
            if (a > b) { int t = a; a = b; b = t; }
            
            long d = SwapDelta(p, a, b);
            if (d < 0 || rng.NextDouble() < Math.Exp(-d / T))
            {
                int tmp = p[a]; p[a] = p[b]; p[b] = tmp;
                cost += d;
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestP = (int[])p.Clone();
                    noImproveCount = 0;
                }
                else
                {
                    noImproveCount++;
                }
            }
            else
            {
                noImproveCount++;
            }
            
            T *= coolingRate;
            saIter++;

            // Reheat if stuck
            if (noImproveCount > noImproveThreshold)
            {
                T = T0 * 0.3;
                noImproveCount = 0;
            }
        }

        // Final local search on best solution
        if (N <= 256)
        {
            p = (int[])bestP.Clone();
            cost = bestCost;
            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < N - 1; i++)
                    for (int j = i + 1; j < N; j++)
                    {
                        long d = SwapDelta(p, i, j);
                        if (d < 0)
                        {
                            int tmp = p[i]; p[i] = p[j]; p[j] = tmp;
                            cost += d;
                            improved = true;
                        }
                    }
            }
            if (cost < bestCost) { bestCost = cost; bestP = (int[])p.Clone(); }
        }

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestP));
    }
}
