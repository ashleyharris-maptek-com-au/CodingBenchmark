type Pt = (f64, f64);

fn rotate_point(p: Pt, angle: f64) -> Pt {
    let (x, y) = p;
    let c = angle.cos();
    let s = angle.sin();
    (x * c - y * s, x * s + y * c)
}

fn translate_polygon(poly: &[Pt], dx: f64, dy: f64) -> Vec<Pt> {
    poly.iter().map(|&(x, y)| (x + dx, y + dy)).collect()
}

fn rotate_polygon(poly: &[Pt], angle: f64) -> Vec<Pt> {
    poly.iter().map(|&p| rotate_point(p, angle)).collect()
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut area = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        area += poly[i].0 * poly[j].1;
        area -= poly[j].0 * poly[i].1;
    }
    area / 2.0
}

fn bounding_box(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::MAX;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::MAX;
    let mut max_y = f64::NEG_INFINITY;
    for &(x, y) in poly {
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }
    (min_x, max_x, min_y, max_y)
}

fn point_in_polygon(px: f64, py: f64, poly: &[Pt]) -> bool {

        }

fn segments_intersect(a1: Pt, a2: Pt, b1: Pt, b2: Pt) -> bool {
    let cross = |o: Pt, a: Pt, b: Pt| -> f64 {
        (a.0 - o.0) * (b.1 - o.1) - (a.1 - o.1) * (b.0 - o.0)
    };
    
    let d1 = cross(b1, b2, a1);
    let d2 = cross(b1, b2, a2);
    let d3 = cross(a1, a2, b1);
    let d4 = cross(a1, a2, b2);
    
    if ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
       ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0)) {
        return true;
    }

    let eps = 1e-10;
    if d1.abs() < eps && on_segment(b1, a1, b2) { return true; }
    if d2.abs() < eps && on_segment(b1, a2, b2) { return true; }
    if d3.abs() < eps && on_segment(a1, b1, a2) { return true; }
    if d4.abs() < eps && on_segment(a1, b2, a2) { return true; }

}
