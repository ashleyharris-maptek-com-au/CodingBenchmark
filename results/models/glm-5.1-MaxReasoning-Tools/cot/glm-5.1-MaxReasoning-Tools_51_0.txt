```python
# Module-level state for PID controllers and persistence
_pid_state = {}

def _get_pid(name, kp, ki, kd, output_limit, integral_limit=None):
    """Get or create PID controller state"""
    if name not in _pid_state:
        _pid_state[name] = {
            'integral': 0.0,
            'prev_error': None,
            'kp': kp, 'ki': ki, 'kd': kd,
            'output_limit': output_limit,
            'integral_limit': integral_limit or output_limit
        }
    return _pid_state[name]

def _pid_update(name, error, dt, kp=None, ki=None, kd=None):
    """Update PID and return output"""
    pid = _get_pid(name, kp or 0, ki or 0, kd or 0, 1.0)
    if kp is not None: pid['kp'] = kp
    if ki is not None: pid['ki'] = ki
    if kd is not None: pid['kd'] = kd
    
    pid['integral'] += error * dt
    # Anti-windup
    if pid['integral_limit']:
        pid['integral'] = max(-pid['integral_limit'], min(pid['integral_limit'], pid['integral']))
    
    derivative = 0.0
    if pid['prev_error'] is not None:
        derivative = (error - pid['prev_error']) / dt
    pid['prev_error'] = error
    
    output = pid['kp'] * error + pid['ki'] * pid['integral'] + pid['kd'] * derivative
    return max(-pid['output_limit'], min(pid['output_limit'], output))
```

```python
def _robust_vote(values, valid_flags, tolerance=None):
    """
    Robust triple modular redundancy voting.
    Even if valid_flags say all valid, detect and reject outliers.
    """
    n = len(values)
    assert n == 3
    
    # If we have a tolerance and all values are within tolerance, just median
    if tolerance is not None:
        spread = max(values) - min(values)
        if spread <= tolerance:
            return sorted(values)[1]  # median
    
    # Try to find a pair that agrees
    pairs = [(0,1,2), (0,2,1), (1,2,0)]
    for i, j, k in pairs:
        if abs(values[i] - values[j]) <= (tolerance or float('inf')):
            # i and j agree, k is the outlier
            # If k is flagged invalid or i,j are flagged valid, good
            # Even if flags say otherwise, agreement beats flags
            return (values[i] + values[j]) / 2
    
    # No pair agrees - use flags to guide
    valid_vals = [values[i] for i in range(3) if valid_flags[i]]
    if valid_vals:
        return sum(valid_vals) / len(valid_vals)
    
    # All invalid and all disagree - return median as best guess
    return sorted(values)[1]
```

```python
    if tol is not None and len(valid_pairs) >= 2:
        # Try to find agreeing pair
        for i in range(len(valid_pairs)):
            for j in range(i+1, len(valid_pairs)):
                vi, ii = valid_pairs[i]
                vj, ij = valid_pairs[j]
                if abs(vi - vj) <= tol:
                    return (vi + vj) / 2, 0.9
        
        # No valid pair agrees - maybe one valid sensor is wrong
        # Try all pairs including invalid-flagged ones
        for i in range(3):
            for j in range(i+1, 3):
                if abs(values[i] - values[j]) <= tol:
                    return (values[i] + values[j]) / 2, 0.7
        
        # Nothing agrees - fall back to median of valid values
        if valid_pairs:
            v = sorted([p[0] for p in valid_pairs])
            return v[len(v)//2], 0.3
    
    if len(valid_pairs) >= 3:
        vals = sorted([p[0] for p in valid_pairs])
        return vals[1], 0.9
    elif len(valid_pairs) >= 2:
        return sum(p[0] for p in valid_pairs) / 2, 0.7
    elif len(valid_pairs) == 1:
        return valid_pairs[0][0], 0.3
    else:
        return _median3(values), 0.1

def _circular_vote3(values, valids, tol):
    """Vote on circular quantities like heading."""
    # Try to find a pair that agrees (considering wraparound)
    for i in range(3):
        for j in range(i+1, 3):
            diff = abs(values[i] - values[j])
            diff = min(diff, 360 - diff)  # Handle wraparound
            if diff <= tol:
                return _circular_avg(values[i], values[j]), 0.9
    
    # No pair agrees - median
    valid_vals = [values[i] for i in range(3) if valids[i]]
    if len(valid_vals) >= 2:
        return _circular_avg(valid_vals[0], valid_vals[1]), 0.5
    elif valid_vals:
        return valid_vals[0], 0.3
    return values[0], 0.1


def _circular_avg(a, b):
    """Average of two angles handling wraparound."""
    diff = b - a
    if diff > 180:
        diff -= 360
    elif diff < -180:
        diff += 360
    return (a + diff / 2) % 360
```

def _clamp(val, lo, hi):
    return max(lo, min(hi, val))

def _wrap_angle(angle):
    """Wrap angle to [-180, 180]"""
    while angle > 180: angle -= 360
    while angle < -180: angle += 360
    return angle

def _angle_diff(from_angle, to_angle):
    """Shortest angular difference from from_angle to to_angle"""
    diff = to_angle - from_angle
    while diff > 180: diff -= 360
    while diff < -180: diff += 360
    return diff

def _median3(a, b, c):
    """Median of 3 values."""
    if a <= b <= c or c <= b <= a: return b
    if b <= a <= c or c <= a <= b: return a
    return c

def _vote3(values, valids, tol=None):
    """
    Triple modular redundancy voting.
    Returns (best_value, confidence 0-1)
    """
    n = len(values)
    
    # If tolerance given, try to find agreeing pair
    if tol is not None:
        best_pair = None
        best_pair_quality = -1
        for i in range(n):
            for j in range(i+1, n):
                if abs(values[i] - values[j]) <= tol:
                    quality = 0
                    if valids[i]: quality += 1
                    if valids[j]: quality += 1
                    if quality > best_pair_quality:
                        best_pair = (values[i], values[j])
                        best_pair_quality = quality
        
        if best_pair is not None:
            return (best_pair[0] + best_pair[1]) / 2, min(1.0, 0.5 + 0.2 * best_pair_quality)
    
    # Use valid values
    valid_vals = [(values[i], i) for i in range(n) if valids[i]]
    
    if len(valid_vals) >= 3:
        v = sorted([p[0] for p in valid_vals])
        return v[1], 0.9
    elif len(valid_vals) >= 2:
        return (valid_vals[0][0] + valid_vals[1][0]) / 2, 0.6
    elif len(valid_vals) == 1:
        return valid_vals[0][0], 0.3
    
    # All invalid - use median as last resort
    return _median3(*values), 0.1


def _circular_vote3(values, valids, tol=10):
    """Vote on circular quantities (headings) considering wraparound."""
    n = len(values)
    
    if tol is not None:
        best_pair = None
        best_pair_quality = -1
        for i in range(n):
            for j in range(i+1, n):
                diff = abs(_angle_diff(values[i], values[j]))
                if diff <= tol:
                    quality = 0
                    if valids[i]: quality += 1
                    if valids[j]: quality += 1
                    if quality > best_pair_quality:
                        best_pair = (i, j)
                        best_pair_quality = quality

            i, j = best_pair
            # Average two angles
            avg = values[i] + _angle_diff(values[i], values[j]) / 2
            return avg % 360, min(1.0, 0.5 + 0.2 * best_pair_quality)
    
    # Fall back
    valid_vals = [(values[i], i) for i in range(n) if valids[i]]
    if valid_vals:
        return valid_vals[0][0], 0.3
    return values[0], 0.1

def _pid(name, error, dt, kp, ki=0, kd=0, ilimit=None, olimit=1.0):
    """
    PID controller with persistent state.
    name: unique identifier for this PID instance
    error: current error value
    dt: timestep
    kp, ki, kd: gains
    ilimit: integral anti-windup limit
    olimit: output limit
    Returns: control output
    """
    ikey = f'pid_{name}_i'
    pkey = f'pid_{name}_pe'
    
    integral = _get_p(ikey, 0.0)
    prev_err = _get_p(pkey, None)
    
    # Proportional
    p_term = kp * error
    
    # Integral
    integral += error * dt
    if ilimit is not None:
        integral = _clamp(integral, -ilimit, ilimit)
    i_term = ki * integral
    
    # Derivative
    d_term = 0.0
    if prev_err is not None and dt > 0:
        d_term = kd * (error - prev_err) / dt
    
    # Store state
    _set_p(ikey, integral)
    _set_p(pkey, error)
    
    output = p_term + i_term + d_term
    return _clamp(output, -olimit, olimit)
