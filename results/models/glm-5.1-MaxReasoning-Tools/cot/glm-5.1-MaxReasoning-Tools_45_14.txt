        name_bytes += b'\x00'
    num_name_words = len(name_bytes) // 4

```python
import struct

def float_to_word(f):
    return struct.unpack('<I', struct.pack('<f', f))[0]

class SPIRVBuilder:
    def __init__(self):
        self.instructions = []
        self.id_counter = 0
        self.bound = 0
    
    def next_id(self):
        self.id_counter += 1
        return self.id_counter
    
    def emit(self, word_count, opcode, *operands):
        self.instructions.append([(word_count << 16) | opcode] + list(operands))
    
    def finalize(self):
        self.bound = self.id_counter + 1
        all_words = [
            0x07230203,  # Magic
            0x00010000,  # Version 1.0
            0x00000000,  # Generator
            self.bound,
            0x00000000,  # Reserved
        ]
        for instr in self.instructions:
            all_words.extend(instr)
        return all_words
    
    def to_hex(self):
        words = self.finalize()
        result = b''
        for w in words:
            result += struct.pack('<I', w)
        return result.hex()
```
