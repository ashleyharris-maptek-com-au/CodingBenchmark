import struct

class SPIRVBuilder:
    def __init__(self):
        self.words = []
        self.next_id = 1
        self.id_bound = 1  # will be updated
        
    def alloc_id(self):
        id = self.next_id
        self.next_id += 1
        self.id_bound = self.next_id
        return id
    
    def emit(self, *args):
        for a in args:
            self.words.append(a)
    
    def instruction(self, word_count, opcode, *operands):
        self.emit((word_count << 16) | opcode, *operands)
    
    def finalize(self):
        # Update header bound
        self.words[3] = self.id_bound
        return self.words

import struct

class SPIRVBuilder:
    def __init__(self):
        self.words = []
        self.next_id = 1
        
    def alloc_id(self):
        id_val = self.next_id
        self.next_id += 1
        return id_val
    
    def emit(self, *args):
        for a in args:
            self.words.append(a)
    
    def instr(self, word_count, opcode, *operands):
        self.emit((word_count << 16) | opcode, *operands)
    
    def encode_string(self, s):
        encoded = s.encode('ascii') + b'\x00'
        while len(encoded) % 4 != 0:
            encoded += b'\x00'
        words = []
        for i in range(0, len(encoded), 4):
            words.append(struct.unpack('<I', encoded[i:i+4])[0])
        return words
    
    def float_to_word(self, f):
        return struct.unpack('<I', struct.pack('<f', f))[0]
    
    def finalize(self):
        # Update bound in header
        self.words[3] = self.next_id
        # Convert to bytes
        result = b''
        for w in self.words:
            result += struct.pack('<I', w)
        return result.hex()
