#include <iostream>
#include <string>
#include <vector>
#include <unordered_set>
#include <algorithm>
#include <cstdint>

// Rolling hash with two moduli
// Mod1 = 2^61 - 1 (Mersenne prime)
// Mod2 = 10^9 + 7 or similar

// Actually, let me use 64-bit hashes with different bases and rely on unsigned overflow (mod 2^64)
// This is simpler and faster, though less collision-resistant.
// For safety, I'll use two independent 64-bit hashes.

struct HashPair {
    uint64_t h1, h2;
    bool operator==(const HashPair& o) const {
        return h1 == o.h1 && h2 == o.h2;
    }
};

struct HashPairHash {
    size_t operator()(const HashPair& hp) const {
        return hp.h1 ^ (hp.h2 * 0x9e3779b97f4a7c15ULL);
    }
};

using HashSet = std::unordered_set<HashPair, HashPairHash>;

// Compute all rolling hashes of length L for string s
std::vector<HashPair> computeHashes(const std::string& s, int L, uint64_t base1, uint64_t base2) {
    if ((int)s.size() < L) return {};
    std::vector<HashPair> hashes;
    hashes.reserve(s.size() - L + 1);
    
    uint64_t h1 = 0, h2 = 0;
    uint64_t pw1 = 1, pw2 = 1;
    for (int i = 0; i < L; i++) {
        h1 = h1 * base1 + (unsigned char)s[i];
        h2 = h2 * base2 + (unsigned char)s[i];
        if (i < L - 1) {
            pw1 *= base1;
            pw2 *= base2;
        }
    }
    hashes.push_back({h1, h2});
    
    for (int i = L; i < (int)s.size(); i++) {
        h1 = (h1 - (unsigned char)s[i - L] * pw1) * base1 + (unsigned char)s[i];
        h2 = (h2 - (unsigned char)s[i - L] * pw2) * base2 + (unsigned char)s[i];
        hashes.push_back({h1, h2});
    }
    
    return hashes;
}

bool checkLength(int L, const std::vector<std::string>& strings, int minIdx, uint64_t base1, uint64_t base2) {
    // Compute hashes for shortest string
    auto candidates = computeHashes(strings[minIdx], L, base1, base2);
    if (candidates.empty()) return false;
    
    HashSet candSet;
    candSet.reserve(candidates.size() * 2);
    for (auto& hp : candidates) {
        candSet.insert(hp);
    }
    
    for (int i = 0; i < (int)strings.size(); i++) {
        if (i == minIdx) continue;
        auto hashes = computeHashes(strings[i], L, base1, base2);
        HashSet curSet;
        curSet.reserve(hashes.size() * 2);
        for (auto& hp : hashes) {
            curSet.insert(hp);
        }
        
        // Filter candSet to only include elements in curSet
        HashSet newCand;
        newCand.reserve(candSet.size() * 2);
        for (auto& hp : candSet) {
            if (curSet.count(hp)) {
                newCand.insert(hp);
            }
        }
        candSet = std::move(newCand);
        
        if (candSet.empty()) return false;
    }
    
    return !candSet.empty();
}

class FlatHashSet {
    vector<uint64_t> table;
    int mask;
    static const uint64_t EMPTY = UINT64_MAX;
    
public:
    FlatHashSet(int capacity) {
        int sz = 1;
        while (sz < capacity * 2) sz *= 2;
        table.assign(sz, EMPTY);
        mask = sz - 1;
    }
    
    void insert(uint64_t val) {
        int idx = val & mask;
        while (table[idx] != EMPTY) {
            if (table[idx] == val) return;
            idx = (idx + 1) & mask;
        }
        table[idx] = val;
    }
    
    bool contains(uint64_t val) const {
        int idx = val & mask;
        while (table[idx] != EMPTY) {
            if (table[idx] == val) return true;
            idx = (idx + 1) & mask;
        }
        return false;
    }
};

// Rolling hash with two bases, mod 2^64
const uint64_t BASE1 = 911382629ULL;
const uint64_t BASE2 = 972663749ULL;

struct HashPair {
    uint64_t h1, h2;
    bool operator==(const HashPair& o) const { return h1 == o.h1 && h2 == o.h2; }
};

struct HashPairHash {
    size_t operator()(const HashPair& hp) const {
        return (size_t)(hp.h1 ^ (hp.h2 * 0x9e3779b97f4a7c15ULL + 0x9e3779b97f4a7c15ULL + (hp.h1 << 6) + (hp.h1 >> 2)));
    }
};

using HashSet = unordered_set<HashPair, HashPairHash>;

// Compute all rolling hashes of length L for string s using two bases
void computeHashes(const string& s, int L, vector<HashPair>& out) {
    out.clear();
    int n = (int)s.size();
    if (n < L) return;
    out.reserve(n - L + 1);
    
    uint64_t h1 = 0, h2 = 0;
    uint64_t pw1 = 1, pw2 = 1;
    for (int i = 0; i < L; i++) {
        h1 = h1 * BASE1 + (unsigned char)s[i];
        h2 = h2 * BASE2 + (unsigned char)s[i];
        pw1 *= BASE1;
        pw2 *= BASE2;
    }
    out.push_back({h1, h2});
    
    for (int i = L; i < n; i++) {
        h1 = h1 * BASE1 - (unsigned char)s[i - L] * pw1 + (unsigned char)s[i];
        h2 = h2 * BASE2 - (unsigned char)s[i - L] * pw2 + (unsigned char)s[i];
        out.push_back({h1, h2});
    }
}

    candSet.reserve(candidates.size() * 2);

        candSet.insert(hp);
    }

    }

}
