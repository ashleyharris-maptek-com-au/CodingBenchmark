                                  const unordered_set<uint64_t>& blocked,
                                  const unordered_set<uint64_t>& food,
                                  int max_nodes) {
    uint64_t start_code = encode(start);
    
    unordered_map<uint64_t, uint64_t> parent;
    unordered_set<uint64_t> visited;
    queue<uint64_t> q;
    
    visited.insert(start_code);
    q.push(start_code);
    
    int nodes = 0;
    
    while (!q.empty() && nodes < max_nodes) {
        uint64_t code = q.front();
        q.pop();
        nodes++;
        
        vector<int> pos = decode(code);
        
        for (int axis = 0; axis < D; axis++) {
            for (int dir = -1; dir <= 1; dir += 2) {
                vector<int> next = pos;
                next[axis] += dir;
                if (!in_bounds(next)) continue;
                
                uint64_t next_code = encode(next);
                if (blocked.count(next_code)) continue;
                if (visited.count(next_code)) continue;
                
                visited.insert(next_code);

                if (food.count(next_code)) {

                    uint64_t cur = next_code;
                    while (parent[cur] != start_code) {
                        cur = parent[cur];

                            return {a, first_step[a] > start[a] ? 1 : -1};
                        }
                    }
                }
                
                q.push(next_code);
            }
        }
    }
    
    return {-1, -1};
}
