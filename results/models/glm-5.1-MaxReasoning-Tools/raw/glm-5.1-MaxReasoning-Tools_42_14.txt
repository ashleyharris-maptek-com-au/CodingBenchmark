               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inColor %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inWorldPos Location 0
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               OpExtInstImport %GLSLstd450 "GLSL.std.450"

        %void = OpTypeVoid
     %void_fn = OpTypeFunction %void
       %float = OpTypeFloat 32
     %v3float = OpTypeVector %float 3
     %v4float = OpTypeVector %float 4
         %int = OpTypeInt 32 1
       %int_0 = OpConstant %int 0
       %int_2 = OpConstant %int 2
       %int_4 = OpConstant %int 4
        %uint = OpTypeInt 32 0
       %uint_6 = OpConstant %uint 6

%gl_PerVertex_in = OpTypeStruct %v4float
   %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
 %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
        %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
  %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
       %gl_out = OpVariable %ptr_out_pv Output

   %arr_v3_6 = OpTypeArray %v3float %uint_6
 %ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
  %inWorldPos = OpVariable %ptr_in_arr_v3 Input
     %inColor = OpVariable %ptr_in_arr_v3 Input

  %ptr_out_v3 = OpTypePointer Output %v3float
    %outColor = OpVariable %ptr_out_v3 Output

   %ptr_in_v4 = OpTypePointer Input %v4float
   %ptr_in_v3 = OpTypePointer Input %v3float
  %ptr_out_v4 = OpTypePointer Output %v4float

     %float_0 = OpConstant %float 0.0
   %float_0_2 = OpConstant %float 0.2
   %float_0_8 = OpConstant %float 0.8
     %float_1 = OpConstant %float 1.0
   %float_1_3 = OpConstant %float 0.333333334
   %v3_light = OpConstantComposite %v3float %float_1 %float_1 %float_1

        %main = OpFunction %void None %void_fn
      %entry = OpLabel

; --- Load world positions for main triangle vertices (0, 2, 4) ---
   %wp0_ptr = OpAccessChain %ptr_in_v3 %inWorldPos %int_0
       %wp0 = OpLoad %v3float %wp0_ptr
   %wp2_ptr = OpAccessChain %ptr_in_v3 %inWorldPos %int_2
       %wp2 = OpLoad %v3float %wp2_ptr
   %wp4_ptr = OpAccessChain %ptr_in_v3 %inWorldPos %int_4
       %wp4 = OpLoad %v3float %wp4_ptr

; --- Edge vectors ---
    %edge1 = OpFSub %v3float %wp2 %wp0
    %edge2 = OpFSub %v3float %wp4 %wp0

; --- Manual cross product: cross(edge1, edge2) ---
    %e1_x = OpCompositeExtract %float %edge1 0
    %e1_y = OpCompositeExtract %float %edge1 1
    %e1_z = OpCompositeExtract %float %edge1 2
    %e2_x = OpCompositeExtract %float %edge2 0
    %e2_y = OpCompositeExtract %float %edge2 1
    %e2_z = OpCompositeExtract %float %edge2 2

    %cx_a = OpFMul %float %e1_y %e2_z
    %cx_b = OpFMul %float %e1_z %e2_y
       %cx = OpFSub %float %cx_a %cx_b

    %cy_a = OpFMul %float %e1_z %e2_x
    %cy_b = OpFMul %float %e1_x %e2_z
       %cy = OpFSub %float %cy_a %cy_b

    %cz_a = OpFMul %float %e1_x %e2_y
    %cz_b = OpFMul %float %e1_y %e2_x
       %cz = OpFSub %float %cz_a %cz_b

    %cross = OpCompositeConstruct %v3float %cx %cy %cz

; --- Normalize face normal ---
         %N = OpExtInst %v3float %GLSLstd450 69 %cross

; --- Light direction L = normalize(vec3(1,1,1)) ---
         %L = OpExtInst %v3float %GLSLstd450 69 %v3_light

; --- Diffuse = max(dot(N, L), 0.0) ---
    %dot_NL = OpDot %float %N %L
    %diffuse = OpExtInst %float %GLSLstd450 40 %dot_NL %float_0

; --- Average color from indices 0, 2, 4 ---
    %c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
        %c0 = OpLoad %v3float %c0_ptr
    %c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
        %c2 = OpLoad %v3float %c2_ptr
    %c4_ptr = OpAccessChain %ptr_in_v3 %inColor %int_4
        %c4 = OpLoad %v3float %c4_ptr

   %c_sum02 = OpFAdd %v3float %c0 %c2
    %c_sum = OpFAdd %v3float %c_sum02 %c4
 %baseColor = OpVectorTimesScalar %v3float %c_sum %float_1_3

; --- Final color = base * (0.2 + 0.8 * diffuse) ---
 %diff_term = OpFMul %float %float_0_8 %diffuse
    %light_f = OpFAdd %float %float_0_2 %diff_term
   %light_v3 = OpCompositeConstruct %v3float %light_f %light_f %light_f
  %finalColor = OpFMul %v3float %baseColor %light_v3

; --- Output position pointer (reused for all emitted vertices) ---
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; --- Emit vertex 0 ---
   %pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
       %pos0 = OpLoad %v4float %pos0_ptr
               OpStore %out_pos_ptr %pos0
               OpStore %outColor %finalColor
               OpEmitVertex

; --- Emit vertex 2 ---
   %pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
       %pos2 = OpLoad %v4float %pos2_ptr
               OpStore %out_pos_ptr %pos2
               OpStore %outColor %finalColor
               OpEmitVertex

; --- Emit vertex 4 ---
   %pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
       %pos4 = OpLoad %v4float %pos4_ptr
               OpStore %out_pos_ptr %pos4
               OpStore %outColor %finalColor
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
