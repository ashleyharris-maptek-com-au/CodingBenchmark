; SPIR-V 1.0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func_type = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0_04 = OpConstant %float 0.04
%float_n0_04 = OpConstant %float -0.04
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_input_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_in
%inWorldPos = OpVariable %ptr_input_arr_v3 Input
%inNormal = OpVariable %ptr_input_arr_v3 Input
%inColor = OpVariable %ptr_input_arr_v3 Input

%ptr_output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3 Output

%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_input_v3 = OpTypePointer Input %v3float

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load gl_in[0].gl_Position
%pos_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1

; Load color[0]
%color_ptr = OpAccessChain %ptr_input_v3 %inColor %int_0
%color = OpLoad %v3float %color_ptr

; Compute corner offsets
%xmh = OpFAdd %float %x %float_n0_04
%xph = OpFAdd %float %x %float_0_04
%ymh = OpFAdd %float %y %float_n0_04
%yph = OpFAdd %float %y %float_0_04

; Vertex 0: bottom-left (x-h, y-h)
%v0 = OpCompositeConstruct %v4float %xmh %ymh %float_0 %float_1
%gl_pos_0 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %gl_pos_0 %v0
OpStore %outColor %color
OpEmitVertex

; Vertex 1: bottom-right (x+h, y-h)
%v1 = OpCompositeConstruct %v4float %xph %ymh %float_0 %float_1
%gl_pos_1 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %gl_pos_1 %v1
OpStore %outColor %color
OpEmitVertex

; Vertex 2: top-left (x-h, y+h)
%v2 = OpCompositeConstruct %v4float %xmh %yph %float_0 %float_1
%gl_pos_2 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %gl_pos_2 %v2
OpStore %outColor %color
OpEmitVertex

; Vertex 3: top-right (x+h, y+h)
%v3 = OpCompositeConstruct %v4float %xph %yph %float_0 %float_1
%gl_pos_3 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %gl_pos_3 %v3
OpStore %outColor %color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
