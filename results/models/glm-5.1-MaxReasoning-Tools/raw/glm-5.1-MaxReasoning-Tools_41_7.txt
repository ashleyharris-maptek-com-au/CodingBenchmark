```spirv
; SPIR-V
; Version: 1.0
OpCapability Shader
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %in_worldPos "in.worldPos"
OpName %in_normal "in.normal"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBO "UBO"

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_3 = OpConstant %float 3

%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%vec3_base = OpConstantComposite %vec3 %float_0_2 %float_0_3 %float_0_8
%vec3_rim_tint = OpConstantComposite %vec3 %float_1 %float_0_8 %float_0_5

%main = OpFunction %void None %void_fn
%entry = OpLabel

%worldPos = OpLoad %vec3 %in_worldPos
%normal_raw = OpLoad %vec3 %in_normal

%N = OpExtInst %vec3 %ext Normalize %normal_raw

%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_4
%lightPos_v4 = OpLoad %vec4 %lightPos_ptr
%cameraPos_v4 = OpLoad %vec4 %cameraPos_ptr

%lightPos = OpVectorShuffle %vec3 %lightPos_v4 %lightPos_v4 0 1 2
%cameraPos = OpVectorShuffle %vec3 %cameraPos_v4 %cameraPos_v4 0 1 2

%toLight = OpFSub %vec3 %lightPos %worldPos
%L = OpExtInst %vec3 %ext Normalize %toLight

%toCamera = OpFSub %vec3 %cameraPos %worldPos
%V = OpExtInst %vec3 %ext Normalize %toCamera

%NdL = OpDot %float %N %L
%diffuse = OpExtInst %float %ext FMax %NdL %float_0

%NdV = OpDot %float %N %V
%NdV_clamped = OpExtInst %float %ext FMax %NdV %float_0

%one_minus_NdV = OpFSub %float %float_1 %NdV_clamped
%rim = OpExtInst %float %ext Pow %one_minus_NdV %float_3

%diffuse_scaled = OpFMul %float %float_0_5 %diffuse
%modulation = OpFAdd %float %float_0_3 %diffuse_scaled

%base = OpVectorTimesScalar %vec3 %vec3_base %modulation
%rim_contrib = OpVectorTimesScalar %vec3 %vec3_rim_tint %rim

%final_rgb = OpFAdd %vec3 %base %rim_contrib
%final_rgba = OpCompositeConstruct %vec4 %final_rgb %float_1

OpStore %outColor %final_rgba
OpReturn
OpFunctionEnd
```
