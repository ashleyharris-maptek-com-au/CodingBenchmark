; SPIR-V 1.0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_1 = OpConstant %float 1.0
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005

; Input gl_PerVertex block (array of 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Input color array (location 2)
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%inColor = OpVariable %ptr_in_arr_v3 Input

; Output color (location 0)
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointer types for access chains
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %fn_void
%label_entry = OpLabel

; ---- Load gl_in[0].gl_Position and gl_in[1].gl_Position ----
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr

; ---- Load per-vertex colors ----
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

; ---- Compute perpendicular direction in NDC space ----
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p0_zw = OpVectorShuffle %v2float %p0 %p0 2 3
%p1_zw = OpVectorShuffle %v2float %p1 %p1 2 3
%p0_w = OpCompositeExtract %float %p0 3
%p1_w = OpCompositeExtract %float %p1 3

; NDC positions: xy / w
%p0_w_inv = OpFDiv %float %float_1 %p0_w
%p1_w_inv = OpFDiv %float %float_1 %p1_w
%p0_ndc = OpVectorTimesScalar %v2float %p0_xy %p0_w_inv
%p1_ndc = OpVectorTimesScalar %v2float %p1_xy %p1_w_inv

; Direction and perpendicular
%dir = OpFSub %v2float %p1_ndc %p0_ndc
%dir_len = OpLength %float %dir
%dir_len_inv = OpFDiv %float %float_1 %dir_len
%dir_norm = OpVectorTimesScalar %v2float %dir %dir_len_inv

%dn_x = OpCompositeExtract %float %dir_norm 0
%dn_y = OpCompositeExtract %float %dir_norm 1
%neg_dn_y = OpFNegate %float %dn_y
%perp = OpCompositeConstruct %v2float %neg_dn_y %dn_x

; ---- Clip-space offsets: perp * thickness * w ----
%perp_start = OpVectorTimesScalar %v2float %perp %float_0_04
%offset_start = OpVectorTimesScalar %v2float %perp_start %p0_w

%perp_end = OpVectorTimesScalar %v2float %perp %float_0_005
%offset_end = OpVectorTimesScalar %v2float %perp_end %p1_w

; ---- Compute 4 ribbon vertices ----
; Vertex 0: p0 + perp * thick_start
%v0_xy = OpFAdd %v2float %p0_xy %offset_start
%v0 = OpVectorShuffle %v4float %v0_xy %p0_zw 0 1 2 3

; Vertex 1: p0 - perp * thick_start
%v1_xy = OpFSub %v2float %p0_xy %offset_start
%v1 = OpVectorShuffle %v4float %v1_xy %p0_zw 0 1 2 3

; Vertex 2: p1 + perp * thick_end
%v2_xy = OpFAdd %v2float %p1_xy %offset_end
%v2 = OpVectorShuffle %v4float %v2_xy %p1_zw 0 1 2 3

; Vertex 3: p1 - perp * thick_end
%v3_xy = OpFSub %v2float %p1_xy %offset_end
%v3 = OpVectorShuffle %v4float %v3_xy %p1_zw 0 1 2 3

; ---- Output position pointer (reused for all emits) ----
%gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- Emit vertex 0: p0 + perp, color c0 ----
OpStore %gl_out_pos %v0
OpStore %outColor %c0
OpEmitVertex

; ---- Emit vertex 1: p0 - perp, color c0 ----
OpStore %gl_out_pos %v1
OpStore %outColor %c0
OpEmitVertex

; ---- Emit vertex 2: p1 + perp, color c1 ----
OpStore %gl_out_pos %v2
OpStore %outColor %c1
OpEmitVertex

; ---- Emit vertex 3: p1 - perp, color c1 ----
OpStore %gl_out_pos %v3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
