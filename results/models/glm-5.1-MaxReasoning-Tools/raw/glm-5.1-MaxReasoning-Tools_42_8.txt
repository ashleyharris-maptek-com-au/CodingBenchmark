; SPIR-V
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations for gl_PerVertex input block
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for gl_PerVertex output block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; User-defined I/O locations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0_7 = OpConstant %float 0.7
%float_1_over_3 = OpConstant %float 0.333333343

; Input gl_PerVertex struct and array of 3
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

; Output gl_PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

; User input arrays (vec3[3])
%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
%ptr_in_v3 = OpTypePointer Input %v3float

; User output
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

; --- Load gl_Position for each input vertex ---
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

; --- Compute centroid: C = (p0 + p1 + p2) / 3 ---
%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%centroid = OpVectorTimesScalar %v4float %sum012 %float_1_over_3

; --- Emit vertex 0: new_pos = C + s * (p0 - C) ---
%delta0 = OpFSub %v4float %p0 %centroid
%scaled0 = OpVectorTimesScalar %v4float %delta0 %float_0_7
%newpos0 = OpFAdd %v4float %centroid %scaled0
%outPos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outPos0_ptr %newpos0
%color0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%color0 = OpLoad %v3float %color0_ptr
OpStore %outColor %color0
OpEmitVertex

; --- Emit vertex 1: new_pos = C + s * (p1 - C) ---
%delta1 = OpFSub %v4float %p1 %centroid
%scaled1 = OpVectorTimesScalar %v4float %delta1 %float_0_7
%newpos1 = OpFAdd %v4float %centroid %scaled1
%outPos1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outPos1_ptr %newpos1
%color1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%color1 = OpLoad %v3float %color1_ptr
OpStore %outColor %color1
OpEmitVertex

; --- Emit vertex 2: new_pos = C + s * (p2 - C) ---
%delta2 = OpFSub %v4float %p2 %centroid
%scaled2 = OpVectorTimesScalar %v4float %delta2 %float_0_7
%newpos2 = OpFAdd %v4float %centroid %scaled2
%outPos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outPos2_ptr %newpos2
%color2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%color2 = OpLoad %v3float %color2_ptr
OpStore %outColor %color2
OpEmitVertex

; --- Complete the triangle strip ---
OpEndPrimitive

OpReturn
OpFunctionEnd
