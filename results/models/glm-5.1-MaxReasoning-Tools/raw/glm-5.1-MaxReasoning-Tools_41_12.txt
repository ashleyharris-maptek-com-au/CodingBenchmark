; SPIR-V
; Version: 1.0
; Generator: 0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4v4 = OpTypeMatrix %vec4 4
%uint = OpTypeInt 32 0
%UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %vec4 %vec4 %vec4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%float_0 = OpConstant %float 0
%float_0_05 = OpConstant %float 0.05
%float_0_15 = OpConstant %float 0.15
%float_0_3 = OpConstant %float 0.3
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_1 = OpConstant %float 1
%uint_3 = OpConstant %uint 3

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %void_fn
%entry = OpLabel

%n_raw = OpLoad %vec3 %in_normal
%N = OpNormalize %vec3 %n_raw

%Ny = OpCompositeExtract %float %N 1
%Ny_half = OpFMul %float %Ny %float_0_5
%blend = OpFAdd %float %Ny_half %float_0_5

%one_minus_blend = OpFSub %float %float_1 %blend

%ground_r = OpFMul %float %float_0_3 %one_minus_blend
%ground_g = OpFMul %float %float_0_15 %one_minus_blend
%ground_b = OpFMul %float %float_0_05 %one_minus_blend

%sky_r = OpFMul %float %float_0_4 %blend
%sky_g = OpFMul %float %float_0_6 %blend
%sky_b = OpFMul %float %float_1 %blend

%amb_r = OpFAdd %float %ground_r %sky_r
%amb_g = OpFAdd %float %ground_g %sky_g
%amb_b = OpFAdd %float %ground_b %sky_b

%wp = OpLoad %vec3 %in_worldPos
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %uint_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lp_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

%toLight = OpFSub %vec3 %lp_xyz %wp
%L = OpNormalize %vec3 %toLight

%NdL = OpDot %float %N %L
%d = OpFMax %float %NdL %float_0

%diffuse = OpFMul %float %float_0_3 %d

%final_r = OpFAdd %float %amb_r %diffuse
%final_g = OpFAdd %float %amb_g %diffuse
%final_b = OpFAdd %float %amb_b %diffuse

%result = OpCompositeConstruct %vec4 %final_r %final_g %final_b %float_1

OpStore %outColor %result
OpReturn
OpFunctionEnd
