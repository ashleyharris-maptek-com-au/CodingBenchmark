; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_pi = OpConstant %float 3.14159
%float_16 = OpConstant %float 16.0
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_9 = OpConstant %float 0.9
%float_0_1 = OpConstant %float 0.1

%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo_var = OpVariable %ptr_Uniform_UBO Uniform

%main = OpFunction %void None %void_fn
%entry = OpLabel

%uv_val = OpLoad %v2float %in_uv
%normal_val = OpLoad %v3float %in_normal
%worldPos_val = OpLoad %v3float %in_worldPos

%norm_n = OpExtInst %v3float %1 Normalize %normal_val

%lightPos_ptr = OpAccessChain %ptr_Uniform_v4float %ubo_var %int_3
%lightPos_val = OpLoad %v4float %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %v3float %lightPos_val %lightPos_val 0 1 2

%to_light = OpFSub %v3float %lightPos_xyz %worldPos_val
%light_dir = OpExtInst %v3float %1 Normalize %to_light

%NdL = OpDot %float %norm_n %light_dir
%diffuse = OpFMax %float %NdL %float_0

%temp = OpFMul %float %float_0_6 %diffuse
%shading = OpFAdd %float %float_0_4 %temp

%uv_y = OpCompositeExtract %float %uv_val 1
%uv_y_pi = OpFMul %float %uv_y %float_pi
%angle = OpFMul %float %uv_y_pi %float_16
%sin_val = OpExtInst %float %1 Sin %angle

%is_red = OpFOrdGreaterThanEqual %bool %sin_val %float_0
%red = OpCompositeConstruct %v3float %float_0_9 %float_0_1 %float_0_1
%blue = OpCompositeConstruct %v3float %float_0_1 %float_0_1 %float_0_9
%stripe = OpSelect %v3float %is_red %red %blue

%shading3 = OpCompositeConstruct %v3float %shading %shading %shading
%final_rgb = OpFMul %v3float %stripe %shading3

%fr = OpCompositeExtract %float %final_rgb 0
%fg = OpCompositeExtract %float %final_rgb 1
%fb = OpCompositeExtract %float %final_rgb 2
%out_vec = OpCompositeConstruct %v4float %fr %fg %fb %float_1

OpStore %outColor %out_vec
OpReturn
OpFunctionEnd
