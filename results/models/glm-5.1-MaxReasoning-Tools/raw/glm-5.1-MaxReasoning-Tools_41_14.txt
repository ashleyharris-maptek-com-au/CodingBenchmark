%fnType = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1

%float_1_5 = OpConstant %float 1.5
%float_3_0 = OpConstant %float 3.0

%inputVec3Ptr = OpTypePointer Input %vec3
%outputVec4Ptr = OpTypePointer Output %vec4
%uniformVec4Ptr = OpTypePointer Uniform %vec4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%uboPtr = OpTypePointer Uniform %UBO

%inWorldPos = OpVariable %inputVec3Ptr Input
%inNormal = OpVariable %inputVec3Ptr Input
%outColor = OpVariable %outputVec4Ptr Output
%uboVar = OpVariable %uboPtr Uniform

%main = OpFunction %void None %fnType
%label = OpLabel

%worldPos = OpLoad %vec3 %inWorldPos
%normalRaw = OpLoad %vec3 %inNormal
%normal = OpNormalize %vec3 %normalRaw

%lightPosPtr = OpAccessChain %uniformVec4Ptr %uboVar %int_3
%lightPosV4 = OpLoad %vec4 %lightPosPtr
%lightPos = OpVectorShuffle %vec3 %lightPosV4 %lightPosV4 0 1 2

%cameraPosPtr = OpAccessChain %uniformVec4Ptr %uboVar %int_4
%cameraPosV4 = OpLoad %vec4 %cameraPosPtr
%cameraPos = OpVectorShuffle %vec3 %cameraPosV4 %cameraPosV4 0 1 2

%toLight = OpFSub %vec3 %lightPos %worldPos
%lightDir = OpNormalize %vec3 %toLight

%NdotL = OpDot %float %normal %lightDir
%diffuse = OpFMax %float %NdotL %float_0

%term = OpFMul %float %float_0_7 %diffuse
%factor = OpFAdd %float %float_0_3 %term

%baseR = OpFMul %float %float_0_8 %factor
%baseG = OpFMul %float %float_0_2 %factor
%baseB = OpFMul %float %float_0_2 %factor
%litColor = OpCompositeConstruct %vec3 %baseR %baseG %baseB

%toFrag = OpFSub %vec3 %worldPos %cameraPos
%dist = OpLength %float %toFrag

%distMinus = OpFSub %float %dist %float_1_5
%fogRaw = OpFDiv %float %distMinus %float_3_0
%fogFactor = OpFClamp %float %fogRaw %float_0 %float_1

%litR = OpCompositeExtract %float %litColor 0
%litG = OpCompositeExtract %float %litColor 1
%litB = OpCompositeExtract %float %litColor 2

%invFog = OpFSub %float %float_1 %fogFactor

%mixRa = OpFMul %float %litR %invFog
%mixRb = OpFMul %float %float_0_7 %fogFactor
%outR = OpFAdd %float %mixRa %mixRb

%mixGa = OpFMul %float %litG %invFog
%mixGb = OpFMul %float %float_0_7 %fogFactor
%outG = OpFAdd %float %mixGa %mixGb

%mixBa = OpFMul %float %litB %invFog
%mixBb = OpFMul %float %float_0_8 %fogFactor
%outB = OpFAdd %float %mixBa %mixBb

%finalColor = OpCompositeConstruct %vec4 %outR %outG %outB %float_1

OpReturn
