; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 16
OpMemberDecorate %UBO 2 Offset 32
OpMemberDecorate %UBO 3 Offset 48
OpMemberDecorate %UBO 4 Offset 64
OpMemberDecorate %UBO 5 Offset 80
OpMemberDecorate %UBO 6 Offset 96
OpMemberDecorate %UBO 7 Offset 112
OpMemberDecorate %UBO 8 Offset 128
OpMemberDecorate %UBO 9 Offset 144
OpMemberDecorate %UBO 10 Offset 160
OpMemberDecorate %UBO 11 Offset 176
OpMemberDecorate %UBO 12 Offset 192
OpMemberDecorate %UBO 13 Offset 208
OpMemberDecorate %UBO 14 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%UBO = OpTypeStruct %vec4 %vec4 %vec4 %vec4 %vec4 %vec4 %vec4 %vec4 %vec4 %vec4 %vec4 %vec4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO

%void_fn = OpTypeFunction %void

%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo_var = OpVariable %ptr_uniform_ubo Uniform

%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9
%float_1 = OpConstant %float 1.0
%float_8 = OpConstant %float 8.0
%uint_0 = OpConstant %uint 0
%uint_2 = OpConstant %uint 2
%uint_12 = OpConstant %uint 12

%glsl450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %void_fn
%entry = OpLabel

%uv = OpLoad %vec2 %in_uv
%uv8 = OpVectorTimesScalar %vec2 %uv %float_8
%floor_uv = OpExtInst %vec2 %glsl450 Floor %uv8
%fx = OpCompositeExtract %float %floor_uv 0
%fy = OpCompositeExtract %float %floor_uv 1
%ix = OpConvertFToU %uint %fx
%iy = OpConvertFToU %uint %fy
%sum = OpIAdd %uint %ix %iy
%parity = OpUMod %uint %sum %uint_2
%is_even = OpIEqual %bool %parity %uint_0
%checker = OpSelect %float %is_even %float_0_9 %float_0_2

%wp = OpLoad %vec3 %in_worldPos
%n_raw = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %glsl450 Normalize %n_raw

%lp_ptr = OpAccessChain %ptr_uniform_vec4 %ubo_var %uint_12
%lp_v4 = OpLoad %vec4 %lp_ptr
%lp = OpVectorShuffle %vec3 %lp_v4 %lp_v4 0 1 2

%ld = OpFSub %vec3 %lp %wp
%L = OpExtInst %vec3 %glsl450 Normalize %ld

%NdotL = OpDot %float %N %L
%diffuse = OpFMax %float %NdotL %float_0

%half_diff = OpFMul %float %float_0_5 %diffuse
%shade = OpFAdd %float %float_0_5 %half_diff

%cr = OpFMul %float %checker %shade
%cg = OpFMul %float %checker %shade
%cb = OpFMul %float %checker %shade

%result = OpCompositeConstruct %vec4 %cr %cg %cb %float_1
OpStore %outColor %result

OpReturn
OpFunctionEnd
