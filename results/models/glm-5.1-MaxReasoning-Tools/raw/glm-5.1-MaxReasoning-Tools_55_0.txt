void orderMv(const Pos& p, vector<Move>& mv, int ply, const Move& ttm) {
    size_t start = 0;

        for (size_t i = 0; i < mv.size(); i++)
            if (mv[i] == ttm) { swap(mv[0], mv[i]); break; }
        start = 1;
    }
    for (size_t i = start; i < mv.size(); i++) {
        size_t bi = i; int bs = moveSc(p, mv[i], ply);
        for (size_t j = i + 1; j < mv.size(); j++) { int s = moveSc(p, mv[j], ply); if (s > bs) { bs = s; bi = j; } }
        if (bi != i) swap(mv[i], mv[bi]);
    }
}

            if (tt[ti].fl == TT_AL && ts <= a) return a;
            if (tt[ti].fl == TT_BE && ts >= b) return b;
        }
    }

    if (nullOk && !chk && depth >= 3 && ply > 0) {
        Pos np = p; np.side = (Color)(1 - np.side); np.ep = -1;
        int ns = -abSearch(np, depth - 3, -b, -b + 1, ply + 1, false);
        if (stopS.load()) return 0;
        if (ns >= b) return b;
    }

    auto mv = genPseudo(p);
    orderMv(p, mv, ply, ttm);

    Move bestMv{}; int bestSc = -99999, origA = a;
    bool hasLegal = false;

    for (size_t i = 0; i < mv.size(); i++) {
        Pos np = doMove(p, mv[i]);
        if (inCheck(np, p.side)) continue;
        hasLegal = true;
        int sc;
        if (i >= 4 && depth >= 3 && !chk && !isCap(mv[i].flags) && !isPromo(mv[i].flags) && !inCheck(np, np.side)) {
            sc = -abSearch(np, depth - 2, -a - 1, -a, ply + 1, true);
            if (sc > a) sc = -abSearch(np, depth - 1, -b, -a, ply + 1, true);
        }
        else if (i > 0) {
            sc = -abSearch(np, depth - 1, -a - 1, -a, ply + 1, true);
            if (sc > a && sc < b) sc = -abSearch(np, depth - 1, -b, -a, ply + 1, true);
        }
        else {
            sc = -abSearch(np, depth - 1, -b, -a, ply + 1, true);
        }
        if (stopS.load()) return 0;
        if (sc > bestSc) { bestSc = sc; bestMv = mv[i]; }
        if (sc > a) a = sc;
        if (a >= b) {
            if (!isCap(mv[i].flags) && ply < 128) {
                killers[ply][1] = killers[ply][0]; killers[ply][0] = mv[i];
                hist[p.side][mv[i].from][mv[i].to] += depth * depth;
            }
            break;
        }
    }
    if (!hasLegal) return chk ? (-20000 + ply) : 0;

    tt[ti].key = h; tt[ti].sc = (int16_t)bestSc; tt[ti].dep = (int8_t)depth; tt[ti].best = bestMv;
    tt[ti].fl = bestSc <= origA ? TT_AL : bestSc >= b ? TT_BE : TT_EX;
    return bestSc;
}

Move findBest(Pos& p, int tMs) {
    tLimMs = tMs; tStart = chrono::steady_clock::now();
    stopS.store(false); nodes = 0; initSearch();

    auto lg = genLegal(p);
    if (lg.empty()) return {};
    if (lg.size() == 1) return lg[0];

    Move best = lg[0]; int bestSc = -99999;

    for (int depth = 1; depth <= 100; depth++) {
        Move iBest = lg[0]; int iSc = -99999;
        auto mv = lg;
        for (size_t i = 0; i < mv.size(); i++) if (mv[i] == best) { swap(mv[0], mv[i]); break; }

        bool done = true;
        for (size_t i = 0; i < mv.size(); i++) {
            Pos np = doMove(p, mv[i]);
            int sc;
            if (i == 0) sc = -abSearch(np, depth - 1, -99999, -iSc, 1, true);
            else {
                sc = -abSearch(np, depth - 1, -iSc - 1, -iSc, 1, true);
                if (sc > iSc) sc = -abSearch(np, depth - 1, -99999, -iSc, 1, true);
            }
            if (stopS.load()) { done = false; break; }
            if (sc > iSc) { iSc = sc; iBest = mv[i]; }
        }
        if (done || iSc > bestSc) { best = iBest; bestSc = iSc; }
        if (stopS.load()) break;
        auto el = chrono::duration_cast<chrono::milliseconds>(chrono::steady_clock::now() - tStart).count();
        if (el > tMs * 2 / 3) break;
        if (abs(bestSc) > 19000) break;
    }
    return best;
}

Pos parseFEN(const string& fen) {
    Pos p; istringstream ss(fen);
    string board, side, castle, ep; int hm, fm;
    ss >> board >> side >> castle >> ep >> hm >> fm;
    int rank = 7, file = 0;
    for (char c : board) {
        if (c == '/') { rank--; file = 0; }
        else if (c >= '1' && c <= '8') file += c - '0';
        else {
            Color col = (c >= 'a' && c <= 'z') ? BLACK : WHITE;
            char lo = c | 32; PieceType pt = PAWN;
            if (lo == 'n') pt = KNIGHT; else if (lo == 'b') pt = BISHOP;
            else if (lo == 'r') pt = ROOK; else if (lo == 'q') pt = QUEEN;
            else if (lo == 'k') pt = KING;
            int sq = sqMake(rank, file);
            p.b[sq] = makePiece(col, pt);
            if (pt == KING) p.ksq[col] = sq;
            file++;
        }
    }
    p.side = side == "b" ? BLACK : WHITE;
    for (char c : castle) {
        if (c == 'K') p.castle |= WKC; if (c == 'Q') p.castle |= WQC;
        if (c == 'k') p.castle |= BKC; if (c == 'q') p.castle |= BQC;
    }
    if (ep != "-" && ep.size() >= 2) p.ep = sqMake(ep[1] - '1', ep[0] - 'a');
    p.hm = hm; p.fm = fm;
    return p;
}

int main() {
    int timeout; string fen;
    cin >> timeout;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    getline(cin, fen);
    Pos p = parseFEN(fen);
    Move best = findBest(p, timeout * 900);
    cout << fmtMove(p, best) << endl;
    return 0;
}
