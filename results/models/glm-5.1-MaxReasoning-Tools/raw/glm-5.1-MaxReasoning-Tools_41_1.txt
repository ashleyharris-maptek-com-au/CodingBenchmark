; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 22
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inNormal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%funcType = OpTypeFunction %void
%ptrInputVec3 = OpTypePointer Input %vec3
%ptrOutputVec4 = OpTypePointer Output %vec4

%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%vec3_0_5 = OpConstantComposite %vec3 %float_0_5 %float_0_5 %float_0_5

%inNormal = OpVariable %ptrInputVec3 Input
%outColor = OpVariable %ptrOutputVec4 Output

%main = OpFunction %void None %funcType
%entry = OpLabel
%normal = OpLoad %vec3 %inNormal
%normalNorm = OpNormalize %vec3 %normal
%halfNormal = OpVectorTimesScalar %vec3 %normalNorm %float_0_5
%mapped = OpFAdd %vec3 %halfNormal %vec3_0_5
%result = OpCompositeConstruct %vec4 %mapped %float_1
OpStore %outColor %result
OpReturn
OpFunctionEnd
