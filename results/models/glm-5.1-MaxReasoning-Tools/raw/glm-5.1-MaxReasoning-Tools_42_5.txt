; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 128
; Schema: 0

OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %inNormal %inColor %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void      = OpTypeVoid
%void_fn   = OpTypeFunction %void
%float     = OpTypeFloat 32
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%uint      = OpTypeInt 32 0
%int       = OpTypeInt 32 1

; Constants
%uint_2    = OpConstant %uint 2
%int_0     = OpConstant %int 0
%int_1     = OpConstant %int 1
%float_0   = OpConstant %float 0
%float_1   = OpConstant %float 1
%float_015 = OpConstant %float 0.015
%float_075 = OpConstant %float 0.75
%float_05  = OpConstant %float 0.05

; Input gl_PerVertex block
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in      = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv  = OpTypePointer Input %arr_pv_in
%gl_in          = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; User-defined input arrays
%arr_v3_2     = OpTypeArray %v3float %uint_2
%ptr_in_arrv3 = OpTypePointer Input %arr_v3_2
%worldPos     = OpVariable %ptr_in_arrv3 Input
%inNormal     = OpVariable %ptr_in_arrv3 Input
%inColor      = OpVariable %ptr_in_arrv3 Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
%gl_out           = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor   = OpVariable %ptr_out_v3 Output

; ---- main ----
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Read endpoints p0, p1 from worldPos
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0     = OpLoad %v3float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p1     = OpLoad %v3float %p1_ptr

; Read colors c0, c1
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0     = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1     = OpLoad %v3float %c1_ptr

; dir = p1 - p0
%dir = OpFSub %v3float %p1 %p0

; perp = normalize(vec3(-dir.y, dir.x, 0.0))
%dir_x     = OpCompositeExtract %float %dir 0
%dir_y     = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw  = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0
%perp      = OpExtInst %v3float %glsl450 Normalize %perp_raw

; mid = p0 + 0.75 * dir
%dir_075 = OpVectorTimesScalar %v3float %dir %float_075
%mid     = OpFAdd %v3float %p0 %dir_075

; Scale perpendicular by thicknesses
%perp_015 = OpVectorTimesScalar %v3float %perp %float_015
%perp_05  = OpVectorTimesScalar %v3float %perp %float_05

; ---- Shaft quad (4 vertices, triangle strip) ----
%sv0 = OpFAdd %v3float %p0  %perp_015
%sv1 = OpFSub %v3float %p0  %perp_015
%sv2 = OpFAdd %v3float %mid %perp_015
%sv3 = OpFSub %v3float %mid %perp_015

; Output position pointer (reused for all emits)
%gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit shaft vertex 0  (p0 + perp*0.015)
%sv0_v4 = OpCompositeConstruct %v4float %sv0 %float_1
OpStore %gl_out_pos %sv0_v4
OpStore %outColor %c0
OpEmitVertex

; Emit shaft vertex 1  (p0 - perp*0.015)
%sv1_v4 = OpCompositeConstruct %v4float %sv1 %float_1
OpStore %gl_out_pos %sv1_v4
OpStore %outColor %c0
OpEmitVertex

; Emit shaft vertex 2  (mid + perp*0.015)
%sv2_v4 = OpCompositeConstruct %v4float %sv2 %float_1
OpStore %gl_out_pos %sv2_v4
OpStore %outColor %c0
OpEmitVertex

; Emit shaft vertex 3  (mid - perp*0.015)
%sv3_v4 = OpCompositeConstruct %v4float %sv3 %float_1
OpStore %gl_out_pos %sv3_v4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; ---- Arrowhead triangle (3 vertices, triangle strip) ----
%av0 = OpFAdd %v3float %mid %perp_05
%av1 = OpFSub %v3float %mid %perp_05

; Emit arrowhead vertex 0  (mid + perp*0.05)
%av0_v4 = OpCompositeConstruct %v4float %av0 %float_1
OpStore %gl_out_pos %av0_v4
OpStore %outColor %c1
OpEmitVertex

; Emit arrowhead vertex 1  (mid - perp*0.05)
%av1_v4 = OpCompositeConstruct %v4float %av1 %float_1
OpStore %gl_out_pos %av1_v4
OpStore %outColor %c1
OpEmitVertex

; Emit arrowhead vertex 2  (p1, the tip)
%tip_v4 = OpCompositeConstruct %v4float %p1 %float_1
OpStore %gl_out_pos %tip_v4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
