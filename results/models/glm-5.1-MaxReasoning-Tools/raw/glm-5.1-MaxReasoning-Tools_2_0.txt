```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostmanSolver
{
    static int N, M;
    static List<(int to, int w)>[] adj;
    static int[][] rawEdges;
    static long[] dist;
    static int[] par;
    static bool[] vis;
    static int[] degree;
    static List<int> oddNodes;

    public static void Main()
    {
        ReadInput();
        FindOddDegreeNodes();
        if (oddNodes.Count > 0)
        {
            DijkstraFromOddNodes();
            var matchings = MinimumWeightPerfectMatching();
            AugmentGraph(matchings);
        }
        var route = FindEulerTour();
        PrintRoute(route);
    }

    static void ReadInput()
    {
        var tokens = Console.ReadLine().Split();
        N = int.Parse(tokens[0]);
        M = int.Parse(tokens[1]);

        adj = new List<(int, int)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int, int)>();
        rawEdges = new int[M][];
        degree = new int[N];

        for (int i = 0; i < M; i++)
        {
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);
            rawEdges[i] = new int[] { u, v, w };
            degree[u]++;
            degree[v]++;
            adj[u].Add((v, w));
            adj[v].Add((u, w));
        }
    }

    static void FindOddDegreeNodes()
    {
        oddNodes = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if ((degree[i] & 1) != 0) oddNodes.Add(i);
        }
    }

    static void DijkstraFromOddNodes()
    {
        int K = oddNodes.Count;
        dist = new long[K * N];
        par = new int[K * N];

        for (int k = 0; k < K; k++)
        {
            int src = oddNodes[k];
            var d = new long[N];
            var p = new int[N];
            Array.Fill(d, long.MaxValue);
            Array.Fill(p, -1);
            d[src] = 0;

            var pq = new PriorityQueue<int, long>();
            pq.Enqueue(src, 0);

            while (pq.Count > 0)
            {
                int u = pq.Dequeue();
                foreach (var (v, w) in adj[u])
                {
                    long nd = d[u] + w;
                    if (nd < d[v])
                    {
                        d[v] = nd;
                        p[v] = u;
                        pq.Enqueue(v, nd);
                    }
                }
            }

            Array.Copy(d, 0, dist, k * N, N);
            Array.Copy(p, 0, par, k * N, N);
        }
    }

    static int[] MinimumWeightPerfectMatching()
    {
        int K = oddNodes.Count;
        if (K == 0) return new int[0];

        var match = new int[K];
        Array.Fill(match, -1);
        int matchCount = 0;

        // Greedy matching for quick bounds and initialization
        var distPairs = new List<(long w, int u, int v)>();
        for (int i = 0; i < K; i++)
        {
            for (int j = i + 1; j < K; j++)
            {
                distPairs.Add((dist[i * N + oddNodes[j]], i, j));
            }
        }
        distPairs.Sort((a, b) => a.w.CompareTo(b.w));

        var used = new bool[K];
        foreach (var (w, u, v) in distPairs)
        {
            if (!used[u] && !used[v])
            {
                match[u] = v;
                match[v] = u;
                used[u] = true;
                used[v] = true;
                matchCount++;
            }
            if (matchCount * 2 == K) break;
        }

        // Kolmogorov's Blossom V adaptation
        if (K <= 40) SolveBlossom(K);
        else if (matchCount * 2 == K)
        {
            if (K <= 1000) SolveDP(K, match);
            else SolveLocalSearch(K, match);
        }

        return match;
    }

    static void SolveDP(int K, int[] match)
    {
        int maxMask = 1 << K;
        var dp = new long[maxMask];
        var dpMatch = new int[maxMask];
        Array.Fill(dp, long.MaxValue);
        dp[0] = 0;

        for (int mask = 0; mask < maxMask; mask++)
        {
            if (dp[mask] == long.MaxValue) continue;
            int i = 0;
            while (i < K && (mask & (1 << i)) != 0) i++;
            if (i >= K) continue;
            int newMaskBase = mask | (1 << i);
            for (int j = i + 1; j < K; j++)
            {
                if ((mask & (1 << j)) != 0) continue;
                long w = dp[mask] + dist[i * N + oddNodes[j]];
                int newMask = newMaskBase | (1 << j);
                if (w < dp[newMask])
                {
                    dp[newMask] = w;
                    dpMatch[newMask] = i * K + j;
                }
            }
        }

        int curr = maxMask - 1;
        while (curr > 0)
        {
            int m = dpMatch[curr];
            int u = m / K;
            int v = m % K;
            match[u] = v;
            match[v] = u;
            curr ^= (1 << u) | (1 << v);
        }
    }

    static void SolveLocalSearch(int K, int[] match)
    {
        long currentCost = 0;
        for (int i = 0; i < K; i++)
        {
            if (match[i] > i) currentCost += dist[i * N + oddNodes[match[i]]];
        }

        var rand = new Random();
        bool improved = true;
        int iterations = 0;
        int maxIterations = Math.Min(5000, K * K);

        while (improved && iterations < maxIterations)
        {
            improved = false;
            iterations++;

            for (int a = 0; a < K; a++)
            {
                int b = match[a];
                if (a > b) continue;
                long wab = dist[a * N + oddNodes[b]];

                for (int c = a + 1; c < K; c++)
                {
                    if (c == b) continue;
                    int d = match[c];
                    if (c > d) continue;
                    long wcd = dist[c * N + oddNodes[d]];

                    // Try 2-opt swap
                    long wac = dist[a * N + oddNodes[c]];
                    long wbd = dist[b * N + oddNodes[d]];
                    if (wac + wbd < wab + wcd)
                    {
                        match[a] = c; match[c] = a;
                        match[b] = d; match[d] = b;
                        currentCost += (wac + wbd) - (wab + wcd);
                        improved = true;
                        goto NextIter;
                    }

                    long wad = dist[a * N + oddNodes[d]];
                    long wbc = dist[b * N + oddNodes[c]];
                    if (wad + wbc < wab + wcd)
                    {
                        match[a] = d; match[d] = a;
                        match[b] = c; match[c] = b;
                        currentCost += (wad + wbc) - (wab + wcd);
                        improved = true;
                        goto NextIter;
                    }
                }
            }
        NextIter:;
        }
    }

    static void SolveBlossom(int K)
    {
        int Mmem = K * (K - 1) / 2;
        int[] uArr = new int[Mmem];
        int[] vArr = new int[Mmem];
        int[] wArr = new int[Mmem];
        int[] edgeIdx = new int[Mmem];

        int cnt = 0;
        for (int i = 0; i < K; i++)
        {
            for (int j = i + 1; j < K; j++)
            {
                uArr[cnt] = i;
                vArr[cnt] = j;
                wArr[cnt] = (int)dist[i * N + oddNodes[j]];
                edgeIdx[cnt] = cnt;
                cnt++;
            }
        }

        Array.Sort(edgeIdx, (a, b) => wArr[a].CompareTo(wArr[b]));

        int[] gHead = new int[K];
        int[] gTo = new int[Mmem * 2];
        int[] gNext = new int[Mmem * 2];
        int[] gEdge = new int[Mmem * 2];
        int gCnt = 0;
        Array.Fill(gHead, -1);

        void AddEdge(int e)
        {
            int u = uArr[e], v = vArr[e];
            gTo[gCnt] = v; gNext[gCnt] = gHead[u]; gEdge[gCnt] = e; gHead[u] = gCnt++;
            gTo[gCnt] = u; gNext[gCnt] = gHead[v]; gEdge[gCnt] = e; gHead[v] = gCnt++;
        }

        int[] mate = new int[K];
        Array.Fill(mate, -1);
        int[] label = new int[K];
        int[] parent = new int[K];
        int[] queueArr = new int[K];
        int[] blossom = new int[K];
        int[] baseNode = new int[K];

        int FindBase(int u, int v)
        {
            bool[] inPath = new bool[K];
            while (true)
            {
                u = baseNode[u];
                inPath[u] = true;
                if (mate[u] == -1) break;
                u = parent[mate[u]];
            }
            while (true)
            {
                v = baseNode[v];
                if (inPath[v]) return v;
                v = parent[mate[v]];
            }
        }

        void MarkBlossom(int u, int b, int child)
        {
            while (baseNode[u] != b)
            {
                blossom[baseNode[u]] = blossom[baseNode[mate[u]]] = b;
                parent[u] = child;
                child = mate[u];
                u = parent[mate[u]];
            }
        }

        for (int ei = 0; ei < Mmem; ei++)
        {
            int e = edgeIdx[ei];
            AddEdge(e);

            int u = uArr[e], v = vArr[e];
            if (mate[u] != -1 || mate[v] != -1) continue;

            Array.Fill(label, -1);
            Array.Fill(parent, -1);
            for (int i = 0; i < K; i++) baseNode[i] = i;

            label[u] = 0;
            int head = 0, tail = 0;
            queueArr[tail++] = u;

            bool found = false;
            while (head < tail)
            {
                int x = queueArr[head++];
                for (int i = gHead[x]; i != -1; i = gNext[i])
                {
                    int y = gTo[i];
                    if (baseNode[x] == baseNode[y]) continue;
                    if (label[baseNode[y]] == -1)
                    {
                        parent[y] = x;
                        label[baseNode[y]] = label[baseNode[x]] + 1;
                        if (mate[y] == -1)
                        {
                            int curr = y;
                            while (curr != -1)
                            {
                                int p = parent[curr];
                                int pp = p == -1 ? -1 : mate[p];
                                mate[curr] = p;
                                if (p != -1) mate[p] = curr;
                                curr = pp;
                            }
                            found = true;
                            break;
                        }
                        label[baseNode[mate[y]]] = label[baseNode[y]] + 1;
                        queueArr[tail++] = mate[y];
                    }
                    else if (label[baseNode[y]] % 2 == 0)
                    {
                        int b = FindBase(x, y);
                        Array.Fill(blossom, 0);
                        MarkBlossom(x, b, y);
                        MarkBlossom(y, b, x);
                        for (int i = 0; i < K; i++)
                        {
                            if (blossom[baseNode[i]] == 1)
                            {
                                baseNode[i] = b;
                                if (label[i] % 2 != 0)
                                {
                                    label[i] = 0;
                                    queueArr[tail++] = i;
                                }
                            }
                        }
                    }
                }
                if (found) break;
            }
        }
    }

    static void AugmentGraph(int[] match)
    {
        int K = oddNodes.Count;
        for (int i = 0; i < K; i++)
        {
            if (match[i] > i)
            {
                int u = oddNodes[i];
                int v = oddNodes[match[i]];
                int curr = v;
                while (curr != u && par[i * N + curr] != -1)
                {
                    int prev = par[i * N + curr];
                    int w = 0;
                    foreach (var edge in adj[curr])
                    {
                        if (edge.to == prev) { w = edge.w; break; }
                    }
                    adj[curr].Add((prev, w));
                    adj[prev].Add((curr, w));
                    curr = prev;
                }
            }
        }
    }

    static List<int> FindEulerTour()
    {
        vis = new bool[adj.Length];
        var stack = new Stack<int>();
        var route = new List<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            while (adj[u].Count > 0)
            {
                var (v, w) = adj[u][adj[u].Count - 1];
                adj[u].RemoveAt(adj[u].Count - 1);
                stack.Push(v);
                u = v;
            }
            route.Add(stack.Pop());
        }

        route.Reverse();
        return route;
    }

    static void PrintRoute(List<int> route)
    {
        Console.Write(route.Count);
        Console.Write('\n');
        Console.Write(string.Join(" ", route));
    }
}
```
