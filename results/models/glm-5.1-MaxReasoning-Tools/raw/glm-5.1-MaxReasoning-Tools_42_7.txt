               OpCapability Geometry
     %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

     %void = OpTypeVoid
  %void_fn = OpTypeFunction %void
    %float = OpTypeFloat 32
  %v3float = OpTypeVector %float 3
  %v4float = OpTypeVector %float 4
      %int = OpTypeInt 32 1
     %uint = OpTypeInt 32 0

    %int_0 = OpConstant %int 0
    %int_1 = OpConstant %int 1
    %int_2 = OpConstant %int 2
   %uint_3 = OpConstant %uint 3
%float_0_08 = OpConstant %float 0.08
  %float_0 = OpConstant %float 0.0

%gl_PerVertex_in = OpTypeStruct %v4float
   %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
        %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
   %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
       %gl_out = OpVariable %ptr_out_pv Output

   %arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
     %inColor = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
    %outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

        %main = OpFunction %void None %void_fn
      %entry = OpLabel

; Load gl_in[0].gl_Position, gl_in[1].gl_Position, gl_in[2].gl_Position
    %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
   %p0_val = OpLoad %v4float %p0_ptr
    %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
   %p1_val = OpLoad %v4float %p1_ptr
    %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
   %p2_val = OpLoad %v4float %p2_ptr

; Extract xyz as vec3
   %p0_xyz = OpVectorShuffle %v3float %p0_val %p0_val 0 1 2
   %p1_xyz = OpVectorShuffle %v3float %p1_val %p1_val 0 1 2
   %p2_xyz = OpVectorShuffle %v3float %p2_val %p2_val 0 1 2

; Compute edges e1 = p1 - p0, e2 = p2 - p0
       %e1 = OpFSub %v3float %p1_xyz %p0_xyz
       %e2 = OpFSub %v3float %p2_xyz %p0_xyz

; Face normal: N = normalize(cross(e1, e2))
   %N_raw = OpExtInst %v3float %glsl 68 %e1 %e2
        %N = OpExtInst %v3float %glsl 69 %N_raw

; offset = N * 0.08
%offset_v3 = OpVectorTimesScalar %v3float %N %float_0_08

; Construct vec4(offset, 0.0)
       %ox = OpCompositeExtract %float %offset_v3 0
       %oy = OpCompositeExtract %float %offset_v3 1
       %oz = OpCompositeExtract %float %offset_v3 2
 %offset_v4 = OpCompositeConstruct %v4float %ox %oy %oz %float_0

; Common pointer to gl_out.gl_Position
%gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- Emit vertex 0: gl_Position = p0 + offset, color = inColor[0] ----
  %new_p0 = OpFAdd %v4float %p0_val %offset_v4
              OpStore %gl_out_pos %new_p0
  %c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
  %c0_val = OpLoad %v3float %c0_ptr
              OpStore %outColor %c0_val
              OpEmitVertex

; ---- Emit vertex 1: gl_Position = p1 + offset, color = inColor[1] ----
  %new_p1 = OpFAdd %v4float %p1_val %offset_v4
              OpStore %gl_out_pos %new_p1
  %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
  %c1_val = OpLoad %v3float %c1_ptr
              OpStore %outColor %c1_val
              OpEmitVertex

; ---- Emit vertex 2: gl_Position = p2 + offset, color = inColor[2] ----
  %new_p2 = OpFAdd %v4float %p2_val %offset_v4
              OpStore %gl_out_pos %new_p2
  %c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
  %c2_val = OpLoad %v3float %c2_ptr
              OpStore %outColor %c2_val
              OpEmitVertex

; End the triangle strip primitive
              OpEndPrimitive
              OpReturn
              OpFunctionEnd
