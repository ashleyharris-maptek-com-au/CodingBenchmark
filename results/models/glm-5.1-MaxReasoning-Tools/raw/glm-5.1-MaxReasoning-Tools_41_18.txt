; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO_struct Block
OpMemberDecorate %UBO_struct 0 Offset 0
OpMemberDecorate %UBO_struct 0 ColMajor
OpMemberDecorate %UBO_struct 0 MatrixStride 16
OpMemberDecorate %UBO_struct 1 Offset 64
OpMemberDecorate %UBO_struct 1 ColMajor
OpMemberDecorate %UBO_struct 1 MatrixStride 16
OpMemberDecorate %UBO_struct 2 Offset 128
OpMemberDecorate %UBO_struct 2 ColMajor
OpMemberDecorate %UBO_struct 2 MatrixStride 16
OpMemberDecorate %UBO_struct 3 Offset 192
OpMemberDecorate %UBO_struct 4 Offset 208
OpMemberDecorate %UBO_struct 5 Offset 224

%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4 = OpTypeMatrix %v4float 4
%bool = OpTypeBool
%uint = OpTypeInt 32 0

%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_5 = OpConstant %float 0.5
%float_2 = OpConstant %float 2.0

%UBO_struct = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float

%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_uniform_v4float = OpTypePointer Uniform %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO_struct

%in_worldPos = OpVariable %ptr_input_v3float Input
%in_normal = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%glsl450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %void_fn
%entry = OpLabel

%worldPos = OpLoad %v3float %in_worldPos
%norm_val = OpLoad %v3float %in_normal

%cam_ptr = OpAccessChain %ptr_uniform_v4float %ubo %uint_4
%cam_v4 = OpLoad %v4float %cam_ptr
%cam_pos = OpVectorShuffle %v3float %cam_v4 %cam_v4 0 1 2

%N = OpExtInst %v3float %glsl450 62 %norm_val
%V_dir = OpFSub %v3float %cam_pos %worldPos
%V = OpExtInst %v3float %glsl450 62 %V_dir

%dot_NV = OpDot %float %N %V
%NdotV = OpExtInst %float %glsl450 40 %dot_NV %float_0

%cond = OpFOrdLessThan %bool %NdotV %float_0_5

OpSelectionMerge %merge None
OpBranchConditional %cond %if_block %else_block

%if_block = OpLabel
%g_if = OpFMul %float %float_2 %NdotV
OpBranch %merge

%else_block = OpLabel
%delta_raw = OpFSub %float %NdotV %float_0_5
%delta = OpFMul %float %delta_raw %float_2
%r_else = OpFSub %float %float_1 %delta
%g_else = OpFSub %float %float_1 %delta
OpBranch %merge

%merge = OpLabel
%r = OpPhi %float %float_1 %if_block %r_else %else_block
%g = OpPhi %float %g_if %if_block %g_else %else_block
%b = OpPhi %float %float_0 %if_block %delta %else_block

%color = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %color

OpReturn
OpFunctionEnd
