```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard: out-of-range invocation does nothing
    if (i >= N) return;

    // Unpack parameters
    float dt         = radii.x;
    float sepRadius  = radii.y;
    float alignRadius = radii.z;
    float cohRadius  = radii.w;

    float sepWeight   = weights.x;
    float alignWeight = weights.y;
    float cohWeight   = weights.z;
    float maxSpeed    = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    // Read current boid state
    vec3 pos = inData[i * 2u].xyz;
    vec3 vel = inData[i * 2u + 1u].xyz;

    // Accumulators for each flocking rule
    vec3  sepAccum   = vec3(0.0);
    vec3  alignAccum = vec3(0.0);
    vec3  cohAccum   = vec3(0.0);
    uint  sepCount   = 0u;
    uint  alignCount = 0u;
    uint  cohCount   = 0u;

    // Iterate over all other boids
    for (uint j = 0u; j < N; j++)
    {
        // Skip self
        if (j == i) continue;

        vec3 otherPos = inData[j * 2u].xyz;
        vec3 otherVel = inData[j * 2u + 1u].xyz;

        vec3  diff = pos - otherPos;
        float dist = length(diff);

        // Skip zero-distance neighbors to avoid NaN
        if (dist < 1e-6) continue;

        // Separation: accumulate unit direction away from neighbor
        if (dist < sepRadius)
        {
            sepAccum += diff / dist;
            sepCount++;
        }

        // Alignment: accumulate neighbor velocity
        if (dist < alignRadius)
        {
            alignAccum += otherVel;
            alignCount++;
        }

        // Cohesion: accumulate neighbor position
        if (dist < cohRadius)
        {
            cohAccum += otherPos;
            cohCount++;
        }
    }

    // Compute acceleration from averaged forces
    vec3 accel = vec3(0.0);

    // Separation: average repulsion direction, scaled by weight
    if (sepCount > 0u)
    {
        accel += (sepAccum / float(sepCount)) * sepWeight;
    }

    // Alignment: steer toward average neighbor velocity
    if (alignCount > 0u)
    {
        vec3 avgVel = alignAccum / float(alignCount);
        accel += (avgVel - vel) * alignWeight;
    }

    // Cohesion: steer toward average neighbor position (center of mass)
    if (cohCount > 0u)
    {
        vec3 avgPos = cohAccum / float(cohCount);
        accel += (avgPos - pos) * cohWeight;
    }

    // Boundary force: push boids back toward center when outside
    // Apply per-component when |pos_comp| > boundSize * 0.4
    {
        float threshold = boundSize * 0.4;
        float scale     = boundSize * 0.5;

        if (abs(pos.x) > threshold)
        {
            accel.x += -boundForce * pos.x / scale;
        }
        if (abs(pos.y) > threshold)
        {
            accel.y += -boundForce * pos.y / scale;
        }
        if (abs(pos.z) > threshold)
        {
            accel.z += -boundForce * pos.z / scale;
        }
    }

    // Integrate velocity: v += a * dt
    vel += accel * dt;

    // Clamp velocity magnitude to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed)
    {
        vel *= maxSpeed / speed;
    }

    // Integrate position: p += v * dt
    pos += vel * dt;

    // Write updated state to output buffer
    outData[i * 2u]     = vec4(pos, 0.0);
    outData[i * 2u + 1u] = vec4(vel, 0.0);
}
```

## Design Walkthrough

### Data Layout
Each boid occupies two consecutive `vec4` slots—position at index `2i`, velocity at `2i+1`—matching the ping-pong buffer contract. One invocation processes exactly one boid.

### Flocking Rules (Reynolds)

| Rule | Neighbor contribution | Averaged result | Steering force |
|------|----------------------|-----------------|----------------|
| **Separation** | Unit vector `(pos - otherPos) / dist` pointing away from neighbor | Average repulsion direction | `avgSep × sepWeight` |
| **Alignment** | Neighbor velocity `otherVel` | Average neighbor velocity `avgVel` | `(avgVel − vel) × alignWeight` |
| **Cohesion** | Neighbor position `otherPos` | Center of mass `avgPos` | `(avgPos − pos) × cohWeight` |

Each accumulated sum is divided by its respective neighbor count (guarded against zero), ensuring that more neighbors don't just create stronger forces—they create better-informed ones.

### Safety Guards
- **Out-of-range invocation** (`i ≥ N`): early return, no write.
- **Self-interaction** (`j == i`): skipped.
- **Zero-distance neighbors** (`dist < 1e-6`): skipped to prevent NaN from `diff/dist`.
- **Division by zero**: each accumulator is only divided when its count is > 0.
- **Speed clamping**: after integration, `vel` is rescaled if `length(vel) > maxSpeed`.

### Boundary Containment
When any position component exceeds `boundSize × 0.4`, a spring-like restoring acceleration is applied per-component:

```
accel_comp = −boundForce × pos_comp / (boundSize × 0.5)
```

This pushes the boid toward the origin with strength proportional to its displacement, scaled so that at the boundary edge the force equals `boundForce × 0.8` and grows linearly beyond.

### Integration Order
1. Compute total acceleration from all three rules + boundary
2. `vel += accel × dt`
3. Clamp `‖vel‖ ≤ maxSpeed`
4. `pos += vel × dt`

Clamping before position integration ensures the boid never moves faster than `maxSpeed × dt` per step.
