Here is the complete, runnable Python code for the CVRP solver:

```python
import sys
import math
import random

def solve():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    CAPACITY = int(next(it))

    depot_x = float(next(it))
    depot_y = float(next(it))

    customers = []
    for i in range(N):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        customers.append((x, y, d))

    n_nodes = N + 1
    dist = [[0.0] * n_nodes for _ in range(n_nodes)]
    coords = [(depot_x, depot_y)] + [(c[0], c[1]) for c in customers]
    for i in range(n_nodes):
        for j in range(n_nodes):
            dx = coords[i][0] - coords[j][0]
            dy = coords[i][1] - coords[j][1]
            dist[i][j] = math.sqrt(dx * dx + dy * dy)

    demands = [0.0] + [c[2] for c in customers]

    def route_demand(route):
        return sum(demands[c] for c in route)

    def route_distance(route):
        if not route:
            return 0.0
        d = dist[0][route[0]]
        for i in range(len(route) - 1):
            d += dist[route[i]][route[i + 1]]
        d += dist[route[-1]][0]
        return d

    def total_distance(routes):
        return sum(route_distance(r) for r in routes)

    # ---- Clarke-Wright Savings ----
    def clarke_wright():
        savings = []
        for i in range(1, N + 1):
            for j in range(i + 1, N + 1):
                s = dist[i][0] + dist[0][j] - dist[i][j]
                savings.append((s, i, j))
        savings.sort(reverse=True)

        customer_route = {i: i for i in range(1, N + 1)}
        route_set = {i: [i] for i in range(1, N + 1)}

        for s, i, j in savings:
            ri = customer_route.get(i)
            rj = customer_route.get(j)
            if ri is None or rj is None or ri == rj:
                continue
            route_i = route_set.get(ri)
            route_j = route_set.get(rj)
            if route_i is None or route_j is None:
                continue
            if route_demand(route_i) + route_demand(route_j) > CAPACITY:
                continue
            merged = None
            if route_i[-1] == i and route_j[0] == j:
                merged = route_i + route_j
            elif route_i[0] == i and route_j[-1] == j:
                merged = route_j + route_i
            elif route_i[-1] == i and route_j[-1] == j:
                merged = route_i + list(reversed(route_j))
            elif route_i[0] == i and route_j[0] == j:
                merged = list(reversed(route_i)) + route_j
            if merged is not None:
                del route_set[ri]
                del route_set[rj]
                route_set[ri] = merged
                for c in merged:
                    customer_route[c] = ri
        return list(route_set.values())

    # ---- Nearest Neighbor ----
    def nearest_neighbor():
        unvisited = set(range(1, N + 1))
        routes = []
        while unvisited:
            route, cap, current = [], 0, 0
            while True:
                best, best_d = None, float('inf')
                for c in unvisited:
                    if cap + demands[c] <= CAPACITY and dist[current][c] < best_d:
                        best_d = dist[current][c]
                        best = c
                if best is None:
                    break
                route.append(best)
                cap += demands[best]
                unvisited.remove(best)
                current = best
            if route:
                routes.append(route)
        return routes

    # ---- Cheapest Insertion ----
    def cheapest_insertion():
        unvisited = set(range(1, N + 1))
        routes = []
        while unvisited:
            farthest = max(unvisited, key=lambda c: dist[0][c])
            route, cap = [farthest], demands[farthest]
            unvisited.remove(farthest)
            while True:
                best_pos, best_cust, best_delta = -1, -1, float('inf')
                for c in unvisited:
                    if cap + demands[c] > CAPACITY:
                        continue
                    for pos in range(len(route) + 1):
                        a = 0 if pos == 0 else route[pos - 1]
                        b = route[pos] if pos < len(route) else 0
                        delta = dist[a][c] + dist[c][b] - dist[a][b]
                        if delta < best_delta:
                            best_delta, best_pos, best_cust = delta, pos, c
                if best_cust == -1:
                    break
                route.insert(best_pos, best_cust)
                cap += demands[best_cust]
                unvisited.remove(best_cust)
            routes.append(route)
        return routes

    # ---- Intra-route: 2-opt ----
    def two_opt_route(r):
        r = list(r)
        if len(r) <= 2:
            return r
        improved = True
        while improved:
            improved = False
            for i in range(len(r) - 1):
                for j in range(i + 2, len(r)):
                    a, b = r[i], r[i + 1]
                    c, d = r[j], r[j + 1] if j + 1 < len(r) else 0
                    if dist[a][c] + dist[b][d] < dist[a][b] + dist[c][d] - 1e-10:
                        r[i + 1:j + 1] = reversed(r[i + 1:j + 1])
                        improved = True
        return r

    # ---- Intra-route: or-opt ----
    def or_opt_route(route):
        r = list(route)
        improved = True
        while improved:
            improved = False
            for seg_len in [1, 2, 3]:
                for i in range(len(r) - seg_len + 1):
                    segment = r[i:i + seg_len]
                    remaining = r[:i] + r[i + seg_len:]
                    a = 0 if i == 0 else r[i - 1]
                    b = r[i + seg_len] if i + seg_len < len(r) else 0
                    remove_cost = dist[a][segment[0]] + dist[segment[-1]][b]
                    add_no_seg = dist[a][b]
                    for j in range(len(remaining) + 1):
                        c = 0 if j == 0 else remaining[j - 1]
                        d = remaining[j] if j < len(remaining) else 0
                        delta = (add_no_seg - remove_cost) + (dist[c][segment[0]] + dist[segment[-1]][d] - dist[c][d])
                        if delta < -1e-10:
                            r = remaining[:j] + segment + remaining[j:]
                            improved = True
                            break
                    if improved:
                        break
                if improved:
                    break
        return r

    # ---- Inter-route: relocate ----
    def relocate_customers(routes):
        improved = True
        while improved:
            improved = False
            for ri in range(len(routes)):
                if not routes[ri]:
                    continue
                for pos_i in range(len(routes[ri])):
                    cust = routes[ri][pos_i]
                    a = 0 if pos_i == 0 else routes[ri][pos_i - 1]
                    b = routes[ri][pos_i + 1] if pos_i + 1 < len(routes[ri]) else 0
                    remove_gain = dist[a][cust] + dist[cust][b] - dist[a][b]
                    for rj in range(len(routes)):
                        if ri == rj:
                            continue
                        if route_demand(routes[rj]) + demands[cust] > CAPACITY:
                            continue
                        for pos_j in range(len(routes[rj]) + 1):
                            c = 0 if pos_j == 0 else routes[rj][pos_j - 1]
                            d = routes[rj][pos_j] if pos_j < len(routes[rj]) else 0
                            insert_cost = dist[c][cust] + dist[cust][d] - dist[c][d]
                            if insert_cost - remove_gain < -1e-10:
                                routes[ri].pop(pos_i)
                                routes[rj].insert(pos_j, cust)
                                improved = True
                                break
                        if improved:
                            break
                    if improved:
                        break
                if improved:
                    break
        routes[:] = [r for r in routes if r]
        return routes

    # ---- Inter-route: exchange ----
    def exchange_customers(routes):
        improved = True
        while improved:
            improved = False
            for ri in range(len(routes)):
                for rj in range(ri + 1, len(routes)):
                    for pos_i in range(len(routes[ri])):
                        for pos_j in range(len(routes[rj])):
                            ci, cj = routes[ri][pos_i], routes[rj][pos_j]
                            if route_demand(routes[ri]) - demands[ci] + demands[cj] > CAPACITY:
                                continue
                            if route_demand(routes[rj]) - demands[cj] + demands[ci] > CAPACITY:
                                continue
                            a_i = 0 if pos_i == 0 else routes[ri][pos_i - 1]
                            b_i = routes[ri][pos_i + 1] if pos_i + 1 < len(routes[ri]) else 0
                            a_j = 0 if pos_j == 0 else routes[rj][pos_j - 1]
                            b_j = routes[rj][pos_j + 1] if pos_j + 1 < len(routes[rj]) else 0
                            delta = (dist[a_i][cj] + dist[cj][b_i] - dist[a_i][ci] - dist[ci][b_i]) + \
                                   (dist[a_j][ci] + dist[ci][b_j] - dist[a_j][cj] - dist[cj][b_j])
                            if delta < -1e-10:
                                routes[ri][pos_i], routes[rj][pos_j] = cj, ci
                                improved = True
                                break
                        if improved:
                            break
                    if improved:
                        break
                if improved:
                    break
        return routes

    # ---- Inter-route: cross (move segment between routes) ----
    def cross_exchange(routes):
        improved = True
        while improved:
            improved = False
            for ri in range(len(routes)):
                for rj in range(len(routes)):
                    if ri == rj:
                        continue
                    for seg_len in [1, 2]:
                        for pos_i in range(len(routes[ri]) - seg_len + 1):
                            seg = routes[ri][pos_i:pos_i + seg_len]
                            seg_d = sum(demands[c] for c in seg)
                            if route_demand(routes[rj]) + seg_d > CAPACITY:
                                continue
                            a = 0 if pos_i == 0 else routes[ri][pos_i - 1]
                            b = routes[ri][pos_i + seg_len] if pos_i + seg_len < len(routes[ri]) else 0
                            remove_gain = dist[a][seg[0]] + dist[seg[-1]][b] - dist[a][b]
                            best_pos, best_cost = -1, float('inf')
                            for pos_j in range(len(routes[rj]) + 1):
                                c = 0 if pos_j == 0 else routes[rj][pos_j - 1]
                                d = routes[rj][pos_j] if pos_j < len(routes[rj]) else 0
                                insert_cost = dist[c][seg[0]] + dist[seg[-1]][d] - dist[c][d]
                                if insert_cost < best_cost:
                                    best_cost, best_pos = insert_cost, pos_j
                            if best_cost - remove_gain < -1e-10:
                                routes[ri] = routes[ri][:pos_i] + routes[ri][pos_i + seg_len:]
                                routes[rj] = routes[rj][:best_pos] + seg + routes[rj][best_pos:]
                                improved = True
                                break
                        if improved:
                            break
                    if improved:
                        break
                if improved:
                    break
        routes[:] = [r for r in routes if r]
        return routes

    # ---- Full improvement loop ----
    def improve_routes(routes):
        for i in range(len(routes)):
            routes[i] = two_opt_route(routes[i])
            routes[i] = or_opt_route(routes[i])
        old_dist = total_distance(routes)
        for _ in range(20):
            relocate_customers(routes)
            exchange_customers(routes)
            cross_exchange(routes)
            for i in range(len(routes)):
                routes[i] = two_opt_route(routes[i])
                routes[i] = or_opt_route(routes[i])
            new_dist = total_distance(routes)
            if new_dist >= old_dist - 1e-10:
                break
            old_dist = new_dist
        return routes

    # ---- Try multiple construction heuristics, keep best ----
    best_routes, best_dist = None, float('inf')

    def try_solution(routes):
        nonlocal best_routes, best_dist
        routes = improve_routes(routes)
        d = total_distance(routes)
        if d < best_dist:
            best_dist = d
            best_routes = [list(r) for r in routes]

    try:
        try_solution(clarke_wright())
    except Exception:
        pass
    try:
        try_solution(nearest_neighbor())
    except Exception:
        pass
    try:
        try_solution(cheapest_insertion())
    except Exception:
        pass

    # Randomized NN
    random.seed(42)
    num_trials = min(30, max(5, N // 3))
    for trial in range(num_trials):
        try:
            unvisited = set(range(1, N + 1))
            routes = []
            while unvisited:
                route, cap, current = [], 0, 0
                while True:
                    candidates = [c for c in unvisited if cap + demands[c] <= CAPACITY]
                    if not candidates:
                        break
                    dists = sorted([(dist[current][c], c) for c in candidates])
                    top_k = max(1, min(len(dists), max(1, len(dists) // 3 + trial % 3)))
                    best = dists[random.randint(0, top_k - 1)][1]
                    route.append(best)
                    cap += demands[best]
                    unvisited.remove(best)
                    current = best
                if route:
                    routes.append(route)
            try_solution(routes)
        except Exception:
            pass

    # Enforce K route limit
    if len(best_routes) > K:
        while len(best_routes) > K:
            best_merge, best_saving = None, -float('inf')
            for i in range(len(best_routes)):
                for j in range(i + 1, len(best_routes)):
                    if route_demand(best_routes[i]) + route_demand(best_routes[j]) > CAPACITY:
                        continue
                    ri, rj = best_routes[i], best_routes[j]
                    old = route_distance(ri) + route_distance(rj)
                    for merged in [ri + rj, rj + ri, ri + list(reversed(rj)), list(reversed(ri)) + rj]:
                        new = route_distance(merged)
                        if old - new > best_saving:
                            best_saving = old - new
                            best_merge = (i, j, merged)
            if best_merge is None:
                break
            i, j, merged = best_merge
            best_routes[i] = merged
            del best_routes[j]
        if len(best_routes) > K:
            while len(best_routes) > K:
                min_pair, min_sum = None, float('inf')
                for i in range(len(best_routes)):
                    for j in range(i + 1, len(best_routes)):
                        s = route_demand(best_routes[i]) + route_demand(best_routes[j])
                        if s < min_sum:
                            min_sum, min_pair = s, (i, j)
                if min_pair:
                    i, j = min_pair
                    best_routes[i] = best_routes[i] + best_routes[j]
                    del best_routes[j]

    best_routes = improve_routes(best_routes)
    best_dist = total_distance(best_routes)

    print(f"{best_dist:.2f}")
    print(len(best_routes))
    for route in best_routes:
        print(' '.join(str(c - 1) for c in route))

solve()
```

**How it works:**

1. **Multiple Construction Heuristics** — builds initial solutions using:
   - **Clarke-Wright Savings**: merges singleton routes by savings (depot-trip overlap), respecting capacity
   - **Nearest Neighbor**: greedily extends each route with the closest feasible customer
   - **Cheapest Insertion**: seed with farthest customer, then insert each next customer at its cheapest position
   - **Randomized NN** (30 trials): like NN but randomly picks from the top-k nearest to diversify solutions

2. **Local Search Improvement** applied to every candidate solution:
   - **2-opt** (intra-route): reverses a sub-segment to eliminate crossings
   - **Or-opt** (intra-route): relocates 1–3 consecutive customers to a better position in the same route
   - **Relocate** (inter-route): moves a single customer from one route to another
   - **Exchange** (inter-route): swaps two customers between routes
   - **Cross exchange** (inter-route): moves a 1–2 customer segment from one route into another
   - These are iterated until convergence (up to 20 rounds)

3. **Best-of-all selection**: the best distance across all construction+improvement runs wins.

4. **Route limit enforcement**: if more than K routes exist, merges pairs with the best orientation and least cost increase; falls back to forced merging if no capacity-feasible merge exists.
