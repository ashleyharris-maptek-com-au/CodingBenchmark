; SPIR-V
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void      = OpTypeVoid
%fn_void   = OpTypeFunction %void
%float     = OpTypeFloat 32
%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0

%int_0     = OpConstant %int 0
%int_1     = OpConstant %int 1
%uint_2    = OpConstant %uint 2
%float_0   = OpConstant %float 0.0
%float_0_03 = OpConstant %float 0.03

; --- Input gl_PerVertex ---
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in       = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv   = OpTypePointer Input %arr_pv_in
%gl_in           = OpVariable %ptr_in_arr_pv Input

; --- Output gl_PerVertex ---
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
%gl_out           = OpVariable %ptr_out_pv Output

; --- Input color[2] ---
%arr_v3_2      = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%inColor       = OpVariable %ptr_in_arr_v3 Input

; --- Output outColor ---
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor   = OpVariable %ptr_out_v3 Output

; --- Pointer types for access chains ---
%ptr_in_v4  = OpTypePointer Input %v4float
%ptr_in_v3  = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %fn_void
%entry = OpLabel

; ---- Read inputs ----
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0     = OpLoad %v4float %p0_ptr

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1     = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0     = OpLoad %v3float %c0_ptr

%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1     = OpLoad %v3float %c1_ptr

; ---- Compute line direction (xy only) ----
%p0_xy  = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy  = OpVectorShuffle %v2float %p1 %p1 0 1
%dir    = OpFSub %v2float %p1_xy %p0_xy

; ---- Compute perpendicular: (-dir.y, dir.x), normalize, scale ----
%dir_x      = OpCompositeExtract %float %dir 0
%dir_y      = OpCompositeExtract %float %dir 1
%neg_dir_y  = OpFNegate %float %dir_y
%perp_raw   = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_norm  = OpNormalize %v2float %perp_raw
%perp       = OpVectorTimesScalar %v2float %perp_norm %float_0_03

; ---- Build perp as vec4 (perp.xy, 0, 0) for addition to clip-space positions ----
%perp_x    = OpCompositeExtract %float %perp 0
%perp_y    = OpCompositeExtract %float %perp 1
%perp_v4   = OpCompositeConstruct %v4float %perp_x %perp_y %float_0 %float_0
%neg_perp_v4 = OpFNegate %v4float %perp_v4

; ---- Output position pointer (reused for every vertex) ----
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- Vertex 0: p0 + perp, color c0 ----
%v0_pos = OpFAdd %v4float %p0 %perp_v4
OpStore %out_pos_ptr %v0_pos
OpStore %outColor %c0
OpEmitVertex

; ---- Vertex 1: p0 - perp, color c0 ----
%v1_pos = OpFAdd %v4float %p0 %neg_perp_v4
OpStore %out_pos_ptr %v1_pos
OpStore %outColor %c0
OpEmitVertex

; ---- Vertex 2: p1 + perp, color c1 ----
%v2_pos = OpFAdd %v4float %p1 %perp_v4
OpStore %out_pos_ptr %v2_pos
OpStore %outColor %c1
OpEmitVertex

; ---- Vertex 3: p1 - perp, color c1 ----
%v3_pos = OpFAdd %v4float %p1 %neg_perp_v4
OpStore %out_pos_ptr %v3_pos
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
