```
; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0

%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_0_9 = OpConstant %float 0.9
%float_1 = OpConstant %float 1.0

%worldPos = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %void_fn
%entry = OpLabel

%pos = OpLoad %vec3 %worldPos
%pos_y = OpCompositeExtract %float %pos 1

%y_half = OpFMul %float %pos_y %float_0_5
%y_shifted = OpFAdd %float %y_half %float_0_5
%t = OpFClamp %float %y_shifted %float_0 %float_1

%red_t = OpFMul %float %t %float_0_9
%red = OpFAdd %float %float_0_1 %red_t

%green_t = OpFMul %float %t %float_0_6
%green = OpFAdd %float %float_0_4 %green_t

%blue_t = OpFMul %float %t %float_0_9
%blue = OpFAdd %float %float_0_1 %blue_t

%color = OpCompositeConstruct %vec4 %red %green %blue %float_1

OpStore %outColor %color

OpReturn
OpFunctionEnd
```
