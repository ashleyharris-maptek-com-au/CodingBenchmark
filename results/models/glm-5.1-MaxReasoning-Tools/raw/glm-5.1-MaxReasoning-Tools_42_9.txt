```
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %inColor %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0.0
%float_0_025 = OpConstant %float 0.025
%float_1 = OpConstant %float 1.0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load all 4 positions from gl_in
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0

%pos0 = OpLoad %v4float %pos0_ptr
%pos1 = OpLoad %v4float %pos1_ptr
%pos2 = OpLoad %v4float %pos2_ptr
%pos3 = OpLoad %v4float %pos3_ptr

; Tangent at start: t0 = normalize(gl_in[2].pos - gl_in[0].pos) using XY
%delta0 = OpFSub %v4float %pos2 %pos0
%delta0_xy = OpVectorShuffle %v2float %delta0 %delta0 0 1
%len0 = OpLength %float %delta0_xy
%inv_len0 = OpFDiv %float %float_1 %len0
%t0 = OpVectorTimesScalar %v2float %delta0_xy %inv_len0

; Tangent at end: t1 = normalize(gl_in[3].pos - gl_in[1].pos) using XY
%delta1 = OpFSub %v4float %pos3 %pos1
%delta1_xy = OpVectorShuffle %v2float %delta1 %delta1 0 1
%len1 = OpLength %float %delta1_xy
%inv_len1 = OpFDiv %float %float_1 %len1
%t1 = OpVectorTimesScalar %v2float %delta1_xy %inv_len1

; Perpendicular at start: perp0 = vec2(-t0.y, t0.x)
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

; Perpendicular at end: perp1 = vec2(-t1.y, t1.x)
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Offsets
%offset0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
%offset1 = OpVectorTimesScalar %v2float %perp1 %float_0_025

; Extract XY from start and end positions
%pos1_xy = OpVectorShuffle %v2float %pos1 %pos1 0 1
%pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1

; Compute 4 vertex XY positions
%v0_xy = OpFAdd %v2float %pos1_xy %offset0
%v1_xy = OpFSub %v2float %pos1_xy %offset0
%v2_xy = OpFAdd %v2float %pos2_xy %offset1
%v3_xy = OpFSub %v2float %pos2_xy %offset1

; Reconstruct vec4 with original Z and W
; v0_xy(vec2) + pos1(vec4): components 0,1 from v0_xy; 4,5 from pos1 (z,w)
%v0_pos = OpVectorShuffle %v4float %v0_xy %pos1 0 1 4 5
%v1_pos = OpVectorShuffle %v4float %v1_xy %pos1 0 1 4 5
%v2_pos = OpVectorShuffle %v4float %v2_xy %pos2 0 1 4 5
%v3_pos = OpVectorShuffle %v4float %v3_xy %pos2 0 1 4 5

; Load colors for start and end vertices
%color1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%color2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%color1 = OpLoad %v3float %color1_ptr
%color2 = OpLoad %v3float %color2_ptr

; Emit vertex 0: start + perp0*thickness
%out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0 %v0_pos
OpStore %outColor %color1
OpEmitVertex

; Emit vertex 1: start - perp0*thickness
%out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos1 %v1_pos
OpStore %outColor %color1
OpEmitVertex

; Emit vertex 2: end + perp1*thickness
%out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2 %v2_pos
OpStore %outColor %color2
OpEmitVertex

; Emit vertex 3: end - perp1*thickness
%out_pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos3 %v3_pos
OpStore %outColor %color2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
