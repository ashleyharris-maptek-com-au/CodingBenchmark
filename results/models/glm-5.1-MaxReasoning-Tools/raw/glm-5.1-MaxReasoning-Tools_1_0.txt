Looking at this problem, I need to build a TSP solver that adapts its strategy based on problem size, ranging from brute-force optimal for tiny instances to fast Hilbert curve approximations for massive ones.

**My adaptive strategy:**
- **N ≤ 11**: Brute force (optimal, 10! feasible)
- **N ≤ 5000**: Nearest Neighbor + full 2-opt (O(N²) feasible)
- **N ≤ 80000**: Hilbert curve + full 2-opt
- **N ≤ 1M**: Hilbert curve + limited 2-opt (K=100)
- **N > 1M**: Hilbert curve + very limited 2-opt (K=30)

Key design decisions: fast custom I/O for millions of cities, squared distances for comparisons (avoid sqrt), don't-look bits for 2-opt acceleration, and bidirectional Hilbert traversal selection.

```cpp
#include <cstdio>
#include <cmath>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <vector>
#include <numeric>
#include <cstdint>

using namespace std;

static int N;
struct City { double x, y; };
static City* cities;

inline double dist2(int a, int b) {
    double dx = cities[a].x - cities[b].x;
    double dy = cities[a].y - cities[b].y;
    return dx * dx + dy * dy;
}

// ---- Fast I/O ----
static char rbuf[1 << 22];
static int rpos = 0, rlen = 0;

inline char rchar() {
    if (rpos == rlen) {
        rlen = (int)fread(rbuf, 1, sizeof(rbuf), stdin);
        rpos = 0;
        if (rlen == 0) return 0;
    }
    return rbuf[rpos++];
}

inline int read_int() {
    char c = rchar();
    while (c >= 0 && c <= ' ') c = rchar();
    int v = 0;
    while (c > ' ') { v = v * 10 + (c - '0'); c = rchar(); }
    return v;
}

inline double read_double() {
    char c = rchar();
    while (c >= 0 && c <= ' ') c = rchar();
    bool neg = false;
    if (c == '-') { neg = true; c = rchar(); }
    double v = 0;
    while (c >= '0' && c <= '9') { v = v * 10 + (c - '0'); c = rchar(); }
    if (c == '.') {
        c = rchar();
        double f = 0.1;
        while (c >= '0' && c <= '9') { v += (c - '0') * f; f *= 0.1; c = rchar(); }
    }
    if (c == 'e' || c == 'E') {
        c = rchar();
        bool eneg = false;
        if (c == '-') { eneg = true; c = rchar(); }
        else if (c == '+') c = rchar();
        int e = 0;
        while (c >= '0' && c <= '9') { e = e * 10 + (c - '0'); c = rchar(); }
        v *= pow(10.0, eneg ? -e : e);
    }
    return neg ? -v : v;
}

static char wbuf[1 << 22];
static int wpos = 0;

inline void wflush() { if (wpos > 0) { fwrite(wbuf, 1, wpos, stdout); wpos = 0; } }

inline void wchar(char c) {
    if (wpos == (int)sizeof(wbuf)) wflush();
    wbuf[wpos++] = c;
}

inline void write_int(int x) {
    if (x == 0) { wchar('0'); return; }
    char tmp[12]; int len = 0;
    while (x > 0) { tmp[len++] = '0' + x % 10; x /= 10; }
    for (int i = len - 1; i >= 0; i--) wchar(tmp[i]);
}

// ---- Brute Force (N <= 11) ----
void brute_force(vector<int>& tour) {
    vector<int> perm(N - 1);
    iota(perm.begin(), perm.end(), 1);
    double best = 1e18;
    vector<int> best_perm = perm;
    do {
        double d = sqrt(dist2(0, perm[0]));
        for (int i = 1; i < N - 1; i++)
            d += sqrt(dist2(perm[i - 1], perm[i]));
        d += sqrt(dist2(perm[N - 2], 0));
        if (d < best) { best = d; best_perm = perm; }
    } while (next_permutation(perm.begin(), perm.end()));
    tour[0] = 0;
    for (int i = 0; i < N - 1; i++) tour[i + 1] = best_perm[i];
}

// ---- Hilbert Curve Sort ----
static uint64_t hilbert_xy2d(int order, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (uint64_t s = (1ULL << (order - 1)); s > 0; s >>= 1) {
        uint64_t rx = (x & s) ? 1 : 0;
        uint64_t ry = (y & s) ? 1 : 0;
        d += s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            uint64_t t = x; x = y; y = t;
        }
    }
    return d;
}

void hilbert_tour(vector<int>& tour) {
    double minx = 1e18, miny = 1e18, maxx = -1e18, maxy = -1e18;
    for (int i = 0; i < N; i++) {
        if (cities[i].x < minx) minx = cities[i].x;
        if (cities[i].y < miny) miny = cities[i].y;
        if (cities[i].x > maxx) maxx = cities[i].x;
        if (cities[i].y > maxy) maxy = cities[i].y;
    }
    double range = maxx - minx;
    if (maxy - miny > range) range = maxy - miny;
    if (range < 1e-10) range = 1.0;

    int order = 1;
    while ((1ULL << order) < (uint64_t)N * 2) order++;
    if (order > 31) order = 31;

    double scale = (double)((1ULL << order) - 1) / range;
    uint64_t maxc = (1ULL << order) - 1;

    struct HI { uint64_t h; int idx; };
    vector<HI> hidx(N);
    for (int i = 0; i < N; i++) {
        uint64_t ix = (uint64_t)((cities[i].x - minx) * scale);
        uint64_t iy = (uint64_t)((cities[i].y - miny) * scale);
        if (ix > maxc) ix = maxc;
        if (iy > maxc) iy = maxc;
        hidx[i].h = hilbert_xy2d(order, ix, iy);
        hidx[i].idx = i;
    }
    sort(hidx.begin(), hidx.end(), [](const HI& a, const HI& b) {
        return a.h < b.h || (a.h == b.h && a.idx < b.idx);
    });

    int start_pos = 0;
    for (int i = 0; i < N; i++) {
        if (hidx[i].idx == 0) { start_pos = i; break; }
    }

    double fwd = 0, bwd = 0;
    for (int i = 0; i < N; i++) {
        fwd += dist2(hidx[(start_pos + i) % N].idx, hidx[(start_pos + i + 1) % N].idx);
        bwd += dist2(hidx[(start_pos - i + N) % N].idx, hidx[(start_pos - i - 1 + N) % N].idx);
    }

    if (fwd <= bwd) {
        for (int i = 0; i < N; i++) tour[i] = hidx[(start_pos + i) % N].idx;
    } else {
        for (int i = 0; i < N; i++) tour[i] = hidx[(start_pos - i + N) % N].idx;
    }
}

// ---- Nearest Neighbor ----
void nn_tour(vector<int>& tour) {
    vector<char> visited(N, 0);
    tour[0] = 0;
    visited[0] = 1;
    int cur = 0;
    for (int step = 1; step < N; step++) {
        double best_d = 1e18;
        int best_next = -1;
        for (int i = 0; i < N; i++) {
            if (!visited[i]) {
                double d = dist2(cur, i);
                if (d < best_d) { best_d = d; best_next = i; }
            }
        }
        tour[step] = best_next;
        visited[best_next] = 1;
        cur = best_next;
    }
}

// ---- Full 2-opt with don't-look bits ----
void two_opt(vector<int>& tour) {
    vector<char> dont_look(N, 0);
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N; i++) {
            int ci = tour[i];
            if (dont_look[ci]) continue;
            bool local_improved = false;
            int ci1 = tour[(i + 1) % N];
            for (int j = i + 2; j < N; j++) {
                if (i == 0 && j == N - 1) continue;
                int cj = tour[j];
                int cj1 = tour[(j + 1) % N];
                double d_old = dist2(ci, ci1) + dist2(cj, cj1);
                double d_new = dist2(ci, cj) + dist2(ci1, cj1);
                if (d_new < d_old - 1e-10) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    dont_look[ci] = 0;
                    dont_look[cj] = 0;
                    dont_look[tour[(i + 1) % N]] = 0;
                    dont_look[cj1] = 0;
                    improved = true;
                    local_improved = true;
                    break;
                }
            }
            if (!local_improved) dont_look[ci] = 1;
        }
    }
}

// ---- Limited 2-opt (nearby positions only) ----
void limited_two_opt(vector<int>& tour, int K, int max_passes) {
    bool improved = true;
    int passes = 0;
    while (improved && passes < max_passes) {
        improved = false;
        passes++;
        for (int i = 0; i < N; i++) {
            bool local_imp = true;
            int tries = 0;
            while (local_imp && tries < 3) {
                local_imp = false;
                tries++;
                int ci = tour[i], ci1 = tour[(i + 1) % N];
                for (int dj = 2; dj <= K && i + dj < N; dj++) {
                    int j = i + dj;
                    if (i == 0 && j == N - 1) continue;
                    int cj = tour[j], cj1 = tour[(j + 1) % N];
                    double d_old = dist2(ci, ci1) + dist2(cj, cj1);
                    double d_new = dist2(ci, cj) + dist2(ci1, cj1);
                    if (d_new < d_old - 1e-10) {
                        reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        improved = true;
                        local_imp = true;
                        break;
                    }
                }
            }
        }
    }
}

int main() {
    N = read_int();
    cities = new City[N];
    for (int i = 0; i < N; i++) {
        cities[i].x = read_double();
        cities[i].y = read_double();
    }

    vector<int> tour(N);

    if (N <= 1) {
        tour[0] = 0;
    } else if (N <= 11) {
        brute_force(tour);
    } else if (N <= 5000) {
        nn_tour(tour);
        two_opt(tour);
    } else if (N <= 80000) {
        hilbert_tour(tour);
        two_opt(tour);
    } else if (N <= 1000000) {
        hilbert_tour(tour);
        limited_two_opt(tour, 100, 20);
    } else {
        hilbert_tour(tour);
        limited_two_opt(tour, 30, 5);
    }

    for (int i = 0; i < N; i++) {
        if (i > 0) wchar(' ');
        write_int(tour[i]);
    }
    wchar('\n');
    wflush();

    delete[] cities;
    return 0;
}
```
