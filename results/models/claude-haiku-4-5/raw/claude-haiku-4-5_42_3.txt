; SPIR-V Assembly - Geometry Shader for Line Expansion to Quad
; Input: InputLines (2 vertices per invocation)
; Output: OutputTriangleStrip (4 vertices per line)
; Task: Expand thin lines into thick ribbons

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor_out
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"

; ============ Type Definitions ============
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; ============ Constants ============
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0p03 = OpConstant %float 0.03
%float_neg1 = OpConstant %float -1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2

; ============ Struct Types ============
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; ============ Array Types ============
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float

; ============ Pointer Types ============
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; ============ Variables ============
%gl_in = OpVariable %ptr_in_arr_pv Input
%color_in = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor_out = OpVariable %ptr_out_v3 Output
OpDecorate %outColor_out Location 0

; ============ Function ============
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input positions: gl_in[0].gl_Position and gl_in[1].gl_Position
%p0_ac = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ac

%p1_ac = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ac

; Load input colors: color[0] and color[1]
%c0_ac = OpAccessChain %ptr_in_v3 %color_in %int_0
%c0 = OpLoad %v3float %c0_ac

%c1_ac = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_ac

; Extract XY coordinates (2D positions from 4D homogeneous coords)
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1

; Compute line direction: dir = p1 - p0
%dir = OpFSub %v2float %p1_xy %p0_xy

; Extract direction components
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

; Compute perpendicular: perp = normalize(-dir.y, dir.x)
%perp_x = OpFNegate %float %dir_y
%perp_y = %dir_x
%perp_unnorm = OpCompositeConstruct %v2float %perp_x %perp_y

; Compute length of direction vector for normalization
%dir_x_sq = OpFMul %float %dir_x %dir_x
%dir_y_sq = OpFMul %float %dir_y %dir_y
%dir_sq = OpFAdd %float %dir_x_sq %dir_y_sq

; Import extended instruction set for Sqrt
%ext = OpExtInstImport "GLSL.std.450"
%dir_len = OpExtInst %float %ext Sqrt %dir_sq

; Normalize perpendicular vector
%perp_norm_x = OpFDiv %float %perp_x %dir_len
%perp_norm_y = OpFDiv %float %perp_y %dir_len
%perp_norm = OpCompositeConstruct %v2float %perp_norm_x %perp_norm_y

; Scale by thickness (0.03)
%perp = OpVectorTimesScalar %v2float %perp_norm %float_0p03

; Extract Z and W from original positions for reconstruction
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

; Create negated perpendicular for opposite offset
%perp_neg = OpVectorTimesScalar %v2float %perp %float_neg1

; ============ Emit Triangle Strip ============
; Vertex 0: p0 + perp
%v0_xy = OpFAdd %v2float %p0_xy %perp
%v0_x = OpCompositeExtract %float %v0_xy 0
%v0_y = OpCompositeExtract %float %v0_xy 1
%v0 = OpCompositeConstruct %v4float %v0_x %v0_y %p0_z %p0_w
%v0_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v0_out %v0
OpStore %outColor_out %c0
OpEmitVertex

; Vertex 1: p0 - perp
%v1_xy = OpFAdd %v2float %p0_xy %perp_neg
%v1_x = OpCompositeExtract %float %v1_xy 0
%v1_y = OpCompositeExtract %float %v1_xy 1
%v1 = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
%v1_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v1_out %v1
OpStore %outColor_out %c0
OpEmitVertex

; Vertex 2: p1 + perp
%v2_xy = OpFAdd %v2float %p1_xy %perp
%v2_x = OpCompositeExtract %float %v2_xy 0
%v2_y = OpCompositeExtract %float %v2_xy 1
%v2 = OpCompositeConstruct %v4float %v2_x %v2_y %p1_z %p1_w
%v2_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v2_out %v2
OpStore %outColor_out %c1
OpEmitVertex

; Vertex 3: p1 - perp
%v3_xy = OpFAdd %v2float %p1_xy %perp_neg
%v3_x = OpCompositeExtract %float %v3_xy 0
%v3_y = OpCompositeExtract %float %v3_xy 1
%v3 = OpCompositeConstruct %v4float %v3_x %v3_y %p1_z %p1_w
%v3_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v3_out %v3
OpStore %outColor_out %c1
OpEmitVertex

; End the primitive (triangle strip)
OpEndPrimitive

OpReturn
OpFunctionEnd
