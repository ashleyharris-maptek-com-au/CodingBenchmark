#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 6x6 polka-dot pattern
    // cellUv is the position within the current cell (0-1 range)
    vec2 cellUv = fract(uv * 6.0);

    // Distance from the center of the cell
    vec2 cellCenter = vec2(0.5, 0.5);
    float distToCenter = distance(cellUv, cellCenter);

    // Determine base color: yellow dot or dark blue background
    vec3 baseColor;
    if (distToCenter < 0.3) {
        baseColor = vec3(1.0, 0.8, 0.0);  // Yellow
    } else {
        baseColor = vec3(0.1, 0.1, 0.3);  // Dark blue
    }

    // Lambertian shading
    vec3 normalNorm = normalize(normal);
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(0.0, dot(normalNorm, lightDir));
    float shadingFactor = 0.3 + 0.7 * diffuse;

    // Apply shading to base color
    vec3 finalColor = baseColor * shadingFactor;

    outColor = vec4(finalColor, 1.0);
}
