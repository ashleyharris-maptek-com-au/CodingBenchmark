OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

%glsl = OpExtInstImport "GLSL.std.450"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%void_func = OpTypeFunction %void

%worldPos = OpVariable %ptr_Input_vec3 Input
%normal = OpVariable %ptr_Input_vec3 Input
%uv = OpVariable %ptr_Input_vec2 Input
%tangent = OpVariable %ptr_Input_vec3 Input
%color = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%uboVar = OpVariable %ptr_Uniform_UBO Uniform

%const_0_0 = OpConstant %float 0.0
%const_1_0 = OpConstant %float 1.0
%const_0_1 = OpConstant %float 0.1
%const_0_05 = OpConstant %float 0.05
%const_32_0 = OpConstant %float 32.0
%const_0_int = OpConstant %int 0
%const_3_int = OpConstant %int 3
%const_4_int = OpConstant %int 4

%ambient_const = OpConstantComposite %vec3 %const_0_1 %const_0_05 %const_0_05
%specular_const = OpConstantComposite %vec3 %const_1_0 %const_1_0 %const_1_0

%main = OpFunction %void None %void_func
%block = OpLabel

%ld_worldPos = OpLoad %vec3 %worldPos
%ld_normal = OpLoad %vec3 %normal
%ld_uv = OpLoad %vec2 %uv
%ld_tangent = OpLoad %vec3 %tangent
%ld_color = OpLoad %vec3 %color

%ubo_lightPos_ptr = OpAccessChain %ptr_Uniform_vec4 %uboVar %const_3_int
%ubo_lightPos = OpLoad %vec4 %ubo_lightPos_ptr
%ubo_cameraPos_ptr = OpAccessChain %ptr_Uniform_vec4 %uboVar %const_4_int
%ubo_cameraPos = OpLoad %vec4 %ubo_cameraPos_ptr

%lightPos = OpVectorShuffle %vec3 %ubo_lightPos %ubo_lightPos 0 1 2
%cameraPos = OpVectorShuffle %vec3 %ubo_cameraPos %ubo_cameraPos 0 1 2

%normal_norm = OpExtInst %vec3 %glsl Normalize %ld_normal

%L_raw = OpFSub %vec3 %lightPos %ld_worldPos
%L = OpExtInst %vec3 %glsl Normalize %L_raw

%V_raw = OpFSub %vec3 %cameraPos %ld_worldPos
%V = OpExtInst %vec3 %glsl Normalize %V_raw

%NL = OpDot %float %normal_norm %L
%NL_clamped = OpExtInst %float %glsl FMax %NL %const_0_0
%diffuse = OpVectorTimesScalar %vec3 %ld_color %NL_clamped

%neg_L = OpFNegate %vec3 %L
%R = OpExtInst %vec3 %glsl Reflect %neg_L %normal_norm
%RV = OpDot %float %R %V
%RV_clamped = OpExtInst %float %glsl FMax %RV %const_0_0
%spec_intensity = OpExtInst %float %glsl Pow %RV_clamped %const_32_0
%specular = OpVectorTimesScalar %vec3 %specular_const %spec_intensity

%ad = OpFAdd %vec3 %ambient_const %diffuse
%rgb = OpFAdd %vec3 %ad %specular

%rgb_clamped = OpExtInst %vec3 %glsl FClamp %rgb %const_0_0 %const_1_0

%final_color = OpCompositeConstruct %vec4 %rgb_clamped %const_1_0

OpStore %outColor %final_color

OpReturn
OpFunctionEnd
