use std::io::{self, BufRead};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read H and W
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let h: usize = parts[0].parse().unwrap();
    let w: usize = parts[1].parse().unwrap();

    // Read the maze
    let mut maze = Vec::new();
    let mut start = (0, 0);
    let mut end = (0, 0);

    for i in 0..h {
        let line = lines.next().unwrap().unwrap();

        // Find start (A) and end (B)
        for (j, ch) in line.chars().enumerate().take(w) {
            if ch == 'A' {
                start = (i, j);
            } else if ch == 'B' {
                end = (i, j);
            }
        }

        maze.push(line);
    }

    // BFS to find shortest path
    let path = bfs(&maze, start, end, h, w);

    // Output
    println!("{}", path.len());
    for (row, col) in path {
        println!("{} {}", col, row);  // x y format (x=col, y=row)
    }
}

fn bfs(maze: &[String], start: (usize, usize), end: (usize, usize), h: usize, w: usize) -> Vec<(usize, usize)> {
    let mut visited = vec![vec![false; w]; h];
    let mut parent = vec![vec![None; w]; h];

    let mut queue = VecDeque::new();
    queue.push_back(start);
    visited[start.0][start.1] = true;

    let directions = [(0, 1), (1, 0), (0, -1), (-1, 0)];  // right, down, left, up

    while let Some((row, col)) = queue.pop_front() {
        if (row, col) == end {
            // Reconstruct path
            let mut path = vec![(row, col)];
            let mut current = (row, col);

            while let Some(prev) = parent[current.0][current.1] {
                path.push(prev);
                current = prev;
            }

            path.reverse();
            return path;
        }

        for (dr, dc) in &directions {
            let nr = (row as i32 + dr) as usize;
            let nc = (col as i32 + dc) as usize;

            if nr < h && nc < w && !visited[nr][nc] {
                if let Some(ch) = maze[nr].chars().nth(nc) {
                    if ch != '#' {
                        visited[nr][nc] = true;
                        parent[nr][nc] = Some((row, col));
                        queue.push_back((nr, nc));
                    }
                }
            }
        }
    }

    // No path found
    vec![]
}
