#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <set>
#include <cstring>

using namespace std;

class GraphColorer {
private:
    int num_vertices;
    int num_edges;
    int num_colors;
    vector<vector<int>> adj;
    vector<int> colors;

public:
    GraphColorer(int v, int e, int c) : num_vertices(v), num_edges(e), num_colors(c) {
        adj.resize(v);
        colors.assign(v, -1);
    }

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    void colorGreedySimple() {
        colors[0] = 0;
        vector<bool> available(num_colors, false);

        for (int u = 1; u < num_vertices; u++) {
            fill(available.begin(), available.end(), false);

            for (int neighbor : adj[u]) {
                if (colors[neighbor] != -1) {
                    available[colors[neighbor]] = true;
                }
            }

            int color = 0;
            for (int c = 0; c < num_colors; c++) {
                if (!available[c]) {
                    color = c;
                    break;
                }
            }
            colors[u] = color;
        }
    }

    void colorWelshPowell() {
        vector<pair<int, int>> degreeOrder;
        for (int i = 0; i < num_vertices; i++) {
            degreeOrder.push_back({adj[i].size(), i});
        }
        sort(degreeOrder.rbegin(), degreeOrder.rend());

        colors.assign(num_vertices, -1);
        vector<bool> available(num_colors, false);

        for (auto& p : degreeOrder) {
            int u = p.second;
            fill(available.begin(), available.end(), false);

            for (int neighbor : adj[u]) {
                if (colors[neighbor] != -1) {
                    available[colors[neighbor]] = true;
                }
            }

            int color = 0;
            for (int c = 0; c < num_colors; c++) {
                if (!available[c]) {
                    color = c;
                    break;
                }
            }
            colors[u] = color;
        }
    }

    void colorSaturationDegree() {
        colors.assign(num_vertices, -1);
        vector<int> saturation(num_vertices, 0);
        vector<bool> available(num_colors);

        int u = 0;
        for (int i = 1; i < num_vertices; i++) {
            if (adj[i].size() > adj[u].size()) {
                u = i;
            }
        }

        colors[u] = 0;

        for (int i = 1; i < num_vertices; i++) {
            int maxSat = -1;
            int nextVertex = -1;

            for (int v = 0; v < num_vertices; v++) {
                if (colors[v] == -1) {
                    if (saturation[v] > maxSat ||
                        (saturation[v] == maxSat && adj[v].size() > adj[nextVertex].size())) {
                        maxSat = saturation[v];
                        nextVertex = v;
                    }
                }
            }

            if (nextVertex == -1) break;

            fill(available.begin(), available.end(), false);
            for (int neighbor : adj[nextVertex]) {
                if (colors[neighbor] != -1) {
                    available[colors[neighbor]] = true;
                }
            }

            int color = 0;
            for (int c = 0; c < num_colors; c++) {
                if (!available[c]) {
                    color = c;
                    break;
                }
            }

            colors[nextVertex] = color;

            for (int neighbor : adj[nextVertex]) {
                if (colors[neighbor] == -1) {
                    saturation[neighbor]++;
                }
            }
        }
    }

    void solve() {
        if (num_vertices < 1000) {
            colorSaturationDegree();
        } else if (num_vertices < 100000) {
            colorWelshPowell();
        } else {
            colorGreedySimple();
        }
    }

    void printColors() {
        for (int i = 0; i < num_vertices; i++) {
            if (i > 0) cout << " ";
            cout << colors[i];
        }
        cout << "\n";
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int num_vertices, num_edges, num_colors;
    cin >> num_vertices >> num_edges >> num_colors;

    GraphColorer colorer(num_vertices, num_edges, num_colors);

    for (int i = 0; i < num_edges; i++) {
        int u, v;
        cin >> u >> v;
        colorer.addEdge(u, v);
    }

    colorer.solve();
    colorer.printColors();

    return 0;
}
