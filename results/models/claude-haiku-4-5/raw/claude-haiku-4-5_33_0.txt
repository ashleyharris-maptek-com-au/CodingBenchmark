using System;
using System.Collections.Generic;
using System.Linq;

class QAPSolver
{
    static void Main()
    {
        int n = int.Parse(Console.ReadLine());

        long[][] flow = new long[n][];
        long[][] distance = new long[n][];

        for (int i = 0; i < n; i++)
        {
            flow[i] = Console.ReadLine().Split().Select(long.Parse).ToArray();
        }

        for (int i = 0; i < n; i++)
        {
            distance[i] = Console.ReadLine().Split().Select(long.Parse).ToArray();
        }

        var solver = new QAPSolver();
        var (cost, permutation) = solver.Solve(flow, distance, n);

        Console.WriteLine(cost);
        Console.WriteLine(string.Join(" ", permutation));
    }

    (long cost, int[] permutation) Solve(long[][] flow, long[][] distance, int n)
    {
        int[] bestPermutation = new int[n];
        for (int i = 0; i < n; i++) bestPermutation[i] = i;
        long bestCost = CalculateCost(bestPermutation, flow, distance);

        if (n <= 10)
        {
            return (bestCost, bestPermutation);
        }

        var random = new Random(42);
        int iterations = Math.Min(100, Math.Max(5, 5000 / n));

        for (int iter = 0; iter < iterations; iter++)
        {
            int[] perm = new int[n];
            for (int i = 0; i < n; i++) perm[i] = i;

            for (int i = n - 1; i > 0; i--)
            {
                int j = random.Next(i + 1);
                int temp = perm[i];
                perm[i] = perm[j];
                perm[j] = temp;
            }

            long cost = TwoOptImprove(perm, flow, distance);

            if (cost < bestCost)
            {
                bestCost = cost;
                Array.Copy(perm, bestPermutation, n);
            }
        }

        return (bestCost, bestPermutation);
    }

    long TwoOptImprove(int[] perm, long[][] flow, long[][] distance)
    {
        int n = perm.Length;
        long cost = CalculateCost(perm, flow, distance);
        bool improved = true;
        int maxIterations = Math.Min(1000, n * n);
        int iteration = 0;

        while (improved && iteration < maxIterations)
        {
            improved = false;
            iteration++;

            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    long delta = CalculateDelta(perm, flow, distance, i, j);

                    if (delta < -1e-10)
                    {
                        Reverse(perm, i + 1, j);
                        cost += delta;
                        improved = true;
                    }
                }
            }
        }

        return cost;
    }

    long CalculateDelta(int[] perm, long[][] flow, long[][] distance, int i, int j)
    {
        int n = perm.Length;
        long delta = 0;

        int a = perm[i];
        int b = perm[i + 1];
        int c = perm[j];
        int d = (j + 1 < n) ? perm[j + 1] : -1;

        delta -= flow[a][b] * distance[perm[i]][perm[i + 1]];
        delta -= flow[b][a] * distance[perm[i + 1]][perm[i]];
        delta -= flow[c][d] * distance[perm[j]][perm[j + 1]];
        delta -= flow[d][c] * distance[perm[j + 1]][perm[j]];

        delta += flow[a][c] * distance[perm[i]][perm[j]];
        delta += flow[c][a] * distance[perm[j]][perm[i]];
        delta += flow[b][d] * distance[perm[i + 1]][perm[j + 1]];
        delta += flow[d][b] * distance[perm[j + 1]][perm[i + 1]];

        for (int k = i + 2; k < j; k++)
        {
            int pk = perm[k];
            delta -= flow[a][pk] * distance[perm[i]][pk];
            delta -= flow[pk][a] * distance[pk][perm[i]];
            delta -= flow[c][pk] * distance[perm[j]][pk];
            delta -= flow[pk][c] * distance[pk][perm[j]];

            delta += flow[a][pk] * distance[perm[i]][pk];
            delta += flow[pk][a] * distance[pk][perm[i]];
            delta += flow[c][pk] * distance[perm[j]][pk];
            delta += flow[pk][c] * distance[pk][perm[j]];
        }

        return delta;
    }

    void Reverse(int[] arr, int start, int end)
    {
        while (start < end)
        {
            int temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
            start++;
            end--;
        }
    }

    long CalculateCost(int[] perm, long[][] flow, long[][] distance)
    {
        int n = perm.Length;
        long cost = 0;

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                cost += flow[i][j] * distance[perm[i]][perm[j]];
            }
        }

        return cost;
    }
}
