using System;
using System.Collections.Generic;
using System.Linq;

class OrbitalTSP
{
    const double MU = 398600.4418;
    const double EARTH_RADIUS = 6371.0;

    struct Orbit
    {
        public double X, Y, Z, Vx, Vy, Vz;

        public double GetRadius() => Math.Sqrt(X*X + Y*Y + Z*Z);
    }

    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        double mu = double.Parse(firstLine[1]);

        var orbits = new Orbit[n + 1];

        // Read start orbit
        string[] startLine = Console.ReadLine().Split();
        orbits[0] = new Orbit
        {
            X = double.Parse(startLine[0]),
            Y = double.Parse(startLine[1]),
            Z = double.Parse(startLine[2]),
            Vx = double.Parse(startLine[3]),
            Vy = double.Parse(startLine[4]),
            Vz = double.Parse(startLine[5])
        };

        // Read station orbits
        for (int i = 1; i <= n; i++)
        {
            string[] line = Console.ReadLine().Split();
            orbits[i] = new Orbit
            {
                X = double.Parse(line[0]),
                Y = double.Parse(line[1]),
                Z = double.Parse(line[2]),
                Vx = double.Parse(line[3]),
                Vy = double.Parse(line[4]),
                Vz = double.Parse(line[5])
            };
        }

        // Calculate distance matrix (using simplified delta-V approximation)
        double[,] deltaV = new double[n + 1, n + 1];
        for (int i = 0; i <= n; i++)
        {
            for (int j = 0; j <= n; j++)
            {
                if (i != j)
                {
                    deltaV[i, j] = CalculateDeltaV(orbits[i], orbits[j]);
                }
            }
        }

        // Solve TSP based on problem size
        int[] solution;
        if (n <= 10)
        {
            solution = SolveSmall(n, deltaV);
        }
        else if (n <= 20)
        {
            solution = SolveMedium(n, deltaV);
        }
        else
        {
            solution = SolveLarge(n, deltaV);
        }

        Console.WriteLine(string.Join(" ", solution));
    }

    static double CalculateDeltaV(Orbit from, Orbit to)
    {
        // Euclidean distance based approximation
        double dx = to.X - from.X;
        double dy = to.Y - from.Y;
        double dz = to.Z - from.Z;
        double posDist = Math.Sqrt(dx*dx + dy*dy + dz*dz);

        double dvx = to.Vx - from.Vx;
        double dvy = to.Vy - from.Vy;
        double dvz = to.Vz - from.Vz;
        double velDiff = Math.Sqrt(dvx*dvx + dvy*dvy + dvz*dvz);

        // Simple approximation: weighted combination of distance and velocity difference
        return posDist * 0.001 + velDiff;
    }

    static int[] SolveSmall(int n, double[,] deltaV)
    {
        // Nearest neighbor with 2-opt improvement
        int[] best = GreedyNearest(n, deltaV);
        best = TwoOpt(best, deltaV);
        return best;
    }

    static int[] SolveMedium(int n, double[,] deltaV)
    {
        // Greedy nearest neighbor with local search
        int[] best = GreedyNearest(n, deltaV);
        best = TwoOpt(best, deltaV);

        // Try a few random restarts
        for (int attempt = 0; attempt < Math.Min(5, n); attempt++)
        {
            int[] candidate = GreedyNearestFrom(n, deltaV, attempt + 1);
            candidate = TwoOpt(candidate, deltaV);
            if (TourCost(candidate, deltaV) < TourCost(best, deltaV))
            {
                best = candidate;
            }
        }

        return best;
    }

    static int[] SolveLarge(int n, double[,] deltaV)
    {
        // Start with greedy nearest neighbor
        int[] best = GreedyNearest(n, deltaV);
        best = TwoOptLimited(best, deltaV, 100);

        // Limited random restarts
        int attempts = Math.Min(3, n / 10);
        for (int attempt = 0; attempt < attempts; attempt++)
        {
            int[] candidate = GreedyNearestFrom(n, deltaV, attempt + 1);
            candidate = TwoOptLimited(candidate, deltaV, 50);
            if (TourCost(candidate, deltaV) < TourCost(best, deltaV))
            {
                best = candidate;
            }
        }

        return best;
    }

    static int[] GreedyNearest(int n, double[,] deltaV)
    {
        return GreedyNearestFrom(n, deltaV, 0);
    }

    static int[] GreedyNearestFrom(int n, double[,] deltaV, int startStation)
    {
        bool[] visited = new bool[n];
        int[] path = new int[n];

        int current = startStation % n;
        path[0] = current;
        visited[current] = true;

        for (int i = 1; i < n; i++)
        {
            double bestCost = double.MaxValue;
            int bestNext = -1;

            for (int j = 0; j < n; j++)
            {
                if (!visited[j] && deltaV[current, j] < bestCost)
                {
                    bestCost = deltaV[current, j];
                    bestNext = j;
                }
            }

            if (bestNext == -1) bestNext = FindUnvisited(visited, n);
            path[i] = bestNext;
            visited[bestNext] = true;
            current = bestNext;
        }

        return path;
    }

    static int FindUnvisited(bool[] visited, int n)
    {
        for (int i = 0; i < n; i++)
            if (!visited[i]) return i;
        return 0;
    }

    static int[] TwoOpt(int[] path, double[,] deltaV)
    {
        int n = path.Length;
        bool improved = true;

        while (improved)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    double currentCost = deltaV[path[i], path[i + 1]] + deltaV[path[j], path[(j + 1) % n]];
                    double newCost = deltaV[path[i], path[j]] + deltaV[path[i + 1], path[(j + 1) % n]];

                    if (newCost < currentCost)
                    {
                        Array.Reverse(path, i + 1, j - i);
                        improved = true;
                    }
                }
            }
        }

        return path;
    }

    static int[] TwoOptLimited(int[] path, double[,] deltaV, int iterations)
    {
        int n = path.Length;

        for (int iter = 0; iter < iterations; iter++)
        {
            bool improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    double currentCost = deltaV[path[i], path[i + 1]] + deltaV[path[j], path[(j + 1) % n]];
                    double newCost = deltaV[path[i], path[j]] + deltaV[path[i + 1], path[(j + 1) % n]];

                    if (newCost < currentCost)
                    {
                        Array.Reverse(path, i + 1, j - i);
                        improved = true;
                        break;
                    }
                }
                if (improved) break;
            }
            if (!improved) break;
        }

        return path;
    }

    static double TourCost(int[] path, double[,] deltaV)
    {
        double cost = 0;
        int n = path.Length;
        for (int i = 0; i < n; i++)
        {
            cost += deltaV[path[i], path[(i + 1) % n]];
        }
        return cost;
    }
}
