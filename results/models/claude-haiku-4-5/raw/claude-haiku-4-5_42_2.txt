OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %color_in %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos_in Location 0
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_04 = OpConstant %float 0.04
%float_0_02 = OpConstant %float 0.02
%float_0_0346 = OpConstant %float 0.0346410162
%float_neg0_02 = OpConstant %float -0.02
%float_neg0_04 = OpConstant %float -0.04
%float_neg0_0346 = OpConstant %float -0.0346410162

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_6 = OpConstant %int 6

%uint_1 = OpConstant %uint 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr Input

%worldPos_arr = OpTypeArray %v3float %uint_1
%ptr_in_worldPos_arr = OpTypePointer Input %worldPos_arr
%worldPos_in = OpVariable %ptr_in_worldPos_arr Input

%color_arr = OpTypeArray %v3float %uint_1
%ptr_in_color_arr = OpTypePointer Input %color_arr
%color_in = OpVariable %ptr_in_color_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_Function_int = OpTypePointer Function %int
%ptr_Function_v4float = OpTypePointer Function %v4float

%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input point position and color
%pos_chain = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_chain
%color_chain = OpAccessChain %ptr_in_v3float %color_in %int_0
%color = OpLoad %v3float %color_chain

; Extract x, y, z from position
%pos_x = OpCompositeExtract %float %pos 0
%pos_y = OpCompositeExtract %float %pos 1
%pos_z = OpCompositeExtract %float %pos 2

; Hexagon vertices (relative to center, 6 vertices in order)
; Using 0°, 60°, 120°, 180°, 240°, 300° angles
; Offsets: (0.04, 0), (0.02, 0.0346), (-0.02, 0.0346), (-0.04, 0), (-0.02, -0.0346), (0.02, -0.0346)

; Loop counter
%i = OpVariable %ptr_Function_int Function %int_0

OpBranch %loop_start
%loop_start = OpLabel
%i_load = OpLoad %int %i
%cond = OpSLessThan %bool %i_load %int_6
OpLoopMerge %loop_exit %loop_continue None
OpBranchConditional %cond %loop_body %loop_exit

%loop_body = OpLabel

; Emit center vertex
%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos_ptr %pos
OpStore %outColor %color
OpEmitVertex

; Calculate offsets for first edge vertex based on loop index
; Index 0: edge at (0.04, 0), next at (0.02, 0.0346)
; Index 1: edge at (0.02, 0.0346), next at (-0.02, 0.0346)
; Index 2: edge at (-0.02, 0.0346), next at (-0.04, 0)
; Index 3: edge at (-0.04, 0), next at (-0.02, -0.0346)
; Index 4: edge at (-0.02, -0.0346), next at (0.02, -0.0346)
; Index 5: edge at (0.02, -0.0346), next at (0.04, 0)

%i_val = OpLoad %int %i
%switch_cond = OpSwitch %i_val %default 0 %case0 1 %case1 2 %case2 3 %case3 4 %case4 5 %case5

%case0 = OpLabel
%edge0_offset_x = OpFAdd %float %pos_x %float_0_04
%edge0_offset_y = OpFAdd %float %pos_y %float_0
%edge0_offset_z = %pos_z
%edge1_offset_x = OpFAdd %float %pos_x %float_0_02
%edge1_offset_y = OpFAdd %float %pos_y %float_0_0346
%edge1_offset_z = %pos_z
OpBranch %emit_edges

%case1 = OpLabel
%edge0_offset_x_1 = OpFAdd %float %pos_x %float_0_02
%edge0_offset_y_1 = OpFAdd %float %pos_y %float_0_0346
%edge0_offset_z_1 = %pos_z
%edge1_offset_x_1 = OpFAdd %float %pos_x %float_neg0_02
%edge1_offset_y_1 = OpFAdd %float %pos_y %float_0_0346
%edge1_offset_z_1 = %pos_z
%edge0_offset_x = %edge0_offset_x_1
%edge0_offset_y = %edge0_offset_y_1
%edge0_offset_z = %edge0_offset_z_1
%edge1_offset_x = %edge1_offset_x_1
%edge1_offset_y = %edge1_offset_y_1
%edge1_offset_z = %edge1_offset_z_1
OpBranch %emit_edges

%case2 = OpLabel
%edge0_offset_x_2 = OpFAdd %float %pos_x %float_neg0_02
%edge0_offset_y_2 = OpFAdd %float %pos_y %float_0_0346
%edge0_offset_z_2 = %pos_z
%edge1_offset_x_2 = OpFAdd %float %pos_x %float_neg0_04
%edge1_offset_y_2 = OpFAdd %float %pos_y %float_0
%edge1_offset_z_2 = %pos_z
%edge0_offset_x = %edge0_offset_x_2
%edge0_offset_y = %edge0_offset_y_2
%edge0_offset_z = %edge0_offset_z_2
%edge1_offset_x = %edge1_offset_x_2
%edge1_offset_y = %edge1_offset_y_2
%edge1_offset_z = %edge1_offset_z_2
OpBranch %emit_edges

%case3 = OpLabel
%edge0_offset_x_3 = OpFAdd %float %pos_x %float_neg0_04
%edge0_offset_y_3 = OpFAdd %float %pos_y %float_0
%edge0_offset_z_3 = %pos_z
%edge1_offset_x_3 = OpFAdd %float %pos_x %float_neg0_02
%edge1_offset_y_3 = OpFAdd %float %pos_y %float_neg0_0346
%edge1_offset_z_3 = %pos_z
%edge0_offset_x = %edge0_offset_x_3
%edge0_offset_y = %edge0_offset_y_3
%edge0_offset_z = %edge0_offset_z_3
%edge1_offset_x = %edge1_offset_x_3
%edge1_offset_y = %edge1_offset_y_3
%edge1_offset_z = %edge1_offset_z_3
OpBranch %emit_edges

%case4 = OpLabel
%edge0_offset_x_4 = OpFAdd %float %pos_x %float_neg0_02
%edge0_offset_y_4 = OpFAdd %float %pos_y %float_neg0_0346
%edge0_offset_z_4 = %pos_z
%edge1_offset_x_4 = OpFAdd %float %pos_x %float_0_02
%edge1_offset_y_4 = OpFAdd %float %pos_y %float_neg0_0346
%edge1_offset_z_4 = %pos_z
%edge0_offset_x = %edge0_offset_x_4
%edge0_offset_y = %edge0_offset_y_4
%edge0_offset_z = %edge0_offset_z_4
%edge1_offset_x = %edge1_offset_x_4
%edge1_offset_y = %edge1_offset_y_4
%edge1_offset_z = %edge1_offset_z_4
OpBranch %emit_edges

%case5 = OpLabel
%edge0_offset_x_5 = OpFAdd %float %pos_x %float_0_02
%edge0_offset_y_5 = OpFAdd %float %pos_y %float_neg0_0346
%edge0_offset_z_5 = %pos_z
%edge1_offset_x_5 = OpFAdd %float %pos_x %float_0_04
%edge1_offset_y_5 = OpFAdd %float %pos_y %float_0
%edge1_offset_z_5 = %pos_z
%edge0_offset_x = %edge0_offset_x_5
%edge0_offset_y = %edge0_offset_y_5
%edge0_offset_z = %edge0_offset_z_5
%edge1_offset_x = %edge1_offset_x_5
%edge1_offset_y = %edge1_offset_y_5
%edge1_offset_z = %edge1_offset_z_5
OpBranch %emit_edges

%default = OpLabel
OpBranch %loop_continue

%emit_edges = OpLabel

; Emit first edge vertex
%edge0_pos = OpCompositeConstruct %v4float %edge0_offset_x %edge0_offset_y %edge0_offset_z %float_1
OpStore %out_pos_ptr %edge0_pos
OpStore %outColor %color
OpEmitVertex

; Emit second edge vertex
%edge1_pos = OpCompositeConstruct %v4float %edge1_offset_x %edge1_offset_y %edge1_offset_z %float_1
OpStore %out_pos_ptr %edge1_pos
OpStore %outColor %color
OpEmitVertex

OpBranch %loop_continue

%loop_continue = OpLabel
%i_next = OpLoad %int %i
%i_incremented = OpIAdd %int %i_next %int_1
OpStore %i %i_incremented
OpBranch %loop_start

%loop_exit = OpLabel
OpEndPrimitive
OpReturn

OpFunctionEnd
