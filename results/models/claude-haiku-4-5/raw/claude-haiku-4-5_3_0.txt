use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let n: usize = parts[0].parse().unwrap();
    let m: usize = parts[1].parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let parts: Vec<&str> = line.split_whitespace().collect();
            if parts.len() >= 2 {
                if let (Ok(u), Ok(v)) = (parts[0].parse::<usize>(), parts[1].parse::<usize>()) {
                    if u < n && v < n {
                        adj[u].push(v);
                        if u != v {
                            adj[v].push(u);
                        }
                    }
                }
            }
        }
    }

    let positions = layout_graph(n, &adj);

    for (x, y) in positions {
        println!("{} {}", x, y);
    }
}

fn layout_graph(n: usize, adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    // Initialize in a circle
    let mut pos: Vec<(f64, f64)> = (0..n)
        .map(|i| {
            let angle = 2.0 * std::f64::consts::PI * i as f64 / n as f64;
            let r = 100.0;
            (r * angle.cos(), r * angle.sin())
        })
        .collect();

    // Apply spring layout refinement based on graph size
    if n <= 100 {
        spring_layout(&mut pos, adj, 100);
    } else if n <= 500 {
        spring_layout(&mut pos, adj, 50);
    } else if n <= 1000 {
        spring_layout(&mut pos, adj, 30);
    } else if n <= 5000 {
        spring_layout(&mut pos, adj, 15);
    } else if n <= 10000 {
        spring_layout(&mut pos, adj, 8);
    } else if n <= 50000 {
        spring_layout(&mut pos, adj, 3);
    }

    pos
}

fn spring_layout(pos: &mut [(f64, f64)], adj: &[Vec<usize>], iterations: usize) {
    let n = pos.len();

    for iter in 0..iterations {
        let cooling = 1.0 - iter as f64 / (iterations as f64 + 1.0);
        let max_dist_repulsion = if n > 5000 { 150.0 } else if n > 1000 { 300.0 } else { 500.0 };

        let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

        // Repulsive forces between non-adjacent nodes
        for i in 0..n {
            for j in (i + 1)..n {
                let dx = pos[j].0 - pos[i].0;
                let dy = pos[j].1 - pos[i].1;
                let dist_sq = dx * dx + dy * dy;
                let dist = dist_sq.sqrt();

                if dist > 0.01 && dist < max_dist_repulsion {
                    let repel_strength = 100.0 / (dist_sq + 1.0);
                    let fx = repel_strength * dx / (dist + 0.1);
                    let fy = repel_strength * dy / (dist + 0.1);

                    disp[i].0 -= fx;
                    disp[i].1 -= fy;
                    disp[j].0 += fx;
                    disp[j].1 += fy;
                }
            }
        }

        // Attractive forces along edges (spring forces)
        for u in 0..n {
            for &v in &adj[u] {
                if u < v {
                    let dx = pos[v].0 - pos[u].0;
                    let dy = pos[v].1 - pos[u].1;
                    let dist = (dx * dx + dy * dy).sqrt();
                    let dist = dist.max(0.01);

                    let ideal_distance = 50.0;
                    let attract_strength = 0.1 * (dist - ideal_distance);
                    let fx = attract_strength * dx / dist;
                    let fy = attract_strength * dy / dist;

                    disp[u].0 += fx;
                    disp[u].1 += fy;
                    disp[v].0 -= fx;
                    disp[v].1 -= fy;
                }
            }
        }

        // Update positions with cooling schedule
        for i in 0..n {
            let mag = (disp[i].0 * disp[i].0 + disp[i].1 * disp[i].1).sqrt();
            if mag > 0.01 {
                let step = cooling * 2.0 / mag;
                let clamped_step = step.min(1.0);
                pos[i].0 += disp[i].0 * clamped_step;
                pos[i].1 += disp[i].1 * clamped_step;
            }
        }
    }
}
