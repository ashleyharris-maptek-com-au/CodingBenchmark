using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        var adj = new List<int>[n];
        var revAdj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            revAdj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
            revAdj[v].Add(u);
        }

        var removed = FindMinimalFeedbackVertexSet(n, adj, revAdj);

        var result = removed.OrderBy(x => x).ToList();

        Console.WriteLine(result.Count);
        if (result.Count > 0)
        {
            Console.WriteLine(string.Join(" ", result));
        }
    }

    static List<int> FindMinimalFeedbackVertexSet(int n, List<int>[] adj, List<int>[] revAdj)
    {
        var removed = new HashSet<int>();

        while (HasCycle(n, adj, removed))
        {
            int bestVertex = FindBestVertexToRemove(n, adj, revAdj, removed);
            removed.Add(bestVertex);
        }

        return removed.ToList();
    }

    static int FindBestVertexToRemove(int n, List<int>[] adj, List<int>[] revAdj, HashSet<int> removed)
    {
        int bestVertex = 0;
        int maxScore = -1;

        for (int u = 0; u < n; u++)
        {
            if (removed.Contains(u)) continue;

            int outDegree = adj[u].Count(v => !removed.Contains(v));
            int inDegree = revAdj[u].Count(v => !removed.Contains(v));

            int score = inDegree + outDegree;
            if (score > maxScore)
            {
                maxScore = score;
                bestVertex = u;
            }
        }

        return bestVertex;
    }

    static bool HasCycle(int n, List<int>[] adj, HashSet<int> removed)
    {
        var state = new int[n];

        for (int i = 0; i < n; i++)
        {
            if (removed.Contains(i) || state[i] != 0) continue;
            if (HasCycleDFS(i, adj, state, removed))
                return true;
        }

        return false;
    }

    static bool HasCycleDFS(int u, List<int>[] adj, int[] state, HashSet<int> removed)
    {
        state[u] = 1;

        foreach (var v in adj[u])
        {
            if (removed.Contains(v)) continue;

            if (state[v] == 1)
                return true;

            if (state[v] == 0 && HasCycleDFS(v, adj, state, removed))
                return true;
        }

        state[u] = 2;
        return false;
    }
}
