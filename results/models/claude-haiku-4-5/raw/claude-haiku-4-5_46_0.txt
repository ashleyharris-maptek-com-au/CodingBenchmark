#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint bodyIdx = gl_GlobalInvocationID.x;

    // Guard against out-of-bounds access
    if (bodyIdx >= counts.x) {
        return;
    }

    uint N = counts.x;
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    uint posIdx = bodyIdx * 2;
    uint velIdx = bodyIdx * 2 + 1;

    vec4 posData = inData[posIdx];
    vec4 velData = inData[velIdx];

    vec3 pos = posData.xyz;
    float mass = posData.w;
    vec3 vel = velData.xyz;

    // Compute acceleration from all other bodies
    vec3 acc = vec3(0.0);

    for (uint j = 0; j < N; ++j) {
        if (j == bodyIdx) {
            continue;
        }

        uint jPosIdx = j * 2;
        vec4 jPosData = inData[jPosIdx];

        vec3 posJ = jPosData.xyz;
        float massJ = jPosData.w;

        vec3 diff = posJ - pos;
        float distSq = dot(diff, diff);
        float distSqSoftened = distSq + softening * softening;
        float dist = sqrt(distSqSoftened);
        float distCubed = dist * dist * dist;

        acc += G * massJ * diff / distCubed;
    }

    // Update velocity and position
    vec3 newVel = vel + acc * dt;
    vec3 newPos = pos + vel * dt;

    // Write output
    outData[posIdx] = vec4(newPos, mass);
    outData[velIdx] = vec4(newVel, 0.0);
}
