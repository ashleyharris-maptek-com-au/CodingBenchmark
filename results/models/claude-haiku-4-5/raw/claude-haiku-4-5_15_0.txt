use std::io::{self, BufRead};

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Vec2 { x, y }
    }

    fn sub(&self, other: &Vec2) -> Vec2 {
        Vec2 { x: self.x - other.x, y: self.y - other.y }
    }

    fn dot(&self, other: &Vec2) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn len(&self) -> f64 {
        (self.x * self.x + self.y * self.y).sqrt()
    }
}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Vec3 { x, y, z }
    }

    fn normalize(&self) -> Vec3 {
        let len = (self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if len > 1e-10 {
            Vec3 {
                x: self.x / len,
                y: self.y / len,
                z: self.z / len,
            }
        } else {
            *self
        }
    }

    fn scale(&self, s: f64) -> Vec3 {
        Vec3 {
            x: self.x * s,
            y: self.y * s,
            z: self.z * s,
        }
    }

    fn add(&self, other: &Vec3) -> Vec3 {
        Vec3 {
            x: self.x + other.x,
            y: self.y + other.y,
            z: self.z + other.z,
        }
    }
}

#[derive(Clone, Copy, Debug)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quat {
    fn identity() -> Self {
        Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 }
    }

    fn rotate_vec(&self, v: &Vec3) -> Vec3 {
        let qx = self.x;
        let qy = self.y;
        let qz = self.z;
        let qw = self.w;

        let vx = v.x;
        let vy = v.y;
        let vz = v.z;

        let uvx = qy * vz - qz * vy;
        let uvy = qz * vx - qx * vz;
        let uvz = qx * vy - qy * vx;

        let uuvx = qy * uvz - qz * uvy;
        let uuvy = qz * uvx - qx * uvz;
        let uuvz = qx * uvy - qy * uvx;

        Vec3 {
            x: vx + 2.0 * (qw * uvx + uuvx),
            y: vy + 2.0 * (qw * uvy + uuvy),
            z: vz + 2.0 * (qw * uvz + uuvz),
        }
    }
}

fn project_to_z0(p: &Vec3, sun_dir: &Vec3) -> Vec2 {
    if sun_dir.z.abs() < 1e-10 {
        return Vec2::new(p.x, p.y);
    }

    let t = -p.z / sun_dir.z;
    Vec2::new(
        p.x + t * sun_dir.x,
        p.y + t * sun_dir.y,
    )
}

fn get_tetrahedron_shadow(pos: &Vec3, rot: &Quat, scale: f64, sun_dir: &Vec3) -> Vec<Vec2> {
    let tet_verts = vec![
        Vec3::new(1.0, 0.0, -0.707),
        Vec3::new(-1.0, 0.0, -0.707),
        Vec3::new(0.0, 1.0, 0.707),
        Vec3::new(0.0, -1.0, 0.707),
    ];

    let mut shadow = Vec::new();
    for v in &tet_verts {
        let rotated = rot.rotate_vec(v);
        let scaled = rotated.scale(scale);
        let world = scaled.add(pos);
        let shadow_pt = project_to_z0(&world, sun_dir);
        shadow.push(shadow_pt);
    }

    shadow
}

fn convex_hull(mut points: Vec<Vec2>) -> Vec<Vec2> {
    if points.len() < 3 {
        return points;
    }

    points.sort_by(|a, b| {
        if (a.x - b.x).abs() < 1e-10 {
            a.y.partial_cmp(&b.y).unwrap()
        } else {
            a.x.partial_cmp(&b.x).unwrap()
        }
    });

    let mut lower = Vec::new();
    for p in &points {
        while lower.len() >= 2 {
            let o = lower[lower.len() - 1];
            let a = lower[lower.len() - 2];
            let cross = (o.x - a.x) * (p.y - a.y) - (o.y - a.y) * (p.x - a.x);
            if cross <= 0.0 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(*p);
    }

    let mut upper = Vec::new();
    for p in points.iter().rev() {
        while upper.len() >= 2 {
            let o = upper[upper.len() - 1];
            let a = upper[upper.len() - 2];
            let cross = (o.x - a.x) * (p.y - a.y) - (o.y - a.y) * (p.x - a.x);
            if cross <= 0.0 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(*p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn polygon_area(poly: &[Vec2]) -> f64 {
    if poly.len() < 3 {
        return 0.0;
    }

    let mut area = 0.0;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        area += poly[i].x * poly[j].y;
        area -= poly[j].x * poly[i].y;
    }
    area.abs() / 2.0
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();
    let mut polygon = Vec::new();

    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.trim().split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        polygon.push(Vec2::new(x, y));
    }

    let sun_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = sun_line.trim().split_whitespace().collect();
    let sx: f64 = parts[0].parse().unwrap();
    let sy: f64 = parts[1].parse().unwrap();
    let sz: f64 = parts[2].parse().unwrap();
    let sun_dir = Vec3::new(sx, sy, sz).normalize();

    let poly_area = polygon_area(&polygon);

    let mut min_x = polygon[0].x;
    let mut max_x = polygon[0].x;
    let mut min_y = polygon[0].y;
    let mut max_y = polygon[0].y;

    for p in &polygon {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }

    let mut centroid = Vec2::new(0.0, 0.0);
    for p in &polygon {
        centroid.x += p.x;
        centroid.y += p.y;
    }
    centroid.x /= polygon.len() as f64;
    centroid.y /= polygon.len() as f64;

    let width = max_x - min_x;
    let height = max_y - min_y;
    let diagonal = (width * width + height * height).sqrt();

    let mut tetrahedrons = Vec::new();

    let shadow_sample = get_tetrahedron_shadow(&Vec3::new(0.0, 0.0, 3.0), &Quat::identity(), 1.0, &sun_dir);
    let shadow_hull = convex_hull(shadow_sample);
    let shadow_area = polygon_area(&shadow_hull);

    if shadow_area > 1e-10 {
        let scale_factor = ((poly_area / shadow_area).sqrt() * 1.2).min(10.0).max(0.1);

        tetrahedrons.push((
            Vec3::new(centroid.x, centroid.y, 3.0),
            Quat::identity(),
            scale_factor,
        ));
    }

    println!("{}", tetrahedrons.len());
    for (pos, rot, scale) in tetrahedrons {
        println!("{} {} {} {} {} {} {} {}",
            pos.x, pos.y, pos.z,
            rot.w, rot.x, rot.y, rot.z,
            scale
        );
    }
}
