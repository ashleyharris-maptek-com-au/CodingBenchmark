#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // Map UVs to complex plane: c = (uv.x*3-2, uv.y*2-1)
    vec2 c = vec2(uv.x * 3.0 - 2.0, uv.y * 2.0 - 1.0);

    // Mandelbrot iteration
    vec2 z = vec2(0.0, 0.0);
    int iterations = 0;

    for (int i = 0; i < 20; i++) {
        // z = z^2 + c
        // (a + bi)^2 = a^2 - b^2 + 2abi
        float real = z.x * z.x - z.y * z.y + c.x;
        float imag = 2.0 * z.x * z.y + c.y;
        z = vec2(real, imag);

        iterations++;

        // Check escape condition
        if (dot(z, z) > 4.0) {
            break;
        }
    }

    // Color based on iterations
    float t = float(iterations) / 20.0;
    vec3 finalColor = vec3(t, t * 0.5, 1.0 - t);

    outColor = vec4(finalColor, 1.0);
}
