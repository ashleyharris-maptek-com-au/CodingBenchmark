#include <iostream>
#include <vector>
#include <set>
#include <deque>
#include <cmath>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    cin >> D >> max_turns;

    vector<int> bounds(D);
    for (int i = 0; i < D; i++) {
        cin >> bounds[i];
    }

    int snake_length;
    vector<int> head(D);
    cin >> snake_length;
    for (int i = 0; i < D; i++) {
        cin >> head[i];
    }

    int F;
    cin >> F;
    vector<vector<int>> food;
    for (int i = 0; i < F; i++) {
        vector<int> pos(D);
        for (int j = 0; j < D; j++) {
            cin >> pos[j];
        }
        food.push_back(pos);
    }

    int O;
    cin >> O;
    set<vector<int>> obstacles;
    for (int i = 0; i < O; i++) {
        vector<int> pos(D);
        for (int j = 0; j < D; j++) {
            cin >> pos[j];
        }
        obstacles.insert(pos);
    }

    deque<vector<int>> body;
    body.push_back(head);

    for (int turn = 0; turn < max_turns; turn++) {
        if (food.empty()) break;

        // Find closest food using Manhattan distance
        int best_food_idx = 0;
        long long best_dist = LLONG_MAX;
        for (int i = 0; i < food.size(); i++) {
            long long dist = 0;
            for (int j = 0; j < D; j++) {
                dist += abs(food[i][j] - head[j]);
            }
            if (dist < best_dist) {
                best_dist = dist;
                best_food_idx = i;
            }
        }

        vector<int>& target = food[best_food_idx];

        int move_axis = -1;
        int move_dir = 0;

        // Strategy 1: Move along axis that brings us closer to target
        for (int axis = 0; axis < D; axis++) {
            if (target[axis] == head[axis]) continue;

            int dir = (target[axis] > head[axis]) ? 1 : -1;
            vector<int> next_head = head;
            next_head[axis] += dir;

            // Check bounds
            if (next_head[axis] < 0 || next_head[axis] >= bounds[axis]) continue;

            // Check obstacles
            if (obstacles.count(next_head)) continue;

            // Check self-collision (body except the tail which will be removed)
            bool collision = false;
            for (size_t i = 0; i < body.size() - 1; i++) {
                if (body[i] == next_head) {
                    collision = true;
                    break;
                }
            }
            if (collision) continue;

            move_axis = axis;
            move_dir = dir;
            break;
        }

        // Strategy 2: If can't move towards food, try any valid move
        if (move_axis == -1) {
            for (int axis = 0; axis < D; axis++) {
                for (int dir : {-1, 1}) {
                    vector<int> next_head = head;
                    next_head[axis] += dir;

                    if (next_head[axis] < 0 || next_head[axis] >= bounds[axis]) continue;
                    if (obstacles.count(next_head)) continue;

                    bool collision = false;
                    for (size_t i = 0; i < body.size() - 1; i++) {
                        if (body[i] == next_head) {
                            collision = true;
                            break;
                        }
                    }
                    if (collision) continue;

                    move_axis = axis;
                    move_dir = dir;
                    break;
                }
                if (move_axis != -1) break;
            }
        }

        // If no valid move found, we're stuck
        if (move_axis == -1) break;

        // Execute the move
        vector<int> next_head = head;
        next_head[move_axis] += move_dir;
        head = next_head;
        body.push_front(head);

        // Check if we ate food
        bool ate_food = false;
        for (size_t i = 0; i < food.size(); i++) {
            if (food[i] == head) {
                food.erase(food.begin() + i);
                ate_food = true;
                break;
            }
        }

        // Remove tail if we didn't eat food (otherwise snake grows)
        if (!ate_food) {
            body.pop_back();
        }

        // Output the move
        cout << move_axis << " " << move_dir << "\n";
    }

    return 0;
}
