use std::io::{self, BufRead};
use std::collections::HashMap;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse input: first line contains num_numbers and target
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let _n: usize = parts.next().unwrap().parse().unwrap();
    let target: i64 = parts.next().unwrap().parse().unwrap();

    // Parse second line: space-separated numbers
    let numbers: Vec<i64> = lines
        .next()
        .unwrap()
        .unwrap()
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    // Solve subset sum problem
    if let Some(indices) = solve_subset_sum(&numbers, target) {
        println!("YES");
        print!("[");
        for (i, idx) in indices.iter().enumerate() {
            if i > 0 { print!(" "); }
            print!("{}", idx);
        }
        println!("]");
    } else {
        println!("NO");
    }
}

fn solve_subset_sum(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    // Edge case: empty set
    if numbers.is_empty() {
        return if target == 0 { Some(vec![]) } else { None };
    }

    // Dynamic programming using HashMap to track achievable sums
    // dp[sum] = indices that produce this sum
    let mut dp: HashMap<i64, Vec<usize>> = HashMap::new();
    dp.insert(0, vec![]);

    // Process each number
    for (i, &num) in numbers.iter().enumerate() {
        // Collect current sums to avoid modifying HashMap during iteration
        let sums_to_update: Vec<i64> = dp.keys().copied().collect();

        // Try adding current number to each achievable sum
        for sum in sums_to_update {
            let new_sum = sum + num;

            // Only add if this sum hasn't been achieved yet
            if !dp.contains_key(&new_sum) {
                let mut indices = dp[&sum].clone();
                indices.push(i);
                dp.insert(new_sum, indices);
            }
        }
    }

    // Return the indices for target sum if it exists
    dp.get(&target).cloned()
}
