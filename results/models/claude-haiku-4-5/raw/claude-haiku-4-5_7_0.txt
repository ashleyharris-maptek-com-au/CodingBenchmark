#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <unordered_map>

using namespace std;

const double EPS = 1e-9;

struct Vec3 {
    double x, y, z;

    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double t) const { return Vec3(x * t, y * t, z * t); }
    Vec3 operator/(double t) const { return Vec3(x / t, y / t, z / t); }

    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }
    double length() const { return sqrt(x * x + y * y + z * z); }
    double length2() const { return x * x + y * y + z * z; }
    Vec3 normalized() const {
        double len = length();
        if (len < EPS) return Vec3(0, 0, 0);
        return Vec3(x / len, y / len, z / len);
    }
};

struct Face {
    int a, b, c;
};

struct Mesh {
    vector<Vec3> vertices;
    vector<Face> faces;
};

double signedDistance(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 ab = b - a;
    Vec3 ac = c - a;
    Vec3 ap = p - a;
    Vec3 normal = ab.cross(ac);
    return normal.dot(ap);
}

bool pointInMesh(const Vec3& p, const Mesh& mesh) {
    if (mesh.faces.empty()) return false;

    int count = 0;
    for (const auto& face : mesh.faces) {
        double d = signedDistance(p, mesh.vertices[face.a], mesh.vertices[face.b], mesh.vertices[face.c]);
        if (d < -EPS) count++;
    }

    return count > (int)mesh.faces.size() / 2;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh A, B;

    A.vertices.resize(VA);
    for (int i = 0; i < VA; i++) {
        cin >> A.vertices[i].x >> A.vertices[i].y >> A.vertices[i].z;
    }
    A.faces.resize(FA);
    for (int i = 0; i < FA; i++) {
        cin >> A.faces[i].a >> A.faces[i].b >> A.faces[i].c;
    }

    B.vertices.resize(VB);
    for (int i = 0; i < VB; i++) {
        cin >> B.vertices[i].x >> B.vertices[i].y >> B.vertices[i].z;
    }
    B.faces.resize(FB);
    for (int i = 0; i < FB; i++) {
        cin >> B.faces[i].a >> B.faces[i].b >> B.faces[i].c;
    }

    Mesh result;
    result.vertices = A.vertices;

    for (const auto& face : A.faces) {
        Vec3 center = (A.vertices[face.a] + A.vertices[face.b] + A.vertices[face.c]) * (1.0/3.0);
        if (!pointInMesh(center, B)) {
            result.faces.push_back(face);
        }
    }

    int offset = A.vertices.size();
    for (const auto& v : B.vertices) {
        result.vertices.push_back(v);
    }

    for (const auto& face : B.faces) {
        Vec3 center = (B.vertices[face.a] + B.vertices[face.b] + B.vertices[face.c]) * (1.0/3.0);
        if (!pointInMesh(center, A)) {
            result.faces.push_back({face.a + offset, face.b + offset, face.c + offset});
        }
    }

    cout << result.vertices.size() << " " << result.faces.size() << "\n";
    for (const auto& v : result.vertices) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& f : result.faces) {
        cout << f.a << " " << f.b << " " << f.c << "\n";
    }

    return 0;
}
