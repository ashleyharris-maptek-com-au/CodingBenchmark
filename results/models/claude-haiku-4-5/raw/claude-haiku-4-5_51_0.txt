import math
from collections import defaultdict

# Module-level PID state persistence
pid_state = {
    'alt_integral': 0,
    'hdg_integral': 0,
    'ias_integral': 0,
    'vs_integral': 0,
    'roll_integral': 0,
    'pitch_integral': 0,
    'prev_alt_error': 0,
    'prev_hdg_error': 0,
    'prev_ias_error': 0,
    'prev_vs_error': 0,
    'prev_roll_error': 0,
    'prev_pitch_error': 0,
    'trim_state': 0,
}

def get_sensor_value(state, keys, prefer_valid=True):
    """Sensor fusion: pick best value from redundant sensors."""
    values = []
    valids = []

    for key in keys:
        if key in state:
            values.append(state[key])
            valids.append(state.get(f"{key}_valid", True))

    if not values:
        return None

    if prefer_valid:
        valid_values = [v for v, valid in zip(values, valids) if valid]
        if valid_values:
            return sum(valid_values) / len(valid_values)

    return sum(values) / len(values)

def get_altitude(state):
    """Fuse barometric, radar, and GPS altitudes."""
    baro_alt = get_sensor_value(state, ['baro_alt_1_ft', 'baro_alt_2_ft', 'baro_alt_3_ft'])
    radar_alt = get_sensor_value(state, ['radar_alt_1_ft', 'radar_alt_2_ft', 'radar_alt_3_ft'])
    gps_alt = get_sensor_value(state, ['gps_1_alt_ft', 'gps_2_alt_ft', 'gps_3_alt_ft'])

    # Low altitude: prefer radar
    if radar_alt and radar_alt < 2500:
        return radar_alt

    # High altitude: prefer baro
    if baro_alt:
        return baro_alt

    return gps_alt or 0

def get_airspeed(state):
    """Fuse IAS from three ADCs."""
    return get_sensor_value(state, ['ias_1_kt', 'ias_2_kt', 'ias_3_kt']) or 0

def get_heading(state):
    """Fuse magnetic heading from IRS units."""
    hdg = get_sensor_value(state, ['irs_1_heading_deg', 'irs_2_heading_deg', 'irs_3_heading_deg'])
    return hdg or 0

def get_vertical_speed(state):
    """Fuse vertical speed."""
    vs = get_sensor_value(state, ['irs_1_vs_fpm', 'irs_2_vs_fpm', 'irs_3_vs_fpm'])
    return vs or 0

def get_pitch(state):
    """Fuse pitch attitude."""
    pitch = get_sensor_value(state, ['irs_1_pitch_deg', 'irs_2_pitch_deg', 'irs_3_pitch_deg'])
    return pitch or 0

def get_roll(state):
    """Fuse roll attitude."""
    roll = get_sensor_value(state, ['irs_1_roll_deg', 'irs_2_roll_deg', 'irs_3_roll_deg'])
    return roll or 0

def pid_control(error, integral, prev_error, kp, ki, kd, dt, max_output=1.0):
    """Standard PID controller."""
    integral += error * dt
    integral = max(-max_output, min(max_output, integral))

    derivative = (error - prev_error) / dt if dt > 0 else 0

    output = kp * error + ki * integral + kd * derivative
    output = max(-max_output, min(max_output, output))

    return output, integral, derivative

def normalize_heading_error(error):
    """Normalize heading error to -180 to 180."""
    while error > 180:
        error -= 360
    while error < -180:
        error += 360
    return error

def autopilot_step(state: dict, dt: float) -> dict:
    global pid_state

    # Extract current state
    current_alt = get_altitude(state)
    target_alt = state.get('target_alt_ft', current_alt)

    current_ias = get_airspeed(state)
    target_ias = state.get('target_ias_kt', current_ias)

    current_hdg = get_heading(state)
    target_hdg = state.get('target_hdg_deg', current_hdg)

    current_vs = get_vertical_speed(state)
    target_vs = state.get('target_vs_fpm', 0)

    current_pitch = get_pitch(state)
    current_roll = get_roll(state)

    elevator_trim = state.get('elevator_trim_deg', 0)
    weight_on_wheels = (state.get('weight_on_wheels_left', 0) +
                        state.get('weight_on_wheels_right', 0) +
                        state.get('weight_on_wheels_nose', 0)) > 0

    # Check engine status
    eng1_running = state.get('engine_1_running', False)
    eng2_running = state.get('engine_2_running', False)
    engines_running = eng1_running or eng2_running

    # Default outputs
    output = {}

    if not state.get('ap_engaged', False) or not engines_running:
        return output

    # === ALTITUDE CONTROL ===
    alt_error = target_alt - current_alt

    # Target VS based on altitude error (lazy leveling)
    if abs(alt_error) > 500:
        target_vs_calc = max(-2000, min(2000, alt_error * 4))
    else:
        target_vs_calc = 0

    vs_error = target_vs_calc - current_vs
    pitch_cmd, pid_state['vs_integral'], _ = pid_control(
        vs_error, pid_state['vs_integral'], pid_state['prev_vs_error'],
        kp=0.02, ki=0.001, kd=0.1, dt=dt, max_output=1.0
    )
    pid_state['prev_vs_error'] = vs_error

    # === PITCH CONTROL (for trim) ===
    pitch_error = -pitch_cmd  # Negative feedback
    trim_adjustment, pid_state['pitch_integral'], _ = pid_control(
        pitch_error, pid_state['pitch_integral'], pid_state['prev_pitch_error'],
        kp=0.01, ki=0.0001, kd=0.05, dt=dt, max_output=1.0
    )
    pid_state['prev_pitch_error'] = pitch_error

    output['elevator_trim'] = max(-1.0, min(1.0, pid_state['trim_state'] + trim_adjustment * dt))
    pid_state['trim_state'] = output['elevator_trim']

    output['elevator'] = max(-1.0, min(1.0, pitch_cmd))

    # === AIRSPEED CONTROL ===
    ias_error = target_ias - current_ias

    # Throttle for speed control
    throttle_cmd, pid_state['ias_integral'], _ = pid_control(
        ias_error, pid_state['ias_integral'], pid_state['prev_ias_error'],
        kp=0.005, ki=0.0001, kd=0.1, dt=dt, max_output=1.0
    )
    pid_state['prev_ias_error'] = ias_error

    current_throttle = (state.get('throttle_1_pos', 0.3) + state.get('throttle_2_pos', 0.3)) / 2
    new_throttle = max(0.1, min(1.0, current_throttle + throttle_cmd * dt))

    output['throttle_1'] = new_throttle
    output['throttle_2'] = new_throttle

    # === HEADING CONTROL ===
    hdg_error = normalize_heading_error(target_hdg - current_hdg)

    roll_cmd, pid_state['hdg_integral'], _ = pid_control(
        hdg_error, pid_state['hdg_integral'], pid_state['prev_hdg_error'],
        kp=0.5, ki=0.01, kd=0.2, dt=dt, max_output=1.0
    )
    pid_state['prev_hdg_error'] = hdg_error

    # === ROLL CONTROL ===
    roll_error = -current_roll - roll_cmd
    aileron_cmd, pid_state['roll_integral'], _ = pid_control(
        roll_error, pid_state['roll_integral'], pid_state['prev_roll_error'],
        kp=0.05, ki=0.001, kd=0.1, dt=dt, max_output=1.0
    )
    pid_state['prev_roll_error'] = roll_error

    output['aileron'] = max(-1.0, min(1.0, aileron_cmd))

    # === YAW DAMPING ===
    gyro_yaw = get_sensor_value(state, ['gyro_1_yaw_rate_dps', 'gyro_2_yaw_rate_dps', 'gyro_3_yaw_rate_dps']) or 0
    output['rudder'] = max(-0.5, min(0.5, -gyro_yaw * 0.01))

    # === STALL PREVENTION ===
    aoa = get_sensor_value(state, ['aoa_1_deg', 'aoa_2_deg', 'aoa_3_deg']) or 0
    if aoa > 16:  # Stall warning threshold
        output['elevator'] = max(-1.0, output.get('elevator', 0) - 0.3)

    # === FLAPS & SLATS ===
    if current_ias > 250:
        output['flaps'] = 0
    elif current_ias > 200:
        output['flaps'] = 15
    elif current_ias > 150:
        output['flaps'] = 25
    else:
        output['flaps'] = 40

    # === LANDING GEAR ===
    if weight_on_wheels:
        output['gear'] = 1
    elif current_ias < 180 and current_alt < 5000:
        output['gear'] = 1
    else:
        output['gear'] = 0

    # === ASYMMETRIC THRUST COMPENSATION ===
    if eng1_running and eng2_running:
        eng1_thrust = state.get('engine_1_thrust_pct', 50)
        eng2_thrust = state.get('engine_2_thrust_pct', 50)
        thrust_diff = (eng1_thrust - eng2_thrust) / 100.0
        output['rudder'] = max(-1.0, min(1.0, output.get('rudder', 0) + thrust_diff * 0.3))

    return output
