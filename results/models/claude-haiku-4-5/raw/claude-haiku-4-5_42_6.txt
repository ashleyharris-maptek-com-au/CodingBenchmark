OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0

%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float

%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3float Output

%func_void = OpTypeFunction %void
%main = OpFunction %void None %func_void
%entry = OpLabel

; Vertex 0
%pos_0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos_0 = OpLoad %v4float %pos_0_ptr
%out_pos_0_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos_0_ptr %pos_0

%col_0_ptr = OpAccessChain %ptr_in_v3float %color %int_0
%col_0 = OpLoad %v3float %col_0_ptr
OpStore %outColor %col_0

OpEmitVertex

; Vertex 1
%pos_1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%pos_1 = OpLoad %v4float %pos_1_ptr
%out_pos_1_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos_1_ptr %pos_1

%col_1_ptr = OpAccessChain %ptr_in_v3float %color %int_1
%col_1 = OpLoad %v3float %col_1_ptr
OpStore %outColor %col_1

OpEmitVertex

; Vertex 2
%pos_2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%pos_2 = OpLoad %v4float %pos_2_ptr
%out_pos_2_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos_2_ptr %pos_2

%col_2_ptr = OpAccessChain %ptr_in_v3float %color %int_2
%col_2 = OpLoad %v3float %col_2_ptr
OpStore %outColor %col_2

OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
