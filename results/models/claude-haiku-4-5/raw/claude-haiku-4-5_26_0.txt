use std::io::{BufReader, BufWriter, Read, Write};

fn main() {
    let stdin = std::io::stdin();
    let stdout = std::io::stdout();
    let mut reader = BufReader::new(stdin);
    let mut writer = BufWriter::new(stdout);

    // Read num_points and num_clusters
    let mut buf = [0u8; 8];
    reader.read_exact(&mut buf).unwrap();
    let num_points = u32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]) as usize;
    let num_clusters = u32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]) as usize;

    // Adaptive algorithm selection based on dataset size
    let assignments = if num_points <= 10_000_000 {
        full_kmeans(num_points, num_clusters, &mut reader)
    } else if num_points <= 100_000_000 {
        minibatch_kmeans(num_points, num_clusters, &mut reader)
    } else {
        streaming_kmeans(num_points, num_clusters, &mut reader)
    };

    // Ensure all clusters are represented
    let assignments = ensure_all_clusters(assignments, num_clusters);

    // Write binary output
    for &cluster in &assignments {
        let bytes = (cluster as u32).to_le_bytes();
        writer.write_all(&bytes).unwrap();
    }
    writer.flush().unwrap();
}

fn ensure_all_clusters(mut assignments: Vec<usize>, num_clusters: usize) -> Vec<usize> {
    if num_clusters == 0 || assignments.is_empty() {
        return assignments;
    }

    let mut cluster_counts = vec![0usize; num_clusters];
    for &c in &assignments {
        cluster_counts[c] += 1;
    }

    let mut empty_clusters: Vec<usize> = (0..num_clusters)
        .filter(|i| cluster_counts[*i] == 0)
        .collect();

    while !empty_clusters.is_empty() {
        let empty = empty_clusters.pop().unwrap();

        for point_idx in 0..assignments.len() {
            let c = assignments[point_idx];
            if cluster_counts[c] > 1 {
                cluster_counts[c] -= 1;
                cluster_counts[empty] += 1;
                assignments[point_idx] = empty;
                break;
            }
        }
    }

    assignments
}

fn full_kmeans(num_points: usize, num_clusters: usize, reader: &mut BufReader<std::io::Stdin>) -> Vec<usize> {
    let mut points = Vec::with_capacity(num_points);
    let mut buf = [0u8; 12];

    for _ in 0..num_points {
        reader.read_exact(&mut buf).unwrap();
        let x = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
        let y = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
        let z = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);
        points.push([x, y, z]);
    }

    kmeans_clustering(&points, num_clusters)
}

fn minibatch_kmeans(num_points: usize, num_clusters: usize, reader: &mut BufReader<std::io::Stdin>) -> Vec<usize> {
    let batch_size = 100_000.min(num_points / 10).max(1000);
    let mut assignments = vec![0usize; num_points];
    let mut centroids = vec![[0.0; 3]; num_clusters];
    let mut counts = vec![0u32; num_clusters];

    let mut buffer = vec![0u8; batch_size * 12];
    let mut point_idx = 0;
    let mut remaining = num_points;

    while remaining > 0 {
        let batch_len = batch_size.min(remaining);
        let read_len = batch_len * 12;
        let buf = &mut buffer[..read_len];
        reader.read_exact(buf).unwrap();

        for i in 0..batch_len {
            let base = i * 12;
            let x = f32::from_le_bytes([buf[base], buf[base+1], buf[base+2], buf[base+3]]);
            let y = f32::from_le_bytes([buf[base+4], buf[base+5], buf[base+6], buf[base+7]]);
            let z = f32::from_le_bytes([buf[base+8], buf[base+9], buf[base+10], buf[base+11]]);

            let mut min_dist = f32::INFINITY;
            let mut best = 0;

            for j in 0..num_clusters {
                let d = distance(x, y, z, centroids[j]);
                if d < min_dist {
                    min_dist = d;
                    best = j;
                }
            }

            assignments[point_idx] = best;
            centroids[best][0] += x;
            centroids[best][1] += y;
            centroids[best][2] += z;
            counts[best] += 1;
            point_idx += 1;
        }

        remaining -= batch_len;
    }

    for j in 0..num_clusters {
        if counts[j] > 0 {
            centroids[j][0] /= counts[j] as f32;
            centroids[j][1] /= counts[j] as f32;
            centroids[j][2] /= counts[j] as f32;
        }
    }

    assignments
}

fn streaming_kmeans(num_points: usize, num_clusters: usize, reader: &mut BufReader<std::io::Stdin>) -> Vec<usize> {
    let mut assignments = vec![0usize; num_points];
    let mut centroids = vec![[0.0; 3]; num_clusters];
    let mut counts = vec![0u32; num_clusters];
    let mut buf = [0u8; 12];

    for i in 0..num_points {
        reader.read_exact(&mut buf).unwrap();
        let x = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
        let y = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
        let z = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);

        let mut min_dist = f32::INFINITY;
        let mut best = 0;

        for j in 0..num_clusters {
            let d = distance(x, y, z, centroids[j]);
            if d < min_dist {
                min_dist = d;
                best = j;
            }
        }

        assignments[i] = best;

        let alpha = 1.0 / (counts[best] as f32 + 1.0);
        centroids[best][0] += alpha * (x - centroids[best][0]);
        centroids[best][1] += alpha * (y - centroids[best][1]);
        centroids[best][2] += alpha * (z - centroids[best][2]);
        counts[best] += 1;
    }

    assignments
}

fn kmeans_clustering(points: &[[f32; 3]], num_clusters: usize) -> Vec<usize> {
    let mut centroids = vec![[0.0; 3]; num_clusters.min(points.len())];

    for i in 0..centroids.len() {
        let idx = (i * 7919) % points.len();
        centroids[i] = points[idx];
    }

    for _ in 0..10 {
        let mut new_centroids = vec![[0.0; 3]; centroids.len()];
        let mut counts = vec![0u32; centroids.len()];

        for &point in points {
            let best = find_nearest(&point, &centroids);
            new_centroids[best][0] += point[0];
            new_centroids[best][1] += point[1];
            new_centroids[best][2] += point[2];
            counts[best] += 1;
        }

        for j in 0..centroids.len() {
            if counts[j] > 0 {
                new_centroids[j][0] /= counts[j] as f32;
                new_centroids[j][1] /= counts[j] as f32;
                new_centroids[j][2] /= counts[j] as f32;
            }
        }

        centroids = new_centroids;
    }

    let mut assignments = vec![0usize; points.len()];
    for (i, &point) in points.iter().enumerate() {
        assignments[i] = find_nearest(&point, &centroids);
    }

    assignments
}

fn find_nearest(point: &[f32; 3], centroids: &[[f32; 3]]) -> usize {
    let mut best = 0;
    let mut min_dist = f32::INFINITY;

    for (j, &centroid) in centroids.iter().enumerate() {
        let d = distance(point[0], point[1], point[2], centroid);
        if d < min_dist {
            min_dist = d;
            best = j;
        }
    }

    best
}

fn distance(x: f32, y: f32, z: f32, centroid: [f32; 3]) -> f32 {
    let dx = x - centroid[0];
    let dy = y - centroid[1];
    let dz = z - centroid[2];
    dx * dx + dy * dy + dz * dz
}
