OpCapability Geometry
OpCapability Shader
OpExtension "SPV_KHR_storage_buffer_block"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpExtInstImport %extInst "GLSL.std.450"

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%uint_6 = OpConstant %uint 6

%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_1_3 = OpConstant %float 0.333333333

%v3float_light = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Structures
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Arrays
%arr_6_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_6_v3 = OpTypeArray %v3float %uint_6

; Pointers
%ptr_in_arr_pv = OpTypePointer Input %arr_6_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_6_v3
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Function
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load the three main triangle vertices (indices 0, 2, 4)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p2_v3 = OpVectorShuffle %v3float %p2 %p2 0 1 2

%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr
%p4_v3 = OpVectorShuffle %v3float %p4 %p4 0 1 2

; Compute edge vectors: e1 = p2 - p0, e2 = p4 - p0
%e1 = OpFSub %v3float %p2_v3 %p0_v3
%e2 = OpFSub %v3float %p4_v3 %p0_v3

; Compute cross product: N = cross(e1, e2)
%cross = OpExtInst %v3float %extInst CrossProduct %e1 %e2

; Normalize the normal
%N = OpExtInst %v3float %extInst Normalize %cross

; Normalize light direction
%L = OpExtInst %v3float %extInst Normalize %v3float_light

; Compute diffuse: max(dot(N, L), 0.0)
%dot_result = OpDot %float %N %L
%diffuse = OpExtInst %float %extInst FMax %dot_result %float_0

; Load colors for vertices 0, 2, 4
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr

%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr

%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %c4_ptr

; Compute average color: (c0 + c2 + c4) / 3
%c_sum = OpFAdd %v3float %c0 %c2
%c_sum2 = OpFAdd %v3float %c_sum %c4
%base_color = OpVectorTimesScalar %v3float %c_sum2 %float_1_3

; Compute final color: base * (0.2 + 0.8 * diffuse)
%diffuse_factor = OpFMul %float %float_0_8 %diffuse
%brightness = OpFAdd %float %float_0_2 %diffuse_factor
%final_color = OpVectorTimesScalar %v3float %base_color %brightness

; Emit the three triangle vertices
; Vertex 0
%pos0_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos0_out %p0
OpStore %outColor %final_color
OpEmitVertex

; Vertex 2
%pos2_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos2_out %p2
OpStore %outColor %final_color
OpEmitVertex

; Vertex 4
%pos4_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos4_out %p4
OpStore %outColor %final_color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
