; SPIR-V
; Version: 1.0
; Generator: Custom
; Bound: 200
; Schema: 0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0
OpDecorate %ubo Block
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224
OpDecorate %ubo_var Binding 0
OpDecorate %ubo_var DescriptorSet 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Pointer types
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_ubo = OpTypePointer Uniform %ubo
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

; Function type
%void_fn = OpTypeFunction %void

; Vector type
%vec2 = OpTypeVector %float 2

; UBO struct
%ubo = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Global variables
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output
%ubo_var = OpVariable %ptr_Uniform_ubo Uniform

; Constants
%c0 = OpConstant %int 0
%c3 = OpConstant %int 3
%c4 = OpConstant %int 4

%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%f0_2 = OpConstant %float 0.2
%f0_3 = OpConstant %float 0.3
%f0_7 = OpConstant %float 0.7
%f0_8 = OpConstant %float 0.8
%f1_5 = OpConstant %float 1.5
%f3 = OpConstant %float 3.0

; Main function
%main = OpFunction %void None %void_fn
%entry_block = OpLabel

; Load inputs
%ld_worldPos = OpLoad %vec3 %in_worldPos
%ld_normal = OpLoad %vec3 %in_normal
%ld_color = OpLoad %vec3 %in_color

; Load camera position from UBO (member 4)
%camera_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo_var %c4
%ld_cameraPos = OpLoad %vec4 %camera_ptr

; Extract xyz from camera position
%camera_3d = OpVectorShuffle %vec3 %ld_cameraPos %ld_cameraPos 0 1 2

; Calculate vector from camera to fragment
%to_frag = OpFSub %vec3 %ld_worldPos %camera_3d

; Calculate distance
%distance = OpExtInst %float %1 Length %to_frag

; Calculate fog factor: clamp((distance - 1.5) / 3.0, 0, 1)
%dist_minus_1_5 = OpFSub %float %distance %f1_5
%fog_div = OpFDiv %float %dist_minus_1_5 %f3
%fog_factor = OpExtInst %float %1 FClamp %fog_div %f0 %f1

; Load light position from UBO (member 3)
%light_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo_var %c3
%ld_lightPos = OpLoad %vec4 %light_ptr

; Extract xyz from light position
%light_3d = OpVectorShuffle %vec3 %ld_lightPos %ld_lightPos 0 1 2

; Calculate light direction (from fragment to light)
%light_dir = OpFSub %vec3 %light_3d %ld_worldPos

; Normalize light direction
%light_dir_norm = OpExtInst %vec3 %1 Normalize %light_dir

; Normalize the normal
%normal_norm = OpExtInst %vec3 %1 Normalize %ld_normal

; Calculate diffuse factor: dot(normal, light_direction)
%diffuse = OpDot %float %normal_norm %light_dir_norm

; Clamp diffuse to [0, 1]
%diffuse_clamped = OpExtInst %float %1 FClamp %diffuse %f0 %f1

; Calculate Lambertian shading factor: 0.3 + 0.7 * diffuse
%lamb_mul = OpFMul %float %f0_7 %diffuse_clamped
%lamb_factor = OpFAdd %float %lamb_mul %f0_3

; Base color (0.8, 0.2, 0.2)
%base_color = OpConstantComposite %vec3 %f0_8 %f0_2 %f0_2

; Apply Lambertian shading
%shaded_color = OpVectorTimesScalar %vec3 %base_color %lamb_factor

; Fog color (0.7, 0.7, 0.8)
%fog_color_val = OpConstantComposite %vec3 %f0_7 %f0_7 %f0_8

; Mix: result = shaded_color * (1 - fog_factor) + fog_color * fog_factor
%inv_fog = OpFSub %float %f1 %fog_factor

%shaded_contrib = OpVectorTimesScalar %vec3 %shaded_color %inv_fog
%fog_contrib = OpVectorTimesScalar %vec3 %fog_color_val %fog_factor

%final_rgb = OpFAdd %vec3 %shaded_contrib %fog_contrib

; Construct final color with alpha = 1.0
%final_color = OpCompositeConstruct %vec4 %final_rgb %f1

; Store the result
OpStore %out_color %final_color

OpReturn
OpFunctionEnd
