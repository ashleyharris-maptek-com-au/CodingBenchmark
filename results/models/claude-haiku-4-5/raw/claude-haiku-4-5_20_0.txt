use std::io::{self, BufRead};
use std::collections::HashMap;

#[derive(Clone)]
struct Sample {
    hole_id: usize,
    sample_idx: usize,
    values: Vec<f64>,
    x: f64,
    y: f64,
    z: f64,
}

fn parse_input() -> io::Result<(Vec<Sample>, Vec<String>)> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap()?.trim().to_string();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    let property_line = lines.next().unwrap()?.trim().to_string();
    let property_names: Vec<String> = property_line.split_whitespace().map(|s| s.to_string()).collect();

    let mut all_samples = Vec::new();

    for _ in 0..num_holes {
        let hole_line = lines.next().unwrap()?.trim().to_string();
        let parts: Vec<&str> = hole_line.split_whitespace().collect();

        let hole_id: usize = parts[0].parse().unwrap();
        let start_x: f64 = parts[1].parse().unwrap();
        let start_y: f64 = parts[2].parse().unwrap();
        let start_z: f64 = parts[3].parse().unwrap();
        let dir_x: f64 = parts[4].parse().unwrap();
        let dir_y: f64 = parts[5].parse().unwrap();
        let dir_z: f64 = parts[6].parse().unwrap();
        let length: f64 = parts[7].parse().unwrap();
        let num_samples: usize = parts[8].parse().unwrap();

        for sample_idx in 0..num_samples {
            let sample_line = lines.next().unwrap()?.trim().to_string();
            let parts: Vec<&str> = sample_line.split_whitespace().collect();

            let depth: f64 = parts[0].parse().unwrap();
            let t = if length > 0.0 { (depth / length).min(1.0).max(0.0) } else { 0.0 };

            let x = start_x + t * dir_x;
            let y = start_y + t * dir_y;
            let z = start_z + t * dir_z;

            let mut values = Vec::new();
            for i in 1..=num_properties {
                values.push(parts[i].parse().unwrap());
            }

            all_samples.push(Sample {
                hole_id,
                sample_idx,
                values,
                x,
                y,
                z,
            });
        }
    }

    Ok((all_samples, property_names))
}

fn grid_key(x: f64, y: f64, z: f64, grid_size: f64) -> (i32, i32, i32) {
    (
        (x / grid_size).floor() as i32,
        (y / grid_size).floor() as i32,
        (z / grid_size).floor() as i32,
    )
}

fn find_outliers(
    samples: &[Sample],
    property_names: &[String],
) -> Vec<(usize, usize, String, f64)> {
    let mut results = Vec::new();

    let sample_count = samples.len();

    let (search_radius, z_threshold, min_neighbors) = if sample_count > 50000 {
        (100.0, 2.0, 2)
    } else if sample_count > 10000 {
        (75.0, 2.5, 3)
    } else {
        (50.0, 3.0, 3)
    };

    let search_radius_sq = search_radius * search_radius;
    let grid_size = search_radius * 2.0;

    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
    for (idx, sample) in samples.iter().enumerate() {
        let key = grid_key(sample.x, sample.y, sample.z, grid_size);
        grid.entry(key).or_insert_with(Vec::new).push(idx);
    }

    for (sample_idx, sample) in samples.iter().enumerate() {
        let grid_key_val = grid_key(sample.x, sample.y, sample.z, grid_size);

        let mut nearby_indices = Vec::new();
        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    let key = (grid_key_val.0 + dx, grid_key_val.1 + dy, grid_key_val.2 + dz);
                    if let Some(indices) = grid.get(&key) {
                        nearby_indices.extend(indices);
                    }
                }
            }
        }

        for (prop_idx, prop_name) in property_names.iter().enumerate() {
            let value = sample.values[prop_idx];

            if !value.is_finite() {
                continue;
            }

            let mut neighbors = Vec::new();

            for &other_idx in &nearby_indices {
                if other_idx == sample_idx {
                    continue;
                }

                let other = &samples[other_idx];
                let dx = sample.x - other.x;
                let dy = sample.y - other.y;
                let dz = sample.z - other.z;
                let dist_sq = dx * dx + dy * dy + dz * dz;

                if dist_sq <= search_radius_sq && other.values[prop_idx].is_finite() {
                    neighbors.push(other.values[prop_idx]);
                }
            }

            if neighbors.len() < min_neighbors {
                continue;
            }

            let n = neighbors.len() as f64;
            let mean = neighbors.iter().sum::<f64>() / n;
            let variance = neighbors.iter()
                .map(|v| (v - mean) * (v - mean))
                .sum::<f64>() / n;
            let std_dev = variance.sqrt();

            if std_dev > 1e-6 {
                let z_score = (value - mean).abs() / std_dev;

                if z_score > z_threshold {
                    results.push((
                        sample.hole_id,
                        sample.sample_idx,
                        prop_name.clone(),
                        z_score,
                    ));
                }
            }
        }
    }

    results.sort_by(|a, b| b.3.partial_cmp(&a.3).unwrap_or(std::cmp::Ordering::Equal));
    results
}

fn main() -> io::Result<()> {
    let (samples, property_names) = parse_input()?;
    let outliers = find_outliers(&samples, &property_names);

    println!("{}", outliers.len());
    for (hole_id, sample_idx, property_name, confidence) in outliers {
        println!("{} {} {} {:.6}", hole_id, sample_idx, property_name, confidence);
    }

    Ok(())
}
