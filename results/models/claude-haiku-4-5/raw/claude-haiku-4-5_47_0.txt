@group(0) @binding(0) var<storage, read> baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result : array<u32, 4>;
@group(0) @binding(2) var<uniform> params : vec4<u32>;

fn countTrailingZeros(x : u32) -> u32 {
    if (x == 0u) {
        return 32u;
    }
    var count : u32 = 0u;
    var v : u32 = x;

    if ((v & 0xFFFFu) == 0u) {
        count = count + 16u;
        v = v >> 16u;
    }
    if ((v & 0xFFu) == 0u) {
        count = count + 8u;
        v = v >> 8u;
    }
    if ((v & 0xFu) == 0u) {
        count = count + 4u;
        v = v >> 4u;
    }
    if ((v & 0x3u) == 0u) {
        count = count + 2u;
        v = v >> 2u;
    }
    if ((v & 0x1u) == 0u) {
        count = count + 1u;
    }

    return count;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let nonce = params.y + gid.x;

    var state : array<u32, 16>;
    for (var i : u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    for (var round : u32 = 0u; round < 8u; round = round + 1u) {
        // Group 0: state[0,1,2,3]
        state[0] = state[0] + state[1];
        state[3] = state[3] ^ state[0];
        state[3] = (state[3] << 16u) | (state[3] >> 16u);

        state[2] = state[2] + state[3];
        state[1] = state[1] ^ state[2];
        state[1] = (state[1] << 12u) | (state[1] >> 20u);

        state[0] = state[0] + state[1];
        state[3] = state[3] ^ state[0];
        state[3] = (state[3] << 8u) | (state[3] >> 24u);

        state[2] = state[2] + state[3];
        state[1] = state[1] ^ state[2];
        state[1] = (state[1] << 7u) | (state[1] >> 25u);

        // Group 1: state[4,5,6,7]
        state[4] = state[4] + state[5];
        state[7] = state[7] ^ state[4];
        state[7] = (state[7] << 16u) | (state[7] >> 16u);

        state[6] = state[6] + state[7];
        state[5] = state[5] ^ state[6];
        state[5] = (state[5] << 12u) | (state[5] >> 20u);

        state[4] = state[4] + state[5];
        state[7] = state[7] ^ state[4];
        state[7] = (state[7] << 8u) | (state[7] >> 24u);

        state[6] = state[6] + state[7];
        state[5] = state[5] ^ state[6];
        state[5] = (state[5] << 7u) | (state[5] >> 25u);

        // Group 2: state[8,9,10,11]
        state[8] = state[8] + state[9];
        state[11] = state[11] ^ state[8];
        state[11] = (state[11] << 16u) | (state[11] >> 16u);

        state[10] = state[10] + state[11];
        state[9] = state[9] ^ state[10];
        state[9] = (state[9] << 12u) | (state[9] >> 20u);

        state[8] = state[8] + state[9];
        state[11] = state[11] ^ state[8];
        state[11] = (state[11] << 8u) | (state[11] >> 24u);

        state[10] = state[10] + state[11];
        state[9] = state[9] ^ state[10];
        state[9] = (state[9] << 7u) | (state[9] >> 25u);

        // Group 3: state[12,13,14,15]
        state[12] = state[12] + state[13];
        state[15] = state[15] ^ state[12];
        state[15] = (state[15] << 16u) | (state[15] >> 16u);

        state[14] = state[14] + state[15];
        state[13] = state[13] ^ state[14];
        state[13] = (state[13] << 12u) | (state[13] >> 20u);

        state[12] = state[12] + state[13];
        state[15] = state[15] ^ state[12];
        state[15] = (state[15] << 8u) | (state[15] >> 24u);

        state[14] = state[14] + state[15];
        state[13] = state[13] ^ state[14];
        state[13] = (state[13] << 7u) | (state[13] >> 25u);
    }

    let h0 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    var trailing_zeros : u32;
    if (h0 != 0u) {
        trailing_zeros = countTrailingZeros(h0);
    } else if (h1 != 0u) {
        trailing_zeros = 32u + countTrailingZeros(h1);
    } else {
        trailing_zeros = 64u;
    }

    if (trailing_zeros >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
