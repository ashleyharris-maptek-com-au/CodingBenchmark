using System;
using System.Collections.Generic;
using System.Linq;

class TetrisPlayer
{
    private int width;
    private int height;
    private int[] board;
    private int maxHeight;

    private static readonly Dictionary<char, int[][][]> pieces = new()
    {
        {'I', new[] { new[] {new[]{1,1,1,1}}, new[] {new[]{1}, new[]{1}, new[]{1}, new[]{1}} }},
        {'O', new[] { new[] {new[]{1,1}, new[]{1,1}} }},
        {'T', new[] {
            new[] {new[]{0,1,0}, new[]{1,1,1}},
            new[] {new[]{1,0}, new[]{1,1}, new[]{1,0}},
            new[] {new[]{1,1,1}, new[]{0,1,0}},
            new[] {new[]{0,1}, new[]{1,1}, new[]{0,1}}
        }},
        {'S', new[] {
            new[] {new[]{0,1,1}, new[]{1,1,0}},
            new[] {new[]{1,0}, new[]{1,1}, new[]{0,1}}
        }},
        {'Z', new[] {
            new[] {new[]{1,1,0}, new[]{0,1,1}},
            new[] {new[]{0,1}, new[]{1,1}, new[]{1,0}}
        }},
        {'J', new[] {
            new[] {new[]{1,0,0}, new[]{1,1,1}},
            new[] {new[]{1,1}, new[]{1,0}, new[]{1,0}},
            new[] {new[]{1,1,1}, new[]{0,0,1}},
            new[] {new[]{0,1}, new[]{0,1}, new[]{1,1}}
        }},
        {'L', new[] {
            new[] {new[]{0,0,1}, new[]{1,1,1}},
            new[] {new[]{1,0}, new[]{1,0}, new[]{1,1}},
            new[] {new[]{1,1,1}, new[]{1,0,0}},
            new[] {new[]{1,1}, new[]{0,1}, new[]{0,1}}
        }}
    };

    public TetrisPlayer(int w, int h)
    {
        width = w;
        height = h;
        board = new int[w];
    }

    public (int rotation, int column) FindBestPlacement(char pieceType)
    {
        var shape = pieces[pieceType];
        (int rotation, int column, double score) best = (-1, -1, double.MinValue);

        for (int r = 0; r < shape.Length; r++)
        {
            int[] rotShape = shape[r];
            int pieceWidth = rotShape[0].Length;
            int pieceHeight = rotShape.Length;

            for (int col = 0; col <= width - pieceWidth; col++)
            {
                if (CanPlace(rotShape, col))
                {
                    var (newBoard, clearedLines) = SimulatePlacement(rotShape, col);
                    double score = EvaluatePlacement(newBoard, clearedLines);

                    if (score > best.score)
                    {
                        best = (r, col, score);
                    }
                }
            }
        }

        return best.rotation >= 0 ? (best.rotation, best.column) : (0, 0);
    }

    private bool CanPlace(int[][] shape, int col)
    {
        for (int row = 0; row < shape.Length; row++)
        {
            for (int c = 0; c < shape[row].Length; c++)
            {
                if (shape[row][c] == 1)
                {
                    int boardCol = col + c;
                    if (boardCol < 0 || boardCol >= width)
                        return false;

                    int landRow = board[boardCol];
                    if (landRow + shape.Length - row > height)
                        return false;
                }
            }
        }
        return true;
    }

    private (int[], int) SimulatePlacement(int[][] shape, int col)
    {
        int[] newBoard = (int[])board.Clone();

        for (int row = 0; row < shape.Length; row++)
        {
            for (int c = 0; c < shape[row].Length; c++)
            {
                if (shape[row][c] == 1)
                {
                    int boardCol = col + c;
                    newBoard[boardCol] = Math.Max(newBoard[boardCol], board[boardCol] + shape.Length - row);
                }
            }
        }

        int clearedLines = 0;
        maxHeight = newBoard.Max();

        return (newBoard, clearedLines);
    }

    private double EvaluatePlacement(int[] newBoard, int clearedLines)
    {
        int newMaxHeight = newBoard.Max();
        double heightPenalty = newMaxHeight * 10;
        double clearReward = clearedLines * 100;

        int holes = 0;
        for (int col = 0; col < width; col++)
        {
            int colHeight = newBoard[col];
            for (int row = 0; row < colHeight; row++)
            {
                if (row < newBoard.Min() && newBoard[col] > newBoard.Min())
                    holes++;
            }
        }
        double holePenalty = holes * 20;

        int bumpiness = 0;
        for (int col = 0; col < width - 1; col++)
        {
            bumpiness += Math.Abs(newBoard[col] - newBoard[col + 1]);
        }
        double bumpPenalty = bumpiness * 5;

        return clearReward - heightPenalty - holePenalty - bumpPenalty;
    }

    public void PlacePiece(char pieceType, int rotation, int column, string result)
    {
        var shape = pieces[pieceType];
        int[] rotShape = shape[rotation];

        for (int row = 0; row < rotShape.Length; row++)
        {
            for (int c = 0; c < rotShape[row].Length; c++)
            {
                if (rotShape[row][c] == 1)
                {
                    int boardCol = column + c;
                    board[boardCol] = Math.Max(board[boardCol], board[boardCol] + rotShape.Length - row);
                }
            }
        }

        if (result.StartsWith("cleared_"))
        {
            int clearedCount = int.Parse(result.Split('_')[1]);
            for (int col = 0; col < width; col++)
            {
                board[col] = Math.Max(0, board[col] - clearedCount);
            }
        }
    }

    static void Main()
    {
        var input = Console.ReadLine().Split();
        int width = int.Parse(input[0]);
        int height = int.Parse(input[1]);
        int numPieces = int.Parse(input[2]);

        var player = new TetrisPlayer(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            char pieceType = Console.ReadLine()[0];
            var (rotation, column) = player.FindBestPlacement(pieceType);

            Console.WriteLine($"{rotation} {column}");
            Console.Out.Flush();

            string result = Console.ReadLine();
            if (result == "gameover")
                break;

            player.PlacePiece(pieceType, rotation, column, result);
        }
    }
}
