use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let _num_vertices: usize = parts[0].parse().unwrap();
    let num_edges: usize = parts[1].parse().unwrap();

    let mut edges = Vec::new();
    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u: usize = parts[0].parse().unwrap();
        let v: usize = parts[1].parse().unwrap();
        edges.push((u, v));
    }

    let mut cover = HashSet::new();
    let mut uncovered = (0..edges.len()).collect::<HashSet<_>>();

    while !uncovered.is_empty() {
        let edge_idx = *uncovered.iter().next().unwrap();
        let (u, v) = edges[edge_idx];
        cover.insert(u);
        cover.insert(v);
        uncovered.retain(|&i| {
            let (a, b) = edges[i];
            a != u && a != v && b != u && b != v
        });
    }

    println!("{}", cover.len());
    let mut vertices: Vec<usize> = cover.into_iter().collect();
    vertices.sort();
    println!("{}", vertices.iter().map(|v| v.to_string()).collect::<Vec<_>>().join(" "));
}
