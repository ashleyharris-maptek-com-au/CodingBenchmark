OpMemberDecorate %uboType 5 Offset 224

; Type definitions
%void = OpTypeVoid

%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%uboType = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptrUniformType = OpTypePointer Uniform %uboType
%ptrInputV3 = OpTypePointer Input %v3float
%ptrInputV2 = OpTypePointer Input %v2float
%ptrOutputV4 = OpTypePointer Output %v4float
%ptrUniformV4 = OpTypePointer Uniform %v4float

%mainType = OpTypeFunction %void

; Constants
%uint_0 = OpConstant %uint 0
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0

; Global variables
%ubo = OpVariable %ptrUniformType Uniform
%inWorldPos = OpVariable %ptrInputV3 Input
%inNormal = OpVariable %ptrInputV3 Input
%inUV = OpVariable %ptrInputV2 Input
%inTangent = OpVariable %ptrInputV3 Input
%inColor = OpVariable %ptrInputV3 Input
%outColor = OpVariable %ptrOutputV4 Output

%main = OpFunction %void None %mainType
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %inWorldPos
%normal = OpLoad %v3float %inNormal
%uv = OpLoad %v2float %inUV
%tangent = OpLoad %v3float %inTangent
%color = OpLoad %v3float %inColor

; Load camera position from UBO[4]
%cameraPos_ptr = OpAccessChain %ptrUniformV4 %ubo %uint_4
%cameraPos_v4 = OpLoad %v4float %cameraPos_ptr
%cameraPos = OpVectorShuffle %v3float %cameraPos_v4 %cameraPos_v4 0 1 2

; Normalize normal vector
%normal_dot = OpDot %float %normal %normal
%normal_sqrt = OpExtInst %float %glsl Sqrt %normal_dot
%inv_normal_length = OpFDiv %float %float_1 %normal_sqrt
%normal_normalized = OpVectorTimesScalar %v3float %normal %inv_normal_length

; Calculate view direction: V = normalize(cameraPos - worldPos)
%view_unnormalized = OpFSub %v3float %cameraPos %worldPos
%view_dot = OpDot %float %view_unnormalized %view_unnormalized
%view_sqrt = OpExtInst %float %glsl Sqrt %view_dot
%inv_view_length = OpFDiv %float %float_1 %view_sqrt
%view_normalized = OpVectorTimesScalar %v3float %view_unnormalized %inv_view_length

; Calculate NdotV = max(dot(N, V), 0)
%ndotv = OpDot %float %normal_normalized %view_normalized
%ndotv_clamped = OpExtInst %float %glsl FMax %ndotv %float_0

; Check if NdotV < 0.5
%is_less = OpFOrdLessThan %bool %ndotv_clamped %float_0_5

; Branch for the two cases
OpSelectionMerge %merge None
OpBranchConditional %is_less %less_block %greater_block

; NdotV < 0.5: r=1, g=2*NdotV, b=0
%less_block = OpLabel
%color_less_g = OpFMul %float %ndotv_clamped %float_2
%color_less_v = OpCompositeConstruct %v3float %float_1 %color_less_g %float_0
OpBranch %merge

; NdotV >= 0.5: r=1-(NdotV-0.5)*2, g=1-(NdotV-0.5)*2, b=(NdotV-0.5)*2
%greater_block = OpLabel
%ndotv_offset = OpFSub %float %ndotv_clamped %float_0_5
%offset_times_2 = OpFMul %float %ndotv_offset %float_2
%color_greater_rg = OpFSub %float %float_1 %offset_times_2
%color_greater_b = OpFMul %float %ndotv_offset %float_2
%color_greater_v = OpCompositeConstruct %v3float %color_greater_rg %color_greater_rg %color_greater_b
OpBranch %merge

; Merge point
%merge = OpLabel
%final_color_v = OpPhi %v3float %color_less_v %less_block %color_greater_v %greater_block

; Extract components and construct final RGBA
%final_r = OpCompositeExtract %float %final_color_v 0
%final_g = OpCompositeExtract %float %final_color_v 1
%final_b = OpCompositeExtract %float %final_color_v 2
%result = OpCompositeConstruct %v4float %final_r %final_g %final_b %float_1

; Store result
OpStore %outColor %result

OpReturn
OpFunctionEnd
