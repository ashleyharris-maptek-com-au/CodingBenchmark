OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %Matrices BufferBlock

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4x4 = OpTypeMatrix %v4float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4

%float_0 = OpConstant %float 0.0
%float_0p02 = OpConstant %float 0.02
%float_0p333333 = OpConstant %float 0.333333
%float_0p666667 = OpConstant %float 0.666667
%float_1 = OpConstant %float 1.0

%Matrices = OpTypeStruct %mat4x4 %mat4x4 %mat4x4 %v3float %v3float %v4float
OpMemberDecorate %Matrices 0 Offset 0
OpMemberDecorate %Matrices 1 Offset 64
OpMemberDecorate %Matrices 2 Offset 128
OpMemberDecorate %Matrices 3 Offset 192
OpMemberDecorate %Matrices 4 Offset 208
OpMemberDecorate %Matrices 5 Offset 224

%ptr_Uniform_Matrices = OpTypePointer Uniform %Matrices
%ubo = OpVariable %ptr_Uniform_Matrices Uniform

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%glsl_ext = OpExtInstImport "GLSL.std.450"

%fn_void = OpTypeFunction %void

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load input positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

; Load colors
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr

; Extract xyz components
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p3_xyz = OpVectorShuffle %v3float %p3 %p3 0 1 2

; Main segment direction
%seg_dir = OpFSub %v3float %p2_xyz %p1_xyz

; Compute smooth tangent at p1 using adjacency
%t1 = OpFSub %v3float %p2_xyz %p0_xyz
%t1_len = OpExtInst %float %glsl_ext Length %t1
%t1_inv = OpFDiv %float %float_1 %t1_len
%t1_norm = OpVectorTimesScalar %v3float %t1 %t1_inv

; Perpendicular direction: cross segment with up vector (0, 0, 1)
%up = OpCompositeConstruct %v3float %float_0 %float_0 %float_1
%cross = OpExtInst %v3float %glsl_ext Cross %seg_dir %up
%cross_len = OpExtInst %float %glsl_ext Length %cross
%cross_inv = OpFDiv %float %float_1 %cross_len
%cross_norm = OpVectorTimesScalar %v3float %cross %cross_inv

; Scale perpendicular by thickness
%perp = OpVectorTimesScalar %v3float %cross_norm %float_0p02

; Compute segment thirds
%third = OpVectorTimesScalar %v3float %seg_dir %float_0p333333
%two_third = OpVectorTimesScalar %v3float %seg_dir %float_0p666667

; Part 1 segment: p1 to p1 + third
%part1_start = %p1_xyz
%part1_end = OpFAdd %v3float %p1_xyz %third

; Part 3 segment: p1 + two_third to p2
%part3_start = OpFAdd %v3float %p1_xyz %two_third
%part3_end = %p2_xyz

; ===== EMIT QUAD 1 (Part 1) =====

; Vertex 0: start - perp
%v0p = OpFSub %v3float %part1_start %perp
%v0_homo = OpCompositeConstruct %v4float %v0p.x %v0p.y %v0p.z %float_1
%v0_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v0_pos_ptr %v0_homo
%col_ptr = OpAccessChain %ptr_out_v3 %outColor
OpStore %col_ptr %c1
OpEmitVertex

; Vertex 1: start + perp
%v1p = OpFAdd %v3float %part1_start %perp
%v1_homo = OpCompositeConstruct %v4float %v1p.x %v1p.y %v1p.z %float_1
OpStore %v0_pos_ptr %v1_homo
OpStore %col_ptr %c1
OpEmitVertex

; Vertex 2: end - perp
%v2p = OpFSub %v3float %part1_end %perp
%v2_homo = OpCompositeConstruct %v4float %v2p.x %v2p.y %v2p.z %float_1
OpStore %v0_pos_ptr %v2_homo
OpStore %col_ptr %c1
OpEmitVertex

; Vertex 3: end + perp
%v3p = OpFAdd %v3float %part1_end %perp
%v3_homo = OpCompositeConstruct %v4float %v3p.x %v3p.y %v3p.z %float_1
OpStore %v0_pos_ptr %v3_homo
OpStore %col_ptr %c1
OpEmitVertex

OpEndPrimitive

; ===== EMIT QUAD 2 (Part 3) =====

; Vertex 4: start - perp
%v4p = OpFSub %v3float %part3_start %perp
%v4_homo = OpCompositeConstruct %v4float %v4p.x %v4p.y %v4p.z %float_1
OpStore %v0_pos_ptr %v4_homo
OpStore %col_ptr %c2
OpEmitVertex

; Vertex 5: start + perp
%v5p = OpFAdd %v3float %part3_start %perp
%v5_homo = OpCompositeConstruct %v4float %v5p.x %v5p.y %v5p.z %float_1
OpStore %v0_pos_ptr %v5_homo
OpStore %col_ptr %c2
OpEmitVertex

; Vertex 6: end - perp
%v6p = OpFSub %v3float %part3_end %perp
%v6_homo = OpCompositeConstruct %v4float %v6p.x %v6p.y %v6p.z %float_1
OpStore %v0_pos_ptr %v6_homo
OpStore %col_ptr %c2
OpEmitVertex

; Vertex 7: end + perp
%v7p = OpFAdd %v3float %part3_end %perp
%v7_homo = OpCompositeConstruct %v4float %v7p.x %v7p.y %v7p.z %float_1
OpStore %v0_pos_ptr %v7_homo
OpStore %col_ptr %c2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
