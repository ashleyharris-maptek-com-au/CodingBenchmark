; SPIR-V

; Generator: Khronos; 11
; Bound: 60
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %in_worldPos %out_color
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %out_color Location 0

OpDecorate %ubo_block Block
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

OpMemberDecorate %ubo_block 0 Offset 0
OpMemberDecorate %ubo_block 1 Offset 64
OpMemberDecorate %ubo_block 2 Offset 128
OpMemberDecorate %ubo_block 3 Offset 192
OpMemberDecorate %ubo_block 4 Offset 208
OpMemberDecorate %ubo_block 5 Offset 224
OpMemberDecorate %ubo_block 0 ColMajor
OpMemberDecorate %ubo_block 0 MatrixStride 16
OpMemberDecorate %ubo_block 1 ColMajor
OpMemberDecorate %ubo_block 1 MatrixStride 16
OpMemberDecorate %ubo_block 2 ColMajor
OpMemberDecorate %ubo_block 2 MatrixStride 16

; Type definitions
%void = OpTypeVoid
%func_void = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Uniform buffer struct
%ubo_block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ubo_ptr_type = OpTypePointer Uniform %ubo_block
%ubo_var = OpVariable %ubo_ptr_type Uniform

; Input/Output pointers
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

; Variables
%in_worldPos = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output

; Constants
%const_0_5 = OpConstant %float 0.5
%const_0_1 = OpConstant %float 0.1
%const_0_4 = OpConstant %float 0.4
%const_1_0 = OpConstant %float 1.0
%const_0_0 = OpConstant %float 0.0

; Main function
%main = OpFunction %void None %func_void
%label_entry = OpLabel

; Load world position
%worldPos = OpLoad %vec3 %in_worldPos

; Extract Y component
%y_value = OpCompositeExtract %float %worldPos 1

; Calculate t = clamp(y * 0.5 + 0.5, 0.0, 1.0)
%y_scaled = OpFMul %float %y_value %const_0_5
%y_offset = OpFAdd %float %y_scaled %const_0_5

; Clamp using min/max
%y_clamped_low = OpFMax %float %y_offset %const_0_0
%t = OpFMin %float %y_clamped_low %const_1_0

; Create bottom color (0.1, 0.4, 0.1)
%bottom = OpCompositeConstruct %vec3 %const_0_1 %const_0_4 %const_0_1

; Create top color (1.0, 1.0, 1.0)
%top = OpCompositeConstruct %vec3 %const_1_0 %const_1_0 %const_1_0

; Calculate 1.0 - t for interpolation
%one_minus_t = OpFSub %float %const_1_0 %t

; Mix: bottom * (1 - t) + top * t
%bottom_scaled = OpVectorTimesScalar %vec3 %bottom %one_minus_t
%top_scaled = OpVectorTimesScalar %vec3 %top %t
%color_rgb = OpFAdd %vec3 %bottom_scaled %top_scaled

; Extract components to create vec4
%r = OpCompositeExtract %float %color_rgb 0
%g = OpCompositeExtract %float %color_rgb 1
%b = OpCompositeExtract %float %color_rgb 2

; Combine into vec4 with alpha = 1.0
%final_color = OpCompositeConstruct %vec4 %r %g %b %const_1_0

; Store result
OpStore %out_color %final_color

OpReturn
OpFunctionEnd
