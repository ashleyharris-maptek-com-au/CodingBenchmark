OpCapability Geometry
OpExtInstImport %glsl "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %inColor "inColor"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4

%int_0 = OpConstant %int 0

%float_0_025 = OpConstant %float 0.025
%float_1_0 = OpConstant %float 1.0

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_4
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%inColor = OpVariable %ptr_in_arr_v3 Input
OpDecorate %inColor Location 2

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

%func_void = OpTypeFunction %void

%main = OpFunction %void None %func_void
%entry = OpLabel

; Load positions from input array
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %int_0
%p0 = OpLoad %v4float %pos0_ptr

%pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_1 %int_0
%p1 = OpLoad %v4float %pos1_ptr

%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %int_0
%p2 = OpLoad %v4float %pos2_ptr

%pos3_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_3 %int_0
%p3 = OpLoad %v4float %pos3_ptr

; Load colors
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %uint_1
%c1 = OpLoad %v3float %c1_ptr

%c2_ptr = OpAccessChain %ptr_in_v3 %inColor %uint_2
%c2 = OpLoad %v3float %c2_ptr

; Compute tangent t0 = normalize(p2 - p0)
%diff0 = OpFSub %v4float %p2 %p0
%diff0_v3 = OpVectorShuffle %v3float %diff0 %diff0 0 1 2
%t0_raw = OpExtInst %v3float %glsl 69 %diff0_v3
%t0 = OpVectorShuffle %v2float %t0_raw %t0_raw 0 1

; Compute tangent t1 = normalize(p3 - p1)
%diff1 = OpFSub %v4float %p3 %p1
%diff1_v3 = OpVectorShuffle %v3float %diff1 %diff1 0 1 2
%t1_raw = OpExtInst %v3float %glsl 69 %diff1_v3
%t1 = OpVectorShuffle %v2float %t1_raw %t1_raw 0 1

; Compute perpendicular: perp0 = (-t0.y, t0.x)
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%t0_y_neg = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %t0_y_neg %t0_x

; Compute perpendicular: perp1 = (-t1.y, t1.x)
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%t1_y_neg = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %t1_y_neg %t1_x

; Scale perpendiculars by thickness (0.025)
%perp0_thick = OpVectorTimesScalar %v2float %perp0 %float_0_025
%perp1_thick = OpVectorTimesScalar %v2float %perp1 %float_0_025

; Extract XY components from positions
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1

; Compute output positions for the triangle strip
%out0_xy = OpFAdd %v2float %p1_xy %perp0_thick
%out1_xy = OpFSub %v2float %p1_xy %perp0_thick
%out2_xy = OpFAdd %v2float %p2_xy %perp1_thick
%out3_xy = OpFSub %v2float %p2_xy %perp1_thick

; Get Z components
%p1_z = OpCompositeExtract %float %p1 2
%p2_z = OpCompositeExtract %float %p2 2

; Build output position 0
%out0_x = OpCompositeExtract %float %out0_xy 0
%out0_y = OpCompositeExtract %float %out0_xy 1
%out0_v4 = OpCompositeConstruct %v4float %out0_x %out0_y %p1_z %float_1_0

; Build output position 1
%out1_x = OpCompositeExtract %float %out1_xy 0
%out1_y = OpCompositeExtract %float %out1_xy 1
%out1_v4 = OpCompositeConstruct %v4float %out1_x %out1_y %p1_z %float_1_0

; Build output position 2
%out2_x = OpCompositeExtract %float %out2_xy 0
%out2_y = OpCompositeExtract %float %out2_xy 1
%out2_v4 = OpCompositeConstruct %v4float %out2_x %out2_y %p2_z %float_1_0

; Build output position 3
%out3_x = OpCompositeExtract %float %out3_xy 0
%out3_y = OpCompositeExtract %float %out3_xy 1
%out3_v4 = OpCompositeConstruct %v4float %out3_x %out3_y %p2_z %float_1_0

; Emit vertex 0
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %out0_v4
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 1
OpStore %out_pos_ptr %out1_v4
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 2
OpStore %out_pos_ptr %out2_v4
OpStore %outColor %c2
OpEmitVertex

; Emit vertex 3
OpStore %out_pos_ptr %out3_v4
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
