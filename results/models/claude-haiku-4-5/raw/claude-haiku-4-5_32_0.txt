#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <set>
#include <map>

using namespace std;

typedef long long ll;
typedef pair<ll, int> pli;
typedef pair<int, int> pii;

const ll INF = 1e18;

vector<ll> dijkstra(int src, const vector<vector<pair<int, ll>>>& adj) {
    int n = adj.size();
    vector<ll> dist(n, INF);
    priority_queue<pli, vector<pli>, greater<pli>> pq;

    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dist[u]) continue;

        for (auto [v, w] : adj[u]) {
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }

    return dist;
}

vector<int> getPath(int src, int dst, const vector<vector<pair<int, ll>>>& adj) {
    int n = adj.size();
    vector<ll> dist(n, INF);
    vector<int> parent(n, -1);
    priority_queue<pli, vector<pli>, greater<pli>> pq;

    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dist[u]) continue;

        for (auto [v, w] : adj[u]) {
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                parent[v] = u;
                pq.push({dist[v], v});
            }
        }
    }

    vector<int> path;
    int cur = dst;
    while (cur != -1) {
        path.push_back(cur);
        cur = parent[cur];
    }
    reverse(path.begin(), path.end());
    return path;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<pair<int, ll>>> adj(n);
    map<pii, ll> edgeWeight;

    for (int i = 0; i < m; i++) {
        int u, v;
        ll w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
        int a = min(u, v), b = max(u, v);
        edgeWeight[{a, b}] = w;
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; i++) {
        cin >> terminals[i];
    }

    if (k == 1) {
        cout << "0 0\n";
        return 0;
    }

    // Compute shortest distances from each terminal
    vector<vector<ll>> dist(k);
    for (int i = 0; i < k; i++) {
        dist[i] = dijkstra(terminals[i], adj);
    }

    // Prim's algorithm on terminal graph
    vector<bool> inMST(k, false);
    set<pii> resultEdges;
    ll totalWeight = 0;

    inMST[0] = true;

    for (int iter = 1; iter < k; iter++) {
        ll bestDist = INF;
        int bestFrom = -1, bestTo = -1;

        for (int i = 0; i < k; i++) {
            if (inMST[i]) {
                for (int j = 0; j < k; j++) {
                    if (!inMST[j] && dist[i][terminals[j]] < bestDist) {
                        bestDist = dist[i][terminals[j]];
                        bestFrom = i;
                        bestTo = j;
                    }
                }
            }
        }

        inMST[bestTo] = true;

        // Get path and add edges
        vector<int> path = getPath(terminals[bestFrom], terminals[bestTo], adj);
        for (size_t i = 0; i + 1 < path.size(); i++) {
            int u = min(path[i], path[i+1]);
            int v = max(path[i], path[i+1]);
            if (resultEdges.find({u, v}) == resultEdges.end()) {
                resultEdges.insert({u, v});
                totalWeight += edgeWeight[{u, v}];
            }
        }
    }

    cout << totalWeight << " " << resultEdges.size() << "\n";
    for (auto [u, v] : resultEdges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
