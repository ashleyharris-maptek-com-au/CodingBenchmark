use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

#[derive(Copy, Clone)]
struct Map {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read map parameters (width height gravity max_thrust fuel)
    let line1 = lines.next().unwrap().unwrap();
    let p1: Vec<f64> = line1.split_whitespace()
        .map(|s| s.parse().unwrap()).collect();

    let mut map = Map {
        width: p1[0] as usize,
        height: p1[1] as usize,
        gravity: p1[2],
        max_thrust: p1[3],
        target_x: 0.0,
        target_y: 0.0,
    };

    // Read start and target coordinates
    let line2 = lines.next().unwrap().unwrap();
    let p2: Vec<f64> = line2.split_whitespace()
        .map(|s| s.parse().unwrap()).collect();

    let start_x = p2[0];
    let start_y = p2[1];
    map.target_x = p2[2];
    map.target_y = p2[3];

    // Skip map grid data
    for _ in 0..map.height {
        let _ = lines.next();
    }

    // Shared state protected by mutex for thread-safe updates
    let state = Arc::new(Mutex::new(State {
        x: start_x,
        y: start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel_remaining: f64::INFINITY,
    }));

    // Spawn reader thread to continuously read state updates from stdin
    // This prevents blocking the control loop
    let state_copy = state.clone();
    thread::spawn(move || {
        for line in lines {
            if let Ok(line) = line {
                let nums: Vec<f64> = line.split_whitespace()
                    .map(|s| s.parse().unwrap_or(0.0))
                    .collect();
                if nums.len() >= 7 {
                    if let Ok(mut s) = state_copy.lock() {
                        s.x = nums[0];
                        s.y = nums[1];
                        s.vx = nums[2];
                        s.vy = nums[3];
                        s.angle = nums[4];
                        s.angular_vel = nums[5];
                        s.fuel_remaining = nums[6];
                    }
                }
            }
        }
    });

    // Control loop running at 20Hz (50ms per iteration)
    loop {
        let loop_start = Instant::now();

        let s = state.lock().unwrap().clone();
        let (thrust, turn) = compute_control(&s, &map);

        // Output control command
        println!("{} {}", thrust, turn);
        io::stdout().flush().unwrap();

        // Check landing condition: velocity < 2 m/s and within 15m of target
        let speed = (s.vx * s.vx + s.vy * s.vy).sqrt();
        let dx = map.target_x - s.x;
        let dy = map.target_y - s.y;
        let dist = (dx * dx + dy * dy).sqrt();
        if speed < 2.0 && dist < 15.0 {
            break;
        }

        // Check failure conditions
        if s.fuel_remaining <= 0.0 || s.y >= map.height as f64 {
            break;
        }

        // Maintain 20Hz output rate (50ms per tick)
        let elapsed = loop_start.elapsed();
        if elapsed < Duration::from_millis(50) {
            thread::sleep(Duration::from_millis(50) - elapsed);
        }
    }
}

fn compute_control(s: &State, m: &Map) -> (f64, f64) {
    let dx = m.target_x - s.x;
    let dy = m.target_y - s.y;
    let dist = (dx * dx + dy * dy).sqrt();

    // Calculate desired angle to point toward target
    // Angle 0 = up (-y direction), π = down (+y direction)
    // atan2(y, x) but adapted for our coordinate system where y increases downward
    let desired_angle = if dist > 1.0 {
        dx.atan2(-dy)
    } else {
        // At target, point down for safe landing
        std::f64::consts::PI
    };

    // Calculate angle error with wraparound handling
    let mut angle_error = desired_angle - s.angle;
    if angle_error > std::f64::consts::PI {
        angle_error -= 2.0 * std::f64::consts::PI;
    }
    if angle_error < -std::f64::consts::PI {
        angle_error += 2.0 * std::f64::consts::PI;
    }

    // Proportional turn control to reduce angle error
    let turn = (angle_error * 1.2).max(-1.0).min(1.0);

    // Calculate current speed
    let speed = (s.vx * s.vx + s.vy * s.vy).sqrt();

    // Adaptive thrust control based on distance and speed
    let thrust = if s.fuel_remaining <= 0.0 {
        0.0
    } else if dist > 100.0 {
        // Far from target, controlled descent to maintain altitude
        (m.gravity / m.max_thrust * 1.1).min(0.85)
    } else if dist > 50.0 {
        // Approaching target, moderate descent rate
        (m.gravity / m.max_thrust * 1.0).min(0.75)
    } else if dist > 25.0 {
        // Close to target, slow approach
        (m.gravity / m.max_thrust * 0.95).min(0.7)
    } else if speed > 8.0 {
        // Close and moving fast, brake hard
        1.0
    } else if speed > 4.0 {
        // Still moving too fast, moderate braking
        0.8
    } else if speed > 2.0 {
        // Getting close to target velocity, light thrust
        0.5
    } else if s.vy > 0.3 {
        // Nearly stopped but still descending, very light thrust
        0.3
    } else {
        // Hovering near target, minimal counteract gravity
        (m.gravity / m.max_thrust * 0.6).min(0.15)
    };

    (thrust, turn)
}
