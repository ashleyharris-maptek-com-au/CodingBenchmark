import sys
from collections import defaultdict

def read_input():
    line = sys.stdin.readline().strip()
    num_vars, num_clauses = map(int, line.split())
    clauses = []
    for _ in range(num_clauses):
        clause = list(map(int, sys.stdin.readline().split()))
        clauses.append(clause)
    return num_vars, clauses

def solve_sat(num_vars, clauses):
    assignment = [None] * (num_vars + 1)  # 1-indexed

    def unit_propagate(assignment):
        changed = True
        while changed:
            changed = False
            for clause in clauses:
                unassigned = []
                satisfied = False
                for lit in clause:
                    var = abs(lit)
                    if assignment[var] is None:
                        unassigned.append(lit)
                    else:
                        val = assignment[var]
                        if (lit > 0 and val) or (lit < 0 and not val):
                            satisfied = True
                            break

                if not satisfied:
                    if len(unassigned) == 0:
                        return False
                    if len(unassigned) == 1:
                        lit = unassigned[0]
                        var = abs(lit)
                        assignment[var] = (lit > 0)
                        changed = True
        return True

    def get_unassigned_var(assignment):
        # Get frequency of variables in unsatisfied clauses
        var_score = defaultdict(int)
        for clause in clauses:
            satisfied = False
            for lit in clause:
                var = abs(lit)
                val = assignment[var]
                if val is not None and ((lit > 0 and val) or (lit < 0 and not val)):
                    satisfied = True
                    break
            if not satisfied:
                for lit in clause:
                    var = abs(lit)
                    if assignment[var] is None:
                        var_score[var] += 1

        if var_score:
            return max(var_score, key=var_score.get)
        return next((v for v in range(1, len(assignment)) if assignment[v] is None), None)

    def dpll(assignment):
        if not unit_propagate(assignment):
            return False

        var = get_unassigned_var(assignment)
        if var is None:
            return True

        # Try True first
        assignment[var] = True
        if dpll(assignment):
            return True

        # Backtrack
        assignment[var] = None

        # Try False
        assignment[var] = False
        if dpll(assignment):
            return True

        assignment[var] = None
        return False

    if dpll(assignment):
        return assignment[1:]

    return None

num_vars, clauses = read_input()
solution = solve_sat(num_vars, clauses)

print("SAT")
print(" ".join(str(1 if x else 0) for x in solution))
