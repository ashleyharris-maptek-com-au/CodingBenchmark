using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostman
{
    struct Edge
    {
        public int to;
        public int weight;
        public int id;
    }

    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        int m = int.Parse(firstLine[1]);

        List<Edge>[] graph = new List<Edge>[n];
        for (int i = 0; i < n; i++)
            graph[i] = new List<Edge>();

        int edgeId = 0;
        for (int i = 0; i < m; i++)
        {
            string[] parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);

            if (u == v)
            {
                graph[u].Add(new Edge { to = v, weight = w, id = edgeId });
                graph[u].Add(new Edge { to = v, weight = w, id = edgeId });
            }
            else
            {
                graph[u].Add(new Edge { to = v, weight = w, id = edgeId });
                graph[v].Add(new Edge { to = u, weight = w, id = edgeId });
            }
            edgeId++;
        }

        List<int> oddVertices = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (graph[i].Count % 2 == 1)
                oddVertices.Add(i);
        }

        if (oddVertices.Count == 0)
        {
            var route = EulerCircuit(n, graph);
            Console.WriteLine(string.Join(" ", route));
            return;
        }

        var distances = new Dictionary<int, Dictionary<int, int>>();
        foreach (int u in oddVertices)
        {
            distances[u] = Dijkstra(u, graph, n);
        }

        var additionalEdges = MinimumWeightMatching(oddVertices, distances);

        var newGraph = new List<Edge>[n];
        for (int i = 0; i < n; i++)
        {
            newGraph[i] = new List<Edge>(graph[i]);
        }

        int nextId = edgeId;
        foreach (var (u, v) in additionalEdges)
        {
            int w = distances[u][v];
            if (u == v)
            {
                newGraph[u].Add(new Edge { to = v, weight = w, id = nextId });
                newGraph[u].Add(new Edge { to = v, weight = w, id = nextId });
            }
            else
            {
                newGraph[u].Add(new Edge { to = v, weight = w, id = nextId });
                newGraph[v].Add(new Edge { to = u, weight = w, id = nextId });
            }
            nextId++;
        }

        var route2 = EulerCircuit(n, newGraph);
        Console.WriteLine(string.Join(" ", route2));
    }

    static Dictionary<int, int> Dijkstra(int start, List<Edge>[] graph, int n)
    {
        var dist = new int[n];
        for (int i = 0; i < n; i++)
            dist[i] = int.MaxValue;
        dist[start] = 0;

        var visited = new bool[n];
        var pq = new PriorityQueue<(int node, int dist), int>();
        pq.Enqueue((start, 0), 0);

        while (pq.Count > 0)
        {
            var (u, d) = pq.Dequeue();

            if (visited[u])
                continue;
            visited[u] = true;

            foreach (var edge in graph[u])
            {
                int v = edge.to;
                int w = edge.weight;
                if (dist[u] != int.MaxValue && dist[u] + w < dist[v])
                {
                    dist[v] = dist[u] + w;
                    pq.Enqueue((v, dist[v]), dist[v]);
                }
            }
        }

        var result = new Dictionary<int, int>();
        for (int i = 0; i < n; i++)
            result[i] = dist[i];
        return result;
    }

    static List<(int, int)> MinimumWeightMatching(List<int> oddVertices, Dictionary<int, Dictionary<int, int>> distances)
    {
        var result = new List<(int, int)>();
        var used = new bool[oddVertices.Count];

        for (int i = 0; i < oddVertices.Count; i++)
        {
            if (used[i])
                continue;

            int bestJ = -1;
            int bestDist = int.MaxValue;

            for (int j = i + 1; j < oddVertices.Count; j++)
            {
                if (used[j])
                    continue;

                int d = distances[oddVertices[i]][oddVertices[j]];
                if (d < bestDist)
                {
                    bestDist = d;
                    bestJ = j;
                }
            }

            if (bestJ != -1)
            {
                used[i] = true;
                used[bestJ] = true;
                result.Add((oddVertices[i], oddVertices[bestJ]));
            }
        }

        return result;
    }

    static List<int> EulerCircuit(int n, List<Edge>[] graph)
    {
        var adj = new List<Edge>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new List<Edge>(graph[i]);

        var circuit = new List<int>();
        var stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();

            if (adj[u].Count > 0)
            {
                var edge = adj[u][0];
                int v = edge.to;
                int edgeId = edge.id;

                adj[u].RemoveAt(0);

                for (int i = 0; i < adj[v].Count; i++)
                {
                    if (adj[v][i].id == edgeId)
                    {
                        adj[v].RemoveAt(i);
                        break;
                    }
                }

                stack.Push(v);
            }
            else
            {
                circuit.Add(stack.Pop());
            }
        }

        circuit.Reverse();
        return circuit;
    }
}
