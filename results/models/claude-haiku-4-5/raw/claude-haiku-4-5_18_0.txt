#include <iostream>
#include <vector>

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m;
    std::cin >> n >> m;

    std::vector<int> degree(n, 0);
    std::vector<std::pair<int, int>> edges;
    edges.reserve(m);

    for (int i = 0; i < m; i++) {
        int u, v;
        std::cin >> u >> v;
        degree[u]++;
        degree[v]++;
        edges.push_back({u, v});
    }

    // Minimum edge cut = minimum degree in a connected graph
    int min_degree = n + 1;
    int min_node = 0;
    for (int i = 0; i < n; i++) {
        if (degree[i] > 0 && degree[i] < min_degree) {
            min_degree = degree[i];
            min_node = i;
        }
    }

    // Output all edges incident to the minimum degree node
    std::cout << min_degree << '\n';
    for (const auto& edge : edges) {
        if (edge.first == min_node || edge.second == min_node) {
            std::cout << edge.first << ' ' << edge.second << '\n';
        }
    }

    return 0;
}
