#include <iostream>
#include <fstream>
#include <vector>
#include <thread>
#include <mutex>
#include <atomic>
#include <cmath>
#include <chrono>
#include <iomanip>
#include <queue>

using namespace std;

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

struct Command {
    double thrust;
    double pitch_cmd;
    double yaw_cmd;
    double roll_cmd;
};

struct TerrainCell {
    uint16_t ground_top;
    uint16_t floating_bottom;
    uint16_t floating_top;
};

class LunarLander {
private:
    int width, depth, height;
    double gravity, max_thrust;
    double target_x, target_y, target_z;
    vector<vector<TerrainCell>> terrain;

    mutex state_mutex;
    State current_state{};
    Command current_command{};

    atomic<bool> running{true};
    atomic<bool> terrain_loaded{false};

public:
    void loadTerrainFile(const string& filepath) {
        ifstream file(filepath, ios::binary);
        if (!file.is_open()) {
            return;
        }

        terrain.resize(width, vector<TerrainCell>(depth));
        for (int x = 0; x < width; x++) {
            for (int y = 0; y < depth; y++) {
                uint16_t a, b, c;
                file.read((char*)&a, 2);
                file.read((char*)&b, 2);
                file.read((char*)&c, 2);
                terrain[x][y] = {a, b, c};
            }
        }

        terrain_loaded = true;
    }

    bool isColliding(double x, double y, double z) const {
        if (x < 0 || x >= width || y < 0 || y >= depth || z < 0 || z >= height)
            return true;

        int xi = (int)floor(x);
        int yi = (int)floor(y);

        if (xi < 0) xi = 0;
        if (xi >= width) xi = width - 1;
        if (yi < 0) yi = 0;
        if (yi >= depth) yi = depth - 1;

        if (!terrain_loaded || terrain.empty()) return false;

        const TerrainCell& cell = terrain[xi][yi];
        int zi = (int)floor(z);

        if (zi < (int)cell.ground_top) return true;
        if (zi >= (int)cell.floating_bottom && zi < (int)cell.floating_top) return true;

        return false;
    }

    Command computeControl(const State& state) {
        double dx = target_x - state.x;
        double dy = target_y - state.y;
        double dz = target_z - state.z;

        double dist_xy = sqrt(dx*dx + dy*dy);
        double height_above_target = state.z - target_z;

        Command cmd{0, 0, 0, 0};

        // Vertical control
        double desired_vz;
        if (height_above_target > 100) {
            desired_vz = -5.0;
        } else if (height_above_target > 50) {
            desired_vz = -3.0;
        } else if (height_above_target > 20) {
            desired_vz = -1.5;
        } else if (height_above_target > 10) {
            desired_vz = -0.5;
        } else {
            desired_vz = -0.2;
        }

        double vz_error = desired_vz - state.vz;
        double vertical_accel = vz_error / 0.1;
        double thrust_needed = (vertical_accel + gravity) / gravity;
        cmd.thrust = max(0.0, min(1.0, thrust_needed));

        // Horizontal control - aim toward target
        double target_heading = atan2(dy, dx);
        double yaw_error = target_heading - state.yaw;

        while (yaw_error > M_PI) yaw_error -= 2*M_PI;
        while (yaw_error < -M_PI) yaw_error += 2*M_PI;

        // Pitch control - tilt in direction of target
        if (dist_xy > 10) {
            double desired_pitch = -sin(yaw_error) * min(M_PI / 6, dist_xy / 200);
            cmd.pitch_cmd = (desired_pitch - state.pitch) * 0.5;
        } else {
            cmd.pitch_cmd = -state.pitch * 0.3;
        }

        // Yaw control - turn toward target
        cmd.yaw_cmd = yaw_error * 0.4;

        // Roll control - keep level
        cmd.roll_cmd = -state.roll * 0.3;

        // Clamp all commands to valid range
        cmd.pitch_cmd = max(-1.0, min(1.0, cmd.pitch_cmd));
        cmd.yaw_cmd = max(-1.0, min(1.0, cmd.yaw_cmd));
        cmd.roll_cmd = max(-1.0, min(1.0, cmd.roll_cmd));

        return cmd;
    }

    void runInputThread() {
        string line;

        cin >> width >> depth >> height >> gravity >> max_thrust;
        double initial_fuel;
        cin >> initial_fuel;

        double start_x, start_y, start_z;
        cin >> start_x >> start_y >> start_z;
        cin >> target_x >> target_y >> target_z;

        cin.ignore();
        getline(cin, line);

        thread terrain_thread([this, line]() {
            this->loadTerrainFile(line);
        });
        terrain_thread.detach();

        string state_marker;
        cin >> state_marker;

        State state;
        while (cin >> state.x >> state.y >> state.z
                   >> state.vx >> state.vy >> state.vz
                   >> state.pitch >> state.yaw >> state.roll
                   >> state.pitch_rate >> state.yaw_rate >> state.roll_rate
                   >> state.fuel) {
            {
                lock_guard<mutex> lock(state_mutex);
                current_state = state;
            }
        }

        running = false;
    }

    void runControlThread() {
        auto last_output = chrono::high_resolution_clock::now();

        while (running) {
            auto now = chrono::high_resolution_clock::now();
            auto elapsed_ms = chrono::duration_cast<chrono::milliseconds>(now - last_output).count();

            if (elapsed_ms >= 95) {
                {
                    lock_guard<mutex> lock(state_mutex);
                    current_command = computeControl(current_state);
                }

                cout << fixed << setprecision(8)
                     << current_command.thrust << " "
                     << current_command.pitch_cmd << " "
                     << current_command.yaw_cmd << " "
                     << current_command.roll_cmd << "\n";
                cout.flush();

                last_output = now;
            }

            this_thread::sleep_for(chrono::milliseconds(5));
        }
    }

    void run() {
        thread input_thread([this]() { this->runInputThread(); });
        thread control_thread([this]() { this->runControlThread(); });

        input_thread.join();
        control_thread.join();
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    LunarLander lander;
    lander.run();

    return 0;
}
