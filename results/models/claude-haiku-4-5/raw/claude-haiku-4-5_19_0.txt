using System;
using System.Collections.Generic;
using System.Linq;

class MiningOptimizer
{
    struct VoxelInfo
    {
        public int Mass;
        public int Value;
    }

    static void Main()
    {
        var header = Console.ReadLine().Split();
        int xCount = int.Parse(header[0]);
        int yCount = int.Parse(header[1]);
        int zCount = int.Parse(header[2]);

        var voxels = new VoxelInfo[xCount, yCount, zCount];

        // Read voxels
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    byte b = (byte)Console.Read();
                    voxels[x, y, z].Mass = b % 16;
                    voxels[x, y, z].Value = b / 16;
                }
            }
        }

        var operations = new List<Operation>();
        var dug = new HashSet<(int, int, int)>();
        var heightMap = new int[xCount, yCount];

        // Initialize height map - height of topmost solid voxel
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                heightMap[x, y] = -1;
                for (int z = zCount - 1; z >= 0; z--)
                {
                    if (voxels[x, y, z].Mass > 0)
                    {
                        heightMap[x, y] = z;
                        break;
                    }
                }
            }
        }

        // Find all minable voxels and sort by profit
        var candidates = new List<(int x, int y, int z, int mass, int value)>();
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    if (voxels[x, y, z].Mass > 0)
                    {
                        candidates.Add((x, y, z, voxels[x, y, z].Mass, voxels[x, y, z].Value));
                    }
                }
            }
        }

        // Sort by value descending
        candidates.Sort((a, b) => (b.value * b.mass).CompareTo(a.value * a.mass));

        // Mine valuable voxels
        foreach (var (digX, digY, digZ, mass, value) in candidates)
        {
            if (dug.Contains((digX, digY, digZ)))
                continue;

            // Check if exposed (adjacent to empty or boundary)
            if (!IsExposed(voxels, dug, digX, digY, digZ, xCount, yCount, zCount))
                continue;

            // Try to find path to processing plant
            if (FindPath(heightMap, dug, voxels, digX, digY, digZ, xCount, yCount, zCount, out var path))
            {
                operations.Add(new Operation
                {
                    DigX = digX,
                    DigY = digY,
                    DigZ = digZ,
                    Path = path
                });
                dug.Add((digX, digY, digZ));
            }
        }

        // Output results
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write($"{op.Path.Count} {op.DigX} {op.DigY} {op.DigZ}");
            foreach (var (px, py) in op.Path)
            {
                Console.Write($" {px} {py}");
            }
            Console.WriteLine(" 0 0");
        }
    }

    static bool IsExposed(VoxelInfo[,,] voxels, HashSet<(int, int, int)> dug,
        int x, int y, int z, int xCount, int yCount, int zCount)
    {
        // Check adjacent cells
        if (z == zCount - 1 || voxels[x, y, z + 1].Mass == 0 || dug.Contains((x, y, z + 1)))
            return true;

        if (x == 0 || voxels[x - 1, y, z].Mass == 0 || dug.Contains((x - 1, y, z)))
            return true;
        if (x == xCount - 1 || voxels[x + 1, y, z].Mass == 0 || dug.Contains((x + 1, y, z)))
            return true;
        if (y == 0 || voxels[x, y - 1, z].Mass == 0 || dug.Contains((x, y - 1, z)))
            return true;
        if (y == yCount - 1 || voxels[x, y + 1, z].Mass == 0 || dug.Contains((x, y + 1, z)))
            return true;

        return false;
    }

    static bool FindPath(int[,] heightMap, HashSet<(int, int, int)> dug, VoxelInfo[,,] voxels,
        int sx, int sy, int sz, int xCount, int yCount, int zCount, out List<(int, int)> path)
    {
        path = new List<(int, int)>();

        var queue = new Queue<State>();
        var visited = new HashSet<(int, int, int)>();

        queue.Enqueue(new State { X = sx, Y = sy, Z = sz, Path = new List<(int, int)>() });
        visited.Add((sx, sy, sz));

        while (queue.Count > 0)
        {
            var state = queue.Dequeue();

            // Reached processing plant
            if (state.X == 0 && state.Y == 0)
            {
                path = state.Path;
                return true;
            }

            // Try moving in 4 directions
            int[][] directions = {
                new[] { 1, 0 }, new[] { -1, 0 },
                new[] { 0, 1 }, new[] { 0, -1 }
            };

            foreach (var dir in directions)
            {
                int nx = state.X + dir[0];
                int ny = state.Y + dir[1];

                if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount)
                    continue;

                // Find target height at destination
                int nz = heightMap[nx, ny];
                if (nz == -1) nz = 0;

                // Check slope constraint: max 1 Z per X/Y step
                if (Math.Abs(nz - state.Z) > 1)
                    continue;

                if (visited.Contains((nx, ny, nz)))
                    continue;

                visited.Add((nx, ny, nz));
                var newPath = new List<(int, int)>(state.Path);
                newPath.Add((nx, ny));

                queue.Enqueue(new State { X = nx, Y = ny, Z = nz, Path = newPath });
            }
        }

        return false;
    }

    class State
    {
        public int X, Y, Z;
        public List<(int, int)> Path;
    }

    class Operation
    {
        public int DigX, DigY, DigZ;
        public List<(int, int)> Path;
    }
}
