OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Type declarations
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

; Struct for UBO
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float

; Pointer types
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Function_v3float = OpTypePointer Function %v3float
%_ptr_Function_float = OpTypePointer Function %float
%_ptr_Function_v4float = OpTypePointer Function %v4float

; Function type
%void_fn = OpTypeFunction %void

; Constants
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_0p2 = OpConstant %float 0.2
%float_0p4 = OpConstant %float 0.4
%float_0p5 = OpConstant %float 0.5
%float_0p7 = OpConstant %float 0.7
%float_0p75 = OpConstant %float 0.75
%float_1 = OpConstant %float 1.0

; Constant composites
%cool = OpConstantComposite %v3float %float_0p2 %float_0p2 %float_0p75
%warm = OpConstantComposite %v3float %float_0p7 %float_0p7 %float_0p4

; Global variables
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %_ptr_Input_v3float Input
%in_normal = OpVariable %_ptr_Input_v3float Input
%in_uv = OpVariable %_ptr_Input_v2float Input
%in_tangent = OpVariable %_ptr_Input_v3float Input
%in_color = OpVariable %_ptr_Input_v3float Input
%out_color = OpVariable %_ptr_Output_v4float Output

; Decorations
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4

OpDecorate %out_color Location 0

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load input data
%worldPos = OpLoad %v3float %in_worldPos
%normal = OpLoad %v3float %in_normal

; Get light position from UBO
%lightPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos = OpLoad %v4float %lightPosPtr

; Extract xyz from light position and normalize
%lightDir_xyz = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2
%lightDir = OpNormalize %v3float %lightDir_xyz

; Normalize the normal
%normalNorm = OpNormalize %v3float %normal

; Calculate dot(N, L)
%dotNL = OpDot %float %normalNorm %lightDir

; t = dot(N,L) * 0.5 + 0.5
%dotNL_scaled = OpFMul %float %dotNL %float_0p5
%t = OpFAdd %float %dotNL_scaled %float_0p5

; Blend cool and warm colors
; result = cool * (1 - t) + warm * t
%one_minus_t = OpFSub %float %float_1 %t
%cool_scaled = OpVectorTimesScalar %v3float %cool %one_minus_t
%warm_scaled = OpVectorTimesScalar %v3float %warm %t
%color = OpFAdd %v3float %cool_scaled %warm_scaled

; Extract components from color vector
%color_x = OpCompositeExtract %float %color 0
%color_y = OpCompositeExtract %float %color 1
%color_z = OpCompositeExtract %float %color 2

; Create output color with alpha = 1.0
%outColor = OpCompositeConstruct %v4float %color_x %color_y %color_z %float_1

; Store output
OpStore %out_color %outColor

; Return
OpReturn
OpFunctionEnd
