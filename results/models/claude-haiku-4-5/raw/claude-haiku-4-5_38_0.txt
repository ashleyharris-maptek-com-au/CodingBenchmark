def solve():
    # Read input
    line = input().split()
    U = int(line[0])
    S = int(line[1])

    sets = []
    for _ in range(S):
        elements = list(map(int, input().split()))
        sets.append(set(elements))

    # Backtracking to find exact cover
    def search(covered, used_indices):
        # Base case: if all elements are covered
        if len(covered) == U:
            return used_indices

        # Find first uncovered element (to guide search)
        uncovered = None
        for i in range(U):
            if i not in covered:
                uncovered = i
                break

        # Try each set that covers this uncovered element
        for i in range(S):
            if i not in used_indices and uncovered in sets[i]:
                # Check if adding this set causes any overlap
                if not (sets[i] & covered):  # No intersection with covered
                    new_covered = covered | sets[i]
                    result = search(new_covered, used_indices | {i})
                    if result is not None:
                        return result

        return None

    result = search(set(), set())

    if result is None:
        print("NO SOLUTION")
    else:
        indices = sorted(list(result))
        print("SOLUTION")
        print(" ".join(map(str, indices)))

solve()
