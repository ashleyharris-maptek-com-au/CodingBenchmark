using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    static void Main()
    {
        string[] dims = Console.ReadLine().Split();
        int h = int.Parse(dims[0]);
        int w = int.Parse(dims[1]);

        char[][] board = new char[h][];
        for (int i = 0; i < h; i++)
        {
            board[i] = Console.ReadLine().ToCharArray();
        }

        HashSet<(int, int)> safe = FindSafeCells(board, h, w);

        Console.WriteLine(safe.Count);
        foreach (var (x, y) in safe.OrderBy(c => c.Item1).ThenBy(c => c.Item2))
        {
            Console.WriteLine($"{x} {y}");
        }
    }

    static HashSet<(int, int)> FindSafeCells(char[][] board, int h, int w)
    {
        HashSet<(int, int)> safe = new HashSet<(int, int)>();
        bool changed = true;

        while (changed)
        {
            changed = false;

            for (int y = 0; y < h; y++)
            {
                for (int x = 0; x < w; x++)
                {
                    if (char.IsDigit(board[y][x]))
                    {
                        int mineCount = board[y][x] - '0';

                        List<(int, int)> unrevealed = new List<(int, int)>();
                        int flaggedCount = 0;

                        for (int dy = -1; dy <= 1; dy++)
                        {
                            for (int dx = -1; dx <= 1; dx++)
                            {
                                if (dy == 0 && dx == 0) continue;
                                int ny = y + dy, nx = x + dx;
                                if (ny >= 0 && ny < h && nx >= 0 && nx < w)
                                {
                                    if (board[ny][nx] == '*')
                                        flaggedCount++;
                                    else if (board[ny][nx] == ' ')
                                        unrevealed.Add((nx, ny));
                                }
                            }
                        }

                        if (flaggedCount == mineCount && unrevealed.Count > 0)
                        {
                            foreach (var (ux, uy) in unrevealed)
                            {
                                if (safe.Add((ux, uy)))
                                    changed = true;
                            }
                        }
                    }
                }
            }

            foreach (var (x, y) in safe)
            {
                if (board[y][x] == ' ')
                    board[y][x] = 'S';
            }
        }

        return safe;
    }
}
