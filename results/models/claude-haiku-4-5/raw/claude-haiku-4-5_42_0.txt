OpCapability Geometry
OpMemoryModel Logical GLSL450

OpName %main "main"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Constants
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_04 = OpConstant %float 0.04

; Input gl_PerVertex structure
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input gl_PerVertex array
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input color array
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%color_in = OpVariable %ptr_in_arr_v3 Input
OpDecorate %color_in Location 0

; Output gl_PerVertex structure
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%color_out = OpVariable %ptr_out_v3 Output
OpDecorate %color_out Location 0

; Entry point
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %color_out

; Execution modes
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Main function
%fn_void = OpTypeFunction %void
%main = OpFunction %void None %fn_void
%entry = OpLabel

; Pointer types for accessors
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; Load input position
%pos_in_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%input_pos = OpLoad %v4float %pos_in_ptr

; Load input color
%color_in_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%input_color = OpLoad %v3float %color_in_ptr

; Extract x, y
%x = OpCompositeExtract %float %input_pos 0
%y = OpCompositeExtract %float %input_pos 1

; Calculate quad vertices (as triangle strip: v0, v1, v2, v3)
; v0: (x-h, y-h)
%x0 = OpFSub %float %x %float_0_04
%y0 = OpFSub %float %y %float_0_04
%pos0 = OpCompositeConstruct %v4float %x0 %y0 %float_0 %float_1

; v1: (x+h, y-h)
%x1 = OpFAdd %float %x %float_0_04
%y1 = OpFSub %float %y %float_0_04
%pos1 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1

; v2: (x-h, y+h)
%x2 = OpFSub %float %x %float_0_04
%y2 = OpFAdd %float %y %float_0_04
%pos2 = OpCompositeConstruct %v4float %x2 %y2 %float_0 %float_1

; v3: (x+h, y+h)
%x3 = OpFAdd %float %x %float_0_04
%y3 = OpFAdd %float %y %float_0_04
%pos3 = OpCompositeConstruct %v4float %x3 %y3 %float_0 %float_1

; Emit vertices
%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit v0
OpStore %pos_out_ptr %pos0
OpStore %color_out %input_color
OpEmitVertex

; Emit v1
OpStore %pos_out_ptr %pos1
OpStore %color_out %input_color
OpEmitVertex

; Emit v2
OpStore %pos_out_ptr %pos2
OpStore %color_out %input_color
OpEmitVertex

; Emit v3
OpStore %pos_out_ptr %pos3
OpStore %color_out %input_color
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
