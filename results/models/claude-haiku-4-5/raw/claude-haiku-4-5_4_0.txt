#include <iostream>
#include <vector>
#include <cmath>
#include <array>

using namespace std;

const double PI = acos(-1.0);
const double EPS = 1e-9;
const double TET_EDGE = 1.0;
const double TET_HEIGHT = TET_EDGE * sqrt(2.0/3.0);
const double TET_CIRCUM = TET_EDGE * sqrt(3.0/8.0);
const double TET_INRAD = TET_EDGE / sqrt(24.0);

struct Vec3 {
    double x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x+v.x, y+v.y, z+v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x-v.x, y-v.y, z-v.z); }
    Vec3 operator*(double t) const { return Vec3(x*t, y*t, z*t); }
    Vec3 operator/(double t) const { return Vec3(x/t, y/t, z/t); }

    double dot(const Vec3& v) const { return x*v.x + y*v.y + z*v.z; }
    Vec3 cross(const Vec3& v) const { return Vec3(y*v.z - z*v.y, z*v.x - x*v.z, x*v.y - y*v.x); }
    double len() const { return sqrt(x*x + y*y + z*z); }
    Vec3 norm() const { double l = len(); return l > EPS ? *this / l : *this; }
};

struct Tetrahedron {
    Vec3 center;
    double rx, ry, rz;

    vector<Vec3> vertices() const {
        double h = TET_HEIGHT / 4;
        double r = TET_CIRCUM * 0.75;

        double cr = cos(rx), sr = sin(rx);
        double cp = cos(ry), sp = sin(ry);
        double cy = cos(rz), sy = sin(rz);

        Vec3 v0(0, 0, TET_HEIGHT * 0.25);
        Vec3 v1(r * sin(2*PI/3 + 0), r * cos(2*PI/3 + 0), -TET_HEIGHT/3);
        Vec3 v2(r * sin(2*PI/3 + 2*PI/3), r * cos(2*PI/3 + 2*PI/3), -TET_HEIGHT/3);
        Vec3 v3(r * sin(2*PI/3 + 4*PI/3), r * cos(2*PI/3 + 4*PI/3), -TET_HEIGHT/3);

        auto rotate = [&](Vec3 p) {
            Vec3 p1(p.x, p.y * cr - p.z * sr, p.y * sr + p.z * cr);
            Vec3 p2(p1.x * cp + p1.z * sp, p1.y, -p1.x * sp + p1.z * cp);
            Vec3 p3(p2.x * cy - p2.y * sy, p2.x * sy + p2.y * cy, p2.z);
            return center + p3;
        };

        return {rotate(v0), rotate(v1), rotate(v2), rotate(v3)};
    }
};

bool pointInPolyhedron(const Vec3& p, const vector<Vec3>& verts, const vector<array<int,3>>& faces) {
    int crossings = 0;
    Vec3 ray(1.0, 0.0, 0.0);

    for (auto& f : faces) {
        Vec3 a = verts[f[0]];
        Vec3 b = verts[f[1]];
        Vec3 c = verts[f[2]];

        Vec3 ab = b - a;
        Vec3 ac = c - a;
        Vec3 n = ab.cross(ac);

        double denom = n.dot(ray);
        if (abs(denom) < EPS) continue;

        double t = n.dot(a - p) / denom;
        if (t < EPS) continue;

        Vec3 q = p + ray * t;

        double d1 = ab.cross(q - a).dot(n);
        double d2 = ac.cross(q - b).dot(n);
        double d3 = (a - c).cross(q - c).dot(n);

        if (d1 >= -EPS && d2 >= -EPS && d3 >= -EPS) crossings++;
    }

    return (crossings % 2) == 1;
}

bool tetsOverlap(const Tetrahedron& t1, const Tetrahedron& t2) {
    double dist = (t1.center - t2.center).len();
    return dist < 2 * TET_CIRCUM - EPS;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge_length;
    cin >> V >> edge_length;

    vector<Vec3> vertices(V);
    for (int i = 0; i < V; i++) {
        cin >> vertices[i].x >> vertices[i].y >> vertices[i].z;
    }

    Vec3 min_v = vertices[0], max_v = vertices[0];
    for (auto& v : vertices) {
        min_v.x = min(min_v.x, v.x); min_v.y = min(min_v.y, v.y); min_v.z = min(min_v.z, v.z);
        max_v.x = max(max_v.x, v.x); max_v.y = max(max_v.y, v.y); max_v.z = max(max_v.z, v.z);
    }

    double vol = (max_v.x - min_v.x) * (max_v.y - min_v.y) * (max_v.z - min_v.z);
    double tet_vol = TET_EDGE * TET_EDGE * TET_EDGE / 12;
    int max_count = max(1, (int)(vol / tet_vol * 0.5));

    vector<Tetrahedron> packed;
    double grid_step = TET_CIRCUM;

    for (double x = min_v.x + grid_step; x <= max_v.x - grid_step; x += grid_step * 0.5) {
        for (double y = min_v.y + grid_step; y <= max_v.y - grid_step; y += grid_step * 0.5) {
            for (double z = min_v.z + grid_step; z <= max_v.z - grid_step; z += grid_step * 0.5) {
                if (packed.size() >= max_count) break;

                Vec3 center(x, y, z);
                if (!pointInPolyhedron(center, vertices, {{0,1,2}})) continue;

                bool overlaps = false;
                for (auto& t : packed) {
                    if (tetsOverlap(t, {center, 0, 0, 0})) {
                        overlaps = true;
                        break;
                    }
                }

                if (!overlaps) {
                    packed.push_back({center, 0, 0, 0});
                }
            }
            if (packed.size() >= max_count) break;
        }
        if (packed.size() >= max_count) break;
    }

    cout << packed.size() << "\n";
    for (auto& t : packed) {
        printf("%.6f %.6f %.6f %.6f %.6f %.6f\n", t.center.x, t.center.y, t.center.z, t.rx, t.ry, t.rz);
    }

    return 0;
}
