[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    float mortarThicknessX = 0.05;
    float mortarThicknessY = 0.1;
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);

    float2 brickUV = input.uv * float2(8.0, 16.0);
    float row = floor(brickUV.y);

    float u = brickUV.x;
    if (fmod(row, 2.0) > 0.5) {
        u += 0.5;
    }

    float2 localUV = frac(float2(u, brickUV.y));

    bool inMortarX = (localUV.x < mortarThicknessX) || (localUV.x > (1.0 - mortarThicknessX));
    bool inMortarY = (localUV.y < mortarThicknessY) || (localUV.y > (1.0 - mortarThicknessY));
    bool inMortar = inMortarX || inMortarY;

    float3 surfaceColor = inMortar ? mortarColor : brickColor;

    float3 normal = normalize(input.normal);
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(normal, lightDir), 0.0);

    float lambertian = 0.4 + 0.6 * diffuse;

    float3 finalColor = surfaceColor * lambertian;

    return float4(finalColor, 1.0);
}
