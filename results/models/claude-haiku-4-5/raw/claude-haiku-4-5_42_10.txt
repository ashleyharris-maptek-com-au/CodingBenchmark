OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %color_in
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

OpExtInstImport "GLSL.std.450"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
OpDecorate %color_in Location 2

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_4 = OpConstant %int 4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3_in = OpTypeArray %v3float %uint_4

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%void_func = OpTypeFunction %void

%gl_in = OpVariable %ptr_in_arr_pv Input
%color_in = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%float_0 = OpConstant %float 0.0
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
%float_1 = OpConstant %float 1.0
%float_0_02 = OpConstant %float 0.02
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%glsl = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %void_func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
%c2 = OpLoad %v3float %c2_ptr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p3_xyz = OpVectorShuffle %v3float %p3 %p3 0 1 2

%t_start = OpFSub %v3float %p2_xyz %p0_xyz
%t_end = OpFSub %v3float %p3_xyz %p1_xyz
%dir = OpFSub %v3float %p2_xyz %p1_xyz

%up = OpConstantComposite %v3float %float_0 %float_1 %float_0

; ========== t=0 ==========
%pos_0 = %p1_xyz
%tang_0 = %t_start
%cross_0 = OpExtInst %v3float %glsl CrossProduct %tang_0 %up
%perp_norm_0 = OpExtInst %v3float %glsl Normalize %cross_0
%perp_0 = OpVectorTimesScalar %v3float %perp_norm_0 %float_0_02

%v0_p = OpFAdd %v3float %pos_0 %perp_0
%v0_p_v4 = OpCompositeConstruct %v4float %v0_p %float_1
%gl_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_pos %v0_p_v4
OpStore %outColor %c1
OpEmitVertex

%v0_n = OpFSub %v3float %pos_0 %perp_0
%v0_n_v4 = OpCompositeConstruct %v4float %v0_n %float_1
OpStore %gl_pos %v0_n_v4
OpStore %outColor %c1
OpEmitVertex

; ========== t=0.25 ==========
%t1_inv = OpFSub %float %float_1 %float_0_25
%offset_1 = OpVectorTimesScalar %v3float %dir %float_0_25
%pos_1 = OpFAdd %v3float %p1_xyz %offset_1
%t_start_scaled_1 = OpVectorTimesScalar %v3float %t_start %t1_inv
%t_end_scaled_1 = OpVectorTimesScalar %v3float %t_end %float_0_25
%tang_1 = OpFAdd %v3float %t_start_scaled_1 %t_end_scaled_1
%cross_1 = OpExtInst %v3float %glsl CrossProduct %tang_1 %up
%perp_norm_1 = OpExtInst %v3float %glsl Normalize %cross_1
%perp_1 = OpVectorTimesScalar %v3float %perp_norm_1 %float_0_02

%c1_diff = OpFSub %v3float %c2 %c1
%c1_scaled = OpVectorTimesScalar %v3float %c1_diff %float_0_25
%color_1 = OpFAdd %v3float %c1 %c1_scaled

%v1_p = OpFAdd %v3float %pos_1 %perp_1
%v1_p_v4 = OpCompositeConstruct %v4float %v1_p %float_1
OpStore %gl_pos %v1_p_v4
OpStore %outColor %color_1
OpEmitVertex

%v1_n = OpFSub %v3float %pos_1 %perp_1
%v1_n_v4 = OpCompositeConstruct %v4float %v1_n %float_1
OpStore %gl_pos %v1_n_v4
OpStore %outColor %color_1
OpEmitVertex

; ========== t=0.5 ==========
%t2_inv = OpFSub %float %float_1 %float_0_5
%offset_2 = OpVectorTimesScalar %v3float %dir %float_0_5
%pos_2 = OpFAdd %v3float %p1_xyz %offset_2
%t_start_scaled_2 = OpVectorTimesScalar %v3float %t_start %t2_inv
%t_end_scaled_2 = OpVectorTimesScalar %v3float %t_end %float_0_5
%tang_2 = OpFAdd %v3float %t_start_scaled_2 %t_end_scaled_2
%cross_2 = OpExtInst %v3float %glsl CrossProduct %tang_2 %up
%perp_norm_2 = OpExtInst %v3float %glsl Normalize %cross_2
%perp_2 = OpVectorTimesScalar %v3float %perp_norm_2 %float_0_02

%c2_diff = OpFSub %v3float %c2 %c1
%c2_scaled = OpVectorTimesScalar %v3float %c2_diff %float_0_5
%color_2 = OpFAdd %v3float %c1 %c2_scaled

%v2_p = OpFAdd %v3float %pos_2 %perp_2
%v2_p_v4 = OpCompositeConstruct %v4float %v2_p %float_1
OpStore %gl_pos %v2_p_v4
OpStore %outColor %color_2
OpEmitVertex

%v2_n = OpFSub %v3float %pos_2 %perp_2
%v2_n_v4 = OpCompositeConstruct %v4float %v2_n %float_1
OpStore %gl_pos %v2_n_v4
OpStore %outColor %color_2
OpEmitVertex

; ========== t=0.75 ==========
%t3_inv = OpFSub %float %float_1 %float_0_75
%offset_3 = OpVectorTimesScalar %v3float %dir %float_0_75
%pos_3 = OpFAdd %v3float %p1_xyz %offset_3
%t_start_scaled_3 = OpVectorTimesScalar %v3float %t_start %t3_inv
%t_end_scaled_3 = OpVectorTimesScalar %v3float %t_end %float_0_75
%tang_3 = OpFAdd %v3float %t_start_scaled_3 %t_end_scaled_3
%cross_3 = OpExtInst %v3float %glsl CrossProduct %tang_3 %up
%perp_norm_3 = OpExtInst %v3float %glsl Normalize %cross_3
%perp_3 = OpVectorTimesScalar %v3float %perp_norm_3 %float_0_02

%c3_diff = OpFSub %v3float %c2 %c1
%c3_scaled = OpVectorTimesScalar %v3float %c3_diff %float_0_75
%color_3 = OpFAdd %v3float %c1 %c3_scaled

%v3_p = OpFAdd %v3float %pos_3 %perp_3
%v3_p_v4 = OpCompositeConstruct %v4float %v3_p %float_1
OpStore %gl_pos %v3_p_v4
OpStore %outColor %color_3
OpEmitVertex

%v3_n = OpFSub %v3float %pos_3 %perp_3
%v3_n_v4 = OpCompositeConstruct %v4float %v3_n %float_1
OpStore %gl_pos %v3_n_v4
OpStore %outColor %color_3
OpEmitVertex

; ========== t=1.0 ==========
%pos_4 = %p2_xyz
%tang_4 = %t_end
%cross_4 = OpExtInst %v3float %glsl CrossProduct %tang_4 %up
%perp_norm_4 = OpExtInst %v3float %glsl Normalize %cross_4
%perp_4 = OpVectorTimesScalar %v3float %perp_norm_4 %float_0_02

%v4_p = OpFAdd %v3float %pos_4 %perp_4
%v4_p_v4 = OpCompositeConstruct %v4float %v4_p %float_1
OpStore %gl_pos %v4_p_v4
OpStore %outColor %c2
OpEmitVertex

%v4_n = OpFSub %v3float %pos_4 %perp_4
%v4_n_v4 = OpCompositeConstruct %v4float %v4_n %float_1
OpStore %gl_pos %v4_n_v4
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
