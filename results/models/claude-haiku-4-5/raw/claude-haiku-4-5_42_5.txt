OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %color
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in ArrayStride 16
OpDecorate %color ArrayStride 16
OpDecorate %gl_out Block
OpMemberDecorate %gl_out 0 BuiltIn Position
OpDecorate %outColor Location 0
OpDecorate %color Location 2

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr = OpTypePointer Input %arr_gl_PerVertex
%ptr_Input_v4 = OpTypePointer Input %v4float
%arr_v3float = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3float
%ptr_Input_v3 = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr Input
%color = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output
%outColor = OpVariable %ptr_Output_v3 Output

%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%entry = OpLabel

%pos0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%pos0_v4 = OpLoad %v4float %pos0_ptr
%pos0 = OpVectorShuffle %v3float %pos0_v4 %pos0_v4 0 1 2

%pos1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%pos1_v4 = OpLoad %v4float %pos1_ptr
%pos1 = OpVectorShuffle %v3float %pos1_v4 %pos1_v4 0 1 2

%c0_ptr = OpAccessChain %ptr_Input_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr

%c1_ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr

%dir = OpFSub %v3float %pos1 %pos0

%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

%perp_x = OpFNegate %float %dir_y
%perp_y = %dir_x
%perp_unnorm = OpCompositeConstruct %v3float %perp_x %perp_y %float_0

%perp_len_sq = OpDot %float %perp_unnorm %perp_unnorm
%perp_len = OpSqrt %float %perp_len_sq
%perp_len_inv = OpFDiv %float %float_1 %perp_len
%perp = OpVectorTimesScalar %v3float %perp_unnorm %perp_len_inv

%dir_scaled = OpVectorTimesScalar %v3float %dir %float_0_75
%mid = OpFAdd %v3float %pos0 %dir_scaled

%shaft_off = OpVectorTimesScalar %v3float %perp %float_0_015
%arrow_off = OpVectorTimesScalar %v3float %perp %float_0_05

%v1_pos = OpFAdd %v3float %pos0 %shaft_off
%v1_x = OpCompositeExtract %float %v1_pos 0
%v1_y = OpCompositeExtract %float %v1_pos 1
%v1_z = OpCompositeExtract %float %v1_pos 2
%v1_v4 = OpCompositeConstruct %v4float %v1_x %v1_y %v1_z %float_1
%v1_gout = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v1_gout %v1_v4
OpStore %outColor %c0
OpEmitVertex

%v2_pos = OpFSub %v3float %pos0 %shaft_off
%v2_x = OpCompositeExtract %float %v2_pos 0
%v2_y = OpCompositeExtract %float %v2_pos 1
%v2_z = OpCompositeExtract %float %v2_pos 2
%v2_v4 = OpCompositeConstruct %v4float %v2_x %v2_y %v2_z %float_1
%v2_gout = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v2_gout %v2_v4
OpStore %outColor %c0
OpEmitVertex

%v3_pos = OpFAdd %v3float %mid %shaft_off
%v3_x = OpCompositeExtract %float %v3_pos 0
%v3_y = OpCompositeExtract %float %v3_pos 1
%v3_z = OpCompositeExtract %float %v3_pos 2
%v3_v4 = OpCompositeConstruct %v4float %v3_x %v3_y %v3_z %float_1
%v3_gout = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v3_gout %v3_v4
OpStore %outColor %c0
OpEmitVertex

%v4_pos = OpFSub %v3float %mid %shaft_off
%v4_x = OpCompositeExtract %float %v4_pos 0
%v4_y = OpCompositeExtract %float %v4_pos 1
%v4_z = OpCompositeExtract %float %v4_pos 2
%v4_v4 = OpCompositeConstruct %v4float %v4_x %v4_y %v4_z %float_1
%v4_gout = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v4_gout %v4_v4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

%v5_pos = OpFAdd %v3float %mid %arrow_off
%v5_x = OpCompositeExtract %float %v5_pos 0
%v5_y = OpCompositeExtract %float %v5_pos 1
%v5_z = OpCompositeExtract %float %v5_pos 2
%v5_v4 = OpCompositeConstruct %v4float %v5_x %v5_y %v5_z %float_1
%v5_gout = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v5_gout %v5_v4
OpStore %outColor %c1
OpEmitVertex

%v6_pos = OpFSub %v3float %mid %arrow_off
%v6_x = OpCompositeExtract %float %v6_pos 0
%v6_y = OpCompositeExtract %float %v6_pos 1
%v6_z = OpCompositeExtract %float %v6_pos 2
%v6_v4 = OpCompositeConstruct %v4float %v6_x %v6_y %v6_z %float_1
%v6_gout = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v6_gout %v6_v4
OpStore %outColor %c1
OpEmitVertex

%v7_x = OpCompositeExtract %float %pos1 0
%v7_y = OpCompositeExtract %float %pos1 1
%v7_z = OpCompositeExtract %float %pos1 2
%v7_v4 = OpCompositeConstruct %v4float %v7_x %v7_y %v7_z %float_1
%v7_gout = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %v7_gout %v7_v4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
