#include <iostream>
#include <vector>
#include <limits>

using namespace std;

int n, m;
vector<long long> c;
vector<pair<int, int>> constraints;
vector<int> best_solution;
long long best_value = LLONG_MIN;

bool is_feasible(const vector<int>& assignment, int vars_assigned) {
    for (const auto& [u, v] : constraints) {
        if (u < vars_assigned && v < vars_assigned) {
            if (assignment[u] == 1 && assignment[v] == 1) {
                return false;
            }
        }
    }
    return true;
}

long long compute_upper_bound(const vector<int>& assignment, int vars_assigned) {
    long long bound = 0;

    for (int i = 0; i < vars_assigned; i++) {
        if (assignment[i] == 1) {
            bound += c[i];
        }
    }

    for (int i = vars_assigned; i < n; i++) {
        if (c[i] > 0) {
            bound += c[i];
        }
    }

    return bound;
}

void branch_and_bound(vector<int>& assignment, int var) {
    if (!is_feasible(assignment, var)) {
        return;
    }

    if (var == n) {
        long long value = 0;
        for (int i = 0; i < n; i++) {
            if (assignment[i] == 1) {
                value += c[i];
            }
        }
        if (value > best_value) {
            best_value = value;
            best_solution = assignment;
        }
        return;
    }

    long long upper_bound = compute_upper_bound(assignment, var);
    if (upper_bound <= best_value) {
        return;
    }

    assignment[var] = 1;
    branch_and_bound(assignment, var + 1);

    assignment[var] = 0;
    branch_and_bound(assignment, var + 1);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;

    c.resize(n);
    for (int i = 0; i < n; i++) {
        cin >> c[i];
    }

    constraints.resize(m);
    for (int i = 0; i < m; i++) {
        cin >> constraints[i].first >> constraints[i].second;
    }

    vector<int> assignment(n, 0);
    branch_and_bound(assignment, 0);

    cout << best_value << "\n";
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << best_solution[i];
    }
    cout << "\n";

    return 0;
}
