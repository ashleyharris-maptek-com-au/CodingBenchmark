; SPIR-V
; Version: 1.0
; Generator: Custom

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %out_outColor
OpExecutionMode %main OriginUpperLeft

OpExtInstImport %glsl "GLSL.std.450"

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4

%main_type = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9
%float_1 = OpConstant %float 1.0
%float_8 = OpConstant %float 8.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

; Variables
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%out_outColor = OpVariable %ptr_Output_vec4 Output

; Entry point
%main = OpFunction %void None %main_type
%entry = OpLabel

; Load inputs
%uv = OpLoad %vec2 %in_uv
%normal = OpLoad %vec3 %in_normal
%worldPos = OpLoad %vec3 %in_worldPos

; Normalize normal
%normal_norm = OpExtInst %vec3 %glsl Normalize %normal

; Scale UV by 8
%uv_scaled = OpVectorTimesScalar %vec2 %uv %float_8

; Floor the scaled UV
%uv_floored = OpExtInst %vec2 %glsl Floor %uv_scaled

; Extract x and y
%uv_x = OpCompositeExtract %float %uv_floored 0
%uv_y = OpCompositeExtract %float %uv_floored 1

; Convert to int
%x_int = OpConvertFToS %int %uv_x
%y_int = OpConvertFToS %int %uv_y

; Check parity - (x + y) % 2 == 0 for checkerboard
%sum = OpIAdd %int %x_int %y_int
%remainder = OpSRem %int %sum %int_2
%is_even = OpIEqual %bool %remainder %int_0

; Select color based on parity
%check_color = OpSelect %float %is_even %float_0_9 %float_0_2

; Load light position from UBO
%light_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo %int_3
%light_pos_full = OpLoad %vec4 %light_ptr
%light_pos = OpVectorShuffle %vec3 %light_pos_full %light_pos_full 0 1 2

; Calculate light direction (light - fragment)
%light_dir = OpFSub %vec3 %light_pos %worldPos

; Normalize light direction
%light_dir_norm = OpExtInst %vec3 %glsl Normalize %light_dir

; Calculate Lambertian diffuse (dot product, clamped to 0)
%dot_nl = OpDot %float %normal_norm %light_dir_norm
%diffuse = OpExtInst %float %glsl FMax %dot_nl %float_0

; Calculate shading factor: 0.5 + 0.5 * diffuse
%factor_scaled = OpFMul %float %float_0_5 %diffuse
%shading_factor = OpFAdd %float %float_0_5 %factor_scaled

; Apply shading to checkerboard color
%final_color = OpFMul %float %check_color %shading_factor

; Create output color (RGB = final_color, A = 1.0)
%output_color = OpCompositeConstruct %vec4 %final_color %final_color %final_color %float_1

; Store output
OpStore %out_outColor %output_color

OpReturn
OpFunctionEnd
