use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let width: usize = parts[0].parse().unwrap();
    let height: usize = parts[1].parse().unwrap();
    let obstacle_count: usize = parts[2].parse().unwrap();
    let require_cycle: bool = parts[3].parse::<u32>().unwrap() != 0;

    let total_cells = width * height;
    let mut obstacles = HashSet::new();

    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<&str> = line.split_whitespace().collect();
        let x: usize = coords[0].parse().unwrap();
        let y: usize = coords[1].parse().unwrap();
        obstacles.insert((x, y));
    }

    if obstacles.contains(&(0, 0)) {
        println!("0");
        return;
    }

    let valid_cells = total_cells - obstacle_count;

    let path = if width * height <= 400 {
        find_hamiltonian_backtrack(width, height, &obstacles, valid_cells, require_cycle)
    } else {
        find_hamiltonian_smart(width, height, &obstacles, valid_cells, require_cycle)
    };

    if let Some(path) = path {
        println!("{}", path.len());
        for (x, y) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn find_hamiltonian_backtrack(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    valid_cells: usize,
    require_cycle: bool,
) -> Option<Vec<(usize, usize)>> {
    let mut visited = HashSet::new();
    let mut path = Vec::new();
    visited.insert((0, 0));
    path.push((0, 0));

    if backtrack(
        width,
        height,
        obstacles,
        &mut visited,
        &mut path,
        valid_cells,
        require_cycle,
    ) {
        Some(path)
    } else {
        None
    }
}

fn backtrack(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    visited: &mut HashSet<(usize, usize)>,
    path: &mut Vec<(usize, usize)>,
    valid_cells: usize,
    require_cycle: bool,
) -> bool {
    if visited.len() == valid_cells {
        if require_cycle {
            let (lx, ly) = path[path.len() - 1];
            is_adjacent(lx, ly, 0, 0)
        } else {
            true
        }
    } else {
        let (cx, cy) = path[path.len() - 1];
        let mut neighbors = get_neighbors(cx, cy, width, height);

        neighbors.sort_by_key(|(x, y)| {
            if visited.contains(&(*x, *y)) {
                1000
            } else {
                count_unvisited(
                    *x, *y, width, height, visited, obstacles,
                )
            }
        });

        for (nx, ny) in neighbors {
            if !visited.contains(&(nx, ny)) && !obstacles.contains(&(nx, ny)) {
                visited.insert((nx, ny));
                path.push((nx, ny));

                if backtrack(width, height, obstacles, visited, path, valid_cells, require_cycle) {
                    return true;
                }

                path.pop();
                visited.remove(&(nx, ny));
            }
        }
        false
    }
}

fn count_unvisited(
    x: usize,
    y: usize,
    width: usize,
    height: usize,
    visited: &HashSet<(usize, usize)>,
    obstacles: &HashSet<(usize, usize)>,
) -> usize {
    let mut count = 0;
    for (nx, ny) in get_neighbors(x, y, width, height) {
        if !visited.contains(&(nx, ny)) && !obstacles.contains(&(nx, ny)) {
            count += 1;
        }
    }
    count
}

fn find_hamiltonian_smart(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    valid_cells: usize,
    require_cycle: bool,
) -> Option<Vec<(usize, usize)>> {
    if let Some(path) = try_snake_pattern(width, height, obstacles, valid_cells, require_cycle) {
        return Some(path);
    }

    greedy_path(width, height, obstacles, valid_cells, require_cycle)
}

fn try_snake_pattern(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    valid_cells: usize,
    require_cycle: bool,
) -> Option<Vec<(usize, usize)>> {
    let mut path = Vec::new();
    let mut visited = HashSet::new();

    for y in 0..height {
        if y % 2 == 0 {
            for x in 0..width {
                if !obstacles.contains(&(x, y)) {
                    path.push((x, y));
                    visited.insert((x, y));
                }
            }
        } else {
            for x in (0..width).rev() {
                if !obstacles.contains(&(x, y)) {
                    path.push((x, y));
                    visited.insert((x, y));
                }
            }
        }
    }

    if visited.len() == valid_cells && !path.is_empty() && path[0] == (0, 0) {
        if require_cycle {
            let (lx, ly) = path[path.len() - 1];
            if is_adjacent(lx, ly, 0, 0) {
                Some(path)
            } else {
                None
            }
        } else {
            Some(path)
        }
    } else {
        None
    }
}

fn greedy_path(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    valid_cells: usize,
    require_cycle: bool,
) -> Option<Vec<(usize, usize)>> {
    let mut visited = HashSet::new();
    let mut path = Vec::new();
    visited.insert((0, 0));
    path.push((0, 0));

    while visited.len() < valid_cells {
        let (cx, cy) = path[path.len() - 1];
        let neighbors = get_neighbors(cx, cy, width, height);

        let mut best = None;
        let mut best_degree = usize::MAX;

        for (nx, ny) in neighbors {
            if !visited.contains(&(nx, ny)) && !obstacles.contains(&(nx, ny)) {
                let degree = count_unvisited(nx, ny, width, height, &visited, obstacles);
                if degree < best_degree || (degree == best_degree && best.is_none()) {
                    best_degree = degree;
                    best = Some((nx, ny));
                }
            }
        }

        if let Some((nx, ny)) = best {
            visited.insert((nx, ny));
            path.push((nx, ny));
        } else {
            break;
        }
    }

    if visited.len() == valid_cells {
        if require_cycle {
            let (lx, ly) = path[path.len() - 1];
            if is_adjacent(lx, ly, 0, 0) {
                Some(path)
            } else {
                None
            }
        } else {
            Some(path)
        }
    } else {
        None
    }
}

fn is_adjacent(x1: usize, y1: usize, x2: usize, y2: usize) -> bool {
    ((x1 as i32 - x2 as i32).abs() + (y1 as i32 - y2 as i32).abs()) == 1
}

fn get_neighbors(x: usize, y: usize, width: usize, height: usize) -> Vec<(usize, usize)> {
    let mut neighbors = Vec::new();
    if x > 0 { neighbors.push((x - 1, y)); }
    if x < width - 1 { neighbors.push((x + 1, y)); }
    if y > 0 { neighbors.push((x, y - 1)); }
    if y < height - 1 { neighbors.push((x, y + 1)); }
    neighbors
}
