#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <unordered_set>

using namespace std;

bool hasCommonSubstring(const vector<string>& strings, int len) {
    if (strings.empty() || len <= 0) return false;

    unordered_set<string> candidates;

    // Get all substrings of length `len` from the first string
    for (size_t i = 0; i + len <= strings[0].length(); i++) {
        candidates.insert(strings[0].substr(i, len));
    }

    if (candidates.empty()) return false;

    // Check if any candidate exists in all other strings
    for (size_t j = 1; j < strings.size(); j++) {
        unordered_set<string> current;
        for (size_t i = 0; i + len <= strings[j].length(); i++) {
            string substr = strings[j].substr(i, len);
            if (candidates.count(substr)) {
                current.insert(substr);
            }
        }
        candidates = current;
        if (candidates.empty()) return false;
    }

    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }

    if (n == 0) {
        cout << "" << endl;
        return 0;
    }

    // Find the minimum length string
    int minLen = INT_MAX;
    for (const auto& s : strings) {
        minLen = min(minLen, (int)s.length());
    }

    if (minLen <= 0) {
        cout << "" << endl;
        return 0;
    }

    // Binary search on the length of the answer
    // If a common substring of length L exists, one of length L-1 also exists
    int low = 0, high = minLen;
    int ansLen = 0;

    while (low <= high) {
        int mid = (low + high) / 2;
        if (hasCommonSubstring(strings, mid)) {
            ansLen = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    // Find the actual substring of length ansLen
    string result = "";

    if (ansLen > 0) {
        // Get all substrings of length ansLen from the first string
        unordered_set<string> candidates;
        for (size_t i = 0; i + ansLen <= strings[0].length(); i++) {
            candidates.insert(strings[0].substr(i, ansLen));
        }

        // Find one that exists in all other strings
        for (const auto& candidate : candidates) {
            bool found = true;
            for (size_t j = 1; j < strings.size(); j++) {
                if (strings[j].find(candidate) == string::npos) {
                    found = false;
                    break;
                }
            }
            if (found) {
                result = candidate;
                break;
            }
        }
    }

    cout << result << endl;

    return 0;
}
