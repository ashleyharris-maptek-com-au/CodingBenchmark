using System;
using System.Collections.Generic;
using System.Linq;

class BinPacking
{
    struct Box
    {
        public int Index;
        public int W, H, D;
    }

    struct Placement
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    static void Main()
    {
        var line = Console.ReadLine().Split();
        int N = int.Parse(line[0]);
        int W = int.Parse(line[1]);
        int H = int.Parse(line[2]);
        int D = int.Parse(line[3]);

        var boxes = new List<Box>();
        for (int i = 0; i < N; i++)
        {
            var dims = Console.ReadLine().Split();
            boxes.Add(new Box
            {
                Index = i,
                W = int.Parse(dims[0]),
                H = int.Parse(dims[1]),
                D = int.Parse(dims[2])
            });
        }

        var packed = PackBoxes(boxes, W, H, D);

        Console.WriteLine(packed.Count);
        foreach (var p in packed)
        {
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
        }
    }

    static List<Placement> PackBoxes(List<Box> boxes, int containerW, int containerH, int containerD)
    {
        var packed = new List<Placement>();
        var occupied = new List<Placement>();

        var sorted = boxes.OrderByDescending(b => (long)b.W * b.H * b.D).ToList();
        long containerVolume = (long)containerW * containerH * containerD;
        bool isSmallContainer = containerVolume <= 50000;

        foreach (var box in sorted)
        {
            var placement = isSmallContainer
                ? FindPlacementExhaustive(box, containerW, containerH, containerD, occupied)
                : FindPlacementOptimized(box, containerW, containerH, containerD, occupied);

            if (placement.HasValue)
            {
                packed.Add(placement.Value);
                occupied.Add(placement.Value);
            }
        }

        return packed;
    }

    static Placement? FindPlacementExhaustive(Box box, int cW, int cH, int cD, List<Placement> occupied)
    {
        var rotations = GetRotations(box);
        int maxZ = Math.Min(cD, 100);
        int maxY = Math.Min(cH, 100);
        int maxX = Math.Min(cW, 100);

        for (int z = 0; z < maxZ; z++)
        {
            for (int y = 0; y < maxY; y++)
            {
                for (int x = 0; x < maxX; x++)
                {
                    foreach (var rot in rotations)
                    {
                        if (x + rot.Item1 <= cW && y + rot.Item2 <= cH && z + rot.Item3 <= cD &&
                            !Overlaps(x, y, z, rot.Item1, rot.Item2, rot.Item3, occupied))
                        {
                            return new Placement
                            {
                                Index = box.Index,
                                X = x, Y = y, Z = z,
                                W = rot.Item1, H = rot.Item2, D = rot.Item3
                            };
                        }
                    }
                }
            }
        }

        return null;
    }

    static Placement? FindPlacementOptimized(Box box, int cW, int cH, int cD, List<Placement> occupied)
    {
        var rotations = GetRotations(box);
        var candidates = GenerateCandidatePositions(cW, cH, cD, occupied);

        foreach (var (x, y, z) in candidates)
        {
            foreach (var rot in rotations)
            {
                if (x + rot.Item1 <= cW && y + rot.Item2 <= cH && z + rot.Item3 <= cD &&
                    !Overlaps(x, y, z, rot.Item1, rot.Item2, rot.Item3, occupied))
                {
                    return new Placement
                    {
                        Index = box.Index,
                        X = x, Y = y, Z = z,
                        W = rot.Item1, H = rot.Item2, D = rot.Item3
                    };
                }
            }
        }

        return null;
    }

    static List<(int, int, int)> GenerateCandidatePositions(int cW, int cH, int cD, List<Placement> occupied)
    {
        var candidates = new HashSet<(int, int, int)>();
        candidates.Add((0, 0, 0));

        foreach (var p in occupied)
        {
            if (p.X + p.W < cW) candidates.Add((p.X + p.W, 0, 0));
            if (p.Y + p.H < cH) candidates.Add((0, p.Y + p.H, 0));
            if (p.Z + p.D < cD) candidates.Add((0, 0, p.Z + p.D));
            if (p.X + p.W < cW && p.Y + p.H < cH) candidates.Add((p.X + p.W, p.Y + p.H, 0));
            if (p.X + p.W < cW && p.Z + p.D < cD) candidates.Add((p.X + p.W, 0, p.Z + p.D));
            if (p.Y + p.H < cH && p.Z + p.D < cD) candidates.Add((0, p.Y + p.H, p.Z + p.D));
            if (p.X + p.W < cW && p.Y + p.H < cH && p.Z + p.D < cD)
                candidates.Add((p.X + p.W, p.Y + p.H, p.Z + p.D));
        }

        return candidates.OrderBy(c => c.Item1 * cH * cD + c.Item2 * cD + c.Item3).ToList();
    }

    static List<(int, int, int)> GetRotations(Box box)
    {
        var rotations = new HashSet<(int, int, int)>();
        var dims = new[] { box.W, box.H, box.D };

        for (int i = 0; i < 3; i++)
            for (int j = 0; j < 3; j++)
                for (int k = 0; k < 3; k++)
                    if (i != j && j != k && i != k)
                        rotations.Add((dims[i], dims[j], dims[k]));

        return rotations.ToList();
    }

    static bool Overlaps(int x, int y, int z, int w, int h, int d, List<Placement> occupied)
    {
        foreach (var p in occupied)
        {
            if (x < p.X + p.W && x + w > p.X &&
                y < p.Y + p.H && y + h > p.Y &&
                z < p.Z + p.D && z + d > p.Z)
            {
                return true;
            }
        }
        return false;
    }
}
