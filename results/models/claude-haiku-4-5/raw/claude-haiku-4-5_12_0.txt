using System;
using System.Collections.Generic;
using System.Linq;

class Vector3
{
    public double X, Y, Z;

    public Vector3(double x = 0, double y = 0, double z = 0)
    {
        X = x; Y = y; Z = z;
    }

    public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3 operator *(double s, Vector3 v) => new Vector3(s * v.X, s * v.Y, s * v.Z);
    public static Vector3 operator *(Vector3 v, double s) => s * v;
    public double Dot(Vector3 other) => X * other.X + Y * other.Y + Z * other.Z;
    public Vector3 Cross(Vector3 other) => new Vector3(Y * other.Z - Z * other.Y, Z * other.X - X * other.Z, X * other.Y - Y * other.X);
    public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
    public Vector3 Normalized() { var len = Length(); return len > 1e-9 ? (1.0 / len) * this : new Vector3(1, 0, 0); }
}

class Quaternion
{
    public double W, X, Y, Z;

    public Quaternion(double w = 1, double x = 0, double y = 0, double z = 0)
    {
        W = w; X = x; Y = y; Z = z;
    }

    public Vector3 Rotate(Vector3 v)
    {
        var q = this;
        var vq = new Quaternion(0, v.X, v.Y, v.Z);
        var qinv = new Quaternion(q.W, -q.X, -q.Y, -q.Z);
        var result = Multiply(Multiply(q, vq), qinv);
        return new Vector3(result.X, result.Y, result.Z);
    }

    public static Quaternion Multiply(Quaternion a, Quaternion b) =>
        new Quaternion(
            a.W * b.W - a.X * b.X - a.Y * b.Y - a.Z * b.Z,
            a.W * b.X + a.X * b.W + a.Y * b.Z - a.Z * b.Y,
            a.W * b.Y - a.X * b.Z + a.Y * b.W + a.Z * b.X,
            a.W * b.Z + a.X * b.Y - a.Y * b.X + a.Z * b.W
        );
}

class Placement
{
    public Vector3 Translation;
    public Quaternion Rotation;

    public Placement(Vector3 t, Quaternion r)
    {
        Translation = t;
        Rotation = r;
    }
}

class BinPacker
{
    private List<Vector3> vertices;
    private List<int[]> faces;
    private double containerW, containerH, containerD;
    private Vector3 polyhedronMin, polyhedronMax;
    private double polyhedronVolume;
    private List<Placement> placements = new();

    static void Main()
    {
        var packer = new BinPacker();
        packer.Run();
    }

    void Run()
    {
        ReadInput();
        ComputePolyhedronProperties();

        var containerVolume = containerW * containerH * containerD;
        var maxPossible = (int)(containerVolume / (polyhedronVolume * 0.7));

        DetermineSizeClass();
        PackPolyhedra();

        WriteOutput();
    }

    void ReadInput()
    {
        var line = Console.ReadLine().Split();
        int V = int.Parse(line[0]);
        int F = int.Parse(line[1]);

        vertices = new List<Vector3>();
        for (int i = 0; i < V; i++)
        {
            var parts = Console.ReadLine().Split();
            vertices.Add(new Vector3(double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2])));
        }

        faces = new List<int[]>();
        for (int i = 0; i < F; i++)
        {
            var parts = Console.ReadLine().Split();
            faces.Add(new[] { int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]) });
        }

        var containerLine = Console.ReadLine().Split();
        containerW = double.Parse(containerLine[0]);
        containerH = double.Parse(containerLine[1]);
        containerD = double.Parse(containerLine[2]);
    }

    void ComputePolyhedronProperties()
    {
        if (vertices.Count == 0) return;

        polyhedronMin = new Vector3(vertices[0].X, vertices[0].Y, vertices[0].Z);
        polyhedronMax = new Vector3(vertices[0].X, vertices[0].Y, vertices[0].Z);

        foreach (var v in vertices)
        {
            polyhedronMin.X = Math.Min(polyhedronMin.X, v.X);
            polyhedronMin.Y = Math.Min(polyhedronMin.Y, v.Y);
            polyhedronMin.Z = Math.Min(polyhedronMin.Z, v.Z);
            polyhedronMax.X = Math.Max(polyhedronMax.X, v.X);
            polyhedronMax.Y = Math.Max(polyhedronMax.Y, v.Y);
            polyhedronMax.Z = Math.Max(polyhedronMax.Z, v.Z);
        }

        var size = polyhedronMax - polyhedronMin;
        polyhedronVolume = Math.Abs(size.X * size.Y * size.Z);
        if (polyhedronVolume < 1e-9) polyhedronVolume = 1;
    }

    void DetermineSizeClass()
    {
    }

    void PackPolyhedra()
    {
        int attempts = 0;
        int maxAttempts = Math.Min(1000, (int)(containerW * containerH * containerD / polyhedronVolume) + 100);

        var random = new Random(42);
        var orientations = GenerateOrientations();

        while (attempts < maxAttempts)
        {
            var orientation = orientations[attempts % orientations.Count];

            Vector3 position = FindBestPosition(orientation, random);
            if (position != null && CanPlace(position, orientation))
            {
                placements.Add(new Placement(position, orientation));
            }

            attempts++;
        }
    }

    List<Quaternion> GenerateOrientations()
    {
        var orientations = new List<Quaternion>();
        orientations.Add(new Quaternion(1, 0, 0, 0));

        double c45 = Math.Cos(Math.PI / 4);
        double s45 = Math.Sin(Math.PI / 4);

        orientations.Add(new Quaternion(c45, 0, 0, s45));
        orientations.Add(new Quaternion(c45, 0, s45, 0));
        orientations.Add(new Quaternion(c45, s45, 0, 0));

        double c90 = Math.Cos(Math.PI / 2);
        double s90 = Math.Sin(Math.PI / 2);

        orientations.Add(new Quaternion(c90, 0, 0, s90));
        orientations.Add(new Quaternion(c90, 0, s90, 0));
        orientations.Add(new Quaternion(c90, s90, 0, 0));

        return orientations;
    }

    Vector3 FindBestPosition(Quaternion orientation, Random random)
    {
        var rotatedMin = orientation.Rotate(polyhedronMin);
        var rotatedMax = orientation.Rotate(polyhedronMax);

        double minX = Math.Min(rotatedMin.X, rotatedMax.X);
        double maxX = Math.Max(rotatedMin.X, rotatedMax.X);
        double minY = Math.Min(rotatedMin.Y, rotatedMax.Y);
        double maxY = Math.Max(rotatedMin.Y, rotatedMax.Y);
        double minZ = Math.Min(rotatedMin.Z, rotatedMax.Z);
        double maxZ = Math.Max(rotatedMin.Z, rotatedMax.Z);

        double sizeX = maxX - minX;
        double sizeY = maxY - minY;
        double sizeZ = maxZ - minZ;

        if (sizeX > containerW || sizeY > containerH || sizeZ > containerD)
            return null;

        for (int i = 0; i < 5; i++)
        {
            double tx = random.NextDouble() * (containerW - sizeX);
            double ty = random.NextDouble() * (containerH - sizeY);
            double tz = random.NextDouble() * (containerD - sizeZ);

            var pos = new Vector3(tx - minX, ty - minY, tz - minZ);
            if (CanPlace(pos, orientation))
                return pos;
        }

        return null;
    }

    bool CanPlace(Vector3 position, Quaternion rotation)
    {
        var rotatedVertices = vertices.Select(v => rotation.Rotate(v) + position).ToList();

        foreach (var v in rotatedVertices)
        {
            if (v.X < -1e-6 || v.X > containerW + 1e-6 ||
                v.Y < -1e-6 || v.Y > containerH + 1e-6 ||
                v.Z < -1e-6 || v.Z > containerD + 1e-6)
                return false;
        }

        foreach (var existing in placements)
        {
            if (CheckCollision(rotatedVertices, existing))
                return false;
        }

        return true;
    }

    bool CheckCollision(List<Vector3> newVerts, Placement existing)
    {
        var existingVerts = vertices.Select(v => existing.Rotation.Rotate(v) + existing.Translation).ToList();

        var newMin = GetBounds(newVerts).min;
        var newMax = GetBounds(newVerts).max;
        var existingMin = GetBounds(existingVerts).min;
        var existingMax = GetBounds(existingVerts).max;

        return !(newMax.X < existingMin.X || newMin.X > existingMax.X ||
                 newMax.Y < existingMin.Y || newMin.Y > existingMax.Y ||
                 newMax.Z < existingMin.Z || newMin.Z > existingMax.Z);
    }

    (Vector3 min, Vector3 max) GetBounds(List<Vector3> verts)
    {
        var min = new Vector3(verts[0].X, verts[0].Y, verts[0].Z);
        var max = new Vector3(verts[0].X, verts[0].Y, verts[0].Z);

        foreach (var v in verts)
        {
            min.X = Math.Min(min.X, v.X);
            min.Y = Math.Min(min.Y, v.Y);
            min.Z = Math.Min(min.Z, v.Z);
            max.X = Math.Max(max.X, v.X);
            max.Y = Math.Max(max.Y, v.Y);
            max.Z = Math.Max(max.Z, v.Z);
        }

        return (min, max);
    }

    void WriteOutput()
    {
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Translation.X:F6} {p.Translation.Y:F6} {p.Translation.Z:F6} {p.Rotation.W:F6} {p.Rotation.X:F6} {p.Rotation.Y:F6} {p.Rotation.Z:F6}");
        }
    }
}
