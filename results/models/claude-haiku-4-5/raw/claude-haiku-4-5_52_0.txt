import statistics

_state = {
    'vel_history': [],
    'steering_integral': 0.0,
    'last_steering': 0.0,
}

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Self-driving autopilot for multi-lane freeway.
    Handles sensor fusion, obstacle avoidance, lane keeping, and graceful degradation.
    """

    # === SENSOR FUSION ===
    velocity_ms = _fuse_velocity(state)
    yaw_rate_dps = _fuse_yaw_rate(state)
    lateral_accel = _fuse_lateral_accel(state)
    steering_angle_deg = _fuse_steering_angle(state)

    # === ENVIRONMENT ===
    mu = state.get('estimated_mu', 1.0)
    speed_limit_kmh = _fuse_speed_limit(state)

    # === PERCEPTION ===
    lane_offset_m, lane_confidence = _estimate_lane_offset(state)
    left_line_dist, right_line_dist = _estimate_line_distances(state)
    front_obstacle_dist = _detect_front_obstacle(state)

    # === TARGET STATE ===
    target_speed_kmh = state.get('target_speed_kmh', 90)
    target_speed_kmh = min(target_speed_kmh, speed_limit_kmh)

    # === DECISION LOGIC ===
    # Critical obstacle - emergency stop
    if front_obstacle_dist < 5.0:
        return {
            'steering': _emergency_steering(lane_offset_m, mu),
            'throttle': 0.0,
            'brake': 1.0,
        }

    # Warning distance - cautious braking
    if front_obstacle_dist < 15.0:
        brake = 0.5 if front_obstacle_dist > 10.0 else 0.7
        throttle = 0.0
        steering = _lane_steering(lane_offset_m, lane_confidence, yaw_rate_dps, mu, dt)
        return {
            'steering': _clamp(steering, -1.0, 1.0),
            'throttle': 0.0,
            'brake': brake,
        }

    # Normal operation - speed control + lane keeping
    throttle, brake = _speed_control(velocity_ms, target_speed_kmh, mu, dt)
    steering = _lane_steering(lane_offset_m, lane_confidence, yaw_rate_dps, mu, dt)

    # === TRACTION MANAGEMENT ===
    if mu < 0.3:  # Ice - very conservative
        throttle *= 0.2
        steering *= 0.3
    elif mu < 0.6:  # Low traction
        throttle *= 0.5
        steering *= 0.6

    return {
        'steering': _clamp(steering, -1.0, 1.0),
        'throttle': _clamp(throttle, 0.0, 1.0),
        'brake': _clamp(brake, 0.0, 1.0),
    }

def _fuse_velocity(state):
    """Fuse wheel speeds, GPS, and vehicle speed sensors."""
    candidates = []

    # Wheel speeds (need at least 2 valid)
    wheel_speeds = []
    for wheel in ['fl', 'fr', 'rl', 'rr']:
        key_speed = f'wheel_speed_{wheel}_kmh'
        key_valid = f'wheel_speed_{wheel}_valid'
        if state.get(key_valid, False):
            ws = state.get(key_speed, 0)
            if ws is not None:
                wheel_speeds.append(ws / 3.6)

    if len(wheel_speeds) >= 2:
        candidates.append(statistics.median(wheel_speeds))
    elif len(wheel_speeds) == 1:
        candidates.append(wheel_speeds[0])

    # GPS speeds
    gps_speeds = []
    for i in [1, 2]:
        if state.get(f'gps_{i}_valid', False):
            gps_speeds.append(state.get(f'gps_{i}_speed_kmh', 0) / 3.6)

    if len(gps_speeds) == 2:
        if abs(gps_speeds[0] - gps_speeds[1]) < 2.0:
            candidates.append(statistics.mean(gps_speeds))
        else:
            candidates.append(max(gps_speeds))
    elif len(gps_speeds) == 1:
        candidates.append(gps_speeds[0])

    # Vehicle speed sensors (always available)
    v1 = state.get('vehicle_speed_1_kmh', 0) / 3.6
    v2 = state.get('vehicle_speed_2_kmh', 0) / 3.6
    if abs(v1 - v2) < 3.0:
        candidates.append((v1 + v2) / 2.0)
    else:
        candidates.append(max(v1, v2))

    if candidates:
        velocity = statistics.median(candidates)
    else:
        velocity = 0.0

    # Low-pass filter for smoothing
    _state['vel_history'].append(velocity)
    if len(_state['vel_history']) > 5:
        _state['vel_history'].pop(0)

    return statistics.mean(_state['vel_history']) if _state['vel_history'] else 0.0

def _fuse_yaw_rate(state):
    """Fuse 3 redundant gyroscopes."""
    rates = []
    for i in [1, 2, 3]:
        if state.get(f'gyro_{i}_valid', False):
            rate = state.get(f'gyro_{i}_yaw_rate_dps', 0)
            if rate is not None:
                rates.append(rate)

    if len(rates) >= 2:
        return statistics.median(rates)
    elif len(rates) == 1:
        return rates[0]
    return 0.0

def _fuse_lateral_accel(state):
    """Fuse 3 redundant lateral accelerometers."""
    accels = []
    for i in [1, 2, 3]:
        if state.get(f'accel_{i}_valid', False):
            accel = state.get(f'accel_{i}_lateral_ms2', 0)
            if accel is not None:
                accels.append(accel)

    if len(accels) >= 2:
        return statistics.median(accels)
    elif len(accels) == 1:
        return accels[0]
    return 0.0

def _fuse_steering_angle(state):
    """Fuse 2 steering angle sensors."""
    angles = []
    for i in [1, 2]:
        if state.get(f'steering_angle_{i}_valid', False):
            angle = state.get(f'steering_angle_{i}_deg', 0)
            if angle is not None:
                angles.append(angle)

    if len(angles) == 2:
        if abs(angles[0] - angles[1]) < 10.0:
            return (angles[0] + angles[1]) / 2.0
        else:
            return angles[0]
    elif len(angles) == 1:
        return angles[0]
    return 0.0

def _fuse_speed_limit(state):
    """Get speed limit from vision systems (median of valid readings)."""
    limits = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_speed_limit_valid', False):
            limit = state.get(f'vision_{i}_speed_limit_kmh', 130)
            if limit and limit > 0:
                limits.append(limit)

    if limits:
        return min(limits)
    return 130

def _estimate_lane_offset(state):
    """Estimate lateral offset from lane center (fuse 3 vision systems)."""
    offsets = []
    valid_count = 0

    for i in [1, 2, 3]:
        left_valid = state.get(f'vision_{i}_left_line_valid', False)
        right_valid = state.get(f'vision_{i}_right_line_valid', False)

        if left_valid and right_valid:
            offset = state.get(f'vision_{i}_lane_offset_m', 0)
            if offset is not None:
                offsets.append(offset)
                valid_count += 1

    if offsets:
        return statistics.median(offsets), valid_count >= 2

    return 0.0, False

def _estimate_line_distances(state):
    """Get distances to left and right lane lines."""
    left_dists = []
    right_dists = []

    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_left_line_valid', False):
            dist = state.get(f'vision_{i}_left_line_dist_m', 2.0)
            if dist is not None and dist > 0:
                left_dists.append(dist)

        if state.get(f'vision_{i}_right_line_valid', False):
            dist = state.get(f'vision_{i}_right_line_dist_m', 2.0)
            if dist is not None and dist > 0:
                right_dists.append(dist)

    left_avg = statistics.mean(left_dists) if left_dists else 2.0
    right_avg = statistics.mean(right_dists) if right_dists else 2.0

    return left_avg, right_avg

def _detect_front_obstacle(state):
    """Detect closest obstacle ahead using vision and lidar."""
    min_dist = 200.0

    # Vision-based obstacle detection (all 3 systems)
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_obstacle_valid', False):
            if state.get(f'vision_{i}_obstacle_detected', False):
                dist = state.get(f'vision_{i}_obstacle_dist_m', 200.0)
                if dist is not None and dist > 0:
                    min_dist = min(min_dist, dist)

    # Lidar front beams (center cone: -30 to +30 degrees)
    lidar_beams = [
        'lidar_+005_deg',
        'lidar_+015_deg',
        'lidar_+030_deg',
        'lidar_-005_deg',
        'lidar_-015_deg',
        'lidar_-030_deg',
    ]

    for beam in lidar_beams:
        if state.get(f'{beam}_valid', False):
            dist = state.get(f'{beam}_range_m', 200.0)
            if dist is not None and 0.5 < dist < 150:
                min_dist = min(min_dist, dist)

    return min_dist

def _speed_control(velocity_ms, target_kmh, mu, dt):
    """Control throttle and brake to match target speed."""
    target_ms = target_kmh / 3.6
    error_ms = target_ms - velocity_ms

    throttle = 0.0
    brake = 0.0

    if error_ms > 0.5:
        throttle = min(1.0, error_ms / 3.0)
        if mu < 0.6:
            throttle *= 0.4
    elif error_ms < -0.5:
        brake_demand = min(1.0, abs(error_ms) / 2.0)
        if mu < 0.3:
            brake = brake_demand * 0.4
        elif mu < 0.6:
            brake = brake_demand * 0.7
        else:
            brake = brake_demand

    return throttle, brake

def _lane_steering(lane_offset, lane_conf, yaw_rate, mu, dt):
    """Control steering to keep centered in lane."""
    if not lane_conf:
        return 0.0

    # Proportional term
    kp = 0.25
    steering = -lane_offset * kp

    # Damping for yaw rate
    kd = 0.08
    steering -= yaw_rate * kd * dt / 10.0

    # Integral wind-up (slow)
    _state['steering_integral'] += lane_offset * dt * 0.02
    _state['steering_integral'] = _clamp(_state['steering_integral'], -0.3, 0.3)
    steering += _state['steering_integral'] * 0.1

    # Traction-based adjustment
    if mu < 0.3:
        steering *= 0.3
    elif mu < 0.6:
        steering *= 0.6

    _state['last_steering'] = steering
    return steering

def _emergency_steering(lane_offset, mu):
    """Minimal steering adjustment during emergency."""
    if mu < 0.4:
        return _clamp(-lane_offset * 0.05, -0.3, 0.3)
    else:
        return _clamp(-lane_offset * 0.1, -0.5, 0.5)

def _clamp(value, min_val, max_val):
    """Clamp value to [min_val, max_val]."""
    return max(min_val, min(max_val, value))
