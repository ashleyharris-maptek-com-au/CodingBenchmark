using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    private static List<int> maxClique = new List<int>();
    private static readonly object syncLock = new object();

    static void Main()
    {
        var cts = new CancellationTokenSource();
        cts.CancelAfter(TimeSpan.FromMinutes(4.5));

        try
        {
            var line = Console.ReadLine().Split();
            int n = int.Parse(line[0]);
            int m = int.Parse(line[1]);

            var adj = new List<HashSet<int>>();
            for (int i = 0; i < n; i++)
                adj.Add(new HashSet<int>());

            for (int i = 0; i < m; i++)
            {
                var edge = Console.ReadLine().Split();
                int u = int.Parse(edge[0]);
                int v = int.Parse(edge[1]);
                adj[u].Add(v);
                adj[v].Add(u);
            }

            if (n <= 300)
                ExactSearch(n, adj, cts.Token);
            else if (n <= 100000)
                HeuristicSearch(n, adj, cts.Token);
            else
                LargeGraphSearch(n, adj, cts.Token);

            Console.WriteLine(maxClique.Count);
            if (maxClique.Count > 0)
                Console.WriteLine(string.Join(" ", maxClique.OrderBy(x => x)));
        }
        catch (OperationCanceledException) { }
    }

    static void ExactSearch(int n, List<HashSet<int>> adj, CancellationToken ct)
    {
        var candidates = Enumerable.Range(0, n).ToList();
        BranchAndBound(adj, candidates, new List<int>(), ct);
    }

    static void BranchAndBound(List<HashSet<int>> adj, List<int> candidates, List<int> current, CancellationToken ct)
    {
        if (ct.IsCancellationRequested) return;
        if (candidates.Count + current.Count <= maxClique.Count) return;

        if (candidates.Count == 0)
        {
            if (current.Count > maxClique.Count)
                lock (syncLock) maxClique = new List<int>(current);
            return;
        }

        for (int i = 0; i < candidates.Count; i++)
        {
            int v = candidates[i];
            current.Add(v);
            var newCandidates = candidates.GetRange(i + 1, candidates.Count - i - 1)
                .Where(u => adj[v].Contains(u)).ToList();
            BranchAndBound(adj, newCandidates, current, ct);
            current.RemoveAt(current.Count - 1);
        }
    }

    static void HeuristicSearch(int n, List<HashSet<int>> adj, CancellationToken ct)
    {
        var tasks = new List<Task>();
        var starts = Enumerable.Range(0, n)
            .OrderByDescending(v => adj[v].Count)
            .Take(Math.Min(n, 50))
            .ToList();

        foreach (var start in starts)
        {
            if (ct.IsCancellationRequested) break;
            tasks.Add(Task.Run(() => TryGreedy(start, adj, ct), ct));
        }

        try { Task.WaitAll(tasks.ToArray(), ct); }
        catch (OperationCanceledException) { }
    }

    static void LargeGraphSearch(int n, List<HashSet<int>> adj, CancellationToken ct)
    {
        var tasks = new List<Task>();
        int numTasks = Environment.ProcessorCount;

        for (int t = 0; t < numTasks && t < n; t++)
        {
            int start = t * n / numTasks;
            tasks.Add(Task.Run(() =>
            {
                for (int i = start; i < Math.Min(start + n / numTasks, n); i++)
                {
                    if (ct.IsCancellationRequested) return;
                    TryGreedy(i, adj, ct);
                }
            }, ct));
        }

        try { Task.WaitAll(tasks.ToArray(), ct); }
        catch (OperationCanceledException) { }
    }

    static void TryGreedy(int start, List<HashSet<int>> adj, CancellationToken ct)
    {
        var clique = new List<int> { start };
        var cand = new HashSet<int>(adj[start]);

        while (cand.Count > 0 && !ct.IsCancellationRequested)
        {
            var best = cand.OrderByDescending(v => cand.Count(u => u != v && adj[v].Contains(u))).First();
            clique.Add(best);
            cand = new HashSet<int>(cand.Where(v => v != best && adj[best].Contains(v)));
        }

        if (clique.Count > maxClique.Count)
            lock (syncLock)
                if (clique.Count > maxClique.Count)
                    maxClique = clique;
    }
}
