import sys
import math

def distance(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def solve_cvrp(n, k, capacity, depot, customers):
    """Solve CVRP using nearest neighbor heuristic"""
    unvisited = set(range(n))
    routes = []

    while unvisited:
        route = []
        current_pos = depot
        current_load = 0

        # Build a route greedily
        while unvisited:
            best_customer = None
            best_distance = float('inf')

            # Find nearest unvisited customer that fits capacity
            for customer_idx in unvisited:
                x, y, demand = customers[customer_idx]
                if current_load + demand <= capacity:
                    d = distance(current_pos, (x, y))
                    if d < best_distance:
                        best_distance = d
                        best_customer = customer_idx

            if best_customer is None:
                break  # No more customers fit, close this route

            route.append(best_customer)
            x, y, demand = customers[best_customer]
            current_pos = (x, y)
            current_load += demand
            unvisited.remove(best_customer)

        if route:
            routes.append(route)

    # Calculate total distance
    total_distance = 0.0
    for route in routes:
        current_pos = depot
        for customer_idx in route:
            x, y, _ = customers[customer_idx]
            total_distance += distance(current_pos, (x, y))
            current_pos = (x, y)
        # Return to depot
        total_distance += distance(current_pos, depot)

    return routes, total_distance

def main():
    # Read input
    line = sys.stdin.readline().strip()
    n, k, capacity = map(int, line.split())

    line = sys.stdin.readline().strip()
    depot_x, depot_y = map(float, line.split())
    depot = (depot_x, depot_y)

    customers = []
    for _ in range(n):
        line = sys.stdin.readline().strip()
        x, y, demand = line.split()
        customers.append((float(x), float(y), int(demand)))

    # Solve
    if n == 0:
        print(0)
        print(0)
    else:
        routes, total_distance = solve_cvrp(n, k, capacity, depot, customers)

        # Output
        print(total_distance)
        print(len(routes))
        for route in routes:
            print(' '.join(map(str, route)))

if __name__ == '__main__':
    main()
