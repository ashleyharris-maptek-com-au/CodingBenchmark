#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out of bounds
    if (idx >= N) return;

    // Load current boid
    vec3 pos = inData[idx * 2].xyz;
    vec3 vel = inData[idx * 2 + 1].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    // Accumulate forces
    vec3 sepDir = vec3(0.0);
    vec3 alignDir = vec3(0.0);
    vec3 cohPos = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    // Iterate through all other boids
    for (uint j = 0; j < N; j++) {
        if (j == idx) continue;  // Skip self

        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;

        vec3 diff = otherPos - pos;
        float dist = length(diff);

        if (dist == 0.0) continue;  // Skip zero-distance neighbors

        // Separation: accumulate direction away from neighbor
        if (dist < sepRadius) {
            sepDir -= normalize(diff);
            sepCount++;
        }

        // Alignment: accumulate neighbor velocity
        if (dist < alignRadius) {
            alignDir += otherVel;
            alignCount++;
        }

        // Cohesion: accumulate neighbor position
        if (dist < cohRadius) {
            cohPos += otherPos;
            cohCount++;
        }
    }

    // Calculate acceleration from Reynolds' rules
    vec3 accel = vec3(0.0);

    // Separation: average accumulated directions and apply weight
    if (sepCount > 0) {
        accel += normalize(sepDir) * sepWeight;
    }

    // Alignment: steer towards average neighbor velocity
    if (alignCount > 0) {
        vec3 avgVel = alignDir / float(alignCount);
        accel += (avgVel - vel) * alignWeight;
    }

    // Cohesion: steer towards center of mass of neighbors
    if (cohCount > 0) {
        vec3 cohCenter = cohPos / float(cohCount);
        accel += (cohCenter - pos) * cohWeight;
    }

    // Boundary forces: repel when approaching boundary
    float boundThreshold = boundSize * 0.4;

    if (abs(pos.x) > boundThreshold) {
        accel.x -= sign(pos.x) * boundForce;
    }
    if (abs(pos.y) > boundThreshold) {
        accel.y -= sign(pos.y) * boundForce;
    }
    if (abs(pos.z) > boundThreshold) {
        accel.z -= sign(pos.z) * boundForce;
    }

    // Update velocity: v = v + a*dt
    vel += accel * dt;

    // Clamp speed to maximum
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    // Update position: p = p + v*dt
    pos += vel * dt;

    // Write output
    outData[idx * 2] = vec4(pos, 0.0);
    outData[idx * 2 + 1] = vec4(vel, 0.0);
}
