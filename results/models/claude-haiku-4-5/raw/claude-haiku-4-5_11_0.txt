use std::io::{self, BufRead};

#[derive(Clone, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }

    fn distance(&self, other: &Point) -> f64 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        (dx * dx + dy * dy).sqrt()
    }

    fn rotate(&self, angle: f64) -> Point {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Point {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Point {
        Point {
            x: self.x + dx,
            y: self.y + dy,
        }
    }
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Polygon { vertices }
    }

    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }

    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            min_x = min_x.min(v.x);
            max_x = max_x.max(v.x);
            min_y = min_y.min(v.y);
            max_y = max_y.max(v.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn rotate(&self, angle: f64) -> Polygon {
        let rotated: Vec<Point> = self.vertices.iter().map(|v| v.rotate(angle)).collect();
        Polygon::new(rotated)
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let translated: Vec<Point> = self
            .vertices
            .iter()
            .map(|v| v.translate(dx, dy))
            .collect();
        Polygon::new(translated)
    }

    fn centroid(&self) -> Point {
        let mut cx = 0.0;
        let mut cy = 0.0;
        for v in &self.vertices {
            cx += v.x;
            cy += v.y;
        }
        let n = self.vertices.len() as f64;
        Point { x: cx / n, y: cy / n }
    }
}

fn point_in_polygon(p: &Point, poly: &Polygon) -> bool {
    let mut inside = false;
    let n = poly.vertices.len();

    for i in 0..n {
        let j = (i + 1) % n;
        let xi = poly.vertices[i].x;
        let yi = poly.vertices[i].y;
        let xj = poly.vertices[j].x;
        let yj = poly.vertices[j].y;

        let intersect = ((yi > p.y) != (yj > p.y))
            && (p.x < (xj - xi) * (p.y - yi) / (yj - yi) + xi);
        if intersect {
            inside = !inside;
        }
    }

    inside
}

fn segments_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {
    let ccw = |a: &Point, b: &Point, c: &Point| -> i32 {
        ((c.y - a.y) * (b.x - a.x) > (b.y - a.y) * (c.x - a.x)) as i32
    };

    ccw(p1, p3, p4) != ccw(p2, p3, p4) && ccw(p1, p2, p3) != ccw(p1, p2, p4)
}

fn polygon_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    for v in &poly1.vertices {
        if point_in_polygon(v, poly2) {
            return true;
        }
    }
    for v in &poly2.vertices {
        if point_in_polygon(v, poly1) {
            return true;
        }
    }

    let n1 = poly1.vertices.len();
    let n2 = poly2.vertices.len();
    for i in 0..n1 {
        for j in 0..n2 {
            if segments_intersect(
                &poly1.vertices[i],
                &poly1.vertices[(i + 1) % n1],
                &poly2.vertices[j],
                &poly2.vertices[(j + 1) % n2],
            ) {
                return true;
            }
        }
    }

    false
}

fn piece_fits_in_stock(piece: &Polygon, stock: &Polygon) -> bool {
    for v in &piece.vertices {
        if !point_in_polygon(v, stock) {
            return false;
        }
    }
    true
}

#[derive(Clone)]
struct Placement {
    piece_idx: usize,
    stock_idx: usize,
    x: f64,
    y: f64,
    rotation: f64,
}

fn find_placement_position(
    stock: &Polygon,
    piece: &Polygon,
    existing_pieces: &[(Polygon, f64, f64, f64)],
    angle: f64,
) -> Option<(f64, f64)> {
    let rotated = piece.rotate(angle);
    let (min_x, min_y, max_x, max_y) = stock.bounding_box();
    let (p_min_x, p_min_y, p_max_x, p_max_y) = rotated.bounding_box();
    let width = p_max_x - p_min_x;
    let height = p_max_y - p_min_y;

    let stock_width = max_x - min_x;
    let stock_height = max_y - min_y;

    let step = (stock_width.max(stock_height) / 20.0).max(0.1);

    let mut x = min_x - p_min_x;
    while x <= max_x - p_min_x {
        let mut y = min_y - p_min_y;
        while y <= max_y - p_min_y {
            let translated = rotated.translate(x, y);

            if piece_fits_in_stock(&translated, stock) {
                let mut valid = true;
                for (existing, _, _, _) in existing_pieces {
                    if polygon_overlap(&translated, existing) {
                        valid = false;
                        break;
                    }
                }

                if valid {
                    return Some((x, y));
                }
            }

            y += step;
        }
        x += step;
    }

    None
}

fn try_place_piece(
    stock: &Polygon,
    piece: &Polygon,
    existing_pieces: &[(Polygon, f64, f64, f64)],
) -> Option<(f64, f64, f64)> {
    let angles = vec![0.0, 1.57, 3.14, 4.71];

    for angle in angles {
        if let Some((x, y)) = find_placement_position(stock, piece, existing_pieces, angle) {
            return Some((x, y, angle));
        }
    }

    None
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first.trim().split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.trim().split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        stock_vertices.push(Point::new(x, y));
    }
    let stock = Polygon::new(stock_vertices);

    let mut pieces = Vec::new();
    for _ in 0..np {
        let line = lines.next().unwrap().unwrap();
        let pv: usize = line.trim().parse().unwrap();
        let mut vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let parts: Vec<&str> = line.trim().split_whitespace().collect();
            let x: f64 = parts[0].parse().unwrap();
            let y: f64 = parts[1].parse().unwrap();
            vertices.push(Point::new(x, y));
        }
        pieces.push(Polygon::new(vertices));
    }

    let mut placements = Vec::new();
    let mut stocks = vec![Vec::new()];

    let mut piece_indices: Vec<usize> = (0..np).collect();
    piece_indices.sort_by(|&a, &b| pieces[b].area().partial_cmp(&pieces[a].area()).unwrap());

    for piece_idx in piece_indices {
        let piece = &pieces[piece_idx];
        let mut placed = false;

        for stock_idx in 0..stocks.len() {
            let existing = stocks[stock_idx].clone();
            if let Some((x, y, rotation)) = try_place_piece(&stock, piece, &existing) {
                let translated = piece.rotate(rotation).translate(x, y);
                stocks[stock_idx].push((translated, x, y, rotation));
                placements.push(Placement {
                    piece_idx,
                    stock_idx,
                    x,
                    y,
                    rotation,
                });
                placed = true;
                break;
            }
        }

        if !placed {
            if let Some((x, y, rotation)) = try_place_piece(&stock, piece, &[]) {
                let translated = piece.rotate(rotation).translate(x, y);
                stocks.push(vec![(translated, x, y, rotation)]);
                placements.push(Placement {
                    piece_idx,
                    stock_idx: stocks.len() - 1,
                    x,
                    y,
                    rotation,
                });
            }
        }
    }

    placements.sort_by_key(|p| p.piece_idx);

    println!("{}", stocks.len());
    for p in placements {
        println!("{} {} {} {}", p.stock_idx, p.x, p.y, p.rotation);
    }
}
