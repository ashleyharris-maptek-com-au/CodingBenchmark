#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Scale UVs: 8x16 pattern
    vec2 scaledUV = uv * vec2(8.0, 16.0);

    // Mortar dimensions
    float mortarThicknessX = 0.05;
    float mortarThicknessY = 0.1;

    // Get the row (integer part of Y)
    float row = floor(scaledUV.y);

    // Stagger offset: half-brick offset on odd rows
    float staggerOffset = mod(row, 2.0) * 0.5;

    // Apply stagger to X coordinate
    float adjustedX = scaledUV.x + staggerOffset;

    // Get fractional part for position within brick
    vec2 brickCoord = fract(vec2(adjustedX, scaledUV.y));

    // Check if we're in mortar (at brick edges)
    bool inMortarX = brickCoord.x < mortarThicknessX ||
                     brickCoord.x > (1.0 - mortarThicknessX);
    bool inMortarY = brickCoord.y < mortarThicknessY ||
                     brickCoord.y > (1.0 - mortarThicknessY);
    bool inMortar = inMortarX || inMortarY;

    // Colors
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 baseColor = inMortar ? mortarColor : brickColor;

    // Lambertian shading
    vec3 n = normalize(normal);
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(0.0, dot(n, lightDir));
    float shadingFactor = 0.4 + 0.6 * diffuse;

    // Final color with shading
    vec3 finalColor = baseColor * shadingFactor;

    outColor = vec4(finalColor, 1.0);
}
