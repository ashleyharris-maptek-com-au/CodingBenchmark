import math

# Module-level persistent state
_state_history = []
_fuel_used = 0.0
_prev_error = {"x": 0, "y": 0, "z": 0}
_phase = "approach"  # approach, close, dock


def autopilot_step(state: dict, dt: float) -> dict:
    """
    Spacecraft docking autopilot that handles orbital mechanics and sensor fusion.

    Key physics:
    - In orbit, thrusting toward target when behind SLOWS you down (raises orbit)
    - Must thrust RETROGRADE to catch up (lower orbit = faster movement)
    - Linear approximation valid <50m (CW equations)
    - Full orbital mechanics >200m
    """
    global _state_history, _fuel_used, _prev_error, _phase

    # Sensor fusion: cross-check redundant sensors
    rel_pos = _fuse_position(state)
    rel_vel = _fuse_velocity(state)
    attitude = _fuse_attitude(state)
    fuel_kg = state.get("fuel_remaining_kg", 500)

    # Early exit if out of fuel
    if fuel_kg < 1:
        return {
            "thrust_x": 0, "thrust_y": 0, "thrust_z": 0,
            "att_rate_roll": 0, "att_rate_pitch": 0, "att_rate_yaw": 0
        }

    range_m = math.sqrt(rel_pos["x"]**2 + rel_pos["y"]**2 + rel_pos["z"]**2)
    speed_ms = math.sqrt(rel_vel["x"]**2 + rel_vel["y"]**2 + rel_vel["z"]**2)

    # Docking success
    if range_m < 2.0 and speed_ms < 0.3:
        return {
            "thrust_x": 0, "thrust_y": 0, "thrust_z": 0,
            "att_rate_roll": 0, "att_rate_pitch": 0, "att_rate_yaw": 0
        }

    # Strategy selection by range
    if range_m > 200:
        thrust = _far_approach(rel_pos, rel_vel, state)
    elif range_m > 50:
        thrust = _medium_approach(rel_pos, rel_vel, state)
    elif range_m > 10:
        thrust = _close_approach(rel_pos, rel_vel, state)
    else:
        thrust = _final_dock(rel_pos, rel_vel, state)

    # Attitude control to align with velocity vector
    att_cmds = _attitude_control(attitude, rel_vel)

    result = {
        "thrust_x": _clamp(thrust["x"], -1.0, 1.0),
        "thrust_y": _clamp(thrust["y"], -1.0, 1.0),
        "thrust_z": _clamp(thrust["z"], -1.0, 1.0),
        "att_rate_roll": _clamp(att_cmds["roll"], -0.035, 0.035),
        "att_rate_pitch": _clamp(att_cmds["pitch"], -0.035, 0.035),
        "att_rate_yaw": _clamp(att_cmds["yaw"], -0.035, 0.035),
    }

    _state_history.append({
        "range": range_m,
        "speed": speed_ms,
        "thrust": thrust,
        "fuel": fuel_kg
    })

    return result


def _fuse_position(state: dict) -> dict:
    """Cross-check radar and lidar position measurements."""
    sources = []

    # Radar position
    if all(state.get(f"radar_rel_{ax}_valid") for ax in ["x", "y", "z"]):
        sources.append({
            "x": state.get("radar_rel_pos_x_m", 0),
            "y": state.get("radar_rel_pos_y_m", 0),
            "z": state.get("radar_rel_pos_z_m", 0),
            "weight": 0.6
        })

    # Lidar position
    if all(state.get(f"lidar_rel_{ax}_valid") for ax in ["x", "y", "z"]):
        sources.append({
            "x": state.get("lidar_rel_pos_x_m", 0),
            "y": state.get("lidar_rel_pos_y_m", 0),
            "z": state.get("lidar_rel_pos_z_m", 0),
            "weight": 0.4
        })

    # Docking camera bearing (within 500m, normalize to position estimate)
    if state.get("dock_camera_active") and state.get("dock_camera_range_m", float('inf')) < 500:
        cam_range = state.get("dock_camera_range_m", 0)
        if all(state.get(f"dock_camera_bearing_{ax}_valid") for ax in ["x", "y", "z"]):
            bx = state.get("dock_camera_bearing_x", 0)
            by = state.get("dock_camera_bearing_y", 0)
            bz = state.get("dock_camera_bearing_z", 0)
            norm = math.sqrt(bx**2 + by**2 + bz**2) or 1.0
            sources.append({
                "x": (bx / norm) * cam_range,
                "y": (by / norm) * cam_range,
                "z": (bz / norm) * cam_range,
                "weight": 0.4
            })

    if not sources:
        return {"x": 0, "y": 0, "z": 0}

    total_weight = sum(s["weight"] for s in sources)
    result = {"x": 0, "y": 0, "z": 0}
    for src in sources:
        w = src["weight"] / total_weight
        result["x"] += src["x"] * w
        result["y"] += src["y"] * w
        result["z"] += src["z"] * w

    return result

def _fuse_attitude(state: dict) -> dict:
    """Cross-check star trackers and gyros for attitude."""
    attitudes = []

    # Star tracker 1
    if all(state.get(f"star_tracker_1_{ax}_valid") for ax in ["pitch_deg", "roll_deg", "yaw_deg"]):
        attitudes.append({
            "pitch": math.radians(state.get("star_tracker_1_pitch_deg", 0)),
            "roll": math.radians(state.get("star_tracker_1_roll_deg", 0)),
            "yaw": math.radians(state.get("star_tracker_1_yaw_deg", 0)),
            "weight": 0.5
        })

    # Star tracker 2
    if all(state.get(f"star_tracker_2_{ax}_valid") for ax in ["pitch_deg", "roll_deg", "yaw_deg"]):
        attitudes.append({
            "pitch": math.radians(state.get("star_tracker_2_pitch_deg", 0)),
            "roll": math.radians(state.get("star_tracker_2_roll_deg", 0)),
            "yaw": math.radians(state.get("star_tracker_2_yaw_deg", 0)),
            "weight": 0.5
        })

    if not attitudes:
        return {"pitch": 0, "roll": 0, "yaw": 0}

    total_weight = sum(a["weight"] for a in attitudes)
    result = {"pitch": 0, "roll": 0, "yaw": 0}
    for att in attitudes:
        w = att["weight"] / total_weight
        result["pitch"] += att["pitch"] * w
        result["roll"] += att["roll"] * w
        result["yaw"] += att["yaw"] * w

    return result


def _far_approach(rel_pos: dict, rel_vel: dict, state: dict) -> dict:
    """
    Far approach (>200m): Hohmann-like transfer.
    If behind (y negative), use retrograde thrust to lower orbit and catch up.
    """
    y = rel_pos["y"]
    vy = rel_vel["y"]
    x = rel_pos["x"]
    vx = rel_vel["x"]

    thrust = {"x": 0, "y": 0, "z": 0}

    # If behind and moving away, need to lower orbit
    if y < -50 and vy > -0.2:
        thrust["y"] = -0.3  # Retrograde to lower orbit
    # If below and moving away radially
    elif x < -50 and vx < 0.2:
        thrust["x"] = 0.1  # Small radial correction

    # Cross-track: gentle correction
    z = rel_pos["z"]
    vz = rel_vel["z"]
    if abs(z) > 50:
        thrust["z"] = -0.2 * _sign(z)

    return thrust


def _medium_approach(rel_pos: dict, rel_vel: dict, state: dict) -> dict:
    """
    Medium range (50-200m): V-bar approach with gentle corrections.
    Target: close velocity to near zero, then gentle range closure.
    """
    y = rel_pos["y"]
    vy = rel_vel["y"]
    x = rel_pos["x"]
    vx = rel_vel["x"]
    z = rel_pos["z"]
    vz = rel_vel["z"]

    thrust = {"x": 0, "y": 0, "z": 0}

    # Along-track: close velocity first, then range
    if vy > 0.1:  # Approaching too fast
        thrust["y"] = -0.4
    elif y < -30:  # Still behind, gently close
        thrust["y"] = -0.15
    elif y > 0:  # Overshot, pull back
        thrust["y"] = 0.1

    # Radial: gentle centering
    if x > 30:
        thrust["x"] = -0.1
    elif x < -30:
        thrust["x"] = 0.1

    # Cross-track: corrections
    if abs(z) > 30:
        thrust["z"] = -0.2 * _sign(z)
    elif abs(vz) > 0.1:
        thrust["z"] = -0.1 * _sign(vz)

    return thrust


def _close_approach(rel_pos: dict, rel_vel: dict, state: dict) -> dict:
    """
    Close range (10-50m): Linear regime, direct thrust corrections.
    Very gentle approach speed, <0.1 m/s target.
    """
    y = rel_pos["y"]
    vy = rel_vel["y"]
    x = rel_pos["x"]
    vx = rel_vel["x"]
    z = rel_pos["z"]
    vz = rel_vel["z"]

    thrust = {"x": 0, "y": 0, "z": 0}

    # Linear damping: proportional to velocity
    speed = math.sqrt(vx**2 + vy**2 + vz**2)

    if speed > 0.2:
        thrust["y"] = -0.2 * _sign(vy) if abs(vy) > 0.05 else 0
        thrust["x"] = -0.1 * _sign(vx) if abs(vx) > 0.05 else 0
        thrust["z"] = -0.1 * _sign(vz) if abs(vz) > 0.05 else 0
    else:
        # Fine position corrections
        if y < -2:
            thrust["y"] = 0.05
        elif y > 2:
            thrust["y"] = -0.05

        if abs(x) > 2:
            thrust["x"] = -0.05 * _sign(x)

        if abs(z) > 2:
            thrust["z"] = -0.05 * _sign(z)

    return thrust


def _final_dock(rel_pos: dict, rel_vel: dict, state: dict) -> dict:
    """
    Final dock (<10m): Extremely gentle, <0.05 m/s approach.
    """
    y = rel_pos["y"]
    vy = rel_vel["y"]
    x = rel_pos["x"]
    vx = rel_vel["x"]
    z = rel_pos["z"]
    vz = rel_vel["z"]

    thrust = {"x": 0, "y": 0, "z": 0}

    # Very light damping
    if vy > 0.05:
        thrust["y"] = -0.08
    elif vy < -0.05:
        thrust["y"] = 0.08

    if abs(vx) > 0.05:
        thrust["x"] = -0.05 * _sign(vx)

    if abs(vz) > 0.05:
        thrust["z"] = -0.05 * _sign(vz)

    # Position holding
    if y < -1:
        thrust["y"] = 0.02
    elif y > 1:
        thrust["y"] = -0.02

    if abs(x) > 1:
        thrust["x"] = -0.02 * _sign(x)

    if abs(z) > 1:
        thrust["z"] = -0.02 * _sign(z)

    return thrust


def _attitude_control(attitude: dict, rel_vel: dict) -> dict:
    """
    Attitude control: align velocity vector with spacecraft x-axis.
    """
    vx = rel_vel["x"]
    vy = rel_vel["y"]
    vz = rel_vel["z"]

    vel_norm = math.sqrt(vx**2 + vy**2 + vz**2) or 1.0

    # Desired pitch: velocity in y-z plane
    vy_norm = vy / vel_norm
    vz_norm = vz / vel_norm

    target_pitch = math.atan2(vz_norm, vy_norm)
    target_yaw = math.atan2(vx / vel_norm, vy_norm)
    target_roll = 0  # No rolling needed

    # PD control
    kp, kd = 0.1, 0.05

    pitch_error = target_pitch - attitude["pitch"]
    yaw_error = target_yaw - attitude["yaw"]
    roll_error = target_roll - attitude["roll"]

    # Clamp errors to [-pi, pi]
    pitch_error = _clamp_angle(pitch_error)
    yaw_error = _clamp_angle(yaw_error)
    roll_error = _clamp_angle(roll_error)

    return {
        "pitch": kp * pitch_error,
        "yaw": kp * yaw_error,
        "roll": kp * roll_error,
    }


def _clamp(val: float, min_val: float, max_val: float) -> float:
    return max(min_val, min(max_val, val))


def _clamp_angle(angle: float) -> float:
    while angle > math.pi:
        angle -= 2 * math.pi
    while angle < -math.pi:
        angle += 2 * math.pi
    return angle


def _sign(val: float) -> float:
    return 1.0 if val > 0 else (-1.0 if val < 0 else 0.0)
