#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <cstring>

using namespace std;

struct City {
    float x, y;
    int idx;
};

inline float dist_sq(const City& a, const City& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// Simple nearest neighbor for small instances
vector<int> solve_small(const vector<City>& cities) {
    int n = cities.size();
    vector<int> tour;
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        float best_dist = 1e30f;
        int best_idx = -1;
        int current = tour.back();

        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                float d = dist_sq(cities[current], cities[j]);
                if (d < best_dist) {
                    best_dist = d;
                    best_idx = j;
                }
            }
        }

        tour.push_back(best_idx);
        visited[best_idx] = true;
    }

    return tour;
}

// 2-opt local improvement
void improve_2opt(vector<int>& tour, const vector<City>& cities, int max_iterations) {
    int n = tour.size();

    for (int iter = 0; iter < max_iterations; iter++) {
        bool improved = false;

        for (int i = 0; i < n - 2; i++) {
            for (int j = i + 2; j < n; j++) {
                float old_cost = dist_sq(cities[tour[i]], cities[tour[i+1]]) +
                                dist_sq(cities[tour[j]], cities[tour[(j+1)%n]]);
                float new_cost = dist_sq(cities[tour[i]], cities[tour[j]]) +
                                dist_sq(cities[tour[i+1]], cities[tour[(j+1)%n]]);

                if (new_cost < old_cost - 1e-9f) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }

        if (!improved) break;
    }
}

// Grid-based nearest neighbor for large instances
vector<int> solve_large(const vector<City>& cities) {
    int n = cities.size();

    // Find bounds
    float min_x = cities[0].x, max_x = cities[0].x;
    float min_y = cities[0].y, max_y = cities[0].y;

    for (const auto& c : cities) {
        min_x = min(min_x, c.x);
        max_x = max(max_x, c.x);
        min_y = min(min_y, c.y);
        max_y = max(max_y, c.y);
    }

    float range_x = max_x - min_x + 1e-6f;
    float range_y = max_y - min_y + 1e-6f;

    // Create spatial grid
    int grid_side = max(1, (int)sqrt(n / 5) + 1);
    grid_side = min(grid_side, 5000);  // Cap grid size to avoid memory issues

    float cell_x = range_x / grid_side;
    float cell_y = range_y / grid_side;

    vector<vector<vector<int>>> grid(grid_side, vector<vector<int>>(grid_side));

    for (int i = 0; i < n; i++) {
        int gx = min(grid_side - 1, (int)((cities[i].x - min_x) / cell_x));
        int gy = min(grid_side - 1, (int)((cities[i].y - min_y) / cell_y));
        grid[gx][gy].push_back(i);
    }

    // Nearest neighbor with spatial acceleration
    vector<int> tour;
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;

    for (int step = 1; step < n; step++) {
        int current = tour.back();
        int cx = min(grid_side - 1, (int)((cities[current].x - min_x) / cell_x));
        int cy = min(grid_side - 1, (int)((cities[current].y - min_y) / cell_y));

        float best_dist = 1e30f;
        int best_next = -1;

        // Search in expanding neighborhoods
        for (int radius = 0; radius < grid_side && best_next == -1; radius++) {
            for (int dx = -radius; dx <= radius; dx++) {
                for (int dy = -radius; dy <= radius; dy++) {
                    if (radius > 0 && abs(dx) != radius && abs(dy) != radius) continue;

                    int gx = cx + dx;
                    int gy = cy + dy;

                    if (gx >= 0 && gx < grid_side && gy >= 0 && gy < grid_side) {
                        for (int idx : grid[gx][gy]) {
                            if (!visited[idx]) {
                                float d = dist_sq(cities[current], cities[idx]);
                                if (d < best_dist) {
                                    best_dist = d;
                                    best_next = idx;
                                }
                            }
                        }
                    }
                }
            }
        }

        tour.push_back(best_next);
        visited[best_next] = true;
    }

    return tour;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<City> cities(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].idx = i;
    }

    vector<int> tour;

    if (n <= 5000) {
        // Small instances: use simple nearest neighbor
        tour = solve_small(cities);

        // Apply 2-opt improvement for smaller instances
        if (n <= 1000) {
            improve_2opt(tour, cities, 5);
        }
    } else {
        // Large instances: use grid-based spatial acceleration
        tour = solve_large(cities);
    }

    // Output tour
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << tour[i];
    }
    cout << "\n";

    return 0;
}
