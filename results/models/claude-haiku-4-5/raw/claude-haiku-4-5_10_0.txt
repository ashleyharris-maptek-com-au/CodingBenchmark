#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Item {
    int x, y, w, h;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, bw, bh;
    cin >> n >> bw >> bh;

    vector<pair<int, int>> rects(n);
    vector<pair<long long, int>> order;

    for (int i = 0; i < n; i++) {
        cin >> rects[i].first >> rects[i].second;
        order.push_back({(long long)rects[i].first * rects[i].second, i});
    }

    sort(order.rbegin(), order.rend());

    vector<int> bid(n), rot(n), px(n), py(n);
    vector<vector<Item>> boards;

    for (auto p : order) {
        int idx = p.second;
        int w = rects[idx].first;
        int h = rects[idx].second;
        bool placed = false;

        for (int b = 0; b < (int)boards.size() && !placed; b++) {
            vector<int> y_levels;
            y_levels.push_back(0);
            for (const auto& item : boards[b]) {
                y_levels.push_back(item.y + item.h);
            }
            sort(y_levels.begin(), y_levels.end());
            y_levels.erase(unique(y_levels.begin(), y_levels.end()), y_levels.end());

            for (int try_y : y_levels) {
                if (try_y + h > bh) break;

                vector<pair<int, int>> x_occ;
                for (const auto& item : boards[b]) {
                    if (!(try_y + h <= item.y || try_y >= item.y + item.h)) {
                        x_occ.push_back({item.x, item.x + item.w});
                    }
                }
                sort(x_occ.begin(), x_occ.end());

                int try_x = 0;
                for (auto xp : x_occ) {
                    if (try_x + w <= xp.first) {
                        boards[b].push_back({try_x, try_y, w, h});
                        bid[idx] = b;
                        rot[idx] = 0;
                        px[idx] = try_x;
                        py[idx] = try_y;
                        placed = true;
                        break;
                    }
                    try_x = max(try_x, xp.second);
                }

                if (placed) break;

                if (try_x + w <= bw) {
                    boards[b].push_back({try_x, try_y, w, h});
                    bid[idx] = b;
                    rot[idx] = 0;
                    px[idx] = try_x;
                    py[idx] = try_y;
                    placed = true;
                    break;
                }
            }

            if (!placed && w != h && h <= bw && w <= bh) {
                for (int try_y : y_levels) {
                    if (try_y + w > bh) break;

                    vector<pair<int, int>> x_occ;
                    for (const auto& item : boards[b]) {
                        if (!(try_y + w <= item.y || try_y >= item.y + item.h)) {
                            x_occ.push_back({item.x, item.x + item.w});
                        }
                    }
                    sort(x_occ.begin(), x_occ.end());

                    int try_x = 0;
                    for (auto xp : x_occ) {
                        if (try_x + h <= xp.first) {
                            boards[b].push_back({try_x, try_y, h, w});
                            bid[idx] = b;
                            rot[idx] = 1;
                            px[idx] = try_x;
                            py[idx] = try_y;
                            placed = true;
                            break;
                        }
                        try_x = max(try_x, xp.second);
                    }

                    if (placed) break;

                    if (try_x + h <= bw) {
                        boards[b].push_back({try_x, try_y, h, w});
                        bid[idx] = b;
                        rot[idx] = 1;
                        px[idx] = try_x;
                        py[idx] = try_y;
                        placed = true;
                        break;
                    }
                }
            }
        }

        if (!placed) {
            vector<Item> board;
            if (w <= bw && h <= bh) {
                board.push_back({0, 0, w, h});
                boards.push_back(board);
                bid[idx] = boards.size() - 1;
                rot[idx] = 0;
                px[idx] = 0;
                py[idx] = 0;
            } else if (h <= bw && w <= bh) {
                board.push_back({0, 0, h, w});
                boards.push_back(board);
                bid[idx] = boards.size() - 1;
                rot[idx] = 1;
                px[idx] = 0;
                py[idx] = 0;
            }
        }
    }

    cout << boards.size() << "\n";
    for (int i = 0; i < n; i++) {
        cout << bid[i] << " " << px[i] << " " << py[i] << " " << rot[i] << "\n";
    }

    return 0;
}
