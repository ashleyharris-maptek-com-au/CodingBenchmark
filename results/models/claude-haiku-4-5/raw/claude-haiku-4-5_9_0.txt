using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);

        var cuts = Console.ReadLine().Split().Select(int.Parse).ToArray();

        // Create array with cut indices and their lengths
        var cutIndices = new List<(int index, int length)>();
        for (int i = 0; i < n; i++)
        {
            cutIndices.Add((i, cuts[i]));
        }

        // Sort by length in decreasing order (First Fit Decreasing)
        cutIndices.Sort((a, b) => b.length.CompareTo(a.length));

        // Bin packing using First Fit Decreasing
        var stocks = new List<List<int>>();
        var stockRemaining = new List<int>();

        foreach (var (originalIndex, length) in cutIndices)
        {
            // Try to fit in first stock that has enough space
            bool placed = false;
            for (int i = 0; i < stocks.Count; i++)
            {
                if (stockRemaining[i] >= length)
                {
                    stocks[i].Add(originalIndex);
                    stockRemaining[i] -= length;
                    placed = true;
                    break;
                }
            }

            // If not placed, create new stock
            if (!placed)
            {
                stocks.Add(new List<int> { originalIndex });
                stockRemaining.Add(stockLength - length);
            }
        }

        // Sort cuts within each stock for consistent output
        foreach (var stock in stocks)
        {
            stock.Sort();
        }

        // Output
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
