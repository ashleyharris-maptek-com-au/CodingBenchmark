use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line: num_vertices and num_edges
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_vertices: usize = parts[0].parse().unwrap();
    let num_edges: usize = parts[1].parse().unwrap();

    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); num_vertices];

    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u: usize = parts[0].parse().unwrap();
        let v: usize = parts[1].parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    // Greedy algorithm: iteratively add the vertex that covers the most uncovered vertices
    let mut dominating_set: HashSet<usize> = HashSet::new();
    let mut covered: HashSet<usize> = HashSet::new();

    while covered.len() < num_vertices {
        // Find the vertex that covers the most uncovered vertices
        let mut best_vertex = 0;
        let mut best_count = 0;

        for v in 0..num_vertices {
            let mut count = 0;
            // Count this vertex if it's not covered
            if !covered.contains(&v) {
                count += 1;
            }
            // Count neighbors that are not covered
            for &u in &adj[v] {
                if !covered.contains(&u) {
                    count += 1;
                }
            }

            if count > best_count {
                best_count = count;
                best_vertex = v;
            }
        }

        // Add best_vertex to dominating set
        dominating_set.insert(best_vertex);
        covered.insert(best_vertex);
        for &u in &adj[best_vertex] {
            covered.insert(u);
        }
    }

    // Local search: remove redundant vertices
    let mut to_remove = Vec::new();
    for &v in &dominating_set {
        // Check if v is redundant
        let mut covered_without_v: HashSet<usize> = HashSet::new();
        for &u in &dominating_set {
            if u != v {
                covered_without_v.insert(u);
                for &w in &adj[u] {
                    covered_without_v.insert(w);
                }
            }
        }

        if covered_without_v.len() == num_vertices {
            to_remove.push(v);
        }
    }

    for v in to_remove {
        dominating_set.remove(&v);
    }

    // Output
    let mut result: Vec<usize> = dominating_set.into_iter().collect();
    result.sort();

    println!("{}", result.len());
    for (i, &v) in result.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
