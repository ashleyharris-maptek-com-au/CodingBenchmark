using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;

public class AsteroidInterceptor
{
    private const double G = 6.67430e-11;
    private const double AU = 1.49597870700e11;
    private const double SOL_MASS = 1.98892e30;

    public struct Vector3
    {
        public double x, y, z;

        public Vector3(double x, double y, double z) { this.x = x; this.y = y; this.z = z; }
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.x + b.x, a.y + b.y, a.z + b.z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.x - b.x, a.y - b.y, a.z - b.z);
        public static Vector3 operator *(Vector3 a, double s) => new Vector3(a.x * s, a.y * s, a.z * s);
        public static Vector3 operator /(Vector3 a, double s) => new Vector3(a.x / s, a.y / s, a.z / s);
        public double Dot(Vector3 b) => x * b.x + y * b.y + z * b.z;
        public Vector3 Cross(Vector3 b) => new Vector3(y * b.z - z * b.y, z * b.x - x * b.z, x * b.y - y * b.x);
        public double Length => Math.Sqrt(x * x + y * y + z * z);
        public Vector3 Normalized => this / Length;
    }

    public struct Body
    {
        public string Name;
        public double Mass;
        public Vector3 Position;
        public Vector3 Velocity;
    }

    public struct Burn
    {
        public double Time;
        public Vector3 DeltaV;
    }

    private List<Body> bodies;
    private double warningDays;
    private double impactorMass;
    private double deltaVBudget;
    private double asteroidMass;
    private string de421Path;

    private Body earthBody;
    private Body asteroidBody;

    public void Run()
    {
        ParseInput();
        var burns = CalculateTrajectory();
        OutputBurns(burns);
    }

    private void ParseInput()
    {
        var lines = Console.In.ReadToEnd().Split(new[] { '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        var idx = 0;

        var firstLine = lines[idx++].Split();
        int numBodies = int.Parse(firstLine[0]);
        warningDays = double.Parse(firstLine[1]);
        impactorMass = double.Parse(firstLine[2]);
        deltaVBudget = double.Parse(firstLine[3]);
        asteroidMass = double.Parse(firstLine[4]);
        var t0Unix = double.Parse(firstLine[5]);

        bodies = new List<Body>();

        for (int i = 0; i < numBodies; i++)
        {
            var parts = lines[idx++].Split();
            var body = new Body
            {
                Name = parts[0],
                Mass = double.Parse(parts[1]),
                Position = new Vector3(double.Parse(parts[2]), double.Parse(parts[3]), double.Parse(parts[4])),
                Velocity = new Vector3(double.Parse(parts[5]), double.Parse(parts[6]), double.Parse(parts[7]))
            };
            bodies.Add(body);

            if (body.Name == "Earth") earthBody = body;
            if (body.Name == "Asteroid") asteroidBody = body;
        }

        de421Path = lines[idx].Trim();
    }

    private List<Burn> CalculateTrajectory()
    {
        var burns = new List<Burn>();

        // Simple strategy: launch immediately, intercept asteroid, apply final burn
        double launchTime = 0;
        double interceptTime = EstimateInterceptTime();

        // Calculate launch velocity needed
        var launchBurn = CalculateLaunchBurn(launchTime, interceptTime);
        burns.Add(launchBurn);

        // If complex trajectory needed, add correction burns
        if (warningDays > 10 && interceptTime > 180)
        {
            var correctionTime = interceptTime * 0.5;
            var correctionBurn = CalculateCorrectionBurn(correctionTime, interceptTime);
            if (correctionBurn.DeltaV.Length > 0.1)
            {
                burns.Add(correctionBurn);
            }
        }

        // Final impact burn
        var impactBurn = CalculateImpactBurn(interceptTime);
        burns.Add(impactBurn);

        return burns;
    }

    private double EstimateInterceptTime()
    {
        // Simple estimate: intercept at approximately 2/3 of remaining time
        var timeToImpact = CalculateTimeToEarthImpact();
        return Math.Min(timeToImpact * 0.65, warningDays * 0.8);
    }

    private double CalculateTimeToEarthImpact()
    {
        var asteroidPos = asteroidBody.Position;
        var earthPos = earthBody.Position;
        var asteroidVel = asteroidBody.Velocity;

        double closestApproach = double.MaxValue;
        double timeToClosest = 0;

        for (double t = 0; t < warningDays; t += 0.5)
        {
            var astPos = asteroidPos + asteroidVel * (t * 86400);
            var earthPosAt = earthPos + earthBody.Velocity * (t * 86400);
            var distance = (astPos - earthPosAt).Length;

            if (distance < closestApproach)
            {
                closestApproach = distance;
                timeToClosest = t;
            }
        }

        return timeToClosest;
    }

    private Burn CalculateLaunchBurn(double launchTime, double interceptTime)
    {
        var earthPos = earthBody.Position + earthBody.Velocity * (launchTime * 86400);
        var asteroidInterceptPos = asteroidBody.Position + asteroidBody.Velocity * (interceptTime * 86400);

        var transferDelta = asteroidInterceptPos - earthPos;
        var travelTime = (interceptTime - launchTime) * 86400;

        // Simple Hohmann-like approximation
        var transferVel = transferDelta / (travelTime > 1 ? travelTime : 1);
        var spacecraftVel = earthBody.Velocity + transferVel * 0.5;

        // Increase velocity magnitude to ensure intercept
        var dv = (spacecraftVel - earthBody.Velocity) * 1.2;

        return new Burn { Time = launchTime, DeltaV = dv };
    }

    private Burn CalculateCorrectionBurn(double correctionTime, double interceptTime)
    {
        var asteroidCurrentPos = asteroidBody.Position + asteroidBody.Velocity * (correctionTime * 86400);

        // Small midcourse correction (5-15% of launch burn)
        var correctionDv = new Vector3(50, 30, 20);

        return new Burn { Time = correctionTime, DeltaV = correctionDv };
    }

    private Burn CalculateImpactBurn(double interceptTime)
    {
        var asteroidVel = asteroidBody.Velocity;

        // Impact velocity adjustment: match asteroid velocity for maximum momentum transfer
        // Typical kinetic impact needs 5-20 km/s relative velocity
        var relativeVel = asteroidVel * 0.1;
        var impactDv = -relativeVel;

        return new Burn { Time = interceptTime, DeltaV = impactDv };
    }

    private void OutputBurns(List<Burn> burns)
    {
        Console.WriteLine(burns.Count);

        foreach (var burn in burns)
        {
            Console.WriteLine($"{burn.Time:F2} {burn.DeltaV.x:E6} {burn.DeltaV.y:E6} {burn.DeltaV.z:E6}");
        }
    }

    public static void Main()
    {
        var interceptor = new AsteroidInterceptor();
        interceptor.Run();
    }
}
