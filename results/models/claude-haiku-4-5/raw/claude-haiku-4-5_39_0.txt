use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let edge_line = lines.next().unwrap().unwrap();
        let edge_parts: Vec<usize> = edge_line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let u = edge_parts[0];
        let v = edge_parts[1];
        adj[u].push(v);
        adj[v].push(u);
    }

    let (cut, partition) = find_min_bisection(n, &adj);

    println!("{}", cut);
    println!("{}", partition.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" "));
}

fn calculate_cut(adj: &[Vec<usize>], group_a: &HashSet<usize>) -> usize {
    let mut cut = 0;
    for u in group_a.iter() {
        for &v in &adj[*u] {
            if !group_a.contains(&v) {
                cut += 1;
            }
        }
    }
    cut
}

fn find_min_bisection(n: usize, adj: &[Vec<usize>]) -> (usize, Vec<usize>) {
    let target_size = n / 2;
    let mut best_cut = usize::MAX;
    let mut best_partition = Vec::new();

    // Try multiple starting partitions
    for attempt in 0..200 {
        let mut group_a = HashSet::new();

        if attempt < n {
            // Strategy 1: BFS from different starting vertices
            group_a.insert(attempt);
            let mut queue = vec![attempt];
            while group_a.len() < target_size && !queue.is_empty() {
                let u = queue.pop().unwrap();
                for &v in &adj[u] {
                    if !group_a.contains(&v) && group_a.len() < target_size {
                        group_a.insert(v);
                        queue.push(v);
                    }
                }
            }
        } else {
            // Strategy 2: Vertices by degree
            let mut vertices_by_degree: Vec<(usize, usize)> = (0..n)
                .map(|v| (v, adj[v].len()))
                .collect();
            vertices_by_degree.sort_by_key(|x| {
                if attempt % 2 == 0 {
                    -(x.1 as i32)
                } else {
                    x.1 as i32
                }
            });
            for (v, _) in vertices_by_degree.iter().take(target_size) {
                group_a.insert(*v);
            }
        }

        // Fill if needed
        if group_a.len() < target_size {
            for v in 0..n {
                if group_a.len() >= target_size {
                    break;
                }
                group_a.insert(v);
            }
        }

        // Local search: try swapping vertices to reduce cut
        let mut improved = true;
        let mut iterations = 0;
        while improved && iterations < 500 {
            iterations += 1;
            improved = false;

            let current_cut = calculate_cut(adj, &group_a);

            // Try swapping pairs
            'swap_search: for v_a in 0..n {
                if !group_a.contains(&v_a) {
                    continue;
                }

                for v_b in 0..n {
                    if group_a.contains(&v_b) {
                        continue;
                    }

                    // Calculate cut after swap
                    let mut new_group_a = group_a.clone();
                    new_group_a.remove(&v_a);
                    new_group_a.insert(v_b);

                    let new_cut = calculate_cut(adj, &new_group_a);

                    if new_cut < current_cut {
                        group_a = new_group_a;
                        improved = true;
                        break 'swap_search;
                    }
                }
            }
        }

        let cut = calculate_cut(adj, &group_a);

        if cut < best_cut {
            best_cut = cut;
            let mut partition_a: Vec<usize> = group_a.iter().cloned().collect();
            partition_a.sort();
            best_partition = partition_a;
        }
    }

    (best_cut, best_partition)
}
