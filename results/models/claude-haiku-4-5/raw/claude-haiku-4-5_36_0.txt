#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<vector<int>> adj;
int best_size = 0;
vector<int> best_set;

void backtrack(vector<int>& current, vector<bool>& candidates) {
    int remaining = 0;
    for (int i = 0; i < n; i++) {
        if (candidates[i]) remaining++;
    }

    if (remaining == 0) {
        if (current.size() > best_size) {
            best_size = current.size();
            best_set = current;
        }
        return;
    }

    // Pruning: can't beat best even if we take all remaining
    if (current.size() + remaining <= best_size) {
        return;
    }

    // Pick vertex with minimum degree in remaining subgraph
    int next_vertex = -1;
    int min_degree = INT_MAX;
    for (int i = 0; i < n; i++) {
        if (candidates[i]) {
            int degree = 0;
            for (int neighbor : adj[i]) {
                if (candidates[neighbor]) degree++;
            }
            if (degree < min_degree) {
                min_degree = degree;
                next_vertex = i;
            }
        }
    }

    // Collect neighbors to mark/unmark efficiently
    vector<int> neighbors;
    for (int neighbor : adj[next_vertex]) {
        if (candidates[neighbor]) {
            neighbors.push_back(neighbor);
        }
    }

    // Branch 1: include next_vertex
    candidates[next_vertex] = false;
    for (int neighbor : neighbors) {
        candidates[neighbor] = false;
    }
    current.push_back(next_vertex);
    backtrack(current, candidates);
    current.pop_back();

    // Restore state
    candidates[next_vertex] = true;
    for (int neighbor : neighbors) {
        candidates[neighbor] = true;
    }

    // Branch 2: exclude next_vertex
    candidates[next_vertex] = false;
    backtrack(current, candidates);
    candidates[next_vertex] = true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    adj.resize(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> current;
    vector<bool> candidates(n, true);
    backtrack(current, candidates);

    cout << best_size << "\n";
    for (int i = 0; i < best_set.size(); i++) {
        if (i > 0) cout << " ";
        cout << best_set[i];
    }
    cout << "\n";

    return 0;
}
