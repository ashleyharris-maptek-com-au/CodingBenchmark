OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 15
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertexIn Block
               OpMemberDecorate %gl_PerVertexIn 0 BuiltIn Position
               OpDecorate %arr_in ArrayStride 16
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %gl_PerVertexOut Block
               OpMemberDecorate %gl_PerVertexOut 0 BuiltIn Position
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
       %bool = OpTypeBool
        %int = OpTypeInt 32 1
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
      %int_5 = OpConstant %int 5
      %int_6 = OpConstant %int 6
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
  %float_0_3 = OpConstant %float 0.3
%float_0_001 = OpConstant %float 0.001
%v3float_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertexIn = OpTypeStruct %v4float
%arr_in = OpTypeArray %gl_PerVertexIn %int_6
%ptr_arr_in = OpTypePointer Input %arr_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3 = OpTypeArray %v3float %int_6
%ptr_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertexOut = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertexOut
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%func_void_type = OpTypeFunction %void
%func_bool_4v4v4 = OpTypeFunction %bool %v4float %v4float %v4float
%func_void_3v4v3 = OpTypeFunction %void %v4float %v4float %v3float

       %gl_in = OpVariable %ptr_arr_in Input
    %worldPos = OpVariable %ptr_arr_v3 Input
     %normal = OpVariable %ptr_arr_v3 Input
      %color = OpVariable %ptr_arr_v3 Input
     %gl_out = OpVariable %ptr_out_pv Output
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %func_void_type
   %bb_entry = OpLabel

; Load the three main triangle vertices (indices 0, 2, 4)
  %pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
       %pos0 = OpLoad %v4float %pos0_ptr
  %pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
       %pos2 = OpLoad %v4float %pos2_ptr
  %pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
       %pos4 = OpLoad %v4float %pos4_ptr

; Load adjacent vertices (indices 1, 3, 5)
  %pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
       %pos1 = OpLoad %v4float %pos1_ptr
  %pos3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
       %pos3 = OpLoad %v4float %pos3_ptr
  %pos5_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_5 %int_0
       %pos5 = OpLoad %v4float %pos5_ptr

; Load colors
 %color0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
      %color0 = OpLoad %v3float %color0_ptr
%color0_dim = OpVectorTimesScalar %v3float %color0 %float_0_3
 %color2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
      %color2 = OpLoad %v3float %color2_ptr
%color2_dim = OpVectorTimesScalar %v3float %color2 %float_0_3
 %color4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
      %color4 = OpLoad %v3float %color4_ptr
%color4_dim = OpVectorTimesScalar %v3float %color4 %float_0_3

; Check edge 0-2 (adjacent vertex 1)
%edge02_boundary = OpFunctionCall %bool %isBoundaryEdge %pos0 %pos2 %pos1
           OpSelectionMerge %edge02_merge None
           OpBranchConditional %edge02_boundary %edge02_true %edge02_merge
   %edge02_true = OpLabel
                  OpFunctionCall %void %emitLineSegment %pos0 %pos2 %v3float_white
           OpBranch %edge02_merge
  %edge02_merge = OpLabel

; Check edge 2-4 (adjacent vertex 3)
%edge24_boundary = OpFunctionCall %bool %isBoundaryEdge %pos2 %pos4 %pos3
           OpSelectionMerge %edge24_merge None
           OpBranchConditional %edge24_boundary %edge24_true %edge24_merge
   %edge24_true = OpLabel
                  OpFunctionCall %void %emitLineSegment %pos2 %pos4 %v3float_white
           OpBranch %edge24_merge
  %edge24_merge = OpLabel

; Check edge 4-0 (adjacent vertex 5)
%edge40_boundary = OpFunctionCall %bool %isBoundaryEdge %pos4 %pos0 %pos5
           OpSelectionMerge %edge40_merge None
           OpBranchConditional %edge40_boundary %edge40_true %edge40_merge
   %edge40_true = OpLabel
                  OpFunctionCall %void %emitLineSegment %pos4 %pos0 %v3float_white
           OpBranch %edge40_merge
  %edge40_merge = OpLabel

; Emit the main triangle with dimmed colors
%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos_out_ptr %pos0
               OpStore %outColor %color0_dim
               OpEmitVertex

               OpStore %pos_out_ptr %pos2
               OpStore %outColor %color2_dim
               OpEmitVertex

               OpStore %pos_out_ptr %pos4
               OpStore %outColor %color4_dim
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd

; isBoundaryEdge function
%isBoundaryEdge = OpFunction %bool None %func_bool_4v4v4
         %v1_in = OpFunctionParameter %v4float
         %v2_in = OpFunctionParameter %v4float
        %adj_in = OpFunctionParameter %v4float
%bb_is_boundary = OpLabel

; Extract XYZ components
%v1_xyz = OpVectorShuffle %v3float %v1_in %v1_in 0 1 2
%v2_xyz = OpVectorShuffle %v3float %v2_in %v2_in 0 1 2
%adj_xyz = OpVectorShuffle %v3float %adj_in %adj_in 0 1 2

; Check if adjacent vertex is close to v1
%diff1 = OpFSub %v3float %adj_xyz %v1_xyz
%dot1 = OpDot %float %diff1 %diff1
%dist1 = OpExtInst %float %extinst_set %Sqrt %dot1
%cond1 = OpFOrdLessThan %bool %dist1 %float_0_001

; Check if adjacent vertex is close to v2
%diff2 = OpFSub %v3float %adj_xyz %v2_xyz
%dot2 = OpDot %float %diff2 %diff2
%dist2 = OpExtInst %float %extinst_set %Sqrt %dot2
%cond2 = OpFOrdLessThan %bool %dist2 %float_0_001

; Return logical OR
%result = OpLogicalOr %bool %cond1 %cond2
           OpReturnValue %result
           OpFunctionEnd

; emitLineSegment function - emit thick line as triangle strip
%emitLineSegment = OpFunction %void None %func_void_3v4v3
         %line_p1 = OpFunctionParameter %v4float
         %line_p2 = OpFunctionParameter %v4float
         %line_col = OpFunctionParameter %v3float
 %bb_emit_line = OpLabel

; Emit 4 vertices for a degenerate quad (simplified thick line)
%line_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %line_out_ptr %line_p1
                OpStore %outColor %line_col
                OpEmitVertex

                OpStore %line_out_ptr %line_p1
                OpStore %outColor %line_col
                OpEmitVertex

                OpStore %line_out_ptr %line_p2
                OpStore %outColor %line_col
                OpEmitVertex

                OpStore %line_out_ptr %line_p2
                OpStore %outColor %line_col
                OpEmitVertex

                OpEndPrimitive
                OpReturn
                OpFunctionEnd

%extinst_set = OpExtInstImport "GLSL.std.450"
