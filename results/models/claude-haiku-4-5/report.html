struct City {
    float x, y;
    int idx;
};

inline float dist_sq(const City& a, const City& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// Simple nearest neighbor for small instances
vector<int> solve_small(const vector<City>& cities) {
    int n = cities.size();
    vector<int> tour;
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        float best_dist = 1e30f;
        int best_idx = -1;
        int current = tour.back();

        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                float d = dist_sq(cities[current], cities[j]);
                if (d < best_dist) {
                    best_dist = d;
                    best_idx = j;
                }
            }
        }

        tour.push_back(best_idx);
        visited[best_idx] = true;
    }

    return tour;
}

// 2-opt local improvement
void improve_2opt(vector<int>& tour, const vector<City>& cities, int max_iterations) {
    int n = tour.size();

    for (int iter = 0; iter < max_iterations; iter++) {
        bool improved = false;

        for (int i = 0; i < n - 2; i++) {
            for (int j = i + 2; j < n; j++) {

                                dist_sq(cities[tour[j]], cities[tour[(j+1)%n]]);
                float new_cost = dist_sq(cities[tour[i]], cities[tour[j]]) +
                                dist_sq(cities[tour[i+1]], cities[tour[(j+1)%n]]);

                if (new_cost < old_cost - 1e-9f) {

}

fn greedy_path(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    valid_cells: usize,
    require_cycle: bool,
) -> Option<Vec<(usize, usize)>> {

    let mut path = Vec::new();
    visited.insert((0, 0));
    path.push((0, 0));

    while visited.len() < valid_cells {
        let (cx, cy) = path[path.len() - 1];

        let mut best = None;
        let mut best_degree = usize::MAX;

        for (nx, ny) in neighbors {
            if !visited.contains(&(nx, ny)) && !obstacles.contains(&(nx, ny)) {
                let degree = count_unvisited(nx, ny, width, height, &visited, obstacles);
                if degree < best_degree || (degree == best_degree && best.is_none()) {

            } else {

            }

            Some(path)
        }
    } else {
        None
    }
}

    let mut end = (0, 0);

            if ch == 'A' {

fn bfs(maze: &[String], start: (usize, usize), end: (usize, usize), h: usize, w: usize) -> Vec<(usize, usize)> {
    let mut visited = vec![vec![false; w]; h];
    let mut parent = vec![vec![None; w]; h];

    let mut queue = VecDeque::new();
    queue.push_back(start);
    visited[start.0][start.1] = true;

    let directions = [(0, 1), (1, 0), (0, -1), (-1, 0)];  // right, down, left, up

    while let Some((row, col)) = queue.pop_front() {
        if (row, col) == end {
            // Reconstruct path
            let mut path = vec![(row, col)];
            let mut current = (row, col);

            while let Some(prev) = parent[current.0][current.1] {
                path.push(prev);
                current = prev;
            }

            path.reverse();
            return path;
        }

        for (dr, dc) in &directions {
            let nr = (row as i32 + dr) as usize;
            let nc = (col as i32 + dc) as usize;

            if nr < h && nc < w && !visited[nr][nc] {
                if let Some(ch) = maze[nr].chars().nth(nc) {
                    if ch != '#' {
                        visited[nr][nc] = true;
                        parent[nr][nc] = Some((row, col));
                        queue.push_back((nr, nc));
                    }
                }
            }
        }
    }

    // No path found
    vec![]
}

        int stockLength = int.Parse(line1[1]);

        // Create array with cut indices and their lengths

        for (int i = 0; i < n; i++)
        {
            cutIndices.Add((i, cuts[i]));

        // Sort by length in decreasing order (First Fit Decreasing)

        foreach (var (originalIndex, length) in cutIndices)

struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }

    fn distance(&self, other: &Point) -> f64 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        (dx * dx + dy * dy).sqrt()
    }

    fn rotate(&self, angle: f64) -> Point {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Point {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Point {
        Point {
            x: self.x + dx,
            y: self.y + dy,
        }
    }
}

    fn new(vertices: Vec<Point>) -> Self {

    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }

    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;

        let mut max_y = f64::NEG_INFINITY;

            max_x = max_x.max(v.x);
            min_y = min_y.min(v.y);
            max_y = max_y.max(v.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn rotate(&self, angle: f64) -> Polygon {
        let rotated: Vec<Point> = self.vertices.iter().map(|v| v.rotate(angle)).collect();
        Polygon::new(rotated)
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let translated: Vec<Point> = self

            .map(|v| v.translate(dx, dy))
            .collect();
        Polygon::new(translated)
    }

fn point_in_polygon(p: &Point, poly: &Polygon) -> bool {
    let mut inside = false;
    let n = poly.vertices.len();

    for i in 0..n {
        let j = (i + 1) % n;
        let xi = poly.vertices[i].x;
        let yi = poly.vertices[i].y;
        let xj = poly.vertices[j].x;
        let yj = poly.vertices[j].y;

        let intersect = ((yi > p.y) != (yj > p.y))
            && (p.x < (xj - xi) * (p.y - yi) / (yj - yi) + xi);
        if intersect {
            inside = !inside;
        }

    inside

fn segments_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {

    let mut inside = false;

        let yi = poly.vertices[i].y;

fn segments_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {

struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Polygon { vertices }
    }

    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }

    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;

        let mut max_y = f64::NEG_INFINITY;

            max_x = max_x.max(v.x);
            min_y = min_y.min(v.y);
            max_y = max_y.max(v.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn rotate(&self, angle: f64) -> Polygon {
        let rotated: Vec<Point> = self.vertices.iter().map(|v| v.rotate(angle)).collect();
        Polygon::new(rotated)
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let translated: Vec<Point> = self
            .vertices
            .iter()
            .map(|v| v.translate(dx, dy))
            .collect();
        Polygon::new(translated)
    }

        let mut cx = 0.0;
        let mut cy = 0.0;
        for v in &self.vertices {
            cx += v.x;
            cy += v.y;

        let n = self.vertices.len() as f64;

fn point_in_polygon(p: &Point, poly: &Polygon) -> bool {
    let mut inside = false;
    let n = poly.vertices.len();

    for i in 0..n {
        let j = (i + 1) % n;
        let xi = poly.vertices[i].x;
        let yi = poly.vertices[i].y;
        let xj = poly.vertices[j].x;
        let yj = poly.vertices[j].y;

        let intersect = ((yi > p.y) != (yj > p.y))
            && (p.x < (xj - xi) * (p.y - yi) / (yj - yi) + xi);
        if intersect {
            inside = !inside;
        }
    }

    inside
}

fn segments_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {
    let ccw = |a: &Point, b: &Point, c: &Point| -> i32 {
        ((c.y - a.y) * (b.x - a.x) > (b.y - a.y) * (c.x - a.x)) as i32
    };

    ccw(p1, p3, p4) != ccw(p2, p3, p4) && ccw(p1, p2, p3) != ccw(p1, p2, p4)
}

fn polygon_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    for v in &poly1.vertices {
        if point_in_polygon(v, poly2) {
            return true;
        }
    }
    for v in &poly2.vertices {
        if point_in_polygon(v, poly1) {
            return true;
        }
    }

    let n1 = poly1.vertices.len();
    let n2 = poly2.vertices.len();
    for i in 0..n1 {
        for j in 0..n2 {
            if segments_intersect(
                &poly1.vertices[i],
                &poly1.vertices[(i + 1) % n1],
                &poly2.vertices[j],
                &poly2.vertices[(j + 1) % n2],
            ) {
                return true;
            }
        }
    }

    false
}

struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {

        Polygon { vertices }

    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }

    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            min_x = min_x.min(v.x);
            max_x = max_x.max(v.x);
            min_y = min_y.min(v.y);
            max_y = max_y.max(v.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn rotate(&self, angle: f64) -> Polygon {
        let rotated: Vec<Point> = self.vertices.iter().map(|v| v.rotate(angle)).collect();
        Polygon::new(rotated)
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let translated: Vec<Point> = self
            .vertices
            .iter()
            .map(|v| v.translate(dx, dy))
            .collect();
        Polygon::new(translated)
    }

    fn centroid(&self) -> Point {
        let mut cx = 0.0;
        let mut cy = 0.0;
        for v in &self.vertices {
            cx += v.x;
            cy += v.y;
        }
        let n = self.vertices.len() as f64;
        Point { x: cx / n, y: cy / n }
    }
}

fn point_in_polygon(p: &Point, poly: &Polygon) -> bool {
    let mut inside = false;
    let n = poly.vertices.len();

    for i in 0..n {
        let j = (i + 1) % n;
        let xi = poly.vertices[i].x;
        let yi = poly.vertices[i].y;
        let xj = poly.vertices[j].x;
        let yj = poly.vertices[j].y;

        let intersect = ((yi > p.y) != (yj > p.y))
            && (p.x < (xj - xi) * (p.y - yi) / (yj - yi) + xi);
        if intersect {
            inside = !inside;
        }
    }

    inside
}

fn segments_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {
    let ccw = |a: &Point, b: &Point, c: &Point| -> i32 {
        ((c.y - a.y) * (b.x - a.x) > (b.y - a.y) * (c.x - a.x)) as i32
    };

    ccw(p1, p3, p4) != ccw(p2, p3, p4) && ccw(p1, p2, p3) != ccw(p1, p2, p4)
}

fn polygon_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    for v in &poly1.vertices {
        if point_in_polygon(v, poly2) {
            return true;
        }
    }
    for v in &poly2.vertices {
        if point_in_polygon(v, poly1) {
            return true;
        }
    }

    let n1 = poly1.vertices.len();
    let n2 = poly2.vertices.len();
    for i in 0..n1 {
        for j in 0..n2 {
            if segments_intersect(
                &poly1.vertices[i],
                &poly1.vertices[(i + 1) % n1],
                &poly2.vertices[j],
                &poly2.vertices[(j + 1) % n2],
            ) {
                return true;
            }
        }
    }

    false
}

fn piece_fits_in_stock(piece: &Polygon, stock: &Polygon) -> bool {
    for v in &piece.vertices {
        if !point_in_polygon(v, stock) {
            return false;
        }
    }
    true
}

fn find_placement_position(
    stock: &Polygon,
    piece: &Polygon,
    existing_pieces: &[(Polygon, f64, f64, f64)],
    angle: f64,
) -> Option<(f64, f64)> {

    let height = p_max_y - p_min_y;

    let stock_width = max_x - min_x;
    let stock_height = max_y - min_y;

        let mut y = min_y - p_min_y;

    }

}

using namespace std;

bool hasCommonSubstring(const vector<string>& strings, int len) {
    if (strings.empty() || len <= 0) return false;

    unordered_set<string> candidates;

    // Get all substrings of length `len` from the first string
    for (size_t i = 0; i + len <= strings[0].length(); i++) {
        candidates.insert(strings[0].substr(i, len));
    }

    if (candidates.empty()) return false;

    // Check if any candidate exists in all other strings
    for (size_t j = 1; j < strings.size(); j++) {
        unordered_set<string> current;
        for (size_t i = 0; i + len <= strings[j].length(); i++) {
            string substr = strings[j].substr(i, len);
            if (candidates.count(substr)) {
                current.insert(substr);
            }
        }
        candidates = current;
        if (candidates.empty()) return false;
    }

    return true;
}

#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    // Read job definitions: for each job, read number of tasks and their machine assignments
    vector<vector<pair<int, int>>> jobs(J);
    int total_tasks = 0;

    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; t++) {
            cin >> jobs[j][t].first >> jobs[j][t].second;  // machine, duration
            total_tasks++;
        }
    }

    // Greedy scheduling algorithm: O(total_tasks) - efficient for any scale
    // For each task, schedule it as early as possible given constraints

    vector<int> machine_free(M, 0);  // Track when each machine becomes available

    for (int j = 0; j < J; j++) {
        int prev_task_end = 0;  // When the previous task in this job completed

        for (int t = 0; t < jobs[j].size(); t++) {
            int machine = jobs[j][t].first;
            int duration = jobs[j][t].second;

            // Task can start only when:
            // 1. Previous task in same job has completed (prev_task_end)
            // 2. The required machine is available (machine_free[machine])
            int start_time = max(prev_task_end, machine_free[machine]);
            int end_time = start_time + duration;

            // Update constraints for next task

        }

    {
        var packed = new List<Placement>();
        var occupied = new List<Placement>();

        var sorted = boxes.OrderByDescending(b => (long)b.W * b.H * b.D).ToList();

                z < p.Z + p.D && z + d > p.Z)
            {
                return true;
            }
        }
        return false;
    }

using System.Linq;

class MiningOptimizer
{
    struct VoxelInfo
    {
        public int Mass;
        public int Value;
    }

    static void Main()
    {
        var header = Console.ReadLine().Split();
        int xCount = int.Parse(header[0]);
        int yCount = int.Parse(header[1]);
        int zCount = int.Parse(header[2]);

        var voxels = new VoxelInfo[xCount, yCount, zCount];

        // Read voxels
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    byte b = (byte)Console.Read();
                    voxels[x, y, z].Mass = b % 16;
                    voxels[x, y, z].Value = b / 16;
                }
            }
        }

        var operations = new List<Operation>();
        var dug = new HashSet<(int, int, int)>();
        var heightMap = new int[xCount, yCount];

        // Initialize height map - height of topmost solid voxel
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                heightMap[x, y] = -1;
                for (int z = zCount - 1; z >= 0; z--)
                {
                    if (voxels[x, y, z].Mass > 0)
                    {
                        heightMap[x, y] = z;
                        break;
                    }
                }
            }
        }

        // Find all minable voxels and sort by profit
        var candidates = new List<(int x, int y, int z, int mass, int value)>();
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    if (voxels[x, y, z].Mass > 0)
                    {
                        candidates.Add((x, y, z, voxels[x, y, z].Mass, voxels[x, y, z].Value));
                    }
                }
            }
        }

        // Sort by value descending
        candidates.Sort((a, b) => (b.value * b.mass).CompareTo(a.value * a.mass));

        // Mine valuable voxels
        foreach (var (digX, digY, digZ, mass, value) in candidates)

                continue;

    static bool IsExposed(VoxelInfo[,,] voxels, HashSet<(int, int, int)> dug,
        int x, int y, int z, int xCount, int yCount, int zCount)
    {
        // Check adjacent cells
        if (z == zCount - 1 || voxels[x, y, z + 1].Mass == 0 || dug.Contains((x, y, z + 1)))
            return true;

        if (x == 0 || voxels[x - 1, y, z].Mass == 0 || dug.Contains((x - 1, y, z)))
            return true;
        if (x == xCount - 1 || voxels[x + 1, y, z].Mass == 0 || dug.Contains((x + 1, y, z)))
            return true;
        if (y == 0 || voxels[x, y - 1, z].Mass == 0 || dug.Contains((x, y - 1, z)))
            return true;
        if (y == yCount - 1 || voxels[x, y + 1, z].Mass == 0 || dug.Contains((x, y + 1, z)))

        return false;

        var visited = new HashSet<(int, int, int)>();

        queue.Enqueue(new State { X = sx, Y = sy, Z = sz, Path = new List<(int, int)>() });
        visited.Add((sx, sy, sz));

        while (queue.Count > 0)
        {
            var state = queue.Dequeue();

            // Reached processing plant
            if (state.X == 0 && state.Y == 0)
            {
                path = state.Path;
                return true;
            }

            // Try moving in 4 directions
            int[][] directions = {
                new[] { 1, 0 }, new[] { -1, 0 },
                new[] { 0, 1 }, new[] { 0, -1 }
            };

            foreach (var dir in directions)
            {
                int nx = state.X + dir[0];
                int ny = state.Y + dir[1];

                    continue;

                    continue;

                newPath.Add((nx, ny));

            }
        }
