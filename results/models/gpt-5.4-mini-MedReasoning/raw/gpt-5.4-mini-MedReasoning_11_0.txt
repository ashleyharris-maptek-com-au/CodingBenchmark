use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Default)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
}

use std::ops::{Add, AddAssign, Mul, Sub, SubAssign};

impl Add for Pt {
    type Output = Pt;
    fn add(self, rhs: Pt) -> Pt {
        Pt::new(self.x + rhs.x, self.y + rhs.y)
    }
}

impl Sub for Pt {
    type Output = Pt;
    fn sub(self, rhs: Pt) -> Pt {
        Pt::new(self.x - rhs.x, self.y - rhs.y)
    }
}

impl AddAssign for Pt {
    fn add_assign(&mut self, rhs: Pt) {
        self.x += rhs.x;
        self.y += rhs.y;
    }
}

impl SubAssign for Pt {
    fn sub_assign(&mut self, rhs: Pt) {
        self.x -= rhs.x;
        self.y -= rhs.y;
    }
}

impl Mul<f64> for Pt {
    type Output = Pt;
    fn mul(self, rhs: f64) -> Pt {
        Pt::new(self.x * rhs, self.y * rhs)
    }
}

fn cross(a: Pt, b: Pt) -> f64 {
    a.x * b.y - a.y * b.x
}

fn cross3(a: Pt, b: Pt, c: Pt) -> f64 {
    cross(b - a, c - a)
}

fn rotate(p: Pt, ang: f64) -> Pt {
    let (s, c) = ang.sin_cos();
    Pt::new(p.x * c - p.y * s, p.x * s + p.y * c)
}

fn polygon_centroid(poly: &[Pt]) -> Pt {
    let n = poly.len();
    let mut a2 = 0.0;
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..n {
        let p = poly[i];
        let q = poly[(i + 1) % n];
        let cr = cross(p, q);
        a2 += cr;
        cx += (p.x + q.x) * cr;
        cy += (p.y + q.y) * cr;
    }
    if a2.abs() < 1e-12 {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for p in poly {
            sx += p.x;
            sy += p.y;
        }
        let n = poly.len() as f64;
        Pt::new(sx / n, sy / n)
    } else {
        Pt::new(cx / (3.0 * a2), cy / (3.0 * a2))
    }
}

fn polygon_bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in poly {
        minx = minx.min(p.x);
        maxx = maxx.max(p.x);
        miny = miny.min(p.y);
        maxy = maxy.max(p.y);
    }
    (minx, maxx, miny, maxy)
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    let eps = 1e-9;
    cross3(a, b, p).abs() <= eps
        && p.x >= a.x.min(b.x) - eps
        && p.x <= a.x.max(b.x) + eps
        && p.y >= a.y.min(b.y) - eps
        && p.y <= a.y.max(b.y) + eps
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if on_segment(a, b, p) {
            return true;
        }
        let intersects = ((a.y > p.y) != (b.y > p.y))
            && (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y + 1e-300) + a.x);
        if intersects {
            inside = !inside;
        }
    }
    inside
}

fn principal_axis_angle(poly: &[Pt]) -> f64 {
    let c = polygon_centroid(poly);
    let mut sxx = 0.0;
    let mut syy = 0.0;
    let mut sxy = 0.0;
    for &p in poly {
        let x = p.x - c.x;
        let y = p.y - c.y;
        sxx += x * x;
        syy += y * y;
        sxy += x * y;
    }
    0.5 * (2.0 * sxy).atan2(sxx - syy)
}

fn unique_angles(mut angs: Vec<f64>) -> Vec<f64> {
    for a in &mut angs {
        while *a <= -PI {
            *a += 2.0 * PI;
        }
        while *a > PI {
            *a -= 2.0 * PI;
        }
    }
    angs.sort_by(|a, b| a.partial_cmp(b).unwrap());
    let mut out = Vec::with_capacity(angs.len());
    for a in angs {
        if out.iter().all(|b: &f64| (a - *b).abs() > 1e-9) {
            out.push(a);
        }
    }
    out
}

fn candidate_angles(piece: &[Pt]) -> Vec<f64> {
    let n = piece.len();
    let base = principal_axis_angle(piece);
    let mut angs = vec![0.0, PI * 0.5, PI, PI * 1.5, base, base + PI * 0.5];
    if n <= 20 {
        for k in 0..16 {
            angs.push((k as f64) * 2.0 * PI / 16.0);
        }
    } else if n <= 80 {
        for k in 0..8 {
            angs.push((k as f64) * 2.0 * PI / 8.0);
        }
    } else {
        for k in 0..4 {
            angs.push((k as f64) * PI * 0.5);
        }
    }
    unique_angles(angs)
}

fn stock_samples(stock: &[Pt]) -> Vec<Pt> {
    let (minx, maxx, miny, maxy) = polygon_bbox(stock);
    let centroid = polygon_centroid(stock);
    let mut pts = Vec::new();
    pts.push(centroid);
    pts.push(Pt::new((minx + maxx) * 0.5, (miny + maxy) * 0.5));
    for &v in stock {
        pts.push(v);
    }
    let grid = 12usize;
    for i in 0..grid {
        for j in 0..grid {
            let x = minx + (i as f64 + 0.5) * (maxx - minx) / (grid as f64);
            let y = miny + (j as f64 + 0.5) * (maxy - miny) / (grid as f64);
            let p = Pt::new(x, y);
            if point_in_poly(p, stock) {
                pts.push(p);
            }
        }
    }
    let mut rng = XorShift64::new(0x9E3779B97F4A7C15);
    let random_trials = 160usize;
    for _ in 0..random_trials {
        let x = minx + rng.next_f64() * (maxx - minx);
        let y = miny + rng.next_f64() * (maxy - miny);
        let p = Pt::new(x, y);
        if point_in_poly(p, stock) {
            pts.push(p);
        }
    }
    // Small deduplication keeps the search fast on repeated grids.
    let mut uniq = Vec::with_capacity(pts.len());
    for p in pts {
        if uniq
            .iter()
            .all(|q: &Pt| (p.x - q.x).abs() > 1e-9 || (p.y - q.y).abs() > 1e-9)
        {
            uniq.push(p);
        }
    }
    uniq
}

fn fits(piece_rel_rot: &[Pt], anchor: Pt, stock: &[Pt]) -> bool {
    let n = piece_rel_rot.len();
    let samples_per_edge = if n <= 20 { 4 } else if n <= 80 { 3 } else { 2 };
    for &q in piece_rel_rot {
        if !point_in_poly(anchor + q, stock) {
            return false;
        }
    }
    for i in 0..n {
        let a = anchor + piece_rel_rot[i];
        let b = anchor + piece_rel_rot[(i + 1) % n];
        for s in 1..=samples_per_edge {
            let t = s as f64 / (samples_per_edge as f64 + 1.0);
            let p = Pt::new(a.x + (b.x - a.x) * t, a.y + (b.y - a.y) * t);
            if !point_in_poly(p, stock) {
                return false;
            }
        }
    }
    true
}

struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: seed.max(1) }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    fn next_f64(&mut self) -> f64 {
        let v = self.next_u64() >> 11;
        (v as f64) * (1.0 / ((1u64 << 53) as f64))
    }
}

fn parse_input() -> (Vec<Pt>, Vec<Vec<Pt>>) {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();
    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Pt::new(x, y));
    }
    let mut pieces = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Pt::new(x, y));
        }
        pieces.push(poly);
    }
    (stock, pieces)
}

fn solve_piece(piece: &[Pt], stock: &[Pt], samples: &[Pt]) -> (f64, f64, f64) {
    let ref_pt = polygon_centroid(piece);
    let rel: Vec<Pt> = piece.iter().map(|&p| p - ref_pt).collect();
    let angles = candidate_angles(piece);

    // First try a bounded deterministic search over promising angles and points.
    for &ang in &angles {
        let rot_ref = rotate(ref_pt, ang);
        let rotated: Vec<Pt> = rel.iter().copied().map(|p| rotate(p, ang)).collect();
        for &anchor in samples {
            if fits(&rotated, anchor, stock) {
                let t = anchor - rot_ref;
                return (t.x, t.y, ang);
            }
        }
    }

    // If the stock is large enough, the centroid-based position usually works.
    let fallback_points = {
        let mut pts = Vec::new();
        let c = polygon_centroid(stock);
        pts.push(c);
        let (minx, maxx, miny, maxy) = polygon_bbox(stock);
        let mut rng = XorShift64::new(0xD1B54A32D192ED03);
        for _ in 0..256 {
            let x = minx + rng.next_f64() * (maxx - minx);
            let y = miny + rng.next_f64() * (maxy - miny);
            let p = Pt::new(x, y);
            if point_in_poly(p, stock) {
                pts.push(p);
            }
        }
        pts
    };
    for &ang in &angles {
        let rot_ref = rotate(ref_pt, ang);
        let rotated: Vec<Pt> = rel.iter().copied().map(|p| rotate(p, ang)).collect();
        for &anchor in &fallback_points {
            if fits(&rotated, anchor, stock) {
                let t = anchor - rot_ref;
                return (t.x, t.y, ang);
            }
        }
    }

    // Last resort: best-effort placement at the stock centroid.
    let ang = 0.0;
    let t = polygon_centroid(stock) - rotate(ref_pt, ang);
    (t.x, t.y, ang)
}

fn main() {
    let (stock, pieces) = parse_input();
    let samples = stock_samples(&stock);

    let mut out = String::new();
    out.push_str(&format!("{}\n", pieces.len()));

    for piece in &pieces {
        let (x, y, rot) = solve_piece(piece, &stock, &samples);
        out.push_str(&format!("0 {:.10} {:.10} {:.10}\n", x, y, rot));
    }

    print!("{}", out);
}
