; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %ubo "ubo"
OpName %UBO "UBO"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %outColor "outColor"

OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float

%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_3 = OpConstant %float 3.0

%v3_0_2_0_3_0_8 = OpConstantComposite %v3float %float_0_2 %float_0_3 %float_0_8
%v3_1_0_8_0_5 = OpConstantComposite %v3float %float_1 %float_0_8 %float_0_5
%v3_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0
%v4_1 = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1

%glsl = OpExtInstImport "GLSL.std.450"

%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

%main = OpFunction %void None %func
%entry = OpLabel

%worldPos_val = OpLoad %v3float %worldPos
%normal_val = OpLoad %v3float %normal
%normal_n = OpExtInst %v3float %glsl Normalize %normal_val

%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos_val = OpLoad %v4float %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %v3float %lightPos_val %lightPos_val 0 1 2
%lightVec = OpFSub %v3float %lightPos_xyz %worldPos_val
%lightDir = OpExtInst %v3float %glsl Normalize %lightVec

%cameraPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%cameraPos_val = OpLoad %v4float %cameraPos_ptr
%cameraPos_xyz = OpVectorShuffle %v3float %cameraPos_val %cameraPos_val 0 1 2
%viewVec = OpFSub %v3float %cameraPos_xyz %worldPos_val
%viewDir = OpExtInst %v3float %glsl Normalize %viewVec

%ndotl = OpDot %float %normal_n %lightDir
%diffuse = OpExtInst %float %glsl FMax %ndotl %float_0

%diffuseHalf = OpFMul %float %float_0_5 %diffuse
%diffuseScale = OpFAdd %float %float_0_3 %diffuseHalf
%baseLit = OpVectorTimesScalar %v3float %v3_0_2_0_3_0_8 %diffuseScale

%ndotv = OpDot %float %normal_n %viewDir
%ndotv_clamped = OpExtInst %float %glsl FMax %ndotv %float_0
%one_minus_ndotv = OpFSub %float %float_1 %ndotv_clamped
%rim = OpExtInst %float %glsl Pow %one_minus_ndotv %float_3
%rimTint = OpVectorTimesScalar %v3float %v3_1_0_8_0_5 %rim

%rgb = OpFAdd %v3float %baseLit %rimTint
%color = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %color

OpReturn
OpFunctionEnd
