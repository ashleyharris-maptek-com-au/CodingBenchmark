// Minimum Edge Cut solver
// Strategy:
// 1) Find any bridge. If one exists, the minimum cut is 1.
// 2) For smaller graphs, run Stoer-Wagner exactly.
// 3) For larger graphs, fall back to randomized contraction (Karger-Stein style).

#include <algorithm>
#include <chrono>
#include <cstdint>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <limits>
#include <numeric>
#include <random>
#include <utility>
#include <vector>

using namespace std;

struct FastScanner {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline char getChar() {
        if (idx >= size) {
            size = fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T &out) {
        char c;
        T sign = 1;
        T val = 0;
        c = getChar();
        if (!c) return false;
        while (c != '-' && (c < '0' || c > '9')) {
            c = getChar();
            if (!c) return false;
        }
        if (c == '-') {
            sign = -1;
            c = getChar();
        }
        for (; c >= '0' && c <= '9'; c = getChar()) {
            val = val * 10 + (c - '0');
        }
        out = val * sign;
        return true;
    }
};

struct Edge {
    int u, v;
};

struct CutResult {
    long long value = (1LL << 60);
    vector<char> side;
};

struct GraphPack {
    int n = 0;
    vector<Edge> edges;
    vector<int> oldToNew;
};

struct DSU {
    vector<int> p, r;
    int comps;
    explicit DSU(int n = 0) { init(n); }
    void init(int n) {
        p.resize(n);
        r.assign(n, 0);
        iota(p.begin(), p.end(), 0);
        comps = n;
    }
    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        --comps;
        return true;
    }
};

static CutResult bruteForceCut(int n, const vector<Edge>& edges) {
    CutResult best;
    if (n == 0) return best;
    if (n == 1) {
        best.value = 0;
        best.side.assign(1, 0);
        return best;
    }

    const uint64_t limit = 1ULL << (n - 1);
    long long bestVal = numeric_limits<long long>::max();
    vector<char> bestSide(n, 0);
    vector<char> side(n, 0);
    side[0] = 1;

    for (uint64_t mask = 0; mask < limit; ++mask) {
        for (int i = 1; i < n; ++i) {
            side[i] = (mask >> (i - 1)) & 1ULL;
        }
        long long cut = 0;
        for (const auto& e : edges) {
            if (side[e.u] != side[e.v]) ++cut;
        }
        if (cut < bestVal) {
            bestVal = cut;
            bestSide = side;
        }
    }

    best.value = bestVal;
    best.side = move(bestSide);
    return best;
}

static GraphPack contractToTarget(int n, const vector<Edge>& edges, int target, mt19937_64& rng) {
    DSU dsu(n);
    if (!edges.empty()) {
        uniform_int_distribution<size_t> pick(0, edges.size() - 1);
        while (dsu.comps > target) {
            const Edge& e = edges[pick(rng)];
            dsu.unite(e.u, e.v);
        }
    }

    vector<int> rootToNew(n, -1);
    vector<int> oldToNew(n, -1);
    int newN = 0;
    for (int v = 0; v < n; ++v) {
        int r = dsu.find(v);
        if (rootToNew[r] == -1) rootToNew[r] = newN++;
        oldToNew[v] = rootToNew[r];
    }

    vector<Edge> newEdges;
    newEdges.reserve(edges.size());
    for (const auto& e : edges) {
        int a = oldToNew[e.u];
        int b = oldToNew[e.v];
        if (a != b) newEdges.push_back({a, b});
    }

    GraphPack pack;
    pack.n = newN;
    pack.edges = move(newEdges);
    pack.oldToNew = move(oldToNew);
    return pack;
}

static CutResult liftCut(const CutResult& child, const vector<int>& oldToNew) {
    CutResult res;
    res.value = child.value;
    res.side.assign(oldToNew.size(), 0);
    for (size_t i = 0; i < oldToNew.size(); ++i) {
        res.side[i] = child.side[oldToNew[i]];
    }
    return res;
}

static CutResult kargerStein(int n, const vector<Edge>& edges, mt19937_64& rng);

static CutResult kargerSteinBase(int n, const vector<Edge>& edges, mt19937_64& rng) {
    (void)rng;
    if (n <= 20) return bruteForceCut(n, edges);
    // Small fallback if the recursion bottoms out a bit above the brute-force threshold.
    // This is still manageable because the contracted graph is already tiny.
    return bruteForceCut(n, edges);
}

static CutResult kargerStein(int n, const vector<Edge>& edges, mt19937_64& rng) {
    if (n <= 20) return kargerSteinBase(n, edges, rng);

    int target = max(2, (int)ceil(n / sqrt(2.0)));
    if (target >= n) target = n - 1;

    GraphPack g1 = contractToTarget(n, edges, target, rng);
    GraphPack g2 = contractToTarget(n, edges, target, rng);

    CutResult c1 = kargerStein(g1.n, g1.edges, rng);
    CutResult c2 = kargerStein(g2.n, g2.edges, rng);

    c1 = liftCut(c1, g1.oldToNew);
    c2 = liftCut(c2, g2.oldToNew);
    if (c2.value < c1.value) return c2;
    return c1;
}

static bool findBridge(int n, const vector<vector<pair<int,int>>>& adj, const vector<Edge>& edges, Edge& bridge) {
    vector<int> disc(n, 0), low(n, 0), parent(n, -1), parentEdge(n, -1), it(n, 0);
    int timer = 0;
    vector<int> st;
    st.reserve(n);

    for (int s = 0; s < n; ++s) {
        if (disc[s] != 0) continue;
        st.push_back(s);
        while (!st.empty()) {
            int v = st.back();
            if (disc[v] == 0) {
                disc[v] = low[v] = ++timer;
            }

            if (it[v] < (int)adj[v].size()) {
                auto [to, eid] = adj[v][it[v]++];
                if (eid == parentEdge[v]) continue;
                if (disc[to] == 0) {
                    parent[to] = v;
                    parentEdge[to] = eid;
                    st.push_back(to);
                } else {
                    low[v] = min(low[v], disc[to]);
                }
            } else {
                st.pop_back();
                int p = parent[v];
                if (p != -1) {
                    low[p] = min(low[p], low[v]);
                    if (low[v] > disc[p]) {
                        bridge = edges[parentEdge[v]];
                        return true;
                    }
                }
            }
        }
    }
    return false;
}

static CutResult stoerWagner(int n, const vector<Edge>& edges) {
    vector<vector<long long>> w(n, vector<long long>(n, 0));
    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        ++w[e.u][e.v];
        ++w[e.v][e.u];
    }

    vector<int> active(n);
    iota(active.begin(), active.end(), 0);
    vector<vector<int>> group(n);
    for (int i = 0; i < n; ++i) group[i].push_back(i);

    long long bestVal = numeric_limits<long long>::max();
    vector<char> bestSide(n, 0);

    int activeCount = n;
    while (activeCount > 1) {
        vector<char> used(n, 0);
        vector<long long> weights(n, 0);
        int prev = -1, sel = -1;

        for (int iter = 0; iter < activeCount; ++iter) {
            sel = -1;
            for (int i = 0; i < activeCount; ++i) {
                int v = active[i];
                if (used[v]) continue;
                if (sel == -1 || weights[v] > weights[sel]) sel = v;
            }

            used[sel] = 1;

            if (iter == activeCount - 1) {
                if (weights[sel] < bestVal) {
                    bestVal = weights[sel];
                    fill(bestSide.begin(), bestSide.end(), 0);
                    for (int x : group[sel]) bestSide[x] = 1;
                }

                if (prev != -1) {
                    for (int i = 0; i < activeCount; ++i) {
                        int v = active[i];
                        if (v == sel || v == prev) continue;
                        w[sel][v] += w[prev][v];
                        w[v][sel] = w[sel][v];
                    }
                    group[sel].insert(group[sel].end(), group[prev].begin(), group[prev].end());

                    int posPrev = -1;
                    for (int i = 0; i < activeCount; ++i) {
                        if (active[i] == prev) {
                            posPrev = i;
                            break;
                        }
                    }
                    if (posPrev != -1) {
                        swap(active[posPrev], active[activeCount - 1]);
                        --activeCount;
                    }
                } else {
                    --activeCount;
                }
                break;
            }

            prev = sel;
            for (int i = 0; i < activeCount; ++i) {
                int v = active[i];
                if (!used[v]) weights[v] += w[sel][v];
            }
        }
    }

    CutResult res;
    res.value = bestVal;
    res.side = move(bestSide);
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    FastScanner fs;
    int n, m;
    if (!fs.readInt(n)) return 0;
    fs.readInt(m);

    vector<Edge> edges;
    edges.reserve(m);
    vector<vector<pair<int,int>>> adj(n);
    adj.reserve(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        fs.readInt(u);
        fs.readInt(v);
        edges.push_back({u, v});
        if (u >= 0 && u < n && v >= 0 && v < n) {
            adj[u].push_back({v, i});
            adj[v].push_back({u, i});
        }
    }

    if (n <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    Edge bridge{-1, -1};
    if (findBridge(n, adj, edges, bridge)) {
        cout << 1 << '\n' << bridge.u << ' ' << bridge.v << '\n';
        return 0;
    }

    CutResult ans;
    const int SW_N_LIMIT = 1800;
    const long long SW_MATRIX_LIMIT = 4500000LL; // n^2 around this size
    if (n <= SW_N_LIMIT && 1LL * n * n <= SW_MATRIX_LIMIT) {
        ans = stoerWagner(n, edges);
    } else {
        mt19937_64 rng((uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count());
        int trials = 3;
        if (n > 5000) trials = 4;
        if (n > 20000) trials = 5;
        for (int t = 0; t < trials; ++t) {
            CutResult cur = kargerStein(n, edges, rng);
            if (cur.value < ans.value) ans = move(cur);
        }
        if (ans.side.empty()) {
            // Safety fallback for degenerate cases: output a valid cut.
            ans.side.assign(n, 0);
            ans.side[0] = 1;
            ans.value = 0;
            for (const auto& e : edges) {
                if (ans.side[e.u] != ans.side[e.v]) ++ans.value;
            }
        }
    }

    vector<Edge> cutEdges;
    cutEdges.reserve((size_t)max(1LL, ans.value));
    for (const auto& e : edges) {
        if (ans.side[e.u] != ans.side[e.v]) cutEdges.push_back(e);
    }

    cout << cutEdges.size() << '\n';
    for (const auto& e : cutEdges) {
        cout << e.u << ' ' << e.v << '\n';
    }
    return 0;
}
