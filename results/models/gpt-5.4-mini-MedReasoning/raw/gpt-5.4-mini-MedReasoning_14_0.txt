using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private const byte Unknown = 0;
    private const byte Mine = 1;
    private const byte Safe = 2;

    private sealed class Constraint
    {
        public int Target;
        public int[] Vars = Array.Empty<int>();
    }

    public static void Main()
    {
        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.WriteLine(0);
            return;
        }

        int[] hw = ParseInts(first);
        int h = hw[0];
        int w = hw[1];

        string[] board = new string[h];
        for (int r = 0; r < h; r++)
        {
            string? line = Console.ReadLine();
            if (line == null) line = string.Empty;
            if (line.Length < w) line = line.PadRight(w, ' ');
            else if (line.Length > w) line = line.Substring(0, w);
            board[r] = line;
        }

        int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

        var varIndex = new Dictionary<int, int>();
        var varRows = new List<int>();
        var varCols = new List<int>();
        var constraints = new List<Constraint>();

        for (int r = 0; r < h; r++)
        {
            string row = board[r];
            for (int c = 0; c < w; c++)
            {
                char ch = row[c];
                if (ch < '0' || ch > '8') continue;

                int clue = ch - '0';
                int flags = 0;
                var vars = new List<int>(8);

                for (int k = 0; k < 8; k++)
                {
                    int nr = r + dr[k];
                    int nc = c + dc[k];
                    if ((uint)nr >= (uint)h || (uint)nc >= (uint)w) continue;
                    char adj = board[nr][nc];
                    if (adj == '*')
                    {
                        flags++;
                    }
                    else if (adj == ' ')
                    {
                        int key = nr * w + nc;
                        if (!varIndex.TryGetValue(key, out int id))
                        {
                            id = varRows.Count;
                            varIndex[key] = id;
                            varRows.Add(nr);
                            varCols.Add(nc);
                        }
                        vars.Add(id);
                    }
                }

                if (vars.Count == 0) continue;

                int target = clue - flags;
                if (target < 0 || target > vars.Count)
                {
                    continue;
                }

                constraints.Add(new Constraint
                {
                    Target = target,
                    Vars = vars.ToArray()
                });
            }
        }

        int varCount = varRows.Count;
        int constraintCount = constraints.Count;
        var varToConstraints = new List<int>[varCount];
        for (int i = 0; i < varCount; i++) varToConstraints[i] = new List<int>();
        for (int i = 0; i < constraintCount; i++)
        {
            foreach (int v in constraints[i].Vars)
            {
                varToConstraints[v].Add(i);
            }
        }

        byte[] state = new byte[varCount];
        int[] knownMines = new int[constraintCount];
        int[] unknownCounts = new int[constraintCount];
        for (int i = 0; i < constraintCount; i++) unknownCounts[i] = constraints[i].Vars.Length;

        var queue = new Queue<int>();
        bool conflict = false;

        void EnqueueAssign(int v, byte newState)
        {
            if (conflict) return;
            byte cur = state[v];
            if (cur == newState) return;
            if (cur != Unknown)
            {
                conflict = true;
                return;
            }

            state[v] = newState;
            queue.Enqueue(v);

            foreach (int c in varToConstraints[v])
            {
                unknownCounts[c]--;
                if (newState == Mine) knownMines[c]++;
                if (knownMines[c] > constraints[c].Target || knownMines[c] + unknownCounts[c] < constraints[c].Target)
                {
                    conflict = true;
                    return;
                }
            }
        }

        void Propagate()
        {
            while (queue.Count > 0 && !conflict)
            {
                int v = queue.Dequeue();
                foreach (int c in varToConstraints[v])
                {
                    if (conflict) return;
                    int target = constraints[c].Target;
                    if (knownMines[c] == target)
                    {
                        foreach (int u in constraints[c].Vars)
                        {
                            if (state[u] == Unknown) EnqueueAssign(u, Safe);
                            if (conflict) return;
                        }
                    }
                    else if (knownMines[c] + unknownCounts[c] == target)
                    {
                        foreach (int u in constraints[c].Vars)
                        {
                            if (state[u] == Unknown) EnqueueAssign(u, Mine);
                            if (conflict) return;
                        }
                    }
                }
            }
        }

        for (int i = 0; i < constraintCount; i++)
        {
            if (constraints[i].Target == 0)
            {
                foreach (int v in constraints[i].Vars) EnqueueAssign(v, Safe);
            }
            else if (constraints[i].Target == constraints[i].Vars.Length)
            {
                foreach (int v in constraints[i].Vars) EnqueueAssign(v, Mine);
            }
        }
        Propagate();

        if (!conflict && varCount > 0)
        {
            var visitedVar = new bool[varCount];
            var visitedConstraint = new bool[constraintCount];
            var components = new List<(List<int> Vars, List<int> Constraints)>();

            for (int start = 0; start < varCount; start++)
            {
                if (state[start] != Unknown || visitedVar[start]) continue;

                var compVars = new List<int>();
                var compConstraints = new List<int>();
                var stack = new Stack<(bool IsVar, int Id)>();
                stack.Push((true, start));
                visitedVar[start] = true;

                while (stack.Count > 0)
                {
                    var item = stack.Pop();
                    if (item.IsVar)
                    {
                        int v = item.Id;
                        compVars.Add(v);
                        foreach (int c in varToConstraints[v])
                        {
                            if (unknownCounts[c] == 0 || visitedConstraint[c]) continue;
                            visitedConstraint[c] = true;
                            stack.Push((false, c));
                        }
                    }
                    else
                    {
                        int c = item.Id;
                        compConstraints.Add(c);
                        foreach (int v in constraints[c].Vars)
                        {
                            if (state[v] != Unknown || visitedVar[v]) continue;
                            visitedVar[v] = true;
                            stack.Push((true, v));
                        }
                    }
                }

                components.Add((compVars, compConstraints));
            }

            const int ExactThreshold = 20;
            const int SubsetThreshold = 80;

            foreach (var comp in components)
            {
                if (comp.Vars.Count == 0 || comp.Constraints.Count == 0) continue;

                if (comp.Vars.Count <= ExactThreshold)
                {
                    RunExactComponent(comp.Vars, comp.Constraints, constraints, state, EnqueueAssign);
                    Propagate();
                    if (conflict) break;
                }
                else if (comp.Vars.Count <= SubsetThreshold)
                {
                    RunSubsetComponent(comp.Vars, comp.Constraints, constraints, state, EnqueueAssign);
                    Propagate();
                    if (conflict) break;
                }
            }
        }

        var output = new List<(int X, int Y)>();
        for (int i = 0; i < varCount; i++)
        {
            if (state[i] == Safe)
            {
                output.Add((varCols[i], varRows[i]));
            }
        }

        output.Sort((a, b) => a.Y != b.Y ? a.Y.CompareTo(b.Y) : a.X.CompareTo(b.X));

        using var writer = new StreamWriter(Console.OpenStandardOutput());
        writer.WriteLine(output.Count);
        foreach (var p in output)
        {
            writer.WriteLine($"{p.X} {p.Y}");
        }
    }

    private static void RunSubsetComponent(
        List<int> compVars,
        List<int> compConstraints,
        List<Constraint> constraints,
        byte[] state,
        Action<int, byte> enqueueAssign)
    {
        int n = compVars.Count;
        int m = compConstraints.Count;
        var localOfGlobal = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++) localOfGlobal[compVars[i]] = i;

        var localConstraints = new int[m][];
        var localTargets = new int[m];
        for (int i = 0; i < m; i++)
        {
            Constraint c = constraints[compConstraints[i]];
            var arr = new int[c.Vars.Length];
            int idx = 0;
            foreach (int gv in c.Vars)
            {
                if (state[gv] != Unknown) continue;
                arr[idx++] = localOfGlobal[gv];
            }
            if (idx != arr.Length)
            {
                Array.Resize(ref arr, idx);
            }
            Array.Sort(arr);
            localConstraints[i] = arr;
            localTargets[i] = c.Target;
            foreach (int gv in c.Vars)
            {
                if (state[gv] == Mine) localTargets[i]--;
            }
        }

        var localSafe = new bool[n];
        var localMine = new bool[n];

        int[] order = new int[m];
        for (int i = 0; i < m; i++) order[i] = i;
        Array.Sort(order, (a, b) => localConstraints[a].Length.CompareTo(localConstraints[b].Length));

        for (int oi = 0; oi < m; oi++)
        {
            int i = order[oi];
            int[] a = localConstraints[i];
            for (int oj = oi + 1; oj < m; oj++)
            {
                int j = order[oj];
                int[] b = localConstraints[j];
                if (a.Length > b.Length) continue;

                if (TrySubset(a, b, out int[] diff))
                {
                    int dt = localTargets[j] - localTargets[i];
                    int ds = diff.Length;
                    if (dt == 0)
                    {
                        foreach (int v in diff) localSafe[v] = true;
                    }
                    else if (dt == ds)
                    {
                        foreach (int v in diff) localMine[v] = true;
                    }
                }
            }
        }

        for (int i = 0; i < n; i++)
        {
            if (localSafe[i] && !localMine[i]) enqueueAssign(compVars[i], Safe);
            else if (localMine[i] && !localSafe[i]) enqueueAssign(compVars[i], Mine);
        }
    }

    private static bool TrySubset(int[] small, int[] large, out int[] diff)
    {
        var temp = new List<int>(large.Length - small.Length);
        int i = 0, j = 0;
        while (i < small.Length && j < large.Length)
        {
            if (small[i] == large[j])
            {
                i++;
                j++;
            }
            else if (small[i] > large[j])
            {
                temp.Add(large[j]);
                j++;
            }
            else
            {
                diff = Array.Empty<int>();
                return false;
            }
        }

        if (i < small.Length)
        {
            diff = Array.Empty<int>();
            return false;
        }

        while (j < large.Length)
        {
            temp.Add(large[j]);
            j++;
        }

        diff = temp.ToArray();
        return true;
    }

    private static void RunExactComponent(
        List<int> compVars,
        List<int> compConstraints,
        List<Constraint> constraints,
        byte[] state,
        Action<int, byte> enqueueAssign)
    {
        int n = compVars.Count;
        int m = compConstraints.Count;
        var localOfGlobal = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++) localOfGlobal[compVars[i]] = i;

        var localToGlobal = new int[n];
        for (int i = 0; i < n; i++) localToGlobal[i] = compVars[i];

        var localConstraints = new int[m][];
        var localTargets = new int[m];
        var varToConstraints = new List<int>[n];
        for (int i = 0; i < n; i++) varToConstraints[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            Constraint c = constraints[compConstraints[i]];
            var arr = new int[c.Vars.Length];
            int idx = 0;
            foreach (int gv in c.Vars)
            {
                if (state[gv] != Unknown) continue;
                int lv = localOfGlobal[gv];
                arr[idx++] = lv;
            }
            if (idx != arr.Length) Array.Resize(ref arr, idx);
            Array.Sort(arr);
            localConstraints[i] = arr;
            int target = c.Target;
            foreach (int gv in c.Vars)
            {
                if (state[gv] == Mine) target--;
            }
            localTargets[i] = target;
            foreach (int lv in arr) varToConstraints[lv].Add(i);
        }

        byte[] assign = new byte[n];
        int[] knownMines = new int[m];
        int[] unknownCounts = new int[m];
        for (int i = 0; i < m; i++) unknownCounts[i] = localConstraints[i].Length;
        bool[] canBeSafe = new bool[n];
        bool[] canBeMine = new bool[n];

        bool Assign(int v, byte newState, Queue<int> q, byte[] a, int[] km, int[] uc)
        {
            byte cur = a[v];
            if (cur == newState) return true;
            if (cur != Unknown)
            {
                return false;
            }

            a[v] = newState;
            q.Enqueue(v);
            foreach (int c in varToConstraints[v])
            {
                uc[c]--;
                if (newState == Mine) km[c]++;
                if (km[c] > localTargets[c] || km[c] + uc[c] < localTargets[c])
                {
                    return false;
                }
            }

            return true;
        }

        bool Propagate(byte[] a, int[] km, int[] uc, Queue<int> q)
        {
            while (q.Count > 0)
            {
                int v = q.Dequeue();
                foreach (int c in varToConstraints[v])
                {
                    if (km[c] == localTargets[c])
                    {
                        foreach (int u in localConstraints[c])
                        {
                            if (a[u] == Unknown && !Assign(u, Safe, q, a, km, uc)) return false;
                        }
                    }
                    else if (km[c] + uc[c] == localTargets[c])
                    {
                        foreach (int u in localConstraints[c])
                        {
                            if (a[u] == Unknown && !Assign(u, Mine, q, a, km, uc)) return false;
                        }
                    }
                }
            }

            return true;
        }

        void Search(byte[] a, int[] km, int[] uc, Queue<int> q)
        {
            if (!Propagate(a, km, uc, q)) return;

            int next = -1;
            int bestDegree = -1;
            for (int i = 0; i < n; i++)
            {
                if (a[i] != Unknown) continue;
                int degree = varToConstraints[i].Count;
                if (degree > bestDegree)
                {
                    bestDegree = degree;
                    next = i;
                }
            }

            if (next == -1)
            {
                for (int i = 0; i < n; i++)
                {
                    if (a[i] == Safe) canBeSafe[i] = true;
                    else if (a[i] == Mine) canBeMine[i] = true;
                }
                return;
            }

            {
                var a0 = (byte[])a.Clone();
                var km0 = (int[])km.Clone();
                var uc0 = (int[])uc.Clone();
                var q0 = new Queue<int>();
                if (Assign(next, Safe, q0, a0, km0, uc0)) Search(a0, km0, uc0, q0);
            }

            {
                var a1 = (byte[])a.Clone();
                var km1 = (int[])km.Clone();
                var uc1 = (int[])uc.Clone();
                var q1 = new Queue<int>();
                if (Assign(next, Mine, q1, a1, km1, uc1)) Search(a1, km1, uc1, q1);
            }
        }

        {
            var a = (byte[])assign.Clone();
            var km = (int[])knownMines.Clone();
            var uc = (int[])unknownCounts.Clone();
            var q = new Queue<int>();
            for (int i = 0; i < m; i++)
            {
                if (localTargets[i] == 0)
                {
                    foreach (int v in localConstraints[i]) if (!Assign(v, Safe, q, a, km, uc)) return;
                }
                else if (localTargets[i] == localConstraints[i].Length)
                {
                    foreach (int v in localConstraints[i]) if (!Assign(v, Mine, q, a, km, uc)) return;
                }
            }

            Search(a, km, uc, q);
        }

        for (int i = 0; i < n; i++)
        {
            if (!canBeMine[i] && canBeSafe[i])
            {
                enqueueAssign(localToGlobal[i], Safe);
            }
            else if (canBeMine[i] && !canBeSafe[i])
            {
                enqueueAssign(localToGlobal[i], Mine);
            }
        }
    }

    private static int[] ParseInts(string s)
    {
        var parts = s.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int[] values = new int[parts.Length];
        for (int i = 0; i < parts.Length; i++) values[i] = int.Parse(parts[i]);
        return values;
    }
}
