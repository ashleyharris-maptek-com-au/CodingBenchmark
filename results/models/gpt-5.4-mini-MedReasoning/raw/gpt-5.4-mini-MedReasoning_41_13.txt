OpCapability Shader
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v4float = OpTypePointer Output %v4float

%worldPos = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output

%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%bottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
%top = OpConstantComposite %v3float %float_1 %float_1 %float_1

%main = OpFunction %void None %func
%label = OpLabel
%pos = OpLoad %v3float %worldPos
%y = OpCompositeExtract %float %pos 1
%half_y = OpFMul %float %y %float_0_5
%t_raw = OpFAdd %float %half_y %float_0_5
%t = OpExtInst %float %glsl450 FClamp %t_raw %float_0 %float_1
%diff = OpFSub %v3float %top %bottom
%scaled = OpVectorTimesScalar %v3float %diff %t
%color = OpFAdd %v3float %bottom %scaled
%rgba = OpCompositeConstruct %v4float %color %float_1
OpStore %outColor %rgba
OpReturn
OpFunctionEnd
