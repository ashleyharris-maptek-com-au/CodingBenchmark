; SPIR-V
; Version: 1.0
; Generator: Codex
; Target: Vulkan 1.0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft

OpSource GLSL 450
%glsl450 = OpExtInstImport "GLSL.std.450"

OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"

OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%v2int = OpTypeVector %int 2
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.200000003
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.699999988
%float_0_9 = OpConstant %float 0.899999976
%float_1 = OpConstant %float 1
%float_8 = OpConstant %float 8
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %funct
%entry = OpLabel

%worldPos_val = OpLoad %v3float %worldPos
%normal_val = OpLoad %v3float %normal
%uv_val = OpLoad %v2float %uv

%normal_n = OpExtInst %v3float %glsl450 Normalize %normal_val

%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos_val = OpLoad %v4float %lightPos_ptr
%lightPos_x = OpCompositeExtract %float %lightPos_val 0
%lightPos_y = OpCompositeExtract %float %lightPos_val 1
%lightPos_z = OpCompositeExtract %float %lightPos_val 2
%lightPos_xyz = OpCompositeConstruct %v3float %lightPos_x %lightPos_y %lightPos_z

%lightDir = OpFSub %v3float %lightPos_xyz %worldPos_val
%lightDir_n = OpExtInst %v3float %glsl450 Normalize %lightDir
%diffuse_raw = OpExtInst %float %glsl450 Dot %normal_n %lightDir_n
%diffuse = OpExtInst %float %glsl450 FMax %diffuse_raw %float_0
%diffuse_half = OpFMul %float %float_0_5 %diffuse
%shade = OpFAdd %float %float_0_5 %diffuse_half

%uv_scaled = OpVectorTimesScalar %v2float %uv_val %float_8
%uv_cell = OpExtInst %v2float %glsl450 Floor %uv_scaled
%uv_cell_i = OpConvertFToS %v2int %uv_cell
%cell_x = OpCompositeExtract %int %uv_cell_i 0
%cell_y = OpCompositeExtract %int %uv_cell_i 1
%cell_sum = OpIAdd %int %cell_x %cell_y
%cell_parity = OpBitwiseAnd %int %cell_sum %int_1
%cell_parity_f = OpConvertSToF %float %cell_parity
%parity_term = OpFMul %float %cell_parity_f %float_0_7
%checker = OpFSub %float %float_0_9 %parity_term

%lit = OpFMul %float %checker %shade
%rgb = OpCompositeConstruct %v4float %lit %lit %lit %float_1
OpStore %outColor %rgb

OpReturn
OpFunctionEnd
