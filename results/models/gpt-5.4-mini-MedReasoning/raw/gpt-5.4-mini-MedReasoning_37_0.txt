using System;
using System.Collections.Generic;
using System.IO;

public class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        var g = new List<int>[n];
        var rg = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            g[i] = new List<int>();
            rg[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            g[u].Add(v);
            rg[v].Add(u);
        }

        var order = new int[n];
        int orderSize = 0;
        var seen = new bool[n];
        for (int i = 0; i < n; i++)
        {
            if (seen[i]) continue;
            Dfs1(i, g, seen, order, ref orderSize);
        }

        Array.Fill(seen, false);
        var components = new List<List<int>>();
        for (int i = orderSize - 1; i >= 0; i--)
        {
            int v = order[i];
            if (seen[v]) continue;
            var comp = new List<int>();
            Dfs2(v, rg, seen, comp);
            components.Add(comp);
        }

        var answer = new List<int>();
        var solver = new Solver();
        foreach (var comp in components)
        {
            if (comp.Count == 1)
            {
                int v = comp[0];
                bool selfLoop = false;
                foreach (var to in g[v])
                {
                    if (to == v)
                    {
                        selfLoop = true;
                        break;
                    }
                }
                if (selfLoop)
                    answer.Add(v);
                continue;
            }

            var sub = BuildSubgraph(comp, g, rg);
            var removed = solver.Solve(sub);
            answer.AddRange(removed);
        }

        var sw = new StreamWriter(Console.OpenStandardOutput());
        sw.WriteLine(answer.Count);
        if (answer.Count > 0)
        {
            for (int i = 0; i < answer.Count; i++)
            {
                if (i > 0) sw.Write(' ');
                sw.Write(answer[i]);
            }
            sw.WriteLine();
        }
        else
        {
            sw.WriteLine();
        }
        sw.Flush();
    }

    private static void Dfs1(int start, List<int>[] g, bool[] seen, int[] order, ref int orderSize)
    {
        var stackV = new int[g.Length];
        var stackI = new int[g.Length];
        int sp = 0;
        stackV[sp] = start;
        stackI[sp] = 0;
        sp++;
        seen[start] = true;
        while (sp > 0)
        {
            int v = stackV[sp - 1];
            int idx = stackI[sp - 1];
            if (idx < g[v].Count)
            {
                int to = g[v][idx];
                stackI[sp - 1] = idx + 1;
                if (!seen[to])
                {
                    seen[to] = true;
                    stackV[sp] = to;
                    stackI[sp] = 0;
                    sp++;
                }
            }
            else
            {
                order[orderSize++] = v;
                sp--;
            }
        }
    }

    private static void Dfs2(int start, List<int>[] rg, bool[] seen, List<int> comp)
    {
        var stack = new int[rg.Length];
        int sp = 0;
        stack[sp++] = start;
        seen[start] = true;
        while (sp > 0)
        {
            int v = stack[--sp];
            comp.Add(v);
            for (int i = 0; i < rg[v].Count; i++)
            {
                int to = rg[v][i];
                if (!seen[to])
                {
                    seen[to] = true;
                    stack[sp++] = to;
                }
            }
        }
    }

    private static Subgraph BuildSubgraph(List<int> comp, List<int>[] g, List<int>[] rg)
    {
        int k = comp.Count;
        var map = new int[g.Length];
        Array.Fill(map, -1);
        for (int i = 0; i < k; i++) map[comp[i]] = i;

        var adj = new int[k][];
        var radj = new int[k][];
        for (int i = 0; i < k; i++)
        {
            int u = comp[i];
            var outs = new List<int>();
            for (int j = 0; j < g[u].Count; j++)
            {
                int v = g[u][j];
                int mv = map[v];
                if (mv >= 0) outs.Add(mv);
            }
            adj[i] = outs.ToArray();

            var ins = new List<int>();
            for (int j = 0; j < rg[u].Count; j++)
            {
                int v = rg[u][j];
                int mv = map[v];
                if (mv >= 0) ins.Add(mv);
            }
            radj[i] = ins.ToArray();
        }

        var global = new int[k];
        for (int i = 0; i < k; i++) global[i] = comp[i];
        return new Subgraph(adj, radj, global);
    }
}

public sealed class Solver
{
    public List<int> Solve(Subgraph sub)
    {
        var alive = new bool[sub.N];
        for (int i = 0; i < sub.N; i++) alive[i] = true;
        return SolveState(sub, alive);
    }

    private List<int> SolveState(Subgraph sub, bool[] alive)
    {
        var forced = new List<int>();
        Reduce(sub, alive, forced);

        if (forced.Count == 0 && IsAcyclic(sub, alive))
            return ConvertToGlobal(sub, forced);

        var comps = GetNonTrivialSccs(sub, alive);
        if (comps.Count == 0)
            return ConvertToGlobal(sub, forced);

        if (comps.Count > 1)
        {
            var result = ConvertToGlobal(sub, forced);
            for (int i = 0; i < comps.Count; i++)
            {
                var child = BuildSubgraph(sub, comps[i]);
                var childAlive = new bool[child.N];
                for (int j = 0; j < child.N; j++) childAlive[j] = true;
                var childSol = SolveState(child, childAlive);
                for (int j = 0; j < childSol.Count; j++) result.Add(childSol[j]);
            }
            return result;
        }

        var active = comps[0];
        var heuristic = GreedyUpperBound(sub, alive);
        var best = ConvertToGlobal(sub, forced);
        for (int i = 0; i < heuristic.Count; i++)
            best.Add(sub.Global[heuristic[i]]);

        int bestSize = best.Count;
        var cycle = FindCycle(sub, alive, active);
        if (cycle == null)
            return best;

        var order = OrderCycleVertices(sub, alive, cycle);
        for (int i = 0; i < order.Count; i++)
        {
            int v = order[i];
            var childAlive = CopyAlive(alive);
            childAlive[v] = false;

            int lb = LowerBound(sub, childAlive, bestSize - forced.Count - 1);
            if (forced.Count + 1 + lb >= bestSize)
                continue;

            var child = SolveState(sub, childAlive);
            int candSize = forced.Count + 1 + child.Count;
            if (candSize < bestSize)
            {
                var cand = new List<int>(candSize);
                cand.AddRange(ConvertToGlobal(sub, forced));
                cand.Add(sub.Global[v]);
                cand.AddRange(child);
                best = cand;
                bestSize = candSize;
            }
        }

        return best;
    }

    private static List<int> ConvertToGlobal(Subgraph sub, List<int> local)
    {
        var result = new List<int>(local.Count);
        for (int i = 0; i < local.Count; i++)
            result.Add(sub.Global[local[i]]);
        return result;
    }

    private static bool[] CopyAlive(bool[] alive)
    {
        var copy = new bool[alive.Length];
        Array.Copy(alive, copy, alive.Length);
        return copy;
    }

    private void Reduce(Subgraph sub, bool[] alive, List<int> forced)
    {
        while (true)
        {
            bool changed = false;
            var indeg = new int[sub.N];
            var outdeg = new int[sub.N];
            var selfLoop = new bool[sub.N];

            for (int v = 0; v < sub.N; v++)
            {
                if (!alive[v]) continue;
                var adj = sub.Adj[v];
                for (int i = 0; i < adj.Length; i++)
                {
                    int to = adj[i];
                    if (!alive[to]) continue;
                    outdeg[v]++;
                    indeg[to]++;
                    if (to == v) selfLoop[v] = true;
                }
            }

            for (int v = 0; v < sub.N; v++)
            {
                if (!alive[v]) continue;
                if (selfLoop[v])
                {
                    alive[v] = false;
                    forced.Add(v);
                    changed = true;
                }
                else if (indeg[v] == 0 || outdeg[v] == 0)
                {
                    alive[v] = false;
                    changed = true;
                }
            }

            if (!changed) break;
        }
    }

    private bool IsAcyclic(Subgraph sub, bool[] alive)
    {
        return FindCycle(sub, alive, null) == null;
    }

    private List<List<int>> GetNonTrivialSccs(Subgraph sub, bool[] alive)
    {
        int n = sub.N;
        var seen = new bool[n];
        var order = new int[n];
        int ord = 0;

        for (int i = 0; i < n; i++)
        {
            if (alive[i] && !seen[i])
                DfsOrder(sub, alive, i, seen, order, ref ord);
        }

        Array.Fill(seen, false);
        var comps = new List<List<int>>();
        for (int i = ord - 1; i >= 0; i--)
        {
            int v = order[i];
            if (!alive[v] || seen[v]) continue;
            var comp = new List<int>();
            DfsCollectReverse(sub, alive, v, seen, comp);
            if (comp.Count > 1)
                comps.Add(comp);
        }

        return comps;
    }

    private static void DfsOrder(Subgraph sub, bool[] alive, int start, bool[] seen, int[] order, ref int ord)
    {
        var stackV = new int[sub.N];
        var stackI = new int[sub.N];
        int sp = 0;
        stackV[sp] = start;
        stackI[sp] = 0;
        sp++;
        seen[start] = true;
        while (sp > 0)
        {
            int v = stackV[sp - 1];
            int idx = stackI[sp - 1];
            var adj = sub.Adj[v];
            bool advanced = false;
            while (idx < adj.Length)
            {
                int to = adj[idx++];
                if (!alive[to] || seen[to]) continue;
                seen[to] = true;
                stackI[sp - 1] = idx;
                stackV[sp] = to;
                stackI[sp] = 0;
                sp++;
                advanced = true;
                break;
            }
            if (!advanced)
            {
                order[ord++] = v;
                sp--;
            }
        }
    }

    private static void DfsCollectReverse(Subgraph sub, bool[] alive, int start, bool[] seen, List<int> comp)
    {
        var stack = new int[sub.N];
        int sp = 0;
        stack[sp++] = start;
        seen[start] = true;
        while (sp > 0)
        {
            int v = stack[--sp];
            comp.Add(v);
            var adj = sub.RAdj[v];
            for (int i = 0; i < adj.Length; i++)
            {
                int to = adj[i];
                if (!alive[to] || seen[to]) continue;
                seen[to] = true;
                stack[sp++] = to;
            }
        }
    }

    private static Subgraph BuildSubgraph(Subgraph sub, List<int> verts)
    {
        int k = verts.Count;
        var map = new int[sub.N];
        Array.Fill(map, -1);
        for (int i = 0; i < k; i++) map[verts[i]] = i;

        var adj = new int[k][];
        var radj = new int[k][];
        for (int i = 0; i < k; i++)
        {
            int u = verts[i];
            var outs = new List<int>();
            var a = sub.Adj[u];
            for (int j = 0; j < a.Length; j++)
            {
                int v = a[j];
                int mv = map[v];
                if (mv >= 0) outs.Add(mv);
            }
            adj[i] = outs.ToArray();

            var ins = new List<int>();
            var ra = sub.RAdj[u];
            for (int j = 0; j < ra.Length; j++)
            {
                int v = ra[j];
                int mv = map[v];
                if (mv >= 0) ins.Add(mv);
            }
            radj[i] = ins.ToArray();
        }

        var global = new int[k];
        for (int i = 0; i < k; i++) global[i] = sub.Global[verts[i]];
        return new Subgraph(adj, radj, global);
    }

    private List<int>? FindCycle(Subgraph sub, bool[] alive, List<int>? restrictTo)
    {
        int n = sub.N;
        var inSet = new bool[n];
        if (restrictTo != null)
        {
            for (int i = 0; i < restrictTo.Count; i++) inSet[restrictTo[i]] = true;
        }

        var color = new byte[n];
        var parent = new int[n];
        Array.Fill(parent, -1);
        var stackV = new int[n];
        var stackI = new int[n];
        int sp = 0;

        for (int s = 0; s < n; s++)
        {
            if (!alive[s] || color[s] != 0) continue;
            if (restrictTo != null && !inSet[s]) continue;

            stackV[sp] = s;
            stackI[sp] = 0;
            sp++;
            color[s] = 1;
            parent[s] = -1;

            while (sp > 0)
            {
                int v = stackV[sp - 1];
                int idx = stackI[sp - 1];
                var adj = sub.Adj[v];
                bool advanced = false;
                while (idx < adj.Length)
                {
                    int to = adj[idx++];
                    if (!alive[to]) continue;
                    if (restrictTo != null && !inSet[to]) continue;
                    if (color[to] == 0)
                    {
                        color[to] = 1;
                        parent[to] = v;
                        stackI[sp - 1] = idx;
                        stackV[sp] = to;
                        stackI[sp] = 0;
                        sp++;
                        advanced = true;
                        break;
                    }
                    if (color[to] == 1)
                    {
                        var cycle = new List<int>();
                        int cur = v;
                        cycle.Add(to);
                        while (cur != to && cur != -1)
                        {
                            cycle.Add(cur);
                            cur = parent[cur];
                        }
                        if (cur == to)
                        {
                            cycle.Reverse();
                            return cycle;
                        }
                    }
                }
                if (!advanced)
                {
                    color[v] = 2;
                    sp--;
                }
            }
        }
        return null;
    }

    private List<int> GreedyUpperBound(Subgraph sub, bool[] alive)
    {
        var temp = CopyAlive(alive);
        var answer = new List<int>();
        while (true)
        {
            var cycle = FindCycle(sub, temp, null);
            if (cycle == null) break;

            int pick = cycle[0];
            int bestScore = -1;
            for (int i = 0; i < cycle.Count; i++)
            {
                int v = cycle[i];
                int score = DegreeScore(sub, temp, v);
                if (score > bestScore)
                {
                    bestScore = score;
                    pick = v;
                }
            }
            temp[pick] = false;
            answer.Add(pick);
        }
        return answer;
    }

    private int LowerBound(Subgraph sub, bool[] alive, int threshold)
    {
        var temp = CopyAlive(alive);
        int count = 0;
        while (count < threshold)
        {
            var cycle = FindCycle(sub, temp, null);
            if (cycle == null) break;
            count++;
            for (int i = 0; i < cycle.Count; i++)
                temp[cycle[i]] = false;
        }
        return count;
    }

    private static int DegreeScore(Subgraph sub, bool[] alive, int v)
    {
        int score = 0;
        var a = sub.Adj[v];
        for (int i = 0; i < a.Length; i++)
            if (alive[a[i]]) score++;
        var r = sub.RAdj[v];
        for (int i = 0; i < r.Length; i++)
            if (alive[r[i]]) score++;
        return score;
    }

    private List<int> OrderCycleVertices(Subgraph sub, bool[] alive, List<int> cycle)
    {
        var order = new List<(int v, int score)>(cycle.Count);
        for (int i = 0; i < cycle.Count; i++)
            order.Add((cycle[i], DegreeScore(sub, alive, cycle[i])));
        order.Sort((a, b) => b.score.CompareTo(a.score));
        var res = new List<int>(cycle.Count);
        for (int i = 0; i < order.Count; i++) res.Add(order[i].v);
        return res;
    }
}

public sealed class Subgraph
{
    public readonly int N;
    public readonly int[][] Adj;
    public readonly int[][] RAdj;
    public readonly int[] Global;

    public Subgraph(int[][] adj, int[][] radj, int[] global)
    {
        Adj = adj;
        RAdj = radj;
        Global = global;
        N = adj.Length;
    }
}

public sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _len;
    private int _ptr;

    public FastScanner(Stream stream)
    {
        _stream = stream;
    }

    private byte Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len == 0) return 0;
        }
        return _buffer[_ptr++];
    }

    public int NextInt()
    {
        byte c;
        do
        {
            c = Read();
        } while (c <= 32 && c != 0);

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int val = 0;
        while (c > 32)
        {
            val = val * 10 + (c - (byte)'0');
            c = Read();
        }
        return val * sign;
    }
}
