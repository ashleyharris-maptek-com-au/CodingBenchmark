use std::cmp::Ordering;
use std::io::{self, Read};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut num = 0usize;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            num = num * 10 + (self.buf[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        num
    }
}

fn lower_bound(
    idx: usize,
    used_a: usize,
    used_b: usize,
    current_cut: i64,
    order: &[usize],
    a_count: &[i32],
    b_count: &[i32],
    k: usize,
    n: usize,
) -> i64 {
    let rem_a = k - used_a;
    let rem_b = k - used_b;
    if idx == order.len() {
        return current_cut;
    }

    let mut freq = vec![0i32; 2 * n + 1];
    let offset = n as i32;
    let mut base = 0i64;

    if rem_a <= rem_b {
        for &u in &order[idx..] {
            base += a_count[u] as i64;
            let d = b_count[u] - a_count[u];
            freq[(d + offset) as usize] += 1;
        }
        let mut need = rem_a as i32;
        let mut add = 0i64;
        for (i, &cnt) in freq.iter().enumerate() {
            if need == 0 {
                break;
            }
            let take = need.min(cnt);
            add += (i as i32 - offset) as i64 * take as i64;
            need -= take;
        }
        current_cut + base + add
    } else {
        for &u in &order[idx..] {
            base += b_count[u] as i64;
            let d = a_count[u] - b_count[u];
            freq[(d + offset) as usize] += 1;
        }
        let mut need = rem_b as i32;
        let mut add = 0i64;
        for (i, &cnt) in freq.iter().enumerate() {
            if need == 0 {
                break;
            }
            let take = need.min(cnt);
            add += (i as i32 - offset) as i64 * take as i64;
            need -= take;
        }
        current_cut + base + add
    }
}

fn assign_vertex(
    v: usize,
    side_value: u8,
    adj: &[Vec<usize>],
    side: &mut [u8],
    a_count: &mut [i32],
    b_count: &mut [i32],
) -> (i64, Vec<usize>) {
    let mut added = 0i64;
    let mut touched = Vec::new();
    side[v] = side_value;
    for &w in &adj[v] {
        if side[w] == 0 {
            if side_value == 1 {
                a_count[w] += 1;
            } else {
                b_count[w] += 1;
            }
            touched.push(w);
        } else if side[w] != side_value {
            added += 1;
        }
    }
    (added, touched)
}

fn undo_assignment(
    v: usize,
    side_value: u8,
    touched: Vec<usize>,
    side: &mut [u8],
    a_count: &mut [i32],
    b_count: &mut [i32],
) {
    for w in touched {
        if side_value == 1 {
            a_count[w] -= 1;
        } else {
            b_count[w] -= 1;
        }
    }
    side[v] = 0;
}

fn dfs(
    idx: usize,
    used_a: usize,
    used_b: usize,
    current_cut: i64,
    order: &[usize],
    adj: &[Vec<usize>],
    side: &mut [u8],
    a_count: &mut [i32],
    b_count: &mut [i32],
    best_cut: &mut i64,
    best_side: &mut [u8],
    k: usize,
    n: usize,
) {
    let lb = lower_bound(idx, used_a, used_b, current_cut, order, a_count, b_count, k, n);
    if lb >= *best_cut {
        return;
    }

    if idx == order.len() {
        if current_cut < *best_cut {
            *best_cut = current_cut;
            best_side.copy_from_slice(side);
        }
        return;
    }

    let v = order[idx];
    let can_a = used_a < k;
    let can_b = used_b < k;

    if can_a && can_b {
        let a_cost = b_count[v] as i64;
        let b_cost = a_count[v] as i64;
        match a_cost.cmp(&b_cost) {
            Ordering::Less | Ordering::Equal => {
                let (add_a, touched_a) = assign_vertex(v, 1, adj, side, a_count, b_count);
                dfs(
                    idx + 1,
                    used_a + 1,
                    used_b,
                    current_cut + add_a,
                    order,
                    adj,
                    side,
                    a_count,
                    b_count,
                    best_cut,
                    best_side,
                    k,
                    n,
                );
                undo_assignment(v, 1, touched_a, side, a_count, b_count);

                let (add_b, touched_b) = assign_vertex(v, 2, adj, side, a_count, b_count);
                dfs(
                    idx + 1,
                    used_a,
                    used_b + 1,
                    current_cut + add_b,
                    order,
                    adj,
                    side,
                    a_count,
                    b_count,
                    best_cut,
                    best_side,
                    k,
                    n,
                );
                undo_assignment(v, 2, touched_b, side, a_count, b_count);
            }
            Ordering::Greater => {
                let (add_b, touched_b) = assign_vertex(v, 2, adj, side, a_count, b_count);
                dfs(
                    idx + 1,
                    used_a,
                    used_b + 1,
                    current_cut + add_b,
                    order,
                    adj,
                    side,
                    a_count,
                    b_count,
                    best_cut,
                    best_side,
                    k,
                    n,
                );
                undo_assignment(v, 2, touched_b, side, a_count, b_count);

                let (add_a, touched_a) = assign_vertex(v, 1, adj, side, a_count, b_count);
                dfs(
                    idx + 1,
                    used_a + 1,
                    used_b,
                    current_cut + add_a,
                    order,
                    adj,
                    side,
                    a_count,
                    b_count,
                    best_cut,
                    best_side,
                    k,
                    n,
                );
                undo_assignment(v, 1, touched_a, side, a_count, b_count);
            }
        }
    } else if can_a {
        let (add_a, touched_a) = assign_vertex(v, 1, adj, side, a_count, b_count);
        dfs(
            idx + 1,
            used_a + 1,
            used_b,
            current_cut + add_a,
            order,
            adj,
            side,
            a_count,
            b_count,
            best_cut,
            best_side,
            k,
            n,
        );
        undo_assignment(v, 1, touched_a, side, a_count, b_count);
    } else {
        let (add_b, touched_b) = assign_vertex(v, 2, adj, side, a_count, b_count);
        dfs(
            idx + 1,
            used_a,
            used_b + 1,
            current_cut + add_b,
            order,
            adj,
            side,
            a_count,
            b_count,
            best_cut,
            best_side,
            k,
            n,
        );
        undo_assignment(v, 2, touched_b, side, a_count, b_count);
    }
}

fn greedy_solution(order: &[usize], adj: &[Vec<usize>], k: usize, n: usize) -> (i64, Vec<u8>) {
    let mut side = vec![0u8; n];
    let mut a_count = vec![0i32; n];
    let mut b_count = vec![0i32; n];
    let mut used_a = 0usize;
    let mut used_b = 0usize;
    let mut cut = 0i64;

    let start = order[0];
    side[start] = 1;
    used_a += 1;
    for &w in &adj[start] {
        if side[w] == 0 {
            a_count[w] += 1;
        }
    }

    for &v in &order[1..] {
        let can_a = used_a < k;
        let can_b = used_b < k;
        let choose_a = if can_a && can_b {
            let a_cost = b_count[v];
            let b_cost = a_count[v];
            a_cost <= b_cost
        } else {
            can_a
        };

        if choose_a {
            side[v] = 1;
            used_a += 1;
            cut += b_count[v] as i64;
            for &w in &adj[v] {
                if side[w] == 0 {
                    a_count[w] += 1;
                } else if side[w] == 2 {
                    cut += 1;
                }
            }
        } else {
            side[v] = 2;
            used_b += 1;
            cut += a_count[v] as i64;
            for &w in &adj[v] {
                if side[w] == 0 {
                    b_count[w] += 1;
                } else if side[w] == 1 {
                    cut += 1;
                }
            }
        }
    }

    (cut, side)
}

fn main() {
    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        adj[u].push(v);
        adj[v].push(u);
    }

    let k = n / 2;
    let mut order: Vec<usize> = (0..n).collect();
    order[1..].sort_by(|&a, &b| {
        let da = adj[a].len();
        let db = adj[b].len();
        db.cmp(&da).then_with(|| a.cmp(&b))
    });

    let (mut best_cut, greedy_side) = greedy_solution(&order, &adj, k, n);

    let mut side = vec![0u8; n];
    let mut a_count = vec![0i32; n];
    let mut b_count = vec![0i32; n];

    let start = order[0];
    side[start] = 1;
    for &w in &adj[start] {
        if side[w] == 0 {
            a_count[w] += 1;
        }
    }

    let mut best_side = greedy_side;

    dfs(
        1,
        1,
        0,
        0,
        &order,
        &adj,
        &mut side,
        &mut a_count,
        &mut b_count,
        &mut best_cut,
        &mut best_side,
        k,
        n,
    );

    let mut part_a = Vec::new();
    for v in 0..n {
        if best_side[v] == 1 {
            part_a.push(v);
        }
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", best_cut));
    for (i, v) in part_a.iter().enumerate() {
        if i > 0 {
            out.push(' ');
        }
        out.push_str(&v.to_string());
    }
    out.push('\n');
    print!("{}", out);
}
