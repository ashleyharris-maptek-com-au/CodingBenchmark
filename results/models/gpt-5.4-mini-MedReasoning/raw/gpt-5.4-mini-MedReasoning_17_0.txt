using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private struct Box
    {
        public int Index;
        public long A;
        public long B;
        public long C;
        public long Sum;
        public long Max;
    }

    private struct Space
    {
        public long X;
        public long Y;
        public long Z;
        public long W;
        public long H;
        public long D;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                {
                    return 0;
                }
            }

            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do
            {
                c = Read();
            } while (c <= 32 && c != 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - (byte)'0');
                c = Read();
            }

            return value * sign;
        }

        public long NextLong()
        {
            byte c;
            do
            {
                c = Read();
            } while (c <= 32 && c != 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - (byte)'0');
                c = Read();
            }

            return value * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        long containerW = fs.NextLong();
        long containerH = fs.NextLong();
        long containerD = fs.NextLong();

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            long w = fs.NextLong();
            long h = fs.NextLong();
            long d = fs.NextLong();
            boxes[i] = new Box
            {
                Index = i,
                A = w,
                B = h,
                C = d,
                Sum = w + h + d,
                Max = Math.Max(w, Math.Max(h, d))
            };
        }

        Array.Sort(boxes, (l, r) =>
        {
            int cmp = l.Sum.CompareTo(r.Sum);
            if (cmp != 0) return cmp;
            cmp = r.Max.CompareTo(l.Max);
            if (cmp != 0) return cmp;
            return l.Index.CompareTo(r.Index);
        });

        var freeSpaces = new List<Space>(Math.Max(4, n * 2 + 1))
        {
            new Space { X = 0, Y = 0, Z = 0, W = containerW, H = containerH, D = containerD }
        };

        var packed = new List<(int Index, long X, long Y, long Z)>(n);

        for (int bi = 0; bi < n; bi++)
        {
            Box box = boxes[bi];

            Sort3Values(box.A, box.B, box.C, out long b0, out long b1, out long b2);

            int bestSpaceIndex = -1;
            long bestScore0 = long.MaxValue;
            long bestScore1 = long.MaxValue;
            long bestScore2 = long.MaxValue;
            int bestO0 = 0, bestO1 = 1, bestO2 = 2;
            long bestEx = 0, bestEy = 0, bestEz = 0;

            for (int si = 0; si < freeSpaces.Count; si++)
            {
                Space sp = freeSpaces[si];

                Sort3Axes(sp.W, sp.H, sp.D, out int s0, out int s1, out int s2, out long f0, out long f1, out long f2);

                if (b0 > f0 || b1 > f1 || b2 > f2)
                {
                    continue;
                }

                long ex = 0, ey = 0, ez = 0;
                AssignAxis(ref ex, ref ey, ref ez, s0, b0);
                AssignAxis(ref ex, ref ey, ref ez, s1, b1);
                AssignAxis(ref ex, ref ey, ref ez, s2, b2);

                long slackX = sp.W - ex;
                long slackY = sp.H - ey;
                long slackZ = sp.D - ez;

                long maxSlack = slackX;
                if (slackY > maxSlack) maxSlack = slackY;
                if (slackZ > maxSlack) maxSlack = slackZ;

                long minSlack = slackX;
                if (slackY < minSlack) minSlack = slackY;
                if (slackZ < minSlack) minSlack = slackZ;

                long midSlack = slackX + slackY + slackZ - maxSlack - minSlack;

                if (bestSpaceIndex < 0 ||
                    maxSlack < bestScore0 ||
                    (maxSlack == bestScore0 && (midSlack < bestScore1 ||
                    (midSlack == bestScore1 && minSlack < bestScore2))))
                {
                    bestSpaceIndex = si;
                    bestScore0 = maxSlack;
                    bestScore1 = midSlack;
                    bestScore2 = minSlack;
                    bestO0 = AxisOfLargestSlack(slackX, slackY, slackZ, 0, 1, 2, out bestO1, out bestO2);
                    bestEx = ex;
                    bestEy = ey;
                    bestEz = ez;

                    if (bestScore0 == 0 && bestScore1 == 0 && bestScore2 == 0)
                    {
                        break;
                    }
                }
            }

            if (bestSpaceIndex < 0)
            {
                continue;
            }

            Space chosen = freeSpaces[bestSpaceIndex];
            int lastIndex = freeSpaces.Count - 1;
            freeSpaces[bestSpaceIndex] = freeSpaces[lastIndex];
            freeSpaces.RemoveAt(lastIndex);

            packed.Add((box.Index, chosen.X, chosen.Y, chosen.Z));

            AddChildren(freeSpaces, chosen, bestEx, bestEy, bestEz, bestO0, bestO1, bestO2);
        }

        var sb = new StringBuilder();
        sb.AppendLine(packed.Count.ToString());
        for (int i = 0; i < packed.Count; i++)
        {
            var p = packed[i];
            sb.Append(p.Index).Append(' ').Append(p.X).Append(' ').Append(p.Y).Append(' ').Append(p.Z).Append('\n');
        }

        Console.Write(sb.ToString());
    }

    private static void Sort3Values(long a, long b, long c, out long v0, out long v1, out long v2)
    {
        v0 = a;
        v1 = b;
        v2 = c;

        if (v0 > v1)
        {
            (v0, v1) = (v1, v0);
        }

        if (v1 > v2)
        {
            (v1, v2) = (v2, v1);
        }

        if (v0 > v1)
        {
            (v0, v1) = (v1, v0);
        }
    }

    private static void Sort3Axes(long a, long b, long c, out int i0, out int i1, out int i2, out long v0, out long v1, out long v2)
    {
        i0 = 0;
        i1 = 1;
        i2 = 2;
        v0 = a;
        v1 = b;
        v2 = c;

        if (v0 > v1)
        {
            (v0, v1) = (v1, v0);
            (i0, i1) = (i1, i0);
        }

        if (v1 > v2)
        {
            (v1, v2) = (v2, v1);
            (i1, i2) = (i2, i1);
        }

        if (v0 > v1)
        {
            (v0, v1) = (v1, v0);
            (i0, i1) = (i1, i0);
        }
    }

    private static void AssignAxis(ref long ex, ref long ey, ref long ez, int axis, long value)
    {
        if (axis == 0)
        {
            ex = value;
        }
        else if (axis == 1)
        {
            ey = value;
        }
        else
        {
            ez = value;
        }
    }

    private static int AxisOfLargestSlack(long sx, long sy, long sz, int a0, int a1, int a2, out int o1, out int o2)
    {
        int o0 = a0;
        o1 = a1;
        o2 = a2;

        long v0 = sx;
        long v1 = sy;
        long v2 = sz;

        if (v0 < v1)
        {
            (v0, v1) = (v1, v0);
            (o0, o1) = (o1, o0);
        }

        if (v1 < v2)
        {
            (v1, v2) = (v2, v1);
            (o1, o2) = (o2, o1);
        }

        if (v0 < v1)
        {
            (v0, v1) = (v1, v0);
            (o0, o1) = (o1, o0);
        }

        return o0;
    }

    private static void AddChildren(List<Space> freeSpaces, Space parent, long bx, long by, long bz, int o0, int o1, int o2)
    {
        // First slab: consume the first axis in the chosen order.
        {
            long sx = parent.X;
            long sy = parent.Y;
            long sz = parent.Z;
            long sw = parent.W;
            long sh = parent.H;
            long sd = parent.D;

            if (o0 == 0)
            {
                sx += bx;
                sw -= bx;
            }
            else if (o0 == 1)
            {
                sy += by;
                sh -= by;
            }
            else
            {
                sz += bz;
                sd -= bz;
            }

            if (sw > 0 && sh > 0 && sd > 0)
            {
                freeSpaces.Add(new Space { X = sx, Y = sy, Z = sz, W = sw, H = sh, D = sd });
            }
        }

        // Second slab: within the prefix of the first axis, consume the second axis.
        {
            long sx = parent.X;
            long sy = parent.Y;
            long sz = parent.Z;
            long sw = parent.W;
            long sh = parent.H;
            long sd = parent.D;

            if (o0 == 0)
            {
                sw = bx;
            }
            else if (o0 == 1)
            {
                sh = by;
            }
            else
            {
                sd = bz;
            }

            if (o1 == 0)
            {
                sx += bx;
                sw -= bx;
            }
            else if (o1 == 1)
            {
                sy += by;
                sh -= by;
            }
            else
            {
                sz += bz;
                sd -= bz;
            }

            if (sw > 0 && sh > 0 && sd > 0)
            {
                freeSpaces.Add(new Space { X = sx, Y = sy, Z = sz, W = sw, H = sh, D = sd });
            }
        }

        // Third slab: within the prefix of the first two axes, consume the third axis.
        {
            long sx = parent.X;
            long sy = parent.Y;
            long sz = parent.Z;
            long sw = parent.W;
            long sh = parent.H;
            long sd = parent.D;

            if (o0 == 0)
            {
                sw = bx;
            }
            else if (o0 == 1)
            {
                sh = by;
            }
            else
            {
                sd = bz;
            }

            if (o1 == 0)
            {
                sw = bx;
            }
            else if (o1 == 1)
            {
                sh = by;
            }
            else
            {
                sd = bz;
            }

            if (o2 == 0)
            {
                sx += bx;
                sw -= bx;
            }
            else if (o2 == 1)
            {
                sy += by;
                sh -= by;
            }
            else
            {
                sz += bz;
                sd -= bz;
            }

            if (sw > 0 && sh > 0 && sd > 0)
            {
                freeSpaces.Add(new Space { X = sx, Y = sy, Z = sz, W = sw, H = sh, D = sd });
            }
        }
    }
}
