import math
import sys
from copy import deepcopy


def read_input():
    data = sys.stdin.buffer.read().split()
    if not data:
        return None
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = float(next(it))
    depot_x = float(next(it))
    depot_y = float(next(it))
    xs = [depot_x]
    ys = [depot_y]
    demand = [0.0]
    for _ in range(n):
        xs.append(float(next(it)))
        ys.append(float(next(it)))
        demand.append(float(next(it)))
    return n, k, cap, xs, ys, demand


def build_dist(xs, ys):
    n = len(xs)
    d = [[0.0] * n for _ in range(n)]
    for i in range(n):
        xi = xs[i]
        yi = ys[i]
        row = d[i]
        for j in range(i + 1, n):
            val = math.hypot(xi - xs[j], yi - ys[j])
            row[j] = val
            d[j][i] = val
    return d


def route_cost(seq, d):
    if not seq:
        return 0.0
    total = d[0][seq[0]]
    for a, b in zip(seq, seq[1:]):
        total += d[a][b]
    total += d[seq[-1]][0]
    return total


def nn_order(customers, d):
    remaining = set(customers)
    seq = []
    cur = 0
    while remaining:
        nxt = min(remaining, key=lambda x: (d[cur][x], x))
        remaining.remove(nxt)
        seq.append(nxt)
        cur = nxt
    return seq


def two_opt(seq, d):
    if len(seq) < 4:
        return seq
    improved = True
    while improved:
        improved = False
        m = len(seq)
        for i in range(m - 2):
            a = 0 if i == 0 else seq[i - 1]
            b = seq[i]
            for j in range(i + 2, m):
                c = seq[j]
                e = 0 if j == m - 1 else seq[j + 1]
                delta = d[a][c] + d[b][e] - d[a][b] - d[c][e]
                if delta < -1e-12:
                    seq[i : j + 1] = reversed(seq[i : j + 1])
                    improved = True
                    break
            if improved:
                break
    return seq


def build_state(routes, demand, d, n):
    loads = [0.0] * len(routes)
    costs = [0.0] * len(routes)
    active = [False] * len(routes)
    cust_route = [0] * (n + 1)
    for idx, seq in enumerate(routes):
        if seq:
            active[idx] = True
            loads[idx] = sum(demand[c] for c in seq)
            costs[idx] = route_cost(seq, d)
            for c in seq:
                cust_route[c] = idx
    return {
        "routes": routes,
        "loads": loads,
        "costs": costs,
        "active": active,
        "cust_route": cust_route,
    }


def active_ids(state):
    return [i for i, a in enumerate(state["active"]) if a and state["routes"][i]]


def active_count(state):
    return sum(1 for i in active_ids(state))


def best_insertion_pos(seq, c, d):
    if not seq:
        return 0, 2.0 * d[0][c]
    best_pos = 0
    best_delta = float("inf")
    for pos in range(len(seq) + 1):
        left = 0 if pos == 0 else seq[pos - 1]
        right = 0 if pos == len(seq) else seq[pos]
        delta = d[left][c] + d[c][right] - d[left][right]
        if delta < best_delta:
            best_delta = delta
            best_pos = pos
    return best_pos, best_delta


def removal_delta(seq, pos, d):
    c = seq[pos]
    left = 0 if pos == 0 else seq[pos - 1]
    right = 0 if pos + 1 == len(seq) else seq[pos + 1]
    return d[left][c] + d[c][right] - d[left][right]


def apply_move(state, ai, apos, bi, bpos, demand, d):
    routes = state["routes"]
    loads = state["loads"]
    costs = state["costs"]
    active = state["active"]
    cust_route = state["cust_route"]

    c = routes[ai].pop(apos)
    loads[ai] -= demand[c]
    routes[bi].insert(bpos, c)
    loads[bi] += demand[c]
    cust_route[c] = bi

    costs[ai] = route_cost(routes[ai], d) if routes[ai] else 0.0
    costs[bi] = route_cost(routes[bi], d)
    if not routes[ai]:
        active[ai] = False


def apply_swap(state, ai, apos, bi, bpos, demand, d):
    routes = state["routes"]
    loads = state["loads"]
    costs = state["costs"]
    cust_route = state["cust_route"]

    ca = routes[ai][apos]
    cb = routes[bi][bpos]
    routes[ai][apos] = cb
    routes[bi][bpos] = ca
    loads[ai] += demand[cb] - demand[ca]
    loads[bi] += demand[ca] - demand[cb]
    cust_route[ca] = bi
    cust_route[cb] = ai
    costs[ai] = route_cost(routes[ai], d)
    costs[bi] = route_cost(routes[bi], d)


def merge_routes(seq_a, seq_b, d):
    candidates = [
        seq_a + seq_b,
        seq_a + seq_b[::-1],
        seq_a[::-1] + seq_b,
        seq_a[::-1] + seq_b[::-1],
    ]
    best_seq = None
    best_cost = float("inf")
    for cand in candidates:
        cost = route_cost(cand, d)
        if cost < best_cost:
            best_cost = cost
            best_seq = cand
    return best_seq, best_cost


def merge_into(state, keep, drop, merged_seq, merged_cost, demand):
    routes = state["routes"]
    loads = state["loads"]
    costs = state["costs"]
    active = state["active"]
    cust_route = state["cust_route"]

    routes[keep] = merged_seq
    loads[keep] += loads[drop]
    costs[keep] = merged_cost
    active[drop] = False
    loads[drop] = 0.0
    costs[drop] = 0.0
    for c in merged_seq:
        cust_route[c] = keep


def cw_initial(n, k, cap, demand, d):
    routes = [[i] for i in range(1, n + 1)]
    state = build_state(routes, demand, d, n)
    savings = []
    for i in range(1, n + 1):
        di = d[0][i]
        for j in range(i + 1, n + 1):
            savings.append((di + d[0][j] - d[i][j], i, j))
    savings.sort(reverse=True)

    for _, i, j in savings:
        ri = state["cust_route"][i]
        rj = state["cust_route"][j]
        if ri == rj or not state["active"][ri] or not state["active"][rj]:
            continue
        if state["loads"][ri] + state["loads"][rj] > cap + 1e-12:
            continue
        seq_a = state["routes"][ri]
        seq_b = state["routes"][rj]
        merged = None
        if seq_a and seq_b:
            if seq_a[-1] == i and seq_b[0] == j:
                merged = seq_a + seq_b
            elif seq_a[-1] == i and seq_b[-1] == j:
                merged = seq_a + seq_b[::-1]
            elif seq_a[0] == i and seq_b[0] == j:
                merged = seq_a[::-1] + seq_b
            elif seq_a[0] == i and seq_b[-1] == j:
                merged = seq_a[::-1] + seq_b[::-1]
        if merged is not None:
            merge_into(state, ri, rj, merged, route_cost(merged, d), demand)

    state = compress_to_k(state, k, cap, demand, d)
    state = optimize_state(state, k, cap, demand, d)
    return state


def bfd_initial(n, k, cap, demand, d):
    order = list(range(1, n + 1))
    order.sort(key=lambda x: (-demand[x], -d[0][x], x))
    routes = []
    loads = []
    for c in order:
        best_idx = None
        best_rem = None
        for idx, load in enumerate(loads):
            if load + demand[c] <= cap + 1e-12:
                rem = cap - (load + demand[c])
                if best_idx is None or rem < best_rem:
                    best_idx = idx
                    best_rem = rem
        if best_idx is None:
            routes.append([c])
            loads.append(demand[c])
        else:
            routes[best_idx].append(c)
            loads[best_idx] += demand[c]

    for i in range(len(routes)):
        routes[i] = nn_order(routes[i], d)

    state = build_state(routes, demand, d, n)
    state = compress_to_k(state, k, cap, demand, d)
    state = optimize_state(state, k, cap, demand, d)
    return state


def try_eliminate_route(state, rid, cap, demand, d):
    if not state["active"][rid] or not state["routes"][rid]:
        return None

    trial = {
        "routes": [seq[:] for seq in state["routes"]],
        "loads": state["loads"][:],
        "costs": state["costs"][:],
        "active": state["active"][:],
        "cust_route": state["cust_route"][:],
    }

    customers = sorted(trial["routes"][rid], key=lambda c: (-demand[c], c))
    for c in customers:
        from_r = trial["cust_route"][c]
        if from_r != rid:
            continue
        pos = trial["routes"][from_r].index(c)
        rem = removal_delta(trial["routes"][from_r], pos, d)
        best = None
        best_delta = float("inf")
        for to_r in active_ids(trial):
            if to_r == rid:
                continue
            if trial["loads"][to_r] + demand[c] > cap + 1e-12:
                continue
            ins_pos, ins_delta = best_insertion_pos(trial["routes"][to_r], c, d)
            delta = ins_delta - rem
            if delta < best_delta:
                best_delta = delta
                best = (to_r, ins_pos)
        if best is None:
            return None
        apply_move(trial, from_r, pos, best[0], best[1], demand, d)

    trial["routes"][rid] = []
    trial["loads"][rid] = 0.0
    trial["costs"][rid] = 0.0
    trial["active"][rid] = False
    return trial


def compress_to_k(state, k, cap, demand, d):
    while active_count(state) > k:
        changed = False
        candidates = sorted(active_ids(state), key=lambda i: (state["loads"][i], len(state["routes"][i]), i))
        for rid in candidates:
            trial = try_eliminate_route(state, rid, cap, demand, d)
            if trial is not None:
                state = trial
                changed = True
                break
        if not changed:
            break
    return state


def optimize_state(state, k, cap, demand, d):
    routes = state["routes"]
    for rid in active_ids(state):
        routes[rid] = two_opt(routes[rid], d)
        state["costs"][rid] = route_cost(routes[rid], d)

    limit = max(1, len(demand) * 2)
    for _ in range(limit):
        improved = False
        ids = active_ids(state)

        best_delta = -1e-12
        best_move = None
        for ai in ids:
            seq_a = state["routes"][ai]
            if not seq_a:
                continue
            for apos, c in enumerate(seq_a):
                rem = removal_delta(seq_a, apos, d)
                for bi in ids:
                    if bi == ai:
                        continue
                    if state["loads"][bi] + demand[c] > cap + 1e-12:
                        continue
                    bpos, ins = best_insertion_pos(state["routes"][bi], c, d)
                    delta = ins - rem
                    if delta < best_delta:
                        best_delta = delta
                        best_move = (ai, apos, bi, bpos)

        if best_move is not None:
            apply_move(state, best_move[0], best_move[1], best_move[2], best_move[3], demand, d)
            rid_a = best_move[0]
            rid_b = best_move[2]
            if state["active"][rid_a]:
                routes[rid_a] = two_opt(routes[rid_a], d)
                state["costs"][rid_a] = route_cost(routes[rid_a], d)
            routes[rid_b] = two_opt(routes[rid_b], d)
            state["costs"][rid_b] = route_cost(routes[rid_b], d)
            improved = True
            continue

        best_delta = -1e-12
        best_swap = None
        ids = active_ids(state)
        for idx_a, ai in enumerate(ids):
            seq_a = state["routes"][ai]
            for idx_b in range(idx_a + 1, len(ids)):
                bi = ids[idx_b]
                seq_b = state["routes"][bi]
                for apos, ca in enumerate(seq_a):
                    for bpos, cb in enumerate(seq_b):
                        new_load_a = state["loads"][ai] - demand[ca] + demand[cb]
                        new_load_b = state["loads"][bi] - demand[cb] + demand[ca]
                        if new_load_a > cap + 1e-12 or new_load_b > cap + 1e-12:
                            continue
                        la = 0 if apos == 0 else seq_a[apos - 1]
                        ra = 0 if apos + 1 == len(seq_a) else seq_a[apos + 1]
                        lb = 0 if bpos == 0 else seq_b[bpos - 1]
                        rb = 0 if bpos + 1 == len(seq_b) else seq_b[bpos + 1]
                        delta_a = d[la][cb] + d[cb][ra] - d[la][ca] - d[ca][ra]
                        delta_b = d[lb][ca] + d[ca][rb] - d[lb][cb] - d[cb][rb]
                        delta = delta_a + delta_b
                        if delta < best_delta:
                            best_delta = delta
                            best_swap = (ai, apos, bi, bpos)

        if best_swap is not None:
            apply_swap(state, best_swap[0], best_swap[1], best_swap[2], best_swap[3], demand, d)
            for rid in {best_swap[0], best_swap[2]}:
                routes[rid] = two_opt(routes[rid], d)
                state["costs"][rid] = route_cost(routes[rid], d)
            improved = True

        if not improved:
            break

    state = compress_to_k(state, k, cap, demand, d)
    for rid in active_ids(state):
        routes[rid] = two_opt(routes[rid], d)
        state["costs"][rid] = route_cost(routes[rid], d)
    return state


def total_cost(state):
    return sum(state["costs"][i] for i in active_ids(state))


def extract_routes(state):
    res = []
    for i in active_ids(state):
        if state["routes"][i]:
            res.append(state["routes"][i])
    return res


def solve():
    parsed = read_input()
    if parsed is None:
        return
    n, k, cap, xs, ys, demand = parsed
    if n == 0:
        sys.stdout.write("0.000000\n0\n")
        return

    d = build_dist(xs, ys)

    cw = cw_initial(n, k, cap, demand, d)
    bfd = bfd_initial(n, k, cap, demand, d)

    cand = []
    for state in (cw, bfd):
        routes = extract_routes(state)
        cost = total_cost(state)
        cand.append((len(routes), cost, routes))

    valid = [x for x in cand if x[0] <= k]
    if valid:
        valid.sort(key=lambda x: (x[1], x[0]))
        best_routes = valid[0][2]
        best_cost = valid[0][1]
    else:
        cand.sort(key=lambda x: (x[0], x[1]))
        best_routes = cand[0][2]
        best_cost = cand[0][1]

    # Final safety pass: if the chosen solution still exceeds K, try compressing once more.
    if len(best_routes) > k:
        state = build_state([r[:] for r in best_routes], demand, d, n)
        state = compress_to_k(state, k, cap, demand, d)
        state = optimize_state(state, k, cap, demand, d)
        best_routes = extract_routes(state)
        best_cost = total_cost(state)

    out = [f"{best_cost:.6f}", str(len(best_routes))]
    for route in best_routes:
        out.append(" ".join(str(c - 1) for c in route))
    sys.stdout.write("\n".join(out) + "\n")


if __name__ == "__main__":
    solve()
