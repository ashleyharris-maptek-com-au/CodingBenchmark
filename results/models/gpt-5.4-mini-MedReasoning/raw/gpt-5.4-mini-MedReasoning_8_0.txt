use std::io::{self, BufRead, BufReader, Write};

fn main() {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    let mut first_line = Vec::new();
    if reader.read_until(b'\n', &mut first_line).unwrap() == 0 {
        return;
    }
    while matches!(first_line.last(), Some(b'\n' | b'\r')) {
        first_line.pop();
    }

    let first = String::from_utf8(first_line).unwrap();
    let mut it = first.split_whitespace();
    let h: usize = it.next().unwrap().parse().unwrap();
    let w: usize = it.next().unwrap().parse().unwrap();
    let total = h.checked_mul(w).expect("grid too large");

    // Cell encoding:
    // 254 = wall
    // 255 = unvisited open cell
    // 252 = start cell
    // 0..3 = parent direction to reach the previous cell
    let mut cells = vec![255u8; total];
    let mut start = None::<usize>;
    let mut goal = None::<usize>;

    for r in 0..h {
        let mut line = Vec::new();
        reader.read_until(b'\n', &mut line).unwrap();
        while matches!(line.last(), Some(b'\n' | b'\r')) {
            line.pop();
        }

        let base = r * w;
        for c in 0..w {
            let idx = base + c;
            let ch = *line.get(c).unwrap_or(&b' ');
            cells[idx] = match ch {
                b'#' => 254,
                b'A' => {
                    start = Some(idx);
                    252
                }
                b'B' => {
                    goal = Some(idx);
                    255
                }
                b' ' => 255,
                _ => 254,
            };
        }
    }

    let start = match start {
        Some(v) => v,
        None => {
            let mut out = io::BufWriter::new(io::stdout().lock());
            writeln!(out, "0").unwrap();
            return;
        }
    };
    let goal = match goal {
        Some(v) => v,
        None => {
            let mut out = io::BufWriter::new(io::stdout().lock());
            writeln!(out, "0").unwrap();
            return;
        }
    };

    if start == goal {
        let mut out = io::BufWriter::new(io::stdout().lock());
        writeln!(out, "1").unwrap();
        writeln!(out, "{} {}", start / w, start % w).unwrap();
        return;
    }

    let mut queue = Vec::<u32>::new();
    queue.push(start as u32);
    let mut head = 0usize;
    let mut found = false;

    while head < queue.len() {
        let cur = queue[head] as usize;
        head += 1;
        if cur == goal {
            found = true;
            break;
        }

        let r = cur / w;
        let c = cur % w;

        if r > 0 {
            let ni = cur - w;
            if cells[ni] == 255 {
                cells[ni] = 1; // parent is below
                queue.push(ni as u32);
            }
        }
        if r + 1 < h {
            let ni = cur + w;
            if cells[ni] == 255 {
                cells[ni] = 0; // parent is above
                queue.push(ni as u32);
            }
        }
        if c > 0 {
            let ni = cur - 1;
            if cells[ni] == 255 {
                cells[ni] = 3; // parent is right
                queue.push(ni as u32);
            }
        }
        if c + 1 < w {
            let ni = cur + 1;
            if cells[ni] == 255 {
                cells[ni] = 2; // parent is left
                queue.push(ni as u32);
            }
        }
    }

    let mut out = io::BufWriter::new(io::stdout().lock());
    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut path = Vec::<usize>::new();
    let mut cur = goal;
    loop {
        path.push(cur);
        if cur == start {
            break;
        }
        cur = match cells[cur] {
            0 => cur - w,
            1 => cur + w,
            2 => cur - 1,
            3 => cur + 1,
            _ => {
                writeln!(out, "0").unwrap();
                return;
            }
        };
    }

    path.reverse();
    writeln!(out, "{}", path.len()).unwrap();
    for idx in path {
        writeln!(out, "{} {}", idx / w, idx % w).unwrap();
    }
}
