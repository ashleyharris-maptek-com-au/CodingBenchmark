import math
from statistics import median


_STATE = {
    "speed_ema_kmh": None,
    "offset_ema_m": None,
    "yaw_rate_ema_dps": 0.0,
    "heading_ema_deg": None,
    "speed_integral": 0.0,
    "lane_integral": 0.0,
    "prev_speed_err": 0.0,
    "prev_lane_err": 0.0,
    "prev_heading_deg": None,
    "prev_cmd": {"steering": 0.0, "throttle": 0.0, "brake": 0.0},
    "evasive_lane": None,
    "evasive_timer": 0.0,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _finite_number(x):
    try:
        x = float(x)
    except (TypeError, ValueError):
        return None
    if math.isfinite(x):
        return x
    return None


def _median_or(values, default=None):
    vals = [v for v in values if v is not None]
    if not vals:
        return default
    return float(median(vals))


def _wrap_deg(angle):
    if angle is None:
        return None
    return (angle + 180.0) % 360.0 - 180.0


def _sigmoid(x):
    if x >= 0:
        z = math.exp(-x)
        return 1.0 / (1.0 + z)
    z = math.exp(x)
    return z / (1.0 + z)


def _read_numeric(state, keys, lo=None, hi=None):
    vals = []
    for key in keys:
        v = _finite_number(state.get(key))
        if v is None:
            continue
        if lo is not None and v < lo:
            continue
        if hi is not None and v > hi:
            continue
        vals.append(v)
    return _median_or(vals)


def _read_boolish(state, keys):
    vals = []
    for key in keys:
        v = state.get(key)
        if isinstance(v, bool):
            vals.append(1.0 if v else 0.0)
            continue
        fv = _finite_number(v)
        if fv is None:
            continue
        vals.append(1.0 if fv >= 0.5 else 0.0)
    return _median_or(vals)


def _fuse_speed_kmh(state):
    wheel = _read_numeric(
        state,
        [
            "wheel_speed_fl_kmh",
            "wheel_speed_fr_kmh",
            "wheel_speed_rl_kmh",
            "wheel_speed_rr_kmh",
        ],
        0.0,
        350.0,
    )
    gps = _read_numeric(state, ["gps_1_speed_kmh", "gps_2_speed_kmh"], 0.0, 350.0)
    v1 = _read_numeric(state, ["vehicle_speed_1_kmh"], 0.0, 350.0)
    v2 = _read_numeric(state, ["vehicle_speed_2_kmh"], 0.0, 350.0)
    candidates = [wheel, gps, v1, v2]
    fused = _median_or(candidates)
    if fused is None:
        fused = _STATE["speed_ema_kmh"]
    if fused is None:
        fused = 0.0
    prev = _STATE["speed_ema_kmh"]
    alpha = 0.45 if prev is not None else 1.0
    fused = fused if prev is None else prev + alpha * (fused - prev)
    _STATE["speed_ema_kmh"] = fused
    return fused


def _fuse_yaw_rate_dps(state, dt):
    gyros = _read_numeric(
        state,
        [
            "gyro_1_yaw_rate_dps",
            "gyro_2_yaw_rate_dps",
            "gyro_3_yaw_rate_dps",
        ],
        -250.0,
        250.0,
    )
    if gyros is None:
        gyros = 0.0

    heading = _read_numeric(
        state,
        ["gps_1_heading_deg", "gps_2_heading_deg"],
        -3600.0,
        3600.0,
    )
    if heading is not None and _STATE["prev_heading_deg"] is not None and dt > 1e-6:
        delta = _wrap_deg(heading - _STATE["prev_heading_deg"])
        heading_rate = delta / dt
        gyros = 0.7 * gyros + 0.3 * heading_rate
    if heading is not None:
        _STATE["prev_heading_deg"] = heading
        prev_h = _STATE["heading_ema_deg"]
        _STATE["heading_ema_deg"] = heading if prev_h is None else prev_h + 0.3 * _wrap_deg(heading - prev_h)

    prev = _STATE["yaw_rate_ema_dps"]
    fused = prev + 0.35 * (gyros - prev)
    _STATE["yaw_rate_ema_dps"] = fused
    return fused


def _fuse_lane_offset_m(state):
    offsets = []
    for i in (1, 2, 3):
        off = _finite_number(state.get(f"vision_{i}_lane_offset_m"))
        if off is not None and abs(off) < 10.0:
            offsets.append(off)
        left = _finite_number(state.get(f"vision_{i}_left_line_dist_m"))
        right = _finite_number(state.get(f"vision_{i}_right_line_dist_m"))
        if left is not None and right is not None and 0.0 <= left < 20.0 and 0.0 <= right < 20.0:
            offsets.append(0.5 * (left - right))
    fused = _median_or(offsets)
    if fused is None:
        fused = _STATE["offset_ema_m"]
    if fused is None:
        fused = 0.0
    prev = _STATE["offset_ema_m"]
    alpha = 0.25 if prev is not None else 1.0
    fused = fused if prev is None else prev + alpha * (fused - prev)
    _STATE["offset_ema_m"] = fused
    return fused


def _fuse_curvature_1pm(state):
    vals = []
    for i in (1, 2, 3):
        v = _finite_number(state.get(f"vision_{i}_road_curvature_1pm"))
        if v is not None and abs(v) < 0.5:
            vals.append(v)
    return _median_or(vals, 0.0)


def _fuse_speed_limit_kmh(state):
    vals = []
    for i in (1, 2, 3):
        v = _finite_number(state.get(f"vision_{i}_speed_limit_kmh"))
        if v is not None and 20.0 <= v <= 220.0:
            vals.append(v)
    return _median_or(vals)


def _fuse_mu(state):
    mu = _finite_number(state.get("estimated_mu"))
    if mu is None:
        mu = 0.75
    return _clamp(mu, 0.05, 1.2)


def _front_obstacle_distance_m(state):
    distances = []
    for i in (1, 2, 3):
        detected = state.get(f"vision_{i}_obstacle_detected")
        valid = state.get(f"vision_{i}_obstacle_valid")
        dist = _finite_number(state.get(f"vision_{i}_obstacle_dist_m"))
        if dist is not None and 0.0 < dist < 200.0:
            if detected is True or valid is True or isinstance(detected, bool):
                distances.append(dist)
    front_angles = ["+005", "-005", "+015", "-015", "+030", "-030"]
    for ang in front_angles:
        dist = _finite_number(state.get(f"lidar_{ang}_deg_range_m"))
        if dist is not None and 0.2 < dist < 250.0:
            distances.append(dist)
    return _median_or(distances)


def _side_clearance_m(state, side):
    if side == "left":
        angles = ["+030", "+045", "+060", "+090"]
    else:
        angles = ["-030", "-045", "-060", "-090"]
    vals = []
    for ang in angles:
        dist = _finite_number(state.get(f"lidar_{ang}_deg_range_m"))
        if dist is not None and 0.2 < dist < 250.0:
            vals.append(dist)
    return _median_or(vals)


def _steering_feedback_norm(state):
    measured = _read_numeric(
        state,
        ["steering_angle_1_deg", "steering_angle_2_deg"],
        -90.0,
        90.0,
    )
    if measured is None:
        return None
    return _clamp(measured / 35.0, -1.5, 1.5)


def autopilot_step(state: dict, dt: float) -> dict:
    if not isinstance(dt, (int, float)) or not math.isfinite(float(dt)) or dt <= 0.0:
        dt = _finite_number(state.get("dt")) or 0.1
    dt = max(1e-3, float(dt))

    speed_kmh = _fuse_speed_kmh(state)
    yaw_rate_dps = _fuse_yaw_rate_dps(state, dt)
    lane_offset_m = _fuse_lane_offset_m(state)
    curvature_1pm = _fuse_curvature_1pm(state)
    speed_limit_kmh = _fuse_speed_limit_kmh(state)
    mu = _fuse_mu(state)

    rain = 1.0 if bool(state.get("rain_sensor")) else 0.0
    temperature_c = _finite_number(state.get("temperature_c"))
    if temperature_c is None:
        temperature_c = 15.0

    current_lane = int(_clamp(_finite_number(state.get("current_lane")) or 1.0, 0.0, 2.0))
    target_lane = int(_clamp(_finite_number(state.get("target_lane")) or current_lane, 0.0, 2.0))

    vision_count = 0
    for i in (1, 2, 3):
        lane_off = _finite_number(state.get(f"vision_{i}_lane_offset_m"))
        left_valid = _finite_number(state.get(f"vision_{i}_left_line_valid"))
        right_valid = _finite_number(state.get(f"vision_{i}_right_line_valid"))
        obs_valid = _finite_number(state.get(f"vision_{i}_obstacle_valid"))
        sl_valid = _finite_number(state.get(f"vision_{i}_speed_limit_valid"))
        if any(v is not None for v in (lane_off, left_valid, right_valid, obs_valid, sl_valid)):
            vision_count += 1

    # Speed planning
    grip_speed_cap = 25.0 + 75.0 * mu
    curvature_cap = 140.0 / (1.0 + 35.0 * abs(curvature_1pm))
    visibility_factor = 1.0 if vision_count >= 2 else 0.85 if vision_count == 1 else 0.72
    weather_factor = 0.92 if rain > 0.5 else 1.0
    temp_factor = 0.9 if temperature_c < 2.0 else 1.0
    speed_cap = min(grip_speed_cap, curvature_cap) * visibility_factor * weather_factor * temp_factor
    desired_speed = speed_cap
    if speed_limit_kmh is not None:
        desired_speed = min(desired_speed, speed_limit_kmh)
    target_speed_kmh = _finite_number(state.get("target_speed_kmh"))
    if target_speed_kmh is not None and target_speed_kmh > 0.0:
        desired_speed = min(desired_speed, target_speed_kmh)

    front_dist = _front_obstacle_distance_m(state)
    front_rel_speed = _read_numeric(
        state,
        [
            "vision_1_obstacle_rel_speed_kmh",
            "vision_2_obstacle_rel_speed_kmh",
            "vision_3_obstacle_rel_speed_kmh",
        ],
        -250.0,
        250.0,
    )

    evasive_lane = _STATE["evasive_lane"]
    evasive_timer = _STATE["evasive_timer"]
    if evasive_timer > 0.0:
        evasive_timer = max(0.0, evasive_timer - dt)
    if evasive_timer <= 0.0:
        evasive_lane = None

    obstacle_close = front_dist is not None and front_dist < 30.0
    emergency = front_dist is not None and front_dist < 8.0

    if obstacle_close and mu < 0.35:
        # On very low grip, prioritize stability over lane changes.
        desired_speed = min(desired_speed, 40.0)

    if front_dist is not None:
        decel = 2.0 + 5.5 * mu
        if front_dist < 60.0:
            safe_speed_ms = math.sqrt(max(0.0, 2.0 * decel * max(0.0, front_dist - 5.0)))
            safe_speed_kmh = safe_speed_ms * 3.6
            desired_speed = min(desired_speed, safe_speed_kmh)
        if front_rel_speed is not None and front_rel_speed < 0.0:
            desired_speed = min(desired_speed, max(0.0, speed_kmh + 0.7 * front_rel_speed))

    side_left = _side_clearance_m(state, "left")
    side_right = _side_clearance_m(state, "right")
    left_clear = side_left is None or side_left > 10.0
    right_clear = side_right is None or side_right > 10.0

    desired_lane = target_lane
    if obstacle_close:
        # Prefer an open adjacent lane when the lead gap is shrinking.
        if current_lane > 0 and left_clear:
            desired_lane = min(desired_lane, current_lane - 1)
            evasive_lane = current_lane - 1
            evasive_timer = max(evasive_timer, 2.5)
        elif current_lane < 2 and right_clear:
            desired_lane = max(desired_lane, current_lane + 1)
            evasive_lane = current_lane + 1
            evasive_timer = max(evasive_timer, 2.5)
        elif left_clear and not right_clear:
            desired_lane = max(0, current_lane - 1)
            evasive_lane = desired_lane
            evasive_timer = max(evasive_timer, 2.5)
        elif right_clear and not left_clear:
            desired_lane = min(2, current_lane + 1)
            evasive_lane = desired_lane
            evasive_timer = max(evasive_timer, 2.5)

    if evasive_lane is not None:
        desired_lane = evasive_lane
    _STATE["evasive_lane"] = evasive_lane
    _STATE["evasive_timer"] = evasive_timer

    lane_delta = float(desired_lane - current_lane)
    lane_error = lane_offset_m
    lane_error_int = _STATE["lane_integral"] + lane_error * dt
    _STATE["lane_integral"] = _clamp(lane_error_int, -3.0, 3.0)
    lane_deriv = 0.0 if dt <= 1e-6 else (lane_error - _STATE["prev_lane_err"]) / dt
    _STATE["prev_lane_err"] = lane_error

    # Base steering from lane centering, lane change intent, curvature, and yaw damping.
    steer_lane = -1.15 * lane_error - 0.18 * _STATE["lane_integral"] - 0.18 * lane_deriv
    steer_change = 0.33 * lane_delta
    steer_curve = 3.4 * curvature_1pm
    steer_damp = -0.012 * yaw_rate_dps

    if mu < 0.35:
        steer_lane *= 0.6
        steer_change *= 0.55
        steer_curve *= 0.75
        steer_damp *= 0.8

    if obstacle_close and not emergency:
        steer_change *= 1.25

    steer_cmd = steer_lane + steer_change + steer_curve + steer_damp

    measured_steer = _steering_feedback_norm(state)
    if measured_steer is not None:
        # Compensate for partial steering jam or actuator lag.
        steer_cmd += 0.55 * (steer_cmd - measured_steer)

    steer_limit = _clamp(0.30 + 0.50 * mu, 0.30, 0.85)
    steer_cmd = _clamp(steer_cmd, -steer_limit, steer_limit)
    steer_cmd = 0.72 * _STATE["prev_cmd"]["steering"] + 0.28 * steer_cmd
    steer_cmd = _clamp(steer_cmd, -steer_limit, steer_limit)

    # Speed control.
    speed_err = desired_speed - speed_kmh
    _STATE["speed_integral"] = _clamp(_STATE["speed_integral"] + speed_err * dt, -250.0, 250.0)
    speed_deriv = (speed_err - _STATE["prev_speed_err"]) / dt
    _STATE["prev_speed_err"] = speed_err

    throttle = 0.0
    brake = 0.0
    if emergency:
        brake = 1.0
        throttle = 0.0
    else:
        if speed_err >= 0.0:
            throttle = 0.018 * speed_err + 0.004 * _STATE["speed_integral"] + 0.002 * speed_deriv
            throttle = _clamp(throttle, 0.0, 1.0)
        else:
            brake = _clamp((-speed_err) * 0.03 + (-speed_deriv) * 0.0015, 0.0, 1.0)
        if front_dist is not None and front_dist < 20.0:
            brake = max(brake, _clamp((20.0 - front_dist) / 12.0, 0.0, 1.0))
            throttle = min(throttle, 0.15)
        if front_dist is not None and front_dist < 12.0:
            brake = max(brake, 0.8)
            throttle = 0.0
        if front_dist is not None and front_dist < 6.0:
            brake = 1.0
            throttle = 0.0

    throttle_limit = _clamp(0.35 + 0.65 * mu, 0.35, 1.0)
    throttle = _clamp(throttle, 0.0, throttle_limit)
    if brake > 0.0:
        throttle = min(throttle, 0.25 * (1.0 - brake))

    # Stability handling in low grip or poor visibility.
    if mu < 0.25 or visibility_factor < 0.8:
        throttle = min(throttle, 0.45 if mu >= 0.25 else 0.28)
        brake = max(brake, 0.0)

    # Use actual brake pressure readback only as a hint; if brakes are weak, stay conservative.
    brake_fb = _read_numeric(state, ["brake_pressure_1_pct", "brake_pressure_2_pct"], 0.0, 100.0)
    if brake_fb is not None and brake_fb > 70.0 and brake < 0.2 and speed_err < -2.0:
        brake = min(1.0, brake + 0.2)

    cmd = {
        "steering": float(_clamp(steer_cmd, -1.0, 1.0)),
        "throttle": float(_clamp(throttle, 0.0, 1.0)),
        "brake": float(_clamp(brake, 0.0, 1.0)),
    }

    # Light command smoothing for flickering sensor conditions.
    prev = _STATE["prev_cmd"]
    cmd["steering"] = 0.65 * prev["steering"] + 0.35 * cmd["steering"]
    cmd["throttle"] = 0.60 * prev["throttle"] + 0.40 * cmd["throttle"]
    cmd["brake"] = 0.55 * prev["brake"] + 0.45 * cmd["brake"]

    cmd["steering"] = float(_clamp(cmd["steering"], -1.0, 1.0))
    cmd["throttle"] = float(_clamp(cmd["throttle"], 0.0, 1.0))
    cmd["brake"] = float(_clamp(cmd["brake"], 0.0, 1.0))

    if cmd["brake"] > 0.15:
        cmd["throttle"] = min(cmd["throttle"], 0.12)

    _STATE["prev_cmd"] = cmd.copy()
    return cmd
