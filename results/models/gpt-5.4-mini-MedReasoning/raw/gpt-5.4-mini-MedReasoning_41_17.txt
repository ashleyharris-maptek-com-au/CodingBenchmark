%c_f_0 = OpConstant %float 0.0

%c_v2_scale = OpConstantComposite %v2float %c_f_8 %c_f_16
%c_v3_mortar = OpConstantComposite %v3float %c_f_07 %c_f_07 %c_f_07
%c_v3_brick = OpConstantComposite %v3float %c_f_07 %c_f_02 %c_f_01

%main = OpFunction %void None %funct
%label = OpLabel

%worldPos = OpLoad %v3float %inWorldPos
%normalIn = OpLoad %v3float %inNormal
%uv = OpLoad %v2float %inUV

%normal = OpExtInst %v3float %1 Normalize %normalIn

%lightPosPtr = OpAccessChain %ptrUniformV4 %ubo %int_3
%lightPos = OpLoad %v4float %lightPosPtr
%lightPosXYZ = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2
%lightVec = OpFSub %v3float %lightPosXYZ %worldPos
%lightDir = OpExtInst %v3float %1 Normalize %lightVec
%diffuseDot = OpExtInst %float %1 Dot %normal %lightDir
%diffuse = OpExtInst %float %1 FMax %diffuseDot %c_f_0
%shadeMul = OpFMul %float %c_f_06 %diffuse
%shade = OpFAdd %float %c_f_04 %shadeMul

%scaledUV = OpFMul %v2float %uv %c_v2_scale
%scaledX = OpCompositeExtract %float %scaledUV 0
%scaledY = OpCompositeExtract %float %scaledUV 1
%row = OpExtInst %float %1 Floor %scaledY
%rowMod = OpExtInst %float %1 FMod %row %c_f_2
%isOdd = OpFOrdGreaterThan %bool %rowMod %c_f_05
%rowOffset = OpSelect %float %isOdd %c_f_05 %c_f_0
%xOffset = OpFAdd %float %scaledX %rowOffset
%brickX = OpExtInst %float %1 Fract %xOffset
%brickY = OpExtInst %float %1 Fract %scaledY

%xLow = OpFOrdLessThan %bool %brickX %c_f_005
%xHigh = OpFOrdGreaterThan %bool %brickX %c_f_095
%yLow = OpFOrdLessThan %bool %brickY %c_f_01
%yHigh = OpFOrdGreaterThan %bool %brickY %c_f_09
%xMortar = OpLogicalOr %bool %xLow %xHigh
%yMortar = OpLogicalOr %bool %yLow %yHigh
%isMortar = OpLogicalOr %bool %xMortar %yMortar

%baseColor = OpSelect %v3float %isMortar %c_v3_mortar %c_v3_brick
%litColor = OpVectorTimesScalar %v3float %baseColor %shade
%finalColor = OpCompositeConstruct %v4float %litColor %c_f_1
OpStore %outColor %finalColor
OpReturn
