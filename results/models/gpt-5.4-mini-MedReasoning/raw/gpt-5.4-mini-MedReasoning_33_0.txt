using System;
using System.Diagnostics;
using System.Text;

public class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = (int)fs.NextLong();
        int nn = n * n;

        long[] flow = new long[nn];
        long[] dist = new long[nn];

        long[] flowRow = new long[n];
        long[] flowCol = new long[n];
        long[] distRow = new long[n];
        long[] distCol = new long[n];

        for (int i = 0; i < n; i++)
        {
            int baseIdx = i * n;
            for (int j = 0; j < n; j++)
            {
                long v = fs.NextLong();
                flow[baseIdx + j] = v;
                flowRow[i] += v;
                flowCol[j] += v;
            }
        }

        for (int i = 0; i < n; i++)
        {
            int baseIdx = i * n;
            for (int j = 0; j < n; j++)
            {
                long v = fs.NextLong();
                dist[baseIdx + j] = v;
                distRow[i] += v;
                distCol[j] += v;
            }
        }

        Candidate best = null;

        best = Better(best, BuildCandidate((long[])flowRow.Clone(), (long[])distRow.Clone(), n, flow, dist));
        best = Better(best, BuildCandidate((long[])flowCol.Clone(), (long[])distCol.Clone(), n, flow, dist));
        best = Better(best, BuildCandidate((long[])flowRow.Clone(), (long[])distCol.Clone(), n, flow, dist));
        best = Better(best, BuildCandidate((long[])flowCol.Clone(), (long[])distRow.Clone(), n, flow, dist));
        best = Better(best, BuildCandidate(BuildCombined(flowRow, flowCol), BuildCombined(distRow, distCol), n, flow, dist));

        Improve(best, flow, dist, n);

        long finalCost = Evaluate(best.Perm, flow, dist, n);
        var sb = new StringBuilder();
        sb.Append(finalCost).Append('\n');
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(best.Perm[i]);
        }
        sb.Append('\n');
        Console.Write(sb.ToString());
    }

    private static long[] BuildCombined(long[] a, long[] b)
    {
        int n = a.Length;
        long[] r = new long[n];
        for (int i = 0; i < n; i++) r[i] = a[i] + b[i];
        return r;
    }

    private static Candidate BuildCandidate(long[] facScore, long[] locScore, int n, long[] flow, long[] dist)
    {
        int[] facOrder = InitOrder(n);
        int[] locOrder = InitOrder(n);
        Array.Sort(facScore, facOrder);
        Array.Sort(locScore, locOrder);

        Array.Reverse(facOrder);

        int[] perm = new int[n];
        for (int i = 0; i < n; i++)
        {
            perm[facOrder[i]] = locOrder[i];
        }

        long cost = Evaluate(perm, flow, dist, n);
        return new Candidate { Perm = perm, FacRank = facOrder, Cost = cost };
    }

    private static Candidate Better(Candidate a, Candidate b)
    {
        if (a == null) return b;
        return b.Cost < a.Cost ? b : a;
    }

    private static int[] InitOrder(int n)
    {
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        return order;
    }

    private static long Evaluate(int[] perm, long[] flow, long[] dist, int n)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int fi = i * n;
            int pi = perm[i] * n;
            for (int j = 0; j < n; j++)
            {
                cost += flow[fi + j] * dist[pi + perm[j]];
            }
        }
        return cost;
    }

    private static void Improve(Candidate cand, long[] flow, long[] dist, int n)
    {
        if (cand == null || n <= 1) return;

        var sw = Stopwatch.StartNew();
        int window = Math.Min(12, n - 1);
        bool changed = true;
        long currentCost = cand.Cost;

        while (changed && sw.ElapsedMilliseconds < 4000)
        {
            changed = false;
            for (int ai = 0; ai < n && sw.ElapsedMilliseconds < 4000; ai++)
            {
                int a = cand.FacRank[ai];
                int pa = cand.Perm[a];
                int end = Math.Min(n - 1, ai + window);

                for (int bi = ai + 1; bi <= end; bi++)
                {
                    int b = cand.FacRank[bi];
                    int pb = cand.Perm[b];
                    long delta = SwapDelta(a, b, pa, pb, cand.Perm, flow, dist, n);
                    if (delta < 0)
                    {
                        cand.Perm[a] = pb;
                        cand.Perm[b] = pa;
                        currentCost += delta;
                        changed = true;
                        break;
                    }
                }
            }
        }

        cand.Cost = currentCost;
    }

    private static long SwapDelta(int a, int b, int pa, int pb, int[] perm, long[] flow, long[] dist, int n)
    {
        int rowA = a * n;
        int rowB = b * n;
        int rowPa = pa * n;
        int rowPb = pb * n;

        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;

            int pk = perm[k];
            int rowK = k * n;
            int rowPk = pk * n;

            long d1 = dist[rowPb + pk] - dist[rowPa + pk];
            long d2 = dist[rowPk + pb] - dist[rowPk + pa];

            delta += (flow[rowA + k] - flow[rowB + k]) * d1;
            delta += (flow[rowK + a] - flow[rowK + b]) * d2;
        }

        delta += flow[rowA + a] * (dist[rowPb + pb] - dist[rowPa + pa]);
        delta += flow[rowB + b] * (dist[rowPa + pa] - dist[rowPb + pb]);
        delta += flow[rowA + b] * (dist[rowPb + pa] - dist[rowPa + pb]);
        delta += flow[rowB + a] * (dist[rowPa + pb] - dist[rowPb + pa]);

        return delta;
    }

    private sealed class Candidate
    {
        public int[] Perm = Array.Empty<int>();
        public int[] FacRank = Array.Empty<int>();
        public long Cost;
    }

    private sealed class FastScanner
    {
        private readonly System.IO.Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len;
        private int ptr;

        public FastScanner(System.IO.Stream stream)
        {
            this.stream = stream;
        }

        private byte Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return 0;
            }
            return buffer[ptr++];
        }

        public long NextLong()
        {
            byte c;
            do
            {
                c = Read();
            } while (c <= 32 && c != 0);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }
    }
}
