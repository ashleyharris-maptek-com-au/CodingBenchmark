// N-dimensional snake planner.
// Reads a complete static board description and emits a precomputed move list.

#include <algorithm>
#include <cstdlib>
#include <cstdint>
#include <iostream>
#include <limits>
#include <optional>
#include <queue>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using std::int64_t;

using Pos = std::vector<int>;

struct CoordHash {
    std::size_t operator()(const Pos& v) const noexcept {
        std::uint64_t h = 1469598103934665603ULL;
        for (int x : v) {
            std::uint64_t y = static_cast<std::uint64_t>(static_cast<std::int64_t>(x));
            y ^= (y >> 33);
            y *= 0xff51afd7ed558ccdULL;
            y ^= (y >> 33);
            y *= 0xc4ceb9fe1a85ec53ULL;
            y ^= (y >> 33);
            h ^= y + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        }
        return static_cast<std::size_t>(h);
    }
};

struct Move {
    int axis = 0;
    int dir = 1;
};

struct Parent {
    Pos prev;
    int axis = 0;
    int dir = 1;
};

struct Node {
    Pos pos;
    int g = 0;
    int f = 0;
};

struct NodeCmp {
    bool operator()(const Node& a, const Node& b) const noexcept {
        if (a.f != b.f) return a.f > b.f;
        return a.g < b.g;
    }
};

static int manhattan(const Pos& a, const Pos& b) {
    int h = 0;
    for (std::size_t i = 0; i < a.size(); ++i) {
        h += std::abs(a[i] - b[i]);
    }
    return h;
}

static std::optional<std::vector<Move>> find_path(
    const Pos& start,
    const Pos& target,
    const std::unordered_set<Pos, CoordHash>& obstacles,
    const std::unordered_set<Pos, CoordHash>& reserved,
    bool strict,
    const Pos& bounds
) {
    if (start == target) return std::vector<Move>{};

    std::priority_queue<Node, std::vector<Node>, NodeCmp> pq;
    std::unordered_map<Pos, int, CoordHash> best_g;
    std::unordered_map<Pos, Parent, CoordHash> parent;

    best_g.emplace(start, 0);
    pq.push(Node{start, 0, manhattan(start, target)});

    while (!pq.empty()) {
        Node cur = pq.top();
        pq.pop();

        auto it = best_g.find(cur.pos);
        if (it == best_g.end() || it->second != cur.g) {
            continue;
        }

        if (cur.pos == target) {
            break;
        }

        for (int axis = 0; axis < static_cast<int>(bounds.size()); ++axis) {
            for (int dir : {-1, 1}) {
                Pos next = cur.pos;
                next[axis] += dir;
                if (next[axis] < 0 || next[axis] >= bounds[axis]) {
                    continue;
                }
                if (obstacles.find(next) != obstacles.end()) {
                    continue;
                }
                if (strict && reserved.find(next) != reserved.end()) {
                    continue;
                }

                int ng = cur.g + 1;
                auto jt = best_g.find(next);
                if (jt == best_g.end() || ng < jt->second) {
                    best_g[next] = ng;
                    parent[next] = Parent{cur.pos, axis, dir};
                    pq.push(Node{next, ng, ng + manhattan(next, target)});
                }
            }
        }
    }

    if (best_g.find(target) == best_g.end()) {
        return std::nullopt;
    }

    std::vector<Move> path;
    Pos cur = target;
    while (cur != start) {
        const Parent& p = parent.at(cur);
        path.push_back(Move{p.axis, p.dir});
        cur = p.prev;
    }
    std::reverse(path.begin(), path.end());
    return path;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int D = 0;
    int max_turns = 0;
    if (!(std::cin >> D >> max_turns)) {
        return 0;
    }

    Pos bounds(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> bounds[i];
    }

    int snake_length = 0;
    Pos start(D);
    std::cin >> snake_length;
    for (int i = 0; i < D; ++i) {
        std::cin >> start[i];
    }

    int F = 0;
    std::cin >> F;
    std::vector<Pos> foods(F, Pos(D));
    for (int i = 0; i < F; ++i) {
        for (int d = 0; d < D; ++d) {
            std::cin >> foods[i][d];
        }
    }

    int O = 0;
    std::cin >> O;
    std::unordered_set<Pos, CoordHash> obstacles;
    obstacles.reserve(static_cast<std::size_t>(O) * 2 + 8);
    for (int i = 0; i < O; ++i) {
        Pos p(D);
        for (int d = 0; d < D; ++d) {
            std::cin >> p[d];
        }
        obstacles.insert(std::move(p));
    }

    (void)snake_length;

    std::unordered_set<Pos, CoordHash> reserved;
    reserved.reserve(1024);
    reserved.insert(start);

    std::vector<Move> output;
    output.reserve(std::max(0, max_turns));

    Pos current = start;
    int used_turns = 0;

    std::vector<int> order;
    order.reserve(foods.size());
    for (int i = 0; i < F; ++i) {
        order.push_back(i);
    }

    while (used_turns < max_turns) {
        bool consumed_any = false;
        for (int pass = 0; pass < 2 && used_turns < max_turns; ++pass) {
            bool strict = (pass == 0);
            std::vector<int> candidates;
            candidates.reserve(order.size());
            for (int idx : order) {
                candidates.push_back(idx);
            }

            std::sort(candidates.begin(), candidates.end(), [&](int a, int b) {
                int da = manhattan(current, foods[a]);
                int db = manhattan(current, foods[b]);
                if (da != db) return da < db;
                return a < b;
            });

            std::optional<std::vector<Move>> chosen_path;
            int chosen_idx = -1;

            for (int idx : candidates) {
                auto path = find_path(current, foods[idx], obstacles, reserved, strict, bounds);
                if (path) {
                    chosen_path = std::move(path);
                    chosen_idx = idx;
                    break;
                }
            }

            if (!chosen_path) {
                continue;
            }

            const std::vector<Move>& path = *chosen_path;
            int take = std::min<int>(path.size(), max_turns - used_turns);
            Pos sim = current;
            for (int i = 0; i < take; ++i) {
                const Move& m = path[i];
                sim[m.axis] += m.dir;
                output.push_back(m);
                reserved.insert(sim);
                ++used_turns;
            }
            current = sim;

            if (take == static_cast<int>(path.size())) {
                order.erase(std::remove(order.begin(), order.end(), chosen_idx), order.end());
                foods[chosen_idx].clear();
                consumed_any = true;
            }
            break;
        }

        if (!consumed_any) {
            break;
        }
    }

    for (const Move& m : output) {
        std::cout << m.axis << ' ' << m.dir << '\n';
    }

    return 0;
}
