use std::cmp::Ordering;
use std::io::{self, BufRead, BufReader, BufWriter, Read, Write};

#[derive(Clone, Copy, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

#[derive(Clone, Copy, Default)]
struct Sum {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy)]
struct Node {
    idx: usize,
    left: Option<usize>,
    right: Option<usize>,
    axis: u8,
    min: [f32; 3],
    max: [f32; 3],
}

struct KdTree {
    nodes: Vec<Node>,
    root: usize,
}

struct Input<R: Read> {
    reader: BufReader<R>,
}

impl<R: Read> Input<R> {
    fn new(inner: R) -> Self {
        Self {
            reader: BufReader::new(inner),
        }
    }

    fn read_header(&mut self) -> io::Result<(usize, usize)> {
        let mut line = String::new();
        self.reader.read_line(&mut line)?;
        let mut it = line.split_whitespace();
        let n: usize = it
            .next()
            .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "missing num_points"))?
            .parse()
            .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid num_points"))?;
        let k: usize = it
            .next()
            .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "missing num_clusters"))?
            .parse()
            .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid num_clusters"))?;
        Ok((n, k))
    }

    fn read_points(&mut self, count: usize) -> io::Result<Vec<Point>> {
        let mut bytes = vec![0u8; count * 12];
        self.reader.read_exact(&mut bytes)?;
        let mut points = Vec::with_capacity(count);
        for i in 0..count {
            let base = i * 12;
            points.push(Point {
                x: f32::from_le_bytes(bytes[base..base + 4].try_into().unwrap()),
                y: f32::from_le_bytes(bytes[base + 4..base + 8].try_into().unwrap()),
                z: f32::from_le_bytes(bytes[base + 8..base + 12].try_into().unwrap()),
            });
        }
        Ok(points)
    }

    fn read_points_batch(&mut self, count: usize, out: &mut [Point]) -> io::Result<()> {
        let mut bytes = vec![0u8; count * 12];
        self.reader.read_exact(&mut bytes)?;
        for i in 0..count {
            let base = i * 12;
            out[i] = Point {
                x: f32::from_le_bytes(bytes[base..base + 4].try_into().unwrap()),
                y: f32::from_le_bytes(bytes[base + 4..base + 8].try_into().unwrap()),
                z: f32::from_le_bytes(bytes[base + 8..base + 12].try_into().unwrap()),
            };
        }
        Ok(())
    }

}

#[inline(always)]
fn coord(p: Point, axis: usize) -> f32 {
    match axis {
        0 => p.x,
        1 => p.y,
        _ => p.z,
    }
}

#[inline(always)]
fn dist2(a: Point, b: Point) -> f32 {
    let dx = a.x - b.x;
    let dy = a.y - b.y;
    let dz = a.z - b.z;
    dx * dx + dy * dy + dz * dz
}

#[inline(always)]
fn bbox_dist2(p: Point, min: [f32; 3], max: [f32; 3]) -> f32 {
    let dx = if p.x < min[0] {
        min[0] - p.x
    } else if p.x > max[0] {
        p.x - max[0]
    } else {
        0.0
    };
    let dy = if p.y < min[1] {
        min[1] - p.y
    } else if p.y > max[1] {
        p.y - max[1]
    } else {
        0.0
    };
    let dz = if p.z < min[2] {
        min[2] - p.z
    } else if p.z > max[2] {
        p.z - max[2]
    } else {
        0.0
    };
    dx * dx + dy * dy + dz * dz
}

#[inline(always)]
fn sum_point(s: &mut Sum, p: Point) {
    s.x += p.x as f64;
    s.y += p.y as f64;
    s.z += p.z as f64;
}

fn init_centers(points: &[Point], k: usize) -> Vec<Point> {
    let mut centers = Vec::with_capacity(k);
    if points.is_empty() {
        centers.resize(k, Point::default());
        return centers;
    }
    for i in 0..k {
        let idx = (i * points.len()) / k;
        centers.push(points[idx.min(points.len() - 1)]);
    }
    centers
}

fn update_centers(centers: &mut [Point], sums: &[Sum], counts: &[u64]) {
    for i in 0..centers.len() {
        if counts[i] > 0 {
            let denom = counts[i] as f64;
            centers[i] = Point {
                x: (sums[i].x / denom) as f32,
                y: (sums[i].y / denom) as f32,
                z: (sums[i].z / denom) as f32,
            };
        }
    }
}

impl KdTree {
    fn build(centers: &[Point]) -> Option<Self> {
        if centers.is_empty() {
            return None;
        }
        let mut indices: Vec<usize> = (0..centers.len()).collect();
        let mut nodes = Vec::with_capacity(centers.len());
        let (root, _, _) = Self::build_rec(&mut indices, 0, centers, &mut nodes);
        Some(Self { nodes, root })
    }

    fn build_rec(
        indices: &mut [usize],
        depth: usize,
        centers: &[Point],
        nodes: &mut Vec<Node>,
    ) -> (usize, [f32; 3], [f32; 3]) {
        let axis = depth % 3;
        indices.sort_unstable_by(|&a, &b| {
            coord(centers[a], axis)
                .partial_cmp(&coord(centers[b], axis))
                .unwrap_or(Ordering::Equal)
        });

        let mid = indices.len() / 2;
        let idx = indices[mid];
        let p = centers[idx];
        let mut min = [p.x, p.y, p.z];
        let mut max = [p.x, p.y, p.z];

        let left = if mid > 0 {
            let (li, lmin, lmax) = Self::build_rec(&mut indices[..mid], depth + 1, centers, nodes);
            for j in 0..3 {
                if lmin[j] < min[j] {
                    min[j] = lmin[j];
                }
                if lmax[j] > max[j] {
                    max[j] = lmax[j];
                }
            }
            Some(li)
        } else {
            None
        };

        let right = if mid + 1 < indices.len() {
            let (ri, rmin, rmax) =
                Self::build_rec(&mut indices[mid + 1..], depth + 1, centers, nodes);
            for j in 0..3 {
                if rmin[j] < min[j] {
                    min[j] = rmin[j];
                }
                if rmax[j] > max[j] {
                    max[j] = rmax[j];
                }
            }
            Some(ri)
        } else {
            None
        };

        let node_idx = nodes.len();
        nodes.push(Node {
            idx,
            left,
            right,
            axis: axis as u8,
            min,
            max,
        });
        (node_idx, min, max)
    }

    fn nearest(&self, q: Point, centers: &[Point]) -> usize {
        let mut best_idx = 0usize;
        let mut best_dist = f32::INFINITY;
        self.nearest_rec(self.root, q, centers, &mut best_idx, &mut best_dist);
        best_idx
    }

    fn nearest_rec(
        &self,
        node_idx: usize,
        q: Point,
        centers: &[Point],
        best_idx: &mut usize,
        best_dist: &mut f32,
    ) {
        let node = &self.nodes[node_idx];
        if bbox_dist2(q, node.min, node.max) >= *best_dist {
            return;
        }

        let p = centers[node.idx];
        let d = dist2(q, p);
        if d < *best_dist {
            *best_dist = d;
            *best_idx = node.idx;
        }

        let axis = node.axis as usize;
        let qv = coord(q, axis);
        let pv = coord(p, axis);
        let (first, second) = if qv <= pv {
            (node.left, node.right)
        } else {
            (node.right, node.left)
        };

        if let Some(child) = first {
            self.nearest_rec(child, q, centers, best_idx, best_dist);
        }
        if let Some(child) = second {
            let child_node = &self.nodes[child];
            if bbox_dist2(q, child_node.min, child_node.max) < *best_dist {
                self.nearest_rec(child, q, centers, best_idx, best_dist);
            }
        }
    }
}

fn nearest_center(q: Point, centers: &[Point], tree: Option<&KdTree>) -> usize {
    if let Some(tree) = tree {
        tree.nearest(q, centers)
    } else {
        let mut best = 0usize;
        let mut best_dist = f32::INFINITY;
        for (i, &c) in centers.iter().enumerate() {
            let d = dist2(q, c);
            if d < best_dist {
                best_dist = d;
                best = i;
            }
        }
        best
    }
}

fn assign_points(
    points: &[Point],
    centers: &[Point],
    tree: Option<&KdTree>,
    sums: &mut [Sum],
    counts: &mut [u64],
    mut out: Option<&mut [u8]>,
) {
    for (i, &p) in points.iter().enumerate() {
        let idx = nearest_center(p, centers, tree);
        counts[idx] += 1;
        sum_point(&mut sums[idx], p);
        if let Some(ref mut buf) = out {
            let bytes = (idx as u32).to_le_bytes();
            let base = i * 4;
            buf[base..base + 4].copy_from_slice(&bytes);
        }
    }
}

fn run_in_memory(points: &[Point], k: usize, writer: &mut BufWriter<impl Write>) -> io::Result<()> {
    let mut centers = init_centers(points, k);
    let use_tree = k > 32;
    let iters = if points.len() <= 200_000 { 6 } else { 3 };
    let mut sums = vec![Sum::default(); k];
    let mut counts = vec![0u64; k];

    for _ in 0..iters {
        sums.fill(Sum::default());
        counts.fill(0);
        let tree = if use_tree {
            KdTree::build(&centers)
        } else {
            None
        };
        let batch = 65_536.min(points.len().max(1));
        for chunk in points.chunks(batch) {
            assign_points(chunk, &centers, tree.as_ref(), &mut sums, &mut counts, None);
        }
        update_centers(&mut centers, &sums, &counts);
    }

    sums.fill(Sum::default());
    counts.fill(0);
    let tree = if use_tree {
        KdTree::build(&centers)
    } else {
        None
    };
    let batch = 65_536.min(points.len().max(1));
    for chunk in points.chunks(batch) {
        let mut bytes = vec![0u8; chunk.len() * 4];
        assign_points(
            chunk,
            &centers,
            tree.as_ref(),
            &mut sums,
            &mut counts,
            Some(&mut bytes),
        );
        writer.write_all(&bytes)?;
    }
    Ok(())
}

fn stream_identity(
    input: &mut Input<impl Read>,
    n: usize,
    writer: &mut BufWriter<impl Write>,
) -> io::Result<()> {
    const BATCH: usize = 65_536;
    let mut points = vec![Point::default(); BATCH];
    let mut out = vec![0u8; BATCH * 4];
    let mut done = 0usize;
    while done < n {
        let take = (n - done).min(BATCH);
        input.read_points_batch(take, &mut points[..take])?;
        for i in 0..take {
            let label = (done + i) as u32;
            out[i * 4..i * 4 + 4].copy_from_slice(&label.to_le_bytes());
        }
        writer.write_all(&out[..take * 4])?;
        done += take;
    }
    Ok(())
}

fn stream_constant_zero(
    input: &mut Input<impl Read>,
    n: usize,
    writer: &mut BufWriter<impl Write>,
) -> io::Result<()> {
    const BATCH: usize = 65_536;
    let mut points = vec![Point::default(); BATCH];
    let out = vec![0u8; BATCH * 4];
    let mut done = 0usize;
    while done < n {
        let take = (n - done).min(BATCH);
        input.read_points_batch(take, &mut points[..take])?;
        writer.write_all(&out[..take * 4])?;
        done += take;
    }
    Ok(())
}

fn run_streaming(
    input: &mut Input<impl Read>,
    n: usize,
    k: usize,
    writer: &mut BufWriter<impl Write>,
) -> io::Result<()> {
    let warmup_cap = (4usize.saturating_mul(k)).max(32_768).min(1_000_000);
    let warmup = warmup_cap.min(n);
    let points = input.read_points(warmup)?;
    let mut centers = init_centers(&points, k);
    let use_tree = k > 32;
    let mut sums = vec![Sum::default(); k];
    let mut counts = vec![0u64; k];
    let warmup_iters = if k <= 512 { 4 } else if k <= 4096 { 3 } else { 2 };

    for _ in 0..warmup_iters {
        sums.fill(Sum::default());
        counts.fill(0);
        let tree = if use_tree {
            KdTree::build(&centers)
        } else {
            None
        };
        assign_points(&points, &centers, tree.as_ref(), &mut sums, &mut counts, None);
        update_centers(&mut centers, &sums, &counts);
    }

    sums.fill(Sum::default());
    counts.fill(0);
    let tree = if use_tree {
        KdTree::build(&centers)
    } else {
        None
    };
    let mut out = vec![0u8; points.len() * 4];
    assign_points(
        &points,
        &centers,
        tree.as_ref(),
        &mut sums,
        &mut counts,
        Some(&mut out),
    );
    update_centers(&mut centers, &sums, &counts);
    writer.write_all(&out)?;

    let remaining = n - warmup;
    if remaining == 0 {
        return Ok(());
    }

    let batch_size = (4usize.saturating_mul(k)).max(8_192).min(262_144);
    let mut batch = vec![Point::default(); batch_size];
    let mut label_bytes = vec![0u8; batch_size * 4];

    let mut left = remaining;
    while left > 0 {
        let take = left.min(batch_size);
        input.read_points_batch(take, &mut batch[..take])?;

        let tree = if use_tree {
            KdTree::build(&centers)
        } else {
            None
        };
        let mut batch_sums = vec![Sum::default(); k];
        let mut batch_counts = vec![0u64; k];
        assign_points(
            &batch[..take],
            &centers,
            tree.as_ref(),
            &mut batch_sums,
            &mut batch_counts,
            Some(&mut label_bytes[..take * 4]),
        );

        for i in 0..k {
            sums[i].x += batch_sums[i].x;
            sums[i].y += batch_sums[i].y;
            sums[i].z += batch_sums[i].z;
            counts[i] += batch_counts[i];
        }
        update_centers(&mut centers, &sums, &counts);
        writer.write_all(&label_bytes[..take * 4])?;
        left -= take;
    }

    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut input = Input::new(stdin.lock());
    let mut writer = BufWriter::new(stdout.lock());
    let (n, k) = input.read_header()?;

    if n == 0 || k == 0 {
        return Ok(());
    }

    if k == 1 {
        return stream_constant_zero(&mut input, n, &mut writer);
    }

    if k >= n && n <= u32::MAX as usize {
        return stream_identity(&mut input, n, &mut writer);
    }

    if n <= 2_000_000 {
        let points = input.read_points(n)?;
        run_in_memory(&points, k, &mut writer)?;
    } else {
        run_streaming(&mut input, n, k, &mut writer)?;
    }

    writer.flush()?;
    Ok(())
}
