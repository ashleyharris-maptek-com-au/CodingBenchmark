// Adaptive graph coloring solver.
// Reads: n m k, then m edges (0-indexed).
// Writes: one color per vertex.

#include <algorithm>
#include <charconv>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <utility>
#include <string>
#include <vector>

#if defined(_MSC_VER)
#include <intrin.h>
#endif

using namespace std;

struct Graph {
    int n = 0, m = 0, k = 0;
    vector<int> deg;
    vector<int> off;
    vector<int> adj;
    vector<pair<int, int>> edges;
};

static inline uint64_t splitmix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

static inline int popcount64(uint64_t x) {
#if defined(_MSC_VER)
    return (int)__popcnt64(x);
#else
    return __builtin_popcountll(x);
#endif
}

static inline bool timed_out(const chrono::steady_clock::time_point& start, int limit_ms) {
    return chrono::duration_cast<chrono::milliseconds>(chrono::steady_clock::now() - start).count() >= limit_ms;
}

static Graph read_graph() {
    Graph g;
    cin >> g.n >> g.m >> g.k;
    g.deg.assign(g.n, 0);
    g.edges.reserve(g.m);
    for (int i = 0; i < g.m; ++i) {
        int u, v;
        cin >> u >> v;
        g.edges.emplace_back(u, v);
        if (u >= 0 && u < g.n && v >= 0 && v < g.n) {
            ++g.deg[u];
            ++g.deg[v];
        }
    }

    g.off.assign(g.n + 1, 0);
    for (int i = 0; i < g.n; ++i) g.off[i + 1] = g.off[i] + g.deg[i];
    g.adj.assign(2 * g.m, 0);
    vector<int> cur = g.off;
    for (auto [u, v] : g.edges) {
        if (u < 0 || u >= g.n || v < 0 || v >= g.n) continue;
        g.adj[cur[u]++] = v;
        g.adj[cur[v]++] = u;
    }
    return g;
}

static vector<int> degree_order(const Graph& g, uint64_t seed) {
    vector<int> order(g.n);
    for (int i = 0; i < g.n; ++i) order[i] = i;
    vector<uint64_t> key(g.n);
    for (int i = 0; i < g.n; ++i) key[i] = splitmix64((uint64_t)i ^ seed);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (g.deg[a] != g.deg[b]) return g.deg[a] > g.deg[b];
        return key[a] < key[b];
    });
    return order;
}

static vector<int> smallest_last_order(const Graph& g) {
    vector<int> cur = g.deg;
    vector<char> removed(g.n, 0);
    vector<int> rem;
    rem.reserve(g.n);
    using Node = pair<int, int>;
    priority_queue<Node, vector<Node>, greater<Node>> pq;
    for (int v = 0; v < g.n; ++v) pq.push({cur[v], v});

    while (!pq.empty()) {
        auto [d, v] = pq.top();
        pq.pop();
        if (removed[v] || d != cur[v]) continue;
        removed[v] = 1;
        rem.push_back(v);
        for (int e = g.off[v]; e < g.off[v + 1]; ++e) {
            int u = g.adj[e];
            if (!removed[u]) {
                --cur[u];
                pq.push({cur[u], u});
            }
        }
    }
    reverse(rem.begin(), rem.end());
    return rem;
}

static bool valid_coloring(const Graph& g, const vector<int>& color) {
    for (auto [u, v] : g.edges) {
        if (u >= 0 && u < g.n && v >= 0 && v < g.n && color[u] == color[v]) return false;
    }
    return true;
}

static bool greedy_assign_order(const Graph& g, const vector<int>& order, vector<int>& color, uint64_t seed) {
    const int n = g.n;
    const int k = g.k;
    color.assign(n, -1);
    vector<int> cnt(k, 0);
    vector<int> rep(k, -1);
    vector<int> repDeg(k, numeric_limits<int>::max());

    for (int idx = 0; idx < n; ++idx) {
        int v = order[idx];
        fill(cnt.begin(), cnt.end(), 0);
        fill(rep.begin(), rep.end(), -1);
        fill(repDeg.begin(), repDeg.end(), numeric_limits<int>::max());

        for (int e = g.off[v]; e < g.off[v + 1]; ++e) {
            int u = g.adj[e];
            int c = color[u];
            if (c >= 0) {
                ++cnt[c];
                if (g.deg[u] < repDeg[c]) {
                    repDeg[c] = g.deg[u];
                    rep[c] = u;
                }
            }
        }

        int best = -1;
        int bestCnt = numeric_limits<int>::max();
        vector<int> candidates;
        candidates.reserve(8);
        for (int c = 0; c < k; ++c) {
            if (cnt[c] < bestCnt) {
                bestCnt = cnt[c];
                candidates.clear();
                candidates.push_back(c);
            } else if (cnt[c] == bestCnt) {
                candidates.push_back(c);
            }
        }

        if (bestCnt == 0) {
            uint64_t h = splitmix64(seed ^ (uint64_t)v);
            best = candidates[(size_t)(h % candidates.size())];
        } else {
            for (int c = 0; c < k; ++c) {
                if (cnt[c] == 1 && rep[c] != -1) {
                    int u = rep[c];
                    int chosen = -1;
                    for (int d = 0; d < k; ++d) {
                        if (d == color[u]) continue;
                        bool ok = true;
                        for (int ee = g.off[u]; ee < g.off[u + 1]; ++ee) {
                            int w = g.adj[ee];
                            if (color[w] == d) {
                                ok = false;
                                break;
                            }
                        }
                        if (ok) {
                            chosen = d;
                            break;
                        }
                    }
                    if (chosen != -1) {
                        color[u] = chosen;
                        best = c;
                        break;
                    }
                }
            }
            if (best == -1) {
                best = candidates[(size_t)(splitmix64(seed ^ ((uint64_t)v << 1)) % candidates.size())];
            }
        }
        color[v] = best;
    }
    return true;
}

static bool repair_coloring(const Graph& g, vector<int>& color, const chrono::steady_clock::time_point& start, int limit_ms, uint64_t seed) {
    const int n = g.n;
    const int k = g.k;
    vector<int> conflict(n, 0);
    priority_queue<pair<int, int>> pq;

    for (int v = 0; v < n; ++v) {
        int c = color[v];
        int cnt = 0;
        for (int e = g.off[v]; e < g.off[v + 1]; ++e) {
            int u = g.adj[e];
            if (color[u] == c) ++cnt;
        }
        conflict[v] = cnt;
        if (cnt > 0) pq.push({cnt, v});
    }

    vector<int> cntColor(k, 0);
    vector<int> rep(k, -1);
    vector<int> repDeg(k, numeric_limits<int>::max());
    vector<int> tmpColor(k, 0);

    auto recolor_vertex = [&](int v, int newColor) {
        int oldColor = color[v];
        if (oldColor == newColor) return;
        for (int e = g.off[v]; e < g.off[v + 1]; ++e) {
            int u = g.adj[e];
            if (color[u] == oldColor) {
                --conflict[u];
                if (conflict[u] > 0) pq.push({conflict[u], u});
            }
            if (color[u] == newColor) {
                ++conflict[u];
                pq.push({conflict[u], u});
            }
        }
        color[v] = newColor;
        int cnt = 0;
        for (int e = g.off[v]; e < g.off[v + 1]; ++e) {
            int u = g.adj[e];
            if (color[u] == newColor) ++cnt;
        }
        conflict[v] = cnt;
        if (cnt > 0) pq.push({cnt, v});
    };

    int iter = 0;
    int maxIter = max(50000, 3 * n);

    while (!pq.empty() && iter < maxIter && !timed_out(start, limit_ms)) {
        auto [cc, v] = pq.top();
        pq.pop();
        if (cc != conflict[v] || cc == 0) continue;
        ++iter;

        fill(cntColor.begin(), cntColor.end(), 0);
        fill(rep.begin(), rep.end(), -1);
        fill(repDeg.begin(), repDeg.end(), numeric_limits<int>::max());
        for (int e = g.off[v]; e < g.off[v + 1]; ++e) {
            int u = g.adj[e];
            int c = color[u];
            ++cntColor[c];
            if (g.deg[u] < repDeg[c]) {
                repDeg[c] = g.deg[u];
                rep[c] = u;
            }
        }

        int current = cntColor[color[v]];
        int bestCnt = current;
        vector<int> bestColors;
        bestColors.reserve(8);
        for (int c = 0; c < k; ++c) {
            if (cntColor[c] < bestCnt) {
                bestCnt = cntColor[c];
                bestColors.clear();
                bestColors.push_back(c);
            } else if (cntColor[c] == bestCnt) {
                bestColors.push_back(c);
            }
        }

        int chosen = color[v];
        if (bestCnt == 0) {
            chosen = bestColors[(size_t)(splitmix64(seed ^ (uint64_t)(v + iter)) % bestColors.size())];
        } else {
            bool improved = bestCnt < current;
            if (improved) {
                chosen = bestColors[(size_t)(splitmix64(seed ^ (uint64_t)(v + iter * 3)) % bestColors.size())];
            } else if (bestCnt == current && bestCnt > 0) {
                // Sideways move to escape local minima.
                if ((iter & 7) == 0 && !bestColors.empty()) {
                    int alt = bestColors[(size_t)(splitmix64(seed ^ (uint64_t)(v + iter * 5)) % bestColors.size())];
                    if (alt != color[v]) chosen = alt;
                }
            }

            if (chosen == color[v]) {
                for (int c = 0; c < k; ++c) {
                    if (cntColor[c] == 1 && rep[c] != -1) {
                        int u = rep[c];
                        int freeColor = -1;
                        fill(tmpColor.begin(), tmpColor.end(), 0);
                        for (int e = g.off[u]; e < g.off[u + 1]; ++e) {
                            int w = g.adj[e];
                            ++tmpColor[color[w]];
                        }
                        for (int d = 0; d < k; ++d) {
                            if (d != color[u] && tmpColor[d] == 0) {
                                freeColor = d;
                                break;
                            }
                        }
                        if (freeColor != -1) {
                            recolor_vertex(u, freeColor);
                            chosen = c;
                            break;
                        }
                    }
                }
            }
        }

        if (chosen != color[v]) recolor_vertex(v, chosen);
    }

    return valid_coloring(g, color);
}

static bool dsatur_attempt(const Graph& g, vector<int>& color, const chrono::steady_clock::time_point& start, int limit_ms, uint64_t seed) {
    if (g.k > 63) return false;
    const int n = g.n;
    const int k = g.k;
    vector<int> sat(n, 0);
    vector<uint64_t> usedMask(n, 0);
    vector<char> colored(n, 0);
    vector<uint64_t> tieKey(n);
    for (int i = 0; i < n; ++i) tieKey[i] = splitmix64((uint64_t)i ^ seed);

    struct Node {
        int sat;
        int deg;
        int v;
        uint64_t tie;
    };
    struct Cmp {
        bool operator()(const Node& a, const Node& b) const {
            if (a.sat != b.sat) return a.sat < b.sat;
            if (a.deg != b.deg) return a.deg < b.deg;
            return a.tie > b.tie;
        }
    };

    priority_queue<Node, vector<Node>, Cmp> pq;
    for (int v = 0; v < n; ++v) pq.push({0, g.deg[v], v, tieKey[v]});
    color.assign(n, -1);
    vector<int> cntColor(k, 0);
    int coloredCount = 0;

    while (coloredCount < n && !pq.empty() && !timed_out(start, limit_ms)) {
        Node cur = pq.top();
        pq.pop();
        int v = cur.v;
        if (colored[v]) continue;
        int realSat = popcount64(usedMask[v]);
        if (cur.sat != realSat) {
            pq.push({realSat, g.deg[v], v, tieKey[v]});
            continue;
        }

        fill(cntColor.begin(), cntColor.end(), 0);
        uint64_t mask = usedMask[v];
        int best = -1;
        for (int e = g.off[v]; e < g.off[v + 1]; ++e) {
            int u = g.adj[e];
            int c = color[u];
            if (c >= 0) ++cntColor[c];
        }
        for (int c = 0; c < k; ++c) {
            if (((mask >> c) & 1ULL) == 0ULL) {
                best = c;
                break;
            }
        }
        if (best == -1) {
            int bestCnt = numeric_limits<int>::max();
            vector<int> options;
            for (int c = 0; c < k; ++c) {
                if (cntColor[c] < bestCnt) {
                    bestCnt = cntColor[c];
                    options.clear();
                    options.push_back(c);
                } else if (cntColor[c] == bestCnt) {
                    options.push_back(c);
                }
            }
            best = options[(size_t)(splitmix64(seed ^ (uint64_t)(v + coloredCount)) % options.size())];
        }

        color[v] = best;
        colored[v] = 1;
        ++coloredCount;

        uint64_t bit = 1ULL << best;
        for (int e = g.off[v]; e < g.off[v + 1]; ++e) {
            int u = g.adj[e];
            if (!colored[u]) {
                if ((usedMask[u] & bit) == 0ULL) {
                    usedMask[u] |= bit;
                    int s = popcount64(usedMask[u]);
                    sat[u] = s;
                    pq.push({s, g.deg[u], u, tieKey[u]});
                }
            }
        }
    }
    return coloredCount == n;
}

static vector<int> make_solution(const Graph& g) {
    if (g.k <= 0) return vector<int>(g.n, 0);
    const auto start = chrono::steady_clock::now();
    const int timeLimitMs = 290000;

    vector<int> best(g.n, 0);
    vector<int> cand;

    auto try_order = [&](const vector<int>& order, uint64_t seed) -> bool {
        cand.clear();
        greedy_assign_order(g, order, cand, seed);
        if (repair_coloring(g, cand, start, timeLimitMs, seed) && valid_coloring(g, cand)) {
            best = cand;
            return true;
        }
        return false;
    };

    if (g.n <= 50000 && g.m <= 500000 && g.k <= 63) {
        if (dsatur_attempt(g, cand, start, timeLimitMs, 0x123456789abcdef0ULL) &&
            repair_coloring(g, cand, start, timeLimitMs, 0x123456789abcdef0ULL) &&
            valid_coloring(g, cand)) {
            return cand;
        }
    }

    if (g.n <= 200000 && g.m <= 2000000) {
        vector<int> sl = smallest_last_order(g);
        if (try_order(sl, 0x9e3779b97f4a7c15ULL)) return cand;
        if (timed_out(start, timeLimitMs)) return cand;
    }

    if (try_order(degree_order(g, 1), 0x1111111111111111ULL)) return cand;
    if (timed_out(start, timeLimitMs)) return cand;
    if (try_order(degree_order(g, 2), 0x2222222222222222ULL)) return cand;
    if (timed_out(start, timeLimitMs)) return cand;
    if (try_order(degree_order(g, 3), 0x3333333333333333ULL)) return cand;

    // Final fallback: if no attempt reached a valid coloring, return the best
    // candidate we have. The benchmark expects a valid solution, but this keeps
    // the solver total and deterministic.
    if (!cand.empty()) return cand;
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Graph g = read_graph();
    vector<int> color = make_solution(g);

    if ((int)color.size() != g.n) color.assign(g.n, 0);

    string out;
    out.reserve((size_t)g.n * 4);
    char buf[32];
    for (int i = 0; i < g.n; ++i) {
        auto res = to_chars(buf, buf + 32, color[i]);
        out.append(buf, (size_t)(res.ptr - buf));
        out.push_back(i + 1 == g.n ? '\n' : ' ');
    }
    cout << out;
    return 0;
}
