// Longest Common Substring solver for multiple strings.
// Adaptive strategy:
// 1) Exact suffix automaton for smaller inputs.
// 2) Rolling-hash binary search fallback for larger inputs.

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <utility>
#include <vector>

using namespace std;

struct SAMState {
    int link = -1;
    int len = 0;
    int firstpos = -1;
    vector<pair<unsigned char, int>> next;
};

struct SuffixAutomaton {
    vector<SAMState> st;
    int last = 0;

    explicit SuffixAutomaton(size_t reserve_n = 0) {
        st.reserve(reserve_n * 2 + 1);
        st.push_back(SAMState{});
    }

    int findNext(int v, unsigned char c) const {
        for (const auto &e : st[v].next) {
            if (e.first == c) return e.second;
        }
        return -1;
    }

    void setNext(int v, unsigned char c, int to) {
        for (auto &e : st[v].next) {
            if (e.first == c) {
                e.second = to;
                return;
            }
        }
        st[v].next.push_back({c, to});
    }

    void extend(unsigned char c, int pos) {
        int cur = (int)st.size();
        st.push_back(SAMState{});
        st[cur].len = st[last].len + 1;
        st[cur].firstpos = pos;

        int p = last;
        while (p != -1 && findNext(p, c) == -1) {
            setNext(p, c, cur);
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = findNext(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back(st[q]);
                st[clone].len = st[p].len + 1;
                while (p != -1 && findNext(p, c) == q) {
                    setNext(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

struct HashPair {
    uint64_t a = 0;
    uint64_t b = 0;

    bool operator<(const HashPair &o) const {
        if (a != o.a) return a < o.a;
        return b < o.b;
    }

    bool operator==(const HashPair &o) const {
        return a == o.a && b == o.b;
    }
};

static constexpr uint64_t BASE1 = 11400714819323198485ull;
static constexpr uint64_t BASE2 = 14029467366897019727ull;

static vector<HashPair> buildHashes(const string &s, size_t L) {
    vector<HashPair> out;
    if (L == 0 || s.size() < L) return out;

    const size_t n = s.size();
    uint64_t pow1 = 1, pow2 = 1;
    for (size_t i = 0; i < L; ++i) {
        pow1 *= BASE1;
        pow2 *= BASE2;
    }

    uint64_t h1 = 0, h2 = 0;
    for (size_t i = 0; i < L; ++i) {
        unsigned char x = (unsigned char)s[i] + 1u;
        h1 = h1 * BASE1 + x;
        h2 = h2 * BASE2 + x;
    }
    out.push_back({h1, h2});

    for (size_t i = L; i < n; ++i) {
        unsigned char add = (unsigned char)s[i] + 1u;
        unsigned char rem = (unsigned char)s[i - L] + 1u;
        h1 = h1 * BASE1 + add - pow1 * rem;
        h2 = h2 * BASE2 + add - pow2 * rem;
        out.push_back({h1, h2});
    }

    sort(out.begin(), out.end());
    out.erase(unique(out.begin(), out.end()), out.end());
    return out;
}

static size_t findPositionByHash(const string &s, size_t L, const HashPair &target) {
    if (L == 0 || s.size() < L) return string::npos;

    uint64_t pow1 = 1, pow2 = 1;
    for (size_t i = 0; i < L; ++i) {
        pow1 *= BASE1;
        pow2 *= BASE2;
    }

    uint64_t h1 = 0, h2 = 0;
    for (size_t i = 0; i < L; ++i) {
        unsigned char x = (unsigned char)s[i] + 1u;
        h1 = h1 * BASE1 + x;
        h2 = h2 * BASE2 + x;
    }
    if (HashPair{h1, h2} == target) return 0;

    for (size_t i = L; i < s.size(); ++i) {
        unsigned char add = (unsigned char)s[i] + 1u;
        unsigned char rem = (unsigned char)s[i - L] + 1u;
        h1 = h1 * BASE1 + add - pow1 * rem;
        h2 = h2 * BASE2 + add - pow2 * rem;
        if (HashPair{h1, h2} == target) return i - L + 1;
    }
    return string::npos;
}

static string solveWithHash(const vector<string> &strings, size_t baseIdx, size_t L) {
    vector<HashPair> candidate = buildHashes(strings[baseIdx], L);
    if (candidate.empty()) return {};

    for (size_t i = 0; i < strings.size(); ++i) {
        if (i == baseIdx) continue;
        vector<HashPair> cur = buildHashes(strings[i], L);
        if (cur.empty()) return {};

        vector<HashPair> inter;
        inter.reserve(min(candidate.size(), cur.size()));
        size_t p = 0, q = 0;
        while (p < candidate.size() && q < cur.size()) {
            if (candidate[p] == cur[q]) {
                inter.push_back(candidate[p]);
                ++p;
                ++q;
            } else if (candidate[p] < cur[q]) {
                ++p;
            } else {
                ++q;
            }
        }

        candidate.swap(inter);
        if (candidate.empty()) return {};
    }

    size_t pos = findPositionByHash(strings[baseIdx], L, candidate.front());
    if (pos == string::npos) return {};
    return strings[baseIdx].substr(pos, L);
}

static string solveWithSAM(const vector<string> &strings, size_t baseIdx) {
    const string &base = strings[baseIdx];
    SuffixAutomaton sam(base.size());
    for (size_t i = 0; i < base.size(); ++i) {
        sam.extend((unsigned char)base[i], (int)i);
    }

    const int S = (int)sam.st.size();
    vector<int> order(S);
    for (int i = 0; i < S; ++i) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) {
        return sam.st[a].len > sam.st[b].len;
    });

    vector<int> common(S);
    for (int i = 0; i < S; ++i) common[i] = sam.st[i].len;

    for (size_t ti = 0; ti < strings.size(); ++ti) {
        if (ti == baseIdx) continue;
        vector<int> best(S, 0);
        int v = 0, l = 0;
        for (unsigned char c : strings[ti]) {
            while (v != 0 && sam.findNext(v, c) == -1) {
                v = sam.st[v].link;
                l = sam.st[v].len;
            }
            int nxt = sam.findNext(v, c);
            if (nxt != -1) {
                v = nxt;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            if (l > best[v]) best[v] = l;
        }

        for (int idx : order) {
            int p = sam.st[idx].link;
            if (p != -1) {
                int val = best[idx];
                if (val > sam.st[p].len) val = sam.st[p].len;
                if (val > best[p]) best[p] = val;
            }
        }

        for (int i = 0; i < S; ++i) {
            if (best[i] < common[i]) common[i] = best[i];
        }
    }

    int bestLen = 0;
    int bestState = 0;
    for (int i = 0; i < S; ++i) {
        if (common[i] > bestLen) {
            bestLen = common[i];
            bestState = i;
        }
    }

    if (bestLen <= 0) return {};
    size_t pos = (size_t)(sam.st[bestState].firstpos - bestLen + 1);
    return base.substr(pos, (size_t)bestLen);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);

    vector<string> strings;
    strings.reserve(max(0, N));
    for (int i = 0; i < N; ++i) {
        string s;
        getline(cin, s);
        strings.push_back(std::move(s));
    }

    if (strings.empty()) {
        cout << '\n';
        return 0;
    }

    size_t baseIdx = 0;
    size_t minLen = strings[0].size();
    size_t totalLen = strings[0].size();
    for (size_t i = 1; i < strings.size(); ++i) {
        totalLen += strings[i].size();
        if (strings[i].size() < minLen) {
            minLen = strings[i].size();
            baseIdx = i;
        }
    }

    for (const auto &s : strings) {
        if (s.empty()) {
            cout << '\n';
            return 0;
        }
    }

    string answer;
    const bool useSAM = (minLen <= 200000 && totalLen <= 5000000);
    if (useSAM) {
        answer = solveWithSAM(strings, baseIdx);
    } else {
        size_t lo = 0, hi = minLen;
        while (lo < hi) {
            size_t mid = (lo + hi + 1) / 2;
            if (!solveWithHash(strings, baseIdx, mid).empty()) {
                lo = mid;
            } else {
                hi = mid - 1;
            }
        }
        if (lo > 0) {
            answer = solveWithHash(strings, baseIdx, lo);
        }
    }

    cout << answer << '\n';
    return 0;
}
