OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1
OpSource GLSL 450
OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpDecorate %gl_PerVertex_in Block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%1 = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%func = OpTypeFunction %void

%bool = OpTypeBool
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_third = OpConstant %float 0.333333343
%float_two_third = OpConstant %float 0.666666687
%float_thickness = OpConstant %float 0.0199999996

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_gl_PerVertex_in
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3float = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel

%start_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%start = OpLoad %v4float %start_ptr
%end_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%end = OpLoad %v4float %end_ptr
%prev_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%prev = OpLoad %v4float %prev_ptr
%next_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%next = OpLoad %v4float %next_ptr

%start_xy = OpVectorShuffle %v2float %start %start 0 1
%end_xy = OpVectorShuffle %v2float %end %end 0 1
%prev_xy = OpVectorShuffle %v2float %prev %prev 0 1
%next_xy = OpVectorShuffle %v2float %next %next 0 1

%dir0 = OpFSub %v2float %start_xy %prev_xy
%dir1 = OpFSub %v2float %next_xy %end_xy
%ndir0 = OpExtInst %v2float %1 Normalize %dir0
%ndir1 = OpExtInst %v2float %1 Normalize %dir1
%tangent_sum = OpFAdd %v2float %ndir0 %ndir1
%tangent = OpExtInst %v2float %1 Normalize %tangent_sum

%tangent_x = OpCompositeExtract %float %tangent 0
%tangent_y = OpCompositeExtract %float %tangent 1
%neg_tangent_y = OpFNegate %float %tangent_y
%perp = OpCompositeConstruct %v2float %neg_tangent_y %tangent_x

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr

%start_part0 = OpVectorTimesScalar %v4float %start %float_1
%end_part0 = OpVectorTimesScalar %v4float %end %float_0
%p0 = OpFAdd %v4float %start_part0 %end_part0
%p0_w = OpCompositeExtract %float %p0 3
%p0_off_scale = OpFMul %float %float_thickness %p0_w
%p0_off2 = OpVectorTimesScalar %v2float %perp %p0_off_scale
%p0_off2_x = OpCompositeExtract %float %p0_off2 0
%p0_off2_y = OpCompositeExtract %float %p0_off2 1
%p0_off4 = OpCompositeConstruct %v4float %p0_off2_x %p0_off2_y %float_0 %float_0
%p0_minus = OpFSub %v4float %p0 %p0_off4
%p0_plus = OpFAdd %v4float %p0 %p0_off4

OpStore %outColor %c0
%out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0 %p0_minus
OpEmitVertex
OpStore %outColor %c0
OpStore %out_pos0 %p0_plus
OpEmitVertex

%p1_start_scale = OpFSub %float %float_1 %float_third
%start_part1 = OpVectorTimesScalar %v4float %start %p1_start_scale
%end_part1 = OpVectorTimesScalar %v4float %end %float_third
%p1 = OpFAdd %v4float %start_part1 %end_part1
%p1_w = OpCompositeExtract %float %p1 3
%p1_off_scale = OpFMul %float %float_thickness %p1_w
%p1_off2 = OpVectorTimesScalar %v2float %perp %p1_off_scale
%p1_off2_x = OpCompositeExtract %float %p1_off2 0
%p1_off2_y = OpCompositeExtract %float %p1_off2 1
%p1_off4 = OpCompositeConstruct %v4float %p1_off2_x %p1_off2_y %float_0 %float_0
%p1_minus = OpFSub %v4float %p1 %p1_off4
%p1_plus = OpFAdd %v4float %p1 %p1_off4

%c0_p1 = OpVectorTimesScalar %v3float %c0 %p1_start_scale
%c1_p1 = OpVectorTimesScalar %v3float %c1 %float_third
%c_p1 = OpFAdd %v3float %c0_p1 %c1_p1

OpStore %outColor %c_p1
OpStore %out_pos0 %p1_minus
OpEmitVertex
OpStore %outColor %c_p1
OpStore %out_pos0 %p1_plus
OpEmitVertex
OpEndPrimitive

%p2_start_scale = OpFSub %float %float_1 %float_two_third
%start_part2 = OpVectorTimesScalar %v4float %start %p2_start_scale
%end_part2 = OpVectorTimesScalar %v4float %end %float_two_third
%p2 = OpFAdd %v4float %start_part2 %end_part2
%p2_w = OpCompositeExtract %float %p2 3
%p2_off_scale = OpFMul %float %float_thickness %p2_w
%p2_off2 = OpVectorTimesScalar %v2float %perp %p2_off_scale
%p2_off2_x = OpCompositeExtract %float %p2_off2 0
%p2_off2_y = OpCompositeExtract %float %p2_off2 1
%p2_off4 = OpCompositeConstruct %v4float %p2_off2_x %p2_off2_y %float_0 %float_0
%p2_minus = OpFSub %v4float %p2 %p2_off4
%p2_plus = OpFAdd %v4float %p2 %p2_off4

%c0_p2 = OpVectorTimesScalar %v3float %c0 %p2_start_scale
%c1_p2 = OpVectorTimesScalar %v3float %c1 %float_two_third
%c_p2 = OpFAdd %v3float %c0_p2 %c1_p2

OpStore %outColor %c_p2
OpStore %out_pos0 %p2_minus
OpEmitVertex
OpStore %outColor %c_p2
OpStore %out_pos0 %p2_plus
OpEmitVertex

%start_part3 = OpVectorTimesScalar %v4float %start %float_0
%end_part3 = OpVectorTimesScalar %v4float %end %float_1
%p3 = OpFAdd %v4float %start_part3 %end_part3
%p3_w = OpCompositeExtract %float %p3 3
%p3_off_scale = OpFMul %float %float_thickness %p3_w
%p3_off2 = OpVectorTimesScalar %v2float %perp %p3_off_scale
%p3_off2_x = OpCompositeExtract %float %p3_off2 0
%p3_off2_y = OpCompositeExtract %float %p3_off2 1
%p3_off4 = OpCompositeConstruct %v4float %p3_off2_x %p3_off2_y %float_0 %float_0
%p3_minus = OpFSub %v4float %p3 %p3_off4
%p3_plus = OpFAdd %v4float %p3 %p3_off4

OpStore %outColor %c1
OpStore %out_pos0 %p3_minus
OpEmitVertex
OpStore %outColor %c1
OpStore %out_pos0 %p3_plus
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
