using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;

public sealed class Program
{
    private static readonly Shape[][] Shapes = BuildShapes();

    public static void Main()
    {
        var input = new FastScanner(Console.OpenStandardInput());
        int width = input.NextInt();
        int height = input.NextInt();
        int numPieces = input.NextInt();

        var board = new BoardState(width, height);
        var output = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = true };

        for (int turn = 0; turn < numPieces; turn++)
        {
            string pieceToken = input.Next();
            if (pieceToken == null)
            {
                break;
            }

            int piece = PieceIndex(pieceToken[0]);
            Move choice = ChooseMove(board, piece, turn, numPieces);

            output.WriteLine(choice.Rotation + " " + choice.Column);
            output.Flush();

            string result = input.Next();
            if (result == null || result == "gameover")
            {
                break;
            }

            if (!board.TryApply(Shapes[piece][choice.Rotation], choice.Column, out _))
            {
                break;
            }
        }
    }

    private static Move ChooseMove(BoardState board, int piece, int turn, int totalPieces)
    {
        double boardComplexity = ComputeComplexity(board);
        bool useLookahead = boardComplexity < 0.78 || turn < 6;
        double futureWeight = useLookahead ? (0.35 - 0.20 * boardComplexity) : 0.0;
        if (futureWeight < 0.10)
        {
            futureWeight = 0.10;
        }

        Move best = default;
        double bestScore = double.NegativeInfinity;
        double bestTiebreak = double.NegativeInfinity;

        for (int rotation = 0; rotation < 4; rotation++)
        {
            Shape shape = Shapes[piece][rotation];
            if (shape.Width > board.Width || shape.Height > board.Height)
            {
                continue;
            }

            int maxColumn = board.Width - shape.Width;
            for (int column = 0; column <= maxColumn; column++)
            {
                if (!board.TrySimulate(shape, column, out BoardState nextBoard, out int cleared, out int landingY))
                {
                    continue;
                }

                double score = ScoreBoard(nextBoard, cleared, column, shape, turn, totalPieces);
                if (useLookahead)
                {
                    score += futureWeight * EvaluateFuture(nextBoard, turn, totalPieces);
                }

                double tiebreak = -nextBoard.MaxHeight * 1000.0
                                  - nextBoard.Holes * 25.0
                                  - Math.Abs(column + (shape.Width - 1) * 0.5 - (board.Width - 1) * 0.5) * 2.0
                                  - landingY * 0.1;

                if (score > bestScore || (score == bestScore && tiebreak > bestTiebreak))
                {
                    bestScore = score;
                    bestTiebreak = tiebreak;
                    best = new Move(rotation, column);
                }
            }
        }

        return best;
    }

    private static double EvaluateFuture(BoardState board, int turn, int totalPieces)
    {
        double sum = 0.0;
        int considered = 0;

        for (int piece = 0; piece < 7; piece++)
        {
            double best = double.NegativeInfinity;
            bool found = false;

            for (int rotation = 0; rotation < 4; rotation++)
            {
                Shape shape = Shapes[piece][rotation];
                if (shape.Width > board.Width || shape.Height > board.Height)
                {
                    continue;
                }

                int maxColumn = board.Width - shape.Width;
                for (int column = 0; column <= maxColumn; column++)
                {
                    if (!board.TrySimulate(shape, column, out BoardState nextBoard, out int cleared, out _))
                    {
                        continue;
                    }

                    double score = ScoreBoard(nextBoard, cleared, column, shape, turn + 1, totalPieces);
                    if (score > best)
                    {
                        best = score;
                    }

                    found = true;
                }
            }

            if (!found)
            {
                best = -1_000_000.0;
            }

            sum += best;
            considered++;
        }

        return considered == 0 ? 0.0 : sum / considered;
    }

    private static double ScoreBoard(BoardState board, int cleared, int column, Shape shape, int turn, int totalPieces)
    {
        int aggregateHeight = 0;
        int maxHeight = 0;
        for (int i = 0; i < board.Width; i++)
        {
            int h = board.Heights[i];
            aggregateHeight += h;
            if (h > maxHeight)
            {
                maxHeight = h;
            }
        }

        int holes = board.Holes;
        int bumpiness = ComputeBumpiness(board.Heights);
        int wells = ComputeWells(board.Heights, board.Height);
        double centerDist = Math.Abs((column + (shape.Width - 1) * 0.5) - ((board.Width - 1) * 0.5));

        double complexity = ComputeComplexity(board);

        double lineWeight;
        double aggregateWeight;
        double holeWeight;
        double bumpWeight;
        double maxWeight;
        double wellWeight;
        double centerWeight;

        if (complexity < 0.28)
        {
            lineWeight = 1200.0;
            aggregateWeight = 0.90;
            holeWeight = 75.0;
            bumpWeight = 1.30;
            maxWeight = 2.25;
            wellWeight = 0.70;
            centerWeight = 4.00;
        }
        else if (complexity < 0.62)
        {
            lineWeight = 1450.0;
            aggregateWeight = 1.15;
            holeWeight = 150.0;
            bumpWeight = 2.10;
            maxWeight = 4.75;
            wellWeight = 1.35;
            centerWeight = 1.60;
        }
        else
        {
            lineWeight = 1850.0;
            aggregateWeight = 1.85;
            holeWeight = 520.0;
            bumpWeight = 4.20;
            maxWeight = 9.50;
            wellWeight = 3.25;
            centerWeight = 0.20;
        }

        double score = 0.0;
        score += cleared * lineWeight;
        score += cleared * cleared * 120.0;
        score -= aggregateHeight * aggregateWeight;
        score -= holes * holeWeight;
        score -= bumpiness * bumpWeight;
        score -= maxHeight * maxWeight;
        score -= wells * wellWeight;
        score -= centerDist * centerWeight;

        if (turn > totalPieces - 12)
        {
            score += cleared * 50.0;
            score -= maxHeight * 1.5;
        }

        return score;
    }

    private static double ComputeComplexity(BoardState board)
    {
        int maxHeight = board.MaxHeight;
        int holes = board.Holes;
        int bumpiness = ComputeBumpiness(board.Heights);

        double heightRatio = board.Height <= 0 ? 0.0 : maxHeight / (double)board.Height;
        double holeRatio = board.Width == 0 || board.Height == 0 ? 0.0 : holes / (double)(board.Width * board.Height);
        double roughRatio = board.Width < 2 || board.Height == 0 ? 0.0 : bumpiness / (double)((board.Width - 1) * board.Height);

        double complexity = heightRatio * 0.60 + holeRatio * 0.25 + roughRatio * 0.15;
        if (complexity < 0.0)
        {
            return 0.0;
        }

        if (complexity > 1.0)
        {
            return 1.0;
        }

        return complexity;
    }

    private static int ComputeBumpiness(int[] heights)
    {
        int bumpiness = 0;
        for (int i = 0; i + 1 < heights.Length; i++)
        {
            bumpiness += Math.Abs(heights[i] - heights[i + 1]);
        }

        return bumpiness;
    }

    private static int ComputeWells(int[] heights, int boardHeight)
    {
        int wells = 0;
        for (int i = 0; i < heights.Length; i++)
        {
            int left = i == 0 ? boardHeight : heights[i - 1];
            int right = i + 1 == heights.Length ? boardHeight : heights[i + 1];
            int depth = Math.Min(left, right) - heights[i];
            if (depth > 0)
            {
                wells += depth * (depth + 1) / 2;
            }
        }

        return wells;
    }

    private static int PieceIndex(char c)
    {
        return c switch
        {
            'I' => 0,
            'O' => 1,
            'T' => 2,
            'S' => 3,
            'Z' => 4,
            'J' => 5,
            'L' => 6,
            _ => 2
        };
    }

    private static Shape[][] BuildShapes()
    {
        return new[]
        {
            BuildRotations(new (int x, int y)[] { (0, 0), (1, 0), (2, 0), (3, 0) }),
            BuildRotations(new (int x, int y)[] { (0, 0), (1, 0), (0, 1), (1, 1) }),
            BuildRotations(new (int x, int y)[] { (0, 0), (1, 0), (2, 0), (1, 1) }),
            BuildRotations(new (int x, int y)[] { (1, 0), (2, 0), (0, 1), (1, 1) }),
            BuildRotations(new (int x, int y)[] { (0, 0), (1, 0), (1, 1), (2, 1) }),
            BuildRotations(new (int x, int y)[] { (0, 0), (0, 1), (1, 1), (2, 1) }),
            BuildRotations(new (int x, int y)[] { (2, 0), (0, 1), (1, 1), (2, 1) })
        };
    }

    private static Shape[] BuildRotations((int x, int y)[] cells)
    {
        var result = new Shape[4];
        result[0] = NormalizeShape(cells);
        for (int i = 1; i < 4; i++)
        {
            result[i] = RotateShape(result[i - 1]);
        }

        return result;
    }

    private static Shape RotateShape(Shape shape)
    {
        var rotated = new (int x, int y)[shape.CellsX.Length];
        int height = shape.Height;
        for (int i = 0; i < shape.CellsX.Length; i++)
        {
            rotated[i] = (height - 1 - shape.CellsY[i], shape.CellsX[i]);
        }

        return NormalizeShape(rotated);
    }

    private static Shape NormalizeShape((int x, int y)[] cells)
    {
        int minX = int.MaxValue;
        int minY = int.MaxValue;
        int maxX = int.MinValue;
        int maxY = int.MinValue;

        for (int i = 0; i < cells.Length; i++)
        {
            if (cells[i].x < minX) minX = cells[i].x;
            if (cells[i].y < minY) minY = cells[i].y;
            if (cells[i].x > maxX) maxX = cells[i].x;
            if (cells[i].y > maxY) maxY = cells[i].y;
        }

        var xs = new int[cells.Length];
        var ys = new int[cells.Length];
        for (int i = 0; i < cells.Length; i++)
        {
            xs[i] = cells[i].x - minX;
            ys[i] = cells[i].y - minY;
        }

        return new Shape(maxX - minX + 1, maxY - minY + 1, xs, ys);
    }

    private readonly struct Move
    {
        public readonly int Rotation;
        public readonly int Column;

        public Move(int rotation, int column)
        {
            Rotation = rotation;
            Column = column;
        }
    }

    private sealed class Shape
    {
        public readonly int Width;
        public readonly int Height;
        public readonly int[] CellsX;
        public readonly int[] CellsY;

        public Shape(int width, int height, int[] cellsX, int[] cellsY)
        {
            Width = width;
            Height = height;
            CellsX = cellsX;
            CellsY = cellsY;
        }
    }

    private sealed class BoardState
    {
        private readonly ulong _fullMask;

        public int Width { get; }
        public int Height { get; }
        public List<ulong> Rows { get; private set; }
        public int[] Heights { get; private set; }
        public int MaxHeight { get; private set; }
        public int Holes { get; private set; }

        public BoardState(int width, int height)
        {
            Width = width;
            Height = height;
            Rows = new List<ulong>();
            Heights = new int[width];
            _fullMask = width == 64 ? ulong.MaxValue : ((1UL << width) - 1UL);
            MaxHeight = 0;
            Holes = 0;
        }

        private BoardState(BoardState other)
        {
            Width = other.Width;
            Height = other.Height;
            Rows = new List<ulong>(other.Rows);
            Heights = (int[])other.Heights.Clone();
            _fullMask = other._fullMask;
            MaxHeight = other.MaxHeight;
            Holes = other.Holes;
        }

        public bool TrySimulate(Shape shape, int column, out BoardState next, out int cleared, out int landingY)
        {
            landingY = ComputeLandingY(shape, column);
            if (landingY < 0 || landingY + shape.Height > Height)
            {
                next = null;
                cleared = 0;
                return false;
            }

            next = new BoardState(this);
            next.ApplyInPlace(shape, column, landingY, out cleared);
            return true;
        }

        public bool TryApply(Shape shape, int column, out int cleared)
        {
            int landingY = ComputeLandingY(shape, column);
            if (landingY < 0 || landingY + shape.Height > Height)
            {
                cleared = 0;
                return false;
            }

            ApplyInPlace(shape, column, landingY, out cleared);
            return true;
        }

        private int ComputeLandingY(Shape shape, int column)
        {
            int y = 0;
            for (int i = 0; i < shape.CellsX.Length; i++)
            {
                int col = column + shape.CellsX[i];
                int required = Heights[col] - shape.CellsY[i];
                if (required > y)
                {
                    y = required;
                }
            }

            return y;
        }

        private void ApplyInPlace(Shape shape, int column, int landingY, out int cleared)
        {
            int neededRows = landingY + shape.Height;
            while (Rows.Count < neededRows)
            {
                Rows.Add(0UL);
            }

            for (int i = 0; i < shape.CellsX.Length; i++)
            {
                int row = landingY + shape.CellsY[i];
                int col = column + shape.CellsX[i];
                Rows[row] |= 1UL << col;
            }

            if (Rows.Count > 0)
            {
                var compact = new List<ulong>(Rows.Count);
                cleared = 0;
                for (int i = 0; i < Rows.Count; i++)
                {
                    ulong rowMask = Rows[i];
                    if (rowMask == _fullMask)
                    {
                        cleared++;
                        continue;
                    }

                    compact.Add(rowMask);
                }

                Rows = compact;
            }
            else
            {
                cleared = 0;
            }

            int last = Rows.Count - 1;
            while (last >= 0 && Rows[last] == 0UL)
            {
                last--;
            }

            if (last + 1 < Rows.Count)
            {
                Rows.RemoveRange(last + 1, Rows.Count - last - 1);
            }

            RecomputeHeights();
        }

        private void RecomputeHeights()
        {
            Array.Clear(Heights, 0, Heights.Length);
            MaxHeight = 0;

            for (int row = 0; row < Rows.Count; row++)
            {
                ulong mask = Rows[row];
                while (mask != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(mask);
                    if (bit < Heights.Length)
                    {
                        int h = row + 1;
                        Heights[bit] = h;
                        if (h > MaxHeight)
                        {
                            MaxHeight = h;
                        }
                    }

                    mask &= mask - 1;
                }
            }

            Holes = ComputeHoles();
        }

        private int ComputeHoles()
        {
            int holes = 0;
            for (int col = 0; col < Width; col++)
            {
                ulong bit = 1UL << col;
                bool seenBlock = false;
                for (int row = 0; row < Rows.Count; row++)
                {
                    if ((Rows[row] & bit) != 0)
                    {
                        seenBlock = true;
                    }
                    else if (seenBlock)
                    {
                        holes++;
                    }
                }
            }

            return holes;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _offset;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public string Next()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1)
                {
                    return null;
                }
            }
            while (c <= 32);

            var chars = new System.Text.StringBuilder();
            while (c > 32 && c != -1)
            {
                chars.Append((char)c);
                c = Read();
            }

            return chars.ToString();
        }

        public int NextInt()
        {
            string token = Next();
            return token == null ? 0 : int.Parse(token);
        }

        private int Read()
        {
            if (_offset >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _offset = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_offset++];
        }
    }
}
