using System;
using System.IO;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _ptr;
        private int _len;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_ptr++];
        }

        public long NextLong()
        {
            int c;
            do
            {
                c = Read();
            }
            while (c <= 32 && c >= 0);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        public int NextInt() => (int)NextLong();
    }

    private sealed class FastOutput : IDisposable
    {
        private readonly Stream _stream = new BufferedStream(Console.OpenStandardOutput(), 1 << 20);
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _ptr;

        public void WriteByte(byte b)
        {
            if (_ptr == _buffer.Length)
            {
                Flush();
            }

            _buffer[_ptr++] = b;
        }

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            uint v = (uint)value;
            Span<byte> tmp = stackalloc byte[11];
            int len = 0;
            while (v != 0)
            {
                tmp[len++] = (byte)('0' + (v % 10));
                v /= 10;
            }

            for (int i = len - 1; i >= 0; i--)
            {
                WriteByte(tmp[i]);
            }
        }

        public void WriteLine() => WriteByte((byte)'\n');

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buffer, 0, _ptr);
                _ptr = 0;
            }
        }

        public void Dispose()
        {
            Flush();
            _stream.Flush();
        }
    }

    private static long[] _key = Array.Empty<long>();
    private static int[] _prio = Array.Empty<int>();
    private static int[] _left = Array.Empty<int>();
    private static int[] _right = Array.Empty<int>();
    private static int _nodeCount;
    private static uint _rng = 2463534242u;

    private static int NextRand()
    {
        _rng ^= _rng << 13;
        _rng ^= _rng >> 17;
        _rng ^= _rng << 5;
        return (int)_rng & int.MaxValue;
    }

    private static int Compare(long keyA, int idA, long keyB, int idB)
    {
        if (keyA < keyB) return -1;
        if (keyA > keyB) return 1;
        if (idA < idB) return -1;
        if (idA > idB) return 1;
        return 0;
    }

    private static int Merge(int a, int b)
    {
        if (a == 0) return b;
        if (b == 0) return a;

        if (_prio[a] > _prio[b])
        {
            _right[a] = Merge(_right[a], b);
            return a;
        }
        else
        {
            _left[b] = Merge(a, _left[b]);
            return b;
        }
    }

    private static void Split(int root, long key, int id, out int leftTree, out int rightTree)
    {
        if (root == 0)
        {
            leftTree = 0;
            rightTree = 0;
            return;
        }

        if (Compare(_key[root], root, key, id) < 0)
        {
            Split(_right[root], key, id, out _right[root], out rightTree);
            leftTree = root;
        }
        else
        {
            Split(_left[root], key, id, out leftTree, out _left[root]);
            rightTree = root;
        }
    }

    private static int Insert(int root, int node)
    {
        if (root == 0)
        {
            return node;
        }

        if (_prio[node] > _prio[root])
        {
            Split(root, _key[node], node, out _left[node], out _right[node]);
            return node;
        }

        if (Compare(_key[node], node, _key[root], root) < 0)
        {
            _left[root] = Insert(_left[root], node);
        }
        else
        {
            _right[root] = Insert(_right[root], node);
        }

        return root;
    }

    private static int Erase(int root, long key, int id)
    {
        if (root == 0)
        {
            return 0;
        }

        int cmp = Compare(key, id, _key[root], root);
        if (cmp < 0)
        {
            _left[root] = Erase(_left[root], key, id);
            return root;
        }

        if (cmp > 0)
        {
            _right[root] = Erase(_right[root], key, id);
            return root;
        }

        return Merge(_left[root], _right[root]);
    }

    private static int LowerBound(int root, long target)
    {
        int current = root;
        int best = 0;
        while (current != 0)
        {
            if (_key[current] >= target)
            {
                best = current;
                current = _left[current];
            }
            else
            {
                current = _right[current];
            }
        }

        return best;
    }

    private static void RadixSort(long[] keys, int[] values)
    {
        int n = keys.Length;
        if (n <= 1)
        {
            return;
        }

        long[] tmpKeys = new long[n];
        int[] tmpValues = new int[n];
        int[] count = new int[1 << 16];

        for (int pass = 0; pass < 4; pass++)
        {
            Array.Clear(count, 0, count.Length);
            int shift = pass * 16;

            for (int i = 0; i < n; i++)
            {
                int bucket = (int)(((ulong)keys[i] >> shift) & 0xFFFFUL);
                count[bucket]++;
            }

            int sum = 0;
            for (int i = 0; i < count.Length; i++)
            {
                int c = count[i];
                count[i] = sum;
                sum += c;
            }

            for (int i = 0; i < n; i++)
            {
                int bucket = (int)(((ulong)keys[i] >> shift) & 0xFFFFUL);
                int pos = count[bucket]++;
                tmpKeys[pos] = keys[i];
                tmpValues[pos] = values[i];
            }

            var swapKeys = keys;
            keys = tmpKeys;
            tmpKeys = swapKeys;

            var swapValues = values;
            values = tmpValues;
            tmpValues = swapValues;
        }
    }

    public static void Main()
    {
        var input = new FastScanner();
        int n = input.NextInt();
        long stockLength = input.NextLong();

        long[] lengths = new long[n];
        int[] indices = new int[n];
        int[] binOfCut = new int[n];

        for (int i = 0; i < n; i++)
        {
            lengths[i] = input.NextLong();
            indices[i] = i;
        }

        RadixSort(lengths, indices);

        _key = new long[n + 1];
        _prio = new int[n + 1];
        _left = new int[n + 1];
        _right = new int[n + 1];
        _nodeCount = 0;

        int root = 0;
        int binCount = 0;

        for (int p = n - 1; p >= 0; p--)
        {
            long len = lengths[p];
            int originalIndex = indices[p];

            int node = LowerBound(root, len);
            if (node == 0)
            {
                node = ++_nodeCount;
                _key[node] = stockLength - len;
                _prio[node] = NextRand();
                _left[node] = 0;
                _right[node] = 0;
                root = Insert(root, node);
                binOfCut[originalIndex] = binCount++;
            }
            else
            {
                root = Erase(root, _key[node], node);
                _key[node] -= len;
                root = Insert(root, node);
                binOfCut[originalIndex] = node - 1;
            }
        }

        int[] counts = new int[binCount];
        for (int i = 0; i < n; i++)
        {
            counts[binOfCut[i]]++;
        }

        int[] start = new int[binCount + 1];
        for (int i = 0; i < binCount; i++)
        {
            start[i + 1] = start[i] + counts[i];
        }

        int[] next = new int[binCount];
        Array.Copy(start, next, binCount);

        int[] ordered = new int[n];
        for (int i = 0; i < n; i++)
        {
            int b = binOfCut[i];
            ordered[next[b]++] = i;
        }

        using var output = new FastOutput();
        output.WriteInt(binCount);
        output.WriteLine();

        for (int b = 0; b < binCount; b++)
        {
            int begin = start[b];
            int end = start[b + 1];
            if (begin < end)
            {
                output.WriteInt(ordered[begin]);
                for (int i = begin + 1; i < end; i++)
                {
                    output.WriteByte((byte)' ');
                    output.WriteInt(ordered[i]);
                }
            }
            output.WriteLine();
        }
    }
}
