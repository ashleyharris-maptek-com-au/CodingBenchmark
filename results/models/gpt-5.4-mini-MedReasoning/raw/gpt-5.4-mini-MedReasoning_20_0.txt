use std::io::{self, Read};

struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn skip_ws(&mut self) {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }

    fn next_token(&mut self) -> Option<&str> {
        self.skip_ws();
        if self.idx >= self.buf.len() {
            return None;
        }
        let start = self.idx;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        std::str::from_utf8(&self.buf[start..self.idx]).ok()
    }

    fn next_usize(&mut self) -> usize {
        self.next_token().and_then(|t| t.parse().ok()).unwrap_or(0)
    }

    fn next_f64(&mut self) -> f64 {
        self.next_token().and_then(|t| t.parse().ok()).unwrap_or(f64::NAN)
    }

    fn next_string(&mut self) -> String {
        self.next_token().unwrap_or("").to_string()
    }

    fn next_opt_f64(&mut self) -> Option<f64> {
        let t = self.next_token()?;
        if t.eq_ignore_ascii_case("na")
            || t.eq_ignore_ascii_case("nan")
            || t.eq_ignore_ascii_case("null")
            || t == "-"
        {
            None
        } else {
            t.parse::<f64>().ok().filter(|v| v.is_finite())
        }
    }
}

#[derive(Clone)]
struct Sample {
    hole_idx: usize,
    sample_idx_in_hole: usize,
    depth: f64,
    pos: [f64; 3],
}

#[derive(Clone)]
struct Suspect {
    hole_idx: usize,
    sample_idx: usize,
    prop_idx: usize,
    confidence: f64,
}

#[derive(Clone, Copy)]
struct PropStats {
    center: f64,
    scale: f64,
    count: usize,
}

fn median_in_place(values: &mut [f64]) -> f64 {
    let n = values.len();
    if n == 0 {
        return f64::NAN;
    }
    let mid = n / 2;
    values.select_nth_unstable_by(mid, |a, b| a.total_cmp(b));
    if n % 2 == 1 {
        values[mid]
    } else {
        let upper = values[mid];
        values[..mid].select_nth_unstable_by(mid - 1, |a, b| a.total_cmp(b));
        (values[mid - 1] + upper) * 0.5
    }
}

fn robust_scale(values: &[f64], center: f64) -> f64 {
    if values.len() < 2 {
        return (center.abs() + 1.0).max(1.0) * 1e-6;
    }
    let mut devs: Vec<f64> = values.iter().map(|v| (v - center).abs()).collect();
    let mad = median_in_place(&mut devs);
    let mut scale = 1.4826 * mad;
    let floor = (center.abs() + 1.0) * 1e-9;
    if !scale.is_finite() || scale < floor {
        scale = floor;
    }
    scale
}

fn score_from_center(value: f64, center: f64, scale: f64) -> f64 {
    if !value.is_finite() || !center.is_finite() || !scale.is_finite() || scale <= 0.0 {
        return 0.0;
    }
    (value - center).abs() / scale
}

fn confidence_from_score(score: f64) -> f64 {
    if !score.is_finite() || score <= 0.0 {
        0.0
    } else {
        1.0 - (-0.5 * score * score).exp()
    }
}

fn local_score(
    order: &[usize],
    pos: usize,
    values: &[Option<f64>],
    window: usize,
    global_scale: f64,
) -> Option<f64> {
    let current = values[order[pos]]?;
    let mut neigh = Vec::with_capacity(window * 2);
    for step in 1..=window {
        if pos >= step {
            if let Some(v) = values[order[pos - step]] {
                neigh.push(v);
            }
        }
        if pos + step < order.len() {
            if let Some(v) = values[order[pos + step]] {
                neigh.push(v);
            }
        }
    }
    if neigh.len() < 2 {
        return None;
    }
    let center = median_in_place(&mut neigh);
    let mut devs: Vec<f64> = neigh.iter().map(|v| (v - center).abs()).collect();
    let mut scale = 1.4826 * median_in_place(&mut devs);
    let floor = global_scale.max((center.abs() + 1.0) * 1e-9);
    if !scale.is_finite() || scale < floor {
        scale = floor;
    }
    Some(score_from_center(current, center, scale))
}

fn main() {
    let mut input = FastInput::new();
    let num_holes = input.next_usize();
    let num_props = input.next_usize();

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(input.next_string());
    }

    let mut hole_ids = Vec::with_capacity(num_holes);
    let mut hole_samples: Vec<Vec<usize>> = Vec::with_capacity(num_holes);
    let mut samples: Vec<Sample> = Vec::new();
    let mut prop_values: Vec<Vec<Option<f64>>> = vec![Vec::new(); num_props];

    for hole_idx in 0..num_holes {
        let hole_id = input.next_string();
        let start_x = input.next_f64();
        let start_y = input.next_f64();
        let start_z = input.next_f64();
        let dir_x = input.next_f64();
        let dir_y = input.next_f64();
        let dir_z = input.next_f64();
        let _length = input.next_f64();
        let num_samples = input.next_usize();

        let norm = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let unit = if norm > 0.0 {
            [dir_x / norm, dir_y / norm, dir_z / norm]
        } else {
            [0.0, 0.0, 0.0]
        };

        hole_ids.push(hole_id);
        let mut indices = Vec::with_capacity(num_samples);
        for sample_idx_in_hole in 0..num_samples {
            let depth = input.next_f64();
            let mut vals = Vec::with_capacity(num_props);
            for _ in 0..num_props {
                vals.push(input.next_opt_f64());
            }
            let pos = [
                start_x + unit[0] * depth,
                start_y + unit[1] * depth,
                start_z + unit[2] * depth,
            ];
            let sample_idx = samples.len();
            samples.push(Sample {
                hole_idx,
                sample_idx_in_hole,
                depth,
                pos,
            });
            for p in 0..num_props {
                prop_values[p].push(vals[p]);
            }
            indices.push(sample_idx);
            let _ = sample_idx_in_hole;
        }
        hole_samples.push(indices);
    }

    let total_samples = samples.len();
    if total_samples == 0 || num_props == 0 {
        println!("0");
        return;
    }

    let mut prop_stats = Vec::with_capacity(num_props);
    for p in 0..num_props {
        let mut vals: Vec<f64> = prop_values[p]
            .iter()
            .copied()
            .flatten()
            .filter(|v| v.is_finite())
            .collect();
        if vals.is_empty() {
            prop_stats.push(PropStats {
                center: 0.0,
                scale: 1.0,
                count: 0,
            });
            continue;
        }
        let center = median_in_place(&mut vals);
        let scale = robust_scale(&vals, center);
        prop_stats.push(PropStats {
            center,
            scale,
            count: vals.len(),
        });
    }

    let window = if total_samples <= 20_000 {
        5
    } else if total_samples <= 200_000 {
        4
    } else {
        3
    };

    let mut hole_orders: Vec<Vec<usize>> = Vec::with_capacity(num_holes);
    for hole_idx in 0..num_holes {
        let mut order = hole_samples[hole_idx].clone();
        order.sort_by(|&a, &b| {
            samples[a]
                .depth
                .total_cmp(&samples[b].depth)
                .then_with(|| a.cmp(&b))
        });
        hole_orders.push(order);
    }

    let mut suspects = Vec::new();
    for p in 0..num_props {
        let stats = prop_stats[p];
        if stats.count == 0 {
            continue;
        }

        let threshold = if stats.count < 20 { 2.2 } else { 2.7 };
        for hole_idx in 0..num_holes {
            let order = &hole_orders[hole_idx];
            for pos in 0..order.len() {
                let sample_idx = order[pos];
                let Some(value) = prop_values[p][sample_idx] else {
                    continue;
                };
                let global_score = score_from_center(value, stats.center, stats.scale);
                let mut best_score = global_score;
                if let Some(local) = local_score(order, pos, &prop_values[p], window, stats.scale) {
                    if local.is_finite() {
                        best_score = best_score.max(local);
                    }
                }

                if best_score >= threshold {
                    let confidence = confidence_from_score(best_score);
                    suspects.push(Suspect {
                        hole_idx,
                        sample_idx: samples[sample_idx].sample_idx_in_hole,
                        prop_idx: p,
                        confidence,
                    });
                }
            }
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .total_cmp(&a.confidence)
            .then_with(|| a.hole_idx.cmp(&b.hole_idx))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.prop_idx.cmp(&b.prop_idx))
    });

    let mut out = String::new();
    out.push_str(&format!("{}\n", suspects.len()));
    for s in suspects {
        out.push_str(&hole_ids[s.hole_idx]);
        out.push(' ');
        out.push_str(&s.sample_idx.to_string());
        out.push(' ');
        out.push_str(&prop_names[s.prop_idx]);
        out.push(' ');
        out.push_str(&format!("{:.6}\n", s.confidence));
    }
    print!("{out}");
}
