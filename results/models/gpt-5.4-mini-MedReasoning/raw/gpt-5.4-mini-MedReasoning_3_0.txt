use std::cmp::Reverse;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read, Write};

struct Scanner {
    input: Vec<u8>,
    index: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            input: input.into_bytes(),
            index: 0,
        }
    }

    fn next_usize(&mut self) -> usize {
        while self.index < self.input.len() && self.input[self.index].is_ascii_whitespace() {
            self.index += 1;
        }
        let mut value = 0usize;
        while self.index < self.input.len() && !self.input[self.index].is_ascii_whitespace() {
            value = value * 10 + (self.input[self.index] - b'0') as usize;
            self.index += 1;
        }
        value
    }
}

struct BfsResult {
    far_local: usize,
    far_global: usize,
    dist: Vec<i32>,
    parent: Vec<usize>,
}

fn bfs_farthest(
    start_global: usize,
    cid: usize,
    comp_of: &[usize],
    comp_pos: &[usize],
    comp_nodes: &[usize],
    adj: &[Vec<usize>],
) -> BfsResult {
    let k = comp_nodes.len();
    let mut dist = vec![-1i32; k];
    let mut parent = vec![usize::MAX; k];
    let start_local = comp_pos[start_global];
    let mut q = VecDeque::new();
    dist[start_local] = 0;
    q.push_back(start_local);
    let mut far_local = start_local;

    while let Some(u_local) = q.pop_front() {
        let u = comp_nodes[u_local];
        let nd = dist[u_local] + 1;
        for &v in &adj[u] {
            if comp_of[v] != cid {
                continue;
            }
            let v_local = comp_pos[v];
            if dist[v_local] == -1 {
                dist[v_local] = nd;
                parent[v_local] = u_local;
                q.push_back(v_local);
                if dist[v_local] > dist[far_local] {
                    far_local = v_local;
                }
            }
        }
    }

    BfsResult {
        far_local,
        far_global: comp_nodes[far_local],
        dist,
        parent,
    }
}

fn refresh_positions(layer: &[usize], pos: &mut [f64]) {
    for (i, &u) in layer.iter().enumerate() {
        pos[u] = i as f64;
    }
}

fn sort_layer_by_reference(
    layer: &mut Vec<usize>,
    ref_depth: i32,
    depth: &[i32],
    pos: &[f64],
    cid: usize,
    comp_of: &[usize],
    adj: &[Vec<usize>],
) {
    let mut keyed: Vec<(f64, usize)> = Vec::with_capacity(layer.len());
    for &u in layer.iter() {
        let mut sum = 0.0;
        let mut cnt = 0.0;
        for &v in &adj[u] {
            if comp_of[v] == cid && depth[v] == ref_depth {
                sum += pos[v];
                cnt += 1.0;
            }
        }
        let key = if cnt > 0.0 { sum / cnt } else { pos[u] };
        keyed.push((key, u));
    }
    keyed.sort_by(|a, b| a.0.total_cmp(&b.0).then_with(|| a.1.cmp(&b.1)));
    layer.clear();
    layer.extend(keyed.into_iter().map(|(_, u)| u));
}

fn main() {
    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let mut comp_of = vec![usize::MAX; n];
    let mut comp_pos = vec![0usize; n];
    let mut components: Vec<Vec<usize>> = Vec::new();

    for v in 0..n {
        if comp_of[v] != usize::MAX {
            continue;
        }
        let cid = components.len();
        let mut stack = vec![v];
        comp_of[v] = cid;
        let mut nodes = Vec::new();

        while let Some(u) = stack.pop() {
            nodes.push(u);
            for &w in &adj[u] {
                if comp_of[w] == usize::MAX {
                    comp_of[w] = cid;
                    stack.push(w);
                }
            }
        }

        for (i, &u) in nodes.iter().enumerate() {
            comp_pos[u] = i;
        }
        components.push(nodes);
    }

    let mut order: Vec<usize> = (0..components.len()).collect();
    order.sort_by_key(|&cid| Reverse(components[cid].len()));

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];
    let mut depth = vec![-1i32; n];
    let mut pos = vec![0.0f64; n];

    let mut cur_x = 0.0f64;
    let mut cur_y = 0.0f64;
    let mut row_h = 0.0f64;
    let row_limit = 20_000.0f64;
    let margin = 120.0f64;

    for cid in order {
        let comp_nodes = &components[cid];
        let k = comp_nodes.len();

        if k == 1 {
            let side = 200.0f64;
            if cur_x > 0.0 && cur_x + side > row_limit {
                cur_x = 0.0;
                cur_y += row_h + margin;
                row_h = 0.0;
            }
            let ox = cur_x + side * 0.5;
            let oy = cur_y + side * 0.5;
            let u = comp_nodes[0];
            x[u] = ox;
            y[u] = oy;
            cur_x += side + margin;
            row_h = row_h.max(side);
            continue;
        }

        let mut start = comp_nodes[0];
        for &u in comp_nodes.iter() {
            if adj[u].len() > adj[start].len() {
                start = u;
            }
        }

        let bfs1 = bfs_farthest(start, cid, &comp_of, &comp_pos, comp_nodes, &adj);
        let bfs2 = bfs_farthest(bfs1.far_global, cid, &comp_of, &comp_pos, comp_nodes, &adj);

        let mut path = Vec::new();
        let mut cur = bfs2.far_local;
        loop {
            path.push(comp_nodes[cur]);
            if bfs2.parent[cur] == usize::MAX {
                break;
            }
            cur = bfs2.parent[cur];
        }
        let root = path[path.len() / 2];

        let bfs = bfs_farthest(root, cid, &comp_of, &comp_pos, comp_nodes, &adj);
        let max_depth = *bfs.dist.iter().max().unwrap_or(&0) as usize;
        let mut layers: Vec<Vec<usize>> = vec![Vec::new(); max_depth + 1];

        for (local, &d) in bfs.dist.iter().enumerate() {
            let u = comp_nodes[local];
            depth[u] = d;
            layers[d as usize].push(u);
        }

        refresh_positions(&layers[0], &mut pos);

        let sweeps = if k < 64 {
            8
        } else if k < 512 {
            6
        } else if k < 5_000 {
            4
        } else {
            2
        };

        for _ in 0..sweeps {
            for l in 1..layers.len() {
                sort_layer_by_reference(
                    &mut layers[l],
                    (l - 1) as i32,
                    &depth,
                    &pos,
                    cid,
                    &comp_of,
                    &adj,
                );
                refresh_positions(&layers[l], &mut pos);
            }
            for l in (0..layers.len().saturating_sub(1)).rev() {
                sort_layer_by_reference(
                    &mut layers[l],
                    (l + 1) as i32,
                    &depth,
                    &pos,
                    cid,
                    &comp_of,
                    &adj,
                );
                refresh_positions(&layers[l], &mut pos);
            }
        }

        let max_layer_size = layers.iter().map(|v| v.len()).max().unwrap_or(1);
        let scale =
            40.0 + 0.18 * (max_layer_size as f64) + 2.0 * (max_depth as f64) + 2.5 * (k as f64).sqrt();
        let side = 2.0 * scale + 100.0;

        if cur_x > 0.0 && cur_x + side > row_limit {
            cur_x = 0.0;
            cur_y += row_h + margin;
            row_h = 0.0;
        }

        let ox = cur_x + side * 0.5;
        let oy = cur_y + side * 0.5;
        let max_depth_f = max_depth.max(1) as f64;

        for (l, layer) in layers.iter().enumerate() {
            let frac = if max_depth == 0 {
                0.0
            } else {
                (l as f64 / max_depth_f).sqrt()
            };
            let r = scale * frac;
            let cnt = layer.len();
            for (i, &u) in layer.iter().enumerate() {
                let ang = if cnt == 1 {
                    0.0
                } else {
                    2.0 * PI * (i as f64) / (cnt as f64)
                };
                x[u] = ox + r * ang.cos();
                y[u] = oy + r * ang.sin();
            }
        }

        cur_x += side + margin;
        row_h = row_h.max(side);
    }

    let mut out = io::BufWriter::new(io::stdout());
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", x[i], y[i]).unwrap();
    }
}
