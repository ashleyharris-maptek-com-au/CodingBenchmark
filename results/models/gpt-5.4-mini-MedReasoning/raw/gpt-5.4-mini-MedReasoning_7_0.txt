static constexpr double EPS = 1e-8;
static constexpr double VTX_EPS = 1e-8;

struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;

    Vec3() = default;
    Vec3(double xx, double yy, double zz) : x(xx), y(yy), z(zz) {}

    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
    Vec3& operator+=(const Vec3& o) {
        x += o.x; y += o.y; z += o.z;
        return *this;
    }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double norm2(const Vec3& a) {
    return dot(a, a);
}

struct Plane {
    Vec3 n;
    double w = 0.0;

    Plane() = default;
    Plane(const Vec3& nn, double ww) : n(nn), w(ww) {}

    static Plane fromPoints(const Vec3& a, const Vec3& b, const Vec3& c) {
        Vec3 nn = cross(b - a, c - a);
        double len = std::sqrt(norm2(nn));
        if (len <= 0.0) {
            return Plane{};
        }
        nn = nn / len;
        return Plane(nn, dot(nn, a));
    }

    void flip() {
        n = n * -1.0;
        w = -w;
    }

    double dist(const Vec3& p) const {
        return dot(n, p) - w;
    }
};

struct Polygon {
    vector<Vec3> vertices;
    Plane plane;
};

static inline void sanitizePolygon(vector<Vec3>& v) {
    if (v.size() < 2) return;
    vector<Vec3> out;
    out.reserve(v.size());
    auto same = [](const Vec3& a, const Vec3& b) {
        return norm2(a - b) <= VTX_EPS * VTX_EPS;
    };
    for (const auto& p : v) {
        if (out.empty() || !same(out.back(), p)) {
            out.push_back(p);
        }
    }
    if (out.size() >= 2 && same(out.front(), out.back())) {
        out.pop_back();
    }
    v.swap(out);
}

struct BSPNode {
    bool hasPlane = false;
    Plane plane;
    vector<Polygon> polygons;
    unique_ptr<BSPNode> front;
    unique_ptr<BSPNode> back;

    enum Side : int { COPLANAR = 0, FRONT = 1, BACK = 2, SPANNING = 3 };

    void splitPolygon(const Polygon& polygon,
                      vector<Polygon>& coplanarFront,
                      vector<Polygon>& coplanarBack,
                      vector<Polygon>& frontPolys,
                      vector<Polygon>& backPolys) const
    {
        const auto& verts = polygon.vertices;
        const size_t n = verts.size();
        if (n < 3) return;

        int polygonType = 0;
        vector<int> types(n);
        vector<double> dists(n);
        for (size_t i = 0; i < n; ++i) {
            double d = plane.dist(verts[i]);
            dists[i] = d;
            int t = (d < -EPS) ? BACK : (d > EPS ? FRONT : COPLANAR);
            types[i] = t;
            polygonType |= t;
        }

        if (polygonType == COPLANAR) {
            if (dot(plane.n, polygon.plane.n) >= 0.0) {
                coplanarFront.push_back(polygon);
            } else {
                coplanarBack.push_back(polygon);
            }
        } else if (polygonType == FRONT) {
            frontPolys.push_back(polygon);
        } else if (polygonType == BACK) {
            backPolys.push_back(polygon);
        } else {
            vector<Vec3> f;
            vector<Vec3> b;
            f.reserve(n + 1);
            b.reserve(n + 1);
            for (size_t i = 0; i < n; ++i) {
                size_t j = (i + 1 == n) ? 0 : (i + 1);
                const Vec3& vi = verts[i];
                const Vec3& vj = verts[j];
                int ti = types[i];
                int tj = types[j];

                if (ti != BACK) f.push_back(vi);
                if (ti != FRONT) b.push_back(vi);

                if ((ti | tj) == SPANNING) {
                    double t = dists[i] / (dists[i] - dists[j]);
                    Vec3 v = vi + (vj - vi) * t;
                    f.push_back(v);
                    b.push_back(v);
                }
            }
            if (f.size() >= 3) {
                Polygon p;
                p.vertices = std::move(f);
                p.plane = polygon.plane;
                sanitizePolygon(p.vertices);
                if (p.vertices.size() >= 3) frontPolys.push_back(std::move(p));
            }
            if (b.size() >= 3) {
                Polygon p;
                p.vertices = std::move(b);
                p.plane = polygon.plane;
                sanitizePolygon(p.vertices);
                if (p.vertices.size() >= 3) backPolys.push_back(std::move(p));
            }
        }
    }

    void build(vector<Polygon> polys) {
        if (polys.empty()) return;
        if (!hasPlane) {
            plane = polys[0].plane;
            hasPlane = true;
        }

        vector<Polygon> frontList;
        vector<Polygon> backList;
        frontList.reserve(polys.size());
        backList.reserve(polys.size());

        for (const auto& p : polys) {
            splitPolygon(p, polygons, polygons, frontList, backList);
        }

        if (!frontList.empty()) {
            if (!front) front = make_unique<BSPNode>();
            front->build(std::move(frontList));
        }
        if (!backList.empty()) {
            if (!back) back = make_unique<BSPNode>();
            back->build(std::move(backList));
        }
    }

    vector<Polygon> clipPolygons(vector<Polygon> polys) const {
        if (!hasPlane) return polys;

        vector<Polygon> frontList;
        vector<Polygon> backList;
        frontList.reserve(polys.size());
        backList.reserve(polys.size());

        for (const auto& p : polys) {
            splitPolygon(p, frontList, backList, frontList, backList);
        }

        if (front) {
            frontList = front->clipPolygons(std::move(frontList));
        }
        if (back) {
            backList = back->clipPolygons(std::move(backList));
        } else {
            backList.clear();
        }

        frontList.insert(frontList.end(),
                         std::make_move_iterator(backList.begin()),
                         std::make_move_iterator(backList.end()));
        return frontList;
    }

    void clipTo(const BSPNode& other) {
        polygons = other.clipPolygons(std::move(polygons));
        if (front) front->clipTo(other);
        if (back) back->clipTo(other);
    }

    void invert() {
        for (auto& p : polygons) {
            std::reverse(p.vertices.begin(), p.vertices.end());
            p.plane.flip();
        }
        plane.flip();
        if (front) front->invert();
        if (back) back->invert();
        std::swap(front, back);
    }

    void collect(vector<Polygon>& out) const {
        out.insert(out.end(), polygons.begin(), polygons.end());
        if (front) front->collect(out);
        if (back) back->collect(out);
    }

    vector<Polygon> allPolygons() const {
        vector<Polygon> out;
        collect(out);
        return out;
    }
};

static inline vector<Polygon> makePolygons(const vector<Vec3>& verts, const vector<array<int, 3>>& faces) {
    vector<Polygon> polys;
    polys.reserve(faces.size());
    for (const auto& f : faces) {
        Polygon p;
        p.vertices = {verts[f[0]], verts[f[1]], verts[f[2]]};
        p.plane = Plane::fromPoints(p.vertices[0], p.vertices[1], p.vertices[2]);
        if (p.plane.n.x == 0.0 && p.plane.n.y == 0.0 && p.plane.n.z == 0.0) continue;
        polys.push_back(std::move(p));
    }
    return polys;
}

static inline long double signedVolume(const vector<Vec3>& verts, const vector<array<int, 3>>& faces) {
    long double vol = 0.0L;
    for (const auto& f : faces) {
        const Vec3& a = verts[f[0]];
        const Vec3& b = verts[f[1]];
        const Vec3& c = verts[f[2]];
        vol += (long double)dot(a, cross(b, c));
    }
    return vol / 6.0L;
}

struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    FILE* fp = stdin;
    char buf[BUFSIZE];
    size_t idx = 0, len = 0;

    inline char getChar() {
        if (idx >= len) {
            len = std::fread(buf, 1, BUFSIZE, fp);
            idx = 0;
            if (len == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T& out) {
        char c;
        do {
            c = getChar();
            if (!c) return false;
        } while (c <= ' ');
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = getChar();
        }
        T val = 0;
        while (c > ' ') {
            val = val * 10 + (c - '0');
            c = getChar();
        }
        out = neg ? -val : val;
        return true;
    }

    bool readDouble(double& out) {
        char c;
        do {
            c = getChar();
            if (!c) return false;
        } while (c <= ' ');

        bool neg = false;
        if (c == '-') {
            neg = true;
            c = getChar();
        } else if (c == '+') {
            c = getChar();
        }

        double val = 0.0;
        while (c >= '0' && c <= '9') {
            val = val * 10.0 + double(c - '0');
            c = getChar();
        }

        if (c == '.') {
            double base = 0.1;
            c = getChar();
            while (c >= '0' && c <= '9') {
                val += double(c - '0') * base;
                base *= 0.1;
                c = getChar();
            }
        }

        if (c == 'e' || c == 'E') {
            int expSign = 1;
            int exp = 0;
            c = getChar();
            if (c == '-') {
                expSign = -1;
                c = getChar();
            } else if (c == '+') {
                c = getChar();
            }
            while (c >= '0' && c <= '9') {
                exp = exp * 10 + (c - '0');
                c = getChar();
            }
            val *= std::pow(10.0, expSign * exp);
        }

        out = neg ? -val : val;
        return true;
    }
};

struct FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0;

    ~FastOutput() {
        flush();
    }

    inline void flush() {
        if (idx) {
            std::fwrite(buf, 1, idx, stdout);
            idx = 0;
        }
    }

    inline void putChar(char c) {
        if (idx == BUFSIZE) flush();
        buf[idx++] = c;
    }

    inline void append(const char* s, const char* e) {
        size_t n = size_t(e - s);
        while (n) {
            if (idx == BUFSIZE) flush();
            size_t room = BUFSIZE - idx;
            size_t m = std::min(room, n);
            std::memcpy(buf + idx, s, m);
            idx += m;
            s += m;
            n -= m;
        }
    }

    inline void writeInt(long long x) {
        char tmp[32];
        auto res = std::to_chars(tmp, tmp + 32, x);
        append(tmp, res.ptr);
    }

    inline void writeDouble(double x) {
        char tmp[64];
        auto res = std::to_chars(tmp, tmp + 64, x, std::chars_format::general, 17);
        if (res.ec == std::errc()) {
            append(tmp, res.ptr);
        } else {
            int n = std::snprintf(tmp, sizeof(tmp), "%.17g", x);
            append(tmp, tmp + n);
        }
    }
};

struct Key {
    long long x, y, z;
    bool operator==(const Key& o) const noexcept {
        return x == o.x && y == o.y && z == o.z;
    }
};

struct KeyHash {
    size_t operator()(const Key& k) const noexcept {
        auto mix = [](size_t h, size_t v) {
            h ^= v + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
            return h;
        };
        size_t h = 0;
        h = mix(h, std::hash<long long>{}(k.x));
        h = mix(h, std::hash<long long>{}(k.y));
        h = mix(h, std::hash<long long>{}(k.z));
        return h;
    }
};

struct FaceKey {
    int a, b, c;
    bool operator==(const FaceKey& o) const noexcept {
        return a == o.a && b == o.b && c == o.c;
    }
};

struct FaceKeyHash {
    size_t operator()(const FaceKey& f) const noexcept {
        auto mix = [](size_t h, size_t v) {
            h ^= v + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
            return h;
        };
        size_t h = 0;
        h = mix(h, std::hash<int>{}(f.a));
        h = mix(h, std::hash<int>{}(f.b));
        h = mix(h, std::hash<int>{}(f.c));
        return h;
    }
};

static inline Key quantize(const Vec3& v) {
    return {
        (long long)std::llround(v.x / VTX_EPS),
        (long long)std::llround(v.y / VTX_EPS),
        (long long)std::llround(v.z / VTX_EPS)
    };
}

static inline FaceKey faceKey(int a, int b, int c) {
    if (a > b) std::swap(a, b);
    if (b > c) std::swap(b, c);
    if (a > b) std::swap(a, b);
    return {a, b, c};
}

static inline vector<Polygon> booleanUnion(vector<Polygon> a, vector<Polygon> b) {
    BSPNode A;
    BSPNode B;
    A.build(std::move(a));
    B.build(std::move(b));
    A.clipTo(B);
    B.clipTo(A);
    B.invert();
    B.clipTo(A);
    B.invert();
    A.build(B.allPolygons());
    return A.allPolygons();
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    FastInput in;
    FastOutput out;

    int VA = 0, FA = 0, VB = 0, FB = 0;
    if (!in.readInt(VA)) return 0;
    in.readInt(FA);
    in.readInt(VB);
    in.readInt(FB);

    vector<Vec3> vertsA(VA), vertsB(VB);
    vector<array<int, 3>> facesA(FA), facesB(FB);

    for (int i = 0; i < VA; ++i) {
        double x, y, z;
        in.readDouble(x);
        in.readDouble(y);
        in.readDouble(z);
        vertsA[i] = {x, y, z};
    }
    for (int i = 0; i < FA; ++i) {
        int a, b, c;
        in.readInt(a);
        in.readInt(b);
        in.readInt(c);
        facesA[i] = {a, b, c};
    }
    for (int i = 0; i < VB; ++i) {
        double x, y, z;
        in.readDouble(x);
        in.readDouble(y);
        in.readDouble(z);
        vertsB[i] = {x, y, z};
    }
    for (int i = 0; i < FB; ++i) {
        int a, b, c;
        in.readInt(a);
        in.readInt(b);
        in.readInt(c);
        facesB[i] = {a, b, c};
    }

    if (signedVolume(vertsA, facesA) < 0.0L) {
        for (auto& f : facesA) std::swap(f[1], f[2]);
    }
    if (signedVolume(vertsB, facesB) < 0.0L) {
        for (auto& f : facesB) std::swap(f[1], f[2]);
    }

    vector<Polygon> polysA = makePolygons(vertsA, facesA);
    vector<Polygon> polysB = makePolygons(vertsB, facesB);

    vector<Polygon> result = booleanUnion(std::move(polysA), std::move(polysB));

    vector<Vec3> outVerts;
    vector<array<int, 3>> outFaces;
    outVerts.reserve(result.size() * 3);
    outFaces.reserve(result.size() * 2);

    std::unordered_map<Key, int, KeyHash> vmap;
    vmap.reserve(result.size() * 3);
    std::unordered_set<FaceKey, FaceKeyHash> fset;
    fset.reserve(result.size() * 2);

    auto addVertex = [&](const Vec3& v) -> int {
        Key k = quantize(v);
        auto it = vmap.find(k);
        if (it != vmap.end()) {
            return it->second;
        }
        int idx = (int)outVerts.size();
        outVerts.push_back(v);
        vmap.emplace(k, idx);
        return idx;
    };

    for (const auto& poly : result) {
        vector<Vec3> verts = poly.vertices;
        sanitizePolygon(verts);
        if (verts.size() < 3) continue;
        int base = addVertex(verts[0]);
        for (size_t i = 1; i + 1 < verts.size(); ++i) {
            const Vec3& a = verts[0];
            const Vec3& b = verts[i];
            const Vec3& c = verts[i + 1];
            Vec3 cr = cross(b - a, c - a);
            if (norm2(cr) <= 1e-20) continue;
            int ia = base;
            int ib = addVertex(b);
            int ic = addVertex(c);
            FaceKey key = faceKey(ia, ib, ic);
            if (fset.insert(key).second) {
                outFaces.push_back({ia, ib, ic});
            }
        }
    }

    out.writeInt((long long)outVerts.size());
    out.putChar(' ');
    out.writeInt((long long)outFaces.size());
    out.putChar('\n');

    for (const auto& v : outVerts) {
        out.writeDouble(v.x);
        out.putChar(' ');
        out.writeDouble(v.y);
        out.putChar(' ');
        out.writeDouble(v.z);
        out.putChar('\n');
    }
    for (const auto& f : outFaces) {
        out.writeInt(f[0]);
        out.putChar(' ');
        out.writeInt(f[1]);
        out.putChar(' ');
        out.writeInt(f[2]);
        out.putChar('\n');
    }

    out.flush();
    return 0;
}
