using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public sealed class Program
{
    private sealed class FastInput
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _length;
        private int _position;

        public int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }

        public void Skip(int count)
        {
            while (count-- > 0)
            {
                if (Read() < 0)
                {
                    throw new EndOfStreamException();
                }
            }
        }

        public string ReadLineAscii()
        {
            var bytes = new List<byte>(64);
            while (true)
            {
                int c = Read();
                if (c < 0 || c == '\n')
                {
                    break;
                }
                if (c != '\r')
                {
                    bytes.Add((byte)c);
                }
            }

            return Encoding.ASCII.GetString(bytes.ToArray());
        }
    }

    private struct Candidate
    {
        public int X;
        public int Y;
        public long Profit;
    }

    private sealed class CandidateHeap
    {
        private readonly Candidate[] _data;
        private int _count;

        public CandidateHeap(int capacity)
        {
            _data = new Candidate[capacity];
        }

        public int Count => _count;

        public void Push(Candidate candidate)
        {
            if (_data.Length == 0)
            {
                return;
            }

            if (_count < _data.Length)
            {
                _data[_count] = candidate;
                SiftUp(_count);
                _count++;
                return;
            }

            if (candidate.Profit <= _data[0].Profit)
            {
                return;
            }

            _data[0] = candidate;
            SiftDown(0);
        }

        public List<Candidate> ToList()
        {
            var list = new List<Candidate>(_count);
            for (int i = 0; i < _count; i++)
            {
                list.Add(_data[i]);
            }
            return list;
        }

        private void SiftUp(int index)
        {
            while (index > 0)
            {
                int parent = (index - 1) >> 1;
                if (_data[parent].Profit <= _data[index].Profit)
                {
                    break;
                }

                (_data[parent], _data[index]) = (_data[index], _data[parent]);
                index = parent;
            }
        }

        private void SiftDown(int index)
        {
            while (true)
            {
                int left = (index << 1) + 1;
                if (left >= _count)
                {
                    return;
                }

                int right = left + 1;
                int smallest = left;
                if (right < _count && _data[right].Profit < _data[left].Profit)
                {
                    smallest = right;
                }

                if (_data[index].Profit <= _data[smallest].Profit)
                {
                    return;
                }

                (_data[index], _data[smallest]) = (_data[smallest], _data[index]);
                index = smallest;
            }
        }
    }

    public static void Main()
    {
        var input = new FastInput();
        string header = input.ReadLineAscii();
        if (string.IsNullOrWhiteSpace(header))
        {
            return;
        }

        string[] parts = header.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int xCount = int.Parse(parts[0], CultureInfo.InvariantCulture);
        int yCount = int.Parse(parts[1], CultureInfo.InvariantCulture);
        int zCount = int.Parse(parts[2], CultureInfo.InvariantCulture);

        long topCellCount = (long)xCount * yCount;
        int zSkips = zCount - 1;

        // Small worlds: keep every profitable top voxel.
        // Larger worlds: keep only the best candidates so output stays bounded.
        int budget;
        if (topCellCount <= 10_000)
        {
            budget = int.MaxValue;
        }
        else if (topCellCount <= 250_000)
        {
            budget = 20_000;
        }
        else
        {
            budget = 5_000;
        }

        List<Candidate>? allCandidates = budget == int.MaxValue ? new List<Candidate>() : null;
        CandidateHeap? heap = budget == int.MaxValue ? null : new CandidateHeap(budget);

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                if (zSkips > 0)
                {
                    input.Skip(zSkips);
                }

                int voxel = input.Read();
                if (voxel < 0)
                {
                    throw new EndOfStreamException();
                }

                int valuePerTonne = voxel >> 4;
                int mass = voxel & 15;
                if (valuePerTonne == 0 || mass == 0)
                {
                    continue;
                }

                int dist = x + y;
                if (dist == 0)
                {
                    continue;
                }

                long profit = (long)mass * (1000L * valuePerTonne - dist);
                if (profit <= 0)
                {
                    continue;
                }

                Candidate candidate = new Candidate
                {
                    X = x,
                    Y = y,
                    Profit = profit
                };

                if (budget == int.MaxValue)
                {
                    allCandidates!.Add(candidate);
                }
                else
                {
                    heap!.Push(candidate);
                }
            }
        }

        List<Candidate> selected = budget == int.MaxValue ? allCandidates! : heap!.ToList();
        selected.Sort((a, b) => b.Profit.CompareTo(a.Profit));

        using var output = new StreamWriter(Console.OpenStandardOutput())
        {
            NewLine = "\n"
        };

        output.WriteLine(selected.Count);
        foreach (Candidate candidate in selected)
        {
            WriteOperation(output, candidate, zCount - 1);
        }
        output.Flush();
    }

    private static void WriteOperation(StreamWriter output, Candidate candidate, int digZ)
    {
        int x = candidate.X;
        int y = candidate.Y;
        int steps = x + y;
        if (steps == 0)
        {
            return;
        }

        // The format uses N waypoint pairs followed by the final destination.
        // We emit the first steps of the Manhattan route and leave the last move
        // implicit in the destination pair.
        output.Write(steps - 1);
        output.Write(' ');
        output.Write(x);
        output.Write(' ');
        output.Write(y);
        output.Write(' ');
        output.Write(digZ);

        int cx = x;
        int cy = y;
        for (int i = 0; i < steps - 1; i++)
        {
            if (cx > 0)
            {
                cx--;
            }
            else
            {
                cy--;
            }

            output.Write(' ');
            output.Write(cx);
            output.Write(' ');
            output.Write(cy);
        }

        output.Write(' ');
        output.Write(0);
        output.Write(' ');
        output.Write(0);
        output.WriteLine();
    }
}
