OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_thickness = OpConstant %float 0.025

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3_in = OpTypeArray %v3float %uint_4

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn_void
%label = OpLabel

%prev_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%start_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%end_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%next_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0

%prev_pos = OpLoad %v4float %prev_ptr
%start_pos = OpLoad %v4float %start_ptr
%end_pos = OpLoad %v4float %end_ptr
%next_pos = OpLoad %v4float %next_ptr

%start_color_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%end_color_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%world_pos_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%normal_ptr = OpAccessChain %ptr_in_v3 %normal %int_1
%start_color = OpLoad %v3float %start_color_ptr
%end_color = OpLoad %v3float %end_color_ptr
%world_pos = OpLoad %v3float %world_pos_ptr
%normal_val = OpLoad %v3float %normal_ptr

%prev_xy = OpVectorShuffle %v2float %prev_pos %prev_pos 0 1
%start_xy = OpVectorShuffle %v2float %start_pos %start_pos 0 1
%end_xy = OpVectorShuffle %v2float %end_pos %end_pos 0 1
%next_xy = OpVectorShuffle %v2float %next_pos %next_pos 0 1

%diff0 = OpFSub %v2float %end_xy %prev_xy
%diff1 = OpFSub %v2float %next_xy %start_xy
%t0 = OpExtInst %v2float %glsl450 Normalize %diff0
%t1 = OpExtInst %v2float %glsl450 Normalize %diff1

%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1

%neg_t0_y = OpFNegate %float %t0_y
%neg_t1_y = OpFNegate %float %t1_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

%off0 = OpVectorTimesScalar %v2float %perp0 %float_thickness
%off1 = OpVectorTimesScalar %v2float %perp1 %float_thickness

%sx = OpCompositeExtract %float %start_pos 0
%sy = OpCompositeExtract %float %start_pos 1
%sz = OpCompositeExtract %float %start_pos 2
%sw = OpCompositeExtract %float %start_pos 3
%ex = OpCompositeExtract %float %end_pos 0
%ey = OpCompositeExtract %float %end_pos 1
%ez = OpCompositeExtract %float %end_pos 2
%ew = OpCompositeExtract %float %end_pos 3

%off0x = OpCompositeExtract %float %off0 0
%off0y = OpCompositeExtract %float %off0 1
%off1x = OpCompositeExtract %float %off1 0
%off1y = OpCompositeExtract %float %off1 1

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%v0x = OpFAdd %float %sx %off0x
%v0y = OpFAdd %float %sy %off0y
%v0pos = OpCompositeConstruct %v4float %v0x %v0y %sz %sw
OpStore %out_pos_ptr %v0pos
OpStore %outColor %start_color
OpEmitVertex

%v1x = OpFSub %float %sx %off0x
%v1y = OpFSub %float %sy %off0y
%v1pos = OpCompositeConstruct %v4float %v1x %v1y %sz %sw
OpStore %out_pos_ptr %v1pos
OpStore %outColor %start_color
OpEmitVertex

%v2x = OpFAdd %float %ex %off1x
%v2y = OpFAdd %float %ey %off1y
%v2pos = OpCompositeConstruct %v4float %v2x %v2y %ez %ew
OpStore %out_pos_ptr %v2pos
OpStore %outColor %end_color
OpEmitVertex

%v3x = OpFSub %float %ex %off1x
%v3y = OpFSub %float %ey %off1y
%v3pos = OpCompositeConstruct %v4float %v3x %v3y %ez %ew
OpStore %out_pos_ptr %v3pos
OpStore %outColor %end_color
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
