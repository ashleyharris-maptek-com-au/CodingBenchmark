OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%functy = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_v3 Input
%normal = OpVariable %ptr_in_v3 Input
%color = OpVariable %ptr_in_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%main = OpFunction %void None %functy
%label = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p0w = OpCompositeExtract %float %p0 3
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1z = OpCompositeExtract %float %p1 2
%p1w = OpCompositeExtract %float %p1 3
%dx = OpFSub %float %p1x %p0x
%dy = OpFSub %float %p1y %p0y
%dir = OpCompositeConstruct %v2float %dx %dy
%ndir = OpExtInst %v2float %glsl Normalize %dir
%ndirx = OpCompositeExtract %float %ndir 0
%ndiry = OpCompositeExtract %float %ndir 1
%neg_ndiry = OpFNegate %float %ndiry
%perp = OpCompositeConstruct %v2float %neg_ndiry %ndirx
%off_start = OpVectorTimesScalar %v2float %perp %float_0_04
%off_end = OpVectorTimesScalar %v2float %perp %float_0_005
%p0xy = OpCompositeConstruct %v2float %p0x %p0y
%p1xy = OpCompositeConstruct %v2float %p1x %p1y
%splus = OpFAdd %v2float %p0xy %off_start
%sminus = OpFSub %v2float %p0xy %off_start
%eplus = OpFAdd %v2float %p1xy %off_end
%eminus = OpFSub %v2float %p1xy %off_end
%splusx = OpCompositeExtract %float %splus 0
%splusy = OpCompositeExtract %float %splus 1
%sminusx = OpCompositeExtract %float %sminus 0
%sminusy = OpCompositeExtract %float %sminus 1
%eplusx = OpCompositeExtract %float %eplus 0
%eplusy = OpCompositeExtract %float %eplus 1
%eminusx = OpCompositeExtract %float %eminus 0
%eminusy = OpCompositeExtract %float %eminus 1
%pos0a = OpCompositeConstruct %v4float %splusx %splusy %p0z %p0w
%pos0b = OpCompositeConstruct %v4float %sminusx %sminusy %p0z %p0w
%pos1a = OpCompositeConstruct %v4float %eplusx %eplusy %p1z %p1w
%pos1b = OpCompositeConstruct %v4float %eminusx %eminusy %p1z %p1w
%gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos %pos0a
OpStore %outColor %c0
OpEmitVertex
OpStore %gl_out_pos %pos0b
OpStore %outColor %c0
OpEmitVertex
OpStore %gl_out_pos %pos1a
OpStore %outColor %c1
OpEmitVertex
OpStore %gl_out_pos %pos1b
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
