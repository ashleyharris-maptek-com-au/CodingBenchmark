use std::collections::VecDeque;
use std::io::{self, Read};
use std::thread;

fn main() {
    thread::Builder::new()
        .stack_size(64 * 1024 * 1024)
        .spawn(run)
        .unwrap()
        .join()
        .unwrap();
}

fn run() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = it.next().unwrap().parse::<u32>().unwrap() != 0;

    let n_total = w * h;
    let mut blocked = vec![false; n_total];
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < w && y < h {
            blocked[y * w + x] = true;
        }
    }

    let start = 0usize;
    if blocked[start] {
        println!("0");
        return;
    }

    let active: Vec<bool> = blocked.iter().map(|&b| !b).collect();
    let free_count = active.iter().filter(|&&b| b).count();

    if free_count == 0 {
        println!("0");
        return;
    }
    if free_count == 1 {
        if require_cycle {
            println!("0");
        } else {
            println!("1");
            println!("0 0");
        }
        return;
    }

    let mut color_cnt = [0usize; 2];
    for y in 0..h {
        for x in 0..w {
            let idx = y * w + x;
            if active[idx] {
                color_cnt[(x + y) & 1] += 1;
            }
        }
    }
    let start_color = 0usize;
    if require_cycle {
        if free_count % 2 == 1 || color_cnt[0] != color_cnt[1] {
            println!("0");
            return;
        }
    } else {
        let need_start = (free_count + 1) / 2;
        let need_other = free_count / 2;
        if color_cnt[start_color] != need_start || color_cnt[1 - start_color] != need_other {
            println!("0");
            return;
        }
    }

    let mut neighbors: Vec<[usize; 4]> = vec![[usize::MAX; 4]; n_total];
    let mut degree = vec![0u8; n_total];
    for y in 0..h {
        for x in 0..w {
            let idx = y * w + x;
            if !active[idx] {
                continue;
            }
            let mut k = 0usize;
            if x > 0 {
                let j = idx - 1;
                if active[j] {
                    neighbors[idx][k] = j;
                    k += 1;
                }
            }
            if x + 1 < w {
                let j = idx + 1;
                if active[j] {
                    neighbors[idx][k] = j;
                    k += 1;
                }
            }
            if y > 0 {
                let j = idx - w;
                if active[j] {
                    neighbors[idx][k] = j;
                    k += 1;
                }
            }
            if y + 1 < h {
                let j = idx + w;
                if active[j] {
                    neighbors[idx][k] = j;
                    k += 1;
                }
            }
            degree[idx] = k as u8;
        }
    }

    let mut deg0 = 0usize;
    let mut deg1 = 0usize;
    for i in 0..n_total {
        if active[i] {
            match degree[i] {
                0 => deg0 += 1,
                1 => deg1 += 1,
                _ => {}
            }
        }
    }

    let mut solver = Solver {
        start,
        cycle: require_cycle,
        active,
        visited: vec![false; n_total],
        neighbors,
        degree,
        remaining: free_count,
        deg0,
        deg1,
        path: Vec::with_capacity(free_count),
        seen: vec![0u32; n_total],
        stamp: 1,
    };

    solver.visit_start();

    let mut out = String::new();

    if !require_cycle && obstacle_count == 0 {
        let mut path = Vec::with_capacity(free_count);
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    path.push((x, y));
                }
            } else {
                for x in (0..w).rev() {
                    path.push((x, y));
                }
            }
        }
        out.push_str(&format!("{}\n", path.len()));
        for (x, y) in path {
            out.push_str(&format!("{} {}\n", x, y));
        }
        print!("{}", out);
        return;
    }

    if solver.dfs(start) {
        out.push_str(&format!("{}\n", solver.path.len()));
        for idx in solver.path {
            let x = idx % w;
            let y = idx / w;
            out.push_str(&format!("{} {}\n", x, y));
        }
        print!("{}", out);
    } else {
        println!("0");
    }
}

struct Solver {
    start: usize,
    cycle: bool,
    active: Vec<bool>,
    visited: Vec<bool>,
    neighbors: Vec<[usize; 4]>,
    degree: Vec<u8>,
    remaining: usize,
    deg0: usize,
    deg1: usize,
    path: Vec<usize>,
    seen: Vec<u32>,
    stamp: u32,
}

impl Solver {
    fn visit_start(&mut self) {
        let s = self.start;
        self.visited[s] = true;
        self.path.push(s);
        self.remaining -= 1;
        self.remove_vertex_contribution(s);
        self.decrement_neighbors(s);
    }

    fn remove_vertex_contribution(&mut self, v: usize) {
        if !self.active[v] {
            return;
        }
        match self.degree[v] {
            0 => self.deg0 -= 1,
            1 => self.deg1 -= 1,
            _ => {}
        }
    }

    fn restore_vertex_contribution(&mut self, v: usize) {
        if !self.active[v] {
            return;
        }
        match self.degree[v] {
            0 => self.deg0 += 1,
            1 => self.deg1 += 1,
            _ => {}
        }
    }

    fn decrement_neighbors(&mut self, v: usize) {
        for &u in self.neighbors[v].iter() {
            if u == usize::MAX || !self.active[u] || self.visited[u] {
                continue;
            }
            match self.degree[u] {
                1 => {
                    self.deg1 -= 1;
                    self.deg0 += 1;
                }
                2 => {
                    self.deg1 += 1;
                }
                _ => {}
            }
            self.degree[u] -= 1;
        }
    }

    fn increment_neighbors(&mut self, v: usize) {
        for &u in self.neighbors[v].iter() {
            if u == usize::MAX || !self.active[u] || self.visited[u] {
                continue;
            }
            match self.degree[u] {
                0 => {
                    self.deg0 -= 1;
                    self.deg1 += 1;
                }
                1 => {
                    self.deg1 -= 1;
                }
                _ => {}
            }
            self.degree[u] += 1;
        }
    }

    fn apply(&mut self, v: usize) {
        self.visited[v] = true;
        self.path.push(v);
        self.remaining -= 1;
        self.remove_vertex_contribution(v);
        self.decrement_neighbors(v);
    }

    fn undo(&mut self, v: usize) {
        self.increment_neighbors(v);
        self.restore_vertex_contribution(v);
        self.remaining += 1;
        self.path.pop();
        self.visited[v] = false;
    }

    fn connected(&mut self) -> bool {
        if self.remaining == 0 {
            return true;
        }
        self.stamp = self.stamp.wrapping_add(1);
        if self.stamp == 0 {
            self.stamp = 1;
            for s in self.seen.iter_mut() {
                *s = 0;
            }
        }
        let stamp = self.stamp;
        let mut start = None;
        for i in 0..self.active.len() {
            if self.active[i] && !self.visited[i] {
                start = Some(i);
                break;
            }
        }
        let s = match start {
            Some(v) => v,
            None => return true,
        };
        let mut q = VecDeque::new();
        q.push_back(s);
        self.seen[s] = stamp;
        let mut seen_count = 0usize;
        while let Some(v) = q.pop_front() {
            seen_count += 1;
            for &u in self.neighbors[v].iter() {
                if u == usize::MAX || !self.active[u] || self.visited[u] || self.seen[u] == stamp {
                    continue;
                }
                self.seen[u] = stamp;
                q.push_back(u);
            }
        }
        seen_count == self.remaining
    }

    fn is_neighbor(&self, a: usize, b: usize) -> bool {
        for &u in self.neighbors[a].iter() {
            if u == b {
                return true;
            }
        }
        false
    }

    fn collect_degree1_neighbors(&self, curr: usize) -> Vec<usize> {
        let mut out = Vec::new();
        for i in 0..self.active.len() {
            if self.active[i] && !self.visited[i] && self.degree[i] == 1 && self.is_neighbor(curr, i) {
                out.push(i);
            }
        }
        out
    }

    fn score(&self, cand: usize) -> i32 {
        let mut predicted_deg1 = self.deg1 as i32;
        let mut sum = 0i32;
        if self.degree[cand] == 1 {
            predicted_deg1 -= 1;
        }
        for &u in self.neighbors[cand].iter() {
            if u == usize::MAX || !self.active[u] || self.visited[u] {
                continue;
            }
            sum += self.degree[u] as i32;
            match self.degree[u] {
                1 => predicted_deg1 -= 1,
                2 => predicted_deg1 += 1,
                _ => {}
            }
        }
        predicted_deg1 * 1000 + (self.degree[cand] as i32) * 10 + sum
    }

    fn dfs(&mut self, curr: usize) -> bool {
        if self.remaining == 0 {
            return !self.cycle || self.is_neighbor(curr, self.start);
        }

        if self.remaining > 1 && self.deg0 > 0 {
            return false;
        }
        if self.deg1 > 2 {
            return false;
        }

        let mut candidates = Vec::new();
        for &u in self.neighbors[curr].iter() {
            if u != usize::MAX && self.active[u] && !self.visited[u] {
                candidates.push(u);
            }
        }
        if candidates.is_empty() {
            return false;
        }

        if self.deg1 == 2 {
            let forced = self.collect_degree1_neighbors(curr);
            if forced.is_empty() {
                return false;
            }
            candidates = forced;
        }

        candidates.sort_by_key(|&u| (self.score(u), u));

        for cand in candidates {
            self.apply(cand);

            let ok = if self.remaining > 1 {
                self.deg0 == 0 && self.connected()
            } else {
                self.connected()
            };

            if ok && self.dfs(cand) {
                return true;
            }

            self.undo(cand);
        }

        false
    }
}
