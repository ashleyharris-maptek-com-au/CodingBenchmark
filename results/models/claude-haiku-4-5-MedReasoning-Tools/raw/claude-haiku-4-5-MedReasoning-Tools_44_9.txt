#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // Create 8x8 checkerboard pattern
    float checkX = floor(uv.x * 8.0);
    float checkY = floor(uv.y * 8.0);
    float parity = mod(checkX + checkY, 2.0);

    // Alternate between 0.9 and 0.2 based on parity
    float checkColor = mix(0.9, 0.2, parity);

    // Normalize normal (may be unnormalized from interpolation)
    vec3 norm = normalize(normal);

    // Calculate light direction
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);

    // Lambertian diffuse
    float diffuse = max(0.0, dot(norm, lightDir));

    // Shading factor: 0.5 + 0.5 * diffuse
    float shadingFactor = 0.5 + 0.5 * diffuse;

    // Apply shading to checkerboard color
    float finalColor = checkColor * shadingFactor;

    // Output with alpha 1.0
    outColor = vec4(vec3(finalColor), 1.0);
}
