declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Pkg {
  id: string;
  x: number;
  y: number;
}

const { depot, truckCount, packages } = data as {
  depot: [number, number];
  truckCount: number;
  packages: Pkg[];
};

function dist(a: [number, number], b: [number, number]): number {
  const dx = a[0] - b[0];
  const dy = a[1] - b[1];
  return Math.sqrt(dx * dx + dy * dy);
}

function kmeans(pkgs: Pkg[], k: number): Pkg[][] {
  const clusters: Pkg[][] = Array(k).fill(null).map(() => []);
  const centers: [number, number][] = [];

  const sorted = [...pkgs].sort((a, b) => a.x + a.y - b.x - b.y);
  for (let i = 0; i < k; i++) {
    const p = sorted[Math.floor((i * pkgs.length) / k)];
    centers.push([p.x, p.y]);
  }

  for (let iter = 0; iter < 3; iter++) {
    for (let i = 0; i < k; i++) clusters[i] = [];

    for (const pkg of pkgs) {
      let best = 0;
      let minDist = Infinity;
      for (let i = 0; i < k; i++) {
        const d = dist([pkg.x, pkg.y], centers[i]);
        if (d < minDist) {
          minDist = d;
          best = i;
        }
      }
      clusters[best].push(pkg);
    }

    for (let i = 0; i < k; i++) {
      if (clusters[i].length > 0) {
        const sx = clusters[i].reduce((s, p) => s + p.x, 0);
        const sy = clusters[i].reduce((s, p) => s + p.y, 0);
        centers[i] = [sx / clusters[i].length, sy / clusters[i].length];
      }
    }
  }

  return clusters;
}

function routeCost(route: string[]): number {
  let cost = 0;
  let pos = depot;
  for (const id of route) {
    const p = packages.find(x => x.id === id)!;
    cost += dist(pos, [p.x, p.y]);
    pos = [p.x, p.y];
  }
  cost += dist(pos, depot);
  return cost;
}

function twoOpt(route: string[]): string[] {
  if (route.length <= 2) return route;

  let best = [...route];
  let improved = true;
  let iterations = 0;

  while (improved && iterations < 50) {
    improved = false;
    for (let i = 0; i < best.length - 2; i++) {
      for (let j = i + 2; j < best.length; j++) {
        const cand = [
          ...best.slice(0, i + 1),
          ...best.slice(i + 1, j + 1).reverse(),
          ...best.slice(j + 1)
        ];
        if (routeCost(cand) < routeCost(best)) {
          best = cand;
          improved = true;
          break;
        }
      }
      if (improved) break;
    }
    iterations++;
  }

  return best;
}

function buildRoute(pkgs: Pkg[]): string[] {
  if (pkgs.length === 0) return [];

  const route: string[] = [];
  const used = new Set<string>();
  let pos = depot;

  while (used.size < pkgs.length) {
    let best: Pkg | null = null;
    let minDist = Infinity;

    for (const p of pkgs) {
      if (!used.has(p.id)) {
        const d = dist(pos, [p.x, p.y]);
        if (d < minDist) {
          minDist = d;
          best = p;
        }
      }
    }

    if (best) {
      route.push(best.id);
      used.add(best.id);
      pos = [best.x, best.y];
    }
  }

  return twoOpt(route);
}

const clusters = kmeans(packages, Math.min(truckCount, packages.length));
const routes = clusters.filter(c => c.length > 0).map(c => buildRoute(c));
