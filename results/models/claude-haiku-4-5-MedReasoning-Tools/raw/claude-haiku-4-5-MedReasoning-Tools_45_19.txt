#!/usr/bin/env python3
"""Generate SPIR-V binary for Phong lighting fragment shader"""
import struct
import io

class SPIRVBuilder:
    def __init__(self):
        self.words = []
        self.next_id = 1
        self.header_bound_idx = None

    def emit_word(self, w):
        self.words.append(w & 0xFFFFFFFF)

    def emit_instruction(self, opcode, *operands):
        wc = 1 + len(operands)
        self.emit_word((wc << 16) | opcode)
        for op in operands:
            self.emit_word(op)

    def emit_string(self, s):
        """Emit SPIR-V string (null-terminated, padded to 4-byte boundary)"""
        b = (s + '\0').encode('utf-8')
        while len(b) % 4 != 0:
            b += b'\x00'
        for i in range(0, len(b), 4):
            w = struct.unpack('<I', b[i:i+4])[0]
            self.emit_word(w)

    def alloc_id(self):
        self.next_id += 1
        return self.next_id - 1

    def float_to_bits(self, f):
        return struct.unpack('I', struct.pack('<f', f))[0]

    def build(self):
        # Header
        self.emit_word(0x07230203)  # Magic number
        self.emit_word(0x00010000)  # Version 1.0
        self.emit_word(0x00000000)  # Generator
        self.header_bound_idx = len(self.words)
        self.emit_word(0)  # Bound (updated at end)
        self.emit_word(0)  # Schema

        # Capability
        self.emit_instruction(17, 1)  # OpCapability Shader

        # Extension import
        ext_inst_id = self.alloc_id()
        self.emit_instruction(11, ext_inst_id)
        self.emit_string("GLSL.std.450")

        # Memory model
        self.emit_instruction(14, 0, 1)  # OpMemoryModel Logical GLSL450

        # Type IDs
        tid_void = self.alloc_id()
        tid_float = self.alloc_id()
        tid_vec2 = self.alloc_id()
        tid_vec3 = self.alloc_id()
        tid_vec4 = self.alloc_id()
        tid_mat4 = self.alloc_id()
        tid_ubo_struct = self.alloc_id()

        # OpTypeVoid
        self.emit_instruction(19, tid_void)

        # OpTypeFloat 32-bit
        self.emit_instruction(22, tid_float, 32)

        # OpTypeVector vec2
        self.emit_instruction(23, tid_vec2, tid_float, 2)

        # OpTypeVector vec3
        self.emit_instruction(23, tid_vec3, tid_float, 3)

        # OpTypeVector vec4
        self.emit_instruction(23, tid_vec4, tid_float, 4)

        # OpTypeMatrix mat4x4
        self.emit_instruction(24, tid_mat4, tid_vec4, 4)

        # OpTypeStruct for UBO (mat4 model, mat4 view, mat4 projection, vec4 lightPos, vec4 cameraPos, vec4 params)
        self.emit_instruction(30, tid_ubo_struct, tid_mat4, tid_mat4, tid_mat4, tid_vec4, tid_vec4, tid_vec4)

        # Pointer types
        tid_ptr_uniform_struct = self.alloc_id()
        tid_ptr_input_vec3 = self.alloc_id()
        tid_ptr_input_vec2 = self.alloc_id()
        tid_ptr_output_vec4 = self.alloc_id()
        tid_ptr_function_vec3 = self.alloc_id()
        tid_ptr_function_vec4 = self.alloc_id()
        tid_ptr_function_float = self.alloc_id()
        tid_ptr_uniform_vec4 = self.alloc_id()

        # OpTypePointer for Uniform storage class
        self.emit_instruction(32, tid_ptr_uniform_struct, 2, tid_ubo_struct)

        # OpTypePointer for Input storage class
        self.emit_instruction(32, tid_ptr_input_vec3, 1, tid_vec3)
        self.emit_instruction(32, tid_ptr_input_vec2, 1, tid_vec2)

        # OpTypePointer for Output storage class
        self.emit_instruction(32, tid_ptr_output_vec4, 3, tid_vec4)

        # OpTypePointer for Function storage class
        self.emit_instruction(32, tid_ptr_function_vec3, 7, tid_vec3)
        self.emit_instruction(32, tid_ptr_function_vec4, 7, tid_vec4)
        self.emit_instruction(32, tid_ptr_function_float, 7, tid_float)
        self.emit_instruction(32, tid_ptr_uniform_vec4, 2, tid_vec4)

        # OpTypeFunction void -> void
        tid_fn_void_void = self.alloc_id()
        self.emit_instruction(33, tid_fn_void_void, tid_void, tid_void)

        # Constants
        const_0p0 = self.alloc_id()
        const_0p05 = self.alloc_id()
        const_0p1 = self.alloc_id()
        const_1p0 = self.alloc_id()
        const_32p0 = self.alloc_id()

        self.emit_instruction(43, tid_float, const_0p0, self.float_to_bits(0.0))
        self.emit_instruction(43, tid_float, const_0p05, self.float_to_bits(0.05))
        self.emit_instruction(43, tid_float, const_0p1, self.float_to_bits(0.1))
        self.emit_instruction(43, tid_float, const_1p0, self.float_to_bits(1.0))
        self.emit_instruction(43, tid_float, const_32p0, self.float_to_bits(32.0))

        # Composite constants
        const_ambient = self.alloc_id()
        const_white = self.alloc_id()

        self.emit_instruction(80, tid_vec3, const_ambient, const_0p1, const_0p05, const_0p05)
        self.emit_instruction(80, tid_vec4, const_white, const_1p0, const_1p0, const_1p0, const_1p0)

        # Decorations for UBO struct
        self.emit_instruction(71, tid_ubo_struct, 2)  # Block decoration
        # Member offsets: 0, 64, 128, 192, 208, 224
        for i, offset in enumerate([0, 64, 128, 192, 208, 224]):
            self.emit_instruction(71, tid_ubo_struct, 35, i, offset)

        # Variables
        var_worldpos = self.alloc_id()
        var_normal = self.alloc_id()
        var_uv = self.alloc_id()
        var_tangent = self.alloc_id()
        var_color = self.alloc_id()
        var_outcolor = self.alloc_id()
        var_ubo = self.alloc_id()

        # OpVariable for inputs
        self.emit_instruction(59, tid_ptr_input_vec3, var_worldpos, 1)  # Input
        self.emit_instruction(59, tid_ptr_input_vec3, var_normal, 1)
        self.emit_instruction(59, tid_ptr_input_vec2, var_uv, 1)
        self.emit_instruction(59, tid_ptr_input_vec3, var_tangent, 1)
        self.emit_instruction(59, tid_ptr_input_vec3, var_color, 1)

        # OpVariable for output
        self.emit_instruction(59, tid_ptr_output_vec4, var_outcolor, 3)  # Output

        # OpVariable for UBO
        self.emit_instruction(59, tid_ptr_uniform_struct, var_ubo, 2)  # Uniform

        # Decorations for variables
        self.emit_instruction(71, var_worldpos, 30, 0)  # Location 0
        self.emit_instruction(71, var_normal, 30, 1)
        self.emit_instruction(71, var_uv, 30, 2)
        self.emit_instruction(71, var_tangent, 30, 3)
        self.emit_instruction(71, var_color, 30, 4)
        self.emit_instruction(71, var_outcolor, 30, 0)

        # Decorations for UBO
        self.emit_instruction(71, var_ubo, 34, 0)  # DescriptorSet 0
        self.emit_instruction(71, var_ubo, 33, 0)  # Binding 0

        # Entry point
        main_fn = self.alloc_id()
        self.emit_instruction(15, 4, main_fn)  # OpEntryPoint Fragment
        self.emit_string("main")

        # Execution mode
        self.emit_instruction(16, main_fn, 7)  # OriginUpperLeft

        # Function definition
        self.emit_instruction(54, tid_void, main_fn, 2, tid_fn_void_void)

        # Entry label
        label_entry = self.alloc_id()
        self.emit_instruction(248, label_entry)

        # Load inputs
        val_worldpos = self.alloc_id()
        val_normal = self.alloc_id()
        val_color = self.alloc_id()

        self.emit_instruction(61, tid_vec3, val_worldpos, var_worldpos)
        self.emit_instruction(61, tid_vec3, val_normal, var_normal)
        self.emit_instruction(61, tid_vec3, val_color, var_color)

        # Access UBO members
        ptr_lightpos = self.alloc_id()
        ptr_camerapos = self.alloc_id()

        # OpAccessChain for lightPos (member 3)
        self.emit_instruction(65, tid_ptr_uniform_vec4, ptr_lightpos, var_ubo, 3)
        self.emit_instruction(65, tid_ptr_uniform_vec4, ptr_camerapos, var_ubo, 4)

        val_lightpos = self.alloc_id()
        val_camerapos = self.alloc_id()

        self.emit_instruction(61, tid_vec4, val_lightpos, ptr_lightpos)
        self.emit_instruction(61, tid_vec4, val_camerapos, ptr_camerapos)

        # Shuffle to vec3 (xyz)
        light_xyz = self.alloc_id()
        camera_xyz = self.alloc_id()

        # Use composite extract for each component and reconstruct
        light_x = self.alloc_id()
        light_y = self.alloc_id()
        light_z = self.alloc_id()
        self.emit_instruction(81, tid_float, light_x, val_lightpos, 0)
        self.emit_instruction(81, tid_float, light_y, val_lightpos, 1)
        self.emit_instruction(81, tid_float, light_z, val_lightpos, 2)
        self.emit_instruction(80, tid_vec3, light_xyz, light_x, light_y, light_z)

        cam_x = self.alloc_id()
        cam_y = self.alloc_id()
        cam_z = self.alloc_id()
        self.emit_instruction(81, tid_float, cam_x, val_camerapos, 0)
        self.emit_instruction(81, tid_float, cam_y, val_camerapos, 1)
        self.emit_instruction(81, tid_float, cam_z, val_camerapos, 2)
        self.emit_instruction(80, tid_vec3, camera_xyz, cam_x, cam_y, cam_z)

        # Normalize normal
        n = self.alloc_id()
        self.emit_instruction(12, tid_vec3, n, ext_inst_id, 69, val_normal)  # normalize

        # L = normalize(lightPos - worldPos)
        light_diff = self.alloc_id()
        l = self.alloc_id()
        self.emit_instruction(131, tid_vec3, light_diff, light_xyz, val_worldpos)  # OpFSub
        self.emit_instruction(12, tid_vec3, l, ext_inst_id, 69, light_diff)  # normalize

        # V = normalize(cameraPos - worldPos)
        view_diff = self.alloc_id()
        v = self.alloc_id()
        self.emit_instruction(131, tid_vec3, view_diff, camera_xyz, val_worldpos)  # OpFSub
        self.emit_instruction(12, tid_vec3, v, ext_inst_id, 69, view_diff)  # normalize

        # R = reflect(-L, N)
        neg_l = self.alloc_id()
        r = self.alloc_id()
        self.emit_instruction(127, tid_vec3, neg_l, l)  # OpFNegate
        self.emit_instruction(12, tid_vec3, r, ext_inst_id, 67, neg_l, n)  # reflect

        # Diffuse: color * max(dot(N,L), 0)
        n_dot_l = self.alloc_id()
        diff_factor = self.alloc_id()
        diffuse = self.alloc_id()

        self.emit_instruction(148, tid_float, n_dot_l, n, l)  # OpDot
        self.emit_instruction(12, tid_float, diff_factor, ext_inst_id, 40, n_dot_l, const_0p0)  # max
        self.emit_instruction(133, tid_vec3, diffuse, val_color, diff_factor)  # OpFMul

        # Specular: (1,1,1) * pow(max(dot(R,V), 0), 32)
        r_dot_v = self.alloc_id()
        spec_base = self.alloc_id()
        spec_factor = self.alloc_id()
        spec = self.alloc_id()
        white_vec3 = self.alloc_id()

        self.emit_instruction(148, tid_float, r_dot_v, r, v)  # OpDot
        self.emit_instruction(12, tid_float, spec_base, ext_inst_id, 40, r_dot_v, const_0p0)  # max
        self.emit_instruction(12, tid_float, spec_factor, ext_inst_id, 26, spec_base, const_32p0)  # pow
        self.emit_instruction(80, tid_vec3, white_vec3, const_1p0, const_1p0, const_1p0)
        self.emit_instruction(133, tid_vec3, spec, white_vec3, spec_factor)  # OpFMul

        # Sum: ambient + diffuse + specular
        amb_diff = self.alloc_id()
        result = self.alloc_id()

        self.emit_instruction(129, tid_vec3, amb_diff, const_ambient, diffuse)  # OpFAdd
        self.emit_instruction(129, tid_vec3, result, amb_diff, spec)  # OpFAdd

        # Clamp to [0, 1]
        zero_vec3 = self.alloc_id()
        one_vec3 = self.alloc_id()
        clamped = self.alloc_id()

        self.emit_instruction(80, tid_vec3, zero_vec3, const_0p0, const_0p0, const_0p0)
        self.emit_instruction(80, tid_vec3, one_vec3, const_1p0, const_1p0, const_1p0)
        self.emit_instruction(12, tid_vec3, clamped, ext_inst_id, 43, result, zero_vec3, one_vec3)  # clamp

        # Build output vec4 with alpha 1.0
        output = self.alloc_id()
        self.emit_instruction(80, tid_vec4, output, clamped, const_1p0)  # OpCompositeConstruct

        # Store to output variable
        self.emit_instruction(62, var_outcolor, output)  # OpStore

        # Return
        self.emit_instruction(253)  # OpReturn

        # End function
        self.emit_instruction(56)  # OpFunctionEnd

        # Update bound
        self.words[self.header_bound_idx] = self.next_id

        # Convert to hex string
        return "".join(f"{w:08x}" for w in self.words)

if __name__ == "__main__":
    builder = SPIRVBuilder()
    hex_output = builder.build()
    print(hex_output)
