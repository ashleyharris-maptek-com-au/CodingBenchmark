use std::io::{self, Read};
use std::f64::consts::PI;

#[derive(Debug, Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn translate(&self, dx: f64, dy: f64) -> Self {
        Polygon {
            vertices: self
                .vertices
                .iter()
                .map(|p| Point {
                    x: p.x + dx,
                    y: p.y + dy,
                })
                .collect(),
        }
    }

    fn rotate(&self, angle: f64) -> Self {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Polygon {
            vertices: self
                .vertices
                .iter()
                .map(|p| Point {
                    x: p.x * cos_a - p.y * sin_a,
                    y: p.x * sin_a + p.y * cos_a,
                })
                .collect(),
        }
    }

    fn bounds(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for p in &self.vertices {
            min_x = min_x.min(p.x);
            min_y = min_y.min(p.y);
            max_x = max_x.max(p.x);
            max_y = max_y.max(p.y);
        }
        (min_x, min_y, max_x, max_y)
    }

    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }
}

fn point_in_polygon(p: Point, polygon: &Polygon) -> bool {
    let mut inside = false;
    let n = polygon.vertices.len();
    let mut p1 = polygon.vertices[0];

    for i in 1..=n {
        let p2 = polygon.vertices[i % n];
        if p.y > p1.y.min(p2.y) {
            if p.y <= p1.y.max(p2.y) {
                if p.x <= p1.x.max(p2.x) {
                    if (p1.y - p2.y).abs() > 1e-10 {
                        let xinters =
                            (p.y - p1.y) * (p2.x - p1.x) / (p2.y - p1.y) + p1.x;
                        if (p1.x - p2.x).abs() < 1e-10 || p.x <= xinters {
                            inside = !inside;
                        }
                    }
                }
            }
        }
        p1 = p2;
    }
    inside
}

fn polygon_fits_in_stock(piece: &Polygon, stock: &Polygon) -> bool {
    piece.vertices.iter().all(|v| point_in_polygon(*v, stock))
}

fn cross_product(o: Point, a: Point, b: Point) -> f64 {
    (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x)
}

fn segments_intersect(p1: Point, p2: Point, p3: Point, p4: Point) -> bool {
    let d1 = cross_product(p3, p4, p1);
    let d2 = cross_product(p3, p4, p2);
    let d3 = cross_product(p1, p2, p3);
    let d4 = cross_product(p1, p2, p4);

    if ((d1 > 1e-10 && d2 < -1e-10) || (d1 < -1e-10 && d2 > 1e-10))
        && ((d3 > 1e-10 && d4 < -1e-10) || (d3 < -1e-10 && d4 > 1e-10))
    {
        return true;
    }
    false
}

fn polygons_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    for v in &poly1.vertices {
        if point_in_polygon(*v, poly2) {
            return true;
        }
    }
    for v in &poly2.vertices {
        if point_in_polygon(*v, poly1) {
            return true;
        }
    }
    for i in 0..poly1.vertices.len() {
        let j = (i + 1) % poly1.vertices.len();
        for k in 0..poly2.vertices.len() {
            let l = (k + 1) % poly2.vertices.len();
            if segments_intersect(
                poly1.vertices[i],
                poly1.vertices[j],
                poly2.vertices[k],
                poly2.vertices[l],
            ) {
                return true;
            }
        }
    }
    false
}

fn try_place_piece(
    piece: &Polygon,
    stock: &Polygon,
    stock_idx: usize,
    placed_pieces: &[(Polygon, usize)],
) -> Option<(f64, f64, f64)> {
    let num_rotations = if placed_pieces.len() < 500 { 16 } else { 8 };

    for rot_idx in 0..num_rotations {
        let angle = (rot_idx as f64) * PI * 2.0 / (num_rotations as f64);
        let rotated = piece.rotate(angle);

        let (px1, py1, px2, py2) = rotated.bounds();
        let piece_w = px2 - px1;
        let piece_h = py2 - py1;

        let (sx1, sy1, sx2, sy2) = stock.bounds();
        let stock_w = sx2 - sx1;
        let stock_h = sy2 - sy1;

        let dx = sx1 - px1 + (stock_w - piece_w) / 2.0;
        let dy = sy1 - py1 + (stock_h - piece_h) / 2.0;

        let placed_piece = rotated.translate(dx, dy);

        if !polygon_fits_in_stock(&placed_piece, stock) {
            continue;
        }

        let mut valid = true;
        for (other, other_stock) in placed_pieces {
            if stock_idx == *other_stock && polygons_overlap(&placed_piece, other) {
                valid = false;
                break;
            }
        }

        if valid {
            return Some((dx, dy, angle));
        }
    }

    None
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();

    let lines: Vec<&str> = input.trim().lines().collect();
    let mut idx = 0;

    let header: Vec<&str> = lines[idx].split_whitespace().collect();
    let sv = header[0].parse::<usize>().unwrap();
    let np = header[1].parse::<usize>().unwrap();
    idx += 1;

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let coords: Vec<&str> = lines[idx].split_whitespace().collect();
        stock_vertices.push(Point {
            x: coords[0].parse().unwrap(),
            y: coords[1].parse().unwrap(),
        });
        idx += 1;
    }
    let stock = Polygon { vertices: stock_vertices };

    let mut pieces = Vec::new();
    for _ in 0..np {
        let pv = lines[idx].parse::<usize>().unwrap();
        idx += 1;
        let mut vertices = Vec::new();
        for _ in 0..pv {
            let coords: Vec<&str> = lines[idx].split_whitespace().collect();
            vertices.push(Point {
                x: coords[0].parse().unwrap(),
                y: coords[1].parse().unwrap(),
            });
            idx += 1;
        }
        pieces.push(Polygon { vertices });
    }

    let mut placed_pieces: Vec<(Polygon, usize)> = Vec::new();
    let mut placements: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); np];
    let mut num_stocks = 0;

    for (piece_idx, piece) in pieces.iter().enumerate() {
        let mut placed = false;

        for stock_idx in 0..num_stocks {
            if let Some((dx, dy, angle)) = try_place_piece(piece, &stock, stock_idx, &placed_pieces)
            {
                let placed_piece = piece.rotate(angle).translate(dx, dy);
                placed_pieces.push((placed_piece, stock_idx));
                placements[piece_idx] = (stock_idx, dx, dy, angle);
                placed = true;
                break;
            }
        }

        if !placed {
            if let Some((dx, dy, angle)) = try_place_piece(piece, &stock, num_stocks, &placed_pieces)
            {
                let placed_piece = piece.rotate(angle).translate(dx, dy);
                placed_pieces.push((placed_piece, num_stocks));
                placements[piece_idx] = (num_stocks, dx, dy, angle);
                num_stocks += 1;
            }
        }
    }

    println!("{}", num_stocks);
    for (stock_idx, dx, dy, angle) in placements {
        println!("{} {} {} {}", stock_idx, dx, dy, angle);
    }
}
