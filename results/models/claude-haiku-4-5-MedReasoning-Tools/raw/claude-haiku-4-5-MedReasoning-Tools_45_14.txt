#!/usr/bin/env python3
import struct

class SPIRVModule:
    def __init__(self):
        self.instructions = []
        self.id_counter = 0
        self.strings = {}

    def new_id(self):
        self.id_counter += 1
        return self.id_counter

    def add_instruction(self, opcode, *operands):
        """Add an instruction with word count header"""
        word_count = 1 + len(operands)
        header = (word_count << 16) | opcode
        self.instructions.append((header, operands))

    def add_string_instruction(self, opcode, result_id, string):
        """Add instruction with embedded string"""
        word_count = 2  # opcode + result_id
        s_bytes = (string + '\0').encode('utf-8')
        # Pad to 4-byte boundary
        padding = (4 - len(s_bytes) % 4) % 4
        s_bytes += b'\0' * padding
        word_count += len(s_bytes) // 4

        header = (word_count << 16) | opcode
        self.instructions.append((header, (result_id,), s_bytes))

    def to_binary(self):
        """Convert to binary with header"""
        binary = bytearray()

        # Header
        binary.extend(struct.pack('<I', 0x07230203))  # Magic
        binary.extend(struct.pack('<I', 0x00010000))  # Version 1.0
        binary.extend(struct.pack('<I', 0))           # Generator
        binary.extend(struct.pack('<I', self.id_counter + 10))  # Bound
        binary.extend(struct.pack('<I', 0))           # Reserved

        # Instructions
        for item in self.instructions:
            if len(item) == 2:
                header, operands = item
                binary.extend(struct.pack('<I', header))
                for op in operands:
                    if isinstance(op, float):
                        binary.extend(struct.pack('<f', op))
                    else:
                        binary.extend(struct.pack('<I', op))
            else:
                header, operands, s_bytes = item
                binary.extend(struct.pack('<I', header))
                for op in operands:
                    binary.extend(struct.pack('<I', op))
                binary.extend(s_bytes)

        return binary

def generate_shader():
    m = SPIRVModule()

    # OpCapability Shader
    m.add_instruction(17, 1)

    # OpExtInstImport
    id_glsl = m.new_id()
    m.add_string_instruction(11, id_glsl, 'GLSL.std.450')

    # OpMemoryModel Logical GLSL450
    m.add_instruction(14, 0, 1)

    # Type IDs
    id_void = m.new_id()
    id_float = m.new_id()
    id_vec3 = m.new_id()
    id_vec4 = m.new_id()
    id_mat4 = m.new_id()
    id_ptr_uniform = m.new_id()
    id_ptr_input = m.new_id()
    id_ptr_output = m.new_id()
    id_fn_type = m.new_id()

    # OpTypeVoid
    m.add_instruction(19)

    # OpTypeFloat 32
    m.add_instruction(22, 32)

    # OpTypeVector float 3
    m.add_instruction(23, id_float, 3)

    # OpTypeVector float 4
    m.add_instruction(23, id_float, 4)

    # OpTypeMatrix vec4 4
    m.add_instruction(24, id_vec4, 4)

    # OpTypePointer Uniform vec4
    m.add_instruction(32, 0, id_vec4)

    # OpTypePointer Input vec3
    m.add_instruction(32, 1, id_vec3)

    # OpTypePointer Output vec4
    m.add_instruction(32, 3, id_vec4)

    # OpTypeFunction void ()
    m.add_instruction(33, id_void)

    # Constant floats
    id_const_0p3 = m.new_id()
    id_const_0p7 = m.new_id()
    id_const_0p8 = m.new_id()
    id_const_0p2 = m.new_id()
    id_const_1p5 = m.new_id()
    id_const_3p0 = m.new_id()
    id_const_1p0 = m.new_id()
    id_const_0p0 = m.new_id()

    # OpConstant float
    m.add_instruction(43, id_float, 0.3)
    m.add_instruction(43, id_float, 0.7)
    m.add_instruction(43, id_float, 0.8)
    m.add_instruction(43, id_float, 0.2)
    m.add_instruction(43, id_float, 1.5)
    m.add_instruction(43, id_float, 3.0)
    m.add_instruction(43, id_float, 1.0)
    m.add_instruction(43, id_float, 0.0)

    # Input variables
    id_worldPos = m.new_id()
    id_normal = m.new_id()
    id_lightPos = m.new_id()
    id_cameraPos = m.new_id()

    # OpVariable Input vec3
    m.add_instruction(59, id_ptr_input, 1, id_worldPos)
    m.add_instruction(59, id_ptr_input, 1, id_normal)
    m.add_instruction(59, id_ptr_uniform, 0, id_lightPos)
    m.add_instruction(59, id_ptr_uniform, 0, id_cameraPos)

    # Output variable
    id_outColor = m.new_id()
    m.add_instruction(59, id_ptr_output, 3, id_outColor)

    # Decorations
    m.add_instruction(71, id_worldPos, 30, 0)  # Location 0
    m.add_instruction(71, id_normal, 30, 1)   # Location 1
    m.add_instruction(71, id_lightPos, 34, 0) # DescriptorSet 0
    m.add_instruction(71, id_lightPos, 33, 0) # Binding 0
    m.add_instruction(71, id_cameraPos, 34, 0)
    m.add_instruction(71, id_cameraPos, 33, 0)
    m.add_instruction(71, id_outColor, 30, 0) # Location 0

    # OpEntryPoint Fragment
    m.add_instruction(15, 4, m.new_id(), 0)  # Fragment, main, "main"

    # OpExecutionMode OriginUpperLeft
    entry_id = m.id_counter
    m.add_instruction(16, entry_id, 7)

    # OpFunction void
    id_main_fn = m.new_id()
    m.add_instruction(54, id_void, id_main_fn, 0)

    # OpLabel
    id_label = m.new_id()
    m.add_instruction(248, id_label)

    # OpLoad inputs
    id_worldPos_val = m.new_id()
    id_normal_val = m.new_id()
    id_lightPos_val = m.new_id()
    id_cameraPos_val = m.new_id()

    m.add_instruction(61, id_vec3, id_worldPos_val, id_worldPos)
    m.add_instruction(61, id_vec3, id_normal_val, id_normal)
    m.add_instruction(61, id_vec4, id_lightPos_val, id_lightPos)
    m.add_instruction(61, id_vec4, id_cameraPos_val, id_cameraPos)

    # Calculate light direction
    id_light_xyz = m.new_id()
    id_diff = m.new_id()
    m.add_instruction(81, id_vec3, id_light_xyz, id_lightPos_val, 0)
    m.add_instruction(129, id_vec3, id_diff, id_light_xyz, id_worldPos_val)

    # Normalize light direction
    id_light_dir = m.new_id()
    m.add_instruction(12, id_vec3, id_light_dir, id_glsl, 69, id_diff)  # Normalize

    # Dot product for diffuse
    id_diffuse = m.new_id()
    m.add_instruction(148, id_float, id_diffuse, id_normal_val, id_light_dir)

    # Max(0, diffuse)
    id_diffuse_clamped = m.new_id()
    m.add_instruction(12, id_float, id_diffuse_clamped, id_glsl, 40, id_diffuse, id_const_0p0)  # Max

    # Calculate Lambertian: 0.3 + 0.7 * diffuse
    id_diffuse_scaled = m.new_id()
    id_lambertian = m.new_id()
    m.add_instruction(133, id_float, id_diffuse_scaled, id_const_0p7, id_diffuse_clamped)
    m.add_instruction(129, id_float, id_lambertian, id_const_0p3, id_diffuse_scaled)

    # Base color construction
    id_base_color = m.new_id()
    m.add_instruction(80, id_vec3, id_base_color, id_const_0p8, id_const_0p2, id_const_0p2)

    # Apply Lambertian to base color
    id_color_with_light = m.new_id()
    m.add_instruction(133, id_vec3, id_color_with_light, id_base_color, id_lambertian)

    # Calculate distance from camera
    id_camera_xyz = m.new_id()
    id_camera_diff = m.new_id()
    m.add_instruction(81, id_vec3, id_camera_xyz, id_cameraPos_val, 0)
    m.add_instruction(131, id_vec3, id_camera_diff, id_camera_xyz, id_worldPos_val)

    # Length of difference
    id_dist = m.new_id()
    m.add_instruction(12, id_float, id_dist, id_glsl, 66, id_camera_diff)  # Length

    # Calculate fog factor: (dist - 1.5) / 3.0
    id_dist_adjusted = m.new_id()
    id_fog_factor = m.new_id()
    m.add_instruction(131, id_float, id_dist_adjusted, id_dist, id_const_1p5)
    m.add_instruction(136, id_float, id_fog_factor, id_dist_adjusted, id_const_3p0)

    # Clamp fog factor
    id_fog_factor_clamped = m.new_id()
    m.add_instruction(12, id_float, id_fog_factor_clamped, id_glsl, 43, id_fog_factor, id_const_0p0, id_const_1p0)  # Clamp

    # Fog color
    id_fog_color = m.new_id()
    m.add_instruction(80, id_vec3, id_fog_color, id_const_0p7, id_const_0p7, id_const_0p8)

    # Mix colors
    id_final_color = m.new_id()
    m.add_instruction(12, id_vec3, id_final_color, id_glsl, 46, id_color_with_light, id_fog_color, id_fog_factor_clamped)  # Mix

    # Construct output vec4
    id_output = m.new_id()
    m.add_instruction(80, id_vec4, id_output, id_final_color, id_const_1p0)

    # Store output
    m.add_instruction(62, id_outColor, id_output)

    # OpReturn
    m.add_instruction(253)

    # OpFunctionEnd
    m.add_instruction(56)

    return m.to_binary().hex()

result = generate_shader()
print(result)
