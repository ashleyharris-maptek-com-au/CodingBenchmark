const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { width, height, grid, ignitions, towns, cutBudget } = data;

function getFireSpread(firebreaks) {
  const burned = new Set();
  const fbSet = new Set(firebreaks.map(([x, y]) => `${x},${y}`));

  const queue = [];
  for (const [x, y] of ignitions) {
    if (grid[y][x] === '.' && !fbSet.has(`${x},${y}`)) {
      const key = `${x},${y}`;
      burned.add(key);
      queue.push([x, y]);
    }
  }

  let idx = 0;
  while (idx < queue.length) {
    const [x, y] = queue[idx++];

    for (const [dx, dy] of [[0,1], [1,0], [0,-1], [-1,0]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
        const key = `${nx},${ny}`;
        if (!burned.has(key) && !fbSet.has(key) && grid[ny][nx] === '.') {
          burned.add(key);
          queue.push([nx, ny]);
        }
      }
    }
  }

  return burned;
}

function getTownValue(firebreaks) {
  const burned = getFireSpread(firebreaks);
  let value = 0;
  for (const town of towns) {
    if (!burned.has(`${town.x},${town.y}`)) {
      value += town.value;
    }
  }
  return value;
}

const firebreaks = [];
const used = new Set();

for (let i = 0; i < cutBudget; i++) {
  const currentValue = getTownValue(firebreaks);

  const burned = getFireSpread(firebreaks);
  const candidates = new Set();

  // Add frontier cells (adjacent to fire)
  for (const cellKey of burned) {
    const [x, y] = cellKey.split(',').map(Number);
    for (const [dx, dy] of [[0,1], [1,0], [0,-1], [-1,0]]) {
      const nx = x + dx;
      const ny = y + dy;
      if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
          grid[ny][nx] === '.' && !used.has(`${nx},${ny}`)) {
        candidates.add(`${nx},${ny}`);
      }
    }
  }

  // Add cells around threatened towns
  for (const town of towns) {
    for (let dy = -5; dy <= 5; dy++) {
      for (let dx = -5; dx <= 5; dx++) {
        const nx = town.x + dx;
        const ny = town.y + dy;
        if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
            grid[ny][nx] === '.' && !used.has(`${nx},${ny}`)) {
          candidates.add(`${nx},${ny}`);
        }
      }
    }
  }

  let best = null;
  let bestValue = currentValue;

  for (const cellKey of candidates) {
    const [x, y] = cellKey.split(',').map(Number);
    const value = getTownValue([...firebreaks, [x, y]]);
    if (value > bestValue) {
      bestValue = value;
      best = [x, y];
    }
  }

  if (!best) break;

  firebreaks.push(best);
  used.add(`${best[0]},${best[1]}`);
}
