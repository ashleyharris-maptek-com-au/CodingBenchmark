use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_vertices: usize = parts[0].parse().unwrap();
    let num_edges: usize = parts[1].parse().unwrap();

    let mut edges = Vec::new();
    let mut adj = vec![Vec::new(); num_vertices];

    // Read edges
    for i in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u: usize = parts[0].parse().unwrap();
        let v: usize = parts[1].parse().unwrap();
        edges.push((u, v));
        adj[u].push(i);
        adj[v].push(i);
    }

    // Greedy vertex cover using degree heuristic
    let mut covered = vec![false; num_edges];
    let mut cover = Vec::new();
    let mut degree = vec![0usize; num_vertices];

    // Initialize degrees
    for &(u, v) in &edges {
        degree[u] += 1;
        degree[v] += 1;
    }

    loop {
        // Find first uncovered edge
        let mut uncovered_idx = None;
        for (i, &is_covered) in covered.iter().enumerate() {
            if !is_covered {
                uncovered_idx = Some(i);
                break;
            }
        }

        match uncovered_idx {
            None => break,
            Some(idx) => {
                let (u, v) = edges[idx];
                let chosen = if degree[u] >= degree[v] { u } else { v };

                cover.push(chosen);

                // Mark incident edges as covered and update degrees
                for &edge_idx in &adj[chosen] {
                    if !covered[edge_idx] {
                        let (a, b) = edges[edge_idx];
                        covered[edge_idx] = true;
                        degree[a] = degree[a].saturating_sub(1);
                        degree[b] = degree[b].saturating_sub(1);
                    }
                }
            }
        }
    }

    // Output
    println!("{}", cover.len());
    let output = cover.iter().map(|v| v.to_string()).collect::<Vec<_>>().join(" ");
    println!("{}", output);
}
