using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    struct Box
    {
        public int Index;
        public int W, H, D;
        public int Volume => W * H * D;
    }

    struct Placement
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    static void Main()
    {
        var line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int containerW = int.Parse(line[1]);
        int containerH = int.Parse(line[2]);
        int containerD = int.Parse(line[3]);

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            var dims = Console.ReadLine().Split();
            boxes[i] = new Box
            {
                Index = i,
                W = int.Parse(dims[0]),
                H = int.Parse(dims[1]),
                D = int.Parse(dims[2])
            };
        }

        var placements = PackBoxes(boxes, containerW, containerH, containerD);

        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
        }
    }

    static List<Placement> PackBoxes(Box[] boxes, int cw, int ch, int cd)
    {
        var placements = new List<Placement>();
        var placed = new HashSet<int>();

        // Sort boxes by volume descending for better packing
        var sortedIndices = Enumerable.Range(0, boxes.Length)
            .OrderByDescending(i => boxes[i].Volume)
            .ToList();

        foreach (int idx in sortedIndices)
        {
            if (placed.Contains(idx)) continue;

            // Try all 6 orientations
            var orientations = GetOrientations(boxes[idx].W, boxes[idx].H, boxes[idx].D);

            bool placed_box = false;
            foreach (var (w, h, d) in orientations)
            {
                if (TryPlaceBox(idx, w, h, d, placements, cw, ch, cd, out int x, out int y, out int z))
                {
                    placements.Add(new Placement { Index = idx, X = x, Y = y, Z = z, W = w, H = h, D = d });
                    placed.Add(idx);
                    placed_box = true;
                    break;
                }
            }
        }

        return placements;
    }

    static List<(int, int, int)> GetOrientations(int w, int h, int d)
    {
        var set = new HashSet<(int, int, int)>();
        var dims = new[] { w, h, d };

        for (int i = 0; i < 3; i++)
            for (int j = 0; j < 3; j++)
                for (int k = 0; k < 3; k++)
                    if (i != j && j != k && i != k)
                        set.Add((dims[i], dims[j], dims[k]));

        return set.ToList();
    }

    static bool TryPlaceBox(int index, int w, int h, int d, List<Placement> placements, int cw, int ch, int cd, out int ox, out int oy, out int oz)
    {
        ox = oy = oz = 0;

        if (w > cw || h > ch || d > cd) return false;

        var candidates = new List<(int, int, int)> { (0, 0, 0) };

        foreach (var p in placements)
        {
            candidates.Add((p.X + p.W, p.Y, p.Z));
            candidates.Add((p.X, p.Y + p.H, p.Z));
            candidates.Add((p.X, p.Y, p.Z + p.D));
        }

        // Try candidates
        foreach (var (cx, cy, cz) in candidates.OrderBy(c => c.Item1 + c.Item2 + c.Item3))
        {
            if (cx + w > cw || cy + h > ch || cz + d > cd) continue;

            bool valid = true;
            foreach (var p in placements)
            {
                if (BoxesOverlap(cx, cy, cz, w, h, d, p.X, p.Y, p.Z, p.W, p.H, p.D))
                {
                    valid = false;
                    break;
                }
            }

            if (valid)
            {
                ox = cx;
                oy = cy;
                oz = cz;
                return true;
            }
        }

        return false;
    }

    static bool BoxesOverlap(int x1, int y1, int z1, int w1, int h1, int d1, int x2, int y2, int z2, int w2, int h2, int d2)
    {
        return !(x1 + w1 <= x2 || x2 + w2 <= x1 ||
                 y1 + h1 <= y2 || y2 + h2 <= y1 ||
                 z1 + d1 <= z2 || z2 + d2 <= z1);
    }
}
