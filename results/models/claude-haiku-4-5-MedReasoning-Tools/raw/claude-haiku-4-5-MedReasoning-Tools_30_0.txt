import sys
import random

def solve_3sat():
    # Read input
    first_line = sys.stdin.readline().split()
    num_variables = int(first_line[0])
    num_clauses = int(first_line[1])

    clauses = []
    for _ in range(num_clauses):
        clause = list(map(int, sys.stdin.readline().split()))
        clauses.append(clause)

    def is_satisfied(assignment, clause):
        """Check if a clause is satisfied by the assignment"""
        for lit in clause:
            if lit > 0:
                if assignment[lit]:
                    return True
            else:
                if not assignment[-lit]:
                    return True
        return False

    def count_unsatisfied(assignment):
        """Count unsatisfied clauses"""
        count = 0
        for clause in clauses:
            if not is_satisfied(assignment, clause):
                count += 1
        return count

    # Attempt multiple times with different strategies
    for attempt in range(20):
        # Initialize assignment
        if attempt == 0:
            # Frequency-based heuristic: set variables based on positive/negative occurrence count
            pos_count = {}
            neg_count = {}
            for var in range(1, num_variables + 1):
                pos_count[var] = 0
                neg_count[var] = 0

            for clause in clauses:
                for lit in clause:
                    if lit > 0:
                        pos_count[lit] += 1
                    else:
                        neg_count[-lit] += 1

            assignment = {}
            for var in range(1, num_variables + 1):
                assignment[var] = 1 if pos_count[var] >= neg_count[var] else 0
        else:
            # Random assignment
            assignment = {i: random.randint(0, 1) for i in range(1, num_variables + 1)}

        # Local search with random walk
        for iteration in range(100000):
            unsatisfied_count = count_unsatisfied(assignment)

            if unsatisfied_count == 0:
                # Found satisfying assignment
                print("SAT")
                print(" ".join(str(assignment[i]) for i in range(1, num_variables + 1)))
                return

            # Find unsatisfied clauses
            unsatisfied_clauses = []
            for clause in clauses:
                if not is_satisfied(assignment, clause):
                    unsatisfied_clauses.append(clause)

            if not unsatisfied_clauses:
                print("SAT")
                print(" ".join(str(assignment[i]) for i in range(1, num_variables + 1)))
                return

            # Pick a random unsatisfied clause
            clause = random.choice(unsatisfied_clauses)

            # Pick a random literal from that clause and flip its variable
            lit = random.choice(clause)
            var = abs(lit)
            assignment[var] = 1 - assignment[var]

solve_3sat()
