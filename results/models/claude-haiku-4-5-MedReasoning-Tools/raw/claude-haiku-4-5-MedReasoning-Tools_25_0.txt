using System;
using System.Collections.Generic;
using System.Linq;
using System.Globalization;

class AsteroidInterception
{
    const double G = 6.67430e-11;
    const double AU = 1.496e11;
    const double DAY = 86400;

    struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator *(Vector3 v, double s) => new Vector3(v.X * s, v.Y * s, v.Z * s);
        public static Vector3 operator *(double s, Vector3 v) => v * s;
        public double Dot(Vector3 other) => X * other.X + Y * other.Y + Z * other.Z;
        public Vector3 Cross(Vector3 other) => new Vector3(Y * other.Z - Z * other.Y, Z * other.X - X * other.Z, X * other.Y - Y * other.X);
        public double Length => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vector3 Normalize() { double len = Length; return len > 0 ? this * (1.0 / len) : this; }
    }

    struct Body
    {
        public string Name;
        public double Mass;
        public Vector3 Position;
        public Vector3 Velocity;
    }

    static void Main()
    {
        var input = Console.In.ReadToEnd().Split(new[] { '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;

        var parts = input[idx++].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int numBodies = int.Parse(parts[0]);
        double warningDays = double.Parse(parts[1], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(parts[2], CultureInfo.InvariantCulture);
        double deltaVBudget = double.Parse(parts[3], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(parts[4], CultureInfo.InvariantCulture);
        long t0Unix = long.Parse(parts[5]);

        var bodies = new List<Body>();
        Body? earth = null, asteroid = null;

        for (int i = 0; i < numBodies; i++)
        {
            parts = input[idx++].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            string name = parts[0];
            double mass = double.Parse(parts[1], CultureInfo.InvariantCulture);
            var pos = new Vector3(
                double.Parse(parts[2], CultureInfo.InvariantCulture),
                double.Parse(parts[3], CultureInfo.InvariantCulture),
                double.Parse(parts[4], CultureInfo.InvariantCulture));
            var vel = new Vector3(
                double.Parse(parts[5], CultureInfo.InvariantCulture),
                double.Parse(parts[6], CultureInfo.InvariantCulture),
                double.Parse(parts[7], CultureInfo.InvariantCulture));

            var body = new Body { Name = name, Mass = mass, Position = pos, Velocity = vel };
            bodies.Add(body);

            if (name == "Earth") earth = body;
            if (name == "Asteroid") asteroid = body;
        }

        string de421Path = input[idx++].Trim();

        var burns = SolveInterception(bodies, earth.Value, asteroid.Value, warningDays, deltaVBudget, asteroidMass);

        Console.WriteLine(burns.Count);
        foreach (var burn in burns)
        {
            Console.WriteLine($"{burn.Time:F6} {burn.DeltaV.X:E15} {burn.DeltaV.Y:E15} {burn.DeltaV.Z:E15}");
        }
    }

    struct Burn
    {
        public double Time;
        public Vector3 DeltaV;
    }

    static List<Burn> SolveInterception(List<Body> bodies, Body earth, Body asteroid, double warningDays, double deltaVBudget, double asteroidMass)
    {
        var burns = new List<Burn>();

        double sunMass = bodies.First(b => b.Name == "Sun").Mass;

        Vector3 earthPos = earth.Position;
        Vector3 asteroidPos = asteroid.Position;
        Vector3 asteroidVel = asteroid.Velocity;

        double impactTime = EstimateImpactTime(earthPos, asteroidPos, asteroidVel, warningDays * DAY);

        if (impactTime < 0 || impactTime > warningDays * DAY)
        {
            impactTime = warningDays * 0.5 * DAY;
        }

        Vector3 earthVel = earth.Velocity;
        double launchTime = 10 * DAY;

        Vector3 interceptPos = asteroidPos + asteroidVel * (impactTime - 0);

        Vector3 launchDeltaV = CalculateLaunchDeltaV(earthPos, earthVel, interceptPos, launchTime, sunMass);

        double launchDVMag = launchDeltaV.Length;
        if (launchDVMag > 0.1)
        {
            burns.Add(new Burn { Time = launchTime / DAY, DeltaV = launchDeltaV });
        }

        double remainingDeltaV = deltaVBudget - launchDVMag;
        if (remainingDeltaV > 100 && impactTime - launchTime > 30 * DAY)
        {
            double correctionTime = launchTime + (impactTime - launchTime) * 0.5;
            Vector3 correctionDeltaV = CalculateCorrectionDeltaV(remainingDeltaV * 0.3);
            burns.Add(new Burn { Time = correctionTime / DAY, DeltaV = correctionDeltaV });
            remainingDeltaV -= correctionDeltaV.Length;
        }

        Vector3 finalDeltaV = CalculateFinalDeltaV(remainingDeltaV * 0.7);
        burns.Add(new Burn { Time = impactTime / DAY, DeltaV = finalDeltaV });

        return burns;
    }

    static double EstimateImpactTime(Vector3 earthPos, Vector3 asteroidPos, Vector3 asteroidVel, double maxTime)
    {
        Vector3 relPos = asteroidPos - earthPos;
        Vector3 relVel = asteroidVel;

        double a = relVel.Dot(relVel);
        double b = 2 * relPos.Dot(relVel);
        double c = relPos.Dot(relPos);

        if (a < 1e-10) return maxTime * 0.5;

        double discriminant = b * b - 4 * a * c;
        if (discriminant < 0) return maxTime * 0.5;

        double t1 = (-b - Math.Sqrt(discriminant)) / (2 * a);
        double t2 = (-b + Math.Sqrt(discriminant)) / (2 * a);

        double t = (t1 > 0) ? t1 : t2;
        return Math.Max(0, Math.Min(t, maxTime));
    }

    static Vector3 CalculateLaunchDeltaV(Vector3 earthPos, Vector3 earthVel, Vector3 targetPos, double launchTime, double sunMass)
    {
        Vector3 relativePos = targetPos - earthPos;
        double distance = relativePos.Length;

        if (distance < 1e6) return new Vector3(0, 0, 0);

        Vector3 direction = relativePos.Normalize();

        double travelTime = launchTime;
        double neededSpeed = distance / (travelTime + 1);
        Vector3 currentVelocity = earthVel;
        Vector3 targetVelocity = direction * neededSpeed;

        Vector3 deltaV = targetVelocity - currentVelocity;

        double mag = deltaV.Length;
        if (mag > 20000) deltaV = deltaV * (20000 / mag);

        return deltaV;
    }

    static Vector3 CalculateCorrectionDeltaV(double budget)
    {
        return new Vector3(Math.Sqrt(budget) * 0.1, Math.Sqrt(budget) * 0.05, 0);
    }

    static Vector3 CalculateFinalDeltaV(double budget)
    {
        double mag = Math.Sqrt(budget * budget);
        return new Vector3(mag * 0.6, mag * 0.8, 0);
    }
}
