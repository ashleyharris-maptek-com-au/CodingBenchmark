#!/usr/bin/env python3
import struct

class SPIRVBuilder:
    def __init__(self):
        self.words = []
        self.id_counter = 0

    def next_id(self):
        self.id_counter += 1
        return self.id_counter

    def emit_word(self, value):
        self.words.append(value & 0xFFFFFFFF)

    def emit_words(self, *values):
        for v in values:
            self.emit_word(v)

    def emit_string(self, s):
        b = s.encode('utf-8') + b'\x00'
        while len(b) % 4 != 0:
            b += b'\x00'
        for i in range(0, len(b), 4):
            word = struct.unpack('<I', b[i:i+4])[0]
            self.emit_word(word)

    def emit_instruction(self, word_count, opcode, *operands):
        self.emit_word((word_count << 16) | opcode)
        for op in operands:
            self.emit_word(op)

    def get_bound(self):
        return self.id_counter + 1

    def to_hex(self):
        data = b''.join(struct.pack('<I', w) for w in self.words)
        return data.hex()

def build_spirv():
    b = SPIRVBuilder()

    # Pre-allocate IDs
    void_t = b.next_id()              # 1
    float_t = b.next_id()             # 2
    vec2_t = b.next_id()              # 3
    vec3_t = b.next_id()              # 4
    vec4_t = b.next_id()              # 5
    glsl_ext = b.next_id()            # 6
    func_void_t = b.next_id()         # 7
    ptr_input_vec3_t = b.next_id()    # 8
    ptr_input_vec2_t = b.next_id()    # 9
    ptr_output_vec4_t = b.next_id()   # 10

    var_world_pos = b.next_id()       # 11
    var_normal = b.next_id()          # 12
    var_uv = b.next_id()              # 13
    var_tangent = b.next_id()         # 14
    var_color = b.next_id()           # 15
    var_out_color = b.next_id()       # 16

    const_float_0 = b.next_id()       # 17
    const_float_1 = b.next_id()       # 18
    const_float_6 = b.next_id()       # 19
    const_float_05 = b.next_id()      # 20
    const_float_03 = b.next_id()      # 21
    const_float_yellow_g = b.next_id() # 22
    const_float_01 = b.next_id()      # 23
    const_vec2_05 = b.next_id()       # 24
    const_vec3_yellow = b.next_id()   # 25
    const_vec3_blue = b.next_id()     # 26

    main_func = b.next_id()           # 27
    label_entry = b.next_id()         # 28

    load_uv = b.next_id()             # 29
    load_normal = b.next_id()         # 30
    cell_uv = b.next_id()             # 31
    cell_id = b.next_id()             # 32
    local_uv = b.next_id()            # 33
    center_offset = b.next_id()       # 34
    dist = b.next_id()                # 35
    is_dot = b.next_id()              # 36
    dot_color = b.next_id()           # 37
    lambert = b.next_id()             # 38
    dot_r = b.next_id()               # 39
    dot_g = b.next_id()               # 40
    dot_b = b.next_id()               # 41
    r = b.next_id()                   # 42
    g = b.next_id()                   # 43
    b2 = b.next_id()                  # 44
    final_rgb = b.next_id()           # 45
    final_color = b.next_id()         # 46

    # Header
    b.emit_words(
        0x07230203,     # Magic number
        0x00010000,     # Version 1.0
        0x00000000,     # Generator
        0x00000000,     # Bound (will update)
        0x00000000      # Reserved
    )

    # Capabilities
    b.emit_instruction(2, 17, 1)

    # Extension import
    b.emit_instruction(3, 11, glsl_ext)
    b.emit_string("GLSL.std.450")

    # Memory model
    b.emit_instruction(3, 14, 0, 1)

    # Entry point
    b.emit_instruction(3, 15, 4, main_func)
    b.emit_string("main")

    # Execution mode
    b.emit_instruction(3, 16, main_func, 7)

    # Decorations
    b.emit_instruction(3, 71, var_world_pos, 30, 0)
    b.emit_instruction(3, 71, var_normal, 30, 1)
    b.emit_instruction(3, 71, var_uv, 30, 2)
    b.emit_instruction(3, 71, var_tangent, 30, 3)
    b.emit_instruction(3, 71, var_color, 30, 4)
    b.emit_instruction(3, 71, var_out_color, 30, 0)

    # Type definitions
    b.emit_instruction(1, 19, void_t)
    b.emit_instruction(2, 22, float_t, 0x3F800000)
    b.emit_instruction(3, 23, vec2_t, float_t, 2)
    b.emit_instruction(3, 23, vec3_t, float_t, 3)
    b.emit_instruction(3, 23, vec4_t, float_t, 4)
    b.emit_instruction(3, 33, func_void_t, void_t)
    b.emit_instruction(4, 32, ptr_input_vec3_t, 1, vec3_t)
    b.emit_instruction(4, 32, ptr_input_vec2_t, 1, vec2_t)
    b.emit_instruction(4, 32, ptr_output_vec4_t, 3, vec4_t)

    # Variables
    b.emit_instruction(4, 59, ptr_input_vec3_t, var_world_pos, 1)
    b.emit_instruction(4, 59, ptr_input_vec3_t, var_normal, 1)
    b.emit_instruction(4, 59, ptr_input_vec2_t, var_uv, 1)
    b.emit_instruction(4, 59, ptr_input_vec3_t, var_tangent, 1)
    b.emit_instruction(4, 59, ptr_input_vec3_t, var_color, 1)
    b.emit_instruction(4, 59, ptr_output_vec4_t, var_out_color, 3)

    # Constants
    b.emit_instruction(3, 43, float_t, const_float_0, 0x00000000)
    b.emit_instruction(3, 43, float_t, const_float_1, 0x3F800000)
    b.emit_instruction(3, 43, float_t, const_float_6, 0x40C00000)
    b.emit_instruction(3, 43, float_t, const_float_05, 0x3F000000)
    b.emit_instruction(3, 43, float_t, const_float_03, 0x3E99999A)
    b.emit_instruction(3, 43, float_t, const_float_yellow_g, 0x3F4CCCCD)
    b.emit_instruction(3, 43, float_t, const_float_01, 0x3DCCCCCD)

    b.emit_instruction(5, 80, vec2_t, const_vec2_05, const_float_05, const_float_05)
    b.emit_instruction(5, 80, vec3_t, const_vec3_yellow,
                       const_float_1, const_float_yellow_g, const_float_0)
    b.emit_instruction(5, 80, vec3_t, const_vec3_blue,
                       const_float_01, const_float_01, const_float_03)

    # Function
    b.emit_instruction(5, 54, void_t, main_func, 0, func_void_t)
    b.emit_instruction(2, 248, label_entry)

    # Load inputs
    b.emit_instruction(4, 61, vec2_t, load_uv, var_uv)
    b.emit_instruction(4, 61, vec3_t, load_normal, var_normal)

    # Multiply uv by 6.0
    b.emit_instruction(5, 133, vec2_t, cell_uv, load_uv, const_float_6)

    # Floor
    b.emit_instruction(5, 12, vec2_t, cell_id, glsl_ext, 8, cell_uv)

    # Fract
    b.emit_instruction(5, 12, vec2_t, local_uv, glsl_ext, 10, cell_uv)

    # Subtract 0.5
    b.emit_instruction(4, 131, vec2_t, center_offset, local_uv, const_vec2_05)

    # Length
    b.emit_instruction(5, 12, float_t, dist, glsl_ext, 66, center_offset)

    # Compare < 0.3
    b.emit_instruction(5, 172, 1, is_dot, dist, const_float_03, 1)

    # Select color
    b.emit_instruction(5, 169, vec3_t, dot_color, is_dot, const_vec3_yellow, const_vec3_blue)

    # Lambertian: clamp normal components
    b.emit_instruction(5, 12, float_t, lambert, glsl_ext, 43, const_float_01, const_float_1, load_normal)

    # Extract components
    b.emit_instruction(5, 81, float_t, dot_r, dot_color, 0)
    b.emit_instruction(5, 81, float_t, dot_g, dot_color, 1)
    b.emit_instruction(5, 81, float_t, dot_b, dot_color, 2)

    # Apply shading
    b.emit_instruction(5, 133, float_t, r, dot_r, lambert)
    b.emit_instruction(5, 133, float_t, g, dot_g, lambert)
    b.emit_instruction(5, 133, float_t, b2, dot_b, lambert)

    # Construct final colors
    b.emit_instruction(5, 80, vec3_t, final_rgb, r, g, b2)
    b.emit_instruction(5, 80, vec4_t, final_color, final_rgb, const_float_1)

    # Store and return
    b.emit_instruction(3, 62, var_out_color, final_color)
    b.emit_instruction(1, 253)
    b.emit_instruction(1, 56)

    # Update bound
    b.words[3] = b.get_bound()

    return b.to_hex()

if __name__ == "__main__":
    print(build_spirv())
