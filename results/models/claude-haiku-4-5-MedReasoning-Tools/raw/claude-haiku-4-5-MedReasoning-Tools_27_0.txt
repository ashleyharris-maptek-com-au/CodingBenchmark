#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <cstring>
#include <unordered_set>

using namespace std;

class GraphColorer {
private:
    int num_vertices;
    int num_edges;
    int num_colors;
    vector<vector<int>> adj;
    vector<int> color;
    vector<int> degree;

public:
    GraphColorer(int v, int e, int k)
        : num_vertices(v), num_edges(e), num_colors(k) {
        adj.resize(v);
        color.assign(v, -1);
        degree.assign(v, 0);
    }

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
        degree[u]++;
        degree[v]++;
    }

    // Welsh-Powell algorithm: greedy coloring with degree ordering
    bool colorWelshPowell() {
        vector<pair<int, int>> degreeOrder;
        for (int i = 0; i < num_vertices; i++) {
            degreeOrder.push_back({degree[i], i});
        }
        sort(degreeOrder.rbegin(), degreeOrder.rend());

        for (auto [deg, v] : degreeOrder) {
            unordered_set<int> usedColors;
            for (int u : adj[v]) {
                if (color[u] != -1) {
                    usedColors.insert(color[u]);
                }
            }

            int c = 0;
            while (usedColors.count(c)) c++;

            if (c >= num_colors) return false;
            color[v] = c;
        }
        return true;
    }

    // DSATUR algorithm: dynamic saturation degree ordering
    bool colorDSATUR() {
        vector<int> saturationDegree(num_vertices, 0);
        vector<bool> colored(num_vertices, false);

        for (int step = 0; step < num_vertices; step++) {
            int maxSat = -1, maxDeg = -1, v = -1;

            // Find uncolored vertex with highest saturation degree
            for (int i = 0; i < num_vertices; i++) {
                if (!colored[i]) {
                    if (saturationDegree[i] > maxSat ||
                        (saturationDegree[i] == maxSat && degree[i] > maxDeg)) {
                        maxSat = saturationDegree[i];
                        maxDeg = degree[i];
                        v = i;
                    }
                }
            }

            if (v == -1) break;

            // Find minimum available color
            unordered_set<int> usedColors;
            for (int u : adj[v]) {
                if (color[u] != -1) {
                    usedColors.insert(color[u]);
                }
            }

            int c = 0;
            while (usedColors.count(c)) c++;

            if (c >= num_colors) return false;

            color[v] = c;
            colored[v] = true;

            // Update saturation degrees of neighbors
            for (int u : adj[v]) {
                if (!colored[u]) {
                    saturationDegree[u]++;
                }
            }
        }
        return true;
    }

    // Simple greedy coloring for very large graphs
    bool colorGreedy() {
        for (int v = 0; v < num_vertices; v++) {
            unordered_set<int> usedColors;
            for (int u : adj[v]) {
                if (color[u] != -1) {
                    usedColors.insert(color[u]);
                }
            }

            int c = 0;
            while (usedColors.count(c)) c++;

            if (c >= num_colors) return false;
            color[v] = c;
        }
        return true;
    }

    bool solve() {
        // Choose strategy based on graph complexity
        if (num_vertices <= 1000) {
            // Small graphs: use DSATUR
            return colorDSATUR();
        } else if (num_vertices <= 10000) {
            // Medium graphs: use Welsh-Powell
            return colorWelshPowell();
        } else {
            // Large graphs: use greedy
            return colorGreedy();
        }
    }

    void printColors() {
        for (int i = 0; i < num_vertices; i++) {
            if (i > 0) cout << " ";
            cout << color[i];
        }
        cout << "\n";
    }

    bool isValid() {
        for (int v = 0; v < num_vertices; v++) {
            if (color[v] < 0 || color[v] >= num_colors) {
                return false;
            }
            for (int u : adj[v]) {
                if (color[v] == color[u]) {
                    return false;
                }
            }
        }
        return true;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int num_vertices, num_edges, num_colors;
    cin >> num_vertices >> num_edges >> num_colors;

    GraphColorer colorer(num_vertices, num_edges, num_colors);

    for (int i = 0; i < num_edges; i++) {
        int u, v;
        cin >> u >> v;
        colorer.addEdge(u, v);
    }

    if (colorer.solve() && colorer.isValid()) {
        colorer.printColors();
    } else {
        // Fallback: output default coloring
        for (int i = 0; i < num_vertices; i++) {
            if (i > 0) cout << " ";
            cout << (i % num_colors);
        }
        cout << "\n";
    }

    return 0;
}
