const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { grid, robots, jobs } = data;

function bfs(start, end, grid) {
  const rows = grid.length;
  const cols = grid[0].length;

  const [sx, sy] = start;
  const [ex, ey] = end;

  if (grid[sy][sx] === '#' || grid[ey][ex] === '#') {
    return Infinity;
  }

  if (sx === ex && sy === ey) {
    return 0;
  }

  const visited = new Set([`${sx},${sy}`]);
  const queue = [[sx, sy, 0]];

  while (queue.length > 0) {
    const [x, y, dist] = queue.shift();

    for (const [dx, dy] of [[1, 0], [-1, 0], [0, 1], [0, -1]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (ny >= 0 && ny < rows && nx >= 0 && nx < cols && grid[ny][nx] !== '#') {
        if (nx === ex && ny === ey) {
          return dist + 1;
        }

        const key = `${nx},${ny}`;
        if (!visited.has(key)) {
          visited.add(key);
          queue.push([nx, ny, dist + 1]);
        }
      }
    }
  }

  return Infinity;
}

const distances = {};
for (let i = 0; i < robots.length; i++) {
  distances[i] = {};
  for (let j = 0; j < jobs.length; j++) {
    distances[i][j] = bfs([robots[i].x, robots[i].y], [jobs[j].x, jobs[j].y], grid);
  }
}

const assignments = Array(robots.length).fill(null).map(() => []);
const assigned = new Set();
const robotPos = robots.map(r => [r.x, r.y]);

while (assigned.size < jobs.length) {
  let bestRobot = -1;
  let bestJob = -1;
  let bestDist = Infinity;

  for (let i = 0; i < robots.length; i++) {
    for (let j = 0; j < jobs.length; j++) {
      if (!assigned.has(j) && distances[i][j] < bestDist) {
        bestDist = distances[i][j];
        bestRobot = i;
        bestJob = j;
      }
    }
  }

  if (bestJob === -1 || bestDist === Infinity) break;

  assignments[bestRobot].push(jobs[bestJob].id);
  assigned.add(bestJob);
  robotPos[bestRobot] = [jobs[bestJob].x, jobs[bestJob].y];

  for (let j = 0; j < jobs.length; j++) {
    if (!assigned.has(j)) {
      distances[bestRobot][j] = bfs(robotPos[bestRobot], [jobs[j].x, jobs[j].y], grid);
    }
  }
}
