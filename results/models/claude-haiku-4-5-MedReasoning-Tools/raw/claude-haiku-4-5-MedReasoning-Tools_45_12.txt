import struct

def encode_dword(value):
    """Encode a 32-bit integer as little-endian hex string (8 characters)"""
    return ''.join(f'{b:02x}' for b in struct.pack('<I', value & 0xFFFFFFFF))

def encode_string(s):
    """Encode a string as SPIR-V format (padded to 4-byte words, null-terminated)"""
    s_bytes = (s + '\0').encode('ascii')
    while len(s_bytes) % 4 != 0:
        s_bytes += b'\0'
    return ''.join(f'{b:02x}' for b in s_bytes)

hex_str = ""

# Header
hex_str += encode_dword(0x07230203)  # Magic
hex_str += encode_dword(0x00010000)  # Version 1.0
hex_str += encode_dword(0x00000000)  # Generator ID
hex_str += encode_dword(70)          # Bound (max ID + 1)
hex_str += encode_dword(0x00000000)  # Schema (reserved)

# OpCapability Shader (opcode 17)
hex_str += encode_dword((2 << 16) | 17)
hex_str += encode_dword(1)  # Capability = Shader

# OpExtInstImport (opcode 11)
glsl_str = encode_string("GLSL.std.450")
glsl_word_count = 2 + len(glsl_str) // 8
hex_str += encode_dword((glsl_word_count << 16) | 11)
hex_str += encode_dword(1)  # Result ID
hex_str += glsl_str

# OpMemoryModel (opcode 14)
hex_str += encode_dword((3 << 16) | 14)
hex_str += encode_dword(0)  # Logical
hex_str += encode_dword(1)  # GLSL450

# OpEntryPoint (opcode 15)
entry_str = encode_string("main")
interface_ids = [11, 12]
word_count = 3 + (len(entry_str) // 8) + len(interface_ids)
hex_str += encode_dword((word_count << 16) | 15)
hex_str += encode_dword(4)  # Fragment
hex_str += encode_dword(2)  # Entry point ID
hex_str += entry_str
for iface_id in interface_ids:
    hex_str += encode_dword(iface_id)

# OpExecutionMode (opcode 16)
hex_str += encode_dword((3 << 16) | 16)
hex_str += encode_dword(2)  # Entry point ID
hex_str += encode_dword(7)  # OriginUpperLeft

# OpDecorate Location (opcode 71, decoration 30)
hex_str += encode_dword((4 << 16) | 71)
hex_str += encode_dword(11)  # normal ID
hex_str += encode_dword(30)  # Location
hex_str += encode_dword(1)   # location value = 1

hex_str += encode_dword((4 << 16) | 71)
hex_str += encode_dword(12)  # outColor ID
hex_str += encode_dword(30)  # Location
hex_str += encode_dword(0)   # location value = 0

# Types
# OpTypeVoid (opcode 19)
hex_str += encode_dword((2 << 16) | 19)
hex_str += encode_dword(1)  # ID 1 = void

# OpTypeFloat (opcode 22)
hex_str += encode_dword((3 << 16) | 22)
hex_str += encode_dword(2)  # ID 2 = float
hex_str += encode_dword(32) # width

# OpTypeVector vec3 (opcode 23)
hex_str += encode_dword((4 << 16) | 23)
hex_str += encode_dword(3)  # ID 3 = vec3
hex_str += encode_dword(2)  # float
hex_str += encode_dword(3)  # count

# OpTypeVector vec4 (opcode 23)
hex_str += encode_dword((4 << 16) | 23)
hex_str += encode_dword(4)  # ID 4 = vec4
hex_str += encode_dword(2)  # float
hex_str += encode_dword(4)  # count

# OpTypePointer Input vec3 (opcode 32)
hex_str += encode_dword((4 << 16) | 32)
hex_str += encode_dword(7)  # ID 7
hex_str += encode_dword(1)  # Input
hex_str += encode_dword(3)  # vec3

# OpTypePointer Output vec4 (opcode 32)
hex_str += encode_dword((4 << 16) | 32)
hex_str += encode_dword(8)  # ID 8
hex_str += encode_dword(3)  # Output
hex_str += encode_dword(4)  # vec4

# OpTypeFunction void (opcode 33)
hex_str += encode_dword((3 << 16) | 33)
hex_str += encode_dword(9)  # ID 9
hex_str += encode_dword(1)  # return type void

# Constants
const_id = 10
floats = [0.3, 0.15, 0.05, 0.4, 0.6, 1.0, 0.5, 0.0]
float_ids = {}
for f in floats:
    float_ids[f] = const_id
    bits = struct.unpack('<I', struct.pack('<f', f))[0]
    hex_str += encode_dword((4 << 16) | 43)  # OpConstant
    hex_str += encode_dword(2)  # type float
    hex_str += encode_dword(const_id)  # ID
    hex_str += encode_dword(bits)  # value
    const_id += 1

# Composite constants
# OpCompositeConstruct skyColor vec3(0.4, 0.6, 1.0)
sky_id = const_id
const_id += 1
hex_str += encode_dword((5 << 16) | 80)  # OpCompositeConstruct
hex_str += encode_dword(3)  # type vec3
hex_str += encode_dword(sky_id)
hex_str += encode_dword(float_ids[0.4])
hex_str += encode_dword(float_ids[0.6])
hex_str += encode_dword(float_ids[1.0])

# OpCompositeConstruct groundColor vec3(0.3, 0.15, 0.05)
ground_id = const_id
const_id += 1
hex_str += encode_dword((5 << 16) | 80)
hex_str += encode_dword(3)  # type vec3
hex_str += encode_dword(ground_id)
hex_str += encode_dword(float_ids[0.3])
hex_str += encode_dword(float_ids[0.15])
hex_str += encode_dword(float_ids[0.05])

# Variables
# OpVariable Input vec3 normal
hex_str += encode_dword((4 << 16) | 59)  # OpVariable
hex_str += encode_dword(7)  # type ptr Input vec3
hex_str += encode_dword(11) # ID 11 = normal
hex_str += encode_dword(1)  # Input

# OpVariable Output vec4 outColor
hex_str += encode_dword((4 << 16) | 59)
hex_str += encode_dword(8)  # type ptr Output vec4
hex_str += encode_dword(12) # ID 12 = outColor
hex_str += encode_dword(3)  # Output

# Function
# OpFunction void
func_id = 2
hex_str += encode_dword((5 << 16) | 54)  # OpFunction
hex_str += encode_dword(1)  # return type void
hex_str += encode_dword(func_id)  # ID 2 = main
hex_str += encode_dword(0)  # control
hex_str += encode_dword(9)  # function type

# OpLabel
label_id = const_id
const_id += 1
hex_str += encode_dword((2 << 16) | 248)
hex_str += encode_dword(label_id)

# OpLoad normal
normal_loaded_id = const_id
const_id += 1
hex_str += encode_dword((4 << 16) | 61)  # OpLoad
hex_str += encode_dword(3)  # type vec3
hex_str += encode_dword(normal_loaded_id)
hex_str += encode_dword(11)  # pointer normal

# OpCompositeExtract normal.y
normal_y_id = const_id
const_id += 1
hex_str += encode_dword((5 << 16) | 81)  # OpCompositeExtract
hex_str += encode_dword(2)  # type float
hex_str += encode_dword(normal_y_id)
hex_str += encode_dword(normal_loaded_id)
hex_str += encode_dword(1)  # index y

# OpFMul normal.y * 0.5
mul_id = const_id
const_id += 1
hex_str += encode_dword((4 << 16) | 133)  # OpFMul
hex_str += encode_dword(2)  # type float
hex_str += encode_dword(mul_id)
hex_str += encode_dword(normal_y_id)
hex_str += encode_dword(float_ids[0.5])

# OpFAdd (normal.y * 0.5) + 0.5
blend_factor_id = const_id
const_id += 1
hex_str += encode_dword((4 << 16) | 129)  # OpFAdd
hex_str += encode_dword(2)  # type float
hex_str += encode_dword(blend_factor_id)
hex_str += encode_dword(mul_id)
hex_str += encode_dword(float_ids[0.5])

# OpFSub skyColor - groundColor
diff_id = const_id
const_id += 1
hex_str += encode_dword((4 << 16) | 131)  # OpFSub
hex_str += encode_dword(3)  # type vec3
hex_str += encode_dword(diff_id)
hex_str += encode_dword(sky_id)
hex_str += encode_dword(ground_id)

# OpCompositeConstruct blend_factor as vec3
blend_vec_id = const_id
const_id += 1
hex_str += encode_dword((5 << 16) | 80)  # OpCompositeConstruct
hex_str += encode_dword(3)  # type vec3
hex_str += encode_dword(blend_vec_id)
hex_str += encode_dword(blend_factor_id)
hex_str += encode_dword(blend_factor_id)
hex_str += encode_dword(blend_factor_id)

# OpFMul (skyColor - groundColor) * blendFactor
weighted_diff_id = const_id
const_id += 1
hex_str += encode_dword((4 << 16) | 133)  # OpFMul
hex_str += encode_dword(3)  # type vec3
hex_str += encode_dword(weighted_diff_id)
hex_str += encode_dword(diff_id)
hex_str += encode_dword(blend_vec_id)

# OpFAdd groundColor + weighted_diff
ambient_id = const_id
const_id += 1
hex_str += encode_dword((4 << 16) | 129)  # OpFAdd
hex_str += encode_dword(3)  # type vec3
hex_str += encode_dword(ambient_id)
hex_str += encode_dword(ground_id)
hex_str += encode_dword(weighted_diff_id)

# Add diffuse term: multiply ambient by (1 + 0.3) to incorporate small diffuse boost
diffuse_factor_id = const_id
const_id += 1
diffuse_boost = 0.3
bits_boost = struct.unpack('<I', struct.pack('<f', diffuse_boost))[0]
hex_str += encode_dword((4 << 16) | 43)  # OpConstant
hex_str += encode_dword(2)  # type float
hex_str += encode_dword(diffuse_factor_id)
hex_str += encode_dword(bits_boost)

# OpCompositeConstruct diffuse boost as vec3
diffuse_vec_id = const_id
const_id += 1
hex_str += encode_dword((5 << 16) | 80)
hex_str += encode_dword(3)  # type vec3
hex_str += encode_dword(diffuse_vec_id)
hex_str += encode_dword(diffuse_factor_id)
hex_str += encode_dword(diffuse_factor_id)
hex_str += encode_dword(diffuse_factor_id)

# OpFMul ambient * (1 + diffuse)
final_color_id = const_id
const_id += 1
hex_str += encode_dword((4 << 16) | 133)  # OpFMul
hex_str += encode_dword(3)  # type vec3
hex_str += encode_dword(final_color_id)
hex_str += encode_dword(ambient_id)
hex_str += encode_dword(diffuse_vec_id)

# Convert vec3 to vec4 by extracting components and adding alpha = 1.0
x_id = const_id
const_id += 1
hex_str += encode_dword((5 << 16) | 81)  # OpCompositeExtract
hex_str += encode_dword(2)  # type float
hex_str += encode_dword(x_id)
hex_str += encode_dword(final_color_id)
hex_str += encode_dword(0)  # index x

y_id = const_id
const_id += 1
hex_str += encode_dword((5 << 16) | 81)
hex_str += encode_dword(2)
hex_str += encode_dword(y_id)
hex_str += encode_dword(final_color_id)
hex_str += encode_dword(1)  # index y

z_id = const_id
const_id += 1
hex_str += encode_dword((5 << 16) | 81)
hex_str += encode_dword(2)
hex_str += encode_dword(z_id)
hex_str += encode_dword(final_color_id)
hex_str += encode_dword(2)  # index z

# OpCompositeConstruct final vec4
output_id = const_id
const_id += 1
hex_str += encode_dword((5 << 16) | 80)  # OpCompositeConstruct
hex_str += encode_dword(4)  # type vec4
hex_str += encode_dword(output_id)
hex_str += encode_dword(x_id)
hex_str += encode_dword(y_id)
hex_str += encode_dword(z_id)
hex_str += encode_dword(float_ids[1.0])  # alpha = 1.0

# OpStore outColor
hex_str += encode_dword((3 << 16) | 62)  # OpStore
hex_str += encode_dword(12)  # pointer outColor
hex_str += encode_dword(output_id)

# OpReturn
hex_str += encode_dword((1 << 16) | 253)

# OpFunctionEnd
hex_str += encode_dword((1 << 16) | 56)

print(hex_str)
