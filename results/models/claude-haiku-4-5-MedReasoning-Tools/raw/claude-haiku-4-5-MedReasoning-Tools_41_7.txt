OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations for inputs/outputs
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0

; Decorations for UBO
OpDecorate %UBO Block
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Pointer types
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

; Function type
%void_func = OpTypeFunction %void

; UBO struct type
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Variables
%worldPos = OpVariable %ptr_Input_vec3 Input
%normal = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%uboVar = OpVariable %ptr_Uniform_UBO Uniform

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_3 = OpConstant %float 3.0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%baseColor = OpConstantComposite %vec3 %float_0_2 %float_0_3 %float_0_8
%rimTint = OpConstantComposite %vec3 %float_1 %float_0_8 %float_0_5

; Main function
%main = OpFunction %void None %void_func
%entry = OpLabel

; External function extensions must be declared here
%GLSL = OpExtInstImport "GLSL.std.450"

; Load inputs
%wPos = OpLoad %vec3 %worldPos
%nml = OpLoad %vec3 %normal

; Normalize normal
%N = OpExtInst %vec3 %GLSL normalize %nml

; Load light position from UBO[3]
%lightPos_ptr = OpAccessChain %ptr_Uniform_vec4 %uboVar %int_3
%lightPos_vec4 = OpLoad %vec4 %lightPos_ptr
%lightPos = OpVectorShuffle %vec3 %lightPos_vec4 %lightPos_vec4 0 1 2

; Load camera position from UBO[4]
%cameraPos_ptr = OpAccessChain %ptr_Uniform_vec4 %uboVar %int_4
%cameraPos_vec4 = OpLoad %vec4 %cameraPos_ptr
%cameraPos = OpVectorShuffle %vec3 %cameraPos_vec4 %cameraPos_vec4 0 1 2

; Calculate view direction: V = normalize(cameraPos - worldPos)
%viewDir_raw = OpFSub %vec3 %cameraPos %wPos
%V = OpExtInst %vec3 %GLSL normalize %viewDir_raw

; Calculate light direction: L = normalize(lightPos - worldPos)
%lightDir_raw = OpFSub %vec3 %lightPos %wPos
%L = OpExtInst %vec3 %GLSL normalize %lightDir_raw

; Calculate diffuse: max(dot(N, L), 0)
%dot_NL = OpDot %float %N %L
%diffuse = OpExtInst %float %GLSL fmax %dot_NL %float_0

; Calculate rim: pow(1 - max(dot(N,V),0), 3)
%dot_NV = OpDot %float %N %V
%dot_NV_clamped = OpExtInst %float %GLSL fmax %dot_NV %float_0
%one_minus_dot = OpFSub %float %float_1 %dot_NV_clamped
%rim = OpExtInst %float %GLSL pow %one_minus_dot %float_3

; Calculate modulation: 0.3 + 0.5 * diffuse
%mod_mul_diffuse = OpFMul %float %float_0_5 %diffuse
%modulation = OpFAdd %float %float_0_3 %mod_mul_diffuse

; Calculate base color with modulation: (0.2, 0.3, 0.8) * modulation
%baseColor_modulated = OpVectorTimesScalar %vec3 %baseColor %modulation

; Calculate rim contribution: (1.0, 0.8, 0.5) * rim
%rim_contribution = OpVectorTimesScalar %vec3 %rimTint %rim

; Final color: baseColor_modulated + rim_contribution
%finalColor = OpFAdd %vec3 %baseColor_modulated %rim_contribution

; Convert to vec4 with alpha 1.0
%outputColor = OpCompositeConstruct %vec4 %finalColor %float_1

; Store output
OpStore %outColor %outputColor

OpReturn
OpFunctionEnd
