"""
A320-class Twin-Engine Autopilot with Sensor Fusion and Safety Systems
"""

import statistics
from collections import defaultdict

# ============ MODULE PERSISTENT STATE ============

# PID controller integrals (persist across calls)
pid_integral = defaultdict(float)

# Sensor history for anomaly detection and smoothing
sensor_history = defaultdict(list)

# Track engine failure state (once detected, remains failed)
engine_failure_state = [False, False]

# Trim deflection integral for gradual trim management
trim_deflection_integral = 0.0

# Last valid readings (fallback when all sensors fail)
last_valid_state = {
    'altitude': 5000.0,
    'ias': 250.0,
    'heading': 0.0,
    'pitch': 0.0,
    'roll': 0.0,
    'vs': 0.0,
}


def autopilot_step(state: dict, dt: float) -> dict:
    """
    Main autopilot control loop for A320-class twin-engine airliner.

    Handles:
    - Redundant sensor fusion with failure detection
    - Altitude, heading, airspeed control
    - Engine failure compensation
    - Stall and overspeed protection
    - Trim management
    - Landing gear and flaps
    """
    global engine_failure_state, trim_deflection_integral, last_valid_state, pid_integral

    dt = max(dt, 0.001)
    safe_dt = min(dt, 0.1)  # Cap timestep for numerical stability

    # ===== SENSOR FUSION WITH REDUNDANCY AND ANOMALY DETECTION =====

    # Altitude: vote between barometric, radar, and GPS
    alt_baro = sensor_vote(
        [state.get(f'baro_alt_{i}_ft') for i in range(1, 4)],
        [state.get(f'baro_alt_{i}_valid', False) for i in range(1, 4)],
        key='alt_baro', history_len=5
    )

    alt_radar = sensor_vote(
        [state.get(f'radar_alt_{i}_ft') for i in range(1, 4)],
        [state.get(f'radar_alt_{i}_valid', False) for i in range(1, 4)],
        key='alt_radar', history_len=5
    )

    alt_gps = sensor_vote(
        [state.get(f'gps_{i}_alt_ft') for i in range(1, 4)],
        [state.get(f'gps_{i}_valid', False) for i in range(1, 4)],
        key='alt_gps', history_len=5
    )

    # Cross-check altitude sources: radar is accurate below 2500 ft, barometric higher
    altitude = select_best_altitude(alt_baro, alt_radar, alt_gps, last_valid_state['altitude'])
    last_valid_state['altitude'] = altitude

    # Airspeed from redundant Air Data Computers
    ias = sensor_vote(
        [state.get(f'ias_{i}_kt') for i in range(1, 4)],
        [state.get(f'ias_{i}_valid', False) for i in range(1, 4)],
        key='ias', history_len=10
    )
    if ias is None:
        ias = last_valid_state['ias']
    last_valid_state['ias'] = ias

    # Pitch attitude from IRS/INS redundancy
    pitch = sensor_vote(
        [state.get(f'irs_{i}_pitch_deg') for i in range(1, 4)],
        [state.get(f'irs_{i}_valid', False) for i in range(1, 4)],
        key='pitch', history_len=5
    ) or 0.0
    last_valid_state['pitch'] = pitch

    # Roll attitude
    roll = sensor_vote(
        [state.get(f'irs_{i}_roll_deg') for i in range(1, 4)],
        [state.get(f'irs_{i}_valid', False) for i in range(1, 4)],
        key='roll', history_len=5
    ) or 0.0
    last_valid_state['roll'] = roll

    # Heading from IRS or compass fallback
    heading = sensor_vote(
        [state.get(f'irs_{i}_heading_deg') for i in range(1, 4)],
        [state.get(f'irs_{i}_valid', False) for i in range(1, 4)],
        key='heading', history_len=5
    )
    if heading is None:
        heading = state.get('magnetic_heading_deg', 0)
    last_valid_state['heading'] = heading

    # Vertical speed from IRS
    vs = sensor_vote(
        [state.get(f'irs_{i}_vs_fpm') for i in range(1, 4)],
        [state.get(f'irs_{i}_valid', False) for i in range(1, 4)],
        key='vs', history_len=5
    ) or 0.0
    last_valid_state['vs'] = vs

    # ===== FLIGHT STATE DETECTION =====

    # Weight-on-wheels (landing gear contact sensors)
    on_ground = any([
        state.get('weight_on_wheels_left', 0),
        state.get('weight_on_wheels_nose', 0),
        state.get('weight_on_wheels_right', 0),
    ])

    # Engine status and failure detection
    engine_1_running = state.get('engine_1_running', True)
    engine_2_running = state.get('engine_2_running', True)

    # Detect engine failures (once detected, stays failed for this scenario)
    if not engine_1_running:
        engine_failure_state[0] = True
    if not engine_2_running:
        engine_failure_state[1] = True

    num_engines_available = sum([not engine_failure_state[0], not engine_failure_state[1]])

    # ===== AUTOPILOT TARGETS =====

    ap_engaged = state.get('ap_engaged', False)
    at_engaged = state.get('at_engaged', False)

    target_altitude = state.get('target_alt_ft', altitude)
    target_heading = state.get('target_hdg_deg', heading)
    target_ias = state.get('target_ias_kt', 250)
    target_vs = state.get('target_vs_fpm', 0)

    # ===== SAFETY LIMITS =====

    # Stall speed: A320 empty ~95 KIAS, max landing weight ~110 KIAS
    # Use conservative estimate with safety margin
    stall_speed_estimate = 110  # KIAS
    min_safe_ias = stall_speed_estimate * 1.3  # 30% safety margin

    # Maximum operating speed (Vne limit)
    vne_limit = 365  # KIAS (as specified)

    # ===== ALTITUDE HOLD / VERTICAL SPEED CONTROL =====

    alt_error = target_altitude - altitude

    # Proportional-integral control for altitude
    pid_integral['altitude'] += alt_error * safe_dt
    pid_integral['altitude'] = clamp(pid_integral['altitude'], -1500, 1500)

    # Altitude error drives desired vertical speed
    # Rule of thumb: 5-10 fpm per foot of error
    kp_alt = 8.0
    ki_alt = 0.2
    desired_vs = clamp(
        alt_error * kp_alt + pid_integral['altitude'] * ki_alt,
        -2000, 2000
    )

    # If target VS is set, blend towards it
    if abs(alt_error) < 100:
        desired_vs = target_vs

    # ===== PITCH CONTROL FOR VERTICAL SPEED =====

    vs_error = desired_vs - vs
    pid_integral['vs'] += vs_error * safe_dt
    pid_integral['vs'] = clamp(pid_integral['vs'], -500, 500)

    # Relationship: ~200 fpm per degree of pitch (typical for A320 in cruise)
    pitch_per_fpm_vs = 1.0 / 200.0
    desired_pitch = clamp(desired_vs * pitch_per_fpm_vs, -25, 25)

    pitch_error = desired_pitch - pitch
    pid_integral['pitch'] += pitch_error * safe_dt
    pid_integral['pitch'] = clamp(pid_integral['pitch'], -50, 50)

    # Elevator control law
    kp_pitch = 0.15
    ki_pitch = 0.02
    elevator_cmd = pitch_error * kp_pitch + pid_integral['pitch'] * ki_pitch

    # ===== HEADING / ROLL CONTROL =====

    hdg_error = normalize_angle(target_heading - heading)

    # Roll command: ~0.1 degrees bank per degree heading error
    max_bank = 30  # degrees (typical autopilot limit)
    desired_bank = clamp(hdg_error * 0.15, -max_bank, max_bank)

    roll_error = desired_bank - roll
    pid_integral['roll'] += roll_error * safe_dt
    pid_integral['roll'] = clamp(pid_integral['roll'], -100, 100)

    # Aileron control law
    kp_roll = 0.08
    ki_roll = 0.005
    aileron_cmd = roll_error * kp_roll + pid_integral['roll'] * ki_roll

    # Rudder for yaw damping and coordination
    # Compensate for adverse yaw and maintain coordinated flight
    yaw_damping = -roll_error * 0.05
    rudder_cmd = clamp(yaw_damping, -1.0, 1.0)

    # ===== AIRSPEED / THROTTLE CONTROL =====

    speed_error = target_ias - ias
    pid_integral['speed'] += speed_error * safe_dt
    pid_integral['speed'] = clamp(pid_integral['speed'], -500, 500)

    # Throttle response: typically slower than pitch
    kp_speed = 0.0015
    ki_speed = 0.0002
    throttle_cmd = 0.5 + speed_error * kp_speed + pid_integral['speed'] * ki_speed

    # ===== SAFETY PROTECTION SYSTEMS =====

    # STALL PROTECTION: Extremely aggressive if airspeed low
    if ias < min_safe_ias:
        # Full throttle, lower nose, no speedbrake
        throttle_cmd = 1.0
        elevator_cmd = clamp(elevator_cmd - 0.8, -1.0, 1.0)
        speedbrake_cmd = 0.0
        aileron_cmd = clamp(aileron_cmd, -0.5, 0.5)  # Reduce roll aggressiveness

    # OVERSPEED PROTECTION: Stop engines, extend controls
    elif ias > vne_limit - 10:
        throttle_cmd = 0.0
        elevator_cmd = clamp(elevator_cmd + 0.5, -1.0, 1.0)
        speedbrake_cmd = 1.0

    else:
        speedbrake_cmd = 0.0

    # PITCH ATTITUDE PROTECTION: Prevent extreme pitch attitudes
    if pitch > 30:
        elevator_cmd = min(elevator_cmd, -0.3)
    elif pitch < -30:
        elevator_cmd = max(elevator_cmd, 0.3)

    # ROLL ATTITUDE PROTECTION: Prevent excessive bank
    if abs(roll) > 60:
        aileron_cmd = -0.2 if roll > 0 else 0.2

    # ===== ENGINE FAILURE COMPENSATION =====

    throttle_1 = throttle_cmd
    throttle_2 = throttle_cmd

    # With one engine failed, increase the other
    if engine_failure_state[0]:
        throttle_1 = 0.0
        if num_engines_available > 0:
            throttle_2 = min(throttle_cmd / num_engines_available * 1.1, 1.0)
        # Asymmetric thrust compensation: more aileron to right
        aileron_cmd = clamp(aileron_cmd - 0.15, -1.0, 1.0)

    if engine_failure_state[1]:
        throttle_2 = 0.0
        if num_engines_available > 0:
            throttle_1 = min(throttle_cmd / num_engines_available * 1.1, 1.0)
        # Asymmetric thrust compensation: more aileron to left
        aileron_cmd = clamp(aileron_cmd + 0.15, -1.0, 1.0)

    # Single-engine approach: increase available engine
    if num_engines_available == 1:
        if not engine_failure_state[0]:
            throttle_1 = min(throttle_cmd * 1.1, 1.0)
        if not engine_failure_state[1]:
            throttle_2 = min(throttle_cmd * 1.1, 1.0)

    # ===== TRIM MANAGEMENT =====

    # Gradually adjust trim to follow sustained elevator deflection
    # This reduces elevator workload and improves handling
    trim_adjustment_rate = 0.05
    elevator_trim_cmd = elevator_cmd * trim_adjustment_rate
    trim_deflection_integral += elevator_trim_cmd * safe_dt
    trim_deflection_integral = clamp(trim_deflection_integral, -1.0, 1.0)

    # ===== LANDING GEAR MANAGEMENT =====

    if on_ground:
        gear_cmd = 1  # Down
    elif altitude < 1000:
        # Below 1000 ft, always down
        gear_cmd = 1
    elif altitude > 3000:
        # Above 3000 ft, assume up
        gear_cmd = 0
    else:
        # Hysteresis: keep last state
        gear_cmd = state.get('gear_handle_down', 0)

    # ===== FLAPS MANAGEMENT =====

    # Simple rule: extend flaps as altitude decreases
    if on_ground or altitude < 500:
        target_flaps = 40  # Landing flaps
    elif altitude < 1500:
        target_flaps = 30
    elif altitude < 3000:
        target_flaps = 15
    else:
        target_flaps = 0  # Cruise

    # ===== BUILD OUTPUT COMMAND DICT =====

    output = {
        'elevator': clamp(elevator_cmd - trim_deflection_integral, -1.0, 1.0),
        'aileron': clamp(aileron_cmd, -1.0, 1.0),
        'rudder': clamp(rudder_cmd, -1.0, 1.0),
        'throttle_1': clamp(throttle_1, 0.0, 1.0),
        'throttle_2': clamp(throttle_2, 0.0, 1.0),
        'elevator_trim': clamp(trim_deflection_integral, -1.0, 1.0),
        'flaps': int(target_flaps),
        'speedbrake': clamp(speedbrake_cmd, 0.0, 1.0),
        'gear': int(gear_cmd),
    }

    return output


def sensor_vote(values, valid_flags, key='', history_len=10):
    """
    Vote on sensor value using redundancy with anomaly detection.

    Strategy:
    1. Prefer sensors flagged as valid
    2. Use median for robustness against single failure
    3. Detect outliers and remove them
    4. Fallback to any non-None reading if needed

    Args:
        values: List of 3 sensor readings
        valid_flags: List of 3 validity flags (can be wrong!)
        key: Sensor key for history tracking
        history_len: Keep last N readings for smoothing

    Returns:
        Voted value or None if all sensors failed
    """
    global sensor_history

    # Track history
    if key not in sensor_history:
        sensor_history[key] = []

    for v, valid in zip(values, valid_flags):
        if v is not None:
            sensor_history[key].append((v, valid))

    # Keep only recent history
    sensor_history[key] = sensor_history[key][-history_len:]

    # Collect non-None values
    readings = [v for v in values if v is not None]

    if not readings:
        return None

    if len(readings) == 1:
        return readings[0]

    # Use median as primary vote
    median_val = statistics.median(readings)

    # Check for outliers: if any reading is far from median, likely a failure
    threshold = max(abs(median_val) * 0.05, 10)  # 5% or 10 unit tolerance
    valid_readings = [v for v in readings if abs(v - median_val) < threshold]

    if valid_readings:
        return statistics.median(valid_readings)

    return median_val


def select_best_altitude(alt_baro, alt_radar, alt_gps, fallback):
    """
    Select the most reliable altitude source based on sensor agreement.

    Radar is most accurate near ground (<2500 ft).
    Barometric is most reliable in cruise.
    GPS can drift or lose lock.
    """
    sources = [alt_baro, alt_radar, alt_gps]
    valid_sources = [s for s in sources if s is not None]

    if not valid_sources:
        return fallback

    if len(valid_sources) == 1:
        return valid_sources[0]

    # Cross-check agreement
    median_alt = statistics.median(valid_sources)

    # Remove outliers (sensor failure detection)
    consensus_sources = [s for s in valid_sources if abs(s - median_alt) < 500]

    if consensus_sources:
        return statistics.median(consensus_sources)

    return median_alt


def normalize_angle(angle):
    """
    Normalize angle to range (-180, 180] degrees.
    """
    while angle > 180:
        angle -= 360
    while angle <= -180:
        angle += 360
    return angle


def clamp(value, min_val, max_val):
    """
    Clamp a value to [min_val, max_val].
    """
    return max(min_val, min(value, max_val))
