#!/usr/bin/env python3
import struct

class SPIRVBuilder:
    def __init__(self):
        self.words = []
        self.id_counter = 1

    def get_id(self):
        id_val = self.id_counter
        self.id_counter += 1
        return id_val

    def add_word(self, val):
        """Add a 32-bit word as integer, will be converted to bytes later."""
        self.words.append(val & 0xFFFFFFFF)

    def add_words(self, vals):
        """Add multiple 32-bit words."""
        for val in vals:
            self.add_word(val)

    def add_instr(self, opcode, operands):
        """Add an instruction."""
        word_count = 1 + len(operands)
        header = (word_count << 16) | opcode
        self.add_word(header)
        for op in operands:
            if isinstance(op, str):
                self.add_string(op)
            elif isinstance(op, float):
                # Pack as little-endian IEEE 754
                bits = struct.unpack('<I', struct.pack('<f', op))[0]
                self.add_word(bits)
            else:
                self.add_word(op)

    def add_string(self, s):
        """Add a null-terminated string as 32-bit words."""
        s_bytes = (s + '\0').encode('utf-8')
        # Pad to multiple of 4 bytes
        padding = (4 - (len(s_bytes) % 4)) % 4
        s_bytes += b'\x00' * padding
        for i in range(0, len(s_bytes), 4):
            chunk = s_bytes[i:i+4]
            val = struct.unpack('<I', chunk)[0]
            self.add_word(val)

    def to_hex(self):
        """Convert to hex string (little-endian words as bytes)."""
        result_bytes = bytearray()
        for word_val in self.words:
            # Pack each word as 4 bytes in little-endian order
            result_bytes.extend(struct.pack('<I', word_val))
        # Return as hex string
        return result_bytes.hex()

    def get_bound(self):
        return self.id_counter

# Build SPIR-V shader
builder = SPIRVBuilder()

# Pre-allocate all IDs
TYPE_VOID = builder.get_id()
TYPE_FLOAT = builder.get_id()
TYPE_VEC3 = builder.get_id()
TYPE_VEC4 = builder.get_id()
TYPE_MAT4 = builder.get_id()
TYPE_PTR_INPUT_VEC3 = builder.get_id()
TYPE_PTR_OUTPUT_VEC4 = builder.get_id()
TYPE_PTR_UNIFORM_VEC4 = builder.get_id()
TYPE_INT = builder.get_id()
TYPE_STRUCT = builder.get_id()
TYPE_PTR_UNIFORM_STRUCT = builder.get_id()
TYPE_FUNC_VOID = builder.get_id()
TYPE_PTR_UNIFORM_MAT4 = builder.get_id()

CONST_0F = builder.get_id()
CONST_02F = builder.get_id()
CONST_05F = builder.get_id()
CONST_08F = builder.get_id()
CONST_1F = builder.get_id()
CONST_32F = builder.get_id()
CONST_0I = builder.get_id()
CONST_3I = builder.get_id()
CONST_4I = builder.get_id()

VAR_WORLDPOS = builder.get_id()
VAR_NORMAL = builder.get_id()
VAR_OUTCOLOR = builder.get_id()
VAR_UBO = builder.get_id()

ENTRY_POINT = builder.get_id()
LABEL_MAIN = builder.get_id()

# Now build from header
# Header
builder.add_word(0x07230203)  # Magic (already in proper little-endian value)
builder.add_word(0x00010000)  # Version 1.0
builder.add_word(0x00000000)  # Generator
builder.add_word(builder.get_bound())  # Bound
builder.add_word(0x00000000)  # Reserved

# OpCapability Shader
builder.add_instr(17, [1])

# OpExtInstImport result=1 "GLSL.std.450"
EXT_IMPORT_ID = 1
builder.add_instr(11, [EXT_IMPORT_ID, "GLSL.std.450"])

# OpMemoryModel Logical GLSL450
builder.add_instr(14, [0, 1])

# OpEntryPoint Fragment result=ENTRY_POINT "main"
builder.add_instr(15, [4, ENTRY_POINT, "main"])

# OpExecutionMode ENTRY_POINT OriginUpperLeft
builder.add_instr(16, [ENTRY_POINT, 7])

# Decorations
builder.add_instr(71, [VAR_WORLDPOS, 30, 0])    # worldPos @ location 0
builder.add_instr(71, [VAR_NORMAL, 30, 1])      # normal @ location 1
builder.add_instr(71, [VAR_OUTCOLOR, 30, 0])    # outColor @ location 0
builder.add_instr(71, [VAR_UBO, 34, 0])         # UBO @ descriptor set 0
builder.add_instr(71, [VAR_UBO, 33, 0])         # UBO @ binding 0
builder.add_instr(71, [TYPE_STRUCT, 2])         # Block decoration

# Member decorations
builder.add_instr(72, [TYPE_STRUCT, 0, 35, 0])     # mat4 offset 0
builder.add_instr(72, [TYPE_STRUCT, 1, 35, 64])    # mat4 offset 64
builder.add_instr(72, [TYPE_STRUCT, 2, 35, 128])   # mat4 offset 128
builder.add_instr(72, [TYPE_STRUCT, 3, 35, 192])   # vec4 offset 192
builder.add_instr(72, [TYPE_STRUCT, 4, 35, 208])   # vec4 offset 208
builder.add_instr(72, [TYPE_STRUCT, 5, 35, 224])   # vec4 offset 224

# Type declarations
builder.add_instr(19, [TYPE_VOID])                  # OpTypeVoid
builder.add_instr(22, [TYPE_FLOAT, 32])             # OpTypeFloat
builder.add_instr(23, [TYPE_VEC3, TYPE_FLOAT, 3])   # OpTypeVector float x3
builder.add_instr(23, [TYPE_VEC4, TYPE_FLOAT, 4])   # OpTypeVector float x4
builder.add_instr(24, [TYPE_MAT4, TYPE_VEC4, 4])    # OpTypeMatrix
builder.add_instr(32, [TYPE_PTR_INPUT_VEC3, 1, TYPE_VEC3])       # OpTypePointer Input
builder.add_instr(32, [TYPE_PTR_OUTPUT_VEC4, 3, TYPE_VEC4])      # OpTypePointer Output
builder.add_instr(32, [TYPE_PTR_UNIFORM_VEC4, 2, TYPE_VEC4])     # OpTypePointer Uniform
builder.add_instr(21, [TYPE_INT, 32, 1])            # OpTypeInt
builder.add_instr(30, [TYPE_STRUCT, TYPE_MAT4, TYPE_MAT4, TYPE_MAT4, TYPE_VEC4, TYPE_VEC4, TYPE_VEC4])  # OpTypeStruct
builder.add_instr(32, [TYPE_PTR_UNIFORM_STRUCT, 2, TYPE_STRUCT])  # OpTypePointer Uniform struct
builder.add_instr(33, [TYPE_FUNC_VOID, TYPE_VOID])   # OpTypeFunction
builder.add_instr(32, [TYPE_PTR_UNIFORM_MAT4, 2, TYPE_MAT4])     # OpTypePointer Uniform mat4

# Constants
builder.add_instr(43, [TYPE_FLOAT, CONST_0F, 0.0])    # 0.0
builder.add_instr(43, [TYPE_FLOAT, CONST_02F, 0.2])   # 0.2
builder.add_instr(43, [TYPE_FLOAT, CONST_05F, 0.5])   # 0.5
builder.add_instr(43, [TYPE_FLOAT, CONST_08F, 0.8])   # 0.8
builder.add_instr(43, [TYPE_FLOAT, CONST_1F, 1.0])    # 1.0
builder.add_instr(43, [TYPE_FLOAT, CONST_32F, 32.0])  # 32.0
builder.add_instr(43, [TYPE_INT, CONST_0I, 0])        # 0
builder.add_instr(43, [TYPE_INT, CONST_3I, 3])        # 3
builder.add_instr(43, [TYPE_INT, CONST_4I, 4])        # 4

# Variables
builder.add_instr(59, [TYPE_PTR_INPUT_VEC3, 1, VAR_WORLDPOS])    # worldPos input
builder.add_instr(59, [TYPE_PTR_INPUT_VEC3, 1, VAR_NORMAL])      # normal input
builder.add_instr(59, [TYPE_PTR_OUTPUT_VEC4, 3, VAR_OUTCOLOR])   # outColor output
builder.add_instr(59, [TYPE_PTR_UNIFORM_STRUCT, 2, VAR_UBO])     # UBO

# Function
builder.add_instr(54, [TYPE_VOID, 0, TYPE_FUNC_VOID, ENTRY_POINT])  # OpFunction
builder.add_instr(248, [LABEL_MAIN])  # OpLabel

# Load inputs
ID = builder.get_id()
builder.add_instr(61, [TYPE_VEC3, ID, VAR_WORLDPOS])  # load worldPos
WORLDPOS = ID

ID = builder.get_id()
builder.add_instr(61, [TYPE_VEC3, ID, VAR_NORMAL])    # load normal
NORMAL_RAW = ID

# Normalize normal
ID = builder.get_id()
builder.add_instr(12, [TYPE_VEC3, ID, EXT_IMPORT_ID, 63, NORMAL_RAW])  # normalize
NORMAL = ID

# Get light position from UBO
ID = builder.get_id()
builder.add_instr(65, [TYPE_PTR_UNIFORM_VEC4, ID, VAR_UBO, CONST_0I])  # access chain lightPos
LIGHT_PTR = ID

ID = builder.get_id()
builder.add_instr(61, [TYPE_VEC4, ID, LIGHT_PTR])  # load lightPos
LIGHT_POS4 = ID

# Extract xyz from light position
ID = builder.get_id()
builder.add_instr(79, [TYPE_VEC3, ID, LIGHT_POS4, LIGHT_POS4, CONST_0I, CONST_0I, CONST_0I])  # vector shuffle x3 times for xyz
LIGHT_POS = ID

# Light direction vector
ID = builder.get_id()
builder.add_instr(131, [TYPE_VEC3, ID, LIGHT_POS, WORLDPOS])  # FSub
LIGHT_DIR = ID

# Normalize light direction
ID = builder.get_id()
builder.add_instr(12, [TYPE_VEC3, ID, EXT_IMPORT_ID, 63, LIGHT_DIR])  # normalize
L = ID

# Get camera position from UBO (member index 4)
ID = builder.get_id()
builder.add_instr(65, [TYPE_PTR_UNIFORM_VEC4, ID, VAR_UBO, CONST_4I])  # access chain cameraPos (member 4)
CAM_PTR = ID

ID = builder.get_id()
builder.add_instr(61, [TYPE_VEC4, ID, CAM_PTR])  # load cameraPos
CAM_POS4 = ID

# Extract xyz from camera position
ID = builder.get_id()
builder.add_instr(79, [TYPE_VEC3, ID, CAM_POS4, CAM_POS4, CONST_0I, CONST_0I, CONST_0I])  # vector shuffle
CAM_POS = ID

# View direction vector
ID = builder.get_id()
builder.add_instr(131, [TYPE_VEC3, ID, CAM_POS, WORLDPOS])  # FSub
VIEW_DIR = ID

# Normalize view direction
ID = builder.get_id()
builder.add_instr(12, [TYPE_VEC3, ID, EXT_IMPORT_ID, 63, VIEW_DIR])  # normalize
V = ID

# Negate L for reflect
ID = builder.get_id()
builder.add_instr(127, [TYPE_VEC3, ID, L])  # FNegate
NEG_L = ID

# Reflect: R = reflect(-L, N)
ID = builder.get_id()
builder.add_instr(12, [TYPE_VEC3, ID, EXT_IMPORT_ID, 64, NEG_L, NORMAL])  # reflect
R = ID

# Diffuse: dot(N, L)
ID = builder.get_id()
builder.add_instr(148, [TYPE_FLOAT, ID, NORMAL, L])  # Dot
NL = ID

# max(dot(N,L), 0)
ID = builder.get_id()
builder.add_instr(12, [TYPE_FLOAT, ID, EXT_IMPORT_ID, 40, NL, CONST_0F])  # FMax
DIFFUSE = ID

# Specular: dot(R, V)
ID = builder.get_id()
builder.add_instr(148, [TYPE_FLOAT, ID, R, V])  # Dot
RV = ID

# max(dot(R,V), 0)
ID = builder.get_id()
builder.add_instr(12, [TYPE_FLOAT, ID, EXT_IMPORT_ID, 40, RV, CONST_0F])  # FMax
RV_MAX = ID

# pow(max(dot(R,V), 0), 32)
ID = builder.get_id()
builder.add_instr(12, [TYPE_FLOAT, ID, EXT_IMPORT_ID, 26, RV_MAX, CONST_32F])  # Pow
SPECULAR = ID

# 0.5 * diffuse
ID = builder.get_id()
builder.add_instr(133, [TYPE_FLOAT, ID, CONST_05F, DIFFUSE])  # FMul
TERM_DIFF = ID

# 0.8 * specular
ID = builder.get_id()
builder.add_instr(133, [TYPE_FLOAT, ID, CONST_08F, SPECULAR])  # FMul
TERM_SPEC = ID

# 0.2 + (0.5 * diffuse)
ID = builder.get_id()
builder.add_instr(129, [TYPE_FLOAT, ID, CONST_02F, TERM_DIFF])  # FAdd
INTENSITY1 = ID

# intensity + (0.8 * specular)
ID = builder.get_id()
builder.add_instr(129, [TYPE_FLOAT, ID, INTENSITY1, TERM_SPEC])  # FAdd
INTENSITY = ID

# Construct output color vec4(intensity, intensity, intensity, 1.0)
ID = builder.get_id()
builder.add_instr(80, [TYPE_VEC4, ID, INTENSITY, INTENSITY, INTENSITY, CONST_1F])  # CompositeConstruct
OUT_COLOR = ID

# Store to output
builder.add_instr(62, [VAR_OUTCOLOR, OUT_COLOR])  # Store

# Return
builder.add_instr(253, [])  # OpReturn
builder.add_instr(56, [])   # OpFunctionEnd

# Output hex
print(builder.to_hex())
