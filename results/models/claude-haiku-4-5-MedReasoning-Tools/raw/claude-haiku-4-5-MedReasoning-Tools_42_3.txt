    %float_1 = OpConstant %float 1

      %glsl = OpExtInstImport "GLSL.std.450"
       %main = OpFunction %void None %FuncType
      %entry = OpLabel
    %p0_ptr = OpAccessChain %ptr_in_v4f32 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4f32 %p0_ptr
    %p1_ptr = OpAccessChain %ptr_in_v4f32 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4f32 %p1_ptr

         %c0 = OpLoad %v3f32 %c0_ptr
    %c1_ptr = OpAccessChain %ptr_in_v3f32 %color %int_1
         %c1 = OpLoad %v3f32 %c1_ptr

    %p1_xy = OpVectorShuffle %v2f32 %p1 %p1 0 1
        %dir = OpFSub %v2f32 %p1_xy %p0_xy

    %dir_y = OpCompositeExtract %float %dir 1

   %len_sq = OpDot %float %dir %dir
     %len = OpExtInst %float %glsl Sqrt %len_sq
   %len_inv = OpFDiv %float %float_1 %len
     %perp = OpVectorTimesScalar %v2f32 %perp_un %len_inv
   %offset = OpVectorTimesScalar %v2f32 %perp %float_003
 %offset_x = OpCompositeExtract %float %offset 0
 %offset_y = OpCompositeExtract %float %offset 1
  %p0_z = OpCompositeExtract %float %p0 2
  %p0_w = OpCompositeExtract %float %p0 3
  %p1_z = OpCompositeExtract %float %p1 2
  %p1_w = OpCompositeExtract %float %p1 3

  %v0py = OpFAdd %float %v0_y %offset_y
  %v0_plus = OpCompositeConstruct %v4f32 %v0px %v0py %p0_z %p0_w
  %v0mx = OpFSub %float %v0_x %offset_x
  %v0my = OpFSub %float %v0_y %offset_y
  %v0_minus = OpCompositeConstruct %v4f32 %v0mx %v0my %p0_z %p0_w
  %v1_x = OpCompositeExtract %float %p1_xy 0
  %v1_y = OpCompositeExtract %float %p1_xy 1
  %v1px = OpFAdd %float %v1_x %offset_x
  %v1py = OpFAdd %float %v1_y %offset_y
  %v1_plus = OpCompositeConstruct %v4f32 %v1px %v1py %p1_z %p1_w
  %v1mx = OpFSub %float %v1_x %offset_x
  %v1my = OpFSub %float %v1_y %offset_y
  %v1_minus = OpCompositeConstruct %v4f32 %v1mx %v1my %p1_z %p1_w
  %out_ptr = OpAccessChain %ptr_out_v4f32 %gl_out %int_0
              OpStore %out_ptr %v0_plus
              OpStore %outColor %c0
              OpEmitVertex
              OpStore %out_ptr %v0_minus
              OpStore %outColor %c0
              OpEmitVertex
              OpStore %out_ptr %v1_plus
              OpStore %outColor %c1
              OpEmitVertex
              OpStore %out_ptr %v1_minus
              OpStore %outColor %c1
