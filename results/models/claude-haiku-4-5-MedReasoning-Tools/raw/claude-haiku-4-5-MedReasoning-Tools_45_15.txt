#!/usr/bin/env python3
"""Generate SPIR-V shader for Gooch shading."""

import struct

class SPIRVAssembler:
    def __init__(self):
        self.words = []
        self.next_id = 1

    def new_id(self):
        """Get next available ID."""
        self.id = self.next_id
        self.next_id += 1
        return self.id

    def float_word(self, f):
        """Convert float to SPIR-V word."""
        return struct.unpack('<I', struct.pack('<f', f))[0]

    def string_words(self, s):
        """Convert string to SPIR-V words."""
        b = (s + '\0').encode('ascii')
        while len(b) % 4:
            b += b'\0'
        return [struct.unpack('<I', b[i:i+4])[0] for i in range(0, len(b), 4)]

    def emit(self, *words):
        """Emit words to output."""
        self.words.extend(words)

    def emit_header(self, bound):
        """Emit SPIR-V header."""
        self.emit(0x07230203)  # Magic
        self.emit(0x00010000)  # Version 1.0
        self.emit(0x00000000)  # Generator
        self.emit(bound)       # Bound
        self.emit(0x00000000)  # Reserved

    def emit_op(self, opcode, *operands):
        """Emit instruction."""
        word_count = 1 + len(operands)
        self.emit((word_count << 16) | opcode)
        self.emit(*operands)

    def emit_op_result(self, opcode, result, *operands):
        """Emit instruction with result ID."""
        word_count = 2 + len(operands)
        self.emit((word_count << 16) | opcode)
        self.emit(result)
        self.emit(*operands)

    def emit_op_type_result(self, opcode, type_id, result, *operands):
        """Emit instruction with type and result."""
        word_count = 3 + len(operands)
        self.emit((word_count << 16) | opcode)
        self.emit(type_id)
        self.emit(result)
        self.emit(*operands)

    def to_hex(self):
        """Convert to hex string."""
        data = b''
        for word in self.words:
            data += struct.pack('<I', word)
        return ''.join(f'{b:02x}' for b in data)

# Build shader
asm = SPIRVAssembler()

# Reserve IDs
void_id = 1
float_id = 2
vec3_id = 3
vec4_id = 4
mat4_id = 5
ubo_struct_id = 6
ptr_uniform_ubo_id = 7
ptr_input_vec3_id = 8
ptr_output_vec4_id = 9
ptr_func_float_id = 10
ptr_uniform_vec4_id = 11
void_func_id = 12

# Constants
const_0_id = 13
const_1_id = 14
const_05_id = 15
const_02_id = 16
const_075_id = 17
const_07_id = 18
const_04_id = 19
cool_color_id = 20
warm_color_id = 21

# Variables
ubo_var_id = 22
normal_input_id = 23
worldpos_input_id = 24
outcolor_output_id = 25
main_func_id = 26

# Header with bound of 70
asm.emit_header(70)

# Decorations and capabilities
asm.emit_op(17, 1)  # OpCapability Shader

# OpExtInstImport
ext_words = asm.string_words("GLSL.std.450")
asm.emit((3 + len(ext_words)) << 16 | 11)
asm.emit(1)
asm.emit(*ext_words)

# OpMemoryModel
asm.emit_op(14, 0, 1)  # Logical, GLSL450

# OpEntryPoint
main_words = asm.string_words("main")
asm.emit((4 + len(main_words)) << 16 | 15)
asm.emit(4)  # Fragment execution model
asm.emit(main_func_id)
asm.emit(*main_words)
asm.emit(normal_input_id, worldpos_input_id)  # Interface variables

# OpExecutionMode
asm.emit_op(16, main_func_id, 7)  # OriginUpperLeft

# Decorations
asm.emit_op(71, normal_input_id, 30, 1)  # Location 1
asm.emit_op(71, worldpos_input_id, 30, 0)  # Location 0
asm.emit_op(71, outcolor_output_id, 30, 0)  # Location 0
asm.emit_op(71, ubo_struct_id, 2)  # Block
asm.emit_op(71, ubo_var_id, 34, 0)  # DescriptorSet 0
asm.emit_op(71, ubo_var_id, 33, 0)  # Binding 0

# Member decorations
asm.emit_op(72, ubo_struct_id, 0, 35, 0)    # mat4 model
asm.emit_op(72, ubo_struct_id, 1, 35, 64)   # mat4 view
asm.emit_op(72, ubo_struct_id, 2, 35, 128)  # mat4 projection
asm.emit_op(72, ubo_struct_id, 3, 35, 192)  # vec4 lightPos
asm.emit_op(72, ubo_struct_id, 4, 35, 208)  # vec4 cameraPos
asm.emit_op(72, ubo_struct_id, 5, 35, 224)  # vec4 params

# Types
asm.emit_op_result(19, void_id)  # OpTypeVoid
asm.emit_op_result(22, float_id, 32)  # OpTypeFloat
asm.emit_op_result(23, vec3_id, float_id, 3)  # OpTypeVector vec3
asm.emit_op_result(23, vec4_id, float_id, 4)  # OpTypeVector vec4
asm.emit_op_result(24, mat4_id, vec4_id, 4)  # OpTypeMatrix

# OpTypeStruct
asm.emit((9 << 16) | 30)
asm.emit(ubo_struct_id)
asm.emit(mat4_id, mat4_id, mat4_id, vec4_id, vec4_id, vec4_id)

# Pointers
asm.emit_op_result(32, ptr_uniform_ubo_id, 0, ubo_struct_id)  # Uniform
asm.emit_op_result(32, ptr_input_vec3_id, 1, vec3_id)  # Input
asm.emit_op_result(32, ptr_output_vec4_id, 3, vec4_id)  # Output
asm.emit_op_result(32, ptr_func_float_id, 7, float_id)  # Function
asm.emit_op_result(32, ptr_uniform_vec4_id, 0, vec4_id)  # Uniform

# Function type
asm.emit_op_result(33, void_func_id, void_id)  # void()

# Constants
asm.emit_op_type_result(43, float_id, const_0_id, asm.float_word(0.0))
asm.emit_op_type_result(43, float_id, const_1_id, asm.float_word(1.0))
asm.emit_op_type_result(43, float_id, const_05_id, asm.float_word(0.5))
asm.emit_op_type_result(43, float_id, const_02_id, asm.float_word(0.2))
asm.emit_op_type_result(43, float_id, const_075_id, asm.float_word(0.75))
asm.emit_op_type_result(43, float_id, const_07_id, asm.float_word(0.7))
asm.emit_op_type_result(43, float_id, const_04_id, asm.float_word(0.4))

# Composite constants
asm.emit((5 << 16) | 44)  # OpCompositeConstruct
asm.emit(vec3_id, cool_color_id, const_02_id, const_02_id, const_075_id)

asm.emit((5 << 16) | 44)  # OpCompositeConstruct
asm.emit(vec3_id, warm_color_id, const_07_id, const_07_id, const_04_id)

# Variables
asm.emit_op_result(59, ubo_var_id, 0, ubo_struct_id)  # Uniform
asm.emit_op_result(59, normal_input_id, 1, vec3_id)  # Input
asm.emit_op_result(59, worldpos_input_id, 1, vec3_id)  # Input
asm.emit_op_result(59, outcolor_output_id, 3, vec4_id)  # Output

# Function
asm.emit((5 << 16) | 54)  # OpFunction
asm.emit(void_id, main_func_id, 0, void_func_id)

# Entry label
asm.emit((2 << 16) | 248)  # OpLabel
asm.emit(27)

# Load inputs
asm.emit_op_type_result(61, vec3_id, 28, normal_input_id)  # Load normal
asm.emit_op_type_result(61, vec3_id, 29, worldpos_input_id)  # Load worldPos

# Access UBO lightPos
const_3_id = 30
asm.emit_op_type_result(43, float_id, const_3_id, 3)  # Const for index
asm.emit((4 << 16) | 65)  # OpAccessChain
asm.emit(ptr_uniform_vec4_id, 31, ubo_var_id, const_3_id)

asm.emit_op_type_result(61, vec4_id, 32, 31)  # Load lightPos

# Extract xyz
asm.emit((6 << 16) | 79)  # OpVectorShuffle
asm.emit(vec3_id, 33, 32, 32, 0, 1, 2)

# Subtract: lightPos.xyz - worldPos
asm.emit_op_type_result(131, vec3_id, 34, 33, 29)  # OpFSub

# Normalize
asm.emit((5 << 16) | 12)  # OpExtInst Normalize
asm.emit(vec3_id, 35, 1, 69, 34)

# Dot product
asm.emit_op_type_result(148, float_id, 36, 28, 35)  # OpDot

# t = dot(N,L) * 0.5 + 0.5
asm.emit_op_type_result(133, float_id, 37, 36, const_05_id)  # Mul 0.5
asm.emit_op_type_result(129, float_id, 38, 37, const_05_id)  # Add 0.5

# Interpolation: cool * (1-t) + warm * t
asm.emit_op_type_result(131, float_id, 39, const_1_id, 38)  # 1 - t
asm.emit_op_type_result(133, vec3_id, 40, cool_color_id, 39)  # cool * (1-t)
asm.emit_op_type_result(133, vec3_id, 41, warm_color_id, 38)  # warm * t
asm.emit_op_type_result(129, vec3_id, 42, 40, 41)  # Add

# Construct final color
asm.emit((4 << 16) | 80)  # OpCompositeConstruct
asm.emit(vec4_id, 43, 42, const_1_id)

# Store result
asm.emit_op(62, outcolor_output_id, 43)  # OpStore

# Return
asm.emit_op(253)  # OpReturn

# Function end
asm.emit_op(56)  # OpFunctionEnd

print(asm.to_hex())
