#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int n, m;
vector<vector<int>> adj;
vector<int> best_set;

void backtrack(int idx, vector<int>& current, set<int>& forbidden) {
    // Pruning: if we can't possibly beat the current best
    if ((int)current.size() + (n - idx) <= (int)best_set.size()) {
        return;
    }

    if (idx == n) {
        if (current.size() > best_set.size()) {
            best_set = current;
        }
        return;
    }

    // Skip this vertex if it's forbidden (adjacent to included vertices)
    if (forbidden.count(idx)) {
        backtrack(idx + 1, current, forbidden);
        return;
    }

    // Try including vertex idx
    current.push_back(idx);
    set<int> old_forbidden = forbidden;

    // Add all neighbors of idx to forbidden set
    for (int neighbor : adj[idx]) {
        forbidden.insert(neighbor);
    }

    backtrack(idx + 1, current, forbidden);

    // Restore state and try excluding idx
    current.pop_back();
    forbidden = old_forbidden;
    backtrack(idx + 1, current, forbidden);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    adj.resize(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    set<int> forbidden;
    vector<int> current;
    backtrack(0, current, forbidden);

    cout << best_set.size() << "\n";
    for (int i = 0; i < (int)best_set.size(); i++) {
        if (i > 0) cout << " ";
        cout << best_set[i];
    }
    if (best_set.size() > 0) cout << "\n";

    return 0;
}
