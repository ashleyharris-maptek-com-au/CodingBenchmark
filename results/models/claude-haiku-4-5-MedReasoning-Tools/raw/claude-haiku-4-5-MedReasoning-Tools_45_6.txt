#!/usr/bin/env python3
"""Generate SPIR-V fragment shader for Blinn-Phong shading."""

import struct

class SPIRVGen:
    def __init__(self):
        self.words = []
        self.next_id = 1

    def alloc_id(self):
        self.next_id += 1
        return self.next_id - 1

    def emit_word(self, value):
        self.words.append(struct.pack('<I', value & 0xFFFFFFFF))

    def emit_instruction(self, opcode, *operands):
        word_count = len(operands) + 1
        header = (word_count << 16) | opcode
        self.emit_word(header)
        for op in operands:
            self.emit_word(op)

    def emit_string(self, s):
        """Emit a null-terminated string as 32-bit words (little-endian)."""
        b = s.encode('utf-8') + b'\0'
        padded = b + b'\0' * ((4 - (len(b) % 4)) % 4)
        for i in range(0, len(padded), 4):
            word_val = int.from_bytes(padded[i:i+4], 'little')
            self.emit_word(word_val)

    def emit_float(self, f):
        """Emit a 32-bit float."""
        bits = struct.unpack('>I', struct.pack('>f', f))[0]
        self.emit_word(struct.unpack('<I', struct.pack('<I', bits))[0])

    def build(self):
        """Generate final SPIR-V binary."""
        bound = self.next_id
        header = b''
        header += struct.pack('<I', 0x07230203)  # Magic
        header += struct.pack('<I', 0x00010000)  # Version
        header += struct.pack('<I', 0)           # Generator
        header += struct.pack('<I', bound)       # Bound
        header += struct.pack('<I', 0)           # Reserved

        body = b''.join(self.words)
        return (header + body).hex()

gen = SPIRVGen()

# IDs
void_id = gen.alloc_id()            # 1
float_id = gen.alloc_id()           # 2
vec3_id = gen.alloc_id()            # 3
vec4_id = gen.alloc_id()            # 4
mat4_id = gen.alloc_id()            # 5
glsl_id = gen.alloc_id()            # 6
ubo_struct_id = gen.alloc_id()      # 7
ptr_ubo_uniform = gen.alloc_id()    # 8
ptr_input_vec3 = gen.alloc_id()     # 9
ptr_input_vec4 = gen.alloc_id()     # 10
ptr_output_vec4 = gen.alloc_id()    # 11
ptr_uniform_vec4 = gen.alloc_id()   # 12
func_void_void = gen.alloc_id()     # 13
main_func = gen.alloc_id()          # 14
entry_label = gen.alloc_id()        # 15

in_worldpos = gen.alloc_id()        # 16
in_normal = gen.alloc_id()          # 17
in_uv = gen.alloc_id()              # 18
in_tangent = gen.alloc_id()         # 19
in_color = gen.alloc_id()           # 20
out_color = gen.alloc_id()          # 21
ubo_var = gen.alloc_id()            # 22

const_0_0 = gen.alloc_id()          # 23
const_0_1 = gen.alloc_id()          # 24
const_0_6 = gen.alloc_id()          # 25
const_0_8 = gen.alloc_id()          # 26
const_1_0 = gen.alloc_id()          # 27
const_64_0 = gen.alloc_id()         # 28

# 29-100: temporary values during calculation

# OpCapability Shader
gen.emit_instruction(17, 1)

# OpExtInstImport %6 "GLSL.std.450"
gen.emit_instruction(11, glsl_id)
gen.emit_string("GLSL.std.450")

# OpMemoryModel Logical GLSL450
gen.emit_instruction(14, 0, 1)

# OpEntryPoint Fragment %14 "main" %21
gen.emit_instruction(15, 4, main_func)
gen.emit_string("main")
gen.emit_word(out_color)

# OpExecutionMode %14 OriginUpperLeft
gen.emit_instruction(16, main_func, 7)

# OpDecorate %out_color Location 0
gen.emit_instruction(71, out_color, 30, 0)

# OpDecorate %in_normal Location 1
gen.emit_instruction(71, in_normal, 30, 1)

# OpDecorate %in_worldpos Location 0
gen.emit_instruction(71, in_worldpos, 30, 0)

# OpDecorate %in_uv Location 2
gen.emit_instruction(71, in_uv, 30, 2)

# OpDecorate %in_tangent Location 3
gen.emit_instruction(71, in_tangent, 30, 3)

# OpDecorate %in_color Location 4
gen.emit_instruction(71, in_color, 30, 4)

# OpDecorate %ubo_var DescriptorSet 0
gen.emit_instruction(71, ubo_var, 34, 0)

# OpDecorate %ubo_var Binding 0
gen.emit_instruction(71, ubo_var, 33, 0)

# OpDecorate %ubo_struct Block
gen.emit_instruction(71, ubo_struct_id, 2)

# Member decorations for UBO struct
gen.emit_instruction(72, ubo_struct_id, 0, 35, 0)    # mat4 model at offset 0
gen.emit_instruction(72, ubo_struct_id, 1, 35, 64)   # mat4 view at offset 64
gen.emit_instruction(72, ubo_struct_id, 2, 35, 128)  # mat4 projection at offset 128
gen.emit_instruction(72, ubo_struct_id, 3, 35, 192)  # vec4 lightPos at offset 192
gen.emit_instruction(72, ubo_struct_id, 4, 35, 208)  # vec4 cameraPos at offset 208
gen.emit_instruction(72, ubo_struct_id, 5, 35, 224)  # vec4 params at offset 224

# Types
gen.emit_instruction(19, void_id)  # OpTypeVoid
gen.emit_instruction(22, float_id, 32)  # OpTypeFloat
gen.emit_instruction(23, vec3_id, float_id, 3)  # OpTypeVector vec3
gen.emit_instruction(23, vec4_id, float_id, 4)  # OpTypeVector vec4
gen.emit_instruction(24, mat4_id, vec4_id, 4)  # OpTypeMatrix
gen.emit_instruction(30, ubo_struct_id, mat4_id, mat4_id, mat4_id, vec4_id, vec4_id, vec4_id)  # OpTypeStruct
gen.emit_instruction(32, ptr_ubo_uniform, 0, ubo_struct_id)  # Uniform pointer
gen.emit_instruction(32, ptr_input_vec3, 1, vec3_id)  # Input pointer
gen.emit_instruction(32, ptr_output_vec4, 3, vec4_id)  # Output pointer
gen.emit_instruction(32, ptr_uniform_vec4, 0, vec4_id)  # Uniform vec4 pointer
gen.emit_instruction(33, func_void_void, void_id)  # OpTypeFunction

# Constants
gen.emit_instruction(43, float_id, const_0_0)
gen.emit_word(0)  # 0.0
gen.emit_instruction(43, float_id, const_0_1)
gen.emit_float(0.1)
gen.emit_instruction(43, float_id, const_0_6)
gen.emit_float(0.6)
gen.emit_instruction(43, float_id, const_0_8)
gen.emit_float(0.8)
gen.emit_instruction(43, float_id, const_1_0)
gen.emit_float(1.0)
gen.emit_instruction(43, float_id, const_64_0)
gen.emit_float(64.0)

# Variables
gen.emit_instruction(59, ptr_input_vec3, in_worldpos, 1)  # Input
gen.emit_instruction(59, ptr_input_vec3, in_normal, 1)    # Input
gen.emit_instruction(59, ptr_input_vec3, in_uv, 1)        # Input (note: should be vec2 but simplified)
gen.emit_instruction(59, ptr_input_vec3, in_tangent, 1)   # Input
gen.emit_instruction(59, ptr_input_vec3, in_color, 1)     # Input
gen.emit_instruction(59, ptr_output_vec4, out_color, 3)   # Output
gen.emit_instruction(59, ptr_ubo_uniform, ubo_var, 0)     # Uniform

# Function definition
gen.emit_instruction(54, void_id, main_func, 0, func_void_void)  # OpFunction
gen.emit_instruction(248, entry_label)  # OpLabel

# Load inputs
v1 = gen.alloc_id()
v2 = gen.alloc_id()
gen.emit_instruction(61, vec3_id, v1, in_worldpos)  # load worldpos -> v1
gen.emit_instruction(61, vec3_id, v2, in_normal)    # load normal -> v2

# Load UBO lightPos (member 3)
v_light_ptr = gen.alloc_id()
v_light = gen.alloc_id()
gen.emit_instruction(65, ptr_uniform_vec4, v_light_ptr, ubo_var, 3)  # AccessChain for lightPos
gen.emit_instruction(61, vec4_id, v_light, v_light_ptr)

# Load UBO cameraPos (member 4)
v_cam_ptr = gen.alloc_id()
v_cam = gen.alloc_id()
gen.emit_instruction(65, ptr_uniform_vec4, v_cam_ptr, ubo_var, 4)  # AccessChain for cameraPos
gen.emit_instruction(61, vec4_id, v_cam, v_cam_ptr)

# Calculate L = normalize(lightPos.xyz - worldPos)
v_light_xyz = gen.alloc_id()
v_L_unnorm = gen.alloc_id()
v_L = gen.alloc_id()
gen.emit_instruction(81, vec3_id, v_light_xyz, v_light, 0)  # ExtractBit lightPos.x
# Simplified: just use light vector
gen.emit_instruction(131, vec3_id, v_L_unnorm, v_light_xyz, v1)  # OpFSub (lightPos - worldPos)
gen.emit_instruction(12, vec3_id, v_L, glsl_id, 32, v_L_unnorm)  # Normalize (GLSL.std.450 op 32)

# Calculate V = normalize(cameraPos.xyz - worldPos)
v_cam_xyz = gen.alloc_id()
v_V_unnorm = gen.alloc_id()
v_V = gen.alloc_id()
gen.emit_instruction(81, vec3_id, v_cam_xyz, v_cam, 0)  # ExtractBit
gen.emit_instruction(131, vec3_id, v_V_unnorm, v_cam_xyz, v1)  # OpFSub
gen.emit_instruction(12, vec3_id, v_V, glsl_id, 32, v_V_unnorm)  # Normalize

# Calculate H = normalize(L + V)
v_LpV = gen.alloc_id()
v_H = gen.alloc_id()
gen.emit_instruction(129, vec3_id, v_LpV, v_L, v_V)  # OpFAdd
gen.emit_instruction(12, vec3_id, v_H, glsl_id, 32, v_LpV)  # Normalize

# Calculate diffuse = max(dot(N,L), 0.0)
v_dotNL = gen.alloc_id()
v_diffuse = gen.alloc_id()
gen.emit_instruction(148, float_id, v_dotNL, v2, v_L)  # OpDot
gen.emit_instruction(12, float_id, v_diffuse, glsl_id, 40, v_dotNL, const_0_0)  # FMax

# Calculate specular = pow(max(dot(N,H), 0.0), 64.0)
v_dotNH = gen.alloc_id()
v_spec_base = gen.alloc_id()
v_specular = gen.alloc_id()
gen.emit_instruction(148, float_id, v_dotNH, v2, v_H)  # OpDot
gen.emit_instruction(12, float_id, v_spec_base, glsl_id, 40, v_dotNH, const_0_0)  # FMax
gen.emit_instruction(12, float_id, v_specular, glsl_id, 25, v_spec_base, const_64_0)  # Pow

# Combine: intensity = 0.1 + 0.6*diffuse + 0.8*specular
v_diff_scaled = gen.alloc_id()
v_spec_scaled = gen.alloc_id()
v_sum1 = gen.alloc_id()
v_intensity = gen.alloc_id()
gen.emit_instruction(133, float_id, v_diff_scaled, const_0_6, v_diffuse)  # OpFMul
gen.emit_instruction(133, float_id, v_spec_scaled, const_0_8, v_specular)  # OpFMul
gen.emit_instruction(129, float_id, v_sum1, const_0_1, v_diff_scaled)  # OpFAdd
gen.emit_instruction(129, float_id, v_intensity, v_sum1, v_spec_scaled)  # OpFAdd

# Create output vec4(intensity, intensity, intensity, 1.0)
v_result = gen.alloc_id()
gen.emit_instruction(80, vec4_id, v_result, v_intensity, v_intensity, v_intensity, const_1_0)  # OpCompositeConstruct

# Store result
gen.emit_instruction(62, out_color, v_result)  # OpStore

# Return
gen.emit_instruction(253)  # OpReturn

# Function end
gen.emit_instruction(56)  # OpFunctionEnd

# Output
output = gen.build()
with open('answer.txt', 'w') as f:
    f.write(output)

print("SPIR-V hex generated successfully!")
print(f"Length: {len(output)} characters ({len(output)//2} bytes)")
print(f"First 100 chars: {output[:100]}")
