use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse first line: num_vertices num_edges
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let num_vertices = parts[0];
    let num_edges = parts[1];

    // Build adjacency list
    let mut adj: Vec<HashSet<usize>> = vec![HashSet::new(); num_vertices];

    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let u = parts[0];
        let v = parts[1];
        adj[u].insert(v);
        adj[v].insert(u);
    }

    // Greedy dominating set algorithm
    let mut covered: HashSet<usize> = HashSet::new();
    let mut dominating_set: Vec<usize> = Vec::new();

    while covered.len() < num_vertices {
        let mut best_vertex = 0;
        let mut best_score = 0;

        // Find the vertex that covers the most uncovered vertices
        for v in 0..num_vertices {
            if dominating_set.contains(&v) {
                continue;
            }

            let mut score = 0;

            // Count uncovered neighbors
            for &neighbor in &adj[v] {
                if !covered.contains(&neighbor) {
                    score += 1;
                }
            }

            // Bonus if vertex itself is uncovered
            if !covered.contains(&v) {
                score += 1;
            }

            if score > best_score {
                best_score = score;
                best_vertex = v;
            }
        }

        // Add the best vertex to dominating set and mark it and neighbors as covered
        dominating_set.push(best_vertex);
        covered.insert(best_vertex);
        for &neighbor in &adj[best_vertex] {
            covered.insert(neighbor);
        }
    }

    // Output
    dominating_set.sort();

    println!("{}", dominating_set.len());
    println!("{}", dominating_set.iter()
        .map(|v| v.to_string())
        .collect::<Vec<_>>()
        .join(" "));
}
