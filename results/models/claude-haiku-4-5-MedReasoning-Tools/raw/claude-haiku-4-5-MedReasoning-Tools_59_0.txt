const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { probability, drones, maxStepsPerDrone } = data;
const rows = probability.length;
const cols = rows > 0 ? probability[0].length : 0;

const covered = new Set();

const getKey = (x, y) => `${x},${y}`;

const isValid = (x, y) => {
  return x >= 0 && x < cols && y >= 0 && y < rows && probability[y][x] !== -1;
};

const scoreCell = (x, y) => {
  const key = getKey(x, y);
  const value = probability[y][x];

  // Already covered cells score 0
  if (covered.has(key)) {
    return 0;
  }

  // Empty or negative cells score 0
  if (value <= 0) {
    return 0;
  }

  // Base score is the cell's value
  let score = value;

  // Bonus for being adjacent to other uncovered valuable cells
  // This encourages exploration of high-value clusters
  const directions = [
    { dx: 1, dy: 0 },
    { dx: -1, dy: 0 },
    { dx: 0, dy: 1 },
    { dx: 0, dy: -1 }
  ];

  for (const { dx, dy } of directions) {
    const nx = x + dx, ny = y + dy;
    if (isValid(nx, ny)) {
      const nKey = getKey(nx, ny);
      if (!covered.has(nKey) && probability[ny][nx] > 0) {
        score += probability[ny][nx] * 0.1;
      }
    }
  }

  return score;
};

const paths = drones.map(drone => {
  const path = [[drone.x, drone.y]];
  let x = drone.x;
  let y = drone.y;
  covered.add(getKey(x, y));

  for (let step = 0; step < maxStepsPerDrone; step++) {
    // Consider all possible moves: 4 directions + stay in place
    const options = [
      { x: x + 1, y },
      { x: x - 1, y },
      { x, y: y + 1 },
      { x, y: y - 1 },
      { x, y }  // stay in place
    ];

    let bestOption = options[4]; // default to staying in place
    let bestScore = 0;

    // Find best move (highest score)
    for (let i = 0; i < 4; i++) {
      if (isValid(options[i].x, options[i].y)) {
        const score = scoreCell(options[i].x, options[i].y);
        if (score > bestScore) {
          bestScore = score;
          bestOption = options[i];
        }
      }
    }

    x = bestOption.x;
    y = bestOption.y;
    path.push([x, y]);
    covered.add(getKey(x, y));
  }

  return path;
});
