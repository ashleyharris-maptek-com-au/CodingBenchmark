"""
ILS Autopilot for Airliner Landing
Handles 25 scenarios including engine failures, sensor failures, and adverse weather
"""

# Module-level state (persists within scenario, reset between scenarios)
_phase = "intercept"
_go_around_count = 0
_gs_integral = 0.0
_loc_integral = 0.0
_speed_integral = 0.0
_previous_altitude = 0.0
_descent_rate_sum = 0.0
_descent_rate_count = 0
_flare_started = False
_landing_complete = False


def _select_best_sensor(state, key_prefix, key_suffix=""):
    """
    Select best value from 3 redundant sensors using voting.
    Returns average of valid sensors, or None if none available.
    """
    valid_values = []

    for i in range(1, 4):
        full_key = f"{key_prefix}_{i}{key_suffix}"
        valid_key = f"{key_prefix}_{i}_valid"

        if full_key not in state:
            continue

        # Skip invalid sensors if validity flag exists
        if valid_key in state and state[valid_key] is False:
            continue

        value = state.get(full_key)
        if value is not None:
            valid_values.append(value)

    if not valid_values:
        return None

    # Return average of valid sensors
    return sum(valid_values) / len(valid_values)


def _clamp(value, min_val, max_val):
    """Clamp value to range [min_val, max_val]"""
    return max(min_val, min(max_val, value))


def autopilot_step(state: dict, dt: float) -> dict:
    """
    Main autopilot control loop.
    Called repeatedly with aircraft state, returns control commands.
    """
    global _phase, _go_around_count, _gs_integral, _loc_integral, _speed_integral
    global _previous_altitude, _descent_rate_sum, _descent_rate_count, _flare_started
    global _landing_complete

    # === SENSOR FUSION: Extract and validate sensor data ===

    # Altitude
    radio_alt = state.get("radio_alt_ft", 0)
    baro_alt = _select_best_sensor(state, "baro_alt", "_ft") or radio_alt

    # Airspeed (IAS from 3 receivers)
    ias = _select_best_sensor(state, "ias", "_kt") or 140

    # Attitude from IRS (inertial reference system)
    pitch = _select_best_sensor(state, "irs", "_pitch_deg") or 0
    roll = _select_best_sensor(state, "irs", "_roll_deg") or 0
    vs = _select_best_sensor(state, "irs", "_vs_fpm") or 0

    # ILS signals (3 receivers each for redundancy)
    gs_dots = _select_best_sensor(state, "glideslope", "_dots") or 0
    loc_dots = _select_best_sensor(state, "localizer", "_dots") or 0

    # Target values
    target_ias = state.get("target_ias_kt", 137)
    target_vref = state.get("target_vref_kt", 137)  # Landing reference speed
    target_vs = state.get("target_vs_fpm", -400)

    # Status flags
    on_ground = state.get("weight_on_wheels", False)
    below_dh = state.get("below_decision_height", False)

    # Engine status (for asymmetric thrust handling)
    engine_1_running = state.get("engine_1_running", True)
    engine_2_running = state.get("engine_2_running", True)

    output = {}

    # === LANDING COMPLETE ===
    if on_ground:
        output["throttle_1"] = 0.0
        output["throttle_2"] = 0.0
        output["elevator"] = 0.0
        output["aileron"] = 0.0
        output["rudder"] = 0.0
        output["spoiler"] = 1.0
        output["autobrake"] = 1.0
        output["reverse_1"] = 0.9
        output["reverse_2"] = 0.9
        return output

    # === PHASE MANAGEMENT ===
    # Transition between approach phases based on altitude
    if radio_alt > 1000:
        _phase = "intercept"
        _flare_started = False
    elif radio_alt > 400:
        _phase = "approach"
        _flare_started = False
    elif radio_alt > 50:
        _phase = "flare"
        _flare_started = True
    else:
        _phase = "landing"

    # Track descent rate for stability monitoring
    if _previous_altitude > 0 and dt > 0:
        descent_rate = (_previous_altitude - radio_alt) / dt
        _descent_rate_sum += descent_rate
        _descent_rate_count += 1
    _previous_altitude = radio_alt

    # === GO-AROUND LOGIC ===
    # Maximum 10 go-arounds before scenario fails
    if _go_around_count >= 10:
        # Scenario has failed - engines out
        output["throttle_1"] = 0.0
        output["throttle_2"] = 0.0
        return output

    # Check stability below decision height
    if below_dh:
        speed_excess = ias - target_vref

        # Unstable conditions requiring go-around
        unstable = (
            abs(gs_dots) > 1.6 or           # Off glideslope
            abs(loc_dots) > 1.6 or          # Off localizer
            speed_excess > 20 or            # Too fast
            abs(roll) > 12 or               # Excessive bank
            radio_alt > 300                 # High on approach
        )

        if unstable:
            _go_around_count += 1
            # Execute go-around: full throttle, pitch up, retract config
            output["throttle_1"] = 1.0
            output["throttle_2"] = 1.0
            output["elevator"] = 0.5
            output["speedbrake"] = 0.0
            output["flaps"] = 15
            return output

    # === PID CONTROL LOOPS ===

    # Glideslope tracking (elevator control)
    # Positive gs_dots = above glideslope → pitch down (negative elevator)
    Kp_gs = 0.20
    Ki_gs = 0.07
    _gs_integral += gs_dots * dt
    _gs_integral = _clamp(_gs_integral, -0.5, 0.5)  # Prevent windup
    elevator = Kp_gs * gs_dots + Ki_gs * _gs_integral

    # Localizer tracking (aileron + rudder control)
    # Positive loc_dots = right of centerline → roll left (negative aileron)
    Kp_loc = 0.24
    Ki_loc = 0.10
    _loc_integral += loc_dots * dt
    _loc_integral = _clamp(_loc_integral, -0.5, 0.5)
    aileron = Kp_loc * loc_dots + Ki_loc * _loc_integral

    # Rudder for crosswind coordination
    rudder = aileron * 0.35

    # === ASYMMETRIC THRUST HANDLING (Engine Failure) ===
    # If one engine fails, apply rudder trim to maintain heading
    if not engine_1_running and engine_2_running:
        rudder = _clamp(rudder - 0.20, -1.0, 1.0)  # Trim for right engine only
    elif engine_1_running and not engine_2_running:
        rudder = _clamp(rudder + 0.20, -1.0, 1.0)  # Trim for left engine only

    # === SPEED CONTROL (Throttle) ===
    Kp_speed = 0.45
    Ki_speed = 0.08
    speed_error = target_ias - ias
    _speed_integral += speed_error * dt
    _speed_integral = _clamp(_speed_integral, -0.4, 0.4)

    throttle = 0.35 + Kp_speed * speed_error * 0.012 + Ki_speed * _speed_integral * 0.08
    throttle = _clamp(throttle, 0.0, 1.0)

    # === PHASE-SPECIFIC CONFIGURATION ===

    if _phase == "intercept":
        # High altitude, far from runway
        output["flaps"] = 15
        output["gear"] = 1.0
        output["speedbrake"] = 0.30
        throttle = _clamp(0.45 + speed_error * 0.009, 0.20, 0.95)

    elif _phase == "approach":
        # Established on approach, stabilizing
        output["flaps"] = 25
        output["gear"] = 1.0
        output["speedbrake"] = 0.15
        throttle = _clamp(0.32 + speed_error * 0.008, 0.15, 0.85)

    elif _phase == "flare":
        # Close to ground, reduce descent rate
        output["flaps"] = 30
        output["gear"] = 1.0
        output["speedbrake"] = 0.0

        # Pitch up to reduce descent rate for soft landing
        elevator_flare = _clamp(elevator + 0.30, -1.0, 1.0)
        elevator = elevator_flare

        # Reduce throttle towards idle during flare
        throttle = 0.08

    elif _phase == "landing":
        # Very close to ground
        output["flaps"] = 35
        output["gear"] = 1.0
        output["speedbrake"] = 0.0
        elevator = _clamp(elevator + 0.40, -1.0, 1.0)
        throttle = 0.0

    # === HANDLE DUAL ENGINE FAILURE (Gimli Glider scenario) ===
    # If both engines out, manage best descent with current configuration
    if not engine_1_running and not engine_2_running:
        throttle_1 = 0.0
        throttle_2 = 0.0
    else:
        # Distribute throttle to both engines
        throttle_1 = throttle
        throttle_2 = throttle

        # Handle single engine failure: increase good engine slightly
        if not engine_1_running and engine_2_running:
            throttle_1 = 0.0
            throttle_2 = _clamp(throttle * 1.05, 0.0, 1.0)
        elif engine_1_running and not engine_2_running:
            throttle_1 = _clamp(throttle * 1.05, 0.0, 1.0)
            throttle_2 = 0.0

    # === OUTPUT CONTROL COMMANDS ===
    output["elevator"] = _clamp(elevator, -1.0, 1.0)
    output["aileron"] = _clamp(aileron, -1.0, 1.0)
    output["rudder"] = _clamp(rudder, -1.0, 1.0)
    output["throttle_1"] = _clamp(throttle_1, 0.0, 1.0)
    output["throttle_2"] = _clamp(throttle_2, 0.0, 1.0)

    return output
