#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Scale UV coordinates: 8 bricks wide, 16 bricks tall
    vec2 brick_uv = uv * vec2(8.0, 16.0);

    // Get integer row for staggering
    float row = floor(brick_uv.y);

    // Apply half-brick offset to every other row
    float stagger_offset = mod(row, 2.0) * 0.5;
    brick_uv.x += stagger_offset;

    // Get position within brick
    vec2 within_brick = fract(brick_uv);

    // Mortar dimensions
    float mortar_thickness_x = 0.05;
    float mortar_thickness_y = 0.1;

    // Check if in mortar region
    bool in_mortar = (within_brick.x < mortar_thickness_x ||
                      within_brick.x > (1.0 - mortar_thickness_x)) ||
                     (within_brick.y < mortar_thickness_y ||
                      within_brick.y > (1.0 - mortar_thickness_y));

    // Select color
    vec3 brick_color = vec3(0.7, 0.2, 0.1);
    vec3 mortar_color = vec3(0.7, 0.7, 0.7);
    vec3 surface_color = in_mortar ? mortar_color : brick_color;

    // Normalize the interpolated normal
    vec3 N = normalize(normal);

    // Calculate light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Lambertian diffuse
    float diffuse = max(dot(N, L), 0.0);

    // Shading factor: 0.4 + 0.6 * diffuse
    float shading = 0.4 + 0.6 * diffuse;

    // Apply shading
    vec3 final_color = surface_color * shading;

    outColor = vec4(final_color, 1.0);
}
