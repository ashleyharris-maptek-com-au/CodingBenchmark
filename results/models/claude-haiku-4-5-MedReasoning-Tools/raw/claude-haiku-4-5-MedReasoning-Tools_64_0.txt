declare const require: any;
const fs = require('fs');

interface Drone {
  id: number;
  x: number;
  y: number;
}

interface Input {
  probability: number[][];
  drones: Drone[];
  maxStepsPerDrone: number;
}

interface Output {
  paths: Array<Array<[number, number]>>;
}

function solve(data: Input): Output {
  const { probability, drones, maxStepsPerDrone } = data;
  const rows = probability.length;
  const cols = probability[0].length;

  const covered = new Set<string>();
  const paths: Array<Array<[number, number]>> = [];

  for (const drone of drones) {
    const path: Array<[number, number]> = [[drone.x, drone.y]];
    covered.add(`${drone.x},${drone.y}`);

    let x = drone.x;
    let y = drone.y;

    for (let step = 0; step < maxStepsPerDrone; step++) {
      let bestCell: [number, number] | null = null;
      let bestValue = -Infinity;

      // Check current cell and all 4 adjacent cells
      const candidates = [
        [x, y],
        [x - 1, y],
        [x + 1, y],
        [x, y - 1],
        [x, y + 1]
      ];

      for (const [nx, ny] of candidates) {
        // Boundary check
        if (nx < 0 || nx >= rows || ny < 0 || ny >= cols) continue;

        // Obstacle check
        if (probability[nx][ny] === -1) continue;

        // Value is probability if uncovered, 0 if already covered
        const key = `${nx},${ny}`;
        const value = covered.has(key) ? 0 : probability[nx][ny];

        if (value > bestValue) {
          bestValue = value;
          bestCell = [nx, ny];
        }
      }

      // Move to best cell if it has positive uncovered value
      if (bestCell && bestValue > 0) {
        x = bestCell[0];
        y = bestCell[1];
        path.push([x, y]);
        covered.add(`${x},${y}`);
      } else {
        // No profitable move available, stop this drone
        break;
      }
    }

    paths.push(path);
  }

  return { paths };
}

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);
const output = solve(data);
