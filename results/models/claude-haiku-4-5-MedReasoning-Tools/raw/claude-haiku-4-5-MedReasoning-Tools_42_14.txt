OpCapability Geometry
OpCapability Shader

OpExtInstImport "GLSL.std.450"

OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor

OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %outColor Location 0

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Constants (must come before types that reference them)
%uint_0 = OpConstant %uint 0
%uint_6 = OpConstant %uint 6

%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0p2 = OpConstant %float 0.2
%float_0p8 = OpConstant %float 0.8
%float_3_inv = OpConstant %float 0.33333333

; gl_PerVertex types (before array types)
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Array types
%arr_gl_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_6 = OpTypeArray %v3float %uint_6

; Pointer types
%ptr_in_arr_pv = OpTypePointer Input %arr_gl_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Function type
%void_func_type = OpTypeFunction %void

; Constant composite
%light_dir_vec = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%glsl_ext = OpExtInstImport "GLSL.std.450"

; Main function
%main = OpFunction %void None %void_func_type
%entry_label = OpLabel

; Load gl_in[0].position
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2

; Load gl_in[2].position
%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Load gl_in[4].position
%ptr_p4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %ptr_p4
%p4_xyz = OpVectorShuffle %v3float %p4 %p4 0 1 2

; Compute edge vectors: p2-p0, p4-p0
%edge1 = OpFSub %v3float %p2_xyz %p0_xyz
%edge2 = OpFSub %v3float %p4_xyz %p0_xyz

; Manually compute cross product: (a×b) = (a.y*b.z-a.z*b.y, a.z*b.x-a.x*b.z, a.x*b.y-a.y*b.x)
%e1_x = OpCompositeExtract %float %edge1 0
%e1_y = OpCompositeExtract %float %edge1 1
%e1_z = OpCompositeExtract %float %edge1 2

%e2_x = OpCompositeExtract %float %edge2 0
%e2_y = OpCompositeExtract %float %edge2 1
%e2_z = OpCompositeExtract %float %edge2 2

; Cross X: a.y*b.z - a.z*b.y
%cx_p1 = OpFMul %float %e1_y %e2_z
%cx_p2 = OpFMul %float %e1_z %e2_y
%cross_x = OpFSub %float %cx_p1 %cx_p2

; Cross Y: a.z*b.x - a.x*b.z
%cy_p1 = OpFMul %float %e1_z %e2_x
%cy_p2 = OpFMul %float %e1_x %e2_z
%cross_y = OpFSub %float %cy_p1 %cy_p2

; Cross Z: a.x*b.y - a.y*b.x
%cz_p1 = OpFMul %float %e1_x %e2_y
%cz_p2 = OpFMul %float %e1_y %e2_x
%cross_z = OpFSub %float %cz_p1 %cz_p2

%cross_vec = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z

; Normalize the face normal
%normal_vec = OpExtInst %v3float %glsl_ext 69 %cross_vec

; Normalize light direction
%light_normalized = OpExtInst %v3float %glsl_ext 69 %light_dir_vec

; Compute diffuse: dot(N, L)
%dot_result = OpDot %float %normal_vec %light_normalized

; max(dot, 0.0)
%diffuse = OpExtInst %float %glsl_ext 40 %dot_result %float_0

; Load color[0], color[2], color[4]
%ptr_c0 = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %ptr_c0

%ptr_c2 = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %ptr_c2

%ptr_c4 = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %ptr_c4

; Average colors: (c0 + c2 + c4) / 3
%sum1 = OpFAdd %v3float %c0 %c2
%sum2 = OpFAdd %v3float %sum1 %c4
%avg_color = OpVectorTimesScalar %v3float %sum2 %float_3_inv

; Apply lighting: base * (0.2 + 0.8 * diffuse)
%diffuse_scaled = OpFMul %float %float_0p8 %diffuse
%light_factor = OpFAdd %float %float_0p2 %diffuse_scaled
%final_color = OpVectorTimesScalar %v3float %avg_color %light_factor

; Emit vertex 0 with gl_in[0].position
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos %p0
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 1 with gl_in[2].position
OpStore %ptr_out_pos %p2
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 2 with gl_in[4].position
OpStore %ptr_out_pos %p4
OpStore %outColor %final_color
OpEmitVertex

; End the primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
