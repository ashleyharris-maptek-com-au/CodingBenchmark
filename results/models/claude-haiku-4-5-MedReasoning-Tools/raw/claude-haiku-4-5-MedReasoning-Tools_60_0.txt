const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { machineCount, tasks } = data;

// Track scheduling state
const schedule = [];
const taskFinishTime = new Map();
const machineEndTime = new Array(machineCount).fill(0);
const scheduled = new Set();

// Get earliest possible start time for a task (respecting dependencies)
function getEarliestStart(task) {
  if (!task.deps || task.deps.length === 0) {
    return 0;
  }
  return Math.max(...task.deps.map(depId => taskFinishTime.get(depId) || 0));
}

// Greedy list scheduling
while (scheduled.size < tasks.length) {
  let bestTask = null;
  let bestPriority = -Infinity;

  // Find next unscheduled task with all dependencies satisfied
  for (const task of tasks) {
    if (scheduled.has(task.id)) {
      continue;
    }

    // Check if all dependencies are scheduled
    if (task.deps && !task.deps.every(depId => scheduled.has(depId))) {
      continue;
    }

    // Prioritize by deadline first, then by weight
    // Lower deadline = higher priority (more negative deadline)
    // Higher weight = higher priority (tiebreaker)
    const deadline = task.deadline ?? Infinity;
    const weight = task.weight ?? 1;
    const priority = -deadline * 10000 + weight;

    if (priority > bestPriority) {
      bestPriority = priority;
      bestTask = task;
    }
  }

  if (!bestTask) {
    break;
  }

  // Find machine that will be free earliest
  let bestMachine = 0;
  let earliestTime = machineEndTime[0];
  for (let m = 1; m < machineCount; m++) {
    if (machineEndTime[m] < earliestTime) {
      earliestTime = machineEndTime[m];
      bestMachine = m;
    }
  }

  // Calculate start time (must respect dependencies and machine availability)
  const depMinTime = getEarliestStart(bestTask);
  const startTime = Math.max(depMinTime, machineEndTime[bestMachine]);
  const finishTime = startTime + bestTask.duration;

  // Record schedule
  schedule.push({
    id: bestTask.id,
    start: startTime,
    machine: bestMachine
  });

  // Update state
  machineEndTime[bestMachine] = finishTime;
  taskFinishTime.set(bestTask.id, finishTime);
  scheduled.add(bestTask.id);
}

console.log(JSON.stringify({ schedule }));
