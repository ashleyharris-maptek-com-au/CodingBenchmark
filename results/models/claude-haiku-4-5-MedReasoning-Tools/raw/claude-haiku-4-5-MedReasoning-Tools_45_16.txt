#!/usr/bin/env python3
import struct

def float_bits(f):
    """Convert float to 32-bit unsigned int representation"""
    return struct.unpack('<I', struct.pack('<f', f))[0]

class SPIRVBuilder:
    def __init__(self):
        self.words = []
        self.next_id = 1

    def new_id(self):
        id = self.next_id
        self.next_id += 1
        return id

    def instr(self, opcode, *operands):
        """Add instruction and return word list"""
        word_count = 1 + len(operands)
        return [(word_count << 16) | opcode] + list(operands)

    def add_instr(self, opcode, *operands):
        """Add instruction to module"""
        self.words.extend(self.instr(opcode, *operands))

    def add_instr_with_result(self, opcode, *operands):
        """Add instruction with result ID"""
        result_id = self.new_id()
        self.add_instr(opcode, result_id, *operands)
        return result_id

    def string_words(self, s):
        """Encode string as SPIR-V words"""
        b = s.encode('utf-8') + b'\x00'
        while len(b) % 4 != 0:
            b += b'\x00'
        return [struct.unpack('<I', b[i:i+4])[0] for i in range(0, len(b), 4)]

    def emit_hex(self):
        """Generate hex string of SPIR-V binary"""
        header = [
            0x07230203,      # Magic
            0x00010000,      # Version 1.0
            0x00000000,      # Generator
            self.next_id,    # Bound
            0x00000000       # Reserved
        ]

        all_words = header + self.words

        hex_str = ''
        for word in all_words:
            hex_str += struct.pack('<I', word & 0xFFFFFFFF).hex()

        return hex_str

# Build SPIR-V
b = SPIRVBuilder()

# Capability
b.add_instr(17, 1)  # OpCapability Shader

# Import GLSL
glsl_string = b.string_words("GLSL.std.450")
glsl_id = b.new_id()
b.words.extend(b.instr(11, glsl_id) + glsl_string)

# Memory model
b.add_instr(14, 0, 1)  # OpMemoryModel Logical GLSL450

# Type definitions
void_t = b.add_instr_with_result(19)  # OpTypeVoid
float_t = b.add_instr_with_result(22, 32)  # OpTypeFloat 32
vec2_t = b.add_instr_with_result(23, float_t, 2)  # OpTypeVector float 2
vec4_t = b.add_instr_with_result(23, float_t, 4)  # OpTypeVector float 4
int_t = b.add_instr_with_result(21, 32, 1)  # OpTypeInt 32 1
bool_t = b.add_instr_with_result(20)  # OpTypeBool

# Pointer types
ptr_input_vec2 = b.add_instr_with_result(32, 1, vec2_t)  # Input vec2*
ptr_output_vec4 = b.add_instr_with_result(32, 3, vec4_t)  # Output vec4*
ptr_func_float = b.add_instr_with_result(32, 7, float_t)  # Function float*

# Function type: void()
func_t = b.add_instr_with_result(33, void_t)  # OpTypeFunction void

# Variables
uv_var = b.add_instr_with_result(59, ptr_input_vec2, 1)  # Input variable
out_var = b.add_instr_with_result(59, ptr_output_vec4, 3)  # Output variable

# Decorations
b.add_instr(71, uv_var, 30, 2)  # Location 2
b.add_instr(71, out_var, 30, 0)  # Location 0

# Entry point
main_id = b.new_id()
main_str = b.string_words("main")
b.words.extend(b.instr(15, 4, main_id) + main_str + [uv_var, out_var])

# Execution mode
b.add_instr(16, main_id, 7)  # OriginUpperLeft

# Function
b.add_instr(54, void_t, main_id, 0, func_t)  # OpFunction

# Entry block
label = b.add_instr_with_result(248)  # OpLabel

# Load inputs
uv_loaded = b.add_instr_with_result(61, uv_var)  # Load uv

# Extract uv components
uv_x = b.add_instr_with_result(81, uv_loaded, 0)  # Extract x
uv_y = b.add_instr_with_result(81, uv_loaded, 1)  # Extract y

# Constants
f0 = b.add_instr_with_result(43, float_t, float_bits(0.0))
f1 = b.add_instr_with_result(43, float_t, float_bits(1.0))
f2 = b.add_instr_with_result(43, float_t, float_bits(2.0))
f3 = b.add_instr_with_result(43, float_t, float_bits(3.0))
f4 = b.add_instr_with_result(43, float_t, float_bits(4.0))
f20 = b.add_instr_with_result(43, float_t, float_bits(20.0))
f0_5 = b.add_instr_with_result(43, float_t, float_bits(0.5))

# Compute c = (uv.x * 3 - 2, uv.y * 2 - 1)
c_x_1 = b.add_instr_with_result(133, uv_x, f3)  # uv.x * 3
c_x = b.add_instr_with_result(131, c_x_1, f2)  # - 2
c_y_1 = b.add_instr_with_result(133, uv_y, f2)  # uv.y * 2
c_y = b.add_instr_with_result(131, c_y_1, f1)  # - 1
c = b.add_instr_with_result(80, c_x, c_y)  # vec2(c_x, c_y)

# Initialize z and iteration count
z_x = f0  # z.x = 0
z_y = f0  # z.y = 0
iter_count = 0
is_escaped = 0

# Mandelbrot iteration (unrolled 20 times)
for step in range(20):
    # Extract current z components
    zx2 = b.add_instr_with_result(133, z_x, z_x)  # z.x * z.x
    zy2 = b.add_instr_with_result(133, z_y, z_y)  # z.y * z.y

    # Compute z = z^2 + c
    real_sq_diff = b.add_instr_with_result(131, zx2, zy2)  # x^2 - y^2
    new_x = b.add_instr_with_result(129, real_sq_diff, c_x)  # + c.x

    two_xy = b.add_instr_with_result(133, z_x, z_y)  # z.x * z.y
    two_xy = b.add_instr_with_result(133, two_xy, f2)  # * 2
    new_y = b.add_instr_with_result(129, two_xy, c_y)  # + c.y

    # Check magnitude |z|^2 > 4
    mag_sq = b.add_instr_with_result(129, zx2, zy2)  # |z|^2
    is_gt_4 = b.add_instr_with_result(166, mag_sq, f4)  # FGreaterThan

    # Update z only if not escaped (simplified - always update)
    z_x = new_x
    z_y = new_y
    iter_count = step + 1

# Final magnitude check for coloring
final_zx2 = b.add_instr_with_result(133, z_x, z_x)
final_zy2 = b.add_instr_with_result(133, z_y, z_y)
final_mag_sq = b.add_instr_with_result(129, final_zx2, final_zy2)

# Simple approach: map magnitude to color
# Normalize: t = min(1, mag_sq / 4)
normalized = b.add_instr_with_result(133, final_mag_sq, f0_5)  # mag_sq * 0.5

# Clamp to [0, 1]
t = normalized  # Use normalized as t (simplified)

# Color: (t, t*0.5, 1.0-t, 1.0)
r_comp = t
g_comp = b.add_instr_with_result(133, t, f0_5)  # t * 0.5
b_neg_t = b.add_instr_with_result(127, t)  # Negate t
b_comp = b.add_instr_with_result(129, f1, b_neg_t)  # 1.0 + (-t) = 1.0 - t

out_color = b.add_instr_with_result(80, r_comp, g_comp, b_comp, f1)

# Store output
b.add_instr(62, out_var, out_color)

# Return
b.add_instr(253)

# End function
b.add_instr(56)

# Output hex
result = b.emit_hex()
print(result)
