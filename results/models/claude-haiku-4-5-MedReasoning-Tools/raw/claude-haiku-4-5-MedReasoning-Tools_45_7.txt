#!/usr/bin/env python3
import struct

class SPIRVGenerator:
    def __init__(self):
        self.instructions = []
        self.next_id = 1

    def get_id(self):
        id = self.next_id
        self.next_id += 1
        return id

    def add_instruction(self, word_count, opcode, *operands):
        """Add an instruction with opcode and operands"""
        header = (word_count << 16) | opcode
        self.instructions.append(struct.pack('<I', header))
        for op in operands:
            if isinstance(op, bytes):
                self.instructions.append(op)
            else:
                self.instructions.append(struct.pack('<I', op))

    def float_to_bits(self, f):
        """Convert float to 32-bit integer bits"""
        return struct.unpack('<I', struct.pack('<f', f))[0]

    def encode_string(self, s):
        """Encode string as SPIR-V format (null-terminated, padded to 4 bytes)"""
        data = s.encode('utf-8') + b'\x00'
        while len(data) % 4 != 0:
            data += b'\x00'
        return data

    def to_hex(self):
        """Convert all instructions to hex string"""
        binary = b''.join(self.instructions)
        return binary.hex()
