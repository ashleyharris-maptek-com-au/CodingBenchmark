#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-bounds invocation
    if (idx >= N) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Read input: body i is stored at indices i*2 and i*2+1
    uint bodyBaseIdx = idx * 2;
    vec4 positionMass = inData[bodyBaseIdx];
    vec4 velocityZero = inData[bodyBaseIdx + 1];

    vec3 position = positionMass.xyz;
    float mass = positionMass.w;
    vec3 velocity = velocityZero.xyz;

    // Compute gravitational acceleration from all other bodies
    vec3 acceleration = vec3(0.0);

    for (uint j = 0; j < N; j++) {
        // Skip self-interaction
        if (j == idx) {
            continue;
        }

        uint otherBaseIdx = j * 2;
        vec4 otherPosData = inData[otherBaseIdx];

        vec3 otherPos = otherPosData.xyz;
        float otherMass = otherPosData.w;

        // Compute displacement vector
        vec3 delta = otherPos - position;
        float distSq = dot(delta, delta);

        // Apply softening parameter to avoid singularities
        float denominator = distSq + softening * softening;

        // Compute distance^3 for the gravitational formula
        float distance = sqrt(denominator);
        float distCubed = denominator * distance;

        // Acceleration formula: acc += G * mass_j * (pos_j - pos_i) / (|pos_j - pos_i|^2 + softening^2)^(3/2)
        float scaleFactor = G * otherMass / distCubed;
        acceleration += scaleFactor * delta;
    }

    // Integrate velocity: v_new = v_old + a * dt
    velocity += acceleration * dt;

    // Integrate position: p_new = p_old + v * dt
    position += velocity * dt;

    // Write output: must write both position+mass and velocity vec4s
    outData[bodyBaseIdx] = vec4(position, mass);
    outData[bodyBaseIdx + 1] = vec4(velocity, 0.0);
}
