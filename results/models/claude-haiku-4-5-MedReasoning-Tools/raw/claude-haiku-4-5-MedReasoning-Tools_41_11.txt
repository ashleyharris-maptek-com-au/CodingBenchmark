OpCapability Shader
OpExtInstImport %glsl450 "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %in_normal %out_color
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_uv Location 2
OpDecorate %in_normal Location 1
OpDecorate %out_color Location 0
OpDecorate %ubo Block
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ubo_ptr = OpTypePointer Uniform %ubo
%v2float_ptr = OpTypePointer Input %v2float
%v3float_ptr = OpTypePointer Input %v3float
%v4float_ptr = OpTypePointer Output %v4float
%float_ptr = OpTypePointer Function %float
%v2float_func_ptr = OpTypePointer Function %v2float
%v3float_func_ptr = OpTypePointer Function %v3float
%v4float_func_ptr = OpTypePointer Function %v4float

%in_uv = OpVariable %v2float_ptr Input
%in_normal = OpVariable %v3float_ptr Input
%out_color = OpVariable %v4float_ptr Output
%ubo_var = OpVariable %ubo_ptr Uniform

%c_0 = OpConstant %float 0.0
%c_0_1 = OpConstant %float 0.1
%c_0_3 = OpConstant %float 0.3
%c_0_5 = OpConstant %float 0.5
%c_0_7 = OpConstant %float 0.7
%c_0_8 = OpConstant %float 0.8
%c_1 = OpConstant %float 1.0
%c_6 = OpConstant %float 6.0

%main_type = OpTypeFunction %void
%main = OpFunction %void None %main_type
%block = OpLabel

%uv = OpLoad %v2float %in_uv
%normal = OpLoad %v3float %in_normal

; Normalize normal
%normal_norm = OpExtInst %v3float %glsl450 Normalize %normal

; Scale UV by 6
%scaled_uv = OpVectorTimesScalar %v2float %uv %c_6

; Get fractional part for position within cell
%local_pos = OpExtInst %v2float %glsl450 Fract %scaled_uv

; Calculate distance from cell center (0.5, 0.5)
%center = OpConstantComposite %v2float %c_0_5 %c_0_5
%diff = OpFSub %v2float %local_pos %center
%dist_sq = OpDot %float %diff %diff
%dist = OpExtInst %float %glsl450 Sqrt %dist_sq

; Check if in dot
%in_dot = OpFOrdLessThan %bool %dist %c_0_3

; Select base color: yellow (1.0, 0.8, 0.0) or dark blue (0.1, 0.1, 0.3)
%yellow = OpConstantComposite %v3float %c_1 %c_0_8 %c_0
%dark_blue = OpConstantComposite %v3float %c_0_1 %c_0_1 %c_0_3
%base_color = OpSelect %v3float %in_dot %yellow %dark_blue

; Light direction (upward)
%light_dir = OpConstantComposite %v3float %c_0 %c_1 %c_0

; Lambertian diffuse: max(0, dot(normal, lightDir))
%diffuse = OpExtInst %float %glsl450 Dot %normal_norm %light_dir
%diffuse_clamped = OpExtInst %float %glsl450 FClamp %diffuse %c_0 %c_1

; Lighting factor: 0.3 + 0.7 * diffuse
%diffuse_contrib = OpFMul %float %c_0_7 %diffuse_clamped
%lighting_factor = OpFAdd %float %c_0_3 %diffuse_contrib

; Apply lighting
%lit_color = OpVectorTimesScalar %v3float %base_color %lighting_factor

; Create output with alpha 1.0
%result = OpCompositeConstruct %v4float %lit_color %c_1

; Store output
OpStore %out_color %result
OpReturn
OpFunctionEnd
