#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <set>
#include <functional>

using namespace std;

typedef long long ll;
typedef pair<ll, int> pli;
const ll INF = 1e18;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<pair<int, ll>>> adj(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        ll w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; i++) {
        cin >> terminals[i];
    }

    if (k == 1) {
        cout << "0 0\n";
        return 0;
    }

    // Dreyfus-Wagner DP
    // dp[mask][v] = minimum cost to connect terminals in mask with v in the tree
    vector<vector<ll>> dp(1 << k, vector<ll>(n, INF));

    // parent[mask][v] stores reconstruction info:
    // (type, arg1, arg2) where:
    // type 0 = base case (single terminal)
    // type 1 = Dijkstra edge from arg1
    // type 2 = merge from subsets arg1 and arg2
    vector<vector<tuple<int, int, int>>> parent(1 << k, vector<tuple<int, int, int>>(n, {0, -1, -1}));

    // Initialize: single terminals
    for (int i = 0; i < k; i++) {
        int t = terminals[i];
        dp[1 << i][t] = 0;
        parent[1 << i][t] = {0, -1, -1};
    }

    // Process all subsets
    for (int mask = 1; mask < (1 << k); mask++) {
        // Merge phase: combine smaller subsets
        for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
            int comp = mask ^ sub;
            for (int v = 0; v < n; v++) {
                if (dp[sub][v] < INF && dp[comp][v] < INF) {
                    ll new_cost = dp[sub][v] + dp[comp][v];
                    if (new_cost < dp[mask][v]) {
                        dp[mask][v] = new_cost;
                        parent[mask][v] = {2, sub, comp};
                    }
                }
            }
        }

        // Dijkstra phase: extend tree to other vertices
        priority_queue<pli, vector<pli>, greater<pli>> pq;

        for (int v = 0; v < n; v++) {
            if (dp[mask][v] < INF) {
                pq.push({dp[mask][v], v});
            }
        }

        while (!pq.empty()) {
            auto [cost, u] = pq.top();
            pq.pop();

            if (cost > dp[mask][u]) continue;

            for (auto [v, w] : adj[u]) {
                if (dp[mask][u] + w < dp[mask][v]) {
                    dp[mask][v] = dp[mask][u] + w;
                    parent[mask][v] = {1, u, -1};
                    pq.push({dp[mask][v], v});
                }
            }
        }
    }

    // Find minimum cost to connect all terminals
    ll ans = INF;
    int ans_v = -1;
    int full_mask = (1 << k) - 1;

    for (int v = 0; v < n; v++) {
        if (dp[full_mask][v] < ans) {
            ans = dp[full_mask][v];
            ans_v = v;
        }
    }

    // Reconstruct the tree edges
    set<pair<int, int>> edges;

    function<void(int, int)> reconstruct = [&](int mask, int v) {
        auto [type, arg1, arg2] = parent[mask][v];

        if (type == 0) {
            // Base case: single terminal, nothing to add
            return;
        } else if (type == 1) {
            // Dijkstra edge from arg1 to v
            int u = arg1;
            edges.insert({min(u, v), max(u, v)});
            reconstruct(mask, u);
        } else if (type == 2) {
            // Merge of two subsets at vertex v
            int sub = arg1, comp = arg2;
            reconstruct(sub, v);
            reconstruct(comp, v);
        }
    };

    reconstruct(full_mask, ans_v);

    cout << ans << " " << edges.size() << "\n";
    for (auto [u, v] : edges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
