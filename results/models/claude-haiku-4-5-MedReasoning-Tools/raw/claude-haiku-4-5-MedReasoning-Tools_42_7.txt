; SPIR-V Geometry Shader - Triangle Explosion
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

%glsl = OpExtInstImport "GLSL.std.450"

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

; Type definitions
%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0_08 = OpConstant %float 0.08

; Struct definitions
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Array types
%arr_PerVertex_3 = OpTypeArray %gl_PerVertex_in %int_3
%arr_v3float_3 = OpTypeArray %v3float %int_3

; Pointer types
%ptr_Input_arr_PerVertex = OpTypePointer Input %arr_PerVertex_3
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3float_3
%ptr_Output_PerVertex = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v3 = OpTypePointer Output %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float

; Function type
%func_void = OpTypeFunction %void

; Variables
%gl_in = OpVariable %ptr_Input_arr_PerVertex Input
%inColor = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_PerVertex Output
%outColor = OpVariable %ptr_Output_v3 Output

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load gl_in[0].gl_Position
%addr_p0 = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %addr_p0

; Load gl_in[1].gl_Position
%addr_p1 = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %addr_p1

; Load gl_in[2].gl_Position
%addr_p2 = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %addr_p2

; Extract xyz components
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Compute edge vectors: e1 = p1 - p0, e2 = p2 - p0
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz

; Compute cross product (opcode 68)
%cross = OpExtInst %v3float %glsl 68 %e1 %e2

; Normalize (opcode 69)
%normal = OpExtInst %v3float %glsl 69 %cross

; Compute offset = normal * 0.08
%offset = OpVectorTimesScalar %v3float %normal %float_0_08

; Load colors
%addr_color0 = OpAccessChain %ptr_Input_v3 %inColor %int_0
%color0 = OpLoad %v3float %addr_color0

%addr_color1 = OpAccessChain %ptr_Input_v3 %inColor %int_1
%color1 = OpLoad %v3float %addr_color1

%addr_color2 = OpAccessChain %ptr_Input_v3 %inColor %int_2
%color2 = OpLoad %v3float %addr_color2

; Emit vertex 0
%p0_offset_xyz = OpFAdd %v3float %p0_xyz %offset
%p0_offset_v4 = OpVectorShuffle %v4float %p0_offset_xyz %p0 0 1 2 6
%addr_out_pos = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %addr_out_pos %p0_offset_v4
OpStore %outColor %color0
OpEmitVertex

; Emit vertex 1
%p1_offset_xyz = OpFAdd %v3float %p1_xyz %offset
%p1_offset_v4 = OpVectorShuffle %v4float %p1_offset_xyz %p1 0 1 2 6
OpStore %addr_out_pos %p1_offset_v4
OpStore %outColor %color1
OpEmitVertex

; Emit vertex 2
%p2_offset_xyz = OpFAdd %v3float %p2_xyz %offset
%p2_offset_v4 = OpVectorShuffle %v4float %p2_offset_xyz %p2 0 1 2 6
OpStore %addr_out_pos %p2_offset_v4
OpStore %outColor %color2
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
