declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Robot {
  id: number;
  x: number;
  y: number;
}

interface Job {
  id: number;
  x: number;
  y: number;
  priority: number;
}

interface GridData {
  grid: string[];
  robots: Robot[];
  jobs: Job[];
}

const gridData: GridData = data;

function shortestPathDistance(grid: string[], fromX: number, fromY: number, toX: number, toY: number): number {
  if (fromX === toX && fromY === toY) return 0;

  const rows = grid.length;
  if (rows === 0) return Infinity;
  const cols = grid[0].length;

  if (toX < 0 || toX >= cols || toY < 0 || toY >= rows) return Infinity;
  if (grid[toY][toX] === '#') return Infinity;

  const visited = new Set<number>();
  const queue: [number, number, number][] = [[fromX, fromY, 0]];
  visited.add(fromY * cols + fromX);

  while (queue.length > 0) {
    const [x, y, dist] = queue.shift()!;

    const directions = [[0, 1], [1, 0], [0, -1], [-1, 0]];
    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && grid[ny][nx] === '.') {
        const key = ny * cols + nx;
        if (!visited.has(key)) {
          if (nx === toX && ny === toY) {
            return dist + 1;
          }
          visited.add(key);
          queue.push([nx, ny, dist + 1]);
        }
      }
    }
  }

  return Infinity;
}

const jobMap = new Map<number, Job>();
for (const job of gridData.jobs) {
  jobMap.set(job.id, job);
}

const assignments: number[][] = Array(gridData.robots.length).fill(null).map(() => []);
const assignedJobIds = new Set<number>();
const robotPositions: [number, number][] = gridData.robots.map(r => [r.x, r.y]);

while (assignedJobIds.size < gridData.jobs.length) {
  let bestRobotIdx = -1;
  let bestJobId = -1;
  let bestDistance = Infinity;

  for (let i = 0; i < gridData.robots.length; i++) {
    const [rx, ry] = robotPositions[i];
    for (const job of gridData.jobs) {
      if (!assignedJobIds.has(job.id)) {
        const distance = shortestPathDistance(gridData.grid, rx, ry, job.x, job.y);

        if (distance < bestDistance) {
          bestDistance = distance;
          bestRobotIdx = i;
          bestJobId = job.id;
        }
      }
    }
  }

  if (bestRobotIdx !== -1 && bestJobId !== -1) {
    const job = jobMap.get(bestJobId)!;
    assignments[bestRobotIdx].push(bestJobId);
    assignedJobIds.add(bestJobId);
    robotPositions[bestRobotIdx] = [job.x, job.y];
  } else {
    break;
  }
}
