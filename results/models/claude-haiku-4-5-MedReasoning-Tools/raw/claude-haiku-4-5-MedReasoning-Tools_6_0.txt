using System;
using System.Collections.Generic;
using System.Linq;
using static System.Math;

class Program
{
    const double MU = 398600.4418;
    const double EARTH_RADIUS = 6371;

    struct Orbit
    {
        public double X, Y, Z, Vx, Vy, Vz;
        public double Radius => Sqrt(X * X + Y * Y + Z * Z);
        public double Speed => Sqrt(Vx * Vx + Vy * Vy + Vz * Vz);
    }

    static void Main()
    {
        var line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);
        double mu = double.Parse(line1[1]);

        var start = ParseOrbit(Console.ReadLine());
        var stations = new Orbit[n];
        for (int i = 0; i < n; i++)
            stations[i] = ParseOrbit(Console.ReadLine());

        double[,] cost = new double[n, n];
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (i != j)
                    cost[i, j] = TransferCost(stations[i], stations[j], mu);

        int[] solution = SolveTSP(cost, n);
        Console.WriteLine(string.Join(" ", solution));
    }

    static Orbit ParseOrbit(string line)
    {
        var values = line.Split().Select(double.Parse).ToArray();
        return new Orbit
        {
            X = values[0],
            Y = values[1],
            Z = values[2],
            Vx = values[3],
            Vy = values[4],
            Vz = values[5]
        };
    }

    static double TransferCost(Orbit from, Orbit to, double mu)
    {
        double r1 = from.Radius;
        double r2 = to.Radius;

        if (r1 < 1 || r2 < 1)
            return 1e10;

        double v1 = Sqrt(mu / r1);
        double v2 = Sqrt(mu / r2);

        double a = (r1 + r2) / 2;
        double vt1 = Sqrt(mu * (2 / r1 - 1 / a));
        double vt2 = Sqrt(mu * (2 / r2 - 1 / a));

        return Abs(vt1 - v1) + Abs(v2 - vt2);
    }

    static int[] SolveTSP(double[,] cost, int n)
    {
        if (n <= 5)
            return SolveBruteForce(cost, n);
        else if (n <= 12)
            return SolveGreedyWith2Opt(cost, n);
        else
            return SolveGreedyWith2OptLimited(cost, n, 50);
    }

    static int[] SolveBruteForce(double[,] cost, int n)
    {
        var perm = Enumerable.Range(0, n).ToArray();
        var best = (int[])perm.Clone();
        double bestCost = TourCost(best, cost);

        while (NextPermutation(perm))
        {
            double c = TourCost(perm, cost);
            if (c < bestCost)
            {
                bestCost = c;
                Array.Copy(perm, best, n);
            }
        }

        return best;
    }

    static int[] SolveGreedyWith2Opt(double[,] cost, int n)
    {
        int[] tour = GreedyTour(cost, n);
        Optimize2Opt(tour, cost);
        return tour;
    }

    static int[] SolveGreedyWith2OptLimited(double[,] cost, int n, int maxIterations)
    {
        int[] tour = GreedyTour(cost, n);
        Optimize2OptLimited(tour, cost, maxIterations);
        return tour;
    }

    static int[] GreedyTour(double[,] cost, int n)
    {
        int[] tour = new int[n];
        bool[] used = new bool[n];
        tour[0] = 0;
        used[0] = true;

        for (int i = 1; i < n; i++)
        {
            int last = tour[i - 1];
            int next = -1;
            double minCost = double.MaxValue;

            for (int j = 0; j < n; j++)
            {
                if (!used[j] && cost[last, j] < minCost)
                {
                    minCost = cost[last, j];
                    next = j;
                }
            }

            tour[i] = next;
            used[next] = true;
        }

        return tour;
    }

    static void Optimize2Opt(int[] tour, double[,] cost)
    {
        int n = tour.Length;
        bool improved = true;

        while (improved)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    if (j == n - 1 && i == 0) continue;

                    int a = tour[i], b = tour[i + 1];
                    int c = tour[j], d = tour[(j + 1) % n];

                    double oldCost = cost[a, b] + cost[c, d];
                    double newCost = cost[a, c] + cost[b, d];

                    if (newCost < oldCost - 1e-9)
                    {
                        Array.Reverse(tour, i + 1, j - i);
                        improved = true;
                    }
                }
            }
        }
    }

    static void Optimize2OptLimited(int[] tour, double[,] cost, int maxIterations)
    {
        int n = tour.Length;
        int iterations = 0;

        bool improved = true;
        while (improved && iterations < maxIterations)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    if (j == n - 1 && i == 0) continue;

                    int a = tour[i], b = tour[i + 1];
                    int c = tour[j], d = tour[(j + 1) % n];

                    double oldCost = cost[a, b] + cost[c, d];
                    double newCost = cost[a, c] + cost[b, d];

                    if (newCost < oldCost - 1e-9)
                    {
                        Array.Reverse(tour, i + 1, j - i);
                        improved = true;
                        break;
                    }
                }
                if (improved) break;
            }
            iterations++;
        }
    }

    static double TourCost(int[] tour, double[,] cost)
    {
        double total = 0;
        for (int i = 0; i < tour.Length; i++)
        {
            total += cost[tour[i], tour[(i + 1) % tour.Length]];
        }
        return total;
    }

    static bool NextPermutation(int[] array)
    {
        int i = array.Length - 2;
        while (i >= 0 && array[i] >= array[i + 1]) i--;
        if (i < 0) return false;

        int j = array.Length - 1;
        while (array[j] <= array[i]) j--;

        int temp = array[i];
        array[i] = array[j];
        array[j] = temp;

        Array.Reverse(array, i + 1, array.Length - i - 1);
        return true;
    }
}
