import statistics
from collections import deque

_state = {
    'speed_history': deque(maxlen=15),
    'lane_offset_history': deque(maxlen=15),
    'speed_error_integral': 0.0,
    'steering_error_integral': 0.0,
    'last_steering': 0.0,
    'last_throttle': 0.0,
    'last_brake': 0.0,
    'obstacle_frames': 0,
    'heading': 0.0,
}

def _fuse_redundant(sensor_data_list):
    """Fuse redundant sensors: keep only valid readings, use median."""
    valid = [v for v, is_valid in sensor_data_list if is_valid]
    if not valid:
        return None
    return statistics.median(valid) if len(valid) > 1 else valid[0]

def _get_speed(state):
    """Fuse speed from wheels, vehicle sensors, and GPS."""
    wheel_speeds = []
    for wheel in ['fl', 'fr', 'rl', 'rr']:
        val = state.get(f'wheel_speed_{wheel}_kmh', 0)
        valid = state.get(f'wheel_speed_{wheel}_valid', False)
        wheel_speeds.append((val, valid))

    fused_wheel = _fuse_redundant(wheel_speeds)

    gps_speeds = []
    for i in [1, 2]:
        val = state.get(f'gps_{i}_speed_kmh', 0)
        valid = state.get(f'gps_{i}_valid', False)
        gps_speeds.append((val, valid))
    fused_gps = _fuse_redundant(gps_speeds)

    all_speeds = []
    if fused_wheel is not None:
        all_speeds.append(fused_wheel)
    if fused_gps is not None:
        all_speeds.append(fused_gps)
    if not all_speeds:
        all_speeds.append(statistics.median([
            state.get('vehicle_speed_1_kmh', 0),
            state.get('vehicle_speed_2_kmh', 0)
        ]))

    return statistics.median(all_speeds)

def _get_lane_offset(state):
    """Fuse lane offset from vision systems."""
    offsets = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_left_line_valid', False) and \
           state.get(f'vision_{i}_right_line_valid', False):
            val = state.get(f'vision_{i}_lane_offset_m', 0)
            offsets.append(val)

    if offsets:
        return statistics.median(offsets)
    return 0.0

def _get_speed_limit(state):
    """Get speed limit from vision systems."""
    limits = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_speed_limit_valid', False):
            limits.append(state.get(f'vision_{i}_speed_limit_kmh', 130))
    return statistics.median(limits) if limits else 130

def _get_road_curvature(state):
    """Get road curvature for feedforward steering."""
    curvatures = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_left_line_valid', False) and \
           state.get(f'vision_{i}_right_line_valid', False):
            curvatures.append(state.get(f'vision_{i}_road_curvature_1pm', 0))
    return statistics.median(curvatures) if curvatures else 0

def _detect_closest_obstacle(state):
    """Detect closest obstacle from vision and lidar."""
    obstacles = []

    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_obstacle_valid', False) and \
           state.get(f'vision_{i}_obstacle_detected', False):
            dist = state.get(f'vision_{i}_obstacle_dist_m', 100)
            rel_speed = state.get(f'vision_{i}_obstacle_rel_speed_kmh', 0)
            obstacles.append(('vision', dist, rel_speed))

    front_angles = [-15, -10, -5, 0, 5, 10, 15]
    for angle in front_angles:
        angle_str = f"{angle:+03d}"
        key_range = f'lidar_{angle_str}_deg_range_m'
        key_valid = f'lidar_{angle_str}_deg_valid'
        if state.get(key_valid, False):
            dist = state.get(key_range, float('inf'))
            if dist < 60:
                obstacles.append(('lidar', dist, 0))

    if not obstacles:
        return None

    obstacles.sort(key=lambda x: x[1])
    return obstacles[0]

def autopilot_step(state: dict, dt: float) -> dict:
    dt = max(dt, 0.001)

    current_speed = _get_speed(state)
    _state['speed_history'].append(current_speed)
    filtered_speed = statistics.mean(_state['speed_history']) if _state['speed_history'] else current_speed

    lane_offset = _get_lane_offset(state)
    _state['lane_offset_history'].append(lane_offset)
    filtered_lane_offset = statistics.mean(_state['lane_offset_history']) if _state['lane_offset_history'] else lane_offset

    mu = state.get('estimated_mu', 1.0)
    target_speed = state.get('target_speed_kmh', 100)
    speed_limit = _get_speed_limit(state)
    target_speed = min(target_speed, speed_limit)

    if mu < 0.3:
        target_speed = min(target_speed, 40)
    elif mu < 0.5:
        target_speed = min(target_speed, 70)

    obstacle = _detect_closest_obstacle(state)

    if obstacle:
        dist = obstacle[1]
        _state['obstacle_frames'] = 50

        if dist < 12:
            target_speed = 0
        elif dist < 25:
            target_speed = min(target_speed, 20)
        elif dist < 40:
            target_speed = min(target_speed, 40)
    else:
        _state['obstacle_frames'] = max(0, _state['obstacle_frames'] - 1)

    speed_error = target_speed - filtered_speed
    _state['speed_error_integral'] += speed_error * dt
    _state['speed_error_integral'] = max(-30, min(30, _state['speed_error_integral']))

    throttle = 0.0
    brake = 0.0

    if _state['obstacle_frames'] == 0:
        if speed_error > 2:
            throttle = 0.3 + 0.04 * speed_error + 0.02 * _state['speed_error_integral']
            throttle = max(0, min(1.0, throttle))
        elif speed_error < -2:
            brake = 0.25 - 0.04 * speed_error - 0.02 * _state['speed_error_integral']
            brake = max(0, min(1.0, brake))
    else:
        if obstacle and obstacle[1] < 15:
            brake = 1.0
        elif speed_error < 0:
            brake = 0.3 - 0.03 * speed_error
            brake = max(0, min(1.0, brake))

    steering_error = -filtered_lane_offset
    _state['steering_error_integral'] += steering_error * dt
    _state['steering_error_integral'] = max(-1.5, min(1.5, _state['steering_error_integral']))

    Kp = 0.5 if filtered_speed < 50 else 0.3
    Ki = 0.15
    steering = Kp * steering_error + Ki * _state['steering_error_integral']

    curvature = _get_road_curvature(state)
    if abs(curvature) > 0.001:
        steering += curvature * (filtered_speed / 100.0) * 0.2

    if filtered_speed > 110:
        steering *= 0.4
    elif filtered_speed > 80:
        steering *= 0.65

    if mu < 0.5:
        steering *= 0.65
        throttle *= 0.65

    max_steering_delta = 1.5 * dt
    steering = max(_state['last_steering'] - max_steering_delta,
                  min(_state['last_steering'] + max_steering_delta, steering))

    steering = max(-1.0, min(1.0, steering))

    brake_1_valid = state.get('brake_pressure_1_valid', True)
    brake_2_valid = state.get('brake_pressure_2_valid', True)
    if not brake_1_valid and not brake_2_valid:
        brake = 0
        if filtered_speed > target_speed:
            throttle = 0

    steering_angles = []
    for i in [1, 2]:
        val = state.get(f'steering_angle_{i}_deg', 0)
        valid = state.get(f'steering_angle_{i}_valid', False)
        if valid:
            steering_angles.append(val)

    if len(steering_angles) == 2 and abs(steering_angles[0] - steering_angles[1]) > 15:
        steering *= 0.7

    _state['last_steering'] = steering
    _state['last_throttle'] = throttle
    _state['last_brake'] = brake

    return {
        'steering': steering,
        'throttle': throttle,
        'brake': brake,
    }
