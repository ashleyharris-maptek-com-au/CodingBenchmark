#!/usr/bin/env python3
"""
Generate SPIR-V fragment shader for procedural brick pattern.
"""

import struct

def float_to_bits(f):
    """Convert float to IEEE 754 bits."""
    return struct.unpack('>I', struct.pack('>f', f))[0]

class SpirvBuilder:
    def __init__(self):
        self.id_counter = 0
        self.words = []

    def new_id(self):
        self.id_counter += 1
        return self.id_counter

    def add_instruction(self, opcode, *operands):
        """Add instruction: [word_count|opcode][operands...]"""
        word_count = 1 + len(operands)
        header = (word_count << 16) | (opcode & 0xffff)
        self.words.append(header & 0xffffffff)
        for op in operands:
            self.words.append(op & 0xffffffff)

    def add_string(self, s):
        """Encode string and return word operands."""
        encoded = s.encode('utf-8') + b'\x00'
        # Pad to 4-byte boundary
        while len(encoded) % 4 != 0:
            encoded += b'\x00'
        result = []
        for i in range(0, len(encoded), 4):
            word = struct.unpack('<I', encoded[i:i+4])[0]
            result.append(word)
        return result

    def to_hex(self):
        """Generate final SPIR-V hex string."""
        bound = self.id_counter + 1
        header = [
            0x07230203,  # Magic
            0x00010000,  # Version 1.0
            0,           # Generator
            bound,       # Bound
            0            # Reserved
        ]

        all_words = header + self.words
        binary = b''
        for w in all_words:
            binary += struct.pack('<I', w)
        return binary.hex()

b = SpirvBuilder()

# === Type IDs ===
void_t = b.new_id()
float_t = b.new_id()
vec2_t = b.new_id()
vec3_t = b.new_id()
vec4_t = b.new_id()
bool_t = b.new_id()

# Pointer types (storage classes: Input=1, Output=3, Private=6)
ptr_in_vec2 = b.new_id()
ptr_in_vec3 = b.new_id()
ptr_out_vec4 = b.new_id()
ptr_priv_vec2 = b.new_id()
ptr_priv_vec3 = b.new_id()
ptr_priv_float = b.new_id()
ptr_priv_bool = b.new_id()

func_type = b.new_id()

# === Constant IDs ===
c0 = b.new_id()
c1 = b.new_id()
c0p5 = b.new_id()
c2 = b.new_id()
c05 = b.new_id()
c0p1 = b.new_id()
c0p2 = b.new_id()
c0p7 = b.new_id()
c8 = b.new_id()
c16 = b.new_id()
c0p95 = b.new_id()
c0p9 = b.new_id()

# Constant vectors
col_mortar = b.new_id()
col_brick = b.new_id()

# === Variables ===
var_uv = b.new_id()
var_normal = b.new_id()
var_wpos = b.new_id()
var_out = b.new_id()

main_func = b.new_id()
entry_label = b.new_id()

# === Instructions ===

# Capability
b.add_instruction(17, 1)  # OpCapability Shader

# Extension
ext_glsl = b.new_id()
b.add_instruction(11, ext_glsl, *b.add_string("GLSL.std.450"))

# Memory model
b.add_instruction(14, 0, 1)  # OpMemoryModel Logical GLSL450

# Type declarations
b.add_instruction(19, void_t)  # OpTypeVoid
b.add_instruction(22, float_t, 32)  # OpTypeFloat
b.add_instruction(23, vec2_t, float_t, 2)  # OpTypeVector vec2
b.add_instruction(23, vec3_t, float_t, 3)  # OpTypeVector vec3
b.add_instruction(23, vec4_t, float_t, 4)  # OpTypeVector vec4
b.add_instruction(20, bool_t)  # OpTypeBool

# Pointer types
b.add_instruction(32, ptr_in_vec2, 1, vec2_t)  # Input vec2*
b.add_instruction(32, ptr_in_vec3, 1, vec3_t)  # Input vec3*
b.add_instruction(32, ptr_out_vec4, 3, vec4_t)  # Output vec4*
b.add_instruction(32, ptr_priv_vec2, 6, vec2_t)  # Private vec2*
b.add_instruction(32, ptr_priv_vec3, 6, vec3_t)  # Private vec3*
b.add_instruction(32, ptr_priv_float, 6, float_t)  # Private float*
b.add_instruction(32, ptr_priv_bool, 6, bool_t)  # Private bool*

# Function type: () -> void
b.add_instruction(33, func_type, void_t)

# Constants
b.add_instruction(43, float_t, c0, 0)  # 0.0
b.add_instruction(43, float_t, c1, float_to_bits(1.0))  # 1.0
b.add_instruction(43, float_t, c0p5, float_to_bits(0.5))  # 0.5
b.add_instruction(43, float_t, c2, float_to_bits(2.0))  # 2.0
b.add_instruction(43, float_t, c05, float_to_bits(0.05))  # 0.05 (mortar X)
b.add_instruction(43, float_t, c0p1, float_to_bits(0.1))  # 0.1 (mortar Y)
b.add_instruction(43, float_t, c0p2, float_to_bits(0.2))  # 0.2 (brick R)
b.add_instruction(43, float_t, c0p7, float_to_bits(0.7))  # 0.7
b.add_instruction(43, float_t, c8, float_to_bits(8.0))  # 8.0
b.add_instruction(43, float_t, c16, float_to_bits(16.0))  # 16.0
b.add_instruction(43, float_t, c0p95, float_to_bits(0.95))  # 1.0 - 0.05
b.add_instruction(43, float_t, c0p9, float_to_bits(0.9))  # 1.0 - 0.1

# Constant vectors
b.add_instruction(80, vec3_t, col_mortar, c0p7, c0p7, c0p7)  # (0.7, 0.7, 0.7)
b.add_instruction(80, vec3_t, col_brick, c0p7, c0p2, c0p1)  # (0.7, 0.2, 0.1)

# Variables
b.add_instruction(59, ptr_in_vec2, var_uv, 1)  # Input
b.add_instruction(59, ptr_in_vec3, var_normal, 1)  # Input
b.add_instruction(59, ptr_in_vec3, var_wpos, 1)  # Input
b.add_instruction(59, ptr_out_vec4, var_out, 3)  # Output

# Decorations (locations)
b.add_instruction(71, var_uv, 30, 2)  # location 2
b.add_instruction(71, var_normal, 30, 1)  # location 1
b.add_instruction(71, var_wpos, 30, 0)  # location 0
b.add_instruction(71, var_out, 30, 0)  # location 0

# Entry point
b.add_instruction(15, 4, main_func, *b.add_string("main"), var_out)

# Execution mode
b.add_instruction(16, main_func, 7)  # OriginUpperLeft

# === Function Definition ===
b.add_instruction(54, void_t, main_func, 0, func_type)  # OpFunction

# Entry block
b.add_instruction(248, entry_label)  # OpLabel

# Load inputs
load_uv = b.new_id()
b.add_instruction(61, vec2_t, load_uv, var_uv)  # OpLoad

load_normal = b.new_id()
b.add_instruction(61, vec3_t, load_normal, var_normal)

load_wpos = b.new_id()
b.add_instruction(61, vec3_t, load_wpos, var_wpos)

# === Brick pattern logic ===

# Scale UV: scaled_uv = uv * (8, 16)
scale_vec = b.new_id()
b.add_instruction(80, vec2_t, scale_vec, c8, c16)

scaled_uv = b.new_id()
b.add_instruction(133, vec2_t, scaled_uv, load_uv, scale_vec)  # OpFMul

# Extract Y for row detection
uv_y = b.new_id()
b.add_instruction(81, float_t, uv_y, scaled_uv, 1)  # OpCompositeExtract Y

# Extract X
uv_x = b.new_id()
b.add_instruction(81, float_t, uv_x, scaled_uv, 0)  # OpCompositeExtract X

# Get row: floor(uv_y)
row = b.new_id()
b.add_instruction(12, float_t, row, ext_glsl, 8, uv_y)  # OpExtInst floor

# Check if row is even: mod(floor(uv_y/2), 2) < 1
row_half = b.new_id()
b.add_instruction(133, float_t, row_half, row, c0p5)  # row / 2

row_fract = b.new_id()
b.add_instruction(12, float_t, row_fract, ext_glsl, 10, row_half)  # fract

is_even = b.new_id()
b.add_instruction(112, bool_t, is_even, row_fract, 1, c0p5)  # OrdLessThan (fract < 0.5)

# If odd row, offset X by 0.5
offset_x = b.new_id()
b.add_instruction(129, float_t, offset_x, uv_x, c0p5)  # OpFAdd

adjusted_x = b.new_id()
b.add_instruction(171, float_t, adjusted_x, is_even, uv_x, offset_x)  # OpSelect

# Reconstruct adjusted UV
adj_uv = b.new_id()
b.add_instruction(80, vec2_t, adj_uv, adjusted_x, uv_y)

# Get fractional parts
fract_uv = b.new_id()
b.add_instruction(12, vec2_t, fract_uv, ext_glsl, 10, adj_uv)  # fract

fract_x = b.new_id()
b.add_instruction(81, float_t, fract_x, fract_uv, 0)

fract_y = b.new_id()
b.add_instruction(81, float_t, fract_y, fract_uv, 1)

# Detect mortar: x < 0.05 or x > 0.95
mortar_x_min = b.new_id()
b.add_instruction(112, bool_t, mortar_x_min, fract_x, 1, c05)  # OrdLessThan

mortar_x_max = b.new_id()
b.add_instruction(113, bool_t, mortar_x_max, fract_x, 0, c0p95)  # OrdGreaterThan

is_mortar_x = b.new_id()
b.add_instruction(139, bool_t, is_mortar_x, mortar_x_min, mortar_x_max)  # OpLogicalOr

# y < 0.1 or y > 0.9
mortar_y_min = b.new_id()
b.add_instruction(112, bool_t, mortar_y_min, fract_y, 1, c0p1)

mortar_y_max = b.new_id()
b.add_instruction(113, bool_t, mortar_y_max, fract_y, 0, c0p9)

is_mortar_y = b.new_id()
b.add_instruction(139, bool_t, is_mortar_y, mortar_y_min, mortar_y_max)

# is_mortar = is_mortar_x OR is_mortar_y
is_mortar = b.new_id()
b.add_instruction(139, bool_t, is_mortar, is_mortar_x, is_mortar_y)

# Select color
color = b.new_id()
b.add_instruction(171, vec3_t, color, is_mortar, col_mortar, col_brick)

# === Lambertian shading ===

# Use fixed light at (1, 1, 1)
light = b.new_id()
b.add_instruction(80, vec3_t, light, c1, c1, c1)

# light_dir = light - wpos
light_vec = b.new_id()
b.add_instruction(131, vec3_t, light_vec, light, load_wpos)  # OpFSub

# Normalize
light_dir = b.new_id()
b.add_instruction(12, vec3_t, light_dir, ext_glsl, 69, light_vec)  # normalize

# dot(normal, light_dir)
dot_val = b.new_id()
b.add_instruction(148, float_t, dot_val, load_normal, light_dir)

# max(dot, 0)
diffuse = b.new_id()
b.add_instruction(12, float_t, diffuse, ext_glsl, 40, dot_val, c0)  # fmax

# Multiply color by diffuse
shaded = b.new_id()
shaded_r = b.new_id()
r = b.new_id()
b.add_instruction(81, float_t, r, color, 0)
b.add_instruction(133, float_t, shaded_r, r, diffuse)

shaded_g = b.new_id()
g = b.new_id()
b.add_instruction(81, float_t, g, color, 1)
b.add_instruction(133, float_t, shaded_g, g, diffuse)

shaded_b = b.new_id()
bl = b.new_id()
b.add_instruction(81, float_t, bl, color, 2)
b.add_instruction(133, float_t, shaded_b, bl, diffuse)

shaded_rgb = b.new_id()
b.add_instruction(80, vec3_t, shaded_rgb, shaded_r, shaded_g, shaded_b)

# Convert to vec4(rgb, 1.0)
out_vec4 = b.new_id()
b.add_instruction(80, vec4_t, out_vec4, shaded_rgb, c1)

# Store output
b.add_instruction(62, var_out, out_vec4)  # OpStore

# Return
b.add_instruction(253)  # OpReturn

# Function end
b.add_instruction(56)  # OpFunctionEnd

# Output
print(b.to_hex())
