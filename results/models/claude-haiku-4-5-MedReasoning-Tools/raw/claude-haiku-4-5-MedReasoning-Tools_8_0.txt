use std::io::{self, BufRead};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();

    let mut line = String::new();
    reader.read_line(&mut line).unwrap();
    let dims: Vec<usize> = line.trim().split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let h = dims[0];
    let w = dims[1];

    let mut maze = Vec::with_capacity(h);
    let mut start = (0, 0);
    let mut goal = (0, 0);

    for y in 0..h {
        line.clear();
        reader.read_line(&mut line).unwrap();
        let row: Vec<char> = line.trim_end().chars().collect();

        for (x, &ch) in row.iter().enumerate() {
            match ch {
                'A' => start = (x, y),
                'B' => goal = (x, y),
                _ => {}
            }
        }

        maze.push(row);
    }

    if let Some(path) = bfs(&maze, start, goal, h, w) {
        println!("{}", path.len());
        for (x, y) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn bfs(maze: &[Vec<char>], start: (usize, usize), goal: (usize, usize), h: usize, w: usize) -> Option<Vec<(usize, usize)>> {
    if start == goal {
        return Some(vec![start]);
    }

    let mut queue = VecDeque::new();
    let mut visited = vec![vec![false; w]; h];
    let mut parent = vec![vec![None; w]; h];

    queue.push_back(start);
    visited[start.1][start.0] = true;

    let directions = [(0, 1), (0, -1), (1, 0), (-1, 0)];

    while let Some((x, y)) = queue.pop_front() {
        if (x, y) == goal {
            let mut path = Vec::new();
            let mut curr = goal;
            while let Some(p) = parent[curr.1][curr.0] {
                path.push(curr);
                curr = p;
            }
            path.push(start);
            path.reverse();
            return Some(path);
        }

        for (dx, dy) in &directions {
            let nx = (x as i32 + dx) as usize;
            let ny = (y as i32 + dy) as usize;

            if nx < w && ny < h && !visited[ny][nx] && maze[ny][nx] != '#' {
                visited[ny][nx] = true;
                parent[ny][nx] = Some((x, y));
                queue.push_back((nx, ny));
            }
        }
    }

    None
}
