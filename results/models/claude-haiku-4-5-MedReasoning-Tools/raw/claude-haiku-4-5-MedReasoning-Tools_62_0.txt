declare const require: any;
const fs = require('fs');

interface Input {
  width: number;
  height: number;
  grid: string[];
  ignitions: [number, number][];
  towns: Array<{ id: string; x: number; y: number; value: number }>;
  cutBudget: number;
}

function simulateFire(
  grid: string[],
  width: number,
  height: number,
  ignitions: [number, number][],
  cuts: Set<string>
): Set<string> {
  const burned = new Set<string>();
  const queue: [number, number][] = [];

  // Start fire from all ignition points
  for (const [x, y] of ignitions) {
    if (grid[y]?.[x] === '.') {
      const key = `${x},${y}`;
      burned.add(key);
      queue.push([x, y]);
    }
  }

  // BFS to spread fire through burnable cells
  while (queue.length > 0) {
    const [x, y] = queue.shift()!;

    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
        const key = `${nx},${ny}`;
        if (!burned.has(key) && grid[ny][nx] === '.' && !cuts.has(key)) {
          burned.add(key);
          queue.push([nx, ny]);
        }
      }
    }
  }

  return burned;
}

function calculateSavedValue(
  towns: Array<{ id: string; x: number; y: number; value: number }>,
  burned: Set<string>
): number {
  let saved = 0;
  for (const town of towns) {
    if (!burned.has(`${town.x},${town.y}`)) {
      saved += town.value;
    }
  }
  return saved;
}

function solve(input: Input): [number, number][] {
  const { grid, width, height, ignitions, towns, cutBudget } = input;
  const cuts = new Set<string>();

  for (let iter = 0; iter < cutBudget; iter++) {
    const burned = simulateFire(grid, width, height, ignitions, cuts);
    const currentValue = calculateSavedValue(towns, burned);

    let bestCell: [number, number] | null = null;
    let bestValue = currentValue;

    // Find candidate cells adjacent to the fire boundary
    const candidates = new Set<string>();
    for (const cellStr of burned) {
      const [x, y] = cellStr.split(',').map(Number);
      for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
        const nx = x + dx;
        const ny = y + dy;
        const key = `${nx},${ny}`;
        if (
          nx >= 0 &&
          nx < width &&
          ny >= 0 &&
          ny < height &&
          grid[ny][nx] === '.' &&
          !cuts.has(key)
        ) {
          candidates.add(key);
        }
      }
    }

    // Evaluate each candidate
    for (const cellStr of candidates) {
      cuts.add(cellStr);
      const testBurned = simulateFire(grid, width, height, ignitions, cuts);
      const testValue = calculateSavedValue(towns, testBurned);

      if (testValue > bestValue) {
        bestValue = testValue;
        const [x, y] = cellStr.split(',').map(Number);
        bestCell = [x, y];
      }

      cuts.delete(cellStr);
    }

    if (bestCell === null) break;

    cuts.add(`${bestCell[0]},${bestCell[1]}`);
  }

  return Array.from(cuts).map((cellStr) => {
    const [x, y] = cellStr.split(',').map(Number);
    return [x, y];
  });
}
