#include <iostream>
#include <vector>
#include <algorithm>

struct Rect {
    int w, h, id;
};

struct PlacedRect {
    int x, y, w, h;
};

struct Board {
    int width, height;
    std::vector<PlacedRect> placed;

    Board(int w, int h) : width(w), height(h) {}

    bool overlaps(int x, int y, int w, int h) const {
        for (const auto& r : placed) {
            if (!(x + w <= r.x || x >= r.x + r.w ||
                  y + h <= r.y || y >= r.y + r.h)) {
                return true;
            }
        }
        return false;
    }

    bool canPlace(int x, int y, int w, int h) const {
        return x >= 0 && y >= 0 && x + w <= width && y + h <= height &&
               !overlaps(x, y, w, h);
    }

    bool tryPlace(int w, int h, int& outX, int& outY, bool useBrute) {
        if (useBrute && (long long)width * height <= 500 * 500) {
            // Brute force for small boards
            for (int y = 0; y <= height - h; y++) {
                for (int x = 0; x <= width - w; x++) {
                    if (canPlace(x, y, w, h)) {
                        placed.push_back({x, y, w, h});
                        outX = x;
                        outY = y;
                        return true;
                    }
                }
            }
        } else {
            // Heuristic: try positions adjacent to placed rectangles
            std::vector<std::pair<int, int>> candidates;
            candidates.push_back({0, 0});

            for (const auto& r : placed) {
                if (r.x + r.w <= width) candidates.push_back({r.x + r.w, r.y});
                if (r.y + r.h <= height) candidates.push_back({r.x, r.y + r.h});
            }

            // Try candidate positions
            for (auto [x, y] : candidates) {
                if (canPlace(x, y, w, h)) {
                    placed.push_back({x, y, w, h});
                    outX = x;
                    outY = y;
                    return true;
                }
            }
        }
        return false;
    }
};

int main() {
    int n, boardW, boardH;
    std::cin >> n >> boardW >> boardH;

    std::vector<Rect> rects(n);
    for (int i = 0; i < n; i++) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }

    // Sort by area descending (larger rectangles first)
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return (long long)a.w * a.h > (long long)b.w * b.h;
    });

    bool useBruteForce = n <= 100;

    std::vector<Board> boards;
    std::vector<std::array<int, 4>> placements(n); // {board, x, y, rotated}

    for (const auto& rect : rects) {
        bool placed = false;

        // Try to place in existing boards
        for (int b = 0; b < boards.size() && !placed; b++) {
            int x, y;

            // Try without rotation
            if (boards[b].tryPlace(rect.w, rect.h, x, y, useBruteForce)) {
                placements[rect.id] = {b, x, y, 0};
                placed = true;
            }
            // Try with rotation
            else if (rect.w != rect.h && boards[b].tryPlace(rect.h, rect.w, x, y, useBruteForce)) {
                placements[rect.id] = {b, x, y, 1};
                placed = true;
            }
        }

        // Create new board if needed
        if (!placed) {
            boards.emplace_back(boardW, boardH);
            int b = boards.size() - 1;
            int x, y;

            if (boards[b].tryPlace(rect.w, rect.h, x, y, useBruteForce)) {
                placements[rect.id] = {b, x, y, 0};
            } else if (rect.w != rect.h && boards[b].tryPlace(rect.h, rect.w, x, y, useBruteForce)) {
                placements[rect.id] = {b, x, y, 1};
            }
        }
    }

    // Output results
    std::cout << boards.size() << "\n";
    for (int i = 0; i < n; i++) {
        auto p = placements[i];
        std::cout << p[0] << " " << p[1] << " " << p[2] << " " << p[3] << "\n";
    }

    return 0;
}
