#!/usr/bin/env python3

def hex_word(val):
    """Convert 32-bit value to 8-char hex string (little-endian)."""
    bytes_val = [(val >> (8 * i)) & 0xFF for i in range(4)]
    return ''.join(f'{b:02x}' for b in bytes_val)

class SPIRVShader:
    def __init__(self):
        self.words = []
        self.next_id = 1

    def new_id(self):
        id_val = self.next_id
        self.next_id += 1
        return id_val

    def add_word(self, val):
        self.words.append(hex_word(val))

    def add_string(self, s):
        """Add null-terminated string as SPIR-V words."""
        s_bytes = (s + '\0').encode('utf-8')
        # Pad to 4-byte boundary
        while len(s_bytes) % 4 != 0:
            s_bytes += b'\0'
        for i in range(0, len(s_bytes), 4):
            w = int.from_bytes(s_bytes[i:i+4], 'little')
            self.add_word(w)

    def add_instruction(self, opcode, *operands):
        """Add instruction with word count and opcode."""
        word_count = len(operands) + 1
        self.add_word((word_count << 16) | opcode)
        for op in operands:
            self.add_word(op)

    def get_hex(self):
        return ''.join(self.words)

# Create shader builder
shader = SPIRVShader()

# Allocate IDs
ID_GLSL_EXT = shader.new_id()       # 1
ID_VOID = shader.new_id()            # 2
ID_FLOAT = shader.new_id()           # 3
ID_INT = shader.new_id()             # 4
ID_VEC3 = shader.new_id()            # 5
ID_VEC4 = shader.new_id()            # 6
ID_MAT4 = shader.new_id()            # 7
ID_STRUCT = shader.new_id()          # 8
ID_PTR_UNI_STRUCT = shader.new_id()  # 9
ID_PTR_IN_VEC3 = shader.new_id()     # 10
ID_PTR_OUT_VEC4 = shader.new_id()    # 11
ID_PTR_UNI_VEC4 = shader.new_id()    # 12
ID_PTR_UNI_FLOAT = shader.new_id()   # 13
ID_FUNC_TYPE = shader.new_id()       # 14
ID_UNIFORMS = shader.new_id()        # 15
ID_IN_NORMAL = shader.new_id()       # 16
ID_IN_WORLDPOS = shader.new_id()     # 17
ID_OUT_COLOR = shader.new_id()       # 18
ID_MAIN = shader.new_id()            # 19
ID_LABEL = shader.new_id()           # 20

# Temporaries
ID_NORMAL = shader.new_id()          # 21
ID_WORLDPOS = shader.new_id()        # 22
ID_LIGHTPOS_PTR = shader.new_id()    # 23
ID_LIGHTPOS = shader.new_id()        # 24
ID_LIGHTPOS_XYZ = shader.new_id()    # 25
ID_L_UNNORM = shader.new_id()        # 26
ID_L = shader.new_id()               # 27
ID_N = shader.new_id()               # 28
ID_DOT_RESULT = shader.new_id()      # 29
ID_CONST_ZERO = shader.new_id()      # 30
ID_D = shader.new_id()               # 31
ID_CONST_ONE = shader.new_id()       # 32
ID_OUTPUT = shader.new_id()          # 33

BOUND = 40

# Header
shader.add_word(0x07230203)          # Magic number
shader.add_word(0x00010000)          # Version 1.0
shader.add_word(0x00000000)          # Generator
shader.add_word(BOUND)               # Bound
shader.add_word(0x00000000)          # Schema

# OpCapability Shader (17)
shader.add_instruction(17, 1)

# OpExtInstImport %1 "GLSL.std.450" (11)
word_count = 1 + 1 + 3  # opcode + result_id + 3 words for string
shader.add_word((word_count << 16) | 11)
shader.add_word(ID_GLSL_EXT)
shader.add_string("GLSL.std.450")

# OpMemoryModel (14): Logical GLSL450
shader.add_instruction(14, 0, 1)

# OpEntryPoint Fragment %19 "main" %16 %17 %18 (15)
word_count = 1 + 1 + 1 + 1 + 3  # opcode + exec_model + entry_id + string + 3 interface IDs
shader.add_word((word_count << 16) | 15)
shader.add_word(4)  # Fragment execution model
shader.add_word(ID_MAIN)
shader.add_string("main")
shader.add_word(ID_IN_NORMAL)
shader.add_word(ID_IN_WORLDPOS)
shader.add_word(ID_OUT_COLOR)

# OpExecutionMode %19 OriginUpperLeft (16)
shader.add_instruction(16, ID_MAIN, 7)

# Decorations

# OpDecorate %15 DescriptorSet 0 (71)
shader.add_instruction(71, ID_UNIFORMS, 34, 0)

# OpDecorate %15 Binding 0 (71)
shader.add_instruction(71, ID_UNIFORMS, 33, 0)

# OpDecorate %8 Block (71)
shader.add_instruction(71, ID_STRUCT, 2)

# OpMemberDecorate %8 0 Offset 0 (72)
shader.add_instruction(72, ID_STRUCT, 0, 35, 0)

# OpMemberDecorate %8 1 Offset 64 (72)
shader.add_instruction(72, ID_STRUCT, 1, 35, 64)

# OpMemberDecorate %8 2 Offset 128 (72)
shader.add_instruction(72, ID_STRUCT, 2, 35, 128)

# OpMemberDecorate %8 3 Offset 192 (72)
shader.add_instruction(72, ID_STRUCT, 3, 35, 192)

# OpMemberDecorate %8 4 Offset 208 (72)
shader.add_instruction(72, ID_STRUCT, 4, 35, 208)

# OpMemberDecorate %8 5 Offset 224 (72)
shader.add_instruction(72, ID_STRUCT, 5, 35, 224)

# OpDecorate %16 Location 1 (71)
shader.add_instruction(71, ID_IN_NORMAL, 30, 1)

# OpDecorate %17 Location 0 (71)
shader.add_instruction(71, ID_IN_WORLDPOS, 30, 0)

# OpDecorate %18 Location 0 (71)
shader.add_instruction(71, ID_OUT_COLOR, 30, 0)

# Type definitions

# OpTypeVoid %2 (19)
shader.add_instruction(19, ID_VOID)

# OpTypeFloat %3 32 (22)
shader.add_instruction(22, ID_FLOAT, 32)

# OpTypeInt %4 32 1 (21)
shader.add_instruction(21, ID_INT, 32, 1)

# OpTypeVector %5 %3 3 (23)
shader.add_instruction(23, ID_VEC3, ID_FLOAT, 3)

# OpTypeVector %6 %3 4 (23)
shader.add_instruction(23, ID_VEC4, ID_FLOAT, 4)

# OpTypeMatrix %7 %6 4 (24)
shader.add_instruction(24, ID_MAT4, ID_VEC4, 4)

# OpTypeStruct %8 %7 %7 %7 %6 %6 %6 (30)
shader.add_instruction(30, ID_STRUCT, ID_MAT4, ID_MAT4, ID_MAT4, ID_VEC4, ID_VEC4, ID_VEC4)

# OpTypePointer %9 Uniform %8 (32)
shader.add_instruction(32, ID_PTR_UNI_STRUCT, 0, ID_STRUCT)

# OpTypePointer %10 Input %5 (32)
shader.add_instruction(32, ID_PTR_IN_VEC3, 1, ID_VEC3)

# OpTypePointer %11 Output %6 (32)
shader.add_instruction(32, ID_PTR_OUT_VEC4, 3, ID_VEC4)

# OpTypePointer %12 Uniform %6 (32)
shader.add_instruction(32, ID_PTR_UNI_VEC4, 0, ID_VEC4)

# OpTypePointer %13 Uniform %3 (32)
shader.add_instruction(32, ID_PTR_UNI_FLOAT, 0, ID_FLOAT)

# OpTypeFunction %14 %2 (33)
shader.add_instruction(33, ID_FUNC_TYPE, ID_VOID)

# Variable declarations

# OpVariable %9 Uniform %15 (59)
shader.add_instruction(59, ID_UNIFORMS, ID_PTR_UNI_STRUCT, 0)

# OpVariable %10 Input %16 (59)
shader.add_instruction(59, ID_IN_NORMAL, ID_PTR_IN_VEC3, 1)

# OpVariable %10 Input %17 (59)
shader.add_instruction(59, ID_IN_WORLDPOS, ID_PTR_IN_VEC3, 1)

# OpVariable %11 Output %18 (59)
shader.add_instruction(59, ID_OUT_COLOR, ID_PTR_OUT_VEC4, 3)

# Function definition

# OpFunction %2 %19 None %14 (54)
shader.add_instruction(54, ID_VOID, ID_MAIN, 0, ID_FUNC_TYPE)

# OpLabel %20 (248)
shader.add_instruction(248, ID_LABEL)

# Load inputs
# %21 = OpLoad %5 %16 (61)
shader.add_instruction(61, ID_NORMAL, ID_VEC3, ID_IN_NORMAL)

# %22 = OpLoad %5 %17 (61)
shader.add_instruction(61, ID_WORLDPOS, ID_VEC3, ID_IN_WORLDPOS)

# Access lightPos from UBO
# %23 = OpAccessChain %12 %15 %constant_3 (65)
shader.add_instruction(65, ID_LIGHTPOS_PTR, ID_PTR_UNI_VEC4, ID_UNIFORMS, 3)

# %24 = OpLoad %6 %23 (61)
shader.add_instruction(61, ID_LIGHTPOS, ID_VEC4, ID_LIGHTPOS_PTR)

# Extract xyz from lightPos
# %25 = OpVectorShuffle %5 %24 %24 0 1 2 (79)
shader.add_instruction(79, ID_LIGHTPOS_XYZ, ID_VEC3, ID_LIGHTPOS, ID_LIGHTPOS, 0, 1, 2)

# Compute L = lightPos.xyz - worldPos
# %26 = OpFSub %5 %25 %22 (131)
shader.add_instruction(131, ID_L_UNNORM, ID_VEC3, ID_LIGHTPOS_XYZ, ID_WORLDPOS)

# Normalize L
# %27 = OpExtInst %5 %1 Normalize %26 (12)
shader.add_instruction(12, ID_L, ID_VEC3, ID_GLSL_EXT, 67, ID_L_UNNORM)

# Normalize N
# %28 = OpExtInst %5 %1 Normalize %21 (12)
shader.add_instruction(12, ID_N, ID_VEC3, ID_GLSL_EXT, 67, ID_NORMAL)

# Compute dot(N, L)
# %29 = OpDot %3 %28 %27 (148)
shader.add_instruction(148, ID_DOT_RESULT, ID_FLOAT, ID_N, ID_L)

# %30 = OpConstant %3 0.0 (43)
shader.add_instruction(43, ID_CONST_ZERO, ID_FLOAT, 0)

# %31 = OpExtInst %3 %1 FMax %29 %30 (12)
shader.add_instruction(12, ID_D, ID_FLOAT, ID_GLSL_EXT, 40, ID_DOT_RESULT, ID_CONST_ZERO)

# %32 = OpConstant %3 1.0 (43)
shader.add_instruction(43, ID_CONST_ONE, ID_FLOAT, 0x3F800000)

# Create output vector
# %33 = OpCompositeConstruct %6 %31 %31 %31 %32 (80)
shader.add_instruction(80, ID_OUTPUT, ID_VEC4, ID_D, ID_D, ID_D, ID_CONST_ONE)

# Store output
# OpStore %18 %33 (62)
shader.add_instruction(62, ID_OUT_COLOR, ID_OUTPUT)

# OpReturn (253)
shader.add_instruction(253)

# OpFunctionEnd (56)
shader.add_instruction(56)

# Output the hex string
print(shader.get_hex())
