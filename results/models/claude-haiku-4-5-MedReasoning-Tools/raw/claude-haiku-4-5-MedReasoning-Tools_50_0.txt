#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (idx >= N) return;

    // Read input boid data
    vec3 pos = inData[idx * 2].xyz;
    vec3 vel = inData[idx * 2 + 1].xyz;

    // Extract parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    float boundSize = bounds.x;
    float boundForce = bounds.y;

    // Accumulate flocking forces
    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);
    int sepCount = 0, alignCount = 0, cohCount = 0;

    // Evaluate all other boids
    for (uint j = 0; j < N; ++j) {
        if (j == idx) continue;

        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;
        vec3 diff = otherPos - pos;
        float dist = length(diff);

        if (dist < 0.001) continue;

        // Separation: steer away from neighbors
        if (dist < sepRadius) {
            sepForce += normalize(-diff);
            sepCount++;
        }
        // Alignment: steer toward average heading
        if (dist < alignRadius) {
            alignForce += otherVel;
            alignCount++;
        }
        // Cohesion: steer toward average position
        if (dist < cohRadius) {
            cohForce += otherPos;
            cohCount++;
        }
    }

    // Compute acceleration from flocking rules
    vec3 accel = vec3(0.0);

    if (sepCount > 0) {
        accel += (sepForce / float(sepCount)) * sepWeight;
    }
    if (alignCount > 0) {
        accel += (alignForce / float(alignCount)) * alignWeight;
    }
    if (cohCount > 0) {
        vec3 cohCenter = cohForce / float(cohCount);
        accel += (cohCenter - pos) * cohWeight;
    }

    // Boundary restoring force
    float boundThreshold = boundSize * 0.4;
    float boundScale = boundSize * 0.5;

    if (abs(pos.x) > boundThreshold) {
        float distBeyond = abs(pos.x) - boundThreshold;
        accel.x -= sign(pos.x) * boundForce * (distBeyond / boundScale);
    }
    if (abs(pos.y) > boundThreshold) {
        float distBeyond = abs(pos.y) - boundThreshold;
        accel.y -= sign(pos.y) * boundForce * (distBeyond / boundScale);
    }
    if (abs(pos.z) > boundThreshold) {
        float distBeyond = abs(pos.z) - boundThreshold;
        accel.z -= sign(pos.z) * boundForce * (distBeyond / boundScale);
    }

    // Update velocity with acceleration
    vel += accel * dt;

    // Clamp velocity magnitude
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    // Update position
    pos += vel * dt;

    // Write output
    outData[idx * 2] = vec4(pos, 0.0);
    outData[idx * 2 + 1] = vec4(vel, 0.0);
}
