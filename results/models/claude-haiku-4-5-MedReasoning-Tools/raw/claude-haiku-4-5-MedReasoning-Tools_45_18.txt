#!/usr/bin/env python3
"""Generate SPIR-V fragment shader for NdotV heatmap visualization."""

import struct

def f32_to_bits(f):
    """Convert float to IEEE 754 32-bit representation (little-endian)."""
    return struct.unpack('<I', struct.pack('<f', f))[0]

class SPIRVGenerator:
    def __init__(self):
        self.words = []
        self.id_counter = 0

    def next_id(self):
        """Allocate a new ID."""
        self.id_counter += 1
        return self.id_counter

    def emit(self, opcode, operands=None):
        """Emit a SPIR-V instruction."""
        if operands is None:
            operands = []
        word_count = 1 + len(operands)
        header = (word_count << 16) | opcode
        self.words.append(header)
        self.words.extend(operands)

    def emit_string(self, s):
        """Emit a null-terminated string."""
        data = s.encode('utf-8') + b'\x00'
        # Pad to 4-byte boundary
        while len(data) % 4 != 0:
            data += b'\x00'
        for i in range(0, len(data), 4):
            word = int.from_bytes(data[i:i+4], 'little')
            self.words.append(word)

    def to_hex(self):
        """Convert to hex string."""
        return ''.join(f'{word & 0xFFFFFFFF:08x}' for word in self.words)

def generate():
    gen = SPIRVGenerator()

    # Header
    gen.words.append(0x07230203)  # Magic number
    gen.words.append(0x00010000)  # Version 1.0
    gen.words.append(0x00000000)  # Generator (0 = unknown)
    bound_idx = len(gen.words)
    gen.words.append(0)  # Placeholder for bound (max ID + 1)
    gen.words.append(0x00000000)  # Schema (reserved)

    # Allocate IDs upfront
    glsl_import = gen.next_id()  # 1
    void_type = gen.next_id()  # 2
    float_type = gen.next_id()  # 3
    vec3_type = gen.next_id()  # 4
    vec4_type = gen.next_id()  # 5
    mat4_type = gen.next_id()  # 6
    ubo_struct = gen.next_id()  # 7
    ptr_vec3_in = gen.next_id()  # 8
    ptr_vec4_out = gen.next_id()  # 9
    ptr_vec4_uniform = gen.next_id()  # 10
    ptr_struct_uniform = gen.next_id()  # 11
    bool_type = gen.next_id()  # 12
    int_type = gen.next_id()  # 13
    func_type = gen.next_id()  # 14
    const_0 = gen.next_id()  # 15
    const_half = gen.next_id()  # 16
    const_1 = gen.next_id()  # 17
    const_2 = gen.next_id()  # 18
    const_4_int = gen.next_id()  # 19
    main_func = gen.next_id()  # 20
    var_normal = gen.next_id()  # 21
    var_worldpos = gen.next_id()  # 22
    var_color = gen.next_id()  # 23
    var_outcolor = gen.next_id()  # 24
    var_ubo = gen.next_id()  # 25

    # OpCapability Shader
    gen.emit(17, [1])

    # OpExtInstImport
    gen.emit(11, [glsl_import])
    gen.emit_string("GLSL.std.450")

    # OpMemoryModel Logical GLSL450
    gen.emit(14, [0, 1])

    # OpEntryPoint Fragment %main "main" interfaces
    gen.emit(15, [4, main_func])
    gen.emit_string("main")
    gen.words.extend([var_normal, var_worldpos, var_color, var_outcolor])

    # OpExecutionMode %main OriginUpperLeft
    gen.emit(16, [main_func, 7])

    # OpDecorate for inputs and outputs
    gen.emit(71, [var_normal, 30, 1])  # Location 1
    gen.emit(71, [var_worldpos, 30, 0])  # Location 0
    gen.emit(71, [var_color, 30, 4])  # Location 4
    gen.emit(71, [var_outcolor, 30, 0])  # Location 0

    # OpTypeVoid
    gen.emit(19, [void_type])

    # OpTypeFloat 32
    gen.emit(22, [float_type, 32])

    # OpTypeInt 32 0
    gen.emit(21, [int_type, 32, 0])

    # OpTypeVector %vec3 %float 3
    gen.emit(23, [vec3_type, float_type, 3])

    # OpTypeVector %vec4 %float 4
    gen.emit(23, [vec4_type, float_type, 4])

    # OpTypeMatrix %mat4 %vec4 4
    gen.emit(24, [mat4_type, vec4_type, 4])

    # OpTypeStruct for UBO
    gen.emit(30, [ubo_struct, mat4_type, mat4_type, mat4_type, vec4_type, vec4_type, vec4_type])

    # OpTypePointer Input %vec3
    gen.emit(32, [ptr_vec3_in, 1, vec3_type])

    # OpTypePointer Output %vec4
    gen.emit(32, [ptr_vec4_out, 3, vec4_type])

    # OpTypePointer Uniform %vec4 (for AccessChain result)
    gen.emit(32, [ptr_vec4_uniform, 0, vec4_type])

    # OpTypePointer Uniform %ubo_struct
    gen.emit(32, [ptr_struct_uniform, 0, ubo_struct])

    # OpTypeBool
    gen.emit(20, [bool_type])

    # OpTypeFunction %func_type %void
    gen.emit(33, [func_type, void_type])

    # OpConstant %float 0.0
    gen.emit(43, [const_0, float_type, f32_to_bits(0.0)])

    # OpConstant %float 0.5
    gen.emit(43, [const_half, float_type, f32_to_bits(0.5)])

    # OpConstant %float 1.0
    gen.emit(43, [const_1, float_type, f32_to_bits(1.0)])

    # OpConstant %float 2.0
    gen.emit(43, [const_2, float_type, f32_to_bits(2.0)])

    # OpConstant %int 4 (for struct field index)
    gen.emit(43, [const_4_int, int_type, 4])

    # OpVariable for inputs
    gen.emit(59, [var_normal, ptr_vec3_in, 1])  # Input storage class
    gen.emit(59, [var_worldpos, ptr_vec3_in, 1])  # Input storage class
    gen.emit(59, [var_color, ptr_vec3_in, 1])  # Input storage class

    # OpVariable for output
    gen.emit(59, [var_outcolor, ptr_vec4_out, 3])  # Output storage class

    # OpVariable for UBO
    gen.emit(59, [var_ubo, ptr_struct_uniform, 0])  # Uniform storage class

    # Function definition
    gen.emit(54, [main_func, void_type, 0, func_type])  # OpFunction

    # Entry label
    entry_label = gen.next_id()
    gen.emit(248, [entry_label])  # OpLabel

    # Load inputs
    load_normal = gen.next_id()
    gen.emit(61, [vec3_type, load_normal, var_normal])

    load_worldpos = gen.next_id()
    gen.emit(61, [vec3_type, load_worldpos, var_worldpos])

    # Access cameraPos from UBO (field 4, offset 208 bytes)
    camera_ptr = gen.next_id()
    gen.emit(65, [ptr_vec4_uniform, camera_ptr, var_ubo, const_4_int])  # OpAccessChain

    camera_pos_vec4 = gen.next_id()
    gen.emit(61, [vec4_type, camera_pos_vec4, camera_ptr])

    # Extract xyz components from camera position vec4
    camera_pos_vec3 = gen.next_id()
    gen.emit(79, [vec3_type, camera_pos_vec3, camera_pos_vec4, camera_pos_vec4, 0, 1, 2])  # OpVectorShuffle

    # Compute view direction: V = normalize(cameraPos.xyz - worldPos)
    view_diff = gen.next_id()
    gen.emit(131, [vec3_type, view_diff, camera_pos_vec3, load_worldpos])  # OpFSub

    view_normalized = gen.next_id()
    gen.emit(12, [vec3_type, view_normalized, glsl_import, 32, view_diff])  # OpExtInst Normalize (GLSL.std.450 #32)

    # Compute dot(normal, view)
    dot_result = gen.next_id()
    gen.emit(148, [float_type, dot_result, load_normal, view_normalized])  # OpDot

    # Compute NdotV = max(dot, 0)
    ndotv = gen.next_id()
    gen.emit(12, [float_type, ndotv, glsl_import, 40, dot_result, const_0])  # OpExtInst FMax (GLSL.std.450 #40)

    # Compare: NdotV < 0.5
    is_less = gen.next_id()
    gen.emit(52, [bool_type, is_less, ndotv, const_half])  # OpFOrdLessThan

    # Conditional branch
    then_label = gen.next_id()
    else_label = gen.next_id()
    merge_label = gen.next_id()
    gen.emit(247, [is_less, then_label, else_label, merge_label])  # OpBranchConditional

    # Then branch: NdotV < 0.5, color = (1, 2*NdotV, 0, 1)
    gen.emit(248, [then_label])

    g_then = gen.next_id()
    gen.emit(133, [float_type, g_then, const_2, ndotv])  # OpFMul

    color_then = gen.next_id()
    gen.emit(80, [vec4_type, color_then, const_1, g_then, const_0, const_1])  # OpCompositeConstruct

    gen.emit(246, [merge_label])  # OpBranch

    # Else branch: NdotV >= 0.5, color = (1-temp, 1-temp, temp, 1) where temp = (NdotV-0.5)*2
    gen.emit(248, [else_label])

    diff_else = gen.next_id()
    gen.emit(131, [float_type, diff_else, ndotv, const_half])  # OpFSub

    temp_else = gen.next_id()
    gen.emit(133, [float_type, temp_else, diff_else, const_2])  # OpFMul

    one_minus_temp = gen.next_id()
    gen.emit(131, [float_type, one_minus_temp, const_1, temp_else])  # OpFSub

    color_else = gen.next_id()
    gen.emit(80, [vec4_type, color_else, one_minus_temp, one_minus_temp, temp_else, const_1])  # OpCompositeConstruct

    gen.emit(246, [merge_label])  # OpBranch

    # Merge block
    gen.emit(248, [merge_label])

    color_final = gen.next_id()
    gen.emit(245, [vec4_type, color_final, color_then, then_label, color_else, else_label])  # OpPhi

    # Store final color to output
    gen.emit(62, [var_outcolor, color_final])  # OpStore

    # Return
    gen.emit(253, [])  # OpReturn

    # End function
    gen.emit(56, [])  # OpFunctionEnd

    # Update bound
    gen.words[bound_idx] = gen.id_counter + 1

    return gen.to_hex()

if __name__ == '__main__':
    hex_output = generate()
    # Write to answer file
    with open('answer.txt', 'w') as f:
        f.write(hex_output)
    print("SPIR-V shader generated successfully")
    print(f"Length: {len(hex_output)} characters ({len(hex_output)//2} bytes)")
