#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <cstring>
#include <random>
#include <map>

using namespace std;

struct Point {
    float x, y;
};

inline float dist2(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

vector<int> nearestNeighbor(const vector<Point>& cities) {
    int n = cities.size();
    vector<int> tour;
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;
    int current = 0;

    for (int i = 1; i < n; ++i) {
        int nearest = -1;
        float minDist = 1e20f;

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                float d = dist2(cities[current], cities[j]);
                if (d < minDist) {
                    minDist = d;
                    nearest = j;
                }
            }
        }

        tour.push_back(nearest);
        visited[nearest] = true;
        current = nearest;
    }

    return tour;
}

vector<int> nearestNeighborGridBased(const vector<Point>& cities) {
    int n = cities.size();

    // Find bounds
    float minX = cities[0].x, maxX = cities[0].x;
    float minY = cities[0].y, maxY = cities[0].y;

    for (const auto& p : cities) {
        minX = min(minX, p.x);
        maxX = max(maxX, p.x);
        minY = min(minY, p.y);
        maxY = max(maxY, p.y);
    }

    // Grid size: balance between granularity and memory
    int gridSize = max(10, (int)sqrt(n / 100.0f));
    float cellW = (maxX - minX + 1e-6f) / gridSize;
    float cellH = (maxY - minY + 1e-6f) / gridSize;

    // Build grid
    vector<vector<vector<int>>> grid(gridSize, vector<vector<int>>(gridSize));
    for (int i = 0; i < n; ++i) {
        int gx = (int)((cities[i].x - minX) / cellW);
        int gy = (int)((cities[i].y - minY) / cellH);
        gx = min(gx, gridSize - 1);
        gy = min(gy, gridSize - 1);
        grid[gx][gy].push_back(i);
    }

    vector<int> tour;
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;
    int current = 0;

    for (int i = 1; i < n; ++i) {
        int nearest = -1;
        float minDist = 1e20f;

        int cgx = (int)((cities[current].x - minX) / cellW);
        int cgy = (int)((cities[current].y - minY) / cellH);
        cgx = min(cgx, gridSize - 1);
        cgy = min(cgy, gridSize - 1);

        // Search in expanding radii with limit for very large N
        int maxRadius = (n > 1000000) ? 5 : gridSize;
        for (int radius = 0; radius < maxRadius; ++radius) {
            for (int gx = max(0, cgx - radius); gx <= min(gridSize - 1, cgx + radius); ++gx) {
                for (int gy = max(0, cgy - radius); gy <= min(gridSize - 1, cgy + radius); ++gy) {
                    if (radius == 0 || abs(gx - cgx) == radius || abs(gy - cgy) == radius) {
                        for (int j : grid[gx][gy]) {
                            if (!visited[j]) {
                                float d = dist2(cities[current], cities[j]);
                                if (d < minDist) {
                                    minDist = d;
                                    nearest = j;
                                }
                            }
                        }
                    }
                }
            }
            if (nearest != -1) break;
        }

        if (nearest == -1) {
            for (int j = 0; j < n; ++j) {
                if (!visited[j]) {
                    nearest = j;
                    break;
                }
            }
        }

        tour.push_back(nearest);
        visited[nearest] = true;
        current = nearest;
    }

    return tour;
}

void twoOpt(vector<int>& tour, const vector<Point>& cities, int maxIterations) {
    int n = tour.size();
    bool improved = true;
    int iterations = 0;

    while (improved && iterations < maxIterations) {
        improved = false;
        iterations++;

        for (int i = 0; i < n - 2; ++i) {
            for (int j = i + 2; j < n; ++j) {
                int a = tour[i];
                int b = tour[i + 1];
                int c = tour[j];
                int d = tour[(j + 1) % n];

                float oldDist = dist2(cities[a], cities[b]) + dist2(cities[c], cities[d]);
                float newDist = dist2(cities[a], cities[c]) + dist2(cities[b], cities[d]);

                if (newDist < oldDist) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<Point> cities(n);
    for (int i = 0; i < n; ++i) {
        cin >> cities[i].x >> cities[i].y;
    }

    vector<int> tour;

    if (n <= 1000) {
        // Small: Use full nearest neighbor + 2-opt
        tour = nearestNeighbor(cities);
        twoOpt(tour, cities, min(5, n / 50));
    } else if (n <= 100000) {
        // Medium: Use nearest neighbor + light 2-opt
        tour = nearestNeighbor(cities);
        twoOpt(tour, cities, 1);
    } else {
        // Large/Extreme: Use grid-based nearest neighbor
        tour = nearestNeighborGridBased(cities);
    }

    for (int i = 0; i < n; ++i) {
        if (i > 0) cout << " ";
        cout << tour[i];
    }
    cout << "\n";

    return 0;
}
