[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal : TEXCOORD1;
    [[vk::location(2)]] float2 uv : TEXCOORD2;
    [[vk::location(3)]] float3 tangent : TEXCOORD3;
    [[vk::location(4)]] float3 color : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0 {
    float c_real = input.uv.x * 3.0 - 2.0;
    float c_imag = input.uv.y * 2.0 - 1.0;

    float z_real = 0.0;
    float z_imag = 0.0;
    int iterations = 0;

    for (int i = 0; i < 20; i++) {
        float z_real_sq = z_real * z_real;
        float z_imag_sq = z_imag * z_imag;

        if (z_real_sq + z_imag_sq > 4.0) {
            break;
        }

        float z_real_new = z_real_sq - z_imag_sq + c_real;
        float z_imag_new = 2.0 * z_real * z_imag + c_imag;

        z_real = z_real_new;
        z_imag = z_imag_new;
        iterations++;
    }

    float t = float(iterations) / 20.0;
    return float4(t, t * 0.5, 1.0 - t, 1.0);
}
