const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { depot, truckCount, packages } = data;

const pkgById = new Map();
for (const pkg of packages) {
  pkgById.set(pkg.id, pkg);
}

function distance(p1, p2) {
  const dx = p1[0] - p2[0];
  const dy = p1[1] - p2[1];
  return Math.sqrt(dx * dx + dy * dy);
}

function routeDistance(route) {
  let dist = 0;
  let current = depot;
  for (const pkgId of route) {
    const pkg = pkgById.get(pkgId);
    dist += distance(current, [pkg.x, pkg.y]);
    current = [pkg.x, pkg.y];
  }
  dist += distance(current, depot);
  return dist;
}

function buildRoutes() {
  const routes = Array(truckCount).fill(null).map(() => []);
  const assigned = new Set();

  while (assigned.size < packages.length) {
    let best = { pkg: -1, truck: -1, dist: Infinity };

    // Find nearest unassigned package to any route endpoint
    for (let t = 0; t < truckCount; t++) {
      const route = routes[t];
      const current = route.length === 0
        ? depot
        : (() => {
            const pkg = pkgById.get(route[route.length - 1]);
            return [pkg.x, pkg.y];
          })();

      for (let i = 0; i < packages.length; i++) {
        if (!assigned.has(i)) {
          const pkg = packages[i];
          const d = distance(current, [pkg.x, pkg.y]);
          if (d < best.dist) {
            best = { pkg: i, truck: t, dist: d };
          }
        }
      }
    }

    if (best.pkg === -1) break;

    routes[best.truck].push(packages[best.pkg].id);
    assigned.add(best.pkg);
  }

  return routes.filter(r => r.length > 0);
}

function optimize2Opt(route) {
  if (route.length <= 2) return route;

  const maxIterations = packages.length > 2000 ? 3 : packages.length > 1000 ? 10 : packages.length > 500 ? 30 : 100;
  let improved = true;
  let iterations = 0;

  while (improved && iterations < maxIterations) {
    improved = false;
    iterations++;

    for (let i = 0; i < route.length - 1; i++) {
      for (let j = i + 2; j < route.length; j++) {
        const newRoute = [
          ...route.slice(0, i + 1),
          ...route.slice(i + 1, j + 1).reverse(),
          ...route.slice(j + 1)
        ];

        const oldDist = routeDistance(route);
        const newDist = routeDistance(newRoute);

        if (newDist < oldDist - 1e-9) {
          route = newRoute;
          improved = true;
          break;
        }
      }
      if (improved) break;
    }
  }

  return route;
}

const routes = buildRoutes();

// Optimize routes with 2-opt local search
for (let i = 0; i < routes.length; i++) {
  routes[i] = optimize2Opt(routes[i]);
}

console.log(JSON.stringify({ routes }));
