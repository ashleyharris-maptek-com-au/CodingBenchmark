#!/usr/bin/env python3
"""Generate SPIR-V binary for fragment shader with stripes and Lambertian shading"""

import struct

def u32(value):
    """Convert to unsigned 32-bit"""
    return value & 0xFFFFFFFF

def word_count_opcode(count, opcode):
    """Encode SPIR-V instruction header: (count << 16) | opcode"""
    return u32((count << 16) | opcode)

def emit_instruction(opcode, *args):
    """Generate SPIR-V instruction words"""
    words = [word_count_opcode(len(args) + 1, opcode)]
    for arg in args:
        if isinstance(arg, str):
            # Null-terminated string padded to 4-byte boundary
            encoded = (arg + '\0').encode('utf-8')
            # Pad to multiple of 4 bytes
            encoded = encoded.ljust((len(encoded) + 3) // 4 * 4, b'\0')
            for i in range(0, len(encoded), 4):
                word = struct.unpack('<I', encoded[i:i+4])[0]
                words.append(word)
        else:
            words.append(u32(arg))
    return words

def to_hex_string(words):
    """Convert list of 32-bit words to hex string (little-endian)"""
    hex_str = ''
    for word in words:
        hex_str += struct.pack('<I', word).hex()
    return hex_str

# Opcode constants
OP_CAPABILITY = 17
OP_EXT_INST_IMPORT = 11
OP_MEMORY_MODEL = 14
OP_ENTRY_POINT = 15
OP_EXECUTION_MODE = 16
OP_DECORATE = 71
OP_MEMBER_DECORATE = 72
OP_TYPE_VOID = 19
OP_TYPE_BOOL = 20
OP_TYPE_INT = 21
OP_TYPE_FLOAT = 22
OP_TYPE_VECTOR = 23
OP_TYPE_MATRIX = 24
OP_TYPE_ARRAY = 28
OP_TYPE_STRUCT = 30
OP_TYPE_POINTER = 32
OP_TYPE_FUNCTION = 33
OP_CONSTANT = 43
OP_CONSTANT_COMPOSITE = 44
OP_VARIABLE = 59
OP_LOAD = 61
OP_STORE = 62
OP_ACCESS_CHAIN = 65
OP_FUNCTION = 54
OP_FUNCTION_END = 56
OP_LABEL = 248
OP_RETURN = 253
OP_RETURN_VALUE = 254
OP_COMPOSITE_CONSTRUCT = 80
OP_COMPOSITE_EXTRACT = 81
OP_EXT_INST = 12
OP_DOT = 148
OP_FMUL = 133
OP_FADD = 129
OP_FSUB = 131
OP_FNEGATE = 127
OP_VEC_SHUFFLE = 79

# ID allocation
ID_COUNTER = 0

def next_id():
    global ID_COUNTER
    ID_COUNTER += 1
    return ID_COUNTER

# Start building SPIR-V
all_words = []

# Header
all_words.append(0x07230203)      # Magic
all_words.append(0x00010000)      # Version 1.0
all_words.append(0x00000000)      # Generator (manual)
BOUND_IDX = len(all_words)
all_words.append(0)               # Bound (will update)
all_words.append(0x00000000)      # Reserved

# Capabilities and extensions
all_words.extend(emit_instruction(OP_CAPABILITY, 1))  # Shader capability

ID_GLSL = next_id()
all_words.extend(emit_instruction(OP_EXT_INST_IMPORT, ID_GLSL, "GLSL.std.450"))

# Memory model: Logical GLSL450
all_words.extend(emit_instruction(OP_MEMORY_MODEL, 0, 1))

# Type definitions
ID_VOID = next_id()
all_words.extend(emit_instruction(OP_TYPE_VOID, ID_VOID))

ID_FLOAT = next_id()
all_words.extend(emit_instruction(OP_TYPE_FLOAT, ID_FLOAT, 32))

ID_VEC3 = next_id()
all_words.extend(emit_instruction(OP_TYPE_VECTOR, ID_VEC3, ID_FLOAT, 3))

ID_VEC2 = next_id()
all_words.extend(emit_instruction(OP_TYPE_VECTOR, ID_VEC2, ID_FLOAT, 2))

ID_VEC4 = next_id()
all_words.extend(emit_instruction(OP_TYPE_VECTOR, ID_VEC4, ID_FLOAT, 4))

# Matrix type for UBO
ID_VEC4_ARRAY = next_id()
ID_4_CONST = next_id()
all_words.extend(emit_instruction(OP_CONSTANT, ID_FLOAT, ID_4_CONST, 4))
all_words.extend(emit_instruction(OP_TYPE_ARRAY, ID_VEC4_ARRAY, ID_VEC4, ID_4_CONST))

ID_MAT4 = next_id()
all_words.extend(emit_instruction(OP_TYPE_MATRIX, ID_MAT4, ID_VEC4, 4))

# UBO struct type with decorations
ID_UBO_STRUCT = next_id()
all_words.extend(emit_instruction(OP_TYPE_STRUCT, ID_UBO_STRUCT, ID_MAT4, ID_MAT4, ID_MAT4, ID_VEC4, ID_VEC4, ID_VEC4))

ID_PTR_UBO = next_id()
all_words.extend(emit_instruction(OP_TYPE_POINTER, ID_PTR_UBO, 0, ID_UBO_STRUCT))  # 0=Uniform

# Input/Output pointer types
ID_PTR_IN_VEC3 = next_id()
all_words.extend(emit_instruction(OP_TYPE_POINTER, ID_PTR_IN_VEC3, 1, ID_VEC3))  # 1=Input

ID_PTR_IN_VEC2 = next_id()
all_words.extend(emit_instruction(OP_TYPE_POINTER, ID_PTR_IN_VEC2, 1, ID_VEC2))

ID_PTR_OUT_VEC4 = next_id()
all_words.extend(emit_instruction(OP_TYPE_POINTER, ID_PTR_OUT_VEC4, 3, ID_VEC4))  # 3=Output

# Function type
ID_FUNC_VOID = next_id()
all_words.extend(emit_instruction(OP_TYPE_FUNCTION, ID_FUNC_VOID, ID_VOID))

# Constants
ID_0 = next_id()
all_words.extend(emit_instruction(OP_CONSTANT, ID_FLOAT, ID_0, 0))

ID_1 = next_id()
all_words.extend(emit_instruction(OP_CONSTANT, ID_FLOAT, ID_1, 1065353216))  # 1.0 as IEEE float

ID_0_5 = next_id()
all_words.extend(emit_instruction(OP_CONSTANT, ID_FLOAT, ID_0_5, 1056964608))  # 0.5

ID_0_9 = next_id()
all_words.extend(emit_instruction(OP_CONSTANT, ID_FLOAT, ID_0_9, 1065353216 - 51380224))  # ~0.9

ID_0_1 = next_id()
all_words.extend(emit_instruction(OP_CONSTANT, ID_FLOAT, ID_0_1, 1036831949))  # ~0.1

ID_PI_16 = next_id()
all_words.extend(emit_instruction(OP_CONSTANT, ID_FLOAT, ID_PI_16, 1140457472))  # 16*pi

ID_16 = next_id()
all_words.extend(emit_instruction(OP_CONSTANT, ID_FLOAT, ID_16, 1098776576))  # 16.0

# Global variables
ID_WORLDPOS = next_id()
all_words.extend(emit_instruction(OP_VARIABLE, ID_PTR_IN_VEC3, ID_WORLDPOS, 1))

ID_NORMAL = next_id()
all_words.extend(emit_instruction(OP_VARIABLE, ID_PTR_IN_VEC3, ID_NORMAL, 1))

ID_UV = next_id()
all_words.extend(emit_instruction(OP_VARIABLE, ID_PTR_IN_VEC2, ID_UV, 1))

ID_OUT_COLOR = next_id()
all_words.extend(emit_instruction(OP_VARIABLE, ID_PTR_OUT_VEC4, ID_OUT_COLOR, 3))

ID_UBO = next_id()
all_words.extend(emit_instruction(OP_VARIABLE, ID_PTR_UBO, ID_UBO, 0))

# Decorations
all_words.extend(emit_instruction(OP_DECORATE, ID_WORLDPOS, 30, 0))  # Location 0
all_words.extend(emit_instruction(OP_DECORATE, ID_NORMAL, 30, 1))    # Location 1
all_words.extend(emit_instruction(OP_DECORATE, ID_UV, 30, 2))        # Location 2
all_words.extend(emit_instruction(OP_DECORATE, ID_OUT_COLOR, 30, 0)) # Location 0

all_words.extend(emit_instruction(OP_DECORATE, ID_UBO_STRUCT, 2))    # Block
all_words.extend(emit_instruction(OP_DECORATE, ID_UBO, 34, 0))       # DescriptorSet 0
all_words.extend(emit_instruction(OP_DECORATE, ID_UBO, 33, 0))       # Binding 0

# Entry point
ID_MAIN = next_id()
all_words.extend(emit_instruction(OP_ENTRY_POINT, 4, ID_MAIN, "main", ID_WORLDPOS, ID_NORMAL, ID_UV, ID_OUT_COLOR))

# Execution mode
all_words.extend(emit_instruction(OP_EXECUTION_MODE, ID_MAIN, 7))  # OriginUpperLeft

# Main function
all_words.extend(emit_instruction(OP_FUNCTION, ID_VOID, ID_MAIN, 0, ID_FUNC_VOID))

ID_ENTRY_BLOCK = next_id()
all_words.extend(emit_instruction(OP_LABEL, ID_ENTRY_BLOCK))

# Load inputs
ID_LOAD_UV = next_id()
all_words.extend(emit_instruction(OP_LOAD, ID_VEC2, ID_LOAD_UV, ID_UV))

ID_LOAD_NORMAL = next_id()
all_words.extend(emit_instruction(OP_LOAD, ID_VEC3, ID_LOAD_NORMAL, ID_NORMAL))

ID_LOAD_WORLDPOS = next_id()
all_words.extend(emit_instruction(OP_LOAD, ID_VEC3, ID_LOAD_WORLDPOS, ID_WORLDPOS))

# Extract UV.y: OpCompositeExtract
ID_UV_Y = next_id()
all_words.extend(emit_instruction(OP_COMPOSITE_EXTRACT, ID_FLOAT, ID_UV_Y, ID_LOAD_UV, 1))

# sin(uv.y * pi * 16)
ID_UV_Y_SCALED = next_id()
all_words.extend(emit_instruction(OP_FMUL, ID_FLOAT, ID_UV_Y_SCALED, ID_UV_Y, ID_PI_16))

# Call sin using OpExtInst
ID_SIN_RESULT = next_id()
all_words.extend(emit_instruction(OP_EXT_INST, ID_FLOAT, ID_SIN_RESULT, ID_GLSL, 13, ID_UV_Y_SCALED))  # 13=sin

# Convert sin result [-1,1] to [0,1]: result * 0.5 + 0.5
ID_SIN_SCALED = next_id()
all_words.extend(emit_instruction(OP_FMUL, ID_FLOAT, ID_SIN_SCALED, ID_SIN_RESULT, ID_0_5))

ID_SIN_01 = next_id()
all_words.extend(emit_instruction(OP_FADD, ID_FLOAT, ID_SIN_01, ID_SIN_SCALED, ID_0_5))

# Create color vectors: red (0.9, 0.1, 0.1) and blue (0.1, 0.1, 0.9)
ID_RED = next_id()
all_words.extend(emit_instruction(OP_COMPOSITE_CONSTRUCT, ID_VEC3, ID_RED, ID_0_9, ID_0_1, ID_0_1))

ID_BLUE = next_id()
all_words.extend(emit_instruction(OP_COMPOSITE_CONSTRUCT, ID_VEC3, ID_BLUE, ID_0_1, ID_0_1, ID_0_9))

# Mix between red and blue: mix(red, blue, sin_01)
ID_STRIPE_COLOR = next_id()
all_words.extend(emit_instruction(OP_EXT_INST, ID_VEC3, ID_STRIPE_COLOR, ID_GLSL, 46, ID_RED, ID_BLUE, ID_SIN_01))  # 46=mix

# Lambertian shading
# Load light position from UBO
ID_LIGHT_POS_CHAIN = next_id()
ID_3_CONST = next_id()
all_words.extend(emit_instruction(OP_CONSTANT, ID_FLOAT, ID_3_CONST, 3))
all_words.extend(emit_instruction(OP_ACCESS_CHAIN, ID_PTR_OUT_VEC4, ID_LIGHT_POS_CHAIN, ID_UBO, ID_3_CONST))

ID_LIGHT_POS = next_id()
all_words.extend(emit_instruction(OP_LOAD, ID_VEC4, ID_LIGHT_POS, ID_LIGHT_POS_CHAIN))

# Extract xyz from light position
ID_LIGHT_POS_XYZ = next_id()
all_words.extend(emit_instruction(OP_VEC_SHUFFLE, ID_VEC3, ID_LIGHT_POS_XYZ, ID_LIGHT_POS, ID_LIGHT_POS, 0, 1, 2))

# Calculate light direction: normalize(lightPos - worldPos)
ID_LIGHT_DIR_DIFF = next_id()
all_words.extend(emit_instruction(OP_FSUB, ID_VEC3, ID_LIGHT_DIR_DIFF, ID_LIGHT_POS_XYZ, ID_LOAD_WORLDPOS))

ID_LIGHT_DIR = next_id()
all_words.extend(emit_instruction(OP_EXT_INST, ID_VEC3, ID_LIGHT_DIR, ID_GLSL, 69, ID_LIGHT_DIR_DIFF))  # 69=normalize

# Diffuse: max(0, dot(normal, lightDir))
ID_DOT_RESULT = next_id()
all_words.extend(emit_instruction(OP_DOT, ID_FLOAT, ID_DOT_RESULT, ID_LOAD_NORMAL, ID_LIGHT_DIR))

ID_DIFFUSE = next_id()
all_words.extend(emit_instruction(OP_EXT_INST, ID_FLOAT, ID_DIFFUSE, ID_GLSL, 42, ID_0, ID_DOT_RESULT))  # 42=max

# Apply diffuse to stripe color: stripeColor * diffuse
ID_SHADED_COLOR = next_id()
all_words.extend(emit_instruction(OP_FMUL, ID_VEC3, ID_SHADED_COLOR, ID_STRIPE_COLOR, ID_DIFFUSE))

# Create output vec4: (shaded_color, 1.0)
ID_FINAL_COLOR = next_id()
all_words.extend(emit_instruction(OP_COMPOSITE_CONSTRUCT, ID_VEC4, ID_FINAL_COLOR, ID_SHADED_COLOR, ID_1))

# Store result
all_words.extend(emit_instruction(OP_STORE, ID_OUT_COLOR, ID_FINAL_COLOR))

# Return
all_words.extend(emit_instruction(OP_RETURN, ID_VOID))

# End function
all_words.extend(emit_instruction(OP_FUNCTION_END))

# Update bound
all_words[BOUND_IDX] = ID_COUNTER + 1

# Convert to hex string
hex_output = to_hex_string(all_words)
print(hex_output)
