using System;
using System.Collections.Generic;
using System.Linq;

class BinPacker3D
{
    static void Main()
    {
        var (polyVerts, polyFaces, containerW, containerH, containerD) = ParseInput();

        var solver = new Solver(containerW, containerH, containerD, polyVerts, polyFaces);
        var packed = solver.Pack();

        Console.WriteLine(packed.Count);
        foreach (var item in packed)
        {
            Console.WriteLine($"{item.tx} {item.ty} {item.tz} {item.qw} {item.qx} {item.qy} {item.qz}");
        }
    }

    static (List<Vector3>, List<(int,int,int)>, float, float, float) ParseInput()
    {
        var vf = Console.ReadLine().Split();
        int v = int.Parse(vf[0]);
        int f = int.Parse(vf[1]);

        var verts = new List<Vector3>();
        for (int i = 0; i < v; i++)
        {
            var xyz = Console.ReadLine().Split();
            verts.Add(new Vector3(float.Parse(xyz[0]), float.Parse(xyz[1]), float.Parse(xyz[2])));
        }

        var faces = new List<(int, int, int)>();
        for (int i = 0; i < f; i++)
        {
            var abc = Console.ReadLine().Split();
            faces.Add((int.Parse(abc[0]), int.Parse(abc[1]), int.Parse(abc[2])));
        }

        var whd = Console.ReadLine().Split();
        return (verts, faces, float.Parse(whd[0]), float.Parse(whd[1]), float.Parse(whd[2]));
    }
}

struct Vector3
{
    public float x, y, z;
    public Vector3(float x, float y, float z) { this.x = x; this.y = y; this.z = z; }
    public static Vector3 operator+(Vector3 a, Vector3 b) => new Vector3(a.x+b.x, a.y+b.y, a.z+b.z);
    public static Vector3 operator-(Vector3 a, Vector3 b) => new Vector3(a.x-b.x, a.y-b.y, a.z-b.z);
    public static Vector3 operator*(Vector3 a, float s) => new Vector3(a.x*s, a.y*s, a.z*s);
    public static float Dot(Vector3 a, Vector3 b) => a.x*b.x + a.y*b.y + a.z*b.z;
    public float Length => (float)Math.Sqrt(x*x + y*y + z*z);
}

struct Quaternion
{
    public float w, x, y, z;
    public Quaternion(float w, float x, float y, float z) { this.w = w; this.x = x; this.y = y; this.z = z; }
    public static Quaternion Identity => new Quaternion(1, 0, 0, 0);

    public Quaternion Conjugate => new Quaternion(w, -x, -y, -z);

    public static Quaternion operator*(Quaternion a, Quaternion b)
    {
        return new Quaternion(
            a.w*b.w - a.x*b.x - a.y*b.y - a.z*b.z,
            a.w*b.x + a.x*b.w + a.y*b.z - a.z*b.y,
            a.w*b.y - a.x*b.z + a.y*b.w + a.z*b.x,
            a.w*b.z + a.x*b.y - a.y*b.x + a.z*b.w
        );
    }

    public Vector3 RotateVector(Vector3 v)
    {
        var qv = new Quaternion(0, v.x, v.y, v.z);
        var result = this * qv * Conjugate;
        return new Vector3(result.x, result.y, result.z);
    }

    public static Quaternion FromAxisAngle(Vector3 axis, float angle)
    {
        float halfAngle = angle / 2;
        float sinHalf = (float)Math.Sin(halfAngle);
        return new Quaternion((float)Math.Cos(halfAngle), axis.x*sinHalf, axis.y*sinHalf, axis.z*sinHalf);
    }
}

struct AABB
{
    public float minX, minY, minZ, maxX, maxY, maxZ;

    public AABB(float minX, float minY, float minZ, float maxX, float maxY, float maxZ)
    {
        this.minX = minX; this.minY = minY; this.minZ = minZ;
        this.maxX = maxX; this.maxY = maxY; this.maxZ = maxZ;
    }

    public bool Intersects(AABB other)
    {
        return !(maxX < other.minX || minX > other.maxX ||
                 maxY < other.minY || minY > other.maxY ||
                 maxZ < other.minZ || minZ > other.maxZ);
    }

    public bool FitsInContainer(float w, float h, float d)
    {
        return minX >= 0 && maxX <= w && minY >= 0 && maxY <= h && minZ >= 0 && maxZ <= d;
    }
}

class Solver
{
    float containerW, containerH, containerD;
    List<Vector3> polyVerts;
    List<(int, int, int)> polyFaces;
    AABB polyAABB;
    List<PlacedItem> placed = new List<PlacedItem>();

    class PlacedItem
    {
        public float tx, ty, tz;
        public Quaternion q;
        public AABB aabb;
    }

    public Solver(float w, float h, float d, List<Vector3> verts, List<(int,int,int)> faces)
    {
        containerW = w;
        containerH = h;
        containerD = d;
        polyVerts = verts;
        polyFaces = faces;
        ComputeAABB();
    }

    void ComputeAABB()
    {
        float minX = float.MaxValue, minY = float.MaxValue, minZ = float.MaxValue;
        float maxX = float.MinValue, maxY = float.MinValue, maxZ = float.MinValue;

        foreach (var v in polyVerts)
        {
            minX = Math.Min(minX, v.x);
            minY = Math.Min(minY, v.y);
            minZ = Math.Min(minZ, v.z);
            maxX = Math.Max(maxX, v.x);
            maxY = Math.Max(maxY, v.y);
            maxZ = Math.Max(maxZ, v.z);
        }

        polyAABB = new AABB(minX, minY, minZ, maxX, maxY, maxZ);
    }

    AABB TransformAABB(float tx, float ty, float tz, Quaternion q)
    {
        float minX = float.MaxValue, minY = float.MaxValue, minZ = float.MaxValue;
        float maxX = float.MinValue, maxY = float.MinValue, maxZ = float.MinValue;

        foreach (var v in polyVerts)
        {
            var rotated = q.RotateVector(v);
            var transformed = new Vector3(rotated.x + tx, rotated.y + ty, rotated.z + tz);

            minX = Math.Min(minX, transformed.x);
            minY = Math.Min(minY, transformed.y);
            minZ = Math.Min(minZ, transformed.z);
            maxX = Math.Max(maxX, transformed.x);
            maxY = Math.Max(maxY, transformed.y);
            maxZ = Math.Max(maxZ, transformed.z);
        }

        return new AABB(minX, minY, minZ, maxX, maxY, maxZ);
    }

    bool CanPlace(float tx, float ty, float tz, Quaternion q)
    {
        var aabb = TransformAABB(tx, ty, tz, q);

        if (!aabb.FitsInContainer(containerW, containerH, containerD))
            return false;

        foreach (var item in placed)
        {
            if (aabb.Intersects(item.aabb))
                return false;
        }

        return true;
    }

    public List<(float tx, float ty, float tz, float qw, float qx, float qy, float qz)> Pack()
    {
        int maxAttempts = DetermineMaxAttempts();

        for (int attempt = 0; attempt < maxAttempts; attempt++)
        {
            if (!TryPlaceItem())
                break;
        }

        return placed.Select(p => (p.tx, p.ty, p.tz, p.q.w, p.q.x, p.q.y, p.q.z)).ToList();
    }

    int DetermineMaxAttempts()
    {
        float volume = containerW * containerH * containerD;
        float polySize = (polyAABB.maxX - polyAABB.minX) *
                        (polyAABB.maxY - polyAABB.minY) *
                        (polyAABB.maxZ - polyAABB.minZ);

        if (polySize <= 0) return 1;

        int estimate = (int)(volume / polySize * 0.6f);

        if (volume < 1000) return Math.Min(estimate, 50);
        if (volume < 10000) return Math.Min(estimate, 200);
        if (volume < 100000) return Math.Min(estimate, 500);
        return Math.Min(estimate, 2000);
    }

    bool TryPlaceItem()
    {
        var rotations = GenerateRotations();

        int gridStep = containerW > 100 ? 10 : containerW > 50 ? 5 : 2;

        for (float x = 0; x < containerW; x += gridStep)
        {
            for (float y = 0; y < containerH; y += gridStep)
            {
                for (float z = 0; z < containerD; z += gridStep)
                {
                    foreach (var q in rotations)
                    {
                        if (CanPlace(x, y, z, q))
                        {
                            var aabb = TransformAABB(x, y, z, q);
                            placed.Add(new PlacedItem { tx = x, ty = y, tz = z, q = q, aabb = aabb });
                            return true;
                        }
                    }
                }
            }
        }

        return false;
    }

    Quaternion[] GenerateRotations()
    {
        var rotations = new List<Quaternion>
        {
            Quaternion.Identity,
            Quaternion.FromAxisAngle(new Vector3(1,0,0), (float)Math.PI/2),
            Quaternion.FromAxisAngle(new Vector3(0,1,0), (float)Math.PI/2),
            Quaternion.FromAxisAngle(new Vector3(0,0,1), (float)Math.PI/2),
            Quaternion.FromAxisAngle(new Vector3(1,0,0), (float)Math.PI),
            Quaternion.FromAxisAngle(new Vector3(0,1,0), (float)Math.PI),
            Quaternion.FromAxisAngle(new Vector3(0,0,1), (float)Math.PI),
        };

        return rotations.ToArray();
    }
}
