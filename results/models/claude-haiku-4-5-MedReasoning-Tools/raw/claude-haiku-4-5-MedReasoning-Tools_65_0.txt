declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Task {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
}

const tasks: Task[] = data.tasks;
const taskMap = new Map(tasks.map(t => [t.id, t]));

// Calculate earliest finish time for each task
const eft = new Map<string, number>();

function calcEFT(taskId: string): number {
  if (eft.has(taskId)) return eft.get(taskId)!;

  const task = taskMap.get(taskId);
  if (!task) return 0;

  let est = 0;
  for (const dep of task.deps) {
    est = Math.max(est, calcEFT(dep));
  }

  const result = est + task.duration;
  eft.set(taskId, result);
  return result;
}

for (const task of tasks) {
  calcEFT(task.id);
}

// Topological sort with slack-based priority
const inDegree = new Map<string, number>(tasks.map(t => [t.id, t.deps.length]));
const outEdges = new Map<string, string[]>(tasks.map(t => [t.id, []]));

for (const task of tasks) {
  for (const dep of task.deps) {
    outEdges.get(dep)!.push(task.id);
  }
}

const ready = tasks.filter(t => t.deps.length === 0).map(t => t.id);
const order: string[] = [];

while (ready.length > 0) {
  // Sort by priority: slack first (lower = higher priority), then weight (higher = higher priority)
  ready.sort((a, b) => {
    const ta = taskMap.get(a)!;
    const tb = taskMap.get(b)!;

    const slackA = ta.deadline - eft.get(a)!;
    const slackB = tb.deadline - eft.get(b)!;

    if (slackA !== slackB) return slackA - slackB;
    return tb.weight - ta.weight;
  });

  const current = ready.shift()!;
  order.push(current);

  for (const next of outEdges.get(current)!) {
    inDegree.set(next, inDegree.get(next)! - 1);
    if (inDegree.get(next) === 0) {
      ready.push(next);
    }
  }
}
