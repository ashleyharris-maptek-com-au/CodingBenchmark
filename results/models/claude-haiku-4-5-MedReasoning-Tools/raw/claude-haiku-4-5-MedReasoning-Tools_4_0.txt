#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>
#include <chrono>

using namespace std;

struct Point {
    double x, y, z;
    Point() : x(0), y(0), z(0) {}
    Point(double x, double y, double z) : x(x), y(y), z(z) {}
    Point operator+(const Point& p) const { return Point(x+p.x, y+p.y, z+p.z); }
    Point operator-(const Point& p) const { return Point(x-p.x, y-p.y, z-p.z); }
    Point operator*(double s) const { return Point(x*s, y*s, z*s); }
    double dot(const Point& p) const { return x*p.x + y*p.y + z*p.z; }
    Point cross(const Point& p) const { return Point(y*p.z-z*p.y, z*p.x-x*p.z, x*p.y-y*p.x); }
    double length() const { return sqrt(x*x + y*y + z*z); }
    Point normalize() const { double l = length(); return l > 1e-9 ? Point(x/l, y/l, z/l) : Point(); }
};

struct Tetrahedron {
    Point center;
    double rx, ry, rz;
    Tetrahedron() : rx(0), ry(0), rz(0) {}
};

vector<Point> container_vertices;
vector<Tetrahedron> placed;
double edge_length_global;

Point rotate_point(const Point& p, double rx, double ry, double rz) {
    double sx = sin(rx), cx = cos(rx);
    double sy = sin(ry), cy = cos(ry);
    double sz = sin(rz), cz = cos(rz);

    double m[3][3] = {
        {cy*cz, -cy*sz, sy},
        {sx*sy*cz + cx*sz, -sx*sy*sz + cx*cz, -sx*cy},
        {-cx*sy*cz + sx*sz, cx*sy*sz + sx*cz, cx*cy}
    };

    return Point(
        m[0][0]*p.x + m[0][1]*p.y + m[0][2]*p.z,
        m[1][0]*p.x + m[1][1]*p.y + m[1][2]*p.z,
        m[2][0]*p.x + m[2][1]*p.y + m[2][2]*p.z
    );
}

vector<Point> get_tet_vertices(Point center, double rx, double ry, double rz) {
    double R = edge_length_global * sqrt(3.0/8.0);
    vector<Point> v = {
        Point(1, 1, 1), Point(1, -1, -1),
        Point(-1, 1, -1), Point(-1, -1, 1)
    };
    for (auto& p : v) {
        p = p.normalize() * R;
        p = rotate_point(p, rx, ry, rz);
        p = center + p;
    }
    return v;
}

bool point_in_box(const Point& p) {
    double minx = container_vertices[0].x, maxx = minx;
    double miny = container_vertices[0].y, maxy = miny;
    double minz = container_vertices[0].z, maxz = minz;

    for (auto& v : container_vertices) {
        minx = min(minx, v.x); maxx = max(maxx, v.x);
        miny = min(miny, v.y); maxy = max(maxy, v.y);
        minz = min(minz, v.z); maxz = max(maxz, v.z);
    }

    const double eps = 0.01;
    return p.x >= minx + eps && p.x <= maxx - eps &&
           p.y >= miny + eps && p.y <= maxy - eps &&
           p.z >= minz + eps && p.z <= maxz - eps;
}

bool can_place(Point center, double rx, double ry, double rz) {
    auto verts = get_tet_vertices(center, rx, ry, rz);
    for (auto& v : verts) {
        if (!point_in_box(v)) return false;
    }

    for (auto& p : placed) {
        double dist = (center - p.center).length();
        if (dist < edge_length_global * 1.3) return false;
    }

    return true;
}

int main() {
    int V;
    cin >> V >> edge_length_global;

    container_vertices.resize(V);
    for (int i = 0; i < V; i++) {
        cin >> container_vertices[i].x >> container_vertices[i].y >> container_vertices[i].z;
    }

    double minx = container_vertices[0].x, maxx = minx;
    double miny = container_vertices[0].y, maxy = miny;
    double minz = container_vertices[0].z, maxz = minz;
    for (auto& v : container_vertices) {
        minx = min(minx, v.x); maxx = max(maxx, v.x);
        miny = min(miny, v.y); maxy = max(maxy, v.y);
        minz = min(minz, v.z); maxz = max(maxz, v.z);
    }

    double spacing = edge_length_global * 1.5;
    auto start = chrono::steady_clock::now();

    for (double x = minx + spacing; x < maxx - spacing; x += spacing) {
        for (double y = miny + spacing; y < maxy - spacing; y += spacing) {
            for (double z = minz + spacing; z < maxz - spacing; z += spacing) {
                Point c(x, y, z);
                for (double rx = 0; rx < 2*M_PI; rx += M_PI/3) {
                    for (double ry = 0; ry < 2*M_PI; ry += M_PI/3) {
                        for (double rz = 0; rz < 2*M_PI; rz += M_PI/3) {
                            if (can_place(c, rx, ry, rz)) {
                                Tetrahedron t;
                                t.center = c;
                                t.rx = rx; t.ry = ry; t.rz = rz;
                                placed.push_back(t);
                                goto next_pos;
                            }
                        }
                    }
                }
                next_pos:;
                if (chrono::steady_clock::now() - start > chrono::seconds(28)) goto done;
            }
        }
    }

    done:
    cout << placed.size() << "\n";
    for (auto& t : placed) {
        printf("%.6g %.6g %.6g %.6g %.6g %.6g\n", t.center.x, t.center.y, t.center.z, t.rx, t.ry, t.rz);
    }

    return 0;
}
