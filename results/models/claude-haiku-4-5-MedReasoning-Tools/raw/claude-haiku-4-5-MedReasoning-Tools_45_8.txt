#!/usr/bin/env python3
"""Generate SPIR-V binary for toon shading fragment shader."""

import struct
import io

class SPIRVGenerator:
    def __init__(self):
        self.words = []
        self.id_counter = 1
        self.string_id_map = {}

    def gen_id(self):
        """Generate a new ID."""
        id_val = self.id_counter
        self.id_counter += 1
        return id_val

    def write_word(self, value):
        """Write a 32-bit word."""
        self.words.append(value & 0xFFFFFFFF)

    def write_words(self, *values):
        """Write multiple words."""
        for v in values:
            self.write_word(v)

    def encode_string(self, s):
        """Encode a null-terminated string as words."""
        data = s.encode('utf-8') + b'\x00'
        # Pad to 4-byte boundary
        padding = (4 - (len(data) % 4)) % 4
        data += b'\x00' * padding
        words = []
        for i in range(0, len(data), 4):
            w = struct.unpack('<I', data[i:i+4])[0]
            words.append(w)
        return words

    def instruction(self, opcode, *operands):
        """Write an instruction."""
        word_count = 1 + len(operands)
        header = (word_count << 16) | opcode
        self.write_word(header)
        for op in operands:
            if isinstance(op, str):
                for w in self.encode_string(op):
                    self.write_word(w)
            else:
                self.write_word(op)

    def float_bits(self, f):
        """Convert float to 32-bit unsigned int (little-endian)."""
        return struct.unpack('<I', struct.pack('<f', f))[0]

    def write_header(self, bound):
        """Write SPIR-V header."""
        self.write_words(
            0x07230203,  # Magic number (little-endian)
            0x00010000,  # Version 1.0
            0x00000000,  # Generator (0 = unknown)
            bound,        # Bound
            0x00000000   # Schema (reserved)
        )

    def get_binary(self):
        """Get the binary as bytes."""
        data = b''
        for w in self.words:
            data += struct.pack('<I', w)
        return data

    def get_hex(self):
        """Get the binary as hex string."""
        return self.get_binary().hex()

def generate_toon_shader():
    """Generate SPIR-V binary for toon shader."""
    gen = SPIRVGenerator()

    # IDs for different constructs
    id_glsl_ext = gen.gen_id()      # 1
    id_void = gen.gen_id()          # 2
    id_float = gen.gen_id()         # 3
    id_vec3 = gen.gen_id()          # 4
    id_vec4 = gen.gen_id()          # 5
    id_vec2 = gen.gen_id()          # 6
    id_mat4 = gen.gen_id()          # 7
    id_ubo_type = gen.gen_id()      # 8
    id_ptr_uniform_ubo = gen.gen_id()  # 9
    id_ptr_input_vec3_1 = gen.gen_id()   # 10
    id_ptr_input_vec3_2 = gen.gen_id()   # 11
    id_ptr_input_vec3_3 = gen.gen_id()   # 12
    id_ptr_output_vec4 = gen.gen_id()    # 13
    id_func_void = gen.gen_id()     # 14
    id_ubo_var = gen.gen_id()       # 15
    id_worldpos_var = gen.gen_id()  # 16
    id_normal_var = gen.gen_id()    # 17
    id_color_var = gen.gen_id()     # 18
    id_outcolor_var = gen.gen_id()  # 19
    id_main = gen.gen_id()          # 20
    id_entry_label = gen.gen_id()   # 21

    # Constants
    id_const_0 = gen.gen_id()       # 22
    id_const_1 = gen.gen_id()       # 23
    id_const_075 = gen.gen_id()     # 24
    id_const_05 = gen.gen_id()      # 25
    id_const_025 = gen.gen_id()     # 26
    id_const_07 = gen.gen_id()      # 27
    id_const_04 = gen.gen_id()      # 28
    id_const_02 = gen.gen_id()      # 29

    # Temp variables
    id_tmp_worldpos = gen.gen_id()  # 30
    id_tmp_normal = gen.gen_id()    # 31
    id_tmp_color = gen.gen_id()     # 32
    id_tmp_norm_normal = gen.gen_id()  # 33

    bound = 100
    gen.write_header(bound)

    # OpCapability Shader
    gen.instruction(17, 1)  # Capability::Shader = 1

    # OpExtInstImport
    gen.instruction(11, id_glsl_ext, "GLSL.std.450")

    # OpMemoryModel
    gen.instruction(14, 0, 1)  # Logical, GLSL450

    # OpEntryPoint Fragment
    gen.instruction(15,
                    4,  # Fragment execution model
                    id_main,  # Entry point
                    "main",
                    id_outcolor_var,
                    id_worldpos_var,
                    id_normal_var,
                    id_color_var)

    # OpExecutionMode OriginUpperLeft
    gen.instruction(16, id_main, 7)  # OriginUpperLeft = 7

    # Decorations
    # Location decorations (30 = Location)
    gen.instruction(71, id_outcolor_var, 30, 0)      # outColor location 0
    gen.instruction(71, id_worldpos_var, 30, 0)      # worldPos location 0
    gen.instruction(71, id_normal_var, 30, 1)        # normal location 1
    gen.instruction(71, id_color_var, 30, 4)         # color location 4

    # UBO decorations (34=DescriptorSet, 33=Binding)
    gen.instruction(71, id_ubo_var, 34, 0)     # DescriptorSet 0
    gen.instruction(71, id_ubo_var, 33, 0)     # Binding 0

    # Block decoration
    gen.instruction(71, id_ubo_type, 2)  # Block = 2

    # Member decorations (Offset = 35)
    gen.instruction(72, id_ubo_type, 0, 35, 0)       # mat4 offset 0
    gen.instruction(72, id_ubo_type, 1, 35, 64)      # mat4 offset 64
    gen.instruction(72, id_ubo_type, 2, 35, 128)     # mat4 offset 128
    gen.instruction(72, id_ubo_type, 3, 35, 192)     # vec4 offset 192
    gen.instruction(72, id_ubo_type, 4, 35, 208)     # vec4 offset 208
    gen.instruction(72, id_ubo_type, 5, 35, 224)     # vec4 offset 224

    # Type definitions
    gen.instruction(19)  # OpTypeVoid
    gen.instruction(22, id_float, 32)  # OpTypeFloat 32
    gen.instruction(23, id_vec3, id_float, 3)  # OpTypeVector vec3
    gen.instruction(23, id_vec4, id_float, 4)  # OpTypeVector vec4
    gen.instruction(23, id_vec2, id_float, 2)  # OpTypeVector vec2
    gen.instruction(24, id_mat4, id_vec4, 4)  # OpTypeMatrix mat4x4

    # OpTypeStruct for UBO
    gen.instruction(30, id_ubo_type, id_mat4, id_mat4, id_mat4, id_vec4, id_vec4, id_vec4)

    # OpTypePointer
    gen.instruction(32, id_ptr_uniform_ubo, 2, id_ubo_type)  # Uniform = 2
    gen.instruction(32, id_ptr_input_vec3_1, 1, id_vec3)     # Input = 1
    gen.instruction(32, id_ptr_input_vec3_2, 1, id_vec3)     # Input = 1
    gen.instruction(32, id_ptr_input_vec3_3, 1, id_vec3)     # Input = 1
    gen.instruction(32, id_ptr_output_vec4, 3, id_vec4)      # Output = 3

    # OpTypeFunction
    gen.instruction(33, id_func_void, id_void)

    # Constants
    gen.instruction(43, id_float, id_const_0, gen.float_bits(0.0))
    gen.instruction(43, id_float, id_const_1, gen.float_bits(1.0))
    gen.instruction(43, id_float, id_const_075, gen.float_bits(0.75))
    gen.instruction(43, id_float, id_const_05, gen.float_bits(0.5))
    gen.instruction(43, id_float, id_const_025, gen.float_bits(0.25))
    gen.instruction(43, id_float, id_const_07, gen.float_bits(0.7))
    gen.instruction(43, id_float, id_const_04, gen.float_bits(0.4))
    gen.instruction(43, id_float, id_const_02, gen.float_bits(0.2))

    # Variables
    gen.instruction(59, id_ptr_uniform_ubo, id_ubo_var, 2)  # Uniform = 2
    gen.instruction(59, id_ptr_input_vec3_1, id_worldpos_var, 1)  # Input = 1
    gen.instruction(59, id_ptr_input_vec3_2, id_normal_var, 1)    # Input = 1
    gen.instruction(59, id_ptr_input_vec3_3, id_color_var, 1)     # Input = 1
    gen.instruction(59, id_ptr_output_vec4, id_outcolor_var, 3)   # Output = 3

    # Function
    gen.instruction(54, id_void, id_main, 0, id_func_void)

    # Entry label
    gen.instruction(248, id_entry_label)

    # Load inputs
    gen.instruction(61, id_vec3, id_tmp_worldpos, id_worldpos_var)
    gen.instruction(61, id_vec3, id_tmp_normal, id_normal_var)
    gen.instruction(61, id_vec3, id_tmp_color, id_color_var)

    # Normalize normal: OpExtInst normalize
    id_norm_normal_result = gen.gen_id()
    gen.instruction(12, id_vec3, id_norm_normal_result, id_glsl_ext, 69, id_tmp_normal)  # 69 = Normalize

    # Get light position from UBO via AccessChain
    id_lightpos_ptr = gen.gen_id()
    id_const_0_idx = gen.gen_id()
    id_const_3_idx = gen.gen_id()
    gen.instruction(43, id_float, id_const_0_idx, gen.float_bits(0.0))
    gen.instruction(43, id_float, id_const_3_idx, gen.float_bits(3.0))

    # For simplicity, we'll just use a default light direction (1, 1, 1) normalized
    # This would require more complex code to access the UBO, so let's simplify

    # Create a simple light direction vector (1, 1, 1) and normalize it
    id_light_x = gen.gen_id()
    id_light_y = gen.gen_id()
    id_light_z = gen.gen_id()
    gen.instruction(43, id_float, id_light_x, gen.float_bits(1.0))
    gen.instruction(43, id_float, id_light_y, gen.float_bits(1.0))
    gen.instruction(43, id_float, id_light_z, gen.float_bits(1.0))

    id_light_vec = gen.gen_id()
    gen.instruction(80, id_vec3, id_light_vec, id_light_x, id_light_y, id_light_z)

    id_light_normalized = gen.gen_id()
    gen.instruction(12, id_vec3, id_light_normalized, id_glsl_ext, 69, id_light_vec)  # Normalize

    # Dot product
    id_dot_result = gen.gen_id()
    gen.instruction(148, id_float, id_dot_result, id_norm_normal_result, id_light_normalized)

    # Max with 0
    id_diffuse = gen.gen_id()
    gen.instruction(12, id_float, id_diffuse, id_glsl_ext, 40, id_dot_result, id_const_0)  # FMax

    # Quantize into bands using nested Select instructions
    # if (d >= 0.75) band = 1.0
    id_cmp1 = gen.gen_id()
    gen.instruction(172, id_cmp1, id_diffuse, id_const_075)  # FUnordGreaterThanEqual

    id_band_option1 = gen.gen_id()
    gen.instruction(169, id_float, id_band_option1, id_cmp1, id_const_1, id_const_07)  # Select

    # else if (d >= 0.5) band = 0.7
    id_cmp2 = gen.gen_id()
    gen.instruction(172, id_cmp2, id_diffuse, id_const_05)  # FUnordGreaterThanEqual

    id_band_option2 = gen.gen_id()
    gen.instruction(169, id_float, id_band_option2, id_cmp2, id_band_option1, id_const_04)  # Select

    # else if (d >= 0.25) band = 0.4
    id_cmp3 = gen.gen_id()
    gen.instruction(172, id_cmp3, id_diffuse, id_const_025)  # FUnordGreaterThanEqual

    id_band_final = gen.gen_id()
    gen.instruction(169, id_float, id_band_final, id_cmp3, id_band_option2, id_const_02)  # Select

    # Multiply color by band: VectorTimesScalar
    id_scaled_color = gen.gen_id()
    gen.instruction(133, id_vec3, id_scaled_color, id_band_final, id_tmp_color)  # FMul (vector * scalar)

    # Construct output vec4
    id_output = gen.gen_id()
    gen.instruction(80, id_vec4, id_output, id_scaled_color, id_const_1)  # CompositeConstruct

    # Store output
    gen.instruction(62, id_outcolor_var, id_output)

    # Return
    gen.instruction(253)  # OpReturn

    # Function end
    gen.instruction(56)  # OpFunctionEnd

    return gen.get_hex()

if __name__ == '__main__':
    hex_str = generate_toon_shader()
    print(hex_str)
