#!/usr/bin/env python3
"""
SPIR-V compute shader builder for inclusive prefix sum.
Generates valid SPIR-V binary with proper encoding.
"""

class SPIRVBuilder:
    def __init__(self):
        self.words = []
        self.next_id = 1

    def new_id(self):
        self.next_id += 1
        return self.next_id - 1

    def emit_word(self, w):
        self.words.append(w & 0xFFFFFFFF)

    def emit_words(self, *ws):
        for w in ws:
            self.emit_word(w)

    def emit_instruction(self, opcode, *operands):
        word_count = 1 + len(operands)
        self.emit_word((opcode << 16) | word_count)
        self.emit_words(*operands)

    def emit_string(self, s):
        b = (s + '\0').encode('utf-8')
        padding = (4 - len(b) % 4) % 4
        b = b + b'\0' * padding
        for i in range(0, len(b), 4):
            self.emit_word(int.from_bytes(b[i:i+4], 'little'))

    def build(self):
        # Allocate IDs
        void_ty = self.new_id()
        uint_ty = self.new_id()
        ptr_storage_uint = self.new_id()
        ptr_uniform_uint = self.new_id()
        ptr_workgroup_uint = self.new_id()

        c0 = self.new_id()
        c1 = self.new_id()
        c256 = self.new_id()

        # Storage arrays: runtime array of uint
        arr_uint = self.new_id()
        ptr_storage_arr = self.new_id()

        # UBO struct
        ubo_struct = self.new_id()
        ptr_ubo = self.new_id()

        # Variables
        var_input = self.new_id()
        var_output = self.new_id()
        var_ubo = self.new_id()
        var_shared = self.new_id()

        # Function
        main_fn = self.new_id()
        fn_type = self.new_id()
        main_entry = self.new_id()

        # --- OpCapability
        self.emit_instruction(0x11, 1)  # Shader

        # --- OpMemoryModel
        self.emit_instruction(0x0E, 0, 1)  # Physical32, GLSL450

        # --- Entry point (will update word count after)
        entry_start = len(self.words)
        self.emit_instruction(0x0F, 5, main_fn)  # GLCompute, main
        self.emit_string("main")
        self.emit_words(var_input, var_output, var_ubo)

        # --- Execution mode
        self.emit_instruction(0x10, main_fn, 17, 256, 1, 1)  # LocalSize

        # --- Decorations
        self.emit_instruction(0x47, var_input, 33, 0)   # Binding 0
        self.emit_instruction(0x47, var_input, 34, 0)   # DescriptorSet 0
        self.emit_instruction(0x47, var_output, 33, 1)  # Binding 1
        self.emit_instruction(0x47, var_output, 34, 0)
        self.emit_instruction(0x47, var_ubo, 33, 2)     # Binding 2
        self.emit_instruction(0x47, var_ubo, 34, 0)
        self.emit_instruction(0x47, ubo_struct, 2)      # Block
        self.emit_instruction(0x48, ubo_struct, 0, 6, 4) # Offset 0, ArrayStride 4

        # --- Types
        self.emit_instruction(0x19, void_ty)             # OpTypeVoid
        self.emit_instruction(0x15, uint_ty, 32, 0)      # OpTypeInt 32 unsigned
        self.emit_instruction(0x1D, arr_uint, uint_ty)   # OpTypeRuntimeArray
        self.emit_instruction(0x20, ubo_struct, uint_ty) # OpTypeStruct

        # --- Type pointers
        self.emit_instruction(0x1B, ptr_storage_uint, 12, uint_ty)    # StorageBuffer
        self.emit_instruction(0x1B, ptr_storage_arr, 12, arr_uint)
        self.emit_instruction(0x1B, ptr_ubo, 2, ubo_struct)           # Uniform
        self.emit_instruction(0x1B, ptr_workgroup_uint, 3, uint_ty)   # Workgroup

        # --- Constants
        self.emit_instruction(0x2B, uint_ty, c0, 0)
        self.emit_instruction(0x2B, uint_ty, c1, 1)
        self.emit_instruction(0x2B, uint_ty, c256, 256)

        # --- Workgroup shared array type
        arr_256 = self.new_id()
        ptr_arr_256 = self.new_id()
        self.emit_instruction(0x1C, arr_256, uint_ty, c256)
        self.emit_instruction(0x1B, ptr_arr_256, 3, arr_256)

        # --- Variables
        self.emit_instruction(0x3B, ptr_storage_arr, var_input, 12, None)
        self.emit_instruction(0x3B, ptr_storage_arr, var_output, 12, None)
        self.emit_instruction(0x3B, ptr_ubo, var_ubo, 2, None)
        self.emit_instruction(0x3B, ptr_arr_256, var_shared, 3, None)

        # --- Function type
        self.emit_instruction(0x21, fn_type, void_ty)

        # --- Function definition
        self.emit_instruction(0x36, void_ty, main_fn, 0, fn_type)

        # --- Entry block
        self.emit_instruction(0x0B, main_entry)

        # Get global invocation ID
        vec3_uint = self.new_id()
        self.emit_instruction(0x16, vec3_uint, uint_ty, 3)
        gid_builtin = self.new_id()
        self.emit_instruction(0x57, vec3_uint, gid_builtin, 3, c0)  # GlobalInvocationId
        gid = self.new_id()
        self.emit_instruction(0x65, uint_ty, gid, gid_builtin, 0)   # Extract x

        # Load N
        n_ptr = self.new_id()
        self.emit_instruction(0x41, ptr_uniform_uint, n_ptr, var_ubo, c0)
        n_val = self.new_id()
        self.emit_instruction(0x3D, uint_ty, n_val, n_ptr)

        # Inclusive scan: shared memory version for N <= 256
        # Local barrier synchronization and parallel scan
        # For simplicity, fallback to sequential: output[i] = sum of input[0..i]

        # Load input[gid] to shared memory
        load_ptr = self.new_id()
        self.emit_instruction(0x41, ptr_storage_uint, load_ptr, var_input, gid)
        in_val = self.new_id()
        self.emit_instruction(0x3D, uint_ty, in_val, load_ptr)

        # Store to shared[gid]
        shared_ptr = self.new_id()
        self.emit_instruction(0x41, ptr_workgroup_uint, shared_ptr, var_shared, gid)
        self.emit_instruction(0x3E, shared_ptr, in_val)

        # Barrier
        self.emit_instruction(0x2B8, 2, 8)  # ControlBarrier with Workgroup|Acquire

        # Accumulate: scan pattern
        # For each thread, sum all elements in [0, gid]
        sum_var = self.new_id()
        self.emit_instruction(0x3F, ptr_workgroup_uint, sum_var)
        self.emit_instruction(0x3E, sum_var, c0)

        # Loop counter
        i_var = self.new_id()
        self.emit_instruction(0x3F, ptr_workgroup_uint, i_var)
        self.emit_instruction(0x3E, i_var, c0)

        # Loop header
        loop_header = self.new_id()
        loop_body = self.new_id()
        loop_continue = self.new_id()
        loop_merge = self.new_id()

        self.emit_instruction(0xf5, loop_header)

        # Header
        self.emit_instruction(0x0B, loop_header)
        self.emit_instruction(0xf7, loop_merge, loop_continue, 1)  # LoopMerge

        # Check i <= gid
        i_load = self.new_id()
        self.emit_instruction(0x3D, uint_ty, i_load, i_var)
        cmp = self.new_id()
        self.emit_instruction(0x9a, uint_ty, cmp, 2, i_load, gid)  # ULessThanEqual

        self.emit_instruction(0xf6, cmp, loop_body, loop_merge)

        # Body
        self.emit_instruction(0x0B, loop_body)

        # shared_sum = shared[i]
        shared_load_ptr = self.new_id()
        self.emit_instruction(0x41, ptr_workgroup_uint, shared_load_ptr, var_shared, i_load)
        val = self.new_id()
        self.emit_instruction(0x3D, uint_ty, val, shared_load_ptr)

        # sum += val
        sum_load = self.new_id()
        self.emit_instruction(0x3D, uint_ty, sum_load, sum_var)
        new_sum = self.new_id()
        self.emit_instruction(0x7f, uint_ty, new_sum, sum_load, val)  # IAdd
        self.emit_instruction(0x3E, sum_var, new_sum)

        # i++
        i_next = self.new_id()
        self.emit_instruction(0x7f, uint_ty, i_next, i_load, c1)
        self.emit_instruction(0x3E, i_var, i_next)

        # Branch to continue
        self.emit_instruction(0xf5, loop_continue)

        # Continue
        self.emit_instruction(0x0B, loop_continue)
        self.emit_instruction(0xf5, loop_header)

        # After loop
        self.emit_instruction(0x0B, loop_merge)

        # Load final sum
        final_sum = self.new_id()
        self.emit_instruction(0x3D, uint_ty, final_sum, sum_var)

        # Store to output[gid]
        out_ptr = self.new_id()
        self.emit_instruction(0x41, ptr_storage_uint, out_ptr, var_output, gid)
        self.emit_instruction(0x3E, out_ptr, final_sum)

        # Return
        self.emit_instruction(0x37)

        # Function end
        self.emit_instruction(0x39)

        # Build final module
        bound = self.next_id
        header = [
            0x07230203,  # Magic
            0x00010000,  # Version
            0x00070000,  # Generator (LLVM 7.0)
            bound,
            0,           # Schema
        ]

        final = header + self.words
        return ''.join(f'{w:08x}' for w in final)

if __name__ == '__main__':
    builder = SPIRVBuilder()
    result = builder.build()
    print(result)
