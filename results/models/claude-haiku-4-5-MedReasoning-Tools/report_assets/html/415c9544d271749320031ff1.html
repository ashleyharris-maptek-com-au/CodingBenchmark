using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Threading.Tasks;

class MaximumClique
{
    private int numVertices;
    private List<int>[] adjacencyList;
    private bool[,] adjacencyMatrix;
    private int[] bestClique;
    private int bestCliqueSize;
    private Stopwatch timer;
    private const long TIME_LIMIT_MS = 280000; // 280 seconds (5 minutes - 20 second buffer)

    static void Main()
    {
        var solver = new MaximumClique();
        solver.Solve();
    }

    void Solve()
    {
        timer = Stopwatch.StartNew();
        ReadInput();

        if (numVertices == 0)
        {
            Console.WriteLine("0");
            return;
        }

        bestClique = new int[0];
        bestCliqueSize = 0;

        // Strategy selection based on graph size and density
        if (numVertices <= 100)
        {
            // Small graphs: use exact algorithm (Bron-Kerbosch with pivoting)
            BronKerbosch();
        }
        else if (numVertices <= 10000)
        {
            // Medium graphs: use branch and bound with greedy initialization
            GreedyCliqueFinding();
            if (timer.ElapsedMilliseconds < TIME_LIMIT_MS * 0.7)
            {
                BranchAndBoundSearch();
            }
        }
        else
        {
            // Large graphs: greedy and multi-start greedy
            MultiStartGreedy();
        }

        OutputResult();
    }

    void ReadInput()
    {
        string line = Console.ReadLine()!.TrimStart('﻿').Trim();
        string[] parts = line.Split();
        numVertices = int.Parse(parts[0]);
        int numEdges = int.Parse(parts[1]);

        adjacencyList = new List<int>[numVertices];
        for (int i = 0; i < numVertices; i++)
            adjacencyList[i] = new List<int>();

        if (numVertices <= 5000)
            adjacencyMatrix = new bool[numVertices, numVertices];

        for (int i = 0; i < numEdges; i++)
        {
            parts = Console.ReadLine()!.Trim().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);

            adjacencyList[u].Add(v);
            adjacencyList[v].Add(u);

            if (adjacencyMatrix != null)
            {
                adjacencyMatrix[u, v] = true;
                adjacencyMatrix[v, u] = true;
            }
        }

        // Sort adjacency lists by degree (descending) for better pruning
        for (int i = 0; i < numVertices; i++)
            adjacencyList[i].Sort((a, b) => adjacencyList[b].Count.CompareTo(adjacencyList[a].Count));
    }

    void GreedyCliqueFinding()
    {
        // Simple greedy: start from highest degree vertex
        int[] degrees = new int[numVertices];
        for (int i = 0; i < numVertices; i++)
            degrees[i] = adjacencyList[i].Count;

        for (int start = 0; start < Math.Min(numVertices, 20) && timer.ElapsedMilliseconds < TIME_LIMIT_MS; start++)
        {
            int startVertex = Array.IndexOf(degrees, degrees.Max());
            degrees[startVertex] = -1;

            var clique = new List<int> { startVertex };
            var candidates = new HashSet<int>(adjacencyList[startVertex]);

            while (candidates.Count > 0)
            {
                int nextVertex = candidates.OrderByDescending(v => candidates.Count(c => IsAdjacent(v, c))).First();
                clique.Add(nextVertex);

                var newCandidates = new HashSet<int>();
                foreach (var v in candidates)
                {
                    if (IsAdjacent(nextVertex, v))
                        newCandidates.Add(v);
                }
                candidates = newCandidates;
            }

            if (clique.Count > bestCliqueSize)
            {
                bestCliqueSize = clique.Count;
                bestClique = clique.ToArray();
            }
        }
    }

    void MultiStartGreedy()
    {
        // Multi-start greedy with random ordering
        var random = new Random(42);
        var vertices = Enumerable.Range(0, numVertices).OrderBy(_ => random.Next()).ToArray();

        Parallel.For(0, Math.Min(numVertices, 50), new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, start =>
        {
            if (timer.ElapsedMilliseconds >= TIME_LIMIT_MS) return;

            var clique = new List<int> { vertices[start] };
            var candidates = new HashSet<int>(adjacencyList[vertices[start]]);

            while (candidates.Count > 0 && timer.ElapsedMilliseconds < TIME_LIMIT_MS)
            {
                int nextVertex = candidates.OrderByDescending(v => candidates.Count(c => IsAdjacent(v, c))).First();
                clique.Add(nextVertex);

                var newCandidates = new HashSet<int>();
                foreach (var v in candidates)
                {
                    if (IsAdjacent(nextVertex, v))
                        newCandidates.Add(v);
                }
                candidates = newCandidates;
            }

            lock (this)
            {
                if (clique.Count > bestCliqueSize)
                {
                    bestCliqueSize = clique.Count;
                    bestClique = clique.ToArray();
                }
            }
        });
    }

    void BronKerbosch()
    {
        var candidates = new HashSet<int>(Enumerable.Range(0, numVertices));
        var clique = new HashSet<int>();
        var excluded = new HashSet<int>();

        BronKerboschPivot(clique, candidates, excluded);
    }

    void BronKerboschPivot(HashSet<int> clique, HashSet<int> candidates, HashSet<int> excluded)
    {
        if (timer.ElapsedMilliseconds >= TIME_LIMIT_MS) return;

        if (candidates.Count == 0 && excluded.Count == 0)
        {
            if (clique.Count > bestCliqueSize)
            {
                bestCliqueSize = clique.Count;
                bestClique = clique.ToArray();
            }
            return;
        }

        // Choose pivot
        var unionSet = candidates.Union(excluded);
        int pivot = unionSet.OrderByDescending(v => candidates.Count(c => IsAdjacent(v, c))).FirstOrDefault();

        var candidates_minus_pivot = candidates.Except(adjacencyList[pivot]).ToList();

        foreach (int vertex in candidates_minus_pivot)
        {
            if (timer.ElapsedMilliseconds >= TIME_LIMIT_MS) return;

            var newClique = new HashSet<int>(clique) { vertex };
            var newCandidates = new HashSet<int>(candidates.Intersect(adjacencyList[vertex]));
            var newExcluded = new HashSet<int>(excluded.Intersect(adjacencyList[vertex]));

            BronKerboschPivot(newClique, newCandidates, newExcluded);

            candidates.Remove(vertex);
            excluded.Add(vertex);
        }
    }

    void BranchAndBoundSearch()
    {
        var verticesByDegree = Enumerable.Range(0, numVertices)
            .OrderByDescending(v => adjacencyList[v].Count)
            .ToList();

        BranchAndBoundRecursive(new List<int>(), new HashSet<int>(verticesByDegree), 0);
    }

    void BranchAndBoundRecursive(List<int> clique, HashSet<int> candidates, int depth)
    {
        if (timer.ElapsedMilliseconds >= TIME_LIMIT_MS) return;
        if (depth > 100) return; // Limit recursion depth

        if (candidates.Count == 0)
        {
            if (clique.Count > bestCliqueSize)
            {
                bestCliqueSize = clique.Count;
                bestClique = clique.ToArray();
            }
            return;
        }

        // Upper bound pruning
        if (clique.Count + candidates.Count <= bestCliqueSize)
            return;

        int vertex = candidates.First();
        var neighbors = new HashSet<int>(adjacencyList[vertex]);
        var newCandidates = new HashSet<int>(candidates.Intersect(neighbors));

        clique.Add(vertex);
        BranchAndBoundRecursive(clique, newCandidates, depth + 1);
        clique.RemoveAt(clique.Count - 1);

        candidates.Remove(vertex);
        BranchAndBoundRecursive(clique, candidates, depth + 1);
    }

    bool IsAdjacent(int u, int v)
    {
        if (adjacencyMatrix != null)
            return adjacencyMatrix[u, v];
        return adjacencyList[u].Contains(v);
    }

    void OutputResult()
    {
        Console.WriteLine(bestCliqueSize);
        if (bestCliqueSize > 0)
            Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
    }
}
