Below is a complete C++ program (C++17-compatible) that computes the **exact Maximum Independent Set** by reducing it to a **Maximum Clique** problem on the complement graph.  
It uses a branch‑and‑bound algorithm with greedy coloring upper bounds (Tomita‑style), which is among the fastest exact methods for moderately sized graphs.

```cpp
#include <iostream>
#include <vector>
#include <cstdint>
#include <algorithm>
#include <cstdio>
#include <cstring>

#ifdef _MSC_VER
#include <intrin.h>
#endif

using namespace std;

// Fast scanner
struct FastScanner {
    static const int BUFSIZE = 1 << 20;
    int idx, size;
    char buf[BUFSIZE];
    FastScanner() : idx(0), size(0) {}
    inline char read() {
        if (idx >= size) {
            size = (int)fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }
    bool nextInt(int &out) {
        char c;
        do {
            c = read();
            if (!c) return false;
        } while (c <= ' ');
        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = read();
        }
        int val = 0;
        while (c > ' ') {
            val = val * 10 + (c - '0');
            c = read();
        }
        out = val * sign;
        return true;
    }
};

#ifdef _MSC_VER
inline int ctz64(uint64_t x) {
    unsigned long idx;
    _BitScanForward64(&idx, x);
    return (int)idx;
}
#else
inline int ctz64(uint64_t x) {
    return __builtin_ctzll(x);
}
#endif

int N, M;
int blocks;
vector<vector<uint64_t>> adj;  // adjacency in complement graph

int best_size = 0;
vector<int> best_clique;

inline void clear_bit(vector<uint64_t>& bs, int v) {
    bs[v >> 6] &= ~(1ULL << (v & 63));
}

inline bool is_empty(const vector<uint64_t>& bs) {
    for (uint64_t x : bs) if (x) return false;
    return true;
}

inline int first_bit(const vector<uint64_t>& bs) {
    for (int i = 0; i < blocks; ++i) {
        uint64_t x = bs[i];
        if (x) {
            return i * 64 + ctz64(x);
        }
    }
    return -1;
}

inline vector<uint64_t> intersect(const vector<uint64_t>& a, const vector<uint64_t>& b) {
    vector<uint64_t> res(blocks);
    for (int i = 0; i < blocks; ++i) res[i] = a[i] & b[i];
    return res;
}

// Greedy coloring for upper bound
void color(const vector<uint64_t>& P, vector<int>& order, vector<int>& colors) {
    order.clear();
    colors.clear();
    vector<uint64_t> U = P;
    int color_id = 0;
    while (!is_empty(U)) {
        ++color_id;
        vector<uint64_t> Q = U;
        while (!is_empty(Q)) {
            int v = first_bit(Q);
            order.push_back(v);
            colors.push_back(color_id);
            clear_bit(U, v);
            clear_bit(Q, v);
            for (int i = 0; i < blocks; ++i) {
                Q[i] &= ~adj[v][i];
            }
        }
    }
}

void expand(vector<uint64_t> P, vector<int>& clique) {
    if (is_empty(P)) {
        if ((int)clique.size() > best_size) {
            best_size = (int)clique.size();
            best_clique = clique;
        }
        return;
    }

    vector<int> order, colors;
    color(P, order, colors);

    for (int i = (int)order.size() - 1; i >= 0; --i) {
        int v = order[i];
        if ((int)clique.size() + colors[i] <= best_size) return;

        clique.push_back(v);
        vector<uint64_t> P2 = intersect(P, adj[v]);
        expand(P2, clique);
        clique.pop_back();

        clear_bit(P, v);
    }
}

int main() {
    FastScanner fs;
    if (!fs.nextInt(N)) return 0;
    fs.nextInt(M);

    if (N == 0) {
        cout << 0 << "\n\n";
        return 0;
    }

    blocks = (N + 63) / 64;
    adj.assign(N, vector<uint64_t>(blocks, ~0ULL));

    // remove self-loops in complement (no self edges)
    for (int v = 0; v < N; ++v) clear_bit(adj[v], v);

    // read edges of original graph and remove them from complement
    for (int i = 0; i < M; ++i) {
        int u, v;
        fs.nextInt(u);
        fs.nextInt(v);
        if (u == v) continue;
        clear_bit(adj[u], v);
        clear_bit(adj[v], u);
    }

    // mask unused bits in last block
    if (N % 64) {
        uint64_t mask = (1ULL << (N % 64)) - 1;
        for (int v = 0; v < N; ++v) adj[v][blocks - 1] &= mask;
    }

    // Initial candidate set = all vertices
    vector<uint64_t> P(blocks, ~0ULL);
    if (N % 64) {
        P[blocks - 1] &= (1ULL << (N % 64)) - 1;
    }

    // greedy initial clique for lower bound
    vector<uint64_t> temp = P;
    vector<int> greedy;
    while (!is_empty(temp)) {
        int v = first_bit(temp);
        greedy.push_back(v);
        temp = intersect(temp, adj[v]);
    }
    best_size = (int)greedy.size();
    best_clique = greedy;

    vector<int> current;
    expand(P, current);

    cout << best_size << "\n";
    for (int i = 0; i < best_size; ++i) {
        if (i) cout << ' ';
        cout << best_clique[i];
    }
    cout << "\n";

    return 0;
}
```

### Notes
- The algorithm is **exact** and guarantees the maximum independent set.
- It uses the equivalence:  
  **Maximum Independent Set in G = Maximum Clique in complement of G**
- Output vertices are the maximum independent set in the original graph.

This program compiles with MSVC `/std:c++20` (and is also fine in C++17 mode).
