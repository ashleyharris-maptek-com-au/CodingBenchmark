OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%uint_6 = OpConstant %uint 6

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015

%white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3 = OpTypeArray %v3float %uint_6

%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3
%ptr_input_v3 = OpTypePointer Input %v3float

%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_output_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_input_arr_pv Input
%worldPos = OpVariable %ptr_input_arr_v3 Input
%normal = OpVariable %ptr_input_arr_v3 Input
%color = OpVariable %ptr_input_arr_v3 Input
%gl_out = OpVariable %ptr_output_pv Output
%outColor = OpVariable %ptr_output_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel

%outposptr = OpAccessChain %ptr_output_v4 %gl_out %int_0

; --- Emit filled triangle (vertices 0,2,4) ---
%p0ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0ptr
%c0ptr = OpAccessChain %ptr_input_v3 %color %int_0
%c0 = OpLoad %v3float %c0ptr
%c0dim = OpVectorTimesScalar %v3float %c0 %float_0_3
OpStore %outposptr %p0
OpStore %outColor %c0dim
OpEmitVertex

%p2ptr = OpAccessChain %ptr_input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2ptr
%c2ptr = OpAccessChain %ptr_input_v3 %color %int_2
%c2 = OpLoad %v3float %c2ptr
%c2dim = OpVectorTimesScalar %v3float %c2 %float_0_3
OpStore %outposptr %p2
OpStore %outColor %c2dim
OpEmitVertex

%p4ptr = OpAccessChain %ptr_input_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4ptr
%c4ptr = OpAccessChain %ptr_input_v3 %color %int_4
%c4 = OpLoad %v3float %c4ptr
%c4dim = OpVectorTimesScalar %v3float %c4 %float_0_3
OpStore %outposptr %p4
OpStore %outColor %c4dim
OpEmitVertex
OpEndPrimitive

; --- Edge 0-2 ---
%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
%dir0 = OpFSub %v2float %p2xy %p0xy
%dot0 = OpDot %float %dir0 %dir0
%len0 = OpExtInst %float %glsl450 Sqrt %dot0
%invlen0 = OpFDiv %float %float_1 %len0
%ndir0 = OpVectorTimesScalar %v2float %dir0 %invlen0
%dx0 = OpCompositeExtract %float %ndir0 0
%dy0 = OpCompositeExtract %float %ndir0 1
%negdy0 = OpFNegate %float %dy0
%perp0 = OpCompositeConstruct %v2float %negdy0 %dx0
%off0 = OpVectorTimesScalar %v2float %perp0 %float_0_015
%offx0 = OpCompositeExtract %float %off0 0
%offy0 = OpCompositeExtract %float %off0 1
%offv40 = OpCompositeConstruct %v4float %offx0 %offy0 %float_0 %float_0
%p0p = OpFAdd %v4float %p0 %offv40
%p0m = OpFSub %v4float %p0 %offv40
%p2p = OpFAdd %v4float %p2 %offv40
%p2m = OpFSub %v4float %p2 %offv40
OpStore %outposptr %p0p
OpStore %outColor %white
OpEmitVertex
OpStore %outposptr %p0m
OpStore %outColor %white
OpEmitVertex
OpStore %outposptr %p2p
OpStore %outColor %white
OpEmitVertex
OpStore %outposptr %p2m
OpStore %outColor %white
OpEmitVertex
OpEndPrimitive

; --- Edge 2-4 ---
%p2xy2 = OpVectorShuffle %v2float %p2 %p2 0 1
%p4xy = OpVectorShuffle %v2float %p4 %p4 0 1
%dir1 = OpFSub %v2float %p4xy %p2xy2
%dot1 = OpDot %float %dir1 %dir1
%len1 = OpExtInst %float %glsl450 Sqrt %dot1
%invlen1 = OpFDiv %float %float_1 %len1
%ndir1 = OpVectorTimesScalar %v2float %dir1 %invlen1
%dx1 = OpCompositeExtract %float %ndir1 0
%dy1 = OpCompositeExtract %float %ndir1 1
%negdy1 = OpFNegate %float %dy1
%perp1 = OpCompositeConstruct %v2float %negdy1 %dx1
%off1 = OpVectorTimesScalar %v2float %perp1 %float_0_015
%offx1 = OpCompositeExtract %float %off1 0
%offy1 = OpCompositeExtract %float %off1 1
%offv41 = OpCompositeConstruct %v4float %offx1 %offy1 %float_0 %float_0
%p2p1 = OpFAdd %v4float %p2 %offv41
%p2m1 = OpFSub %v4float %p2 %offv41
%p4p1 = OpFAdd %v4float %p4 %offv41
%p4m1 = OpFSub %v4float %p4 %offv41
OpStore %outposptr %p2p1
OpStore %outColor %white
OpEmitVertex
OpStore %outposptr %p2m1
OpStore %outColor %white
OpEmitVertex
OpStore %outposptr %p4p1
OpStore %outColor %white
OpEmitVertex
OpStore %outposptr %p4m1
OpStore %outColor %white
OpEmitVertex
OpEndPrimitive

; --- Edge 4-0 ---
%p4xy2 = OpVectorShuffle %v2float %p4 %p4 0 1
%p0xy2 = OpVectorShuffle %v2float %p0 %p0 0 1
%dir2 = OpFSub %v2float %p0xy2 %p4xy2
%dot2 = OpDot %float %dir2 %dir2
%len2 = OpExtInst %float %glsl450 Sqrt %dot2
%invlen2 = OpFDiv %float %float_1 %len2
%ndir2 = OpVectorTimesScalar %v2float %dir2 %invlen2
%dx2 = OpCompositeExtract %float %ndir2 0
%dy2 = OpCompositeExtract %float %ndir2 1
%negdy2 = OpFNegate %float %dy2
%perp2 = OpCompositeConstruct %v2float %negdy2 %dx2
%off2 = OpVectorTimesScalar %v2float %perp2 %float_0_015
%offx2 = OpCompositeExtract %float %off2 0
%offy2 = OpCompositeExtract %float %off2 1
%offv42 = OpCompositeConstruct %v4float %offx2 %offy2 %float_0 %float_0
%p4p2 = OpFAdd %v4float %p4 %offv42
%p4m2 = OpFSub %v4float %p4 %offv42
%p0p2 = OpFAdd %v4float %p0 %offv42
%p0m2 = OpFSub %v4float %p0 %offv42
OpStore %outposptr %p4p2
OpStore %outColor %white
OpEmitVertex
OpStore %outposptr %p4m2
OpStore %outColor %white
OpEmitVertex
OpStore %outposptr %p0p2
OpStore %outColor %white
OpEmitVertex
OpStore %outposptr %p0m2
OpStore %outColor %white
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
