using System;
using System.Collections.Generic;
using System.IO;

public class Program
{
    struct Cut
    {
        public int Len;
        public int Idx;
    }

    struct BinEntry
    {
        public int Rem;
        public int Id;
    }

    sealed class BinEntryComparer : IComparer<BinEntry>
    {
        public int Compare(BinEntry a, BinEntry b)
        {
            int c = a.Rem.CompareTo(b.Rem);
            if (c != 0) return c;
            return a.Id.CompareTo(b.Id);
        }
    }

    static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int stock = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        Cut[] cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();
            cuts[i] = new Cut { Len = len, Idx = i };
        }

        Array.Sort(cuts, (a, b) =>
        {
            int c = b.Len.CompareTo(a.Len);
            if (c != 0) return c;
            return a.Idx.CompareTo(b.Idx);
        });

        var comparer = new BinEntryComparer();
        var set = new SortedSet<BinEntry>(comparer);
        var bins = new List<List<int>>();

        for (int i = 0; i < n; i++)
        {
            int len = cuts[i].Len;
            var lower = new BinEntry { Rem = len, Id = int.MinValue };
            var upper = new BinEntry { Rem = int.MaxValue, Id = int.MaxValue };
            var view = set.GetViewBetween(lower, upper);

            if (view.Count > 0)
            {
                BinEntry entry = view.Min;
                set.Remove(entry);

                bins[entry.Id].Add(cuts[i].Idx);
                int newRem = entry.Rem - len;
                if (newRem > 0)
                    set.Add(new BinEntry { Rem = newRem, Id = entry.Id });
            }
            else
            {
                int id = bins.Count;
                var list = new List<int>();
                list.Add(cuts[i].Idx);
                bins.Add(list);
                int rem = stock - len;
                if (rem > 0)
                    set.Add(new BinEntry { Rem = rem, Id = id });
            }
        }

        var sw = new StreamWriter(Console.OpenStandardOutput());
        sw.NewLine = "\n";
        sw.WriteLine(bins.Count);
        for (int i = 0; i < bins.Count; i++)
        {
            var list = bins[i];
            for (int j = 0; j < list.Count; j++)
            {
                if (j > 0) sw.Write(' ');
                sw.Write(list[j]);
            }
            sw.WriteLine();
        }
        sw.Flush();
    }

    sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner()
        {
            _stream = Console.OpenStandardInput();
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= ' ' && c != -1);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
