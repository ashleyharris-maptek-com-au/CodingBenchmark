```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (input.length === 0) {
  console.log(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid || [];
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = data.cutBudget | 0;

const N = width * height;

const isTown = new Uint8Array(N);
const townValue = new Float64Array(N);
const isIgnition = new Uint8Array(N);
const burnable = new Uint8Array(N);
const cuttable = new Uint8Array(N);

// Initialize burnable/cuttable from grid
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    const idx = y * width + x;
    const ch = row.charCodeAt(x);
    if (ch === 46) { // '.'
      burnable[idx] = 1;
      cuttable[idx] = 1;
    }
  }
}

// Mark towns
const townIndices = [];
for (const t of towns) {
  const x = t.x | 0;
  const y = t.y | 0;
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  const idx = y * width + x;
  if (!isTown[idx]) {
    isTown[idx] = 1;
    townIndices.push(idx);
  }
  townValue[idx] += +t.value || 0;
  burnable[idx] = 1; // towns are burnable cells
  cuttable[idx] = 0; // do not cut town cells
}

// Mark ignitions
for (const ig of ignitions) {
  const x = ig[0] | 0;
  const y = ig[1] | 0;
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  const idx = y * width + x;
  isIgnition[idx] = 1;
}

// BFS from ignitions to find reachable cells
const reachable = new Uint8Array(N);
const queue = new Int32Array(N);
let head = 0, tail = 0;
for (const ig of ignitions) {
  const x = ig[0] | 0;
  const y = ig[1] | 0;
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  const idx = y * width + x;
  if (burnable[idx] && !reachable[idx]) {
    reachable[idx] = 1;
    queue[tail++] = idx;
  }
}
while (head < tail) {
  const idx = queue[head++];
  const x = idx % width;
  const y = (idx / width) | 0;

  if (x > 0) {
    const n = idx - 1;
    if (burnable[n] && !reachable[n]) {
      reachable[n] = 1;
      queue[tail++] = n;
    }
  }
  if (x + 1 < width) {
    const n = idx + 1;
    if (burnable[n] && !reachable[n]) {
      reachable[n] = 1;
      queue[tail++] = n;
    }
  }
  if (y > 0) {
    const n = idx - width;
    if (burnable[n] && !reachable[n]) {
      reachable[n] = 1;
      queue[tail++] = n;
    }
  }
  if (y + 1 < height) {
    const n = idx + width;
    if (burnable[n] && !reachable[n]) {
      reachable[n] = 1;
      queue[tail++] = n;
    }
  }
}

// Find town components and their boundary cuts
const visitedTown = new Uint8Array(N);
const boundaryMark = new Int32Array(N);
let markId = 1;

const candidates = [];

for (const startIdx of townIndices) {
  if (visitedTown[startIdx]) continue;

  const q = [];
  let qh = 0;
  q.push(startIdx);
  visitedTown[startIdx] = 1;

  let totalValue = 0;
  let reachableFlag = false;
  let ignitionInside = false;

  const boundary = [];
  markId++;
  if (markId === 2147483647) {
    boundaryMark.fill(0);
    markId = 1;
  }

  while (qh < q.length) {
    const idx = q[qh++];
    totalValue += townValue[idx];
    if (reachable[idx]) reachableFlag = true;
    if (isIgnition[idx]) ignitionInside = true;

    const x = idx % width;
    const y = (idx / width) | 0;

    // Left
    if (x > 0) {
      const n = idx - 1;
      if (isTown[n]) {
        if (!visitedTown[n]) {
          visitedTown[n] = 1;
          q.push(n);
        }
      } else if (cuttable[n]) {
        if (boundaryMark[n] !== markId) {
          boundaryMark[n] = markId;
          boundary.push(n);
        }
      }
    }
    // Right
    if (x + 1 < width) {
      const n = idx + 1;
      if (isTown[n]) {
        if (!visitedTown[n]) {
          visitedTown[n] = 1;
          q.push(n);
        }
      } else if (cuttable[n]) {
        if (boundaryMark[n] !== markId) {
          boundaryMark[n] = markId;
          boundary.push(n);
        }
      }
    }
    // Up
    if (y > 0) {
      const n = idx - width;
      if (isTown[n]) {
        if (!visitedTown[n]) {
          visitedTown[n] = 1;
          q.push(n);
        }
      } else if (cuttable[n]) {
        if (boundaryMark[n] !== markId) {
          boundaryMark[n] = markId;
          boundary.push(n);
        }
      }
    }
    // Down
    if (y + 1 < height) {
      const n = idx + width;
      if (isTown[n]) {
        if (!visitedTown[n]) {
          visitedTown[n] = 1;
          q.push(n);
        }
      } else if (cuttable[n]) {
        if (boundaryMark[n] !== markId) {
          boundaryMark[n] = markId;
          boundary.push(n);
        }
      }
    }
  }

  const cost = boundary.length;
  if (reachableFlag && !ignitionInside && cost > 0) {
    candidates.push({ value: totalValue, cost, boundary });
  }
}

// Sort by value/cost ratio descending
candidates.sort((a, b) => (b.value * a.cost) - (a.value * b.cost));

// Select firebreaks greedily
let remaining = cutBudget;
const cutSelected = new Uint8Array(N);
const cutList = [];

for (const cand of candidates) {
  if (remaining <= 0) break;
  const boundary = cand.boundary;
  const additional = [];
  for (let i = 0; i < boundary.length; i++) {
    const idx = boundary[i];
    if (!cutSelected[idx]) additional.push(idx);
  }
  const addCost = additional.length;
  if (addCost === 0) continue;
  if (addCost <= remaining) {
    for (const idx of additional) {
      cutSelected[idx] = 1;
      cutList.push(idx);
    }
    remaining -= addCost;
  }
}

// Prepare output
const firebreaks = cutList.map(idx => [idx % width, (idx / width) | 0]);

console.log(JSON.stringify({ firebreaks }));
```
