; SPIR-V
; Version: 1.0
; Generator: None
; Bound: 200
; Schema: 0
OpCapability Shader
%GLSL450 = OpExtInstImport "GLSL450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4 = OpTypePointer Uniform %v4
%ptr_Input_v3 = OpTypePointer Input %v3
%ptr_Output_v4 = OpTypePointer Output %v4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_5 = OpConstant %float 0.5
%float_0_3 = OpConstant %float 0.3
%float_0_15 = OpConstant %float 0.15
%float_0_05 = OpConstant %float 0.05
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6

%int_3 = OpConstant %int 3

%ground = OpConstantComposite %v3 %float_0_3 %float_0_15 %float_0_05
%sky = OpConstantComposite %v3 %float_0_4 %float_0_6 %float_1

%ubo = OpVariable %ptr_Uniform_UBO Uniform
%inWorldPos = OpVariable %ptr_Input_v3 Input
%inNormal = OpVariable %ptr_Input_v3 Input
%outColor = OpVariable %ptr_Output_v4 Output

%main = OpFunction %void None %func
%entry = OpLabel

%norm_in = OpLoad %v3 %inNormal
%N = OpExtInst %v3 %GLSL450 Normalize %norm_in

%world = OpLoad %v3 %inWorldPos

%lightPtr = OpAccessChain %ptr_Uniform_v4 %ubo %int_3
%light4 = OpLoad %v4 %lightPtr
%light3 = OpVectorShuffle %v3 %light4 %light4 0 1 2

%Ldir = OpFSub %v3 %light3 %world
%L = OpExtInst %v3 %GLSL450 Normalize %Ldir
%dot = OpDot %float %N %L
%diffmax = OpExtInst %float %GLSL450 FMax %dot %float_0
%diffuse = OpFMul %float %diffmax %float_0_3

%Ny = OpCompositeExtract %float %N 1
%Ny_half = OpFMul %float %Ny %float_0_5
%t = OpFAdd %float %Ny_half %float_0_5

%sky_minus_ground = OpFSub %v3 %sky %ground
%scaled = OpVectorTimesScalar %v3 %sky_minus_ground %t
%ambient = OpFAdd %v3 %ground %scaled

%diffuse3 = OpCompositeConstruct %v3 %diffuse %diffuse %diffuse
%color3 = OpFAdd %v3 %ambient %diffuse3

%r = OpCompositeExtract %float %color3 0
%g = OpCompositeExtract %float %color3 1
%b = OpCompositeExtract %float %color3 2
%out = OpCompositeConstruct %v4 %r %g %b %float_1
OpStore %outColor %out

OpReturn
OpFunctionEnd
