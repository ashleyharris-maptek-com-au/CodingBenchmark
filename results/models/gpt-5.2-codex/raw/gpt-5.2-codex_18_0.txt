Below is a complete, compilable C++20 solution that **adapts to graph size**:

- **Small graphs** → exact **Stoer–Wagner** algorithm (deterministic, optimal).
- **Large graphs** → randomized **Karger–Stein** contraction with repetition for high probability of optimality.
- Efficient buffered I/O is used for speed.
- Works with MSVC C++20 and avoids non‑standard headers.

> **Note:** The Minimum Edge Cut (global min cut) for extremely large graphs is inherently expensive. This solver automatically switches to a randomized contraction method for scalability while still producing valid cuts.

### ✅ C++20 Implementation (MSVC-compatible)

```cpp
#include <iostream>
#include <vector>
#include <random>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <limits>
#include <chrono>
#include <utility>
#include <cstdio>
#include <cstring>

struct FastScanner {
    static const size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline char read() {
        if (idx >= size) {
            size = fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <typename T>
    bool readInt(T &out) {
        char c;
        T sign = 1, val = 0;
        c = read();
        if (!c) return false;
        while (c <= ' ') {
            c = read();
            if (!c) return false;
        }
        if (c == '-') {
            sign = -1;
            c = read();
        }
        for (; c > ' '; c = read()) {
            val = val * 10 + (c - '0');
        }
        out = val * sign;
        return true;
    }
};

struct FastOutput {
    static const size_t BUFSIZE = 1 << 20;
    std::string buf;
    void flush() {
        if (!buf.empty()) {
            fwrite(buf.data(), 1, buf.size(), stdout);
            buf.clear();
        }
    }
    ~FastOutput() { flush(); }

    void writeInt(long long x) {
        if (x == 0) {
            buf.push_back('0');
            return;
        }
        if (x < 0) {
            buf.push_back('-');
            x = -x;
        }
        char tmp[32];
        int len = 0;
        while (x) {
            tmp[len++] = char('0' + x % 10);
            x /= 10;
        }
        for (int i = len - 1; i >= 0; --i) buf.push_back(tmp[i]);
    }

    void writeLine(long long x) {
        writeInt(x);
        buf.push_back('\n');
        if (buf.size() > BUFSIZE) flush();
    }

    void writeEdge(uint32_t u, uint32_t v) {
        writeInt(u);
        buf.push_back(' ');
        writeInt(v);
        buf.push_back('\n');
        if (buf.size() > BUFSIZE) flush();
    }
};

struct Edge {
    uint32_t u, v;
};

struct Graph {
    int n;
    int originalN;
    std::vector<Edge> edges;
    std::vector<std::vector<int>> members;
};

struct CutResult {
    long long weight;
    std::vector<char> side;
};

std::pair<long long, std::vector<char>> stoerWagner(std::vector<std::vector<long long>> &w) {
    int n = (int)w.size();
    std::vector<int> v(n);
    for (int i = 0; i < n; ++i) v[i] = i;
    long long best = std::numeric_limits<long long>::max();
    std::vector<char> bestSide(n, false);

    for (int phase = n; phase > 1; --phase) {
        std::vector<long long> weights(n, 0);
        std::vector<char> added(n, false);
        int prev = -1, sel = -1;

        for (int i = 0; i < phase; ++i) {
            long long maxw = -1;
            sel = -1;
            for (int j = 0; j < phase; ++j) {
                int vtx = v[j];
                if (!added[vtx] && weights[vtx] > maxw) {
                    maxw = weights[vtx];
                    sel = vtx;
                }
            }
            if (sel == -1) break;
            added[sel] = true;

            if (i == phase - 1) {
                if (weights[sel] < best) {
                    best = weights[sel];
                    bestSide.assign(n, false);
                    for (int j = 0; j < phase; ++j) {
                        int vtx = v[j];
                        if (added[vtx] && vtx != sel)
                            bestSide[vtx] = true;
                    }
                }
                if (prev != -1) {
                    for (int j = 0; j < phase; ++j) {
                        int vtx = v[j];
                        if (vtx != sel) {
                            w[prev][vtx] += w[sel][vtx];
                            w[vtx][prev] = w[prev][vtx];
                        }
                    }
                }
                for (int j = 0; j < phase; ++j) {
                    if (v[j] == sel) {
                        v[j] = v[phase - 1];
                        break;
                    }
                }
            } else {
                for (int j = 0; j < phase; ++j) {
                    int vtx = v[j];
                    if (!added[vtx])
                        weights[vtx] += w[sel][vtx];
                }
                prev = sel;
            }
        }
    }

    return {best, bestSide};
}

CutResult exactMinCut(const Graph &g) {
    int n = g.n;
    std::vector<std::vector<long long>> w(n, std::vector<long long>(n, 0));
    for (auto &e : g.edges) {
        w[e.u][e.v]++;
        w[e.v][e.u]++;
    }
    auto [weight, sideSuper] = stoerWagner(w);
    std::vector<char> sideOrig(g.originalN, false);
    for (int i = 0; i < n; ++i) {
        if (sideSuper[i]) {
            for (int v : g.members[i])
                sideOrig[v] = true;
        }
    }
    return {weight, std::move(sideOrig)};
}

Graph contractGraph(const Graph &g, int target, std::mt19937_64 &rng) {
    int n = g.n;
    std::vector<int> parent(n), sz(n);
    std::vector<std::vector<int>> members = g.members;
    for (int i = 0; i < n; ++i) {
        parent[i] = i;
        sz[i] = 1;
    }
    int active = n;
    std::uniform_int_distribution<size_t> dist(0, g.edges.size() - 1);

    auto find = [&](int x) {
        while (parent[x] != x) {
            parent[x] = parent[parent[x]];
            x = parent[x];
        }
        return x;
    };

    while (active > target) {
        const Edge &e = g.edges[dist(rng)];
        int ru = find(e.u), rv = find(e.v);
        if (ru == rv) continue;
        if (sz[ru] < sz[rv]) std::swap(ru, rv);
        parent[rv] = ru;
        sz[ru] += sz[rv];
        if (!members[rv].empty()) {
            members[ru].insert(members[ru].end(), members[rv].begin(), members[rv].end());
            members[rv].clear();
        }
        active--;
    }

    std::vector<int> rootToNew(n, -1);
    std::vector<std::vector<int>> newMembers;
    newMembers.reserve(active);
    for (int i = 0; i < n; ++i) {
        if (parent[i] == i) {
            rootToNew[i] = (int)newMembers.size();
            newMembers.push_back(std::move(members[i]));
        }
    }

    std::vector<Edge> newEdges;
    newEdges.reserve(g.edges.size());
    for (auto &e : g.edges) {
        int ru = find(e.u), rv = find(e.v);
        if (ru == rv) continue;
        newEdges.push_back({(uint32_t)rootToNew[ru], (uint32_t)rootToNew[rv]});
    }

    Graph ng;
    ng.n = (int)newMembers.size();
    ng.originalN = g.originalN;
    ng.edges = std::move(newEdges);
    ng.members = std::move(newMembers);
    return ng;
}

CutResult kargerStein(const Graph &g, std::mt19937_64 &rng) {
    if (g.n <= 20) return exactMinCut(g);
    int t = (int)std::ceil(g.n / std::sqrt(2.0));
    Graph g1 = contractGraph(g, t, rng);
    Graph g2 = contractGraph(g, t, rng);
    CutResult c1 = kargerStein(g1, rng);
    CutResult c2 = kargerStein(g2, rng);
    return (c1.weight <= c2.weight) ? std::move(c1) : std::move(c2);
}

int main() {
    FastScanner fs;
    uint64_t N64, M64;
    if (!fs.readInt(N64) || !fs.readInt(M64)) return 0;
    int N = (int)N64;
    uint64_t M = M64;

    std::vector<Edge> edges;
    edges.reserve((size_t)M);
    for (uint64_t i = 0; i < M; ++i) {
        uint32_t u, v;
        fs.readInt(u);
        fs.readInt(v);
        edges.push_back({u, v});
    }

    std::vector<char> bestSide(N, false);

    if (N <= 4000) {
        std::vector<std::vector<long long>> w(N, std::vector<long long>(N, 0));
        for (auto &e : edges) {
            w[e.u][e.v]++;
            w[e.v][e.u]++;
        }
        auto [mincut, side] = stoerWagner(w);
        bestSide = std::move(side);
    } else {
        Graph g;
        g.n = N;
        g.originalN = N;
        g.edges = edges;
        g.members.resize(N);
        for (int i = 0; i < N; ++i) g.members[i].push_back(i);

        std::mt19937_64 rng((uint64_t)std::chrono::high_resolution_clock::now().time_since_epoch().count());
        int reps = std::max(1, (int)(2 * std::log2((double)N)));
        CutResult best{std::numeric_limits<long long>::max(), std::vector<char>()};

        for (int r = 0; r < reps; ++r) {
            CutResult cur = kargerStein(g, rng);
            if (cur.weight < best.weight) best = std::move(cur);
        }
        bestSide = std::move(best.side);
    }

    std::vector<Edge> cutEdges;
    cutEdges.reserve(edges.size());
    for (auto &e : edges) {
        if (bestSide[e.u] != bestSide[e.v]) {
            cutEdges.push_back(e);
        }
    }

    FastOutput out;
    out.writeLine((long long)cutEdges.size());
    for (auto &e : cutEdges) {
        out.writeEdge(e.u, e.v);
    }
    return 0;
}
```
