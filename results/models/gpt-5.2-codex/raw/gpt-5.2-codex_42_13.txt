OpCapability Shader
OpCapability Geometry
%GLSL450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3 = OpTypeArray %v3float %uint_6

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_half = OpConstant %float 0.5
%float_thickness = OpConstant %float 0.008

%main = OpFunction %void None %func
%entry = OpLabel
%out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Edge 0 -> 2
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%p1 = OpLoad %v3float %p1_ptr
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c_sum = OpFAdd %v3float %c0 %c1
%c_avg = OpVectorTimesScalar %v3float %c_sum %float_half
%dir0 = OpFSub %v3float %p1 %p0
%dir0_x = OpCompositeExtract %float %dir0 0
%dir0_y = OpCompositeExtract %float %dir0 1
%neg_dir0_y = OpFNegate %float %dir0_y
%perp0 = OpCompositeConstruct %v3float %neg_dir0_y %dir0_x %float_0
%perp0_n = OpExtInst %v3float %GLSL450 Normalize %perp0
%off0 = OpVectorTimesScalar %v3float %perp0_n %float_thickness
%v0 = OpFAdd %v3float %p0 %off0
%v1 = OpFSub %v3float %p0 %off0
%v2 = OpFAdd %v3float %p1 %off0
%v3 = OpFSub %v3float %p1 %off0

%pos0 = OpCompositeConstruct %v4float %v0 %float_1
OpStore %outColor %c_avg
OpStore %out_pos %pos0
OpEmitVertex
%pos1 = OpCompositeConstruct %v4float %v1 %float_1
OpStore %outColor %c_avg
OpStore %out_pos %pos1
OpEmitVertex
%pos2 = OpCompositeConstruct %v4float %v2 %float_1
OpStore %outColor %c_avg
OpStore %out_pos %pos2
OpEmitVertex
%pos3 = OpCompositeConstruct %v4float %v3 %float_1
OpStore %outColor %c_avg
OpStore %out_pos %pos3
OpEmitVertex
OpEndPrimitive

; Edge 2 -> 4
%p2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%p2 = OpLoad %v3float %p2_ptr
%p4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p4 = OpLoad %v3float %p4_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %c4_ptr
%c_sum2 = OpFAdd %v3float %c2 %c4
%c_avg2 = OpVectorTimesScalar %v3float %c_sum2 %float_half
%dir1 = OpFSub %v3float %p4 %p2
%dir1_x = OpCompositeExtract %float %dir1 0
%dir1_y = OpCompositeExtract %float %dir1 1
%neg_dir1_y = OpFNegate %float %dir1_y
%perp1 = OpCompositeConstruct %v3float %neg_dir1_y %dir1_x %float_0
%perp1_n = OpExtInst %v3float %GLSL450 Normalize %perp1
%off1 = OpVectorTimesScalar %v3float %perp1_n %float_thickness
%v4a = OpFAdd %v3float %p2 %off1
%v5a = OpFSub %v3float %p2 %off1
%v6a = OpFAdd %v3float %p4 %off1
%v7a = OpFSub %v3float %p4 %off1

%pos4 = OpCompositeConstruct %v4float %v4a %float_1
OpStore %outColor %c_avg2
OpStore %out_pos %pos4
OpEmitVertex
%pos5 = OpCompositeConstruct %v4float %v5a %float_1
OpStore %outColor %c_avg2
OpStore %out_pos %pos5
OpEmitVertex
%pos6 = OpCompositeConstruct %v4float %v6a %float_1
OpStore %outColor %c_avg2
OpStore %out_pos %pos6
OpEmitVertex
%pos7 = OpCompositeConstruct %v4float %v7a %float_1
OpStore %outColor %c_avg2
OpStore %out_pos %pos7
OpEmitVertex
OpEndPrimitive

; Edge 4 -> 0
%p4b_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p4b = OpLoad %v3float %p4b_ptr
%p0b_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0b = OpLoad %v3float %p0b_ptr
%c4b_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c4b = OpLoad %v3float %c4b_ptr
%c0b_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0b = OpLoad %v3float %c0b_ptr
%c_sum3 = OpFAdd %v3float %c4b %c0b
%c_avg3 = OpVectorTimesScalar %v3float %c_sum3 %float_half
%dir2 = OpFSub %v3float %p0b %p4b
%dir2_x = OpCompositeExtract %float %dir2 0
%dir2_y = OpCompositeExtract %float %dir2 1
%neg_dir2_y = OpFNegate %float %dir2_y
%perp2 = OpCompositeConstruct %v3float %neg_dir2_y %dir2_x %float_0
%perp2_n = OpExtInst %v3float %GLSL450 Normalize %perp2
%off2 = OpVectorTimesScalar %v3float %perp2_n %float_thickness
%v8a = OpFAdd %v3float %p4b %off2
%v9a = OpFSub %v3float %p4b %off2
%v10a = OpFAdd %v3float %p0b %off2
%v11a = OpFSub %v3float %p0b %off2

%pos8 = OpCompositeConstruct %v4float %v8a %float_1
OpStore %outColor %c_avg3
OpStore %out_pos %pos8
OpEmitVertex
%pos9 = OpCompositeConstruct %v4float %v9a %float_1
OpStore %outColor %c_avg3
OpStore %out_pos %pos9
OpEmitVertex
%pos10 = OpCompositeConstruct %v4float %v10a %float_1
OpStore %outColor %c_avg3
OpStore %out_pos %pos10
OpEmitVertex
%pos11 = OpCompositeConstruct %v4float %v11a %float_1
OpStore %outColor %c_avg3
OpStore %out_pos %pos11
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
