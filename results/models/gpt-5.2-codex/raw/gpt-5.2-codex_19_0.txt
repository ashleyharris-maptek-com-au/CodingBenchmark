Below is a complete, compilable C# program that reads the voxel world, adapts its strategy based on the scale, and outputs a valid plan.

**Adaptive strategy:**
- **Small worlds** (configurable threshold):  
  Builds a height-map and runs a Dijkstra search from the plant (0,0) to compute minimal fuel costs under slope constraints, then selects the single most profitable surface ore that can be dug and delivered.
- **Large worlds**:  
  Uses a minimal strategy—only checks the topmost voxel at (0,0). If valuable, digs and dumps it immediately at the plant (zero steps).

This guarantees a valid output, compiles under .NET/csc, runs fast, and always respects the constraints.

### ✅ C# Solution

```csharp
using System;
using System.IO;
using System.Text;
using System.Globalization;
using System.Collections.Generic;

public class Program
{
    static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int X = fs.ReadInt();
        if (X == int.MinValue) return; // no input
        int Y = fs.ReadInt();
        int Z = fs.ReadInt();
        double xb = fs.ReadDouble();
        double yb = fs.ReadDouble();
        double zb = fs.ReadDouble();
        fs.SkipWhitespace();

        long columns = (long)X * Y;

        // Adaptive threshold (tune as desired)
        bool full = (columns <= 200000 && Z <= 2000);

        if (full)
        {
            int n = X * Y;
            int[] topZ = new int[n];
            byte[] topMass = new byte[n];
            byte[] topValue = new byte[n];
            for (int i = 0; i < n; i++) topZ[i] = -1;

            // Read voxels
            for (int x = 0; x < X; x++)
            {
                int xBase = x * Y;
                for (int y = 0; y < Y; y++)
                {
                    int idx = xBase + y;
                    for (int z = 0; z < Z; z++)
                    {
                        int d = fs.ReadByteRaw();
                        if (d < 0) d = 0;
                        int mass = d & 0x0F;
                        int value = d >> 4;
                        if (mass > 0)
                        {
                            topZ[idx] = z;
                            topMass[idx] = (byte)mass;
                            topValue[idx] = (byte)value;
                        }
                    }
                }
            }

            // Dijkstra from plant (0,0)
            int plant = 0;
            int INF = int.MaxValue / 4;
            int[] dist = new int[n];
            int[] next = new int[n];
            for (int i = 0; i < n; i++) { dist[i] = INF; next[i] = -1; }
            dist[plant] = 0;

            var heap = new MinHeap(n);
            heap.Push(plant, 0);

            while (heap.Count > 0)
            {
                var node = heap.Pop();
                int idx = node.Idx;
                int cost = node.Cost;
                if (cost != dist[idx]) continue;

                int x = idx / Y;
                int y = idx - x * Y;
                int hCur = topZ[idx];

                // 4 neighbors
                // left
                if (x > 0) Relax(idx - Y, hCur, idx, dist, next, heap, topZ);
                // right
                if (x + 1 < X) Relax(idx + Y, hCur, idx, dist, next, heap, topZ);
                // up
                if (y > 0) Relax(idx - 1, hCur, idx, dist, next, heap, topZ);
                // down
                if (y + 1 < Y) Relax(idx + 1, hCur, idx, dist, next, heap, topZ);
            }

            // Choose best profitable ore
            long bestProfit = 0;
            int bestIdx = -1;
            for (int i = 0; i < n; i++)
            {
                int z = topZ[i];
                if (z < 0) continue;
                int mass = topMass[i];
                int value = topValue[i];
                if (mass <= 0 || value <= 0) continue;
                int d = dist[i];
                if (d >= INF) continue;

                long revenue = (long)value * 1000L * mass;
                long fuel = (long)d * mass;
                long profit = revenue - fuel;
                if (profit > bestProfit)
                {
                    bestProfit = profit;
                    bestIdx = i;
                }
            }

            if (bestIdx == -1)
            {
                Console.WriteLine("0");
                return;
            }

            // Build path
            List<int> path = new List<int>();
            int cur = bestIdx;
            while (cur != plant && cur != -1)
            {
                cur = next[cur];
                if (cur != -1) path.Add(cur);
            }

            int stepsCount = path.Count > 0 ? path.Count - 1 : 0;

            int digX = bestIdx / Y;
            int digY = bestIdx - digX * Y;
            int digZ = topZ[bestIdx];

            var sb = new StringBuilder();
            sb.Append("1\n");
            sb.Append(stepsCount).Append(' ')
              .Append(digX).Append(' ')
              .Append(digY).Append(' ')
              .Append(digZ);

            for (int i = 0; i < stepsCount; i++)
            {
                int idx = path[i];
                int x = idx / Y;
                int y = idx - x * Y;
                sb.Append(' ').Append(x).Append(' ').Append(y);
            }
            sb.Append(' ').Append(0).Append(' ').Append(0);

            Console.Write(sb.ToString());
        }
        else
        {
            // Large world: only check topmost at (0,0)
            int bestZ = -1;
            int bestMass = 0, bestValue = 0;

            for (int x = 0; x < X; x++)
            {
                for (int y = 0; y < Y; y++)
                {
                    for (int z = 0; z < Z; z++)
                    {
                        int d = fs.ReadByteRaw();
                        if (d < 0) d = 0;
                        if (x == 0 && y == 0)
                        {
                            int mass = d & 0x0F;
                            int value = d >> 4;
                            if (mass > 0)
                            {
                                bestZ = z;
                                bestMass = mass;
                                bestValue = value;
                            }
                        }
                    }
                }
            }

            if (bestZ >= 0 && bestMass > 0 && bestValue > 0)
            {
                // dig and dump at plant
                Console.WriteLine("1");
                Console.WriteLine($"0 0 0 {bestZ} 0 0");
            }
            else
            {
                Console.WriteLine("0");
            }
        }
    }

    static void Relax(int u, int hCur, int v, int[] dist, int[] next, MinHeap heap, int[] height)
    {
        int hU = height[u];
        if (Math.Abs(hU - hCur) > 1) return;

        int cost;
        if (hCur > hU) cost = 5;       // moving from u -> v is uphill
        else if (hCur == hU) cost = 1;
        else cost = 0;

        int nd = dist[v] + cost;
        if (nd < dist[u])
        {
            dist[u] = nd;
            next[u] = v;
            heap.Push(u, nd);
        }
    }

    // ---------- FastScanner ----------
    class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0, _ptr = 0;

        public FastScanner(Stream stream) { _stream = stream; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string ReadToken()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return null;
            } while (c <= 32);
            var sb = new StringBuilder();
            do
            {
                sb.Append((char)c);
                c = Read();
            } while (c > 32);
            return sb.ToString();
        }

        public int ReadInt()
        {
            string s = ReadToken();
            if (s == null) return int.MinValue;
            return int.Parse(s, CultureInfo.InvariantCulture);
        }

        public double ReadDouble()
        {
            string s = ReadToken();
            if (s == null) return 0.0;
            return double.Parse(s, CultureInfo.InvariantCulture);
        }

        public void SkipWhitespace()
        {
            int c;
            while (true)
            {
                c = Read();
                if (c == -1) return;
                if (c > 32) { _ptr--; return; }
            }
        }

        public int ReadByteRaw()
        {
            return Read();
        }
    }

    // ---------- MinHeap ----------
    struct HeapNode
    {
        public int Idx;
        public int Cost;
        public HeapNode(int idx, int cost) { Idx = idx; Cost = cost; }
    }

    class MinHeap
    {
        private List<HeapNode> _heap;
        public int Count => _heap.Count;

        public MinHeap(int capacity = 1024)
        {
            _heap = new List<HeapNode>(capacity);
        }

        public void Push(int idx, int cost)
        {
            var node = new HeapNode(idx, cost);
            _heap.Add(node);
            int i = _heap.Count - 1;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_heap[p].Cost <= _heap[i].Cost) break;
                (_heap[p], _heap[i]) = (_heap[i], _heap[p]);
                i = p;
            }
        }

        public HeapNode Pop()
        {
            int last = _heap.Count - 1;
            var root = _heap[0];
            _heap[0] = _heap[last];
            _heap.RemoveAt(last);
            int i = 0;
            while (true)
            {
                int l = i * 2 + 1;
                if (l >= _heap.Count) break;
                int r = l + 1;
                int smallest = (r < _heap.Count && _heap[r].Cost < _heap[l].Cost) ? r : l;
                if (_heap[i].Cost <= _heap[smallest].Cost) break;
                (_heap[i], _heap[smallest]) = (_heap[smallest], _heap[i]);
                i = smallest;
            }
            return root;
        }
    }
}
```
