Below is a **complete, compilable C++20 solution** that **adapts strategy** based on input scale and satisfies all requirements.  

✅ **Small case (2 strings, short)** → fast DP  
✅ **General / large / extreme** → **Suffix Automaton** on the *shortest string* (linear in total input size)  
✅ Handles huge input sizes and large N (100+ strings, 10MB+ each)

---

### ✅ C++20 Solution

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <limits>

using namespace std;

// ------------------ Small-case DP (2 strings only) ------------------
static string lcs2_dp(const string &a, const string &b) {
    if (a.empty() || b.empty()) return "";
    int n = (int)a.size(), m = (int)b.size();
    vector<int> dp(m + 1, 0);
    int best = 0, endPos = 0;

    for (int i = 1; i <= n; ++i) {
        for (int j = m; j >= 1; --j) {
            if (a[i - 1] == b[j - 1]) {
                dp[j] = dp[j - 1] + 1;
                if (dp[j] > best) {
                    best = dp[j];
                    endPos = i - 1;
                }
            } else {
                dp[j] = 0;
            }
        }
    }
    return a.substr(endPos - best + 1, best);
}

// ------------------ Suffix Automaton ------------------
class SuffixAutomaton {
public:
    struct State {
        int link;
        int len;
        int head;
        int firstpos;
    };
    struct Edge {
        int to;
        int next;
        unsigned char ch;
    };

    vector<State> st;
    vector<Edge> edges;
    int last;

    SuffixAutomaton(size_t maxLen = 0) { init(maxLen); }

    void init(size_t maxLen) {
        st.clear();
        edges.clear();
        st.reserve(2 * maxLen + 1);
        edges.reserve(4 * maxLen + 1);
        st.push_back({-1, 0, -1, -1});  // root
        last = 0;
    }

    inline int get_next(int v, unsigned char c) const {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) return edges[e].to;
        }
        return -1;
    }

    inline void add_edge(int v, unsigned char c, int to) {
        edges.push_back({to, st[v].head, c});
        st[v].head = (int)edges.size() - 1;
    }

    inline void set_edge(int v, unsigned char c, int to) {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                edges[e].to = to;
                return;
            }
        }
        add_edge(v, c, to);
    }

    void copy_edges(int from, int to) {
        for (int e = st[from].head; e != -1; e = edges[e].next) {
            add_edge(to, edges[e].ch, edges[e].to);
        }
    }

    void extend(unsigned char c, int pos) {
        int cur = (int)st.size();
        st.push_back({0, st[last].len + 1, -1, pos});

        int p = last;
        while (p != -1 && get_next(p, c) == -1) {
            add_edge(p, c, cur);
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = get_next(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back({st[q].link, st[p].len + 1, -1, st[q].firstpos});
                copy_edges(q, clone);

                while (p != -1 && get_next(p, c) == q) {
                    set_edge(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

// ------------------ Main ------------------
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');

    vector<string> strs;
    strs.reserve(N);

    for (int i = 0; i < N; ++i) {
        string s;
        getline(cin, s);
        if (!s.empty() && s.back() == '\r') s.pop_back();
        strs.push_back(s);
    }

    if (N == 0) {
        cout << "\n";
        return 0;
    }

    // If any string empty => no common substring
    for (auto &s : strs) {
        if (s.empty()) {
            cout << "\n";
            return 0;
        }
    }

    if (N == 1) {
        cout << strs[0] << "\n";
        return 0;
    }

    // --- Adaptive Strategy: Small 2-string DP ---
    if (N == 2) {
        long long len1 = strs[0].size(), len2 = strs[1].size();
        if (len1 * len2 <= 20000000LL) { // ~20M operations threshold
            // Use shorter string as "a" to reduce memory
            string a = strs[0], b = strs[1];
            if (b.size() < a.size()) swap(a, b);
            cout << lcs2_dp(a, b) << "\n";
            return 0;
        }
    }

    // --- Large / General Case: Suffix Automaton ---
    // Pick shortest string as base
    int base_idx = 0;
    for (int i = 1; i < N; ++i) {
        if (strs[i].size() < strs[base_idx].size()) base_idx = i;
    }
    if (base_idx != 0) swap(strs[0], strs[base_idx]);

    const string &base = strs[0];

    SuffixAutomaton sam(base.size());
    for (int i = 0; i < (int)base.size(); ++i) {
        sam.extend((unsigned char)base[i], i);
    }

    int sz = (int)sam.st.size();

    // build order by length
    int maxLen = 0;
    for (auto &st : sam.st) maxLen = max(maxLen, st.len);
    vector<int> cnt(maxLen + 1, 0);
    for (auto &st : sam.st) cnt[st.len]++;
    for (int i = 1; i <= maxLen; ++i) cnt[i] += cnt[i - 1];

    vector<int> order(sz);
    for (int i = 0; i < sz; ++i) {
        order[--cnt[sam.st[i].len]] = i;
    }

    vector<int> best(sz);
    for (int i = 0; i < sz; ++i) best[i] = sam.st[i].len;

    vector<int> mx(sz);

    for (int idx = 1; idx < N; ++idx) {
        fill(mx.begin(), mx.end(), 0);
        int v = 0, l = 0;

        for (unsigned char c : strs[idx]) {
            int nxt = sam.get_next(v, c);
            while (v != -1 && nxt == -1) {
                v = sam.st[v].link;
                if (v != -1) {
                    l = sam.st[v].len;
                    nxt = sam.get_next(v, c);
                }
            }
            if (v == -1) {
                v = 0;
                l = 0;
                continue;
            }
            v = nxt;
            l++;
            if (l > mx[v]) mx[v] = l;
        }

        // propagate
        for (int i = sz - 1; i > 0; --i) {
            int state = order[i];
            int link = sam.st[state].link;
            if (link >= 0) {
                int val = min(mx[state], sam.st[link].len);
                if (val > mx[link]) mx[link] = val;
            }
        }

        for (int i = 0; i < sz; ++i) {
            best[i] = min(best[i], mx[i]);
        }
    }

    int ansLen = 0, ansState = 0;
    for (int i = 1; i < sz; ++i) {
        if (best[i] > ansLen) {
            ansLen = best[i];
            ansState = i;
        }
    }

    if (ansLen == 0) {
        cout << "\n";
    } else {
        int endPos = sam.st[ansState].firstpos;
        cout << base.substr(endPos - ansLen + 1, ansLen) << "\n";
    }

    return 0;
}
```

---

### ✅ Why this works

- **Small inputs** → O(n·m) DP is simplest and fastest  
- **Large/extreme inputs** → SAM on the shortest string gives **O(total input size)** runtime  
- **Memory** scaled by base size (shortest string), making it feasible for massive input
