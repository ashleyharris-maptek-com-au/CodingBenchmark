Below is a complete C++20 program (MSVC compatible) that **constructs a valid Steiner tree** (not necessarily optimal) by:

1. Running Dijkstra from each terminal.
2. Building the MST over the **metric closure** of terminals.
3. Expanding MST edges into shortest paths.
4. Taking the union and running Kruskal to remove cycles.

This guarantees **connected + acyclic** edges containing all terminals.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <unordered_map>
#include <cstdint>
#include <limits>

using namespace std;

struct Edge {
    int to;
    long long w;
};

struct UEdge {
    int u, v;
    long long w;
};

struct DSU {
    vector<int> p, r;
    DSU(int n = 0) { init(n); }
    void init(int n) { p.resize(n); r.assign(n, 0); for (int i = 0; i < n; ++i) p[i] = i; }
    int find(int x) { return (p[x] == x) ? x : (p[x] = find(p[x])); }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) r[a]++;
        return true;
    }
};

const long long INF = (long long)4e18;

void dijkstra(int src, const vector<vector<Edge>>& adj,
              vector<long long>& dist,
              vector<int>& parent,
              vector<long long>& parentW) {
    int n = (int)adj.size();
    dist.assign(n, INF);
    parent.assign(n, -1);
    parentW.assign(n, 0);

    priority_queue<pair<long long,int>, vector<pair<long long,int>>, greater<pair<long long,int>>> pq;
    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, v] = pq.top(); pq.pop();
        if (d != dist[v]) continue;
        for (auto &e : adj[v]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parent[e.to] = v;
                parentW[e.to] = e.w;
                pq.push({nd, e.to});
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    // Dijkstra from each terminal
    vector<vector<long long>> distK(k);
    vector<vector<int>> parentK(k);
    vector<vector<long long>> parentWK(k);

    for (int i = 0; i < k; ++i) {
        dijkstra(terminals[i], adj, distK[i], parentK[i], parentWK[i]);
    }

    // Build distance matrix between terminals
    vector<vector<long long>> termDist(k, vector<long long>(k, INF));
    for (int i = 0; i < k; ++i)
        for (int j = 0; j < k; ++j)
            termDist[i][j] = distK[i][terminals[j]];

    // Prim MST on terminals
    vector<long long> key(k, INF);
    vector<int> mstParent(k, -1);
    vector<bool> inMST(k, false);
    key[0] = 0;

    for (int iter = 0; iter < k; ++iter) {
        long long best = INF;
        int u = -1;
        for (int i = 0; i < k; ++i)
            if (!inMST[i] && key[i] < best) {
                best = key[i];
                u = i;
            }
        if (u == -1) break;
        inMST[u] = true;

        for (int v = 0; v < k; ++v) {
            if (!inMST[v] && termDist[u][v] < key[v]) {
                key[v] = termDist[u][v];
                mstParent[v] = u;
            }
        }
    }

    // Collect union of paths
    unordered_map<uint64_t, long long> edgeMap;
    auto addEdge = [&](int a, int b, long long w) {
        if (a > b) swap(a, b);
        uint64_t key = (uint64_t(a) << 32) | uint32_t(b);
        auto it = edgeMap.find(key);
        if (it == edgeMap.end() || w < it->second)
            edgeMap[key] = w;
    };

    for (int v = 1; v < k; ++v) {
        int u = mstParent[v];
        if (u < 0) continue;
        int srcIdx = u;
        int dst = terminals[v];
        int src = terminals[u];

        int cur = dst;
        while (cur != src) {
            int p = parentK[srcIdx][cur];
            if (p == -1) break;
            long long w = parentWK[srcIdx][cur];
            addEdge(p, cur, w);
            cur = p;
        }
    }

    vector<UEdge> unionEdges;
    unionEdges.reserve(edgeMap.size());
    for (auto &kv : edgeMap) {
        uint64_t key = kv.first;
        int u = int(key >> 32);
        int v = int(key & 0xffffffffu);
        unionEdges.push_back({u, v, kv.second});
    }

    // Kruskal on union graph
    sort(unionEdges.begin(), unionEdges.end(),
         [](const UEdge& a, const UEdge& b){ return a.w < b.w; });

    DSU dsu(n);
    vector<UEdge> result;
    long long totalWeight = 0;

    for (auto &e : unionEdges) {
        if (dsu.unite(e.u, e.v)) {
            result.push_back(e);
            totalWeight += e.w;
        }
    }

    // Output
    cout << totalWeight << " " << result.size() << "\n";
    for (auto &e : result) {
        cout << e.u << " " << e.v << "\n";
    }
    return 0;
}
```

### Notes
- Produces a **valid Steiner tree** (connected, acyclic, includes all terminals).
- Uses only standard C++20 library (MSVC-friendly).
- Scales better than exact DP for large terminal sets.

If you'd like an exact solution for small terminal counts, I can also provide a Dreyfus–Wagner implementation with reconstruction.
