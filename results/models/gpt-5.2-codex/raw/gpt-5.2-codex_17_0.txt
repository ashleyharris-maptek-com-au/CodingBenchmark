Below is a complete, compilable C# solution that implements a fast, scalable 3D bin packing heuristic with axis-aligned boxes and allowed dimension permutations. It uses a free-space list and a guillotine split strategy, with automatic adaptation between Best‑Fit and First‑Fit for performance on large inputs.

```csharp
using System;
using System.Collections.Generic;
using System.Text;

public class Program
{
    struct Orientation
    {
        public int W, H, D;
        public Orientation(int w, int h, int d) { W = w; H = h; D = d; }
    }

    struct Box
    {
        public int Id;
        public int W, H, D;
        public long Volume;
        public Orientation[] Orients;
    }

    struct Space
    {
        public int X, Y, Z, W, H, D;
        public long Volume;
    }

    struct Placement
    {
        public int Id, X, Y, Z;
        public Placement(int id, int x, int y, int z) { Id = id; X = x; Y = y; Z = z; }
    }

    static void Main()
    {
        var fs = new FastScanner();
        int N = fs.NextInt();
        if (N <= 0) { Console.WriteLine(0); return; }
        int containerW = fs.NextInt();
        int containerH = fs.NextInt();
        int containerD = fs.NextInt();

        var boxes = new Box[N];
        for (int i = 0; i < N; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();
            boxes[i] = new Box
            {
                Id = i,
                W = w,
                H = h,
                D = d,
                Volume = (long)w * h * d,
                Orients = GenerateOrientations(w, h, d)
            };
        }

        Array.Sort(boxes, (a, b) =>
        {
            int cmp = a.Volume.CompareTo(b.Volume);
            if (cmp != 0) return cmp;
            return a.Id.CompareTo(b.Id);
        });

        bool useBestFit = N <= 5000; // adaptive speed/quality

        List<Space> spaces = new List<Space>(1024);
        spaces.Add(new Space { X = 0, Y = 0, Z = 0, W = containerW, H = containerH, D = containerD, Volume = (long)containerW * containerH * containerD });

        List<Placement> placements = new List<Placement>(N);

        foreach (var box in boxes)
        {
            int bestIndex = -1;
            Orientation bestOrient = default;
            long bestScore = long.MaxValue;
            int bestSideScore = int.MaxValue;

            for (int si = 0; si < spaces.Count; si++)
            {
                Space sp = spaces[si];
                if (sp.Volume < box.Volume) continue;

                foreach (var o in box.Orients)
                {
                    if (o.W <= sp.W && o.H <= sp.H && o.D <= sp.D)
                    {
                        if (!useBestFit)
                        {
                            bestIndex = si;
                            bestOrient = o;
                            goto PlaceFound;
                        }

                        long leftover = sp.Volume - box.Volume;
                        int sideScore = (sp.W - o.W) + (sp.H - o.H) + (sp.D - o.D);

                        if (leftover < bestScore || (leftover == bestScore && sideScore < bestSideScore))
                        {
                            bestScore = leftover;
                            bestSideScore = sideScore;
                            bestIndex = si;
                            bestOrient = o;
                        }
                    }
                }
            }

        PlaceFound:
            if (bestIndex >= 0)
            {
                Space sp = spaces[bestIndex];
                // Remove used space (swap last)
                int last = spaces.Count - 1;
                spaces[bestIndex] = spaces[last];
                spaces.RemoveAt(last);

                int bw = bestOrient.W;
                int bh = bestOrient.H;
                int bd = bestOrient.D;

                placements.Add(new Placement(box.Id, sp.X, sp.Y, sp.Z));

                // Split into new spaces
                AddSpace(spaces, new Space
                {
                    X = sp.X + bw,
                    Y = sp.Y,
                    Z = sp.Z,
                    W = sp.W - bw,
                    H = sp.H,
                    D = sp.D,
                    Volume = (long)(sp.W - bw) * sp.H * sp.D
                });

                AddSpace(spaces, new Space
                {
                    X = sp.X,
                    Y = sp.Y + bh,
                    Z = sp.Z,
                    W = bw,
                    H = sp.H - bh,
                    D = sp.D,
                    Volume = (long)bw * (sp.H - bh) * sp.D
                });

                AddSpace(spaces, new Space
                {
                    X = sp.X,
                    Y = sp.Y,
                    Z = sp.Z + bd,
                    W = bw,
                    H = bh,
                    D = sp.D - bd,
                    Volume = (long)bw * bh * (sp.D - bd)
                });
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(placements.Count.ToString());
        foreach (var p in placements)
            sb.AppendLine($"{p.Id} {p.X} {p.Y} {p.Z}");
        Console.Write(sb.ToString());
    }

    static Orientation[] GenerateOrientations(int w, int h, int d)
    {
        Orientation[] tmp = new Orientation[]
        {
            new Orientation(w,h,d),
            new Orientation(w,d,h),
            new Orientation(h,w,d),
            new Orientation(h,d,w),
            new Orientation(d,w,h),
            new Orientation(d,h,w)
        };

        List<Orientation> unique = new List<Orientation>(6);
        for (int i = 0; i < tmp.Length; i++)
        {
            bool exists = false;
            for (int j = 0; j < unique.Count; j++)
            {
                if (tmp[i].W == unique[j].W && tmp[i].H == unique[j].H && tmp[i].D == unique[j].D)
                {
                    exists = true;
                    break;
                }
            }
            if (!exists) unique.Add(tmp[i]);
        }
        return unique.ToArray();
    }

    static void AddSpace(List<Space> spaces, Space s)
    {
        if (s.W <= 0 || s.H <= 0 || s.D <= 0) return;

        for (int i = 0; i < spaces.Count; i++)
        {
            Space o = spaces[i];
            if (IsContained(s, o)) return;
            if (IsContained(o, s))
            {
                int last = spaces.Count - 1;
                spaces[i] = spaces[last];
                spaces.RemoveAt(last);
                i--;
            }
        }
        spaces.Add(s);
    }

    static bool IsContained(Space a, Space b)
    {
        return a.X >= b.X && a.Y >= b.Y && a.Z >= b.Z
            && a.X + a.W <= b.X + b.W
            && a.Y + a.H <= b.Y + b.H
            && a.Z + a.D <= b.Z + b.D;
    }

    // Fast input parser
    private class FastScanner
    {
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = Console.OpenStandardInput().Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= ' ' && c != -1);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
```

### Features
- **Adaptive packing strategy**: Best‑fit for smaller input, First‑fit for large input.
- **Rotation allowed**: All unique permutations of w/h/d are considered.
- **Fast and scalable**: Efficient space list with containment pruning.
- **Valid packing**: No overlaps, all boxes within bounds.

This solution is optimized for both large-scale scenarios and small containers, and will run efficiently under the provided environment constraints.
