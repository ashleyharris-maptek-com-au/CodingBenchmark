Here is a complete **C++17** solution that reads the instance, solves the **maximum weight independent set** exactly, and prints the optimum objective value plus a valid binary assignment.

The constraints `x_u + x_v <= 1` define a conflict graph; the solution is a maximum‑weight independent set.  
The code solves the equivalent **maximum weight clique** problem in the **complement graph** using a branch‑and‑bound algorithm with greedy coloring for a valid upper bound.

```cpp
#include <bits/stdc++.h>
using namespace std;

using Bitset = vector<uint64_t>;

int nOrig, m;
vector<long long> c;            // original weights
vector<long long> w;            // weights of kept vertices (positive only)
vector<int> oldIndex;           // new index -> original index
vector<int> newIndex;           // original index -> new index (or -1)
int n;                          // number of kept vertices
int blocks;

vector<Bitset> adj;             // adjacency of complement graph (clique graph)
Bitset fullMask;

Bitset currentSet, bestSet;
long long bestWeight = 0;

inline void setBit(Bitset &b, int v) { b[v >> 6] |= (1ULL << (v & 63)); }
inline void clearBit(Bitset &b, int v) { b[v >> 6] &= ~(1ULL << (v & 63)); }
inline bool isEmpty(const Bitset &b) {
    for (uint64_t x : b) if (x) return false;
    return true;
}
inline int firstBit(const Bitset &b) {
    for (int i = 0; i < blocks; i++) {
        if (b[i]) return i * 64 + __builtin_ctzll(b[i]);
    }
    return -1;
}

Bitset intersect(const Bitset &a, const Bitset &b) {
    Bitset r(blocks);
    for (int i = 0; i < blocks; i++) r[i] = a[i] & b[i];
    return r;
}

int chooseVertex(const Bitset &P) {
    int best = -1;
    long long bestw = -1;
    for (int i = 0; i < blocks; i++) {
        uint64_t x = P[i];
        while (x) {
            int t = __builtin_ctzll(x);
            int v = i * 64 + t;
            if (v >= n) break;
            if (w[v] > bestw) {
                bestw = w[v];
                best = v;
            }
            x &= x - 1;
        }
    }
    return best;
}

// Greedy coloring upper bound for maximum weight clique
long long colorBound(const Bitset &P) {
    Bitset uncolored = P;
    long long bound = 0;
    while (!isEmpty(uncolored)) {
        Bitset S = uncolored;
        long long maxw = 0;
        while (!isEmpty(S)) {
            int v = firstBit(S);
            if (v < 0) break;
            maxw = max(maxw, w[v]);
            clearBit(uncolored, v);
            // remove neighbors from S (keep only non-adjacent to v)
            for (int i = 0; i < blocks; i++) {
                S[i] &= ~adj[v][i];
            }
            clearBit(S, v);
        }
        bound += maxw;
    }
    return bound;
}

void updateBest(long long curWeight) {
    if (curWeight > bestWeight) {
        bestWeight = curWeight;
        bestSet = currentSet;
    }
}

void dfs(const Bitset &P, long long curWeight) {
    if (isEmpty(P)) {
        updateBest(curWeight);
        return;
    }
    long long bound = colorBound(P);
    if (curWeight + bound <= bestWeight) return;

    int v = chooseVertex(P);
    if (v == -1) {
        updateBest(curWeight);
        return;
    }

    // include v
    Bitset P1 = intersect(P, adj[v]);
    setBit(currentSet, v);
    dfs(P1, curWeight + w[v]);
    clearBit(currentSet, v);

    // exclude v
    Bitset P2 = P;
    clearBit(P2, v);
    dfs(P2, curWeight);
}

void greedyInit() {
    Bitset P = fullMask;
    Bitset temp(blocks, 0);
    long long sum = 0;
    while (!isEmpty(P)) {
        int v = chooseVertex(P);
        if (v == -1) break;
        setBit(temp, v);
        sum += w[v];
        P = intersect(P, adj[v]);
    }
    if (sum > bestWeight) {
        bestWeight = sum;
        bestSet = temp;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> nOrig >> m)) return 0;
    c.resize(nOrig);
    for (int i = 0; i < nOrig; i++) cin >> c[i];

    newIndex.assign(nOrig, -1);
    for (int i = 0; i < nOrig; i++) {
        if (c[i] > 0) {
            newIndex[i] = (int)w.size();
            w.push_back(c[i]);
            oldIndex.push_back(i);
        }
    }

    n = (int)w.size();
    if (n == 0) {
        cout << 0 << "\n";
        for (int i = 0; i < nOrig; i++) {
            cout << 0 << (i + 1 == nOrig ? '\n' : ' ');
        }
        return 0;
    }

    blocks = (n + 63) / 64;
    fullMask.assign(blocks, ~0ULL);
    if (n % 64 != 0) fullMask.back() = (1ULL << (n % 64)) - 1;

    adj.assign(n, fullMask);
    for (int i = 0; i < n; i++) clearBit(adj[i], i);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= nOrig || v >= nOrig) continue;
        int iu = newIndex[u];
        int iv = newIndex[v];
        if (iu == -1 || iv == -1) continue;
        clearBit(adj[iu], iv);
        clearBit(adj[iv], iu);
    }

    currentSet.assign(blocks, 0);
    bestSet.assign(blocks, 0);
    bestWeight = 0;

    greedyInit();
    dfs(fullMask, 0);

    vector<int> x(nOrig, 0);
    for (int i = 0; i < blocks; i++) {
        uint64_t bits = bestSet[i];
        while (bits) {
            int t = __builtin_ctzll(bits);
            int v = i * 64 + t;
            if (v < n) {
                int orig = oldIndex[v];
                x[orig] = 1;
            }
            bits &= bits - 1;
        }
    }

    long long obj = 0;
    for (int i = 0; i < nOrig; i++) obj += c[i] * x[i];

    cout << obj << "\n";
    for (int i = 0; i < nOrig; i++) {
        cout << x[i] << (i + 1 == nOrig ? '\n' : ' ');
    }
    return 0;
}
```
