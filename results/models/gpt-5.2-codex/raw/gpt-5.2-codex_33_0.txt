Below is a complete, compilable C# solution that reads the QAP instance, builds a good initial permutation using a common heuristic (flow–distance ranking), and then improves it with a fast local-search swap heuristic. It handles small and large N efficiently and always outputs a valid permutation.

```csharp
using System;
using System.IO;
using System.Text;
using System.Diagnostics;

public class Program
{
    // ---------- Fast Scanner ----------
    private class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0, _ptr = 0;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    static long ComputeCost(int[] p, int[][] flow, int[][] dist)
    {
        int n = p.Length;
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = p[i];
            int[] distRow = dist[pi];
            int[] flowRow = flow[i];
            for (int j = 0; j < n; j++)
                cost += (long)flowRow[j] * distRow[p[j]];
        }
        return cost;
    }

    static long DeltaSwap(int i, int j, int[] p, int[][] flow, int[][] dist)
    {
        if (i == j) return 0;
        int n = p.Length;
        int pi = p[i], pj = p[j];
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == i || k == j) continue;
            int pk = p[k];
            delta += (long)flow[i][k] * (dist[pj][pk] - dist[pi][pk]);
            delta += (long)flow[j][k] * (dist[pi][pk] - dist[pj][pk]);
            delta += (long)flow[k][i] * (dist[pk][pj] - dist[pk][pi]);
            delta += (long)flow[k][j] * (dist[pk][pi] - dist[pk][pj]);
        }

        delta += (long)flow[i][i] * (dist[pj][pj] - dist[pi][pi]);
        delta += (long)flow[j][j] * (dist[pi][pi] - dist[pj][pj]);
        delta += (long)flow[i][j] * (dist[pj][pi] - dist[pi][pj]);
        delta += (long)flow[j][i] * (dist[pi][pj] - dist[pj][pi]);

        return delta;
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        if (n == int.MinValue || n <= 0) return;

        int[][] flow = new int[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = new int[n];
            for (int j = 0; j < n; j++)
                flow[i][j] = fs.NextInt();
        }

        int[][] dist = new int[n][];
        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            for (int j = 0; j < n; j++)
                dist[i][j] = fs.NextInt();
        }

        // Compute flow sums (row+col) and distance sums
        long[] flowSum = new long[n];
        long[] distSum = new long[n];
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                flowSum[i] += flow[i][j];
                flowSum[j] += flow[i][j]; // column contribution
                distSum[i] += dist[i][j];
                distSum[j] += dist[i][j];
            }
        }

        // Sort facilities by descending flow, locations by ascending distance
        int[] fac = new int[n];
        int[] loc = new int[n];
        for (int i = 0; i < n; i++) { fac[i] = i; loc[i] = i; }

        Array.Sort(fac, (a, b) => flowSum[b].CompareTo(flowSum[a]));
        Array.Sort(loc, (a, b) => distSum[a].CompareTo(distSum[b]));

        int[] p = new int[n];
        for (int k = 0; k < n; k++)
            p[fac[k]] = loc[k];

        long cost = ComputeCost(p, flow, dist);

        // Local improvement
        Random rng = new Random(1);
        if (n <= 200)
        {
            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < n - 1; i++)
                {
                    for (int j = i + 1; j < n; j++)
                    {
                        long delta = DeltaSwap(i, j, p, flow, dist);
                        if (delta < 0)
                        {
                            int tmp = p[i]; p[i] = p[j]; p[j] = tmp;
                            cost += delta;
                            improved = true;
                        }
                    }
                }
            }
        }
        else
        {
            Stopwatch sw = Stopwatch.StartNew();
            double timeLimit = n <= 500 ? 5.0 : 3.0;
            while (sw.Elapsed.TotalSeconds < timeLimit)
            {
                int i = rng.Next(n);
                int j = rng.Next(n - 1);
                if (j >= i) j++;
                long delta = DeltaSwap(i, j, p, flow, dist);
                if (delta < 0)
                {
                    int tmp = p[i]; p[i] = p[j]; p[j] = tmp;
                    cost += delta;
                }
            }
        }

        // Output
        StringBuilder sb = new StringBuilder();
        sb.AppendLine(cost.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(p[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
```

**Notes:**
- Always prints a valid permutation.
- Uses a fast heuristic and local search, suitable for N up to ~2000.
- For small N it performs stronger improvement; for large N it uses randomized swap hill-climbing.
- Uses only the standard library.
