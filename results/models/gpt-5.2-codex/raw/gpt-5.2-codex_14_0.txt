Below is a complete, compilable C# Minesweeper solver that:

✅ Handles tiny to enormous boards  
✅ Uses fast logical propagation for all sizes  
✅ Adds exact CSP reasoning for small/medium boards  
✅ Never guesses  
✅ Only outputs provably safe unknown cells

---

### ✅ **C# Solution (Compilable, Adaptive, Efficient)**

```csharp
using System;
using System.Collections.Generic;
using System.Text;

public class Program
{
    static int H, W;
    static char[] board;
    static Queue<int> queue;
    static bool[] inQueue;
    static List<int> safeCells = new List<int>();

    static bool IsNumber(char c) => c >= '0' && c <= '8';

    static void Enqueue(int idx)
    {
        if (!inQueue[idx])
        {
            inQueue[idx] = true;
            queue.Enqueue(idx);
        }
    }

    static void AddNeighborNumbers(int idx)
    {
        int x = idx % W;
        int y = idx / W;
        int y0 = Math.Max(0, y - 1);
        int y1 = Math.Min(H - 1, y + 1);
        int x0 = Math.Max(0, x - 1);
        int x1 = Math.Min(W - 1, x + 1);

        for (int ny = y0; ny <= y1; ny++)
        {
            int row = ny * W;
            for (int nx = x0; nx <= x1; nx++)
            {
                if (nx == x && ny == y) continue;
                int nidx = row + nx;
                if (IsNumber(board[nidx]))
                    Enqueue(nidx);
            }
        }
    }

    static void SetSafe(int idx)
    {
        if (board[idx] != ' ') return;
        board[idx] = 's';
        safeCells.Add(idx);
        AddNeighborNumbers(idx);
    }

    static void SetMine(int idx)
    {
        if (board[idx] != ' ') return;
        board[idx] = '*';
        AddNeighborNumbers(idx);
    }

    static bool BasicPropagation()
    {
        bool changed = false;

        while (queue.Count > 0)
        {
            int idx = queue.Dequeue();
            inQueue[idx] = false;
            char c = board[idx];
            if (!IsNumber(c)) continue;

            int number = c - '0';
            int mines = 0;
            int unknown = 0;

            int x = idx % W;
            int y = idx / W;
            int y0 = Math.Max(0, y - 1);
            int y1 = Math.Min(H - 1, y + 1);
            int x0 = Math.Max(0, x - 1);
            int x1 = Math.Min(W - 1, x + 1);

            for (int ny = y0; ny <= y1; ny++)
            {
                int row = ny * W;
                for (int nx = x0; nx <= x1; nx++)
                {
                    if (nx == x && ny == y) continue;
                    char nc = board[row + nx];
                    if (nc == '*') mines++;
                    else if (nc == ' ') unknown++;
                }
            }

            if (unknown == 0) continue;

            if (number == mines)
            {
                for (int ny = y0; ny <= y1; ny++)
                {
                    int row = ny * W;
                    for (int nx = x0; nx <= x1; nx++)
                    {
                        if (nx == x && ny == y) continue;
                        int nidx = row + nx;
                        if (board[nidx] == ' ')
                        {
                            SetSafe(nidx);
                            changed = true;
                        }
                    }
                }
            }
            else if (number == mines + unknown)
            {
                for (int ny = y0; ny <= y1; ny++)
                {
                    int row = ny * W;
                    for (int nx = x0; nx <= x1; nx++)
                    {
                        if (nx == x && ny == y) continue;
                        int nidx = row + nx;
                        if (board[nidx] == ' ')
                        {
                            SetMine(nidx);
                            changed = true;
                        }
                    }
                }
            }
        }

        return changed;
    }

    class Constraint
    {
        public int[] Vars;
        public int Required;
    }

    static bool ExactSolver()
    {
        List<Constraint> constraints = new List<Constraint>();
        Dictionary<int, int> varIndex = new Dictionary<int, int>();
        List<int> varCells = new List<int>();

        for (int y = 0; y < H; y++)
        {
            int row = y * W;
            for (int x = 0; x < W; x++)
            {
                char c = board[row + x];
                if (!IsNumber(c)) continue;

                int number = c - '0';
                int mines = 0;
                List<int> unknowns = new List<int>();

                int y0 = Math.Max(0, y - 1);
                int y1 = Math.Min(H - 1, y + 1);
                int x0 = Math.Max(0, x - 1);
                int x1 = Math.Min(W - 1, x + 1);

                for (int ny = y0; ny <= y1; ny++)
                {
                    int r2 = ny * W;
                    for (int nx = x0; nx <= x1; nx++)
                    {
                        if (nx == x && ny == y) continue;
                        int nidx = r2 + nx;
                        char nc = board[nidx];
                        if (nc == '*') mines++;
                        else if (nc == ' ')
                        {
                            if (!varIndex.TryGetValue(nidx, out int vi))
                            {
                                vi = varCells.Count;
                                varIndex[nidx] = vi;
                                varCells.Add(nidx);
                            }
                            unknowns.Add(vi);
                        }
                    }
                }

                if (unknowns.Count > 0)
                {
                    int required = number - mines;
                    if (required >= 0 && required <= unknowns.Count)
                    {
                        constraints.Add(new Constraint { Vars = unknowns.ToArray(), Required = required });
                    }
                }
            }
        }

        int varCount = varCells.Count;
        if (varCount == 0 || constraints.Count == 0) return false;

        List<int>[] varCons = new List<int>[varCount];
        for (int i = 0; i < varCount; i++) varCons[i] = new List<int>();
        for (int i = 0; i < constraints.Count; i++)
        {
            foreach (int v in constraints[i].Vars)
                varCons[v].Add(i);
        }

        bool[] varVisited = new bool[varCount];
        bool[] consVisited = new bool[constraints.Count];
        bool changed = false;

        const int MAX_COMPONENT = 20;

        for (int v0 = 0; v0 < varCount; v0++)
        {
            if (varVisited[v0]) continue;

            Stack<int> stack = new Stack<int>();
            stack.Push(v0);
            varVisited[v0] = true;

            List<int> compVars = new List<int>();
            List<int> compCons = new List<int>();

            while (stack.Count > 0)
            {
                int v = stack.Pop();
                compVars.Add(v);
                foreach (int ci in varCons[v])
                {
                    if (!consVisited[ci])
                    {
                        consVisited[ci] = true;
                        compCons.Add(ci);
                    }
                    foreach (int v2 in constraints[ci].Vars)
                    {
                        if (!varVisited[v2])
                        {
                            varVisited[v2] = true;
                            stack.Push(v2);
                        }
                    }
                }
            }

            foreach (int ci in compCons) consVisited[ci] = false;

            if (compVars.Count > MAX_COMPONENT) continue;

            int k = compVars.Count;
            int m = compCons.Count;

            int[] globalToLocal = new int[varCount];
            for (int i = 0; i < varCount; i++) globalToLocal[i] = -1;
            for (int i = 0; i < k; i++) globalToLocal[compVars[i]] = i;

            int[][] consVars = new int[m][];
            int[] consReq = new int[m];
            int[] consTotal = new int[m];

            for (int i = 0; i < m; i++)
            {
                var cons = constraints[compCons[i]];
                int[] vlist = new int[cons.Vars.Length];
                for (int j = 0; j < cons.Vars.Length; j++)
                    vlist[j] = globalToLocal[cons.Vars[j]];
                consVars[i] = vlist;
                consReq[i] = cons.Required;
                consTotal[i] = vlist.Length;
            }

            List<int>[] localVarCons = new List<int>[k];
            for (int i = 0; i < k; i++) localVarCons[i] = new List<int>();
            for (int i = 0; i < m; i++)
                foreach (int v in consVars[i])
                    localVarCons[v].Add(i);

            int[] order = new int[k];
            for (int i = 0; i < k; i++) order[i] = i;
            Array.Sort(order, (a, b) => localVarCons[b].Count - localVarCons[a].Count);

            bool[] possibleSafe = new bool[k];
            bool[] possibleMine = new bool[k];
            int[] assign = new int[k];
            for (int i = 0; i < k; i++) assign[i] = -1;
            int[] consAssigned = new int[m];
            int[] consMines = new int[m];

            bool stop = false;
            bool anySolution = false;

            bool TryAssign(int v, int val)
            {
                foreach (int ci in localVarCons[v])
                {
                    consAssigned[ci]++;
                    if (val == 1) consMines[ci]++;
                }

                foreach (int ci in localVarCons[v])
                {
                    if (consMines[ci] > consReq[ci]) { Undo(v, val); return false; }
                    int rem = consTotal[ci] - consAssigned[ci];
                    if (consMines[ci] + rem < consReq[ci]) { Undo(v, val); return false; }
                }
                return true;
            }

            void Undo(int v, int val)
            {
                foreach (int ci in localVarCons[v])
                {
                    consAssigned[ci]--;
                    if (val == 1) consMines[ci]--;
                }
            }

            void Search(int depth)
            {
                if (stop) return;
                if (depth == k)
                {
                    anySolution = true;
                    for (int i = 0; i < k; i++)
                    {
                        if (assign[i] == 1) possibleMine[i] = true;
                        else possibleSafe[i] = true;
                    }

                    bool all = true;
                    for (int i = 0; i < k; i++)
                    {
                        if (!(possibleMine[i] && possibleSafe[i])) { all = false; break; }
                    }
                    if (all) stop = true;
                    return;
                }

                int v = order[depth];
                for (int val = 0; val <= 1; val++)
                {
                    if (TryAssign(v, val))
                    {
                        assign[v] = val;
                        Search(depth + 1);
                        assign[v] = -1;
                        Undo(v, val);
                        if (stop) return;
                    }
                }
            }

            Search(0);
            if (!anySolution) continue;

            for (int i = 0; i < k; i++)
            {
                int idx = varCells[compVars[i]];
                if (possibleSafe[i] && !possibleMine[i])
                {
                    SetSafe(idx);
                    changed = true;
                }
                else if (possibleMine[i] && !possibleSafe[i])
                {
                    SetMine(idx);
                    changed = true;
                }
            }
        }

        return changed;
    }

    public static void Main()
    {
        string first = Console.ReadLine();
        if (first == null) return;
        string[] parts = first.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        board = new char[H * W];

        for (int y = 0; y < H; y++)
        {
            string line = Console.ReadLine() ?? "";
            if (line.Length < W) line = line.PadRight(W, ' ');
            for (int x = 0; x < W; x++)
            {
                board[y * W + x] = line[x];
            }
        }

        queue = new Queue<int>();
        inQueue = new bool[H * W];

        // Initialize queue with number cells that have unknown neighbors
        for (int y = 0; y < H; y++)
        {
            int row = y * W;
            for (int x = 0; x < W; x++)
            {
                int idx = row + x;
                char c = board[idx];
                if (!IsNumber(c)) continue;

                bool hasUnknown = false;
                int y0 = Math.Max(0, y - 1);
                int y1 = Math.Min(H - 1, y + 1);
                int x0 = Math.Max(0, x - 1);
                int x1 = Math.Min(W - 1, x + 1);
                for (int ny = y0; ny <= y1 && !hasUnknown; ny++)
                {
                    int r2 = ny * W;
                    for (int nx = x0; nx <= x1; nx++)
                    {
                        if (nx == x && ny == y) continue;
                        if (board[r2 + nx] == ' ')
                        {
                            hasUnknown = true;
                            break;
                        }
                    }
                }
                if (hasUnknown)
                    Enqueue(idx);
            }
        }

        long totalCells = (long)H * W;
        bool allowExact = totalCells <= 2_000_000;

        bool changed;
        int iterations = 0;

        do
        {
            changed = BasicPropagation();
            if (allowExact)
            {
                if (ExactSolver()) changed = true;
            }
            iterations++;
        }
        while (changed && iterations < 10);

        StringBuilder sb = new StringBuilder();
        sb.AppendLine(safeCells.Count.ToString());
        foreach (int idx in safeCells)
        {
            int x = idx % W;
            int y = idx / W;
            sb.Append(x).Append(' ').Append(y).AppendLine();
        }
        Console.Write(sb.ToString());
    }
}
```
