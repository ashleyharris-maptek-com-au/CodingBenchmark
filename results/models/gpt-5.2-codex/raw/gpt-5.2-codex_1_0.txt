Below is a **complete, compilable C++20 TSP solver** that adapts its strategy based on the number of cities:

- **N ≤ 10** → brute-force optimal
- **N ≤ 2000** → nearest-neighbor heuristic
- **N ≤ 5,000,000** → sort by x (space-filling-like order)
- **N > 5,000,000** → output input order (fastest possible)

It uses **fast I/O**, **minimal memory**, and always outputs a valid permutation starting at city **0**.

---

### ✅ C++20 Code (MSVC compatible)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdint>
#include <cstring>

class FastScanner {
    static const size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx, size;
public:
    FastScanner() : idx(0), size(0) {}
    inline char getChar() {
        if (idx >= size) {
            size = fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }
    long long nextLong() {
        char c;
        do {
            c = getChar();
            if (!c) return 0;
        } while (c <= ' ');
        int sign = 1;
        if (c == '-') { sign = -1; c = getChar(); }
        long long val = 0;
        while (c > ' ') {
            val = val * 10 + (c - '0');
            c = getChar();
        }
        return val * sign;
    }
    double nextDouble() {
        char c;
        do {
            c = getChar();
            if (!c) return 0.0;
        } while (c <= ' ');
        int sign = 1;
        if (c == '-') { sign = -1; c = getChar(); }
        double val = 0.0;
        while (c >= '0' && c <= '9') {
            val = val * 10.0 + (c - '0');
            c = getChar();
        }
        double frac = 0.0, base = 1.0;
        if (c == '.') {
            c = getChar();
            while (c >= '0' && c <= '9') {
                base *= 0.1;
                frac += (c - '0') * base;
                c = getChar();
            }
        }
        return (val + frac) * sign;
    }
};

class FastOutput {
    static const size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx;
public:
    FastOutput() : idx(0) {}
    ~FastOutput() { flush(); }
    inline void flush() {
        if (idx) fwrite(buf, 1, idx, stdout);
        idx = 0;
    }
    inline void writeChar(char c) {
        if (idx >= BUFSIZE) flush();
        buf[idx++] = c;
    }
    inline void writeInt(long long v) {
        if (v == 0) { writeChar('0'); return; }
        if (v < 0) { writeChar('-'); v = -v; }
        char tmp[24];
        int len = 0;
        while (v > 0) {
            tmp[len++] = char('0' + (v % 10));
            v /= 10;
        }
        for (int i = len - 1; i >= 0; --i) writeChar(tmp[i]);
    }
    inline void writeIntSep(long long v, char sep) {
        writeInt(v);
        writeChar(sep);
    }
};

struct Node {
    double x, y;
    int idx;
};

int main() {
    FastScanner fs;
    FastOutput fo;

    long long Nll = fs.nextLong();
    if (Nll <= 0) return 0;
    int N = (int)Nll;

    const int HUGE_THRESHOLD = 5000000;

    // Extremely large case: no storage, output sequential order
    if (Nll > HUGE_THRESHOLD) {
        for (long long i = 0; i < Nll; ++i) {
            fs.nextDouble();
            fs.nextDouble();
        }
        for (long long i = 0; i < Nll; ++i) {
            fo.writeIntSep(i, (i + 1 == Nll) ? '\n' : ' ');
        }
        return 0;
    }

    // Read all points (for small to large manageable N)
    std::vector<Node> nodes(N);
    for (int i = 0; i < N; ++i) {
        nodes[i].x = fs.nextDouble();
        nodes[i].y = fs.nextDouble();
        nodes[i].idx = i;
    }

    // Case 1: brute force for very small N
    if (N <= 10) {
        std::vector<int> perm;
        for (int i = 1; i < N; ++i) perm.push_back(i);

        auto dist = [&](int a, int b) {
            double dx = nodes[a].x - nodes[b].x;
            double dy = nodes[a].y - nodes[b].y;
            return std::sqrt(dx * dx + dy * dy);
        };

        double bestLen = 1e100;
        std::vector<int> bestPerm;

        do {
            double len = dist(0, perm[0]);
            for (int i = 1; i < (int)perm.size(); ++i)
                len += dist(perm[i - 1], perm[i]);
            len += dist(perm.back(), 0);

            if (len < bestLen) {
                bestLen = len;
                bestPerm = perm;
            }
        } while (std::next_permutation(perm.begin(), perm.end()));

        fo.writeIntSep(0, (N == 1 ? '\n' : ' '));
        for (int i = 0; i < (int)bestPerm.size(); ++i) {
            fo.writeIntSep(bestPerm[i], (i + 1 == (int)bestPerm.size()) ? '\n' : ' ');
        }
        return 0;
    }

    // Case 2: Nearest Neighbor for small N
    if (N <= 2000) {
        std::vector<int> route;
        std::vector<char> used(N, 0);
        route.reserve(N);
        int cur = 0;
        used[0] = 1;
        route.push_back(0);

        for (int step = 1; step < N; ++step) {
            int best = -1;
            double bestDist = 1e100;
            for (int i = 0; i < N; ++i) {
                if (!used[i]) {
                    double dx = nodes[cur].x - nodes[i].x;
                    double dy = nodes[cur].y - nodes[i].y;
                    double d = dx * dx + dy * dy;
                    if (d < bestDist) {
                        bestDist = d;
                        best = i;
                    }
                }
            }
            used[best] = 1;
            route.push_back(best);
            cur = best;
        }

        for (int i = 0; i < N; ++i) {
            fo.writeIntSep(route[i], (i + 1 == N) ? '\n' : ' ');
        }
        return 0;
    }

    // Case 3: Sort by X then Y for larger sizes
    std::sort(nodes.begin(), nodes.end(), [](const Node &a, const Node &b) {
        if (a.x != b.x) return a.x < b.x;
        return a.y < b.y;
    });

    int pos0 = 0;
    for (int i = 0; i < N; ++i) {
        if (nodes[i].idx == 0) { pos0 = i; break; }
    }

    fo.writeIntSep(0, (N == 1 ? '\n' : ' '));
    for (int i = pos0 + 1; i < N; ++i)
        fo.writeIntSep(nodes[i].idx, ' ');
    for (int i = 0; i < pos0; ++i)
        fo.writeIntSep(nodes[i].idx, (i + 1 == pos0) ? '\n' : ' ');

    return 0;
}
```
