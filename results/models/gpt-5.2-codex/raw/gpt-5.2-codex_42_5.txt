OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_75 = OpConstant %float 0.75
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v3 = OpTypePointer Input %v3float
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %functype
%label = OpLabel

%p0ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0 = OpLoad %v3float %p0ptr
%p1ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p1 = OpLoad %v3float %p1ptr

%c0ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0ptr
%c1ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1ptr

%dir = OpFSub %v3float %p1 %p0
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp2 = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%len = OpExtInst %float %glsl450 Length %perp2
%inv_len = OpFDiv %float %float_1 %len
%perp2n = OpVectorTimesScalar %v2float %perp2 %inv_len
%perp_x = OpCompositeExtract %float %perp2n 0
%perp_y = OpCompositeExtract %float %perp2n 1
%perp3 = OpCompositeConstruct %v3float %perp_x %perp_y %float_0

%dir_scaled = OpVectorTimesScalar %v3float %dir %float_0_75
%mid = OpFAdd %v3float %p0 %dir_scaled

%off_shaft = OpVectorTimesScalar %v3float %perp3 %float_0_015
%p0_plus = OpFAdd %v3float %p0 %off_shaft
%p0_minus = OpFSub %v3float %p0 %off_shaft
%mid_plus = OpFAdd %v3float %mid %off_shaft
%mid_minus = OpFSub %v3float %mid %off_shaft

%pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%p0_plus4 = OpCompositeConstruct %v4float %p0_plus %float_1
OpStore %pos_ptr %p0_plus4
OpStore %outColor %c0
OpEmitVertex

%p0_minus4 = OpCompositeConstruct %v4float %p0_minus %float_1
OpStore %pos_ptr %p0_minus4
OpStore %outColor %c0
OpEmitVertex

%mid_plus4 = OpCompositeConstruct %v4float %mid_plus %float_1
OpStore %pos_ptr %mid_plus4
OpStore %outColor %c0
OpEmitVertex

%mid_minus4 = OpCompositeConstruct %v4float %mid_minus %float_1
OpStore %pos_ptr %mid_minus4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

%off_arrow = OpVectorTimesScalar %v3float %perp3 %float_0_05
%mid_plus_a = OpFAdd %v3float %mid %off_arrow
%mid_minus_a = OpFSub %v3float %mid %off_arrow

%mid_plus4a = OpCompositeConstruct %v4float %mid_plus_a %float_1
OpStore %pos_ptr %mid_plus4a
OpStore %outColor %c1
OpEmitVertex

%mid_minus4a = OpCompositeConstruct %v4float %mid_minus_a %float_1
OpStore %pos_ptr %mid_minus4a
OpStore %outColor %c1
OpEmitVertex

%p1_4 = OpCompositeConstruct %v4float %p1 %float_1
OpStore %pos_ptr %p1_4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
