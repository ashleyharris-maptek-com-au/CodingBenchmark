OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%glsl450 = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%fn = OpTypeFunction %void

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0_3333333 = OpConstant %float 0.333333333

%v3_ones = OpConstantComposite %v3float %float_1 %float_1 %float_1

%main = OpFunction %void None %fn
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%p2 = OpLoad %v3float %p2_ptr
%p4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p4 = OpLoad %v3float %p4_ptr

%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0

%a_x = OpCompositeExtract %float %a 0
%a_y = OpCompositeExtract %float %a 1
%a_z = OpCompositeExtract %float %a 2
%b_x = OpCompositeExtract %float %b 0
%b_y = OpCompositeExtract %float %b 1
%b_z = OpCompositeExtract %float %b 2

%mul1 = OpFMul %float %a_y %b_z
%mul2 = OpFMul %float %a_z %b_y
%cross_x = OpFSub %float %mul1 %mul2

%mul3 = OpFMul %float %a_z %b_x
%mul4 = OpFMul %float %a_x %b_z
%cross_y = OpFSub %float %mul3 %mul4

%mul5 = OpFMul %float %a_x %b_y
%mul6 = OpFMul %float %a_y %b_x
%cross_z = OpFSub %float %mul5 %mul6

%cross = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z
%N = OpExtInst %v3float %glsl450 Normalize %cross
%L = OpExtInst %v3float %glsl450 Normalize %v3_ones
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl450 FMax %dotNL %float_0

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %c4_ptr

%sum01 = OpFAdd %v3float %c0 %c2
%sum = OpFAdd %v3float %sum01 %c4
%base = OpVectorTimesScalar %v3float %sum %float_0_3333333

%diffuse_scaled = OpFMul %float %diffuse %float_0_8
%scale = OpFAdd %float %diffuse_scaled %float_0_2
%finalColor = OpVectorTimesScalar %v3float %base %scale

; emit vertex 0
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %pos0
OpStore %outColor %finalColor
OpEmitVertex

; emit vertex 2
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
OpStore %out_pos_ptr %pos2
OpStore %outColor %finalColor
OpEmitVertex

; emit vertex 4
%pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr
OpStore %out_pos_ptr %pos4
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
