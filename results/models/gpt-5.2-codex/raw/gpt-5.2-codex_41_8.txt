; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%funcType = OpTypeFunction %void

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_75 = OpConstant %float 0.75
%float_0_5 = OpConstant %float 0.5
%float_0_25 = OpConstant %float 0.25
%float_0_7 = OpConstant %float 0.7
%float_0_4 = OpConstant %float 0.4
%float_0_2 = OpConstant %float 0.2

%int_3 = OpConstant %int 3

%inWorldPos = OpVariable %_ptr_Input_v3float Input
%inNormal = OpVariable %_ptr_Input_v3float Input
%inColor = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %funcType
%entry = OpLabel
%wp = OpLoad %v3float %inWorldPos
%norm = OpLoad %v3float %inNormal
%col = OpLoad %v3float %inColor
%lpPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lp = OpLoad %v4float %lpPtr
%lp_xyz = OpVectorShuffle %v3float %lp %lp 0 1 2
%Lvec = OpFSub %v3float %lp_xyz %wp
%L = OpExtInst %v3float %glsl Normalize %Lvec
%N = OpExtInst %v3float %glsl Normalize %norm
%dot = OpDot %float %N %L
%d = OpExtInst %float %glsl FMax %dot %float_0
%cmp1 = OpFOrdGreaterThanEqual %bool %d %float_0_75
%cmp2 = OpFOrdGreaterThanEqual %bool %d %float_0_5
%cmp3 = OpFOrdGreaterThanEqual %bool %d %float_0_25
%val2 = OpSelect %float %cmp2 %float_0_7 %float_0_4
%val3 = OpSelect %float %cmp3 %val2 %float_0_2
%band = OpSelect %float %cmp1 %float_1 %val3
%bandVec = OpCompositeConstruct %v3float %band %band %band
%colMul = OpFMul %v3float %col %bandVec
%colx = OpCompositeExtract %float %colMul 0
%coly = OpCompositeExtract %float %colMul 1
%colz = OpCompositeExtract %float %colMul 2
%outv = OpCompositeConstruct %v4float %colx %coly %colz %float_1
OpStore %outColor %outv
OpReturn
OpFunctionEnd
