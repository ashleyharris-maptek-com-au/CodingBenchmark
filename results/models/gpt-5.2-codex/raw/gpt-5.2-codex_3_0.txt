Below is a complete Rust solver that adapts its layout strategy based on component size and graph scale.  
It uses:

- **Union-Find** to detect components (works for massive graphs)
- **Force-directed (Fruchterman–Reingold)** for small components (≤2000 nodes)
- **Spiral/Fibonacci layout** for large components (fast and scalable)
- **Component packing** to avoid overlap

It handles self-loops and parallel edges safely and runs fast even for very large inputs.

```rust
use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read, Write};

#[derive(Clone)]
struct DSU {
    parent: Vec<usize>,
    rank: Vec<u8>,
}
impl DSU {
    fn new(n: usize) -> Self {
        Self { parent: (0..n).collect(), rank: vec![0; n] }
    }
    fn find(&mut self, x: usize) -> usize {
        if self.parent[x] != x {
            let root = self.find(self.parent[x]);
            self.parent[x] = root;
        }
        self.parent[x]
    }
    fn union(&mut self, a: usize, b: usize) {
        let mut ra = self.find(a);
        let mut rb = self.find(b);
        if ra == rb { return; }
        if self.rank[ra] < self.rank[rb] {
            std::mem::swap(&mut ra, &mut rb);
        }
        self.parent[rb] = ra;
        if self.rank[ra] == self.rank[rb] {
            self.rank[ra] += 1;
        }
    }
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self { buf: input.into_bytes(), idx: 0 }
    }
    fn next_usize(&mut self) -> Option<usize> {
        let n = self.buf.len();
        while self.idx < n && !self.buf[self.idx].is_ascii_digit() {
            self.idx += 1;
        }
        if self.idx >= n { return None; }
        let mut val = 0usize;
        while self.idx < n && self.buf[self.idx].is_ascii_digit() {
            val = val * 10 + (self.buf[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        Some(val)
    }
}

#[derive(Clone, Copy)]
struct BBox {
    minx: f64, maxx: f64, miny: f64, maxy: f64,
    width: f64, height: f64,
}

fn compute_bbox(nodes: &[usize], pos: &[(f64, f64)]) -> BBox {
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &u in nodes {
        let (x, y) = pos[u];
        if x < minx { minx = x; }
        if x > maxx { maxx = x; }
        if y < miny { miny = y; }
        if y > maxy { maxy = y; }
    }
    let mut width = maxx - minx;
    let mut height = maxy - miny;
    if width < 1e-6 { width = 1.0; }
    if height < 1e-6 { height = 1.0; }
    BBox { minx, maxx, miny, maxy, width, height }
}

fn fr_layout(
    nodes: &[usize],
    edges: &[(usize, usize)],
    pos: &mut Vec<(f64, f64)>,
    temp_index: &mut Vec<usize>,
) {
    let s = nodes.len();
    if s == 0 { return; }
    let radius = (s as f64).sqrt().max(1.0);
    for (i, &u) in nodes.iter().enumerate() {
        let angle = 2.0 * PI * (i as f64) / (s as f64);
        pos[u] = (radius * angle.cos(), radius * angle.sin());
        temp_index[u] = i;
    }
    let k = 2.0;
    let iterations = if s <= 500 { 80 } else { 50 };
    let mut t = radius;
    let mut disp = vec![(0.0f64, 0.0f64); s];

    for _ in 0..iterations {
        for d in disp.iter_mut() { d.0 = 0.0; d.1 = 0.0; }

        for i in 0..s {
            let u = nodes[i];
            let (ux, uy) = pos[u];
            for j in (i + 1)..s {
                let v = nodes[j];
                let (vx, vy) = pos[v];
                let dx = ux - vx;
                let dy = uy - vy;
                let dist = (dx * dx + dy * dy).sqrt() + 1e-6;
                let force = k * k / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }

        for &(u, v) in edges.iter() {
            let i = temp_index[u];
            let j = temp_index[v];
            if i == usize::MAX || j == usize::MAX { continue; }
            let (ux, uy) = pos[u];
            let (vx, vy) = pos[v];
            let dx = ux - vx;
            let dy = uy - vy;
            let dist = (dx * dx + dy * dy).sqrt() + 1e-6;
            let force = dist * dist / k;
            let fx = dx / dist * force;
            let fy = dy / dist * force;
            disp[i].0 -= fx;
            disp[i].1 -= fy;
            disp[j].0 += fx;
            disp[j].1 += fy;
        }

        for i in 0..s {
            let u = nodes[i];
            let dx = disp[i].0;
            let dy = disp[i].1;
            let norm = (dx * dx + dy * dy).sqrt();
            if norm > 1e-6 {
                let limited = norm.min(t);
                let nx = dx / norm * limited;
                let ny = dy / norm * limited;
                pos[u].0 += nx;
                pos[u].1 += ny;
            }
        }
        t *= 0.9;
    }

    for &u in nodes.iter() {
        temp_index[u] = usize::MAX;
    }
}

fn spiral_layout(nodes: &[usize], pos: &mut Vec<(f64, f64)>) {
    let golden_angle = PI * (3.0 - 5.0_f64.sqrt());
    let spacing = 1.5;
    for (i, &u) in nodes.iter().enumerate() {
        let r = spacing * (i as f64).sqrt();
        let angle = i as f64 * golden_angle;
        pos[u] = (r * angle.cos(), r * angle.sin());
    }
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = sc.next_usize().unwrap_or(0);

    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    let mut dsu = DSU::new(n);

    for _ in 0..m {
        let u = sc.next_usize().unwrap_or(0);
        let v = sc.next_usize().unwrap_or(0);
        edges.push((u, v));
        if u != v {
            dsu.union(u, v);
        }
    }

    // Build component mapping
    let mut comp_map = std::collections::HashMap::<usize, usize>::new();
    let mut comp_id = vec![0usize; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();

    for i in 0..n {
        let root = dsu.find(i);
        let cid = *comp_map.entry(root).or_insert_with(|| {
            let id = comps.len();
            comps.push(Vec::new());
            id
        });
        comp_id[i] = cid;
        comps[cid].push(i);
    }

    let comp_count = comps.len();
    let mut comp_size: Vec<usize> = comps.iter().map(|c| c.len()).collect();
    let mut comp_edges: Vec<Vec<(usize, usize)>> = vec![Vec::new(); comp_count];

    // Only store edges for small components (<=2000)
    for &(u, v) in edges.iter() {
        if u == v { continue; }
        let c = comp_id[u];
        if c == comp_id[v] && comp_size[c] <= 2000 {
            comp_edges[c].push((u, v));
        }
    }

    let mut pos: Vec<(f64, f64)> = vec![(0.0, 0.0); n];
    let mut temp_index = vec![usize::MAX; n];
    let mut bbox: Vec<BBox> = Vec::with_capacity(comp_count);

    // Layout each component
    for (cid, nodes) in comps.iter().enumerate() {
        if nodes.len() <= 2000 {
            fr_layout(nodes, &comp_edges[cid], &mut pos, &mut temp_index);
        } else {
            spiral_layout(nodes, &mut pos);
        }
        bbox.push(compute_bbox(nodes, &pos));
    }

    // Arrange components in a grid-like packing
    let mut order: Vec<usize> = (0..comp_count).collect();
    order.sort_by(|&a, &b| {
        bbox[b].width.partial_cmp(&bbox[a].width).unwrap_or(Ordering::Equal)
    });

    let margin = 5.0;
    let row_limit = (n as f64).sqrt() * 10.0 + 10.0;
    let mut offsets = vec![(0.0f64, 0.0f64); comp_count];

    let mut cur_x = 0.0;
    let mut cur_y = 0.0;
    let mut row_height = 0.0;

    for &cid in order.iter() {
        let b = bbox[cid];
        if cur_x + b.width > row_limit && cur_x > 0.0 {
            cur_x = 0.0;
            cur_y += row_height + margin;
            row_height = 0.0;
        }
        offsets[cid] = (cur_x - b.minx + margin, cur_y - b.miny + margin);
        cur_x += b.width + margin;
        if b.height > row_height {
            row_height = b.height;
        }
    }

    // Apply offsets
    for (cid, nodes) in comps.iter().enumerate() {
        let (ox, oy) = offsets[cid];
        for &u in nodes {
            pos[u].0 += ox;
            pos[u].1 += oy;
        }
    }

    let mut output = String::new();
    for i in 0..n {
        let (x, y) = pos[i];
        output.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    let _ = io::stdout().write_all(output.as_bytes());
}
```

### Notes:
- Works from tiny graphs to **200k nodes**.
- Uses **cheap layouts** for huge graphs to stay fast.
- Avoids memory blowups by not storing edge lists for large components.
- Handles parallel edges and self-loops safely.

You can compile with:

```
rustc -O -C target-cpu=native main.rs
```

This will run efficiently across all graph sizes required.
