Below is a **complete, compilable Rust (edition 2021)** program that adapts to dataset size and cluster count.  

It supports:

✅ Small datasets: full in‑memory Lloyd k‑means  
✅ Medium datasets: streaming + temporary file + multi‑pass k‑means  
✅ Large datasets: online streaming k‑means (single pass, minimal memory)  

Binary I/O, adaptive strategy selection, and memory‑safe behavior are all included.

```rust
use std::fs::File;
use std::io::{self, Read, Write, BufReader, BufWriter, StdinLock, Stdout};
use std::path::PathBuf;

// ------------------------- Fast Input Reader -------------------------
struct FastInput<R: Read> {
    reader: R,
    buf: Vec<u8>,
    idx: usize,
    len: usize,
}

impl<R: Read> FastInput<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buf: vec![0u8; 1 << 20], // 1MB buffer
            idx: 0,
            len: 0,
        }
    }

    fn refill(&mut self) -> bool {
        self.len = self.reader.read(&mut self.buf).unwrap_or(0);
        self.idx = 0;
        self.len > 0
    }

    fn read_byte(&mut self) -> Option<u8> {
        if self.idx >= self.len {
            if !self.refill() {
                return None;
            }
        }
        let b = self.buf[self.idx];
        self.idx += 1;
        Some(b)
    }

    fn read_u64_ascii(&mut self) -> Option<u64> {
        let mut b = self.read_byte()?;
        while b <= b' ' {
            b = self.read_byte()?;
        }
        let mut num = 0u64;
        while b >= b'0' && b <= b'9' {
            num = num * 10 + (b - b'0') as u64;
            match self.read_byte() {
                Some(nb) => b = nb,
                None => return Some(num),
            }
        }
        Some(num)
    }

    fn read_f32(&mut self) -> Option<f32> {
        let mut bytes = [0u8; 4];
        for i in 0..4 {
            bytes[i] = self.read_byte()?;
        }
        Some(f32::from_le_bytes(bytes))
    }

    fn read_point(&mut self) -> Option<[f32; 3]> {
        let x = self.read_f32()?;
        let y = self.read_f32()?;
        let z = self.read_f32()?;
        Some([x, y, z])
    }
}

// ------------------------- Simple RNG -------------------------
struct SimpleRng {
    state: u64,
}
impl SimpleRng {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }
    fn next_u64(&mut self) -> u64 {
        self.state = self.state
            .wrapping_mul(6364136223846793005)
            .wrapping_add(1);
        self.state
    }
    fn next_f64(&mut self) -> f64 {
        (self.next_u64() as f64) / (u64::MAX as f64)
    }
    fn next_usize(&mut self, max: usize) -> usize {
        (self.next_u64() % max as u64) as usize
    }
}

// ------------------------- Utility Functions -------------------------
fn dist2(p: &[f32; 3], c: &[f32; 3]) -> f32 {
    let dx = p[0] - c[0];
    let dy = p[1] - c[1];
    let dz = p[2] - c[2];
    dx * dx + dy * dy + dz * dz
}

fn nearest(p: &[f32; 3], centroids: &[[f32; 3]]) -> usize {
    let mut best = 0usize;
    let mut best_d = dist2(p, &centroids[0]);
    for i in 1..centroids.len() {
        let d = dist2(p, &centroids[i]);
        if d < best_d {
            best_d = d;
            best = i;
        }
    }
    best
}

// ------------------------- KMeans++ Initialization -------------------------
fn kmeans_plus_plus(points: &Vec<[f32; 3]>, k: usize, rng: &mut SimpleRng) -> Vec<[f32; 3]> {
    let n = points.len();
    let mut centroids = Vec::with_capacity(k);
    let first = rng.next_usize(n);
    centroids.push(points[first]);
    let mut dist = vec![f32::MAX; n];

    for _ in 1..k {
        let last = *centroids.last().unwrap();
        for i in 0..n {
            let d = dist2(&points[i], &last);
            if d < dist[i] {
                dist[i] = d;
            }
        }
        let total: f64 = dist.iter().map(|&d| d as f64).sum();
        let mut r = rng.next_f64() * total;
        let mut idx = 0;
        for i in 0..n {
            r -= dist[i] as f64;
            if r <= 0.0 {
                idx = i;
                break;
            }
        }
        centroids.push(points[idx]);
    }
    centroids
}

// ------------------------- Lloyd Iteration -------------------------
fn lloyd(points: &Vec<[f32; 3]>, mut centroids: Vec<[f32; 3]>, max_iter: usize, rng: &mut SimpleRng) -> Vec<[f32; 3]> {
    let k = centroids.len();
    for _ in 0..max_iter {
        let mut sums = vec![[0f64; 3]; k];
        let mut counts = vec![0u64; k];
        for p in points.iter() {
            let idx = nearest(p, &centroids);
            sums[idx][0] += p[0] as f64;
            sums[idx][1] += p[1] as f64;
            sums[idx][2] += p[2] as f64;
            counts[idx] += 1;
        }
        let mut changed = 0.0;
        for i in 0..k {
            if counts[i] > 0 {
                let newc = [
                    (sums[i][0] / counts[i] as f64) as f32,
                    (sums[i][1] / counts[i] as f64) as f32,
                    (sums[i][2] / counts[i] as f64) as f32,
                ];
                changed += dist2(&centroids[i], &newc);
                centroids[i] = newc;
            } else {
                centroids[i] = points[rng.next_usize(points.len())];
            }
        }
        if changed < 1e-4 {
            break;
        }
    }
    centroids
}

// ------------------------- Main -------------------------
fn main() {
    const MAX_IN_MEMORY: u64 = 10_000_000;   // ~120MB
    const MAX_DISK: u64 = 100_000_000;       // ~1.2GB

    let stdin = io::stdin();
    let mut input = FastInput::new(stdin.lock());

    let n_points = match input.read_u64_ascii() {
        Some(v) => v,
        None => return,
    };
    let k = match input.read_u64_ascii() {
        Some(v) => v as usize,
        None => return,
    };

    if n_points == 0 || k == 0 {
        return;
    }

    let mut out = BufWriter::new(io::stdout());

    // Special case: K=1
    if k == 1 {
        for _ in 0..n_points {
            let _ = input.read_point();
            out.write_all(&0u32.to_le_bytes()).unwrap();
        }
        out.flush().unwrap();
        return;
    }

    // Strategy Selection
    if n_points <= MAX_IN_MEMORY {
        // ---------------- Small: In-Memory Lloyd ----------------
        let mut points = Vec::with_capacity(n_points as usize);
        for _ in 0..n_points {
            if let Some(p) = input.read_point() {
                points.push(p);
            }
        }
        let mut rng = SimpleRng::new(1234567);
        let centroids = kmeans_plus_plus(&points, k, &mut rng);
        let centroids = lloyd(&points, centroids, 15, &mut rng);

        for p in points.iter() {
            let idx = nearest(p, &centroids);
            out.write_all(&(idx as u32).to_le_bytes()).unwrap();
        }
        out.flush().unwrap();
    } else if n_points <= MAX_DISK {
        // ---------------- Medium: Stream to Temp File ----------------
        let mut rng = SimpleRng::new(987654);
        let mut reservoir: Vec<[f32; 3]> = Vec::with_capacity(k);

        let mut tmp_path = PathBuf::from(std::env::temp_dir());
        tmp_path.push("kmeans_stream_tmp.bin");
        let file = File::create(&tmp_path).unwrap();
        let mut writer = BufWriter::new(file);

        for i in 0..n_points {
            let p = input.read_point().unwrap();
            writer.write_all(&p[0].to_le_bytes()).unwrap();
            writer.write_all(&p[1].to_le_bytes()).unwrap();
            writer.write_all(&p[2].to_le_bytes()).unwrap();

            if reservoir.len() < k {
                reservoir.push(p);
            } else {
                let j = rng.next_u64() % (i + 1);
                if j < k as u64 {
                    reservoir[j as usize] = p;
                }
            }
        }
        writer.flush().unwrap();

        let mut centroids = reservoir.clone();
        if centroids.len() < k {
            while centroids.len() < k {
                centroids.push(reservoir[0]);
            }
        }

        for _ in 0..4 {
            let file = File::open(&tmp_path).unwrap();
            let mut reader = FastInput::new(BufReader::new(file));
            let mut sums = vec![[0f64; 3]; k];
            let mut counts = vec![0u64; k];

            for _ in 0..n_points {
                let p = reader.read_point().unwrap();
                let idx = nearest(&p, &centroids);
                sums[idx][0] += p[0] as f64;
                sums[idx][1] += p[1] as f64;
                sums[idx][2] += p[2] as f64;
                counts[idx] += 1;
            }

            for i in 0..k {
                if counts[i] > 0 {
                    centroids[i] = [
                        (sums[i][0] / counts[i] as f64) as f32,
                        (sums[i][1] / counts[i] as f64) as f32,
                        (sums[i][2] / counts[i] as f64) as f32,
                    ];
                } else {
                    centroids[i] = reservoir[rng.next_usize(reservoir.len())];
                }
            }
        }

        let file = File::open(&tmp_path).unwrap();
        let mut reader = FastInput::new(BufReader::new(file));
        for _ in 0..n_points {
            let p = reader.read_point().unwrap();
            let idx = nearest(&p, &centroids);
            out.write_all(&(idx as u32).to_le_bytes()).unwrap();
        }
        out.flush().unwrap();
        let _ = std::fs::remove_file(&tmp_path);
    } else {
        // ---------------- Large: Pure Streaming Online ----------------
        let mut centroids = vec![[0f32; 3]; k];
        let mut counts = vec![1u64; k];

        for i in 0..k {
            let p = input.read_point().unwrap();
            centroids[i] = p;
            counts[i] = 1;
            out.write_all(&(i as u32).to_le_bytes()).unwrap();
        }

        for _ in k as u64..n_points {
            let p = input.read_point().unwrap();
            let idx = nearest(&p, &centroids);
            counts[idx] += 1;
            let n = counts[idx] as f32;
            let c = &mut centroids[idx];
            c[0] += (p[0] - c[0]) / n;
            c[1] += (p[1] - c[1]) / n;
            c[2] += (p[2] - c[2]) / n;
            out.write_all(&(idx as u32).to_le_bytes()).unwrap();
        }
        out.flush().unwrap();
    }
}
```
