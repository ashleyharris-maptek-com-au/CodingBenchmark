```python
import math
import statistics

# Persistent controller states
_int_alt = 0.0
_int_vs = 0.0
_int_spd = 0.0
_int_hdg = 0.0

_prev_vs_err = 0.0
_prev_spd_err = 0.0
_prev_roll_err = 0.0

_prev_trim = 0.0

def _median(values):
    return statistics.median(values) if values else None

def _robust_fuse(values, valids=None):
    # Use all values, but if valids is provided, prefer those; still include all in case flags are wrong.
    if valids is None:
        arr = [v for v in values if v is not None]
    else:
        arr = [v for v, ok in zip(values, valids) if v is not None and ok] or [v for v in values if v is not None]
    if not arr:
        return None
    med = _median(arr)
    mad = _median([abs(v - med) for v in arr]) or 0.0
    if mad < 1e-6:
        return sum(arr) / len(arr)
    filt = [v for v in arr if abs(v - med) <= 3.0 * mad]
    if not filt:
        filt = arr
    return sum(filt) / len(filt)

def _wrap_deg(a):
    a = (a + 180) % 360 - 180
    return a

def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x

def autopilot_step(state: dict, dt: float) -> dict:
    global _int_alt, _int_vs, _int_spd, _int_hdg
    global _prev_vs_err, _prev_spd_err, _prev_roll_err, _prev_trim

    # Engage logic
    ap_engaged = state.get("ap_engaged", False)
    at_engaged = state.get("at_engaged", False)
    if not ap_engaged and not at_engaged:
        return {}

    # --- Sensor Fusion ---
    baro_alt = _robust_fuse(
        [state.get("baro_alt_1_ft"), state.get("baro_alt_2_ft"), state.get("baro_alt_3_ft")],
        [state.get("baro_alt_1_valid"), state.get("baro_alt_2_valid"), state.get("baro_alt_3_valid")]
    )
    radar_alt = _robust_fuse(
        [state.get("radar_alt_1_ft"), state.get("radar_alt_2_ft"), state.get("radar_alt_3_ft")],
        [state.get("radar_alt_1_valid"), state.get("radar_alt_2_valid"), state.get("radar_alt_3_valid")]
    )
    gps_alt = _robust_fuse(
        [state.get("gps_1_alt_ft"), state.get("gps_2_alt_ft"), state.get("gps_3_alt_ft")],
        [state.get("gps_1_valid"), state.get("gps_2_valid"), state.get("gps_3_valid")]
    )
    alt = baro_alt if baro_alt is not None else gps_alt
    # Blend radar at low altitude
    if radar_alt is not None and alt is not None and radar_alt < 2500:
        alt = 0.7 * radar_alt + 0.3 * alt

    ias = _robust_fuse(
        [state.get("ias_1_kt"), state.get("ias_2_kt"), state.get("ias_3_kt")],
        [state.get("ias_1_valid"), state.get("ias_2_valid"), state.get("ias_3_valid")]
    )
    if ias is None:
        ias = state.get("ground_speed_kt", 250)

    vs = _robust_fuse(
        [state.get("irs_1_vs_fpm"), state.get("irs_2_vs_fpm"), state.get("irs_3_vs_fpm"),
         state.get("vertical_speed_fpm")],
        [state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid"), True]
    )
    vs = vs if vs is not None else 0.0

    hdg = _robust_fuse(
        [state.get("irs_1_heading_deg"), state.get("irs_2_heading_deg"), state.get("irs_3_heading_deg"),
         state.get("magnetic_heading_deg")],
        [state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid"), True]
    )
    hdg = hdg if hdg is not None else state.get("true_heading_deg", 0.0)

    roll = _robust_fuse(
        [state.get("irs_1_roll_deg"), state.get("irs_2_roll_deg"), state.get("irs_3_roll_deg")],
        [state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid")]
    ) or 0.0
    pitch = _robust_fuse(
        [state.get("irs_1_pitch_deg"), state.get("irs_2_pitch_deg"), state.get("irs_3_pitch_deg")],
        [state.get("irs_1_valid"), state.get("irs_2_valid"), state.get("irs_3_valid")]
    ) or 0.0
    roll_rate = _robust_fuse(
        [state.get("gyro_1_roll_rate_dps"), state.get("gyro_2_roll_rate_dps"), state.get("gyro_3_roll_rate_dps")],
        [state.get("gyro_1_valid"), state.get("gyro_2_valid"), state.get("gyro_3_valid")]
    ) or 0.0
    yaw_rate = _robust_fuse(
        [state.get("gyro_1_yaw_rate_dps"), state.get("gyro_2_yaw_rate_dps"), state.get("gyro_3_yaw_rate_dps")],
        [state.get("gyro_1_valid"), state.get("gyro_2_valid"), state.get("gyro_3_valid")]
    ) or 0.0
    drift = state.get("drift_angle_deg", 0.0)

    # --- Targets ---
    tgt_alt = state.get("target_alt_ft", alt)
    tgt_hdg = state.get("target_hdg_deg", hdg)
    tgt_ias = state.get("target_ias_kt", ias)
    tgt_vs = state.get("target_vs_fpm", 0.0)

    # --- Lateral Control (Heading Hold) ---
    hdg_err = _wrap_deg(tgt_hdg - hdg)
    _int_hdg += hdg_err * dt
    bank_cmd = _clamp(hdg_err * 0.02, -25.0, 25.0)
    roll_err = bank_cmd - roll
    aileron = _clamp(0.04 * roll_err - 0.1 * roll_rate, -1.0, 1.0)

    # --- Vertical Control (Altitude/VS) ---
    alt_err = (tgt_alt - alt) if alt is not None else 0.0
    if abs(alt_err) > 200:
        vs_cmd = _clamp(alt_err * 0.5, -2000.0, 2000.0)
    else:
        vs_cmd = tgt_vs
    vs_err = vs_cmd - vs
    _int_vs += vs_err * dt
    _int_vs = _clamp(_int_vs, -1000, 1000)

    # Elevator PID on VS error with pitch damping
    kp_vs, ki_vs, kd_vs = 0.0007, 0.00002, 0.0005
    dv = (vs_err - _prev_vs_err) / dt if dt > 0 else 0.0
    elevator = kp_vs * vs_err + ki_vs * _int_vs + kd_vs * dv - 0.01 * pitch
    elevator = _clamp(elevator, -1.0, 1.0)
    _prev_vs_err = vs_err

    # Trim bias
    trim_rate = _clamp(elevator, -0.2, 0.2) * 0.02
    _prev_trim = _clamp(_prev_trim + trim_rate * dt, -1.0, 1.0)

    # --- Speed / Autothrottle ---
    throttle_cmd = None
    if at_engaged:
        spd_err = tgt_ias - ias
        _int_spd += spd_err * dt
        _int_spd = _clamp(_int_spd, -50, 50)
        kp_spd, ki_spd, kd_spd = 0.01, 0.002, 0.02
        dspd = (spd_err - _prev_spd_err) / dt if dt > 0 else 0.0
        throttle_cmd = kp_spd * spd_err + ki_spd * _int_spd + kd_spd * dspd
        throttle_cmd = _clamp(0.5 + throttle_cmd, 0.0, 1.0)
        _prev_spd_err = spd_err

    # --- Rudder (Yaw Damper / Slip) ---
    rudder = _clamp(-0.08 * yaw_rate - 0.02 * drift, -1.0, 1.0)

    # --- Gear / Flaps / Speedbrake ---
    gear = None
    flaps = None
    speedbrake = None

    if radar_alt is not None and radar_alt < 1500:
        gear = 1
    elif tgt_alt and tgt_alt > 2000:
        gear = 0

    if ias is not None:
        if ias < 110:
            flaps = 40
        elif ias < 120:
            flaps = 30
        elif ias < 140:
            flaps = 20
        elif ias < 160:
            flaps = 10
        elif ias < 180:
            flaps = 5
        else:
            flaps = 0

    if ias is not None and ias > tgt_ias + 15 and vs < -500:
        speedbrake = 0.5
    else:
        speedbrake = 0.0

    out = {
        "aileron": aileron,
        "elevator": elevator,
        "rudder": rudder,
        "elevator_trim": _prev_trim,
    }

    if throttle_cmd is not None:
        out["throttle_1"] = throttle_cmd
        out["throttle_2"] = throttle_cmd

    if gear is not None:
        out["gear"] = gear
    if flaps is not None:
        out["flaps"] = flaps
    if speedbrake is not None:
        out["speedbrake"] = speedbrake

    return out
```
