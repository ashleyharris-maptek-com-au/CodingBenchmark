```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;

public class Program
{
    static int n, m;
    static int[] deg;
    static int[] adjStart;
    static int[] adj;
    static ulong[][] adjBits;
    static int words;
    static bool hasBits;

    static Stopwatch sw;
    static TimeSpan timeLimit;
    static bool timeExceeded;

    static List<int> bestClique = new List<int>();
    static List<int> currClique = new List<int>();
    static int bestSize;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        n = fs.NextInt();
        m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine("0\n");
            return;
        }

        deg = new int[n];
        int[] eu = new int[m];
        int[] ev = new int[m];
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u == v) { i--; continue; } // ignore self-loops if any
            eu[i] = u;
            ev[i] = v;
            deg[u]++;
            deg[v]++;
        }

        // Build adjacency list
        adjStart = new int[n + 1];
        for (int i = 0; i < n; i++)
            adjStart[i + 1] = adjStart[i] + deg[i];
        adj = new int[2 * m];
        int[] temp = (int[])adjStart.Clone();
        for (int i = 0; i < m; i++)
        {
            int u = eu[i], v = ev[i];
            adj[temp[u]++] = v;
            adj[temp[v]++] = u;
        }
        for (int i = 0; i < n; i++)
        {
            Array.Sort(adj, adjStart[i], deg[i]);
        }

        // Build bitset if feasible
        if (n <= 20000)
        {
            words = (n + 63) >> 6;
            adjBits = new ulong[n][];
            for (int i = 0; i < n; i++)
                adjBits[i] = new ulong[words];
            for (int i = 0; i < m; i++)
            {
                int u = eu[i], v = ev[i];
                adjBits[u][v >> 6] |= 1UL << (v & 63);
                adjBits[v][u >> 6] |= 1UL << (u & 63);
            }
            hasBits = true;
        }
        else
        {
            hasBits = false;
        }

        // Heuristic best
        var greedy = GreedyDegree();
        bestClique = greedy;
        bestSize = bestClique.Count;

        // Exact search for small graphs
        if (n <= 2000)
        {
            // stronger heuristic start
            if (n <= 20000)
            {
                var g2 = GreedyIntersection(5);
                if (g2.Count > bestSize)
                {
                    bestClique = g2;
                    bestSize = g2.Count;
                }
            }

            sw = Stopwatch.StartNew();
            timeLimit = TimeSpan.FromSeconds(270);
            timeExceeded = false;

            int[] all = new int[n];
            for (int i = 0; i < n; i++) all[i] = i;
            Expand(all, n);
        }

        // Output
        Console.WriteLine(bestClique.Count);
        Console.WriteLine(string.Join(" ", bestClique));
    }

    // Exact algorithm (branch & bound with coloring)
    static void Expand(int[] list, int len)
    {
        if (timeExceeded) return;
        if (sw.Elapsed > timeLimit)
        {
            timeExceeded = true;
            return;
        }
        if (len == 0)
        {
            if (currClique.Count > bestSize)
            {
                bestSize = currClique.Count;
                bestClique = new List<int>(currClique);
            }
            return;
        }

        ColorSort(list, len, out int[] order, out int[] colors);

        for (int i = len - 1; i >= 0; i--)
        {
            if (currClique.Count + colors[i] <= bestSize)
                return;
            int v = order[i];

            // build new candidate list
            int[] newList = new int[i];
            int newLen = 0;
            for (int j = 0; j < i; j++)
            {
                int u = order[j];
                if (IsAdjacentBit(v, u))
                    newList[newLen++] = u;
            }

            currClique.Add(v);
            Expand(newList, newLen);
            currClique.RemoveAt(currClique.Count - 1);

            if (timeExceeded) return;
        }
    }

    static void ColorSort(int[] list, int len, out int[] order, out int[] colors)
    {
        int maxColor = 0;
        List<ulong[]> classBits = new List<ulong[]>(len);
        int[] colorOf = new int[len];

        for (int i = 0; i < len; i++)
        {
            int v = list[i];
            int c;
            for (c = 0; c < maxColor; c++)
            {
                if (!Intersects(adjBits[v], classBits[c]))
                    break;
            }
            if (c == maxColor)
            {
                classBits.Add(new ulong[words]);
                maxColor++;
            }
            colorOf[i] = c + 1;
            classBits[c][v >> 6] |= 1UL << (v & 63);
        }

        int[] counts = new int[maxColor + 1];
        for (int i = 0; i < len; i++) counts[colorOf[i]]++;

        int[] pos = new int[maxColor + 1];
        int sum = 0;
        for (int c = 1; c <= maxColor; c++)
        {
            pos[c] = sum;
            sum += counts[c];
        }

        order = new int[len];
        colors = new int[len];
        for (int i = 0; i < len; i++)
        {
            int c = colorOf[i];
            int idx = pos[c]++;
            order[idx] = list[i];
            colors[idx] = c;
        }
    }

    static bool Intersects(ulong[] a, ulong[] b)
    {
        for (int i = 0; i < words; i++)
        {
            if ((a[i] & b[i]) != 0) return true;
        }
        return false;
    }

    static bool IsAdjacentBit(int u, int v)
    {
        return (adjBits[u][v >> 6] & (1UL << (v & 63))) != 0;
    }

    static bool IsAdjacent(int u, int v)
    {
        if (hasBits) return IsAdjacentBit(u, v);
        return IsAdjacentList(u, v);
    }

    static bool IsAdjacentList(int u, int v)
    {
        int l = adjStart[u], r = adjStart[u + 1] - 1;
        while (l <= r)
        {
            int mid = (l + r) >> 1;
            int val = adj[mid];
            if (val == v) return true;
            if (val < v) l = mid + 1;
            else r = mid - 1;
        }
        return false;
    }

    // Greedy heuristic based on degree order
    static List<int> GreedyDegree()
    {
        int[] order = Enumerable.Range(0, n).ToArray();
        Array.Sort(order, (a, b) => deg[b].CompareTo(deg[a]));

        List<int> clique = new List<int>();
        foreach (var v in order)
        {
            bool ok = true;
            for (int i = 0; i < clique.Count; i++)
            {
                if (!IsAdjacent(v, clique[i]))
                {
                    ok = false; break;
                }
            }
            if (ok) clique.Add(v);
        }
        return clique;
    }

    // Greedy intersection heuristic (best for n <= 20000)
    static List<int> GreedyIntersection(int iterations)
    {
        int[] order = Enumerable.Range(0, n).ToArray();
        Array.Sort(order, (a, b) => deg[b].CompareTo(deg[a]));

        List<int> best = new List<int>();

        Random rnd = new Random(1);
        for (int it = 0; it < iterations; it++)
        {
            if (it > 0)
            {
                // shuffle for variation
                for (int i = order.Length - 1; i > 0; i--)
                {
                    int j = rnd.Next(i + 1);
                    int tmp = order[i]; order[i] = order[j]; order[j] = tmp;
                }
            }

            List<int> candidates = new List<int>(order);
            List<int> clique = new List<int>();

            while (candidates.Count > 0)
            {
                int v = candidates[0];
                clique.Add(v);

                List<int> newCand = new List<int>();
                for (int i = 1; i < candidates.Count; i++)
                {
                    int u = candidates[i];
                    if (IsAdjacent(v, u)) newCand.Add(u);
                }
                candidates = newCand;
            }
            if (clique.Count > best.Count) best = clique;
        }
        return best;
    }

    // Fast input
    public class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;

        public FastScanner(Stream stream) { this.stream = stream; }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            while ((c = Read()) != -1 && c <= ' ') { }
            if (c == -1) return 0;
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
```

This solution:

- Uses an exact branch-and-bound algorithm with coloring for graphs up to 2000 vertices.
- Automatically falls back to robust heuristics for larger graphs.
- Handles dense and sparse graphs efficiently.
- Uses bitsets where feasible for speed and adjacency checks.
- Ensures correctness of output format and clique validity.
