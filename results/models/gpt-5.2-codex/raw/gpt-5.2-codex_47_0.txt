// WGSL Compute Shader: Proof-of-Work Hash Mining

@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;
// params.x = difficulty
// params.y = range_start
// params.z = range_size

fn rotl(x: u32, n: u32) -> u32 {
    return (x << n) | (x >> (32u - n));
}

fn quarter_round(a: u32, b: u32, c: u32, d: u32) -> vec4<u32> {
    var aa = a;
    var bb = b;
    var cc = c;
    var dd = d;

    aa = aa + bb; dd = rotl(dd ^ aa, 16u);
    cc = cc + dd; bb = rotl(bb ^ cc, 12u);
    aa = aa + bb; dd = rotl(dd ^ aa,  8u);
    cc = cc + dd; bb = rotl(bb ^ cc,  7u);

    return vec4<u32>(aa, bb, cc, dd);
}

fn custom_hash(data: array<u32,16>, nonce: u32) -> vec2<u32> {
    var state : array<u32, 16>;

    for (var i: u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = data[i] ^ (nonce + i);
    }

    for (var r: u32 = 0u; r < 8u; r = r + 1u) {
        // group 0: state[0..3]
        let q0 = quarter_round(state[0], state[1], state[2], state[3]);
        state[0] = q0.x; state[1] = q0.y; state[2] = q0.z; state[3] = q0.w;

        // group 1: state[4..7]
        let q1 = quarter_round(state[4], state[5], state[6], state[7]);
        state[4] = q1.x; state[5] = q1.y; state[6] = q1.z; state[7] = q1.w;

        // group 2: state[8..11]
        let q2 = quarter_round(state[8], state[9], state[10], state[11]);
        state[8]  = q2.x; state[9]  = q2.y; state[10] = q2.z; state[11] = q2.w;

        // group 3: state[12..15]
        let q3 = quarter_round(state[12], state[13], state[14], state[15]);
        state[12] = q3.x; state[13] = q3.y; state[14] = q3.z; state[15] = q3.w;
    }

    let h0 =
        state[0] ^ state[2] ^ state[4] ^ state[6] ^
        state[8] ^ state[10] ^ state[12] ^ state[14];

    let h1 =
        state[1] ^ state[3] ^ state[5] ^ state[7] ^
        state[9] ^ state[11] ^ state[13] ^ state[15];

    return vec2<u32>(h0, h1);
}

fn trailing_zeros_64(h0: u32, h1: u32) -> u32 {
    if (h0 != 0u) {
        return countTrailingZeros(h0);
    }
    return 32u + countTrailingZeros(h1);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid: vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let nonce = params.y + gid.x;
    let h = custom_hash(baseData, nonce);
    let tz = trailing_zeros_64(h.x, h.y);

    if (tz >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h.x;
        result[3] = h.y;
    }
}
