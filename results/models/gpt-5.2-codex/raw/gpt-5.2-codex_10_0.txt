Below is a complete **C++20** solver for the 2D Cutting Stock / Rectangle Bin Packing problem as specified.  
It adapts strategy based on problem size:

- **Small/Medium (≤ 5000 rectangles):** best‑fit shelf packing across all boards  
- **Large/Ludicrous (> 5000 rectangles):** fast next‑fit shelf packing per board

This keeps performance under 30 seconds while still producing decent packing quality.  
It supports rotation, avoids overlaps, respects board bounds, and outputs valid placements.

---

### ✅ C++20 Solver (MSVC Compatible)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>

struct Rect {
    int w, h, idx;
};

struct Shelf {
    int y;
    int height;
    int used; // used width
};

struct Board {
    std::vector<Shelf> shelves;
    int usedHeight = 0;
};

struct Placement {
    int board;
    int x;
    int y;
    int rot;
};

static bool chooseOrientationShelf(const Rect& r, int shelfHeight, int remWidth,
                                   int& outW, int& outH, int& outRot) {
    bool fit1 = (r.w <= remWidth && r.h <= shelfHeight);
    bool fit2 = (r.h <= remWidth && r.w <= shelfHeight);

    if (!fit1 && !fit2) return false;
    if (fit1 && fit2) {
        // choose larger width to reduce leftover
        if (r.w >= r.h) {
            outW = r.w; outH = r.h; outRot = 0;
        } else {
            outW = r.h; outH = r.w; outRot = 1;
        }
    } else if (fit1) {
        outW = r.w; outH = r.h; outRot = 0;
    } else {
        outW = r.h; outH = r.w; outRot = 1;
    }
    return true;
}

static bool chooseOrientationNewShelf(const Rect& r, int boardW, int remH,
                                      int& outW, int& outH, int& outRot) {
    bool fit1 = (r.w <= boardW && r.h <= remH);
    bool fit2 = (r.h <= boardW && r.w <= remH);

    if (!fit1 && !fit2) return false;
    if (fit1 && fit2) {
        // choose smaller height to save vertical space
        if (r.h <= r.w) {
            outW = r.w; outH = r.h; outRot = 0;
        } else {
            outW = r.h; outH = r.w; outRot = 1;
        }
    } else if (fit1) {
        outW = r.w; outH = r.h; outRot = 0;
    } else {
        outW = r.h; outH = r.w; outRot = 1;
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, BW, BH;
    if (!(std::cin >> N >> BW >> BH)) return 0;

    std::vector<Rect> rects(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].idx = i;
    }

    // Sort by decreasing max side, then min side, then area
    std::sort(rects.begin(), rects.end(),
              [](const Rect& a, const Rect& b) {
                  int amax = std::max(a.w, a.h);
                  int bmax = std::max(b.w, b.h);
                  if (amax != bmax) return amax > bmax;
                  int amin = std::min(a.w, a.h);
                  int bmin = std::min(b.w, b.h);
                  if (amin != bmin) return amin > bmin;
                  long long areaA = 1LL * a.w * a.h;
                  long long areaB = 1LL * b.w * b.h;
                  return areaA > areaB;
              });

    std::vector<Placement> placements(N);
    std::vector<Board> boards;

    bool largeMode = (N > 5000);

    for (const auto& r : rects) {
        bool placed = false;

        if (!largeMode) {
            // Best-fit across all shelves in all boards
            int bestBoard = -1, bestShelf = -1;
            int bestW = 0, bestH = 0, bestRot = 0;
            int bestRem = std::numeric_limits<int>::max();

            for (int bi = 0; bi < (int)boards.size(); ++bi) {
                Board& b = boards[bi];
                for (int si = 0; si < (int)b.shelves.size(); ++si) {
                    Shelf& s = b.shelves[si];
                    int remWidth = BW - s.used;

                    int w, h, rot;
                    if (chooseOrientationShelf(r, s.height, remWidth, w, h, rot)) {
                        int rem = remWidth - w;
                        if (rem < bestRem) {
                            bestRem = rem;
                            bestBoard = bi;
                            bestShelf = si;
                            bestW = w; bestH = h; bestRot = rot;
                        }
                    }
                }
            }

            if (bestBoard != -1) {
                Shelf& s = boards[bestBoard].shelves[bestShelf];
                int x = s.used;
                int y = s.y;
                s.used += bestW;
                placements[r.idx] = {bestBoard, x, y, bestRot};
                placed = true;
            } else {
                // Try to open new shelf in existing board
                int bestBoardNS = -1;
                int bestWNS = 0, bestHNS = 0, bestRotNS = 0;
                int bestRemainH = std::numeric_limits<int>::max();

                for (int bi = 0; bi < (int)boards.size(); ++bi) {
                    Board& b = boards[bi];
                    int remH = BH - b.usedHeight;
                    int w, h, rot;
                    if (chooseOrientationNewShelf(r, BW, remH, w, h, rot)) {
                        int rem = remH - h;
                        if (rem < bestRemainH) {
                            bestRemainH = rem;
                            bestBoardNS = bi;
                            bestWNS = w; bestHNS = h; bestRotNS = rot;
                        }
                    }
                }

                if (bestBoardNS != -1) {
                    Board& b = boards[bestBoardNS];
                    Shelf newS;
                    newS.y = b.usedHeight;
                    newS.height = bestHNS;
                    newS.used = bestWNS;
                    b.shelves.push_back(newS);
                    int x = 0;
                    int y = newS.y;
                    b.usedHeight += bestHNS;
                    placements[r.idx] = {bestBoardNS, x, y, bestRotNS};
                    placed = true;
                }
            }
        } else {
            // Large mode: only use current board (next-fit)
            if (boards.empty()) {
                boards.push_back(Board());
            }

            Board& b = boards.back();

            int bestShelf = -1;
            int bestW = 0, bestH = 0, bestRot = 0;
            int bestRem = std::numeric_limits<int>::max();

            for (int si = 0; si < (int)b.shelves.size(); ++si) {
                Shelf& s = b.shelves[si];
                int remWidth = BW - s.used;

                int w, h, rot;
                if (chooseOrientationShelf(r, s.height, remWidth, w, h, rot)) {
                    int rem = remWidth - w;
                    if (rem < bestRem) {
                        bestRem = rem;
                        bestShelf = si;
                        bestW = w; bestH = h; bestRot = rot;
                    }
                }
            }

            if (bestShelf != -1) {
                Shelf& s = b.shelves[bestShelf];
                int x = s.used;
                int y = s.y;
                s.used += bestW;
                placements[r.idx] = {(int)boards.size()-1, x, y, bestRot};
                placed = true;
            } else {
                int remH = BH - b.usedHeight;
                int w, h, rot;
                if (chooseOrientationNewShelf(r, BW, remH, w, h, rot)) {
                    Shelf newS;
                    newS.y = b.usedHeight;
                    newS.height = h;
                    newS.used = w;
                    b.shelves.push_back(newS);
                    int x = 0;
                    int y = newS.y;
                    b.usedHeight += h;
                    placements[r.idx] = {(int)boards.size()-1, x, y, rot};
                    placed = true;
                } else {
                    // New board
                    Board nb;
                    int w2, h2, rot2;
                    // Should always fit in new board if input valid
                    chooseOrientationNewShelf(r, BW, BH, w2, h2, rot2);
                    Shelf s;
                    s.y = 0;
                    s.height = h2;
                    s.used = w2;
                    nb.shelves.push_back(s);
                    nb.usedHeight = h2;
                    boards.push_back(nb);
                    placements[r.idx] = {(int)boards.size()-1, 0, 0, rot2};
                    placed = true;
                }
            }
        }

        // Fallback: if not placed (shouldn't happen), force new board
        if (!placed) {
            Board nb;
            int w2, h2, rot2;
            chooseOrientationNewShelf(r, BW, BH, w2, h2, rot2);
            Shelf s;
            s.y = 0; s.height = h2; s.used = w2;
            nb.shelves.push_back(s);
            nb.usedHeight = h2;
            boards.push_back(nb);
            placements[r.idx] = {(int)boards.size()-1, 0, 0, rot2};
        }
    }

    // Output
    std::cout << boards.size() << "\n";
    for (int i = 0; i < N; ++i) {
        std::cout << placements[i].board << " "
                  << placements[i].x << " "
                  << placements[i].y << " "
                  << placements[i].rot << "\n";
    }
    return 0;
}
```
