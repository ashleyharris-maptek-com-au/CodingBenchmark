Below is a complete C# program that compiles with `csc` and implements a greedy, adaptive Tetris AI.  
It reads the game setup and each piece, evaluates all placements, and outputs the rotation and column for the best move.

The evaluation function adapts weights based on board height and complexity.

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    public static void Main()
    {
        var fs = new FastScanner();
        string s = fs.Next();
        if (s == null) return;

        int width = int.Parse(s);
        int height = int.Parse(fs.Next());
        int numPieces = int.Parse(fs.Next());

        var ai = new TetrisAI(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            string p = fs.Next();
            if (p == null) break;

            char type = p[0];
            Move move = ai.FindBestMove(type);

            Console.WriteLine($"{move.Rotation} {move.Column}");
            Console.Out.Flush();

            // Read result if provided (interactive environments)
            string result = fs.Next();
            if (result == null) continue;
            if (result.StartsWith("gameover")) break;
        }
    }
}

public struct Move
{
    public int Rotation;
    public int Column;
}

public class TetrisAI
{
    private readonly int width;
    private readonly int height;
    private bool[,] board;
    private readonly Dictionary<char, Piece> pieces;

    public TetrisAI(int w, int h)
    {
        width = w;
        height = h;
        board = new bool[h, w];
        pieces = Piece.CreatePieces();
    }

    public Move FindBestMove(char type)
    {
        Piece piece = pieces[type];
        double bestScore = double.NegativeInfinity;
        int bestRot = 0;
        int bestCol = 0;
        bool found = false;

        for (int rot = 0; rot < 4; rot++)
        {
            var rotation = piece.Rotations[rot];
            int maxCol = width - rotation.Width;
            if (maxCol < 0) continue;

            for (int col = 0; col <= maxCol; col++)
            {
                int row = FindDropRow(rotation, col);
                if (row < 0) continue;

                var sim = Simulate(board, rotation, col, row);
                BoardFeatures feat = ComputeFeatures(sim.Board);
                double score = Evaluate(sim.Cleared, feat);

                if (!found || score > bestScore)
                {
                    bestScore = score;
                    bestRot = rot;
                    bestCol = col;
                    found = true;
                }
            }
        }

        if (!found)
        {
            return new Move { Rotation = 0, Column = 0 };
        }

        // Apply the best move to internal board
        var bestRotation = piece.Rotations[bestRot];
        int bestRow = FindDropRow(bestRotation, bestCol);
        if (bestRow >= 0)
        {
            var sim = Simulate(board, bestRotation, bestCol, bestRow);
            board = sim.Board;
        }

        return new Move { Rotation = bestRot, Column = bestCol };
    }

    private int FindDropRow(Rotation rot, int col)
    {
        int maxRow = height - rot.Height;
        if (maxRow < 0) return -1;
        int row = maxRow;
        if (Collision(rot, col, row)) return -1;

        while (row > 0 && !Collision(rot, col, row - 1))
            row--;

        return row;
    }

    private bool Collision(Rotation rot, int col, int row)
    {
        foreach (var b in rot.Blocks)
        {
            int x = col + b.X;
            int y = row + b.Y;
            if (x < 0 || x >= width || y < 0 || y >= height) return true;
            if (board[y, x]) return true;
        }
        return false;
    }

    private (bool[,] Board, int Cleared) Simulate(bool[,] source, Rotation rot, int col, int row)
    {
        bool[,] temp = CloneBoard(source);

        foreach (var b in rot.Blocks)
        {
            temp[row + b.Y, col + b.X] = true;
        }

        int cleared;
        temp = ClearLines(temp, out cleared);
        return (temp, cleared);
    }

    private bool[,] ClearLines(bool[,] b, out int cleared)
    {
        cleared = 0;
        bool[,] newBoard = new bool[height, width];
        int newRow = 0;

        for (int y = 0; y < height; y++)
        {
            bool full = true;
            for (int x = 0; x < width; x++)
            {
                if (!b[y, x]) { full = false; break; }
            }

            if (full)
            {
                cleared++;
                continue;
            }

            for (int x = 0; x < width; x++)
                newBoard[newRow, x] = b[y, x];
            newRow++;
        }

        return newBoard;
    }

    private bool[,] CloneBoard(bool[,] src)
    {
        bool[,] dst = new bool[height, width];
        for (int y = 0; y < height; y++)
            for (int x = 0; x < width; x++)
                dst[y, x] = src[y, x];
        return dst;
    }

    private BoardFeatures ComputeFeatures(bool[,] b)
    {
        int[] heights = new int[width];
        int aggregate = 0, maxHeight = 0, holes = 0;

        for (int x = 0; x < width; x++)
        {
            bool seenBlock = false;
            int h = 0;

            for (int y = height - 1; y >= 0; y--)
            {
                if (b[y, x])
                {
                    if (!seenBlock)
                    {
                        h = y + 1;
                        seenBlock = true;
                    }
                }
                else if (seenBlock)
                {
                    holes++;
                }
            }

            heights[x] = h;
            aggregate += h;
            if (h > maxHeight) maxHeight = h;
        }

        int bumpiness = 0;
        for (int x = 0; x < width - 1; x++)
            bumpiness += Math.Abs(heights[x] - heights[x + 1]);

        return new BoardFeatures
        {
            Heights = heights,
            AggregateHeight = aggregate,
            MaxHeight = maxHeight,
            Holes = holes,
            Bumpiness = bumpiness
        };
    }

    private double Evaluate(int cleared, BoardFeatures f)
    {
        double ratio = (double)f.MaxHeight / height;
        double wLines, wAgg, wHoles, wBump;

        if (ratio > 0.7)
        {
            wLines = 2.5;
            wAgg = -0.9;
            wHoles = -1.6;
            wBump = -0.5;
        }
        else if (ratio > 0.4)
        {
            wLines = 1.6;
            wAgg = -0.6;
            wHoles = -1.2;
            wBump = -0.4;
        }
        else
        {
            wLines = 1.2;
            wAgg = -0.4;
            wHoles = -1.0;
            wBump = -0.3;
        }

        if (f.Holes > width) wHoles *= 1.3;

        double score = wLines * cleared * cleared
                     + wAgg * f.AggregateHeight
                     + wHoles * f.Holes
                     + wBump * f.Bumpiness;

        if (cleared == 4) score += 4; // Tetris bonus
        return score;
    }
}

public struct BoardFeatures
{
    public int[] Heights;
    public int AggregateHeight;
    public int MaxHeight;
    public int Holes;
    public int Bumpiness;
}

public struct Point
{
    public int X, Y;
    public Point(int x, int y) { X = x; Y = y; }
}

public class Rotation
{
    public Point[] Blocks;
    public int Width;
    public int Height;
}

public class Piece
{
    public Rotation[] Rotations;

    public static Dictionary<char, Piece> CreatePieces()
    {
        var dict = new Dictionary<char, Piece>();

        dict['I'] = new Piece(new[]
        {
            new Point(0,0), new Point(1,0), new Point(2,0), new Point(3,0)
        });
        dict['O'] = new Piece(new[]
        {
            new Point(0,0), new Point(1,0), new Point(0,1), new Point(1,1)
        });
        dict['T'] = new Piece(new[]
        {
            new Point(0,0), new Point(1,0), new Point(2,0), new Point(1,1)
        });
        dict['S'] = new Piece(new[]
        {
            new Point(1,0), new Point(2,0), new Point(0,1), new Point(1,1)
        });
        dict['Z'] = new Piece(new[]
        {
            new Point(0,0), new Point(1,0), new Point(1,1), new Point(2,1)
        });
        dict['J'] = new Piece(new[]
        {
            new Point(0,0), new Point(1,0), new Point(2,0), new Point(0,1)
        });
        dict['L'] = new Piece(new[]
        {
            new Point(0,0), new Point(1,0), new Point(2,0), new Point(2,1)
        });

        return dict;
    }

    public Piece(Point[] baseBlocks)
    {
        Rotations = GenerateRotations(baseBlocks);
    }

    private static Rotation[] GenerateRotations(Point[] baseBlocks)
    {
        Rotation[] rots = new Rotation[4];
        Point[] current = baseBlocks;

        for (int i = 0; i < 4; i++)
        {
            current = Normalize(current);
            rots[i] = CreateRotation(current);
            current = Rotate(current);
        }

        return rots;
    }

    private static Point[] Rotate(Point[] blocks)
    {
        Point[] rotated = new Point[blocks.Length];
        for (int i = 0; i < blocks.Length; i++)
        {
            int x = blocks[i].X;
            int y = blocks[i].Y;
            rotated[i] = new Point(y, -x);
        }
        return rotated;
    }

    private static Point[] Normalize(Point[] blocks)
    {
        int minX = int.MaxValue, minY = int.MaxValue;
        foreach (var b in blocks)
        {
            if (b.X < minX) minX = b.X;
            if (b.Y < minY) minY = b.Y;
        }

        Point[] norm = new Point[blocks.Length];
        for (int i = 0; i < blocks.Length; i++)
            norm[i] = new Point(blocks[i].X - minX, blocks[i].Y - minY);

        return norm;
    }

    private static Rotation CreateRotation(Point[] blocks)
    {
        int maxX = 0, maxY = 0;
        foreach (var b in blocks)
        {
            if (b.X > maxX) maxX = b.X;
            if (b.Y > maxY) maxY = b.Y;
        }

        return new Rotation
        {
            Blocks = blocks,
            Width = maxX + 1,
            Height = maxY + 1
        };
    }
}

public class FastScanner
{
    private readonly Stream stream;
    private readonly byte[] buffer = new byte[1 << 16];
    private int ptr = 0, len = 0;

    public FastScanner()
    {
        stream = Console.OpenStandardInput();
    }

    private int Read()
    {
        if (ptr >= len)
        {
            len = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (len == 0) return -1;
        }
        return buffer[ptr++];
    }

    public string Next()
    {
        int c;
        do
        {
            c = Read();
            if (c == -1) return null;
        } while (c <= 32);

        StringBuilder sb = new StringBuilder();
        while (c > 32)
        {
            sb.Append((char)c);
            c = Read();
            if (c == -1) break;
        }
        return sb.ToString();
    }
}
```
