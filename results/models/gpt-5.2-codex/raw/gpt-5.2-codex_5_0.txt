Below is a complete Rust 2024 solution that adapts to grid size and obstacle density.  
It uses **constructive patterns** for obstacle‑free grids (very fast and scalable),  
**exact backtracking** with strong pruning for small grids, and a **greedy heuristic** for large obstructed grids.

It always validates input and necessary conditions, and outputs either a valid Hamiltonian path/cycle or `0`.

```rust
use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

fn idx(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn coord(i: usize, w: usize) -> (usize, usize) {
    (i % w, i / w)
}

/// Construct a simple serpentine Hamiltonian path for a rectangle (no obstacles).
fn construct_snake(w: usize, h: usize) -> Vec<(usize, usize)> {
    let mut path = Vec::with_capacity(w * h);
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                path.push((x, y));
            }
        } else {
            for x in (0..w).rev() {
                path.push((x, y));
            }
        }
    }
    path
}

/// Construct a Hamiltonian path whose last vertex is adjacent to the first
/// (i.e. Hamiltonian cycle path) when height is even (no obstacles).
fn construct_cycle_even_height(w: usize, h: usize) -> Option<Vec<(usize, usize)>> {
    if h % 2 == 1 {
        return None;
    }
    if w == 1 {
        // Only possible if h == 2 (two cells line)
        if h == 2 {
            return Some(vec![(0, 0), (0, 1)]);
        } else {
            return None;
        }
    }
    let mut path = Vec::with_capacity(w * h);
    // Top row
    for x in 0..w {
        path.push((x, 0));
    }
    // Rows 1..h-1 without column 0
    for y in 1..h {
        if y % 2 == 1 {
            for x in (1..w).rev() {
                path.push((x, y));
            }
        } else {
            for x in 1..w {
                path.push((x, y));
            }
        }
    }
    // Column 0 bottom to top (excluding (0,0))
    for y in (1..h).rev() {
        path.push((0, y));
    }

    if path.len() == w * h {
        Some(path)
    } else {
        None
    }
}

/// Construct Hamiltonian cycle path for obstacle‑free rectangles.
/// Returns None if impossible.
fn construct_cycle(w: usize, h: usize) -> Option<Vec<(usize, usize)>> {
    if w * h <= 1 {
        return None;
    }
    if (w * h) % 2 == 1 {
        return None;
    }
    if h % 2 == 0 {
        return construct_cycle_even_height(w, h);
    }
    if w % 2 == 0 {
        // transpose, build, then swap back
        let p = construct_cycle_even_height(h, w)?;
        let mut out = Vec::with_capacity(p.len());
        for (x, y) in p {
            out.push((y, x));
        }
        return Some(out);
    }
    None
}

#[derive(Clone)]
struct RNG {
    state: u64,
}
impl RNG {
    fn new(seed: u64) -> Self {
        RNG { state: seed }
    }
    fn next(&mut self) -> u64 {
        self.state = self.state.wrapping_mul(6364136223846793005).wrapping_add(1);
        self.state
    }
}

/// Backtracking solver for small grids
struct Backtracker {
    w: usize,
    h: usize,
    total: usize,
    n_free: usize,
    neighbors: Vec<Vec<usize>>,
    free_cells: Vec<usize>,
    visited: Vec<bool>,
    degree: Vec<u8>,
    path: Vec<usize>,
    start: usize,
    require_cycle: bool,
    start_time: Instant,
    time_limit: Duration,
    timed_out: bool,
}

impl Backtracker {
    fn check_dead_end(&self) -> bool {
        for &c in &self.free_cells {
            if !self.visited[c] && self.degree[c] == 0 {
                return false;
            }
        }
        true
    }

    fn check_connectivity(&self, current: usize, visited_count: usize) -> bool {
        let remaining = self.n_free - visited_count;
        if remaining == 0 {
            return true;
        }
        let mut seen = vec![false; self.total];
        let mut q = VecDeque::new();
        seen[current] = true;
        q.push_back(current);
        let mut count = 0usize;

        while let Some(v) = q.pop_front() {
            if v == current || !self.visited[v] {
                count += 1;
            }
            for &u in &self.neighbors[v] {
                if !self.visited[u] && !seen[u] {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        count == remaining + 1
    }

    fn dfs(&mut self, current: usize, visited_count: usize) -> bool {
        if self.start_time.elapsed() > self.time_limit {
            self.timed_out = true;
            return false;
        }
        if visited_count == self.n_free {
            if !self.require_cycle {
                return true;
            }
            // must be adjacent to start
            return self.neighbors[current].iter().any(|&u| u == self.start);
        }

        let mut cand: Vec<usize> = self.neighbors[current]
            .iter()
            .filter(|&&v| !self.visited[v])
            .copied()
            .collect();
        if cand.is_empty() {
            return false;
        }
        cand.sort_by_key(|&v| self.degree[v]);

        for v in cand {
            self.visited[v] = true;
            self.path.push(v);
            let mut affected = Vec::new();
            for &u in &self.neighbors[v] {
                if !self.visited[u] {
                    self.degree[u] -= 1;
                    affected.push(u);
                }
            }

            let mut ok = true;
            if self.require_cycle && visited_count + 1 < self.n_free {
                let mut has_unvisited_neighbor = false;
                for &u in &self.neighbors[self.start] {
                    if !self.visited[u] {
                        has_unvisited_neighbor = true;
                        break;
                    }
                }
                if !has_unvisited_neighbor {
                    ok = false;
                }
            }

            if ok && !self.check_dead_end() {
                ok = false;
            }
            if ok && !self.check_connectivity(v, visited_count + 1) {
                ok = false;
            }

            if ok {
                if self.dfs(v, visited_count + 1) {
                    return true;
                }
            }

            // restore
            for u in affected {
                self.degree[u] += 1;
            }
            self.visited[v] = false;
            self.path.pop();

            if self.timed_out {
                return false;
            }
        }
        false
    }
}

/// Greedy heuristic for large obstructed grids
fn greedy_attempt(
    neighbors: &Vec<Vec<usize>>,
    blocked: &Vec<bool>,
    start: usize,
    n_free: usize,
    require_cycle: bool,
    rng: &mut RNG,
) -> Option<Vec<usize>> {
    let total = blocked.len();
    let mut visited = blocked.clone();
    let mut path = Vec::with_capacity(n_free);
    let mut current = start;
    visited[current] = true;
    path.push(current);

    for _ in 1..n_free {
        let mut best = Vec::new();
        let mut best_deg = 5usize;
        for &v in &neighbors[current] {
            if !visited[v] {
                let deg = neighbors[v].iter().filter(|&&u| !visited[u]).count();
                if deg < best_deg {
                    best_deg = deg;
                    best.clear();
                    best.push(v);
                } else if deg == best_deg {
                    best.push(v);
                }
            }
        }
        if best.is_empty() {
            return None;
        }
        let idx = if best.len() == 1 {
            0
        } else {
            (rng.next() as usize) % best.len()
        };
        let v = best[idx];
        visited[v] = true;
        current = v;
        path.push(v);
    }

    if require_cycle {
        let adj = neighbors[current].iter().any(|&u| u == start);
        if !adj {
            return None;
        }
    }

    Some(path)
}

fn main() {
    // ----- Read input -----
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let obs: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = it.next().unwrap().parse::<u32>().unwrap() != 0;

    let total = w * h;
    let mut blocked = vec![false; total];
    for _ in 0..obs {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < w && y < h {
            blocked[idx(x, y, w)] = true;
        }
    }

    let start = idx(0, 0, w);
    if blocked[start] {
        println!("0");
        return;
    }

    // ----- Build graph -----
    let mut neighbors = vec![Vec::new(); total];
    let mut free_cells = Vec::new();
    for y in 0..h {
        for x in 0..w {
            let i = idx(x, y, w);
            if blocked[i] {
                continue;
            }
            free_cells.push(i);
            if x > 0 && !blocked[idx(x - 1, y, w)] {
                neighbors[i].push(idx(x - 1, y, w));
            }
            if x + 1 < w && !blocked[idx(x + 1, y, w)] {
                neighbors[i].push(idx(x + 1, y, w));
            }
            if y > 0 && !blocked[idx(x, y - 1, w)] {
                neighbors[i].push(idx(x, y - 1, w));
            }
            if y + 1 < h && !blocked[idx(x, y + 1, w)] {
                neighbors[i].push(idx(x, y + 1, w));
            }
        }
    }

    let n_free = free_cells.len();
    if n_free == 0 {
        println!("0");
        return;
    }

    // ----- Basic necessary checks -----
    let mut black = 0usize;
    let mut white = 0usize;
    for &i in &free_cells {
        let (x, y) = coord(i, w);
        if (x + y) % 2 == 0 {
            black += 1;
        } else {
            white += 1;
        }
    }
    if require_cycle {
        if n_free % 2 == 1 || black != white {
            println!("0");
            return;
        }
        if n_free > 2 {
            for &i in &free_cells {
                if neighbors[i].len() < 2 {
                    println!("0");
                    return;
                }
            }
        }
    } else {
        let diff = if black > white { black - white } else { white - black };
        if diff > 1 {
            println!("0");
            return;
        }
        if diff == 1 {
            let (sx, sy) = (0usize, 0usize);
            let start_color_black = (sx + sy) % 2 == 0;
            if (black > white && !start_color_black) || (white > black && start_color_black) {
                println!("0");
                return;
            }
        }
    }

    // connectivity check
    {
        let mut seen = vec![false; total];
        let mut q = VecDeque::new();
        q.push_back(start);
        seen[start] = true;
        let mut count = 0usize;
        while let Some(v) = q.pop_front() {
            count += 1;
            for &u in &neighbors[v] {
                if !seen[u] {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        if count != n_free {
            println!("0");
            return;
        }
    }

    // ----- Obstacle-free fast constructive -----
    if obs == 0 {
        if require_cycle {
            if let Some(p) = construct_cycle(w, h) {
                println!("{}", p.len());
                for (x, y) in p {
                    println!("{} {}", x, y);
                }
                return;
            } else {
                println!("0");
                return;
            }
        } else {
            let p = construct_snake(w, h);
            println!("{}", p.len());
            for (x, y) in p {
                println!("{} {}", x, y);
            }
            return;
        }
    }

    // ----- Exact backtracking for small grids -----
    const EXACT_LIMIT: usize = 70;
    if n_free <= EXACT_LIMIT {
        let mut visited = blocked.clone();
        visited[start] = true;
        let mut degree = vec![0u8; total];
        for &i in &free_cells {
            degree[i] = neighbors[i].len() as u8;
        }
        // update degrees for start visited
        for &u in &neighbors[start] {
            if !visited[u] {
                degree[u] -= 1;
            }
        }

        let mut solver = Backtracker {
            w,
            h,
            total,
            n_free,
            neighbors: neighbors.clone(),
            free_cells,
            visited,
            degree,
            path: vec![start],
            start,
            require_cycle,
            start_time: Instant::now(),
            time_limit: Duration::from_secs(270),
            timed_out: false,
        };

        let ok = solver.dfs(start, 1);
        if ok {
            println!("{}", solver.path.len());
            for idx in solver.path {
                let (x, y) = coord(idx, w);
                println!("{} {}", x, y);
            }
            return;
        } else {
            println!("0");
            return;
        }
    }

    // ----- Greedy heuristic for large obstructed grids -----
    let attempts = if n_free <= 500 { 200 } else { 500 };
    let mut rng = RNG::new(0xC0FFEE);
    for _ in 0..attempts {
        if let Some(path) = greedy_attempt(&neighbors, &blocked, start, n_free, require_cycle, &mut rng) {
            println!("{}", path.len());
            for idx in path {
                let (x, y) = coord(idx, w);
                println!("{} {}", x, y);
            }
            return;
        }
    }

    println!("0");
}
```

---

### ✅ Summary of Strategy

- **Obstacle‑free:** Uses deterministic, linear-time constructions for massive grids.
- **Small with obstacles (≤70):** Exact backtracking with pruning & connectivity checks.
- **Large with obstacles:** Fast greedy heuristic attempts.

This adapts to size & density and runs fast on 2500+ cells for open grids while still handling small obstructed grids correctly.

If you want a *purely exact* solver for larger obstructed grids, that’s NP‑complete and may exceed 5 minutes — but this design balances correctness with scalability.
