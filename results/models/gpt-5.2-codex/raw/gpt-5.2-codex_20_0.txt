The solution uses:

✅ **Local sliding‑window statistics** per hole  
✅ **Adaptive window size + thresholds** based on dataset scale  
✅ **Robust handling of missing/invalid data**  
✅ **Sorting by anomaly confidence**

### ✅ Rust Program

```rust
use std::cmp::Ordering;
use std::io::{self, Read};

#[derive(Clone, Debug)]
struct Sample {
    depth: f64,
    props: Vec<Option<f64>>,
}

#[derive(Clone, Debug)]
struct Hole {
    id: String,
    samples: Vec<Sample>,
}

#[derive(Clone, Debug)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    prop_name: String,
    confidence: f64,
}

#[derive(Clone, Debug)]
struct Welford {
    n: usize,
    mean: f64,
    m2: f64,
}

impl Welford {
    fn new() -> Self {
        Self { n: 0, mean: 0.0, m2: 0.0 }
    }
    fn push(&mut self, x: f64) {
        self.n += 1;
        let delta = x - self.mean;
        self.mean += delta / self.n as f64;
        let delta2 = x - self.mean;
        self.m2 += delta * delta2;
    }
    fn mean(&self) -> f64 { self.mean }
    fn var(&self) -> f64 {
        if self.n < 2 { 0.0 } else { self.m2 / (self.n as f64 - 1.0) }
    }
    fn std(&self) -> f64 { self.var().sqrt() }
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self { buf: input.into_bytes(), idx: 0 }
    }

    fn next<T: std::str::FromStr>(&mut self) -> Option<T> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let start = self.idx;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        std::str::from_utf8(&self.buf[start..self.idx]).ok()?.parse().ok()
    }

    fn next_string(&mut self) -> Option<String> {
        self.next::<String>()
    }
}

fn main() {
    let mut sc = Scanner::new();

    let num_holes: usize = match sc.next() { Some(v) => v, None => return };
    let num_props: usize = sc.next().unwrap_or(0);

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(sc.next_string().unwrap_or_else(|| "prop".to_string()));
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut globals: Vec<Welford> = vec![Welford::new(); num_props];
    let mut total_samples: usize = 0;

    for _ in 0..num_holes {
        let hole_id = sc.next_string().unwrap_or_else(|| "H".to_string());
        // read geometry (ignored for now)
        let _sx: f64 = sc.next().unwrap_or(0.0);
        let _sy: f64 = sc.next().unwrap_or(0.0);
        let _sz: f64 = sc.next().unwrap_or(0.0);
        let _dx: f64 = sc.next().unwrap_or(0.0);
        let _dy: f64 = sc.next().unwrap_or(0.0);
        let _dz: f64 = sc.next().unwrap_or(0.0);
        let _length: f64 = sc.next().unwrap_or(0.0);
        let nsamples: usize = sc.next().unwrap_or(0);

        let mut samples = Vec::with_capacity(nsamples);
        for _ in 0..nsamples {
            let depth: f64 = sc.next().unwrap_or(0.0);
            let mut props = Vec::with_capacity(num_props);
            for p in 0..num_props {
                let tok: String = sc.next_string().unwrap_or_default();
                if tok == "NA" || tok == "NaN" || tok == "" {
                    props.push(None);
                } else {
                    let val: f64 = tok.parse().unwrap_or(f64::NAN);
                    if val.is_finite() {
                        globals[p].push(val);
                        props.push(Some(val));
                    } else {
                        props.push(None);
                    }
                }
            }
            samples.push(Sample { depth, props });
        }

        total_samples += nsamples;
        holes.push(Hole { id: hole_id, samples });
    }

    let global_means: Vec<f64> = globals.iter().map(|g| g.mean()).collect();
    let global_stds: Vec<f64> = globals.iter().map(|g| g.std()).collect();

    // adaptive strategy
    let (window_radius, threshold) = if total_samples < 5_000 {
        (2usize, 2.5f64)
    } else if total_samples < 200_000 {
        (4usize, 3.0f64)
    } else {
        (6usize, 3.5f64)
    };

    let mut suspects: Vec<Suspect> = Vec::new();

    for hole in &holes {
        let n = hole.samples.len();
        if n == 0 { continue; }

        for p in 0..num_props {
            // prefix sums for this property
            let mut pref_count = vec![0usize; n + 1];
            let mut pref_sum = vec![0.0f64; n + 1];
            let mut pref_sq = vec![0.0f64; n + 1];

            for i in 0..n {
                pref_count[i + 1] = pref_count[i];
                pref_sum[i + 1] = pref_sum[i];
                pref_sq[i + 1] = pref_sq[i];
                if let Some(v) = hole.samples[i].props[p] {
                    pref_count[i + 1] += 1;
                    pref_sum[i + 1] += v;
                    pref_sq[i + 1] += v * v;
                }
            }

            for i in 0..n {
                let val = match hole.samples[i].props[p] {
                    Some(v) => v,
                    None => continue,
                };

                let start = if i >= window_radius { i - window_radius } else { 0 };
                let end = usize::min(n, i + window_radius + 1);
                let count = pref_count[end] - pref_count[start];
                if count < 3 { continue; }

                let sum = pref_sum[end] - pref_sum[start];
                let sq = pref_sq[end] - pref_sq[start];
                let mean = sum / count as f64;
                let var = (sq / count as f64) - mean * mean;
                let std = if var > 0.0 { var.sqrt() } else { 0.0 };

                let mut z_local = 0.0;
                if std > 1e-9 { z_local = (val - mean).abs() / std; }

                let gstd = global_stds[p];
                let gmean = global_means[p];
                let z_global = if gstd > 1e-9 { (val - gmean).abs() / gstd } else { 0.0 };

                let confidence = z_local.max(z_global);

                if confidence >= threshold {
                    suspects.push(Suspect {
                        hole_id: hole.id.clone(),
                        sample_idx: i,
                        prop_name: prop_names[p].clone(),
                        confidence,
                    });
                }
            }
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence.partial_cmp(&a.confidence).unwrap_or(Ordering::Equal)
    });

    println!("{}", suspects.len());
    for s in suspects {
        println!(
            "{} {} {} {:.4}",
            s.hole_id, s.sample_idx, s.prop_name, s.confidence
        );
    }
}
```

---

### ✅ Notes on Behavior

- **Confidence score** is derived from the maximum of:
  - local z‑score (nearby samples in same hole)
  - global z‑score (entire dataset)

- **Adaptive logic**: as dataset grows, window size and thresholds increase to remain efficient and conservative.

- **Missing values**: `"NA"` or invalid numbers are ignored.

- **Duplicate entries**: treated as separate samples.
