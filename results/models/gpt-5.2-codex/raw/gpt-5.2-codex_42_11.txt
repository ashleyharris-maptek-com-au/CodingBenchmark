OpCapability Shader
OpCapability Geometry
%GLSL450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_02 = OpConstant %float 0.02
%float_0_33333334 = OpConstant %float 0.33333334
%float_0_6666667 = OpConstant %float 0.6666667

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%func = OpTypeFunction %void

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1

%diff_prev = OpFSub %v2float %p1_xy %p0_xy
%dot_prev = OpDot %float %diff_prev %diff_prev
%invlen_prev = OpExtInst %float %GLSL450 InverseSqrt %dot_prev
%dir_prev = OpVectorTimesScalar %v2float %diff_prev %invlen_prev

%diff_curr = OpFSub %v2float %p2_xy %p1_xy
%dot_curr = OpDot %float %diff_curr %diff_curr
%invlen_curr = OpExtInst %float %GLSL450 InverseSqrt %dot_curr
%dir_curr = OpVectorTimesScalar %v2float %diff_curr %invlen_curr

%diff_next = OpFSub %v2float %p3_xy %p2_xy
%dot_next = OpDot %float %diff_next %diff_next
%invlen_next = OpExtInst %float %GLSL450 InverseSqrt %dot_next
%dir_next = OpVectorTimesScalar %v2float %diff_next %invlen_next

%sum0 = OpFAdd %v2float %dir_prev %dir_curr
%dot_sum0 = OpDot %float %sum0 %sum0
%invlen_sum0 = OpExtInst %float %GLSL450 InverseSqrt %dot_sum0
%t0 = OpVectorTimesScalar %v2float %sum0 %invlen_sum0

%sum1 = OpFAdd %v2float %dir_curr %dir_next
%dot_sum1 = OpDot %float %sum1 %sum1
%invlen_sum1 = OpExtInst %float %GLSL450 InverseSqrt %dot_sum1
%t1 = OpVectorTimesScalar %v2float %sum1 %invlen_sum1

%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%neg_t0_y = OpFNegate %float %t0_y
%n0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%n1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

%dir_curr_x = OpCompositeExtract %float %dir_curr 0
%dir_curr_y = OpCompositeExtract %float %dir_curr 1
%neg_dir_curr_y = OpFNegate %float %dir_curr_y
%nmid = OpCompositeConstruct %v2float %neg_dir_curr_y %dir_curr_x

%off0 = OpVectorTimesScalar %v2float %n0 %float_0_02
%off1 = OpVectorTimesScalar %v2float %n1 %float_0_02
%offm = OpVectorTimesScalar %v2float %nmid %float_0_02

%off0_x = OpCompositeExtract %float %off0 0
%off0_y = OpCompositeExtract %float %off0 1
%off0_v4 = OpCompositeConstruct %v4float %off0_x %off0_y %float_0 %float_0
%off0_v4_neg = OpFNegate %v4float %off0_v4

%off1_x = OpCompositeExtract %float %off1 0
%off1_y = OpCompositeExtract %float %off1 1
%off1_v4 = OpCompositeConstruct %v4float %off1_x %off1_y %float_0 %float_0
%off1_v4_neg = OpFNegate %v4float %off1_v4

%offm_x = OpCompositeExtract %float %offm 0
%offm_y = OpCompositeExtract %float %offm 1
%offm_v4 = OpCompositeConstruct %v4float %offm_x %offm_y %float_0 %float_0
%offm_v4_neg = OpFNegate %v4float %offm_v4

%delta = OpFSub %v4float %p2 %p1
%delta_third = OpVectorTimesScalar %v4float %delta %float_0_33333334
%delta_twothird = OpVectorTimesScalar %v4float %delta %float_0_6666667
%pA = OpFAdd %v4float %p1 %delta_third
%pB = OpFAdd %v4float %p1 %delta_twothird

%col1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%col2 = OpLoad %v3float %col2_ptr
%col_delta = OpFSub %v3float %col2 %col1
%col_delta_third = OpVectorTimesScalar %v3float %col_delta %float_0_33333334
%col_delta_twothird = OpVectorTimesScalar %v3float %col_delta %float_0_6666667
%colA = OpFAdd %v3float %col1 %col_delta_third
%colB = OpFAdd %v3float %col1 %col_delta_twothird

%gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

%Aplus = OpFAdd %v4float %p1 %off0_v4
%Aminus = OpFAdd %v4float %p1 %off0_v4_neg
%Bplus = OpFAdd %v4float %pA %offm_v4
%Bminus = OpFAdd %v4float %pA %offm_v4_neg

OpStore %gl_out_pos %Aplus
OpStore %outColor %col1
OpEmitVertex

OpStore %gl_out_pos %Aminus
OpStore %outColor %col1
OpEmitVertex

OpStore %gl_out_pos %Bplus
OpStore %outColor %colA
OpEmitVertex

OpStore %gl_out_pos %Bminus
OpStore %outColor %colA
OpEmitVertex

OpEndPrimitive

%Cplus = OpFAdd %v4float %pB %offm_v4
%Cminus = OpFAdd %v4float %pB %offm_v4_neg
%Dplus = OpFAdd %v4float %p2 %off1_v4
%Dminus = OpFAdd %v4float %p2 %off1_v4_neg

OpStore %gl_out_pos %Cplus
OpStore %outColor %colB
OpEmitVertex

OpStore %gl_out_pos %Cminus
OpStore %outColor %colB
OpEmitVertex

OpStore %gl_out_pos %Dplus
OpStore %outColor %col2
OpEmitVertex

OpStore %gl_out_pos %Dminus
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
