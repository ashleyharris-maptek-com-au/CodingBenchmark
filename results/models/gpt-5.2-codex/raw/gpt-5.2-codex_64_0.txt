```typescript
declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Cell = [number, number];

const grid: number[][] = data.probability;
const drones: { id: any; x: number; y: number }[] = data.drones;
const maxSteps: number = data.maxStepsPerDrone;

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < width && y < height;
}
function isObstacle(x: number, y: number): boolean {
  return grid[y][x] === -1;
}

const visited: boolean[][] = Array.from({ length: height }, () =>
  Array.from({ length: width }, () => false)
);

function markVisited(x: number, y: number) {
  if (grid[y][x] > 0 && !visited[y][x]) {
    visited[y][x] = true;
  }
}

function bfsNextStep(sx: number, sy: number): Cell | null {
  const n = width * height;
  const q = new Int32Array(n);
  const prev = new Int32Array(n);
  prev.fill(-1);
  const seen = new Uint8Array(n);
  const startIdx = sy * width + sx;
  let head = 0,
    tail = 0;
  q[tail++] = startIdx;
  seen[startIdx] = 1;

  let found = -1;

  const dirs = [
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];

  while (head < tail) {
    const idx = q[head++];
    const x = idx % width;
    const y = (idx / width) | 0;

    if (!(x === sx && y === sy) && grid[y][x] > 0 && !visited[y][x]) {
      found = idx;
      break;
    }

    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (!inBounds(nx, ny) || isObstacle(nx, ny)) continue;
      const nidx = ny * width + nx;
      if (!seen[nidx]) {
        seen[nidx] = 1;
        prev[nidx] = idx;
        q[tail++] = nidx;
      }
    }
  }

  if (found === -1) return null;

  // Walk back to find the first step from start
  let cur = found;
  while (prev[cur] !== startIdx && prev[cur] !== -1) {
    cur = prev[cur];
  }
  const nx = cur % width;
  const ny = (cur / width) | 0;
  return [nx, ny];
}

const paths: Cell[][] = [];

for (const drone of drones) {
  let x = drone.x;
  let y = drone.y;
  const path: Cell[] = [];
  path.push([x, y]);
  markVisited(x, y);

  for (let step = 0; step < maxSteps; step++) {
    let bestScore = -1;
    let bestMove: Cell | null = null;

    const candidates: Cell[] = [
      [x, y], // stay
      [x + 1, y],
      [x - 1, y],
      [x, y + 1],
      [x, y - 1],
    ];

    for (const [nx, ny] of candidates) {
      if (!inBounds(nx, ny) || isObstacle(nx, ny)) continue;
      const score = grid[ny][nx] > 0 && !visited[ny][nx] ? grid[ny][nx] : 0;
      if (score > bestScore) {
        bestScore = score;
        bestMove = [nx, ny];
      }
    }

    if (bestScore === 0) {
      const next = bfsNextStep(x, y);
      if (next) {
        bestMove = next;
      } else {
        bestMove = [x, y];
      }
    } else if (bestMove === null) {
      bestMove = [x, y];
    }

    x = bestMove[0];
    y = bestMove[1];
    path.push([x, y]);
    markVisited(x, y);
  }

  paths.push(path);
}

const output = { paths };
process.stdout.write(JSON.stringify(output));
```
