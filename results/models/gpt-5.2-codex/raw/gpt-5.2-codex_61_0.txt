```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8').trim();
const data: any = input.length ? JSON.parse(input) : {};

const depot: [number, number] = data.depot || [0, 0];
const truckCount: number = Math.max(0, data.truckCount || 0);
const packages: any[] = Array.isArray(data.packages) ? data.packages : [];

const n = packages.length;
const k = Math.min(truckCount || n, n);

function nearestNeighborOrder(list: any[], depotX: number, depotY: number): any[] {
    const remaining = list.slice();
    const route: any[] = [];
    let cx = depotX, cy = depotY;

    while (remaining.length > 0) {
        let bestIdx = 0;
        let bestDist = Number.POSITIVE_INFINITY;
        for (let i = 0; i < remaining.length; i++) {
            const p = remaining[i];
            const dx = p.x - cx;
            const dy = p.y - cy;
            const d = dx * dx + dy * dy;
            if (d < bestDist) {
                bestDist = d;
                bestIdx = i;
            }
        }
        const chosen = remaining[bestIdx];
        // remove chosen in O(1)
        remaining[bestIdx] = remaining[remaining.length - 1];
        remaining.pop();
        route.push(chosen);
        cx = chosen.x;
        cy = chosen.y;
    }
    return route;
}

let routes: any[][] = [];

if (n === 0 || k === 0) {
    routes = [];
} else {
    const depotX = depot[0];
    const depotY = depot[1];

    const sorted = packages.map(p => ({
        id: p.id,
        x: p.x,
        y: p.y,
        angle: Math.atan2(p.y - depotY, p.x - depotX)
    })).sort((a, b) => a.angle - b.angle);

    const baseSize = Math.floor(n / k);
    const remainder = n % k;
    let index = 0;

    for (let i = 0; i < k; i++) {
        const size = baseSize + (i < remainder ? 1 : 0);
        const slice = sorted.slice(index, index + size);
        index += size;

        const ordered = nearestNeighborOrder(slice, depotX, depotY);
        routes.push(ordered.map(p => p.id));
    }
}

const output = { routes };
process.stdout.write(JSON.stringify(output));
```
