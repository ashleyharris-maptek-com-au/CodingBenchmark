OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1

%float_0 = OpConstant %float 0.0
%float_0_8 = OpConstant %float 0.8
%float_r = OpConstant %float 0.04
%float_rhalf = OpConstant %float 0.02
%float_ry = OpConstant %float 0.034641016
%float_nr = OpConstant %float -0.04
%float_nrhalf = OpConstant %float -0.02
%float_nry = OpConstant %float -0.034641016

%v4_e0 = OpConstantComposite %v4float %float_r %float_0 %float_0 %float_0
%v4_e1 = OpConstantComposite %v4float %float_rhalf %float_ry %float_0 %float_0
%v4_e2 = OpConstantComposite %v4float %float_nrhalf %float_ry %float_0 %float_0
%v4_e3 = OpConstantComposite %v4float %float_nr %float_0 %float_0 %float_0
%v4_e4 = OpConstantComposite %v4float %float_nrhalf %float_nry %float_0 %float_0
%v4_e5 = OpConstantComposite %v4float %float_rhalf %float_nry %float_0 %float_0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel

%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %in_pos_ptr

%in_col_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%col = OpLoad %v3float %in_col_ptr
%edgeCol = OpVectorTimesScalar %v3float %col %float_0_8

%pos_e0 = OpFAdd %v4float %pos %v4_e0
%pos_e1 = OpFAdd %v4float %pos %v4_e1
%pos_e2 = OpFAdd %v4float %pos %v4_e2
%pos_e3 = OpFAdd %v4float %pos %v4_e3
%pos_e4 = OpFAdd %v4float %pos %v4_e4
%pos_e5 = OpFAdd %v4float %pos %v4_e5

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; triangle 0: center, e0, e1
OpStore %out_pos_ptr %pos
OpStore %outColor %col
OpEmitVertex

OpStore %out_pos_ptr %pos_e0
OpStore %outColor %edgeCol
OpEmitVertex

OpStore %out_pos_ptr %pos_e1
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

; triangle 1: center, e1, e2
OpStore %out_pos_ptr %pos
OpStore %outColor %col
OpEmitVertex

OpStore %out_pos_ptr %pos_e1
OpStore %outColor %edgeCol
OpEmitVertex

OpStore %out_pos_ptr %pos_e2
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

; triangle 2: center, e2, e3
OpStore %out_pos_ptr %pos
OpStore %outColor %col
OpEmitVertex

OpStore %out_pos_ptr %pos_e2
OpStore %outColor %edgeCol
OpEmitVertex

OpStore %out_pos_ptr %pos_e3
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

; triangle 3: center, e3, e4
OpStore %out_pos_ptr %pos
OpStore %outColor %col
OpEmitVertex

OpStore %out_pos_ptr %pos_e3
OpStore %outColor %edgeCol
OpEmitVertex

OpStore %out_pos_ptr %pos_e4
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

; triangle 4: center, e4, e5
OpStore %out_pos_ptr %pos
OpStore %outColor %col
OpEmitVertex

OpStore %out_pos_ptr %pos_e4
OpStore %outColor %edgeCol
OpEmitVertex

OpStore %out_pos_ptr %pos_e5
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

; triangle 5: center, e5, e0
OpStore %out_pos_ptr %pos
OpStore %outColor %col
OpEmitVertex

OpStore %out_pos_ptr %pos_e5
OpStore %outColor %edgeCol
OpEmitVertex

OpStore %out_pos_ptr %pos_e0
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
