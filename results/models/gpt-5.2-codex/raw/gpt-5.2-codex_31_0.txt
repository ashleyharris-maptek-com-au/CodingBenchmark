use std::collections::hash_map::Entry;
use std::collections::HashMap;
use std::io::{self, Read};

const MITM_LIMIT: usize = 40;
const BITSET_TARGET_LIMIT: i64 = 5_000_000;
const RANGE_LIMIT: i64 = 2_000_000;

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let target = match input.next_i64() {
        Some(v) => v,
        None => return,
    };

    let mut nums = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(v) = input.next_i64() {
            nums.push(v);
        } else {
            break;
        }
    }

    let result = solve_subset_sum(&nums, target);

    let mut out = String::new();
    if let Some(indices) = result {
        out.push_str("YES\n");
        if !indices.is_empty() {
            for (i, idx) in indices.iter().enumerate() {
                if i > 0 {
                    out.push(' ');
                }
                out.push_str(&idx.to_string());
            }
        }
        out.push('\n');
    } else {
        out.push_str("NO\n");
    }
    print!("{out}");
}

fn solve_subset_sum(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    let n = nums.len();
    if n == 0 {
        return None;
    }

    if n <= MITM_LIMIT {
        return meet_in_middle(nums, target);
    }

    let mut min_sum = 0i64;
    let mut max_sum = 0i64;
    let mut all_nonneg = true;

    for (i, &v) in nums.iter().enumerate() {
        if v == target {
            return Some(vec![i]);
        }
        if v < 0 {
            all_nonneg = false;
            min_sum += v;
        } else {
            max_sum += v;
        }
    }

    if target < min_sum || target > max_sum {
        return None;
    }

    if all_nonneg && target >= 0 && target <= BITSET_TARGET_LIMIT {
        return bitset_dp(nums, target);
    }

    let range = max_sum - min_sum;
    if range <= RANGE_LIMIT {
        return offset_dp(nums, target, min_sum, max_sum);
    }

    hashmap_dp(nums, target)
}

fn meet_in_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let n1 = n / 2;
    let n2 = n - n1;

    let total1 = 1usize << n1;
    let mut map: HashMap<i64, u64> = HashMap::with_capacity(total1);

    for mask in 0..total1 {
        let mut sum = 0i64;
        for i in 0..n1 {
            if (mask >> i) & 1 == 1 {
                sum += nums[i];
            }
        }
        map.entry(sum).or_insert(mask as u64);
    }

    let total2 = 1usize << n2;
    for mask in 0..total2 {
        let mut sum = 0i64;
        for i in 0..n2 {
            if (mask >> i) & 1 == 1 {
                sum += nums[n1 + i];
            }
        }
        let need = target - sum;
        if let Some(&mask1) = map.get(&need) {
            let mut res = Vec::new();
            for i in 0..n1 {
                if (mask1 >> i) & 1 == 1 {
                    res.push(i);
                }
            }
            for i in 0..n2 {
                if (mask >> i) & 1 == 1 {
                    res.push(n1 + i);
                }
            }
            return Some(res);
        }
    }
    None
}

fn bitset_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let t = target as usize;
    let len = t / 64 + 1;

    let mut bits = vec![0u64; len];
    bits[0] = 1;

    let mut prev_idx = vec![-1i32; t + 1];
    let mut prev_sum = vec![0i32; t + 1];
    prev_idx[0] = -2;

    let mut shifted = vec![0u64; len];

    let last_mask = if t % 64 == 63 {
        u64::MAX
    } else {
        (1u64 << (t % 64 + 1)) - 1
    };

    for (idx, &v) in nums.iter().enumerate() {
        if v <= 0 {
            continue;
        }
        let shift = v as usize;
        if shift > t {
            continue;
        }

        for x in shifted.iter_mut() {
            *x = 0;
        }

        let word_shift = shift / 64;
        let bit_shift = shift % 64;

        for i in word_shift..len {
            let mut val = bits[i - word_shift] << bit_shift;
            if bit_shift > 0 && i > word_shift {
                val |= bits[i - word_shift - 1] >> (64 - bit_shift);
            }
            shifted[i] = val;
        }

        for i in 0..len {
            let mut new_word = shifted[i] & !bits[i];
            if i == len - 1 {
                new_word &= last_mask;
            }
            if new_word != 0 {
                let base = i * 64;
                while new_word != 0 {
                    let tz = new_word.trailing_zeros() as usize;
                    let sum = base + tz;
                    if sum <= t {
                        prev_idx[sum] = idx as i32;
                        prev_sum[sum] = (sum - shift) as i32;
                    }
                    new_word &= new_word - 1;
                }
            }
            if i == len - 1 {
                bits[i] = (bits[i] | shifted[i]) & last_mask;
            } else {
                bits[i] |= shifted[i];
            }
        }

        if prev_idx[t] != -1 {
            break;
        }
    }

    if prev_idx[t] == -1 {
        None
    } else {
        Some(reconstruct(&prev_idx, &prev_sum, t))
    }
}

fn offset_dp(nums: &[i64], target: i64, min_sum: i64, max_sum: i64) -> Option<Vec<usize>> {
    if target < min_sum || target > max_sum {
        return None;
    }
    let range = (max_sum - min_sum) as usize;
    let offset = -min_sum;
    let size = range + 1;

    let mut prev_idx = vec![-1i32; size];
    let mut prev_sum = vec![0i32; size];

    let start = offset as usize;
    prev_idx[start] = -2;

    let mut reachable: Vec<usize> = Vec::new();
    reachable.push(start);

    let target_idx = (target + offset) as usize;

    for (idx, &v) in nums.iter().enumerate() {
        let current_len = reachable.len();
        for j in 0..current_len {
            let sum_idx = reachable[j];
            let sum = sum_idx as i64 - offset;
            let new_sum = sum + v;
            if new_sum < min_sum || new_sum > max_sum {
                continue;
            }
            let new_idx = (new_sum + offset) as usize;
            if prev_idx[new_idx] == -1 {
                prev_idx[new_idx] = idx as i32;
                prev_sum[new_idx] = sum_idx as i32;
                reachable.push(new_idx);
            }
        }
        if prev_idx[target_idx] != -1 {
            break;
        }
    }

    if prev_idx[target_idx] == -1 {
        None
    } else {
        Some(reconstruct(&prev_idx, &prev_sum, target_idx))
    }
}

fn hashmap_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut map: HashMap<i64, (i64, usize)> = HashMap::new();
    let mut sums: Vec<i64> = Vec::new();
    map.insert(0, (i64::MIN, usize::MAX));
    sums.push(0);

    for (idx, &v) in nums.iter().enumerate() {
        if v == 0 {
            continue;
        }
        let current_len = sums.len();
        for i in 0..current_len {
            let s = sums[i];
            let new_s = s + v;
            match map.entry(new_s) {
                Entry::Vacant(e) => {
                    e.insert((s, idx));
                    sums.push(new_s);
                    if new_s == target {
                        return Some(reconstruct_hashmap(&map, target));
                    }
                }
                Entry::Occupied(_) => {}
            }
        }
    }

    if map.contains_key(&target) {
        Some(reconstruct_hashmap(&map, target))
    } else {
        None
    }
}

fn reconstruct(prev_idx: &[i32], prev_sum: &[i32], mut cur: usize) -> Vec<usize> {
    let mut res = Vec::new();
    while prev_idx[cur] != -2 {
        let idx = prev_idx[cur] as usize;
        res.push(idx);
        cur = prev_sum[cur] as usize;
    }
    res.reverse();
    res
}

fn reconstruct_hashmap(map: &HashMap<i64, (i64, usize)>, target: i64) -> Vec<usize> {
    if target == 0 {
        return Vec::new();
    }
    let mut res = Vec::new();
    let mut cur = target;
    while cur != 0 {
        if let Some(&(prev, idx)) = map.get(&cur) {
            if prev == i64::MIN {
                break;
            }
            res.push(idx);
            cur = prev;
        } else {
            break;
        }
    }
    res.reverse();
    res
}

struct FastInput {
    data: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            data: input.into_bytes(),
            pos: 0,
        }
    }

    fn next_i64(&mut self) -> Option<i64> {
        let len = self.data.len();
        while self.pos < len && self.data[self.pos] <= b' ' {
            self.pos += 1;
        }
        if self.pos >= len {
            return None;
        }
        let mut sign = 1i64;
        if self.data[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        } else if self.data[self.pos] == b'+' {
            self.pos += 1;
        }
        let mut num = 0i64;
        while self.pos < len {
            let c = self.data[self.pos];
            if c < b'0' || c > b'9' {
                break;
            }
            num = num * 10 + (c - b'0') as i64;
            self.pos += 1;
        }
        Some(num * sign)
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_i64().map(|v| v as usize)
    }
}
