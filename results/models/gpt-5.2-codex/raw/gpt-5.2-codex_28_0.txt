```rust
use std::fmt::Write;
use std::io::{self, Read};

struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        FastInput { buf: input, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut val = 0usize;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            val = val * 10 + (self.buf[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        Some(val)
    }
}

fn main() {
    let mut input = FastInput::new();

    let n = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = match input.next_usize() {
        Some(v) => v,
        None => return,
    };

    let mut selected = vec![false; n];

    for _ in 0..m {
        let u = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match input.next_usize() {
            Some(v) => v,
            None => break,
        };

        if u < n && v < n && !selected[u] && !selected[v] {
            selected[u] = true;
            selected[v] = true;
        }
    }

    let mut vertices = Vec::new();
    vertices.reserve(n / 2);
    for (i, &sel) in selected.iter().enumerate() {
        if sel {
            vertices.push(i);
        }
    }

    let mut out = String::new();
    let _ = writeln!(out, "{}", vertices.len());
    if !vertices.is_empty() {
        for (i, v) in vertices.iter().enumerate() {
            if i > 0 {
                out.push(' ');
            }
            let _ = write!(out, "{}", v);
        }
    }
    out.push('\n');

    print!("{out}");
}
```

This program computes a valid vertex cover in a single pass using a greedy maximal matching approach.  
It guarantees all edges are covered and runs in linear time with very low memory overhead, suitable for graphs with millions of vertices and edges.
