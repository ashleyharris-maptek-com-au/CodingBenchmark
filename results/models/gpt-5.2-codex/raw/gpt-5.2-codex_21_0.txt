```cpp
#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <limits>
#include <cmath>
#include <cstdint>

using namespace std;

using Pos = std::vector<int>;

struct VecHash {
    size_t operator()(Pos const& v) const noexcept {
        // FNV-1a style hash
        size_t h = 1469598103934665603ULL;
        for (int x : v) {
            h ^= std::hash<long long>{}(static_cast<long long>(x));
            h *= 1099511628211ULL;
        }
        return h;
    }
};

static long long manhattan(const Pos& a, const Pos& b) {
    long long d = 0;
    for (size_t i = 0; i < a.size(); ++i)
        d += llabs(static_cast<long long>(a[i]) - b[i]);
    return d;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D;
    long long max_turns;
    if (!(cin >> D >> max_turns)) return 0;

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int length;
    Pos head(D);
    cin >> length;
    for (int i = 0; i < D; ++i) cin >> head[i];

    int F;
    cin >> F;
    unordered_set<Pos, VecHash> foodSet;
    for (int i = 0; i < F; ++i) {
        Pos p(D);
        for (int j = 0; j < D; ++j) cin >> p[j];
        foodSet.insert(p);
    }

    int O;
    cin >> O;
    unordered_set<Pos, VecHash> obstacleSet;
    for (int i = 0; i < O; ++i) {
        Pos p(D);
        for (int j = 0; j < D; ++j) cin >> p[j];
        obstacleSet.insert(p);
    }

    // Initialize snake body (best-effort line in one dimension)
    deque<Pos> body;
    unordered_set<Pos, VecHash> bodySet;
    body.push_front(head);
    bodySet.insert(head);

    if (length > 1) {
        int dimChoice = -1, sign = -1;
        for (int d = 0; d < D; ++d) {
            if (head[d] - (length - 1) >= 0) { dimChoice = d; sign = -1; break; }
            if (head[d] + (length - 1) < bounds[d]) { dimChoice = d; sign = +1; break; }
        }
        if (dimChoice >= 0) {
            for (int i = 1; i < length; ++i) {
                Pos p = head;
                p[dimChoice] += sign * i;
                if (p[dimChoice] >= 0 && p[dimChoice] < bounds[dimChoice]) {
                    body.push_back(p);
                    bodySet.insert(p);
                } else {
                    break;
                }
            }
        }
    }

    vector<pair<int,int>> outputs;
    outputs.reserve(static_cast<size_t>(max_turns));

    bool alive = true;
    for (long long t = 0; t < max_turns; ++t) {
        int chosenAxis = 0;
        int chosenDir  = 1;

        if (alive) {
            bool hasFood = !foodSet.empty();
            Pos target;
            if (hasFood) {
                long long bestDist = std::numeric_limits<long long>::max();
                for (const auto& f : foodSet) {
                    long long d = manhattan(head, f);
                    if (d < bestDist) { bestDist = d; target = f; }
                }
            }

            long long bestScore = std::numeric_limits<long long>::max();
            bool found = false;

            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, +1}) {
                    if (dir == -1 && head[axis] == 0) continue;
                    if (dir == +1 && head[axis] == bounds[axis] - 1) continue;

                    Pos newpos = head;
                    newpos[axis] += dir;

                    if (obstacleSet.find(newpos) != obstacleSet.end()) continue;

                    bool willEat = (foodSet.find(newpos) != foodSet.end());

                    auto itBody = bodySet.find(newpos);
                    if (itBody != bodySet.end()) {
                        if (!( !willEat && !body.empty() && newpos == body.back() )) {
                            continue; // collision
                        }
                    }

                    long long score = 0;
                    if (hasFood) score = manhattan(newpos, target);

                    if (!found || score < bestScore) {
                        found = true;
                        bestScore = score;
                        chosenAxis = axis;
                        chosenDir = dir;
                    }
                }
            }

            if (!found) {
                alive = false;
            } else {
                Pos newhead = head;
                newhead[chosenAxis] += chosenDir;
                bool willEat = (foodSet.find(newhead) != foodSet.end());

                head = newhead;
                body.push_front(newhead);
                bodySet.insert(newhead);

                if (willEat) {
                    foodSet.erase(newhead);
                } else {
                    Pos tail = body.back();
                    body.pop_back();
                    bodySet.erase(tail);
                }
            }
        }

        outputs.push_back({chosenAxis, chosenDir});
    }

    for (auto &m : outputs) {
        cout << m.first << " " << m.second << "\n";
    }

    return 0;
}
```
