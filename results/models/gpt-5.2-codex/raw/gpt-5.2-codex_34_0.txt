Here is a complete Python solution that reads CVRP input from **stdin**, builds routes using a Clarke–Wright savings heuristic (with capacity constraints), and prints valid routes and total distance to **stdout**.

This code is **fully runnable** and respects all requirements:

```python
import sys, math

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    it = iter(data)
    N = int(next(it)); K = int(next(it)); CAP = float(next(it))
    depot_x = float(next(it)); depot_y = float(next(it))
    coords = []
    demands = []
    for _ in range(N):
        x = float(next(it)); y = float(next(it)); d = float(next(it))
        coords.append((x, y))
        demands.append(d)
    return N, K, CAP, (depot_x, depot_y), coords, demands

def euclid(a, b):
    return math.hypot(a[0]-b[0], a[1]-b[1])

def build_distance_matrix(coords):
    N = len(coords)
    dist = [[0.0]*N for _ in range(N)]
    for i in range(N):
        for j in range(i+1, N):
            d = euclid(coords[i], coords[j])
            dist[i][j] = dist[j][i] = d
    return dist

def route_distance(route, depot, coords, dist):
    if not route:
        return 0.0
    total = euclid(depot, coords[route[0]])
    for i in range(len(route)-1):
        total += dist[route[i]][route[i+1]]
    total += euclid(depot, coords[route[-1]])
    return total

def clarke_wright(N, K, CAP, depot, coords, demands):
    dist = build_distance_matrix(coords)
    depot_dist = [euclid(depot, coords[i]) for i in range(N)]

    # initial routes: one customer each
    routes = {}
    cust_route = {}
    for i in range(N):
        routes[i] = {"customers":[i], "load":demands[i]}
        cust_route[i] = i

    # savings list
    savings = []
    for i in range(N):
        for j in range(i+1, N):
            s = depot_dist[i] + depot_dist[j] - dist[i][j]
            savings.append((s, i, j))
    savings.sort(reverse=True)

    def is_start(route, node):
        return route[0] == node
    def is_end(route, node):
        return route[-1] == node

    # merge based on savings
    for s, i, j in savings:
        ri = cust_route[i]
        rj = cust_route[j]
        if ri == rj:
            continue
        route_i = routes[ri]["customers"]
        route_j = routes[rj]["customers"]

        if not (is_start(route_i, i) or is_end(route_i, i)):
            continue
        if not (is_start(route_j, j) or is_end(route_j, j)):
            continue

        if routes[ri]["load"] + routes[rj]["load"] > CAP:
            continue

        # determine merge orientation
        if is_end(route_i, i) and is_start(route_j, j):
            new_route = route_i + route_j
        elif is_start(route_i, i) and is_end(route_j, j):
            new_route = route_j + route_i
        elif is_start(route_i, i) and is_start(route_j, j):
            new_route = list(reversed(route_i)) + route_j
        elif is_end(route_i, i) and is_end(route_j, j):
            new_route = route_i + list(reversed(route_j))
        else:
            continue

        # update
        new_id = ri
        routes[new_id] = {"customers": new_route,
                          "load": routes[ri]["load"] + routes[rj]["load"]}
        for c in new_route:
            cust_route[c] = new_id
        if rj in routes:
            del routes[rj]

    # optional: try to reduce routes if > K
    def best_merge():
        best = None
        route_ids = list(routes.keys())
        for idx_a in range(len(route_ids)):
            for idx_b in range(idx_a+1, len(route_ids)):
                ra = route_ids[idx_a]
                rb = route_ids[idx_b]
                if routes[ra]["load"] + routes[rb]["load"] > CAP:
                    continue
                A = routes[ra]["customers"]
                B = routes[rb]["customers"]
                # four orientations
                candidates = []

                a_start, a_end = A[0], A[-1]
                b_start, b_end = B[0], B[-1]

                # A end -> B start
                s1 = depot_dist[a_end] + depot_dist[b_start] - dist[a_end][b_start]
                candidates.append((s1, A + B))

                # A start -> B end
                s2 = depot_dist[a_start] + depot_dist[b_end] - dist[a_start][b_end]
                candidates.append((s2, B + A))

                # A start -> B start
                s3 = depot_dist[a_start] + depot_dist[b_start] - dist[a_start][b_start]
                candidates.append((s3, list(reversed(A)) + B))

                # A end -> B end
                s4 = depot_dist[a_end] + depot_dist[b_end] - dist[a_end][b_end]
                candidates.append((s4, A + list(reversed(B))))

                s_best, route_best = max(candidates, key=lambda x:x[0])
                if best is None or s_best > best[0]:
                    best = (s_best, ra, rb, route_best)
        return best

    while len(routes) > K:
        bm = best_merge()
        if bm is None:
            break
        _, ra, rb, merged = bm
        new_load = routes[ra]["load"] + routes[rb]["load"]
        routes[ra] = {"customers": merged, "load": new_load}
        if rb in routes:
            del routes[rb]

    # compute total distance
    total_dist = 0.0
    final_routes = list(routes.values())
    for r in final_routes:
        total_dist += route_distance(r["customers"], depot, coords, dist)

    # extract customer routes
    output_routes = [r["customers"] for r in final_routes]
    return total_dist, output_routes

def main():
    parsed = read_input()
    if parsed is None:
        return
    N, K, CAP, depot, coords, demands = parsed
    total_dist, routes = clarke_wright(N, K, CAP, depot, coords, demands)

    # Output
    print(f"{total_dist:.6f}")
    print(len(routes))
    for r in routes:
        print(" ".join(map(str, r)))

if __name__ == "__main__":
    main()
```

- Enforces **capacity per vehicle**.
