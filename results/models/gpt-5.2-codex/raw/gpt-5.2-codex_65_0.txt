```typescript
// Mission-control scheduling benchmark solution in TypeScript

declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type IdType = string | number;

interface TaskInput {
    id: IdType;
    duration: number;
    deps: IdType[];
    deadline: number;
    weight: number;
}

class BinaryHeap<T> {
    private arr: T[] = [];
    private cmp: (a: T, b: T) => number;
    constructor(cmp: (a: T, b: T) => number) {
        this.cmp = cmp;
    }
    size(): number { return this.arr.length; }
    peek(): T | undefined { return this.arr[0]; }
    push(x: T): void {
        const a = this.arr;
        a.push(x);
        this.bubbleUp(a.length - 1);
    }
    pop(): T | undefined {
        const a = this.arr;
        if (a.length === 0) return undefined;
        const top = a[0];
        const last = a.pop()!;
        if (a.length > 0) {
            a[0] = last;
            this.bubbleDown(0);
        }
        return top;
    }
    private bubbleUp(i: number): void {
        const a = this.arr;
        while (i > 0) {
            const p = (i - 1) >> 1;
            if (this.cmp(a[i], a[p]) < 0) {
                const tmp = a[i]; a[i] = a[p]; a[p] = tmp;
                i = p;
            } else break;
        }
    }
    private bubbleDown(i: number): void {
        const a = this.arr;
        const n = a.length;
        while (true) {
            let l = (i << 1) + 1;
            let r = l + 1;
            let best = i;
            if (l < n && this.cmp(a[l], a[best]) < 0) best = l;
            if (r < n && this.cmp(a[r], a[best]) < 0) best = r;
            if (best !== i) {
                const tmp = a[i]; a[i] = a[best]; a[best] = tmp;
                i = best;
            } else break;
        }
    }
}

const machineCount: number = data.machineCount || 1;
const taskInputs: TaskInput[] = data.tasks || [];
const n = taskInputs.length;

const idToIndex = new Map<string, number>();
const ids: IdType[] = new Array(n);
for (let i = 0; i < n; i++) {
    ids[i] = taskInputs[i].id;
    idToIndex.set(String(taskInputs[i].id), i);
}

const duration: number[] = new Array(n);
const deadline: number[] = new Array(n);
const weight: number[] = new Array(n);
const indeg: number[] = new Array(n).fill(0);
const adj: number[][] = new Array(n);
for (let i = 0; i < n; i++) adj[i] = [];

for (let i = 0; i < n; i++) {
    const t = taskInputs[i];
    duration[i] = t.duration || 0;
    deadline[i] = (t.deadline != null) ? t.deadline : Number.MAX_SAFE_INTEGER;
    weight[i] = (t.weight != null) ? t.weight : 1;
    const deps = t.deps || [];
    for (const d of deps) {
        const j = idToIndex.get(String(d));
        if (j !== undefined) {
            adj[j].push(i);
            indeg[i]++;
        }
    }
}

// Ready queue priority: earliest deadline, then highest weight, then shortest duration
interface ReadyTask { index: number; }
const readyHeap = new BinaryHeap<ReadyTask>((a, b) => {
    const i = a.index, j = b.index;
    if (deadline[i] !== deadline[j]) return deadline[i] - deadline[j];
    if (weight[i] !== weight[j]) return weight[j] - weight[i];
    if (duration[i] !== duration[j]) return duration[i] - duration[j];
    const idA = String(ids[i]), idB = String(ids[j]);
    if (idA < idB) return -1;
    if (idA > idB) return 1;
    return 0;
});

// Running tasks: min-heap by end time
interface RunningTask { end: number; index: number; machine: number; }
const runningHeap = new BinaryHeap<RunningTask>((a, b) => {
    if (a.end !== b.end) return a.end - b.end;
    return a.machine - b.machine;
});

// initialize ready tasks
for (let i = 0; i < n; i++) {
    if (indeg[i] === 0) readyHeap.push({ index: i });
}

let time = 0;
const freeMachines: number[] = [];
for (let m = 0; m < machineCount; m++) freeMachines.push(m);

const schedule: { id: IdType; start: number; machine: number }[] = [];
let scheduledCount = 0;

function processCompletion(taskIndex: number): void {
    for (const v of adj[taskIndex]) {
        indeg[v]--;
        if (indeg[v] === 0) {
            readyHeap.push({ index: v });
        }
    }
}

while (scheduledCount < n) {
    if (readyHeap.size() > 0 && freeMachines.length > 0) {
        while (readyHeap.size() > 0 && freeMachines.length > 0) {
            const r = readyHeap.pop()!;
            const i = r.index;
            const machine = freeMachines.pop()!;
            const start = time;
            const end = start + duration[i];
            schedule.push({ id: ids[i], start, machine });
            runningHeap.push({ end, index: i, machine });
            scheduledCount++;
        }
    } else {
        if (runningHeap.size() === 0) {
            // Cycle or no runnable tasks; break to avoid infinite loop
            break;
        }
        const first = runningHeap.pop()!;
        time = first.end;
        freeMachines.push(first.machine);
        processCompletion(first.index);
        while (runningHeap.size() > 0 && runningHeap.peek()!.end === time) {
            const r = runningHeap.pop()!;
            freeMachines.push(r.machine);
            processCompletion(r.index);
        }
    }
}

// Output schedule
const output = { schedule };
process.stdout.write(JSON.stringify(output));
```
