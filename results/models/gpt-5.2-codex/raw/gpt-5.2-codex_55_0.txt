Below is a complete, compilable C++20 chess bot that:

✅ Parses a FEN position  
✅ Searches for the best move using iterative deepening + alpha-beta  
✅ Handles legal move generation (castling, en passant, promotions)  
✅ Outputs moves in the required disambiguated algebraic notation  

---

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <array>
#include <chrono>
#include <algorithm>

using namespace std;

struct Move {
    int from, to;
    char piece;
    char capture;
    char promo;
    bool enpassant;
    bool castle;
    bool castleLong;
    int score;
    Move() : from(0), to(0), piece('.'), capture('.'), promo(0),
             enpassant(false), castle(false), castleLong(false), score(0) {}
};

struct Position {
    array<char, 64> board;
    bool white;
    bool wks, wqs, bks, bqs;
    int enpassant; // -1 if none
};

const int INF = 100000;
chrono::steady_clock::time_point deadline;
bool stopSearch = false;
Move bestMoveRoot;
int rootDepth = 0;

bool isWhitePiece(char c) { return c >= 'A' && c <= 'Z'; }
bool isBlackPiece(char c) { return c >= 'a' && c <= 'z'; }
bool isEmpty(char c) { return c == '.'; }

int pieceValue(char c) {
    switch (tolower(c)) {
        case 'p': return 100;
        case 'n': return 320;
        case 'b': return 330;
        case 'r': return 500;
        case 'q': return 900;
        case 'k': return 0;
    }
    return 0;
}

string squareToStr(int sq) {
    string s;
    s += char('a' + (sq % 8));
    s += char('1' + (sq / 8));
    return s;
}

bool timeUp() {
    return chrono::steady_clock::now() >= deadline;
}

bool squareAttacked(const Position& pos, int sq, bool byWhite) {
    int f = sq % 8, r = sq / 8;
    // Pawns
    if (byWhite) {
        if (f > 0 && sq - 9 >= 0 && pos.board[sq - 9] == 'P') return true;
        if (f < 7 && sq - 7 >= 0 && pos.board[sq - 7] == 'P') return true;
    } else {
        if (f > 0 && sq + 7 < 64 && pos.board[sq + 7] == 'p') return true;
        if (f < 7 && sq + 9 < 64 && pos.board[sq + 9] == 'p') return true;
    }
    // Knights
    const int knOffsets[8][2] = {{1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}};
    for (auto& o : knOffsets) {
        int nf = f + o[0], nr = r + o[1];
        if (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            char pc = pos.board[nr*8 + nf];
            if (byWhite && pc == 'N') return true;
            if (!byWhite && pc == 'n') return true;
        }
    }
    // Bishops / Queens (diagonals)
    const int diagDirs[4][2] = {{1,1},{-1,1},{1,-1},{-1,-1}};
    for (auto& d : diagDirs) {
        int nf = f + d[0], nr = r + d[1];
        while (nf>=0 && nf<8 && nr>=0 && nr<8) {
            char pc = pos.board[nr*8 + nf];
            if (!isEmpty(pc)) {
                if (byWhite && (pc=='B' || pc=='Q')) return true;
                if (!byWhite && (pc=='b' || pc=='q')) return true;
                break;
            }
            nf += d[0]; nr += d[1];
        }
    }
    // Rooks / Queens (orthogonal)
    const int orthoDirs[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for (auto& d : orthoDirs) {
        int nf = f + d[0], nr = r + d[1];
        while (nf>=0 && nf<8 && nr>=0 && nr<8) {
            char pc = pos.board[nr*8 + nf];
            if (!isEmpty(pc)) {
                if (byWhite && (pc=='R' || pc=='Q')) return true;
                if (!byWhite && (pc=='r' || pc=='q')) return true;
                break;
            }
            nf += d[0]; nr += d[1];
        }
    }
    // King
    for (int df=-1; df<=1; df++)
        for (int dr=-1; dr<=1; dr++)
            if (df!=0 || dr!=0) {
                int nf=f+df, nr=r+dr;
                if (nf>=0&&nf<8&&nr>=0&&nr<8) {
                    char pc = pos.board[nr*8+nf];
                    if (byWhite && pc=='K') return true;
                    if (!byWhite && pc=='k') return true;
                }
            }
    return false;
}

bool inCheck(const Position& pos, bool white) {
    for (int i=0;i<64;i++) {
        char pc = pos.board[i];
        if ((white && pc=='K') || (!white && pc=='k')) {
            return squareAttacked(pos, i, !white);
        }
    }
    return false;
}

Position makeMove(Position pos, const Move& m) {
    bool side = pos.white;
    char piece = pos.board[m.from];
    pos.board[m.from] = '.';
    pos.enpassant = -1;

    if (m.castle) {
        if (side) {
            if (m.to == 6) { // O-O
                pos.board[4]='.'; pos.board[6]='K';
                pos.board[7]='.'; pos.board[5]='R';
            } else {
                pos.board[4]='.'; pos.board[2]='K';
                pos.board[0]='.'; pos.board[3]='R';
            }
            pos.wks = pos.wqs = false;
        } else {
            if (m.to == 62) {
                pos.board[60]='.'; pos.board[62]='k';
                pos.board[63]='.'; pos.board[61]='r';
            } else {
                pos.board[60]='.'; pos.board[58]='k';
                pos.board[56]='.'; pos.board[59]='r';
            }
            pos.bks = pos.bqs = false;
        }
        pos.white = !side;
        return pos;
    }

    if (m.enpassant) {
        if (side) pos.board[m.to - 8] = '.';
        else pos.board[m.to + 8] = '.';
    }

    char placed = piece;
    if (m.promo) placed = side ? m.promo : (char)tolower(m.promo);

    pos.board[m.to] = placed;

    // update castling rights
    if (tolower(piece)=='k') {
        if (side) pos.wks=pos.wqs=false;
        else pos.bks=pos.bqs=false;
    }
    if (tolower(piece)=='r') {
        if (side) {
            if (m.from==0) pos.wqs=false;
            else if (m.from==7) pos.wks=false;
        } else {
            if (m.from==56) pos.bqs=false;
            else if (m.from==63) pos.bks=false;
        }
    }
    if (m.capture!='.' && !m.enpassant) {
        if (m.to==0) pos.wqs=false;
        else if (m.to==7) pos.wks=false;
        else if (m.to==56) pos.bqs=false;
        else if (m.to==63) pos.bks=false;
    }

    if (tolower(piece)=='p') {
        int fr = m.from/8, tr = m.to/8;
        if (side && fr==1 && tr==3) pos.enpassant = m.from + 8;
        if (!side && fr==6 && tr==4) pos.enpassant = m.from - 8;
    }

    pos.white = !side;
    return pos;
}

void addMove(vector<Move>& moves, int from, int to, char piece, char capture='.', char promo=0,
             bool enp=false, bool castle=false, bool castleLong=false) {
    Move m;
    m.from = from; m.to = to;
    m.piece = piece; m.capture = capture;
    m.promo = promo;
    m.enpassant = enp;
    m.castle = castle;
    m.castleLong = castleLong;
    moves.push_back(m);
}

vector<Move> generatePseudoMoves(const Position& pos) {
    vector<Move> moves;
    bool side = pos.white;
    for (int sq=0; sq<64; sq++) {
        char pc = pos.board[sq];
        if (pc=='.') continue;
        if (side && !isWhitePiece(pc)) continue;
        if (!side && !isBlackPiece(pc)) continue;

        int f = sq % 8, r = sq / 8;
        char p = tolower(pc);

        if (p=='p') {
            int dir = side ? 1 : -1;
            int startRank = side ? 1 : 6;
            int promoRank = side ? 6 : 1;
            int one = sq + dir*8;
            if (one>=0 && one<64 && isEmpty(pos.board[one])) {
                if (r==promoRank) {
                    for (char pr : {'Q','R','B','N'})
                        addMove(moves,sq,one,pc,'.',pr);
                } else addMove(moves,sq,one,pc);
                int two = sq + dir*16;
                if (r==startRank && isEmpty(pos.board[two]))
                    addMove(moves,sq,two,pc);
            }
            for (int df=-1; df<=1; df+=2) {
                int nf = f+df;
                int to = sq + dir*8 + df;
                if (nf<0||nf>7||to<0||to>=64) continue;
                char target = pos.board[to];
                if (!isEmpty(target) && (side?isBlackPiece(target):isWhitePiece(target))) {
                    if (r==promoRank) {
                        for (char pr : {'Q','R','B','N'})
                            addMove(moves,sq,to,pc,target,pr);
                    } else addMove(moves,sq,to,pc,target);
                }
                if (to == pos.enpassant)
                    addMove(moves,sq,to,pc, side?'p':'P',0,true);
            }
        } else if (p=='n') {
            const int knOffsets[8][2]={{1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}};
            for (auto& o: knOffsets) {
                int nf=f+o[0], nr=r+o[1];
                if(nf<0||nf>7||nr<0||nr>7) continue;
                int to=nr*8+nf;
                char target=pos.board[to];
                if(isEmpty(target) || (side?isBlackPiece(target):isWhitePiece(target)))
                    addMove(moves,sq,to,pc,target);
            }
        } else if (p=='b' || p=='r' || p=='q') {
            const int dirsB[4][2]={{1,1},{-1,1},{1,-1},{-1,-1}};
            const int dirsR[4][2]={{1,0},{-1,0},{0,1},{0,-1}};
            const int (*dirs)[2] = (p=='b')?dirsB:(p=='r')?dirsR:nullptr;
            int dirCount = (p=='q')?8:4;
            int allDirs[8][2];
            if (p=='q') {
                int idx=0;
                for(auto& d:dirsB) {allDirs[idx][0]=d[0];allDirs[idx++][1]=d[1];}
                for(auto& d:dirsR) {allDirs[idx][0]=d[0];allDirs[idx++][1]=d[1];}
                dirs = allDirs;
            }
            for (int d=0; d<dirCount; d++) {
                int df=dirs[d][0], dr=dirs[d][1];
                int nf=f+df,nr=r+dr;
                while(nf>=0&&nf<8&&nr>=0&&nr<8){
                    int to=nr*8+nf;
                    char target=pos.board[to];
                    if(isEmpty(target))
                        addMove(moves,sq,to,pc);
                    else {
                        if(side?isBlackPiece(target):isWhitePiece(target))
                            addMove(moves,sq,to,pc,target);
                        break;
                    }
                    nf+=df; nr+=dr;
                }
            }
        } else if (p=='k') {
            for(int df=-1;df<=1;df++)
                for(int dr=-1;dr<=1;dr++)
                    if(df||dr){
                        int nf=f+df,nr=r+dr;
                        if(nf<0||nf>7||nr<0||nr>7) continue;
                        int to=nr*8+nf;
                        char target=pos.board[to];
                        if(isEmpty(target) || (side?isBlackPiece(target):isWhitePiece(target)))
                            addMove(moves,sq,to,pc,target);
                    }
            // castling
            if (side) {
                if (pos.wks && pos.board[5]=='.' && pos.board[6]=='.' &&
                    !squareAttacked(pos,4,false) && !squareAttacked(pos,5,false) && !squareAttacked(pos,6,false))
                    addMove(moves,4,6,'K','.',0,false,true,false);
                if (pos.wqs && pos.board[3]=='.' && pos.board[2]=='.' && pos.board[1]=='.' &&
                    !squareAttacked(pos,4,false) && !squareAttacked(pos,3,false) && !squareAttacked(pos,2,false))
                    addMove(moves,4,2,'K','.',0,false,true,true);
            } else {
                if (pos.bks && pos.board[61]=='.' && pos.board[62]=='.' &&
                    !squareAttacked(pos,60,true) && !squareAttacked(pos,61,true) && !squareAttacked(pos,62,true))
                    addMove(moves,60,62,'k','.',0,false,true,false);
                if (pos.bqs && pos.board[59]=='.' && pos.board[58]=='.' && pos.board[57]=='.' &&
                    !squareAttacked(pos,60,true) && !squareAttacked(pos,59,true) && !squareAttacked(pos,58,true))
                    addMove(moves,60,58,'k','.',0,false,true,true);
            }
        }
    }
    return moves;
}

vector<Move> generateLegalMoves(const Position& pos) {
    vector<Move> pseudo = generatePseudoMoves(pos);
    vector<Move> legal;
    for (auto& m : pseudo) {
        Position np = makeMove(pos, m);
        if (!inCheck(np, !np.white)) legal.push_back(m);
    }
    return legal;
}

int evaluate(const Position& pos) {
    int score = 0;
    for (int i=0;i<64;i++) {
        char pc = pos.board[i];
        if (pc=='.') continue;
        int val = pieceValue(pc);
        score += isWhitePiece(pc) ? val : -val;
        // simple pawn advancement bonus
        if (tolower(pc)=='p') {
            int rank = i/8;
            if (isWhitePiece(pc)) score += rank*5;
            else score -= (7-rank)*5;
        }
    }
    return pos.white ? score : -score;
}

int negamax(Position pos, int depth, int alpha, int beta, int ply) {
    if (timeUp()) { stopSearch = true; return evaluate(pos); }
    if (depth==0) return evaluate(pos);

    vector<Move> moves = generateLegalMoves(pos);
    if (moves.empty()) {
        if (inCheck(pos, pos.white)) return -INF + ply;
        return 0;
    }

    // Order moves
    for (auto& m : moves) {
        int capVal = (m.capture!='.') ? pieceValue(m.capture) : 0;
        int promoVal = (m.promo) ? pieceValue(m.promo) : 0;
        m.score = capVal + promoVal;
    }
    sort(moves.begin(), moves.end(), [](const Move& a, const Move& b){return a.score>b.score;});

    int best = -INF;
    for (auto& m : moves) {
        Position np = makeMove(pos, m);
        int score = -negamax(np, depth-1, -beta, -alpha, ply+1);
        if (stopSearch) return evaluate(pos);
        if (score > best) best = score;
        if (score > alpha) {
            alpha = score;
            if (depth == rootDepth) bestMoveRoot = m;
        }
        if (alpha >= beta) break;
    }
    return best;
}

Move searchBestMove(const Position& pos, double timeout) {
    deadline = chrono::steady_clock::now() + chrono::milliseconds((int)(timeout*1000 - 50));
    stopSearch = false;
    Move best;
    vector<Move> legal = generateLegalMoves(pos);
    if (!legal.empty()) best = legal[0];
    for (int depth=1; depth<=6 && !stopSearch; depth++) {
        rootDepth = depth;
        negamax(pos, depth, -INF, INF, 0);
        if (!stopSearch) best = bestMoveRoot;
    }
    return best;
}

string moveToString(const Move& m, const Position& pos, bool check, bool mate) {
    if (m.castle) {
        string s = m.castleLong ? "O-O-O" : "O-O";
        if (mate) s += "#";
        else if (check) s += "+";
        return s;
    }
    string s;
    char piece = pos.board[m.from];
    if (tolower(piece)=='p') s += 'p';
    else s += (char)toupper(piece);
    s += squareToStr(m.from);
    if (m.capture!='.' || m.enpassant) s += "x";
    s += squareToStr(m.to);
    if (m.promo) {
        s += "/";
        s += (char)toupper(m.promo);
    }
    if (m.enpassant) s += "ep";
    if (mate) s += "#";
    else if (check) s += "+";
    return s;
}

Position parseFEN(const string& fen) {
    Position pos;
    pos.board.fill('.');
    pos.enpassant = -1;
    pos.wks=pos.wqs=pos.bks=pos.bqs=false;

    vector<string> parts;
    string tmp;
    for(char c:fen) {
        if (c==' ') { parts.push_back(tmp); tmp.clear(); }
        else tmp+=c;
    }
    parts.push_back(tmp);

    // Board
    int r=7,f=0;
    for(char c:parts[0]) {
        if (c=='/') { r--; f=0; continue; }
        if (isdigit(c)) { f += c-'0'; continue; }
        pos.board[r*8+f] = c;
        f++;
    }

    pos.white = (parts[1]=="w");

    string cast = parts[2];
    if (cast.find('K')!=string::npos) pos.wks=true;
    if (cast.find('Q')!=string::npos) pos.wqs=true;
    if (cast.find('k')!=string::npos) pos.bks=true;
    if (cast.find('q')!=string::npos) pos.bqs=true;

    if (parts[3]!="-") {
        int file = parts[3][0]-'a';
        int rank = parts[3][1]-'1';
        pos.enpassant = rank*8 + file;
    }

    return pos;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeout;
    string fen;
    if (!(cin>>timeout)) return 0;
    cin.ignore();
    getline(cin, fen);

    Position pos = parseFEN(fen);
    Move best = searchBestMove(pos, timeout);

    Position np = makeMove(pos, best);
    bool check = inCheck(np, np.white);
    bool mate = false;
    if (check) {
        auto oppMoves = generateLegalMoves(np);
        if (oppMoves.empty()) mate = true;
    }
    cout << moveToString(best, pos, check, mate) << "\n";
    return 0;
}
```
