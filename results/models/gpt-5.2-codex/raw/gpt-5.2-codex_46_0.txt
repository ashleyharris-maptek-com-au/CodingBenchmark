```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.
layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    // Load position and velocity for body i
    vec4 pi4 = inData[i * 2u];
    vec4 vi4 = inData[i * 2u + 1u];

    vec3 pos_i = pi4.xyz;
    float mass_i = pi4.w; // not used directly in acceleration sum
    vec3 vel_i = vi4.xyz;

    float dt = physics.x;
    float soft = physics.y;
    float G = physics.z;

    vec3 acc = vec3(0.0);

    // Compute acceleration from all other bodies
    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue;

        vec4 pj4 = inData[j * 2u];
        vec3 pos_j = pj4.xyz;
        float mass_j = pj4.w;

        vec3 r = pos_j - pos_i;
        float dist2 = dot(r, r) + soft * soft;
        float invDist = inversesqrt(dist2);
        float invDist3 = invDist * invDist * invDist;

        acc += G * mass_j * r * invDist3;
    }

    // Integrate (semi-implicit Euler / leapfrog-style step)
    vel_i += acc * dt;
    pos_i += vel_i * dt;

    // Write updated position+mass and velocity
    outData[i * 2u]     = vec4(pos_i, mass_i);
    outData[i * 2u + 1u]= vec4(vel_i, 0.0);
}
```
