```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    uint idx = i * 2u;
    vec3 pos = inData[idx].xyz;
    vec3 vel = inData[idx + 1u].xyz;

    float dt         = radii.x;
    float sepRadius  = radii.y;
    float aliRadius  = radii.z;
    float cohRadius  = radii.w;

    float sepWeight  = weights.x;
    float aliWeight  = weights.y;
    float cohWeight  = weights.z;
    float maxSpeed   = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    vec3 sepAcc = vec3(0.0);
    vec3 aliAcc = vec3(0.0);
    vec3 cohAcc = vec3(0.0);

    uint sepCount = 0u;
    uint aliCount = 0u;
    uint cohCount = 0u;

    // Naive O(N^2) neighbor scan
    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue;
        uint jdx = j * 2u;
        vec3 posJ = inData[jdx].xyz;
        vec3 velJ = inData[jdx + 1u].xyz;

        vec3 diff = posJ - pos;
        float dist = length(diff);
        if (dist <= 0.00001) continue;

        if (dist < sepRadius) {
            // Push away from neighbors
            sepAcc -= diff / dist;
            sepCount++;
        }
        if (dist < aliRadius) {
            aliAcc += velJ;
            aliCount++;
        }
        if (dist < cohRadius) {
            cohAcc += posJ;
            cohCount++;
        }
    }

    // Average and weight
    if (sepCount > 0u) {
        sepAcc = (sepAcc / float(sepCount)) * sepWeight;
    } else {
        sepAcc = vec3(0.0);
    }

    if (aliCount > 0u) {
        vec3 avgVel = aliAcc / float(aliCount);
        aliAcc = (avgVel - vel) * aliWeight;
    } else {
        aliAcc = vec3(0.0);
    }

    if (cohCount > 0u) {
        vec3 avgPos = cohAcc / float(cohCount);
        cohAcc = (avgPos - pos) * cohWeight;
    } else {
        cohAcc = vec3(0.0);
    }

    vec3 accel = sepAcc + aliAcc + cohAcc;

    // Boundary force: push back toward center if outside boundSize*0.4
    float limit = boundSize * 0.4;
    float scale = boundSize * 0.5;

    if (abs(pos.x) > limit) accel.x += (-pos.x / scale) * boundForce;
    if (abs(pos.y) > limit) accel.y += (-pos.y / scale) * boundForce;
    if (abs(pos.z) > limit) accel.z += (-pos.z / scale) * boundForce;

    // Integrate
    vec3 newVel = vel + accel * dt;

    // Clamp speed
    float speed = length(newVel);
    if (speed > maxSpeed && speed > 0.0) {
        newVel = newVel / speed * maxSpeed;
    }

    vec3 newPos = pos + newVel * dt;

    outData[idx]     = vec4(newPos, 0.0);
    outData[idx + 1u]= vec4(newVel, 0.0);
}
```
