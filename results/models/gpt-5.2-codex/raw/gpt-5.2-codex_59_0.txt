const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const prob = data.probability;
const drones = data.drones || [];
const maxStepsPerDrone = data.maxStepsPerDrone || 0;

const height = prob.length;
const width = height > 0 ? prob[0].length : 0;

function valid(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height && prob[y][x] !== -1;
}

// visited cells with positive probability
const visited = Array.from({ length: height }, () => Array(width).fill(false));

const paths = [];

for (let d = 0; d < drones.length; d++) {
  const drone = drones[d];
  let x = drone.x, y = drone.y;
  const path = [];
  // Add start
  path.push([x, y]);
  if (prob[y][x] > 0 && !visited[y][x]) visited[y][x] = true;

  for (let step = 0; step < maxStepsPerDrone; step++) {
    const moves = [
      [0, 0],
      [1, 0],
      [-1, 0],
      [0, 1],
      [0, -1]
    ];
    let best = null;
    for (const [dx, dy] of moves) {
      const nx = x + dx, ny = y + dy;
      if (!valid(nx, ny)) continue;
      const val = prob[ny][nx];
      const unvisited = val > 0 && !visited[ny][nx];
      const s1 = unvisited ? 1 : 0;
      if (
        best === null ||
        s1 > best.s1 ||
        (s1 === best.s1 && val > best.val)
      ) {
        best = { nx, ny, s1, val };
      }
    }
    if (best === null) {
      // no valid moves (shouldn't happen), stay
      best = { nx: x, ny: y, s1: 0, val: prob[y][x] };
    }
    x = best.nx;
    y = best.ny;
    path.push([x, y]);
    if (prob[y][x] > 0 && !visited[y][x]) visited[y][x] = true;
  }

  paths.push(path);
}

const output = { paths };
process.stdout.write(JSON.stringify(output));
