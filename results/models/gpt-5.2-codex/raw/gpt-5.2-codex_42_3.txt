OpCapability Shader
OpCapability Geometry
%ext = OpExtInstImport "GLSL450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%f_thick = OpConstant %float 0.03

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%label = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy

%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_base = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_norm = OpExtInst %v2float %ext Normalize %perp_base
%perp = OpVectorTimesScalar %v2float %perp_norm %f_thick

%p0_plus = OpFAdd %v2float %p0_xy %perp
%p0_minus = OpFSub %v2float %p0_xy %perp
%p1_plus = OpFAdd %v2float %p1_xy %perp
%p1_minus = OpFSub %v2float %p1_xy %perp

%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

%p0p_x = OpCompositeExtract %float %p0_plus 0
%p0p_y = OpCompositeExtract %float %p0_plus 1
%pos0p = OpCompositeConstruct %v4float %p0p_x %p0p_y %p0_z %p0_w

%p0m_x = OpCompositeExtract %float %p0_minus 0
%p0m_y = OpCompositeExtract %float %p0_minus 1
%pos0m = OpCompositeConstruct %v4float %p0m_x %p0m_y %p0_z %p0_w

%p1p_x = OpCompositeExtract %float %p1_plus 0
%p1p_y = OpCompositeExtract %float %p1_plus 1
%pos1p = OpCompositeConstruct %v4float %p1p_x %p1p_y %p1_z %p1_w

%p1m_x = OpCompositeExtract %float %p1_minus 0
%p1m_y = OpCompositeExtract %float %p1_minus 1
%pos1m = OpCompositeConstruct %v4float %p1m_x %p1m_y %p1_z %p1_w

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %out_pos_ptr %pos0p
OpStore %outColor %c0
OpEmitVertex

OpStore %out_pos_ptr %pos0m
OpStore %outColor %c0
OpEmitVertex

OpStore %out_pos_ptr %pos1p
OpStore %outColor %c1
OpEmitVertex

OpStore %out_pos_ptr %pos1m
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
