```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <random>
#include <cstdio>
#include <cstring>
#include <string>

// Fast input
class FastScanner {
    static const int BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    int idx, size;
public:
    FastScanner() : idx(0), size(0) {}
    inline char read() {
        if (idx >= size) {
            size = (int)fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }
    template <typename T>
    bool nextInt(T &out) {
        char c;
        T sign = 1;
        T val = 0;
        c = read();
        if (!c) return false;
        while (c != '-' && (c < '0' || c > '9')) {
            c = read();
            if (!c) return false;
        }
        if (c == '-') { sign = -1; c = read(); }
        for (; c >= '0' && c <= '9'; c = read()) {
            val = val * 10 + (c - '0');
        }
        out = val * sign;
        return true;
    }
};

// Append integer to string (fast output)
static void appendInt(std::string &s, int x) {
    if (x == 0) { s.push_back('0'); return; }
    if (x < 0) { s.push_back('-'); x = -x; }
    char buf[20];
    int len = 0;
    while (x > 0) { buf[len++] = char('0' + (x % 10)); x /= 10; }
    for (int i = len - 1; i >= 0; --i) s.push_back(buf[i]);
}

// Graph coloring solver
class GraphColorer {
public:
    int n, m, k;
    std::vector<std::vector<int>> adj;
    std::vector<int> deg;
    std::vector<int> color;

    GraphColorer(int n_, int m_, int k_) : n(n_), m(m_), k(k_) {
        adj.assign(n, {});
        deg.assign(n, 0);
        color.assign(n, -1);
    }

    // Exact DSATUR backtracking for small graphs
    bool dsaturBacktrack(int colored) {
        if (colored == n) return true;

        int best = -1;
        int bestSat = -1, bestDeg = -1;

        std::vector<int> mark(k, -1);
        int markId = 0;

        for (int v = 0; v < n; ++v) {
            if (color[v] != -1) continue;
            int sat = 0;
            markId++;
            for (int u : adj[v]) {
                int c = color[u];
                if (c != -1 && mark[c] != markId) {
                    mark[c] = markId;
                    sat++;
                }
            }
            if (sat > bestSat || (sat == bestSat && deg[v] > bestDeg)) {
                best = v;
                bestSat = sat;
                bestDeg = deg[v];
            }
        }

        if (best == -1) return false;

        std::vector<char> used(k, 0);
        for (int u : adj[best]) {
            int c = color[u];
            if (c != -1) used[c] = 1;
        }

        for (int c = 0; c < k; ++c) {
            if (!used[c]) {
                color[best] = c;
                if (dsaturBacktrack(colored + 1)) return true;
                color[best] = -1;
            }
        }
        return false;
    }

    // DSATUR greedy (no backtracking)
    bool dsaturGreedy() {
        std::fill(color.begin(), color.end(), -1);

        for (int step = 0; step < n; ++step) {
            int best = -1;
            int bestSat = -1, bestDeg = -1;

            std::vector<int> mark(k, -1);
            int markId = 0;

            for (int v = 0; v < n; ++v) {
                if (color[v] != -1) continue;
                int sat = 0;
                markId++;
                for (int u : adj[v]) {
                    int c = color[u];
                    if (c != -1 && mark[c] != markId) {
                        mark[c] = markId;
                        sat++;
                    }
                }
                if (sat > bestSat || (sat == bestSat && deg[v] > bestDeg)) {
                    best = v;
                    bestSat = sat;
                    bestDeg = deg[v];
                }
            }

            if (best == -1) return false;

            std::vector<char> used(k, 0);
            for (int u : adj[best]) {
                int c = color[u];
                if (c != -1) used[c] = 1;
            }

            int chosen = -1;
            for (int c = 0; c < k; ++c) {
                if (!used[c]) { chosen = c; break; }
            }
            if (chosen == -1) return false;
            color[best] = chosen;
        }
        return true;
    }

    // Greedy with fixed order
    bool greedyOrder(const std::vector<int> &order) {
        std::fill(color.begin(), color.end(), -1);
        std::vector<int> mark(k, -1);

        for (int idx = 0; idx < n; ++idx) {
            int v = order[idx];
            for (int u : adj[v]) {
                int c = color[u];
                if (c != -1) mark[c] = idx;
            }
            int chosen = -1;
            for (int c = 0; c < k; ++c) {
                if (mark[c] != idx) { chosen = c; break; }
            }
            if (chosen == -1) return false;
            color[v] = chosen;
        }
        return true;
    }

    // Multi-attempt greedy with randomized ordering
    bool greedyMulti(int attempts) {
        std::vector<int> order(n);
        for (int i = 0; i < n; ++i) order[i] = i;

        std::mt19937 rng(1234567);

        // Attempt 0: deterministic by degree
        std::sort(order.begin(), order.end(),
                  [&](int a, int b) { return deg[a] > deg[b]; });
        if (greedyOrder(order)) return true;

        // Randomized attempts
        for (int t = 1; t < attempts; ++t) {
            std::shuffle(order.begin(), order.end(), rng);
            std::stable_sort(order.begin(), order.end(),
                             [&](int a, int b) { return deg[a] > deg[b]; });
            if (greedyOrder(order)) return true;
        }
        return false;
    }
};

int main() {
    FastScanner fs;
    int n, m, k;
    if (!fs.nextInt(n)) return 0;
    fs.nextInt(m);
    fs.nextInt(k);

    GraphColorer solver(n, m, k);

    std::vector<std::pair<int,int>> edges;
    edges.reserve(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        fs.nextInt(u);
        fs.nextInt(v);
        if (u == v) continue;
        edges.push_back({u, v});
        solver.deg[u]++;
        solver.deg[v]++;
    }

    solver.adj.assign(n, {});
    for (int i = 0; i < n; ++i) solver.adj[i].reserve(solver.deg[i]);

    for (auto &e : edges) {
        solver.adj[e.first].push_back(e.second);
        solver.adj[e.second].push_back(e.first);
    }

    bool success = false;

    if (n <= 200) {
        success = solver.dsaturBacktrack(0);
    } else if (n <= 5000) {
        success = solver.dsaturGreedy();
        if (!success) success = solver.greedyMulti(5);
    } else {
        int attempts = (n < 50000 ? 8 : 3);
        success = solver.greedyMulti(attempts);
    }

    // Output (even if unsuccessful, still prints whatever is in color[])
    std::string out;
    out.reserve(n * 3);
    for (int i = 0; i < n; ++i) {
        if (i) out.push_back(' ');
        appendInt(out, solver.color[i] == -1 ? 0 : solver.color[i]);
    }
    std::cout << out;
    return 0;
}
```

**Notes:**
- Uses adaptive strategy:
  - **Exact DSATUR backtracking** for very small graphs
  - **DSATUR greedy** for medium graphs
  - **Greedy with random restarts** for large graphs  
- Fully standard C++17 (no non-standard headers).
- Designed to scale up to million-node graphs efficiently.
