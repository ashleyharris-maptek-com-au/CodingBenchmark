const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (input.length === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}
const data = JSON.parse(input);
const depot = data.depot || [0, 0];
const truckCount = Math.max(1, data.truckCount || 1);
const packages = Array.isArray(data.packages) ? data.packages.slice() : [];

const n = packages.length;
if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

// compute angles
for (const p of packages) {
  p._angle = Math.atan2(p.y - depot[1], p.x - depot[0]);
}
packages.sort((a, b) => a._angle - b._angle);

const numRoutes = Math.min(truckCount, n);
const chunkSize = Math.ceil(n / numRoutes);
const routes = [];

for (let i = 0; i < numRoutes; i++) {
  const start = i * chunkSize;
  const end = Math.min(n, start + chunkSize);
  if (start >= n) break;
  const chunk = packages.slice(start, end);
  // keep angle order
  routes.push(chunk.map(p => p.id));
}

process.stdout.write(JSON.stringify({ routes }));
