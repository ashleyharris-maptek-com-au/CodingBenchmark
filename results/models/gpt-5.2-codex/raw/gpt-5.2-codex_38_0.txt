```python
import sys

def main():
    data = sys.stdin.read().splitlines()
    if not data:
        return
    first = data[0].split()
    if len(first) < 2:
        return
    U = int(first[0])
    S = int(first[1])

    set_masks = []
    element_to_sets = [[] for _ in range(U)]

    for i in range(S):
        line = data[i+1] if i+1 < len(data) else ""
        elems = set()
        for tok in line.split():
            try:
                e = int(tok)
            except:
                continue
            if 0 <= e < U:
                elems.add(e)
        mask = 0
        for e in elems:
            mask |= 1 << e
        set_masks.append(mask)
        for e in elems:
            element_to_sets[e].append(i)

    universe_mask = (1 << U) - 1 if U > 0 else 0

    # If any element is not present in any set, no solution
    for e in range(U):
        if not element_to_sets[e]:
            print("NO SOLUTION")
            return

    sys.setrecursionlimit(10000)
    memo = set()

    def search(covered_mask):
        if covered_mask == universe_mask:
            return []
        if covered_mask in memo:
            return None

        best_candidates = None
        best_count = 10**9

        for e in range(U):
            if not (covered_mask >> e) & 1:
                candidates = []
                for idx in element_to_sets[e]:
                    if set_masks[idx] & covered_mask == 0:
                        candidates.append(idx)
                c = len(candidates)
                if c < best_count:
                    best_count = c
                    best_candidates = candidates
                    if c == 0:
                        break

        if best_candidates is None or best_count == 0:
            memo.add(covered_mask)
            return None

        best_candidates.sort()
        for idx in best_candidates:
            res = search(covered_mask | set_masks[idx])
            if res is not None:
                return [idx] + res

        memo.add(covered_mask)
        return None

    result = search(0)

    if result is None:
        print("NO SOLUTION")
    else:
        result_sorted = sorted(result)
        print("SOLUTION")
        if result_sorted:
            print(" ".join(map(str, result_sorted)))
        else:
            print()

if __name__ == "__main__":
    main()
```
