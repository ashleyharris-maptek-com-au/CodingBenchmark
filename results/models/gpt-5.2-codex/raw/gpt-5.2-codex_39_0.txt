Below is a complete, correct Rust solution that **always produces a minimum‑cut bisection**.  
Because minimum bisection is NP‑hard, the graph size is assumed small (as is typical for this problem).  
The code exhaustively enumerates all subsets of size `n/2` (with symmetry reduction by fixing vertex `0` in the first part) and keeps the minimum cut.

```rust
use std::io::{self, Read};

fn dfs(
    idx: usize,
    chosen: usize,
    mask: u64,
    cut: i32,
    n: usize,
    k: usize,
    adj: &[u64],
    deg: &[i32],
    best_cut: &mut i32,
    best_mask: &mut u64,
) {
    if chosen == k {
        if cut < *best_cut {
            *best_cut = cut;
            *best_mask = mask;
        }
        return;
    }
    if idx == n {
        return;
    }
    let remaining = n - idx;
    if chosen + remaining < k {
        return; // not enough vertices left
    }

    // Option 1: include idx
    if chosen < k {
        let edges_to_s = (adj[idx] & mask).count_ones() as i32;
        let new_cut = cut + deg[idx] - 2 * edges_to_s;
        dfs(
            idx + 1,
            chosen + 1,
            mask | (1u64 << idx),
            new_cut,
            n,
            k,
            adj,
            deg,
            best_cut,
            best_mask,
        );
    }

    // Option 2: exclude idx
    if chosen + remaining - 1 >= k {
        dfs(
            idx + 1,
            chosen,
            mask,
            cut,
            n,
            k,
            adj,
            deg,
            best_cut,
            best_mask,
        );
    }
}

fn main() {
    // Read input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n_opt = it.next();
    if n_opt.is_none() {
        return;
    }
    let n: usize = n_opt.unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    // Build adjacency bitmasks
    let mut adj = vec![0u64; n];
    let mut deg = vec![0i32; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        let bitu = 1u64 << v;
        let bitv = 1u64 << u;
        if (adj[u] & bitu) == 0 {
            adj[u] |= bitu;
            adj[v] |= bitv;
            deg[u] += 1;
            deg[v] += 1;
        }
    }

    let k = n / 2;
    if n == 0 {
        return;
    }

    // Fix vertex 0 in partition A (symmetry reduction)
    let start_mask: u64 = 1;
    let start_cut: i32 = deg[0];

    let mut best_cut = i32::MAX;
    let mut best_mask: u64 = 0;

    dfs(
        1,
        1,
        start_mask,
        start_cut,
        n,
        k,
        &adj,
        &deg,
        &mut best_cut,
        &mut best_mask,
    );

    // Recompute cut for safety
    let full_mask = if n == 64 {
        u64::MAX
    } else {
        (1u64 << n) - 1
    };
    let comp = full_mask ^ best_mask;
    let mut actual_cut = 0i32;
    for v in 0..n {
        if (best_mask >> v) & 1 == 1 {
            actual_cut += (adj[v] & comp).count_ones() as i32;
        }
    }

    // Output
    let mut out = String::new();
    out.push_str(&format!("{}\n", actual_cut));
    let mut first = true;
    for v in 0..n {
        if (best_mask >> v) & 1 == 1 {
            if !first {
                out.push(' ');
            }
            first = false;
            out.push_str(&v.to_string());
        }
    }
    out.push('\n');
    print!("{}", out);
}
```

### Notes
- The algorithm is exact and always outputs a **minimum** bisection.
- The symmetry fix (`vertex 0` always in A) halves the search space.
- Works efficiently for typical contest constraints on this NP‑hard task.
