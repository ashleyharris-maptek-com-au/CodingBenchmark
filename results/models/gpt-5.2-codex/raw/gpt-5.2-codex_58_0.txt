const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  process.stdout.write(JSON.stringify({ assignments: [] }));
  return;
}
const data = JSON.parse(input);

const grid = data.grid || [];
const H = grid.length;
const W = H > 0 ? grid[0].length : 0;
const nCells = W * H;

const passable = new Uint8Array(nCells);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    if (row[x] !== '#') passable[y * W + x] = 1;
  }
}

function cellIndex(x, y) {
  return y * W + x;
}

function bfs(startIdx) {
  const dist = new Int32Array(nCells);
  dist.fill(-1);
  const q = new Int32Array(nCells);
  let head = 0, tail = 0;
  dist[startIdx] = 0;
  q[tail++] = startIdx;

  while (head < tail) {
    const idx = q[head++];
    const d = dist[idx] + 1;
    const x = idx % W;
    const y = (idx / W) | 0;
    let n;
    if (x > 0) {
      n = idx - 1;
      if (passable[n] && dist[n] === -1) {
        dist[n] = d; q[tail++] = n;
      }
    }
    if (x + 1 < W) {
      n = idx + 1;
      if (passable[n] && dist[n] === -1) {
        dist[n] = d; q[tail++] = n;
      }
    }
    if (y > 0) {
      n = idx - W;
      if (passable[n] && dist[n] === -1) {
        dist[n] = d; q[tail++] = n;
      }
    }
    if (y + 1 < H) {
      n = idx + W;
      if (passable[n] && dist[n] === -1) {
        dist[n] = d; q[tail++] = n;
      }
    }
  }
  return dist;
}

const robots = data.robots || [];
const jobs = data.jobs || [];

let maxRobotId = -1;
const robotIds = [];
const robotPosIdx = [];
const robotDist = [];

for (const r of robots) {
  if (r.id > maxRobotId) maxRobotId = r.id;
  robotIds.push(r.id);
  const idx = cellIndex(r.x, r.y);
  robotPosIdx.push(idx);
  robotDist.push(bfs(idx));
}

const assignments = Array.from({ length: maxRobotId + 1 }, () => []);

const jobsSorted = jobs.slice().sort((a, b) => {
  const p = b.priority - a.priority;
  if (p !== 0) return p;
  return a.id - b.id;
});

const INF = 1e9;

for (const job of jobsSorted) {
  const jobIdx = cellIndex(job.x, job.y);
  let bestR = 0;
  let bestDist = INF;
  for (let r = 0; r < robotIds.length; r++) {
    let d = robotDist[r][jobIdx];
    if (d < 0) d = INF;
    if (d < bestDist || (d === bestDist && robotIds[r] < robotIds[bestR])) {
      bestDist = d;
      bestR = r;
    }
  }
  const rid = robotIds[bestR];
  assignments[rid].push(job.id);
  robotPosIdx[bestR] = jobIdx;
  robotDist[bestR] = bfs(jobIdx);
}

process.stdout.write(JSON.stringify({ assignments }));
