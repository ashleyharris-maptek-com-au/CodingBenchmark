; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%GLSL450 = OpExtInstImport "GLSL.std.450"
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%functype = OpTypeFunction %void

%c0 = OpConstant %float 0.0
%c1 = OpConstant %float 1.0
%c2 = OpConstant %float 2.0
%c8 = OpConstant %float 8.0
%c16 = OpConstant %float 16.0
%c0_5 = OpConstant %float 0.5
%c0_05 = OpConstant %float 0.05
%c0_95 = OpConstant %float 0.95
%c0_1 = OpConstant %float 0.1
%c0_9 = OpConstant %float 0.9
%c0_7 = OpConstant %float 0.7
%c0_2 = OpConstant %float 0.2
%c0_4 = OpConstant %float 0.4
%c0_6 = OpConstant %float 0.6
%scale = OpConstantComposite %v2float %c8 %c16
%brickColor = OpConstantComposite %v3float %c0_7 %c0_2 %c0_1
%mortarColor = OpConstantComposite %v3float %c0_7 %c0_7 %c0_7

%c0i = OpConstant %int 0
%c3i = OpConstant %int 3

%inWorldPos = OpVariable %_ptr_Input_v3float Input
%inNormal = OpVariable %_ptr_Input_v3float Input
%inUV = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %functype
%entry = OpLabel
%uv = OpLoad %v2float %inUV
%uvScaled = OpFMul %v2float %uv %scale
%y = OpCompositeExtract %float %uvScaled 1
%row = OpExtInst %float %GLSL450 Floor %y
%mod = OpFMod %float %row %c2
%stepRow = OpExtInst %float %GLSL450 Step %c0_5 %mod
%offset = OpFMul %float %stepRow %c0_5
%x = OpCompositeExtract %float %uvScaled 0
%xoff = OpFAdd %float %x %offset
%uvRow = OpCompositeConstruct %v2float %xoff %y
%frac = OpExtInst %v2float %GLSL450 Fract %uvRow
%fracx = OpCompositeExtract %float %frac 0
%fracy = OpCompositeExtract %float %frac 1
%stepx1 = OpExtInst %float %GLSL450 Step %c0_05 %fracx
%invx1 = OpFSub %float %c1 %stepx1
%stepx2 = OpExtInst %float %GLSL450 Step %c0_95 %fracx
%mortarX = OpFAdd %float %invx1 %stepx2
%stepy1 = OpExtInst %float %GLSL450 Step %c0_1 %fracy
%invy1 = OpFSub %float %c1 %stepy1
%stepy2 = OpExtInst %float %GLSL450 Step %c0_9 %fracy
%mortarY = OpFAdd %float %invy1 %stepy2
%mortarMask = OpExtInst %float %GLSL450 Max %mortarX %mortarY
%mortarMask3 = OpCompositeConstruct %v3float %mortarMask %mortarMask %mortarMask
%baseColor = OpExtInst %v3float %GLSL450 Mix %brickColor %mortarColor %mortarMask3
%norm = OpLoad %v3float %inNormal
%normN = OpExtInst %v3float %GLSL450 Normalize %norm
%wpos = OpLoad %v3float %inWorldPos
%lightPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %c3i
%light4 = OpLoad %v4float %lightPtr
%light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%lightDir = OpFSub %v3float %light3 %wpos
%lightDirN = OpExtInst %v3float %GLSL450 Normalize %lightDir
%dot = OpDot %float %normN %lightDirN
%diff = OpExtInst %float %GLSL450 Max %dot %c0
%mul = OpFMul %float %c0_6 %diff
%factor = OpFAdd %float %c0_4 %mul
%factor3 = OpCompositeConstruct %v3float %factor %factor %factor
%colorLit = OpFMul %v3float %baseColor %factor3
%r = OpCompositeExtract %float %colorLit 0
%g = OpCompositeExtract %float %colorLit 1
%b = OpCompositeExtract %float %colorLit 2
%outv = OpCompositeConstruct %v4float %r %g %b %c1
OpStore %outColor %outv
OpReturn
OpFunctionEnd
