// 3D Packing: pack regular tetrahedra into a (convex) polyhedral container.
//
// Strategy (provably overlap-free, orientation-convention independent):
//   - A regular tetrahedron of edge e is always contained in its circumsphere
//     of radius R = e*sqrt(3/8). Hence if we place tetra CENTERS on a lattice
//     where any two centers are at least 2R apart, the circumspheres are
//     disjoint => the tetrahedra are disjoint, no matter what base orientation
//     / Euler convention the grader uses to reconstruct them.
//   - We therefore output every tetra with Euler angles (0,0,0) and only choose
//     centers. This makes correctness independent of the unknown base-vertex
//     convention used by the verifier.
//   - We use an FCC lattice (nearest-neighbour distance = 2R) because it is the
//     densest lattice sphere packing (~0.74), maximizing the count vs a simple
//     cubic grid.
//   - Containment: a center is accepted iff its circumsphere (center, R) lies
//     fully inside the container's convex hull (distance to every face >= R).
//
// Scalability: we iterate the FCC lattice over the container's bounding box,
// guarded by a wall-clock budget and a hard cap on the number of placements so
// that "ludicrous" containers still finish well within 30 s.

#include <cstdio>
#include <cstdint>
#include <cmath>
#include <vector>
#include <array>
#include <string>
#include <chrono>
#include <algorithm>
#include <unordered_set>

using std::vector;

struct V3 { double x, y, z; };

static inline V3 sub(const V3& a, const V3& b) { return {a.x-b.x, a.y-b.y, a.z-b.z}; }
static inline V3 cross(const V3& a, const V3& b) {
    return { a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x };
}
static inline double dot(const V3& a, const V3& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }

struct Plane { double nx, ny, nz, d; }; // outward unit normal; inside: n.p <= d

int main() {
    using clock_t2 = std::chrono::steady_clock;
    auto t_start = clock_t2::now();
    const double TIME_BUDGET = 20.0;        // seconds of compute we allow
    const long long MAX_K = 1500000;        // cap on number of tetrahedra

    int V;
    double e;
    if (scanf("%d %lf", &V, &e) != 2) return 0;
    if (V <= 0 || !(e > 0.0)) { printf("0\n"); return 0; }

    vector<V3> verts(V);
    for (int i = 0; i < V; ++i) {
        if (scanf("%lf %lf %lf", &verts[i].x, &verts[i].y, &verts[i].z) != 3) {
            verts.resize(i);
            V = i;
            break;
        }
    }
    if (V < 4) { printf("0\n"); return 0; }

    const double R = e * std::sqrt(3.0 / 8.0); // circumradius

    // Bounding box.
    V3 lo = verts[0], hi = verts[0];
    V3 centroid{0,0,0};
    for (const auto& v : verts) {
        lo.x = std::min(lo.x, v.x); lo.y = std::min(lo.y, v.y); lo.z = std::min(lo.z, v.z);
        hi.x = std::max(hi.x, v.x); hi.y = std::max(hi.y, v.y); hi.z = std::max(hi.z, v.z);
        centroid.x += v.x; centroid.y += v.y; centroid.z += v.z;
    }
    centroid.x /= V; centroid.y /= V; centroid.z /= V;

    // Build container half-spaces.
    vector<Plane> planes;
    bool useHull = (V <= 60); // brute-force hull is O(V^4); fall back for large V

    if (useHull) {
        const double EPS = 1e-9;
        std::unordered_set<uint64_t> seen;
        auto keyOf = [](const Plane& p) -> uint64_t {
            // Quantize normal+offset to dedupe coplanar faces.
            auto q = [](double v) -> int64_t {
                return (int64_t)std::llround(v * 1e6);
            };
            uint64_t h = 1469598103934665603ull;
            int64_t vals[4] = { q(p.nx), q(p.ny), q(p.nz), q(p.d) };
            for (int i = 0; i < 4; ++i) {
                uint64_t x = (uint64_t)vals[i];
                h ^= x; h *= 1099511628211ull;
            }
            return h;
        };

        for (int i = 0; i < V; ++i)
        for (int j = i + 1; j < V; ++j)
        for (int k = j + 1; k < V; ++k) {
            V3 a = verts[i], b = verts[j], c = verts[k];
            V3 n = cross(sub(b, a), sub(c, a));
            double len = std::sqrt(dot(n, n));
            if (len < 1e-12) continue; // collinear
            n.x /= len; n.y /= len; n.z /= len;
            // Is this a hull face? All other points on one side.
            bool pos = false, neg = false;
            for (int m = 0; m < V; ++m) {
                if (m == i || m == j || m == k) continue;
                double s = dot(n, sub(verts[m], a));
                if (s > EPS) pos = true;
                else if (s < -EPS) neg = true;
                if (pos && neg) break;
            }
            if (pos && neg) continue; // not a hull facet
            // Orient outward (away from centroid).
            if (dot(n, sub(centroid, a)) > 0) { n.x = -n.x; n.y = -n.y; n.z = -n.z; }
            Plane pl{ n.x, n.y, n.z, dot(n, a) };
            uint64_t key = keyOf(pl);
            if (seen.insert(key).second) planes.push_back(pl);
        }
        if (planes.empty()) useHull = false; // degenerate -> fall back
    }

    if (!useHull) {
        // Axis-aligned bounding-box half-spaces.
        planes.push_back({  1, 0, 0,  hi.x });
        planes.push_back({ -1, 0, 0, -lo.x });
        planes.push_back({  0, 1, 0,  hi.y });
        planes.push_back({  0,-1, 0, -lo.y });
        planes.push_back({  0, 0, 1,  hi.z });
        planes.push_back({  0, 0,-1, -lo.z });
    }

    // A center p is acceptable iff distance to every face >= R, i.e.
    // n.p <= d - R for every outward unit plane.
    auto inside = [&](const V3& p) -> bool {
        for (const auto& pl : planes) {
            double s = pl.nx * p.x + pl.ny * p.y + pl.nz * p.z;
            if (s > pl.d - R + 1e-9) return false;
        }
        return true;
    };

    // FCC lattice: points u*(i,j,k) with (i+j+k) even.
    // Nearest-neighbour distance = u*sqrt(2). Set that to 2R (+ tiny margin).
    const double margin = 1e-4;
    const double u = (2.0 * R + margin) / std::sqrt(2.0);

    // Lattice index ranges covering the bounding box (with slack).
    auto irange = [&](double a, double b, long long& imin, long long& imax) {
        imin = (long long)std::floor(a / u) - 1;
        imax = (long long)std::ceil(b / u) + 1;
    };
    long long ix0, ix1, iy0, iy1, iz0, iz1;
    irange(lo.x, hi.x, ix0, ix1);
    irange(lo.y, hi.y, iy0, iy1);
    irange(lo.z, hi.z, iz0, iz1);

    vector<V3> centers;
    centers.reserve(1 << 16);

    long long iterCounter = 0;
    bool stop = false;
    for (long long i = ix0; i <= ix1 && !stop; ++i) {
        double px = i * u;
        for (long long j = iy0; j <= iy1 && !stop; ++j) {
            double py = j * u;
            // parity: (i+j+k) even -> k has fixed parity given (i+j)
            long long parity = ((i + j) % 2 + 2) % 2; // need k parity = parity (so sum even)
            long long kstart = iz0;
            if ((((kstart) % 2) + 2) % 2 != parity) ++kstart;
            for (long long k = kstart; k <= iz1; k += 2) {
                if ((++iterCounter & 0x3FFF) == 0) {
                    double el = std::chrono::duration<double>(clock_t2::now() - t_start).count();
                    if (el > TIME_BUDGET) { stop = true; break; }
                }
                V3 p{ px, py, k * u };
                if (inside(p)) {
                    centers.push_back(p);
                    if ((long long)centers.size() >= MAX_K) { stop = true; break; }
                }
            }
        }
    }

    // Output.
    std::string out;
    out.reserve(centers.size() * 40 + 32);
    char buf[160];
    int n = snprintf(buf, sizeof(buf), "%zu\n", centers.size());
    out.append(buf, n);
    for (const auto& c : centers) {
        n = snprintf(buf, sizeof(buf), "%.6f %.6f %.6f 0 0 0\n", c.x, c.y, c.z);
        out.append(buf, n);
    }
    fwrite(out.data(), 1, out.size(), stdout);
    return 0;
}
