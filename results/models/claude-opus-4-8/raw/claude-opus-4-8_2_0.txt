using System;
using System.IO;
using System.Collections.Generic;

// Chinese Postman Problem (Route Inspection) solver.
//
// Strategy (adapts to graph size):
//   * Find odd-degree vertices. If none, the connected component of node 0 is
//     already Eulerian.
//   * To make it Eulerian we add a "T-join": a set of duplicated edges whose
//     odd-degree endpoints are exactly the odd vertices. Adding it makes every
//     vertex even, after which an Euler circuit (Hierholzer) traverses every
//     edge exactly once.
//   * Choosing the T-join:
//       - oddCount <= 18            : exact min-weight matching (bitmask DP) over
//                                     shortest-path distances -> near-optimal.
//       - oddCount <= 1000 & N<=2000: greedy nearest-neighbour matching over
//                                     shortest paths -> good quality, fast.
//       - otherwise (extreme/epic)  : spanning-tree T-join, O(N+M) -> always
//                                     valid, scales to 100k nodes / 1M edges.
//   Any T-join yields a valid route; the tiers trade optimality for speed.

class Program
{
    const long LMAX = long.MaxValue / 4;

    // ----- input buffer -----
    static byte[] buf;
    static int bpos, blen;

    // ----- original graph CSR -----
    static int N, M;
    static int[] off;      // CSR offsets, size N+1
    static int[] aTo;      // adjacency target
    static int[] aEid;     // adjacency edge id
    static int[] EU, EV;   // edge endpoints
    static long[] EW;      // edge weights

    static long ReadLong()
    {
        while (bpos < blen)
        {
            byte c = buf[bpos];
            if ((c >= '0' && c <= '9') || c == '-') break;
            bpos++;
        }
        bool neg = false;
        if (bpos < blen && buf[bpos] == '-') { neg = true; bpos++; }
        long v = 0;
        while (bpos < blen)
        {
            byte c = buf[bpos];
            if (c < '0' || c > '9') break;
            v = v * 10 + (c - '0');
            bpos++;
        }
        return neg ? -v : v;
    }

    static byte[] ReadAllInput()
    {
        using (var ms = new MemoryStream())
        {
            var s = Console.OpenStandardInput();
            byte[] tmp = new byte[1 << 16];
            int r;
            while ((r = s.Read(tmp, 0, tmp.Length)) > 0) ms.Write(tmp, 0, r);
            return ms.ToArray();
        }
    }

    static void Main()
    {
        buf = ReadAllInput();
        bpos = 0; blen = buf.Length;

        N = (int)ReadLong();
        M = (int)ReadLong();
        if (N <= 0) { Console.Out.Write("1\n0\n"); return; }

        EU = new int[M]; EV = new int[M]; EW = new long[M];
        int[] deg = new int[N];
        for (int i = 0; i < M; i++)
        {
            int u = (int)ReadLong();
            int v = (int)ReadLong();
            long w = ReadLong();
            EU[i] = u; EV[i] = v; EW[i] = w;
            if (u == v) deg[u] += 2;
            else { deg[u]++; deg[v]++; }
        }

        // Build CSR adjacency for original graph.
        off = new int[N + 1];
        for (int i = 0; i < N; i++) off[i + 1] = off[i] + deg[i];
        int totalAdj = off[N];
        aTo = new int[totalAdj];
        aEid = new int[totalAdj];
        int[] cur = (int[])off.Clone();
        for (int i = 0; i < M; i++)
        {
            int u = EU[i], v = EV[i];
            if (u == v)
            {
                aTo[cur[u]] = u; aEid[cur[u]] = i; cur[u]++;
                aTo[cur[u]] = u; aEid[cur[u]] = i; cur[u]++;
            }
            else
            {
                aTo[cur[u]] = v; aEid[cur[u]] = i; cur[u]++;
                aTo[cur[v]] = u; aEid[cur[v]] = i; cur[v]++;
            }
        }

        // BFS from 0: component + rooted spanning tree.
        int[] parent = new int[N];
        int[] parentEdge = new int[N];
        bool[] vis = new bool[N];
        int[] order = new int[N];
        int oc = 0;
        {
            int[] q = new int[N];
            int qh = 0, qt = 0;
            vis[0] = true; q[qt++] = 0; parent[0] = -1; parentEdge[0] = -1;
            while (qh < qt)
            {
                int v = q[qh++];
                order[oc++] = v;
                for (int e = off[v]; e < off[v + 1]; e++)
                {
                    int to = aTo[e];
                    if (!vis[to])
                    {
                        vis[to] = true;
                        parent[to] = v;
                        parentEdge[to] = aEid[e];
                        q[qt++] = to;
                    }
                }
            }
        }

        // Odd-degree vertices.
        List<int> odd = new List<int>();
        for (int v = 0; v < N; v++) if ((deg[v] & 1) == 1) odd.Add(v);
        int k = odd.Count;

        // Duplicate edge-ids (each duplicates an original edge).
        List<int> dup = new List<int>();

        if (k > 0)
        {
            if (k <= 18)
                ExactMatch(odd, dup);
            else if (k <= 1000 && N <= 2000)
                GreedyMatch(odd, dup);
            else
                TreeJoin(deg, order, oc, parent, parentEdge, dup);
        }

        // Build final multigraph (original + duplicates).
        int TI = M + dup.Count;
        int[] FU = new int[TI], FV = new int[TI];
        for (int i = 0; i < M; i++) { FU[i] = EU[i]; FV[i] = EV[i]; }
        for (int d = 0; d < dup.Count; d++)
        {
            int e = dup[d];
            FU[M + d] = EU[e]; FV[M + d] = EV[e];
        }

        int[] deg2 = new int[N];
        for (int i = 0; i < TI; i++)
        {
            int u = FU[i], v = FV[i];
            if (u == v) deg2[u] += 2;
            else { deg2[u]++; deg2[v]++; }
        }
        int[] off2 = new int[N + 1];
        for (int i = 0; i < N; i++) off2[i + 1] = off2[i] + deg2[i];
        int tot2 = off2[N];
        int[] adjTo = new int[tot2];
        int[] adjE = new int[tot2];
        int[] cur2 = (int[])off2.Clone();
        for (int i = 0; i < TI; i++)
        {
            int u = FU[i], v = FV[i];
            if (u == v)
            {
                adjTo[cur2[u]] = u; adjE[cur2[u]] = i; cur2[u]++;
                adjTo[cur2[u]] = u; adjE[cur2[u]] = i; cur2[u]++;
            }
            else
            {
                adjTo[cur2[u]] = v; adjE[cur2[u]] = i; cur2[u]++;
                adjTo[cur2[v]] = u; adjE[cur2[v]] = i; cur2[v]++;
            }
        }

        // Hierholzer's algorithm from node 0.
        bool[] used = new bool[TI];
        int[] iter = (int[])off2.Clone();
        int[] stack = new int[TI + 2];
        int sp = 0;
        int[] circuit = new int[TI + 2];
        int cc = 0;
        stack[sp++] = 0;
        while (sp > 0)
        {
            int v = stack[sp - 1];
            bool found = false;
            while (iter[v] < off2[v + 1])
            {
                int idx = iter[v];
                int e = adjE[idx];
                iter[v]++;
                if (!used[e])
                {
                    used[e] = true;
                    stack[sp++] = adjTo[idx];
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                circuit[cc++] = v;
                sp--;
            }
        }

        // Output: L then the route (circuit is reversed -> iterate backwards).
        long obsize = (long)(cc + 2) * 8 + 16;
        byte[] ob = new byte[obsize];
        int p = 0;
        p = WriteInt(ob, p, cc);
        ob[p++] = (byte)'\n';
        for (int i = cc - 1; i >= 0; i--)
        {
            p = WriteInt(ob, p, circuit[i]);
            ob[p++] = (byte)' ';
        }
        ob[p++] = (byte)'\n';
        var os = Console.OpenStandardOutput();
        os.Write(ob, 0, p);
        os.Flush();
    }

    // ---- Spanning-tree T-join (scales to any size) ----
    static void TreeJoin(int[] deg, int[] order, int oc, int[] parent, int[] parentEdge, List<int> dup)
    {
        int[] cnt = new int[N];
        for (int v = 0; v < N; v++) cnt[v] = deg[v] & 1;
        for (int idx = oc - 1; idx >= 0; idx--)
        {
            int v = order[idx];
            int pa = parent[v];
            if (pa >= 0)
            {
                if ((cnt[v] & 1) == 1) dup.Add(parentEdge[v]);
                cnt[pa] += cnt[v];
            }
        }
    }

    // ---- Exact min-weight matching via bitmask DP (oddCount <= 18) ----
    static void ExactMatch(List<int> odd, List<int> dup)
    {
        int k = odd.Count;
        int[][] PE = new int[k][];
        int[][] PN = new int[k][];
        long[,] dm = new long[k, k];
        for (int i = 0; i < k; i++)
        {
            long[] dist; int[] pe, pn;
            Dijkstra(odd[i], out dist, out pe, out pn);
            PE[i] = pe; PN[i] = pn;
            for (int j = 0; j < k; j++) dm[i, j] = dist[odd[j]];
        }

        int full = 1 << k;
        long[] dp = new long[full];
        int[] choice = new int[full];
        for (int m = 1; m < full; m++) dp[m] = LMAX;
        dp[0] = 0;
        for (int m = 0; m < full; m++)
        {
            if (dp[m] >= LMAX) continue;
            int i = 0;
            while (i < k && (m & (1 << i)) != 0) i++;
            if (i == k) continue;
            for (int j = i + 1; j < k; j++)
            {
                if ((m & (1 << j)) != 0) continue;
                long nd = dp[m] + dm[i, j];
                int nm = m | (1 << i) | (1 << j);
                if (nd < dp[nm]) { dp[nm] = nd; choice[nm] = (i << 16) | j; }
            }
        }

        int mask = full - 1;
        while (mask != 0)
        {
            int c = choice[mask];
            int i = c >> 16, j = c & 0xffff;
            DuplicatePath(odd[i], odd[j], PE[i], PN[i], dup);
            mask ^= (1 << i) | (1 << j);
        }
    }

    // ---- Greedy nearest-neighbour matching ----
    static void GreedyMatch(List<int> odd, List<int> dup)
    {
        int k = odd.Count;
        int[][] PE = new int[k][];
        int[][] PN = new int[k][];
        long[,] dm = new long[k, k];
        for (int i = 0; i < k; i++)
        {
            long[] dist; int[] pe, pn;
            Dijkstra(odd[i], out dist, out pe, out pn);
            PE[i] = pe; PN[i] = pn;
            for (int j = 0; j < k; j++) dm[i, j] = dist[odd[j]];
        }

        bool[] matched = new bool[k];
        for (int i = 0; i < k; i++)
        {
            if (matched[i]) continue;
            long best = LMAX; int bj = -1;
            for (int j = 0; j < k; j++)
            {
                if (j == i || matched[j]) continue;
                if (dm[i, j] < best) { best = dm[i, j]; bj = j; }
            }
            if (bj < 0) continue;
            matched[i] = true; matched[bj] = true;
            DuplicatePath(odd[i], odd[bj], PE[i], PN[i], dup);
        }
    }

    // Duplicate the edges along the shortest path from src to dst, using src's
    // parent arrays (walk back from dst).
    static void DuplicatePath(int src, int dst, int[] parE, int[] parN, List<int> dup)
    {
        int node = dst;
        while (node != src)
        {
            int e = parE[node];
            if (e < 0) break; // unreachable (disconnected) - best effort
            dup.Add(e);
            node = parN[node];
        }
    }

    // Dijkstra over the original graph (negative weights clamped to 0 so the
    // algorithm stays valid; the resulting path is still valid for duplication).
    static void Dijkstra(int src, out long[] dist, out int[] parE, out int[] parN)
    {
        int n = N;
        dist = new long[n];
        parE = new int[n];
        parN = new int[n];
        for (int i = 0; i < n; i++) { dist[i] = LMAX; parE[i] = -1; parN[i] = -1; }
        dist[src] = 0;

        int cap = off[n] + 8;
        long[] hd = new long[cap];
        int[] hn = new int[cap];
        int hs = 0;
        hd[0] = 0; hn[0] = src; hs = 1;

        while (hs > 0)
        {
            long d = hd[0]; int v = hn[0];
            hs--;
            hd[0] = hd[hs]; hn[0] = hn[hs];
            // sift down
            int ii = 0;
            while (true)
            {
                int l = 2 * ii + 1, r = 2 * ii + 2, s = ii;
                if (l < hs && hd[l] < hd[s]) s = l;
                if (r < hs && hd[r] < hd[s]) s = r;
                if (s == ii) break;
                long t1 = hd[ii]; hd[ii] = hd[s]; hd[s] = t1;
                int t2 = hn[ii]; hn[ii] = hn[s]; hn[s] = t2;
                ii = s;
            }
            if (d > dist[v]) continue;

            for (int e = off[v]; e < off[v + 1]; e++)
            {
                int to = aTo[e];
                long w = EW[aEid[e]];
                if (w < 0) w = 0;
                long nd = d + w;
                if (nd < dist[to])
                {
                    dist[to] = nd;
                    parE[to] = aEid[e];
                    parN[to] = v;
                    if (hs >= cap)
                    {
                        cap *= 2;
                        Array.Resize(ref hd, cap);
                        Array.Resize(ref hn, cap);
                    }
                    int j = hs;
                    hd[j] = nd; hn[j] = to;
                    hs++;
                    // sift up
                    while (j > 0)
                    {
                        int pp = (j - 1) / 2;
                        if (hd[pp] <= hd[j]) break;
                        long t1 = hd[pp]; hd[pp] = hd[j]; hd[j] = t1;
                        int t2 = hn[pp]; hn[pp] = hn[j]; hn[j] = t2;
                        j = pp;
                    }
                }
            }
        }
    }

    static int WriteInt(byte[] b, int p, int val)
    {
        if (val < 0) { b[p++] = (byte)'-'; val = -val; }
        if (val == 0) { b[p++] = (byte)'0'; return p; }
        int s = p;
        while (val > 0) { b[p++] = (byte)('0' + (val % 10)); val /= 10; }
        int a = s, e = p - 1;
        while (a < e) { byte t = b[a]; b[a] = b[e]; b[e] = t; a++; e--; }
        return p;
    }
}
