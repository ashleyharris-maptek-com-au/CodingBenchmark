using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

public class Program
{
    // Graph storage
    private static int _numVertices;
    private static int[][] _adjacencyList; // Sorted adjacency lists for intersection
    private static BitArray[] _adjacencyMatrix; // BitArray for O(1) edge checks
    private static int[] _coreNumbers;
    
    // Solution tracking
    private static volatile int _globalBestSize = 0;
    private static int[] _globalBestClique = new int[0];
    private static readonly object _bestLock = new object();
    
    // Cancellation
    private static readonly CancellationTokenSource _cts = new CancellationTokenSource();
    private static readonly Stopwatch _sw = new Stopwatch();
    private static readonly int _timeLimitMs = 270000; // 4.5 minutes

    public static void Main(string[] args)
    {
        _sw.Start();
        
        // Run timeout task
        Task.Run(() => {
            Thread.Sleep(_timeLimitMs);
            _cts.Cancel();
        });

        try
        {
            ReadInput();
            ComputeCoreDecomposition();
            FindMaximumClique();
            OutputResult();
        }
        catch (OperationCanceledException)
        {
            OutputResult(); // Output best found so far
        }
        catch (Exception)
        {
            // If crashed, output current best
            OutputResult();
        }
    }

    private static void ReadInput()
    {
        // Use BufferedStream for faster reading
        using (var stream = new BufferedStream(Console.OpenStandardInput()))
        using (var reader = new StreamReader(stream))
        {
            var firstLine = reader.ReadLine();
            if (string.IsNullOrWhiteSpace(firstLine)) return;
            
            var parts = firstLine.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            _numVertices = int.Parse(parts[0]);
            int numEdges = int.Parse(parts[1]);

            _adjacencyList = new int[_numVertices][];
            _adjacencyMatrix = new BitArray[_numVertices];
            
            var tempNeighbors = new List<int>[_numVertices];
            for (int i = 0; i < _numVertices; i++)
            {
                tempNeighbors[i] = new List<int>();
            }

            for (int i = 0; i < numEdges; i++)
            {
                var line = reader.ReadLine();
                if (string.IsNullOrWhiteSpace(line)) continue;
                var tokens = line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
                int u = int.Parse(tokens[0]);
                int v = int.Parse(tokens[1]);
                
                if (u == v) continue;
                
                tempNeighbors[u].Add(v);
                tempNeighbors[v].Add(u);
            }

            for (int i = 0; i < _numVertices; i++)
            {
                tempNeighbors[i].Sort();
                _adjacencyList[i] = tempNeighbors[i].ToArray();
                _adjacencyMatrix[i] = new BitArray(_numVertices);
                foreach (var n in tempNeighbors[i])
                {
                    _adjacencyMatrix[i][n] = true;
                }
            }
        }
    }

    private static void ComputeCoreDecomposition()
    {
        _coreNumbers = new int[_numVertices];
        int maxDegree = 0;
        int[] degrees = new int[_numVertices];
        
        for (int i = 0; i < _numVertices; i++)
        {
            degrees[i] = _adjacencyList[i].Length;
            if (degrees[i] > maxDegree) maxDegree = degrees[i];
        }

        var buckets = new List<int>[maxDegree + 1];
        for (int i = 0; i <= maxDegree; i++) buckets[i] = new List<int>();
        
        for (int i = 0; i < _numVertices; i++)
        {
            buckets[degrees[i]].Add(i);
        }

        int processed = 0;
        for (int d = 0; d <= maxDegree; d++)
        {
            while (buckets[d].Count > 0)
            {
                int u = buckets[d][buckets[d].Count - 1];
                buckets[d].RemoveAt(buckets[d].Count - 1);
                
                if (_coreNumbers[u] > 0) continue; // Already processed
                
                _coreNumbers[u] = d;
                processed++;
                
                foreach (var v in _adjacencyList[u])
                {
                    if (_coreNumbers[v] > 0) continue;
                    
                    if (degrees[v] > d)
                    {
                        // Move v to lower bucket
                        buckets[degrees[v]].Remove(v);
                        degrees[v]--;
                        buckets[degrees[v]].Add(v);
                    }
                }
            }
            
            if (processed == _numVertices) break;
        }
    }

    private static void FindMaximumClique()
    {
        // Greedy heuristic to get a good initial solution
        RunGreedyHeuristic();
        
        // Order vertices by core number for branching
        var vertices = Enumerable.Range(0, _numVertices)
            .OrderByDescending(v => _coreNumbers[v])
            .ThenByDescending(v => _adjacencyList[v].Length)
            .ToList();

        // Parallel processing with limited concurrency
        int degreeParallelism = Math.Min(Environment.ProcessorCount, 16);
        var options = new ParallelOptions { 
            CancellationToken = _cts.Token,
            MaxDegreeOfParallelism = degreeParallelism 
        };

        // Order vertices by core number for branching
        vertices.Sort((a, b) => _coreNumbers[b].CompareTo(_coreNumbers[a]));

        try
        {
            Parallel.For(0, vertices.Count, options, i =>
            {
                if (_cts.IsCancellationRequested) return;
                
                int v = vertices[i];
                int ub = _coreNumbers[v] + 1;
                
                // Pruning: if upper bound <= current best, skip
                if (ub <= _globalBestSize) return;
                
                // Build initial candidate set from neighbors
                var candidates = new List<int>();
                foreach (var n in _adjacencyList[v])
                {
                    if (_coreNumbers[n] >= _globalBestSize)
                        candidates.Add(n);
                }
                
                // Branch
                BranchAndBound(new List<int> { v }, candidates, 1);
            });
        }
        catch (OperationCanceledException) { }
    }

    private static void RunGreedyHeuristic()
    {
        // Run multiple greedy strategies to find a good initial clique
        var strategies = new Func<int, int>[] {
            v => -_coreNumbers[v],                    // Highest core first
            v => -_adjacencyList[v].Length,           // Highest degree first
            v => _coreNumbers[v],                     // Lowest core first
            v => _adjacencyList[v].Length             // Lowest degree first
        };

        foreach (var strategy in strategies)
        {
            var order = Enumerable.Range(0, _numVertices).OrderBy(strategy).ToList();
            
            for (int i = 0; i < _numVertices; i++)
            {
                int start = order[i];
                var clique = new List<int> { start };
                var candidates = new List<int>(_adjacencyList[start]);
                
                while (candidates.Count > 0)
                {
                    // Select vertex with most connections to other candidates
                    int best = -1;
                    int bestCount = -1;
                    
                    foreach (var c in candidates)
                    {
                        int cnt = 0;
                        foreach (var n in _adjacencyList[c])
                        {
                            if (candidates.Contains(n)) cnt++;
                        }
                        
                        if (cnt > bestCount)
                        {
                            bestCount = cnt;
                            best = c;
                        }
                    }
                    
                    if (best == -1) break;
                    
                    clique.Add(best);
                    candidates.Remove(best);
                    
                    // Filter candidates to those connected to 'best'
                    var newCandidates = new List<int>();
                    foreach (var c in candidates)
                    {
                        if (_adjacencyMatrix[best][c])
                            newCandidates.Add(c);
                    }
                    candidates = newCandidates;
                }
                
                UpdateBest(clique);
            }
        }
    }

    private static void BranchAndBound(List<int> currentClique, List<int> candidates, int currentSize)
    {
        if (_cts.IsCancellationRequested) return;
        
        // Update best if improved
        UpdateBest(currentClique);
        
        if (candidates.Count == 0) return;

        // Upper bound: current size + number of candidates
        // More refined bound: find max clique in candidate induced subgraph is expensive,
        // so we use coloring approximation
        int ub = currentSize + candidates.Count;
        if (ub <= _globalBestSize) return;

        // Order candidates by core number (descending) for better pruning
        candidates.Sort((a, b) => _coreNumbers[b].CompareTo(_coreNumbers[a]));

        // Branch on candidates
        for (int i = 0; i < candidates.Count; i++)
        {
            if (_cts.IsCancellationRequested) return;
            
            int v = candidates[i];
            
            // Pruning based on position in sorted candidates
            int potentialSize = currentSize + (candidates.Count - i);
            if (potentialSize <= _globalBestSize) break;
            
            // Check if this vertex can potentially improve solution
            if (_coreNumbers[v] + 1 <= _globalBestSize) continue;

            // New candidate set: neighbors of v that are also in current candidates
            var newCandidates = new List<int>();
            
            // Use binary search for efficiency since adjacency lists are sorted
            foreach (var n in _adjacencyList[v])
            {
                if (candidates.BinarySearch(n) >= 0 && _coreNumbers[n] >= _globalBestSize)
                {
                    newCandidates.Add(n);
                }
            }

            var newClique = new List<int>(currentClique) { v };
            BranchAndBound(newClique, newCandidates, currentSize + 1);
            
            // Remove v from candidates for subsequent branches (avoid duplicates)
            // This is implicitly handled by the loop
        }
    }

    private static void UpdateBest(List<int> clique)
    {
        if (clique.Count > _globalBestSize)
        {
            lock (_bestLock)
            {
                if (clique.Count > _globalBestSize)
                {
                    _globalBestSize = clique.Count;
                    _globalBestClique = clique.ToArray();
                }
            }
        }
    }

    private static void OutputResult()
    {
        // Verify the clique is valid
        if (_globalBestClique.Length > 0)
        {
            for (int i = 0; i < _globalBestClique.Length; i++)
            {
                for (int j = i + 1; j < _globalBestClique.Length; j++)
                {
                    int u = _globalBestClique[i];
                    int v = _globalBestClique[j];
                    if (!_adjacencyMatrix[u][v])
                    {
                        // Invalid clique, output single vertex
                        Console.WriteLine("1");
                        Console.WriteLine("0");
                        return;
                    }
                }
            }
        }

        using (var writer = new StreamWriter(Console.OpenStandardOutput()))
        {
            writer.WriteLine(_globalBestSize);
            if (_globalBestSize > 0)
            {
                writer.WriteLine(string.Join(" ", _globalBestClique));
            }
            else
            {
                writer.WriteLine();
            }
        }
    }
}

// Simple BitArray implementation for .NET compatibility
public class BitArray
{
    private readonly int[] _data;
    private readonly int _length;

    public BitArray(int length)
    {
        _length = length;
        _data = new int[(length + 31) / 32];
    }

    public bool this[int index]
    {
        get
        {
            return (_data[index >> 5] & (1 << (index & 31))) != 0;
        }
        set
        {
            if (value)
                _data[index >> 5] |= (1 << (index & 31));
            else
                _data[index >> 5] &= ~(1 << (index & 31));
        }
    }
}
