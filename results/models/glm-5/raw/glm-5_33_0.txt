```csharp
using System;
using System.Collections.Generic;
using System.Linq;

namespace QAPSolver
{
    class Program
    {
        static void Main(string[] args)
        {
            // 1. Read Input
            Solver solver = new Solver();
            solver.ReadInput();
            
            // 2. Solve
            solver.Solve();
            
            // 3. Output
            solver.WriteOutput();
        }
    }

    class Solver
    {
        private int N;
        private int[,] flow;
        private int[,] dist;
        
        // solution[i] = location assigned to facility i
        private int[] solution; 
        
        // reverse mapping: locToFac[l] = facility assigned to location l
        private int[] locToFac;

        private long currentCost;
        private Random random = new Random(12345);

        public void ReadInput()
        {
            string line;
            // Skip empty lines at the start if any
            while (string.IsNullOrWhiteSpace(line = Console.ReadLine())) ;
            
            N = int.Parse(line.Trim());
            
            flow = new int[N, N];
            dist = new int[N, N];
            
            // Read Flow Matrix
            for (int i = 0; i < N; i++)
            {
                // Handle potential line breaks or multiple spaces
                var parts = ReadInts();
                for (int j = 0; j < N; j++)
                {
                    flow[i, j] = parts[j];
                }
            }
            
            // Read Distance Matrix
            for (int i = 0; i < N; i++)
            {
                var parts = ReadInts();
                for (int j = 0; j < N; j++)
                {
                    dist[i, j] = parts[j];
                }
            }
        }

        private int[] ReadInts()
        {
            string line;
            while (string.IsNullOrWhiteSpace(line = Console.ReadLine())) ;
            return line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries)
                       .Select(s => int.Parse(s)).ToArray();
        }

        public void Solve()
        {
            solution = new int[N];
            locToFac = new int[N];
            
            // Initialize with a greedy heuristic
            GreedyInit();

            if (N > 1)
            {
                // Iterative Improvement (2-opt)
                bool improved = true;
                // Limit iterations to prevent infinite loops on oscillations 
                // (though strict 'if better' usually converges)
                int maxPasses = 100; 
                int pass = 0;

                // We use a do-while loop. 
                // For very large N, a single pass of 2-opt is often sufficient 
                // to get a good solution quickly. Full convergence might be slow.
                // However, 2-opt is O(N^2). N=2000 -> 4M ops per pass. Fast enough.
                do
                {
                    improved = false;
                    pass++;
                    
                    // Standard 2-opt neighborhood search
                    // We iterate through all pairs (i, j) where i < j
                    // Swap facilities at locations i and j (Wait, standard swap is swapping facilities)
                    // Standard QAP Swap:
                    // Currently: Fac A at Loc r, Fac B at Loc s
                    // Swap: Fac A to Loc s, Fac B at Loc r
                    
                    // Implementation detail:
                    // solution[fac] = loc.
                    // To swap facilities assigned to loc_r and loc_s:
                    // fac_r = locToFac[r], fac_s = locToFac[s]
                    // New: solution[fac_r] = s, solution[fac_s] = r.
                    
                    // We iterate r from 0 to N-1, s from r+1 to N-1.
                    // This corresponds to swapping the facilities currently at location r and s.

                    for (int r = 0; r < N - 1; r++)
                    {
                        for (int s = r + 1; s < N; s++)
                        {
                            long delta = CalculateSwapDelta(r, s);

                            if (delta < 0)
                            {
                                PerformSwap(r, s);
                                currentCost += delta;
                                improved = true;
                            }
                        }
                    }
                } while (improved && pass < maxPasses);
            }
            
            currentCost = CalculateFullCost();
        }

        private void GreedyInit()
        {
            // Heuristic:
            // Assign facility with highest flow to location with lowest average distance?
            // Or simpler: Sort facilities by total flow, Locations by total distance.
            // Match them.
            
            var facIndices = Enumerable.Range(0, N).ToList();
            var locIndices = Enumerable.Range(0, N).ToList();

            // Calculate total flow for each facility
            long[] flowSum = new long[N];
            for(int i=0; i<N; i++)
            {
                long sum = 0;
                for(int j=0; j<N; j++) sum += flow[i,j];
                flowSum[i] = sum;
            }

            // Calculate total distance for each location
            long[] distSum = new long[N];
            for(int i=0; i<N; i++)
            {
                long sum = 0;
                for(int j=0; j<N; j++) sum += dist[i,j];
                distSum[i] = sum;
            }

            // Sort facilities descending by flow
            facIndices.Sort((a, b) => flowSum[b].CompareTo(flowSum[a]));
            
            // Sort locations ascending by distance
            locIndices.Sort((a, b) => distSum[a].CompareTo(distSum[b]));

            // Assign
            for(int i=0; i<N; i++)
            {
                int fac = facIndices[i];
                int loc = locIndices[i];
                solution[fac] = loc;
                locToFac[loc] = fac;
            }

            currentCost = CalculateFullCost();
        }

        // Calculates the change in cost if we swap facilities at locations r and s
        private long CalculateSwapDelta(int r, int s)
        {
            if (r == s) return 0;
            
            int u = locToFac[r]; // Facility currently at location r
            int v = locToFac[s]; // Facility currently at location s
            
            // We are swapping assignments of u and v.
            // u goes to s, v goes to r.
            
            long delta = 0;
            
            // Contributions involving u and v.
            // Iterate over all facilities k.
            for (int k = 0; k < N; k++)
            {
                if (k == u || k == v) continue;

                int w = locToFac[solution[k]]; // Location of facility k (Wait, solution[k] IS location)
                int locK = solution[k];

                // Interaction u-k
                int f_uk = flow[u, k];
                int d_old_uk = dist[r, locK];
                int d_new_uk = dist[s, locK];
                
                delta += (long)f_uk * (d_new_uk - d_old_uk);

                // Interaction v-k
                int f_vk = flow[v, k];
                int d_old_vk = dist[s, locK];
                int d_new_vk = dist[r, locK];
                
                delta += (long)f_vk * (d_new_vk - d_old_vk);
            }

            // Interaction u-v
            int f_uv = flow[u, v];
            int d_old_uv = dist[r, s]; // dist between their current locations
            int d_new_uv = dist[s, r]; // dist between swapped locations (same value, indices swapped)
            
            // Note dist[r,s] == dist[s,r] usually, but generally:
            // Old: u at r, v at s -> cost flow[u,v] * dist[r,s]
            // New: u at s, v at r -> cost flow[u,v] * dist[s,r]
            // Also flow[v,u] * dist[s,r] (old) vs flow[v,u] * dist[r,s] (new)
            // If matrices symmetric, this part is 0 change.
            // If asymmetric:
            // f_uv * (dist(s,r) - dist(r,s)) + f_vu * (dist(r,s) - dist(s,r))
            // = (f_uv - f_vu) * (dist(s,r) - dist(r,s))
            
            // Standard formula logic:
            // Cost change for pair (u,v):
            // Old: f_uv*d_rs + f_vu*d_sr
            // New: f_uv*d_sr + f_vu*d_rs
            // Change: f_uv*(d_sr - d_rs) + f_vu*(d_rs - d_sr)
            
            delta += (long)f_uv * (dist[s, r] - dist[r, s]);
            delta += (long)flow[v, u] * (dist[r, s] - dist[s, r]);
            
            return delta;
        }

        private void PerformSwap(int r, int s)
        {
            int u = locToFac[r];
            int v = locToFac[s];

            // Swap locations in solution
            solution[u] = s;
            solution[v] = r;

            // Swap facilities in reverse mapping
            locToFac[r] = v;
            locToFac[s] = u;
        }

        private long CalculateFullCost()
        {
            long cost = 0;
            for (int i = 0; i < N; i++)
            {
                for (int j = 0; j < N; j++)
                {
                    cost += (long)flow[i, j] * dist[solution[i], solution[j]];
                }
            }
            return cost;
        }

        public void WriteOutput()
        {
            // Verify permutation (debug safety)
            #if DEBUG
            bool[] used = new bool[N];
            foreach(var x in solution) {
                if(x < 0 || x >= N || used[x]) throw new Exception("Invalid perm");
                used[x] = true;
            }
            #endif

            Console.WriteLine(currentCost);
            Console.WriteLine(string.Join(" ", solution));
        }
    }
}
```
