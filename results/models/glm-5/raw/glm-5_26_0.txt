Here's a comprehensive Rust solution that adapts its clustering strategy based on dataset scale:

```rust
use std::io::{self, BufReader, BufWriter, Read, Write};
use std::time::Instant;

const BATCH_SIZE: usize = 100_000;
const SMALL_THRESHOLD: usize = 10_000_000;
const MEDIUM_THRESHOLD: usize = 100_000_000;
const LARGE_THRESHOLD: usize = 1_000_000_000;

/// 3D point representation
#[derive(Clone, Copy, Default)]
struct Point {
    x: f64,
    y: f64,
    z: f64,
}

impl Point {
    #[inline]
    fn distance_squared(&self, other: &Point) -> f64 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }

    #[inline]
    fn add(&self, other: &Point) -> Point {
        Point {
            x: self.x + other.x,
            y: self.y + other.y,
            z: self.z + other.z,
        }
    }

    #[inline]
    fn scale(&self, s: f64) -> Point {
        Point {
            x: self.x * s,
            y: self.y * s,
            z: self.z * s,
        }
    }
}

/// Read a batch of points from binary input
fn read_point_batch(reader: &mut BufReader<io::Stdin>, buffer: &mut Vec<Point>) -> io::Result<usize> {
    buffer.clear();
    let point_size = 12usize;
    let mut raw_buffer = vec![0u8; BATCH_SIZE * point_size];
    
    let mut bytes_read = 0usize;
    let target_bytes = BATCH_SIZE * point_size;
    
    while bytes_read < target_bytes {
        match reader.read(&mut raw_buffer[bytes_read..target_bytes]) {
            Ok(0) => break,
            Ok(n) => bytes_read += n,
            Err(e) => return Err(e),
        }
    }
    
    let num_points = bytes_read / point_size;
    buffer.reserve(num_points);
    
    for i in 0..num_points {
        let offset = i * point_size;
        let x = f32::from_le_bytes([
            raw_buffer[offset], raw_buffer[offset + 1],
            raw_buffer[offset + 2], raw_buffer[offset + 3],
        ]);
        let y = f32::from_le_bytes([
            raw_buffer[offset + 4], raw_buffer[offset + 5],
            raw_buffer[offset + 6], raw_buffer[offset + 7],
        ]);
        let z = f32::from_le_bytes([
            raw_buffer[offset + 8], raw_buffer[offset + 9],
            raw_buffer[offset + 10], raw_buffer[offset + 11],
        ]);
        buffer.push(Point {
            x: x as f64,
            y: y as f64,
            z: z as f64,
        });
    }
    
    Ok(num_points)
}

/// Initialize centroids using k-means++ initialization
fn initialize_centroids_kmeanspp(points: &[Point], k: usize) -> Vec<Point> {
    let n = points.len();
    if n == 0 || k == 0 {
        return vec![Point::default(); k];
    }
    
    let mut centroids = Vec::with_capacity(k);
    let mut rng_state: u64 = 0x853c49e6748fea9b;
    
    // Fast xorshift64 RNG
    let next_random = |state: &mut u64| -> f64 {
        *state ^= *state >> 12;
        *state ^= *state << 25;
        *state ^= *state >> 27;
        (*state).wrapping_mul(0x2545F4914F6CDD1D) as f64 / u64::MAX as f64
    };
    
    // Pick first centroid randomly
    let first_idx = (next_random(&mut rng_state) * n as f64) as usize;
    centroids.push(points[first_idx]);
    
    let mut distances = vec![f64::MAX; n];
    
    for _ in 1..k {
        // Update distances to nearest centroid
        let new_centroid = centroids.last().unwrap();
        let mut total_dist = 0.0;
        
        for (i, point) in points.iter().enumerate() {
            let dist = point.distance_squared(new_centroid);
            distances[i] = distances[i].min(dist);
            total_dist += distances[i];
        }
        
        // Weighted random selection
        let threshold = next_random(&mut rng_state) * total_dist;
        let mut cumulative = 0.0;
        let mut next_idx = 0;
        
        for (i, &d) in distances.iter().enumerate() {
            cumulative += d;
            if cumulative >= threshold {
                next_idx = i;
                break;
            }
        }
        
        centroids.push(points[next_idx]);
    }
    
    centroids
}

/// Streaming initialization: reservoir sampling then refine
fn initialize_centroids_streaming(
    reader: &mut BufReader<io::Stdin>,
    num_points: usize,
    k: usize,
) -> io::Result<Vec<Point>> {
    // Sample approximately min(k*100, sqrt(n)) points for initialization
    let sample_size = ((k * 100).min((num_points as f64).sqrt() as usize)).max(k).min(10000);
    
    let mut reservoir: Vec<Point> = Vec::with_capacity(sample_size);
    let mut buffer = Vec::with_capacity(BATCH_SIZE);
    let mut points_seen = 0usize;
    let mut rng_state: u64 = 0x853c49e6748fea9b;
    
    let next_random = |state: &mut u64| -> f64 {
        *state ^= *state >> 12;
        *state ^= *state << 25;
        *state ^= *state >> 27;
        (*state).wrapping_mul(0x2545F4914F6CDD1D) as f64 / u64::MAX as f64
    };
    
    // Reservoir sampling
    loop {
        let count = read_point_batch(reader, &mut buffer)?;
        if count == 0 {
            break;
        }
        
        for point in buffer.iter().take(count) {
            if reservoir.len() < sample_size {
                reservoir.push(*point);
            } else {
                let j = (next_random(&mut rng_state) * (points_seen + 1) as f64) as usize;
                if j < sample_size {
                    reservoir[j] = *point;
                }
            }
            points_seen += 1;
        }
    }
    
    // Run k-means++ on the sample
    Ok(initialize_centroids_kmeanspp(&reservoir, k))
}

/// Find nearest centroid to a point
#[inline]
fn find_nearest_centroid(point: &Point, centroids: &[Point]) -> usize {
    let mut best_idx = 0;
    let mut best_dist = point.distance_squared(&centroids[0]);
    
    for (i, centroid) in centroids.iter().enumerate().skip(1) {
        let dist = point.distance_squared(centroid);
        if dist < best_dist {
            best_dist = dist;
            best_idx = i;
        }
    }
    
    best_idx
}

/// Lloyd's algorithm for k-means
fn lloyd_kmeans(points: &[Point], k: usize, max_iterations: usize) -> Vec<usize> {
    if points.is_empty() || k == 0 {
        return vec![];
    }
    
    let n = points.len();
    let mut centroids = initialize_centroids_kmeanspp(points, k.min(n));
    let actual_k = centroids.len();
    
    let mut assignments = vec![0u32; n];
    let mut new_centroids = vec![Point::default(); actual_k];
    let mut counts = vec![0usize; actual_k];
    
    for _ in 0..max_iterations {
        // Assignment step
        let mut changed = false;
        for (i, point) in points.iter().enumerate() {
            let new_assignment = find_nearest_centroid(point, &centroids) as u32;
            if assignments[i] != new_assignment {
                assignments[i] = new_assignment;
                changed = true;
            }
        }
        
        if !changed {
            break;
        }
        
        // Update step
        for c in new_centroids.iter_mut() {
            *c = Point::default();
        }
        counts.fill(0);
        
        for (i, &assignment) in assignments.iter().enumerate() {
            let idx = assignment as usize;
            new_centroids[idx].x += points[i].x;
            new_centroids[idx].y += points[i].y;
            new_centroids[idx].z += points[i].z;
            counts[idx] += 1;
        }
        
        for (i, (centroid, &count)) in new_centroids.iter_mut().zip(counts.iter()).enumerate() {
            if count > 0 {
                centroids[i] = centroid.scale(1.0 / count as f64);
            }
        }
    }
    
    assignments.into_iter().map(|x| x as usize).collect()
}

/// Mini-batch k-means (streaming variant)
fn minibatch_kmeans_streaming(
    reader: &mut BufReader<io::Stdin>,
    num_points: usize,
    k: usize,
    max_iterations: usize,
) -> io::Result<Vec<Point>> {
    let mut buffer = Vec::with_capacity(BATCH_SIZE);
    let mut centroids = initialize_centroids_streaming(reader, num_points, k)?;
    
    // Per-centroid counts for incremental averaging
    let mut counts = vec![1usize; k];
    
    let mut rng_state: u64 = 0x853c49e6748fea9b;
    let next_random = |state: &mut u64| -> f64 {
        *state ^= *state >> 12;
        *state ^= *state << 25;
        *state ^= *state >> 27;
        (*state).wrapping_mul(0x2545F4914F6CDD1D) as f64 / u64::MAX as f64
    };
    
    for iteration in 0..max_iterations {
        // Mini-batch learning rate (decreasing)
        let lr = 1.0 / (iteration + 1) as f64;
        
        // Process data in batches
        let mut points_processed = 0usize;
        let mut max_iterations_for_size = max_iterations;
        
        // Adjust iterations based on dataset size
        if num_points > LARGE_THRESHOLD {
            max_iterations_for_size = max_iterations.min(5);
        } else if num_points > MEDIUM_THRESHOLD {
            max_iterations_for_size = max_iterations.min(10);
        }
        
        loop {
            let count = read_point_batch(reader, &mut buffer)?;
            if count == 0 {
                break;
            }
            
            for point in buffer.iter().take(count) {
                let nearest = find_nearest_centroid(point, &centroids);
                counts[nearest] += 1;
                let eta = lr / counts[nearest] as f64;
                
                // Incremental centroid update
                centroids[nearest].x += eta * (point.x - centroids[nearest].x);
                centroids[nearest].y += eta * (point.y - centroids[nearest].y);
                centroids[nearest].z += eta * (point.z - centroids[nearest].z);
            }
            
            points_processed += count;
        }
        
        // Stop early for very large datasets
        if iteration >= max_iterations_for_size - 1 {
            break;
        }
        
        // Rewind for next iteration (we need to re-read data)
        // For true streaming, we'd use single-pass; here we simulate
        if iteration < max_iterations_for_size - 1 {
            // In a real streaming scenario, we'd just continue with new data
            // For this implementation, we process in a single pass with adaptive learning
            break;
        }
    }
    
    Ok(centroids)
}

/// Streaming assignment: assign points to nearest centroids
fn streaming_assign(
    reader: &mut BufReader<io::Stdin>,
    centroids: &[Point],
    writer: &mut BufWriter<io::Stdout>,
) -> io::Result<()> {
    let mut buffer = Vec::with_capacity(BATCH_SIZE);
    
    loop {
        let count = read_point_batch(reader, &mut buffer)?;
        if count == 0 {
            break;
        }
        
        let mut output_buffer = vec![0u8; count * 4];
        for (i, point) in buffer.iter().take(count).enumerate() {
            let assignment = find_nearest_centroid(point, centroids) as u32;
            let bytes = assignment.to_le_bytes();
            output_buffer[i * 4..i * 4 + 4].copy_from_slice(&bytes);
        }
        
        writer.write_all(&output_buffer)?;
    }
    
    writer.flush()?;
    Ok(())
}

/// Small dataset strategy: load all into memory
fn cluster_small(num_points: usize, k: usize) -> io::Result<()> {
    let mut reader = BufReader::new(io::stdin());
    let mut writer = BufWriter::new(io::stdout());
    
    // Read all points
    let mut points = Vec::with_capacity(num_points);
    let mut buffer = Vec::with_capacity(BATCH_SIZE);
    
    while points.len() < num_points {
        let count = read_point_batch(&mut reader, &mut buffer)?;
        if count == 0 {
            break;
        }
        points.extend(buffer.iter().take(count));
    }
    
    // Run Lloyd's algorithm
    let assignments = lloyd_kmeans(&points, k, 100);
    
    // Write output
    let mut output_buffer = vec![0u8; assignments.len() * 4];
    for (i, &assignment) in assignments.iter().enumerate() {
        let bytes = (assignment as u32).to_le_bytes();
        output_buffer[i * 4..i * 4 + 4].copy_from_slice(&bytes);
    }
    writer.write_all(&output_buffer)?;
    writer.flush()?;
    
    Ok(())
}

/// Medium/Large dataset strategy: streaming mini-batch k-means
fn cluster_streaming(num_points: usize, k: usize) -> io::Result<()> {
    let mut reader = BufReader::new(io::stdin());
    let mut writer = BufWriter::new(io::stdout());
    
    // Determine number of passes based on dataset size
    let max_iterations = if num_points > LARGE_THRESHOLD {
        3
    } else if num_points > MEDIUM_THRESHOLD {
        10
    } else {
        20
    };
    
    // Phase 1: Learn centroids (this consumes stdin)
    let centroids = minibatch_kmeans_streaming(&mut reader, num_points, k, max_iterations)?;
    
    // Note: For true single-pass streaming, we'd need to store assignments during learning
    // or re-open the input. Since we're reading from stdin, we use a two-phase approach:
    // In production, you'd either: 
    // 1. Store assignments during learning (requires O(n) memory)
    // 2. Accept the limitation of a single pass with online assignments
    
    // For this challenge, we output assignments based on learned centroids
    // Since stdin is exhausted, we'll simulate by re-reading or using stored assignments
    
    // Actually, for a proper streaming solution, let's use a different approach:
    // Store all assignments during the centroid learning phase
    
    // Reinitialize and do a proper single-pass with assignment storage
    streaming_assign_with_storage(num_points, k, max_iterations)
}

/// Single-pass streaming with assignment storage for moderate datasets
fn streaming_assign_with_storage(num_points: usize, k: usize, max_iterations: usize) -> io::Result<()> {
    let mut reader = BufReader::new(io::stdin());
    let mut writer = BufWriter::new(io::stdout());
    
    let mut buffer = Vec::with_capacity(BATCH_SIZE);
    
    // Initialize centroids using first batch
    let initial_count = read_point_batch(&mut reader, &mut buffer)?;
    let initial_points: Vec<Point> = buffer.iter().take(initial_count).copied().collect();
    let mut centroids = initialize_centroids_kmeanspp(&initial_points, k.min(initial_points.len()));
    let actual_k = centroids.len();
    
    // Store initial assignments
    let mut all_assignments: Vec<u32> = Vec::with_capacity(num_points);
    for point in initial_points.iter() {
        all_assignments.push(find_nearest_centroid(point, &centroids) as u32);
    }
    
    // Per-centroid statistics
    let mut centroid_sums: Vec<Point> = centroids.clone();
    let mut centroid_counts: Vec<usize> = vec![1; actual_k];
    
    // Single-pass streaming update
    let mut total_processed = initial_count;
    
    loop {
        let count = read_point_batch(&mut reader, &mut buffer)?;
        if count == 0 {
            break;
        }
        
        for point in buffer.iter().take(count) {
            let nearest = find_nearest_centroid(point, &centroids);
            all_assignments.push(nearest as u32);
            
            // Update centroid incrementally
            centroid_sums[nearest].x += point.x;
            centroid_sums[nearest].y += point.y;
            centroid_sums[nearest].z += point.z;
            centroid_counts[nearest] += 1;
            
            total_processed += 1;
            
            // Periodically update centroids
            if total_processed % BATCH_SIZE == 0 {
                for i in 0..actual_k {
                    if centroid_counts[i] > 0 {
                        centroids[i] = centroid_sums[i].scale(1.0 / centroid_counts[i] as f64);
                    }
                }
            }
        }
    }
    
    // Final centroid update
    for i in 0..actual_k {
        if centroid_counts[i] > 0 {
            centroids[i] = centroid_sums[i].scale(1.0 / centroid_counts[i] as f64);
        }
    }
    
    // Reassign based on final centroids (optional refinement)
    if num_points <= MEDIUM_THRESHOLD {
        // Refine assignments with final centroids
        let mut idx = 0;
        let points_per_batch = BATCH_SIZE;
        
        // We need to re-read the input for refinement - but stdin is consumed
        // So we skip refinement for now and use the streaming assignments
    }
    
    // Write output
    let mut output_buffer = vec![0u8; all_assignments.len() * 4];
    for (i, &assignment) in all_assignments.iter().enumerate() {
        let bytes = assignment.to_le_bytes();
        output_buffer[i * 4..i * 4 + 4].copy_from_slice(&bytes);
    }
    writer.write_all(&output_buffer)?;
    writer.flush()?;
    
    Ok(())
}

/// Very large dataset: streaming with minimal memory, no assignment storage
fn cluster_very_large_streaming(num_points: usize, k: usize) -> io::Result<()> {
    let mut reader = BufReader::new(io::stdin());
    let mut writer = BufWriter::new(io::stdout());
    
    let mut buffer = Vec::with_capacity(BATCH_SIZE);
    
    // Read initial batch for centroid initialization
    let initial_count = read_point_batch(&mut reader, &mut buffer)?;
    let initial_points: Vec<Point> = buffer.iter().take(initial_count).copied().collect();
    let mut centroids = initialize_centroids_kmeanspp(&initial_points, k.min(initial_points.len()));
    let actual_k = centroids.len();
    
    // Centroid statistics
    let mut centroid_sums: Vec<Point> = centroids.clone();
    let mut centroid_counts: Vec<usize> = vec![1; actual_k];
    
    // Output initial assignments
    let mut output_batch = vec![0u8; initial_count * 4];
    for (i, point) in initial_points.iter().enumerate() {
        let nearest = find_nearest_centroid(point, &centroids);
        output_batch[i * 4..i * 4 + 4].copy_from_slice(&(nearest as u32).to_le_bytes());
    }
    writer.write_all(&output_batch)?;
    
    // Process rest of stream
    loop {
        let count = read_point_batch(&mut reader, &mut buffer)?;
        if count == 0 {
            break;
        }
        
        let mut output_batch = vec![0u8; count * 4];
        
        for (i, point) in buffer.iter().take(count).enumerate() {
            let nearest = find_nearest_centroid(point, &centroids);
            output_batch[i * 4..i * 4 + 4].copy_from_slice(&(nearest as u32).to_le_bytes());
            
            // Update centroid incrementally
            centroid_sums[nearest].x += point.x;
            centroid_sums[nearest].y += point.y;
            centroid_sums[nearest].z += point.z;
            centroid_counts[nearest] += 1;
        }
        
        writer.write_all(&output_batch)?;
        
        // Update centroids periodically
        for i in 0..actual_k {
            if centroid_counts[i] > 0 {
                centroids[i] = centroid_sums[i].scale(1.0 / centroid_counts[i] as f64);
            }
        }
    }
    
    writer.flush()?;
    Ok(())
}

fn main() -> io::Result<()> {
    let start = Instant::now();
    
    // Read header
    let mut header = String::new();
    let stdin = io::stdin();
    let mut handle = stdin.lock();
    
    // Read until newline for header
    let mut byte = [0u8; 1];
    loop {
        match handle.read(&mut byte)? {
            0 => break,
            _ => {
                if byte[0] == b'\n' {
                    break;
                }
                header.push(byte[0] as char);
            }
        }
    }
    
    let parts: Vec<&str> = header.trim().split_whitespace().collect();
    if parts.len() < 2 {
        eprintln!("Error: Invalid header format");
        return Err(io::Error::new(io::ErrorKind::InvalidData, "Invalid header"));
    }
    
    let num_points: usize = parts[0].parse().unwrap_or(0);
    let num_clusters: usize = parts[1].parse().unwrap_or(0);
    
    if num_clusters == 0 {
        eprintln!("Error: Number of clusters must be > 0");
        return Err(io::Error::new(io::ErrorKind::InvalidData, "Invalid cluster count"));
    }
    
    if num_points == 0 {
        return Ok(());
    }
    
    // Choose strategy based on dataset size
    let result = if num_points <= SMALL_THRESHOLD {
        // Small dataset: load all into memory
        cluster_small(num_points, num_clusters)
    } else if num_points <= MEDIUM_THRESHOLD {
        // Medium dataset: streaming with assignment storage
        streaming_assign_with_storage(num_points, num_clusters, 20)
    } else {
        // Large/Very large dataset: pure streaming
        cluster_very_large_streaming(num_points, num_clusters)
    };
    
    match result {
        Ok(_) => {
            eprintln!("Clustering completed in {:?}", start.elapsed());
        }
        Err(e) => {
            eprintln!("Error: {}", e);
            return Err(e);
        }
    }
    
    Ok(())
}
```

## Algorithm Explanation

This solution implements **adaptive k-means clustering** that scales from millions to billions of points:

### Strategy Selection

| Dataset Size | Strategy | Memory Usage |
|-------------|----------|--------------|
| ≤10M points | In-memory Lloyd's k-means | O(n + k) |
| 10M-100M | Streaming with assignment storage | O(n + k) |
| >100M | Pure streaming | O(k + batch) |

### Key Techniques

1. **K-means++ initialization**: Provides better initial centroids, reducing convergence iterations

2. **Streaming updates**: Uses incremental centroid averaging:
   $$\mu_i \leftarrow \mu_i + \frac{1}{n_i}(x - \mu_i)$$

3. **Binary I/O**: Efficient reading/writing using `BufReader`/`BufWriter` with batch processing

4. **Memory efficiency**: 
   - Small: Store all points + assignments
   - Medium: Store only assignments
   - Large: Output directly, no storage

5. **Distance optimization**: Uses squared distance to avoid sqrt operations

### Performance Characteristics

- **Time**: O(n·k·i) where i = iterations (bounded by dataset size)
- **Space**: O(k + batch_size) for large datasets
- **Single-pass** for large datasets to minimize I/O

### Edge Cases Handled

- Empty clusters (centroids persist)
- k > n (uses min(k, n) clusters)
- Zero points (graceful exit)
- Streaming convergence (periodic centroid updates)
