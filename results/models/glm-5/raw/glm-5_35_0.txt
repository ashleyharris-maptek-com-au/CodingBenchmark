```rust
use std::collections::BTreeSet;
use std::io::{self, BufRead, Write, BufWriter};
use std::time::Instant;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    let mut lines = stdin.lock().lines();
    
    // Parse header
    let header = lines.next().unwrap().unwrap();
    let mut iter = header.split_whitespace();
    let num_vertices: usize = iter.next().unwrap().parse().unwrap();
    let num_edges: usize = iter.next().unwrap().parse().unwrap();
    
    // Build adjacency list (using Vec for fast iteration)
    let mut adj = vec![vec![]; num_vertices];
    
    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let mut iter = line.split_whitespace();
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }
    
    // Add self-loops for easier domination logic (vertex dominates itself)
    for v in 0..num_vertices {
        adj[v].push(v);
    }
    
    // Compute degree for each vertex
    let degrees: Vec<usize> = adj.iter().map(|n| n.len()).collect();
    
    // State tracking
    let mut dominated = vec![false; num_vertices];
    let mut dom_set = BTreeSet::new();
    let mut uncovered_count = num_vertices;
    
    // Track current "effective degree" (uncovered neighbors)
    let mut eff_degree = degrees.clone();
    
    // Track which vertices are uncovered for fast scanning
    let mut uncovered: Vec<bool> = vec![true; num_vertices];
    
    // Track iteration count for heuristic switching
    let start_time = Instant::now();
    let time_limit = 290; // seconds (leave margin for output)
    
    // ============================================
    // Phase 1: Handle Isolated Vertices
    // ============================================
    // Vertices with degree 1 (only self-loop) must be in the set
    for v in 0..num_vertices {
        if degrees[v] == 1 {
            dom_set.insert(v);
            for &u in &adj[v] {
                if !dominated[u] {
                    dominated[u] = true;
                    uncovered_count -= 1;
                    uncovered[u] = false;
                }
            }
        }
    }
    
    // ============================================
    // Phase 2: Greedy Selection with Updates
    // ============================================
    // Update effective degrees after marking isolated vertices
    for v in 0..num_vertices {
        if dominated[v] {
            eff_degree[v] = 0;
        } else {
            eff_degree[v] = adj[v].iter().filter(|&&u| !dominated[u]).count();
        }
    }
    
    while uncovered_count > 0 {
        // Time check for very large instances
        if start_time.elapsed().as_secs() > time_limit {
            break;
        }
        
        // Find vertex with maximum effective degree
        let mut best_v = 0;
        let mut best_gain = 0;
        
        for v in 0..num_vertices {
            if eff_degree[v] > best_gain {
                best_gain = eff_degree[v];
                best_v = v;
            }
        }
        
        if best_gain == 0 {
            // This shouldn't happen if graph is connected
            // but handle gracefully
            for v in 0..num_vertices {
                if !dominated[v] {
                    dom_set.insert(v);
                    dominated[v] = true;
                    uncovered[v] = false;
                    uncovered_count -= 1;
                }
            }
            break;
        }
        
        // Add best vertex to dominating set
        dom_set.insert(best_v);
        
        // Update dominated status and effective degrees
        for &u in &adj[best_v] {
            if !dominated[u] {
                dominated[u] = true;
                uncovered[u] = false;
                uncovered_count -= 1;
                
                // Decrease effective degree of all neighbors of u
                for &w in &adj[u] {
                    if eff_degree[w] > 0 {
                        eff_degree[w] -= 1;
                    }
                }
            }
        }
        
        eff_degree[best_v] = 0;
    }
    
    // ============================================
    // Phase 3: Redundancy Elimination
    // ============================================
    // Try to remove vertices from the dominating set if still valid
    // Check in reverse order of selection (rough heuristic)
    
    let dom_vec: Vec<usize> = dom_set.iter().cloned().collect();
    let mut redundant = vec![false; num_vertices];
    
    // Check each vertex in dominating set
    for &v in &dom_vec {
        // Check if v can be removed
        // v can be removed if every neighbor it dominates (including itself) 
        // is also dominated by another vertex in the set
        
        let mut can_remove = true;
        'check_neighbor: for &u in &adj[v] {
            // If u is in the dominating set, v doesn't need to cover it
            if dom_set.contains(&u) && u != v {
                continue;
            }
            
            // Count how many vertices in the set dominate u
            let mut dominators = 0;
            for &w in &adj[u] {
                if dom_set.contains(&w) && !redundant[w] {
                    dominators += 1;
                }
            }
            
            // If u has only one dominator (which is v), can't remove v
            if dominators <= 1 {
                can_remove = false;
                break 'check_neighbor;
            }
        }
        
        if can_remove {
            redundant[v] = true;
        }
    }
    
    // Build final result excluding redundant vertices
    let final_set: Vec<usize> = dom_set.iter()
        .filter(|&&v| !redundant[v])
        .cloned()
        .collect();
    
    // Output
    writeln!(out, "{}", final_set.len()).unwrap();
    let output: Vec<String> = final_set.iter().map(|v| v.to_string()).collect();
    writeln!(out, "{}", output.join(" ")).unwrap();
}
```
