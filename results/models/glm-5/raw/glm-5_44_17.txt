```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Constants for brick pattern
    vec2 brickCount = vec2(8.0, 16.0);
    vec2 mortarThickness = vec2(0.05, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 brickColor = vec3(0.7, 0.2, 0.1);

    // Scale UVs
    vec2 scaledUV = uv * brickCount;

    // Staggered rows: offset every other row by half a brick
    if (mod(floor(scaledUV.y), 2.0) == 1.0) {
        scaledUV.x += 0.5;
    }

    // Determine brick/mortar regions using nearest neighbor logic
    vec2 brickPos = fract(scaledUV);
    vec2 edge = step(brickPos, mortarThickness) + step(1.0 - mortarThickness, brickPos);
    float isMortar = max(edge.x, edge.y);

    // Color selection
    vec3 baseColor = mix(brickColor, mortarColor, isMortar);

    // Lighting calculations
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    float NdotL = max(dot(N, L), 0.0);
    float shadingFactor = 0.4 + 0.6 * NdotL;

    vec3 finalColor = baseColor * shadingFactor;

    outColor = vec4(finalColor, 1.0);
}
```
