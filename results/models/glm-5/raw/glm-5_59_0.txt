// Simple A* pathfinding to find path to a target
function findPath(sx, sy, tx, ty, startTime, agentIdx) {
    const maxTime = maxStepsPerDrone;
    
    // Priority queue: [fScore, time, x, y, path]
    const openSet = [];
    const openMap = new Map();
    
    // Heuristic: Manhattan distance
    function heuristic(x, y) {
        return Math.abs(x - tx) + Math.abs(y - ty);
    }
    
    const startH = heuristic(sx, sy);
    openSet.push({ f: startH, g: 0, x: sx, y: sy, t: startTime, path: [[sx, sy]] });
    openMap.set(`${sx},${sy},${startTime}`, 0);
    
    const closedSet = new Set();
    
    while (openSet.length > 0) {
        // Find minimum f score
        let minIdx = 0;
        for (let i = 1; i < openSet.length; i++) {
            if (openSet[i].f < openSet[minIdx].f || 
                (openSet[i].f === openSet[minIdx].f && openSet[i].g > openSet[minIdx].g)) {
                minIdx = i;
            }
        }
        
        const current = openSet[minIdx];
        openSet.splice(minIdx, 1);
        
        const { x, y, t, g, path } = current;
        const stateKey = `${x},${y},${t}`;
        
        if (closedSet.has(stateKey)) continue;
        closedSet.add(stateKey);
        
        // Check if reached target
        if (x === tx && y === ty && t > startTime) {
            return path.slice(1); // Return path without start position
        }
        
        // Check time limit
        if (t >= maxTime - 1) continue;
        
        // Explore neighbors
        for (let d = 0; d < 5; d++) {
            const nx = x + DX[d];
            const ny = y + DY[d];
            const nt = t + 1;
            
            if (!isValidCell(nx, ny)) continue;
            
            // Check collision with other drones at time nt
            let collision = false;
            for (let i = 0; i < paths.length; i++) {
                if (i === agentIdx) continue;
                const otherPath = paths[i];
                if (otherPath.length > nt) {
                    const otherPos = otherPath[nt];
                    if (otherPos[0] === nx && otherPos[1] === ny) {
                        collision = true;
                        break;
                    }
                }
            }
            if (collision) continue;
            
            const nkey = `${nx},${ny},${nt}`;
            if (closedSet.has(nkey)) continue;
            
            const ng = g + 1;
            const nh = heuristic(nx, ny);
            const nf = ng + nh;
            
            const existing = openMap.get(nkey);
            if (existing !== undefined && openSet[existing] && openSet[existing].g <= ng) {
                continue;
            }
            
            const newPath = [...path, [nx, ny]];
            openSet.push({ f: nf, g: ng, x: nx, y: ny, t: nt, path: newPath });
            openMap.set(nkey, openSet.length - 1);
        }
    }
    
    return null; // No path found
}

// Find path avoiding obstacles only (simpler version)
function findSimplePath(sx, sy, tx, ty) {
    const queue = [[sx, sy, [[sx, sy]]]];
    const visitedBFS = new Set();
    visitedBFS.add(`${sx},${sy}`);
    
    while (queue.length > 0) {
        const [x, y, path] = queue.shift();
        
        if (x === tx && y === ty && path.length > 1) {
            return path.slice(1);
        }
        
        if (path.length >= maxStepsPerDrone) continue;
        
        for (let d = 1; d < 5; d++) { // Skip stay for BFS
            const nx = x + DX[d];
            const ny = y + DY[d];
            
            if (!isValidCell(nx, ny)) continue;
            if (visitedBFS.has(`${nx},${ny}`)) continue;
            
            visitedBFS.add(`${nx},${ny}`);
            queue.push([nx, ny, [...path, [nx, ny]]]);
        }
    }
    
    return null;
}

// BFS to find nearest unvisited valuable cell
function findNearestTarget(sx, sy, visited, maxDist) {
    const queue = [[sx, sy, 0]];
    const bfsVisited = new Set();
    bfsVisited.add(`${sx},${sy}`);
    
    let bestTarget = null;
    let bestValue = -1;
    let bestDist = Infinity;
    
    // Also collect candidates for batch evaluation
    const candidates = [];
    
    while (queue.length > 0) {
        const [x, y, dist] = queue.shift();
        
        if (dist > maxDist) continue;
        
        // Check if this cell is valuable and unvisited
        if (!visited[y][x] && probability[y][x] > 0) {
            candidates.push({ x, y, value: probability[y][x], dist });
        }
        
        // Explore neighbors
        for (let d = 1; d < 5; d++) {
            const nx = x + DX[d];
            const ny = y + DY[d];
            
            if (!isValidCell(nx, ny)) continue;
            const key = `${nx},${ny}`;
            if (bfsVisited.has(key)) continue;
            
            bfsVisited.add(key);
            queue.push([nx, ny, dist + 1]);
        }
    }
    
    // Find best candidate considering value and distance
    for (const cand of candidates) {
        // Score: value minus distance penalty
        const score = cand.value - cand.dist * 0.1;
        if (bestTarget === null || score > bestValue - bestDist * 0.1) {
            bestTarget = cand;
            bestValue = cand.value;
            bestDist = cand.dist;
        }
    }
    
    return bestTarget;
}

// Advanced search using cooperative multi-agent path finding concepts
function runCooperativeSearch() {
    const numDrones = drones.length;
    const remainingSteps = new Array(numDrones).fill(maxStepsPerDrone - 1);
    
    // Reserve initial positions for all time steps
    const reservations = new Map();
    for (let t = 0; t <= maxStepsPerDrone; t++) {
        for (let i = 0; i < numDrones; i++) {
            if (t === 0) {
                reservations.set(`${drones[i].x},${drones[i].y},${t}`, i);
            }
        }
    }
    
    // Main loop for each time step
    for (let step = 1; step < maxStepsPerDrone; step++) {
        // Plan moves for all drones
        const plannedMoves = new Array(numDrones);
        
        for (let i = 0; i < numDrones; i++) {
            const cx = currentX[i];
            const cy = currentY[i];
            
            // Find valid moves
            const validMoves = [];
            for (let d = 0; d < 5; d++) {
                const nx = cx + DX[d];
                const ny = cy + DY[d];
                
                if (!isValidCell(nx, ny)) continue;
                
                // Check reservation
                const resKey = `${nx},${ny},${step}`;
                if (reservations.has(resKey)) continue;
                
                // Calculate score
                let score = 0;
                if (!visited[ny][nx]) {
                    score += probability[ny][nx] * 10;
                }
                
                // Add exploration bonus - prefer moving toward unexplored areas
                let nearbyUnexplored = 0;
                for (let d2 = 1; d2 < 5; d2++) {
                    const nx2 = nx + DX[d2];
                    const ny2 = ny + DY[d2];
                    if (isValidCell(nx2, ny2) && !visited[ny2][nx2]) {
                        nearbyUnexplored += probability[ny2][nx2];
                    }
                }
                score += nearbyUnexplored * 0.5;
                
                // Small penalty for staying in place
                if (d === 0) score -= 0.1;
                
                validMoves.push({ d, nx, ny, score });
            }
            
            // Sort by score descending
            validMoves.sort((a, b) => b.score - a.score);
            
            // Select best valid move
            if (validMoves.length > 0) {
                plannedMoves[i] = validMoves[0];
            } else {
                // Stay in place
                plannedMoves[i] = { d: 0, nx: cx, ny: cy, score: -1 };
            }
        }
        
        // Resolve conflicts - drones trying to move to same cell
        const targetCells = new Map();
        for (let i = 0; i < numDrones; i++) {
            const key = `${plannedMoves[i].nx},${plannedMoves[i].ny}`;
            if (!targetCells.has(key)) {
                targetCells.set(key, []);
            }
            targetCells.get(key).push(i);
        }
        
        // For conflicts, assign to highest scoring drone, others re-plan
        const finalMoves = new Array(numDrones);
        const conflictSet = new Set();
        
        for (const [key, droneList] of targetCells) {
            if (droneList.length === 1) {
                finalMoves[droneList[0]] = plannedMoves[droneList[0]];
            } else {
                // Sort by score
                droneList.sort((a, b) => plannedMoves[b].score - plannedMoves[a].score);
                finalMoves[droneList[0]] = plannedMoves[droneList[0]];
                for (let j = 1; j < droneList.length; j++) {
                    conflictSet.add(droneList[j]);
                }
            }
        }
        
        // Re-plan for conflicted drones
        for (const i of conflictSet) {
            const cx = currentX[i];
            const cy = currentY[i];
            
            // Find alternative move
            let bestAlt = { d: 0, nx: cx, ny: cy, score: -1000 };
            
            for (let d = 0; d < 5; d++) {
                const nx = cx + DX[d];
                const ny = cy + DY[d];
                
                if (!isValidCell(nx, ny)) continue;
                
                const resKey = `${nx},${ny},${step}`;
                if (reservations.has(resKey)) continue;
                
                // Check if already taken by another drone this step
                let taken = false;
                for (let j = 0; j < numDrones; j++) {
                    if (j !== i && finalMoves[j] && finalMoves[j].nx === nx && finalMoves[j].ny === ny) {
                        taken = true;
                        break;
                    }
                }
                if (taken) continue;
                
                let score = 0;
                if (!visited[ny][nx]) {
                    score += probability[ny][nx] * 10;
                }
                if (d === 0) score -= 0.1;
                
                if (score > bestAlt.score) {
                    bestAlt = { d, nx, ny, score };
                }
            }
            
            finalMoves[i] = bestAlt;
        }
        
        // Apply moves
        for (let i = 0; i < numDrones; i++) {
            const move = finalMoves[i];
            paths[i].push([move.nx, move.ny]);
            currentX[i] = move.nx;
            currentY[i] = move.ny;
            visited[move.ny][move.nx] = 1;
            reservations.set(`${move.nx},${move.ny},${step}`, i);
        }
    }
}

// Alternative: Use greedy search with BFS for targets
function runGreedySearch() {
    const numDrones = drones.length;
    
    for (let step = 1; step < maxStepsPerDrone; step++) {
        for (let i = 0; i < numDrones; i++) {
            const cx = currentX[i];
            const cy = currentY[i];
            
            // Find best immediate move
            let bestMove = { d: 0, nx: cx, ny: cy, score: -Infinity };
            
            for (let d = 0; d < 5; d++) {
                const nx = cx + DX[d];
                const ny = cy + DY[d];
                
                if (!isValidCell(nx, ny)) continue;
                
                // Check collision with other drones
                let collision = false;
                for (let j = 0; j < numDrones; j++) {
                    if (j === i) continue;
                    // Check current position of other drones
                    if (currentX[j] === nx && currentY[j] === ny) {
                        // Other drone is there, but might move
                        // Simple check: avoid if other drone is staying
                        if (paths[j].length > step && 
                            paths[j][step][0] === currentX[j] && 
                            paths[j][step][1] === currentY[j]) {
                            collision = true;
                        }
                    }
                    // Check if other drone is moving to this position
                    if (paths[j].length >= step) {
                        const prevX = paths[j][step - 1][0];
                        const prevY = paths[j][step - 1][1];
                        if (prevX === nx && prevY === ny && currentX[j] === cx && currentY[j] === cy && d > 0) {
                            // Swap collision
                            collision = true;
                        }
                    }
                }
                if (collision) continue;
                
                let score = 0;
                if (!visited[ny][nx] && probability[ny][nx] > 0) {
                    score = probability[ny][nx] * 10;
                }
                
                // Exploration bonus
                for (let d2 = 1; d2 < 5; d2++) {
                    const nx2 = nx + DX[d2];
                    const ny2 = ny + DY[d2];
                    if (isValidCell(nx2, ny2) && !visited[ny2][nx2]) {
                        score += probability[ny2][nx2] * 0.5;
                    }
                }
                
                // Random tie-breaking
                score += Math.random() * 0.01;
                
                if (score > bestMove.score) {
                    bestMove = { d, nx, ny, score };
                }
            }
            
            paths[i].push([bestMove.nx, bestMove.ny]);
            currentX[i] = bestMove.nx;
            currentY[i] = bestMove.ny;
            visited[bestMove.ny][bestMove.nx] = 1;
        }
    }
}
