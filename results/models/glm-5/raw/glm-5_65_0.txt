declare const require: any;
const fs = require('fs');

// Read input from stdin
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Extract inputs
const machineCount: number = data.machineCount;
const tasksInput: Array<{ id: string, duration: number, deps: string[], deadline?: number, weight?: number }> = data.tasks;

// Types
interface Task {
    id: string;
    duration: number;
    deps: Set<string>;
    deadline: number;
    weight: number;
    rank: number;
    earliestStart: number;
    processedDeps: number;
}

// Initialize tasks map
const tasks: Map<string, Task> = new Map();
const taskIds: string[] = [];

for (const t of tasksInput) {
    const task: Task = {
        id: t.id,
        duration: t.duration || 1,
        deps: new Set(t.deps || []),
        deadline: t.deadline !== undefined ? t.deadline : 1000000,
        weight: t.weight !== undefined ? t.weight : 1,
        rank: 0,
        earliestStart: 0,
        processedDeps: 0
    };
    tasks.set(t.id, task);
    taskIds.push(t.id);
}

// Build reverse dependency graph and count dependencies
const reverseDeps: Map<string, string[]> = new Map();
const depCount: Map<string, number> = new Map();

for (const id of taskIds) {
    reverseDeps.set(id, []);
    depCount.set(id, 0);
}

for (const id of taskIds) {
    const task = tasks.get(id)!;
    for (const dep of task.deps) {
        reverseDeps.get(dep)!.push(id);
    }
    depCount.set(id, task.deps.size);
}

// Calculate ranks using critical path (longest path to end)
// We need to process in reverse topological order
const reverseTopoOrder: string[] = [];
const queue: string[] = [];

// Find tasks with no successors (no one depends on them)
const hasSuccessor: Set<string> = new Set();
for (const id of taskIds) {
    for (const dep of tasks.get(id)!.deps) {
        hasSuccessor.add(dep);
    }
}

for (const id of taskIds) {
    if (!hasSuccessor.has(id)) {
        queue.push(id);
    }
}

const visitedReverse: Set<string> = new Set();
while (queue.length > 0) {
    const current = queue.shift()!;
    reverseTopoOrder.push(current);
    visitedReverse.add(current);
    
    for (const dep of tasks.get(current)!.deps) {
        const revDeps = reverseDeps.get(dep)!;
        let allProcessed = true;
        for (const succ of revDeps) {
            if (!visitedReverse.has(succ)) {
                allProcessed = false;
                break;
            }
        }
        // Check if all successors of 'dep' are visited
        // Actually, we need to check if all nodes that depend on 'dep' are visited
        // Let's use a counter approach
    }
}

// Better approach: use Kahn's algorithm in reverse
// Rebuild: count successors
const successorCount: Map<string, number> = new Map();
for (const id of taskIds) {
    successorCount.set(id, 0);
}

for (const id of taskIds) {
    for (const dep of tasks.get(id)!.deps) {
        successorCount.set(dep, successorCount.get(dep)! + 1);
    }
}

const reverseQueue: string[] = [];
for (const id of taskIds) {
    if (successorCount.get(id) === 0) {
        reverseQueue.push(id);
    }
}

const reverseOrder: string[] = [];
while (reverseQueue.length > 0) {
    const current = reverseQueue.shift()!;
    reverseOrder.push(current);
    
    for (const dep of tasks.get(current)!.deps) {
        const newCount = successorCount.get(dep)! - 1;
        successorCount.set(dep, newCount);
        if (newCount === 0) {
            reverseQueue.push(dep);
        }
    }
}

// Calculate ranks (longest path from task to end)
for (const id of reverseOrder) {
    const task = tasks.get(id)!;
    const successors = reverseDeps.get(id)!;
    
    if (successors.length === 0) {
        task.rank = task.duration;
    } else {
        let maxSuccRank = 0;
        for (const succ of successors) {
            const succTask = tasks.get(succ)!;
            if (succTask.rank > maxSuccRank) {
                maxSuccRank = succTask.rank;
            }
        }
        task.rank = task.duration + maxSuccRank;
    }
}

// Topological sort using critical path rank as priority
// Priority queue implementation (max-heap by rank)
interface HeapNode {
    id: string;
    rank: number;
}

class MaxHeap {
    private heap: HeapNode[] = [];
    
    push(node: HeapNode): void {
        this.heap.push(node);
        this.bubbleUp(this.heap.length - 1);
    }
    
    pop(): HeapNode | undefined {
        if (this.heap.length === 0) return undefined;
        const top = this.heap[0];
        const last = this.heap.pop()!;
        if (this.heap.length > 0) {
            this.heap[0] = last;
            this.bubbleDown(0);
        }
        return top;
    }
    
    get length(): number {
        return this.heap.length;
    }
    
    private bubbleUp(index: number): void {
        while (index > 0) {
            const parent = Math.floor((index - 1) / 2);
            if (this.heap[parent].rank >= this.heap[index].rank) break;
            [this.heap[parent], this.heap[index]] = [this.heap[index], this.heap[parent]];
            index = parent;
        }
    }
    
    private bubbleDown(index: number): void {
        const length = this.heap.length;
        while (true) {
            const left = 2 * index + 1;
            const right = 2 * index + 2;
            let largest = index;
            
            if (left < length && this.heap[left].rank > this.heap[largest].rank) {
                largest = left;
            }
            if (right < length && this.heap[right].rank > this.heap[largest].rank) {
                largest = right;
            }
            
            if (largest === index) break;
            [this.heap[largest], this.heap[index]] = [this.heap[index], this.heap[largest]];
            index = largest;
        }
    }
}

// Find all tasks with no dependencies (ready to start)
const ready: MaxHeap = new MaxHeap();
const order: string[] = [];
const completed: Set<string> = new Set();

// Initialize: find tasks with no dependencies
for (const id of taskIds) {
    if (depCount.get(id) === 0) {
        ready.push({ id, rank: tasks.get(id)!.rank });
    }
}

// Process tasks in topological order with priority
while (ready.length > 0) {
    const node = ready.pop()!;
    const id = node.id;
    
    order.push(id);
    completed.add(id);
    
    // Update dependents
    for (const dependent of reverseDeps.get(id)!) {
        const newCount = depCount.get(dependent)! - 1;
        depCount.set(dependent, newCount);
        
        if (newCount === 0) {
            ready.push({ id: dependent, rank: tasks.get(dependent)!.rank });
        }
    }
}

// Output the topological order
// The grader will use greedy list scheduling with this order
const output = {
    order: order
};

console.log(JSON.stringify(output));
