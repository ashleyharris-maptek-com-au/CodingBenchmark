```spirv
; SPIR-V
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04
%float_0_8 = OpConstant %float 0.8
%float_0_6 = OpConstant %float 0.6 ; For darker edge color
%float_pi = OpConstant %float 3.14159265
%float_2 = OpConstant %float 2
%float_6 = OpConstant %float 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_1 = OpConstant %uint 1

; Input Blocks
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User Inputs (Arrays of size 1 for InputPoints)
%arr_v3float_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_1
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

; Output Block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; UBO
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float ; model, view, proj, lightPos, cameraPos, params
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo_var = OpVariable %ptr_ubo Uniform
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16

; Pointers
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_undef_v3 = OpTypePointer Function %v3float

; Functions
%main = OpFunction %void None %void_func_type
%void_func_type = OpTypeFunction %void
%label_entry = OpLabel

; --- Read Inputs ---
; gl_in[0].Position
%ptr_in_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos_in = OpLoad %v4float %ptr_in_pos

; inColor[0]
%ptr_in_color = OpAccessChain %ptr_in_v3 %inColor %int_0
%color_base = OpLoad %v3float %ptr_in_color

; Prepare colors
%color_edge = OpVectorTimesScalar %v3float %color_base %float_0_8
%color_center = OpVectorTimesScalar %v3float %color_base %float_0_6

; Center position (XY offset base)
%center_xy = OpVectorShuffle %v2float %pos_in %pos_in 0 1

; --- Helper to emit vertex ---
; Expects %angle, %rgb_color on stack

; Loop unrolled for 6 triangles (segments)
; Angle calculations: angle = 2*pi/6 * i

; Variables for calculation
%two_pi_over_6 = OpFDiv %float %float_2 %float_6 ; Actually 2*pi/6
%angle_step = OpFMul %float %two_pi_over_6 %float_pi ; Wait, 2*pi/6 is just pi/3. 
; angle_step = 60 degrees.

; Let's compute angles manually for clarity in unrolled code.
; i=0: angle=0. a1=0, a2=60.
; i=1: angle=60. a1=60, a2=120.
; ...
; Cos/Sin constants needed? No, runtime compute is safer for SPIR-V assembly generation.

; Emit Triangle 0
; Vertex 0: Center
%ptr_out_pos_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos_0 %pos_in
OpStore %outColor %color_center
OpEmitVertex

; Vertex 1: Edge at angle 0
%angle_0 = OpFMul %float %float_0 %float_pi ; 0
%cos_0 = OpCos %float %angle_0
%sin_0 = OpSin %float %angle_0
%offset_0 = OpCompositeConstruct %v2float %cos_0 %sin_0
%off_s_0 = OpVectorTimesScalar %v2float %offset_0 %float_0_04
%pos_0_xy = OpFAdd %v2float %center_xy %off_s_0
%pos_0 = OpCompositeConstruct %v4float %pos_0_xy_0 %pos_0_xy_1 %float_0 %float_1 ; Z=0, W=1
; Wait, CompositeConstruct needs components.
%pos_0_x = OpCompositeExtract %float %pos_0_xy 0
%pos_0_y = OpCompositeExtract %float %pos_0_xy 1
%pos_0_final = OpCompositeConstruct %v4float %pos_0_x %pos_0_y %float_0 %float_1
%ptr_out_pos_1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos_1 %pos_0_final
OpStore %outColor %color_edge
OpEmitVertex

; Vertex 2: Edge at angle 60
%angle_1 = OpFMul %float %float_pi %float_0_6 ; pi * 0.6? No.
; angle = 1.047 (pi/3)
%angle_1_val = OpFDiv %float %float_pi %float_6 ; No.
; Let's use immediate values for steps.
; pi/3 approx 1.047. But we defined float_pi.
%angle_1_val = OpFDiv %float %float_pi %float_6 ; No.
; Let's calculate step.
%step = OpFDiv %float %float_pi %float_6 ; pi/6 = 30deg. No.
; 2*pi/6 = pi/3.
%step_angle = OpFDiv %float %float_pi %float_6 
; Wait. 2*pi / 6.
%step_angle = OpFMul %float %float_pi %float_0_6 ; No.
; Correct: step = 2*pi/6.
; float_pi is 3.14. 2*3.14/6 = 1.04.
; Let's just define constant for PI/3? No, derive.
%two_pi = OpFMul %float %float_2 %float_pi
%step_angle = OpFDiv %float %two_pi %float_6

; V2 angle: 1 * step
%angle_2_val = OpFMul %float %step_angle %float_1
%cos_2 = OpCos %float %angle_2_val
%sin_2 = OpSin %float %angle_2_val
%offset_2 = OpCompositeConstruct %v2float %cos_2 %sin_2
%off_s_2 = OpVectorTimesScalar %v2float %offset_2 %float_0_04
%pos_2_xy = OpFAdd %v2float %center_xy %off_s_2
%p2x = OpCompositeExtract %float %pos_2_xy 0
%p2y = OpCompositeExtract %float %pos_2_xy 1
%pos_2_final = OpCompositeConstruct %v4float %p2x %p2y %float_0 %float_1
%ptr_out_pos_2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos_2 %pos_2_final
OpStore %outColor %color_edge
OpEmitVertex
OpEndPrimitive

; Triangle 1
; Center
%ptr_out_p3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p3 %pos_in
OpStore %outColor %color_center
OpEmitVertex

; Edge 1 (reuse previous angle_2_val)
; Re-use %pos_2_final? Need to re-emit.
OpStore %ptr_out_p3 %pos_2_final ; Assume ptr is valid, yes.
; Actually we need a fresh store to Output.
%ptr_out_p4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p4 %pos_2_final
OpStore %outColor %color_edge
OpEmitVertex

; Edge 2 (angle 120)
%angle_3_val = OpFMul %float %step_angle %float_2
%cos_3 = OpCos %float %angle_3_val
%sin_3 = OpSin %float %angle_3_val
%off_3 = OpCompositeConstruct %v2float %cos_3 %sin_3
%off_s_3 = OpVectorTimesScalar %v2float %off_3 %float_0_04
%pos_3_xy = OpFAdd %v2float %center_xy %off_s_3
%p3x = OpCompositeExtract %float %pos_3_xy 0
%p3y = OpCompositeExtract %float %pos_3_xy 1
%pos_3_final = OpCompositeConstruct %v4float %p3x %p3y %float_0 %float_1
%ptr_out_p5 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p5 %pos_3_final
OpStore %outColor %color_edge
OpEmitVertex
OpEndPrimitive

; Triangle 2
%ptr_out_p6 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p6 %pos_in
OpStore %outColor %color_center
OpEmitVertex

OpStore %ptr_out_p6 %pos_3_final
OpStore %outColor %color_edge
OpEmitVertex

; Edge 3 (angle 180)
%angle_4_val = OpFMul %float %step_angle %float_3 ; 3 * pi/3 = pi
%cos_4 = OpCos %float %angle_4_val
%sin_4 = OpSin %float %angle_4_val
%off_4 = OpCompositeConstruct %v2float %cos_4 %sin_4
%off_s_4 = OpVectorTimesScalar %v2float %off_4 %float_0_04
%pos_4_xy = OpFAdd %v2float %center_xy %off_s_4
%p4x = OpCompositeExtract %float %pos_4_xy 0
%p4y = OpCompositeExtract %float %pos_4_xy 1
%pos_4_final = OpCompositeConstruct %v4float %p4x %p4y %float_0 %float_1
%ptr_out_p7 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p7 %pos_4_final
OpStore %outColor %color_edge
OpEmitVertex
OpEndPrimitive

; Triangle 3
%ptr_out_p8 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p8 %pos_in
OpStore %outColor %color_center
OpEmitVertex

OpStore %ptr_out_p8 %pos_4_final
OpStore %outColor %color_edge
OpEmitVertex

; Edge 4 (angle 240)
%angle_5_val = OpFMul %float %step_angle %float_4
%cos_5 = OpCos %float %angle_5_val
%sin_5 = OpSin %float %angle_5_val
%off_5 = OpCompositeConstruct %v2float %cos_5 %sin_5
%off_s_5 = OpVectorTimesScalar %v2float %off_5 %float_0_04
%pos_5_xy = OpFAdd %v2float %center_xy %off_s_5
%p5x = OpCompositeExtract %float %pos_5_xy 0
%p5y = OpCompositeExtract %float %pos_5_xy 1
%pos_5_final = OpCompositeConstruct %v4float %p5x %p5y %float_0 %float_1
%ptr_out_p9 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p9 %pos_5_final
OpStore %outColor %color_edge
OpEmitVertex
OpEndPrimitive

; Triangle 4
%ptr_out_p10 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p10 %pos_in
OpStore %outColor %color_center
OpEmitVertex

OpStore %ptr_out_p10 %pos_5_final
OpStore %outColor %color_edge
OpEmitVertex

; Edge 5 (angle 300)
%angle_6_val = OpFMul %float %step_angle %float_5
%cos_6 = OpCos %float %angle_6_val
%sin_6 = OpSin %float %angle_6_val
%off_6 = OpCompositeConstruct %v2float %cos_6 %sin_6
%off_s_6 = OpVectorTimesScalar %v2float %off_6 %float_0_04
%pos_6_xy = OpFAdd %v2float %center_xy %off_s_6
%p6x = OpCompositeExtract %float %pos_6_xy 0
%p6y = OpCompositeExtract %float %pos_6_xy 1
%pos_6_final = OpCompositeConstruct %v4float %p6x %p6y %float_0 %float_1
%ptr_out_p11 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p11 %pos_6_final
OpStore %outColor %color_edge
OpEmitVertex
OpEndPrimitive

; Triangle 5
%ptr_out_p12 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p12 %pos_in
OpStore %outColor %color_center
OpEmitVertex

OpStore %ptr_out_p12 %pos_6_final
OpStore %outColor %color_edge
OpEmitVertex

; Edge 0 again (angle 0) to close
%ptr_out_p13 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_p13 %pos_0_final ; Reuse calculated pos for angle 0
OpStore %outColor %color_edge
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
