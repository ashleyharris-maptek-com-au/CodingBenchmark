; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %ubo_block Block
OpMemberDecorate %ubo_block 0 Offset 0
OpMemberDecorate %ubo_block 1 Offset 64
OpMemberDecorate %ubo_block 2 Offset 128
OpMemberDecorate %ubo_block 3 Offset 192
OpMemberDecorate %ubo_block 4 Offset 208
OpMemberDecorate %ubo_block 5 Offset 224

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%ubo_block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %ubo_block

%in_worldPos = OpVariable %_ptr_Input_vec3 Input
%in_normal = OpVariable %_ptr_Input_vec3 Input
%in_uv = OpVariable %_ptr_Input_vec2 Input
%in_tangent = OpVariable %_ptr_Input_vec3 Input
%in_color = OpVariable %_ptr_Input_vec3 Input
%outColor = OpVariable %_ptr_Output_vec4 Output
%ubo = OpVariable %ptr_ubo Uniform

%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1

%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%normal_raw = OpLoad %vec3 %in_normal

; Normalize the interpolated normal
%N = OpExtInst %vec3 %1 Normalize %normal_raw

; Load lightPos (member 3, offset 192 = 3 * 64)
%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %int_3
%lightPos_vec4 = OpLoad %vec4 %lightPos_ptr
%lightPos = OpVectorShuffle %vec3 %lightPos_vec4 %lightPos_vec4 0 1 2

; Compute L = normalize(lightPos - worldPos)
%L_vec = OpFSub %vec3 %lightPos %worldPos
%L = OpExtInst %vec3 %1 Normalize %L_vec

; Compute dot(N, L)
%NdotL = OpDot %float %N %L

; Compute t = dot(N, L) * 0.5 + 0.5
%t_half = OpFMul %float %NdotL %float_0_5
%t = OpFAdd %float %t_half %float_0_5

; Cool color: (0.2, 0.2, 0.75)
%cool = OpCompositeConstruct %vec3 %float_0_2 %float_0_2 %float_0_75

; Warm color: (0.7, 0.7, 0.4)
%warm = OpCompositeConstruct %vec3 %float_0_7 %float_0_7 %float_0_4

; Gooch shading: (1-t)*cool + t*warm
%one_minus_t = OpFSub %float %float_1 %t
%cool_term = OpVectorTimesScalar %vec3 %cool %one_minus_t
%warm_term = OpVectorTimesScalar %vec3 %warm %t
%color = OpFAdd %vec3 %cool_term %warm_term

; Output with alpha 1.0
%finalColor = OpCompositeConstruct %vec4 %color %float_1
OpStore %outColor %finalColor

OpReturn
OpFunctionEnd

%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2

%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
