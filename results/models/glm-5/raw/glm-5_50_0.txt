```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint index = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-bounds invocation
    if (index >= N) {
        return;
    }

    // --- Read Current State ---
    uint baseIdx = index * 2;
    vec3 pos = inData[baseIdx].xyz;
    vec3 vel = inData[baseIdx + 1].xyz;

    // --- Parameters ---
    float dt = radii.x;
    float sepRad = radii.y;
    float alignRad = radii.z;
    float cohRad = radii.w;
    
    float sepW = weights.x;
    float alignW = weights.y;
    float cohW = weights.z;
    float maxSpeed = weights.w;
    
    float boundSize = bounds.x;
    float boundForce = bounds.y;

    // --- Accumulators ---
    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);
    
    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    // --- Neighbor Iteration ---
    for (uint j = 0; j < N; ++j) {
        // Skip self
        if (j == index) continue;

        uint jBase = j * 2;
        vec3 otherPos = inData[jBase].xyz;
        vec3 otherVel = inData[jBase + 1].xyz;

        vec3 diff = pos - otherPos;
        float dist = length(diff);

        // Skip zero-distance neighbors (protects divide-by-zero later)
        if (dist < 1e-5) continue;

        // Separation
        if (dist < sepRad) {
            // Normalize direction (diff/dist) and weight by 1/dist (closer = stronger)
            // Equivalent to diff / (dist * dist)
            sepForce += diff / (dist * dist);
            sepCount++;
        }

        // Alignment
        if (dist < alignRad) {
            alignForce += otherVel;
            alignCount++;
        }

        // Cohesion
        if (dist < cohRad) {
            cohForce += otherPos;
            cohCount++;
        }
    }

    // --- Calculate Acceleration ---
    vec3 acceleration = vec3(0.0);

    // Average separation
    if (sepCount > 0) {
        // Normalize the accumulated vector to get a direction,
        // scaled by weight. (Standard Reynolds separation steering).
        acceleration += normalize(sepForce) * sepW;
    }

    // Average alignment
    if (alignCount > 0) {
        alignForce /= alignCount;
        // Steering = desired - current
        acceleration += (alignForce - vel) * alignW;
    }

    // Average cohesion
    if (cohCount > 0) {
        cohForce /= cohCount;
        vec3 desired = cohForce - pos;
        // Steering = desired - current
        acceleration += (desired - vel) * cohW;
    }

    // --- Boundary Force ---
    // Boundary threshold: boundSize * 0.4
    float thresh = boundSize * 0.4;
    float halfBound = boundSize * 0.5;

    // Apply force for each component if it exceeds threshold
    // Push toward center (0,0,0) scaled by halfBound
    if (abs(pos.x) > thresh) {
        acceleration.x -= sign(pos.x) * boundForce * (abs(pos.x) - thresh) / halfBound;
    }
    if (abs(pos.y) > thresh) {
        acceleration.y -= sign(pos.y) * boundForce * (abs(pos.y) - thresh) / halfBound;
    }
    if (abs(pos.z) > thresh) {
        acceleration.z -= sign(pos.z) * boundForce * (abs(pos.z) - thresh) / halfBound;
    }

    // --- Integration ---
    vel += acceleration * dt;

    // --- Speed Clamp ---
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = (vel / speed) * maxSpeed;
    }

    pos += vel * dt;

    // --- Write Output ---
    outData[baseIdx] = vec4(pos, 0.0);
    outData[baseIdx + 1] = vec4(vel, 0.0);
}
```
