```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);
    
    // Calculate light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Lambertian diffuse
    float NdotL = max(dot(N, L), 0.0);
    
    // Shading factor: 0.5 + 0.5 * diffuse
    float shading = 0.5 + 0.5 * NdotL;
    
    // 8x8 UV checkerboard
    // Scale UVs by frequency
    vec2 checkerUV = uv * 8.0;
    
    // Determine parity using floor
    float parity = floor(checkerUV.x) + floor(checkerUV.y);
    
    // Alternate colors 0.9 and 0.2 based on parity
    float checker = mod(parity, 2.0) < 1.0 ? 0.2 : 0.9;
    
    // Apply shading to checker color
    float finalColor = checker * shading;
    
    outColor = vec4(finalColor, finalColor, finalColor, 1.0);
}
```
