```python
import sys
import math
import random
import time

def solve_cvrp():
    # 1. Read Input
    input_data = sys.stdin.read().split()
    if not input_data:
        return

    iterator = iter(input_data)
    try:
        N = int(next(iterator))
        K = int(next(iterator))
        CAPACITY = int(next(iterator))
        
        depot_x = float(next(iterator))
        depot_y = float(next(iterator))
        depot = (depot_x, depot_y, 0) # demand 0
        
        customers = []
        for i in range(N):
            x = float(next(iterator))
            y = float(next(iterator))
            d = int(next(iterator))
            customers.append((x, y, d))
    except StopIteration:
        return

    # 2. Pre-compute Distance Matrix
    # Locations: 0 is depot, 1..N are customers
    # Map internal 0..N to locations. 
    # Actually, let's stick to 0-based indexing for customers as requested in output.
    # distance_matrix[i][j] where i, j are customer indices 0..N-1.
    # dist_to_depot[i] distance from customer i to depot.
    
    points = [depot] + customers
    num_nodes = N + 1
    dist_matrix = [[0.0] * num_nodes for _ in range(num_nodes)]
    
    for i in range(num_nodes):
        for j in range(num_nodes):
            if i == j:
                dist_matrix[i][j] = 0.0
            else:
                dx = points[i][0] - points[j][0]
                dy = points[i][1] - points[j][1]
                dist_matrix[i][j] = math.sqrt(dx*dx + dy*dy)

    # Helper to calculate route cost
    def calculate_route_cost(route):
        if not route:
            return 0.0
        cost = 0.0
        # depot to first
        cost += dist_matrix[0][route[0] + 1]
        # between customers
        for i in range(len(route) - 1):
            cost += dist_matrix[route[i] + 1][route[i+1] + 1]
        # last to depot
        cost += dist_matrix[route[-1] + 1][0]
        return cost

    def calculate_demand(route):
        return sum(customers[i][2] for i in route)

    # 3. Initial Solution: Nearest Neighbor Heuristic
    def nearest_neighbor_init():
        routes = []
        visited = [False] * N
        visited_count = 0
        
        while visited_count < N:
            current_route = []
            current_load = 0
            current_loc = 0 # Start at depot (index 0 in points)
            
            while True:
                best_next = -1
                best_dist = float('inf')
                
                # Find nearest feasible customer
                for i in range(N):
                    if not visited[i]:
                        demand = customers[i][2]
                        if current_load + demand <= CAPACITY:
                            dist = dist_matrix[current_loc][i + 1]
                            if dist < best_dist:
                                best_dist = dist
                                best_next = i
                
                if best_next == -1:
                    break # No more customers fit in this truck
                
                current_route.append(best_next)
                current_load += customers[best_next][2]
                visited[best_next] = True
                visited_count += 1
                current_loc = best_next + 1
            
            routes.append(current_route)
            if len(routes) > K:
                # Infeasible start, but let's proceed, local search might fix it
                pass
        return routes

    # 4. Local Search (2-opt for each route)
    def two_opt(route):
        best_route = route
        improved = True
        while improved:
            improved = False
            for i in range(len(best_route) - 1):
                for j in range(i + 1, len(best_route)):
                    new_route = best_route[:i] + best_route[i:j+1][::-1] + best_route[j+1:]
                    if calculate_route_cost(new_route) < calculate_route_cost(best_route):
                        best_route = new_route
                        improved = True
                        break
                if improved:
                    break
        return best_route

    # Initial solution
    current_routes = nearest_neighbor_init()
    
    # Refine routes with 2-opt
    optimized_routes = [two_opt(r) for r in current_routes]
    
    # Calculate total distance
    total_distance = sum(calculate_route_cost(r) for r in optimized_routes)
    
    # 5. Output
    print("{:.10f}".format(total_distance))
    print(len(optimized_routes))
    for r in optimized_routes:
        print(" ".join(map(str, r)))

if __name__ == "__main__":
    solve_cvrp()
```
