; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%int = OpTypeInt 32 1
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%in_uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_4 = OpConstant %float 4
%float_neg1 = OpConstant %float -1
%float_neg2 = OpConstant %float -2
%float_0_5 = OpConstant %float 0.5
%float_20 = OpConstant %float 20
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20
%bool = OpTypeBool
%main = OpFunction %void None %func
%label = OpLabel
%uv = OpLoad %vec2 %in_uv
%uv_x = OpCompositeExtract %float %uv 0
%uv_y = OpCompositeExtract %float %uv 1
%cx_1 = OpFMul %float %uv_x %float_3
%cx = OpFAdd %float %cx_1 %float_neg2
%cy_1 = OpFMul %float %uv_y %float_2
%cy = OpFAdd %float %cy_1 %float_neg1
%zx = OpCopyObject %float %float_0
%zy = OpCopyObject %float %float_0
%i = OpCopyObject %int %int_0
OpBranch %loop_check
%loop_check = OpLabel
%loop_phi = OpPhi %float %zx %label %zx_next %loop_body
%loop_phi_zy = OpPhi %float %zy %label %zy_next %loop_body
%loop_phi_i = OpPhi %int %i %label %i_next %loop_body
%i_check = OpSLessThan %bool %loop_phi_i %int_20
OpLoopMerge %loop_end %loop_body None
OpBranchConditional %i_check %loop_body %loop_end
%loop_body = OpLabel
%zx2 = OpFMul %float %loop_phi %loop_phi
%zy2 = OpFMul %float %loop_phi_zy %loop_phi_zy
%mag2 = OpFAdd %float %zx2 %zy2
%esc_check = OpFGreaterThan %bool %mag2 %float_4
OpSelectionMerge %sel_end None
OpBranchConditional %esc_check %sel_break %sel_end
%sel_break = OpLabel
OpBranch %loop_end
%sel_end = OpLabel
%mul = OpFMul %float %loop_phi %loop_phi_zy
%two_mul = OpFMul %float %mul %float_2
%new_zy = OpFAdd %float %two_mul %cy
%sq_diff = OpFSub %float %zx2 %zy2
%new_zx = OpFAdd %float %sq_diff %cx
%zx_next = OpCopyObject %float %new_zx
%zy_next = OpCopyObject %float %new_zy
%i_next = OpIAdd %int %loop_phi_i %int_1
OpBranch %loop_check
%loop_end = OpLabel
%final_i = OpPhi %int %loop_phi_i %loop_check %loop_phi_i %sel_break
%iter_float = OpConvertSToF %float %final_i
%t = OpFDiv %float %iter_float %float_20
%color_g = OpFMul %float %t %float_0_5
%one_minus_t = OpFSub %float %float_1 %t
%color = OpCompositeConstruct %vec4 %t %color_g %one_minus_t %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
