; SPIR-V
; Version: 1.0
; Generator: Khronos Glslang Reference Front End
; Bound: 100

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 0
OpDecorate %outColor Location 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3

; PerVertex Input Struct
%gl_PerVertex_in = OpTypeStruct %v4float
%uint_1 = OpConstant %uint 1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; PerVertex Output Struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User Inputs
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%inColor = OpVariable %ptr_in_arr_v3 Input

; User Outputs
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Constants
%int_0 = OpConstant %int 0
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04
%float_n0_04 = OpConstant %float -0.04

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %3
%5 = OpLabel

; Access gl_in[0].gl_Position
%gl_in_0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%center_pos = OpLoad %v4float %gl_in_0_ptr

; Access inColor[0]
%color_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%base_color = OpLoad %v3float %color_ptr

; Extract x, y from center position
%cx = OpCompositeExtract %float %center_pos 0
%cy = OpCompositeExtract %float %center_pos 1

; Calculate offsets
%x_min = OpFAdd %float %cx %float_n0_04
%x_max = OpFAdd %float %cx %float_0_04
%y_min = OpFAdd %float %cy %float_n0_04
%y_max = OpFAdd %float %cy %float_0_04

; --- Vertex 0: (x-h, y-h) ---
%v0_pos = OpCompositeConstruct %v4float %x_min %y_min %float_0 %float_1
%out_pos_ptr_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr_0 %v0_pos
OpStore %outColor %base_color
OpEmitVertex

; --- Vertex 1: (x+h, y-h) ---
%v1_pos = OpCompositeConstruct %v4float %x_max %y_min %float_0 %float_1
%out_pos_ptr_1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr_1 %v1_pos
OpStore %outColor %base_color
OpEmitVertex

; --- Vertex 2: (x-h, y+h) ---
%v2_pos = OpCompositeConstruct %v4float %x_min %y_max %float_0 %float_1
%out_pos_ptr_2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr_2 %v2_pos
OpStore %outColor %base_color
OpEmitVertex

; --- Vertex 3: (x+h, y+h) ---
%v3_pos = OpCompositeConstruct %v4float %x_max %y_max %float_0 %float_1
%out_pos_ptr_3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr_3 %v3_pos
OpStore %outColor %base_color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
