#include <iostream>

using namespace std;

// --- Constants and Types ---

const int INF = 1000000;
const int MATE_SCORE = 100000;

enum PieceType : int {
    NONE = 0,
    PAWN = 1,
    KNIGHT = 2,
    BISHOP = 3,
    ROOK = 4,
    QUEEN = 5,
    KING = 6
};

enum Color : int {
    WHITE = 0,
    BLACK = 1
};

struct Piece {
    PieceType type = NONE;
    Color color = WHITE;
    
    Piece() = default;
    Piece(PieceType t, Color c) : type(t), color(c) {}
    
    bool operator==(const Piece& other) const {
        return type == other.type && color == other.color;
    }
    bool operator!=(const Piece& other) const {
        return !(*this == other);
    }
};

const Piece NO_PIECE(NONE, WHITE);

struct Move {
    int from = -1;
    int to = -1;
    PieceType promotion = NONE;
    bool is_castling = false;
    bool is_en_passant = false;
    Piece captured = NO_PIECE; // For unmake
};

// --- Board Representation ---

class Board {
public:
    Piece squares[64];
    Color side_to_move = WHITE;
    bool castle_rights[4] = {true, true, true, true}; // K, Q, k, q
    int en_passant_sq = -1;
    int halfmove_clock = 0;
    int fullmove_number = 1;
    
    // For unmake
    struct History {
        Move move;
        Piece captured;
        bool castle_rights[4];
        int en_passant_sq;
        int halfmove_clock;
    };
    vector<History> history;
    
    int king_sq[2] = {-1, -1};

    Board() {
        for(int i=0; i<64; ++i) squares[i] = NO_PIECE;
    }

    void load_fen(const string& fen) {
        fill(begin(squares), end(squares), NO_PIECE);
        history.clear();
        side_to_move = WHITE;
        fill(begin(castle_rights), end(castle_rights), false);
        en_passant_sq = -1;
        halfmove_clock = 0;
        fullmove_number = 1;
        king_sq[0] = -1;
        king_sq[1] = -1;

        stringstream ss(fen);
        string token;
        
        // 1. Piece placement
        ss >> token;
        int rank = 7, file = 0;
        for (char c : token) {
            if (c == '/') {
                rank--;
                file = 0;
            } else if (isdigit(c)) {
                file += c - '0';
            } else {
                int sq = rank * 8 + file;
                PieceType pt = NONE;
                Color col = isupper(c) ? WHITE : BLACK;
                char lc = tolower(c);
                switch (lc) {
                    case 'p': pt = PAWN; break;
                    case 'n': pt = KNIGHT; break;
                    case 'b': pt = BISHOP; break;
                    case 'r': pt = ROOK; break;
                    case 'q': pt = QUEEN; break;
                    case 'k': pt = KING; break;
                }
                squares[sq] = Piece(pt, col);
                if (pt == KING) king_sq[col] = sq;
                file++;
            }
        }
        
        // 2. Side to move
        ss >> token;
        if (token == "b") side_to_move = BLACK;
        
        // 3. Castling rights
        ss >> token;
        for (char c : token) {
            switch (c) {
                case 'K': castle_rights[0] = true; break;
                case 'Q': castle_rights[1] = true; break;
                case 'k': castle_rights[2] = true; break;
                case 'q': castle_rights[3] = true; break;
            }
        }
        
        // 4. En passant
        ss >> token;
        if (token != "-") {
            int f = token[0] - 'a';
            int r = token[1] - '1';
            en_passant_sq = r * 8 + f;
        }
        
        // 5. Halfmove clock
        ss >> token;
        if (!token.empty()) halfmove_clock = stoi(token);
        
        // 6. Fullmove number
        ss >> token;
        if (!token.empty()) fullmove_number = stoi(token);
    }
    
    bool is_square_attacked(int sq, Color by_color) const {
        // Check knights
        const int knight_dr[8] = { -2, -1, 1, 2, 2, 1, -1, -2 };
        const int knight_dc[8] = { 1, 2, 2, 1, -1, -2, -2, -1 };
        int r = sq / 8, c = sq % 8;
        for (int i = 0; i < 8; ++i) {
            int nr = r + knight_dr[i], nc = c + knight_dc[i];
            if (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
                int target = nr * 8 + nc;
                if (squares[target].type == KNIGHT && squares[target].color == by_color)
                    return true;
            }
        }
        
        // Check king
        for (int dr = -1; dr <= 1; ++dr) {
            for (int dc = -1; dc <= 1; ++dc) {
                if (dr == 0 && dc == 0) continue;
                int nr = r + dr, nc = c + dc;
                if (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
                    int target = nr * 8 + nc;
                    if (squares[target].type == KING && squares[target].color == by_color)
                        return true;
                }
            }
        }
        
        // Check pawns
        int pawn_dir = (by_color == WHITE) ? -1 : 1; // Direction *from* the attacker
        int pawn_rank_check = (by_color == WHITE) ? 1 : -1; // Relative rank diff
        // Attacker is at sq + pawn_dir*8 +/- 1
        int pr = r + pawn_dir;
        if (pr >= 0 && pr < 8) {
            for (int dc : {-1, 1}) {
                int pc = c + dc;
                if (pc >= 0 && pc < 8) {
                    int target = pr * 8 + pc;
                    if (squares[target].type == PAWN && squares[target].color == by_color)
                        return true;
                }
            }
        }
        
        // Check sliding pieces (Rook/Queen on ranks/files, Bishop/Queen on diagonals)
        // Rook/Queen
        const int rd[4] = {0, 0, 1, -1};
        const int cd[4] = {1, -1, 0, 0};
        for (int d = 0; d < 4; ++d) {
            for (int k = 1; k < 8; ++k) {
                int nr = r + rd[d] * k, nc = c + cd[d] * k;
                if (nr < 0 || nr >= 8 || nc < 0 || nc >= 8) break;
                int target = nr * 8 + nc;
                Piece p = squares[target];
                if (p.type != NONE) {
                    if (p.color == by_color && (p.type == ROOK || p.type == QUEEN)) return true;
                    break;
                }
            }
        }
        
        // Bishop/Queen
        const int rd2[4] = {1, 1, -1, -1};
        const int cd2[4] = {1, -1, 1, -1};
        for (int d = 0; d < 4; ++d) {
            for (int k = 1; k < 8; ++k) {
                int nr = r + rd2[d] * k, nc = c + cd2[d] * k;
                if (nr < 0 || nr >= 8 || nc < 0 || nc >= 8) break;
                int target = nr * 8 + nc;
                Piece p = squares[target];
                if (p.type != NONE) {
                    if (p.color == by_color && (p.type == BISHOP || p.type == QUEEN)) return true;
                    break;
                }
            }
        }
        
        return false;
    }
    
    bool is_in_check(Color color) const {
        if (king_sq[color] == -1) return false;
        return is_square_attacked(king_sq[color], (Color)(1 - color));
    }
    
    // Returns true if legal. Generates pseudolegal moves and checks validity.
    void make_move(const Move& m) {
        History h;
        h.move = m;
        h.captured = squares[m.to]; // Could be NO_PIECE
        h.en_passant_sq = en_passant_sq;
        h.halfmove_clock = halfmove_clock;
        copy(begin(castle_rights), end(castle_rights), begin(h.castle_rights));
        
        // Handle en passant capture
        if (m.is_en_passant) {
            int cap_sq = (side_to_move == WHITE) ? m.to - 8 : m.to + 8;
            h.captured = squares[cap_sq];
            squares[cap_sq] = NO_PIECE;
        }
        
        // Handle castling move
        if (m.is_castling) {
            // Move King
            squares[m.to] = squares[m.from];
            squares[m.from] = NO_PIECE;
            king_sq[side_to_move] = m.to;
            
            // Move Rook
            int rook_from, rook_to;
            if (m.to > m.from) { // Kingside
                rook_from = m.from + 3;
                rook_to = m.from + 1;
            } else { // Queenside
                rook_from = m.from - 4;
                rook_to = m.from - 1;
            }
            squares[rook_to] = squares[rook_from];
            squares[rook_from] = NO_PIECE;
        } else {
            squares[m.to] = squares[m.from];
            squares[m.from] = NO_PIECE;
            if (squares[m.to].type == KING) {
                king_sq[side_to_move] = m.to;
            }
            
            // Promotion
            if (m.promotion != NONE) {
                squares[m.to].type = m.promotion;
            }
        }
        
        // Update castling rights
        if (m.from == 0 || m.to == 0) castle_rights[3] = false; // q
        if (m.from == 7 || m.to == 7) castle_rights[2] = false; // k
        if (m.from == 56 || m.to == 56) castle_rights[1] = false; // Q
        if (m.from == 63 || m.to == 63) castle_rights[0] = false; // K
        
        if (squares[m.to].type == KING) {
            if (side_to_move == WHITE) {
                castle_rights[0] = false;
                castle_rights[1] = false;
            } else {
                castle_rights[2] = false;
                castle_rights[3] = false;
            }
        }
        
        // Update en passant square
        en_passant_sq = -1;
        if (squares[m.to].type == PAWN) {
            if (abs(m.to - m.from) == 16) {
                en_passant_sq = (m.to + m.from) / 2;
            }
        }
        
        if (squares[m.to].type == PAWN || h.captured.type != NONE) {
            halfmove_clock = 0;
        } else {
            halfmove_clock++;
        }
        
        if (side_to_move == BLACK) fullmove_number++;
        
        side_to_move = (Color)(1 - side_to_move);
        history.push_back(h);
    }
    
    void unmake_move() {
        History h = history.back();
        history.pop_back();
        Move m = h.move;
        
        side_to_move = (Color)(1 - side_to_move);
        if (side_to_move == BLACK) fullmove_number--;
        
        // Restore King/Rook for castling
        if (m.is_castling) {
            squares[m.from] = squares[m.to];
            squares[m.to] = NO_PIECE;
            king_sq[side_to_move] = m.from;
            
            int rook_from, rook_to;
            if (m.to > m.from) { // Kingside
                rook_from = m.from + 3;
                rook_to = m.from + 1;
            } else { // Queenside
                rook_from = m.from - 4;
                rook_to = m.from - 1;
            }
            squares[rook_from] = squares[rook_to];
            squares[rook_to] = NO_PIECE;
        } else {
            squares[m.from] = squares[m.to];
            
            // Un-promote
            if (m.promotion != NONE) {
                squares[m.from].type = PAWN;
            }
            
            // Restore captured piece
            if (m.is_en_passant) {
                int cap_sq = (side_to_move == WHITE) ? m.to - 8 : m.to + 8;
                squares[cap_sq] = h.captured;
                squares[m.to] = NO_PIECE;
            } else {
                squares[m.to] = h.captured;
            }
            
            if (squares[m.from].type == KING) {
                king_sq[side_to_move] = m.from;
            }
        }
        
        copy(begin(h.castle_rights), end(h.castle_rights), begin(castle_rights));
        en_passant_sq = h.en_passant_sq;
        halfmove_clock = h.halfmove_clock;
    }
};

// --- Move Generation ---

void generate_moves(const Board& b, vector<Move>& moves) {
    Color us = b.side_to_move;
    Color them = (Color)(1 - us);
    
    for (int sq = 0; sq < 64; ++sq) {
        Piece p = b.squares[sq];
        if (p.type == NONE || p.color != us) continue;
        
        int r = sq / 8, c = sq % 8;
        
        switch (p.type) {
            case PAWN: {
                int dir = (us == WHITE) ? 1 : -1;
                int start_rank = (us == WHITE) ? 1 : 6;
                int promo_rank = (us == WHITE) ? 7 : 0;
                
                // Forward
                int to = sq + dir * 8;
                if (to >= 0 && to < 64 && b.squares[to].type == NONE) {
                    if ((to / 8) == promo_rank) {
                        for (PieceType pt : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                            moves.push_back({sq, to, pt, false, false});
                        }
                    } else {
                        moves.push_back({sq, to, NONE, false, false});
                    }
                    
                    // Double push
                    if (r == start_rank) {
                        int to2 = sq + dir * 16;
                        if (b.squares[to2].type == NONE) {
                            moves.push_back({sq, to2, NONE, false, false});
                        }
                    }
                }
                
                // Captures
                for (int dc : {-1, 1}) {
                    int nc = c + dc;
                    if (nc < 0 || nc >= 8) continue;
                    int to_cap = sq + dir * 8 + dc;
                    if (to_cap < 0 || to_cap >= 64) continue;
                    
                    if (b.squares[to_cap].type != NONE && b.squares[to_cap].color == them) {
                        if ((to_cap / 8) == promo_rank) {
                            for (PieceType pt : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                                moves.push_back({sq, to_cap, pt, false, false});
                            }
                        } else {
                            moves.push_back({sq, to_cap, NONE, false, false});
                        }
                    } else if (to_cap == b.en_passant_sq) {
                        moves.push_back({sq, to_cap, NONE, false, true});
                    }
                }
                break;
            }
            case KNIGHT: {
                const int knight_dr[8] = { -2, -1, 1, 2, 2, 1, -1, -2 };
                const int knight_dc[8] = { 1, 2, 2, 1, -1, -2, -2, -1 };
                for (int i = 0; i < 8; ++i) {
                    int nr = r + knight_dr[i], nc = c + knight_dc[i];
                    if (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
                        int to = nr * 8 + nc;
                        if (b.squares[to].color != us) {
                            moves.push_back({sq, to, NONE, false, false});
                        }
                    }
                }
                break;
            }
            case KING: {
                for (int dr = -1; dr <= 1; ++dr) {
                    for (int dc = -1; dc <= 1; ++dc) {
                        if (dr == 0 && dc == 0) continue;
                        int nr = r + dr, nc = c + dc;
                        if (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
                            int to = nr * 8 + nc;
                            if (b.squares[to].color != us) {
                                moves.push_back({sq, to, NONE, false, false});
                            }
                        }
                    }
                }
                
                // Castling
                if (!b.is_in_check(us)) {
                    // Kingside
                    int idx = (us == WHITE) ? 0 : 2;
                    if (b.castle_rights[idx]) {
                        bool path_clear = b.squares[sq + 1].type == NONE && b.squares[sq + 2].type == NONE;
                        // Rook on sq+3 is assumed by right
                        if (path_clear) {
                            if (!b.is_square_attacked(sq + 1, them) && !b.is_square_attacked(sq + 2, them)) {
                                moves.push_back({sq, sq + 2, NONE, true, false});
                            }
                        }
                    }
                    // Queenside
                    idx = (us == WHITE) ? 1 : 3;
                    if (b.castle_rights[idx]) {
                        bool path_clear = b.squares[sq - 1].type == NONE && b.squares[sq - 2].type == NONE && b.squares[sq - 3].type == NONE;
                        if (path_clear) {
                            if (!b.is_square_attacked(sq - 1, them) && !b.is_square_attacked(sq - 2, them)) {
                                moves.push_back({sq, sq - 2, NONE, true, false});
                            }
                        }
                    }
                }
                break;
            }
            case BISHOP:
            case ROOK:
            case QUEEN: {
                int start_d = 0, end_d = 8;
                if (p.type == ROOK) { start_d = 0; end_d = 4; }
                else if (p.type == BISHOP) { start_d = 4; end_d = 8; }
                
                const int dr[8] = {0, 0, 1, -1, 1, 1, -1, -1};
                const int dc[8] = {1, -1, 0, 0, 1, -1, 1, -1};
                
                for (int d = start_d; d < end_d; ++d) {
                    for (int k = 1; k < 8; ++k) {
                        int nr = r + dr[d] * k, nc = c + dc[d] * k;
                        if (nr < 0 || nr >= 8 || nc < 0 || nc >= 8) break;
                        int to = nr * 8 + nc;
                        if (b.squares[to].type == NONE) {
                            moves.push_back({sq, to, NONE, false, false});
                        } else {
                            if (b.squares[to].color != us) {
                                moves.push_back({sq, to, NONE, false, false});
                            }
                            break;
                        }
                    }
                }
                break;
            }
            default: break;
        }
    }
}

vector<Move> generate_legal_moves(Board& b) {
    vector<Move> pseudo_moves;
    generate_moves(b, pseudo_moves);
    
    vector<Move> legal_moves;
    for (const Move& m : pseudo_moves) {
        b.make_move(m);
        if (!b.is_in_check((Color)(1 - b.side_to_move))) {
            legal_moves.push_back(m);
        }
        b.unmake_move();
    }
    return legal_moves;
}

const int KingMiddleTable[64] = {

    -30,-40,-40,-50,-50,-40,-40,-30,
    -30,-40,-40,-50,-50,-40,-40,-30,
    -30,-40,-40,-50,-50,-40,-40,-30,
    -20,-30,-30,-40,-40,-30,-30,-20,
    -10,-20,-20,-20,-20,-20,-20,-10,
    20, 20,  0,  0,  0,  0, 20, 20,
    20, 30, 10,  0,  0, 10, 30, 20

int evaluate(Board& b) {
    int score = 0;
    
    for (int sq = 0; sq < 64; ++sq) {
        Piece p = b.squares[sq];
        if (p.type == NONE) continue;
        
        int val = PieceValues[p.type];
        int mirror_sq = (7 - sq / 8) * 8 + (sq % 8);
        int map_sq = (p.color == WHITE) ? mirror_sq : sq; // Maps to White's perspective for table lookup
        
        switch (p.type) {
            case PAWN: val += PawnTable[map_sq]; break;
            case KNIGHT: val += KnightTable[map_sq]; break;
            case BISHOP: val += BishopTable[map_sq]; break;
            case ROOK: val += RookTable[map_sq]; break;
            case QUEEN: val += QueenTable[map_sq]; break;
            case KING: val += KingMiddleTable[map_sq]; break;
            default: break;
        }
        
        if (p.color == WHITE) score += val;
        else score -= val;
    }
    
    return (b.side_to_move == WHITE) ? score : -score;
}

// --- Search ---

atomic<bool> stop_search(false);
chrono::steady_clock::time_point search_start_time;
int search_time_limit_ms = 5000;

bool time_up() {
    if (stop_search) return true;
    auto now = chrono::steady_clock::now();
    auto elapsed = chrono::duration_cast<chrono::milliseconds>(now - search_start_time).count();
    if (elapsed >= search_time_limit_ms) {
        stop_search = true;
        return true;
    }
    return false;
}

int quiescence(Board& b, int alpha, int beta) {
    if (time_up()) return 0;
    
    int stand_pat = evaluate(b);
    if (stand_pat >= beta) return beta;
    if (alpha < stand_pat) alpha = stand_pat;
    
    vector<Move> moves;
    generate_moves(b, moves);
    
    // Only captures
    vector<Move> captures;
    for (const auto& m : moves) {
        if (m.is_en_passant || (m.to >= 0 && m.to < 64 && b.squares[m.to].type != NONE)) {
            captures.push_back(m);
        }
    }
    
    // Sort captures (MVV-LVA)
    sort(captures.begin(), captures.end(), [&](const Move& a, const Move& b) {
        int va = PieceValues[b.squares[a.to].type]; // Victim
        int vb = PieceValues[b.squares[b.to].type];
        // simple MVV
        return va > vb;
    });
    
    for (const Move& m : captures) {
        b.make_move(m);
        if (b.is_in_check((Color)(1 - b.side_to_move))) {
            b.unmake_move();
            continue;
        }
        
        int score = -quiescence(b, -beta, -alpha);
        b.unmake_move();
        
        if (stop_search) return 0;
        
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    
    return alpha;
}

int negamax(Board& b, int depth, int alpha, int beta) {
    if (time_up()) return 0;
    
    if (depth == 0) return quiescence(b, alpha, beta);
    
    vector<Move> moves = generate_legal_moves(b);
    
    if (moves.empty()) {
        if (b.is_in_check(b.side_to_move)) return -MATE_SCORE - depth; // Checkmate
        return 0; // Stalemate
    }
    
    // Move ordering: MVV-LVA for captures, promotions first
    sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
        if (a.promotion != b.promotion) return a.promotion > b.promotion;
        int va = PieceValues[b.squares[a.to].type];
        int vb = PieceValues[b.squares[b.to].type];
        return va > vb;
    });
    
    for (const Move& m : moves) {
        b.make_move(m);
        int score = -negamax(b, depth - 1, -beta, -alpha);
        b.unmake_move();
        
        if (stop_search) return 0;
        
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    
    return alpha;
}

Move find_best_move(Board& b, int timeout_ms) {
    search_time_limit_ms = timeout_ms;
    search_start_time = chrono::steady_clock::now();
    stop_search = false;
    
    vector<Move> legal_moves = generate_legal_moves(b);
    if (legal_moves.empty()) return Move();
    if (legal_moves.size() == 1) return legal_moves[0];
    
    Move best_move = legal_moves[0]; // Default
    
    // Iterative Deepening
    for (int depth = 1; depth <= 100; ++depth) {
        if (time_up()) break;
        
        int best_score = -INF - 1;
        Move current_best = legal_moves[0];
        
        // Ordering based on previous iteration could be done here
        // For simplicity, re-order every depth
        
        bool finished_depth = true;
        
        for (const Move& m : legal_moves) {
            if (time_up()) {
                finished_depth = false;
                break;
            }
            
            b.make_move(m);
            int score = -negamax(b, depth - 1, -INF, -best_score);
            b.unmake_move();
            
            if (stop_search) {
                finished_depth = false;
                break;
            }
            
            if (score > best_score) {
                best_score = score;
                current_best = m;
            }
        }
        
        if (finished_depth) {
            best_move = current_best;
            // Optional: reorder legal_moves to search best_move first next iteration
            // Move best to front
            for (size_t i = 0; i < legal_moves.size(); ++i) {
                if (legal_moves[i].from == best_move.from && legal_moves[i].to == best_move.to && legal_moves[i].promotion == best_move.promotion) {
                    swap(legal_moves[i], legal_moves[0]);
                    break;
                }
            }
        }
    }
    
    return best_move;
}

// --- Output Formatting ---

string to_algebraic(const Board& b, const Move& m) {
    stringstream ss;
    
    if (m.is_castling) {
        if (m.to > m.from) ss << "O-O";
        else ss << "O-O-O";
    } else {
        Piece p = b.squares[m.from];
        
        // Piece letter
        char piece_char = '?';
        switch (p.type) {
            case PAWN: piece_char = 'p'; break;
            case KNIGHT: piece_char = 'N'; break;
            case BISHOP: piece_char = 'B'; break;
            case ROOK: piece_char = 'R'; break;
            case QUEEN: piece_char = 'Q'; break;
            case KING: piece_char = 'K'; break;
            default: break;
        }
        ss << piece_char;
        
        // Source square
        ss << (char)('a' + m.from % 8) << (char)('1' + m.from / 8);
        
        // Capture
        bool is_capture = (m.is_en_passant || b.squares[m.to].type != NONE);
        if (is_capture) ss << 'x';
        
        // Destination square
        ss << (char)('a' + m.to % 8) << (char)('1' + m.to / 8);
        
        // Promotion
        if (m.promotion != NONE) {
            ss << '/';
            switch (m.promotion) {
                case QUEEN: ss << 'Q'; break;
                case ROOK: ss << 'R'; break;
                case BISHOP: ss << 'B'; break;
                case KNIGHT: ss << 'N'; break;
                default: break;
            }
        }
        
        // En passant
        if (m.is_en_passant) ss << "ep";
    }
    
    // Check/Checkmate
    // We need to apply the move to the board to check this
    // Board is currently in state before move.
    // We cast away const because make_move/unmake is non-const, or copy.
    // Let's assume we have a non-const b in `to_algebraic` caller context, 
    // but signature says `const Board&`.
    // Let's copy or use the move generation logic.
    // Since `find_best_move` returns a move, we apply it there to print?
    // No, we apply it here.
    // For simplicity in this single-file solution, we cast or make `mutable`.
    // Actually, we can just compute check status.
    
    Board temp_b = b; // Copy
    temp_b.make_move(m);
    if (temp_b.is_in_check(temp_b.side_to_move)) {
        vector<Move> next_moves = generate_legal_moves(temp_b);
        if (next_moves.empty()) ss << '#';
        else ss << '+';
    }
    
    return ss.str();
}

// --- Main ---

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int timeout_seconds;
    string fen_line;
    
    if (!(cin >> timeout_seconds)) return 0;
    cin.ignore(); // consume newline
    
    getline(cin, fen_line);
    // Handle potential empty line or malformed input
    if (fen_line.empty()) {
        // Try reading again?
        getline(cin, fen_line);
    }

    Board board;
    board.load_fen(fen_line);
    
    // Search
    int timeout_ms = timeout_seconds * 1000;
    // Leave some buffer to output results
    timeout_ms = max(100, timeout_ms - 500);
    
    Move best_move = find_best_move(board, timeout_ms);
    
    // Output
    // We must apply the move to a copy to generate the notation correctly (check/checkmate)
    // But `to_algebraic` does the copy internally now.
    cout << to_algebraic(board, best_move) << endl;
    
    return 0;
}
