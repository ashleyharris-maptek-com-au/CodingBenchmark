```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    // Guard against threads exceeding the number of bodies
    uint index = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (index >= N) {
        return;
    }

    // Unpack parameters
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Read current body data
    // Body 'index' starts at 'index * 2'
    vec4 posMass = inData[index * 2];
    vec4 velPad = inData[index * 2 + 1];

    vec3 pos_i = posMass.xyz;
    float mass_i = posMass.w; // Mass is carried through, not used for acc on self
    vec3 vel_i = velPad.xyz;

    // Compute acceleration
    vec3 acc = vec3(0.0);
    float softeningSq = softening * softening;

    // Iterate over all other bodies
    for (uint j = 0; j < N; ++j) {
        // Skip self-interaction
        if (index == j) {
            continue;
        }

        // Read body j data
        vec4 posMass_j = inData[j * 2];
        vec3 pos_j = posMass_j.xyz;
        float mass_j = posMass_j.w;

        // Compute interaction vector
        vec3 r = pos_j - pos_i;

        // distSq = |r|^2
        float distSq = dot(r, r);

        // denominator = (|r|^2 + softening^2)^(3/2)
        // We compute this as: sqrt(distSq + softeningSq)^3
        // Or equivalently: (distSq + softeningSq) * sqrt(distSq + softeningSq) * G * mass_j
        float denom = distSq + softeningSq;
        float invDistCube = inversesqrt(denom * denom * denom); 
        
        // Accumulate acceleration: G * m_j * r / denom^(3/2)
        // Formula: acc += G * mass_j * r * invDistCube
        // Note: invDistCube currently holds 1/(denom^(3/2)) without G and mass factors
        // To optimize, we apply G and mass_j to the scalar factor
        
        // Force magnitude scalar = G * mass_j * invDistCube
        acc += (G * mass_j * invDistCube) * r;
    }

    // Integration (Symplectic Euler / Leapfrog Kick-Drift)
    // Advance velocity
    vec3 newVel = vel_i + acc * dt;
    
    // Advance position
    vec3 newPos = pos_i + newVel * dt;

    // Write results to output buffer
    // Preserve mass in the position vec4's w component
    outData[index * 2]     = vec4(newPos, mass_i);
    outData[index * 2 + 1] = vec4(newVel, 0.0);
}
```
