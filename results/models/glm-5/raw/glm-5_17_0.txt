using System;
using System.Collections.Generic;
using System.Linq;

public class Box
{
    public int Index;
    public int W, H, D;
    public int Volume;
    public int MaxDim;
}

public class BoxPlacement
{
    public int Index;
    public int X, Y, Z;
    public int W, H, D;
}

public class Corner
{
    public int X, Y, Z;
    
    public Corner(int x, int y, int z)
    {
        X = x;
        Y = y;
        Z = z;
    }
    
    public override bool Equals(object obj)
    {
        if (obj is Corner other)
            return X == other.X && Y == other.Y && Z == other.Z;
        return false;
    }
    
    public override int GetHashCode()
    {
        return HashCode.Combine(X, Y, Z);
    }
}

public class PlacedBox
{
    public int X1, Y1, Z1;
    public int X2, Y2, Z2;
    public int Index;
}

public class Program
{
    static int ContainerW, ContainerH, ContainerD;
    static List<PlacedBox> placedBoxes = new List<PlacedBox>();
    static int[] volumeClassThresholds = { 8, 27, 64, 125, 216, 343, 512, 1000, 2000, 5000, 10000 };
    
    public static void Main()
    {
        string header = Console.ReadLine();
        if (string.IsNullOrEmpty(header)) return;
        
        string[] parts = header.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(parts[0]);
        ContainerW = int.Parse(parts[1]);
        ContainerH = int.Parse(parts[2]);
        ContainerD = int.Parse(parts[3]);
        
        Box[] boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            parts = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int w = int.Parse(parts[0]);
            int h = int.Parse(parts[1]);
            int d = int.Parse(parts[2]);
            
            int[] dims = new int[] { w, h, d };
            Array.Sort(dims);
            
            boxes[i] = new Box
            {
                Index = i,
                W = dims[0],
                H = dims[1],
                D = dims[2],
                Volume = w * h * d,
                MaxDim = dims[2]
            };
        }
        
        Array.Sort(boxes, (a, b) => b.Volume.CompareTo(a.Volume));
        
        var placements = Solve(boxes);
        
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
        }
    }
    
    static List<BoxPlacement> Solve(Box[] boxes)
    {
        var placements = new List<BoxPlacement>();
        var cornerSet = new HashSet<Corner>();
        var cornerList = new List<Corner>();
        
        cornerSet.Add(new Corner(0, 0, 0));
        cornerList.Add(new Corner(0, 0, 0));
        
        int gridSize = ComputeGridSize(boxes);
        var spatialIndex = new SpatialIndex(gridSize);
        
        foreach (var box in boxes)
        {
            if (cornerList.Count == 0) break;
            
            bool placed = false;
            int bestCornerIdx = -1;
            int bestScore = int.MaxValue;
            int bestOrient = 0;
            
            int maxCornersToCheck = Math.Min(cornerList.Count, GetMaxCornersToCheck(boxes.Length));
            
            for (int ci = 0; ci < maxCornersToCheck; ci++)
            {
                var corner = cornerList[ci];
                
                for (int orient = 0; orient < 6; orient++)
                {
                    int bw, bh, bd;
                    GetOrientation(box, orient, out bw, out bh, out bd);
                    
                    if (corner.X + bw > ContainerW || corner.Y + bh > ContainerH || corner.Z + bd > ContainerD)
                        continue;
                    
                    if (spatialIndex.CheckOverlap(corner.X, corner.Y, corner.Z, bw, bh, bd, box.Index))
                        continue;
                    
                    int score = (corner.X + corner.Y + corner.Z) * 1000 + ci;
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestCornerIdx = ci;
                        bestOrient = orient;
                    }
                }
            }
            
            if (bestCornerIdx >= 0)
            {
                var corner = cornerList[bestCornerIdx];
                int bw, bh, bd;
                GetOrientation(box, bestOrient, out bw, out bh, out bd);
                
                var placement = new BoxPlacement
                {
                    Index = box.Index,
                    X = corner.X,
                    Y = corner.Y,
                    Z = corner.Z,
                    W = bw,
                    H = bh,
                    D = bd
                };
                placements.Add(placement);
                
                var placedBox = new PlacedBox
                {
                    X1 = corner.X,
                    Y1 = corner.Y,
                    Z1 = corner.Z,
                    X2 = corner.X + bw,
                    Y2 = corner.Y + bh,
                    Z2 = corner.Z + bd,
                    Index = box.Index
                };
                placedBoxes.Add(placedBox);
                spatialIndex.Add(placedBox);
                
                AddCorner(cornerSet, cornerList, corner.X + bw, corner.Y, corner.Z);
                AddCorner(cornerSet, cornerList, corner.X, corner.Y + bh, corner.Z);
                AddCorner(cornerSet, cornerList, corner.X, corner.Y, corner.Z + bd);
                AddCorner(cornerSet, cornerList, corner.X + bw, corner.Y + bh, corner.Z);
                AddCorner(cornerSet, cornerList, corner.X + bw, corner.Y, corner.Z + bd);
                AddCorner(cornerSet, cornerList, corner.X, corner.Y + bh, corner.Z + bd);
                AddCorner(cornerSet, cornerList, corner.X + bw, corner.Y + bh, corner.Z + bd);
                
                placed = true;
            }
        }
        
        return placements;
    }
    
    static int ComputeGridSize(Box[] boxes)
    {
        if (boxes.Length == 0) return 50;
        
        long totalVol = 0;
        int minDim = int.MaxValue;
        foreach (var b in boxes)
        {
            totalVol += b.Volume;
            minDim = Math.Min(minDim, Math.Min(b.W, Math.Min(b.H, b.D)));
        }
        
        long containerVol = (long)ContainerW * ContainerH * ContainerD;
        double fillRatio = totalVol > 0 ? (double)totalVol / containerVol : 0.5;
        
        int baseSize = Math.Max(20, (int)Math.Sqrt(boxes.Length));
        
        if (fillRatio > 0.8) baseSize = Math.Max(baseSize, 40);
        if (fillRatio < 0.1) baseSize = Math.Max(baseSize, 10);
        
        if (minDim < 10) baseSize = Math.Max(baseSize, 50);
        
        return Math.Min(baseSize, 100);
    }
    
    static int GetMaxCornersToCheck(int boxCount)
    {
        if (boxCount < 100) return 1000;
        if (boxCount < 1000) return 500;
        if (boxCount < 10000) return 200;
        return 100;
    }
    
    static void GetOrientation(Box box, int orient, out int w, out int h, out int d)
    {
        switch (orient)
        {
            case 0: w = box.W; h = box.H; d = box.D; break;
            case 1: w = box.W; h = box.D; d = box.H; break;
            case 2: w = box.H; h = box.W; d = box.D; break;
            case 3: w = box.H; h = box.D; d = box.W; break;
            case 4: w = box.D; h = box.W; d = box.H; break;
            default: w = box.D; h = box.H; d = box.W; break;
        }
    }
    
    static void AddCorner(HashSet<Corner> cornerSet, List<Corner> cornerList, int x, int y, int z)
    {
        if (x >= ContainerW || y >= ContainerH || z >= ContainerD) return;
        var c = new Corner(x, y, z);
        if (cornerSet.Add(c))
        {
            cornerList.Add(c);
        }
    }
}

public class SpatialIndex
{
    int cellSize;
    Dictionary<long, List<PlacedBox>> cells = new Dictionary<long, List<PlacedBox>>();
    
    public SpatialIndex(int cellSize)
    {
        this.cellSize = Math.Max(1, cellSize);
    }
    
    public void Add(PlacedBox box)
    {
        int minX = box.X1 / cellSize;
        int minY = box.Y1 / cellSize;
        int minZ = box.Z1 / cellSize;
        int maxX = box.X2 / cellSize;
        int maxY = box.Y2 / cellSize;
        int maxZ = box.Z2 / cellSize;
        
        for (int x = minX; x <= maxX; x++)
        {
            for (int y = minY; y <= maxY; y++)
            {
                for (int z = minZ; z <= maxZ; z++)
                {
                    long key = GetKey(x, y, z);
                    if (!cells.TryGetValue(key, out var list))
                    {
                        list = new List<PlacedBox>();
                        cells[key] = list;
                    }
                    list.Add(box);
                }
            }
        }
    }
    
    public bool CheckOverlap(int x, int y, int z, int w, int h, int d, int boxIndex)
    {
        int minX = x / cellSize;
        int minY = y / cellSize;
        int minZ = z / cellSize;
        int maxX = (x + w) / cellSize;
        int maxY = (y + h) / cellSize;
        int maxZ = (z + d) / cellSize;
        
        for (int cx = minX; cx <= maxX; cx++)
        {
            for (int cy = minY; cy <= maxY; cy++)
            {
                for (int cz = minZ; cz <= maxZ; cz++)
                {
                    long key = GetKey(cx, cy, cz);
                    if (cells.TryGetValue(key, out var list))
                    {
                        foreach (var box in list)
                        {
                            if (box.Index == boxIndex) continue;
                            if (BoxesOverlap(x, y, z, x + w, y + h, z + d, box))
                                return true;
                        }
                    }
                }
            }
        }
        return false;
    }
    
    long GetKey(int x, int y, int z)
    {
        return ((long)x * 1000000 + y) * 1000000 + z;
    }
    
    bool BoxesOverlap(int x1, int y1, int z1, int x2, int y2, int z2, PlacedBox box)
    {
        return x1 < box.X2 && x2 > box.X1 &&
               y1 < box.Y2 && y2 > box.Y1 &&
               z1 < box.Z2 && z2 > box.Z1;
    }
}
