; Generator: Khronos SPIR-V Tools Assembler
; Bound: 100

OpCapability Geometry

OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%3 = OpTypeFunction %void

%float = OpTypeFloat 32

%uint = OpTypeInt 32 0

; PerVertex Input Block

%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

; User Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_333333 = OpConstant %float 0.333333
%float_0_7 = OpConstant %float 0.7

; Pointers
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %3
%5 = OpLabel

; --- Load Positions ---
%p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %p0
%p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %p1
%p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0

%sum_p2 = OpFAdd %v4float %sum_p %pos2

; --- Process Vertex 0 ---
; Load Input Color 0

; diff = pos0 - centroid

; scaled_diff = 0.7 * diff
%scaled0 = OpVectorTimesScalar %v4float %diff0 %float_0_7
; new_pos = centroid + scaled_diff
%new_pos0 = OpFAdd %v4float %centroid %scaled0

; Store Output 0
%out_p0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_p0 %new_pos0
OpStore %outColor %color0
OpEmitVertex

; --- Process Vertex 1 ---
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%color1 = OpLoad %v3float %c1_ptr

%diff1 = OpFSub %v4float %pos1 %centroid
%scaled1 = OpVectorTimesScalar %v4float %diff1 %float_0_7
%new_pos1 = OpFAdd %v4float %centroid %scaled1

%out_p1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_p1 %new_pos1
OpStore %outColor %color1

; --- Process Vertex 2 ---

%out_p2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_p2 %new_pos2
OpStore %outColor %color2
