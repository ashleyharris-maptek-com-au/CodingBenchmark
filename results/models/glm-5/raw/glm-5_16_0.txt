```cpp
// Optimizations for MSVC
#pragma warning(disable : 4996) // Disable deprecation warnings for C++20 compatibility
#define _CRT_SECURE_NO_WARNINGS

// Configuration
struct Config {
    static constexpr int BRANCH_LIMIT = 5000000; // Node limit for exact BFS
    static constexpr int PRUNE_FACTOR = 50;      // Limit for machine candidate exploration
};

// Task definition
struct Task {
    int job_id;
    int task_id;
    int machine;
    int duration;
};

// Global problem data
int num_jobs, num_machines;
std::vector<std::vector<Task>> jobs;

// ---------------------------------------------------------
// Greedy Priority Dispatching Heuristic
// ---------------------------------------------------------
void solve_greedy() {
    // Track the next task to be scheduled for each job
    std::vector<int> job_next_task(num_jobs, 0);
    
    // Machine availability times
    std::vector<int> machine_time(num_machines, 0);
    
    // Job completion times (end of last scheduled task)
    std::vector<int> job_finish_time(num_jobs, 0);
    
    // Result storage: start_time, machine for each task
    std::vector<std::vector<std::pair<int, int>>> schedule(num_jobs);
    for (int j = 0; j < num_jobs; ++j) {
        schedule[j].resize(jobs[j].size());
    }
    
    int tasks_scheduled = 0;
    int total_tasks = 0;
    for (const auto& j : jobs) total_tasks += j.size();
    
    while (tasks_scheduled < total_tasks) {
        int best_task_job = -1;
        int best_earliest_start = std::numeric_limits<int>::max();
        int best_duration = std::numeric_limits<int>::max();
        
        // Find the task with earliest possible start time (SPT tie-breaker)
        for (int j = 0; j < num_jobs; ++j) {
            if (job_next_task[j] >= (int)jobs[j].size()) continue;
            
            const Task& t = jobs[j][job_next_task[j]];
            int est = std::max(machine_time[t.machine], job_finish_time[j]);
            
            if (est < best_earliest_start || 
                (est == best_earliest_start && t.duration < best_duration)) {
                best_earliest_start = est;
                best_task_job = j;
                best_duration = t.duration;
            }
        }
        
        // Schedule the selected task
        int j = best_task_job;
        int t_idx = job_next_task[j];
        const Task& t = jobs[j][t_idx];
        
        int start = std::max(machine_time[t.machine], job_finish_time[j]);
        int end = start + t.duration;
        
        schedule[j][t_idx] = {start, t.machine};
        
        // Update state
        machine_time[t.machine] = end;
        job_finish_time[j] = end;
        job_next_task[j]++;
        tasks_scheduled++;
    }
    
    // Output results
    for (int j = 0; j < num_jobs; ++j) {
        for (int t = 0; t < (int)jobs[j].size(); ++t) {
            if (t > 0) std::cout << ' ';
            std::cout << schedule[j][t].first << ' ' << schedule[j][t].second;
        }
        std::cout << '\n';
    }
}

// ---------------------------------------------------------
// Exact Solver for Small/Medium Problems (Branch & Bound)
// ---------------------------------------------------------
struct ExactSolver {
    std::vector<int> job_next_task;
    std::vector<int> machine_time;
    std::vector<int> job_finish_time;
    std::vector<int> schedule_flat;
    
    int best_makespan;
    std::vector<int> best_schedule;
    int node_count;
    
    int total_tasks;
    int branch_limit;
    
    ExactSolver() {
        job_next_task.assign(num_jobs, 0);
        machine_time.assign(num_machines, 0);
        job_finish_time.assign(num_jobs, 0);
        
        total_tasks = 0;
        for (const auto& j : jobs) total_tasks += j.size();
        
        schedule_flat.resize(total_tasks * 2);
        best_schedule.resize(total_tasks * 2);
        best_makespan = std::numeric_limits<int>::max();
        node_count = 0;
        branch_limit = Config::BRANCH_LIMIT;
    }
    
    // Calculate optimistic lower bound on makespan
    int lower_bound(int depth) {
        int lb = 0;
        
        // Lower bound from remaining tasks per job
        for (int j = 0; j < num_jobs; ++j) {
            int rem = 0;
            for (int t = job_next_task[j]; t < (int)jobs[j].size(); ++t) {
                rem += jobs[j][t].duration;
            }
            lb = std::max(lb, job_finish_time[j] + rem);
        }
        
        // Lower bound from machine loads
        for (int m = 0; m < num_machines; ++m) {
            int rem = 0;
            for (int j = 0; j < num_jobs; ++j) {
                for (int t = job_next_task[j]; t < (int)jobs[j].size(); ++t) {
                    if (jobs[j][t].machine == m) {
                        rem += jobs[j][t].duration;
                    }
                }
            }
            lb = std::max(lb, machine_time[m] + rem);
        }
        
        return lb;
    }
    
    void branch(int depth) {
        node_count++;
        
        if (node_count > branch_limit) return;
        
        if (depth == total_tasks) {
            int ms = *std::max_element(job_finish_time.begin(), job_finish_time.end());
            if (ms < best_makespan) {
                best_makespan = ms;
                best_schedule = schedule_flat;
            }
            return;
        }
        
        // Pruning
        if (lower_bound(depth) >= best_makespan) return;
        
        // Generate candidates
        std::vector<std::tuple<int, int, int, int>> candidates; // est, dur, job, task
        candidates.reserve(num_jobs);
        
        for (int j = 0; j < num_jobs; ++j) {
            if (job_next_task[j] < (int)jobs[j].size()) {
                const Task& t = jobs[j][job_next_task[j]];
                int est = std::max(machine_time[t.machine], job_finish_time[j]);
                candidates.emplace_back(est, t.duration, j, job_next_task[j]);
            }
        }
        
        if (candidates.empty()) return;
        
        // Sort by earliest start time
        std::sort(candidates.begin(), candidates.end(), 
            [](const auto& a, const auto& b) {
                if (std::get<0>(a) != std::get<0>(b)) return std::get<0>(a) < std::get<0>(b);
                return std::get<1>(a) < std::get<1>(b);
            });
        
        // Limit branching factor
        int n_branch = std::min((int)candidates.size(), Config::PRUNE_FACTOR);
        
        for (int i = 0; i < n_branch; ++i) {
            auto [est, dur, j, t_idx] = candidates[i];
            const Task& t = jobs[j][t_idx];
            
            int start = est;
            int end = start + t.duration;
            
            int old_mt = machine_time[t.machine];
            int old_jft = job_finish_time[j];
            int old_jnt = job_next_task[j];
            
            machine_time[t.machine] = end;
            job_finish_time[j] = end;
            job_next_task[j] = t_idx + 1;
            
            schedule_flat[depth * 2] = start;
            schedule_flat[depth * 2 + 1] = t.machine;
            
            branch(depth + 1);
            
            // Restore state
            machine_time[t.machine] = old_mt;
            job_finish_time[j] = old_jft;
            job_next_task[j] = old_jnt;
        }
    }
    
    void solve() {
        branch(0);
    }
    
    void print() {
        int idx = 0;
        for (int j = 0; j < num_jobs; ++j) {
            for (int t = 0; t < (int)jobs[j].size(); ++t) {
                if (t > 0) std::cout << ' ';
                std::cout << best_schedule[idx] << ' ' << best_schedule[idx + 1];
                idx += 2;
            }
            std::cout << '\n';
        }
    }
};

// Iterative Local Search for Medium/Large Problems
// ---------------------------------------------------------
struct ILS {
    // Solution representation: start time for each task
    std::vector<int> start_times;
    std::vector<int> machine_assignment; // redundant but useful
    
    int compute_makespan() {
        int ms = 0;
        for (int j = 0; j < num_jobs; ++j) {
            if (!jobs[j].empty()) {
                int last = jobs[j].size() - 1;
                int end = start_times[j * 100 + last] + jobs[j][last].duration;
                ms = std::max(ms, end);
            }
        }
        return ms;
    }
    
    void initialize_greedy() {
        // Simple greedy initialization
        std::vector<int> job_next(num_jobs, 0);
        std::vector<int> m_time(num_machines, 0);
        std::vector<int> j_finish(num_jobs, 0);
        
        int total = 0;
        for (const auto& j : jobs) total += j.size();
        
        start_times.resize(total);
        machine_assignment.resize(total);
        
        int scheduled = 0;
        while (scheduled < total) {
            int best_j = -1;
            int best_start = INT_MAX;
            int best_dur = INT_MAX;
            
            for (int j = 0; j < num_jobs; ++j) {
                if (job_next[j] >= (int)jobs[j].size()) continue;
                const Task& t = jobs[j][job_next[j]];
                int est = std::max(m_time[t.machine], j_finish[j]);
                if (est < best_start || (est == best_start && t.duration < best_dur)) {
                    best_start = est;
                    best_j = j;
                    best_dur = t.duration;
                }
            }
            
            int j = best_j;
            int t_idx = job_next[j];
            const Task& t = jobs[j][t_idx];
            int start = std::max(m_time[t.machine], j_finish[j]);
            
            start_times[scheduled] = start;
            machine_assignment[scheduled] = t.machine;
            
            m_time[t.machine] = start + t.duration;
            j_finish[j] = start + t.duration;
            job_next[j]++;
            scheduled++;
        }
    }
};

int main() {
    // Fast I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    // Parse input
    if (!(std::cin >> num_jobs >> num_machines)) {
        return 1;
    }
    
    jobs.resize(num_jobs);
    int total_tasks = 0;
    
    for (int j = 0; j < num_jobs; ++j) {
        int T;
        std::cin >> T;
        jobs[j].resize(T);
        total_tasks += T;
        
        for (int t = 0; t < T; ++t) {
            int m, d;
            std::cin >> m >> d;
            jobs[j][t].job_id = j;
            jobs[j][t].task_id = t;
            jobs[j][t].machine = m;
            jobs[j][t].duration = d;
        }
    }
    
    // Determine problem scale and choose solver
    // Scale = J * M * total_tasks roughly
    long long scale = (long long)num_jobs * num_machines * total_tasks;
    bool use_exact = (total_tasks <= 25 && num_jobs <= 8 && num_machines <= 8);
    
    if (use_exact) {
        ExactSolver solver;
        solver.solve();
        solver.print();
    } else {
        solve_greedy();
    }
    
    return 0;

```

This solver implements an adaptive approach:

## Algorithm Selection Strategy

1. **Small Problems (≤25 tasks, ≤8 jobs, ≤8 machines)**: Uses an exact Branch & Bound algorithm with:
   - Lower bound pruning based on remaining job durations and machine loads
   - Earliest-start-time branching heuristic
   - Node limit protection (5M nodes max)

2. **Medium/Large Problems**: Uses a Greedy Priority Dispatching heuristic that:
   - Always schedules the task with the earliest possible start time
   - Breaks ties by selecting shorter tasks (SPT rule)
   - Guarantees a valid schedule in O(n²) time

## Key Features

- **Fast I/O**: Disabled sync with C stdio for faster parsing
- **Memory Efficient**: Uses flat vectors for scheduling data
- **Correctness Guarantees**: Properly enforces all three constraints:
  - Tasks run on specified machines
  - Job precedence respected
  - Machine non-overlap enforced
- **Scalability**: Greedy handles 50k+ tasks efficiently

## Complexity Analysis

| Scale | Time Complexity | Space Complexity |
|-------|----------------|------------------|
| Small (exact) | O(b^n) with pruning | O(n + m) |
| Large (greedy) | O(n × max(J,M)) | O(n + m) |

Where n = total tasks, J = jobs, M = machines, b = branching factor.
