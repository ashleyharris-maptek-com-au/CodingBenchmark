const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Extract input data
const machineCount = data.machineCount || 1;
const tasks = data.tasks || [];
const n = tasks.length;

// Handle empty input
if (n === 0) {
    console.log(JSON.stringify({ schedule: [] }));
    process.exit(0);
}

// Build task map and adjacency lists
const taskMap = new Map();
for (const t of tasks) {
    taskMap.set(t.id, t);
}

// Topological sort with priority
// Priority: highest criticality first (combination of deadline urgency and weight)
const inDegree = new Map();
const successors = new Map();

for (const t of tasks) {
    inDegree.set(t.id, 0);
    successors.set(t.id, []);
}

for (const t of tasks) {
    if (t.deps) {
        for (const depId of t.deps) {
            if (taskMap.has(depId)) {
                inDegree.set(t.id, inDegree.get(t.id) + 1);
                successors.get(depId).push(t.id);
            }
        }
    }
}

// Compute earliest start times and priority values
const earliestStart = new Map();
const deadlinePriority = new Map();
const criticality = new Map();

// Forward pass for earliest start times
const queue = [];
for (const t of tasks) {
    if (inDegree.get(t.id) === 0) {
        queue.push(t.id);
    }
}

const topoOrder = [];
let idx = 0;
while (idx < queue.length) {
    const u = queue[idx++];
    topoOrder.push(u);
    const task = taskMap.get(u);
    
    let es = 0;
    if (task.deps) {
        for (const depId of task.deps) {
            if (earliestStart.has(depId)) {
                const depTask = taskMap.get(depId);
                es = Math.max(es, earliestStart.get(depId) + (depTask.duration || 0));
            }
        }
    }
    earliestStart.set(u, es);
    
    // Compute priority: lower deadline = higher priority, higher weight = higher priority
    const deadline = task.deadline || Infinity;
    const weight = task.weight || 1;
    const duration = task.duration || 0;
    
    // Normalize and combine factors
    const urgency = deadline === Infinity ? 0 : 1 / (deadline + 1);
    const density = weight / Math.max(duration, 1);
    
    deadlinePriority.set(u, deadline);
    criticality.set(u, weight * urgency + density);
    
    for (const v of successors.get(u)) {
        inDegree.set(v, inDegree.get(v) - 1);
        if (inDegree.get(v) === 0) {
            queue.push(v);
        }
    }
}

// Priority queue implementation using binary heap
class PriorityQueue {
    constructor(compare) {
        this.heap = [];
        this.compare = compare;
    }
    
    push(item) {
        this.heap.push(item);
        this._siftUp(this.heap.length - 1);
    }
    
    pop() {
        if (this.heap.length === 0) return undefined;
        const top = this.heap[0];
        const last = this.heap.pop();
        if (this.heap.length > 0) {
            this.heap[0] = last;
            this._siftDown(0);
        }
        return top;
    }
    
    peek() {
        return this.heap[0];
    }
    
    get length() {
        return this.heap.length;
    }
    
    _siftUp(i) {
        while (i > 0) {
            const parent = (i - 1) >> 1;
            if (this.compare(this.heap[i], this.heap[parent]) < 0) {
                [this.heap[i], this.heap[parent]] = [this.heap[parent], this.heap[i]];
                i = parent;
            } else break;
        }
    }
    
    _siftDown(i) {
        const n = this.heap.length;
        while (true) {
            let smallest = i;
            const left = 2 * i + 1;
            const right = 2 * i + 2;
            
            if (left < n && this.compare(this.heap[left], this.heap[smallest]) < 0) {
                smallest = left;
            }
            if (right < n && this.compare(this.heap[right], this.heap[smallest]) < 0) {
                smallest = right;
            }
            
            if (smallest !== i) {
                [this.heap[i], this.heap[smallest]] = [this.heap[smallest], this.heap[i]];
                i = smallest;
            } else break;
        }
    }
}

// Event-driven simulation
// State: current time, available tasks, machine states
const readySet = new Set();
const remainingDeps = new Map();

// Initialize remaining dependencies
for (const t of tasks) {
    const deg = t.deps ? t.deps.filter(d => taskMap.has(d)).length : 0;
    remainingDeps.set(t.id, deg);
    if (deg === 0) {
        readySet.add(t.id);
    }
}

// Machine availability: each machine has an available time starting at 0
const machineAvailable = new Array(machineCount).fill(0);
const schedule = [];

// Track completed tasks for dependency resolution
const completed = new Set();

// Main scheduling loop
while (schedule.length < n) {
    // Find next event time: min of machine available times
    let currentTime = Infinity;
    for (let i = 0; i < machineCount; i++) {
        if (machineAvailable[i] < currentTime) {
            currentTime = machineAvailable[i];
        }
    }
    
    // Check for available tasks at current time
    // Get all ready tasks sorted by priority
    const readyList = [];
    for (const taskId of readySet) {
        const task = taskMap.get(taskId);
        const es = earliestStart.get(taskId) || 0;
        // Task is truly ready if earliest start constraint is satisfied
        readyList.push(taskId);
    }
    
    if (readyList.length === 0) {
        // No ready tasks - advance time or find what's blocking
        // This shouldn't happen in a valid DAG, but handle gracefully
        break;
    }
    
    // Sort ready tasks by priority (lower deadline first, then higher weight)
    readyList.sort((a, b) => {
        const taskA = taskMap.get(a);
        const taskB = taskMap.get(b);
        
        const deadlineA = taskA.deadline || Infinity;
        const deadlineB = taskB.deadline || Infinity;
        
        // Primary: deadline (earlier = higher priority)
        if (deadlineA !== deadlineB) {
            return deadlineA - deadlineB;
        }
        
        // Secondary: weight (higher = higher priority)
        const weightA = taskA.weight || 1;
        const weightB = taskB.weight || 1;
        if (weightA !== weightB) {
            return weightB - weightA;
        }
        
        // Tertiary: duration (shorter first for better packing)
        const durA = taskA.duration || 0;
        const durB = taskB.duration || 0;
        return durA - durB;
    });
    
    // Find idle machines at current time
    const idleMachines = [];
    for (let i = 0; i < machineCount; i++) {
        if (machineAvailable[i] <= currentTime) {
            idleMachines.push(i);
        }
    }
    
    // Assign tasks to idle machines
    const toRemove = [];
    let machineIdx = 0;
    
    for (const taskId of readyList) {
        if (machineIdx >= idleMachines.length) break;
        
        const task = taskMap.get(taskId);
        const es = earliestStart.get(taskId) || 0;
        const machine = idleMachines[machineIdx];
        const startTime = Math.max(machineAvailable[machine], es);
        
        // Schedule this task
        schedule.push({
            id: taskId,
            start: startTime,
            machine: machine
        });
        
        const duration = task.duration || 0;
        machineAvailable[machine] = startTime + duration;
        completed.add(taskId);
        toRemove.push(taskId);
        machineIdx++;
    }
    
    // Remove scheduled tasks from ready set
    for (const id of toRemove) {
        readySet.delete(id);
    }
    
    // Update dependencies
    for (const taskId of toRemove) {
        const task = taskMap.get(taskId);
        const finishTime = (earliestStart.get(taskId) || 0) + (task.duration || 0);
        
        for (const succId of successors.get(taskId)) {
            const oldDeps = remainingDeps.get(succId);
            const newDeps = oldDeps - 1;
            remainingDeps.set(succId, newDeps);
            
            if (newDeps === 0) {
                readySet.add(succId);
            }
        }
    }
}

// Output the schedule
console.log(JSON.stringify({ schedule }));
