// Greedy coloring with ordering
void greedy_coloring(const vector<int>& order) {
    vector<bool> used(num_colors);
    for (int u : order) {
        fill(used.begin(), used.end(), false);
        for (int v : adj[u]) {
            if (colors[v] >= 0 && colors[v] < num_colors) {
                used[colors[v]] = true;
            }
        }
        for (int c = 0; c < num_colors; ++c) {
            if (!used[c]) {
                colors[u] = c;
                break;
            }
        }
    }
}

class TabuSearch {
    struct TabuEntry {
        int vertex, color;
        int expiry;
        
        bool operator<(const TabuEntry& other) const {
            return expiry < other_expiry;
        }
        int other_expiry;
    };
    
    vector<int> tabu_list;
    int tabu_tenure;
    int iteration;
    int best_conflicts;
    vector<int> conflict_counts;
    
public:
    bool run(int max_iterations) {
        iteration = 0;
        best_conflicts = INT_MAX;
        tabu_tenure = max(5, num_vertices / 50);
        tabu_list.assign(num_vertices * num_colors, 0);
        
        conflict_counts.assign(num_vertices, 0);
        int total_conflicts = 0;
        
        // Count initial conflicts
        for (int u = 0; u < num_vertices; ++u) {
            for (int v : adj[u]) {
                if (u < v && colors[u] == colors[v]) {
                    conflict_counts[u]++;
                    conflict_counts[v]++;
                    total_conflicts++;
                }
            }
        }
        total_conflicts /= 2;
        best_conflicts = total_conflicts;
        
        if (best_conflicts == 0) return true;
        
        while (iteration < max_iterations) {
            if (iteration % 1000 == 0 && check_time()) return false;
            
            // Find best move
            int best_u = -1, best_c = -1;
            int best_delta = INT_MAX;
            bool is_tabu = true;
            
            // Sample vertices with conflicts
            int samples = min(100, num_vertices);
            for (int i = 0; i < samples; ++i) {
                int u = rng() % num_vertices;
                if (conflict_counts[u] == 0) continue;
                
                int old_c = colors[u];
                for (int c = 0; c < num_colors; ++c) {
                    if (c == old_c) continue;
                    
                    int delta = compute_delta(u, c, old_c);
                    bool tabu = tabu_list[u * num_colors + c] > iteration;
                    
                    if (!tabu && delta < best_delta) {
                        best_delta = delta;
                        best_u = u;
                        best_c = c;
                        is_tabu = false;
                    } else if (tabu && delta < best_delta) {
                        // Aspiration criterion
                        if (total_conflicts + delta < best_conflicts) {
                            best_delta = delta;
                            best_u = u;
                            best_c = c;
                            is_tabu = true;
                        }
                    }
                }
            }
            
            if (best_u == -1) {
                // Random move
                best_u = rng() % num_vertices;
                best_c = rng() % num_colors;
                while (best_c == colors[best_u]) {
                    best_c = rng() % num_colors;
                }
            }
            
            // Apply move
            int old_color = colors[best_u];
            update_conflicts(best_u, old_color, best_c, conflict_counts, total_conflicts);
            colors[best_u] = best_c;
            
            // Update tabu list
            tabu_list[best_u * num_colors + old_color] = iteration + tabu_tenure + (rng() % 10);
            
            if (total_conflicts < best_conflicts) {
                best_conflicts = total_conflicts;
                if (best_conflicts == 0) return true;
            }
            
            iteration++;
        }
        
        return best_conflicts == 0;
    }
    
private:
    int compute_delta(int u, int new_c, int old_c) {
        int delta = 0;
        for (int v : adj[u]) {
            if (colors[v] == old_c) delta--;
            if (colors[v] == new_c) delta++;
        }
        return delta;
    }
    
    void update_conflicts(int u, int old_c, int new_c, vector<int>& cc, int& total) {
        for (int v : adj[u]) {
            if (colors[v] == old_c) {
                cc[u]--;
                cc[v]--;
                total--;
            }
            if (colors[v] == new_c) {
                cc[u]++;
                cc[v]++;
                total++;
            }
        }
    }

Actually, looking at this more carefully, the DSATUR approach doesn't work well for k-coloring when k is fixed and potentially less than the chromatic number. Let me reconsider the algorithm.

For the graph k-coloring problem where we must use exactly k colors (and k might be less than the chromatic number), we need a different approach. The key is:

1. If k >= chromatic number, we can always find a valid coloring
2. If k < chromatic number, we need to minimize conflicts and the problem becomes finding if a valid k-coloring exists
