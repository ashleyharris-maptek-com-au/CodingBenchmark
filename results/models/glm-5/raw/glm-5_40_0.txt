#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <algorithm>
#include <numeric>
#include <set>
#include <queue>
#include <map>
#include <memory>

using namespace std;

typedef long long ll;

// IO Helpers
int read_int() {
    int x;
    cin >> x;
    return x;
}

// Solve 0/1 ILP using Branch and Bound
// Objective: Maximize c^T x
// Subject to: x_u + x_v <= 1

struct Node {
    vector<int> assignment; // -1: unassigned, 0: false, 1: true
    ll lb; // Lower bound (objective value of valid solution found in/above this node)
    int depth;
    int branching_var;
    bool lb_is_valid; // True if lb represents a feasible solution value

    // Priority queue needs largest LB first. 
    // std::priority_queue orders largest element to the top.
    // We want to prioritize nodes with valid LBs and high LB values.
    // If LB is not valid, it's just an upper bound estimate, but with accurate bounds it should be fine.
    // To ensure valid solutions are prioritized:
    bool operator<(const Node& other) const {
        // If one is valid and other isn't, valid goes first? 
        // Actually standard B&B just uses the bound value.
        // Let's stick to simple bound comparison for now, but we must ensure we don't pop invalid nodes indefinitely.
        // Actually, we only push valid bounds (feasible relaxed solutions).
        return lb < other.lb;
    }
};

int main() {
    // Optimize I/O operations
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<int> c(n);
    for (int i = 0; i < n; ++i) {
        cin >> c[i];
    }

    vector<vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    // Branch and Bound
    // Maximize sum c_i x_i
    // Constraints: x_u + x_v <= 1 (Independent Set)

    // Initial state: all variables unassigned
    Node root;
    root.assignment.resize(n, -1);
    root.lb = 0;
    root.depth = 0;
    root.branching_var = -1;
    root.lb_is_valid = true; // Empty assignment has value 0 (feasible)

    // Priority Queue: stores nodes ordered by Lower Bound (descending)
    priority_queue<Node> pq;
    pq.push(root);

    ll global_best = 0;
    vector<int> best_assignment(n, 0);

    // The bounding function:
    // For any unassigned variable i, if we set it to 1, we gain c[i].
    // Is this always possible? 
    // Constraint violation occurs only if a neighbor j is already assigned 1.
    // If all neighbors j are assigned 0 or -1, we can tentatively set x_i=1.
    // Since the constraint is x_u + x_v <= 1, setting x_j=0 never hurts the feasibility of setting x_i=1.
    // Thus, for the relaxation upper bound, we can iterate all unassigned i.
    // If no neighbor is assigned 1, we add c[i] (assuming c[i] > 0).
    // This creates a valid upper bound (relaxation).
    // Crucially, this relaxed solution is actually feasible for the relaxed constraints.
    // Wait, if we greedily add c[i] for unassigned i, we might pick x_u=1 and x_v=1 if u, v are neighbors.
    // BUT in the Branch and Bound tree:
    // If we are at a node, we have fixed assignments.
    // When computing the bound:
    // We iterate over unassigned variables.
    // We can add c[i] to the bound if we can independently set it to 1.
    // However, we must respect the constraints between *unassigned* variables too for an Upper Bound.
    // The "greedy independent set" on unassigned variables is a valid Upper Bound for the remaining capacity.
    // Actually, solving Max Weight Independent Set on the induced subgraph of unassigned variables + fixed 1s constraint.
    // Since MWIS is the problem itself, we just use a greedy approach for the bound.
    // Greedy MWIS: sort by weight, pick if possible.
    // This gives a valid Upper Bound for the subproblem.
    // And it also yields a valid feasible solution for the subproblem (the greedy set).
    // If the greedy solution value equals the upper bound (impossible unless perfect), or we just use the greedy solution value as the Node's LB (Lower Bound on Optimum).
    // Wait, Priority Queue prioritizes largest LB.
    // If we compute a greedy solution for the subproblem, that is a valid lower bound for the optimum of that subproblem.
    // If we push nodes with this LB, we explore promising regions.
    // The "Upper Bound" logic is typically used to prune.
    // Pruning: If Node.lb <= global_best, prune.
    // So if we use the Greedy solution value as the node's value in PQ:
    // 1. It's a valid feasible solution.
    // 2. We explore nodes that look promising.
    // This effectively becomes a "Best-First Search" using a heuristic valuation.
    // This is safe for correctness (exhaustive, modulo pruning).
    // Pruning condition: If the theoretical maximum possible value (Upper Bound) is <= global_best, we prune.
    // Theoretical Max: Sum of positive weights of unassigned variables that are compatible with current assignment.
    // Let's refine:
    // For each node:
    // 1. Calculate a valid Lower Bound (constructive heuristic).
    // 2. Calculate a valid Upper Bound (relaxation).
    // If Upper Bound <= global_best: Prune.
    // Else: Push to PQ. (Using LB or UB for sorting is a choice; LB guides towards finding good solutions fast, UB guides towards proving optimality).
    // With the greedy MWIS on unassigned variables:
    // We get a solution (LB).
    // Do we get an UB easily?
    // Sum of weights of unassigned variables that are compatible with current assignment?
    // No, need to respect unassigned-unassigned constraints.
    // Just summing positive compatible weights is an over-estimate (UB).
    // Let's compute both.
    // UB = current_value + sum_{i compatible with fixed 1s} max(0, c[i]).
    // This UB assumes we can take all compatible items. (Relaxes unassigned-unassigned constraints).
    // This is fast to compute.
    
    while (!pq.empty()) {
        Node node = pq.top();
        pq.pop();

        // Pruning check
        // If the lower bound (best solution we found in this subtree) is not better than global best, we might still need to explore if we used UB for PQ ordering.
        // But here we assume we push valid feasible LBs into PQ.
        // If we push UBs, we need a separate check.
        // Let's push nodes based on Upper Bound to ensure optimality proof.
        // Correction:
        // If we use LB for PQ ordering, we are doing "Best-First Search" on heuristic.
        // This finds good solutions fast but might struggle to prove optimality (exhausting search space).
        // If we use UB for PQ ordering, we pick the node with highest potential.
        // If the top node has UB <= global_best, we are done (Optimality Proof).
        // Let's use UB for ordering. We need to store LB (actual solution value found) separately or re-evaluate.
        // Actually, let's stick to the structure:
        // Node.lb will store the Upper Bound.
        // We also need to find a feasible solution to update global_best.
        
        // Re-calculating values for the node (since we only stored assignment in node, or minimal info)
        // Actually, we stored assignment. 
        // Calculate current fixed value
        ll current_val = 0;
        for(int i=0; i<n; ++i) {
            if (node.assignment[i] == 1) current_val += c[i];
        }

        // Check consistency (should be consistent)
        // Neighbors of 1 must be 0
        // During branching, we enforced this.
        
        // Identify unassigned variables and their status
        vector<int> unassigned;
        for(int i=0; i<n; ++i) {
            if (node.assignment[i] == -1) unassigned.push_back(i);
        }

        // Calculate Upper Bound
        // UB = current_val + sum of positive weights of unassigned variables that are not adjacent to any fixed 1.
        // This relaxes constraints between unassigned variables.
        ll potential_gain = 0;
        bool can_improve = false;
        
        // To refine UB, we can use the greedy MWIS on the subgraph of unassigned variables.
        // This gives a solution (LB) and a value.
        // Is greedy MWIS value an Upper Bound? No, it's a Lower Bound.
        // Is sum of compatible positive weights an Upper Bound? Yes.
        // Let's compute the sum of compatible positive weights (Simple UB).
        // And also run a greedy pass to find a feasible solution (LB) to update global_best.
        
        // Check compatibility with fixed 1s
        vector<bool> is_compatible(n, true);
        for(int i=0; i<n; ++i) {
            if (node.assignment[i] == 1) {
                for(int neighbor : adj[i]) {
                    is_compatible[neighbor] = false;
                }
            }
        }
        
        ll ub_extra = 0;
        for(int u : unassigned) {
            if (is_compatible[u] && c[u] > 0) {
                ub_extra += c[u];
            }
        }
        
        ll ub = current_val + ub_extra;
        
        // Optimality check
        if (ub <= global_best) continue;
        
        // Try to improve global_best with a greedy solution on unassigned
        // Greedy MWIS on unassigned compatible vertices
        // We need to sort unassigned by weight?
        // Or just iterate and pick.
        // To keep it simple and efficient:
        // Make a local copy of relevant state for greedy
        // Since we are already iterating, a simple greedy is fast.
        
        vector<int> greedy_assignment = node.assignment; // Copy
        ll greedy_val = current_val;
        
        // Heuristic: pick high value vertices first?
        // We don't have them sorted by weight globally without sorting.
        // But we can just iterate.
        // For a better bound, we should pick max weight first.
        // Let's sort unassigned indices by weight descending.
        vector<int> sorted_unassigned = unassigned;
        sort(sorted_unassigned.begin(), sorted_unassigned.end(), [&](int a, int b) {
            return c[a] > c[b];
        });
        
        vector<int> status(n, 0); // 0: available, 1: taken, -1: blocked
        // Initialize status from assignment
        for(int i=0; i<n; ++i) {
            if (greedy_assignment[i] == 1) status[i] = 1;
            if (greedy_assignment[i] == 0) status[i] = -1; // blocked or just 0
        }
        // Note: if it's -1 in assignment, it's 0 in status (available)
        
        for(int u : sorted_unassigned) {
            if (status[u] != 0) continue; // blocked or taken
            
            // Check if we can take it
            bool possible = true;
            for(int v : adj[u]) {
                if (status[v] == 1) { // neighbor already taken
                    possible = false;
                    break;
                }
            }
            
            if (possible) {
                // Check weight (optimization: only take if positive?)
                // Actually, for MWIS, we only take if beneficial.
                if (c[u] > 0) {
                    status[u] = 1;
                    greedy_val += c[u];
                    for(int v : adj[u]) {
                        if (status[v] == 0) status[v] = -1; // block neighbors
                    }
                }
            }
        }
        
        if (greedy_val > global_best) {
            global_best = greedy_val;
            for(int i=0; i<n; ++i) {
                if (status[i] == 1) best_assignment[i] = 1;
                else if (status[i] == -1 && greedy_assignment[i] == 1) best_assignment[i] = 1; // shouldn't happen
                else if (status[i] == -1) best_assignment[i] = 0; // blocked
                else if (status[i] == 0) best_assignment[i] = 0; // available but not taken
                else best_assignment[i] = greedy_assignment[i]; // 1 or 0
            }
            // Simplified assignment construction
            // status[i] == 1 -> 1
            // else -> 0
            // Wait, we must respect original assignment.
            // The status array reflects the greedy choices + original assignment.
            for(int i=0; i<n; ++i) {
                best_assignment[i] = (status[i] == 1) ? 1 : 0;
            }
        }
        
        // Pruning again
        if (ub <= global_best) continue;
        
        // Branching
        if (unassigned.empty()) continue;
        
        // Pick a variable to branch on
        // Heuristic: pick variable with highest weight among unassigned?
        int branch_u = -1;
        int max_c = -1; // Use -1 to handle 0 weights correctly if all are 0?
                        // Actually we want to pick something that splits the problem well.
                        // Highest weight is a good candidate.
        for(int u : unassigned) {
            // If we pick a variable, we create two children:
            // x_u = 0: easy, just remove u.
            // x_u = 1: gain c[u], remove neighbors.
            // If c[u] is high, the bounds change significantly.
            if (c[u] > max_c) {
                max_c = c[u];
                branch_u = u;
            }
        }
        
        if (branch_u == -1) branch_u = unassigned[0]; // fallback
        
        // Create two children
        // Child 1: x_branch_u = 0
        Node child0;
        child0.assignment = node.assignment;
        child0.assignment[branch_u] = 0;
        child0.depth = node.depth + 1;
        // Calculate UB for child0 to use as priority key
        // We can simply propagate parent UB minus contribution of branch_u?
        // But branch_u might have been 0 in the UB calculation anyway if c[u] < 0.
        // Safe to recompute or just use parent UB logic.
        // Let's recompute simply:
        // current_val stays same.
        // unassigned reduces by 1.
        // compatible set loses branch_u.
        ll child0_val = current_val;
        ll child0_ub_extra = 0;
        bool child0_compatible;
        // We can just assume child0.lb is an Upper Bound estimate.
        // To avoid complex logic here, we just push it and evaluate at pop time.
        // We can estimate UB to push roughly correct.
        // Pushing with parent UB is safe (upper bound is valid for children).
        // But we want good ordering.
        
        // Heuristic push value:
        // For x_u = 0: UB loses c[u] (if c[u]>0).
        // For x_u = 1: UB is at least current_val + c[u]. Neighbors lose their potential.
        
        // Let's just push. We re-evaluate at start of loop.
        // To guide search better:
        // Child 0 (x=0): value roughly current + (UB_extra - c[u])
        // Child 1 (x=1): value roughly current + c[u] + (UB_extra of neighbors of neighbors - neighbors potential)
        
        // Simplified push logic:
        // For x=0: priority = current_val + (ub_extra - (c[branch_u] > 0 ? c[branch_u] : 0));
        // For x=1:
        //   gain = c[branch_u];
        //   loss = sum of positive c[v] for neighbors v in unassigned.
        //   priority = current_val + gain + (ub_extra - loss - (c[branch_u]>0 ? c[branch_u]:0));
        // Note: neighbors are lost, branch_u is gained.
        
        ll c_u = c[branch_u];
        ll c_u_pos = (c_u > 0) ? c_u : 0;
        
        // Child 0
        // We set x_u = 0. We lose potential c_u_pos.
        // New UB approx current_val + ub_extra - c_u_pos.
        // This is valid only if c_u was part of the compatible sum.
        // Since branch_u is unassigned, it was compatible iff not adjacent to fixed 1s.
        // We can check:
        bool u_was_compatible = true;
        for(int v : adj[branch_u]) if(node.assignment[v] == 1) u_was_compatible = false;
        
        ll push_val_0 = ub;
        if (u_was_compatible) push_val_0 -= c_u_pos;
        // push_val_0 is an Upper Bound.
        
        child0.lb = push_val_0;
        pq.push(child0);
        
        // Child 1
        // x_branch_u = 1
        // Feasibility: neighbors must be 0.
        // If any neighbor is 1, this node is infeasible.
        bool feasible = true;
        for(int v : adj[branch_u]) {
            if (node.assignment[v] == 1) {
                feasible = false;
                break;
            }
        }
        
        if (feasible) {
            Node child1;
            child1.assignment = node.assignment;
            child1.assignment[branch_u] = 1;
            for(int v : adj[branch_u]) {
                if (child1.assignment[v] == -1) child1.assignment[v] = 0;
            }
            child1.depth = node.depth + 1;
            
            // Estimate UB for child 1
            // Gain c_u.
            // Lose potential of unassigned neighbors (v where assignment was -1).
            ll loss = 0;
            for(int v : adj[branch_u]) {
                // if v was unassigned and compatible and positive
                // We need to check if v was compatible in parent node
                bool v_was_compatible = true;
                for(int w : adj[v]) if(node.assignment[w] == 1) v_was_compatible = false;
                
                if (node.assignment[v] == -1 && v_was_compatible && c[v] > 0) {
                    loss += c[v];
                }
            }
            
            ll push_val_1 = current_val + c_u + (ub_extra - loss);
            // We must also subtract c_u_pos from ub_extra if it was counted?
            // ub_extra counted branch_u if it was compatible and positive.
            // In child 1, branch_u is taken.
            // So new potential = (old potential - loss - c_u_pos) + c_u.
            // Correct update:
            // New Val = current + c_u
            // New Potential = ub_extra - items removed
            // Items removed: branch_u (if it was in potential), neighbors (if they were in potential).
            // New UB = current + c_u + (ub_extra - loss - c_u_pos).
            
            // Check if c_u was in ub_extra.
            // It was if u_was_compatible and c_u > 0.
            // If we take it, it moves from 'potential' to 'current'.
            // So current part increases by c_u. Potential decreases by c_u.
            // Net change from "current + potential" due to branch_u: +c_u - c_u_pos.
            // If c_u > 0, +c_u - c_u = 0.
            // If c_u <= 0, +c_u - 0 = c_u (decrease in bound, correct).
            
            // But we also lose neighbors potential 'loss'.
            // So New UB = ub + c_u - c_u_pos - loss.
            
            if (!u_was_compatible) {
                // c_u was not in ub_extra.
                // So we add c_u, subtract loss.
                push_val_1 = ub + c_u - loss;
            } else {
                push_val_1 = ub + c_u - c_u_pos - loss;
            }
            
            child1.lb = push_val_1;
            pq.push(child1);
        }
    }

    // Output result
    cout << global_best << "\n";
    for (int i = 0; i < n; ++i) {
        cout << best_assignment[i] << (i == n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
