; SPIR-V
Version: 1.0
Generator: 0
Bound: 100
Schema: 0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpSource GLSL 450
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inColor "color"
OpName %outColor "outColor"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; --- Input Interface ---
%gl_PerVertex_in = OpTypeStruct %v4float
%uint_3 = OpConstant %uint 3
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

; --- Output Interface ---
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; --- Constants ---
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_08 = OpConstant %float 0.08
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_333 = OpConstant %float 0.333333

; --- Pointer Types ---
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %3
%5 = OpLabel

; --- Load Positions ---
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %pos0_ptr
%p1 = OpLoad %v4float %pos1_ptr
%p2 = OpLoad %v4float %pos2_ptr

; --- Compute Face Normal ---
; Extract .xyz
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; e1 = p1 - p0
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
; e2 = p2 - p0
%e2 = OpFSub %v3float %p2_xyz %p0_xyz

; Cross Product: N = cross(e1, e2)
%N_raw = OpExtInst %v3float %1 Cross %e1 %e2
%N = OpExtInst %v3float %1 Normalize %N_raw

; Offset = N * 0.08
%offset = OpVectorTimesScalar %v3float %N %float_0_08

; --- Emit Vertex 0 ---
%color0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %color0_ptr
; p0 + vec4(offset, 0)
%pos0_out = OpCompositeConstruct %v4float %offset %float_0
%pos0_final = OpFAdd %v4float %p0 %pos0_out
; Store Output
%outPos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outPos0_ptr %pos0_final
OpStore %outColor %c0
OpEmitVertex

; --- Emit Vertex 1 ---
%color1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %color1_ptr
%pos1_out = OpCompositeConstruct %v4float %offset %float_0
%pos1_final = OpFAdd %v4float %p1 %pos1_out
%outPos1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outPos1_ptr %pos1_final
OpStore %outColor %c1
OpEmitVertex

; --- Emit Vertex 2 ---
%color2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2 = OpLoad %v3float %color2_ptr
%pos2_out = OpCompositeConstruct %v4float %offset %float_0
%pos2_final = OpFAdd %v4float %p2 %pos2_out
%outPos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outPos2_ptr %pos2_final
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
