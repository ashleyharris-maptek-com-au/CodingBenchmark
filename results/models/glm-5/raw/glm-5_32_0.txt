```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <string>
#include <fstream>
#include <limits>
#include <bitset>

// Optimization for speed
void speed_up() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
}

// Weighted Undirected Graph representation
struct Edge {
    int to;
    int weight;
};

// Constants
const int INF = 1e9;

int main() {
    speed_up();

    int num_vertices, num_edges, num_terminals;
    if (!(std::cin >> num_vertices >> num_edges >> num_terminals)) {
        return 0;
    }

    // Adjacency list for the graph
    std::vector<std::vector<Edge>> adj(num_vertices);

    for (int i = 0; i < num_edges; ++i) {
        int u, v, w;
        std::cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    // Read terminals
    std::vector<int> terminals(num_terminals);
    for (int i = 0; i < num_terminals; ++i) {
        std::cin >> terminals[i];
    }

    // If no terminals, weight is 0.
    if (num_terminals == 0) {
        std::cout << "0 0\n";
        return 0;
    }

    // Sort terminals to map them to indices 0..K-1 consistent with bits
    std::sort(terminals.begin(), terminals.end());

    // Heuristic Strategy:
    // 1. Compute shortest paths between all pairs of terminals.
    // 2. Construct a complete graph K where nodes are terminals and weights are shortest path distances.
    // 3. Compute Minimum Spanning Tree (MST) of K (using Prim's algorithm).
    // 4. Expand the MST edges back to the original graph to form the Steiner Tree.
    //    - Union shortest paths for all edges in the MST.
    //    - Compute MST on the induced subgraph to ensure acyclic property and minimal weight.

    // Step 1: Compute Shortest Paths from each terminal to all vertices
    // dists[t][v] = distance from terminal t to vertex v
    std::vector<std::vector<int>> dists(num_terminals, std::vector<int>(num_vertices, INF));
    std::vector<std::vector<int>> parent(num_terminals, std::vector<int>(num_vertices, -1));

    for (int i = 0; i < num_terminals; ++i) {
        int source = terminals[i];
        dists[i][source] = 0;
        // Min-priority queue: (distance, vertex)
        std::priority_queue<std::pair<int, int>, std::vector<std::pair<int, int>>, std::greater<std::pair<int, int>>> pq;
        pq.push({0, source});

        while (!pq.empty()) {
            auto top = pq.top();
            pq.pop();
            int d = top.first;
            int u = top.second;

            if (d > dists[i][u]) continue;

            for (const auto& edge : adj[u]) {
                if (dists[i][edge.to] > d + edge.weight) {
                    dists[i][edge.to] = d + edge.weight;
                    parent[i][edge.to] = u; // Store predecessor for path reconstruction
                    pq.push({dists[i][edge.to], edge.to});
                }
            }
        }
    }

    // Step 2 & 3: MST on the Metric Closure (Complete Graph on Terminals)
    // Prim's algorithm on the complete graph of terminals
    std::vector<int> min_edge_w(num_terminals, INF);
    std::vector<int> selected_source(num_terminals, -1);
    std::vector<bool> in_mst(num_terminals, false);

    min_edge_w[0] = 0; // Start from terminal 0
    
    int mst_edges_count = 0;
    // Stores pairs of terminal indices (index in sorted 'terminals' vector) that form the MST
    std::vector<std::pair<int, int>> mst_terminal_pairs;

    for (int i = 0; i < num_terminals; ++i) {
        int v = -1;
        // Find the vertex with minimum weight edge connecting it to the growing MST
        for (int j = 0; j < num_terminals; ++j) {
            if (!in_mst[j] && (v == -1 || min_edge_w[j] < min_edge_w[v])) {
                v = j;
            }
        }

        // If the smallest available edge is INF, graph is disconnected (shouldn't happen for valid input)
        if (min_edge_w[v] == INF) break; 

        in_mst[v] = true;
        mst_edges_count++;

        if (selected_source[v] != -1) {
            mst_terminal_pairs.push_back({selected_source[v], v});
        }

        // Update adjacent weights
        for (int to = 0; to < num_terminals; ++to) {
            if (!in_mst[to] && dists[v][terminals[to]] < min_edge_w[to]) {
                min_edge_w[to] = dists[v][terminals[to]];
                selected_source[to] = v;
            }
        }
    }

    // Step 4: Construct the Steiner Tree in the original graph
    // 4a. Identify all vertices and edges in the union of shortest paths
    std::vector<bool> used_vertices(num_vertices, false);
    
    // We extract edges. Using a set or sorting to handle duplicates (since we union paths).
    // Since N <= 100, O(N^2 log N) or O(N^2) is fine. We'll use a boolean matrix for edges.
    // Resize dynamically or use vector of vectors.
    std::vector<std::vector<bool>> edge_used(num_vertices, std::vector<bool>(num_vertices, false));
    std::vector<std::pair<int, int>> steiner_edges_list;

    for (const auto& pair : mst_terminal_pairs) {
        int u_idx = pair.first; // Index in 'terminals'
        int v_idx = pair.second; // Index in 'terminals'
        
        int target_u = terminals[u_idx];
        int target_v = terminals[v_idx];

        // Reconstruct path from target_v to target_u using the predecessor tree of u_idx
        // We traced paths from u_idx. 
        // The path is target_v -> ... -> target_u
        int curr = target_v;
        while (curr != target_u) {
            int prev = parent[u_idx][curr];
            if (prev == -1) break; // Should not happen if connectivity exists
            
            if (!edge_used[curr][prev]) {
                edge_used[curr][prev] = true;
                edge_used[prev][curr] = true;
                steiner_edges_list.push_back({curr, prev});
                used_vertices[curr] = true;
                used_vertices[prev] = true;
            }
            curr = prev;
        }
    }

    // 4b. The union of shortest paths might contain cycles. 
    // We must compute an MST on the induced subgraph to satisfy the "acyclic" requirement
    // and further minimize weight.
    
    // Map local indices for the induced subgraph
    std::vector<int> mapped_vertices; // mapped_vertices[i] -> original vertex id
    for(int i = 0; i < num_vertices; ++i) {
        if (used_vertices[i]) mapped_vertices.push_back(i);
    }
    int subgraph_size = mapped_vertices.size();

    // If subgraph size is small (e.g., 1 or 2), no cycle reduction needed really, but let's handle generally.
    if (subgraph_size > 1) {
        // Prim's on the induced subgraph
        // We need to recover original edge weights.
        // Build adjacency for used vertices to find weights quickly?
        // Or just iterate neighbors. Since degree sum is small, iterating neighbors is fine.
        
        std::vector<int> sub_min_w(subgraph_size, INF);
        std::vector<int> sub_parent(subgraph_size, -1);
        std::vector<bool> sub_visited(subgraph_size, false);
        
        sub_min_w[0] = 0;
        int mst_count = 0;
        long long final_weight = 0;
        std::vector<std::pair<int, int>> final_edges;

        // Optimization: Create a map from original vertex to subgraph index
        std::vector<int> original_to_sub(num_vertices, -1);
        for(int i = 0; i < subgraph_size; ++i) {
            original_to_sub[mapped_vertices[i]] = i;
        }

        while(mst_count < subgraph_size) {
            int v = -1;
            for(int i=0; i<subgraph_size; ++i) {
                if(!sub_visited[i] && (v == -1 || sub_min_w[i] < sub_min_w[v])) {
                    v = i;
                }
            }
            
            // Disconnected component in the union (shouldn't happen)
            if (sub_min_w[v] == INF) break; 

            sub_visited[v] = true;
            mst_count++;
            
            if (sub_parent[v] != -1) {
                final_weight += sub_min_w[v];
                final_edges.push_back({mapped_vertices[v], mapped_vertices[sub_parent[v]]});
            }

            // Update neighbors
            int orig_v = mapped_vertices[v];
            for(const auto& edge : adj[orig_v]) {
                // Only consider edges connecting to other used vertices
                if (original_to_sub[edge.to] != -1) {
                    int to = original_to_sub[edge.to];
                    if (!sub_visited[to] && edge.weight < sub_min_w[to]) {
                        sub_min_w[to] = edge.weight;
                        sub_parent[to] = v;
                    }
                }
            }
        }

        // Output results
        std::cout << final_weight << " " << final_edges.size() << "\n";
        for(const auto& edge : final_edges) {
            std::cout << edge.first << " " << edge.second << "\n";
        }

    } else {
        // Subgraph has 0 or 1 vertices (no edges)
        // Or terminals count was 1 initially.
        std::cout << 0 << " " << 0 << "\n";
    }

    return 0;
}
```
