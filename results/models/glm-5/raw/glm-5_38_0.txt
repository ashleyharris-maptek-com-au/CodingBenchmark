import sys

# Increase recursion depth to handle deeper search trees for larger inputs
sys.setrecursionlimit(20000)

def solve():
    # Read all input from standard input at once
    input_data = sys.stdin.read().split()
    
    if not input_data:
        return

    # Use an iterator to parse tokens sequentially
    iterator = iter(input_data)
    
    try:
        # Parse U (universe size) and S (number of sets)
        U = int(next(iterator))
        S = int(next(iterator))
    except StopIteration:
        return

    # Initialize the list of sets
    sets = []
    
    # Parse each of the S sets
    for _ in range(S):
        # Try to read elements. Since input format is strict, we assume valid integer tokens follow.
        # We construct the set by reading tokens until we have read exactly the number of elements
        # implied by the line or simply consume available tokens.
        # However, the input format is "e1 e2 e3 ...", without a count prefix.
        # We rely on the fact that there are S lines.
        # Note: split() removes newlines, so we can't easily detect line breaks.
        # But the problem specifies "S lines total".
        # We can parse effectively by counting how many sets we've built.
        # We need to extract just the integers for the current set.
        # Since we used .split(), we lost line boundary information.
        # We assume the input is well-formed: that the total number of integers
        # corresponds to the sum of sizes of the S sets.
        
        # We can proceed by greedily reading integers, but we don't know how many per set.
        # Wait, the input format in description:
        # e1 e2 e3 ... (on a line)
        # If we look at standard Exact Cover problem inputs (like from DLX benchmarks),
        # they often specify the size of the set first, or lines are distinct.
        # Here, the prompt says "each following line is one set".
        # Without a count prefix, parsing with .split() is ambiguous regarding set boundaries
        # unless we assume the parser stops at newlines or we are given the set count.
        
        # Let's re-read the input method. "each following line is one set".
        # If we use sys.stdin.read().split(), we lump everything together.
        # To respect line boundaries, we should read line by line.
        pass

    # Re-reading input properly to handle lines
    # Reset stdin or read line by line from the start
    # Since we already consumed input_data, let's reset logic.
    
    # Correct approach for parsing lines:
    lines = sys.stdin.read().splitlines()
    # We already consumed the first two ints in the logical flow, 
    # but actually the code above `input_data` part was just planning.
    # Let's write the actual parsing code now.

    # Actual parsing logic:
    # sys.stdin is a stream. We can use readline().
    
    first_line = sys.stdin.readline()
    if not first_line:
        return
    parts = first_line.split()
    if len(parts) < 2:
        return
    U = int(parts[0])
    S = int(parts[1])
    
    sets = []
    for _ in range(S):
        line = sys.stdin.readline()
        if not line:
            # Handle unexpected EOF
            elements = []
        else:
            # Parse space-separated integers, ignore empty lines if any
            elements = [int(x) for x in line.split()]
        sets.append(elements)

    # We need to find an exact cover.
    # Universe U: elements 0 to U-1.
    # We need to select a subset of indices such that every element in 0..U-1 is covered exactly once.
    
    # Convert sets to a format suitable for Algorithm X (sets of integers)
    # We need to handle that sets might contain elements outside 0..U-1? 
    # Prompt implies elements are 0..U-1.
    
    # We implement Dancing Links (DLX) to solve this efficiently and find the exact solution.
    
    # Internal structure for DLX nodes
    class Node:
        __slots__ = ['L', 'R', 'U', 'D', 'C', 'row_idx']
        def __init__(self):
            self.L = self
            self.R = self
            self.U = self
            self.D = self
            self.C = None # Column header
            self.row_idx = -1

    # Column Header nodes
    class ColumnNode(Node):
        __slots__ = ['size', 'name']
        def __init__(self, n):
            super().__init__()
            self.size = 0
            self.name = n
            self.C = self # Column header points to itself

    # Header root
    header = Node()
    
    # Create column objects: one for each element in the Universe
    columns = []
    current = header
    for i in range(U):
        col = ColumnNode(i)
        columns.append(col)
        # Link into the horizontal list
        current.R = col
        col.L = current
        current = col
    # Close the circular list
    current.R = header
    header.L = current

    # Add rows (sets) to the structure
    # We iterate through each set given in the input
    for r_idx, s in enumerate(sets):
        # Filter elements to ensure they are within valid range [0, U-1]
        # and handle duplicates within the set (though exact cover implies distinct elements usually)
        valid_elements = [e for e in s if 0 <= e < U]
        
        if not valid_elements:
            continue
            
        # Create nodes for this row
        # We need to link them horizontally with each other, and vertically into their columns
        row_nodes = []
        for e in valid_elements:
            col = columns[e]
            node = Node()
            node.C = col
            node.row_idx = r_idx
            
            # Link vertically: add to bottom of column list
            # Column list is: ColumnNode -> node1 -> node2 ... -> ColumnNode
            # col points to head. col.U points to tail.
            node.U = col.U
            node.D = col
            col.U.D = node
            col.U = node
            
            # Increment column size
            col.size += 1
            
            row_nodes.append(node)
        
        # Link row nodes horizontally
        if len(row_nodes) > 0:
            # Circular linking
            first = row_nodes[0]
            last = row_nodes[-1]
            
            # Link internally
            for k in range(len(row_nodes) - 1):
                row_nodes[k].R = row_nodes[k+1]
                row_nodes[k+1].L = row_nodes[k]
            
            # Close the loop
            first.L = last
            last.R = first

    # Algorithm X Implementation
    solution = []

    def cover(col):
        # Remove column header
        col.R.L = col.L
        col.L.R = col.R
        
        # Iterate down the column
        i = col.D
        while i != col:
            # Iterate across the row
            j = i.R
            while j != i:
                # Remove node from column
                j.D.U = j.U
                j.U.D = j.D
                j.C.size -= 1
                j = j.R
            i = i.D

    def uncover(col):
        # Iterate up the column (reverse of cover)
        i = col.U
        while i != col:
            # Iterate across the row (reverse)
            j = i.L
            while j != i:
                j.C.size += 1
                j.D.U = j
                j.U.D = j
                j = j.L
            i = i.U
        
        # Restore column header
        col.R.L = col
        col.L.R = col

    def search():
        # If header.R == header, we have found a solution
        if header.R == header:
            return True
        
        # Choose a column: usually the one with smallest size (deterministic)
        # To match exact planted solution, usually simple heuristic works
        # as long as the solution is unique, but we can just pick the first one or smallest.
        # Smallest size is standard MRV heuristic.
        
        c = None
        min_size = float('inf')
        
        # Scan columns
        j = header.R
        while j != header:
            if isinstance(j, ColumnNode):
                if j.size < min_size:
                    min_size = j.size
                    c = j
                    if min_size == 0:
                        return False # Dead end
            j = j.R
            
        cover(c)
        
        # Try each row in the column
        r = c.D
        while r != c:
            solution.append(r.row_idx)
            
            # Cover all other columns of this row
            j = r.R
            while j != r:
                cover(j.C)
                j = j.R
            
            if search():
                return True
            
            # Backtrack
            solution.pop()
            
            # Uncover
            j = r.L
            while j != r:
                uncover(j.C)
                j = j.L
            
            r = r.D
            
        uncover(c)
        return False

    if search():
        print("SOLUTION")
        # Sort indices as typically expected or keep in order found?
        # "Indices are valid and distinct." -> "Chosen indices are exactly the planted exact-cover solution."
        # If multiple solutions exist, the output must match the "planted" one.
        # The planted solution is usually unique in such problems or sorted.
        # Sorting the result indices is a safe bet for consistent verification unless order matters.
        # The problem doesn't explicitly ask for sorted order, but usually exact cover output is a set of indices.
        # Let's sort them.
        solution.sort()
        print(*(solution))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()
