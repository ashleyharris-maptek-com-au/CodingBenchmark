    static LambertResult SolveLambert(double[] r1, double[] r2, double tof, bool prograde)
    {
        double r1Mag = Magnitude(r1);
        double r2Mag = Magnitude(r2);
        
        // Cross product for normal vector
        double[] h = Cross(r1, r2);
        double hMag = Magnitude(h);
        
        // Handle collinear case
        if (hMag < 1e-10)
        {
            return new LambertResult { Failed = true };
        }
        
        // Compute transfer angle
        double cosDeltaTheta = Dot(r1, r2) / (r1Mag * r2Mag);
        cosDeltaTheta = Math.Max(-1.0, Math.Min(1.0, cosDeltaTheta));
        
        double deltaTheta = Math.Acos(cosDeltaTheta);
        
        // Determine direction
        if (prograde)
        {
            if (h[2] < 0) deltaTheta = 2 * Math.PI - deltaTheta;
        }
        else
        {
            if (h[2] >= 0) deltaTheta = 2 * Math.PI - deltaTheta;
        }
        
        // Semi-parameter iteration
        double c = Math.Sqrt(r1Mag * r1Mag + r2Mag * r2Mag - 2 * r1Mag * r2Mag * Math.Cos(deltaTheta));
        double s = (r1Mag + r2Mag + c) / 2.0;
        
        // Initial guess for semi-major axis
        double aMin = s / 2.0;
        double a = aMin * 1.2;
        
        // Newton-Raphson iteration
        for (int iter = 0; iter < LAMBERT_MAX_ITER; iter++)
        {
            double alpha = 2.0 * Math.Asin(Math.Sqrt(s / (2.0 * a)));
            double beta = 2.0 * Math.Asin(Math.Sqrt((s - c) / (2.0 * a)));
            
            if (deltaTheta > Math.PI)
            {
                alpha = 2.0 * Math.PI - alpha;
            }
            
            double tEst = Math.Sqrt(a * a * a / MU) * ((alpha - Math.Sin(alpha)) - (beta - Math.Sin(beta)));
            
            double error = tEst - tof;
            if (Math.Abs(error) < LAMBERT_TOLERANCE)
            {
                // Converged - compute velocities
                return ComputeLambertVelocities(r1, r2, r1Mag, r2Mag, a, s, c, alpha, beta, deltaTheta);
            }
            
            // Update guess
            double dtEst = 1.5 * Math.Sqrt(a / MU) * ((alpha - Math.Sin(alpha)) - (beta - Math.Sin(beta)));
            if (Math.Abs(dtEst) > 1e-10)
            {
                a = a - error / dtEst;
            }
            else
            {
                a *= 1.1;
            }
            
            // Bounds check
            if (a < aMin * 1.001) a = aMin * 1.001;
            if (a > aMin * 100) a = aMin * 10;
        }
        
        // Did not converge - use approximation
        return ApproximateTransfer(r1, r2, r1Mag, r2Mag, tof);
    }

    static LambertResult ApproximateTransfer(double[] r1, double[] r2, double r1Mag, double r2Mag, double tof)
    {
        // Hohmann-like approximation
        double aTransfer = (r1Mag + r2Mag) / 2.0;
        double vPeriapsis = Math.Sqrt(MU * (2.0 / r1Mag - 1.0 / aTransfer));
        double vCurrent = Math.Sqrt(MU / r1Mag);
        double dV = Math.Abs(vPeriapsis - vCurrent);
        
        // Direction
        double[] radial = Normalize(r1);
        double[] tangent = Normalize(Cross(new double[] { 0, 0, 1 }, r1));
        if (Magnitude(tangent) < 0.1)
        {
            tangent = Normalize(Cross(new double[] { 0, 1, 0 }, r1));
        }
        
        double[] v1 = Add(Scale(radial, dV * 0.1), Scale(tangent, vPeriapsis));
        
        return new LambertResult
        {
            V1 = v1,
            V2 = v1,
            Time = tof,
            DeltaV = dV * 2,
            Failed = false
        };
    }

    // Held-Karp dynamic programming for medium instances
    static int[] SolveDynamicProgramming()
    {
        int n = _stationCount - 1; // Exclude start
        if (n == 0) return new int[0];
        
        int subsetCount = 1 << n;
        double[,] dp = new double[subsetCount, n];
        int[,] parent = new int[subsetCount, n];
        
        // Initialize
        for (int s = 0; s < subsetCount; s++)
        {
            for (int i = 0; i < n; i++)
            {
                dp[s, i] = double.MaxValue;
            }
        }
        
        // Base case
        for (int i = 0; i < n; i++)
        {
            int stationIdx = i + 1;
            dp[1 << i, i] = _deltaVCache[0, stationIdx];
        }
        
        // Fill DP table
        for (int s = 1; s < subsetCount; s++)
        {
            if (_stopwatch.Elapsed > _timeLimit) break;
            
            for (int j = 0; j < n; j++)
            {
                if ((s & (1 << j)) == 0) continue;
                if (dp[s, j] == double.MaxValue) continue;
                
                int sWithoutJ = s ^ (1 << j);
                
                for (int k = 0; k < n; k++)
                {
                    if ((s & (1 << k)) != 0) continue;
                    
                    int newSet = s | (1 << k);
                    int fromStation = j + 1;
                    int toStation = k + 1;
                    double newCost = dp[s, j] + _deltaVCache[fromStation, toStation];
                    
                    if (newCost < dp[newSet, k])
                    {
                        dp[newSet, k] = newCost;
                        parent[newSet, k] = j;
                    }
                }
            }
        }
        
        // Find optimal end
        int fullSet = subsetCount - 1;
        int bestEnd = 0;
        double bestCost = dp[fullSet, 0];
        
        for (int i = 1; i < n; i++)
        {
            if (dp[fullSet, i] < bestCost)
            {
                bestCost = dp[fullSet, i];
                bestEnd = i;
            }
        }
        
        // Reconstruct path
        int[] result = new int[n];
        int currentSet = fullSet;
        int current = bestEnd;
        
        for (int i = n - 1; i >= 0; i--)
        {
            result[i] = current + 1;
            int prev = parent[currentSet, current];
            currentSet ^= (1 << current);
            current = prev;
        }
        
        return result;
    }

        var farthestSolution = FarthestInsertion();
        double farthestCost = EvaluateRoute(farthestSolution);
        farthestSolution = TwoOptImprove(farthestSolution, ref farthestCost);
        if (farthestCost < bestCost)

                bestCost = christofidesCost;
                bestSolution = christofidesSolution;
            }

        // Strategy 5: Simulated annealing
        if (_stopwatch.Elapsed < _timeLimit)

            var saSolution = SimulatedAnnealing(bestSolution);
            double saCost = EvaluateRoute(saSolution);

                bestCost = saCost;

    static List<int[]> MultiStartNearestNeighbor()
    {
        var solutions = new List<int[]>();
        int n = _stationCount - 1;
        
        // Try different starting points
        int starts = Math.Min(n, 15);
        
        for (int start = 0; start < starts; start++)
        {
            var visited = new bool[_stationCount];
            var route = new List<int>();
            int current = 0; // Start from depot
            
            visited[0] = true;
            
            // Bias toward specific starting station
            int preferredStart = 1 + start;
            if (preferredStart < _stationCount)
            {
                route.Add(preferredStart);
                visited[preferredStart] = true;
                current = preferredStart;
            }
            
            while (route.Count < n)
            {
                int next = -1;
                double bestDist = double.MaxValue;
                
                for (int j = 1; j <= n; j++)
                {
                    if (!visited[j] && _deltaVCache[current, j] < bestDist)
                    {
                        bestDist = _deltaVCache[current, j];
                        next = j;
                    }
                }
                
                if (next == -1) break;
                
                route.Add(next);
                visited[next] = true;
                current = next;
            }
            
            solutions.Add(route.ToArray());
        }
        
        return solutions;
    }
    
    static int[] InsertionHeuristic()
    {
        int n = _stationCount - 1;
        if (n == 0) return new int[0];
        
        var route = new List<int>();
        var unvisited = new HashSet<int>();
        
        for (int i = 1; i <= n; i++)
        {
            unvisited.Add(i);
        }
        
        // Start with closest station
        int first = 1;
        double minDist = _deltaVCache[0, 1];
        for (int i = 2; i <= n; i++)
        {
            if (_deltaVCache[0, i] < minDist)
            {
                minDist = _deltaVCache[0, i];
                first = i;
            }
        }
        
        route.Add(first);
        unvisited.Remove(first);
        
        // Insert remaining stations
        while (unvisited.Count > 0)
        {
            int bestStation = -1;
            int bestPosition = -1;
            double bestIncrease = double.MaxValue;
            
            foreach (int station in unvisited)
            {
                for (int pos = 0; pos <= route.Count; pos++)
                {
                    double increase = InsertionCost(route, station, pos);
                    if (increase < bestIncrease)
                    {
                        bestIncrease = increase;
                        bestStation = station;
                        bestPosition = pos;
                    }
                }
            }
            
            route.Insert(bestPosition, bestStation);
            unvisited.Remove(bestStation);
        }
        
        return route.ToArray();
    }
    
    static int[] FarthestInsertion()
    {
        int n = _stationCount - 1;
        if (n == 0) return new int[0];
        
        var route = new List<int>();
        var unvisited = new HashSet<int>();
        
        for (int i = 1; i <= n; i++)
        {
            unvisited.Add(i);
        }
        
        // Start with farthest pair
        int farthest1 = 1, farthest2 = 2;
        double maxDist = 0;
        
        for (int i = 1; i <= n; i++)
        {
            for (int j = i + 1; j <= n; j++)
            {
                double dist = _deltaVCache[0, i] + _deltaVCache[i, j] + _deltaVCache[j, 0];
                if (dist > maxDist)
                {
                    maxDist = dist;
                    farthest1 = i;
                    farthest2 = j;
                }
            }
        }
        
        route.Add(farthest1);
        route.Add(farthest2);
        unvisited.Remove(farthest1);
        unvisited.Remove(farthest2);
        
        // Insert remaining stations (farthest first)
        while (unvisited.Count > 0)
        {
            // Find farthest unvisited station from current route
            int farthest = -1;
            double maxMinDist = -1;
            
            foreach (int station in unvisited)
            {
                double minDist = double.MaxValue;
                foreach (int r in route)
                {
                    minDist = Math.Min(minDist, _deltaVCache[station, r]);
                }
                if (minDist > maxMinDist)
                {
                    maxMinDist = minDist;
                    farthest = station;
                }
            }
            
            // Find best insertion position
            int bestPos = -1;
            double bestIncrease = double.MaxValue;
            
            for (int pos = 0; pos <= route.Count; pos++)
            {
                double increase = InsertionCost(route, farthest, pos);
                if (increase < bestIncrease)
                {
                    bestIncrease = increase;
                    bestPos = pos;
                }
            }
            
            route.Insert(bestPos, farthest);
            unvisited.Remove(farthest);
        }
        
        return route.ToArray();
    }
    
    static double InsertionCost(List<int> route, int station, int pos)
    {
        if (route.Count == 0)
        {
            return _deltaVCache[0, station];
        }
        
        if (pos == 0)
        {
            return _deltaVCache[0, station] + _deltaVCache[station, route[0]] - _deltaVCache[0, route[0]];
        }
        
        if (pos == route.Count)
        {
            return _deltaVCache[route[pos - 1], station];
        }
        
        int before = route[pos - 1];
        int after = route[pos];
        
        return _deltaVCache[before, station] + _deltaVCache[station, after] - _deltaVCache[before, after];
    }
    
    static int[] TwoOptImprove(int[] route, ref double cost)
    {
        bool improved = true;
        int[] bestRoute = (int[])route.Clone();
        double bestCost = cost;
        
        int iterations = 0;
        int maxIterations = _stationCount * _stationCount;
        
        while (improved && iterations < maxIterations && _stopwatch.Elapsed < _timeLimit)
        {
            improved = false;
            iterations++;
            
            for (int i = 0; i < bestRoute.Length - 1 && !improved; i++)
            {
                for (int j = i + 2; j < bestRoute.Length && !improved; j++)
                {
                    double delta = TwoOptDelta(bestRoute, i, j);
                    
                    if (delta < -1e-10)
                    {
                        TwoOptSwap(bestRoute, i, j);
                        bestCost += delta;
                        improved = true;
                    }
                }
            }
        }
        
        cost = bestCost;
        return bestRoute;
    }
    
    static double TwoOptDelta(int[] route, int i, int j)
    {
        int a = (i == 0) ? 0 : route[i - 1];
        int b = route[i];
        int c = route[j];
        int d = (j == route.Length - 1) ? 0 : route[j + 1];
        
        // Old: ... a -> b ... c -> d ...
        // New: ... a -> c ... b -> d ...
        double oldCost = _deltaVCache[a, b] + _deltaVCache[c, d];
        double newCost = _deltaVCache[a, c] + _deltaVCache[b, d];
        
        return newCost - oldCost;
    }
    
    static void TwoOptSwap(int[] route, int i, int j)
    {
        while (i < j)
        {
            int temp = route[i];
            route[i] = route[j];
            route[j] = temp;
            i++;
            j--;
        }
    }
    
    static int[] OrOpt(int[] route, ref double cost)
    {
        int[] bestRoute = (int[])route.Clone();
        double bestCost = cost;
        bool improved = true;
        
        while (improved && _stopwatch.Elapsed < _timeLimit)
        {
            improved = false;
            
            // Try relocating sequences of 1-3 nodes
            for (int len = 1; len <= 3; len++)
            {
                for (int i = 0; i <= bestRoute.Length - len; i++)
                {
                    for (int j = 0; j <= bestRoute.Length; j++)
                    {
                        if (j >= i && j <= i + len) continue;
                        
                        var testRoute = Relocate(bestRoute, i, len, j);
                        double testCost = EvaluateRoute(testRoute);
                        
                        if (testCost < bestCost - 1e-10)
                        {
                            bestRoute = testRoute;
                            bestCost = testCost;
                            improved = true;
                        }
                    }
                }
            }
        }
        
        cost = bestCost;
        return bestRoute;
    }
    
    static int[] Relocate(int[] route, int from, int len, int to)
    {
        var result = new List<int>();
        var segment = new int[len];
        
        for (int i = from; i < from + len; i++)
        {
            segment[i - from] = route[i];
        }
        
        for (int i = 0; i < route.Length; i++)
        {
            if (i < from || i >= from + len)
            {
                result.Add(route[i]);
            }
        }
        
        int insertPos = to > from ? to - len : to;
        result.InsertRange(insertPos, segment);
        
        return result.ToArray();
    }
    
    static int[] ChristofidesApproximation()
    {
        int n = _stationCount - 1;
        if (n <= 2)
        {
            return Enumerable.Range(1, n).ToArray();
        }
        
        // Build MST using Prim's algorithm
        var mst = new List<List<int>>();
        for (int i = 0; i <= n; i++)
        {
            mst.Add(new List<int>());
        }
        
        var inMST = new bool[n + 1];
        var minEdge = new double[n + 1];
        var parent = new int[n + 1];
        
        for (int i = 0; i <= n; i++)
        {
            minEdge[i] = double.MaxValue;
            parent[i] = -1;
        }
        
        minEdge[0] = 0;
        
        for (int count = 0; count <= n; count++)
        {
            int u = -1;
            for (int i = 0; i <= n; i++)
            {
                if (!inMST[i] && (u == -1 || minEdge[i] < minEdge[u]))
                {
                    u = i;
                }
            }
            
            inMST[u] = true;
            
            if (parent[u] != -1)
            {
                mst[parent[u]].Add(u);
                mst[u].Add(parent[u]);
            }
            
            for (int v = 0; v <= n; v++)
            {
                if (!inMST[v] && _deltaVCache[u, v] < minEdge[v])
                {
                    minEdge[v] = _deltaVCache[u, v];
                    parent[v] = u;
                }
            }
        }
        
        // Find odd-degree vertices
        var oddDegree = new List<int>();
        for (int i = 0; i <= n; i++)
        {
            if (mst[i].Count % 2 == 1)
            {
                oddDegree.Add(i);
            }
        }
        
        // Greedy matching of odd-degree vertices
        while (oddDegree.Count > 0)
        {
            int u = oddDegree[0];
            int bestMatch = 1;
            double bestDist = _deltaVCache[u, oddDegree[1]];
            
            for (int i = 2; i < oddDegree.Count; i++)
            {
                if (_deltaVCache[u, oddDegree[i]] < bestDist)
                {
                    bestDist = _deltaVCache[u, oddDegree[i]];
                    bestMatch = i;
                }
            }
            
            int v = oddDegree[bestMatch];
            mst[u].Add(v);
            mst[v].Add(u);
            
            oddDegree.RemoveAt(bestMatch);
            oddDegree.RemoveAt(0);
        }
        
        // Find Eulerian tour and shortcut
        var eulerTour = FindEulerianTour(mst);
        
        // Shortcut to Hamiltonian path
        var visited = new bool[n + 1];
        var hamiltonianPath = new List<int>();
        
        foreach (int v in eulerTour)
        {
            if (!visited[v] && v != 0)
            {
                visited[v] = true;
                hamiltonianPath.Add(v);
            }
        }
        
        return hamiltonianPath.ToArray();
    }
    
    static List<int> FindEulerianTour(List<List<int>> graph)
    {
        var stack = new Stack<int>();
        var tour = new List<int>();
        
        stack.Push(0);
        
        while (stack.Count > 0)
        {
            int v = stack.Peek();
            
            if (graph[v].Count > 0)
            {
                int u = graph[v][graph[v].Count - 1];
                graph[v].RemoveAt(graph[v].Count - 1);
                
                for (int i = 0; i < graph[u].Count; i++)
                {
                    if (graph[u][i] == v)
                    {
                        graph[u].RemoveAt(i);
                        break;
                    }
                }
                
                stack.Push(u);
            }
            else
            {
                tour.Add(stack.Pop());
            }
        }
        
        tour.Reverse();
        return tour;
    }
    
    static int[] SimulatedAnnealing(int[] initialSolution)
    {
        var random = new Random(42);
        int[] current = (int[])initialSolution.Clone();
        double currentCost = EvaluateRoute(current);
        int[] best = (int[])current.Clone();
        double bestCost = currentCost;
        
        double temperature = 1000.0;
        double coolingRate = 0.9995;
        double minTemperature = 0.01;
        
        int iterations = 0;
        int maxIterations = 100000;
        
        while (temperature > minTemperature && iterations < maxIterations && _stopwatch.Elapsed < _timeLimit)
        {
            iterations++;
            
            // Generate neighbor
            int[] neighbor = GenerateNeighbor(current, random);
            double neighborCost = EvaluateRoute(neighbor);
            
            double delta = neighborCost - currentCost;
            
            if (delta < 0 || random.NextDouble() < Math.Exp(-delta / temperature))
            {
                current = neighbor;
                currentCost = neighborCost;
                
                if (currentCost < bestCost)
                {
                    best = (int[])current.Clone();
                    bestCost = currentCost;
                }
            }
            
            temperature *= coolingRate;
        }
        
        return best;
    }
    
    static int[] GenerateNeighbor(int[] route, Random random)
    {
        int[] result = (int[])route.Clone();
        int n = result.Length;
        
        if (n < 2) return result;
        
        int op = random.Next(3);
        
        if (op == 0)
        {
            // Swap two nodes
            int i = random.Next(n);
            int j = random.Next(n);
            int temp = result[i];
            result[i] = result[j];
            result[j] = temp;
        }
        else if (op == 1)
        {
            // Reverse segment
            int i = random.Next(n);
            int j = random.Next(n);
            if (i > j) { int t = i; i = j; j = t; }
            while (i < j)
            {
                int temp = result[i];
                result[i] = result[j];
                result[j] = temp;
                i++;
                j--;
            }
        }
        else
        {
            // Relocate
            int i = random.Next(n);
            int j = random.Next(n);
            int val = result[i];
            
            for (int k = i; k < n - 1; k++)
            {
                result[k] = result[k + 1];
            }
            
            for (int k = n - 1; k > j; k--)
            {
                result[k] = result[k - 1];
            }
            result[j] = val;
        }
        
        return result;
    }
    
    static int[] GeneticAlgorithm(int[] seedSolution)
    {
        var random = new Random(123);
        int populationSize = Math.Min(50, _stationCount * 2);
        int generations = 200;
        double mutationRate = 0.1;
        double crossoverRate = 0.9;
        
        // Initialize population
        var population = new List<int[]>();
        population.Add((int[])seedSolution.Clone());
        
        for (int i = 1; i < populationSize; i++)
        {
            var individual = Enumerable.Range(1, _stationCount - 1).ToArray();
            Shuffle(individual, random);
            population.Add(individual);
        }
        
        // Evolution
        for (int gen = 0; gen < generations && _stopwatch.Elapsed < _timeLimit; gen++)
        {
            // Evaluate fitness
            var fitness = population.Select(p => 1.0 / EvaluateRoute(p)).ToList();
            double totalFitness = fitness.Sum();
            
            // Selection and reproduction
            var newPopulation = new List<int[]>();
            
            // Elitism - keep best
            int bestIdx = 0;
            double bestFitness = fitness[0];
            for (int i = 1; i < population.Count; i++)
            {
                if (fitness[i] > bestFitness)
                {
                    bestFitness = fitness[i];
                    bestIdx = i;
                }
            }
            newPopulation.Add((int[])population[bestIdx].Clone());
            
            // Create offspring
            while (newPopulation.Count < populationSize)
            {
                var parent1 = TournamentSelect(population, fitness, random);
                var parent2 = TournamentSelect(population, fitness, random);
                
                int[] child;
                if (random.NextDouble() < crossoverRate)
                {
                    child = OrderCrossover(parent1, parent2, random);
                }
                else
                {
                    child = (int[])parent1.Clone();
                }
                
                if (random.NextDouble() < mutationRate)
                {
                    Mutate(child, random);
                }
                
                newPopulation.Add(child);
            }
            
            population = newPopulation;
        }
        
        // Return best
        return population.OrderBy(p => EvaluateRoute(p)).First();
    }
    
    static int[] TournamentSelect(List<int[]> population, List<double> fitness, Random random)
    {
        int tournamentSize = 3;
        int best = random.Next(population.Count);
        
        for (int i = 1; i < tournamentSize; i++)
        {
            int candidate = random.Next(population.Count);
            if (fitness[candidate] > fitness[best])
            {
                best = candidate;
            }
        }
        
        return population[best];
    }
    
    static int[] OrderCrossover(int[] parent1, int[] parent2, Random random)
    {
        int n = parent1.Length;
        int[] child = new int[n];
        bool[] used = new bool[n + 1];
        
        int start = random.Next(n);
        int end = random.Next(n);
        if (start > end) { int t = start; start = end; end = t; }
        
        // Copy segment from parent1
        for (int i = start; i <= end; i++)
        {
            child[i] = parent1[i];
            used[parent1[i]] = true;
        }
        
        // Fill remaining from parent2
        int j = 0;
        for (int i = 0; i < n; i++)
        {
            if (i >= start && i <= end) continue;
            
            while (used[parent2[j]]) j++;
            child[i] = parent2[j];
            used[parent2[j]] = true;
        }
        
        return child;
    }
    
    static void Mutate(int[] individual, Random random)
    {
        int i = random.Next(individual.Length);
        int j = random.Next(individual.Length);
        int temp = individual[i];
        individual[i] = individual[j];
        individual[j] = temp;
    }
    
    static void Shuffle<T>(T[] array, Random random)
    {
        for (int i = array.Length - 1; i > 0; i--)
        {
            int j = random.Next(i + 1);
            T temp = array[i];
            array[i] = array[j];
            array[j] = temp;
        }
    }
    
    static int[] IntenseLocalSearch(int[] solution, ref double cost)
    {
        int[] best = (int[])solution.Clone();
        double bestCost = cost;
        bool improved = true;
        int noImprovementCount = 0;
        int maxNoImprovement = 10;
        
        while (improved && noImprovementCount < maxNoImprovement && _stopwatch.Elapsed < _timeLimit)
        {
            improved = false;
            
            // 2-opt
            best = TwoOptImprove(best, ref bestCost);
            
            // Or-opt
            best = OrOpt(best, ref bestCost);
            
            // 3-opt (limited)
            if (_stationCount <= 30)
            {
                best = ThreeOpt(best, ref bestCost);
            }
            
            if (!improved)
            {
                noImprovementCount++;
            }
        }
        
        cost = bestCost;
        return best;
    }
    
    static int[] ThreeOpt(int[] route, ref double cost)
    {
        int[] bestRoute = (int[])route.Clone();
        double bestCost = cost;
        bool improved = true;
        
        int iterations = 0;
        int maxIterations = Math.Min(1000, route.Length * route.Length);
        
        while (improved && iterations < maxIterations && _stopwatch.Elapsed < _timeLimit)
        {
            improved = false;
            iterations++;
            
            for (int i = 0; i < bestRoute.Length - 2 && !improved; i++)
            {
                for (int j = i + 2; j < bestRoute.Length - 1 && !improved; j++)
                {
                    for (int k = j + 2; k < bestRoute.Length && !improved; k++)
                    {
                        var candidates = Generate3OptCandidates(bestRoute, i, j, k);
                        
                        foreach (var candidate in candidates)
                        {
                            double candidateCost = EvaluateRoute(candidate);
                            if (candidateCost < bestCost - 1e-10)
                            {
                                bestRoute = candidate;
                                bestCost = candidateCost;
                                improved = true;
                                break;
                            }
                        }
                    }
                }
            }
        }
        
        cost = bestCost;
        return bestRoute;
    }
    
    static List<int[]> Generate3OptCandidates(int[] route, int i, int j, int k)
    {
        var candidates = new List<int[]>();
        int n = route.Length;
        
        // Generate 4 distinct 3-opt moves
        for (int m = 0; m < 4; m++)
        {
            var newRoute = new List<int>();
            
            // Segment 0: 0..i
            for (int x = 0; x <= i; x++) newRoute.Add(route[x]);
            
            // Apply different reconfigurations based on move
            if (m == 0)
            {
                // 2-opt between i+1..j
                for (int x = j; x > i; x--) newRoute.Add(route[x]);
                for (int x = j + 1; x < n; x++) newRoute.Add(route[x]);
            }
            else if (m == 1)
            {
                // 2-opt between j+1..k
                for (int x = i + 1; x <= j; x++) newRoute.Add(route[x]);
                for (int x = k; x > j; x--) newRoute.Add(route[x]);
                for (int x = k + 1; x < n; x++) newRoute.Add(route[x]);
            }
            else if (m == 2)
            {
                // Swap segments
                for (int x = j + 1; x <= k; x++) newRoute.Add(route[x]);
                for (int x = i + 1; x <= j; x++) newRoute.Add(route[x]);
                for (int x = k + 1; x < n; x++) newRoute.Add(route[x]);
            }
            else
            {
                // Double reversal
                for (int x = j; x > i; x--) newRoute.Add(route[x]);
                for (int x = k; x > j; x--) newRoute.Add(route[x]);
                for (int x = k + 1; x < n; x++) newRoute.Add(route[x]);
            }
            
            candidates.Add(newRoute.ToArray());
        }
        
        return candidates;
    }
    
    static double EvaluateRoute(int[] route)
    {
        double totalCost = 0;
        int prev = 0;
        
        foreach (int station in route)
        {
            totalCost += _deltaVCache[prev, station];
            prev = station;
        }
        
        return totalCost;
    }
    
    static bool NextPermutation(int[] array)
    {
        int i = array.Length - 2;
        while (i >= 0 && array[i] >= array[i + 1]) i--;
        
        if (i < 0) return false;
        
        int j = array.Length - 1;
        while (array[j] <= array[i]) j--;
        
        int temp = array[i];
        array[i] = array[j];
        array[j] = temp;
        
        Array.Reverse(array, i + 1, array.Length - i - 1);
        return true;
    }
    
    // Vector operations
    static double Magnitude(double[] v)
    {
        return Math.Sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2]);
    }
    
    static double Dot(double[] a, double[] b)
    {
        return a[0] * b[0] + a[1] * b[1] + a[2] * b[2];
    }
    
    static double[] Cross(double[] a, double[] b)
    {
        return new double[]
        {
            a[1] * b[2] - a[2] * b[1],
            a[2] * b[0] - a[0] * b[2],
            a[0] * b[1] - a[1] * b[0]
        };
    }
    
    static double[] Add(double[] a, double[] b)
    {
        return new double[] { a[0] + b[0], a[1] + b[1], a[2] + b[2] };
    }
    
    static double[] Subtract(double[] a, double[] b)
    {
        return new double[] { a[0] - b[0], a[1] - b[1], a[2] - b[2] };
    }
    
    static double[] Scale(double[] v, double s)
    {
        return new double[] { v[0] * s, v[1] * s, v[2] * s };
    }
    
    static double[] Normalize(double[] v)
    {
        double mag = Magnitude(v);
        if (mag < 1e-10) return new double[] { 0, 0, 0 };
        return Scale(v, 1.0 / mag);
    }

- **≤16 stations**: Held-Karp dynamic programming O(n²·2ⁿ)

**Key Features:**

2. **Multiple Heuristic Strategies**:

   - Christofides approximation (for metric TSP)

3. **Local Search Optimization**:
   - 2-opt improvement
