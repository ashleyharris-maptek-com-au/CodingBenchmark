Here's a complete, optimized maze solver in Rust that handles mazes from 5x5 to 50,000x50,000 efficiently:

```rust
use std::io::{self, BufRead, BufWriter, Write, BufReader};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut stdout = BufWriter::new(stdout.lock());
    
    let mut lines = BufReader::new(stdin.lock()).lines();
    
    // Parse dimensions
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let h: usize = parts.next().unwrap().parse().unwrap();
    let w: usize = parts.next().unwrap().parse().unwrap();
    
    // Allocate a single flat buffer: each cell stores 4 bits
    // Bit 0-3: visited directions (up/down/left/right) for BFS path reconstruction
    // Bit 4: is wall
    // Bit 5: is start (A)
    // Bit 6: is end (B)
    const WALL: u8 = 0x10;
    const START: u8 = 0x20;
    const END: u8 = 0x40;
    const DIR_UP: u8 = 0x01;
    const DIR_DOWN: u8 = 0x02;
    const DIR_LEFT: u8 = 0x04;
    const DIR_RIGHT: u8 = 0x08;
    
    let mut grid: Vec<u8> = vec![0; h * w];
    let mut start_pos: usize = 0;
    let mut end_pos: usize = 0;
    let mut start_found = false;
    let mut end_found = false;
    
    // Parse maze
    for (row, line_result) in lines.take(h).enumerate() {
        let line = line_result.unwrap();
        let row_start = row * w;
        
        for (col, ch) in line.as_bytes().iter().take(w).enumerate() {
            let idx = row_start + col;
            match *ch {
                b'#' => grid[idx] = WALL,
                b'A' => {
                    start_pos = idx;
                    start_found = true;
                }
                b'B' => {
                    grid[idx] = END;
                    end_pos = idx;
                    end_found = true;
                }
                _ => {} // Space or anything else is open
            }
        }
    }
    
    // Handle edge cases
    if !start_found || !end_found {
        writeln!(stdout, "0").unwrap();
        return;
    }
    
    // Special case: start == end
    if start_pos == end_pos {
        writeln!(stdout, "1").unwrap();
        writeln!(stdout, "{} {}", start_pos % w, start_pos / w).unwrap();
        return;
    }
    
    // BFS with in-place visited tracking
    // We reuse the grid to store which direction we came from
    let mut queue: VecDeque<usize> = VecDeque::with_capacity((h * w).min(1_000_000));
    queue.push_back(start_pos);
    
    let mut found = false;
    
    while let Some(pos) = queue.pop_front() {
        let row = pos / w;
        let col = pos % w;
        let cell = grid[pos];
        
        // Try all 4 directions
        // Up
        if row > 0 {
            let next = pos - w;
            let next_cell = grid[next];
            if next_cell & WALL == 0 && next_cell < DIR_UP {
                // Not wall and not visited
                grid[next] |= DIR_DOWN; // Came from below
                if next_cell & END != 0 {
                    found = true;
                    break;
                }
                queue.push_back(next);
            }
        }
        
        // Down
        if row + 1 < h {
            let next = pos + w;
            let next_cell = grid[next];
            if next_cell & WALL == 0 && next_cell < DIR_UP {
                grid[next] |= DIR_UP; // Came from above
                if next_cell & END != 0 {
                    found = true;
                    break;
                }
                queue.push_back(next);
            }
        }
        
        // Left
        if col > 0 {
            let next = pos - 1;
            let next_cell = grid[next];
            if next_cell & WALL == 0 && next_cell < DIR_UP {
                grid[next] |= DIR_RIGHT; // Came from right
                if next_cell & END != 0 {
                    found = true;
                    break;
                }
                queue.push_back(next);
            }
        }
        
        // Right
        if col + 1 < w {
            let next = pos + 1;
            let next_cell = grid[next];
            if next_cell & WALL == 0 && next_cell < DIR_UP {
                grid[next] |= DIR_LEFT; // Came from left
                if next_cell & END != 0 {
                    found = true;
                    break;
                }
                queue.push_back(next);
            }
        }
    }
    
    if !found {
        writeln!(stdout, "0").unwrap();
        return;
    }
    
    // Reconstruct path from end to start
    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut current = end_pos;
    
    // Add end position
    path.push((current % w, current / w));
    
    while current != start_pos {
        let dir = grid[current] & 0x0F;
        let row = current / w;
        let col = current % w;
        
        let prev = match dir {
            DIR_UP => current - w,    // Came from above, go up
            DIR_DOWN => current + w,  // Came from below, go down
            DIR_LEFT => current - 1,  // Came from left, go left
            DIR_RIGHT => current + 1, // Came from right, go right
            _ => break, // Shouldn't happen
        };
        
        path.push((prev % w, prev / w));
        current = prev;
    }
    
    // Reverse path (currently end -> start, need start -> end)
    path.reverse();
    
    // Output
    writeln!(stdout, "{}", path.len()).unwrap();
    for (x, y) in path {
        writeln!(stdout, "{} {}", x, y).unwrap();
    }
}
```

## Key Optimizations

1. **Flat vector storage**: Uses a single `Vec<u8>` instead of 2D vectors, reducing allocation overhead and improving cache locality.

2. **Bit-packing**: Each cell stores both the wall info and visited direction in a single byte:
   - Bits 0-3: Direction we came from (for path reconstruction)
   - Bit 4: Wall flag
   - Bit 5: Start flag
   - Bit 6: End flag

3. **Single-pass BFS**: The BFS stores the direction we came from directly in the grid, eliminating the need for a separate parent array.

4. **Early termination**: Stops immediately when reaching the goal, rather than exploring the entire maze.

5. **Buffered I/O**: Uses `BufReader` and `BufWriter` for efficient I/O on large inputs.

6. **Memory-efficient queue**: Uses `VecDeque` with pre-allocation to avoid repeated reallocations.

## Complexity

- **Time**: O(H × W) - each cell visited at most once
- **Space**: O(H × W) - single grid + queue (worst case queue size)

## Performance

For a 50,000 × 50,000 maze:
- Memory: ~2.5 GB for the grid
- Well within your 128 GB RAM limit
- BFS will find the shortest path efficiently
