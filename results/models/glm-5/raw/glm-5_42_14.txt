```spirv
; SPIR-V
; Version: 1.0
; Generator: Khronos Glslang Reference Front End
; Bound: 100

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Imports
%1 = OpExtInstImport "GLSL.std.450"
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Constants
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_0 = OpConstant %float 0

; Input Interface
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; Output Interface
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointer types for access
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; UBO
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%uboVar = OpVariable %ptr_ubo Uniform
OpDecorate %UBO Block
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%main = OpFunction %void None %3
%5 = OpLabel

    ; Load positions for main triangle vertices (0, 2, 4)
    %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
    %p0 = OpLoad %v4float %p0_ptr
    
    %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
    %p2 = OpLoad %v4float %p2_ptr
    
    %p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
    %p4 = OpLoad %v4float %p4_ptr

    ; Compute Face Normal
    ; p2 - p0
    %vec_p20 = OpFSub %v4float %p2 %p0
    ; p4 - p0
    %vec_p40 = OpFSub %v4float %p4 %p0

    ; Extract XYZ for cross product
    %v_p20 = OpVectorShuffle %v3float %vec_p20 %vec_p20 0 1 2
    %v_p40 = OpVectorShuffle %v3float %vec_p40 %vec_p40 0 1 2

    ; Cross Product Manual: (a.y*b.z-a.z*b.y, a.z*b.x-a.x*b.z, a.x*b.y-a.y*b.x)
    ; x = a.y*b.z - a.z*b.y
    %a_y = OpCompositeExtract %float %v_p20 1
    %b_z = OpCompositeExtract %float %v_p40 2
    %a_z = OpCompositeExtract %float %v_p20 2
    %b_y = OpCompositeExtract %float %v_p40 1
    %term_x1 = OpFMul %float %a_y %b_z
    %term_x2 = OpFMul %float %a_z %b_y
    %x = OpFSub %float %term_x1 %term_x2

    ; y = a.z*b.x - a.x*b.z
    %b_x = OpCompositeExtract %float %v_p40 0
    %a_x = OpCompositeExtract %float %v_p20 0
    %term_y1 = OpFMul %float %a_z %b_x
    %term_y2 = OpFMul %float %a_x %b_z
    %y = OpFSub %float %term_y1 %term_y2

    ; z = a.x*b.y - a.y*b.x
    %term_z1 = OpFMul %float %a_x %b_y
    %term_z2 = OpFMul %float %a_y %b_x
    %z = OpFSub %float %term_z1 %term_z2

    %cross_vec = OpCompositeConstruct %v3float %x %y %z
    %faceNormal = OpExtInst %v3float %1 Normalize %cross_vec

    ; Lighting
    ; L = normalize(vec3(1, 1, 1))
    %lightVec = OpCompositeConstruct %v3float %float_1 %float_1 %float_1
    %L = OpExtInst %v3float %1 Normalize %lightVec

    ; Diffuse = max(dot(N, L), 0.0)
    %dot_NL = OpDot %float %faceNormal %L
    %diffuse = OpExtInst %float %1 FMax %dot_NL %float_0

    ; Base color = average of color[0], color[2], color[4]
    %c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
    %c0 = OpLoad %v3float %c0_ptr
    %c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
    %c2 = OpLoad %v3float %c2_ptr
    %c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
    %c4 = OpLoad %v3float %c4_ptr

    %sum1 = OpFAdd %v3float %c0 %c2
    %sum2 = OpFAdd %v3float %sum1 %c4
    ; Divide by 3 (Mul by 1/3 is faster, but constructing constant is verbose, let's assume valid constant or just use 3.0 division for assembly clarity if no 1/3 constant)
    ; Since 1/3 is not defined in constants, I'll use OpCompositeConstruct for 3.0 and FDiv
    %three = OpCompositeConstruct %v3float %float_1 %float_1 %float_1
    ; 3.0 = 1.0 + 1.0 + 1.0? No, simpler to construct 1.0,1.0,1.0 and add... 
    ; Let's just make a 3.0 constant.
    %float_3 = OpConstant %float 3
    %v3_3 = OpCompositeConstruct %v3float %float_3 %float_3 %float_3
    %baseColor = OpFDiv %v3float %sum2 %v3_3

    ; Final color = base * (0.2 + 0.8 * diffuse)
    ; term = 0.2 + 0.8 * diffuse
    %diff_scale = OpFMul %float %float_0_8 %diffuse
    %ambient_term = OpFAdd %float %float_0_2 %diff_scale
    
    %finalColor = OpVectorTimesScalar %v3float %baseColor %ambient_term

    ; Emit Vertex 0
    %outPos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
    OpStore %outPos0_ptr %p0
    OpStore %outColor %finalColor
    OpEmitVertex

    ; Emit Vertex 2
    %outPos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
    OpStore %outPos2_ptr %p2
    OpStore %outColor %finalColor
    OpEmitVertex

    ; Emit Vertex 4
    %outPos4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
    OpStore %outPos4_ptr %p4
    OpStore %outColor %finalColor
    OpEmitVertex

    OpEndPrimitive

OpReturn
OpFunctionEnd
```
