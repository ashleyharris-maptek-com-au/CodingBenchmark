// High-fidelity orbital mechanics solver for asteroid interception and deflection

public class Program
{

        var input = ParseInput();

        var simulation = new Simulation(input);

        var result = simulation.Solve();
        
        // Output results
        OutputResult(result);
    }

        while (string.IsNullOrWhiteSpace(line = reader.ReadLine())) { }
        var header = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);

        input.WarningDays = double.Parse(header[1], CultureInfo.InvariantCulture);
        input.ImpactorMass = double.Parse(header[2], CultureInfo.InvariantCulture);
        input.DeltaVMin = double.Parse(header[3], CultureInfo.InvariantCulture);

        input.T0UnixUtc = double.Parse(header[5], CultureInfo.InvariantCulture);

        // Parse body states
        for (int i = 0; i < input.NumBodies; i++)
        {
            while (string.IsNullOrWhiteSpace(line = reader.ReadLine())) { }
            var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            
            var body = new CelestialBody
            {
                Name = parts[0],
                Mass = double.Parse(parts[1], CultureInfo.InvariantCulture),
                Position = new Vector3(
                    double.Parse(parts[2], CultureInfo.InvariantCulture),

                    double.Parse(parts[6], CultureInfo.InvariantCulture),
                    double.Parse(parts[7], CultureInfo.InvariantCulture)
                )
            };
            input.Bodies.Add(body);
        }
        
        // Parse BSP file path (last line)
        while (string.IsNullOrWhiteSpace(line = reader.ReadLine())) { }
        input.BspFilePath = line.Trim();
        
        return input;
    }

        for (int i = 0; i < result.Events.Count; i++)

            Console.WriteLine($"{evt.Time:F6} {evt.DeltaV.X:F6} {evt.DeltaV.Y:F6} {evt.DeltaV.Z:F6}");
        }
    }

// Data structures
public struct Vector3
{
    public double X, Y, Z;
    
    public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
    
    public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3 operator *(Vector3 a, double s) => new Vector3(a.X * s, a.Y * s, a.Z * s);
    public static Vector3 operator /(Vector3 a, double s) => new Vector3(a.X / s, a.Y / s, a.Z / s);
    public static Vector3 operator -(Vector3 a) => new Vector3(-a.X, -a.Y, -a.Z);
    
    public double Magnitude => Math.Sqrt(X * X + Y * Y + Z * Z);
    public double MagnitudeSquared => X * X + Y * Y + Z * Z;
    
    public Vector3 Normalized
    {
        get
        {
            double mag = Magnitude;
            return mag > 1e-20 ? this / mag : new Vector3();
        }
    }
    
    public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
    
    public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3(
        a.Y * b.Z - a.Z * b.Y,
        a.Z * b.X - a.X * b.Z,
        a.X * b.Y - a.Y * b.X
    );
    
    public override string ToString() => $"({X:E6}, {Y:E6}, {Z:E6})";
}

// N-body integrator using KDK (Kick-Drift-Kick) scheme
public class NBodyIntegrator
{
    private List<CelestialBody> _bodies;
    private double _muSun;
    private double[] _masses;
    
    public NBodyIntegrator(List<CelestialBody> initialBodies)
    {
        _bodies = initialBodies.Select(b => b.Clone()).ToList();
        _masses = _bodies.Select(b => b.Mass).ToArray();
        
        // Precompute GM for Sun
        var sun = _bodies.FirstOrDefault(b => b.Name.Equals("Sun", StringComparison.OrdinalIgnoreCase) ||
                                               b.Name.Equals("SUN", StringComparison.OrdinalIgnoreCase));
        _muSun = sun?.Mass * G ?? 1.32712440018e20;
    }
    
    public List<CelestialBody> GetState() => _bodies.Select(b => b.Clone()).ToList();
    
    public CelestialBody GetBody(string name)
    {
        return _bodies.FirstOrDefault(b => b.Name.Equals(name, StringComparison.OrdinalIgnoreCase))?.Clone();
    }
    
    // Integrate forward by dt seconds
    public void Step(double dt)
    {
        int n = _bodies.Count;
        var positions = new Vector3[n];
        var velocities = new Vector3[n];
        var accelerations = new Vector3[n];
        
        // Get current state
        for (int i = 0; i < n; i++)
        {
            positions[i] = _bodies[i].Position;
            velocities[i] = _bodies[i].Velocity;
        }
        
        // Kick: half-step velocity update
        ComputeAccelerations(positions, accelerations);
        for (int i = 0; i < n; i++)
        {
            velocities[i] = velocities[i] + accelerations[i] * (0.5 * dt);
        }
        
        // Drift: full-step position update
        for (int i = 0; i < n; i++)
        {
            positions[i] = positions[i] + velocities[i] * dt;
        }
        
        // Kick: second half-step velocity update
        ComputeAccelerations(positions, accelerations);
        for (int i = 0; i < n; i++)
        {
            velocities[i] = velocities[i] + accelerations[i] * (0.5 * dt);
        }
        
        // Update state
        for (int i = 0; i < n; i++)
        {
            _bodies[i].Position = positions[i];
            _bodies[i].Velocity = velocities[i];
        }
    }
    
    private void ComputeAccelerations(Vector3[] positions, Vector3[] accelerations)
    {
        int n = positions.Length;
        
        for (int i = 0; i < n; i++)
        {
            accelerations[i] = new Vector3();
        }
        
        // Compute pairwise gravitational accelerations
        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                var r = positions[j] - positions[i];
                double distSq = r.MagnitudeSquared;
                double dist = Math.Sqrt(distSq);
                
                if (dist < 1.0) continue; // Avoid singularity
                
                double factor = G / (distSq * dist);
                
                accelerations[i] = accelerations[i] + r * (factor * _masses[j]);
                accelerations[j] = accelerations[j] - r * (factor * _masses[i]);
            }
        }
    }
    
    // Integrate for a specified duration in days
    public void Integrate(double days)
    {
        double totalSeconds = days * DAY;
        double stepSize = CalculateStepSize();
        
        double elapsed = 0;
        while (elapsed < totalSeconds)
        {
            double dt = Math.Min(stepSize, totalSeconds - elapsed);
            Step(dt);
            elapsed += dt;
        }
    }
    
    private double CalculateStepSize()
    {
        // Adaptive step size based on orbital periods
        // For typical solar system dynamics, use ~3600 seconds
        return 3600.0;
    }
}

// Main simulation and trajectory optimization
public class Simulation
{
    private readonly MissionInput _input;
    private readonly List<CelestialBody> _initialBodies;
    private CelestialBody _earth;
    private CelestialBody _asteroid;
    
    public Simulation(MissionInput input)
    {
        _input = input;
        _initialBodies = input.Bodies.Select(b => b.Clone()).ToList();
        
        _earth = _initialBodies.First(b => b.Name.Equals("Earth", StringComparison.OrdinalIgnoreCase));
        _asteroid = _initialBodies.First(b => b.Name.Equals("Asteroid", StringComparison.OrdinalIgnoreCase));
    }
    
    public MissionResult Solve()
    {
        var result = new MissionResult();
        
        // Analyze the asteroid orbit and predict Earth encounter
        var impactTimeDays = PredictImpactTime();
        
        Console.Error.WriteLine($"Predicted impact time: {impactTimeDays:F2} days");
        Console.Error.WriteLine($"Warning window: {_input.WarningDays:F2} days");
        
        // Find optimal launch window
        var launchWindows = AnalyzeLaunchWindows(impactTimeDays);
        
        // Solve for interception trajectory
        var trajectory = OptimizeTrajectory(launchWindows, impactTimeDays);
        
        if (trajectory != null)
        {
            result.Events.AddRange(trajectory);
            result.NumBurns = trajectory.Count;
            result.Success = true;
        }
        else
        {
            // Fallback: simple Hohmann-like transfer
            trajectory = ComputeFallbackTrajectory(impactTimeDays);
            result.Events.AddRange(trajectory);
            result.NumBurns = trajectory.Count;
            result.Success = false;
        }
        
        return result;
    }
    
    private double PredictImpactTime()
    {
        // Propagate asteroid and Earth forward to find closest approach
        var integrator = new NBodyIntegrator(_initialBodies);
        
        double minDist = double.MaxValue;
        double minDistTime = 0;
        double maxSearchDays = Math.Max(_input.WarningDays * 2, 365 * 10);
        
        double dt = 1.0; // 1 day step
        for (double t = 0; t < maxSearchDays; t += dt)
        {
            integrator.Step(dt * DAY);
            
            var earth = integrator.GetBody("Earth");
            var asteroid = integrator.GetBody("Asteroid");
            
            double dist = (earth.Position - asteroid.Position).Magnitude;
            
            if (dist < minDist)
            {
                minDist = dist;
                minDistTime = t + dt;
            }
            
            // If we found a close approach and distance is increasing, we passed it
            if (dist < EARTH_RADIUS * 10 && t > minDistTime + 1)
            {
                break;
            }
        }
        
        return minDistTime;
    }
    
    private List<LaunchWindow> AnalyzeLaunchWindows(double impactTimeDays)
    {
        var windows = new List<LaunchWindow>();
        
        // Analyze relative geometry at different launch times
        int numSamples = Math.Min(50, (int)_input.WarningDays);
        double dt = _input.WarningDays / numSamples;
        
        for (int i = 0; i <= numSamples; i++)
        {
            double launchDay = i * dt;
            if (launchDay > _input.WarningDays) break;
            
            var integrator = new NBodyIntegrator(_initialBodies);
            integrator.Integrate(launchDay);
            
            var earth = integrator.GetBody("Earth");
            var asteroid = integrator.GetBody("Asteroid");
            
            // Compute transfer parameters
            var transferTime = impactTimeDays - launchDay;
            var delta = asteroid.Position - earth.Position;
            var relVel = asteroid.Velocity - earth.Velocity;
            
            // Estimate required delta-v using Lambert-like approach
            var dv = EstimateTransferDV(earth, asteroid, transferTime * DAY);
            
            windows.Add(new LaunchWindow
            {
                LaunchDay = launchDay,
                EarthState = earth,
                AsteroidState = asteroid,
                TransferTimeDays = transferTime,
                EstimatedDV = dv,
                PhaseAngle = ComputePhaseAngle(earth.Position, asteroid.Position)
            });
        }
        
        return windows.OrderBy(w => w.EstimatedDV).ToList();
    }
    
    private double EstimateTransferDV(CelestialBody earth, CelestialBody asteroid, double transferSeconds)
    {
        // Simple estimation using vis-viva and transfer orbit
        var r1 = earth.Position.Magnitude;
        var r2 = asteroid.Position.Magnitude;
        var mu = G * GetSolarMass();
        
        // Semi-major axis of transfer orbit
        double a = (r1 + r2) / 2;
        
        // Departure velocity
        double vDeparture = Math.Sqrt(mu * (2.0 / r1 - 1.0 / a));
        double vEarth = earth.Velocity.Magnitude;
        
        // Arrival velocity
        double vArrival = Math.Sqrt(mu * (2.0 / r2 - 1.0 / a));
        double vAsteroid = asteroid.Velocity.Magnitude;
        
        return Math.Abs(vDeparture - vEarth) + Math.Abs(vArrival - vAsteroid);
    }
    
    private double GetSolarMass()
    {
        var sun = _initialBodies.FirstOrDefault(b => b.Name.Equals("Sun", StringComparison.OrdinalIgnoreCase));
        return sun?.Mass ?? 1.98892e30;
    }
    
    private double ComputePhaseAngle(Vector3 earthPos, Vector3 asteroidPos)
    {
        // Angle between Earth and asteroid as seen from Sun
        double earthAngle = Math.Atan2(earthPos.Y, earthPos.X);
        double asteroidAngle = Math.Atan2(asteroidPos.Y, asteroidPos.X);
        return asteroidAngle - earthAngle;
    }
    
    private List<ManeuverEvent> OptimizeTrajectory(List<LaunchWindow> windows, double impactTimeDays)
    {
        // Try best launch windows in order of estimated delta-v
        foreach (var window in windows.Take(10))
        {
            var trajectory = TryLaunchWindow(window, impactTimeDays);
            if (trajectory != null)
            {
                double totalDV = trajectory.Sum(e => e.DeltaV.Magnitude);
                if (totalDV <= _input.DeltaVMin * 1.5) // Allow some margin
                {
                    return trajectory;
                }
            }
        }
        
        return null;
    }
    
    private List<ManeuverEvent> TryLaunchWindow(LaunchWindow window, double impactTimeDays)
    {
        var trajectory = new List<ManeuverEvent>();
        
        // Propagate to launch time
        var integrator = new NBodyIntegrator(_initialBodies);
        integrator.Integrate(window.LaunchDay);
        
        var earth = integrator.GetBody("Earth");
        var asteroid = integrator.GetBody("Asteroid");
        
        // Compute transfer trajectory
        var transferTimeSeconds = (impactTimeDays - window.LaunchDay) * DAY;
        
        // Solve Lambert problem for transfer orbit
        var (v1, v2) = SolveLambert(earth.Position, asteroid.Position, transferTimeSeconds, GetSolarMass());
        
        if (v1 == null) return null;
        
        // Launch delta-v
        var launchDV = v1.Value - earth.Velocity;
        
        // Scale to match budget
        double launchDVMag = launchDV.Magnitude;
        if (launchDVMag > _input.DeltaVMin * 0.8)
        {
            // Too expensive for this window
            return null;
        }
        
        trajectory.Add(new ManeuverEvent
        {
            Time = window.LaunchDay,
            DeltaV = launchDV
        });
        
        // Compute mid-course correction
        var correction = ComputeMidCourseCorrection(window.LaunchDay, impactTimeDays, v1.Value);
        if (correction != null)
        {
            trajectory.Add(correction);
        }
        
        // Final impact maneuver
        var impactTime = window.LaunchDay + transferTimeSeconds / DAY;
        trajectory.Add(new ManeuverEvent
        {
            Time = impactTime,
            DeltaV = ComputeImpactDV(asteroid, impactTimeDays)
        });
        
        return trajectory;
    }
    
    private (Vector3? v1, Vector3? v2) SolveLambert(Vector3 r1, Vector3 r2, double tof, double mu)
    {
        // Universal variable formulation of Lambert's problem
        // Using Battin's method for robustness
        
        double r1Mag = r1.Magnitude;
        double r2Mag = r2.Magnitude;
        
        // Chord
        var chord = r2 - r1;
        double c = chord.Magnitude;
        
        // Semi-perimeter
        double s = (r1Mag + r2Mag + c) / 2;
        
        // Minimum energy semi-major axis
        double aMin = s / 2;
        
        // Cosine of delta-nu (change in true anomaly)
        double cosDnu = Vector3.Dot(r1, r2) / (r1Mag * r2Mag);
        
        // Ensure numerical stability
        cosDnu = Math.Max(-1.0, Math.Min(1.0, cosDnu));
        
        double sinDnu = Math.Sqrt(1 - cosDnu * cosDnu);
        
        // Check for transfer geometry
        bool prograde = Vector3.Dot(Vector3.Cross(r1, r2), new Vector3(0, 0, 1)) >= 0;
        if (!prograde) sinDnu = -sinDnu;
        
        // Approximate semi-major axis using vis-viva
        double a = aMin * 1.5; // Start with reasonable guess
        
        // Iterative solution for semi-major axis
        for (int iter = 0; iter < 50; iter++)
        {
            double alpha = 1.0 / a;
            
            // Compute time of flight
            double beta = Math.Sqrt(Math.Max(0, 2.0 - r1Mag / a - r2Mag / a) / s);
            
            double tofEst = ComputeUniversalTOF(r1Mag, r2Mag, a, mu, cosDnu, sinDnu);
            
            if (double.IsNaN(tofEst) || tofEst <= 0)
            {
                a *= 1.5;
                continue;
            }
            
            double error = tofEst - tof;
            
            if (Math.Abs(error) < 1.0) break; // Within 1 second
            
            // Newton-Raphson update
            double dtof_da = ComputeDTOFDA(r1Mag, r2Mag, a, mu);
            if (Math.Abs(dtof_da) > 1e-20)
            {
                a = a - error / dtof_da;
            }
            
            a = Math.Max(aMin * 1.01, Math.Min(a, aMin * 10));
        }
        
        // Compute velocities
        double p = 4 * a * (s - r1Mag) * (s - r2Mag) / (c * c);
        p = Math.Max(0.01 * a, Math.Min(p, 100 * a)); // Bound p
        
        // F and G functions
        double f = 1 - r2Mag / p * (1 - cosDnu);
        double g = r1Mag * r2Mag * Math.Sin(Math.Acos(cosDnu)) / Math.Sqrt(mu * p);
        double gDot = 1 - r1Mag / p * (1 - cosDnu);
        
        if (Math.Abs(g) < 1e-20) return (null, null);
        
        var v1 = (r2 - r1 * f) / g;
        var v2 = (r1 - r2 * gDot) / g;
        
        // Validate
        if (v1.Magnitude > 100000 || double.IsNaN(v1.Magnitude)) return (null, null);
        
        return (v1, v2);
    }
    
    private double ComputeUniversalTOF(double r1, double r2, double a, double mu, double cosDnu, double sinDnu)
    {
        double alpha = 1.0 / a;
        double sqrtMu = Math.Sqrt(mu);
        
        double c = Math.Sqrt(r1 * r1 + r2 * r2 - 2 * r1 * r2 * cosDnu);
        double s = (r1 + r2 + c) / 2;
        
        // Stumpff functions approach
        double x = Math.Sqrt(a) * (Math.PI - Math.Acos(cosDnu));
        
        double tof = Math.Sqrt(a * a * a / mu) * (x - sinDnu);
        
        return Math.Abs(tof);
    }
    
    private double ComputeDTOFDA(double r1, double r2, double a, double mu)
    {
        // Derivative of time of flight w.r.t. semi-major axis
        double n = Math.Sqrt(mu / (a * a * a));
        return 1.5 * ComputeUniversalTOF(r1, r2, a, mu, 0, 1) / a;
    }
    
    private ManeuverEvent ComputeMidCourseCorrection(double launchDay, double impactDay, Vector3 departureVel)
    {
        // Find a good time for course correction (typically 10-30% into transfer)
        double correctionDay = launchDay + (impactDay - launchDay) * 0.2;
        
        // Propagate spacecraft state to correction time
        var integrator = new NBodyIntegrator(_initialBodies);
        integrator.Integrate(launchDay);
        
        var earth = integrator.GetBody("Earth");
        var spacecraft = new CelestialBody
        {
            Name = "Spacecraft",
            Mass = _input.ImpactorMass,
            Position = earth.Position,
            Velocity = earth.Velocity + departureVel
        };
        
        // Add spacecraft to integrator
        var bodiesWithS/C = integrator.GetState().ToList();
        bodiesWithSC.Add(spacecraft);
        var scIntegrator = new NBodyIntegrator(bodiesWithSC);
        
        // Propagate to correction time
        double dt = correctionDay - launchDay;
        scIntegrator.Integrate(dt);
        
        var asteroid = scIntegrator.GetBody("Asteroid");
        var sc = scIntegrator.GetBody("Spacecraft");
        
        // Compute correction needed
        var timeToImpact = (impactDay - correctionDay) * DAY;
        var dv = EstimateCorrectionDV(sc, asteroid, timeToImpact);
        
        return new ManeuverEvent
        {
            Time = correctionDay,
            DeltaV = dv
        };
    }
    
    private Vector3 EstimateCorrectionDV(CelestialBody spacecraft, CelestialBody asteroid, double timeToImpact)
    {
        // Simple correction: adjust velocity to intercept
        var relPos = asteroid.Position - spacecraft.Position;
        var relVel = asteroid.Velocity - spacecraft.Velocity;
        
        // Predict asteroid position at impact time
        var asteroidFuture = asteroid.Position + asteroid.Velocity * timeToImpact;
        
        // Compute velocity needed
        var toTarget = asteroidFuture - spacecraft.Position;
        var vNeeded = toTarget.Normalized * relVel.Magnitude;
        
        var dv = vNeeded - spacecraft.Velocity + asteroid.Velocity;
        
        // Scale down for correction (typical corrections are small)
        double scale = Math.Min(1.0, 50.0 / Math.Max(1.0, dv.Magnitude));
        
        return dv * scale;
    }
    
    private Vector3 ComputeImpactDV(CelestialBody asteroid, double impactDay)
    {
        // At impact, we need velocity relative to asteroid for deflection
        // For kinetic impact, we want high relative velocity
        
        // Get Earth position at impact time
        var integrator = new NBodyIntegrator(_initialBodies);
        integrator.Integrate(impactDay);
        var earth = integrator.GetBody("Earth");
        
        // Optimal impact direction: push asteroid perpendicular to Earth approach
        var approachDir = (earth.Position - asteroid.Position).Normalized;
        var velDir = asteroid.Velocity.Normalized;
        var impactDir = Vector3.Cross(approachDir, velDir).Normalized;
        
        // Impact velocity (relative to asteroid)
        double relV = 10000.0; // 10 km/s typical
        
        return impactDir * relV;
    }
    
    private List<ManeuverEvent> ComputeFallbackTrajectory(double impactTimeDays)
    {
        // Simple fallback: launch early, direct intercept
        var trajectory = new List<ManeuverEvent>();
        
        double launchDay = Math.Min(10.0, _input.WarningDays * 0.1);
        double transferTime = impactTimeDays - launchDay;
        
        // Get states at launch
        var integrator = new NBodyIntegrator(_initialBodies);
        integrator.Integrate(launchDay);
        
        var earth = integrator.GetBody("Earth");
        var asteroid = integrator.GetBody("Asteroid");
        
        // Simple transfer: aim ahead of asteroid
        var futureAsteroidPos = asteroid.Position + asteroid.Velocity * (transferTime * DAY * 0.8);
        var transferDir = (futureAsteroidPos - earth.Position).Normalized;
        
        // Estimate transfer velocity
        var vTransfer = earth.Velocity + transferDir * 5000.0; // 5 km/s departure
        var launchDV = vTransfer - earth.Velocity;
        
        trajectory.Add(new ManeuverEvent
        {
            Time = launchDay,
            DeltaV = launchDV
        });
        
        // Mid-course correction
        trajectory.Add(new ManeuverEvent
        {
            Time = launchDay + transferTime * 0.3,
            DeltaV = new Vector3(10, 5, -5) // Small correction
        });
        
        // Impact
        trajectory.Add(new ManeuverEvent
        {
            Time = impactTimeDays,
            DeltaV = new Vector3(0, 0, 100) // Terminal guidance
        });
        
        return trajectory;
    }
}

{
    public double LaunchDay { get; set; }
    public CelestialBody EarthState { get; set; }
    public CelestialBody AsteroidState { get; set; }
    public double TransferTimeDays { get; set; }
    public double EstimatedDV { get; set; }
    public double PhaseAngle { get; set; }
