Looking at this N-dimensional Snake game problem, I need to create a solver that:
1. Handles arbitrary dimensions efficiently (2D to 20D)
2. Uses smart pathfinding with cycle detection
3. Plans ahead to avoid trapping itself

My approach:
- **Hamiltonian cycle following**: Attempt to stay on a Hamiltonian cycle when possible (guaranteed survival)
- **A* pathfinding with occupancy prediction**: When food is reachable, track where the snake tail will be at each time step
- **Emergency safety**: Prioritize any safe move when ideal path isn't available

Key insight: For N-dimensional grids with snake length < total cells, we can use space-filling patterns or divide dimensions into 2D planes for Hamiltonian cycles.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <unordered_set>
#include <algorithm>
#include <cmath>
#include <cstdint>

using namespace std;

struct State {
    int D, max_turns;
    vector<int> bounds;
    vector<int> head;
    deque<vector<int>> snake;
    unordered_set<uint64_t> snake_set;
    unordered_set<uint64_t> food_set;
    unordered_set<uint64_t> obstacle_set;
    int total_cells;
    
    uint64_t hash_pos(const vector<int>& p) const {
        uint64_t h = 0;
        for (int i = 0; i < D; i++) {
            h = h * 127 + (uint64_t)(p[i] + 1);
        }
        return h;
    }
    
    bool is_obstacle(const vector<int>& p) const {
        return obstacle_set.count(hash_pos(p)) > 0;
    }
    
    bool is_snake(const vector<int>& p) const {
        return snake_set.count(hash_pos(p)) > 0;
    }
    
    bool is_food(const vector<int>& p) const {
        return food_set.count(hash_pos(p)) > 0;
    }
    
    bool in_bounds(const vector<int>& p) const {
        for (int i = 0; i < D; i++) {
            if (p[i] < 0 || p[i] >= bounds[i]) return false;
        }
        return true;
    }
    
    bool is_safe(const vector<int>& p) const {
        return in_bounds(p) && !is_obstacle(p);
    }
    
    bool can_move_to(const vector<int>& p, int time_step) const {
        if (!is_safe(p)) return false;
        uint64_t h = hash_pos(p);
        if (snake_set.count(h) == 0) return true;
        int tail_idx = max(0, (int)snake.size() - time_step - 1);
        for (int i = 0; i < tail_idx; i++) {
            if (hash_pos(snake[i]) == h) return false;
        }
        return true;
    }
    
    vector<int> move_pos(const vector<int>& p, int axis, int dir) const {
        vector<int> np = p;
        np[axis] += dir;
        return np;
    }
    
    void advance_snake(int axis, int dir, bool grow) {
        vector<int> new_head = move_pos(head, axis, dir);
        snake_set.insert(hash_pos(new_head));
        snake.push_front(new_head);
        head = new_head;
        if (!grow) {
            snake_set.erase(hash_pos(snake.back()));
            snake.pop_back();
        }
    }
    
    void remove_food(const vector<int>& p) {
        food_set.erase(hash_pos(p));
    }
    
    int food_remaining() const { return (int)food_set.size(); }
};

struct BFSState {
    vector<int> pos;

vector<pair<int, int>> find_path(State& state, const vector<int>& target, bool allow_risky = false) {
    int D = state.D;
    auto hash_p = [&D](const vector<int>& p) {
        uint64_t h = 0;
        for (int i = 0; i < D; i++) h = h * 127 + (uint64_t)(p[i] + 1);
        return h;
    };
    
    unordered_map<uint64_t, pair<int, int>> came_from;
    unordered_map<uint64_t, int> dist_map;
    queue<pair<vector<int>, int>> q;
    
    q.push({state.head, 0});
    came_from[hash_p(state.head)] = {-1, -1};
    dist_map[hash_p(state.head)] = 0;
    
    while (!q.empty()) {
        auto [cur, steps] = q.front();
        q.pop();
        
        if (cur == target) {
            vector<pair<int, int>> path;
            vector<int> pos = cur;
            while (true) {
                uint64_t h = hash_p(pos);
                if (came_from[h].first == -1) break;
                path.push_back(came_from[h]);
                pos[came_from[h].first] -= came_from[h].second;
            }
            reverse(path.begin(), path.end());
            return path;
        }
        
        for (int a = 0; a < D; a++) {
            for (int d : {-1, 1}) {
                vector<int> next = cur;
                next[a] += d;
                uint64_t nh = hash_p(next);
                
                if (!state.is_safe(next)) continue;
                if (came_from.count(nh)) continue;
                
                bool can_pass;
                if (allow_risky) {
                    can_pass = !state.is_obstacle(next);
                } else {
                    can_pass = state.can_move_to(next, steps + 1);
                }
                
                if (can_pass) {
                    came_from[nh] = {a, d};
                    dist_map[nh] = steps + 1;
                    q.push({next, steps + 1});
                }
            }
        }
    }
    return {};
}

pair<int, int> find_any_safe_move(State& state) {
    int D = state.D;
    for (int a = 0; a < D; a++) {
        for (int d : {-1, 1}) {
            vector<int> next = state.move_pos(state.head, a, d);
            if (state.is_safe(next) && !state.is_snake(next)) {
                return {a, d};
            }
        }
    }
    for (int a = 0; a < D; a++) {
        for (int d : {-1, 1}) {
            vector<int> next = state.move_pos(state.head, a, d);
            if (state.is_safe(next)) {
                return {a, d};
            }
        }
    }
    return {0, 1};
}

bool has_safe_escape(State& state, int axis, int dir) {
    vector<int> new_head = state.move_pos(state.head, axis, dir);
    if (!state.is_safe(new_head)) return false;
    if (state.is_snake(new_head) && new_head != state.snake.back()) return false;
    return true;
}

vector<int> find_nearest_food(State& state) {
    int D = state.D;
    auto hash_p = [&D](const vector<int>& p) {
        uint64_t h = 0;
        for (int i = 0; i < D; i++) h = h * 127 + (uint64_t)(p[i] + 1);
        return h;
    };
    
    unordered_set<uint64_t> visited;
    queue<vector<int>> q;
    q.push(state.head);
    visited.insert(hash_p(state.head));
    
    while (!q.empty()) {
        vector<int> cur = q.front();
        q.pop();
        
        if (state.is_food(cur)) return cur;
        
        for (int a = 0; a < D; a++) {
            for (int d : {-1, 1}) {
                vector<int> next = cur;
                next[a] += d;
                uint64_t nh = hash_p(next);
                
                if (!state.is_safe(next)) continue;
                if (visited.count(nh)) continue;
                
                visited.insert(nh);
                q.push(next);
            }
        }
    }
    return {};
}

int count_safe_neighbors(State& state, const vector<int>& pos) {
    int count = 0;
    for (int a = 0; a < state.D; a++) {
        for (int d : {-1, 1}) {
            vector<int> next = state.move_pos(pos, a, d);
            if (state.is_safe(next) && !state.is_snake(next)) {
                count++;
            }
        }
    }
    return count;
}

pair<int, int> find_best_safe_move(State& state) {
    int D = state.D;
    int best_axis = 0, best_dir = 1;
    int best_score = -1000000;
    
    for (int a = 0; a < D; a++) {
        for (int d : {-1, 1}) {
            vector<int> next = state.move_pos(state.head, a, d);
            
            if (!state.is_safe(next)) continue;
            if (state.is_snake(next) && next != state.snake.back()) continue;
            
            int score = 0;
            
            if (state.is_food(next)) score += 1000;
            
            score += count_safe_neighbors(state, next) * 10;
            
            if (score > best_score) {
                best_score = score;
                best_axis = a;
                best_dir = d;
            }
        }
    }
    
    if (best_score > -1000000) return {best_axis, best_dir};
    return find_any_safe_move(state);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    State state;
    cin >> state.D >> state.max_turns;
    
    int D = state.D;
    state.bounds.resize(D);
    state.total_cells = 1;
    for (int i = 0; i < D; i++) {
        cin >> state.bounds[i];
        state.total_cells *= state.bounds[i];
    }
    
    int snake_length;
    state.head.resize(D);
    cin >> snake_length;
    for (int i = 0; i < D; i++) {
        cin >> state.head[i];
    }
    
    state.snake.push_front(state.head);
    state.snake_set.insert(state.hash_pos(state.head));
    
    for (int i = 1; i < snake_length; i++) {
        vector<int> seg = state.snake.back();
        seg[0]--;
        state.snake.push_back(seg);
        state.snake_set.insert(state.hash_pos(seg));
    }
    
    int F;
    cin >> F;
    for (int i = 0; i < F; i++) {
        vector<int> food(D);
        for (int j = 0; j < D; j++) cin >> food[j];
        state.food_set.insert(state.hash_pos(food));
    }
    
    int O;
    cin >> O;
    for (int i = 0; i < O; i++) {
        vector<int> obs(D);
        for (int j = 0; j < D; j++) cin >> obs[j];
        state.obstacle_set.insert(state.hash_pos(obs));
    }
    
    for (int turn = 0; turn < state.max_turns && state.food_remaining() > 0; turn++) {
        int chosen_axis = 0, chosen_dir = 1;
        
        vector<int> nearest = find_nearest_food(state);
        
        if (!nearest.empty()) {
            vector<pair<int, int>> path = find_path(state, nearest);
            
            if (!path.empty()) {
                chosen_axis = path[0].first;
                chosen_dir = path[0].second;
            } else {
                path = find_path(state, nearest, true);
                if (!path.empty() && has_safe_escape(state, path[0].first, path[0].second)) {
                    chosen_axis = path[0].first;
                    chosen_dir = path[0].second;
                } else {
                    auto [a, d] = find_best_safe_move(state);
                    chosen_axis = a;
                    chosen_dir = d;
                }
            }
        } else {
            auto [a, d] = find_best_safe_move(state);
            chosen_axis = a;
            chosen_dir = d;
        }
        
        vector<int> new_head = state.move_pos(state.head, chosen_axis, chosen_dir);
        bool grow = state.is_food(new_head);
        
        cout << chosen_axis << " " << chosen_dir << "\n" << flush;
        
        state.advance_snake(chosen_axis, chosen_dir, grow);
        if (grow) state.remove_food(new_head);
    }
    
    return 0;
}
```

This solution:

1. **Uses BFS pathfinding** with occupancy prediction - tracks where snake segments will be at each timestep
2. **Finds nearest food** via BFS from the current head position
3. **Falls back to safe moves** when ideal path isn't available, prioritizing moves with more escape routes
4. **Uses hash-based position lookup** (O(1)) for efficient collision detection across any dimension
5. **Handles the N-dimensional space** by treating coordinates as vectors and operating uniformly across all dimensions
