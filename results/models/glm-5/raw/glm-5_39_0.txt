use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    
    // Read n and m
    let first_line = lines.next().unwrap().unwrap();
    let mut iter = first_line.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    // Adjacency list for the original graph
    let mut adj = vec
![vec
![]; n];
    
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut iter = line.split_whitespace();
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut best_partition: Vec<bool> = vec
![false; n];
    let mut best_cut = m + 1; // Initialize with a value larger than max possible cut

    // Heuristic: Greedy Deterministic Construction
    // Start with the first vertex in A, then iteratively add the vertex
    // that minimizes the increase in cut size to A.
    {
        let mut current_partition = vec
![false; n];
        let mut current_cut = 0;
        current_partition[0] = true;

        for _ in 1..n/2 {
            let mut best_node = 0;
            let mut min_delta = m + 1;

            for u in 0..n {
                // Check if u is already in A
                if current_partition[u] { continue; }

                // Calculate delta if we move u to A
                let mut delta = 0;
                for &v in &adj[u] {
                    if current_partition[v] {
                        // Neighbor is in A, edge u-v is currently a cut edge.
                        // Moving u to A makes it internal, reducing cut.
                        delta -= 1;
                    } else {
                        // Neighbor is in B, edge u-v is currently internal.
                        // Moving u to A makes it a cut edge, increasing cut.
                        delta += 1;
                    }
                }

                if delta < min_delta {
                    min_delta = delta;
                    best_node = u;
                }
            }

            current_partition[best_node] = true;
            current_cut += min_delta;
        }

        best_cut = current_cut;
        best_partition = current_partition;
    }

    // Local Search (Fiduccia-Mattheyses / Kernighan-Lin inspired)
    // We maintain `gain[u]` for each vertex, which is the change in cut size
    // if we move `u` to the other partition.
    // We aim to find a sequence of swaps that reduces the cut size.

    // Precompute initial gains.
    // gain[u] = (edges to opposite partition) - (edges to same partition)
    // Moving u flips these contributions.
    let mut gain = vec
![0i32; n];
    for u in 0..n {
        let mut g = 0;
        for &v in &adj[u] {
            if best_partition[u] != best_partition[v] {
                g += 1;
            } else {
                g -= 1;
            }
        }
        gain[u] = g;
    }

    let mut improved = true;
    while improved {
        improved = false;
        
        // Try to find a single best swap
        // Iterate over pairs (u, v) where u in A and v in B
        // The gain of swapping u and v is gain[u] + gain[v] - 2*I(u,v)
        // where I(u,v) is 1 if they are adjacent, 0 otherwise.
        
        // Since full O(N^2) pass is expensive, we use a greedy strategy:
        // Pick the best u from A, then pick the best v from B to swap with u.
        
        // Find best u in A
        let mut best_u = None;
        let mut max_g_u = i32::MIN;
        for i in 0..n {
            if best_partition[i] {
                if gain[i] > max_g_u {
                    max_g_u = gain[i];
                    best_u = Some(i);
                }
            }
        }

        if let Some(u) = best_u {
            // Find best v in B to swap with u
            let mut best_v = None;
            let mut max_g_pair = i32::MIN;

            for j in 0..n {
                if !best_partition[j] {
                    // Calculate gain of swapping u and j
                    // gain_swap = gain[u] + gain[j] - 2 * adjacency
                    let mut g_swap = gain[u] + gain[j];
                    
                    // Check adjacency
                    // Small optimization: check if j is in adj[u]
                    let adj_u = &adj[u];
                    let is_adj = adj_u.binary_search(&j).is_ok(); // adj lists are not sorted, linear scan
                    // Given small N, linear scan is fine within outer loop structure
                    // but let's do binary search logic if we sorted adj, or just iterate.
                    // Actually, just iterate adj[u] to check.
                    let mut connected = false;
                    for &nei in &adj[u] {
                        if nei == j { connected = true; break; }
                    }
                    if connected {
                        g_swap -= 2;
                    }

                    if g_swap > max_g_pair {
                        max_g_pair = g_swap;
                        best_v = Some(j);
                    }
                }
            }

            if let Some(v) = best_v {
                // Check if this swap improves the cut
                if max_g_pair > 0 {
                    improved = true;
                    
                    // Update cut size
                    // Since we compare strictly, we only commit if strictly better.
                    // Note: best_cut is i32, m fits in i32.
                    best_cut -= max_g_pair;

                    // Perform swap: move u to B, v to A
                    best_partition[u] = false;
                    best_partition[v] = true;

                    // Update gains
                    // For u (moving A->B):
                    //   Neighbors in A were internal, become cut (-1 gain contrib? No, logic below)
                    //   Neighbors in B were cut, become internal.
                    //   gain change for neighbor x:
                    //     if x in A: edge u-x was internal, now cut. x gains edges to opposite (+1), loses same (-1). gain[x] += 2.
                    //     if x in B: edge u-x was cut, now internal. gain[x] -= 2.
                    //   gain[u] negates effectively.
                    
                    // Neighbors of u
                    for &x in &adj[u] {
                        if best_partition[x] { 
                            // x is in A (since u moved to B, and if x is in A, edge was cut? No.)
                            // Wait. u moved A -> B.
                            // Old state: u in A.
                            // Neighbor x:
                            //   If x in A: edge was internal. Now cut.
                            //     For x: edge to same -> edge to diff. gain[x] += 2.
                            //   If x in B: edge was cut. Now internal.
                            //     For x: edge to diff -> edge to same. gain[x] -= 2.
                            gain[x] += 2;
                        } else {
                            gain[x] -= 2;
                        }
                    }
                    gain[u] = -gain[u]; // gain[u] flips roughly, but with adjacency correction exact.
                    // Actually simple flip is correct only if we consider the full state change.
                    // Let's recompute or trust flip:
                    // If u moves A->B.
                    // gain_new[u] = (edges to A) - (edges to B).
                    // gain_old[u] = (edges to B) - (edges to A).
                    // Yes, gain[u] = -gain_old[u].
                    // But wait, we updated neighbors. 
                    // Neighbors update assumes u is now in B.
                    // Let's verify gain[u].
                    // gain[u] should be recalculated.
                    let mut new_g_u = 0;
                    for &x in &adj[u] {
                        if best_partition[u] != best_partition[x] { new_g_u += 1; } else { new_g_u -= 1; }
                    }
                    gain[u] = new_g_u;

                    // Neighbors of v (moving B->A)
                    // Old state: v in B.
                    // Neighbor x:
                    //   If x in A: edge was cut. Now internal.
                    //     gain[x] -= 2.
                    //   If x in B: edge was internal. Now cut.
                    //     gain[x] += 2.
                    for &x in &adj[v] {
                        if best_partition[x] {
                            gain[x] -= 2;
                        } else {
                            gain[x] += 2;
                        }
                    }
                    // Update gain[v]
                    let mut new_g_v = 0;
                    for &x in &adj[v] {
                        if best_partition[v] != best_partition[x] { new_g_v += 1; } else { new_g_v -= 1; }
                    }
                    gain[v] = new_g_v;
                }
            }
        }
    }

    // Output result
    // Collect vertices in partition A
    let mut part_a: Vec<usize> = Vec::with_capacity(n / 2);
    for i in 0..n {
        if best_partition[i] {
            part_a.push(i);
        }
    }
    
    // Adjust size if needed (theoretical guarantee of balance maintained by swaps)
    // If greedy initialization was wrong size (unlikely) or swaps broke it,
    // we truncate or extend. But swaps preserve size.
    // The greedy init is strictly size N/2.
    
    writeln!(out, "{}", best_cut).unwrap();
    let s: Vec<String> = part_a.iter().map(|x| x.to_string()).collect();
    writeln!(out, "{}", s.join(" ")).unwrap();
}
