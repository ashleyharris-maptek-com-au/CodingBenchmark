```spirv
; SPIR-V
Version: 1.0
Generator: 0
Bound: 150
Schema: 0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 204
OpMemberDecorate %UBO 5 Offset 216
OpMemberDecorate %UBO 6 Offset 228
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%uint = OpTypeInt 32 0
%uint_6 = OpConstant %uint 6

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0_008 = OpConstant %float 0.008
%float_0_5 = OpConstant %float 0.5

; Input Blocks
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Output Block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Interface Variables
%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; UBO
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform

; Pointers
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; Main Function
%main = OpFunction %void None %func
%label = OpLabel

; Indices for main triangle vertices: 0, 2, 4
; We loop manually for 3 edges: (0,2), (2,4), (4,0)

; --- Edge 0: i=0, j=2 ---
%p0_0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0_0 = OpLoad %v4float %p0_0
%p0_2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos0_2 = OpLoad %v4float %p0_2

%wp0_0 = OpAccessChain %ptr_in_v3 %inWorldPos %int_0
%wp0_2 = OpAccessChain %ptr_in_v3 %inWorldPos %int_2
%wp0_0_val = OpLoad %v3float %wp0_0
%wp0_2_val = OpLoad %v3float %wp0_2

%c0_0 = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0_2 = OpAccessChain %ptr_in_v3 %inColor %int_2
%c0_0_val = OpLoad %v3float %c0_0
%c0_2_val = OpLoad %v3float %c0_2

%edgeCol0 = OpFMul %v3float %c0_0_val %float_0_5
%edgeCol0 = OpFAdd %v3float %edgeCol0 %c0_2_val

; Geometry for edge 0
%diff0 = OpFSub %v3float %wp0_2_val %wp0_0_val
%diff0_swiz = OpVectorShuffle %v2float %diff0 %diff0 0 1
%n0 = OpCompositeExtract %float %diff0_swiz 1
%neg_n0 = OpFNegate %float %n0
%m0 = OpCompositeExtract %float %diff0_swiz 0
%perp0 = OpCompositeConstruct %v2float %neg_n0 %m0
%len0_sq = OpDot %v2float %perp0 %perp0
%len0 = OpFDiv %float %float_1_0 %len0_sq ; Assuming length is not zero
%len0_sqrt = OpExtInst %float %1 Sqrt %len0_sq
%len0_safe = OpFDiv %float %float_1_0 %len0_sqrt
%perpNorm0 = OpVectorTimesScalar %v2float %perp0 %len0_safe
%offset0 = OpVectorTimesScalar %v2float %perpNorm0 %float_0_008

%pos0_0_xy = OpVectorShuffle %v2float %pos0_0 %pos0_0 0 1
%pos0_2_xy = OpVectorShuffle %v2float %pos0_2 %pos0_2 0 1

; V0: pos0_0 + offset
%v0_xy = OpFAdd %v2float %pos0_0_xy %offset0
%v0 = OpCompositeConstruct %v4float %v0_xy %pos0_0
; V1: pos0_0 - offset
%v1_xy = OpFSub %v2float %pos0_0_xy %offset0
%v1 = OpCompositeConstruct %v4float %v1_xy %pos0_0
; V2: pos0_2 + offset
%v2_xy = OpFAdd %v2float %pos0_2_xy %offset0
%v2 = OpCompositeConstruct %v4float %v2_xy %pos0_2
; V3: pos0_2 - offset
%v3_xy = OpFSub %v2float %pos0_2_xy %offset0
%v3 = OpCompositeConstruct %v4float %v3_xy %pos0_2

; Emit Edge 0
%outP = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outP %v0
OpStore %outColor %edgeCol0
OpEmitVertex

OpStore %outP %v1
OpStore %outColor %edgeCol0
OpEmitVertex

OpStore %outP %v2
OpStore %outColor %edgeCol0
OpEmitVertex

OpStore %outP %v3
OpStore %outColor %edgeCol0
OpEmitVertex

OpEndPrimitive

; --- Edge 1: i=2, j=4 ---
%p1_2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos1_2 = OpLoad %v4float %p1_2
%p1_4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos1_4 = OpLoad %v4float %p1_4

%wp1_2 = OpAccessChain %ptr_in_v3 %inWorldPos %int_2
%wp1_4 = OpAccessChain %ptr_in_v3 %inWorldPos %int_4
%wp1_2_val = OpLoad %v3float %wp1_2
%wp1_4_val = OpLoad %v3float %wp1_4

%c1_2 = OpAccessChain %ptr_in_v3 %inColor %int_2
%c1_4 = OpAccessChain %ptr_in_v3 %inColor %int_4
%c1_2_val = OpLoad %v3float %c1_2
%c1_4_val = OpLoad %v3float %c1_4

%edgeCol1 = OpFMul %v3float %c1_2_val %float_0_5
%edgeCol1 = OpFAdd %v3float %edgeCol1 %c1_4_val

%diff1 = OpFSub %v3float %wp1_4_val %wp1_2_val
%diff1_swiz = OpVectorShuffle %v2float %diff1 %diff1 0 1
%n1 = OpCompositeExtract %float %diff1_swiz 1
%neg_n1 = OpFNegate %float %n1
%m1 = OpCompositeExtract %float %diff1_swiz 0
%perp1 = OpCompositeConstruct %v2float %neg_n1 %m1
%len1_sq = OpDot %v2float %perp1 %perp1
%len1_sqrt = OpExtInst %float %1 Sqrt %len1_sq
%len1_safe = OpFDiv %float %float_1_0 %len1_sqrt
%perpNorm1 = OpVectorTimesScalar %v2float %perp1 %len1_safe
%offset1 = OpVectorTimesScalar %v2float %perpNorm1 %float_0_008

%pos1_2_xy = OpVectorShuffle %v2float %pos1_2 %pos1_2 0 1
%pos1_4_xy = OpVectorShuffle %v2float %pos1_4 %pos1_4 0 1

%v4_xy = OpFAdd %v2float %pos1_2_xy %offset1
%v4 = OpCompositeConstruct %v4float %v4_xy %pos1_2
%v5_xy = OpFSub %v2float %pos1_2_xy %offset1
%v5 = OpCompositeConstruct %v4float %v5_xy %pos1_2
%v6_xy = OpFAdd %v2float %pos1_4_xy %offset1
%v6 = OpCompositeConstruct %v4float %v6_xy %pos1_4
%v7_xy = OpFSub %v2float %pos1_4_xy %offset1
%v7 = OpCompositeConstruct %v4float %v7_xy %pos1_4

OpStore %outP %v4
OpStore %outColor %edgeCol1
OpEmitVertex

OpStore %outP %v5
OpStore %outColor %edgeCol1
OpEmitVertex

OpStore %outP %v6
OpStore %outColor %edgeCol1
OpEmitVertex

OpStore %outP %v7
OpStore %outColor %edgeCol1
OpEmitVertex

OpEndPrimitive

; --- Edge 2: i=4, j=0 ---
%p2_4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos2_4 = OpLoad %v4float %p2_4
%p2_0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos2_0 = OpLoad %v4float %p2_0

%wp2_4 = OpAccessChain %ptr_in_v3 %inWorldPos %int_4
%wp2_0 = OpAccessChain %ptr_in_v3 %inWorldPos %int_0
%wp2_4_val = OpLoad %v3float %wp2_4
%wp2_0_val = OpLoad %v3float %wp2_0

%c2_4 = OpAccessChain %ptr_in_v3 %inColor %int_4
%c2_0 = OpAccessChain %ptr_in_v3 %inColor %int_0
%c2_4_val = OpLoad %v3float %c2_4
%c2_0_val = OpLoad %v3float %c2_0

%edgeCol2 = OpFMul %v3float %c2_4_val %float_0_5
%edgeCol2 = OpFAdd %v3float %edgeCol2 %c2_0_val

%diff2 = OpFSub %v3float %wp2_0_val %wp2_4_val
%diff2_swiz = OpVectorShuffle %v2float %diff2 %diff2 0 1
%n2 = OpCompositeExtract %float %diff2_swiz 1
%neg_n2 = OpFNegate %float %n2
%m2 = OpCompositeExtract %float %diff2_swiz 0
%perp2 = OpCompositeConstruct %v2float %neg_n2 %m2
%len2_sq = OpDot %v2float %perp2 %perp2
%len2_sqrt = OpExtInst %float %1 Sqrt %len2_sq
%len2_safe = OpFDiv %float %float_1_0 %len2_sqrt
%perpNorm2 = OpVectorTimesScalar %v2float %perp2 %len2_safe
%offset2 = OpVectorTimesScalar %v2float %perpNorm2 %float_0_008

%pos2_4_xy = OpVectorShuffle %v2float %pos2_4 %pos2_4 0 1
%pos2_0_xy = OpVectorShuffle %v2float %pos2_0 %pos2_0 0 1

%v8_xy = OpFAdd %v2float %pos2_4_xy %offset2
%v8 = OpCompositeConstruct %v4float %v8_xy %pos2_4
%v9_xy = OpFSub %v2float %pos2_4_xy %offset2
%v9 = OpCompositeConstruct %v4float %v9_xy %pos2_4
%v10_xy = OpFAdd %v2float %pos2_0_xy %offset2
%v10 = OpCompositeConstruct %v4float %v10_xy %pos2_0
%v11_xy = OpFSub %v2float %pos2_0_xy %offset2
%v11 = OpCompositeConstruct %v4float %v11_xy %pos2_0

OpStore %outP %v8
OpStore %outColor %edgeCol2
OpEmitVertex

OpStore %outP %v9
OpStore %outColor %edgeCol2
OpEmitVertex

OpStore %outP %v10
OpStore %outColor %edgeCol2
OpEmitVertex

OpStore %outP %v11
OpStore %outColor %edgeCol2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
