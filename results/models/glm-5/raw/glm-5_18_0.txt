const int BUFSIZE = 1 << 20; // 1MB buffer

struct FastReader {
    FILE* fp;
    char* buf;
    int len, pos;
    bool eof;

    FastReader(FILE* f) : fp(f), len(0), pos(0), eof(false) {
        buf = (char*)malloc(BUFSIZE);
        refill();
    }
    ~FastReader() { free(buf); }

    void refill() {
        len = (int)fread(buf, 1, BUFSIZE, fp);
        pos = 0;
        if (len == 0) eof = true;
    }

    int readInt() {
        // Skip whitespace
        while (true) {
            if (pos == len) {
                if (eof) return -1; // End of file or error
                refill();
                if (eof) return -1;
            }
            if (buf[pos] > ' ') break;
            pos++;
        }

        // Read number
        int val = 0;
        bool neg = false;
        while (!eof && pos < len && buf[pos] > ' ') {
            if (buf[pos] == '-') {
                neg = true;
                pos++;
            } else {
                val = val * 10 + (buf[pos] - '0');
                pos++;
            }
            if (pos == len && !eof) refill();
        }
        return neg ? -val : val;
    }
};

using IntMatrix = std::vector<std::vector<int>>;

// ============================================================================
// Stoer-Wagner Minimum Cut Algorithm
// Best for small dense graphs
// O(N^3) time, O(N^2) memory
// ============================================================================

int stoer_wagner(const IntMatrix& graph, int n) {
    if (n <= 1) return 0;
    
    std::vector<int> vertices(n);
    std::iota(vertices.begin(), vertices.end(), 0);
    
    IntMatrix weights = graph; // Copy to modify
    int best_cut = std::numeric_limits<int>::max();
    
    std::vector<int> added(n);
    std::vector<int> merged_weights(n);
    
    int remaining = n;
    
    while (remaining > 1) {
        std::fill(added.begin(), added.end(), 0);
        std::fill(merged_weights.begin(), merged.end(), 0);
        
        int prev = -1;
        for (int i = 0; i < remaining; ++i) {
            int sel = -1;
            // Select vertex with maximum merged weight
            for (int j = 0; j < remaining; ++j) {
                if (!added[vertices[j]]) {
                    if (sel == -1 || merged_weights[vertices[j]] > merged_weights[vertices[sel]]) {
                        sel = vertices[j];
                    }
                }
            }
            
            if (i == remaining - 1) {
                // Last vertex added - this is the cut
                best_cut = std::min(best_cut, merged_weights[vertices[sel]]);
                
                // Merge prev and sel
                for (int j = 0; j < remaining; ++j) {
                    weights[vertices[j]][prev] += weights[vertices[j]][vertices[sel]];
                    weights[prev][vertices[j]] += weights[vertices[vertices[sel]]][vertices[j]];
                }
                weights[prev][prev] = 0;
                
                // Remove sel from vertices list
                for (int j = 0; j < remaining; ++j) {
                    if (vertices[j] == vertices[sel]) {
                        std::swap(vertices[j], vertices[remaining - 1]);
                        break;
                    }
                }
            } else {
                added[vertices[sel]] = 1;
                prev = vertices[sel];
                
                // Update merged weights
                for (int j = 0; j < remaining; ++j) {
                    if (!added[vertices[j]]) {
                        merged_weights[vertices[j]] += weights[vertices[vertices[j]]][vertices[sel]];
                    }
                }
            }
        }
        remaining--;
    }
    
    return best_cut;
}

// ============================================================================
// Disjoint Set Union (DSU) for Karger's Algorithm
// ============================================================================

class DSU {
    std::vector<int> parent, rank;
public:
    DSU(int n) : parent(n), rank(n, 0) {
        std::iota(parent.begin(), parent.end(), 0);
    }
    
    int find(int x) {
        while (parent[x] != x) {
            parent[x] = parent[parent[x]];
            x = parent[x];
        }
        return x;
    }
    
    bool unite(int x, int y) {
        x = find(x);
        y = find(y);
        if (x == y) return false;
        if (rank[x] < rank[y]) std::swap(x, y);
        parent[y] = x;
        if (rank[x] == rank[y]) rank[x]++;
        return true;
    }
    
    void reset(int n) {
        std::iota(parent.begin(), parent.begin() + n, 0);
        std::fill(rank.begin(), rank.begin() + n, 0);
    }
};

// ============================================================================
// Karger-Stein Algorithm
// Best for large graphs
// O(N^2 log N) time, O(N) memory per iteration
// ============================================================================

struct KargerResult {
    int cut_size;
    std::vector<std::pair<int, int>> cut_edges;
};

// Single run of Karger's contraction algorithm
int karger_run(const std::vector<int>& u_edges, const std::vector<int>& v_edges, 
               int n, size_t m, std::mt19937& rng) {
    DSU dsu(n);
    int components = n;
    
    // Create edge index permutation
    std::vector<size_t> perm(m);
    std::iota(perm.begin(), perm.end(), 0);
    
    // Fisher-Yates shuffle - only shuffle as needed
    for (size_t i = 0; i < m && components > 2; ++i) {
        size_t j = i + rng() % (m - i);
        std::swap(perm[i], perm[j]);
        
        if (dsu.unite(u_edges[perm[i]], v_edges[perm[i]])) {
            components--;
        }
    }
    
    // Count cut edges
    int cut = 0;
    for (size_t i = 0; i < m; ++i) {
        if (dsu.find(u_edges[i]) != dsu.find(v_edges[i])) {
            cut++;
        }
    }
    
    return cut;
}

// Karger with cut edge tracking
KargerResult karger_run_with_edges(const std::vector<int>& u_edges, const std::vector<int>& v_edges,
                                   int n, size_t m, std::mt19937& rng) {
    DSU dsu(n);
    int components = n;
    
    std::vector<size_t> perm(m);
    std::iota(perm.begin(), perm.end(), 0);
    
    for (size_t i = 0; i < m && components > 2; ++i) {
        size_t j = i + rng() % (m - i);
        std::swap(perm[i], perm[j]);
        
        if (dsu.unite(u_edges[perm[i]], v_edges[perm[i]])) {
            components--;
        }
    }
    
    KargerResult result;
    for (size_t i = 0; i < m; ++i) {
        if (dsu.find(u_edges[i]) != dsu.find(v_edges[i])) {
            result.cut_edges.emplace_back(u_edges[i], v_edges[i]);
        }
    }
    result.cut_size = (int)result.cut_edges.size();
    return result;
}

// ============================================================================
// Parallel Karger Runner
// ============================================================================

int run_parallel_karger(const std::vector<int>& u_edges, const std::vector<int>& v_edges,
                        int n, size_t m, int num_threads) {
    
    // Calculate number of iterations needed
    // Probability of finding min cut: ~1/n for each run
    // Need ~n log n runs for high probability
    int log_n = 1;
    for (int x = n; x > 1; x >>= 1) log_n++;
    
    int iterations = std::min(n * KARGER_ITERATIONS_SCALE * log_n, 1000000);
    iterations = std::max(iterations, 100); // At least 100 iterations
    
    if (m == 0) return 0;
    
    std::atomic<int> best_cut{std::numeric_limits<int>::max()};
    std::atomic<int> completed{0};
    
    // Thread-safe minimum tracking
    
    auto worker = [&](int thread_id, int start_iter, int end_iter) {
        std::random_device rd;
        std::mt19937 rng(rd() + thread_id * 10007);
        
        int local_best = std::numeric_limits<int>::max();
        
        for (int i = start_iter; i < end_iter; ++i) {
            int cut = karger_run(u_edges, v_edges, n, m, rng);
            
            // Early termination if we found cut of size 1 (minimum possible for unweighted)
            if (cut == 1) {
                int expected = best_cut.load();
                while (expected > 1) {
                    if (best_cut.compare_exchange_weak(expected, 1)) break;
                }
                return;
            }
            
            if (cut < local_best) {
                local_best = cut;
            }
        }
        
        // Update global best
        int expected = best_cut.load();
        while (local_best < expected) {
            if (best_cut.compare_exchange_weak(expected, local_best)) break;
        }
    };
    
    // Launch threads
    std::vector<std::thread> threads;
    int iter_per_thread = (iterations + num_threads - 1) / num_threads;
    
    for (int t = 0; t < num_threads; ++t) {
        int start = t * iter_per_thread;
        int end = std::min(start + iter_per_thread, iterations);
        if (start < end) {
            threads.emplace_back(worker, t, start, end);
        }
    }
    
    for (auto& t : threads) {
        t.join();
    }
    
    return best_cut.load();
}

// Find cut edges after determining minimum cut size
KargerResult find_cut_edges(const std::vector<int>& u_edges, const std::vector<int>& v_edges,
                           int n, size_t m, int target_cut, int num_threads) {
    
    std::atomic<bool> found{false};
    KargerResult result;
    std::mutex result_mutex;
    
    auto worker = [&](int thread_id) {
        std::random_device rd;
        std::mt19937 rng(rd() + thread_id * 10007 + 999);
        
        while (!found.load()) {
            auto res = karger_run_with_edges(u_edges, v_edges, n, m, rng);
            if (res.cut_size == target_cut) {
                std::lock_guard<std::mutex> lock(result_mutex);
                if (!found.load()) {
                    result = std::move(res);
                    found.store(true);
                }
                return;
            }
        }
    };
    
    std::vector<std::thread> threads;
    for (int t = 0; t < num_threads; ++t) {
        threads.emplace_back(worker, t);
    }
    
    for (auto& t : threads) {
        t.join();
    }
    
    return result;
}

class MinCutSolver {
public:
    void solve() {
        read_input();
        
        if (g_N <= 1 || g_M == 0) {
            std::cout << "0\n";
            return;
        }
        
        // Check if graph is already disconnected
        if (!is_connected()) {
            std::cout << "0\n";
            return;
        }
        
        int min_cut = compute_min_cut();
        auto cut_edges = extract_cut_edges(min_cut);
        
        // Output
        std::cout << min_cut << "\n";
        for (const auto& [u, v] : cut_edges) {
            std::cout << u << " " << v << "\n";
        }
    }

private:
    void read_input() {
        // Use fast I/O
        FastReader reader(stdin);
        
        g_N = reader.readInt();
        int m_int = reader.readInt();
        g_M = (size_t)m_int;
        
        g_u_edges.reserve(g_M);
        g_v_edges.reserve(g_M);
        
        for (size_t i = 0; i < g_M; ++i) {
            int u = reader.readInt();
            int v = reader.readInt();
            g_u_edges.push_back(u);
            g_v_edges.push_back(v);
        }
    }
    
    bool is_connected() {
        // BFS/DFS to check connectivity
        std::vector<std::vector<int>> adj(g_N);
        for (size_t i = 0; i < g_M; ++i) {
            adj[g_u_edges[i]].push_back(g_v_edges[i]);
            adj[g_v_edges[i]].push_back(g_u_edges[i]);
        }
        
        std::vector<int> visited(g_N, 0);
        std::vector<int> stack;
        stack.push_back(0);
        visited[0] = 1;
        int count = 1;
        
        while (!stack.empty()) {
            int u = stack.back();
            stack.pop_back();
            
            for (int v : adj[u]) {
                if (!visited[v]) {
                    visited[v] = 1;
                    count++;
                    stack.push_back(v);
                }
            }
        }
        
        return count == g_N;
    }
    
    int compute_min_cut() {
        int num_threads = std::thread::hardware_concurrency();
        if (num_threads == 0) num_threads = 8;
        
        // For small graphs, use Stoer-Wagner
        size_t stoer_wagner_memory = (size_t)g_N * g_N * sizeof(int);
        
        if (g_N <= 500 && stoer_wagner_memory < MAX_MEMORY_USAGE / 2) {
            return stoer_wagner_solve();
        }
        
        // For larger graphs, use parallel Karger
        return run_parallel_karger(g_u_edges, g_v_edges, g_N, g_M, num_threads);
    }
    
    int stoer_wagner_solve() {
        // Build adjacency matrix
        IntMatrix graph(g_N, std::vector<int>(g_N, 0));
        
        for (size_t i = 0; i < g_M; ++i) {
            int u = g_u_edges[i];
            int v = g_v_edges[i];
            if (u != v) {
                graph[u][v]++;
                graph[v][u]++;
            }
        }
        
        return stoer_wagner(graph, g_N);
    }
    
    std::vector<std::pair<int, int>> extract_cut_edges(int min_cut) {
        if (min_cut == 0) return {};
        
        int num_threads = std::thread::hardware_concurrency();
        if (num_threads == 0) num_threads = 8;
        
        auto result = find_cut_edges(g_u_edges, g_v_edges, g_N, g_M, min_cut, num_threads);
        return result.cut_edges;
    }
};

    std::ios::sync_with_stdio(false);

        MinCutSolver solver;
        solver.solve();

        return 1;
    }
    
    return 0;
}
