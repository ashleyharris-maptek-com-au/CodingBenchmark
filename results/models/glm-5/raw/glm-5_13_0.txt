// Robust string reading to handle large inputs and platform differences
std::string read_string(std::istream& is) {
    std::string s;
    // Reserve space to avoid reallocations
    s.reserve(4096); 
    
    // Read character by character to handle all whitespace scenarios correctly
    // and to avoid issues with '\r' on Windows
    int c;
    while ((c = is.get()) != EOF) {
        if (c == '\n') break;
        if (c == '\r') {
            // Peek next char to see if it's \n (Windows line ending)
            if (is.peek() == '\n') is.get();
            break;
        }
        s += static_cast<char>(c);
    }
    return s;
}

// ==========================================
// Strategy 1: Binary Search + Rolling Hash
// Time: O(N * L * log(MinLen))
// Space: O(N * L)
// ==========================================

class RollingHashSolver {
    static constexpr uint64_t BASE = 131;
    static constexpr uint64_t MOD = (1ULL << 61) - 1;

    const std::vector<std::string>& strings;
    size_t min_len;
    int n_strings;

    // Modular arithmetic helpers for uint128 simulation or logic
    // Using __int128 if available, otherwise standard overflow arithmetic
    // MSVC supports __int128 in 64-bit mode
    static uint64_t mod(uint64_t x) {
        x = (x >> 61) + (x & MOD);
        if (x >= MOD) x -= MOD;
        return x;
    }

    static uint64_t mul(uint64_t a, uint64_t b) {
#if defined(_M_X64) || defined(__SIZEOF_INT128__)
        unsigned __int128 res = (unsigned __int128)a * b;
        return mod(res);
#else
        // Fallback for 32-bit or other architectures
        uint64_t a_hi = a >> 31, a_lo = a & ((1ULL << 31) - 1);
        uint64_t b_hi = b >> 31, b_lo = b & ((1ULL << 31) - 1);
        uint64_t m1 = a_hi * b_hi, m2 = a_lo * b_hi, m3 = a_hi * b_lo, m4 = a_lo * b_lo;
        uint64_t res = m1 + (m2 >> 30) + (m3 >> 30) + (m4 >> 31) + ((m4 << 33) >> 33);
        return mod(res);
#endif
    }

public:
    RollingHashSolver(const std::vector<std::string>& s) : strings(s), n_strings((int)s.size()) {
        min_len = s[0].size();
        for (const auto& str : s) min_len = std::min(min_len, str.size());
    }

    std::string solve() {
        if (min_len == 0) return "";

        size_t low = 1, high = min_len;
        std::string result;

        std::vector<uint64_t> powers(min_len + 1);
        powers[0] = 1;
        for (size_t i = 1; i <= min_len; ++i) powers[i] = mul(powers[i - 1], BASE);

        // Binary search on length
        while (low <= high) {
            size_t mid = low + (high - low) / 2;
            std::string temp_res = check(mid, powers);
            if (!temp_res.empty()) {
                result = temp_res;
                low = mid + 1;
            } else {
                if (mid == 0) break;
                high = mid - 1;
            }
        }
        return result;
    }

private:
    std::string check(size_t len, const std::vector<uint64_t>& powers) {
        if (len == 0) return "";
        
        // Find the reference string (smallest one) to enumerate substrings from
        int ref_idx = 0;
        size_t smallest_len = strings[0].size();
        for(int i=1; i<n_strings; ++i) {
            if(strings[i].size() < smallest_len) {
                smallest_len = strings[i].size();
                ref_idx = i;
            }
        }
        
        const std::string& ref_str = strings[ref_idx];
        std::vector<uint64_t> ref_hashes;
        ref_hashes.reserve(ref_str.size() - len + 1);

        // Precompute hashes for the reference string
        uint64_t h = 0;
        for (size_t i = 0; i < len; ++i) h = mod(mul(h, BASE) + ref_str[i]);
        ref_hashes.push_back(h);

        uint64_t power_len = powers[len];
        for (size_t i = len; i < ref_str.size(); ++i) {
            h = mod(mul(h, BASE) + ref_str[i] + MOD - mul(ref_str[i - len], power_len));
            ref_hashes.push_back(h);
        }

        // For each hash in reference, check if it exists in all other strings
        for (size_t i = 0; i < ref_hashes.size(); ++i) {
            uint64_t target_hash = ref_hashes[i];
            bool found_in_all = true;

            for (int k = 0; k < n_strings; ++k) {
                if (k == ref_idx) continue;

                const std::string& curr = strings[k];
                if (curr.size() < len) { found_in_all = false; break; }

                bool found_in_curr = false;
                uint64_t curr_h = 0;
                
                // Compute first window
                for (size_t j = 0; j < len; ++j) curr_h = mod(mul(curr_h, BASE) + curr[j]);
                if (curr_h == target_hash) found_in_curr = true;
                else {
                    // Roll
                    for (size_t j = len; j < curr.size(); ++j) {
                        curr_h = mod(mul(curr_h, BASE) + curr[j] + MOD - mul(curr[j - len], power_len));
                        if (curr_h == target_hash) {
                            found_in_curr = true;
                            break;
                        }
                    }
                }
                
                if (!found_in_curr) {
                    found_in_all = false;
                    break;
                }
            }

            if (found_in_all) {
                return ref_str.substr(i, len);
            }
        }
        return "";
    }
};

// ==========================================
// Strategy 2: Suffix Automaton (SAM)
// Time: O(L_total) construction, O(N * L_state) query
// Space: O(L_total)
// ==========================================

class SAMSolver {
    struct State {
        int len = 0;
        int link = -1;
        std::array<int, 26> next{};
        State() { next.fill(-1); }
    };

    std::vector<State> st;
    int last = 0;
    int sz = 1;
    std::string source_str;

public:
    SAMSolver(const std::string& s) : source_str(s) {
        size_t n = s.size();
        st.resize(2 * n);
        st[0].len = 0;
        st[0].link = -1;

        for (char c : s) extend(c - 'a');
    }

    std::string solve(const std::vector<std::string>& strings) {
        std::vector<int> lcs(st.size(), 0);
        std::vector<int> order(st.size());
        
        // Topological sort by length
        int max_len = 0;
        for(int i=0; i<sz; ++i) max_len = std::max(max_len, st[i].len);
        std::vector<int> cnt(max_len + 1, 0);
        for(int i=0; i<sz; ++i) cnt[st[i].len]++;
        for(int i=1; i<=max_len; ++i) cnt[i] += cnt[i-1];
        for(int i=0; i<sz; ++i) order[--cnt[st[i].len]] = i;

        // Initialize with lengths of SAM states (representing occurrences in source)
        for(int i=0; i<sz; ++i) lcs[i] = st[i].len;

        // Process other strings
        for (size_t i = 0; i < strings.size(); ++i) {
            // Skip the string used to build SAM
            if (strings[i] == source_str) continue; 
            process_string(strings[i], lcs, order);
        }

        int max_len_res = 0;
        for (int i = 0; i < sz; ++i) {
            if (lcs[i] < st[i].len) {
                // The common substring ends within this state
                max_len_res = std::max(max_len_res, lcs[i]);
            } else {
                // The common substring spans the full length of this state
                max_len_res = std::max(max_len_res, st[i].len);
            }
        }

        if (max_len_res == 0) return "";

        // Retrieve the substring
        // Find a state that can represent a substring of length max_len_res
        // We can just walk the SAM again or re-track
        // Simpler: rerun the matching logic to recover the string
        return recover_substring(strings, max_len_res);
    }

private:
    void extend(int c) {
        int cur = sz++;
        st[cur].len = st[last].len + 1;
        int p = last;
        while (p != -1 && st[p].next[c] == -1) {
            st[p].next[c] = cur;
            p = st[p].link;
        }
        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = sz++;
                st[clone].len = st[p].len + 1;
                st[clone].next = st[q].next;
                st[clone].link = st[q].link;
                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = clone;
                st[cur].link = clone;
            }
        }
        last = cur;
    }

    void process_string(const std::string& t, std::vector<int>& lcs, const std::vector<int>& order) {
        int v = 0;
        int l = 0;
        std::vector<int> curr_match(st.size(), 0);

        for (char ch : t) {
            int c = ch - 'a';
            while (v != 0 && st[v].next[c] == -1) {
                v = st[v].link;
                l = st[v].len;
            }
            if (st[v].next[c] != -1) {
                v = st[v].next[c];
                l++;
            }
            curr_match[v] = std::max(curr_match[v], l);
        }

        // Propagate values up the suffix links (in reverse topological order)
        for (int i = order.size() - 1; i >= 0; --i) {
            int u = order[i];
            if (curr_match[u] > 0 && st[u].link != -1) {
                int p = st[u].link;
                curr_match[p] = std::max(curr_match[p], std::min(curr_match[u], st[p].len));
            }
        }

        for (int i = 0; i < sz; ++i) {
            lcs[i] = std::min(lcs[i], curr_match[i]);
        }
    }

    std::string recover_substring(const std::vector<std::string>& strings, int target_len) {
        // We find any substring of target_len that is common to all
        // We can just use the shortest string to generate candidates and check
        // This is cheap enough since we know it exists
        // Or use the SAM itself:
        // We need to find a state v such that lcs[v] >= target_len (effectively)
        // and then trace back.
        // Alternatively, iterate through the shortest string.
        
        int shortest_idx = 0;
        size_t min_s = strings[0].size();
        for(int i=1; i<strings.size(); ++i) {
            if(strings[i].size() < min_s) {
                min_s = strings[i].size();
                shortest_idx = i;
            }
        }
        
        const std::string& ref = strings[shortest_idx];
        std::string temp = ref + "#" + source_str; // Concatenate to use SAM for checking?
        // No, simpler: just take any substring of len target_len from ref and check against SAM.
        // Since SAM contains source_str, if source_str is not ref, we are checking if ref's substring is in source.
        // Wait, the check logic needs to be "common to ALL".
        
        // Let's just iterate substrings of the reference string.
        // Since N is small, and we verified length exists, this is fast.
        
        for (size_t i = 0; i + target_len <= ref.size(); ++i) {
            std::string cand = ref.substr(i, target_len);
            bool ok = true;
            for (int j = 0; j < strings.size(); ++j) {
                if (strings[j].find(cand) == std::string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) return cand;
        }
        return ""; // Should not happen
    }
};

// ==========================================
// Strategy 3: Brute Force (Small Input)
// ==========================================

class BruteForceSolver {
    const std::vector<std::string>& strings;
public:
    BruteForceSolver(const std::vector<std::string>& s) : strings(s) {}

    std::string solve() {
        int n = strings.size();
        int shortest_idx = 0;
        size_t min_len = strings[0].size();
        for(int i=1; i<n; ++i) {
            if(strings[i].size() < min_len) {
                min_len = strings[i].size();
                shortest_idx = i;
            }
        }

        const std::string& ref = strings[shortest_idx];
        
        // From longest to shortest length
        for (size_t len = min_len; len >= 1; --len) {
            // Iterate all substrings of this length
            // Early exit on first find
            for (size_t i = 0; i + len <= ref.size(); ++i) {
                std::string sub = ref.substr(i, len);
                bool found = true;
                for (int j = 0; j < n; ++j) {
                    if (j == shortest_idx) continue;
                    if (strings[j].find(sub) == std::string::npos) {
                        found = false;
                        break;
                    }
                }
                if (found) return sub;
            }
        }
        return "";
    }
};

int main() {
    fast_io();

    int n;
    if (!(std::cin >> n)) return 0;
    std::cin.ignore(); // Consume newline after N

    std::vector<std::string> strings(n);
    size_t total_len = 0;
    size_t max_len = 0;
    size_t min_len = SIZE_MAX;

    for (int i = 0; i < n; ++i) {
        strings[i] = read_string(std::cin);
        total_len += strings[i].size();
        max_len = std::max(max_len, strings[i].size());
        min_len = std::min(min_len, strings[i].size());
    }

    if (n == 0 || min_len == 0) {
        std::cout << "" << std::endl;
        return 0;
    }

    std::string result;

    // Heuristic for strategy selection
    // Limit for Brute Force: Very small strings
    // Limit for SAM: Moderate N, moderate total length. SAM is O(L) construction but O(N * L) query.
    // If N is large, the query part becomes heavy.
    // Limit for Hash: Works well for large L, but O(N * L) per binary search step.
    
    // Thresholds tuned for 300s time limit and 128GB RAM
    // 1. Trivial: max_len < 50 -> Brute Force
    // 2. Small N (<= 20) & Total Len < 2,000,000 -> SAM (Linear time complexity)
    // 3. Else -> Rolling Hash (Logarithmic factor helps on huge strings, memory efficient)

    bool use_brute = (max_len < 50);
    bool use_sam = (n <= 20 && total_len < 2000000);

    if (use_brute) {
        BruteForceSolver solver(strings);
        result = solver.solve();
    } else if (use_sam) {
        // Build SAM on the shortest string to minimize state count
        int shortest_idx = 0;
        for(int i=1; i<n; ++i) if(strings[i].size() < strings[shortest_idx].size()) shortest_idx = i;
        
        SAMSolver solver(strings[shortest_idx]);
        result = solver.solve(strings);
    } else {
        RollingHashSolver solver(strings);
        result = solver.solve();
    }

    std::cout << result << "\n";

    return 0;
}
