using System;
using System.Collections.Generic;
using System.IO;
using System.Threading.Tasks;

public class Program
{
    // Buffer size for optimized I/O
    private const int BufferSize = 65536;

    public static void Main()
    {
        // Use optimized stream reader
        using var stream = new BufferedStream(Console.OpenStandardInput(), BufferSize);
        using var reader = new StreamReader(stream);

        string line1 = reader.ReadLine();
        if (string.IsNullOrEmpty(line1)) return;

        // Parse N and stock length from first line
        int spaceIdx = line1.IndexOf(' ');
        int n = int.Parse(line1.AsSpan(0, spaceIdx));
        int stockLength = int.Parse(line1.AsSpan(spaceIdx + 1));

        // Pre-allocate cuts array
        int[] cuts = new int[n];
        
        // Parse all cuts from the input
        int cutsRead = 0;
        
        // Continue reading until we have all cuts (handles line wrapping)
        while (cutsRead < n)
        {
            string line = reader.ReadLine();
            if (string.IsNullOrEmpty(line)) break;
            
            int start = 0;
            int end = 0;
            
            while (start < line.Length && cutsRead < n)
            {
                // Skip leading whitespace
                while (start < line.Length && line[start] == ' ') start++;
                if (start >= line.Length) break;
                
                // Find end of number
                end = start;
                while (end < line.Length && line[end] != ' ') end++;
                
                // Parse and store
#if NETCOREAPP
                cuts[cutsRead++] = int.Parse(line.AsSpan(start, end - start));
#else
                cuts[cutsRead++] = int.Parse(line.Substring(start, end - start));
#endif
                start = end + 1;
            }
        }

        // Solve using First Fit Decreasing with parallel sorting
        List<List<int>> bins = SolveFFD(cuts, stockLength);

        // Output results
        using var outStream = new BufferedStream(Console.OpenStandardOutput(), BufferSize);
        using var writer = new StreamWriter(outStream);
        
        writer.WriteLine(bins.Count);
        foreach (var bin in bins)
        {
            writer.WriteLine(string.Join(" ", bin));
        }
    }

    /// <summary>
    /// First Fit Decreasing algorithm with optimizations.
    /// Time complexity: O(n log n) for sorting + O(n log k) for bin packing where k is number of bins.
    /// </summary>
    private static List<List<int>> SolveFFD(int[] cuts, int stockLength)
    {
        int n = cuts.Length;
        if (n == 0) return new List<List<int>>();

        // Create array of indices and sort by cut length (descending) in parallel
        int[] indices = new int[n];
        for (int i = 0; i < n; i++) indices[i] = i;

        // Parallel sort for large arrays
        if (n > 50000)
        {
            ParallelSort(indices, cuts);
        }
        else
        {
            Array.Sort(indices, (a, b) => cuts[b].CompareTo(cuts[a]));
        }

        // Bin packing structures
        // Using simple arrays for maximum performance
        int[] binRemaining = new int[n]; // Remaining capacity in each bin
        int binCount = 0;

        // We need to track which cuts go in which bin
        // Using jagged array approach with lists
        var bins = new List<List<int>>(n / 4 + 1);

        // First Fit: try to place in first bin that fits
        // For large problems, use binary search optimization
        if (n > 100000)
        {
            // Use binary search tree for efficient first-fit
            return FirstFitWithTree(indices, cuts, stockLength);
        }
        else
        {
            // Simple linear first-fit for smaller problems
            for (int i = 0; i < n; i++)
            {
                int cutIndex = indices[i];
                int cutLen = cuts[cutIndex];

                // Find first bin with enough capacity
                int foundBin = -1;
                for (int b = 0; b < binCount; b++)
                {
                    if (binRemaining[b] >= cutLen)
                    {
                        foundBin = b;
                        break;
                    }
                }

                if (foundBin == -1)
                {
                    // Create new bin
                    binRemaining[binCount] = stockLength - cutLen;
                    bins.Add(new List<int> { cutIndex });
                    binCount++;
                }
                else
                {
                    binRemaining[foundBin] -= cutLen;
                    bins[foundBin].Add(cutIndex);
                }
            }
        }

        return bins;
    }

    /// <summary>
    /// Optimized First Fit using a balanced tree structure for O(log n) bin lookup.
    /// </summary>
    private static List<List<int>> FirstFitWithTree(int[] sortedIndices, int[] cuts, int stockLength)
    {
        int n = sortedIndices.Length;
        var bins = new List<List<int>>();
        
        // Use SortedSet with duplicate handling via tuple
        // Key: remaining capacity, Value: bin index
        var availableBins = new SortedSet<(int remaining, int index)>();
        int binCounter = 0;

        for (int i = 0; i < n; i++)
        {
            int cutIndex = sortedIndices[i];
            int cutLen = cuts[cutIndex];

            // Find bin with smallest remaining capacity that still fits
            var candidate = new (int, int)Comparer 
            { 
                Compare = (a, b) => 
                {
                    int cmp = a.Item1.CompareTo(b.Item1);
                    return cmp != 0 ? cmp : a.Item2.CompareTo(b.Item2);
                }
            };
            
            // Find first bin with remaining >= cutLen
            var found = FindFirstFit(availableBins, cutLen);
            
            if (found != default)
            {
                availableBins.Remove(found);
                int newRemaining = found.remaining - cutLen;
                bins[found.index].Add(cutIndex);
                if (newRemaining > 0)
                {
                    availableBins.Add((newRemaining, found.index));
                }
            }
            else
            {
                // Create new bin
                bins.Add(new List<int> { cutIndex });
                int newRemaining = stockLength - cutLen;
                if (newRemaining > 0)
                {
                    availableBins.Add((newRemaining, binCounter));
                }
                binCounter++;
            }
        }

        return bins;
    }

    private static (int remaining, int index) FindFirstFit(SortedSet<(int remaining, int index)> set, int cutLen)
    {
        foreach (var item in set)
        {
            if (item.remaining >= cutLen)
                return item;
        }
        return default;
    }

    /// <summary>
    /// Parallel quicksort for large arrays.
    /// </summary>
    private static void ParallelSort(int[] indices, int[] values)
    {
        int n = indices.Length;
        
        // Use parallel quicksort with threshold
        ParallelQuickSort(indices, values, 0, n - 1, (int)(Math.Log(n, 2) * 2));
    }

    private static void ParallelQuickSort(int[] indices, int[] values, int left, int right, int depth)
    {
        const int threshold = 10000;
        
        if (right - left <= threshold || depth <= 0)
        {
            // Sequential sort for small partitions
            Array.Sort(indices, left, right - left + 1, (a, b) => values[b].CompareTo(values[a]));
            return;
        }

        int pivotIndex = Partition(indices, values, left, right);
        
        if (pivotIndex > 1)
        {
            if (depth > 0)
            {
                Parallel.Invoke(
                    () => ParallelQuickSort(indices, values, left, pivotIndex - 1, depth - 1),
                    () => ParallelQuickSort(indices, values, pivotIndex + 1, right, depth - 1)
                );
            }
            else
            {
                ParallelQuickSort(indices, values, left, pivotIndex - 1, 0);
                ParallelQuickSort(indices, values, pivotIndex + 1, right, 0);
            }
        }
    }

    private static int Partition(int[] indices, int[] values, int left, int right)
    {
        int pivotValue = values[indices[right]];
        int i = left - 1;
        
        for (int j = left; j < right; j++)
        {
            if (values[indices[j]] >= pivotValue)
            {
                i++;
                int temp = indices[i];
                indices[i] = indices[j];
                indices[j] = temp;
            }
        }
        
        int temp2 = indices[i + 1];
        indices[i + 1] = indices[right];
        indices[right] = temp2;
        
        return i + 1;
    }
}
