```spirv
; SPIR-V
; Version: 1.0
; Generator: Khronos Glslang Reference Front End
; Bound: 128

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %arr_worldPos ArrayStride 12
OpDecorate %arr_normal ArrayStride 12
OpDecorate %arr_color ArrayStride 12
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%float_0_025 = OpConstant %float 0.025

; Input Blocks/Variables
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_worldPos = OpTypeArray %v3float %uint_4
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos
%worldPos = OpVariable %ptr_in_arr_worldPos Input

%arr_normal = OpTypeArray %v3float %uint_4
%ptr_in_arr_normal = OpTypePointer Input %arr_normal
%normal = OpVariable %ptr_in_arr_normal Input

%arr_color = OpTypeArray %v3float %uint_4
%ptr_in_arr_color = OpTypePointer Input %arr_color
%color = OpVariable %ptr_in_arr_color Input

; Output Blocks/Variables
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

; Pointers for access
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %3
%5 = OpLabel

; --- Load Positions ---
; gl_in[0].xyz (prev)
%pos0_full = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos0_v4 = OpLoad %v4float %pos0_full
%pos0 = OpVectorShuffle %v3float %pos0_v4 %pos0_v4 0 1 2

; gl_in[1].xyz (start)
%pos1_full = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%pos1_v4 = OpLoad %v4float %pos1_full
%pos1 = OpVectorShuffle %v3float %pos1_v4 %pos1_v4 0 1 2

; gl_in[2].xyz (end)
%pos2_full = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%pos2_v4 = OpLoad %v4float %pos2_full
%pos2 = OpVectorShuffle %v3float %pos2_v4 %pos2_v4 0 1 2

; gl_in[3].xyz (next)
%pos3_full = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
%pos3_v4 = OpLoad %v4float %pos3_full
%pos3 = OpVectorShuffle %v3float %pos3_v4 %pos3_v4 0 1 2

; --- Compute Tangents ---
; t0 = normalize(pos2 - pos0)
%diff0 = OpFSub %v3float %pos2 %pos0
%t0_3d = OpExtInst %v3float %1 Normalize %diff0

; t1 = normalize(pos3 - pos1)
%diff1 = OpFSub %v3float %pos3 %pos1
%t1_3d = OpExtInst %v3float %1 Normalize %diff1

; Extract 2D tangents (xy plane)
%t0 = OpVectorShuffle %v2float %t0_3d %t0_3d 0 1
%t1 = OpVectorShuffle %v2float %t1_3d %t1_3d 0 1

; --- Compute Perpendiculars ---
; perp = vec2(-y, x)
%t0_y = OpCompositeExtract %float %t0 1
%t0_x = OpCompositeExtract %t0 0
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%t1_y = OpCompositeExtract %float %t1 1
%t1_x = OpCompositeExtract %float %t1 0
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; --- Load Colors ---
%color1_ptr = OpAccessChain %ptr_in_v3float %color %int_1
%color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_in_v3float %color %int_2
%color2 = OpLoad %v3float %color2_ptr

; --- Calculate Offsets ---
%offset0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
%offset1 = OpVectorTimesScalar %v2float %perp1 %float_0_025

; --- Emit Vertex 1 (Start, +Perp) ---
; pos1.xy + offset0
%pos1_xy = OpVectorShuffle %v2float %pos1 %pos1 0 1
%p1_xy = OpFAdd %v2float %pos1_xy %offset0
%pos1_z = OpCompositeExtract %float %pos1 2
%pos1_w = OpCompositeExtract %float %pos1_v4 3
%pos1_new_v3 = OpCompositeConstruct %v3float %p1_xy %pos1_z
%v1_pos = OpCompositeConstruct %v4float %pos1_new_v3 %pos1_w

%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos_ptr %v1_pos
OpStore %outColor %color1
OpEmitVertex

; --- Emit Vertex 2 (Start, -Perp) ---
%neg_offset0 = OpFNegate %v2float %offset0
%p2_xy = OpFAdd %v2float %pos1_xy %neg_offset0
%pos2_new_v3 = OpCompositeConstruct %v3float %p2_xy %pos1_z
%v2_pos = OpCompositeConstruct %v4float %pos2_new_v3 %pos1_w

OpStore %out_pos_ptr %v2_pos
OpStore %outColor %color1
OpEmitVertex

; --- Emit Vertex 3 (End, +Perp) ---
%pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
%p3_xy = OpFAdd %v2float %pos2_xy %offset1
%pos2_z = OpCompositeExtract %float %pos2 2
%pos2_w = OpCompositeExtract %float %pos2_v4 3
%pos3_new_v3 = OpCompositeConstruct %v3float %p3_xy %pos2_z
%v3_pos = OpCompositeConstruct %v4float %pos3_new_v3 %pos2_w

OpStore %out_pos_ptr %v3_pos
OpStore %outColor %color2
OpEmitVertex

; --- Emit Vertex 4 (End, -Perp) ---
%neg_offset1 = OpFNegate %v2float %offset1
%p4_xy = OpFAdd %v2float %pos2_xy %neg_offset1
%pos4_new_v3 = OpCompositeConstruct %v3float %p4_xy %pos2_z
%v4_pos = OpCompositeConstruct %v4float %pos4_new_v3 %pos2_w

OpStore %out_pos_ptr %v4_pos
OpStore %outColor %color2
OpEmitVertex

OpEndPrimitive

OpFunctionEnd
```
