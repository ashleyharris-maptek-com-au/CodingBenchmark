        private static void GreedyInitialOrder()
        {

                    degIn[v]--;

            // Reset state for this logic

            // Or just a simple loop.

            var sources = new HashSet<int>();
            var sinks = new HashSet<int>();
            for(int i=0; i<n; i++) sources.Add(i);
            for(int i=0; i<n; i++) sinks.Add(i);
            
            // This is getting complex to maintain efficient data structures.
            // Fallback: 
            // The `TryImprove` logic is powerful. It can fix a random order or a simple default order.
            // We can just initialize with 0..n-1 or random. `TryImprove` will find cycles and resolve them.
            
            // Initialize: Assume no nodes removed (order is 0..n-1).
            // Wait, `currentOrderUsed` tracks the *order*, not the FVS.
            // `currentOrderUsed[i] == true` means `i` is in the valid Topological Order (NOT in FVS).
            // Yes.
            
            // Initial state: Valid Topological Order is empty.
            // All nodes are effectively in FVS.
            // `TryImprove` finds a cycle. If cycle exists, and we have all nodes in FVS, that's wrong.
            // `TryImprove` logic: "If cycle found, swap something into `currentOrder`".
            // If `currentOrderCount == 0`, any cycle found will be swapped in.
            
            // Correct Logic for Initialization:
            // Start with `currentOrder` containing all vertices that are not in cycles? No.
            // Start with `currentOrder` empty.
            // Use `TryImprove` which calls `FindAnyCycle`.
            // Since `currentOrder` is empty, `FindAnyCycle` searches the whole graph.
            // It finds a cycle.
            // Then `TryImprove` picks a node `u` from cycle.
            // If `u` is NOT in `currentOrder`, we add it.
            // This builds up the DAG.
            // We repeat this. `FindAnyCycle` ignores nodes in `currentOrder`.
            // Eventually `FindAnyCycle` returns null (no more cycles).
            // Result: a maximal DAG.
            // Is it maximum? Not necessarily. That's what the iteration in `Solve` is for.
            // The iteration tries to perturb/exchange.
            
            // This simple bootstrapping works.
            // `TryImprove` initially just builds a maximal DAG greedily.
            // Then it tries to improve it.
            
            // Wait, `TryImprove` swaps `u` with a node in cycle.
            // If `u` is NOT in `currentOrder` (which is true initially for any `u`), `add = true`.
            // It adds `u` to `currentOrder`.
            // This increases the size of the Topological Order (Decreases FVS).
            // This is exactly what we want: grow the DAG as large as possible.
            
            // So we can just rely on the loop in `Solve`.
            // Start with `currentOrderCount = 0`.
            // `TryImprove` will add nodes until no cycles remain.
            // Then it will try swaps.

        private static bool TryImprove()
        {
            bool changed = false;
            
            // We iterate through nodes.
            // Strategy:
            // 1. Find a cycle in the remaining graph.
            // 2. If found, we must break it.
            //    Since we maintain a valid partial topological order (set T),
            //    the remaining graph G \ T contains at least one cycle.
            //    We pick a node `u` in the cycle.
            //    Ideally, we want to add `u` to T.
            //    But adding `u` creates a cycle in T if `u` has edges to/from T that violate order?
            //    Wait, `FindAnyCycle` only looks at G \ T.
            //    If we add `u` to T, does it conflict with T?
            //    The definition of FVS S is: G \ S is DAG.
            //    So T = V \ S.
            //    We construct T.
            //    Any node in G \ T can be added to T if and only if it doesn't create a cycle?
            //    No. If `u` is in G \ T, it means we decided to remove it (or haven't added it yet).
            //    The graph G \ T contains cycles. `u` is part of one.
            //    We can simply add `u` to T.
            //    The cycle in G \ T that contained `u` is now broken in G \ (T u {u}).
            //    Is G \ (T u {u}) necessarily a DAG?
            //    No, G \ T might have other cycles not involving `u`.
            //    So adding `u` to T is valid?
            //    We just need T to be a DAG.
            //    T is currently a DAG.
            //    If we add `u` to T, is T u {u} a DAG?
            //    Not necessarily! `u` might have edges (u -> v) and (w -> u) where v, w in T
            //    such that v comes before w in T. This creates a cycle in T u {u}.
            //    
            //    This implies we must maintain topological sort constraints for T.
            //    It's not enough to just "remove nodes". We must order them.
            //    My state `currentOrder` IS the topological sort of T.
            //    
            //    Algorithm:
            //    1. Find cycle in G \ T. (Nodes not in `currentOrder`).
            //    2. If no cycle, G \ T is a DAG.
            //       Can we pick a node u in G \ T and add to T?
            //       Yes, if we can fit `u` into the topological order.
            //       We can always insert `u` into the order if it doesn't create a cycle.
            //       Since G \ T is a DAG, `u` has specific relations.
            //       If `u` connects to T, we must check constraints.
            //       Actually, if G \ T is a DAG, we can just topologically sort G \ T
            //       and merge it with T! This would reduce FVS size to 0 (if graph is DAG).
            //       
            //       Wait, my `FindAnyCycle` logic was:
            //       If cycle found, pick `u`. If `u` not in T, add it.
            //       BUT `u` might conflict with T!
            //       
            //       Correct logic for `TryImprove`:
            //       We need to check if we can add `u` to T.
            //       Or, better: Swap.
            //       If we find a cycle C in G \ T.
            //       Every node in C is currently "removed" (in FVS).
            //       We want to keep nodes in T (DAG part).
            //       We want to move `u` from FVS to T.
            //       But `u` is in a cycle C.
            //       Moving `u` to T breaks cycle C.
            //       However, `u` might form a new cycle with T.
            //       
            //       Let's refine the "valid swap" condition.
            //       Swap v \in T with u \in S (FVS).
            //       New T' = T \ {v} U {u}.
            //       We want T' to be a DAG.
            //       1. T \ {v} is DAG.
            //       2. (T \ {v}) U {u} is DAG.
            //       This requires `u` not to create a cycle with T \ {v}.
            //       
            //       This is hard to check efficiently.
            //       
            //       Alternative: Just use the cycle finding to guide a simple heuristic.
            //       We want to build a Maximal DAG T.
            //       Repeat:
            //         Pick a node `u` not in T.
            //         Can we add `u` to T? (Does T U {u} have a cycle?)
            //         We check this by verifying if adding `u` creates a backedge in `currentOrder`.
            //         Or just run DFS for cycle on T U {u}. O(N+M). Slow.
            //         
            //         Faster check:
            //         `u` can be appended to `currentOrder` if there is no edge `u -> v` where v \in T.
            //         Wait, for topological order, `u` can be last if it has no outgoing edges to T?
            //         No. Topological order `u` comes last means no edges `u -> v` where v in T.
            //         Actually, standard definition:
            //         If `u` is last, no `u -> v` exists.
            //         If `u` is first, no `v -> u` exists.
            //         
            //         So we can try to append `u` to `currentOrder` if `u` has no outgoing edges to T.
            //         (Check `adj[u]` intersection with T).
            //         Or prepend `u` if `u` has no incoming edges from T.
            //         
            //         If we can add `u` to T, we do it. FVS size decreases.
            //         
            //       If we cannot add any more nodes to T (no valid append/prepend), T is "saturated".
            //       Then T is a Maximal Induced DAG?
            //       Not necessarily maximum.
            //       
            //       But it's a good starting point.
            //       Then we can try swaps: remove `v` from T, try to add `u` from S.
            //       
            //       
            // Let's implement this "Saturate T" logic.
            
            // Reset order if empty? No, build it up.
            
            bool action = true;
            while (action)
            {
                action = false;
                
                // Try to grow T
                // Iterate all nodes not in T
                for (int i = 0; i < n; i++)
                {
                    if (currentOrderUsed[i]) continue;
                    
                    // Can we append i?
                    // Condition: No edge i -> v for any v in T.
                    bool canAppend = true;
                    foreach (var v in adj[i])
                    {
                        if (currentOrderUsed[v]) { canAppend = false; break; }
                    }
                    
                    if (canAppend)
                    {
                        currentOrder[currentOrderCount++] = i;
                        currentOrderUsed[i] = true;
                        action = true;
                        changed = true;
                        continue;
                    }
                    
                    // Can we prepend i?
                    // Condition: No edge v -> i for any v in T.
                    bool canPrepend = true;
                    foreach (var v in radj[i])
                    {
                        if (currentOrderUsed[v]) { canPrepend = false; break; }
                    }
                    
                    if (canPrepend)
                    {
                        // Shift order
                        for (int k = currentOrderCount; k > 0; k--) currentOrder[k] = currentOrder[k - 1];
                        currentOrder[0] = i;
                        currentOrderCount++;
                        currentOrderUsed[i] = true;
                        action = true;
                        changed = true;
                        continue;
                    }
                }
                
                if (action) continue; // Keep growing
                
                // Saturated. Try to Swap to enable growth.
                // Pick a node `u` not in T.
                // `u` has edges to T and from T.
                // Identify the conflict.
                // `u` has `out_edges` to T (blocking append).
                // `u` has `in_edges` from T (blocking prepend).
                
                // If we remove a node `v` from T:
                // Maybe `u` can fit in?
                // This is complex.
                
                // Fallback: Local Search on FVS set directly.
                // Current FVS S = V \ T.
                // Pick `u` in S.
                // If we move `u` to T, T gets a cycle.
                // Find a cycle in T U {u} involving `u`.
                // Cycle C.
                // Remove any `v` in C, v != u.
                // Swap u, v.
                // Size of T stays same.
                // But maybe this unlocks future growth?
                
                // We need to find a cycle involving `u` efficiently.
                // `u` is added. Find path in T from successor of `u` to predecessor of `u`.
                // Path P in T.
                // If P exists, cycle is `u -> ... -> u`.
                // We can remove any node on P.
                
                // Random heuristic:
                // Pick random `u` in S.
                // Try to find `v` in T such that swapping is valid.
                // Valid swap: T \ {v} U {u} is DAG.
                // 1. Check T U {u} for cycle.
                //    This happens if there is a path in T from adj[u] to radj[u].
                //    i.e. exists `x` in adj[u], `y` in radj[u] such that path `x -> ... -> y` exists in T.
                //    If no such path, `u` can be added (FVS size decreases!).
                //    So we check this path existence.
                //    If path exists, cycle C exists.
                //    Any `v` on path `x...y` is a candidate for swap.
                //    Pick one `v`.
                //    New T' = T \ {v} U {u}.
                //    T' is DAG.
                //    Did we improve?
                //    We want to minimize |S|.
                //    Swapping keeps |S| same.
                //    Why do it?
                //    To escape local optimum.
                //    Maybe removing `v` allows neighbors of `v` in S to join T later.
                
                // We perform a limited number of swaps or random moves.
                break; // Exit while loop for this iteration of TryImprove
            }
            
            // Try to find a direct reduction (move S -> T)
            for (int i = 0; i < n; i++)
            {
                if (currentOrderUsed[i]) continue;
                if (CanAddToOrder(i))
                {
                    AddToOrder(i);
                    changed = true;
                }
            }
            
            // If stuck, try a swap
            if (!changed)
            {
                // Pick a candidate from S
                // Heuristic: pick one with high degree?
                // Random is fine.
                for(int iter=0; iter<10; iter++)
                {
                    int u = random.Next(n);
                    if (currentOrderUsed[u]) continue;
                    
                    // Try to swap u with a neighbor in T?
                    // Check if we can add u to T by removing one v.
                    // Find a cycle involving u in T U {u}.
                    // Cycle must exist if CanAddToOrder returned false?
                    // Not necessarily. CanAddToOrder checks strict append/prepend.
                    // Maybe u can be inserted in middle.
                    
                    // Check if T U {u} has cycle.
                    // Path exists from adj[u] to radj[u]?
                    int x = -1, y = -1;
                    bool pathExists = false;
                    
                    // Mark adj[u] in T
                    // For each pred in radj[u] in T, check reachability.
                    // We need fast reachability in DAG.
                    // With currentOrder, checking path x -> y is easy?
                    // x must appear before y in topological order.
                    // Path x -> y exists iff reachable(x, y).
                    // We don't store transitive closure.
                    // BFS/DFS from adj[u] to see if we hit radj[u].
                    
                    // Optimization: T is topologically sorted.
                    // Forward edges only go from smaller index to larger index.
                    // Path x -> ... -> y implies Index(x) < Index(y).
                    // So we only check pairs (succ, pred) where Index(succ) < Index(pred).
                    
                    var successors = new List<int>();
                    foreach(var v in adj[u]) if (currentOrderUsed[v]) successors.Add(v);
                    
                    var predecessors = new HashSet<int>();
                    foreach(var v in radj[u]) if (currentOrderUsed[v]) predecessors.Add(v);
                    
                    // BFS from successors
                    var q = new Queue<int>();
                    var visited = new bool[n];
                    foreach(var s in successors) 
                    {
                        // Check strict ordering constraint for DAG path
                        // If we BFS, we follow edges.
                        if(predecessors.Contains(s)) 
                        {
                            // Cycle u -> s -> u
                            // Swap u with s.
                            RemoveFromOrder(s);
                            AddToOrder(u);
                            return true; // Changed
                        }
                        q.Enqueue(s);
                        visited[s] = true;
                    }
                    
                    int nodeInPath = -1;
                    
                    while(q.Count > 0)
                    {
                        int curr = q.Dequeue();
                        foreach(var next in adj[curr])
                        {
                            if (!currentOrderUsed[next]) continue;
                            if (visited[next]) continue;
                            
                            if (predecessors.Contains(next))
                            {
                                // Found cycle u -> ... -> next -> ... -> u
                                nodeInPath = next;
                                break; // Found a path
                            }
                            visited[next] = true;
                            q.Enqueue(next);
                        }
                        if (nodeInPath != -1) break;
                    }
                    
                    if (nodeInPath != -1)
                    {
                        // Cycle involves u and nodeInPath.
                        // Swap u and nodeInPath?
                        // If we remove nodeInPath, can we add u?
                        // Yes, removing nodeInPath breaks the path.
                        // Does T \ {nodeInPath} U {u} have other cycles?
                        // Maybe. But for local search, breaking one cycle is a good step.
                        // However, we prefer swapping if it leads to size reduction.
                        // But here we are just swapping.
                        // Is it a valid swap?
                        // Yes. T' is DAG.
                        RemoveFromOrder(nodeInPath);
                        AddToOrder(u);
                        changed = true;
                        break; 
                    }
                    else
                    {
                        // No path from adj[u] to radj[u] in T.
                        // So T U {u} is a DAG!
                        // Why was CanAddToOrder false?
                        // Because u couldn't be appended (had outgoing edges to T)
                        // and couldn't be prepended (had incoming edges from T).
                        // But it can be inserted in the middle!
                        // We should add it!
                        // This decreases FVS size.
                        AddToOrder(u); // Need a general insert function
                        changed = true;
                        break;
                    }
                }
            }
            
            return changed;
        }

        private static void RemoveFromOrder(int v)
        {
            currentOrderUsed[v] = false;
            // Rebuilding array is lazy but correct and simple.
            // Since T is topologically sorted, we can find index and shift.
            int idx = -1;
            for(int i=0; i<currentOrderCount; i++)
            {
                if(currentOrder[i] == v) { idx = i; break; }
            }
            if (idx != -1)
            {
                for(int i=idx; i<currentOrderCount-1; i++) currentOrder[i] = currentOrder[i+1];
                currentOrderCount--;
            }
        }

        private static void AddToOrder(int u)
        {
            // General insertion into topological sort.
            // Find a valid position.
            // Index must be > all predecessors in T.
            // Index must be < all successors in T.
            
            int minIndex = 0;
            foreach(var v in radj[u])
            {
                if(currentOrderUsed[v])
                {
                    // Find index of v
                    // This is O(N). Slow.
                    // Optimization: Maintain rank/position map.
                    // For now O(N) is acceptable for insertion logic given N is manageable or updates rare.
                    int idxV = GetOrderIndex(v);
                    if (idxV + 1 > minIndex) minIndex = idxV + 1;
                }
            }
            
            int maxIndex = currentOrderCount;
            foreach(var v in adj[u])
            {
                if(currentOrderUsed[v])
                {
                    int idxV = GetOrderIndex(v);
                    if (idxV < maxIndex) maxIndex = idxV;
                }
            }
            
            if (minIndex <= maxIndex) // Valid slot exists
            {
                // Insert at minIndex
                // Shift elements
                for(int i=currentOrderCount; i>minIndex; i--) currentOrder[i] = currentOrder[i-1];
                currentOrder[minIndex] = u;
                currentOrderCount++;
                currentOrderUsed[u] = true;
            }
            else
            {
                 // Conflict. Should not happen if called after cycle check.
                 // But if it does, we just don't add.
            }
        }
        
        private static int GetOrderIndex(int v)
        {
            for(int i=0; i<currentOrderCount; i++) if(currentOrder[i] == v) return i;
            return -1;
        }

        private static bool CanAddToOrder(int u)
        {
            // Check if u can be added to T without violating DAG property.
            // Essentially: Is there a path in T from adj[u] to radj[u]?
            // If yes, cycle.
            // If no, DAG.
            
            // Simple check for Append/Prepend first (fast)
            bool canAppend = true;
            foreach (var v in adj[u]) if (currentOrderUsed[v]) { canAppend = false; break; }
            if (canAppend) return true;
            
            bool canPrepend = true;
            foreach (var v in radj[u]) if (currentOrderUsed[v]) { canPrepend = false; break; }
            if (canPrepend) return true;
            
            // Check middle insertion
            // Path from adj[u] to radj[u]?
            // Use BFS.
            var q = new Queue<int>();
            var visited = new bool[n];
            
            var targets = new HashSet<int>();
            foreach(var v in radj[u]) if(currentOrderUsed[v]) targets.Add(v);
            if(targets.Count == 0) return true; // No predecessors in T
            
            foreach(var v in adj[u])
            {
                if(currentOrderUsed[v])
                {
                    if(targets.Contains(v)) return false; // 2-cycle u->v->u
                    q.Enqueue(v);
                    visited[v] = true;
                }
            }
            
            while(q.Count > 0)
            {
                int curr = q.Dequeue();
                foreach(var next in adj[curr])
                {
                    if(!currentOrderUsed[next] || visited[next]) continue;
                    if(targets.Contains(next)) return false; // Cycle found
                    visited[next] = true;
                    q.Enqueue(next);
                }
            }
            
            return true; // No cycle
        }
    }
