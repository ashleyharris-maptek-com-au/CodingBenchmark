; Input Arrays (Size 2 for InputLines)
%uint = OpTypeInt 32 0

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2

; UBO

%main = OpFunction %void None %3

; --- Load Inputs ---
; P0 (Vertex 0)
%p0_pos = OpLoad %v4float %gl_in_0_pos
%p0_world = OpLoad %v3float %gl_in_0_world
%p0_color = OpLoad %v3float %gl_in_0_color

; P1 (Vertex 1)
%p1_pos = OpLoad %v4float %gl_in_1_pos
%p1_world = OpLoad %v3float %gl_in_1_world
%p1_color = OpLoad %v3float %gl_in_1_color

; --- Calculate Perpendicular Direction ---
; View space calculation for constant screen-space thickness
; We access the View matrix (Member 1)
%view_mat = OpLoad %mat4v4float %ubo_view

; Transform world positions to view space
%v4_p0_w = OpCompositeConstruct %v4float %p0_world_0 %p0_world_1 %p0_world_2 %float_1
%v4_p1_w = OpCompositeConstruct %v4float %p1_world_0 %p1_world_1 %p1_world_2 %float_1

%p0_view_v4 = OpMatrixTimesVector %v4float %view_mat %v4_p0_w
%p1_view_v4 = OpMatrixTimesVector %v4float %view_mat %v4_p1_w

; Extract XY for 2D vector math in view space
%p0_view_xy = OpVectorShuffle %v2float %p0_view_v4 %p0_view_v4 0 1
%p1_view_xy = OpVectorShuffle %v2float %p1_view_v4 %p1_view_v4 0 1

; Vector from p0 to p1
%line_vec = OpFSub %v2float %p1_view_xy %p0_view_xy

; Rotate 90 degrees to get perpendicular: (x, y) -> (-y, x)
%perp_x = OpCompositeExtract %float %line_vec 1
%perp_y = OpCompositeExtract %float %line_vec 0
%neg_perp_y = OpFNegate %float %perp_y
%perp = OpCompositeConstruct %v2float %neg_perp_y %perp_x

; Normalize perpendicular
%perp_len = OpLength %float %perp
%perp_norm = OpVectorTimesScalar %v2float %perp %perp_len_inv

; --- Emit Vertex 0 (P0 + Thick) ---
%offset_start = OpVectorTimesScalar %v2float %perp_norm %float_0_04
%pos_start_off_xy = OpFAdd %v2float %p0_view_xy %offset_start
; Reconstruct clip space position:
; XY: modified. ZW: copy from original P0 (keeps depth and perspective).
%pos_v0_xy = OpVectorShuffle %v2float %pos_start_off_xy %pos_start_off_xy 0 1
%gl_pos_v0 = OpVectorShuffle %v4float %pos_v0_xy %p0_pos 0 1 2 3

; Output
OpStore %gl_out_0 %gl_pos_v0
OpStore %outColor_var %p0_color
OpEmitVertex

; --- Emit Vertex 1 (P0 - Thick) ---
%pos_start_off_xy_neg = OpFSub %v2float %p0_view_xy %offset_start
%pos_v1_xy = OpVectorShuffle %v2float %pos_start_off_xy_neg %pos_start_off_xy_neg 0 1
%gl_pos_v1 = OpVectorShuffle %v4float %pos_v1_xy %p0_pos 0 1 2 3

; Output
OpStore %gl_out_0 %gl_pos_v1
OpStore %outColor_var %p0_color
OpEmitVertex

; --- Emit Vertex 2 (P1 + Thin) ---
%offset_end = OpVectorTimesScalar %v2float %perp_norm %float_0_005
%pos_end_off_xy = OpFAdd %v2float %p1_view_xy %offset_end
%pos_v2_xy = OpVectorShuffle %v2float %pos_end_off_xy %pos_end_off_xy 0 1
%gl_pos_v2 = OpVectorShuffle %v4float %pos_v2_xy %p1_pos 0 1 2 3

; Output
OpStore %gl_out_0 %gl_pos_v2
OpStore %outColor_var %p1_color
OpEmitVertex

; --- Emit Vertex 3 (P1 - Thin) ---
%pos_end_off_xy_neg = OpFSub %v2float %p1_view_xy %offset_end
%pos_v3_xy = OpVectorShuffle %v2float %pos_end_off_xy_neg %pos_end_off_xy_neg 0 1
%gl_pos_v3 = OpVectorShuffle %v4float %pos_v3_xy %p1_pos 0 1 2 3

; Output
OpStore %gl_out_0 %gl_pos_v3
OpStore %outColor_var %p1_color

OpEndPrimitive

OpReturn
