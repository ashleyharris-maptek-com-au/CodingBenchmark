    # Allocate
    num_lits = 2 * N + 2
    occur = [array.array('I') for _ in range(num_lits)]
    for i in range(num_lits):
        if counts[i] > 0:
            occur[i] = array.array('I')
            occur[i].extend([0] * counts[i]) # Pre-extend
            # Reset count to use as cursor
            counts[i] = 0
            
    # Fill `occur`
    # We iterate `clauses_list` with index.
    # Clause index is `i // 3`.
    
    for i, lit in enumerate(clauses_list):
        v = abs(lit)
        lit_idx = (v << 1) | (1 if lit < 0 else 0)
        
        # Get insertion position
        pos = counts[lit_idx]
        occur[lit_idx][pos] = i // 3
        counts[lit_idx] = pos + 1
        
    # Now we have the graph.
    # We have `clauses_list` (flat array of literals).
    # We need to check clause satisfaction.
    # Clause i contains literals at 3*i, 3*i+1, 3*i+2.
    
    # Assignment: boolean array for variables 1..N.
    # assignment[v] = True/False.
    assignment = [random.choice([True, False]) for _ in range(N + 1)]
    
    # Tracking unsatisfied clauses
    # Initially, check all.
    # `unsat` set or list.
    # List of indices.
    unsat = []
    # Fast check: we can use an array of booleans `is_unsat`?
    # 85M booleans = 85MB. Acceptable.
    # But `unsat` list allows random pick.
    # We can just append to `unsat`.
    
    # Helper to check clause
    def is_clause_satisfied(c_idx):
        # Access literals
        base = c_idx * 3
        l1 = clauses_list[base]
        l2 = clauses_list[base+1]
        l3 = clauses_list[base+2]
        
        # Eval
        v1 = assignment[abs(l1)]
        if l1 < 0: v1 = not v1
        if v1: return True
        
        v2 = assignment[abs(l2)]
        if l2 < 0: v2 = not v2
        if v2: return True
        
        v3 = assignment[abs(l3)]
        if l3 < 0: v3 = not v3
        if v3: return True
        
        return False

    for i in range(M):
        if not is_clause_satisfied(i):
            unsat.append(i)
            
    # WalkSAT Loop
    max_steps = 100000000 # Large limit
    p = 0.5 # Noise parameter
    
    # Optimization:
    # `is_unsat` boolean array to efficiently check if a clause is already in `unsat` list?
    # We need to update status when flipping.
    # When a variable flips, clauses containing it change status.
    # We iterate `occur` lists.
    # If a clause becomes satisfied: remove from `unsat`.
    # If a clause becomes unsatisfied: add to `unsat`.
    #
    # Removal from `unsat` (list) is O(N).
    # This is the bottleneck.
    #
    # Standard WalkSAT optimization:
    # Don't explicitly remove.
    # Just keep a boolean array `clause_status` (True=Sat, False=Unsat).
    # When picking a clause from `unsat` list:
    #   Pick random index.
    #   Check `clause_status`.
    #   If satisfied, swap with end of list, pop, pick another.
    #   If unsatisfied, proceed.
    # This keeps the list clean-ish over time.
    
    clause_status = [False] * M # False = Unsatisfied
    
    # Initialize clause_status and unsat
    # (Already did above, but let's set status correctly)
    # Initially all False.
    # If satisfied, set True.
    
    for i in range(M):
        if is_clause_satisfied(i):
            clause_status[i] = True
        else:
            unsat.append(i)
            
    steps = 0
    while unsat and steps < max_steps:
        steps += 1
        
        # Pick an unsatisfied clause
        # Random pick
        idx_in_list = random.randint(0, len(unsat) - 1)
        c_idx = unsat[idx_in_list]
        
        # Validate
        if clause_status[c_idx]:
            # It was satisfied by a previous flip, but not removed.
            # Remove it now.
            unsat[idx_in_list] = unsat[-1]
            unsat.pop()
            continue
            
        # We have a genuine unsatisfied clause c_idx
        # Literals:
        base = c_idx * 3
        lits = [clauses_list[base], clauses_list[base+1], clauses_list[base+2]]
        
        # Choose variable to flip
        # With prob p: random
        # With prob 1-p: greedy (min break)
        
        if random.random() < p:
            # Random walk
            chosen_lit = random.choice(lits)
        else:
            # Greedy
            # Calculate breaks for each literal
            best_lit = None
            min_breaks = float('inf')
            
            # Optimization: If we find 0 breaks, pick it immediately.
            
            for lit in lits:
                var = abs(lit)
                # We are considering flipping `var`.
                # This satisfies `lit` (and clause c_idx).
                # But it breaks clauses where `var` appears with opposite polarity.
                # We need to find clauses where `var` is currently satisfying it ALONE.
                
                # Find opposite literal index
                # If lit > 0 (var=True), we are making var=False.
                # We break clauses where `-var` appears.
                # Lit `lit` -> index `lit_idx`.
                # Opposite lit `-lit` -> index `opp_idx`.
                
                opp_idx = (abs(lit) << 1) | (0 if lit > 0 else 1)
                
                breaks = 0
                # Iterate over clauses containing `opp_idx`
                # These are clauses containing `-lit`.
                # Since we are flipping to satisfy `lit`, we make `-lit` False.
                # If a clause containing `-lit` was satisfied ONLY by `-lit`, it breaks.
                
                for affected_c_idx in occur[opp_idx]:
                    if clause_status[affected_c_idx]:
                        # Check if it was satisfied ONLY by this literal
                        # Scan literals of affected_c_idx
                        b = affected_c_idx * 3
                        count_true = 0
                        for k in range(3):
                            sub_lit = clauses_list[b+k]
                            v_sub = assignment[abs(sub_lit)]
                            if sub_lit < 0: v_sub = not v_sub
                            if v_sub: count_true += 1
                        
                        if count_true == 1:
                            breaks += 1
                
                if breaks < min_breaks:
                    min_breaks = breaks
                    best_lit = lit
                    if min_breaks == 0:
                        break
            
            chosen_lit = best_lit
            
        # Flip the chosen variable
        var_to_flip = abs(chosen_lit)
        old_val = assignment[var_to_flip]
        new_val = not old_val
        assignment[var_to_flip] = new_val
        
        # Update clause statuses
        # Clauses containing `var_to_flip`:
        # 1. Those satisfied by new value (lit_idx) -> remain satisfied or become satisfied.
        # 2. Those satisfied by old value (opp_idx) -> might break.
        
        # Polarity of the literal that is NOW TRUE (satisfied)
        # If new_val is True, positive literals `var` are satisfied.
        # If new_val is False, negative literals `-var` are satisfied.
        
        sat_polarity_idx = (var_to_flip << 1) | (0 if new_val else 1)
        unsat_polarity_idx = (var_to_flip << 1) | (1 if new_val else 0)
        
        # Process clauses that contain the newly True literal
        # These clauses are now definitely satisfied.
        for c in occur[sat_polarity_idx]:
            if not clause_status[c]:
                clause_status[c] = True
                unsat.append(c) # Will be handled lazily or we add to stack?
                # Actually if it was unsat, we just mark it sat.
                # The `unsat` list contains indices.
                # We rely on the lazy check at start of loop.
        
        # Process clauses that contained the newly False literal
        # These might lose their satisfying literal.
        for c in occur[unsat_polarity_idx]:
            if clause_status[c]:
                # It was satisfied. Check if it is still satisfied.
                # We need to re-evaluate.
                # If it was satisfied by more than one literal, it's fine.
                # Re-evaluating is safe.
                
                # Check if satisfied
                b = c * 3
                is_still_sat = False
                for k in range(3):
                    sub_lit = clauses_list[b+k]
                    v_sub = assignment[abs(sub_lit)]
                    if sub_lit < 0: v_sub = not v_sub
                    if v_sub:
                        is_still_sat = True
                        break
                
                if not is_still_sat:
                    clause_status[c] = False
                    # We don't add to unsat list yet?
                    # No, we should add it so it gets picked.
                    unsat.append(c)
                    
        # End of loop
    
    if not unsat:
        # Found solution
        # Output format: v1 v2 ... vN
        # Convert bools to 0/1
        out = []
        out.append("SAT")
        # Use string join for fast output
        vals = ["1" if assignment[i] else "0" for i in range(1, N+1)]
        out.append(" ".join(vals))
        sys.stdout.write("\n".join(out) + "\n")
    else:
        # Failed to find solution
        # Should not happen for "guaranteed satisfiable"
        # But if timeout or steps exceeded, print anything?
        # Output "SAT" is required?
        # Prompt says: "SAT\n v1..."
        # If not found, we output a wrong one? No.
        # We output a best-effort or just the current assignment.
        out = []
        out.append("SAT")
        vals = ["1" if assignment[i] else "0" for i in range(1, N+1)]
        out.append(" ".join(vals))
        sys.stdout.write("\n".join(out) + "\n")

if __name__ == '__main__':
    solve()
