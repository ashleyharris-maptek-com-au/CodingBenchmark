using System;
using System.Collections.Generic;
using System.Diagnostics;

public class Program
{
    // Configuration
    private const double TimeLimitSeconds = 280.0;
    private const double PlacementResolution = 0.5;
    private const int HeuristicInterval = 2;

    // Data structures
    private struct Vec3 { public double X, Y, Z; }
    private struct Quat { public double W, X, Y, Z; }
    private struct Face { public int A, B, C; }
    private struct OBB
    {
        public Vec3 Center;
        public Vec3 Extents; // Half-sizes
        public Quat Orientation;
    }

    // State
    private static Vec3[] _baseVertices;
    private static Face[] _baseFaces;
    private static Vec3 _containerSize;
    private static Random _rng = new Random(12345);
    private static Stopwatch _sw;
    private static double _voxelScale = 1.0;
    private static bool _smallScaleMode = false;

    public static void Main(string[] args)
    {
        _sw = Stopwatch.StartNew();
        
        try
        {
            ReadInput();

    private static void ReadInput()
    {
        // Fast buffered reading
        var buffer = new byte[65536];
        Console.OpenStandardInput().Read(buffer, 0, buffer.Length);
        
        var content = Encoding.ASCII.GetString(buffer);
        var tokens = content.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);

        int idx = 0;
        
        if (tokens.Length < 2) throw new Exception("Invalid input: missing V F");
        int vCount = ParseIntFast(tokens[idx++]);
        int fCount = ParseIntFast(tokens[idx++]);

        _baseVertices = new Vec3[vCount];
        _baseFaces = new Face[fCount];

        for (int i = 0; i < vCount; i++)
        {
            _baseVertices[i] = new Vec3 
            { 
                X = ParseDoubleFast(tokens[idx++]),
                Y = ParseDoubleFast(tokens[idx++]),
                Z = ParseDoubleFast(tokens[idx++])
            };
        }

        for (int i = 0; i < fCount; i++)
        {
            _baseFaces[i] = new Face
            {
                A = ParseIntFast(tokens[idx++]),
                B = ParseIntFast(tokens[idx++]),
                C = ParseIntFast(tokens[idx++])
            };
        }

        _containerSize = new Vec3
        {
            X = ParseDoubleFast(tokens[idx++]),
            Y = ParseDoubleFast(tokens[idx++]),
            Z = ParseDoubleFast(tokens[idx++])
        };

        // Determine strategy based on scale
        double maxDim = Math.Max(_containerSize.X, Math.Max(_containerSize.Y, _containerSize.Z));
        if (maxDim < 25) _smallScaleMode = true;
        
        // Voxel scale adapts to container size
        _voxelScale = Math.Max(1.0, Math.Floor(maxDim / 100.0));
        if (_voxelScale < 1) _voxelScale = 1.0;
    }

    private static void SolveAndOutput()
    {
        // 1. Pre-compute rotations
        // Use 24 orientations for small scale (quality), fewer for massive scale (speed)
        int rotationCount = _smallScaleMode ? 24 : 12;
        Quat[] rotations = GenerateRotations(rotationCount);

        // 2. Compute OBBs for rotations
        OBB[] obbs = new OBB[rotations.Length];
        double minDim = double.MaxValue;

        for (int i = 0; i < rotations.Length; i++)
        {
            obbs[i] = ComputeOBB(rotations[i]);
            minDim = Math.Min(minDim, Math.Min(obbs[i].Extents.X, Math.Min(obbs[i].Extents.Y, obbs[i].Extents.Z)));
        }

        // 3. Packing
        var packedTransforms = new List<(Vec3 T, Quat R)>();
        
        // Calculate grid dimensions based on voxel scale
        int gX = Math.Max(1, (int)(_containerSize.X / _voxelScale));
        int gY = Math.Max(1, (int)(_containerSize.Y / _voxelScale));
        int gZ = Math.Max(1, (int)(_containerSize.Z / _voxelScale));
        
        int gridSize = gX * gY * gZ;
        
        // For huge grids, use BitArray (memory efficient)
        BitArray grid = (gridSize > 100000000) ? new BitArray(gridSize) : new BitArray(gridSize);
        
        // Fill scan pattern
        int[] zOrder = Enumerable.Range(0, gZ).ToArray();
        int[] yOrder = Enumerable.Range(0, gY).ToArray();
        int[] xOrder = Enumerable.Range(0, gX).ToArray();

        // Max items estimation to prevent infinite loops
        double itemVolume = obbs[0].Extents.X * obbs[0].Extents.Y * obbs[0].Extents.Z * 8.0;
        double containerVolume = _containerSize.X * _containerSize.Y * _containerSize.Z;
        int maxItems = (int)(containerVolume / Math.Max(0.01, itemVolume) * 1.5) + 100;
        maxItems = Math.Min(maxItems, 1000000); // Hard cap

        int count = 0;
        bool timeOut = false;

        // Layer-based packing
        for (int z = 0; z < gZ && !timeOut; z++)
        {
            if (_sw.Elapsed.TotalSeconds > TimeLimitSeconds) break;

            for (int y = 0; y < gY && !timeOut; y++)
            {
                for (int x = 0; x < gX; x++)
                {
                    if (count >= maxItems) { timeOut = true; break; }

                    // Quick time check every ~1000 iterations
                    if ((x + y + z) % 500 == 0 && _sw.Elapsed.TotalSeconds > TimeLimitSeconds)
                    {
                        timeOut = true;
                        break;
                    }

                    int gridIdx = x + y * gX + z * gX * gY;
                    if (grid[gridIdx]) continue; // Spot taken

                    // Try placements
                    bool placed = false;
                    
                    // Optimization: Try best rotation first
                    int bestRot = 0; // Default identity
                    for (int r = 0; r < rotations.Length; r++)
                    {
                        var obb = obbs[r];
                        
                        // Calculate required cell size for this rotation
                        int rx = (int)Math.Ceiling(obb.Extents.X * 2.0 / _voxelScale);
                        int ry = (int)Math.Ceiling(obb.Extents.Y * 2.0 / _voxelScale);
                        int rz = (int)Math.Ceiling(obb.Extents.Z * 2.0 / _voxelScale);

                        // Check bounds
                        if (x + rx > gX || y + ry > gY || z + rz > gZ) continue;

                        // Check collision
                        bool collision = false;
                        for (int dx = 0; dx < rx && !collision; dx++)
                        {
                            for (int dy = 0; dy < ry && !collision; dy++)
                            {
                                for (int dz = 0; dz < rz; dz++)
                                {
                                    int checkIdx = (x + dx) + (y + dy) * gX + (z + dz) * gX * gY;
                                    if (grid[checkIdx]) { collision = true; break; }
                                }
                            }
                        }

                        if (!collision)
                        {
                            // Mark grid
                            for (int dx = 0; dx < rx; dx++)
                            {
                                for (int dy = 0; dy < ry; dy++)
                                {
                                    for (int dz = 0; dz < rz; dz++)
                                    {
                                        int fillIdx = (x + dx) + (y + dy) * gX + (z + dz) * gX * gY;
                                        grid[fillIdx] = true;
                                    }
                                }
                            }

                            // Calculate world position
                            double wx = x * _voxelScale + obb.Extents.X;
                            double wy = y * _voxelScale + obb.Extents.Y;
                            double wz = z * _voxelScale + obb.Extents.Z;

                            // Verify AABB bounds strictly
                            if (wx + obb.Extents.X <= _containerSize.X + 0.001 &&
                                wy + obb.Extents.Y <= _containerSize.Y + 0.001 &&
                                wz + obb.Extents.Z <= _containerSize.Z + 0.001)
                            {
                                packedTransforms.Add((new Vec3 { X = wx, Y = wy, Z = wz }, rotations[r]));
                                count++;
                                placed = true;
                                break;
                            }
                            else
                            {
                                // Out of bounds, unmark grid
                                for (int dx = 0; dx < rx; dx++)
                                {
                                    for (int dy = 0; dy < ry; dy++)
                                    {
                                        for (int dz = 0; dz < rz; dz++)
                                        {
                                            int fillIdx = (x + dx) + (y + dy) * gX + (z + dz) * gX * gY;
                                            grid[fillIdx] = false;
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }

        // Output
        Console.WriteLine(count);
        var sb = new StringBuilder();
        for (int i = 0; i < count; i++)
        {
            var t = packedTransforms[i].T;
            var r = packedTransforms[i].R;
            sb.AppendLine($"{t.X:F6} {t.Y:F6} {t.Z:F6} {r.W:F6} {r.X:F6} {r.Y:F6} {r.Z:F6}");
        }
        Console.Write(sb.ToString());
    }

    private static Quat[] GenerateRotations(int count)
    {
        var list = new List<Quat>();
        list.Add(new Quat { W = 1, X = 0, Y = 0, Z = 0 }); // Identity

        // Generate unique rotations by rotating face normals to align with axes
        // This creates the 24 orientation classes of a cube
        double sqrt2 = Math.Sqrt(2.0) / 2.0;
        double sqrt3_2 = Math.Sqrt(3.0) / 2.0; // Not used for 90deg rotations

        // Basic 90-degree rotations
        var rots = new List<Quat>();
        
        // Rotations around Z
        rots.Add(new Quat { W = Math.Cos(Math.PI / 2), X = 0, Y = 0, Z = Math.Sin(Math.PI / 2) / 1 }); // 90
        rots.Add(new Quat { W = Math.Cos(Math.PI), X = 0, Y = 0, Z = Math.Sin(Math.PI) / 1 }); // 180
        rots.Add(new Quat { W = Math.Cos(3 * Math.PI / 2), X = 0, Y = 0, Z = Math.Sin(3 * Math.PI / 2) / 1 }); // 270

        // Rotations around X
        rots.Add(new Quat { W = Math.Cos(Math.PI / 2), X = Math.Sin(Math.PI / 2), Y = 0, Z = 0 });
        rots.Add(new Quat { W = Math.Cos(Math.PI), X = Math.Sin(Math.PI), Y = 0, Z = 0 });
        rots.Add(new Quat { W = Math.Cos(3 * Math.PI / 2), X = Math.Sin(3 * Math.PI / 2), Y = 0, Z = 0 });

        // Rotations around Y
        rots.Add(new Quat { W = Math.Cos(Math.PI / 2), X = 0, Y = Math.Sin(Math.PI / 2), Z = 0 });
        rots.Add(new Quat { W = Math.Cos(Math.PI), X = 0, Y = Math.Sin(Math.PI), Z = 0 });
        rots.Add(new Quat { W = Math.Cos(3 * Math.PI / 2), X = 0, Y = Math.Sin(3 * Math.PI / 2), Z = 0 });

        // Compound rotations (face up)
        // 90 Y then 90 X
        rots.Add(new Quat { W = 0.5, X = 0.5, Y = 0.5, Z = 0.5 });
        rots.Add(new Quat { W = 0.5, X = -0.5, Y = 0.5, Z = -0.5 });
        rots.Add(new Quat { W = 0.5, X = 0.5, Y = -0.5, Z = -0.5 });
        rots.Add(new Quat { W = 0.5, X = -0.5, Y = -0.5, Z = 0.5 });

        // Generate full set by combining
        var generated = new HashSet<string>();
        foreach (var r in rots)
        {
            if (generated.Add(NormalizeQuat(r)))
                list.Add(r);
        }

        // Ensure we have at least 'count' rotations
        // For the solver, we just return what we have (simplified for robustness)
        return list.ToArray();
    }

    private static string NormalizeQuat(Quat q)
    {
        double mag = Math.Sqrt(q.W * q.W + q.X * q.X + q.Y * q.Y + q.Z * q.Z);
        if (mag < 1e-10) return "0";
        return $"{q.W / mag:F5}_{q.X / mag:F5}_{q.Y / mag:F5}_{q.Z / mag:F5}";
    }

    private static OBB ComputeOBB(Quat rotation)
    {
        Vec3 min = new Vec3 { X = double.MaxValue, Y = double.MaxValue, Z = double.MaxValue };
        Vec3 max = new Vec3 { X = double.MinValue, Y = double.MinValue, Z = double.MinValue };

        for (int i = 0; i < _baseVertices.Length; i++)
        {
            Vec3 v = RotatePoint(_baseVertices[i], rotation);
            min.X = Math.Min(min.X, v.X);
            min.Y = Math.Min(min.Y, v.Y);
            min.Z = Math.Min(min.Z, v.Z);
            max.X = Math.Max(max.X, v.X);
            max.Y = Math.Max(max.Y, v.Y);
            max.Z = Math.Max(max.Z, v.Z);
        }

        return new OBB
        {
            Center = new Vec3 { X = (min.X + max.X) / 2, Y = (min.Y + max.Y) / 2, Z = (min.Z + max.Z) / 2 },
            Extents = new Vec3 { X = (max.X - min.X) / 2, Y = (max.Y - min.Y) / 2, Z = (max.Z - min.Z) / 2 },
            Orientation = rotation
        };
    }

    [System.Runtime.CompilerServices.MethodImpl(System.Runtime.CompilerServices.MethodImplOptions.AggressiveInlining)]
    private static Vec3 RotatePoint(Vec3 p, Quat q)
    {
        // q * p * q^-1
        // Optimized formula
        double dx = 2 * (q.X * p.Z - q.Z * p.Y);
        double dy = 2 * (q.Z * p.X - q.Z * p.X); // Fixed formula below
        double dz = 2 * (q.Y * p.X - q.X * p.Y);

        // Correct quaternion rotation formula derivation
        // t = 2 * cross(q.xyz, v)
        // v' = v + q.w * t + cross(q.xyz, t)
        
        double tx = 2 * (q.Y * p.Z - q.Z * p.Y);
        double ty = 2 * (q.Z * p.X - q.X * p.Z);
        double tz = 2 * (q.X * p.Y - q.Y * p.X);

        return new Vec3
        {
            X = p.X + q.W * tx + (q.Y * tz - q.Z * ty),
            Y = p.Y + q.W * ty + (q.Z * tx - q.X * tz),
            Z = p.Z + q.W * tz + (q.X * ty - q.Y * tx)
        };
    }

    private static int ParseIntFast(string s)
    {
        int val = 0;
        int sign = 1;
        int i = 0;
        if (s[0] == '-') { sign = -1; i++; }
        while (i < s.Length && s[i] >= '0' && s[i] <= '9')
        {
            val = val * 10 + (s[i] - '0');
            i++;
        }
        return val * sign;
    }

    private static double ParseDoubleFast(string s)
    {
        return double.Parse(s, System.Globalization.CultureInfo.InvariantCulture);
    }
