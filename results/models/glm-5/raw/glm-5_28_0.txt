```rust
use std::io::{self, BufRead, BufWriter, Write};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    let mut lines = stdin.lock().lines();
    
    // Parse header
    let header = lines.next().expect("Missing header").expect("IO error");
    let mut iter = header.split_whitespace();
    let n: usize = iter.next().expect("Missing n").parse().expect("Invalid n");
    let m: usize = iter.next().expect("Missing m").parse().expect("Invalid m");
    
    // Adjacency lists
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    
    // Parse edges
    for _ in 0..m {
        let line = lines.next().expect("Missing edge line").expect("IO error");
        let mut edge_iter = line.split_whitespace();
        let u: usize = edge_iter.next().expect("Missing u").parse().expect("Invalid u");
        let v: usize = edge_iter.next().expect("Missing v").parse().expect("Invalid v");
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }
    
    // Clean up adjacency lists (remove duplicates, shrink)
    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
        neighbors.shrink_to_fit();
    }
    
    // Compute degrees
    let mut degree: Vec<i32> = adj.iter().map(|v| v.len() as i32).collect();
    let mut removed = vec![false; n];
    let mut in_cover = vec![false; n];
    
    // Use usize::MAX as sentinel for removed nodes
    const REMOVED: usize = usize::MAX;
    
    // Bin sort structures
    let max_deg: usize = degree.iter().map(|&d| d as usize).max().unwrap_or(0);
    
    if n == 0 {
        writeln!(out, "0").unwrap();
        writeln!(out, "").unwrap();
        return;
    }
    
    // Use sentinel value to avoid Option checks
    let mut bin_head: Vec<usize> = vec![REMOVED; max_deg + 1];
    let mut next_in_bin: Vec<usize> = vec![REMOVED; n];
    let mut prev_in_bin: Vec<usize> = vec![REMOVED; n];
    
    // Initialize bins
    for u in 0..n {
        let d = degree[u] as usize;
        if d > 0 {
            next_in_bin[u] = bin_head[d];
            if bin_head[d] != REMOVED {
                prev_in_bin[bin_head[d]] = u;
            }
            bin_head[d] = u;
        }
    }
    
    // Track maximum degree
    let mut current_max_deg = max_deg;
    let mut cover_count = 0usize;
    
    // Helper function to remove vertex from bin
    let remove_from_bin = |u: usize, deg: usize, bin_head: &mut [usize], next_in_bin: &mut [usize], prev_in_bin: &mut [usize]| {
        let prev = prev_in_bin[u];
        let nxt = next_in_bin[u];
        
        if prev != REMOVED {
            next_in_bin[prev] = nxt;
        } else {
            // u was head of bin
            bin_head[deg] = nxt;
        }
        
        if nxt != REMOVED {
            prev_in_bin[nxt] = prev;
        }
    };
    
    // Main reduction loop
    while current_max_deg > 0 {
        // Find highest degree vertex
        while current_max_deg > 0 && bin_head[current_max_deg] == REMOVED {
            current_max_deg -= 1;
        }
        
        if current_max_deg == 0 {
            break;
        }
        
        let u = bin_head[current_max_deg];
        
        // Degree 1: take neighbor instead
        if current_max_deg == 1 {
            let v = adj[u].iter().find(|&&w| !removed[w]).copied().unwrap_or(u);
            in_cover[v] = true;
            cover_count += 1;
            
            // Remove v and all its neighbors from graph
            if !removed[v] {
                removed[v] = true;
                let v_deg = degree[v] as usize;
                remove_from_bin(v, v_deg, &mut bin_head, &mut next_in_bin, &mut prev_in_bin);
                
                // Decrease degree of v's neighbors
                for &w in &adj[v] {
                    if !removed[w] {
                        let old_deg = degree[w] as usize;
                        remove_from_bin(w, old_deg, &mut bin_head, &mut next_in_bin, &mut prev_in_bin);
                        
                        degree[w] -= 1;
                        let new_deg = degree[w] as usize;
                        if new_deg > 0 {
                            next_in_bin[w] = bin_head[new_deg];
                            if bin_head[new_deg] != REMOVED {
                                prev_in_bin[bin_head[new_deg]] = w;
                            }
                            bin_head[new_deg] = w;
                        }
                        removed[w] = degree[w] == 0;
                        if removed[w] && !in_cover[w] {
                            // Check if isolated vertex should be added
                            // Isolated vertices don't need to be in cover
                        }
                    }
                }
            }
            
            // Remove u
            if !removed[u] {
                removed[u] = true;
                remove_from_bin(u, 1, &mut bin_head, &mut next_in_bin, &mut prev_in_bin);
            }
        } else {
            // Higher degree: add to cover
            in_cover[u] = true;
            cover_count += 1;
            
            // Remove u from graph
            removed[u] = true;
            remove_from_bin(u, current_max_deg, &mut bin_head, &mut next_in_bin, &mut prev_in_bin);
            
            // Decrease degree of neighbors
            for &v in &adj[u] {
                if !removed[v] {
                    let old_deg = degree[v] as usize;
                    remove_from_bin(v, old_deg, &mut bin_head, &mut next_in_bin, &mut prev_in_bin);
                    
                    degree[v] -= 1;
                    let new_deg = degree[v] as usize;
                    if new_deg > 0 {
                        next_in_bin[v] = bin_head[new_deg];
                        if bin_head[new_deg] != REMOVED {
                            prev_in_bin[bin_head[new_deg]] = v;
                        }
                        bin_head[new_deg] = v;
                    }
                    removed[v] = degree[v] == 0;
                }
            }
        }
    }
    
    // Output result
    writeln!(out, "{}", cover_count).unwrap();
    let mut first = true;
    for u in 0..n {
        if in_cover[u] {
            if !first {
                write!(out, " ").unwrap();
            }
            write!(out, "{}", u).unwrap();
            first = false;
        }
    }
    writeln!(out, "").unwrap();
}
```
