; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%v3input = OpTypePointer Input %vec3
%in_worldPos = OpVariable %v3input Input
%in_normal = OpVariable %v3input Input

%v4output = OpTypePointer Output %vec4
%outColor = OpVariable %v4output Output

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ubo_ptr = OpTypePointer Uniform %UBO
%ubo_var = OpVariable %ubo_ptr Uniform

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_1_5 = OpConstant %float 1.5
%float_3_0 = OpConstant %float 3.0

%base_r = OpConstant %float 0.8
%base_g = OpConstant %float 0.2
%base_b = OpConstant %float 0.2

%fog_r = OpConstant %float 0.7
%fog_g = OpConstant %float 0.7
%fog_b = OpConstant %float 0.8

%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%pos = OpLoad %vec3 %in_worldPos
%nrm = OpLoad %vec3 %in_normal

; Load UBO lightPos (member 3, offset 192)
%ubo_light_ptr = OpTypePointer Uniform %vec4
%lightPos_ptr = OpAccessChain %ubo_light_ptr %ubo_var %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

; Normalize normal
%normal_norm = OpExtInst %vec3 %glsl Normalize %nrm

; Light vector and distance
%light_vec = OpExtInst %vec3 %glsl FSub %lightPos_xyz %pos
%light_dir = OpExtInst %vec3 %glsl Normalize %light_vec

; Lambertian diffuse
%diff = OpDot %float %normal_norm %light_dir
%diff_max = OpExtInst %float %glsl FMax %diff %float_0

; Calculate lighting factor: 0.3 + 0.7 * diffuse
%diff_scaled = OpExtInst %float %glsl FMul %float_0_7 %diff_max
%light_factor = OpExtInst %float %glsl FAdd %float_0_3 %diff_scaled

; Apply lighting to base color
%base_color = OpCompositeConstruct %vec3 %base_r %base_g %base_b
%lit_color = OpVectorTimesScalar %vec3 %base_color %light_factor

; Calculate fog distance
%dist = OpExtInst %float %glsl Length %light_vec

; Fog factor: clamp((dist - 1.5) / 3.0, 0, 1)
%dist_minus = OpExtInst %float %glsl FSub %dist %float_1_5
%fog_raw = OpExtInst %float %glsl FDiv %dist_minus %float_3_0
%fog_factor = OpExtInst %float %glsl FClamp %fog_raw %float_0 %float_1

; Mix lit color with fog color
%fog_color = OpCompositeConstruct %vec3 %fog_r %fog_g %fog_b
%fog_inv = OpExtInst %float %glsl FSub %float_1 %fog_factor
%fog_inv_vec = OpCompositeConstruct %vec3 %fog_inv %fog_inv %fog_inv
%fog_factor_vec = OpCompositeConstruct %vec3 %fog_factor %fog_factor %fog_factor

%lit_mul = OpExtInst %vec3 %glsl FMul %lit_color %fog_inv_vec
%fog_mul = OpExtInst %vec3 %glsl FMul %fog_color %fog_factor_vec
%final_color = OpExtInst %vec3 %glsl FAdd %lit_mul %fog_mul

; Output with alpha 1.0
%final_vec4 = OpCompositeConstruct %vec4 %final_color %float_1
OpStore %outColor %final_vec4

OpReturn
OpFunctionEnd

%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3
%glsl = OpExtInstImport "GLSL.std.450"
