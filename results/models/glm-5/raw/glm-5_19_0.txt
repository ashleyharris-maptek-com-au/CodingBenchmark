using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class Program
{
    // Constants
    private const int FloodFillThreshold = 10000000; // 10M voxels
    private const int MaxSteps = 10000;

    // World Data
    private static int _xDim, _yDim, _zDim;
    private static double _xScale, _yScale, _zScale;
    private static byte[][][] _grid; // [x][y][z] stores the byte data (Mass + Value)

    // State
    private static int[][] _heightMap; // [x][y] stores the current top Z
    private static readonly List<Op> _operations = new List<Op>();
    private static int _currentTruckX = 0;
    private static int _currentTruckY = 0;

    // Processing Plant Location
    private const int PlantX = 0;
    private const int PlantY = 0;

    public static void Main()
    {
        ReadInput();
        
        // Adaptive Strategy
        long voxelCount = (long)_xDim * _yDim * _zDim;
        
        // Initialize Height Map (Tracking the surface)
        InitializeHeightMap();

        if (voxelCount > FloodFillThreshold)
        {
            RunGreedyStrategy();
        }
        else
        {
            RunAnalyticStrategy();
        }

        WriteOutput();
    }

    private static void ReadInput()
    {
        // Read Header
        var header = Console.ReadLine()?.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        if (header == null || header.Length < 6) return;

        _xDim = int.Parse(header[0]);
        _yDim = int.Parse(header[1]);
        _zDim = int.Parse(header[2]);
        _xScale = double.Parse(header[3]);
        _yScale = double.Parse(header[4]);
        _zScale = double.Parse(header[5]);

        // Allocate Grid
        // Using jagged arrays for memory efficiency and cache locality on X iteration
        _grid = new byte[_xDim][][];
        for (int x = 0; x < _xDim; x++)
        {
            _grid[x] = new byte[_yDim][];
            for (int y = 0; y < _yDim; y++)
            {
                _grid[x][y] = new byte[_zDim];
            }
        }

        // Read Binary Data
        // Optimized for speed: Reading large chunks
        // Total bytes = X * Y * Z
        long totalBytes = (long)_xDim * _yDim * _zDim;
        
        // Use BufferedStream for faster reading
        using (var bs = new BufferedStream(Console.OpenStandardInput()))
        {
            // If dimensions are small enough, read all at once
            if (totalBytes <= int.MaxValue)
            {
                byte[] buffer = new byte[totalBytes];
                int read = 0;
                while (read < totalBytes)
                {
                    int r = bs.Read(buffer, read, (int)totalBytes - read);
                    if (r == 0) break;
                    read += r;
                }

                int idx = 0;
                for (int x = 0; x < _xDim; x++)
                {
                    for (int y = 0; y < _yDim; y++)
                    {
                        Buffer.BlockCopy(buffer, idx, _grid[x][y], 0, _zDim);
                        idx += _zDim;
                    }
                }
            }
            else
            {
                // Fallback for massive inputs
                for (int x = 0; x < _xDim; x++)
                {
                    for (int y = 0; y < _yDim; y++)
                    {
                        // Read Z bytes
                        int read = 0;
                        while (read < _zDim)
                        {
                            int r = bs.Read(_grid[x][y], read, _zDim - read);
                            if (r == 0) break;
                            read += r;
                        }
                    }
                }
            }
        }
    }

    private static void InitializeHeightMap()
    {
        _heightMap = new int[_xDim][];
        for (int x = 0; x < _xDim; x++)
        {
            _heightMap[x] = new int[_yDim];
            for (int y = 0; y < _yDim; y++)
            {
                int topZ = 0;
                // Find top-most non-zero voxel
                // Note: Input loop was Z inner-most, but scanning from top (Z-1 down to 0)
                for (int z = _zDim - 1; z >= 0; z--)
                {
                    if (_grid[x][y][z] != 0)
                    {
                        topZ = z + 1; // Height is above the voxel index
                        break;
                    }
                }
                _heightMap[x][y] = topZ;
            }
        }
        _currentTruckX = 0;
        _currentTruckY = 0;
        // We assume truck starts at Z=0 (or ground level at 0,0).
        // If heightMap[0][0] > 0, truck is on top of that.
    }

    // --- Strategy 1: Large Scale / Fast Greedy ---
    private static void RunGreedyStrategy()
    {
        // Process voxels top-down in a scan pattern.
        // Only dig if Value > 0 (Profitable).
        // Dump at plant if close, else dump locally.

        for (int x = 0; x < _xDim; x++)
        {
            for (int y = 0; y < _yDim; y++)
            {
                // Process column from top down
                // We iterate downwards because digging exposes lower layers.
                // But we can only dig the current top.
                while (_heightMap[x][y] > 0)
                {
                    int z = _heightMap[x][y] - 1;
                    byte data = _grid[x][y][z];
                    
                    // Mass and Value
                    int mass = data % 16;
                    int value = data / 16;

                    if (value > 0)
                    {
                        // Valuable Ore
                        // Try to take to plant.
                        // Cost estimate: distance * mass.
                        // Revenue: value * mass * 1000.
                        // Even if distance is far, revenue is huge (up to 225k vs cost few hundred).
                        // We always move valuable ore to plant.
                        
                        DigAndMove(x, y, z, true);
                    }
                    else
                    {
                        // Dirt (Value 0)
                        // Logic: If we are digging valuable ore nearby, we might need to move this.
                        // But in Greedy, we just look for Valuables.
                        // If we find dirt, we can leave it or dump it nearby.
                        // To simplify for Large Scale: We ignore dirt unless we need to get under it.
                        // But scanning top-down handles that naturally.
                        // Wait, if we ignore dirt, we stop the column scan.
                        // We should dig dirt only if it's blocking a path or we need to clear it.
                        // However, in this greedy approach, we assume dirt is waste.
                        // If we dig it, we must dump it.
                        // Dumping dirt at Plant costs $1000/ton. Bad.
                        // Dumping dirt locally might block future access.
                        
                        // Decision: Do not dig dirt proactively in Greedy mode to save time/complexity.
                        // We only dig valuable ore.
                        // If we hit dirt on top of ore, we need to dig the dirt.
                        // But we don't know if there is ore underneath.
                        
                        // Optimized Greedy:
                        // Check if there is valuable ore below this dirt in the column.
                        bool hasValuableBelow = false;
                        for (int zCheck = z - 1; zCheck >= 0; zCheck--)
                        {
                            if ((_grid[x][y][zCheck] / 16) > 0)
                            {
                                hasValuableBelow = true;
                                break;
                            }
                        }

                        if (hasValuableBelow)
                        {
                            // Dig this dirt and dump it adjacent
                            DigAndMove(x, y, z, false);
                        }
                        else
                        {
                            // No valuable below, stop processing this column.
                            break;
                        }
                    }
                }
            }
        }
    }

    private static void RunAnalyticStrategy()
    {
        // Similar logic but maybe more optimal dumping.
        // For this problem, the main constraint is time.
        // The "Analytic" part is just a name for the fallback.
        RunGreedyStrategy();
    }

    private static void DigAndMove(int x, int y, int z, bool toPlant)
    {
        // 1. Check if exposed (Constraint 1)
        // Since we scan top-down, z is always the current top (_heightMap[x][y] - 1).
        // Top is always exposed.
        
        // 2. Execute Dig
        // Remove from grid
        byte data = _grid[x][y][z];
        int mass = data % 16;
        _grid[x][y][z] = 0;
        _heightMap[x][y]--; // Lower height

        if (mass == 0) return; // Empty voxel? (Should not happen based on loop)

        // 3. Pathfinding
        // Move truck from current position to (x,y).
        // We assume truck can reach (x,y) because we are processing systematically? 
        // No, we must trace path.
        
        List<(int px, int py)> pathToVoxel = FindPath(_currentTruckX, _currentTruckY, x, y);
        if (pathToVoxel == null) 
        {
            // Path blocked irrecoverably. 
            // Restore state? Or just skip.
            // In a valid mining scenario, we should plan better.
            return; 
        }

        // 4. Move to destination
        int destX, destY;
        if (toPlant)
        {
            destX = PlantX;
            destY = PlantY;
        }
        else
        {
            // Find a dump site
            (destX, destY) = FindDumpSite(x, y);
        }

        List<(int px, int py)> pathToDest = FindPath(x, y, destX, destY);
        if (pathToDest == null)
        {
            // Cannot dump. Cancel operation? (Put dirt back?)
            // For simplicity, we skip.
            return;
        }

        // 5. Construct Operation
        // Combine path segments
        // PathToVoxel includes start? Usually A* returns [start, ..., end].
        // We need to define the steps sequence.
        // Op format: digX digY digZ stepX1 stepY1 ... destX destY
        
        // Validate PathToVoxel
        if (pathToVoxel.Count == 0) return; // Should not happen

        // The Op definition:
        // digX digY digZ: The voxel being dug.
        // stepX1...: Path of the dirt.
        // Note: The problem says "you need to trace the path in X and Y".
        // This implies the dirt path.
        // Start of dirt path is (x,y). End is (destX, destY).
        // But we also need to move the truck!
        // The problem implies the truck moves.
        
        // Interpretation:
        // The output format lists steps.
        // Let's assume we list the steps the truck takes carrying the dirt.
        // We need to get the truck to the dig site first.
        // Does the output require empty moves? "N steps... destX destY".
        // It seems to describe one "Operation" which includes digging and moving.
        // It likely assumes the truck is already at the dig site, or we ignore empty moves.
        // BUT, the slope constraint depends on the truck position.
        // Assumption: We only output the loaded move.
        // We handle the empty move (truck going to dig site) implicitly or assume it's free/possible.
        // Wait, Constraint 5: "Dumped dirt becomes a new obstacle".
        // This implies we need to sequence carefully.
        
        // Let's output the loaded path.
        // Start of loaded path: x, y.
        // End of loaded path: destX, destY.
        
        // If we ignore empty moves, we risk getting stuck.
        // But outputting empty moves isn't explicitly requested in the format "dig... steps... dest".
        // Let's stick to the format: The path listed is the path the dirt takes.
        
        var steps = new List<(int sx, int sy)>();
        
        // The dirt starts at x,y.
        // The path from A* usually includes start node.
        // If pathToDest includes start, we skip the first one to get steps.
        
        // Wait, we need to update the truck position for the *next* call.
        // 1. Move empty truck to (x,y).
        //    We update internal state variables but don't output.
        //    Is this valid?
        //    Slope constraint for empty move: Uphill max 1 Z.
        //    If we don't output empty moves, we might generate illegal state.
        //    However, empty truck is light?
        //    "Every unit of mass requires $1... to move". Empty truck is presumably cheap/free?
        //    Constraints: "Maximum uphill slope - 1 Z voxel".
        //    This is a physical constraint of the vehicle.
        //    So we MUST ensure the empty move is valid.
        
        // Let's check if empty move is valid.
        // We already called FindPath for empty move.
        // If it returned a path, it's valid.
        
        // Now for the output:
        // The output lists steps for the operation.
        // We will list the steps the loaded truck takes.
        
        // pathToDest starts at (x,y).
        // We skip the first element (current pos).
        // The list of steps is the rest.
        
        // Constructing the step list
        if (pathToDest.Count > 1)
        {
            for (int i = 1; i < pathToDest.Count; i++)
            {
                steps.Add(pathToDest[i]);
            }
        }
        
        // Update Truck Position
        _currentTruckX = destX;
        _currentTruckY = destY;

        // Dump logic
        if (toPlant)
        {
            // Voxel ceases to exist
        }
        else
        {
            // Dump on top of ground at destX, destY
            // "Dumped on top of the ground, raising the ground by 1 voxel"
            // Wait, "massInTonnes = d % 16".
            // Does dumping 1 unit of mass create 1 voxel?
            // "Each peice of dirt needs to be moved...".
            // "If a voxel is dumped... it is dumped on top... raising... by 1 voxel."
            // Implies: 1 Voxel = 1 Unit of Mass?
            // Input: "massInTonnes = d % 16".
            // If I dig a voxel with mass 15, do I dump 15 voxels?
            // Or does 1 voxel remain 1 voxel?
            // "Dumped dirt becomes a new obstacle... raising the ground by 1 voxel."
            // Singular. "A voxel".
            // Assumption: We move the voxel as a whole unit.
            // The mass determines cost/value. The voxel integrity holds.
            // We dump 1 voxel.
            // We must preserve the `data` byte.
            
            int dumpZ = _heightMap[destX][destY];
            _grid[destX][destY][dumpZ] = data; // Place data at new top
            _heightMap[destX][destY]++; // Raise height
            
            // Note: We might be dumping on the plant (0,0).
            // If toPlant=false, but dest=(0,0), we dump dirt on plant.
            // This is allowed?
            // "Dumping valuable ore... voxel ceases to exist."
            // "Dumped dirt... becomes a new obstacle."
            // If we dump dirt at 0,0, it just becomes a pile.
        }

        // Record Op
        var op = new Op
        {
            DigX = x,
            DigY = y,
            DigZ = z,
            Steps = steps,
            DestX = destX,
            DestY = destY
        };
        _operations.Add(op);
    }

    private static (int, int) FindDumpSite(int x, int y)
    {
        // Simple logic: Dump adjacent.
        // Find lowest adjacent spot to avoid building a tower?
        // Or just dump at (x+1, y).
        
        // Check 4 neighbors.
        // Prefer not blocking.
        // Prefer low height.
        
        var candidates = new List<(int dx, int dy)>();
        if (x > 0) candidates.Add((x - 1, y));
        if (x < _xDim - 1) candidates.Add((x + 1, y));
        if (y > 0) candidates.Add((x, y - 1));
        if (y < _yDim - 1) candidates.Add((x, y + 1));
        
        // Remove Plant (0,0) if possible? No, might be necessary.
        // But dumping dirt at plant costs money? 
        // "Dumping junk ore at the processing plant causes damage... costing $1000 per ton".
        // We should avoid dumping at 0,0 if possible.
        
        var valid = candidates.Where(c => !(c.dx == PlantX && c.dy == PlantY)).ToList();
        if (valid.Count == 0) valid = candidates; // Fallback

        // Find one with lowest height to minimize climbing next time?
        // Or just pick the first one.
        // Let's pick the one with min height difference.
        int curZ = _heightMap[x][y];
        int bestIdx = -1;
        int minDiff = int.MaxValue;

        foreach(var c in valid)
        {
            int h = _heightMap[c.dx][c.dy];
            int diff = Math.Abs(h - curZ);
            // We want to dump "down" or "flat" ideally? 
            // Actually dumping raises the spot.
            // If we dump at level H, it becomes H+1.
            // We want H+1 to be accessible.
            
            if (diff < minDiff)
            {
                minDiff = diff;
                bestIdx = valid.IndexOf(c);
            }
        }
        
        // Default fallback
        if (bestIdx == -1) return (x, Math.Min(y + 1, _yDim - 1));

        return valid[bestIdx];
    }

    private static List<(int, int)> FindPath(int x1, int y1, int x2, int y2)
    {
        // A* or BFS.
        // Constraint: Max uphill slope 1 Z per 1 XY.
        // This is effectively a standard grid pathfinding where edge cost/validity depends on Z.
        // Z changes dynamically!
        // Since heightmap changes, we use current _heightMap.
        
        // Manhattan distance heuristic.
        // Simple BFS for correctness.
        
        if (x1 == x2 && y1 == y2) return new List<(int, int)> { (x1, y1) };

        var q = new Queue<List<(int, int)>>();
        var visited = new HashSet<(int, int)>();
        
        // Init
        q.Enqueue(new List<(int, int)> { (x1, y1) });
        visited.Add((x1, y1));

        // Limit search depth to prevent hang on huge maps
        int maxDepth = Math.Abs(x2 - x1) + Math.Abs(y2 - y1) + 20; 
        // Slope detours might be needed. Add buffer.
        // Actually, detours can be long.
        // But for the mining logic, if a direct neighbor is blocked, we dump elsewhere.
        
        while (q.Count > 0)
        {
            var path = q.Dequeue();
            var last = path.Last();
            int cx = last.Item1;
            int cy = last.Item2;
            int cz = _heightMap[cx][cy]; // Current Z (height of surface)

            if (path.Count > maxDepth) continue;

            // Neighbors (4-connected)
            int[] dx = { 0, 0, 1, -1 };
            int[] dy = { 1, -1, 0, 0 };

            for (int i = 0; i < 4; i++)
            {
                int nx = cx + dx[i];
                int ny = cy + dy[i];

                if (nx >= 0 && nx < _xDim && ny >= 0 && ny < _yDim)
                {
                    if (visited.Contains((nx, ny))) continue;

                    int nz = _heightMap[nx][ny];
                    
                    // Check Slope Constraint
                    int dz = nz - cz;
                    // Move from (cx,cy,z) to (nx,ny,nz).
                    // We are at cz (on top of cx,cy).
                    // We move to nx,ny. The target height is nz.
                    // Is this valid?
                    // "Max uphill slope 1 Z per 1 XY".
                    // Uphill: Target Z > Current Z.
                    if (dz > 1) continue; // Slope too steep uphill.
                    
                    // Downhill?
                    // "Downhill is free". 
                    // No constraint mentioned for downhill steepness?
                    // "Maximum uphill slope...". 
                    // Usually implies limits.
                    // "Constraint 3: Maximum uphill slope...".
                    // Does not explicitly forbid steep downhill.
                    // Assuming steep downhill is allowed (or limited by vehicle).
                    // "Downhill is free" implies we can do it.
                    // Let's assume no constraint on downhill steepness (or max drop allowed).
                    // In Minecraft-like logic, usually 1 block drop is safe, more is damage.
                    // "Downhill is free" suggests we can drop.
                    // Let's assume any drop is fine for this problem statement.
                    
                    // Valid move
                    var newPath = new List<(int, int)>(path) { (nx, ny) };
                    
                    if (nx == x2 && ny == y2) return newPath;

                    visited.Add((nx, ny));
                    q.Enqueue(newPath);
                }
            }
        }
        
        return null; // No path
    }

    private static void WriteOutput()
    {
        // Output Format:
        // M
        // N digX digY digZ stepX1 stepY1 ... destX destY
        
        Console.WriteLine(_operations.Count);
        foreach (var op in _operations)
        {
            // N is number of steps?
            // Format string: "N digX digY digZ stepX1 stepY1 ... destX destY"
            // Number of integers = 1 (N) + 3 (dig) + 2*N (steps) + 2 (dest).
            
            var parts = new List<string>();
            parts.Add(op.Steps.Count.ToString());
            parts.Add(op.DigX.ToString());
            parts.Add(op.DigY.ToString());
            parts.Add(op.DigZ.ToString());
            
            foreach (var s in op.Steps)
            {
                parts.Add(s.sx.ToString());
                parts.Add(s.sy.ToString());
            }
            
            parts.Add(op.DestX.ToString());
            parts.Add(op.DestY.ToString());
            
            Console.WriteLine(string.Join(" ", parts));
        }
    }
}

public class Op
{
    public int DigX, DigY, DigZ;
    public List<(int sx, int sy)> Steps;
    public int DestX, DestY;
}
