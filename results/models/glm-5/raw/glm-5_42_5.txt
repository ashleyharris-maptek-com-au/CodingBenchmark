```spirv
; SPIR-V Assembly for Geometry Shader: Line to Arrow (Thick Shaft + Head)
; Vulkan 1.0

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%float_1 = OpConstant %float 1

; Input Interface Block (gl_in)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined Inputs
%arr_v3 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

; Output Interface Block (gl_out)
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User-defined Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointer types for accessing data
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

; --- Load Inputs ---
; p0 = gl_in[0].Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

; p1 = gl_in[1].Position
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; c0 = inColor[0]
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr

; c1 = inColor[1]
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr

; --- Geometry Calculations ---
; vec3 p0_3 = p0.xyz
%p0_3 = OpVectorShuffle %v3float %p0 %p0 0 1 2

; vec3 p1_3 = p1.xyz
%p1_3 = OpVectorShuffle %v3float %p1 %p1 0 1 2

; vec3 dir = p1_3 - p0_3
%dir = OpFSub %v3float %p1_3 %p0_3

; vec2 dir2 = dir.xy
%dir2 = OpVectorShuffle %v2float %dir %dir 0 1

; vec2 perp_raw = vec2(-dir2.y, dir2.x)
%dir_y = OpCompositeExtract %float %dir2 1
%dir_x = OpCompositeExtract %float %dir2 0
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeInsert %v2float %neg_dir_y %dir2 0
%perp_raw_x = OpCompositeExtract %float %perp_raw 0
%perp_raw_final = OpCompositeInsert %v2float %dir_x %perp_raw 1 ; Result: (-dir.y, dir.x)

; float len = length(dir2)
%len = OpExtInst %float %1 Length %dir2

; Avoid division by zero
%len_safe = OpFMax %float %len %float_0_0001

; vec2 perp = perp_raw / len
%perp_norm = OpCompositeConstruct %v2float %len_safe %len_safe
%perp = OpFDiv %v2float %perp_raw_final %perp_norm

; --- Shaft Calculations ---
; offset_s = perp * 0.015
%scale_s = OpCompositeConstruct %v2float %float_0_015 %float_0_015
%offset_s = OpFMul %v2float %perp %scale_s

; vec3 mid3 = p0 + 0.75 * (p1 - p0) = p0 + 0.75 * dir
%dir_scaled = OpVectorTimesScalar %v3float %dir %float_0_75
%mid3 = OpFAdd %v3float %p0_3 %dir_scaled

; p0_swizzle = vec3(p0_3.x, p0_3.y, p0_3.z) (same)
; We need to manipulate XY of p0 and mid.

; Construct shaft vertices (as vec3)
; V0: p0 - offset
%off_s_neg = OpFNegate %v2float %offset_s
%p0_xy = OpVectorShuffle %v2float %p0_3 %p0_3 0 1
%v0_xy = OpFAdd %v2float %p0_xy %off_s_neg
%v0_z = OpCompositeExtract %float %p0_3 2
%v0 = OpCompositeConstruct %v3float %v0_xy %v0_z

; V1: p0 + offset
%v1_xy = OpFAdd %v2float %p0_xy %offset_s
%v1 = OpCompositeConstruct %v3float %v1_xy %v0_z

; V2: mid - offset
%mid_xy = OpVectorShuffle %v2float %mid3 %mid3 0 1
%v2_xy = OpFAdd %v2float %mid_xy %off_s_neg
%v2_z = OpCompositeExtract %float %mid3 2
%v2 = OpCompositeConstruct %v3float %v2_xy %v2_z

; V3: mid + offset
%v3_xy = OpFAdd %v2float %mid_xy %offset_s
%v3 = OpCompositeConstruct %v3float %v3_xy %v2_z

; --- Output Shaft (4 vertices) ---
; Store V0
%out_pos_v0 = OpCompositeConstruct %v4float %v0 %float_1
%gl_out_v0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_v0_ptr %out_pos_v0
OpStore %outColor %c0
OpEmitVertex

; Store V1
%out_pos_v1 = OpCompositeConstruct %v4float %v1 %float_1
%gl_out_v1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_v1_ptr %out_pos_v1
OpStore %outColor %c0
OpEmitVertex

; Store V2
%out_pos_v2 = OpCompositeConstruct %v4float %v2 %float_1
%gl_out_v2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_v2_ptr %out_pos_v2
OpStore %outColor %c0
OpEmitVertex

; Store V3
%out_pos_v3 = OpCompositeConstruct %v4float %v3 %float_1
%gl_out_v3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_v3_ptr %out_pos_v3
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; --- Arrowhead Calculations ---
; offset_a = perp * 0.05
%scale_a = OpCompositeConstruct %v2float %float_0_05 %float_0_05
%offset_a = OpFMul %v2float %perp %scale_a

; V4: mid - offset_a
%off_a_neg = OpFNegate %v2float %offset_a
%v4_xy = OpFAdd %v2float %mid_xy %off_a_neg
%v4 = OpCompositeConstruct %v3float %v4_xy %v2_z

; V5: mid + offset_a
%v5_xy = OpFAdd %v2float %mid_xy %offset_a
%v5 = OpCompositeConstruct %v3float %v5_xy %v2_z

; V6: p1 (Tip)
%v6 = OpCompositeConstruct %v3float %p1_3

; --- Output Arrowhead (3 vertices) ---
; Store V4
%out_pos_v4 = OpCompositeConstruct %v4float %v4 %float_1
%gl_out_v4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_v4_ptr %out_pos_v4
OpStore %outColor %c1
OpEmitVertex

; Store V5
%out_pos_v5 = OpCompositeConstruct %v4float %v5 %float_1
%gl_out_v5_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_v5_ptr %out_pos_v5
OpStore %outColor %c1
OpEmitVertex

; Store V6
%out_pos_v6 = OpCompositeConstruct %v4float %v6 %float_1
%gl_out_v6_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_v6_ptr %out_pos_v6
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
