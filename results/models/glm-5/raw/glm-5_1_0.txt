#include <iostream>

using namespace std;

// Configuration constants
constexpr double TIME_LIMIT_SECONDS = 290.0;
constexpr size_t CACHE_BLOCK_SIZE = 8;
constexpr int SIMULATED_ANNEALING_ITERATIONS = 5000000;

// Global state
struct Point {
    double x, y;
};

vector<Point> cities;
vector<int> tour;
vector<double> dist_cache;
int N;
bool use_cache;
chrono::steady_clock::time_point start_time;

// Distance calculation with optional cache
inline double dist(int i, int j) {
    if (i == j) return 0.0;
    if (use_cache) {
        if (i > j) swap(i, j);
        size_t idx = (size_t)i * N + j;
        return dist_cache[idx];
    }
    double dx = cities[i].x - cities[j].x;
    double dy = cities[i].y - cities[j].y;
    return sqrt(dx * dx + dy * dy);
}

// Precompute distance cache for small-medium instances
void build_distance_cache() {
    if (N > 8000) {
        use_cache = false;
        return;
    }
    
    use_cache = true;
    dist_cache.resize((size_t)N * N);
    
    // Parallel-friendly cache construction (rows)
    for (int i = 0; i < N; ++i) {
        double ix = cities[i].x;
        double iy = cities[i].y;
        size_t row_start = (size_t)i * N;
        
        for (int j = i + 1; j < N; ++j) {
            double dx = ix - cities[j].x;
            double dy = iy - cities[j].y;
            dist_cache[row_start + j] = sqrt(dx * dx + dy * dy);
        }
        dist_cache[row_start + i] = 0.0;
    }
}

// Check remaining time
inline bool time_remaining() {
    auto now = chrono::steady_clock::now();
    chrono::duration<double> elapsed = now - start_time;
    return elapsed.count() < TIME_LIMIT_SECONDS;
}

// Greedy insertion heuristic - O(N^2) but very fast
void greedy_insertion() {
    tour.resize(N);
    vector<bool> used(N, false);
    
    tour[0] = 0;
    used[0] = true;
    
    // Find nearest neighbor for second city
    double best_d = numeric_limits<double>::max();
    int best_j = -1;
    for (int j = 1; j < N; ++j) {
        double d = dist(0, j);
        if (d < best_d) {
            best_d = d;
            best_j = j;
        }
    }
    tour[1] = best_j;
    used[best_j] = true;
    
    int tour_len = 2;
    
    for (int i = 2; i < N; ++i) {
        // Find next unvisited city
        int next_city = -1;
        for (int j = 1; j < N; ++j) {
            if (!used[j]) {
                next_city = j;
                break;
            }
        }
        if (next_city == -1) break;
        
        // Find best insertion position
        double best_incr = numeric_limits<double>::max();
        int best_pos = tour_len;
        
        for (int pos = 0; pos < tour_len; ++pos) {
            int a = tour[pos];
            int b = tour[(pos + 1) % tour_len];
            double incr = dist(a, next_city) + dist(next_city, b) - dist(a, b);
            if (incr < best_incr) {
                best_incr = incr;
                best_pos = pos + 1;
                if (best_incr < 1e-9) break; // Early exit for very good insertions
            }
        }
        
        // Insert city
        for (int k = tour_len; k > best_pos; --k) {
            tour[k] = tour[k - 1];
        }
        tour[best_pos] = next_city;
        used[next_city] = true;
        tour_len++;
        
        if (i % 10000 == 0 && !time_remaining()) return;
    }
}

// Simulated annealing 2-opt
void sa_2opt() {
    mt19937 rng(42);
    uniform_int_distribution<int> dist_n(1, N - 1);
    uniform_real_distribution<double> dist_real(0.0, 1.0);
    
    double temp = 1000.0;
    double cooling = 0.99999;
    int accepted = 0;
    
    for (int iter = 0; iter < SIMULATED_ANNEALING_ITERATIONS && time_remaining(); ++iter) {
        int i = dist_n(rng);
        int j = dist_n(rng);
        if (i == j) continue;
        if (i > j) swap(i, j);
        
        // Calculate delta for 2-opt swap
        int pi = (i == 0) ? tour[N - 1] : tour[i - 1];
        int ni = tour[i];
        int pj = tour[j];
        int nj = tour[(j + 1) % N];
        
        double delta = dist(pi, pj) + dist(ni, nj) - dist(pi, ni) - dist(pj, nj);
        
        if (delta < 0 || dist_real(rng) < exp(-delta / temp)) {
            reverse(tour.begin() + i, tour.begin() + j + 1);
            accepted++;
        }
        
        temp *= cooling;
    }
}

// Simple 2-opt improvement for medium instances
void simple_2opt() {
    bool improved = true;
    int iterations = 0;
    const int max_iterations = 50;
    
    while (improved && iterations < max_iterations && time_remaining()) {
        improved = false;
        iterations++;
        
        for (int i = 1; i < N - 1 && time_remaining(); ++i) {
            for (int j = i + 1; j < N; ++j) {
                int pi = tour[i - 1];
                int ni = tour[i];
                int pj = tour[j];
                int nj = tour[(j + 1) % N];
                
                double delta = dist(pi, pj) + dist(ni, nj) - dist(pi, ni) - dist(pj, nj);
                
                if (delta < -1e-9) {
                    reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// Quick nearest neighbor for massive instances
void fast_nearest_neighbor() {
    tour.resize(N);
    vector<bool> used(N, false);
    
    tour[0] = 0;
    used[0] = true;
    
    for (int i = 1; i < N; ++i) {
        int prev = tour[i - 1];
        double best_d = numeric_limits<double>::max();
        int best_j = -1;
        
        // For very large N, only check a sample
        int check_limit = min(N, 5000);
        int step = max(1, N / check_limit);
        
        for (int j = 1; j < N; ++j) {
            if (!used[j]) {
                double d = dist(prev, j);
                if (d < best_d) {
                    best_d = d;
                    best_j = j;
                }
            }
        }
        
        tour[i] = best_j;
        used[best_j] = true;
    }
}

// Space-filling curve sort for extreme cases
double hilbert_order(int x, int y, int bits) {
    double d = 0;
    for (int s = bits - 1; s >= 0; --s) {
        int rx = (x >> s) & 1;
        int ry = (y >> s) & 1;
        d += d + (rx * 2) + (ry ^ rx);
        if (ry == 0) {
            if (rx == 1) {
                x = (1 << bits) - 1 - x;
                y = (1 << bits) - 1 - y;
            }
            swap(x, y);
        }
    }
    return d;
}

void space_filling_sort() {
    tour.resize(N);
    
    // Find bounding box
    double min_x = cities[0].x, max_x = cities[0].x;
    double min_y = cities[0].y, max_y = cities[0].y;
    
    for (int i = 1; i < N; ++i) {
        min_x = min(min_x, cities[i].x);
        max_x = max(max_x, cities[i].x);
        min_y = min(min_y, cities[i].y);
        max_y = max(max_y, cities[i].y);
    }
    
    double range_x = max_x - min_x + 1e-9;
    double range_y = max_y - min_y + 1e-9;
    
    int bits = max(10, (int)ceil(log2(N)));
    bits = min(bits, 20); // Cap at 20 bits
    
    vector<pair<double, int>> order(N);
    
    for (int i = 0; i < N; ++i) {
        int ix = (int)((cities[i].x - min_x) / range_x * ((1 << bits) - 1));
        int iy = (int)((cities[i].y - min_y) / range_y * ((1 << bits) - 1));
        ix = max(0, min(ix, (1 << bits) - 1));
        iy = max(0, min(iy, (1 << bits) - 1));
        order[i] = {hilbert_order(ix, iy, bits), i};
    }
    
    sort(order.begin(), order.end());
    
    // Ensure city 0 is first
    int start_pos = 0;
    for (int i = 0; i < N; ++i) {
        if (order[i].second == 0) {
            start_pos = i;
            break;
        }
    }
    
    for (int i = 0; i < N; ++i) {
        tour[i] = order[(start_pos + i) % N].second;
    }
}

// Faster greedy for large instances
void fast_greedy() {
    tour.resize(N);
    vector<bool> used(N, false);
    
    tour[0] = 0;
    used[0] = true;
    
    // Build initial tour using nearest neighbor
    for (int i = 1; i < N; ++i) {
        int prev = tour[i - 1];
        double best_d = numeric_limits<double>::max();
        int best_j = -1;
        
        for (int j = 1; j < N; ++j) {
            if (!used[j]) {
                double d = dist(prev, j);
                if (d < best_d) {
                    best_d = d;
                    best_j = j;
                    if (best_d < 1e-6) break;
                }
            }
        }
        
        tour[i] = best_j;
        used[best_j] = true;
        
        if (i % 50000 == 0 && !time_remaining()) return;
    }
}

// Main solver
void solve() {
    start_time = chrono::steady_clock::now();
    
    if (N == 1) {
        tour = {0};
        return;
    }
    
    if (N == 2) {
        tour = {0, 1};
        return;
    }
    
    // Choose algorithm based on problem size
    if (N <= 12) {
        // Exact brute-force for tiny instances
        build_distance_cache();
        tour.resize(N);
        for (int i = 0; i < N; ++i) tour[i] = i;
        
        vector<int> best = tour;
        double best_len = numeric_limits<double>::max();
        
        do {
            if (tour[0] != 0) continue;
            double len = 0;
            for (int i = 0; i < N - 1; ++i) {
                len += dist(tour[i], tour[i + 1]);
            }
            len += dist(tour[N - 1], tour[0]);
            
            if (len < best_len) {
                best_len = len;
                best = tour;
            }
        } while (next_permutation(tour.begin(), tour.end()));
        
        tour = best;
    }
    else if (N <= 100) {
        // Small: build cache, greedy + 2-opt
        build_distance_cache();
        greedy_insertion();
        simple_2opt();
    }
    else if (N <= 2000) {
        // Medium: cache, greedy + SA
        build_distance_cache();
        greedy_insertion();
        sa_2opt();
    }
    else if (N <= 50000) {
        // Large: cache, fast greedy + limited SA
        build_distance_cache();
        fast_greedy();
        if (time_remaining()) {
            sa_2opt();
        }
    }
    else if (N <= 500000) {
        // Very large: space-filling sort with local refinement
        use_cache = false;
        space_filling_sort();
        
        // Quick local optimization if time permits
        if (time_remaining() && N <= 100000) {
            simple_2opt();
        }
    }
    else {
        // Extreme: space-filling curve only
        use_cache = false;
        space_filling_sort();
    }
}
