Looking at this problem, I need to implement the Chinese Postman Problem (CPP) solver that scales from tiny to massive graphs. Let me break down the approach:

**Algorithm Overview:**
1. Check if graph is Eulerian (all vertices have even degree)
2. If not, find minimum weight matching on odd-degree vertices
3. Add "artificial" edges for the matching pairs
4. Find Euler circuit on the augmented graph

**Key Optimizations:**
- For small odd-vertex counts: Hungarian algorithm
- For medium: Dijkstra with priority queue + Greedy matching refinement
- For large: Multi-threaded shortest paths + optimized greedy matching

```csharp
using System;
using System.Collections.Generic;
using System.Threading.Tasks;

public sealed class Program
{
    const int INF = 2000000000;
    static int n, m;
    static List<(int to, int w)>[] adj;
    static int[] edgeU, edgeV, edgeW;
    static long totalWeight;
    static bool[] edgeUsed;
    static int edgePtr;
    static List<(int to, int w, int edgeId)>[] adjWithIds;
    static int[] edgeFrom, edgeTo;

    public static void Main()
    {
        ReadInput();
        int[] odd = FindOddVertices();

        if (odd.Length > 0)
            AddMatchingEdges(odd);

        FindEulerCircuit();
    }

    static void ReadInput()
    {
        var reader = new FastReader(Console.OpenStandardInput());
        n = reader.NextInt();
        m = reader.NextInt();

        adj = new List<(int, int)>[n];
        adjWithIds = new List<(int, int, int)>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<(int, int)>();
            adjWithIds[i] = new List<(int, int, int)>();
        }

        edgeU = new int[m];
        edgeV = new int[m];
        edgeW = new int[m];
        edgeFrom = new int[m];
        edgeTo = new int[m];
        totalWeight = 0;

        for (int i = 0; i < m; i++)
        {
            int u = reader.NextInt();
            int v = reader.NextInt();
            int w = reader.NextInt();

            edgeU[i] = u;
            edgeV[i] = v;
            edgeW[i] = w;
            totalWeight += w;

            adj[u].Add((v, w));
            adj[v].Add((u, w));
        }
    }

    static int[] FindOddVertices()
    {
        var odd = new List<int>();
        for (int i = 0; i < n; i++)
            if ((adj[i].Count & 1) == 1)
                odd.Add(i);
        return odd.ToArray();
    }

    static void AddMatchingEdges(int[] odd)
    {
        int k = odd.Length;

        // Compute pairwise distances
        int[,] dist = new int[k, k];
        bool useParallel = k >= 20 && n >= 50;

        if (useParallel)
        {
            int threads = Math.Min(Environment.ProcessorCount, (k + 9) / 10);
            threads = Math.Max(1, Math.Min(threads, k));
            Parallel.For(0, k, new ParallelOptions { MaxDegreeOfParallelism = threads }, i =>
            {
                int source = odd[i];
                int[] d = Dijkstra(source);
                for (int j = 0; j < k; j++)
                    dist[i, j] = d[odd[j]];
            });
        }
        else
        {
            for (int i = 0; i < k; i++)
            {
                int source = odd[i];
                int[] d = Dijkstra(source);
                for (int j = 0; j < k; j++)
                    dist[i, j] = d[odd[j]];
            }
        }

        // Find minimum weight matching
        int[] match;
        if (k <= 14)
            match = HungarianMatching(k, dist);
        else
            match = GreedyMatching(k, dist, odd);

        // Add artificial edges for matched pairs
        int eid = m;
        for (int i = 0; i < k; i++)
        {
            if (match[i] > i)
            {
                int u = odd[i], v = odd[match[i]];
                int w = dist[i, match[i]];
                adjWithIds[u].Add((v, w, eid));
                adjWithIds[v].Add((u, w, eid));
                eid++;
            }
        }
    }

    static int[] Dijkstra(int source)
    {
        int[] dist = new int[n];
        for (int i = 0; i < n; i++) dist[i] = INF;
        dist[source] = 0;

        var pq = new PriorityQueue(n);
        pq.Push(source, 0);

        while (pq.Count > 0)
        {
            var (u, d) = pq.Pop();
            if (d != dist[u]) continue;

            foreach (var (v, w) in adj[u])
            {
                int nd = d + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    pq.Push(v, nd);
                }
            }
        }
        return dist;
    }

    static int[] HungarianMatching(int k, int[,] cost)
    {
        int max = k / 2;
        int[,] d = new int[max + 1, max + 1];
        for (int i = 0; i < max; i++)
            for (int j = 0; j < max; j++)
                d[i + 1, j + 1] = cost[i, j + max];

        int[] u = new int[max + 1];
        int[] v = new int[max + 1];
        int[] p = new int[max + 1];
        int[] way = new int[max + 1];

        for (int i = 1; i <= max; i++)
        {
            int[] minv = new int[max + 1];
            bool[] used = new bool[max + 1];
            for (int j = 0; j <= max; j++) minv[j] = INF;
            p[0] = i;
            int j0 = 0;

            while (true)
            {
                int j1 = 0, delta = INF;
                used[j0] = true;
                int i0 = p[j0];

                for (int j = 1; j <= max; j++)
                {
                    if (!used[j])
                    {
                        int cur = d[i0, j] - u[i0] - v[j];
                        if (cur < minv[j]) minv[j] = cur;
                        if (minv[j] < delta) { delta = minv[j]; j1 = j; }
                    }
                }

                for (int j = 0; j <= max; j++)
                {
                    if (used[j]) { u[p[j]] += delta; v[j] -= delta; }
                    else minv[j] -= delta;
                }

                j0 = j1;
                if (p[j0] == 0) break;
            }

            while (j0 != 0)
            {
                int j1 = way[j0];
                p[j0] = p[j1];
                j0 = j1;
            }
        }

        int[] ans = new int[max + 1];
        for (int j = 1; j <= max; j++) ans[p[j]] = j;

        int[] match = new int[k];
        for (int i = 1; i <= max; i++)
        {
            int j = ans[i];
            match[i - 1] = j - 1 + max;
            match[j - 1 + max] = i - 1;
        }
        return match;
    }

    static int[] GreedyMatching(int k, int[,] dist, int[] odd)
    {
        int[] match = new int[k];
        bool[] done = new bool[k];

        var heap = new PairHeap();
        for (int i = 0; i < k; i++)
            for (int j = i + 1; j < k; j++)
                heap.Push((i << 20) + j, dist[i, j]);

        while (heap.Count > 0)
        {
            var (key, _) = heap.Pop();
            int i = key >> 20, j = key & 0xFFFFF;
            if (!done[i] && !done[j])
            {
                match[i] = j;
                match[j] = i;
                done[i] = done[j] = true;
            }
        }
        return match;
    }

    static void FindEulerCircuit()
    {
        // Build adjacency with edge IDs
        for (int i = 0; i < m; i++)
        {
            int u = edgeU[i], v = edgeV[i], w = edgeW[i];
            adjWithIds[u].Add((v, w, i));
            adjWithIds[v].Add((u, w, i));
            edgeFrom[i] = u;
            edgeTo[i] = v;
        }

        edgeUsed = new bool[m + 200000];
        edgePtr = m;

        var stack = new FastStack(n + m + 100);
        var circuit = new FastList(m * 2 + 200000);
        stack.Push(0);

        int[] curIdx = new int[n];
        while (stack.Count > 0)
        {
            int u = stack.Peek();
            var list = adjWithIds[u];
            int ci = curIdx[u];

            while (ci < list.Count && edgeUsed[list[ci].edgeId])
                ci++;

            if (ci < list.Count)
            {
                curIdx[u] = ci + 1;
                var e = list[ci];
                edgeUsed[e.edgeId] = true;
                stack.Push(e.to);
            }
            else
            {
                stack.Pop();
                circuit.Add(u);
            }
        }

        WriteOutput(circuit);
    }

    static void WriteOutput(FastList circuit)
    {
        var writer = new FastWriter();
        writer.Write(circuit.Count);
        writer.Write('\n');
        for (int i = 0; i < circuit.Count; i++)
        {
            writer.Write(circuit[i]);
            writer.Write('\n');
        }
        writer.Flush();
    }

    // Fast I/O structures
    sealed class FastReader
    {
        byte[] buf = new byte[1 << 16];
        int pos, len;
        System.IO.Stream s;

        public FastReader(System.IO.Stream stream) { s = stream; }

        public int NextInt()
        {
            int c = ReadByte();
            while (c < '0' || c > '9') c = ReadByte();
            int r = 0;
            while (c >= '0' && c <= '9') { r = r * 10 + c - '0'; c = ReadByte(); }
            return r;
        }

        int ReadByte()
        {
            if (pos >= len) { pos = 0; len = s.Read(buf, 0, buf.Length); }
            return pos < len ? buf[pos++] : -1;
        }
    }

    sealed class FastWriter
    {
        byte[] buf = new byte[1 << 20];
        int pos;

        public void Write(int v)
        {
            if (v == 0) { buf[pos++] = (byte)'0'; return; }
            int start = pos;
            while (v > 0) { buf[pos++] = (byte)('0' + v % 10); v /= 10; }
            Array.Reverse(buf, start, pos - start);
        }

        public void Write(char c) => buf[pos++] = (byte)c;
        public void Flush() => Console.OpenStandardOutput().Write(buf, 0, pos);
    }

    sealed class FastList
    {
        int[] a;
        public int Count;
        public FastList(int cap) { a = new int[cap]; }
        public void Add(int v) { if (Count >= a.Length) Array.Resize(ref a, Count * 2); a[Count++] = v; }
        public int this[int i] => a[i];
    }

    sealed class FastStack
    {
        int[] a;
        public int Count;
        public FastStack(int cap) { a = new int[cap]; }
        public void Push(int v) => a[Count++] = v;
        public int Pop() => a[--Count];
        public int Peek() => a[Count - 1];
    }

    sealed class PriorityQueue
    {
        (int node, int dist)[] heap;
        int count;
        public PriorityQueue(int cap) { heap = new (int, int)[cap + 1]; }
        public int Count => count;
        public void Push(int node, int dist) { heap[++count] = (node, dist); Up(count); }
        public (int, int) Pop() { var r = heap[1]; heap[1] = heap[count--]; if (count > 0) Down(1); return r; }

        void Up(int i) { while (i > 1 && heap[i].dist < heap[i >> 1].dist) { Swap(ref i, i >> 1); } }
        void Down(int i) { int j; while ((j = i << 1) <= count) { if (j < count && heap[j + 1].dist < heap[j].dist) j++; if (heap[i].dist <= heap[j].dist) break; Swap(ref i, j); } }
        void Swap(ref int i, int j) { var t = heap[i]; heap[i] = heap[j]; heap[j] = t; i = j; }
    }

    sealed class PairHeap
    {
        (int key, int val)[] heap;
        int count;
        public PairHeap() { heap = new (int, int)[65536]; }
        public int Count => count;
        public void Push(int key, int val) { if (count >= heap.Length) Array.Resize(ref heap, count * 2); heap[++count] = (key, val); Up(count); }
        public (int, int) Pop() { var r = heap[1]; heap[1] = heap[count--]; if (count > 0) Down(1); return r; }

        void Up(int i) { while (i > 1 && heap[i].val < heap[i >> 1].val) { Swap(ref i, i >> 1); } }
        void Down(int i) { int j; while ((j = i << 1) <= count) { if (j < count && heap[j + 1].val < heap[j].val) j++; if (heap[i].val <= heap[j].val) break; Swap(ref i, j); } }
        void Swap(ref int i, int j) { var t = heap[i]; heap[i] = heap[j]; heap[j] = t; i = j; }
    }
}
```

**Key Features:**
1. **Fast I/O**: Custom buffered reader/writer to handle massive input/output
2. **Scalable Matching**: Uses Hungarian algorithm (exact) for small odd-vertex sets, greedy with heap for larger
3. **Parallel Dijkstra**: Multi-threaded shortest path computation for large graphs
4. **Memory Efficient**: Uses arrays instead of objects where possible
5. **Hierholzer's Algorithm**: Linear-time Euler circuit finding with edge tracking

**Complexity:**
- Time: O(V² + E log V) for sparse, O(V² log V + VE) worst case
- Space: O(V + E)
