struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);

    // Calculate light direction
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Calculate Lambertian diffuse factor
    float NdotL = dot(N, L);
    float diffuse = max(NdotL, 0.0);

    // Polka-dot pattern calculation
    // Create 6x6 grid cells
    float2 cellUV = input.uv * 6.0;
    
    // Determine grid cell coordinates (0 to 5)
    float2 cellIndex = floor(cellUV);
    
    // Calculate local UV coordinates within the cell [0, 1]
    float2 localUV = frac(cellUV);
    
    // Calculate distance from center of the cell (0.5, 0.5)
    float dist = distance(localUV, float2(0.5, 0.5));
    
    // Determine base color based on radius 0.3
    // If distance < 0.3, it's a dot (yellow), otherwise background (dark blue)
    float3 baseColor = (dist < 0.3) ? float3(1.0, 0.8, 0.0) : float3(0.1, 0.1, 0.3);

    // Apply shading factor: 0.3 + 0.7 * diffuse
    float shadeFactor = 0.3 + 0.7 * diffuse;
    
    float3 finalColor = baseColor * shadeFactor;

    return float4(finalColor, 1.0);
}
