```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal vector
    float3 N = normalize(input.normal);

    // Calculate Light Vector (L) - direction from surface to light
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Calculate View Vector (V) - direction from surface to camera
    float3 V = normalize(cameraPos.xyz - input.worldPos);

    // Calculate Reflection Vector (R) - reflection of -L around N
    float3 R = reflect(-L, N);

    // Calculate Diffuse component
    float diffuse = max(dot(N, L), 0.0);

    // Calculate Specular component with shininess 32
    float specular = pow(max(dot(R, V), 0.0), 32.0);

    // Combine into grayscale intensity
    // Formula: 0.2 (ambient) + 0.5 * diffuse + 0.8 * specular
    float intensity = 0.2 + 0.5 * diffuse + 0.8 * specular;

    // Output RGBA with full opacity
    return float4(intensity.rrr, 1.0);
}
```
