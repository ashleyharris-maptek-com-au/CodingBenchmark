const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot;
const truckCount = data.truckCount;
const packages = data.packages;

// Utility functions
const distance = (p1, p2) => {
    const dx = p1.x - p2.x;
    const dy = p1.y - p2.y;
    return Math.sqrt(dx * dx + dy * dy);
};

// Pre-calculate distances from depot to all packages
const depotDist = new Map();
for (const p of packages) {
    depotDist.set(p.id, distance(depot, p));
}

// Nearest Neighbor Heuristic to build initial routes
function nearestNeighbor(nodes, startNode) {
    if (nodes.length === 0) return [];
    
    const remaining = new Set(nodes);
    const path = [];
    let current = startNode;
    
    // If startNode is in the set, remove it first
    if (remaining.has(startNode)) {
        remaining.delete(startNode);
        path.push(startNode);
    }
    
    while (remaining.size > 0) {
        let best = null;
        let bestDist = Infinity;
        
        for (const node of remaining) {
            const d = current ? distance(current, node) : depotDist.get(node.id);
            if (d < bestDist) {
                bestDist = d;
                best = node;
            }
        }
        
        remaining.delete(best);
        path.push(best);
        current = best;
    }
    
    return path;
}

// 2-opt optimization for a single route
function twoOpt(route) {
    if (route.length < 2) return route;
    
    const getIndex = (i) => {
        if (i < 0) return depot;
        if (i >= route.length) return depot;
        return route[i];
    };
    
    const getDist = (a, b) => {
        if (a === depot) return depotDist.get(b.id);
        if (b === depot) return depotDist.get(a.id);
        return distance(a, b);
    };
    
    let improved = true;
    let currentRoute = [...route];
    
    while (improved) {
        improved = false;
        
        for (let i = 0; i < currentRoute.length - 1; i++) {
            for (let j = i + 1; j < currentRoute.length; j++) {
                // Calculate current distance
                const d1 = getDist(getIndex(i), getIndex(i + 1)) + 
                          getDist(getIndex(j), getIndex(j + 1));
                
                // Calculate new distance after reversing segment [i+1, j]
                const d2 = getDist(getIndex(i), getIndex(j)) + 
                          getDist(getIndex(i + 1), getIndex(j + 1));
                
                if (d2 < d1 - 1e-10) {
                    // Reverse the segment
                    const segment = currentRoute.slice(i + 1, j + 1);
                    segment.reverse();
                    currentRoute = [
                        ...currentRoute.slice(0, i + 1),
                        ...segment,
                        ...currentRoute.slice(j + 1)
                    ];
                    improved = true;
                }
            }
        }
    }
    
    return currentRoute;
}

// Calculate route cost
function routeCost(route) {
    if (route.length === 0) return 0;
    let cost = depotDist.get(route[0].id);
    for (let i = 0; i < route.length - 1; i++) {
        cost += distance(route[i], route[i + 1]);
    }
    cost += depotDist.get(route[route.length - 1].id);
    return cost;
}

// Simple angle-based clustering for route initialization
function initializeRoutes() {
    if (packages.length === 0) return [];
    
    // Calculate angles from depot
    const withAngles = packages.map(p => ({
        ...p,
        angle: Math.atan2(p.y - depot[1], p.x - depot[0])
    }));
    
    // Sort by angle
    withAngles.sort((a, b) => a.angle - b.angle);
    
    // Initialize routes
    const routes = [];
    const packagesPerRoute = Math.ceil(withAngles.length / truckCount);
    
    for (let i = 0; i < truckCount; i++) {
        const start = i * packagesPerRoute;
        const end = Math.min(start + packagesPerRoute, withAngles.length);
        if (start < withAngles.length) {
            routes.push(withAngles.slice(start, end));
        }
    }
    
    // Ensure we have at least some routes if packages exist
    while (routes.length === 0 && packages.length > 0) {
        routes.push([]);
    }
    
    return routes;
}

// Save-operator: try moving a node from one route to best position in another
function saveOperator(routes) {
    let improved = false;
    
    for (let r1 = 0; r1 < routes.length; r1++) {
        if (routes[r1].length === 0) continue;
        
        for (let idx = 0; idx < routes[r1].length; idx++) {
            const node = routes[r1][idx];
            let bestRoute = -1;
            let bestPos = -1;
            let bestImprovement = 0;
            
            // Calculate removal cost
            const prev = idx > 0 ? routes[r1][idx - 1] : null;
            const next = idx < routes[r1].length - 1 ? routes[r1][idx + 1] : null;
            
            const distPrev = prev ? distance(prev, node) : depotDist.get(node.id);
            const distNext = next ? distance(node, next) : depotDist.get(node.id);
            const distPrevNext = prev && next ? distance(prev, next) : 
                                prev ? depotDist.get(prev.id) : 
                                next ? depotDist.get(next.id) : 0;
            
            const removalDelta = distPrevNext - distPrev - distNext;
            
            // Try inserting into other routes
            for (let r2 = 0; r2 < routes.length; r2++) {
                if (r1 === r2) continue;
                
                for (let pos = 0; pos <= routes[r2].length; pos++) {
                    const prev2 = pos > 0 ? routes[r2][pos - 1] : null;
                    const next2 = pos < routes[r2].length ? routes[r2][pos] : null;
                    
                    const distPrev2 = prev2 ? distance(prev2, node) : depotDist.get(node.id);
                    const distNext2 = next2 ? distance(node, next2) : depotDist.get(node.id);
                    const distPrevNext2 = prev2 && next2 ? distance(prev2, next2) : 
                                         prev2 ? depotDist.get(prev2.id) : 
                                         next2 ? depotDist.get(next2.id) : 0;
                    
                    const insertionDelta = distPrev2 + distNext2 - distPrevNext2;
                    const totalDelta = removalDelta + insertionDelta;
                    
                    if (totalDelta < bestImprovement) {
                        bestImprovement = totalDelta;
                        bestRoute = r2;
                        bestPos = pos;
                    }
                }
            }
            
            if (bestRoute !== -1 && bestImprovement < -1e-10) {
                // Perform the move
                routes[r1].splice(idx, 1);
                routes[bestRoute].splice(bestPos, 0, node);
                improved = true;
                idx--; // Adjust index since we removed an element
            }
        }
    }
    
    return improved;
}

// Main optimization
function optimize() {
    if (packages.length === 0) {
        return [];
    }
    
    // Initialize routes using angle-based clustering
    let routes = initializeRoutes();
    
    // Optimize each route with 2-opt
    for (let i = 0; i < routes.length; i++) {
        routes[i] = twoOpt(routes[i]);
    }
    
    // Apply save operator
    let iterations = 0;
    const maxIterations = packages.length * 2;
    
    while (iterations < maxIterations && saveOperator(routes)) {
        // Re-optimize affected routes with 2-opt
        for (let i = 0; i < routes.length; i++) {
            routes[i] = twoOpt(routes[i]);
        }
        iterations++;
    }
    
    return routes;
}

const routes = optimize();

// Extract package IDs
const result = {
    routes: routes.map(route => route.map(p => p.id))
};

console.log(JSON.stringify(result));
