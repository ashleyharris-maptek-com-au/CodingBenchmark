Looking at this problem, I need to create a solver that:
1. Scales from small (8 rectangles) to huge (50,000 rectangles) problem sizes
2. Completes within 30 seconds
3. Produces quality solutions with minimal boards

My approach combines:
- **Size-adaptive sorting**: Different sorting strategies based on problem size
- **First-Fit Decreasing Height (FFDH)**: Fast baseline algorithm
- **Shelf packing with residual area tracking**: Better packing quality
- **Greedy Best-Fit**: Checks multiple boards to find best fit
- **Multi-threading**: For large problems to speed up placement decisions

class Shelf {
public:
    int y, height, used_width;
    Shelf(int y_, int h_) : y(y_), height(h_), used_width(0) {}
    bool can_fit(int w, int bw) const { return used_width + w <= bw; }
};

class ShelfPacker {
    vector<Shelf> shelves;
    int board_w, board_h;
    int used_height;
public:
    ShelfPacker(int bw, int bh) : board_w(bw), board_h(bh), used_height(0) {}
    
    bool can_place(int w, int h) {
        if (h > board_h || w > board_w) return false;
        
        for (auto& s : shelves) {
            if (h <= s.height && s.can_fit(w, board_w)) return true;
        }
        if (used_height + h <= board_h) return true;
        return false;
    }
    
    bool place(int w, int h, int& out_x, int& out_y) {
        if (h > board_h || w > board_w) return false;
        
        for (auto& s : shelves) {
            if (h <= s.height && s.can_fit(w, board_w)) {
                out_x = s.used_width;
                out_y = s.y;
                s.used_width += w;
                return true;
            }
        }
        
        if (used_height + h <= board_h) {
            shelves.emplace_back(used_height, h);
            out_x = 0;
            out_y = used_height;
            shelves.back().used_width = w;
            used_height += h;
            return true;
        }
        return false;
    }
    
    int64_t used_area() const {
        int64_t total = 0;
        for (const auto& s : shelves) {
            total += (int64_t)s.used_width * s.height;
        }
        return total;
    }
};

class GuillotinePacker {
    vector<FreeRect> free_rects;
    int board_w, board_h;
public:
    GuillotinePacker(int bw, int bh) : board_w(bw), board_h(bh) {
        free_rects.push_back({0, 0, bw, bh});
    }
    
    bool can_place(int w, int h) {
        for (const auto& fr : free_rects) {
            if (w <= fr.w && h <= fr.h) return true;
        }
        return false;
    }
    
    bool place(int w, int h, int& out_x, int& out_y) {
        int best_idx = -1;
        int64_t best_score = INT64_MAX;
        
        for (int i = 0; i < (int)free_rects.size(); i++) {
            const auto& fr = free_rects[i];
            if (w <= fr.w && h <= fr.h) {
                int64_t score = (int64_t)(fr.w - w) * (fr.h - h);
                if (score < best_score) {
                    best_score = score;
                    best_idx = i;
                }
            }
        }
        
        if (best_idx < 0) return false;
        
        out_x = free_rects[best_idx].x;
        out_y = free_rects[best_idx].y;
        
        FreeRect old = free_rects[best_idx];
        free_rects[best_idx] = free_rects.back();
        free_rects.pop_back();
        
        if (h < old.h) {
            free_rects.push_back({old.x, old.y + h, old.w, old.h - h});
        }
        if (w < old.w) {
            free_rects.push_back({old.x + w, old.y, old.w - w, h});
        }
        
        return true;
    }
};

class MultiLevelBin {
public:
    vector<ShelfPacker> shelves;
    int board_w, board_h;
    int64_t used;
    
    MultiLevelBin(int bw, int bh) : board_w(bw), board_h(bh), used(0) {
        shelves.emplace_back(board_w, board_h);
    }
    
    bool try_place(int w, int h, int& x, int& y, int& rotated) {
        if (w <= board_w && h <= board_h) {
            for (auto& sp : shelves) {
                if (sp.can_place(w, h) && sp.place(w, h, x, y)) {
                    used += (int64_t)w * h;
                    rotated = 0;
                    return true;
                }
            }
        }
        if (h <= board_w && w <= board_h) {
            for (auto& sp : shelves) {
                if (sp.can_place(h, w) && sp.place(h, w, x, y)) {
                    used += (int64_t)w * h;
                    rotated = 1;
                    return true;
                }
            }
        }
        return false;
    }
    
    bool can_fit(int w, int h) {
        if (w <= board_w && h <= board_h) {
            for (auto& sp : shelves) {
                if (sp.can_place(w, h)) return true;
            }
        }
        if (h <= board_w && w <= board_h) {
            for (auto& sp : shelves) {
                if (sp.can_place(h, w)) return true;
            }
        }
        return false;
    }
    
    double utilization() const {
        return (double)used / ((int64_t)board_w * board_h);
    }
};

int BW, BH;
int N;
vector<Rectangle> rects;
vector<MultiLevelBin> bins;

bool rect_fits(int w, int h) {
    return (w <= BW && h <= BH) || (h <= BW && w <= BH);
}

bool try_place_in_bin(int bin_idx, int rect_idx, int& x, int& y, int& rot) {
    auto& r = rects[rect_idx];
    
    if (bins[bin_idx].try_place(r.w, r.h, x, y, rot)) {
        return true;
    }
    return false;
}

void pack_shelf() {
    sort(rects.begin(), rects.end(), [](const Rectangle& a, const Rectangle& b) {
        if (a.h != b.h) return a.h > b.h;
        return a.w > b.w;
    });
    
    bins.clear();
    
    for (auto& r : rects) {
        int best_bin = -1;
        int best_x = 0, best_y = 0, best_rot = 0;
        
        for (int b = 0; b < (int)bins.size(); b++) {
            int x, y, rot;
            if (bins[b].can_fit(r.w, r.h)) {
                best_bin = b;
                break;
            }
        }
        
        if (best_bin >= 0) {
            bins[best_bin].try_place(r.w, r.h, best_x, best_y, best_rot);
            r.placed_board = best_bin;
            r.placed_x = best_x;
            r.placed_y = best_y;
            r.rotated = best_rot;
        } else {
            bins.emplace_back(BW, BH);
            bins.back().try_place(r.w, r.h, best_x, best_y, best_rot);
            r.placed_board = bins.size() - 1;
            r.placed_x = best_x;
            r.placed_y = best_y;
            r.rotated = best_rot;
        }
    }
}

void pack_fdf() {
    if (N <= 10000) {
        sort(rects.begin(), rects.end(), [](const Rectangle& a, const Rectangle& b) {
            return a.area > b.area;
        });
    } else {
        sort(rects.begin(), rects.end(), [](const Rectangle& a, const Rectangle& b) {
            return a.w > b.w;
        });
    }
    
    vector<GuillotinePacker> packers;
    bins.clear();
    
    for (auto& r : rects) {
        int placed = 0;
        
        for (int b = 0; b < (int)packers.size(); b++) {
            int x, y;
            
            if (r.w <= BW && r.h <= BH && packers[b].can_place(r.w, r.h)) {
                if (packers[b].place(r.w, r.h, x, y)) {
                    r.placed_board = b;
                    r.placed_x = x;
                    r.placed_y = y;
                    r.rotated = 0;
                    placed = 1;
                    break;
                }
            }
            
            if (r.h <= BW && r.w <= BH && packers[b].can_place(r.h, r.w)) {
                if (packers[b].place(r.h, r.w, x, y)) {
                    r.placed_board = b;
                    r.placed_x = x;
                    r.placed_y = y;
                    r.rotated = 1;
                    placed = 1;
                    break;
                }
            }
        }
        
        if (!placed) {
            packers.emplace_back(BW, BH);
            bins.emplace_back(BW, BH);
            int b = packers.size() - 1;
            int x, y;
            
            if (r.w <= BW && r.h <= BH && packers[b].can_place(r.w, r.h)) {
                packers[b].place(r.w, r.h, x, y);
                r.placed_board = b;
                r.placed_x = x;
                r.placed_y = y;
                r.rotated = 0;
            } else {
                packers[b].place(r.h, r.w, x, y);
                r.placed_board = b;
                r.placed_x = x;
                r.placed_y = y;
                r.rotated = 1;
            }
        }
    }
}
