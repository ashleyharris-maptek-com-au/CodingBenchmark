static inline uint64_t hilbertD(uint32_t n, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = n / 2; s > 0; s >>= 1) {
        uint32_t rx = (x & s) ? 1u : 0u;
        uint32_t ry = (y & s) ? 1u : 0u;
        d += (uint64_t)s * (uint64_t)s * ((3u * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

struct Solver {
    w: usize,
    h: usize,
    n: usize,
    blocked: Vec<bool>,
    target: usize,
    require_cycle: bool,
    visited: Vec<bool>,
    path: Vec<usize>,
    deadline: Instant,
    time_limit_secs: u64,
    found: bool,
}

impl Solver {
    fn xy(&self, i: usize) -> (usize, usize) { (i % self.w, i / self.w) }

    fn neighbors(&self, i: usize) -> [Option<usize>; 4] {
        let (x, y) = self.xy(i);
        let mut r = [None; 4];
        if x > 0 { r[0] = Some(i - 1); }
        if x + 1 < self.w { r[1] = Some(i + 1); }
        if y > 0 { r[2] = Some(i - self.w); }
        if y + 1 < self.h { r[3] = Some(i + self.w); }
        r
    }

    fn parity_ok(&self) -> bool {
        let mut c0 = 0i64;
        let mut c1 = 0i64;
        for i in 0..self.n {
            if !self.blocked[i] {
                let (x, y) = self.xy(i);
                if (x + y) & 1 == 0 { c0 += 1; } else { c1 += 1; }
            }
        }
        // start (0,0) is color 0
        if self.require_cycle {
            if c0 != c1 { return false; }
        } else {
            // path starts at color 0
            // diff must be 0 (ends opposite) or 1 (ends same)
            let diff = c0 - c1;
            if diff != 0 && diff != 1 { return false; }
        }
        true
    }

    fn connected_check(&self, head: usize) -> bool {
        let mut stack = vec![head];
        let mut seen = vec![false; self.n];
        seen[head] = true;
        let mut count = 0usize;
        while let Some(u) = stack.pop() {
            for nb in self.neighbors(u).iter().flatten() {
                let v = *nb;
                if self.blocked[v] || seen[v] || self.visited[v] { continue; }
                seen[v] = true;
                count += 1;
                stack.push(v);
            }
        }
        count == self.target - self.path.len()
    }

    fn dfs(&mut self, current: usize) -> bool {
        if self.found { return true; }
        if self.path.len() & 63 == 0 {
            if self.deadline.elapsed().as_secs() >= self.time_limit_secs {
                return false;
            }
        }

        if self.path.len() == self.target {
            if self.require_cycle {
                for nb in self.neighbors(current).iter().flatten() {
                    if *nb == 0 { self.found = true; return true; }
                }
                return false;
            } else {
                self.found = true;
                return true;
            }
        }

        // Periodic connectivity check (expensive, do every few steps)
        if self.target >= 30 && (self.path.len() & 7) == 0 {
            if !self.connected_check(current) { return false; }
        }

        // Dead-end pruning: any unvisited cell (not head) with 0 available neighbors => fail
        // any with 1 neighbor must be endpoint
        let mut forced: Option<usize> = None;
        let mut dead_ones = 0usize;
        for i in 0..self.n {
            if self.blocked[i] || self.visited[i] { continue; }
            let mut d = 0;
            let mut adj_head = false;
            for nb in self.neighbors(i).iter().flatten() {
                let v = *nb;
                if self.blocked[v] { continue; }
                if !self.visited[v] { d += 1; }
                if v == current { adj_head = true; }
            }
            if d == 0 {
                // isolated unvisited; only OK if it's adjacent to head AND will be next
                if !adj_head { return false; }
                if forced.is_some() && forced != Some(i) { return false; }
                forced = Some(i);
            } else if d == 1 {
                dead_ones += 1;
                // it must be the endpoint (or adjacent-to-head if not endpoint scenario)
                if !self.require_cycle {
                    // allow one as endpoint
                    if dead_ones > 1 {
                        // multiple dead-ends — only OK if remaining cells small; fail
                        if dead_ones > 2 { return false; }
                    }
                } else {
                    // cycle: all degrees must be >=2 ultimately; one dead-end of degree 1 in remaining is suspicious unless adj to head
                    if !adj_head { return false; }
                }
            }
        }

        let mut cands: Vec<(usize, usize)> = Vec::with_capacity(4);
        if let Some(f) = forced {
            // must go to f if adjacent
            let mut found_adj = false;
            for nb in self.neighbors(current).iter().flatten() {
                if *nb == f { found_adj = true; break; }
            }
            if !found_adj { return false; }
            cands.push((0, f));
        } else {
            for nb in self.neighbors(current).iter().flatten() {
                let v = *nb;
                if self.blocked[v] || self.visited[v] { continue; }
                let mut d = 0;
                for nn in self.neighbors(v).iter().flatten() {
                    let w = *nn;
                    if !self.blocked[w] && !self.visited[w] && w != current { d += 1; }
                }
                cands.push((d, v));
            }
            cands.sort_by_key(|x| x.0);

            self.visited[v] = true;
            self.path.push(v);
            if self.dfs(v) { return true; }
            self.path.pop();
            self.visited[v] = false;
            if self.found { return true; }
        }
        false
    }

struct SkylineBin {
    int W, H;
    vector<SkylineNode> skyline;

    void init(int w, int h) {
        W = w; H = h;
        skyline.clear();
        skyline.push_back({0, 0, w});
    }

    int fit(int i, int w, int h) const {
        int x = skyline[i].x;
        if (x + w > W) return -1;
        int widthLeft = w;
        int y = skyline[i].y;
        int j = i;
        while (widthLeft > 0) {
            if (j >= (int)skyline.size()) return -1;
            if (skyline[j].y > y) y = skyline[j].y;
            if (y + h > H) return -1;
            widthLeft -= skyline[j].width;
            j++;
        }
        return y;
    }

    void addLevel(int i, int x, int y, int w, int h) {
        SkylineNode newNode = {x, y + h, w};
        skyline.insert(skyline.begin() + i, newNode);
        for (int j = i + 1; j < (int)skyline.size();) {
            if (skyline[j].x < skyline[i].x + skyline[i].width) {
                int shrink = skyline[i].x + skyline[i].width - skyline[j].x;
                skyline[j].x += shrink;
                skyline[j].width -= shrink;
                if (skyline[j].width <= 0) {
                    skyline.erase(skyline.begin() + j);
                } else break;
            } else break;

        let x = next_f(); let y = next_f();
        stock.push(Pt{x,y});
    }

        let pv = next_f() as usize;

fn bbox(poly: &[Pt]) -> (f64,f64,f64,f64) {
    let mut mnx = f64::INFINITY; let mut mny = f64::INFINITY;

        if p.x < mnx { mnx = p.x; }
        if p.y < mny { mny = p.y; }
        if p.x > mxx { mxx = p.x; }
        if p.y > mxy { mxy = p.y; }

}

    uint64_t hh = a_hi*b_hi;
    uint64_t mid = lh + hl;
    uint64_t mid_carry = (mid < lh) ? (1ULL<<32) : 0;
    uint64_t mid_hi = (mid >> 32) | mid_carry;
    uint64_t mid_lo = mid << 32;
    uint64_t lo = ll + mid_lo;
    uint64_t carry = (lo < ll) ? 1 : 0;
    uint64_t hi = hh + mid_hi + carry;
    uint64_t r = (lo & MOD) + ((lo >> 61) | (hi << 3));
    if (r >= MOD) r -= MOD;
    if (r >= MOD) r -= MOD;
    return r;
}

static constexpr uint64_t MOD1 = (1ULL<<61) - 1;
static constexpr uint64_t MOD2 = 1000000007ULL;
static constexpr uint64_t B1 = 131;
static constexpr uint64_t B2 = 137;

static void compute_hashes(const string& s, uint64_t base, uint64_t mod,
                           vector<uint64_t>& H, vector<uint64_t>& P, bool use61) {
    size_t n = s.size();
    H.assign(n+1, 0);
    P.assign(n+1, 0);
    P[0] = 1;
    for (size_t i = 0; i < n; i++) {
        if (use61) {
            uint64_t hh = mulmod61(H[i], base);
            hh += (uint8_t)s[i] + 1;
            if (hh >= mod) hh -= mod;
            H[i+1] = hh;
            P[i+1] = mulmod61(P[i], base);
        } else {
            H[i+1] = (H[i]*base + (uint8_t)s[i] + 1) % mod;
            P[i+1] = (P[i]*base) % mod;
        }
    }
}

static inline uint64_t sub_hash(const vector<uint64_t>& H, const vector<uint64_t>& P,
                                size_t l, size_t len, uint64_t mod, bool use61) {
    if (use61) {
        uint64_t h = mulmod61(H[l], P[len]);
        uint64_t v = H[l+len] + mod - h;
        if (v >= mod) v -= mod;
        return v;
    } else {
        uint64_t h2 = (H[l]*P[len]) % mod;
        uint64_t v = (H[l+len] + mod - h2) % mod;
        return v;
    }
}

static bool try_length(const vector<string>& S, size_t L,
                       const vector<vector<uint64_t>>& H1,
                       const vector<vector<uint64_t>>& P1,
                       const vector<vector<uint64_t>>& H2,
                       const vector<vector<uint64_t>>& P2,
                       size_t& best_pos) {
    if (L == 0) { best_pos = 0; return true; }
    const string& s0 = S[0];
    if (s0.size() < L) return false;
    unordered_map<uint64_t, vector<pair<uint64_t,uint64_t>>> cur;
    cur.reserve(s0.size()-L+1);
    for (size_t i = 0; i + L <= s0.size(); i++) {
        uint64_t a = sub_hash(H1[0], P1[0], i, L, MOD1, true);
        uint64_t b = sub_hash(H2[0], P2[0], i, L, MOD2, false);
        cur[a].push_back({b, (uint64_t)i});
    }
    for (size_t k = 1; k < S.size(); k++) {
        const string& s = S[k];
        if (s.size() < L) return false;
        unordered_map<uint64_t, unordered_set<uint64_t>> mp;
        mp.reserve(s.size()-L+1);
        for (size_t i = 0; i + L <= s.size(); i++) {
            uint64_t a = sub_hash(H1[k], P1[k], i, L, MOD1, true);
            uint64_t b = sub_hash(H2[k], P2[k], i, L, MOD2, false);
            mp[a].insert(b);
        }
        unordered_map<uint64_t, vector<pair<uint64_t,uint64_t>>> next_;
        next_.reserve(cur.size());
        for (auto& kv : cur) {
            auto it = mp.find(kv.first);
            if (it == mp.end()) continue;
            for (auto& bp : kv.second) {
                if (it->second.count(bp.first)) {
                    next_[kv.first].push_back(bp);
                }
            }
        }
        if (next_.empty()) return false;
        cur = std::move(next_);
    }
    for (auto& kv : cur) {
        if (!kv.second.empty()) {
            best_pos = (size_t)kv.second.front().second;
            return true;
        }
    }
    return false;
}

static bool verify_substring(const vector<string>& S, const string& cand) {
    for (size_t k = 1; k < S.size(); k++) {
        if (S[k].find(cand) == string::npos) return false;
    }
    return true;
}

static string lcs_dp_two(const string& a, const string& b) {
    if (a.empty() || b.empty()) return "";
    size_t n = a.size(), m = b.size();
    vector<int> prev(m+1, 0), cur(m+1, 0);
    int best = 0;
    size_t end_i = 0;
    for (size_t i = 1; i <= n; i++) {
        for (size_t j = 1; j <= m; j++) {
            if (a[i-1] == b[j-1]) {
                cur[j] = prev[j-1] + 1;
                if (cur[j] > best) { best = cur[j]; end_i = i; }
            } else cur[j] = 0;
        }
        swap(prev, cur);
        fill(cur.begin(), cur.end(), 0);
    }
    if (best == 0) return "";
    return a.substr(end_i - best, best);
}

int main() {
    auto S = read_all_strings();
    int N = (int)S.size();
    if (N == 0) { cout << "\n"; return 0; }
    if (N == 1) { cout << S[0] << "\n"; return 0; }

    size_t minLen = SIZE_MAX;
    int minIdx = 0;
    for (int i = 0; i < N; i++) {
        if (S[i].size() < minLen) { minLen = S[i].size(); minIdx = i; }
    }
    if (minLen == 0) { cout << "\n"; return 0; }
    if (minIdx != 0) swap(S[0], S[minIdx]);

    if (N == 2 && (uint64_t)S[0].size() * (uint64_t)S[1].size() <= 50000000ULL) {
        cout << lcs_dp_two(S[0], S[1]) << "\n";
        return 0;
    }

    vector<vector<uint64_t>> H1(N), P1(N), H2(N), P2(N);
    for (int i = 0; i < N; i++) {
        compute_hashes(S[i], B1, MOD1, H1[i], P1[i], true);
        compute_hashes(S[i], B2, MOD2, H2[i], P2[i], false);
    }

    size_t lo = 1, hi = minLen;
    string best_str = "";
    while (lo <= hi) {
        size_t mid = lo + (hi - lo) / 2;
        size_t pos;
        bool ok = try_length(S, mid, H1, P1, H2, P2, pos);
        if (ok) {
            string cand = S[0].substr(pos, mid);
            if (verify_substring(S, cand)) {
                best_str = cand;
                lo = mid + 1;
            } else {
                hi = mid - 1;
            }
        } else {
            if (mid == 0) break;
            hi = mid - 1;
        }
        if (hi == (size_t)-1) break;
    }

    cout << best_str << "\n";
    return 0;
}
