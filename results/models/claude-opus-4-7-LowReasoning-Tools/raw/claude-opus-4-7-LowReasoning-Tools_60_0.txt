const fs = require('fs');

function main() {
  const data = JSON.parse(fs.readFileSync(0, 'utf8'));
  const machineCount = Math.max(1, data.machineCount | 0);
  const tasks = data.tasks || [];
  const n = tasks.length;

  const idx = new Map();
  for (let i = 0; i < n; i++) idx.set(tasks[i].id, i);

  const deps = new Array(n);
  const children = new Array(n);
  const indeg = new Int32Array(n);
  for (let i = 0; i < n; i++) { deps[i] = []; children[i] = []; }
  for (let i = 0; i < n; i++) {
    const d = tasks[i].deps || [];
    for (const dep of d) {
      if (idx.has(dep)) {
        const j = idx.get(dep);
        deps[i].push(j);
        children[j].push(i);
        indeg[i]++;
      }
    }
  }

  const order = [];
  const indeg2 = Int32Array.from(indeg);
  const q = [];
  for (let i = 0; i < n; i++) if (indeg2[i] === 0) q.push(i);
  let head = 0;
  while (head < q.length) {
    const v = q[head++];
    order.push(v);
    for (const c of children[v]) if (--indeg2[c] === 0) q.push(c);
  }

  const downstream = new Float64Array(n);
  for (let k = order.length - 1; k >= 0; k--) {
    const v = order[k];
    let m = 0;
    for (const c of children[v]) if (downstream[c] > m) m = downstream[c];
    downstream[v] = m + (tasks[v].duration || 0);
  }

  function priority(i, time) {
    const t = tasks[i];
    const w = t.weight || 1;
    const dl = (t.deadline != null) ? t.deadline : 1e15;
    const slack = dl - time - downstream[i];
    return slack / (w + 1) - downstream[i];
  }

  const machineFree = new Float64Array(machineCount);
  const finishTime = new Float64Array(n);
  const remainingDeps = Int32Array.from(indeg);
  const ready = [];
  for (let i = 0; i < n; i++) if (remainingDeps[i] === 0) ready.push(i);

  const schedule = [];
  let done = 0;

  while (done < n && ready.length > 0) {
    let mi = 0;
    for (let k = 1; k < machineCount; k++) if (machineFree[k] < machineFree[mi]) mi = k;
    const now = machineFree[mi];

    let bestReadyNow = -1, bestReadyNowScore = Infinity;
    let bestAny = -1, bestAnyEarliest = Infinity, bestAnyScore = Infinity;
    for (let k = 0; k < ready.length; k++) {
      const i = ready[k];
      let earliest = 0;
      for (const d of deps[i]) if (finishTime[d] > earliest) earliest = finishTime[d];
      const startIfNow = Math.max(now, earliest);
      const sc = priority(i, startIfNow);
      if (earliest <= now) {
        if (sc < bestReadyNowScore) { bestReadyNowScore = sc; bestReadyNow = k; }
      }
      if (earliest < bestAnyEarliest || (earliest === bestAnyEarliest && sc < bestAnyScore)) {
        bestAnyEarliest = earliest; bestAnyScore = sc; bestAny = k;
      }
    }

    const pickIdx = bestReadyNow >= 0 ? bestReadyNow : bestAny;
    const taskI = ready[pickIdx];
    ready[pickIdx] = ready[ready.length - 1];
    ready.pop();

    let earliest = 0;
    for (const d of deps[taskI]) if (finishTime[d] > earliest) earliest = finishTime[d];
    const start = Math.max(now, earliest);
    const dur = tasks[taskI].duration || 0;
    const finish = start + dur;
    finishTime[taskI] = finish;
    machineFree[mi] = finish;
    done++;
    schedule.push({ id: tasks[taskI].id, start: Math.round(start), machine: mi });

    for (const c of children[taskI]) if (--remainingDeps[c] === 0) ready.push(c);
  }

  process.stdout.write(JSON.stringify({ schedule }));
}

main();
