using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static int XC, YC, ZC;
    static double BX, BY, BZ;
    static byte[,,] data;
    static bool[,,] solid;
    static int[,] topZ;

    static void Main()
    {
        var stdin = Console.OpenStandardInput();
        var br = new BinaryReader(stdin);
        var sb = new StringBuilder();
        int b;
        while ((b = br.ReadByte()) != '\n')
        {
            if (b == -1) break;
            if (b == '\r') continue;
            sb.Append((char)b);
        }
        var parts = sb.ToString().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        XC = int.Parse(parts[0]);
        YC = int.Parse(parts[1]);
        ZC = int.Parse(parts[2]);
        BX = double.Parse(parts[3], System.Globalization.CultureInfo.InvariantCulture);
        BY = double.Parse(parts[4], System.Globalization.CultureInfo.InvariantCulture);
        BZ = double.Parse(parts[5], System.Globalization.CultureInfo.InvariantCulture);

        long totalVox = (long)XC * YC * ZC;
        var sw = new StreamWriter(Console.OpenStandardOutput());

        if (totalVox > 200_000_000L)
        {
            sw.WriteLine(0);
            sw.Flush();
            return;
        }

        data = new byte[XC, YC, ZC];
        solid = new bool[XC, YC, ZC];
        topZ = new int[XC, YC];

        for (int x = 0; x < XC; x++)
            for (int y = 0; y < YC; y++)
            {
                int top = 0;
                for (int z = 0; z < ZC; z++)
                {
                    int v = br.ReadByte();
                    if (v < 0) v = 0;
                    data[x, y, z] = (byte)v;
                    int mass = v & 0xF;
                    if (mass > 0)
                    {
                        solid[x, y, z] = true;
                        if (z + 1 > top) top = z + 1;
                    }
                }
                topZ[x, y] = top;
            }

        var ops = new List<string>();
        var cols = new List<(int x, int y, int dist)>();
        for (int x = 0; x < XC; x++)
            for (int y = 0; y < YC; y++)
                cols.Add((x, y, x + y));
        cols.Sort((a, c) => a.dist.CompareTo(c.dist));

        int maxOps = 100000;
        foreach (var (cx, cy, _) in cols)
        {
            if (ops.Count >= maxOps) break;
            if (cx == 0 && cy == 0) continue;
            while (topZ[cx, cy] > 0)
            {
                int z = topZ[cx, cy] - 1;
                if (!solid[cx, cy, z]) { topZ[cx, cy] = z; continue; }
                byte v = data[cx, cy, z];
                int mass = v & 0xF;
                int val = (v >> 4) & 0xF;
                if (val == 0) break;
                if (mass == 0) { solid[cx, cy, z] = false; topZ[cx, cy] = z; continue; }

                var steps = new List<(int sx, int sy)>();
                int px = cx, py = cy;
                int pz = topZ[px, py];
                bool ok = true;
                while (px != 0 || py != 0)
                {
                    int nx = px, ny = py;
                    if (px > 0) nx = px - 1;
                    else if (py > 0) ny = py - 1;
                    int nz = topZ[nx, ny];
                    if (Math.Abs(nz - pz) > 1) { ok = false; break; }
                    steps.Add((nx, ny));
                    px = nx; py = ny; pz = nz;
                    if (steps.Count > XC + YC + 10) { ok = false; break; }
                }
                if (!ok) break;

                var sob = new StringBuilder();
                sob.Append(steps.Count).Append(' ').Append(cx).Append(' ').Append(cy).Append(' ').Append(z);
                foreach (var s in steps) sob.Append(' ').Append(s.sx).Append(' ').Append(s.sy);
                sob.Append(' ').Append(0).Append(' ').Append(0);
                ops.Add(sob.ToString());

                solid[cx, cy, z] = false;
                topZ[cx, cy] = z;

                if (ops.Count >= maxOps) break;
            }
        }

        sw.WriteLine(ops.Count);
        foreach (var o in ops) sw.WriteLine(o);
        sw.Flush();
    }
}
