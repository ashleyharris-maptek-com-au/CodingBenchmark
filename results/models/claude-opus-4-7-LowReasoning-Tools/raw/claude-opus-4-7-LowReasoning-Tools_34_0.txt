import sys
import math


def dist(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_distance(route, depot, coords):
    if not route:
        return 0.0
    d = dist(depot, coords[route[0]])
    for i in range(len(route) - 1):
        d += dist(coords[route[i]], coords[route[i + 1]])
    d += dist(coords[route[-1]], depot)
    return d


def total_distance(routes, depot, coords):
    return sum(route_distance(r, depot, coords) for r in routes)


def clarke_wright(N, capacity, depot, coords, demands):
    routes = {i: [i] for i in range(N)}
    route_of = {i: i for i in range(N)}
    route_demand = {i: demands[i] for i in range(N)}

    d0 = [dist(depot, coords[i]) for i in range(N)]
    savings = []
    for i in range(N):
        for j in range(i + 1, N):
            s = d0[i] + d0[j] - dist(coords[i], coords[j])
            savings.append((s, i, j))
    savings.sort(reverse=True)

    for s, i, j in savings:
        ri = route_of[i]
        rj = route_of[j]
        if ri == rj:
            continue
        if route_demand[ri] + route_demand[rj] > capacity + 1e-9:
            continue
        route_i = routes[ri]
        route_j = routes[rj]

        if route_i[-1] == i and route_j[0] == j:
            new_route = route_i + route_j
        elif route_i[0] == i and route_j[-1] == j:
            new_route = route_j + route_i
        elif route_i[0] == i and route_j[0] == j:
            new_route = list(reversed(route_i)) + route_j
        elif route_i[-1] == i and route_j[-1] == j:
            new_route = route_i + list(reversed(route_j))
        else:
            continue

        routes[ri] = new_route
        route_demand[ri] = route_demand[ri] + route_demand[rj]
        del routes[rj]
        del route_demand[rj]
        for c in new_route:
            route_of[c] = ri

    return list(routes.values())


def two_opt(route, depot, coords, max_iter=100):
    if len(route) < 2:
        return route
    r = list(route)
    n = len(r)
    improved = True
    it = 0
    while improved and it < max_iter:
        improved = False
        it += 1
        for i in range(n - 1):
            for j in range(i + 1, n):
                a = depot if i == 0 else coords[r[i - 1]]
                b = coords[r[i]]
                c = coords[r[j]]
                d = depot if j == n - 1 else coords[r[j + 1]]
                d1 = dist(a, b) + dist(c, d)
                d2 = dist(a, c) + dist(b, d)
                if d2 + 1e-12 < d1:
                    r[i:j + 1] = r[i:j + 1][::-1]
                    improved = True
    return r


def solve():
    data = sys.stdin.read().split()
    idx = 0
    N = int(data[idx]); idx += 1
    K = int(data[idx]); idx += 1
    capacity = float(data[idx]); idx += 1
    depot = (float(data[idx]), float(data[idx + 1])); idx += 2
    coords = []
    demands = []
    for _ in range(N):
        x = float(data[idx]); idx += 1
        y = float(data[idx]); idx += 1
        dm = float(data[idx]); idx += 1
        coords.append((x, y))
        demands.append(dm)

    routes = clarke_wright(N, capacity, depot, coords, demands)

    # Try to merge if too many routes
    while len(routes) > K:
        best = None
        for i in range(len(routes)):
            for j in range(len(routes)):
                if i == j:
                    continue
                di = sum(demands[c] for c in routes[i])
                dj = sum(demands[c] for c in routes[j])
                if di + dj > capacity + 1e-9:
                    continue
                merged = two_opt(routes[i] + routes[j], depot, coords, max_iter=20)
                extra = (route_distance(merged, depot, coords)
                         - route_distance(routes[i], depot, coords)
                         - route_distance(routes[j], depot, coords))
                if best is None or extra < best[0]:
                    best = (extra, i, j, merged)
        if best is None:
            break
        _, i, j, merged = best
        routes = [r for k, r in enumerate(routes) if k != i and k != j] + [merged]

    routes = [two_opt(r, depot, coords, max_iter=100) for r in routes]

    total = total_distance(routes, depot, coords)
    out = [f"{total:.6f}", str(len(routes))]
    for r in routes:
        out.append(" ".join(str(c) for c in r))
    sys.stdout.write("\n".join(out) + "\n")


if __name__ == "__main__":
    solve()
