using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        using var sin = new StreamReader(Console.OpenStandardInput(), Encoding.ASCII, false, 1 << 20);
        using var sout = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1 << 20);
        sout.NewLine = "\n";

        string all = sin.ReadToEnd();
        int pos = 0;
        int len = all.Length;

        long ReadLong()
        {
            while (pos < len && (all[pos] == ' ' || all[pos] == '\n' || all[pos] == '\r' || all[pos] == '\t')) pos++;
            long v = 0; bool neg = false;
            if (pos < len && all[pos] == '-') { neg = true; pos++; }
            while (pos < len)
            {
                char c = all[pos];
                if (c < '0' || c > '9') break;
                v = v * 10 + (c - '0');
                pos++;
            }
            return neg ? -v : v;
        }

        int N = (int)ReadLong();
        long stockLen = ReadLong();
        long[] cuts = new long[N];
        for (int i = 0; i < N; i++) cuts[i] = ReadLong();

        int[] idx = new int[N];
        for (int i = 0; i < N; i++) idx[i] = i;
        Array.Sort(idx, (a, b) => cuts[b].CompareTo(cuts[a]));

        int cap = 1;
        while (cap < Math.Max(1, N)) cap <<= 1;
        long[] tree = new long[2 * cap];
        for (int i = 0; i < 2 * cap; i++) tree[i] = -1;

        int numStocks = 0;
        List<List<int>> assignments = new List<List<int>>();
        long[] remaining = new long[N];

        void Update(int p, long val)
        {
            p += cap;
            tree[p] = val;
            p >>= 1;
            while (p > 0)
            {
                long l = tree[2 * p], r = tree[2 * p + 1];
                tree[p] = l >= r ? l : r;
                p >>= 1;
            }
        }

        int Query(long need)
        {
            int p = 1;
            if (tree[p] < need) return -1;
            while (p < cap)
            {
                if (tree[2 * p] >= need) p = 2 * p;
                else p = 2 * p + 1;
            }
            return p - cap;
        }

        for (int k = 0; k < N; k++)
        {
            int ci = idx[k];
            long c = cuts[ci];
            if (c > stockLen) continue;
            int s = Query(c);
            if (s == -1)
            {
                s = numStocks++;
                assignments.Add(new List<int>());
                remaining[s] = stockLen - c;
                Update(s, remaining[s]);
            }
            else
            {
                remaining[s] -= c;
                Update(s, remaining[s]);
            }
            assignments[s].Add(ci);
        }

        var sb = new StringBuilder();
        sb.Append(numStocks).Append('\n');
        for (int i = 0; i < numStocks; i++)
        {
            var lst = assignments[i];
            for (int j = 0; j < lst.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(lst[j]);
            }
            sb.Append('\n');
        }
        sout.Write(sb.ToString());
    }
}
