use std::io::{self, Read, Write, BufWriter};
use std::collections::BinaryHeap;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let n: usize = match it.next() { Some(s) => s.parse().unwrap_or(0), None => 0 };
    let m: usize = match it.next() { Some(s) => s.parse().unwrap_or(0), None => 0 };

    if n == 0 {
        writeln!(out, "0").unwrap();
        writeln!(out).unwrap();
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = match it.next() { Some(s) => s.parse().unwrap_or(0), None => break };
        let v: usize = match it.next() { Some(s) => s.parse().unwrap_or(0), None => break };
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }
    for a in adj.iter_mut() {
        a.sort_unstable();
        a.dedup();
    }

    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];
    let mut remaining = n;

    let mut heap: BinaryHeap<(usize, usize, u64)> = BinaryHeap::new();
    let mut version: Vec<u64> = vec![0; n];
    let mut score: Vec<usize> = vec![0; n];

    for v in 0..n {
        let s = adj[v].len() + 1;
        score[v] = s;
        heap.push((s, v, 0));
    }

    let mut result: Vec<usize> = Vec::new();

    while remaining > 0 {
        let mut best: Option<usize> = None;
        while let Some((s, v, ver)) = heap.pop() {
            if ver != version[v] { continue; }
            if in_set[v] { continue; }
            if s != score[v] { continue; }
            if s == 0 { break; }
            best = Some(v);
            break;
        }

        let pick = match best {
            Some(v) => v,
            None => {
                let mut found = None;
                for v in 0..n {
                    if !dominated[v] { found = Some(v); break; }
                }
                match found { Some(v) => v, None => break }
            }
        };

        in_set[pick] = true;
        result.push(pick);

        let mut newly: Vec<usize> = Vec::new();
        if !dominated[pick] {
            dominated[pick] = true;
            remaining -= 1;
            newly.push(pick);
        }
        for &u in &adj[pick] {
            if !dominated[u] {
                dominated[u] = true;
                remaining -= 1;
                newly.push(u);
            }
        }

        for &w in &newly {
            if !in_set[w] && score[w] > 0 {
                score[w] -= 1;
                version[w] += 1;
                heap.push((score[w], w, version[w]));
            }
            for &x in &adj[w] {
                if !in_set[x] && score[x] > 0 {
                    score[x] -= 1;
                    version[x] += 1;
                    heap.push((score[x], x, version[x]));
                }
            }
        }
    }

    writeln!(out, "{}", result.len()).unwrap();
    let strs: Vec<String> = result.iter().map(|v| v.to_string()).collect();
    writeln!(out, "{}", strs.join(" ")).unwrap();
}
