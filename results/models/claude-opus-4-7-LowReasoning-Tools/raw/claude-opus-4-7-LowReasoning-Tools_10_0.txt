#include <cstdio>
#include <cstdlib>
#include <cstdint>
#include <vector>
#include <algorithm>
#include <chrono>
#include <climits>

using namespace std;

struct Rect { int w, h, idx; };
struct Placement { int board, x, y, rotated; };
struct SkylineNode { int x, y, width; };

struct SkylineBin {
    int W, H;
    vector<SkylineNode> skyline;

    void init(int w, int h) {
        W = w; H = h;
        skyline.clear();
        skyline.push_back({0, 0, w});
    }

    int fit(int i, int w, int h) const {
        int x = skyline[i].x;
        if (x + w > W) return -1;
        int widthLeft = w;
        int y = skyline[i].y;
        int j = i;
        while (widthLeft > 0) {
            if (j >= (int)skyline.size()) return -1;
            if (skyline[j].y > y) y = skyline[j].y;
            if (y + h > H) return -1;
            widthLeft -= skyline[j].width;
            j++;
        }
        return y;
    }

    void addLevel(int i, int x, int y, int w, int h) {
        SkylineNode newNode = {x, y + h, w};
        skyline.insert(skyline.begin() + i, newNode);
        for (int j = i + 1; j < (int)skyline.size();) {
            if (skyline[j].x < skyline[i].x + skyline[i].width) {
                int shrink = skyline[i].x + skyline[i].width - skyline[j].x;
                skyline[j].x += shrink;
                skyline[j].width -= shrink;
                if (skyline[j].width <= 0) {
                    skyline.erase(skyline.begin() + j);
                } else break;
            } else break;
        }
        for (int j = 0; j + 1 < (int)skyline.size();) {
            if (skyline[j].y == skyline[j+1].y) {
                skyline[j].width += skyline[j+1].width;
                skyline.erase(skyline.begin() + j + 1);
            } else j++;
        }
    }

    bool tryPlace(int rw, int rh, bool allowRot,
                  int &outIdx, int &outY, int &outW, int &outH, int &outRot) {
        int bestTop = INT_MAX, bestSW = INT_MAX, bestIdx = -1;
        int bestPW = 0, bestPH = 0, bestRot = 0;
        for (int i = 0; i < (int)skyline.size(); i++) {
            if (rw <= W) {
                int y = fit(i, rw, rh);
                if (y >= 0) {
                    int top = y + rh;
                    if (top < bestTop || (top == bestTop && skyline[i].width < bestSW)) {
                        bestTop = top; bestSW = skyline[i].width; bestIdx = i;
                        bestPW = rw; bestPH = rh; bestRot = 0;
                    }
                }
            }
            if (allowRot && rw != rh && rh <= W) {
                int y2 = fit(i, rh, rw);
                if (y2 >= 0) {
                    int top = y2 + rw;
                    if (top < bestTop || (top == bestTop && skyline[i].width < bestSW)) {
                        bestTop = top; bestSW = skyline[i].width; bestIdx = i;
                        bestPW = rh; bestPH = rw; bestRot = 1;
                    }
                }
            }
        }
        if (bestIdx < 0) return false;
        outIdx = bestIdx;
        outY = bestTop - bestPH;
        outW = bestPW;
        outH = bestPH;
        outRot = bestRot;
        return true;
    }
};

int main() {
    int N, BW, BH;
    if (scanf("%d %d %d", &N, &BW, &BH) != 3) return 0;

    vector<Rect> rects(N);
    for (int i = 0; i < N; i++) {
        scanf("%d %d", &rects[i].w, &rects[i].h);
        rects[i].idx = i;
    }

    vector<Rect> sorted_rects = rects;
    sort(sorted_rects.begin(), sorted_rects.end(), [](const Rect&a, const Rect&b){
        int ma = max(a.w, a.h), mb = max(b.w, b.h);
        if (ma != mb) return ma > mb;
        return (a.w * a.h) > (b.w * b.h);
    });

    vector<Placement> result(N);
    vector<SkylineBin> bins;

    auto t0 = chrono::steady_clock::now();
    auto elapsed = [&](){
        return chrono::duration<double>(chrono::steady_clock::now() - t0).count();
    };

    for (int k = 0; k < N; k++) {
        Rect r = sorted_rects[k];
        int rw = r.w, rh = r.h;

        int bestBin = -1, bestIdx = -1, bestTop = INT_MAX, bestSW = INT_MAX;
        int bestPW = 0, bestPH = 0, bestRot = 0, bestX = 0;

        int startBin = 0;
        if (N > 10000 && (int)bins.size() > 4) startBin = (int)bins.size() - 4;
        else if (N > 2000 && (int)bins.size() > 16) startBin = (int)bins.size() - 16;

        for (int b = startBin; b < (int)bins.size(); b++) {
            int idx, y, pw, ph, rot;
            if (bins[b].tryPlace(rw, rh, true, idx, y, pw, ph, rot)) {
                int top = y + ph;
                int sw = bins[b].skyline[idx].width;
                if (top < bestTop || (top == bestTop && sw < bestSW)) {
                    bestTop = top; bestSW = sw; bestBin = b;
                    bestIdx = idx; bestPW = pw; bestPH = ph; bestRot = rot;
                    bestX = bins[b].skyline[idx].x;
                }
            }
        }

        if (bestBin < 0) {
            SkylineBin nb;
            nb.init(BW, BH);
            int idx, y, pw, ph, rot;
            if (!nb.tryPlace(rw, rh, true, idx, y, pw, ph, rot)) {
                result[r.idx] = {0, 0, 0, 0};
                continue;
            }
            int x = nb.skyline[idx].x;
            nb.addLevel(idx, x, y, pw, ph);
            bins.push_back(std::move(nb));
            result[r.idx] = {(int)bins.size() - 1, x, y, rot};
        } else {
            bins[bestBin].addLevel(bestIdx, bestX, bestTop - bestPH, bestPW, bestPH);
            result[r.idx] = {bestBin, bestX, bestTop - bestPH, bestRot};
        }

        if (elapsed() > 25.0) {
            for (int kk = k + 1; kk < N; kk++) {
                Rect rr = sorted_rects[kk];
                SkylineBin nb;
                nb.init(BW, BH);
                int idx, y, pw, ph, rot;
                if (!nb.tryPlace(rr.w, rr.h, true, idx, y, pw, ph, rot)) {
                    result[rr.idx] = {0,0,0,0}; continue;
                }
                int x = nb.skyline[idx].x;
                nb.addLevel(idx, x, y, pw, ph);
                bins.push_back(std::move(nb));
                result[rr.idx] = {(int)bins.size() - 1, x, y, rot};
            }
            break;
        }
    }

    printf("%d\n", (int)bins.size());
    for (int i = 0; i < N; i++) {
        printf("%d %d %d %d\n", result[i].board, result[i].x, result[i].y, result[i].rotated);
    }
    return 0;
}
