use std::io::{self, Read, Write, BufWriter};

#[derive(Clone, Copy, Debug)]
struct Pt { x: f64, y: f64 }

fn read_input() -> (Vec<Pt>, Vec<Vec<Pt>>) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_ascii_whitespace();
    let mut next_f = || -> f64 { it.next().unwrap().parse().unwrap() };
    let sv = next_f() as usize;
    let np = next_f() as usize;
    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x = next_f(); let y = next_f();
        stock.push(Pt{x,y});
    }
    let mut pieces = Vec::with_capacity(np);
    for _ in 0..np {
        let pv = next_f() as usize;
        let mut p = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x = next_f(); let y = next_f();
            p.push(Pt{x,y});
        }
        pieces.push(p);
    }
    (stock, pieces)
}

fn bbox(poly: &[Pt]) -> (f64,f64,f64,f64) {
    let mut mnx = f64::INFINITY; let mut mny = f64::INFINITY;
    let mut mxx = f64::NEG_INFINITY; let mut mxy = f64::NEG_INFINITY;
    for p in poly {
        if p.x < mnx { mnx = p.x; }
        if p.y < mny { mny = p.y; }
        if p.x > mxx { mxx = p.x; }
        if p.y > mxy { mxy = p.y; }
    }
    (mnx,mny,mxx,mxy)
}

fn main() {
    let (stock, pieces) = read_input();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let (smnx, smny, smxx, smxy) = bbox(&stock);
    let sw = smxx - smnx;
    let sh = smxy - smny;

    let n = pieces.len();
    let mut order: Vec<usize> = (0..n).collect();
    let pb: Vec<(f64,f64,f64,f64)> = pieces.iter().map(|p| bbox(p)).collect();
    order.sort_by(|&a, &b| {
        let ha = pb[a].3 - pb[a].1;
        let hb = pb[b].3 - pb[b].1;
        hb.partial_cmp(&ha).unwrap()
    });

    struct Stock { shelves: Vec<(f64,f64,f64)>, used_h: f64 }
    let mut stocks: Vec<Stock> = Vec::new();
    let mut placements: Vec<(usize, f64, f64, f64)> = vec![(0,0.0,0.0,0.0); n];

    for &i in &order {
        let (mnx, mny, mxx, mxy) = pb[i];
        let pw = mxx - mnx;
        let ph = mxy - mny;
        if pw > sw || ph > sh {
            stocks.push(Stock{ shelves: vec![(0.0, ph, pw)], used_h: ph });
            placements[i] = (stocks.len()-1, smnx - mnx, smny - mny, 0.0);
            continue;
        }
        let mut placed = false;
        for (si, st) in stocks.iter_mut().enumerate() {
            for sh_idx in 0..st.shelves.len() {
                let (y_off, sh_h, used_w) = st.shelves[sh_idx];
                if ph <= sh_h && used_w + pw <= sw {
                    let tx = smnx + used_w - mnx;
                    let ty = smny + y_off - mny;
                    st.shelves[sh_idx].2 = used_w + pw;
                    placements[i] = (si, tx, ty, 0.0);
                    placed = true;
                    break;
                }
            }
            if placed { break; }
            if st.used_h + ph <= sh {
                let y_off = st.used_h;
                let tx = smnx - mnx;
                let ty = smny + y_off - mny;
                st.shelves.push((y_off, ph, pw));
                st.used_h += ph;
                placements[i] = (si, tx, ty, 0.0);
                placed = true;
                break;
            }
        }
        if !placed {
            stocks.push(Stock{ shelves: vec![(0.0, ph, pw)], used_h: ph });
            let tx = smnx - mnx;
            let ty = smny - mny;
            placements[i] = (stocks.len()-1, tx, ty, 0.0);
        }
    }

    writeln!(out, "{}", stocks.len()).unwrap();
    for i in 0..n {
        let (si, tx, ty, r) = placements[i];
        writeln!(out, "{} {} {} {}", si, tx, ty, r).unwrap();
    }
}
