; SPIR-V
; Version: 1.0
               OpCapability Shader
       %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4f 4

        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
   %_ptr_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_ptr_UBO Uniform

 %_ptr_in_v3 = OpTypePointer Input %v3f
 %_ptr_in_v2 = OpTypePointer Input %v2f
%_ptr_out_v4 = OpTypePointer Output %v4f
  %_ptr_u_v4 = OpTypePointer Uniform %v4f

   %worldPos = OpVariable %_ptr_in_v3 Input
     %normal = OpVariable %_ptr_in_v3 Input
         %uv = OpVariable %_ptr_in_v2 Input
    %tangent = OpVariable %_ptr_in_v3 Input
      %color = OpVariable %_ptr_in_v3 Input
   %outColor = OpVariable %_ptr_out_v4 Output

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3

        %f_0 = OpConstant %float 0.0
        %f_1 = OpConstant %float 1.0
       %f_04 = OpConstant %float 0.4
       %f_06 = OpConstant %float 0.6
     %f_pi16 = OpConstant %float 50.26548
       %f_09 = OpConstant %float 0.9
       %f_01 = OpConstant %float 0.1
      %red_c = OpConstantComposite %v3f %f_09 %f_01 %f_01
     %blue_c = OpConstantComposite %v3f %f_01 %f_01 %f_09

       %bool = OpTypeBool

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

       %uv_v = OpLoad %v2f %uv
        %uvy = OpCompositeExtract %float %uv_v 1
        %arg = OpFMul %float %uvy %f_pi16
          %s = OpExtInst %float %glsl Sin %arg

        %n_v = OpLoad %v3f %normal
        %n_n = OpExtInst %v3f %glsl Normalize %n_v

         %wp = OpLoad %v3f %worldPos
     %lp_ptr = OpAccessChain %_ptr_u_v4 %ubo %int_3
        %lp4 = OpLoad %v4f %lp_ptr
        %lp3 = OpVectorShuffle %v3f %lp4 %lp4 0 1 2
       %ldir = OpFSub %v3f %lp3 %wp
        %ldn = OpExtInst %v3f %glsl Normalize %ldir

         %dt = OpDot %float %n_n %ldn
       %diff = OpExtInst %float %glsl FMax %dt %f_0
        %mul = OpFMul %float %f_06 %diff
         %lf = OpFAdd %float %f_04 %mul

        %cmp = OpFOrdGreaterThan %bool %s %f_0
       %base = OpSelect %v3f %cmp %red_c %blue_c

     %shaded = OpVectorTimesScalar %v3f %base %lf
         %rr = OpCompositeExtract %float %shaded 0
         %gg = OpCompositeExtract %float %shaded 1
         %bb = OpCompositeExtract %float %shaded 2
      %final = OpCompositeConstruct %v4f %rr %gg %bb %f_1
               OpStore %outColor %final
               OpReturn
               OpFunctionEnd
