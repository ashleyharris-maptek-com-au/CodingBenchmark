using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

class Solver
{
    static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        var sb = new StringBuilder();
        using (var sr = new StreamReader(Console.OpenStandardInput()))
        {
            string allText = sr.ReadToEnd();
            var tokens = allText.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
            int idx = 0;
            int V = int.Parse(tokens[idx++], ci);
            int F = int.Parse(tokens[idx++], ci);
            double[] xs = new double[V];
            double[] ys = new double[V];
            double[] zs = new double[V];
            for (int i = 0; i < V; i++)
            {
                xs[i] = double.Parse(tokens[idx++], ci);
                ys[i] = double.Parse(tokens[idx++], ci);
                zs[i] = double.Parse(tokens[idx++], ci);
            }
            for (int i = 0; i < F; i++) { idx += 3; }
            double W = double.Parse(tokens[idx++], ci);
            double H = double.Parse(tokens[idx++], ci);
            double D = double.Parse(tokens[idx++], ci);

            double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
            double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;
            for (int i = 0; i < V; i++)
            {
                if (xs[i] < minX) minX = xs[i];
                if (ys[i] < minY) minY = ys[i];
                if (zs[i] < minZ) minZ = zs[i];
                if (xs[i] > maxX) maxX = xs[i];
                if (ys[i] > maxY) maxY = ys[i];
                if (zs[i] > maxZ) maxZ = zs[i];
            }
            double bx = maxX - minX;
            double by = maxY - minY;
            double bz = maxZ - minZ;
            if (bx <= 0) bx = 1e-9;
            if (by <= 0) by = 1e-9;
            if (bz <= 0) bz = 1e-9;

            double[] dims = { bx, by, bz };

            double[][] quats = new double[6][];
            quats[0] = new double[] { 1, 0, 0, 0 };
            quats[1] = new double[] { 0.7071067811865476, 0.7071067811865476, 0, 0 };
            quats[2] = new double[] { 0.7071067811865476, 0, 0, 0.7071067811865476 };
            quats[3] = new double[] { 0.5, 0.5, 0.5, 0.5 };
            quats[4] = new double[] { 0.5, -0.5, -0.5, -0.5 };
            quats[5] = new double[] { 0.7071067811865476, 0, 0.7071067811865476, 0 };

            int[][] permMap = new int[][]
            {
                new[]{0,1,2},
                new[]{0,2,1},
                new[]{1,0,2},
                new[]{2,0,1},
                new[]{1,2,0},
                new[]{2,1,0},
            };

            long bestCount = -1;
            int bestPerm = 0;
            long bestNx = 0, bestNy = 0, bestNz = 0;
            for (int p = 0; p < 6; p++)
            {
                double dx = dims[permMap[p][0]];
                double dy = dims[permMap[p][1]];
                double dz = dims[permMap[p][2]];
                long nx = (long)Math.Floor(W / dx);
                long ny = (long)Math.Floor(H / dy);
                long nz = (long)Math.Floor(D / dz);
                if (nx < 0) nx = 0; if (ny < 0) ny = 0; if (nz < 0) nz = 0;
                long count;
                try { count = checked(nx * ny * nz); }
                catch { count = long.MaxValue; }
                if (count > bestCount)
                {
                    bestCount = count;
                    bestPerm = p;
                    bestNx = nx; bestNy = ny; bestNz = nz;
                }
            }

            long maxOut = 2_000_000L;
            long total = bestCount;
            if (total > maxOut) total = maxOut;

            double[] q = quats[bestPerm];
            double qw = q[0], qx = q[1], qy = q[2], qz = q[3];
            int pp = bestPerm;
            double pdx = dims[permMap[pp][0]];
            double pdy = dims[permMap[pp][1]];
            double pdz = dims[permMap[pp][2]];

            double rMinX = double.PositiveInfinity, rMinY = double.PositiveInfinity, rMinZ = double.PositiveInfinity;
            for (int i = 0; i < V; i++)
            {
                double px = xs[i], py = ys[i], pz = zs[i];
                double tx = 2 * (qy * pz - qz * py);
                double ty = 2 * (qz * px - qx * pz);
                double tz = 2 * (qx * py - qy * px);
                double rx = px + qw * tx + (qy * tz - qz * ty);
                double ry = py + qw * ty + (qz * tx - qx * tz);
                double rz = pz + qw * tz + (qx * ty - qy * tx);
                if (rx < rMinX) rMinX = rx;
                if (ry < rMinY) rMinY = ry;
                if (rz < rMinZ) rMinZ = rz;
            }
            double offX = -rMinX;
            double offY = -rMinY;
            double offZ = -rMinZ;

            sb.Append(total).Append('\n');
            long emitted = 0;
            string qStr = " " + qw.ToString("R", ci) + " " + qx.ToString("R", ci) + " " + qy.ToString("R", ci) + " " + qz.ToString("R", ci) + "\n";
            for (long k = 0; k < bestNz && emitted < total; k++)
            {
                double cz = k * pdz + offZ;
                for (long j = 0; j < bestNy && emitted < total; j++)
                {
                    double cy = j * pdy + offY;
                    for (long i = 0; i < bestNx && emitted < total; i++)
                    {
                        double cx = i * pdx + offX;
                        sb.Append(cx.ToString("R", ci)).Append(' ');
                        sb.Append(cy.ToString("R", ci)).Append(' ');
                        sb.Append(cz.ToString("R", ci));
                        sb.Append(qStr);
                        emitted++;
                    }
                }
            }
            Console.Out.Write(sb.ToString());
        }
    }
}
