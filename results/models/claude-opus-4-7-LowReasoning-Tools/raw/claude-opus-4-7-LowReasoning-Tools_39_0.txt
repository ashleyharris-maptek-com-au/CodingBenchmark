use std::io::{self, Read, Write, BufWriter};

fn read_input() -> (usize, Vec<Vec<usize>>, Vec<(usize,usize)>) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_ascii_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let mut adj = vec![Vec::new(); n];
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
            edges.push((u,v));
        }
    }
    (n, adj, edges)
}

fn cut_size(part: &[u8], edges: &[(usize,usize)]) -> i64 {
    let mut c = 0i64;
    for &(u,v) in edges {
        if part[u] != part[v] { c += 1; }
    }
    c
}

fn kl_improve(part: &mut Vec<u8>, n: usize, adj: &[Vec<usize>], edges: &[(usize,usize)]) -> i64 {
    loop {
        let mut d = vec![0i64; n];
        for v in 0..n {
            for &u in &adj[v] {
                if part[u] != part[v] { d[v] += 1; } else { d[v] -= 1; }
            }
        }
        let mut locked = vec![false; n];
        let mut gains: Vec<i64> = Vec::new();
        let mut swaps: Vec<(usize,usize)> = Vec::new();
        let half = n / 2;

        for _ in 0..half {
            let mut best_gain = i64::MIN;
            let mut best_a = usize::MAX;
            let mut best_b = usize::MAX;
            for a in 0..n {
                if locked[a] || part[a] != 0 { continue; }
                for b in 0..n {
                    if locked[b] || part[b] != 1 { continue; }
                    let mut g = d[a] + d[b];
                    let ab_count = adj[a].iter().filter(|&&x| x == b).count() as i64;
                    if ab_count > 0 { g -= 2 * ab_count; }
                    if g > best_gain {
                        best_gain = g;
                        best_a = a;
                        best_b = b;
                    }
                }
            }
            if best_a == usize::MAX { break; }
            gains.push(best_gain);
            swaps.push((best_a, best_b));
            locked[best_a] = true;
            locked[best_b] = true;
            for v in 0..n {
                if locked[v] { continue; }
                let cnt_a = adj[v].iter().filter(|&&x| x == best_a).count() as i64;
                let cnt_b = adj[v].iter().filter(|&&x| x == best_b).count() as i64;
                let mut delta = 0i64;
                if cnt_a > 0 {
                    if part[v] == 0 { delta += 2 * cnt_a; } else { delta -= 2 * cnt_a; }
                }
                if cnt_b > 0 {
                    if part[v] == 1 { delta += 2 * cnt_b; } else { delta -= 2 * cnt_b; }
                }
                d[v] += delta;
            }
        }

        let mut best_sum = 0i64;
        let mut cur_sum = 0i64;
        let mut best_k = 0;
        for (i, &g) in gains.iter().enumerate() {
            cur_sum += g;
            if cur_sum > best_sum {
                best_sum = cur_sum;
                best_k = i + 1;
            }
        }

        if best_sum <= 0 {
            return cut_size(part, edges);
        }

        for i in 0..best_k {
            let (a,b) = swaps[i];
            part[a] = 1;
            part[b] = 0;
        }
    }
}

fn main() {
    let (n, adj, edges) = read_input();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    if n == 0 {
        writeln!(out, "0").unwrap();
        writeln!(out, "").unwrap();
        return;
    }

    let half = n / 2;

    // Brute force for small n (fix vertex 0 in part A)
    if n <= 22 {
        let mut best_cut = i64::MAX;
        let mut best_part: Vec<u8> = vec![0; n];

        if n == 2 {
            let mut part = vec![0u8; n];
            part[1] = 1;
            best_cut = cut_size(&part, &edges);
            best_part = part;
        } else {
            let rest: Vec<usize> = (1..n).collect();
            let k = half - 1;
            let total = rest.len();
            // generate combinations of size k from rest
            let mut indices: Vec<usize> = (0..k).collect();
            loop {
                let mut part = vec![1u8; n];
                part[0] = 0;
                for &i in &indices {
                    part[rest[i]] = 0;
                }
                let c = cut_size(&part, &edges);
                if c < best_cut {
                    best_cut = c;
                    best_part = part;
                }
                // next combination
                let mut i_opt: Option<usize> = None;
                for i in (0..k).rev() {
                    if indices[i] < i + total - k {
                        i_opt = Some(i);
                        break;
                    }
                }
                match i_opt {
                    None => break,
                    Some(i) => {
                        indices[i] += 1;
                        for j in (i+1)..k {
                            indices[j] = indices[j-1] + 1;
                        }
                    }
                }
            }
        }

        let part_a: Vec<usize> = (0..n).filter(|&i| best_part[i] == 0).collect();
        writeln!(out, "{}", best_cut).unwrap();
        let s: Vec<String> = part_a.iter().map(|x| x.to_string()).collect();
        writeln!(out, "{}", s.join(" ")).unwrap();
        return;
    }

    let mut best_cut = i64::MAX;
    let mut best_part = vec![0u8; n];

    let mut seed: u64 = 0x9E3779B97F4A7C15;
    let mut rand = || -> u64 {
        seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        seed
    };

    let restarts = if n <= 40 { 300 } else if n <= 80 { 100 } else if n <= 150 { 40 } else if n <= 300 { 15 } else { 5 };

    for _ in 0..restarts {
        let mut perm: Vec<usize> = (0..n).collect();
        for i in (1..n).rev() {
            let j = (rand() as usize) % (i+1);
            perm.swap(i, j);
        }
        let mut part = vec![1u8; n];
        for i in 0..half {
            part[perm[i]] = 0;
        }
        let c = kl_improve(&mut part, n, &adj, &edges);
        if c < best_cut {
            best_cut = c;
            best_part = part;
        }
    }

    let part_a: Vec<usize> = (0..n).filter(|&i| best_part[i] == 0).collect();
    writeln!(out, "{}", best_cut).unwrap();
    let s: Vec<String> = part_a.iter().map(|x| x.to_string()).collect();
    writeln!(out, "{}", s.join(" ")).unwrap();
}
