; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %vcolor %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %vcolor Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
       %bool = OpTypeBool
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
        %m4f = OpTypeMatrix %v4f 4

        %UBO = OpTypeStruct %m4f %m4f %m4f %v4f %v4f %v4f
   %ptr_ubo  = OpTypePointer Uniform %UBO
       %ubo  = OpVariable %ptr_ubo Uniform
   %ptr_u_v4 = OpTypePointer Uniform %v4f

  %ptr_in_v3 = OpTypePointer Input %v3f
  %ptr_in_v2 = OpTypePointer Input %v2f
 %ptr_out_v4 = OpTypePointer Output %v4f

   %worldPos = OpVariable %ptr_in_v3 Input
     %normal = OpVariable %ptr_in_v3 Input
         %uv = OpVariable %ptr_in_v2 Input
    %tangent = OpVariable %ptr_in_v3 Input
     %vcolor = OpVariable %ptr_in_v3 Input
   %outColor = OpVariable %ptr_out_v4 Output

       %i_3  = OpConstant %int 3
       %u_0  = OpConstant %uint 0
       %u_2  = OpConstant %uint 2
       %f_0  = OpConstant %float 0.0
       %f_1  = OpConstant %float 1.0
       %f_0_5 = OpConstant %float 0.5
       %f_0_2 = OpConstant %float 0.2
       %f_0_9 = OpConstant %float 0.9
       %f_8  = OpConstant %float 8.0

       %main = OpFunction %void None %fn_void
      %label = OpLabel

       ; --- Checkerboard ---
       %uvVal = OpLoad %v2f %uv
       %uvScaled = OpVectorTimesScalar %v2f %uvVal %f_8
       %uS = OpCompositeExtract %float %uvScaled 0
       %vS = OpCompositeExtract %float %uvScaled 1
       %uFloor = OpExtInst %float %1 Floor %uS
       %vFloor = OpExtInst %float %1 Floor %vS
       %uI = OpConvertFToS %int %uFloor
       %vI = OpConvertFToS %int %vFloor
       %sumI = OpIAdd %int %uI %vI
       %sumU = OpBitcast %uint %sumI
       %parity = OpBitwiseAnd %uint %sumU %u_2
       %parityMod = OpUMod %uint %sumU %u_2
       %isOdd = OpINotEqual %bool %parityMod %u_0
       %cVal = OpSelect %float %isOdd %f_0_2 %f_0_9
       %baseColor = OpCompositeConstruct %v3f %cVal %cVal %cVal

       ; --- Lambertian shading ---
       %nRaw = OpLoad %v3f %normal
       %nNorm = OpExtInst %v3f %1 Normalize %nRaw

       %lightPosPtr = OpAccessChain %ptr_u_v4 %ubo %i_3
       %lightPos4 = OpLoad %v4f %lightPosPtr
       %lightPos3 = OpVectorShuffle %v3f %lightPos4 %lightPos4 0 1 2
       %wp = OpLoad %v3f %worldPos
       %lDir = OpFSub %v3f %lightPos3 %wp
       %lDirN = OpExtInst %v3f %1 Normalize %lDir

       %ndotl = OpDot %float %nNorm %lDirN
       %diffuseClamp = OpExtInst %float %1 FMax %ndotl %f_0
       %halfDiff = OpFMul %float %diffuseClamp %f_0_5
       %factor = OpFAdd %float %f_0_5 %halfDiff

       %litRGB = OpVectorTimesScalar %v3f %baseColor %factor

       %r = OpCompositeExtract %float %litRGB 0
       %g = OpCompositeExtract %float %litRGB 1
       %b = OpCompositeExtract %float %litRGB 2
       %outRGBA = OpCompositeConstruct %v4f %r %g %b %f_1
       OpStore %outColor %outRGBA

       OpReturn
       OpFunctionEnd
