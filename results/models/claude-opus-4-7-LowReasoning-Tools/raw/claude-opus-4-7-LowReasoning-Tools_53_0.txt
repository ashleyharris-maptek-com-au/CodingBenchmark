import math

_state = {
    'step': 0,
    'prev_pos': None,
    'est_pos': [0.0, 0.0, 0.0],
    'est_vel': [0.0, 0.0, 0.0],
    'have_est': False,
    'bias_x': 0.0,
    'bias_y': 0.0,
    'bias_z': 0.0,
}


def _median(vals):
    vals = sorted(vals)
    n = len(vals)
    if n == 0:
        return None
    if n % 2 == 1:
        return vals[n // 2]
    return 0.5 * (vals[n // 2 - 1] + vals[n // 2])


def _get_pos(state):
    cx, cy, cz = [], [], []
    if state.get('lidar_rel_x_valid'):
        cx.append(state.get('lidar_rel_pos_x_m', 0.0))
    if state.get('lidar_rel_y_valid'):
        cy.append(state.get('lidar_rel_pos_y_m', 0.0))
    if state.get('lidar_rel_z_valid'):
        cz.append(state.get('lidar_rel_pos_z_m', 0.0))
    if state.get('radar_rel_x_valid'):
        cx.append(state.get('radar_rel_pos_x_m', 0.0))
    if state.get('radar_rel_y_valid'):
        cy.append(state.get('radar_rel_pos_y_m', 0.0))
    if state.get('radar_rel_z_valid'):
        cz.append(state.get('radar_rel_pos_z_m', 0.0))
    x = _median(cx) if cx else None
    y = _median(cy) if cy else None
    z = _median(cz) if cz else None

    if state.get('dock_camera_active'):
        rng = state.get('dock_camera_range_m', 0.0)
        if rng and rng > 0:
            bx = state.get('dock_camera_bearing_x', 0.0) if state.get('dock_camera_bearing_x_valid') else 0.0
            by = state.get('dock_camera_bearing_y', 0.0) if state.get('dock_camera_bearing_y_valid') else 0.0
            bz = state.get('dock_camera_bearing_z', 0.0) if state.get('dock_camera_bearing_z_valid') else 0.0
            mag = math.sqrt(bx * bx + by * by + bz * bz)
            if mag > 0.01:
                if x is None:
                    x = bx / mag * rng
                if y is None:
                    y = by / mag * rng
                if z is None:
                    z = bz / mag * rng
    return x, y, z


def _get_vel(state):
    cx, cy, cz = [], [], []
    if state.get('lidar_relv_x_valid'):
        cx.append(state.get('lidar_rel_vel_x_ms', 0.0))
    if state.get('lidar_relv_y_valid'):
        cy.append(state.get('lidar_rel_vel_y_ms', 0.0))
    if state.get('lidar_relv_z_valid'):
        cz.append(state.get('lidar_rel_vel_z_ms', 0.0))
    if state.get('radar_relv_x_valid'):
        cx.append(state.get('radar_rel_vel_x_ms', 0.0))
    if state.get('radar_relv_y_valid'):
        cy.append(state.get('radar_rel_vel_y_ms', 0.0))
    if state.get('radar_relv_z_valid'):
        cz.append(state.get('radar_rel_vel_z_ms', 0.0))
    vx = _median(cx) if cx else None
    vy = _median(cy) if cy else None
    vz = _median(cz) if cz else None
    return vx, vy, vz


def _clamp(v, lo=-1.0, hi=1.0):
    return max(lo, min(hi, v))


def autopilot_step(state: dict, dt: float) -> dict:
    global _state
    _state['step'] += 1

    a = state.get('ground_semi_major_axis_m', 6.778e6)
    if not state.get('ground_semi_major_axis_m_valid') or not a or a <= 0:
        a = 6.778e6
    mu = 3.986e14
    n = math.sqrt(mu / (a ** 3))

    px, py, pz = _get_pos(state)
    vx, vy, vz = _get_vel(state)

    # Fallback to propagation
    if px is None or py is None or pz is None:
        if _state['have_est']:
            ex, ey, ez = _state['est_pos']
            evx_, evy_, evz_ = _state['est_vel']
            ex2 = ex + evx_ * dt
            ey2 = ey + evy_ * dt
            ez2 = ez + evz_ * dt
            if px is None:
                px = ex2
            if py is None:
                py = ey2
            if pz is None:
                pz = ez2
        else:
            px = px if px is not None else 0.0
            py = py if py is not None else -100.0
            pz = pz if pz is not None else 0.0

    if vx is None or vy is None or vz is None:
        if _state['prev_pos'] is not None:
            ppx, ppy, ppz = _state['prev_pos']
            if vx is None:
                vx = (px - ppx) / max(dt, 1e-3)
            if vy is None:
                vy = (py - ppy) / max(dt, 1e-3)
            if vz is None:
                vz = (pz - ppz) / max(dt, 1e-3)
        else:
            vx = vx if vx is not None else 0.0
            vy = vy if vy is not None else 0.0
            vz = vz if vz is not None else 0.0

    _state['prev_pos'] = (px, py, pz)
    _state['est_pos'] = [px, py, pz]
    _state['est_vel'] = [vx, vy, vz]
    _state['have_est'] = True

    rng = math.sqrt(px * px + py * py + pz * pz)
    speed = math.sqrt(vx * vx + vy * vy + vz * vz)

    if rng > 500:
        v_des = 2.0
    elif rng > 200:
        v_des = 1.0
    elif rng > 50:
        v_des = 0.4
    elif rng > 10:
        v_des = 0.15
    elif rng > 2:
        v_des = 0.07
    else:
        v_des = 0.03

    if rng > 0.1:
        ux = -px / rng
        uy = -py / rng
        uz = -pz / rng
    else:
        ux = uy = uz = 0.0

    vdes_x = ux * v_des
    vdes_y = uy * v_des
    vdes_z = uz * v_des

    evx = vdes_x - vx
    evy = vdes_y - vy
    evz = vdes_z - vz

    ff_x = -3.0 * n * n * px - 2.0 * n * vy
    ff_y = 2.0 * n * vx
    ff_z = n * n * pz

    max_accel = 0.22
    kp_v = 1.5
    ax_cmd = kp_v * evx + ff_x
    ay_cmd = kp_v * evy + ff_y
    az_cmd = kp_v * evz + ff_z

    if rng < 80 and speed < 1.0:
        _state['bias_x'] += 0.0008 * evx
        _state['bias_y'] += 0.0008 * evy
        _state['bias_z'] += 0.0008 * evz
        _state['bias_x'] = _clamp(_state['bias_x'], -0.3, 0.3)
        _state['bias_y'] = _clamp(_state['bias_y'], -0.3, 0.3)
        _state['bias_z'] = _clamp(_state['bias_z'], -0.3, 0.3)

    tx = ax_cmd / max_accel + _state['bias_x']
    ty = ay_cmd / max_accel + _state['bias_y']
    tz = az_cmd / max_accel + _state['bias_z']

    fuel_pct = state.get('fuel_remaining_pct', 100.0)
    if fuel_pct < 15:
        if abs(evx) < 0.03:
            tx = 0.0
        if abs(evy) < 0.03:
            ty = 0.0
        if abs(evz) < 0.03:
            tz = 0.0

    if rng < 1.8 and speed < 0.25:
        tx = ty = tz = 0.0

    tx = _clamp(tx)
    ty = _clamp(ty)
    tz = _clamp(tz)

    gp, gr, gy = [], [], []
    for i in (1, 2, 3):
        if state.get(f'gyro_{i}_pitch_valid'):
            gp.append(state.get(f'gyro_{i}_pitch_dps', 0.0))
        if state.get(f'gyro_{i}_roll_valid'):
            gr.append(state.get(f'gyro_{i}_roll_dps', 0.0))
        if state.get(f'gyro_{i}_yaw_valid'):
            gy.append(state.get(f'gyro_{i}_yaw_dps', 0.0))
    pitch_rate = _median(gp) if gp else 0.0
    roll_rate = _median(gr) if gr else 0.0
    yaw_rate = _median(gy) if gy else 0.0

    d2r = math.pi / 180.0
    att_pitch = _clamp(-pitch_rate * d2r * 0.5, -0.035, 0.035)
    att_roll = _clamp(-roll_rate * d2r * 0.5, -0.035, 0.035)
    att_yaw = _clamp(-yaw_rate * d2r * 0.5, -0.035, 0.035)

    return {
        'thrust_x': tx,
        'thrust_y': ty,
        'thrust_z': tz,
        'att_rate_roll': att_roll,
        'att_rate_pitch': att_pitch,
        'att_rate_yaw': att_yaw,
    }
