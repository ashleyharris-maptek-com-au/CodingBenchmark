OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_out %gl_in %inColor %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2
%float_thick = OpConstant %float 0.03

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float

   %arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
  %ptr_in_v3 = OpTypePointer Input %v3float
    %inColor = OpVariable %ptr_in_arr_v3 Input

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

     %ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %ptr_p0
     %ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %ptr_p1

     %ptr_c0 = OpAccessChain %ptr_in_v3 %inColor %int_0
         %c0 = OpLoad %v3float %ptr_c0
     %ptr_c1 = OpAccessChain %ptr_in_v3 %inColor %int_1
         %c1 = OpLoad %v3float %ptr_c1

        %p0x = OpCompositeExtract %float %p0 0
        %p0y = OpCompositeExtract %float %p0 1
        %p0z = OpCompositeExtract %float %p0 2
        %p0w = OpCompositeExtract %float %p0 3
        %p1x = OpCompositeExtract %float %p1 0
        %p1y = OpCompositeExtract %float %p1 1
        %p1z = OpCompositeExtract %float %p1 2
        %p1w = OpCompositeExtract %float %p1 3
         %dx = OpFSub %float %p1x %p0x
         %dy = OpFSub %float %p1y %p0y
      %neg_dy = OpFNegate %float %dy
    %perp_raw = OpCompositeConstruct %v2float %neg_dy %dx
      %perp_n = OpExtInst %v2float %1 Normalize %perp_raw
        %perp = OpVectorTimesScalar %v2float %perp_n %float_thick
       %perpx = OpCompositeExtract %float %perp 0
       %perpy = OpCompositeExtract %float %perp 1

        %v0x = OpFAdd %float %p0x %perpx
        %v0y = OpFAdd %float %p0y %perpy
       %pos0 = OpCompositeConstruct %v4float %v0x %v0y %p0z %p0w
    %op_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %op_pos0 %pos0
               OpStore %outColor %c0
               OpEmitVertex

        %v1x = OpFSub %float %p0x %perpx
        %v1y = OpFSub %float %p0y %perpy
       %pos1 = OpCompositeConstruct %v4float %v1x %v1y %p0z %p0w
    %op_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %op_pos1 %pos1
               OpStore %outColor %c0
               OpEmitVertex

        %v2x = OpFAdd %float %p1x %perpx
        %v2y = OpFAdd %float %p1y %perpy
       %pos2 = OpCompositeConstruct %v4float %v2x %v2y %p1z %p1w
    %op_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %op_pos2 %pos2
               OpStore %outColor %c1
               OpEmitVertex

        %v3x = OpFSub %float %p1x %perpx
        %v3y = OpFSub %float %p1y %perpy
       %pos3 = OpCompositeConstruct %v4float %v3x %v3y %p1z %p1w
    %op_pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %op_pos3 %pos3
               OpStore %outColor %c1
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
