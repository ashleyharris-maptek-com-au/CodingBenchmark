use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u: u32 = it.next().unwrap().parse().unwrap();
        let v: u32 = it.next().unwrap().parse().unwrap();
        if u == v { continue; }
        let (a, b) = if u < v { (u, v) } else { (v, u) };
        edges.push((a, b));
    }
    edges.sort_unstable();
    edges.dedup();

    let mut adj: Vec<Vec<u32>> = vec![Vec::new(); n];
    for &(u, v) in &edges {
        adj[u as usize].push(v);
        adj[v as usize].push(u);
    }
    drop(edges);

    let mut in_cover: Vec<bool> = vec![false; n];
    let mut removed: Vec<bool> = vec![false; n];
    let mut deg: Vec<u32> = adj.iter().map(|a| a.len() as u32).collect();

    // Reduction: degree-1 -> include neighbor; degree-0 -> drop
    let mut queue: Vec<u32> = Vec::new();
    for i in 0..n {
        if deg[i] <= 1 {
            queue.push(i as u32);
        }
    }

    while let Some(v) = queue.pop() {
        let vu = v as usize;
        if removed[vu] { continue; }
        if deg[vu] == 0 {
            removed[vu] = true;
            continue;
        }
        if deg[vu] == 1 {
            let mut nb: u32 = u32::MAX;
            for &w in &adj[vu] {
                if !removed[w as usize] {
                    nb = w;
                    break;
                }
            }
            if nb == u32::MAX {
                removed[vu] = true;
                deg[vu] = 0;
                continue;
            }
            in_cover[nb as usize] = true;
            removed[nb as usize] = true;
            removed[vu] = true;
            deg[vu] = 0;
            let nb_neighbors = std::mem::take(&mut adj[nb as usize]);
            for &w in &nb_neighbors {
                let wu = w as usize;
                if !removed[wu] {
                    if deg[wu] > 0 { deg[wu] -= 1; }
                    if deg[wu] <= 1 {
                        queue.push(w);
                    }
                }
            }
        }
    }

    // Greedy max-degree using bucket structure
    let mut max_deg: u32 = 0;
    for i in 0..n {
        if !removed[i] && deg[i] > max_deg {
            max_deg = deg[i];
        }
    }

    if max_deg > 0 {
        let mut buckets: Vec<Vec<u32>> = vec![Vec::new(); (max_deg as usize) + 1];
        let mut pos_in_bucket: Vec<u32> = vec![u32::MAX; n];
        let mut bucket_of: Vec<u32> = vec![u32::MAX; n];

        for i in 0..n {
            if !removed[i] && deg[i] > 0 {
                let d = deg[i] as usize;
                pos_in_bucket[i] = buckets[d].len() as u32;
                bucket_of[i] = d as u32;
                buckets[d].push(i as u32);
            }
        }

        let mut cur_max = max_deg as i64;

        loop {
            while cur_max >= 1 && buckets[cur_max as usize].is_empty() {
                cur_max -= 1;
            }
            if cur_max < 1 { break; }
            let v = *buckets[cur_max as usize].last().unwrap();
            let vu = v as usize;
            // remove v from its bucket
            {
                let b = bucket_of[vu] as usize;
                let p = pos_in_bucket[vu] as usize;
                let bvec = &mut buckets[b];
                let last = *bvec.last().unwrap();
                bvec.swap_remove(p);
                if (last as usize) != vu {
                    pos_in_bucket[last as usize] = p as u32;
                }
                pos_in_bucket[vu] = u32::MAX;
                bucket_of[vu] = u32::MAX;
            }
            in_cover[vu] = true;
            removed[vu] = true;
            let neighbors = std::mem::take(&mut adj[vu]);
            for &w in &neighbors {
                let wu = w as usize;
                if !removed[wu] && deg[wu] > 0 {
                    let old_d = deg[wu] as usize;
                    let new_d = old_d - 1;
                    deg[wu] = new_d as u32;
                    let b = bucket_of[wu] as usize;
                    let p = pos_in_bucket[wu] as usize;
                    let bvec = &mut buckets[b];
                    let last = *bvec.last().unwrap();
                    bvec.swap_remove(p);
                    if (last as usize) != wu {
                        pos_in_bucket[last as usize] = p as u32;
                    }
                    pos_in_bucket[wu] = u32::MAX;
                    bucket_of[wu] = u32::MAX;
                    if new_d > 0 {
                        pos_in_bucket[wu] = buckets[new_d].len() as u32;
                        bucket_of[wu] = new_d as u32;
                        buckets[new_d].push(w);
                    } else {
                        removed[wu] = true;
                    }
                }
            }
        }
    }

    let cover: Vec<usize> = (0..n).filter(|&i| in_cover[i]).collect();
    writeln!(out, "{}", cover.len()).unwrap();
    let mut first = true;
    for v in cover {
        if !first { write!(out, " ").unwrap(); }
        write!(out, "{}", v).unwrap();
        first = false;
    }
    writeln!(out).unwrap();
}
