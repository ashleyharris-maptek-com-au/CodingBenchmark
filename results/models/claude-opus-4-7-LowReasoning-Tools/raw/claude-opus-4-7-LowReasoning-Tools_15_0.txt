use std::io::{self, Read, Write, BufWriter};

#[derive(Clone, Copy, Debug)]
struct V2 { x: f64, y: f64 }

#[derive(Clone, Copy, Debug)]
struct V3 { x: f64, y: f64, z: f64 }

fn polygon_area(poly: &[V2]) -> f64 {
    let n = poly.len();
    let mut a = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        a += poly[i].x * poly[j].y - poly[j].x * poly[i].y;
    }
    a * 0.5
}

fn bbox(poly: &[V2]) -> (f64, f64, f64, f64) {
    let mut mnx = f64::INFINITY;
    let mut mny = f64::INFINITY;
    let mut mxx = f64::NEG_INFINITY;
    let mut mxy = f64::NEG_INFINITY;
    for p in poly {
        if p.x < mnx { mnx = p.x; }
        if p.y < mny { mny = p.y; }
        if p.x > mxx { mxx = p.x; }
        if p.y > mxy { mxy = p.y; }
    }
    (mnx, mny, mxx, mxy)
}

fn point_in_poly(p: V2, poly: &[V2]) -> bool {
    let n = poly.len();
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let pi = poly[i];
        let pj = poly[j];
        if ((pi.y > p.y) != (pj.y > p.y))
            && (p.x < (pj.x - pi.x) * (p.y - pi.y) / (pj.y - pi.y + 1e-30) + pi.x)
        {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut iter = input.split_ascii_whitespace();
    let nv: usize = iter.next().unwrap().parse().unwrap();
    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = iter.next().unwrap().parse().unwrap();
        let y: f64 = iter.next().unwrap().parse().unwrap();
        poly.push(V2 { x, y });
    }
    let sx: f64 = iter.next().unwrap().parse().unwrap();
    let sy: f64 = iter.next().unwrap().parse().unwrap();
    let sz: f64 = iter.next().unwrap().parse().unwrap();
    let sun = V3 { x: sx, y: sy, z: sz };

    let (mnx, mny, mxx, mxy) = bbox(&poly);
    let w = mxx - mnx;
    let h = mxy - mny;
    let area = polygon_area(&poly).abs();
    let max_dim = w.max(h).max(1e-9);

    // Tetra rotated 45 deg about z casts an axis-aligned square shadow
    // (under vertical sun) of side = scale * sqrt(2).
    // Max scale = 10 -> max shadow side ~ 14.14.
    let max_cell = 10.0 * (2.0_f64).sqrt() * 0.9;

    let target_cells = if area < 4.0 {
        4.0
    } else if area < 100.0 {
        16.0
    } else if area < 10000.0 {
        64.0
    } else {
        256.0
    };
    let mut cell = (max_dim / target_cells.sqrt()).max(max_dim / 100.0);
    if cell > max_cell { cell = max_cell; }
    if cell <= 0.0 { cell = 1.0; }

    let nx = ((w / cell).ceil() as i64).max(1);
    let ny = ((h / cell).ceil() as i64).max(1);

    let sun_z = if sun.z.abs() < 1e-9 { -1.0 } else { sun.z };
    let shear = (1.0 + (sun.x/sun_z).powi(2) + (sun.y/sun_z).powi(2)).sqrt();
    let mut tetra_scale = (cell / (2.0_f64).sqrt()) * 1.15 * shear;
    if tetra_scale > 10.0 { tetra_scale = 10.0; }
    if tetra_scale < 0.01 { tetra_scale = 0.01; }

    // 45-deg rotation about z: q = (cos(pi/8), 0, 0, sin(pi/8))
    let half = std::f64::consts::FRAC_PI_8;
    let q: [f64; 4] = [half.cos(), 0.0, 0.0, half.sin()];

    let mut cells: Vec<(f64, f64)> = Vec::new();
    for j in 0..ny {
        for i in 0..nx {
            let cx = mnx + (i as f64 + 0.5) * cell;
            let cy = mny + (j as f64 + 0.5) * cell;
            let pts = [
                V2 { x: cx, y: cy },
                V2 { x: cx - cell*0.5, y: cy - cell*0.5 },
                V2 { x: cx + cell*0.5, y: cy - cell*0.5 },
                V2 { x: cx - cell*0.5, y: cy + cell*0.5 },
                V2 { x: cx + cell*0.5, y: cy + cell*0.5 },
            ];
            let mut hit = false;
            for p in &pts {
                if point_in_poly(*p, &poly) { hit = true; break; }
            }
            if !hit {
                for v in &poly {
                    if v.x >= cx - cell*0.5 - 1e-12 && v.x <= cx + cell*0.5 + 1e-12
                        && v.y >= cy - cell*0.5 - 1e-12 && v.y <= cy + cell*0.5 + 1e-12 {
                        hit = true; break;
                    }
                }
            }
            if hit {
                cells.push((cx, cy));
            }
        }
    }
    if cells.is_empty() {
        let cx = (mnx + mxx) * 0.5;
        let cy = (mny + mxy) * 0.5;
        cells.push((cx, cy));
        let needed = (max_dim / (2.0_f64).sqrt()) * 1.15 * shear;
        tetra_scale = needed.min(10.0).max(0.01);
    }

    // Stack along +z to avoid 3D intersections (z-extent per tetra is +-0.707*scale).
    let dz = 1.5 * tetra_scale;

    let count = cells.len();
    writeln!(out, "{}", count).unwrap();
    for (idx, (cx, cy)) in cells.iter().enumerate() {
        let cz = (idx as f64 + 1.0) * dz;
        // Compensate xy so projected shadow center lands at (cx, cy):
        // shadow.x = tx - cz*sx/sz, shadow.y = ty - cz*sy/sz
        let tx = cx + cz * sun.x / sun_z;
        let ty = cy + cz * sun.y / sun_z;
        writeln!(
            out,
            "{} {} {} {} {} {} {} {}",
            tx, ty, cz, q[0], q[1], q[2], q[3], tetra_scale
        ).unwrap();
    }
}
