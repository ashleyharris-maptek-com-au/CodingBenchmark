const fs = require('fs');
const data = JSON.parse(fs.readFileSync(0, 'utf8'));

const grid = data.grid;
const robots = data.robots;
const jobs = data.jobs;
const H = grid.length;
const W = H > 0 ? grid[0].length : 0;

function bfs(sx, sy) {
  const dist = new Int32Array(W * H);
  dist.fill(-1);
  if (sx < 0 || sy < 0 || sx >= W || sy >= H) return dist;
  if (grid[sy].charCodeAt(sx) === 35) return dist;
  const q = new Int32Array(W * H);
  let head = 0, tail = 0;
  const start = sy * W + sx;
  dist[start] = 0;
  q[tail++] = start;
  while (head < tail) {
    const cur = q[head++];
    const cy = (cur / W) | 0;
    const cx = cur - cy * W;
    const d = dist[cur] + 1;
    if (cx + 1 < W && grid[cy].charCodeAt(cx + 1) !== 35) {
      const n = cur + 1;
      if (dist[n] === -1) { dist[n] = d; q[tail++] = n; }
    }
    if (cx - 1 >= 0 && grid[cy].charCodeAt(cx - 1) !== 35) {
      const n = cur - 1;
      if (dist[n] === -1) { dist[n] = d; q[tail++] = n; }
    }
    if (cy + 1 < H && grid[cy + 1].charCodeAt(cx) !== 35) {
      const n = cur + W;
      if (dist[n] === -1) { dist[n] = d; q[tail++] = n; }
    }
    if (cy - 1 >= 0 && grid[cy - 1].charCodeAt(cx) !== 35) {
      const n = cur - W;
      if (dist[n] === -1) { dist[n] = d; q[tail++] = n; }
    }
  }
  return dist;
}

const sortedJobs = jobs.slice().sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.id - b.id;
});

const assignments = robots.map(() => []);
const robotPos = robots.map(r => ({ x: r.x, y: r.y }));
const distCache = new Array(robots.length).fill(null);

for (const job of sortedJobs) {
  let bestRobot = -1;
  let bestDist = Infinity;
  for (let i = 0; i < robots.length; i++) {
    if (!distCache[i]) {
      distCache[i] = bfs(robotPos[i].x, robotPos[i].y);
    }
    const d = distCache[i][job.y * W + job.x];
    if (d < 0) continue;
    if (d < bestDist) {
      bestDist = d;
      bestRobot = i;
    }
  }
  if (bestRobot === -1) bestRobot = 0;
  assignments[bestRobot].push(job.id);
  robotPos[bestRobot] = { x: job.x, y: job.y };
  distCache[bestRobot] = null;
}

process.stdout.write(JSON.stringify({ assignments }));
