#include <iostream>
#include <vector>
#include <deque>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <cstdint>
#include <algorithm>
#include <cstdio>
#include <climits>

using namespace std;

static int D;
static int MAX_TURNS;
static vector<int> BOUNDS;

typedef vector<int> Pos;

struct PosHash {
    size_t operator()(const Pos& p) const noexcept {
        uint64_t h = 1469598103934665603ULL;
        for (int v : p) {
            h ^= (uint64_t)(uint32_t)v;
            h *= 1099511628211ULL;
        }
        return (size_t)h;
    }
};

static inline bool inBounds(const Pos& p) {
    for (int i = 0; i < D; ++i) if (p[i] < 0 || p[i] >= BOUNDS[i]) return false;
    return true;
}

static inline int manhattan(const Pos& a, const Pos& b) {
    int s = 0;
    for (int i = 0; i < D; ++i) s += abs(a[i] - b[i]);
    return s;
}

int main() {
    ios::sync_with_stdio(false);

    if (!(cin >> D >> MAX_TURNS)) return 0;
    BOUNDS.assign(D, 0);
    for (int i = 0; i < D; ++i) cin >> BOUNDS[i];

    int snake_length;
    cin >> snake_length;
    Pos head(D);
    for (int i = 0; i < D; ++i) cin >> head[i];

    int F;
    cin >> F;
    vector<Pos> foods(F, Pos(D));
    for (int i = 0; i < F; ++i)
        for (int j = 0; j < D; ++j)
            cin >> foods[i][j];

    int O;
    cin >> O;
    unordered_set<Pos, PosHash> obstacles;
    obstacles.reserve((size_t)O * 2 + 16);
    for (int i = 0; i < O; ++i) {
        Pos p(D);
        for (int j = 0; j < D; ++j) cin >> p[j];
        obstacles.insert(p);
    }

    deque<Pos> snake;
    snake.push_front(head);
    int pending_growth = max(0, snake_length - 1);

    unordered_set<Pos, PosHash> bodySet;
    bodySet.reserve(64);
    bodySet.insert(head);

    vector<char> eaten(F, 0);
    int foods_left = F;

    auto isBlocked = [&](const Pos& p, bool tailWillMove) {
        if (!inBounds(p)) return true;
        if (obstacles.count(p)) return true;
        if (bodySet.count(p)) {
            if (tailWillMove && !snake.empty() && p == snake.back()) return false;
            return true;
        }
        return false;
    };

    auto bfsStep = [&](const Pos& target) -> pair<int,int> {
        const size_t NODE_CAP = 200000;
        unordered_map<Pos, pair<int,int>, PosHash> firstMove;
        unordered_set<Pos, PosHash> visited;
        firstMove.reserve(1024);
        visited.reserve(1024);
        queue<Pos> q;
        q.push(head);
        visited.insert(head);

        while (!q.empty() && visited.size() < NODE_CAP) {
            Pos cur = q.front(); q.pop();
            if (cur == target) {
                auto it = firstMove.find(cur);
                if (it == firstMove.end()) return {-1,-1};
                return it->second;
            }
            for (int a = 0; a < D; ++a) {
                for (int d = -1; d <= 1; d += 2) {
                    Pos nxt = cur;
                    nxt[a] += d;
                    if (!inBounds(nxt)) continue;
                    if (obstacles.count(nxt)) continue;
                    if (bodySet.count(nxt)) {
                        if (!(pending_growth == 0 && !snake.empty() && nxt == snake.back())) {
                            continue;
                        }
                    }
                    if (visited.count(nxt)) continue;
                    visited.insert(nxt);
                    if (cur == head) firstMove[nxt] = {a, d};
                    else firstMove[nxt] = firstMove[cur];
                    if (nxt == target) {
                        return firstMove[nxt];
                    }
                    q.push(nxt);
                }
            }
        }
        return {-1, -1};
    };

    auto greedyStep = [&](const Pos& target) -> pair<int,int> {
        vector<pair<int,int>> order;
        for (int a = 0; a < D; ++a) {
            int diff = target[a] - head[a];
            order.push_back({-abs(diff), a});
        }
        sort(order.begin(), order.end());
        for (auto& pr : order) {
            int a = pr.second;
            int diff = target[a] - head[a];
            if (diff == 0) continue;
            int d = (diff > 0) ? 1 : -1;
            Pos nxt = head; nxt[a] += d;
            if (!isBlocked(nxt, pending_growth == 0)) return {a, d};
        }
        for (int a = 0; a < D; ++a) {
            for (int d = -1; d <= 1; d += 2) {
                Pos nxt = head; nxt[a] += d;
                if (!isBlocked(nxt, pending_growth == 0)) return {a, d};
            }
        }
        return {-1, -1};
    };

    int turns_used = 0;
    while (turns_used < MAX_TURNS && foods_left > 0) {
        int best = -1; int bestDist = INT_MAX;
        for (int i = 0; i < F; ++i) {
            if (eaten[i]) continue;
            int d = manhattan(head, foods[i]);
            if (d < bestDist) { bestDist = d; best = i; }
        }
        if (best < 0) break;

        pair<int,int> move = {-1,-1};
        bool tryBFS = (D <= 6);
        if (tryBFS) move = bfsStep(foods[best]);
        if (move.first < 0) move = greedyStep(foods[best]);
        if (move.first < 0) {
            cout << 0 << " " << 1 << "\n";
            break;
        }

        cout << move.first << " " << move.second << "\n";
        cout.flush();

        Pos newHead = head;
        newHead[move.first] += move.second;

        snake.push_front(newHead);
        bodySet.insert(newHead);

        bool ate = false;
        for (int i = 0; i < F; ++i) {
            if (!eaten[i] && newHead == foods[i]) {
                eaten[i] = 1;
                --foods_left;
                ate = true;
                break;
            }
        }

        if (ate) pending_growth += 1;

        if (pending_growth > 0) {
            pending_growth -= 1;
        } else {
            Pos tail = snake.back();
            snake.pop_back();
            bodySet.erase(tail);
        }

        head = newHead;
        ++turns_used;
    }

    return 0;
}
