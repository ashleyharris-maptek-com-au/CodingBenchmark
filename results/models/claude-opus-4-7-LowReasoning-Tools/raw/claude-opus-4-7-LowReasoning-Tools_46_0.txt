#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    float dt        = physics.x;
    float softening = physics.y;
    float G         = physics.z;
    float soft2     = softening * softening;

    vec4 pi = inData[i * 2u + 0u];
    vec4 vi = inData[i * 2u + 1u];

    vec3 pos_i = pi.xyz;
    vec3 vel_i = vi.xyz;
    float mass_i = pi.w;

    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }
        vec4 pj = inData[j * 2u + 0u];
        vec3 r  = pj.xyz - pos_i;
        float mass_j = pj.w;

        float distSqr = dot(r, r) + soft2;
        float invDist = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        acc += G * mass_j * r * invDist3;
    }

    vec3 new_vel = vel_i + acc * dt;
    vec3 new_pos = pos_i + new_vel * dt;

    outData[i * 2u + 0u] = vec4(new_pos, mass_i);
    outData[i * 2u + 1u] = vec4(new_vel, 0.0);
}
