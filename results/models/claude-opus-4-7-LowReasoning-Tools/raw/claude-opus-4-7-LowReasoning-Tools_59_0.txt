const fs = require('fs');

function main() {
  const data = JSON.parse(fs.readFileSync(0, 'utf8'));
  const prob = data.probability;
  const drones = data.drones || [];
  const maxSteps = data.maxStepsPerDrone || 0;
  const H = prob.length;
  const W = H > 0 ? prob[0].length : 0;

  const isObstacle = (x, y) => {
    if (x < 0 || y < 0 || x >= W || y >= H) return true;
    return prob[y][x] < 0;
  };

  const claimed = new Set();
  const key = (x, y) => y * W + x;

  const paths = new Array(drones.length);
  const HORIZON = 6;

  function bestNeighborScore(sx, sy, localClaimed) {
    const seen = new Uint8Array(W * H);
    let frontier = [[sx, sy]];
    seen[key(sx, sy)] = 1;
    let best = 0;
    if (!localClaimed.has(key(sx, sy)) && prob[sy][sx] > 0) best = prob[sy][sx];
    let depth = 0;
    while (frontier.length && depth < HORIZON) {
      const next = [];
      for (const [x, y] of frontier) {
        const dirs = [[1,0],[-1,0],[0,1],[0,-1]];
        for (const [dx, dy] of dirs) {
          const nx = x + dx, ny = y + dy;
          if (isObstacle(nx, ny)) continue;
          const k = key(nx, ny);
          if (seen[k]) continue;
          seen[k] = 1;
          const v = (!localClaimed.has(k) && prob[ny][nx] > 0) ? prob[ny][nx] : 0;
          if (v > best) best = v;
          next.push([nx, ny]);
        }
      }
      frontier = next;
      depth++;
    }
    return best;
  }

  for (let i = 0; i < drones.length; i++) {
    const d = drones[i];
    let x = d.x, y = d.y;
    const path = [[x, y]];
    if (!isObstacle(x, y) && prob[y][x] > 0) claimed.add(key(x, y));

    for (let s = 0; s < maxSteps; s++) {
      const candidates = [
        [x, y],
        [x + 1, y],
        [x - 1, y],
        [x, y + 1],
        [x, y - 1],
      ];
      let bestMove = [x, y];
      let bestImmediate = -1;
      let bestLook = -1;
      for (const [nx, ny] of candidates) {
        if (isObstacle(nx, ny)) continue;
        const k = key(nx, ny);
        const imm = (!claimed.has(k) && prob[ny][nx] > 0) ? prob[ny][nx] : 0;
        let look = 0;
        if (imm === 0) {
          look = bestNeighborScore(nx, ny, claimed);
        }
        if (imm > bestImmediate || (imm === bestImmediate && look > bestLook)) {
          bestImmediate = imm;
          bestLook = look;
          bestMove = [nx, ny];
        }
      }
      x = bestMove[0]; y = bestMove[1];
      path.push([x, y]);
      const k = key(x, y);
      if (!claimed.has(k) && prob[y][x] > 0) claimed.add(k);
    }
    paths[i] = path;
  }

  process.stdout.write(JSON.stringify({ paths }));
}

main();
