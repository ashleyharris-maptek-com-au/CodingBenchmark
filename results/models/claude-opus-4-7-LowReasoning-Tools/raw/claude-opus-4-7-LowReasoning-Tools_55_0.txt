using namespace std;
using Clock = chrono::steady_clock;

enum { EMPTY=0, WP=1, WN, WB, WR, WQ, WK, BP=7, BN, BB, BR, BQ, BK };

inline int pieceColor(int p){ return p==0?-1:(p<=6?0:1);}
inline int pieceType(int p){ return p==0?0:((p-1)%6)+1; }

struct Move {
    int from, to;
    int piece;
    int captured;
    int promo;
    bool isEP;
    bool isCastleK;
    bool isCastleQ;
    bool isDoublePush;
};

struct Undo {
    Move m;
    int castling;
    int ep;
    int halfmove;
    int capturedSquare;
    int capturedPiece;
};

struct Board {
    array<int,64> sq;
    int side;
    int castling;
    int ep;
    int halfmove;
    int fullmove;
    int kingSq[2];
};

inline int FILE_OF(int sq){ return sq & 7; }
inline int RANK_OF(int sq){ return sq >> 3; }

string sqName(int sq){ string s; s.push_back('a'+FILE_OF(sq)); s.push_back('1'+RANK_OF(sq)); return s; }
int sqFromName(const string& s){ return (s[1]-'1')*8 + (s[0]-'a'); }

bool parseFEN(const string& fen, Board& b){
    for(auto& s: b.sq) s = EMPTY;
    stringstream ss(fen);
    string board, stm, cast, ep;
    int hm=0, fm=1;
    ss >> board >> stm >> cast >> ep;
    if(!(ss >> hm)) hm = 0;
    if(!(ss >> fm)) fm = 1;
    int rank = 7, file = 0;
    for(char c: board){
        if(c=='/'){ rank--; file=0; }
        else if(isdigit((unsigned char)c)){ file += c-'0'; }
        else {
            int p=0;
            switch(c){
                case 'P': p=WP; break; case 'N': p=WN; break; case 'B': p=WB; break;
                case 'R': p=WR; break; case 'Q': p=WQ; break; case 'K': p=WK; break;
                case 'p': p=BP; break; case 'n': p=BN; break; case 'b': p=BB; break;
                case 'r': p=BR; break; case 'q': p=BQ; break; case 'k': p=BK; break;
            }
            int sq = rank*8 + file;
            b.sq[sq] = p;
            if(p==WK) b.kingSq[0]=sq;
            if(p==BK) b.kingSq[1]=sq;
            file++;
        }
    }
    b.side = (stm=="w")?0:1;
    b.castling = 0;
    for(char c: cast){
        if(c=='K') b.castling |= 1;
        else if(c=='Q') b.castling |= 2;
        else if(c=='k') b.castling |= 4;
        else if(c=='q') b.castling |= 8;
    }
    b.ep = (ep=="-")? -1 : sqFromName(ep);
    b.halfmove = hm;
    b.fullmove = fm;
    return true;
}

bool isAttacked(const Board& b, int sq, int bySide){
    int rank = RANK_OF(sq), file = FILE_OF(sq);
    if(bySide==0){
        if(rank>=1){
            if(file>0 && b.sq[sq-9]==WP) return true;
            if(file<7 && b.sq[sq-7]==WP) return true;
        }
    } else {
        if(rank<=6){
            if(file>0 && b.sq[sq+7]==BP) return true;
            if(file<7 && b.sq[sq+9]==BP) return true;
        }
    }
    int knightP = (bySide==0)?WN:BN;
    static const int knDF[8]={1,-1,2,-2,2,-2,1,-1};
    static const int knDR[8]={2,2,1,1,-1,-1,-2,-2};
    for(int i=0;i<8;i++){
        int nf=file+knDF[i], nr=rank+knDR[i];
        if(nf<0||nf>7||nr<0||nr>7) continue;
        if(b.sq[nr*8+nf]==knightP) return true;
    }
    int kingP = (bySide==0)?WK:BK;
    for(int dr=-1;dr<=1;dr++) for(int df=-1;df<=1;df++){
        if(!dr && !df) continue;
        int nf=file+df, nr=rank+dr;
        if(nf<0||nf>7||nr<0||nr>7) continue;
        if(b.sq[nr*8+nf]==kingP) return true;
    }
    int rookP = (bySide==0)?WR:BR;
    int queenP = (bySide==0)?WQ:BQ;
    int bishopP = (bySide==0)?WB:BB;
    static const int ortho[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for(auto& d: ortho){
        int nf=file, nr=rank;
        while(true){
            nf+=d[0]; nr+=d[1];
            if(nf<0||nf>7||nr<0||nr>7) break;
            int p = b.sq[nr*8+nf];
            if(p){ if(p==rookP||p==queenP) return true; break; }
        }
    }
    static const int diag[4][2] = {{1,1},{-1,1},{1,-1},{-1,-1}};
    for(auto& d: diag){
        int nf=file, nr=rank;
        while(true){
            nf+=d[0]; nr+=d[1];
            if(nf<0||nf>7||nr<0||nr>7) break;
            int p = b.sq[nr*8+nf];
            if(p){ if(p==bishopP||p==queenP) return true; break; }
        }
    }
    return false;
}

inline bool inCheck(const Board& b, int side){
    return isAttacked(b, b.kingSq[side], side^1);
}

void addMove(vector<Move>& mvs, int from, int to, int piece, int captured, int promo, bool isEP, bool ck, bool cq, bool dp){
    Move m; m.from=from; m.to=to; m.piece=piece; m.captured=captured; m.promo=promo;
    m.isEP=isEP; m.isCastleK=ck; m.isCastleQ=cq; m.isDoublePush=dp;
    mvs.push_back(m);
}

void generatePseudoMoves(const Board& b, vector<Move>& mvs){
    int side = b.side;
    int myColor = side;
    for(int sq=0; sq<64; sq++){
        int p = b.sq[sq];
        if(!p) continue;
        if(pieceColor(p)!=myColor) continue;
        int pt = pieceType(p);
        int file = FILE_OF(sq), rank = RANK_OF(sq);
        if(pt==1){
            int dir = (side==0)?1:-1;
            int startRank = (side==0)?1:6;
            int promoRank = (side==0)?7:0;
            int one = sq + dir*8;
            if(one>=0 && one<64 && b.sq[one]==0){
                if(RANK_OF(one)==promoRank){
                    for(int pr=5;pr>=2;pr--) addMove(mvs,sq,one,p,0,pr,false,false,false,false);
                } else {
                    addMove(mvs,sq,one,p,0,0,false,false,false,false);
                    if(rank==startRank){
                        int two = sq + dir*16;
                        if(b.sq[two]==0) addMove(mvs,sq,two,p,0,0,false,false,false,true);
                    }
                }
            }
            for(int df=-1; df<=1; df+=2){
                int nf=file+df, nr=rank+dir;
                if(nf<0||nf>7||nr<0||nr>7) continue;
                int dst = nr*8+nf;
                int tp = b.sq[dst];
                if(tp && pieceColor(tp)!=myColor){
                    if(nr==promoRank){
                        for(int pr=5;pr>=2;pr--) addMove(mvs,sq,dst,p,tp,pr,false,false,false,false);
                    } else {
                        addMove(mvs,sq,dst,p,tp,0,false,false,false,false);
                    }
                } else if(dst==b.ep && tp==0){
                    addMove(mvs,sq,dst,p,(side==0?BP:WP),0,true,false,false,false);
                }
            }
        }
        else if(pt==2){
            static const int knDF[8]={1,-1,2,-2,2,-2,1,-1};
            static const int knDR[8]={2,2,1,1,-1,-1,-2,-2};
            for(int i=0;i<8;i++){
                int nf=file+knDF[i], nr=rank+knDR[i];
                if(nf<0||nf>7||nr<0||nr>7) continue;
                int dst = nr*8+nf;
                int tp = b.sq[dst];
                if(tp==0) addMove(mvs,sq,dst,p,0,0,false,false,false,false);
                else if(pieceColor(tp)!=myColor) addMove(mvs,sq,dst,p,tp,0,false,false,false,false);
            }
        }
        else if(pt==3 || pt==4 || pt==5){
            vector<pair<int,int>> dirs;
            if(pt!=4){ dirs.push_back({1,1}); dirs.push_back({1,-1}); dirs.push_back({-1,1}); dirs.push_back({-1,-1}); }
            if(pt!=3){ dirs.push_back({1,0}); dirs.push_back({-1,0}); dirs.push_back({0,1}); dirs.push_back({0,-1}); }
            for(auto& d: dirs){
                int nf=file, nr=rank;
                while(true){
                    nf+=d.first; nr+=d.second;
                    if(nf<0||nf>7||nr<0||nr>7) break;
                    int dst=nr*8+nf;
                    int tp=b.sq[dst];
                    if(tp==0) addMove(mvs,sq,dst,p,0,0,false,false,false,false);
                    else { if(pieceColor(tp)!=myColor) addMove(mvs,sq,dst,p,tp,0,false,false,false,false); break; }
                }
            }
        }
        else if(pt==6){
            for(int dr=-1;dr<=1;dr++) for(int df=-1;df<=1;df++){
                if(!dr && !df) continue;
                int nf=file+df, nr=rank+dr;
                if(nf<0||nf>7||nr<0||nr>7) continue;
                int dst=nr*8+nf;
                int tp=b.sq[dst];
                if(tp==0) addMove(mvs,sq,dst,p,0,0,false,false,false,false);
                else if(pieceColor(tp)!=myColor) addMove(mvs,sq,dst,p,tp,0,false,false,false,false);
            }
            if(side==0 && sq==4){
                if((b.castling&1) && b.sq[5]==0 && b.sq[6]==0 && b.sq[7]==WR
                    && !isAttacked(b,4,1) && !isAttacked(b,5,1) && !isAttacked(b,6,1))
                    addMove(mvs,4,6,p,0,0,false,true,false,false);
                if((b.castling&2) && b.sq[3]==0 && b.sq[2]==0 && b.sq[1]==0 && b.sq[0]==WR
                    && !isAttacked(b,4,1) && !isAttacked(b,3,1) && !isAttacked(b,2,1))
                    addMove(mvs,4,2,p,0,0,false,false,true,false);
            } else if(side==1 && sq==60){
                if((b.castling&4) && b.sq[61]==0 && b.sq[62]==0 && b.sq[63]==BR
                    && !isAttacked(b,60,0) && !isAttacked(b,61,0) && !isAttacked(b,62,0))
                    addMove(mvs,60,62,p,0,0,false,true,false,false);
                if((b.castling&8) && b.sq[59]==0 && b.sq[58]==0 && b.sq[57]==0 && b.sq[56]==BR
                    && !isAttacked(b,60,0) && !isAttacked(b,59,0) && !isAttacked(b,58,0))
                    addMove(mvs,60,58,p,0,0,false,false,true,false);
            }
        }
    }
}

void makeMove(Board& b, const Move& m, Undo& u){
    u.m = m;
    u.castling = b.castling;
    u.ep = b.ep;
    u.halfmove = b.halfmove;
    u.capturedSquare = -1;
    u.capturedPiece = 0;
    int side = b.side;
    int from = m.from, to = m.to;
    int piece = m.piece;
    if(m.isEP){
        int capSq = (side==0)? to-8 : to+8;
        u.capturedSquare = capSq;
        u.capturedPiece = b.sq[capSq];
        b.sq[capSq] = 0;
    } else if(m.captured){
        u.capturedSquare = to;
        u.capturedPiece = b.sq[to];
        b.sq[to] = 0;
    }
    b.sq[from] = 0;
    int placed = piece;
    if(m.promo){
        placed = (side==0)? m.promo : m.promo+6;
    }
    b.sq[to] = placed;
    if(pieceType(piece)==6){
        b.kingSq[side] = to;
        if(m.isCastleK){
            if(side==0){ b.sq[5]=WR; b.sq[7]=0; }
            else { b.sq[61]=BR; b.sq[63]=0; }
        } else if(m.isCastleQ){
            if(side==0){ b.sq[3]=WR; b.sq[0]=0; }
            else { b.sq[59]=BR; b.sq[56]=0; }
        }
        if(side==0) b.castling &= ~3; else b.castling &= ~12;
    }
    if(from==0 || to==0) b.castling &= ~2;
    if(from==7 || to==7) b.castling &= ~1;
    if(from==56 || to==56) b.castling &= ~8;
    if(from==63 || to==63) b.castling &= ~4;
    b.ep = -1;
    if(m.isDoublePush){
        b.ep = (side==0)? from+8 : from-8;
    }
    if(pieceType(piece)==1 || m.captured || m.isEP) b.halfmove=0;
    else b.halfmove++;
    if(side==1) b.fullmove++;
    b.side ^= 1;
}

void unmakeMove(Board& b, const Undo& u){
    const Move& m = u.m;
    b.side ^= 1;
    int side = b.side;
    b.castling = u.castling;
    b.ep = u.ep;
    b.halfmove = u.halfmove;
    if(side==1) b.fullmove--;
    int from = m.from, to = m.to;
    b.sq[from] = m.piece;
    b.sq[to] = 0;
    if(u.capturedSquare>=0){
        b.sq[u.capturedSquare] = u.capturedPiece;
    }
    if(pieceType(m.piece)==6){
        b.kingSq[side] = from;
        if(m.isCastleK){
            if(side==0){ b.sq[7]=WR; b.sq[5]=0; }
            else { b.sq[63]=BR; b.sq[61]=0; }
        } else if(m.isCastleQ){
            if(side==0){ b.sq[0]=WR; b.sq[3]=0; }
            else { b.sq[56]=BR; b.sq[59]=0; }
        }
    }
}

inline int mirror(int sq){ return sq ^ 56; }

int evaluate(const Board& b){
    int score = 0;
    for(int sq=0; sq<64; sq++){
        int p = b.sq[sq];
        if(!p) continue;
        int pt = pieceType(p);
        int c = pieceColor(p);
        int v = PVAL[pt];
        int pst = 0;
        int psq = (c==0)? sq : mirror(sq);
        switch(pt){
            case 1: pst = PST_P[psq]; break;
            case 2: pst = PST_N[psq]; break;
            case 3: pst = PST_B[psq]; break;
            case 4: pst = PST_R[psq]; break;
            case 5: pst = PST_Q[psq]; break;
            case 6: pst = PST_K[psq]; break;
        }
        if(c==0) score += v+pst; else score -= v+pst;
    }
    return (b.side==0)? score : -score;
}

static atomic<bool> g_stop{false};
static Clock::time_point g_deadline;
inline bool timeUp(){
    if(g_stop.load(memory_order_relaxed)) return true;
    return Clock::now() >= g_deadline;
}

int scoreMoveOrd(const Move& m){
    int s = 0;
    if(m.captured) s = 100000 + PVAL[pieceType(m.captured)]*10 - PVAL[pieceType(m.piece)];
    if(m.promo) s += 90000 + PVAL[m.promo];
    return s;
}

int quiesce(Board& b, int alpha, int beta){
    if(timeUp()) return 0;
    int stand = evaluate(b);
    if(stand >= beta) return beta;
    if(alpha < stand) alpha = stand;
    vector<Move> mvs;
    generatePseudoMoves(b, mvs);
    vector<pair<int,int>> ord;
    for(size_t i=0;i<mvs.size();i++){
        if(mvs[i].captured || mvs[i].promo || mvs[i].isEP){
            ord.push_back({scoreMoveOrd(mvs[i]),(int)i});
        }
    }
    sort(ord.begin(), ord.end(), [](auto&a,auto&b){return a.first>b.first;});
    for(auto& o: ord){
        Move& m = mvs[o.second];
        Undo u;
        makeMove(b,m,u);
        if(inCheck(b, b.side^1)){ unmakeMove(b,u); continue; }
        int score = -quiesce(b, -beta, -alpha);
        unmakeMove(b,u);
        if(timeUp()) return 0;
        if(score >= beta) return beta;
        if(score > alpha) alpha = score;
    }
    return alpha;
}

int alphaBeta(Board& b, int depth, int alpha, int beta, int ply){
    if(timeUp()) return 0;
    if(depth<=0) return quiesce(b, alpha, beta);
    vector<Move> mvs;
    generatePseudoMoves(b, mvs);
    vector<pair<int,int>> ord;
    ord.reserve(mvs.size());
    for(size_t i=0;i<mvs.size();i++) ord.push_back({scoreMoveOrd(mvs[i]),(int)i});
    sort(ord.begin(), ord.end(), [](auto&a,auto&b){return a.first>b.first;});
    int legalCount = 0;
    int bestScore = -1000000;
    for(auto& o: ord){
        Move& m = mvs[o.second];
        Undo u;
        makeMove(b,m,u);
        if(inCheck(b, b.side^1)){ unmakeMove(b,u); continue; }
        legalCount++;
        int score = -alphaBeta(b, depth-1, -beta, -alpha, ply+1);
        unmakeMove(b,u);
        if(timeUp()) return 0;
        if(score > bestScore) bestScore = score;
        if(score > alpha) alpha = score;
        if(alpha >= beta) break;
    }
    if(legalCount==0){
        if(inCheck(b, b.side)) return -100000 + ply;
        return 0;
    }
    return bestScore;
}

vector<Move> generateLegalMoves(Board& b){
    vector<Move> ps;
    generatePseudoMoves(b, ps);
    vector<Move> legal;
    for(auto& m: ps){
        Undo u; makeMove(b,m,u);
        if(!inCheck(b, b.side^1)) legal.push_back(m);
        unmakeMove(b,u);
    }
    return legal;
}

string moveToAlgebraic(Board& b, const Move& m){
    string base;
    if(m.isCastleK) base = "O-O";
    else if(m.isCastleQ) base = "O-O-O";
    else {
        char pchar = 'p';
        switch(pieceType(m.piece)){
            case 1: pchar='p'; break;
            case 2: pchar='N'; break;
            case 3: pchar='B'; break;
            case 4: pchar='R'; break;
            case 5: pchar='Q'; break;
            case 6: pchar='K'; break;
        }
        base.push_back(pchar);
        base += sqName(m.from);
        if(m.captured || m.isEP) base.push_back('x');
        base += sqName(m.to);
        if(m.promo){
            base.push_back('/');
            char pr='Q';
            switch(m.promo){
                case 2: pr='N'; break;
                case 3: pr='B'; break;
                case 4: pr='R'; break;
                case 5: pr='Q'; break;
            }
            base.push_back(pr);
        }
        if(m.isEP) base += "ep";
    }
    Undo u; makeMove(b,m,u);
    if(inCheck(b, b.side)){
        auto lm = generateLegalMoves(b);
        base += lm.empty()?"#":"+";
    }
    unmakeMove(b,u);
    return base;
}

            int s = scoreMoveOrd(legal[i]);
            if(legal[i].from==best.from && legal[i].to==best.to && legal[i].promo==best.promo) s = 1000000;
            ord.push_back({s,(int)i});
        }
        sort(ord.begin(), ord.end(), [](auto&a,auto&b){return a.first>b.first;});
        bool completed = true;
        for(auto& o: ord){
            Move& m = legal[o.second];
            Undo u;
            makeMove(b,m,u);
            int score = -alphaBeta(b, depth-1, -beta, -alpha, 1);
            unmakeMove(b,u);
            if(timeUp()){ completed = false; break; }
            if(score > bestThisIter){
                bestThisIter = score;
                bestMoveThisIter = m;
            }
            if(score > alpha) alpha = score;
        }
        if(completed){
            best = bestMoveThisIter;
            bestScore = bestThisIter;
            if(bestScore > 90000) break;
        } else {
            break;
        }
    }

    cout << moveToAlgebraic(b, best) << endl;
    return 0;
}
