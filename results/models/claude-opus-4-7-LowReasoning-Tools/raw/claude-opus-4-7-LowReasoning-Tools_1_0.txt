using namespace std;
using Clock = chrono::steady_clock;

static double elapsed(Clock::time_point t0) {
    return chrono::duration<double>(Clock::now() - t0).count();
}

struct Pt { float x, y; };

static inline float dist(const Pt& a, const Pt& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return sqrtf(dx*dx + dy*dy);
}

// ---------------- Fast I/O ----------------
static char inbuf[1 << 22];
static int inpos = 0, inlen = 0;
static inline int gc() {
    if (inpos >= inlen) {
        inlen = (int)fread(inbuf, 1, sizeof(inbuf), stdin);
        inpos = 0;
        if (inlen <= 0) return -1;
    }
    return inbuf[inpos++];
}
static bool readFloat(float& out) {
    int c = gc();
    while (c != -1 && (c == ' ' || c == '\n' || c == '\r' || c == '\t')) c = gc();
    if (c == -1) return false;
    bool neg = false;
    if (c == '-') { neg = true; c = gc(); }
    else if (c == '+') c = gc();
    double v = 0;
    while (c >= '0' && c <= '9') { v = v*10 + (c - '0'); c = gc(); }
    if (c == '.') {
        c = gc();
        double f = 0, p = 1;
        while (c >= '0' && c <= '9') { f = f*10 + (c-'0'); p *= 10; c = gc(); }
        v += f / p;
    }
    if (c == 'e' || c == 'E') {
        c = gc();
        bool eneg = false;
        if (c == '-') { eneg = true; c = gc(); }
        else if (c == '+') c = gc();
        int e = 0;
        while (c >= '0' && c <= '9') { e = e*10 + (c-'0'); c = gc(); }
        double m = 1;
        for (int i = 0; i < e; i++) m *= 10;
        v = eneg ? v / m : v * m;
    }
    out = (float)(neg ? -v : v);
    return true;
}
static bool readInt(long long& out) {
    int c = gc();
    while (c != -1 && (c == ' '||c=='\n'||c=='\r'||c=='\t')) c = gc();
    if (c == -1) return false;
    bool neg = false;
    if (c == '-') { neg = true; c = gc(); }
    long long v = 0;
    while (c >= '0' && c <= '9') { v = v*10 + (c-'0'); c = gc(); }
    out = neg ? -v : v;
    return true;
}

static char outbuf[1 << 22];
static int outpos = 0;
static void flushOut() {
    if (outpos) { fwrite(outbuf, 1, outpos, stdout); outpos = 0; }
}
static void writeInt(int v) {
    if (outpos > (int)sizeof(outbuf) - 16) flushOut();
    if (v == 0) { outbuf[outpos++] = '0'; return; }
    char tmp[12]; int n = 0;
    while (v > 0) { tmp[n++] = '0' + (v % 10); v /= 10; }
    while (n--) outbuf[outpos++] = tmp[n];
}

// ---------------- Hilbert curve ----------------
static inline uint64_t hilbertD(uint32_t n, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = n / 2; s > 0; s >>= 1) {
        uint32_t rx = (x & s) ? 1u : 0u;
        uint32_t ry = (y & s) ? 1u : 0u;
        d += (uint64_t)s * (uint64_t)s * ((3u * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

static vector<int> hilbertOrder(const vector<Pt>& pts) {
    int N = (int)pts.size();
    float minx = pts[0].x, maxx = pts[0].x, miny = pts[0].y, maxy = pts[0].y;
    for (auto& p : pts) {
        if (p.x < minx) minx = p.x;
        if (p.x > maxx) maxx = p.x;
        if (p.y < miny) miny = p.y;
        if (p.y > maxy) maxy = p.y;
    }
    float rx = maxx - minx; if (rx <= 0) rx = 1;
    float ry = maxy - miny; if (ry <= 0) ry = 1;
    uint32_t order = 1u << 16;
    float sx = (order - 1) / rx;
    float sy = (order - 1) / ry;
    vector<pair<uint64_t,int>> keys((size_t)N);
    for (int i = 0; i < N; i++) {
        uint32_t x = (uint32_t)((pts[i].x - minx) * sx);
        uint32_t y = (uint32_t)((pts[i].y - miny) * sy);
        keys[i] = { hilbertD(order, x, y), i };
    }
    sort(keys.begin(), keys.end(), [](const pair<uint64_t,int>& a, const pair<uint64_t,int>& b){ return a.first < b.first; });
    vector<int> tour(N);
    for (int i = 0; i < N; i++) tour[i] = keys[i].second;
    int z = 0;
    for (int i = 0; i < N; i++) if (tour[i] == 0) { z = i; break; }
    if (z) rotate(tour.begin(), tour.begin() + z, tour.end());
    return tour;
}

// ---------------- KD-tree ----------------
struct KDNode { int lo, hi; int axis; float split; int left, right; };
struct KDTree {
    const vector<Pt>* pts;
    vector<int> idx;
    vector<KDNode> nodes;
    int build(int lo, int hi) {
        int id = (int)nodes.size();
        nodes.push_back({lo, hi, 0, 0.f, -1, -1});
        if (hi - lo <= 16) return id;
        float minx = (*pts)[idx[lo]].x, maxx = minx;
        float miny = (*pts)[idx[lo]].y, maxy = miny;
        for (int i = lo+1; i < hi; i++) {
            float x = (*pts)[idx[i]].x, y = (*pts)[idx[i]].y;
            if (x<minx)minx=x; if(x>maxx)maxx=x;
            if (y<miny)miny=y; if(y>maxy)maxy=y;
        }
        int axis = (maxx-minx) > (maxy-miny) ? 0 : 1;
        int mid = (lo + hi) / 2;
        const vector<Pt>& P = *pts;
        nth_element(idx.begin()+lo, idx.begin()+mid, idx.begin()+hi,
            [&](int a, int b){ return axis==0 ? P[a].x < P[b].x : P[a].y < P[b].y; });
        nodes[id].axis = axis;
        nodes[id].split = axis==0 ? P[idx[mid]].x : P[idx[mid]].y;
        int L = build(lo, mid);
        int R = build(mid, hi);
        nodes[id].left = L;
        nodes[id].right = R;
        return id;
    }
    void init(const vector<Pt>& p) {
        pts = &p;
        idx.resize(p.size());
        iota(idx.begin(), idx.end(), 0);
        nodes.reserve(2 * p.size() + 4);
        if (!p.empty()) build(0, (int)p.size());
    }
    int best; float bestD2;
    const vector<char>* visited;
    Pt target;
    void search(int id) {
        const KDNode& n = nodes[id];
        if (n.left == -1) {
            const vector<Pt>& P = *pts;
            for (int i = n.lo; i < n.hi; i++) {
                int c = idx[i];
                if ((*visited)[c]) continue;
                float dx = P[c].x - target.x, dy = P[c].y - target.y;
                float d2 = dx*dx + dy*dy;
                if (d2 < bestD2) { bestD2 = d2; best = c; }
            }
            return;
        }
        float diff = (n.axis == 0 ? target.x : target.y) - n.split;
        int first = n.left, second = n.right;
        if (diff > 0) { int t=first; first=second; second=t; }
        search(first);
        if (diff * diff < bestD2) search(second);
    }
    int nearest(Pt q, const vector<char>& vis) {
        target = q; visited = &vis;
        best = -1; bestD2 = 1e38f;
        search(0);
        return best;
    }
};

static vector<int> nnTour(const vector<Pt>& pts) {
    int N = (int)pts.size();
    KDTree kd; kd.init(pts);
    vector<char> visited(N, 0);
    vector<int> tour; tour.reserve(N);
    int cur = 0;
    tour.push_back(0);
    visited[0] = 1;
    for (int i = 1; i < N; i++) {
        int nx = kd.nearest(pts[cur], visited);
        if (nx < 0) {
            for (int j = 0; j < N; j++) if (!visited[j]) { nx = j; break; }
        }
        visited[nx] = 1;
        tour.push_back(nx);
        cur = nx;
    }
    return tour;
}

static void twoOpt(vector<int>& tour, const vector<Pt>& pts, double deadline, Clock::time_point t0) {
    int N = (int)tour.size();
    if (N < 4) return;
    int K = N > 50000 ? 8 : (N > 5000 ? 12 : 18);
    if (K >= N) K = N - 1;

    KDTree kd; kd.init(pts);
    vector<int> neigh((size_t)N * K, -1);

    struct Heap {
        int K;
        vector<pair<float,int>> a;
        void push(float d, int i) {
            if ((int)a.size() < K) {
                a.push_back({d,i});
                push_heap(a.begin(), a.end());
            } else if (d < a.front().first) {
                pop_heap(a.begin(), a.end());
                a.back() = {d,i};
                push_heap(a.begin(), a.end());
            }
        }
    };
    function<void(int, Pt, Heap&)> kdsearch = [&](int id, Pt q, Heap& h) {
        const KDNode& n = kd.nodes[id];
        if (n.left == -1) {
            for (int i = n.lo; i < n.hi; i++) {
                int c = kd.idx[i];
                float dx = pts[c].x - q.x, dy = pts[c].y - q.y;
                float d2 = dx*dx + dy*dy;
                h.push(d2, c);
            }
            return;
        }
        float diff = (n.axis == 0 ? q.x : q.y) - n.split;
        int first = n.left, second = n.right;
        if (diff > 0) { int t=first; first=second; second=t; }
        kdsearch(first, q, h);
        float bd = ((int)h.a.size() < h.K) ? 1e38f : h.a.front().first;
        if (diff * diff < bd) kdsearch(second, q, h);
    };

    for (int i = 0; i < N; i++) {
        Heap h; h.K = K + 1;
        kdsearch(0, pts[i], h);
        sort(h.a.begin(), h.a.end());
        int cnt = 0;
        for (auto& pr : h.a) {
            if (pr.second == i) continue;
            neigh[(size_t)i*K + cnt] = pr.second;
            cnt++;
            if (cnt >= K) break;
        }
        while (cnt < K) neigh[(size_t)i*K + cnt++] = -1;
        if ((i & 8191) == 0 && elapsed(t0) > deadline) return;
    }

    vector<int> pos(N);
    for (int i = 0; i < N; i++) pos[tour[i]] = i;

    bool improved = true;
    int passes = 0;
    while (improved) {
        improved = false;
        passes++;
        for (int i = 0; i < N; i++) {
            if ((i & 4095) == 0 && elapsed(t0) > deadline) return;
            int a = tour[i];
            int b = tour[(i+1) % N];
            float dab = dist(pts[a], pts[b]);
            for (int k = 0; k < K; k++) {
                int c = neigh[(size_t)a*K + k];
                if (c < 0) break;
                if (c == a || c == b) continue;
                float dac = dist(pts[a], pts[c]);
                if (dac >= dab) break;
                int j = pos[c];
                int d = tour[(j+1) % N];
                if (d == a) continue;
                float dcd = dist(pts[c], pts[d]);
                float dbd = dist(pts[b], pts[d]);
                if (dab + dcd > dac + dbd + 1e-7f) {
                    int lo = (i+1) % N;
                    int hi = j;
                    int len;
                    if (hi >= lo) len = hi - lo + 1;
                    else len = N - lo + hi + 1;
                    if (len > N/2) {
                        lo = (j+1) % N;
                        hi = i;
                        if (hi >= lo) len = hi - lo + 1;
                        else len = N - lo + hi + 1;
                    }
                    int l = lo, r = hi;
                    for (int s = 0; s < len/2; s++) {
                        int tmp = tour[l]; tour[l] = tour[r]; tour[r] = tmp;
                        pos[tour[l]] = l;
                        pos[tour[r]] = r;
                        l = (l+1) % N;
                        r = (r-1+N) % N;
                    }
                    improved = true;
                    b = tour[(i+1) % N];
                    dab = dist(pts[a], pts[b]);
                }
            }
        }
        if (passes > 30) break;
    }
}

int main() {
    auto t0 = Clock::now();
    double TIME_BUDGET = 280.0;

    long long Nll;
    if (!readInt(Nll)) return 0;
    int N = (int)Nll;
    if (N <= 0) return 0;

    vector<Pt> pts((size_t)N);
    for (int i = 0; i < N; i++) {
        readFloat(pts[i].x);
        readFloat(pts[i].y);
    }

    if (N == 1) {
        writeInt(0); outbuf[outpos++] = '\n'; flushOut(); return 0;
    }

    vector<int> tour;

    if (N <= 200000) {
        tour = nnTour(pts);
        double remain = TIME_BUDGET - elapsed(t0) - 3.0;
        if (remain > 0) twoOpt(tour, pts, elapsed(t0) + remain, t0);
    } else if (N <= 1500000) {
        tour = hilbertOrder(pts);
        double remain = TIME_BUDGET - elapsed(t0) - 10.0;
        if (N <= 500000 && remain > 30) {
            twoOpt(tour, pts, elapsed(t0) + remain, t0);
        }
    } else {
        tour = hilbertOrder(pts);
    }

    if (!tour.empty() && tour[0] != 0) {
        int z = 0;
        for (int i = 0; i < (int)tour.size(); i++) if (tour[i] == 0) { z = i; break; }
        rotate(tour.begin(), tour.begin() + z, tour.end());
    }

    for (int i = 0; i < (int)tour.size(); i++) {
        if (i) outbuf[outpos++] = ' ';
        writeInt(tour[i]);
        if (outpos > (int)sizeof(outbuf) - 16) flushOut();
    }
    outbuf[outpos++] = '\n';
    flushOut();
    return 0;
}
