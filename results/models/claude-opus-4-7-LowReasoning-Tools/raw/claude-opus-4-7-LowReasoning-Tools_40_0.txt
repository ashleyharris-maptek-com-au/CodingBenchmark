#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<double> c;
vector<vector<int>> adj;
vector<uint64_t> adjMask;
int blocks;

double bestVal;
vector<int> bestX;
vector<int> curX;

static inline void setBit(vector<uint64_t>& v, int i){ v[i>>6] |= (1ULL<<(i&63)); }

double upperBound(const vector<uint64_t>& allowed, double cur){
    double s = cur;
    for(int b=0;b<blocks;b++){
        uint64_t x = allowed[b];
        while(x){
            int t = __builtin_ctzll(x);
            x &= x-1;
            int v = (b<<6)|t;
            if(v<n && c[v]>0) s += c[v];
        }
    }
    return s;
}

int pickVertex(const vector<uint64_t>& allowed){
    int best=-1, bestDeg=-1;
    for(int b=0;b<blocks;b++){
        uint64_t x = allowed[b];
        while(x){
            int t = __builtin_ctzll(x);
            x &= x-1;
            int v = (b<<6)|t;
            if(v>=n) continue;
            if(c[v]<=0) continue;
            int deg=0;
            for(int bb=0;bb<blocks;bb++){
                deg += __builtin_popcountll(allowed[bb] & adjMask[(long long)v*blocks+bb]);
            }
            if(deg>bestDeg){bestDeg=deg; best=v;}
        }
    }
    return best;
}

void solve(vector<uint64_t> allowed, double curVal){
    double ub = upperBound(allowed, curVal);
    if(ub <= bestVal + 1e-12) return;

    int v = pickVertex(allowed);
    if(v<0){
        if(curVal > bestVal){
            bestVal = curVal;
            bestX = curX;
        }
        return;
    }

    vector<uint64_t> a1 = allowed;
    for(int b=0;b<blocks;b++) a1[b] &= ~adjMask[(long long)v*blocks+b];
    a1[v>>6] &= ~(1ULL<<(v&63));
    curX[v]=1;
    solve(a1, curVal + c[v]);
    curX[v]=0;

    vector<uint64_t> a2 = allowed;
    a2[v>>6] &= ~(1ULL<<(v&63));
    solve(a2, curVal);
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    if(!(cin>>n>>m)) return 0;
    c.assign(n,0);
    for(int i=0;i<n;i++) cin>>c[i];
    adj.assign(n,{});
    blocks = (n+63)/64;
    if(blocks<1) blocks=1;
    adjMask.assign((long long)n*blocks, 0ULL);
    for(int i=0;i<m;i++){
        int u,v; cin>>u>>v;
        if(u==v) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
        adjMask[(long long)u*blocks + (v>>6)] |= (1ULL<<(v&63));
        adjMask[(long long)v*blocks + (u>>6)] |= (1ULL<<(u&63));
    }

    bestVal = 0;
    bestX.assign(n,0);
    curX.assign(n,0);

    {
        vector<int> ord(n); iota(ord.begin(),ord.end(),0);
        sort(ord.begin(),ord.end(),[&](int a,int b){return c[a]>c[b];});
        vector<int> taken(n,0);
        double val=0;
        for(int v: ord){
            if(c[v]<=0) break;
            bool ok=true;
            for(int u: adj[v]) if(taken[u]){ok=false;break;}
            if(ok){taken[v]=1; val+=c[v];}
        }
        if(val>bestVal){bestVal=val; bestX=taken;}
    }

    vector<uint64_t> allowed(blocks,0ULL);
    for(int i=0;i<n;i++) setBit(allowed,i);

    solve(allowed, 0.0);

    if(fabs(bestVal - llround(bestVal)) < 1e-9){
        cout << (long long)llround(bestVal) << "\n";
    } else {
        cout << setprecision(15) << bestVal << "\n";
    }
    for(int i=0;i<n;i++){
        if(i) cout<<' ';
        cout<<bestX[i];
    }
    cout<<"\n";
    return 0;
}
