OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor %worldPos %normal
               OpExecutionMode %main InputPoints
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

      %int_0 = OpConstant %int 0
     %uint_1 = OpConstant %uint 1
 %darkfactor = OpConstant %float 0.7

       %ox0 = OpConstant %float 0.04
       %oy0 = OpConstant %float 0.0
       %ox1 = OpConstant %float 0.02
       %oy1 = OpConstant %float 0.034641016
       %ox2 = OpConstant %float -0.02
       %oy2 = OpConstant %float 0.034641016
       %ox3 = OpConstant %float -0.04
       %oy3 = OpConstant %float 0.0
       %ox4 = OpConstant %float -0.02
       %oy4 = OpConstant %float -0.034641016
       %ox5 = OpConstant %float 0.02
       %oy5 = OpConstant %float -0.034641016

%gl_PerVertex_in  = OpTypeStruct %v4float
      %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
  %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
          %gl_in = OpVariable %ptr_in_arr_pv Input
      %ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
     %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
         %gl_out = OpVariable %ptr_out_pv Output
     %ptr_out_v4 = OpTypePointer Output %v4float

      %arr_v3_in = OpTypeArray %v3float %uint_1
  %ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
       %worldPos = OpVariable %ptr_in_arr_v3 Input
         %normal = OpVariable %ptr_in_arr_v3 Input
        %inColor = OpVariable %ptr_in_arr_v3 Input
      %ptr_in_v3 = OpTypePointer Input %v3float

     %ptr_out_v3 = OpTypePointer Output %v3float
       %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
        %lbl = OpLabel

       %p_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %center = OpLoad %v4float %p_ptr

       %c_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
        %ccol = OpLoad %v3float %c_ptr

          %dk = OpCompositeConstruct %v3float %darkfactor %darkfactor %darkfactor
        %ecol = OpFMul %v3float %ccol %dk

          %cx = OpCompositeExtract %float %center 0
          %cy = OpCompositeExtract %float %center 1
          %cz = OpCompositeExtract %float %center 2
          %cw = OpCompositeExtract %float %center 3

         %e0x = OpFAdd %float %cx %ox0
         %e0y = OpFAdd %float %cy %oy0
       %edge0 = OpCompositeConstruct %v4float %e0x %e0y %cz %cw

         %e1x = OpFAdd %float %cx %ox1
         %e1y = OpFAdd %float %cy %oy1
       %edge1 = OpCompositeConstruct %v4float %e1x %e1y %cz %cw

         %e2x = OpFAdd %float %cx %ox2
         %e2y = OpFAdd %float %cy %oy2
       %edge2 = OpCompositeConstruct %v4float %e2x %e2y %cz %cw

         %e3x = OpFAdd %float %cx %ox3
         %e3y = OpFAdd %float %cy %oy3
       %edge3 = OpCompositeConstruct %v4float %e3x %e3y %cz %cw

         %e4x = OpFAdd %float %cx %ox4
         %e4y = OpFAdd %float %cy %oy4
       %edge4 = OpCompositeConstruct %v4float %e4x %e4y %cz %cw

         %e5x = OpFAdd %float %cx %ox5
         %e5y = OpFAdd %float %cy %oy5
       %edge5 = OpCompositeConstruct %v4float %e5x %e5y %cz %cw

        %pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos1 %center
               OpStore %outColor %ccol
               OpEmitVertex
        %pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos2 %edge0
               OpStore %outColor %ecol
               OpEmitVertex
        %pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos3 %edge1
               OpStore %outColor %ecol
               OpEmitVertex
               OpEndPrimitive

        %pos4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos4 %center
               OpStore %outColor %ccol
               OpEmitVertex
        %pos5 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos5 %edge1
               OpStore %outColor %ecol
               OpEmitVertex
        %pos6 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos6 %edge2
               OpStore %outColor %ecol
               OpEmitVertex
               OpEndPrimitive

        %pos7 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos7 %center
               OpStore %outColor %ccol
               OpEmitVertex
        %pos8 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos8 %edge2
               OpStore %outColor %ecol
               OpEmitVertex
        %pos9 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos9 %edge3
               OpStore %outColor %ecol
               OpEmitVertex
               OpEndPrimitive

       %pos10 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos10 %center
               OpStore %outColor %ccol
               OpEmitVertex
       %pos11 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos11 %edge3
               OpStore %outColor %ecol
               OpEmitVertex
       %pos12 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos12 %edge4
               OpStore %outColor %ecol
               OpEmitVertex
               OpEndPrimitive

       %pos13 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos13 %center
               OpStore %outColor %ccol
               OpEmitVertex
       %pos14 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos14 %edge4
               OpStore %outColor %ecol
               OpEmitVertex
       %pos15 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos15 %edge5
               OpStore %outColor %ecol
               OpEmitVertex
               OpEndPrimitive

       %pos16 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos16 %center
               OpStore %outColor %ccol
               OpEmitVertex
       %pos17 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos17 %edge5
               OpStore %outColor %ecol
               OpEmitVertex
       %pos18 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos18 %edge0
               OpStore %outColor %ecol
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
