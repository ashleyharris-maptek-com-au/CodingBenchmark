declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Drone { id: number; x: number; y: number; }

const prob: number[][] = data.probability;
const drones: Drone[] = data.drones;
const maxSteps: number = data.maxStepsPerDrone;

const H = prob.length;
const W = H > 0 ? prob[0].length : 0;

function inBounds(x: number, y: number): boolean {
    return x >= 0 && x < W && y >= 0 && y < H;
}

function isObstacle(x: number, y: number): boolean {
    return prob[y][x] === -1;
}

const claimed: Uint8Array = new Uint8Array(W * H);

const DX = [0, 1, 0, -1, 0];
const DY = [-1, 0, 1, 0, 0];

function planDrone(startX: number, startY: number): [number, number][] {
    const path: [number, number][] = [[startX, startY]];
    let cx = startX, cy = startY;
    const localClaimed = new Uint8Array(W * H);
    if (prob[cy][cx] > 0 && !claimed[cy * W + cx]) {
        localClaimed[cy * W + cx] = 1;
    }

    const horizon = Math.min(8, maxSteps);

    for (let step = 0; step < maxSteps; step++) {
        let bestDir = 4;
        let bestScore = -Infinity;

        for (let d = 0; d < 5; d++) {
            const nx = cx + DX[d];
            const ny = cy + DY[d];
            if (!inBounds(nx, ny) || isObstacle(nx, ny)) continue;

            let score = 0;
            const visitedRollout = new Uint8Array(W * H);
            const idx0 = ny * W + nx;
            if (!claimed[idx0] && !localClaimed[idx0] && prob[ny][nx] > 0) {
                score += prob[ny][nx];
            }
            visitedRollout[idx0] = 1;
            let rx = nx, ry = ny;
            for (let h = 1; h < horizon; h++) {
                let bx = rx, by = ry, bv = 0;
                for (let dd = 0; dd < 4; dd++) {
                    const ax = rx + DX[dd];
                    const ay = ry + DY[dd];
                    if (!inBounds(ax, ay) || isObstacle(ax, ay)) continue;
                    const ai = ay * W + ax;
                    if (visitedRollout[ai]) continue;
                    let v = 0;
                    if (!claimed[ai] && !localClaimed[ai] && prob[ay][ax] > 0) {
                        v = prob[ay][ax];
                    }
                    const decay = Math.pow(0.85, h);
                    const sv = v * decay;
                    if (sv > bv) { bv = sv; bx = ax; by = ay; }
                }
                if (bx === rx && by === ry) break;
                rx = bx; ry = by;
                visitedRollout[ry * W + rx] = 1;
                score += bv;
            }

            if (d === 4) score -= 1e-9;

            if (score > bestScore) {
                bestScore = score;
                bestDir = d;
            }
        }

        cx = cx + DX[bestDir];
        cy = cy + DY[bestDir];
        path.push([cx, cy]);
        const idx = cy * W + cx;
        if (prob[cy][cx] > 0 && !claimed[idx] && !localClaimed[idx]) {
            localClaimed[idx] = 1;
        }
    }

    for (let i = 0; i < W * H; i++) {
        if (localClaimed[i]) claimed[i] = 1;
    }

    return path;
}

const paths: [number, number][][] = [];
for (const drone of drones) {
    paths.push(planDrone(drone.x, drone.y));
}

process.stdout.write(JSON.stringify({ paths }));
