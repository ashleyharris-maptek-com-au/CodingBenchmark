#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <vector>
#include <algorithm>
#include <chrono>
#include <random>
#include <climits>
#include <string>

using namespace std;

struct Task { int machine, dur; };

static int J, M;
static vector<vector<Task>> jobs;
static vector<int> jobTaskCount;
static int totalTasks = 0;

static inline int readInt() {
    int c = getchar();
    while (c != EOF && (c < '0' || c > '9') && c != '-') c = getchar();
    int sign = 1;
    if (c == '-') { sign = -1; c = getchar(); }
    int x = 0;
    while (c >= '0' && c <= '9') { x = x*10 + (c - '0'); c = getchar(); }
    return x * sign;
}

struct Schedule {
    vector<vector<int>> starts;
    long long makespan = 0;
};

// Giffler-Thompson active scheduler with priority rule
// rule: 0=MWR, 1=SPT, 2=LPT, 3=LWR, 4=random
static Schedule giffler_thompson(int rule, mt19937 &rng) {
    Schedule S;
    S.starts.assign(J, {});
    for (int j = 0; j < J; ++j) S.starts[j].assign(jobTaskCount[j], 0);

    vector<int> nextOp(J, 0);
    vector<long long> jobReady(J, 0);
    vector<long long> machReady(M, 0);
    vector<long long> remWork(J, 0);
    for (int j = 0; j < J; ++j)
        for (auto &t : jobs[j]) remWork[j] += t.dur;

    int scheduled = 0;
    while (scheduled < totalTasks) {
        long long Cstar = LLONG_MAX;
        int machStar = -1;
        for (int j = 0; j < J; ++j) {
            if (nextOp[j] >= jobTaskCount[j]) continue;
            Task &t = jobs[j][nextOp[j]];
            long long est = max(jobReady[j], machReady[t.machine]);
            long long ect = est + t.dur;
            if (ect < Cstar) { Cstar = ect; machStar = t.machine; }
        }
        if (machStar < 0) break;

        int bestJob = -1;
        double bestKey = -1e18;
        for (int j = 0; j < J; ++j) {
            if (nextOp[j] >= jobTaskCount[j]) continue;
            Task &t = jobs[j][nextOp[j]];
            if (t.machine != machStar) continue;
            long long est = max(jobReady[j], machReady[t.machine]);
            if (est >= Cstar) continue;
            double key;
            switch (rule) {
                case 0: key = (double)remWork[j]; break;
                case 1: key = -(double)t.dur; break;
                case 2: key = (double)t.dur; break;
                case 3: key = -(double)remWork[j]; break;
                default: key = uniform_real_distribution<double>(0,1)(rng); break;
            }
            if (key > bestKey) { bestKey = key; bestJob = j; }
        }
        if (bestJob < 0) {
            // fallback: pick any
            for (int j = 0; j < J; ++j) {
                if (nextOp[j] >= jobTaskCount[j]) continue;
                Task &t = jobs[j][nextOp[j]];
                if (t.machine == machStar) { bestJob = j; break; }
            }
            if (bestJob < 0) break;
        }

        Task &t = jobs[bestJob][nextOp[bestJob]];
        long long est = max(jobReady[bestJob], machReady[t.machine]);
        S.starts[bestJob][nextOp[bestJob]] = (int)est;
        long long finish = est + t.dur;
        jobReady[bestJob] = finish;
        machReady[t.machine] = finish;
        remWork[bestJob] -= t.dur;
        nextOp[bestJob]++;
        scheduled++;
        if (finish > S.makespan) S.makespan = finish;
    }
    return S;
}

int main() {
    auto t0 = chrono::steady_clock::now();
    J = readInt(); M = readInt();
    jobs.assign(J, {});
    jobTaskCount.assign(J, 0);
    for (int j = 0; j < J; ++j) {
        int T = readInt();
        jobTaskCount[j] = T;
        jobs[j].reserve(T);
        for (int k = 0; k < T; ++k) {
            int m = readInt();
            int d = readInt();
            jobs[j].push_back({m, d});
            totalTasks++;
        }
    }

    // Adaptive budget based on scale
    double budget;
    if (totalTasks <= 200) budget = 2.0;
    else if (totalTasks <= 5000) budget = 5.0;
    else if (totalTasks <= 50000) budget = 10.0;
    else budget = 15.0;

    mt19937 rng(2026);

    Schedule best;
    best.makespan = LLONG_MAX;

    // Try deterministic priority rules first
    int rulesToTry = (totalTasks > 50000) ? 2 : 4;
    for (int r = 0; r < rulesToTry; ++r) {
        double elapsed = chrono::duration<double>(chrono::steady_clock::now() - t0).count();
        if (elapsed > budget * 0.5) break;
        Schedule S = giffler_thompson(r, rng);
        if (S.makespan < best.makespan) best = move(S);
    }

    // Random restarts
    int maxRestarts;
    if (totalTasks <= 200) maxRestarts = 500;
    else if (totalTasks <= 5000) maxRestarts = 80;
    else if (totalTasks <= 50000) maxRestarts = 15;
    else maxRestarts = 3;

    for (int it = 0; it < maxRestarts; ++it) {
        double elapsed = chrono::duration<double>(chrono::steady_clock::now() - t0).count();
        if (elapsed > budget * 0.92) break;
        Schedule S = giffler_thompson(4, rng);
        if (S.makespan < best.makespan) best = move(S);
    }

    // Output
    string out;
    out.reserve((size_t)totalTasks * 12);
    char buf[64];
    for (int j = 0; j < J; ++j) {
        for (int k = 0; k < jobTaskCount[j]; ++k) {
            if (k) out.push_back(' ');
            int len = snprintf(buf, sizeof(buf), "%d %d", best.starts[j][k], jobs[j][k].machine);
            out.append(buf, len);
        }
        out.push_back('\n');
    }
    fwrite(out.data(), 1, out.size(), stdout);
    return 0;
}
