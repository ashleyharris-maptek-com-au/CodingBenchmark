use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let n: usize = match it.next() { Some(s) => s.parse().unwrap_or(0), None => 0 };
    let m: usize = match it.next() { Some(s) => s.parse().unwrap_or(0), None => 0 };

    if n == 0 { return; }

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u: u32 = match it.next() { Some(s) => s.parse().unwrap_or(0), None => break };
        let v: u32 = match it.next() { Some(s) => s.parse().unwrap_or(0), None => break };
        if (u as usize) >= n || (v as usize) >= n { continue; }
        if u == v { continue; }
        edges.push((u, v));
    }

    let mut x: Vec<f64> = vec![0.0; n];
    let mut y: Vec<f64> = vec![0.0; n];

    let mut seed: u64 = 0x9E3779B97F4A7C15;
    let mut rng = || -> f64 {
        seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        ((seed >> 33) as u32 as f64) / (u32::MAX as f64)
    };

    let side = (n as f64).sqrt().max(1.0);
    for i in 0..n {
        x[i] = rng() * side;
        y[i] = rng() * side;
    }

    if n == 1 {
        writeln!(out, "1").unwrap();
        writeln!(out, "0.0 0.0").unwrap();
        return;
    }

    let iters: usize = if n <= 50 {
        500
    } else if n <= 200 {
        300
    } else if n <= 1000 {
        150
    } else if n <= 10000 {
        80
    } else if n <= 50000 {
        40
    } else {
        20
    };

    let k = side / (n as f64).sqrt().max(1.0);
    let k2 = k * k;
    let mut t = side / 10.0;
    let cool = (t - 0.01) / (iters as f64).max(1.0);

    let mut dx_arr = vec![0.0f64; n];
    let mut dy_arr = vec![0.0f64; n];

    let use_grid = n > 500;

    for _iter in 0..iters {
        for i in 0..n {
            dx_arr[i] = 0.0;
            dy_arr[i] = 0.0;
        }

        if use_grid {
            let cell_size = 2.0 * k;
            let mut min_x = f64::INFINITY;
            let mut min_y = f64::INFINITY;
            let mut max_x = f64::NEG_INFINITY;
            let mut max_y = f64::NEG_INFINITY;
            for i in 0..n {
                if x[i] < min_x { min_x = x[i]; }
                if y[i] < min_y { min_y = y[i]; }
                if x[i] > max_x { max_x = x[i]; }
                if y[i] > max_y { max_y = y[i]; }
            }
            let w = (((max_x - min_x) / cell_size).ceil() as usize + 1).max(1);
            let h = (((max_y - min_y) / cell_size).ceil() as usize + 1).max(1);
            let total = w.saturating_mul(h);
            if total < 50_000_000 {
                let mut grid: Vec<Vec<u32>> = vec![Vec::new(); w * h];
                for i in 0..n {
                    let cx = (((x[i] - min_x) / cell_size) as usize).min(w - 1);
                    let cy = (((y[i] - min_y) / cell_size) as usize).min(h - 1);
                    grid[cy * w + cx].push(i as u32);
                }
                for i in 0..n {
                    let cx = (((x[i] - min_x) / cell_size) as usize).min(w - 1);
                    let cy = (((y[i] - min_y) / cell_size) as usize).min(h - 1);
                    let x0 = cx.saturating_sub(1);
                    let x1 = (cx + 1).min(w - 1);
                    let y0 = cy.saturating_sub(1);
                    let y1 = (cy + 1).min(h - 1);
                    let xi = x[i];
                    let yi = y[i];
                    let mut ddx = 0.0;
                    let mut ddy = 0.0;
                    for gy in y0..=y1 {
                        for gx in x0..=x1 {
                            for &jn in &grid[gy * w + gx] {
                                let j = jn as usize;
                                if j == i { continue; }
                                let dx = xi - x[j];
                                let dy = yi - y[j];
                                let mut d2 = dx * dx + dy * dy;
                                if d2 < 1e-9 { d2 = 1e-9; }
                                let f = k2 / d2;
                                ddx += dx * f;
                                ddy += dy * f;
                            }
                        }
                    }
                    dx_arr[i] += ddx;
                    dy_arr[i] += ddy;
                }
            }
        } else {
            for i in 0..n {
                let xi = x[i];
                let yi = y[i];
                let mut ddx = 0.0;
                let mut ddy = 0.0;
                for j in 0..n {
                    if i == j { continue; }
                    let dx = xi - x[j];
                    let dy = yi - y[j];
                    let mut d2 = dx * dx + dy * dy;
                    if d2 < 1e-9 { d2 = 1e-9; }
                    let f = k2 / d2;
                    ddx += dx * f;
                    ddy += dy * f;
                }
                dx_arr[i] = ddx;
                dy_arr[i] = ddy;
            }
        }

        for &(u, v) in &edges {
            let ui = u as usize;
            let vi = v as usize;
            let dx = x[ui] - x[vi];
            let dy = y[ui] - y[vi];
            let d = (dx * dx + dy * dy).sqrt().max(1e-9);
            let f = (d * d) / k;
            let fx = (dx / d) * f;
            let fy = (dy / d) * f;
            dx_arr[ui] -= fx;
            dy_arr[ui] -= fy;
            dx_arr[vi] += fx;
            dy_arr[vi] += fy;
        }

        for i in 0..n {
            let d = (dx_arr[i] * dx_arr[i] + dy_arr[i] * dy_arr[i]).sqrt().max(1e-9);
            let lim = d.min(t);
            x[i] += dx_arr[i] / d * lim;
            y[i] += dy_arr[i] / d * lim;
        }

        t = (t - cool).max(0.01);
    }

    writeln!(out, "{}", n).unwrap();
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", x[i], y[i]).unwrap();
    }
}
