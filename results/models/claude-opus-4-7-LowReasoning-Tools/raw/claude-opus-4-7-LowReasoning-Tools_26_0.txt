use std::io::{self, Read, Write, BufReader, BufWriter};

const POINT_SIZE: usize = 12;
const OUT_SIZE: usize = 4;

#[inline]
fn dist2(a: &[f32; 3], b: &[f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

#[inline]
fn assign(p: &[f32; 3], centroids: &[[f32; 3]]) -> u32 {
    let mut best = 0u32;
    let mut best_d = f32::INFINITY;
    for (i, c) in centroids.iter().enumerate() {
        let d = dist2(p, c);
        if d < best_d {
            best_d = d;
            best = i as u32;
        }
    }
    best
}

struct Rng(u64);
impl Rng {
    fn new(seed: u64) -> Self { Rng(seed | 1) }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.0;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.0 = x;
        x
    }
    fn next_usize(&mut self, n: usize) -> usize {
        (self.next_u64() as usize) % n
    }
}

fn read_header<R: Read>(r: &mut R) -> io::Result<(u64, u32)> {
    let mut header = String::new();
    let mut byte = [0u8; 1];
    loop {
        let n = r.read(&mut byte)?;
        if n == 0 { break; }
        if byte[0] == b'\n' { break; }
        header.push(byte[0] as char);
    }
    let parts: Vec<&str> = header.split_whitespace().collect();
    let np: u64 = parts[0].parse().unwrap();
    let nc: u32 = parts[1].parse().unwrap();
    Ok((np, nc))
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());
    let (num_points, num_clusters) = read_header(&mut reader)?;
    let k = num_clusters as usize;
    let n = num_points as usize;

    let stdout = io::stdout();
    let mut writer = BufWriter::with_capacity(1 << 20, stdout.lock());

    let in_memory_threshold: usize = 20_000_000;

    if n == 0 {
        return Ok(());
    }

    let mut rng = Rng::new(0x9E3779B97F4A7C15);

    if n <= in_memory_threshold {
        let mut points: Vec<[f32; 3]> = Vec::with_capacity(n);
        let batch = 8192.min(n);
        let mut buf = vec![0u8; POINT_SIZE * batch];
        let mut remaining = n;
        while remaining > 0 {
            let chunk = remaining.min(buf.len() / POINT_SIZE);
            let bytes = chunk * POINT_SIZE;
            reader.read_exact(&mut buf[..bytes])?;
            for i in 0..chunk {
                let off = i * POINT_SIZE;
                let x = f32::from_le_bytes([buf[off], buf[off+1], buf[off+2], buf[off+3]]);
                let y = f32::from_le_bytes([buf[off+4], buf[off+5], buf[off+6], buf[off+7]]);
                let z = f32::from_le_bytes([buf[off+8], buf[off+9], buf[off+10], buf[off+11]]);
                points.push([x, y, z]);
            }
            remaining -= chunk;
        }

        let kk = k.min(points.len());
        let mut centroids: Vec<[f32; 3]> = Vec::with_capacity(kk);
        for _ in 0..kk {
            let idx = rng.next_usize(points.len());
            centroids.push(points[idx]);
        }

        let max_iters = if n > 1_000_000 { 10 } else if n > 100_000 { 20 } else { 50 };
        let mut sums = vec![[0f64; 3]; kk];
        let mut counts = vec![0u64; kk];

        for _iter in 0..max_iters {
            for s in sums.iter_mut() { *s = [0.0; 3]; }
            for c in counts.iter_mut() { *c = 0; }
            for p in points.iter() {
                let a = assign(p, &centroids) as usize;
                sums[a][0] += p[0] as f64;
                sums[a][1] += p[1] as f64;
                sums[a][2] += p[2] as f64;
                counts[a] += 1;
            }
            let mut moved = 0.0f32;
            for i in 0..kk {
                if counts[i] > 0 {
                    let c = counts[i] as f64;
                    let nc = [(sums[i][0] / c) as f32, (sums[i][1] / c) as f32, (sums[i][2] / c) as f32];
                    moved += dist2(&centroids[i], &nc);
                    centroids[i] = nc;
                } else {
                    let idx = rng.next_usize(points.len());
                    centroids[i] = points[idx];
                }
            }
            if moved < 1e-6 { break; }
        }

        for p in points.iter() {
            let a = assign(p, &centroids);
            writer.write_all(&a.to_le_bytes())?;
        }
    } else {
        let kk = k;
        let mut centroids: Vec<[f32; 3]> = Vec::with_capacity(kk);
        let mut counts: Vec<u64> = vec![0; kk];

        let batch_size: usize = 8192;
        let mut buf = vec![0u8; POINT_SIZE * batch_size];
        let mut out_buf: Vec<u8> = Vec::with_capacity(OUT_SIZE * batch_size);

        let mut remaining = n;
        let warmup_target = (kk * 4).max(100_000).min(n);
        let mut warmup: Vec<[f32; 3]> = Vec::with_capacity(warmup_target);

        while warmup.len() < warmup_target && remaining > 0 {
            let chunk = remaining.min(batch_size).min(warmup_target - warmup.len());
            let bytes = chunk * POINT_SIZE;
            reader.read_exact(&mut buf[..bytes])?;
            for i in 0..chunk {
                let off = i * POINT_SIZE;
                let x = f32::from_le_bytes([buf[off], buf[off+1], buf[off+2], buf[off+3]]);
                let y = f32::from_le_bytes([buf[off+4], buf[off+5], buf[off+6], buf[off+7]]);
                let z = f32::from_le_bytes([buf[off+8], buf[off+9], buf[off+10], buf[off+11]]);
                warmup.push([x, y, z]);
            }
            remaining -= chunk;
        }

        let kk_eff = kk.min(warmup.len());
        for _ in 0..kk_eff {
            let idx = rng.next_usize(warmup.len());
            centroids.push(warmup[idx]);
        }
        while centroids.len() < kk {
            centroids.push([0.0, 0.0, 0.0]);
        }

        let mut sums = vec![[0f64; 3]; kk];
        let mut cnts = vec![0u64; kk];
        for _iter in 0..8 {
            for s in sums.iter_mut() { *s = [0.0; 3]; }
            for c in cnts.iter_mut() { *c = 0; }
            for p in warmup.iter() {
                let a = assign(p, &centroids) as usize;
                sums[a][0] += p[0] as f64;
                sums[a][1] += p[1] as f64;
                sums[a][2] += p[2] as f64;
                cnts[a] += 1;
            }
            for i in 0..kk {
                if cnts[i] > 0 {
                    let c = cnts[i] as f64;
                    centroids[i] = [(sums[i][0] / c) as f32, (sums[i][1] / c) as f32, (sums[i][2] / c) as f32];
                } else if !warmup.is_empty() {
                    let idx = rng.next_usize(warmup.len());
                    centroids[i] = warmup[idx];
                }
            }
        }

        for p in warmup.iter() {
            let a = assign(p, &centroids);
            out_buf.extend_from_slice(&a.to_le_bytes());
            counts[a as usize] += 1;
            if out_buf.len() >= OUT_SIZE * batch_size {
                writer.write_all(&out_buf)?;
                out_buf.clear();
            }
        }
        drop(warmup);

        while remaining > 0 {
            let chunk = remaining.min(batch_size);
            let bytes = chunk * POINT_SIZE;
            reader.read_exact(&mut buf[..bytes])?;
            for i in 0..chunk {
                let off = i * POINT_SIZE;
                let x = f32::from_le_bytes([buf[off], buf[off+1], buf[off+2], buf[off+3]]);
                let y = f32::from_le_bytes([buf[off+4], buf[off+5], buf[off+6], buf[off+7]]);
                let z = f32::from_le_bytes([buf[off+8], buf[off+9], buf[off+10], buf[off+11]]);
                let p = [x, y, z];
                let a = assign(&p, &centroids);
                let ai = a as usize;
                counts[ai] += 1;
                let lr = (1.0 / (counts[ai] as f32).sqrt()).min(0.5);
                let c = &mut centroids[ai];
                c[0] += lr * (p[0] - c[0]);
                c[1] += lr * (p[1] - c[1]);
                c[2] += lr * (p[2] - c[2]);
                out_buf.extend_from_slice(&a.to_le_bytes());
            }
            if out_buf.len() >= OUT_SIZE * batch_size {
                writer.write_all(&out_buf)?;
                out_buf.clear();
            }
            remaining -= chunk;
        }
        if !out_buf.is_empty() {
            writer.write_all(&out_buf)?;
        }
    }

    writer.flush()?;
    Ok(())
}
