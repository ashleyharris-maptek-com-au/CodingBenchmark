#include <cstdio>
#include <vector>
#include <queue>
#include <cstdint>
#include <algorithm>
#include <utility>
#include <tuple>
#include <functional>

using namespace std;

typedef long long ll;
typedef pair<ll,int> PLI;

static const ll INF = (ll)1e18;

int N, M, K;
struct Edge { int u, v; ll w; };
vector<Edge> edges;
vector<vector<tuple<int,ll,int>>> adjE;

int main() {
    if (scanf("%d %d %d", &N, &M, &K) != 3) return 0;
    edges.resize(M);
    adjE.assign(N, {});
    for (int i = 0; i < M; i++) {
        int u, v; ll w;
        scanf("%d %d %lld", &u, &v, &w);
        edges[i] = {u, v, w};
        adjE[u].push_back({v, w, i});
        adjE[v].push_back({u, w, i});
    }
    vector<int> terms(K);
    for (int i = 0; i < K; i++) scanf("%d", &terms[i]);

    if (K <= 1) {
        printf("0 0\n");
        return 0;
    }

    vector<char> inTree(N, 0);
    vector<char> isTerm(N, 0);
    for (int t : terms) isTerm[t] = 1;

    vector<char> usedEdge(M, 0);
    int start = terms[0];
    inTree[start] = 1;
    int remaining = K - 1;

    while (remaining > 0) {
        vector<ll> dist(N, INF);
        vector<int> parEdge(N, -1);
        priority_queue<PLI, vector<PLI>, greater<PLI>> pq;
        for (int i = 0; i < N; i++) {
            if (inTree[i]) {
                dist[i] = 0;
                pq.push({0, i});
            }
        }
        int found = -1;
        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d > dist[u]) continue;
            if (isTerm[u] && !inTree[u]) { found = u; break; }
            for (auto& [v, w, eid] : adjE[u]) {
                ll nd = d + w;
                if (nd < dist[v]) {
                    dist[v] = nd;
                    parEdge[v] = eid;
                    pq.push({nd, v});
                }
            }
        }
        if (found < 0) break;
        int cur = found;
        while (!inTree[cur]) {
            int eid = parEdge[cur];
            usedEdge[eid] = 1;
            inTree[cur] = 1;
            int other = (edges[eid].u == cur) ? edges[eid].v : edges[eid].u;
            cur = other;
        }
        remaining--;
    }

    vector<int> eids;
    for (int i = 0; i < M; i++) if (usedEdge[i]) eids.push_back(i);
    sort(eids.begin(), eids.end(), [&](int a, int b){ return edges[a].w < edges[b].w; });
    vector<int> par(N), rnk(N, 0);
    for (int i = 0; i < N; i++) par[i] = i;
    function<int(int)> find = [&](int x){ while (par[x] != x) { par[x] = par[par[x]]; x = par[x]; } return x; };
    vector<int> mstEdges;
    for (int eid : eids) {
        int a = find(edges[eid].u), b = find(edges[eid].v);
        if (a != b) {
            if (rnk[a] < rnk[b]) swap(a, b);
            par[b] = a;
            if (rnk[a] == rnk[b]) rnk[a]++;
            mstEdges.push_back(eid);
        }
    }

    vector<vector<pair<int,int>>> tadj(N);
    vector<char> edgeAlive(M, 0);
    for (int eid : mstEdges) {
        edgeAlive[eid] = 1;
        tadj[edges[eid].u].push_back({edges[eid].v, eid});
        tadj[edges[eid].v].push_back({edges[eid].u, eid});
    }
    vector<int> deg(N, 0);
    for (int eid : mstEdges) { deg[edges[eid].u]++; deg[edges[eid].v]++; }
    queue<int> q;
    for (int i = 0; i < N; i++) {
        if (deg[i] == 1 && !isTerm[i]) q.push(i);
    }
    while (!q.empty()) {
        int u = q.front(); q.pop();
        if (deg[u] != 1 || isTerm[u]) continue;
        for (auto& [v, eid] : tadj[u]) {
            if (edgeAlive[eid]) {
                edgeAlive[eid] = 0;
                deg[u]--;
                deg[v]--;
                if (deg[v] == 1 && !isTerm[v]) q.push(v);
                break;
            }
        }
    }

    ll total = 0;
    vector<int> finalEdges;
    for (int eid : mstEdges) {
        if (edgeAlive[eid]) {
            finalEdges.push_back(eid);
            total += edges[eid].w;
        }
    }

    printf("%lld %d\n", total, (int)finalEdges.size());
    for (int eid : finalEdges) {
        printf("%d %d\n", edges[eid].u, edges[eid].v);
    }
    return 0;
}
