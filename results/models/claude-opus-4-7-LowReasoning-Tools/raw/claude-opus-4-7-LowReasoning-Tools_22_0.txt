using System;
using System.Collections.Generic;
using System.IO;

class Tetris
{
    static int W, H;
    static int[] board;
    static int fullMask;
    static int[][][][] shapes;
    static int[][] widths;

    static void Main()
    {
        InitShapes();
        var tokens = new Queue<string>();
        Func<string> nextTok = () => {
            while (tokens.Count == 0)
            {
                string line = Console.In.ReadLine();
                if (line == null) return null;
                foreach (var t in line.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries))
                    tokens.Enqueue(t);
            }
            return tokens.Dequeue();
        };

        W = int.Parse(nextTok());
        H = int.Parse(nextTok());
        int numPieces = int.Parse(nextTok());
        fullMask = (1 << W) - 1;
        board = new int[H];

        var outW = new StreamWriter(Console.OpenStandardOutput());
        outW.AutoFlush = true;

        for (int p = 0; p < numPieces; p++)
        {
            string pt = nextTok();
            if (pt == null) break;
            int type = TypeIndex(pt);
            var best = ChooseMove(type);
            outW.WriteLine(best.rot + " " + best.col);
            string res = nextTok();
            if (res == null) break;
            if (res == "gameover") break;
            ApplyPiece(type, best.rot, best.col);
            ClearLinesOn(board);
        }
    }

    static int TypeIndex(string s)
    {
        switch (s)
        {
            case "I": return 0;
            case "O": return 1;
            case "T": return 2;
            case "S": return 3;
            case "Z": return 4;
            case "J": return 5;
            case "L": return 6;
        }
        return 0;
    }

    static void InitShapes()
    {
        var bases = new int[7][][]
        {
            new[]{ new[]{0,0}, new[]{1,0}, new[]{2,0}, new[]{3,0} },
            new[]{ new[]{0,0}, new[]{1,0}, new[]{0,1}, new[]{1,1} },
            new[]{ new[]{0,0}, new[]{1,0}, new[]{2,0}, new[]{1,1} },
            new[]{ new[]{1,0}, new[]{2,0}, new[]{0,1}, new[]{1,1} },
            new[]{ new[]{0,0}, new[]{1,0}, new[]{1,1}, new[]{2,1} },
            new[]{ new[]{0,0}, new[]{0,1}, new[]{1,1}, new[]{2,1} },
            new[]{ new[]{2,0}, new[]{0,1}, new[]{1,1}, new[]{2,1} },
        };

        shapes = new int[7][][][];
        widths = new int[7][];
        for (int t = 0; t < 7; t++)
        {
            shapes[t] = new int[4][][];
            widths[t] = new int[4];
            var cur = bases[t];
            for (int r = 0; r < 4; r++)
            {
                int minX = int.MaxValue, minY = int.MaxValue, maxX = int.MinValue;
                foreach (var c in cur) { if (c[0] < minX) minX = c[0]; if (c[1] < minY) minY = c[1]; if (c[0] > maxX) maxX = c[0]; }
                var norm = new int[cur.Length][];
                for (int i = 0; i < cur.Length; i++) norm[i] = new[] { cur[i][0] - minX, cur[i][1] - minY };
                shapes[t][r] = norm;
                widths[t][r] = maxX - minX + 1;

                int mY = int.MinValue;
                foreach (var c in cur) if (c[1] > mY) mY = c[1];
                var nxt = new int[cur.Length][];
                for (int i = 0; i < cur.Length; i++) nxt[i] = new[] { mY - cur[i][1], cur[i][0] };
                cur = nxt;
            }
        }
    }

    struct Move { public int rot; public int col; }

    static Move ChooseMove(int type)
    {
        double bestScore = double.NegativeInfinity;
        Move bestMove = new Move { rot = 0, col = 0 };
        bool any = false;
        int rotMax = type == 1 ? 1 : (type == 0 || type == 3 || type == 4 ? 2 : 4);

        for (int r = 0; r < rotMax; r++)
        {
            int w = widths[type][r];
            for (int c = 0; c + w <= W; c++)
            {
                int top = DropPiece(type, r, c, board);
                if (top < 0) continue;
                int[] sim = (int[])board.Clone();
                PlacePieceOn(sim, type, r, c, top);
                int cleared = ClearLinesOn(sim);
                double score = Evaluate(sim, cleared);
                if (!any || score > bestScore)
                {
                    bestScore = score;
                    bestMove = new Move { rot = r, col = c };
                    any = true;
                }
            }
        }
        return bestMove;
    }

    static int DropPiece(int type, int rot, int col, int[] bd)
    {
        var cells = shapes[type][rot];
        int top = int.MaxValue;
        int wPiece = widths[type][rot];
        int[] colMaxDy = new int[wPiece];
        bool[] colHas = new bool[wPiece];
        for (int i = 0; i < cells.Length; i++)
        {
            int dx = cells[i][0];
            if (!colHas[dx] || cells[i][1] > colMaxDy[dx]) { colMaxDy[dx] = cells[i][1]; colHas[dx] = true; }
        }
        for (int dx = 0; dx < wPiece; dx++)
        {
            if (!colHas[dx]) continue;
            int boardCol = col + dx;
            int topFilled = H;
            for (int row = 0; row < H; row++)
                if ((bd[row] & (1 << boardCol)) != 0) { topFilled = row; break; }
            int candidateTop = topFilled - colMaxDy[dx] - 1;
            if (candidateTop < top) top = candidateTop;
        }
        if (top < 0) return -1;
        foreach (var cc in cells)
        {
            int rr = top + cc[1];
            if (rr < 0 || rr >= H) return -1;
            int boardCol = col + cc[0];
            if ((bd[rr] & (1 << boardCol)) != 0) return -1;
        }
        return top;
    }

    static void PlacePieceOn(int[] bd, int type, int rot, int col, int top)
    {
        var cells = shapes[type][rot];
        foreach (var c in cells)
        {
            int r = top + c[1];
            int bc = col + c[0];
            bd[r] |= (1 << bc);
        }
    }

    static int ClearLinesOn(int[] bd)
    {
        int cleared = 0;
        int wr = H - 1;
        for (int r = H - 1; r >= 0; r--)
        {
            if (bd[r] == fullMask) { cleared++; continue; }
            bd[wr--] = bd[r];
        }
        for (int r = wr; r >= 0; r--) bd[r] = 0;
        return cleared;
    }

    static void ApplyPiece(int type, int rot, int col)
    {
        int top = DropPiece(type, rot, col, board);
        if (top < 0) return;
        PlacePieceOn(board, type, rot, col, top);
    }

    static double Evaluate(int[] bd, int cleared)
    {
        int[] heights = new int[W];
        int aggH = 0, maxH = 0;
        for (int c = 0; c < W; c++)
        {
            int h = 0;
            for (int r = 0; r < H; r++)
                if ((bd[r] & (1 << c)) != 0) { h = H - r; break; }
            heights[c] = h;
            aggH += h;
            if (h > maxH) maxH = h;
        }
        int holes = 0;
        for (int c = 0; c < W; c++)
        {
            bool seen = false;
            for (int r = 0; r < H; r++)
            {
                bool filled = (bd[r] & (1 << c)) != 0;
                if (filled) seen = true;
                else if (seen) holes++;
            }
        }
        int bumpiness = 0;
        for (int c = 0; c < W - 1; c++) bumpiness += Math.Abs(heights[c] - heights[c + 1]);

        int wells = 0;
        for (int c = 0; c < W; c++)
        {
            int left = c == 0 ? H : heights[c - 1];
            int right = c == W - 1 ? H : heights[c + 1];
            int minN = Math.Min(left, right);
            if (heights[c] < minN) wells += (minN - heights[c]);
        }

        double score = 0;
        score += -0.510066 * aggH;
        score += 0.760666 * cleared;
        score += -0.35663 * holes;
        score += -0.184483 * bumpiness;
        score += -0.15 * wells;
        score += -0.05 * maxH;
        return score;
    }
}
