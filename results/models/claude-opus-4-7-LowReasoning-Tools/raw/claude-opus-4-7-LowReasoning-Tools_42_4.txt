; SPIR-V
; Version: 1.0
               OpCapability Geometry
       %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_out %gl_in %inWorldPos %inNormal %inColor %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2
    %f_zero  = OpConstant %float 0.0
    %f_one   = OpConstant %float 1.0
    %f_ts    = OpConstant %float 0.04
    %f_te    = OpConstant %float 0.005
    %f_eps   = OpConstant %float 0.00001

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
 %inWorldPos = OpVariable %ptr_in_arr_v3 Input
   %inNormal = OpVariable %ptr_in_arr_v3 Input
    %inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

      %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %p0     = OpLoad %v4float %p0_ptr
      %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
      %p1     = OpLoad %v4float %p1_ptr

      %c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
      %c0     = OpLoad %v3float %c0_ptr
      %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
      %c1     = OpLoad %v3float %c1_ptr

      %p0xy   = OpVectorShuffle %v2float %p0 %p0 0 1
      %p1xy   = OpVectorShuffle %v2float %p1 %p1 0 1
      %dxy    = OpFSub %v2float %p1xy %p0xy
      %dx     = OpCompositeExtract %float %dxy 0
      %dy     = OpCompositeExtract %float %dxy 1
      %ndy    = OpFNegate %float %dy
      %perp_raw = OpCompositeConstruct %v2float %ndy %dx
      %len2   = OpDot %float %perp_raw %perp_raw
      %len    = OpExtInst %float %glsl Sqrt %len2
      %lensafe = OpExtInst %float %glsl FMax %len %f_eps
      %invlen = OpFDiv %float %f_one %lensafe
      %perp   = OpVectorTimesScalar %v2float %perp_raw %invlen

      %off_s  = OpVectorTimesScalar %v2float %perp %f_ts
      %off_e  = OpVectorTimesScalar %v2float %perp %f_te
      %off_s_x = OpCompositeExtract %float %off_s 0
      %off_s_y = OpCompositeExtract %float %off_s 1
      %off_e_x = OpCompositeExtract %float %off_e 0
      %off_e_y = OpCompositeExtract %float %off_e 1
      %off_s4 = OpCompositeConstruct %v4float %off_s_x %off_s_y %f_zero %f_zero
      %off_e4 = OpCompositeConstruct %v4float %off_e_x %off_e_y %f_zero %f_zero

      %v0     = OpFAdd %v4float %p0 %off_s4
      %v0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %v0_ptr %v0
                OpStore %outColor %c0
                OpEmitVertex

      %v1     = OpFSub %v4float %p0 %off_s4
      %v1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %v1_ptr %v1
                OpStore %outColor %c0
                OpEmitVertex

      %v2v    = OpFAdd %v4float %p1 %off_e4
      %v2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %v2_ptr %v2v
                OpStore %outColor %c1
                OpEmitVertex

      %v3v    = OpFSub %v4float %p1 %off_e4
      %v3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %v3_ptr %v3v
                OpStore %outColor %c1
                OpEmitVertex

                OpEndPrimitive
                OpReturn
                OpFunctionEnd
