using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static Dictionary<int, HashSet<int>> outAdj;
    static Dictionary<int, HashSet<int>> inAdj;
    static HashSet<int> bestSolution;

    static void Main()
    {
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int n = int.Parse(tokens[idx++]);
        int m = int.Parse(tokens[idx++]);
        outAdj = new Dictionary<int, HashSet<int>>();
        inAdj = new Dictionary<int, HashSet<int>>();
        for (int i = 0; i < n; i++)
        {
            outAdj[i] = new HashSet<int>();
            inAdj[i] = new HashSet<int>();
        }
        var solution = new HashSet<int>();
        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(tokens[idx++]);
            int v = int.Parse(tokens[idx++]);
            if (u == v)
            {
                if (outAdj.ContainsKey(u)) { RemoveVertex(outAdj, inAdj, u); solution.Add(u); }
            }
            else
            {
                if (outAdj.ContainsKey(u) && outAdj.ContainsKey(v))
                {
                    outAdj[u].Add(v);
                    inAdj[v].Add(u);
                }
            }
        }

        ReduceLocal(outAdj, inAdj, solution);

        var sccs = ComputeSCCs(outAdj);
        foreach (var scc in sccs)
        {
            if (scc.Count <= 1) continue;
            var sccSet = new HashSet<int>(scc);
            var subOut = new Dictionary<int, HashSet<int>>();
            var subIn = new Dictionary<int, HashSet<int>>();
            foreach (var v in scc) { subOut[v] = new HashSet<int>(); subIn[v] = new HashSet<int>(); }
            foreach (var u in scc)
                foreach (var v in outAdj[u])
                    if (sccSet.Contains(v))
                    {
                        subOut[u].Add(v);
                        subIn[v].Add(u);
                    }
            var sub = SolveSCC(subOut, subIn);
            foreach (var v in sub) solution.Add(v);
        }

        var sb = new StringBuilder();
        sb.Append(solution.Count).Append('\n');
        bool first = true;
        foreach (var v in solution)
        {
            if (!first) sb.Append(' ');
            sb.Append(v);
            first = false;
        }
        sb.Append('\n');
        Console.Write(sb.ToString());
    }

    static void RemoveVertex(Dictionary<int, HashSet<int>> g, Dictionary<int, HashSet<int>> gi, int v)
    {
        if (!g.ContainsKey(v)) return;
        foreach (var w in g[v]) gi[w].Remove(v);
        foreach (var u in gi[v]) g[u].Remove(v);
        g.Remove(v); gi.Remove(v);
    }

    static void ReduceLocal(Dictionary<int, HashSet<int>> g, Dictionary<int, HashSet<int>> gi, HashSet<int> sol)
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            var keys = new List<int>(g.Keys);
            foreach (var v in keys)
            {
                if (!g.ContainsKey(v)) continue;
                if (g[v].Contains(v)) { sol.Add(v); RemoveVertex(g, gi, v); changed = true; continue; }
                if (gi[v].Count == 0 || g[v].Count == 0) { RemoveVertex(g, gi, v); changed = true; continue; }
                if (gi[v].Count == 1)
                {
                    int u = -1; foreach (var x in gi[v]) { u = x; break; }
                    foreach (var w in new List<int>(g[v])) { if (w == u) continue; g[u].Add(w); gi[w].Add(u); }
                    RemoveVertex(g, gi, v); changed = true; continue;
                }
                if (g[v].Count == 1)
                {
                    int w = -1; foreach (var x in g[v]) { w = x; break; }
                    foreach (var u in new List<int>(gi[v])) { if (u == w) continue; g[u].Add(w); gi[w].Add(u); }
                    RemoveVertex(g, gi, v); changed = true; continue;
                }
            }
        }
    }

    static List<List<int>> ComputeSCCs(Dictionary<int, HashSet<int>> g)
    {
        var index = new Dictionary<int, int>();
        var low = new Dictionary<int, int>();
        var onStack = new HashSet<int>();
        var stack = new Stack<int>();
        int idxC = 0;
        var result = new List<List<int>>();
        var verts = new List<int>(g.Keys);
        foreach (var start in verts)
        {
            if (index.ContainsKey(start)) continue;
            var work = new Stack<(int v, IEnumerator<int> it)>();
            index[start] = idxC; low[start] = idxC; idxC++;
            stack.Push(start); onStack.Add(start);
            work.Push((start, g[start].GetEnumerator()));
            while (work.Count > 0)
            {
                var (v, it) = work.Peek();
                bool advanced = false;
                while (it.MoveNext())
                {
                    int w = it.Current;
                    if (!g.ContainsKey(w)) continue;
                    if (!index.ContainsKey(w))
                    {
                        index[w] = idxC; low[w] = idxC; idxC++;
                        stack.Push(w); onStack.Add(w);
                        work.Push((w, g[w].GetEnumerator()));
                        advanced = true;
                        break;
                    }
                    else if (onStack.Contains(w))
                    {
                        if (index[w] < low[v]) low[v] = index[w];
                    }
                }
                if (advanced) continue;
                work.Pop();
                if (low[v] == index[v])
                {
                    var comp = new List<int>();
                    while (true)
                    {
                        int w = stack.Pop();
                        onStack.Remove(w);
                        comp.Add(w);
                        if (w == v) break;
                    }
                    result.Add(comp);
                }
                if (work.Count > 0)
                {
                    var parent = work.Peek().v;
                    if (low[v] < low[parent]) low[parent] = low[v];
                }
            }
        }
        return result;
    }

    static Dictionary<int, HashSet<int>> Clone(Dictionary<int, HashSet<int>> a)
    {
        var r = new Dictionary<int, HashSet<int>>(a.Count);
        foreach (var kv in a) r[kv.Key] = new HashSet<int>(kv.Value);
        return r;
    }

    static List<int> GreedyDFVS(Dictionary<int, HashSet<int>> g, Dictionary<int, HashSet<int>> gi)
    {
        var solList = new List<int>();
        var solSet = new HashSet<int>();
        while (true)
        {
            ReduceLocal(g, gi, solSet);
            foreach (var v in solSet) if (!solList.Contains(v)) solList.Add(v);
            if (g.Count == 0 || IsDAG(g)) break;
            int bestV = -1; long bestScore = -1;
            foreach (var kv in g)
            {
                long score = (long)kv.Value.Count * gi[kv.Key].Count;
                if (score > bestScore) { bestScore = score; bestV = kv.Key; }
            }
            solList.Add(bestV); solSet.Add(bestV);
            RemoveVertex(g, gi, bestV);
        }
        return solList;
    }

    static bool IsDAG(Dictionary<int, HashSet<int>> g)
    {
        var indeg = new Dictionary<int, int>();
        foreach (var kv in g) indeg[kv.Key] = 0;
        foreach (var kv in g) foreach (var w in kv.Value) if (indeg.ContainsKey(w)) indeg[w]++;
        var q = new Queue<int>();
        foreach (var kv in indeg) if (kv.Value == 0) q.Enqueue(kv.Key);
        int seen = 0;
        while (q.Count > 0)
        {
            var v = q.Dequeue(); seen++;
            foreach (var w in g[v])
            {
                if (!indeg.ContainsKey(w)) continue;
                indeg[w]--;
                if (indeg[w] == 0) q.Enqueue(w);
            }
        }
        return seen == g.Count;
    }

    static List<int> FindShortCycle(Dictionary<int, HashSet<int>> g)
    {
        List<int> bestCyc = null;
        foreach (var start in g.Keys)
        {
            var parent = new Dictionary<int, int>();
            var visited = new HashSet<int>();
            var q = new Queue<int>();
            q.Enqueue(start); visited.Add(start); parent[start] = -1;
            while (q.Count > 0)
            {
                var v = q.Dequeue();
                if (bestCyc != null && parent.Count >= bestCyc.Count) break;
                foreach (var w in g[v])
                {
                    if (w == start)
                    {
                        var cyc = new List<int>();
                        int cur = v;
                        while (cur != -1) { cyc.Add(cur); cur = parent[cur]; }
                        if (bestCyc == null || cyc.Count < bestCyc.Count) bestCyc = cyc;
                        break;
                    }
                    if (!visited.Contains(w))
                    {
                        visited.Add(w); parent[w] = v; q.Enqueue(w);
                    }
                }
            }
            if (bestCyc != null && bestCyc.Count <= 3) return bestCyc;
        }
        return bestCyc;
    }

    static List<int> SolveSCC(Dictionary<int, HashSet<int>> sOut, Dictionary<int, HashSet<int>> sIn)
    {
        var ub = GreedyDFVS(Clone(sOut), Clone(sIn));
        bestSolution = new HashSet<int>(ub);
        var current = new HashSet<int>();
        Branch(Clone(sOut), Clone(sIn), current, 0);
        return new List<int>(bestSolution);
    }

    static int branchCount;
    const int BRANCH_LIMIT = 2000000;

    static void Branch(Dictionary<int, HashSet<int>> g, Dictionary<int, HashSet<int>> gi, HashSet<int> current, int depth)
    {
        branchCount++;
        if (branchCount > BRANCH_LIMIT) return;

        var localAdds = new HashSet<int>();
        ReduceLocal(g, gi, localAdds);
        foreach (var x in localAdds) current.Add(x);

        if (current.Count >= bestSolution.Count) { foreach (var x in localAdds) current.Remove(x); return; }

        if (g.Count == 0 || IsDAG(g))
        {
            if (current.Count < bestSolution.Count) bestSolution = new HashSet<int>(current);
            foreach (var x in localAdds) current.Remove(x);
            return;
        }

        var cycle = FindShortCycle(g);
        if (cycle == null)
        {
            if (current.Count < bestSolution.Count) bestSolution = new HashSet<int>(current);
            foreach (var x in localAdds) current.Remove(x);
            return;
        }

        foreach (var v in cycle)
        {
            if (current.Count + 1 >= bestSolution.Count) continue;
            if (branchCount > BRANCH_LIMIT) break;
            var gc = Clone(g);
            var gic = Clone(gi);
            RemoveVertex(gc, gic, v);
            current.Add(v);
            Branch(gc, gic, current, depth + 1);
            current.Remove(v);
        }

        foreach (var x in localAdds) current.Remove(x);
    }
}
