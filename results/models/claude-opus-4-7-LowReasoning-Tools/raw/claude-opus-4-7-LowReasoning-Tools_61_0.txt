declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Pkg { id: any; x: number; y: number; }
const depot: [number, number] = data.depot;
const truckCount: number = data.truckCount;
const packages: Pkg[] = data.packages;
const n = packages.length;

const dx = depot[0], dy = depot[1];

function dist(ax: number, ay: number, bx: number, by: number): number {
  const dxv = ax - bx, dyv = ay - by;
  return Math.sqrt(dxv * dxv + dyv * dyv);
}

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const d0: number[] = new Array(n);
for (let i = 0; i < n; i++) d0[i] = dist(packages[i].x, packages[i].y, dx, dy);

type Route = { nodes: number[]; };
const routeOf: (Route | null)[] = new Array(n);
const allRoutes: Route[] = [];
for (let i = 0; i < n; i++) {
  const r: Route = { nodes: [i] };
  allRoutes.push(r);
  routeOf[i] = r;
}

const MAX_SAVINGS_N = 3000;

if (n <= MAX_SAVINGS_N && n > 1) {
  const savings: { s: number; i: number; j: number }[] = [];
  for (let i = 0; i < n; i++) {
    const pi = packages[i];
    for (let j = i + 1; j < n; j++) {
      const pj = packages[j];
      const dij = dist(pi.x, pi.y, pj.x, pj.y);
      const s = d0[i] + d0[j] - dij;
      if (s > 0) savings.push({ s, i, j });
    }
  }
  savings.sort((a, b) => b.s - a.s);

  for (const sv of savings) {
    const ri = routeOf[sv.i];
    const rj = routeOf[sv.j];
    if (!ri || !rj || ri === rj) continue;
    const ni = ri.nodes, nj = rj.nodes;
    const iAtStart = ni[0] === sv.i;
    const iAtEnd = ni[ni.length - 1] === sv.i;
    const jAtStart = nj[0] === sv.j;
    const jAtEnd = nj[nj.length - 1] === sv.j;
    if (!(iAtStart || iAtEnd) || !(jAtStart || jAtEnd)) continue;

    let merged: number[];
    if (iAtEnd && jAtStart) merged = ni.concat(nj);
    else if (iAtEnd && jAtEnd) merged = ni.concat(nj.slice().reverse());
    else if (iAtStart && jAtEnd) merged = nj.concat(ni);
    else merged = ni.slice().reverse().concat(nj);

    const newRoute: Route = { nodes: merged };
    for (const idx of merged) routeOf[idx] = newRoute;
    ri.nodes = [];
    rj.nodes = [];
  }
} else if (n > MAX_SAVINGS_N) {
  // Angular sweep for very large n
  const order: number[] = [];
  for (let i = 0; i < n; i++) order.push(i);
  order.sort((a, b) => {
    const angA = Math.atan2(packages[a].y - dy, packages[a].x - dx);
    const angB = Math.atan2(packages[b].y - dy, packages[b].x - dx);
    return angA - angB;
  });
  const perRoute = Math.ceil(n / Math.max(1, truckCount));
  const newRoutes: Route[] = [];
  for (let i = 0; i < order.length; i += perRoute) {
    const chunk = order.slice(i, i + perRoute);
    const r: Route = { nodes: chunk };
    for (const idx of chunk) routeOf[idx] = r;
    newRoutes.push(r);
  }
  // mark old singletons dead
  for (const r of allRoutes) r.nodes = [];
  for (const r of newRoutes) allRoutes.push(r);
}

let live: number[][] = [];
const seen = new Set<Route>();
for (let i = 0; i < n; i++) {
  const r = routeOf[i];
  if (r && !seen.has(r) && r.nodes.length > 0) {
    seen.add(r);
    live.push(r.nodes.slice());
  }
}

function insertionCost(route: number[], idx: number): { cost: number; pos: number } {
  const p = packages[idx];
  let best = Infinity, bestPos = 0;
  for (let pos = 0; pos <= route.length; pos++) {
    const prev = pos === 0 ? -1 : route[pos - 1];
    const next = pos === route.length ? -1 : route[pos];
    const px = prev < 0 ? dx : packages[prev].x;
    const py = prev < 0 ? dy : packages[prev].y;
    const nx = next < 0 ? dx : packages[next].x;
    const ny = next < 0 ? dy : packages[next].y;
    const dPrev = dist(px, py, p.x, p.y);
    const dNext = dist(p.x, p.y, nx, ny);
    const dRemove = dist(px, py, nx, ny);
    const cost = dPrev + dNext - dRemove;
    if (cost < best) { best = cost; bestPos = pos; }
  }
  return { cost: best, pos: bestPos };
}

while (live.length > truckCount && truckCount > 0) {
  live.sort((a, b) => a.length - b.length);
  const small = live.shift()!;
  for (const idx of small) {
    let bestR = -1, bestCost = Infinity, bestPos = 0;
    for (let r = 0; r < live.length; r++) {
      const { cost, pos } = insertionCost(live[r], idx);
      if (cost < bestCost) { bestCost = cost; bestR = r; bestPos = pos; }
    }
    if (bestR >= 0) {
      live[bestR].splice(bestPos, 0, idx);
    } else {
      live.push([idx]);
    }
  }
}

function twoOpt(route: number[]): number[] {
  const m = route.length;
  if (m < 4) return route;
  let improved = true;
  let iters = 0;
  const maxIters = m > 200 ? 2 : 20;
  while (improved && iters < maxIters) {
    improved = false;
    iters++;
    for (let i = 0; i < m - 1; i++) {
      for (let j = i + 1; j < m; j++) {
        const a = i - 1, b = i, c = j, d = j + 1;
        const ax = a < 0 ? dx : packages[route[a]].x;
        const ay = a < 0 ? dy : packages[route[a]].y;
        const bx = packages[route[b]].x, by = packages[route[b]].y;
        const cx = packages[route[c]].x, cy = packages[route[c]].y;
        const dxe = d >= m ? dx : packages[route[d]].x;
        const dye = d >= m ? dy : packages[route[d]].y;
        const before = dist(ax, ay, bx, by) + dist(cx, cy, dxe, dye);
        const after = dist(ax, ay, cx, cy) + dist(bx, by, dxe, dye);
        if (after + 1e-12 < before) {
          let lo = i, hi = j;
          while (lo < hi) { const t = route[lo]; route[lo] = route[hi]; route[hi] = t; lo++; hi--; }
          improved = true;
        }
      }
    }
  }
  return route;
}

for (let r = 0; r < live.length; r++) {
  if (live[r].length <= 300) live[r] = twoOpt(live[r]);
}

const outRoutes: any[][] = live.map(r => r.map(idx => packages[idx].id));
process.stdout.write(JSON.stringify({ routes: outRoutes }));
