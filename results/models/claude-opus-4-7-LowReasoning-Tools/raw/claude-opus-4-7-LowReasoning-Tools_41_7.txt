; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
     %fnvoid = OpTypeFunction %void
      %float = OpTypeFloat 32
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4f 4

        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
    %ptr_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %ptr_UBO Uniform

   %ptr_in3  = OpTypePointer Input %v3f
   %ptr_in2  = OpTypePointer Input %v2f
   %ptr_out4 = OpTypePointer Output %v4f
   %ptr_u_v4 = OpTypePointer Uniform %v4f

   %worldPos = OpVariable %ptr_in3 Input
     %normal = OpVariable %ptr_in3 Input
         %uv = OpVariable %ptr_in2 Input
    %tangent = OpVariable %ptr_in3 Input
      %color = OpVariable %ptr_in3 Input
   %outColor = OpVariable %ptr_out4 Output

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
        %f_0 = OpConstant %float 0.0
        %f_1 = OpConstant %float 1.0
        %f_3 = OpConstant %float 3.0
      %f_0_2 = OpConstant %float 0.2
      %f_0_3 = OpConstant %float 0.3
      %f_0_5 = OpConstant %float 0.5
      %f_0_8 = OpConstant %float 0.8
   %baseCol  = OpConstantComposite %v3f %f_0_2 %f_0_3 %f_0_8
   %rimTint  = OpConstantComposite %v3f %f_1 %f_0_8 %f_0_5

       %main = OpFunction %void None %fnvoid
      %entry = OpLabel

      %wpVal = OpLoad %v3f %worldPos
       %nRaw = OpLoad %v3f %normal
          %N = OpExtInst %v3f %1 Normalize %nRaw

   %camPtr   = OpAccessChain %ptr_u_v4 %ubo %int_4
   %camPos4  = OpLoad %v4f %camPtr
   %camPos3  = OpVectorShuffle %v3f %camPos4 %camPos4 0 1 2
      %vDir  = OpFSub %v3f %camPos3 %wpVal
          %V = OpExtInst %v3f %1 Normalize %vDir

   %lightPtr = OpAccessChain %ptr_u_v4 %ubo %int_3
   %lightP4  = OpLoad %v4f %lightPtr
   %lightP3  = OpVectorShuffle %v3f %lightP4 %lightP4 0 1 2
      %lDir  = OpFSub %v3f %lightP3 %wpVal
          %L = OpExtInst %v3f %1 Normalize %lDir

      %NdotL = OpDot %float %N %L
    %diffuse = OpExtInst %float %1 FMax %NdotL %f_0

      %NdotV = OpDot %float %N %V
   %NdotVcl  = OpExtInst %float %1 FMax %NdotV %f_0
   %oneMinus = OpFSub %float %f_1 %NdotVcl
        %rim = OpExtInst %float %1 Pow %oneMinus %f_3

    %diffMul = OpFMul %float %diffuse %f_0_5
    %diffAdd = OpFAdd %float %f_0_3 %diffMul
    %baseMod = OpVectorTimesScalar %v3f %baseCol %diffAdd

    %rimVec  = OpVectorTimesScalar %v3f %rimTint %rim
    %sumRGB  = OpFAdd %v3f %baseMod %rimVec

         %rC = OpCompositeExtract %float %sumRGB 0
         %gC = OpCompositeExtract %float %sumRGB 1
         %bC = OpCompositeExtract %float %sumRGB 2
       %rgba = OpCompositeConstruct %v4f %rC %gC %bC %f_1
               OpStore %outColor %rgba
               OpReturn
               OpFunctionEnd
