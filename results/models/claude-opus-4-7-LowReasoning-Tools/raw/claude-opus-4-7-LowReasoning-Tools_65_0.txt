declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Task {
  id: any;
  duration: number;
  deps: any[];
  deadline: number;
  weight: number;
}

const machineCount: number = data.machineCount;
const tasks: Task[] = data.tasks;
const n = tasks.length;

const idToIdx = new Map<any, number>();
for (let i = 0; i < n; i++) idToIdx.set(tasks[i].id, i);

const children: number[][] = Array.from({ length: n }, () => []);
const indeg: number[] = new Array(n).fill(0);
for (let i = 0; i < n; i++) {
  const deps = tasks[i].deps || [];
  for (const d of deps) {
    const j = idToIdx.get(d);
    if (j !== undefined) {
      children[j].push(i);
      indeg[i]++;
    }
  }
}

const order: number[] = [];
{
  const q: number[] = [];
  const d = indeg.slice();
  let head = 0;
  for (let i = 0; i < n; i++) if (d[i] === 0) q.push(i);
  while (head < q.length) {
    const u = q[head++];
    order.push(u);
    for (const v of children[u]) {
      if (--d[v] === 0) q.push(v);
    }
  }
}

const cpl: number[] = new Array(n).fill(0);
const downW: number[] = new Array(n).fill(0);
for (let i = order.length - 1; i >= 0; i--) {
  const u = order[i];
  let best = 0;
  let wsum = 0;
  for (const v of children[u]) {
    if (cpl[v] > best) best = cpl[v];
    wsum += downW[v];
  }
  cpl[u] = tasks[u].duration + best;
  downW[u] = tasks[u].weight + wsum;
}

function priority(i: number): number {
  const t = tasks[i];
  return (t.weight + downW[i] * 0.1) * 1000 + cpl[i] * 10 - t.deadline;
}

const depFinish: number[] = new Array(n).fill(0);
const remaining = indeg.slice();
const machineFree: number[] = new Array(machineCount).fill(0);

const ready: number[] = [];
for (let i = 0; i < n; i++) if (remaining[i] === 0) ready.push(i);

const schedule: { id: any; start: number; machine: number }[] = [];
let done = 0;

while (done < n && ready.length > 0) {
  ready.sort((a, b) => priority(b) - priority(a));
  let m = 0;
  for (let k = 1; k < machineCount; k++) if (machineFree[k] < machineFree[m]) m = k;
  const idx = ready.shift()!;
  const start = Math.max(machineFree[m], depFinish[idx]);
  const fin = start + tasks[idx].duration;
  machineFree[m] = fin;
  schedule.push({ id: tasks[idx].id, start, machine: m });
  done++;
  for (const v of children[idx]) {
    if (fin > depFinish[v]) depFinish[v] = fin;
    if (--remaining[v] === 0) ready.push(v);
  }
}

process.stdout.write(JSON.stringify({ schedule }));
