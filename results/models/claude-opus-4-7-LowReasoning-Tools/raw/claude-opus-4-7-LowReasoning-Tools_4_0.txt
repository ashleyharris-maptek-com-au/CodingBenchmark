#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <array>

using namespace std;

struct Vec3 { double x, y, z; };

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double e;
    if(!(cin >> V >> e)) return 0;
    vector<Vec3> verts(V);
    double minx=1e18,miny=1e18,minz=1e18;
    double maxx=-1e18,maxy=-1e18,maxz=-1e18;
    for(int i=0;i<V;i++){
        cin >> verts[i].x >> verts[i].y >> verts[i].z;
        minx=min(minx,verts[i].x); maxx=max(maxx,verts[i].x);
        miny=min(miny,verts[i].y); maxy=max(maxy,verts[i].y);
        minz=min(minz,verts[i].z); maxz=max(maxz,verts[i].z);
    }

    // Regular tetrahedron geometry
    double R = e * sqrt(3.0/8.0);  // circumradius

    // Use cubic cells large enough that a tet of edge e fits with axis-aligned orientation.
    // The tet's bounding sphere has radius R ~ 0.612e, so a cube of side 2R ensures no
    // overlap regardless of orientation. Use slightly larger for safety.
    double cell = 2.0 * R + 1e-4;

    // Shrink bbox by R so tets stay inside the container's bounding box.
    double pad = R + 1e-6;
    double sx = minx + pad, sy = miny + pad, sz = minz + pad;
    double ex = maxx - pad, ey = maxy - pad, ez = maxz - pad;

    vector<array<double,6>> out;

    if(sx <= ex && sy <= ey && sz <= ez){
        double stride = cell;
        long long cap = 150000;

        // Estimate count and rescale stride if needed.
        auto estimate = [&](double s)->long long{
            long long a = max(1LL, (long long)floor((ex-sx)/s) + 1);
            long long b = max(1LL, (long long)floor((ey-sy)/s) + 1);
            long long c = max(1LL, (long long)floor((ez-sz)/s) + 1);
            return a*b*c;
        };

        if(estimate(stride) > cap){
            double lo = stride, hi = stride * 1000.0;
            for(int it=0; it<60; it++){
                double mid = 0.5*(lo+hi);
                if(estimate(mid) > cap) lo = mid; else hi = mid;
            }
            stride = hi;
        }

        for(double cz = sz; cz <= ez + 1e-9; cz += stride){
            for(double cy = sy; cy <= ey + 1e-9; cy += stride){
                for(double cx = sx; cx <= ex + 1e-9; cx += stride){
                    out.push_back({cx,cy,cz,0.0,0.0,0.0});
                    if((long long)out.size() >= cap) goto done;
                }
            }
        }
    }
done:

    cout << out.size() << "\n";
    cout.setf(std::ios::fixed);
    cout.precision(6);
    for(auto&t:out){
        cout << t[0] << " " << t[1] << " " << t[2] << " "
             << t[3] << " " << t[4] << " " << t[5] << "\n";
    }
    return 0;
}
