import math

_state = {
    'prev_steering': 0.0,
    'speed_est': 0.0,
    'lane_offset_est': 0.0,
    'integral_speed': 0.0,
    'prev_speed_err': 0.0,
    'prev_lane_err': 0.0,
    'last_speed_limit': 130.0,
    'last_curvature': 0.0,
    'last_obs_dist': 1e9,
    'last_obs_rel': 0.0,
    'obs_count': 0,
    'no_obs_count': 10,
    'brake_fail_count': 0,
}


def _median(vals):
    vals = sorted(vals)
    n = len(vals)
    if n == 0:
        return None
    if n % 2 == 1:
        return vals[n // 2]
    return 0.5 * (vals[n // 2 - 1] + vals[n // 2])


def _get(state, key, default=0.0):
    v = state.get(key, default)
    if v is None:
        return default
    try:
        if isinstance(v, float) and (math.isnan(v) or math.isinf(v)):
            return default
    except Exception:
        pass
    return v


def _clip(x, lo, hi):
    return lo if x < lo else (hi if x > hi else x)


def autopilot_step(state: dict, dt: float) -> dict:
    global _state
    if dt is None or dt <= 0:
        dt = _get(state, 'dt', 0.05)
    if dt <= 0:
        dt = 0.05

    # ---- SPEED FUSION ----
    wheel_speeds = []
    for w in ('fl', 'fr', 'rl', 'rr'):
        v = _get(state, f'wheel_speed_{w}_kmh', None)
        valid = _get(state, f'wheel_speed_{w}_valid', False)
        if v is not None and valid and 0 <= v < 300:
            wheel_speeds.append(v)
    speed_vals = []
    if wheel_speeds:
        speed_vals.append(_median(wheel_speeds))
    for i in (1, 2):
        if _get(state, f'gps_{i}_valid', False):
            s = _get(state, f'gps_{i}_speed_kmh', None)
            if s is not None and 0 <= s < 300:
                speed_vals.append(s)
    for i in (1, 2):
        s = _get(state, f'vehicle_speed_{i}_kmh', None)
        if s is not None and 0 <= s < 300:
            speed_vals.append(s)
    if speed_vals:
        speed_kmh = _median(speed_vals)
    else:
        speed_kmh = _state['speed_est']
    speed_kmh = 0.6 * speed_kmh + 0.4 * _state['speed_est']
    _state['speed_est'] = speed_kmh
    speed_ms = speed_kmh / 3.6

    # ---- ACCEL ----
    lon_accels = []
    for i in (1, 2, 3):
        if _get(state, f'accel_{i}_valid', False):
            lo = _get(state, f'accel_{i}_longitudinal_ms2', None)
            if lo is not None and abs(lo) < 30:
                lon_accels.append(lo)
    lon_accel = _median(lon_accels) if lon_accels else 0.0

    # ---- VISION FUSION ----
    lane_offsets = []
    curvatures = []
    speed_limits = []
    obs_dists = []
    obs_rels = []
    obs_votes = 0
    obs_total = 0
    for i in (1, 2, 3):
        lo = _get(state, f'vision_{i}_lane_offset_m', None)
        ll_v = _get(state, f'vision_{i}_left_line_valid', False)
        rl_v = _get(state, f'vision_{i}_right_line_valid', False)
        if (ll_v or rl_v) and lo is not None and abs(lo) < 5:
            lane_offsets.append(lo)
        c = _get(state, f'vision_{i}_road_curvature_1pm', None)
        if c is not None and abs(c) < 0.1:
            curvatures.append(c)
        if _get(state, f'vision_{i}_speed_limit_valid', False):
            sl = _get(state, f'vision_{i}_speed_limit_kmh', None)
            if sl is not None and 0 < sl < 250:
                speed_limits.append(sl)
        if _get(state, f'vision_{i}_obstacle_valid', False):
            obs_total += 1
            det = _get(state, f'vision_{i}_obstacle_detected', False)
            if det:
                obs_votes += 1
                od = _get(state, f'vision_{i}_obstacle_dist_m', None)
                ors = _get(state, f'vision_{i}_obstacle_rel_speed_kmh', None)
                if od is not None and 0 < od < 500:
                    obs_dists.append(od)
                if ors is not None and abs(ors) < 300:
                    obs_rels.append(ors)

    lane_offset = _median(lane_offsets) if lane_offsets else _state['lane_offset_est']
    lane_offset = 0.5 * lane_offset + 0.5 * _state['lane_offset_est']
    _state['lane_offset_est'] = lane_offset

    curvature = _median(curvatures) if curvatures else _state['last_curvature']
    _state['last_curvature'] = curvature

    if speed_limits:
        speed_limit = _median(speed_limits)
        _state['last_speed_limit'] = speed_limit
    else:
        speed_limit = _state['last_speed_limit']

    obstacle = False
    obs_dist = 1e9
    obs_rel = 0.0
    if obs_total > 0 and obs_votes * 2 >= obs_total:
        obstacle = True
        if obs_dists:
            obs_dist = _median(obs_dists)
        if obs_rels:
            obs_rel = _median(obs_rels)
        _state['obs_count'] += 1
        _state['no_obs_count'] = 0
        _state['last_obs_dist'] = obs_dist
        _state['last_obs_rel'] = obs_rel
    else:
        _state['no_obs_count'] += 1
        _state['obs_count'] = max(0, _state['obs_count'] - 1)
        if _state['obs_count'] > 0 and _state['no_obs_count'] < 3:
            obstacle = True
            obs_dist = _state['last_obs_dist']
            obs_rel = _state['last_obs_rel']

    # ---- LIDAR FRONT ----
    front_min = 1e9
    for ang in ('+005', '-005', '+015', '-015', '+030', '-030'):
        if _get(state, f'lidar_{ang}_deg_valid', False):
            r = _get(state, f'lidar_{ang}_deg_range_m', None)
            if r is not None and 0 < r < 200 and r < front_min:
                front_min = r
    if front_min < 60:
        if not obstacle or front_min < obs_dist:
            obstacle = True
            obs_dist = min(obs_dist, front_min)
            if obs_rel == 0.0:
                obs_rel = -speed_kmh

    # ---- TRACTION ----
    mu = _get(state, 'estimated_mu', 1.0)
    if mu is None or mu <= 0:
        mu = 1.0
    rain = _get(state, 'rain_sensor', 0.0) or 0.0
    low_traction = mu < 0.6 or rain > 0.5

    # ---- TARGETS ----
    target_speed = _get(state, 'target_speed_kmh', 100.0) or 100.0
    target_lane = _get(state, 'target_lane', _get(state, 'current_lane', 1))
    current_lane = _get(state, 'current_lane', target_lane)

    effective_target = min(target_speed, speed_limit)
    if mu < 0.3:
        effective_target = min(effective_target, 40.0)
    elif mu < 0.5:
        effective_target = min(effective_target, 70.0)
    elif mu < 0.7:
        effective_target = min(effective_target, 90.0)

    if abs(curvature) > 1e-5:
        radius = 1.0 / max(abs(curvature), 1e-6)
        max_lat = max(0.5, mu * 9.81 * 0.5)
        v_curve_kmh = math.sqrt(max_lat * radius) * 3.6
        effective_target = min(effective_target, v_curve_kmh)

    # ---- OBSTACLE HANDLING ----
    brake = 0.0
    throttle_cap = 1.0
    if obstacle and obs_dist < 200:
        rel_ms = obs_rel / 3.6
        closing_ms = max(0.0, -rel_ms)
        ttc = obs_dist / max(closing_ms, 0.1)
        follow_dist = max(8.0, 2.0 * speed_ms)
        decel_max = max(2.0, mu * 9.81 * 0.8)
        stop_dist = (speed_ms * speed_ms) / (2 * decel_max) + 5.0

        if obs_dist < stop_dist * 1.2 or ttc < 2.5:
            urgency = 1.0
            if ttc > 0:
                urgency = _clip(1.5 / ttc, 0.3, 1.0)
            if obs_dist < stop_dist:
                urgency = max(urgency, 0.9)
            if obs_dist < 10:
                urgency = 1.0
            brake = urgency
            throttle_cap = 0.0
            effective_target = min(effective_target, max(0.0, speed_kmh + obs_rel - 5))
        elif obs_dist < follow_dist * 2:
            target_follow = speed_kmh + obs_rel
            effective_target = min(effective_target, max(0.0, target_follow))
            if obs_dist < follow_dist:
                brake = _clip((follow_dist - obs_dist) / follow_dist * 0.5, 0.0, 0.4)
                throttle_cap = 0.0

    # ---- SPEED PID ----
    speed_err = effective_target - speed_kmh
    _state['integral_speed'] = _clip(_state['integral_speed'] + speed_err * dt, -50, 50)
    dspeed = (speed_err - _state['prev_speed_err']) / max(dt, 1e-3)
    _state['prev_speed_err'] = speed_err

    cmd = 0.05 * speed_err + 0.005 * _state['integral_speed'] + 0.005 * dspeed

    if cmd >= 0:
        throttle = _clip(cmd, 0.0, 1.0)
    else:
        throttle = 0.0
        brake = max(brake, _clip(-cmd * 0.5, 0.0, 1.0))

    throttle = min(throttle, throttle_cap)
    if low_traction:
        throttle *= 0.6
        brake = min(brake, 0.7)

    # ---- STEERING ----
    lane_change = 0.0
    if target_lane is not None and current_lane is not None and target_lane != current_lane:
        direction = 1.0 if target_lane > current_lane else -1.0
        lane_change = direction * 1.0

    lane_err = lane_offset
    dlane = (lane_err - _state['prev_lane_err']) / max(dt, 1e-3)
    _state['prev_lane_err'] = lane_err

    steer_lane = -(0.25 * lane_err + 0.15 * dlane)
    steer_ff = _clip(curvature * 30.0, -0.5, 0.5)
    steer_lc = lane_change * 0.15

    steering = steer_lane + steer_ff + steer_lc
    steering = _clip(steering, -1.0, 1.0)

    max_rate = 4.0 * dt
    if low_traction:
        max_rate *= 0.5
    delta = _clip(steering - _state['prev_steering'], -max_rate, max_rate)
    steering = _state['prev_steering'] + delta

    # Brake failure heuristic
    if brake > 0.5 and lon_accel > -0.5 and speed_kmh > 5:
        _state['brake_fail_count'] += 1
    else:
        _state['brake_fail_count'] = max(0, _state['brake_fail_count'] - 1)
    if _state['brake_fail_count'] > 20:
        throttle = 0.0

    throttle = _clip(throttle, 0.0, 1.0)
    brake = _clip(brake, 0.0, 1.0)
    steering = _clip(steering, -1.0, 1.0)
    if brake > 0.1:
        throttle = 0.0

    _state['prev_steering'] = steering

    return {'steering': steering, 'throttle': throttle, 'brake': brake}
