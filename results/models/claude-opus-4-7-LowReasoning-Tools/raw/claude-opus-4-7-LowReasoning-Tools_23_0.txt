use std::io::{self, BufRead, Write, BufWriter};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    ang_vel: f64,
    fuel: f64,
    t: f64,
}

#[derive(Clone, Debug, Default)]
struct MapInfo {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    init_fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
    grid: Vec<u8>,
}

impl MapInfo {
    fn ground_y(&self, x: i64) -> i64 {
        if x < 0 || (x as usize) >= self.width {
            return self.height as i64;
        }
        for y in 0..self.height {
            let c = self.grid[y * self.width + x as usize];
            if c == 1 || c == 2 || c == 3 {
                return y as i64;
            }
        }
        self.height as i64
    }
}

fn read_initial<R: BufRead>(r: &mut R) -> io::Result<MapInfo> {
    let mut tokens: Vec<String> = Vec::new();
    let mut map = MapInfo::default();
    let mut line = String::new();
    while tokens.len() < 9 {
        line.clear();
        let n = r.read_line(&mut line)?;
        if n == 0 { break; }
        for tok in line.split_whitespace() {
            tokens.push(tok.to_string());
            if tokens.len() == 9 { break; }
        }
    }
    if tokens.len() < 9 {
        return Err(io::Error::new(io::ErrorKind::UnexpectedEof, "bad header"));
    }
    map.width = tokens[0].parse().unwrap_or(0);
    map.height = tokens[1].parse().unwrap_or(0);
    map.gravity = tokens[2].parse().unwrap_or(0.0);
    map.max_thrust = tokens[3].parse().unwrap_or(0.0);
    map.init_fuel = tokens[4].parse().unwrap_or(0.0);
    map.start_x = tokens[5].parse().unwrap_or(0.0);
    map.start_y = tokens[6].parse().unwrap_or(0.0);
    map.target_x = tokens[7].parse().unwrap_or(0.0);
    map.target_y = tokens[8].parse().unwrap_or(0.0);

    map.grid = vec![0u8; map.width.saturating_mul(map.height)];

    let mut rows_read = 0usize;
    while rows_read < map.height {
        line.clear();
        let n = r.read_line(&mut line)?;
        if n == 0 { break; }
        let mut col = 0usize;
        for ch in line.chars() {
            if col >= map.width { break; }
            let v = match ch {
                '.' => 0u8,
                '#' => 1u8,
                'L' => 2u8,
                'T' => 3u8,
                'S' => 0u8,
                _ => continue,
            };
            map.grid[rows_read * map.width + col] = v;
            col += 1;
        }
        if col > 0 { rows_read += 1; }
    }
    Ok(map)
}

fn parse_state(line: &str) -> Option<State> {
    let mut it = line.split_whitespace();
    let x: f64 = it.next()?.parse().ok()?;
    let y: f64 = it.next()?.parse().ok()?;
    let vx: f64 = it.next()?.parse().ok()?;
    let vy: f64 = it.next()?.parse().ok()?;
    let ang: f64 = it.next()?.parse().ok()?;
    let av: f64 = it.next()?.parse().ok()?;
    let f: f64 = it.next()?.parse().ok()?;
    Some(State { x, y, vx, vy, angle: ang, ang_vel: av, fuel: f, t: 0.0 })
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo { lo } else if v > hi { hi } else { v }
}

fn control(s: &State, m: &MapInfo) -> (f64, f64) {
    let g = m.gravity.max(0.01);
    let tmax = m.max_thrust.max(0.01);

    let dx = m.target_x - s.x;

    let dist_x = dx.abs();
    let max_vx = clamp(0.6 * dist_x.sqrt() + 0.5, 0.0, 12.0);
    let desired_vx = max_vx * dx.signum();
    let vx_err = desired_vx - s.vx;

    let a_x_des = clamp(vx_err * 1.2, -0.6 * g.max(1.0), 0.6 * g.max(1.0));

    let height_above = (m.target_y - s.y).max(0.0);
    let desired_vy_mag = clamp(0.5 + 0.7 * height_above.sqrt(), 0.5, 18.0);
    let desired_vy_signed = if s.y < m.target_y { desired_vy_mag } else { -1.0 };
    let vy_err = desired_vy_signed - s.vy;
    let a_y_net_des = clamp(-vy_err * 1.5, -2.5 * g, 2.0 * g);
    let an_up = (g - a_y_net_des).max(0.0);

    let an_eff = an_up.max(0.5);
    let mut angle_des = (a_x_des / an_eff).atan();
    let max_tilt = 0.6;
    angle_des = clamp(angle_des, -max_tilt, max_tilt);

    if dx.abs() < 1.5 && height_above < 5.0 {
        angle_des = 0.0;
    }

    let ground = m.ground_y(s.x as i64) as f64;
    let dist_to_ground = (ground - s.y).max(0.0);
    let _ = dist_to_ground;

    let cos_a = s.angle.cos().abs().max(0.2);
    let thrust_acc = an_up / cos_a;
    let mut thrust_frac = clamp(thrust_acc / tmax, 0.0, 1.0);

    if s.vy < 0.0 && height_above > 10.0 {
        thrust_frac *= 0.9;
    }
    if s.fuel <= 0.05 {
        thrust_frac = 0.0;
    }

    let ang_err = angle_des - s.angle;
    let turn = clamp(ang_err * 4.0 - s.ang_vel * 1.5, -1.0, 1.0);

    (thrust_frac, turn)
}

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let stdout = io::stdout();
    let writer = Arc::new(Mutex::new(BufWriter::new(stdout.lock())));

    let map = match read_initial(&mut reader) {
        Ok(v) => v,
        Err(_) => {
            let mut w = writer.lock().unwrap();
            for _ in 0..50 {
                let _ = writeln!(w, "0.0 0.0");
            }
            let _ = w.flush();
            return;
        }
    };
    drop(reader);
    let map = Arc::new(map);

    let state = Arc::new(Mutex::new(State {
        x: map.start_x,
        y: map.start_y,
        vx: 0.0, vy: 0.0, angle: 0.0, ang_vel: 0.0,
        fuel: map.init_fuel, t: 0.0,
    }));

    {
        let state_cl = Arc::clone(&state);
        thread::spawn(move || {
            let stdin = io::stdin();
            let mut r = stdin.lock();
            let mut buf = String::new();
            loop {
                buf.clear();
                match r.read_line(&mut buf) {
                    Ok(0) => break,
                    Ok(_) => {
                        if let Some(s) = parse_state(&buf) {
                            let mut st = state_cl.lock().unwrap();
                            let prev_t = st.t;
                            *st = s;
                            st.t = prev_t + 0.1;
                        }
                    }
                    Err(_) => break,
                }
            }
        });
    }

    let period = Duration::from_millis(20);
    let start = Instant::now();
    let mut next = start + period;
    loop {
        let s_now = { *state.lock().unwrap() };
        let (thrust, turn) = control(&s_now, &map);
        {
            let mut w = writer.lock().unwrap();
            let _ = writeln!(w, "{:.3} {:.3}", thrust, turn);
            let _ = w.flush();
        }
        let now = Instant::now();
        if next > now {
            thread::sleep(next - now);
        }
        next += period;
    }
}
