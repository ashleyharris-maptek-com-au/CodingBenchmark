#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <vector>
#include <algorithm>
#include <random>
#include <chrono>
#include <cstdint>

using namespace std;
using Clock = chrono::steady_clock;

static int N, M, K;
static vector<vector<int>> adj;
static Clock::time_point t_start;
static double time_limit_sec = 270.0;

static inline double elapsed() {
    return chrono::duration<double>(Clock::now() - t_start).count();
}

static bool greedy_order(const vector<int>& order, vector<int>& out) {
    out.assign(N, -1);
    vector<int> used(K + 1, -1);
    int marker = 0;
    bool full = true;
    for (int v : order) {
        marker++;
        for (int u : adj[v]) {
            if (out[u] >= 0) used[out[u]] = marker;
        }
        int c = -1;
        for (int i = 0; i < K; i++) {
            if (used[i] != marker) { c = i; break; }
        }
        if (c < 0) { full = false; out[v] = 0; }
        else out[v] = c;
    }
    return full;
}

static bool dsatur_color(vector<int>& out) {
    out.assign(N, -1);
    vector<int> sat(N, 0);
    vector<int> deg(N);
    for (int i = 0; i < N; i++) deg[i] = (int)adj[i].size();
    vector<vector<uint8_t>> nbcol(N);
    for (int i = 0; i < N; i++) nbcol[i].assign(K, 0);

    int colored = 0;
    while (colored < N) {
        int best = -1;
        int bestSat = -1, bestDeg = -1;
        for (int v = 0; v < N; v++) {
            if (out[v] >= 0) continue;
            if (sat[v] > bestSat || (sat[v] == bestSat && deg[v] > bestDeg)) {
                bestSat = sat[v]; bestDeg = deg[v]; best = v;
            }
        }
        if (best < 0) break;
        int c = -1;
        for (int i = 0; i < K; i++) if (!nbcol[best][i]) { c = i; break; }
        if (c < 0) return false;
        out[best] = c;
        colored++;
        for (int u : adj[best]) {
            if (out[u] < 0 && !nbcol[u][c]) {
                nbcol[u][c] = 1;
                sat[u]++;
            }
        }
    }
    return true;
}

static long long count_conflicts(const vector<int>& col) {
    long long c = 0;
    for (int u = 0; u < N; u++) {
        for (int v : adj[u]) if (v > u && col[u] == col[v]) c++;
    }
    return c;
}

static bool repair_min_conflicts(vector<int>& col, double budget_sec) {
    mt19937 rng(12345);
    uniform_int_distribution<int> dc(0, K - 1);
    for (int i = 0; i < N; i++) if (col[i] < 0) col[i] = dc(rng);

    vector<int> vcon(N, 0);
    vector<int> conflictVerts;
    for (int v = 0; v < N; v++) {
        int c = 0;
        for (int u : adj[v]) if (col[u] == col[v]) c++;
        vcon[v] = c;
        if (c > 0) conflictVerts.push_back(v);
    }

    double t0 = elapsed();
    long long iter = 0;
    vector<int> cnt(K, 0);
    while (!conflictVerts.empty()) {
        if ((iter & 1023) == 0 && elapsed() - t0 > budget_sec) break;
        iter++;
        int idx = (int)(rng() % conflictVerts.size());
        int v = conflictVerts[idx];
        if (vcon[v] == 0) {
            conflictVerts[idx] = conflictVerts.back();
            conflictVerts.pop_back();
            continue;
        }
        fill(cnt.begin(), cnt.end(), 0);
        for (int u : adj[v]) cnt[col[u]]++;
        int oldc = col[v];
        int bestc = oldc;
        int bestv = cnt[oldc];
        int offs = (int)(rng() % K);
        for (int i = 0; i < K; i++) {
            int c = (i + offs) % K;
            if (cnt[c] < bestv) { bestv = cnt[c]; bestc = c; }
        }
        if (bestc == oldc) {
            if ((rng() & 15) == 0) {
                bestc = (oldc + 1 + (int)(rng() % (K - 1))) % K;
                bestv = cnt[bestc];
            } else {
                continue;
            }
        }
        col[v] = bestc;
        vcon[v] = bestv;
        for (int u : adj[v]) {
            int cu = col[u];
            int delta = 0;
            if (cu == oldc) delta -= 1;
            if (cu == bestc) delta += 1;
            if (delta != 0) {
                int before = vcon[u];
                vcon[u] = before + delta;
                if (before == 0 && vcon[u] > 0) {
                    conflictVerts.push_back(u);
                }
            }
        }
        if (bestv == 0) {
            // remove v from list
            // find current index of v (could have moved); we know idx still refers to v unless push_back grew
            // safer: skip removal; cleanup later
        }
        if ((iter & 0xFFFF) == 0) {
            int w = 0;
            for (int i = 0; i < (int)conflictVerts.size(); i++) {
                int x = conflictVerts[i];
                if (vcon[x] > 0) conflictVerts[w++] = x;
            }
            conflictVerts.resize(w);
        }
    }
    int w = 0;
    for (int i = 0; i < (int)conflictVerts.size(); i++) {
        int x = conflictVerts[i];
        if (vcon[x] > 0) conflictVerts[w++] = x;
    }
    conflictVerts.resize(w);
    return conflictVerts.empty();
}

int main() {
    t_start = Clock::now();
    if (scanf("%d %d %d", &N, &M, &K) != 3) return 1;
    adj.assign(N, {});
    vector<int> deg(N, 0);
    vector<pair<int,int>> edges;
    edges.reserve(M);
    for (int i = 0; i < M; i++) {
        int u, v;
        if (scanf("%d %d", &u, &v) != 2) return 1;
        if (u == v) continue;
        edges.emplace_back(u, v);
        deg[u]++; deg[v]++;
    }
    for (int i = 0; i < N; i++) adj[i].reserve(deg[i]);
    for (auto& e : edges) {
        adj[e.first].push_back(e.second);
        adj[e.second].push_back(e.first);
    }
    edges.clear(); edges.shrink_to_fit();

    vector<int> best;
    bool ok = false;

    bool small = (N <= 2000);
    if (small) {
        vector<int> c1;
        if (dsatur_color(c1) && count_conflicts(c1) == 0) {
            best = c1; ok = true;
        } else if (!c1.empty()) {
            best = c1;
        }
    }

    if (!ok) {
        vector<int> order(N);
        for (int i = 0; i < N; i++) order[i] = i;
        sort(order.begin(), order.end(), [&](int a, int b){
            return adj[a].size() > adj[b].size();
        });
        vector<int> c2;
        bool full = greedy_order(order, c2);
        if (full && count_conflicts(c2) == 0) {
            best = c2; ok = true;
        } else {
            best = c2;
        }
    }

    if (!ok) {
        double remaining = time_limit_sec - elapsed();
        double budget = max(5.0, remaining - 15.0);
        if (repair_min_conflicts(best, budget)) ok = true;
    }

    if (!ok) {
        mt19937 rng(7);
        int attempts = 3;
        for (int attempt = 0; attempt < attempts && elapsed() < time_limit_sec - 5; attempt++) {
            vector<int> order(N);
            for (int i = 0; i < N; i++) order[i] = i;
            shuffle(order.begin(), order.end(), rng);
            vector<int> c;
            greedy_order(order, c);
            for (int i = 0; i < N; i++) if (c[i] < 0) c[i] = (int)(rng() % K);
            double rem = time_limit_sec - elapsed() - 3.0;
            if (rem < 2) break;
            if (repair_min_conflicts(c, rem / (attempts - attempt))) {
                best = c; ok = true; break;
            }
            if (count_conflicts(c) < count_conflicts(best)) best = c;
        }
    }

    for (int i = 0; i < N; i++) {
        if (best[i] < 0 || best[i] >= K) best[i] = 0;
    }

    static char buf[1 << 20];
    setvbuf(stdout, buf, _IOFBF, sizeof(buf));
    for (int i = 0; i < N; i++) {
        if (i) putchar(' ');
        printf("%d", best[i]);
    }
    putchar('\n');
    return 0;
}
