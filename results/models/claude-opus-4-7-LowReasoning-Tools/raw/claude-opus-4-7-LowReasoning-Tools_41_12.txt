; SPIR-V
; Version: 1.0
               OpCapability Shader
       %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
     %fnType = OpTypeFunction %void
      %float = OpTypeFloat 32
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
        %m4f = OpTypeMatrix %v4f 4

        %UBO = OpTypeStruct %m4f %m4f %m4f %v4f %v4f %v4f
    %_pt_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_pt_UBO Uniform
    %_pt_v4u = OpTypePointer Uniform %v4f

   %_pti_v3f = OpTypePointer Input %v3f
   %_pti_v2f = OpTypePointer Input %v2f
   %_pto_v4f = OpTypePointer Output %v4f

   %worldPos = OpVariable %_pti_v3f Input
     %normal = OpVariable %_pti_v3f Input
         %uv = OpVariable %_pti_v2f Input
    %tangent = OpVariable %_pti_v3f Input
      %color = OpVariable %_pti_v3f Input
   %outColor = OpVariable %_pto_v4f Output

        %int = OpTypeInt 32 1
         %c3 = OpConstant %int 3

         %f0 = OpConstant %float 0.0
         %f1 = OpConstant %float 1.0
        %fh  = OpConstant %float 0.5
       %f03  = OpConstant %float 0.3
       %skyR = OpConstant %float 0.4
       %skyG = OpConstant %float 0.6
       %skyB = OpConstant %float 1.0
       %grR  = OpConstant %float 0.3
       %grG  = OpConstant %float 0.15
       %grB  = OpConstant %float 0.05
       %skyC = OpConstantComposite %v3f %skyR %skyG %skyB
        %grC = OpConstantComposite %v3f %grR %grG %grB
       %oneV = OpConstantComposite %v3f %f1 %f1 %f1

       %main = OpFunction %void None %fnType
        %lbl = OpLabel
         %N0 = OpLoad %v3f %normal
          %N = OpExtInst %v3f %glsl Normalize %N0
         %Ny = OpCompositeExtract %float %N 1
        %Ny2 = OpFMul %float %Ny %fh
          %t = OpFAdd %float %Ny2 %fh
         %tv = OpCompositeConstruct %v3f %t %t %t
        %omt = OpFSub %v3f %oneV %tv
       %skyM = OpFMul %v3f %skyC %tv
        %grM = OpFMul %v3f %grC %omt
        %amb = OpFAdd %v3f %skyM %grM

        %pLp = OpAccessChain %_pt_v4u %ubo %c3
        %lp4 = OpLoad %v4f %pLp
        %lp3 = OpVectorShuffle %v3f %lp4 %lp4 0 1 2
         %wp = OpLoad %v3f %worldPos
        %ld0 = OpFSub %v3f %lp3 %wp
         %ld = OpExtInst %v3f %glsl Normalize %ld0
       %dRaw = OpDot %float %N %ld
        %dCl = OpExtInst %float %glsl FMax %dRaw %f0
         %dT = OpFMul %float %dCl %f03
         %dV = OpCompositeConstruct %v3f %dT %dT %dT
        %rgb = OpFAdd %v3f %amb %dV
          %r = OpCompositeExtract %float %rgb 0
          %g = OpCompositeExtract %float %rgb 1
          %b = OpCompositeExtract %float %rgb 2
        %out = OpCompositeConstruct %v4f %r %g %b %f1
               OpStore %outColor %out
               OpReturn
               OpFunctionEnd
