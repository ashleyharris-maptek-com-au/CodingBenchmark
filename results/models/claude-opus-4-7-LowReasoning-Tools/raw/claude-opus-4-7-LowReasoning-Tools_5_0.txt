use std::io::{self, Read, Write, BufWriter};
use std::time::Instant;

struct Solver {
    w: usize,
    h: usize,
    n: usize,
    blocked: Vec<bool>,
    target: usize,
    require_cycle: bool,
    visited: Vec<bool>,
    path: Vec<usize>,
    deadline: Instant,
    time_limit_secs: u64,
    found: bool,
}

impl Solver {
    fn xy(&self, i: usize) -> (usize, usize) { (i % self.w, i / self.w) }

    fn neighbors(&self, i: usize) -> [Option<usize>; 4] {
        let (x, y) = self.xy(i);
        let mut r = [None; 4];
        if x > 0 { r[0] = Some(i - 1); }
        if x + 1 < self.w { r[1] = Some(i + 1); }
        if y > 0 { r[2] = Some(i - self.w); }
        if y + 1 < self.h { r[3] = Some(i + self.w); }
        r
    }

    fn parity_ok(&self) -> bool {
        let mut c0 = 0i64;
        let mut c1 = 0i64;
        for i in 0..self.n {
            if !self.blocked[i] {
                let (x, y) = self.xy(i);
                if (x + y) & 1 == 0 { c0 += 1; } else { c1 += 1; }
            }
        }
        // start (0,0) is color 0
        if self.require_cycle {
            if c0 != c1 { return false; }
        } else {
            // path starts at color 0
            // diff must be 0 (ends opposite) or 1 (ends same)
            let diff = c0 - c1;
            if diff != 0 && diff != 1 { return false; }
        }
        true
    }

    fn connected_check(&self, head: usize) -> bool {
        let mut stack = vec![head];
        let mut seen = vec![false; self.n];
        seen[head] = true;
        let mut count = 0usize;
        while let Some(u) = stack.pop() {
            for nb in self.neighbors(u).iter().flatten() {
                let v = *nb;
                if self.blocked[v] || seen[v] || self.visited[v] { continue; }
                seen[v] = true;
                count += 1;
                stack.push(v);
            }
        }
        count == self.target - self.path.len()
    }

    fn dfs(&mut self, current: usize) -> bool {
        if self.found { return true; }
        if self.path.len() & 63 == 0 {
            if self.deadline.elapsed().as_secs() >= self.time_limit_secs {
                return false;
            }
        }

        if self.path.len() == self.target {
            if self.require_cycle {
                for nb in self.neighbors(current).iter().flatten() {
                    if *nb == 0 { self.found = true; return true; }
                }
                return false;
            } else {
                self.found = true;
                return true;
            }
        }

        // Periodic connectivity check (expensive, do every few steps)
        if self.target >= 30 && (self.path.len() & 7) == 0 {
            if !self.connected_check(current) { return false; }
        }

        // Dead-end pruning: any unvisited cell (not head) with 0 available neighbors => fail
        // any with 1 neighbor must be endpoint
        let mut forced: Option<usize> = None;
        let mut dead_ones = 0usize;
        for i in 0..self.n {
            if self.blocked[i] || self.visited[i] { continue; }
            let mut d = 0;
            let mut adj_head = false;
            for nb in self.neighbors(i).iter().flatten() {
                let v = *nb;
                if self.blocked[v] { continue; }
                if !self.visited[v] { d += 1; }
                if v == current { adj_head = true; }
            }
            if d == 0 {
                // isolated unvisited; only OK if it's adjacent to head AND will be next
                if !adj_head { return false; }
                if forced.is_some() && forced != Some(i) { return false; }
                forced = Some(i);
            } else if d == 1 {
                dead_ones += 1;
                // it must be the endpoint (or adjacent-to-head if not endpoint scenario)
                if !self.require_cycle {
                    // allow one as endpoint
                    if dead_ones > 1 {
                        // multiple dead-ends — only OK if remaining cells small; fail
                        if dead_ones > 2 { return false; }
                    }
                } else {
                    // cycle: all degrees must be >=2 ultimately; one dead-end of degree 1 in remaining is suspicious unless adj to head
                    if !adj_head { return false; }
                }
            }
        }

        let mut cands: Vec<(usize, usize)> = Vec::with_capacity(4);
        if let Some(f) = forced {
            // must go to f if adjacent
            let mut found_adj = false;
            for nb in self.neighbors(current).iter().flatten() {
                if *nb == f { found_adj = true; break; }
            }
            if !found_adj { return false; }
            cands.push((0, f));
        } else {
            for nb in self.neighbors(current).iter().flatten() {
                let v = *nb;
                if self.blocked[v] || self.visited[v] { continue; }
                let mut d = 0;
                for nn in self.neighbors(v).iter().flatten() {
                    let w = *nn;
                    if !self.blocked[w] && !self.visited[w] && w != current { d += 1; }
                }
                cands.push((d, v));
            }
            cands.sort_by_key(|x| x.0);
        }

        for (_, v) in cands {
            self.visited[v] = true;
            self.path.push(v);
            if self.dfs(v) { return true; }
            self.path.pop();
            self.visited[v] = false;
            if self.found { return true; }
        }
        false
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut iter = input.split_ascii_whitespace();
    let w: usize = iter.next().unwrap().parse().unwrap();
    let h: usize = iter.next().unwrap().parse().unwrap();
    let obs: usize = iter.next().unwrap().parse().unwrap();
    let rc: usize = iter.next().unwrap().parse().unwrap();
    let require_cycle = rc != 0;

    let n = w * h;
    let mut blocked = vec![false; n];
    for _ in 0..obs {
        let x: usize = iter.next().unwrap().parse().unwrap();
        let y: usize = iter.next().unwrap().parse().unwrap();
        if x < w && y < h {
            blocked[y * w + x] = true;
        }
    }

    let target = blocked.iter().filter(|b| !**b).count();

    if blocked[0] || target == 0 {
        writeln!(out, "0").unwrap();
        return;
    }
    if target == 1 {
        if require_cycle {
            writeln!(out, "0").unwrap();
        } else {
            writeln!(out, "1").unwrap();
            writeln!(out, "0 0").unwrap();
        }
        return;
    }

    let mut solver = Solver {
        w, h, n,
        blocked,
        target,
        require_cycle,
        visited: vec![false; n],
        path: Vec::with_capacity(target),
        deadline: Instant::now(),
        time_limit_secs: 280,
        found: false,
    };

    if !solver.parity_ok() {
        writeln!(out, "0").unwrap();
        return;
    }

    solver.visited[0] = true;
    solver.path.push(0);

    let ok = solver.dfs(0);

    if ok && solver.path.len() == solver.target {
        writeln!(out, "{}", solver.path.len()).unwrap();
        for &i in &solver.path {
            let (x, y) = solver.xy(i);
            writeln!(out, "{} {}", x, y).unwrap();
        }
    } else {
        writeln!(out, "0").unwrap();
    }
}
