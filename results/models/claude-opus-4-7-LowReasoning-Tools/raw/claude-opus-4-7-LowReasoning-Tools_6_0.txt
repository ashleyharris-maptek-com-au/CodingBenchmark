using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.IO;
using System.Linq;

public class OrbitalTSP
{
    const double MU = 398600.4418;

    struct State
    {
        public double X, Y, Z, VX, VY, VZ;
        public State(double x, double y, double z, double vx, double vy, double vz)
        { X = x; Y = y; Z = z; VX = vx; VY = vy; VZ = vz; }
    }

    static State[] stations;
    static State start;
    static int N;

    static double EstimateCost(State a, State b)
    {
        double dx = a.X - b.X, dy = a.Y - b.Y, dz = a.Z - b.Z;
        double dist = Math.Sqrt(dx * dx + dy * dy + dz * dz);
        double dvx = a.VX - b.VX, dvy = a.VY - b.VY, dvz = a.VZ - b.VZ;
        double dv = Math.Sqrt(dvx * dvx + dvy * dvy + dvz * dvz);
        return dist * 0.001 + dv;
    }

    public static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        var lines = new List<string>();
        string line;
        while ((line = Console.In.ReadLine()) != null) lines.Add(line);

        var first = lines[0].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        N = int.Parse(first[0], ci);

        var startNums = new List<double>();
        foreach (var t in lines[1].Split(new[] { ' ', '\t', ':', ',' }, StringSplitOptions.RemoveEmptyEntries))
            if (double.TryParse(t, NumberStyles.Float, ci, out double v)) startNums.Add(v);
        start = new State(startNums[0], startNums[1], startNums[2], startNums[3], startNums[4], startNums[5]);

        stations = new State[N];
        for (int i = 0; i < N; i++)
        {
            var nums = new List<double>();
            foreach (var t in lines[2 + i].Split(new[] { ' ', '\t', ':', ',' }, StringSplitOptions.RemoveEmptyEntries))
                if (double.TryParse(t, NumberStyles.Float, ci, out double v)) nums.Add(v);
            stations[i] = new State(nums[0], nums[1], nums[2], nums[3], nums[4], nums[5]);
        }

        var order = NearestNeighbor();

        int twoOptIters = N <= 20 ? 2000 : (N <= 50 ? 500 : 100);
        order = TwoOpt(order, twoOptIters);

        Console.WriteLine(string.Join(" ", order));
    }

    static int[] NearestNeighbor()
    {
        var visited = new bool[N];
        var order = new int[N];
        State current = start;
        for (int step = 0; step < N; step++)
        {
            int best = -1;
            double bestCost = double.MaxValue;
            for (int i = 0; i < N; i++)
            {
                if (visited[i]) continue;
                double c = EstimateCost(current, stations[i]);
                if (c < bestCost) { bestCost = c; best = i; }
            }
            order[step] = best;
            visited[best] = true;
            current = stations[best];
        }
        return order;
    }

    static double TotalCost(int[] order)
    {
        double total = 0;
        State current = start;
        for (int i = 0; i < order.Length; i++)
        {
            total += EstimateCost(current, stations[order[i]]);
            current = stations[order[i]];
        }
        return total;
    }

    static int[] TwoOpt(int[] order, int maxIters)
    {
        var sw = Stopwatch.StartNew();
        long limitMs = 60_000;
        bool improved = true;
        int iters = 0;
        double bestCost = TotalCost(order);
        while (improved && iters < maxIters && sw.ElapsedMilliseconds < limitMs)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    if (sw.ElapsedMilliseconds > limitMs) break;
                    Array.Reverse(order, i, j - i + 1);
                    double c = TotalCost(order);
                    if (c < bestCost - 1e-9)
                    {
                        bestCost = c;
                        improved = true;
                    }
                    else
                    {
                        Array.Reverse(order, i, j - i + 1);
                    }
                }
                if (sw.ElapsedMilliseconds > limitMs) break;
            }
            iters++;
        }
        return order;
    }
}
