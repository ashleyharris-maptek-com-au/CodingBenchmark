using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Numerics;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

class MaxClique
{
    static int N;
    static int W;
    static volatile int bestSize;
    static int[] bestClique = new int[0];
    static readonly object bestLock = new object();
    static Stopwatch sw;
    static long timeLimitMs = 4 * 60 * 1000 + 30 * 1000;
    static volatile bool timeUp = false;

    static void Main()
    {
        sw = Stopwatch.StartNew();
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        N = int.Parse(tokens[idx++]);
        int M = int.Parse(tokens[idx++]);

        if (N == 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var adjList = new List<int>[N];
        for (int i = 0; i < N; i++) adjList[i] = new List<int>();

        for (int i = 0; i < M; i++)
        {
            int u = int.Parse(tokens[idx++]);
            int v = int.Parse(tokens[idx++]);
            if (u == v) continue;
            adjList[u].Add(v);
            adjList[v].Add(u);
        }

        var degree = new int[N];
        for (int i = 0; i < N; i++)
        {
            adjList[i].Sort();
            int w = 0; int prev = -1;
            for (int j = 0; j < adjList[i].Count; j++)
            {
                int x = adjList[i][j];
                if (x != prev && x != i) { adjList[i][w++] = x; prev = x; }
            }
            adjList[i].RemoveRange(w, adjList[i].Count - w);
            degree[i] = adjList[i].Count;
        }

        bestSize = 1;
        bestClique = new int[] { 0 };
        GreedyHeuristic(adjList, degree);

        var ct = new CancellationTokenSource();
        Task.Run(() =>
        {
            while (!ct.IsCancellationRequested)
            {
                if (sw.ElapsedMilliseconds > timeLimitMs) { timeUp = true; return; }
                Thread.Sleep(100);
            }
        });

        int[] coreMap; ulong[][] coreAdj;
        BuildCore(adjList, degree, out coreMap, out coreAdj);

        if (coreMap.Length > 0 && coreAdj != null)
        {
            RunBranchAndBound(coreMap, coreAdj);
        }

        ct.Cancel();

        var sb = new StringBuilder();
        sb.Append(bestSize);
        sb.Append('\n');
        for (int i = 0; i < bestClique.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(bestClique[i]);
        }
        sb.Append('\n');
        Console.Write(sb.ToString());
    }

    static void GreedyHeuristic(List<int>[] adjList, int[] degree)
    {
        int[] order = Enumerable.Range(0, N).OrderByDescending(v => degree[v]).ToArray();
        int seeds = Math.Min(N, 128);
        object localLock = new object();
        Parallel.For(0, seeds, si =>
        {
            if (timeUp) return;
            int seed = order[si];
            var clique = new List<int> { seed };
            var cand = new HashSet<int>(adjList[seed]);
            while (cand.Count > 0)
            {
                int bestV = -1, bestD = -1;
                foreach (var v in cand)
                {
                    int d = 0;
                    foreach (var u in adjList[v]) if (cand.Contains(u)) d++;
                    if (d > bestD) { bestD = d; bestV = v; }
                }
                if (bestV < 0) break;
                clique.Add(bestV);
                cand.IntersectWith(adjList[bestV]);
            }
            lock (localLock)
            {
                if (clique.Count > bestSize)
                {
                    bestSize = clique.Count;
                    bestClique = clique.ToArray();
                }
            }
        });
    }

    static void BuildCore(List<int>[] adjList, int[] degree, out int[] coreMap, out ulong[][] coreAdj)
    {
        int minDeg = bestSize;
        bool[] alive = new bool[N];
        int[] curDeg = new int[N];
        for (int i = 0; i < N; i++) { alive[i] = true; curDeg[i] = degree[i]; }

        var queue = new Queue<int>();
        for (int i = 0; i < N; i++) if (curDeg[i] < minDeg) queue.Enqueue(i);

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (!alive[v]) continue;
            alive[v] = false;
            foreach (var u in adjList[v])
            {
                if (alive[u])
                {
                    curDeg[u]--;
                    if (curDeg[u] < minDeg) queue.Enqueue(u);
                }
            }
        }

        var keep = new List<int>();
        for (int i = 0; i < N; i++) if (alive[i]) keep.Add(i);
        int nc = keep.Count;
        coreMap = keep.ToArray();

        if (nc == 0) { coreAdj = new ulong[0][]; return; }

        long bytes = (long)nc * nc / 8;
        if (bytes > 2L * 1024 * 1024 * 1024)
        {
            coreMap = new int[0];
            coreAdj = null;
            return;
        }

        int[] oldToNew = new int[N];
        for (int i = 0; i < N; i++) oldToNew[i] = -1;
        for (int i = 0; i < nc; i++) oldToNew[coreMap[i]] = i;

        W = (nc + 63) / 64;
        coreAdj = new ulong[nc][];
        for (int i = 0; i < nc; i++)
        {
            coreAdj[i] = new ulong[W];
            int old = coreMap[i];
            foreach (var u in adjList[old])
            {
                int nu = oldToNew[u];
                if (nu >= 0) coreAdj[i][nu >> 6] |= 1UL << (nu & 63);
            }
        }
    }

    static ulong[][] gAdj;
    static int[] gMap;

    static void RunBranchAndBound(int[] coreMap, ulong[][] coreAdj)
    {
        gAdj = coreAdj; gMap = coreMap;
        int nc = coreMap.Length;
        int[] deg = new int[nc];
        for (int i = 0; i < nc; i++)
        {
            int d = 0;
            for (int w = 0; w < W; w++) d += BitOperations.PopCount(coreAdj[i][w]);
            deg[i] = d;
        }
        int[] order = Enumerable.Range(0, nc).OrderByDescending(i => deg[i]).ToArray();
        int[] rank = new int[nc];
        for (int i = 0; i < order.Length; i++) rank[order[i]] = i;

        Parallel.For(0, order.Length, new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount },
            (oi, state) =>
        {
            if (timeUp) { state.Stop(); return; }
            int v = order[oi];
            ulong[] cand = new ulong[W];
            for (int u = 0; u < nc; u++)
            {
                if (rank[u] > oi && (coreAdj[v][u >> 6] & (1UL << (u & 63))) != 0)
                    cand[u >> 6] |= 1UL << (u & 63);
            }
            var clique = new List<int> { v };
            Expand(cand, clique);
        });
    }

    static void Expand(ulong[] P, List<int> R)
    {
        if (timeUp) return;
        int pc = 0;
        for (int w = 0; w < W; w++) pc += BitOperations.PopCount(P[w]);
        if (pc == 0) { TryUpdate(R); return; }
        if (R.Count + pc <= bestSize) return;

        var verts = new List<int>(pc);
        for (int w = 0; w < W; w++)
        {
            ulong b = P[w];
            while (b != 0)
            {
                int bit = BitOperations.TrailingZeroCount(b);
                verts.Add((w << 6) | bit);
                b &= b - 1;
            }
        }

        var colorClasses = new List<List<int>>();
        foreach (var v in verts)
        {
            int chosen = -1;
            for (int c = 0; c < colorClasses.Count; c++)
            {
                bool ok = true;
                foreach (var u in colorClasses[c])
                {
                    if ((gAdj[v][u >> 6] & (1UL << (u & 63))) != 0) { ok = false; break; }
                }
                if (ok) { chosen = c; break; }
            }
            if (chosen < 0) { chosen = colorClasses.Count; colorClasses.Add(new List<int>()); }
            colorClasses[chosen].Add(v);
        }

        var ordered = new List<(int v, int c)>(verts.Count);
        for (int c = 0; c < colorClasses.Count; c++)
            foreach (var v in colorClasses[c]) ordered.Add((v, c + 1));

        for (int i = ordered.Count - 1; i >= 0; i--)
        {
            if (timeUp) return;
            int v = ordered[i].v;
            int cnum = ordered[i].c;
            if (R.Count + cnum <= bestSize) return;

            ulong[] newP = new ulong[W];
            ulong[] mask = new ulong[W];
            for (int j = 0; j <= i; j++)
            {
                int u = ordered[j].v;
                mask[u >> 6] |= 1UL << (u & 63);
            }
            for (int w = 0; w < W; w++) newP[w] = mask[w] & gAdj[v][w];

            R.Add(v);
            Expand(newP, R);
            R.RemoveAt(R.Count - 1);
        }
    }

    static void TryUpdate(List<int> R)
    {
        if (R.Count > bestSize)
        {
            lock (bestLock)
            {
                if (R.Count > bestSize)
                {
                    bestSize = R.Count;
                    var arr = new int[R.Count];
                    for (int i = 0; i < R.Count; i++) arr[i] = gMap[R[i]];
                    bestClique = arr;
                }
            }
        }
    }
}
