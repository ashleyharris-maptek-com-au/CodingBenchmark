using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

class Body
{
    public string Name;
    public double Mass;
    public double[] R = new double[3];
    public double[] V = new double[3];
}

class Program
{
    const double G = 6.67430e-11;
    const double DAY = 86400.0;

    static void Main()
    {
        var inp = Console.In.ReadToEnd();
        var tokens = inp.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;
        int numBodies = int.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double warningDays = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double dvBudget = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double t0Unix = double.Parse(tokens[p++], CultureInfo.InvariantCulture);

        var bodies = new List<Body>();
        int idxAsteroid = -1, idxEarth = -1, idxSun = -1;
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = tokens[p++];
            b.Mass = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
            for (int k = 0; k < 3; k++) b.R[k] = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
            for (int k = 0; k < 3; k++) b.V[k] = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
            bodies.Add(b);
            if (b.Name == "Asteroid") idxAsteroid = i;
            if (b.Name == "Earth") idxEarth = i;
            if (b.Name == "Sun") idxSun = i;
        }

        double mSun = idxSun >= 0 ? bodies[idxSun].Mass : 1.98892e30;
        double muSun = G * mSun;

        double[] sunR = idxSun >= 0 ? bodies[idxSun].R : new double[3];
        double[] sunV = idxSun >= 0 ? bodies[idxSun].V : new double[3];

        double[] rE0 = Sub(bodies[idxEarth].R, sunR);
        double[] vE0 = Sub(bodies[idxEarth].V, sunV);
        double[] rA0 = Sub(bodies[idxAsteroid].R, sunR);
        double[] vA0 = Sub(bodies[idxAsteroid].V, sunV);

        // Choose launch at t=0, impact at midpoint of warning window
        double tLaunch = 0.0;
        double tImpact = Math.Max(30.0, warningDays * 0.5);
        if (tImpact > warningDays) tImpact = warningDays;
        double tof = (tImpact - tLaunch) * DAY;

        var (rE_L, vE_L) = Kepler(rE0, vE0, tLaunch * DAY, muSun);
        var (rA_I, vA_I) = Kepler(rA0, vA0, tImpact * DAY, muSun);

        double[] vSC = new double[3];
        for (int k = 0; k < 3; k++) vSC[k] = (rA_I[k] - rE_L[k]) / tof;

        double[] dvLaunch = Sub(vSC, vE_L);
        double dvLaunchMag = Norm(dvLaunch);

        double launchCap = dvBudget * 0.9;
        if (dvLaunchMag > launchCap)
        {
            double s = launchCap / dvLaunchMag;
            for (int k = 0; k < 3; k++) dvLaunch[k] *= s;
            dvLaunchMag = launchCap;
        }

        double dvRemaining = Math.Max(0, dvBudget - dvLaunchMag);
        double vAmag = Norm(vA_I);
        double[] dvImpact = new double[3];
        if (vAmag > 0)
            for (int k = 0; k < 3; k++) dvImpact[k] = -dvRemaining * vA_I[k] / vAmag;

        var sb = new System.Text.StringBuilder();
        sb.AppendLine("2");
        sb.AppendLine(Fmt(tLaunch, dvLaunch));
        sb.AppendLine(Fmt(tImpact, dvImpact));
        Console.Write(sb.ToString());
    }

    static string Fmt(double t, double[] dv) =>
        string.Format(CultureInfo.InvariantCulture, "{0:R} {1:R} {2:R} {3:R}", t, dv[0], dv[1], dv[2]);

    static double[] Sub(double[] a, double[] b) => new double[] { a[0] - b[0], a[1] - b[1], a[2] - b[2] };
    static double Norm(double[] a) => Math.Sqrt(a[0] * a[0] + a[1] * a[1] + a[2] * a[2]);
    static double Dot(double[] a, double[] b) => a[0] * b[0] + a[1] * b[1] + a[2] * b[2];

    static (double[], double[]) Kepler(double[] r0, double[] v0, double dt, double mu)
    {
        if (Math.Abs(dt) < 1e-9) return ((double[])r0.Clone(), (double[])v0.Clone());
        double r0m = Norm(r0);
        double v0m = Norm(v0);
        double vr0 = Dot(r0, v0) / r0m;
        double alpha = 2.0 / r0m - v0m * v0m / mu;

        double x = Math.Sqrt(mu) * Math.Abs(alpha) * dt;
        for (int i = 0; i < 300; i++)
        {
            double x2 = x * x;
            double z = alpha * x2;
            double C = StumpffC(z);
            double S = StumpffS(z);
            double F = r0m * vr0 / Math.Sqrt(mu) * x2 * C + (1 - alpha * r0m) * x2 * x * S + r0m * x - Math.Sqrt(mu) * dt;
            double Fp = r0m * vr0 / Math.Sqrt(mu) * x * (1 - alpha * x2 * S) + (1 - alpha * r0m) * x2 * C + r0m;
            double dx = F / Fp;
            x -= dx;
            if (Math.Abs(dx) < 1e-8) break;
        }

        double x2f = x * x;
        double zf = alpha * x2f;
        double Cf = StumpffC(zf);
        double Sf = StumpffS(zf);
        double f = 1 - x2f / r0m * Cf;
        double g = dt - 1.0 / Math.Sqrt(mu) * x2f * x * Sf;
        double[] r = new double[] { f * r0[0] + g * v0[0], f * r0[1] + g * v0[1], f * r0[2] + g * v0[2] };
        double rm = Norm(r);
        double fdot = Math.Sqrt(mu) / (rm * r0m) * (zf * Sf - 1) * x;
        double gdot = 1 - x2f / rm * Cf;
        double[] v = new double[] { fdot * r0[0] + gdot * v0[0], fdot * r0[1] + gdot * v0[1], fdot * r0[2] + gdot * v0[2] };
        return (r, v);
    }

    static double StumpffC(double z)
    {
        if (z > 1e-6) return (1 - Math.Cos(Math.Sqrt(z))) / z;
        if (z < -1e-6) return (Math.Cosh(Math.Sqrt(-z)) - 1) / (-z);
        return 0.5 - z / 24.0 + z * z / 720.0;
    }
    static double StumpffS(double z)
    {
        if (z > 1e-6) { double sz = Math.Sqrt(z); return (sz - Math.Sin(sz)) / (sz * sz * sz); }
        if (z < -1e-6) { double sz = Math.Sqrt(-z); return (Math.Sinh(sz) - sz) / (sz * sz * sz); }
        return 1.0 / 6.0 - z / 120.0 + z * z / 5040.0;
    }
}
