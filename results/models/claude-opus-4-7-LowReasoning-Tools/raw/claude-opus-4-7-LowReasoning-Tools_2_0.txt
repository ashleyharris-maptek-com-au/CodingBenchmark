using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static int[] head;
    static int[] next_;
    static int[] to;
    static long[] w;
    static bool[] used;
    static int edgeCount;
    static int N;

    static void AddEdge(int u, int v, long weight)
    {
        to[edgeCount] = v; w[edgeCount] = weight; next_[edgeCount] = head[u]; head[u] = edgeCount; edgeCount++;
        to[edgeCount] = u; w[edgeCount] = weight; next_[edgeCount] = head[v]; head[v] = edgeCount; edgeCount++;
    }

    static void EnsureCap(int needed)
    {
        if (needed <= to.Length) return;
        int newCap = to.Length;
        while (newCap < needed) newCap *= 2;
        Array.Resize(ref next_, newCap);
        Array.Resize(ref to, newCap);
        Array.Resize(ref w, newCap);
    }

    static void Main()
    {
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        N = int.Parse(tokens[idx++]);
        int M = int.Parse(tokens[idx++]);

        int cap = Math.Max(16, (M * 2 + 16) * 2);
        head = new int[N];
        for (int i = 0; i < N; i++) head[i] = -1;
        next_ = new int[cap];
        to = new int[cap];
        w = new long[cap];
        edgeCount = 0;

        int[] deg = new int[N];
        for (int i = 0; i < M; i++)
        {
            int u = int.Parse(tokens[idx++]);
            int v = int.Parse(tokens[idx++]);
            long weight = long.Parse(tokens[idx++]);
            AddEdge(u, v, weight);
            deg[u]++; deg[v]++;
        }
        int origHalfEdges = edgeCount;

        var odd = new List<int>();
        for (int i = 0; i < N; i++) if ((deg[i] & 1) == 1) odd.Add(i);

        if (odd.Count > 0)
        {
            bool[] isOdd = new bool[N];
            foreach (var o in odd) isOdd[o] = true;
            bool[] paired = new bool[N];

            var pq = new PriorityQueue<int, long>();
            long[] dist = new long[N];
            int[] parentEdge = new int[N];
            int[] parentNode = new int[N];

            foreach (var s in odd)
            {
                if (paired[s]) continue;
                for (int i = 0; i < N; i++) { dist[i] = long.MaxValue; parentEdge[i] = -1; parentNode[i] = -1; }
                dist[s] = 0;
                pq.Clear();
                pq.Enqueue(s, 0);
                int found = -1;
                while (pq.Count > 0)
                {
                    pq.TryDequeue(out int u, out long d);
                    if (d > dist[u]) continue;
                    if (u != s && isOdd[u] && !paired[u]) { found = u; break; }
                    for (int e = head[u]; e != -1; e = next_[e])
                    {
                        if (e >= origHalfEdges) continue;
                        int v = to[e];
                        long nd = d + w[e];
                        if (nd < dist[v])
                        {
                            dist[v] = nd;
                            parentEdge[v] = e;
                            parentNode[v] = u;
                            pq.Enqueue(v, nd);
                        }
                    }
                }
                if (found == -1) continue;
                paired[s] = true; paired[found] = true;
                int cur = found;
                while (cur != s)
                {
                    int e = parentEdge[cur];
                    int p = parentNode[cur];
                    EnsureCap(edgeCount + 2);
                    AddEdge(p, cur, w[e]);
                    cur = p;
                }
            }
        }

        used = new bool[edgeCount];
        int[] iter = new int[N];
        Array.Copy(head, iter, N);

        var stack = new Stack<int>();
        var circuit = new List<int>(edgeCount / 2 + 1);
        stack.Push(0);
        while (stack.Count > 0)
        {
            int v = stack.Peek();
            while (iter[v] != -1 && used[iter[v]]) iter[v] = next_[iter[v]];
            if (iter[v] == -1)
            {
                circuit.Add(v);
                stack.Pop();
            }
            else
            {
                int e = iter[v];
                used[e] = true;
                used[e ^ 1] = true;
                iter[v] = next_[e];
                stack.Push(to[e]);
            }
        }

        var sb = new StringBuilder();
        sb.Append(circuit.Count);
        sb.Append('\n');
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(circuit[i]);
        }
        sb.Append('\n');
        Console.Out.Write(sb.ToString());
    }
}
