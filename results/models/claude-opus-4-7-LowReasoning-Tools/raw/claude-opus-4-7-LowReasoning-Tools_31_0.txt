use std::io::{self, Read, Write, BufWriter};

struct Tokens<'a> {
    bytes: &'a [u8],
    pos: usize,
}
impl<'a> Tokens<'a> {
    fn new(s: &'a str) -> Self { Tokens { bytes: s.as_bytes(), pos: 0 } }
    fn next_i64(&mut self) -> Option<i64> {
        let b = self.bytes;
        while self.pos < b.len() && (b[self.pos] as char).is_whitespace() { self.pos += 1; }
        if self.pos >= b.len() { return None; }
        let mut neg = false;
        if b[self.pos] == b'-' { neg = true; self.pos += 1; }
        else if b[self.pos] == b'+' { self.pos += 1; }
        let mut v: i64 = 0;
        let mut any = false;
        while self.pos < b.len() && b[self.pos].is_ascii_digit() {
            v = v * 10 + (b[self.pos] - b'0') as i64;
            self.pos += 1;
            any = true;
        }
        if !any { return None; }
        Some(if neg { -v } else { v })
    }
}

fn mitm(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let half = n / 2;
    let an = half;
    let bn = n - half;
    if an > 30 || bn > 30 { return None; }
    let a = &nums[..half];
    let b = &nums[half..];
    let mut bsums: Vec<(i64, u32)> = Vec::with_capacity(1usize << bn);
    for m in 0u32..(1u32 << bn) {
        let mut s = 0i64;
        for i in 0..bn { if (m >> i) & 1 == 1 { s += b[i]; } }
        bsums.push((s, m));
    }
    bsums.sort_by_key(|&(s, _)| s);
    for m in 0u32..(1u32 << an) {
        let mut s = 0i64;
        for i in 0..an { if (m >> i) & 1 == 1 { s += a[i]; } }
        let need = target - s;
        if let Ok(idx) = bsums.binary_search_by_key(&need, |&(x, _)| x) {
            let bm = bsums[idx].1;
            let mut indices = Vec::new();
            for i in 0..an { if (m >> i) & 1 == 1 { indices.push(i); } }
            for i in 0..bn { if (bm >> i) & 1 == 1 { indices.push(half + i); } }
            return Some(indices);
        }
    }
    None
}

fn bitset_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    if target < 0 { return None; }
    let t = target as usize;
    let n = nums.len();
    let words = (t / 64) + 1;
    let mem = (n as u64).saturating_mul(words as u64).saturating_mul(8);
    if mem > 2_000_000_000 { return None; }
    let mut layers: Vec<Vec<u64>> = Vec::with_capacity(n + 1);
    let mut cur = vec![0u64; words];
    cur[0] = 1;
    layers.push(cur.clone());
    let last_word = words - 1;
    let last_bit = t % 64;
    let mask_last: u64 = if last_bit == 63 { !0u64 } else { (1u64 << (last_bit + 1)) - 1 };
    for i in 0..n {
        let v = nums[i];
        if v < 0 || (v as usize) > t {
            layers.push(cur.clone());
            continue;
        }
        let shift = v as usize;
        let wshift = shift / 64;
        let bshift = shift % 64;
        let mut next = cur.clone();
        if bshift == 0 {
            for k in (wshift..words).rev() {
                next[k] |= cur[k - wshift];
            }
        } else {
            for k in (wshift..words).rev() {
                let lo = cur[k - wshift] << bshift;
                let hi = if k > wshift { cur[k - wshift - 1] >> (64 - bshift) } else { 0 };
                next[k] |= lo | hi;
            }
        }
        next[last_word] &= mask_last;
        cur = next;
        layers.push(cur.clone());
    }
    let tw = t / 64;
    let tb = t % 64;
    if (layers[n][tw] >> tb) & 1 == 0 { return None; }
    let mut indices = Vec::new();
    let mut rem = t;
    for i in (0..n).rev() {
        let v = nums[i];
        if v < 0 || (v as usize) > rem { continue; }
        let prev_rem = rem - v as usize;
        let pw = prev_rem / 64;
        let pb = prev_rem % 64;
        if (layers[i][pw] >> pb) & 1 == 1 {
            indices.push(i);
            rem = prev_rem;
            if rem == 0 { break; }
        }
    }
    indices.reverse();
    Some(indices)
}

fn solve(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    if target == 0 { return Some(Vec::new()); }
    if n == 0 { return None; }

    let total: i128 = nums.iter().map(|&x| x as i128).sum();
    if total == target as i128 { return Some((0..n).collect()); }
    for (i, &v) in nums.iter().enumerate() {
        if v == target { return Some(vec![i]); }
    }

    if n <= 50 {
        return mitm(nums, target);
    }

    let all_nonneg = nums.iter().all(|&x| x >= 0);
    if all_nonneg && target > 0 && target <= 50_000_000 {
        if let Some(r) = bitset_dp(nums, target) { return Some(r); }
    }

    // Heuristic: try pair sums
    if n <= 5000 {
        use std::collections::HashMap;
        let mut map: HashMap<i64, usize> = HashMap::new();
        for (i, &v) in nums.iter().enumerate() { map.insert(v, i); }
        for (i, &v) in nums.iter().enumerate() {
            let need = target - v;
            if let Some(&j) = map.get(&need) {
                if j != i { return Some(vec![i.min(j), i.max(j)]); }
            }
        }
    }

    None
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut tok = Tokens::new(&input);
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let n = match tok.next_i64() { Some(v) => v as usize, None => { writeln!(out, "NO").unwrap(); return; } };
    let target = match tok.next_i64() { Some(v) => v, None => { writeln!(out, "NO").unwrap(); return; } };
    let mut nums = Vec::with_capacity(n);
    for _ in 0..n {
        match tok.next_i64() {
            Some(v) => nums.push(v),
            None => break,
        }
    }

    match solve(&nums, target) {
        Some(indices) => {
            writeln!(out, "YES").unwrap();
            write!(out, "[").unwrap();
            for (i, idx) in indices.iter().enumerate() {
                if i > 0 { write!(out, " ").unwrap(); }
                write!(out, "{}", idx).unwrap();
            }
            writeln!(out, "]").unwrap();
        }
        None => {
            writeln!(out, "NO").unwrap();
        }
    }
}
