using System;
using System.Diagnostics;
using System.IO;
using System.Text;

class QAP
{
    static int N;
    static int[][] F;
    static int[][] D;

    static void Main()
    {
        var sw = Stopwatch.StartNew();
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        N = int.Parse(tokens[idx++]);
        F = new int[N][];
        D = new int[N][];
        for (int i = 0; i < N; i++)
        {
            F[i] = new int[N];
            for (int j = 0; j < N; j++) F[i][j] = int.Parse(tokens[idx++]);
        }
        for (int i = 0; i < N; i++)
        {
            D[i] = new int[N];
            for (int j = 0; j < N; j++) D[i][j] = int.Parse(tokens[idx++]);
        }

        long timeLimitMs;
        if (N <= 30) timeLimitMs = 20_000;
        else if (N <= 100) timeLimitMs = 60_000;
        else if (N <= 300) timeLimitMs = 150_000;
        else timeLimitMs = 260_000;

        var rng = new Random(12345);

        int[] best = RandomPerm(rng);
        long bestCost = ComputeCost(best);
        LocalSearch2Opt(best, ref bestCost, sw, timeLimitMs);

        int[] cur = (int[])best.Clone();
        long curCost = bestCost;

        while (sw.ElapsedMilliseconds < timeLimitMs)
        {
            Array.Copy(best, cur, N);
            int kicks = 2 + rng.Next(Math.Max(2, N / 8));
            for (int k = 0; k < kicks; k++)
            {
                int a = rng.Next(N), b = rng.Next(N);
                (cur[a], cur[b]) = (cur[b], cur[a]);
            }
            curCost = ComputeCost(cur);
            LocalSearch2Opt(cur, ref curCost, sw, timeLimitMs);
            if (curCost < bestCost)
            {
                bestCost = curCost;
                Array.Copy(cur, best, N);
            }
        }

        var sb = new StringBuilder();
        sb.Append(bestCost).Append('\n');
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(best[i]);
        }
        sb.Append('\n');
        Console.Write(sb.ToString());
    }

    static int[] RandomPerm(Random rng)
    {
        int[] p = new int[N];
        for (int i = 0; i < N; i++) p[i] = i;
        for (int i = N - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            (p[i], p[j]) = (p[j], p[i]);
        }
        return p;
    }

    static long ComputeCost(int[] p)
    {
        long c = 0;
        for (int i = 0; i < N; i++)
        {
            var Fi = F[i];
            int pi = p[i];
            var Dpi = D[pi];
            for (int j = 0; j < N; j++)
            {
                c += (long)Fi[j] * Dpi[p[j]];
            }
        }
        return c;
    }

    static long SwapDelta(int[] p, int r, int s)
    {
        int pr = p[r], ps = p[s];
        long d = (long)(F[r][r] - F[s][s]) * (D[ps][ps] - D[pr][pr])
               + (long)(F[r][s] - F[s][r]) * (D[ps][pr] - D[pr][ps]);
        for (int k = 0; k < N; k++)
        {
            if (k == r || k == s) continue;
            int pk = p[k];
            d += (long)(F[r][k] - F[s][k]) * (D[ps][pk] - D[pr][pk])
               + (long)(F[k][r] - F[k][s]) * (D[pk][ps] - D[pk][pr]);
        }
        return d;
    }

    static void LocalSearch2Opt(int[] p, ref long cost, Stopwatch sw, long timeLimit)
    {
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                if ((i & 15) == 0 && sw.ElapsedMilliseconds >= timeLimit) return;
                for (int j = i + 1; j < N; j++)
                {
                    long d = SwapDelta(p, i, j);
                    if (d < 0)
                    {
                        (p[i], p[j]) = (p[j], p[i]);
                        cost += d;
                        improved = true;
                    }
                }
            }
        }
    }
}
