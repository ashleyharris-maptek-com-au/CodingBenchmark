@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

fn rotl(x: u32, s: u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let nonce: u32 = params.y + gid.x;
    let difficulty: u32 = params.x;

    var state : array<u32, 16>;
    for (var i: u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    var a: u32;
    var b: u32;
    var c: u32;
    var d: u32;

    for (var r: u32 = 0u; r < 8u; r = r + 1u) {
        // Group 0: state[0..3]
        a = state[0]; b = state[1]; c = state[2]; d = state[3];
        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);
        state[0] = a; state[1] = b; state[2] = c; state[3] = d;

        // Group 1: state[4..7]
        a = state[4]; b = state[5]; c = state[6]; d = state[7];
        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);
        state[4] = a; state[5] = b; state[6] = c; state[7] = d;

        // Group 2: state[8..11]
        a = state[8]; b = state[9]; c = state[10]; d = state[11];
        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);
        state[8] = a; state[9] = b; state[10] = c; state[11] = d;

        // Group 3: state[12..15]
        a = state[12]; b = state[13]; c = state[14]; d = state[15];
        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);
        state[12] = a; state[13] = b; state[14] = c; state[15] = d;
    }

    let h0: u32 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1: u32 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    var tz: u32 = 0u;
    if (h0 == 0u) {
        tz = 32u;
        if (h1 == 0u) {
            tz = 64u;
        } else {
            var v: u32 = h1;
            loop {
                if ((v & 1u) != 0u) { break; }
                v = v >> 1u;
                tz = tz + 1u;
            }
        }
    } else {
        var v: u32 = h0;
        loop {
            if ((v & 1u) != 0u) { break; }
            v = v >> 1u;
            tz = tz + 1u;
        }
    }

    if (tz >= difficulty) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
