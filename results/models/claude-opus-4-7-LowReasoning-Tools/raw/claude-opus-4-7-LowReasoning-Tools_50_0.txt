#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    float dt          = radii.x;
    float sepRadius   = radii.y;
    float alignRadius = radii.z;
    float cohRadius   = radii.w;

    float sepWeight   = weights.x;
    float alignWeight = weights.y;
    float cohWeight   = weights.z;
    float maxSpeed    = weights.w;

    float boundSize   = bounds.x;
    float boundForce  = bounds.y;

    vec3 pos = inData[i * 2u].xyz;
    vec3 vel = inData[i * 2u + 1u].xyz;

    vec3 sepAccum   = vec3(0.0);
    vec3 alignAccum = vec3(0.0);
    vec3 cohAccum   = vec3(0.0);
    int sepCount   = 0;
    int alignCount = 0;
    int cohCount   = 0;

    float sepR2   = sepRadius   * sepRadius;
    float alignR2 = alignRadius * alignRadius;
    float cohR2   = cohRadius   * cohRadius;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue;
        vec3 pj = inData[j * 2u].xyz;
        vec3 vj = inData[j * 2u + 1u].xyz;
        vec3 diff = pos - pj;
        float d2 = dot(diff, diff);
        if (d2 == 0.0) continue;

        if (d2 < sepR2) {
            sepAccum += diff / d2;
            sepCount++;
        }
        if (d2 < alignR2) {
            alignAccum += vj;
            alignCount++;
        }
        if (d2 < cohR2) {
            cohAccum += pj;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0) {
        sepAccum /= float(sepCount);
        accel += sepAccum * sepWeight;
    }
    if (alignCount > 0) {
        alignAccum /= float(alignCount);
        accel += alignAccum * alignWeight;
    }
    if (cohCount > 0) {
        cohAccum /= float(cohCount);
        vec3 toCenter = cohAccum - pos;
        accel += toCenter * cohWeight;
    }

    float boundThresh = boundSize * 0.4;
    float boundScale  = boundSize * 0.5;
    if (abs(pos.x) > boundThresh) {
        accel.x += -sign(pos.x) * boundForce * (abs(pos.x) / boundScale);
    }
    if (abs(pos.y) > boundThresh) {
        accel.y += -sign(pos.y) * boundForce * (abs(pos.y) / boundScale);
    }
    if (abs(pos.z) > boundThresh) {
        accel.z += -sign(pos.z) * boundForce * (abs(pos.z) / boundScale);
    }

    vel += accel * dt;

    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = vel * (maxSpeed / speed);
    }

    pos += vel * dt;

    outData[i * 2u]      = vec4(pos, 0.0);
    outData[i * 2u + 1u] = vec4(vel, 0.0);
}
