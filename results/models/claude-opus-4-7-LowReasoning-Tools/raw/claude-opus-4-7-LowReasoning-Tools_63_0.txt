declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid: string[] = data.grid;
const robots: { id: number; x: number; y: number }[] = data.robots;
const jobs: { id: number; x: number; y: number; priority: number }[] = data.jobs;

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;

function bfs(sx: number, sy: number): Int32Array {
    const dist = new Int32Array(H * W);
    dist.fill(-1);
    if (sx < 0 || sy < 0 || sx >= W || sy >= H) return dist;
    if (grid[sy].charCodeAt(sx) === 35) return dist;
    const queue = new Int32Array(H * W);
    let qh = 0, qt = 0;
    const start = sy * W + sx;
    dist[start] = 0;
    queue[qt++] = start;
    while (qh < qt) {
        const cur = queue[qh++];
        const cy = (cur / W) | 0;
        const cx = cur - cy * W;
        const d = dist[cur] + 1;
        if (cx > 0) {
            const n = cur - 1;
            if (dist[n] === -1 && grid[cy].charCodeAt(cx - 1) !== 35) { dist[n] = d; queue[qt++] = n; }
        }
        if (cx + 1 < W) {
            const n = cur + 1;
            if (dist[n] === -1 && grid[cy].charCodeAt(cx + 1) !== 35) { dist[n] = d; queue[qt++] = n; }
        }
        if (cy > 0) {
            const n = cur - W;
            if (dist[n] === -1 && grid[cy - 1].charCodeAt(cx) !== 35) { dist[n] = d; queue[qt++] = n; }
        }
        if (cy + 1 < H) {
            const n = cur + W;
            if (dist[n] === -1 && grid[cy + 1].charCodeAt(cx) !== 35) { dist[n] = d; queue[qt++] = n; }
        }
    }
    return dist;
}

const sortedJobs = jobs.slice().sort((a, b) => {
    if (b.priority !== a.priority) return b.priority - a.priority;
    return a.id - b.id;
});

const assignments: number[][] = robots.map(() => []);
const robotPos = robots.map(r => ({ x: r.x, y: r.y }));

for (const job of sortedJobs) {
    let bestRobot = -1;
    let bestDist = Infinity;
    for (let i = 0; i < robotPos.length; i++) {
        const rp = robotPos[i];
        const dist = bfs(rp.x, rp.y);
        const d = dist[job.y * W + job.x];
        if (d >= 0 && d < bestDist) {
            bestDist = d;
            bestRobot = i;
        }
    }
    if (bestRobot === -1) bestRobot = 0;
    assignments[bestRobot].push(job.id);
    robotPos[bestRobot] = { x: job.x, y: job.y };
}

process.stdout.write(JSON.stringify({ assignments }));
