; SPIR-V
; Version: 1.0
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_out %gl_in %outColor %inColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
     %uint_4 = OpConstant %uint 4
      %f_thk = OpConstant %float 0.025

  %gl_PerVertex_in = OpTypeStruct %v4float
        %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
     %ptr_in_arr   = OpTypePointer Input %arr_pv_in
            %gl_in = OpVariable %ptr_in_arr Input
        %ptr_in_v4 = OpTypePointer Input %v4float

 %gl_PerVertex_out = OpTypeStruct %v4float
      %ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
           %gl_out = OpVariable %ptr_out_pv Output
       %ptr_out_v4 = OpTypePointer Output %v4float

        %arr_v3_in = OpTypeArray %v3float %uint_4
    %ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
          %inColor = OpVariable %ptr_in_arr_v3 Input
        %ptr_in_v3 = OpTypePointer Input %v3float

       %ptr_out_v3 = OpTypePointer Output %v3float
         %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

       %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
           %p0 = OpLoad %v4float %p0_ptr
       %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
           %p1 = OpLoad %v4float %p1_ptr
       %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
           %p2 = OpLoad %v4float %p2_ptr
       %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
           %p3 = OpLoad %v4float %p3_ptr

       %p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
       %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
       %p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
       %p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1

       %d0    = OpFSub %v2float %p2_xy %p0_xy
       %t0    = OpExtInst %v2float %1 Normalize %d0
       %d1    = OpFSub %v2float %p3_xy %p1_xy
       %t1    = OpExtInst %v2float %1 Normalize %d1

       %t0x   = OpCompositeExtract %float %t0 0
       %t0y   = OpCompositeExtract %float %t0 1
     %neg_t0y = OpFNegate %float %t0y
       %perp0 = OpCompositeConstruct %v2float %neg_t0y %t0x

       %t1x   = OpCompositeExtract %float %t1 0
       %t1y   = OpCompositeExtract %float %t1 1
     %neg_t1y = OpFNegate %float %t1y
       %perp1 = OpCompositeConstruct %v2float %neg_t1y %t1x

       %off0  = OpVectorTimesScalar %v2float %perp0 %f_thk
       %off1  = OpVectorTimesScalar %v2float %perp1 %f_thk

       %off0x = OpCompositeExtract %float %off0 0
       %off0y = OpCompositeExtract %float %off0 1
       %off1x = OpCompositeExtract %float %off1 0
       %off1y = OpCompositeExtract %float %off1 1

       %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
           %c1 = OpLoad %v3float %c1_ptr
       %c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
           %c2 = OpLoad %v3float %c2_ptr

       %p1x = OpCompositeExtract %float %p1 0
       %p1y = OpCompositeExtract %float %p1 1
       %p1z = OpCompositeExtract %float %p1 2
       %p1w = OpCompositeExtract %float %p1 3
       %p2x = OpCompositeExtract %float %p2 0
       %p2y = OpCompositeExtract %float %p2 1
       %p2z = OpCompositeExtract %float %p2 2
       %p2w = OpCompositeExtract %float %p2 3

       %v1x = OpFAdd %float %p1x %off0x
       %v1y = OpFAdd %float %p1y %off0y
       %vert1 = OpCompositeConstruct %v4float %v1x %v1y %p1z %p1w
       %posp1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
       OpStore %posp1 %vert1
       OpStore %outColor %c1
       OpEmitVertex

       %v2x = OpFSub %float %p1x %off0x
       %v2y = OpFSub %float %p1y %off0y
       %vert2 = OpCompositeConstruct %v4float %v2x %v2y %p1z %p1w
       %posp2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
       OpStore %posp2 %vert2
       OpStore %outColor %c1
       OpEmitVertex

       %v3x = OpFAdd %float %p2x %off1x
       %v3y = OpFAdd %float %p2y %off1y
       %vert3 = OpCompositeConstruct %v4float %v3x %v3y %p2z %p2w
       %posp3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
       OpStore %posp3 %vert3
       OpStore %outColor %c2
       OpEmitVertex

       %v4x = OpFSub %float %p2x %off1x
       %v4y = OpFSub %float %p2y %off1y
       %vert4 = OpCompositeConstruct %v4float %v4x %v4y %p2z %p2w
       %posp4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
       OpStore %posp4 %vert4
       OpStore %outColor %c2
       OpEmitVertex

       OpEndPrimitive
       OpReturn
       OpFunctionEnd
