using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Text;

class Program
{
    struct Box
    {
        public int Index;
        public int W, H, D;
    }

    struct Placement
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    struct FreeSpace
    {
        public int X, Y, Z;
        public int W, H, D;
        public long Volume => (long)W * H * D;
    }

    static int CW, CH, CD;
    static Stopwatch SW;
    static long TimeLimitMs = 27000;

    static void Main()
    {
        SW = Stopwatch.StartNew();
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int ti = 0;
        int N = int.Parse(tokens[ti++]);
        CW = int.Parse(tokens[ti++]);
        CH = int.Parse(tokens[ti++]);
        CD = int.Parse(tokens[ti++]);

        var boxes = new Box[N];
        for (int i = 0; i < N; i++)
        {
            boxes[i].Index = i;
            boxes[i].W = int.Parse(tokens[ti++]);
            boxes[i].H = int.Parse(tokens[ti++]);
            boxes[i].D = int.Parse(tokens[ti++]);
        }

        var sortedBoxes = new Box[N];
        Array.Copy(boxes, sortedBoxes, N);
        Array.Sort(sortedBoxes, (a, b) =>
        {
            long va = (long)a.W * a.H * a.D;
            long vb = (long)b.W * b.H * b.D;
            return vb.CompareTo(va);
        });

        var best = PackGuillotine(sortedBoxes);

        if (SW.ElapsedMilliseconds < TimeLimitMs / 2)
        {
            var alt = new Box[N];
            Array.Copy(boxes, alt, N);
            Array.Sort(alt, (a, b) =>
            {
                int ma = Math.Max(a.W, Math.Max(a.H, a.D));
                int mb = Math.Max(b.W, Math.Max(b.H, b.D));
                if (mb != ma) return mb.CompareTo(ma);
                long va = (long)a.W * a.H * a.D;
                long vb = (long)b.W * b.H * b.D;
                return vb.CompareTo(va);
            });
            var alt2 = PackGuillotine(alt);
            if (alt2.Count > best.Count) best = alt2;
        }

        var sb = new StringBuilder();
        sb.Append(best.Count).Append('\n');
        foreach (var p in best)
        {
            sb.Append(p.Index).Append(' ').Append(p.X).Append(' ').Append(p.Y).Append(' ').Append(p.Z).Append('\n');
        }
        Console.Write(sb.ToString());
    }

    static List<Placement> PackGuillotine(Box[] boxes)
    {
        var placed = new List<Placement>();
        var free = new List<FreeSpace>();
        free.Add(new FreeSpace { X = 0, Y = 0, Z = 0, W = CW, H = CH, D = CD });

        foreach (var box in boxes)
        {
            if (SW.ElapsedMilliseconds > TimeLimitMs) break;

            int bestFs = -1;
            int bestBw = 0, bestBh = 0, bestBd = 0;
            long bestScore = long.MaxValue;

            var orients = new (int w, int h, int d)[]
            {
                (box.W, box.H, box.D),
                (box.W, box.D, box.H),
                (box.H, box.W, box.D),
                (box.H, box.D, box.W),
                (box.D, box.W, box.H),
                (box.D, box.H, box.W),
            };

            for (int o = 0; o < 6; o++)
            {
                var (bw, bh, bd) = orients[o];
                bool dup = false;
                for (int pp = 0; pp < o; pp++)
                {
                    if (orients[pp].w == bw && orients[pp].h == bh && orients[pp].d == bd) { dup = true; break; }
                }
                if (dup) continue;

                for (int f = 0; f < free.Count; f++)
                {
                    var fs = free[f];
                    if (fs.W >= bw && fs.H >= bh && fs.D >= bd)
                    {
                        long score = (long)(fs.Z) * 1000000L + (long)(fs.Y) * 1000L + fs.X;
                        if (score < bestScore)
                        {
                            bestScore = score;
                            bestFs = f;
                            bestBw = bw; bestBh = bh; bestBd = bd;
                        }
                    }
                }
            }

            if (bestFs < 0) continue;

            var chosen = free[bestFs];
            var pl = new Placement
            {
                Index = box.Index,
                X = chosen.X, Y = chosen.Y, Z = chosen.Z,
                W = bestBw, H = bestBh, D = bestBd
            };
            placed.Add(pl);

            free.RemoveAt(bestFs);

            int rw = chosen.W - bestBw;
            int rh = chosen.H - bestBh;
            int rd = chosen.D - bestBd;

            if (rw > 0)
                free.Add(new FreeSpace { X = chosen.X + bestBw, Y = chosen.Y, Z = chosen.Z, W = rw, H = chosen.H, D = chosen.D });
            if (rh > 0)
                free.Add(new FreeSpace { X = chosen.X, Y = chosen.Y + bestBh, Z = chosen.Z, W = bestBw, H = rh, D = chosen.D });
            if (rd > 0)
                free.Add(new FreeSpace { X = chosen.X, Y = chosen.Y, Z = chosen.Z + bestBd, W = bestBw, H = bestBh, D = rd });

            if (free.Count > 5000)
            {
                free.Sort((a, b) => b.Volume.CompareTo(a.Volume));
                free.RemoveRange(3000, free.Count - 3000);
            }
        }

        return placed;
    }
}
