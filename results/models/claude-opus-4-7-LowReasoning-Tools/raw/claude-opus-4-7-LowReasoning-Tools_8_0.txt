use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = input.split('\n');
    let header = match lines.next() {
        Some(l) => l,
        None => return,
    };
    let mut it = header.split_whitespace();
    let h: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let w: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };

    let n = h.checked_mul(w).expect("overflow");
    let mut grid: Vec<u8> = Vec::with_capacity(n);
    let mut start: usize = usize::MAX;
    let mut goal: usize = usize::MAX;

    for y in 0..h {
        let line = lines.next().unwrap_or("");
        let bytes = line.as_bytes();
        let mut len = bytes.len();
        if len > 0 && bytes[len - 1] == b'\r' {
            len -= 1;
        }
        for x in 0..w {
            let c = if x < len { bytes[x] } else { b' ' };
            let idx = y * w + x;
            match c {
                b'A' => { start = idx; grid.push(0); }
                b'B' => { goal = idx; grid.push(0); }
                b'#' => { grid.push(1); }
                _ => { grid.push(0); }
            }
        }
    }

    if start == usize::MAX || goal == usize::MAX {
        let _ = writeln!(out, "0");
        return;
    }
    if start == goal {
        let _ = writeln!(out, "1");
        let _ = writeln!(out, "{} {}", start % w, start / w);
        return;
    }

    // parent direction: 0=parent is above (we moved down), 1=parent below (moved up),
    // 2=parent left (moved right), 3=parent right (moved left), 255=unvisited, 254=start
    let mut parent: Vec<u8> = vec![255u8; n];
    parent[start] = 254;

    let mut queue: Vec<usize> = Vec::with_capacity(1024);
    queue.push(start);
    let mut head = 0usize;
    let mut found = false;

    'bfs: while head < queue.len() {
        let cur = queue[head];
        head += 1;
        let x = cur % w;
        let y = cur / w;

        if y > 0 {
            let nb = cur - w;
            if grid[nb] == 0 && parent[nb] == 255 {
                parent[nb] = 1;
                if nb == goal { found = true; break 'bfs; }
                queue.push(nb);
            }
        }
        if y + 1 < h {
            let nb = cur + w;
            if grid[nb] == 0 && parent[nb] == 255 {
                parent[nb] = 0;
                if nb == goal { found = true; break 'bfs; }
                queue.push(nb);
            }
        }
        if x > 0 {
            let nb = cur - 1;
            if grid[nb] == 0 && parent[nb] == 255 {
                parent[nb] = 3;
                if nb == goal { found = true; break 'bfs; }
                queue.push(nb);
            }
        }
        if x + 1 < w {
            let nb = cur + 1;
            if grid[nb] == 0 && parent[nb] == 255 {
                parent[nb] = 2;
                if nb == goal { found = true; break 'bfs; }
                queue.push(nb);
            }
        }
    }

    if !found {
        let _ = writeln!(out, "0");
        return;
    }

    let mut path: Vec<usize> = Vec::new();
    let mut cur = goal;
    loop {
        path.push(cur);
        if cur == start { break; }
        let p = parent[cur];
        cur = match p {
            0 => cur - w,
            1 => cur + w,
            2 => cur - 1,
            3 => cur + 1,
            _ => break,
        };
    }
    path.reverse();

    let _ = writeln!(out, "{}", path.len());
    let mut buf = String::with_capacity(path.len() * 12);
    for idx in path {
        let x = idx % w;
        let y = idx / w;
        buf.push_str(&x.to_string());
        buf.push(' ');
        buf.push_str(&y.to_string());
        buf.push('\n');
    }
    let _ = out.write_all(buf.as_bytes());
}
