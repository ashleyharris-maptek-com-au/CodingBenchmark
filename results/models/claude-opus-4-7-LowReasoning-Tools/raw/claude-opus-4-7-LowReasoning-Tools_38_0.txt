import sys

def run():
    lines = sys.stdin.read().split('\n')
    first = lines[0].split()
    U = int(first[0]); S = int(first[1])
    sets = []
    for i in range(1, S + 1):
        parts = lines[i].split() if i < len(lines) else []
        sets.append([int(x) for x in parts])

    masks = [0] * S
    for i, s in enumerate(sets):
        m = 0
        for e in s:
            m |= (1 << e)
        masks[i] = m
    full = (1 << U) - 1

    elem_sets = [[] for _ in range(U)]
    for i, m in enumerate(masks):
        x = m
        while x:
            b = x & -x
            e = b.bit_length() - 1
            elem_sets[e].append(i)
            x ^= b

    solution = []

    def solve(covered, chosen):
        if covered == full:
            solution.extend(chosen)
            return True
        remaining = full ^ covered
        best_e = -1
        best_count = 10**9
        x = remaining
        while x:
            b = x & -x
            e = b.bit_length() - 1
            cnt = 0
            for i in elem_sets[e]:
                if masks[i] & covered == 0:
                    cnt += 1
            if cnt < best_count:
                best_count = cnt
                best_e = e
                if cnt <= 1:
                    break
            x ^= b
        if best_count == 0:
            return False
        for i in elem_sets[best_e]:
            if masks[i] & covered == 0:
                chosen.append(i)
                if solve(covered | masks[i], chosen):
                    return True
                chosen.pop()
        return False

    sys.setrecursionlimit(100000)
    if solve(0, []):
        solution.sort()
        print("SOLUTION")
        print(" ".join(str(i) for i in solution))
    else:
        print("NO SOLUTION")

run()
