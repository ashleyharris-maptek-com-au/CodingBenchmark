OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 8
       %ext = OpExtInstImport "GLSL.std.450"

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
     %uint_4 = OpConstant %uint 4

    %float_0 = OpConstant %float 0.0
    %float_1 = OpConstant %float 1.0
  %float_thk = OpConstant %float 0.02
%float_third = OpConstant %float 0.33333333
%float_2third = OpConstant %float 0.66666667

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

  %arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
    %inColor = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

  %ps_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
  %p_start = OpLoad %v4float %ps_ptr
  %pe_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
  %p_end = OpLoad %v4float %pe_ptr
  %pp_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
  %p_prev = OpLoad %v4float %pp_ptr
  %pn_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
  %p_next = OpLoad %v4float %pn_ptr

  %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
  %c1 = OpLoad %v3float %c1_ptr
  %c2_ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
  %c2 = OpLoad %v3float %c2_ptr

  %ps_xy = OpVectorShuffle %v2float %p_start %p_start 0 1
  %pe_xy = OpVectorShuffle %v2float %p_end %p_end 0 1
  %pp_xy = OpVectorShuffle %v2float %p_prev %p_prev 0 1
  %pn_xy = OpVectorShuffle %v2float %p_next %p_next 0 1

  %t_s = OpFSub %v2float %pe_xy %pp_xy
  %t_s_n = OpExtInst %v2float %ext Normalize %t_s
  %t_e = OpFSub %v2float %pn_xy %ps_xy
  %t_e_n = OpExtInst %v2float %ext Normalize %t_e

  %ts_x = OpCompositeExtract %float %t_s_n 0
  %ts_y = OpCompositeExtract %float %t_s_n 1
  %ts_ny = OpFNegate %float %ts_y
  %perp_s = OpCompositeConstruct %v2float %ts_ny %ts_x

  %te_x = OpCompositeExtract %float %t_e_n 0
  %te_y = OpCompositeExtract %float %t_e_n 1
  %te_ny = OpFNegate %float %te_y
  %perp_e = OpCompositeConstruct %v2float %te_ny %te_x

  %off_s = OpVectorTimesScalar %v2float %perp_s %float_thk
  %off_e = OpVectorTimesScalar %v2float %perp_e %float_thk

  ; Build positions at t=0, 1/3, 2/3, 1 by linear interp
  %ps_z = OpCompositeExtract %float %p_start 2
  %ps_w = OpCompositeExtract %float %p_start 3
  %pe_z = OpCompositeExtract %float %p_end 2
  %pe_w = OpCompositeExtract %float %p_end 3

  ; pos_a at t=1/3
  %pa_xy = OpExtInst %v2float %ext FMix %ps_xy %pe_xy %float_third_v2
  ; need scalar mix to vector form; use scalar:
  %d_xy = OpFSub %v2float %pe_xy %ps_xy
  %d_z = OpFSub %float %pe_z %ps_z
  %d_w = OpFSub %float %pe_w %ps_w

  %step1_xy = OpVectorTimesScalar %v2float %d_xy %float_third
  %step2_xy = OpVectorTimesScalar %v2float %d_xy %float_2third
  %step1_z = OpFMul %float %d_z %float_third
  %step2_z = OpFMul %float %d_z %float_2third
  %step1_w = OpFMul %float %d_w %float_third
  %step2_w = OpFMul %float %d_w %float_2third

  %pa_xy_r = OpFAdd %v2float %ps_xy %step1_xy
  %pa_z = OpFAdd %float %ps_z %step1_z
  %pa_w = OpFAdd %float %ps_w %step1_w
  %pb_xy = OpFAdd %v2float %ps_xy %step2_xy
  %pb_z = OpFAdd %float %ps_z %step2_z
  %pb_w = OpFAdd %float %ps_w %step2_w

  ; offsets at t=1/3 and t=2/3: lerp between off_s and off_e
  %do_xy = OpFSub %v2float %off_e %off_s
  %odo1 = OpVectorTimesScalar %v2float %do_xy %float_third
  %odo2 = OpVectorTimesScalar %v2float %do_xy %float_2third
  %offa = OpFAdd %v2float %off_s %odo1
  %offb = OpFAdd %v2float %off_s %odo2

  ; Colors at t=0, 1/3, 2/3, 1
  %dc = OpFSub %v3float %c2 %c1
  %dc1 = OpVectorTimesScalar %v3float %dc %float_third
  %dc2 = OpVectorTimesScalar %v3float %dc %float_2third
  %col_a = OpFAdd %v3float %c1 %dc1
  %col_b = OpFAdd %v3float %c1 %dc2

  ; Helpers to build v4 = (xy + off, z, w)
  ; QUAD 1: from t=0 (start) to t=1/3 (a)
  %q1v0_xy = OpFAdd %v2float %ps_xy %off_s
  %q1v0_x = OpCompositeExtract %float %q1v0_xy 0
  %q1v0_y = OpCompositeExtract %float %q1v0_xy 1
  %q1v0 = OpCompositeConstruct %v4float %q1v0_x %q1v0_y %ps_z %ps_w

  %neg_off_s = OpFNegate %v2float %off_s
  %q1v1_xy = OpFAdd %v2float %ps_xy %neg_off_s
  %q1v1_x = OpCompositeExtract %float %q1v1_xy 0
  %q1v1_y = OpCompositeExtract %float %q1v1_xy 1
  %q1v1 = OpCompositeConstruct %v4float %q1v1_x %q1v1_y %ps_z %ps_w

  %q1v2_xy = OpFAdd %v2float %pa_xy_r %offa
  %q1v2_x = OpCompositeExtract %float %q1v2_xy 0
  %q1v2_y = OpCompositeExtract %float %q1v2_xy 1
  %q1v2 = OpCompositeConstruct %v4float %q1v2_x %q1v2_y %pa_z %pa_w

  %neg_offa = OpFNegate %v2float %offa
  %q1v3_xy = OpFAdd %v2float %pa_xy_r %neg_offa
  %q1v3_x = OpCompositeExtract %float %q1v3_xy 0
  %q1v3_y = OpCompositeExtract %float %q1v3_xy 1
  %q1v3 = OpCompositeConstruct %v4float %q1v3_x %q1v3_y %pa_z %pa_w

  ; emit quad 1: order (v0,v1,v2,v3) as triangle strip
  %gp0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
        OpStore %gp0 %q1v0
        OpStore %outColor %c1
        OpEmitVertex
  %gp1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
        OpStore %gp1 %q1v1
        OpStore %outColor %c1
        OpEmitVertex
  %gp2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
        OpStore %gp2 %q1v2
        OpStore %outColor %col_a
        OpEmitVertex
  %gp3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
        OpStore %gp3 %q1v3
        OpStore %outColor %col_a
        OpEmitVertex
        OpEndPrimitive

  ; QUAD 2: from t=2/3 (b) to t=1 (end)
  %q2v0_xy = OpFAdd %v2float %pb_xy %offb
  %q2v0_x = OpCompositeExtract %float %q2v0_xy 0
  %q2v0_y = OpCompositeExtract %float %q2v0_xy 1
  %q2v0 = OpCompositeConstruct %v4float %q2v0_x %q2v0_y %pb_z %pb_w

  %neg_offb = OpFNegate %v2float %offb
  %q2v1_xy = OpFAdd %v2float %pb_xy %neg_offb
  %q2v1_x = OpCompositeExtract %float %q2v1_xy 0
  %q2v1_y = OpCompositeExtract %float %q2v1_xy 1
  %q2v1 = OpCompositeConstruct %v4float %q2v1_x %q2v1_y %pb_z %pb_w

  %q2v2_xy = OpFAdd %v2float %pe_xy %off_e
  %q2v2_x = OpCompositeExtract %float %q2v2_xy 0
  %q2v2_y = OpCompositeExtract %float %q2v2_xy 1
  %q2v2 = OpCompositeConstruct %v4float %q2v2_x %q2v2_y %pe_z %pe_w

  %neg_off_e = OpFNegate %v2float %off_e
  %q2v3_xy = OpFAdd %v2float %pe_xy %neg_off_e
  %q2v3_x = OpCompositeExtract %float %q2v3_xy 0
  %q2v3_y = OpCompositeExtract %float %q2v3_xy 1
  %q2v3 = OpCompositeConstruct %v4float %q2v3_x %q2v3_y %pe_z %pe_w

  %gp4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
        OpStore %gp4 %q2v0
        OpStore %outColor %col_b
        OpEmitVertex
  %gp5 = OpAccessChain %ptr_out_v4 %gl_out %int_0
        OpStore %gp5 %q2v1
        OpStore %outColor %col_b
        OpEmitVertex
  %gp6 = OpAccessChain %ptr_out_v4 %gl_out %int_0
        OpStore %gp6 %q2v2
        OpStore %outColor %c2
        OpEmitVertex
  %gp7 = OpAccessChain %ptr_out_v4 %gl_out %int_0
        OpStore %gp7 %q2v3
        OpStore %outColor %c2
        OpEmitVertex
        OpEndPrimitive

        OpReturn
        OpFunctionEnd
