import sys
import numpy as np

def main():
    data = sys.stdin.buffer.read().split()
    idx = 0
    N = int(data[idx]); idx += 1
    M = int(data[idx]); idx += 1
    arr = np.array(data[idx:idx + 3 * M], dtype=np.int32).reshape(M, 3)

    var_idx = np.abs(arr) - 1
    pos = (arr > 0)

    assign = np.ones(N, dtype=np.bool_)
    rng = np.random.default_rng(0xC0FFEE)

    for it in range(300):
        vals = assign[var_idx]
        lit_true = vals == pos
        clause_sat = lit_true.any(axis=1)
        unsat_mask = ~clause_sat
        if not unsat_mask.any():
            break
        unsat_clauses = np.nonzero(unsat_mask)[0]
        choices = rng.integers(0, 3, size=unsat_clauses.shape[0], dtype=np.int8)
        vars_to_flip = var_idx[unsat_clauses, choices]
        counts = np.bincount(vars_to_flip, minlength=N)
        flip = (counts & 1).astype(np.bool_)
        assign ^= flip

    out = np.where(assign, b'1', b'0')
    sys.stdout.write("SAT\n")
    sys.stdout.buffer.write(b' '.join(out) + b'\n')

main()
