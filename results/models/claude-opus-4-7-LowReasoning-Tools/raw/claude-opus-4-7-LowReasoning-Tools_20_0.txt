use std::io::{self, Read, Write, BufWriter};
use std::collections::HashMap;

#[derive(Clone, Copy)]
struct Vec3 { x: f64, y: f64, z: f64 }

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self { Vec3 { x, y, z } }
    fn add(self, o: Vec3) -> Vec3 { Vec3::new(self.x+o.x, self.y+o.y, self.z+o.z) }
    fn scale(self, s: f64) -> Vec3 { Vec3::new(self.x*s, self.y*s, self.z*s) }
    fn norm(self) -> Vec3 {
        let l = (self.x*self.x + self.y*self.y + self.z*self.z).sqrt();
        if l > 0.0 { Vec3::new(self.x/l, self.y/l, self.z/l) } else { Vec3::new(0.0,0.0,1.0) }
    }
}

struct Sample {
    pos: Vec3,
    values: Vec<f64>,
    hole_idx: usize,
    sample_idx: usize,
}

struct Hole {
    id: String,
    sample_start: usize,
    sample_count: usize,
}

fn parse_f64(s: &str) -> f64 { s.parse::<f64>().unwrap_or(f64::NAN) }

fn transform(v: f64, use_log: bool, minp: f64) -> f64 {
    if !v.is_finite() { return f64::NAN; }
    if use_log {
        let vv = if v <= 0.0 { minp * 0.5 } else { v };
        vv.ln()
    } else { v }
}

fn median_sorted(v: &[f64]) -> f64 {
    if v.is_empty() { return 0.0; }
    v[v.len()/2]
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).ok();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut tokens = input.split_ascii_whitespace();

    let num_holes: usize = match tokens.next().and_then(|t| t.parse().ok()) {
        Some(v) => v, None => { writeln!(out, "0").ok(); return; }
    };
    let num_props: usize = match tokens.next().and_then(|t| t.parse().ok()) {
        Some(v) => v, None => { writeln!(out, "0").ok(); return; }
    };

    let mut prop_names: Vec<String> = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(tokens.next().unwrap_or("?").to_string());
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut samples: Vec<Sample> = Vec::new();

    for h in 0..num_holes {
        let id = match tokens.next() { Some(t) => t.to_string(), None => break };
        let sx = parse_f64(tokens.next().unwrap_or("0"));
        let sy = parse_f64(tokens.next().unwrap_or("0"));
        let sz = parse_f64(tokens.next().unwrap_or("0"));
        let dx = parse_f64(tokens.next().unwrap_or("0"));
        let dy = parse_f64(tokens.next().unwrap_or("0"));
        let dz = parse_f64(tokens.next().unwrap_or("1"));
        let _length = parse_f64(tokens.next().unwrap_or("0"));
        let ns: usize = tokens.next().and_then(|t| t.parse().ok()).unwrap_or(0);
        let start_pt = Vec3::new(sx, sy, sz);
        let dir = Vec3::new(dx, dy, dz).norm();
        let sample_start = samples.len();
        for si in 0..ns {
            let depth = parse_f64(tokens.next().unwrap_or("0"));
            let mut vals = Vec::with_capacity(num_props);
            for _ in 0..num_props {
                let tok = tokens.next().unwrap_or("nan");
                let v = tok.parse::<f64>().unwrap_or(f64::NAN);
                vals.push(v);
            }
            let pos = start_pt.add(dir.scale(depth));
            samples.push(Sample { pos, values: vals, hole_idx: h, sample_idx: si });
        }
        holes.push(Hole { id, sample_start, sample_count: ns });
    }

    let n = samples.len();
    if n == 0 || num_props == 0 { writeln!(out, "0").ok(); return; }

    // Per-property log scaling decision and global robust stats
    let mut prop_log = vec![false; num_props];
    let mut prop_min_pos = vec![1e-6f64; num_props];

    for p in 0..num_props {
        let mut vals: Vec<f64> = samples.iter().filter_map(|s| {
            let v = s.values[p];
            if v.is_finite() { Some(v) } else { None }
        }).collect();
        if vals.is_empty() { continue; }
        vals.sort_by(|a,b| a.partial_cmp(b).unwrap());
        let mn = vals[0];
        let mx = vals[vals.len()-1];
        if mn > 0.0 && mx / mn.max(1e-300) > 100.0 {
            prop_log[p] = true;
        }
        prop_min_pos[p] = if mn > 0.0 { mn } else { 1e-6 };
    }

    // Global robust stats on transformed scale
    let mut tmed = vec![0.0f64; num_props];
    let mut tmad = vec![1.0f64; num_props];
    for p in 0..num_props {
        let mut vals: Vec<f64> = samples.iter().filter_map(|s| {
            let v = s.values[p];
            if v.is_finite() { Some(transform(v, prop_log[p], prop_min_pos[p])) } else { None }
        }).collect();
        if vals.is_empty() { continue; }
        vals.sort_by(|a,b| a.partial_cmp(b).unwrap());
        let med = median_sorted(&vals);
        tmed[p] = med;
        let mut devs: Vec<f64> = vals.iter().map(|v| (v - med).abs()).collect();
        devs.sort_by(|a,b| a.partial_cmp(b).unwrap());
        let mad = median_sorted(&devs);
        tmad[p] = if mad > 1e-12 { mad * 1.4826 } else {
            let m = vals.iter().sum::<f64>() / vals.len() as f64;
            let var = vals.iter().map(|v| (v-m)*(v-m)).sum::<f64>() / vals.len() as f64;
            var.sqrt().max(1e-9)
        };
    }

    // Spatial bounding box and grid for neighbor lookup
    let mut minp = Vec3::new(f64::INFINITY, f64::INFINITY, f64::INFINITY);
    let mut maxp = Vec3::new(f64::NEG_INFINITY, f64::NEG_INFINITY, f64::NEG_INFINITY);
    for s in &samples {
        if s.pos.x < minp.x { minp.x = s.pos.x; }
        if s.pos.y < minp.y { minp.y = s.pos.y; }
        if s.pos.z < minp.z { minp.z = s.pos.z; }
        if s.pos.x > maxp.x { maxp.x = s.pos.x; }
        if s.pos.y > maxp.y { maxp.y = s.pos.y; }
        if s.pos.z > maxp.z { maxp.z = s.pos.z; }
    }
    let ex = (maxp.x - minp.x).max(1.0);
    let ey = (maxp.y - minp.y).max(1.0);
    let ez = (maxp.z - minp.z).max(1.0);
    let vol = ex * ey * ez;
    let cell = ((vol / (n as f64)).cbrt() * 2.5).max(0.5);

    // Adaptive parameters
    let scale_large = n > 200_000;
    let k_neighbors = if scale_large { 8 } else if n > 20_000 { 12 } else { 20 };

    let mut grid: HashMap<(i32,i32,i32), Vec<u32>> = HashMap::with_capacity(n);
    let inv_cell = 1.0 / cell;
    for (i, s) in samples.iter().enumerate() {
        let key = (((s.pos.x-minp.x)*inv_cell) as i32,
                   ((s.pos.y-minp.y)*inv_cell) as i32,
                   ((s.pos.z-minp.z)*inv_cell) as i32);
        grid.entry(key).or_insert_with(Vec::new).push(i as u32);
    }

    struct Suspect {
        hole_idx: usize,
        sample_idx: usize,
        prop_idx: usize,
        confidence: f64,
    }
    let mut suspects: Vec<Suspect> = Vec::new();

    for i in 0..n {
        let s = &samples[i];
        let key = (((s.pos.x-minp.x)*inv_cell) as i32,
                   ((s.pos.y-minp.y)*inv_cell) as i32,
                   ((s.pos.z-minp.z)*inv_cell) as i32);

        let mut cands: Vec<u32> = Vec::with_capacity(32);
        for dx in -1..=1i32 {
            for dy in -1..=1i32 {
                for dz in -1..=1i32 {
                    if let Some(v) = grid.get(&(key.0+dx, key.1+dy, key.2+dz)) {
                        cands.extend_from_slice(v);
                    }
                }
            }
        }
        let mut dists: Vec<(f64, u32)> = cands.iter().filter_map(|&j| {
            if j as usize == i { return None; }
            let o = &samples[j as usize];
            let dx = o.pos.x - s.pos.x;
            let dy = o.pos.y - s.pos.y;
            let dz = o.pos.z - s.pos.z;
            Some((dx*dx + dy*dy + dz*dz, j))
        }).collect();
        if dists.len() < 3 {
            // expand to same-hole neighbors
            let h = &holes[s.hole_idx];
            for k in 0..h.sample_count {
                let j = h.sample_start + k;
                if j == i { continue; }
                let o = &samples[j];
                let dx = o.pos.x - s.pos.x;
                let dy = o.pos.y - s.pos.y;
                let dz = o.pos.z - s.pos.z;
                dists.push((dx*dx + dy*dy + dz*dz, j as u32));
            }
        }
        if dists.len() < 3 { continue; }
        dists.sort_by(|a,b| a.0.partial_cmp(&b.0).unwrap());
        let k = k_neighbors.min(dists.len());
        let nbrs = &dists[..k];

        for p in 0..num_props {
            let v = s.values[p];
            if !v.is_finite() { continue; }
            let tv = transform(v, prop_log[p], prop_min_pos[p]);

            let mut nvals: Vec<f64> = nbrs.iter().filter_map(|&(_,j)| {
                let vv = samples[j as usize].values[p];
                if vv.is_finite() { Some(transform(vv, prop_log[p], prop_min_pos[p])) } else { None }
            }).collect();
            if nvals.len() < 3 { continue; }
            nvals.sort_by(|a,b| a.partial_cmp(b).unwrap());
            let lmed = median_sorted(&nvals);
            let mut ldev: Vec<f64> = nvals.iter().map(|x| (x-lmed).abs()).collect();
            ldev.sort_by(|a,b| a.partial_cmp(b).unwrap());
            let lmad = (median_sorted(&ldev) * 1.4826).max(tmad[p] * 0.1);

            let z_local = ((tv - lmed) / lmad).abs();
            let z_global = ((tv - tmed[p]) / tmad[p]).abs();

            if z_local >= 4.0 && z_global >= 3.0 {
                let conf = (z_local * 0.6 + z_global * 0.4).min(1e6);
                suspects.push(Suspect {
                    hole_idx: s.hole_idx,
                    sample_idx: s.sample_idx,
                    prop_idx: p,
                    confidence: conf,
                });
            }
        }
    }

    suspects.sort_by(|a,b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));

    writeln!(out, "{}", suspects.len()).ok();
    for s in &suspects {
        writeln!(out, "{} {} {} {:.4}",
            holes[s.hole_idx].id, s.sample_idx, prop_names[s.prop_idx], s.confidence).ok();
    }
}
