// CSG Union via BSP tree (classic csg.js algorithm, ported to C++).
// Compile: cl /std:c++20 /O2 /EHsc

#include <cstdint>

using std::vector;

struct V3 {
    double x,y,z;
    V3():x(0),y(0),z(0){}
    V3(double a,double b,double c):x(a),y(b),z(c){}
    V3 operator+(const V3&o)const{return V3(x+o.x,y+o.y,z+o.z);}
    V3 operator-(const V3&o)const{return V3(x-o.x,y-o.y,z-o.z);}
    V3 operator*(double s)const{return V3(x*s,y*s,z*s);}
    double dot(const V3&o)const{return x*o.x+y*o.y+z*o.z;}
    V3 cross(const V3&o)const{return V3(y*o.z-z*o.y,z*o.x-x*o.z,x*o.y-y*o.x);}
    double norm()const{return std::sqrt(x*x+y*y+z*z);}
    V3 unit()const{double n=norm(); return n>0?V3(x/n,y/n,z/n):V3();}
};

struct Plane {
    V3 n; double w;
    Plane():w(0){}
    Plane(const V3&a,const V3&b,const V3&c){ n=(b-a).cross(c-a).unit(); w=n.dot(a); }
    bool ok() const { return n.norm() > 0.5; }
};

static const double EPS = 1e-7;

struct Poly {
    vector<V3> v;
    Plane p;
    Poly() {}
    void flip(){ std::reverse(v.begin(), v.end()); p.n=V3(-p.n.x,-p.n.y,-p.n.z); p.w=-p.w; }
};

static void splitPolygon(const Plane& plane, const Poly& poly,
                         vector<Poly>& coF, vector<Poly>& coB,
                         vector<Poly>& F, vector<Poly>& B) {
    enum { CO=0, FR=1, BA=2, SP=3 };
    int polyType = 0;
    int n = (int)poly.v.size();
    vector<int> types(n);
    for (int i=0;i<n;i++){
        double t = plane.n.dot(poly.v[i]) - plane.w;
        int tp = (t < -EPS)?BA:((t>EPS)?FR:CO);
        polyType |= tp;
        types[i] = tp;
    }
    if (polyType == CO) {
        if (plane.n.dot(poly.p.n) > 0) coF.push_back(poly); else coB.push_back(poly);
    } else if (polyType == FR) {
        F.push_back(poly);
    } else if (polyType == BA) {
        B.push_back(poly);
    } else {
        vector<V3> f, b;
        for (int i=0;i<n;i++){
            int j=(i+1)%n;
            int ti = types[i], tj = types[j];
            const V3& vi = poly.v[i]; const V3& vj = poly.v[j];
            if (ti != BA) f.push_back(vi);
            if (ti != FR) b.push_back(vi);
            if ((ti|tj) == SP) {
                double t = (plane.w - plane.n.dot(vi)) / plane.n.dot(vj - vi);
                V3 vmid = vi + (vj - vi) * t;
                f.push_back(vmid);
                b.push_back(vmid);
            }
        }
        if (f.size() >= 3) {
            for (size_t k=1;k+1<f.size();k++){
                Poly q; q.v = {f[0], f[k], f[k+1]}; q.p = poly.p;
                F.push_back(std::move(q));
            }
        }
        if (b.size() >= 3) {
            for (size_t k=1;k+1<b.size();k++){
                Poly q; q.v = {b[0], b[k], b[k+1]}; q.p = poly.p;
                B.push_back(std::move(q));
            }
        }
    }
}

struct Node {
    Plane plane;
    bool hasPlane=false;
    std::unique_ptr<Node> front, back;
    vector<Poly> polys;

    void build(vector<Poly>&& list) {
        struct Frame { Node* node; vector<Poly> polys; };
        vector<Frame> stk;
        stk.push_back({this, std::move(list)});
        while (!stk.empty()) {
            Frame f = std::move(stk.back()); stk.pop_back();
            if (f.polys.empty()) continue;
            if (!f.node->hasPlane) {
                f.node->plane = f.polys[0].p;
                f.node->hasPlane = true;
            }
            vector<Poly> frontList, backList;
            for (auto& p : f.polys) {
                splitPolygon(f.node->plane, p, f.node->polys, f.node->polys, frontList, backList);
            }
            if (!frontList.empty()) {
                if (!f.node->front) f.node->front.reset(new Node());
                stk.push_back({f.node->front.get(), std::move(frontList)});
            }
            if (!backList.empty()) {
                if (!f.node->back) f.node->back.reset(new Node());
                stk.push_back({f.node->back.get(), std::move(backList)});
            }
        }
    }

    vector<Poly> clipPolygons(vector<Poly> list) const {
        struct Frame { const Node* node; vector<Poly> polys; };
        vector<Poly> result;
        vector<Frame> stk;
        stk.push_back({this, std::move(list)});
        while (!stk.empty()) {
            Frame f = std::move(stk.back()); stk.pop_back();
            if (!f.node->hasPlane) {
                for (auto& p : f.polys) result.push_back(std::move(p));
                continue;
            }
            vector<Poly> frontList, backList;
            for (auto& p : f.polys) {
                splitPolygon(f.node->plane, p, frontList, backList, frontList, backList);
            }
            if (f.node->front) stk.push_back({f.node->front.get(), std::move(frontList)});
            else for (auto& p : frontList) result.push_back(std::move(p));
            if (f.node->back) stk.push_back({f.node->back.get(), std::move(backList)});
        }
        return result;
    }

    void clipTo(const Node& other) {
        vector<Node*> stk; stk.push_back(this);
        while (!stk.empty()) {
            Node* n = stk.back(); stk.pop_back();
            n->polys = other.clipPolygons(std::move(n->polys));
            if (n->front) stk.push_back(n->front.get());
            if (n->back) stk.push_back(n->back.get());
        }
    }

    void invert() {
        vector<Node*> stk; stk.push_back(this);
        while (!stk.empty()) {
            Node* n = stk.back(); stk.pop_back();
            for (auto& p : n->polys) p.flip();
            if (n->hasPlane) { n->plane.n = V3(-n->plane.n.x,-n->plane.n.y,-n->plane.n.z); n->plane.w = -n->plane.w; }
            std::swap(n->front, n->back);
            if (n->front) stk.push_back(n->front.get());
            if (n->back) stk.push_back(n->back.get());
        }
    }

    vector<Poly> allPolygons() const {
        vector<Poly> out;
        vector<const Node*> stk; stk.push_back(const_cast<Node*>(this));
        while (!stk.empty()) {
            const Node* n = stk.back(); stk.pop_back();
            for (auto& p : n->polys) out.push_back(p);
            if (n->front) stk.push_back(n->front.get());
            if (n->back) stk.push_back(n->back.get());
        }
        return out;
    }
};

static char* readAll(size_t& outLen) {
    size_t cap = 1<<20, len = 0;
    char* buf = (char*)std::malloc(cap);
    for(;;){
        if (len + 65536 > cap){ cap*=2; buf=(char*)std::realloc(buf,cap); }
        size_t r = std::fread(buf+len, 1, 65536, stdin);
        len += r;
        if (r==0) break;
    }
    outLen = len;
    return buf;
}

struct Reader {
    const char* p; const char* e;
    Reader(const char* a, const char* b):p(a),e(b){}
    void skip(){ while(p<e && (*p==' '||*p=='\n'||*p=='\r'||*p=='\t')) p++; }
    long long readInt(){ skip(); long long s=1; if(p<e&&*p=='-'){s=-1;p++;} long long v=0; while(p<e && *p>='0'&&*p<='9'){v=v*10+(*p-'0'); p++;} return v*s; }
    double readDouble(){ skip(); const char* st=p; while(p<e && *p!=' '&&*p!='\n'&&*p!='\r'&&*p!='\t') p++; std::string s(st,p-st); return std::atof(s.c_str()); }
};

int main(){
    size_t inLen; char* inBuf = readAll(inLen);
    Reader R(inBuf, inBuf+inLen);
    long long VA=R.readInt(), FA=R.readInt(), VB=R.readInt(), FB=R.readInt();
    vector<V3> va(VA), vb(VB);
    for (long long i=0;i<VA;i++){ va[i].x=R.readDouble(); va[i].y=R.readDouble(); va[i].z=R.readDouble(); }
    vector<std::array<int,3>> fa(FA);
    for (long long i=0;i<FA;i++){ fa[i][0]=(int)R.readInt(); fa[i][1]=(int)R.readInt(); fa[i][2]=(int)R.readInt(); }
    for (long long i=0;i<VB;i++){ vb[i].x=R.readDouble(); vb[i].y=R.readDouble(); vb[i].z=R.readDouble(); }
    vector<std::array<int,3>> fb(FB);
    for (long long i=0;i<FB;i++){ fb[i][0]=(int)R.readInt(); fb[i][1]=(int)R.readInt(); fb[i][2]=(int)R.readInt(); }
    std::free(inBuf);

    auto bbox = [](const vector<V3>& V, V3& lo, V3& hi){
        lo = V[0]; hi = V[0];
        for (auto& p : V){
            lo.x=std::min(lo.x,p.x); lo.y=std::min(lo.y,p.y); lo.z=std::min(lo.z,p.z);
            hi.x=std::max(hi.x,p.x); hi.y=std::max(hi.y,p.y); hi.z=std::max(hi.z,p.z);
        }
    };
    V3 aL,aH,bL,bH;
    if (VA>0) bbox(va,aL,aH);
    if (VB>0) bbox(vb,bL,bH);
    bool disjoint = (VA==0||VB==0) ||
        (aH.x < bL.x - 1e-12 || bH.x < aL.x - 1e-12 ||
         aH.y < bL.y - 1e-12 || bH.y < aL.y - 1e-12 ||
         aH.z < bL.z - 1e-12 || bH.z < aL.z - 1e-12);

    vector<V3> outV;
    vector<std::array<int,3>> outF;

    if (disjoint) {
        outV.reserve(VA+VB);
        outF.reserve(FA+FB);
        for (auto& p : va) outV.push_back(p);
        for (auto& f : fa) outF.push_back(f);
        int off = (int)VA;
        for (auto& p : vb) outV.push_back(p);
        for (auto& f : fb) outF.push_back({f[0]+off,f[1]+off,f[2]+off});
    } else {
        vector<Poly> A; A.reserve(FA);
        for (auto& f : fa) {
            Poly p; p.v = {va[f[0]], va[f[1]], va[f[2]]};
            p.p = Plane(p.v[0],p.v[1],p.v[2]);
            if (p.p.ok()) A.push_back(std::move(p));
        }
        vector<Poly> Bp; Bp.reserve(FB);
        for (auto& f : fb) {
            Poly p; p.v = {vb[f[0]], vb[f[1]], vb[f[2]]};
            p.p = Plane(p.v[0],p.v[1],p.v[2]);
            if (p.p.ok()) Bp.push_back(std::move(p));
        }

        Node a, b;
        a.build(std::move(A));
        b.build(std::move(Bp));

        a.clipTo(b);
        b.clipTo(a);
        b.invert();
        b.clipTo(a);
        b.invert();
        a.build(b.allPolygons());

        auto polys = a.allPolygons();

        struct Key3 { int64_t a,b,c; bool operator==(const Key3&o)const{return a==o.a&&b==o.b&&c==o.c;} };
        struct K3H { size_t operator()(const Key3& k)const{
            uint64_t h=(uint64_t)k.a*1469598103934665603ull;
            h^=(uint64_t)k.b + 0x9e3779b97f4a7c15ull + (h<<6)+(h>>2);
            h^=(uint64_t)k.c + 0x9e3779b97f4a7c15ull + (h<<6)+(h>>2);
            return (size_t)h;
        }};
        std::unordered_map<Key3,int,K3H> vmap;
        vmap.reserve(polys.size()*2);
        auto getIdx = [&](const V3& v)->int{
            const double Q = 1e7;
            Key3 k{(int64_t)std::llround(v.x*Q),(int64_t)std::llround(v.y*Q),(int64_t)std::llround(v.z*Q)};
            auto it = vmap.find(k);
            if (it != vmap.end()) return it->second;
            int id = (int)outV.size();
            outV.push_back(v);
            vmap.emplace(k,id);
            return id;
        };

        for (auto& p : polys) {
            if (p.v.size() < 3) continue;
            int i0 = getIdx(p.v[0]);
            for (size_t k=1;k+1<p.v.size();k++){
                int i1 = getIdx(p.v[k]);
                int i2 = getIdx(p.v[k+1]);
                if (i0==i1||i1==i2||i0==i2) continue;
                outF.push_back({i0,i1,i2});
            }
        }
    }

    static char obuf[1<<16];
    setvbuf(stdout, obuf, _IOFBF, sizeof(obuf));
    std::printf("%zu %zu\n", outV.size(), outF.size());
    for (auto& v : outV) std::printf("%.17g %.17g %.17g\n", v.x, v.y, v.z);
    for (auto& f : outF) std::printf("%d %d %d\n", f[0], f[1], f[2]);
    return 0;
}
