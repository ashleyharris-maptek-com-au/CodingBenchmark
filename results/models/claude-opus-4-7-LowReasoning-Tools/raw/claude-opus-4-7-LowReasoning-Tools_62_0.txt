declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const W: number = data.width;
const H: number = data.height;
const gridRows: string[] = data.grid;
const ignitions: [number, number][] = data.ignitions;
const towns: { id: any; x: number; y: number; value: number }[] = data.towns;
const cutBudget: number = data.cutBudget;

const N = W * H;
const blocked = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = gridRows[y];
  for (let x = 0; x < W; x++) {
    if (row.charCodeAt(x) === 126) blocked[y * W + x] = 1;
  }
}

const townIdx = new Int32Array(N);
for (let i = 0; i < N; i++) townIdx[i] = -1;
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (t.x >= 0 && t.x < W && t.y >= 0 && t.y < H) {
    townIdx[t.y * W + t.x] = i;
  }
}

const ignIdx: number[] = [];
for (const ig of ignitions) {
  const ix = ig[0], iy = ig[1];
  if (ix >= 0 && ix < W && iy >= 0 && iy < H) ignIdx.push(iy * W + ix);
}

function simulate(extraCuts: Set<number>): { burned: Uint8Array; lostValue: number } {
  const burned = new Uint8Array(N);
  const queue = new Int32Array(N);
  let qh = 0, qt = 0;
  for (const s of ignIdx) {
    if (blocked[s] || extraCuts.has(s)) continue;
    if (!burned[s]) { burned[s] = 1; queue[qt++] = s; }
  }
  while (qh < qt) {
    const c = queue[qh++];
    const x = c % W, y = (c / W) | 0;
    if (y > 0) { const n = c - W; if (!burned[n] && !blocked[n] && !extraCuts.has(n)) { burned[n] = 1; queue[qt++] = n; } }
    if (y < H - 1) { const n = c + W; if (!burned[n] && !blocked[n] && !extraCuts.has(n)) { burned[n] = 1; queue[qt++] = n; } }
    if (x > 0) { const n = c - 1; if (!burned[n] && !blocked[n] && !extraCuts.has(n)) { burned[n] = 1; queue[qt++] = n; } }
    if (x < W - 1) { const n = c + 1; if (!burned[n] && !blocked[n] && !extraCuts.has(n)) { burned[n] = 1; queue[qt++] = n; } }
  }
  let lost = 0;
  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    if (t.x < 0 || t.x >= W || t.y < 0 || t.y >= H) continue;
    if (burned[t.y * W + t.x]) lost += t.value;
  }
  return { burned, lostValue: lost };
}

const cuts = new Set<number>();
let sim0 = simulate(cuts);
let curBurned = sim0.burned;
let curLost = sim0.lostValue;

const firebreaks: [number, number][] = [];

function atRiskTowns(burned: Uint8Array): number[] {
  const out: number[] = [];
  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    if (t.x < 0 || t.x >= W || t.y < 0 || t.y >= H) continue;
    if (burned[t.y * W + t.x]) out.push(i);
  }
  return out;
}

const TIME_LIMIT_MS = 8000;
const startTime = Date.now();

while (cuts.size < cutBudget && curLost > 0) {
  if (Date.now() - startTime > TIME_LIMIT_MS) break;

  const risky = atRiskTowns(curBurned);
  if (risky.length === 0) break;

  const dist = new Int32Array(N);
  for (let i = 0; i < N; i++) dist[i] = -1;
  const q = new Int32Array(N);
  let qh = 0, qt = 0;
  for (const ti of risky) {
    const t = towns[ti];
    const c = t.y * W + t.x;
    if (curBurned[c] && !blocked[c] && !cuts.has(c)) { dist[c] = 0; q[qt++] = c; }
  }
  const candidates: number[] = [];
  while (qh < qt) {
    const c = q[qh++];
    candidates.push(c);
    const x = c % W, y = (c / W) | 0;
    const d = dist[c] + 1;
    if (y > 0) { const n = c - W; if (dist[n] === -1 && curBurned[n] && !blocked[n] && !cuts.has(n)) { dist[n] = d; q[qt++] = n; } }
    if (y < H - 1) { const n = c + W; if (dist[n] === -1 && curBurned[n] && !blocked[n] && !cuts.has(n)) { dist[n] = d; q[qt++] = n; } }
    if (x > 0) { const n = c - 1; if (dist[n] === -1 && curBurned[n] && !blocked[n] && !cuts.has(n)) { dist[n] = d; q[qt++] = n; } }
    if (x < W - 1) { const n = c + 1; if (dist[n] === -1 && curBurned[n] && !blocked[n] && !cuts.has(n)) { dist[n] = d; q[qt++] = n; } }
  }

  const remaining = cutBudget - cuts.size;
  const cap = Math.max(80, Math.min(candidates.length, Math.floor(4000 / Math.max(1, remaining)) + 150));
  const evalList = candidates.slice(0, cap);

  let bestCell = -1;
  let bestSave = 0;
  for (const c of evalList) {
    if (townIdx[c] !== -1) continue;
    cuts.add(c);
    const r = simulate(cuts);
    cuts.delete(c);
    const save = curLost - r.lostValue;
    if (save > bestSave) {
      bestSave = save;
      bestCell = c;
    }
    if (Date.now() - startTime > TIME_LIMIT_MS) break;
  }

  if (bestCell === -1 || bestSave <= 0) break;
  cuts.add(bestCell);
  firebreaks.push([bestCell % W, (bestCell / W) | 0]);
  const r = simulate(cuts);
  curBurned = r.burned;
  curLost = r.lostValue;
}

process.stdout.write(JSON.stringify({ firebreaks }));
