; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
        %m4f = OpTypeMatrix %v4f 4

        %UBO = OpTypeStruct %m4f %m4f %m4f %v4f %v4f %v4f
   %ptr_UBO  = OpTypePointer Uniform %UBO
        %ubo = OpVariable %ptr_UBO Uniform

  %ptr_in_v3 = OpTypePointer Input %v3f
  %ptr_in_v2 = OpTypePointer Input %v2f
 %ptr_out_v4 = OpTypePointer Output %v4f
   %ptr_u_v4 = OpTypePointer Uniform %v4f

   %worldPos = OpVariable %ptr_in_v3 Input
     %normal = OpVariable %ptr_in_v3 Input
         %uv = OpVariable %ptr_in_v2 Input
    %tangent = OpVariable %ptr_in_v3 Input
      %color = OpVariable %ptr_in_v3 Input
   %outColor = OpVariable %ptr_out_v4 Output

        %int = OpTypeInt 32 1
      %int_4 = OpConstant %int 4
        %f_0 = OpConstant %float 0.0
        %f_1 = OpConstant %float 1.0
        %f_2 = OpConstant %float 2.0
       %f_05 = OpConstant %float 0.5
       %bool = OpTypeBool

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

         %n0 = OpLoad %v3f %normal
          %N = OpExtInst %v3f %1 Normalize %n0
         %wp = OpLoad %v3f %worldPos
   %cam_ptr  = OpAccessChain %ptr_u_v4 %ubo %int_4
       %cam4 = OpLoad %v4f %cam_ptr
       %cam3 = OpVectorShuffle %v3f %cam4 %cam4 0 1 2
       %vdir = OpFSub %v3f %cam3 %wp
          %V = OpExtInst %v3f %1 Normalize %vdir
       %dotp = OpDot %float %N %V
      %ndotv = OpExtInst %float %1 FMax %dotp %f_0

       %cond = OpFOrdLessThan %bool %ndotv %f_05
               OpSelectionMerge %merge None
               OpBranchConditional %cond %then %else
       %then = OpLabel
       %g_lo = OpFMul %float %f_2 %ndotv
     %col_lo = OpCompositeConstruct %v3f %f_1 %g_lo %f_0
               OpBranch %merge
       %else = OpLabel
         %t1 = OpFSub %float %ndotv %f_05
          %t = OpFMul %float %t1 %f_2
        %omt = OpFSub %float %f_1 %t
     %col_hi = OpCompositeConstruct %v3f %omt %omt %t
               OpBranch %merge
      %merge = OpLabel
        %col = OpPhi %v3f %col_lo %then %col_hi %else
          %r = OpCompositeExtract %float %col 0
          %g = OpCompositeExtract %float %col 1
          %b = OpCompositeExtract %float %col 2
       %outv = OpCompositeConstruct %v4f %r %g %b %f_1
               OpStore %outColor %outv
               OpReturn
               OpFunctionEnd
