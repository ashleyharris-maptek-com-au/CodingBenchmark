// Minimum Edge Cut solver - adaptive across graph scales.
// Strategy:
//   1) Bridge detection (iterative Tarjan): if a bridge exists, K=1.
//   2) Min-degree cut: always an upper bound (cut around lowest-degree vertex).
//   3) Small (N <= 400): exact Stoer-Wagner.
//   4) Medium: random s-t Dinic min-cuts to refine.
//   5) Huge: keep min-degree cut (and limited s-t sampling if memory allows).

#include <cstdio>
#include <cstdlib>
#include <cstdint>
#include <cstring>
#include <vector>
#include <queue>
#include <stack>
#include <algorithm>
#include <random>
#include <climits>
#include <utility>
#include <iostream>

using namespace std;
using i64 = long long;

struct FastIn {
    static const size_t BUF = 1 << 20;
    char buf[BUF];
    size_t pos = 0, len = 0;
    FILE* f;
    FastIn(FILE* fp) : f(fp) {}
    int gc() {
        if (pos == len) {
            len = fread(buf, 1, BUF, f);
            pos = 0;
            if (len == 0) return -1;
        }
        return (unsigned char)buf[pos++];
    }
    bool readInt(i64& x) {
        int c = gc();
        while (c != -1 && (c == ' ' || c == '\n' || c == '\r' || c == '\t')) c = gc();
        if (c == -1) return false;
        int sign = 1;
        if (c == '-') { sign = -1; c = gc(); }
        x = 0;
        while (c >= '0' && c <= '9') { x = x * 10 + (c - '0'); c = gc(); }
        x *= sign;
        return true;
    }
};

static i64 N, M;
static vector<int> E_u, E_v;
static vector<i64> adj_head;
static vector<int> adj_nbr;
static vector<int> adj_eid;

static void buildCSR() {
    adj_head.assign((size_t)N + 1, 0);
    for (i64 i = 0; i < M; ++i) {
        adj_head[E_u[i] + 1]++;
        adj_head[E_v[i] + 1]++;
    }
    for (i64 i = 0; i < N; ++i) adj_head[i + 1] += adj_head[i];
    adj_nbr.assign((size_t)2 * M, 0);
    adj_eid.assign((size_t)2 * M, 0);
    vector<i64> cur(adj_head.begin(), adj_head.end() - 1);
    for (i64 i = 0; i < M; ++i) {
        int u = E_u[i], v = E_v[i];
        adj_nbr[cur[u]] = v; adj_eid[cur[u]++] = (int)i;
        adj_nbr[cur[v]] = u; adj_eid[cur[v]++] = (int)i;
    }
}

static bool findBridge(int& outEid) {
    vector<int> disc(N, -1), low(N, 0), parentEdge(N, -1);
    vector<int> it(N, 0);
    int timer = 0;
    for (i64 s = 0; s < N; ++s) {
        if (disc[s] != -1) continue;
        stack<int> stk;
        stk.push((int)s);
        disc[s] = low[s] = timer++;
        it[s] = 0;
        while (!stk.empty()) {
            int u = stk.top();
            i64 base = adj_head[u];
            i64 end = adj_head[u + 1];
            if ((i64)(base + it[u]) < end) {
                i64 idx = base + it[u]++;
                int v = adj_nbr[idx];
                int eid = adj_eid[idx];
                if (eid == parentEdge[u]) continue;
                if (disc[v] == -1) {
                    disc[v] = low[v] = timer++;
                    parentEdge[v] = eid;
                    it[v] = 0;
                    stk.push(v);
                } else {
                    if (disc[v] < low[u]) low[u] = disc[v];
                }
            } else {
                stk.pop();
                if (!stk.empty()) {
                    int p = stk.top();
                    if (low[u] < low[p]) low[p] = low[u];
                    if (low[u] > disc[p]) {
                        outEid = parentEdge[u];
                        return true;
                    }
                }
            }
        }
    }
    return false;
}

// Stoer-Wagner with dense weight matrix; tracks one side of the best cut.
static int stoerWagner(int n, vector<vector<int>>& w, vector<int>& bestSide) {
    vector<vector<int>> groups(n);
    for (int i = 0; i < n; ++i) groups[i].push_back(i);
    int bestCut = INT_MAX;
    vector<int> bestGroup;
    int curN = n;
    vector<char> alive(n, 1);

    while (curN > 1) {
        vector<int> wts(n, 0);
        vector<char> added(n, 0);
        int prev = -1;
        int start = -1;
        for (int i = 0; i < n; ++i) if (alive[i]) { start = i; break; }
        for (int step = 0; step < curN; ++step) {
            int sel = -1;
            if (step == 0) sel = start;
            else {
                for (int i = 0; i < n; ++i) {
                    if (alive[i] && !added[i]) {
                        if (sel == -1 || wts[i] > wts[sel]) sel = i;
                    }
                }
            }
            added[sel] = 1;
            if (step == curN - 1) {
                if (wts[sel] < bestCut) {
                    bestCut = wts[sel];
                    bestGroup = groups[sel];
                }
                if (prev != -1) {
                    for (int i = 0; i < n; ++i) {
                        if (alive[i] && i != prev && i != sel) {
                            w[prev][i] += w[sel][i];
                            w[i][prev] = w[prev][i];
                        }
                    }
                    for (int x : groups[sel]) groups[prev].push_back(x);
                    alive[sel] = 0;
                    curN--;
                }
                break;
            }
            for (int i = 0; i < n; ++i) {
                if (alive[i] && !added[i]) wts[i] += w[sel][i];
            }
            prev = sel;
        }
    }
    bestSide.assign(n, 0);
    for (int x : bestGroup) bestSide[x] = 1;
    return bestCut;
}

static int solveStoerWagner(vector<pair<int,int>>& outCut) {
    vector<vector<int>> w(N, vector<int>(N, 0));
    for (i64 i = 0; i < M; ++i) {
        int u = E_u[i], v = E_v[i];
        if (u == v) continue;
        w[u][v]++;
        w[v][u]++;
    }
    vector<int> side;
    int cut = stoerWagner((int)N, w, side);
    outCut.clear();
    for (i64 i = 0; i < M; ++i) {
        int u = E_u[i], v = E_v[i];
        if (side[u] != side[v]) outCut.emplace_back(u, v);
    }
    return cut;
}

struct Dinic {
    struct Edge { int to, rev; int cap; };
    int n;
    vector<vector<Edge>> g;
    vector<int> level, iter;
    Dinic(int n_) : n(n_), g(n_), level(n_), iter(n_) {}
    void addUndirected(int u, int v) {
        g[u].push_back({v, (int)g[v].size(), 1});
        g[v].push_back({u, (int)g[u].size() - 1, 1});
    }
    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0; q.push(s);
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (auto& e : g[u]) if (e.cap > 0 && level[e.to] < 0) {
                level[e.to] = level[u] + 1; q.push(e.to);
            }
        }
        return level[t] >= 0;
    }
    int dfs(int u, int t, int f) {
        if (u == t) return f;
        for (int& i = iter[u]; i < (int)g[u].size(); ++i) {
            Edge& e = g[u][i];
            if (e.cap > 0 && level[e.to] == level[u] + 1) {
                int d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) { e.cap -= d; g[e.to][e.rev].cap += d; return d; }
            }
        }
        return 0;
    }
    int maxflow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            int f;
            while ((f = dfs(s, t, INT_MAX)) > 0) flow += f;
        }
        return flow;
    }
    vector<char> sideOfS(int s) {
        vector<char> side(n, 0);
        queue<int> q; q.push(s); side[s] = 1;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (auto& e : g[u]) if (e.cap > 0 && !side[e.to]) {
                side[e.to] = 1; q.push(e.to);
            }
        }
        return side;
    }
};

static int minSTCut(int s, int t, vector<pair<int,int>>& cutEdges) {
    Dinic D((int)N);
    for (i64 i = 0; i < M; ++i) {
        int u = E_u[i], v = E_v[i];
        if (u == v) continue;
        D.addUndirected(u, v);
    }
    int flow = D.maxflow(s, t);
    auto side = D.sideOfS(s);
    cutEdges.clear();
    for (i64 i = 0; i < M; ++i) {
        int u = E_u[i], v = E_v[i];
        if (u == v) continue;
        if (side[u] != side[v]) cutEdges.emplace_back(u, v);
    }
    return flow;
}

static int solveGlobalViaSTCuts(vector<pair<int,int>>& bestCut, int trials) {
    int best = INT_MAX;
    int s = 0;
    mt19937 rng(12345);
    vector<int> ts;
    for (i64 i = 1; i < N; ++i) ts.push_back((int)i);
    shuffle(ts.begin(), ts.end(), rng);
    if ((int)ts.size() > trials) ts.resize(trials);

    for (int t : ts) {
        vector<pair<int,int>> cut;
        int f = minSTCut(s, t, cut);
        if (f < best) {
            best = f;
            bestCut = cut;
            if (best <= 1) break;
        }
    }
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    FastIn in(stdin);
    if (!in.readInt(N) || !in.readInt(M)) {
        printf("0\n");
        return 0;
    }
    if (N <= 1 || M == 0) {
        printf("0\n");
        return 0;
    }

    E_u.resize(M);
    E_v.resize(M);
    for (i64 i = 0; i < M; ++i) {
        i64 a, b;
        if (!in.readInt(a) || !in.readInt(b)) { E_u[i] = 0; E_v[i] = 0; continue; }
        E_u[i] = (int)a;
        E_v[i] = (int)b;
    }

    buildCSR();

    int bridgeEid = -1;
    if (findBridge(bridgeEid) && bridgeEid >= 0) {
        printf("1\n%d %d\n", E_u[bridgeEid], E_v[bridgeEid]);
        return 0;
    }

    vector<pair<int,int>> bestCut;
    int bestK = INT_MAX;

    int minDeg = INT_MAX, minDegV = 0;
    for (i64 i = 0; i < N; ++i) {
        int d = (int)(adj_head[i + 1] - adj_head[i]);
        if (d < minDeg) { minDeg = d; minDegV = (int)i; }
    }
    bestK = minDeg;
    bestCut.clear();
    for (i64 i = adj_head[minDegV]; i < adj_head[minDegV + 1]; ++i) {
        int eid = adj_eid[i];
        bestCut.emplace_back(E_u[eid], E_v[eid]);
    }

    if (N <= 400) {
        vector<pair<int,int>> swCut;
        int k = solveStoerWagner(swCut);
        if (k < bestK) { bestK = k; bestCut = swCut; }
    } else if ((i64)N * M <= 50'000'000LL) {
        vector<pair<int,int>> stCut;
        int k = solveGlobalViaSTCuts(stCut, (int)min<i64>(N - 1, 32));
        if (k < bestK) { bestK = k; bestCut = stCut; }
    } else if (M <= 5'000'000) {
        vector<pair<int,int>> stCut;
        int k = solveGlobalViaSTCuts(stCut, 8);
        if (k < bestK) { bestK = k; bestCut = stCut; }
    }

    printf("%d\n", bestK);
    for (auto& e : bestCut) printf("%d %d\n", e.first, e.second);
    return 0;
}
