; SPIR-V
; Version: 1.0
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 7

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2
        %f_0 = OpConstant %float 0.0
    %f_0_015 = OpConstant %float 0.015
     %f_0_05 = OpConstant %float 0.05
     %f_0_75 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

  %arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

       %p0p = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %p0 = OpLoad %v4float %p0p
       %p1p = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %p1 = OpLoad %v4float %p1p

       %c0p = OpAccessChain %ptr_in_v3 %color %int_0
        %c0 = OpLoad %v3float %c0p
       %c1p = OpAccessChain %ptr_in_v3 %color %int_1
        %c1 = OpLoad %v3float %c1p

       %p0x = OpCompositeExtract %float %p0 0
       %p0y = OpCompositeExtract %float %p0 1
       %p1x = OpCompositeExtract %float %p1 0
       %p1y = OpCompositeExtract %float %p1 1

        %dx = OpFSub %float %p1x %p0x
        %dy = OpFSub %float %p1y %p0y
       %ndy = OpFNegate %float %dy
   %perpRaw = OpCompositeConstruct %v2float %ndy %dx
      %perp = OpExtInst %v2float %1 Normalize %perpRaw
      %perpX = OpCompositeExtract %float %perp 0
      %perpY = OpCompositeExtract %float %perp 1

       %ox_s = OpFMul %float %perpX %f_0_015
       %oy_s = OpFMul %float %perpY %f_0_015
       %ox_a = OpFMul %float %perpX %f_0_05
       %oy_a = OpFMul %float %perpY %f_0_05

       %diff = OpFSub %v4float %p1 %p0
     %scaled = OpVectorTimesScalar %v4float %diff %f_0_75
        %mid = OpFAdd %v4float %p0 %scaled
       %midx = OpCompositeExtract %float %mid 0
       %midy = OpCompositeExtract %float %mid 1
       %midz = OpCompositeExtract %float %mid 2
       %midw = OpCompositeExtract %float %mid 3

       %p0z = OpCompositeExtract %float %p0 2
       %p0w = OpCompositeExtract %float %p0 3

       ; v0 = p0 + perp*0.015
       %v0x = OpFAdd %float %p0x %ox_s
       %v0y = OpFAdd %float %p0y %oy_s
       %v0  = OpCompositeConstruct %v4float %v0x %v0y %p0z %p0w
      %gp0  = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %gp0 %v0
              OpStore %outColor %c0
              OpEmitVertex

       ; v1 = p0 - perp*0.015
       %v1x = OpFSub %float %p0x %ox_s
       %v1y = OpFSub %float %p0y %oy_s
       %v1  = OpCompositeConstruct %v4float %v1x %v1y %p0z %p0w
      %gp1  = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %gp1 %v1
              OpStore %outColor %c0
              OpEmitVertex

       ; v2 = mid + perp*0.015
       %v2x = OpFAdd %float %midx %ox_s
       %v2y = OpFAdd %float %midy %oy_s
       %v2  = OpCompositeConstruct %v4float %v2x %v2y %midz %midw
      %gp2  = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %gp2 %v2
              OpStore %outColor %c0
              OpEmitVertex

       ; v3 = mid - perp*0.015
       %v3x = OpFSub %float %midx %ox_s
       %v3y = OpFSub %float %midy %oy_s
       %v3  = OpCompositeConstruct %v4float %v3x %v3y %midz %midw
      %gp3  = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %gp3 %v3
              OpStore %outColor %c0
              OpEmitVertex
              OpEndPrimitive

       ; v4 = mid + perp*0.05
       %v4x = OpFAdd %float %midx %ox_a
       %v4y = OpFAdd %float %midy %oy_a
       %v4v = OpCompositeConstruct %v4float %v4x %v4y %midz %midw
      %gp4  = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %gp4 %v4v
              OpStore %outColor %c1
              OpEmitVertex

       ; v5 = mid - perp*0.05
       %v5x = OpFSub %float %midx %ox_a
       %v5y = OpFSub %float %midy %oy_a
       %v5v = OpCompositeConstruct %v4float %v5x %v5y %midz %midw
      %gp5  = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %gp5 %v5v
              OpStore %outColor %c1
              OpEmitVertex

       ; v6 = p1 (tip)
      %gp6  = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %gp6 %p1
              OpStore %outColor %c1
              OpEmitVertex
              OpEndPrimitive

              OpReturn
              OpFunctionEnd
