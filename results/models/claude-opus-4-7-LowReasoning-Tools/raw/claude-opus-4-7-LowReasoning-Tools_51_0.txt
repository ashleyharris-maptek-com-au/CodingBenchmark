import math

_state = {
    "init": False,
    "ias_i": 0.0,
    "roll_i": 0.0,
    "pitch_i": 0.0,
    "trim": 0.0,
    "time": 0.0,
    "throttle": 0.6,
    "flaps": 0,
    "gear": 0,
    "speedbrake": 0.0,
    "stall_recovery_until": 0.0,
    "overspeed_until": 0.0,
    "last_alt": None,
}


def _median(vals):
    vals = sorted(vals)
    n = len(vals)
    if n == 0:
        return None
    if n % 2 == 1:
        return vals[n // 2]
    return 0.5 * (vals[n // 2 - 1] + vals[n // 2])


def _fuse(state, keys, valid_keys=None, plausible=None):
    readings = []
    all_readings = []
    for i, k in enumerate(keys):
        v = state.get(k)
        if v is None:
            continue
        try:
            fv = float(v)
        except (TypeError, ValueError):
            continue
        if not math.isfinite(fv):
            continue
        if plausible is not None and not (plausible[0] <= fv <= plausible[1]):
            continue
        all_readings.append(fv)
        if valid_keys is not None:
            if not state.get(valid_keys[i], True):
                continue
        readings.append(fv)
    pool = readings if readings else all_readings
    if not pool:
        return None
    if len(pool) >= 3:
        med = _median(pool)
        tol = max(500.0, abs(med) * 0.2 + 50.0)
        filt = [v for v in pool if abs(v - med) <= tol]
        return _median(filt) if len(filt) >= 2 else med
    if len(pool) == 2:
        if abs(pool[0] - pool[1]) > max(200.0, abs(pool[0]) * 0.1 + 20.0) and all_readings:
            m = _median(all_readings)
            return min(pool, key=lambda x: abs(x - m))
        return 0.5 * (pool[0] + pool[1])
    return pool[0]


def _clamp(x, lo, hi):
    return max(lo, min(hi, x))


def _angle_diff(a, b):
    d = (a - b) % 360.0
    if d > 180.0:
        d -= 360.0
    return d


def autopilot_step(state: dict, dt: float) -> dict:
    if dt is None or dt <= 0:
        dt = float(state.get("dt", 0.05) or 0.05)
    dt = max(1e-3, min(0.5, dt))

    s = _state
    s["time"] += dt

    baro_alt = _fuse(state,
        ["baro_alt_1_ft", "baro_alt_2_ft", "baro_alt_3_ft"],
        ["baro_alt_1_valid", "baro_alt_2_valid", "baro_alt_3_valid"],
        plausible=(-2000, 60000))
    radar_alt = _fuse(state,
        ["radar_alt_1_ft", "radar_alt_2_ft", "radar_alt_3_ft"],
        ["radar_alt_1_valid", "radar_alt_2_valid", "radar_alt_3_valid"],
        plausible=(-10, 5000))
    gps_alt = _fuse(state,
        ["gps_1_alt_ft", "gps_2_alt_ft", "gps_3_alt_ft"],
        ["gps_1_valid", "gps_2_valid", "gps_3_valid"],
        plausible=(-2000, 60000))

    if baro_alt is not None and gps_alt is not None and abs(baro_alt - gps_alt) < 1000:
        altitude = 0.5 * (baro_alt + gps_alt)
    elif baro_alt is not None and gps_alt is not None:
        if radar_alt is not None and radar_alt < 2500 and abs(radar_alt - gps_alt) < abs(radar_alt - baro_alt):
            altitude = gps_alt
        else:
            altitude = baro_alt
    elif baro_alt is not None:
        altitude = baro_alt
    elif gps_alt is not None:
        altitude = gps_alt
    elif radar_alt is not None:
        altitude = radar_alt
    else:
        altitude = s.get("last_alt") or 5000.0
    s["last_alt"] = altitude

    ias = _fuse(state,
        ["ias_1_kt", "ias_2_kt", "ias_3_kt"],
        ["ias_1_valid", "ias_2_valid", "ias_3_valid"],
        plausible=(0, 500))
    if ias is None:
        gs = _fuse(state,
            ["gps_1_gs_kt", "gps_2_gs_kt", "gps_3_gs_kt"],
            ["gps_1_valid", "gps_2_valid", "gps_3_valid"],
            plausible=(0, 600))
        ias = gs if gs is not None else 250.0

    mach = _fuse(state, ["mach_1", "mach_2", "mach_3"], None, plausible=(0, 1.0))
    aoa = _fuse(state,
        ["aoa_1_deg", "aoa_2_deg", "aoa_3_deg"],
        ["aoa_1_valid", "aoa_2_valid", "aoa_3_valid"],
        plausible=(-20, 40))

    heading = _fuse(state,
        ["irs_1_heading_deg", "irs_2_heading_deg", "irs_3_heading_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        plausible=(0, 360))
    if heading is None:
        heading = state.get("magnetic_heading_deg") or state.get("true_heading_deg") or 0.0

    pitch = _fuse(state,
        ["irs_1_pitch_deg", "irs_2_pitch_deg", "irs_3_pitch_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        plausible=(-30, 45)) or 0.0
    roll = _fuse(state,
        ["irs_1_roll_deg", "irs_2_roll_deg", "irs_3_roll_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        plausible=(-90, 90)) or 0.0
    vs = _fuse(state,
        ["irs_1_vs_fpm", "irs_2_vs_fpm", "irs_3_vs_fpm"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        plausible=(-10000, 10000))
    if vs is None:
        vs = state.get("vertical_speed_fpm") or 0.0

    pitch_rate = _fuse(state,
        ["gyro_1_pitch_rate_dps", "gyro_2_pitch_rate_dps", "gyro_3_pitch_rate_dps"],
        ["gyro_1_valid", "gyro_2_valid", "gyro_3_valid"],
        plausible=(-60, 60)) or 0.0
    roll_rate = _fuse(state,
        ["gyro_1_roll_rate_dps", "gyro_2_roll_rate_dps", "gyro_3_roll_rate_dps"],
        ["gyro_1_valid", "gyro_2_valid", "gyro_3_valid"],
        plausible=(-90, 90)) or 0.0
    yaw_rate = _fuse(state,
        ["gyro_1_yaw_rate_dps", "gyro_2_yaw_rate_dps", "gyro_3_yaw_rate_dps"],
        ["gyro_1_valid", "gyro_2_valid", "gyro_3_valid"],
        plausible=(-30, 30)) or 0.0

    target_alt = state.get("target_alt_ft")
    if target_alt is None:
        target_alt = altitude
    target_hdg = state.get("target_hdg_deg")
    if target_hdg is None:
        target_hdg = heading
    target_ias = state.get("target_ias_kt")
    if target_ias is None:
        target_ias = max(180.0, min(280.0, ias))
    target_vs = state.get("target_vs_fpm")

    wow = (state.get("weight_on_wheels_left") or state.get("weight_on_wheels_right")
           or state.get("weight_on_wheels_nose"))
    radar_low = radar_alt if radar_alt is not None else 99999.0

    eng1_run = state.get("engine_1_running", True) and not state.get("engine_1_fire", False)
    eng2_run = state.get("engine_2_running", True) and not state.get("engine_2_fire", False)

    if not s["init"]:
        s["init"] = True
        s["throttle"] = 0.6

    stall = False
    if aoa is not None and aoa > 14.0:
        stall = True
    if ias < 130.0 and radar_low > 200 and not wow:
        stall = True
    if stall:
        s["stall_recovery_until"] = s["time"] + 2.0

    overspeed = False
    if ias > 350.0:
        overspeed = True
    if mach is not None and mach > 0.80:
        overspeed = True
    if overspeed:
        s["overspeed_until"] = s["time"] + 1.0

    in_stall = s["time"] < s["stall_recovery_until"]
    in_overspeed = s["time"] < s["overspeed_until"]

    if in_stall:
        thr_cmd = 1.0
    elif in_overspeed:
        thr_cmd = 0.05
    else:
        err = target_ias - ias
        s["ias_i"] = _clamp(s["ias_i"] + err * dt, -200.0, 200.0)
        if target_vs is not None:
            vs_ff = target_vs / 4000.0
        else:
            vs_ff = (target_alt - altitude) / 50000.0
        base = s["throttle"]
        thr_cmd = base + (0.01 * err + 0.0015 * s["ias_i"]) * dt * 20.0 + vs_ff * 0.1
        thr_cmd = _clamp(thr_cmd, 0.15, 1.0)
    s["throttle"] = 0.9 * s["throttle"] + 0.1 * thr_cmd
    throttle = _clamp(s["throttle"], 0.0, 1.0)

    t1 = throttle if eng1_run else 0.0
    t2 = throttle if eng2_run else 0.0
    if eng1_run and not eng2_run:
        t1 = min(1.0, throttle + 0.2)
    if eng2_run and not eng1_run:
        t2 = min(1.0, throttle + 0.2)

    alt_err = target_alt - altitude
    if target_vs is not None:
        desired_vs = target_vs
    else:
        desired_vs = _clamp(alt_err * 10.0, -2500.0, 2500.0)
        if abs(alt_err) < 200:
            desired_vs = alt_err * 5.0

    if in_stall:
        pitch_target = 2.0
    elif in_overspeed:
        pitch_target = max(0.0, pitch + 1.0)
    else:
        tas = ias if ias > 50 else 250.0
        ratio = _clamp(desired_vs / (tas * 101.3 + 1e-6), -0.3, 0.3)
        pitch_for_vs = math.degrees(math.asin(ratio))
        pitch_target = _clamp(pitch_for_vs + 2.0, -10.0, 15.0)

    pitch_err = pitch_target - pitch
    s["pitch_i"] = _clamp(s["pitch_i"] + pitch_err * dt, -10.0, 10.0)
    elevator = 0.08 * pitch_err + 0.01 * s["pitch_i"] - 0.04 * pitch_rate
    elevator = _clamp(elevator, -1.0, 1.0)

    if abs(pitch_err) < 3.0:
        s["trim"] += _clamp(elevator * 0.1, -0.05, 0.05) * dt
        s["trim"] = _clamp(s["trim"], -1.0, 1.0)

    hdg_err = _angle_diff(target_hdg, heading)
    desired_bank = _clamp(hdg_err * 1.5, -25.0, 25.0)
    roll_err = desired_bank - roll
    s["roll_i"] = _clamp(s["roll_i"] + roll_err * dt, -20.0, 20.0)
    aileron = 0.04 * roll_err + 0.002 * s["roll_i"] - 0.02 * roll_rate
    aileron = _clamp(aileron, -1.0, 1.0)

    rudder = -0.05 * yaw_rate
    thrust_asym = t1 - t2
    rudder += 0.4 * thrust_asym
    rudder += 0.005 * desired_bank
    rudder = _clamp(rudder, -1.0, 1.0)

    sb = 0.0
    if in_overspeed:
        sb = 1.0
    elif ias > target_ias + 20 and alt_err < -200:
        sb = _clamp((ias - target_ias - 20) / 30.0, 0.0, 0.7)
    if in_stall:
        sb = 0.0
    s["speedbrake"] = sb

    if wow:
        flaps_cmd = s["flaps"]
    elif radar_low < 1500 and ias < 180:
        flaps_cmd = 30
    elif radar_low < 3000 and ias < 200:
        flaps_cmd = 15
    elif ias < 180:
        flaps_cmd = 10
    elif ias < 210:
        flaps_cmd = 5
    else:
        flaps_cmd = 0
    s["flaps"] = flaps_cmd

    if wow:
        gear_cmd = 1
    elif radar_low < 1500 and vs < 0 and ias < 200:
        gear_cmd = 1
    elif radar_low > 2500 or ias > 220:
        gear_cmd = 0
    else:
        gear_cmd = s["gear"]
    s["gear"] = gear_cmd

    return {
        "elevator": float(elevator),
        "aileron": float(aileron),
        "rudder": float(rudder),
        "throttle_1": float(_clamp(t1, 0.0, 1.0)),
        "throttle_2": float(_clamp(t2, 0.0, 1.0)),
        "elevator_trim": float(_clamp(s["trim"], -1.0, 1.0)),
        "flaps": int(flaps_cmd),
        "speedbrake": float(_clamp(sb, 0.0, 1.0)),
        "gear": int(gear_cmd),
    }
