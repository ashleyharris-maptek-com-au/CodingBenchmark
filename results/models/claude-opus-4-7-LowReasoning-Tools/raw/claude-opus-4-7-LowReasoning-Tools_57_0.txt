const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const W = data.width, H = data.height;
const grid = data.grid;
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = data.cutBudget | 0;

function inb(x, y) { return x >= 0 && y >= 0 && x < W && y < H; }
function isForest(x, y) { return inb(x, y) && grid[y][x] === '.'; }

const ignSet = new Set();
for (const ig of ignitions) ignSet.add(ig[1] * W + ig[0]);

function reachableFrom(cutSet) {
  const seen = new Uint8Array(W * H);
  const q = new Int32Array(W * H);
  let qh = 0, qt = 0;
  for (const ig of ignitions) {
    const x = ig[0], y = ig[1];
    if (!isForest(x, y)) continue;
    const k = y * W + x;
    if (cutSet.has(k)) continue;
    if (!seen[k]) { seen[k] = 1; q[qt++] = k; }
  }
  while (qh < qt) {
    const k = q[qh++];
    const x = k % W, y = (k - x) / W;
    const ns = [[x+1,y],[x-1,y],[x,y+1],[x,y-1]];
    for (const [nx, ny] of ns) {
      if (!isForest(nx, ny)) continue;
      const nk = ny * W + nx;
      if (cutSet.has(nk) || seen[nk]) continue;
      seen[nk] = 1;
      q[qt++] = nk;
    }
  }
  return seen;
}

function minCutForTown(townX, townY, cutSet, budget) {
  const townK = townY * W + townX;
  const N = 2 * W * H + 2;
  const SRC = 2 * W * H;
  const SNK = 2 * W * H + 1;
  const INF = 1e9;
  const head = new Int32Array(N).fill(-1);
  const nextE = [];
  const toE = [];
  const capE = [];
  function addEdge(u, v, c) {
    nextE.push(head[u]); toE.push(v); capE.push(c); head[u] = nextE.length - 1;
    nextE.push(head[v]); toE.push(u); capE.push(0); head[v] = nextE.length - 1;
  }
  for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
      if (!isForest(x, y)) continue;
      const k = y * W + x;
      if (cutSet.has(k)) continue;
      const isIg = ignSet.has(k);
      const isTown = (k === townK);
      const cap = (isIg || isTown) ? INF : 1;
      addEdge(2 * k, 2 * k + 1, cap);
      if (x + 1 < W && isForest(x + 1, y) && !cutSet.has(k + 1)) {
        addEdge(2 * k + 1, 2 * (k + 1), INF);
        addEdge(2 * (k + 1) + 1, 2 * k, INF);
      }
      if (y + 1 < H && isForest(x, y + 1) && !cutSet.has(k + W)) {
        addEdge(2 * k + 1, 2 * (k + W), INF);
        addEdge(2 * (k + W) + 1, 2 * k, INF);
      }
    }
  }
  for (const ig of ignitions) {
    if (isForest(ig[0], ig[1])) addEdge(SRC, 2 * (ig[1] * W + ig[0]), INF);
  }
  addEdge(2 * townK + 1, SNK, INF);

  const parentE = new Int32Array(N);
  const parentN = new Int32Array(N);
  const bfsq = new Int32Array(N);
  let flow = 0;
  while (true) {
    parentE.fill(-1);
    parentN[SRC] = SRC;
    parentE[SRC] = -2;
    let qh = 0, qt = 0;
    bfsq[qt++] = SRC;
    let found = false;
    while (qh < qt && !found) {
      const u = bfsq[qh++];
      for (let e = head[u]; e !== -1; e = nextE[e]) {
        if (capE[e] > 0 && parentE[toE[e]] === -1) {
          parentE[toE[e]] = e;
          parentN[toE[e]] = u;
          if (toE[e] === SNK) { found = true; break; }
          bfsq[qt++] = toE[e];
        }
      }
    }
    if (parentE[SNK] === -1) break;
    let bn = INF;
    for (let v = SNK; v !== SRC; v = parentN[v]) {
      const e = parentE[v];
      if (capE[e] < bn) bn = capE[e];
    }
    for (let v = SNK; v !== SRC; v = parentN[v]) {
      const e = parentE[v];
      capE[e] -= bn;
      capE[e ^ 1] += bn;
    }
    flow += bn;
    if (flow > budget) return null;
  }
  if (flow > budget) return null;
  const reach = new Uint8Array(N);
  reach[SRC] = 1;
  let qh = 0, qt = 0;
  bfsq[qt++] = SRC;
  while (qh < qt) {
    const u = bfsq[qh++];
    for (let e = head[u]; e !== -1; e = nextE[e]) {
      if (capE[e] > 0 && !reach[toE[e]]) {
        reach[toE[e]] = 1;
        bfsq[qt++] = toE[e];
      }
    }
  }
  const cuts = [];
  for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
      if (!isForest(x, y)) continue;
      const k = y * W + x;
      if (cutSet.has(k)) continue;
      if (ignSet.has(k) || k === townK) continue;
      if (reach[2 * k] && !reach[2 * k + 1]) cuts.push([x, y]);
    }
  }
  return cuts;
}

const cutSet = new Set();
let remaining = cutBudget;

const sortedTowns = towns.slice().sort((a, b) => b.value - a.value);

for (const t of sortedTowns) {
  if (remaining <= 0) break;
  if (!isForest(t.x, t.y)) continue;
  const reach = reachableFrom(cutSet);
  if (!reach[t.y * W + t.x]) continue;
  const cuts = minCutForTown(t.x, t.y, cutSet, remaining);
  if (cuts && cuts.length <= remaining) {
    for (const [x, y] of cuts) cutSet.add(y * W + x);
    remaining -= cuts.length;
  }
}

const firebreaks = [];
for (const k of cutSet) firebreaks.push([k % W, Math.floor(k / W)]);
process.stdout.write(JSON.stringify({ firebreaks }));
