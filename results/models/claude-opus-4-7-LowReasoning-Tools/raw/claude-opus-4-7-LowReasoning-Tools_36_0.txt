#include <bits/stdc++.h>
using namespace std;

static int N;
static vector<vector<uint64_t>> Adj; // complement adjacency for clique
static int W;

static inline bool bsGet(const vector<uint64_t>& b, int i){ return (b[i>>6]>>(i&63))&1ULL; }

static int bestSize;
static vector<int> bestSet;
static vector<int> curSet;

static void colorSort(const vector<int>& P, vector<int>& order, vector<int>& color){
    int n = (int)P.size();
    order.resize(n); color.resize(n);
    vector<vector<int>> cls;
    for(int v : P){
        bool placed=false;
        for(size_t k=0;k<cls.size();k++){
            bool ok=true;
            for(int u : cls[k]) if(bsGet(Adj[v], u)){ ok=false; break; }
            if(ok){ cls[k].push_back(v); placed=true; break; }
        }
        if(!placed) cls.push_back({v});
    }
    int idx=0;
    for(size_t k=0;k<cls.size();k++){
        for(int v : cls[k]){ order[idx]=v; color[idx]=(int)k+1; idx++; }
    }
}

static void expand(vector<int> P){
    vector<int> order, color;
    colorSort(P, order, color);
    for(int i=(int)order.size()-1; i>=0; i--){
        if((int)curSet.size() + color[i] <= bestSize) return;
        int v = order[i];
        curSet.push_back(v);
        vector<int> P2;
        P2.reserve(i);
        for(int j=0;j<i;j++){
            int u = order[j];
            if(bsGet(Adj[v], u)) P2.push_back(u);
        }
        if(P2.empty()){
            if((int)curSet.size() > bestSize){
                bestSize = (int)curSet.size();
                bestSet = curSet;
            }
        } else {
            expand(P2);
        }
        curSet.pop_back();
    }
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    if(!(cin>>n>>m)) return 0;
    N = n;
    W = (n+63)>>6;
    if(n==0){ cout<<0<<"\n\n"; return 0; }
    vector<vector<uint64_t>> G(n, vector<uint64_t>(W, 0));
    for(int i=0;i<m;i++){
        int u,v; cin>>u>>v;
        if(u==v) continue;
        if(u<0||v<0||u>=n||v>=n) continue;
        G[u][v>>6] |= (1ULL<<(v&63));
        G[v][u>>6] |= (1ULL<<(u&63));
    }

    Adj.assign(n, vector<uint64_t>(W, 0));
    int lastBit = n & 63;
    uint64_t topMask = lastBit ? ((1ULL<<lastBit)-1ULL) : ~0ULL;
    for(int i=0;i<n;i++){
        for(int w=0;w<W;w++) Adj[i][w] = ~G[i][w];
        Adj[i][i>>6] &= ~(1ULL<<(i&63));
        Adj[i][W-1] &= topMask;
    }

    vector<int> deg(n);
    for(int i=0;i<n;i++){
        int d=0; for(auto x:Adj[i]) d+=__builtin_popcountll(x); deg[i]=d;
    }
    vector<int> P(n);
    iota(P.begin(), P.end(), 0);
    sort(P.begin(), P.end(), [&](int a, int b){ return deg[a] < deg[b]; });

    bestSize = 0;
    expand(P);

    cout << bestSet.size() << "\n";
    for(size_t i=0;i<bestSet.size();i++){
        if(i) cout<<' ';
        cout<<bestSet[i];
    }
    cout<<"\n";
    return 0;
}
