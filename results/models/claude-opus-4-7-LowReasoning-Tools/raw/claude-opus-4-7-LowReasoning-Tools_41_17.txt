; SPIR-V
; Version: 1.0
               OpCapability Shader
        %ext = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
     %fnvoid = OpTypeFunction %void
      %float = OpTypeFloat 32
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
        %m4f = OpTypeMatrix %v4f 4

        %UBO = OpTypeStruct %m4f %m4f %m4f %v4f %v4f %v4f
    %ptr_ubo = OpTypePointer Uniform %UBO
        %ubo = OpVariable %ptr_ubo Uniform

  %ptr_in_v3 = OpTypePointer Input %v3f
  %ptr_in_v2 = OpTypePointer Input %v2f
 %ptr_out_v4 = OpTypePointer Output %v4f
   %ptr_u_v4 = OpTypePointer Uniform %v4f

   %worldPos = OpVariable %ptr_in_v3 Input
     %normal = OpVariable %ptr_in_v3 Input
         %uv = OpVariable %ptr_in_v2 Input
    %tangent = OpVariable %ptr_in_v3 Input
      %color = OpVariable %ptr_in_v3 Input
   %outColor = OpVariable %ptr_out_v4 Output

        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
       %bool = OpTypeBool

       %f_0  = OpConstant %float 0.0
       %f_1  = OpConstant %float 1.0
      %f_05  = OpConstant %float 0.5
       %f_8  = OpConstant %float 8.0
      %f_16  = OpConstant %float 16.0
     %f_005  = OpConstant %float 0.05
      %f_01  = OpConstant %float 0.1
      %f_07  = OpConstant %float 0.7
      %f_02  = OpConstant %float 0.2
      %f_b   = OpConstant %float 0.1
      %f_04  = OpConstant %float 0.4
      %f_06  = OpConstant %float 0.6
     %f_095  = OpConstant %float 0.95
      %f_09  = OpConstant %float 0.9

     %mortar = OpConstantComposite %v3f %f_07 %f_07 %f_07
     %brickC = OpConstantComposite %v3f %f_07 %f_02 %f_b

       %main = OpFunction %void None %fnvoid
      %entry = OpLabel

        %uvv = OpLoad %v2f %uv
          %u = OpCompositeExtract %float %uvv 0
          %v = OpCompositeExtract %float %uvv 1

         %us = OpFMul %float %u %f_8
         %vs = OpFMul %float %v %f_16

         %vf = OpExtInst %float %ext Floor %vs
        %vfi = OpConvertFToS %int %vf
      %vmod2 = OpSMod %int %vfi %int_2
      %isOdd = OpINotEqual %bool %vmod2 %int_0

       %uoff = OpSelect %float %isOdd %f_05 %f_0
        %ush = OpFAdd %float %us %uoff

      %ufrac = OpExtInst %float %ext Fract %ush
      %vfrac = OpExtInst %float %ext Fract %vs

         %t1 = OpFOrdLessThan %bool %ufrac %f_005
         %t2 = OpFOrdGreaterThan %bool %ufrac %f_095
         %t3 = OpFOrdLessThan %bool %vfrac %f_01
         %t4 = OpFOrdGreaterThan %bool %vfrac %f_09
        %t12 = OpLogicalOr %bool %t1 %t2
        %t34 = OpLogicalOr %bool %t3 %t4
   %isMortar = OpLogicalOr %bool %t12 %t34

    %baseCol = OpSelect %v3f %isMortar %mortar %brickC

     %lp_ptr = OpAccessChain %ptr_u_v4 %ubo %int_3
        %lp4 = OpLoad %v4f %lp_ptr
        %lp3 = OpVectorShuffle %v3f %lp4 %lp4 0 1 2

         %wp = OpLoad %v3f %worldPos
       %ldir = OpFSub %v3f %lp3 %wp
      %lnorm = OpExtInst %v3f %ext Normalize %ldir

        %nrm = OpLoad %v3f %normal
         %nn = OpExtInst %v3f %ext Normalize %nrm

       %dotv = OpDot %float %nn %lnorm
       %diff = OpExtInst %float %ext FMax %dotv %f_0

    %dscaled = OpFMul %float %diff %f_06
       %fact = OpFAdd %float %f_04 %dscaled

      %factv = OpCompositeConstruct %v3f %fact %fact %fact
     %shaded = OpFMul %v3f %baseCol %factv

          %r = OpCompositeExtract %float %shaded 0
          %g = OpCompositeExtract %float %shaded 1
          %b = OpCompositeExtract %float %shaded 2
       %outv = OpCompositeConstruct %v4f %r %g %b %f_1
               OpStore %outColor %outv
               OpReturn
               OpFunctionEnd
