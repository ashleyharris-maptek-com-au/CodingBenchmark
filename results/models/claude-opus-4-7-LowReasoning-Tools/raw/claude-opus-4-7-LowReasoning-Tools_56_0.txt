const fs = require('fs');
const data = JSON.parse(fs.readFileSync(0, 'utf8'));

const depot = data.depot;
const dx = depot[0], dy = depot[1];
const K = data.truckCount;
const pkgs = data.packages;
const n = pkgs.length;

function dist(ax, ay, bx, by) {
  const a = ax - bx, b = ay - by;
  return Math.sqrt(a * a + b * b);
}

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const d0 = new Float64Array(n);
for (let i = 0; i < n; i++) d0[i] = dist(dx, dy, pkgs[i].x, pkgs[i].y);

const routeOf = new Int32Array(n);
const routes = [];
for (let i = 0; i < n; i++) {
  routes.push([i]);
  routeOf[i] = i;
}

const K_NEIGH = Math.min(n - 1, 30);
const savings = [];

if (n <= 1500) {
  for (let i = 0; i < n; i++) {
    const pi = pkgs[i];
    for (let j = i + 1; j < n; j++) {
      const pj = pkgs[j];
      const dij = dist(pi.x, pi.y, pj.x, pj.y);
      const s = d0[i] + d0[j] - dij;
      if (s > 0) savings.push([s, i, j]);
    }
  }
} else {
  for (let i = 0; i < n; i++) {
    const pi = pkgs[i];
    const cand = [];
    for (let j = 0; j < n; j++) {
      if (i === j) continue;
      const pj = pkgs[j];
      cand.push([dist(pi.x, pi.y, pj.x, pj.y), j]);
    }
    cand.sort((a, b) => a[0] - b[0]);
    for (let k = 0; k < K_NEIGH && k < cand.length; k++) {
      const j = cand[k][1];
      if (j > i) {
        const s = d0[i] + d0[j] - cand[k][0];
        if (s > 0) savings.push([s, i, j]);
      }
    }
  }
}

savings.sort((a, b) => b[0] - a[0]);

function isEndpoint(route, idx) {
  return route[0] === idx || route[route.length - 1] === idx;
}

for (const sav of savings) {
  const i = sav[1], j = sav[2];
  const ri = routeOf[i], rj = routeOf[j];
  if (ri === rj) continue;
  const Ri = routes[ri], Rj = routes[rj];
  if (!Ri || !Rj) continue;
  if (!isEndpoint(Ri, i) || !isEndpoint(Rj, j)) continue;
  let A = Ri[0] === i ? Ri.slice().reverse() : Ri.slice();
  let B = Rj[Rj.length - 1] === j ? Rj.slice().reverse() : Rj.slice();
  const merged = A.concat(B);
  routes[ri] = merged;
  for (const idx of merged) routeOf[idx] = ri;
  routes[rj] = null;
}

let finalRoutes = routes.filter(r => r && r.length > 0);

while (finalRoutes.length > K) {
  let bestInc = Infinity, bestA = -1, bestB = -1, bestOpt = 0;
  for (let a = 0; a < finalRoutes.length; a++) {
    const Ra = finalRoutes[a];
    const aS = pkgs[Ra[0]], aE = pkgs[Ra[Ra.length - 1]];
    const aSi = Ra[0], aEi = Ra[Ra.length - 1];
    for (let b = a + 1; b < finalRoutes.length; b++) {
      const Rb = finalRoutes[b];
      const bS = pkgs[Rb[0]], bE = pkgs[Rb[Rb.length - 1]];
      const bSi = Rb[0], bEi = Rb[Rb.length - 1];
      const c1 = dist(aE.x, aE.y, bS.x, bS.y) - d0[aEi] - d0[bSi];
      const c2 = dist(aE.x, aE.y, bE.x, bE.y) - d0[aEi] - d0[bEi];
      const c3 = dist(aS.x, aS.y, bS.x, bS.y) - d0[aSi] - d0[bSi];
      const c4 = dist(aS.x, aS.y, bE.x, bE.y) - d0[aSi] - d0[bEi];
      const opts = [c1, c2, c3, c4];
      for (let k = 0; k < 4; k++) {
        if (opts[k] < bestInc) { bestInc = opts[k]; bestA = a; bestB = b; bestOpt = k; }
      }
    }
  }
  if (bestA < 0) break;
  const Ra = finalRoutes[bestA], Rb = finalRoutes[bestB];
  let merged;
  if (bestOpt === 0) merged = Ra.concat(Rb);
  else if (bestOpt === 1) merged = Ra.concat(Rb.slice().reverse());
  else if (bestOpt === 2) merged = Ra.slice().reverse().concat(Rb);
  else merged = Ra.slice().reverse().concat(Rb.slice().reverse());
  finalRoutes[bestA] = merged;
  finalRoutes.splice(bestB, 1);
}

function twoOpt(r) {
  if (r.length < 4) return r;
  for (let iter = 0; iter < 3; iter++) {
    let improved = false;
    const m = r.length;
    for (let i = 0; i < m - 1; i++) {
      for (let j = i + 1; j < m; j++) {
        const aPrevX = i === 0 ? dx : pkgs[r[i - 1]].x;
        const aPrevY = i === 0 ? dy : pkgs[r[i - 1]].y;
        const aCur = pkgs[r[i]];
        const bCur = pkgs[r[j]];
        const bNextX = j === m - 1 ? dx : pkgs[r[j + 1]].x;
        const bNextY = j === m - 1 ? dy : pkgs[r[j + 1]].y;
        const before = dist(aPrevX, aPrevY, aCur.x, aCur.y) + dist(bCur.x, bCur.y, bNextX, bNextY);
        const after = dist(aPrevX, aPrevY, bCur.x, bCur.y) + dist(aCur.x, aCur.y, bNextX, bNextY);
        if (after + 1e-12 < before) {
          let lo = i, hi = j;
          while (lo < hi) { const t = r[lo]; r[lo] = r[hi]; r[hi] = t; lo++; hi--; }
          improved = true;
        }
      }
    }
    if (!improved) break;
  }
  return r;
}

if (n <= 3000) {
  for (let i = 0; i < finalRoutes.length; i++) finalRoutes[i] = twoOpt(finalRoutes[i]);
}

while (finalRoutes.length < K) finalRoutes.push([]);

const out = finalRoutes.map(r => r.map(idx => pkgs[idx].id));
process.stdout.write(JSON.stringify({ routes: out }));
