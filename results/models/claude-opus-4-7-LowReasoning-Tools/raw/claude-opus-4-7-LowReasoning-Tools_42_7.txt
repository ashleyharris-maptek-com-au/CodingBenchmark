; SPIR-V
; Version: 1.0
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
     %uint_3 = OpConstant %uint 3
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
 %float_0_08 = OpConstant %float 0.08
    %float_0 = OpConstant %float 0

%gl_PerVertex_in = OpTypeStruct %v4float
   %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

   %arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
    %inColor = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

      %p0ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0ptr
      %p1ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1ptr
      %p2ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2ptr

       %p0x3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
       %p1x3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
       %p2x3 = OpVectorShuffle %v3float %p2 %p2 0 1 2

         %e1 = OpFSub %v3float %p1x3 %p0x3
         %e2 = OpFSub %v3float %p2x3 %p0x3
      %cross = OpExtInst %v3float %1 Cross %e1 %e2
       %norm = OpExtInst %v3float %1 Normalize %cross

       %ofs3 = OpVectorTimesScalar %v3float %norm %float_0_08
       %ofs4 = OpCompositeConstruct %v4float %ofs3 %float_0

        %np0 = OpFAdd %v4float %p0 %ofs4
        %np1 = OpFAdd %v4float %p1 %ofs4
        %np2 = OpFAdd %v4float %p2 %ofs4

        %c0p = OpAccessChain %ptr_in_v3 %inColor %int_0
         %c0 = OpLoad %v3float %c0p
        %c1p = OpAccessChain %ptr_in_v3 %inColor %int_1
         %c1 = OpLoad %v3float %c1p
        %c2p = OpAccessChain %ptr_in_v3 %inColor %int_2
         %c2 = OpLoad %v3float %c2p

        %op0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %op0 %np0
               OpStore %outColor %c0
               OpEmitVertex

        %op1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %op1 %np1
               OpStore %outColor %c1
               OpEmitVertex

        %op2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %op2 %np2
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
