using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

class MinesweeperSolver
{
    static int H, W;
    static byte[,] grid;
    static sbyte[,] num;
    const byte UNKNOWN = 0;
    const byte MINE = 1;
    const byte NUMBER = 2;

    static void Main()
    {
        using var sr = new StreamReader(Console.OpenStandardInput(), Encoding.ASCII, false, 1 << 20);
        string firstLine = sr.ReadLine();
        if (firstLine == null) { Console.WriteLine(0); return; }
        var parts = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        grid = new byte[H, W];
        num = new sbyte[H, W];
        for (int r = 0; r < H; r++)
        {
            string line = sr.ReadLine() ?? "";
            if (line.Length < W) line = line.PadRight(W, ' ');
            for (int c = 0; c < W; c++)
            {
                char ch = line[c];
                if (ch == ' ') { grid[r, c] = UNKNOWN; num[r, c] = -1; }
                else if (ch == '*') { grid[r, c] = MINE; num[r, c] = -1; }
                else if (ch >= '0' && ch <= '8') { grid[r, c] = NUMBER; num[r, c] = (sbyte)(ch - '0'); }
                else { grid[r, c] = UNKNOWN; num[r, c] = -1; }
            }
        }

        bool[,] safe = new bool[H, W];
        int safeCount = 0;

        var queue = new Queue<int>();
        bool[,] inQueue = new bool[H, W];

        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (grid[r, c] == NUMBER)
                {
                    queue.Enqueue(r * W + c);
                    inQueue[r, c] = true;
                }

        while (queue.Count > 0)
        {
            int idx = queue.Dequeue();
            int r = idx / W, c = idx % W;
            inQueue[r, c] = false;
            if (grid[r, c] != NUMBER) continue;
            int n = num[r, c];
            int mines = 0, unknowns = 0;
            int r0 = r > 0 ? r - 1 : r;
            int r1 = r < H - 1 ? r + 1 : r;
            int c0 = c > 0 ? c - 1 : c;
            int c1 = c < W - 1 ? c + 1 : c;
            for (int rr = r0; rr <= r1; rr++)
                for (int cc = c0; cc <= c1; cc++)
                {
                    if (rr == r && cc == c) continue;
                    byte g = grid[rr, cc];
                    if (g == MINE) mines++;
                    else if (g == UNKNOWN && !safe[rr, cc]) unknowns++;
                }
            if (unknowns == 0) continue;
            if (mines == n)
            {
                for (int rr = r0; rr <= r1; rr++)
                    for (int cc = c0; cc <= c1; cc++)
                    {
                        if (rr == r && cc == c) continue;
                        if (grid[rr, cc] == UNKNOWN && !safe[rr, cc])
                        {
                            safe[rr, cc] = true;
                            safeCount++;
                            int nr0 = rr > 0 ? rr - 1 : rr;
                            int nr1 = rr < H - 1 ? rr + 1 : rr;
                            int nc0 = cc > 0 ? cc - 1 : cc;
                            int nc1 = cc < W - 1 ? cc + 1 : cc;
                            for (int rrr = nr0; rrr <= nr1; rrr++)
                                for (int ccc = nc0; ccc <= nc1; ccc++)
                                    if (grid[rrr, ccc] == NUMBER && !inQueue[rrr, ccc])
                                    { queue.Enqueue(rrr * W + ccc); inQueue[rrr, ccc] = true; }
                        }
                    }
            }
            else if (n - mines == unknowns)
            {
                for (int rr = r0; rr <= r1; rr++)
                    for (int cc = c0; cc <= c1; cc++)
                    {
                        if (rr == r && cc == c) continue;
                        if (grid[rr, cc] == UNKNOWN && !safe[rr, cc])
                        {
                            grid[rr, cc] = MINE;
                            int nr0 = rr > 0 ? rr - 1 : rr;
                            int nr1 = rr < H - 1 ? rr + 1 : rr;
                            int nc0 = cc > 0 ? cc - 1 : cc;
                            int nc1 = cc < W - 1 ? cc + 1 : cc;
                            for (int rrr = nr0; rrr <= nr1; rrr++)
                                for (int ccc = nc0; ccc <= nc1; ccc++)
                                    if (grid[rrr, ccc] == NUMBER && !inQueue[rrr, ccc])
                                    { queue.Enqueue(rrr * W + ccc); inQueue[rrr, ccc] = true; }
                        }
                    }
            }
        }

        // Subset-pair deductions
        var active = new List<int>();
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
            {
                if (grid[r, c] != NUMBER) continue;
                int r0 = r > 0 ? r - 1 : r;
                int r1 = r < H - 1 ? r + 1 : r;
                int c0 = c > 0 ? c - 1 : c;
                int c1 = c < W - 1 ? c + 1 : c;
                bool any = false;
                for (int rr = r0; rr <= r1 && !any; rr++)
                    for (int cc = c0; cc <= c1 && !any; cc++)
                    {
                        if (rr == r && cc == c) continue;
                        if (grid[rr, cc] == UNKNOWN && !safe[rr, cc]) any = true;
                    }
                if (any) active.Add(r * W + c);
            }

        int A = active.Count;
        var unkList = new (int r, int c)[A][];
        var remMines = new int[A];
        var keyByPos = new Dictionary<int, int>(A);

        void RebuildConstraints()
        {
            for (int i = 0; i < A; i++)
            {
                int idx = active[i];
                int r = idx / W, c = idx % W;
                int n = num[r, c];
                int mines = 0;
                var tmp = new List<(int, int)>(8);
                int rr0 = r > 0 ? r - 1 : r;
                int rr1 = r < H - 1 ? r + 1 : r;
                int cc0 = c > 0 ? c - 1 : c;
                int cc1 = c < W - 1 ? c + 1 : c;
                for (int rr = rr0; rr <= rr1; rr++)
                    for (int cc = cc0; cc <= cc1; cc++)
                    {
                        if (rr == r && cc == c) continue;
                        byte g = grid[rr, cc];
                        if (g == MINE) mines++;
                        else if (g == UNKNOWN && !safe[rr, cc]) tmp.Add((rr, cc));
                    }
                unkList[i] = tmp.ToArray();
                remMines[i] = n - mines;
            }
        }

        for (int i = 0; i < A; i++) keyByPos[active[i]] = i;
        RebuildConstraints();

        bool subsetProgressed = true;
        int subsetIters = 0;
        while (subsetProgressed && subsetIters < 50)
        {
            subsetProgressed = false;
            subsetIters++;
            for (int i = 0; i < A; i++)
            {
                if (unkList[i] == null || unkList[i].Length == 0) continue;
                int idx = active[i];
                int r = idx / W, c = idx % W;
                int r0 = Math.Max(0, r - 2), r1 = Math.Min(H - 1, r + 2);
                int c0 = Math.Max(0, c - 2), c1 = Math.Min(W - 1, c + 2);
                for (int rr = r0; rr <= r1; rr++)
                    for (int cc = c0; cc <= c1; cc++)
                    {
                        if (rr == r && cc == c) continue;
                        if (grid[rr, cc] != NUMBER) continue;
                        if (!keyByPos.TryGetValue(rr * W + cc, out int j)) continue;
                        if (unkList[j] == null || unkList[j].Length == 0) continue;
                        var Ua = unkList[i];
                        var Ub = unkList[j];
                        if (Ua.Length >= Ub.Length) continue;
                        bool subset = true;
                        for (int k = 0; k < Ua.Length && subset; k++)
                        {
                            bool found = false;
                            for (int m = 0; m < Ub.Length; m++)
                                if (Ub[m].r == Ua[k].r && Ub[m].c == Ua[k].c) { found = true; break; }
                            if (!found) subset = false;
                        }
                        if (!subset) continue;
                        int diff = remMines[j] - remMines[i];
                        int sizeDiff = Ub.Length - Ua.Length;
                        if (diff == 0)
                        {
                            for (int m = 0; m < Ub.Length; m++)
                            {
                                bool inA = false;
                                for (int k = 0; k < Ua.Length; k++)
                                    if (Ua[k].r == Ub[m].r && Ua[k].c == Ub[m].c) { inA = true; break; }
                                if (!inA)
                                {
                                    int br = Ub[m].r, bc = Ub[m].c;
                                    if (!safe[br, bc] && grid[br, bc] == UNKNOWN)
                                    { safe[br, bc] = true; safeCount++; subsetProgressed = true; }
                                }
                            }
                        }
                        else if (diff == sizeDiff)
                        {
                            for (int m = 0; m < Ub.Length; m++)
                            {
                                bool inA = false;
                                for (int k = 0; k < Ua.Length; k++)
                                    if (Ua[k].r == Ub[m].r && Ua[k].c == Ub[m].c) { inA = true; break; }
                                if (!inA)
                                {
                                    int br = Ub[m].r, bc = Ub[m].c;
                                    if (grid[br, bc] == UNKNOWN && !safe[br, bc])
                                    { grid[br, bc] = MINE; subsetProgressed = true; }
                                }
                            }
                        }
                    }
            }
            if (subsetProgressed)
            {
                RebuildConstraints();
                bool tp = true;
                while (tp)
                {
                    tp = false;
                    for (int i = 0; i < A; i++)
                    {
                        if (unkList[i] == null || unkList[i].Length == 0) continue;
                        if (remMines[i] == 0)
                        {
                            foreach (var (rr, cc) in unkList[i])
                                if (!safe[rr, cc]) { safe[rr, cc] = true; safeCount++; tp = true; }
                            unkList[i] = Array.Empty<(int, int)>();
                        }
                        else if (remMines[i] == unkList[i].Length)
                        {
                            foreach (var (rr, cc) in unkList[i])
                                if (grid[rr, cc] == UNKNOWN && !safe[rr, cc]) { grid[rr, cc] = MINE; tp = true; }
                            unkList[i] = Array.Empty<(int, int)>();
                        }
                    }
                    if (tp) RebuildConstraints();
                }
            }
        }

        // CSP enumeration on small components
        var consVars = new List<List<int>>();
        var consMines = new List<int>();
        var frontierIdOf = new Dictionary<int, int>();
        var frontierCells = new List<int>();

        for (int i = 0; i < A; i++)
        {
            if (unkList[i] == null || unkList[i].Length == 0) continue;
            var varsIdx = new List<int>(unkList[i].Length);
            foreach (var (rr, cc) in unkList[i])
            {
                int cid = rr * W + cc;
                if (!frontierIdOf.TryGetValue(cid, out int fid))
                {
                    fid = frontierCells.Count;
                    frontierCells.Add(cid);
                    frontierIdOf[cid] = fid;
                }
                varsIdx.Add(fid);
            }
            consVars.Add(varsIdx);
            consMines.Add(remMines[i]);
        }

        int F = frontierCells.Count;
        int Cn = consVars.Count;

        if (F > 0 && Cn > 0)
        {
            int[] uf = new int[F];
            for (int i = 0; i < F; i++) uf[i] = i;
            int Find(int x) { while (uf[x] != x) { uf[x] = uf[uf[x]]; x = uf[x]; } return x; }
            void Union(int a, int b) { a = Find(a); b = Find(b); if (a != b) uf[a] = b; }
            foreach (var vs in consVars)
                for (int k = 1; k < vs.Count; k++) Union(vs[0], vs[k]);

            var compVars = new Dictionary<int, List<int>>();
            for (int i = 0; i < F; i++)
            {
                int rt = Find(i);
                if (!compVars.TryGetValue(rt, out var lst)) { lst = new List<int>(); compVars[rt] = lst; }
                lst.Add(i);
            }
            var compCons = new Dictionary<int, List<int>>();
            for (int ci = 0; ci < Cn; ci++)
            {
                int rt = Find(consVars[ci][0]);
                if (!compCons.TryGetValue(rt, out var lst)) { lst = new List<int>(); compCons[rt] = lst; }
                lst.Add(ci);
            }

            const int MAX_ENUM_VARS = 22;

            foreach (var kv in compVars)
            {
                var vars = kv.Value;
                if (vars.Count == 0 || vars.Count > MAX_ENUM_VARS) continue;
                var cons = compCons.ContainsKey(kv.Key) ? compCons[kv.Key] : new List<int>();
                var local = new Dictionary<int, int>(vars.Count);
                for (int k = 0; k < vars.Count; k++) local[vars[k]] = k;
                int Cm = cons.Count;
                int[][] localCons = new int[Cm][];
                int[] localMines = new int[Cm];
                for (int k = 0; k < Cm; k++)
                {
                    var src = consVars[cons[k]];
                    var arr = new int[src.Count];
                    for (int t = 0; t < src.Count; t++) arr[t] = local[src[t]];
                    localCons[k] = arr;
                    localMines[k] = consMines[cons[k]];
                }
                int V = vars.Count;
                long[] mineCount = new long[V];
                long totalValid = 0;

                int[] order = Enumerable.Range(0, V).ToArray();
                int[] posInOrder = new int[V];
                for (int k = 0; k < V; k++) posInOrder[order[k]] = k;
                int[] consCount = new int[Cm];
                for (int k = 0; k < Cm; k++) consCount[k] = localCons[k].Length;
                List<int>[] consByVarOrder = new List<int>[V];
                for (int k = 0; k < V; k++) consByVarOrder[k] = new List<int>();
                for (int k = 0; k < Cm; k++)
                    foreach (var v in localCons[k]) consByVarOrder[posInOrder[v]].Add(k);

                int[] consMinesSoFar = new int[Cm];
                int[] consVarsAssigned = new int[Cm];
                int[] assignByOrder = new int[V];

                long MAX_ENUM = 5_000_000L;
                long enumCount = 0;
                bool overflow = false;

                void Recurse(int depth)
                {
                    if (overflow) return;
                    if (depth == V)
                    {
                        for (int k = 0; k < Cm; k++)
                            if (consMinesSoFar[k] != localMines[k]) return;
                        totalValid++;
                        for (int k = 0; k < V; k++) if (assignByOrder[k] == 1) mineCount[order[k]]++;
                        enumCount++;
                        if (enumCount > MAX_ENUM) overflow = true;
                        return;
                    }
                    for (int val = 0; val <= 1; val++)
                    {
                        assignByOrder[depth] = val;
                        var clist = consByVarOrder[depth];
                        foreach (var k in clist) { consVarsAssigned[k]++; if (val == 1) consMinesSoFar[k]++; }
                        bool ok = true;
                        foreach (var k in clist)
                        {
                            int remaining = consCount[k] - consVarsAssigned[k];
                            if (consMinesSoFar[k] > localMines[k]) { ok = false; break; }
                            if (consMinesSoFar[k] + remaining < localMines[k]) { ok = false; break; }
                        }
                        if (ok) Recurse(depth + 1);
                        foreach (var k in clist) { consVarsAssigned[k]--; if (val == 1) consMinesSoFar[k]--; }
                        if (overflow) return;
                    }
                }

                Recurse(0);
                if (overflow || totalValid == 0) continue;

                for (int v = 0; v < V; v++)
                {
                    if (mineCount[v] == 0)
                    {
                        int cid = frontierCells[vars[v]];
                        int r = cid / W, c = cid % W;
                        if (grid[r, c] == UNKNOWN && !safe[r, c])
                        { safe[r, c] = true; safeCount++; }
                    }
                }
            }
        }

        var sb = new StringBuilder();
        sb.Append(safeCount).Append('\n');
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (safe[r, c]) sb.Append(c).Append(' ').Append(r).Append('\n');

        var stdout = Console.OpenStandardOutput();
        var bytes = Encoding.ASCII.GetBytes(sb.ToString());
        stdout.Write(bytes, 0, bytes.Length);
        stdout.Flush();
    }
}
