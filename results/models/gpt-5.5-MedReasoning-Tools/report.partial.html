static inline double dist2(const Point &a, const Point &b) {
    const double dx = a.x - b.x;
    const double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline double dist(const Point &a, const Point &b) {
    return std::sqrt(dist2(a, b));
}

static inline uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t morton(uint32_t x, uint32_t y) {
    return part1by1(x) | (part1by1(y) << 1);

std::vector<uint32_t> exactSmallTour(const std::vector<Point> &p) {
    const int n = static_cast<int>(p.size());
    if (n <= 1) return std::vector<uint32_t>{0};
    const int m = n - 1;
    const int states = 1 << m;
    const double INF = std::numeric_limits<double>::infinity();
    std::vector<double> dp(static_cast<size_t>(states) * m, INF);
    std::vector<int> parent(static_cast<size_t>(states) * m, -1);

    auto at = [m](int mask, int j) -> size_t {
        return static_cast<size_t>(mask) * m + j;
    };

    for (int j = 0; j < m; ++j) {
        dp[at(1 << j, j)] = dist(p[0], p[j + 1]);
    }

    for (int mask = 1; mask < states; ++mask) {
        for (int j = 0; j < m; ++j) {
            if (!(mask & (1 << j))) continue;
            const double cur = dp[at(mask, j)];
            if (!std::isfinite(cur)) continue;
            const int nextMaskBase = mask;
            for (int k = 0; k < m; ++k) {
                if (nextMaskBase & (1 << k)) continue;
                const int nm = nextMaskBase | (1 << k);
                const double cand = cur + dist(p[j + 1], p[k + 1]);
                const size_t idx = at(nm, k);
                if (cand < dp[idx]) {
                    dp[idx] = cand;
                    parent[idx] = j;
                }
            }
        }
    }

    const int full = states - 1;
    int bestEnd = 0;
    double best = INF;
    for (int j = 0; j < m; ++j) {
        const double cand = dp[at(full, j)] + dist(p[j + 1], p[0]);
        if (cand < best) {
            best = cand;
            bestEnd = j;
        }
    }

    std::vector<uint32_t> rev;
    int mask = full;
    int cur = bestEnd;
    while (cur != -1) {
        rev.push_back(static_cast<uint32_t>(cur + 1));
        const int prev = parent[at(mask, cur)];
        mask ^= (1 << cur);
        cur = prev;
    }
    std::reverse(rev.begin(), rev.end());

    std::vector<uint32_t> order;
    order.reserve(n);
    order.push_back(0);
    order.insert(order.end(), rev.begin(), rev.end());
    return order;
}

std::vector<uint32_t> mortonTour(const std::vector<Point> &p, double minX, double maxX, double minY, double maxY) {
    const size_t n = p.size();
    std::vector<Node> nodes(n);
    const double sx = (maxX > minX) ? (4294967295.0 / (maxX - minX)) : 0.0;
    const double sy = (maxY > minY) ? (4294967295.0 / (maxY - minY)) : 0.0;

    for (size_t i = 0; i < n; ++i) {
        double nx = (p[i].x - minX) * sx;
        double ny = (p[i].y - minY) * sy;
        if (!(nx >= 0.0)) nx = 0.0;
        if (!(ny >= 0.0)) ny = 0.0;
        if (nx > 4294967295.0) nx = 4294967295.0;
        if (ny > 4294967295.0) ny = 4294967295.0;
        nodes[i].key = morton(static_cast<uint32_t>(nx), static_cast<uint32_t>(ny));
        nodes[i].id = static_cast<uint32_t>(i);
    }

    std::sort(nodes.begin(), nodes.end(), [&](const Node &a, const Node &b) {
        if (a.key != b.key) return a.key < b.key;
        const Point &pa = p[a.id];
        const Point &pb = p[b.id];
        if (pa.x != pb.x) return pa.x < pb.x;
        if (pa.y != pb.y) return pa.y < pb.y;
        return a.id < b.id;
    });

    size_t zeroPos = 0;
    for (size_t i = 0; i < n; ++i) {
        if (nodes[i].id == 0) {
            zeroPos = i;
            break;
        }
    }

    std::vector<uint32_t> order;
    order.reserve(n);
    for (size_t k = 0; k < n; ++k) {
        order.push_back(nodes[(zeroPos + k) % n].id);
    }
    return order;
}

void boundedTwoOpt(std::vector<uint32_t> &order, const std::vector<Point> &p, int window, int passes) {
    const size_t n = order.size();

    for (int pass = 0; pass < passes; ++pass) {
        bool changed = false;
        for (size_t i = 0; i + 2 < n; ++i) {
            const uint32_t a = order[i];
            const uint32_t b = order[i + 1];
            const size_t upper = std::min(n - 1, i + static_cast<size_t>(window));

                const uint32_t d = order[(j + 1) % n];
                const double oldLen = dist(p[a], p[b]) + dist(p[c], p[d]);
                const double newLen = dist(p[a], p[c]) + dist(p[b], p[d]);

        stack.Add(0); route.Add(0);

    static (long[] dist, int[] prev) Dijkstra(int src, List<(int to, int w)>[] g)
    {
        int n = g.Length;
        var dist = new long[n]; var prev = new int[n];
        Array.Fill(dist, INF); Array.Fill(prev, -1);
        var pq = new PriorityQueue<int, long>();
        dist[src] = 0; pq.Enqueue(src, 0);
        while (pq.Count > 0)
        {
            pq.TryDequeue(out int v, out long d);
            if (d != dist[v]) continue;
            foreach (var e in g[v])
            {
                long nd = d + e.w;
                if (nd < dist[e.to])
                {
                    dist[e.to] = nd; prev[e.to] = v; pq.Enqueue(e.to, nd);
                }
            }
        }
        return (dist, prev);
    }

    static List<int>? TrySmallExact(int n, int m, int[] eu, int[] ev, int[] ew, List<int> odd)
    {
        int k = odd.Count;
        if (k == 0)
        {
            var mg0 = new MultiGraph(n, m);
            for (int i = 0; i < m; i++) mg0.AddEdge(eu[i], ev[i]);
            return mg0.EulerCircuit(0);
        }

        var g = new List<(int to, int w)>[n];
        for (int i = 0; i < n; i++) g[i] = new List<(int to, int w)>();
        for (int i = 0; i < m; i++) { g[eu[i]].Add((ev[i], ew[i])); g[ev[i]].Add((eu[i], ew[i])); }
        long[][] d = new long[k][]; int[][] prev = new int[k][];
        for (int i = 0; i < k; i++) { var r = Dijkstra(odd[i], g); d[i] = r.dist; prev[i] = r.prev; }

        Array.Fill(memo, -1); Array.Fill(choice, -1);
        long Solve(int mask)
        {
            if (mask == 0) return 0;
            if (memo[mask] >= 0) return memo[mask];
            int i = 0; while (((mask >> i) & 1) == 0) i++;
            long best = INF; int bj = -1;
            int withoutI = mask ^ (1 << i);
            for (int j = i + 1; j < k; j++) if (((mask >> j) & 1) != 0)
            {
                long dij = d[i][odd[j]];
                if (dij >= INF) continue;
                long cand = dij + Solve(withoutI ^ (1 << j));
                if (cand < best) { best = cand; bj = j; }
            }
            choice[mask] = bj;
            return memo[mask] = best;
        }
        if (Solve(full) >= INF) return null;
        var mg = new MultiGraph(n, m + n * k);
        for (int i = 0; i < m; i++) mg.AddEdge(eu[i], ev[i]);
        int mask2 = full;

                mg.AddEdge(p, cur); cur = p;
            }
            mask2 ^= 1 << i; mask2 ^= 1 << j;
        }
        return mg.EulerCircuit(0);
    }
