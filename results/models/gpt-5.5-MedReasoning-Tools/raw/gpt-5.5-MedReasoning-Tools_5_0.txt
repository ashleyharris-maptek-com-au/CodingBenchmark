use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct Solver {
    n: usize,
    start: usize,
    require_cycle: bool,
    xy: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
    visited: Vec<bool>,
    path: Vec<usize>,
    deadline: Instant,
    timed_out: bool,
}

impl Solver {
    fn residual_degree(&self, v: usize) -> usize {
        self.adj[v].iter().filter(|&&u| !self.visited[u]).count()
    }

    fn connected_remainder(&self, cur: usize) -> bool {
        if self.path.len() == self.n {
            return true;
        }
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        seen[cur] = true;
        q.push_back(cur);
        let mut count = 0usize;
        while let Some(v) = q.pop_front() {
            if !self.visited[v] {
                count += 1;
            }
            for &u in &self.adj[v] {
                if !seen[u] && (!self.visited[u] || u == cur) {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        count == self.n - self.path.len()
    }

    fn cheap_prune(&self, cur: usize) -> bool {
        if !self.connected_remainder(cur) {
            return true;
        }
        let left = self.n - self.path.len();
        if left == 0 {
            return false;
        }

        let mut dead = 0usize;
        for v in 0..self.n {
            if self.visited[v] {
                continue;
            }
            let mut d = self.residual_degree(v);
            if self.adj[v].contains(&cur) {
                d += 1;
            }
            if d == 0 {
                return true;
            }
            if d == 1 {
                dead += 1;
            }
        }
        dead > 1
    }

    fn dfs(&mut self, cur: usize) -> bool {
        if Instant::now() >= self.deadline {
            self.timed_out = true;
            return false;
        }
        if self.path.len() == self.n {
            return !self.require_cycle || self.adj[cur].contains(&self.start);
        }
        if self.cheap_prune(cur) {
            return false;
        }

        let mut next: Vec<usize> = self.adj[cur]
            .iter()
            .copied()
            .filter(|&u| !self.visited[u])
            .collect();
        next.sort_by_key(|&u| {
            let forced = self.adj[u]
                .iter()
                .filter(|&&v| !self.visited[v] && v != cur)
                .count();
            (forced, self.residual_degree(u))
        });

        for u in next {
            self.visited[u] = true;
            self.path.push(u);
            if self.dfs(u) {
                return true;
            }
            self.path.pop();
            self.visited[u] = false;
            if self.timed_out {
                return false;
            }
        }
        false
    }
}

fn id(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn print_path(path: &[(usize, usize)]) {
    println!("{}", path.len());
    for &(x, y) in path {
        println!("{} {}", x, y);
    }
}

fn rectangular_path(w: usize, h: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    if w == 0 || h == 0 {
        return None;
    }
    if !require_cycle {
        let mut p = Vec::with_capacity(w * h);
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    p.push((x, y));
                }
            } else {
                for x in (0..w).rev() {
                    p.push((x, y));
                }
            }
        }
        return Some(p);
    }

    if w == 1 || h == 1 || (w % 2 == 1 && h % 2 == 1) {
        return None;
    }

    if w % 2 == 0 {
        let mut p = Vec::with_capacity(w * h);
        for x in 0..w {
            if x % 2 == 0 {
                let first = if x == 0 { 0 } else { 1 };
                for y in first..h {
                    p.push((x, y));
                }
            } else {
                for y in (1..h).rev() {
                    p.push((x, y));
                }
            }
        }
        for x in (1..w).rev() {
            p.push((x, 0));
        }
        Some(p)
    } else {
        let q = rectangular_path(h, w, true)?;
        Some(q.into_iter().map(|(y, x)| (x, y)).collect())
    }
}

fn main() {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();
    let w: usize = match it.next().and_then(|v| v.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle = it.next().unwrap().parse::<u32>().unwrap() != 0;

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    let mut blocked = vec![false; w * h];
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < w && y < h {
            blocked[id(x, y, w)] = true;
        }
    }

    if obstacle_count == 0 {
        if let Some(p) = rectangular_path(w, h, require_cycle) {
            print_path(&p);
        } else {
            println!("0");
        }
        return;
    }

    if blocked[0] {
        println!("0");
        return;
    }

    let mut map = vec![usize::MAX; w * h];
    let mut xy = Vec::new();
    let mut black = 0usize;
    let mut white = 0usize;
    for y in 0..h {
        for x in 0..w {
            let old = id(x, y, w);
            if !blocked[old] {
                map[old] = xy.len();
                xy.push((x, y));
                if (x + y) % 2 == 0 {
                    black += 1;
                } else {
                    white += 1;
                }
            }
        }
    }

    let n = xy.len();
    if n == 0 {
        println!("0");
        return;
    }
    if n == 1 {
        if require_cycle {
            println!("0");
        } else {
            println!("1\n0 0");
        }
        return;
    }

    if require_cycle {
        if black != white {
            println!("0");
            return;
        }
    } else if black.abs_diff(white) > 1 {
        println!("0");
        return;
    }

    let mut adj = vec![Vec::new(); n];
    for (v, &(x, y)) in xy.iter().enumerate() {
        let dirs = [(1isize, 0isize), (-1, 0), (0, 1), (0, -1)];
        for (dx, dy) in dirs {
            let nx = x as isize + dx;
            let ny = y as isize + dy;
            if nx >= 0 && ny >= 0 && (nx as usize) < w && (ny as usize) < h {
                let old = id(nx as usize, ny as usize, w);
                if !blocked[old] {
                    adj[v].push(map[old]);
                }
            }
        }
    }

    if adj[0].is_empty() {
        println!("0");
        return;
    }

    let mut seen = vec![false; n];
    let mut q = VecDeque::new();
    seen[0] = true;
    q.push_back(0usize);
    let mut reached = 0usize;
    while let Some(v) = q.pop_front() {
        reached += 1;
        for &u in &adj[v] {
            if !seen[u] {
                seen[u] = true;
                q.push_back(u);
            }
        }
    }
    if reached != n {
        println!("0");
        return;
    }

    let leaves: Vec<usize> = (0..n).filter(|&v| adj[v].len() <= 1).collect();
    if require_cycle && !leaves.is_empty() {
        println!("0");
        return;
    }
    if !require_cycle && leaves.iter().any(|&v| v == 0) && n > 1 && leaves.len() > 1 {
        println!("0");
        return;
    }
    if !require_cycle && leaves.len() > 2 {
        println!("0");
        return;
    }

    let budget = if n <= 80 {
        Duration::from_secs(285)
    } else if n <= 400 {
        Duration::from_secs(120)
    } else {
        Duration::from_secs(30)
    };
    let mut solver = Solver {
        n,
        start: 0,
        require_cycle,
        xy,
        adj,
        visited: vec![false; n],
        path: vec![0],
        deadline: Instant::now() + budget,
        timed_out: false,
    };
    solver.visited[0] = true;

    if solver.dfs(0) {
        let out: Vec<(usize, usize)> = solver.path.iter().map(|&v| solver.xy[v]).collect();
        print_path(&out);
    } else {
        println!("0");
    }
}
