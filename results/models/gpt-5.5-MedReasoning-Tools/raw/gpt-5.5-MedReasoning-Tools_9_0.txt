using System;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _len, _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c >= 0 && c != '-' && (c < '0' || c > '9'));

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c >= '0' && c <= '9')
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int stockLength = fs.NextInt();

        var items = new long[n];
        for (int i = 0; i < n; i++)
        {
            int length = fs.NextInt();
            // Sorting this single key array is faster and lighter than sorting
            // pairs: high bits sort by descending length, low bits keep index.
            items[i] = ((long)(int.MaxValue - length) << 32) | (uint)i;
        }

        Array.Sort(items);

        int baseSize = 1;
        while (baseSize < Math.Max(1, n)) baseSize <<= 1;

        var segMax = new int[baseSize << 1];
        var remaining = new int[n];
        var head = new int[n];
        var next = new int[n];
        Array.Fill(head, -1);

        int stockCount = 0;

        for (int p = 0; p < n; p++)
        {
            long key = items[p];
            int cutIndex = (int)key;
            int cutLength = int.MaxValue - (int)(key >> 32);

            int stock = FindFirstFit(segMax, baseSize, cutLength);
            if (stock < 0)
            {
                stock = stockCount++;
                remaining[stock] = stockLength;
            }

            remaining[stock] -= cutLength;
            next[cutIndex] = head[stock];
            head[stock] = cutIndex;
            Update(segMax, baseSize, stock, remaining[stock]);
        }

        using var output = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        output.WriteLine(stockCount);

        for (int s = 0; s < stockCount; s++)
        {
            bool first = true;
            for (int idx = head[s]; idx != -1; idx = next[idx])
            {
                if (!first) output.Write(' ');
                output.Write(idx);
                first = false;
            }
            output.WriteLine();
        }
    }

    private static int FindFirstFit(int[] segMax, int baseSize, int need)
    {
        if (segMax[1] < need) return -1;

        int node = 1;
        while (node < baseSize)
        {
            int left = node << 1;
            node = segMax[left] >= need ? left : left + 1;
        }
        return node - baseSize;
    }

    private static void Update(int[] segMax, int baseSize, int index, int value)
    {
        int node = baseSize + index;
        segMax[node] = value;
        node >>= 1;

        while (node > 0)
        {
            int left = segMax[node << 1];
            int right = segMax[(node << 1) + 1];
            int max = left >= right ? left : right;
            if (segMax[node] == max) break;
            segMax[node] = max;
            node >>= 1;
        }
    }
}
