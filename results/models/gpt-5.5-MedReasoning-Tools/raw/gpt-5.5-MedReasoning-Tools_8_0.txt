use std::fs::{remove_file, File, OpenOptions};
use std::io::{self, BufRead, BufReader, BufWriter, Read, Seek, SeekFrom, Write};

const UNVISITED: u8 = 0xff;

fn bit_is_set(bits: &[u64], i: usize) -> bool {
    ((bits[i >> 6] >> (i & 63)) & 1) != 0
}

fn set_bit(bits: &mut [u64], i: usize) {
    bits[i >> 6] |= 1u64 << (i & 63);
}

fn pack(parent_dir: u8, next_dir: u8) -> u8 {
    parent_dir | (next_dir << 2)
}

fn parent_dir(v: u8) -> u8 {
    v & 3
}

fn next_dir(v: u8) -> u8 {
    (v >> 2) & 7
}

fn with_next(v: u8, n: u8) -> u8 {
    (v & 3) | (n << 2)
}

fn parent_index(i: usize, dir: u8, w: usize) -> usize {
    match dir {
        0 => i - w,
        1 => i + 1,
        2 => i + w,
        3 => i - 1,
        _ => unreachable!(),
    }
}

fn emit_path(mut goal: usize, start: usize, meta: &[u8], w: usize) -> io::Result<()> {
    let mut count: u64 = 1;
    let mut cur = goal;
    while cur != start {
        cur = parent_index(cur, parent_dir(meta[cur]), w);
        count += 1;
    }

    let mut out = BufWriter::new(io::stdout().lock());
    writeln!(out, "{count}")?;

    let tmp_path = std::env::temp_dir().join(format!(
        "rust_maze_path_{}_{}.bin",
        std::process::id(),
        count
    ));

    {
        let mut tmp = BufWriter::new(File::create(&tmp_path)?);
        loop {
            tmp.write_all(&(goal as u64).to_le_bytes())?;
            if goal == start {
                break;
            }
            goal = parent_index(goal, parent_dir(meta[goal]), w);
        }
        tmp.flush()?;
    }

    let mut tmp = OpenOptions::new().read(true).open(&tmp_path)?;
    let mut buf = [0u8; 8];
    for rec in (0..count).rev() {
        tmp.seek(SeekFrom::Start(rec * 8))?;
        tmp.read_exact(&mut buf)?;
        let idx = u64::from_le_bytes(buf) as usize;
        writeln!(out, "{} {}", idx % w, idx / w)?;
    }
    out.flush()?;

    let _ = remove_file(tmp_path);
    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut input = BufReader::new(stdin.lock());

    let mut header = String::new();
    input.read_line(&mut header)?;
    let mut parts = header.split_whitespace();
    let h: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return Ok(()),
    };
    let w: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return Ok(()),
    };

    let total = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return Ok(());
        }
    };

    let mut walls = vec![0u64; (total + 63) >> 6];
    let mut start = None;
    let mut goal = None;
    let mut line = Vec::with_capacity(w + 2);

    for y in 0..h {
        line.clear();
        input.read_until(b'\n', &mut line)?;
        while matches!(line.last(), Some(b'\n' | b'\r')) {
            line.pop();
        }

        for x in 0..w {
            let ch = line.get(x).copied().unwrap_or(b'#');
            let idx = y * w + x;
            match ch {
                b'#' => set_bit(&mut walls, idx),
                b'A' => start = Some(idx),
                b'B' => goal = Some(idx),
                _ => {}
            }
        }
    }

    let (start, goal) = match (start, goal) {
        (Some(a), Some(b)) => (a, b),
        _ => {
            println!("0");
            return Ok(());
        }
    };

    let mut meta = vec![UNVISITED; total];
    meta[start] = pack(0, 0);
    let mut cur = start;

    loop {
        if cur == goal {
            return emit_path(goal, start, &meta, w);
        }

        let nd = next_dir(meta[cur]);
        if nd >= 4 {
            if cur == start {
                println!("0");
                return Ok(());
            }
            cur = parent_index(cur, parent_dir(meta[cur]), w);
            continue;
        }

        meta[cur] = with_next(meta[cur], nd + 1);

        let mut next = None;
        let mut back = 0u8;
        match nd {
            0 if cur >= w => {
                next = Some(cur - w);
                back = 2;
            }
            1 if cur % w + 1 < w => {
                next = Some(cur + 1);
                back = 3;
            }
            2 if cur + w < total => {
                next = Some(cur + w);
                back = 0;
            }
            3 if cur % w > 0 => {
                next = Some(cur - 1);
                back = 1;
            }
            _ => {}
        }

        if let Some(n) = next {
            if !bit_is_set(&walls, n) && meta[n] == UNVISITED {
                meta[n] = pack(back, 0);
                cur = n;
            }
        }
    }
}
