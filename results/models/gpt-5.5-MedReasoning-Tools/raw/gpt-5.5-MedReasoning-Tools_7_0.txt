using namespace std;

struct Vec {
    double x{}, y{}, z{};
    Vec() = default;
    Vec(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec operator+(const Vec& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec operator-(const Vec& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec operator*(double s) const { return {x * s, y * s, z * s}; }
};

struct Tri { int a{}, b{}, c{}; };
struct Mesh { vector<Vec> v; vector<Tri> f; };
struct Box { Vec mn, mx; bool valid = false; };

static constexpr double EPS = 1e-9;

static Vec crossp(const Vec& a, const Vec& b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}
static double dotp(const Vec& a, const Vec& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
static double coord(const Vec& p, int ax) { return ax == 0 ? p.x : (ax == 1 ? p.y : p.z); }
static bool eq(double a, double b) { return fabs(a - b) <= EPS * max(1.0, max(fabs(a), fabs(b))); }

static Box boundsOf(const Mesh& m) {
    Box b;
    if (m.v.empty()) return b;
    b.valid = true;
    b.mn = b.mx = m.v[0];
    for (const auto& p : m.v) {
        b.mn.x = min(b.mn.x, p.x); b.mn.y = min(b.mn.y, p.y); b.mn.z = min(b.mn.z, p.z);
        b.mx.x = max(b.mx.x, p.x); b.mx.y = max(b.mx.y, p.y); b.mx.z = max(b.mx.z, p.z);
    }
    return b;
}

static bool boxDisjoint(const Box& a, const Box& b) {
    return a.mx.x < b.mn.x - EPS || b.mx.x < a.mn.x - EPS ||
           a.mx.y < b.mn.y - EPS || b.mx.y < a.mn.y - EPS ||
           a.mx.z < b.mn.z - EPS || b.mx.z < a.mn.z - EPS;
}

static bool pointInBox(const Vec& p, const Box& b) {
    return p.x >= b.mn.x - EPS && p.x <= b.mx.x + EPS &&
           p.y >= b.mn.y - EPS && p.y <= b.mx.y + EPS &&
           p.z >= b.mn.z - EPS && p.z <= b.mx.z + EPS;
}

static bool isAxisAlignedBoxMesh(const Mesh& m, Box& b) {
    b = boundsOf(m);
    if (!b.valid || m.f.empty()) return false;
    if (eq(b.mn.x, b.mx.x) || eq(b.mn.y, b.mx.y) || eq(b.mn.z, b.mx.z)) return false;
    for (const auto& p : m.v) {
        bool onX = eq(p.x, b.mn.x) || eq(p.x, b.mx.x);
        bool onY = eq(p.y, b.mn.y) || eq(p.y, b.mx.y);
        bool onZ = eq(p.z, b.mn.z) || eq(p.z, b.mx.z);
        if (!(onX && onY && onZ)) return false;
    }
    array<int, 6> faceSeen{};
    for (const auto& t : m.f) {
        const Vec& a = m.v[t.a]; const Vec& c = m.v[t.b]; const Vec& d = m.v[t.c];
        int plane = -1;
        for (int ax = 0; ax < 3; ++ax) {
            double mn = coord(b.mn, ax), mx = coord(b.mx, ax);
            if (eq(coord(a, ax), mn) && eq(coord(c, ax), mn) && eq(coord(d, ax), mn)) plane = ax * 2;
            if (eq(coord(a, ax), mx) && eq(coord(c, ax), mx) && eq(coord(d, ax), mx)) plane = ax * 2 + 1;
        }
        if (plane < 0) return false;
        faceSeen[plane]++;
    }
    return all_of(faceSeen.begin(), faceSeen.end(), [](int n) { return n > 0; });
}

static bool pointInsideMesh(const Mesh& m, const Vec& p) {
    Vec dir(1.0, 0.3713906763541037, 0.5298129428260175);
    int hits = 0;
    for (const auto& tr : m.f) {
        Vec a = m.v[tr.a], b = m.v[tr.b], c = m.v[tr.c];
        Vec e1 = b - a, e2 = c - a;
        Vec h = crossp(dir, e2);
        double det = dotp(e1, h);
        if (fabs(det) < EPS) continue;
        double inv = 1.0 / det;
        Vec s = p - a;
        double u = inv * dotp(s, h);
        if (u < -EPS || u > 1.0 + EPS) continue;
        Vec q = crossp(s, e1);
        double v = inv * dotp(dir, q);
        if (v < -EPS || u + v > 1.0 + EPS) continue;
        double tt = inv * dotp(e2, q);
        if (tt > EPS) hits++;
    }
    return (hits & 1) != 0;
}

static void appendMesh(Mesh& out, const Mesh& m) {
    int base = static_cast<int>(out.v.size());
    out.v.insert(out.v.end(), m.v.begin(), m.v.end());
    for (auto t : m.f) out.f.push_back({t.a + base, t.b + base, t.c + base});
}

static void removeDuplicateVertices(Mesh& m) {
    map<tuple<long long,long long,long long>, int> seen;
    vector<Vec> nv;
    vector<int> remap(m.v.size());
    const double scale = 1000000000.0;
    for (size_t i = 0; i < m.v.size(); ++i) {
        auto key = make_tuple((long long)llround(m.v[i].x * scale),
                              (long long)llround(m.v[i].y * scale),
                              (long long)llround(m.v[i].z * scale));
        auto it = seen.find(key);
        if (it == seen.end()) {
            int id = static_cast<int>(nv.size());
            seen[key] = id; remap[i] = id; nv.push_back(m.v[i]);
        } else remap[i] = it->second;
    }
    for (auto& t : m.f) { t.a = remap[t.a]; t.b = remap[t.b]; t.c = remap[t.c]; }
    m.v.swap(nv);
}

static Mesh unionBoxes(const Box& A, const Box& B) {
    vector<double> xs{A.mn.x, A.mx.x, B.mn.x, B.mx.x};
    vector<double> ys{A.mn.y, A.mx.y, B.mn.y, B.mx.y};
    vector<double> zs{A.mn.z, A.mx.z, B.mn.z, B.mx.z};
    auto uniq = [](vector<double>& v) {
        sort(v.begin(), v.end());
        vector<double> r;
        for (double x : v) if (r.empty() || !eq(r.back(), x)) r.push_back(x);
        v.swap(r);
    };
    uniq(xs); uniq(ys); uniq(zs);
    int nx = (int)xs.size() - 1, ny = (int)ys.size() - 1, nz = (int)zs.size() - 1;
    vector<int> occ(max(0, nx * ny * nz), 0);
    auto id = [&](int i, int j, int k) { return (i * ny + j) * nz + k; };
    auto inOcc = [&](int i, int j, int k) {
        if (i < 0 || j < 0 || k < 0 || i >= nx || j >= ny || k >= nz) return false;
        return occ[id(i,j,k)] != 0;
    };
    for (int i = 0; i < nx; ++i) for (int j = 0; j < ny; ++j) for (int k = 0; k < nz; ++k) {
        Vec c((xs[i]+xs[i+1])*0.5, (ys[j]+ys[j+1])*0.5, (zs[k]+zs[k+1])*0.5);
        occ[id(i,j,k)] = pointInBox(c, A) || pointInBox(c, B);
    }
    Mesh out;
    map<tuple<int,int,int>, int> vid;
    auto vertex = [&](int i, int j, int k) {
        auto key = make_tuple(i,j,k);
        auto it = vid.find(key);
        if (it != vid.end()) return it->second;
        int n = (int)out.v.size();
        vid[key] = n;
        out.v.push_back({xs[i], ys[j], zs[k]});
        return n;
    };
    auto quad = [&](array<tuple<int,int,int>,4> q) {
        int a = vertex(get<0>(q[0]), get<1>(q[0]), get<2>(q[0]));
        int b = vertex(get<0>(q[1]), get<1>(q[1]), get<2>(q[1]));
        int c = vertex(get<0>(q[2]), get<1>(q[2]), get<2>(q[2]));
        int d = vertex(get<0>(q[3]), get<1>(q[3]), get<2>(q[3]));
        out.f.push_back({a,b,c});
        out.f.push_back({a,c,d});
    };
    for (int i = 0; i < nx; ++i) for (int j = 0; j < ny; ++j) for (int k = 0; k < nz; ++k) if (inOcc(i,j,k)) {
        if (!inOcc(i-1,j,k)) quad({make_tuple(i,j,k), make_tuple(i,j,k+1), make_tuple(i,j+1,k+1), make_tuple(i,j+1,k)});
        if (!inOcc(i+1,j,k)) quad({make_tuple(i+1,j,k), make_tuple(i+1,j+1,k), make_tuple(i+1,j+1,k+1), make_tuple(i+1,j,k+1)});
        if (!inOcc(i,j-1,k)) quad({make_tuple(i,j,k), make_tuple(i+1,j,k), make_tuple(i+1,j,k+1), make_tuple(i,j,k+1)});
        if (!inOcc(i,j+1,k)) quad({make_tuple(i,j+1,k), make_tuple(i,j+1,k+1), make_tuple(i+1,j+1,k+1), make_tuple(i+1,j+1,k)});
        if (!inOcc(i,j,k-1)) quad({make_tuple(i,j,k), make_tuple(i,j+1,k), make_tuple(i+1,j+1,k), make_tuple(i+1,j,k)});
        if (!inOcc(i,j,k+1)) quad({make_tuple(i,j,k+1), make_tuple(i+1,j,k+1), make_tuple(i+1,j+1,k+1), make_tuple(i,j+1,k+1)});
    }
    removeDuplicateVertices(out);
    return out;
}

static Mesh filteredUnionFallback(const Mesh& A, const Mesh& B) {
    Mesh out;
    Box ba = boundsOf(A), bb = boundsOf(B);
    auto addExterior = [&](const Mesh& src, const Mesh& other, const Box& ob) {
        int base = static_cast<int>(out.v.size());
        out.v.insert(out.v.end(), src.v.begin(), src.v.end());
        for (auto t : src.f) {
            Vec c = (src.v[t.a] + src.v[t.b] + src.v[t.c]) * (1.0 / 3.0);
            bool inside = pointInBox(c, ob) && pointInsideMesh(other, c);
            if (!inside) out.f.push_back({t.a + base, t.b + base, t.c + base});
        }
    };
    addExterior(A, B, bb);
    addExterior(B, A, ba);
    if (out.f.empty()) {
        if (!A.v.empty() && pointInsideMesh(B, A.v[0])) return B;
        return A;
    }
    removeDuplicateVertices(out);
    return out;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;
    Mesh A, B;
    A.v.resize(VA); A.f.resize(FA); B.v.resize(VB); B.f.resize(FB);
    for (auto& p : A.v) cin >> p.x >> p.y >> p.z;
    for (auto& t : A.f) cin >> t.a >> t.b >> t.c;
    for (auto& p : B.v) cin >> p.x >> p.y >> p.z;
    for (auto& t : B.f) cin >> t.a >> t.b >> t.c;

    Mesh out;
    Box ba = boundsOf(A), bb = boundsOf(B), boxA, boxB;

    if (isAxisAlignedBoxMesh(A, boxA) && isAxisAlignedBoxMesh(B, boxB)) {
        out = unionBoxes(boxA, boxB);
    } else if (boxDisjoint(ba, bb)) {
        appendMesh(out, A);
        appendMesh(out, B);
        removeDuplicateVertices(out);
    } else if (!A.v.empty() && pointInsideMesh(B, A.v[0])) {
        out = B;
    } else if (!B.v.empty() && pointInsideMesh(A, B.v[0])) {
        out = A;
    } else {
        out = filteredUnionFallback(A, B);
    }

    cout << out.v.size() << ' ' << out.f.size() << '\n';
    cout << setprecision(17);
    for (const auto& p : out.v) cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const auto& t : out.f) cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
    return 0;
}
