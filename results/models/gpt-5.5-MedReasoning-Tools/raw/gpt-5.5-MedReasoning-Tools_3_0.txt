use std::collections::VecDeque;
use std::io::{self, Read, Write};
use std::time::Instant;

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, idx: 0 }
    }
    fn usize(&mut self) -> usize {
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_digit() {
            self.idx += 1;
        }
        let mut v = 0usize;
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_digit() {
            v = v * 10 + (self.buf[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        v
    }
}

struct Dsu {
    p: Vec<usize>,
    sz: Vec<usize>,
}

impl Dsu {
    fn new(n: usize) -> Self {
        Self { p: (0..n).collect(), sz: vec![1; n] }
    }
    fn find(&mut self, x: usize) -> usize {
        if self.p[x] != x {
            let r = self.find(self.p[x]);
            self.p[x] = r;
        }
        self.p[x]
    }
    fn union(&mut self, a: usize, b: usize) {
        let mut ra = self.find(a);
        let mut rb = self.find(b);
        if ra == rb {
            return;
        }
        if self.sz[ra] < self.sz[rb] {
            std::mem::swap(&mut ra, &mut rb);
        }
        self.p[rb] = ra;
        self.sz[ra] += self.sz[rb];
    }
}

#[derive(Clone)]
struct Layout {
    nodes: Vec<usize>,
    x: Vec<f64>,
    y: Vec<f64>,
}

fn bfs_order(nodes: &[usize], adj: &[Vec<usize>], deg: &[usize], seen: &mut [u32], mark: u32) -> Vec<usize> {
    let mut seeds = nodes.to_vec();
    seeds.sort_unstable_by(|&a, &b| deg[b].cmp(&deg[a]).then_with(|| a.cmp(&b)));
    let mut order = Vec::with_capacity(nodes.len());
    let mut q = VecDeque::new();
    for s in seeds {
        if seen[s] == mark {
            continue;
        }
        seen[s] = mark;
        q.push_back(s);
        while let Some(v) = q.pop_front() {
            order.push(v);
            for &to in &adj[v] {
                if seen[to] != mark {
                    seen[to] = mark;
                    q.push_back(to);
                }
            }
        }
    }
    order
}

fn tree_layout(
    nodes: &[usize],
    adj: &[Vec<usize>],
    parent: &mut [usize],
    depth: &mut [usize],
    lx: &mut [f64],
) -> Layout {
    if nodes.len() == 1 {
        return Layout { nodes: nodes.to_vec(), x: vec![0.0], y: vec![0.0] };
    }

    for &v in nodes {
        parent[v] = usize::MAX;
        depth[v] = 0;
    }
    let root = nodes[0];
    parent[root] = root;
    let mut stack = vec![root];
    let mut order = Vec::with_capacity(nodes.len());
    while let Some(v) = stack.pop() {
        order.push(v);
        for &to in &adj[v] {
            if parent[to] == usize::MAX {
                parent[to] = v;
                depth[to] = depth[v] + 1;
                stack.push(to);
            }
        }
    }

    let mut next_leaf = 0.0;
    for &v in order.iter().rev() {
        let mut sum = 0.0;
        let mut cnt = 0.0;
        for &to in &adj[v] {
            if parent[to] == v {
                sum += lx[to];
                cnt += 1.0;
            }
        }
        if cnt == 0.0 {
            lx[v] = next_leaf;
            next_leaf += 1.0;
        } else {
            lx[v] = sum / cnt;
        }
    }

    let mut out_nodes = Vec::with_capacity(order.len());
    let mut xs = Vec::with_capacity(order.len());
    let mut ys = Vec::with_capacity(order.len());
    for &v in &order {
        out_nodes.push(v);
        xs.push(lx[v]);
        ys.push(depth[v] as f64 * 1.35);
    }
    Layout { nodes: out_nodes, x: xs, y: ys }
}

fn cycle_layout(nodes: &[usize], adj: &[Vec<usize>]) -> Layout {
    let k = nodes.len();
    if k <= 2 {
        return Layout {
            nodes: nodes.to_vec(),
            x: (0..k).map(|i| i as f64).collect(),
            y: vec![0.0; k],
        };
    }
    let start = nodes[0];
    let mut order = Vec::with_capacity(k);
    let mut prev = usize::MAX;
    let mut cur = start;
    for _ in 0..k {
        order.push(cur);
        let mut nxt = usize::MAX;
        for &to in &adj[cur] {
            if to != prev {
                nxt = to;
                break;
            }
        }
        prev = cur;
        cur = nxt;
        if cur == usize::MAX || cur == start {
            break;
        }
    }
    if order.len() != k {
        order = nodes.to_vec();
    }
    let r = (k as f64) / std::f64::consts::PI.max(1.0);
    let mut xs = Vec::with_capacity(k);
    let mut ys = Vec::with_capacity(k);
    for i in 0..k {
        let a = 2.0 * std::f64::consts::PI * (i as f64) / (k as f64);
        xs.push(r * a.cos());
        ys.push(r * a.sin());
    }
    Layout { nodes: order, x: xs, y: ys }
}

fn general_layout(
    nodes: &[usize],
    comp_edges: &[(usize, usize)],
    adj: &[Vec<usize>],
    deg: &[usize],
    seen: &mut [u32],
    index: &mut [usize],
    mark: u32,
    start_time: Instant,
) -> Layout {
    let k = nodes.len();
    if k == 1 {
        return Layout { nodes: nodes.to_vec(), x: vec![0.0], y: vec![0.0] };
    }

    let order = bfs_order(nodes, adj, deg, seen, mark);
    for (i, &v) in order.iter().enumerate() {
        index[v] = i;
    }

    let mut x = vec![0.0; k];
    let mut y = vec![0.0; k];
    if k > 1800 {
        let golden = std::f64::consts::PI * (3.0 - 5.0f64.sqrt());
        for i in 0..k {
            let r = (i as f64).sqrt() * 1.7;
            let a = i as f64 * golden;
            x[i] = r * a.cos();
            y[i] = r * a.sin();
        }
    } else {
        let r = (k as f64).sqrt().max(2.0) * 1.8;
        for i in 0..k {
            let a = 2.0 * std::f64::consts::PI * (i as f64) / (k as f64);
            x[i] = r * a.cos();
            y[i] = r * a.sin();
        }
    }

    let mut edges = Vec::new();
    edges.reserve(comp_edges.len().min(k * 16 + 1));
    for &(a, b) in comp_edges {
        let ia = index[a];
        let ib = index[b];
        if ia != usize::MAX && ib != usize::MAX && ia != ib {
            edges.push((ia, ib));
        }
    }

    let iters = if k <= 120 {
        320
    } else if k <= 500 {
        170
    } else if k <= 1200 {
        80
    } else if k <= 2500 {
        35
    } else {
        0
    };

    if iters > 0 {
        let ideal = (k as f64).sqrt().max(1.0);
        let mut dx = vec![0.0; k];
        let mut dy = vec![0.0; k];
        let edge_stride = (edges.len() / (k * 10 + 1)).max(1);

        for it in 0..iters {
            if start_time.elapsed().as_secs_f64() > 27.0 {
                break;
            }
            dx.fill(0.0);
            dy.fill(0.0);

            if k <= 750 {
                for i in 0..k {
                    for j in (i + 1)..k {
                        let vx = x[i] - x[j];
                        let vy = y[i] - y[j];
                        let d2 = (vx * vx + vy * vy).max(0.01);
                        let f = ideal * ideal / d2;
                        let inv_d = d2.sqrt().recip();
                        let fx = vx * inv_d * f;
                        let fy = vy * inv_d * f;
                        dx[i] += fx;
                        dy[i] += fy;
                        dx[j] -= fx;
                        dy[j] -= fy;
                    }
                }
            } else {
                let samples = 28usize;
                for i in 0..k {
                    let mut seed = (i as u64).wrapping_mul(6364136223846793005).wrapping_add(1);
                    for _ in 0..samples {
                        seed = seed.wrapping_mul(2862933555777941757).wrapping_add(3037000493);
                        let j = (seed as usize) % k;
                        if i == j {
                            continue;
                        }
                        let vx = x[i] - x[j];
                        let vy = y[i] - y[j];
                        let d2 = (vx * vx + vy * vy).max(0.01);
                        let f = (ideal * ideal / d2) * (k as f64 / samples as f64);
                        let inv_d = d2.sqrt().recip();
                        dx[i] += vx * inv_d * f;
                        dy[i] += vy * inv_d * f;
                    }
                }
            }

            for (ei, &(a, b)) in edges.iter().enumerate() {
                if ei % edge_stride != 0 {
                    continue;
                }
                let vx = x[a] - x[b];
                let vy = y[a] - y[b];
                let d = (vx * vx + vy * vy).sqrt().max(0.01);
                let f = (d * d / ideal) * 0.045 * edge_stride as f64;
                let fx = vx / d * f;
                let fy = vy / d * f;
                dx[a] -= fx;
                dy[a] -= fy;
                dx[b] += fx;
                dy[b] += fy;
            }

            let temp = (1.0 - it as f64 / iters as f64).max(0.03) * ideal * 0.35;
            for i in 0..k {
                let len = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt().max(0.001);
                x[i] += dx[i] / len * len.min(temp);
                y[i] += dy[i] / len * len.min(temp);
            }
        }
    }

    for &v in nodes {
        index[v] = usize::MAX;
    }
    Layout { nodes: order, x, y }
}

fn normalize(mut l: Layout) -> Layout {
    if l.nodes.is_empty() {
        return l;
    }
    let min_x = l.x.iter().copied().fold(f64::INFINITY, f64::min);
    let max_x = l.x.iter().copied().fold(f64::NEG_INFINITY, f64::max);
    let min_y = l.y.iter().copied().fold(f64::INFINITY, f64::min);
    let max_y = l.y.iter().copied().fold(f64::NEG_INFINITY, f64::max);
    let cx = (min_x + max_x) * 0.5;
    let cy = (min_y + max_y) * 0.5;
    let span = (max_x - min_x).max(max_y - min_y).max(1.0);
    let scale = (l.nodes.len() as f64).sqrt().max(1.0) * 2.0 / span;
    for i in 0..l.nodes.len() {
        l.x[i] = (l.x[i] - cx) * scale;
        l.y[i] = (l.y[i] - cy) * scale;
    }
    l
}

fn main() {
    let start_time = Instant::now();
    let mut sc = Scanner::new();
    let n = sc.usize();
    let m = sc.usize();

    let mut dsu = Dsu::new(n);
    let mut raw_edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u = sc.usize();
        let v = sc.usize();
        if u < n && v < n {
            raw_edges.push((u, v));
            if u != v {
                dsu.union(u, v);
            }
        }
    }

    let mut adj = vec![Vec::<usize>::new(); n];
    let mut edges = Vec::new();
    for &(u, v) in &raw_edges {
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
            edges.push((u, v));
        }
    }
    let deg: Vec<usize> = adj.iter().map(Vec::len).collect();

    let mut root_to_id = vec![usize::MAX; n];
    let mut comp_id = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    for v in 0..n {
        let r = dsu.find(v);
        if root_to_id[r] == usize::MAX {
            root_to_id[r] = comps.len();
            comps.push(Vec::new());
        }
        let id = root_to_id[r];
        comp_id[v] = id;
        comps[id].push(v);
    }

    let mut comp_edges: Vec<Vec<(usize, usize)>> = vec![Vec::new(); comps.len()];
    for &(u, v) in &edges {
        comp_edges[comp_id[u]].push((u, v));
    }

    let mut order_ids: Vec<usize> = (0..comps.len()).collect();
    order_ids.sort_unstable_by(|&a, &b| comps[b].len().cmp(&comps[a].len()));

    let mut ans_x = vec![0.0; n];
    let mut ans_y = vec![0.0; n];
    let mut parent = vec![usize::MAX; n];
    let mut depth = vec![0usize; n];
    let mut scratch_x = vec![0.0; n];
    let mut seen = vec![0u32; n];
    let mut index = vec![usize::MAX; n];
    let mut layouts = Vec::with_capacity(comps.len());

    for (mark0, &cid) in order_ids.iter().enumerate() {
        let nodes = &comps[cid];
        let ecount = comp_edges[cid].len();
        let is_tree = nodes.len() > 1 && ecount == nodes.len() - 1;
        let is_cycle = nodes.len() > 2 && ecount == nodes.len() && nodes.iter().all(|&v| deg[v] == 2);
        let layout = if nodes.len() == 1 {
            Layout { nodes: nodes.to_vec(), x: vec![0.0], y: vec![0.0] }
        } else if is_tree {
            tree_layout(nodes, &adj, &mut parent, &mut depth, &mut scratch_x)
        } else if is_cycle {
            cycle_layout(nodes, &adj)
        } else {
            general_layout(
                nodes,
                &comp_edges[cid],
                &adj,
                &deg,
                &mut seen,
                &mut index,
                (mark0 as u32).wrapping_add(1),
                start_time,
            )
        };
        layouts.push(normalize(layout));
    }

    let total_area: f64 = layouts.iter().map(|l| l.nodes.len() as f64 + 16.0).sum();
    let target_row = total_area.sqrt() * 5.0;
    let mut cursor_x = 0.0;
    let mut cursor_y = 0.0;
    let mut row_h = 0.0;

    for l in layouts {
        let min_x = l.x.iter().copied().fold(f64::INFINITY, f64::min);
        let max_x = l.x.iter().copied().fold(f64::NEG_INFINITY, f64::max);
        let min_y = l.y.iter().copied().fold(f64::INFINITY, f64::min);
        let max_y = l.y.iter().copied().fold(f64::NEG_INFINITY, f64::max);
        let w = (max_x - min_x).max(1.0);
        let h = (max_y - min_y).max(1.0);
        let margin = 4.0;
        if cursor_x > 0.0 && cursor_x + w + margin > target_row {
            cursor_x = 0.0;
            cursor_y += row_h + margin;
            row_h = 0.0;
        }
        for i in 0..l.nodes.len() {
            let v = l.nodes[i];
            ans_x[v] = cursor_x + l.x[i] - min_x;
            ans_y[v] = cursor_y + l.y[i] - min_y;
        }
        cursor_x += w + margin;
        row_h = row_h.max(h);
    }

    let mut out = String::with_capacity(n * 24);
    for i in 0..n {
        if !ans_x[i].is_finite() || !ans_y[i].is_finite() {
            ans_x[i] = i as f64;
            ans_y[i] = 0.0;
        }
        out.push_str(&format!("{:.6} {:.6}\n", ans_x[i], ans_y[i]));
    }
    io::stdout().write_all(out.as_bytes()).unwrap();
}
