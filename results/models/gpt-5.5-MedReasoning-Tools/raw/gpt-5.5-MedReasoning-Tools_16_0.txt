#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <string>
#include <vector>

using namespace std;

struct Task {
    int machine;
    long long duration;
};

struct Job {
    vector<Task> tasks;
    vector<long long> suffixWork;
};

struct Schedule {
    vector<vector<long long>> start;
    long long makespan = numeric_limits<long long>::max();
};

enum class Rule {
    FIFO,
    EarliestReady,
    ShortestNext,
    LongestNext,
    MostRemaining,
    LeastRemaining,
    MostTasksRemaining,
    LeastTasksRemaining,
    RandomBiased
};

struct Solver {
    int J = 0, M = 0;
    long long totalTasks = 0;
    vector<Job> jobs;

    static uint64_t mix64(uint64_t x) {
        x += 0x9e3779b97f4a7c15ULL;
        x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
        x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
        return x ^ (x >> 31);
    }

    long long keyForJob(int j, int next, long long ready, Rule rule, uint64_t seed) const {
        const Job& job = jobs[j];
        const Task& task = job.tasks[next];
        const int tasksLeft = static_cast<int>(job.tasks.size()) - next;
        const long long rem = job.suffixWork[next];

        switch (rule) {
            case Rule::FIFO:
                return static_cast<long long>(j);
            case Rule::EarliestReady:
                return ready * 1000003LL + j;
            case Rule::ShortestNext:
                return task.duration * 1000003LL + ready;
            case Rule::LongestNext:
                return -task.duration * 1000003LL + ready;
            case Rule::MostRemaining:
                return -rem * 1000003LL + ready;
            case Rule::LeastRemaining:
                return rem * 1000003LL + ready;
            case Rule::MostTasksRemaining:
                return -static_cast<long long>(tasksLeft) * 1000003LL + ready;
            case Rule::LeastTasksRemaining:
                return static_cast<long long>(tasksLeft) * 1000003LL + ready;
            case Rule::RandomBiased: {
                uint64_t h = mix64(seed ^ (static_cast<uint64_t>(j) << 32) ^ static_cast<uint64_t>(next));
                long long noise = static_cast<long long>(h & 0xfffffULL);
                return -rem * 1000003LL + noise;
            }
        }
        return 0;
    }

    Schedule buildWithRule(Rule rule, uint64_t seed = 0) const {
        struct Node {
            long long key;
            int job;
            int version;
            bool operator<(const Node& other) const {
                if (key != other.key) return key > other.key; // min-heap behavior
                return job > other.job;
            }
        };

        vector<int> next(J, 0), version(J, 0);
        vector<long long> jobReady(J, 0), machineReady(M, 0);
        Schedule schedule;
        schedule.start.resize(J);
        priority_queue<Node> pq;

        for (int j = 0; j < J; ++j) {
            schedule.start[j].assign(jobs[j].tasks.size(), 0);
            if (!jobs[j].tasks.empty()) {
                pq.push({keyForJob(j, 0, 0, rule, seed), j, 0});
            }
        }

        long long placed = 0;
        long long makespan = 0;
        while (placed < totalTasks && !pq.empty()) {
            Node node = pq.top();
            pq.pop();

            int j = node.job;
            if (node.version != version[j] || next[j] >= static_cast<int>(jobs[j].tasks.size())) {
                continue;
            }

            int t = next[j];
            const Task& task = jobs[j].tasks[t];
            long long s = max(jobReady[j], machineReady[task.machine]);
            long long e = s + task.duration;

            schedule.start[j][t] = s;
            jobReady[j] = e;
            machineReady[task.machine] = e;
            makespan = max(makespan, e);
            ++next[j];
            ++version[j];
            ++placed;

            if (next[j] < static_cast<int>(jobs[j].tasks.size())) {
                pq.push({keyForJob(j, next[j], jobReady[j], rule, seed), j, version[j]});
            }
        }

        schedule.makespan = makespan;
        return schedule;
    }

    Schedule solve() const {
        vector<Rule> rules;

        if (totalTasks <= 20) {
            rules = {
                Rule::FIFO, Rule::EarliestReady, Rule::ShortestNext, Rule::LongestNext,
                Rule::MostRemaining, Rule::LeastRemaining, Rule::MostTasksRemaining,
                Rule::LeastTasksRemaining
            };
        } else if (totalTasks <= 2000) {
            rules = {
                Rule::MostRemaining, Rule::ShortestNext, Rule::LongestNext, Rule::EarliestReady,
                Rule::LeastRemaining, Rule::MostTasksRemaining, Rule::LeastTasksRemaining, Rule::FIFO
            };
        } else if (totalTasks <= 50000) {
            rules = {
                Rule::MostRemaining, Rule::ShortestNext, Rule::LongestNext,
                Rule::EarliestReady, Rule::MostTasksRemaining
            };
        } else {
            rules = {Rule::MostRemaining, Rule::ShortestNext};
        }

        Schedule best;
        for (Rule rule : rules) {
            Schedule candidate = buildWithRule(rule);
            if (candidate.makespan < best.makespan) {
                best = move(candidate);
            }
        }

        // Small and medium instances can afford extra randomized dispatching passes.
        int randomPasses = 0;
        if (totalTasks <= 200) randomPasses = 120;
        else if (totalTasks <= 2000) randomPasses = 30;
        else if (totalTasks <= 10000) randomPasses = 6;

        uint64_t seed = 0x123456789abcdefULL ^ static_cast<uint64_t>(J) << 17 ^ static_cast<uint64_t>(M);
        for (int i = 0; i < randomPasses; ++i) {
            Schedule candidate = buildWithRule(Rule::RandomBiased, seed + static_cast<uint64_t>(i) * 1000003ULL);
            if (candidate.makespan < best.makespan) {
                best = move(candidate);
            }
        }

        return best;
    }

    bool readInput() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);

        if (!(cin >> J >> M)) return false;
        jobs.resize(J);
        totalTasks = 0;

        for (int j = 0; j < J; ++j) {
            int T;
            cin >> T;
            jobs[j].tasks.resize(T);
            totalTasks += T;
            for (int t = 0; t < T; ++t) {
                cin >> jobs[j].tasks[t].machine >> jobs[j].tasks[t].duration;
            }

            jobs[j].suffixWork.assign(T + 1, 0);
            for (int t = T - 1; t >= 0; --t) {
                jobs[j].suffixWork[t] = jobs[j].suffixWork[t + 1] + jobs[j].tasks[t].duration;
            }
        }
        return true;
    }

    void printSchedule(const Schedule& schedule) const {
        for (int j = 0; j < J; ++j) {
            for (int t = 0; t < static_cast<int>(jobs[j].tasks.size()); ++t) {
                if (t) cout << ' ';
                cout << schedule.start[j][t] << ' ' << jobs[j].tasks[t].machine;
            }
            cout << '\n';
        }
    }
};

int main() {
    Solver solver;
    if (!solver.readInput()) return 0;

    Schedule schedule = solver.solve();
    solver.printSchedule(schedule);
    return 0;
}
