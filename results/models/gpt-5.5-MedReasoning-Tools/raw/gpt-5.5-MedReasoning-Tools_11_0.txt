use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-7;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
}

impl BBox {
    fn width(&self) -> f64 {
        self.maxx - self.minx
    }
    fn height(&self) -> f64 {
        self.maxy - self.miny
    }
    fn area(&self) -> f64 {
        self.width().max(0.0) * self.height().max(0.0)
    }
    fn overlaps(&self, other: &BBox) -> bool {
        self.minx < other.maxx - EPS
            && self.maxx > other.minx + EPS
            && self.miny < other.maxy - EPS
            && self.maxy > other.miny + EPS
    }
}

#[derive(Clone)]
struct Piece {
    idx: usize,
    poly: Vec<Pt>,
    bbox_area: f64,
}

#[derive(Clone)]
struct Placed {
    poly: Vec<Pt>,
    bbox: BBox,
}

struct StockState {
    placed: Vec<Placed>,
    candidates: Vec<Pt>,
}

#[derive(Clone, Copy)]
struct Placement {
    stock: usize,
    x: f64,
    y: f64,
    rot: f64,
}

struct Scanner {
    data: Vec<String>,
    pos: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            data: input.split_whitespace().map(|s| s.to_string()).collect(),
            pos: 0,
        }
    }
    fn next_usize(&mut self) -> usize {
        let v = self.data[self.pos].parse::<usize>().unwrap();
        self.pos += 1;
        v
    }
    fn next_f64(&mut self) -> f64 {
        let v = self.data[self.pos].parse::<f64>().unwrap();
        self.pos += 1;
        v
    }
}

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    cross(a, b, p).abs() <= EPS
        && p.x >= a.x.min(b.x) - EPS
        && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS
        && p.y <= a.y.max(b.y) + EPS
}

fn proper_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let c1 = cross(a, b, c);
    let c2 = cross(a, b, d);
    let c3 = cross(c, d, a);
    let c4 = cross(c, d, b);
    c1 * c2 < -EPS && c3 * c4 < -EPS
}

fn point_on_boundary(poly: &[Pt], p: Pt) -> bool {
    let n = poly.len();
    for i in 0..n {
        if on_segment(poly[i], poly[(i + 1) % n], p) {
            return true;
        }
    }
    false
}

fn point_inside_strict(poly: &[Pt], p: Pt) -> bool {
    if point_on_boundary(poly, p) {
        return false;
    }
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if (a.y > p.y) != (b.y > p.y) {
            let x = a.x + (p.y - a.y) * (b.x - a.x) / (b.y - a.y);
            if x > p.x {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_inside_or_boundary(poly: &[Pt], p: Pt) -> bool {
    point_on_boundary(poly, p) || point_inside_strict(poly, p)
}

fn bbox(poly: &[Pt]) -> BBox {
    let mut b = BBox {
        minx: f64::INFINITY,
        miny: f64::INFINITY,
        maxx: f64::NEG_INFINITY,
        maxy: f64::NEG_INFINITY,
    };
    for &p in poly {
        b.minx = b.minx.min(p.x);
        b.miny = b.miny.min(p.y);
        b.maxx = b.maxx.max(p.x);
        b.maxy = b.maxy.max(p.y);
    }
    b
}

fn rotate(p: Pt, a: f64) -> Pt {
    let (s, c) = a.sin_cos();
    Pt {
        x: p.x * c - p.y * s,
        y: p.x * s + p.y * c,
    }
}

fn transformed(poly: &[Pt], a: f64, tx: f64, ty: f64) -> Vec<Pt> {
    poly.iter()
        .map(|&p| {
            let r = rotate(p, a);
            Pt {
                x: r.x + tx,
                y: r.y + ty,
            }
        })
        .collect()
}

fn rotated_bbox(poly: &[Pt], a: f64) -> BBox {
    let pts: Vec<Pt> = poly.iter().map(|&p| rotate(p, a)).collect();
    bbox(&pts)
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        s += a.x * b.y - a.y * b.x;
    }
    0.5 * s
}

fn polygon_centroid(poly: &[Pt]) -> Pt {
    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut a_sum = 0.0;
    for i in 0..poly.len() {
        let p = poly[i];
        let q = poly[(i + 1) % poly.len()];
        let cr = p.x * q.y - q.x * p.y;
        a_sum += cr;
        cx += (p.x + q.x) * cr;
        cy += (p.y + q.y) * cr;
    }
    if a_sum.abs() < EPS {
        let b = bbox(poly);
        return Pt {
            x: (b.minx + b.maxx) * 0.5,
            y: (b.miny + b.maxy) * 0.5,
        };
    }
    Pt {
        x: cx / (3.0 * a_sum),
        y: cy / (3.0 * a_sum),
    }
}

fn inside_stock(stock: &[Pt], poly: &[Pt]) -> bool {
    for &p in poly {
        if !point_inside_or_boundary(stock, p) {
            return false;
        }
    }
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        for j in 0..stock.len() {
            if proper_intersect(a, b, stock[j], stock[(j + 1) % stock.len()]) {
                return false;
            }
        }
    }
    true
}

fn polygons_overlap(a: &[Pt], ab: &BBox, b: &[Pt], bb: &BBox) -> bool {
    if !ab.overlaps(bb) {
        return false;
    }
    for i in 0..a.len() {
        for j in 0..b.len() {
            if proper_intersect(a[i], a[(i + 1) % a.len()], b[j], b[(j + 1) % b.len()]) {
                return true;
            }
        }
    }
    for &p in a {
        if point_inside_strict(b, p) {
            return true;
        }
    }
    for &p in b {
        if point_inside_strict(a, p) {
            return true;
        }
    }
    false
}

fn clear_of_placed(poly: &[Pt], pb: &BBox, placed: &[Placed]) -> bool {
    for q in placed {
        if polygons_overlap(poly, pb, &q.poly, &q.bbox) {
            return false;
        }
    }
    true
}

fn rotations(n: usize) -> Vec<f64> {
    if n > 1000 {
        vec![0.0, PI * 0.5]
    } else if n > 100 {
        vec![0.0, PI * 0.5, PI, PI * 1.5]
    } else {
        (0..16).map(|i| i as f64 * PI / 8.0).collect()
    }
}

fn try_place_in_state(
    piece: &Piece,
    stock: &[Pt],
    stock_box: &BBox,
    state: &mut StockState,
    stock_idx: usize,
    rots: &[f64],
    large: bool,
) -> Option<Placement> {
    let mut candidates = state.candidates.clone();
    candidates.push(Pt {
        x: (stock_box.minx + stock_box.maxx) * 0.5,
        y: (stock_box.miny + stock_box.maxy) * 0.5,
    });
    candidates.sort_by(|a, b| {
        a.y.partial_cmp(&b.y)
            .unwrap_or(Ordering::Equal)
            .then(a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal))
    });
    candidates.dedup_by(|a, b| (a.x - b.x).abs() < 1e-6 && (a.y - b.y).abs() < 1e-6);
    let limit = if large { 80 } else { 700 };
    if candidates.len() > limit {
        candidates.truncate(limit);
    }

    for &r in rots {
        let rb = rotated_bbox(&piece.poly, r);
        if rb.width() > stock_box.width() + EPS || rb.height() > stock_box.height() + EPS {
            continue;
        }
        for c in &candidates {
            let target_min = if (c.x - (stock_box.minx + stock_box.maxx) * 0.5).abs() < EPS
                && (c.y - (stock_box.miny + stock_box.maxy) * 0.5).abs() < EPS
            {
                Pt {
                    x: c.x - rb.width() * 0.5,
                    y: c.y - rb.height() * 0.5,
                }
            } else {
                *c
            };
            if target_min.x + rb.width() > stock_box.maxx + EPS
                || target_min.y + rb.height() > stock_box.maxy + EPS
            {
                continue;
            }
            let tx = target_min.x - rb.minx;
            let ty = target_min.y - rb.miny;
            let poly = transformed(&piece.poly, r, tx, ty);
            let pb = bbox(&poly);
            if inside_stock(stock, &poly) && clear_of_placed(&poly, &pb, &state.placed) {
                state.placed.push(Placed { poly, bbox: pb });
                state.candidates.push(Pt {
                    x: pb.maxx,
                    y: pb.miny,
                });
                state.candidates.push(Pt {
                    x: pb.minx,
                    y: pb.maxy,
                });
                state.candidates.push(Pt {
                    x: pb.maxx,
                    y: pb.maxy,
                });
                state.candidates.retain(|p| {
                    p.x >= stock_box.minx - EPS
                        && p.x <= stock_box.maxx + EPS
                        && p.y >= stock_box.miny - EPS
                        && p.y <= stock_box.maxy + EPS
                });
                return Some(Placement {
                    stock: stock_idx,
                    x: tx,
                    y: ty,
                    rot: r,
                });
            }
        }
    }
    None
}

fn find_any_placement(piece: &Piece, stock: &[Pt], stock_box: &BBox, rots: &[f64], large: bool) -> Option<(Placement, Placed)> {
    let centroid = polygon_centroid(stock);
    let grid = if large { 4 } else { 9 };
    let mut centers = Vec::new();
    centers.push(centroid);
    centers.push(Pt {
        x: (stock_box.minx + stock_box.maxx) * 0.5,
        y: (stock_box.miny + stock_box.maxy) * 0.5,
    });
    for iy in 0..grid {
        for ix in 0..grid {
            let x = stock_box.minx + stock_box.width() * (ix as f64 + 0.5) / grid as f64;
            let y = stock_box.miny + stock_box.height() * (iy as f64 + 0.5) / grid as f64;
            centers.push(Pt { x, y });
        }
    }
    centers.retain(|&p| point_inside_or_boundary(stock, p));
    for &r in rots {
        let rb = rotated_bbox(&piece.poly, r);
        for c in &centers {
            let tx = c.x - (rb.minx + rb.maxx) * 0.5;
            let ty = c.y - (rb.miny + rb.maxy) * 0.5;
            let poly = transformed(&piece.poly, r, tx, ty);
            let pb = bbox(&poly);
            if inside_stock(stock, &poly) {
                return Some((
                    Placement {
                        stock: 0,
                        x: tx,
                        y: ty,
                        rot: r,
                    },
                    Placed { poly, bbox: pb },
                ));
            }
        }
    }
    None
}

fn main() {
    let mut sc = Scanner::new();
    if sc.data.is_empty() {
        return;
    }
    let sv = sc.next_usize();
    let np = sc.next_usize();
    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        stock.push(Pt {
            x: sc.next_f64(),
            y: sc.next_f64(),
        });
    }
    if polygon_area(&stock) < 0.0 {
        stock.reverse();
    }
    let stock_box = bbox(&stock);

    let mut pieces = Vec::with_capacity(np);
    for idx in 0..np {
        let pv = sc.next_usize();
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            poly.push(Pt {
                x: sc.next_f64(),
                y: sc.next_f64(),
            });
        }
        if polygon_area(&poly) < 0.0 {
            poly.reverse();
        }
        let b = bbox(&poly);
        pieces.push(Piece {
            idx,
            poly,
            bbox_area: b.area(),
        });
    }

    let large = np > 1000 || pieces.iter().map(|p| p.poly.len()).sum::<usize>() > 100_000;
    let rots = rotations(np);
    pieces.sort_by(|a, b| {
        b.bbox_area
            .partial_cmp(&a.bbox_area)
            .unwrap_or(Ordering::Equal)
    });

    let mut states: Vec<StockState> = Vec::new();
    let mut out = vec![
        Placement {
            stock: 0,
            x: 0.0,
            y: 0.0,
            rot: 0.0,
        };
        np
    ];

    for piece in &pieces {
        let mut found = None;
        for s in 0..states.len() {
            if let Some(p) = try_place_in_state(piece, &stock, &stock_box, &mut states[s], s, &rots, large) {
                found = Some(p);
                break;
            }
        }
        if found.is_none() {
            let new_idx = states.len();
            let mut state = StockState {
                placed: Vec::new(),
                candidates: vec![Pt {
                    x: stock_box.minx,
                    y: stock_box.miny,
                }],
            };
            if let Some(mut p) = try_place_in_state(piece, &stock, &stock_box, &mut state, new_idx, &rots, large) {
                p.stock = new_idx;
                found = Some(p);
            } else if let Some((mut p, placed)) = find_any_placement(piece, &stock, &stock_box, &rots, large) {
                p.stock = new_idx;
                state.placed.push(placed);
                found = Some(p);
            } else {
                let rb = rotated_bbox(&piece.poly, 0.0);
                let tx = (stock_box.minx + stock_box.maxx - rb.minx - rb.maxx) * 0.5;
                let ty = (stock_box.miny + stock_box.maxy - rb.miny - rb.maxy) * 0.5;
                let poly = transformed(&piece.poly, 0.0, tx, ty);
                let pb = bbox(&poly);
                state.placed.push(Placed { poly, bbox: pb });
                found = Some(Placement {
                    stock: new_idx,
                    x: tx,
                    y: ty,
                    rot: 0.0,
                });
            }
            states.push(state);
        }
        out[piece.idx] = found.unwrap();
    }

    println!("{}", states.len());
    for p in out {
        println!("{} {:.10} {:.10} {:.10}", p.stock, p.x, p.y, p.rot);
    }
}
