#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <vector>

using namespace std;

struct Item {
    int id;
    int w;
    int h;
    long long area;
};

struct Placement {
    int board = 0;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Rect {
    int x;
    int y;
    int w;
    int h;
};

static bool contains_rect(const Rect& a, const Rect& b) {
    return a.x <= b.x && a.y <= b.y &&
           a.x + a.w >= b.x + b.w &&
           a.y + a.h >= b.y + b.h;
}

struct MaxRectsBoard {
    vector<Rect> free_rects;

    MaxRectsBoard(int W, int H) {
        free_rects.push_back({0, 0, W, H});
    }

    bool find_position(const Item& it, int W, int H, Rect& out, int& rot, long long& score) const {
        (void)W;
        (void)H;
        bool found = false;
        score = numeric_limits<long long>::max();

        for (const Rect& f : free_rects) {
            for (int r = 0; r < 2; ++r) {
                int rw = r ? it.h : it.w;
                int rh = r ? it.w : it.h;
                if (rw > f.w || rh > f.h) continue;

                long long leftover = 1LL * f.w * f.h - 1LL * rw * rh;
                int short_side = min(f.w - rw, f.h - rh);
                long long cur = leftover * 1024LL + short_side;
                if (!found || cur < score) {
                    found = true;
                    score = cur;
                    out = {f.x, f.y, rw, rh};
                    rot = r;
                }
            }
        }
        return found;
    }

    void place(const Rect& used) {
        vector<Rect> next;
        next.reserve(free_rects.size() + 4);

        for (const Rect& f : free_rects) {
            bool separate = used.x >= f.x + f.w || used.x + used.w <= f.x ||
                            used.y >= f.y + f.h || used.y + used.h <= f.y;
            if (separate) {
                next.push_back(f);
                continue;
            }

            if (used.x > f.x)
                next.push_back({f.x, f.y, used.x - f.x, f.h});
            if (used.x + used.w < f.x + f.w)
                next.push_back({used.x + used.w, f.y, f.x + f.w - (used.x + used.w), f.h});
            if (used.y > f.y)
                next.push_back({f.x, f.y, f.w, used.y - f.y});
            if (used.y + used.h < f.y + f.h)
                next.push_back({f.x, used.y + used.h, f.w, f.y + f.h - (used.y + used.h)});
        }

        free_rects.clear();
        for (const Rect& r : next) {
            if (r.w > 0 && r.h > 0) free_rects.push_back(r);
        }
        prune();
    }

    void prune() {
        sort(free_rects.begin(), free_rects.end(), [](const Rect& a, const Rect& b) {
            if (a.x != b.x) return a.x < b.x;
            if (a.y != b.y) return a.y < b.y;
            if (a.w != b.w) return a.w > b.w;
            return a.h > b.h;
        });

        vector<char> removed(free_rects.size(), 0);
        for (size_t i = 0; i < free_rects.size(); ++i) {
            if (removed[i]) continue;
            for (size_t j = i + 1; j < free_rects.size(); ++j) {
                if (removed[j]) continue;
                if (contains_rect(free_rects[i], free_rects[j])) {
                    removed[j] = 1;
                } else if (contains_rect(free_rects[j], free_rects[i])) {
                    removed[i] = 1;
                    break;
                }
            }
        }

        vector<Rect> kept;
        kept.reserve(free_rects.size());
        for (size_t i = 0; i < free_rects.size(); ++i)
            if (!removed[i]) kept.push_back(free_rects[i]);
        free_rects.swap(kept);
    }
};

struct Shelf {
    int y;
    int h;
    int used;
};

struct ShelfBoard {
    vector<Shelf> shelves;
    int next_y = 0;
};

static bool can_fit_item(const Item& it, int W, int H) {
    return (it.w <= W && it.h <= H) || (it.h <= W && it.w <= H);
}

static void solve_maxrects(const vector<Item>& order, int W, int H, vector<Placement>& ans) {
    vector<MaxRectsBoard> boards;
    boards.reserve(max<size_t>(1, order.size() / 8));

    for (const Item& it : order) {
        Rect best_rect{0, 0, 0, 0};
        int best_rot = 0;
        int best_board = -1;
        long long best_score = numeric_limits<long long>::max();

        int scan_boards = (int)boards.size();
        for (int b = 0; b < scan_boards; ++b) {
            Rect candidate;
            int rot = 0;
            long long score = 0;
            if (boards[b].find_position(it, W, H, candidate, rot, score)) {
                long long board_bias = 1LL * b * 8LL;
                if (score + board_bias < best_score) {
                    best_score = score + board_bias;
                    best_rect = candidate;
                    best_rot = rot;
                    best_board = b;
                }
            }
        }

        if (best_board < 0) {
            boards.emplace_back(W, H);
            best_board = (int)boards.size() - 1;
            int rw = it.w, rh = it.h;
            best_rot = 0;
            if ((it.w > W || it.h > H) && it.h <= W && it.w <= H) {
                rw = it.h;
                rh = it.w;
                best_rot = 1;
            } else if (it.h <= W && it.w <= H) {
                int waste0 = min(W - it.w, H - it.h);
                int waste1 = min(W - it.h, H - it.w);
                if (waste1 < waste0) {
                    rw = it.h;
                    rh = it.w;
                    best_rot = 1;
                }
            }
            best_rect = {0, 0, rw, rh};
        }

        boards[best_board].place(best_rect);
        ans[it.id] = {best_board, best_rect.x, best_rect.y, best_rot};
    }
}

static void solve_shelves(const vector<Item>& order, int W, int H, vector<Placement>& ans) {
    vector<ShelfBoard> boards;
    boards.reserve(max<size_t>(1, order.size() / 20));

    for (const Item& it : order) {
        int best_b = -1, best_s = -1, best_rot = 0;
        long long best_score = numeric_limits<long long>::max();

        for (int b = 0; b < (int)boards.size(); ++b) {
            ShelfBoard& bd = boards[b];
            for (int s = 0; s < (int)bd.shelves.size(); ++s) {
                const Shelf& sh = bd.shelves[s];
                for (int r = 0; r < 2; ++r) {
                    int rw = r ? it.h : it.w;
                    int rh = r ? it.w : it.h;
                    if (rw <= W - sh.used && rh <= sh.h) {
                        long long width_left = W - (sh.used + rw);
                        long long height_slack = sh.h - rh;
                        long long score = width_left * 4096LL + height_slack * 32LL + b;
                        if (score < best_score) {
                            best_score = score;
                            best_b = b;
                            best_s = s;
                            best_rot = r;
                        }
                    }
                }
            }
        }

        if (best_b >= 0) {
            Shelf& sh = boards[best_b].shelves[best_s];
            int x = sh.used;
            int y = sh.y;
            int rw = best_rot ? it.h : it.w;
            sh.used += rw;
            ans[it.id] = {best_b, x, y, best_rot};
            continue;
        }

        int new_b = -1, new_rot = 0;
        long long best_new_score = numeric_limits<long long>::max();
        for (int b = 0; b < (int)boards.size(); ++b) {
            for (int r = 0; r < 2; ++r) {
                int rw = r ? it.h : it.w;
                int rh = r ? it.w : it.h;
                if (rw <= W && boards[b].next_y + rh <= H) {
                    long long score = 1LL * b * W * H + 1LL * rh * W + (W - rw);
                    if (score < best_new_score) {
                        best_new_score = score;
                        new_b = b;
                        new_rot = r;
                    }
                }
            }
        }

        if (new_b < 0) {
            boards.push_back(ShelfBoard{});
            new_b = (int)boards.size() - 1;
            new_rot = 0;
            if ((it.w > W || it.h > H) && it.h <= W && it.w <= H) {
                new_rot = 1;
            } else if (it.h <= W && it.w <= H) {
                int a0 = it.h * W + (W - it.w);
                int a1 = it.w * W + (W - it.h);
                if (a1 < a0) new_rot = 1;
            }
        }

        int rw = new_rot ? it.h : it.w;
        int rh = new_rot ? it.w : it.h;
        ShelfBoard& bd = boards[new_b];
        int y = bd.next_y;
        bd.shelves.push_back({y, rh, rw});
        bd.next_y += rh;
        ans[it.id] = {new_b, 0, y, new_rot};
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, W, H;
    if (!(cin >> N >> W >> H)) return 0;

    vector<Item> items;
    items.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        items.push_back({i, w, h, 1LL * w * h});
    }

    vector<Placement> ans(N);
    vector<Item> packable;
    packable.reserve(N);
    for (const Item& it : items) {
        if (can_fit_item(it, W, H)) {
            packable.push_back(it);
        } else {
            ans[it.id] = {0, 0, 0, 0};
        }
    }

    sort(packable.begin(), packable.end(), [](const Item& a, const Item& b) {
        int amax = max(a.w, a.h), bmax = max(b.w, b.h);
        int amin = min(a.w, a.h), bmin = min(b.w, b.h);
        if (amax != bmax) return amax > bmax;
        if (amin != bmin) return amin > bmin;
        if (a.area != b.area) return a.area > b.area;
        return a.id < b.id;
    });

    if (N <= 2000) {
        solve_maxrects(packable, W, H, ans);
    } else {
        solve_shelves(packable, W, H, ans);
    }

    int num_boards = 0;
    for (const Placement& p : ans) num_boards = max(num_boards, p.board + 1);

    cout << num_boards << '\n';
    for (int i = 0; i < N; ++i) {
        cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rotated << '\n';
    }

    return 0;
}
