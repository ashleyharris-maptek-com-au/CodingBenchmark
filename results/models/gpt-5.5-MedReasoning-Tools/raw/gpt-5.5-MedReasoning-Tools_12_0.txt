using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    struct Vec
    {
        public double X, Y, Z;
        public Vec(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    struct Mat
    {
        public double M00, M01, M02, M10, M11, M12, M20, M21, M22;

        public Vec Mul(Vec p)
        {
            return new Vec(
                M00 * p.X + M01 * p.Y + M02 * p.Z,
                M10 * p.X + M11 * p.Y + M12 * p.Z,
                M20 * p.X + M21 * p.Y + M22 * p.Z);
        }

        public static Mat Mul(Mat a, Mat b)
        {
            return new Mat
            {
                M00 = a.M00 * b.M00 + a.M01 * b.M10 + a.M02 * b.M20,
                M01 = a.M00 * b.M01 + a.M01 * b.M11 + a.M02 * b.M21,
                M02 = a.M00 * b.M02 + a.M01 * b.M12 + a.M02 * b.M22,
                M10 = a.M10 * b.M00 + a.M11 * b.M10 + a.M12 * b.M20,
                M11 = a.M10 * b.M01 + a.M11 * b.M11 + a.M12 * b.M21,
                M12 = a.M10 * b.M02 + a.M11 * b.M12 + a.M12 * b.M22,
                M20 = a.M20 * b.M00 + a.M21 * b.M10 + a.M22 * b.M20,
                M21 = a.M20 * b.M01 + a.M21 * b.M11 + a.M22 * b.M21,
                M22 = a.M20 * b.M02 + a.M21 * b.M12 + a.M22 * b.M22
            };
        }
    }

    struct Candidate
    {
        public Mat R;
        public double MinX, MinY, MinZ, SizeX, SizeY, SizeZ;
        public long NX, NY, NZ, Count;
        public double QW, QX, QY, QZ;
    }

    sealed class FastScanner
    {
        readonly Stream input = Console.OpenStandardInput();
        readonly byte[] buffer = new byte[1 << 16];
        int len, ptr;

        int Read()
        {
            if (ptr >= len)
            {
                len = input.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = Read();
            }
            return v * sign;
        }

        public double NextDouble()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            double v = 0;
            while (c >= '0' && c <= '9')
            {
                v = v * 10 + (c - '0');
                c = Read();
            }
            if (c == '.')
            {
                double scale = 1;
                c = Read();
                while (c >= '0' && c <= '9')
                {
                    scale *= 0.1;
                    v += (c - '0') * scale;
                    c = Read();
                }
            }
            if (c == 'e' || c == 'E')
            {
                int esign = 1;
                int exp = 0;
                c = Read();
                if (c == '-') { esign = -1; c = Read(); }
                else if (c == '+') c = Read();
                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }
                v *= Math.Pow(10, esign * exp);
            }
            return sign * v;
        }
    }

    static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        CultureInfo.CurrentUICulture = CultureInfo.InvariantCulture;

        var fs = new FastScanner();
        int vCount = fs.NextInt();
        int fCount = fs.NextInt();
        var vertices = new Vec[vCount];
        for (int i = 0; i < vCount; i++)
            vertices[i] = new Vec(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());

        for (int i = 0; i < fCount; i++)
        {
            fs.NextInt();
            fs.NextInt();
            fs.NextInt();
        }

        double W = fs.NextDouble(), H = fs.NextDouble(), D = fs.NextDouble();
        if (vCount == 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var rotations = BuildRotations(W * H * D, vCount, fCount);
        Candidate best = default;
        bool hasBest = false;
        double gap = 0.0;

        foreach (var r in rotations)
        {
            Candidate c = Evaluate(vertices, r, W, H, D, gap);
            if (!hasBest || c.Count > best.Count)
            {
                best = c;
                hasBest = true;
            }
        }

        if (!hasBest || best.Count <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        long outputCount = Math.Min(best.Count, OutputBudget(W, H, D, vCount, fCount));
        using var writer = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1 << 20);
        writer.WriteLine(outputCount.ToString(CultureInfo.InvariantCulture));

        long written = 0;
        double pitchX = best.SizeX + gap, pitchY = best.SizeY + gap, pitchZ = best.SizeZ + gap;
        string quat = F(best.QW) + " " + F(best.QX) + " " + F(best.QY) + " " + F(best.QZ);

        for (long iz = 0; iz < best.NZ && written < outputCount; iz++)
        {
            double tzBase = iz * pitchZ - best.MinZ;
            for (long iy = 0; iy < best.NY && written < outputCount; iy++)
            {
                double tyBase = iy * pitchY - best.MinY;
                for (long ix = 0; ix < best.NX && written < outputCount; ix++)
                {
                    double tx = ix * pitchX - best.MinX;
                    writer.Write(F(tx)); writer.Write(' ');
                    writer.Write(F(tyBase)); writer.Write(' ');
                    writer.Write(F(tzBase)); writer.Write(' ');
                    writer.WriteLine(quat);
                    written++;
                }
            }
        }
    }

    static Candidate Evaluate(Vec[] vertices, Mat r, double W, double H, double D, double gap)
    {
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;
        foreach (var p in vertices)
        {
            Vec q = r.Mul(p);
            if (q.X < minX) minX = q.X; if (q.X > maxX) maxX = q.X;
            if (q.Y < minY) minY = q.Y; if (q.Y > maxY) maxY = q.Y;
            if (q.Z < minZ) minZ = q.Z; if (q.Z > maxZ) maxZ = q.Z;
        }

        double sx = Math.Max(0.0, maxX - minX);
        double sy = Math.Max(0.0, maxY - minY);
        double sz = Math.Max(0.0, maxZ - minZ);
        long nx = Fits(sx, W, gap), ny = Fits(sy, H, gap), nz = Fits(sz, D, gap);
        long count = SafeMul(nx, SafeMul(ny, nz));
        (double qw, double qx, double qy, double qz) = Quaternion(r);
        return new Candidate { R = r, MinX = minX, MinY = minY, MinZ = minZ, SizeX = sx, SizeY = sy, SizeZ = sz, NX = nx, NY = ny, NZ = nz, Count = count, QW = qw, QX = qx, QY = qy, QZ = qz };
    }

    static long Fits(double size, double container, double gap)
    {
        if (size < 1e-12) size = 1e-12;
        if (size > container + 1e-9) return 0;
        return Math.Max(1, (long)Math.Floor((container - size + 1e-9) / (size + gap)) + 1);
    }

    static long SafeMul(long a, long b)
    {
        if (a <= 0 || b <= 0) return 0;
        if (a > long.MaxValue / b) return long.MaxValue;
        return a * b;
    }

    static long OutputBudget(double W, double H, double D, int v, int f)
    {
        double volume = W * H * D;
        if (volume <= 8000 && v + f <= 2000) return 5_000_000;
        if (volume <= 1_000_000 && v + f <= 10000) return 3_000_000;
        return 2_000_000;
    }

    static List<Mat> BuildRotations(double volume, int v, int f)
    {
        var list = new List<Mat>(256);
        AddCubeRotations(list);

        int complexity = v + f;
        if (volume <= 27000 && complexity <= 3000)
        {
            for (int axis = 0; axis < 3; axis++)
                for (int deg = 15; deg < 360; deg += 15)
                    list.Add(AxisRotation(axis, deg * Math.PI / 180.0));
        }
        else if (volume <= 1_000_000 && complexity <= 10000)
        {
            for (int axis = 0; axis < 3; axis++)
                for (int deg = 30; deg < 360; deg += 30)
                    list.Add(AxisRotation(axis, deg * Math.PI / 180.0));
        }

        return list;
    }

    static void AddCubeRotations(List<Mat> list)
    {
        int[] p = { 0, 1, 2 };
        foreach (var perm in Permutations())
        {
            for (int sx = -1; sx <= 1; sx += 2)
                for (int sy = -1; sy <= 1; sy += 2)
                    for (int sz = -1; sz <= 1; sz += 2)
                    {
                        double[,] a = new double[3, 3];
                        a[0, perm[0]] = sx;
                        a[1, perm[1]] = sy;
                        a[2, perm[2]] = sz;
                        double det =
                            a[0, 0] * (a[1, 1] * a[2, 2] - a[1, 2] * a[2, 1]) -
                            a[0, 1] * (a[1, 0] * a[2, 2] - a[1, 2] * a[2, 0]) +
                            a[0, 2] * (a[1, 0] * a[2, 1] - a[1, 1] * a[2, 0]);
                        if (det > 0.5)
                        {
                            list.Add(new Mat { M00 = a[0, 0], M01 = a[0, 1], M02 = a[0, 2], M10 = a[1, 0], M11 = a[1, 1], M12 = a[1, 2], M20 = a[2, 0], M21 = a[2, 1], M22 = a[2, 2] });
                        }
                    }
        }
    }

    static IEnumerable<int[]> Permutations()
    {
        yield return new[] { 0, 1, 2 };
        yield return new[] { 0, 2, 1 };
        yield return new[] { 1, 0, 2 };
        yield return new[] { 1, 2, 0 };
        yield return new[] { 2, 0, 1 };
        yield return new[] { 2, 1, 0 };
    }

    static Mat AxisRotation(int axis, double a)
    {
        double c = Math.Cos(a), s = Math.Sin(a);
        if (axis == 0) return new Mat { M00 = 1, M11 = c, M12 = -s, M21 = s, M22 = c };
        if (axis == 1) return new Mat { M11 = 1, M00 = c, M02 = s, M20 = -s, M22 = c };
        return new Mat { M22 = 1, M00 = c, M01 = -s, M10 = s, M11 = c };
    }

    static (double, double, double, double) Quaternion(Mat m)
    {
        double qw, qx, qy, qz;
        double tr = m.M00 + m.M11 + m.M22;
        if (tr > 0)
        {
            double s = Math.Sqrt(tr + 1.0) * 2.0;
            qw = 0.25 * s; qx = (m.M21 - m.M12) / s; qy = (m.M02 - m.M20) / s; qz = (m.M10 - m.M01) / s;
        }
        else if (m.M00 > m.M11 && m.M00 > m.M22)
        {
            double s = Math.Sqrt(1.0 + m.M00 - m.M11 - m.M22) * 2.0;
            qw = (m.M21 - m.M12) / s; qx = 0.25 * s; qy = (m.M01 + m.M10) / s; qz = (m.M02 + m.M20) / s;
        }
        else if (m.M11 > m.M22)
        {
            double s = Math.Sqrt(1.0 + m.M11 - m.M00 - m.M22) * 2.0;
            qw = (m.M02 - m.M20) / s; qx = (m.M01 + m.M10) / s; qy = 0.25 * s; qz = (m.M12 + m.M21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m.M22 - m.M00 - m.M11) * 2.0;
            qw = (m.M10 - m.M01) / s; qx = (m.M02 + m.M20) / s; qy = (m.M12 + m.M21) / s; qz = 0.25 * s;
        }
        double n = Math.Sqrt(qw * qw + qx * qx + qy * qy + qz * qz);
        if (qw < 0) n = -n;
        return (qw / n, qx / n, qy / n, qz / n);
    }

    static string F(double x)
    {
        if (Math.Abs(x) < 5e-13) x = 0;
        return x.ToString("G17", CultureInfo.InvariantCulture);
    }
}
