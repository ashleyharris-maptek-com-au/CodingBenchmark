use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Vec2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

struct Tet {
    p: Vec3,
    q: Quat,
    scale: f64,
}

fn dot(a: Vec3, b: Vec3) -> f64 {
    a.x * b.x + a.y * b.y + a.z * b.z
}

fn cross(a: Vec3, b: Vec3) -> Vec3 {
    Vec3 {
        x: a.y * b.z - a.z * b.y,
        y: a.z * b.x - a.x * b.z,
        z: a.x * b.y - a.y * b.x,
    }
}

fn sub(a: Vec3, b: Vec3) -> Vec3 {
    Vec3 {
        x: a.x - b.x,
        y: a.y - b.y,
        z: a.z - b.z,
    }
}

fn norm(v: Vec3) -> f64 {
    dot(v, v).sqrt()
}

fn normalize(v: Vec3) -> Vec3 {
    let n = norm(v);
    if n <= 1.0e-12 {
        Vec3 { x: 0.0, y: 0.0, z: 1.0 }
    } else {
        Vec3 {
            x: v.x / n,
            y: v.y / n,
            z: v.z / n,
        }
    }
}

fn q_normalize(q: Quat) -> Quat {
    let n = (q.w * q.w + q.x * q.x + q.y * q.y + q.z * q.z).sqrt();
    Quat {
        w: q.w / n,
        x: q.x / n,
        y: q.y / n,
        z: q.z / n,
    }
}

fn q_mul(a: Quat, b: Quat) -> Quat {
    Quat {
        w: a.w * b.w - a.x * b.x - a.y * b.y - a.z * b.z,
        x: a.w * b.x + a.x * b.w + a.y * b.z - a.z * b.y,
        y: a.w * b.y - a.x * b.z + a.y * b.w + a.z * b.x,
        z: a.w * b.z + a.x * b.y - a.y * b.x + a.z * b.w,
    }
}

fn q_from_axis_angle(axis: Vec3, angle: f64) -> Quat {
    let half = angle * 0.5;
    let s = half.sin();
    let a = normalize(axis);
    q_normalize(Quat {
        w: half.cos(),
        x: a.x * s,
        y: a.y * s,
        z: a.z * s,
    })
}

fn q_from_to(from: Vec3, to: Vec3) -> Quat {
    let f = normalize(from);
    let t = normalize(to);
    let d = dot(f, t);
    if d > 1.0 - 1.0e-12 {
        return Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 };
    }
    if d < -1.0 + 1.0e-12 {
        return q_from_axis_angle(Vec3 { x: 0.0, y: 1.0, z: 0.0 }, PI);
    }
    let c = cross(f, t);
    q_normalize(Quat {
        w: 1.0 + d,
        x: c.x,
        y: c.y,
        z: c.z,
    })
}

fn q_rotate(q: Quat, v: Vec3) -> Vec3 {
    let u = Vec3 { x: q.x, y: q.y, z: q.z };
    let uv = cross(u, v);
    let uuv = cross(u, uv);
    Vec3 {
        x: v.x + 2.0 * (q.w * uv.x + uuv.x),
        y: v.y + 2.0 * (q.w * uv.y + uuv.y),
        z: v.z + 2.0 * (q.w * uv.z + uuv.z),
    }
}

fn make_tet(center_shadow: Vec2, scale: f64, z: f64, yaw: f64, kx: f64, ky: f64, align: Quat) -> Tet {
    let yaw_q = q_from_axis_angle(Vec3 { x: 0.0, y: 0.0, z: 1.0 }, yaw);
    let q = q_normalize(q_mul(yaw_q, align));
    let a = 0.707_f64;
    let v0 = Vec3 { x: 1.0, y: 0.0, z: -a };
    let v2 = Vec3 { x: 0.0, y: 1.0, z: a };
    let v3 = Vec3 { x: 0.0, y: -1.0, z: a };
    let r0 = q_rotate(q, v0);
    let r2 = q_rotate(q, v2);
    let r3 = q_rotate(q, v3);
    let face_centroid = Vec3 {
        x: (r0.x + r2.x + r3.x) / 3.0,
        y: (r0.y + r2.y + r3.y) / 3.0,
        z: (r0.z + r2.z + r3.z) / 3.0,
    };
    let face_z = z + scale * face_centroid.z;
    Tet {
        p: Vec3 {
            x: center_shadow.x + kx * face_z - scale * face_centroid.x,
            y: center_shadow.y + ky * face_z - scale * face_centroid.y,
            z,
        },
        q,
        scale,
    }
}

fn polygon_area(poly: &[Vec2]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        s += poly[i].x * poly[j].y - poly[j].x * poly[i].y;
    }
    s * 0.5
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("stdin");
    let mut it = input.split_whitespace();
    let nv = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    if nv == 0 {
        println!("0");
        return;
    }

    let mut poly = Vec::with_capacity(nv);
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for _ in 0..nv {
        let x = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let y = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        poly.push(Vec2 { x, y });
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }
    let sx = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let sy = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let sz = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(-1.0);
    let denom = if sz.abs() < 1.0e-9 { if sz.is_sign_negative() { -1.0e-9 } else { 1.0e-9 } } else { sz };
    let kx = sx / denom;
    let ky = sy / denom;

    let a = 0.707_f64;
    let face_a = Vec3 { x: 1.0, y: 0.0, z: -a };
    let face_b = Vec3 { x: 0.0, y: 1.0, z: a };
    let face_c = Vec3 { x: 0.0, y: -1.0, z: a };
    let normal = cross(sub(face_b, face_a), sub(face_c, face_a));
    let align = q_from_to(normal, Vec3 { x: 0.0, y: 0.0, z: 1.0 });

    let width = (max_x - min_x).max(0.0);
    let height = (max_y - min_y).max(0.0);
    let cx = (min_x + max_x) * 0.5;
    let cy = (min_y + max_y) * 0.5;
    let base_inradius = 3.0_f64.sqrt() / 3.0;
    let diag = (width * width + height * height).sqrt();
    let mut tets: Vec<Tet> = Vec::new();

    if diag <= 11.2 {
        let scale = (diag * 0.5 / base_inradius).max(0.05).min(10.0);
        tets.push(make_tet(Vec2 { x: cx, y: cy }, scale, 25.0, 0.0, kx, ky, align));
    } else {
        let area = polygon_area(&poly).abs();
        let complexity = nv as f64 + area.sqrt() * 0.02;
        let scale = if width.max(height) < 35.0 && complexity < 80.0 { 7.0 } else { 10.0 };
        let inradius = base_inradius * scale;
        let step = inradius * 1.30;
        let margin = inradius + step;
        let start_x = min_x - margin;
        let end_x = max_x + margin;
        let start_y = min_y - margin;
        let end_y = max_y + margin;
        let mut row = 0usize;
        let mut y = start_y;
        while y <= end_y {
            let offset = if row % 2 == 0 { 0.0 } else { step * 0.5 };
            let mut x = start_x + offset;
            while x <= end_x {
                let idx = tets.len() as f64;
                let z = 25.0 + idx * (scale * 4.0 + 10.0);
                let yaw = if row % 2 == 0 { 0.0 } else { PI / 3.0 };
                tets.push(make_tet(Vec2 { x, y }, scale, z, yaw, kx, ky, align));
                x += step;
            }
            y += step;
            row += 1;
        }
    }

    println!("{}", tets.len());
    for tet in tets {
        println!(
            "{:.9} {:.9} {:.9} {:.12} {:.12} {:.12} {:.12} {:.9}",
            tet.p.x, tet.p.y, tet.p.z, tet.q.w, tet.q.x, tet.q.y, tet.q.z, tet.scale
        );
    }
}
