using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    private static readonly int[] Dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
    private static readonly int[] Dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

    private struct Clue
    {
        public int R, C, N;
        public Clue(int r, int c, int n) { R = r; C = c; N = n; }
    }

    private sealed class Constraint
    {
        public int[] Vars = Array.Empty<int>();
        public int Need;
    }

    public static void Main()
    {
        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first)) return;
        string[] hw = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int h = int.Parse(hw[0]);
        int w = int.Parse(hw[1]);

        string[] board = new string[h];
        var clues = new List<Clue>();

        for (int r = 0; r < h; r++)
        {
            string line = Console.ReadLine() ?? "";
            if (line.Length < w) line = line.PadRight(w);
            if (line.Length > w) line = line.Substring(0, w);
            board[r] = line;

            for (int c = 0; c < w; c++)
            {
                char ch = line[c];
                if (ch >= '0' && ch <= '8')
                    clues.Add(new Clue(r, c, ch - '0'));
            }
        }

        var safe = new HashSet<int>();
        var mines = new HashSet<int>();

        RunLocalFixpoint(board, h, w, clues, safe, mines);

        // Expensive reasoning is useful on dense/complex fronts but harmful on huge open boards.
        // The limits keep memory and exponential search bounded while still solving hard local CSPs.
        RunSubsetReasoning(board, h, w, clues, safe, mines, maxConstraints: 250_000, maxVars: 250_000);
        RunLocalFixpoint(board, h, w, clues, safe, mines);

        RunSmallComponentCsp(board, h, w, clues, safe, mines, maxVars: 80_000, maxComponentVars: 24);
        RunLocalFixpoint(board, h, w, clues, safe, mines);

        using var output = new StreamWriter(Console.OpenStandardOutput()) { NewLine = "\n" };
        output.WriteLine(safe.Count);
        foreach (int id in safe)
        {
            int y = id / w;
            int x = id - y * w;
            output.Write(x);
            output.Write(' ');
            output.WriteLine(y);
        }
    }

    private static void RunLocalFixpoint(string[] board, int h, int w, List<Clue> clues, HashSet<int> safe, HashSet<int> mines)
    {
        bool changed;
        int passes = 0;
        do
        {
            changed = false;
            passes++;

            foreach (var clue in clues)
            {
                int flagged = 0;
                int unknownCount = 0;

                for (int k = 0; k < 8; k++)
                {
                    int nr = clue.R + Dr[k], nc = clue.C + Dc[k];
                    if ((uint)nr >= (uint)h || (uint)nc >= (uint)w) continue;
                    int id = nr * w + nc;
                    char ch = board[nr][nc];
                    if (ch == '*' || mines.Contains(id))
                    {
                        flagged++;
                    }
                    else if (ch == ' ' && !safe.Contains(id))
                    {
                        unknownCount++;
                    }
                }

                int need = clue.N - flagged;
                if (need < 0 || need > unknownCount) continue; // Inconsistent input/flags: do not infer from it.

                if (need == 0 || need == unknownCount)
                {
                    for (int k = 0; k < 8; k++)
                    {
                        int nr = clue.R + Dr[k], nc = clue.C + Dc[k];
                        if ((uint)nr >= (uint)h || (uint)nc >= (uint)w) continue;
                        int id = nr * w + nc;
                        if (board[nr][nc] != ' ' || safe.Contains(id) || mines.Contains(id)) continue;

                        if (need == 0)
                        {
                            if (safe.Add(id)) changed = true;
                        }
                        else
                        {
                            if (mines.Add(id)) changed = true;
                        }
                    }
                }
            }

            // Most boards settle quickly. This is a safety valve for pathological full-board rescans.
            if (passes > 64 && clues.Count > 1_000_000) break;
        } while (changed);
    }

    private static List<Constraint> BuildConstraints(
        string[] board, int h, int w, List<Clue> clues, HashSet<int> safe, HashSet<int> mines,
        Dictionary<int, int> varId, List<int> cells, int maxVars, int maxConstraints)
    {
        var constraints = new List<Constraint>();

        foreach (var clue in clues)
        {
            int flagged = 0;
            int[] tmp = new int[8];
            int n = 0;

            for (int k = 0; k < 8; k++)
            {
                int nr = clue.R + Dr[k], nc = clue.C + Dc[k];
                if ((uint)nr >= (uint)h || (uint)nc >= (uint)w) continue;
                int id = nr * w + nc;
                char ch = board[nr][nc];
                if (ch == '*' || mines.Contains(id)) flagged++;
                else if (ch == ' ' && !safe.Contains(id)) tmp[n++] = id;
            }

            int need = clue.N - flagged;
            if (need < 0 || need > n || n == 0) continue;

            var vars = new int[n];
            for (int i = 0; i < n; i++)
            {
                int cell = tmp[i];
                if (!varId.TryGetValue(cell, out int v))
                {
                    if (varId.Count >= maxVars) return constraints;
                    v = varId.Count;
                    varId[cell] = v;
                    cells.Add(cell);
                }
                vars[i] = v;
            }
            Array.Sort(vars);
            constraints.Add(new Constraint { Vars = vars, Need = need });
            if (constraints.Count >= maxConstraints) return constraints;
        }

        return constraints;
    }

    private static void RunSubsetReasoning(
        string[] board, int h, int w, List<Clue> clues, HashSet<int> safe, HashSet<int> mines,
        int maxConstraints, int maxVars)
    {
        bool changed;
        int rounds = 0;
        do
        {
            changed = false;
            rounds++;

            var varId = new Dictionary<int, int>();
            var cells = new List<int>();
            var constraints = BuildConstraints(board, h, w, clues, safe, mines, varId, cells, maxVars, maxConstraints);
            if (constraints.Count == 0 || constraints.Count >= maxConstraints || varId.Count >= maxVars) return;

            var exact = new Dictionary<string, List<Constraint>>();
            foreach (var c in constraints)
            {
                string key = Key(c.Vars);
                if (!exact.TryGetValue(key, out var list))
                    exact[key] = list = new List<Constraint>();
                list.Add(c);
            }

            foreach (var b in constraints)
            {
                if (b.Vars.Length <= 1) continue;
                int masks = 1 << b.Vars.Length;
                for (int mask = 1; mask < masks - 1; mask++)
                {
                    int[] sub = SelectByMask(b.Vars, mask);
                    if (!exact.TryGetValue(Key(sub), out var candidates)) continue;

                    foreach (var a in candidates)
                    {
                        if (a.Need > b.Need) continue;
                        int diffNeed = b.Need - a.Need;
                        int diffSize = b.Vars.Length - a.Vars.Length;
                        if (diffNeed != 0 && diffNeed != diffSize) continue;

                        foreach (int v in Difference(b.Vars, a.Vars))
                        {
                            int cell = cells[v];
                            if (diffNeed == 0)
                            {
                                if (safe.Add(cell)) changed = true;
                            }
                            else
                            {
                                if (mines.Add(cell)) changed = true;
                            }
                        }
                    }
                }
            }

            if (changed) RunLocalFixpoint(board, h, w, clues, safe, mines);
        } while (changed && rounds < 8);
    }

    private static string Key(int[] vars) => string.Join(",", vars);

    private static int[] SelectByMask(int[] vars, int mask)
    {
        int count = 0;
        for (int i = 0; i < vars.Length; i++)
            if (((mask >> i) & 1) != 0) count++;

        int[] result = new int[count];
        int p = 0;
        for (int i = 0; i < vars.Length; i++)
            if (((mask >> i) & 1) != 0) result[p++] = vars[i];
        return result;
    }

    private static IEnumerable<int> Difference(int[] b, int[] a)
    {
        int i = 0, j = 0;
        while (i < b.Length)
        {
            if (j >= a.Length || b[i] < a[j]) yield return b[i++];
            else if (b[i] == a[j]) { i++; j++; }
            else j++;
        }
    }

    private static void RunSmallComponentCsp(
        string[] board, int h, int w, List<Clue> clues, HashSet<int> safe, HashSet<int> mines,
        int maxVars, int maxComponentVars)
    {
        var varId = new Dictionary<int, int>();
        var cells = new List<int>();
        var constraints = BuildConstraints(board, h, w, clues, safe, mines, varId, cells, maxVars, maxConstraints: 500_000);
        int n = cells.Count;
        if (n == 0 || n >= maxVars) return;

        var parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = i;

        foreach (var c in constraints)
            for (int i = 1; i < c.Vars.Length; i++)
                Union(parent, c.Vars[0], c.Vars[i]);

        var compVars = new Dictionary<int, List<int>>();
        for (int i = 0; i < n; i++)
        {
            int root = Find(parent, i);
            if (!compVars.TryGetValue(root, out var list))
                compVars[root] = list = new List<int>();
            list.Add(i);
        }

        var varToConstraints = new List<int>[n];
        for (int i = 0; i < n; i++) varToConstraints[i] = new List<int>();
        for (int ci = 0; ci < constraints.Count; ci++)
            foreach (int v in constraints[ci].Vars)
                varToConstraints[v].Add(ci);

        foreach (var vars in compVars.Values)
        {
            if (vars.Count > maxComponentVars) continue;

            var cset = new HashSet<int>();
            foreach (int v in vars)
                foreach (int ci in varToConstraints[v])
                    cset.Add(ci);

            var compConstraints = cset.Select(ci => constraints[ci]).ToList();
            SolveComponent(vars, compConstraints, cells, safe, mines);
        }
    }

    private static void SolveComponent(List<int> globalVars, List<Constraint> constraints, List<int> cells, HashSet<int> safe, HashSet<int> mines)
    {
        int n = globalVars.Count;
        var local = new Dictionary<int, int>();
        for (int i = 0; i < n; i++) local[globalVars[i]] = i;

        var cons = new List<int[]>();
        var need = new List<int>();
        foreach (var c in constraints)
        {
            var arr = c.Vars.Select(v => local[v]).OrderBy(v => v).ToArray();
            cons.Add(arr);
            need.Add(c.Need);
        }

        var consByVar = new List<int>[n];
        for (int i = 0; i < n; i++) consByVar[i] = new List<int>();
        for (int ci = 0; ci < cons.Count; ci++)
            foreach (int v in cons[ci])
                consByVar[v].Add(ci);

        int[] assignedMines = new int[cons.Count];
        int[] unassigned = cons.Select(c => c.Length).ToArray();
        int[] assignment = Enumerable.Repeat(-1, n).ToArray();
        long solutions = 0;
        long[] mineInSolutions = new long[n];

        void Search(int pos)
        {
            if (solutions > 2_000_000) return; // Enough evidence; avoids runaway components.

            if (pos == n)
            {
                for (int ci = 0; ci < cons.Count; ci++)
                    if (assignedMines[ci] != need[ci]) return;

                solutions++;
                for (int i = 0; i < n; i++)
                    if (assignment[i] == 1) mineInSolutions[i]++;
                return;
            }

            int v = PickVariable(assignment, consByVar, assignedMines, unassigned, need);
            for (int val = 0; val <= 1; val++)
            {
                assignment[v] = val;
                bool ok = true;
                foreach (int ci in consByVar[v])
                {
                    unassigned[ci]--;
                    assignedMines[ci] += val;
                    if (assignedMines[ci] > need[ci] || assignedMines[ci] + unassigned[ci] < need[ci])
                        ok = false;
                }

                if (ok) Search(pos + 1);

                foreach (int ci in consByVar[v])
                {
                    assignedMines[ci] -= val;
                    unassigned[ci]++;
                }
                assignment[v] = -1;
            }
        }

        Search(0);
        if (solutions == 0 || solutions > 2_000_000) return;

        for (int i = 0; i < n; i++)
        {
            int cell = cells[globalVars[i]];
            if (mineInSolutions[i] == 0) safe.Add(cell);
            else if (mineInSolutions[i] == solutions) mines.Add(cell);
        }
    }

    private static int PickVariable(int[] assignment, List<int>[] consByVar, int[] assignedMines, int[] unassigned, List<int> need)
    {
        int best = -1;
        int bestScore = int.MinValue;
        for (int v = 0; v < assignment.Length; v++)
        {
            if (assignment[v] != -1) continue;
            int score = consByVar[v].Count * 4;
            foreach (int ci in consByVar[v])
            {
                if (need[ci] - assignedMines[ci] == 0 || need[ci] - assignedMines[ci] == unassigned[ci])
                    score += 8;
            }
            if (score > bestScore)
            {
                bestScore = score;
                best = v;
            }
        }
        return best;
    }

    private static int Find(int[] parent, int x)
    {
        while (parent[x] != x)
        {
            parent[x] = parent[parent[x]];
            x = parent[x];
        }
        return x;
    }

    private static void Union(int[] parent, int a, int b)
    {
        int ra = Find(parent, a), rb = Find(parent, b);
        if (ra != rb) parent[rb] = ra;
    }
}
