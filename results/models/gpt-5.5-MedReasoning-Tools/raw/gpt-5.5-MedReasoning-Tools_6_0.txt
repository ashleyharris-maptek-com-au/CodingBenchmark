using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    const double EPS = 1e-12;

    struct Vec
    {
        public double X, Y, Z;
        public Vec(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec operator +(Vec a, Vec b) => new Vec(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec operator -(Vec a, Vec b) => new Vec(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec operator *(Vec a, double s) => new Vec(a.X * s, a.Y * s, a.Z * s);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double Norm2() => X * X + Y * Y + Z * Z;
        public static double Dot(Vec a, Vec b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec Cross(Vec a, Vec b) => new Vec(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    sealed class State
    {
        public Vec R, V;
        public double Rmag, Vmag, A, E, Inc, Raan, ArgP, TrueAnom, Period, Energy;
        public Vec Hhat, Evec;

        public State(Vec r, Vec v, double mu)
        {
            R = r; V = v;
            Rmag = r.Norm();
            Vmag = v.Norm();
            Vec h = Vec.Cross(r, v);
            double hmag = h.Norm();
            Hhat = hmag > EPS ? h * (1.0 / hmag) : new Vec(0, 0, 1);

            Energy = 0.5 * Vmag * Vmag - mu / Math.Max(Rmag, EPS);
            A = Math.Abs(Energy) > EPS ? -mu / (2.0 * Energy) : 1e12;

            Evec = (Vec.Cross(v, h) * (1.0 / mu)) - (r * (1.0 / Math.Max(Rmag, EPS)));
            E = Evec.Norm();

            Inc = Math.Acos(Clamp(Hhat.Z, -1, 1));
            Vec n = Vec.Cross(new Vec(0, 0, 1), h);
            double nmag = n.Norm();
            Raan = nmag > EPS ? NormAngle(Math.Atan2(n.Y, n.X)) : 0.0;

            if (nmag > EPS && E > 1e-8)
            {
                double c = Clamp(Vec.Dot(n, Evec) / (nmag * E), -1, 1);
                ArgP = Math.Acos(c);
                if (Evec.Z < 0) ArgP = 2.0 * Math.PI - ArgP;
            }
            else ArgP = 0.0;

            if (E > 1e-8)
            {
                double c = Clamp(Vec.Dot(Evec, r) / (E * Rmag), -1, 1);
                TrueAnom = Math.Acos(c);
                if (Vec.Dot(r, v) < 0) TrueAnom = 2.0 * Math.PI - TrueAnom;
            }
            else
            {
                TrueAnom = NormAngle(Math.Atan2(r.Y, r.X));
            }

            Period = A > 0 ? 2.0 * Math.PI * Math.Sqrt(A * A * A / mu) : 1e12;
        }
    }

    public static void Main()
    {
        string all = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(all)) return;
        all = all.TrimStart('\uFEFF');

        string[] tok = all.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;
        int n = int.Parse(tok[p++], CultureInfo.InvariantCulture);
        double mu = double.Parse(tok[p++], CultureInfo.InvariantCulture);

        State start = ReadState(tok, ref p, mu);
        State[] s = new State[n];
        for (int i = 0; i < n; i++) s[i] = ReadState(tok, ref p, mu);

        if (n == 0) { Console.WriteLine(); return; }
        if (n == 1) { Console.WriteLine("0"); return; }

        double[,] c = BuildCostMatrix(start, s, mu);
        int[] order = n <= 11 ? SolveExactPath(c, n) : SolveHeuristic(c, n);

        Console.WriteLine(string.Join(" ", order));
    }

    static State ReadState(string[] tok, ref int p, double mu)
    {
        double x = double.Parse(tok[p++], CultureInfo.InvariantCulture);
        double y = double.Parse(tok[p++], CultureInfo.InvariantCulture);
        double z = double.Parse(tok[p++], CultureInfo.InvariantCulture);
        double vx = double.Parse(tok[p++], CultureInfo.InvariantCulture);
        double vy = double.Parse(tok[p++], CultureInfo.InvariantCulture);
        double vz = double.Parse(tok[p++], CultureInfo.InvariantCulture);
        return new State(new Vec(x, y, z), new Vec(vx, vy, vz), mu);
    }

    static double[,] BuildCostMatrix(State start, State[] s, double mu)
    {
        int n = s.Length;
        double[,] c = new double[n + 1, n];
        for (int i = 0; i < n; i++) c[0, i] = TransferCost(start, s[i], mu);
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                c[i + 1, j] = i == j ? 0.0 : TransferCost(s[i], s[j], mu);
        return c;
    }

    static double TransferCost(State a, State b, double mu)
    {
        double r1 = Math.Max(a.Rmag, 1.0);
        double r2 = Math.Max(b.Rmag, 1.0);
        double sma1 = Math.Max(1.0, Math.Abs(a.A));
        double sma2 = Math.Max(1.0, Math.Abs(b.A));

        double vCirc1 = Math.Sqrt(mu / r1);
        double vCirc2 = Math.Sqrt(mu / r2);
        double xferA = 0.5 * (r1 + r2);
        double dv1 = Math.Abs(Math.Sqrt(mu * (2.0 / r1 - 1.0 / xferA)) - vCirc1);
        double dv2 = Math.Abs(vCirc2 - Math.Sqrt(mu * (2.0 / r2 - 1.0 / xferA)));
        double hohmann = dv1 + dv2;

        double plane = Math.Acos(Clamp(Vec.Dot(a.Hhat, b.Hhat), -1, 1));
        double planeDv = 2.0 * Math.Min(a.Vmag, b.Vmag) * Math.Sin(0.5 * plane);

        double eccPenalty = 1.7 * Math.Abs(a.E - b.E);
        double energyPenalty = 0.16 * Math.Abs(a.Energy - b.Energy);
        double smaPenalty = 0.00018 * Math.Abs(sma1 - sma2);

        double node = AngleDiff(a.Raan, b.Raan);
        double peri = AngleDiff(a.ArgP, b.ArgP);
        double phase = AngleDiff(a.TrueAnom, b.TrueAnom);
        double angularPenalty = 0.025 * (node + 0.55 * peri + 0.25 * phase) * Math.Min(a.Vmag, b.Vmag);

        double euclid = (a.R - b.R).Norm() * 0.000035 + (a.V - b.V).Norm() * 0.18;
        double periodPenalty = 0.000002 * Math.Abs(a.Period - b.Period);

        return hohmann + planeDv + eccPenalty + energyPenalty + smaPenalty + angularPenalty + euclid + periodPenalty;
    }

    static int[] SolveExactPath(double[,] c, int n)
    {
        int masks = 1 << n;
        double[,] dp = new double[masks, n];
        int[,] parent = new int[masks, n];
        for (int m = 0; m < masks; m++)
            for (int i = 0; i < n; i++)
            {
                dp[m, i] = double.PositiveInfinity;
                parent[m, i] = -1;
            }

        for (int i = 0; i < n; i++) dp[1 << i, i] = c[0, i];

        for (int mask = 1; mask < masks; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                double cur = dp[mask, last];
                if (double.IsInfinity(cur)) continue;
                int rem = ((masks - 1) ^ mask);
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int nxt = BitIndex(bit);
                    int nm = mask | bit;
                    double cand = cur + c[last + 1, nxt];
                    if (cand < dp[nm, nxt])
                    {
                        dp[nm, nxt] = cand;
                        parent[nm, nxt] = last;
                    }
                    rem -= bit;
                }
            }
        }

        int full = masks - 1, best = 0;
        double bestVal = dp[full, 0];
        for (int i = 1; i < n; i++)
            if (dp[full, i] < bestVal) { bestVal = dp[full, i]; best = i; }

        int[] order = new int[n];
        int mask2 = full, curLast = best;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            order[pos] = curLast;
            int prev = parent[mask2, curLast];
            mask2 ^= 1 << curLast;
            curLast = prev;
        }
        return order;
    }

    static int[] SolveHeuristic(double[,] c, int n)
    {
        int[] nn = NearestNeighbor(c, n);
        int[] ins = CheapestInsertion(c, n);
        int[] order = PathCost(c, ins) < PathCost(c, nn) ? ins : nn;

        TwoOpt(c, order, n <= 35 ? 90 : n <= 70 ? 55 : 35);
        Relocate(c, order, n <= 35 ? 45 : 25);
        TwoOpt(c, order, n <= 35 ? 45 : 20);
        OrOpt2(c, order, n <= 60 ? 20 : 12);
        TwoOpt(c, order, 15);
        return order;
    }

    static int[] NearestNeighbor(double[,] c, int n)
    {
        bool[] used = new bool[n];
        int[] order = new int[n];
        int last = -1;
        for (int k = 0; k < n; k++)
        {
            int best = -1;
            double bv = double.PositiveInfinity;
            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                double v = last < 0 ? c[0, i] : c[last + 1, i];
                if (v < bv) { bv = v; best = i; }
            }
            order[k] = best;
            used[best] = true;
            last = best;
        }
        return order;
    }

    static int[] CheapestInsertion(double[,] c, int n)
    {
        bool[] used = new bool[n];
        List<int> path = new List<int>(n);
        int first = 0;
        double bf = c[0, 0];
        for (int i = 1; i < n; i++) if (c[0, i] < bf) { bf = c[0, i]; first = i; }
        path.Add(first);
        used[first] = true;

        while (path.Count < n)
        {
            int bestNode = -1, bestPos = 0;
            double bestDelta = double.PositiveInfinity;
            for (int node = 0; node < n; node++)
            {
                if (used[node]) continue;
                for (int pos = 0; pos <= path.Count; pos++)
                {
                    double oldCost = 0, newCost = 0;
                    if (pos == 0)
                    {
                        oldCost = c[0, path[0]];
                        newCost = c[0, node] + c[node + 1, path[0]];
                    }
                    else if (pos == path.Count)
                    {
                        oldCost = 0;
                        newCost = c[path[pos - 1] + 1, node];
                    }
                    else
                    {
                        oldCost = c[path[pos - 1] + 1, path[pos]];
                        newCost = c[path[pos - 1] + 1, node] + c[node + 1, path[pos]];
                    }
                    double delta = newCost - oldCost;
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestNode = node;
                        bestPos = pos;
                    }
                }
            }
            path.Insert(bestPos, bestNode);
            used[bestNode] = true;
        }
        return path.ToArray();
    }

    static void TwoOpt(double[,] c, int[] o, int maxPass)
    {
        int n = o.Length;
        for (int pass = 0; pass < maxPass; pass++)
        {
            bool changed = false;
            for (int i = 0; i < n - 2; i++)
            {
                int a = i == 0 ? -1 : o[i - 1];
                int b = o[i];
                double leftOld = a < 0 ? c[0, b] : c[a + 1, b];
                for (int j = i + 1; j < n; j++)
                {
                    int d = o[j];
                    int e = j == n - 1 ? -1 : o[j + 1];
                    double oldCost = leftOld + (e < 0 ? 0 : c[d + 1, e]);
                    double newCost = (a < 0 ? c[0, d] : c[a + 1, d]) + (e < 0 ? 0 : c[b + 1, e]);
                    if (newCost + 1e-10 < oldCost)
                    {
                        Array.Reverse(o, i, j - i + 1);
                        changed = true;
                    }
                }
            }
            if (!changed) break;
        }
    }

    static void Relocate(double[,] c, int[] o, int maxPass)
    {
        int n = o.Length;
        for (int pass = 0; pass < maxPass; pass++)
        {
            bool changed = false;
            for (int from = 0; from < n; from++)
            {
                int node = o[from];
                int prev = from == 0 ? -1 : o[from - 1];
                int next = from == n - 1 ? -1 : o[from + 1];
                double removeOld = (prev < 0 ? c[0, node] : c[prev + 1, node]) + (next < 0 ? 0 : c[node + 1, next]);
                double removeNew = next < 0 ? 0 : (prev < 0 ? c[0, next] : c[prev + 1, next]);
                double removeGain = removeNew - removeOld;

                for (int to = 0; to <= n; to++)
                {
                    if (to == from || to == from + 1) continue;
                    int before = to == 0 ? -1 : o[to - 1];
                    int after = to == n ? -1 : o[to];
                    if (before == node || after == node) continue;
                    double insertOld = after < 0 ? 0 : (before < 0 ? c[0, after] : c[before + 1, after]);
                    double insertNew = (before < 0 ? c[0, node] : c[before + 1, node]) + (after < 0 ? 0 : c[node + 1, after]);
                    if (removeGain + insertNew - insertOld < -1e-10)
                    {
                        MoveOne(o, from, to);
                        changed = true;
                        goto NextPass;
                    }
                }
            }
        NextPass:
            if (!changed) break;
        }
    }

    static void OrOpt2(double[,] c, int[] o, int maxPass)
    {
        int n = o.Length;
        if (n < 4) return;
        for (int pass = 0; pass < maxPass; pass++)
        {
            bool changed = false;
            for (int from = 0; from < n - 1; from++)
            {
                int x = o[from], y = o[from + 1];
                int prev = from == 0 ? -1 : o[from - 1];
                int next = from + 2 >= n ? -1 : o[from + 2];
                double remOld = (prev < 0 ? c[0, x] : c[prev + 1, x]) + c[x + 1, y] + (next < 0 ? 0 : c[y + 1, next]);
                double remNew = next < 0 ? 0 : (prev < 0 ? c[0, next] : c[prev + 1, next]);

                for (int to = 0; to <= n; to++)
                {
                    if (to >= from && to <= from + 2) continue;
                    int before = to == 0 ? -1 : o[to - 1];
                    int after = to == n ? -1 : o[to];
                    if (before == x || before == y || after == x || after == y) continue;
                    double insOld = after < 0 ? 0 : (before < 0 ? c[0, after] : c[before + 1, after]);
                    double insNew = (before < 0 ? c[0, x] : c[before + 1, x]) + c[x + 1, y] + (after < 0 ? 0 : c[y + 1, after]);
                    if ((remNew - remOld) + (insNew - insOld) < -1e-10)
                    {
                        MoveBlock(o, from, 2, to);
                        changed = true;
                        goto NextPass;
                    }
                }
            }
        NextPass:
            if (!changed) break;
        }
    }

    static void MoveOne(int[] o, int from, int to)
    {
        int node = o[from];
        if (to > from) to--;
        if (from < to) Array.Copy(o, from + 1, o, from, to - from);
        else Array.Copy(o, to, o, to + 1, from - to);
        o[to] = node;
    }

    static void MoveBlock(int[] o, int from, int len, int to)
    {
        int[] tmp = new int[len];
        Array.Copy(o, from, tmp, 0, len);
        if (to > from) to -= len;
        if (from < to) Array.Copy(o, from + len, o, from, to - from);
        else Array.Copy(o, to, o, to + len, from - to);
        Array.Copy(tmp, 0, o, to, len);
    }

    static double PathCost(double[,] c, int[] o)
    {
        if (o.Length == 0) return 0;
        double sum = c[0, o[0]];
        for (int i = 1; i < o.Length; i++) sum += c[o[i - 1] + 1, o[i]];
        return sum;
    }

    static int BitIndex(int bit)
    {
        int idx = 0;
        while ((bit >>= 1) != 0) idx++;
        return idx;
    }

    static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);
    static double NormAngle(double a)
    {
        double t = 2.0 * Math.PI;
        a %= t;
        return a < 0 ? a + t : a;
    }
    static double AngleDiff(double a, double b)
    {
        double d = Math.Abs(NormAngle(a) - NormAngle(b));
        return d > Math.PI ? 2.0 * Math.PI - d : d;
    }
}
