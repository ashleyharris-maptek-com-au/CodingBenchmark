using System;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_ptr >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_length == 0) return -1;
            }

            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }

            return value * sign;
        }
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        int m = scanner.NextInt();

        if (n <= 0)
        {
            return;
        }

        if (m == 0)
        {
            Console.Write("0");
            return;
        }

        int[] head = new int[n];
        Array.Fill(head, -1);

        int arcCount = 2 * m;
        int[] to = new int[arcCount];
        int[] next = new int[arcCount];
        int[] edgeId = new int[arcCount];

        int arc = 0;
        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            scanner.NextInt(); // Weight is irrelevant for this valid inspection walk.

            to[arc] = v;
            edgeId[arc] = i;
            next[arc] = head[u];
            head[u] = arc++;

            to[arc] = u;
            edgeId[arc] = i;
            next[arc] = head[v];
            head[v] = arc++;
        }

        // Duplicating every edge makes all degrees even. An Euler tour of that
        // doubled multigraph is a closed route that traverses every input edge.
        byte[] usedCopies = new byte[m];
        int[] cursor = new int[n];
        Array.Copy(head, cursor, n);

        int maxRouteNodes = 2 * m + 1;
        int[] stack = new int[maxRouteNodes];
        int[] route = new int[maxRouteNodes];
        int sp = 0;
        int routeCount = 0;

        stack[sp++] = 0;
        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = cursor[v];

            while (e != -1 && usedCopies[edgeId[e]] == 2)
            {
                e = next[e];
            }

            cursor[v] = e;
            if (e == -1)
            {
                route[routeCount++] = v;
                sp--;
            }
            else
            {
                usedCopies[edgeId[e]]++;
                stack[sp++] = to[e];
            }
        }

        var output = new BufferedStream(Console.OpenStandardOutput(), 1 << 20);
        WriteRoute(output, route, routeCount);
    }

    private static void WriteRoute(Stream output, int[] route, int count)
    {
        byte[] buffer = new byte[1 << 16];
        int pos = 0;

        for (int i = 0; i < count; i++)
        {
            if (i > 0)
            {
                if (pos == buffer.Length) Flush(output, buffer, ref pos);
                buffer[pos++] = (byte)' ';
            }

            WriteInt(route[i], output, buffer, ref pos);
        }

        if (pos > 0)
        {
            output.Write(buffer, 0, pos);
        }

        output.Flush();
    }

    private static void WriteInt(int value, Stream output, byte[] buffer, ref int pos)
    {
        if (value == 0)
        {
            if (pos == buffer.Length) Flush(output, buffer, ref pos);
            buffer[pos++] = (byte)'0';
            return;
        }

        if (value < 0)
        {
            if (pos == buffer.Length) Flush(output, buffer, ref pos);
            buffer[pos++] = (byte)'-';
            value = -value;
        }

        Span<byte> digits = stackalloc byte[11];
        int len = 0;
        while (value > 0)
        {
            digits[len++] = (byte)('0' + (value % 10));
            value /= 10;
        }

        for (int i = len - 1; i >= 0; i--)
        {
            if (pos == buffer.Length) Flush(output, buffer, ref pos);
            buffer[pos++] = digits[i];
        }
    }

    private static void Flush(Stream output, byte[] buffer, ref int pos)
    {
        output.Write(buffer, 0, pos);
        pos = 0;
    }
}
