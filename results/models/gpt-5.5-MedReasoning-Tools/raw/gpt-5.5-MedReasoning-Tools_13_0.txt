#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <unordered_set>
#include <vector>

class SuffixAutomaton {
public:
    struct State {
        int link = -1;
        int len = 0;
        int head = -1;
        int first_pos = -1;
    };

    struct Edge {
        int to;
        int next;
        unsigned char ch;
    };

    std::vector<State> st;
    std::vector<Edge> edges;
    int last = 0;

    explicit SuffixAutomaton(const std::string& s) {
        st.reserve(std::max<std::size_t>(2, 2 * s.size()));
        edges.reserve(std::max<std::size_t>(8, 3 * s.size()));
        st.push_back(State{});
        for (unsigned char c : s) {
            extend(c);
        }
    }

    int transition(int v, unsigned char c) const {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) return edges[e].to;
        }
        return -1;
    }

    const std::vector<State>& states() const {
        return st;
    }

private:
    void add_edge(int from, unsigned char c, int to) {
        edges.push_back(Edge{to, st[from].head, c});
        st[from].head = static_cast<int>(edges.size()) - 1;
    }

    void set_transition(int from, unsigned char c, int to) {
        for (int e = st[from].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                edges[e].to = to;
                return;
            }
        }
        add_edge(from, c, to);
    }

    int copy_transitions(int from) {
        int new_head = -1;
        for (int e = st[from].head; e != -1; e = edges[e].next) {
            edges.push_back(Edge{edges[e].to, new_head, edges[e].ch});
            new_head = static_cast<int>(edges.size()) - 1;
        }
        return new_head;
    }

    void extend(unsigned char c) {
        int cur = static_cast<int>(st.size());
        st.push_back(State{});
        st[cur].len = st[last].len + 1;
        st[cur].first_pos = st[cur].len - 1;

        int p = last;
        while (p != -1 && transition(p, c) == -1) {
            add_edge(p, c, cur);
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = transition(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = static_cast<int>(st.size());
                st.push_back(st[q]);
                st[clone].len = st[p].len + 1;
                st[clone].first_pos = st[q].first_pos;
                st[clone].head = copy_transitions(q);

                while (p != -1 && transition(p, c) == q) {
                    set_transition(p, c, clone);
                    p = st[p].link;
                }

                st[q].link = clone;
                st[cur].link = clone;
            }
        }

        last = cur;
    }
};

static std::string solve_tiny(const std::vector<std::string>& strings, int base_idx) {
    const std::string& base = strings[base_idx];
    for (std::size_t len = base.size(); len > 0; --len) {
        for (std::size_t pos = 0; pos + len <= base.size(); ++pos) {
            std::string candidate = base.substr(pos, len);
            bool ok = true;
            for (std::size_t i = 0; i < strings.size(); ++i) {
                if (static_cast<int>(i) == base_idx) continue;
                if (strings[i].find(candidate) == std::string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) return candidate;
        }
    }
    return "";
}

static std::vector<int> length_order(const SuffixAutomaton& sam, int max_len) {
    const auto& st = sam.states();
    std::vector<int> count(max_len + 1, 0), order(st.size());
    for (const auto& state : st) ++count[state.len];
    for (int i = 1; i <= max_len; ++i) count[i] += count[i - 1];
    for (int i = static_cast<int>(st.size()) - 1; i >= 0; --i) {
        order[--count[st[i].len]] = i;
    }
    return order;
}

static std::string solve_with_sam(const std::vector<std::string>& strings, int base_idx) {
    const std::string& base = strings[base_idx];
    if (base.empty()) return "";
    if (strings.size() == 1) return base;

    SuffixAutomaton sam(base);
    const auto& st = sam.states();
    const int state_count = static_cast<int>(st.size());
    std::vector<int> order = length_order(sam, static_cast<int>(base.size()));

    std::vector<int> common(state_count);
    for (int i = 0; i < state_count; ++i) common[i] = st[i].len;

    std::vector<int> best(state_count);
    for (std::size_t si = 0; si < strings.size(); ++si) {
        if (static_cast<int>(si) == base_idx) continue;

        std::fill(best.begin(), best.end(), 0);
        int v = 0;
        int matched = 0;

        for (unsigned char c : strings[si]) {
            int to = sam.transition(v, c);
            if (to != -1) {
                v = to;
                ++matched;
            } else {
                while (v != -1 && (to = sam.transition(v, c)) == -1) {
                    v = st[v].link;
                }
                if (v == -1) {
                    v = 0;
                    matched = 0;
                    continue;
                }
                matched = st[v].len + 1;
                v = to;
            }
            if (matched > best[v]) best[v] = matched;
        }

        for (int i = state_count - 1; i > 0; --i) {
            int node = order[i];
            int parent = st[node].link;
            if (parent >= 0) {
                int propagated = std::min(best[node], st[parent].len);
                if (propagated > best[parent]) best[parent] = propagated;
            }
        }

        for (int i = 0; i < state_count; ++i) {
            if (best[i] < common[i]) common[i] = best[i];
        }
    }

    int answer_len = 0;
    int answer_state = 0;
    for (int i = 1; i < state_count; ++i) {
        if (common[i] > answer_len) {
            answer_len = common[i];
            answer_state = i;
        }
    }

    if (answer_len == 0) return "";
    int end = st[answer_state].first_pos;
    return base.substr(static_cast<std::size_t>(end - answer_len + 1),
                       static_cast<std::size_t>(answer_len));
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    if (!(std::cin >> n)) {
        std::cout << '\n';
        return 0;
    }
    std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');

    std::vector<std::string> strings;
    strings.reserve(std::max(0, n));
    for (int i = 0; i < n; ++i) {
        std::string s;
        if (!std::getline(std::cin, s)) s.clear();
        strings.push_back(std::move(s));
    }

    if (strings.empty()) {
        std::cout << '\n';
        return 0;
    }

    int base_idx = 0;
    std::size_t total = 0;
    for (int i = 0; i < n; ++i) {
        total += strings[i].size();
        if (strings[i].size() < strings[base_idx].size()) base_idx = i;
    }

    std::string answer;
    if (strings[base_idx].empty()) {
        answer = "";
    } else if (strings[base_idx].size() <= 160 && total <= 1200) {
        answer = solve_tiny(strings, base_idx);
    } else {
        answer = solve_with_sam(strings, base_idx);
    }

    std::cout << answer << '\n';
    return 0;
}
