static vector<Plane> buildHullPlanes(const vector<Vec>& p) {
    vector<Plane> planes;
    unordered_set<string> seen;
    const int n = static_cast<int>(p.size());
    const double eps = 1e-8;

        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec normal = crossp(p[j] - p[i], p[k] - p[i]);
                double len = norm(normal);
                if (len <= eps) continue;

                double mx = -numeric_limits<double>::infinity();

    n: usize,
    start: usize,
    require_cycle: bool,
    xy: Vec<(usize, usize)>,

impl Solver {
    fn residual_degree(&self, v: usize) -> usize {
        self.adj[v].iter().filter(|&&u| !self.visited[u]).count()
    }

    fn connected_remainder(&self, cur: usize) -> bool {
        if self.path.len() == self.n {
            return true;
        }
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        seen[cur] = true;
        q.push_back(cur);
        let mut count = 0usize;
        while let Some(v) = q.pop_front() {
            if !self.visited[v] {
                count += 1;
            }
            for &u in &self.adj[v] {
                if !seen[u] && (!self.visited[u] || u == cur) {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        count == self.n - self.path.len()
    }

    fn cheap_prune(&self, cur: usize) -> bool {
        if !self.connected_remainder(cur) {
            return true;
        }
        let left = self.n - self.path.len();
        if left == 0 {
            return false;
        }

        let mut dead = 0usize;
        for v in 0..self.n {
            if self.visited[v] {
                continue;
            }
            let mut d = self.residual_degree(v);
            if self.adj[v].contains(&cur) {
                d += 1;
            }
            if d == 0 {
                return true;
            }
            if d == 1 {
                dead += 1;
            }
        }
        dead > 1
    }

    fn dfs(&mut self, cur: usize) -> bool {
        if Instant::now() >= self.deadline {
            self.timed_out = true;
            return false;
        }
        if self.path.len() == self.n {
            return !self.require_cycle || self.adj[cur].contains(&self.start);
        }
        if self.cheap_prune(cur) {
            return false;
        }

        let mut next: Vec<usize> = self.adj[cur]
            .iter()
            .copied()
            .filter(|&u| !self.visited[u])
            .collect();
        next.sort_by_key(|&u| {
            let forced = self.adj[u]
                .iter()
                .filter(|&&v| !self.visited[v] && v != cur)
                .count();
            (forced, self.residual_degree(u))
        });

            self.visited[u] = true;
            self.path.push(u);
            if self.dfs(u) {
                return true;
            }
            self.path.pop();
            self.visited[u] = false;
            if self.timed_out {
                return false;
            }
        }

    }

    static int[] SolveExactPath(double[,] c, int n)
    {
        int masks = 1 << n;
        double[,] dp = new double[masks, n];
        int[,] parent = new int[masks, n];
        for (int m = 0; m < masks; m++)
            for (int i = 0; i < n; i++)
            {
                dp[m, i] = double.PositiveInfinity;
                parent[m, i] = -1;
            }

        for (int i = 0; i < n; i++) dp[1 << i, i] = c[0, i];

        for (int mask = 1; mask < masks; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                double cur = dp[mask, last];
                if (double.IsInfinity(cur)) continue;
                int rem = ((masks - 1) ^ mask);

                {

                    int nm = mask | bit;

                    if (cand < dp[nm, nxt])
                    {
                        dp[nm, nxt] = cand;
                        parent[nm, nxt] = last;
                    }
                    rem -= bit;
                }
            }
        }

        int full = masks - 1, best = 0;
        double bestVal = dp[full, 0];
        for (int i = 1; i < n; i++)
            if (dp[full, i] < bestVal) { bestVal = dp[full, i]; best = i; }

        int[] order = new int[n];
        int mask2 = full, curLast = best;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            order[pos] = curLast;
            int prev = parent[mask2, curLast];
            mask2 ^= 1 << curLast;
            curLast = prev;
        }
        return order;
    }

struct MaxRectsBoard {
    vector<Rect> free_rects;

    MaxRectsBoard(int W, int H) {
        free_rects.push_back({0, 0, W, H});
    }

    bool find_position(const Item& it, int W, int H, Rect& out, int& rot, long long& score) const {
        (void)W;
        (void)H;
        bool found = false;
        score = numeric_limits<long long>::max();

        for (const Rect& f : free_rects) {
            for (int r = 0; r < 2; ++r) {
                int rw = r ? it.h : it.w;
                int rh = r ? it.w : it.h;
                if (rw > f.w || rh > f.h) continue;

                long long leftover = 1LL * f.w * f.h - 1LL * rw * rh;
                int short_side = min(f.w - rw, f.h - rh);
                long long cur = leftover * 1024LL + short_side;
                if (!found || cur < score) {
                    found = true;
                    score = cur;

                }

        next.reserve(free_rects.size() + 4);

        for (const Rect& f : free_rects) {

    minx: f64,

impl BBox {
    fn width(&self) -> f64 {
        self.maxx - self.minx
    }

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    cross(a, b, p).abs() <= EPS
        && p.x >= a.x.min(b.x) - EPS
        && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS
        && p.y <= a.y.max(b.y) + EPS
}

fn proper_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let c1 = cross(a, b, c);
    let c2 = cross(a, b, d);
    let c3 = cross(c, d, a);
    let c4 = cross(c, d, b);
    c1 * c2 < -EPS && c3 * c4 < -EPS
}

fn point_on_boundary(poly: &[Pt], p: Pt) -> bool {
    let n = poly.len();
    for i in 0..n {
        if on_segment(poly[i], poly[(i + 1) % n], p) {
            return true;
        }
    }
    false
}

fn point_inside_strict(poly: &[Pt], p: Pt) -> bool {
    if point_on_boundary(poly, p) {
        return false;
    }
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if (a.y > p.y) != (b.y > p.y) {
            let x = a.x + (p.y - a.y) * (b.x - a.x) / (b.y - a.y);
            if x > p.x {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_inside_or_boundary(poly: &[Pt], p: Pt) -> bool {
    point_on_boundary(poly, p) || point_inside_strict(poly, p)
}

fn bbox(poly: &[Pt]) -> BBox {
    let mut b = BBox {
        minx: f64::INFINITY,
        miny: f64::INFINITY,
        maxx: f64::NEG_INFINITY,
        maxy: f64::NEG_INFINITY,
    };
    for &p in poly {
        b.minx = b.minx.min(p.x);
        b.miny = b.miny.min(p.y);
        b.maxx = b.maxx.max(p.x);
        b.maxy = b.maxy.max(p.y);
    }
    b
}

fn rotate(p: Pt, a: f64) -> Pt {
    let (s, c) = a.sin_cos();
    Pt {
        x: p.x * c - p.y * s,
        y: p.x * s + p.y * c,
    }
}

fn transformed(poly: &[Pt], a: f64, tx: f64, ty: f64) -> Vec<Pt> {
    poly.iter()

                x: r.x + tx,
                y: r.y + ty,

        .collect()
}

fn rotated_bbox(poly: &[Pt], a: f64) -> BBox {
    let pts: Vec<Pt> = poly.iter().map(|&p| rotate(p, a)).collect();
    bbox(&pts)
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        s += a.x * b.y - a.y * b.x;
    }
    0.5 * s
}

fn polygon_centroid(poly: &[Pt]) -> Pt {
    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut a_sum = 0.0;
    for i in 0..poly.len() {
        let p = poly[i];
        let q = poly[(i + 1) % poly.len()];
        let cr = p.x * q.y - q.x * p.y;
        a_sum += cr;
        cx += (p.x + q.x) * cr;
        cy += (p.y + q.y) * cr;
    }
    if a_sum.abs() < EPS {
        let b = bbox(poly);
        return Pt {
            x: (b.minx + b.maxx) * 0.5,
            y: (b.miny + b.maxy) * 0.5,
        };
    }
    Pt {
        x: cx / (3.0 * a_sum),
        y: cy / (3.0 * a_sum),
    }
}

fn inside_stock(stock: &[Pt], poly: &[Pt]) -> bool {
    for &p in poly {
        if !point_inside_or_boundary(stock, p) {
            return false;
        }
    }
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        for j in 0..stock.len() {
            if proper_intersect(a, b, stock[j], stock[(j + 1) % stock.len()]) {
                return false;
            }
        }
    }
    true
}

fn polygons_overlap(a: &[Pt], ab: &BBox, b: &[Pt], bb: &BBox) -> bool {
    if !ab.overlaps(bb) {
        return false;
    }
    for i in 0..a.len() {
        for j in 0..b.len() {
            if proper_intersect(a[i], a[(i + 1) % a.len()], b[j], b[(j + 1) % b.len()]) {
                return true;
            }
        }
    }
    for &p in a {
        if point_inside_strict(b, p) {
            return true;
        }
    }
    for &p in b {
        if point_inside_strict(a, p) {
            return true;
        }
    }
    false
}

fn clear_of_placed(poly: &[Pt], pb: &BBox, placed: &[Placed]) -> bool {
    for q in placed {
        if polygons_overlap(poly, pb, &q.poly, &q.bbox) {
            return false;
        }

}

        int to;

    int last = 0;

    explicit SuffixAutomaton(const std::string& s) {
        st.reserve(std::max<std::size_t>(2, 2 * s.size()));
        edges.reserve(std::max<std::size_t>(8, 3 * s.size()));

            extend(c);
        }

        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) return edges[e].to;
        }
        return -1;
    }

        return st;

        edges.push_back(Edge{to, st[from].head, c});
        st[from].head = static_cast<int>(edges.size()) - 1;

        for (int e = st[from].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                edges[e].to = to;
                return;
            }
        }

            new_head = static_cast<int>(edges.size()) - 1;

        {

            if (pos == n)
            {
                for (int ci = 0; ci < cons.Count; ci++)
                    if (assignedMines[ci] != need[ci]) return;

                for (int i = 0; i < n; i++)
                    if (assignment[i] == 1) mineInSolutions[i]++;
                return;

            {

                bool ok = true;

                {

                    assignedMines[ci] += val;
                    if (assignedMines[ci] > need[ci] || assignedMines[ci] + unassigned[ci] < need[ci])
                        ok = false;
                }

                if (ok) Search(pos + 1);

                foreach (int ci in consByVar[v])
                {
                    assignedMines[ci] -= val;
                    unassigned[ci]++;
                }
                assignment[v] = -1;
            }
        }

        {

            if (mineInSolutions[i] == 0) safe.Add(cell);
            else if (mineInSolutions[i] == solutions) mines.Add(cell);

            }

    Tet {
        p: Vec3 {
            x: center_shadow.x + kx * face_z - scale * face_centroid.x,
            y: center_shadow.y + ky * face_z - scale * face_centroid.y,
            z,
        },
        q,
        scale,
    }
}

fn polygon_area(poly: &[Vec2]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        s += poly[i].x * poly[j].y - poly[j].x * poly[i].y;
    }
    s * 0.5
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("stdin");
    let mut it = input.split_whitespace();
    let nv = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    if nv == 0 {
        println!("0");
        return;
    }

    let mut poly = Vec::with_capacity(nv);
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for _ in 0..nv {
        let x = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let y = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        poly.push(Vec2 { x, y });
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }
    let sx = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let sy = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let sz = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(-1.0);
    let denom = if sz.abs() < 1.0e-9 { if sz.is_sign_negative() { -1.0e-9 } else { 1.0e-9 } } else { sz };
    let kx = sx / denom;
    let ky = sy / denom;

    let a = 0.707_f64;
    let face_a = Vec3 { x: 1.0, y: 0.0, z: -a };
    let face_b = Vec3 { x: 0.0, y: 1.0, z: a };
    let face_c = Vec3 { x: 0.0, y: -1.0, z: a };
    let normal = cross(sub(face_b, face_a), sub(face_c, face_a));
    let align = q_from_to(normal, Vec3 { x: 0.0, y: 0.0, z: 1.0 });

    let width = (max_x - min_x).max(0.0);
    let height = (max_y - min_y).max(0.0);
    let cx = (min_x + max_x) * 0.5;
    let cy = (min_y + max_y) * 0.5;
    let base_inradius = 3.0_f64.sqrt() / 3.0;
    let diag = (width * width + height * height).sqrt();
    let mut tets: Vec<Tet> = Vec::new();

    if diag <= 11.2 {
        let scale = (diag * 0.5 / base_inradius).max(0.05).min(10.0);
        tets.push(make_tet(Vec2 { x: cx, y: cy }, scale, 25.0, 0.0, kx, ky, align));
    } else {
        let area = polygon_area(&poly).abs();
        let complexity = nv as f64 + area.sqrt() * 0.02;
        let scale = if width.max(height) < 35.0 && complexity < 80.0 { 7.0 } else { 10.0 };
        let inradius = base_inradius * scale;
        let step = inradius * 1.30;
        let margin = inradius + step;
        let start_x = min_x - margin;
        let end_x = max_x + margin;
        let start_y = min_y - margin;
        let end_y = max_y + margin;
        let mut row = 0usize;
        let mut y = start_y;
        while y <= end_y {
            let offset = if row % 2 == 0 { 0.0 } else { step * 0.5 };
            let mut x = start_x + offset;
            while x <= end_x {
                let idx = tets.len() as f64;
                let z = 25.0 + idx * (scale * 4.0 + 10.0);
                let yaw = if row % 2 == 0 { 0.0 } else { PI / 3.0 };
                tets.push(make_tet(Vec2 { x, y }, scale, z, yaw, kx, ky, align));
                x += step;
            }
            y += step;
            row += 1;
        }
    }

    println!("{}", tets.len());
    for tet in tets {
        println!(
            "{:.9} {:.9} {:.9} {:.12} {:.12} {:.12} {:.12} {:.9}",
            tet.p.x, tet.p.y, tet.p.z, tet.q.w, tet.q.x, tet.q.y, tet.q.z, tet.scale
        );
    }
}
