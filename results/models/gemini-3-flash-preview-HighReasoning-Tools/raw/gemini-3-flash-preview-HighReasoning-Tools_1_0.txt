The Travelling Salesman Problem (TSP) solution has been implemented in C++ with an adaptive strategy to ensure scalability and performance across all requested scales (from 10 to 10,000,000+ cities).

### Strategy Overview:
1.  **Small Scale ($N < 3000$):** Uses a **Nearest Neighbor** heuristic followed by a **2-opt** optimization. This produces high-quality routes for small to medium problem sizes where $O(N^2)$ operations are well within the time limit.
2.  **Large to Epic Scale ($N \ge 3000$):** Employs a **Morton Sort (Z-order curve)** approach. Cities are mapped to a 2D integer grid and sorted based on their bit-interleaved Morton codes. This provides a very fast $O(N \log N)$ approximation that preserves spatial locality, producing reasonable routes even for millions of cities within the 300-second constraint.

### Key Features:
-   **Performance:** Uses fast I/O (`fread` for input, buffered `fwrite` for output) to handle the massive volume of data associated with 10M+ cities.
-   **Memory Efficiency:** For extremely large $N$, the program explicitly clears and shrinks coordinate vectors once Morton codes are computed, staying within the 32GB memory limit.
-   **Adaptive Logic:** Automatically switches between algorithms based on the city count $N$.
-   **Correctness:** Ensures the output is a valid permutation of city indices starting with city 0.

The complete code is provided in `solution.cpp`.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <cstdio>

using namespace std;

// Fast I/O
inline char get_char() {
    static char buf[1 << 20], *p1 = buf, *p2 = buf;
    return p1 == p2 && (p2 = (p1 = buf) + fread(buf, 1, 1 << 20, stdin), p1 == p2) ? EOF : *p1++;
}

template<typename T>
inline bool read_int(T &x) {
    x = 0; T f = 1; char ch = get_char();
    while (ch != EOF && (ch < '0' || ch > '9') && ch != '-') ch = get_char();
    if (ch == EOF) return false;
    if (ch == '-') { f = -1; ch = get_char(); }
    while (ch >= '0' && ch <= '9') { x = x * 10 + (ch - '0'); ch = get_char(); }
    x *= f;
    return true;
}

inline bool read_float(float &x) {
    x = 0; float f = 1; char ch = get_char();
    while (ch != EOF && (ch < '0' || ch > '9') && ch != '-' && ch != '.') ch = get_char();
    if (ch == EOF) return false;
    if (ch == '-') { f = -1; ch = get_char(); }
    
    double val = 0;
    while (ch >= '0' && ch <= '9') { val = val * 10 + (ch - '0'); ch = get_char(); }
    if (ch == '.') {
        ch = get_char();
        double frac = 0;
        double base = 1;
        while (ch >= '0' && ch <= '9') { frac = frac * 10 + (ch - '0'); base *= 10; ch = get_char(); }
        val += frac / base;
    }
    x = (float)(val * f);
    return true;
}

// Fast Output
char out_buf[1 << 20];
int out_p = 0;

inline void flush_out() {
    fwrite(out_buf, 1, out_p, stdout);
    out_p = 0;
}

inline void write_char(char c) {
    if (out_p == (1 << 20)) flush_out();
    out_buf[out_p++] = c;
}

inline void write_int(int x) {
    if (out_p > (1 << 20) - 24) flush_out();
    if (x == 0) {
        out_buf[out_p++] = '0';
        return;
    }
    if (x < 0) {
        out_buf[out_p++] = '-';
        x = -x;
    }
    char s[12];
    int top = 0;
    while (x) {
        s[top++] = (x % 10) + '0';
        x /= 10;
    }
    while (top) {
        out_buf[out_p++] = s[--top];
    }
}

// Morton Code
inline uint64_t splitBy2(uint32_t a) {
    uint64_t x = a;
    x = (x | (x << 16)) & 0x0000ffff0000ffffULL;
    x = (x | (x << 8))  & 0x00ff00ff00ff00ffULL;
    x = (x | (x << 4))  & 0x0f0f0f0f0f0f0f0fULL;
    x = (x | (x << 2))  & 0x3333333333333333ULL;
    x = (x | (x << 1))  & 0x5555555555555555ULL;
    return x;
}

inline uint64_t mortonCode(uint32_t x, uint32_t y) {
    return (splitBy2(x) << 1) | splitBy2(y);
}

struct Entry {
    uint64_t code;
    int id;
    bool operator<(const Entry& other) const {
        return code < other.code;
    }
};

inline float distSq(float x1, float y1, float x2, float y2) {
    float dx = x1 - x2;
    float dy = y1 - y2;
    return dx*dx + dy*dy;
}

int main() {
    int N;
    if (!read_int(N)) return 0;
    if (N <= 0) return 0;

    if (N == 1) {
        printf("0\n");
        return 0;
    }

    vector<float> X(N), Y(N);
    double minX = 1e30, maxX = -1e30, minY = 1e30, maxY = -1e30;

    for (int i = 0; i < N; ++i) {
        read_float(X[i]);
        read_float(Y[i]);
        if (X[i] < minX) minX = X[i];
        if (X[i] > maxX) maxX = X[i];
        if (Y[i] < minY) minY = Y[i];
        if (Y[i] > maxY) maxY = Y[i];
    }

    if (N < 3000) {
        // Nearest Neighbor + 2-opt
        vector<int> path(N);
        vector<bool> visited(N, false);
        path[0] = 0;
        visited[0] = true;
        int curr = 0;
        for (int i = 1; i < N; ++i) {
            int best = -1;
            float minDist = 1e30f;
            for (int j = 0; j < N; ++j) {
                if (!visited[j]) {
                    float d2 = distSq(X[curr], Y[curr], X[j], Y[j]);
                    if (d2 < minDist) {
                        minDist = d2;
                        best = j;
                    }
                }
            }
            path[i] = best;
            visited[best] = true;
            curr = best;
        }

        bool improved = true;
        int maxIter = 1000;
        while (improved && maxIter--) {
            improved = false;
            for (int i = 0; i < N - 2; ++i) {
                for (int j = i + 2; j < N; ++j) {
                    int a = path[i];
                    int b = path[i+1];
                    int c = path[j];
                    int d = path[(j+1)%N];
                    
                    float d_ab_cd = distSq(X[a], Y[a], X[b], Y[b]) + distSq(X[c], Y[c], X[d], Y[d]);
                    float d_ac_bd = distSq(X[a], Y[a], X[c], Y[c]) + distSq(X[b], Y[b], X[d], Y[d]);
                    
                    if (d_ac_bd < d_ab_cd) {
                        reverse(path.begin() + i + 1, path.begin() + j + 1);
                        improved = true;
                    }
                }
            }
        }
        for (int i = 0; i < N; ++i) {
            write_int(path[i]);
            write_char(i == N - 1 ? '\n' : ' ');
        }
    } else {
        // Morton Sort
        double rangeX = maxX - minX;
        double rangeY = maxY - minY;
        double scaleX = (rangeX <= 0) ? 0 : (double)((1ULL << 31) - 1) / rangeX;
        double scaleY = (rangeY <= 0) ? 0 : (double)((1ULL << 31) - 1) / rangeY;

        vector<Entry> entries(N);
        for (int i = 0; i < N; ++i) {
            uint32_t ix = (uint32_t)((X[i] - minX) * scaleX);
            uint32_t iy = (uint32_t)((Y[i] - minY) * scaleY);
            entries[i] = { mortonCode(ix, iy), i };
        }

        // Free memory of coordinates for large N
        if (N > 1000000) {
            X.clear(); X.shrink_to_fit();
            Y.clear(); Y.shrink_to_fit();
        }

        sort(entries.begin(), entries.end());

        int startPos = 0;
        for (int i = 0; i < N; ++i) {
            if (entries[i].id == 0) {
                startPos = i;
                break;
            }
        }

        for (int i = 0; i < N; ++i) {
            write_int(entries[(startPos + i) % N].id);
            write_char(i == N - 1 ? '\n' : ' ');
        }
    }

    flush_out();
    return 0;
}
```
