import math
import sys

class Customer:
    def __init__(self, id, x, y, demand):
        self.id = id
        self.x = x
        self.y = y
        self.demand = demand
        self.is_visited = False

class Depot:
    def __init__(self, x, y):
        self.x = x
        self.y = y

def calculate_distance(point1, point2):
    """Calculates the Euclidean distance between two points."""
    return math.sqrt((point1.x - point2.x)**2 + (point1.y - point2.y)**2)

def read_input():
    """Reads CVRP problem data from stdin."""
    line1 = sys.stdin.readline().split()
    N = int(line1[0])
    K = int(line1[1])
    CAPACITY = int(line1[2])

    depot_coords = sys.stdin.readline().split()
    depot = Depot(float(depot_coords[0]), float(depot_coords[1]))

    customers = []
    for i in range(N):
        customer_data = sys.stdin.readline().split()
        x = float(customer_data[0])
        y = float(customer_data[1])
        demand = int(customer_data[2])
        customers.append(Customer(i, x, y, demand))

    return N, K, CAPACITY, depot, customers

def solve_cvrp():
    N, K, CAPACITY, depot, customers = read_input()

    total_distance = 0
    routes = []
    num_routes = 0

    unvisited_customers = set(customer.id for customer in customers)

    while unvisited_customers and num_routes < K:
        num_routes += 1
        current_route = []
        current_capacity = CAPACITY
        current_location = depot
        route_distance = 0

        # Start from depot
        current_route_customer_ids = []
        
        while True:
            nearest_customer = None
            min_distance = float('inf')

            for customer_id in list(unvisited_customers):
                customer = customers[customer_id]
                if customer.demand <= current_capacity:
                    dist = calculate_distance(current_location, customer)
                    if dist < min_distance:
                        min_distance = dist
                        nearest_customer = customer

            if nearest_customer is None:
                # No more customers can be added to this route
                break
            
            # Add nearest customer to route
            current_route_customer_ids.append(nearest_customer.id)
            current_capacity -= nearest_customer.demand
            route_distance += min_distance
            current_location = nearest_customer
            unvisited_customers.remove(nearest_customer.id)

        # Return to depot from the last customer
        if current_route_customer_ids:
            route_distance += calculate_distance(current_location, depot)
            total_distance += route_distance
            # Prepend depot (0) and append depot (0) if needed for output
            # The problem asks for customer indices, so 0-based. Depot itself is not a customer.
            # So, the route should be customer indices.
            routes.append(current_route_customer_ids)
        else:
            # If a route was started but no customers were added (e.g., all remaining are too large)
            num_routes -= 1 # Don't count this empty route

    return total_distance, num_routes, routes

if __name__ == "__main__":
    total_distance, num_routes, routes = solve_cvrp()
    print(f"{total_distance}")
    print(f"{num_routes}")
    for route in routes:
        print(" ".join(map(str, route)))
