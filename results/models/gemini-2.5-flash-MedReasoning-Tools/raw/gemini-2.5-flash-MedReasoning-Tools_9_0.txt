using System;
using System.Collections.Generic;
using System.Linq;

public class Cut
{
    public int Length { get; set; }
    public int OriginalIndex { get; set; }
}

public class Stock
{
    public int RemainingCapacity { get; set; }
    public List<int> CutIndices { get; set; }

    public Stock(int stockLength)
    {
        RemainingCapacity = stockLength;
        CutIndices = new List<int>();
    }

    public void AddCut(int cutLength, int originalIndex)
    {
        RemainingCapacity -= cutLength;
        CutIndices.Add(originalIndex);
    }
}

public class Solution
{
    public static void Main(string[] args)
    {
        // Read N and stock_length
        string[] line1 = Console.ReadLine()!.Split(' ');
        int N = int.Parse(line1[0]);
        int stock_length = int.Parse(line1[1]);

        // Read cut lengths
        string[] line2 = Console.ReadLine()!.Split(' ');
        List<Cut> cuts = new List<Cut>(N);
        for (int i = 0; i < N; i++)
        {
            cuts.Add(new Cut { Length = int.Parse(line2[i]), OriginalIndex = i });
        }

        // Sort cuts in descending order of length for FFD
        // This is crucial for the First Fit Decreasing heuristic
        cuts.Sort((a, b) => b.Length.CompareTo(a.Length));

        List<Stock> stocks = new List<Stock>();

        foreach (var cut in cuts)
        {
            bool placed = false;
            // Iterate through existing stocks to find the first one that can fit the cut
            foreach (var stock in stocks)
            {
                if (stock.RemainingCapacity >= cut.Length)
                {
                    stock.AddCut(cut.Length, cut.OriginalIndex);
                    placed = true;
                    break; // Found a fit, move to the next cut
                }
            }

            if (!placed)
            {
                // No existing stock could fit the cut, create a new one
                Stock newStock = new Stock(stock_length);
                newStock.AddCut(cut.Length, cut.OriginalIndex);
                stocks.Add(newStock);
            }
        }

        // Output results
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock.CutIndices));
        }
    }
}
