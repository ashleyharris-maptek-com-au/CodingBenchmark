import sys

# Global variables for watches and trail
# watches[literal] = list of (clause_idx, other_watched_literal_in_clause)
watches = {} 
# trail: stores (var_idx, old_value) for assignments, enabling efficient backtracking.
trail = []   
# decision_level: current depth of decision tree
decision_level = -1 # Start at -1, so first decision is level 0
# assigned_vars_at_level[level] = list of (var_idx, old_value) assigned at this decision level (including propagations)
assigned_vars_at_level = [] 

# Helper function to evaluate a literal given the current assignment
# Returns True if literal evaluates to True, False if to False, None if unassigned
def evaluate_literal(literal, assignment):
    var_index = abs(literal)
    var_assignment = assignment[var_index]

    if var_assignment == 0:  # Variable is unassigned
        return None
    elif literal > 0:  # Positive literal (e.g., 5 means x5)
        return var_assignment == 1
    else:  # Negative literal (e.g., -5 means not x5)
        return var_assignment == -1

def assign_variable(var_idx, value, assignment):
    global trail, decision_level, assigned_vars_at_level
    if assignment[var_idx] == 0:
        assignment[var_idx] = value
        trail.append((var_idx, 0)) # Record that it was unassigned before
        assigned_vars_at_level[decision_level].append((var_idx, 0))
        return True
    elif assignment[var_idx] != value:
        return False # Conflict: trying to assign a different value
    return True # Already assigned correctly


def backtrack_one_level(assignment):
    global trail, decision_level, assigned_vars_at_level
    if decision_level < 0: 
        return

    # Revert assignments made at the current decision level
    # These assignments were added to the `trail` starting from the point `assigned_vars_at_level[decision_level]` began.
    # The `assigned_vars_at_level` list stores tuples (var_idx, old_value)
    # The `trail` itself stores (var_idx, old_value) but it's used to undo ALL assignments
    # made in a decision level, including those from propagation.
    
    # We need to precisely undo the assignments from the current decision level.
    # The `assigned_vars_at_level` records the variables specifically assigned at *this* level,
    # either by decision or propagation.
    
    # Iterate through assigned variables at this level and revert them.
    for var_idx, old_value in assigned_vars_at_level[decision_level]:
        assignment[var_idx] = old_value
    
    assigned_vars_at_level.pop()
    decision_level -= 1


def propagate(assigned_literal, clauses, assignment):
    # This function is now responsible for handling all cascading propagations
    # triggered by 'assigned_literal'.
    # Returns True if no conflict, False if conflict.
    
    # The literal that was just assigned is `assigned_literal`.
    # We need to check clauses watching its opposite (`-assigned_literal`).
    
    propagation_queue = [assigned_literal]

    while propagation_queue:
        current_literal = propagation_queue.pop(0)
        negated_current_literal = -current_literal
        
        # We need to iterate through a copy of the list of clauses watched by `negated_current_literal`
        # because we might modify the `watches` for `negated_current_literal` or `new_watched_literal`.
        
        # Clauses that still need to be watched by negated_current_literal after processing
        clauses_to_keep_watching = [] 

        for clause_info in list(watches.get(negated_current_literal, [])): # Iterate over a copy
            clause_idx, other_watched_literal = clause_info
            
            # 1. Check if the clause is already satisfied by the other watched literal.
            if evaluate_literal(other_watched_literal, assignment) is True:
                clauses_to_keep_watching.append(clause_info) # Keep watching this clause
                continue

            # 2. Try to find a new literal to watch in this clause
            new_watched_literal_found = False
            clause_literals = clauses[clause_idx] # Actual literals in the clause

            # Search for a replacement watch literal (must not be the current assigned_literal or other_watched_literal)
            for literal_in_clause in clause_literals:
                # We are looking for a literal that is not `negated_current_literal` (the one that became false)
                # and not `other_watched_literal` (which might be false or unassigned)
                # and is either True or Unassigned.
                if literal_in_clause != negated_current_literal and literal_in_clause != other_watched_literal:
                    lit_val = evaluate_literal(literal_in_clause, assignment)
                    if lit_val is not False: # Can be True or None (unassigned)
                        # Found a new literal to watch.
                        watches[literal_in_clause].append((clause_idx, other_watched_literal))
                        # The clause_info needs to be removed from negated_current_literal's watches.
                        # This removal will happen implicitly when we reassign watches[negated_current_literal]
                        # with clauses_to_keep_watching.
                        new_watched_literal_found = True
                        break
            
            if new_watched_literal_found:
                continue # Moved the watch, continue to next clause
            
            # If we reach here, no new literal was found to watch.
            # Now, check the `other_watched_literal`.
            # If other_watched_literal is unassigned, it becomes a unit literal.
            if evaluate_literal(other_watched_literal, assignment) is None:
                var_to_assign = abs(other_watched_literal)
                value_to_assign = 1 if other_watched_literal > 0 else -1

                if assignment[var_to_assign] == 0: # If unassigned, assign it
                    if not assign_variable(var_to_assign, value_to_assign, assignment):
                        return False # Conflict during unit propagation
                    propagation_queue.append(other_watched_literal)
                elif assignment[var_to_assign] != value_to_assign: # Already assigned a conflicting value
                    return False # Conflict
                
                # In either case (unassigned or already assigned but consistent), keep watching.
                clauses_to_keep_watching.append(clause_info) 
            else:
                # `other_watched_literal` is False, and no new literal found to watch. Conflict!
                return False
        
        # After processing all clauses watched by `negated_current_literal`, update its watch list
        watches[negated_current_literal] = clauses_to_keep_watching

    return True


def dpll(clauses, assignment, num_variables):
    global decision_level, assigned_vars_at_level

    # Check if all clauses are satisfied - this is the base case for recursion
    # This check is more efficient with 2WL. We only need to check active clauses.
    # For now, let's keep the simple iteration over all clauses, as the problem guarantees satisfiability.
    all_clauses_satisfied = True
    for clause in clauses:
        clause_satisfied = False
        for literal in clause:
            if evaluate_literal(literal, assignment) is True:
                clause_satisfied = True
                break
        if not clause_satisfied:
            all_clauses_satisfied = False
            break

    if all_clauses_satisfied:
        return True # Solution found

    # --- Decision ---
    # Find an unassigned variable to make a decision on.
    decision_var = -1
    for i in range(1, num_variables + 1):
        if assignment[i] == 0:
            decision_var = i
            break

    if decision_var == -1: # All variables assigned, but not all clauses satisfied (should have been caught above)
        return False # Should not reach here if all_clauses_satisfied was properly checked

    # Step 3: Branch (try assigning True, then False)
    
    # Store the index in the trail before making a decision.
    # This helps in backtracking efficiently.
    trail_mark = len(trail)

    # Try assigning True (1)
    decision_level += 1
    assigned_vars_at_level.append([]) # Start a new list for this decision level
    
    if assign_variable(decision_var, 1, assignment):
        # Now propagate this assignment
        if propagate(decision_var, clauses, assignment):
            if dpll(clauses, assignment, num_variables):
                return True
    
    # Backtrack if assigning True failed (or propagation failed)
    # Undo all assignments made *since* `trail_mark`
    while len(trail) > trail_mark:
        var_idx, old_value = trail.pop()
        assignment[var_idx] = old_value
    assigned_vars_at_level.pop() 
    decision_level -= 1


    # Try assigning False (-1)
    decision_level += 1
    assigned_vars_at_level.append([])

    if assign_variable(decision_var, -1, assignment):
        # Now propagate this assignment
        if propagate(-decision_var, clauses, assignment):
            if dpll(clauses, assignment, num_variables):
                return True
    
    # Backtrack if assigning False also failed (or propagation failed)
    while len(trail) > trail_mark:
        var_idx, old_value = trail.pop()
        assignment[var_idx] = old_value
    assigned_vars_at_level.pop() 
    decision_level -= 1
    
    return False # Both branches failed


def solve_3sat():
    global watches, trail, decision_level, assigned_vars_at_level

    line = sys.stdin.readline().split()
    num_variables = int(line[0])
    num_clauses = int(line[1])

    assignment = [0] * (num_variables + 1) # 0: unassigned, 1: True, -1: False
    
    clauses = []
    # Initialize watches dictionary
    watches = {i: [] for i in range(-num_variables, num_variables + 1) if i != 0}

    for clause_idx in range(num_clauses):
        clause = list(map(int, sys.stdin.readline().split()))
        clauses.append(clause)

        # Initially, make the first two literals in each clause watched.
        # This assumes clauses have at least two literals, which is true for 3-SAT.
        # If a clause has only 1 literal, it's a unit clause from the start.
        # We need to handle this separately for initial propagation.
        
        if len(clause) == 1:
            # Handle as an initial unit clause
            literal = clause[0]
            var_to_assign = abs(literal)
            value_to_assign = 1 if literal > 0 else -1
            if not assign_variable(var_to_assign, value_to_assign, assignment):
                print("UNSAT") 
                return
            if not propagate(literal, clauses, assignment):
                print("UNSAT")
                return
            
        elif len(clause) == 2: # Binary clause, watch both
            literal1 = clause[0]
            literal2 = clause[1]
            watches[literal1].append((clause_idx, literal2))
            watches[literal2].append((clause_idx, literal1))
        elif len(clause) == 3: # Ternary clause, watch first two
            literal1 = clause[0]
            literal2 = clause[1]
            watches[literal1].append((clause_idx, literal2))
            watches[literal2].append((clause_idx, literal1))
        # Clauses with more than 3 literals are not 3-SAT, but for robustness
        # one would initialize 2 watched literals. Problem states "exactly 3".

    # Initialize backtracking structures. Level 0 is for initial propagations before any decision.
    trail = []
    decision_level = -1
    assigned_vars_at_level = [] # This will be managed by assign_variable and dpll
    
    # At this point, initial unit propagations (if any) from len(clause) == 1 would have run.
    # The `propagate` function should handle further cascading effects.

    # Start the DPLL search
    # Set up decision level 0 for the first call to dpll
    decision_level = 0
    assigned_vars_at_level.append([]) # For decision level 0

    solution_found = dpll(clauses, assignment, num_variables)

    if solution_found:
        print("SAT")
        output_assignment = [1 if assignment[i] == 1 else 0 for i in range(1, num_variables + 1)]
        print(*output_assignment)
    else:
        print("UNSAT") # Should not happen based on problem statement

if __name__ == "__main__":
    solve_3sat()
