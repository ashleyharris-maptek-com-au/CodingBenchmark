import sys

def solve_exact_cover(universe_size, input_sets):
    sets_list = input_sets 

    # solution stores the indices of the sets in the exact cover
    solution = []

    # This function will attempt to find a solution recursively
    # 'current_elements_to_cover' is the set of elements that still need to be covered
    # 'current_available_set_indices' is the set of indices of sets that are still available to be chosen
    def find_solution_recursive(current_elements_to_cover, current_available_set_indices):
        if not current_elements_to_cover:
            # All elements are covered, a solution has been found
            return list(solution) # Return a copy of the solution found

        # Heuristic: Choose the element with the smallest index that still needs to be covered
        element_to_cover = min(current_elements_to_cover)

        # Iterate through the available sets that contain the element_to_cover
        # We make a copy of current_available_set_indices because we might remove elements during iteration
        for set_idx in list(current_available_set_indices): 
            current_set = sets_list[set_idx]

            # If this set covers the chosen element
            if element_to_cover in current_set:
                # 1. Add this set to the current solution
                solution.append(set_idx)

                # 2. Prepare for the next recursive call:
                #    - elements to be covered next
                #    - sets available next
                
                # Elements that will be covered by choosing current_set
                # These are elements from current_set that are also in current_elements_to_cover
                newly_covered_by_this_set = current_set.intersection(current_elements_to_cover)
                
                # Elements remaining to be covered in the next step
                next_elements_to_cover = current_elements_to_cover - newly_covered_by_this_set
                
                # Sets that become unavailable:
                #   - The chosen set itself (set_idx)
                #   - Any other available set that overlaps with current_set
                #     (i.e., shares any element with current_set, to ensure exact cover)
                sets_to_remove_for_next_step = {set_idx}
                for other_set_idx in current_available_set_indices:
                    if other_set_idx == set_idx:
                        continue
                    if not sets_list[other_set_idx].isdisjoint(current_set):
                        sets_to_remove_for_next_step.add(other_set_idx)
                
                # Available sets for the next step
                next_available_set_indices = current_available_set_indices - sets_to_remove_for_next_step

                # Recurse
                result = find_solution_recursive(next_elements_to_cover, next_available_set_indices)
                if result is not None:
                    return result
                
                # Backtrack: if the recursive call didn't find a solution, undo the choice
                solution.pop() 
        
        return None

    # Initial call to the recursive function
    initial_elements_to_cover = set(range(universe_size))
    initial_available_set_indices = set(range(len(sets_list)))

    found_solution = find_solution_recursive(initial_elements_to_cover, initial_available_set_indices)
    return found_solution


def main():
    U, S = map(int, sys.stdin.readline().split())
    
    input_sets_raw = []
    for _ in range(S):
        line = list(map(int, sys.stdin.readline().split()))
        input_sets_raw.append(set(line))
    
    solution_indices = solve_exact_cover(U, input_sets_raw)
    
    if solution_indices:
        print(" ".join(map(str, sorted(solution_indices))))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
