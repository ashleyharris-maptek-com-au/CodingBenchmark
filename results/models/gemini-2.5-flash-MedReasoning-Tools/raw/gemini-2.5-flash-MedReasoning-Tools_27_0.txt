#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm> // For std::sort

// Function to check if a color can be assigned to a vertex
bool isSafe(int v, const std::vector<std::vector<int>>& adj, const std::vector<int>& colors, int c) {
    for (int neighbor : adj[v]) {
        if (colors[neighbor] == c) {
            return false;
        }
    }
    return true;
}

// Recursive backtracking function to solve graph coloring problem
bool solveGraphColoring(int v, int num_vertices, int num_colors,
                        const std::vector<std::vector<int>>& adj, std::vector<int>& colors) {
    // Base case: All vertices are colored
    if (v == num_vertices) {
        return true;
    }

    // Try all possible colors for the current vertex v
    for (int c = 0; c < num_colors; ++c) {
        if (isSafe(v, adj, colors, c)) {
            colors[v] = c;

            // Recur for the next vertex
            if (solveGraphColoring(v + 1, num_vertices, num_colors, adj, colors)) {
                return true;
            }

            // If coloring with color c doesn't lead to a solution, backtrack
            colors[v] = -1; // Unassign color
        }
    }

    // If no color can be assigned to this vertex
    return false;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_vertices, num_edges, num_colors;
    std::cin >> num_vertices >> num_edges >> num_colors;

    std::vector<std::vector<int>> adj(num_vertices);
    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        std::cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    std::vector<int> colors(num_vertices, -1); // Initialize all colors to -1 (unassigned)

    // Attempt to color the graph
    if (solveGraphColoring(0, num_vertices, num_colors, adj, colors)) {
        for (int i = 0; i < num_vertices; ++i) {
            std::cout << colors[i] << (i == num_vertices - 1 ? "" : " ");
        }
        std::cout << std::endl;
    } else {
        // This case should ideally not happen if a coloring exists, but good for debugging
        // For this problem, if no coloring is found, output may be empty or error state
        // According to the problem, "Failure Criteria: Cannot find valid coloring within time limit"
        // So, if we reach here, it implies no solution found or timeout.
        // For now, I'll print an error message or an empty line as per common contest practices.
        // The problem description doesn't specify output for failure,
        // but typically it's an empty line or specific error code.
        // Let's assume an empty line as a default if no coloring is found.
        std::cout << std::endl;
    }

    return 0;
}
