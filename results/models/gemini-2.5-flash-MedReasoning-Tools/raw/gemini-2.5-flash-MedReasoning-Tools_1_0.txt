#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <numeric>
#include <limits> // For numeric_limits

// Structure to represent a city
struct Point {
    float x, y;
    int id; // Original index of the city
};

// Function to calculate Euclidean distance between two points
double distance(const Point& p1, const Point& p2) {
    return std::hypot(p1.x - p2.x, p1.y - p2.y);
}

// Placeholder for Nearest Neighbor algorithm
std::vector<int> nearest_neighbor(int N, const std::vector<Point>& cities) {
    std::vector<int> tour;
    tour.reserve(N);
    std::vector<bool> visited(N, false);

    int current_city_idx = 0; // Start at city 0
    tour.push_back(cities[current_city_idx].id);
    visited[current_city_idx] = true;

    for (int i = 1; i < N; ++i) {
        double min_dist = std::numeric_limits<double>::max();
        int next_city_idx = -1;

        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                double dist = distance(cities[current_city_idx], cities[j]);
                if (dist < min_dist) {
                    min_dist = dist;
                    next_city_idx = j;
                }
            }
        }
        tour.push_back(cities[next_city_idx].id);
        visited[next_city_idx] = true;
        current_city_idx = next_city_idx;
    }
    return tour;
}

// Helper function to calculate the total length of a tour
double calculate_tour_length(int N, const std::vector<Point>& cities, const std::vector<int>& tour) {
    double total_dist = 0.0;
    for (int i = 0; i < N; ++i) {
        total_dist += distance(cities[tour[i]], cities[tour[(i + 1) % N]]);
    }
    return total_dist;
}

// 2-Opt local search
std::vector<int> two_opt(int N, const std::vector<Point>& cities, std::vector<int> current_tour) {
    std::vector<int> best_tour = current_tour;
    double best_dist = calculate_tour_length(N, cities, best_tour);

    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N - 1; ++i) { // First edge: (tour[i], tour[i+1])
            for (int k = i + 2; k < N; ++k) { // Second edge: (tour[k], tour[k+1])
                                            // Ensure k is not adjacent to i. k+1 will wrap around for last city.

                // Get the actual point objects using their IDs
                // (tour[i], tour[i+1]) and (tour[k], tour[k+1]) are the edges to swap
                Point p_i = cities[best_tour[i]];
                Point p_i_plus_1 = cities[best_tour[(i + 1) % N]];
                Point p_k = cities[best_tour[k]];
                Point p_k_plus_1 = cities[best_tour[(k + 1) % N]];

                // Calculate old segment distances
                double old_segment_dist = distance(p_i, p_i_plus_1) + distance(p_k, p_k_plus_1);

                // Calculate new segment distances after reversal
                // The new edges are (tour[i], tour[k]) and (tour[i+1], tour[k+1])
                double new_segment_dist = distance(p_i, p_k) + distance(p_i_plus_1, p_k_plus_1);

                if (new_segment_dist < old_segment_dist) {
                    std::reverse(best_tour.begin() + i + 1, best_tour.begin() + k + 1);
                    best_dist = calculate_tour_length(N, cities, best_tour); // Recalculate full length
                    improved = true;
                    // Restart search from beginning to allow further improvements
                    i = 0; // Reset i to re-scan for improvements
                    break; // Break inner loop
                }
            }
        }
    }
    return best_tour;
}


// Placeholder for Angle-Sweep algorithm
std::vector<int> angle_sweep(int N, const std::vector<Point>& cities) {
    if (N <= 1) return {0};
    if (N == 2) return {0, 1};

    // We use city 0 as the reference point for angles
    Point ref_point = cities[0];

    // Create a vector of pairs: {angle, original_index} for cities other than city 0
    std::vector<std::pair<double, int>> angled_cities;
    for (int i = 1; i < N; ++i) {
        double angle = std::atan2(cities[i].y - ref_point.y, cities[i].x - ref_point.x);
        angled_cities.push_back({angle, cities[i].id});
    }

    // Sort cities based on their angle
    std::sort(angled_cities.begin(), angled_cities.end());

    std::vector<int> tour;
    tour.reserve(N);
    tour.push_back(ref_point.id); // Start with city 0

    for (const auto& ac : angled_cities) {
        tour.push_back(ac.second); // Add sorted cities to the tour
    }

    return tour;
}


int main() {
    // Fast I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N;
    std::cin >> N;

    std::vector<Point> cities(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
        cities[i].id = i;
    }

    // Handle trivial cases
    if (N == 0) {
        return 0; // No cities, no output
    }
    if (N == 1) {
        std::cout << "0" << std::endl;
        return 0;
    }

    std::vector<int> tour;

    if (N <= 1000) { // Small to Medium scale: Nearest Neighbor + 2-Opt
        tour = nearest_neighbor(N, cities);
        tour = two_opt(N, cities, tour);
    } else { // Large, Extreme, Epic scale: Angle-Sweep Algorithm
        tour = angle_sweep(N, cities);
    }

    // Print the tour
    for (int i = 0; i < N; ++i) {
        std::cout << tour[i] << (i == N - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
