#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>

// Global variables to store problem data and best solution found
int n_vars;
int m_constraints;
std::vector<int> c_coeffs;
std::vector<std::vector<bool>> adj_matrix; // Adjacency matrix for constraints

long long best_objective_value = -1;
std::vector<int> best_assignment;

// Recursive backtracking function with pruning
// k: current variable index to assign (from 0 to n_vars-1)
// current_assignment: the assignment for variables x_0 to x_{k-1}
// current_objective_value: the objective value for current_assignment
void solve(int k, std::vector<int>& current_assignment, long long current_objective_value) {
    // Base case: all variables assigned
    if (k == n_vars) {
        if (current_objective_value > best_objective_value) {
            best_objective_value = current_objective_value;
            best_assignment = current_assignment;
        }
        return;
    }

    // Pruning: if current objective + max possible remaining is less than best found,
    // then this branch cannot lead to a better solution.
    // A simple upper bound: sum of all remaining positive coefficients.
    long long remaining_potential_max_value = 0;
    for (int i = k; i < n_vars; ++i) {
        if (c_coeffs[i] > 0) {
            remaining_potential_max_value += c_coeffs[i];
        }
    }
    if (current_objective_value + remaining_potential_max_value <= best_objective_value && best_objective_value != -1) {
        return;
    }

    // Option 1: Set x_k = 0
    current_assignment[k] = 0;
    solve(k + 1, current_assignment, current_objective_value);

    // Option 2: Set x_k = 1
    // Check feasibility with previously assigned variables (x_0 to x_{k-1})
    bool feasible = true;
    for (int i = 0; i < k; ++i) {
        if (current_assignment[i] == 1 && adj_matrix[i][k]) {
            feasible = false;
            break;
        }
    }

    if (feasible) {
        current_assignment[k] = 1;
        solve(k + 1, current_assignment, current_objective_value + c_coeffs[k]);
    }
    // Backtrack (no need to explicitly undo current_assignment[k]=1 as it's overwritten by x_k=0 in next branch or function returns)
    // The current_assignment[k]=0 for Option 1 call is implicitly handled by the next iteration of the loop or return.
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    std::cin >> n_vars >> m_constraints;

    c_coeffs.resize(n_vars);
    for (int i = 0; i < n_vars; ++i) {
        std::cin >> c_coeffs[i];
    }

    adj_matrix.resize(n_vars, std::vector<bool>(n_vars, false));
    for (int i = 0; i < m_constraints; ++i) {
        int u, v;
        std::cin >> u >> v;
        adj_matrix[u][v] = true;
        adj_matrix[v][u] = true; // Constraints are symmetric
    }

    std::vector<int> current_assignment(n_vars);
    best_assignment.resize(n_vars);

    solve(0, current_assignment, 0);

    std::cout << best_objective_value << std::endl;
    for (int i = 0; i < n_vars; ++i) {
        std::cout << best_assignment[i] << (i == n_vars - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
