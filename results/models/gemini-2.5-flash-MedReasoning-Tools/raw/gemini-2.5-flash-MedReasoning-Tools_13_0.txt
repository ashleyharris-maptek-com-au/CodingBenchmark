#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

// Function to find the longest common substring of two strings
std::string longestCommonSubstring(const std::string& s1, const std::string& s2) {
    int m = s1.length();
    int n = s2.length();

    // Create a 2D DP table to store the lengths of common suffixes
    // dp[i][j] stores the length of the longest common suffix of s1[0...i-1] and s2[0...j-1]
    std::vector<std::vector<int>> dp(m + 1, std::vector<int>(n + 1, 0));

    int maxLength = 0;      // Stores the length of the longest common substring
    int endIndex = 0;       // Stores the ending index of the longest common substring in s1

    // Fill the dp table
    for (int i = 1; i <= m; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (s1[i - 1] == s2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
                if (dp[i][j] > maxLength) {
                    maxLength = dp[i][j];
                    endIndex = i - 1; // Current ending index in s1
                }
            } else {
                dp[i][j] = 0; // Reset if characters do not match
            }
        }
    }

    // If no common substring is found, return an empty string
    if (maxLength == 0) {
        return "";
    }

    // Extract the longest common substring using the maxLength and endIndex
    return s1.substr(endIndex - maxLength + 1, maxLength);
}

int main() {
    std::string s1, s2;

    // Get input strings from the user
    std::cout << "Enter the first string: ";
    std::cin >> s1;
    std::cout << "Enter the second string: ";
    std::cin >> s2;

    // Find and print the longest common substring
    std::string lcs = longestCommonSubstring(s1, s2);
    std::cout << "Longest Common Substring: " << lcs << std::endl;

    return 0;
}
