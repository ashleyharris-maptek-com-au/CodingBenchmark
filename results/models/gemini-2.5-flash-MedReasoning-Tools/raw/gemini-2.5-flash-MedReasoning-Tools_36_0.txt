#include <iostream>
#include <vector>
#include <algorithm>

// Function to find the maximum independent set using backtracking
void findMaxIndependentSet(
    const std::vector<std::vector<int>>& graph,
    std::vector<int>& current_set,
    std::vector<int>& max_set,
    std::vector<bool>& is_in_set,
    int vertex_idx
) {
    // If the current set is larger than the maximum set found so far, update max_set
    if (current_set.size() > max_set.size()) {
        max_set = current_set;
    }

    // Iterate through the remaining vertices
    for (int i = vertex_idx; i < graph.size(); ++i) {
        bool can_add = true;
        // Check if the current vertex can be added to the independent set
        // It can be added if it has no edges to any vertex already in current_set
        for (int node_in_set : current_set) {
            for (int neighbor : graph[i]) {
                if (neighbor == node_in_set) {
                    can_add = false;
                    break;
                }
            }
            if (!can_add) {
                break;
            }
        }

        if (can_add) {
            current_set.push_back(i);
            is_in_set[i] = true;
            findMaxIndependentSet(graph, current_set, max_set, is_in_set, i + 1);
            current_set.pop_back(); // Backtrack
            is_in_set[i] = false;
        }
    }
}

int main() {
    // Example graph (adjacency list representation)
    // 0 -- 1
    // |  / |
    // 2 -- 3
    // |  /
    // 4
    std::vector<std::vector<int>> graph = {
        {1, 2},      // Neighbors of vertex 0
        {0, 2, 3},   // Neighbors of vertex 1
        {0, 1, 3, 4},// Neighbors of vertex 2
        {1, 2, 4},   // Neighbors of vertex 3
        {2, 3}       // Neighbors of vertex 4
    };

    std::vector<int> current_set;
    std::vector<int> max_set;
    std::vector<bool> is_in_set(graph.size(), false);

    findMaxIndependentSet(graph, current_set, max_set, is_in_set, 0);

    std::cout << "Maximum Independent Set: { ";
    for (int vertex : max_set) {
        std::cout << vertex << " ";
    }
    std::cout << "}" << std::endl;

    return 0;
}
