using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public struct Voxel
{
    public int X;
    public int Y;
    public int Z;
    public int MassInTonnes;
    public int ValuePerTonne;

    public Voxel(int x, int y, int z, int mass, int value)
    {
        X = x;
        Y = y;
        Z = z;
        MassInTonnes = mass;
        ValuePerTonne = value;
    }

    public bool IsValuableOre => ValuePerTonne > 0;
}

public class World
{
    public int XVoxelCount { get; private set; }
    public int YVoxelCount { get; private set; }
    public int ZVoxelCount { get; private set; }
    public decimal XBlockSize { get; private set; }
    public decimal YBlockSize { get; private set; }
    public decimal ZBlockSize { get; private set; }

    // Using a dictionary for sparse voxel storage
    private Dictionary<(int x, int y, int z), Voxel> _voxels;
    
    // Height map to track the highest solid voxel at each (x,y) coordinate
    // Initialized to -1 to indicate empty space above ground, or actual Z of highest voxel
    private int[,] _heightMap; 

    public World(int xCount, int yCount, int zCount, decimal xBlock, decimal yBlock, decimal zBlock)
    {
        XVoxelCount = xCount;
        YVoxelCount = yCount;
        ZVoxelCount = zCount;
        XBlockSize = xBlock;
        YBlockSize = yBlock;
        ZBlockSize = zBlock;

        _voxels = new Dictionary<(int x, int y, int z), Voxel>();
        _heightMap = new int[xCount, yCount];
        // Initialize height map to lowest possible Z, assuming ground starts at Z=0
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                _heightMap[x, y] = -1; // Represents air/empty space
            }
        }
    }

    public void AddVoxel(Voxel voxel)
    {
        _voxels[(voxel.X, voxel.Y, voxel.Z)] = voxel;
        // Update height map if this voxel is higher than the current recorded height
        if (voxel.Z > _heightMap[voxel.X, voxel.Y])
        {
            _heightMap[voxel.X, voxel.Y] = voxel.Z;
        }
    }

    public Voxel? GetVoxel(int x, int y, int z)
    {
        if (_voxels.TryGetValue((x, y, z), out Voxel voxel))
        {
            return voxel;
        }
        return null;
    }

    public int GetGroundLevel(int x, int y)
    {
        if (x < 0 || x >= XVoxelCount || y < 0 || y >= YVoxelCount)
        {
            return -1; // Outside bounds
        }
        return _heightMap[x, y];
    }
    
    // Checks if a voxel is exposed from above or a side
    public bool IsExposed(int x, int y, int z)
    {
        // Check if there's air above it
        if (GetVoxel(x, y, z + 1) == null) return true;

        // Check if any side is exposed to air
        if (GetVoxel(x + 1, y, z) == null) return true;
        if (GetVoxel(x - 1, y, z) == null) return true;
        if (GetVoxel(x, y + 1, z) == null) return true;
        if (GetVoxel(x, y - 1, z) == null) return true;

        return false;
    }

    public bool TryDigVoxel(int x, int y, int z, out Voxel dugVoxel)
    {
        if (_voxels.TryGetValue((x, y, z), out dugVoxel) && IsExposed(x, y, z))
        {
            _voxels.Remove((x, y, z));
            UpdateHeightMapAfterDig(x, y, z); // Recalculate height map for this column
            return true;
        }
        dugVoxel = default;
        return false;
    }

    private void UpdateHeightMapAfterDig(int x, int y, int zDug)
    {
        // Find the new highest voxel in this column (x, y)
        int newMaxZ = -1;
        for (int z = ZVoxelCount - 1; z >= 0; z--) // Iterate downwards from max Z
        {
            if (_voxels.ContainsKey((x, y, z)))
            {
                newMaxZ = z;
                break;
            }
        }
        _heightMap[x, y] = newMaxZ;
    }

    public void DumpVoxel(Voxel voxel, int destX, int destY)
    {
        if (destX == 0 && destY == 0) // Processing plant
        {
            // Voxel ceases to exist, nothing to add to the world
        }
        else
        {
            // Dump on top of the ground
            int newZ = GetGroundLevel(destX, destY) + 1;
            // Ensure newZ is within world bounds, though problem implies it just adds on top
            if (newZ < ZVoxelCount)
            {
                Voxel dumpedVoxel = new Voxel(destX, destY, newZ, voxel.MassInTonnes, voxel.ValuePerTonne);
                _voxels[(dumpedVoxel.X, dumpedVoxel.Y, dumpedVoxel.Z)] = dumpedVoxel;
                _heightMap[destX, destY] = newZ; // Update height map
            }
            else
            {
                // This case implies dumping outside the valid Z-range if world max Z is strict.
                // For now, assuming world expands upwards or newZ won't exceed ZVoxelCount significantly.
                // Based on "raises the ground by 1 voxel" and ZVoxelCount, it might implicitly expand.
                // For this problem, I'll assume we can't dump outside ZVoxelCount.
                // If it can expand, need to re-evaluate ZVoxelCount as a strict limit.
                // For now, let's treat ZVoxelCount as a hard upper bound.
                Console.Error.WriteLine($"Warning: Attempted to dump voxel at ({destX}, {destY}, {newZ}) which is outside ZVoxelCount limit.");
            }
        }
    }

    // A* Pathfinding for movement on the X/Y plane, considering Z-slope constraint
    public (List<(int x, int y)> path, decimal cost) FindPath(Voxel startVoxel, int targetX, int targetY)
    {
        var start = (startVoxel.X, startVoxel.Y);
        var target = (targetX, targetY);

        var openSet = new PriorityQueue<(int x, int y), decimal>();
        openSet.Enqueue(start, 0);

        var cameFrom = new Dictionary<(int x, int y), (int x, int y)>();
        var gScore = new Dictionary<(int x, int y), decimal>(); // Cost from start to current node
        gScore[start] = 0;

        var fScore = new Dictionary<(int x, int y), decimal>(); // gScore + heuristic
        fScore[start] = Heuristic(start, target, startVoxel.MassInTonnes);

        int[] dx = { 0, 0, 1, -1 };
        int[] dy = { 1, -1, 0, 0 };

        while (openSet.Count > 0)
        {
            var current = openSet.Dequeue();

            if (current == target)
            {
                return ReconstructPath(cameFrom, current, startVoxel.MassInTonnes);
            }

            int currentZ = GetGroundLevel(current.x, current.y);

            for (int i = 0; i < 4; i++)
            {
                int neighborX = current.x + dx[i];
                int neighborY = current.y + dy[i];
                var neighbor = (neighborX, neighborY);

                if (neighborX < 0 || neighborX >= XVoxelCount || neighborY < 0 || neighborY >= YVoxelCount)
                {
                    continue; // Out of bounds
                }
                
                // Get the Z-level of the *surface* at the neighbor location
                int neighborZ = GetGroundLevel(neighborX, neighborY);

                // Calculate movement cost
                decimal movementCost = 0;
                int dz = neighborZ - currentZ;

                if (dz > 1) // Too steep uphill
                {
                    continue; 
                }
                else if (dz == 1) // Uphill
                {
                    movementCost = startVoxel.MassInTonnes * 5;
                }
                else // Flat or downhill
                {
                    movementCost = startVoxel.MassInTonnes * 1; 
                }

                decimal tentative_gScore = gScore.GetValueOrDefault(current, decimal.MaxValue) + movementCost;

                if (tentative_gScore < gScore.GetValueOrDefault(neighbor, decimal.MaxValue))
                {
                    cameFrom[neighbor] = current;
                    gScore[neighbor] = tentative_gScore;
                    fScore[neighbor] = tentative_gScore + Heuristic(neighbor, target, startVoxel.MassInTonnes);
                    openSet.Enqueue(neighbor, fScore[neighbor]);
                }
            }
        }

        return (new List<(int x, int y)>(), decimal.MaxValue); // No path found
    }

    private decimal Heuristic((int x, int y) current, (int x, int y) target, int mass)
    {
        // Manhattan distance heuristic, multiplied by flat movement cost
        return (Math.Abs(current.x - target.x) + Math.Abs(current.y - target.y)) * mass;
    }

    private (List<(int x, int y)> path, decimal cost) ReconstructPath(Dictionary<(int x, int y), (int x, int y)> cameFrom, (int x, int y) current, int mass)
    {
        List<(int x, int y)> totalPath = new List<(int x, int y)>();
        decimal totalCost = 0;

        while (cameFrom.ContainsKey(current))
        {
            (int prevX, int prevY) = cameFrom[current];
            
            // Calculate cost for this step (from prev to current)
            int currentZ = GetGroundLevel(current.x, current.y);
            int prevZ = GetGroundLevel(prevX, prevY);
            int dz = currentZ - prevZ;
            
            decimal stepCost = 0;
            if (dz == 1) // Uphill
            {
                stepCost = mass * 5;
            }
            else // Flat or downhill
            {
                stepCost = mass * 1;
            }
            totalCost += stepCost;

            totalPath.Add(current);
            current = (prevX, prevY);
        }
        totalPath.Add(current); // Add the starting point
        totalPath.Reverse(); // Path from start to target

        return (totalPath, totalCost);
    }

    // Helper method to find a good dump spot for non-valuable voxels
    public (int x, int y) FindBestDumpSpotForNonValuable(int initialX, int initialY)
    {
        // For simplicity, let's try to dump it at the lowest adjacent spot.
        // This is a basic strategy. A more advanced one would consider long-term impact.

        int bestX = initialX;
        int bestY = initialY;
        int minZ = GetGroundLevel(initialX, initialY);

        int[] dx = { 0, 0, 1, -1 };
        int[] dy = { 1, -1, 0, 0 };

        for (int i = 0; i < 4; i++)
        {
            int neighborX = initialX + dx[i];
            int neighborY = initialY + dy[i];

            if (neighborX >= 0 && neighborX < XVoxelCount && neighborY >= 0 && neighborY < YVoxelCount)
            {
                int neighborZ = GetGroundLevel(neighborX, neighborY);
                if (neighborZ < minZ)
                {
                    minZ = neighborZ;
                    bestX = neighborX;
                    bestY = neighborY;
                }
            }
        }
        return (bestX, bestY);
    }
}
// A more efficient PriorityQueue implementation using a min-heap
public class PriorityQueue<TItem, TPriority> where TPriority : IComparable<TPriority>
{
    private List<(TItem item, TPriority priority)> _elements = new List<(TItem item, TPriority priority)>();

    public int Count => _elements.Count;

    public void Enqueue(TItem item, TPriority priority)
    {
        _elements.Add((item, priority));
        int childIndex = _elements.Count - 1;
        while (childIndex > 0)
        {
            int parentIndex = (childIndex - 1) / 2;
            if (_elements[childIndex].priority.CompareTo(_elements[parentIndex].priority) < 0)
            {
                Swap(childIndex, parentIndex);
                childIndex = parentIndex;
            }
            else
            {
                break;
            }
        }
    }

    public TItem Dequeue()
    {
        if (_elements.Count == 0)
        {
            throw new InvalidOperationException("Queue is empty.");
        }

        TItem rootItem = _elements[0].item;
        _elements[0] = _elements[_elements.Count - 1];
        _elements.RemoveAt(_elements.Count - 1);

        int parentIndex = 0;
        while (true)
        {
            int leftChildIndex = 2 * parentIndex + 1;
            int rightChildIndex = 2 * parentIndex + 2;
            int smallestIndex = parentIndex;

            if (leftChildIndex < _elements.Count && _elements[leftChildIndex].priority.CompareTo(_elements[smallestIndex].priority) < 0)
            {
                smallestIndex = leftChildIndex;
            }

            if (rightChildIndex < _elements.Count && _elements[rightChildIndex].priority.CompareTo(_elements[smallestIndex].priority) < 0)
            {
                smallestIndex = rightChildIndex;
            }

            if (smallestIndex != parentIndex)
            {
                Swap(parentIndex, smallestIndex);
                parentIndex = smallestIndex;
            }
            else
            {
                break;
            }
        }
        return rootItem;
    }

    private void Swap(int index1, int index2)
    {
        var temp = _elements[index1];
        _elements[index1] = _elements[index2];
        _elements[index2] = temp;
    }
}
