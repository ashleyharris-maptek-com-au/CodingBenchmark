using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class MiningSolver
{
    private struct Voxel
    {
        public byte Mass; // 0-15
        public byte Value; // 0-15
        public bool IsDug;
        
        public Voxel(byte data)
        {
            Mass = (byte)(data % 16);
            Value = (byte)(data / 16);
            IsDug = false;
        }
        
        public int Profit => Value * 1000;
        public int Cost => Mass * 1000; // damage if junk dumped at plant
        public int NetProfit => Value > 0 ? Profit : -Cost;
    }
    
    private struct Pos
    {
        public int X, Y, Z;
        public Pos(int x, int y, int z) { X = x; Y = y; Z = z; }
        
        public override int GetHashCode() => X * 10007 + Y * 17 + Z;
        public override bool Equals(object obj) => obj is Pos p && p.X == X && p.Y == Y && p.Z == Z;
        
        public Pos Move(int dx, int dy, int dz) => new Pos(X + dx, Y + dy, Z + dz);
    }
    
    private readonly Voxel[,,] world;
    private readonly int XV, YV, ZV;
    private readonly double XB, YB, ZB;
    private readonly List<string> operations = new List<string>();
    private readonly HashSet<Pos> obstacles = new HashSet<Pos>();
    
    public MiningSolver(int xv, int yv, int zv, double xb, double yb, double zb, byte[] data)
    {
        XV = xv; YV = yv; ZV = zv;
        XB = xb; YB = yb; ZB = zb;
        
        world = new Voxel[xv, yv, zv];
        int idx = 0;
        for (int x = 0; x < xv; x++)
            for (int y = 0; y < yv; y++)
                for (int z = 0; z < zv; z++)
                    world[x, y, z] = new Voxel(data[idx++]);
    }
    
    public static void Main()
    {
        string header = Console.In.ReadLine();
        var parts = header.Split(' ');
        int xv = int.Parse(parts[0]), yv = int.Parse(parts[1]), zv = int.Parse(parts[2]);
        double xb = double.Parse(parts[3]), yb = double.Parse(parts[4]), zb = double.Parse(parts[5]);
        
        using var stdin = Console.OpenStandardInput();
        byte[] data = new byte[xv * yv * zv];
        stdin.Read(data, 0, data.Length);
        
        var solver = new MiningSolver(xv, yv, zv, xb, yb, zb, data);
        solver.Solve();
        
        Console.WriteLine(solver.operations.Count);
        foreach (var op in solver.operations)
            Console.WriteLine(op);
    }
    
    private void Solve()
    {
        bool isSmall = XV * YV * ZV < 1000000;
        
        if (isSmall)
            SolveSmall();
        else
            SolveLarge();
    }
    
    private void SolveSmall()
    {
        // Exhaustive search for small problems
        var valuable = new List<Pos>();
        
        for (int x = 0; x < XV; x++)
            for (int y = 0; y < YV; y++)
                for (int z = 0; z < ZV; z++)
                    if (world[x, y, z].Value > 0 && !world[x, y, z].IsDug)
                        valuable.Add(new Pos(x, y, z));
        
        // Sort by net profit descending
        valuable.Sort((a, b) => world[b.X, b.Y, b.Z].NetProfit.CompareTo(world[a.X, a.Y, a.Z].NetProfit));
        
        foreach (var pos in valuable)
        {
            if (CanReachPlant(pos))
            {
                DigAndMoveToPlant(pos);
            }
        }
    }
    
    private void SolveLarge()
    {
        // Greedy strategy for large problems: scan in layers from surface down
        for (int z = ZV - 1; z >= 0; z--)
        {
            for (int x = 0; x < XV; x++)
            {
                for (int y = 0; y < YV; y++)
                {
                    var pos = new Pos(x, y, z);
                    ProcessVoxel(pos);
                }
            }
        }
    }
    
    private void ProcessVoxel(Pos pos)
    {
        if (world[pos.X, pos.Y, pos.Z].IsDug || world[pos.X, pos.Y, pos.Z].Mass == 0)
            return;
            
        int netProfit = world[pos.X, pos.Y, pos.Z].NetProfit;
        
        if (netProfit > 0 && CanReachPlant(pos))
        {
            DigAndMoveToPlant(pos);
        }
        else if (netProfit < 0)
        {
            // Junk: dump far away in corner
            DumpToCorner(pos);
        }
    }
    
    private bool CanReachPlant(Pos start)
    {
        // Simple connectivity check: can we path to (0,0) without steep slopes?
        var queue = new Queue<Pos>();
        var visited = new HashSet<Pos>();
        queue.Enqueue(start);
        visited.Add(start);
        
        while (queue.Count > 0)
        {
            var curr = queue.Dequeue();
            if (curr.X == 0 && curr.Y == 0)
                return true;
                
            foreach (var dir in new[] { new Pos(0,0,0), new Pos(1,0,0), new Pos(-1,0,0), new Pos(0,1,0), new Pos(0,-1,0) })
            {
                var next = curr.Move(dir.X, dir.Y, dir.Z);
                if (IsValid(next) && !visited.Contains(next) && 
                    Math.Abs(next.Z - curr.Z) <= Math.Abs(dir.X) + Math.Abs(dir.Y))
                {
                    queue.Enqueue(next);
                    visited.Add(next);
                }
            }
        }
        return false;
    }
    
    private bool IsValid(Pos p)
    {
        return p.X >= 0 && p.X < XV && p.Y >= 0 && p.Y < YV && p.Z >= 0 && p.Z < ZV;
    }
    
    private void DigAndMoveToPlant(Pos pos)
    {
        var path = FindPath(pos, new Pos(0, 0, pos.Z));
        if (path == null) return;
        
        var steps = new List<int>();
        for (int i = 1; i < path.Count; i++)
        {
            steps.Add(path[i].X);
            steps.Add(path[i].Y);
        }
        
        var voxel = world[pos.X, pos.Y, pos.Z];
        string op = $"{steps.Count/2} {pos.X} {pos.Y} {pos.Z} " + 
                   string.Join(" ", steps) + $" 0 0";
        
        operations.Add(op);
        world[pos.X, pos.Y, pos.Z].IsDug = true;
        obstacles.Remove(pos);
    }
    
    private void DumpToCorner(Pos pos)
    {
        // Dump junk to far corner (XV-1, YV-1, surface)
        int targetZ = ZV - 1;
        var path = FindPath(pos, new Pos(XV - 1, YV - 1, targetZ));
        if (path == null) return;
        
        var steps = new List<int>();
        for (int i = 1; i < path.Count; i++)
        {
            steps.Add(path[i].X);
            steps.Add(path[i].Y);
        }
        
        string op = $"{steps.Count/2} {pos.X} {pos.Y} {pos.Z} " + 
                   string.Join(" ", steps) + $" {XV-1} {YV-1}";
        
        operations.Add(op);
        world[pos.X, pos.Y, pos.Z].IsDug = true;
    }
    
    private List<Pos> FindPath(Pos start, Pos goal)
    {
        var open = new PriorityQueue<Pos, int>();
        var cameFrom = new Dictionary<Pos, Pos>();
        var gScore = new Dictionary<Pos, int>();
        
        gScore[start] = 0;
        open.Enqueue(start, 0);
        
        while (open.Count > 0)
        {
            var current = open.Dequeue();
            
            if (current.X == goal.X && current.Y == goal.Y)
            {
                return ReconstructPath(cameFrom, current);
            }
            
            foreach (var dir in GetNeighbors(current))
            {
                int tentativeG = gScore[current] + Cost(current, dir);
                
                if (!gScore.TryGetValue(dir, out int existingG) || tentativeG < existingG)
                {
                    cameFrom[dir] = current;
                    gScore[dir] = tentativeG;
                    int fScore = tentativeG + Heuristic(dir, goal);
                    open.Enqueue(dir, fScore);
                }
            }
        }
        return null;
    }
    
    private List<Pos> GetNeighbors(Pos p)
    {
        var neighbors = new List<Pos>();
        for (int dx = -1; dx <= 1; dx++)
        for (int dy = -1; dy <= 1; dy++)
        {
            int dzMax = Math.Abs(dx) + Math.Abs(dy); // 1:1 slope max
            for (int dz = -dzMax; dz <= dzMax; dz++)
            {
                var n = p.Move(dx, dy, dz);
                if (IsValid(n) && !obstacles.Contains(n))
                    neighbors.Add(n);
            }
        }
        return neighbors;
    }
    
    private int Cost(Pos from, Pos to)
    {
        int dx = Math.Abs(to.X - from.X);
        int dy = Math.Abs(to.Y - from.Y);
        int dz = to.Z - from.Z;
        
        int flatCost = (dx + dy) * 1;
        int slopeCost = Math.Max(0, dz) * 5; // uphill only
        
        return flatCost + slopeCost;
    }
    
    private int Heuristic(Pos a, Pos b)
    {
        int dx = Math.Abs(a.X - b.X);
        int dy = Math.Abs(a.Y - b.Y);
        int dz = Math.Abs(a.Z - b.Z);
        return (dx + dy) * 1 + dz * 5;
    }
    
    private List<Pos> ReconstructPath(Dictionary<Pos, Pos> cameFrom, Pos current)
    {
        var path = new List<Pos> { current };
        while (cameFrom.TryGetValue(current, out var prev))
        {
            current = prev;
            path.Add(current);
        }
        path.Reverse();
        return path;
    }
}

public class PriorityQueue<TK, TPriority> where TPriority : IComparable<TPriority>
{
    private readonly List<(TK key, TPriority priority)> elements = new();
    
    public int Count => elements.Count;
    
    public void Enqueue(TK key, TPriority priority)
    {
        elements.Add((key, priority));
        int i = elements.Count - 1;
        while (i > 0)
        {
            int parent = (i - 1) / 2;
            if (elements[parent].priority.CompareTo(elements[i].priority) <= 0)
                break;
            (elements[i], elements[parent]) = (elements[parent], elements[i]);
            i = parent;
        }
    }
    
    public TK Dequeue()
    {
        if (elements.Count == 0) throw new InvalidOperationException();
        
        var result = elements[0].key;
        elements[0] = elements[^1];
        elements.RemoveAt(elements.Count - 1);
        
        int i = 0;
        while (true)
        {
            int left = 2 * i + 1;
            int right = 2 * i + 2;
            int smallest = i;
            
            if (left < elements.Count && elements[left].priority.CompareTo(elements[smallest].priority) < 0)
                smallest = left;
            if (right < elements.Count && elements[right].priority.CompareTo(elements[smallest].priority) < 0)
                smallest = right;
                
            if (smallest == i) break;
            
            (elements[i], elements[smallest]) = (elements[smallest], elements[i]);
            i = smallest;
        }
        
        return result;
    }
}
