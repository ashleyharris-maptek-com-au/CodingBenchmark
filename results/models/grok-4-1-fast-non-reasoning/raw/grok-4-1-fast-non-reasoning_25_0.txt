using System;

public struct Vector3D
{
    public double X, Y, Z;
    
    public Vector3D(double x, double y, double z)
    {
        X = x; Y = y; Z = z;
    }
    
    public static Vector3D operator +(Vector3D a, Vector3D b) => new Vector3D(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3D operator -(Vector3D a, Vector3D b) => new Vector3D(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3D operator *(Vector3D a, double s) => new Vector3D(a.X * s, a.Y * s, a.Z * s);
    public static double Dot(Vector3D a, Vector3D b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
    public static Vector3D Cross(Vector3D a, Vector3D b) => new Vector3D(
        a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    public double Magnitude() => Math.Sqrt(X*X + Y*Y + Z*Z);
    public Vector3D Normalize() 
    {
        double mag = Magnitude();
        return mag > 0 ? new Vector3D(X/mag, Y/mag, Z/mag) : new Vector3D(0,0,0);
    }
}

    static void Solve(List<Body> bodies, double asteroidMass, double deltaVBudget, double warningDays)
    {
        var asteroid = bodies.First(b => b.Name == "Asteroid");
        var earth = bodies.First(b => b.Name == "Earth");
        
        // Simple Hohmann-like transfer + kinetic impact solution

    static TrajectorySolution? OptimizeSimpleTrajectory(List<Body> bodies, Body asteroid, Body earth, 
        double asteroidMass, double deltaVBudget, double warningDays)
    {
        const int NUM_SAMPLES = 100;
        double bestScore = double.MaxValue;
        List<Burn> bestBurns = null;
        double bestTotalDV = 0;
        
        // Sample launch times
        for (int i = 10; i < NUM_SAMPLES; i++)

            double launchTime = warningDays * (double)i / NUM_SAMPLES;
            
            // Propagate asteroid to Earth impact time (estimate)
            double impactTime = EstimateImpactTime(bodies, asteroid, earth, launchTime);
            if (impactTime > warningDays * 1.2) continue;
            
            // Calculate required spacecraft velocity at intercept
            var interceptState = PropagateSimple(asteroid, impactTime - launchTime);
            var earthState = PropagateSimple(earth, impactTime);
            
            // Rendezvous velocity
            Vector3D relPos = interceptState.Position - earthState.Position;
            Vector3D relVel = interceptState.Velocity - earthState.Velocity;
            
            double r = relPos.Magnitude();
            if (r < INTERCEPT_RADIUS) continue;
            
            // Lambert solver approximation for transfer
            var transfer = LambertTransfer(earthState.Position, earthState.Velocity, 
                                         interceptState.Position, launchTime, impactTime);
            
            if (transfer.TotalDV > deltaVBudget) continue;
            
            // Kinetic impact delta-v for asteroid deflection
            double impactDV = CalculateKineticImpactDV(relVel.Magnitude(), asteroidMass, 1000.0); // 1ton spacecraft
            
            if (transfer.TotalDV + impactDV > deltaVBudget) continue;
            
            // Score based on DV margin and timing
            double score = transfer.TotalDV + impactDV * 2 + Math.Abs(launchTime - warningDays * 0.5);
            
            if (score < bestScore)
            {
                bestScore = score;
                bestBurns = new List<Burn>

                    new Burn { Time = launchTime, DeltaV = transfer.LaunchDV },
                    new Burn { Time = (launchTime + impactTime) * 0.5, DeltaV = transfer.MidcourseDV * 0.1 },
                    new Burn { Time = impactTime, DeltaV = new Vector3D(0,0,impactDV) }
                };
                bestTotalDV = transfer.TotalDV + impactDV;
            }
        }
        
        if (bestBurns != null && bestTotalDV <= deltaVBudget)
        {
            return new TrajectorySolution { Burns = bestBurns };
        }
        
        return null;
    }
    
    static double EstimateImpactTime(List<Body> bodies, Body asteroid, Body earth, double fromTime)
    {
        // Simple minimum distance estimation
        var astPos = PropagateSimple(asteroid, fromTime).Position;
        var earthPos = PropagateSimple(earth, fromTime).Position;
        Vector3D relPos = astPos - earthPos;
        
        double v_rel = (asteroid.Velocity - earth.Velocity).Magnitude();
        double timeToImpact = relPos.Magnitude() / Math.Max(v_rel, 1e-6);
        
        return fromTime + timeToImpact / DAY;
    }
    
    static State PropagateSimple(Body body, double days)
    {
        double t = days * DAY;
        // Keplerian propagation approximation (2-body Sun)
        double mu = 1.3271244e20; // Solar GM
        Vector3D r0 = body.Position;
        Vector3D v0 = body.Velocity;
        
        double r0_mag = r0.Magnitude();
        Vector3D h = Vector3D.Cross(r0, v0);
        Vector3D e_vec = (Vector3D.Cross(v0, h) * (1.0 / mu) - r0.Normalize()) * (v0.Magnitude() * v0.Magnitude() / mu - 1.0 / r0_mag);
        
        double e = e_vec.Magnitude();
        double a = e < 0.99 ? r0_mag * (1 - e*e) / (1 - Math.Cos(Math.Acos(Vector3D.Dot(e_vec.Normalize(), r0.Normalize())))) : -mu / (v0.Magnitude()*v0.Magnitude());
        
        double dt = t * Math.Sqrt(Math.Abs(a*a*a / mu));
        double theta = Math.Acos((e*e - 1 + (1 - e*e) / (1 + e * Math.Cos(MeanAnomalyToTrue(dt / Math.Sqrt(Math.Abs(a*a*a/mu)))))) / e);
        
        Vector3D r = r0_mag * (Vector3D.Cross(h.Normalize(), e_vec.Normalize()) * Math.Sin(theta) + 
                              h.Normalize() * ((1 - e*e) / (1 + e * Math.Cos(theta))));
        
        return new State { Position = r, Velocity = v0 }; // Simplified
    }
    
    static double MeanAnomalyToTrue(double M) => M; // Simplified
    
    static (Vector3D LaunchDV, Vector3D MidcourseDV, double TotalDV) LambertTransfer(
        Vector3D r1, Vector3D v1, Vector3D r2, double t1, double t2)
    {
        double tof = (t2 - t1) * DAY;
        double r1m = r1.Magnitude();
        double r2m = r2.Magnitude();
        
        // Simplified Lambert solver using universal variable approximation
        double cte = 1.0 / Math.Sqrt(1.3271244e20);
        double A = Math.Sqrt(r1m * r2m * (1 + Vector3D.Dot(r1.Normalize(), r2.Normalize())));
        double tof2 = tof * tof;
        
        double y = tof * Math.Sqrt(1.3271244e20);
        double C = (r1m + r2m + A) / 2;
        double S = (r1m + r2m - A) / 2;
        
        double x = Math.Pow(y / Math.Sqrt(C), 2.0 / 3.0);
        double lambda = Math.Sqrt(2 * x / C);
        
        Vector3D v1_transfer = (r2 - r1 * (S / C)) * (lambda / tof);
        Vector3D v2_transfer = (r1 * (S / C) - r2) * (lambda / tof);
        
        Vector3D launchDV = v1_transfer - v1;
        Vector3D midcourseDV = (v2_transfer - v1_transfer) * 0.3; // Conservative
        
        return (launchDV, midcourseDV, launchDV.Magnitude() + midcourseDV.Magnitude());
    }
    
    static double CalculateKineticImpactDV(double relSpeed, double targetMass, double spacecraftMass)
    {
        // Momentum transfer: dv_target = (m_sc * v_rel) / m_target
        return (spacecraftMass * relSpeed) / targetMass;
    }
}
