import math


_MEM = {
    "est_pos": [0.0, -200.0, 0.0],
    "est_vel": [0.0, 0.0, 0.0],
    "est_att": [0.0, 0.0, 0.0],  # roll, pitch, yaw radians
    "bias_accel": [0.0, 0.0, 0.0],
    "last_range": None,
    "last_dt": 1.0,
    "mode": "far",
    "integral": [0.0, 0.0, 0.0],
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _norm(v):
    return math.sqrt(sum(a * a for a in v))


def _median(values):
    vals = sorted(values)
    n = len(vals)
    if n == 0:
        return None
    if n % 2:
        return vals[n // 2]
    return 0.5 * (vals[n // 2 - 1] + vals[n // 2])


def _valid_values(state, triples):
    vals = []
    for value_key, valid_key in triples:
        if state.get(valid_key, False):
            val = state.get(value_key)
            if isinstance(val, (int, float)) and math.isfinite(val):
                vals.append(float(val))
    return vals


def _fuse_axis(state, radar_key, radar_valid, lidar_key, lidar_valid, pred, max_jump):
    candidates = []
    if state.get(radar_valid, False):
        v = state.get(radar_key)
        if isinstance(v, (int, float)) and math.isfinite(v):
            candidates.append(float(v))
    if state.get(lidar_valid, False):
        v = state.get(lidar_key)
        if isinstance(v, (int, float)) and math.isfinite(v):
            candidates.append(float(v))

    if not candidates:
        return pred
    if len(candidates) == 1:
        meas = candidates[0]
    else:
        if abs(candidates[0] - candidates[1]) > max_jump:
            meas = candidates[0] if abs(candidates[0] - pred) < abs(candidates[1] - pred) else candidates[1]
        else:
            meas = 0.5 * (candidates[0] + candidates[1])
    return meas


def _fuse_relative_state(state, dt):
    pred_pos = [
        _MEM["est_pos"][i] + _MEM["est_vel"][i] * dt for i in range(3)
    ]
    pred_vel = list(_MEM["est_vel"])

    pos = [
        _fuse_axis(state, "radar_rel_pos_x_m", "radar_rel_x_valid", "lidar_rel_pos_x_m", "lidar_rel_x_valid", pred_pos[0], 80.0),
        _fuse_axis(state, "radar_rel_pos_y_m", "radar_rel_y_valid", "lidar_rel_pos_y_m", "lidar_rel_y_valid", pred_pos[1], 120.0),
        _fuse_axis(state, "radar_rel_pos_z_m", "radar_rel_z_valid", "lidar_rel_pos_z_m", "lidar_rel_z_valid", pred_pos[2], 80.0),
    ]
    vel = [
        _fuse_axis(state, "radar_rel_vel_x_ms", "radar_relv_x_valid", "lidar_rel_vel_x_ms", "lidar_relv_x_valid", pred_vel[0], 2.0),
        _fuse_axis(state, "radar_rel_vel_y_ms", "radar_relv_y_valid", "lidar_rel_vel_y_ms", "lidar_relv_y_valid", pred_vel[1], 2.0),
        _fuse_axis(state, "radar_rel_vel_z_ms", "radar_relv_z_valid", "lidar_rel_vel_z_ms", "lidar_relv_z_valid", pred_vel[2], 2.0),
    ]

    range_vals = _valid_values(
        state,
        [
            ("range_1_m", "range_1_valid"),
            ("range_2_m", "range_2_valid"),
            ("range_3_m", "range_3_valid"),
        ],
    )
    scalar_range = _median(range_vals)
    vec_range = _norm(pos)
    if scalar_range is not None and vec_range > 1.0:
        ratio = scalar_range / vec_range
        if 0.6 <= ratio <= 1.4:
            pos = [p * (0.85 + 0.15 * ratio) for p in pos]

    _MEM["est_pos"] = pos
    _MEM["est_vel"] = vel
    _MEM["last_range"] = scalar_range if scalar_range is not None else vec_range
    return pos, vel, _MEM["last_range"]


def _estimate_attitude(state, dt):
    att = list(_MEM["est_att"])
    gyro_axes = [
        [("gyro_1_roll_dps", "gyro_1_roll_valid"), ("gyro_2_roll_dps", "gyro_2_roll_valid"), ("gyro_3_roll_dps", "gyro_3_roll_valid")],
        [("gyro_1_pitch_dps", "gyro_1_pitch_valid"), ("gyro_2_pitch_dps", "gyro_2_pitch_valid"), ("gyro_3_pitch_dps", "gyro_3_pitch_valid")],
        [("gyro_1_yaw_dps", "gyro_1_yaw_valid"), ("gyro_2_yaw_dps", "gyro_2_yaw_valid"), ("gyro_3_yaw_dps", "gyro_3_yaw_valid")],
    ]
    star_axes = [
        [("star_tracker_1_roll_deg", "star_tracker_1_roll_valid"), ("star_tracker_2_roll_deg", "star_tracker_2_roll_valid")],
        [("star_tracker_1_pitch_deg", "star_tracker_1_pitch_valid"), ("star_tracker_2_pitch_deg", "star_tracker_2_pitch_valid")],
        [("star_tracker_1_yaw_deg", "star_tracker_1_yaw_valid"), ("star_tracker_2_yaw_deg", "star_tracker_2_yaw_valid")],
    ]

    for i in range(3):
        gyro_vals = _valid_values(state, gyro_axes[i])
        rate = math.radians(_median(gyro_vals)) if gyro_vals else 0.0
        att[i] += rate * dt
        star_vals = _valid_values(state, star_axes[i])
        if star_vals:
            meas = math.radians(_median(star_vals))
            att[i] = 0.9 * att[i] + 0.1 * meas

    _MEM["est_att"] = att
    return att


def _target_speed(range_m, fuel_pct):
    if range_m < 2.0:
        return 0.05
    if range_m < 10.0:
        return 0.08
    if range_m < 50.0:
        return 0.15
    if range_m < 200.0:
        return 0.35
    if range_m < 800.0:
        return 0.8
    return 1.2 if fuel_pct > 20.0 else 0.7


def _choose_mode(range_m):
    if range_m < 10.0:
        return "final"
    if range_m < 50.0:
        return "close"
    if range_m < 200.0:
        return "medium"
    return "far"


def autopilot_step(state: dict, dt: float) -> dict:
    dt = 1.0 if not isinstance(dt, (int, float)) or dt <= 0 else float(dt)
    dt = _clamp(dt, 0.05, 5.0)
    _MEM["last_dt"] = dt

    pos, vel, range_m = _fuse_relative_state(state, dt)
    _estimate_attitude(state, dt)

    x, y, z = pos
    vx, vy, vz = vel
    fuel_pct = float(state.get("fuel_remaining_pct", 100.0) or 100.0)
    target_range = float(state.get("target_dock_range_m", 0.0) or 0.0)
    target_speed = float(state.get("target_dock_speed_ms", 0.0) or 0.0)
    if target_range > 0.0:
        range_m = min(range_m, target_range)

    mode = _choose_mode(range_m)
    _MEM["mode"] = mode

    desired = [0.0, 0.0, 0.0]
    speed_cmd = min(_target_speed(range_m, fuel_pct), max(target_speed, 0.05) if target_speed > 0 else 9.0)

    if mode == "far":
        desired[0] = _clamp(-0.010 * x - 0.080 * vx, -0.25, 0.25)
        desired[2] = _clamp(-0.010 * z - 0.080 * vz, -0.25, 0.25)

        if y < -20.0:
            desired_y_vel = speed_cmd
            desired[1] = _clamp(-0.020 * (vy - desired_y_vel), -0.55, 0.05)
        elif y > 20.0:
            desired[1] = _clamp(-0.015 * (vy + speed_cmd * 0.5), -0.05, 0.55)
        else:
            desired[1] = _clamp(-0.015 * y - 0.12 * vy, -0.2, 0.2)

        if abs(x) > 100.0 and y < 0.0:
            desired[1] += -0.08 * math.copysign(1.0, y)

    elif mode == "medium":
        desired[0] = _clamp(-0.020 * x - 0.18 * vx, -0.35, 0.35)
        desired[2] = _clamp(-0.020 * z - 0.18 * vz, -0.35, 0.35)
        closing_vel = min(speed_cmd, max(0.08, 0.004 * abs(y) + 0.05))
        desired_y = 0.0
        if y < -2.0:
            desired_y = closing_vel
        elif y > 2.0:
            desired_y = -min(0.2, closing_vel)
        desired[1] = _clamp(-0.25 * (vy - desired_y) - 0.010 * y, -0.28, 0.28)

    elif mode == "close":
        desired[0] = _clamp(-0.040 * x - 0.35 * vx, -0.25, 0.25)
        desired[2] = _clamp(-0.040 * z - 0.35 * vz, -0.25, 0.25)
        desired_y = 0.0
        if y < -0.5:
            desired_y = min(speed_cmd, 0.12)
        elif y > 0.5:
            desired_y = -0.08
        desired[1] = _clamp(-0.40 * (vy - desired_y) - 0.030 * y, -0.18, 0.18)

    else:  # final
        desired[0] = _clamp(-0.080 * x - 0.50 * vx, -0.12, 0.12)
        desired[2] = _clamp(-0.080 * z - 0.50 * vz, -0.12, 0.12)
        desired_y = 0.0
        if y < -0.15:
            desired_y = min(speed_cmd, 0.06)
        elif y > 0.15:
            desired_y = -0.04
        desired[1] = _clamp(-0.55 * (vy - desired_y) - 0.060 * y, -0.10, 0.10)

    if range_m < 3.0 and _norm(vel) < 0.25:
        desired = [
            _clamp(-0.10 * x - 0.60 * vx, -0.08, 0.08),
            _clamp(-0.10 * y - 0.60 * vy, -0.08, 0.08),
            _clamp(-0.10 * z - 0.60 * vz, -0.08, 0.08),
        ]

    if fuel_pct < 15.0:
        desired = [0.65 * a for a in desired]
    elif fuel_pct < 5.0:
        desired = [0.4 * a for a in desired]

    if state.get("dock_camera_active", False) and range_m < 500.0:
        cam = []
        ok = True
        for k, vk in [
            ("dock_camera_bearing_x", "dock_camera_bearing_x_valid"),
            ("dock_camera_bearing_y", "dock_camera_bearing_y_valid"),
            ("dock_camera_bearing_z", "dock_camera_bearing_z_valid"),
        ]:
            if not state.get(vk, False):
                ok = False
                break
            cam.append(float(state.get(k, 0.0)))
        if ok and _norm(cam) > 1e-6:
            n = _norm(cam)
            cam = [c / n for c in cam]
            if range_m < 30.0:
                desired[0] += _clamp(0.08 * cam[0], -0.03, 0.03)
                desired[2] += _clamp(0.08 * cam[2], -0.03, 0.03)

    att_rate_roll = _clamp(-0.6 * _MEM["est_att"][0], -0.035, 0.035)
    att_rate_pitch = _clamp(-0.6 * _MEM["est_att"][1], -0.035, 0.035)
    att_rate_yaw = _clamp(-0.6 * _MEM["est_att"][2], -0.035, 0.035)

    return {
        "thrust_x": _clamp(desired[0], -1.0, 1.0),
        "thrust_y": _clamp(desired[1], -1.0, 1.0),
        "thrust_z": _clamp(desired[2], -1.0, 1.0),
        "att_rate_roll": att_rate_roll,
        "att_rate_pitch": att_rate_pitch,
        "att_rate_yaw": att_rate_yaw,
    }
