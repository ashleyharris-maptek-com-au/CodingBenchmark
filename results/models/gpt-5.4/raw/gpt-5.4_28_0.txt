use std::cmp::Reverse;
use std::collections::BinaryHeap;
use std::io::{self, Read};

#[derive(Clone)]
struct Graph {
    n: usize,
    edges: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
    degree: Vec<usize>,
}

impl Graph {
    fn new(n: usize, edges: Vec<(usize, usize)>) -> Self {
        let mut adj = vec![Vec::new(); n];
        for &(u, v) in &edges {
            adj[u].push(v);
            adj[v].push(u);
        }
        let degree = adj.iter().map(Vec::len).collect();
        Self { n, edges, adj, degree }
    }
}

fn parse_input() -> Graph {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n = match it.next() {
        Some(v) => v.parse::<usize>().unwrap(),
        None => return Graph::new(0, Vec::new()),
    };
    let m = it.next().unwrap_or("0").parse::<usize>().unwrap();
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u = match it.next() {
            Some(v) => v.parse::<usize>().unwrap(),
            None => break,
        };
        let v = match it.next() {
            Some(v) => v.parse::<usize>().unwrap(),
            None => break,
        };
        if u != v && u < n && v < n {
            edges.push((u, v));
        }
    }
    Graph::new(n, edges)
}

fn maximal_matching_cover(g: &Graph) -> Vec<bool> {
    let mut used = vec![false; g.n];
    let mut cover = vec![false; g.n];
    for &(u, v) in &g.edges {
        if !used[u] && !used[v] {
            used[u] = true;
            used[v] = true;
            cover[u] = true;
            cover[v] = true;
        }
    }
    cover
}

fn greedy_cover(g: &Graph) -> Vec<bool> {
    let mut covered = vec![false; g.edges.len()];
    let mut remaining_deg = g.degree.clone();
    let mut heap = BinaryHeap::new();
    for (v, &deg) in remaining_deg.iter().enumerate() {
        heap.push((deg, v));
    }

    let mut cover = vec![false; g.n];
    let mut uncovered_edges = g.edges.len();

    while uncovered_edges > 0 {
        let (deg, v) = loop {
            match heap.pop() {
                Some((d, x)) if d == remaining_deg[x] => break (d, x),
                Some(_) => {}
                None => break (0, 0),
            }
        };
        if deg == 0 {
            break;
        }
        if cover[v] {
            continue;
        }
        cover[v] = true;
        for (ei, &(a, b)) in g.edges.iter().enumerate() {
            if covered[ei] {
                continue;
            }
            if a == v || b == v {
                covered[ei] = true;
                uncovered_edges -= 1;
                let other = if a == v { b } else { a };
                if remaining_deg[other] > 0 {
                    remaining_deg[other] -= 1;
                    heap.push((remaining_deg[other], other));
                }
            }
        }
        remaining_deg[v] = 0;
    }

    for (ei, &(u, v)) in g.edges.iter().enumerate() {
        if !covered[ei] {
            cover[u] = true;
            cover[v] = true;
        }
    }
    cover
}

fn build_subgraph(g: &Graph, alive: &[bool]) -> (Vec<usize>, Vec<(usize, usize)>) {
    let mut map = vec![usize::MAX; g.n];
    let mut rev = Vec::new();
    for (v, &ok) in alive.iter().enumerate() {
        if ok {
            map[v] = rev.len();
            rev.push(v);
        }
    }
    let mut edges = Vec::new();
    for &(u, v) in &g.edges {
        if alive[u] && alive[v] {
            edges.push((map[u], map[v]));
        }
    }
    (rev, edges)
}

fn reduced_exact_cover(g: &Graph, initial_upper: &[bool]) -> Option<Vec<bool>> {
    const MAX_EXACT_VERTICES: usize = 64;
    const MAX_EXACT_EDGES: usize = 256;

    let mut forced = vec![false; g.n];
    let mut alive = vec![true; g.n];
    let mut adj_alive: Vec<Vec<usize>> = g.adj.clone();
    let mut deg: Vec<usize> = g.degree.clone();

    loop {
        let mut changed = false;
        for v in 0..g.n {
            if !alive[v] {
                continue;
            }
            if deg[v] == 0 {
                alive[v] = false;
                changed = true;
                continue;
            }
            if deg[v] == 1 {
                let mut neighbor = usize::MAX;
                for &u in &adj_alive[v] {
                    if alive[u] {
                        neighbor = u;
                        break;
                    }
                }
                if neighbor != usize::MAX && alive[neighbor] {
                    forced[neighbor] = true;
                    alive[neighbor] = false;
                    changed = true;
                    for &x in &adj_alive[neighbor] {
                        if alive[x] && deg[x] > 0 {
                            deg[x] -= 1;
                        }
                    }
                    deg[neighbor] = 0;
                }
                if alive[v] {
                    alive[v] = false;
                    deg[v] = 0;
                }
            }
        }
        if !changed {
            break;
        }
    }

    let (rev, edges_small) = build_subgraph(g, &alive);
    if rev.len() > MAX_EXACT_VERTICES || edges_small.len() > MAX_EXACT_EDGES {
        return None;
    }

    let upper_bound = initial_upper.iter().filter(|&&x| x).count();
    let forced_count = forced.iter().filter(|&&x| x).count();
    if forced_count > upper_bound {
        return None;
    }

    let mut edge_masks = vec![0_u64; rev.len()];
    for &(u, v) in &edges_small {
        edge_masks[u] |= 1_u64 << v;
        edge_masks[v] |= 1_u64 << u;
    }

    let mut best_mask = if rev.len() == 64 { u64::MAX } else { (1_u64 << rev.len()) - 1 };
    let mut best_size = upper_bound.saturating_sub(forced_count);

    fn dfs(
        edge_masks: &[u64],
        alive_mask: u64,
        chosen_mask: u64,
        chosen_size: usize,
        best_mask: &mut u64,
        best_size: &mut usize,
    ) {
        if chosen_size >= *best_size {
            return;
        }
        let mut pivot = None;
        let mut max_deg = 0usize;
        let mut mask = alive_mask;
        while mask != 0 {
            let v = mask.trailing_zeros() as usize;
            mask &= mask - 1;
            let deg = (edge_masks[v] & alive_mask).count_ones() as usize;
            if deg > max_deg {
                max_deg = deg;
                pivot = Some(v);
            }
        }
        if max_deg == 0 {
            *best_size = chosen_size;
            *best_mask = chosen_mask;
            return;
        }
        let v = pivot.unwrap_or(0);
        let neighbors = edge_masks[v] & alive_mask;

        let alive_without_v = alive_mask & !(1_u64 << v);
        dfs(
            edge_masks,
            alive_without_v,
            chosen_mask | (1_u64 << v),
            chosen_size + 1,
            best_mask,
            best_size,
        );

        let take_neighbors = neighbors.count_ones() as usize;
        if chosen_size + take_neighbors < *best_size {
            let alive_without_closed = alive_mask & !neighbors & !(1_u64 << v);
            dfs(
                edge_masks,
                alive_without_closed,
                chosen_mask | neighbors,
                chosen_size + take_neighbors,
                best_mask,
                best_size,
            );
        }
    }

    let all_mask = if rev.len() == 64 { u64::MAX } else { (1_u64 << rev.len()) - 1 };
    dfs(&edge_masks, all_mask, 0, 0, &mut best_mask, &mut best_size);

    let mut cover = forced;
    for (i, &orig) in rev.iter().enumerate() {
        if ((best_mask >> i) & 1) != 0 {
            cover[orig] = true;
        }
    }
    Some(cover)
}

fn local_improve(g: &Graph, cover: &mut [bool]) {
    let mut uncovered = vec![0usize; g.n];
    for &(u, v) in &g.edges {
        if !cover[u] && !cover[v] {
            cover[u] = true;
        }
    }

    for &(u, v) in &g.edges {
        if !cover[u] || !cover[v] {
            let chosen = if cover[u] { u } else { v };
            uncovered[chosen] += 0;
        }
    }

    let mut order: Vec<usize> = (0..g.n).filter(|&v| cover[v]).collect();
    order.sort_by_key(|&v| Reverse(g.degree[v]));
    for v in order {
        if !cover[v] {
            continue;
        }
        let mut needed = false;
        for &u in &g.adj[v] {
            if !cover[u] {
                needed = true;
                break;
            }
        }
        if !needed {
            cover[v] = false;
        }
    }
}

fn verify_cover(g: &Graph, cover: &[bool]) -> bool {
    g.edges.iter().all(|&(u, v)| cover[u] || cover[v])
}

fn main() {
    let g = parse_input();
    if g.n == 0 {
        println!("0");
        println!();
        return;
    }

    let matching_cover = maximal_matching_cover(&g);
    let greedy = greedy_cover(&g);

    let mut best = if greedy.iter().filter(|&&x| x).count() < matching_cover.iter().filter(|&&x| x).count() {
        greedy
    } else {
        matching_cover
    };

    if let Some(exact) = reduced_exact_cover(&g, &best) {
        if exact.iter().filter(|&&x| x).count() < best.iter().filter(|&&x| x).count() {
            best = exact;
        }
    }

    local_improve(&g, &mut best);

    if !verify_cover(&g, &best) {
        for &(u, v) in &g.edges {
            if !best[u] && !best[v] {
                best[u] = true;
            }
        }
    }

    let result: Vec<usize> = best
        .iter()
        .enumerate()
        .filter_map(|(v, &take)| if take { Some(v) } else { None })
        .collect();

    println!("{}", result.len());
    for (i, v) in result.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{v}");
    }
    println!();
}
