#include <algorithm>
#include <cstdint>
#include <iostream>
#include <vector>

using namespace std;

struct BitSet {
    int n = 0;
    int words = 0;
    vector<uint64_t> a;

    BitSet() = default;
    explicit BitSet(int n_) { init(n_); }

    void init(int n_) {
        n = n_;
        words = (n + 63) >> 6;
        a.assign(words, 0);
    }

    void set(int i) { a[i >> 6] |= 1ULL << (i & 63); }
    void reset(int i) { a[i >> 6] &= ~(1ULL << (i & 63)); }
    bool test(int i) const { return (a[i >> 6] >> (i & 63)) & 1ULL; }

    bool any() const {
        for (uint64_t x : a) {
            if (x) return true;
        }
        return false;
    }

    int count() const {
        int s = 0;
        for (uint64_t x : a) s += static_cast<int>(__popcnt64(x));
        return s;
    }

    int first() const {
        for (int w = 0; w < words; ++w) {
            uint64_t x = a[w];
            if (x) return (w << 6) + static_cast<int>(__builtin_ctzll(x));
        }
        return -1;
    }

    void intersect_with(const BitSet& other) {
        for (int i = 0; i < words; ++i) a[i] &= other.a[i];
    }
};

struct MaxCliqueSolver {
    int n = 0;
    vector<BitSet> g;
    vector<int> best;
    vector<int> cur;

    explicit MaxCliqueSolver(int n_) : n(n_), g(n_, BitSet(n_)) {}

    void add_edge(int u, int v) {
        if (u == v) return;
        g[u].set(v);
        g[v].set(u);
    }

    vector<int> to_vector(const BitSet& bs) const {
        vector<int> out;
        out.reserve(bs.count());
        for (int w = 0; w < bs.words; ++w) {
            uint64_t x = bs.a[w];
            while (x) {
                int b = static_cast<int>(__builtin_ctzll(x));
                int v = (w << 6) + b;
                if (v < n) out.push_back(v);
                x &= x - 1;
            }
        }
        return out;
    }

    int degree_in_set(int v, const BitSet& set) const {
        int d = 0;
        for (int w = 0; w < set.words; ++w) {
            d += static_cast<int>(__popcnt64(g[v].a[w] & set.a[w]));
        }
        return d;
    }

    void color_sort(const BitSet& cand, vector<int>& order, vector<int>& bound) const {
        vector<int> verts = to_vector(cand);
        sort(verts.begin(), verts.end(), [&](int x, int y) {
            int dx = degree_in_set(x, cand);
            int dy = degree_in_set(y, cand);
            if (dx != dy) return dx > dy;
            return x < y;
        });

        order.clear();
        bound.clear();
        vector<int> rem = verts;
        int color = 0;

        while (!rem.empty()) {
            ++color;
            BitSet forbidden(n);
            vector<int> next;
            for (int v : rem) {
                if (!forbidden.test(v)) {
                    order.push_back(v);
                    bound.push_back(color);
                    forbidden.set(v);
                    for (int w = 0; w < forbidden.words; ++w) {
                        forbidden.a[w] |= g[v].a[w] & cand.a[w];
                    }
                } else {
                    next.push_back(v);
                }
            }
            rem.swap(next);
        }
    }

    void expand(BitSet cand) {
        if (!cand.any()) {
            if (cur.size() > best.size()) best = cur;
            return;
        }

        vector<int> order;
        vector<int> bound;
        color_sort(cand, order, bound);

        for (int i = static_cast<int>(order.size()) - 1; i >= 0; --i) {
            if (cur.size() + bound[i] <= best.size()) return;

            int v = order[i];
            if (!cand.test(v)) continue;

            cur.push_back(v);
            BitSet next = cand;
            next.intersect_with(g[v]);
            expand(next);
            cur.pop_back();

            cand.reset(v);
        }
    }

    vector<int> solve() {
        BitSet all(n);
        for (int i = 0; i < n; ++i) all.set(i);
        expand(all);
        sort(best.begin(), best.end());
        return best;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<BitSet> original(n, BitSet(n));
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (0 <= u && u < n && 0 <= v && v < n && u != v) {
            original[u].set(v);
            original[v].set(u);
        }
    }

    MaxCliqueSolver solver(n);
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            if (!original[i].test(j)) solver.add_edge(i, j);
        }
    }

    vector<int> ans = solver.solve();
    cout << ans.size() << '\n';
    for (int i = 0; i < static_cast<int>(ans.size()); ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
