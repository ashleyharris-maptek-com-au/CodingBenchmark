use std::io::{self, Read};

struct Solver {
    n: usize,
    half: usize,
    adj_mask: Vec<u128>,
    assigned: Vec<i8>, // -1 = unassigned, 0 = B, 1 = A
    order: Vec<usize>,
    prefix_a: Vec<usize>,
    prefix_b: Vec<usize>,
    cur_cut: usize,
    best_cut: usize,
    best_mask: u128,
}

impl Solver {
    fn new(n: usize, adj_mask: Vec<u128>) -> Self {
        let half = n / 2;
        let mut order: Vec<usize> = (0..n).collect();
        order.sort_by_key(|&v| std::cmp::Reverse(adj_mask[v].count_ones()));
        Self {
            n,
            half,
            adj_mask,
            assigned: vec![-1; n],
            order,
            prefix_a: vec![0; n + 1],
            prefix_b: vec![0; n + 1],
            cur_cut: 0,
            best_cut: usize::MAX,
            best_mask: 0,
        }
    }

    fn solve(&mut self) {
        self.dfs(0, 0, 0, 0);
    }

    fn dfs(&mut self, idx: usize, a_count: usize, b_count: usize, a_mask: u128) {
        if idx == self.n {
            if a_count == self.half && b_count == self.half && self.cur_cut < self.best_cut {
                self.best_cut = self.cur_cut;
                self.best_mask = a_mask;
            }
            return;
        }

        if self.cur_cut >= self.best_cut {
            return;
        }

        let rem = self.n - idx;
        if a_count > self.half || b_count > self.half {
            return;
        }
        if a_count + rem < self.half || b_count + rem < self.half {
            return;
        }

        let lower = self.cur_cut + self.lower_bound(idx, a_count, b_count);
        if lower >= self.best_cut {
            return;
        }

        let v = self.order[idx];
        let deg_a = self.prefix_a[idx];
        let deg_b = self.prefix_b[idx];

        let can_a = a_count < self.half;
        let can_b = b_count < self.half;

        if can_a && can_b {
            if deg_a <= deg_b {
                self.assign_and_recurse(idx, v, 1, a_count, b_count, a_mask);
                self.assign_and_recurse(idx, v, 0, a_count, b_count, a_mask);
            } else {
                self.assign_and_recurse(idx, v, 0, a_count, b_count, a_mask);
                self.assign_and_recurse(idx, v, 1, a_count, b_count, a_mask);
            }
        } else if can_a {
            self.assign_and_recurse(idx, v, 1, a_count, b_count, a_mask);
        } else {
            self.assign_and_recurse(idx, v, 0, a_count, b_count, a_mask);
        }
    }

    fn assign_and_recurse(
        &mut self,
        idx: usize,
        v: usize,
        side: i8,
        a_count: usize,
        b_count: usize,
        a_mask: u128,
    ) {
        let added_cut = if side == 1 {
            self.prefix_b[idx]
        } else {
            self.prefix_a[idx]
        };
        let next_mask = if side == 1 { a_mask | (1u128 << v) } else { a_mask };

        self.assigned[v] = side;
        self.cur_cut += added_cut;

        let touched = self.update_prefix(idx + 1, v, side, 1);
        if side == 1 {
            self.dfs(idx + 1, a_count + 1, b_count, next_mask);
        } else {
            self.dfs(idx + 1, a_count, b_count + 1, next_mask);
        }
        self.update_prefix_list(&touched, side, -1);

        self.cur_cut -= added_cut;
        self.assigned[v] = -1;
    }

    fn update_prefix(&mut self, next_idx: usize, v: usize, side: i8, delta: isize) -> Vec<usize> {
        let mut touched = Vec::new();
        let mask_v = self.adj_mask[v];
        for j in next_idx..self.n {
            let u = self.order[j];
            if ((mask_v >> u) & 1) != 0 {
                if side == 1 {
                    self.prefix_a[j] = ((self.prefix_a[j] as isize) + delta) as usize;
                } else {
                    self.prefix_b[j] = ((self.prefix_b[j] as isize) + delta) as usize;
                }
                touched.push(j);
            }
        }
        touched
    }

    fn update_prefix_list(&mut self, touched: &[usize], side: i8, delta: isize) {
        for &j in touched {
            if side == 1 {
                self.prefix_a[j] = ((self.prefix_a[j] as isize) + delta) as usize;
            } else {
                self.prefix_b[j] = ((self.prefix_b[j] as isize) + delta) as usize;
            }
        }
    }

    fn lower_bound(&self, idx: usize, a_count: usize, b_count: usize) -> usize {
        let need_a = self.half - a_count;
        let need_b = self.half - b_count;
        let rem = self.n - idx;
        debug_assert_eq!(need_a + need_b, rem);

        let mut forced = 0usize;
        let mut penalties = Vec::with_capacity(rem);

        for j in idx..self.n {
            let ca = self.prefix_b[j];
            let cb = self.prefix_a[j];
            forced += ca.min(cb);
            penalties.push(ca.abs_diff(cb));
        }

        penalties.sort_unstable();
        forced + penalties.iter().take(need_a.min(need_b)).sum::<usize>()
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    assert!(n % 2 == 0, "n must be even");
    assert!(n <= 128, "this exact solver supports n <= 128");

    let mut adj_mask = vec![0u128; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj_mask[u] |= 1u128 << v;
        adj_mask[v] |= 1u128 << u;
    }

    let mut solver = Solver::new(n, adj_mask);
    solver.solve();

    let mut part = Vec::with_capacity(solver.half);
    for v in 0..n {
        if ((solver.best_mask >> v) & 1) != 0 {
            part.push(v);
        }
    }

    println!("{}", solver.best_cut);
    for (i, v) in part.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
