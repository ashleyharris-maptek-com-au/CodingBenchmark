#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct Task {
    int machine = 0;
    int duration = 0;
};

struct JobShop {
    int jobs = 0;
    int machines = 0;
    vector<vector<Task>> job_tasks;
    vector<vector<long long>> prefix;
    vector<long long> total_job_duration;
    vector<int> task_count;
    long long total_tasks = 0;
    long long total_duration = 0;

    void build_prefix() {
        prefix.assign(jobs, {});
        total_job_duration.assign(jobs, 0);
        task_count.assign(jobs, 0);
        total_tasks = 0;
        total_duration = 0;
        for (int j = 0; j < jobs; ++j) {
            int t = static_cast<int>(job_tasks[j].size());
            task_count[j] = t;
            prefix[j].assign(t + 1, 0);
            for (int i = 0; i < t; ++i) {
                prefix[j][i + 1] = prefix[j][i] + job_tasks[j][i].duration;
            }
            total_job_duration[j] = prefix[j][t];
            total_duration += total_job_duration[j];
            total_tasks += t;
        }
    }
};

struct Schedule {
    vector<vector<long long>> start;
    long long makespan = 0;
};

struct Candidate {
    int job = -1;
    int task = -1;
    int machine = -1;
    int duration = 0;
    long long ready = 0;
    long long score = 0;
};

static long long compute_lower_bound(const JobShop& js) {
    vector<long long> machine_load(js.machines, 0);
    long long max_job = 0;
    for (int j = 0; j < js.jobs; ++j) {
        max_job = max(max_job, js.total_job_duration[j]);
        for (const Task& task : js.job_tasks[j]) {
            machine_load[task.machine] += task.duration;
        }
    }
    long long max_machine = 0;
    for (long long load : machine_load) {
        max_machine = max(max_machine, load);
    }
    return max(max_job, max_machine);
}

static Schedule build_schedule_by_job_order(const JobShop& js, const vector<int>& order, bool use_score) {
    vector<vector<long long>> start(js.jobs);
    for (int j = 0; j < js.jobs; ++j) {
        start[j].assign(js.task_count[j], 0);
    }

    vector<long long> job_ready(js.jobs, 0);
    vector<long long> machine_ready(js.machines, 0);
    vector<int> next_task(js.jobs, 0);

    auto remaining_work = [&](int job, int task_idx) -> long long {
        return js.total_job_duration[job] - js.prefix[job][task_idx];
    };

    for (int preferred_job : order) {
        if (preferred_job < 0 || preferred_job >= js.jobs) {
            continue;
        }
        while (next_task[preferred_job] < js.task_count[preferred_job]) {
            const int current_task = next_task[preferred_job];
            const Task& target = js.job_tasks[preferred_job][current_task];

            Candidate best;
            best.score = numeric_limits<long long>::max();
            best.ready = numeric_limits<long long>::max();

            for (int j = 0; j < js.jobs; ++j) {
                int ti = next_task[j];
                if (ti >= js.task_count[j]) {
                    continue;
                }
                const Task& task = js.job_tasks[j][ti];
                if (task.machine != target.machine) {
                    continue;
                }
                long long est = max(job_ready[j], machine_ready[task.machine]);
                long long tail = remaining_work(j, ti + 1);
                long long slack = js.total_job_duration[j] - js.prefix[j][ti];
                long long score = est;
                if (use_score) {
                    score = est * 8 - static_cast<long long>(task.duration) * 3 - tail * 2 - slack;
                }
                if (score < best.score ||
                    (score == best.score && est < best.ready) ||
                    (score == best.score && est == best.ready && task.duration > best.duration)) {
                    best = {j, ti, task.machine, task.duration, est, score};
                }
            }

            if (best.job == -1) {
                break;
            }

            long long st = max(job_ready[best.job], machine_ready[best.machine]);
            start[best.job][best.task] = st;
            long long finish = st + best.duration;
            job_ready[best.job] = finish;
            machine_ready[best.machine] = finish;
            ++next_task[best.job];
        }
    }

    bool progress = true;
    while (progress) {
        progress = false;
        Candidate best;
        best.score = numeric_limits<long long>::max();
        best.ready = numeric_limits<long long>::max();

        for (int j = 0; j < js.jobs; ++j) {
            int ti = next_task[j];
            if (ti >= js.task_count[j]) {
                continue;
            }
            const Task& task = js.job_tasks[j][ti];
            long long est = max(job_ready[j], machine_ready[task.machine]);
            long long tail = remaining_work(j, ti + 1);
            long long score = est;
            if (use_score) {
                score = est * 8 - static_cast<long long>(task.duration) * 3 - tail * 2;
            }
            if (score < best.score ||
                (score == best.score && est < best.ready) ||
                (score == best.score && est == best.ready && task.duration > best.duration)) {
                best = {j, ti, task.machine, task.duration, est, score};
            }
        }

        if (best.job != -1) {
            progress = true;
            long long st = max(job_ready[best.job], machine_ready[best.machine]);
            start[best.job][best.task] = st;
            long long finish = st + best.duration;
            job_ready[best.job] = finish;
            machine_ready[best.machine] = finish;
            ++next_task[best.job];
        }
    }

    long long makespan = 0;
    for (long long t : job_ready) {
        makespan = max(makespan, t);
    }
    return {move(start), makespan};
}

static vector<int> order_jobs_by_total_work(const JobShop& js, bool descending) {
    vector<int> order(js.jobs);
    iota(order.begin(), order.end(), 0);
    stable_sort(order.begin(), order.end(), [&](int a, int b) {
        if (js.total_job_duration[a] != js.total_job_duration[b]) {
            return descending ? js.total_job_duration[a] > js.total_job_duration[b]
                              : js.total_job_duration[a] < js.total_job_duration[b];
        }
        return a < b;
    });
    return order;
}

static vector<int> order_jobs_by_bottleneck(const JobShop& js) {
    vector<long long> machine_load(js.machines, 0);
    for (int j = 0; j < js.jobs; ++j) {
        for (const Task& task : js.job_tasks[j]) {
            machine_load[task.machine] += task.duration;
        }
    }

    vector<pair<long long, int>> ranked;
    ranked.reserve(js.jobs);
    for (int j = 0; j < js.jobs; ++j) {
        long long score = 0;
        for (const Task& task : js.job_tasks[j]) {
            score += machine_load[task.machine] * 2 + task.duration;
        }
        ranked.push_back({-score, j});
    }
    sort(ranked.begin(), ranked.end());

    vector<int> order;
    order.reserve(js.jobs);
    for (auto [_, j] : ranked) {
        order.push_back(j);
    }
    return order;
}

static Schedule greedy_dispatch(const JobShop& js, int mode) {
    vector<int> order;
    if (mode == 0) {
        order.resize(js.jobs);
        iota(order.begin(), order.end(), 0);
        return build_schedule_by_job_order(js, order, false);
    }
    if (mode == 1) {
        order = order_jobs_by_total_work(js, true);
        return build_schedule_by_job_order(js, order, true);
    }
    if (mode == 2) {
        order = order_jobs_by_total_work(js, false);
        return build_schedule_by_job_order(js, order, true);
    }
    order = order_jobs_by_bottleneck(js);
    return build_schedule_by_job_order(js, order, true);
}

static Schedule local_search(const JobShop& js, Schedule best, int iterations, uint64_t seed) {
    if (js.jobs <= 1 || iterations <= 0) {
        return best;
    }

    vector<int> base = order_jobs_by_bottleneck(js);
    mt19937_64 rng(seed);

    for (int it = 0; it < iterations; ++it) {
        vector<int> order = base;
        int a = static_cast<int>(rng() % js.jobs);
        int b = static_cast<int>(rng() % js.jobs);
        if (a != b) {
            swap(order[a], order[b]);
        }
        if (js.jobs >= 4 && (rng() & 1ULL)) {
            int l = static_cast<int>(rng() % js.jobs);
            int r = static_cast<int>(rng() % js.jobs);
            if (l > r) {
                swap(l, r);
            }
            reverse(order.begin() + l, order.begin() + r + 1);
        }
        Schedule candidate = build_schedule_by_job_order(js, order, true);
        if (candidate.makespan < best.makespan) {
            best = move(candidate);
            base = move(order);
        }
    }
    return best;
}

static Schedule solve_small(const JobShop& js) {
    Schedule best = greedy_dispatch(js, 0);
    best = min(best, greedy_dispatch(js, 1), [](const Schedule& a, const Schedule& b) {
        return a.makespan < b.makespan;
    });
    Schedule alt = greedy_dispatch(js, 2);
    if (alt.makespan < best.makespan) {
        best = move(alt);
    }
    alt = greedy_dispatch(js, 3);
    if (alt.makespan < best.makespan) {
        best = move(alt);
    }
    return local_search(js, move(best), 400, 123456789ULL);
}

static Schedule solve_medium(const JobShop& js) {
    Schedule best = greedy_dispatch(js, 1);
    Schedule alt = greedy_dispatch(js, 3);
    if (alt.makespan < best.makespan) {
        best = move(alt);
    }
    alt = greedy_dispatch(js, 0);
    if (alt.makespan < best.makespan) {
        best = move(alt);
    }
    int iterations = max(100, min(2000, js.jobs * 6));
    return local_search(js, move(best), iterations, 987654321ULL);
}

static Schedule solve_large(const JobShop& js) {
    Schedule best = greedy_dispatch(js, 3);
    Schedule alt = greedy_dispatch(js, 1);
    if (alt.makespan < best.makespan) {
        best = move(alt);
    }
    if (js.total_tasks <= 20000) {
        int iterations = max(40, min(400, js.jobs));
        best = local_search(js, move(best), iterations, 42424242ULL);
    }
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    JobShop js;
    if (!(cin >> js.jobs >> js.machines)) {
        return 0;
    }

    js.job_tasks.assign(js.jobs, {});
    for (int j = 0; j < js.jobs; ++j) {
        int t;
        cin >> t;
        js.job_tasks[j].resize(t);
        for (int i = 0; i < t; ++i) {
            cin >> js.job_tasks[j][i].machine >> js.job_tasks[j][i].duration;
        }
    }
    js.build_prefix();

    Schedule result;
    if (js.total_tasks <= 120) {
        result = solve_small(js);
    } else if (js.total_tasks <= 5000) {
        result = solve_medium(js);
    } else {
        result = solve_large(js);
    }

    long long lower_bound = compute_lower_bound(js);
    if (result.makespan < lower_bound) {
        result = greedy_dispatch(js, 0);
    }

    for (int j = 0; j < js.jobs; ++j) {
        for (int i = 0; i < js.task_count[j]; ++i) {
            if (i) {
                cout << ' ';
            }
            cout << result.start[j][i] << ' ' << js.job_tasks[j][i].machine;
        }
        cout << '\n';
    }

    return 0;
}
