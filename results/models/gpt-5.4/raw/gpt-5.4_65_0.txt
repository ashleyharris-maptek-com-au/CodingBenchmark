const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Task {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
}

interface ScheduleEntry {
  id: string;
  start: number;
  machine: number;
}

interface StateTask {
  idx: number;
  id: string;
  duration: number;
  deadline: number;
  weight: number;
  deps: number[];
  succs: number[];
  indegree: number;
  depCount: number;
  level: number;
  bottom: number;
  release: number;
  remainingDeps: number;
}

class MaxHeap<T> {
  private data: T[] = [];
  constructor(private readonly compare: (a: T, b: T) => number) {}

  size(): number {
    return this.data.length;
  }

  push(value: T): void {
    const arr = this.data;
    arr.push(value);
    let i = arr.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.compare(arr[i], arr[p]) <= 0) break;
      [arr[i], arr[p]] = [arr[p], arr[i]];
      i = p;
    }
  }

  pop(): T | undefined {
    const arr = this.data;
    if (arr.length === 0) return undefined;
    const top = arr[0];
    const last = arr.pop()!;
    if (arr.length > 0) {
      arr[0] = last;
      let i = 0;
      for (;;) {
        let best = i;
        const l = i * 2 + 1;
        const r = l + 1;
        if (l < arr.length && this.compare(arr[l], arr[best]) > 0) best = l;
        if (r < arr.length && this.compare(arr[r], arr[best]) > 0) best = r;
        if (best === i) break;
        [arr[i], arr[best]] = [arr[best], arr[i]];
        i = best;
      }
    }
    return top;
  }
}

class MinHeap<T> {
  private data: T[] = [];
  constructor(private readonly compare: (a: T, b: T) => number) {}

  size(): number {
    return this.data.length;
  }

  push(value: T): void {
    const arr = this.data;
    arr.push(value);
    let i = arr.length - 1;
    while (i > 0) {

      if (this.compare(arr[i], arr[p]) >= 0) break;
      [arr[i], arr[p]] = [arr[p], arr[i]];
      i = p;
    }
  }

  pop(): T | undefined {
    const arr = this.data;
    if (arr.length === 0) return undefined;
    const top = arr[0];
    const last = arr.pop()!;
    if (arr.length > 0) {
      arr[0] = last;
      let i = 0;
      for (;;) {
        let best = i;
        const l = i * 2 + 1;
        const r = l + 1;
        if (l < arr.length && this.compare(arr[l], arr[best]) < 0) best = l;
        if (r < arr.length && this.compare(arr[r], arr[best]) < 0) best = r;

        [arr[i], arr[best]] = [arr[best], arr[i]];

      }
    }
    return top;
  }
}

function asFiniteNumber(value: unknown, fallback = 0): number {
  return typeof value === 'number' && Number.isFinite(value) ? value : fallback;
}

function buildTasks(rawTasks: any[]): StateTask[] {
  const tasks: StateTask[] = [];
  const idToIndex = new Map<string, number>();

  for (let i = 0; i < rawTasks.length; i++) {
    const raw = rawTasks[i] ?? {};
    const id = String(raw.id);
    idToIndex.set(id, i);
    tasks.push({
      idx: i,
      id,
      duration: Math.max(0, Math.floor(asFiniteNumber(raw.duration, 0))),
      deadline: Math.floor(asFiniteNumber(raw.deadline, Number.MAX_SAFE_INTEGER / 4)),
      weight: Math.max(0, asFiniteNumber(raw.weight, 0)),
      deps: [],
      succs: [],
      indegree: 0,
      depCount: 0,
      level: 0,
      bottom: 0,
      release: 0,
      remainingDeps: 0,
    });
  }

  for (let i = 0; i < rawTasks.length; i++) {
    const raw = rawTasks[i] ?? {};
    const depIds = Array.isArray(raw.deps) ? raw.deps : [];
    const seen = new Set<number>();
    for (const depId of depIds) {
      const depIdx = idToIndex.get(String(depId));
      if (depIdx === undefined || depIdx === i || seen.has(depIdx)) continue;
      seen.add(depIdx);
      tasks[i].deps.push(depIdx);
      tasks[depIdx].succs.push(i);
    }
  }

  for (const t of tasks) {
    t.indegree = t.deps.length;
    t.depCount = t.deps.length;
    t.remainingDeps = t.deps.length;
  }

  return tasks;
}

function topoOrder(tasks: StateTask[]): number[] {
  const indegree = tasks.map((t) => t.indegree);
  const heap = new MaxHeap<number>((a, b) => {
    const ta = tasks[a];
    const tb = tasks[b];
    if (ta.weight !== tb.weight) return ta.weight - tb.weight;
    if (ta.duration !== tb.duration) return ta.duration - tb.duration;
    return tb.idx - ta.idx;
  });

  for (const t of tasks) {
    if (indegree[t.idx] === 0) heap.push(t.idx);
  }

  const order: number[] = [];
  while (heap.size() > 0) {
    const idx = heap.pop()!;
    order.push(idx);
    for (const s of tasks[idx].succs) {
      indegree[s]--;
      if (indegree[s] === 0) heap.push(s);
    }
  }
  return order;
}

function computeMetrics(tasks: StateTask[], topo: number[]): void {
  for (let i = topo.length - 1; i >= 0; i--) {
    const idx = topo[i];
    const task = tasks[idx];
    let maxBottom = 0;
    let maxLevel = 0;
    for (const s of task.succs) {
      if (tasks[s].bottom > maxBottom) maxBottom = tasks[s].bottom;
      if (tasks[s].level + 1 > maxLevel) maxLevel = tasks[s].level + 1;
    }
    task.bottom = task.duration + maxBottom;
    task.level = maxLevel;
  }
}

function makePriority(tasks: StateTask[], machineCount: number, avgDur: number) {
  const pressureScale = Math.max(1, avgDur * Math.max(1, machineCount * 0.5));
  return (aIdx: number, bIdx: number, now: number) => {
    const a = tasks[aIdx];
    const b = tasks[bIdx];
    const slackA = a.deadline - (now + a.bottom);
    const slackB = b.deadline - (now + b.bottom);
    const urgA = slackA < 0 ? (-slackA) * (a.weight + 1) : (a.weight + 1) / (slackA + pressureScale);
    const urgB = slackB < 0 ? (-slackB) * (b.weight + 1) : (b.weight + 1) / (slackB + pressureScale);
    if (urgA !== urgB) return urgA - urgB;
    const scoreA = a.weight * 8 + a.bottom * 2 + a.level * 3 - a.duration * 0.15 - a.depCount * 0.05;
    const scoreB = b.weight * 8 + b.bottom * 2 + b.level * 3 - b.duration * 0.15 - b.depCount * 0.05;
    if (scoreA !== scoreB) return scoreA - scoreB;
    if (a.weight !== b.weight) return a.weight - b.weight;
    if (a.bottom !== b.bottom) return a.bottom - b.bottom;
    if (a.duration !== b.duration) return a.duration - b.duration;
    return b.idx - a.idx;
  };
}

function scheduleTasks(tasks: StateTask[], machineCount: number): ScheduleEntry[] {
  const n = tasks.length;
  if (n === 0) return [];
  const topo = topoOrder(tasks);
  if (topo.length !== n) {
    return topo.map((idx, pos) => ({ id: tasks[idx].id, start: pos, machine: pos % Math.max(1, machineCount) }));
  }

  computeMetrics(tasks, topo);

  const totalDuration = tasks.reduce((sum, t) => sum + t.duration, 0);
  const avgDur = totalDuration / Math.max(1, n);

  for (const task of tasks) {
    task.release = 0;
    task.remainingDeps = task.indegree;
  }

  const machineAvail = new Array<number>(Math.max(1, machineCount)).fill(0);
  const availableAt = new MinHeap<number>((a, b) => {
    if (machineAvail[a] !== machineAvail[b]) return machineAvail[a] - machineAvail[b];
    return a - b;
  });
  for (let m = 0; m < machineAvail.length; m++) availableAt.push(m);

  const comparePriority = makePriority(tasks, machineAvail.length, avgDur);
  const ready = new MaxHeap<number>((a, b) => comparePriority(a, b, currentTime));
  const future = new MinHeap<number>((a, b) => {
    const ra = tasks[a].release;
    const rb = tasks[b].release;
    if (ra !== rb) return ra - rb;
    return a - b;
  });

  for (const task of tasks) {
    if (task.remainingDeps === 0) ready.push(task.idx);
  }

  const schedule: ScheduleEntry[] = [];
  let scheduled = 0;
  let currentTime = 0;

  while (scheduled < n) {
    while (future.size() > 0) {
      const idx = future.pop()!;
      if (tasks[idx].release > currentTime) {
        future.push(idx);
        break;
      }
      ready.push(idx);
    }

    const nextMachine = availableAt.pop()!;
    if (machineAvail[nextMachine] > currentTime) currentTime = machineAvail[nextMachine];

    while (future.size() > 0) {
      const idx = future.pop()!;
      if (tasks[idx].release > currentTime) {
        future.push(idx);
        break;
      }
      ready.push(idx);
    }

    if (ready.size() === 0) {
      let nextTime = Number.MAX_SAFE_INTEGER;
      if (future.size() > 0) {
        const idx = future.pop()!;
        nextTime = Math.min(nextTime, tasks[idx].release);
        future.push(idx);
      }
      nextTime = Math.min(nextTime, machineAvail[nextMachine]);
      if (!Number.isFinite(nextTime) || nextTime === Number.MAX_SAFE_INTEGER) break;
      currentTime = Math.max(currentTime, nextTime);
      availableAt.push(nextMachine);
      continue;
    }

    const taskIdx = ready.pop()!;
    const task = tasks[taskIdx];
    const start = Math.max(currentTime, machineAvail[nextMachine], task.release);
    const finish = start + task.duration;

    schedule.push({ id: task.id, start, machine: nextMachine });
    machineAvail[nextMachine] = finish;
    availableAt.push(nextMachine);
    scheduled++;

    for (const succIdx of task.succs) {
      const succ = tasks[succIdx];
      if (finish > succ.release) succ.release = finish;
      succ.remainingDeps--;
      if (succ.remainingDeps === 0) {
        if (succ.release <= currentTime) ready.push(succIdx);
        else future.push(succIdx);
      }
    }
  }

  if (schedule.length !== n) {
    const used = new Set(schedule.map((x) => x.id));
    const remaining = topo.filter((idx) => !used.has(tasks[idx].id));
    let fallbackTime = machineAvail.reduce((a, b) => Math.min(a, b), 0);
    for (const idx of remaining) {
      schedule.push({ id: tasks[idx].id, start: fallbackTime++, machine: fallbackTime % machineAvail.length });
    }
  }

  schedule.sort((a, b) => a.start - b.start || a.machine - b.machine || String(a.id).localeCompare(String(b.id)));
  return schedule;
}

function main(): void {
  const machineCountRaw = Math.floor(asFiniteNumber(data.machineCount, 1));
  const machineCount = Math.max(1, machineCountRaw);
  const rawTasks = Array.isArray(data.tasks) ? data.tasks : [];
  const tasks = buildTasks(rawTasks);
  const schedule = scheduleTasks(tasks, machineCount);
  process.stdout.write(JSON.stringify({ schedule }));
}

main();
