using System;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var flow = new int[n, n];
        var dist = new int[n, n];

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                flow[i, j] = scanner.NextInt();
            }
        }

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                dist[i, j] = scanner.NextInt();
            }
        }

        int[] permutation = Solve(flow, dist, n);
        long cost = ComputeCost(flow, dist, permutation);

        var sb = new StringBuilder();
        sb.AppendLine(cost.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(permutation[i]);
        }

        Console.Write(sb.ToString());
    }

    private static int[] Solve(int[,] flow, int[,] dist, int n)
    {
        if (n <= 10)
        {
            return SolveExact(flow, dist, n);
        }

        long[] facilityWeight = new long[n];
        long[] locationWeight = new long[n];

        for (int i = 0; i < n; i++)
        {
            long fw = 0;
            long lw = 0;
            for (int j = 0; j < n; j++)
            {
                fw += flow[i, j];
                fw += flow[j, i];
                lw += dist[i, j];
                lw += dist[j, i];
            }

            facilityWeight[i] = fw;
            locationWeight[i] = lw;
        }

        int[] permutation = BuildGreedyAssignment(facilityWeight, locationWeight);

        int candidateCount = n <= 80 ? n - 1 : Math.Min(64, n - 1);
        var nearestLocations = BuildNearestLocations(dist, n, candidateCount);

        long currentCost = ComputeCost(flow, dist, permutation);
        var facilityOfLocation = new int[n];
        for (int facility = 0; facility < n; facility++)
        {
            facilityOfLocation[permutation[facility]] = facility;
        }

        var timer = new TimeBudget(n);
        bool improved = true;

        while (improved && !timer.IsExpired)
        {
            improved = false;
            long bestDelta = 0;
            int bestA = -1;
            int bestB = -1;

            for (int a = 0; a < n && !timer.IsExpired; a++)
            {
                int locA = permutation[a];
                var candidates = nearestLocations[locA];
                for (int idx = 0; idx < candidates.Length; idx++)
                {
                    int locB = candidates[idx];
                    int b = facilityOfLocation[locB];
                    if (a >= b)
                    {
                        continue;
                    }

                    long delta = ComputeSwapDelta(flow, dist, permutation, a, b);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestA = a;
                        bestB = b;
                    }
                }
            }

            if (bestDelta < 0)
            {
                ApplySwap(permutation, facilityOfLocation, bestA, bestB);
                currentCost += bestDelta;
                improved = true;
            }
        }

        if (!timer.IsExpired && n <= 400)
        {
            ImproveByRandomSwaps(flow, dist, permutation, facilityOfLocation, ref currentCost, timer);
        }

        return permutation;
    }

    private static int[] SolveExact(int[,] flow, int[,] dist, int n)
    {
        var used = new bool[n];
        var current = new int[n];
        var best = new int[n];
        long bestCost = long.MaxValue;

        void Dfs(int depth, long currentCost)
        {
            if (currentCost >= bestCost)
            {
                return;
            }

            if (depth == n)
            {
                bestCost = currentCost;
                Array.Copy(current, best, n);
                return;
            }

            for (int loc = 0; loc < n; loc++)
            {
                if (used[loc])
                {
                    continue;
                }

                long add = 0;
                for (int i = 0; i < depth; i++)
                {
                    add += (long)flow[depth, i] * dist[loc, current[i]];
                    add += (long)flow[i, depth] * dist[current[i], loc];
                }

                used[loc] = true;
                current[depth] = loc;
                Dfs(depth + 1, currentCost + add + (long)flow[depth, depth] * dist[loc, loc]);
                used[loc] = false;
            }
        }

        Dfs(0, 0);
        return best;
    }

    private static int[] BuildGreedyAssignment(long[] facilityWeight, long[] locationWeight)
    {
        int n = facilityWeight.Length;
        var facilities = new int[n];
        var locations = new int[n];
        for (int i = 0; i < n; i++)
        {
            facilities[i] = i;
            locations[i] = i;
        }

        Array.Sort(facilities, (a, b) =>
        {
            int cmp = facilityWeight[b].CompareTo(facilityWeight[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        Array.Sort(locations, (a, b) =>
        {
            int cmp = locationWeight[a].CompareTo(locationWeight[b]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        var permutation = new int[n];
        for (int i = 0; i < n; i++)
        {
            permutation[facilities[i]] = locations[i];
        }

        return permutation;
    }

    private static int[][] BuildNearestLocations(int[,] dist, int n, int candidateCount)
    {
        var result = new int[n][];
        for (int i = 0; i < n; i++)
        {
            var order = new int[n - 1];
            int ptr = 0;
            for (int j = 0; j < n; j++)
            {
                if (i != j)
                {
                    order[ptr++] = j;
                }
            }

            Array.Sort(order, (a, b) =>
            {
                int cmp = dist[i, a].CompareTo(dist[i, b]);
                return cmp != 0 ? cmp : a.CompareTo(b);
            });

            int len = Math.Min(candidateCount, order.Length);
            var trimmed = new int[len];
            Array.Copy(order, trimmed, len);
            result[i] = trimmed;
        }

        return result;
    }

    private static void ImproveByRandomSwaps(int[,] flow, int[,] dist, int[] permutation, int[] facilityOfLocation, ref long currentCost, TimeBudget timer)
    {
        int n = permutation.Length;
        int attempts = Math.Min(20000, Math.Max(2000, n * 40));
        var rng = new Random(1);

        for (int step = 0; step < attempts && !timer.IsExpired; step++)
        {
            int a = rng.Next(n);
            int b = rng.Next(n - 1);
            if (b >= a) b++;
            if (a > b)
            {
                int tmp = a;
                a = b;
                b = tmp;
            }

            long delta = ComputeSwapDelta(flow, dist, permutation, a, b);
            if (delta < 0)
            {
                ApplySwap(permutation, facilityOfLocation, a, b);
                currentCost += delta;
            }
        }
    }

    private static void ApplySwap(int[] permutation, int[] facilityOfLocation, int a, int b)
    {
        int locA = permutation[a];
        int locB = permutation[b];
        permutation[a] = locB;
        permutation[b] = locA;
        facilityOfLocation[locA] = b;
        facilityOfLocation[locB] = a;
    }

    private static long ComputeSwapDelta(int[,] flow, int[,] dist, int[] permutation, int a, int b)
    {
        int n = permutation.Length;
        int locA = permutation[a];
        int locB = permutation[b];
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b)
            {
                continue;
            }

            int locK = permutation[k];
            delta += (long)flow[a, k] * (dist[locB, locK] - dist[locA, locK]);
            delta += (long)flow[b, k] * (dist[locA, locK] - dist[locB, locK]);
            delta += (long)flow[k, a] * (dist[locK, locB] - dist[locK, locA]);
            delta += (long)flow[k, b] * (dist[locK, locA] - dist[locK, locB]);
        }

        delta += (long)flow[a, a] * (dist[locB, locB] - dist[locA, locA]);
        delta += (long)flow[b, b] * (dist[locA, locA] - dist[locB, locB]);
        delta += (long)flow[a, b] * (dist[locB, locA] - dist[locA, locB]);
        delta += (long)flow[b, a] * (dist[locA, locB] - dist[locB, locA]);

        return delta;
    }

    private static long ComputeCost(int[,] flow, int[,] dist, int[] permutation)
    {
        int n = permutation.Length;
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = permutation[i];
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[i, j] * dist[pi, permutation[j]];
            }
        }

        return cost;
    }

    private sealed class TimeBudget
    {
        private readonly long startTicks;
        private readonly long limitTicks;

        public TimeBudget(int n)
        {
            startTicks = DateTime.UtcNow.Ticks;
            double seconds = n <= 200 ? 1.2 : n <= 800 ? 8.0 : 20.0;
            limitTicks = (long)(seconds * TimeSpan.TicksPerSecond);
        }

        public bool IsExpired
        {
            get { return DateTime.UtcNow.Ticks - startTicks >= limitTicks; }
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int length;
        private int ptr;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1)
                {
                    return 0;
                }
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (ptr >= length)
            {
                length = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (length <= 0)
                {
                    return -1;
                }
            }

            return buffer[ptr++];
        }
    }
}
