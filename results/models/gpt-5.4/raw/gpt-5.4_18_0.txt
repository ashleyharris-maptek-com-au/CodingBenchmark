#include <algorithm>
#include <array>
#include <cstdint>
#include <fstream>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <string>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

namespace {

struct Edge {
    uint32_t u;
    uint32_t v;
};

struct DSU {
    vector<uint32_t> p;
    vector<uint32_t> sz;

    explicit DSU(uint32_t n = 0) : p(n), sz(n, 1) {
        iota(p.begin(), p.end(), 0u);
    }

    uint32_t find(uint32_t x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }

    bool unite(uint32_t a, uint32_t b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (sz[a] < sz[b]) swap(a, b);
        p[b] = a;
        sz[a] += sz[b];
        return true;
    }
};

static uint64_t choose_memory_budget_bytes() {
    // Conservative default for a 32 GB machine. The solver deliberately stays well
    // below physical RAM to leave headroom for allocator overhead and OS caching.
    return 1500ULL * 1024ULL * 1024ULL;
}

static bool file_exists(const string& path) {
    ifstream f(path, ios::binary);
    return f.good();
}

static string make_temp_edge_file() {
    constexpr const char* candidates[] = {
        "mincut_edges.tmp",
        "mincut_edges_pass1.tmp",
        "mincut_edges_spool.tmp"
    };
    for (const char* path : candidates) {
        if (!file_exists(path)) return string(path);
    }
    return "mincut_edges_work.tmp";
}

static vector<int> bfs_levels(const vector<vector<uint32_t>>& adj, uint32_t src) {
    vector<int> dist(adj.size(), -1);
    queue<uint32_t> q;
    dist[src] = 0;
    q.push(src);
    while (!q.empty()) {
        uint32_t u = q.front();
        q.pop();
        for (uint32_t v : adj[u]) {
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    return dist;
}

static vector<char> greedy_component_from_seed(const vector<vector<uint32_t>>& adj, uint32_t seed) {
    const uint32_t n = static_cast<uint32_t>(adj.size());
    vector<char> in_a(n, 0);
    vector<uint32_t> internal(n, 0), external(n, 0);
    vector<uint32_t> frontier;
    frontier.reserve(n);

    in_a[seed] = 1;
    for (uint32_t v : adj[seed]) {
        external[v] += 1;
        frontier.push_back(v);
    }

    vector<char> queued(n, 0);
    for (uint32_t v : frontier) queued[v] = 1;

    while (!frontier.empty()) {
        uint32_t best = numeric_limits<uint32_t>::max();
        int64_t best_gain = numeric_limits<int64_t>::min();
        size_t best_pos = 0;

        for (size_t i = 0; i < frontier.size(); ++i) {
            uint32_t v = frontier[i];
            if (in_a[v]) continue;
            int64_t gain = static_cast<int64_t>(external[v]) - static_cast<int64_t>(adj[v].size() - external[v]);
            if (gain > best_gain) {
                best_gain = gain;
                best = v;
                best_pos = i;
            }
        }

        if (best == numeric_limits<uint32_t>::max() || best_gain <= 0) break;

        frontier[best_pos] = frontier.back();
        frontier.pop_back();
        queued[best] = 0;
        in_a[best] = 1;

        for (uint32_t to : adj[best]) {
            if (in_a[to]) {
                internal[best] += 1;
            } else {
                external[to] += 1;
                if (!queued[to]) {
                    queued[to] = 1;
                    frontier.push_back(to);
                }
            }
        }
    }

    uint32_t count_a = 0;
    for (char x : in_a) count_a += (x != 0);
    if (count_a == 0 || count_a == n) {
        fill(in_a.begin(), in_a.end(), 0);
        in_a[seed] = 1;
    }
    return in_a;
}

static vector<char> heuristic_partition(const vector<vector<uint32_t>>& adj) {
    const uint32_t n = static_cast<uint32_t>(adj.size());
    if (n <= 1) return vector<char>(n, 0);

    uint32_t seed = 0;
    for (uint32_t i = 1; i < n; ++i) {
        if (adj[i].size() < adj[seed].size()) seed = i;
    }

    vector<char> best = greedy_component_from_seed(adj, seed);
    size_t best_cut = numeric_limits<size_t>::max();

    auto eval_cut = [&](const vector<char>& side) -> size_t {
        size_t cut = 0;
        for (uint32_t u = 0; u < n; ++u) {
            for (uint32_t v : adj[u]) {
                if (u < v && side[u] != side[v]) ++cut;
            }
        }
        return cut;
    };

    best_cut = eval_cut(best);

    vector<uint32_t> probes;
    probes.push_back(seed);
    probes.push_back(static_cast<uint32_t>(max_element(adj.begin(), adj.end(), [](const auto& a, const auto& b) {
        return a.size() < b.size();
    }) - adj.begin()));

    auto dist = bfs_levels(adj, probes.back());
    uint32_t far = probes.back();
    for (uint32_t i = 0; i < n; ++i) {
        if (dist[i] > dist[far]) far = i;
    }
    probes.push_back(far);

    for (uint32_t s : probes) {
        vector<char> cand = greedy_component_from_seed(adj, s);
        size_t cut = eval_cut(cand);
        if (cut < best_cut) {
            best_cut = cut;
            best.swap(cand);
        }
    }
    return best;
}

static vector<Edge> cut_edges_from_partition(const vector<Edge>& edges, const vector<char>& side) {
    vector<Edge> cut;
    for (const Edge& e : edges) {
        if (side[e.u] != side[e.v]) cut.push_back(e);
    }
    return cut;
}

static vector<Edge> cut_edges_from_partition_stream(const string& edge_file, const vector<char>& side, uint64_t m) {
    vector<Edge> cut;
    cut.reserve(1024);
    ifstream fin(edge_file, ios::binary);
    for (uint64_t i = 0; i < m; ++i) {
        Edge e{};
        fin.read(reinterpret_cast<char*>(&e), sizeof(Edge));
        if (side[e.u] != side[e.v]) cut.push_back(e);
    }
    return cut;
}

static pair<uint64_t, vector<char>> stoer_wagner_dense(vector<vector<uint32_t>> w) {
    const uint32_t n = static_cast<uint32_t>(w.size());
    vector<vector<uint32_t>> groups(n);
    for (uint32_t i = 0; i < n; ++i) groups[i] = {i};

    uint64_t best = numeric_limits<uint64_t>::max();
    vector<char> best_side(n, 0);
    vector<uint32_t> alive(n);
    iota(alive.begin(), alive.end(), 0u);

    while (alive.size() > 1) {
        vector<uint64_t> weight(n, 0);
        vector<char> added(n, 0);
        uint32_t prev = alive[0];
        uint32_t last = alive[0];

        for (size_t step = 0; step < alive.size(); ++step) {
            uint32_t sel = numeric_limits<uint32_t>::max();
            uint64_t best_w = 0;
            for (uint32_t v : alive) {
                if (!added[v] && (sel == numeric_limits<uint32_t>::max() || weight[v] > best_w)) {
                    sel = v;
                    best_w = weight[v];
                }
            }

            if (step + 1 == alive.size()) {
                if (best_w < best) {
                    best = best_w;
                    fill(best_side.begin(), best_side.end(), 0);
                    for (uint32_t x : groups[sel]) best_side[x] = 1;
                }

                for (uint32_t v : alive) {
                    if (v == sel || v == prev) continue;
                    w[prev][v] += w[sel][v];
                    w[v][prev] = w[prev][v];
                }
                groups[prev].insert(groups[prev].end(), groups[sel].begin(), groups[sel].end());
                alive.erase(find(alive.begin(), alive.end(), sel));
                break;
            }

            added[sel] = 1;
            prev = last;
            last = sel;
            for (uint32_t v : alive) {
                if (!added[v]) weight[v] += w[sel][v];
            }
        }
    }
    return {best, best_side};
}

static pair<uint64_t, vector<char>> exact_min_cut_small(uint32_t n, const vector<Edge>& edges) {
    vector<vector<uint32_t>> w(n, vector<uint32_t>(n, 0));
    for (const Edge& e : edges) {
        ++w[e.u][e.v];
        ++w[e.v][e.u];
    }
    return stoer_wagner_dense(std::move(w));
}

}  // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    uint64_t n64, m;
    if (!(cin >> n64 >> m)) {
        return 0;
    }
    if (n64 == 0) {
        cout << 0 << '\n';
        return 0;
    }
    if (n64 > numeric_limits<uint32_t>::max()) {
        return 0;
    }
    const uint32_t n = static_cast<uint32_t>(n64);

    const uint64_t mem_budget = choose_memory_budget_bytes();
    const uint64_t approx_adj_bytes = 2ULL * m * sizeof(uint32_t);
    const uint64_t approx_edge_bytes = m * sizeof(Edge);
    const bool can_store_full_graph = approx_adj_bytes + approx_edge_bytes + 32ULL * n <= mem_budget;
    const bool can_run_exact_dense = (static_cast<uint64_t>(n) * static_cast<uint64_t>(n) * sizeof(uint32_t) <= mem_budget / 2ULL) && (n <= 6000);

    vector<Edge> edges;
    vector<vector<uint32_t>> adj;
    string edge_file;

    if (can_store_full_graph) {
        edges.reserve(static_cast<size_t>(m));
        adj.assign(n, {});

        vector<uint32_t> degree(n, 0);
        for (uint64_t i = 0; i < m; ++i) {
            uint32_t u, v;
            cin >> u >> v;
            edges.push_back({u, v});
            ++degree[u];
            ++degree[v];
        }

        for (uint32_t i = 0; i < n; ++i) adj[i].reserve(degree[i]);
        for (const Edge& e : edges) {
            adj[e.u].push_back(e.v);
            adj[e.v].push_back(e.u);
        }
    } else {
        edge_file = make_temp_edge_file();
        ofstream fout(edge_file, ios::binary | ios::trunc);
        if (!fout) return 0;

        adj.assign(n, {});
        vector<uint32_t> degree(n, 0);
        for (uint64_t i = 0; i < m; ++i) {
            Edge e{};
            cin >> e.u >> e.v;
            fout.write(reinterpret_cast<const char*>(&e), sizeof(Edge));
            ++degree[e.u];
            ++degree[e.v];
        }
        fout.close();

        uint64_t degree_sum = 0;
        for (uint32_t d : degree) degree_sum += d;
        if (degree_sum * sizeof(uint32_t) > mem_budget / 2ULL) {
            // Memory is too tight for full adjacency. Fall back to a guaranteed-valid cut:
            // isolate the minimum-degree vertex by streaming the edge list once.
            uint32_t best_v = 0;
            for (uint32_t i = 1; i < n; ++i) {
                if (degree[i] < degree[best_v]) best_v = i;
            }

            vector<Edge> cut;
            cut.reserve(degree[best_v]);
            ifstream fin(edge_file, ios::binary);
            for (uint64_t i = 0; i < m; ++i) {
                Edge e{};
                fin.read(reinterpret_cast<char*>(&e), sizeof(Edge));
                if (e.u == best_v || e.v == best_v) cut.push_back(e);
            }

            cout << cut.size() << '\n';
            for (const Edge& e : cut) cout << e.u << ' ' << e.v << '\n';
            return 0;
        }

        for (uint32_t i = 0; i < n; ++i) adj[i].reserve(degree[i]);
        ifstream fin(edge_file, ios::binary);
        for (uint64_t i = 0; i < m; ++i) {
            Edge e{};
            fin.read(reinterpret_cast<char*>(&e), sizeof(Edge));
            adj[e.u].push_back(e.v);
            adj[e.v].push_back(e.u);
        }
    }

    if (can_store_full_graph && can_run_exact_dense) {
        auto [value, side] = exact_min_cut_small(n, edges);
        vector<Edge> cut = cut_edges_from_partition(edges, side);
        cout << cut.size() << '\n';
        for (const Edge& e : cut) cout << e.u << ' ' << e.v << '\n';
        return 0;
    }

    vector<char> side = heuristic_partition(adj);
    uint32_t cnt_a = 0;
    for (char x : side) cnt_a += (x != 0);
    if (cnt_a == 0 || cnt_a == n) {
        uint32_t best_v = 0;
        for (uint32_t i = 1; i < n; ++i) {
            if (adj[i].size() < adj[best_v].size()) best_v = i;
        }
        fill(side.begin(), side.end(), 0);
        side[best_v] = 1;
    }

    vector<Edge> cut = can_store_full_graph ? cut_edges_from_partition(edges, side)
                                            : cut_edges_from_partition_stream(edge_file, side, m);

    cout << cut.size() << '\n';
    for (const Edge& e : cut) {
        cout << e.u << ' ' << e.v << '\n';
    }
    return 0;
}
