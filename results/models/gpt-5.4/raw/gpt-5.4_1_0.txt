#include <array>

struct City {
    float x;
    float y;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static std::vector<int> brute_force_tsp(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());
    if (n <= 1) return {0};

    std::vector<int> perm(n - 1);
    std::iota(perm.begin(), perm.end(), 1);
    std::vector<int> best = perm;
    double best_cost = std::numeric_limits<double>::infinity();

    do {
        double cost = 0.0;
        int prev = 0;
        for (int idx : perm) {
            cost += dist2(cities[prev], cities[idx]);
            prev = idx;
        }
        cost += dist2(cities[prev], cities[0]);
        if (cost < best_cost) {
            best_cost = cost;
            best = perm;
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    route.insert(route.end(), best.begin(), best.end());
    return route;
}

static std::vector<int> nearest_neighbor(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());
    std::vector<int> route;
    route.reserve(n);
    std::vector<unsigned char> used(n, 0);
    int current = 0;
    route.push_back(0);
    used[0] = 1;

    for (int step = 1; step < n; ++step) {
        int best = -1;
        double best_d = std::numeric_limits<double>::infinity();
        for (int i = 1; i < n; ++i) {
            if (used[i]) continue;
            const double d = dist2(cities[current], cities[i]);
            if (d < best_d) {
                best_d = d;
                best = i;
            }
        }
        if (best == -1) break;
        used[best] = 1;
        route.push_back(best);
        current = best;
    }
    return route;
}

static void two_opt_limited(const std::vector<City>& cities, std::vector<int>& route, int max_passes) {
    const int n = static_cast<int>(route.size());
    if (n < 4) return;

    for (int pass = 0; pass < max_passes; ++pass) {
        bool improved = false;
        for (int i = 0; i + 2 < n; ++i) {
            const int a = route[i];
            const int b = route[(i + 1) % n];
            const int j_limit = (i == 0) ? n - 2 : n - 1;
            for (int j = i + 2; j < j_limit; ++j) {
                const int c = route[j];
                const int d = route[(j + 1) % n];
                const double before = dist2(cities[a], cities[b]) + dist2(cities[c], cities[d]);
                const double after = dist2(cities[a], cities[c]) + dist2(cities[b], cities[d]);
                if (after + 1e-12 < before) {
                    std::reverse(route.begin() + i + 1, route.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

struct GridKeyHash {
    std::size_t operator()(const std::pair<int, int>& p) const noexcept {
        const std::uint64_t x = static_cast<std::uint32_t>(p.first);
        const std::uint64_t y = static_cast<std::uint32_t>(p.second);
        return static_cast<std::size_t>((x << 32) ^ y);
    }
};

static std::vector<int> grid_greedy_route(const std::vector<City>& cities, int target_bucket_pop) {
    const int n = static_cast<int>(cities.size());
    if (n <= 1) return {0};

    float min_x = cities[0].x, max_x = cities[0].x;
    float min_y = cities[0].y, max_y = cities[0].y;
    for (const auto& c : cities) {
        min_x = std::min(min_x, c.x);
        max_x = std::max(max_x, c.x);
        min_y = std::min(min_y, c.y);
        max_y = std::max(max_y, c.y);
    }

    const double width = std::max(1e-9, static_cast<double>(max_x) - min_x);
    const double height = std::max(1e-9, static_cast<double>(max_y) - min_y);
    const double area = width * height;
    const double cell_size = std::max(1e-9, std::sqrt(area * static_cast<double>(target_bucket_pop) / std::max(1, n)));

    std::unordered_map<std::pair<int, int>, std::vector<int>, GridKeyHash> grid;
    grid.reserve(static_cast<std::size_t>(n * 1.3));

    auto cell_of = [&](const City& c) -> std::pair<int, int> {
        const int gx = static_cast<int>((static_cast<double>(c.x) - min_x) / cell_size);
        const int gy = static_cast<int>((static_cast<double>(c.y) - min_y) / cell_size);
        return {gx, gy};
    };

    for (int i = 0; i < n; ++i) {
        grid[cell_of(cities[i])].push_back(i);
    }

    std::vector<unsigned char> used(n, 0);
    std::vector<int> route;
    route.reserve(n);
    int current = 0;
    used[0] = 1;
    route.push_back(0);

    auto try_pick_from_bucket = [&](const std::vector<int>& bucket, int from, int& best_idx, double& best_d) {
        for (int idx : bucket) {
            if (used[idx]) continue;
            const double d = dist2(cities[from], cities[idx]);
            if (d < best_d) {
                best_d = d;
                best_idx = idx;
            }
        }
    };

    for (int step = 1; step < n; ++step) {
        int best = -1;
        double best_d = std::numeric_limits<double>::infinity();
        const auto [cx, cy] = cell_of(cities[current]);

        for (int radius = 0; radius <= 64 && best == -1; ++radius) {
            for (int dx = -radius; dx <= radius; ++dx) {
                for (int dy = -radius; dy <= radius; ++dy) {
                    if (std::max(std::abs(dx), std::abs(dy)) != radius) continue;
                    auto it = grid.find({cx + dx, cy + dy});
                    if (it != grid.end()) {
                        try_pick_from_bucket(it->second, current, best, best_d);
                    }
                }
            }
        }

        if (best == -1) {
            for (int i = 1; i < n; ++i) {
                if (used[i]) continue;
                const double d = dist2(cities[current], cities[i]);
                if (d < best_d) {
                    best_d = d;
                    best = i;
                }
            }
        }

        used[best] = 1;
        route.push_back(best);
        current = best;
    }

    return route;
}

static std::vector<int> sort_by_space_filling_curve(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());
    if (n <= 1) return {0};

    float min_x = cities[0].x, max_x = cities[0].x;
    float min_y = cities[0].y, max_y = cities[0].y;
    for (const auto& c : cities) {
        min_x = std::min(min_x, c.x);
        max_x = std::max(max_x, c.x);
        min_y = std::min(min_y, c.y);
        max_y = std::max(max_y, c.y);
    }

    const double scale_x = (max_x > min_x) ? (4294967295.0 / (static_cast<double>(max_x) - min_x)) : 0.0;
    const double scale_y = (max_y > min_y) ? (4294967295.0 / (static_cast<double>(max_y) - min_y)) : 0.0;

    auto part1by1 = [](std::uint32_t x) -> std::uint64_t {
        std::uint64_t v = x;
        v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
        v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
        v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
        v = (v | (v << 2)) & 0x3333333333333333ULL;
        v = (v | (v << 1)) & 0x5555555555555555ULL;
        return v;
    };

    std::vector<std::pair<std::uint64_t, int>> keyed;
    keyed.reserve(n - 1);
    for (int i = 1; i < n; ++i) {
        const std::uint32_t xi = static_cast<std::uint32_t>((static_cast<double>(cities[i].x) - min_x) * scale_x);
        const std::uint32_t yi = static_cast<std::uint32_t>((static_cast<double>(cities[i].y) - min_y) * scale_y);
        const std::uint64_t key = part1by1(xi) | (part1by1(yi) << 1);
        keyed.push_back({key, i});
    }

    std::sort(keyed.begin(), keyed.end(), [](const auto& a, const auto& b) {
        return a.first < b.first;
    });

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    for (const auto& [key, idx] : keyed) {
        (void)key;
        route.push_back(idx);
    }
    return route;
}

static std::vector<int> adaptive_tsp(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());
    if (n == 0) return {};
    if (n == 1) return {0};

    if (n <= 10) {
        return brute_force_tsp(cities);
    }
    if (n <= 2500) {
        auto route = nearest_neighbor(cities);
        two_opt_limited(cities, route, 2);
        return route;
    }
    if (n <= 200000) {
        return grid_greedy_route(cities, 32);
    }
    return sort_by_space_filling_curve(cities);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    if (!(std::cin >> n)) {
        return 0;
    }

    std::vector<City> cities(static_cast<std::size_t>(n));
    for (int i = 0; i < n; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
    }

    std::vector<int> route = adaptive_tsp(cities);
    if (route.empty()) {
        return 0;
    }

    if (route[0] != 0) {
        auto it = std::find(route.begin(), route.end(), 0);
        if (it != route.end()) {
            std::rotate(route.begin(), it, route.end());
        }
    }

    for (int i = 0; i < n; ++i) {
        if (i) std::cout << ' ';
        std::cout << route[i];
    }
    std::cout << '\n';
    return 0;
}
