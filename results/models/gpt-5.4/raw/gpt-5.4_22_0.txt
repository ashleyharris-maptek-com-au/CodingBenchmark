using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var reader = new FastReader(Console.In);
        int width = reader.NextInt();
        int height = reader.NextInt();
        int numPieces = reader.NextInt();

        var board = new Board(width, height);

        for (int pieceIndex = 0; pieceIndex < numPieces; pieceIndex++)
        {
            char piece = reader.NextToken()[0];
            var move = Solver.ChooseMove(board, piece, null, pieceIndex, numPieces);
            Console.WriteLine(move.Rotation.ToString(CultureInfo.InvariantCulture) + " " +
                              move.Column.ToString(CultureInfo.InvariantCulture));
            Console.Out.Flush();

            string result = reader.NextToken();
            if (!board.TryApply(piece, move.Rotation, move.Column, out _))
            {
                ApplyEmergencyPlacement(board, piece);
            }

            if (result.StartsWith("cleared_", StringComparison.OrdinalIgnoreCase))
            {
                // External engine is authoritative, but the internal board should already match.
            }
            else if (string.Equals(result, "gameover", StringComparison.OrdinalIgnoreCase))
            {
                break;
            }
        }
    }

    private static void ApplyEmergencyPlacement(Board board, char piece)
    {
        foreach (var placement in PieceSet.GetPlacements(piece, board.Width))
        {
            if (board.TryApply(piece, placement.Rotation, placement.Column, out _))
            {
                return;
            }
        }
    }
}

public readonly struct Move
{
    public readonly int Rotation;
    public readonly int Column;
    public readonly double Score;

    public Move(int rotation, int column, double score)
    {
        Rotation = rotation;
        Column = column;
        Score = score;
    }
}

public static class Solver
{
    public static Move ChooseMove(Board board, char piece, char? nextPiece, int pieceIndex, int totalPieces)
    {
        var placements = PieceSet.GetPlacements(piece, board.Width);
        Move best = default;
        bool found = false;

        int remaining = totalPieces - pieceIndex;
        int depth = SelectDepth(board, remaining, nextPiece.HasValue);
        var weights = SelectWeights(board, remaining);

        foreach (var placement in placements)
        {
            if (!board.TrySimulate(piece, placement.Rotation, placement.Column, out Board after, out int cleared))
            {
                continue;
            }

            double score = Evaluate(after, cleared, weights);
            if (depth >= 2 && nextPiece.HasValue)
            {
                score += 0.45 * BestReplyScore(after, nextPiece.Value, depth - 1, weights);
            }

            score += TieBreak(placement, board.Width);
            var move = new Move(placement.Rotation, placement.Column, score);
            if (!found || move.Score > best.Score)
            {
                best = move;
                found = true;
            }
        }

        if (found)
        {
            return best;
        }

        return new Move(0, 0, double.NegativeInfinity);
    }

    private static int SelectDepth(Board board, int remaining, bool hasNext)
    {
        if (!hasNext)
        {
            return 1;
        }

        int area = board.Width * board.Height;
        if (area <= 220 && remaining <= 120)
        {
            return 2;
        }

        if (board.Width <= 10 && remaining <= 300)
        {
            return 2;
        }

        return 1;
    }

    private static Weights SelectWeights(Board board, int remaining)
    {
        int maxHeight = board.MaxHeight();
        bool danger = maxHeight >= board.Height - 4;
        bool late = remaining < 40;

        if (danger)
        {
            return new Weights(
                aggregateHeight: -0.70,
                holes: -2.30,
                bumpiness: -0.32,
                lines: 1.35,
                wells: -0.18,
                rowTransitions: -0.28,
                colTransitions: -0.20,
                topHalf: -0.55,
                perfectClear: 2.50,
                tetrisReady: 0.10);
        }

        if (late)
        {
            return new Weights(
                aggregateHeight: -0.46,
                holes: -1.85,
                bumpiness: -0.22,
                lines: 1.10,
                wells: -0.08,
                rowTransitions: -0.20,
                colTransitions: -0.15,
                topHalf: -0.30,
                perfectClear: 3.20,
                tetrisReady: 0.12);
        }

        return new Weights(
            aggregateHeight: -0.50,
            holes: -1.95,
            bumpiness: -0.26,
            lines: 1.25,
            wells: -0.10,
            rowTransitions: -0.22,
            colTransitions: -0.16,
            topHalf: -0.35,
            perfectClear: 2.80,
            tetrisReady: 0.16);
    }

    private static double BestReplyScore(Board board, char piece, int depth, Weights weights)
    {
        double best = double.NegativeInfinity;
        foreach (var placement in PieceSet.GetPlacements(piece, board.Width))
        {
            if (!board.TrySimulate(piece, placement.Rotation, placement.Column, out Board next, out int cleared))
            {
                continue;
            }

            double score = Evaluate(next, cleared, weights) + TieBreak(placement, board.Width);
            if (depth > 1)
            {
                score += 0.20 * Evaluate(next, 0, weights);
            }

            if (score > best)
            {
                best = score;
            }
        }

        return double.IsNegativeInfinity(best) ? -1e9 : best;
    }

    private static double TieBreak(Placement placement, int width)
    {
        double center = (width - placement.Width) / 2.0;
        double centerBias = -0.002 * Math.Abs(placement.Column - center);
        double lowRotationBias = -0.0005 * placement.Rotation;
        return centerBias + lowRotationBias;
    }

    private static double Evaluate(Board board, int cleared, Weights weights)
    {
        int[] heights = board.ColumnHeights();
        int aggregateHeight = 0;
        int bumpiness = 0;
        for (int i = 0; i < heights.Length; i++)
        {
            aggregateHeight += heights[i];
            if (i + 1 < heights.Length)
            {
                bumpiness += Math.Abs(heights[i] - heights[i + 1]);
            }
        }

        int holes = board.CountHoles(heights);
        int wells = board.WellSum(heights);
        int rowTransitions = board.RowTransitions();
        int colTransitions = board.ColumnTransitions(heights);
        int topHalf = board.TopHalfCells();
        bool perfectClear = board.IsEmpty;
        int tetrisReady = board.TetrisReadyScore(heights);

        return aggregateHeight * weights.AggregateHeight +
               holes * weights.Holes +
               bumpiness * weights.Bumpiness +
               cleared * weights.Lines +
               wells * weights.Wells +
               rowTransitions * weights.RowTransitions +
               colTransitions * weights.ColTransitions +
               topHalf * weights.TopHalf +
               (perfectClear ? weights.PerfectClear : 0.0) +
               tetrisReady * weights.TetrisReady;
    }

    private readonly struct Weights
    {
        public readonly double AggregateHeight;
        public readonly double Holes;
        public readonly double Bumpiness;
        public readonly double Lines;
        public readonly double Wells;
        public readonly double RowTransitions;
        public readonly double ColTransitions;
        public readonly double TopHalf;
        public readonly double PerfectClear;
        public readonly double TetrisReady;

        public Weights(
            double aggregateHeight,
            double holes,
            double bumpiness,
            double lines,
            double wells,
            double rowTransitions,
            double colTransitions,
            double topHalf,
            double perfectClear,
            double tetrisReady)
        {
            AggregateHeight = aggregateHeight;
            Holes = holes;
            Bumpiness = bumpiness;
            Lines = lines;
            Wells = wells;
            RowTransitions = rowTransitions;
            ColTransitions = colTransitions;
            TopHalf = topHalf;
            PerfectClear = perfectClear;
            TetrisReady = tetrisReady;
        }
    }
}

public sealed class Board
{
    private readonly bool[,] cells;
    public int Width { get; }
    public int Height { get; }
    public bool IsEmpty { get; private set; }

    public Board(int width, int height)
    {
        Width = width;
        Height = height;
        cells = new bool[height, width];
        IsEmpty = true;
    }

    private Board(int width, int height, bool[,] source, bool isEmpty)
    {
        Width = width;
        Height = height;
        cells = source;
        IsEmpty = isEmpty;
    }

    public Board Clone()
    {
        var copy = new bool[Height, Width];
        Array.Copy(cells, copy, cells.Length);
        return new Board(Width, Height, copy, IsEmpty);
    }

    public bool TryApply(char piece, int rotation, int column, out int cleared)
    {
        if (!TryPlace(PieceSet.GetShape(piece, rotation), column, out int landingRow))
        {
            cleared = 0;
            return false;
        }

        Stamp(PieceSet.GetShape(piece, rotation), column, landingRow);
        cleared = ClearLines();
        IsEmpty = ComputeIsEmpty();
        return true;
    }

    public bool TrySimulate(char piece, int rotation, int column, out Board board, out int cleared)
    {
        board = Clone();
        if (!board.TryApply(piece, rotation, column, out cleared))
        {
            return false;
        }

        return true;
    }

    private bool TryPlace(bool[,] shape, int column, out int landingRow)
    {
        landingRow = -1;
        int shapeHeight = shape.GetLength(0);
        int shapeWidth = shape.GetLength(1);
        if (column < 0 || column + shapeWidth > Width)
        {
            return false;
        }

        int lastValid = -1;
        for (int row = 0; row <= Height - shapeHeight; row++)
        {
            if (Collides(shape, column, row))
            {
                break;
            }

            lastValid = row;
        }

        if (lastValid < 0)
        {
            return false;
        }

        landingRow = lastValid;
        return true;
    }

    private bool Collides(bool[,] shape, int column, int row)
    {
        int shapeHeight = shape.GetLength(0);
        int shapeWidth = shape.GetLength(1);
        for (int y = 0; y < shapeHeight; y++)
        {
            for (int x = 0; x < shapeWidth; x++)
            {
                if (!shape[y, x])
                {
                    continue;
                }

                if (cells[row + y, column + x])
                {
                    return true;
                }
            }
        }

        return false;
    }

    private void Stamp(bool[,] shape, int column, int row)
    {
        int shapeHeight = shape.GetLength(0);
        int shapeWidth = shape.GetLength(1);
        for (int y = 0; y < shapeHeight; y++)
        {
            for (int x = 0; x < shapeWidth; x++)
            {
                if (shape[y, x])
                {
                    cells[row + y, column + x] = true;
                }
            }
        }

        IsEmpty = false;
    }

    private int ClearLines()
    {
        int cleared = 0;
        int writeRow = Height - 1;
        for (int row = Height - 1; row >= 0; row--)
        {
            bool full = true;
            for (int col = 0; col < Width; col++)
            {
                if (!cells[row, col])
                {
                    full = false;
                    break;
                }
            }

            if (full)
            {
                cleared++;
                continue;
            }

            if (writeRow != row)
            {
                for (int col = 0; col < Width; col++)
                {
                    cells[writeRow, col] = cells[row, col];
                }
            }

            writeRow--;
        }

        for (int row = writeRow; row >= 0; row--)
        {
            for (int col = 0; col < Width; col++)
            {
                cells[row, col] = false;
            }
        }

        return cleared;
    }

    private bool ComputeIsEmpty()
    {
        for (int row = 0; row < Height; row++)
        {
            for (int col = 0; col < Width; col++)
            {
                if (cells[row, col])
                {
                    return false;
                }
            }
        }

        return true;
    }

    public int[] ColumnHeights()
    {
        var heights = new int[Width];
        for (int col = 0; col < Width; col++)
        {
            int h = 0;
            for (int row = 0; row < Height; row++)
            {
                if (cells[row, col])
                {
                    h = Height - row;
                    break;
                }
            }

            heights[col] = h;
        }

        return heights;
    }

    public int MaxHeight()
    {
        int max = 0;
        for (int col = 0; col < Width; col++)
        {
            for (int row = 0; row < Height; row++)
            {
                if (cells[row, col])
                {
                    int h = Height - row;
                    if (h > max)
                    {
                        max = h;
                    }
                    break;
                }
            }
        }

        return max;
    }

    public int CountHoles(int[] heights)
    {
        int holes = 0;
        for (int col = 0; col < Width; col++)
        {
            bool seenBlock = false;
            for (int row = 0; row < Height; row++)
            {
                if (cells[row, col])
                {
                    seenBlock = true;
                }
                else if (seenBlock)
                {
                    holes++;
                }
            }
        }

        return holes;
    }

    public int WellSum(int[] heights)
    {
        int sum = 0;
        for (int col = 0; col < Width; col++)
        {
            int left = col == 0 ? Height : heights[col - 1];
            int right = col == Width - 1 ? Height : heights[col + 1];
            int current = heights[col];
            int minSide = Math.Min(left, right);
            if (minSide > current)
            {
                int depth = minSide - current;
                sum += depth * (depth + 1) / 2;
            }
        }

        return sum;
    }

    public int RowTransitions()
    {
        int transitions = 0;
        for (int row = 0; row < Height; row++)
        {
            bool prev = true;
            for (int col = 0; col < Width; col++)
            {
                bool current = cells[row, col];
                if (current != prev)
                {
                    transitions++;
                }
                prev = current;
            }
            if (!prev)
            {
                transitions++;
            }
        }

        return transitions;
    }

    public int ColumnTransitions(int[] heights)
    {
        int transitions = 0;
        for (int col = 0; col < Width; col++)
        {
            bool prev = true;
            for (int row = 0; row < Height; row++)
            {
                bool current = cells[row, col];
                if (current != prev)
                {
                    transitions++;
                }
                prev = current;
            }
            if (!prev)
            {
                transitions++;
            }
        }

        return transitions;
    }

    public int TopHalfCells()
    {
        int count = 0;
        int limit = Height / 2;
        for (int row = 0; row < limit; row++)
        {
            for (int col = 0; col < Width; col++)
            {
                if (cells[row, col])
                {
                    count++;
                }
            }
        }

        return count;
    }

    public int TetrisReadyScore(int[] heights)
    {
        if (Width < 4)
        {
            return 0;
        }

        int best = 0;
        for (int gapCol = 0; gapCol < Width; gapCol++)
        {
            int supportMin = int.MaxValue;
            int supportMax = 0;
            for (int col = 0; col < Width; col++)
            {
                if (col == gapCol)
                {
                    continue;
                }

                int h = heights[col];
                if (h < supportMin)
                {
                    supportMin = h;
                }
                if (h > supportMax)
                {
                    supportMax = h;
                }
            }

            int gap = heights[gapCol];
            int supportLevel = supportMin == int.MaxValue ? 0 : supportMin;
            int score = (supportLevel - gap) * 4 - (supportMax - supportLevel);
            if (score > best)
            {
                best = score;
            }
        }

        return best;
    }
}

public readonly struct Placement
{
    public readonly int Rotation;
    public readonly int Column;
    public readonly int Width;

    public Placement(int rotation, int column, int width)
    {
        Rotation = rotation;
        Column = column;
        Width = width;
    }
}

public static class PieceSet
{
    private static readonly Dictionary<char, bool[][,]> Shapes = new Dictionary<char, bool[][,]>
    {
        ['I'] = UniqueRotations(Parse(
            "####")),
        ['O'] = UniqueRotations(Parse(
            "##",
            "##")),
        ['T'] = UniqueRotations(Parse(
            "###",
            ".#.")),
        ['S'] = UniqueRotations(Parse(
            ".##",
            "##.")),
        ['Z'] = UniqueRotations(Parse(
            "##.",
            ".##")),
        ['J'] = UniqueRotations(Parse(
            "#..",
            "###")),
        ['L'] = UniqueRotations(Parse(
            "..#",
            "###"))
    };

    public static IEnumerable<Placement> GetPlacements(char piece, int boardWidth)
    {
        var rotations = Shapes[piece];
        for (int rotation = 0; rotation < rotations.Length; rotation++)
        {
            int width = rotations[rotation].GetLength(1);
            for (int col = 0; col <= boardWidth - width; col++)
            {
                yield return new Placement(rotation, col, width);
            }
        }
    }

    public static bool[,] GetShape(char piece, int rotation)
    {
        var rotations = Shapes[piece];
        return rotations[((rotation % rotations.Length) + rotations.Length) % rotations.Length];
    }

    private static bool[][,] UniqueRotations(bool[,] shape)
    {
        var list = new List<bool[,]>();
        var seen = new HashSet<string>(StringComparer.Ordinal);
        bool[,] current = Trim(shape);
        for (int i = 0; i < 4; i++)
        {
            string key = Serialize(current);
            if (seen.Add(key))
            {
                list.Add(current);
            }
            current = Trim(Rotate(current));
        }
        return list.ToArray();
    }

    private static bool[,] Parse(params string[] rows)
    {
        int h = rows.Length;
        int w = rows[0].Length;
        var shape = new bool[h, w];
        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                shape[y, x] = rows[y][x] == '#';
            }
        }
        return shape;
    }

    private static bool[,] Rotate(bool[,] shape)
    {
        int h = shape.GetLength(0);
        int w = shape.GetLength(1);
        var rotated = new bool[w, h];
        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                rotated[x, h - 1 - y] = shape[y, x];
            }
        }
        return rotated;
    }

    private static bool[,] Trim(bool[,] shape)
    {
        int h = shape.GetLength(0);
        int w = shape.GetLength(1);
        int minRow = h, maxRow = -1, minCol = w, maxCol = -1;
        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                if (!shape[y, x])
                {
                    continue;
                }
                if (y < minRow) minRow = y;
                if (y > maxRow) maxRow = y;
                if (x < minCol) minCol = x;
                if (x > maxCol) maxCol = x;
            }
        }

        int newH = maxRow - minRow + 1;
        int newW = maxCol - minCol + 1;
        var trimmed = new bool[newH, newW];
        for (int y = 0; y < newH; y++)
        {
            for (int x = 0; x < newW; x++)
            {
                trimmed[y, x] = shape[minRow + y, minCol + x];
            }
        }
        return trimmed;
    }

    private static string Serialize(bool[,] shape)
    {
        int h = shape.GetLength(0);
        int w = shape.GetLength(1);
        char[] data = new char[h * (w + 1)];
        int idx = 0;
        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                data[idx++] = shape[y, x] ? '#' : '.';
            }
            data[idx++] = '/';
        }
        return new string(data);
    }
}

public sealed class FastReader
{
    private readonly TextReader reader;

    public FastReader(TextReader reader)
    {
        this.reader = reader;
    }

    public int NextInt()
    {
        return int.Parse(NextToken(), CultureInfo.InvariantCulture);
    }

    public string NextToken()
    {
        int ch;
        do
        {
            ch = reader.Read();
            if (ch < 0)
            {
                return string.Empty;
            }
        } while (char.IsWhiteSpace((char)ch));

        var chars = new List<char>(8);
        while (ch >= 0 && !char.IsWhiteSpace((char)ch))
        {
            chars.Add((char)ch);
            ch = reader.Read();
        }

        return new string(chars.ToArray());
    }
}
