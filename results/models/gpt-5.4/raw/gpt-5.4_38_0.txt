import sys


def parse_input():
    lines = [line.strip() for line in sys.stdin.read().splitlines()]
    if not lines:
        return 0, 0, []

    first = lines[0].split()
    if len(first) != 2:
        raise ValueError("Invalid header")
    u, s = map(int, first)

    sets = []
    for i in range(s):
        line = lines[i + 2] if i + 2 < len(lines) else ""
        if line:
            elems = list(map(int, line.split()))
        else:
            elems = []
        sets.append(set(elems))
    return u, s, sets


def solve_exact_cover(u, sets):
    universe = set(range(u))

    elem_to_sets = {e: [] for e in range(u)}
    for idx, subset in enumerate(sets):
        for e in subset:
            if e in elem_to_sets:
                elem_to_sets[e].append(idx)

    used = [False] * len(sets)
    covered = set()
    solution = []

    def backtrack():
        if covered == universe:
            return True

        uncovered = [e for e in universe if e not in covered]
        if not uncovered:
            return True

        chosen_elem = min(uncovered, key=lambda e: len([
            idx for idx in elem_to_sets[e]
            if not used[idx] and sets[idx].isdisjoint(covered)
        ]))

        candidates = [
            idx for idx in elem_to_sets[chosen_elem]
            if not used[idx] and sets[idx].isdisjoint(covered)
        ]

        if not candidates:
            return False

        for idx in candidates:
            used[idx] = True
            solution.append(idx)
            prev_covered = covered.copy()
            covered.update(sets[idx])

            if backtrack():
                return True

            covered.clear()
            covered.update(prev_covered)
            solution.pop()
            used[idx] = False

        return False

    if backtrack():
        return solution
    return None


def main():
    u, s, sets = parse_input()
    solution = solve_exact_cover(u, sets)

    if solution is None:
        sys.stdout.write("NO SOLUTION\n")
    else:
        sys.stdout.write("SOLUTION\n")
        sys.stdout.write(" ".join(map(str, solution)) + "\n")


if __name__ == "__main__":
    main()
