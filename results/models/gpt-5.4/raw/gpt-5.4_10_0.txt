#include <algorithm>
#include <array>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct RectInput {
    int w;
    int h;
    int id;
    long long area;
    int max_side;
    int min_side;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct FreeRect {
    int x;
    int y;
    int w;
    int h;
};

struct Candidate {
    int board = -1;
    int slot = -1;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int rotated = 0;
    long long score = numeric_limits<long long>::max();
};

struct Board {
    vector<FreeRect> free_rects;
    long long free_area = 0;
    int max_free_w = 0;
    int max_free_h = 0;
};

static inline bool fits_in(const FreeRect& fr, int w, int h) {
    return w <= fr.w && h <= fr.h;
}

static inline long long score_candidate(const FreeRect& fr, int w, int h) {
    const int dw = fr.w - w;
    const int dh = fr.h - h;
    const int short_left = min(dw, dh);
    const int long_left = max(dw, dh);
    return (static_cast<long long>(short_left) << 44) |
           (static_cast<long long>(long_left) << 24) |
           static_cast<long long>(fr.y + fr.x);
}

static void recompute_board_stats(Board& b) {
    long long area = 0;
    int mw = 0;
    int mh = 0;
    for (const auto& fr : b.free_rects) {
        area += 1LL * fr.w * fr.h;
        mw = max(mw, fr.w);
        mh = max(mh, fr.h);
    }
    b.free_area = area;
    b.max_free_w = mw;
    b.max_free_h = mh;
}

static void prune_free_rects(vector<FreeRect>& rects) {
    vector<char> removed(rects.size(), 0);
    for (size_t i = 0; i < rects.size(); ++i) {
        if (removed[i]) continue;
        for (size_t j = i + 1; j < rects.size(); ++j) {
            if (removed[j]) continue;
            const auto& a = rects[i];
            const auto& b = rects[j];
            const bool a_in_b = a.x >= b.x && a.y >= b.y &&
                                a.x + a.w <= b.x + b.w &&
                                a.y + a.h <= b.y + b.h;
            const bool b_in_a = b.x >= a.x && b.y >= a.y &&
                                b.x + b.w <= a.x + a.w &&
                                b.y + b.h <= a.y + a.h;
            if (a_in_b) {
                removed[i] = 1;
                break;
            }
            if (b_in_a) {
                removed[j] = 1;
            }
        }
    }
    vector<FreeRect> filtered;
    filtered.reserve(rects.size());
    for (size_t i = 0; i < rects.size(); ++i) {
        if (!removed[i] && rects[i].w > 0 && rects[i].h > 0) filtered.push_back(rects[i]);
    }
    rects.swap(filtered);
}

static Candidate find_best_candidate(const vector<Board>& boards, int rw, int rh) {
    Candidate best;
    for (int bi = 0; bi < static_cast<int>(boards.size()); ++bi) {
        const Board& board = boards[bi];
        if (board.free_area < 1LL * rw * rh) continue;
        if ((board.max_free_w < rw || board.max_free_h < rh) &&
            (board.max_free_w < rh || board.max_free_h < rw)) {
            continue;
        }
        for (int si = 0; si < static_cast<int>(board.free_rects.size()); ++si) {
            const FreeRect& fr = board.free_rects[si];
            if (fits_in(fr, rw, rh)) {
                const long long sc = score_candidate(fr, rw, rh);
                if (sc < best.score) {
                    best.board = bi;
                    best.slot = si;
                    best.x = fr.x;
                    best.y = fr.y;
                    best.w = rw;
                    best.h = rh;
                    best.rotated = 0;
                    best.score = sc;
                }
            }
            if (rw != rh && fits_in(fr, rh, rw)) {
                const long long sc = score_candidate(fr, rh, rw);
                if (sc < best.score) {
                    best.board = bi;
                    best.slot = si;
                    best.x = fr.x;
                    best.y = fr.y;
                    best.w = rh;
                    best.h = rw;
                    best.rotated = 1;
                    best.score = sc;
                }
            }
        }
    }
    return best;
}

static Candidate open_new_board_and_place(vector<Board>& boards, int board_w, int board_h, int rw, int rh) {
    Board b;
    b.free_rects.push_back({0, 0, board_w, board_h});
    b.free_area = 1LL * board_w * board_h;
    b.max_free_w = board_w;
    b.max_free_h = board_h;
    boards.push_back(std::move(b));
    return find_best_candidate(boards, rw, rh);
}

static void apply_candidate(Board& board, const Candidate& c) {
    const FreeRect used = board.free_rects[c.slot];
    vector<FreeRect> updated;
    updated.reserve(board.free_rects.size() + 1);
    for (int i = 0; i < static_cast<int>(board.free_rects.size()); ++i) {
        if (i != c.slot) updated.push_back(board.free_rects[i]);
    }

    const int right_w = used.w - c.w;
    const int bottom_h = used.h - c.h;

    if (right_w > 0) {
        updated.push_back({used.x + c.w, used.y, right_w, used.h});
    }
    if (bottom_h > 0) {
        updated.push_back({used.x, used.y + c.h, c.w, bottom_h});
    }

    prune_free_rects(updated);
    board.free_rects.swap(updated);
    recompute_board_stats(board);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, board_w, board_h;
    if (!(cin >> n >> board_w >> board_h)) {
        return 0;
    }

    vector<RectInput> rects;
    rects.reserve(n);
    long long total_area = 0;

    for (int i = 0; i < n; ++i) {
        int w, h;
        cin >> w >> h;
        rects.push_back({w, h, i, 1LL * w * h, max(w, h), min(w, h)});
        total_area += 1LL * w * h;
    }

    const long long board_area = 1LL * board_w * board_h;
    for (const auto& r : rects) {
        if ((r.w > board_w || r.h > board_h) && (r.h > board_w || r.w > board_h)) {
            return 0;
        }
    }

    sort(rects.begin(), rects.end(), [](const RectInput& a, const RectInput& b) {
        if (a.area != b.area) return a.area > b.area;
        if (a.max_side != b.max_side) return a.max_side > b.max_side;
        if (a.min_side != b.min_side) return a.min_side > b.min_side;
        return a.id < b.id;
    });

    vector<Placement> placements(n);
    vector<Board> boards;
    boards.reserve(max(1LL, (total_area + board_area - 1) / board_area));

    for (const auto& r : rects) {
        Candidate best = find_best_candidate(boards, r.w, r.h);
        if (best.board == -1) {
            best = open_new_board_and_place(boards, board_w, board_h, r.w, r.h);
        }
        apply_candidate(boards[best.board], best);
        placements[r.id] = {best.board, best.x, best.y, best.rotated};
    }

    cout << boards.size() << '\n';
    for (int i = 0; i < n; ++i) {
        const auto& p = placements[i];
        cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }
    return 0;
}
