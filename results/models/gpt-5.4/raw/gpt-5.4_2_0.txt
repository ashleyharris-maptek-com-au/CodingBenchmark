using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    struct EdgeInput
    {
        public int U;
        public int V;
        public long W;
    }

    sealed class AdjEdge
    {
        public int To;
        public long W;
        public int EdgeIndex;
        public AdjEdge Next;
    }

    sealed class MinHeap
    {
        private int[] _nodes;
        private long[] _dist;
        public int Count { get; private set; }

        public MinHeap(int capacity)
        {
            _nodes = new int[Math.Max(4, capacity + 1)];
            _dist = new long[_nodes.Length];
        }

        public void Push(int node, long d)
        {
            if (Count + 1 == _nodes.Length)
            {
                Array.Resize(ref _nodes, _nodes.Length * 2);
                Array.Resize(ref _dist, _dist.Length * 2);
            }
            int i = ++Count;
            while (i > 1)
            {
                int p = i >> 1;
                if (_dist[p] <= d) break;
                _nodes[i] = _nodes[p];
                _dist[i] = _dist[p];
                i = p;
            }
            _nodes[i] = node;
            _dist[i] = d;
        }

        public (int node, long dist) Pop()
        {
            int resultNode = _nodes[1];
            long resultDist = _dist[1];
            int lastNode = _nodes[Count];
            long lastDist = _dist[Count];
            Count--;
            int i = 1;
            while (true)
            {
                int left = i << 1;
                if (left > Count) break;
                int right = left + 1;
                int child = right <= Count && _dist[right] < _dist[left] ? right : left;
                if (_dist[child] >= lastDist) break;
                _nodes[i] = _nodes[child];
                _dist[i] = _dist[child];
                i = child;
            }
            if (Count > 0)
            {
                _nodes[i] = lastNode;
                _dist[i] = lastDist;
            }
            return (resultNode, resultDist);
        }
    }

    sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    static int _n;
    static int _m;
    static EdgeInput[] _edges;
    static AdjEdge[] _head;
    static int[] _degree;
    static bool[] _reachable;
    static int[] _queue;

    static long[] _dist;
    static int[] _parentNode;
    static int[] _parentEdge;
    static int[] _seenStamp;
    static int _stamp;

    static bool[] _baseUsed;
    static int[] _edgeU;
    static int[] _edgeV;
    static int[] _edgeNextU;
    static int[] _edgeNextV;

    public static void Main()
    {
        var fs = new FastScanner();
        _n = fs.NextInt();
        _m = fs.NextInt();

        _edges = new EdgeInput[_m];
        _head = new AdjEdge[_n];
        _degree = new int[_n];

        for (int i = 0; i < _m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();
            _edges[i] = new EdgeInput { U = u, V = v, W = w };
            AddAdj(u, v, w, i);
            AddAdj(v, u, w, i);
            _degree[u]++;
            _degree[v]++;
        }

        _reachable = new bool[_n];
        _queue = new int[_n];
        MarkReachableFromZero();

        for (int i = 0; i < _m; i++)
        {
            if (!(_reachable[_edges[i].U] && _reachable[_edges[i].V]))
            {
                // Unreachable edges make the required tour impossible. Fall back to a trivial cycle.
                Console.WriteLine("1 0");
                return;
            }
        }

        var odd = new List<int>();
        for (int i = 0; i < _n; i++)
        {
            if (_reachable[i] && ((_degree[i] & 1) != 0))
                odd.Add(i);
        }

        var duplicates = BuildAugmentation(odd);
        BuildAugmentedGraph(duplicates);
        var route = BuildEulerTour();

        var sb = new StringBuilder(route.Count * 3 + 32);
        sb.Append(route.Count);
        for (int i = 0; i < route.Count; i++)
        {
            sb.Append(' ');
            sb.Append(route[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    static void AddAdj(int u, int v, long w, int edgeIndex)
    {
        _head[u] = new AdjEdge { To = v, W = w, EdgeIndex = edgeIndex, Next = _head[u] };
    }

    static void MarkReachableFromZero()
    {
        int qh = 0, qt = 0;
        _queue[qt++] = 0;
        _reachable[0] = true;
        while (qh < qt)
        {
            int u = _queue[qh++];
            for (AdjEdge e = _head[u]; e != null; e = e.Next)
            {
                int v = e.To;
                if (_reachable[v]) continue;
                _reachable[v] = true;
                _queue[qt++] = v;
            }
        }
    }

    static List<int> BuildAugmentation(List<int> odd)
    {
        int k = odd.Count;
        var duplicates = new List<int>();
        if (k == 0) return duplicates;

        _dist = new long[_n];
        _parentNode = new int[_n];
        _parentEdge = new int[_n];
        _seenStamp = new int[_n];
        _stamp = 0;

        if (k <= 22)
        {
            var pairing = ExactBitmaskMatching(odd);
            for (int i = 0; i < pairing.Count; i += 2)
                AppendShortestPathEdges(pairing[i], pairing[i + 1], duplicates);
            return duplicates;
        }

        if (k <= 4000)
        {
            var remaining = new HashSet<int>(odd);
            while (remaining.Count > 0)
            {
                int a = -1;
                foreach (var x in remaining) { a = x; break; }
                remaining.Remove(a);
                int b = FindNearestOdd(a, remaining);
                remaining.Remove(b);
                AppendShortestPathEdges(a, b, duplicates);
            }
            return duplicates;
        }

        // For very many odd vertices, pair via a DFS-tree style walk over shortest-path tree roots from 0.
        // This avoids quadratic work and still guarantees parity correction.
        var parent = BuildShortestPathTreeFromZero();
        var bucket = new List<int>[_n];
        foreach (int v in odd)
        {
            bucket[v] ??= new List<int>();
            bucket[v].Add(v);
        }

        var order = new int[_n];
        int ordCount = BuildReachableOrder(order);
        for (int idx = ordCount - 1; idx >= 0; idx--)
        {
            int v = order[idx];
            var list = bucket[v];
            if (list != null)
            {
                while (list.Count >= 2)
                {
                    int a = list[list.Count - 1];
                    list.RemoveAt(list.Count - 1);
                    int b = list[list.Count - 1];
                    list.RemoveAt(list.Count - 1);
                    AppendShortestPathEdges(a, b, duplicates);
                }
            }

            int p = parent[v];
            if (p >= 0 && list != null && list.Count == 1)
            {
                bucket[p] ??= new List<int>();
                bucket[p].Add(list[0]);
            }
        }

        var rootList = bucket[0];
        if (rootList != null)
        {
            for (int i = 0; i + 1 < rootList.Count; i += 2)
                AppendShortestPathEdges(rootList[i], rootList[i + 1], duplicates);
        }

        return duplicates;
    }

    static List<int> ExactBitmaskMatching(List<int> odd)
    {
        int k = odd.Count;
        var distMat = new long[k, k];
        for (int i = 0; i < k; i++)
        {
            Dijkstra(odd[i], -1);
            for (int j = i + 1; j < k; j++)
                distMat[i, j] = distMat[j, i] = _dist[odd[j]];
        }

        int size = 1 << k;
        var dp = new long[size];
        var choiceA = new int[size];
        var choiceB = new int[size];
        const long INF = long.MaxValue / 4;
        for (int i = 1; i < size; i++) dp[i] = INF;

        for (int mask = 1; mask < size; mask++)
        {
            int a = 0;
            while (((mask >> a) & 1) == 0) a++;
            int maskWithoutA = mask ^ (1 << a);
            for (int b = a + 1; b < k; b++)
            {
                if (((maskWithoutA >> b) & 1) == 0) continue;
                int nextMask = maskWithoutA ^ (1 << b);
                long cand = dp[nextMask] + distMat[a, b];
                if (cand < dp[mask])
                {
                    dp[mask] = cand;
                    choiceA[mask] = a;
                    choiceB[mask] = b;
                }
            }
        }

        var result = new List<int>(k);
        int cur = size - 1;
        while (cur != 0)
        {
            int a = choiceA[cur];
            int b = choiceB[cur];
            result.Add(odd[a]);
            result.Add(odd[b]);
            cur ^= (1 << a);
            cur ^= (1 << b);
        }
        return result;
    }

    static int FindNearestOdd(int source, HashSet<int> targets)
    {
        Dijkstra(source, -1);
        int bestNode = -1;
        long best = long.MaxValue;
        foreach (int t in targets)
        {
            long d = _dist[t];
            if (d < best)
            {
                best = d;
                bestNode = t;
            }
        }
        return bestNode;
    }

    static void Dijkstra(int source, int stopTarget)
    {
        _stamp++;
        var heap = new MinHeap(64);
        _dist[source] = 0;
        _parentNode[source] = -1;
        _parentEdge[source] = -1;
        _seenStamp[source] = _stamp;
        heap.Push(source, 0);

        while (heap.Count > 0)
        {
            var cur = heap.Pop();
            int u = cur.node;
            long du = cur.dist;
            if (_seenStamp[u] != _stamp || _dist[u] != du) continue;
            if (u == stopTarget) return;

            for (AdjEdge e = _head[u]; e != null; e = e.Next)
            {
                int v = e.To;
                long nd = du + e.W;
                if (_seenStamp[v] != _stamp || nd < _dist[v])
                {
                    _seenStamp[v] = _stamp;
                    _dist[v] = nd;
                    _parentNode[v] = u;
                    _parentEdge[v] = e.EdgeIndex;
                    heap.Push(v, nd);
                }
            }
        }
    }

    static void AppendShortestPathEdges(int a, int b, List<int> duplicates)
    {
        Dijkstra(a, b);
        int cur = b;
        while (cur != a)
        {
            int edge = _parentEdge[cur];
            if (edge < 0) break;
            duplicates.Add(edge);
            cur = _parentNode[cur];
        }
    }

    static int[] BuildShortestPathTreeFromZero()
    {
        Dijkstra(0, -1);
        var parent = new int[_n];
        for (int i = 0; i < _n; i++)
            parent[i] = _seenStamp[i] == _stamp ? _parentNode[i] : -1;
        return parent;
    }

    static int BuildReachableOrder(int[] order)
    {
        int count = 0;
        for (int i = 0; i < _n; i++)
            if (_reachable[i]) order[count++] = i;
        Array.Sort(order, 0, count, Comparer<int>.Create((a, b) => _dist[b].CompareTo(_dist[a])));
        return count;
    }

    static void BuildAugmentedGraph(List<int> duplicates)
    {
        int totalEdges = _m + duplicates.Count;
        _baseUsed = new bool[totalEdges];
        _edgeU = new int[totalEdges];
        _edgeV = new int[totalEdges];
        _edgeNextU = new int[totalEdges];
        _edgeNextV = new int[totalEdges];

        var firstAtNode = new int[_n];
        Array.Fill(firstAtNode, -1);

        for (int i = 0; i < totalEdges; i++)
        {
            int u, v;
            if (i < _m)
            {
                u = _edges[i].U;
                v = _edges[i].V;
            }
            else
            {
                var e = _edges[duplicates[i - _m]];
                u = e.U;
                v = e.V;
            }

            _edgeU[i] = u;
            _edgeV[i] = v;
            _edgeNextU[i] = firstAtNode[u];
            firstAtNode[u] = i;
            _edgeNextV[i] = firstAtNode[v];
            firstAtNode[v] = i;
        }

        _iterHead = firstAtNode;
    }

    static int[] _iterHead;

    static List<int> BuildEulerTour()
    {
        int totalEdges = _edgeU.Length;
        var stackNodes = new int[totalEdges + 2];
        var stackEdges = new int[totalEdges + 2];
        int sp = 0;
        stackNodes[0] = 0;
        stackEdges[0] = -1;
        var route = new List<int>(totalEdges + 1);

        while (sp >= 0)
        {
            int u = stackNodes[sp];
            int ei = NextUnusedEdge(u);
            if (ei == -1)
            {
                route.Add(u);
                sp--;
                continue;
            }

            _baseUsed[ei] = true;
            int v = _edgeU[ei] ^ _edgeV[ei] ^ u;
            sp++;
            stackNodes[sp] = v;
            stackEdges[sp] = ei;
        }

        route.Reverse();
        if (route.Count == 0 || route[0] != 0)
            return new List<int> { 0 };
        return route;
    }

    static int NextUnusedEdge(int u)
    {
        int ei = _iterHead[u];
        while (ei != -1)
        {
            int next = _edgeU[ei] == u ? _edgeNextU[ei] : _edgeNextV[ei];
            _iterHead[u] = next;
            if (!_baseUsed[ei]) return ei;
            ei = next;
        }
        return -1;
    }
}
