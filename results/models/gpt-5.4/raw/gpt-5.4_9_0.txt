using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private readonly struct Cut
    {
        public readonly int Length;
        public readonly int Index;

        public Cut(int length, int index)
        {
            Length = length;
            Index = index;
        }
    }

    private sealed class Stock
    {
        public int Id;
        public int Remaining;
        public readonly List<int> Cuts = new List<int>();
    }

    private sealed class FastReader
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _position;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1)
                {
                    return int.MinValue;
                }
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > ' ')
            {
                value = checked(value * 10 + (c - '0'));
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }
    }

    public static void Main()
    {
        var reader = new FastReader();
        int n = reader.NextInt();
        if (n == int.MinValue)
        {
            return;
        }

        int stockLength = reader.NextInt();
        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int length = reader.NextInt();
            if (length > stockLength)
            {
                throw new InvalidOperationException("Cut length exceeds stock length.");
            }

            cuts[i] = new Cut(length, i);
        }

        Array.Sort(cuts, static (a, b) =>
        {
            int byLength = b.Length.CompareTo(a.Length);
            return byLength != 0 ? byLength : a.Index.CompareTo(b.Index);
        });

        var openStocks = new SortedDictionary<int, Queue<Stock>>();
        var allStocks = new List<Stock>();

        foreach (var cut in cuts)
        {
            Stock stock = TakeBestFit(openStocks, cut.Length);
            if (stock == null)
            {
                stock = new Stock
                {
                    Id = allStocks.Count,
                    Remaining = stockLength
                };
                allStocks.Add(stock);
            }

            stock.Cuts.Add(cut.Index);
            stock.Remaining -= cut.Length;

            if (stock.Remaining > 0)
            {
                if (!openStocks.TryGetValue(stock.Remaining, out Queue<Stock> queue))
                {
                    queue = new Queue<Stock>();
                    openStocks.Add(stock.Remaining, queue);
                }

                queue.Enqueue(stock);
            }
        }

        var output = new StringBuilder(allStocks.Count * 16 + n * 6);
        output.Append(allStocks.Count).Append('\n');

        foreach (var stock in allStocks)
        {
            for (int i = 0; i < stock.Cuts.Count; i++)
            {
                if (i > 0)
                {
                    output.Append(' ');
                }

                output.Append(stock.Cuts[i]);
            }

            output.Append('\n');
        }

        Console.Write(output.ToString());
    }

    private static Stock TakeBestFit(SortedDictionary<int, Queue<Stock>> openStocks, int cutLength)
    {
        foreach (var pair in openStocks)
        {
            if (pair.Key < cutLength)
            {
                continue;
            }

            Queue<Stock> queue = pair.Value;
            Stock stock = queue.Dequeue();
            if (queue.Count == 0)
            {
                openStocks.Remove(pair.Key);
            }

            return stock;
        }

        return null;
    }
}
