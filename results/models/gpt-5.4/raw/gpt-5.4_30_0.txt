import sys
from array import array
from random import Random


def main():
    data = sys.stdin.buffer
    first = data.readline().split()
    if not first:
        return
    n = int(first[0])
    m = int(first[1])

    # Compact storage:
    # - lits keeps all clause literals flattened: 3 integers per clause.
    # - occ_pos / occ_neg store linked lists of clause indices per variable.
    # - next_pos / next_neg provide the next link for each (variable, slot) occurrence.
    lits = array("i", [0]) * (3 * m)
    occ_pos = array("i", [-1]) * (n + 1)
    occ_neg = array("i", [-1]) * (n + 1)
    next_pos = array("i", [-1]) * (3 * m)
    next_neg = array("i", [-1]) * (3 * m)

    idx = 0
    clause_id = 0
    for line in data:
        parts = line.split()
        if not parts:
            continue
        a = int(parts[0])
        b = int(parts[1])
        c = int(parts[2])
        for lit in (a, b, c):
            lits[idx] = lit
            v = lit if lit > 0 else -lit
            if lit > 0:
                next_pos[idx] = occ_pos[v]
                occ_pos[v] = clause_id
            else:
                next_neg[idx] = occ_neg[v]
                occ_neg[v] = clause_id
            idx += 1
        clause_id += 1

    # Guaranteed satisfiable, but m may be smaller than the number of parsed lines
    # if there were blank lines in the input. Use the actual parsed clause count.
    m = clause_id
    if m == 0:
        out = sys.stdout.write
        out("SAT\n")
        out(" ".join("0" for _ in range(n)))
        out("\n")
        return

    rng = Random(123456789)

    # Greedy initial assignment based on literal imbalance.
    assign = bytearray(n + 1)
    for v in range(1, n + 1):
        pos_count = 0
        neg_count = 0
        cidx = occ_pos[v]
        while cidx != -1:
            pos_count += 1
            base = cidx * 3
            if lits[base] == v:
                cidx = next_pos[base]
            elif lits[base + 1] == v:
                cidx = next_pos[base + 1]
            else:
                cidx = next_pos[base + 2]
        cidx = occ_neg[v]
        while cidx != -1:
            neg_count += 1
            base = cidx * 3
            if lits[base] == -v:
                cidx = next_neg[base]
            elif lits[base + 1] == -v:
                cidx = next_neg[base + 1]
            else:
                cidx = next_neg[base + 2]
        assign[v] = 1 if pos_count >= neg_count else 0

    sat_count = array("b", [0]) * m
    unsat = array("i", [0]) * m
    where_unsat = array("i", [-1]) * m
    unsat_size = 0

    def lit_true(lit):
        v = lit if lit > 0 else -lit
        val = assign[v]
        return val if lit > 0 else (1 - val)

    for cid in range(m):
        base = cid * 3
        cnt = lit_true(lits[base]) + lit_true(lits[base + 1]) + lit_true(lits[base + 2])
        sat_count[cid] = cnt
        if cnt == 0:
            where_unsat[cid] = unsat_size
            unsat[unsat_size] = cid
            unsat_size += 1

    def remove_unsat(cid):
        nonlocal unsat_size
        pos = where_unsat[cid]
        if pos == -1:
            return
        unsat_size -= 1
        last = unsat[unsat_size]
        unsat[pos] = last
        where_unsat[last] = pos
        where_unsat[cid] = -1

    def add_unsat(cid):
        nonlocal unsat_size
        if where_unsat[cid] != -1:
            return
        where_unsat[cid] = unsat_size
        unsat[unsat_size] = cid
        unsat_size += 1

    def break_score(var):
        # Number of clauses that would become unsatisfied if var flipped.
        score = 0
        desired_now = assign[var]

        cidx = occ_pos[var] if desired_now else occ_neg[var]
        while cidx != -1:
            base = cidx * 3
            if sat_count[cidx] == 1:
                score += 1
            if desired_now:
                if lits[base] == var:
                    cidx = next_pos[base]
                elif lits[base + 1] == var:
                    cidx = next_pos[base + 1]
                else:
                    cidx = next_pos[base + 2]
            else:
                if lits[base] == -var:
                    cidx = next_neg[base]
                elif lits[base + 1] == -var:
                    cidx = next_neg[base + 1]
                else:
                    cidx = next_neg[base + 2]
        return score

    def flip(var):
        old = assign[var]
        assign[var] = 1 - old

        # Clauses where the literal becomes false.
        cidx = occ_pos[var] if old else occ_neg[var]
        while cidx != -1:
            sat_count[cidx] -= 1
            if sat_count[cidx] == 0:
                add_unsat(cidx)
            base = cidx * 3
            if old:
                if lits[base] == var:
                    cidx = next_pos[base]
                elif lits[base + 1] == var:
                    cidx = next_pos[base + 1]
                else:
                    cidx = next_pos[base + 2]
            else:
                if lits[base] == -var:
                    cidx = next_neg[base]
                elif lits[base + 1] == -var:
                    cidx = next_neg[base + 1]
                else:
                    cidx = next_neg[base + 2]

        # Clauses where the literal becomes true.
        cidx = occ_neg[var] if old else occ_pos[var]
        while cidx != -1:
            if sat_count[cidx] == 0:
                remove_unsat(cidx)
            sat_count[cidx] += 1
            base = cidx * 3
            if old:
                if lits[base] == -var:
                    cidx = next_neg[base]
                elif lits[base + 1] == -var:
                    cidx = next_neg[base + 1]
                else:
                    cidx = next_neg[base + 2]
            else:
                if lits[base] == var:
                    cidx = next_pos[base]
                elif lits[base + 1] == var:
                    cidx = next_pos[base + 1]
                else:
                    cidx = next_pos[base + 2]

    max_steps = max(10_000_000, min(200_000_000, m * 20))
    noise_num = 3
    noise_den = 10

    step = 0
    while unsat_size and step < max_steps:
        step += 1
        cid = unsat[rng.randrange(unsat_size)]
        base = cid * 3
        v1 = lits[base]
        v2 = lits[base + 1]
        v3 = lits[base + 2]
        a = v1 if v1 > 0 else -v1
        b = v2 if v2 > 0 else -v2
        c = v3 if v3 > 0 else -v3

        if rng.randrange(noise_den) < noise_num:
            choice = rng.randrange(3)
            flip(a if choice == 0 else b if choice == 1 else c)
            continue

        sa = break_score(a)
        sb = break_score(b)
        sc = break_score(c)

        if sa <= sb and sa <= sc:
            flip(a)
        elif sb <= sc:
            flip(b)
        else:
            flip(c)

    # Guaranteed satisfiable. If local search failed to converge within the step budget,
    # continue with random diversification rounds instead of giving up.
    while unsat_size:
        for v in range(1, n + 1):
            assign[v] = rng.randrange(2)

        unsat_size = 0
        for cid in range(m):
            base = cid * 3
            cnt = lit_true(lits[base]) + lit_true(lits[base + 1]) + lit_true(lits[base + 2])
            sat_count[cid] = cnt
            if cnt == 0:
                where_unsat[cid] = unsat_size
                unsat[unsat_size] = cid
                unsat_size += 1
            else:
                where_unsat[cid] = -1

        step = 0
        while unsat_size and step < max_steps:
            step += 1
            cid = unsat[rng.randrange(unsat_size)]
            base = cid * 3
            v1 = lits[base]
            v2 = lits[base + 1]
            v3 = lits[base + 2]
            a = v1 if v1 > 0 else -v1
            b = v2 if v2 > 0 else -v2
            c = v3 if v3 > 0 else -v3

            if rng.randrange(noise_den) < noise_num:
                choice = rng.randrange(3)
                flip(a if choice == 0 else b if choice == 1 else c)
                continue

            sa = break_score(a)
            sb = break_score(b)
            sc = break_score(c)
            if sa <= sb and sa <= sc:
                flip(a)
            elif sb <= sc:
                flip(b)
            else:
                flip(c)

    out = sys.stdout.write
    out("SAT\n")
    chunk = []
    limit = 100000
    for v in range(1, n + 1):
        chunk.append("1" if assign[v] else "0")
        if len(chunk) >= limit:
            out(" ".join(chunk))
            out(" ")
            chunk.clear()
    if chunk:
        out(" ".join(chunk))
    out("\n")


if __name__ == "__main__":
    main()
