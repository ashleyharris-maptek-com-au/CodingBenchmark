```python
import math


_AP_STATE = {
    "init": False,
    "alt_int": 0.0,
    "vs_int": 0.0,
    "hdg_int": 0.0,
    "ias_int": 0.0,
    "trim_int": 0.0,
    "last_hdg_err": 0.0,
    "last_ias_err": 0.0,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _wrap180(deg):
    x = (deg + 180.0) % 360.0 - 180.0
    return x


def _median(values):
    vals = sorted(v for v in values if v is not None and math.isfinite(v))
    if not vals:
        return None
    n = len(vals)
    m = n // 2
    return vals[m] if n % 2 else 0.5 * (vals[m - 1] + vals[m])


def _robust_triplet(state, prefixes, suffix, valid_suffix="_valid", spread=None):
    vals = []
    for p in prefixes:
        v = state.get(f"{p}{suffix}")
        valid = state.get(f"{p}{valid_suffix}", True)
        if isinstance(v, (int, float)) and math.isfinite(v) and valid:
            vals.append(float(v))
    if not vals:
        for p in prefixes:
            v = state.get(f"{p}{suffix}")
            if isinstance(v, (int, float)) and math.isfinite(v):
                vals.append(float(v))
    if not vals:
        return None
    if len(vals) == 1:
        return vals[0]
    med = _median(vals)
    if spread is None:
        return med
    kept = [v for v in vals if abs(v - med) <= spread]
    return _median(kept) if kept else med


def _best_altitude(state):
    baro = _robust_triplet(state, ["baro_alt_1", "baro_alt_2", "baro_alt_3"], "_ft", spread=300.0)
    gps = _robust_triplet(state, ["gps_1", "gps_2", "gps_3"], "_alt_ft", spread=400.0)
    radar = _robust_triplet(state, ["radar_alt_1", "radar_alt_2", "radar_alt_3"], "_ft", spread=100.0)
    pressure = _robust_triplet(state, ["adc_1", "adc_2", "adc_3"], "_pressure_alt_ft", spread=300.0)

    wow = any(bool(state.get(k, False)) for k in [
        "weight_on_wheels_left",
        "weight_on_wheels_right",
        "weight_on_wheels_nose",
    ])
    if radar is not None and radar < 2500.0 and not wow:
        if baro is not None and abs(baro - radar) < 800.0:
            return 0.7 * radar + 0.3 * baro
        return radar
    if baro is not None:
        return baro
    if pressure is not None:
        return pressure
    return gps if gps is not None else 0.0


def _best_vs(state):
    irs_vs = _robust_triplet(state, ["irs_1", "irs_2", "irs_3"], "_vs_fpm", spread=1200.0)
    nav_vs = state.get("vertical_speed_fpm")
    if isinstance(nav_vs, (int, float)) and math.isfinite(nav_vs):
        if irs_vs is not None:
            return 0.7 * irs_vs + 0.3 * float(nav_vs)
        return float(nav_vs)
    return irs_vs if irs_vs is not None else 0.0


def _best_ias(state):
    ias = _robust_triplet(state, ["ias_1", "ias_2", "ias_3"], "_kt", spread=20.0)
    if ias is not None:
        return ias
    gs = _robust_triplet(state, ["gps_1", "gps_2", "gps_3"], "_gs_kt", spread=25.0)
    if gs is None:
        gs = _robust_triplet(state, ["irs_1", "irs_2", "irs_3"], "_gs_kt", spread=25.0)
    return gs if gs is not None else 150.0


def _best_mach(state, ias):
    vals = []
    for k in ["mach_1", "mach_2", "mach_3"]:
        v = state.get(k)
        if isinstance(v, (int, float)) and math.isfinite(v) and 0.0 < v < 1.5:
            vals.append(float(v))
    mach = _median(vals)
    if mach is not None:
        return mach
    tas = _median([state.get("tas_1_kt"), state.get("tas_2_kt"), state.get("tas_3_kt")])
    if tas and tas > 1.0:
        return ias / max(tas, 1.0)
    return 0.0


def _best_heading(state):
    irs_hdg = _robust_triplet(state, ["irs_1", "irs_2", "irs_3"], "_heading_deg", spread=15.0)
    nav_hdg = state.get("magnetic_heading_deg")
    if isinstance(nav_hdg, (int, float)) and math.isfinite(nav_hdg):
        if irs_hdg is not None:
            e = _wrap180(float(nav_hdg) - irs_hdg)
            if abs(e) < 10.0:
                return (irs_hdg + e * 0.3) % 360.0
        return float(nav_hdg) % 360.0
    return irs_hdg if irs_hdg is not None else float(state.get("target_hdg_deg", 0.0)) % 360.0


def _best_roll(state):
    return _robust_triplet(state, ["irs_1", "irs_2", "irs_3"], "_roll_deg", spread=8.0) or 0.0


def _best_pitch(state):
    return _robust_triplet(state, ["irs_1", "irs_2", "irs_3"], "_pitch_deg", spread=6.0) or 0.0


def _best_yaw_rate(state):
    return _robust_triplet(state, ["gyro_1", "gyro_2", "gyro_3"], "_yaw_rate_dps", spread=5.0) or 0.0


def _best_roll_rate(state):
    return _robust_triplet(state, ["gyro_1", "gyro_2", "gyro_3"], "_roll_rate_dps", spread=8.0) or 0.0


def _best_pitch_rate(state):
    return _robust_triplet(state, ["gyro_1", "gyro_2", "gyro_3"], "_pitch_rate_dps", spread=6.0) or 0.0


def _best_aoa(state):
    return _robust_triplet(state, ["aoa_1", "aoa_2", "aoa_3"], "_deg", spread=4.0)


def autopilot_step(state: dict, dt: float) -> dict:
    dt = float(dt if dt and dt > 0.0 else state.get("dt", 0.05) or 0.05)
    dt = _clamp(dt, 0.01, 0.2)

    alt = _best_altitude(state)
    vs = _best_vs(state)
    ias = _best_ias(state)
    mach = _best_mach(state, ias)
    hdg = _best_heading(state)
    roll = _best_roll(state)
    pitch = _best_pitch(state)
    roll_rate = _best_roll_rate(state)
    pitch_rate = _best_pitch_rate(state)
    yaw_rate = _best_yaw_rate(state)
    aoa = _best_aoa(state)

    target_alt = float(state.get("target_alt_ft", alt))
    target_hdg = float(state.get("target_hdg_deg", hdg)) % 360.0
    target_ias = float(state.get("target_ias_kt", max(ias, 140.0)))
    target_vs = float(state.get("target_vs_fpm", 0.0))

    wow = any(bool(state.get(k, False)) for k in [
        "weight_on_wheels_left",
        "weight_on_wheels_right",
        "weight_on_wheels_nose",
    ])
    gear_handle = bool(state.get("gear_handle_down", False))

    eng1_running = bool(state.get("engine_1_running", False))
    eng2_running = bool(state.get("engine_2_running", False))
    eng1_fire = bool(state.get("engine_1_fire", False))
    eng2_fire = bool(state.get("engine_2_fire", False))
    thrust1 = float(state.get("engine_1_thrust_pct", 0.0))
    thrust2 = float(state.get("engine_2_thrust_pct", 0.0))

    alt_err = target_alt - alt
    vs_cmd = _clamp(alt_err * 1.5 + target_vs, -2500.0, 2500.0)
    if abs(alt_err) < 120.0:
        vs_cmd = _clamp(target_vs, -1200.0, 1200.0)

    low_energy = ias < 135.0 or (aoa is not None and aoa > 10.0)
    overspeed = ias > 340.0 or mach > 0.80
    severe_overspeed = ias > 360.0 or mach > 0.82

    if low_energy:
        vs_cmd = min(vs_cmd, 300.0)
    if severe_overspeed:
        vs_cmd = max(vs_cmd, -1500.0)

    vs_err = vs_cmd - vs
    _AP_STATE["vs_int"] = _clamp(_AP_STATE["vs_int"] + vs_err * dt * 0.0004, -0.3, 0.3)
    pitch_cmd = (
        0.00055 * vs_err
        + _AP_STATE["vs_int"]
        - 0.055 * pitch
        - 0.020 * pitch_rate
    )

    if low_energy:
        pitch_cmd -= 0.20 + 0.015 * max((aoa or 8.0) - 8.0, 0.0)
    if severe_overspeed:
        pitch_cmd -= 0.10

    elevator = _clamp(pitch_cmd, -1.0, 1.0)

    _AP_STATE["trim_int"] = _clamp(_AP_STATE["trim_int"] + elevator * dt * 0.10, -0.8, 0.8)
    elevator_trim = _clamp(_AP_STATE["trim_int"], -1.0, 1.0)

    hdg_err = _wrap180(target_hdg - hdg)
    bank_limit = 25.0
    if low_energy or alt < 1000.0:
        bank_limit = 15.0
    desired_bank = _clamp(0.45 * hdg_err, -bank_limit, bank_limit)
    bank_err = desired_bank - roll

    aileron = _clamp(0.045 * bank_err - 0.020 * roll_rate, -1.0, 1.0)
    rudder = _clamp(-0.12 * yaw_rate - 0.015 * roll + 0.0015 * (thrust2 - thrust1), -1.0, 1.0)

    if eng1_fire or not eng1_running:
        throttle_1 = 0.0
        throttle_2 = 0.80 if eng2_running else 0.0
    elif eng2_fire or not eng2_running:
        throttle_2 = 0.0
        throttle_1 = 0.80 if eng1_running else 0.0
    else:
        ias_err = target_ias - ias
        if overspeed:
            ias_err = min(ias_err, -10.0)
        _AP_STATE["ias_int"] = _clamp(_AP_STATE["ias_int"] + ias_err * dt * 0.01, -0.25, 0.25)
        thrust_base = (
            0.52
            + 0.0065 * ias_err
            + 0.00010 * vs_cmd
            + _AP_STATE["ias_int"]
        )
        if low_energy:
            thrust_base += 0.20
        if severe_overspeed:
            thrust_base -= 0.35
        throttle_1 = _clamp(thrust_base, 0.0, 1.0)
        throttle_2 = _clamp(thrust_base, 0.0, 1.0)

    if wow:
        gear = 1
        flaps = 15 if target_ias < 170.0 else 5
        speedbrake = 0.0
        elevator_trim = _clamp(elevator_trim, -0.2, 0.4)
    else:
        landing_config = target_alt < 3000.0 and alt < 3000.0 and target_ias < 190.0
        if gear_handle or landing_config:
            gear = 1
        else:
            gear = 0

        if gear or ias < 150.0:
            flaps = 15
        elif ias < 180.0:
            flaps = 5
        else:
            flaps = 0

        if severe_overspeed:
            speedbrake = 0.6 if flaps == 0 and gear == 0 else 0.2
        elif overspeed and alt_err < -500.0:
            speedbrake = 0.3 if flaps == 0 and gear == 0 else 0.0
        else:
            speedbrake = 0.0

    if low_energy:
        speedbrake = 0.0
        if gear == 1 and alt > 500.0 and target_ias > 160.0:
            gear = 0
        flaps = max(flaps, 5) if ias < 170.0 else min(flaps, 5)

    return {
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": _clamp(throttle_1, 0.0, 1.0),
        "throttle_2": _clamp(throttle_2, 0.0, 1.0),
        "elevator_trim": _clamp(elevator_trim, -1.0, 1.0),
        "flaps": int(flaps),
        "speedbrake": _clamp(speedbrake, 0.0, 1.0),
        "gear": int(gear),
    }
```
