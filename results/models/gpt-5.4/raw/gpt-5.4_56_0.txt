const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot || [0, 0];
const truckCount = Math.max(1, Math.floor(data.truckCount || 1));
const packages = Array.isArray(data.packages) ? data.packages.slice() : [];

function dist(a, b) {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
}

function twoOpt(route, pts, depotPt) {
  const n = route.length;
  if (n < 4) return route;

  let improved = true;
  let passes = 0;
  const maxPasses = 3;

  while (improved && passes < maxPasses) {
    improved = false;
    passes++;

    for (let i = 0; i < n - 2; i++) {
      const a = i === 0 ? null : route[i - 1];
      const b = route[i];
      const aPt = a === null ? depotPt : pts[a];
      const bPt = pts[b];

      for (let k = i + 1; k < n; k++) {
        const c = route[k];
        const d = k === n - 1 ? null : route[k + 1];
        const cPt = pts[c];
        const dPt = d === null ? depotPt : pts[d];

        const oldCost = dist(aPt, bPt) + dist(cPt, dPt);
        const newCost = dist(aPt, cPt) + dist(bPt, dPt);

        if (newCost + 1e-12 < oldCost) {
          let left = i;
          let right = k;
          while (left < right) {
            const tmp = route[left];
            route[left] = route[right];
            route[right] = tmp;
            left++;
            right--;
          }
          improved = true;
        }
      }
    }
  }

  return route;
}

function buildRoute(indices, pts, depotPt) {
  if (indices.length <= 1) return indices.slice();

  const remaining = indices.slice();
  let startPos = 0;
  let bestStart = Infinity;

  for (let i = 0; i < remaining.length; i++) {
    const d = dist(depotPt, pts[remaining[i]]);
    if (d < bestStart) {
      bestStart = d;
      startPos = i;
    }
  }

  const route = [remaining[startPos]];
  remaining.splice(startPos, 1);

  while (remaining.length > 0) {
    const last = pts[route[route.length - 1]];
    let bestPos = 0;
    let best = Infinity;

    for (let i = 0; i < remaining.length; i++) {
      const d = dist(last, pts[remaining[i]]);
      if (d < best) {
        best = d;
        bestPos = i;
      }
    }

    route.push(remaining[bestPos]);
    remaining.splice(bestPos, 1);
  }

  return twoOpt(route, pts, depotPt);
}

function assignClusters(pts, depotPt, maxRoutes) {
  const n = pts.length;
  if (n === 0) return [];

  const routeCount = Math.min(maxRoutes, n);
  const ordered = pts.map((p, idx) => {
    const angle = Math.atan2(p.y - depotPt.y, p.x - depotPt.x);
    const radius = dist(p, depotPt);
    return { idx, angle, radius };
  });

  ordered.sort((a, b) => {
    if (a.angle !== b.angle) return a.angle - b.angle;
    if (a.radius !== b.radius) return a.radius - b.radius;
    return a.idx - b.idx;
  });

  const baseSize = Math.floor(n / routeCount);
  const extra = n % routeCount;
  const clusters = [];
  let cursor = 0;

  for (let r = 0; r < routeCount; r++) {
    const size = baseSize + (r < extra ? 1 : 0);
    const cluster = [];
    for (let i = 0; i < size; i++) {
      cluster.push(ordered[cursor++].idx);
    }
    clusters.push(cluster);
  }

  return clusters;
}

function improveBoundaries(clusters, pts, depotPt) {
  if (clusters.length <= 1) return clusters;

  const routeStarts = clusters.map(cluster => cluster.slice());

  for (let iter = 0; iter < 2; iter++) {
    let changed = false;

    for (let i = 0; i < routeStarts.length - 1; i++) {
      const a = routeStarts[i];
      const b = routeStarts[i + 1];
      if (a.length === 0 || b.length === 0) continue;

      const lastA = a[a.length - 1];
      const firstB = b[0];

      const current = dist(pts[lastA], depotPt) + dist(depotPt, pts[firstB]);
      const merged = dist(pts[lastA], pts[firstB]);

      if (merged + 1e-12 < current && a.length > 1) {
        b.unshift(a.pop());
        changed = true;
      }
    }

    if (!changed) break;
  }

  return routeStarts;
}

function solve() {
  const depotPt = { x: depot[0], y: depot[1] };
  const pts = packages.map(p => ({ id: p.id, x: p.x, y: p.y }));

  if (pts.length === 0) {
    return { routes: [] };
  }

  let clusters = assignClusters(pts, depotPt, truckCount);
  clusters = improveBoundaries(clusters, pts, depotPt);

  const routes = clusters.map(cluster => buildRoute(cluster, pts, depotPt).map(idx => pts[idx].id));
  return { routes };
}

const result = solve();
process.stdout.write(JSON.stringify(result));
