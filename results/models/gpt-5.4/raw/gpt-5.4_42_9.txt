%_ptr_Output_v4float = OpTypePointer Output %v4float
%main = OpFunction %void None %3
%5 = OpLabel
%startPosPtr = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%startPos4 = OpLoad %v4float %startPosPtr
%endPosPtr = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
%endPos4 = OpLoad %v4float %endPosPtr
%prevPosPtr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%prevPos4 = OpLoad %v4float %prevPosPtr
%nextPosPtr = OpAccessChain %_ptr_Input_v4float %gl_in %int_3 %int_0

%startXY = OpVectorShuffle %v2float %startPos4 %startPos4 0 1
%endXY = OpVectorShuffle %v2float %endPos4 %endPos4 0 1
%prevXY = OpVectorShuffle %v2float %prevPos4 %prevPos4 0 1
%nextXY = OpVectorShuffle %v2float %nextPos4 %nextPos4 0 1

%delta0 = OpFSub %v2float %endXY %prevXY
%t0 = OpExtInst %v2float %1 Normalize %delta0
%delta1 = OpFSub %v2float %nextXY %startXY
%t1 = OpExtInst %v2float %1 Normalize %delta1

%t0x = OpCompositeExtract %float %t0 0
%t0y = OpCompositeExtract %float %t0 1
%neg_t0y = OpFNegate %float %t0y
%perp0 = OpCompositeConstruct %v2float %neg_t0y %t0x

%t1x = OpCompositeExtract %float %t1 0
%t1y = OpCompositeExtract %float %t1 1
%neg_t1y = OpFNegate %float %t1y
%perp1 = OpCompositeConstruct %v2float %neg_t1y %t1x

%offset0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
%offset1 = OpVectorTimesScalar %v2float %perp1 %float_0_025

%startPlusXY = OpFAdd %v2float %startXY %offset0
%startMinusXY = OpFSub %v2float %startXY %offset0
%endPlusXY = OpFAdd %v2float %endXY %offset1
%endMinusXY = OpFSub %v2float %endXY %offset1

%startZ = OpCompositeExtract %float %startPos4 2
%startW = OpCompositeExtract %float %startPos4 3
%endZ = OpCompositeExtract %float %endPos4 2
%endW = OpCompositeExtract %float %endPos4 3

%spx = OpCompositeExtract %float %startPlusXY 0
%spy = OpCompositeExtract %float %startPlusXY 1
%startPlusPos = OpCompositeConstruct %v4float %spx %spy %startZ %startW
%smx = OpCompositeExtract %float %startMinusXY 0
%smy = OpCompositeExtract %float %startMinusXY 1
%startMinusPos = OpCompositeConstruct %v4float %smx %smy %startZ %startW
%epx = OpCompositeExtract %float %endPlusXY 0
%epy = OpCompositeExtract %float %endPlusXY 1
%endPlusPos = OpCompositeConstruct %v4float %epx %epy %endZ %endW
%emx = OpCompositeExtract %float %endMinusXY 0
%emy = OpCompositeExtract %float %endMinusXY 1
%endMinusPos = OpCompositeConstruct %v4float %emx %emy %endZ %endW

%color1Ptr = OpAccessChain %_ptr_Input_v3float %color %int_1
%color1 = OpLoad %v3float %color1Ptr
%color2Ptr = OpAccessChain %_ptr_Input_v3float %color %int_2
%color2 = OpLoad %v3float %color2Ptr

%outPosPtr0 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %outPosPtr0 %startPlusPos
OpStore %outColor %color1
OpEmitVertex

OpStore %outPosPtr0 %endPlusPos

OpStore %outColor %color2

OpEndPrimitive
