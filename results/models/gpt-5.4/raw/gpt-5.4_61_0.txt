declare const require: any;
const fs = require('fs');

type Point = { x: number; y: number };
type Package = { id: string | number; x: number; y: number };
type Input = {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
};

type IndexedPackage = {
  idx: number;
  id: string | number;
  x: number;
  y: number;
  angle: number;
  radial: number;
};

function dist(a: Point, b: Point): number {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
}

function routeDistance(route: IndexedPackage[], depot: Point): number {
  if (route.length === 0) return 0;
  let total = dist(depot, route[0]);
  for (let i = 1; i < route.length; i++) {
    total += dist(route[i - 1], route[i]);
  }
  total += dist(route[route.length - 1], depot);
  return total;
}

function twoOpt(route: IndexedPackage[], depot: Point): void {
  const n = route.length;
  if (n < 4) return;

  let improved = true;
  let passes = 0;
  const maxPasses = 6;

  while (improved && passes < maxPasses) {
    improved = false;
    passes++;

    for (let i = 0; i < n - 2; i++) {
      const a = i === 0 ? depot : route[i - 1];
      const b = route[i];

      for (let k = i + 1; k < n - 1; k++) {
        const c = route[k];
        const d = route[k + 1];

        const before = dist(a, b) + dist(c, d);
        const after = dist(a, c) + dist(b, d);

        if (after + 1e-9 < before) {
          let l = i;
          let r = k;
          while (l < r) {
            const tmp = route[l];
            route[l] = route[r];
            route[r] = tmp;
            l++;
            r--;
          }
          improved = true;
        }
      }
    }
  }
}

function improveBoundaries(routes: IndexedPackage[][], depot: Point): void {
  if (routes.length < 2) return;

  let changed = true;
  let rounds = 0;
  const maxRounds = 4;

  while (changed && rounds < maxRounds) {
    changed = false;
    rounds++;

    for (let i = 0; i < routes.length - 1; i++) {
      const left = routes[i];
      const right = routes[i + 1];
      if (left.length === 0 || right.length === 0) continue;

      const base = routeDistance(left, depot) + routeDistance(right, depot);

      const leftToRight = left[left.length - 1];
      const newLeft1 = left.slice(0, left.length - 1);
      const newRight1 = [leftToRight, ...right];
      const cost1 = routeDistance(newLeft1, depot) + routeDistance(newRight1, depot);

      const rightToLeft = right[0];
      const newLeft2 = [...left, rightToLeft];
      const newRight2 = right.slice(1);
      const cost2 = routeDistance(newLeft2, depot) + routeDistance(newRight2, depot);

      if (cost1 + 1e-9 < base && newLeft1.length > 0) {
        left.pop();
        right.unshift(leftToRight);
        changed = true;
      } else if (cost2 + 1e-9 < base && newRight2.length > 0) {
        right.shift();
        left.push(rightToLeft);
        changed = true;
      }
    }
  }
}

function solve(data: Input): { routes: Array<Array<string | number>> } {
  const depot: Point = { x: data.depot[0], y: data.depot[1] };
  const packages = data.packages;
  const n = packages.length;

  if (n === 0 || data.truckCount <= 0) {
    return { routes: [] };
  }

  const k = Math.min(Math.max(1, data.truckCount), n);

  const indexed: IndexedPackage[] = packages.map((p, idx) => {
    const dx = p.x - depot.x;
    const dy = p.y - depot.y;
    return {
      idx,
      id: p.id,
      x: p.x,
      y: p.y,
      angle: Math.atan2(dy, dx),
      radial: Math.hypot(dx, dy),
    };
  });

  indexed.sort((a, b) => {
    if (a.angle !== b.angle) return a.angle - b.angle;
    if (a.radial !== b.radial) return a.radial - b.radial;
    return a.idx - b.idx;
  });

  const totalRadial = indexed.reduce((sum, p) => sum + p.radial, 0);
  const targetWeight = totalRadial > 0 ? totalRadial / k : n / k;

  const routes: IndexedPackage[][] = [];
  let current: IndexedPackage[] = [];
  let currentWeight = 0;
  let remaining = n;
  let remainingRoutes = k;

  for (let i = 0; i < indexed.length; i++) {
    const p = indexed[i];
    current.push(p);
    currentWeight += totalRadial > 0 ? p.radial : 1;
    remaining--;

    const minNeededForFuture = remainingRoutes - 1;
    const canClose = remaining >= minNeededForFuture;
    const mustClose = remainingRoutes > 1 && remaining === minNeededForFuture;
    const enoughWeight = currentWeight >= targetWeight;

    if (remainingRoutes > 1 && canClose && (mustClose || enoughWeight)) {
      routes.push(current);
      current = [];
      currentWeight = 0;
      remainingRoutes--;
    }
  }
  if (current.length > 0) routes.push(current);

  while (routes.length < k) {
    routes.push([]);
  }

  for (const route of routes) {
    route.sort((a, b) => a.radial - b.radial);
    if (route.length >= 2) {
      const asc = routeDistance(route, depot);
      const descRoute = route.slice().reverse();
      const desc = routeDistance(descRoute, depot);
      if (desc + 1e-9 < asc) {
        route.reverse();
      }
    }
  }

  improveBoundaries(routes, depot);

  for (const route of routes) {
    twoOpt(route, depot);
  }

  const outputRoutes = routes.filter((route) => route.length > 0).map((route) => route.map((p) => p.id));
  return { routes: outputRoutes };
}

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);
const result = solve(data);
process.stdout.write(JSON.stringify(result));
