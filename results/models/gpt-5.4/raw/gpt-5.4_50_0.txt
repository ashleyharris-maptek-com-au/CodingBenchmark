#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // x=dt, y=sepRadius, z=alignRadius, w=cohRadius
    vec4 weights;      // x=sepWeight, y=alignWeight, z=cohWeight, w=maxSpeed
    vec4 bounds;       // x=boundSize, y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint n = counts.x;

    if (i >= n) {
        return;
    }

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    uint base = i * 2u;
    vec3 pos = inData[base].xyz;
    vec3 vel = inData[base + 1u].xyz;

    vec3 sepAccum = vec3(0.0);
    vec3 alignAccum = vec3(0.0);
    vec3 cohAccum = vec3(0.0);
    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    float sepRadius2 = sepRadius * sepRadius;
    float alignRadius2 = alignRadius * alignRadius;
    float cohRadius2 = cohRadius * cohRadius;

    for (uint j = 0u; j < n; ++j) {
        if (j == i) {
            continue;
        }

        uint otherBase = j * 2u;
        vec3 otherPos = inData[otherBase].xyz;
        vec3 otherVel = inData[otherBase + 1u].xyz;

        vec3 offset = otherPos - pos;
        float dist2 = dot(offset, offset);

        if (dist2 <= 0.0) {
            continue;
        }

        if (dist2 < sepRadius2) {
            float dist = sqrt(dist2);
            sepAccum -= offset / dist;
            sepCount++;
        }

        if (dist2 < alignRadius2) {
            alignAccum += otherVel;
            alignCount++;
        }

        if (dist2 < cohRadius2) {
            cohAccum += otherPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        accel += (sepAccum / float(sepCount)) * sepWeight;
    }

    if (alignCount > 0u) {
        vec3 avgVel = alignAccum / float(alignCount);
        accel += (avgVel - vel) * alignWeight;
    }

    if (cohCount > 0u) {
        vec3 center = cohAccum / float(cohCount);
        accel += (center - pos) * cohWeight;
    }

    float boundThreshold = boundSize * 0.4;
    float boundScale = boundSize * 0.5;

    if (pos.x > boundThreshold) {
        accel.x -= boundForce * ((pos.x - boundThreshold) / boundScale);
    } else if (pos.x < -boundThreshold) {
        accel.x += boundForce * ((-boundThreshold - pos.x) / boundScale);
    }

    if (pos.y > boundThreshold) {
        accel.y -= boundForce * ((pos.y - boundThreshold) / boundScale);
    } else if (pos.y < -boundThreshold) {
        accel.y += boundForce * ((-boundThreshold - pos.y) / boundScale);
    }

    if (pos.z > boundThreshold) {
        accel.z -= boundForce * ((pos.z - boundThreshold) / boundScale);
    } else if (pos.z < -boundThreshold) {
        accel.z += boundForce * ((-boundThreshold - pos.z) / boundScale);
    }

    vec3 newVel = vel + accel * dt;
    float speed = length(newVel);
    if (speed > maxSpeed && speed > 0.0) {
        newVel = (newVel / speed) * maxSpeed;
    }

    vec3 newPos = pos + newVel * dt;

    outData[base] = vec4(newPos, 0.0);
    outData[base + 1u] = vec4(newVel, 0.0);
}
