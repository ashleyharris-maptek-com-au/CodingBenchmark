type Cell = { x: number; y: number };
type RobotInput = { id: number; x: number; y: number };
type JobInput = { id: number; x: number; y: number; priority: number };

declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid: string[] = data.grid;
const robots: RobotInput[] = data.robots;
const jobs: JobInput[] = data.jobs;

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const cellCount = width * height;

function indexOf(x: number, y: number): number {
  return y * width + x;
}

function isOpen(x: number, y: number): boolean {
  return y >= 0 && y < height && x >= 0 && x < width && grid[y][x] !== '#';
}

function assertOpenPoint(x: number, y: number, label: string): void {
  if (!isOpen(x, y)) {
    throw new Error(`${label} is on a blocked or invalid cell`);
  }
}

for (const robot of robots) {
  assertOpenPoint(robot.x, robot.y, `Robot ${robot.id}`);
}
for (const job of jobs) {
  assertOpenPoint(job.x, job.y, `Job ${job.id}`);
}

const dirs = [1, 0, -1, 0, 1];
const queue = new Int32Array(cellCount);

function bfsFrom(startX: number, startY: number): Int32Array {
  const dist = new Int32Array(cellCount);
  dist.fill(-1);

  let head = 0;
  let tail = 0;
  const start = indexOf(startX, startY);
  dist[start] = 0;
  queue[tail++] = start;

  while (head < tail) {
    const cur = queue[head++];
    const cx = cur % width;
    const cy = (cur / width) | 0;
    const nextDist = dist[cur] + 1;

    for (let d = 0; d < 4; d++) {
      const nx = cx + dirs[d];
      const ny = cy + dirs[d + 1];
      if (!isOpen(nx, ny)) continue;
      const ni = indexOf(nx, ny);
      if (dist[ni] !== -1) continue;
      dist[ni] = nextDist;
      queue[tail++] = ni;
    }
  }

  return dist;
}

const robotCount = robots.length;
const jobCount = jobs.length;

const robotIndexById = new Map<number, number>();
for (let i = 0; i < robotCount; i++) {
  robotIndexById.set(robots[i].id, i);
}

const jobIndexById = new Map<number, number>();
for (let i = 0; i < jobCount; i++) {
  jobIndexById.set(jobs[i].id, i);
}

const originPoints: Cell[] = [];
for (const robot of robots) originPoints.push({ x: robot.x, y: robot.y });
for (const job of jobs) originPoints.push({ x: job.x, y: job.y });

const distMaps: Int32Array[] = originPoints.map((p) => bfsFrom(p.x, p.y));

const robotToJob: number[][] = Array.from({ length: robotCount }, () => new Array(jobCount).fill(-1));
const jobToJob: number[][] = Array.from({ length: jobCount }, () => new Array(jobCount).fill(-1));

for (let r = 0; r < robotCount; r++) {
  const dist = distMaps[r];
  for (let j = 0; j < jobCount; j++) {
    robotToJob[r][j] = dist[indexOf(jobs[j].x, jobs[j].y)];
  }
}

for (let a = 0; a < jobCount; a++) {
  const dist = distMaps[robotCount + a];
  for (let b = 0; b < jobCount; b++) {
    jobToJob[a][b] = dist[indexOf(jobs[b].x, jobs[b].y)];
  }
}

type PendingJob = {
  id: number;
  index: number;
  priority: number;
  x: number;
  y: number;
};

const pending: PendingJob[] = jobs.map((job, index) => ({
  id: job.id,
  index,
  priority: job.priority,
  x: job.x,
  y: job.y,
}));

pending.sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  if (a.id !== b.id) return a.id - b.id;
  if (a.y !== b.y) return a.y - b.y;
  return a.x - b.x;
});

const assignments: number[][] = [];
for (let robotId = 0; robotId <= Math.max(-1, ...robots.map((r) => r.id)); robotId++) {
  assignments.push([]);
}

const currentJobIndex: number[] = new Array(robotCount).fill(-1);

function distanceFromRobot(robotIdx: number, jobIdx: number): number {
  const lastJob = currentJobIndex[robotIdx];
  return lastJob === -1 ? robotToJob[robotIdx][jobIdx] : jobToJob[lastJob][jobIdx];
}

for (const job of pending) {
  let bestRobotIdx = -1;
  let bestDist = Number.POSITIVE_INFINITY;
  let bestAssignedCount = Number.POSITIVE_INFINITY;
  let bestRobotId = Number.POSITIVE_INFINITY;

  for (let r = 0; r < robotCount; r++) {
    const dist = distanceFromRobot(r, job.index);
    if (dist < 0) continue;

    const robotId = robots[r].id;
    const assignedCount = assignments[robotId]?.length ?? 0;

    let better = false;
    if (dist < bestDist) better = true;
    else if (dist === bestDist && assignedCount < bestAssignedCount) better = true;
    else if (dist === bestDist && assignedCount === bestAssignedCount && robotId < bestRobotId) better = true;

    if (better) {
      bestRobotIdx = r;
      bestDist = dist;
      bestAssignedCount = assignedCount;
      bestRobotId = robotId;
    }
  }

  if (bestRobotIdx === -1) {
    throw new Error(`Job ${job.id} is unreachable from all robots`);
  }

  const chosenRobotId = robots[bestRobotIdx].id;
  while (assignments.length <= chosenRobotId) assignments.push([]);
  assignments[chosenRobotId].push(job.id);
  currentJobIndex[bestRobotIdx] = job.index;
}
