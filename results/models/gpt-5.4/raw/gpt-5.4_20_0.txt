use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone)]
struct Sample {
    pos: [f64; 3],
    values: Vec<Option<f64>>,
}

struct Hole {
    id: String,
    samples: Vec<Sample>,
}

#[derive(Clone)]
struct PointRef {
    hole_idx: usize,
    sample_idx: usize,
    pos: [f64; 3],
    value: f64,
}

#[derive(Clone)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

struct PropertyStats {
    median: f64,
    mad_sigma: f64,
    min: f64,
    max: f64,
}

#[derive(Default)]
struct GridIndex {
    cell_size: f64,
    cells: HashMap<(i32, i32, i32), Vec<usize>>,
}

impl GridIndex {
    fn from_points(points: &[PointRef]) -> Self {
        if points.is_empty() {
            return Self::default();
        }

        let mut min = [f64::INFINITY; 3];
        let mut max = [f64::NEG_INFINITY; 3];
        for p in points {
            for a in 0..3 {
                min[a] = min[a].min(p.pos[a]);
                max[a] = max[a].max(p.pos[a]);
            }
        }

        let dx = (max[0] - min[0]).max(1.0);
        let dy = (max[1] - min[1]).max(1.0);
        let dz = (max[2] - min[2]).max(1.0);
        let volume = dx * dy * dz;
        let target_points_per_cell = 24.0;
        let approx_cells = (points.len() as f64 / target_points_per_cell).max(1.0);
        let cell_size = (volume / approx_cells).cbrt().max(1.0);

        let mut cells: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
        for (idx, p) in points.iter().enumerate() {
            let key = Self::cell_key_with_size(p.pos, cell_size);
            cells.entry(key).or_default().push(idx);
        }

        Self { cell_size, cells }
    }

    fn cell_key_with_size(pos: [f64; 3], cell_size: f64) -> (i32, i32, i32) {
        (
            (pos[0] / cell_size).floor() as i32,
            (pos[1] / cell_size).floor() as i32,
            (pos[2] / cell_size).floor() as i32,
        )
    }

    fn cell_key(&self, pos: [f64; 3]) -> (i32, i32, i32) {
        Self::cell_key_with_size(pos, self.cell_size)
    }

    fn candidate_indices(&self, pos: [f64; 3], radius_cells: i32, out: &mut Vec<usize>) {
        out.clear();
        let (cx, cy, cz) = self.cell_key(pos);
        for dx in -radius_cells..=radius_cells {
            for dy in -radius_cells..=radius_cells {
                for dz in -radius_cells..=radius_cells {
                    if let Some(bucket) = self.cells.get(&(cx + dx, cy + dy, cz + dz)) {
                        out.extend(bucket.iter().copied());
                    }
                }
            }
        }
    }
}

fn parse_f64(token: Option<&str>) -> Option<f64> {
    let s = token?;
    let v = s.parse::<f64>().ok()?;
    if v.is_finite() { Some(v) } else { None }
}

fn sample_position(start: [f64; 3], dir: [f64; 3], depth: f64, length: f64) -> [f64; 3] {
    let d = depth.clamp(0.0, length.max(0.0));
    [
        start[0] + dir[0] * d,
        start[1] + dir[1] * d,
        start[2] + dir[2] * d,
    ]
}

fn normalize_dir(dir: [f64; 3]) -> [f64; 3] {
    let norm = (dir[0] * dir[0] + dir[1] * dir[1] + dir[2] * dir[2]).sqrt();
    if norm > 0.0 && norm.is_finite() {
        [dir[0] / norm, dir[1] / norm, dir[2] / norm]
    } else {
        [0.0, 0.0, 1.0]
    }
}

fn median_in_place(values: &mut [f64]) -> f64 {
    if values.is_empty() {
        return 0.0;
    }
    let mid = values.len() / 2;
    values.select_nth_unstable_by(mid, |a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let hi = values[mid];
    if values.len() % 2 == 1 {
        hi
    } else {
        let lo_max = values[..mid]
            .iter()
            .copied()
            .max_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal))
            .unwrap_or(hi);
        (lo_max + hi) * 0.5
    }
}

fn robust_stats(points: &[PointRef]) -> Option<PropertyStats> {
    if points.is_empty() {
        return None;
    }
    let mut vals: Vec<f64> = points.iter().map(|p| p.value).filter(|v| v.is_finite()).collect();
    if vals.is_empty() {
        return None;
    }
    let min = vals
        .iter()
        .copied()
        .min_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal))
        .unwrap_or(0.0);
    let max = vals
        .iter()
        .copied()
        .max_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal))
        .unwrap_or(0.0);
    let median = median_in_place(&mut vals);
    let mut devs: Vec<f64> = points
        .iter()
        .map(|p| (p.value - median).abs())
        .filter(|v| v.is_finite())
        .collect();
    let mad = median_in_place(&mut devs);
    let mad_sigma = (mad * 1.4826).max(((max - min).abs()) * 1e-9).max(1e-12);
    Some(PropertyStats {
        median,
        mad_sigma,
        min,
        max,
    })
}

fn dist2(a: [f64; 3], b: [f64; 3]) -> f64 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

fn local_outlier_score(
    point_idx: usize,
    points: &[PointRef],
    grid: &GridIndex,
    stats: &PropertyStats,
    small_mode: bool,
    scratch: &mut Vec<usize>,
) -> Option<f64> {
    let p = &points[point_idx];
    let neighbor_target = if small_mode { 18 } else { 12 };
    let max_radius = if small_mode { 8 } else { 4 };
    let mut neighbors: Vec<(f64, f64)> = Vec::with_capacity(neighbor_target * 4);

    for radius in 0..=max_radius {
        grid.candidate_indices(p.pos, radius, scratch);
        for &idx in scratch.iter() {
            if idx == point_idx {
                continue;
            }
            let q = &points[idx];
            let d2 = dist2(p.pos, q.pos);
            if d2 == 0.0 {
                continue;
            }
            neighbors.push((d2, q.value));
        }
        if neighbors.len() >= neighbor_target || radius == max_radius {
            break;
        }
    }

    if neighbors.len() < 4 {
        let global_z = (p.value - stats.median).abs() / stats.mad_sigma;
        return if global_z >= 8.0 {
            Some((global_z / 12.0).clamp(0.0, 0.98))
        } else {
            None
        };
    }

    neighbors.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
    let limit = neighbors.len().min(neighbor_target.max(4));
    neighbors.truncate(limit);

    let mut weighted_sum = 0.0;
    let mut weight_total = 0.0;
    let mut local_vals = Vec::with_capacity(limit);
    for (d2, value) in neighbors.iter().copied() {
        let w = 1.0 / (d2.sqrt() + 1e-6);
        weighted_sum += value * w;
        weight_total += w;
        local_vals.push(value);
    }
    if weight_total <= 0.0 {
        return None;
    }
    let local_mean = weighted_sum / weight_total;
    let mut devs: Vec<f64> = local_vals.iter().map(|v| (v - local_mean).abs()).collect();
    let local_mad = median_in_place(&mut devs);
    let local_sigma = (local_mad * 1.4826).max(stats.mad_sigma * 0.15).max(1e-12);
    let diff = (p.value - local_mean).abs();
    let local_z = diff / local_sigma;
    let global_z = (p.value - stats.median).abs() / stats.mad_sigma;

    let spread = (stats.max - stats.min).abs();
    let duplicateish = spread > 0.0 && diff <= spread * 1e-12;
    if duplicateish {
        return None;
    }

    let score = local_z * 0.75 + global_z * 0.25;
    if score < 4.5 {
        return None;
    }

    let confidence = (1.0 - (-0.18 * (score - 4.0)).exp()).clamp(0.0, 0.999);
    Some(confidence)
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        println!("0");
        return;
    }
    if input.trim().is_empty() {
        println!("0");
        return;
    }

    let mut it = input.split_whitespace();
    let num_holes = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    let num_properties = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    if num_holes == 0 || num_properties == 0 {
        println!("0");
        return;
    }

    let mut property_names = Vec::with_capacity(num_properties);
    for i in 0..num_properties {
        property_names.push(
            it.next()
                .map(|s| s.to_string())
                .unwrap_or_else(|| format!("property_{}", i + 1)),
        );
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut total_samples = 0usize;

    for hole_idx in 0..num_holes {
        let hole_id = match it.next() {
            Some(s) => s.to_string(),
            None => break,
        };
        let start = [
            parse_f64(it.next()).unwrap_or(0.0),
            parse_f64(it.next()).unwrap_or(0.0),
            parse_f64(it.next()).unwrap_or(0.0),
        ];
        let dir = normalize_dir([
            parse_f64(it.next()).unwrap_or(0.0),
            parse_f64(it.next()).unwrap_or(0.0),
            parse_f64(it.next()).unwrap_or(1.0),
        ]);
        let length = parse_f64(it.next()).unwrap_or(0.0).max(0.0);
        let declared_samples = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);

        let mut samples = Vec::with_capacity(declared_samples);
        for _ in 0..declared_samples {
            let Some(depth_tok) = it.next() else { break };
            let depth = depth_tok.parse::<f64>().ok().unwrap_or(0.0);
            let mut values = Vec::with_capacity(num_properties);
            for _ in 0..num_properties {
                values.push(parse_f64(it.next()));
            }
            let pos = sample_position(start, dir, depth, length);
            samples.push(Sample { pos, values });
        }

        total_samples += samples.len();
        holes.push(Hole { id: if hole_id.is_empty() { format!("hole_{}", hole_idx + 1) } else { hole_id }, samples });
    }

    if holes.is_empty() || total_samples == 0 {
        println!("0");
        return;
    }

    let small_mode = total_samples <= 20_000;
    let huge_mode = total_samples >= 500_000;
    let per_property_cap = if huge_mode { 40 } else if total_samples >= 100_000 { 80 } else { 200 };

    let mut suspects: Vec<Suspect> = Vec::new();

    for prop_idx in 0..num_properties {
        let mut points: Vec<PointRef> = Vec::new();
        points.reserve(total_samples.min(200_000));
        for (hole_idx, hole) in holes.iter().enumerate() {
            for (sample_idx, sample) in hole.samples.iter().enumerate() {
                if let Some(value) = sample.values.get(prop_idx).and_then(|v| *v) {
                    if value.is_finite() {
                        points.push(PointRef {
                            hole_idx,
                            sample_idx,
                            pos: sample.pos,
                            value,
                        });
                    }
                }
            }
        }

        if points.len() < 5 {
            continue;
        }

        let Some(stats) = robust_stats(&points) else { continue };
        let grid = GridIndex::from_points(&points);
        let mut property_suspects: Vec<Suspect> = Vec::new();
        let mut scratch = Vec::new();

        for idx in 0..points.len() {
            if let Some(confidence) =
                local_outlier_score(idx, &points, &grid, &stats, small_mode, &mut scratch)
            {
                let p = &points[idx];
                property_suspects.push(Suspect {
                    hole_id: holes[p.hole_idx].id.clone(),
                    sample_idx: p.sample_idx,
                    property_name: property_names[prop_idx].clone(),
                    confidence,
                });
            }
        }

        property_suspects.sort_by(|a, b| {
            b.confidence
                .partial_cmp(&a.confidence)
                .unwrap_or(Ordering::Equal)
                .then_with(|| a.hole_id.cmp(&b.hole_id))
                .then_with(|| a.sample_idx.cmp(&b.sample_idx))
                .then_with(|| a.property_name.cmp(&b.property_name))
        });

        if property_suspects.len() > per_property_cap {
            property_suspects.truncate(per_property_cap);
        }
        suspects.extend(property_suspects);
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.property_name.cmp(&b.property_name))
    });

    println!("{}", suspects.len());
    for s in suspects {
        println!(
            "{} {} {} {:.6}",
            s.hole_id, s.sample_idx, s.property_name, s.confidence
        );
    }
}
