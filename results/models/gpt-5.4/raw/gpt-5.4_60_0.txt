const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input || '{}');

const machineCount = Math.max(1, data.machineCount | 0);
const rawTasks = Array.isArray(data.tasks) ? data.tasks : [];

if (rawTasks.length === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const taskMap = new Map();
for (const task of rawTasks) {
  const id = task.id;
  if (id === undefined || id === null) continue;
  taskMap.set(id, {
    id,
    duration: Math.max(0, Number(task.duration) || 0),
    deps: Array.isArray(task.deps) ? task.deps.slice() : [],
    deadline: Number.isFinite(Number(task.deadline)) ? Number(task.deadline) : Number.POSITIVE_INFINITY,
    weight: Math.max(0, Number(task.weight) || 0),
    succs: [],
    indegree: 0,
  });
}

const tasks = Array.from(taskMap.values());
const n = tasks.length;
if (n === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const indexOf = new Map();
for (let i = 0; i < n; i++) indexOf.set(tasks[i].id, i);

for (const task of tasks) {
  const filteredDeps = [];
  for (const depId of task.deps) {
    if (taskMap.has(depId) && depId !== task.id) filteredDeps.push(depId);
  }
  task.deps = filteredDeps;
}

for (const task of tasks) {
  task.indegree = task.deps.length;
  for (const depId of task.deps) {
    taskMap.get(depId).succs.push(task.id);
  }
}

const topo = [];
const queue = [];
for (const task of tasks) {
  if (task.indegree === 0) queue.push(task.id);
}
queue.sort((a, b) => {
  const ta = taskMap.get(a);
  const tb = taskMap.get(b);
  return ta.duration - tb.duration || String(a).localeCompare(String(b));
});

for (let head = 0; head < queue.length; head++) {
  const id = queue[head];
  topo.push(id);
  for (const succId of taskMap.get(id).succs) {
    const succ = taskMap.get(succId);
    succ.indegree--;
    if (succ.indegree === 0) queue.push(succId);
  }
}

if (topo.length !== n) {
  const fallback = tasks
    .slice()
    .sort((a, b) => String(a.id).localeCompare(String(b.id)))
    .map((task) => task.id);
  process.stdout.write(JSON.stringify({ order: fallback }));
  process.exit(0);
}

const revTopo = topo.slice().reverse();
const rank = new Map();
const downstreamWeight = new Map();
const succCount = new Map();

for (const id of revTopo) {
  const task = taskMap.get(id);
  let bestSuccRank = 0;
  let totalSuccWeight = 0;
  let totalSuccCount = 0;
  for (const succId of task.succs) {
    bestSuccRank = Math.max(bestSuccRank, rank.get(succId) || 0);
    totalSuccWeight += downstreamWeight.get(succId) || 0;
    totalSuccCount += 1 + (succCount.get(succId) || 0);
  }
  rank.set(id, task.duration + bestSuccRank);
  downstreamWeight.set(id, task.weight + totalSuccWeight);
  succCount.set(id, totalSuccCount);
}

const remainingDeps = new Map();
const ready = [];
for (const id of topo) {
  const task = taskMap.get(id);
  remainingDeps.set(id, task.deps.length);
  if (task.deps.length === 0) ready.push(id);
}

const finishTime = new Map();
const machineFree = new Array(machineCount).fill(0);
const scheduled = [];

function compareReady(aId, bId) {
  const a = taskMap.get(aId);
  const b = taskMap.get(bId);

  const aDeadline = Number.isFinite(a.deadline) ? a.deadline : Number.MAX_SAFE_INTEGER;
  const bDeadline = Number.isFinite(b.deadline) ? b.deadline : Number.MAX_SAFE_INTEGER;
  const aSlack = aDeadline - rank.get(aId);
  const bSlack = bDeadline - rank.get(bId);

  return (
    (b.weight - a.weight) ||
    (aSlack - bSlack) ||
    ((downstreamWeight.get(bId) || 0) - (downstreamWeight.get(aId) || 0)) ||
    ((rank.get(bId) || 0) - (rank.get(aId) || 0)) ||
    ((succCount.get(bId) || 0) - (succCount.get(aId) || 0)) ||
    (b.duration - a.duration) ||
    String(aId).localeCompare(String(bId))
  );
}

function pushReady(id) {
  ready.push(id);
}

while (ready.length > 0) {
  ready.sort(compareReady);
  const id = ready.shift();
  const task = taskMap.get(id);

  let depsDoneAt = 0;
  for (const depId of task.deps) {
    const t = finishTime.get(depId);
    if (t > depsDoneAt) depsDoneAt = t;
  }

  let bestMachine = 0;
  let bestStart = Math.max(machineFree[0], depsDoneAt);
  let bestScore = Infinity;

  for (let machine = 0; machine < machineCount; machine++) {
    const start = Math.max(machineFree[machine], depsDoneAt);
    const completion = start + task.duration;
    const lateness = Math.max(0, completion - task.deadline);
    const score =
      lateness * (task.weight * 1000000 + 1) +
      completion * 1024 +
      start;

    if (
      score < bestScore ||
      (score === bestScore && start < bestStart) ||
      (score === bestScore && start === bestStart && machine < bestMachine)
    ) {
      bestScore = score;
      bestStart = start;
      bestMachine = machine;
    }
  }

  const end = bestStart + task.duration;
  machineFree[bestMachine] = end;
  finishTime.set(id, end);
  scheduled.push({ id, start: bestStart, machine: bestMachine });

  for (const succId of task.succs) {
    const rem = remainingDeps.get(succId) - 1;
    remainingDeps.set(succId, rem);
    if (rem === 0) pushReady(succId);
  }
}

process.stdout.write(JSON.stringify({ schedule: scheduled }));
