using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private const double DefaultMu = 398600.4418;
    private const double EarthRadius = 6371.0;
    private const double TimeScale = 1800.0;
    private const double PlaneWeight = 0.45;
    private const double RadiusWeight = 0.20;
    private const double SpeedWeight = 0.10;
    private const double EccWeight = 0.15;
    private const double LongWeight = 0.10;

    private struct State
    {
        public Vec3 R;
        public Vec3 V;
    }

    private struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);

        public double Dot(Vec3 other) => X * other.X + Y * other.Y + Z * other.Z;

        public Vec3 Cross(Vec3 other)
        {
            return new Vec3(
                Y * other.Z - Z * other.Y,
                Z * other.X - X * other.Z,
                X * other.Y - Y * other.X);
        }

        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);

        public Vec3 Normalized()
        {
            double n = Norm();
            return n > 1e-12 ? this / n : new Vec3(0, 0, 0);
        }
    }

    private sealed class OrbitFeatures
    {
        public int Index;
        public State State;
        public double Radius;
        public double Speed;
        public double Energy;
        public double SemiMajorAxis;
        public double Eccentricity;
        public Vec3 AngularMomentum;
        public Vec3 PlaneNormal;
        public Vec3 EccentricityVector;
        public double Longitude;
        public double Inclination;
        public double Period;
        public double[] CostRow = Array.Empty<double>();
    }

    public static void Main()
    {
        var reader = new TokenReader();
        int n = reader.NextInt();
        double mu = reader.TryNextDouble(out double parsedMu) ? parsedMu : DefaultMu;

        State start = ReadState(reader);
        var stations = new OrbitFeatures[n];
        for (int i = 0; i < n; i++)
        {
            stations[i] = BuildFeatures(ReadState(reader), i, mu);
        }

        var startFeatures = BuildFeatures(start, -1, mu);
        ComputeCosts(startFeatures, stations, mu);

        int[] route = Solve(stations, startFeatures, mu);
        Console.WriteLine(string.Join(" ", route));
    }

    private static State ReadState(TokenReader reader)
    {
        return new State
        {
            R = new Vec3(reader.NextDouble(), reader.NextDouble(), reader.NextDouble()),
            V = new Vec3(reader.NextDouble(), reader.NextDouble(), reader.NextDouble())
        };
    }

    private static OrbitFeatures BuildFeatures(State state, int index, double mu)
    {
        double r = state.R.Norm();
        double v = state.V.Norm();
        Vec3 h = state.R.Cross(state.V);
        double hNorm = h.Norm();
        Vec3 plane = hNorm > 1e-12 ? h / hNorm : new Vec3(0, 0, 1);
        Vec3 eVec = (state.V.Cross(h) / mu) - (state.R / Math.Max(r, 1e-9));
        double e = eVec.Norm();
        double energy = 0.5 * v * v - mu / Math.Max(r, 1e-9);
        double a = Math.Abs(energy) > 1e-9 ? -mu / (2.0 * energy) : 1e18;
        if (a < EarthRadius) a = r;
        double incl = Math.Acos(Clamp(plane.Z, -1.0, 1.0));
        double lon = Math.Atan2(state.R.Y, state.R.X);
        double period = a > EarthRadius ? 2.0 * Math.PI * Math.Sqrt(a * a * a / mu) : 0.0;

        return new OrbitFeatures
        {
            Index = index,
            State = state,
            Radius = r,
            Speed = v,
            Energy = energy,
            SemiMajorAxis = a,
            Eccentricity = e,
            AngularMomentum = h,
            PlaneNormal = plane,
            EccentricityVector = eVec,
            Longitude = lon,
            Inclination = incl,
            Period = period
        };
    }

    private static void ComputeCosts(OrbitFeatures start, OrbitFeatures[] stations, double mu)
    {
        int n = stations.Length;
        for (int i = 0; i < n; i++)
        {
            stations[i].CostRow = new double[n];
        }

        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                double c = TransferCost(stations[i], stations[j], mu);
                stations[i].CostRow[j] = c;
                stations[j].CostRow[i] = c;
            }
        }

        start.CostRow = new double[n];
        for (int i = 0; i < n; i++)
        {
            start.CostRow[i] = TransferCost(start, stations[i], mu);
        }
    }

    private static int[] Solve(OrbitFeatures[] stations, OrbitFeatures start, double mu)
    {
        int n = stations.Length;
        if (n <= 1)
        {
            return n == 0 ? Array.Empty<int>() : new[] { 0 };
        }

        if (n <= 9)
        {
            return SolveExact(stations, start);
        }

        int[] route = BuildGreedyRoute(stations, start);

        if (n <= 18)
        {
            route = ImproveRoute(route, stations, start, passes: 14, candidateCount: Math.Min(10, n - 1));
        }
        else if (n <= 40)
        {
            route = BuildMultiStartRoute(stations, start, restarts: 12);
            route = ImproveRoute(route, stations, start, passes: 10, candidateCount: 14);
        }
        else
        {
            route = BuildClusteredRoute(stations, start);
            route = ImproveRoute(route, stations, start, passes: 8, candidateCount: 16);
        }

        return route;
    }

    private static int[] SolveExact(OrbitFeatures[] stations, OrbitFeatures start)
    {
        int n = stations.Length;
        int fullMask = 1 << n;
        double[,] dp = new double[fullMask, n];
        int[,] parent = new int[fullMask, n];
        for (int mask = 0; mask < fullMask; mask++)
        {
            for (int i = 0; i < n; i++)
            {
                dp[mask, i] = double.PositiveInfinity;
                parent[mask, i] = -1;
            }
        }

        for (int i = 0; i < n; i++)
        {
            dp[1 << i, i] = start.CostRow[i];
        }

        for (int mask = 1; mask < fullMask; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                double cur = dp[mask, last];
                if (double.IsPositiveInfinity(cur)) continue;
                int remaining = ((fullMask - 1) ^ mask);
                while (remaining != 0)
                {
                    int bit = remaining & -remaining;
                    int next = BitOperations.TrailingZeroCount(bit);
                    int nextMask = mask | bit;
                    double cand = cur + stations[last].CostRow[next];
                    if (cand < dp[nextMask, next])
                    {
                        dp[nextMask, next] = cand;
                        parent[nextMask, next] = last;
                    }
                    remaining ^= bit;
                }
            }
        }

        int finalMask = fullMask - 1;
        double best = double.PositiveInfinity;
        int bestLast = 0;
        for (int i = 0; i < n; i++)
        {
            if (dp[finalMask, i] < best)
            {
                best = dp[finalMask, i];
                bestLast = i;
            }
        }

        int[] route = new int[n];
        int maskWalk = finalMask;
        int lastWalk = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = lastWalk;
            int prev = parent[maskWalk, lastWalk];
            maskWalk ^= 1 << lastWalk;
            lastWalk = prev;
        }
        return route;
    }

    private static int[] BuildGreedyRoute(OrbitFeatures[] stations, OrbitFeatures start)
    {
        int n = stations.Length;
        bool[] used = new bool[n];
        int[] route = new int[n];
        int current = -1;

        for (int step = 0; step < n; step++)
        {
            int best = -1;
            double bestScore = double.PositiveInfinity;
            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                double score = current == -1 ? start.CostRow[i] : stations[current].CostRow[i];
                score += 0.08 * RadialBandPenalty(stations, i, used);
                if (score < bestScore)
                {
                    bestScore = score;
                    best = i;
                }
            }
            route[step] = best;
            used[best] = true;
            current = best;
        }

        return route;
    }

    private static int[] BuildMultiStartRoute(OrbitFeatures[] stations, OrbitFeatures start, int restarts)
    {
        int n = stations.Length;
        int[] bestRoute = BuildGreedyRoute(stations, start);
        double bestCost = RouteCost(bestRoute, stations, start);

        int[] seeds = Enumerable.Range(0, n)
            .OrderBy(i => start.CostRow[i])
            .Take(Math.Min(restarts, n))
            .ToArray();

        foreach (int seed in seeds)
        {
            int[] route = BuildSeededRoute(stations, start, seed);
            route = ImproveRoute(route, stations, start, passes: 4, candidateCount: 10);
            double cost = RouteCost(route, stations, start);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestRoute = route;
            }
        }

        return bestRoute;
    }

    private static int[] BuildSeededRoute(OrbitFeatures[] stations, OrbitFeatures start, int seed)
    {
        int n = stations.Length;
        bool[] used = new bool[n];
        int[] route = new int[n];
        route[0] = seed;
        used[seed] = true;
        int current = seed;

        for (int step = 1; step < n; step++)
        {
            int best = -1;
            double bestScore = double.PositiveInfinity;
            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                double score = stations[current].CostRow[i] + 0.15 * start.CostRow[i];
                if (score < bestScore)
                {
                    bestScore = score;
                    best = i;
                }
            }
            route[step] = best;
            used[best] = true;
            current = best;
        }
        return route;
    }

    private static int[] BuildClusteredRoute(OrbitFeatures[] stations, OrbitFeatures start)
    {
        int n = stations.Length;
        int bandCount = Math.Min(8, Math.Max(3, (int)Math.Round(Math.Sqrt(n) / 1.5)));
        var ordered = Enumerable.Range(0, n)
            .OrderBy(i => stations[i].SemiMajorAxis)
            .ThenBy(i => stations[i].Longitude)
            .ToArray();

        var bands = new List<List<int>>(bandCount);
        for (int i = 0; i < bandCount; i++) bands.Add(new List<int>());
        for (int i = 0; i < n; i++) bands[i * bandCount / n].Add(ordered[i]);

        foreach (var band in bands)
        {
            band.Sort((a, b) => stations[a].Longitude.CompareTo(stations[b].Longitude));
            if (band.Count >= 4)
            {
                int[] improved = ImproveRoute(band.ToArray(), stations, start, passes: 3, candidateCount: Math.Min(6, band.Count - 1));
                band.Clear();
                band.AddRange(improved);
            }
        }

        var bandOrder = bands
            .Select((band, idx) => new { idx, band, entry = band.Count == 0 ? double.PositiveInfinity : start.CostRow[band[0]] })
            .OrderBy(x => x.entry)
            .ThenBy(x => x.idx)
            .Select(x => x.idx)
            .ToArray();

        List<int> route = new List<int>(n);
        int previous = -1;
        foreach (int bi in bandOrder)
        {
            var band = bands[bi];
            if (band.Count == 0) continue;
            if (previous != -1 && band.Count > 1)
            {
                double forward = stations[previous].CostRow[band[0]];
                double reverse = stations[previous].CostRow[band[band.Count - 1]];
                if (reverse < forward)
                {
                    band.Reverse();
                }
            }
            route.AddRange(band);
            previous = band[band.Count - 1];
        }

        return route.ToArray();
    }

    private static int[] ImproveRoute(int[] initialRoute, OrbitFeatures[] stations, OrbitFeatures start, int passes, int candidateCount)
    {
        int n = initialRoute.Length;
        int[] route = (int[])initialRoute.Clone();
        var nearest = BuildNearestNeighbors(stations, candidateCount);

        for (int pass = 0; pass < passes; pass++)
        {
            bool changed = false;
            changed |= TwoOptPass(route, stations, start, nearest);
            changed |= RelocatePass(route, stations, start);
            if (!changed) break;
        }

        return route;
    }

    private static List<int>[] BuildNearestNeighbors(OrbitFeatures[] stations, int candidateCount)
    {
        int n = stations.Length;
        var nearest = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            nearest[i] = Enumerable.Range(0, n)
                .Where(j => j != i)
                .OrderBy(j => stations[i].CostRow[j])
                .Take(candidateCount)
                .ToList();
        }
        return nearest;
    }

    private static bool TwoOptPass(int[] route, OrbitFeatures[] stations, OrbitFeatures start, List<int>[] nearest)
    {
        int n = route.Length;
        int[] pos = new int[n];
        for (int i = 0; i < n; i++) pos[route[i]] = i;
        bool changed = false;

        for (int i = 0; i < n - 2; i++)
        {
            int a = route[i];
            int b = route[i + 1];
            double ab = EdgeCost(i == 0 ? -1 : a, b, stations, start, route, i + 1);

            foreach (int c in nearest[b])
            {
                int j = pos[c];
                if (j <= i + 1 || j >= n - 1) continue;
                int d = route[j + 1];

                double current = ab + stations[c].CostRow[d];
                double candidate = EdgeCost(i == 0 ? -1 : a, c, stations, start, route, j) + stations[b].CostRow[d];
                if (candidate + 1e-9 < current)
                {
                    Array.Reverse(route, i + 1, j - i);
                    for (int k = i + 1; k <= j; k++) pos[route[k]] = k;
                    changed = true;
                    b = route[i + 1];
                    ab = EdgeCost(i == 0 ? -1 : a, b, stations, start, route, i + 1);
                }
            }
        }

        return changed;
    }

    private static bool RelocatePass(int[] route, OrbitFeatures[] stations, OrbitFeatures start)
    {
        int n = route.Length;
        bool changed = false;

        for (int i = 0; i < n; i++)
        {
            int node = route[i];
            double removeGain = RemovalGain(route, i, stations, start);
            int bestPos = i;
            double bestDelta = 0.0;

            for (int j = 0; j < n; j++)
            {
                if (j == i || j == i - 1) continue;
                double insertCost = InsertionDelta(route, node, i, j, stations, start);
                double delta = insertCost - removeGain;
                if (delta < bestDelta - 1e-9)
                {
                    bestDelta = delta;
                    bestPos = j;
                }
            }

            if (bestPos != i)
            {
                ApplyRelocate(route, i, bestPos);
                changed = true;
            }
        }

        return changed;
    }

    private static void ApplyRelocate(int[] route, int from, int to)
    {
        int node = route[from];
        if (from < to)
        {
            Array.Copy(route, from + 1, route, from, to - from);
            route[to] = node;
        }
        else
        {
            Array.Copy(route, to + 1, route, to + 2, from - to - 1);
            route[to + 1] = node;
        }
    }

    private static double RemovalGain(int[] route, int index, OrbitFeatures[] stations, OrbitFeatures start)
    {
        int prev = index > 0 ? route[index - 1] : -1;
        int cur = route[index];
        int next = index + 1 < route.Length ? route[index + 1] : -1;
        double current = Edge(prev, cur, stations, start) + Edge(cur, next, stations, start);
        double bridged = Edge(prev, next, stations, start);
        return current - bridged;
    }

    private static double InsertionDelta(int[] route, int node, int currentIndex, int afterIndex, OrbitFeatures[] stations, OrbitFeatures start)
    {
        int left;
        int right;

        if (afterIndex < currentIndex)
        {
            left = route[afterIndex];
            right = afterIndex + 1 == currentIndex ? node : route[afterIndex + 1];
        }
        else
        {
            left = route[afterIndex];
            right = afterIndex + 1 < route.Length ? route[afterIndex + 1] : -1;
        }

        double replaced = Edge(left, right, stations, start);
        double added = Edge(left, node, stations, start) + Edge(node, right, stations, start);
        return added - replaced;
    }

    private static double RouteCost(int[] route, OrbitFeatures[] stations, OrbitFeatures start)
    {
        if (route.Length == 0) return 0.0;
        double total = start.CostRow[route[0]];
        for (int i = 1; i < route.Length; i++)
        {
            total += stations[route[i - 1]].CostRow[route[i]];
        }
        return total;
    }

    private static double Edge(int from, int to, OrbitFeatures[] stations, OrbitFeatures start)
    {
        if (to < 0) return 0.0;
        if (from < 0) return start.CostRow[to];
        return stations[from].CostRow[to];
    }

    private static double EdgeCost(int from, int to, OrbitFeatures[] stations, OrbitFeatures start, int[] route, int routePos)
    {
        return from < 0 ? start.CostRow[to] : stations[from].CostRow[to];
    }

    private static double TransferCost(OrbitFeatures a, OrbitFeatures b, double mu)
    {
        double planeChange = Math.Sqrt(Math.Max(0.0, 0.5 * (1.0 - Clamp(a.PlaneNormal.Dot(b.PlaneNormal), -1.0, 1.0))));
        double radiusDiff = Math.Abs(a.Radius - b.Radius) / Math.Max(1.0, 0.5 * (a.Radius + b.Radius));
        double speedDiff = Math.Abs(a.Speed - b.Speed) / Math.Max(1.0, 0.5 * (a.Speed + b.Speed));
        double eccDiff = Math.Abs(a.Eccentricity - b.Eccentricity);
        double longDiff = WrappedAngle(a.Longitude - b.Longitude) / Math.PI;
        double phasePenalty = EstimatePhasePenalty(a, b);

        double visA = Math.Sqrt(mu / Math.Max(a.Radius, EarthRadius + 100.0));
        double visB = Math.Sqrt(mu / Math.Max(b.Radius, EarthRadius + 100.0));
        double baseVelocity = 0.5 * (visA + visB);

        double weighted =
            PlaneWeight * planeChange +
            RadiusWeight * radiusDiff +
            SpeedWeight * speedDiff +
            EccWeight * eccDiff +
            LongWeight * longDiff +
            0.20 * phasePenalty;

        return baseVelocity * weighted;
    }

    private static double EstimatePhasePenalty(OrbitFeatures a, OrbitFeatures b)
    {
        double periodA = a.Period > 1.0 ? a.Period : TimeScale;
        double periodB = b.Period > 1.0 ? b.Period : TimeScale;
        double drift = Math.Abs((1.0 / periodA) - (1.0 / periodB)) * TimeScale;
        double phase = WrappedAngle(a.Longitude - b.Longitude) / Math.PI;
        return phase / (1.0 + 5.0 * drift);
    }

    private static double RadialBandPenalty(OrbitFeatures[] stations, int candidate, bool[] used)
    {
        double candidateA = stations[candidate].SemiMajorAxis;
        double best = double.PositiveInfinity;
        for (int i = 0; i < stations.Length; i++)
        {
            if (!used[i]) continue;
            double diff = Math.Abs(candidateA - stations[i].SemiMajorAxis) / Math.Max(1.0, candidateA);
            if (diff < best) best = diff;
        }
        return double.IsPositiveInfinity(best) ? 0.0 : best;
    }

    private static double WrappedAngle(double angle)
    {
        angle = Math.Abs(angle);
        while (angle > Math.PI) angle = Math.Abs(angle - 2.0 * Math.PI);
        return angle;
    }

    private static double Clamp(double value, double lo, double hi)
    {
        if (value < lo) return lo;
        if (value > hi) return hi;
        return value;
    }

    private sealed class TokenReader
    {
        private readonly Queue<string> _tokens = new Queue<string>();

        public string Next()
        {
            while (_tokens.Count == 0)
            {
                string? line = Console.ReadLine();
                if (line == null)
                {
                    throw new InvalidOperationException("Unexpected end of input.");
                }
                foreach (string token in line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries))
                {
                    _tokens.Enqueue(token);
                }
            }
            return _tokens.Dequeue();
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);

        public bool TryNextDouble(out double value)
        {
            value = 0.0;
            string token = Next();
            return double.TryParse(token, NumberStyles.Float | NumberStyles.AllowThousands, CultureInfo.InvariantCulture, out value);
        }
    }

    private static class BitOperations
    {
        public static int TrailingZeroCount(int value)
        {
            int count = 0;
            while ((value & 1) == 0)
            {
                value >>= 1;
                count++;
            }
            return count;
        }
    }
}
