using System;
using System.IO;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private const int PlantX = 0;
    private const int PlantY = 0;
    private const int MinOreValue = 1;
    private const int LargeWorldCellBudget = 8_000_000;

    public static void Main()
    {
        var reader = new FastInput(Console.OpenStandardInput());
        int xCount = reader.ReadInt();
        int yCount = reader.ReadInt();
        int zCount = reader.ReadInt();
        double xBlockSize = reader.ReadDouble();
        double yBlockSize = reader.ReadDouble();
        double zBlockSize = reader.ReadDouble();

        if (xCount <= 0 || yCount <= 0 || zCount <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        long totalCells = (long)xCount * yCount * zCount;
        var strategy = ChooseStrategy(totalCells, xCount, yCount, zCount);

        var scanner = new WorldScanner(xCount, yCount, zCount, strategy);
        scanner.Scan(reader);

        var planner = new Planner(
            xCount,
            yCount,
            zCount,
            xBlockSize,
            yBlockSize,
            zBlockSize,
            strategy,
            scanner.Columns,
            scanner.Candidates);

        var operations = planner.Plan();
        WriteOperations(operations);
    }

    private static SolverStrategy ChooseStrategy(long totalCells, int xCount, int yCount, int zCount)
    {
        if (totalCells <= 2_000_000)
        {
            return SolverStrategy.Detailed;
        }

        if ((long)xCount * yCount <= LargeWorldCellBudget / Math.Max(1, zCount))
        {
            return SolverStrategy.Balanced;
        }

        return SolverStrategy.Sparse;
    }

    private static void WriteOperations(List<Operation> operations)
    {
        var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        writer.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            writer.Write(op.StepCount);
            writer.Write(' ');
            writer.Write(op.X);
            writer.Write(' ');
            writer.Write(op.Y);
            writer.Write(' ');
            writer.Write(op.Z);
            for (int i = 0; i < op.Path.Count; i++)
            {
                writer.Write(' ');
                writer.Write(op.Path[i].X);
                writer.Write(' ');
                writer.Write(op.Path[i].Y);
            }
            writer.Write(' ');
            writer.Write(op.DestX);
            writer.Write(' ');
            writer.Write(op.DestY);
            writer.WriteLine();
        }
        writer.Flush();
    }

    private enum SolverStrategy
    {
        Detailed,
        Balanced,
        Sparse
    }

    private readonly struct Step
    {
        public readonly int X;
        public readonly int Y;

        public Step(int x, int y)
        {
            X = x;
            Y = y;
        }
    }

    private sealed class Operation
    {
        public int X;
        public int Y;
        public int Z;
        public List<Step> Path = new List<Step>();
        public int DestX;
        public int DestY;
        public int StepCount => Path.Count;
    }

    private struct ColumnInfo
    {
        public int TopSolidZ;
        public int TopSolidValue;
        public int TopSolidMass;
        public int SurfaceOreCount;
        public int BestOreZ;
        public int BestOreValue;
        public int BestOreMass;
        public long BestOreProfitScore;
    }

    private readonly struct Candidate
    {
        public readonly int X;
        public readonly int Y;
        public readonly int Z;
        public readonly int Mass;
        public readonly int Value;
        public readonly long ProfitScore;

        public Candidate(int x, int y, int z, int mass, int value, long profitScore)
        {
            X = x;
            Y = y;
            Z = z;
            Mass = mass;
            Value = value;
            ProfitScore = profitScore;
        }
    }

    private sealed class WorldScanner
    {
        private readonly int _xCount;
        private readonly int _yCount;
        private readonly int _zCount;
        private readonly SolverStrategy _strategy;

        public ColumnInfo[] Columns { get; }
        public List<Candidate> Candidates { get; }

        public WorldScanner(int xCount, int yCount, int zCount, SolverStrategy strategy)
        {
            _xCount = xCount;
            _yCount = yCount;
            _zCount = zCount;
            _strategy = strategy;
            Columns = new ColumnInfo[xCount * yCount];
            Candidates = new List<Candidate>(EstimateCandidateCapacity(xCount, yCount, strategy));
        }

        public void Scan(FastInput reader)
        {
            for (int x = 0; x < _xCount; x++)
            {
                for (int y = 0; y < _yCount; y++)
                {
                    int index = x * _yCount + y;
                    var column = new ColumnInfo
                    {
                        TopSolidZ = -1,
                        BestOreZ = -1
                    };

                    for (int z = 0; z < _zCount; z++)
                    {
                        int d = reader.ReadByte();
                        int mass = d & 15;
                        int value = d >> 4;
                        if (mass <= 0)
                        {
                            continue;
                        }

                        column.TopSolidZ = z;
                        column.TopSolidValue = value;
                        column.TopSolidMass = mass;

                        if (value >= MinOreValue)
                        {
                            column.SurfaceOreCount++;
                            long score = ((long)value * 1000L - 1L) * mass - z;
                            if (score > column.BestOreProfitScore)
                            {
                                column.BestOreProfitScore = score;
                                column.BestOreZ = z;
                                column.BestOreValue = value;
                                column.BestOreMass = mass;
                            }
                        }
                    }

                    Columns[index] = column;
                    if (column.BestOreZ >= 0 && ShouldKeepCandidate(column))
                    {
                        Candidates.Add(new Candidate(
                            x,
                            y,
                            column.BestOreZ,
                            column.BestOreMass,
                            column.BestOreValue,
                            column.BestOreProfitScore));
                    }
                }
            }
        }

        private bool ShouldKeepCandidate(ColumnInfo column)
        {
            return _strategy switch
            {
                SolverStrategy.Detailed => true,
                SolverStrategy.Balanced => column.BestOreValue >= 2 || column.BestOreZ <= 200,
                _ => column.BestOreValue >= 4 || column.BestOreZ <= 40
            };
        }

        private static int EstimateCandidateCapacity(int xCount, int yCount, SolverStrategy strategy)
        {
            long columns = (long)xCount * yCount;
            long cap = strategy switch
            {
                SolverStrategy.Detailed => columns,
                SolverStrategy.Balanced => columns / 3,
                _ => columns / 10
            };
            if (cap < 16) cap = 16;
            if (cap > 2_000_000) cap = 2_000_000;
            return (int)cap;
        }
    }

    private sealed class Planner
    {
        private readonly int _xCount;
        private readonly int _yCount;
        private readonly SolverStrategy _strategy;
        private readonly ColumnInfo[] _columns;
        private readonly List<Candidate> _candidates;

        public Planner(
            int xCount,
            int yCount,
            int zCount,
            double xBlockSize,
            double yBlockSize,
            double zBlockSize,
            SolverStrategy strategy,
            ColumnInfo[] columns,
            List<Candidate> candidates)
        {
            _xCount = xCount;
            _yCount = yCount;
            _strategy = strategy;
            _columns = columns;
            _candidates = candidates;
        }

        public List<Operation> Plan()
        {
            TrimCandidates();
            _candidates.Sort(static (a, b) => b.ProfitScore.CompareTo(a.ProfitScore));

            int budget = _strategy switch
            {
                SolverStrategy.Detailed => Math.Min(_candidates.Count, 50_000),
                SolverStrategy.Balanced => Math.Min(_candidates.Count, 12_000),
                _ => Math.Min(_candidates.Count, 2_500)
            };

            var operations = new List<Operation>(budget);
            int accepted = 0;
            for (int i = 0; i < _candidates.Count && accepted < budget; i++)
            {
                Candidate c = _candidates[i];
                if (!IsStillTopAccessible(c))
                {
                    continue;
                }

                if (!TryBuildPathToPlant(c.X, c.Y, c.Z, out var path))
                {
                    continue;
                }

                long revenue = 1000L * c.Mass * c.Value;
                long cost = EstimateMoveCost(c.Mass, c.Z, path);
                if (revenue <= cost)
                {
                    continue;
                }

                operations.Add(new Operation
                {
                    X = c.X,
                    Y = c.Y,
                    Z = c.Z,
                    Path = path,
                    DestX = PlantX,
                    DestY = PlantY
                });
                accepted++;

                int idx = c.X * _yCount + c.Y;
                var col = _columns[idx];
                if (col.TopSolidZ == c.Z)
                {
                    col.TopSolidZ--;
                }
                _columns[idx] = col;
            }

            return operations;
        }

        private void TrimCandidates()
        {
            int limit = _strategy switch
            {
                SolverStrategy.Detailed => _candidates.Count,
                SolverStrategy.Balanced => Math.Min(_candidates.Count, 300_000),
                _ => Math.Min(_candidates.Count, 80_000)
            };

            if (_candidates.Count <= limit)
            {
                return;
            }

            _candidates.Sort(static (a, b) => b.ProfitScore.CompareTo(a.ProfitScore));
            _candidates.RemoveRange(limit, _candidates.Count - limit);
        }

        private bool IsStillTopAccessible(Candidate c)
        {
            int idx = c.X * _yCount + c.Y;
            return _columns[idx].TopSolidZ == c.Z;
        }

        private bool TryBuildPathToPlant(int x, int y, int startZ, out List<Step> path)
        {
            path = new List<Step>(Math.Abs(x - PlantX) + Math.Abs(y - PlantY));
            int currentX = x;
            int currentY = y;
            int currentHeight = startZ;

            while (currentX != PlantX || currentY != PlantY)
            {
                int nextX = currentX;
                int nextY = currentY;

                if (currentX > PlantX && currentY > PlantY)
                {
                    int hx = GetTopHeight(currentX - 1, currentY);
                    int hy = GetTopHeight(currentX, currentY - 1);
                    if (hx <= hy)
                    {
                        nextX--;
                    }
                    else
                    {
                        nextY--;
                    }
                }
                else if (currentX > PlantX)
                {
                    nextX--;
                }
                else if (currentY > PlantY)
                {
                    nextY--;
                }
                else
                {
                    return false;
                }

                int nextHeight = GetTopHeight(nextX, nextY) + 1;
                if (nextHeight > currentHeight + 1)
                {
                    return false;
                }

                path.Add(new Step(nextX, nextY));
                currentX = nextX;
                currentY = nextY;
                currentHeight = nextHeight;
            }

            return true;
        }

        private int GetTopHeight(int x, int y)
        {
            if ((uint)x >= (uint)_xCount || (uint)y >= (uint)_yCount)
            {
                return -1;
            }

            return _columns[x * _yCount + y].TopSolidZ;
        }

        private long EstimateMoveCost(int mass, int startZ, List<Step> path)
        {
            if (path.Count == 0)
            {
                return 0;
            }

            long cost = 0;
            int currentX = 0;
            int currentY = 0;
            int previousHeight = startZ;

            for (int i = 0; i < path.Count; i++)
            {
                int nextHeight = GetTopHeight(path[i].X, path[i].Y) + 1;
                int dx = Math.Abs(path[i].X - currentX);
                int dy = Math.Abs(path[i].Y - currentY);
                int distance = dx + dy;
                if (distance != 1)
                {
                    distance = 1;
                }

                if (nextHeight > previousHeight)
                {
                    cost += 5L * mass * distance;
                }
                else
                {
                    cost += 1L * mass * distance;
                }

                previousHeight = nextHeight;
                currentX = path[i].X;
                currentY = path[i].Y;
            }

            return cost;
        }
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _position;

        public FastInput(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        public int ReadByte()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length <= 0)
                {
                    return 0;
                }
            }

            return _buffer[_position++];
        }

        public int ReadInt()
        {
            SkipWhitespace();
            int sign = 1;
            int c = ReadByte();
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32)
            {
                value = checked(value * 10 + (c - '0'));
                c = ReadByte();
            }
            return value * sign;
        }

        public double ReadDouble()
        {
            SkipWhitespace();
            var sb = new ValueStringBuilder(stackalloc char[32]);
            int c = ReadByte();
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByte();
            }
            return double.Parse(sb.AsSpan(), CultureInfo.InvariantCulture);
        }

        private void SkipWhitespace()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= 32);

            _position--;
        }
    }

    private ref struct ValueStringBuilder
    {
        private Span<char> _initial;
        private char[]? _array;
        private int _pos;

        public ValueStringBuilder(Span<char> initial)
        {
            _initial = initial;
            _array = null;
            _pos = 0;
        }

        public void Append(char c)
        {
            if (_pos < Capacity)
            {
                this[_pos++] = c;
                return;
            }

            Grow();
            this[_pos++] = c;
        }

        public ReadOnlySpan<char> AsSpan()
        {
            return _array == null ? _initial.Slice(0, _pos) : _array.AsSpan(0, _pos);
        }

        private int Capacity => _array?.Length ?? _initial.Length;

        private ref char this[int index]
        {
            get
            {
                if (_array == null)
                {
                    return ref _initial[index];
                }
                return ref _array[index];
            }
        }

        private void Grow()
        {
            int newSize = Math.Max(Capacity * 2, 16);
            char[] arr = new char[newSize];
            AsSpan().CopyTo(arr);
            _array = arr;
        }
    }
}
