constexpr int INF = 1000000000;

enum Piece : int {
    EMPTY = 0,
    WP = 1, WN = 2, WB = 3, WR = 4, WQ = 5, WK = 6,
    BP = -1, BN = -2, BB = -3, BR = -4, BQ = -5, BK = -6
};

enum MoveFlag : int {
    FLAG_NONE = 0,
    FLAG_CAPTURE = 1 << 0,
    FLAG_EP = 1 << 1,
    FLAG_CASTLE_K = 1 << 2,
    FLAG_CASTLE_Q = 1 << 3,
    FLAG_PROMOTION = 1 << 4
};

struct Move {
    int from = 0;
    int to = 0;
    int promotion = 0;
    int flags = 0;
    int captured = EMPTY;
};

struct Undo {
    Move move;
    int movedPiece = EMPTY;
    int capturedPiece = EMPTY;
    int prevEnPassant = -1;
    int prevCastling = 0;
    int prevHalfmove = 0;
    int prevFullmove = 1;
    int prevKingSq[2] = { -1, -1 };
};

inline int file_of(int sq) { return sq & 7; }
inline int rank_of(int sq) { return sq >> 3; }
inline bool on_board(int sq) { return sq >= 0 && sq < 64; }
inline int color_of(int piece) { return piece > 0 ? 1 : (piece < 0 ? -1 : 0); }
inline int abs_piece(int piece) { return piece >= 0 ? piece : -piece; }

string square_to_string(int sq) {
    string s;
    s += static_cast<char>('a' + file_of(sq));
    s += static_cast<char>('1' + rank_of(sq));
    return s;
}

char piece_letter_for_move(int piece) {
    switch (abs_piece(piece)) {
    case 1: return 'p';
    case 2: return 'N';
    case 3: return 'B';
    case 4: return 'R';
    case 5: return 'Q';
    case 6: return 'K';
    default: return '?';
    }
}

char promotion_letter(int piece) {
    switch (abs_piece(piece)) {
    case 2: return 'N';
    case 3: return 'B';
    case 4: return 'R';
    case 5: return 'Q';
    default: return 'Q';
    }
}

struct Position {
    array<int, 64> board{};
    int sideToMove = 1;
    int castlingRights = 0;
    int enPassant = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
    int kingSq[2] = { -1, -1 };

    static Position from_fen(const string& fen) {
        Position pos;
        pos.board.fill(EMPTY);

        istringstream iss(fen);
        string boardPart, stmPart, castlePart, epPart;
        iss >> boardPart >> stmPart >> castlePart >> epPart >> pos.halfmoveClock >> pos.fullmoveNumber;

        int rank = 7;
        int file = 0;
        for (char c : boardPart) {
            if (c == '/') {
                --rank;
                file = 0;
                continue;
            }
            if (isdigit(static_cast<unsigned char>(c))) {
                file += c - '0';
                continue;
            }
            int sq = rank * 8 + file;
            int piece = EMPTY;
            switch (c) {
            case 'P': piece = WP; break;
            case 'N': piece = WN; break;
            case 'B': piece = WB; break;
            case 'R': piece = WR; break;
            case 'Q': piece = WQ; break;
            case 'K': piece = WK; break;
            case 'p': piece = BP; break;
            case 'n': piece = BN; break;
            case 'b': piece = BB; break;
            case 'r': piece = BR; break;
            case 'q': piece = BQ; break;
            case 'k': piece = BK; break;
            default: break;
            }
            pos.board[sq] = piece;
            if (piece == WK) pos.kingSq[0] = sq;
            if (piece == BK) pos.kingSq[1] = sq;
            ++file;
        }

        pos.sideToMove = (stmPart == "w") ? 1 : -1;
        pos.castlingRights = 0;
        if (castlePart.find('K') != string::npos) pos.castlingRights |= 1;
        if (castlePart.find('Q') != string::npos) pos.castlingRights |= 2;
        if (castlePart.find('k') != string::npos) pos.castlingRights |= 4;
        if (castlePart.find('q') != string::npos) pos.castlingRights |= 8;

        if (epPart != "-") {
            int epFile = epPart[0] - 'a';
            int epRank = epPart[1] - '1';
            pos.enPassant = epRank * 8 + epFile;
        }
        return pos;
    }
};

bool is_square_attacked(const Position& pos, int sq, int bySide) {
    const auto& b = pos.board;
    const int pawn = bySide == 1 ? WP : BP;
    const int knight = bySide == 1 ? WN : BN;
    const int bishop = bySide == 1 ? WB : BB;
    const int rook = bySide == 1 ? WR : BR;
    const int queen = bySide == 1 ? WQ : BQ;
    const int king = bySide == 1 ? WK : BK;

    if (bySide == 1) {
        if (file_of(sq) > 0) {
            int from = sq - 9;
            if (on_board(from) && b[from] == pawn) return true;
        }
        if (file_of(sq) < 7) {
            int from = sq - 7;
            if (on_board(from) && b[from] == pawn) return true;
        }
    } else {
        if (file_of(sq) > 0) {
            int from = sq + 7;
            if (on_board(from) && b[from] == pawn) return true;
        }
        if (file_of(sq) < 7) {
            int from = sq + 9;
            if (on_board(from) && b[from] == pawn) return true;
        }
    }

    static constexpr int knightOffsets[8] = { -17, -15, -10, -6, 6, 10, 15, 17 };
    for (int off : knightOffsets) {
        int from = sq + off;
        if (!on_board(from)) continue;
        int df = abs(file_of(from) - file_of(sq));
        int dr = abs(rank_of(from) - rank_of(sq));
        if ((df == 1 && dr == 2) || (df == 2 && dr == 1)) {
            if (b[from] == knight) return true;
        }
    }

    static constexpr int bishopDirs[4] = { -9, -7, 7, 9 };
    for (int dir : bishopDirs) {
        int cur = sq;
        while (true) {
            int next = cur + dir;
            if (!on_board(next)) break;
            if (abs(file_of(next) - file_of(cur)) != 1) break;
            cur = next;
            int piece = b[cur];
            if (piece != EMPTY) {
                if (piece == bishop || piece == queen) return true;
                break;
            }
        }
    }

    static constexpr int rookDirs[4] = { -8, -1, 1, 8 };
    for (int dir : rookDirs) {
        int cur = sq;
        while (true) {
            int next = cur + dir;
            if (!on_board(next)) break;
            if ((dir == -1 || dir == 1) && rank_of(next) != rank_of(cur)) break;
            cur = next;
            int piece = b[cur];
            if (piece != EMPTY) {
                if (piece == rook || piece == queen) return true;
                break;
            }
        }
    }

    for (int dr = -1; dr <= 1; ++dr) {
        for (int df = -1; df <= 1; ++df) {
            if (dr == 0 && df == 0) continue;
            int r = rank_of(sq) + dr;
            int f = file_of(sq) + df;
            if (r < 0 || r > 7 || f < 0 || f > 7) continue;
            int from = r * 8 + f;
            if (b[from] == king) return true;
        }
    }
    return false;
}

bool in_check(const Position& pos, int side) {
    int kingSquare = side == 1 ? pos.kingSq[0] : pos.kingSq[1];
    return is_square_attacked(pos, kingSquare, -side);
}

void add_move(vector<Move>& moves, int from, int to, int flags = FLAG_NONE, int promotion = 0, int captured = EMPTY) {
    Move m;
    m.from = from;
    m.to = to;
    m.flags = flags;
    m.promotion = promotion;
    m.captured = captured;
    moves.push_back(m);
}

vector<Move> generate_pseudo_legal(const Position& pos) {
    vector<Move> moves;
    const auto& b = pos.board;
    const int side = pos.sideToMove;

    for (int sq = 0; sq < 64; ++sq) {
        int piece = b[sq];
        if (piece == EMPTY || color_of(piece) != side) continue;
        int r = rank_of(sq);
        int f = file_of(sq);

        switch (abs_piece(piece)) {
        case 1: {
            int dir = side == 1 ? 8 : -8;
            int startRank = side == 1 ? 1 : 6;
            int promoRank = side == 1 ? 6 : 1;
            int one = sq + dir;
            if (on_board(one) && b[one] == EMPTY) {
                if (r == promoRank) {
                    add_move(moves, sq, one, FLAG_PROMOTION, side == 1 ? WQ : BQ);
                    add_move(moves, sq, one, FLAG_PROMOTION, side == 1 ? WR : BR);
                    add_move(moves, sq, one, FLAG_PROMOTION, side == 1 ? WB : BB);
                    add_move(moves, sq, one, FLAG_PROMOTION, side == 1 ? WN : BN);
                } else {
                    add_move(moves, sq, one);
                    int two = sq + 2 * dir;
                    if (r == startRank && b[two] == EMPTY) add_move(moves, sq, two);
                }
            }
            for (int df : { -1, 1 }) {
                int nf = f + df;
                if (nf < 0 || nf > 7) continue;
                int to = sq + dir + df;
                if (!on_board(to)) continue;
                if (b[to] != EMPTY && color_of(b[to]) == -side) {
                    if (r == promoRank) {
                        add_move(moves, sq, to, FLAG_CAPTURE | FLAG_PROMOTION, side == 1 ? WQ : BQ, b[to]);
                        add_move(moves, sq, to, FLAG_CAPTURE | FLAG_PROMOTION, side == 1 ? WR : BR, b[to]);
                        add_move(moves, sq, to, FLAG_CAPTURE | FLAG_PROMOTION, side == 1 ? WB : BB, b[to]);
                        add_move(moves, sq, to, FLAG_CAPTURE | FLAG_PROMOTION, side == 1 ? WN : BN, b[to]);
                    } else {
                        add_move(moves, sq, to, FLAG_CAPTURE, 0, b[to]);
                    }
                }
                if (to == pos.enPassant) {
                    add_move(moves, sq, to, FLAG_CAPTURE | FLAG_EP, 0, side == 1 ? BP : WP);
                }
            }
            break;
        }
        case 2: {
            static constexpr int offsets[8] = { -17, -15, -10, -6, 6, 10, 15, 17 };
            for (int off : offsets) {
                int to = sq + off;
                if (!on_board(to)) continue;
                int df = abs(file_of(to) - f);
                int dr = abs(rank_of(to) - r);
                if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
                if (b[to] == EMPTY) add_move(moves, sq, to);
                else if (color_of(b[to]) == -side) add_move(moves, sq, to, FLAG_CAPTURE, 0, b[to]);
            }
            break;
        }
        case 3:
        case 4:
        case 5: {
            vector<int> dirs;
            if (abs_piece(piece) == 3 || abs_piece(piece) == 5) dirs.insert(dirs.end(), { -9, -7, 7, 9 });
            if (abs_piece(piece) == 4 || abs_piece(piece) == 5) dirs.insert(dirs.end(), { -8, -1, 1, 8 });
            for (int dir : dirs) {
                int cur = sq;
                while (true) {
                    int to = cur + dir;
                    if (!on_board(to)) break;
                    if ((dir == -1 || dir == 1) && rank_of(to) != rank_of(cur)) break;
                    if ((dir == -9 || dir == 7) && file_of(to) != file_of(cur) - 1) break;
                    if ((dir == -7 || dir == 9) && file_of(to) != file_of(cur) + 1) break;
                    cur = to;
                    if (b[to] == EMPTY) {
                        add_move(moves, sq, to);
                    } else {
                        if (color_of(b[to]) == -side) add_move(moves, sq, to, FLAG_CAPTURE, 0, b[to]);
                        break;
                    }
                }
            }
            break;
        }
        case 6: {
            for (int dr = -1; dr <= 1; ++dr) {
                for (int df = -1; df <= 1; ++df) {
                    if (dr == 0 && df == 0) continue;
                    int nr = r + dr;
                    int nf = f + df;
                    if (nr < 0 || nr > 7 || nf < 0 || nf > 7) continue;
                    int to = nr * 8 + nf;
                    if (b[to] == EMPTY) add_move(moves, sq, to);
                    else if (color_of(b[to]) == -side) add_move(moves, sq, to, FLAG_CAPTURE, 0, b[to]);
                }
            }
            if (side == 1) {
                if ((pos.castlingRights & 1) && b[5] == EMPTY && b[6] == EMPTY &&
                    !is_square_attacked(pos, 4, -1) && !is_square_attacked(pos, 5, -1) && !is_square_attacked(pos, 6, -1)) {
                    add_move(moves, 4, 6, FLAG_CASTLE_K);
                }
                if ((pos.castlingRights & 2) && b[3] == EMPTY && b[2] == EMPTY && b[1] == EMPTY &&
                    !is_square_attacked(pos, 4, -1) && !is_square_attacked(pos, 3, -1) && !is_square_attacked(pos, 2, -1)) {
                    add_move(moves, 4, 2, FLAG_CASTLE_Q);
                }
            } else {
                if ((pos.castlingRights & 4) && b[61] == EMPTY && b[62] == EMPTY &&
                    !is_square_attacked(pos, 60, 1) && !is_square_attacked(pos, 61, 1) && !is_square_attacked(pos, 62, 1)) {
                    add_move(moves, 60, 62, FLAG_CASTLE_K);
                }
                if ((pos.castlingRights & 8) && b[59] == EMPTY && b[58] == EMPTY && b[57] == EMPTY &&
                    !is_square_attacked(pos, 60, 1) && !is_square_attacked(pos, 59, 1) && !is_square_attacked(pos, 58, 1)) {
                    add_move(moves, 60, 58, FLAG_CASTLE_Q);
                }
            }
            break;
        }
        default:
            break;
        }
    }
    return moves;
}

Undo make_move(Position& pos, const Move& move) {
    Undo u;
    u.move = move;
    u.prevEnPassant = pos.enPassant;
    u.prevCastling = pos.castlingRights;
    u.prevHalfmove = pos.halfmoveClock;
    u.prevFullmove = pos.fullmoveNumber;
    u.prevKingSq[0] = pos.kingSq[0];
    u.prevKingSq[1] = pos.kingSq[1];
    u.movedPiece = pos.board[move.from];
    u.capturedPiece = (move.flags & FLAG_EP) ? (pos.sideToMove == 1 ? BP : WP) : pos.board[move.to];

    int piece = pos.board[move.from];
    pos.board[move.from] = EMPTY;
    pos.enPassant = -1;

    if (abs_piece(piece) == 1 || move.flags & FLAG_CAPTURE) pos.halfmoveClock = 0;
    else ++pos.halfmoveClock;

    if (move.flags & FLAG_EP) {
        pos.board[move.to] = piece;
        int capSq = move.to + (pos.sideToMove == 1 ? -8 : 8);
        pos.board[capSq] = EMPTY;
    } else if (move.flags & FLAG_CASTLE_K) {
        pos.board[move.to] = piece;
        if (pos.sideToMove == 1) {
            pos.board[7] = EMPTY;
            pos.board[5] = WR;
            pos.kingSq[0] = 6;
        } else {
            pos.board[63] = EMPTY;
            pos.board[61] = BR;
            pos.kingSq[1] = 62;
        }
    } else if (move.flags & FLAG_CASTLE_Q) {
        pos.board[move.to] = piece;
        if (pos.sideToMove == 1) {
            pos.board[0] = EMPTY;
            pos.board[3] = WR;
            pos.kingSq[0] = 2;
        } else {
            pos.board[56] = EMPTY;
            pos.board[59] = BR;
            pos.kingSq[1] = 58;
        }
    } else {
        pos.board[move.to] = (move.flags & FLAG_PROMOTION) ? move.promotion : piece;
        if (piece == WK) pos.kingSq[0] = move.to;
        if (piece == BK) pos.kingSq[1] = move.to;
    }

    if (abs_piece(piece) == 1 && abs(move.to - move.from) == 16) {
        pos.enPassant = (move.from + move.to) / 2;
    }

    if (piece == WK) pos.castlingRights &= ~3;
    if (piece == BK) pos.castlingRights &= ~12;
    if (move.from == 0 || move.to == 0) pos.castlingRights &= ~2;
    if (move.from == 7 || move.to == 7) pos.castlingRights &= ~1;
    if (move.from == 56 || move.to == 56) pos.castlingRights &= ~8;
    if (move.from == 63 || move.to == 63) pos.castlingRights &= ~4;

    pos.sideToMove = -pos.sideToMove;
    if (pos.sideToMove == 1) ++pos.fullmoveNumber;
    return u;
}

void undo_move(Position& pos, const Undo& u) {
    pos.sideToMove = -pos.sideToMove;
    pos.enPassant = u.prevEnPassant;
    pos.castlingRights = u.prevCastling;
    pos.halfmoveClock = u.prevHalfmove;
    pos.fullmoveNumber = u.prevFullmove;
    pos.kingSq[0] = u.prevKingSq[0];
    pos.kingSq[1] = u.prevKingSq[1];

    const Move& move = u.move;
    pos.board[move.from] = u.movedPiece;

    if (move.flags & FLAG_EP) {
        pos.board[move.to] = EMPTY;
        int capSq = move.to + (pos.sideToMove == 1 ? -8 : 8);
        pos.board[capSq] = u.capturedPiece;
    } else if (move.flags & FLAG_CASTLE_K) {
        pos.board[move.to] = EMPTY;
        if (pos.sideToMove == 1) {
            pos.board[7] = WR;
            pos.board[5] = EMPTY;
        } else {
            pos.board[63] = BR;
            pos.board[61] = EMPTY;
        }
    } else if (move.flags & FLAG_CASTLE_Q) {
        pos.board[move.to] = EMPTY;
        if (pos.sideToMove == 1) {
            pos.board[0] = WR;
            pos.board[3] = EMPTY;
        } else {
            pos.board[56] = BR;
            pos.board[59] = EMPTY;
        }
    } else {
        pos.board[move.to] = u.capturedPiece;
    }
}

vector<Move> generate_legal_moves(Position& pos) {
    vector<Move> legal;
    vector<Move> pseudo = generate_pseudo_legal(pos);
    legal.reserve(pseudo.size());
    for (const Move& m : pseudo) {
        Undo u = make_move(pos, m);
        if (!in_check(pos, -pos.sideToMove)) legal.push_back(m);
        undo_move(pos, u);
    }
    return legal;
}

int evaluate(const Position& pos) {
    static constexpr int pieceValue[7] = { 0, 100, 320, 330, 500, 900, 0 };
    static constexpr int pawnTable[64] = {
         0,  0,  0,  0,  0,  0,  0,  0,
        50, 50, 50, 50, 50, 50, 50, 50,
        10, 10, 20, 30, 30, 20, 10, 10,
         5,  5, 10, 25, 25, 10,  5,  5,
         0,  0,  0, 20, 20,  0,  0,  0,
         5, -5,-10,  0,  0,-10, -5,  5,
         5, 10, 10,-20,-20, 10, 10,  5,
         0,  0,  0,  0,  0,  0,  0,  0
    };
    static constexpr int knightTable[64] = {
        -50,-40,-30,-30,-30,-30,-40,-50,
        -40,-20,  0,  5,  5,  0,-20,-40,
        -30,  5, 10, 15, 15, 10,  5,-30,
        -30,  0, 15, 20, 20, 15,  0,-30,
        -30,  5, 15, 20, 20, 15,  5,-30,
        -30,  0, 10, 15, 15, 10,  0,-30,
        -40,-20,  0,  0,  0,  0,-20,-40,
        -50,-40,-30,-30,-30,-30,-40,-50
    };
    static constexpr int bishopTable[64] = {
        -20,-10,-10,-10,-10,-10,-10,-20,
        -10,  5,  0,  0,  0,  0,  5,-10,
        -10, 10, 10, 10, 10, 10, 10,-10,
        -10,  0, 10, 10, 10, 10,  0,-10,
        -10,  5,  5, 10, 10,  5,  5,-10,
        -10,  0,  5, 10, 10,  5,  0,-10,
        -10,  0,  0,  0,  0,  0,  0,-10,
        -20,-10,-10,-10,-10,-10,-10,-20
    };
    static constexpr int rookTable[64] = {
         0,  0,  0,  5,  5,  0,  0,  0,
        -5,  0,  0,  0,  0,  0,  0, -5,
        -5,  0,  0,  0,  0,  0,  0, -5,
        -5,  0,  0,  0,  0,  0,  0, -5,
        -5,  0,  0,  0,  0,  0,  0, -5,
        -5,  0,  0,  0,  0,  0,  0, -5,
         5, 10, 10, 10, 10, 10, 10,  5,
         0,  0,  0,  0,  0,  0,  0,  0
    };
    static constexpr int queenTable[64] = {
        -20,-10,-10, -5, -5,-10,-10,-20,
        -10,  0,  0,  0,  0,  0,  0,-10,
        -10,  0,  5,  5,  5,  5,  0,-10,
         -5,  0,  5,  5,  5,  5,  0, -5,
          0,  0,  5,  5,  5,  5,  0, -5,
        -10,  5,  5,  5,  5,  5,  0,-10,
        -10,  0,  5,  0,  0,  0,  0,-10,
        -20,-10,-10, -5, -5,-10,-10,-20
    };
    static constexpr int kingTable[64] = {
        -30,-40,-40,-50,-50,-40,-40,-30,
        -30,-40,-40,-50,-50,-40,-40,-30,
        -30,-40,-40,-50,-50,-40,-40,-30,
        -30,-40,-40,-50,-50,-40,-40,-30,
        -20,-30,-30,-40,-40,-30,-30,-20,
        -10,-20,-20,-20,-20,-20,-20,-10,
         20, 20,  0,  0,  0,  0, 20, 20,
         20, 30, 10,  0,  0, 10, 30, 20
    };

    auto pst = [&](int piece, int sq) {
        int idx = piece > 0 ? sq : ((7 - rank_of(sq)) * 8 + file_of(sq));
        switch (abs_piece(piece)) {
        case 1: return pawnTable[idx];
        case 2: return knightTable[idx];
        case 3: return bishopTable[idx];
        case 4: return rookTable[idx];
        case 5: return queenTable[idx];
        case 6: return kingTable[idx];
        default: return 0;
        }
    };

    int score = 0;
    for (int sq = 0; sq < 64; ++sq) {
        int piece = pos.board[sq];
        if (piece == EMPTY) continue;
        int val = pieceValue[abs_piece(piece)] + pst(piece, sq);
        score += piece > 0 ? val : -val;
    }

    Position tmp = pos;
    vector<Move> ownMobility = generate_legal_moves(tmp);
    score += static_cast<int>(ownMobility.size()) * 2 * pos.sideToMove;

    tmp.sideToMove = -tmp.sideToMove;
    vector<Move> oppMobility = generate_legal_moves(tmp);
    score -= static_cast<int>(oppMobility.size()) * 2 * pos.sideToMove;

    return pos.sideToMove == 1 ? score : -score;
}

struct Searcher {
    chrono::steady_clock::time_point deadline;
    bool stopped = false;
    Move bestMove{};
    int nodes = 0;

    bool time_up() {
        return chrono::steady_clock::now() >= deadline;
    }

    int quiescence(Position& pos, int alpha, int beta) {
        if ((++nodes & 2047) == 0 && time_up()) {
            stopped = true;
            return 0;
        }
        int standPat = evaluate(pos);
        if (standPat >= beta) return beta;
        if (standPat > alpha) alpha = standPat;

        vector<Move> pseudo = generate_pseudo_legal(pos);
        for (const Move& m : pseudo) {
            if (!(m.flags & FLAG_CAPTURE)) continue;
            Undo u = make_move(pos, m);
            if (in_check(pos, -pos.sideToMove)) {
                undo_move(pos, u);
                continue;
            }
            int score = -quiescence(pos, -beta, -alpha);
            undo_move(pos, u);
            if (stopped) return 0;
            if (score >= beta) return beta;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }

    int search(Position& pos, int depth, int alpha, int beta, int ply) {
        if ((++nodes & 2047) == 0 && time_up()) {
            stopped = true;
            return 0;
        }
        if (depth == 0) return quiescence(pos, alpha, beta);

        vector<Move> moves = generate_legal_moves(pos);
        if (moves.empty()) {
            if (in_check(pos, pos.sideToMove)) return -MATE_SCORE + ply;
            return 0;
        }

        stable_sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            int movedA = pos.board[a.from];
            int movedB = pos.board[b.from];
            int sa = ((a.flags & FLAG_CAPTURE) ? 1000 + abs_piece(a.captured) * 10 - abs_piece(movedA) : 0)
                   + ((a.flags & FLAG_PROMOTION) ? 500 + abs_piece(a.promotion) : 0)
                   + ((a.flags & (FLAG_CASTLE_K | FLAG_CASTLE_Q)) ? 50 : 0);
            int sb = ((b.flags & FLAG_CAPTURE) ? 1000 + abs_piece(b.captured) * 10 - abs_piece(movedB) : 0)
                   + ((b.flags & FLAG_PROMOTION) ? 500 + abs_piece(b.promotion) : 0)
                   + ((b.flags & (FLAG_CASTLE_K | FLAG_CASTLE_Q)) ? 50 : 0);
            return sa > sb;
        });

        int bestScore = -INF;
        for (size_t i = 0; i < moves.size(); ++i) {
            Undo u = make_move(pos, moves[i]);
            int score = -search(pos, depth - 1, -beta, -alpha, ply + 1);
            undo_move(pos, u);
            if (stopped) return 0;

            if (score > bestScore) bestScore = score;
            if (score > alpha) {
                alpha = score;
                if (ply == 0) bestMove = moves[i];
            }
            if (alpha >= beta) break;
        }
        return bestScore;
    }

    Move iterative_deepening(Position& pos, double seconds) {
        deadline = chrono::steady_clock::now() + chrono::milliseconds(static_cast<int>(seconds * 1000.0));
        stopped = false;
        nodes = 0;

        vector<Move> rootMoves = generate_legal_moves(pos);
        if (rootMoves.empty()) return Move{};
        bestMove = rootMoves[0];

        for (int depth = 1; depth <= 64; ++depth) {
            int score = search(pos, depth, -INF, INF, 0);
            (void)score;
            if (stopped) break;
            if (time_up()) break;
        }
        return bestMove;
    }
};

string format_move(Position& pos, const Move& move) {
    if (move.flags & FLAG_CASTLE_K) return "O-O";
    if (move.flags & FLAG_CASTLE_Q) return "O-O-O";

    int movingPiece = pos.board[move.from];
    string out;
    out += piece_letter_for_move(movingPiece);
    out += square_to_string(move.from);
    out += (move.flags & FLAG_CAPTURE) ? "x" : "";
    out += square_to_string(move.to);
    if (move.flags & FLAG_EP) out += "ep";
    if (move.flags & FLAG_PROMOTION) {
        out += '/';
        out += promotion_letter(move.promotion);
    }

    Undo u = make_move(pos, move);
    vector<Move> replies = generate_legal_moves(pos);
    bool givesCheck = in_check(pos, pos.sideToMove);
    bool mate = givesCheck && replies.empty();
    undo_move(pos, u);

    if (mate) out += '#';
    else if (givesCheck) out += '+';
    return out;
}

} // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSeconds = 0.0;
    string fen;
    if (!(cin >> timeoutSeconds)) return 0;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    if (!getline(cin, fen)) return 0;

    Position pos = Position::from_fen(fen);
    Searcher searcher;

    double thinkTime = max(0.05, timeoutSeconds * 0.95);
    Move best = searcher.iterative_deepening(pos, thinkTime);

    vector<Move> legal = generate_legal_moves(pos);
    if (legal.empty()) return 0;

    bool found = false;
    for (const Move& m : legal) {
        if (m.from == best.from && m.to == best.to && m.flags == best.flags && m.promotion == best.promotion) {
            found = true;
            break;
        }
    }
    if (!found) best = legal[0];

    cout << format_move(pos, best) << '\n';
    return 0;
}
