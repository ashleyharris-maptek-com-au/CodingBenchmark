using System;
using System.Collections.Generic;
using System.Linq;

public static class Program
{
    private readonly struct Pos : IEquatable<Pos>
    {
        public readonly int X;
        public readonly int Y;

        public Pos(int x, int y)
        {
            X = x;
            Y = y;
        }

        public bool Equals(Pos other) => X == other.X && Y == other.Y;
        public override bool Equals(object? obj) => obj is Pos other && Equals(other);
        public override int GetHashCode() => HashCode.Combine(X, Y);
    }

    private sealed class Constraint
    {
        public int Required;
        public List<int> Variables = new List<int>();
    }

    private sealed class FrontierComponent
    {
        public List<Pos> Variables = new List<Pos>();
        public List<Constraint> Constraints = new List<Constraint>();
    }

    private static readonly int[] Dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
    private static readonly int[] Dy = { -1, -1, -1, 0, 0, 1, 1, 1 };

    public static void Main()
    {
        var header = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(header))
        {
            Console.WriteLine(0);
            return;
        }

        var parts = header.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int h = int.Parse(parts[0]);
        int w = int.Parse(parts[1]);

        var board = new char[h][];
        for (int y = 0; y < h; y++)
        {
            var line = Console.ReadLine() ?? string.Empty;
            if (line.Length < w)
            {
                line = line.PadRight(w, ' ');
            }
            else if (line.Length > w)
            {
                line = line.Substring(0, w);
            }

            board[y] = line.ToCharArray();
        }

        var safe = Solve(board, h, w);

        Console.WriteLine(safe.Count);
        foreach (var pos in safe.OrderBy(p => p.Y).ThenBy(p => p.X))
        {
            Console.WriteLine($"{pos.X} {pos.Y}");
        }
    }

    private static List<Pos> Solve(char[][] board, int h, int w)
    {
        var safe = new HashSet<Pos>();
        var mines = new HashSet<Pos>();

        RunLocalPropagation(board, h, w, safe, mines);

        var complexity = EstimateComplexity(board, h, w);
        if (complexity == 0)
        {
            return safe.ToList();
        }

        var components = BuildFrontierComponents(board, h, w, safe, mines, complexity);
        foreach (var component in components)
        {
            AnalyzeComponent(component, safe, mines, complexity);
        }

        return safe.ToList();
    }

    private static int EstimateComplexity(char[][] board, int h, int w)
    {
        long unknown = 0;
        long numbered = 0;
        long frontierUnknown = 0;

        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                char c = board[y][x];
                if (c == ' ')
                {
                    unknown++;
                    bool frontier = false;
                    for (int dir = 0; dir < 8 && !frontier; dir++)
                    {
                        int nx = x + Dx[dir];
                        int ny = y + Dy[dir];
                        if ((uint)nx < (uint)w && (uint)ny < (uint)h)
                        {
                            char nc = board[ny][nx];
                            if (nc >= '0' && nc <= '8')
                            {
                                frontier = true;
                            }
                        }
                    }

                    if (frontier)
                    {
                        frontierUnknown++;
                    }
                }
                else if (c >= '0' && c <= '8')
                {
                    numbered++;
                }
            }
        }

        if (frontierUnknown == 0 || numbered == 0)
        {
            return 0;
        }

        if (frontierUnknown <= 128)
        {
            return 1;
        }

        if (frontierUnknown <= 2000)
        {
            return 2;
        }

        return 3;
    }

    private static void RunLocalPropagation(char[][] board, int h, int w, HashSet<Pos> safe, HashSet<Pos> mines)
    {
        bool changed;
        do
        {
            changed = false;
            for (int y = 0; y < h; y++)
            {
                for (int x = 0; x < w; x++)
                {
                    char c = board[y][x];
                    if (c < '0' || c > '8')
                    {
                        continue;
                    }

                    int target = c - '0';
                    Span<Pos> unknownBuf = stackalloc Pos[8];
                    int unknownCount = 0;
                    int mineCount = 0;

                    for (int dir = 0; dir < 8; dir++)
                    {
                        int nx = x + Dx[dir];
                        int ny = y + Dy[dir];
                        if ((uint)nx >= (uint)w || (uint)ny >= (uint)h)
                        {
                            continue;
                        }

                        var pos = new Pos(nx, ny);
                        char nc = board[ny][nx];
                        if (nc == '*' || mines.Contains(pos))
                        {
                            mineCount++;
                        }
                        else if (nc == ' ' && !safe.Contains(pos))
                        {
                            unknownBuf[unknownCount++] = pos;
                        }
                    }

                    if (unknownCount == 0)
                    {
                        continue;
                    }

                    if (mineCount == target)
                    {
                        for (int i = 0; i < unknownCount; i++)
                        {
                            changed |= safe.Add(unknownBuf[i]);
                        }
                    }
                    else if (mineCount + unknownCount == target)
                    {
                        for (int i = 0; i < unknownCount; i++)
                        {
                            changed |= mines.Add(unknownBuf[i]);
                        }
                    }
                }
            }
        } while (changed);
    }

    private static List<FrontierComponent> BuildFrontierComponents(
        char[][] board,
        int h,
        int w,
        HashSet<Pos> knownSafe,
        HashSet<Pos> knownMines,
        int complexity)
    {
        var varIndexByPos = new Dictionary<Pos, int>();
        var constraints = new List<Constraint>();
        var cellToConstraints = new Dictionary<int, List<int>>();

        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                char c = board[y][x];
                if (c < '0' || c > '8')
                {
                    continue;
                }

                int required = c - '0';
                int knownMineCount = 0;
                var vars = new List<int>(8);

                for (int dir = 0; dir < 8; dir++)
                {
                    int nx = x + Dx[dir];
                    int ny = y + Dy[dir];
                    if ((uint)nx >= (uint)w || (uint)ny >= (uint)h)
                    {
                        continue;
                    }

                    var pos = new Pos(nx, ny);
                    char nc = board[ny][nx];
                    if (nc == '*' || knownMines.Contains(pos))
                    {
                        knownMineCount++;
                    }
                    else if (nc == ' ' && !knownSafe.Contains(pos))
                    {
                        if (!varIndexByPos.TryGetValue(pos, out int vid))
                        {
                            vid = varIndexByPos.Count;
                            varIndexByPos[pos] = vid;
                        }

                        vars.Add(vid);
                    }
                }

                int remaining = required - knownMineCount;
                if (vars.Count == 0 || remaining < 0 || remaining > vars.Count)
                {
                    continue;
                }

                vars.Sort();
                var compact = new List<int>(vars.Count);
                int prev = -1;
                foreach (int v in vars)
                {
                    if (v != prev)
                    {
                        compact.Add(v);
                        prev = v;
                    }
                }

                var constraint = new Constraint { Required = remaining, Variables = compact };
                int cid = constraints.Count;
                constraints.Add(constraint);
                foreach (int v in compact)
                {
                    if (!cellToConstraints.TryGetValue(v, out var list))
                    {
                        list = new List<int>();
                        cellToConstraints[v] = list;
                    }

                    list.Add(cid);
                }
            }
        }

        var posByIndex = new Pos[varIndexByPos.Count];
        foreach (var kvp in varIndexByPos)
        {
            posByIndex[kvp.Value] = kvp.Key;
        }

        var components = new List<FrontierComponent>();
        var visitedVar = new bool[posByIndex.Length];
        var visitedConstraint = new bool[constraints.Count];
        var q = new Queue<int>();

        for (int start = 0; start < posByIndex.Length; start++)
        {
            if (visitedVar[start])
            {
                continue;
            }

            var varIds = new List<int>();
            var constraintIds = new List<int>();
            visitedVar[start] = true;
            q.Enqueue(start);

            while (q.Count > 0)
            {
                int v = q.Dequeue();
                varIds.Add(v);
                if (!cellToConstraints.TryGetValue(v, out var linked))
                {
                    continue;
                }

                foreach (int cid in linked)
                {
                    if (!visitedConstraint[cid])
                    {
                        visitedConstraint[cid] = true;
                        constraintIds.Add(cid);
                    }

                    foreach (int nextV in constraints[cid].Variables)
                    {
                        if (!visitedVar[nextV])
                        {
                            visitedVar[nextV] = true;
                            q.Enqueue(nextV);
                        }
                    }
                }
            }

            if (varIds.Count == 0)
            {
                continue;
            }

            if (complexity >= 3 && varIds.Count > 256)
            {
                continue;
            }

            if (complexity == 2 && varIds.Count > 512)
            {
                continue;
            }

            var remap = new Dictionary<int, int>(varIds.Count);
            for (int i = 0; i < varIds.Count; i++)
            {
                remap[varIds[i]] = i;
            }

            var component = new FrontierComponent();
            foreach (int vid in varIds)
            {
                component.Variables.Add(posByIndex[vid]);
            }

            foreach (int cid in constraintIds)
            {
                var old = constraints[cid];
                var nc = new Constraint { Required = old.Required };
                foreach (int oldVid in old.Variables)
                {
                    if (remap.TryGetValue(oldVid, out int newVid))
                    {
                        nc.Variables.Add(newVid);
                    }
                }

                component.Constraints.Add(nc);
            }

            components.Add(component);
        }

        return components;
    }

    private static void AnalyzeComponent(FrontierComponent component, HashSet<Pos> safe, HashSet<Pos> mines, int complexity)
    {
        SimplifyConstraints(component, safe, mines);

        int n = component.Variables.Count;
        if (n == 0)
        {
            return;
        }

        if (complexity <= 1 && n <= 26)
        {
            SolveExactly(component, safe, mines);
            return;
        }

        if (n <= 96)
        {
            SolveByBranching(component, safe, mines, 200000);
            return;
        }

        SolveByConstraintDominance(component, safe, mines);
    }

    private static void SimplifyConstraints(FrontierComponent component, HashSet<Pos> safe, HashSet<Pos> mines)
    {
        bool progress;
        do
        {
            progress = false;
            var nextConstraints = new List<Constraint>(component.Constraints.Count);
            foreach (var source in component.Constraints)
            {
                int required = source.Required;
                var nextVars = new List<int>(source.Variables.Count);
                foreach (int vid in source.Variables)
                {
                    var pos = component.Variables[vid];
                    if (safe.Contains(pos))
                    {
                        progress = true;
                        continue;
                    }

                    if (mines.Contains(pos))
                    {
                        required--;
                        progress = true;
                        continue;
                    }

                    nextVars.Add(vid);
                }

                if (required < 0 || required > nextVars.Count)
                {
                    continue;
                }

                if (nextVars.Count == 0)
                {
                    continue;
                }

                if (required == 0)
                {
                    foreach (int vid in nextVars)
                    {
                        progress |= safe.Add(component.Variables[vid]);
                    }
                }
                else if (required == nextVars.Count)
                {
                    foreach (int vid in nextVars)
                    {
                        progress |= mines.Add(component.Variables[vid]);
                    }
                }
                else
                {
                    nextConstraints.Add(new Constraint { Required = required, Variables = nextVars });
                }
            }

            component.Constraints = nextConstraints;
        } while (progress);
    }

    private static void SolveExactly(FrontierComponent component, HashSet<Pos> safe, HashSet<Pos> mines)
    {
        int n = component.Variables.Count;
        var values = new int[n];
        Array.Fill(values, -1);
        var mineCountPerVar = new int[n];
        long totalSolutions = 0;

        var memberships = BuildMemberships(component, n);
        Search(0);

        if (totalSolutions == 0)
        {
            return;
        }

        for (int i = 0; i < n; i++)
        {
            if (mineCountPerVar[i] == 0)
            {
                safe.Add(component.Variables[i]);
            }
            else if (mineCountPerVar[i] == totalSolutions)
            {
                mines.Add(component.Variables[i]);
            }
        }

        void Search(int idx)
        {
            if (idx == n)
            {
                if (AllConstraintsSatisfied(component, values))
                {
                    totalSolutions++;
                    for (int i = 0; i < n; i++)
                    {
                        if (values[i] == 1)
                        {
                            mineCountPerVar[i]++;
                        }
                    }
                }
                return;
            }

            foreach (int val in new[] { 0, 1 })
            {
                values[idx] = val;
                if (PartialValid(component, memberships, values, idx))
                {
                    Search(idx + 1);
                }
            }

            values[idx] = -1;
        }
    }

    private static void SolveByBranching(FrontierComponent component, HashSet<Pos> safe, HashSet<Pos> mines, int nodeBudget)
    {
        int n = component.Variables.Count;
        var values = new int[n];
        Array.Fill(values, -1);
        var canBeMine = new bool[n];
        var canBeSafe = new bool[n];
        var memberships = BuildMemberships(component, n);
        int nodes = 0;

        Dfs(ChooseNext(values));

        for (int i = 0; i < n; i++)
        {
            if (canBeSafe[i] && !canBeMine[i])
            {
                safe.Add(component.Variables[i]);
            }
            else if (canBeMine[i] && !canBeSafe[i])
            {
                mines.Add(component.Variables[i]);
            }
        }

        int ChooseNext(int[] current)
        {
            int bestVar = -1;
            int bestScore = -1;
            for (int i = 0; i < n; i++)
            {
                if (current[i] != -1)
                {
                    continue;
                }

                int score = memberships[i].Count;
                if (score > bestScore)
                {
                    bestScore = score;
                    bestVar = i;
                }
            }
            return bestVar;
        }

        void Dfs(int var)
        {
            if (nodes >= nodeBudget)
            {
                return;
            }

            if (var == -1)
            {
                if (AllConstraintsSatisfied(component, values))
                {
                    for (int i = 0; i < n; i++)
                    {
                        if (values[i] == 1)
                        {
                            canBeMine[i] = true;
                        }
                        else
                        {
                            canBeSafe[i] = true;
                        }
                    }
                }
                return;
            }

            nodes++;
            values[var] = 0;
            if (PartialValid(component, memberships, values, var))
            {
                Dfs(ChooseNext(values));
            }

            values[var] = 1;
            if (PartialValid(component, memberships, values, var))
            {
                Dfs(ChooseNext(values));
            }

            values[var] = -1;
        }
    }

    private static void SolveByConstraintDominance(FrontierComponent component, HashSet<Pos> safe, HashSet<Pos> mines)
    {
        SimplifyConstraints(component, safe, mines);
        var constraints = component.Constraints;
        for (int i = 0; i < constraints.Count; i++)
        {
            var a = constraints[i];
            if (a.Variables.Count == 0)
            {
                continue;
            }

            var setA = new HashSet<int>(a.Variables);
            for (int j = 0; j < constraints.Count; j++)
            {
                if (i == j)
                {
                    continue;
                }

                var b = constraints[j];
                if (b.Variables.Count == 0 || a.Variables.Count > b.Variables.Count)
                {
                    continue;
                }

                bool subset = true;
                foreach (int v in a.Variables)
                {
                    if (!b.Variables.Contains(v))
                    {
                        subset = false;
                        break;
                    }
                }

                if (!subset)
                {
                    continue;
                }

                int extraRequired = b.Required - a.Required;
                int extraCount = b.Variables.Count - a.Variables.Count;
                if (extraRequired == 0)
                {
                    foreach (int v in b.Variables)
                    {
                        if (!setA.Contains(v))
                        {
                            safe.Add(component.Variables[v]);
                        }
                    }
                }
                else if (extraRequired == extraCount)
                {
                    foreach (int v in b.Variables)
                    {
                        if (!setA.Contains(v))
                        {
                            mines.Add(component.Variables[v]);
                        }
                    }
                }
            }
        }
    }

    private static List<int>[] BuildMemberships(FrontierComponent component, int n)
    {
        var memberships = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            memberships[i] = new List<int>();
        }

        for (int cid = 0; cid < component.Constraints.Count; cid++)
        {
            foreach (int vid in component.Constraints[cid].Variables)
            {
                memberships[vid].Add(cid);
            }
        }

        return memberships;
    }

    private static bool PartialValid(FrontierComponent component, List<int>[] memberships, int[] values, int changedVar)
    {
        foreach (int cid in memberships[changedVar])
        {
            var c = component.Constraints[cid];
            int assignedMines = 0;
            int unassigned = 0;
            foreach (int vid in c.Variables)
            {
                int value = values[vid];
                if (value == 1)
                {
                    assignedMines++;
                }
                else if (value == -1)
                {
                    unassigned++;
                }
            }

            if (assignedMines > c.Required)
            {
                return false;
            }

            if (assignedMines + unassigned < c.Required)
            {
                return false;
            }
        }

        return true;
    }

    private static bool AllConstraintsSatisfied(FrontierComponent component, int[] values)
    {
        foreach (var c in component.Constraints)
        {
            int mines = 0;
            foreach (int vid in c.Variables)
            {
                if (values[vid] == 1)
                {
                    mines++;
                }
            }

            if (mines != c.Required)
            {
                return false;
            }
        }

        return true;
    }
}
