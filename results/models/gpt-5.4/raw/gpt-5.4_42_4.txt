OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBO "UBO"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_uint_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_gl_PerVertex_in_uint_2 = OpTypePointer Input %arr_gl_PerVertex_in_uint_2
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_uint_2 Input
%arr_v3float_uint_2 = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3float_uint_2 = OpTypePointer Input %arr_v3float_uint_2
%worldPos = OpVariable %ptr_Input_arr_v3float_uint_2 Input
%normal = OpVariable %ptr_Input_arr_v3float_uint_2 Input
%color = OpVariable %ptr_Input_arr_v3float_uint_2 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float

%main = OpFunction %void None %3
%5 = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%c1 = OpLoad %v3float %c1_ptr
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %1 Normalize %perp_raw
%off0 = OpVectorTimesScalar %v2float %perp %float_0_04
%off1 = OpVectorTimesScalar %v2float %perp %float_0_005
%p0_plus_xy = OpFAdd %v2float %p0_xy %off0
%p0_minus_xy = OpFSub %v2float %p0_xy %off0
%p1_plus_xy = OpFAdd %v2float %p1_xy %off1
%p1_minus_xy = OpFSub %v2float %p1_xy %off1
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3
%p0_plus_x = OpCompositeExtract %float %p0_plus_xy 0
%p0_plus_y = OpCompositeExtract %float %p0_plus_xy 1
%p0_minus_x = OpCompositeExtract %float %p0_minus_xy 0
%p0_minus_y = OpCompositeExtract %float %p0_minus_xy 1
%p1_plus_x = OpCompositeExtract %float %p1_plus_xy 0
%p1_plus_y = OpCompositeExtract %float %p1_plus_xy 1
%p1_minus_x = OpCompositeExtract %float %p1_minus_xy 0
%p1_minus_y = OpCompositeExtract %float %p1_minus_xy 1
%vtx0 = OpCompositeConstruct %v4float %p0_plus_x %p0_plus_y %p0_z %p0_w
%vtx1 = OpCompositeConstruct %v4float %p0_minus_x %p0_minus_y %p0_z %p0_w
%vtx2 = OpCompositeConstruct %v4float %p1_plus_x %p1_plus_y %p1_z %p1_w
%vtx3 = OpCompositeConstruct %v4float %p1_minus_x %p1_minus_y %p1_z %p1_w
%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %outColor %c0
OpStore %out_pos_ptr %vtx0
OpEmitVertex
OpStore %outColor %c0
OpStore %out_pos_ptr %vtx1
OpEmitVertex
OpStore %outColor %c1
OpStore %out_pos_ptr %vtx2
OpEmitVertex
OpStore %outColor %c1
OpStore %out_pos_ptr %vtx3
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
