use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const REF_VERTICES: [[f64; 3]; 4] = [
    [1.0, 0.0, -0.707],
    [-1.0, 0.0, -0.707],
    [0.0, 1.0, 0.707],
    [0.0, -1.0, 0.707],
];

const MAX_SCALE: f64 = 10.0;
const EPS: f64 = 1.0e-9;

#[derive(Clone, Copy, Debug, Default)]
struct Vec2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug, Default)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy, Debug)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Debug)]
struct Placement {
    center: Vec3,
    rotation: Quaternion,
    scale: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn cross(self, other: Self) -> f64 {
        self.x * other.y - self.y * other.x
    }

    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn length(self) -> f64 {
        self.dot(self).sqrt()
    }
}

impl std::ops::Add for Vec2 {
    type Output = Self;

    fn add(self, rhs: Self) -> Self::Output {
        Self::new(self.x + rhs.x, self.y + rhs.y)
    }
}

impl std::ops::Sub for Vec2 {
    type Output = Self;

    fn sub(self, rhs: Self) -> Self::Output {
        Self::new(self.x - rhs.x, self.y - rhs.y)
    }
}

impl std::ops::Mul<f64> for Vec2 {
    type Output = Self;

    fn mul(self, rhs: f64) -> Self::Output {
        Self::new(self.x * rhs, self.y * rhs)
    }
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }

    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(self, other: Self) -> Self {
        Self::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }

    fn length(self) -> f64 {
        self.dot(self).sqrt()
    }

    fn normalized(self) -> Self {
        let len = self.length();
        if len <= EPS {
            Self::new(0.0, 0.0, 1.0)
        } else {
            self / len
        }
    }
}

impl std::ops::Add for Vec3 {
    type Output = Self;

    fn add(self, rhs: Self) -> Self::Output {
        Self::new(self.x + rhs.x, self.y + rhs.y, self.z + rhs.z)
    }
}

impl std::ops::Sub for Vec3 {
    type Output = Self;

    fn sub(self, rhs: Self) -> Self::Output {
        Self::new(self.x - rhs.x, self.y - rhs.y, self.z - rhs.z)
    }
}

impl std::ops::Mul<f64> for Vec3 {
    type Output = Self;

    fn mul(self, rhs: f64) -> Self::Output {
        Self::new(self.x * rhs, self.y * rhs, self.z * rhs)
    }
}

impl std::ops::Div<f64> for Vec3 {
    type Output = Self;

    fn div(self, rhs: f64) -> Self::Output {
        Self::new(self.x / rhs, self.y / rhs, self.z / rhs)
    }
}

impl Quaternion {
    fn identity() -> Self {
        Self {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        }
    }

    fn normalized(self) -> Self {
        let len = (self.w * self.w + self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if len <= EPS {
            Self::identity()
        } else {
            Self {
                w: self.w / len,
                x: self.x / len,
                y: self.y / len,
                z: self.z / len,
            }
        }
    }

    fn from_axis_angle(axis: Vec3, angle: f64) -> Self {
        let half = angle * 0.5;
        let s = half.sin();
        let unit = axis.normalized();
        Self {
            w: half.cos(),
            x: unit.x * s,
            y: unit.y * s,
            z: unit.z * s,
        }
        .normalized()
    }

    fn from_basis(x_axis: Vec3, y_axis: Vec3, z_axis: Vec3) -> Self {
        let m00 = x_axis.x;
        let m01 = y_axis.x;
        let m02 = z_axis.x;
        let m10 = x_axis.y;
        let m11 = y_axis.y;
        let m12 = z_axis.y;
        let m20 = x_axis.z;
        let m21 = y_axis.z;
        let m22 = z_axis.z;

        let trace = m00 + m11 + m22;
        let q = if trace > 0.0 {
            let s = (trace + 1.0).sqrt() * 2.0;
            Quaternion {
                w: 0.25 * s,
                x: (m21 - m12) / s,
                y: (m02 - m20) / s,
                z: (m10 - m01) / s,
            }
        } else if m00 > m11 && m00 > m22 {
            let s = (1.0 + m00 - m11 - m22).sqrt() * 2.0;
            Quaternion {
                w: (m21 - m12) / s,
                x: 0.25 * s,
                y: (m01 + m10) / s,
                z: (m02 + m20) / s,
            }
        } else if m11 > m22 {
            let s = (1.0 + m11 - m00 - m22).sqrt() * 2.0;
            Quaternion {
                w: (m02 - m20) / s,
                x: (m01 + m10) / s,
                y: 0.25 * s,
                z: (m12 + m21) / s,
            }
        } else {
            let s = (1.0 + m22 - m00 - m11).sqrt() * 2.0;
            Quaternion {
                w: (m10 - m01) / s,
                x: (m02 + m20) / s,
                y: (m12 + m21) / s,
                z: 0.25 * s,
            }
        };
        q.normalized()
    }

    fn mul(self, rhs: Self) -> Self {
        Self {
            w: self.w * rhs.w - self.x * rhs.x - self.y * rhs.y - self.z * rhs.z,
            x: self.w * rhs.x + self.x * rhs.w + self.y * rhs.z - self.z * rhs.y,
            y: self.w * rhs.y - self.x * rhs.z + self.y * rhs.w + self.z * rhs.x,
            z: self.w * rhs.z + self.x * rhs.y - self.y * rhs.x + self.z * rhs.w,
        }
        .normalized()
    }
}

fn polygon_area(poly: &[Vec2]) -> f64 {
    let mut sum = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        sum += a.cross(b);
    }
    0.5 * sum
}

fn point_in_triangle(p: Vec2, a: Vec2, b: Vec2, c: Vec2) -> bool {
    let ab = (b - a).cross(p - a);
    let bc = (c - b).cross(p - b);
    let ca = (a - c).cross(p - c);
    (ab >= -EPS && bc >= -EPS && ca >= -EPS) || (ab <= EPS && bc <= EPS && ca <= EPS)
}

fn is_ear(prev: usize, curr: usize, next: usize, indices: &[usize], poly: &[Vec2]) -> bool {
    let a = poly[prev];
    let b = poly[curr];
    let c = poly[next];
    if (b - a).cross(c - b) <= EPS {
        return false;
    }
    for &idx in indices {
        if idx == prev || idx == curr || idx == next {
            continue;
        }
        if point_in_triangle(poly[idx], a, b, c) {
            return false;
        }
    }
    true
}

fn triangulate(poly: &[Vec2]) -> Vec<[usize; 3]> {
    let n = poly.len();
    let mut result = Vec::new();
    if n < 3 {
        return result;
    }
    if n == 3 {
        result.push([0, 1, 2]);
        return result;
    }

    let mut indices: Vec<usize> = (0..n).collect();
    let mut guard = 0usize;
    while indices.len() > 3 && guard < n * n {
        let m = indices.len();
        let mut ear_found = false;
        for i in 0..m {
            let prev = indices[(i + m - 1) % m];
            let curr = indices[i];
            let next = indices[(i + 1) % m];
            if is_ear(prev, curr, next, &indices, poly) {
                result.push([prev, curr, next]);
                indices.remove(i);
                ear_found = true;
                break;
            }
        }
        if !ear_found {
            break;
        }
        guard += 1;
    }
    if indices.len() == 3 {
        result.push([indices[0], indices[1], indices[2]]);
    }
    result
}

fn convex_hull(points: &[Vec2]) -> Vec<Vec2> {
    let mut pts = points.to_vec();
    pts.sort_by(|a, b| {
        let by_x = a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal);
        if by_x == Ordering::Equal {
            a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal)
        } else {
            by_x
        }
    });
    pts.dedup_by(|a, b| (a.x - b.x).abs() <= EPS && (a.y - b.y).abs() <= EPS);
    if pts.len() <= 1 {
        return pts;
    }

    let mut lower: Vec<Vec2> = Vec::new();
    for &p in &pts {
        while lower.len() >= 2 {
            let l = lower.len();
            let cross = (lower[l - 1] - lower[l - 2]).cross(p - lower[l - 1]);
            if cross <= EPS {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(p);
    }

    let mut upper: Vec<Vec2> = Vec::new();
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let l = upper.len();
            let cross = (upper[l - 1] - upper[l - 2]).cross(p - upper[l - 1]);
            if cross <= EPS {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn regular_ngon(center: Vec2, radius: f64, sides: usize) -> Vec<Vec2> {
    let mut out = Vec::with_capacity(sides);
    for i in 0..sides {
        let angle = 2.0 * PI * (i as f64) / (sides as f64);
        out.push(Vec2::new(
            center.x + radius * angle.cos(),
            center.y + radius * angle.sin(),
        ));
    }
    out
}

fn adaptive_cover(poly: &[Vec2]) -> Vec<[usize; 3]> {
    let area = polygon_area(poly).abs();
    if poly.len() <= 3 {
        return vec![[0, 1, 2]];
    }
    if poly.len() <= 64 {
        return triangulate(poly);
    }

    let hull = convex_hull(poly);
    let hull_area = polygon_area(&hull).abs();
    if hull.len() >= 3 && hull_area <= area * 1.05 {
        return triangulate(&hull);
    }
    if poly.len() > 512 {
        let mut center = Vec2::new(0.0, 0.0);
        for &p in poly {
            center = center + p;
        }
        center = center * (1.0 / poly.len() as f64);
        let mut radius = 0.0;
        for &p in poly {
            let d = (p - center).length();
            if d > radius {
                radius = d;
            }
        }
        let proxy = regular_ngon(center, radius, 32);
        return triangulate(&proxy);
    }
    triangulate(poly)
}

fn triangle_centroid(a: Vec2, b: Vec2, c: Vec2) -> Vec2 {
    Vec2::new((a.x + b.x + c.x) / 3.0, (a.y + b.y + c.y) / 3.0)
}

fn make_basis(shadow_dir: Vec3) -> (Vec3, Vec3, Vec3) {
    let z_axis = shadow_dir.normalized();
    let helper = if z_axis.z.abs() < 0.9 {
        Vec3::new(0.0, 0.0, 1.0)
    } else {
        Vec3::new(1.0, 0.0, 0.0)
    };
    let x_axis = helper.cross(z_axis).normalized();
    let y_axis = z_axis.cross(x_axis).normalized();
    (x_axis, y_axis, z_axis)
}

fn reference_shadow_extents() -> (f64, f64) {
    let min_x = REF_VERTICES.iter().map(|v| v[0]).fold(f64::INFINITY, f64::min);
    let max_x = REF_VERTICES
        .iter()
        .map(|v| v[0])
        .fold(f64::NEG_INFINITY, f64::max);
    let min_y = REF_VERTICES.iter().map(|v| v[1]).fold(f64::INFINITY, f64::min);
    let max_y = REF_VERTICES
        .iter()
        .map(|v| v[1])
        .fold(f64::NEG_INFINITY, f64::max);
    (max_x - min_x, max_y - min_y)
}

fn placement_for_triangle(
    a: Vec2,
    b: Vec2,
    c: Vec2,
    shadow_dir: Vec3,
    layer: usize,
    layers_per_triangle: usize,
) -> Vec<Placement> {
    let centroid = triangle_centroid(a, b, c);
    let edge1 = b - a;
    let edge2 = c - a;
    let ux = edge1.normalized();
    let proj = ux * edge2.dot(ux);
    let mut vy = edge2 - proj;
    if vy.length() <= EPS {
        vy = Vec2::new(-ux.y, ux.x);
    }
    let uy = vy.normalized();

    let p0 = Vec2::new(0.0, 0.0);
    let p1 = Vec2::new(edge1.length(), 0.0);
    let p2 = Vec2::new(edge2.dot(ux), edge2.dot(uy));
    let min_x = p0.x.min(p1.x.min(p2.x));
    let max_x = p0.x.max(p1.x.max(p2.x));
    let min_y = p0.y.min(p1.y.min(p2.y));
    let max_y = p0.y.max(p1.y.max(p2.y));
    let width = (max_x - min_x).max(1.0e-3);
    let height = (max_y - min_y).max(1.0e-3);

    let (ref_w, ref_h) = reference_shadow_extents();
    let mut scale = (width / ref_w).max(height / ref_h);
    if scale > MAX_SCALE {
        scale = MAX_SCALE;
    }

    let pieces_x = ((width / (ref_w * MAX_SCALE)).ceil() as usize).max(1);
    let pieces_y = ((height / (ref_h * MAX_SCALE)).ceil() as usize).max(1);
    let split_count = pieces_x.max(pieces_y);

    let (bx, by, bz) = make_basis(shadow_dir);
    let base_rot = Quaternion::from_basis(bx, by, bz);
    let twist = Quaternion::from_axis_angle(bz, ((layer % 11) as f64) * 0.173);
    let rotation = base_rot.mul(twist);

    let tri_area = ((b - a).cross(c - a)).abs() * 0.5;
    let lift_step = 30.0 + 2.0 * tri_area.sqrt();
    let base_lift = lift_step * (layer as f64 + 1.0);

    let mut placements = Vec::new();
    for sy in 0..split_count {
        for sx in 0..split_count {
            let fx = (sx as f64 + 0.5) / (split_count as f64) - 0.5;
            let fy = (sy as f64 + 0.5) / (split_count as f64) - 0.5;
            let plane_offset =
                bx * (fx * width * 0.35) + by * (fy * height * 0.35) + bz * base_lift;
            let center = Vec3::new(centroid.x, centroid.y, 0.0) + plane_offset;
            placements.push(Placement {
                center,
                rotation,
                scale: scale / split_count as f64,
            });
        }
    }

    if placements.is_empty() {
        let center = Vec3::new(centroid.x, centroid.y, 0.0) + bz * base_lift;
        placements.push(Placement {
            center,
            rotation,
            scale,
        });
    }

    if layers_per_triangle > 1 {
        for extra in 1..layers_per_triangle {
            let shift = bz * (extra as f64 * lift_step * 0.25);
            let clone_base = placements.clone();
            for mut p in clone_base {
                p.center = p.center + shift;
                p.scale *= 0.92;
                placements.push(p);
            }
        }
    }

    placements
}

fn normalized_shadow_dir(sun: Vec3) -> Vec3 {
    let horizontal = Vec3::new(sun.x, sun.y, 0.0);
    if horizontal.length() <= EPS {
        Vec3::new(0.0, 0.0, 1.0)
    } else {
        (horizontal * -1.0 + Vec3::new(0.0, 0.0, 1.0)).normalized()
    }
}

fn solve(poly: &[Vec2], sun: Vec3) -> Vec<Placement> {
    let mut work = poly.to_vec();
    if polygon_area(&work) < 0.0 {
        work.reverse();
    }

    let tris = adaptive_cover(&work);
    let shadow_dir = normalized_shadow_dir(sun);
    let layers_per_triangle = if work.len() > 256 { 1 } else { 1 };

    let mut placements = Vec::new();
    for (i, tri) in tris.iter().enumerate() {
        let a = work[tri[0]];
        let b = work[tri[1]];
        let c = work[tri[2]];
        placements.extend(placement_for_triangle(
            a,
            b,
            c,
            shadow_dir,
            i,
            layers_per_triangle,
        ));
    }
    placements
}

fn parse_input() -> Option<(Vec<Vec2>, Vec3)> {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).ok()?;
    let mut it = input.split_whitespace();
    let nv: usize = it.next()?.parse().ok()?;
    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next()?.parse().ok()?;
        let y: f64 = it.next()?.parse().ok()?;
        poly.push(Vec2::new(x, y));
    }
    let sx: f64 = it.next()?.parse().ok()?;
    let sy: f64 = it.next()?.parse().ok()?;
    let sz: f64 = it.next()?.parse().ok()?;
    Some((poly, Vec3::new(sx, sy, sz)))
}

fn main() {
    let Some((poly, sun)) = parse_input() else {
        println!("0");
        return;
    };
    if poly.len() < 3 {
        println!("0");
        return;
    }

    let placements = solve(&poly, sun);
    println!("{}", placements.len());
    for p in placements {
        println!(
            "{:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9}",
            p.center.x,
            p.center.y,
            p.center.z,
            p.rotation.w,
            p.rotation.x,
            p.rotation.y,
            p.rotation.z,
            p.scale
        );
    }
}
