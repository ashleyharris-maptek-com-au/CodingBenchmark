import math
import sys


def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = int(next(it))
    depot = (float(next(it)), float(next(it)))
    customers = []
    for idx in range(n):
        x = float(next(it))
        y = float(next(it))
        demand = int(next(it))
        customers.append((x, y, demand, idx))
    return n, k, capacity, depot, customers


def euclidean(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_distance(route, points, depot_idx=0):
    if not route:
        return 0.0
    total = points[depot_idx][route[0]]
    for i in range(len(route) - 1):
        total += points[route[i]][route[i + 1]]
    total += points[route[-1]][depot_idx]
    return total


def two_opt(route, points):
    if len(route) < 4:
        return route
    best = route[:]
    best_cost = route_distance(best, points)
    improved = True
    while improved:
        improved = False
        for i in range(len(best) - 2):
            for j in range(i + 2, len(best)):
                if j - i == 1:
                    continue
                candidate = best[:i + 1] + best[i + 1:j + 1][::-1] + best[j + 1:]
                cand_cost = route_distance(candidate, points)
                if cand_cost + 1e-9 < best_cost:
                    best = candidate
                    best_cost = cand_cost
                    improved = True
                    break
            if improved:
                break
    return best


def build_seed_routes(n, k, capacity, depot, customers, points):
    unassigned = set(range(1, n + 1))
    demands = [0] + [c[2] for c in customers]
    angles = []
    for node in range(1, n + 1):
        x, y, _, _ = customers[node - 1]
        angles.append((math.atan2(y - depot[1], x - depot[0]), node))
    angles.sort()

    ordered = [node for _, node in angles]
    routes = []
    idx = 0
    for remaining_routes in range(k):
        if not unassigned:
            break
        route = []
        load = 0
        left_routes = k - remaining_routes
        left_customers = len(unassigned)
        while idx < len(ordered) and ordered[idx] not in unassigned:
            idx += 1
        if idx >= len(ordered):
            break
        start = ordered[idx]
        route.append(start)
        unassigned.remove(start)
        load += demands[start]

        progress = True
        while progress and unassigned:
            progress = False
            current = route[-1]
            candidates = []
            for node in list(unassigned):
                if load + demands[node] > capacity:
                    continue
                if left_routes > 1 and left_customers - 1 < left_routes - 1:
                    continue
                score = points[current][node] + 0.15 * points[0][node]
                candidates.append((score, node))
            if not candidates:
                break
            candidates.sort()
            next_node = candidates[0][1]
            route.append(next_node)
            unassigned.remove(next_node)
            load += demands[next_node]
            left_customers -= 1
            progress = True
        routes.append(route)

    while unassigned:
        node = unassigned.pop()
        best_route = None
        best_pos = None
        best_increase = float('inf')
        for r_idx, route in enumerate(routes):
            load = sum(demands[v] for v in route)
            if load + demands[node] > capacity:
                continue
            for pos in range(len(route) + 1):
                prev_node = 0 if pos == 0 else route[pos - 1]
                next_node = 0 if pos == len(route) else route[pos]
                increase = points[prev_node][node] + points[node][next_node] - points[prev_node][next_node]
                if increase < best_increase:
                    best_increase = increase
                    best_route = r_idx
                    best_pos = pos
        if best_route is None:
            return None
        routes[best_route].insert(best_pos, node)

    return routes


def repair_or_fallback(n, k, capacity, depot, customers, points):
    demands = [c[2] for c in customers]
    if any(d > capacity for d in demands):
        return None

    nodes = list(range(1, n + 1))
    nodes.sort(key=lambda node: points[0][node], reverse=True)
    routes = []
    current = []
    load = 0
    for node in nodes:
        demand = demands[node - 1]
        if load + demand <= capacity:
            current.append(node)
            load += demand
        else:
            routes.append(current)
            current = [node]
            load = demand
    if current:
        routes.append(current)
    if len(routes) > k:
        return None
    return routes


def solve():
    parsed = read_input()
    if parsed is None:
        return
    n, k, capacity, depot, customers = parsed

    coords = [depot] + [(x, y) for x, y, _, _ in customers]
    points = [[0.0] * (n + 1) for _ in range(n + 1)]
    for i in range(n + 1):
        for j in range(i + 1, n + 1):
            d = euclidean(coords[i], coords[j])
            points[i][j] = d
            points[j][i] = d

    routes = build_seed_routes(n, k, capacity, depot, customers, points)
    if routes is None:
        routes = repair_or_fallback(n, k, capacity, depot, customers, points)
    if routes is None:
        print('IMPOSSIBLE')
        print(0)
        return

    routes = [two_opt(route, points) for route in routes if route]
    total = sum(route_distance(route, points) for route in routes)

    print(f"{total:.6f}")
    print(len(routes))
    for route in routes:
        print(' '.join(str(node - 1) for node in route))


if __name__ == '__main__':
    solve()
