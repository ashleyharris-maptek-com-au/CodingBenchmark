using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        var edges = new List<(int From, int To)>(m);
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            edges.Add((u, v));
        }

        var solver = new DirectedFeedbackVertexSetSolver(n, edges);
        var answer = solver.Solve();

        var sw = new StreamWriter(Console.OpenStandardOutput());
        sw.WriteLine(answer.Count);
        for (int i = 0; i < answer.Count; i++)
        {
            if (i > 0) sw.Write(' ');
            sw.Write(answer[i]);
        }
        sw.WriteLine();
        sw.Flush();
    }
}

internal sealed class DirectedFeedbackVertexSetSolver
{
    private readonly int _n;
    private readonly List<int>[] _out;
    private readonly List<int>[] _in;

    public DirectedFeedbackVertexSetSolver(int n, List<(int From, int To)> edges)
    {
        _n = n;
        _out = new List<int>[n];
        _in = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            _out[i] = new List<int>();
            _in[i] = new List<int>();
        }

        foreach (var (from, to) in edges)
        {
            _out[from].Add(to);
            _in[to].Add(from);
        }
    }

    public List<int> Solve()
    {
        var alive = new bool[_n];
        Array.Fill(alive, true);
        PeelTrivial(alive);

        var answer = new List<int>();
        SolveRecursive(alive, answer);
        answer.Sort();
        return answer;
    }

    private void SolveRecursive(bool[] alive, List<int> removed)
    {
        PeelTrivial(alive);
        var sccs = GetNonTrivialSccs(alive);
        if (sccs.Count == 0) return;

        foreach (var comp in sccs)
        {
            if (comp.Count <= 22)
            {
                SolveSmallComponentExactly(alive, comp, removed);
            }
            else
            {
                SolveLargeComponentBranchAndBound(alive, comp, removed);
            }
        }
    }

    private void SolveSmallComponentExactly(bool[] globalAlive, List<int> comp, List<int> removed)
    {
        int k = comp.Count;
        var index = new Dictionary<int, int>(k);
        for (int i = 0; i < k; i++) index[comp[i]] = i;

        ulong[] edgeMask = new ulong[k];
        ulong[] revMask = new ulong[k];
        for (int i = 0; i < k; i++)
        {
            int v = comp[i];
            foreach (int to in _out[v])
            {
                if (index.TryGetValue(to, out int j))
                {
                    edgeMask[i] |= 1UL << j;
                    revMask[j] |= 1UL << i;
                }
            }
        }

        int total = 1 << k;
        var acyclic = new bool[total];
        acyclic[0] = true;
        int bestMask = 0;
        int bestSize = 0;

        for (int mask = 1; mask < total; mask++)
        {
            int bit = TrailingZeroCount(mask);
            int prev = mask & (mask - 1);
            ulong prevMask = (ulong)prev;

            bool source = (edgeMask[bit] & prevMask) == 0;
            bool sink = (revMask[bit] & prevMask) == 0;
            if (acyclic[prev] && (source || sink))
            {
                acyclic[mask] = true;
                int size = PopCount(mask);
                if (size > bestSize)
                {
                    bestSize = size;
                    bestMask = mask;
                }
            }
        }

        for (int i = 0; i < k; i++)
        {
            if (((bestMask >> i) & 1) == 0)
            {
                int v = comp[i];
                if (globalAlive[v])
                {
                    globalAlive[v] = false;
                    removed.Add(v);
                }
            }
        }
    }

    private void SolveLargeComponentBranchAndBound(bool[] globalAlive, List<int> comp, List<int> removed)
    {
        var alive = new HashSet<int>(comp);
        var chosen = new List<int>();
        var best = new List<int>(comp);

        var greedy = GreedyFeedbackSet(alive);
        best = new List<int>(greedy);

        Branch(alive, chosen, ref best);

        foreach (int v in best)
        {
            if (globalAlive[v])
            {
                globalAlive[v] = false;
                removed.Add(v);
            }
        }
    }

    private void Branch(HashSet<int> alive, List<int> chosen, ref List<int> best)
    {
        PeelTrivial(alive);
        var cycle = FindCycle(alive);
        if (cycle.Count == 0)
        {
            if (chosen.Count < best.Count) best = new List<int>(chosen);
            return;
        }

        if (chosen.Count >= best.Count) return;

        int lb = LowerBoundByPacking(alive);
        if (chosen.Count + lb >= best.Count) return;

        cycle.Sort((a, b) => ScoreVertex(alive, b).CompareTo(ScoreVertex(alive, a)));
        int limit = Math.Min(3, cycle.Count);
        for (int i = 0; i < limit; i++)
        {
            int v = cycle[i];
            var nextAlive = new HashSet<int>(alive);
            nextAlive.Remove(v);
            chosen.Add(v);
            Branch(nextAlive, chosen, ref best);
            chosen.RemoveAt(chosen.Count - 1);
        }
    }

    private List<int> GreedyFeedbackSet(HashSet<int> alive)
    {
        var work = new HashSet<int>(alive);
        var removed = new List<int>();
        PeelTrivial(work);

        while (true)
        {
            var cycle = FindCycle(work);
            if (cycle.Count == 0) break;
            int bestV = cycle[0];
            long bestScore = ScoreVertex(work, bestV);
            for (int i = 1; i < cycle.Count; i++)
            {
                int v = cycle[i];
                long score = ScoreVertex(work, v);
                if (score > bestScore)
                {
                    bestScore = score;
                    bestV = v;
                }
            }

            work.Remove(bestV);
            removed.Add(bestV);
            PeelTrivial(work);
        }

        return removed;
    }

    private int LowerBoundByPacking(HashSet<int> alive)
    {
        var work = new HashSet<int>(alive);
        int count = 0;
        var blocked = new HashSet<int>();
        while (true)
        {
            var cycle = FindCycle(work, blocked);
            if (cycle.Count == 0) break;
            count++;
            foreach (int v in cycle) blocked.Add(v);
        }
        return count;
    }

    private long ScoreVertex(HashSet<int> alive, int v)
    {
        int indeg = 0;
        int outdeg = 0;
        foreach (int to in _out[v]) if (alive.Contains(to)) outdeg++;
        foreach (int from in _in[v]) if (alive.Contains(from)) indeg++;
        return (long)indeg * outdeg + indeg + outdeg;
    }

    private void PeelTrivial(bool[] alive)
    {
        var queue = new Queue<int>();
        var indeg = new int[_n];
        var outdeg = new int[_n];

        for (int v = 0; v < _n; v++)
        {
            if (!alive[v]) continue;
            foreach (int to in _out[v]) if (alive[to]) outdeg[v]++;
            foreach (int from in _in[v]) if (alive[from]) indeg[v]++;
        }

        for (int v = 0; v < _n; v++)
        {
            if (alive[v] && (indeg[v] == 0 || outdeg[v] == 0)) queue.Enqueue(v);
        }

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (!alive[v]) continue;
            if (indeg[v] != 0 && outdeg[v] != 0) continue;
            alive[v] = false;
            foreach (int to in _out[v])
            {
                if (!alive[to]) continue;
                indeg[to]--;
                if (indeg[to] == 0 || outdeg[to] == 0) queue.Enqueue(to);
            }
            foreach (int from in _in[v])
            {
                if (!alive[from]) continue;
                outdeg[from]--;
                if (indeg[from] == 0 || outdeg[from] == 0) queue.Enqueue(from);
            }
        }
    }

    private void PeelTrivial(HashSet<int> alive)
    {
        var queue = new Queue<int>();
        var indeg = new Dictionary<int, int>(alive.Count);
        var outdeg = new Dictionary<int, int>(alive.Count);

        foreach (int v in alive)
        {
            int inCount = 0;
            int outCount = 0;
            foreach (int to in _out[v]) if (alive.Contains(to)) outCount++;
            foreach (int from in _in[v]) if (alive.Contains(from)) inCount++;
            indeg[v] = inCount;
            outdeg[v] = outCount;
            if (inCount == 0 || outCount == 0) queue.Enqueue(v);
        }

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (!alive.Contains(v)) continue;
            if (indeg[v] != 0 && outdeg[v] != 0) continue;
            alive.Remove(v);
            foreach (int to in _out[v])
            {
                if (!alive.Contains(to)) continue;
                indeg[to]--;
                if (indeg[to] == 0 || outdeg[to] == 0) queue.Enqueue(to);
            }
            foreach (int from in _in[v])
            {
                if (!alive.Contains(from)) continue;
                outdeg[from]--;
                if (indeg[from] == 0 || outdeg[from] == 0) queue.Enqueue(from);
            }
        }
    }

    private List<List<int>> GetNonTrivialSccs(bool[] alive)
    {
        var order = new List<int>();
        var seen = new bool[_n];
        for (int v = 0; v < _n; v++)
        {
            if (alive[v] && !seen[v]) DfsOrder(v, alive, seen, order);
        }

        var seen2 = new bool[_n];
        var result = new List<List<int>>();
        for (int i = order.Count - 1; i >= 0; i--)
        {
            int start = order[i];
            if (!alive[start] || seen2[start]) continue;
            var comp = new List<int>();
            var stack = new Stack<int>();
            stack.Push(start);
            seen2[start] = true;
            while (stack.Count > 0)
            {
                int v = stack.Pop();
                comp.Add(v);
                foreach (int from in _in[v])
                {
                    if (alive[from] && !seen2[from])
                    {
                        seen2[from] = true;
                        stack.Push(from);
                    }
                }
            }

            if (comp.Count > 1 || HasSelfLoop(comp[0], alive))
            {
                result.Add(comp);
            }
        }
        return result;
    }

    private void DfsOrder(int start, bool[] alive, bool[] seen, List<int> order)
    {
        var stack = new Stack<(int V, int I)>();
        seen[start] = true;
        stack.Push((start, 0));
        while (stack.Count > 0)
        {
            var cur = stack.Pop();
            int v = cur.V;
            int i = cur.I;
            if (i < _out[v].Count)
            {
                stack.Push((v, i + 1));
                int to = _out[v][i];
                if (alive[to] && !seen[to])
                {
                    seen[to] = true;
                    stack.Push((to, 0));
                }
            }
            else
            {
                order.Add(v);
            }
        }
    }

    private bool HasSelfLoop(int v, bool[] alive)
    {
        if (!alive[v]) return false;
        foreach (int to in _out[v]) if (to == v && alive[to]) return true;
        return false;
    }

    private List<int> FindCycle(HashSet<int> alive, HashSet<int> blocked = null)
    {
        var state = new Dictionary<int, byte>(alive.Count);
        var pos = new Dictionary<int, int>(alive.Count);
        var path = new List<int>();

        foreach (int start in alive)
        {
            if (blocked != null && blocked.Contains(start)) continue;
            if (state.ContainsKey(start)) continue;

            var stack = new Stack<(int V, int I)>();
            stack.Push((start, 0));
            state[start] = 1;
            pos[start] = 0;
            path.Clear();
            path.Add(start);

            while (stack.Count > 0)
            {
                var cur = stack.Pop();
                int v = cur.V;
                int i = cur.I;
                if (i < _out[v].Count)
                {
                    stack.Push((v, i + 1));
                    int to = _out[v][i];
                    if (!alive.Contains(to)) continue;
                    if (blocked != null && blocked.Contains(to)) continue;

                    if (!state.TryGetValue(to, out byte st))
                    {
                        state[to] = 1;
                        pos[to] = path.Count;
                        path.Add(to);
                        stack.Push((to, 0));
                    }
                    else if (st == 1)
                    {
                        int p = pos[to];
                        var cycle = new List<int>();
                        for (int j = p; j < path.Count; j++) cycle.Add(path[j]);
                        return cycle;
                    }
                }
                else
                {
                    state[v] = 2;
                    int idx = pos[v];
                    if (idx >= 0 && idx == path.Count - 1)
                    {
                        path.RemoveAt(path.Count - 1);
                    }
                    pos[v] = -1;
                }
            }
        }

        return new List<int>();
    }

    private static int PopCount(int x)
    {
        int c = 0;
        while (x != 0)
        {
            x &= x - 1;
            c++;
        }
        return c;
    }

    private static int TrailingZeroCount(int x)
    {
        int c = 0;
        while ((x & 1) == 0)
        {
            x >>= 1;
            c++;
        }
        return c;
    }
}

internal sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _len;
    private int _ptr;

    public FastScanner(Stream stream)
    {
        _stream = stream;
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
        } while (c >= 0 && c <= 32);

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int value = 0;
        while (c > 32)
        {
            value = value * 10 + c - '0';
            c = Read();
        }
        return value * sign;
    }

    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len == 0) return -1;
        }
        return _buffer[_ptr++];
    }
}
