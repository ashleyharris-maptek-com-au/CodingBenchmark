using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Numerics;
using System.Threading.Tasks;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        int m = scanner.NextInt();

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var neighbors = new int[n][];
        var degree = new int[n];
        var temp = new List<int>[n];
        for (int i = 0; i < n; i++) temp[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if (u == v || (uint)u >= (uint)n || (uint)v >= (uint)n) continue;
            temp[u].Add(v);
            temp[v].Add(u);
        }

        for (int i = 0; i < n; i++)
        {
            if (temp[i].Count > 1)
            {
                temp[i].Sort();
                int write = 1;
                for (int read = 1; read < temp[i].Count; read++)
                {
                    if (temp[i][read] != temp[i][read - 1])
                    {
                        temp[i][write++] = temp[i][read];
                    }
                }
                if (write != temp[i].Count) temp[i].RemoveRange(write, temp[i].Count - write);
            }

            neighbors[i] = temp[i].ToArray();
            degree[i] = neighbors[i].Length;
            temp[i] = null;
        }

        int[] best = SolveMaximumClique(neighbors, degree);

        Array.Sort(best);
        Console.WriteLine(best.Length);
        if (best.Length > 0)
        {
            Console.WriteLine(string.Join(" ", best));
        }
        else
        {
            Console.WriteLine();
        }
    }

    private static int[] SolveMaximumClique(int[][] neighbors, int[] degree)
    {
        int n = neighbors.Length;
        long start = Environment.TickCount64;
        long deadline = start + 270_000;

        if (n == 1) return new[] { 0 };

        bool useDense = n <= 4096 && ShouldUseBitsets(neighbors, degree);
        if (useDense)
        {
            return SolveDense(neighbors, degree, deadline);
        }

        return SolveSparseHeuristic(neighbors, degree, deadline);
    }

    private static bool ShouldUseBitsets(int[][] neighbors, int[] degree)
    {
        long n = neighbors.Length;
        long edges2 = 0;
        foreach (int d in degree) edges2 += d;
        double density = n <= 1 ? 0.0 : (double)edges2 / (n * (n - 1));
        return n <= 768 || density >= 0.06;
    }

    private static int[] SolveDense(int[][] neighbors, int[] degree, long deadline)
    {
        int n = neighbors.Length;
        int words = (n + 63) >> 6;
        var adjacency = new ulong[n][];
        for (int v = 0; v < n; v++)
        {
            var bits = new ulong[words];
            foreach (int u in neighbors[v])
            {
                bits[u >> 6] |= 1UL << (u & 63);
            }
            adjacency[v] = bits;
        }

        int[] order = Enumerable.Range(0, n).ToArray();
        Array.Sort(order, (a, b) =>
        {
            int cmp = degree[b].CompareTo(degree[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        var heuristic = new DenseHeuristic(adjacency, degree, deadline);
        int[] seed = heuristic.BuildInitialClique(order);
        if (seed.Length == n) return seed;

        var solver = new DenseExactSolver(adjacency, degree, deadline, seed);
        solver.Run(order);
        return solver.GetBestClique();
    }

    private static int[] SolveSparseHeuristic(int[][] neighbors, int[] degree, long deadline)
    {
        int n = neighbors.Length;
        var rank = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ThenBy(v => v).ToArray();
        var pos = new int[n];
        for (int i = 0; i < n; i++) pos[rank[i]] = i;

        var clique = new List<int>();
        var best = new List<int>();
        var candidates = new List<int>(rank);
        var marks = new int[n];
        int stamp = 1;

        void ExtendGreedy(List<int> startCandidates)
        {
            clique.Clear();
            var local = new List<int>(startCandidates);
            while (local.Count > 0)
            {
                int chosen = local[0];
                clique.Add(chosen);
                if (clique.Count > best.Count)
                {
                    best.Clear();
                    best.AddRange(clique);
                }

                if (Environment.TickCount64 >= deadline) return;

                stamp++;
                foreach (int u in neighbors[chosen]) marks[u] = stamp;
                int write = 0;
                for (int i = 1; i < local.Count; i++)
                {
                    int v = local[i];
                    if (marks[v] == stamp) local[write++] = v;
                }
                if (write < local.Count) local.RemoveRange(write, local.Count - write);
                local.Sort((a, b) =>
                {
                    int cmp = degree[b].CompareTo(degree[a]);
                    return cmp != 0 ? cmp : pos[a].CompareTo(pos[b]);
                });
            }
        }

        ExtendGreedy(candidates);

        int trials = Math.Min(n, n <= 200_000 ? 256 : 64);
        for (int i = 0; i < trials && Environment.TickCount64 < deadline; i++)
        {
            int pivot = rank[i];
            var start = new List<int> { pivot };
            stamp++;
            foreach (int u in neighbors[pivot]) marks[u] = stamp;
            foreach (int u in neighbors[pivot])
            {
                if (degree[u] + 1 >= best.Count && marks[u] == stamp)
                {
                    start.Add(u);
                }
            }
            start.Sort((a, b) =>
            {
                int cmp = degree[b].CompareTo(degree[a]);
                return cmp != 0 ? cmp : pos[a].CompareTo(pos[b]);
            });
            ExtendGreedy(start);
        }

        return best.ToArray();
    }

    private sealed class DenseHeuristic
    {
        private readonly ulong[][] _adjacency;
        private readonly int[] _degree;
        private readonly long _deadline;
        private readonly int _n;

        public DenseHeuristic(ulong[][] adjacency, int[] degree, long deadline)
        {
            _adjacency = adjacency;
            _degree = degree;
            _deadline = deadline;
            _n = adjacency.Length;
        }

        public int[] BuildInitialClique(int[] order)
        {
            var best = new List<int>();
            int attempts = Math.Min(order.Length, Environment.ProcessorCount * 4);
            object gate = new object();

            Parallel.For(0, attempts, new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, idx =>
            {
                if (Environment.TickCount64 >= _deadline) return;
                int start = order[idx];
                var local = GrowClique(start, idx);
                lock (gate)
                {
                    if (local.Count > best.Count)
                    {
                        best = local;
                    }
                }
            });

            if (best.Count == 0 && _n > 0) best.Add(order[0]);
            return best.ToArray();
        }

        private List<int> GrowClique(int seed, int variant)
        {
            var clique = new List<int> { seed };
            var candidates = new List<int>();
            foreach (int u in ExpandNeighbors(seed))
            {
                candidates.Add(u);
            }

            while (candidates.Count > 0 && Environment.TickCount64 < _deadline)
            {
                int pickIndex = 0;
                for (int i = 1; i < candidates.Count; i++)
                {
                    int a = candidates[i];
                    int b = candidates[pickIndex];
                    int cmp = _degree[a].CompareTo(_degree[b]);
                    if ((variant & 1) == 0)
                    {
                        if (cmp > 0 || (cmp == 0 && a < b)) pickIndex = i;
                    }
                    else
                    {
                        if (cmp < 0 || (cmp == 0 && a < b)) pickIndex = i;
                    }
                }

                int chosen = candidates[pickIndex];
                clique.Add(chosen);
                var next = new List<int>(candidates.Count);
                for (int i = 0; i < candidates.Count; i++)
                {
                    if (i == pickIndex) continue;
                    int v = candidates[i];
                    if (HasEdge(chosen, v)) next.Add(v);
                }
                candidates = next;
            }

            return clique;
        }

        private IEnumerable<int> ExpandNeighbors(int v)
        {
            ulong[] row = _adjacency[v];
            for (int w = 0; w < row.Length; w++)
            {
                ulong bits = row[w];
                while (bits != 0)
                {
                    int bit = BitOperations.TrailingZeroCount(bits);
                    int u = (w << 6) + bit;
                    if (u < _n) yield return u;
                    bits &= bits - 1;
                }
            }
        }

        private bool HasEdge(int a, int b) => (_adjacency[a][b >> 6] & (1UL << (b & 63))) != 0;
    }

    private sealed class DenseExactSolver
    {
        private readonly ulong[][] _adjacency;
        private readonly int[] _degree;
        private readonly long _deadline;
        private readonly List<int> _current;
        private int[] _best;

        public DenseExactSolver(ulong[][] adjacency, int[] degree, long deadline, int[] seed)
        {
            _adjacency = adjacency;
            _degree = degree;
            _deadline = deadline;
            _current = new List<int>(seed.Length + 8);
            _best = (int[])seed.Clone();
        }

        public void Run(int[] order)
        {
            var candidates = new List<int>(order.Length);
            for (int i = 0; i < order.Length; i++)
            {
                int v = order[i];
                if (_degree[v] + 1 > _best.Length) candidates.Add(v);
            }

            Expand(candidates);
        }

        public int[] GetBestClique() => _best;

        private void Expand(List<int> candidates)
        {
            if (Environment.TickCount64 >= _deadline) return;
            if (candidates.Count == 0)
            {
                if (_current.Count > _best.Length) _best = _current.ToArray();
                return;
            }

            ColorSort(candidates, out int[] colors);
            for (int i = candidates.Count - 1; i >= 0; i--)
            {
                if (_current.Count + colors[i] <= _best.Length) return;
                int v = candidates[i];
                _current.Add(v);

                var next = new List<int>(i);
                for (int j = 0; j < i; j++)
                {
                    int u = candidates[j];
                    if (HasEdge(v, u)) next.Add(u);
                }

                if (next.Count == 0)
                {
                    if (_current.Count > _best.Length) _best = _current.ToArray();
                }
                else if (_current.Count + next.Count > _best.Length)
                {
                    Expand(next);
                }

                _current.RemoveAt(_current.Count - 1);
                if (Environment.TickCount64 >= _deadline) return;
            }
        }

        private void ColorSort(List<int> candidates, out int[] colors)
        {
            candidates.Sort((a, b) =>
            {
                int cmp = _degree[b].CompareTo(_degree[a]);
                return cmp != 0 ? cmp : a.CompareTo(b);
            });

            int n = candidates.Count;
            var verts = candidates.ToArray();
            colors = new int[n];
            var colorClasses = new List<List<int>>();

            for (int idx = 0; idx < n; idx++)
            {
                int v = verts[idx];
                int color = 1;
                while (true)
                {
                    if (colorClasses.Count < color) colorClasses.Add(new List<int>());
                    bool fits = true;
                    var cls = colorClasses[color - 1];
                    for (int j = 0; j < cls.Count; j++)
                    {
                        if (HasEdge(v, cls[j]))
                        {
                            fits = false;
                            break;
                        }
                    }

                    if (fits)
                    {
                        cls.Add(v);
                        colors[idx] = color;
                        break;
                    }

                    color++;
                }
            }

            Array.Sort(colors, verts);
            candidates.Clear();
            candidates.AddRange(verts);
        }

        private bool HasEdge(int a, int b) => (_adjacency[a][b >> 6] & (1UL << (b & 63))) != 0;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_length <= 0) return -1;
            }

            return _buffer[_ptr++];
        }
    }
}
