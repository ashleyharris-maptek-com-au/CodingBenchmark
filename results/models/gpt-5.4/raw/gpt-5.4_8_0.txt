use std::collections::VecDeque;
use std::io::{self, Read};

const NONE: u8 = 255;

fn dir_from_to(from: usize, to: usize, width: usize) -> u8 {
    if to + 1 == from {
        0
    } else if to == from + 1 {
        1
    } else if to + width == from {
        2
    } else {
        3
    }
}

fn step(idx: usize, dir: u8, width: usize) -> usize {
    match dir {
        0 => idx - 1,
        1 => idx + 1,
        2 => idx - width,
        3 => idx + width,
        _ => unreachable!(),
    }
}

fn push_neighbors(
    current: usize,
    width: usize,
    height: usize,
    passable: &[u8],
    own_parent: &mut [u8],
    other_parent: &[u8],
    queue: &mut VecDeque<usize>,
) -> Option<(usize, usize)> {
    let x = current % width;
    let y = current / width;

    if x > 0 {
        let next = current - 1;
        if passable[next] != b'#' && own_parent[next] == NONE {
            own_parent[next] = dir_from_to(next, current, width);
            if other_parent[next] != NONE {
                return Some((next, next));
            }
            queue.push_back(next);
        }
    }
    if x + 1 < width {
        let next = current + 1;
        if passable[next] != b'#' && own_parent[next] == NONE {
            own_parent[next] = dir_from_to(next, current, width);
            if other_parent[next] != NONE {
                return Some((next, next));
            }
            queue.push_back(next);
        }
    }
    if y > 0 {
        let next = current - width;
        if passable[next] != b'#' && own_parent[next] == NONE {
            own_parent[next] = dir_from_to(next, current, width);
            if other_parent[next] != NONE {
                return Some((next, next));
            }
            queue.push_back(next);
        }
    }
    if y + 1 < height {
        let next = current + width;
        if passable[next] != b'#' && own_parent[next] == NONE {
            own_parent[next] = dir_from_to(next, current, width);
            if other_parent[next] != NONE {
                return Some((next, next));
            }
            queue.push_back(next);
        }
    }

    None
}

fn expand_layer(
    width: usize,
    height: usize,
    passable: &[u8],
    own_parent: &mut [u8],
    other_parent: &[u8],
    queue: &mut VecDeque<usize>,
) -> Option<(usize, usize)> {
    let layer = queue.len();
    for _ in 0..layer {
        let current = queue.pop_front().unwrap();
        if let Some(hit) = push_neighbors(
            current,
            width,
            height,
            passable,
            own_parent,
            other_parent,
            queue,
        ) {
            return Some(hit);
        }
    }
    None
}

fn build_path(
    meet_from_start: usize,
    meet_from_goal: usize,
    start: usize,
    goal: usize,
    width: usize,
    parent_from_start: &[u8],
    parent_from_goal: &[u8],
) -> Vec<usize> {
    let mut left = Vec::new();
    let mut cur = meet_from_start;
    left.push(cur);
    while cur != start {
        cur = step(cur, parent_from_start[cur], width);
        left.push(cur);
    }
    left.reverse();

    let mut right = Vec::new();
    let mut cur = meet_from_goal;
    while cur != goal {
        cur = step(cur, parent_from_goal[cur], width);
        right.push(cur);
    }

    if meet_from_start != meet_from_goal {
        let mut bridge = Vec::new();
        let mut cur = meet_from_goal;
        bridge.push(cur);
        while cur != meet_from_start {
            cur = step(cur, parent_from_start[cur], width);
            bridge.push(cur);
        }
        bridge.reverse();
        left.pop();
        left.extend(bridge);
    }

    left.extend(right);
    left
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }

    let mut lines = input.lines();
    let header = match lines.next() {
        Some(h) => h,
        None => return,
    };
    let mut hw = header.split_whitespace();
    let height: usize = match hw.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let width: usize = match hw.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };

    let total = match height.checked_mul(width) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut grid = Vec::with_capacity(total);
    let mut start = None;
    let mut goal = None;

    for y in 0..height {
        let line = match lines.next() {
            Some(v) => v,
            None => {
                println!("0");
                return;
            }
        };
        let bytes = line.as_bytes();
        if bytes.len() != width {
            println!("0");
            return;
        }
        for (x, &cell) in bytes.iter().enumerate() {
            let idx = y * width + x;
            if cell == b'A' {
                start = Some(idx);
            } else if cell == b'B' {
                goal = Some(idx);
            }
            grid.push(cell);
        }
    }

    let start = match start {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let goal = match goal {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if start == goal {
        println!("1");
        println!("{} {}", start % width, start / width);
        return;
    }

    let mut parent_from_start = vec![NONE; total];
    let mut parent_from_goal = vec![NONE; total];
    parent_from_start[start] = 254;
    parent_from_goal[goal] = 254;

    let mut q_start = VecDeque::new();
    let mut q_goal = VecDeque::new();
    q_start.push_back(start);
    q_goal.push_back(goal);

    let meet = loop {
        if q_start.is_empty() || q_goal.is_empty() {
            println!("0");
            return;
        }

        let found = if q_start.len() <= q_goal.len() {
            expand_layer(
                width,
                height,
                &grid,
                &mut parent_from_start,
                &parent_from_goal,
                &mut q_start,
            )
        } else {
            expand_layer(
                width,
                height,
                &grid,
                &mut parent_from_goal,
                &parent_from_start,
                &mut q_goal,
            )
        };

        if let Some(hit) = found {
            break hit;
        }
    };

    let path = if parent_from_start[meet.0] != NONE && parent_from_goal[meet.1] != NONE {
        build_path(
            meet.0,
            meet.1,
            start,
            goal,
            width,
            &parent_from_start,
            &parent_from_goal,
        )
    } else {
        println!("0");
        return;
    };

    println!("{}", path.len());
    let mut output = String::new();
    for idx in path {
        let x = idx % width;
        let y = idx / width;
        output.push_str(&format!("{} {}\n", x, y));
    }
    print!("{output}");
}
