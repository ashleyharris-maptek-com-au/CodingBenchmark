#include <algorithm>
#include <array>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <vector>

using namespace std;

namespace {

using Clock = std::chrono::steady_clock;

struct Solver {
    int n = 0;
    int m = 0;
    int k = 0;
    vector<vector<int>> adj;
    vector<int> degree;
    vector<int> color;
    vector<int> order;
    vector<int> pos_in_order;
    vector<int> neighbor_color_count;
    vector<uint64_t> forbid;
    vector<int> parent;
    std::mt19937 rng;
    Clock::time_point start_time;
    double time_limit_seconds = 295.0;

    Solver() : rng(123456789u) {}

    bool timed_out() const {
        double elapsed = chrono::duration<double>(Clock::now() - start_time).count();
        return elapsed >= time_limit_seconds;
    }

    void build_order() {
        order.resize(n);
        iota(order.begin(), order.end(), 0);

        vector<int> score(n, 0);
        for (int v = 0; v < n; ++v) {
            long long s = 1LL * degree[v] * 1024;
            for (int u : adj[v]) {
                s += degree[u];
            }
            score[v] = static_cast<int>(min<long long>(s, numeric_limits<int>::max()));
        }

        stable_sort(order.begin(), order.end(), [&](int a, int b) {
            if (degree[a] != degree[b]) return degree[a] > degree[b];
            if (score[a] != score[b]) return score[a] > score[b];
            return a < b;
        });

        pos_in_order.assign(n, 0);
        for (int i = 0; i < n; ++i) pos_in_order[order[i]] = i;
    }

    bool greedy_with_order(const vector<int>& seq) {
        fill(color.begin(), color.end(), -1);
        vector<int> used_stamp(k, -1);
        for (int idx = 0; idx < static_cast<int>(seq.size()); ++idx) {
            int v = seq[idx];
            for (int u : adj[v]) {
                int c = color[u];
                if (c >= 0) used_stamp[c] = v;
            }
            int chosen = -1;
            for (int c = 0; c < k; ++c) {
                if (used_stamp[c] != v) {
                    chosen = c;
                    break;
                }
            }
            if (chosen < 0) return false;
            color[v] = chosen;
        }
        return true;
    }

    bool greedy_with_dynamic_choice() {
        fill(color.begin(), color.end(), -1);
        vector<uint64_t> unavailable(n, 0);
        vector<int> sat(n, 0);
        vector<char> alive(n, 1);

        for (int step = 0; step < n; ++step) {
            int best = -1;
            int best_sat = -1;
            int best_deg = -1;
            for (int v = 0; v < n; ++v) {
                if (!alive[v]) continue;
                if (sat[v] > best_sat || (sat[v] == best_sat && degree[v] > best_deg)) {
                    best = v;
                    best_sat = sat[v];
                    best_deg = degree[v];
                }
            }
            if (best < 0) break;

            int chosen = -1;
            for (int c = 0; c < k; ++c) {
                if (((unavailable[best] >> c) & 1ULL) == 0ULL) {
                    chosen = c;
                    break;
                }
            }
            if (chosen < 0) return false;

            alive[best] = 0;
            color[best] = chosen;
            for (int u : adj[best]) {
                if (!alive[u]) continue;
                uint64_t before = unavailable[u];
                unavailable[u] |= (1ULL << chosen);
                if (unavailable[u] != before) {
                    sat[u] += 1;
                }
            }
        }
        return true;
    }

    bool try_component_bfs_repair(int root) {
        vector<int> q;
        q.reserve(1024);
        parent.assign(n, -1);
        q.push_back(root);
        parent[root] = root;
        size_t head = 0;
        while (head < q.size() && q.size() <= 50000) {
            int v = q[head++];
            for (int u : adj[v]) {
                if (parent[u] == -1) {
                    parent[u] = v;
                    q.push_back(u);
                }
            }
        }

        vector<int> comp = q;
        sort(comp.begin(), comp.end(), [&](int a, int b) {
            if (degree[a] != degree[b]) return degree[a] > degree[b];
            return a < b;
        });

        vector<int> backup;
        backup.reserve(comp.size());
        for (int v : comp) backup.push_back(color[v]);
        for (int v : comp) color[v] = -1;

        bool ok = true;
        vector<int> used_stamp(k, -1);
        for (int v : comp) {
            for (int u : adj[v]) {
                int c = color[u];
                if (c >= 0) used_stamp[c] = v;
            }
            int chosen = -1;
            int best_conflict = numeric_limits<int>::max();
            for (int c = 0; c < k; ++c) {
                if (used_stamp[c] == v) continue;
                int future_conflicts = 0;
                for (int u : adj[v]) {
                    if (color[u] == -1) {
                        bool free_for_u = true;
                        for (int w : adj[u]) {
                            if (color[w] == c) {
                                free_for_u = false;
                                break;
                            }
                        }
                        if (!free_for_u) ++future_conflicts;
                    }
                }
                if (future_conflicts < best_conflict) {
                    best_conflict = future_conflicts;
                    chosen = c;
                    if (future_conflicts == 0) break;
                }
            }
            if (chosen < 0) {
                ok = false;
                break;
            }
            color[v] = chosen;
        }

        if (!ok) {
            for (size_t i = 0; i < comp.size(); ++i) color[comp[i]] = backup[i];
        }
        return ok;
    }

    bool local_repair_pass() {
        vector<int> bad;
        for (int v = 0; v < n; ++v) {
            if (color[v] < 0) bad.push_back(v);
        }
        if (bad.empty()) return true;

        for (int v : bad) {
            vector<int> blocked(k, 0);
            for (int u : adj[v]) {
                if (color[u] >= 0) blocked[color[u]] = 1;
            }
            for (int c = 0; c < k; ++c) {
                if (!blocked[c]) {
                    color[v] = c;
                    break;
                }
            }
            if (color[v] >= 0) continue;

            bool repaired = false;
            for (int u : adj[v]) {
                int old = color[u];
                if (old < 0) continue;

                vector<int> blocked_u(k, 0);
                for (int w : adj[u]) {
                    if (w == v) continue;
                    if (color[w] >= 0) blocked_u[color[w]] = 1;
                }
                for (int nc = 0; nc < k; ++nc) {
                    if (nc == old || blocked_u[nc]) continue;
                    color[u] = nc;
                    bool ok = true;
                    for (int w : adj[v]) {
                        if (color[w] == old) {
                            ok = false;
                            break;
                        }
                    }
                    if (ok) {
                        color[v] = old;
                        repaired = true;
                        break;
                    }
                }
                if (repaired) break;
                color[u] = old;
            }
            if (!repaired) {
                try_component_bfs_repair(v);
            }
            if (color[v] < 0) return false;
        }
        return true;
    }

    bool verify() const {
        for (int v = 0; v < n; ++v) {
            if (color[v] < 0 || color[v] >= k) return false;
            for (int u : adj[v]) {
                if (u > v && color[u] == color[v]) return false;
            }
        }
        return true;
    }

    bool solve() {
        start_time = Clock::now();
        color.assign(n, -1);
        degree.resize(n);
        for (int v = 0; v < n; ++v) degree[v] = static_cast<int>(adj[v].size());

        if (k <= 0) return false;
        if (n == 0) return true;
        if (k == 1) return m == 0;

        build_order();

        if (k <= 60) {
            if (greedy_with_order(order) && verify()) return true;

            vector<int> shuffled = order;
            for (int attempt = 0; attempt < 6 && !timed_out(); ++attempt) {
                int prefix = min(n, max(8, n / 20));
                shuffle(shuffled.begin(), shuffled.begin() + prefix, rng);
                if (greedy_with_order(shuffled) && verify()) return true;
            }

            if (greedy_with_dynamic_choice() && verify()) return true;
        }

        if (!local_repair_pass()) return false;
        return verify();
    }
};

}  // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Solver solver;
    if (!(cin >> solver.n >> solver.m >> solver.k)) {
        return 0;
    }

    solver.adj.assign(solver.n, {});
    solver.adj.reserve(solver.n);

    for (int i = 0; i < solver.m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= solver.n || v >= solver.n || u == v) continue;
        solver.adj[u].push_back(v);
        solver.adj[v].push_back(u);
    }

    for (int v = 0; v < solver.n; ++v) {
        auto& nbrs = solver.adj[v];
        sort(nbrs.begin(), nbrs.end());
        nbrs.erase(unique(nbrs.begin(), nbrs.end()), nbrs.end());
    }

    solver.m = 0;
    for (int v = 0; v < solver.n; ++v) solver.m += static_cast<int>(solver.adj[v].size());
    solver.m /= 2;

    bool ok = solver.solve();
    if (!ok) {
        vector<int> fallback_order(solver.n);
        iota(fallback_order.begin(), fallback_order.end(), 0);
        solver.color.assign(solver.n, -1);
        solver.greedy_with_order(fallback_order);
    }

    for (int i = 0; i < solver.n; ++i) {
        if (i) cout << ' ';
        int c = solver.color[i];
        if (c < 0) c = 0;
        cout << c;
    }
    cout << '\n';
    return 0;
}
