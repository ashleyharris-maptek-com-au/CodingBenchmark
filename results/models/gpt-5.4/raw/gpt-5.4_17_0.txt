using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    private struct Box
    {
        public int Index;
        public int W;
        public int H;
        public int D;
        public long Volume;
    }

    private struct Placement
    {
        public int BoxIndex;
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
    }

    private struct Space
    {
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;

        public long Volume => (long)W * H * D;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0)
                {
                    return -1;
                }
            }

            return _buffer[_ptr++];
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int containerW = fs.NextInt();
        int containerH = fs.NextInt();
        int containerD = fs.NextInt();

        var boxes = new Box[n];
        long containerVolume = (long)containerW * containerH * containerD;

        for (int i = 0; i < n; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            int c = fs.NextInt();
            var dims = new[] { a, b, c };
            Array.Sort(dims);
            boxes[i] = new Box
            {
                Index = i,
                W = dims[0],
                H = dims[1],
                D = dims[2],
                Volume = (long)dims[0] * dims[1] * dims[2]
            };
        }

        int sampleSpaces = n <= 128 ? 96 : n <= 2000 ? 40 : 12;
        int candidateBoxes = n <= 128 ? 128 : n <= 2000 ? 48 : 16;

        Array.Sort(boxes, (l, r) => CompareBoxes(l, r, containerVolume));

        var spaces = new List<Space>(1024)
        {
            new Space { X = 0, Y = 0, Z = 0, W = containerW, H = containerH, D = containerD }
        };
        var used = new bool[n];
        var placements = new List<Placement>(Math.Min(n, 100000));

        long packedVolume = 0;
        int packedCount = 0;

        while (true)
        {
            if (spaces.Count == 0)
            {
                break;
            }

            int chosenSpaceIndex = -1;
            int chosenBoxSortedIndex = -1;
            int chosenOrientation = -1;
            long bestScore = long.MinValue;

            int spacesToProbe = Math.Min(sampleSpaces, spaces.Count);
            for (int s = 0; s < spacesToProbe; s++)
            {
                int spaceIndex = SelectBestSpaceIndex(spaces, s);
                var space = spaces[spaceIndex];

                int examined = 0;
                for (int i = 0; i < n && examined < candidateBoxes; i++)
                {
                    if (used[boxes[i].Index])
                    {
                        continue;
                    }

                    examined++;
                    if (!TryBestOrientation(boxes[i], space, out int bw, out int bh, out int bd, out int orientationScore))
                    {
                        continue;
                    }

                    long waste = space.Volume - (long)bw * bh * bd;
                    long touchScore = 0;
                    if (bw == space.W) touchScore += 3;
                    if (bh == space.H) touchScore += 3;
                    if (bd == space.D) touchScore += 3;
                    touchScore += orientationScore;

                    long score = touchScore * 1_000_000_000L - waste;
                    if (score > bestScore)
                    {
                        bestScore = score;
                        chosenSpaceIndex = spaceIndex;
                        chosenBoxSortedIndex = i;
                        chosenOrientation = PackOrientation(bw, bh, bd);
                    }
                }
            }

            if (chosenBoxSortedIndex < 0)
            {
                int largestSpaceIndex = SelectBestSpaceIndex(spaces, 0);
                spaces.RemoveAt(largestSpaceIndex);
                continue;
            }

            var targetSpace = spaces[chosenSpaceIndex];
            UnpackOrientation(chosenOrientation, out int placeW, out int placeH, out int placeD);
            var chosenBox = boxes[chosenBoxSortedIndex];
            used[chosenBox.Index] = true;
            placements.Add(new Placement
            {
                BoxIndex = chosenBox.Index,
                X = targetSpace.X,
                Y = targetSpace.Y,
                Z = targetSpace.Z,
                W = placeW,
                H = placeH,
                D = placeD
            });
            packedCount++;
            packedVolume += (long)placeW * placeH * placeD;

            spaces.RemoveAt(chosenSpaceIndex);
            SplitSpace(spaces, targetSpace, placeW, placeH, placeD);
            PruneSpaces(spaces);

            if (packedVolume == containerVolume || packedCount == n)
            {
                break;
            }
        }

        var writer = new StreamWriter(Console.OpenStandardOutput())
        {
            NewLine = "\n"
        };
        writer.WriteLine(placements.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var p in placements)
        {
            writer.Write(p.BoxIndex.ToString(CultureInfo.InvariantCulture));
            writer.Write(' ');
            writer.Write(p.X.ToString(CultureInfo.InvariantCulture));
            writer.Write(' ');
            writer.Write(p.Y.ToString(CultureInfo.InvariantCulture));
            writer.Write(' ');
            writer.WriteLine(p.Z.ToString(CultureInfo.InvariantCulture));
        }
        writer.Flush();
    }

    private static int CompareBoxes(Box l, Box r, long containerVolume)
    {
        bool lTiny = l.Volume * 24 < containerVolume;
        bool rTiny = r.Volume * 24 < containerVolume;
        if (lTiny != rTiny)
        {
            return lTiny ? -1 : 1;
        }

        int cmp = l.Volume.CompareTo(r.Volume);
        if (cmp != 0)
        {
            return cmp;
        }

        cmp = (l.W + l.H + l.D).CompareTo(r.W + r.H + r.D);
        if (cmp != 0)
        {
            return cmp;
        }

        return l.Index.CompareTo(r.Index);
    }

    private static bool TryBestOrientation(Box box, Space space, out int bestW, out int bestH, out int bestD, out int score)
    {
        bestW = 0;
        bestH = 0;
        bestD = 0;
        score = int.MinValue;

        int a = box.W;
        int b = box.H;
        int c = box.D;

        TryOrientation(a, b, c);
        TryOrientation(a, c, b);
        TryOrientation(b, a, c);
        TryOrientation(b, c, a);
        TryOrientation(c, a, b);
        TryOrientation(c, b, a);

        return score != int.MinValue;

        void TryOrientation(int w, int h, int d)
        {
            if (w > space.W || h > space.H || d > space.D)
            {
                return;
            }

            int fitScore = 0;
            fitScore -= (space.W - w) + (space.H - h) + (space.D - d);
            if (w == space.W) fitScore += 60;
            if (h == space.H) fitScore += 60;
            if (d == space.D) fitScore += 60;

            if (fitScore > score)
            {
                score = fitScore;
                bestW = w;
                bestH = h;
                bestD = d;
            }
        }
    }

    private static int SelectBestSpaceIndex(List<Space> spaces, int rank)
    {
        int bestIndex = -1;
        long bestVolume = long.MinValue;
        int bestX = 0;
        int bestY = 0;
        int bestZ = 0;

        for (int i = 0; i < spaces.Count; i++)
        {
            var s = spaces[i];
            long volume = s.Volume;
            if (volume > bestVolume ||
                (volume == bestVolume && (s.Z < bestZ || (s.Z == bestZ && (s.Y < bestY || (s.Y == bestY && s.X < bestX))))))
            {
                bestVolume = volume;
                bestIndex = i;
                bestX = s.X;
                bestY = s.Y;
                bestZ = s.Z;
            }
        }

        if (rank == 0 || spaces.Count <= 1)
        {
            return bestIndex;
        }

        var top = new List<(int Index, long Volume, int X, int Y, int Z)>(spaces.Count);
        for (int i = 0; i < spaces.Count; i++)
        {
            var s = spaces[i];
            top.Add((i, s.Volume, s.X, s.Y, s.Z));
        }

        top.Sort((l, r) =>
        {
            int cmp = r.Volume.CompareTo(l.Volume);
            if (cmp != 0) return cmp;
            cmp = l.Z.CompareTo(r.Z);
            if (cmp != 0) return cmp;
            cmp = l.Y.CompareTo(r.Y);
            if (cmp != 0) return cmp;
            return l.X.CompareTo(r.X);
        });

        return top[Math.Min(rank, top.Count - 1)].Index;
    }

    private static void SplitSpace(List<Space> spaces, Space space, int usedW, int usedH, int usedD)
    {
        int remW = space.W - usedW;
        int remH = space.H - usedH;
        int remD = space.D - usedD;

        if (remW > 0)
        {
            spaces.Add(new Space
            {
                X = space.X + usedW,
                Y = space.Y,
                Z = space.Z,
                W = remW,
                H = space.H,
                D = space.D
            });
        }

        if (remH > 0)
        {
            spaces.Add(new Space
            {
                X = space.X,
                Y = space.Y + usedH,
                Z = space.Z,
                W = usedW,
                H = remH,
                D = space.D
            });
        }

        if (remD > 0)
        {
            spaces.Add(new Space
            {
                X = space.X,
                Y = space.Y,
                Z = space.Z + usedD,
                W = usedW,
                H = usedH,
                D = remD
            });
        }
    }

    private static void PruneSpaces(List<Space> spaces)
    {
        for (int i = spaces.Count - 1; i >= 0; i--)
        {
            var a = spaces[i];
            if (a.W <= 0 || a.H <= 0 || a.D <= 0)
            {
                spaces.RemoveAt(i);
                continue;
            }

            for (int j = 0; j < spaces.Count; j++)
            {
                if (i == j)
                {
                    continue;
                }

                var b = spaces[j];
                if (ContainedIn(a, b))
                {
                    spaces.RemoveAt(i);
                    break;
                }
            }
        }
    }

    private static bool ContainedIn(Space a, Space b)
    {
        return a.X >= b.X && a.Y >= b.Y && a.Z >= b.Z &&
               a.X + a.W <= b.X + b.W &&
               a.Y + a.H <= b.Y + b.H &&
               a.Z + a.D <= b.Z + b.D;
    }

    private static int PackOrientation(int w, int h, int d)
    {
        return (w & 1023) | ((h & 1023) << 10) | ((d & 1023) << 20);
    }

    private static void UnpackOrientation(int packed, out int w, out int h, out int d)
    {
        w = packed & 1023;
        h = (packed >> 10) & 1023;
        d = (packed >> 20) & 1023;
    }
}
