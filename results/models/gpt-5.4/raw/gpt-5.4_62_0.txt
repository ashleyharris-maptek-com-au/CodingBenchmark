declare const require: any;
const fs = require('fs');

type Point = [number, number];

interface Town {
  id: string;
  x: number;
  y: number;
  value: number;
}

interface InputData {
  width: number;
  height: number;
  grid: string[];
  ignitions: Point[];
  towns: Town[];
  cutBudget: number;
}

interface OutputData {
  firebreaks: Point[];
}

interface DSUNode {
  parent: number;
  size: number;
  townValue: number;
  ignitionCount: number;
}

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input) as InputData;

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid;
const cutBudget = Math.max(0, data.cutBudget | 0);

const totalCells = width * height;
const dirs = [1, 0, -1, 0, 1];

function idx(x: number, y: number): number {
  return y * width + x;
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function coord(index: number): Point {
  return [index % width, Math.floor(index / width)];
}

const burnable = new Uint8Array(totalCells);
for (let y = 0; y < height; y++) {
  const row = grid[y] ?? '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) {
      burnable[idx(x, y)] = 1;
    }
  }
}

const ignitionMark = new Uint8Array(totalCells);
for (const [x, y] of data.ignitions) {
  if (inBounds(x, y)) {
    const id = idx(x, y);
    if (burnable[id]) {
      ignitionMark[id] = 1;
    }
  }
}

const townValueAt = new Float64Array(totalCells);
for (const town of data.towns) {
  if (inBounds(town.x, town.y)) {
    const id = idx(town.x, town.y);
    if (burnable[id]) {
      townValueAt[id] += Number.isFinite(town.value) ? town.value : 0;
    }
  }
}

const dsu: DSUNode[] = new Array(totalCells);
for (let i = 0; i < totalCells; i++) {
  dsu[i] = {
    parent: i,
    size: 1,
    townValue: townValueAt[i],
    ignitionCount: ignitionMark[i],
  };
}

function find(a: number): number {
  let root = a;
  while (dsu[root].parent !== root) {
    root = dsu[root].parent;
  }
  while (dsu[a].parent !== a) {
    const next = dsu[a].parent;
    dsu[a].parent = root;
    a = next;
  }
  return root;
}

function union(a: number, b: number): void {
  let ra = find(a);
  let rb = find(b);
  if (ra === rb) return;
  if (dsu[ra].size < dsu[rb].size) {
    const t = ra;
    ra = rb;
    rb = t;
  }
  dsu[rb].parent = ra;
  dsu[ra].size += dsu[rb].size;
  dsu[ra].townValue += dsu[rb].townValue;
  dsu[ra].ignitionCount += dsu[rb].ignitionCount;
}

for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const a = idx(x, y);
    if (!burnable[a]) continue;
    if (x + 1 < width) {
      const b = a + 1;
      if (burnable[b]) union(a, b);
    }
    if (y + 1 < height) {
      const b = a + width;
      if (burnable[b]) union(a, b);
    }
  }
}

const componentTownValue = new Float64Array(totalCells);
const componentIgnitions = new Int32Array(totalCells);
for (let i = 0; i < totalCells; i++) {
  if (!burnable[i]) continue;
  const r = find(i);
  componentTownValue[r] = dsu[r].townValue;
  componentIgnitions[r] = dsu[r].ignitionCount;
}

const initiallyBurned = new Uint8Array(totalCells);
const queue = new Int32Array(totalCells);
let qh = 0;
let qt = 0;
for (let i = 0; i < totalCells; i++) {
  if (ignitionMark[i]) {
    initiallyBurned[i] = 1;
    queue[qt++] = i;
  }
}

while (qh < qt) {
  const cur = queue[qh++];
  const x = cur % width;
  const y = Math.floor(cur / width);
  for (let k = 0; k < 4; k++) {
    const nx = x + dirs[k];
    const ny = y + dirs[k + 1];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (!burnable[ni] || initiallyBurned[ni]) continue;
    initiallyBurned[ni] = 1;
    queue[qt++] = ni;
  }
}

const threatenedRoots: number[] = [];
const rootSeen = new Uint8Array(totalCells);
for (let i = 0; i < totalCells; i++) {
  if (!burnable[i] || !initiallyBurned[i]) continue;
  const r = find(i);
  if (!rootSeen[r]) {
    rootSeen[r] = 1;
    threatenedRoots.push(r);
  }
}

if (cutBudget === 0 || threatenedRoots.length === 0) {
  const result: OutputData = { firebreaks: [] };
  process.stdout.write(JSON.stringify(result));
  process.exit(0);
}

let maxTownValue = 0;
for (const r of threatenedRoots) {
  if (componentTownValue[r] > maxTownValue) maxTownValue = componentTownValue[r];
}

if (maxTownValue <= 0) {
  const result: OutputData = { firebreaks: [] };
  process.stdout.write(JSON.stringify(result));
  process.exit(0);
}

const dist = new Int32Array(totalCells);
dist.fill(-1);
qh = 0;
qt = 0;
for (let i = 0; i < totalCells; i++) {
  if (ignitionMark[i]) {
    dist[i] = 0;
    queue[qt++] = i;
  }
}

while (qh < qt) {
  const cur = queue[qh++];
  const x = cur % width;
  const y = Math.floor(cur / width);
  const nd = dist[cur] + 1;
  for (let k = 0; k < 4; k++) {
    const nx = x + dirs[k];
    const ny = y + dirs[k + 1];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (!burnable[ni] || dist[ni] !== -1) continue;
    dist[ni] = nd;
    queue[qt++] = ni;
  }
}

const threatWeight = new Float64Array(totalCells);
const frontierDegree = new Int32Array(totalCells);

for (let i = 0; i < totalCells; i++) {
  if (!burnable[i] || !initiallyBurned[i]) continue;
  const x = i % width;
  const y = Math.floor(i / width);
  let touchesSafe = 0;
  for (let k = 0; k < 4; k++) {
    const nx = x + dirs[k];
    const ny = y + dirs[k + 1];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (burnable[ni] && !initiallyBurned[ni]) {
      touchesSafe++;
    }
  }
  frontierDegree[i] = touchesSafe;
}

for (const r of threatenedRoots) {
  const value = componentTownValue[r];
  if (value <= 0) continue;
  const ignitionCount = Math.max(1, componentIgnitions[r]);
  const rootWeight = value / ignitionCount;
  for (let i = 0; i < totalCells; i++) {
    if (!burnable[i] || !initiallyBurned[i]) continue;
    if (find(i) !== r) continue;
    if (frontierDegree[i] === 0) continue;
    const localDist = dist[i] >= 0 ? dist[i] : 0;
    threatWeight[i] += rootWeight / (1 + localDist);
  }
}

interface Candidate {
  cell: number;
  score: number;
}

const candidates: Candidate[] = [];
for (let i = 0; i < totalCells; i++) {
  if (!burnable[i] || !initiallyBurned[i]) continue;
  const w = threatWeight[i];
  if (w <= 0) continue;
  const score = w * (1 + frontierDegree[i] * 0.35);
  candidates.push({ cell: i, score });
}

candidates.sort((a, b) => b.score - a.score);

const cuts = new Uint8Array(totalCells);
const firebreaks: Point[] = [];
const maxCandidates = Math.min(candidates.length, Math.max(cutBudget * 12, 64));

function canCut(cell: number): boolean {
  if (!burnable[cell] || cuts[cell] || ignitionMark[cell] || !initiallyBurned[cell]) {
    return false;
  }
  return true;
}

for (let i = 0; i < maxCandidates && firebreaks.length < cutBudget; i++) {
  const cell = candidates[i].cell;
  if (!canCut(cell)) continue;
  cuts[cell] = 1;
  firebreaks.push(coord(cell));
}

if (firebreaks.length < cutBudget) {
  const townCells: Candidate[] = [];
  for (const town of data.towns) {
    if (!inBounds(town.x, town.y)) continue;
    const cell = idx(town.x, town.y);
    if (!burnable[cell] || !initiallyBurned[cell]) continue;
    if (ignitionMark[cell]) continue;
    const d = dist[cell];
    const score = (Number.isFinite(town.value) ? town.value : 0) * (1 + (d >= 0 ? d : 0));
    townCells.push({ cell, score });
  }
  townCells.sort((a, b) => b.score - a.score);
  for (const entry of townCells) {
    if (firebreaks.length >= cutBudget) break;
    if (!canCut(entry.cell)) continue;
    cuts[entry.cell] = 1;
    firebreaks.push(coord(entry.cell));
  }
}

if (firebreaks.length < cutBudget) {
  const frontierCells: Candidate[] = [];
  for (let i = 0; i < totalCells; i++) {
    if (!canCut(i)) continue;
    if (frontierDegree[i] > 0) {
      const score = frontierDegree[i] * 1000 - dist[i];
      frontierCells.push({ cell: i, score });
    }
  }
  frontierCells.sort((a, b) => b.score - a.score);
  for (const entry of frontierCells) {
    if (firebreaks.length >= cutBudget) break;
    if (!canCut(entry.cell)) continue;
    cuts[entry.cell] = 1;
    firebreaks.push(coord(entry.cell));
  }
}

const result: OutputData = { firebreaks };
process.stdout.write(JSON.stringify(result));
