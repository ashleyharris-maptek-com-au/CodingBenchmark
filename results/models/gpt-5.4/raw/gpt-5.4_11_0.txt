use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-8;
const GRID_TARGET: usize = 48;

#[derive(Clone, Copy, Debug, Default)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn add(self, other: Point) -> Self {
        Self::new(self.x + other.x, self.y + other.y)
    }

    fn sub(self, other: Point) -> Self {
        Self::new(self.x - other.x, self.y - other.y)
    }

    fn dot(self, other: Point) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn cross(self, other: Point) -> f64 {
        self.x * other.y - self.y * other.x
    }
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    min_x: f64,
    min_y: f64,
    max_x: f64,
    max_y: f64,
}

impl BBox {
    fn from_points(points: &[Point]) -> Self {
        let mut min_x = f64::INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for &p in points {
            min_x = min_x.min(p.x);
            min_y = min_y.min(p.y);
            max_x = max_x.max(p.x);
            max_y = max_y.max(p.y);
        }
        Self {
            min_x,
            min_y,
            max_x,
            max_y,
        }
    }

    fn width(&self) -> f64 {
        self.max_x - self.min_x
    }

    fn height(&self) -> f64 {
        self.max_y - self.min_y
    }

    fn intersects(&self, other: &Self) -> bool {
        !(self.max_x <= other.min_x + EPS
            || other.max_x <= self.min_x + EPS
            || self.max_y <= other.min_y + EPS
            || other.max_y <= self.min_y + EPS)
    }
}

#[derive(Clone)]
struct Piece {
    id: usize,
    vertices: Vec<Point>,
    area: f64,
    bbox: BBox,
    centroid: Point,
    radius: f64,
}

#[derive(Clone)]
struct OrientedPiece {
    rotation: f64,
    vertices: Vec<Point>,
    bbox: BBox,
    anchor: Point,
    area: f64,
    radius: f64,
}

#[derive(Clone)]
struct PlacedPiece {
    stock_idx: usize,
    translation: Point,
    rotation: f64,
    vertices: Vec<Point>,
    bbox: BBox,
    center: Point,
    radius: f64,
}

struct StockState {
    placed: Vec<PlacedPiece>,
    grid: SpatialGrid,
}

struct SpatialGrid {
    bbox: BBox,
    cols: usize,
    rows: usize,
    cell_w: f64,
    cell_h: f64,
    cells: Vec<Vec<usize>>,
}

impl SpatialGrid {
    fn new(bbox: BBox) -> Self {
        let side = GRID_TARGET.max(8);
        let cols = side;
        let rows = side;
        let cell_w = (bbox.width().max(1.0)) / cols as f64;
        let cell_h = (bbox.height().max(1.0)) / rows as f64;
        Self {
            bbox,
            cols,
            rows,
            cell_w,
            cell_h,
            cells: vec![Vec::new(); cols * rows],
        }
    }

    fn clamp_x(&self, x: f64) -> usize {
        let raw = ((x - self.bbox.min_x) / self.cell_w).floor() as isize;
        raw.clamp(0, self.cols as isize - 1) as usize
    }

    fn clamp_y(&self, y: f64) -> usize {
        let raw = ((y - self.bbox.min_y) / self.cell_h).floor() as isize;
        raw.clamp(0, self.rows as isize - 1) as usize
    }

    fn cell_index(&self, x: usize, y: usize) -> usize {
        y * self.cols + x
    }

    fn bbox_cells(&self, bbox: &BBox) -> (usize, usize, usize, usize) {
        (
            self.clamp_x(bbox.min_x),
            self.clamp_x(bbox.max_x),
            self.clamp_y(bbox.min_y),
            self.clamp_y(bbox.max_y),
        )
    }

    fn insert(&mut self, idx: usize, bbox: &BBox) {
        let (x0, x1, y0, y1) = self.bbox_cells(bbox);
        for y in y0..=y1 {
            for x in x0..=x1 {
                self.cells[self.cell_index(x, y)].push(idx);
            }
        }
    }

    fn query_indices(&self, bbox: &BBox, out: &mut Vec<usize>) {
        out.clear();
        let (x0, x1, y0, y1) = self.bbox_cells(bbox);
        let mut unique = std::collections::HashSet::new();
        for y in y0..=y1 {
            for x in x0..=x1 {
                for &idx in &self.cells[self.cell_index(x, y)] {
                    if unique.insert(idx) {
                        out.push(idx);
                    }
                }
            }
        }
    }
}

fn polygon_area(poly: &[Point]) -> f64 {
    let mut sum = 0.0;
    for i in 0..poly.len() {
        sum += poly[i].cross(poly[(i + 1) % poly.len()]);
    }
    0.5 * sum.abs()
}

fn polygon_centroid(poly: &[Point]) -> Point {
    let mut cross_sum = 0.0;
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let cross = a.cross(b);
        cross_sum += cross;
        cx += (a.x + b.x) * cross;
        cy += (a.y + b.y) * cross;
    }
    if cross_sum.abs() < EPS {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for &p in poly {
            sx += p.x;
            sy += p.y;
        }
        return Point::new(sx / poly.len() as f64, sy / poly.len() as f64);
    }
    let factor = 1.0 / (3.0 * cross_sum);
    Point::new(cx * factor, cy * factor)
}

fn rotate_point(p: Point, angle: f64) -> Point {
    let (s, c) = angle.sin_cos();
    Point::new(c * p.x - s * p.y, s * p.x + c * p.y)
}

fn point_on_segment(p: Point, a: Point, b: Point) -> bool {
    let ap = p.sub(a);
    let ab = b.sub(a);
    ap.cross(ab).abs() <= EPS && ap.dot(ab) >= -EPS && ap.dot(ab) <= ab.dot(ab) + EPS
}

fn point_in_polygon(poly: &[Point], p: Point) -> bool {
    let mut inside = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        if point_on_segment(p, a, b) {
            return true;
        }
        let intersect = ((a.y > p.y) != (b.y > p.y))
            && (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x);
        if intersect {
            inside = !inside;
        }
    }
    inside
}

fn orient(a: Point, b: Point, c: Point) -> f64 {
    b.sub(a).cross(c.sub(a))
}

fn segments_intersect(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let o1 = orient(a1, a2, b1);
    let o2 = orient(a1, a2, b2);
    let o3 = orient(b1, b2, a1);
    let o4 = orient(b1, b2, a2);

    if o1.abs() <= EPS && point_on_segment(b1, a1, a2) {
        return true;
    }
    if o2.abs() <= EPS && point_on_segment(b2, a1, a2) {
        return true;
    }
    if o3.abs() <= EPS && point_on_segment(a1, b1, b2) {
        return true;
    }
    if o4.abs() <= EPS && point_on_segment(a2, b1, b2) {
        return true;
    }

    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn polygons_overlap(a: &[Point], b: &[Point]) -> bool {
    for i in 0..a.len() {
        let a1 = a[i];
        let a2 = a[(i + 1) % a.len()];
        for j in 0..b.len() {
            let b1 = b[j];
            let b2 = b[(j + 1) % b.len()];
            if segments_intersect(a1, a2, b1, b2) {
                let shared_endpoint = ((a1.x - b1.x).abs() <= EPS && (a1.y - b1.y).abs() <= EPS)
                    || ((a1.x - b2.x).abs() <= EPS && (a1.y - b2.y).abs() <= EPS)
                    || ((a2.x - b1.x).abs() <= EPS && (a2.y - b1.y).abs() <= EPS)
                    || ((a2.x - b2.x).abs() <= EPS && (a2.y - b2.y).abs() <= EPS);
                if !shared_endpoint {
                    return true;
                }
            }
        }
    }
    point_in_polygon(a, b[0]) || point_in_polygon(b, a[0])
}

fn normalize_polygon(mut poly: Vec<Point>) -> Vec<Point> {
    let mut area2 = 0.0;
    for i in 0..poly.len() {
        area2 += poly[i].cross(poly[(i + 1) % poly.len()]);
    }
    if area2 < 0.0 {
        poly.reverse();
    }
    poly
}

fn build_piece(id: usize, vertices: Vec<Point>) -> Piece {
    let vertices = normalize_polygon(vertices);
    let area = polygon_area(&vertices);
    let centroid = polygon_centroid(&vertices);
    let bbox = BBox::from_points(&vertices);
    let mut radius: f64 = 0.0;
    for &p in &vertices {
        let d = p.sub(centroid);
        radius = radius.max((d.x * d.x + d.y * d.y).sqrt());
    }
    Piece {
        id,
        vertices,
        area,
        bbox,
        centroid,
        radius,
    }
}

fn generate_rotations(np: usize, complexity: usize) -> Vec<f64> {
    let count = if np <= 50 {
        24
    } else if np <= 300 {
        16
    } else if np <= 1200 {
        8
    } else if complexity <= 5 {
        6
    } else {
        4
    };
    (0..count).map(|i| 2.0 * PI * i as f64 / count as f64).collect()
}

fn oriented_variants(piece: &Piece, rotations: &[f64]) -> Vec<OrientedPiece> {
    let local: Vec<Point> = piece.vertices.iter().map(|&p| p.sub(piece.centroid)).collect();
    rotations
        .iter()
        .map(|&rotation| {
            let vertices: Vec<Point> = local.iter().map(|&p| rotate_point(p, rotation)).collect();
            let bbox = BBox::from_points(&vertices);
            let anchor = Point::new(-bbox.min_x, -bbox.min_y);
            OrientedPiece {
                rotation,
                vertices,
                bbox,
                anchor,
                area: piece.area,
                radius: piece.radius,
            }
        })
        .collect()
}

fn candidate_positions(stock_bbox: &BBox, stock_area: f64, state: &StockState, op: &OrientedPiece, np: usize) -> Vec<Point> {
    let mut result = Vec::new();
    let left = stock_bbox.min_x - op.bbox.min_x;
    let bottom = stock_bbox.min_y - op.bbox.min_y;
    let right = stock_bbox.max_x - op.bbox.max_x;
    let top = stock_bbox.max_y - op.bbox.max_y;

    result.push(Point::new(left, bottom));
    result.push(Point::new(left, top));
    result.push(Point::new(right, bottom));
    result.push(Point::new(right, top));
    result.push(Point::new(
        0.5 * ((stock_bbox.min_x + stock_bbox.max_x) - (op.bbox.min_x + op.bbox.max_x)),
        0.5 * ((stock_bbox.min_y + stock_bbox.max_y) - (op.bbox.min_y + op.bbox.max_y)),
    ));

    let density = (state.placed.len() as f64 * op.area / stock_area.max(1.0)).min(1.0);
    let limit = if np <= 100 {
        64
    } else if np <= 1000 {
        24
    } else {
        10
    };

    for placed in state.placed.iter().take(limit) {
        result.push(Point::new(placed.bbox.max_x - op.bbox.min_x, placed.bbox.min_y - op.bbox.min_y));
        result.push(Point::new(placed.bbox.min_x - op.bbox.max_x, placed.bbox.min_y - op.bbox.min_y));
        result.push(Point::new(placed.bbox.min_x - op.bbox.min_x, placed.bbox.max_y - op.bbox.min_y));
        result.push(Point::new(placed.bbox.min_x - op.bbox.min_x, placed.bbox.min_y - op.bbox.max_y));
        if density < 0.8 {
            result.push(Point::new(placed.center.x - op.anchor.x, placed.bbox.max_y - op.bbox.min_y));
        }
    }

    if state.placed.is_empty() && np > 500 {
        let sx = (stock_bbox.width() / 6.0).max(1.0);
        let sy = (stock_bbox.height() / 6.0).max(1.0);
        let mut x = stock_bbox.min_x;
        while x <= stock_bbox.max_x {
            let mut y = stock_bbox.min_y;
            while y <= stock_bbox.max_y {
                result.push(Point::new(x - op.anchor.x, y - op.anchor.y));
                y += sy;
            }
            x += sx;
        }
    }

    result
}

fn translated_vertices(vertices: &[Point], translation: Point) -> Vec<Point> {
    vertices.iter().map(|&p| p.add(translation)).collect()
}

fn polygon_inside_stock(stock: &[Point], stock_bbox: &BBox, poly: &[Point], poly_bbox: &BBox) -> bool {
    if poly_bbox.min_x < stock_bbox.min_x - EPS
        || poly_bbox.max_x > stock_bbox.max_x + EPS
        || poly_bbox.min_y < stock_bbox.min_y - EPS
        || poly_bbox.max_y > stock_bbox.max_y + EPS
    {
        return false;
    }
    for &p in poly {
        if !point_in_polygon(stock, p) {
            return false;
        }
    }
    true
}

fn try_place(
    stock_poly: &[Point],
    stock_bbox: &BBox,
    stock_area: f64,
    state: &StockState,
    op: &OrientedPiece,
    np: usize,
) -> Option<(Point, Vec<Point>, BBox)> {
    let candidates = candidate_positions(stock_bbox, stock_area, state, op, np);
    let mut neighbor_ids = Vec::new();
    for t in candidates {
        let verts = translated_vertices(&op.vertices, t);
        let bbox = BBox::from_points(&verts);
        if !polygon_inside_stock(stock_poly, stock_bbox, &verts, &bbox) {
            continue;
        }
        state.grid.query_indices(&bbox, &mut neighbor_ids);
        let mut ok = true;
        for &idx in &neighbor_ids {
            let other = &state.placed[idx];
            let dx = other.center.x - (t.x);
            let dy = other.center.y - (t.y);
            let rr = other.radius + op.radius;
            if dx * dx + dy * dy > rr * rr * 4.0 {
                continue;
            }
            if other.bbox.intersects(&bbox) && polygons_overlap(&verts, &other.vertices) {
                ok = false;
                break;
            }
        }
        if ok {
            return Some((t, verts, bbox));
        }
    }
    None
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock_poly = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_poly.push(Point::new(x, y));
    }
    stock_poly = normalize_polygon(stock_poly);
    let stock_bbox = BBox::from_points(&stock_poly);
    let stock_area = polygon_area(&stock_poly);

    let mut pieces = Vec::with_capacity(np);
    let mut max_vertices = 0usize;
    for id in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut vertices = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            vertices.push(Point::new(x, y));
        }
        max_vertices = max_vertices.max(pv);
        pieces.push(build_piece(id, vertices));
    }

    let rotations = generate_rotations(np, max_vertices);
    let variants: Vec<Vec<OrientedPiece>> = pieces
        .iter()
        .map(|piece| oriented_variants(piece, &rotations))
        .collect();

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        pieces[b]
            .area
            .partial_cmp(&pieces[a].area)
            .unwrap_or(Ordering::Equal)
            .then_with(|| pieces[b].bbox.width().partial_cmp(&pieces[a].bbox.width()).unwrap_or(Ordering::Equal))
    });

    let mut placements: Vec<Option<PlacedPiece>> = vec![None; np];
    let mut stocks: Vec<StockState> = Vec::new();

    for &piece_idx in &order {
        let piece = &pieces[piece_idx];
        let mut placed_result: Option<PlacedPiece> = None;

        'search: for stock_idx in 0..=stocks.len() {
            let is_new_stock = stock_idx == stocks.len();
            if is_new_stock {
                stocks.push(StockState {
                    placed: Vec::new(),
                    grid: SpatialGrid::new(stock_bbox),
                });
            }
            let state = &stocks[stock_idx];

            let mut best: Option<(f64, Point, Vec<Point>, BBox, f64)> = None;
            for op in &variants[piece_idx] {
                if op.bbox.width() > stock_bbox.width() + EPS || op.bbox.height() > stock_bbox.height() + EPS {
                    continue;
                }
                if let Some((translation, verts, bbox)) =
                    try_place(&stock_poly, &stock_bbox, stock_area, state, op, np)
                {
                    let score = bbox.max_y * 1_000_000.0 + bbox.max_x - op.area * 1e-3;
                    if best.as_ref().map_or(true, |b| score < b.0) {
                        best = Some((score, translation, verts, bbox, op.rotation));
                    }
                }
            }

            if let Some((_, translation, verts, bbox, rotation)) = best {
                let center = Point::new(
                    0.5 * (bbox.min_x + bbox.max_x),
                    0.5 * (bbox.min_y + bbox.max_y),
                );
                let placed = PlacedPiece { stock_idx, translation, rotation, vertices: verts, bbox, center, radius: piece.radius };
                placed_result = Some(placed);
                break 'search;
            }

            if is_new_stock {
                let _ = stocks.pop();
            }
        }

        let placed = placed_result.expect("every piece should fit in a fresh stock");
        let stock_idx = placed.stock_idx;
        placements[piece_idx] = Some(placed.clone());
        let state = &mut stocks[stock_idx];
        let idx = state.placed.len();
        state.grid.insert(idx, &placed.bbox);
        state.placed.push(placed);
    }

    println!("{}", stocks.len());
    for piece_id in 0..np {
        let placed = placements[piece_id].as_ref().unwrap();
        println!(
            "{} {:.10} {:.10} {:.10}",
            placed.stock_idx, placed.translation.x, placed.translation.y, placed.rotation
        );
    }
}
