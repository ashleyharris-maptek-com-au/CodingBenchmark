use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let state = if seed == 0 { 0x9E3779B97F4A7C15 } else { seed };
        Self { state }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }

    fn next_f64(&mut self) -> f64 {
        const SCALE: f64 = (1u64 << 53) as f64;
        ((self.next_u64() >> 11) as f64) / SCALE
    }
}

fn bfs_component(
    start: usize,
    adj: &[Vec<usize>],
    seen: &mut [bool],
    comp_id: usize,
    node_comp: &mut [usize],
) -> Vec<usize> {
    let mut q = VecDeque::new();
    let mut nodes = Vec::new();
    seen[start] = true;
    node_comp[start] = comp_id;
    q.push_back(start);
    while let Some(u) = q.pop_front() {
        nodes.push(u);
        for &v in &adj[u] {
            if !seen[v] {
                seen[v] = true;
                node_comp[v] = comp_id;
                q.push_back(v);
            }
        }
    }
    nodes
}

fn bfs_farthest(start: usize, adj: &[Vec<usize>], allowed: &[bool]) -> (usize, Vec<i32>) {
    let n = adj.len();
    let mut dist = vec![-1; n];
    let mut q = VecDeque::new();
    dist[start] = 0;
    q.push_back(start);
    let mut best = start;
    while let Some(u) = q.pop_front() {
        if dist[u] > dist[best] {
            best = u;
        }
        for &v in &adj[u] {
            if allowed[v] && dist[v] == -1 {
                dist[v] = dist[u] + 1;
                q.push_back(v);
            }
        }
    }
    (best, dist)
}

fn tree_like_score(component: &[usize], degrees: &[usize], undirected_edges: usize) -> bool {
    if component.len() <= 2 {
        return true;
    }
    let n = component.len();
    let max_edges = n.saturating_sub(1) + n / 8 + 1;
    let leaves = component.iter().filter(|&&u| degrees[u] <= 1).count();
    undirected_edges <= max_edges && leaves * 3 >= n
}

fn order_component(component: &[usize], adj: &[Vec<usize>], degrees: &[usize]) -> Vec<usize> {
    if component.len() <= 2 {
        return component.to_vec();
    }
    let n = adj.len();
    let mut allowed = vec![false; n];
    for &u in component {
        allowed[u] = true;
    }

    let mut seed = component[0];
    for &u in component {
        if degrees[u] > degrees[seed] {
            seed = u;
        }
    }
    let (a, _) = bfs_farthest(seed, adj, &allowed);
    let (b, dist_a) = bfs_farthest(a, adj, &allowed);
    let (_, dist_b) = bfs_farthest(b, adj, &allowed);

    let mut ordered = component.to_vec();
    ordered.sort_by(|&u, &v| {
        let ku = (
            dist_a[u] + dist_b[u],
            dist_a[u].max(dist_b[u]),
            -(degrees[u] as i32),
            u as i32,
        );
        let kv = (
            dist_a[v] + dist_b[v],
            dist_a[v].max(dist_b[v]),
            -(degrees[v] as i32),
            v as i32,
        );
        ku.cmp(&kv)
    });
    ordered
}

fn place_path_like(
    component: &[usize],
    adj: &[Vec<usize>],
    degrees: &[usize],
    coords: &mut [(f64, f64)],
    center_x: f64,
    center_y: f64,
    scale: f64,
) {
    let ordered = order_component(component, adj, degrees);
    let n = ordered.len().max(1);
    let step = if n <= 1 {
        0.0
    } else {
        (2.0 * scale) / (n as f64 - 1.0)
    };
    for (i, &u) in ordered.iter().enumerate() {
        let x = center_x - scale + step * i as f64;
        let layer = (degrees[u] as f64).sqrt() * 0.15 * scale / (n as f64).sqrt().max(1.0);
        let parity = if i % 2 == 0 { -1.0 } else { 1.0 };
        coords[u] = (x, center_y + parity * layer);
    }
}

fn place_circle(
    component: &[usize],
    order: &[usize],
    coords: &mut [(f64, f64)],
    center_x: f64,
    center_y: f64,
    radius: f64,
) {
    let n = component.len().max(1);
    if n == 1 {
        coords[component[0]] = (center_x, center_y);
        return;
    }
    for (i, &u) in order.iter().enumerate() {
        let angle = 2.0 * PI * (i as f64) / (n as f64);
        coords[u] = (center_x + radius * angle.cos(), center_y + radius * angle.sin());
    }
}

fn place_spiral(
    component: &[usize],
    order: &[usize],
    degrees: &[usize],
    coords: &mut [(f64, f64)],
    center_x: f64,
    center_y: f64,
    scale: f64,
) {
    let n = component.len().max(1);
    for (i, &u) in order.iter().enumerate() {
        let t = (i as f64 + 1.0) / (n as f64);
        let angle = 6.0 * PI * t + (u as f64 * 0.017);
        let r = scale * (0.15 + 0.85 * t.sqrt());
        let wobble = 1.0 + 0.08 * (degrees[u] as f64).ln_1p();
        coords[u] = (
            center_x + r * angle.cos() * wobble,
            center_y + r * angle.sin() / wobble,
        );
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => return,
    };
    let m: usize = it.next().and_then(|v| v.parse().ok()).unwrap_or(0);

    if n == 0 {
        return;
    }

    let mut adj = vec![Vec::<usize>::new(); n];
    let mut degrees = vec![0usize; n];
    let mut non_loop_edges = 0usize;

    for _ in 0..m {
        let Some(us) = it.next() else { break };
        let Some(vs) = it.next() else { break };
        let mut u: usize = us.parse().unwrap_or(0);
        let mut v: usize = vs.parse().unwrap_or(0);
        if u >= n {
            u %= n;
        }
        if v >= n {
            v %= n;
        }
        if u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
        degrees[u] += 1;
        degrees[v] += 1;
        non_loop_edges += 1;
    }

    let mut seen = vec![false; n];
    let mut node_comp = vec![0usize; n];
    let mut components: Vec<Vec<usize>> = Vec::new();
    for s in 0..n {
        if !seen[s] {
            let comp_id = components.len();
            let comp = bfs_component(s, &adj, &mut seen, comp_id, &mut node_comp);
            components.push(comp);
        }
    }

    let mut comp_edges = vec![0usize; components.len()];
    for u in 0..n {
        for &v in &adj[u] {
            if u < v && node_comp[u] == node_comp[v] {
                comp_edges[node_comp[u]] += 1;
            }
        }
    }

    let mut coords = vec![(0.0f64, 0.0f64); n];

    if components.len() == n {
        let side = (n as f64).sqrt().ceil() as usize;
        let spacing = 10.0;
        for i in 0..n {
            let row = i / side;
            let col = i % side;
            coords[i] = (col as f64 * spacing, row as f64 * spacing);
        }
    } else {
        let mut comp_indices: Vec<usize> = (0..components.len()).collect();
        comp_indices.sort_by_key(|&ci| std::cmp::Reverse(components[ci].len()));

        let huge = n >= 50_000 || non_loop_edges >= 400_000;
        let large = n >= 4_000 || non_loop_edges >= 80_000;
        let medium = n >= 200 || non_loop_edges >= 2_000;

        let comp_count = components.len().max(1);
        let ring_radius = 80.0 * (comp_count as f64).sqrt();
        let mut rng = XorShift64::new((n as u64).wrapping_mul(0x9E3779B97F4A7C15) ^ non_loop_edges as u64);

        for (rank, &ci) in comp_indices.iter().enumerate() {
            let comp = &components[ci];
            let k = comp.len();
            if k == 1 {
                let angle = 2.0 * PI * (rank as f64) / (comp_count as f64);
                coords[comp[0]] = (
                    ring_radius * angle.cos() + rank as f64 * 0.001,
                    ring_radius * angle.sin(),
                );
                continue;
            }

            let angle = 2.0 * PI * (rank as f64) / (comp_count as f64);
            let center_x = ring_radius * angle.cos();
            let center_y = ring_radius * angle.sin();
            let local_scale = 4.0 + (k as f64).sqrt() * 1.8;

            if tree_like_score(comp, &degrees, comp_edges[ci]) && !huge {
                place_path_like(comp, &adj, &degrees, &mut coords, center_x, center_y, local_scale);
                continue;
            }

            let order = order_component(comp, &adj, &degrees);

            if huge {
                let side = (k as f64).sqrt().ceil() as usize;
                let spacing = 2.5;
                for (idx, &u) in order.iter().enumerate() {
                    let row = idx / side;
                    let col = idx % side;
                    let jitter_x = (rng.next_f64() - 0.5) * 0.2;
                    let jitter_y = (rng.next_f64() - 0.5) * 0.2;
                    coords[u] = (
                        center_x + (col as f64 - side as f64 * 0.5) * spacing + jitter_x,
                        center_y + (row as f64 - side as f64 * 0.5) * spacing + jitter_y,
                    );
                }
            } else if large {
                let radius = local_scale + (k as f64) * 0.08;
                place_circle(comp, &order, &mut coords, center_x, center_y, radius);
            } else if medium {
                place_spiral(comp, &order, &degrees, &mut coords, center_x, center_y, local_scale);
            } else {
                let radius = 8.0 + (k as f64) * 0.6;
                place_circle(comp, &order, &mut coords, center_x, center_y, radius);
            }
        }
    }

    let mut out = String::with_capacity(n * 32);
    for (x, y) in coords {
        out.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    print!("{out}");
}
