using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    private const double G = 6.67430e-11;
    private const double Day = 86400.0;
    private const double Au = 149597870700.0;
    private const double EarthRadius = 6371000.0;
    private const double InterceptTolerance = 1_000_000.0;
    private const double DefaultShipMass = 2000.0;

    public static void Main()
    {
        var scanner = new FastScanner(Console.In);
        int numBodies = scanner.NextInt();
        double warningDays = scanner.NextDouble();
        double impactorMass = scanner.NextDouble();
        double deltaVBudget = scanner.NextDouble();
        double asteroidMass = scanner.NextDouble();
        double t0UnixUtc = scanner.NextDouble();

        var bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            string name = scanner.Next();
            double mass = scanner.NextDouble();
            var pos = new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble());
            var vel = new Vec3(scanner.NextDouble(), scanner.NextDouble(), scanner.NextDouble());
            bodies.Add(new Body(name, mass, pos, vel));
        }

        string bspPath = scanner.Next() ?? string.Empty;

        var earth = bodies.FirstOrDefault(b => string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase));
        var asteroid = bodies.FirstOrDefault(b => string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase));
        if (earth == null || asteroid == null)
        {
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        var environment = BuildEnvironment(bodies, earth, asteroid);
        var mission = SolveMission(environment, warningDays, deltaVBudget, impactorMass, asteroidMass);

        Console.WriteLine(mission.Burns.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var burn in mission.Burns)
        {
            Console.WriteLine(string.Format(
                CultureInfo.InvariantCulture,
                "{0:R} {1:R} {2:R} {3:R}",
                burn.TimeDays,
                burn.DeltaV.X,
                burn.DeltaV.Y,
                burn.DeltaV.Z));
        }
    }

    private static EnvironmentModel BuildEnvironment(List<Body> bodies, Body earth, Body asteroid)
    {
        var gravBodies = bodies
            .Where(b => !string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase))
            .Select(b => new GravBody(b.Name, b.Mass, b.Position, b.Velocity))
            .ToList();
        return new EnvironmentModel(gravBodies, earth, asteroid);
    }

    private static Mission SolveMission(EnvironmentModel env, double warningDays, double deltaVBudget, double impactorMass, double asteroidMass)
    {
        double impactTime = EstimateImpactTimeDays(env, warningDays);
        if (double.IsNaN(impactTime) || impactTime < 2.0)
        {
            impactTime = Math.Max(2.0, warningDays * 0.8);
        }

        var best = SearchDirectIntercept(env, warningDays, impactTime, deltaVBudget, impactorMass, asteroidMass);
        if (best != null)
        {
            return best;
        }

        return BuildFallbackMission(env, warningDays, impactTime, deltaVBudget);
    }

    private static double EstimateImpactTimeDays(EnvironmentModel env, double warningDays)
    {
        double horizonDays = Math.Max(warningDays + 3650.0, warningDays * 2.0);
        double step = horizonDays <= 2000.0 ? 0.25 : 1.0;
        double minDistance = double.MaxValue;
        double bestTime = double.NaN;

        for (double t = 0.0; t <= horizonDays; t += step)
        {
            var earthState = env.PropagateBody(env.Earth, t * Day);
            var asteroidState = env.PropagateAsteroid(t * Day);
            double d = (asteroidState.Position - earthState.Position).Length();
            if (d < minDistance)
            {
                minDistance = d;
                bestTime = t;
            }
            if (d <= EarthRadius)
            {
                return t;
            }
        }

        return bestTime;
    }

    private static Mission SearchDirectIntercept(EnvironmentModel env, double warningDays, double impactTime, double deltaVBudget, double impactorMass, double asteroidMass)
    {
        double shipMass = Math.Max(500.0, Math.Min(impactorMass, DefaultShipMass));
        double desiredAsteroidDelta = Math.Max(0.001, 2.0 * EarthRadius / Math.Max(impactTime * Day, 1.0));
        double momentumNeeded = asteroidMass * desiredAsteroidDelta;
        double requiredRelativeSpeed = momentumNeeded / Math.Max(shipMass, 1.0);
        requiredRelativeSpeed = Math.Min(requiredRelativeSpeed, 25000.0);

        var bestCandidate = (Mission)null;
        double bestScore = double.MaxValue;

        int launchSamples = warningDays <= 30 ? 25 : 40;
        double launchStep = launchSamples <= 1 ? warningDays : warningDays / (launchSamples - 1);
        double earliestIntercept = Math.Min(impactTime - 0.5, warningDays + 0.5);
        if (earliestIntercept <= 0.5)
        {
            earliestIntercept = Math.Max(1.0, impactTime * 0.25);
        }

        for (int li = 0; li < launchSamples; li++)
        {
            double launchTime = Math.Min(warningDays, li * launchStep);
            var earthState = env.PropagateBody(env.Earth, launchTime * Day);

            int interceptSamples = 80;
            double startIntercept = Math.Max(launchTime + 1.0, earliestIntercept);
            double endIntercept = Math.Max(startIntercept + 0.5, impactTime - 0.1);
            double interceptStep = (endIntercept - startIntercept) / interceptSamples;

            for (int ii = 0; ii <= interceptSamples; ii++)
            {
                double interceptTime = startIntercept + ii * interceptStep;
                double tofDays = interceptTime - launchTime;
                if (tofDays <= 0.1)
                {
                    continue;
                }

                var asteroidState = env.PropagateAsteroid(interceptTime * Day);
                Vec3 requiredVelocity = LambertApproximation(earthState.Position, asteroidState.Position, tofDays * Day, env.SunMu);
                Vec3 launchDv = requiredVelocity - earthState.Velocity;
                double dv = launchDv.Length();
                if (dv > deltaVBudget)
                {
                    continue;
                }

                var shipState = new State(earthState.Position, earthState.Velocity + launchDv);
                var propagatedShip = env.PropagateShip(shipState, tofDays * Day);
                Vec3 miss = asteroidState.Position - propagatedShip.Position;
                double missDistance = miss.Length();
                if (missDistance > InterceptTolerance)
                {
                    Vec3 correctionDv = miss / Math.Max(tofDays * Day * 0.5, 1.0);
                    double totalDv = dv + correctionDv.Length();
                    if (totalDv > deltaVBudget)
                    {
                        continue;
                    }

                    var mission = new Mission();
                    mission.Burns.Add(new Burn(launchTime, launchDv));
                    mission.Burns.Add(new Burn((launchTime + interceptTime) * 0.5, correctionDv));
                    mission.Burns.Add(new Burn(interceptTime, Vec3.Zero));

                    double relSpeed = (propagatedShip.Velocity + correctionDv - asteroidState.Velocity).Length();
                    if (relSpeed < requiredRelativeSpeed * 0.25)
                    {
                        continue;
                    }

                    double score = missDistance + totalDv * 1000.0 - relSpeed;
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestCandidate = mission;
                    }
                }
                else
                {
                    double relSpeed = (propagatedShip.Velocity - asteroidState.Velocity).Length();
                    if (relSpeed < requiredRelativeSpeed * 0.25)
                    {
                        continue;
                    }

                    var mission = new Mission();
                    mission.Burns.Add(new Burn(launchTime, launchDv));
                    mission.Burns.Add(new Burn(interceptTime, Vec3.Zero));

                    double score = missDistance + dv * 1000.0 - relSpeed;
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestCandidate = mission;
                    }
                }
            }
        }

        return bestCandidate;
    }

    private static Mission BuildFallbackMission(EnvironmentModel env, double warningDays, double impactTime, double deltaVBudget)
    {
        double launchTime = Math.Min(warningDays, Math.Max(0.0, warningDays * 0.1));
        double interceptTime = Math.Max(launchTime + 5.0, impactTime * 0.8);

        var earthState = env.PropagateBody(env.Earth, launchTime * Day);
        var asteroidState = env.PropagateAsteroid(interceptTime * Day);
        Vec3 transferVelocity = LambertApproximation(earthState.Position, asteroidState.Position, (interceptTime - launchTime) * Day, env.SunMu);
        Vec3 launchDv = transferVelocity - earthState.Velocity;
        if (launchDv.Length() > deltaVBudget)
        {
            launchDv = launchDv.Normalized() * deltaVBudget;
        }

        var mission = new Mission();
        mission.Burns.Add(new Burn(launchTime, launchDv));
        mission.Burns.Add(new Burn(interceptTime, Vec3.Zero));
        return mission;
    }

    private static Vec3 LambertApproximation(Vec3 r1, Vec3 r2, double dt, double mu)
    {
        if (dt <= 1.0)
        {
            return (r2 - r1);
        }

        Vec3 chordVelocity = (r2 - r1) / dt;
        double r1Len = r1.Length();
        double r2Len = r2.Length();
        Vec3 radial1 = r1Len > 0 ? r1 / r1Len : Vec3.Zero;
        Vec3 radial2 = r2Len > 0 ? r2 / r2Len : Vec3.Zero;
        double vCirc1 = r1Len > 0 ? Math.Sqrt(mu / r1Len) : 0.0;
        double vCirc2 = r2Len > 0 ? Math.Sqrt(mu / r2Len) : 0.0;
        Vec3 planeNormal = Vec3.Cross(r1, r2).Normalized();
        Vec3 tangential1 = Vec3.Cross(planeNormal, radial1).Normalized();
        Vec3 tangential2 = Vec3.Cross(planeNormal, radial2).Normalized();
        Vec3 circularBlend = (tangential1 * vCirc1 + tangential2 * vCirc2) * 0.5;
        return chordVelocity + circularBlend * 0.15;
    }

    private readonly record struct Body(string Name, double Mass, Vec3 Position, Vec3 Velocity);
    private readonly record struct GravBody(string Name, double Mass, Vec3 Position, Vec3 Velocity);
    private readonly record struct State(Vec3 Position, Vec3 Velocity);
    private readonly record struct Burn(double TimeDays, Vec3 DeltaV);

    private sealed class Mission
    {
        public List<Burn> Burns { get; } = new List<Burn>();
    }

    private sealed class EnvironmentModel
    {
        private readonly List<GravBody> _bodies;
        private readonly GravBody _sun;

        public Body Earth { get; }
        public Body Asteroid { get; }
        public double SunMu => _sun.Mass * G;

        public EnvironmentModel(List<GravBody> bodies, Body earth, Body asteroid)
        {
            _bodies = bodies;
            Earth = earth;
            Asteroid = asteroid;
            _sun = bodies.FirstOrDefault(b => string.Equals(b.Name, "Sun", StringComparison.OrdinalIgnoreCase));
            if (_sun.Mass <= 0.0)
            {
                _sun = new GravBody("Sun", 1.98847e30, Vec3.Zero, Vec3.Zero);
                _bodies.Add(_sun);
            }
        }

        public State PropagateAsteroid(double seconds)
        {
            return PropagateState(new State(Asteroid.Position, Asteroid.Velocity), seconds, includeEarth: true);
        }

        public State PropagateBody(Body body, double seconds)
        {
            return PropagateState(new State(body.Position, body.Velocity), seconds, includeEarth: true);
        }

        public State PropagateShip(State initial, double seconds)
        {
            return PropagateState(initial, seconds, includeEarth: true);
        }

        private State PropagateState(State initial, double seconds, bool includeEarth)
        {
            int steps = Math.Max(8, (int)Math.Ceiling(Math.Abs(seconds) / (Day * 0.5)));
            double dt = seconds / steps;
            var state = initial;
            for (int i = 0; i < steps; i++)
            {
                state = Rk4(state, dt, includeEarth);
            }
            return state;
        }

        private State Rk4(State s, double dt, bool includeEarth)
        {
            var a1 = Acceleration(s.Position, 0.0, includeEarth);
            var k1v = a1 * dt;
            var k1x = s.Velocity * dt;

            var a2 = Acceleration(s.Position + k1x * 0.5, 0.0, includeEarth);
            var k2v = a2 * dt;
            var k2x = (s.Velocity + k1v * 0.5) * dt;

            var a3 = Acceleration(s.Position + k2x * 0.5, 0.0, includeEarth);
            var k3v = a3 * dt;
            var k3x = (s.Velocity + k2v * 0.5) * dt;

            var a4 = Acceleration(s.Position + k3x, 0.0, includeEarth);
            var k4v = a4 * dt;
            var k4x = (s.Velocity + k3v) * dt;

            var newPos = s.Position + (k1x + k2x * 2.0 + k3x * 2.0 + k4x) / 6.0;
            var newVel = s.Velocity + (k1v + k2v * 2.0 + k3v * 2.0 + k4v) / 6.0;
            return new State(newPos, newVel);
        }

        private Vec3 Acceleration(Vec3 position, double seconds, bool includeEarth)
        {
            Vec3 acc = Vec3.Zero;
            foreach (var body in _bodies)
            {
                if (!includeEarth && string.Equals(body.Name, "Earth", StringComparison.OrdinalIgnoreCase))
                {
                    continue;
                }

                Vec3 bodyPos = body.Position + body.Velocity * seconds;
                Vec3 r = bodyPos - position;
                double distSq = r.LengthSquared() + 1.0;
                double invDist = 1.0 / Math.Sqrt(distSq);
                double invDist3 = invDist * invDist * invDist;
                acc += r * (G * body.Mass * invDist3);
            }
            return acc;
        }
    }

    private readonly record struct Vec3(double X, double Y, double Z)
    {
        public static readonly Vec3 Zero = new Vec3(0.0, 0.0, 0.0);

        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double LengthSquared() => X * X + Y * Y + Z * Z;

        public Vec3 Normalized()
        {
            double len = Length();
            return len > 0.0 ? this / len : Zero;
        }

        public static Vec3 Cross(Vec3 a, Vec3 b)
        {
            return new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X);
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
    }

    private sealed class FastScanner
    {
        private readonly TextReader _reader;
        private readonly Queue<string> _tokens = new Queue<string>();

        public FastScanner(TextReader reader)
        {
            _reader = reader;
        }

        public string Next()
        {
            while (_tokens.Count == 0)
            {
                string line = _reader.ReadLine();
                if (line == null)
                {
                    return null;
                }

                foreach (var token in line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries))
                {
                    _tokens.Enqueue(token);
                }
            }

            return _tokens.Dequeue();
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);
    }
}
