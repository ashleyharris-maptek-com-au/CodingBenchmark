def autopilot_step(state: dict, dt: float) -> dict:
    import math

    global _AP_STATE
    try:
        mem = _AP_STATE
    except NameError:
        mem = _AP_STATE = {
            "mode": "approach",
            "ga_count": 0,
            "last_ra": None,
            "filtered_loc": 0.0,
            "filtered_gs": 0.0,
            "filtered_speed_err": 0.0,
            "int_loc": 0.0,
            "int_gs": 0.0,
            "int_speed": 0.0,
            "flare_timer": 0.0,
            "rollout_timer": 0.0,
            "last_cmd": {},
        }

    def clamp(x, lo, hi):
        return lo if x < lo else hi if x > hi else x

    def pick_median(values):
        values = sorted(values)
        n = len(values)
        if n == 0:
            return None
        if n % 2:
            return values[n // 2]
        return 0.5 * (values[n // 2 - 1] + values[n // 2])

    def voted(prefix, suffix, count=3, valid_suffix='_valid'):
        vals = []
        for i in range(1, count + 1):
            key = f'{prefix}_{i}{suffix}'
            valid_key = f'{prefix}_{i}{valid_suffix}' if valid_suffix else None
            if valid_key is None or state.get(valid_key, True):
                v = state.get(key)
                if isinstance(v, (int, float)):
                    vals.append(float(v))
        return pick_median(vals)

    def fallback_chain(*values):
        for v in values:
            if isinstance(v, (int, float)) and math.isfinite(v):
                return float(v)
        return 0.0

    def angle_diff(a, b):
        d = (a - b + 180.0) % 360.0 - 180.0
        return d

    radio_alt = state.get('radio_alt_ft')
    if not isinstance(radio_alt, (int, float)) or not math.isfinite(radio_alt):
        radio_alt = None
    baro_alt = voted('baro_alt', '_ft')
    gps_alt = voted('gps', '_alt_ft')
    alt = fallback_chain(radio_alt if radio_alt is not None and radio_alt < 2500 else None, baro_alt, gps_alt)

    loc = voted('localizer', '_dots')
    gs = voted('glideslope', '_dots')
    if loc is None:
        loc = state.get('target_loc_dots', 0.0)
    if gs is None:
        gs = state.get('target_gs_dots', 0.0)
    loc = clamp(float(loc), -2.5, 2.5)
    gs = clamp(float(gs), -2.5, 2.5)

    ias = voted('ias', '_kt')
    gs_kt = voted('irs', '_gs_kt')
    if ias is None:
        ias = fallback_chain(gs_kt, state.get('groundspeed_kt'), state.get('target_vref_kt', 140.0))
    vs = voted('irs', '_vs_fpm')
    pitch = voted('irs', '_pitch_deg')
    roll = voted('irs', '_roll_deg')
    heading = voted('irs', '_heading_deg')
    track = voted('irs', '_track_deg')
    aoa = voted('aoa', '_deg')

    pitch = fallback_chain(pitch, 0.0)
    roll = fallback_chain(roll, 0.0)
    heading = fallback_chain(heading, state.get('target_hdg_deg', 0.0))
    track = fallback_chain(track, heading)
    vs = fallback_chain(vs, -700.0)
    aoa = fallback_chain(aoa, 4.0)

    target_vref = fallback_chain(state.get('target_vref_kt'), 137.0)
    target_ias = fallback_chain(state.get('target_ias_kt'), target_vref + 5.0)
    dme = fallback_chain(state.get('dme_1_nm'), state.get('dme_2_nm'), 10.0)
    on_ground = bool(state.get('on_ground') or state.get('weight_on_wheels'))
    below_dh = bool(state.get('below_decision_height'))
    engine1_running = bool(state.get('engine_1_running', True))
    engine2_running = bool(state.get('engine_2_running', True))
    thrust1 = fallback_chain(state.get('engine_1_thrust_pct'), 50.0) / 100.0
    thrust2 = fallback_chain(state.get('engine_2_thrust_pct'), 50.0) / 100.0
    eng_out = (not engine1_running) or (not engine2_running) or thrust1 < 0.12 or thrust2 < 0.12
    dual_out = (not engine1_running and not engine2_running) or (thrust1 < 0.08 and thrust2 < 0.08)

    alpha = clamp(dt / 1.5 if dt > 0 else 1.0, 0.0, 1.0)
    mem['filtered_loc'] += alpha * (loc - mem['filtered_loc'])
    mem['filtered_gs'] += alpha * (gs - mem['filtered_gs'])

    if radio_alt is not None and mem['last_ra'] is not None and dt > 0:
        ra_rate = (radio_alt - mem['last_ra']) / dt * 60.0
    else:
        ra_rate = vs
    mem['last_ra'] = radio_alt

    flare_alt = 50.0 if radio_alt is not None else 35.0
    approach_alt = radio_alt if radio_alt is not None else alt

    desired_speed = target_vref + 5.0
    if eng_out:
        desired_speed += 5.0
    if dual_out:
        desired_speed += 8.0
    if approach_alt > 1000:
        desired_speed = max(desired_speed, target_ias)
    if approach_alt < 40:
        desired_speed = target_vref

    speed_err = clamp(desired_speed - ias, -40.0, 40.0)
    mem['filtered_speed_err'] += clamp(dt / 2.0 if dt > 0 else 1.0, 0.0, 1.0) * (speed_err - mem['filtered_speed_err'])

    unstable = False
    if 200.0 < approach_alt < 1000.0:
        unstable = (
            abs(mem['filtered_loc']) > 1.3 or
            abs(mem['filtered_gs']) > 1.0 or
            ias > target_vref + 25.0 or
            ias < max(target_vref - 8.0, 90.0) or
            abs(roll) > 20.0 or
            ra_rate < -1400.0
        )
    if below_dh and (abs(mem['filtered_loc']) > 1.8 or abs(mem['filtered_gs']) > 1.5):
        unstable = True
    if approach_alt < 80.0:
        unstable = False
    if dual_out:
        unstable = False

    if mem['mode'] != 'go_around' and not on_ground and unstable and mem['ga_count'] < 10:
        mem['mode'] = 'go_around'
        mem['ga_count'] += 1
        mem['flare_timer'] = 0.0
        mem['int_loc'] = 0.0
        mem['int_gs'] = 0.0
        mem['int_speed'] = 0.0

    if on_ground:
        mem['mode'] = 'rollout'
    elif mem['mode'] != 'go_around' and approach_alt <= flare_alt:
        mem['mode'] = 'flare'
    elif mem['mode'] != 'go_around':
        mem['mode'] = 'approach'

    cmd = {
        'gear': 1.0,
        'flaps': 30.0,
        'speedbrake': 0.0,
        'spoiler': 0.0,
        'autobrake': 0.0,
        'reverse_1': 0.0,
        'reverse_2': 0.0,
    }

    if approach_alt > 1500:
        cmd['flaps'] = 15.0
        cmd['gear'] = 0.0 if dme > 8.0 else 1.0
    elif approach_alt > 800:
        cmd['flaps'] = 20.0
    else:
        cmd['flaps'] = 30.0 if not eng_out else 25.0

    if mem['mode'] == 'go_around':
        cmd.update({
            'throttle_1': 1.0 if engine1_running else 0.0,
            'throttle_2': 1.0 if engine2_running else 0.0,
            'elevator': 0.55,
            'elevator_trim': 0.20,
            'aileron': clamp(-roll * 0.04, -0.35, 0.35),
            'rudder': clamp((0.0 if dual_out else angle_diff(track, heading) * 0.02), -0.3, 0.3),
            'speedbrake': 0.0,
            'gear': 1.0 if approach_alt < 100.0 else 0.0,
            'flaps': 20.0,
        })
        mem['last_cmd'] = cmd
        return cmd

    loc_err = mem['filtered_loc']
    gs_err = mem['filtered_gs']

    loc_gain = 0.22 if approach_alt > 500 else 0.18
    gs_gain = 0.20 if approach_alt > 500 else 0.16
    mem['int_loc'] = clamp(mem['int_loc'] + (-loc_err) * dt * 0.08, -0.25, 0.25)
    mem['int_gs'] = clamp(mem['int_gs'] + (-gs_err) * dt * 0.06, -0.25, 0.25)
    mem['int_speed'] = clamp(mem['int_speed'] + mem['filtered_speed_err'] * dt * 0.01, -0.2, 0.2)

    crab = angle_diff(track, heading)
    hdg_to_target = angle_diff(state.get('target_hdg_deg', heading), heading)

    aileron = (
        -roll * 0.045
        - loc_err * loc_gain
        + mem['int_loc']
        + hdg_to_target * 0.01
    )
    rudder = crab * 0.03 - loc_err * 0.04
    if eng_out and not dual_out:
        asym = thrust1 - thrust2
        rudder += clamp(asym * 0.25, -0.3, 0.3)
        aileron += clamp(asym * 0.08, -0.12, 0.12)

    desired_vs = -700.0
    if approach_alt > 1200:
        desired_vs = -850.0
    elif approach_alt < 300:
        desired_vs = -600.0
    if gs_err > 0.3:
        desired_vs -= 220.0 * gs_err
    elif gs_err < -0.3:
        desired_vs -= 160.0 * gs_err
    vs_err = clamp(desired_vs - vs, -1200.0, 1200.0)

    elevator = (
        -pitch * 0.06
        - gs_err * gs_gain
        + mem['int_gs']
        + vs_err * 0.00045
        + clamp(aoa - 5.0, -4.0, 8.0) * 0.01
    )

    throttle_bias = 0.52 if not dual_out else 0.0
    throttle = throttle_bias + mem['filtered_speed_err'] * 0.018 + mem['int_speed'] - vs_err * 0.00008
    if approach_alt < 100:
        throttle += 0.03
    if eng_out and not dual_out:
        throttle += 0.06
    throttle = clamp(throttle, 0.0, 1.0)

    if dual_out:
        throttle1 = 0.0
        throttle2 = 0.0
    else:
        throttle1 = throttle if engine1_running else 0.0
        throttle2 = throttle if engine2_running else 0.0

    if mem['mode'] == 'flare':
        mem['flare_timer'] += max(dt, 0.0)
        flare_factor = clamp((flare_alt - max(0.0, approach_alt)) / max(flare_alt, 1.0), 0.0, 1.0)
        target_sink = -120.0 - 120.0 * (1.0 - flare_factor)
        flare_vs_err = clamp(target_sink - ra_rate, -800.0, 800.0)
        elevator = 0.10 + flare_factor * 0.18 - pitch * 0.03 + flare_vs_err * 0.00055
        throttle1 = 0.0 if not dual_out else throttle1
        throttle2 = 0.0 if not dual_out else throttle2
        if approach_alt < 15.0:
            elevator += 0.05
        cmd['speedbrake'] = 0.0

    if mem['mode'] == 'rollout':
        mem['rollout_timer'] += max(dt, 0.0)
        gs_now = fallback_chain(state.get('groundspeed_kt'), gs_kt, ias)
        cmd.update({
            'spoiler': 1.0,
            'speedbrake': 1.0,
            'autobrake': 1.0 if gs_now > 80.0 else 0.6 if gs_now > 40.0 else 0.25,
            'reverse_1': 1.0 if engine1_running and gs_now > 60.0 else 0.4 if engine1_running and gs_now > 20.0 else 0.0,
            'reverse_2': 1.0 if engine2_running and gs_now > 60.0 else 0.4 if engine2_running and gs_now > 20.0 else 0.0,
            'throttle_1': 0.0,
            'throttle_2': 0.0,
            'elevator': clamp(0.02 - pitch * 0.02, -0.15, 0.15),
            'elevator_trim': 0.0,
            'aileron': clamp(-roll * 0.04 - loc_err * 0.08, -0.4, 0.4),
            'rudder': clamp(crab * 0.03 - loc_err * 0.05, -0.4, 0.4),
        })
        mem['last_cmd'] = cmd
        return cmd

    cmd.update({
        'elevator': clamp(elevator, -1.0, 1.0),
        'aileron': clamp(aileron, -1.0, 1.0),
        'rudder': clamp(rudder, -1.0, 1.0),
        'throttle_1': clamp(throttle1, 0.0, 1.0),
        'throttle_2': clamp(throttle2, 0.0, 1.0),
        'elevator_trim': clamp(pitch * 0.015 + gs_err * 0.02, -0.4, 0.4),
    })

    mem['last_cmd'] = cmd
    return cmd
