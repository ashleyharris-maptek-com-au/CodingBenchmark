#include <algorithm>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <queue>
#include <unordered_set>
#include <vector>

using namespace std;

struct Move {
    int axis = 0;
    int dir = 1;
};

struct EncodedPath {
    vector<int> parent;
    vector<int> move_axis;
    vector<int> move_dir;
    vector<int> mark;
    int stamp = 1;

    explicit EncodedPath(size_t n = 0)
        : parent(n, -2), move_axis(n, -1), move_dir(n, 0), mark(n, 0) {}

    void reset() {
        ++stamp;
        if (stamp == numeric_limits<int>::max()) {
            fill(mark.begin(), mark.end(), 0);
            stamp = 1;
        }
    }

    void ensure(size_t n) {
        if (parent.size() >= n) {
            return;
        }
        parent.resize(n, -2);
        move_axis.resize(n, -1);
        move_dir.resize(n, 0);
        mark.resize(n, 0);
    }

    bool seen(int idx) const {
        return mark[idx] == stamp;
    }

    void set(int idx, int p, int axis, int dir) {
        mark[idx] = stamp;
        parent[idx] = p;
        move_axis[idx] = axis;
        move_dir[idx] = dir;
    }
};

class SnakeSolver {
public:
    void run() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);

        if (!(cin >> D >> max_turns)) {
            return;
        }

        bounds.resize(D);
        for (int i = 0; i < D; ++i) {
            cin >> bounds[i];
        }

        build_strides();

        cin >> snake_length;
        vector<int> start_head(D);
        for (int i = 0; i < D; ++i) {
            cin >> start_head[i];
        }

        int F = 0;
        cin >> F;
        foods.reserve(F);
        for (int i = 0; i < F; ++i) {
            vector<int> p(D);
            for (int j = 0; j < D; ++j) {
                cin >> p[j];
            }
            foods.push_back(encode(p));
        }

        int O = 0;
        cin >> O;
        obstacle.assign(total_cells(), 0);
        for (int i = 0; i < O; ++i) {
            vector<int> p(D);
            for (int j = 0; j < D; ++j) {
                cin >> p[j];
            }
            obstacle[encode(p)] = 1;
        }

        initialize_snake(start_head);

        visit_mark.assign(total_cells(), 0);
        bfs_dist.assign(total_cells(), -1);
        path_state.ensure(total_cells());

        simulate();
    }

private:
    int D = 0;
    int max_turns = 0;
    int snake_length = 1;
    vector<int> bounds;
    vector<int> strides;
    vector<int> foods;
    vector<unsigned char> obstacle;
    deque<int> snake;
    unordered_set<int> food_set;
    vector<int> visit_mark;
    vector<int> bfs_dist;
    int visit_stamp = 1;
    EncodedPath path_state;

    void build_strides() {
        strides.assign(D, 1);
        for (int i = D - 2; i >= 0; --i) {
            strides[i] = strides[i + 1] * bounds[i + 1];
        }
    }

    int total_cells() const {
        long long total = 1;
        for (int b : bounds) {
            total *= b;
        }
        return static_cast<int>(total);
    }

    int encode(const vector<int>& p) const {
        int idx = 0;
        for (int i = 0; i < D; ++i) {
            idx += p[i] * strides[i];
        }
        return idx;
    }

    bool neighbor(int idx, int axis, int dir, int& out) const {
        int coord = (idx / strides[axis]) % bounds[axis];
        int next = coord + dir;
        if (next < 0 || next >= bounds[axis]) {
            return false;
        }
        out = idx + dir * strides[axis];
        return true;
    }

    void initialize_snake(const vector<int>& start_head) {
        int head = encode(start_head);
        snake.clear();
        snake.push_front(head);

        vector<int> cur = start_head;
        for (int segment = 1; segment < snake_length; ++segment) {
            bool placed = false;
            for (int axis = D - 1; axis >= 0 && !placed; --axis) {
                if (cur[axis] + 1 < bounds[axis]) {
                    ++cur[axis];
                    snake.push_back(encode(cur));
                    placed = true;
                } else if (cur[axis] - 1 >= 0) {
                    --cur[axis];
                    snake.push_back(encode(cur));
                    placed = true;
                }
            }
            if (!placed) {
                snake.push_back(head);
            }
        }

        food_set.clear();
        for (int f : foods) {
            food_set.insert(f);
        }
    }

    vector<unsigned char> occupancy_map(int tail_to_ignore = -1) const {
        vector<unsigned char> occ(total_cells(), 0);
        for (int cell : snake) {
            occ[cell] = 1;
        }
        if (tail_to_ignore != -1) {
            occ[tail_to_ignore] = 0;
        }
        return occ;
    }

    bool is_free_cell(int idx, const vector<unsigned char>& occ) const {
        return !obstacle[idx] && !occ[idx];
    }

    int count_free_neighbors(int idx, const vector<unsigned char>& occ) const {
        int count = 0;
        for (int axis = 0; axis < D; ++axis) {
            int nxt = -1;
            if (neighbor(idx, axis, -1, nxt) && is_free_cell(nxt, occ)) {
                ++count;
            }
            if (neighbor(idx, axis, 1, nxt) && is_free_cell(nxt, occ)) {
                ++count;
            }
        }
        return count;
    }

    int flood_fill_size(int start, const vector<unsigned char>& occ, int cap) {
        if (!is_free_cell(start, occ)) {
            return 0;
        }
        ++visit_stamp;
        if (visit_stamp == numeric_limits<int>::max()) {
            fill(visit_mark.begin(), visit_mark.end(), 0);
            visit_stamp = 1;
        }

        queue<int> q;
        q.push(start);
        visit_mark[start] = visit_stamp;
        int count = 0;

        while (!q.empty() && count < cap) {
            int cur = q.front();
            q.pop();
            ++count;
            for (int axis = 0; axis < D; ++axis) {
                int nxt = -1;
                if (neighbor(cur, axis, -1, nxt) && visit_mark[nxt] != visit_stamp &&
                    is_free_cell(nxt, occ)) {
                    visit_mark[nxt] = visit_stamp;
                    q.push(nxt);
                }
                if (neighbor(cur, axis, 1, nxt) && visit_mark[nxt] != visit_stamp &&
                    is_free_cell(nxt, occ)) {
                    visit_mark[nxt] = visit_stamp;
                    q.push(nxt);
                }
            }
        }
        return count;
    }

    bool bfs_to_any_food(const vector<unsigned char>& occ, Move& out_move) {
        const int head = snake.front();
        path_state.reset();
        fill(bfs_dist.begin(), bfs_dist.end(), -1);

        queue<int> q;
        q.push(head);
        path_state.set(head, -1, -1, 0);
        bfs_dist[head] = 0;

        int found = -1;

        while (!q.empty()) {
            int cur = q.front();
            q.pop();

            if (cur != head && food_set.find(cur) != food_set.end()) {
                found = cur;
                break;
            }

            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    int nxt = -1;
                    if (!neighbor(cur, axis, dir, nxt)) {
                        continue;
                    }
                    if (path_state.seen(nxt) || !is_free_cell(nxt, occ)) {
                        continue;
                    }
                    path_state.set(nxt, cur, axis, dir);
                    bfs_dist[nxt] = bfs_dist[cur] + 1;
                    q.push(nxt);
                }
            }
        }

        if (found == -1) {
            return false;
        }

        int cur = found;
        while (path_state.parent[cur] != head) {
            cur = path_state.parent[cur];
            if (cur == -1) {
                return false;
            }
        }

        out_move.axis = path_state.move_axis[cur];
        out_move.dir = path_state.move_dir[cur];
        return true;
    }

    bool tail_reachable_after_move(int next_head, bool grows) {
        deque<int> next_snake = snake;
        next_snake.push_front(next_head);
        if (!grows) {
            next_snake.pop_back();
        }

        vector<unsigned char> occ(total_cells(), 0);
        for (int cell : next_snake) {
            occ[cell] = 1;
        }
        int tail = next_snake.back();
        occ[tail] = 0;

        ++visit_stamp;
        if (visit_stamp == numeric_limits<int>::max()) {
            fill(visit_mark.begin(), visit_mark.end(), 0);
            visit_stamp = 1;
        }

        queue<int> q;
        q.push(next_snake.front());
        visit_mark[next_snake.front()] = visit_stamp;

        while (!q.empty()) {
            int cur = q.front();
            q.pop();
            if (cur == tail) {
                return true;
            }

            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    int nxt = -1;
                    if (!neighbor(cur, axis, dir, nxt)) {
                        continue;
                    }
                    if (visit_mark[nxt] == visit_stamp || obstacle[nxt] || occ[nxt]) {
                        continue;
                    }
                    visit_mark[nxt] = visit_stamp;
                    q.push(nxt);
                }
            }
        }
        return false;
    }

    Move choose_survival_move() {
        const int head = snake.front();
        const int tail = snake.back();
        vector<unsigned char> occ = occupancy_map(tail);

        struct Candidate {
            int score;
            int area;
            int degree;
            int food_bias;
            Move move;
        };

        Candidate best{numeric_limits<int>::min(), -1, -1, -1, {0, 1}};
        bool found = false;

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                int nxt = -1;
                if (!neighbor(head, axis, dir, nxt)) {
                    continue;
                }
                bool grows = food_set.find(nxt) != food_set.end();
                if (obstacle[nxt]) {
                    continue;
                }
                if (nxt != tail && occ[nxt]) {
                    continue;
                }

                vector<unsigned char> local_occ(total_cells(), 0);
                for (int cell : snake) {
                    local_occ[cell] = 1;
                }
                if (!grows) {
                    local_occ[tail] = 0;
                }
                local_occ[nxt] = 1;
                local_occ[head] = 0;

                int degree = count_free_neighbors(nxt, local_occ);
                int area = flood_fill_size(nxt, local_occ, 50000);
                bool tail_ok = tail_reachable_after_move(nxt, grows);
                int food_bias = (food_set.find(nxt) != food_set.end()) ? 1 : 0;
                int score = area * 8 + degree * 4 + (tail_ok ? 1000000 : 0) + food_bias;

                Candidate cur{score, area, degree, food_bias, {axis, dir}};
                if (!found || cur.score > best.score ||
                    (cur.score == best.score && cur.area > best.area) ||
                    (cur.score == best.score && cur.area == best.area && cur.degree > best.degree)) {
                    best = cur;
                    found = true;
                }
            }
        }

        if (found) {
            return best.move;
        }

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                int nxt = -1;
                if (!neighbor(head, axis, dir, nxt)) {
                    continue;
                }
                if (!obstacle[nxt]) {
                    return {axis, dir};
                }
            }
        }

        return {0, 1};
    }

    Move choose_move() {
        const int tail = snake.back();
        vector<unsigned char> occ = occupancy_map(tail);

        Move food_move;
        if (!food_set.empty() && bfs_to_any_food(occ, food_move)) {
            int nxt = snake.front() + food_move.dir * strides[food_move.axis];
            bool grows = food_set.find(nxt) != food_set.end();
            if (tail_reachable_after_move(nxt, grows)) {
                return food_move;
            }
        }

        return choose_survival_move();
    }

    void apply_move(const Move& mv) {
        int next_head = snake.front() + mv.dir * strides[mv.axis];
        snake.push_front(next_head);
        auto it = food_set.find(next_head);
        bool grows = it != food_set.end();
        if (grows) {
            food_set.erase(it);
        } else {
            snake.pop_back();
        }
    }

    void simulate() {
        for (int turn = 0; turn < max_turns && !food_set.empty(); ++turn) {
            Move mv = choose_move();
            cout << mv.axis << ' ' << mv.dir << '\n';
            apply_move(mv);
        }
    }
};

int main() {
    SnakeSolver solver;
    solver.run();
    return 0;
}
