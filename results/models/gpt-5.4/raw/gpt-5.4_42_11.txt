OpCapability Geometry
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_half = OpConstant %float 0.5
%float_thickness = OpConstant %float 0.02
%float_third = OpConstant %float 0.33333334
%float_two_thirds = OpConstant %float 0.6666667

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_input_arr_gl_PerVertex_in_4 = OpTypePointer Input %arr_gl_PerVertex_in_4
%ptr_input_v4float = OpTypePointer Input %v4float
%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_input_arr_v3float_4 = OpTypePointer Input %arr_v3float_4
%ptr_input_v3float = OpTypePointer Input %v3float

%gl_in = OpVariable %ptr_input_arr_gl_PerVertex_in_4 Input
%worldPos = OpVariable %ptr_input_arr_v3float_4 Input
%normal = OpVariable %ptr_input_arr_v3float_4 Input
%color = OpVariable %ptr_input_arr_v3float_4 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_output_v3float = OpTypePointer Output %v3float

%gl_out = OpVariable %ptr_output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_output_v3float Output

%main = OpFunction %void None %fn
%entry = OpLabel
%glin0_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_0 %int_0
%glin0 = OpLoad %v4float %glin0_ptr
%glin1_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_1 %int_0
%glin1 = OpLoad %v4float %glin1_ptr
%glin2_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_2 %int_0
%glin2 = OpLoad %v4float %glin2_ptr
%glin3_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_3 %int_0
%glin3 = OpLoad %v4float %glin3_ptr

%color1_ptr = OpAccessChain %ptr_input_v3float %color %int_1
%color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_input_v3float %color %int_2
%color2 = OpLoad %v3float %color2_ptr

%p0_xy = OpVectorShuffle %v2float %glin0 %glin0 0 1
%p1_xy = OpVectorShuffle %v2float %glin1 %glin1 0 1
%p2_xy = OpVectorShuffle %v2float %glin2 %glin2 0 1
%p3_xy = OpVectorShuffle %v2float %glin3 %glin3 0 1

%p1_z = OpCompositeExtract %float %glin1 2
%p1_w = OpCompositeExtract %float %glin1 3
%p2_z = OpCompositeExtract %float %glin2 2
%p2_w = OpCompositeExtract %float %glin2 3

%dir01 = OpFSub %v2float %p1_xy %p0_xy
%dir12 = OpFSub %v2float %p2_xy %p1_xy
%dir23 = OpFSub %v2float %p3_xy %p2_xy

%tan01 = OpExtInst %v2float %ext Normalize %dir01
%tan12 = OpExtInst %v2float %ext Normalize %dir12
%tan23 = OpExtInst %v2float %ext Normalize %dir23

%sum_start = OpFAdd %v2float %tan01 %tan12
%sum_end = OpFAdd %v2float %tan12 %tan23
%smooth_start = OpExtInst %v2float %ext Normalize %sum_start
%smooth_end = OpExtInst %v2float %ext Normalize %sum_end

%smooth_start_x = OpCompositeExtract %float %smooth_start 0
%smooth_start_y = OpCompositeExtract %float %smooth_start 1
%smooth_end_x = OpCompositeExtract %float %smooth_end 0
%smooth_end_y = OpCompositeExtract %float %smooth_end 1

%neg_smooth_start_y = OpFNegate %float %smooth_start_y
%neg_smooth_end_y = OpFNegate %float %smooth_end_y
%perp_start = OpCompositeConstruct %v2float %neg_smooth_start_y %smooth_start_x
%perp_end = OpCompositeConstruct %v2float %neg_smooth_end_y %smooth_end_x

%offset_start = OpVectorTimesScalar %v2float %perp_start %float_thickness
%offset_end = OpVectorTimesScalar %v2float %perp_end %float_thickness

%segment = OpFSub %v2float %p2_xy %p1_xy
%one_third_vec = OpVectorTimesScalar %v2float %segment %float_third
%two_thirds_vec = OpVectorTimesScalar %v2float %segment %float_two_thirds

%dash0_start_xy = %p1_xy
%dash0_end_xy = OpFAdd %v2float %p1_xy %one_third_vec
%dash1_start_xy = OpFAdd %v2float %p1_xy %two_thirds_vec
%dash1_end_xy = %p2_xy

%two_thirds_p1_z = OpFMul %float %p1_z %float_two_thirds
%one_third_p2_z = OpFMul %float %p2_z %float_third
%dash0_end_z = OpFAdd %float %two_thirds_p1_z %one_third_p2_z
%two_thirds_p1_w = OpFMul %float %p1_w %float_two_thirds
%one_third_p2_w = OpFMul %float %p2_w %float_third
%dash0_end_w = OpFAdd %float %two_thirds_p1_w %one_third_p2_w

%one_third_p1_z = OpFMul %float %p1_z %float_third
%two_thirds_p2_z = OpFMul %float %p2_z %float_two_thirds
%dash1_start_z = OpFAdd %float %one_third_p1_z %two_thirds_p2_z
%one_third_p1_w = OpFMul %float %p1_w %float_third
%two_thirds_p2_w = OpFMul %float %p2_w %float_two_thirds
%dash1_start_w = OpFAdd %float %one_third_p1_w %two_thirds_p2_w

%color_delta = OpFSub %v3float %color2 %color1
%color_third_delta = OpVectorTimesScalar %v3float %color_delta %float_third
%color_two_thirds_delta = OpVectorTimesScalar %v3float %color_delta %float_two_thirds
%dash0_start_color = %color1
%dash0_end_color = OpFAdd %v3float %color1 %color_third_delta
%dash1_start_color = OpFAdd %v3float %color1 %color_two_thirds_delta
%dash1_end_color = %color2

%dash0_v0_xy = OpFAdd %v2float %dash0_start_xy %offset_start
%dash0_v1_xy = OpFSub %v2float %dash0_start_xy %offset_start
%dash0_v2_xy = OpFAdd %v2float %dash0_end_xy %offset_end
%dash0_v3_xy = OpFSub %v2float %dash0_end_xy %offset_end

%dash0_v0_x = OpCompositeExtract %float %dash0_v0_xy 0
%dash0_v0_y = OpCompositeExtract %float %dash0_v0_xy 1
%dash0_v1_x = OpCompositeExtract %float %dash0_v1_xy 0
%dash0_v1_y = OpCompositeExtract %float %dash0_v1_xy 1
%dash0_v2_x = OpCompositeExtract %float %dash0_v2_xy 0
%dash0_v2_y = OpCompositeExtract %float %dash0_v2_xy 1
%dash0_v3_x = OpCompositeExtract %float %dash0_v3_xy 0
%dash0_v3_y = OpCompositeExtract %float %dash0_v3_xy 1

%dash0_v0 = OpCompositeConstruct %v4float %dash0_v0_x %dash0_v0_y %p1_z %p1_w
%dash0_v1 = OpCompositeConstruct %v4float %dash0_v1_x %dash0_v1_y %p1_z %p1_w
%dash0_v2 = OpCompositeConstruct %v4float %dash0_v2_x %dash0_v2_y %dash0_end_z %dash0_end_w
%dash0_v3 = OpCompositeConstruct %v4float %dash0_v3_x %dash0_v3_y %dash0_end_z %dash0_end_w

%dash1_v0_xy = OpFAdd %v2float %dash1_start_xy %offset_start
%dash1_v1_xy = OpFSub %v2float %dash1_start_xy %offset_start
%dash1_v2_xy = OpFAdd %v2float %dash1_end_xy %offset_end
%dash1_v3_xy = OpFSub %v2float %dash1_end_xy %offset_end

%dash1_v0_x = OpCompositeExtract %float %dash1_v0_xy 0
%dash1_v0_y = OpCompositeExtract %float %dash1_v0_xy 1
%dash1_v1_x = OpCompositeExtract %float %dash1_v1_xy 0
%dash1_v1_y = OpCompositeExtract %float %dash1_v1_xy 1
%dash1_v2_x = OpCompositeExtract %float %dash1_v2_xy 0
%dash1_v2_y = OpCompositeExtract %float %dash1_v2_xy 1
%dash1_v3_x = OpCompositeExtract %float %dash1_v3_xy 0
%dash1_v3_y = OpCompositeExtract %float %dash1_v3_xy 1

%dash1_v0 = OpCompositeConstruct %v4float %dash1_v0_x %dash1_v0_y %dash1_start_z %dash1_start_w
%dash1_v1 = OpCompositeConstruct %v4float %dash1_v1_x %dash1_v1_y %dash1_start_z %dash1_start_w
%dash1_v2 = OpCompositeConstruct %v4float %dash1_v2_x %dash1_v2_y %p2_z %p2_w
%dash1_v3 = OpCompositeConstruct %v4float %dash1_v3_x %dash1_v3_y %p2_z %p2_w

%gl_out_pos = OpAccessChain %ptr_output_v4float %gl_out %int_0

OpStore %outColor %dash0_start_color
OpStore %gl_out_pos %dash0_v0
OpEmitVertex
OpStore %outColor %dash0_start_color
OpStore %gl_out_pos %dash0_v1
OpEmitVertex
OpStore %outColor %dash0_end_color
OpStore %gl_out_pos %dash0_v2
OpEmitVertex
OpStore %outColor %dash0_end_color
OpStore %gl_out_pos %dash0_v3
OpEmitVertex
OpEndPrimitive

OpStore %outColor %dash1_start_color
OpStore %gl_out_pos %dash1_v0
OpEmitVertex
OpStore %outColor %dash1_start_color
OpStore %gl_out_pos %dash1_v1
OpEmitVertex
OpStore %outColor %dash1_end_color
OpStore %gl_out_pos %dash1_v2
OpEmitVertex
OpStore %outColor %dash1_end_color
OpStore %gl_out_pos %dash1_v3
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
