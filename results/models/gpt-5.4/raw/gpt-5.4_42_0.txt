OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %inColor "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.0399999991
%float_nh = OpConstant %float -0.0399999991
%uint_1 = OpConstant %uint 1
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_1 = OpTypeArray %gl_PerVertex_in %uint_1
%_ptr_Input_arr_gl_PerVertex_in_1 = OpTypePointer Input %arr_gl_PerVertex_in_1
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in_1 Input
%_ptr_Input_v4float = OpTypePointer Input %v4float
%arr_v3float_1 = OpTypeArray %v3float %uint_1
%_ptr_Input_arr_v3float_1 = OpTypePointer Input %arr_v3float_1
%inColor = OpVariable %_ptr_Input_arr_v3float_1 Input
%_ptr_Input_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %_ptr_Output_v3float Output

%main = OpFunction %void None %func
%entry = OpLabel
%pos_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%color_ptr = OpAccessChain %_ptr_Input_v3float %inColor %int_0
%col = OpLoad %v3float %color_ptr
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%x0 = OpFAdd %float %x %float_nh
%x1 = OpFAdd %float %x %float_h
%y0 = OpFAdd %float %y %float_nh
%y1 = OpFAdd %float %y %float_h
%v0 = OpCompositeConstruct %v4float %x0 %y0 %float_0 %float_1
%v1 = OpCompositeConstruct %v4float %x1 %y0 %float_0 %float_1
%v2 = OpCompositeConstruct %v4float %x0 %y1 %float_0 %float_1
%v3 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
%out_pos_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %outColor %col
OpStore %out_pos_ptr %v0
OpEmitVertex
OpStore %outColor %col
OpStore %out_pos_ptr %v1
OpEmitVertex
OpStore %outColor %col
OpStore %out_pos_ptr %v2
OpEmitVertex
OpStore %outColor %col
OpStore %out_pos_ptr %v3
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
