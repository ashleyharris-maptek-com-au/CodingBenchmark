declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Cell = [number, number];

interface DroneInput {
  id: string;
  x: number;
  y: number;
}

interface InputData {
  probability: number[][];
  drones: DroneInput[];
  maxStepsPerDrone: number;
}

interface Target {
  x: number;
  y: number;
  value: number;
}

interface PlanCandidate {
  path: Cell[];
  totalScore: number;
}

const parsed = data as InputData;
const grid = Array.isArray(parsed.probability) ? parsed.probability : [];
const drones = Array.isArray(parsed.drones) ? parsed.drones : [];
const maxStepsPerDrone = Number.isFinite(parsed.maxStepsPerDrone) ? Math.max(0, parsed.maxStepsPerDrone | 0) : 0;

const height = grid.length;
const width = height > 0 && Array.isArray(grid[0]) ? grid[0].length : 0;
const cellCount = width * height;

function inside(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < width && y < height;
}

function idx(x: number, y: number): number {
  return y * width + x;
}

function decode(index: number): Cell {
  return [index % width, Math.floor(index / width)];
}

function isObstacle(x: number, y: number): boolean {
  return !inside(x, y) || grid[y][x] === -1;
}

function manhattan(ax: number, ay: number, bx: number, by: number): number {
  return Math.abs(ax - bx) + Math.abs(ay - by);
}

const validMoves: Cell[] = [
  [0, 0],
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

const adjacency: number[][] = Array.from({ length: cellCount }, () => []);
const passable = new Uint8Array(cellCount);
const cellValues = new Float64Array(cellCount);
const positiveTargets: Target[] = [];

for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const id = idx(x, y);
    const value = Number(grid[y][x]) || 0;
    cellValues[id] = value;
    if (value !== -1) {
      passable[id] = 1;
      if (value > 0) {
        positiveTargets.push({ x, y, value });
      }
      for (const [dx, dy] of validMoves) {
        const nx = x + dx;
        const ny = y + dy;
        if (inside(nx, ny) && grid[ny][nx] !== -1) {
          adjacency[id].push(idx(nx, ny));
        }
      }
    }
  }
}

function bfsDistances(startIndex: number): Int32Array {
  const dist = new Int32Array(cellCount);
  dist.fill(-1);
  if (startIndex < 0 || startIndex >= cellCount || passable[startIndex] === 0) {
    return dist;
  }
  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;
  dist[startIndex] = 0;
  queue[tail++] = startIndex;
  while (head < tail) {
    const current = queue[head++];
    const nextDist = dist[current] + 1;
    const neighbors = adjacency[current];
    for (let i = 0; i < neighbors.length; i++) {
      const next = neighbors[i];
      if (dist[next] !== -1) {
        continue;
      }
      dist[next] = nextDist;
      queue[tail++] = next;
    }
  }
  return dist;
}

const droneStartIndices: number[] = drones.map((drone) => {
  if (!inside(drone.x, drone.y) || isObstacle(drone.x, drone.y)) {
    return -1;
  }
  return idx(drone.x, drone.y);
});

const droneDistances: Int32Array[] = droneStartIndices.map((start) => bfsDistances(start));

const targetAssignments: number[][] = Array.from({ length: drones.length }, () => []);
const targetCellIndices: number[] = positiveTargets.map((target) => idx(target.x, target.y));

for (let t = 0; t < positiveTargets.length; t++) {
  let bestDrone = -1;
  let bestMetric = Number.POSITIVE_INFINITY;
  for (let d = 0; d < drones.length; d++) {
    const dist = droneDistances[d][targetCellIndices[t]];
    if (dist < 0) {
      continue;
    }
    const metric = dist - positiveTargets[t].value * 0.15;
    if (metric < bestMetric) {
      bestMetric = metric;
      bestDrone = d;
    }
  }
  if (bestDrone >= 0) {
    targetAssignments[bestDrone].push(t);
  }
}

function chooseTargetsForDrone(droneIndex: number): number[] {
  const assigned = targetAssignments[droneIndex];
  if (assigned.length === 0) {
    return [];
  }
  const startIndex = droneStartIndices[droneIndex];
  if (startIndex < 0) {
    return [];
  }

  assigned.sort((a, b) => positiveTargets[b].value - positiveTargets[a].value);
  const limit = Math.min(assigned.length, Math.max(24, Math.min(80, Math.floor(Math.sqrt(cellCount || 1) * 8))));
  const shortlisted = assigned.slice(0, limit);

  const chosen: number[] = [];
  let remaining = maxStepsPerDrone;
  let currentIndex = startIndex;
  let currentDistances = bfsDistances(currentIndex);
  const remainingSet = new Set<number>(shortlisted);

  while (remainingSet.size > 0) {
    let bestTarget = -1;
    let bestScore = -1;
    for (const targetId of remainingSet) {
      const targetIndex = targetCellIndices[targetId];
      const dist = currentDistances[targetIndex];
      if (dist < 0 || dist > remaining) {
        continue;
      }
      const target = positiveTargets[targetId];
      const score = target.value / (dist + 1);
      if (score > bestScore) {
        bestScore = score;
        bestTarget = targetId;
      }
    }
    if (bestTarget < 0) {
      break;
    }
    chosen.push(bestTarget);
    remaining -= currentDistances[targetCellIndices[bestTarget]];
    currentIndex = targetCellIndices[bestTarget];
    currentDistances = bfsDistances(currentIndex);
    remainingSet.delete(bestTarget);
  }

  return chosen;
}

function reconstructPath(startIndex: number, goalIndex: number, distFromStart: Int32Array): number[] {
  if (startIndex === goalIndex) {
    return [startIndex];
  }
  if (goalIndex < 0 || goalIndex >= cellCount || distFromStart[goalIndex] < 0) {
    return [startIndex];
  }
  const path: number[] = [];
  let current = goalIndex;
  path.push(current);
  while (current !== startIndex) {
    const currentDist = distFromStart[current];
    const neighbors = adjacency[current];
    let nextNode = -1;
    let nextValue = -1;
    for (let i = 0; i < neighbors.length; i++) {
      const neighbor = neighbors[i];
      if (distFromStart[neighbor] === currentDist - 1) {
        const value = cellValues[neighbor];
        if (value > nextValue) {
          nextValue = value;
          nextNode = neighbor;
        }
      }
    }
    if (nextNode < 0) {
      break;
    }
    current = nextNode;
    path.push(current);
  }
  path.reverse();
  return path;
}

function buildGreedyCoveragePath(droneIndex: number): PlanCandidate {
  const startIndex = droneStartIndices[droneIndex];
  if (startIndex < 0) {
    return { path: [], totalScore: 0 };
  }

  const chosenTargets = chooseTargetsForDrone(droneIndex);
  const pathIndices: number[] = [startIndex];
  const covered = new Uint8Array(cellCount);
  let score = 0;
  if (cellValues[startIndex] > 0) {
    covered[startIndex] = 1;
    score += cellValues[startIndex];
  }

  let current = startIndex;
  let stepsUsed = 0;

  for (const targetId of chosenTargets) {
    const targetIndex = targetCellIndices[targetId];
    const dist = bfsDistances(current);
    const segment = reconstructPath(current, targetIndex, dist);
    if (segment.length <= 1) {
      continue;
    }
    const extraSteps = segment.length - 1;
    if (stepsUsed + extraSteps > maxStepsPerDrone) {
      break;
    }
    for (let i = 1; i < segment.length; i++) {
      const node = segment[i];
      pathIndices.push(node);
      if (!covered[node] && cellValues[node] > 0) {
        covered[node] = 1;
        score += cellValues[node];
      }
    }
    current = targetIndex;
    stepsUsed += extraSteps;
  }

  while (stepsUsed < maxStepsPerDrone) {
    const distances = bfsDistances(current);
    let bestNode = current;
    let bestGain = 0;
    let bestDist = 0;
    const radius = Math.min(12, maxStepsPerDrone - stepsUsed);
    for (let node = 0; node < cellCount; node++) {
      const dist = distances[node];
      if (dist <= 0 || dist > radius) {
        continue;
      }
      if (covered[node] || cellValues[node] <= 0) {
        continue;
      }
      const gain = cellValues[node] / dist;
      if (gain > bestGain) {
        bestGain = gain;
        bestNode = node;
        bestDist = dist;
      }
    }
    if (bestNode === current || bestGain <= 0) {
      const neighbors = adjacency[current];
      let moved = false;
      for (let i = 0; i < neighbors.length; i++) {
        const next = neighbors[i];
        if (!covered[next] && cellValues[next] >= 0) {
          pathIndices.push(next);
          if (cellValues[next] > 0) {
            covered[next] = 1;
            score += cellValues[next];
          }
          current = next;
          stepsUsed += 1;
          moved = true;
          break;
        }
      }
      if (!moved) {
        pathIndices.push(current);
        stepsUsed += 1;
      }
      continue;
    }
    const segment = reconstructPath(current, bestNode, distances);
    if (segment.length <= 1 || stepsUsed + bestDist > maxStepsPerDrone) {
      break;
    }
    for (let i = 1; i < segment.length; i++) {
      const node = segment[i];
      pathIndices.push(node);
      if (!covered[node] && cellValues[node] > 0) {
        covered[node] = 1;
        score += cellValues[node];
      }
    }
    current = bestNode;
    stepsUsed += bestDist;
  }

  while (pathIndices.length < maxStepsPerDrone + 1) {
    pathIndices.push(current);
  }

  const path = pathIndices.map((node) => decode(node));
  return { path, totalScore: score };
}

function validateAndFixPath(path: Cell[], drone: DroneInput): Cell[] {
  const fixed: Cell[] = [];
  let current: Cell;
  if (inside(drone.x, drone.y) && !isObstacle(drone.x, drone.y)) {
    current = [drone.x, drone.y];
  } else {
    current = [0, 0];
  }
  fixed.push(current);
  const targetLength = maxStepsPerDrone + 1;
  for (let i = 1; i < targetLength; i++) {
    const desired = i < path.length ? path[i] : current;
    let next: Cell = current;
    if (Array.isArray(desired) && desired.length === 2) {
      const [tx, ty] = desired;
      if (inside(tx, ty) && !isObstacle(tx, ty) && manhattan(current[0], current[1], tx, ty) <= 1) {
        next = [tx, ty];
      } else {
        const currentIndex = idx(current[0], current[1]);
        let bestNeighbor = currentIndex;
        let bestValue = -1;
        for (const neighbor of adjacency[currentIndex]) {
          const [nx, ny] = decode(neighbor);
          const value = cellValues[neighbor];
          if (value > bestValue) {
            bestValue = value;
            bestNeighbor = neighbor;
          }
        }
        next = decode(bestNeighbor);
      }
    }
    fixed.push(next);
    current = next;
  }
  return fixed;
}

const paths: Cell[][] = drones.map((drone, droneIndex) => {
  const plan = buildGreedyCoveragePath(droneIndex);
  return validateAndFixPath(plan.path, drone);
});

const output = { paths };
process.stdout.write(JSON.stringify(output));
