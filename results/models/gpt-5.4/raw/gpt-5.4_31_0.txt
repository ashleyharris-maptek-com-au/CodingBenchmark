use std::collections::HashMap;
use std::io::{self, Read};

const SMALL_N_LIMIT: usize = 46;
const EXACT_POS_SUM_LIMIT: i64 = 2_000_000;
const DENSE_DP_TARGET_LIMIT: i64 = 2_000_000;
const LARGE_PAIR_LIMIT: usize = 2_000_000;

struct Scanner {
    data: Vec<u8>,
    index: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data)?;
        Ok(Self { data, index: 0 })
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.index < self.data.len() && self.data[self.index].is_ascii_whitespace() {
            self.index += 1;
        }
        if self.index >= self.data.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.data[self.index] == b'-' {
            sign = -1;
            self.index += 1;
        } else if self.data[self.index] == b'+' {
            self.index += 1;
        }
        let mut value = 0i64;
        let mut saw_digit = false;
        while self.index < self.data.len() {
            let b = self.data[self.index];
            if !b.is_ascii_digit() {
                break;
            }
            saw_digit = true;
            value = value.saturating_mul(10).saturating_add((b - b'0') as i64);
            self.index += 1;
        }
        if !saw_digit {
            None
        } else {
            Some(value.saturating_mul(sign))
        }
    }
}

fn append_indices_from_mask(base: usize, values: &[i64], mask: u64, out: &mut Vec<usize>) {
    let mut bits = mask;
    while bits != 0 {
        let bit = bits.trailing_zeros() as usize;
        if bit < values.len() {
            out.push(base + bit);
        }
        bits &= bits - 1;
    }
}

fn solve_zero_target(values: &[i64]) -> Vec<usize> {
    let mut result = Vec::new();
    for (idx, &value) in values.iter().enumerate() {
        if value == 0 {
            result.push(idx);
        }
    }
    result
}

fn solve_singletons(values: &[i64], target: i64) -> Option<Vec<usize>> {
    for (idx, &value) in values.iter().enumerate() {
        if value == target {
            return Some(vec![idx]);
        }
    }
    None
}

fn solve_greedy_pairs(values: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut seen = HashMap::<i64, usize>::with_capacity(values.len().min(LARGE_PAIR_LIMIT));
    for (idx, &value) in values.iter().enumerate() {
        let need = target - value;
        if let Some(&other) = seen.get(&need) {
            return Some(vec![other, idx]);
        }
        seen.entry(value).or_insert(idx);
        if seen.len() >= LARGE_PAIR_LIMIT {
            break;
        }
    }
    None
}

fn meet_in_the_middle(values: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = values.len();
    let mid = n / 2;
    let left = &values[..mid];
    let right = &values[mid..];

    let left_size = 1usize << left.len();
    let right_size = 1usize << right.len();

    let mut left_sums = Vec::with_capacity(left_size);
    for mask in 0..left_size {
        let mut sum = 0i64;
        for (bit, &value) in left.iter().enumerate() {
            if ((mask >> bit) & 1) != 0 {
                sum += value;
            }
        }
        left_sums.push((sum, mask as u64));
    }
    left_sums.sort_unstable_by_key(|&(sum, _)| sum);

    for mask in 0..right_size {
        let mut sum = 0i64;
        for (bit, &value) in right.iter().enumerate() {
            if ((mask >> bit) & 1) != 0 {
                sum += value;
            }
        }
        let need = target - sum;
        if let Ok(pos) = left_sums.binary_search_by_key(&need, |&(left_sum, _)| left_sum) {
            let mut result = Vec::new();
            append_indices_from_mask(0, left, left_sums[pos].1, &mut result);
            append_indices_from_mask(mid, right, mask as u64, &mut result);
            return Some(result);
        }
    }
    None
}

fn positive_dense_dp(values: &[i64], target: i64) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let target_usize = target as usize;
    let mut prev = vec![usize::MAX; target_usize + 1];
    let mut picked = vec![usize::MAX; target_usize + 1];
    prev[0] = 0;

    for (idx, &value) in values.iter().enumerate() {
        if value <= 0 || value > target {
            continue;
        }
        let v = value as usize;
        for sum in (v..=target_usize).rev() {
            if prev[sum] == usize::MAX && prev[sum - v] != usize::MAX {
                prev[sum] = sum - v;
                picked[sum] = idx;
            }
        }
        if prev[target_usize] != usize::MAX {
            let mut result = Vec::new();
            let mut cur = target_usize;
            while cur != 0 {
                let idx_used = picked[cur];
                if idx_used == usize::MAX {
                    break;
                }
                result.push(idx_used);
                cur = prev[cur];
            }
            result.reverse();
            return Some(result);
        }
    }
    None
}

fn sparse_signed_dp(values: &[i64], target: i64, max_states: usize) -> Option<Vec<usize>> {
    let mut current = HashMap::<i64, (i64, usize)>::with_capacity(1024);
    current.insert(0, (i64::MIN, usize::MAX));
    if target == 0 {
        return Some(Vec::new());
    }

    for (idx, &value) in values.iter().enumerate() {
        let mut next = current.clone();
        for (&sum, _) in &current {
            let new_sum = sum + value;
            next.entry(new_sum).or_insert((sum, idx));
        }
        if let Some(&(prev_sum, picked_idx)) = next.get(&target) {
            let mut result = Vec::new();
            let mut cur = target;
            let mut prev_cur = prev_sum;
            let mut picked_cur = picked_idx;
            while picked_cur != usize::MAX {
                result.push(picked_cur);
                cur = prev_cur;
                let &(p_sum, p_idx) = next.get(&cur).unwrap_or(&(i64::MIN, usize::MAX));
                prev_cur = p_sum;
                picked_cur = p_idx;
                if cur == 0 && picked_cur == usize::MAX {
                    break;
                }
            }
            result.reverse();
            return Some(result);
        }
        if next.len() > max_states {
            return None;
        }
        current = next;
    }
    None
}

fn solve(values: &[i64], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(solve_zero_target(values));
    }

    if let Some(ans) = solve_singletons(values, target) {
        return Some(ans);
    }
    if let Some(ans) = solve_greedy_pairs(values, target) {
        return Some(ans);
    }

    let has_negative = values.iter().any(|&v| v < 0);
    let positive_sum = values
        .iter()
        .filter(|&&v| v > 0)
        .fold(0i64, |acc, &v| acc.saturating_add(v));
    let negative_sum = values
        .iter()
        .filter(|&&v| v < 0)
        .fold(0i64, |acc, &v| acc.saturating_add(v));

    if target > positive_sum || target < negative_sum {
        return None;
    }

    if values.len() <= SMALL_N_LIMIT {
        return meet_in_the_middle(values, target);
    }

    if !has_negative
        && target >= 0
        && target <= DENSE_DP_TARGET_LIMIT
        && positive_sum <= EXACT_POS_SUM_LIMIT
    {
        if let Some(ans) = positive_dense_dp(values, target) {
            return Some(ans);
        }
    }

    if values.len() <= 256 {
        if let Some(ans) = sparse_signed_dp(values, target, 2_000_000) {
            return Some(ans);
        }
    }

    None
}

fn main() -> io::Result<()> {
    let mut scanner = Scanner::new()?;
    let n = match scanner.next_i64() {
        Some(v) if v >= 0 => v as usize,
        _ => {
            println!("NO");
            return Ok(());
        }
    };
    let target = match scanner.next_i64() {
        Some(v) => v,
        None => {
            println!("NO");
            return Ok(());
        }
    };

    let mut values = Vec::with_capacity(n);
    for _ in 0..n {
        match scanner.next_i64() {
            Some(v) => values.push(v),
            None => {
                println!("NO");
                return Ok(());
            }
        }
    }

    if let Some(mut indices) = solve(&values, target) {
        let sum = indices
            .iter()
            .fold(0i64, |acc, &idx| acc.saturating_add(values[idx]));
        if sum != target {
            println!("NO");
            return Ok(());
        }
        indices.sort_unstable();
        indices.dedup();
        let sum_after = indices
            .iter()
            .fold(0i64, |acc, &idx| acc.saturating_add(values[idx]));
        if sum_after != target {
            println!("NO");
            return Ok(());
        }
        println!("YES");
        if indices.is_empty() {
            println!();
        } else {
            let mut first = true;
            for idx in indices {
                if !first {
                    print!(" ");
                }
                first = false;
                print!("{idx}");
            }
            println!();
        }
    } else {
        println!("NO");
    }

    Ok(())
}
