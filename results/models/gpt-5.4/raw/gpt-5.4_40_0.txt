#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <vector>

using namespace std;

struct Bitset {
    int n = 0;
    vector<uint64_t> w;

    Bitset() = default;
    explicit Bitset(int n_) : n(n_), w((n_ + 63) >> 6, 0) {}

    void set(int i) { w[i >> 6] |= 1ULL << (i & 63); }
    void reset(int i) { w[i >> 6] &= ~(1ULL << (i & 63)); }
    bool test(int i) const { return (w[i >> 6] >> (i & 63)) & 1ULL; }
    bool any() const {
        for (uint64_t x : w) {
            if (x) return true;
        }
        return false;
    }
    int count() const {
        int res = 0;
        for (uint64_t x : w) res += __builtin_popcountll(x);
        return res;
    }
};

static Bitset bit_and_not(const Bitset& a, const Bitset& b) {
    Bitset r(a.n);
    for (size_t i = 0; i < a.w.size(); ++i) r.w[i] = a.w[i] & ~b.w[i];
    return r;
}

static Bitset bit_and(const Bitset& a, const Bitset& b) {
    Bitset r(a.n);
    for (size_t i = 0; i < a.w.size(); ++i) r.w[i] = a.w[i] & b.w[i];
    return r;
}

static vector<int> bit_vertices(const Bitset& s) {
    vector<int> res;
    for (int block = 0; block < (int)s.w.size(); ++block) {
        uint64_t x = s.w[block];
        while (x) {
            int b = __builtin_ctzll(x);
            int v = (block << 6) + b;
            if (v < s.n) res.push_back(v);
            x &= x - 1;
        }
    }
    return res;
}

struct Solver {
    int n;
    vector<long long> weight;
    vector<Bitset> adj;
    long long best_value = numeric_limits<long long>::lowest();
    vector<char> best_take;

    explicit Solver(int n_) : n(n_), weight(n_), adj(n_, Bitset(n_)), best_take(n_, 0) {}

    long long greedy_coloring_bound(
        const Bitset& cand,
        vector<int>& order,
        vector<long long>& color_bound
    ) const {
        vector<int> verts = bit_vertices(cand);
        sort(verts.begin(), verts.end(), [&](int a, int b) {
            if (weight[a] != weight[b]) return weight[a] > weight[b];
            int da = adj[a].count();
            int db = adj[b].count();
            if (da != db) return da > db;
            return a < b;
        });

        vector<Bitset> color_sets;
        vector<long long> color_max;
        order.clear();
        color_bound.clear();
        long long prefix = 0;

        for (int v : verts) {
            int placed = -1;
            for (int c = 0; c < (int)color_sets.size(); ++c) {
                if (!bit_and(adj[v], color_sets[c]).any()) {
                    placed = c;
                    break;
                }
            }
            if (placed == -1) {
                placed = (int)color_sets.size();
                color_sets.emplace_back(n);
                color_max.push_back(0);
            }
            color_sets[placed].set(v);
            color_max[placed] = max(color_max[placed], max(0LL, weight[v]));
        }

        vector<int> by_color;
        by_color.reserve(verts.size());
        vector<vector<int>> bucket(color_sets.size());
        for (int v : verts) {
            for (int c = 0; c < (int)color_sets.size(); ++c) {
                if (color_sets[c].test(v)) {
                    bucket[c].push_back(v);
                    break;
                }
            }
        }

        for (int c = 0; c < (int)bucket.size(); ++c) {
            sort(bucket[c].begin(), bucket[c].end(), [&](int a, int b) {
                if (weight[a] != weight[b]) return weight[a] < weight[b];
                return a < b;
            });
            for (int v : bucket[c]) by_color.push_back(v);
        }

        vector<long long> suffix(by_color.size());
        vector<int> color_of_vertex(n, -1);
        for (int c = 0; c < (int)bucket.size(); ++c) {
            for (int v : bucket[c]) color_of_vertex[v] = c;
        }

        vector<long long> rem = color_max;
        long long total = accumulate(rem.begin(), rem.end(), 0LL);
        for (int i = 0; i < (int)by_color.size(); ++i) {
            int v = by_color[i];
            suffix[i] = total;
            int c = color_of_vertex[v];
            if (rem[c] > 0) {
                total -= rem[c];
                rem[c] = 0;
            }
        }

        order = move(by_color);
        color_bound = move(suffix);
        return order.empty() ? 0 : color_bound[0];
    }

    void expand(Bitset cand, long long cur, vector<char>& take) {
        if (!cand.any()) {
            if (cur > best_value) {
                best_value = cur;
                best_take = take;
            }
            return;
        }

        vector<int> order;
        vector<long long> bound;
        greedy_coloring_bound(cand, order, bound);

        for (int idx = (int)order.size() - 1; idx >= 0; --idx) {
            if (cur + bound[idx] <= best_value) return;

            int v = order[idx];
            if (!cand.test(v)) continue;

            take[v] = 1;
            Bitset next = bit_and_not(cand, adj[v]);
            next.reset(v);
            expand(next, cur + weight[v], take);
            take[v] = 0;

            cand.reset(v);
        }

        if (cur > best_value) {
            best_value = cur;
            best_take = take;
        }
    }

    pair<long long, vector<char>> solve() {
        vector<int> pos, zero, neg;
        for (int i = 0; i < n; ++i) {
            if (weight[i] > 0) pos.push_back(i);
            else if (weight[i] == 0) zero.push_back(i);
            else neg.push_back(i);
        }

        Bitset cand(n);
        for (int v : pos) cand.set(v);

        vector<char> take(n, 0);
        best_value = 0;
        best_take.assign(n, 0);

        expand(cand, 0, take);
        return {best_value, best_take};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    Solver solver(n);
    for (int i = 0; i < n; ++i) cin >> solver.weight[i];

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        solver.adj[u].set(v);
        solver.adj[v].set(u);
    }

    auto [value, take] = solver.solve();

    cout << value << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << int(take[i]);
    }
    cout << '\n';

    return 0;
}
