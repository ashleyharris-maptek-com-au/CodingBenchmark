namespace {

constexpr double kPi = 3.141592653589793238462643383279502884;
constexpr double kSqrt3 = 1.73205080756887729353;
constexpr double kSqrt6 = 2.4494897427831780982;
constexpr double kEps = 1e-9;

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;

    Vec3() = default;
    Vec3(double xx, double yy, double zz) : x(xx), y(yy), z(zz) {}

    Vec3 operator+(const Vec3& other) const { return {x + other.x, y + other.y, z + other.z}; }
    Vec3 operator-(const Vec3& other) const { return {x - other.x, y - other.y, z - other.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }

    Vec3& operator+=(const Vec3& other) {
        x += other.x;
        y += other.y;
        z += other.z;
        return *this;
    }
};

double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

double norm2(const Vec3& v) {
    return dot(v, v);
}

double norm(const Vec3& v) {
    return std::sqrt(norm2(v));
}

Vec3 normalized(const Vec3& v) {
    const double n = norm(v);
    if (n <= kEps) return {0.0, 0.0, 0.0};
    return v / n;
}

struct Face {
    int a = 0;
    int b = 0;
    int c = 0;
    Vec3 n;
    double d = 0.0;
};

struct Placement {
    Vec3 center;
    double rx = 0.0;
    double ry = 0.0;
    double rz = 0.0;
};

struct Bounds {
    Vec3 lo;
    Vec3 hi;
};

struct CellKey {
    int x = 0;
    int y = 0;
    int z = 0;

    bool operator==(const CellKey& other) const {
        return x == other.x && y == other.y && z == other.z;
    }
};

struct CellHasher {
    std::size_t operator()(const CellKey& key) const noexcept {
        std::size_t h = 1469598103934665603ull;
        auto mix = [&](std::uint64_t v) {
            h ^= v + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        };
        mix(static_cast<std::uint64_t>(static_cast<std::int64_t>(key.x)));
        mix(static_cast<std::uint64_t>(static_cast<std::int64_t>(key.y)));
        mix(static_cast<std::uint64_t>(static_cast<std::int64_t>(key.z)));
        return h;
    }
};

struct Timer {
    std::chrono::steady_clock::time_point start = std::chrono::steady_clock::now();
    double limit_seconds = 29.2;

    bool expired() const {
        const auto now = std::chrono::steady_clock::now();
        const double elapsed = std::chrono::duration<double>(now - start).count();
        return elapsed >= limit_seconds;
    }
};

Vec3 rotateXYZ(const Vec3& v, double rx, double ry, double rz) {
    const double cx = std::cos(rx), sx = std::sin(rx);
    const double cy = std::cos(ry), sy = std::sin(ry);
    const double cz = std::cos(rz), sz = std::sin(rz);

    Vec3 t = v;
    t = {t.x, cx * t.y - sx * t.z, sx * t.y + cx * t.z};
    t = {cy * t.x + sy * t.z, t.y, -sy * t.x + cy * t.z};
    t = {cz * t.x - sz * t.y, sz * t.x + cz * t.y, t.z};
    return t;
}

std::array<Vec3, 4> tetraTemplate(double edge) {
    const Vec3 a{0.0, 0.0, 0.0};
    const Vec3 b{edge, 0.0, 0.0};
    const Vec3 c{0.5 * edge, 0.5 * kSqrt3 * edge, 0.0};
    const Vec3 d{0.5 * edge, kSqrt3 * edge / 6.0, kSqrt6 * edge / 3.0};
    const Vec3 centroid = (a + b + c + d) / 4.0;
    return {a - centroid, b - centroid, c - centroid, d - centroid};
}

std::array<Vec3, 4> orientedVertices(double edge, double rx, double ry, double rz) {
    auto pts = tetraTemplate(edge);
    for (Vec3& p : pts) {
        p = rotateXYZ(p, rx, ry, rz);
    }
    return pts;
}

bool planeFromPoints(const Vec3& a, const Vec3& b, const Vec3& c, Vec3& n, double& d) {
    n = cross(b - a, c - a);
    const double len = norm(n);
    if (len <= kEps) return false;
    n = n / len;
    d = dot(n, a);
    return true;
}

std::vector<Face> buildHullFaces(const std::vector<Vec3>& pts) {
    const int n = static_cast<int>(pts.size());
    std::vector<Face> faces;
    if (n < 4) return faces;

    Vec3 centroid{0.0, 0.0, 0.0};
    for (const Vec3& p : pts) centroid += p;
    centroid = centroid / static_cast<double>(n);

    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 nrm;
                double d = 0.0;
                if (!planeFromPoints(pts[i], pts[j], pts[k], nrm, d)) continue;

                int pos = 0;
                int neg = 0;
                for (int t = 0; t < n; ++t) {
                    if (t == i || t == j || t == k) continue;
                    const double side = dot(nrm, pts[t]) - d;
                    if (side > 1e-8) ++pos;
                    else if (side < -1e-8) ++neg;
                }
                if (pos > 0 && neg > 0) continue;

                Face f;
                if (dot(nrm, centroid) - d > 0.0) {
                    f = {i, k, j, nrm * -1.0, -d};
                } else {
                    f = {i, j, k, nrm, d};
                }
                bool duplicate = false;
                for (const Face& g : faces) {
                    if ((f.a == g.a || f.a == g.b || f.a == g.c) &&
                        (f.b == g.a || f.b == g.b || f.b == g.c) &&
                        (f.c == g.a || f.c == g.b || f.c == g.c)) {
                        duplicate = true;
                        break;
                    }
                }
                if (!duplicate) faces.push_back(f);
            }
        }
    }
    return faces;
}

bool pointInConvexPoly(const Vec3& p, const std::vector<Face>& faces, const std::vector<Vec3>& verts) {
    for (const Face& f : faces) {
        const Vec3& a = verts[f.a];
        const Vec3 nrm = normalized(cross(verts[f.b] - a, verts[f.c] - a));
        if (dot(nrm, p - a) > 1e-8) return false;
    }
    return true;
}

double volumeByTriangulation(const std::vector<Face>& faces, const std::vector<Vec3>& verts) {
    double sum = 0.0;
    for (const Face& f : faces) {
        sum += dot(verts[f.a], cross(verts[f.b], verts[f.c])) / 6.0;
    }
    return std::abs(sum);
}

Bounds computeBounds(const std::vector<Vec3>& verts) {
    Bounds b;
    b.lo = b.hi = verts.front();
    for (const Vec3& p : verts) {
        b.lo.x = std::min(b.lo.x, p.x);
        b.lo.y = std::min(b.lo.y, p.y);
        b.lo.z = std::min(b.lo.z, p.z);
        b.hi.x = std::max(b.hi.x, p.x);
        b.hi.y = std::max(b.hi.y, p.y);
        b.hi.z = std::max(b.hi.z, p.z);
    }
    return b;
}

struct Solver {
    int vertex_count = 0;
    double edge = 1.0;
    std::vector<Vec3> verts;
    std::vector<Face> faces;
    Bounds bounds;
    double volume = 0.0;
    double tetra_volume = 0.0;
    double circumradius = 0.0;
    double inradius = 0.0;
    double packing_gap = 0.0;
    std::vector<Placement> out;
    Timer timer;

    std::unordered_map<CellKey, std::vector<int>, CellHasher> grid;
    double cell_size = 1.0;

    bool readInput() {
        if (!(std::cin >> vertex_count >> edge)) return false;
        if (vertex_count < 4 || edge <= 0.0) return false;
        verts.resize(vertex_count);
        for (int i = 0; i < vertex_count; ++i) {
            std::cin >> verts[i].x >> verts[i].y >> verts[i].z;
        }
        return true;
    }

    bool prepare() {
        faces = buildHullFaces(verts);
        if (faces.empty()) return false;
        bounds = computeBounds(verts);
        volume = volumeByTriangulation(faces, verts);
        tetra_volume = edge * edge * edge / (6.0 * kSqrt2());
        circumradius = edge * std::sqrt(3.0 / 8.0);
        inradius = edge / std::sqrt(24.0);
        packing_gap = 1.08 * circumradius;
        cell_size = 2.5 * circumradius;
        return true;
    }

    static double kSqrt2() {
        return 1.4142135623730950488;
    }

    CellKey keyFor(const Vec3& p) const {
        return {
            static_cast<int>(std::floor(p.x / cell_size)),
            static_cast<int>(std::floor(p.y / cell_size)),
            static_cast<int>(std::floor(p.z / cell_size))
        };
    }

    bool insideWithMargin(const Vec3& center, const std::array<Vec3, 4>& oriented) const {
        for (const Vec3& local : oriented) {
            if (!pointInConvexPoly(center + local, faces, verts)) return false;
        }
        return true;
    }

    bool overlaps(const Vec3& center) const {
        const CellKey ck = keyFor(center);
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                for (int dz = -1; dz <= 1; ++dz) {
                    const CellKey nk{ck.x + dx, ck.y + dy, ck.z + dz};
                    auto it = grid.find(nk);
                    if (it == grid.end()) continue;
                    for (int idx : it->second) {
                        if (norm2(out[idx].center - center) < packing_gap * packing_gap) {
                            return true;
                        }
                    }
                }
            }
        }
        return false;
    }

    void addPlacement(const Vec3& center, double rx, double ry, double rz) {
        const int idx = static_cast<int>(out.size());
        out.push_back({center, rx, ry, rz});
        grid[keyFor(center)].push_back(idx);
    }

    std::vector<std::array<double, 3>> orientationSet() const {
        std::vector<std::array<double, 3>> result = {
            {0.0, 0.0, 0.0},
            {kPi, 0.0, 0.0},
            {0.0, kPi, 0.0},
            {0.0, 0.0, kPi},
            {0.0, 0.0, kPi / 3.0},
            {0.0, 0.0, 2.0 * kPi / 3.0},
            {0.0, 0.0, kPi},
            {0.0, 0.0, 4.0 * kPi / 3.0},
            {0.0, 0.0, 5.0 * kPi / 3.0},
            {std::acos(1.0 / 3.0), 0.0, 0.0},
            {-std::acos(1.0 / 3.0), 0.0, 0.0},
            {0.0, std::acos(1.0 / 3.0), 0.0},
            {0.0, -std::acos(1.0 / 3.0), 0.0}
        };
        return result;
    }

    void fillGridRegular() {
        const auto orientations = orientationSet();
        std::vector<std::array<Vec3, 4>> oriented_cache;
        oriented_cache.reserve(orientations.size());
        for (const auto& ang : orientations) {
            oriented_cache.push_back(orientedVertices(edge, ang[0], ang[1], ang[2]));
        }

        const double dx = edge;
        const double dy = edge * kSqrt3 / 2.0;
        const double dz = edge * std::sqrt(2.0 / 3.0);
        const double x0 = bounds.lo.x + circumradius;
        const double y0 = bounds.lo.y + circumradius;
        const double z0 = bounds.lo.z + circumradius;

        for (int layer = 0;; ++layer) {
            const double z = z0 + layer * (dz * 0.92);
            if (z > bounds.hi.z - circumradius || timer.expired()) break;
            const double y_shift = (layer & 1) ? dy / 3.0 : 0.0;
            for (int row = 0;; ++row) {
                const double y = y0 + row * dy + y_shift;
                if (y > bounds.hi.y - circumradius || timer.expired()) break;
                const double x_shift = ((row + layer) & 1) ? dx * 0.5 : 0.0;
                for (int col = 0;; ++col) {
                    const double x = x0 + col * dx + x_shift;
                    if (x > bounds.hi.x - circumradius || timer.expired()) break;
                    const Vec3 c{x, y, z};
                    const std::size_t oi = static_cast<std::size_t>((row + 3 * layer + col) % static_cast<int>(orientations.size()));
                    if (overlaps(c)) continue;
                    if (insideWithMargin(c, oriented_cache[oi])) {
                        addPlacement(c, orientations[oi][0], orientations[oi][1], orientations[oi][2]);
                    }
                }
            }
        }
    }

    void fillAdaptiveCenters() {
        const auto orientations = orientationSet();
        std::vector<std::array<Vec3, 4>> oriented_cache;
        oriented_cache.reserve(orientations.size());
        for (const auto& ang : orientations) {
            oriented_cache.push_back(orientedVertices(edge, ang[0], ang[1], ang[2]));
        }

        const double span_x = bounds.hi.x - bounds.lo.x;
        const double span_y = bounds.hi.y - bounds.lo.y;
        const double span_z = bounds.hi.z - bounds.lo.z;
        const double box_volume = std::max(span_x * span_y * span_z, edge * edge * edge);
        const double target_samples = std::clamp(box_volume / std::max(tetra_volume, 1e-9) * 2.0, 5000.0, 250000.0);
        const double step = std::max(edge * 0.65, std::cbrt(box_volume / target_samples));

        std::vector<Vec3> frontier;
        frontier.reserve(static_cast<std::size_t>(target_samples / 2));
        for (double z = bounds.lo.z + circumradius; z <= bounds.hi.z - circumradius && !timer.expired(); z += step) {
            for (double y = bounds.lo.y + circumradius; y <= bounds.hi.y - circumradius && !timer.expired(); y += step) {
                for (double x = bounds.lo.x + circumradius; x <= bounds.hi.x - circumradius && !timer.expired(); x += step) {
                    frontier.push_back({x, y, z});
                }
            }
        }

        std::stable_sort(frontier.begin(), frontier.end(), [&](const Vec3& lhs, const Vec3& rhs) {
            return lhs.z < rhs.z;
        });

        for (std::size_t i = 0; i < frontier.size() && !timer.expired(); ++i) {
            const Vec3& c = frontier[i];
            if (overlaps(c)) continue;
            for (std::size_t oi = 0; oi < orientations.size(); ++oi) {
                if (insideWithMargin(c, oriented_cache[oi])) {
                    addPlacement(c, orientations[oi][0], orientations[oi][1], orientations[oi][2]);
                    break;
                }
            }
        }
    }

    void solve() {
        const double rough_capacity = volume / std::max(tetra_volume, 1e-9);
        if (vertex_count <= 24 && rough_capacity <= 250000.0) {
            fillGridRegular();
            if (!timer.expired() && out.size() < static_cast<std::size_t>(rough_capacity * 0.15)) {
                fillAdaptiveCenters();
            }
        } else {
            fillAdaptiveCenters();
        }
    }

    void printOutput() const {
        std::cout << out.size() << '\n';
        std::cout << std::fixed << std::setprecision(9);
        for (const Placement& p : out) {
            std::cout << p.center.x << ' ' << p.center.y << ' ' << p.center.z << ' '
                      << p.rx << ' ' << p.ry << ' ' << p.rz << '\n';
        }
    }
};

}  // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    Solver solver;
    if (!solver.readInput()) {
        std::cout << 0 << '\n';
        return 0;
    }
    if (!solver.prepare()) {
        std::cout << 0 << '\n';
        return 0;
    }
    solver.solve();
    solver.printOutput();
    return 0;
}
