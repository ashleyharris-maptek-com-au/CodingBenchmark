constexpr double kEps = 1e-9;

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator+(const Vec3& other) const { return {x + other.x, y + other.y, z + other.z}; }
    Vec3 operator-(const Vec3& other) const { return {x - other.x, y - other.y, z - other.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double norm(const Vec3& v) {
    return sqrt(dot(v, v));
}

struct Tri {
    int a = 0, b = 0, c = 0;
};

struct AABB {
    Vec3 mn{
        numeric_limits<double>::infinity(),
        numeric_limits<double>::infinity(),
        numeric_limits<double>::infinity()
    };
    Vec3 mx{
        -numeric_limits<double>::infinity(),
        -numeric_limits<double>::infinity(),
        -numeric_limits<double>::infinity()
    };

    void expand(const Vec3& p) {
        mn.x = min(mn.x, p.x);
        mn.y = min(mn.y, p.y);
        mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x);
        mx.y = max(mx.y, p.y);
        mx.z = max(mx.z, p.z);
    }

    void expand(const AABB& b) {
        expand(b.mn);
        expand(b.mx);
    }
};

struct Mesh {
    vector<Vec3> v;
    vector<Tri> f;
    AABB bounds;
};

static AABB triangle_bounds(const Mesh& m, const Tri& t) {
    AABB b;
    b.expand(m.v[t.a]);
    b.expand(m.v[t.b]);
    b.expand(m.v[t.c]);
    return b;
}

static bool overlap(const AABB& a, const AABB& b) {
    return a.mn.x <= b.mx.x + kEps && a.mx.x + kEps >= b.mn.x &&
           a.mn.y <= b.mx.y + kEps && a.mx.y + kEps >= b.mn.y &&
           a.mn.z <= b.mx.z + kEps && a.mx.z + kEps >= b.mn.z;
}

static Vec3 centroid(const Mesh& m, const Tri& t) {
    return (m.v[t.a] + m.v[t.b] + m.v[t.c]) / 3.0;
}

static bool ray_intersects_triangle(const Vec3& orig, const Vec3& dir,
                                    const Vec3& v0, const Vec3& v1, const Vec3& v2,
                                    double& outT) {
    Vec3 e1 = v1 - v0;
    Vec3 e2 = v2 - v0;
    Vec3 p = cross(dir, e2);
    double det = dot(e1, p);
    if (fabs(det) < kEps) return false;
    double invDet = 1.0 / det;
    Vec3 s = orig - v0;
    double u = dot(s, p) * invDet;
    if (u < -kEps || u > 1.0 + kEps) return false;
    Vec3 q = cross(s, e1);
    double v = dot(dir, q) * invDet;
    if (v < -kEps || u + v > 1.0 + kEps) return false;
    double t = dot(e2, q) * invDet;
    if (t < kEps) return false;
    outT = t;
    return true;
}

static bool point_inside_mesh(const Mesh& m, const Vec3& p) {
    const Vec3 dir(1.0, 0.17364817766693033, 0.3183098861837907);
    int hits = 0;
    for (const Tri& t : m.f) {
        double hitT = 0.0;
        if (ray_intersects_triangle(p, dir, m.v[t.a], m.v[t.b], m.v[t.c], hitT)) {
            ++hits;
        }
    }
    return (hits & 1) != 0;
}

static bool mesh_inside_mesh(const Mesh& inner, const Mesh& outer) {
    if (inner.f.empty()) return true;
    Vec3 c = centroid(inner, inner.f.front());
    return point_inside_mesh(outer, c);
}

static bool segment_triangle_intersection(const Vec3& p0, const Vec3& p1,
                                          const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 dir = p1 - p0;
    double t = 0.0;
    if (!ray_intersects_triangle(p0, dir, a, b, c, t)) return false;
    return t <= 1.0 + kEps;
}

static bool triangle_triangle_intersection(const Vec3& a0, const Vec3& a1, const Vec3& a2,
                                           const Vec3& b0, const Vec3& b1, const Vec3& b2) {
    const array<pair<Vec3, Vec3>, 3> ea = {{{a0, a1}, {a1, a2}, {a2, a0}}};
    const array<pair<Vec3, Vec3>, 3> eb = {{{b0, b1}, {b1, b2}, {b2, b0}}};
    for (const auto& e : ea) {
        if (segment_triangle_intersection(e.first, e.second, b0, b1, b2)) return true;
    }
    for (const auto& e : eb) {
        if (segment_triangle_intersection(e.first, e.second, a0, a1, a2)) return true;
    }
    Vec3 ca = (a0 + a1 + a2) / 3.0;
    Vec3 cb = (b0 + b1 + b2) / 3.0;
    return point_inside_mesh(Mesh{{b0, b1, b2}, {{0, 1, 2}}, {}}, ca) ||
           point_inside_mesh(Mesh{{a0, a1, a2}, {{0, 1, 2}}, {}}, cb);
}

static bool meshes_intersect_bruteforce(const Mesh& a, const Mesh& b, size_t limitChecks) {
    vector<AABB> ab(a.f.size()), bb(b.f.size());
    for (size_t i = 0; i < a.f.size(); ++i) ab[i] = triangle_bounds(a, a.f[i]);
    for (size_t i = 0; i < b.f.size(); ++i) bb[i] = triangle_bounds(b, b.f[i]);

    size_t checks = 0;
    for (size_t i = 0; i < a.f.size(); ++i) {
        for (size_t j = 0; j < b.f.size(); ++j) {
            if (++checks > limitChecks) return true;
            if (!overlap(ab[i], bb[j])) continue;
            const Tri& ta = a.f[i];
            const Tri& tb = b.f[j];
            if (triangle_triangle_intersection(a.v[ta.a], a.v[ta.b], a.v[ta.c],
                                               b.v[tb.a], b.v[tb.b], b.v[tb.c])) {
                return true;
            }
        }
    }
    return false;
}

struct GridKey {
    int x = 0, y = 0, z = 0;
    bool operator==(const GridKey& other) const {
        return x == other.x && y == other.y && z == other.z;
    }
};

struct GridKeyHash {
    size_t operator()(const GridKey& k) const {
        size_t h1 = std::hash<int>{}(k.x);
        size_t h2 = std::hash<int>{}(k.y);
        size_t h3 = std::hash<int>{}(k.z);
        return h1 * 1315423911u ^ (h2 << 1) ^ (h3 << 7);
    }
};

struct VoxelGrid {
    AABB bounds;
    int nx = 0, ny = 0, nz = 0;
    double dx = 1.0, dy = 1.0, dz = 1.0;
    vector<uint8_t> solid;
};

static size_t grid_index(const VoxelGrid& g, int x, int y, int z) {
    return (static_cast<size_t>(z) * g.ny + static_cast<size_t>(y)) * g.nx + static_cast<size_t>(x);
}

static Vec3 cell_center(const VoxelGrid& g, int x, int y, int z) {
    return {
        g.bounds.mn.x + (x + 0.5) * g.dx,
        g.bounds.mn.y + (y + 0.5) * g.dy,
        g.bounds.mn.z + (z + 0.5) * g.dz
    };
}

static VoxelGrid build_grid(const Mesh& a, const Mesh& b) {
    VoxelGrid g;
    g.bounds = a.bounds;
    g.bounds.expand(b.bounds);
    Vec3 ext = g.bounds.mx - g.bounds.mn;
    double volume = max(ext.x * ext.y * ext.z, 1e-6);
    size_t triCount = a.f.size() + b.f.size();
    double targetCells = std::clamp(4.0 * cbrt(static_cast<double>(triCount) + 1.0), 32.0, 192.0);
    double s = cbrt(volume / (targetCells * targetCells * targetCells));
    s = max(s, 1e-6);
    g.nx = max(1, min(192, static_cast<int>(ceil(ext.x / s)) + 2));
    g.ny = max(1, min(192, static_cast<int>(ceil(ext.y / s)) + 2));
    g.nz = max(1, min(192, static_cast<int>(ceil(ext.z / s)) + 2));
    g.dx = max(ext.x, 1e-6) / g.nx;
    g.dy = max(ext.y, 1e-6) / g.ny;
    g.dz = max(ext.z, 1e-6) / g.nz;
    g.solid.assign(static_cast<size_t>(g.nx) * g.ny * g.nz, 0);
    return g;
}

static void voxelize_union(VoxelGrid& g, const Mesh& a, const Mesh& b) {
    for (int z = 0; z < g.nz; ++z) {
        for (int y = 0; y < g.ny; ++y) {
            for (int x = 0; x < g.nx; ++x) {
                Vec3 p = cell_center(g, x, y, z);
                if (point_inside_mesh(a, p) || point_inside_mesh(b, p)) {
                    g.solid[grid_index(g, x, y, z)] = 1;
                }
            }
        }
    }
}

struct OutMesh {
    vector<Vec3> v;
    vector<Tri> f;
};

static int add_vertex(vector<Vec3>& verts, unordered_map<long long, int>& id, int gx, int gy, int gz,
                      const VoxelGrid& g) {
    long long key = (static_cast<long long>(gx) << 42) ^
                    (static_cast<long long>(gy) << 21) ^
                    static_cast<long long>(gz);
    auto it = id.find(key);
    if (it != id.end()) return it->second;
    Vec3 p{
        g.bounds.mn.x + gx * g.dx,
        g.bounds.mn.y + gy * g.dy,
        g.bounds.mn.z + gz * g.dz
    };
    int idx = static_cast<int>(verts.size());
    verts.push_back(p);
    id.emplace(key, idx);
    return idx;
}

static void emit_quad(OutMesh& out, unordered_map<long long, int>& id,
                      int x0, int y0, int z0, int x1, int y1, int z1,
                      int x2, int y2, int z2, int x3, int y3, int z3,
                      const VoxelGrid& g, bool flip) {
    int i0 = add_vertex(out.v, id, x0, y0, z0, g);
    int i1 = add_vertex(out.v, id, x1, y1, z1, g);
    int i2 = add_vertex(out.v, id, x2, y2, z2, g);
    int i3 = add_vertex(out.v, id, x3, y3, z3, g);
    if (!flip) {
        out.f.push_back({i0, i1, i2});
        out.f.push_back({i0, i2, i3});
    } else {
        out.f.push_back({i0, i2, i1});
        out.f.push_back({i0, i3, i2});
    }
}

static OutMesh extract_surface(const VoxelGrid& g) {
    OutMesh out;
    unordered_map<long long, int> vid;
    const int dx[6] = {1, -1, 0, 0, 0, 0};
    const int dy[6] = {0, 0, 1, -1, 0, 0};
    const int dz[6] = {0, 0, 0, 0, 1, -1};

    for (int z = 0; z < g.nz; ++z) {
        for (int y = 0; y < g.ny; ++y) {
            for (int x = 0; x < g.nx; ++x) {
                if (!g.solid[grid_index(g, x, y, z)]) continue;
                for (int dir = 0; dir < 6; ++dir) {
                    int nx = x + dx[dir], ny = y + dy[dir], nz = z + dz[dir];
                    bool exposed = nx < 0 || ny < 0 || nz < 0 || nx >= g.nx || ny >= g.ny || nz >= g.nz ||
                                   !g.solid[grid_index(g, nx, ny, nz)];
                    if (!exposed) continue;
                    switch (dir) {
                        case 0:
                            emit_quad(out, vid, x + 1, y, z, x + 1, y + 1, z, x + 1, y + 1, z + 1, x + 1, y, z + 1, g, false);
                            break;
                        case 1:
                            emit_quad(out, vid, x, y, z, x, y, z + 1, x, y + 1, z + 1, x, y + 1, z, g, false);
                            break;
                        case 2:
                            emit_quad(out, vid, x, y + 1, z, x, y + 1, z + 1, x + 1, y + 1, z + 1, x + 1, y + 1, z, g, false);
                            break;
                        case 3:
                            emit_quad(out, vid, x, y, z, x + 1, y, z, x + 1, y, z + 1, x, y, z + 1, g, true);
                            break;
                        case 4:
                            emit_quad(out, vid, x, y, z + 1, x + 1, y, z + 1, x + 1, y + 1, z + 1, x, y + 1, z + 1, g, false);
                            break;
                        case 5:
                            emit_quad(out, vid, x, y, z, x, y + 1, z, x + 1, y + 1, z, x + 1, y, z, g, false);
                            break;
                    }
                }
            }
        }
    }
    return out;
}

static Mesh read_mesh(int vc, int fc) {
    Mesh m;
    m.v.resize(vc);
    m.f.resize(fc);
    for (int i = 0; i < vc; ++i) {
        cin >> m.v[i].x >> m.v[i].y >> m.v[i].z;
        m.bounds.expand(m.v[i]);
    }
    for (int i = 0; i < fc; ++i) {
        cin >> m.f[i].a >> m.f[i].b >> m.f[i].c;
    }
    return m;
}

static OutMesh mesh_to_out(const Mesh& m) {
    return {m.v, m.f};
}

static OutMesh merge_meshes(const Mesh& a, const Mesh& b) {
    OutMesh out;
    out.v = a.v;
    out.f = a.f;
    int base = static_cast<int>(out.v.size());
    out.v.insert(out.v.end(), b.v.begin(), b.v.end());
    for (const Tri& t : b.f) {
        out.f.push_back({t.a + base, t.b + base, t.c + base});
    }
    return out;
}

static OutMesh solve_union(const Mesh& a, const Mesh& b) {
    if (!overlap(a.bounds, b.bounds)) {
        return merge_meshes(a, b);
    }

    bool aInB = mesh_inside_mesh(a, b);
    if (aInB) return mesh_to_out(b);

    bool bInA = mesh_inside_mesh(b, a);
    if (bInA) return mesh_to_out(a);

    size_t triProduct = a.f.size() * b.f.size();
    bool probablyIntersect = true;
    if (triProduct <= 3000000ULL) {
        probablyIntersect = meshes_intersect_bruteforce(a, b, 3000000ULL);
        if (!probablyIntersect) {
            return merge_meshes(a, b);
        }
    }

    VoxelGrid grid = build_grid(a, b);
    voxelize_union(grid, a, b);
    return extract_surface(grid);
}

}  // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int va, fa, vb, fb;
    if (!(cin >> va >> fa >> vb >> fb)) {
        return 0;
    }

    Mesh a = read_mesh(va, fa);
    Mesh b = read_mesh(vb, fb);
    OutMesh out = solve_union(a, b);

    cout << out.v.size() << ' ' << out.f.size() << '\n';
    cout << setprecision(17);
    for (const Vec3& p : out.v) {
        cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    }
    for (const Tri& t : out.f) {
        cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
    }
    return 0;
}
