using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;
using System.Numerics;

public static class Program
{
    private static readonly QuaternionInfo[] AxisAlignedRotations = BuildAxisAlignedRotations();

    public static void Main()
    {
        var input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
        {
            Console.WriteLine(0);
            return;
        }

        var scanner = new FastScanner(input);
        int vertexCount = scanner.NextInt();
        int faceCount = scanner.NextInt();

        var vertices = new Vector3[vertexCount];
        for (int i = 0; i < vertexCount; i++)
        {
            vertices[i] = new Vector3(scanner.NextFloat(), scanner.NextFloat(), scanner.NextFloat());
        }

        var faces = new (int A, int B, int C)[faceCount];
        for (int i = 0; i < faceCount; i++)
        {
            faces[i] = (scanner.NextInt(), scanner.NextInt(), scanner.NextInt());
        }

        int containerW = scanner.NextInt();
        int containerH = scanner.NextInt();
        int containerD = scanner.NextInt();

        var poly = PolyhedronAnalysis.Analyze(vertices, faces);
        var solver = new AdaptiveSolver(poly, containerW, containerH, containerD);
        var placements = solver.Solve();

        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine(FormattableString.Invariant(
                $"{p.Tx:0.######} {p.Ty:0.######} {p.Tz:0.######} {p.Qw:0.######} {p.Qx:0.######} {p.Qy:0.######} {p.Qz:0.######}"
            ));
        }
    }

    private static QuaternionInfo[] BuildAxisAlignedRotations()
    {
        var basis = new[]
        {
            new Vector3(1, 0, 0),
            new Vector3(0, 1, 0),
            new Vector3(0, 0, 1),
        };

        var result = new List<QuaternionInfo>();
        foreach (var xAxis in SignedAxes())
        {
            foreach (var yAxis in SignedAxes())
            {
                if (Math.Abs(Vector3.Dot(xAxis, yAxis)) > 0.001f)
                {
                    continue;
                }

                var zAxis = Vector3.Cross(xAxis, yAxis);
                if (zAxis.LengthSquared() < 0.5f)
                {
                    continue;
                }

                var matrix = new Matrix4x4(
                    xAxis.X, xAxis.Y, xAxis.Z, 0,
                    yAxis.X, yAxis.Y, yAxis.Z, 0,
                    zAxis.X, zAxis.Y, zAxis.Z, 0,
                    0, 0, 0, 1
                );

                var q = Quaternion.Normalize(Quaternion.CreateFromRotationMatrix(matrix));
                bool duplicate = result.Any(r => Quaternion.Dot(r.Quaternion, q) > 0.9999f || Quaternion.Dot(r.Quaternion, -q) > 0.9999f);
                if (!duplicate)
                {
                    result.Add(new QuaternionInfo(q));
                }
            }
        }

        return result.ToArray();
    }

    private static IEnumerable<Vector3> SignedAxes()
    {
        yield return new Vector3(1, 0, 0);
        yield return new Vector3(-1, 0, 0);
        yield return new Vector3(0, 1, 0);
        yield return new Vector3(0, -1, 0);
        yield return new Vector3(0, 0, 1);
        yield return new Vector3(0, 0, -1);
    }

    private sealed class AdaptiveSolver
    {
        private readonly PolyhedronAnalysis _poly;
        private readonly int _w;
        private readonly int _h;
        private readonly int _d;

        public AdaptiveSolver(PolyhedronAnalysis poly, int w, int h, int d)
        {
            _poly = poly;
            _w = w;
            _h = h;
            _d = d;
        }

        public List<Placement> Solve()
        {
            long containerVolume = 1L * _w * _h * _d;
            long bboxVolume = 1L * Math.Max(1, _poly.BaseDims.Xi) * Math.Max(1, _poly.BaseDims.Yi) * Math.Max(1, _poly.BaseDims.Zi);
            double complexity = _poly.Vertices.Length + _poly.Faces.Length * 2.0;

            if (containerVolume <= 250_000 && complexity <= 500)
            {
                var exact = SolveExactVoxel();
                if (exact.Count > 0)
                {
                    return exact;
                }
            }

            if (containerVolume <= 8_000_000 && bboxVolume > 0)
            {
                return SolveGridPacked();
            }

            return SolveLargeScale();
        }

        private List<Placement> SolveExactVoxel()
        {
            var oriented = BuildOrientedShapes(limitToIntegerRotations: true, voxelize: true);
            if (oriented.Count == 0)
            {
                return new List<Placement>();
            }

            var occupancy = new bool[_w, _h, _d];
            var placements = new List<Placement>();

            while (true)
            {
                bool found = false;
                Placement bestPlacement = default;
                OrientedShape bestShape = default;

                for (int z = 0; z < _d && !found; z++)
                {
                    for (int y = 0; y < _h && !found; y++)
                    {
                        for (int x = 0; x < _w && !found; x++)
                        {
                            if (occupancy[x, y, z])
                            {
                                continue;
                            }

                            foreach (var shape in oriented)
                            {
                                if (x + shape.Dims.Xi > _w || y + shape.Dims.Yi > _h || z + shape.Dims.Zi > _d)
                                {
                                    continue;
                                }

                                if (!CanPlace(shape, x, y, z, occupancy))
                                {
                                    continue;
                                }

                                bestShape = shape;
                                bestPlacement = new Placement(
                                    x - shape.Min.X,
                                    y - shape.Min.Y,
                                    z - shape.Min.Z,
                                    shape.Quaternion.W,
                                    shape.Quaternion.X,
                                    shape.Quaternion.Y,
                                    shape.Quaternion.Z
                                );
                                found = true;
                                break;
                            }
                        }
                    }
                }

                if (!found)
                {
                    break;
                }

                Stamp(bestShape, bestPlacement, occupancy);
                placements.Add(bestPlacement);
            }

            return placements;
        }

        private List<Placement> SolveGridPacked()
        {
            var oriented = BuildOrientedShapes(limitToIntegerRotations: true, voxelize: false);
            var best = SelectBestOrientation(oriented, useVolumeScore: true);
            if (oriented.Count == 0)
            {
                return new List<Placement>();
            }

            int nx = _w / best.Dims.Xi;
            int ny = _h / best.Dims.Yi;
            int nz = _d / best.Dims.Zi;
            var placements = new List<Placement>(Math.Max(0, nx * ny * nz));

            for (int z = 0; z < nz; z++)
            {
                for (int y = 0; y < ny; y++)
                {
                    for (int x = 0; x < nx; x++)
                    {
                        placements.Add(new Placement(
                            x * best.Dims.Xi - best.Min.X,
                            y * best.Dims.Yi - best.Min.Y,
                            z * best.Dims.Zi - best.Min.Z,
                            best.Quaternion.W,
                            best.Quaternion.X,
                            best.Quaternion.Y,
                            best.Quaternion.Z
                        ));
                    }
                }
            }

            return placements;
        }

        private List<Placement> SolveLargeScale()
        {
            var oriented = BuildOrientedShapes(limitToIntegerRotations: true, voxelize: false);
            var best = SelectBestOrientation(oriented, useVolumeScore: false);
            if (oriented.Count == 0)
            {
                return new List<Placement>();
            }

            long nx = _w / best.Dims.Xi;
            long ny = _h / best.Dims.Yi;
            long nz = _d / best.Dims.Zi;
            long count = nx * ny * nz;

            const long hardCap = 2_000_000;
            if (count > hardCap)
            {
                long reducedNz = Math.Max(1, hardCap / Math.Max(1, nx * ny));
                nz = Math.Min(nz, reducedNz);
                count = nx * ny * nz;
            }

            var placements = new List<Placement>((int)Math.Min(count, int.MaxValue));
            for (long z = 0; z < nz; z++)
            {
                for (long y = 0; y < ny; y++)
                {
                    for (long x = 0; x < nx; x++)
                    {
                        placements.Add(new Placement(
                            x * best.Dims.Xi - best.Min.X,
                            y * best.Dims.Yi - best.Min.Y,
                            z * best.Dims.Zi - best.Min.Z,
                            best.Quaternion.W,
                            best.Quaternion.X,
                            best.Quaternion.Y,
                            best.Quaternion.Z
                        ));
                    }
                }
            }

            return placements;
        }

        private static bool CanPlace(OrientedShape shape, int ox, int oy, int oz, bool[,,] occupancy)
        {
            foreach (var c in shape.Cells!)
            {
                if (occupancy[ox + c.X, oy + c.Y, oz + c.Z])
                {
                    return false;
                }
            }

            return true;
        }

        private static void Stamp(OrientedShape shape, Placement placement, bool[,,] occupancy)
        {
            int ox = (int)Math.Round(placement.Tx + shape.Min.X);
            int oy = (int)Math.Round(placement.Ty + shape.Min.Y);
            int oz = (int)Math.Round(placement.Tz + shape.Min.Z);
            foreach (var c in shape.Cells!)
            {
                occupancy[ox + c.X, oy + c.Y, oz + c.Z] = true;
            }
        }

        private List<OrientedShape> BuildOrientedShapes(bool limitToIntegerRotations, bool voxelize)
        {
            var result = new List<OrientedShape>();
            foreach (var qInfo in AxisAlignedRotations)
            {
                var rotated = new Vector3[_poly.Vertices.Length];
                float minX = float.PositiveInfinity;
                float minY = float.PositiveInfinity;
                float minZ = float.PositiveInfinity;
                float maxX = float.NegativeInfinity;
                float maxY = float.NegativeInfinity;
                float maxZ = float.NegativeInfinity;

                for (int i = 0; i < _poly.Vertices.Length; i++)
                {
                    var v = Vector3.Transform(_poly.Vertices[i], qInfo.Quaternion);
                    rotated[i] = v;
                    minX = MathF.Min(minX, v.X);
                    minY = MathF.Min(minY, v.Y);
                    minZ = MathF.Min(minZ, v.Z);
                    maxX = MathF.Max(maxX, v.X);
                    maxY = MathF.Max(maxY, v.Y);
                    maxZ = MathF.Max(maxZ, v.Z);
                }

                int dx = Math.Max(1, (int)Math.Ceiling(maxX - minX - 1e-6));
                int dy = Math.Max(1, (int)Math.Ceiling(maxY - minY - 1e-6));
                int dz = Math.Max(1, (int)Math.Ceiling(maxZ - minZ - 1e-6));

                if (dx > _w || dy > _h || dz > _d)
                {
                    continue;
                }

                VoxelCell[]? cells = null;
                if (voxelize)
                {
                    cells = Voxelizer.Voxelize(rotated, _poly.Faces, minX, minY, minZ, dx, dy, dz);
                    if (cells.Length == 0)
                    {
                        continue;
                    }
                }

                result.Add(new OrientedShape(
                    qInfo.Quaternion,
                    new Int3(dx, dy, dz),
                    new Vector3(minX, minY, minZ),
                    cells
                ));
            }

            return result;
        }

        private static OrientedShape SelectBestOrientation(List<OrientedShape> shapes, bool useVolumeScore)
        {
            if (shapes.Count == 0)
            {
                return default;
            }

            OrientedShape best = shapes[0];
            double bestScore = double.NegativeInfinity;
            foreach (var s in shapes)
            {
                double score = useVolumeScore
                    ? 1.0 / (s.Dims.Xi * s.Dims.Yi * s.Dims.Zi)
                    : 1.0 / (s.Dims.Xi + s.Dims.Yi + s.Dims.Zi);
                if (score > bestScore)
                {
                    best = s;
                    bestScore = score;
                }
            }

            return best;
        }
    }

    private sealed class PolyhedronAnalysis
    {
        public Vector3[] Vertices { get; }
        public (int A, int B, int C)[] Faces { get; }
        public double Volume { get; }
        public Int3 BaseDims { get; }

        private PolyhedronAnalysis(Vector3[] vertices, (int A, int B, int C)[] faces, double volume, Int3 baseDims)
        {
            Vertices = vertices;
            Faces = faces;
            Volume = volume;
            BaseDims = baseDims;
        }

        public static PolyhedronAnalysis Analyze(Vector3[] vertices, (int A, int B, int C)[] faces)
        {
            double signedVolume = 0;
            foreach (var (a, b, c) in faces)
            {
                var va = vertices[a];
                var vb = vertices[b];
                var vc = vertices[c];
                signedVolume += Vector3.Dot(va, Vector3.Cross(vb, vc)) / 6.0;
            }

            float minX = vertices.Min(v => v.X);
            float minY = vertices.Min(v => v.Y);
            float minZ = vertices.Min(v => v.Z);
            float maxX = vertices.Max(v => v.X);
            float maxY = vertices.Max(v => v.Y);
            float maxZ = vertices.Max(v => v.Z);
            int dx = Math.Max(1, (int)Math.Ceiling(maxX - minX - 1e-6));
            int dy = Math.Max(1, (int)Math.Ceiling(maxY - minY - 1e-6));
            int dz = Math.Max(1, (int)Math.Ceiling(maxZ - minZ - 1e-6));

            return new PolyhedronAnalysis(vertices, faces, Math.Abs(signedVolume), new Int3(dx, dy, dz));
        }
    }

    private static class Voxelizer
    {
        public static VoxelCell[] Voxelize(Vector3[] vertices, (int A, int B, int C)[] faces, float minX, float minY, float minZ, int dx, int dy, int dz)
        {
            var cells = new List<VoxelCell>(dx * dy * dz);
            for (int z = 0; z < dz; z++)
            {
                for (int y = 0; y < dy; y++)
                {
                    for (int x = 0; x < dx; x++)
                    {
                        var point = new Vector3(minX + x + 0.5f, minY + y + 0.5f, minZ + z + 0.5f);
                        if (PointInMesh(point, vertices, faces))
                        {
                            cells.Add(new VoxelCell(x, y, z));
                        }
                    }
                }
            }

            return cells.ToArray();
        }

        private static bool PointInMesh(Vector3 p, Vector3[] vertices, (int A, int B, int C)[] faces)
        {
            var dir = new Vector3(1.0f, 0.3125f, 0.15625f);
            int hits = 0;
            foreach (var (a, b, c) in faces)
            {
                if (RayIntersectsTriangle(p, dir, vertices[a], vertices[b], vertices[c]))
                {
                    hits++;
                }
            }

            return (hits & 1) == 1;
        }

        private static bool RayIntersectsTriangle(Vector3 origin, Vector3 direction, Vector3 v0, Vector3 v1, Vector3 v2)
        {
            const float epsilon = 1e-6f;
            var e1 = v1 - v0;
            var e2 = v2 - v0;
            var pvec = Vector3.Cross(direction, e2);
            float det = Vector3.Dot(e1, pvec);
            if (MathF.Abs(det) < epsilon)
            {
                return false;
            }

            float invDet = 1.0f / det;
            var tvec = origin - v0;
            float u = Vector3.Dot(tvec, pvec) * invDet;
            if (u < 0 || u > 1)
            {
                return false;
            }

            var qvec = Vector3.Cross(tvec, e1);
            float v = Vector3.Dot(direction, qvec) * invDet;
            if (v < 0 || u + v > 1)
            {
                return false;
            }

            float t = Vector3.Dot(e2, qvec) * invDet;
            return t > epsilon;
        }
    }

    private readonly record struct QuaternionInfo(Quaternion Quaternion);
    private readonly record struct Int3(int Xi, int Yi, int Zi);
    private readonly record struct VoxelCell(int X, int Y, int Z);
    private readonly record struct Placement(double Tx, double Ty, double Tz, double Qw, double Qx, double Qy, double Qz);
    private readonly record struct OrientedShape(Quaternion Quaternion, Int3 Dims, Vector3 Min, VoxelCell[]? Cells);

    private ref struct FastScanner
    {
        private readonly ReadOnlySpan<char> _span;
        private int _index;

        public FastScanner(string input)
        {
            _span = input.AsSpan();
            _index = 0;
        }

        public int NextInt()
        {
            SkipWhite();
            int sign = 1;
            if (_span[_index] == '-')
            {
                sign = -1;
                _index++;
            }

            int value = 0;
            while (_index < _span.Length && char.IsDigit(_span[_index]))
            {
                value = value * 10 + (_span[_index] - '0');
                _index++;
            }

            return value * sign;
        }

        public float NextFloat()
        {
            SkipWhite();
            int start = _index;
            if (_span[_index] == '-' || _span[_index] == '+')
            {
                _index++;
            }

            while (_index < _span.Length && (char.IsDigit(_span[_index]) || _span[_index] == '.'
                || _span[_index] == 'e' || _span[_index] == 'E' || _span[_index] == '-' || _span[_index] == '+'))
            {
                if ((_span[_index] == '-' || _span[_index] == '+') && _index > start && _span[_index - 1] != 'e' && _span[_index - 1] != 'E')
                {
                    break;
                }
                _index++;
            }

            return float.Parse(_span[start.._index], CultureInfo.InvariantCulture);
        }

        private void SkipWhite()
        {
            while (_index < _span.Length && char.IsWhiteSpace(_span[_index]))
            {
                _index++;
            }
        }
    }
}
