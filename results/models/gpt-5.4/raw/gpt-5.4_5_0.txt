use std::collections::{HashSet, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
struct Pt {
    x: i32,
    y: i32,
}

impl Pt {
    fn manhattan(self, other: Pt) -> i32 {
        (self.x - other.x).abs() + (self.y - other.y).abs()
    }
}

const DIRS: [(i32, i32); 4] = [(1, 0), (-1, 0), (0, 1), (0, -1)];
const SEARCH_LIMIT: usize = 64;

struct Solver {
    w: i32,
    h: i32,
    require_cycle: bool,
    blocked: Vec<bool>,
    free_cells: Vec<Pt>,
    total_free: usize,
}

impl Solver {
    fn new(w: i32, h: i32, require_cycle: bool, blocked_points: &[Pt]) -> Self {
        let mut blocked = vec![false; (w * h) as usize];
        for &p in blocked_points {
            if p.x >= 0 && p.x < w && p.y >= 0 && p.y < h {
                blocked[(p.y * w + p.x) as usize] = true;
            }
        }
        let mut free_cells = Vec::new();
        for y in 0..h {
            for x in 0..w {
                let p = Pt { x, y };
                if !blocked[(y * w + x) as usize] {
                    free_cells.push(p);
                }
            }
        }
        let total_free = free_cells.len();
        Self {
            w,
            h,
            require_cycle,
            blocked,
            free_cells,
            total_free,
        }
    }

    fn idx(&self, p: Pt) -> usize {
        (p.y * self.w + p.x) as usize
    }

    fn in_bounds(&self, p: Pt) -> bool {
        p.x >= 0 && p.x < self.w && p.y >= 0 && p.y < self.h
    }

    fn is_free(&self, p: Pt) -> bool {
        self.in_bounds(p) && !self.blocked[self.idx(p)]
    }

    fn neighbors(&self, p: Pt) -> impl Iterator<Item = Pt> + '_ {
        DIRS.into_iter().filter_map(move |(dx, dy)| {
            let q = Pt {
                x: p.x + dx,
                y: p.y + dy,
            };
            if self.is_free(q) { Some(q) } else { None }
        })
    }

    fn parity(&self, p: Pt) -> i32 {
        (p.x + p.y) & 1
    }

    fn quick_feasibility(&self) -> bool {
        let start = Pt { x: 0, y: 0 };
        if !self.is_free(start) || self.total_free == 0 {
            return false;
        }

        let reachable = self.component_from(start, None, None);
        if reachable.len() != self.total_free {
            return false;
        }

        let mut black = 0usize;
        let mut white = 0usize;
        for &p in &self.free_cells {
            if self.parity(p) == 0 {
                black += 1;
            } else {
                white += 1;
            }
        }
        if self.require_cycle {
            if self.total_free == 1 || black != white {
                return false;
            }
        } else if (black as isize - white as isize).abs() > 1 {
            return false;
        }
        true
    }

    fn component_from(&self, start: Pt, banned: Option<Pt>, allowed_end: Option<Pt>) -> Vec<Pt> {
        let mut seen = vec![false; (self.w * self.h) as usize];
        let mut out = Vec::new();
        let mut q = VecDeque::new();
        if Some(start) == banned && Some(start) != allowed_end {
            return out;
        }
        seen[self.idx(start)] = true;
        q.push_back(start);
        while let Some(p) = q.pop_front() {
            out.push(p);
            for n in self.neighbors(p) {
                if Some(n) == banned && Some(n) != allowed_end {
                    continue;
                }
                let id = self.idx(n);
                if !seen[id] {
                    seen[id] = true;
                    q.push_back(n);
                }
            }
        }
        out
    }

    fn rectangular_snake(&self) -> Option<Vec<Pt>> {
        if self.total_free != (self.w * self.h) as usize {
            return None;
        }
        let start = Pt { x: 0, y: 0 };
        if !self.is_free(start) {
            return None;
        }

        if let Some(path) = self.rect_path_rows() {
            return Some(path);
        }
        if let Some(path) = self.rect_path_cols() {
            return Some(path);
        }
        None
    }

    fn rect_path_rows(&self) -> Option<Vec<Pt>> {
        if self.h <= 0 || self.w <= 0 {
            return None;
        }
        if self.h == 1 {
            let mut path = Vec::new();
            for x in 0..self.w {
                path.push(Pt { x, y: 0 });
            }
            return self.accept_path(path);
        }
        if self.w % 2 == 1 {
            let mut path = Vec::with_capacity(self.total_free);
            for y in 0..self.h {
                if y % 2 == 0 {
                    for x in 0..self.w {
                        path.push(Pt { x, y });
                    }
                } else {
                    for x in (0..self.w).rev() {
                        path.push(Pt { x, y });
                    }
                }
            }
            return self.accept_path(path);
        }
        None
    }

    fn rect_path_cols(&self) -> Option<Vec<Pt>> {
        if self.h <= 0 || self.w <= 0 {
            return None;
        }
        if self.w == 1 {
            let mut path = Vec::new();
            for y in 0..self.h {
                path.push(Pt { x: 0, y });
            }
            return self.accept_path(path);
        }
        if self.h % 2 == 1 {
            let mut path = Vec::with_capacity(self.total_free);
            for x in 0..self.w {
                if x % 2 == 0 {
                    for y in 0..self.h {
                        path.push(Pt { x, y });
                    }
                } else {
                    for y in (0..self.h).rev() {
                        path.push(Pt { x, y });
                    }
                }
            }
            return self.accept_path(path);
        }
        None
    }

    fn rect_cycle(&self) -> Option<Vec<Pt>> {
        if self.total_free != (self.w * self.h) as usize {
            return None;
        }
        if self.total_free < 4 || self.w <= 1 || self.h <= 1 || self.total_free % 2 == 1 {
            return None;
        }

        if self.h % 2 == 0 {
            let mut path = Vec::with_capacity(self.total_free);
            for y in 0..self.h {
                if y % 2 == 0 {
                    path.push(Pt { x: 0, y });
                } else {
                    for x in 1..self.w {
                        path.push(Pt { x, y });
                    }
                }
            }
            for y in (0..self.h).rev() {
                if y % 2 == 0 {
                    for x in (1..self.w).rev() {
                        path.push(Pt { x, y });
                    }
                } else {
                    path.push(Pt { x: 0, y });
                }
            }
            return self.accept_path(path);
        }

        if self.w % 2 == 0 {
            let mut path = Vec::with_capacity(self.total_free);
            for x in 0..self.w {
                if x % 2 == 0 {
                    path.push(Pt { x, y: 0 });
                } else {
                    for y in 1..self.h {
                        path.push(Pt { x, y });
                    }
                }
            }
            for x in (0..self.w).rev() {
                if x % 2 == 0 {
                    for y in (1..self.h).rev() {
                        path.push(Pt { x, y });
                    }
                } else {
                    path.push(Pt { x, y: 0 });
                }
            }
            return self.accept_path(path);
        }

        None
    }

    fn accept_path(&self, path: Vec<Pt>) -> Option<Vec<Pt>> {
        if path.first().copied() != Some(Pt { x: 0, y: 0 }) {
            return None;
        }
        if path.len() != self.total_free {
            return None;
        }
        let mut seen = HashSet::with_capacity(path.len());
        for window in path.windows(2) {
            if window[0].manhattan(window[1]) != 1 {
                return None;
            }
        }
        for &p in &path {
            if !self.is_free(p) || !seen.insert(p) {
                return None;
            }
        }
        if self.require_cycle && path.last().unwrap().manhattan(path[0]) != 1 {
            return None;
        }
        Some(path)
    }

    fn solve(&self) -> Option<Vec<Pt>> {
        if !self.quick_feasibility() {
            return None;
        }

        if self.require_cycle {
            if let Some(path) = self.rect_cycle() {
                return Some(path);
            }
        } else if let Some(path) = self.rectangular_snake() {
            return Some(path);
        }

        if self.total_free <= SEARCH_LIMIT {
            return self.solve_by_search();
        }

        None
    }

    fn solve_by_search(&self) -> Option<Vec<Pt>> {
        let start = Pt { x: 0, y: 0 };
        let mut visited = vec![false; (self.w * self.h) as usize];
        let mut path = Vec::with_capacity(self.total_free);
        visited[self.idx(start)] = true;
        path.push(start);
        if self.dfs(start, &mut visited, &mut path) {
            Some(path)
        } else {
            None
        }
    }

    fn dfs(&self, cur: Pt, visited: &mut [bool], path: &mut Vec<Pt>) -> bool {
        if path.len() == self.total_free {
            return !self.require_cycle || cur.manhattan(Pt { x: 0, y: 0 }) == 1;
        }

        let mut candidates: Vec<Pt> = self
            .neighbors(cur)
            .filter(|&n| !visited[self.idx(n)])
            .collect();

        candidates.sort_by_key(|&n| {
            let onward = self
                .neighbors(n)
                .filter(|&m| !visited[self.idx(m)] && m != cur)
                .count();
            let back_to_start = if self.require_cycle && n.manhattan(Pt { x: 0, y: 0 }) == 1 {
                0usize
            } else {
                1usize
            };
            (onward, back_to_start)
        });

        for next in candidates {
            visited[self.idx(next)] = true;
            path.push(next);
            if self.search_state_ok(cur, next, visited, path.len()) && self.dfs(next, visited, path)
            {
                return true;
            }
            path.pop();
            visited[self.idx(next)] = false;
        }
        false
    }

    fn search_state_ok(&self, prev: Pt, cur: Pt, visited: &[bool], used: usize) -> bool {
        let mut unvisited = Vec::new();
        let mut black = 0usize;
        let mut white = 0usize;
        for &p in &self.free_cells {
            if !visited[self.idx(p)] {
                if self.parity(p) == 0 {
                    black += 1;
                } else {
                    white += 1;
                }
                unvisited.push(p);
            }
        }

        let start = Pt { x: 0, y: 0 };
        if self.require_cycle {
            if used < self.total_free && !unvisited.iter().any(|&p| p.manhattan(start) == 1) {
                return false;
            }
            if (black as isize - white as isize).abs() > 1 {
                return false;
            }
        } else {
            let cur_par = self.parity(cur);
            let diff = black as isize - white as isize;
            if self.total_free - used == 0 {
                return true;
            }
            if diff.abs() > 1 {
                return false;
            }
            if diff == 1 && cur_par != 1 {
                return false;
            }
            if diff == -1 && cur_par != 0 {
                return false;
            }
        }

        if unvisited.is_empty() {
            return true;
        }

        let seed = unvisited[0];
        let mut seen = HashSet::new();
        let mut q = VecDeque::new();
        seen.insert(seed);
        q.push_back(seed);
        while let Some(p) = q.pop_front() {
            for n in self.neighbors(p) {
                if visited[self.idx(n)] {
                    continue;
                }
                if seen.insert(n) {
                    q.push_back(n);
                }
            }
        }
        if seen.len() != unvisited.len() {
            return false;
        }

        let mut forced_degree_one = 0usize;
        for &p in &unvisited {
            let deg = self
                .neighbors(p)
                .filter(|&n| !visited[self.idx(n)] || n == cur || n == prev)
                .count();
            if deg == 0 {
                return false;
            }
            if deg == 1 {
                forced_degree_one += 1;
                if self.require_cycle || forced_degree_one > 1 {
                    return false;
                }
            }
        }
        true
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        println!("0");
        return;
    }
    let mut it = input.split_whitespace();
    let w: i32 = it.next().unwrap().parse().unwrap();
    let h: i32 = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle = match it.next().unwrap() {
        "0" => false,
        "1" => true,
        v => panic!("invalid require_cycle flag: {v}"),
    };
    let mut blocked = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        let x: i32 = it.next().unwrap().parse().unwrap();
        let y: i32 = it.next().unwrap().parse().unwrap();
        blocked.push(Pt { x, y });
    }

    let solver = Solver::new(w, h, require_cycle, &blocked);
    if let Some(path) = solver.solve() {
        println!("{}", path.len());
        for p in path {
            println!("{} {}", p.x, p.y);
        }
    } else {
        println!("0");
    }
}
