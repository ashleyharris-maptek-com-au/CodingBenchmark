use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct Graph {
    n: usize,
    closed: Vec<Vec<usize>>,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let graph = parse_graph(&input);
    let solution = solve(&graph);

    println!("{}", solution.len());
    if solution.is_empty() {
        println!();
    } else {
        let line = solution
            .iter()
            .map(|v| v.to_string())
            .collect::<Vec<_>>()
            .join(" ");
        println!("{line}");
    }
}

fn parse_graph(input: &str) -> Graph {
    let mut it = input.split_whitespace();
    let n = it.next().and_then(|x| x.parse::<usize>().ok()).unwrap_or(0);
    let m = it.next().and_then(|x| x.parse::<usize>().ok()).unwrap_or(0);

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let Some(us) = it.next() else { break };
        let Some(vs) = it.next() else { break };
        let Ok(u) = us.parse::<usize>() else { continue };
        let Ok(v) = vs.parse::<usize>() else { continue };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    for nbrs in &mut adj {
        nbrs.sort_unstable();
        nbrs.dedup();
    }

    let mut closed = vec![Vec::<usize>::new(); n];
    for v in 0..n {
        let mut c = Vec::with_capacity(adj[v].len() + 1);
        c.push(v);
        c.extend_from_slice(&adj[v]);
        closed[v] = c;
    }

    Graph { n, closed }
}

fn solve(graph: &Graph) -> Vec<usize> {
    if graph.n == 0 {
        return Vec::new();
    }

    let start = Instant::now();
    let budget = Duration::from_millis(4500);

    let mut best = greedy_solution(graph, None);
    improve(graph, &mut best, start, budget);

    let mut degree_order: Vec<usize> = (0..graph.n).collect();
    degree_order.sort_unstable_by_key(|&v| std::cmp::Reverse(graph.closed[v].len()));
    let candidate_limit = degree_order.len().min(12);

    for &seed in degree_order.iter().take(candidate_limit) {
        if start.elapsed() >= budget {
            break;
        }
        let mut candidate = greedy_solution(graph, Some(seed));
        improve(graph, &mut candidate, start, budget);
        if candidate.len() < best.len() {
            best = candidate;
        }
    }

    best.sort_unstable();
    best
}

fn greedy_solution(graph: &Graph, forced_seed: Option<usize>) -> Vec<usize> {
    let n = graph.n;
    let mut covered = vec![false; n];
    let mut cover_count = vec![0u16; n];
    let mut in_set = vec![false; n];
    let mut solution = Vec::new();
    let mut uncovered = n;

    if let Some(seed) = forced_seed {
        if seed < n {
            add_vertex(graph, seed, &mut in_set, &mut covered, &mut cover_count, &mut uncovered, &mut solution);
        }
    }

    while uncovered > 0 {
        let mut best_v = None;
        let mut best_gain = 0usize;
        let mut best_secondary = 0usize;

        for v in 0..n {
            if in_set[v] {
                continue;
            }
            let gain = graph.closed[v].iter().filter(|&&u| !covered[u]).count();
            if gain == 0 {
                continue;
            }
            let secondary = graph.closed[v].len();
            if gain > best_gain || (gain == best_gain && secondary > best_secondary) {
                best_gain = gain;
                best_secondary = secondary;
                best_v = Some(v);
            }
        }

        let chosen = if let Some(v) = best_v {
            v
        } else {
            let mut fallback = 0usize;
            while fallback < n && covered[fallback] {
                fallback += 1;
            }
            if fallback == n {
                break;
            }
            let mut best = fallback;
            let mut best_deg = graph.closed[fallback].len();
            for &nbr in &graph.closed[fallback] {
                let deg = graph.closed[nbr].len();
                if deg > best_deg {
                    best = nbr;
                    best_deg = deg;
                }
            }
            best
        };

        add_vertex(graph, chosen, &mut in_set, &mut covered, &mut cover_count, &mut uncovered, &mut solution);
    }

    solution
}

fn improve(graph: &Graph, solution: &mut Vec<usize>, start: Instant, budget: Duration) {
    let n = graph.n;
    let mut in_set = vec![false; n];
    let mut cover_count = vec![0u16; n];
    for &v in solution.iter() {
        in_set[v] = true;
        for &u in &graph.closed[v] {
            cover_count[u] += 1;
        }
    }

    prune_redundant(graph, solution, &mut in_set, &mut cover_count);
    if start.elapsed() >= budget {
        return;
    }

    let mut changed = true;
    while changed && start.elapsed() < budget {
        changed = false;

        let mut order = solution.clone();
        order.sort_unstable_by_key(|&v| graph.closed[v].len());
        for v in order {
            if start.elapsed() >= budget {
                break;
            }
            if !in_set[v] {
                continue;
            }

            remove_vertex(graph, v, &mut in_set, &mut cover_count, solution);
            let undominated = first_undominated(&cover_count);
            if undominated.is_none() {
                changed = true;
                continue;
            }

            let target = undominated.unwrap();
            let mut best_rep = None;
            let mut best_gain = 0usize;
            let mut best_overlap = 0usize;

            for &cand in &graph.closed[target] {
                if in_set[cand] {
                    continue;
                }
                let gain = graph.closed[cand]
                    .iter()
                    .filter(|&&u| cover_count[u] == 0)
                    .count();
                if gain == 0 {
                    continue;
                }
                let overlap = graph.closed[cand]
                    .iter()
                    .filter(|&&u| cover_count[u] > 0)
                    .count();
                if gain > best_gain || (gain == best_gain && overlap > best_overlap) {
                    best_gain = gain;
                    best_overlap = overlap;
                    best_rep = Some(cand);
                }
            }

            if let Some(rep) = best_rep {
                add_existing_vertex(graph, rep, &mut in_set, &mut cover_count, solution);
                prune_redundant(graph, solution, &mut in_set, &mut cover_count);
                changed = true;
            } else {
                add_existing_vertex(graph, v, &mut in_set, &mut cover_count, solution);
            }
        }
    }

    prune_redundant(graph, solution, &mut in_set, &mut cover_count);
    solution.sort_unstable();
}

fn prune_redundant(
    graph: &Graph,
    solution: &mut Vec<usize>,
    in_set: &mut [bool],
    cover_count: &mut [u16],
) {
    let mut order = solution.clone();
    order.sort_unstable_by_key(|&v| graph.closed[v].len());
    for v in order {
        if !in_set[v] {
            continue;
        }
        let removable = graph.closed[v].iter().all(|&u| cover_count[u] >= 2);
        if removable {
            remove_vertex(graph, v, in_set, cover_count, solution);
        }
    }
}

fn first_undominated(cover_count: &[u16]) -> Option<usize> {
    cover_count.iter().position(|&c| c == 0)
}

fn add_vertex(
    graph: &Graph,
    v: usize,
    in_set: &mut [bool],
    covered: &mut [bool],
    cover_count: &mut [u16],
    uncovered: &mut usize,
    solution: &mut Vec<usize>,
) {
    if in_set[v] {
        return;
    }
    in_set[v] = true;
    solution.push(v);
    for &u in &graph.closed[v] {
        cover_count[u] += 1;
        if !covered[u] {
            covered[u] = true;
            *uncovered -= 1;
        }
    }
}

fn add_existing_vertex(
    graph: &Graph,
    v: usize,
    in_set: &mut [bool],
    cover_count: &mut [u16],
    solution: &mut Vec<usize>,
) {
    if in_set[v] {
        return;
    }
    in_set[v] = true;
    solution.push(v);
    for &u in &graph.closed[v] {
        cover_count[u] += 1;
    }
}

fn remove_vertex(
    graph: &Graph,
    v: usize,
    in_set: &mut [bool],
    cover_count: &mut [u16],
    solution: &mut Vec<usize>,
) {
    if !in_set[v] {
        return;
    }
    in_set[v] = false;
    for &u in &graph.closed[v] {
        cover_count[u] -= 1;
    }
    if let Some(pos) = solution.iter().position(|&x| x == v) {
        solution.swap_remove(pos);
    }
}
