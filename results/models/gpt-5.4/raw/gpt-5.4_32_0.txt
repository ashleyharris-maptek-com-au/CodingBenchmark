using namespace std;

struct Edge {
    int to;
    long long w;
    int id;
};

struct InputEdge {
    int u;
    int v;
    long long w;
};

struct UF {
    vector<int> parent;
    vector<int> rankv;

    explicit UF(int n) : parent(n), rankv(n, 0) {
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }

    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) {
            return false;
        }
        if (rankv[a] < rankv[b]) {
            swap(a, b);
        }
        parent[b] = a;
        if (rankv[a] == rankv[b]) {
            ++rankv[a];
        }
        return true;
    }
};

static constexpr long long INF = numeric_limits<long long>::max() / 4;

vector<long long> dijkstra_all(int source, const vector<vector<Edge>>& graph) {
    const int n = static_cast<int>(graph.size());
    vector<long long> dist(n, INF);
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    dist[source] = 0;
    pq.push({0, source});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) {
            continue;
        }
        for (const Edge& e : graph[u]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                pq.push({nd, e.to});
            }
        }
    }
    return dist;
}

struct PathResult {
    vector<long long> dist;
    vector<int> parent_vertex;
    vector<int> parent_edge;
};

PathResult dijkstra_with_parent(int source, const vector<vector<Edge>>& graph) {
    const int n = static_cast<int>(graph.size());
    vector<long long> dist(n, INF);
    vector<int> parent_vertex(n, -1);
    vector<int> parent_edge(n, -1);
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    dist[source] = 0;
    pq.push({0, source});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) {
            continue;
        }
        for (const Edge& e : graph[u]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parent_vertex[e.to] = u;
                parent_edge[e.to] = e.id;
                pq.push({nd, e.to});
            }
        }
    }

    return {move(dist), move(parent_vertex), move(parent_edge)};
}

pair<long long, vector<int>> steiner_exact(
    int n,
    const vector<vector<Edge>>& graph,
    const vector<InputEdge>& edges,
    const vector<int>& terminals
) {
    const int k = static_cast<int>(terminals.size());
    const int full = 1 << k;

    vector<vector<long long>> dp(full, vector<long long>(n, INF));
    vector<vector<int>> split_mask(full, vector<int>(n, -1));
    vector<vector<int>> relax_parent(full, vector<int>(n, -1));
    vector<vector<int>> relax_edge(full, vector<int>(n, -1));

    for (int i = 0; i < k; ++i) {
        dp[1 << i][terminals[i]] = 0;
    }

    for (int mask = 1; mask < full; ++mask) {
        if ((mask & (mask - 1)) != 0) {
            int first = __builtin_ctz(mask);
            for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
                if (((sub >> first) & 1) == 0) {
                    continue;
                }
                int other = mask ^ sub;
                if (other == 0) {
                    continue;
                }
                for (int v = 0; v < n; ++v) {
                    long long cand = dp[sub][v] + dp[other][v];
                    if (cand < dp[mask][v]) {
                        dp[mask][v] = cand;
                        split_mask[mask][v] = sub;
                        relax_parent[mask][v] = -1;
                        relax_edge[mask][v] = -1;
                    }
                }
            }
        }

        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] < INF) {
                pq.push({dp[mask][v], v});
            }
        }

        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (d != dp[mask][u]) {
                continue;
            }
            for (const Edge& e : graph[u]) {
                long long nd = d + e.w;
                if (nd < dp[mask][e.to]) {
                    dp[mask][e.to] = nd;
                    split_mask[mask][e.to] = -1;
                    relax_parent[mask][e.to] = u;
                    relax_edge[mask][e.to] = e.id;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    int best_vertex = 0;
    long long best_cost = dp[full - 1][0];
    for (int v = 1; v < n; ++v) {
        if (dp[full - 1][v] < best_cost) {
            best_cost = dp[full - 1][v];
            best_vertex = v;
        }
    }

    vector<char> used(edges.size(), 0);
    function<void(int, int)> build = [&](int mask, int v) {
        if ((mask & (mask - 1)) == 0) {
            return;
        }
        if (relax_parent[mask][v] != -1) {
            int e = relax_edge[mask][v];
            used[e] = 1;
            build(mask, relax_parent[mask][v]);
            return;
        }
        int sub = split_mask[mask][v];
        if (sub == -1) {
            return;
        }
        int other = mask ^ sub;
        build(sub, v);
        build(other, v);
    };

    build(full - 1, best_vertex);

    vector<int> selected;
    for (int i = 0; i < static_cast<int>(edges.size()); ++i) {
        if (used[i]) {
            selected.push_back(i);
        }
    }

    return {best_cost, selected};
}

pair<long long, vector<int>> steiner_metric_closure(
    int n,
    const vector<vector<Edge>>& graph,
    const vector<InputEdge>& edges,
    const vector<int>& terminals
) {
    const int k = static_cast<int>(terminals.size());
    vector<PathResult> terminal_runs;
    terminal_runs.reserve(k);
    for (int t : terminals) {
        terminal_runs.push_back(dijkstra_with_parent(t, graph));
    }

    struct MetricEdge {
        long long w;
        int a;
        int b;
    };

    vector<MetricEdge> metric_edges;
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            metric_edges.push_back({terminal_runs[i].dist[terminals[j]], i, j});
        }
    }

    sort(metric_edges.begin(), metric_edges.end(), [](const MetricEdge& lhs, const MetricEdge& rhs) {
        return lhs.w < rhs.w;
    });

    UF uf_metric(k);
    vector<pair<int, int>> mst_pairs;
    for (const MetricEdge& me : metric_edges) {
        if (uf_metric.unite(me.a, me.b)) {
            mst_pairs.push_back({me.a, me.b});
            if (static_cast<int>(mst_pairs.size()) == k - 1) {
                break;
            }
        }
    }

    vector<char> used(edges.size(), 0);
    for (auto [a, b] : mst_pairs) {
        int cur = terminals[b];
        while (cur != terminals[a]) {
            int e = terminal_runs[a].parent_edge[cur];
            if (e == -1) {
                break;
            }
            used[e] = 1;
            cur = terminal_runs[a].parent_vertex[cur];
        }
    }

    vector<int> candidate_edges;
    for (int i = 0; i < static_cast<int>(edges.size()); ++i) {
        if (used[i]) {
            candidate_edges.push_back(i);
        }
    }

    vector<int> relevant_vertices = terminals;
    for (int id : candidate_edges) {
        relevant_vertices.push_back(edges[id].u);
        relevant_vertices.push_back(edges[id].v);
    }
    sort(relevant_vertices.begin(), relevant_vertices.end());
    relevant_vertices.erase(unique(relevant_vertices.begin(), relevant_vertices.end()), relevant_vertices.end());

    vector<char> is_terminal(n, 0);
    for (int t : terminals) {
        is_terminal[t] = 1;
    }

    vector<int> degree(n, 0);
    vector<vector<pair<int, int>>> subgraph(n);
    for (int id : candidate_edges) {
        const auto& e = edges[id];
        subgraph[e.u].push_back({e.v, id});
        subgraph[e.v].push_back({e.u, id});
        ++degree[e.u];
        ++degree[e.v];
    }

    queue<int> q;
    vector<char> removed(n, 0);
    for (int v : relevant_vertices) {
        if (!is_terminal[v] && degree[v] <= 1) {
            q.push(v);
        }
    }

    while (!q.empty()) {
        int v = q.front();
        q.pop();
        if (removed[v] || is_terminal[v] || degree[v] > 1) {
            continue;
        }
        removed[v] = 1;
        for (auto [to, id] : subgraph[v]) {
            if (!used[id]) {
                continue;
            }
            used[id] = 0;
            --degree[v];
            --degree[to];
            if (!is_terminal[to] && !removed[to] && degree[to] <= 1) {
                q.push(to);
            }
        }
    }

    vector<int> pruned_edges;
    for (int i = 0; i < static_cast<int>(edges.size()); ++i) {
        if (used[i]) {
            pruned_edges.push_back(i);
        }
    }

    if (pruned_edges.empty() && !terminals.empty()) {
        return {0, {}};
    }

    vector<int> tree_edges;
    vector<tuple<long long, int, int>> weighted_edges;
    weighted_edges.reserve(pruned_edges.size());
    for (int id : pruned_edges) {
        weighted_edges.push_back({edges[id].w, edges[id].u, edges[id].v});
    }
    sort(weighted_edges.begin(), weighted_edges.end());

    UF uf_final(n);
    long long total = 0;
    for (const auto& [w, u, v] : weighted_edges) {
        if (uf_final.unite(u, v)) {
            total += w;
            for (int id : pruned_edges) {
                if (((edges[id].u == u && edges[id].v == v) || (edges[id].u == v && edges[id].v == u)) && edges[id].w == w) {
                    tree_edges.push_back(id);
                    break;
                }
            }
        }
    }

    return {total, tree_edges};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    vector<InputEdge> edges(m);
    vector<vector<Edge>> graph(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges[i] = {u, v, w};
        graph[u].push_back({v, w, i});
        graph[v].push_back({u, w, i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
    }

    sort(terminals.begin(), terminals.end());
    terminals.erase(unique(terminals.begin(), terminals.end()), terminals.end());
    k = static_cast<int>(terminals.size());

    if (k <= 1) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    pair<long long, vector<int>> result;
    if (k <= 18) {
        result = steiner_exact(n, graph, edges, terminals);
    } else {
        result = steiner_metric_closure(n, graph, edges, terminals);
    }

    long long total_weight = result.first;
    vector<int>& chosen = result.second;

    cout << total_weight << ' ' << chosen.size() << '\n';
    for (int id : chosen) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }

    return 0;
}
