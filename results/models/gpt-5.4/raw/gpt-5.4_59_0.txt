const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const probability = Array.isArray(data.probability) ? data.probability : [];
const drones = Array.isArray(data.drones) ? data.drones : [];
const maxStepsPerDrone = Math.max(0, data.maxStepsPerDrone | 0);

const height = probability.length;
const width = height > 0 && Array.isArray(probability[0]) ? probability[0].length : 0;

function makeStayPaths() {
  const paths = [];
  for (let i = 0; i < drones.length; i++) {
    const d = drones[i] || { x: 0, y: 0 };
    const path = [];
    const x = d.x | 0;
    const y = d.y | 0;
    for (let s = 0; s <= maxStepsPerDrone; s++) path.push([x, y]);
    paths.push(path);
  }
  return { paths };
}

if (height === 0 || width === 0 || drones.length === 0) {
  process.stdout.write(JSON.stringify(makeStayPaths()));
  process.exit(0);
}

function inBounds(x, y) {
  return y >= 0 && y < height && x >= 0 && x < width;
}

function passable(x, y) {
  return inBounds(x, y) && probability[y][x] !== -1;
}

const cellCount = width * height;
const cellId = (x, y) => y * width + x;
const idX = id => id % width;
const idY = id => (id / width) | 0;

const moves = [
  [0, 0],
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1]
];

const neighborCache = new Array(cellCount);
const passableIds = [];
let totalPositive = 0;

for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const id = cellId(x, y);
    if (!passable(x, y)) {
      neighborCache[id] = [];
      continue;
    }
    passableIds.push(id);
    const val = Number(probability[y][x]);
    if (val > 0) totalPositive += val;
    const neighbors = [];
    for (let i = 0; i < moves.length; i++) {
      const nx = x + moves[i][0];
      const ny = y + moves[i][1];
      if (passable(nx, ny)) neighbors.push(cellId(nx, ny));
    }
    neighborCache[id] = neighbors;
  }
}

if (passableIds.length === 0) {
  process.stdout.write(JSON.stringify(makeStayPaths()));
  process.exit(0);
}

const distMaps = new Array(drones.length);
const nearestDrone = new Int32Array(cellCount);
const nearestDist = new Int32Array(cellCount);
nearestDrone.fill(-1);
nearestDist.fill(1 << 30);

function bfsFrom(startX, startY) {
  const dist = new Int32Array(cellCount);
  dist.fill(-1);
  if (!passable(startX, startY)) return dist;

  const startId = cellId(startX, startY);
  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;

  dist[startId] = 0;
  queue[tail++] = startId;

  while (head < tail) {
    const id = queue[head++];
    const nextDist = dist[id] + 1;
    const neighbors = neighborCache[id];
    for (let i = 0; i < neighbors.length; i++) {
      const nid = neighbors[i];
      if (dist[nid] !== -1) continue;
      dist[nid] = nextDist;
      queue[tail++] = nid;
    }
  }

  return dist;
}

for (let i = 0; i < drones.length; i++) {
  const d = drones[i] || { x: 0, y: 0 };
  const dist = bfsFrom(d.x | 0, d.y | 0);
  distMaps[i] = dist;
  for (let id = 0; id < cellCount; id++) {
    const dd = dist[id];
    if (dd < 0) continue;
    if (dd < nearestDist[id]) {
      nearestDist[id] = dd;
      nearestDrone[id] = i;
    }
  }
}

const assignedCells = Array.from({ length: drones.length }, () => []);

for (let k = 0; k < passableIds.length; k++) {
  const id = passableIds[k];
  const owner = nearestDrone[id];
  if (owner >= 0 && nearestDist[id] <= maxStepsPerDrone) {
    assignedCells[owner].push(id);
  }
}

for (let i = 0; i < drones.length; i++) {
  assignedCells[i].sort((a, b) => {
    const va = Number(probability[idY(a)][idX(a)]);
    const vb = Number(probability[idY(b)][idX(b)]);
    if (vb !== va) return vb - va;
    return distMaps[i][a] - distMaps[i][b];
  });
}

const globallyVisited = new Uint8Array(cellCount);

function reconstructMove(dist, fromId, toId) {
  if (fromId === toId) return [fromId];
  if (dist[fromId] < 0 || dist[toId] < 0) return [fromId];

  const seen = new Uint8Array(cellCount);
  const parent = new Int32Array(cellCount);
  parent.fill(-1);
  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;

  queue[tail++] = fromId;
  seen[fromId] = 1;

  while (head < tail) {
    const id = queue[head++];
    if (id === toId) break;
    const neighbors = neighborCache[id];
    for (let i = 0; i < neighbors.length; i++) {
      const nid = neighbors[i];
      if (seen[nid]) continue;
      seen[nid] = 1;
      parent[nid] = id;
      queue[tail++] = nid;
      if (nid === toId) {
        head = tail;
        break;
      }
    }
  }

  if (!seen[toId]) return [fromId];

  const rev = [];
  let cur = toId;
  while (cur !== -1) {
    rev.push(cur);
    if (cur === fromId) break;
    cur = parent[cur];
  }
  rev.reverse();
  return rev;
}

function scoreCell(id, stepsUsed, distFromStart) {
  if (distFromStart < 0) return -1e18;
  const val = Number(probability[idY(id)][idX(id)]);
  const fresh = globallyVisited[id] ? 0 : Math.max(0, val);
  const remaining = maxStepsPerDrone - stepsUsed;
  if (remaining < distFromStart) return -1e18;
  const closeness = remaining - distFromStart;
  return fresh * 1000000 + closeness * 16 - distFromStart;
}

function buildPathForDrone(droneIndex) {
  const d = drones[droneIndex] || { x: 0, y: 0 };
  const sx = d.x | 0;
  const sy = d.y | 0;

  if (!passable(sx, sy)) {
    const path = [];
    for (let s = 0; s <= maxStepsPerDrone; s++) path.push([sx, sy]);
    return path;
  }

  const startId = cellId(sx, sy);
  const dist = distMaps[droneIndex];
  const candidates = assignedCells[droneIndex];

  const route = [startId];
  let current = startId;
  let stepsUsed = 0;
  globallyVisited[current] = 1;

  let pointer = 0;
  while (stepsUsed < maxStepsPerDrone) {
    let target = -1;
    let bestScore = -1e18;

    for (let examined = 0; examined < candidates.length; examined++) {
      const idx = (pointer + examined) % candidates.length;
      const id = candidates[idx];
      const sc = scoreCell(id, stepsUsed, dist[id]);
      if (sc > bestScore) {
        bestScore = sc;
        target = id;
      }
      if (examined > 48 && bestScore > 0) break;
    }

    if (target === -1 || bestScore < 0) {
      let bestNeighbor = current;
      let bestNeighborScore = -1e18;
      const neighbors = neighborCache[current];
      for (let i = 0; i < neighbors.length; i++) {
        const nid = neighbors[i];
        const val = globallyVisited[nid] ? 0 : Math.max(0, Number(probability[idY(nid)][idX(nid)]));
        const sc = val * 1000000 + dist[nid];
        if (sc > bestNeighborScore) {
          bestNeighborScore = sc;
          bestNeighbor = nid;
        }
      }
      current = bestNeighbor;
      route.push(current);
      globallyVisited[current] = 1;
      stepsUsed++;
      continue;
    }

    pointer = (pointer + 1) % Math.max(1, candidates.length);
    const segment = reconstructMove(dist, current, target);
    if (segment.length <= 1) {
      current = target;
      route.push(current);
      globallyVisited[current] = 1;
      stepsUsed++;
      continue;
    }

    let moved = false;
    for (let i = 1; i < segment.length && stepsUsed < maxStepsPerDrone; i++) {
      current = segment[i];
      route.push(current);
      globallyVisited[current] = 1;
      stepsUsed++;
      moved = true;
    }
    if (!moved) break;
  }

  while (route.length < maxStepsPerDrone + 1) route.push(current);

  const out = new Array(route.length);
  for (let i = 0; i < route.length; i++) out[i] = [idX(route[i]), idY(route[i])];
  return out;
}

const paths = new Array(drones.length);
for (let i = 0; i < drones.length; i++) paths[i] = buildPathForDrone(i);

process.stdout.write(JSON.stringify({ paths }));
