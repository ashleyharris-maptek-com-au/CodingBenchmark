use std::f64::consts::PI;
use std::io::{self, BufRead, BufReader, BufWriter, Read, Write};

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone, Copy, Debug)]
struct Config {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
    landing_left: f64,
    landing_right: f64,
    ground_y: f64,
}

fn clamp(value: f64, lo: f64, hi: f64) -> f64 {
    value.max(lo).min(hi)
}

fn normalize_angle(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn read_initial_config<R: BufRead>(reader: &mut R) -> io::Result<Config> {
    let mut line = String::new();

    line.clear();
    reader.read_line(&mut line)?;
    let first: Vec<_> = line.split_whitespace().collect();
    let width = first[0].parse::<usize>().unwrap();
    let height = first[1].parse::<usize>().unwrap();
    let gravity = first[2].parse::<f64>().unwrap();
    let max_thrust = first[3].parse::<f64>().unwrap();
    let fuel = first[4].parse::<f64>().unwrap();

    line.clear();
    reader.read_line(&mut line)?;
    let second: Vec<_> = line.split_whitespace().collect();
    let start_x = second[0].parse::<f64>().unwrap();
    let start_y = second[1].parse::<f64>().unwrap();
    let target_x = second[2].parse::<f64>().unwrap();
    let target_y = second[3].parse::<f64>().unwrap();

    let mut landing_left = target_x.floor();
    let mut landing_right = target_x.ceil().max(landing_left + 1.0);
    let mut ground_y = target_y;
    let target_row = target_y.floor() as isize;

    for row in 0..height {
        line.clear();
        reader.read_line(&mut line)?;
        let row_str = line.trim_end_matches(&['\r', '\n'][..]);
        if row as isize == target_row {
            let bytes = row_str.as_bytes();
            let mut idx = 0usize;
            while idx < bytes.len() {
                if bytes[idx] == b'L' || bytes[idx] == b'T' {
                    let start = idx;
                    while idx < bytes.len() && (bytes[idx] == b'L' || bytes[idx] == b'T') {
                        idx += 1;
                    }
                    if target_x >= start as f64 - 0.5 && target_x <= idx as f64 + 0.5 {
                        landing_left = start as f64;
                        landing_right = idx as f64;
                        ground_y = row as f64;
                        break;
                    }
                } else {
                    idx += 1;
                }
            }
        }
    }

    Ok(Config {
        width,
        height,
        gravity,
        max_thrust,
        fuel,
        start_x,
        start_y,
        target_x,
        target_y,
        landing_left,
        landing_right,
        ground_y,
    })
}

fn compute_command(cfg: &Config, s: State) -> (f64, f64) {
    let landing_center = 0.5 * (cfg.landing_left + cfg.landing_right);
    let dx = landing_center - s.x;
    let dy = cfg.ground_y - s.y;
    let altitude = dy.max(0.0);

    let zone_half_width = ((cfg.landing_right - cfg.landing_left) * 0.5).max(1.0);
    let far_horizontal = dx.abs() > zone_half_width * 1.5;

    let target_vx = if far_horizontal {
        clamp(dx * 0.22, -8.0, 8.0)
    } else {
        clamp(dx * 0.10, -2.0, 2.0)
    };
    let vx_error = target_vx - s.vx;

    let mut desired_ax = clamp(vx_error * 0.9, -cfg.max_thrust * 0.55, cfg.max_thrust * 0.55);
    if altitude < 8.0 {
        desired_ax = clamp(desired_ax, -cfg.max_thrust * 0.18, cfg.max_thrust * 0.18);
    }

    let speed = (s.vx * s.vx + s.vy * s.vy).sqrt();
    let target_vy = if altitude > 80.0 {
        12.0 + dx.abs() * 0.02
    } else if altitude > 30.0 {
        7.0 + dx.abs() * 0.015
    } else if altitude > 10.0 {
        3.2
    } else if altitude > 4.0 {
        1.6
    } else {
        0.8
    };

    let mut desired_ay = cfg.gravity + clamp((target_vy - s.vy) * 1.2, -cfg.max_thrust, cfg.max_thrust);
    if altitude < 4.0 && speed < 2.5 {
        desired_ay = cfg.gravity;
    }

    let mut thrust_vec_x = desired_ax;
    let mut thrust_vec_y = clamp(desired_ay, 0.0, cfg.max_thrust);

    let thrust_mag = (thrust_vec_x * thrust_vec_x + thrust_vec_y * thrust_vec_y).sqrt();
    if thrust_mag > cfg.max_thrust && thrust_mag > 0.0 {
        let scale = cfg.max_thrust / thrust_mag;
        thrust_vec_x *= scale;
        thrust_vec_y *= scale;
    }

    let desired_angle = clamp(f64::atan2(thrust_vec_x, thrust_vec_y), -1.45, 1.45);
    let angle_error = normalize_angle(desired_angle - s.angle);
    let turn = clamp(angle_error * 2.4 - s.angular_vel * 0.9, -1.0, 1.0);

    let aligned = (angle_error.abs() < 0.35) as i32 as f64;
    let mut thrust = if thrust_mag <= 1e-6 {
        0.0
    } else {
        let facing_error = normalize_angle(desired_angle - s.angle).abs();
        let projected = if facing_error < 1.2 {
            thrust_mag * facing_error.cos().max(0.25)
        } else {
            thrust_mag * 0.25
        };
        clamp(projected / cfg.max_thrust, 0.0, 1.0)
    };

    if altitude < 15.0 && dx.abs() < zone_half_width && s.vy > 2.5 {
        thrust = thrust.max(0.75 + 0.2 * aligned);
    }
    if altitude < 5.0 && dx.abs() < zone_half_width && s.vy > 1.5 {
        thrust = 1.0;
    }
    if s.fuel <= 0.05 {
        thrust = 0.0;
    }

    (thrust, turn)
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());
    let cfg = read_initial_config(&mut reader)?;

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut rest = String::new();
    reader.read_to_string(&mut rest)?;
    let mut nums = Vec::new();
    for tok in rest.split_whitespace() {
        if let Ok(v) = tok.parse::<f64>() {
            nums.push(v);
        }
    }

    let mut i = 0usize;
    while i + 6 < nums.len() {
        let state = State {
            x: nums[i],
            y: nums[i + 1],
            vx: nums[i + 2],
            vy: nums[i + 3],
            angle: nums[i + 4],
            angular_vel: nums[i + 5],
            fuel: nums[i + 6],
        };
        i += 7;

        let (thrust, turn) = compute_command(&cfg, state);

        // Emit multiple commands per state update so the engine still sees >=20 Hz output.
        for _ in 0..3 {
            writeln!(out, "{:.4} {:.4}", thrust, turn)?;
        }
        out.flush()?;
    }

    Ok(())
}
