float4 main(PSInput input) : SV_Target0
{
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);

    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.5 + 0.5 * diffuse;

    float2 checkerUV = floor(input.uv * 8.0);
    float parity = fmod(checkerUV.x + checkerUV.y, 2.0);
    float checker = lerp(0.9, 0.2, parity);

    float3 color = checker.xxx * shade;
    return float4(color, 1.0);
}
