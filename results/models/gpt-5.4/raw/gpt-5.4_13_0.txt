using namespace std;

namespace {

struct HashPair {
    uint64_t a;
    uint64_t b;

    bool operator==(const HashPair& other) const noexcept {
        return a == other.a && b == other.b;
    }
};

struct HashPairHasher {
    size_t operator()(const HashPair& h) const noexcept {
        uint64_t x = h.a ^ (h.b + 0x9e3779b97f4a7c15ULL + (h.a << 6U) + (h.a >> 2U));
        return static_cast<size_t>(x ^ (x >> 32U));
    }
};

class RollingHash {
public:
    RollingHash() = default;

    explicit RollingHash(const string& s) {
        build(s);
    }

    void build(const string& s) {
        const size_t n = s.size();
        pref1.assign(n + 1, 0);
        pref2.assign(n + 1, 0);
        pow1.assign(n + 1, 1);
        pow2.assign(n + 1, 1);

        for (size_t i = 0; i < n; ++i) {
            const uint64_t v = static_cast<unsigned char>(s[i]) + 1ULL;
            pref1[i + 1] = pref1[i] * base1 + v;
            pref2[i + 1] = pref2[i] * base2 + v;
            pow1[i + 1] = pow1[i] * base1;
            pow2[i + 1] = pow2[i] * base2;
        }
    }

    HashPair get(size_t l, size_t len) const {
        return {
            pref1[l + len] - pref1[l] * pow1[len],
            pref2[l + len] - pref2[l] * pow2[len]
        };
    }

private:
    static constexpr uint64_t base1 = 1469598103934665603ULL;
    static constexpr uint64_t base2 = 1099511628211ULL;

    vector<uint64_t> pref1;
    vector<uint64_t> pref2;
    vector<uint64_t> pow1;
    vector<uint64_t> pow2;
};

struct CandidateLocation {
    int stringIndex = -1;
    size_t pos = 0;
};

struct SearchResult {
    bool found = false;
    string value;
};

bool verify_in_all(const string& candidate, const vector<string>& strings) {
    for (const string& s : strings) {
        if (s.find(candidate) == string::npos) {
            return false;
        }
    }
    return true;
}

SearchResult try_length_small(
    int len,
    const vector<string>& strings,
    const vector<RollingHash>& hashes,
    const vector<int>& order
) {
    if (len == 0) {
        return {true, ""};
    }

    unordered_map<HashPair, vector<size_t>, HashPairHasher> positions;
    const string& base = strings[order[0]];
    const RollingHash& baseHash = hashes[order[0]];
    const size_t baseCount = base.size() >= static_cast<size_t>(len)
        ? base.size() - static_cast<size_t>(len) + 1
        : 0;

    positions.reserve(baseCount * 2 + 1);
    for (size_t i = 0; i < baseCount; ++i) {
        positions[baseHash.get(i, static_cast<size_t>(len))].push_back(i);
    }

    unordered_set<HashPair, HashPairHasher> active;
    active.reserve(positions.size() * 2 + 1);
    for (const auto& entry : positions) {
        active.insert(entry.first);
    }

    for (size_t idx = 1; idx < order.size() && !active.empty(); ++idx) {
        const int si = order[idx];
        const string& cur = strings[si];
        if (cur.size() < static_cast<size_t>(len)) {
            return {};
        }

        unordered_set<HashPair, HashPairHasher> next;
        const size_t count = cur.size() - static_cast<size_t>(len) + 1;
        next.reserve(min(active.size(), count) * 2 + 1);
        for (size_t i = 0; i < count; ++i) {
            HashPair h = hashes[si].get(i, static_cast<size_t>(len));
            if (active.find(h) != active.end()) {
                next.insert(h);
            }
        }
        active.swap(next);
    }

    if (active.empty()) {
        return {};
    }

    for (const HashPair& h : active) {
        const auto it = positions.find(h);
        if (it == positions.end()) {
            continue;
        }
        for (size_t pos : it->second) {
            string candidate = base.substr(pos, static_cast<size_t>(len));
            if (verify_in_all(candidate, strings)) {
                return {true, std::move(candidate)};
            }
        }
    }

    return {};
}

SearchResult try_length_large(
    int len,
    const vector<string>& strings,
    const vector<RollingHash>& hashes,
    const vector<int>& order
) {
    if (len == 0) {
        return {true, ""};
    }

    const int baseIdx = order[0];
    const string& base = strings[baseIdx];
    if (base.size() < static_cast<size_t>(len)) {
        return {};
    }

    unordered_map<HashPair, CandidateLocation, HashPairHasher> candidates;
    const size_t baseCount = base.size() - static_cast<size_t>(len) + 1;
    candidates.reserve(baseCount * 2 + 1);
    for (size_t i = 0; i < baseCount; ++i) {
        candidates.emplace(hashes[baseIdx].get(i, static_cast<size_t>(len)), CandidateLocation{baseIdx, i});
    }

    for (size_t idx = 1; idx < order.size() && !candidates.empty(); ++idx) {
        const int si = order[idx];
        const string& cur = strings[si];
        if (cur.size() < static_cast<size_t>(len)) {
            return {};
        }

        unordered_set<HashPair, HashPairHasher> seen;
        const size_t count = cur.size() - static_cast<size_t>(len) + 1;
        seen.reserve(min(candidates.size(), count) * 2 + 1);
        for (size_t i = 0; i < count; ++i) {
            HashPair h = hashes[si].get(i, static_cast<size_t>(len));
            if (candidates.find(h) != candidates.end()) {
                seen.insert(h);
            }
        }

        for (auto it = candidates.begin(); it != candidates.end(); ) {
            if (seen.find(it->first) == seen.end()) {
                it = candidates.erase(it);
            } else {
                ++it;
            }
        }
    }

    if (candidates.empty()) {
        return {};
    }

    for (const auto& entry : candidates) {
        const CandidateLocation& loc = entry.second;
        string candidate = strings[loc.stringIndex].substr(loc.pos, static_cast<size_t>(len));
        if (verify_in_all(candidate, strings)) {
            return {true, std::move(candidate)};
        }
    }

    return {};
}

string longest_common_substring(vector<string> strings) {
    if (strings.empty()) {
        return "";
    }

    vector<int> order(strings.size());
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int lhs, int rhs) {
        if (strings[lhs].size() != strings[rhs].size()) {
            return strings[lhs].size() < strings[rhs].size();
        }
        return lhs < rhs;
    });

    vector<RollingHash> hashes(strings.size());
    for (size_t i = 0; i < strings.size(); ++i) {
        hashes[i].build(strings[i]);
    }

    const size_t shortest = strings[order[0]].size();
    uint64_t totalLength = 0;
    for (const string& s : strings) {
        totalLength += static_cast<uint64_t>(s.size());
    }

    const bool useSmallMode = strings.size() <= 8 && totalLength <= 2'000'000ULL;

    int lo = 0;
    int hi = static_cast<int>(shortest);
    string best;

    while (lo <= hi) {
        const int mid = lo + (hi - lo) / 2;
        SearchResult result = useSmallMode
            ? try_length_small(mid, strings, hashes, order)
            : try_length_large(mid, strings, hashes, order);

        if (result.found) {
            best = std::move(result.value);
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    return best;
}

}  // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n = 0;
    if (!(cin >> n)) {
        return 0;
    }
    cin.ignore(numeric_limits<streamsize>::max(), '\n');

    vector<string> strings;
    strings.reserve(max(n, 0));
    for (int i = 0; i < n; ++i) {
        string s;
        getline(cin, s);
        strings.push_back(std::move(s));
    }

    cout << longest_common_substring(std::move(strings)) << '\n';
    return 0;
}
