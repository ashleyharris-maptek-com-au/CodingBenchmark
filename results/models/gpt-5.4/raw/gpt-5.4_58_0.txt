const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = Array.isArray(data.grid) ? data.grid : [];
const robots = Array.isArray(data.robots) ? data.robots : [];
const jobs = Array.isArray(data.jobs) ? data.jobs : [];

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const cellCount = width * height;

function indexOf(x, y) {
  return y * width + x;
}

function isOpen(x, y) {
  return y >= 0 && y < height && x >= 0 && x < width && grid[y][x] !== '#';
}

const neighbors = new Int32Array(cellCount * 4);
for (let i = 0; i < neighbors.length; i++) neighbors[i] = -1;

for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    if (!isOpen(x, y)) continue;
    const idx = indexOf(x, y);
    let p = idx * 4;
    if (y > 0 && isOpen(x, y - 1)) neighbors[p++] = indexOf(x, y - 1);
    if (x + 1 < width && isOpen(x + 1, y)) neighbors[p++] = indexOf(x + 1, y);
    if (y + 1 < height && isOpen(x, y + 1)) neighbors[p++] = indexOf(x, y + 1);
    if (x > 0 && isOpen(x - 1, y)) neighbors[p++] = indexOf(x - 1, y);
  }
}

const queue = new Int32Array(Math.max(cellCount, 1));
const dist = new Int32Array(Math.max(cellCount, 1));

function bfsFrom(startIdx) {
  dist.fill(-1);
  let head = 0;
  let tail = 0;
  queue[tail++] = startIdx;
  dist[startIdx] = 0;

  while (head < tail) {
    const cur = queue[head++];
    const nextDist = dist[cur] + 1;
    const base = cur * 4;
    for (let i = 0; i < 4; i++) {
      const nxt = neighbors[base + i];
      if (nxt === -1 || dist[nxt] !== -1) continue;
      dist[nxt] = nextDist;
      queue[tail++] = nxt;
    }
  }

  return Int32Array.from(dist);
}

let maxRobotId = -1;
for (const robot of robots) {
  if (robot.id > maxRobotId) maxRobotId = robot.id;
}

const assignments = Array.from({ length: maxRobotId + 1 }, () => []);
const robotStates = [];

for (const robot of robots) {
  const startIdx = indexOf(robot.x, robot.y);
  if (!isOpen(robot.x, robot.y)) {
    throw new Error(`Robot ${robot.id} is on a blocked cell`);
  }
  robotStates.push({
    id: robot.id,
    totalTravel: 0,
    distMap: bfsFrom(startIdx),
  });
}

const sortedJobs = jobs.slice().sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  if (a.id !== b.id) return a.id - b.id;
  if (a.y !== b.y) return a.y - b.y;
  return a.x - b.x;
});

for (const job of sortedJobs) {
  const jobIdx = indexOf(job.x, job.y);
  if (!isOpen(job.x, job.y)) {
    throw new Error(`Job ${job.id} is on a blocked cell`);
  }
  let bestRobot = 0;
  let bestDist = Infinity;
  let bestTravel = Infinity;
  let bestId = Infinity;

  for (let i = 0; i < robotStates.length; i++) {
    const state = robotStates[i];
    const d = state.distMap[jobIdx];
    const travel = d >= 0 ? d : Number.MAX_SAFE_INTEGER;

    if (
      travel < bestDist ||
      (travel === bestDist && state.totalTravel < bestTravel) ||
      (travel === bestDist && state.totalTravel === bestTravel && state.id < bestId)
    ) {
      bestRobot = i;
      bestDist = travel;
      bestTravel = state.totalTravel;
      bestId = state.id;
    }
  }

  const chosen = robotStates[bestRobot];
  assignments[chosen.id].push(job.id);
  if (bestDist !== Number.MAX_SAFE_INTEGER) {
    chosen.totalTravel += bestDist;
    chosen.distMap = bfsFrom(jobIdx);
  }
}

process.stdout.write(JSON.stringify({ assignments }));
