use std::io::{self, BufRead, BufReader, Write};
use std::thread;
use std::time::{Duration, Instant};
use std::sync::mpsc::{self, Sender, Receiver};

// Simple data structures for map and state updates
#[derive(Clone, Debug)]
struct WorldMap {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    fuel: f64,
    start: (f64, f64),
    target: (f64, f64),
    grid: Vec<Vec<u8>>, // 0 = empty, 1 = obstacle
}

#[derive(Clone, Debug)]
struct StateUpdate {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

// Helpers
fn clamp(x: f64, lo: f64, hi: f64) -> f64 {
    if x < lo { lo } else if x > hi { hi } else { x }
}

fn wrap_angle(mut a: f64) -> f64 {
    use std::f64::consts::PI;
    while a > PI { a -= 2.0 * PI; }
    while a < -PI { a += 2.0 * PI; }
    a
}

fn dist(a: (f64, f64), b: (f64, f64)) -> f64 {
    let dx = a.0 - b.0;
    let dy = a.1 - b.1;
    (dx * dx + dy * dy).sqrt()
}

fn main() {
    // Communication channels:
    // map_rx -> receives the initial map
    // state_rx -> receives continuous state updates
    // The reader thread parses stdin and sends data through these channels.
    let (map_tx, map_rx): (Sender<WorldMap>, Receiver<WorldMap>) = std::sync::mpsc::channel();
    let (state_tx, state_rx): (Sender<StateUpdate>, Receiver<StateUpdate>) = std::sync::mpsc::channel();

    // Spawn a helper thread to read input (map + state updates) from stdin.
    thread::spawn(move || {
        // Read lines from stdin
        let stdin = io::stdin();
        let mut lines = BufReader::new(stdin.lock()).lines();

        // 1) Map header line: width height gravity max_thrust fuel
        // Skip blank lines if any
        let mut line1 = String::new();
        loop {
            let opt = lines.next();
            match opt {
                Some(Ok(l)) => {
                    if !l.trim().is_empty() {
                        line1 = l;
                        break;
                    }
                }
                Some(Err(_)) | None => {
                    // Input ended unexpectedly
                    return;
                }
            }
        }

        let mut it1 = line1.split_whitespace();
        let width: usize = it1.next().unwrap().parse().unwrap();
        let height: usize = it1.next().unwrap().parse().unwrap();
        let gravity: f64 = it1.next().unwrap().parse().unwrap();
        let max_thrust: f64 = it1.next().unwrap().parse().unwrap();
        let fuel: f64 = it1.next().unwrap().parse().unwrap();

        // 2) Start and Target line: start_x start_y target_x target_y
        let mut line2 = String::new();
        loop {
            match lines.next() {
                Some(Ok(l)) => {
                    if !l.trim().is_empty() {
                        line2 = l;
                        break;
                    }
                }
                Some(Err(_)) | None => return,
            }
        }
        let mut it2 = line2.split_whitespace();
        let start_x: f64 = it2.next().unwrap().parse().unwrap();
        let start_y: f64 = it2.next().unwrap().parse().unwrap();
        let target_x: f64 = it2.next().unwrap().parse().unwrap();
        let target_y: f64 = it2.next().unwrap().parse().unwrap();

        // 3) Map grid: height lines
        let mut grid: Vec<Vec<u8>> = vec![vec![0u8; width]; height];
        for y in 0..height {
            let mut row_line = String::new();
            // read next non-empty line (keep even if it contains spaces)
            loop {
                match lines.next() {
                    Some(Ok(l)) => {
                        // Allow empty lines in between by skipping them
                        row_line = l;
                        break;
                    }
                    Some(Err(_)) | None => return,
                }
            }
            // Parse characters, ignoring spaces. Use first `width` characters.
            let mut col = 0usize;
            for ch in row_line.chars() {
                if ch == ' ' || ch == '\t' || ch == '\r' { continue; }
                if col >= width { break; }
                grid[y][col] = match ch {
                    '#' => 1,
                    '.' => 0,
                    'S' | 'T' | 'L' => 0,
                    _   => 0, // treat unknown as empty
                };
                col += 1;
            }
            // If line didn't provide enough columns, fill rest with empty
            while col < width {
                grid[y][col] = 0;
                col += 1;
            }
        }

        // Create map and send to main thread
        let map = WorldMap {
            width,
            height,
            gravity,
            max_thrust,
            fuel,
            start: (start_x, start_y),
            target: (target_x, target_y),
            grid,
        };
        let _ = map_tx.send(map);

        // 4) Read continuous state updates: x y vx vy angle angular_vel fuel_remaining
        // Each update is on its own line, with numbers separated by whitespace.
        while let Some(Ok(sline)) = lines.next() {
            let sline = sline.trim();
            if sline.is_empty() { continue; }
            // Parse into numbers
            let mut nums: Vec<f64> = Vec::new();
            for tok in sline.split_whitespace() {
                if let Ok(v) = tok.parse::<f64>() {
                    nums.push(v);
                }
            }
            if nums.len() >= 7 {
                let upd = StateUpdate {
                    x: nums[0],
                    y: nums[1],
                    vx: nums[2],
                    vy: nums[3],
                    angle: nums[4],
                    angular_vel: nums[5],
                    fuel_remaining: nums[6],
                };
                let _ = state_tx.send(upd);
            }
            // If fewer than 7 numbers, skip this line (wait for a proper line)
        }
    });

    // Main control loop: wait for map, then run at ~20 Hz, using latest state data
    // Set up a simple PD-ish controller as described in the analysis.
    let map = match map_rx.recv() {
        Ok(m) => m,
        Err(_) => {
            // If map couldn't be read, nothing to do
            return;
        }
    };

    // Control parameters (tuned conservatively)
    let k_p: f64 = 0.8;      // position proportional
    let k_v: f64 = 0.4;      // velocity damping
    let mut x = map.start.0;
    let mut y = map.start.1;
    let mut vx = 0.0f64;
    let mut vy = 0.0f64;
    let mut angle = 0.0f64;
    let mut angular_vel = 0.0f64;
    let mut fuel_remaining = map.fuel;

    // For output
    let mut stdout = io::BufWriter::new(io::stdout());

    // Timing
    let period = Duration::from_millis(50); // 20 Hz
    let mut last_tick = Instant::now();

    // Keep last state in case updates lag
    let mut has_state = false;

    loop {
        // Try to retrieve latest state (non-blocking)
        if let Ok(update) = state_rx.try_recv() {
            x = update.x;
            y = update.y;
            vx = update.vx;
            vy = update.vy;
            angle = update.angle;
            angular_vel = update.angular_vel;
            fuel_remaining = update.fuel_remaining;
            has_state = true;
        }

        // Compute commands
        // If we have no state yet, issue a safe default (hover-ish)
        let (thrust, turn) = if !has_state {
            (0.0f64, 0.0f64)
        } else {
            // Target vector toward landing pad
            let dx = map.target.0 - x;
            let dy = map.target.1 - y;

            // Desired heading so that forward points toward target
            // For angle 0, forward is (0,-1). The mapping is:
            // theta_target such that (sin(theta), -cos(theta)) ~ (dx, dy)
            // So theta_target = atan2(dx, -dy)
            let theta_target = dx.atan2(-dy);

            // Angle error (wrap to [-PI, PI])
            let angle_error = wrap_angle(theta_target - angle);
            // Turning command: scale the angular error to [-1, 1]
            // A simple proportional term with clamping
            let mut turn_cmd = angle_error * 2.0;
            turn_cmd = clamp(turn_cmd, -1.0, 1.0);

            // Simple PD to generate desired acceleration toward the target
            // a_des = Kp*(pos_error) - Kv*(vel)
            let a_des_x = k_p * dx - k_v * vx;
            let a_des_y = k_p * dy - k_v * vy;

            // Forward unit vector given current angle
            // 0 rad -> facing up (0,-1)
            let f_x = angle.sin();
            let f_y = -angle.cos();

            // Component of desired acceleration along forward direction
            let a_des_along = a_des_x * f_x + a_des_y * f_y;

            // Gravity component along the forward direction
            let g_along = map.gravity * f_y; // gravity is (0, gravity)

            // Required thrust along forward direction to achieve a_des along F
            let mut t_needed = a_des_along - g_along;
            if t_needed < 0.0 { t_needed = 0.0; } // cannot pull backwards with thrust

            // Normalize by max thrust to get [0,1]
            let mut thrust_cmd = if map.max_thrust > 0.0 {
                t_needed / map.max_thrust
            } else {
                0.0
            };
            thrust_cmd = clamp(thrust_cmd, 0.0, 1.0);

            // If out of fuel, reduce thrust
            if fuel_remaining <= 1e-6 {
                thrust_cmd = 0.0;
            }

            // Optional: near target, soften thrust for a gentle landing
            let near_target = dist((x, y), map.target) < 2.0;
            if near_target {
                thrust_cmd = clamp(thrust_cmd * 0.6, 0.0, 1.0);
            }

            (thrust_cmd, turn_cmd)
        };

        // Before outputting, ensure numbers are finite
        let thrust_out = if thrust.is_finite() { thrust } else { 0.0 };
        let turn_out = if thrust_out.is_finite() { thrust_out } else { 0.0 }; // placeholder; we'll compute proper turn

        // The above variable naming had a small mistake: we need to output thrust_out and turn_cmd.
        // We'll recompute properly:
        let (thrust_out, turn_out) = if !has_state {
            (0.0, 0.0)
        } else {
            // Recompute to capture the correct "turn" value
            let dx = map.target.0 - x;
            let dy = map.target.1 - y;
            let theta_target = dx.atan2(-dy);
            let angle_error = wrap_angle(theta_target - angle);
            let turn_cmd = clamp(angle_error * 2.0, -1.0, 1.0);

            // Recompute thrust as above (duplicate to ensure correct values)
            let a_des_x = k_p * dx - k_v * vx;
            let a_des_y = k_p * dy - k_v * vy;
            let f_x = angle.sin();
            let f_y = -angle.cos();
            let a_des_along = a_des_x * f_x + a_des_y * f_y;
            let g_along = map.gravity * f_y;
            let mut t_needed = a_des_along - g_along;
            if t_needed < 0.0 { t_needed = 0.0; }
            let mut thrust_cmd = if map.max_thrust > 0.0 { t_needed / map.max_thrust } else { 0.0 };
            thrust_cmd = clamp(thrust_cmd, 0.0, 1.0);
            if fuel_remaining <= 1e-6 { thrust_cmd = 0.0; }

            let near_target = dist((x, y), map.target) < 2.0;
            if near_target {
                thrust_cmd = clamp(thrust_cmd * 0.6, 0.0, 1.0);
            }

            (thrust_cmd, turn_cmd)
        };

        // Send output (two decimals, per spec)
        // Note: turn_out is in [-1,1], thrust_out in [0,1]
        // We print with two decimals as requested
        let _ = writeln!(stdout, "{:.2} {:.2}", thrust_out, turn_out);
        stdout.flush().ok();

        // Sleep until next tick, but adjust for drift
        let now = Instant::now();
        let elapsed = now.duration_since(last_tick);
        if elapsed < period {
            thread::sleep(period - elapsed);
        }
        last_tick = Instant::now();

        // Optional: Stop condition (very soft landing)
        // If very close to target and velocity is small, you could keep outputting zeros.
        // This is left as a simple heuristic; the host environment determines success.
    }
}
