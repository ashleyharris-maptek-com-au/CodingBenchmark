```rust
use std::collections::{HashMap, VecDeque};
use std::io::{self, Read, Write};

fn is_wall(walls: &Vec<Vec<u8>>, r: usize, c: usize, w: usize) -> bool {
    // walls[r] is a bit-packed row where each bit represents a cell:
    // 1 means wall, 0 means free.
    let byte = c / 8;
    let bit = c % 8;
    (walls[r][byte] & (1u8 << bit)) != 0
}

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut lines = input.lines();

    // Line 1: H W
    let first = lines.next().unwrap_or("0 0");
    let mut wh = first.split_whitespace();
    let h: usize = wh.next().unwrap_or("0").parse().unwrap_or(0);
    let w: usize = wh.next().unwrap_or("0").parse().unwrap_or(0);

    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    // Each row is bit-packed: (w + 7) / 8 bytes
    let row_bytes = (w + 7) / 8;
    let mut walls: Vec<Vec<u8>> = Vec::with_capacity(h);

    let mut ax: usize = 0;
    let mut ay: usize = 0;
    let mut bx: usize = 0;
    let mut by: usize = 0;
    let mut found_a = false;
    let mut found_b = false;

    for r in 0..h {
        let line = lines.next().unwrap_or("");
        // Pad line if it is shorter than W to avoid panics
        let mut s = line.to_string();
        if s.len() < w {
            s.push_str(&" ".repeat(w - s.len()));
        }

        let bytes = s.as_bytes();
        let mut row = vec![0u8; row_bytes];

        for c in 0..w {
            let ch = bytes[c];
            if ch == b'#' {
                // mark as wall
                let idx = c;
                let byte = idx / 8;
                let bit = idx % 8;
                row[byte] |= 1u8 << bit;
            } else if ch == b'A' {
                ax = r;
                ay = c;
                found_a = true;
            } else if ch == b'B' {
                bx = r;
                by = c;
                found_b = true;
            }
            // spaces and other characters are treated as open paths
        }

        walls.push(row);
    }

    // If start or end not found, there's no valid path
    if !found_a || !found_b {
        println!("0");
        return;
    }

    // Start and end indices (flattened)
    let start: u32 = (ax * w + ay) as u32;
    let end: u32 = (bx * w + by) as u32;
    let NONE: u32 = u32::MAX;

    // Bidirectional BFS
    let mut qf: VecDeque<u32> = VecDeque::new();
    let mut qb: VecDeque<u32> = VecDeque::new();
    let mut vis_fwd: HashMap<u32, u32> = HashMap::new(); // node -> parent
    let mut vis_bwd: HashMap<u32, u32> = HashMap::new(); // node -> parent

    qf.push_back(start);
    vis_fwd.insert(start, NONE);

    qb.push_back(end);
    vis_bwd.insert(end, NONE);

    let mut meeting: Option<u32> = None;

    // Helpers to know grid bounds
    // Pre-calculate total cells for index-to-coordinates
    // Note: W and H are up to 50k in the problem statement; we use u32 indices.

    while !qf.is_empty() && !qb.is_empty() && meeting.is_none() {
        if qf.len() <= qb.len() {
            // Expand forward frontier one layer
            let layer_size = qf.len();
            for _ in 0..layer_size {
                let cur = qf.pop_front().unwrap();
                let r = (cur as usize) / w;
                let c = (cur as usize) % w;

                // Up
                if r > 0 && !is_wall(&walls, r - 1, c, w) {
                    let nidx = ((r - 1) * w + c) as u32;
                    if !vis_fwd.contains_key(&nidx) {
                        vis_fwd.insert(nidx, cur);
                        if vis_bwd.contains_key(&nidx) {
                            meeting = Some(nidx);
                            break;
                        }
                        qf.push_back(nidx);
                    }
                }
                // Down
                if r + 1 < h && !is_wall(&walls, r + 1, c, w) {
                    let nidx = ((r + 1) * w + c) as u32;
                    if !vis_fwd.contains_key(&nidx) {
                        vis_fwd.insert(nidx, cur);
                        if vis_bwd.contains_key(&nidx) {
                            meeting = Some(nidx);
                            break;
                        }
                        qf.push_back(nidx);
                    }
                }
                // Left
                if c > 0 && !is_wall(&walls, r, c - 1, w) {
                    let nidx = (r * w + (c - 1)) as u32;
                    if !vis_fwd.contains_key(&nidx) {
                        vis_fwd.insert(nidx, cur);
                        if vis_bwd.contains_key(&nidx) {
                            meeting = Some(nidx);
                            break;
                        }
                        qf.push_back(nidx);
                    }
                }
                // Right
                if c + 1 < w && !is_wall(&walls, r, c + 1, w) {
                    let nidx = (r * w + (c + 1)) as u32;
                    if !vis_fwd.contains_key(&nidx) {
                        vis_fwd.insert(nidx, cur);
                        if vis_bwd.contains_key(&nidx) {
                            meeting = Some(nidx);
                            break;
                        }
                        qf.push_back(nidx);
                    }
                }
            }
        } else {
            // Expand backward frontier one layer
            let layer_size = qb.len();
            for _ in 0..layer_size {
                let cur = qb.pop_front().unwrap();
                let r = (cur as usize) / w;
                let c = (cur as usize) % w;

                // Up
                if r > 0 && !is_wall(&walls, r - 1, c, w) {
                    let nidx = ((r - 1) * w + c) as u32;
                    if !vis_bwd.contains_key(&nidx) {
                        vis_bwd.insert(nidx, cur);
                        if vis_fwd.contains_key(&nidx) {
                            meeting = Some(nidx);
                            break;
                        }
                        qb.push_back(nidx);
                    }
                }
                // Down
                if r + 1 < h && !is_wall(&walls, r + 1, c, w) {
                    let nidx = ((r + 1) * w + c) as u32;
                    if !vis_bwd.contains_key(&nidx) {
                        vis_bwd.insert(nidx, cur);
                        if vis_fwd.contains_key(&nidx) {
                            meeting = Some(nidx);
                            break;
                        }
                        qb.push_back(nidx);
                    }
                }
                // Left
                if c > 0 && !is_wall(&walls, r, c - 1, w) {
                    let nidx = (r * w + (c - 1)) as u32;
                    if !vis_bwd.contains_key(&nidx) {
                        vis_bwd.insert(nidx, cur);
                        if vis_fwd.contains_key(&nidx) {
                            meeting = Some(nidx);
                            break;
                        }
                        qb.push_back(nidx);
                    }
                }
                // Right
                if c + 1 < w && !is_wall(&walls, r, c + 1, w) {
                    let nidx = (r * w + (c + 1)) as u32;
                    if !vis_bwd.contains_key(&nidx) {
                        vis_bwd.insert(nidx, cur);
                        if vis_fwd.contains_key(&nidx) {
                            meeting = Some(nidx);
                            break;
                        }
                        qb.push_back(nidx);
                    }
                }
            }
        }
    }

    // Prepare output
    let mut out = String::new();
    if let Some(meet) = meeting {
        // Reconstruct path from start to meeting via vis_fwd
        let mut path1: Vec<u32> = Vec::new();
        let mut cur = meet;
        loop {
            path1.push(cur);
            let parent = vis_fwd[&cur];
            if parent == NONE {
                break;
            } else {
                cur = parent;
            }
        }
        path1.reverse();

        // Reconstruct path from meeting to end via vis_bwd
        let mut path2: Vec<u32> = Vec::new();
        let mut cur2 = meet;
        loop {
            path2.push(cur2);
            let parent = vis_bwd[&cur2];
            if parent == NONE {
                break;
            } else {
                cur2 = parent;
            }
        }

        // Combine: path1 (start..meeting) + path2 (meeting..end) skipping duplicate meeting
        let mut full_path: Vec<u32> = Vec::with_capacity(path1.len() + path2.len());
        full_path.extend(path1);
        for idx in path2.iter().skip(1) {
            full_path.push(*idx);
        }

        // Output
        let n = full_path.len();
        out.push_str(&format!("{}", n));
        out.push('\n');
        for idx in full_path {
            let r = (idx as usize) / w;
            let c = (idx as usize) % w;
            out.push_str(&format!("{} {}", r, c));
            out.push('\n');
        }
    } else {
        // No path
        out.push_str("0\n");
    }

    let mut writer = io::BufWriter::new(io::stdout());
    writer.write_all(out.as_bytes()).unwrap();
}
```
