OpCapability Geometry
OpMemoryModel Logical GLSL450

; GLSL std 450 helpers
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0

; gl_PerVertex input structure (for gl_Position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input array: 6 vertices (InputTrianglesAdjacency)
%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Per-vertex input arrays from vertex shader
%ptr_in_v3float = OpTypePointer Input %v3float
worldPos %worldPos = OpVariable %ptr_in_v3float Input
normal   %normal   = OpVariable %ptr_in_v3float Input
color    %color    = OpVariable %ptr_in_v3float Input

; Decorations for inputs (locations)
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Output: per-vertex gl_Position (in gl_PerVertex_out)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; Access to gl_out.gl_Position (first and only member)
%ptr_out_v4 = OpTypePointer Output %v4float

; Flat, location-scoped output color passed to fragment shader
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0
OpDecorate %outColor Flat

; Entry point
%main = OpFunction %void None %func
%entry = OpLabel
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %outColor

; Execution modes for GS
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Local temporaries for computations
; Positions
%pos0 = %v4float
%pos2 = %v4float
%pos4 = %v4float
; XYZ components as floats
%x0 = %float; %y0 = %float; %z0 = %float
%x2 = %float; %y2 = %float; %z2 = %float
%x4 = %float; %y4 = %float; %z4 = %float
; Vector differences a = p2.xyz - p0.xyz; b = p4.xyz - p0.xyz
%ax = %float; %ay = %float; %az = %float
%bx = %float; %by = %float; %bz = %float
; Cross product components
%cx = %float; %cy = %float; %cz = %float
; Normal and lighting
%N_raw = %v3float
%N = %v3float
%L = %v3float
%dotNL = %float
%diffuse = %float
%color0 = %v3float
%color2 = %v3float
%color4 = %v3float
%baseColor = %v3float
%finalColor = %v3float
; Color accumulation / averaging
%sum01 = %v3float
%sum = %v3float
%inv3 = OpConstant %float 0.3333333333
; Constants for lighting factors
%f02 = OpConstant %float 0.2
%f08 = OpConstant %float 0.8

; Load helper constants
%uint_0 = OpConstant %uint 0
%uint_2 = OpConstant %uint 2
%uint_4 = OpConstant %uint 4

; Load input positions from gl_in[0].gl_Position, gl_in[2].gl_Position, gl_in[4].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %uint_0
%p0 = OpLoad %v4float %p0_ptr

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %uint_0
%p2 = OpLoad %v4float %p2_ptr

%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_4 %uint_0
%p4 = OpLoad %v4float %p4_ptr

; Extract xyz for each
%x0 = OpCompositeExtract %float %p0 0
%y0 = OpCompositeExtract %float %p0 1
%z0 = OpCompositeExtract %float %p0 2

%x2 = OpCompositeExtract %float %p2 0
%y2 = OpCompositeExtract %float %p2 1
%z2 = OpCompositeExtract %float %p2 2

%x4 = OpCompositeExtract %float %p4 0
%y4 = OpCompositeExtract %float %p4 1
%z4 = OpCompositeExtract %float %p4 2

; a = p2.xyz - p0.xyz
%ax = OpFSub %float %x2 %x0
%ay = OpFSub %float %y2 %y0
%az = OpFSub %float %z2 %z0

; b = p4.xyz - p0.xyz
%bx = OpFSub %float %x4 %x0
%by = OpFSub %float %y4 %y0
%bz = OpFSub %float %z4 %z0

; cross = a x b
%ay_bz = OpFMul %float %ay %bz
%az_by = OpFMul %float %az %by
%cx = OpFSub %float %ay_bz % %az_by

%az_bx = OpFMul %float %az %bx
%ax_bz = OpFMul %float %ax %bz
%cy = OpFSub %float %az_bx % %ax_bz

%ax_by = OpFMul %float %ax %by
%ay_bx = OpFMul %float %ay %bx
%cz = OpFSub %float %ax_by % %ay_bx

; N_raw = (cx, cy, cz)
%N_raw = OpCompositeConstruct %v3float %cx %cy %cz
; N = Normalize(N_raw)
%N = OpExtInst %v3float %GLSLstd450 Normalize %N_raw

; L = Normalize(vec3(1,1,1))
%L_dir = OpCompositeConstruct %v3float %float_1 %float_1 %float_1
%L = OpExtInst %v3float %GLSLstd450 Normalize %L_dir

; dotNL = Dot(N, L)
%dotNL = OpExtInst %float %GLSLstd450 Dot %N %L

; diffuse = max(dotNL, 0.0)
%zero_f = OpConstant %float 0.0
%diffuse = OpExtInst %float %GLSLstd450 FMax %dotNL %zero_f

; colors: color[0], color[2], color[4]
%color0_ptr = OpAccessChain %ptr_in_v3float %color %uint_0
%color0 = OpLoad %v3float %color0_ptr

%color2_ptr = OpAccessChain %ptr_in_v3float %color %uint_2
%color2 = OpLoad %v3float %color2_ptr

%color4_ptr = OpAccessChain %ptr_in_v3float %color %uint_4
%color4 = OpLoad %v3float %color4_ptr

; baseColor = (color0 + color2 + color4) / 3
%sum01 = OpFAdd %v3float %color0 %color2
%sum = OpFAdd %v3float %sum01 %color4
;baseColor = sum * (1/3)
%baseColor = OpVectorTimesScalar %v3float %sum %inv3

; finalColor = baseColor * (0.2 + 0.8 * diffuse)
%term = OpFMul %float %diffuse %f08
%coef = OpFAdd %float %term %f02
%finalColor = OpVectorTimesScalar %v3float %baseColor %coef

; Emit three vertices of the triangle (p0, p2, p4) with the same finalColor
; Vertex 0
%gl_out_pos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_pos0_ptr %p0
%outColor0_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outColor0_ptr %finalColor
OpEmitVertex

; Vertex 2
%gl_out_pos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_pos2_ptr %p2
OpStore %outColor0_ptr %finalColor
OpEmitVertex

; Vertex 4
%gl_out_pos4_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_pos4_ptr %p4
OpStore %outColor0_ptr %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

; Helpers (constants)
float_1 = 1.0
float_0_2 = 0.2
float_0_8 = 0.8
float_0 = 0.0
float_3333 = 0.3333333333
