struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Mandelbrot set render using UVs mapped to c = (uv.x*3-2, uv.y*2-1)
    float2 c = float2(input.uv.x * 3.0 - 2.0, input.uv.y * 2.0 - 1.0);
    float2 z = float2(0.0, 0.0);
    int i;
    for (i = 0; i < 20; ++i) {
        float x = z.x;
        float y = z.y;
        z = float2(x * x - y * y + c.x, 2.0 * x * y + c.y);
        if (dot(z, z) > 4.0) break;
    }
    float t = float(i) / 20.0;
    float3 color = float3(t, t * 0.5, 1.0 - t);
    return float4(color, 1.0);
}
