struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    float3 N = normalize(input.normal);
    float3 pos = input.worldPos;

    float3 L = normalize(lightPos.xyz - pos);
    float diff = max(dot(N, L), 0.0);

    float lambert = 0.4 + 0.6 * diff;

    const float PI = 3.14159f;
    float s = sin(input.uv.y * PI * 16.0f);

    float3 red = float3(0.9f, 0.1f, 0.1f);
    float3 blue = float3(0.1f, 0.1f, 0.9f);

    float t = step(0.0f, s); // 1.0 when s >= 0, else 0.0
    float3 color = lerp(blue, red, t);

    float3 rgb = color * lambert;

    return float4(rgb, 1.0f);
}
