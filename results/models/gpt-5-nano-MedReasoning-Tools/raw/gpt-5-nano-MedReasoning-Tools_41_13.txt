; SPIR-V
; Version: 1.0
; Generator: Custom
; Bound: 999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450
%GLSL = OpExtInstImport "GLSL.std.450"

; Types
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%mat4 = OpTypeMatrix %v4float 4

; Uniform block (std140)
%UBOBlock = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBOBlock
; Pointers
%ptrInputV3 = OpTypePointer Input %v3float
%ptrOutputV4 = OpTypePointer Output %v4float

; Variables
%worldPosIn = OpVariable %ptrInputV3 Input
$outColor = OpVariable %ptrOutputV4 Output
%UBO = OpVariable %ptrUniformUBO Uniform

; Decorations
OpDecorate %worldPosIn Location 0
OpDecorate %outColor Location 0
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0
OpDecorate %UBOBlock Block
OpMemberDecorate %UBOBlock 0 Offset 0
OpMemberDecorate %UBOBlock 1 Offset 64
OpMemberDecorate %UBOBlock 2 Offset 128
OpMemberDecorate %UBOBlock 3 Offset 192
OpMemberDecorate %UBOBlock 4 Offset 208
OpMemberDecorate %UBOBlock 5 Offset 224

; Constants
%zero = OpConstant %float 0.0
%one = OpConstant %float 1.0
%half = OpConstant %float 0.5

; Bottom/top colors
%botR = OpConstant %float 0.1
%botG = OpConstant %float 0.4
%botB = OpConstant %float 0.1
%bottomVec3 = OpConstantComposite %v3float %botR %botG %botB

%topR = OpConstant %float 1.0
%topG = OpConstant %float 1.0
%topB = OpConstant %float 1.0
%topVec3 = OpConstantComposite %v3float %topR %topG %topB

; Function types
%void = OpTypeVoid
%funcType = OpTypeFunction %void

; Main function
%main = OpFunction %void None %funcType
%entry = OpLabel
; Compute t = clamp(worldPos.y * 0.5 + 0.5)
%worldPosVal = OpLoad %v3float %worldPosIn
%worldPosY = OpCompositeExtract %float %worldPosVal 1
%t1 = OpFMul %float %worldPosY %half
%t = OpFAdd %float %t1 %half
%tClamped = OpExtInst %float %GLSL FClamp %t %zero %one
; Color = mix(bottomVec3, topVec3, tClamped)
%colorVec3 = OpExtInst %v3float %GLSL FMix %bottomVec3 %topVec3 %tClamped
; Final vec4 with alpha = 1.0
%finalColor = OpCompositeConstruct %v4float %colorVec3 %one
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd

; Entry point and execution mode
OpEntryPoint Fragment %main "main" %worldPosIn %outColor
OpExecutionMode %main OriginUpperLeft
