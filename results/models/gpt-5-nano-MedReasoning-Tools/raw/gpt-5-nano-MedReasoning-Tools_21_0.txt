#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <functional>
#include <cmath>
#include <limits>
#include <algorithm>

// N-D Point with dynamic dimension
struct Point {
    std::vector<int> c; // coordinates, size = D
    bool operator==(const Point& other) const {
        return c == other.c;
    }
};

// Hash for Point so we can store in unordered_set
struct PointHash {
    std::size_t operator()(const Point& p) const noexcept {
        // FNV-1a like combine for speed and simplicity
        std::size_t h = 1469598103934665603ULL;
        for (int v : p.c) {
            h ^= static_cast<std::size_t>(v) + 0x9e3779b97f4a7c15ULL;
            h *= 1099511628211ULL;
        }
        return h;
    }
};

// Manhattan distance in N-D
inline int manhattan(const Point& a, const Point& b) {
    int s = 0;
    for (size_t i = 0; i < a.c.size(); ++i) {
        s += std::abs(a.c[i] - b.c[i]);
    }
    return s;
}

int main() {
    // Fast IO
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    // Input parsing (whitespace separated)
    int D;
    long long max_turns_ll;
    if (!(std::cin >> D >> max_turns_ll)) return 0;
    int max_turns = static_cast<int>(max_turns_ll);

    // Bounds per dimension
    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];

    // Snake initial: length and head position
    int initial_length;
    std::vector<int> head_coords(D);
    std::cin >> initial_length;
    for (int i = 0; i < D; ++i) std::cin >> head_coords[i];

    // Build initial body: along axis 0 decreasing from head
    // Body order: front() is head, back() is tail
    std::deque<Point> body;
    {
        Point head;
        head.c = head_coords;
        // Create L segments: head, head-(1,0,0,...), ..., head-(L-1,0,0,...)
        for (int t = 0; t < initial_length; ++t) {
            Point p = head;
            p.c[0] = head_coords[0] - t;
            body.push_back(p);
        }
    }

    // Foods
    int F;
    std::unordered_set<Point, PointHash> foods;
    std::vector<Point> foods_vec; // for quick iteration if needed
    std::cin >> F;
    foods.reserve(static_cast<size_t>(F) * 2 + 4);
    for (int i = 0; i < F; ++i) {
        Point f;
        f.c.resize(D);
        for (int k = 0; k < D; ++k) std::cin >> f.c[k];
        if (foods.insert(f).second) {
            foods_vec.push_back(f);
        }
    }

    // Obstacles
    int O;
    std::unordered_set<Point, PointHash> obstacles;
    obstacles.reserve(static_cast<size_t>(O) * 2 + 4);
    std::cin >> O;
    for (int i = 0; i < O; ++i) {
        Point o;
        o.c.resize(D);
        for (int k = 0; k < D; ++k) std::cin >> o.c[k];
        obstacles.insert(o);
    }

    // Pre-calculate initial head
    auto head = [&](const std::deque<Point>& b) -> const Point& { return b.front(); };

    // If there are no foods to begin with, nothing to output
    if (foods.empty()) return 0;

    // Helpers
    auto in_bounds = [&](const Point& p) -> bool {
        for (int i = 0; i < D; ++i) {
            if (p.c[i] < 0 || p.c[i] >= bounds[i]) return false;
        }
        return true;
    };

    // Main loop: greedy path toward nearest food with simple collision checks
    int turn = 0;
    while (turn < max_turns) {
        // Re-evaluate current head
        head = [&](const std::deque<Point>& b) -> const Point& { return b.front(); }(body);

        // If no foods remain, we ended with a win previously; break.
        if (foods.empty()) break;

        // Generate candidate moves: (axis, dir)
        struct Candidate { int axis; int dir; int dist; };
        std::vector<Candidate> candidates;
        const Point& h = head;
        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, +1}) {
                Point nh = h;
                nh.c[axis] += dir;
                if (!in_bounds(nh)) continue;

                // Collision checks
                bool is_food = (foods.find(nh) != foods.end());

                // Determine if nh collides with body, with tail exception depending on growth
                bool in_body = false;
                bool equals_tail = false;
                // Iterate body to see if nh matches any cell
                size_t idx = 0;
                for (const auto& seg : body) {
                    if (seg == nh) {
                        in_body = true;
                        if (idx == body.size() - 1) equals_tail = true; // tail
                        break;
                    }
                    ++idx;
                }

                bool allowed = true;
                if (in_body) {
                    // Allowed only if nh equals current tail AND we are not growing (no food at nh)
                    if (equals_tail && !is_food) {
                        allowed = true;
                    } else {
                        allowed = false;
                    }
                }

                // If not allowed, skip
                if (!allowed) continue;

                // Heuristic: distance to nearest food after this move
                int bestdist = std::numeric_limits<int>::max();
                if (!foods.empty()) {
                    for (const auto& f : foods) {
                        int d = 0;
                        for (int k = 0; k < D; ++k) d += std::abs(nh.c[k] - f.c[k]);
                        if (d < bestdist) bestdist = d;
                    }
                }

                candidates.push_back({axis, dir, bestdist});
            }
        }

        // If no possible moves, game over (lose)
        if (candidates.empty()) {
            break;
        }

        // Pick best candidate: smallest distance to food, then by axis, dir
        std::sort(candidates.begin(), candidates.end(),
                  [](const Candidate& a, const Candidate& b) {
                      if (a.dist != b.dist) return a.dist < b.dist;
                      if (a.axis != b.axis) return a.axis < b.axis;
                      return a.dir < b.dir;
                  });

        const Candidate best = candidates.front();

        // Output the move for this turn
        std::cout << best.axis << ' ' << best.dir << "\n";

        // Apply move
        Point nh = head;
        nh.c[best.axis] += best.dir;

        bool is_food = (foods.find(nh) != foods.end());

        // Update body: push new head; if not eating, pop tail
        body.push_front(nh);
        if (!is_food) {
            body.pop_back();
        } else {
            // Eat food: remove from set
            foods.erase(nh);
            // If no more foods, we can end after this move
        }

        // If all food eaten, end (win)
        if (foods.empty()) {
            break;
        }

        ++turn;
    }

    return 0;
}
