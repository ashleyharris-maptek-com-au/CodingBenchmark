```csharp
using System;
using System.Globalization;

namespace OrbitalTSP
{
    class Program
    {
        static void Main()
        {
            CultureInfo.InvariantCulture = new CultureInfo("en-US");
            // Read first non-empty line: N mu
            string line = ReadNonEmptyLine();
            if (line == null) return;
            string[] p = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (p.Length < 2) return;
            int N = int.Parse(p[0]);
            // mu is provided but not used in this simplified solver
            double mu = double.Parse(p[1], CultureInfo.InvariantCulture);

            // Read start state (6 numbers)
            double[] startState = ReadSixDoublesLine();
            if (startState == null) return;

            // Read N stations (each 6 numbers)
            double[][] stations = new double[N][];
            for (int i = 0; i < N; i++)
            {
                double[] s = ReadSixDoublesLine();
                if (s == null) return;
                stations[i] = s;
            }

            // Build cost matrix in 6D space: distance in (r,v) state space
            // Dist6(a,b) = sqrt(sum_{k=0..5} (a[k]-b[k])^2)
            double[,] cost = new double[N, N];
            for (int i = 0; i < N; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    double c = Dist6(stations[i], stations[j]);
                    cost[i, j] = c;
                    cost[j, i] = c;
                }
            }

            double[] startTo = new double[N];
            for (int i = 0; i < N; i++)
                startTo[i] = Dist6(startState, stations[i]);

            int[] bestPath;
            if (N <= 8)
            {
                bestPath = SolveBruteForce(N, cost, startTo);
            }
            else
            {
                bestPath = SolveHeuristic(N, cost, startTo);
            }

            // Output: one line with N integers (0..N-1)
            Console.WriteLine(string.Join(" ", bestPath));
        }

        // Helpers

        // Read a line and skip empties
        static string ReadNonEmptyLine()
        {
            string line;
            while ((line = Console.ReadLine()) != null)
            {
                if (!string.IsNullOrWhiteSpace(line))
                    return line;
            }
            return null;
        }

        // Read a line containing 6 doubles
        static double[] ReadSixDoublesLine()
        {
            string line = ReadNonEmptyLine();
            if (line == null) return null;
            string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 6) return null;
            double[] arr = new double[6];
            for (int i = 0; i < 6; i++)
                arr[i] = double.Parse(parts[i], CultureInfo.InvariantCulture);
            return arr;
        }

        // 6D distance between two state vectors
        static double Dist6(double[] a, double[] b)
        {
            double s = 0.0;
            for (int i = 0; i < 6; i++)
            {
                double d = a[i] - b[i];
                s += d * d;
            }
            return Math.Sqrt(s);
        }

        // Brute-force solver for small N (<= 8)
        static int[] SolveBruteForce(int N, double[,] cost, double[] startTo)
        {
            int[] bestPath = null;
            double bestCost = double.MaxValue;

            int[] path = new int[N];
            bool[] used = new bool[N];

            void Dfs(int depth, double currentCost)
            {
                if (depth == N)
                {
                    double total = currentCost;
                    // add cost to first node from start
                    total += startTo[path[0]];
                    if (total < bestCost)
                    {
                        bestCost = total;
                        bestPath = (int[])path.Clone();
                    }
                    return;
                }

                for (int i = 0; i < N; i++)
                {
                    if (used[i]) continue;
                    path[depth] = i;
                    double newCost = currentCost;
                    if (depth > 0)
                        newCost += cost[path[depth - 1], path[depth]];
                    if (newCost >= bestCost) continue; // prune
                    used[i] = true;
                    Dfs(depth + 1, newCost);
                    used[i] = false;
                }
            }

            Dfs(0, 0.0);
            // In extremely rare case bestPath remains null (shouldn't), fallback to identity
            if (bestPath == null)
            {
                bestPath = new int[N];
                for (int i = 0; i < N; i++) bestPath[i] = i;
            }
            return bestPath;
        }

        // Heuristic solver for medium-to-large N
        static int[] SolveHeuristic(int N, double[,] cost, double[] startTo)
        {
            int[] bestPath = null;
            double bestCost = double.MaxValue;

            Random rng = new Random(123456);

            // 2-opt helpers on path with fixed start
            int[] path = new int[N];
            int[] scratch = new int[N];

            // Cost computation for a path
            double PathCost(int[] p)
            {
                double c = startTo[p[0]];
                for (int i = 1; i < N; i++) c += cost[p[i - 1], p[i]];
                return c;
            }

            // 2-opt delta when reversing [i..j]
            double ComputeDeltaTwoOpt(int i, int j, int[] p)
            {
                // affected edges:
                // If i == 0: Start -> p[0] changes to Start -> p[j]
                // else: p[i-1] -> p[i] changes to p[i-1] -> p[j]
                double delta = 0.0;
                if (i == 0)
                    delta += startTo[p[j]] - startTo[p[0]];
                else
                    delta += cost[p[i - 1], p[j]] - cost[p[i - 1], p[i]];

                // edge p[i] -> p[i+1] replaced by p[j] -> p[i+1]
                // this edge exists when i+1 <= j
                delta += cost[p[j], p[i + 1]] - cost[p[i], p[i + 1]];

                // edge p[j] -> p[j+1] replaced by p[i] -> p[j+1] (if j < N-1)
                if (j < N - 1)
                    delta += cost[p[i], p[j + 1]] - cost[p[j], p[j + 1]];

                return delta;
            }

            // Reverse [i..j] in-place
            void ReverseSegment(int[] arr, int i, int j)
            {
                while (i < j)
                {
                    int t = arr[i]; arr[i] = arr[j]; arr[j] = t;
                    i++; j--;
                }
            }

            // Improve a given path by 2-opt
            double TwoOptImprove(int[] p, double currentCost)
            {
                bool improved = true;
                while (improved)
                {
                    improved = false;
                    for (int i = 0; i < N - 1; i++)
                    {
                        for (int j = i + 1; j < N; j++)
                        {
                            double delta = ComputeDeltaTwoOpt(i, j, p);
                            if (delta < -1e-12)
                            {
                                // apply
                                ReverseSegment(p, i, j);
                                currentCost += delta;
                                improved = true;
                                // after an improvement, restart inner loops for stability
                            }
                            if (improved) break;
                        }
                        if (improved) break;
                    }
                }
                return currentCost;
            }

            // Greedy initializer
            int[] GreedyFromStartPath()
            {
                bool[] used = new bool[N];
                int[] p = new int[N];
                // first: nearest to start
                int first = 0;
                double best = startTo[0];
                for (int i = 1; i < N; i++)
                {
                    if (startTo[i] < best)
                    {
                        best = startTo[i];
                        first = i;
                    }
                }
                p[0] = first;
                used[first] = true;
                for (int k = 1; k < N; k++)
                {
                    int last = p[k - 1];
                    int bestIdx = -1;
                    double bestVal = double.MaxValue;
                    for (int i = 0; i < N; i++)
                    {
                        if (used[i]) continue;
                        if (cost[last, i] < bestVal)
                        {
                            bestVal = cost[last, i];
                            bestIdx = i;
                        }
                    }
                    p[k] = bestIdx;
                    used[bestIdx] = true;
                }
                return p;
            }

            int RESTARTS = Math.Min(200, Math.Max(50, N * 2));

            for (int r = 0; r < RESTARTS; r++)
            {
                // Build an initial path
                if (r == 0)
                {
                    path = GreedyFromStartPath();
                }
                else
                {
                    // random permutation
                    path = new int[N];
                    for (int i = 0; i < N; i++) path[i] = i;
                    // Fisher-Yates shuffle
                    for (int i = N - 1; i > 0; i--)
                    {
                        int j = rng.Next(i + 1);
                        int tmp = path[i]; path[i] = path[j]; path[j] = tmp;
                    }
                }

                double c = PathCost(path);
                // Improve with 2-opt
                c = TwoOptImprove(path, c);

                if (c < bestCost)
                {
                    bestCost = c;
                    bestPath = (int[])path.Clone();
                }
            }

            // If somehow nothing found (shouldn't happen), return a simple order
            if (bestPath == null)
            {
                bestPath = new int[N];
                for (int i = 0; i < N; i++) bestPath[i] = i;
            }

            return bestPath;
        }
    }
}
```
