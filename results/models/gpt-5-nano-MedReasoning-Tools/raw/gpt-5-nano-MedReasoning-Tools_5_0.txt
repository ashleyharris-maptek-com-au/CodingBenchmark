```rust
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct Frame {
    current: usize,
    candidates: Vec<usize>,
    next_index: usize,
}

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Parse header
    let width: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let height: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let obstacle_count: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    // require_cycle: 0 or 1
    let require_cycle: i32 = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => 0,
    };

    let total = width * height;

    // Blocked cells
    let mut blocked = vec![false; total];
    for _ in 0..obstacle_count {
        let x: isize = it.next().unwrap().parse().unwrap();
        let y: isize = it.next().unwrap().parse().unwrap();
        if x >= 0 && (x as usize) < width && y >= 0 && (y as usize) < height {
            blocked[(y as usize) * width + (x as usize)] = true;
        }
    }

    // Quick checks
    if blocked[0] {
        println!("0");
        return;
    }

    // Number of free cells we must visit
    let total_free: usize = total - obstacle_count;
    if total_free == 0 {
        // No free cells to visit; no valid path per problem's output format
        println!("0");
        return;
    }

    // Precompute adjacency for all cells (only between unblocked cells)
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); total];
    for id in 0..total {
        if blocked[id] { continue; }
        let x = id % width;
        let y = id / width;
        let mut neigh = Vec::new();
        if x > 0 {
            let nb = id - 1;
            if !blocked[nb] { neigh.push(nb); }
        }
        if x + 1 < width {
            let nb = id + 1;
            if !blocked[nb] { neigh.push(nb); }
        }
        if y > 0 {
            let nb = id - width;
            if !blocked[nb] { neigh.push(nb); }
        }
        if y + 1 < height {
            let nb = id + width;
            if !blocked[nb] { neigh.push(nb); }
        }
        adj[id] = neigh;
    }

    // Helper: number of unvisited neighbors for pruning
    fn unvisited_neighbors_count(nid: usize, visited: &Vec<bool>, blocked: &Vec<bool>, adj: &Vec<Vec<usize>>) -> usize {
        let mut cnt = 0usize;
        for &nb in &adj[nid] {
            if !blocked[nb] && !visited[nb] {
                cnt += 1;
            }
        }
        cnt
    }

    // Helper: connectivity check on remaining unvisited cells
    // Returns true if the set of unvisited (and unblocked) cells is connected.
    fn remaining_connected(visited: &Vec<bool>, blocked: &Vec<bool>, adj: &Vec<Vec<usize>>, total_free_remaining: usize) -> bool {
        // If no remaining cells, it's trivially connected
        if total_free_remaining == 0 { return true; }

        // Find a starting unvisited cell
        let n = visited.len();
        let mut start: Option<usize> = None;
        for i in 0..n {
            if !blocked[i] && !visited[i] {
                start = Some(i);
                break;
            }
        }
        let start_id = match start {
            Some(v) => v,
            None => return true, // nothing remaining
        };

        // DFS over unvisited cells
        let mut stack: Vec<usize> = Vec::new();
        stack.push(start_id);
        let mut seen = vec![false; n];
        seen[start_id] = true;
        let mut count = 1usize;

        while let Some(v) = stack.pop() {
            for &nb in &adj[v] {
                if !blocked[nb] && !visited[nb] && !seen[nb] {
                    seen[nb] = true;
                    stack.push(nb);
                    count += 1;
                }
            }
        }

        count == total_free_remaining
    }

    // Time limit to avoid timeouts (approx 4.5 minutes)
    let time_limit = Duration::from_secs(270); // 4.5 minutes
    let start_time = Instant::now();

    // Special case: if only 1 free cell, path is just (0,0)
    if total_free == 1 {
        // If cycle is required, it's not possible (a single cell isn't adjacent to itself)
        if require_cycle != 0 {
            println!("0");
            return;
        } else {
            println!("1");
            println!("0 0");
            return;
        }
    }

    // DFS with explicit stack to avoid deep recursion
    let mut visited = vec![false; total];
    visited[0] = true;
    let mut path: Vec<usize> = vec![0];

    // Initialize first frame (from start)
    // Only consider unvisited, unblocked neighbors
    let mut initial_candidates: Vec<usize> = adj[0]
        .iter()
        .cloned()
        .filter(|&nb| !blocked[nb] && !visited[nb])
        .collect();

    // Sort initial candidates by their unvisited neighbor count (Warnsdorff-like heuristic)
    initial_candidates.sort_by_key(|&nb| unvisited_neighbors_count(nb, &visited, &blocked, &adj));

    let mut stack: Vec<Frame> = Vec::new();
    stack.push(Frame {
        current: 0,
        candidates: initial_candidates,
        next_index: 0,
    });

    // Main loop
    // We attempt to extend path until we either find a full path or exhaust search/time.
    while let Some(frame) = stack.last_mut() {
        // Time check
        if start_time.elapsed() > time_limit {
            // Time limit reached; report no path found within limit
            println!("0");
            return;
        }

        // If we've exhausted all candidates for current frame, backtrack
        if frame.next_index >= frame.candidates.len() {
            // Backtrack: remove current node from path and stack, unmark visited
            let last_id = path.pop().unwrap();
            visited[last_id] = false;
            stack.pop();
            // If there is no previous frame, search failed
            if stack.is_empty() {
                println!("0");
                return;
            }
            // Continue with the previous frame (its next_index already points to the next candidate)
            continue;
        }

        // Try the next candidate
        let next_id = frame.candidates[frame.next_index];
        // Advance next_index for the frame since we're attempting this candidate now
        frame.next_index += 1;

        // Safety: skip if somehow blocked/visited (shouldn't happen due to filtering)
        if blocked[next_id] || visited[next_id] {
            continue;
        }

        // Move to next_id
        path.push(next_id);
        visited[next_id] = true;

        // Remaining unvisited cells
        let remaining = total_free - path.len();

        // If we've filled all cells, check cycle condition and output
        if remaining == 0 {
            // If require_cycle is True, last must be adjacent to start (0,0)
            if require_cycle != 0 {
                let last = *path.last().unwrap();
                let dx = (last as isize % width as isize) - 0;
                let dy = (last as isize / width as isize) - 0;
                // Check adjacency
                let adjs = (dx.abs() + dy.abs()) == 1;
                if adjs {
                    // Success: print path
                } else {
                    // Not a valid cycle; backtrack
                    path.pop();
                    visited[next_id] = false;
                    // Also remove the current frame and try next candidates
                    // We simulate by continuing; the frame is still on stack with next candidates
                    continue;
                }
            }
            // Print result
            let mut out = String::new();
            out.push_str(&format!("{}", path.len()));
            out.push('\n');
            for &vid in &path {
                let x = vid % width;
                let y = vid / width;
                out.push_str(&format!("{} {}\n", x, y));
            }
            print!("{}", out);
            return;
        }

        // Connectivity pruning: ensure remaining unvisited cells are connected
        // If not connected, prune this branch
        if !remaining_connected(&visited, &blocked, &adj, remaining) {
            // prune
            path.pop();
            visited[next_id] = false;
            // Note: the top frame's next_index already advanced; proceed to try next candidate on this frame
            continue;
        }

        // If connected, push a new frame for the new current cell (next_id)
        // Build its candidate list (unvisited neighbors)
        let mut new_cands: Vec<usize> = adj[next_id]
            .iter()
            .cloned()
            .filter(|&nb| !blocked[nb] && !visited[nb])
            .collect();
        new_cands.sort_by_key(|&nb| unvisited_neighbors_count(nb, &visited, &blocked, &adj));

        stack.push(Frame {
            current: next_id,
            candidates: new_cands,
            next_index: 0,
        });

        // Loop continues
    }

    // If loop ends without returning, no path found within constraints
    println!("0");
}

// Frame struct for iterative DFS
struct Frame {
    current: usize,
    candidates: Vec<usize>,
    next_index: usize,
}
```
