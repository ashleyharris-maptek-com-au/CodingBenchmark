; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_Mat4 = OpTypePointer Uniform %mat4
%_ptr_Uniform_Vec4 = OpTypePointer Uniform %vec4
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%_ptr_Function_float = OpTypePointer Function %float
%_bool = OpTypeBool

%GLSL = OpExtInstImport "GLSL.std.450"

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%ubo = OpVariable %_ptr_Uniform_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%in_worldPos = OpVariable %_ptr_Input_vec3 Input
OpDecorate %in_worldPos Location 0
%in_normal = OpVariable %_ptr_Input_vec3 Input
OpDecorate %in_normal Location 1
%in_color = OpVariable %_ptr_Input_vec3 Input
OpDecorate %in_color Location 4

%outColor = OpVariable %_ptr_Output_vec4 Output
OpDecorate %outColor Location 0

; Function and entry point declarations
%void = OpTypeVoid
%_func_type = OpTypeFunction %void
%main = OpFunction %void None %_func_type
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%norm = OpLoad %vec3 %in_normal
%colorV = OpLoad %vec3 %in_color

; Light pos from UBO
%lightPosPtr = OpAccessChain %_ptr_Uniform_Vec4 %ubo 3
%lightPos = OpLoad %vec4 %lightPosPtr
%Lx = OpCompositeExtract %float %lightPos 0
%Ly = OpCompositeExtract %float %lightPos 1
%Lz = OpCompositeExtract %float %lightPos 2
%Lxyz = OpCompositeConstruct %vec3 %Lx %Ly %Lz

%N = OpExtInst %vec3 %GLSL Normalize %norm
%LtoW = OpFSub %vec3 %Lxyz %worldPos
%Ldir = OpExtInst %vec3 %GLSL Normalize %LtoW
%d = OpExtInst %float %GLSL Dot %N %Ldir

; Constants for toon shading bands
%const_075 = OpConstant %float 0.75
%const_05  = OpConstant %float 0.50
%const_025 = OpConstant %float 0.25
%const_100 = OpConstant %float 1.00
%const_07  = OpConstant %float 0.70
%const_04  = OpConstant %float 0.40
%const_02  = OpConstant %float 0.20

; Booleans for nested selects
%cond75 = OpFOrdGreaterThanEqual %_bool %d %%const_075
%cond50 = OpFOrdGreaterThanEqual %_bool %d %%const_05
%cond25 = OpFOrdGreaterThanEqual %_bool %d %%const_025

; Toon band value computed via nested selects
%bandTemp0 = OpSelect %float %cond25 %%const_04 %%const_02
%bandTemp1 = OpSelect %float %cond50 %%const_07 %bandTemp0
%bandVal = OpSelect %float %cond75 %%const_100 %bandTemp1

; Apply band to vertex color and output with alpha 1.0
%scaled = OpVectorTimesScalar %vec3 %colorV %bandVal
%outVec4 = OpCompositeConstruct %vec4 %scaled %%const_100
OpStore %outColor %outVec4

OpReturn
OpFunctionEnd

OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main OriginUpperLeft
