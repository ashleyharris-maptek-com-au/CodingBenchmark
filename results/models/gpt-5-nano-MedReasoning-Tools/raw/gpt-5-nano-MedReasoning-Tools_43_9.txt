struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normal in world space
    float3 N = normalize(input.normal);

    // Light direction (world space)
    float3 light = lightPos.xyz;
    float3 L = normalize(light - input.worldPos);

    // Lambertian diffuse
    float diff = max(dot(N, L), 0.0);
    float lambert = 0.5 + 0.5 * diff;

    // 8x8 UV checkerboard
    int ix = (int)floor(input.uv.x * 8.0);
    int iy = (int)floor(input.uv.y * 8.0);
    ix = min(ix, 7);
    iy = min(iy, 7);
    int parity = (ix + iy) & 1;

    float3 checkerColor = (parity == 0) ? float3(0.9, 0.9, 0.9) : float3(0.2, 0.2, 0.2);

    float3 finalColor = checkerColor * lambert;

    return float4(finalColor, 1.0);
}
