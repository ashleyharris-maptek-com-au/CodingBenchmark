; SPIR-V Geometry Shader to expand each input point into a 4-vertex diamond (triangle strip)
; Geometry shader: inputs a single point, outputs a 4-vertex triangle strip (diamond)
; Each emitted vertex uses the input point's color

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%one_0 = OpConstant %int 0
%zero_0 = OpConstant %uint 0
%f_0_05 = OpConstant %float 0.05
%f_1_0  = OpConstant %float 1.0

; gl_PerVertex input (position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input arrays (per-vertex)
%uint_1 = OpConstant %uint 1

; World position input array (location = 0)
%worldPos_ty = OpTypeArray %v3float %uint_1
%ptr_in_worldPos = OpTypePointer Input %worldPos_ty
%worldPos = OpVariable %ptr_in_worldPos Input
OpDecorate %worldPos Location 0

; Normal input array (location = 1) - declared for completeness
%normal_ty = OpTypeArray %v3float %uint_1
%ptr_in_normal = OpTypePointer Input %normal_ty
%normal = OpVariable %ptr_in_normal Input
OpDecorate %normal Location 1

; Color input array (location = 2)
%color_ty = OpTypeArray %v3float %uint_1
%ptr_in_color = OpTypePointer Input %color_ty
%color = OpVariable %ptr_in_color Input
OpDecorate %color Location 2

; gl_PerVertex output (position)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

; Output color (location = 0)
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Function types
%void = OpTypeVoid
%func_void = OpTypeFunction %void

; Entry point and execution modes
%main = OpFunction %void None %func_void
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; gl_in declaration for access (input primitive is 1 vertex)
%gl_in = OpVariable %ptr_in_arr_pv Input
; Note: The actual array type %ptr_in_arr_pv is defined below with the gl_PerVertex_in type

; Emit vertex helper pointers (for clarity)
; Pointer to the emitted vertex's gl_Position
%gl_out_pos_ptr = OpTypePointer Output %v4float

; Body of main
%entry = OpLabel

; Load input position: gl_in[0].gl_Position
%gl_in_0_ptr = OpAccessChain %ptr_in_v4 %gl_in %zero_0 %zero_0
%pos_in = OpLoad %v4float %gl_in_0_ptr

; Decompose input position
%x = OpCompositeExtract %float %pos_in 0
%y = OpCompositeExtract %float %pos_in 1
%z = OpCompositeExtract %float %pos_in 2
%w = OpCompositeExtract %float %pos_in 3

; Compute diamond vertices (top, right, left, bottom)
%top_y = OpFAdd %float %y % %f_0_05
%top = OpCompositeConstruct %v4float %x %top_y %z %w

%right_x = OpFAdd %float %x % %f_0_05
%right = OpCompositeConstruct %v4float %right_x %y %z %w

%left_x = OpFSub %float %x % %f_0_05
%left = OpCompositeConstruct %v4float %left_x %y %z %w

%bottom_y = OpFSub %float %y % %f_0_05
%bottom = OpCompositeConstruct %v4float %x %bottom_y %z %w

; Load input color for the single vertex (color[0])
%color_elem_ptr = OpAccessChain %ptr_color_elem %color %zero_0
%color0 = OpLoad %v3float %color_elem_ptr

; Set per-vertex color (same for all emitted vertices)
OpStore %outColor %color0

; Emit top vertex
%gl_out_top_ptr = OpAccessChain %ptr_out_v4 %gl_out %zero_0
OpStore %gl_out_top_ptr %top
OpEmitVertex

; Emit right vertex
%gl_out_right_ptr = OpAccessChain %ptr_out_v4 %gl_out %zero_0
OpStore %gl_out_right_ptr %right
OpEmitVertex

; Emit left vertex
%gl_out_left_ptr = OpAccessChain %ptr_out_v4 %gl_out %zero_0
OpStore %gl_out_left_ptr %left
OpEmitVertex

; Emit bottom vertex
%gl_out_bottom_ptr = OpAccessChain %ptr_out_v4 %gl_out %zero_0
OpStore %gl_out_bottom_ptr %bottom
OpEmitVertex

; End primitive and return
OpEndPrimitive
OpReturn
End

; Decorations for interface variables
OpDecorate %gl_in Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Note: The exact IDs used above (gl_in, worldPos, normal, color, gl_out, outColor) are placeholders in this textual SPIR-V snippet to illustrate the required structure. Replace IDs consistently when assembling.
