#version 450

layout(location = 0) in vec3 worldPos;   // world-space position
layout(location = 1) in vec3 normal;     // world-space normal (interpolated, may need normalize)
layout(location = 2) in vec2 uv;         // texture coordinates in [0,1]
layout(location = 3) in vec3 tangent;    // tangent vector
layout(location = 4) in vec3 color;      // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;          // offset 0
    mat4 view;           // offset 64
    mat4 projection;     // offset 128
    vec4 lightPos;       // value: (5.0, 5.0, 5.0, 1.0)
    vec4 cameraPos;      // value: (0.0, 0.0, 3.0, 1.0)
    vec4 params;         // value: (0.0, 0.5, 0.0, 0.0)
} ubo;

layout(location = 0) out vec4 outColor;  // RGBA float, stored as rgba8unorm

void main() {
    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);

    // Procedural brick texture in UV space
    // 8 bricks across (U) and 16 bricks across (V)
    float px = uv.x * 8.0;
    float py = uv.y * 16.0;

    // Stagger every other row by offsetting U by 0.5
    float row = floor(py);
    float offset = mod(row, 2.0) * 0.5;
    float uInBrick = fract(px + offset);
    float vInBrick = fract(py);

    // Mortar thickness in UV units
    const float mortarX = 0.05;
    const float mortarY = 0.1;

    bool isBrick = (uInBrick >= mortarX) && (uInBrick <= 1.0 - mortarX) &&
                   (vInBrick >= mortarY) && (vInBrick <= 1.0 - mortarY);

    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 baseColor = isBrick ? brickColor : mortarColor;

    // Apply Lambertian shading with the given factor
    float shade = 0.4 + 0.6 * diffuse;
    vec3 finalColor = baseColor * shade;

    outColor = vec4(finalColor, 1.0);
}
