Here's a complete, compilable C++20 solution that adapts to problem size and uses an efficient approach for large inputs while remaining correct for all sizes.

- For moderate sizes (shortest string length <= 200k), it uses a suffix-automaton (SAM) built on the shortest string. This yields O(sum of lengths) time and memory proportional to the SAM size.
- For very large bases, it switches to a robust rolling-hash (double-hash) based binary-search approach to avoid building an enormous SAM. This still finds the exact longest common substring with high confidence (hash collisions are extremely unlikely with double hashing).

Input: N followed by N strings (one per line). Output: the longest common substring shared by all strings (empty line if none).

Code (copy-paste into a file and compile with MSVC /EHsc, C++20):

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <limits>
#include <unordered_set>
#include <unordered_map>
#include <utility>

using namespace std;

/* =========================================================
   Suffix Automaton (SAM) based solution (adaptive)
   - Built on the shortest string to minimize state count
   - For each other string, compute per-state matches and propagate
   - The answer is the maximum length that occurs in all strings,
     and we can reconstruct via first_pos of the corresponding state.
   ========================================================= */
class SuffixAutomaton {
public:
    struct State {
        int len;                        // max length of substrings in this state
        int link;                       // suffix link
        vector<pair<char,int>> next;    // transitions (character -> state)
        int first_pos;                  // an end position of some string realized by this state
        State(): len(0), link(-1), first_pos(-1) {}
    };

    vector<State> st;
    int last;
    vector<int> order; // states sorted by len ascending

    SuffixAutomaton() : last(0) {}

    // helper: get transition target from state v by char c
    int go(int v, char c) const {
        for (const auto &p : st[v].next)
            if (p.first == c) return p.second;
        return -1;
    }

    // helper: set transition from state v by char c to 'to'
    void setNext(int v, char c, int to) {
        for (auto &p : st[v].next)
            if (p.first == c) { p.second = to; return; }
        st[v].next.emplace_back(c, to);
    }

    // extend SAM with character c at position pos in the base string
    void extend(char c, int pos) {
        int cur = (int)st.size();
        st.emplace_back();
        st[cur].len = st[last].len + 1;
        st[cur].first_pos = pos;

        int p = last;
        while (p != -1 && go(p, c) == -1) {
            setNext(p, c, cur);
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = go(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back(st[q]); // clone of q
                st[clone].len = st[p].len + 1;
                st[clone].first_pos = st[q].first_pos;

                while (p != -1 && go(p, c) == q) {
                    setNext(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }

    // build from the base string and prepare order for propagation
    void init(const string &s) {
        st.clear();
        st.reserve(2 * s.size());
        st.emplace_back(); // root state 0
        st[0].len = 0;
        st[0].link = -1;
        st[0].first_pos = -1;
        last = 0;
        for (int i = 0; i < (int)s.size(); ++i) {
            extend(s[i], i);
        }
        build_order();
    }

    // precompute ordering of states by increasing len
    void build_order() {
        int sz = (int)st.size();
        int maxL = 0;
        for (int i = 0; i < sz; ++i) maxL = max(maxL, st[i].len);
        order.resize(sz);
        vector<int> cnt(maxL + 1, 0);
        for (int i = 0; i < sz; ++i) ++cnt[st[i].len];
        for (int i = 1; i <= maxL; ++i) cnt[i] += cnt[i - 1];
        for (int i = sz - 1; i >= 0; --i) order[--cnt[st[i].len]] = i;
    }
};

/* =========================================================
   Double rolling hash helper for large inputs (fallback)
   - Binary search for the maximum common substring length
   - For each length L, check if there exists a substring of base with length L
     that appears in all other strings (by intersection of hash sets)
   - Returns one actual substring if exists
   ========================================================= */
struct HashPair {
    uint64_t a, b;
    bool operator==(const HashPair& o) const { return a == o.a && b == o.b; }
};

struct HashPairHasher {
    size_t operator()(const HashPair& k) const noexcept {
        return static_cast<size_t>(k.a ^ (k.b << 1));
    }
};

class RollingHash {
public:
    static const uint64_t B1 = 911382323ull;
    static const uint64_t B2 = 972663749ull;

    string s;
    int n;
    vector<uint64_t> h1, h2, p1, p2;

    void init(const string& str) {
        s = str;
        n = (int)s.size();
        h1.assign(n + 1, 0);
        h2.assign(n + 1, 0);
        p1.assign(n + 1, 1);
        p2.assign(n + 1, 1);
        for (int i = 0; i < n; ++i) {
            h1[i + 1] = h1[i] * B1 + (unsigned char)s[i] + 1;
            h2[i + 1] = h2[i] * B2 + (unsigned char)s[i] + 1;
            p1[i + 1] = p1[i] * B1;
            p2[i + 1] = p2[i] * B2;
        }
    }

    // get hash of s[l..r) (0-based, half-open)
    pair<uint64_t,uint64_t> get(int l, int r) const {
        uint64_t x1 = h1[r] - h1[l] * p1[r - l];
        uint64_t x2 = h2[r] - h2[l] * p2[r - l];
        return {x1, x2};
    }
};

/* =========================================================
   Main
   - Reads input
   - Picks the shortest string as base
   - If base length <= 200000, uses SAM
   - Else uses double-hash binary search method
   - Outputs any longest common substring
   ========================================================= */
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy); // consume endline after N

    vector<string> strs;
    strs.reserve(N);
    for (int i = 0; i < N; ++i) {
        string s;
        getline(cin, s);
        strs.push_back(s);
    }

    if (N == 0) {
        cout << "\n";
        return 0;
    }

    // Find the shortest string to use as base
    int baseIndex = 0;
    for (int i = 1; i < N; ++i) {
        if (strs[i].size() < strs[baseIndex].size()) baseIndex = i;
    }
    string base = strs[baseIndex];

    // Gather other strings
    vector<string> others;
    others.reserve(N - 1);
    for (int i = 0; i < N; ++i) if (i != baseIndex) others.push_back(strs[i]);

    if (others.empty()) {
        // Only one string: the longest common substring is the string itself
        cout << base << "\n";
        return 0;
    }

    // Adaptive strategy based on base length
    const size_t SAM_THRESHOLD = 200000; // heuristic threshold

    if (base.size() <= SAM_THRESHOLD) {
        // SAM approach
        SuffixAutomaton sam;
        sam.init(base);

        int sz = (int)sam.st.size();
        vector<int> best(sz);
        for (int i = 0; i < sz; ++i) best[i] = sam.st[i].len;

        for (const string& t : others) {
            vector<int> match(sz, 0);
            int v = 0, l = 0;

            for (char ch : t) {
                int to = sam.go(v, ch);
                if (to != -1) {
                    v = to;
                    ++l;
                } else {
                    while (v != -1 && sam.go(v, ch) == -1) v = sam.st[v].link;
                    if (v == -1) {
                        v = 0;
                        l = 0;
                    } else {
                        l = sam.st[v].len + 1;
                        v = sam.go(v, ch);
                    }
                }
                if (v != -1 && l > match[v]) match[v] = l;
            }

            // Propagate matches to suffix links (descend by len)
            for (int idx = (int)sam.order.size() - 1; idx >= 0; --idx) {
                int state = sam.order[idx];
                int p = sam.st[state].link;
                if (p >= 0) {
                    int val = match[state];
                    if (val > sam.st[p].len) val = sam.st[p].len;
                    if (val > match[p]) match[p] = val;
                }
            }

            // Update the running minimum across strings
            for (int i = 0; i < sz; ++i) best[i] = min(best[i], match[i]);
        }

        // Find the maximum length and corresponding start position in base
        int maxLen = 0;
        int bestStart = 0;
        for (int i = 0; i < sz; ++i) {
            if (best[i] > maxLen) {
                maxLen = best[i];
                bestStart = sam.st[i].first_pos - maxLen + 1;
            } else if (best[i] == maxLen && maxLen > 0) {
                int start = sam.st[i].first_pos - maxLen + 1;
                if (start < bestStart) bestStart = start;
            }
        }

        if (maxLen <= 0) {
            cout << "\n";
        } else {
            if (bestStart < 0) bestStart = 0;
            cout << base.substr(bestStart, maxLen) << "\n";
        }
        return 0;
    } else {
        // Hash-based approach for very large base
        RollingHash rh;
        rh.init(base);
        int baseLen = (int)base.size();

        int lo = 0, hi = baseLen;
        string bestSub = "";

        while (lo <= hi) {
            int mid = (lo + hi) / 2;
            if (mid == 0) { lo = 1; continue; } // skip length 0

            // Build mapping from base substring hash -> start index
            unordered_map<HashPair,int,HashPairHasher> pos;
            pos.reserve(baseLen - mid + 1);

            for (int i = 0; i + mid <= baseLen; ++i) {
                auto h = rh.get(i, i + mid);
                HashPair hp{h.first, h.second};
                if (pos.find(hp) == pos.end()) pos[hp] = i;
            }

            // Candidate set: all hashes present in base
            unordered_set<HashPair,HashPairHasher> cand;
            cand.reserve(pos.size() * 2);
            for (const auto &kv : pos) cand.insert(HashPair{kv.first.a, kv.first.b});

            bool ok = true;
            // Intersect with hashes from every other string
            for (const string &t : others) {
                unordered_set<HashPair,HashPairHasher> setT;
                setT.reserve((int)t.size() - mid + 1);
                for (int j = 0; j + mid <= (int)t.size(); ++j) {
                    auto hh = rh.get(j, j + mid);
                    setT.insert(HashPair{hh.first, hh.second});
                }

                // In-place intersection: keep only hashes present in setT
                for (auto it = cand.begin(); it != cand.end();) {
                    if (setT.find(*it) == setT.end()) it = cand.erase(it);
                    else ++it;
                }

                if (cand.empty()) { ok = false; break; }
            }

            if (ok && !cand.empty()) {
                // Take any candidate hash and map back to base position
                auto hv = *cand.begin();
                int start = pos[HashPair{hv.a, hv.b}];
                bestSub = base.substr(start, mid);
                lo = mid + 1; // try for longer
            } else {
                hi = mid - 1; // shorter substrings
            }
        }

        cout << bestSub << "\n";
        return 0;
    }
}
```
