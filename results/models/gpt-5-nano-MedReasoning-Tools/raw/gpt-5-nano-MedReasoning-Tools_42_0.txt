OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%f04 = OpConstant %float 0.04
%f0  = OpConstant %float 0.0
%f1  = OpConstant %float 1.0

%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3

; gl_PerVertex_in (input per-vertex data from vertex shader)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

; Per-vertex inputs from vertex shader (arrays; size 1 for InputPoints)
%arr_worldPos = OpTypeArray %v3float %uint_1
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos
%worldPos = OpVariable %ptr_in_arr_worldPos Input

%arr_normal = OpTypeArray %v3float %uint_1
%ptr_in_arr_normal = OpTypePointer Input %arr_normal
%normal = OpVariable %ptr_in_arr_normal Input

%arr_color = OpTypeArray %v3float %uint_1
%ptr_in_arr_color = OpTypePointer Input %arr_color
%color = OpVariable %ptr_in_arr_color Input

; Decorations for input interfaces (locations)
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; gl_PerVertex_out (output per-vertex data to next stage)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output per-vertex color (to fragment shader)
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Uniform UBO (set=0, binding=0) - minimal placeholder layout
%Mat4 = OpTypeMatrix %v4float 4
%UboBlock = OpTypeStruct %Mat4
OpDecorate %UboBlock Block
%ptr_ubo = OpTypePointer Uniform %UboBlock
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Entry point declaration
%void_ = OpTypeVoid
%func_type = OpTypeFunction %void_
%main = OpFunction %void_ None %func_type
%entry = OpLabel

; Entry point annotation (interface)
OpEntryPoint Geometry %main "main" %worldPos %normal %color %outColor

; Execution modes
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 100
OpExecutionMode %main Invocations 1

; Helper pointers for later use
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; Load gl_Position from gl_in[0].gl_Position
%gl_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%gl_pos = OpLoad %v4float %gl_pos_ptr

; Load color[0]
%color0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%color0 = OpLoad %v3float %color0_ptr

; Compute quad vertices around (x,y) with half-size h = 0.04
%gx = OpCompositeExtract %float %gl_pos 0
%gy = OpCompositeExtract %float %gl_pos 1
%xmin = OpFSub %float %gx %f04
%xmax = OpFAdd %float %gx %f04
%ymin = OpFSub %float %gy %f04
%ymax = OpFAdd %float %gy %f04

; ZW values for all emitted vertices
%z = %f0
%w = %f1

; Vertex 1: (x-h, y-h)
%v1 = OpCompositeConstruct %v4float %xmin %ymin %z %w
%v1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v1_ptr %v1
OpStore %outColor %color0
OpEmitVertex

; Vertex 2: (x+h, y-h)
%v2 = OpCompositeConstruct %v4float %xmax %ymin %z %w
%v2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v2_ptr %v2
OpStore %outColor %color0
OpEmitVertex

; Vertex 3: (x-h, y+h)
%v3 = OpCompositeConstruct %v4float %xmin %ymax %z %w
%v3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v3_ptr %v3
OpStore %outColor %color0
OpEmitVertex

; Vertex 4: (x+h, y+h)
%v4 = OpCompositeConstruct %v4float %xmax %ymax %z %w
%v4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v4_ptr %v4
OpStore %outColor %color0
OpEmitVertex

; End the primitive (quad)
OpEndPrimitive

OpReturn
OpFunctionEnd
