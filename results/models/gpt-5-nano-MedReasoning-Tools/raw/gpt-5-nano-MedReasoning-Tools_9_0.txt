Here's a complete, compilable C# solution using First Fit Decreasing (FFD) to solve the 1D Cutting Stock Problem (minimize the number of stocks used). It reads the input, packs cuts into bins, and prints the assignment per stock.

```csharp
using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    struct Item
    {
        public int length;
        public int index;
        public Item(int l, int i) { length = l; index = i; }
    }

    class Bin
    {
        public int remaining;
        public List<int> indices = new List<int>();
        public Bin(int cap)
        {
            remaining = cap;
        }
    }

    static void Main()
    {
        FastScanner fs = new FastScanner(Console.OpenStandardInput());
        int N = fs.NextInt();
        int stockLength = fs.NextInt();

        int[] cuts = new int[N];
        for (int i = 0; i < N; i++) cuts[i] = fs.NextInt();

        // Create items and sort by decreasing length (FFD)
        Item[] items = new Item[N];
        for (int i = 0; i < N; i++) items[i] = new Item(cuts[i], i);
        Array.Sort(items, (a, b) => b.length.CompareTo(a.length)); // descending

        List<Bin> bins = new List<Bin>();

        foreach (var it in items)
        {
            bool placed = false;
            foreach (var b in bins)
            {
                if (b.remaining >= it.length)
                {
                    b.indices.Add(it.index);
                    b.remaining -= it.length;
                    placed = true;
                    break;
                }
            }
            if (!placed)
            {
                Bin nb = new Bin(stockLength);
                nb.indices.Add(it.index);
                nb.remaining -= it.length;
                bins.Add(nb);
            }
        }

        // Output result
        using (var sw = new StreamWriter(Console.OpenStandardOutput()))
        {
            sw.WriteLine(bins.Count);
            foreach (var b in bins)
            {
                for (int i = 0; i < b.indices.Count; i++)
                {
                    if (i > 0) sw.Write(' ');
                    sw.Write(b.indices[i]);
                }
                sw.WriteLine();
            }
            sw.Flush();
        }
    }

    // Fast integer scanner for performance with large inputs
    class FastScanner
    {
        private readonly Stream _in;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0;
        private int _ptr = 0;

        public FastScanner(Stream s)
        {
            _in = s;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _in.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            int sign = 1;
            int val = 0;

            do { c = Read(); } while (c <= 32 && c != -1);
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
```
