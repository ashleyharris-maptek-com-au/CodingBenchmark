; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 999
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4

; Variables
ubo = OpVariable %_ptr_Uniform_UBO Uniform
inWorldPos = OpVariable %_ptr_Input_vec3 Input
inNormal = OpVariable %_ptr_Input_vec3 Input
inUV = OpVariable %_ptr_Input_vec2 Input
inTangent = OpVariable %_ptr_Input_vec3 Input
inColor = OpVariable %_ptr_Input_vec3 Input
outColor = OpVariable %_ptr_Output_vec4 Output

; Decorations
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; Constants
%one = OpConstant %float 1.0
%zero = OpConstant %float 0.0
%two = OpConstant %float 2.0
%eight = OpConstant %float 8.0
%half = OpConstant %float 0.5
%c09 = OpConstant %float 0.9
%c02 = OpConstant %float 0.2
; Color constants
%colA = OpConstantComposite %vec3 %c09 %c09 %c09
%colB = OpConstantComposite %vec3 %c02 %c02 %c02

; Function
%main = OpFunction %void None %voidFn
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %inWorldPos
%normalIn = OpLoad %vec3 %inNormal
%uvIn = OpLoad %vec2 %inUV
%tangentIn = OpLoad %vec3 %inTangent

; Load UBO fields
%modelPtr = OpAccessChain %_ptr_Uniform_mat4 %ubo 0
%viewPtr = OpAccessChain %_ptr_Uniform_mat4 %ubo 1
%projPtr = OpAccessChain %_ptr_Uniform_mat4 %ubo 2
%lightPosPtr = OpAccessChain %_ptr_Uniform_vec4 %ubo 3
%cameraPosPtr = OpAccessChain %_ptr_Uniform_vec4 %ubo 4
%paramsPtr = OpAccessChain %_ptr_Uniform_vec4 %ubo 5

%model = OpLoad %mat4 %modelPtr
%view = OpLoad %mat4 %viewPtr
%proj = OpLoad %mat4 %projPtr
%lightPos4 = OpLoad %vec4 %lightPosPtr
%cameraPos4 = OpLoad %vec4 %cameraPosPtr
%params4 = OpLoad %vec4 %paramsPtr

; Build light position vec3 from lightPos4
%lp0 = OpCompositeExtract %float %lightPos4 0
%lp1 = OpCompositeExtract %float %lightPos4 1
%lp2 = OpCompositeExtract %float %lightPos4 2
%lightPos3 = OpCompositeConstruct %vec3 %lp0 %lp1 %lp2

; Compute lightDir = normalize(lightPos3 - worldPos)
%lightDirRaw = OpFSub %vec3 %lightPos3 %worldPos
%ldLen2 = OpDot %float %lightDirRaw %lightDirRaw
%ldLen = OpSqrt %float %ldLen2
%invLd = OpFDiv %float %one %ldLen
%invLdVec = OpCompositeConstruct %vec3 %invLd %invLd %invLd
%lightDir = OpFMul %vec3 %lightDirRaw %invLdVec

; Normalize world normal
%normal = OpLoad %vec3 %inNormal
%nLen2 = OpDot %float %normal %normal
%nLen = OpSqrt %float %nLen2
%invN = OpFDiv %float %one %nLen
%invNVec = OpCompositeConstruct %vec3 %invN %invN %invN
%n = OpFMul %vec3 %normal %invNVec

; Diffuse and Lambert factor
%dotNL = OpDot %float %n %lightDir
%zeroF = OpConstant %float 0.0
%diffuse = OpFMax %float %dotNL %zeroF
%lamHalf = OpConstant %float 0.5
%diffTerm = OpFMul %float %lamHalf %diffuse
%lambert = OpFAdd %float %lamHalf %diffTerm

; UV checkerboard 8x8
%uv = OpLoad %vec2 %inUV
%uvx = OpCompositeExtract %float %uv 0
%uvy = OpCompositeExtract %float %uv 1
%u8 = OpFMul %float %uvx %eight
%v8 = OpFMul %float %uvy %eight
%fx = OpFloor %float %u8
%fy = OpFloor %float %v8
%sum = OpFAdd %float %fx %fy
%parRem = OpFRem %float %sum %two
%par = OpFloor %float %parRem

%oneMinusP = OpFSub %float %one %par
%parVec = OpCompositeConstruct %vec3 %par %par %par
%oneMinusPVec = OpCompositeConstruct %vec3 %oneMinusP %oneMinusP %oneMinusP
%colorApart = OpFMul %vec3 %colA %oneMinusPVec
%colorBpart = OpFMul %vec3 %colB %parVec
%checkerColor = OpFAdd %vec3 %colorApart %colorBpart

; Apply Lambert shading
%lambertVec = OpCompositeConstruct %vec3 %lambert %lambert %lambert
%shadedRGB = OpFMul %vec3 %checkerColor %lambertVec

; Output alpha
%alpha = OpConstant %float 1.0
%out = OpCompositeConstruct %vec4 %shadedRGB %alpha
OpStore %outColor %out

OpReturn
OpFunctionEnd

; Entry point and execution mode
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Function prototype (empty to satisfy SPIR-V syntax)
; (Not strictly necessary in many assemblers, but kept for clarity)
%voidFn = OpFunction %void None %voidFnType
%voidFnEnd = OpLabel
OpReturn
OpFunctionEnd
