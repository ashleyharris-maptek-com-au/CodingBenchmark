struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize world-space normal
    float3 N = normalize(input.normal);

    // Light position from UBO (world space)
    float3 lightWorldPos = lightPos.xyz;
    // Light direction from fragment to light
    float3 L = normalize(lightWorldPos - input.worldPos);

    // Lambertian diffuse term
    float d = max(dot(N, L), 0.0);

    // Toon quantization bands: thresholds 0.75/0.5/0.25 with values 1.0/0.7/0.4/0.2
    float band;
    if (d > 0.75)
        band = 1.0;
    else if (d > 0.5)
        band = 0.7;
    else if (d > 0.25)
        band = 0.4;
    else
        band = 0.2;

    // Apply band to vertex color
    float3 finalColor = input.color * band;

    // Output with alpha = 1.0
    return float4(finalColor, 1.0);
}
