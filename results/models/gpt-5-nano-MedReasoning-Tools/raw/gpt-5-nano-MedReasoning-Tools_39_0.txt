Here is a complete, self-contained Rust solution that finds an exact minimum-bisection for small graphs by enumerating all subsets of size n/2 (with a symmetry break to cut the search roughly in half). It uses 128-bit masks to support reasonably large n (up to 128). It prints the minimum cut and the A-side of the partition (vertices in ascending order).

Note: this approach is exponential in n, so it’s practical for modest n (e.g., n ≤ 20–25 in typical online judge tests). If your input can be much larger, a heuristic or approximate method would be needed.

```rust
use std::io::{self, Read};

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // n must be even
    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    // Read edges
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        edges.push((u, v));
    }

    // Trivial case: n == 0
    if n == 0 {
        println!("0\n");
        return;
    }

    let k = n / 2; // size of A

    // Best result
    let mut best_cut: usize = usize::MAX;
    let mut best_mask: u128 = 0;

    // If k == 0 (n == 0) handled above; otherwise we implement a symmetry break:
    // Only consider A-subsets that include vertex 0.
    if k == 0 {
        // Shouldn't happen for n > 0 since n is even; keep for completeness
        println!("0\n");
        return;
    } else if k - 1 == 0 {
        // n == 2: only one partition with vertex 0 in A
        let mask: u128 = 1u128; // bit 0 set
        let cut = compute_cut(mask, &edges);
        best_cut = cut;
        best_mask = mask;
    } else {
        // Enumerate all k-1 subsets of the remaining n-1 vertices (indices 1..n-1),
        // and always include vertex 0 in A.
        let m_bits = n - 1; // number of bits to choose from (vertices 1..n-1)
        let mut t: u128 = (1u128 << (k - 1)) - 1; // initial combination of size k-1
        let limit_t: u128 = 1u128 << m_bits;

        while t < limit_t {
            // Build mask with bit 0 set, and bits corresponding to t placed at positions 1..n-1
            let mut mask: u128 = 1u128; // include vertex 0
            for i in 0..m_bits {
                if ((t >> i) & 1) != 0 {
                    mask |= 1u128 << (i + 1);
                }
            }

            // Evaluate cut
            let cut = compute_cut(mask, &edges);
            if cut < best_cut {
                best_cut = cut;
                best_mask = mask;
            }

            // Gosper's hack to next combination of size k-1 within m_bits
            let c = t & (!t + 1); // least significant 1 bit
            let r = t + c;
            if r == 0 { break; } // overflow safety
            t = (((r ^ t) >> 2) / c) | r;
        }
    }

    // Prepare output: first line is min cut, second line is vertices in A (ascending)
    let mut a_vertices: Vec<usize> = Vec::with_capacity(k);
    for i in 0..n {
        if ((best_mask >> i) & 1) != 0 {
            a_vertices.push(i);
        }
    }

    // Print result
    println!("{}", best_cut);
    for (idx, v) in a_vertices.iter().enumerate() {
        if idx > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}

// Helper: compute the cut size for a given A-set mask
fn compute_cut(mask: u128, edges: &Vec<(usize, usize)>) -> usize {
    let mut cut: usize = 0;
    for &(u, v) in edges.iter() {
        let in_u = ((mask >> u) & 1) != 0;
        let in_v = ((mask >> v) & 1) != 0;
        if in_u != in_v {
            cut += 1;
        }
    }
    cut
}
```
