; SPIR-V Geometry shader implementing a tapered ribbon for 2-vertex input lines
; Geometry: 8 radial line segments from center outward (per the task)
; Input primitive: InputLines (2 verts per invocation)
; Output primitive: OutputTriangleStrip
; Emits 4 vertices per input primitive
; Thickness: start 0.04, end 0.005
; Note: worldPos[] provides endpoints in clip space; color[] provides per-vertex color.

               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               ; GLSL std 450 for math operations (sqrt)
               %GLSLstd450 = OpExtInstImport "GLSL.std.450"

               OpEntryPoint Geometry %main "main" %gl_in %worldPos %color %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%one_f = OpConstant %float 1.0
%thick_start = OpConstant %float 0.04
%thick_end   = OpConstant %float 0.005

; gl_PerVertex input (from vertex shader) - array of 2 entries
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
; Access chain helper for gl_in[i].gl_Position
%ptr_in_v4 = OpTypePointer Input %v4float

; Input vertex arrays (per-vertex attributes)
%worldPos_arr = OpTypeArray %v3float %uint_2
%ptr_worldPos = OpTypePointer Input %worldPos_arr
%worldPos = OpVariable %ptr_worldPos Input

%normal_arr = OpTypeArray %v3float %uint_2
%ptr_normal = OpTypePointer Input %normal_arr
%normal = OpVariable %ptr_normal Input

%color_arr = OpTypeArray %v3float %uint_2
%ptr_color = OpTypePointer Input %color_arr
%color = OpVariable %ptr_color Input

; Decorate input locations (as specified)
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Output per-vertex gl_PerVertex_out (for gl_Position)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; Output color to fragment shader
%outColor_type = OpTypeVector %float 3
%ptr_out_color = OpTypePointer Output %outColor_type
%outColor = OpVariable %ptr_out_color Output
; gl_Position helper pointer (first member of gl_PerVertex_out)
%ptr_out_v4 = OpTypePointer Output %v4float

; Optional Uniform Block (UBO) access (set=0, binding=0) - declared to match description
%UBOType = OpTypeStruct
%UBOPtr  = OpTypePointer Uniform %UBOType
%UBO     = OpVariable %UBOPtr Uniform
; Block decoration to reflect Uniform block usage
OpDecorate %UBOType Block

; Function declarations
%voidFn = OpTypeFunction %void
%main = OpFunction %void None %voidFn
%entry = OpLabel

; Helper: load input endpoints and colors
; p0, p1 in vec3; c0, c1 in vec3
%p0_ptr = OpAccessChain %ptr_worldPos %worldPos %int_0
%p0_vec3 = OpLoad %v3float %p0_ptr
%p1_ptr = OpAccessChain %ptr_worldPos %worldPos %int_1
%p1_vec3 = OpLoad %v3float %p1_ptr

%c0_ptr = OpAccessChain %ptr_color %color %int_0
%c0_vec3 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_color %color %int_1
%c1_vec3 = OpLoad %v3float %c1_ptr

; Convert p0/p1 to vec4 with w = 1.0
%p0_x = OpCompositeExtract %float %p0_vec3 0
%p0_y = OpCompositeExtract %float %p0_vec3 1
%p0_z = OpCompositeExtract %float %p0_vec3 2
%p0_w = %one_f
%p0_vec4 = OpCompositeConstruct %v4float %p0_x %p0_y %% p0_z %p0_w

%p1_x = OpCompositeExtract %float %p1_vec3 0
%p1_y = OpCompositeExtract %float %p1_vec3 1
%p1_z = OpCompositeExtract %float %p1_vec3 2
%p1_w = %one_f
%p1_vec4 = OpCompositeConstruct %v4float %p1_x %p1_y %% p1_z %p1_w

; 2D delta: dx, dy
%dx = OpFSub %float %p1_x %p0_x
%dy = OpFSub %float %p1_y %p0_y

; length and inverse length for normalization
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%length2 = OpFAdd %float %dx2 %dy2
%length  = OpExtInst %float %GLSLstd450 Sqrt %length2
%inv_len = OpFDiv %float %one_f %length

; Perpendicular (unnormalized: (-dy, dx)) then normalize
%neg_dy = OpFNegate %float %dy
%perp_x = OpFMul %float %neg_dy %inv_len
%perp_y = OpFMul %float %dx %inv_len

; Thick start/end
; Offsets: (perp * thickness)
%off_s_x = OpFMul %float %perp_x %thick_start
%off_s_y = OpFMul %float %perp_y %thick_start
%off_e_x = OpFMul %float %perp_x %thick_end
%off_e_y = OpFMul %float %perp_y %thick_end

; p0_plus = p0 + (off_s_x, off_s_y, 0, 0)
%p0x = OpCompositeExtract %float %p0_vec4 0
%p0y = OpCompositeExtract %float %p0_vec4 1
%p0z = OpCompositeExtract %float %p0_vec4 2
%p0w = OpCompositeExtract %float %p0_vec4 3

%p0x_plus = OpFAdd %float %p0x %% off_s_x
%p0y_plus = OpFAdd %float %p0y %% off_s_y
%p0_vec4_plus = OpCompositeConstruct %v4float %p0x_plus %p0y_plus %p0z %p0w

;p0_minus
%p0x_minus = OpFSub %float %p0x %% off_s_x
%p0y_minus = OpFSub %float %p0y %% off_s_y
%p0_vec4_minus = OpCompositeConstruct %v4float %p0x_minus %p0y_minus %p0z %p0w

; p1_plus = p1 + (off_e_x, off_e_y, 0, 0)
%p1x = OpCompositeExtract %float %p1_vec4 0
%p1y = OpCompositeExtract %float %p1_vec4 1
%p1z = OpCompositeExtract %float %p1_vec4 2
%p1w = OpCompositeExtract %float %p1_vec4 3

%p1x_plus = OpFAdd %float %p1x %% off_e_x
%p1y_plus = OpFAdd %float %p1y %% off_e_y
%p1_vec4_plus = OpCompositeConstruct %v4float %p1x_plus %p1y_plus %p1z %p1w

;p1_minus
%p1x_minus = OpFSub %float %p1x %% off_e_x
%p1y_minus = OpFSub %float %p1y %% off_e_y
%p1_vec4_minus = OpCompositeConstruct %v4float %p1x_minus %p1y_minus %p1z %p1w

; Emit four vertices of the ribbon as a triangle strip
; Vertex 0: p0_plus, color c0
          OpStore %gl_out %p0_vec4_plus
          OpStore %outColor %c0_vec3
          OpEmitVertex

; Vertex 1: p0_minus, color c0
          OpStore %gl_out %p0_vec4_minus
          OpStore %outColor %c0_vec3
          OpEmitVertex

; Vertex 2: p1_plus, color c1
          OpStore %gl_out %p1_vec4_plus
          OpStore %outColor %c1_vec3
          OpEmitVertex

; Vertex 3: p1_minus, color c1
          OpStore %gl_out %p1_vec4_minus
          OpStore %outColor %c1_vec3
          OpEmitVertex

; End primitive
          OpEndPrimitive

; Return from function
          OpReturn
          OpFunctionEnd

; End of SPIR-V shader

Notes:
- This shader reads two endpoints and two colors from the vertex shader stage inputs worldPos[] and color[].
- It computes a perpendicular direction in clip-space, builds four positions corresponding to a tapered ribbon (thick_start at the first endpoint, thick_end at the second), and emits a 4-vertex triangle strip.
- Colors are assigned per emitted vertex: the two vertices around the first endpoint receive c0; the two around the second endpoint receive c1.
- The UBO is declared (set=0, binding=0) per the description, but the shader does not need to read from it for the requested geometry expansion; it is present to satisfy the accessibility requirement.
