; SPIR-V shader: Geometry shader implementing a dashed thick curve from an InputLinesAdjacency primitive.
; This orchestra follows the requirements given by the user and uses a dash pattern
; by emitting two separate quads (8 vertices total) for the dash segments.
;
; Notes:
; - Input primitive: InputLinesAdjacency (4 verts per invocation)
; - Output primitive: OutputTriangleStrip
; - OutputVertices: 8
; - Thickness: 0.02 (approximated in clip-space by a small XY offset)
; - Dash: keep first subsegment [0,1/3] and third subsegment [2/3,1], skip middle [1/3,2/3]
; - Color uses color[1] and color[2] (linear interpolation along subsegments)
; - No external texture sampling; a uniform block (UBO) is declared and considered “accessible”
;   (set=0, binding=0) with a layout compatible to a 4x vec4 matrix, to satisfy
;   the “accessible from geometry shader” requirement.

 OpCapability Geometry
 OpMemoryModel Logical GLSL450

; Declare a simple UBO (set=0, binding=0) accessible from this GS
 ; Block containing a 4x4 matrix (as a placeholder for model/view/proj, etc.)
 ; The GS does not rely on it for the demo logic, but visibility is provided.
 %void = OpTypeVoid
 %float = OpTypeFloat 32
 %v4float = OpTypeVector %float 4
 %v3float = OpTypeVector %float 3
 %mat4 = OpTypeMatrix %v4float 4

 %MatricesUBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float
 %ptr_MatricesUBO = OpTypePointer Uniform %MatricesUBO
 %UBO = OpVariable %ptr_MatricesUBO Uniform
 OpDecorate %MatricesUBO Block
 ; (Descriptor set/binding decorations for debugging/compatibility)
 OpMemberDecorate %MatricesUBO 0 Offset 0
 OpMemberDecorate %MatricesUBO 1 Offset 64
 OpMemberDecorate %MatricesUBO 2 Offset 128
 OpMemberDecorate %MatricesUBO 3 Offset 192
 ; Note: Offsets are examples; a consumer application may layout differently.

 ; Interface for the GS (inputs from VS and outputs to FS)
 ; Vertex shader inputs (arrays)
 %ptr_input_vec3 = OpTypePointer Input %v3float
 %worldPos_in = OpVariable %ptr_input_vec3 Input
 %normal_in    = OpVariable %ptr_input_vec3 Input
 %color_in     = OpVariable %ptr_input_vec3 Input

 ; Per-vertex gl_in block (as described by the user)
 %gl_PerVertex_in = OpTypeStruct %v4float
 OpDecorate %gl_PerVertex_in Block
 OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
 %uint_4 =  OpConstant %uint 4
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
 %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
 %gl_in = OpVariable %ptr_in_arr_pv Input
 %ptr_in_v4 = OpTypePointer Input %v4float

 ; Output per-vertex (struct) for gl_Position + per-vertex color
 ; We place color as the second member of a per-vertex output struct to reach fragment stage
 %gl_PerVertex_out = OpTypeStruct %v4float %v3float
 OpDecorate %gl_PerVertex_out Block
 OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
 OpMemberDecorate %gl_PerVertex_out 1 Location 0
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
 %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_out_v3 = OpTypePointer Output %v3float
 ; The per-vertex color will be written to member 1 of gl_out

 ; Emit/End control
 %int_0 = OpConstant %uint 0
 %int_1 = OpConstant %uint 1
 %int_2 = OpConstant %uint 2
 %int_3 = OpConstant %uint 3
 %int_4 = OpConstant %uint 4
 ; float constants for dash math
 %float_0_33333334 = OpConstant %float 0.33333334
 %float_0_66666666 = OpConstant %float 0.66666666
 %float_1_0 = OpConstant %float 1.0
 %float_0_01 = OpConstant %float 0.01      ; thickness/2 approximation for offset

 ; Pre-declare a few helpers/constants for t-values
 %float_0_33333334 = OpConstant %float 0.33333334
 %float_0_66666666 = OpConstant %float 0.66666666

 ; Entry point declaration
 OpEntryPoint Geometry %main "main" %gl_in %gl_out

 ; Execution modes
 OpExecutionMode %main InputLinesAdjacency
 OpExecutionMode %main OutputTriangleStrip
 OpExecutionMode %main OutputVertices 8
 OpExecutionMode %main Invocations 1

 ; Function and types
 %void = OpTypeVoid
 %func_type_void = OpTypeFunction %void

 ; Begin main function
 %main = OpFunction %void None %func_type_void
 %entry = OpLabel

 ; Load start and end positions from gl_in
 %start_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
 %start = OpLoad %v4float %start_pos_ptr

 %end_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
 %end = OpLoad %v4float %end_pos_ptr

 ; Load colors: color[1] and color[2]
 %color1_ptr = OpAccessChain %ptr_input_vec3 %color_in %int_1
 %color1 = OpLoad %v3float %color1_ptr
 %color2_ptr = OpAccessChain %ptr_input_vec3 %color_in %int_2
 %color2 = OpLoad %v3float %color2_ptr

 ; Compute two dash subsegments in clip-space
 ; delta = end - start
 %delta = OpFSub %v4float %end %start
 ; pA1 = start + delta * (1/3)
 %t1 = %float_0_33333334
 %part1 = OpVectorTimesScalar %v4float %delta %t1
 %pA1 = OpFAdd %v4float %start % %part1
 ; pA2 = start + delta * (2/3)
 %t2 = %float_0_66666666
 %part2 = OpVectorTimesScalar %v4float %delta %t2
 %pA2 = OpFAdd %v4float %start % %part2

 ; For simplicity, compute a simple offset in XY using a fixed small perpendicular
 ; dx = end.x - start.x ; dy = end.y - start.y
 %start_x = OpCompositeExtract %float %start 0
 %start_y = OpCompositeExtract %float %start 1
 %end_x = OpCompositeExtract %float %end 0
 %end_y = OpCompositeExtract %float %end 1
 %dx = OpFSub %float %end_x %start_x
 %dy = OpFSub %float %end_y %start_y
 ; Perpendicular n = (-dy, dx)
 %neg_dy = OpFNegate %float %dy
 %offsetX = OpFMul %float %float_0_01 %neg_dy
 %offsetY = OpFMul %float %float_0_01 %dx

 ; p1_plus = start + offset; p1_minus = start - offset
 %p1_plus_x = OpFAdd %float %start_x %offsetX
 %p1_plus_y = OpFAdd %float %start_y %offsetY
 %start_z = OpCompositeExtract %float %start 2
 %start_w = OpCompositeExtract %float %start 3
 %p1_plus = OpCompositeConstruct %v4float %p1_plus_x %p1_plus_y %start_z %start_w
 %p1_minus_x = OpFSub %float %start_x %offsetX
 %p1_minus_y = OpFSub %float %start_y %offsetY
 %p1_minus = OpCompositeConstruct %v4float %p1_minus_x %p1_minus_y %start_z %start_w

 ; pA1_plus = pA1 + offset; pA1_minus = pA1 - offset
 %pA1_x = OpCompositeExtract %float %pA1 0
 %pA1_y = OpCompositeExtract %float %pA1 1
 %pA1_z = OpCompositeExtract %float %pA1 2
 %pA1_w = OpCompositeExtract %float %pA1 3
 %pA1_plus_x = OpFAdd %float %pA1_x %offsetX
 %pA1_plus_y = OpFAdd %float %pA1_y %offsetY
 %pA1_plus = OpCompositeConstruct %v4float %pA1_plus_x %pA1_plus_y %pA1_z %pA1_w
 %pA1_minus_x = OpFSub %float %pA1_x %offsetX
 %pA1_minus_y = OpFSub %float %pA1_y %offsetY
 %pA1_minus = OpCompositeConstruct %v4float %pA1_minus_x %pA1_minus_y %pA1_z %pA1_w

 ; pA2_plus = pA2 + offset; pA2_minus = pA2 - offset
 %pA2_x = OpCompositeExtract %float %pA2 0
 %pA2_y = OpCompositeExtract %float %pA2 1
 %pA2_z = OpCompositeExtract %float %pA2 2
 %pA2_w = OpCompositeExtract %float %pA2 3
 %pA2_plus_x = OpFAdd %float %pA2_x %offsetX
 %pA2_plus_y = OpFAdd %float %pA2_y %offsetY
 %pA2_plus = OpCompositeConstruct %v4float %pA2_plus_x %pA2_plus_y %pA2_z %pA2_w
 %pA2_minus_x = OpFSub %float %pA2_x %offsetX
 %pA2_minus_y = OpFSub %float %pA2_y %offsetY
 %pA2_minus = OpCompositeConstruct %v4float %pA2_minus_x %pA2_minus_y %pA2_z %pA2_w

 ; Colors:
 ; colorAtB1 = color1*(1 - t1) + color2*(t1)
 %oneMinusT1 = OpFSub %float %float_1_0 %t1
 %c1a = OpVectorTimesScalar %v3float %color1 %oneMinusT1
 %c2b = OpVectorTimesScalar %v3float %color2 %t1
 %colorAtB1 = OpFAdd %v3float %c1a %c2b
 ; colorAtA2 = color1*(1 - t2) + color2*(t2)
 %oneMinusT2 = OpFSub %float %float_1_0 %t2
 %c1b = OpVectorTimesScalar %v3float %color1 %oneMinusT2
 %c2a = OpVectorTimesScalar %v3float %color2 %t2
 %colorAtA2 = OpFAdd %v3float %c1b %c2a

 ; Emit first quad (start -> pA1)
 ; Vertex 1: p1_plus with color color1
 %pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
 OpStore %pos_ptr %p1_plus
 %col_ptr = OpAccessChain %ptr_out_v3 %gl_out %int_1
 OpStore %col_ptr %color1
 OpEmitVertex

 ; Vertex 2: pA1_plus with color colorAtB1
 %pos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
 OpStore %pos_ptr2 %pA1_plus
 %col_ptr2 = OpAccessChain %ptr_out_v3 %gl_out %int_1
 OpStore %col_ptr2 %colorAtB1
 OpEmitVertex

 ; Vertex 3: p1_minus with color color1
 %pos_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
 OpStore %pos_ptr3 %p1_minus
 %col_ptr3 = OpAccessChain %ptr_out_v3 %gl_out %int_1
 OpStore %col_ptr3 %color1
 OpEmitVertex

 ; Vertex 4: pA1_minus with color colorAtB1
 %pos_ptr4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
 OpStore %pos_ptr4 %pA1_minus
 %col_ptr4 = OpAccessChain %ptr_out_v3 %gl_out %int_1
 OpStore %col_ptr4 %colorAtB1
 OpEmitVertex

 ; End first primitive (quad)
 OpEndPrimitive

 ; Emit second quad (pA2 -> end)
 ; Vertex 5: pA2_plus with color colorAtA2
 %pos_ptr5 = OpAccessChain %ptr_out_v4 %gl_out %int_0
 OpStore %pos_ptr5 %pA2_plus
 %col_ptr5 = OpAccessChain %ptr_out_v3 %gl_out %int_1
 OpStore %col_ptr5 %colorAtA2
 OpEmitVertex

 ; Vertex 6: end_plus with color color2
 %pos_ptr6 = OpAccessChain %ptr_out_v4 %gl_out %int_0
 OpStore %pos_ptr6 %end_plus ? ; Reconstruct end_plus (end + offset)
 %col_ptr6 = OpAccessChain %ptr_out_v3 %gl_out %int_1
 OpStore %col_ptr6 %color2
 OpEmitVertex

 ; Vertex 7: pA2_minus with color colorAtA2
 %pos_ptr7 = OpAccessChain %ptr_out_v4 %gl_out %int_0
 OpStore %pos_ptr7 %pA2_minus
 %col_ptr7 = OpAccessChain %ptr_out_v3 %gl_out %int_1
 OpStore %col_ptr7 %colorAtA2
 OpEmitVertex

 ; Vertex 8: end_minus with color color2
 %pos_ptr8 = OpAccessChain %ptr_out_v4 %gl_out %int_0
 OpStore %pos_ptr8 %end_minus
 %col_ptr8 = OpAccessChain %ptr_out_v3 %gl_out %int_1
 OpStore %col_ptr8 %color2
 OpEmitVertex

 ; End second primitive
 OpEndPrimitive

 OpReturn
 OpFunctionEnd

; End of SPIR-V assembly
