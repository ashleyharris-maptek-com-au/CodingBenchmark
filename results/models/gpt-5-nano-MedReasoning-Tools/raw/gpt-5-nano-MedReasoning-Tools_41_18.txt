; SPIR-V

; Bound: 9999

OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%_bool = OpTypeBool
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%_struct_UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform__struct_UBO = OpTypePointer Uniform %_struct_UBO
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4

%uint = OpTypeInt 32 0
%idx0 = OpConstant %uint 0
%idx1 = OpConstant %uint 1
%idx2 = OpConstant %uint 2
%idx3 = OpConstant %uint 3
%idx4 = OpConstant %uint 4

; GLSL std450 import for normalize/dot/max/etc
%GLSL = OpExtInstImport "GLSL.std.450"

; Constants
%_float_0 = OpConstant %float 0.0
%_float_1 = OpConstant %float 1.0
%_float_0_5 = OpConstant %float 0.5
%_float_2 = OpConstant %float 2.0

; Inputs
%worldPos = OpVariable %_ptr_Input_vec3 Input
%normalIn = OpVariable %_ptr_Input_vec3 Input
%uv = OpVariable %_ptr_Input_vec2 Input
%tangentIn = OpVariable %_ptr_Input_vec3 Input
%colorIn = OpVariable %_ptr_Input_vec3 Input

; Output
$outColor = OpVariable %_ptr_Output_vec4 Output

; Uniform
%ubo = OpVariable %_ptr_Uniform__struct_UBO Uniform

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normalIn Location 1
OpDecorate %uv Location 2
OpDecorate %tangentIn Location 3
OpDecorate %colorIn Location 4
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %_struct_UBO Block
OpMemberDecorate %_struct_UBO 0 Offset 0
OpMemberDecorate %_struct_UBO 1 Offset 64
OpMemberDecorate %_struct_UBO 2 Offset 128
OpMemberDecorate %_struct_UBO 3 Offset 192
OpMemberDecorate %_struct_UBO 4 Offset 208
OpMemberDecorate %_struct_UBO 5 Offset 224

; Entry point
OpEntryPoint Fragment %main "main" %worldPos %normalIn %uv %tangentIn %colorIn %outColor
OpExecutionMode %main OriginUpperLeft

; Function declarations
%main = OpFunction %void None %_func
%_func = OpTypeFunction %void
%entry = OpLabel

; Load inputs
%worldPosVal = OpLoad %vec3 %worldPos
%normalVal = OpLoad %vec3 %normalIn

; Normalize normal
%N_norm = OpExtInst %vec3 %GLSL Normalize %normalVal

; Load camera position from UBO (uniform block)
%cameraPosPtr = OpAccessChain %_ptr_Uniform_vec4 %ubo %idx4
%cameraPos4 = OpLoad %vec4 %cameraPosPtr
%camX = OpCompositeExtract %float %cameraPos4 0
%camY = OpCompositeExtract %float %cameraPos4 1
%camZ = OpCompositeExtract %float %cameraPos4 2
%camPosXYZ = OpCompositeConstruct %vec3 %camX %camY %camZ

; View vector V = normalize(cameraPos - worldPos)
%Vtmp = OpFSub %vec3 %camPosXYZ %worldPosVal
%V = OpExtInst %vec3 %GLSL Normalize %Vtmp

; N dot V
%dotNV = OpDot %float %N_norm %V

; NdotV = max(dot(N,V), 0.0)
%NdotV = OpExtInst %float %GLSL Max %dotNV %_float_0

; Define branching for heatmap
OpSelectionMerge %labelMerge None
OpBranchConditional %_cond %labelThen %labelElse

; Boolean condition: NdotV < 0.5
%_cond = OpFOrdLessThan %_bool %NdotV %_float_0_5

; Then block (NdotV < 0.5): colorA = (1, 2*NdotV, 0, 1)
%labelThen = OpLabel
%tmpTwoNdotV = OpFMul %float %NdotV %_float_2
%colorA = OpCompositeConstruct %vec4 %_float_1 %tmpTwoNdotV %_float_0 %_float_1
OpBranch %labelMerge

; Else block (NdotV >= 0.5): colorB = (1 - t, 1 - t, t, 1), where t = (NdotV - 0.5) * 2
%labelElse = OpLabel
%diff = OpFSub %float %NdotV %_float_0_5
%t = OpFMul %float %diff %_float_2
%rg = OpFSub %float %_float_1 % %t
%colorB = OpCompositeConstruct %vec4 %rg %rg %% t %_float_1
OpBranch %labelMerge

; Merge block
%labelMerge = OpLabel
%finalColor = OpPhi %vec4 %colorA %labelThen %colorB %labelElse
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
