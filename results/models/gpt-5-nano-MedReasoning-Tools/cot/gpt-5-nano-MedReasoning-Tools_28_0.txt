```rust
use std::io::{self, Write};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();
    let mut in_cover = vec![0u8; n];

    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        if in_cover[u] == 0 && in_cover[v] == 0 {
            in_cover[u] = 1;
            in_cover[v] = 1;
        }
    }

    // Build output
    let selected: Vec<usize> = (0..n).filter(|&i| in_cover[i] == 1).collect();
    let mut out = String::new();
    out.push_str(&format!("{}", selected.len()));
    out.push('\n');

    if !selected.is_empty() {
        out.push_str(&selected.iter().map(|val| val.to_string()).collect::<Vec<_>>().join(" "));
    }
    
    out.push('\n');
    let mut stdout = io::BufWriter::new(io::stdout());
    stdout.write_all(out.as_bytes()).unwrap();
}
```
