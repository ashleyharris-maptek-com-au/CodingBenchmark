// Beam search for a single drone
function planDronePath(startX: number, startY: number, maxSteps: number, remaining: number[][]): [number, number][] {
  const BEAM_WIDTH = 300;
  
  // Beam state: x, y, cumulative score
  interface State {
    x: number;
    y: number;
    score: number;
    parentIdx: number; // index in previous step's beam
  }
  
  // For path reconstruction
  const beamHistory: State[][] = [];
  
  let beam: State[] = [{x: startX, y: startY, score: 0, parentIdx: -1}];
  
  // Collect starting cell value
  const startVal = remaining[startY][startX];
  beam[0].score = startVal;
  
  beamHistory.push([...beam]);
  
  for (let step = 1; step <= maxSteps; step++) {
    const candidates: State[] = [];
    
    for (const state of beam) {
      for (const [dx, dy] of [[0,1],[0,-1],[1,0],[-1,0],[0,0]]) {
        const nx = state.x + dx;
        const ny = state.y + dy;
        
        if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && probability[ny][nx] >= 0) {
          // Value of new cell (approximate - we don't track per-state visited)
          const cellVal = remaining[ny][nx];
          // Penalize staying if no value
          candidates.push({
            x: nx, y: ny,
            score: state.score + cellVal,
            parentIdx: beam.indexOf(state) // This is O(B), bad!
          });
        }
      }
    }
    
    // Sort by score descending
    candidates.sort((a, b) => b.score - a.score);
    
    // Deduplicate by position (keep best score for each position)
    const uniqueStates = new Map<string, State>();
    for (const c of candidates) {
      const key = `${c.x},${c.y}`;
      if (!uniqueStates.has(key) || uniqueStates.get(key)!.score < c.score) {
        uniqueStates.set(key, c);
      }
    }
    
    beam = Array.from(uniqueStates.values()).slice(0, BEAM_WIDTH);
    beamHistory.push(beam);
  }
  
  // Reconstruct best path
  let bestIdx = 0;
  let bestScore = beam[0].score;
  for (let i = 1; i < beam.length; i++) {
    if (beam[i].score > bestScore) {
      bestScore = beam[i].score;
      bestIdx = i;
    }
  }
  
  const path: [number, number][] = [];
  for (let step = maxSteps; step >= 0; step--) {
    const state = beamHistory[step][bestIdx];
    path.push([state.x, state.y]);
    bestIdx = state.parentIdx;
  }
  
  path.reverse();
  return path;
}

    const cy = new Int32Array(maxCands);
    const cpx = new Int32Array(maxCands);
    const cpy = new Int32Array(maxCands);
    const cs = new Float64Array(maxCands);
    const cpi = new Int32Array(maxCands); // parent index in current beam
    let cCount = 0;
    
    for (let i = 0; i < bCount; i++) {
      for (let d = 0; d < 5; d++) {
        const nx = bx[i] + DX[d];
        const ny = by[i] + DY[d];
        if (!isValid(nx, ny)) continue;
        
        let val = remaining[ny][nx];
        // Penalize immediate backtrack
        if (nx === bpx[i] && ny === bpy[i]) val = 0;
        // Slight penalty for staying
        if (d === 4) val *= 0.99;
        
        cx[cCount] = nx;
        cy[cCount] = ny;
        cpx[cCount] = bx[i];
        cpy[cCount] = by[i];
        cs[cCount] = bs[i] + val;
        cpi[cCount] = i;
        cCount++;
      }
    }
    
    if (cCount === 0) {
      // No valid moves, stay
      bx[0] = bx[0]; // stay
      by[0] = by[0];
      bCount = 1;
      histX.push(Int32Array.from(bx.subarray(0, bCount)));
      histY.push(Int32Array.from(by.subarray(0, bCount)));
      histParent.push(new Int32Array(bCount).fill(0));
      continue;
    }
    
    // Sort by score descending (using indices)
    const indices = Array.from({length: cCount}, (_, i) => i);
    indices.sort((a, b) => cs[b] - cs[a]);
    
    // Select top BEAM_WIDTH unique positions
    const newX = new Int32Array(BEAM_WIDTH);
    const newY = new Int32Array(BEAM_WIDTH);
    const newPX = new Int32Array(BEAM_WIDTH);
    const newPY = new Int32Array(BEAM_WIDTH);
    const newS = new Float64Array(BEAM_WIDTH);
    const newParents = new Int32Array(BEAM_WIDTH);
    let newCount = 0;
    
    const seenPos = new Map<number, number>(); // key -> best score
    
    for (const idx of indices) {
      const k = cy[idx] * C + cx[idx];
      
      // Dedup: keep only the best score for each position
      const existing = seenPos.get(k);
      if (existing !== undefined) continue;
      seenPos.set(k, cs[idx]);
      
      newX[newCount] = cx[idx];
      newY[newCount] = cy[idx];
      newPX[newCount] = cpx[idx];
      newPY[newCount] = cpy[idx];
      newS[newCount] = cs[idx];
      newParents[newCount] = cpi[idx];
      newCount++;
      
      if (newCount >= BEAM_WIDTH) break;
    }
    
    bx = newX; by = newY; bpx = newPX; bpy = newPY; bs = newS;
    bCount = newCount;
    
    histX.push(Int32Array.from(bx.subarray(0, bCount)));
    histY.push(Int32Array.from(by.subarray(0, bCount)));
    histParent.push(Int32Array.from(newParents.subarray(0, bCount)));
  }
  
  // Find best final state
  let bestIdx = 0;
  for (let i = 1; i < bCount; i++) {
    if (bs[i] > bs[bestIdx]) bestIdx = i;
  }
  
  // Reconstruct path
  const path: [number, number][] = [];
  let idx = bestIdx;
  for (let step = TOTAL_STEPS; step >= 0; step--) {
    path.push([histX[step][idx], histY[step][idx]]);
    if (step > 0) idx = histParent[step][idx];
  }
  path.reverse();
  
  return path;
}

function updateRemaining(path: [number, number][]) {
  for (const [x, y] of path) {
    remaining[y][x] = 0;
  }
}
