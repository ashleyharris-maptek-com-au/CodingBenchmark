            else if (a[i] > b[j]) j++;
            else { result.Add(a[i]); i++; j++; }

static bool IsNeighbor(int u, int v)
{
    // Binary search in adj[u] for v
    var list = adj[u];
    int lo = 0, hi = list.Count - 1;
    while (lo <= hi)
    {
        int mid = (lo + hi) / 2;
        if (list[mid] == v) return true;
        else if (list[mid] < v) lo = mid + 1;
        else hi = mid - 1;
    }
    return false;
}
