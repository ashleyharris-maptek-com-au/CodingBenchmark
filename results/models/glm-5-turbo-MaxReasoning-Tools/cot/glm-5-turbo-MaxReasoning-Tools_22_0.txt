static bool CanPlace(bool[,] board, int[][] cells, int row, int col)
{
    for (int i = 0; i < cells.Length; i++)
    {
        int r = row + cells[i][0];
        int c = col + cells[i][1];
        if (r < 0 || r >= height || c < 0 || c >= width)
            return false;
        if (board[r, c])
            return false;
    }
    return true;
}

static int FindDropRow(bool[,] board, int[][] cells, int col)
{
    int row = 0;
    if (!CanPlace(board, cells, row, col))
        return -1; // Can't place
    while (CanPlace(board, cells, row + 1, col))
        row++;
    return row;
}

static (int linesCleared, bool[,] newBoard) PlacePiece(bool[,] board, int[][] cells, int row, int col)
{
    // Copy board
    bool[,] newBoard = new bool[height, width];
    for (int r = 0; r < height; r++)
        for (int c = 0; c < width; c++)
            newBoard[r, c] = board[r, c];
    
    // Place piece
    for (int i = 0; i < cells.Length; i++)
    {
        int r = row + cells[i][0];
        int c = col + cells[i][1];
        newBoard[r, c] = true;
    }
    
    // Clear lines
    int linesCleared = 0;
    for (int r = height - 1; r >= 0; r--)
    {
        bool full = true;
        for (int c = 0; c < width; c++)
        {
            if (!newBoard[r, c])
            {
                full = false;
                break;
            }
        }
        if (full)
        {
            linesCleared++;
            // Shift everything down
            for (int rr = r; rr > 0; rr--)
                for (int c = 0; c < width; c++)
                    newBoard[rr, c] = newBoard[rr - 1, c];
            for (int c = 0; c < width; c++)
                newBoard[0, c] = false;
            r++; // Re-check this row
        }
    }
    
    return (linesCleared, newBoard);
}
