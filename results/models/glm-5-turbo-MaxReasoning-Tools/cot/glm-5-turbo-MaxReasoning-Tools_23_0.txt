use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

struct GameState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    updated: bool,
    initialized: bool,
}

struct MapData {

    max_thrust: f64,
    initial_fuel: f64,
    start_x: f64,
    start_y: f64,

    target_y: f64,

    let stdout = io::stdout();

        .map(|s| s.parse().unwrap())
        .collect();
    let width = header[0] as usize;
    let height = header[1] as usize;

    let max_thrust = header[3];
    let initial_fuel = header[4];

        .map(|s| s.parse().unwrap())
        .collect();
    let start_x = pos[0];
    let start_y = pos[1];
    let target_x = pos[2];
    let target_y = pos[3];

    let mut landing_zone_min_x = f64::MAX;
    let mut landing_zone_max_x = f64::MIN;
    let mut landing_zone_y = 0.0;
    
    for y in 0..height {
        let line = lines.next().unwrap().unwrap();
        for (x, ch) in line.chars().enumerate() {
            if ch == 'L' {
                landing_zone_min_x = landing_zone_min_x.min(x as f64);
                landing_zone_max_x = landing_zone_max_x.max(x as f64);
                landing_zone_y = y as f64;
            }
        }
    }
    
    let map_data = MapData {
        gravity,
        max_thrust,
        initial_fuel,
        start_x,
        start_y,
        target_x,
        target_y,
        landing_zone_min_x,
        landing_zone_max_x,
        landing_zone_y,
    };

        y: start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: initial_fuel,
        updated: true,
        initialized: true,

    let reader_thread = thread::spawn(move || {

            let line = match line {
                Ok(l) => l,
                Err(_) => break,

            let vals: Vec<f64> = line.split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            if vals.len() >= 7 {
                let mut s = state_reader.lock().unwrap();
                s.x = vals[0];
                s.y = vals[1];
                s.vx = vals[2];
                s.vy = vals[3];
                s.angle = vals[4];
                s.angular_vel = vals[5];
                s.fuel = vals[6];
                s.updated = true;
            }
        }
    });

    let mut last_output = Instant::now();
    
    // PD controller gains (will be tuned based on physics)
    let net_thrust = max_thrust - gravity;
    
    // Gains for position control
    let kp = 0.3; // Position gain
    let kd = 1.5; // Velocity gain (derivative)
    
    // Angle control gains
    let kp_angle = 3.0;
    let kd_angle = 1.5;
    
    loop {
        let now = Instant::now();
        let elapsed = now - last_output;
        
        if elapsed < control_interval {
            thread::sleep(control_interval - elapsed);
        }
        last_output = Instant::now();
        
        let s = state.lock().unwrap();
        let x = s.x;
        let y = s.y;
        let vx = s.vx;
        let vy = s.vy;

        let angular_vel = s.angular_vel;
        let fuel = s.fuel;

        let ty = map_data.landing_zone_y;

        let ex = tx - x;

        let dist = (ex*ex + ey*ey).sqrt();

        let max_speed = 15.0; // m/s max approach speed

        if dist < 50.0 {

            max_speed *= scale;
            vx_des = (kp * ex).clamp(-max_speed, max_speed);

            let max_v = 2.0; // max 2 m/s for landing
            vx_des = (kp * ex * scale).clamp(-max_v, max_v);
            vy_des = (kp * ey * scale).clamp(-max_v, max_v);

        let evx = vx_des - vx;
        let evy = vy_des - vy;

        let ax_des = kd * evx;
        let ay_des = kd * evy;

        let thrust_ax = ax_des;
        let thrust_ay = ay_des - map_data.gravity;
        
        // Convert to angle and magnitude
        let thrust_mag = (thrust_ax*thrust_ax + thrust_ay*thrust_ay).sqrt();
        let desired_angle = thrust_ax.atan2(-thrust_ay);
        
        let mut thrust_frac = thrust_mag / map_data.max_thrust;
        thrust_frac = thrust_frac.clamp(0.0, 1.0);
        
        // Angle PD control
        let angle_error = desired_angle - angle;
        let turn = (kp_angle * angle_error - kd_angle * angular_vel).clamp(-1.0, 1.0);
        
        // Don't thrust if angle is too far off
        if angle_error.abs() > std::f64::consts::FRAC_PI_2 {
            thrust_frac = 0.0;
        }
        
        // Fuel conservation
        if fuel < 0.1 {
            thrust_frac = 0.0;
        }
        
        writeln!(stdout, "{:.4} {:.4}", thrust_frac, turn).unwrap();
        stdout.flush().unwrap();
    }
}
