                for (let j = i + 2; j < n - 1; j++) {
                    const oldD = dist(path[i], path[i+1]) + dist(path[j], path[j+1]);
                    const newD = dist(path[i], path[j]) + dist(path[i+1], path[j+1]);
                    if (newD < oldD) {

                        while (left < right) {
                            const tmp = path[left];
                            path[left] = path[right];
                            path[right] = tmp;

                        }

}
