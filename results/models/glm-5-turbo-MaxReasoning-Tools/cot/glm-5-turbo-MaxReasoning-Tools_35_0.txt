fn exact_solve(n: usize, adj: &[Vec<usize>]) -> Vec<usize> {
    let full = (1u64 << n) - 1;
    let mut best = n + 1;
    let mut best_mask = 0u64;
    
    // Precompute domination mask for each vertex
    let mut dom_mask: Vec<u64> = vec![0; n];
    for v in 0..n {
        let mut mask = 1u64 << v;
        for &u in &adj[v] {
            mask |= 1u64 << u;
        }
        dom_mask[v] = mask;
    }
    
    // Iterate all subsets
    for mask in 1u64..=(1u64 << n) - 1 {
        // Quick prune: if popcount already >= best, skip
        let pop = mask.count_ones() as usize;
        if pop >= best { continue; }
        
        // Compute union of domination masks
        let mut covered = 0u64;
        let mut m = mask;
        while m != 0 {
            let bit = m & m.wrapping_neg(); // lowest set bit
            let v = bit.trailing_zeros() as usize;
            covered |= dom_mask[v];
            m ^= bit;
        }
        
        if covered == full {
            best = pop;
            best_mask = mask;
        }
    }
    
    // Extract vertices
    let mut result = vec![];
    let mut m = best_mask;
    while m != 0 {
        let bit = m & m.wrapping_neg();
        let v = bit.trailing_zeros() as usize;
        result.push(v);
        m ^= bit;
    }
    
    result
}

fn remove_redundant(n: usize, adj: &[Vec<usize>], solution: &mut Vec<usize>) {
    let mut in_solution: Vec<bool> = vec![false; n];
    for &v in solution { in_solution[v] = true; }
    
    let mut changed = true;
    while changed {
        changed = false;
        let mut i = 0;
        while i < solution.len() {
            let v = solution[i];
            in_solution[v] = false;
            
            // Check if v is redundant
            let mut still_covered = true;
            // Check v itself
            if !in_solution[v] {
                // Is v covered by some other vertex in solution?
                let mut v_covered = false;
                for &u in &adj[v] {
                    if in_solution[u] {
                        v_covered = true;
                        break;
                    }
                }
                if !v_covered {
                    still_covered = false;
                }
            }
            
            if still_covered {
                // Check all neighbors of v
                for &u in &adj[v] {
                    if in_solution[u] { continue; } // u is in solution, so u covers itself
                    // Is u covered by some vertex in solution?
                    let mut u_covered = false;
                    for &w in &adj[u] {
                        if in_solution[w] {
                            u_covered = true;
                            break;
                        }
                    }
                    if !u_covered {
                        still_covered = false;
                        break;
                    }
                }
            }
            
            if still_covered {
                // Remove v
                solution.swap_remove(i);
                changed = true;
            } else {

    let mut cover_count: Vec<u32> = vec![0; n];

        in_solution[v] = true;
    }

        cover_count[v] += 1;
        for &u in &adj[v] {
            cover_count[u] += 1;
        }
    }
    
    let mut changed = true;
    while changed {
        changed = false;
        let mut i = 0;
        while i < solution.len() {
            let v = solution[i];
            
            // Decrement cover counts
            cover_count[v] -= 1;
            for &u in &adj[v] {
                cover_count[u] -= 1;
            }
            
            // Check if any vertex has cover_count == 0
            let mut redundant = true;
            if cover_count[v] == 0 {
                redundant = false;
            } else {
                for &u in &adj[v] {
                    if cover_count[u] == 0 {
                        redundant = false;
                        break;
                    }
                }
            }
            
            if redundant {
                // Permanently remove v
                in_solution[v] = false;
                solution.swap_remove(i);
                changed = true;
            } else {
                // Restore cover counts
                cover_count[v] += 1;
                for &u in &adj[v] {
                    cover_count[u] += 1;
                }
                i += 1;
            }
        }
    }
}

fn branch_and_bound(n: usize, adj: &[Vec<usize>], best: &mut Vec<usize>) {
    let full_cover: u64 = (1u64 << n) - 1;
    // Precompute domination masks
    let dom: Vec<u64> = (0..n).map(|v| {
        let mut mask = 1u64 << v;
        for &u in &adj[v] { mask |= 1u64 << u; }
        mask
    }).collect();
    
    let initial_cover = 0u64;
    let mut current: Vec<usize> = vec![];
    
    fn search(
        v: usize, // next vertex to consider
        current: &mut Vec<usize>,
        cover: u64,
        dom: &[u64],
        n: usize,
        full: u64,
        best: &mut Vec<usize>,
        remaining: &[usize; 64], // vertices not yet considered
        remaining_count: usize,
    ) {
        if cover == full {
            if current.len() < best.len() {
                *best = current.clone();
            }
            return;
        }
        
        if current.len() >= best.len() { return; }
        
        // Lower bound: number of additional vertices needed
        // At best, one vertex covers all uncovered
        // But we need at least ceil(popcount(uncovered) / max_dom) vertices
        let uncovered = !cover & full;
        let uncovered_count = uncovered.count_ones();
        
        if uncovered_count == 0 { return; } // Shouldn't happen
        
        // Find max coverage by a single remaining vertex
        let mut max_cov = 0u32;
        for i in 0..remaining_count {
            let vi = remaining[i];
            let cov = (dom[vi] & uncovered).count_ones();
            if cov > max_cov { max_cov = cov; }
        }
        
        if max_cov == 0 { return; } // No remaining vertex can cover uncovered
        
        let lower_bound = (uncovered_count + max_cov - 1) / max_cov;
        
        if current.len() + lower_bound as usize >= best.len() { return; }
        
        // Branch: either include or exclude remaining[0]
        // Actually, try including first (more likely to find solution)
        for i in 0..remaining_count {
            let vi = remaining[i];
            
            current.push(vi);
            let new_cover = cover | dom[vi];
            
            // Build new remaining list
            let mut new_remaining = [0usize; 64];
            let mut j = 0;
            for k in 0..remaining_count {
                if k != i {
                    new_remaining[j] = remaining[k];
                    j += 1;
                }
            }
            
            search(0, current, new_cover, dom, n, full, best, &new_remaining, j);
            current.pop();
            
            if current.len() + 1 >= best.len() { break; }
        }
    }
    
    let mut remaining = [0usize; 64];
    for i in 0..n {
        remaining[i] = i;
    }
    
    search(0, &mut current, initial_cover, &dom, n, full_cover, best, &remaining, n);
}
