    let n = poly.vertices.len();
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let vi = &poly.vertices[i];
        let vj = &poly.vertices[j];
        if ((vi.y > p.y) != (vj.y > p.y))
            && (p.x < (vj.x - vi.x) * (p.y - vi.y) / (vj.y - vi.y) + vi.x)

            inside = !inside;

        j = i;

    inside

fn point_on_segment(p: &Point, a: &Point, b: &Point, eps: f64) -> bool {
    let cross_val = (b.x - a.x) * (p.y - a.y) - (b.y - a.y) * (p.x - a.x);
    if cross_val.abs() > eps { return false; }
    let dot = (p.x - a.x) * (b.x - a.x) + (p.y - a.y) * (b.y - a.y);
    let len_sq = (b.x - a.x) * (b.x - a.x) + (b.y - a.y) * (b.y - a.y);
    dot >= -eps && dot <= len_sq + eps

    height: usize,

        let (min_x, min_y, max_x, max_y) = bounding_box(stock);
        let width = ((max_x - min_x) / resolution).ceil() as usize + 1;
        let height = ((max_y - min_y) / resolution).ceil() as usize + 1;
        let mut cells = vec![0u8; width * height];
