function interRouteRelocate(routes: number[][]): void {
  const m = routes.length;
  let improved = true;
  
  while (improved) {
    improved = false;
    
    for (let ri = 0; ri < m && !improved; ri++) {
      for (let pi = routes[ri].length - 1; pi >= 0 && !improved; pi--) {
        const customer = routes[ri][pi];
        const custId = customer + 1;
        
        // Cost of removing customer from route ri at position pi
        const prevIdx = pi === 0 ? 0 : routes[ri][pi - 1] + 1;
        const nextIdx = pi === routes[ri].length - 1 ? 0 : routes[ri][pi + 1] + 1;
        const removeCost = dist(prevIdx, custId) + dist(custId, nextIdx) - dist(prevIdx, nextIdx);

          if (rj === ri) continue;

            // Inserting into empty route
            const insertCost = dist(0, custId) + dist(custId, 0);
            if (insertCost < removeCost - 1e-10) {

              routes[rj].push(customer);

function savingsTour(): number[] {
  // Initially each package is its own route
  const next = new Int32Array(n);
  const prev = new Int32Array(n);
  const routeId = new Int32Array(n);
  
  for (let i = 0; i < n; i++) {
    next[i] = -1;
    prev[i] = -1;
    routeId[i] = i;
  }
  
  // Compute savings and sort
  interface Saving {
    i: number;
    j: number;
    s: number;
  }
  
  const savings: Saving[] = [];
  for (let i = 0; i < n; i++) {
    for (let j = i + 1; j < n; j++) {
      const s = dist(0, i + 1) + dist(0, j + 1) - dist(i + 1, j + 1);
      if (s > 0) {
        savings.push({ i, j, s });
      }
    }
  }
  
  savings.sort((a, b) => b.s - a.s);
  
  let routeCount = n;
  
  for (const { i, j } of savings) {
    if (routeCount <= 1) break;
    if (!timeLeft()) break;
    
    const ri = routeId[i];
    const rj = routeId[j];
    if (ri === rj) continue;
    
    const iIsTail = (next[i] === -1);
    const iIsHead = (prev[i] === -1);
    const jIsTail = (next[j] === -1);
    const jIsHead = (prev[j] === -1);
    
    // Must connect a tail to a head
    if (iIsTail && jIsHead) {
      next[i] = j;
      prev[j] = i;
      // Update routeId for j's route members
      let cur = j;
      while (cur !== -1) {
        routeId[cur] = ri;
        cur = next[cur];
      }
      routeCount--;
    } else if (jIsTail && iIsHead) {
      next[j] = i;
      prev[i] = j;
      // Update routeId for i's route members
      let cur = i;
      while (cur !== -1) {
        routeId[cur] = rj;
        cur = next[cur];

    }
  }
  
  // If multiple routes remain, connect them
  if (routeCount > 1) {
    // Find all route heads and tails
    const heads: number[] = [];
    const tails: number[] = [];
    const visited = new Uint8Array(n);
    
    for (let i = 0; i < n; i++) {
      if (visited[i]) continue;
      // Find head
      let h = i;
      while (prev[h] !== -1) {
        h = prev[h];
      }
      // Find tail
      let t = h;
      while (next[t] !== -1) {
        t = next[t];
      }
      heads.push(h);
      tails.push(t);
      // Mark visited
      let cur = h;
      while (cur !== -1) {
        visited[cur] = 1;
        cur = next[cur];
      }
    }
    
    // Connect remaining routes: tail of one to head of next
    for (let r = 0; r < routeCount - 1; r++) {
      next[tails[r]] = heads[r + 1];
      prev[heads[r + 1]] = tails[r];
    }
  }
  
  // Extract tour
  // Find head
  let head = 0;
  while (prev[head] !== -1) head = prev[head];
  
  const tour: number[] = [];
  let cur = head;
  while (cur !== -1) {
    tour.push(cur);
    cur = next[cur];
  }
  
  return tour;
}
