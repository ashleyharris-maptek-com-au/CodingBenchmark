2-opt: Given a tour (v0, v1, ..., v_{n-1}), try swapping edges (v_i, v_{i+1}) and (v_j, v_{j+1}) by reversing the segment v_{i+1}, ..., v_j. If this improves the tour length, accept the swap.

```cpp
bool twoOptImprove(vector<int>& tour, const vector<double>& x, const vector<double>& y) {
    int N = tour.size();
    bool improved = false;
    for (int i = 0; i < N - 1; i++) {
        for (int j = i + 2; j < N; j++) {
            // Skip if j == N-1 and i == 0 (that would be the closing edge)
            if (i == 0 && j == N - 1) continue;
            double d1 = dist(tour[i], tour[i+1]) + dist(tour[j], tour[(j+1)%N]);
            double d2 = dist(tour[i], tour[j]) + dist(tour[i+1], tour[(j+1)%N]);
            if (d2 < d1 - eps) {
                reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                improved = true;
            }
        }
    }
    return improved;
}
```
