def fuse_sensors(values, valid_flags, sanity_range, agreement_tol):
    """Fuse triple-redundant sensors with outlier detection."""
    # Filter by validity AND sanity range
    candidates = []
    for i, (v, valid) in enumerate(zip(values, valid_flags)):
        if valid and sanity_range[0] <= v <= sanity_range[1]:
            candidates.append(v)
    
    if len(candidates) == 0:
        # All flagged invalid or out of range - use valid flags anyway
        for i, (v, valid) in enumerate(zip(values, valid_flags)):
            if sanity_range[0] <= v <= sanity_range[1]:
                candidates.append(v)
    
    if len(candidates) == 0:
        # All out of range - use raw values
        candidates = list(values)
    
    if len(candidates) == 1:
        return candidates[0], 1  # value, confidence
    
    # Sort and check agreement
    candidates.sort()
    
    if len(candidates) == 2:
        if abs(candidates[0] - candidates[1]) <= agreement_tol:
            return (candidates[0] + candidates[1]) / 2, 2
        else:
            return candidates[0], 0.5  # Low confidence
    
    if len(candidates) == 3:
        # Check if any is an outlier
        d01 = abs(candidates[0] - candidates[1])
        d12 = abs(candidates[1] - candidates[2])
        d02 = abs(candidates[0] - candidates[2])
        
        # If the two closest agree
        if d01 <= agreement_tol and d12 > agreement_tol:
            return (candidates[0] + candidates[1]) / 2, 2  # Two agree, third is outlier
        elif d12 <= agreement_tol and d01 > agreement_tol:
            return (candidates[1] + candidates[2]) / 2, 2
        elif d01 <= agreement_tol and d12 <= agreement_tol:
            return candidates[1], 3  # All three agree
        elif d01 <= agreement_tol:
            return (candidates[0] + candidates[1]) / 2, 2
        elif d12 <= agreement_tol:
            return (candidates[1] + candidates[2]) / 2, 2
        else:
            return candidates[1], 1  # Median, low confidence

    'cmd_trim': 0.0,
    'cmd_flaps': 0.0,
    'cmd_gear': 0.0,
    'cmd_speedbrake': 0.0,
    
    # ── Special modes ──
    'go_around': False,
    'go_around_time': None,
    'unusual_attitude': False,
    'stall_recovery': False,
    'overspeed_recovery': False,
    'no_hydraulics': False,
    'radar_alt_suspect': False,
    'baro_alt_suspect': False,
    'ias_suspect': False,
    'all_engine_out': False,
    'failed_engine': -1,
    
    # ── Sensor health scores (0-1, decaying, boosted when consistent) ──
    'sensor_health': {},
    
    # ── Latched decisions ──
    'gear_retracted_after_takeoff': False,
    'flaps_retracted_after_takeoff': False,
}

def _norm_angle(a):
    """Normalize angle to [-180, 180]."""
    while a > 180: a -= 360
    while a < -180: a += 360
    return a

def _clamp(v, lo, hi):
    return max(lo, min(hi, v))

def _median3(a, b, c):
    """Return the median of three values."""
    if a <= b:
        if b <= c:
            return b
        elif a <= c:
            return c
        else:
            return a
    else:
        if a <= c:
            return a
        elif b <= c:
            return c
        else:
            return b

def _fuse_triple(vals, valids, sane_lo, sane_hi, tol):
    """Fuse three redundant sensors with outlier detection.
    Returns (value, confidence, outlier_index_or_-1)."""
    candidates = []
    indices = []
    for i in range(3):
        v = vals[i]
        if v is not None and sane_lo <= v <= sane_hi:
            candidates.append(v)
            indices.append(i)
    
    if len(candidates) == 0:
        # All out of range - try all values
        for i in range(3):
            if vals[i] is not None:
                candidates.append(vals[i])
                indices.append(i)
    
    if len(candidates) == 0:
        return 0, 0, -1
    
    if len(candidates) == 1:
        return candidates[0], 0.3, indices[0]
    
    # Pairwise agreement
    pairs_agree = []
    for i in range(len(candidates)):
        for j in range(i+1, len(candidates)):
            if abs(candidates[i] - candidates[j]) <= tol:
                pairs_agree.append((i, j))
    
    if len(candidates) == 2:
        if abs(candidates[0] - candidates[1]) <= tol:
            return (candidates[0] + candidates[1]) / 2, 0.7, -1
        else:
            # Disagree - return median-ish (first)
            return candidates[0], 0.3, -1
    
    # 3 candidates
    # Check if one is an outlier
    agreements = [0, 0, 0]
    for i in range(3):
        for j in range(3):
            if i != j and abs(candidates[i] - candidates[j]) <= tol:
                agreements[i] += 1
    
    # The outlier has fewer agreements
    if agreements[0] == 0 and agreements[1] >= 1 and agreements[2] >= 1:
        # candidate[0] is outlier
        return (candidates[1] + candidates[2]) / 2, 0.8, indices[0]
    elif agreements[2] == 0 and agreements[0] >= 1 and agreements[1] >= 1:
        # candidate[2] is outlier
        return (candidates[0] + candidates[1]) / 2, 0.8, indices[2]
    elif agreements[1] == 0 and agreements[0] >= 1 and agreements[2] >= 1:
        # candidate[1] is outlier
        return (candidates[0] + candidates[2]) / 2, 0.8, indices[1]
    elif agreements[0] == 2 and agreements[1] == 2 and agreements[2] == 2:
        # All agree
        return candidates[1], 1.0, -1
    else:
        # Ambiguous - return median
        return _median3(candidates[0], candidates[1], candidates[2]), 0.3, -1


def _pid(error, integral_state, prev_error, dt, kp, ki, kd, 
         int_limit=10.0, rate_limit=None):
    """Compute PID output. Returns (output, new_integral, new_prev_error)."""
    new_integral = integral_state + error * dt
    new_integral = _clamp(new_integral, -int_limit, int_limit)
    
    derivative = 0.0
    if dt > 0 and prev_error is not None:
        derivative = (error - prev_error) / dt
    
    output = kp * error + ki * new_integral + kd * derivative
    
    if rate_limit is not None:
        output = _clamp(output, -rate_limit, rate_limit)
    
    return output, new_integral, error


def autopilot_step(state, dt):
    """Main autopilot function. Called at ~20Hz."""
    
    if not _S['initialized']:
        _S['initialized'] = True
        # Initialize default command values
        _S['cmd_throttle1'] = state.get('throttle_1_pos', 0.0)
        _S['cmd_throttle2'] = state.get('throttle_2_pos', 0.0)
        _S['cmd_trim'] = state.get('elevator_trim_deg', 0.0) / 15.0  # Assume ±15° range
        _S['cmd_flaps'] = state.get('flap_pos_left_deg', 0.0)
        _S['prev_time'] = state.get('sim_time_s', 0.0)
    
    _S['step_count'] += 1
    sim_time = state.get('sim_time_s', 0.0)
    
    # ══════════════════════════════════════════════════════════════════
    # 1. SENSOR FUSION
    # ══════════════════════════════════════════════════════════════════
    
    # --- Barometric Altitude ---
    baro_vals = [state.get('baro_alt_1_ft'), state.get('baro_alt_2_ft'), state.get('baro_alt_3_ft')]
    baro_valids = [state.get('baro_alt_1_valid', True), state.get('baro_alt_2_valid', True), state.get('baro_alt_3_valid', True)]
    baro_alt, baro_conf, baro_outlier = _fuse_triple(baro_vals, baro_valids, -1000, 60000, 200)
    
    # --- Radar Altitude (AGL) ---
    radar_vals = [state.get('radar_alt_1_ft'), state.get('radar_alt_2_ft'), state.get('radar_alt_3_ft')]
    radar_valids = [state.get('radar_alt_1_valid', True), state.get('radar_alt_2_valid', True), state.get('radar_alt_3_valid', True)]
    radar_alt, radar_conf, radar_outlier = _fuse_triple(radar_vals, radar_valids, -50, 5000, 50)
    # Use radar alt only if reasonable (it's AGL, so < 5000ft is reasonable)
    radar_alt_reliable = radar_conf > 0.3 and 0 <= radar_alt < 2500
    
    # --- GPS Altitude ---
    gps_vals = [state.get('gps_1_alt_ft'), state.get('gps_2_alt_ft'), state.get('gps_3_alt_ft')]
    gps_valids = [state.get('gps_1_valid', True), state.get('gps_2_valid', True), state.get('gps_3_valid', True)]
    gps_alt, gps_conf, gps_outlier = _fuse_triple(gps_vals, gps_valids, -500, 60000, 500)
    
    # --- IRS Data ---
    irs_alt_vals = []  # IRS doesn't have altitude directly, but has pitch/roll and VS
    irs_vs_vals = [state.get('irs_1_vs_fpm'), state.get('irs_2_vs_fpm'), state.get('irs_3_vs_fpm')]
    irs_vs_valids = [state.get('irs_1_valid', True), state.get('irs_2_valid', True), state.get('irs_3_valid', True)]
    irs_vs, irs_vs_conf, _ = _fuse_triple(irs_vs_vals, irs_vs_valids, -6000, 6000, 500)
    
    irs_pitch_vals = [state.get('irs_1_pitch_deg'), state.get('irs_2_pitch_deg'), state.get('irs_3_pitch_deg')]
    irs_roll_vals = [state.get('irs_1_roll_deg'), state.get('irs_2_roll_deg'), state.get('irs_3_roll_deg')]
    irs_hdg_vals = [state.get('irs_1_heading_deg'), state.get('irs_2_heading_deg'), state.get('irs_3_heading_deg')]
    irs_hdg_valids = [state.get('irs_1_valid', True), state.get('irs_2_valid', True), state.get('irs_3_valid', True)]
    
    irs_pitch, irs_pitch_conf, _ = _fuse_triple(irs_pitch_vals, irs_hdg_valids, -45, 45, 5)
    irs_roll, irs_roll_conf, _ = _fuse_triple(irs_roll_vals, irs_hdg_valids, -90, 90, 5)
    irs_hdg, irs_hdg_conf, _ = _fuse_triple(irs_hdg_vals, irs_hdg_valids, 0, 360, 10)
    
    # --- IAS ---
    ias_vals = [state.get('ias_1_kt'), state.get('ias_2_kt'), state.get('ias_3_kt')]
    ias_valids = [state.get('ias_1_valid', True), state.get('ias_2_valid', True), state.get('ias_3_valid', True)]
    ias, ias_conf, ias_outlier = _fuse_triple(ias_vals, ias_valids, 50, 500, 30)
    
    # --- TAS ---
    tas_vals = [state.get('tas_1_kt'), state.get('tas_2_kt'), state.get('tas_3_kt')]
    tas, tas_conf, _ = _fuse_triple(tas_vals, [True, True, True], 100, 700, 30)
    
    # --- Mach ---
    mach_vals = [state.get('mach_1'), state.get('mach_2'), state.get('mach_3')]
    mach, mach_conf, _ = _fuse_triple(mach_vals, [True, True, True], 0.2, 1.0, 0.05)
    
    # --- AoA ---
    aoa_vals = [state.get('aoa_1_deg'), state.get('aoa_2_deg'), state.get('aoa_3_deg')]
    aoa_valids = [state.get('aoa_1_valid', True), state.get('aoa_2_valid', True), state.get('aoa_3_valid', True)]
    aoa, aoa_conf, aoa_outlier = _fuse_triple(aoa_vals, aoa_valids, -10, 30, 3)
    
    # --- GPS Ground Speed & Track ---
    gps_gs_vals = [state.get('gps_1_gs_kt'), state.get('gps_2_gs_kt'), state.get('gps_3_gs_kt')]
    gps_gs, gps_gs_conf, _ = _fuse_triple(gps_gs_vals, [True, True, True], 0, 700, 20)
    
    gps_track_vals = [state.get('gps_1_track_deg'), state.get('gps_2_track_deg'), state.get('gps_3_track_deg')]
    gps_track, gps_track_conf, _ = _fuse_triple(gps_track_vals, [True, True, True], 0, 360, 10)
    
    # --- IRS Ground Speed ---
    irs_gs_vals = [state.get('irs_1_gs_kt'), state.get('irs_2_gs_kt'), state.get('irs_3_gs_kt')]
    irs_gs, irs_gs_conf, _ = _fuse_triple(irs_gs_vals, irs_hdg_valids, 0, 700, 20)
    
    # --- Accelerometers ---
    accel_x, _, _ = _fuse_triple(
        [state.get('accel_1_x_g'), state.get('accel_2_x_g'), state.get('accel_3_x_g')],
        [state.get('accel_1_valid', True), state.get('accel_2_valid', True), state.get('accel_3_valid', True)],
        -3, 3, 0.3
    )
    accel_y, _, _ = _fuse_triple(
        [state.get('accel_1_y_g'), state.get('accel_2_y_g'), state.get('accel_3_y_g')],
        [state.get('accel_1_valid', True), state.get('accel_2_valid', True), state.get('accel_3_valid', True)],
        -3, 3, 0.3
    )
    accel_z, _, _ = _fuse_triple(
        [state.get('accel_1_z_g'), state.get('accel_2_z_g'), state.get('accel_3_z_g')],
        [state.get('accel_1_valid', True), state.get('accel_2_valid', True), state.get('accel_3_valid', True)],
        -1, 3, 0.3
    )
    
    # --- Navigation data ---
    ground_speed = state.get('ground_speed_kt', gps_gs if gps_gs else irs_gs)
    true_heading = state.get('true_heading_deg', irs_hdg)
    magnetic_heading = state.get('magnetic_heading_deg', irs_hdg)
    track = state.get('track_deg', gps_track)
    vs = state.get('vertical_speed_fpm', irs_vs)
    fpa = state.get('flight_path_angle_deg', 0.0)
    
    # --- Environment ---
    wind_dir = state.get('wind_dir_deg', 0)
    wind_speed = state.get('wind_speed_kt', 0)
    oat = state.get('oat_c', 15)
    
    # --- Engines ---
    eng1_running = state.get('engine_1_running', False)
    eng2_running = state.get('engine_2_running', False)
    eng1_n1 = state.get('engine_1_n1_pct', 0)
    eng2_n1 = state.get('engine_2_n2_pct', 0)
    eng1_thrust = state.get('engine_1_thrust_pct', 0)
    eng2_thrust = state.get('engine_2_thrust_pct', 0)
    eng1_fire = state.get('engine_1_fire', False)
    eng2_fire = state.get('engine_2_fire', False)
    
    # --- Weight on Wheels ---
    wow_left = state.get('weight_on_wheels_left', True)
    wow_nose = state.get('weight_on_wheels_nose', True)
    wow_right = state.get('weight_on_wheels_right', True)
    any_wow = wow_left or wow_nose or wow_right
    all_wow = wow_left and wow_nose and wow_right
    
    # --- Hydraulics ---
    hyd_a = state.get('hyd_a_press_psi', 3000)
    hyd_b = state.get('hyd_b_press_psi', 3000)
    hyd_c = state.get('hyd_c_press_psi', 3000)
    hyd_a_ok = hyd_a > 1500
    hyd_b_ok = hyd_b > 1500
    hyd_c_ok = hyd_c > 1500
    no_hydraulics = not (hyd_a_ok or hyd_b_ok or hyd_c_ok)
    
    # --- Electrical ---
    bus_1_v = state.get('bus_1_voltage', 115)
    bus_2_v = state.get('bus_2_voltage', 115)
    bus_ess_v = state.get('bus_ess_voltage', 115)
    
    # --- Current controls ---
    current_throttle1 = state.get('throttle_1_pos', _S['cmd_throttle1'])
    current_throttle2 = state.get('throttle_2_pos', _S['cmd_throttle2'])
    current_trim = state.get('elevator_trim_deg', _S['cmd_trim'] * 15)
    
    # --- Targets ---
    target_alt = state.get('target_alt_ft', 0)
    target_hdg = state.get('target_hdg_deg', 0)
    target_ias = state.get('target_ias_kt', 250)
    target_vs = state.get('target_vs_fpm', 0)
    ap_engaged = state.get('ap_engaged', False)
    at_engaged = state.get('at_engaged', False)
    
    # --- Fuel ---
    fuel_total = state.get('fuel_total_kg', 10000)
    
    # --- Weight/CG ---
    gross_weight = state.get('gross_weight_kg', 65000)
    cg = state.get('cg_pct_mac', 30)
    
    # ══════════════════════════════════════════════════════════════════
    # 2. CROSS-CHECK SENSORS & DETECT FAILURES
    # ══════════════════════════════════════════════════════════════════
    
    # --- Cross-check altitude sources ---
    # Baro alt and GPS alt should agree within ~500ft (baro setting errors)
    baro_gps_agree = abs(baro_alt - gps_alt) < 500 if (baro_conf > 0.3 and gps_conf > 0.3) else True
    
    # If baro and GPS disagree, check IRS for tie-breaking
    if not baro_gps_agree:
        # IRS VS should be consistent with altitude change
        # If IRS says we're climbing/descending, check which alt source agrees
        if _S['prev_alt'] is not None and _S['step_count'] > 5:
            elapsed = dt * 5  # approximate time over last 5 steps
            if elapsed > 0:
                # Compute altitude rate from baro
                baro_rate = (baro_alt - _S['prev_alt']) / max(elapsed, 0.01) * 60  # fpm
                # IRS VS
                if abs(baro_rate - irs_vs) < abs(baro_rate + irs_vs):
                    # Baro agrees with IRS
                    _S['baro_alt_suspect'] = False
                else:
                    _S['baro_alt_suspect'] = True
    
    # Determine best altitude source
    if baro_gps_agree and baro_conf > 0.3:
        fused_alt = baro_alt
    elif gps_conf > 0.5:
        fused_alt = gps_alt
    elif baro_conf > 0.3:
        fused_alt = baro_alt
    else:
        fused_alt = gps_alt
    
    # If radar alt is available and reliable, use it for low altitude
    # For AGL: radar alt is direct measurement
    if radar_alt_reliable and radar_alt < 2500:
        # Radar alt is more accurate at low altitude
        # But we still need MSL for navigation
        # Use radar for landing and terrain avoidance
        pass  # We'll use radar_alt directly when needed
    
    # --- Cross-check airspeed sources ---
    # IAS should be consistent with TAS and density altitude
    # TAS ≈ IAS * sqrt(1 + 0.00002256 * density_alt) roughly
    # Or: IAS = TAS * sqrt(rho/rho0), rho = rho0 * exp(-density_alt / 34000)
    
    # Also cross-check IAS with ground speed and wind
    if ias_conf > 0.3 and gps_gs_conf > 0.3:
        hdg_rad = math.radians(irs_hdg)
        wind_rad = math.radians(wind_dir)
        headwind = wind_speed * math.cos(wind_rad - hdg_rad)
        # Rough TAS estimate from ground speed
        estimated_tas = gps_gs + headwind  # approximate
        # IAS from estimated TAS (rough)
        density_alt = state.get('adc_1_density_alt_ft', 0)
        rho_ratio = math.exp(-density_alt / 34000)
        estimated_ias = estimated_tas * math.sqrt(rho_ratio)
        
        ias_gs_agree = abs(ias - estimated_ias) < 50  # within 50 kt
    else:
        ias_gs_agree = True
    
    # --- Detect frozen sensors ---
    if _S['prev_ias'] is not None and _S['step_count'] > 10:
        if abs(ias - _S['prev_ias']) < 0.1 and abs(ground_speed - (_S.get('prev_gs', ground_speed))) > 5:
            # IAS is frozen but ground speed is changing
            _S['ias_suspect'] = True
    
    _S['prev_gs'] = ground_speed
    _S['prev_alt'] = fused_alt
    _S['prev_ias'] = ias
    
    # --- Detect radar altimeter stuck at zero ---
    if _S['step_count'] > 20:
        if radar_alt < 1 and fused_alt > 500:
            _S['radar_alt_suspect'] = True
        elif radar_alt > 10 or fused_alt < 100:
            _S['radar_alt_suspect'] = False
    
    # ══════════════════════════════════════════════════════════════════
    # 3. FLIGHT PHASE DETECTION
    # ══════════════════════════════════════════════════════════════════
    
    was_airborne = _S['was_airborne']
    is_airborne = not all_wow and fused_alt > 50 and ias > 100
    
    if is_airborne and not was_airborne:
        # Just took off
        _S['airborne_time'] = sim_time
        _S['phase'] = 'TAKEOFF'
        _S['gear_retracted_after_takeoff'] = False
        _S['flaps_retracted_after_takeoff'] = False
    elif was_airborne and all_wow:
        # Just landed
        _S['phase'] = 'GROUND'
        _S['airborne_time'] = None
    elif is_airborne:
        # Airborne phase determination
        alt_diff = target_alt - fused_alt
        if _S['go_around']:
            _S['phase'] = 'GO_AROUND'
        elif _S['phase'] == 'TAKEOFF' and (sim_time - (_S['airborne_time'] or 0)) < 30:
            _S['phase'] = 'TAKEOFF'
        elif radar_alt_reliable and radar_alt < 1500 and target_alt < fused_alt and abs(vs) > 100:
            _S['phase'] = 'APPROACH'
        elif abs(alt_diff) > 200 and alt_diff > 0:
            _S['phase'] = 'CLIMB'
        elif abs(alt_diff) > 200 and alt_diff < 0:
            _S['phase'] = 'DESCENT'
        elif abs(alt_diff) <= 200:
            _S['phase'] = 'CRUISE'
        else:
            _S['phase'] = 'CRUISE'
    else:
        _S['phase'] = 'GROUND'
    
    _S['was_airborne'] = is_airborne
    
    # ══════════════════════════════════════════════════════════════════
    # 4. ENGINE FAILURE DETECTION
    # ══════════════════════════════════════════════════════════════════
    
    _S['all_engine_out'] = (not eng1_running) and (not eng2_running)
    
    if eng1_running and not eng2_running:
        _S['failed_engine'] = 2
    elif eng2_running and not eng1_running:
        _S['failed_engine'] = 1
    else:
        _S['failed_engine'] = -1
    
    _S['no_hydraulics'] = no_hydraulics
    
    # ══════════════════════════════════════════════════════════════════
    # 5. ENVELOPE PROTECTION CHECKS
    # ══════════════════════════════════════════════════════════════════
    
    stall_warning = False
    overspeed_warning = False
    bank_exceeded = False
    unusual_attitude = False
    
    # Stall: AoA > ~15° for A320, or speed < Vref
    stall_aoa = 15.0  # degrees, approximate
    if aoa_conf > 0.3 and aoa > stall_aoa:
        stall_warning = True
    # Also check for very low speed
    if ias < 120 and is_airborne:
        stall_warning = True
    # If IAS is suspect, use estimated IAS from GS
    if _S['ias_suspect'] and is_airborne:
        if gps_gs < 120:  # Rough proxy
            stall_warning = True
    
    # Overspeed
    vmo = 350  # KIAS
    mmo = 0.82
    if ias > vmo or mach > mmo:
        overspeed_warning = True
    
    # Bank angle protection
    if abs(irs_roll) > 35:
        bank_exceeded = True
    if radar_alt_reliable and radar_alt < 1000 and abs(irs_roll) > 20:
        bank_exceeded = True
    
    # Unusual attitude
    if abs(irs_pitch) > 30 or abs(irs_roll) > 60:
        unusual_attitude = True
    if irs_pitch < -15 and abs(irs_roll) > 45:
        unusual_attitude = True
    # Nose extremely low
    if irs_pitch < -20:
        unusual_attitude = True
    
    # ══════════════════════════════════════════════════════════════════
    # 6. COMPUTE CONTROL COMMANDS
    # ══════════════════════════════════════════════════════════════════
    
    elevator_cmd = 0.0
    aileron_cmd = 0.0
    rudder_cmd = 0.0
    throttle1_cmd = _S['cmd_throttle1']
    throttle2_cmd = _S['cmd_throttle2']
    trim_cmd = _S['cmd_trim']
    flaps_cmd = _S['cmd_flaps']
    gear_cmd = _S['cmd_gear']
    speedbrake_cmd = _S['cmd_speedbrake']
    
    # ── 6a. EMERGENCY: Unusual Attitude Recovery ──
    if unusual_attitude:
        _S['unusual_attitude'] = True
        # Priority 1: Wings level
        if abs(irs_roll) > 10:
            # Roll toward nearest wings-level
            if irs_roll > 0:
                aileron_cmd = -0.8  # Roll left
            else:
                aileron_cmd = 0.8  # Roll right
        
        # Priority 2: Recover pitch
        if irs_pitch > 20:
            elevator_cmd = -0.8  # Push nose down
        elif irs_pitch < -10:
            elevator_cmd = 0.8  # Pull nose up
        
        # Full thrust if low speed
        if ias < 200:
            throttle1_cmd = 1.0
            throttle2_cmd = 1.0
        
        # Speedbrakes in
        speedbrake_cmd = 0.0
        
        # Reset integrators
        _S['int_alt'] = 0
        _S['int_vs'] = 0
        _S['int_pitch'] = 0
        _S['int_roll'] = 0
        _S['int_hdg'] = 0
        
        _S['unusual_attitude'] = unusual_attitude
        
    # ── 6b. EMERGENCY: Stall Recovery ──
    elif stall_warning:
        _S['stall_recovery'] = True
        # Push nose down
        elevator_cmd = -0.5
        # Full thrust
        if eng1_running:
            throttle1_cmd = 1.0
        if eng2_running:
            throttle2_cmd = 1.0
        # Speedbrakes in
        speedbrake_cmd = 0.0
        # Wings level
        if abs(irs_roll) > 15:
            if irs_roll > 0:
                aileron_cmd = -0.5
            else:
                aileron_cmd = 0.5
        
        # Reset integrators
        _S['int_alt'] = 0
        _S['int_vs'] = 0
        
    # ── 6c. EMERGENCY: Overspeed Recovery ──
    elif overspeed_warning:
        _S['overspeed_recovery'] = True
        # Reduce thrust
        throttle1_cmd = 0.0
        throttle2_cmd = 0.0
        # Pull up slightly to slow down
        elevator_cmd = 0.3
        # Deploy speedbrakes
        speedbrake_cmd = 1.0
        
        # Reset integrators
        _S['int_spd'] = 0
    
    # ── 6d. EMERGENCY: No Hydraulics (thrust-only control) ──
    elif no_hydraulics and is_airborne:
        # Can't control surfaces - use differential thrust
        # This is like UAL 232
        # Map elevator to symmetric thrust changes
        # Map aileron to differential thrust
        # Map rudder to differential thrust
        
        # Still compute desired elevator/aileron/rudder from normal control
        # Then convert to thrust commands
        
        # Normal altitude/heading control (simplified)
        alt_err = target_alt - fused_alt
        vs_target = _clamp(alt_err * 0.5, -2000, 2000)
        vs_err = vs_target - vs
        pitch_target = _clamp(vs_err * 0.002, -15, 15)
        pitch_err = pitch_target - irs_pitch
        
        hdg_err = _norm_angle(target_hdg - irs_hdg)
        bank_target = _clamp(hdg_err * 0.5, -25, 25)
        bank_err = bank_target - irs_roll
        
        # Convert to thrust commands
        base_thrust = 0.5  # Maintain some thrust
        thrust_diff = _clamp(bank_err * 0.02, -0.4, 0.4)
        thrust_sym = _clamp(-pitch_err * 0.03, -0.3, 0.3)
        
        throttle1_cmd = _clamp(base_thrust + thrust_sym - thrust_diff, 0, 1)
        throttle2_cmd = _clamp(base_thrust + thrust_sym + thrust_diff, 0, 1)
        
        # Surface commands won't work but output them anyway
        elevator_cmd = pitch_err * 0.1
        aileron_cmd = bank_err * 0.1
        rudder_cmd = thrust_diff * 0.5
        
    # ── 6e. NORMAL CONTROL ──
    elif is_airborne and (ap_engaged or not all_wow):
        
        # ── Altitude / Vertical Speed Control ──
        alt_err = target_alt - fused_alt
        
        # Determine desired VS based on altitude error and phase
        if _S['phase'] == 'APPROACH':
            # Use target_vs for approach (typically from glide slope)
            vs_target = target_vs if abs(target_vs) > 100 else -700
            # Limit VS based on altitude
            if radar_alt_reliable and radar_alt < 100:
                vs_target = max(vs_target, -200)  # Reduce VS near ground
            if radar_alt_reliable and radar_alt < 30:
                vs_target = -50  # Flare
        elif _S['phase'] == 'GO_AROUND':
            vs_target = 2000  # Climb at 2000 fpm
        elif _S['phase'] == 'TAKEOFF':
            vs_target = 2500
        elif abs(alt_err) < 200:
            vs_target = target_vs

            # Altitude capture
            vs_target = _clamp(alt_err * 0.5, -2500, 2500)
            # Reduce VS near target altitude
            if abs(alt_err) < 500:
                vs_target = _clamp(alt_err * 0.3, -1500, 1500)
        
        # VS PID → pitch target
        vs_err = vs_target - vs
        pitch_target = _clamp(vs_err * 0.003, -15, 15)
        
        # Account for bank angle - need more pitch in bank
        bank_compensation = abs(irs_roll) * 0.15  # degrees of pitch per degree of bank
        pitch_target += bank_compensation
        
        # Clamp pitch target
        if _S['phase'] == 'TAKEOFF':
            pitch_target = _clamp(pitch_target, 5, 18)
        elif _S['phase'] == 'GO_AROUND':
            pitch_target = _clamp(pitch_target, 10, 18)
        elif _S['phase'] == 'APPROACH':
            if radar_alt_reliable and radar_alt < 50:
                pitch_target = _clamp(pitch_target, -3, 10)  # Allow slight nose down for flare
            else:
                pitch_target = _clamp(pitch_target, -3, 10)
        else:
            pitch_target = _clamp(pitch_target, -10, 15)
        
        # Pitch PID → elevator
        pitch_err = pitch_target - irs_pitch

        gyro_pitch_rate = _fuse_triple(
            [state.get('gyro_1_pitch_rate_dps'), state.get('gyro_2_pitch_rate_dps'), state.get('gyro_3_pitch_rate_dps')],
            [state.get('gyro_1_valid', True), state.get('gyro_2_valid', True), state.get('gyro_3_valid', True)],
            -20, 20, 5
        )[0] if True else 0
        
        elevator_cmd, _S['int_pitch'], _S['prev_pitch_err'] = _pid(

            kp=0.15, ki=0.005, kd=0.05,
            int_limit=2.0, rate_limit=1.0
        )
        
        # Add pitch rate damping
        elevator_cmd -= gyro_pitch_rate * 0.02
        elevator_cmd = _clamp(elevator_cmd, -1.0, 1.0)
        
        # ── Altitude integral for trim (slow) ──
        trim_adjust, _S['int_alt'], _S['prev_alt_err'] = _pid(
            alt_err, _S['int_alt'], _S['prev_alt_err'], dt,
            kp=0.0, ki=0.0001, kd=0.0,
            int_limit=1.0
        )
        trim_cmd = _clamp(_S['cmd_trim'] + trim_adjust * dt, -1, 1)
        
        # ── Heading / Bank Angle Control ──
        hdg_err = _norm_angle(target_hdg - irs_hdg)
        
        if _S['phase'] == 'APPROACH' and radar_alt_reliable and radar_alt < 1000:
            max_bank = 15
        elif radar_alt_reliable and radar_alt < 1000:
            max_bank = 20
        else:
            max_bank = 30
        
        bank_target = _clamp(hdg_err * 0.5, -max_bank, max_bank)
        
        # Bank PID → aileron
        bank_err = bank_target - irs_roll
        
        gyro_roll_rate = _fuse_triple(
            [state.get('gyro_1_roll_rate_dps'), state.get('gyro_2_roll_rate_dps'), state.get('gyro_3_roll_rate_dps')],
            [state.get('gyro_1_valid', True), state.get('gyro_2_valid', True), state.get('gyro_3_valid', True)],
            -60, 60, 10

        aileron_cmd, _S['int_roll'], _S['prev_roll_err'] = _pid(
            bank_err, _S['int_roll'], _S['prev_roll_err'], dt,
            kp=0.1, ki=0.003, kd=0.04,
            int_limit=1.0, rate_limit=1.0
        )
        
        # Add roll rate damping
        aileron_cmd -= gyro_roll_rate * 0.01
        aileron_cmd = _clamp(aileron_cmd, -1.0, 1.0)
        
        # ── Speed / Throttle Control ──
        # Determine target speed based on phase
        if _S['phase'] == 'TAKEOFF':
            spd_target = max(target_ias, 145)  # At least Vr
        elif _S['phase'] == 'APPROACH':
            spd_target = target_ias  # Should be Vref + 5
        elif _S['phase'] == 'GO_AROUND':
            spd_target = max(target_ias, 150)
        else:
            spd_target = target_ias
        
        spd_err = spd_target - ias
        
        # Speed PID → throttle
        if at_engaged or True:  # Always control speed
            base_throttle, _S['int_spd'], _S['prev_spd_err'] = _pid(
                spd_err, _S['int_spd'], _S['prev_spd_err'], dt,
                kp=0.008, ki=0.001, kd=0.003,
                int_limit=0.5, rate_limit=1.0
            )
            base_throttle = _clamp(base_throttle + 0.5, 0, 1)  # Offset: 0.5 is roughly cruise
        else:
            base_throttle = _S['cmd_throttle1']
        
        # Adjust throttle based on altitude change
        if _S['phase'] == 'CLIMB':
            base_throttle = max(base_throttle, 0.75)
        elif _S['phase'] == 'DESCENT':
            base_throttle = min(base_throttle, 0.5)
            if target_vs < -1500:
                base_throttle = min(base_throttle, 0.3)
        elif _S['phase'] == 'TAKEOFF':
            base_throttle = 0.95  # TOGA
        elif _S['phase'] == 'GO_AROUND':
            base_throttle = 0.95  # TOGA
        
        if _S['all_engine_out']:
            base_throttle = 0.0  # No engines to command
        
        throttle1_cmd = base_throttle
        throttle2_cmd = base_throttle
        
        # ── Rudder Control ──
        # Yaw damper: suppress yaw rate
        gyro_yaw_rate = _fuse_triple(
            [state.get('gyro_1_yaw_rate_dps'), state.get('gyro_2_yaw_rate_dps'), state.get('gyro_3_yaw_rate_dps')],
            [state.get('gyro_1_valid', True), state.get('gyro_2_valid', True), state.get('gyro_3_valid', True)],
            -30, 30, 5
        )[0]
        
        # Yaw damper
        rudder_cmd = -gyro_yaw_rate * 0.03
