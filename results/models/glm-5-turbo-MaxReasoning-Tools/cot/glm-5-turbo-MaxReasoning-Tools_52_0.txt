    vals = [v for v in values if v is not None and not (isinstance(v, float) and math.isnan(v))]
    if not vals:
        return None
    vals.sort()
    n = len(vals)
    if n % 2 == 1:
        return vals[n // 2]
    return (vals[n // 2 - 1] + vals[n // 2]) / 2.0

    """Mean of non-None values."""
    vals = [v for v in values if v is not None and not (isinstance(v, float) and math.isnan(v))]
    if not vals:
        return None

    """Exponential moving average."""
    if new is None:
        return prev
    return alpha * new + (1.0 - alpha) * prev

    median = _safe_median(values)

def _iqr_filter(vals, factor=1.5):
    """Remove outliers using IQR method."""
    vs = [x for x in vals if x is not None and math.isfinite(x)]
    if len(vs) < 3:
        return vs
    vs_sorted = sorted(vs)
    n = len(vs_sorted)
    q1 = vs_sorted[n // 4]
    q3 = vs_sorted[3 * n // 4]
    iqr = q3 - q1
    lo = q1 - factor * iqr
    hi = q3 + factor * iqr
    return [x for x in vs if lo <= x <= hi]

        _S['obs_dist'] = _ema(_S['obs_dist'], obs_dist, 0.4) if obs_dist is not None else _S['obs_dist']
        _S['obs_rel_speed'] = _ema(_S['obs_rel_speed'], obs_rel_speed, 0.4) if obs_rel_speed is not None else _S['obs_rel_speed']
    else:
        _S['obs_detected'] = False
    
    # --------------------------------------------------------
    # 3. DERIVED VALUES
    # --------------------------------------------------------
    cur_speed = _S['speed_kmh']
    cur_speed_ms = cur_speed / 3.6  # Convert to m/s
    
    # Effective speed limit
    eff_speed_limit = target_speed_raw
    if _S['speed_limit'] is not None and _S['speed_limit'] > 0:
        eff_speed_limit = min(eff_speed_limit, _S['speed_limit'])
    
    # Traction-based speed reduction
    traction_factor = mu
    if rain:
        traction_factor *= 0.7
    if temp < 2:
        traction_factor *= 0.8
    
    # Max safe speed based on curvature and traction
    curv = _S['curvature']
    if curv is not None and abs(curv) > 0.0001:
        # curvature in 1/m → radius in m
        R = 1.0 / abs(curv)
        # v_max = sqrt(mu * g * R)
        v_max_curve = math.sqrt(max(traction_factor, 0.1) * 9.81 * R) * 3.6  # km/h
        eff_speed_limit = min(eff_speed_limit, v_max_curve)
    
    # Reduce target speed on low traction
    if traction_factor < 0.5:
        eff_speed_limit = min(eff_speed_limit, 80.0 * traction_factor / 0.5)
    if traction_factor < 0.3:
        eff_speed_limit = min(eff_speed_limit, 40.0)
    
    # --------------------------------------------------------
    # 4. SAFETY MONITORING
    # --------------------------------------------------------
    # Monitor brake effectiveness
    if _S['prev_cmd_brake'] > 0.3 and cur_speed_ms > 1.0:
        expected_decel = _S['prev_cmd_brake'] * 8.0 * traction_factor  # m/s²
        actual_decel = (_S['prev_speed'] - cur_speed_ms) / max(actual_dt, 0.001)
        if actual_decel < expected_decel * 0.3 and _S['prev_speed'] > cur_speed_ms + 0.1:
            _S['speed_not_decreasing_count'] += 1
        else:
            _S['speed_not_decreasing_count'] = max(0, _S['speed_not_decreasing_count'] - 1)
    else:
        _S['speed_not_decreasing_count'] = max(0, _S['speed_not_decreasing_count'] - 1)
    
    if _S['speed_not_decreasing_count'] > 5:
        _S['brake_effective'] = False
    
    # --------------------------------------------------------
    # 5. OBSTACLE ASSESSMENT & DECISION
    # --------------------------------------------------------
    output = {}
    emergency = False
    
    # Calculate TTC (time to collision)
    ttc = float('inf')
    if _S['obs_detected'] and _S['obs_dist'] is not None and _S['obs_dist'] < 200:
        # obs_rel_speed: negative means obstacle is slower (closing)
        closing_speed_ms = -_S['obs_rel_speed'] / 3.6  # Convert to m/s, positive = closing
        if closing_speed_ms > 0.5:
            ttc = _S['obs_dist'] / closing_speed_ms
        elif closing_speed_ms <= 0:
            # Obstacle is moving away or same speed
            ttc = float('inf')
        else:
            # Closing but slowly
            ttc = _S['obs_dist'] / max(closing_speed_ms, 0.1)
    
    # Also check lidar front beams for close obstacles
    min_front = min_front_lidar(lidar)
    if min_front is not None and min_front < 10:
        # Very close obstacle from lidar
        ttc_lidar = min_front / max(cur_speed_ms, 1.0) if cur_speed_ms > 1 else float('inf')
        ttc = min(ttc, ttc_lidar)
    
    # Emergency braking
    if ttc < 1.5 or (min_front is not None and min_front < 5):
        emergency = True
    
    # Obstacle ahead needs action
    obstacle_action = 'none'
    if _S['obs_detected'] and _S['obs_dist'] < 150:
        if ttc < 4.0:
            obstacle_action = 'brake'
        if ttc < 6.0 and _S['obs_dist'] < 80:
            obstacle_action = 'consider_lane_change'
    
    # --------------------------------------------------------
    # 6. LANE CHANGE LOGIC
    # --------------------------------------------------------
    need_lane_change = (target_lane != current_lane)
    
    if _S['lc_state'] == 'idle':
        if need_lane_change and obstacle_action in ('brake', 'consider_lane_change'):
            # Decide direction
            direction = target_lane - current_lane  # -1 or +1
            _S['lc_direction'] = 1 if direction > 0 else -1
            # Wait, convention: positive lane number could be left or right
            # Let's assume lane 0=left, 1=center, 2=right
            # So target > current → move right
            _S['lc_direction'] = 1  # right
            if direction < 0:
                _S['lc_direction'] = -1  # left
            _S['lc_state'] = 'checking'
            _S['lc_check_counter'] = 0
    
    elif _S['lc_state'] == 'checking':
        # Verify adjacent lane is clear for a few frames
        if _S['lc_direction'] < 0:
            clear = check_adjacent_clear(lidar, -1)
        else:
            clear = check_adjacent_clear(lidar, 1)
        
        # Also check vision: the line we need to cross should be dashed
        can_cross = True
        if _S['lc_direction'] < 0:
            if left_type == 'solid':
                can_cross = False
        else:
            if right_type == 'solid':
                can_cross = False
        
        if clear and can_cross:
            _S['lc_check_counter'] += 1
        else:
            _S['lc_check_counter'] = max(0, _S['lc_check_counter'] - 1)
            _S['lc_blocked_counter'] += 1
        
        # Need 3 consecutive clear frames to proceed
        if _S['lc_check_counter'] >= 3:
            _S['lc_state'] = 'executing'
            _S['lc_start_time'] = sim_time
        
        # If blocked for too long, abort
        if _S['lc_blocked_counter'] > 30:
            _S['lc_state'] = 'idle'
            _S['lc_blocked_counter'] = 0
    
    elif _S['lc_state'] == 'executing':
        # Lane change in progress - steer toward target lane
        elapsed = sim_time - _S['lc_start_time']
        
        # Check if still clear during change
        if _S['lc_direction'] < 0:
            still_clear = check_adjacent_clear(lidar, -1)
        else:
            still_clear = check_adjacent_clear(lidar, 1)
        
        if not still_clear and elapsed > 0.5:
            # Abort lane change
            _S['lc_state'] = 'aborting'
        
        # Lane change should take about 3-4 seconds
        if elapsed > 4.0:
            _S['lc_state'] = 'idle'
            _S['lc_check_counter'] = 0
            _S['lc_blocked_counter'] = 0
    
    elif _S['lc_state'] == 'aborting':
        # Return to current lane
        elapsed_since_abort = sim_time - _S['lc_start_time'] - 4.0  # Approximate
        if elapsed_since_abort > 2.0 or _S['lc_state'] == 'aborting':
            # Check if we're back in lane
            if _S['lane_offset'] is not None and abs(_S['lane_offset']) < 0.5:
                _S['lc_state'] = 'idle'
    
    # --------------------------------------------------------
    # 7. EMERGENCY HANDLING
    # --------------------------------------------------------
    if emergency:
        # Emergency brake
        brake_cmd = 1.0
        if not _S['brake_effective']:
            brake_cmd = 0.3  # Partial brake even if not fully effective
        throttle_cmd = 0.0
        
        # Also try to steer away from obstacle
        steer_cmd = 0.0
        # Use lidar to find the direction with more space
        left_space = 0.0
        right_space = 0.0
        for ang in ['+15', '+30', '+45']:
            r = lidar.get(ang)
            if r is not None:
                left_space += r
        for ang in ['-15', '-30', '-45']:
            r = lidar.get(ang)
            if r is not None:
                right_space += r
        
        if left_space > right_space + 10:
            steer_cmd = -0.3  # Steer left (toward more space)
        elif right_space > left_space + 10:
            steer_cmd = 0.3  # Steer right
        
        # Reduce steering on low traction
        if traction_factor < 0.5:
            steer_cmd *= traction_factor * 2
        
        _S['prev_cmd_throttle'] = throttle_cmd
        _S['prev_cmd_brake'] = brake_cmd
        _S['prev_speed'] = cur_speed_ms
        return {'steering': _clamp(steer_cmd, -1, 1),
                'throttle': throttle_cmd,
                'brake': _clamp(brake_cmd, 0, 1)}
    
    # --------------------------------------------------------
    # 8. LATERAL CONTROL (Lane Keeping / Lane Change)
    # --------------------------------------------------------
    
    # Compute lateral error
    if left_line is not None and right_line is not None:
        # Error: positive means we need to steer left
        lane_error = (left_line - right_line) / 2.0
    elif _S['lane_offset'] is not None:
        lane_error = -_S['lane_offset']  # Negative because offset positive = right of center → steer left
    else:
        lane_error = 0.0
    
    # Add curvature feedforward
    curv_ff = 0.0
    if curv is not None:
        # Curvature feedforward: steer into the curve
        # Positive curvature = road curves left
        curv_ff = -curv * 2.7 * 1000  # Scale factor (tune as needed)
    
    # Lane change offset
    lc_offset = 0.0
    if _S['lc_state'] == 'executing':
        elapsed = sim_time - _S['lc_start_time']
        # Smooth lane change trajectory: sinusoidal
        lane_width = _S['left_line'] + _S['right_line'] if (_S['left_line'] and _S['right_line']) else 3.5
        # Target: move one lane width in the direction
        target_offset = _S['lc_direction'] * lane_width
        # Progress: 0 to 1 over ~3 seconds
        progress = _clamp(elapsed / 3.0, 0, 1)
        # Smooth trajectory (sigmoid-like)
        smooth_progress = 0.5 * (1 - math.cos(progress * math.pi))
        lc_offset = target_offset * smooth_progress
        # Add to error (we want to move toward lc_offset from center)
        lane_error -= lc_offset
    
    elif _S['lc_state'] == 'aborting':
        # Return to center
        lc_offset = 0.0
        lane_error = -_S['lane_offset'] if _S['lane_offset'] is not None else 0.0
    
    # PID for lateral control
    Kp_lat = 0.15
    Ki_lat = 0.005
    Kd_lat = 0.3
    
    # Speed-dependent gain scaling
    speed_factor = max(cur_speed_ms, 5.0) / 25.0  # Normalize around 25 m/s (90 km/h)
    Kp_lat_eff = Kp_lat / speed_factor
    
    # Traction-dependent gain scaling
