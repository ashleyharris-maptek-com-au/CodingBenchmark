// ============= FOREST DP =============
void dfs_forest(int v, int parent, const vector<vector<int>>& g, 
                vector<int>& dp_take, vector<int>& dp_not_take) {
    dp_take[v] = 1;
    dp_not_take[v] = 0;
    for (int u : g[v]) {
        if (u == parent) continue;
        dfs_forest(u, v, g, dp_take, dp_not_take);
        dp_take[v] += dp_not_take[u];
        dp_not_take[v] += max(dp_take[u], dp_not_take[u]);
    }
}

void traceback_forest(int v, int parent, bool take, const vector<vector<int>>& g,
                      const vector<int>& dp_take, const vector<int>& dp_not_take,
                      vector<int>& result) {
    if (take) {
        result.push_back(v);
        for (int u : g[v]) {
            if (u == parent) continue;
            traceback_forest(u, v, false, g, dp_take, dp_not_take, result);
        }
    } else {

            bool child_take = (dp_take[u] >= dp_not_take[u]);
            traceback_forest(u, v, child_take, g, dp_take, dp_not_take, result);

// Returns maximum matching size
int hopcroft_karp(const vector<vector<int>>& g, int n_left, int n_right) {
    // g[i] for i in [0, n_left) contains right-side vertices
    vector<int> pair_left(n_left, -1), pair_right(n_right, -1);
    vector<int> dist(n_left);
    
    function<bool()> bfs = [&]() {
        queue<int> q;
        for (int i = 0; i < n_left; i++) {
            if (pair_left[i] == -1) {
                dist[i] = 0;
                q.push(i);
            } else {
                dist[i] = INT_MAX;
            }
        }
        bool found = false;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int v : g[u]) {
                if (pair_right[v] != -1) {
                    if (dist[pair_right[v]] == INT_MAX) {
                        dist[pair_right[v]] = dist[u] + 1;
                        q.push(pair_right[v]);
                    }
                } else {
                    found = true;
                }
            }
        }
        return found;
    };
    
    function<bool(int)> dfs = [&](int u) {
        for (int v : g[u]) {
            if (pair_right[v] == -1 || (dist[pair_right[v]] == dist[u] + 1 && dfs(pair_right[v]))) {
                pair_left[u] = v;
                pair_right[v] = u;
                return true;
            }
        }
        dist[u] = INT_MAX;
        return false;
    };
    
    int matching = 0;
    while (bfs()) {
        for (int i = 0; i < n_left; i++) {
            if (pair_left[i] == -1 && dfs(i)) {
                matching++;
            }
        }
    }
    return matching;
}

                      uint64_t R, int current_size, vector<int>& current_set,
                      int& best_size, vector<int>& best_set,
                      const vector<int>& orig_id) {
    if (R == 0) {
        if (current_size > best_size) {
            best_size = current_size;
            best_set = current_set;
        }
        return;
    }
    
    // Upper bound: |R| - matching_size
    // For now, just use |R| + current_size as trivial upper bound
    int ub = current_size + __popcnt(R);
    if (ub <= best_size) return;
    
    // Better bound: greedy matching

    int matching = 0;

        Rcopy &= ~(1ULL << v);
        uint64_t nbs = nbr[v] & R;
        if (nbs) {

            matching++;

    for (int i = 0; i < n; i++) toLocal[vertices[i]] = i;
    
    // Build local adjacency
    vector<vector<int>> adj(n);
    for (int i = 0; i < n; i++) {
        for (int u : globalAdj[vertices[i]]) {
            if (toLocal.count(u)) {
                adj[i].push_back(toLocal[u]);
            }
        }
    }
    
    // DP
    vector<int> take(n), notake(n);
    vector<bool> visited(n, false);
    
    function<void(int, int)> dp = [&](int v, int p) {
        visited[v] = true;
        take[v] = 1;
        notake[v] = 0;
        for (int u : adj[v]) {
            if (u == p) continue;
            dp(u, v);
            take[v] += notake[u];
            notake[v] += max(take[u], notake[u]);
        }
    };
    
    int total = 0;
    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            dp(i, -1);
            total += max(take[i], notake[i]);
        }
    }
    
    // Traceback
    vector<int> result;
    visited.assign(n, false);
    
    function<void(int, int, bool)> trace = [&](int v, int p, bool doTake) {
        visited[v] = true;
        if (doTake) {
            result.push_back(vertices[v]);
            for (int u : adj[v]) {
                if (u != p && !visited[u]) trace(u, v, false);
            }
        } else {
            for (int u : adj[v]) {
                if (u != p && !visited[u]) trace(u, v, take[u] >= notake[u]);
            }
        }
    };
    
    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            trace(i, -1, take[i] >= notake[i]);
        }
    }
    
    return result;
}

// =================== SOLVE BIPARTITE ===================

vector<int> solveBipartite(const vector<int>& vertices, const vector<vector<int>>& globalAdj, const vector<int>& colors) {
    int n = vertices.size();
    if (n == 0) return {};
    
    map<int, int> toLocal;
    for (int i = 0; i < n; i++) toLocal[vertices[i]] = i;
    
    // Build bipartite graph: left = color 0, right = color 1
    vector<int> leftIds, rightIds;
    map<int, int> toLeft, toRight;
    for (int i = 0; i < n; i++) {
        if (colors[i] == 0) { toLeft[i] = leftIds.size(); leftIds.push_back(i); }
        else { toRight[i] = rightIds.size(); rightIds.push_back(i); }
    }
    
    int nL = leftIds.size(), nR = rightIds.size();
    HopcroftKarp hk(nL, nR);
    
    for (int li = 0; li < nL; li++) {
        int v = leftIds[li];
        for (int u : globalAdj[vertices[v]]) {
            auto it = toLocal.find(u);
            if (it == toLocal.end()) continue;
            int local = it->second;
            if (colors[local] == 1) {
                hk.adj[li].push_back(toRight[local]);
            }
        }
    }
    
    hk.maxMatching();
    
    // König's theorem: find minimum vertex cover
    // BFS from unmatched left vertices
    vector<bool> reachableLeft(nL, false), reachableRight(nR, false);
    queue<int> q;
    
    for (int l = 0; l < nL; l++) {
        if (hk.pairL[l] == -1) {
            reachableLeft[l] = true;
            q.push(l);
        }
    }
    
    while (!q.empty()) {
        int l = q.front(); q.pop();
        // From left, follow non-matching edges to right
        for (int r : hk.adj[l]) {
            if (!reachableRight[r]) {
                reachableRight[r] = true;
                // From right, follow matching edge back to left
                int pl = hk.pairR[r];
                if (pl != -1 && !reachableLeft[pl]) {
                    reachableLeft[pl] = true;
                    q.push(pl);
                }
            }
        }
    }
    
    // Minimum vertex cover = (unreachable left) ∪ (reachable right)
    // Maximum independent set = V \ minimum vertex cover
    vector<int> result;
    for (int l = 0; l < nL; l++) {
        if (!reachableLeft[l]) {
            // In vertex cover → NOT in MIS
        } else {
            result.push_back(vertices[leftIds[l]]);
        }
    }
    for (int r = 0; r < nR; r++) {
        if (reachableRight[r]) {
            // In vertex cover → NOT in MIS
        } else {
            result.push_back(vertices[rightIds[r]]);
        }
    }
    
    return result;
}

// =================== BRANCH AND BOUND (BITMASK) ===================

// For n <= 64
vector<int> solveBB(const vector<int>& vertices, const vector<vector<int>>& globalAdj) {
    int n = vertices.size();
    if (n == 0) return {};
    if (n == 1) return {vertices[0]};
    
    // Build adjacency bitmasks
    map<int, int> toLocal;
    for (int i = 0; i < n; i++) toLocal[vertices[i]] = i;
    
    vector<uint64_t> nbr(n, 0);
    for (int i = 0; i < n; i++) {
        for (int u : globalAdj[vertices[i]]) {
            auto it = toLocal.find(u);
            if (it != toLocal.end()) nbr[i] |= (1ULL << it->second);
        }
    }
    
    vector<int> best_set;
    int best_size = 0;
    vector<int> current;
    
    function<void(uint64_t, int)> solve = [&](uint64_t R, int sz) {
        if (R == 0) {
            if (sz > best_size) {
                best_size = sz;
                best_set = current;
            }
            return;
        }
        
        int cnt = __popcnt64(R);
        if (sz + cnt <= best_size) return;
        
        // Greedy matching upper bound
        uint64_t Rcopy = R;
        int matching = 0;
        while (Rcopy) {
            int v = __builtin_ctzll(Rcopy);
            Rcopy ^= (1ULL << v);
            uint64_t nbs = nbr[v] & R;
            if (nbs) {
                int u = __builtin_ctzll(nbs);
                matching++;
                Rcopy ^= (1ULL << u);
            }
        }
        if (sz + cnt - matching <= best_size) return;
        
        // Find vertex with max degree in R
        int best_v = -1, best_deg = -1;
        uint64_t Rtmp = R;
        while (Rtmp) {
            int v = __builtin_ctzll(Rtmp);
            Rtmp ^= (1ULL << v);
            int deg = __popcnt64(nbr[v] & R);
            if (deg > best_deg) {
                best_deg = deg;
                best_v = v;
            }
        }
        
        // Handle degree-0 vertices immediately
        if (best_deg == 0) {
            // All remaining vertices are isolated
            Rtmp = R;
            while (Rtmp) {
                int v = __builtin_ctzll(Rtmp);
                Rtmp ^= (1ULL << v);
                current.push_back(vertices[v]);
            }
            if (sz + cnt > best_size) {
                best_size = sz + cnt;
                best_set = current;
            }
            // Pop them
            for (int i = 0; i < cnt; i++) current.pop_back();
            return;
        }
        
        int v = best_v;
        
        // Branch 1: include v
        {
            uint64_t R1 = R & ~(1ULL << v) & ~nbr[v];
            current.push_back(vertices[v]);
            solve(R1, sz + 1);
            current.pop_back();
        }
        
        // Branch 2: exclude v
        {
            uint64_t R2 = R & ~(1ULL << v);
            solve(R2, sz);
        }
    };
    
    uint64_t R = (n == 64) ? ~0ULL : ((1ULL << n) - 1);
    solve(R, 0);
    
    return best_set;
}
