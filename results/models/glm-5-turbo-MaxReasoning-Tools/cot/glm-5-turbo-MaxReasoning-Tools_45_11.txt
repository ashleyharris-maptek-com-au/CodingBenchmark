import struct

def word(val):
    return struct.pack('<I', val & 0xFFFFFFFF)

def instruction(opcode, *operands):
    word_count = 1 + len(operands)
    first_word = (word_count << 16) | opcode
    return word(first_word) + b''.join(word(op) for op in operands)

def string_literal(s):
    # Encode string as SPIR-V literal (null-terminated, padded to 4-byte boundary)
    encoded = s.encode('utf-8') + b'\x00'
    padding = (4 - len(encoded) % 4) % 4
    encoded += b'\x00' * padding
    return encoded

        wc = 1 + len(operands)

        for op in operands:

        """Emit a string literal as words (null-terminated, padded)."""
        data = s.encode('utf-8') + b'\x00'
