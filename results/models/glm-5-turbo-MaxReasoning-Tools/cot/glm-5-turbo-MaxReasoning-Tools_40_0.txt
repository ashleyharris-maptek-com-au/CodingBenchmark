vector<unordered_set<int>> adj_set;
long long best_value = 0;
vector<int> best_assignment;
vector<int> current_assignment;

void solve(vector<bool>& avail, long long current_value) {
    // Simple upper bound
    long long remaining = 0;
    for (int i = 0; i < n; i++) {
        if (avail[i]) remaining += c[i];
    }
    if (current_value + remaining <= best_value) return;
    
    // Find vertex with min degree among available
    int best_v = -1;
    int min_deg = INT_MAX;
    long long max_weight = LLONG_MIN;
    for (int i = 0; i < n; i++) {
        if (avail[i]) {
            int deg = 0;
            for (int nb : adj_set[i]) {
                if (avail[nb]) deg++;
            }
            if (deg < min_deg || (deg == min_deg && c[i] > max_weight)) {
                min_deg = deg;
                best_v = i;
                max_weight = c[i];
            }
        }
    }
    
    if (best_v == -1) {
        // No more vertices
        if (current_value > best_value) {
            best_value = current_value;
            best_assignment = current_assignment;
        }
        return;

        // Include this vertex (no neighbors to worry about)
        current_assignment[best_v] = 1;
        avail[best_v] = false;

        avail[best_v] = true;
        current_assignment[best_v] = 0;
        return;

    current_assignment[best_v] = 1;
    avail[best_v] = false;
    vector<int> removed_neighbors;

void solve(int idx, long long current_value, vector<bool>& avail, vector<int>& current) {
    // Simple upper bound: current + sum of available
    long long ub = current_value;

    int best_v = -1;

        if (!avail[i]) continue;
        int deg = 0;
        for (int nb : adj[i]) {
            if (avail[nb]) deg++;
        }
        if (deg < min_deg || (deg == min_deg && c[i] > max_c)) {

            best_v = i;

        }
