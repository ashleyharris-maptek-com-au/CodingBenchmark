```cpp
        int first_pos;
        unordered_map<char, int> next;
```

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <unordered_map>
#include <algorithm>
#include <numeric>
#include <cstring>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    if (!(cin >> n)) {
        cout << "\n";
        return 0;
    }
    cin.ignore();
    
    if (n <= 0) {
        cout << "\n";
        return 0;
    }
    
    vector<string> strs(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strs[i]);
    }
    
    if (n == 1) {
        cout << strs[0] << "\n";
        return 0;
    }
    
    // Find the shortest string
    int min_idx = 0;
    for (int i = 1; i < n; i++) {
        if (strs[i].size() < strs[min_idx].size()) {
            min_idx = i;
        }
    }
    
    const string& base = strs[min_idx];
    int base_len = (int)base.size();
    
    if (base_len == 0) {
        cout << "\n";
        return 0;
    }
    
    // Build Suffix Automaton on the shortest string
    struct State {
        int len;
        int link;
        int first_pos;
        unordered_map<char, int> next;
    };
    
    vector<State> st;
    st.reserve(2 * (size_t)base_len);
    st.push_back({0, -1, -1, {}});
    int last = 0;
    
    for (int i = 0; i < base_len; i++) {
        char c = base[i];
        int cur = (int)st.size();
        st.push_back({st[last].len + 1, -1, i, {}});
        int p = last;
        while (p != -1 && st[p].next.find(c) == st[p].next.end()) {
            st[p].next[c] = cur;
            p = st[p].link;
        }
        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back({st[p].len + 1, st[q].link, st[q].first_pos, std::move(st[q].next)});
                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
    
    int num_states = (int)st.size();
    
    // Precompute order (states sorted by decreasing len)
    vector<int> order(num_states);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        return st[a].len > st[b].len;
    });
    
    // Initialize min_dp: for the base string, all substrings are in the SAM
    // min_dp[v] starts as st[v].len (the max possible match for state v)
    vector<int> min_dp(num_states);
    for (int i = 0; i < num_states; i++) {
        min_dp[i] = st[i].len;
    }
    
    // Allocate dp array once
    vector<int> dp(num_states, 0);
    
    // Process each other string
    for (int si = 0; si < n; si++) {
        if (si == min_idx) continue;
        
        const string& t = strs[si];
        if (t.empty()) {
            cout << "\n";
            return 0;
        }
        
        fill(dp.begin(), dp.end(), 0);
        
        int v = 0, l = 0;
        int t_len = (int)t.size();
        for (int i = 0; i < t_len; i++) {
            char c = t[i];
            auto it = st[v].next.find(c);
            while (it == st[v].next.end() && v != 0) {
                v = st[v].link;
                l = st[v].len;
                it = st[v].next.find(c);
            }
            if (it != st[v].next.end()) {
                v = it->second;
                l++;
            }
            if (l > dp[v]) dp[v] = l;
        }
        
        // Propagate and update min_dp (combined, process in decreasing len order)
        for (int j = 0; j < num_states; j++) {
            int u = order[j];
            if (st[u].link != -1 && dp[u] > 0) {
                int p = st[u].link;
                int val = (dp[u] < st[p].len) ? dp[u] : st[p].len;
                if (val > dp[p]) dp[p] = val;
            }
            if (dp[u] < min_dp[u]) min_dp[u] = dp[u];
        }
    }
    
    // Find best state
    int best_state = 0;
    int best_len = 0;
    for (int i = 1; i < num_states; i++) {
        if (min_dp[i] > best_len) {
            best_len = min_dp[i];
            best_state = i;
        }
    }
    
    if (best_len == 0) {
        cout << "\n";
    } else {
        int start = st[best_state].first_pos - best_len + 1;
        cout << base.substr(start, best_len) << "\n";
    }
    
    return 0;
}
```
