long ComputeDelta(int i, int j)
{
    int a = perm[i], b = perm[j];
    long delta = 0;
    
    for (int k = 0; k < N; k++)
    {
        if (k == i || k == j) continue;
        int pk = perm[k];
        delta += (long)flow[i][k] * ((long)dist[b][pk] - dist[a][pk]);
        delta += (long)flow[j][k] * ((long)dist[a][pk] - dist[b][pk]);
        delta += (long)flow[k][i] * ((long)dist[pk][b] - dist[pk][a]);
        delta += (long)flow[k][j] * ((long)dist[pk][a] - dist[pk][b]);
    }

    delta += (long)flow[i][j] * ((long)dist[b][a] - dist[a][b]);
    // Term (j,i)
    delta += (long)flow[j][i] * ((long)dist[a][b] - dist[b][a]);
    
    return delta;

using System;
using System.Diagnostics;
using System.IO;
using System.Text;

class Program
{
    static int N;
    static int[][] F, D;
    static int[] p;
    static long cost;
    static readonly Random rng = new Random(42);
    const long TIME_LIMIT = 270000; // 4.5 minutes in ms

    static void Main()
    {
        var sw = Stopwatch.StartNew();
        ReadInput();
        
        if (N <= 1)
        {
            Console.WriteLine(N == 1 ? 0 : 0);
            if (N == 1) Console.WriteLine(0);
            return;
        }
        
        // Initialize with best greedy
        long bestCost = long.MaxValue;
        int[] bestP = null;
        
        // Greedy: high flow -> low distance
        Greedy(false);
        long c = FullCost();
        if (c < bestCost) { bestCost = c; bestP = (int[])p.Clone(); }
        
        // Greedy: high flow -> high distance
        Greedy(true);
        c = FullCost();
        if (c < bestCost) { bestCost = c; bestP = (int[])p.Clone(); }
        
        // Random starts
        for (int r = 0; r < 3 && sw.ElapsedMilliseconds < 5000; r++)
        {
            RandomInit();
            c = FullCost();
            if (c < bestCost) { bestCost = c; bestP = (int[])p.Clone(); }
        }
        
        p = bestP;
        cost = bestCost;
        
        // Local search + SA
        LocalSearch(sw);
        
        // Output
        cost = FullCost();
        var sb = new StringBuilder();
        sb.AppendLine(cost.ToString());
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(p[i]);
        }
        Console.Write(sb.ToString());
    }
    
    static void ReadInput()
    {
        // Read all at once for speed
        var input = new StreamReader(Console.OpenStandardInput()).ReadToEnd();
        var tokens = new StringBuilder();
        foreach (char ch in input)
        {
            if (ch >= '0' && ch <= '9' || ch == '-')
                tokens.Append(ch);
            else if (tokens.Length > 0)
                tokens.Append(' ');
        }
        var parts = tokens.ToString().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        N = int.Parse(parts[idx++]);
        F = new int[N][];
        D = new int[N][];
        for (int i = 0; i < N; i++)
        {
            F[i] = new int[N];
            for (int j = 0; j < N; j++)
                F[i][j] = int.Parse(parts[idx++]);
        }
        for (int i = 0; i < N; i++)
        {
            D[i] = new int[N];
            for (int j = 0; j < N; j++)
                D[i][j] = int.Parse(parts[idx++]);
        }
    }
    
    static void Greedy(bool reverseDist)
    {
        p = new int[N];
        var fSum = new long[N];
        var dSum = new long[N];
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
            {
                fSum[i] += F[i][j];
                dSum[i] += D[i][j];
            }
        var fi = new int[N], li = new int[N];
        for (int i = 0; i < N; i++) { fi[i] = i; li[i] = i; }
        Array.Sort(fi, (a, b) => fSum[b].CompareTo(fSum[a]));
        Array.Sort(li, (a, b) => reverseDist ? dSum[b].CompareTo(dSum[a]) : dSum[a].CompareTo(dSum[b]));
        for (int i = 0; i < N; i++)
            p[fi[i]] = li[i];
    }
    
    static void RandomInit()
    {
        p = new int[N];
        for (int i = 0; i < N; i++) p[i] = i;
        // Fisher-Yates shuffle
        for (int i = N - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            int t = p[i]; p[i] = p[j]; p[j] = t;
        }
    }
    
    static long FullCost()
    {
        long c = 0;
        for (int i = 0; i < N; i++)
        {
            int pi = p[i];
            for (int j = 0; j < N; j++)
                c += (long)F[i][j] * D[pi][p[j]];
        }
        return c;
    }
    
    static long SwapDelta(int i, int j)
    {
        int a = p[i], b = p[j];
        long delta = 0;
        for (int k = 0; k < N; k++)
        {
            if (k == i || k == j) continue;
            int pk = p[k];
            int d1 = D[b][pk] - D[a][pk];
            int d2 = D[pk][b] - D[pk][a];
            delta += (long)F[i][k] * d1;
            delta -= (long)F[j][k] * d1;
            delta += (long)F[k][i] * d2;
            delta -= (long)F[k][j] * d2;
        }
        delta += (long)F[i][j] * ((long)D[b][a] - D[a][b]);
        delta += (long)F[j][i] * ((long)D[a][b] - D[b][a]);
        return delta;
    }
    
    static void LocalSearch(Stopwatch sw)
    {
        double remaining = TIME_LIMIT - sw.ElapsedMilliseconds;
        if (remaining < 500) return;
        
        // Phase 1: Fast greedy 2-opt for small N
        if (N <= 150)
        {
            bool improved = true;
            while (improved && sw.ElapsedMilliseconds < TIME_LIMIT * 0.15)
            {
                improved = false;
                for (int i = 0; i < N && !improved; i++)
                    for (int j = i + 1; j < N && !improved; j++)
                    {
                        long d = SwapDelta(i, j);
                        if (d < 0)
                        {
                            int t = p[i]; p[i] = p[j]; p[j] = t;
                            cost += d;
                            improved = true;
                        }
                    }
            }
        }
        
        remaining = TIME_LIMIT - sw.ElapsedMilliseconds;
        if (remaining < 500) return;
        
        // Phase 2: Simulated annealing
        // Estimate ops/ms
        long t0 = sw.ElapsedMilliseconds;
        int estIter = Math.Min(50000, N * N);
        for (int t = 0; t < estIter; t++)
        {
            int i = t % N;
            int j = (t * 7 + 3) % (N > 1 ? N - 1 : 1);
            if (j >= i) j++;
            SwapDelta(i, j);
        }
        long t1 = sw.ElapsedMilliseconds;
        double opsPerMs = estIter / Math.Max(1.0, t1 - t0);
        
        remaining = TIME_LIMIT - sw.ElapsedMilliseconds;
        long totalOps = (long)(opsPerMs * remaining * 0.95);
        
        // Estimate temperature
        long sumAbs = 0;
        int samples = Math.Min(2000, N * (N - 1) / 2);
        for (int s = 0; s < samples; s++)
        {
            int i = rng.Next(N);
            int j = rng.Next(N > 1 ? N - 1 : 1);
            if (j >= i) j++;
            sumAbs += Math.Abs(SwapDelta(i, j));
        }
        double avgAbs = sumAbs / (double)samples;
        double T0 = avgAbs * 1.5;
        double Tf = avgAbs * 1e-5;
        
        long ops = 0;
        while (ops < totalOps)
        {
            double progress = (double)ops / totalOps;
            double T = T0 * Math.Pow(Tf / T0, progress);
            
            long batch = Math.Min(totalOps - ops, Math.Max(2000, totalOps / 100));
            
            for (long b = 0; b < batch; b++)
            {
                int i = rng.Next(N);
                int j = rng.Next(N > 1 ? N - 1 : 1);
                if (j >= i) j++;
                
                long d = SwapDelta(i, j);
                
                if (d < 0)
                {
                    int t = p[i]; p[i] = p[j]; p[j] = t;
                    cost += d;
                }
                else if (T > 1e-12 && rng.NextDouble() < Math.Exp(-(double)d / T))
                {
                    int t = p[i]; p[i] = p[j]; p[j] = t;
                    cost += d;
                }
                ops++;
            }
        }
    }
}
