def _valid_median(values, valids):
    vv = [v for v, valid in zip(values, valids) if valid]
    if not vv:
        return None
    vv.sort()
    n = len(vv)
    if n == 1:
        return vv[0]
    if n == 2:
        return (vv[0] + vv[1]) / 2.0
    return vv[1]  # median of 3

def _valid_mean(values, valids):
    vv = [v for v, valid in zip(values, valids) if valid]
    if not vv:
        return None
    return sum(vv) / len(vv)

def _low_pass(val, prev, alpha=0.3):
    if prev is None:
        return val
    return alpha * val + (1 - alpha) * prev

def _median3(a, b, c):
    """Return median of three values."""
    if a <= b <= c or c <= b <= a:
        return b
    if b <= a <= c or c <= a <= b:
        return a
    return c


def _valid_values(vals, valids):
    """Return list of (value, index) for valid sensors."""
    return [(v, i) for i, (v, valid) in enumerate(zip(vals, valids)) if valid]


def _detect_stuck(hist, valid):
    """Detect if a sensor is stuck (low variance)."""
    if not valid or len(hist) < 10:
        return False
    values = list(hist)
    spread = max(values) - min(values)
    return spread < 0.05  # Less than 0.05 dots variation in 30 samples


def _fuse_sensors(state_key, hist_key, stuck_key, dots_list, valid_list, prev_filt):
    """Fuse redundant sensor data with stuck detection."""
    s = _state
    
    # Update history
    for i, (val, valid) in enumerate(zip(dots_list, valid_list)):
        if valid:
            s[hist_key][i].append(val)
    
    # Detect stuck sensors
    for i in range(3):
        s[stuck_key][i] = _detect_stuck(s[hist_key][i], valid_list[i])
    
    # Get non-stuck valid values
    candidates = []
    for i, (val, valid) in enumerate(zip(dots_list, valid_list)):
        if valid and not s[stuck_key][i]:
            candidates.append(val)
    
    if not candidates:
        # All stuck or invalid - fall back to any valid
        any_valid = [(v, i) for i, (v, valid) in enumerate(zip(dots_list, valid_list)) if valid]
        if any_valid:
            candidates = [v for v, i in any_valid]
        else:
            return prev_filt
    
    # Use median
    if len(candidates) == 1:
        result = candidates[0]
    elif len(candidates) == 2:
        result = (candidates[0] + candidates[1]) / 2.0
    else:
        result = _median3(candidates[0], candidates[1], candidates[2])
    
    # Low-pass filter
    alpha = 0.4
    if prev_filt is None:
        return result
    return alpha * result + (1 - alpha) * prev_filt


def _clamp(val, lo, hi):
    return max(lo, min(hi, val))


def _get_radio_alt(state, baro_alt, gs_dots, dme):
    """Get radio alt, with DME-based fallback."""
    s = _state
    ra = state.get('radio_alt_ft')
    
    ra_valid = ra is not None and 0 <= ra < 10000

    if ra_valid and baro_alt is not None and baro_alt > 0:

        if s['field_elev'] is None:
            s['field_elev'] = est
        else:
            s['field_elev'] = 0.95 * s['field_elev'] + 0.05 * est
    
    if ra_valid:
        return ra, True
    
    # Fallback 1: DME + glideslope angle
    if dme is not None and dme > 0.1 and dme < 15:
        if gs_dots is not None:
            gs_angle = 3.0 + gs_dots * 0.7

            gs_angle = 3.0

    # Fallback 2: baro alt - field elevation

        est_agl = baro_alt - s['field_elev']
        return max(0, est_agl), est_agl > 0
    
    return 0, False
