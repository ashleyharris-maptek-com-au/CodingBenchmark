    vector<vector<int>> ext_par(full + 1, vector<int>(n, -1));
    
    vector<vector<int>> term_pred(K);
    
    for (int i = 0; i < K; i++) {
        int mask = 1 << i;
        auto dist = dijkstra(terminals[i], term_pred[i]);
        for (int v = 0; v < n; v++) {
            dp[mask][v] = dist[v];
        }
    }
    
    for (int mask = 1; mask <= full; mask++) {
        int pc = __popcount(mask);
        if (pc <= 1) continue;
        
        // Split step
        int smallest = __builtin_ctz(mask);
        for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
            if (!(sub & (1 << smallest))) continue;
            int rest = mask ^ sub;
            for (int v = 0; v < n; v++) {
                if (dp[sub][v] < INF && dp[rest][v] < INF) {
                    ll val = dp[sub][v] + dp[rest][v];
                    if (val < dp[mask][v]) {
                        dp[mask][v] = val;
                        ptype[mask][v] = 1;
                        split_sub[mask][v] = sub;
                    }
                }
            }
        }
        
        // Relaxation step
        vector<ll> ndist = dp[mask];
        vector<int> rpred(n, -1);
        priority_queue<pli, vector<pli>, greater<pli>> pq;
        for (int v = 0; v < n; v++) {
            if (ndist[v] < INF) pq.push({ndist[v], v});
        }
        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d > ndist[u]) continue;
            for (auto& [v, w] : adj[u]) {
                if (ndist[v] > d + w) {
                    ndist[v] = d + w;
                    rpred[v] = u;
                    pq.push({ndist[v], v});
                }
            }
        }
        for (int v = 0; v < n; v++) {
            if (ndist[v] < dp[mask][v]) {
                dp[mask][v] = ndist[v];
                ptype[mask][v] = 2;
                ext_par[mask][v] = rpred[v];
            }
        }
    }
    
    ll best = INF;
    int best_v = -1;
    for (int v = 0; v < n; v++) {
        if (dp[full][v] < best) {
            best = dp[full][v];
            best_v = v;
        }
    }
    
    if (K == 1) {
        printf("0 0\n");
        return;
    }
    
    set<pair<int,int>> edges;
    function<void(int, int)> reconstruct = [&](int mask, int v) {
        if (__popcount(mask) == 1) {
            int ti = __builtin_ctz(mask);
            int cur = v;
            while (cur != terminals[ti]) {
                int p = term_pred[ti][cur];
                if (p == -1) break;
                edges.insert({min(p, cur), max(p, cur)});
                cur = p;
            }
        } else if (ptype[mask][v] == 1) {
            int sub = split_sub[mask][v];
            reconstruct(sub, v);
            reconstruct(mask ^ sub, v);
        } else {
            int u = ext_par[mask][v];
            edges.insert({min(u, v), max(u, v)});
            reconstruct(mask, u);
        }
    };
    
    reconstruct(full, best_v);
    
    ll total_weight = 0;
    for (auto& [u, v] : edges) {
        total_weight += edge_weight(u, v);
    }
    
    printf("%lld %d\n", total_weight, (int)edges.size());
    for (auto& [u, v] : edges) {
        printf("%d %d\n", u, v);
    }
}

void solve_heuristic(vector<int>& terminals) {
    // Distance network + MST approach (2-approximation)
    
    // Compute shortest paths from each terminal
    vector<vector<ll>> dist(K, vector<ll>(n, INF));
    vector<vector<int>> pred(K, vector<int>(n, -1));
    for (int i = 0; i < K; i++) {
        dist[i] = dijkstra(terminals[i], pred[i]);
    }
    
    if (K == 1) {
        printf("0 0\n");
        return;
    }
    
    // Build distance network: edges (i, j, dist[i][terminals[j]])
    struct Edge { int u, v; ll w; };
    vector<Edge> edges;
    for (int i = 0; i < K; i++) {
        for (int j = i + 1; j < K; j++) {
            edges.push_back({i, j, dist[i][terminals[j]]});
        }
    }
    
    // Kruskal's MST
    sort(edges.begin(), edges.end(), [](const Edge& a, const Edge& b) { return a.w < b.w; });
    
    vector<int> parent(K), rnk(K, 0);
    iota(parent.begin(), parent.end(), 0);
    function<int(int)> find = [&](int x) { return parent[x] == x ? x : parent[x] = find(parent[x]); };
    auto unite = [&](int x, int y) {
        x = find(x); y = find(y);
        if (x == y) return false;
        if (rnk[x] < rnk[y]) swap(x, y);
        parent[y] = x;
        if (rnk[x] == rnk[y]) rnk[x]++;
        return true;
    };
    
    // Build adjacency list for MST
    vector<vector<pair<int, int>>> mst_adj(K); // (terminal index, path via terminal i)
    // Actually, I need to track which MST edges are selected and their corresponding actual paths
    
    // Set of edges in the subgraph (union of all shortest paths)
    set<pair<int,int>> sub_edges;
    
    for (auto& e : edges) {
        if (unite(e.u, e.v)) {
            // Add the actual shortest path from terminals[e.u] to terminals[e.v]
            int src = terminals[e.u];
            int dst = terminals[e.v];
            int cur = dst;
            while (cur != src && pred[e.u][cur] != -1) {
                int p = pred[e.u][cur];
                sub_edges.insert({min(p, cur), max(p, cur)});
                cur = p;
            }
        }
    }
    
    // Build adjacency list for the subgraph
    vector<vector<pair<int, ll>>> sub_adj(n);
    for (auto& [u, v] : sub_edges) {
        ll w = edge_weight(u, v);
        sub_adj[u].push_back({v, w});
        sub_adj[v].push_back({u, w});
    }
    
    // BFS to find spanning tree
    vector<bool> visited(n, false);
    queue<int> bfs_q;
    set<pair<int,int>> tree_edges;
    bfs_q.push(terminals[0]);
    visited[terminals[0]] = true;
    
    while (!bfs_q.empty()) {
        int u = bfs_q.front(); bfs_q.pop();
        for (auto& [v, w] : sub_adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                tree_edges.insert({min(u, v), max(u, v)});
                bfs_q.push(v);
            }
        }
    }
    
    ll total_weight = 0;
    for (auto& [u, v] : tree_edges) {
        total_weight += edge_weight(u, v);
    }
    
    printf("%lld %d\n", total_weight, (int)tree_edges.size());
    for (auto& [u, v] : tree_edges) {
        printf("%d %d\n", u, v);
    }
}
