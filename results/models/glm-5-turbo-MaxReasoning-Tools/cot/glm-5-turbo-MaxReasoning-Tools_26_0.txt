use std::io::{Read, Write, BufReader, BufWriter};

type Point = [f32; 3];

fn dist_sq(a: &[f32; 3], b: &[f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

fn nearest_centroid(point: &Point, centroids: &[Point]) -> usize {
    let mut best = 0;
    let mut best_dist = dist_sq(point, &centroids[0]);
    for (i, c) in centroids.iter().enumerate().skip(1) {
        let d = dist_sq(point, c);
        if d < best_dist {
            best_dist = d;
            best = i;
        }
    }
    best
}

// Simple RNG
struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Rng { state: seed }
    }
    
    fn next_u64(&mut self) -> u64 {
        self.state = self.state.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        self.state
    }
    
    fn next_f64(&mut self) -> f64 {
        (self.next_u64() >> 11) as f64 / (1u64 << 53) as f64
    }
    
    fn next_usize(&mut self, max: usize) -> usize {
        (self.next_f64() * max as f64) as usize
    }
}

fn kmeans_pp_init(sample: &[Point], k: usize, rng: &mut Rng) -> Vec<Point> {
    let mut centroids = Vec::with_capacity(k);
    // Pick first centroid randomly
    let first = rng.next_usize(sample.len());
    centroids.push(sample[first]);
    
    for _ in 1..k {
        let mut min_dists: Vec<f32> = Vec::with_capacity(sample.len());
        let mut total = 0.0f32;
        for p in sample.iter() {
            let mut min_d = f32::MAX;
            for c in centroids.iter() {
                let d = dist_sq(p, c);
                if d < min_d {
                    min_d = d;
                }
            }
            min_dists.push(min_d);
            total += min_d;
        }
        
        if total <= 0.0 {
            // All points are at centroids, pick random
            centroids.push(sample[rng.next_usize(sample.len())].clone());
            continue;
        }
        
        let mut r = rng.next_f64() * total as f64;
        let mut chosen = sample.len() - 1;
        for (i, d) in min_dists.iter().enumerate() {
            r -= *d as f64;
            if r <= 0.0 {
                chosen = i;
                break;
            }
        }
        centroids.push(sample[chosen].clone());
    }
    centroids
}

// In-memory k-means
fn in_memory_kmeans<R: Read, W: Write>(
    reader: &mut BufReader<R>,
    writer: &mut BufWriter<W>,
    num_points: usize,
    num_clusters: usize,
) {
    // Read all points
    let mut points = vec![[0.0f32; 3]; num_points];
    let mut buf = vec![0u8; num_points * 12];
    reader.read_exact(&mut buf).unwrap();
    for i in 0..num_points {
        let base = i * 12;
        points[i][0] = f32::from_le_bytes(buf[base..base + 4].try_into().unwrap());
        points[i][1] = f32::from_le_bytes(buf[base + 4..base + 8].try_into().unwrap());
        points[i][2] = f32::from_le_bytes(buf[base + 8..base + 12].try_into().unwrap());
    }
    drop(buf); // Free buffer memory
    
    // Determine number of iterations based on data size
    let max_iter = if num_points < 1_000_000 { 30 }
                   else if num_points < 10_000_000 { 20 }
                   else { 15 };
    
    // Initialize centroids
    let sample_size = (num_points.min(100_000)).max(num_clusters * 10);
    let mut rng = Rng::new(42);
    
    // Create sample indices
    let mut sample: Vec<Point> = Vec::with_capacity(sample_size);
    let step = num_points / sample_size;
    for i in 0..sample_size {
        sample.push(points[i * step]);
    }
    
    let mut centroids = kmeans_pp_init(&sample, num_clusters, &mut rng);
    drop(sample);
    
    // Run k-means iterations
    let mut sums = vec![[0.0f64; 3]; num_clusters];
    let mut counts = vec![0usize; num_clusters];
    let mut assignments = vec![0usize; num_points];
    
    for _iter in 0..max_iter {
        // Reset accumulators
        for s in sums.iter_mut() {
            s[0] = 0.0; s[1] = 0.0; s[2] = 0.0;
        }
        for c in counts.iter_mut() {
            *c = 0;
        }
        
        // Assign points to centroids
        for (i, p) in points.iter().enumerate() {
            let j = nearest_centroid(p, &centroids);
            assignments[i] = j;
            sums[j][0] += p[0] as f64;
            sums[j][1] += p[1] as f64;
            sums[j][2] += p[2] as f64;
            counts[j] += 1;
        }
        
        // Update centroids
        let mut empty_clusters = Vec::new();
        for j in 0..num_clusters {
            if counts[j] > 0 {
                centroids[j][0] = (sums[j][0] / counts[j] as f64) as f32;

use std::io::{Read, Write, BufReader, BufWriter};

const INMEM_THRESHOLD: usize = 80_000_000; // ~960MB for points

type Point = [f32; 3];

#[inline]
fn dist_sq(a: &[f32; 3], b: &[f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

#[inline]
fn nearest_centroid(point: &Point, centroids: &[Point]) -> usize {
    let mut best = 0;
    let mut best_dist = dist_sq(point, &centroids[0]);
    for (i, c) in centroids.iter().enumerate().skip(1) {
        let d = dist_sq(point, c);
        if d < best_dist {
            best_dist = d;
            best = i;
        }
    }
    best
}

struct Rng(u64);

impl Rng {
    fn new(seed: u64) -> Self { Rng(seed) }
    fn next(&mut self) -> u64 {
        self.0 ^= self.0 << 13;
        self.0 ^= self.0 >> 7;
        self.0 ^= self.0 << 17;
        self.0
    }
    fn gen_range(&mut self, max: usize) -> usize {
        (self.next() as usize % max)
    }
    fn gen_f64(&mut self) -> f64 {
        self.next() as f64 / (u64::MAX as f64 + 1.0)
    }
}

fn kmeans_pp_init(sample: &[Point], k: usize, rng: &mut Rng) -> Vec<Point> {
    let n = sample.len();
    let mut centroids = Vec::with_capacity(k);
    
    if k >= n {
        for p in sample.iter() {
            centroids.push(*p);
        }
        // Fill remaining with duplicates
        while centroids.len() < k {
            centroids.push(sample[0]);
        }
        return centroids;
    }
    
    // First centroid: random
    centroids.push(sample[rng.gen_range(n)]);
    
    for _ in 1..k {
        // Compute min distances
        let mut dists = vec![0.0f32; n];
        let mut total = 0.0f64;
        for (i, p) in sample.iter().enumerate() {
            let mut min_d = f32::MAX;
            for c in centroids.iter() {
                let d = dist_sq(p, c);
                if d < min_d { min_d = d; }
            }
            dists[i] = min_d;
            total += min_d as f64;
        }
        
        if total <= 0.0 {
            centroids.push(sample[rng.gen_range(n)]);
            continue;
        }
        
        let mut r = rng.gen_f64() * total;
        let mut chosen = n - 1;
        for (i, d) in dists.iter().enumerate() {
            r -= *d as f64;
            if r <= 0.0 { chosen = i; break; }
        }
        centroids.push(sample[chosen]);
    }
    centroids
}

fn read_batch<R: Read>(reader: &mut R, buf: &mut Vec<u8>, points: &mut Vec<Point>, count: usize) {
    buf.resize(count * 12, 0);
    reader.read_exact(buf).unwrap();
    points.clear();
    points.reserve(count);
    for i in 0..count {
        let base = i * 12;
        let p = [
            f32::from_le_bytes(buf[base..base+4].try_into().unwrap()),
            f32::from_le_bytes(buf[base+4..base+8].try_into().unwrap()),
            f32::from_le_bytes(buf[base+8..base+12].try_into().unwrap()),
        ];
        points.push(p);
    }
}

fn write_assignments_batch<W: Write>(writer: &mut W, assignments: &[usize]) {

        }

            }
            writer.write_all(&out_buf[..(end - offset) * 4]).unwrap();
            offset = end;
        }
    }
    writer.flush().unwrap();

This way the first batch gets processed twice (once for init, once for assignment) but it's a small cost.

Actually, even simpler: just process everything in a single stream, including the first batch. The quality difference is negligible.

use std::io::{Read, Write, BufReader, BufWriter};

type Point = [f32; 3];

const INMEM_THRESHOLD: usize = 80_000_000;
const BATCH_SIZE: usize = 131_072; // 128K points
const KMPP_MAX_K: usize = 2000;
const BUF_SIZE: usize = 8 * 1024 * 1024; // 8MB I/O buffer

fn dist_sq(a: &Point, b: &Point) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

fn nearest_centroid(p: &Point, centroids: &[Point]) -> usize {
    let mut best = 0usize;
    let mut best_d = dist_sq(p, &centroids[0]);
    for i in 1..centroids.len() {
        let d = dist_sq(p, &centroids[i]);
        if d < best_d {
            best_d = d;
            best = i;
        }
    }
    best
}

// Xorshift64 PRNG
struct Rng(u64);

impl Rng {
    fn new(seed: u64) -> Self { Rng(seed) }
    fn next_u64(&mut self) -> u64 {
        self.0 ^= self.0 << 13;
        self.0 ^= self.0 >> 7;
        self.0 ^= self.0 << 17;
        self.0
    }
    fn gen_usize(&mut self, n: usize) -> usize {
        (self.next_u64() as usize) % n
    }
    fn gen_f64(&mut self) -> f64 {
        (self.next_u64() >> 11) as f64 * (1.0 / (1u64 << 53) as f64)
    }
}

fn kmeans_pp(sample: &[Point], k: usize, rng: &mut Rng) -> Vec<Point> {
    let n = sample.len();
    if k >= n {
        let mut c: Vec<Point> = sample.to_vec();
        while c.len() < k { c.push(sample[0]); }
        return c;
    }
    
    let mut centroids = Vec::with_capacity(k);
    let mut min_dist = vec![f32::MAX; n];
    
    // First: random
    let first = rng.gen_usize(n);
    centroids.push(sample[first]);
    for i in 0..n {
        min_dist[i] = dist_sq(&sample[i], &centroids[0]);
    }
    
    for _ in 1..k {
        let total: f64 = min_dist.iter().map(|&d| d as f64).sum();
        if total <= 0.0 {
            centroids.push(sample[rng.gen_usize(n)]);
            continue;
        }
        let mut r = rng.gen_f64() * total;
        let mut chosen = n - 1;
        for i in 0..n {
            r -= min_dist[i] as f64;
            if r <= 0.0 { chosen = i; break; }
        }
        centroids.push(sample[chosen]);
        // Update min distances
        for i in 0..n {
            let d = dist_sq(&sample[i], &centroids[centroids.len() - 1]);
            if d < min_dist[i] {
                min_dist[i] = d;
            }
        }
    }
    
    centroids
}

fn random_init(sample: &[Point], k: usize, rng: &mut Rng) -> Vec<Point> {
    let n = sample.len();
    let mut centroids = Vec::with_capacity(k);
    for _ in 0..k {
        centroids.push(sample[rng.gen_usize(n)]);
    }
    centroids
}

fn parse_header(line: &str) -> (usize, usize) {
    let mut parts = line.trim().split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let k: usize = parts.next().unwrap().parse().unwrap();
    (n, k)
}

fn read_points_batch<R: Read>(reader: &mut R, buf: &mut Vec<u8>, points: &mut Vec<Point>, count: usize) {
    buf.resize(count * 12, 0);
    reader.read_exact(buf).unwrap();
    points.clear();
    if points.capacity() < count {
        points.reserve(count - points.len());
    }
    unsafe { points.set_len(count); }
    for i in 0..count {
        let b = i * 12;
        points[i][0] = f32::from_le_bytes(buf[b..b+4].try_into().unwrap());
        points[i][1] = f32::from_le_bytes(buf[b+4..b+8].try_into().unwrap());
        points[i][2] = f32::from_le_bytes(buf[b+8..b+12].try_into().unwrap());
    }
}

fn write_assignments<W: Write>(writer: &mut W, assignments: &[usize]) {
    for &a in assignments {
        writer.write_all(&(a as u32).to_le_bytes()).unwrap();
    }
}

fn in_memory_kmeans<R: Read, W: Write>(
    reader: &mut BufReader<R>,
    writer: &mut BufWriter<W>,
    n: usize,
    k: usize,
) {
    // Read all points
    let mut buf = vec![0u8; n * 12];
    reader.read_exact(&mut buf).unwrap();
    
    let mut points = vec![[0.0f32; 3]; n];
    for i in 0..n {
        let b = i * 12;
        points[i][0] = f32::from_le_bytes(buf[b..b+4].try_into().unwrap());
        points[i][1] = f32::from_le_bytes(buf[b+4..b+8].try_into().unwrap());
        points[i][2] = f32::from_le_bytes(buf[b+8..b+12].try_into().unwrap());
    }
    drop(buf);
    
    // Init
    let mut rng = Rng::new(42);
    let sample_size = n.min(200_000).max(k * 20).min(n);
    let step = n / sample_size;
    let sample: Vec<Point> = (0..sample_size).map(|i| points[i * step]).collect();
    
    let mut centroids = if k <= KMPP_MAX_K {
        kmeans_pp(&sample, k, &mut rng)
    } else {
        random_init(&sample, k, &mut rng)
    };
    drop(sample);
    
    // K-means iterations
    let max_iter = if n < 1_000_000 { 50 }
                   else if n < 10_000_000 { 30 }
                   else { 20 };
    
    let mut sums = vec![[0.0f64; 3]; k];
    let mut counts = vec![0usize; k];
    let mut assignments = vec![0usize; n];
    
    for iter in 0..max_iter {
        // Reset
        for s in sums.iter_mut() { s[0] = 0.0; s[1] = 0.0; s[2] = 0.0; }
        counts.fill(0);
        
        let mut changes = 0usize;
        for (i, p) in points.iter().enumerate() {
            let j = nearest_centroid(p, &centroids);
            if iter > 0 && j != assignments[i] { changes += 1; }
            assignments[i] = j;
            sums[j][0] += p[0] as f64;
            sums[j][1] += p[1] as f64;
            sums[j][2] += p[2] as f64;
            counts[j] += 1;
        }
        
        // Update centroids
        for j in 0..k {
            if counts[j] > 0 {
                centroids[j][0] = (sums[j][0] / counts[j] as f64) as f32;
                centroids[j][1] = (sums[j][1] / counts[j] as f64) as f32;
                centroids[j][2] = (sums[j][2] / counts[j] as f64) as f32;
            } else {
                // Reinitialize empty cluster
                centroids[j] = points[rng.gen_usize(n)];
            }
        }
        
        // Convergence check
        if iter > 0 && changes < n / 1000 {
            break;
        }
    }
    
    // Write output
    let mut out_buf = vec![0u8; BATCH_SIZE * 4];
    for chunk_start in (0..n).step_by(BATCH_SIZE) {
        let chunk_end = (chunk_start + BATCH_SIZE).min(n);
        for i in chunk_start..chunk_end {
            let off = (i - chunk_start) * 4;
            out_buf[off..off+4].copy_from_slice(&(assignments[i] as u32).to_le_bytes());
        }
        writer.write_all(&out_buf[..(chunk_end - chunk_start) * 4]).unwrap();
    }
    writer.flush().unwrap();
}
