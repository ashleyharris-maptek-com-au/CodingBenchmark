import sys

class DLXNode:
    __slots__ = ('L', 'R', 'U', 'D', 'C', 'row')
    def __init__(self):
        self.L = self.R = self.U = self.D = self
        self.C = None
        self.row = -1

class DLX:
    def __init__(self, num_cols):
        self.nodes = [DLXNode() for _ in range(num_cols + 1)]  # 0 = root
        self.root = self.nodes[0]
        self.columns = self.nodes[1:]  # column headers
        
        # Link columns circularly
        if num_cols > 0:
            self.root.R = self.columns[0]
            self.root.L = self.columns[-1]
            self.columns[0].L = self.root
            self.columns[-1].R = self.root
            for i in range(num_cols):
                if i > 0:
                    self.columns[i].L = self.columns[i-1]
                if i < num_cols - 1:
                    self.columns[i].R = self.columns[i+1]
        
        # Each column points to itself vertically
        for c in self.columns:
            c.C = c
            c.U = c.D = c
    
    def add_row(self, row_id, col_indices):
        """Add a row with 1s in the given column indices (0-based)."""
        first = None
        prev = None
        for ci in col_indices:
            col = self.columns[ci]
            node = DLXNode()
            node.C = col
            node.row = row_id
            self.nodes.append(node)
            
            # Insert into column (above header)
            node.U = col.U
            node.D = col
            col.U.D = node
            col.U = node
            
            if first is None:
                first = node
            if prev is not None:
                node.L = prev
                prev.R = node
            prev = node
        
        if prev and first:
            prev.R = first
            first.L = prev
    
    def cover(self, col):
        """Cover column col."""
        col.R.L = col.L
        col.L.R = col.R
        i = col.D
        while i != col:
            j = i.R
            while j != i:
                j.D.U = j.U
                j.U.D = j.D
                j = j.R
            i = i.D

            while j != i:

                j.U.D = j
                j = j.L
            i = i.U
        col.R.L = col
