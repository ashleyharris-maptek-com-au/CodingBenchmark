                     start_x: usize, start_y: usize, 
                     free_count: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    let mut visited = vec![false; width * height];
    let mut path = Vec::with_capacity(free_count);
    
    let mut cx = start_x;
    let mut cy = start_y;
    visited[start_y * width + start_x] = true;
    path.push((cx, cy));
    
    for step in 1..free_count {
        let mut neighbors = Vec::new();
        for &(dx, dy) in &DIRS {
            let nx = cx as i32 + dx;
            let ny = cy as i32 + dy;
            if nx >= 0 && nx < width as i32 && ny >= 0 && ny < height as i32 {
                let nx = nx as usize;
                let ny = ny as usize;
                let ni = ny * width + nx;
                if grid[ni] == 0 && !visited[ni] {
                    // Count unvisited neighbors of this neighbor
                    let mut degree = 0;
                    for &(ddx, ddy) in &DIRS {
                        let nnx = nx as i32 + ddx;
                        let nny = ny as i32 + ddy;
                        if nnx >= 0 && nnx < width as i32 && nny >= 0 && nny < height as i32 {
                            let nni = nny as usize * width + nnx as usize;
                            if grid[nni] == 0 && !visited[nni] {
                                degree += 1;
                            }
                        }
                    }
                    neighbors.push((degree, nx, ny));
                }
            }
        }
        
        if neighbors.is_empty() {
            return None;
        }
        
        // Sort by degree (Warnsdorff's rule)
        neighbors.sort_by_key(|&(d, _, _)| d);
        
        cx = neighbors[0].1;
        cy = neighbors[0].2;
        visited[cy * width + cx] = true;
        path.push((cx, cy));
    }
    
    // Check cycle requirement
    if require_cycle {
        // Last cell must be adjacent to start
        let last = path.last().unwrap();
        let adjacent = DIRS.iter().any(|&(dx, dy)| {
            let nx = last.0 as i32 + dx;
            let ny = last.1 as i32 + dy;
            nx == start_x as i32 && ny == start_y as i32
        });
        if !adjacent {
            return None;
        }
    }
    
    Some(path)
