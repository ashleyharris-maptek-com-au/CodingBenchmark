```rust
struct SpatialGrid {
    cell_size: f64,
    origin: [f64; 3],
    cells: HashMap<[i64; 3], Vec<usize>>,
    positions: Vec<[f64; 3]>,
}

impl SpatialGrid {
    fn new(positions: Vec<[f64; 3]>, target_cells: usize) -> Self {
        if positions.is_empty() {
            return Self {
                cell_size: 1.0,
                origin: [0.0; 3],
                cells: HashMap::new(),
                positions,
            };
        }
        
        // Compute bounding box
        let mut min = [f64::INFINITY; 3];
        let mut max = [f64::NEG_INFINITY; 3];
        for p in &positions {
            for i in 0..3 {
                min[i] = min[i].min(p[i]);
                max[i] = max[i].max(p[i]);
            }
        }
        
        // Compute cell size
        let extent = [
            (max[0] - min[0]).max(1e-10),
            (max[1] - min[1]).max(1e-10),
            (max[2] - min[2]).max(1e-10),
        ];
        
        let n = positions.len();
        let cells_per_dim = ((target_cells as f64 / 3.0).cbrt()).ceil() as i64;
        let cell_size = extent.iter().map(|&e| e / cells_per_dim as f64).fold(f64::INFINITY, f64::min);
        
        // Build grid
        let mut cells: HashMap<[i64; 3], Vec<usize>> = HashMap::new();
        for (idx, p) in positions.iter().enumerate() {
            let cell = [
                ((p[0] - min[0]) / cell_size).floor() as i64,
                ((p[1] - min[1]) / cell_size).floor() as i64,
                ((p[2] - min[2]) / cell_size).floor() as i64,
            ];
            cells.entry(cell).or_default().push(idx);
        }
        
        Self {
            cell_size,
            origin: min,
            cells,
            positions,
        }
    }
    
    fn query(&self, center: [f64; 3], radius: f64) -> Vec<usize> {
        let r_cells = (radius / self.cell_size).ceil() as i64;
        let cc = [
            ((center[0] - self.origin[0]) / self.cell_size).floor() as i64,
            ((center[1] - self.origin[1]) / self.cell_size).floor() as i64,
            ((center[2] - self.origin[2]) / self.cell_size).floor() as i64,
        ];
        
        let mut result = Vec::new();
        let r2 = radius * radius;
        
        for dx in -r_cells..=r_cells {
            for dy in -r_cells..=r_cells {
                for dz in -r_cells..=r_cells {
                    let cell = [cc[0] + dx, cc[1] + dy, cc[2] + dz];
                    if let Some(indices) = self.cells.get(&cell) {
                        for &idx in indices {
                            let p = self.positions[idx];
                            let d2 = (p[0]-center[0]).powi(2) + (p[1]-center[1]).powi(2) + (p[2]-center[2]).powi(2);
                            if d2 <= r2 {
                                result.push(idx);
                            }
                        }
                    }
                }
            }
        }
        
        result
    }
}
```

```rust
    let s = s.trim();
    if s.is_empty() || s.eq_ignore_ascii_case("na") || s.eq_ignore_ascii_case("nan") || 
       s.eq_ignore_ascii_case("n/a") || s == "-" || s == "." {
        return None;
    }
    s.parse::<f64>().ok().filter(|v| v.is_finite())
}
```

```rust
fn parse_float(s: &str) -> Option<f64> {
    let s = s.trim();
    match s.to_lowercase().as_str() {
        "" | "na" | "nan" | "n/a" | "-" | "." | "null" | "nil" => None,
        _ => s.parse::<f64>().ok().filter(|v| v.is_finite()),
    }
}

fn median_of_sorted(sorted: &[f64]) -> f64 {
    let n = sorted.len();
    if n % 2 == 0 {
        (sorted[n / 2 - 1] + sorted[n / 2]) / 2.0
    } else {
        sorted[n / 2]
    }
}

fn modified_z_score(val: f64, neighbors: &[f64]) -> f64 {
    let n = neighbors.len();
    if n < 2 { return 0.0; }
    
    let mut sorted = neighbors.to_vec();
    sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
    
    let med = median_of_sorted(&sorted);
    
    let mut abs_devs: Vec<f64> = sorted.iter().map(|v| (v - med).abs()).collect();
    abs_devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
    let mad = median_of_sorted(&abs_devs);
    
    if mad > 1e-12 {
        0.6745 * (val - med).abs() / mad
    } else {
        // All neighbors have essentially the same value
        let range = sorted[n - 1] - sorted[0];
        if range > 1e-12 {
            (val - med).abs() / range * 20.0
        } else {
            // Truly constant neighbors
            if (val - med).abs() > 1e-10 {
                100.0 // Maximum anomaly
            } else {
                0.0
            }
        }
    }
}

struct SpatialGrid {
    cell_size: f64,
    origin: [f64; 3],
    cells: HashMap<[i64; 3], Vec<usize>>,
}

impl SpatialGrid {
    fn build(positions: &[[f64; 3]], indices: &[usize], target_avg_per_cell: f64) -> Self {
        if indices.is_empty() {
            return Self {
                cell_size: 1.0,
                origin: [0.0; 3],
                cells: HashMap::new(),
            };
        }
        
        // Compute bounding box
        let mut mn = [f64::INFINITY; 3];
        let mut mx = [f64::NEG_INFINITY; 3];
        for &i in indices {
            let p = positions[i];
            for d in 0..3 {
                mn[d] = mn[d].min(p[d]);
                mx[d] = mx[d].max(p[d]);
            }
        }
        
        let extent = [
            (mx[0] - mn[0]).max(1e-6),
            (mx[1] - mn[1]).max(1e-6),
            (mx[2] - mn[2]).max(1e-6),
        ];
        
        let total_vol = extent[0] * extent[1] * extent[2];
        let n = indices.len() as f64;
        let target_vol_per_cell = n / target_avg_per_cell;
        let cell_size = (total_vol / target_vol_per_cell).cbrt().max(1e-6);
        
        let mut cells: HashMap<[i64; 3], Vec<usize>> = HashMap::new();
        for &i in indices {
            let p = positions[i];
            let cell = [
                ((p[0] - mn[0]) / cell_size).floor() as i64,
                ((p[1] - mn[1]) / cell_size).floor() as i64,
                ((p[2] - mn[2]) / cell_size).floor() as i64,
            ];
            cells.entry(cell).or_default().push(i);
        }
        
        Self {
            cell_size,
            origin: mn,
            cells,
        }
    }
    
    fn query(&self, positions: &[[f64; 3]], center: [f64; 3], radius: f64) -> Vec<usize> {
        let rc = (radius / self.cell_size).ceil() as i64;
        let cc = [
            ((center[0] - self.origin[0]) / self.cell_size).floor() as i64,
            ((center[1] - self.origin[1]) / self.cell_size).floor() as i64,
            ((center[2] - self.origin[2]) / self.cell_size).floor() as i64,
        ];
        
        let r2 = radius * radius;
        let mut result = Vec::new();
        
        for dx in -rc..=rc {
            for dy in -rc..=rc {
                for dz in -rc..=rc {
                    let cell = [cc[0] + dx, cc[1] + dy, cc[2] + dz];
                    if let Some(indices) = self.cells.get(&cell) {
                        for &idx in indices {
                            let p = positions[idx];
                            let d2 = (p[0]-center[0]).powi(2) + (p[1]-center[1]).powi(2) + (p[2]-center[2]).powi(2);
                            if d2 <= r2 * 1.01 { // small tolerance for floating point
                                result.push(idx);
                            }
                        }
                    }
                }
            }
        }
        
        result
    }
}
```

fn parse_float(s: &str) -> Option<f64> {
    let s = s.trim();
    match s.to_lowercase().as_str() {
        "" | "na" | "nan" | "n/a" | "-" | "." | "null" | "nil" => None,
        _ => s.parse::<f64>().ok().filter(|v| v.is_finite()),

fn median_of_sorted(sorted: &[f64]) -> f64 {
    let n = sorted.len();
    if n == 0 { return 0.0; }
    if n % 2 == 0 {
        (sorted[n / 2 - 1] + sorted[n / 2]) / 2.0
    } else {
        sorted[n / 2]
    }
}

fn modified_z_score(val: f64, neighbors: &[f64]) -> f64 {
    let n = neighbors.len();
    if n < 2 { return 0.0; }
    
    let mut sorted = neighbors.to_vec();
    sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
    
    let med = median_of_sorted(&sorted);
    
    let mut abs_devs: Vec<f64> = sorted.iter().map(|v| (v - med).abs()).collect();
    abs_devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
    let mad = median_of_sorted(&abs_devs);
    
    if mad > 1e-12 {
        0.6745 * (val - med).abs() / mad
    } else {
        let range = sorted[n - 1] - sorted[0];
        if range > 1e-12 {
            (val - med).abs() / range * 20.0
        } else {
            if (val - med).abs() > 1e-10 {
                100.0
            } else {
                0.0
            }
        }
    }
}
