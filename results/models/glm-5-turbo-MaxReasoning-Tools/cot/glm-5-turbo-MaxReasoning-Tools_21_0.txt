uint64_t encode(const vector<int>& p) {
    uint64_t id = 0;
    for (int i = 0; i < D; i++) id += (uint64_t)p[i] * mult[i];
    return id;
}

vector<int> decode(uint64_t id) {
    vector<int> p(D);
    for (int i = D - 1; i >= 0; i--) {
        p[i] = (int)(id / mult[i]);
        id %= mult[i];
    }
    return p;
}

int manhattan(uint64_t a, uint64_t b) {
    vector<int> pa = decode(a), pb = decode(b);
    int d = 0;
    for (int i = 0; i < D; i++) d += abs(pa[i] - pb[i]);
    return d;
}

// A* returns first move as {axis, dir} or {-1, 0}
pair<int,int> astar(uint64_t start, uint64_t goal, const unordered_set<uint64_t>& blocked, int maxn = 2000000) {
    if (start == goal) return {0, 1};
    if (blocked.count(goal)) return {-1, 0};
    
    auto h = [&](uint64_t id) -> int {
        vector<int> p = decode(id);
        vector<int> g = decode(goal);
        int d = 0;
        for (int i = 0; i < D; i++) d += abs(p[i] - g[i]);
        return d;
    };
    
    struct Entry { int f, g, cnt; uint64_t id; };
    auto cmp = [](const Entry& a, const Entry& b) { return a.f > b.f || (a.f == b.f && a.cnt > b.cnt); };
    priority_queue<Entry, vector<Entry>, decltype(cmp)> pq(cmp);
    
    unordered_map<uint64_t, int> gs;
    unordered_map<uint64_t, uint64_t> par;
    unordered_set<uint64_t> vis;
    
    int cnt = 0;
    gs[start] = 0;
    pq.push({h(start), 0, cnt++, start});
    
    int explored = 0;
    while (!pq.empty() && explored < maxn) {
        auto cur = pq.top(); pq.pop();
        if (vis.count(cur.id)) continue;
        vis.insert(cur.id);
        explored++;
        
        if (cur.id == goal) {
            uint64_t node = cur.id;
            while (par.count(node) && par[node] != start) node = par[node];
            if (!par.count(node)) return {-1, 0};
            
            int64_t diff = (int64_t)node - (int64_t)start;
            for (int d = 0; d < D; d++) {
                if ((int64_t)mult[d] == diff) return {d, 1};
                if (-(int64_t)mult[d] == diff) return {d, -1};
            }
            return {-1, 0};
        }
        
        vector<int> cp = decode(cur.id);
        for (int d = 0; d < D; d++) {
            for (int dir = -1; dir <= 1; dir += 2) {
                int ni = cp[d] + dir;
                if (ni < 0 || ni >= bounds[d]) continue;
                uint64_t nid = (dir == 1) ? cur.id + mult[d] : cur.id - mult[d];
                if (blocked.count(nid) || vis.count(nid)) continue;
                
                int ng = cur.g + 1;
                if (!gs.count(nid) || ng < gs[nid]) {
                    gs[nid] = ng;
                    par[nid] = cur.id;
                    pq.push({ng + h(nid), ng, cnt++, nid});
                }
            }
        }
    }
    return {-1, 0};
}

pair<int,int> get_safe_move() {
    uint64_t head = snake.front();
    vector<int> hp = decode(head);
    
    unordered_set<uint64_t> blocked;
    blocked.insert(obs.begin(), obs.end());
    for (auto s : snake) blocked.insert(s);
    // Remove head from blocked since we're moving away from it
    blocked.erase(head);
    
    for (int d = 0; d < D; d++) {
        for (int dir = -1; dir <= 1; dir += 2) {
            int ni = hp[d] + dir;
            if (ni < 0 || ni >= bounds[d]) continue;
            uint64_t nid = (dir == 1) ? head + mult[d] : head - mult[d];
            if (!blocked.count(nid)) return {d, dir};
        }
    }
    return {0, 1}; // no safe move
}

pair<int,int> find_move() {
    uint64_t head = snake.front();
    if (food_left.empty()) return {0, 1};
    
    // Build blocked set
    unordered_set<uint64_t> blocked;
    blocked.insert(obs.begin(), obs.end());
    for (int i = 1; i < (int)snake.size() - 1; i++) blocked.insert(snake[i]);
    // If snake.size() <= 2, no body segments to block
    
    // Find nearest food
    uint64_t nearest = 0;
    int best = INT_MAX;
    for (auto f : food) {
        if (!food_left.count(f)) continue;
        int d = manhattan(head, f);
        if (d < best) { best = d; nearest = f; }
    }
    
    if (nearest != 0 || food_left.count(nearest)) {
        auto mv = astar(head, nearest, blocked);
        if (mv.first >= 0) return mv;
    }
    
    // Can't reach food, follow tail
    if (snake.size() > 1) {
        auto mv = astar(head, snake.back(), blocked, 500000);
        if (mv.first >= 0) return mv;
    }
    
    return get_safe_move();
}

When I use `pop_back` to remove the tail, the last element is removed.

Actually, for a deque:
- `front()` returns the first element
- `back()` returns the last element
- `push_front(x)` adds x to the front
- `push_back(x)` adds x to the back
- `pop_front()` removes the first element
- `pop_back()` removes the last element

So if I maintain the invariant that `front()` is the head and `back()` is the tail:
- To move: `push_front(new_head)`, `pop_back()` (or not, if food eaten)
- This correctly shifts the entire snake forward
