bool dsatur_backtrack(vector<int>& color, const vector<vector<int>>& adj, int k, 
                       int time_limit_ms = 30000) {
    int n = (int)adj.size();
    if (n == 0) return true;
    fill(color.begin(), color.end(), -1);
    
    vector<uint64_t> ncol(n, 0);
    vector<int> deg(n);
    for (int v = 0; v < n; v++) deg[v] = (int)adj[v].size();
    
    auto start = chrono::steady_clock::now();
    bool timeout = false;
    
    // Track stack of changes for undo
    struct Change { int v; int c; };
    
    function<bool(int)> solve = [&](int num_colored) -> bool {
        if (timeout) return false;
        if (num_colored == n) return true;
        
        // Check timeout
        auto now = chrono::steady_clock::now();
        if (chrono::duration_cast<chrono::milliseconds>(now - start).count() > time_limit_ms) {
            timeout = true;
            return false;
        }
        
        // Select vertex with max saturation, then max degree
        int best = -1, best_sat = -1, best_deg = -1;
        for (int v = 0; v < n; v++) {
            if (color[v] == -1) {
                int s = popcount64(ncol[v]);
                if (s > best_sat || (s == best_sat && deg[v] > best_deg)) {
                    best = v;
                    best_sat = s;
                    best_deg = deg[v];
                }
            }
        }
        
        if (best == -1) return false;
        
        uint64_t used = ncol[best];
        
        // Try each available color
        // Order: try least constraining first? Or just in order 0..k-1?
        for (int c = 0; c < k; c++) {
            if (used & (1ULL << c)) continue;
            
            color[best] = c;
            
            // Update neighbors
            vector<int> affected;
            bool feasible = true;
            for (int v : adj[best]) {
                if (color[v] == -1) {
                    if (ncol[v] & (1ULL << c)) continue; // already has this color in neighbor
                    ncol[v] |= (1ULL << c);
                    affected.push_back(v);
                    if (popcount64(ncol[v]) >= k) {
                        feasible = false;
                    }
                }
            }
            
            if (feasible && solve(num_colored + 1)) return true;
            
            // Undo
            color[best] = -1;
            for (int v : affected) {
                ncol[v] &= ~(1ULL << c);
            }
        }
        
        return false;
    };
    
    return solve(0) && !timeout;
}

// ============ DSATUR GREEDY ============

bool dsatur_greedy(vector<int>& color, const vector<vector<int>>& adj, int k) {
    const int n = (int)adj.size();
    if (n == 0) return true;
    
    fill(color.begin(), color.end(), -1);
    
    vector<uint64_t> ncol(n, 0);
    vector<int> deg(n);
    for (int v = 0; v < n; v++) deg[v] = (int)adj[v].size();
    
    // Bucket by saturation degree
    // buckets[s] = list of uncolored vertices with saturation s
    // pos[v] = position of v in its bucket, -1 if removed/colored
    vector<vector<int>> buckets(k + 1);
    vector<int> sat(n, 0);
    vector<int> pos(n);
    
    for (int v = 0; v < n; v++) {
        buckets[0].push_back(v);
        pos[v] = v;
    }
    
    uint64_t color_mask = (k >= 64) ? ~0ULL : ((1ULL << k) - 1);
    int uncolored = n;
    
    while (uncolored > 0) {
        // Find highest non-empty bucket
        int b = k;
        while (b >= 0 && buckets[b].empty()) b--;
        if (b < 0) return false;
        
        auto& bk = buckets[b];
        
        // Find vertex with max degree in this bucket
        int bi = 0;
        for (int i = 1; i < (int)bk.size(); i++) {
            if (deg[bk[i]] > deg[bk[bi]]) bi = i;
        }
        
        int u = bk[bi];
        
        // Remove from bucket
        int last = bk.back();
        bk[bi] = last;
        pos[last] = bi;
        bk.pop_back();
        pos[u] = -1;
        
        // Find smallest available color
        uint64_t available = color_mask & ~ncol[u];
        if (available == 0) return false;
        int c = ctz64(available);
        
        color[u] = c;
        uncolored--;
        
        // Update neighbors
        uint64_t cbit = 1ULL << c;
        for (int v : adj[u]) {
            if (pos[v] >= 0 && !(ncol[v] & cbit)) {
                int os = sat[v];
                ncol[v] |= cbit;
                int ns = sat[v] = popcount64(ncol[v]);
                
                if (ns > k) return false; // shouldn't happen
                
                // Move v from buckets[os] to buckets[ns]
                int p = pos[v];
                auto& ob = buckets[os];
                int ob_last = ob.back();
                ob[p] = ob_last;
                pos[ob_last] = p;
                ob.pop_back();
                
                buckets[ns].push_back(v);
                pos[v] = (int)buckets[ns].size() - 1;
            }
        }
    }
    
    return true;
}

// ============ DSATUR BACKTRACKING ============

bool dsatur_backtrack(vector<int>& color, const vector<vector<int>>& adj, int k,
                       chrono::steady_clock::time_point deadline) {
    const int n = (int)adj.size();
    if (n == 0) return true;
    
    fill(color.begin(), color.end(), -1);
    
    vector<uint64_t> ncol(n, 0);
    vector<int> deg(n);
    for (int v = 0; v < n; v++) deg[v] = (int)adj[v].size();
    
    uint64_t color_mask = (k >= 64) ? ~0ULL : ((1ULL << k) - 1);
    
    vector<vector<int>> children(n); // For each vertex, list of uncolored neighbors
    
    function<bool(int)> solve = [&](int num_colored) -> bool {
        if (chrono::steady_clock::now() > deadline) return false;
        if (num_colored == n) return true;
        
        // Select vertex with max saturation, then max degree
        int best = -1, best_sat = -1, best_deg = -1;
        for (int v = 0; v < n; v++) {
            if (color[v] == -1) {
                int s = popcount64(ncol[v]);
                if (s > best_sat || (s == best_sat && deg[v] > best_deg)) {
                    best = v; best_sat = s; best_deg = deg[v];
                }
            }
        }
        
        if (best == -1) return false;
        
        uint64_t used = ncol[best];
        uint64_t available = color_mask & ~used;
        
        // Try each available color
        while (available) {
            int c = ctz64(available);
            available &= available - 1; // remove lowest bit
            
            color[best] = c;
            uint64_t cbit = 1ULL << c;
            
            // Update neighbors
            vector<int> affected;
            bool feasible = true;
            for (int v : adj[best]) {
                if (color[v] == -1 && !(ncol[v] & cbit)) {
                    ncol[v] |= cbit;
                    affected.push_back(v);
                    if (popcount64(ncol[v]) >= k) {
                        feasible = false;
                    }
                }
            }
            
            if (feasible && solve(num_colored + 1)) return true;
            
            // Undo
            color[best] = -1;
            for (int v : affected) {
                ncol[v] &= ~cbit;
            }
        }
        
        return false;
    };
    
    return solve(0);
}

// ============ RANDOM GREEDY ============

bool random_greedy(vector<int>& color, const vector<vector<int>>& adj, int k, mt19937& rng) {
    const int n = (int)adj.size();
    if (n == 0) return true;
    fill(color.begin(), color.end(), -1);
    
    // Create order: sort by degree with random perturbation
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    
    uniform_real_distribution<double> dist(0.0, 1.0);
    vector<double> perturbation(n);
    for (int v = 0; v < n; v++) perturbation[v] = dist(rng);
    
    sort(order.begin(), order.end(), [&](int a, int b) {
        double da = adj[a].size() + perturbation[a] * 0.5;
        double db = adj[b].size() + perturbation[b] * 0.5;
        return da > db;
    });
    
    vector<uint64_t> ncol(n, 0);
    uint64_t color_mask = (k >= 64) ? ~0ULL : ((1ULL << k) - 1);
    
    for (int u : order) {
        uint64_t available = color_mask & ~ncol[u];
        if (available == 0) return false;
        int c = ctz64(available);
        
        color[u] = c;
        uint64_t cbit = 1ULL << c;
        for (int v : adj[u]) {
            ncol[v] |= cbit;
        }
    }
    
    return true;
}
