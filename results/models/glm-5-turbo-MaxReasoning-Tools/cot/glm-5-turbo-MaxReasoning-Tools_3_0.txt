fn force_iteration_grid(adj: &[Vec<usize>], x: &mut [f64], y: &mut [f64], k: f64, temp: f64, n: usize) {
    let mut dx = vec![0.0; n];
    let mut dy = vec![0.0; n];
    
    // Determine bounding box
    let (mut min_x, mut max_x) = (f64::MAX, f64::MIN);
    let (mut min_y, mut max_y) = (f64::MAX, f64::MIN);
    for i in 0..n {
        min_x = min_x.min(x[i]);
        max_x = max_x.max(x[i]);
        min_y = min_y.min(y[i]);
        max_y = max_y.max(y[i]);
    }
    
    let width = (max_x - min_x).max(1.0);
    let height = (max_y - min_y).max(1.0);
    let cell_size = k * 2.0;
    let grid_w = ((width / cell_size) as usize + 2).max(3);
    let grid_h = ((height / cell_size) as usize + 2).max(3);
    
    // Assign nodes to grid cells
    let mut grid: Vec<Vec<usize>> = vec![vec![]; grid_w * grid_h];
    for i in 0..n {
        let cx = ((x[i] - min_x) / cell_size) as usize;
        let cy = ((y[i] - min_y) / cell_size) as usize;
        let cx = cx.min(grid_w - 1);
        let cy = cy.min(grid_h - 1);
        grid[cx * grid_h + cy].push(i);
    }
    
    // Repulsive forces (grid-based)
    for i in 0..n {
        let cx = ((x[i] - min_x) / cell_size) as usize;
        let cy = ((y[i] - min_y) / cell_size) as usize;
        let cx = cx.min(grid_w - 1);
        let cy = cy.min(grid_h - 1);
        
        for dcx in cx.saturating_sub(1)..=(cx + 1).min(grid_w - 1) {
            for dcy in cy.saturating_sub(1)..=(cy + 1).min(grid_h - 1) {
                for &j in &grid[dcx * grid_h + dcy] {
                    if j <= i { continue; }
                    let ddx = x[i] - x[j];
                    let ddy = y[i] - y[j];
                    let dist_sq = ddx * ddx + ddy * ddy;
                    let dist = dist_sq.sqrt().max(0.001);
                    let force = k * k / dist;
                    let fx = ddx / dist * force;
                    let fy = ddy / dist * force;
                    dx[i] += fx;
                    dy[i] += fy;
                    dx[j] -= fx;
                    dy[j] -= fy;
                }
            }
        }
    }
    
    // Attractive forces (edge-based)
    for i in 0..n {
        for &j in &adj[i] {
            if j <= i { continue; }
            let ddx = x[i] - x[j];
            let ddy = y[i] - y[j];
            let dist_sq = ddx * ddx + ddy * ddy;
            let dist = dist_sq.sqrt().max(0.001);
            let force = dist / k;
            let fx = ddx / dist * force;
            let fy = ddy / dist * force;
            dx[i] -= fx;
            dy[i] -= fy;
            dx[j] += fx;
            dy[j] += fy;
        }
    }
    
    // Apply displacements with temperature
    for i in 0..n {
        let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt().max(0.001);
        let scale = temp.min(disp) / disp;
        x[i] += dx[i] * scale;
        y[i] += dy[i] * scale;
    }
}

fn spectral_approx(adj: &[Vec<usize>], n: usize, iterations: usize) -> (Vec<f64>, Vec<f64>) {

    // Initialize with pseudo-random values
    let mut x: Vec<f64> = (0..n).map(|i| {
        rng_state = rng_state.wrapping_mul(6364136223846793005).wrapping_add(1);
        (rng_state >> 33) as f64 / (1u64 << 31) as f64 - 1.0
    }).collect();
    let mut y: Vec<f64> = (0..n).map(|i| {
        rng_state = rng_state.wrapping_mul(6364136223846793005).wrapping_add(1);
        (rng_state >> 33) as f64 / (1u64 << 31) as f64 - 1.0
    }).collect();
    
    // Compute degrees
    let deg: Vec<f64> = (0..n).map(|i| adj[i].len() as f64).collect();
    
    for _ in 0..iterations {
        let mut x_new = vec![0.0; n];
        let mut y_new = vec![0.0; n];
        
        // Smooth: weighted average of neighbors
        for i in 0..n {
            if adj[i].is_empty() { continue; }
            let mut sum_x = 0.0;
            let mut sum_y = 0.0;
            for &j in &adj[i] {
                sum_x += x[j];
                sum_y += y[j];
            }
            // Mix between current position and neighbor average
            let mix = 0.5;
            x_new[i] = (1.0 - mix) * x[i] + mix * sum_x / deg[i];
            y_new[i] = (1.0 - mix) * y[i] + mix * sum_y / deg[i];
        }
        
        // Subtract mean
        let mean_x: f64 = x_new.iter().sum::<f64>() / n as f64;
        let mean_y: f64 = y_new.iter().sum::<f64>() / n as f64;
        for i in 0..n {
            x[i] = x_new[i] - mean_x;
            y[i] = y_new[i] - mean_y;
        }
        
        // Normalize
        let mut norm_sq = 0.0;
        for i in 0..n {
            norm_sq += x[i] * x[i] + y[i] * y[i];
        }
        let norm = norm_sq.sqrt().max(1e-10);
        for i in 0..n {
            x[i] /= norm;
            y[i] /= norm;
        }
    }
    
    (x, y)
}

        let cols = (num_components as f64).sqrt().ceil() as usize;
        let gap = max_w.max(max_h).max(1.0) * 2.0;
        
        for ci in 0..num_components {
            let row = ci / cols;
            let col = ci % cols;
            let (min_x, _, min_y, _) = bounds[ci];
            let offset_x = col as f64 * (max_w + gap) - min_x;
            let offset_y = row as f64 * (max_h + gap) - min_y;
            for &node in &components[ci] {
                pos_x[node] += offset_x;
                pos_y[node] += offset_y;
            }
        }
    }
    
    // Output
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    for i in 0..n {
        writeln!(out, "{} {}", pos_x[i], pos_y[i])?;
    }
    
    Ok(())
}

fn layout_component(adj: &[Vec<usize>], n: usize) -> (Vec<f64>, Vec<f64>) {
    if n == 0 {
        return (vec![], vec![]);
    }
    if n == 1 {
        return (vec![0.0], vec![0.0]);
    }
    if n == 2 {
        return (vec![0.0, 1.0], vec![0.0, 0.0]);
    }
    
    // Initialize positions - circular layout
    let (mut x, mut y) = circular_init(n);
    
    // Force-directed parameters
    let area = n as f64 * 4.0;
    let k = (area / n as f64).sqrt(); // = 2.0
    
    // Adaptive iterations
    let iterations = if n <= 10 { 500 }
        else if n <= 30 { 300 }
        else if n <= 100 { 200 }
        else if n <= 300 { 150 }
        else if n <= 1000 { 80 }
        else if n <= 5000 { 50 }
        else if n <= 20000 { 30 }
        else { 20 };
    
    let initial_temp = k * 10.0;
    
    let use_grid = n > 400;
    
    for iter in 0..iterations {
        let progress = iter as f64 / iterations as f64;
        let temp = initial_temp * (1.0 - progress);
        
        if use_grid {
            force_grid(adj, &mut x, &mut y, k, temp, n);
        } else {
            force_exact(adj, &mut x, &mut y, k, temp, n);
        }
    }
    
    // Center the layout
    let cx: f64 = x.iter().sum::<f64>() / n as f64;
    let cy: f64 = y.iter().sum::<f64>() / n as f64;
    for i in 0..n {
        x[i] -= cx;
        y[i] -= cy;
    }
    
    (x, y)
}

fn circular_init(n: usize) -> (Vec<f64>, Vec<f64>) {
    let radius = n as f64 * 0.5;
    let mut x = Vec::with_capacity(n);
    let mut y = Vec::with_capacity(n);
    let golden_ratio = (1.0 + 5.0_f64.sqrt()) / 2.0;
    
    for i in 0..n {
        let angle = 2.0 * std::f64::consts::PI * i as f64 / golden_ratio;
        let r = radius * ((i + 1) as f64 / n as f64).sqrt(); // Sunflower pattern
        x.push(r * angle.cos());
        y.push(r * angle.sin());
    }
    
    (x, y)
}

fn force_exact(adj: &[Vec<usize>], x: &mut [f64], y: &mut [f64], k: f64, temp: f64, n: usize) {
    let mut dx = vec![0.0; n];
    let mut dy = vec![0.0; n];
    
    // Repulsive forces: all pairs
    for i in 0..n {
        for j in (i + 1)..n {
            let ddx = x[i] - x[j];
            let ddy = y[i] - y[j];
            let dist_sq = ddx * ddx + ddy * ddy;
            if dist_sq < 1e-10 {
                continue;
            }
            let dist = dist_sq.sqrt();
            let force = k * k / dist;
            let fx = ddx / dist * force;
            let fy = ddy / dist * force;
            dx[i] += fx;
            dy[i] += fy;
            dx[j] -= fx;
            dy[j] -= fy;
        }
    }
    
    // Attractive forces: edges
    for i in 0..n {
        for &j in &adj[i] {
            if j > i {
                let ddx = x[i] - x[j];
                let ddy = y[i] - y[j];
                let dist_sq = ddx * ddx + ddy * ddy;
                if dist_sq < 1e-10 {
                    continue;
                }
                let dist = dist_sq.sqrt();
                let force = dist / k;
                let fx = ddx / dist * force;
                let fy = ddy / dist * force;
                dx[i] -= fx;
                dy[i] -= fy;
                dx[j] += fx;
                dy[j] += fy;
            }
        }
    }
    
    // Apply with temperature
    for i in 0..n {
        let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt();
        if disp > 0.0 {
            let scale = temp.min(disp) / disp;
            x[i] += dx[i] * scale;
            y[i] += dy[i] * scale;
        }
    }
}

fn force_grid(adj: &[Vec<usize>], x: &mut [f64], y: &mut [f64], k: f64, temp: f64, n: usize) {
    let mut disp_x = vec![0.0; n];
    let mut disp_y = vec![0.0; n];
    
    // Compute bounding box
    let (mut min_x, mut max_x) = (f64::MAX, f64::MIN);
    let (mut min_y, mut max_y) = (f64::MAX, f64::MIN);
    for i in 0..n {
        min_x = min_x.min(x[i]);
        max_x = max_x.max(x[i]);
        min_y = min_y.min(y[i]);
        max_y = max_y.max(y[i]);
    }
    
    let cell_size = k * 2.0;
    let width = (max_x - min_x).max(cell_size);
    let height = (max_y - min_y).max(cell_size);
    let grid_cols = (width / cell_size) as usize + 2;
    let grid_rows = (height / cell_size) as usize + 2;
    
    // Build grid
    let total_cells = grid_cols * grid_rows;
    let mut grid: Vec<Vec<usize>> = vec![Vec::new(); total_cells];
    
    for i in 0..n {
        let col = ((x[i] - min_x) / cell_size) as usize;
        let row = ((y[i] - min_y) / cell_size) as usize;
        let col = col.min(grid_cols - 1);
        let row = row.min(grid_rows - 1);
        grid[col * grid_rows + row].push(i);
    }
    
    // Repulsive forces using grid
    for i in 0..n {
        let col = ((x[i] - min_x) / cell_size) as usize;
        let row = ((y[i] - min_y) / cell_size) as usize;
        let col = col.min(grid_cols - 1);
        let row = row.min(grid_rows - 1);
        
        let col_start = col.saturating_sub(1);
        let col_end = (col + 1).min(grid_cols - 1);
        let row_start = row.saturating_sub(1);
        let row_end = (row + 1).min(grid_rows - 1);
        
        for c in col_start..=col_end {
            for r in row_start..=row_end {
                for &j in &grid[c * grid_rows + r] {
                    if j <= i {
                        continue;
                    }
                    let ddx = x[i] - x[j];
                    let ddy = y[i] - y[j];
                    let dist_sq = ddx * ddx + ddy * ddy;
                    if dist_sq < 1e-10 {
                        continue;
                    }
                    let dist = dist_sq.sqrt();
                    let force = k * k / dist;
                    let fx = ddx / dist * force;
                    let fy = ddy / dist * force;
                    disp_x[i] += fx;
                    disp_y[i] += fy;
                    disp_x[j] -= fx;
                    disp_y[j] -= fy;
                }
            }
        }
    }
    
    // Attractive forces: edges
    for i in 0..n {
        for &j in &adj[i] {
            if j > i {
                let ddx = x[i] - x[j];
                let ddy = y[i] - y[j];
                let dist_sq = ddx * ddx + ddy * ddy;
                if dist_sq < 1e-10 {
                    continue;
                }
                let dist = dist_sq.sqrt();
                let force = dist / k;
                let fx = ddx / dist * force;
                let fy = ddy / dist * force;
                disp_x[i] -= fx;
                disp_y[i] -= fy;
                disp_x[j] += fx;
                disp_y[j] += fy;
            }
        }
    }
    
    // Apply with temperature
    for i in 0..n {
        let disp = (disp_x[i] * disp_x[i] + disp_y[i] * disp_y[i]).sqrt();
        if disp > 0.0 {
            let scale = temp.min(disp) / disp;
            x[i] += disp_x[i] * scale;
            y[i] += disp_y[i] * scale;
        }
    }
}
