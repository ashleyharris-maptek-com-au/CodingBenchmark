    int board_w, board_h;
    vector<SkylineSegment> segs;
    
public:
    Skyline(int bw, int bh) : board_w(bw), board_h(bh) {
        segs.push_back({0, 0, bw});
    }
    
    // Try to place rectangle (rw, rh). Returns (x, y) or (-1, -1) if fails.
    pair<int,int> tryPlace(int rw, int rh) {
        // Find best position
        int best_x = -1, best_y = board_h + 1;
        
        for (int i = 0; i < (int)segs.size(); i++) {
            if (segs[i].x + rw > board_w) break;
            
            // Compute max height in [segs[i].x, segs[i].x + rw]
            int max_h = 0;
            for (int j = i; j < (int)segs.size(); j++) {
                if (segs[j].x >= segs[i].x + rw) break;
                max_h = max(max_h, segs[j].y);
            }
            
            if (max_h + rh <= board_h) {
                // Candidate position
                if (max_h < best_y || (max_h == best_y && segs[i].x < best_x)) {
                    best_y = max_h;
                    best_x = segs[i].x;
                }
            }
        }
        
        if (best_x == -1) return {-1, -1};
        
        // Update skyline
        place(best_x, best_y, rw, rh);
        return {best_x, best_y};
    }
    
    void place(int px, int py, int rw, int rh) {
        int new_h = py + rh;
        
        // Find segments to modify
        int start = -1, end = -1;
        for (int i = 0; i < (int)segs.size(); i++) {
            if (segs[i].x < px + rw && segs[i].x + segs[i].w > px) {
                if (start == -1) start = i;
                end = i;
            }

        // Handle splitting at left edge
        if (segs[start].x < px) {
            // Split: create new segment for [segs[start].x, px)
            int orig_w = segs[start].w;
            segs[start].w = px - segs[start].x;
            segs.insert(segs.begin() + start + 1, {px, segs[start].y, orig_w - segs[start].w});

            end++;

        if (segs[end].x + segs[end].w > px + rw) {
            // Split: create new segment for [px+rw, segs[end].x + segs[end].w)
            int orig_w = segs[end].w;
            int new_seg_w = segs[end].x + segs[end].w - (px + rw);
            segs[end].w = (px + rw) - segs[end].x;
            segs.insert(segs.begin() + end + 1, {px + rw, segs[end].y, new_seg_w});

            segs[i].y = max(segs[i].y, new_h);

        vector<SkylineSegment> merged;
        for (int i = 0; i < (int)segs.size(); i++) {
            if (!merged.empty() && merged.back().y == segs[i].y) {
                merged.back().w += segs[i].w;
            } else {
                merged.push_back(segs[i]);
            }
        }
        segs = merged;
    }

    if (rw > board_w || rh > board_h) return {-1, -1};
    
    // Sliding window max of size rw over skyline
    deque<int> dq;
    int best_x = -1, best_y = board_h + 1;
    
    for (int i = 0; i < board_w; i++) {
        // Add skyline[i] to deque
        while (!dq.empty() && skyline[dq.back()] <= skyline[i]) dq.pop_back();
        dq.push_back(i);
        
        // Remove elements outside window [i-rw+1, i]

        // Window is [i-rw+1, i]

            int max_h = skyline[dq.front()];
            if (max_h + rh <= board_h) {
                if (max_h < best_y || (max_h == best_y && window_start < best_x)) {

    return {best_x, best_y};

**Optimization 4: Better heuristic for starting positions**

Instead of trying every segment as a starting position, only try segments that are "below" the rectangle's potential placement. Specifically, the base height of the placement is the max height in the range, and we want the minimum base height.

Key insight: if segments s_i and s_j have the same height y, and s_i starts before s_j, then starting at s_i is always at least as good as starting at s_j (same height, but more to the left). So we only need to try segments at distinct heights.

This reduces the number of starting positions, but in the worst case all segments have distinct heights, so no improvement.

For BWF, among all positions where the rectangle fits, choose the one where the rectangle width most closely matches the gap width. This tends to create more uniform gaps.
