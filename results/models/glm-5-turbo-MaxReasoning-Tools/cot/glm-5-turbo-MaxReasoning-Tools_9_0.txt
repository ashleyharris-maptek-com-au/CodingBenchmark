class FastReader
{
    private Stream stream;
    private byte[] buffer;
    private int pos;
    private int length;
    
    public FastReader(Stream stream, int bufferSize = 1 << 20)
    {
        this.stream = stream;
        this.buffer = new byte[bufferSize];
        this.pos = 0;
        this.length = 0;
    }
    
    private void Fill()
    {
        length = stream.Read(buffer, 0, buffer.Length);
        pos = 0;
    }
    
    public int ReadInt()
    {
        int c = ReadByte();
        while (c == ' ' || c == '\n' || c == '\r' || c == '\t')
            c = ReadByte();
        
        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = ReadByte();
        }
        
        int value = 0;
        while (c >= '0' && c <= '9')
        {
            value = value * 10 + (c - '0');
            c = ReadByte();
        }
        
        return sign * value;
    }
    
    private int ReadByte()
    {
        if (pos >= length)
        {
            Fill();
            if (length == 0) return -1;
        }
        return buffer[pos++];
    }
}
