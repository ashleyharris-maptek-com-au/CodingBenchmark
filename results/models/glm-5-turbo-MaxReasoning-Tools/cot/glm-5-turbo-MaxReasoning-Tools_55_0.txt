```cpp
Position parseFEN(const string& fen) {
    Position pos;
    memset(&pos, 0, sizeof(pos));
    pos.ep = -1;
    
    istringstream ss(fen);
    string board, sideStr, castleStr, epStr;
    ss >> board >> sideStr >> castleStr >> epStr >> pos.halfmove >> pos.fullmove;
    
    pos.side = (sideStr == "w") ? WHITE : BLACK;
    
    // Parse board
    int rank = 7, file = 0;
    for (char c : board) {
        if (c == '/') {
            rank--;
            file = 0;
        } else if (c >= '1' && c <= '8') {
            file += c - '0';
        } else {
            int sq = mkSq(rank, file);
            int piece = 0;
            switch (c) {
                case 'P': piece = PAWN; break;
                case 'N': piece = KNIGHT; break;
                case 'B': piece = BISHOP; break;
                case 'R': piece = ROOK; break;
                case 'Q': piece = QUEEN; break;
                case 'K': piece = KING; break;
                case 'p': piece = -PAWN; break;
                case 'n': piece = -KNIGHT; break;
                case 'b': piece = -BISHOP; break;
                case 'r': piece = -ROOK; break;
                case 'q': piece = -QUEEN; break;
                case 'k': piece = -KING; break;
            }
            pos.board[sq] = piece;
            if (pieceType(piece) == KING) {
                pos.kingSq[pieceColor(piece)] = sq;
            }
            file++;
        }
    }
    
    // Castle rights
    pos.castle = 0;
    if (castleStr != "-") {
        for (char c : castleStr) {
            switch (c) {
                case 'K': pos.castle |= 1; break;
                case 'Q': pos.castle |= 2; break;
                case 'k': pos.castle |= 4; break;
                case 'q': pos.castle |= 8; break;
            }
        }
    }
    
    // En passant
    if (epStr != "-") {
        int f = epStr[0] - 'a';
        int r = epStr[1] - '1';
        pos.ep = mkSq(r, f);
    }
    
    // Compute hash
    pos.hash = 0;
    for (int sq = 0; sq < 64; sq++) {
        if (pos.board[sq] != 0) {
            pos.hash ^= ZOBR_PIECE[sq][zobPieceIdx(pos.board[sq])];
        }
    }
    if (pos.side == BLACK) pos.hash ^= ZOBR_SIDE;
    pos.hash ^= ZOBR_CASTLE[pos.castle];
    if (pos.ep >= 0) pos.hash ^= ZOBR_EP[pos.ep];
    
    return pos;
}
```

```cpp
bool isAttacked(const Position& pos, int sq, int byColor) {
    // Check if square sq is attacked by color byColor
    int r = rankOf(sq), f = fileOf(sq);
    
    // Pawn attacks
    int pawnDir = byColor == WHITE ? 1 : -1; // direction FROM which a pawn of byColor would attack
    // Actually: white pawns attack diagonally upward (rank+1). 
    // A white pawn on (r-1, f-1) or (r-1, f+1) attacks sq (r,f)
    // So for byColor == WHITE, check (r-1, f±1)
    // For byColor == BLACK, check (r+1, f±1)
    int pawnRank = r - (byColor == WHITE ? 1 : -1);
    for (int df = -1; df <= 1; df += 2) {
        if (validSq(pawnRank, f + df)) {
            int p = pos.board[mkSq(pawnRank, f + df)];
            if (p != 0 && pieceColor(p) == byColor && pieceType(p) == PAWN) return true;
        }
    }
    
    // Knight attacks
    static const int knightDr[] = {-2,-2,-1,-1,1,1,2,2};
    static const int knightDf[] = {-1,1,-2,2,-2,2,-1,1};
    for (int i = 0; i < 8; i++) {
        int nr = r + knightDr[i], nf = f + knightDf[i];
        if (validSq(nr, nf)) {
            int p = pos.board[mkSq(nr, nf)];
            if (p != 0 && pieceColor(p) == byColor && pieceType(p) == KNIGHT) return true;
        }
    }
    
    // King attacks
    for (int dr = -1; dr <= 1; dr++) {
        for (int df = -1; df <= 1; df++) {
            if (dr == 0 && df == 0) continue;
            int nr = r + dr, nf = f + df;
            if (validSq(nr, nf)) {
                int p = pos.board[mkSq(nr, nf)];
                if (p != 0 && pieceColor(p) == byColor && pieceType(p) == KING) return true;
            }
        }
    }
    
    // Sliding pieces: bishop/queen on diagonals, rook/queen on ranks/files
    // Diagonals
    static const int diagDr[] = {-1,-1,1,1};
    static const int diagDf[] = {-1,1,-1,1};
    for (int d = 0; d < 4; d++) {
        int nr = r + diagDr[d], nf = f + diagDf[d];
        while (validSq(nr, nf)) {
            int p = pos.board[mkSq(nr, nf)];
            if (p != 0) {
                if (pieceColor(p) == byColor && (pieceType(p) == BISHOP || pieceType(p) == QUEEN))
                    return true;
                break;
            }
```

```cpp
    uint64_t key;
    int16_t score;
    int8_t depth;
    int8_t flags; // 0: exact, 1: lower bound, 2: upper bound
    uint32_t bestMove; // packed move
};

        timeoutFlag = true;
        return 0;
    }

    if (standPat >= beta) return beta;
    if (standPat > alpha) alpha = standPat;

    int alphaOrig = alpha;
    
    // TT lookup
    auto it = tt.find(pos.hash);
    Move ttMove;
    memset(&ttMove, 0, sizeof(ttMove));
    bool ttHit = false;
    if (it != tt.end() && it->second.key == pos.hash) {
        ttHit = true;
        ttMove.from = it->second.bestMove & 0xFF;
        ttMove.to = (it->second.bestMove >> 8) & 0xFF;
        ttMove.flags = (it->second.bestMove >> 16) & 0xFF;
        ttMove.promotion = (it->second.bestMove >> 20) & 0xF;
        
        if (it->second.depth >= depth) {
            if (it->second.flags == TT_EXACT) return it->second.score;
            if (it->second.flags == TT_LOWER) alpha = max(alpha, it->second.score);
            if (it->second.flags == TT_UPPER) beta = min(beta, it->second.score);
            if (alpha >= beta) return it->second.score;

    vector<Move> moves;
    generateMoves(pos, moves);

        return 0; // stalemate
```
