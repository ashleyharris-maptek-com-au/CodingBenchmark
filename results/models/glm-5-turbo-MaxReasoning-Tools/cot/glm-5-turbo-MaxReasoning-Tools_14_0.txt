- Count flagged neighbors F and unknown neighbors

### Algorithm 2: Set-based Reduction

- If constraint A: unknowns_A, mines_A
- If constraint B: unknowns_B, mines_B, where unknowns_A ⊂ unknowns_B
- Then we get a new constraint: unknowns_B \ unknowns_A, mines_B - mines_A
- If the new constraint has 0 mines: all those cells are safe
- If the new constraint has same mines as cells: all those cells are mines

### Connected Components

After basic propagation, find connected components of unknowns. Two unknowns are connected if they share a numbered neighbor.

```csharp
static void SolveComponents()
{
    // Find remaining unknowns
    var unknownSet = new HashSet<int>();
    for (int i = 0; i < H * W; i++)
        if (state[i] == 0) unknownSet.Add(i);
    
    if (unknownSet.Count == 0) return;
    
    // Union-Find
    int[] parent = new int[H * W];
    int[] rank = new int[H * W];
    for (int i = 0; i < parent.Length; i++) parent[i] = i;
    
    int Find(int x)
    {
        while (parent[x] != x)
        {
            parent[x] = parent[parent[x]];
            x = parent[x];
        }
        return x;
    }
    
    void Union(int a, int b)
    {
        a = Find(a); b = Find(b);
        if (a == b) return;
        if (rank[a] < rank[b]) { int t = a; a = b; b = t; }
        parent[b] = a;
        if (rank[a] == rank[b]) rank[a]++;
    }
    
    // Union unknowns that share a numbered neighbor
    for (int y = 0; y < H; y++)
    {
        for (int x = 0; x < W; x++)
        {
            if (board[y][x] < '0' || board[y][x] > '8') continue;
            
            int idx = y * W + x;
            int number = board[y][x] - '0';
```

```csharp
// For each connected component with k unknowns and m constraints:
// Set up m×(k+1) augmented matrix
// Row reduce
// Check for rows with single variable

static void GaussianElimination(List<(HashSet<int> cells, int mines)> constraints, 
    List<int> unknowns, Dictionary<int, int> cellToIdx)
{
    int m = constraints.Count;
    int k = unknowns.Count;
    if (m == 0 || k == 0) return;
    
    // Build augmented matrix (m × (k+1))
    var matrix = new int[m][];
    for (int i = 0; i < m; i++)
    {
        matrix[i] = new int[k + 1];
        foreach (int cell in constraints[i].cells)
            matrix[i][cellToIdx[cell]] = 1;
        matrix[i][k] = constraints[i].mines;
    }
    
    // Gaussian elimination with partial pivoting
    int[] pivotRow = new int[k];
    for (int i = 0; i < k; i++) pivotRow[i] = -1;
    
    int currentRow = 0;
    for (int col = 0; col < k && currentRow < m; col++)
    {
        // Find pivot
        int pivot = -1;
        for (int row = currentRow; row < m; row++)
            if (matrix[row][col] != 0) { pivot = row; break; }
        
        if (pivot == -1) continue;
        
        // Swap
        var temp = matrix[currentRow];
        matrix[currentRow] = matrix[pivot];
        matrix[pivot] = temp;
        
        pivotRow[col] = currentRow;
        
        // Eliminate
        for (int row = 0; row < m; row++)
        {
            if (row == currentRow || matrix[row][col] == 0) continue;
            int factor = matrix[row][col];
            for (int j = 0; j <= k; j++)
                matrix[row][j] -= factor * matrix[currentRow][j];
        }
        
        currentRow++;
    }
    
    // Check for determined variables
    for (int col = 0; col < k; col++)
    {
        if (pivotRow[col] == -1) continue;
        int row = pivotRow[col];
        
        // Check if this row has only this variable
        bool singleVar = true;
        for (int j = 0; j < k; j++)
            if (j != col && matrix[row][j] != 0) { singleVar = false; break; }
        
        if (singleVar)
        {
            int rhs = matrix[row][k];
            if (matrix[row][col] == 1)
            {
                if (rhs == 0) state[unknowns[col]] = 3; // safe
                else if (rhs == 1) state[unknowns[col]] = 4; // mine
            }
            else if (matrix[row][col] == -1)
            {
                if (rhs == 0) state[unknowns[col]] = 3; // safe (-x = 0 → x = 0)
                else if (rhs == -1) state[unknowns[col]] = 4; // mine (-x = -1 → x = 1)
            }
        }
    }
}
```

                        int idx = y * W + x;
                        worklist.Enqueue(idx);
                        inWorklist[idx] = true;
                    }
                }
        
        while (worklist.Count > 0)
        {
            int idx = worklist.Dequeue();
            inWorklist[idx] = false;
            
            int y = idx / W, x = idx % W;
            if (board[y][x] < '0' || board[y][x] > '8') continue;
            
            int number = board[y][x] - '0';
            
            int mineCount = 0;
            int unknownCount = 0;
            List<int> unknowns = null;
            
            for (int d = 0; d < 8; d++)
            {
                int nx = x + DX[d], ny = y + DY[d];
                if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                {
                    int ni = ny * W + nx;
                    if (state[ni] == 2) mineCount++;
                    else if (board[ny][nx] == ' ' && state[ni] == 0)
                    {
                        unknownCount++;
                        if (unknowns == null) unknowns = new List<int>();
                        unknowns.Add(ni);
                    }
                }
            }
            
            int remaining = number - mineCount;
            
            if (remaining == 0 && unknowns != null)
            {
                foreach (int ui in unknowns)
                {
                    state[ui] = 1; // safe
                    AddNeighborsToWorklist(ui, worklist, inWorklist);
                }
            }
            else if (remaining == unknownCount && unknowns != null)
            {
                foreach (int ui in unknowns)
                {
                    state[ui] = 2; // mine
                    AddNeighborsToWorklist(ui, worklist, inWorklist);
                }
            }
        }
    }
    
    static void AddNeighborsToWorklist(int cellIdx, Queue<int> worklist, bool[] inWorklist)
    {
        int x = cellIdx % W, y = cellIdx / W;
        for (int d = 0; d < 8; d++)
        {
            int nx = x + DX[d], ny = y + DY[d];
            if (nx >= 0 && nx < W && ny >= 0 && ny < H)
            {
                int ni = ny * W + nx;
                if (board[ny][nx] >= '0' && board[ny][nx] <= '8' && !inWorklist[ni])
                {
                    worklist.Enqueue(ni);
                    inWorklist[ni] = true;
                }
            }
        }
    }
    
    static void AdvancedSolve()
    {
        // Find remaining unknowns
        var unknowns = new List<int>();
        for (int i = 0; i < H * W; i++)
            if (board[i / W][i % W] == ' ' && state[i] == 0)
                unknowns.Add(i);
        
        if (unknowns.Count == 0) return;
        
        // Build cell-to-index map
        var cellToIdx = new Dictionary<int, int>();
        for (int i = 0; i < unknowns.Count; i++)
            cellToIdx[unknowns[i]] = i;
        
        // Find connected components using Union-Find
        int[] parent = new int[unknowns.Count];
        for (int i = 0; i < parent.Length; i++) parent[i] = i;
        
        int Find(int x)
        {
            while (parent[x] != x) { parent[x] = parent[parent[x]]; x = parent[x]; }
            return x;
        }
        
        void Union(int a, int b)
        {
            a = Find(a); b = Find(b);
            if (a != b) parent[b] = a;
        }
        
        // For each numbered cell, union its unknown neighbors
        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                if (board[y][x] < '0' || board[y][x] > '8') continue;
                
                int first = -1;
                for (int d = 0; d < 8; d++)
                {
                    int nx = x + DX[d], ny = y + DY[d];
                    if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                    {
                        int ni = ny * W + nx;
                        if (board[ny][nx] == ' ' && state[ni] == 0 && cellToIdx.ContainsKey(ni))
                        {
                            int idx = cellToIdx[ni];
                            if (first == -1) first = idx;
                            else Union(first, idx);
                        }
                    }
                }
            }
        }
        
        // Group by component
        var components = new Dictionary<int, List<int>>();
        for (int i = 0; i < unknowns.Count; i++)
        {
            int root = Find(i);
            if (!components.ContainsKey(root))
                components[root] = new List<int>();
            components[root].Add(i);
        }
        
        // Solve each component
        foreach (var kvp in components)
        {
            var compIndices = kvp.Value;
            var compCells = compIndices.Select(i => unknowns[i]).ToList();
            SolveComponent(compCells);
        }
    }
    
    static void SolveComponent(List<int> compCells)
    {
        // Build constraints for this component
        var cellSet = new HashSet<int>(compCells);
        var constraints = new List<(HashSet<int> cells, int mines)>();
        
        var borderNumbers = new HashSet<int>();
        foreach (int cell in compCells)
        {
            int x = cell % W, y = cell / W;
            for (int d = 0; d < 8; d++)
            {
                int nx = x + DX[d], ny = y + DY[d];
                if (nx >= 0 && nx < W && ny >= 0 && ny < H && board[ny][nx] >= '0' && board[ny][nx] <= '8')
                    borderNumbers.Add(ny * W + nx);
            }
        }
        
        foreach (int ni in borderNumbers)
        {
            int ny = ni / W, nx = ni % W;
            int number = board[ny][nx] - '0';
            
            int mineCount = 0;
            var unknowns = new HashSet<int>();
            
            for (int d = 0; d < 8; d++)
            {
                int nnx = nx + DX[d], nny = ny + DY[d];
                if (nnx >= 0 && nnx < W && nny >= 0 && nny < H)
                {
                    int idx = nny * W + nnx;
                    if (state[idx] == 2) mineCount++;
                    else if (board[nny][nnx] == ' ' && state[idx] == 0 && cellSet.Contains(idx))
                        unknowns.Add(idx);
                }
            }
            
            int remaining = number - mineCount;
            if (unknowns.Count > 0 && remaining >= 0 && remaining <= unknowns.Count)
                constraints.Add((unknowns, remaining));
        }
        
        if (constraints.Count == 0) return;
        
        // Remove trivial constraints and apply deductions
        bool changed = true;
        while (changed)
        {
            changed = false;
            var newConstraints = new List<(HashSet<int>, int)>();
            
            // Check each constraint for trivial deduction
            var activeConstraints = new List<(HashSet<int> cells, int mines)>();
            foreach (var c in constraints)
            {
                // Remove cells that have been deduced
                var remainingCells = new HashSet<int>();
                foreach (int cell in c.cells)
                    if (state[cell] == 0) remainingCells.Add(cell);
                
                int deductedMines = 0;
                foreach (int cell in c.cells)
                    if (state[cell] == 2) deductedMines++;
                
                int newMines = c.mines - deductedMines;
                
                if (newMines == 0 && remainingCells.Count > 0)
                {
                    foreach (int cell in remainingCells) state[cell] = 1;
                    changed = true;
                }
                else if (newMines == remainingCells.Count && remainingCells.Count > 0)
                {
                    foreach (int cell in remainingCells) state[cell] = 2;
                    changed = true;
                }
                else if (remainingCells.Count > 0)
                    activeConstraints.Add((remainingCells, newMines));
            }
            
            if (changed)
            {
                constraints = activeConstraints;
                BasicPropagation();
                continue;
            }
            
            // Subset reduction
            if (activeConstraints.Count < 5000) // Limit for performance
            {
                // Sort by size
                activeConstraints.Sort((a, b) => a.cells.Count.CompareTo(b.cells.Count));
                
                // Build cell-to-constraint index
                var cellToConstraints = new Dictionary<int, List<int>>();
                for (int i = 0; i < activeConstraints.Count; i++)
                    foreach (int cell in activeConstraints[i].cells)
                    {
                        if (!cellToConstraints.ContainsKey(cell))
                            cellToConstraints[cell] = new List<int>();
                        cellToConstraints[cell].Add(i);
                    }
                
                // Check pairs
                for (int i = 0; i < activeConstraints.Count && !changed; i++)
                {
                    if (activeConstraints[i].cells.Count == 0) continue;
                    
                    // Find candidate supersets
                    var candidates = new HashSet<int>();
                    foreach (int cell in activeConstraints[i].cells)
                    {
                        if (cellToConstraints.TryGetValue(cell, out var list))
                            foreach (int j in list)
                                if (j > i && activeConstraints[j].cells.Count > activeConstraints[i].cells.Count)
                                    candidates.Add(j);
                    }
                    
                    foreach (int j in candidates)
                    {
                        if (activeConstraints[j].cells.Count == 0) continue;
                        if (activeConstraints[i].cells.IsSubsetOf(activeConstraints[j].cells))
                        {
                            var diff = new HashSet<int>(activeConstraints[j].cells);
                            foreach (int cell in activeConstraints[i].cells) diff.Remove(cell);
                            int diffMines = activeConstraints[j].mines - activeConstraints[i].mines;
                            
                            if (diff.Count > 0 && diffMines >= 0 && diffMines <= diff.Count)
                            {
                                if (diffMines == 0)
                                {
                                    foreach (int cell in diff) state[cell] = 1;
                                    changed = true;
                                }
                                else if (diffMines == diff.Count)
                                {
                                    foreach (int cell in diff) state[cell] = 2;
                                    changed = true;
                                }
                                else
                                {
                                    newConstraints.Add((diff, diffMines));
                                }
                            }
                        }
                    }
                }
            }
            
            if (changed)
            {
                constraints = activeConstraints;
                constraints.AddRange(newConstraints);
                BasicPropagation();
            }
            else
            {
                constraints = activeConstraints;
                constraints.AddRange(newConstraints);
            }
        }
        
        // Enumeration for small components
        var remaining = compCells.Where(c => state[c] == 0).ToList();
        if (remaining.Count > 0 && remaining.Count <= 25)
        {
            EnumerateSolutions(constraints, remaining);
        }
        else if (remaining.Count > 0)
        {
            // Gaussian elimination for larger components
            GaussianElimination(constraints, remaining);
        }
    }

    {
        int k = unknowns.Count;
        if (k == 0 || k > 25) return;
        
        var cellToIdx = new Dictionary<int, int>();
        for (int i = 0; i < k; i++) cellToIdx[unknowns[i
