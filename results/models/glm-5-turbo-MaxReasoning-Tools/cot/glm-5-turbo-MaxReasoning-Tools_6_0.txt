{
    double ai = orbits[i, 0], aj = orbits[j, 0];
    double ei = orbits[i, 1], ej = orbits[j, 1];
    double ii = orbits[i, 2], ij = orbits[j, 2];
    
    double a_avg = (ai + aj) / 2;
    double V_avg = Math.Sqrt(mu / a_avg);
    
    // Hohmann transfer cost
    double V_ci = Math.Sqrt(mu / ai);
    double V_cj = Math.Sqrt(mu / aj);
    
    double dv_hohmann;

    {
        dv_hohmann = 0;
    }
    else
    {
        double a1 = Math.Min(ai, aj);
        double a2 = Math.Max(ai, aj);
        double dv1 = Math.Sqrt(mu / a1) * (Math.Sqrt(2 * a2 / (a1 + a2)) - 1);
        double dv2 = Math.Sqrt(mu / a2) * (1 - Math.Sqrt(2 * a1 / (a1 + a2)));

    double dv_total_sq = dv_hohmann * dv_hohmann + dv_plane * dv_plane + dv_ecc * dv_ecc;

    static void ComputeElements(double[] r, double[] v, double mu, 
        out double semiMajor, out double eccentricity, out double inclination)
    {
        double rmag2 = r[0]*r[0] + r[1]*r[1] + r[2]*r[2];
        double vmag2 = v[0]*v[0] + v[1]*v[1] + v[2]*v[2];
        double rmag = Math.Sqrt(rmag2);
        
        // Specific orbital energy
        double energy = vmag2 / 2 - mu / rmag;
        
        // Semi-major axis
        semiMajor = (Math.Abs(energy) > 1e-10) ? -mu / (2 * energy) : 1e10;
        
        // Angular momentum h = r × v
        double hx = r[1]*v[2] - r[2]*v[1];
        double hy = r[2]*v[0] - r[0]*v[2];
        double hz = r[0]*v[1] - r[1]*v[0];
        double hmag = Math.Sqrt(hx*hx + hy*hy + hz*hz);
        
        // Eccentricity vector: e = (v × h)/μ - r/|r|
        // v × h:
        double vxhx = v[1]*hz - v[2]*hy;
        double vxhy = v[2]*hx - v[0]*hz;
        double vxhz = v[0]*hy - v[1]*hx;
        
        double ex = vxhx / mu - r[0] / rmag;
        double ey = vxhy / mu - r[1] / rmag;
        double ez = vxhz / mu - r[2] / rmag;
        
        eccentricity = Math.Sqrt(ex*ex + ey*ey + ez*ez);
        
        // Inclination
        inclination = (hmag > 1e-10) ? Math.Acos(Math.Max(-1, Math.Min(1, hz / hmag))) : 0;
    }
    
    static double ComputeCost(double ai, double ei, double ii, double vi, 
        double aj, double ej, double ij, double vj, double mu)
    {
        // Energy-based cost: difference in circular velocities
        // For elliptical orbits, use circular velocity at SMA
        // For hyperbolic (a < 0), use velocity magnitude
        
        double vci, vcj;
        if (ai > 0) vci = Math.Sqrt(mu / ai);
        else vci = vi;
        
        if (aj > 0) vcj = Math.Sqrt(mu / aj);
        else vcj = vj;
        
        // Hohmann-like cost for energy change
        double a1 = Math.Abs(ai), a2 = Math.Abs(aj);
        double amin = Math.Min(a1, a2);
        double amax = Math.Max(a1, a2);
        
        double dv_energy;
        if (amin < 1e-6)
        {
            dv_energy = Math.Abs(vci - vcj);
        }
        else
        {
            // Hohmann transfer delta-V
            double r1 = amin, r2 = amax;
            double dv1 = Math.Sqrt(mu / r1) * (Math.Sqrt(2 * r2 / (r1 + r2)) - 1);
            double dv2 = Math.Sqrt(mu / r2) * (1 - Math.Sqrt(2 * r1 / (r1 + r2)));
            dv_energy = Math.Abs(dv1) + Math.Abs(dv2);
        }
        
        // Plane change cost
        double v_avg = (vci + vcj) / 2;
        double di = Math.Abs(ii - ij);
        if (di > Math.PI) di = 2 * Math.PI - di;
        double dv_plane = v_avg * 2 * Math.Sin(di / 2);
        
        // Eccentricity matching cost
        double dv_ecc = v_avg * Math.Abs(ei - ej);
        
        return Math.Sqrt(dv_energy * dv_energy + dv_plane * dv_plane + dv_ecc * dv_ecc);
    }

    static int[] HeldKarp(int n)
    {
        // n stations, indexed 0..n-1
        // In cost matrix, stations are at indices 1..n (0 is start)
        
        int fullMask = (1 << n) - 1;
        double[,] dp = new double[1 << n, n];
        int[,] parent = new int[1 << n, n];
        
        for (int i = 0; i < (1 << n); i++)
            for (int j = 0; j < n; j++)
                dp[i, j] = double.MaxValue;
        
        // Initialize: go from start (index 0 in cost) to station j
        for (int j = 0; j < n; j++)
        {
            dp[1 << j, j] = cost[0, j + 1];
            parent[1 << j, j] = -1;
        }
        
        // DP
        for (int mask = 1; mask <= fullMask; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0) continue;
                if (dp[mask, last] == double.MaxValue) continue;
                
                for (int next = 0; next < n; next++)
                {
                    if ((mask & (1 << next)) != 0) continue;
                    int newMask = mask | (1 << next);
                    double newCost = dp[mask, last] + cost[last + 1, next + 1];
                    if (newCost < dp[newMask, next])
                    {
                        dp[newMask, next] = newCost;
                        parent[newMask, next] = last;
                    }
                }
            }
        }
        
        // Find best ending station
        double bestCost = double.MaxValue;
        int bestLast = -1;
        for (int last = 0; last < n; last++)
        {
            if (dp[fullMask, last] < bestCost)
            {
                bestCost = dp[fullMask, last];
                bestLast = last;
            }
        }
        
        // Reconstruct path
        int[] path = new int[n];
        int mask = fullMask;
        int cur = bestLast;
        for (int idx = n - 1; idx >= 0; idx--)
        {
            path[idx] = cur;
            int prev = parent[mask, cur];
            mask ^= (1 << cur);
            cur = prev;
        }
        
        return path;
    }

Wait, one more thing about the Held-Karp: for path TSP (not cycle), the formulation is:
- Start at node 0 (start orbit), visit all N stations
- No need to return to start

dp[mask][i] = minimum cost to visit the set of stations indicated by mask, with station i being the last visited.

Initialize: dp[1 << j][j] = cost[0][j+1] for each station j (0-indexed)

Transition: for each state (mask, i), try to go to station j not in mask:
  dp[mask | (1<<j)][j] = min(dp[mask | (1<<j)][j], dp[mask][i] + cost[i+1][j+1])

Answer: min over i of dp[(1<<N)-1][i]

This is correct for the path version.
