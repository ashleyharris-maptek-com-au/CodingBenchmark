function bfs(startX: number, startY: number, grid: string[], W: number, H: number): Int32Array {
    const dist = new Int32Array(W * H).fill(-1);
    const queue: number[] = [];
    const head = startIdx = startY * W + startX;
    dist[head] = 0;
    queue.push(head);
    let front = 0;
    while (front < queue.length) {
        const idx = queue[front++];
        const d = dist[idx];
        const x = idx % W;
        const y = (idx - x) / W;
        // 4 neighbors
        for (const [dx, dy] of [[0,1],[0,-1],[1,0],[-1,0]]) {
            const nx = x + dx, ny = y + dy;
            if (nx >= 0 && nx < W && ny >= 0 && ny < H) {
                const nIdx = ny * W + nx;
                if (dist[nIdx] === -1 && grid[ny][nx] === '.') {
                    dist[nIdx] = d + 1;
                    queue.push(nIdx);
                }
            }
        }
    }
    return dist;
}

function bfs(sx: number, sy: number, grid: Uint8Array, W: number, H: number): Int32Array {

    const dist = new Int32Array(N).fill(-1);
    const q = new Int32Array(N);
    let head = 0, tail = 0;
    const startIdx = sy * W + sx;
    dist[startIdx] = 0;
