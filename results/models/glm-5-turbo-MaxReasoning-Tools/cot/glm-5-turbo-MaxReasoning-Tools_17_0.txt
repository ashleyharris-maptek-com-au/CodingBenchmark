using System;
using System.Collections.Generic;
using System.Linq;

static class Program
{
    struct Box
    {
        public int w, h, d;
        public int Volume => w * h * d;
    }
    
    struct PlacedBox
    {
        public int x, y, z, w, h, d, index;
    }
    
    struct Point3D : IComparable<Point3D>
    {
        public int x, y, z;
        
        public int CompareTo(Point3D other)
        {
            // BLB order: y first, then x, then z
            int cmp = y.CompareTo(other.y);
            if (cmp != 0) return cmp;
            cmp = x.CompareTo(other.x);
            if (cmp != 0) return cmp;
            return z.CompareTo(other.z);
        }
    }
    
    static void Main()
    {
        // Parse input
        string line = Console.ReadLine();
        string[] parts = line.Split();
        int N = int.Parse(parts[0]);
        int W = int.Parse(parts[1]);
        int H = int.Parse(parts[2]);
        int D = int.Parse(parts[3]);
        
        Box[] boxes = new Box[N];
        for (int i = 0; i < N; i++)
        {
            parts = Console.ReadLine().Split();
            boxes[i].w = int.Parse(parts[0]);
            boxes[i].h = int.Parse(parts[1]);
            boxes[i].d = int.Parse(parts[2]);
        }
        
        // Sort by volume descending
        Array.Sort(boxes, (a, b) => b.Volume.CompareTo(a.Volume));
        
        List<PlacedBox> placed = new List<PlacedBox>();
        List<Point3D> extremePoints = new List<Point3D>();
        extremePoints.Add(new Point3D { x = 0, y = 0, z = 0 });
        
        for (int i = 0; i < N; i++)
        {
            Box box = boxes[i];
            int[,] orientations = new int[6, 3];
            orientations[0,0] = box.w; orientations[0,1] = box.h; orientations[0,2] = box.d;
            orientations[1,0] = box.w; orientations[1,1] = box.d; orientations[1,2] = box.h;
            orientations[2,0] = box.h; orientations[2,1] = box.w; orientations[2,2] = box.d;
            orientations[3,0] = box.h; orientations[3,1] = box.d; orientations[3,2] = box.w;
            orientations[4,0] = box.d; orientations[4,1] = box.w; orientations[4,2] = box.h;
            orientations[5,0] = box.d; orientations[5,1] = box.h; orientations[5,2] = box.w;
            
            bool found = false;
            Point3D bestPoint = default;
            int bestOri = -1;
            
            // Sort extreme points by BLB order
            extremePoints.Sort();
            
            for (int ori = 0; ori < 6 && !found; ori++)
            {
                int bw = orientations[ori, 0];
                int bh = orientations[ori, 1];
                int bd = orientations[ori, 2];
                
                foreach (var pt in extremePoints)
                {
                    if (pt.x + bw > W || pt.y + bh > H || pt.z + bd > D)
                        continue;
                    
                    // Check collision
                    bool collision = false;
                    foreach (var pb in placed)
                    {
                        if (pt.x < pb.x + pb.w && pt.x + bw > pb.x &&
                            pt.y < pb.y + pb.h && pt.y + bh > pb.y &&
                            pt.z < pb.z + pb.d && pt.z + bd > pb.z)
                        {
                            collision = true;
                            break;
                        }
                    }
                    
                    if (!collision)
                    {
                        bestPoint = pt;
                        bestOri = ori;
                        found = true;
                        break;
                    }
                }
            }
            
            if (found)
            {
                placed.Add(new PlacedBox
                {
                    index = i,
                    x = bestPoint.x, y = bestPoint.y, z = bestPoint.z,
                    w = orientations[bestOri, 0],
                    h = orientations[bestOri, 1],
                    d = orientations[bestOri, 2]
                });
                
                // Add new extreme points
                extremePoints.Add(new Point3D { x = bestPoint.x + orientations[bestOri, 0], y = bestPoint.y, z = bestPoint.z });
                extremePoints.Add(new Point3D { x = bestPoint.x, y = bestPoint.y + orientations[bestOri, 1], z = bestPoint.z });
                extremePoints.Add(new Point3D { x = bestPoint.x, y = bestPoint.y, z = bestPoint.z + orientations[bestOri, 2] });
            }
        }
        
        Console.WriteLine(placed.Count);
        for (int i = 0; i < placed.Count; i++)
        {
            Console.WriteLine($"{placed[i].index} {placed[i].x} {placed[i].y} {placed[i].z}");
        }
    }
}

class SpatialHash
{
    int cellSize;
    Dictionary<(int, int, int), List<int>> cells = new Dictionary<(int, int, int), List<int>>();
    
    public SpatialHash(int cellSize)
    {
        this.cellSize = cellSize;
    }
    
    (int, int, int) CellKey(int x, int y, int z)
    {
        return (x / cellSize, y / cellSize, z / cellSize);
    }
    
    public void Insert(int boxIdx, int x, int y, int z, int w, int h, int d)
    {
        for (int cx = x; cx < x + w; cx += cellSize)
            for (int cy = y; cy < y + h; cy += cellSize)
                for (int cz = z; cz < z + d; cz += cellSize)
                {
                    var key = CellKey(cx, cy, cz);
                    if (!cells.ContainsKey(key))
                        cells[key] = new List<int>();
                    cells[key].Add(boxIdx);
                }
    }
    
    public IEnumerable<int> Query(int x, int y, int z, int w, int h, int d)
    {
        var seen = new HashSet<int>();
        for (int cx = x; cx < x + w; cx += cellSize)
            for (int cy = y; cy < y + h; cy += cellSize)
                for (int cz = z; cz < z + d; cz += cellSize)
                {
                    var key = CellKey(cx, cy, cz);
                    if (cells.TryGetValue(key, out var list))
                        foreach (int idx in list)
                            if (seen.Add(idx))
                                yield return idx;
                }
    }
}
