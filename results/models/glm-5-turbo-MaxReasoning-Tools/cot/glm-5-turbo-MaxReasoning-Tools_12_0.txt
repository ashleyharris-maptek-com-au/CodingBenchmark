```csharp
List<Quat> Generate24Rotations() {
    List<Quat> rotations = new List<Quat>();
    int[][] perms = {
        new int[]{0,1,2}, new int[]{0,2,1},
        new int[]{1,0,2}, new int[]{1,2,0},
        new int[]{2,0,1}, new int[]{2,1,0}
    };
    for (int p = 0; p < 6; p++) {
        for (int sx = -1; sx <= 1; sx += 2) {
            for (int sy = -1; sy <= 1; sy += 2) {
                for (int sz = -1; sz <= 1; sz += 2) {
                    // Build rotation matrix
                    double[,] m = new double[3,3];
                    int[] perm = perms[p];
                    m[0, perm[0]] = sx;
                    m[1, perm[1]] = sy;
                    m[2, perm[2]] = sz;
                    // Check determinant
                    double det = m[0,0]*(m[1,1]*m[2,2]-m[1,2]*m[2,1])
                               - m[0,1]*(m[1,0]*m[2,2]-m[1,2]*m[2,0])
                               + m[0,2]*(m[1,0]*m[2,1]-m[1,1]*m[2,0]);
                    if (det > 0) {
                        // Convert to quaternion
                        Quat q = MatrixToQuat(m);
                        rotations.Add(q);
                    }
                }
            }
        }
    }
    return rotations;
}
```

```csharp
            double s = Math.Sqrt(tr + 1.0) * 2;
            w = s / 4;
            x = (m[2,1] - m[1,2]) / s;
            y = (m[0,2] - m[2,0]) / s;
            z = (m[1,0] - m[0,1]) / s;
        }
        else if (m[0,0] > m[1,1] && m[0,0] > m[2,2])
        {
            double s = Math.Sqrt(1.0 + m[0,0] - m[1,1] - m[2,2]) * 2;
            w = (m[2,1] - m[1,2]) / s;
            x = s / 4;
            y = (m[0,1] + m[1,0]) / s;
            z = (m[0,2] + m[2,0]) / s;
        }
        else if (m[1,1] > m[2,2])
        {
            double s = Math.Sqrt(1.0 + m[1,1] - m[0,0] - m[2,2]) * 2;
            w = (m[0,2] - m[2,0]) / s;
            x = (m[0,1] + m[1,0]) / s;
            y = s / 4;
            z = (m[1,2] + m[2,1]) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m[2,2] - m[0,0] - m[1,1]) * 2;
            w = (m[1,0] - m[0,1]) / s;
            x = (m[0,2] + m[2,0]) / s;
            y = (m[1,2] + m[2,1]) / s;
            z = s / 4;
        }
        // Normalize
        double len = Math.Sqrt(w*w + x*x + y*y + z*z);
        return new Quat(w/len, x/len, y/len, z/len);
    }
    
    static Vec3[] RotateVerts(Vec3[] verts, Quat q)
    {
        Vec3[] rv = new Vec3[verts.Length];
        for (int i = 0; i < verts.Length; i++)
            rv[i] = q.Rotate(verts[i]);
        return rv;
    }
    
    static void ComputeAABB(Vec3[] verts, out double minX, out double minY, out double minZ,
                             out double maxX, out double maxY, out double maxZ)
    {
        minX = minY = minZ = double.MaxValue;
        maxX = maxY = maxZ = double.MinValue;
        foreach (var v in verts)
        {
            if (v.x < minX) minX = v.x; if (v.x > maxX) maxX = v.x;
            if (v.y < minY) minY = v.y; if (v.y > maxY) maxY = v.y;
            if (v.z < minZ) minZ = v.z; if (v.z > maxZ) maxZ = v.z;
        }
    }
```
