**Finding the critical path:**
The makespan is determined by the critical path in the disjunctive graph. The critical path goes through a sequence of operations, alternating between job-constrained arcs (precedence within a job) and machine-constrained arcs (precedence on a machine).

To find the critical path:
1. For each operation, compute its earliest start time (forward pass) - this is already done by the schedule.
2. Find the operation that finishes last (= makespan - duration, finishing at makespan).
3. Trace backwards: from the last operation, look at what precedes it - either the previous task in the same job, or the previous operation on the same machine. Pick the one with the later finish time.
4. Continue until we reach the start.

**Critical blocks:**
A critical block is a maximal sequence of consecutive operations on the same machine that are all on the critical path.

**Neighborhood move:**
For a critical block with operations (o1, o2, ..., ok) on machine m:
- Try moving o1 to just after ok (i.e., schedule o1 after the block)
- Try moving ok to just before o1 (i.e., schedule ok before the block)
- Try swapping adjacent operations within the block

After each move, we need to recompute the schedule to check feasibility and compute the new makespan.

Solution solve_greedy(int rule, mt19937& rng) {
    vector<int> order;
    order.reserve(total_tasks);
    
    vector<int> machine_avail(M, 0);
    vector<int> job_avail(J, 0);
    vector<int> job_next(J, 0);
    
    auto priority = [&](int j) -> double {
        int t = job_next[j];
        int dur = jobs[j][t].second;
        int rw = rem_work[j][t];
        double rnd = uniform_real_distribution<double>(0, 1)(rng);

            case MWR: return -rw;
            case LWR: return rw;
            case SPT: return dur;
            case LPT: return -dur;
            case FIFO: return j;
            case RANDOM: return -rnd;
            case RANDOM_MWR: return -rw * (0.8 + 0.4 * rnd); // MWR with randomness
            case LPT_MWR: return -(dur + rw) * (0.8 + 0.4 * rnd);

Solution solve_greedy(int rule, mt19937& rng) {
    vector<int> machine_avail(M, 0);
    vector<int> job_next(J, 0);
    vector<int> job_avail_time(J, 0);
    
    struct Entry { int job; double prio; };
    auto cmp = [](const Entry& a, const Entry& b) { return a.prio > b.prio; };
    priority_queue<Entry, vector<Entry>, decltype(cmp)> pq(cmp);
    
    vector<int> order;
    order.reserve(total_tasks);
    vector<int> start_flat(total_tasks);
    int makespan = 0;
    
    auto compute_prio = [&](int j) -> double {
        int t = job_next[j];
        if (t >= (int)jobs[j].size()) return 1e18;
        int dur = jobs[j][t].second;
        // remaining work from task t to end
        int rw = 0;
        for (int k = t; k < (int)jobs[j].size(); k++) rw += jobs[j][k].second;
        
        switch (rule) {
            case 0: return -(double)rw; // MWR
            case 1: return (double)rw;  // LWR
            case 2: return (double)dur; // SPT
            case 3: return -(double)dur; // LPT
            case 4: return (double)j;   // FIFO
            case 5: { // RANDOM
                uniform_real_distribution<double> ud(0, 1);
                return -ud(rng);
            }
            case 6: { // RANDOM_MWR
                uniform_real_distribution<double> ud(0, 1);
                return -(double)rw * (0.7 + 0.6 * ud(rng));
            }
            case 7: { // LPT_MWR
                uniform_real_distribution<double> ud(0, 1);
                return -((double)dur * 2 + (double)rw) * (0.7 + 0.6 * ud(rng));
            }
            case 8: { // SPT_MWR
                uniform_real_distribution<double> ud(0, 1);
                return ((double)dur - (double)rw) * (0.7 + 0.6 * ud(rng));
            }
        }
        return 0;
    };
    
    for (int j = 0; j < J; j++) {
        if (jobs[j].size() > 0) {
            pq.push({j, compute_prio(j)});
        }
    }
    
    while (!pq.empty()) {
        Entry e = pq.top(); pq.pop();
        int j = e.job;
        int t = job_next[j];
        int m = jobs[j][t].first;
        int d = jobs[j][t].second;
        
        int s = max(machine_avail[m], job_avail_time[j]);
        machine_avail[m] = s + d;
        job_avail_time[j] = s + d;
        start_flat[op_index(j, t)] = s;
        makespan = max(makespan, s + d);
        
        order.push_back(j);
        job_next[j]++;
        
        if (job_next[j] < (int)jobs[j].size()) {
            pq.push({j, compute_prio(j)});
        }
    }
    
    Solution sol;
    sol.order = order;
    sol.start_flat = start_flat;
    sol.makespan = makespan;

To find the critical path from a schedule (start_flat):
1. Build machine operation lists (sorted by start time).
2. Find the operation finishing at makespan.
3. Trace back: at each step, the predecessor is either:
   a. The previous task of the same job (job arc)
   b. The previous operation on the same machine (machine arc)
   Pick the one that finishes later.
