double ez = target_z - z;

double omega = clamp(sqrt(2.0 * max_thrust / max(dist, 1.0)), 0.01, 5.0);

double thrust = clamp(a_mag / max_thrust, 0.0, 1.0);

    double scale = min(a_mag, max_thrust) / a_mag;
    ax *= scale; ay *= scale; az *= scale;
}
