```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Scale UV to create the brick grid (8 columns, 16 rows)
    float2 scaledUV = input.uv * float2(8.0, 16.0);

    // Stagger every other row by half a brick
    float row = floor(scaledUV.y);
    float offset = fmod(row, 2.0) * 0.5;
    scaledUV.x += offset;

    // Position within the current brick cell
    float2 cellPos = frac(scaledUV);

    // Mortar thickness thresholds
    float mortarThickX = 0.05;
    float mortarThickY = 0.1;

    // Determine if the current fragment is in mortar
    bool isMortarX = cellPos.x < mortarThickX;
    bool isMortarY = cellPos.y < mortarThickY;
    bool isMortar = isMortarX || isMortarY;

    // Brick and mortar colors
    float3 brickColor  = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);

    // Select base color
    float3 baseColor = isMortar ? mortarColor : brickColor;

    // Lambertian diffuse shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float  diffuse = max(dot(N, L), 0.0);
    float  shade   = 0.4 + 0.6 * diffuse;

    // Final shaded color
    float3 finalColor = baseColor * shade;

    return float4(finalColor, 1.0);
}
```
