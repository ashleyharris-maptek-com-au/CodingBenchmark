; SPIR-V 1.0
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1
               OpSource GLSL 450

; ---- Names ----
               OpName %main "main"
               OpName %gl_PerVertex_in "gl_PerVertex"
               OpMemberName %gl_PerVertex_in 0 "gl_Position"
               OpName %gl_PerVertex_out "gl_PerVertex"
               OpMemberName %gl_PerVertex_out 0 "gl_Position"
               OpName %gl_in "gl_in"
               OpName %in_worldPos "worldPos"
               OpName %in_normal "normal"
               OpName %in_color "color"
               OpName %gl_out "gl_out"
               OpName %outColor "outColor"

; ---- Decorations ----
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %in_color Location 2
               OpDecorate %outColor Location 0

; ---- Types ----
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %uint = OpTypeUInt 32
        %int = OpTypeInt 32 1

; ---- Constants ----
     %uint_3 = OpConstant %uint 3
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
   %float_3 = OpConstant %float 3.0
   %float_s = OpConstant %float 0.7
      %v4_3 = OpConstantComposite %v4float %float_3 %float_3 %float_3 %float_3

; ---- gl_PerVertex input block (array of 3) ----
%gl_PerVertex_in = OpTypeStruct %v4float
    %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
 %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
    %ptr_in_v4 = OpTypePointer Input %v4float

; ---- gl_PerVertex output block ----
%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
   %ptr_out_v4 = OpTypePointer Output %v4float

; ---- Input / output vec3 arrays ----
    %arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
   %ptr_in_v3 = OpTypePointer Input %v3float
  %ptr_out_v3 = OpTypePointer Output %v3float

; ---- Function type ----
   %func_type = OpTypeFunction %void

; ---- Interface variables ----
      %gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
  %in_normal = OpVariable %ptr_in_arr_v3 Input
   %in_color = OpVariable %ptr_in_arr_v3 Input
     %gl_out = OpVariable %ptr_out_pv Output
  %outColor = OpVariable %ptr_out_v3 Output

; ======== main ========
       %main = OpFunction %void None %func_type
      %entry = OpLabel

; ---- Load the 3 input gl_Position values ----
    %ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %p0 = OpLoad %v4float %ptr_p0
    %ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %p1 = OpLoad %v4float %ptr_p1
    %ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %p2 = OpLoad %v4float %ptr_p2

; ---- Centroid: C = (p0 + p1 + p2) / 3 ----
      %sum1 = OpFAdd %v4float %p0 %p1
      %sum2 = OpFAdd %v4float %sum1 %p2
  %centroid = OpFDiv %v4float %sum2 %v4_3

; ---- Pre-compute output pointer (reused for each vertex) ----
   %out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- Emit vertex 0: new_pos = C + 0.7*(p0 - C) ----
    %diff_0 = OpFSub %v4float %p0 %centroid
 %scaled_0 = OpVectorTimesScalar %v4float %diff_0 %float_s
    %new_0 = OpFAdd %v4float %centroid %scaled_0
             OpStore %out_pos %new_0
   %ptr_c0 = OpAccessChain %ptr_in_v3 %in_color %int_0
     %col0 = OpLoad %v3float %ptr_c0
             OpStore %outColor %col0
             OpEmitVertex

; ---- Emit vertex 1: new_pos = C + 0.7*(p1 - C) ----
    %diff_1 = OpFSub %v4float %p1 %centroid
 %scaled_1 = OpVectorTimesScalar %v4float %diff_1 %float_s
    %new_1 = OpFAdd %v4float %centroid %scaled_1
             OpStore %out_pos %new_1
   %ptr_c1 = OpAccessChain %ptr_in_v3 %in_color %int_1
     %col1 = OpLoad %v3float %ptr_c1
             OpStore %outColor %col1
             OpEmitVertex

; ---- Emit vertex 2: new_pos = C + 0.7*(p2 - C) ----
    %diff_2 = OpFSub %v4float %p2 %centroid
 %scaled_2 = OpVectorTimesScalar %v4float %diff_2 %float_s
    %new_2 = OpFAdd %v4float %centroid %scaled_2
             OpStore %out_pos %new_2
   %ptr_c2 = OpAccessChain %ptr_in_v3 %in_color %int_2
     %col2 = OpLoad %v3float %ptr_c2
             OpStore %outColor %col2
             OpEmitVertex

; ---- End the triangle strip (1 triangle, 3 vertices) ----
             OpEndPrimitive
             OpReturn
               OpFunctionEnd
