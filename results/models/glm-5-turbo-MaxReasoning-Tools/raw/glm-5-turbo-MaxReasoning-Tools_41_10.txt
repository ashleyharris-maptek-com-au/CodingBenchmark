; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; Input/Output decorations
               OpDecorate %worldPos Location 0
               OpDecorate %normal   Location 1
               OpDecorate %uv       Location 2
               OpDecorate %tangent  Location 3
               OpDecorate %color    Location 4
               OpDecorate %outColor Location 0

; UBO decorations
               OpDecorate %ubo        DescriptorSet 0
               OpDecorate %ubo        Binding 0
               OpDecorate %UBO        Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224

; ---------- Types ----------
%void       = OpTypeVoid
%float      = OpTypeFloat 32
%int        = OpTypeInt 32 1
%bool       = OpTypeBool
%vec2       = OpTypeVector %float 2
%vec3       = OpTypeVector %float 3
%vec4       = OpTypeVector %float 4
%mat4       = OpTypeMatrix %vec4 4
%UBO        = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%funcType   = OpTypeFunction %void
%inPtrV3    = OpTypePointer Input %vec3
%inPtrV2    = OpTypePointer Input %vec2
%outPtrV4   = OpTypePointer Output %vec4
%uboPtr     = OpTypePointer Uniform %UBO
%uboV4Ptr   = OpTypePointer Uniform %vec4

; ---------- Constants ----------
%c0         = OpConstant %float 0.0
%c1         = OpConstant %float 1.0
%c_pi       = OpConstant %float 3.14159
%c16        = OpConstant %float 16.0
%c0_9       = OpConstant %float 0.9
%c0_1       = OpConstant %float 0.1
%c0_4       = OpConstant %float 0.4
%c0_6       = OpConstant %float 0.6
%i0         = OpConstant %int 0
%i1         = OpConstant %int 1
%i3         = OpConstant %int 3
%v3red      = OpConstantComposite %vec3 %c0_9 %c0_1 %c0_1
%v3blue     = OpConstantComposite %vec3 %c0_1 %c0_1 %c0_9

; ---------- Variables ----------
%worldPos   = OpVariable %inPtrV3  Input
%normal     = OpVariable %inPtrV3  Input
%uv         = OpVariable %inPtrV2  Input
%tangent    = OpVariable %inPtrV3  Input
%color      = OpVariable %inPtrV3  Input
%outColor   = OpVariable %outPtrV4 Output
%ubo        = OpVariable %uboPtr   Uniform

; ========== main ==========
%main       = OpFunction %void None %funcType
%entry      = OpLabel

; --- uv.y ---
%uv_load    = OpLoad %vec2 %uv
%uv_y       = OpCompositeExtract %float %uv_load 1

; --- stripe pattern: sin(uv.y * PI * 16) ---
%pi16       = OpFMul %float %c_pi %c16
%angle      = OpFMul %float %uv_y %pi16
%sin_val    = OpExtInst %float %1 Sin %angle

; select red / blue
%cond       = OpFOrdGreaterThan %bool %sin_val %c0
%stripe     = OpSelect %vec3 %cond %v3red %v3blue

; --- Lambertian shading ---
%n_raw      = OpLoad %vec3 %normal
%N          = OpExtInst %vec3 %1 Normalize %n_raw

%wp         = OpLoad %vec3 %worldPos

%lp_ptr     = OpAccessChain %uboV4Ptr %ubo %i3
%lp         = OpLoad %vec4 %lp_ptr
%lp_xyz     = OpVectorShuffle %vec3 %lp %lp 0 1 2

%to_light   = OpFSub %vec3 %lp_xyz %wp
%L          = OpExtInst %vec3 %1 Normalize %to_light

%ndotl      = OpDot %float %N %L
%diffuse    = OpExtInst %float %1 FMax %ndotl %c0

; factor = 0.4 + 0.6 * diffuse
%scaled_d   = OpFMul %float %c0_6 %diffuse
%factor     = OpFAdd %float %c0_4 %scaled_d

; shaded = stripe * factor
%shaded     = OpVectorTimesScalar %vec3 %stripe %factor

; --- output vec4(shaded, 1.0) ---
%sx         = OpCompositeExtract %float %shaded 0
%sy         = OpCompositeExtract %float %shaded 1
%sz         = OpCompositeExtract %float %shaded 2
%out        = OpCompositeConstruct %vec4 %sx %sy %sz %c1

               OpStore %outColor %out
               OpReturn
               OpFunctionEnd
