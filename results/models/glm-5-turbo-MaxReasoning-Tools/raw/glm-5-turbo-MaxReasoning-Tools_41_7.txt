; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBOType Block
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 1 Offset 16
OpMemberDecorate %UBOType 2 Offset 32
OpMemberDecorate %UBOType 3 Offset 48
OpMemberDecorate %UBOType 4 Offset 64
OpMemberDecorate %UBOType 5 Offset 80
OpMemberDecorate %UBOType 6 Offset 96
OpMemberDecorate %UBOType 7 Offset 112
OpMemberDecorate %UBOType 8 Offset 128
OpMemberDecorate %UBOType 9 Offset 144
OpMemberDecorate %UBOType 10 Offset 160
OpMemberDecorate %UBOType 11 Offset 176
OpMemberDecorate %UBOType 12 Offset 192
OpMemberDecorate %UBOType 13 Offset 208
OpMemberDecorate %UBOType 14 Offset 224

%void = OpTypeVoid
%fnTy = OpTypeFunction %void
%f32 = OpTypeFloat 32
%u32 = OpTypeInt 32 0
%v2f = OpTypeVector %f32 2
%v3f = OpTypeVector %f32 3
%v4f = OpTypeVector %f32 4

%UBOType = OpTypeStruct %v4f %v4f %v4f %v4f %v4f %v4f %v4f %v4f %v4f %v4f %v4f %v4f %v4f %v4f %v4f

%ptrInV3 = OpTypePointer Input %v3f
%ptrInV2 = OpTypePointer Input %v2f
%ptrOutV4 = OpTypePointer Output %v4f
%ptrUniformUBO = OpTypePointer Uniform %UBOType
%ptrUniformV4 = OpTypePointer Uniform %v4f

%glsl450 = OpExtInstImport "GLSL.std.450"

%c0 = OpConstant %f32 0.0
%c0_2 = OpConstant %f32 0.2
%c0_3 = OpConstant %f32 0.3
%c0_5 = OpConstant %f32 0.5
%c0_8 = OpConstant %f32 0.8
%c1 = OpConstant %f32 1.0
%c3 = OpConstant %f32 3.0
%c12 = OpConstant %u32 12
%c13 = OpConstant %u32 13

%cbase = OpConstantComposite %v3f %c0_2 %c0_3 %c0_8
%crimtint = OpConstantComposite %v3f %c1 %c0_8 %c0_5

%worldPos = OpVariable %ptrInV3 Input
%normal = OpVariable %ptrInV3 Input
%uv = OpVariable %ptrInV2 Input
%tangent = OpVariable %ptrInV3 Input
%color = OpVariable %ptrInV3 Input
%outColor = OpVariable %ptrOutV4 Output
%ubo = OpVariable %ptrUniformUBO Uniform

%main = OpFunction %void None %fnTy
%entry = OpLabel

%wp = OpLoad %v3f %worldPos
%n = OpLoad %v3f %normal

%lightPosPtr = OpAccessChain %ptrUniformV4 %ubo %c12
%cameraPosPtr = OpAccessChain %ptrUniformV4 %ubo %c13
%lightPos = OpLoad %v4f %lightPosPtr
%cameraPos = OpLoad %v4f %cameraPosPtr

%lp = OpVectorShuffle %v3f %lightPos %lightPos 0 1 2
%cp = OpVectorShuffle %v3f %cameraPos %cameraPos 0 1 2

%N = OpExtInst %v3f %glsl450 Normalize %n
%Vdir = OpFSub %v3f %cp %wp
%V = OpExtInst %v3f %glsl450 Normalize %Vdir
%Ldir = OpFSub %v3f %lp %wp
%L = OpExtInst %v3f %glsl450 Normalize %Ldir

%NdV = OpDot %f32 %N %V
%NdVc = OpExtInst %f32 %glsl450 FMax %NdV %c0
%NdL = OpDot %f32 %N %L
%diffuse = OpExtInst %f32 %glsl450 FMax %NdL %c0

%oneMinusNdV = OpFSub %f32 %c1 %NdVc
%rim = OpExtInst %f32 %glsl450 Pow %oneMinusNdV %c3

%halfDiff = OpFMul %f32 %c0_5 %diffuse
%baseFactor = OpFAdd %f32 %c0_3 %halfDiff
%baseColor = OpVectorTimesScalar %v3f %cbase %baseFactor
%rimColor = OpVectorTimesScalar %v3f %crimtint %rim
%finalColor = OpFAdd %v3f %baseColor %rimColor

%out = OpCompositeConstruct %v4f %finalColor %c1
OpStore %outColor %out

OpReturn
OpFunctionEnd
