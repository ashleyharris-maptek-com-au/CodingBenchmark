; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %inNormal %inUV %inTangent %inColor %outColor
               OpExecutionMode %main OriginUpperLeft

; decorations for inputs
               OpDecorate %worldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %inUV Location 2
               OpDecorate %inTangent Location 3
               OpDecorate %inColor Location 4

; decoration for output
               OpDecorate %outColor Location 0

; decorations for uniform buffer
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpDecorate %UBO_Block Block
               OpMemberDecorate %UBO_Block 0 Offset 0
               OpMemberDecorate %UBO_Block 1 Offset 64
               OpMemberDecorate %UBO_Block 2 Offset 128
               OpMemberDecorate %UBO_Block 3 Offset 192
               OpMemberDecorate %UBO_Block 4 Offset 208
               OpMemberDecorate %UBO_Block 5 Offset 224

; types
        %void = OpTypeVoid
     %funcTy = OpTypeFunction %void
       %float = OpTypeFloat 32
     %v2float = OpTypeVector %float 2
     %v3float = OpTypeVector %float 3
     %v4float = OpTypeVector %float 4
        %mat4 = OpTypeMatrix %v4float 4
         %int = OpTypeInt 32 1

; UBO struct type
  %UBO_Block = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; pointer types
 %ptrInV3 = OpTypePointer Input %v3float
 %ptrInV2 = OpTypePointer Input %v2float
 %ptrOutV4 = OpTypePointer Output %v4float
 %ptrUnifUBO = OpTypePointer Uniform %UBO_Block
 %ptrUnifV4 = OpTypePointer Uniform %v4float

; constants
  %float_0 = OpConstant %float 0
  %float_1 = OpConstant %float 1
    %int_0 = OpConstant %int 0
    %int_3 = OpConstant %int 3

; input variables
   %worldPos = OpVariable %ptrInV3 Input
   %inNormal = OpVariable %ptrInV3 Input
      %inUV = OpVariable %ptrInV2 Input
  %inTangent = OpVariable %ptrInV3 Input
   %inColor = OpVariable %ptrInV3 Input

; output variable
   %outColor = OpVariable %ptrOutV4 Output

; uniform buffer variable
        %ubo = OpVariable %ptrUnifUBO Uniform

; main function
       %main = OpFunction %void None %funcTy
      %entry = OpLabel

; load world-space position
  %posWorld = OpLoad %v3float %worldPos

; load normal and normalize it
  %rawNormal = OpLoad %v3float %inNormal
          %N = OpExtInst %v3float %1 Normalize %rawNormal

; access lightPos (member 3 of UBO) and extract xyz
%lightPtr = OpAccessChain %ptrUnifV4 %ubo %int_3
%lightFull = OpLoad %v4float %lightPtr
  %lightXYZ = OpVectorShuffle %v3float %lightFull %lightFull 0 1 2

; L = normalize(lightPos.xyz - worldPos)
    %toLight = OpFSub %v3float %lightXYZ %posWorld
          %L = OpExtInst %v3float %1 Normalize %toLight

; d = max(dot(N, L), 0.0)
      %dotNL = OpDot %float %N %L
          %d = OpExtInst %float %1 FMax %dotNL %float_0

; outColor = vec4(d, d, d, 1.0)
     %color = OpCompositeConstruct %v4float %d %d %d %float_1
               OpStore %outColor %color

               OpReturn
               OpFunctionEnd
