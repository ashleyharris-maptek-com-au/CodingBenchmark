```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);

    // Scale UV to brick grid
    vec2 brickUV = uv * vec2(8.0, 16.0);

    // Stagger every other row by half a brick
    float row = floor(brickUV.y);
    if (mod(row, 2.0) >= 1.0) {
        brickUV.x += 0.5;
    }

    // Fractional position within the current brick cell
    vec2 f = fract(brickUV);

    // Test mortar regions
    float mortarX = 0.05;
    float mortarY = 0.1;
    bool isMortar = (f.x < mortarX) || (f.y < mortarY);

    vec3 baseColor = isMortar ? mortarColor : brickColor;

    // Lambertian diffuse shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diff = max(dot(N, L), 0.0);

    // Combine ambient and diffuse: factor = 0.4 + 0.6*diffuse
    vec3 shaded = baseColor * (0.4 + 0.6 * diff);

    outColor = vec4(shaded, 1.0);
}
```
