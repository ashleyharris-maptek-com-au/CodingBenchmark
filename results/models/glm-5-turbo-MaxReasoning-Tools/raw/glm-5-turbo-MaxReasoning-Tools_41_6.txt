; SPIR-V 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %outColor Location 0
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224

%void       = OpTypeVoid
%funcType   = OpTypeFunction %void
%f32        = OpTypeFloat 32
%i32        = OpTypeInt 32 1
%v3f32      = OpTypeVector %f32 3
%v4f32      = OpTypeVector %f32 4
%m4f32      = OpTypeMatrix %v4f32 4
%ptrInV3    = OpTypePointer Input %v3f32
%ptrOutV4   = OpTypePointer Output %v4f32
%ptrUboV4   = OpTypePointer Uniform %v4f32
%UBO        = OpTypeStruct %m4f32 %m4f32 %m4f32 %v4f32 %v4f32 %v4f32
%ptrUBO     = OpTypePointer Uniform %UBO

%c_f0       = OpConstant %f32 0.0
%c_f1       = OpConstant %f32 1.0
%c_f01      = OpConstant %f32 0.1
%c_f06      = OpConstant %f32 0.6
%c_f08      = OpConstant %f32 0.8
%c_f64      = OpConstant %f32 64.0
%c_i3       = OpConstant %i32 3
%c_i4       = OpConstant %i32 4

%inWorldPos = OpVariable %ptrInV3 Input
%inNormal   = OpVariable %ptrInV3 Input
%outColor   = OpVariable %ptrOutV4 Output
%ubo        = OpVariable %ptrUBO Uniform

%main       = OpFunction %void None %funcType
%entry      = OpLabel

%worldPos   = OpLoad %v3f32 %inWorldPos
%N_raw      = OpLoad %v3f32 %inNormal
%N          = OpExtInst %v3f32 %1 Normalize %N_raw

%lpPtr      = OpAccessChain %ptrUboV4 %ubo %c_i3
%cpPtr      = OpAccessChain %ptrUboV4 %ubo %c_i4
%lp4        = OpLoad %v4f32 %lpPtr
%cp4        = OpLoad %v4f32 %cpPtr
%lightPos   = OpVectorShuffle %v3f32 %lp4 %lp4 0 1 2
%cameraPos  = OpVectorShuffle %v3f32 %cp4 %cp4 0 1 2

%Ldir       = OpFSub %v3f32 %lightPos %worldPos
%L          = OpExtInst %v3f32 %1 Normalize %Ldir
%Vdir       = OpFSub %v3f32 %cameraPos %worldPos
%V          = OpExtInst %v3f32 %1 Normalize %Vdir
%Hdir       = OpFAdd %v3f32 %L %V
%H          = OpExtInst %v3f32 %1 Normalize %Hdir

%NdL        = OpDot %f32 %N %L
%diffuse    = OpExtInst %f32 %1 FMax %NdL %c_f0
%NdH        = OpDot %f32 %N %H
%NdHc       = OpExtInst %f32 %1 FMax %NdH %c_f0
%specular   = OpExtInst %f32 %1 Pow %NdHc %c_f64

%dmul       = OpFMul %f32 %c_f06 %diffuse
%smul       = OpFMul %f32 %c_f08 %specular
%dsp        = OpFAdd %f32 %dmul %smul
%intensity  = OpFAdd %f32 %c_f01 %dsp

%outV4      = OpCompositeConstruct %v4f32 %intensity %intensity %intensity %c_f1
               OpStore %outColor %outV4
               OpReturn
               OpFunctionEnd
