```spirv
               OpCapability Geometry
               OpCapability Shader
         %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18
               OpExecutionMode %main Invocations 1

; === Decorations ===
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

; === Types ===
       %void = OpTypeVoid
%func_type = OpTypeFunction %void
      %bool = OpTypeBool
      %int = OpTypeInt 32 1
     %uint = OpTypeInt 32 0
    %float = OpTypeFloat 32
  %v2float = OpTypeVector %float 2
  %v3float = OpTypeVector %float 3
  %v4float = OpTypeVector %float 4

; === Constants ===
    %int_0 = OpConstant %int 0
    %int_1 = OpConstant %int 1
    %int_6 = OpConstant %int 6
   %uint_1 = OpConstant %uint 1
%float_0_04 = OpConstant %float 0.04
%float_0_7 = OpConstant %float 0.7
%float_pi_3 = OpConstant %float 1.0471975512

; === Struct types ===
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
    %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
     %arr_v3 = OpTypeArray %v3float %uint_1

; === Pointer types ===
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_func_int = OpTypePointer Function %int

; === Interface variables ===
     %gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
  %normal_in = OpVariable %ptr_in_arr_v3 Input
   %color_in = OpVariable %ptr_in_arr_v3 Input
    %gl_out = OpVariable %ptr_out_pv Output
   %outColor = OpVariable %ptr_out_v3 Output

; === Main function ===
      %main = OpFunction %void None %func_type
     %entry = OpLabel

; Load center position: gl_in[0].gl_Position
%ptr_center_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
  %center_pos = OpLoad %v4float %ptr_center_pos

; Load center color: color_in[0]
%ptr_center_col = OpAccessChain %ptr_in_v3 %color_in %int_0
 %center_color = OpLoad %v3float %ptr_center_col

; Darker color for edge vertices (70% brightness)
  %dark_color = OpVectorTimesScalar %v3float %center_color %float_0_7

; Extract center x y for offset computation
 %center_xy = OpVectorShuffle %v2float %center_pos %center_pos 0 1
 %center_z = OpCompositeExtract %float %center_pos 2
 %center_w = OpCompositeExtract %float %center_pos 3

; Output gl_Position pointer (reused every emit)
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Loop counter variable
    %i_ptr = OpVariable %ptr_func_int Function
         OpStore %i_ptr %int_0
         OpBranch %loop_header

; --- Loop header ---
%loop_header = OpLabel
   %i_val = OpLoad %int %i_ptr
  %i_lt_6 = OpSLessThan %bool %i_val %int_6
  OpBranchConditional %i_lt_6 %loop_body %loop_merge

; --- Loop body ---
%loop_body = OpLabel

; ---- Compute edge vertex 1 ----
; angle1 = float(i) * (PI / 3)
  %i_float = OpConvertSToF %float %i_val
   %angle1 = OpFMul %float %i_float %float_pi_3
    %cos1 = OpExtInst %float %1 Cos %angle1
    %sin1 = OpExtInst %float %1 Sin %angle1
; offset = radius * (cos, sin)
    %ox1 = OpFMul %float %float_0_04 %cos1
    %oy1 = OpFMul %float %float_0_04 %sin1
 %e1_off = OpCompositeConstruct %v2float %ox1 %oy1
; edge1_xy = center_xy + offset
   %e1_xy = OpFAdd %v2float %center_xy %e1_off
    %e1_x = OpCompositeExtract %float %e1_xy 0
    %e1_y = OpCompositeExtract %float %e1_xy 1
  %e1_pos = OpCompositeConstruct %v4float %e1_x %e1_y %center_z %center_w

; ---- Compute edge vertex 2 (j = (i+1) % 6) ----
   %j_val = OpIAdd %int %i_val %int_1
  %j_eq_6 = OpIEqual %bool %j_val %int_6
 %j_wrap = OpSelect %int %j_eq_6 %int_0 %j_val

  %j_float = OpConvertSToF %float %j_wrap
   %angle2 = OpFMul %float %j_float %float_pi_3
    %cos2 = OpExtInst %float %1 Cos %angle2
    %sin2 = OpExtInst %float %1 Sin %angle2
    %ox2 = OpFMul %float %float_0_04 %cos2
    %oy2 = OpFMul %float %float_0_04 %sin2
 %e2_off = OpCompositeConstruct %v2float %ox2 %oy2
   %e2_xy = OpFAdd %v2float %center_xy %e2_off
    %e2_x = OpCompositeExtract %float %e2_xy 0
    %e2_y = OpCompositeExtract %float %e2_xy 1
  %e2_pos = OpCompositeConstruct %v4float %e2_x %e2_y %center_z %center_w

; ---- Emit triangle strip: center, edge1, edge2 ----
; Vertex 0: center (full color)
         OpStore %ptr_out_pos %center_pos
         OpStore %outColor %center_color
         OpEmitVertex

; Vertex 1: edge1 (dark color)
         OpStore %ptr_out_pos %e1_pos
         OpStore %outColor %dark_color
         OpEmitVertex

; Vertex 2: edge2 (dark color)
         OpStore %ptr_out_pos %e2_pos
         OpStore %outColor %dark_color
         OpEmitVertex

; End this 3-vertex strip (forms one triangle)
         OpEndPrimitive

; ---- Increment and loop ----
    %i_next = OpIAdd %int %i_val %int_1
         OpStore %i_ptr %i_next
         OpBranch %loop_header

; --- Loop merge ---
%loop_merge = OpLabel
         OpReturn
               OpFunctionEnd
```
