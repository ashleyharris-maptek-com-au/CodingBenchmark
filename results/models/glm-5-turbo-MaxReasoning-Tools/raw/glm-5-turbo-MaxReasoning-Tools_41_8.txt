```spirv
; SPIR-V
; Version: 1.0
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inColor %outColor
               OpExecutionMode %main OriginUpperLeft

         %glsl = OpExtInstImport "GLSL.std.450"

; -- Decorations --
               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %inColor Location 4
               OpDecorate %outColor Location 0
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224

; -- Types --
        %void = OpTypeVoid
        %f32 = OpTypeFloat 32
       %bool = OpTypeBool
        %v3f = OpTypeVector %f32 3
        %v4f = OpTypeVector %f32 4
       %mat4 = OpTypeMatrix %v4f 4
        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
 %ptr_in_v3f = OpTypePointer Input %v3f
 %ptr_out_v4 = OpTypePointer Output %v4f
 %ptr_ubo    = OpTypePointer Uniform %UBO
 %ptr_uni_v4 = OpTypePointer Uniform %v4f
        %i32 = OpTypeInt 32 1
    %fn_void  = OpTypeFunction %void

; -- Constants --
         %c0  = OpConstant %f32 0.0
       %c0_2  = OpConstant %f32 0.2
     %c0_25  = OpConstant %f32 0.25
       %c0_4  = OpConstant %f32 0.4
       %c0_5  = OpConstant %f32 0.5
       %c0_7  = OpConstant %f32 0.7
      %c0_75  = OpConstant %f32 0.75
         %c1  = OpConstant %f32 1.0
        %ci3  = OpConstant %i32 3

; -- Variables --
  %inWorldPos = OpVariable %ptr_in_v3f Input
    %inNormal = OpVariable %ptr_in_v3f Input
    %inColor  = OpVariable %ptr_in_v3f Input
   %outColor  = OpVariable %ptr_out_v4 Output
        %ubo  = OpVariable %ptr_ubo Uniform

; -- Function: main --
        %main = OpFunction %void None %fn_void
      %entry = OpLabel

; Load inputs
  %worldPos  = OpLoad %v3f %inWorldPos
  %normalRaw = OpLoad %v3f %inNormal
  %vertexCol = OpLoad %v3f %inColor

; Normalize the interpolated normal
          %N = OpExtInst %v3f %glsl Normalize %normalRaw

; Load light position from UBO (member index 3 = lightPos)
%lpPtr       = OpAccessChain %ptr_uni_v4 %ubo %ci3
  %lightPos  = OpLoad %v4f %lpPtr

; Compute light direction: L = normalize(lightPos.xyz - worldPos)
   %lp_xyz   = OpVectorShuffle %v3f %lightPos %lightPos 0 1 2
   %toLight  = OpFSub %v3f %lp_xyz %worldPos
          %L = OpExtInst %v3f %glsl Normalize %toLight

; Lambertian diffuse: d = max(dot(N, L), 0.0)
     %ndotl  = OpExtInst %f32 %glsl Dot %N %L
          %d = OpExtInst %f32 %glsl FMax %ndotl %c0

; Quantize into toon bands (thresholds 0.75 / 0.5 / 0.25)
      %ge25  = OpFOrdGreaterThanEqual %bool %d %c0_25
       %b25  = OpSelect %f32 %ge25 %c0_4 %c0_2

      %ge50  = OpFOrdGreaterThanEqual %bool %d %c0_5
       %b50  = OpSelect %f32 %ge50 %c0_7 %b25

      %ge75  = OpFOrdGreaterThanEqual %bool %d %c0_75
      %band  = OpSelect %f32 %ge75 %c1 %b50

; Apply band to vertex color
    %shaded  = OpVectorTimesScalar %v3f %vertexCol %band

; Construct output vec4 with alpha = 1.0
         %r  = OpCompositeExtract %f32 %shaded 0
         %g  = OpCompositeExtract %f32 %shaded 1
         %b  = OpCompositeExtract %f32 %shaded 2
      %rgba  = OpCompositeConstruct %v4f %r %g %b %c1

               OpStore %outColor %rgba
               OpReturn
               OpFunctionEnd
```
