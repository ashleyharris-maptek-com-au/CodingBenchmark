```spirv
; SPIR-V
; Version: 1.0
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

; -------- Types --------
%void       = OpTypeVoid
%float      = OpTypeFloat 32
%int        = OpTypeInt 32 1
%uint       = OpTypeUInt 32
%v3float    = OpTypeVector %float 3
%v4float    = OpTypeVector %float 4

; -------- Constants --------
%uint_1     = OpConstant %uint 1
%int_0      = OpConstant %int 0
%float_h    = OpConstant %float 0.05

; -------- Structs --------
%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; -------- Arrays --------
%arr_pv_in  = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3_in  = OpTypeArray %v3float %uint_1

; -------- Pointer types --------
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4     = OpTypePointer Input %v4float
%ptr_in_v3     = OpTypePointer Input %v3float
%ptr_out_pv    = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4    = OpTypePointer Output %v4float
%ptr_out_v3    = OpTypePointer Output %v3float

; -------- Function type --------
%func_type     = OpTypeFunction %void

; -------- Interface variables --------
%gl_in       = OpVariable %ptr_in_arr_pv Input
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in   = OpVariable %ptr_in_arr_v3 Input
%color_in    = OpVariable %ptr_in_arr_v3 Input
%gl_out      = OpVariable %ptr_out_pv Output
%outColor    = OpVariable %ptr_out_v3 Output

; ========================================
; Main function
; ========================================
%main = OpFunction %void None %func_type
%bb   = OpLabel

; --- Load gl_in[0].gl_Position ---
%pos_ptr  = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%orig_pos = OpLoad %v4float %pos_ptr

; --- Extract x, y, z, w ---
%x = OpCompositeExtract %float %orig_pos 0
%y = OpCompositeExtract %float %orig_pos 1
%z = OpCompositeExtract %float %orig_pos 2
%w = OpCompositeExtract %float %orig_pos 3

; --- Precompute offsets ---
%y_plus_h  = OpFAdd %float %y %float_h
%y_minus_h = OpFSub %float %y %float_h
%x_plus_h  = OpFAdd %float %x %float_h
%x_minus_h = OpFSub %float %x %float_h

; --- Load input color from color_in[0] ---
%col_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%color   = OpLoad %v3float %col_ptr

; --- Pointer to gl_out.gl_Position (reused for every vertex) ---
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; =============================
; Vertex 0 – Top    (x, y+h, z, w)
; =============================
%pos_top = OpCompositeConstruct %v4float %x %y_plus_h %z %w
           OpStore %out_pos_ptr %pos_top
           OpStore %outColor %color
           OpEmitVertex

; =============================
; Vertex 1 – Right  (x+h, y, z, w)
; =============================
%pos_right = OpCompositeConstruct %v4float %x_plus_h %y %z %w
            OpStore %out_pos_ptr %pos_right
            OpStore %outColor %color
            OpEmitVertex

; =============================
; Vertex 2 – Left   (x-h, y, z, w)
; =============================
%pos_left = OpCompositeConstruct %v4float %x_minus_h %y %z %w
           OpStore %out_pos_ptr %pos_left
           OpStore %outColor %color
           OpEmitVertex

; =============================
; Vertex 3 – Bottom (x, y-h, z, w)
; =============================
%pos_bottom = OpCompositeConstruct %v4float %x %y_minus_h %z %w
             OpStore %out_pos_ptr %pos_bottom
             OpStore %outColor %color
             OpEmitVertex

; --- Complete the triangle strip ---
OpEndPrimitive
OpReturn
OpFunctionEnd
```
