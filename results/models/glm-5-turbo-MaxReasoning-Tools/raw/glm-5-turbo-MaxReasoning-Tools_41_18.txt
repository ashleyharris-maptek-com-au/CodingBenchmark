```spirv
; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor %ubo
               OpExecutionMode %main OriginUpperLeft

; Source info
               OpSource GLSL 450
               OpName %main "main"
               OpName %inWorldPos "worldPos"
               OpName %inNormal "normal"
               OpName %inUV "uv"
               OpName %inTangent "tangent"
               OpName %inColor "color"
               OpName %outColor "outColor"
               OpName %ubo "ubo"
               OpName %UBO "UBO"

; Input locations
               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %inUV Location 2
               OpDecorate %inTangent Location 3
               OpDecorate %inColor Location 4

; Output location
               OpDecorate %outColor Location 0

; UBO block decorations
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224

; ========== Types ==========
%void       = OpTypeVoid
%fnTy       = OpTypeFunction %void
%f32        = OpTypeFloat 32
%v3f        = OpTypeVector %f32 3
%v4f        = OpTypeVector %f32 4
%bool       = OpTypeBool
%mat4v4     = OpTypeMatrix %v4f 4
%i32        = OpTypeInt 32 0

; Pointer types
%pInV3      = OpTypePointer Input %v3f
%pOutV4     = OpTypePointer Output %v4f
%pUniformV4 = OpTypePointer Uniform %v4f

; UBO struct type (std140)
%UBO        = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4f %v4f %v4f
%pUBO       = OpTypePointer Uniform %UBO

; ========== Constants ==========
%cZero      = OpConstant %f32 0.0
%cHalf      = OpConstant %f32 0.5
%cOne       = OpConstant %f32 1.0
%cTwo       = OpConstant %f32 2.0
%cIdx4      = OpConstant %i32 4

; ========== Variables ==========
%inWorldPos = OpVariable %pInV3 Input
%inNormal   = OpVariable %pInV3 Input
%inUV       = OpVariable %pInV3 Input
%inTangent  = OpVariable %pInV3 Input
%inColor    = OpVariable %pInV3 Input
%outColor   = OpVariable %pOutV4 Output
%ubo        = OpVariable %pUBO Uniform

; ========== Main Function ==========
%main       = OpFunction %void None %fnTy
%entry      = OpLabel

; Load interpolated world-space normal and normalize it
%N_raw      = OpLoad %v3f %inNormal
%N          = OpExtInst %v3f %1 Normalize %N_raw

; Load world-space position
%P          = OpLoad %v3f %inWorldPos

; Access cameraPos (member index 4) from UBO
%camPtr     = OpAccessChain %pUniformV4 %ubo %cIdx4
%camRaw     = OpLoad %v4f %camPtr
%camXYZ     = OpVectorShuffle %v3f %camRaw %camRaw 0 1 2

; Compute view direction V = normalize(cameraPos - worldPos)
%V_unnorm   = OpFSub %v3f %camXYZ %P
%V          = OpExtInst %v3f %1 Normalize %V_unnorm

; NdotV = max(dot(N, V), 0.0)
%NdotV_raw  = OpDot %f32 %N %V
%NdotV      = OpExtInst %f32 %1 FMax %NdotV_raw %cZero

; Compare: NdotV < 0.5
%cmp        = OpFOrdLessThan %bool %NdotV %cHalf

; --- Branch 1 (NdotV < 0.5): r=1, g=2*NdotV, b=0 ---
%g_low      = OpFMul %f32 %cTwo %NdotV        ; 2 * NdotV

; --- Branch 2 (NdotV >= 0.5): r=g=1-2*(NdotV-0.5), b=2*(NdotV-0.5) ---
%temp       = OpFSub %f32 %NdotV %cHalf        ; NdotV - 0.5
%twoTemp    = OpFMul %f32 %cTwo %temp           ; 2 * (NdotV - 0.5)
%rg_high    = OpFSub %f32 %cOne %twoTemp        ; 1 - 2*(NdotV - 0.5)

; Select per-component based on comparison
%r          = OpSelect %f32 %cmp %cOne %rg_high  ; 1.0 or 1-2t
%g          = OpSelect %f32 %cmp %g_low %rg_high ; 2*NdotV or 1-2t
%b          = OpSelect %f32 %cmp %cZero %twoTemp ; 0.0 or 2t

; Construct final RGBA output
%rgba       = OpCompositeConstruct %v4f %r %g %b %cOne
               OpStore %outColor %rgba

               OpReturn
               OpFunctionEnd
```
