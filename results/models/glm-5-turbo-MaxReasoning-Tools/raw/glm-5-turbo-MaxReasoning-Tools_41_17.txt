; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%functy = OpTypeFunction %void
%v3ptr_input = OpTypePointer Input %vec3
%v2ptr_input = OpTypePointer Input %vec2
%v4ptr_output = OpTypePointer Output %vec4
%v4ptr_uniform = OpTypePointer Uniform %vec4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%uboptr = OpTypePointer Uniform %UBO

%worldPos = OpVariable %v3ptr_input Input
%normal = OpVariable %v3ptr_input Input
%uv = OpVariable %v2ptr_input Input
%outColor = OpVariable %v4ptr_output Output
%ubo = OpVariable %uboptr Uniform

%c0 = OpConstant %float 0
%c0_05 = OpConstant %float 0.05
%c0_1 = OpConstant %float 0.1
%c0_2 = OpConstant %float 0.2
%c0_4 = OpConstant %float 0.4
%c0_5 = OpConstant %float 0.5
%c0_6 = OpConstant %float 0.6
%c0_7 = OpConstant %float 0.7
%c1 = OpConstant %float 1
%c8 = OpConstant %float 8
%c16 = OpConstant %float 16
%i3 = OpConstant %int 3
%cmortar = OpConstantComposite %vec3 %c0_7 %c0_7 %c0_7
%cbrick = OpConstantComposite %vec3 %c0_7 %c0_2 %c0_1

%main = OpFunction %void None %functy
%entry = OpLabel

; Scale UV by (8, 16)
%uv_val = OpLoad %vec2 %uv
%scale_vec = OpCompositeConstruct %vec2 %c8 %c16
%scaled_uv = OpFMul %vec2 %uv_val %scale_vec
%su = OpCompositeExtract %float %scaled_uv 0
%sv = OpCompositeExtract %float %scaled_uv 1

; Compute row and check odd for stagger
%row = OpExtInst %float %glsl Floor %sv
%row_half = OpFMul %float %row %c0_5
%row_fract = OpExtInst %float %glsl Fract %row_half
%is_odd = OpFOrdGreaterThanEqual %bool %row_fract %c0_5

; Apply half-brick offset on odd rows
%offset = OpSelect %float %is_odd %c0_5 %c0
%su_shifted = OpFAdd %float %su %offset

; Fractional position within brick cell
%fu = OpExtInst %float %glsl Fract %su_shifted
%fv = OpExtInst %float %glsl Fract %sv

; Mortar test: within mortar thickness in X or Y
%mortar_x = OpFOrdLessThan %bool %fu %c0_05
%mortar_y = OpFOrdLessThan %bool %fv %c0_1
%is_mortar = OpLogicalOr %bool %mortar_x %mortar_y

; Select brick or mortar color
%base_color = OpSelect %vec3 %is_mortar %cmortar %cbrick

; Lambertian shading
%n_val = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %glsl Normalize %n_val
%P = OpLoad %vec3 %worldPos

; Load light position from UBO member 3
%light_ptr = OpAccessChain %v4ptr_uniform %ubo %i3
%light_pos = OpLoad %vec4 %light_ptr
%L_pos = OpVectorShuffle %vec3 %light_pos %light_pos 0 1 2

; Light direction and diffuse
%diff_vec = OpFSub %vec3 %L_pos %P
%L = OpExtInst %vec3 %glsl Normalize %diff_vec
%ndotl = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %ndotl %c0

; Shading factor: 0.4 + 0.6 * diffuse
%scaled_diff = OpFMul %float %c0_6 %diffuse
%shading = OpFAdd %float %c0_4 %scaled_diff

; Final color = base_color * shading, alpha = 1.0
%final_rgb = OpVectorTimesScalar %vec3 %base_color %shading
%final_color = OpCompositeConstruct %vec4 %final_rgb %c1
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
