```spirv
; SPIR-V
; Version: 1.0
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

%glslstd450     = OpExtInstImport "GLSL.std.450"

; Input decorations
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %in_uv Location 2

; Output decoration
               OpDecorate %outColor Location 0

; UBO decorations
               OpDecorate %UBO_type Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpMemberDecorate %UBO_type 0 ColMajor
               OpMemberDecorate %UBO_type 0 MatrixStride 16
               OpMemberDecorate %UBO_type 0 Offset 0
               OpMemberDecorate %UBO_type 1 ColMajor
               OpMemberDecorate %UBO_type 1 MatrixStride 16
               OpMemberDecorate %UBO_type 1 Offset 64
               OpMemberDecorate %UBO_type 2 ColMajor
               OpMemberDecorate %UBO_type 2 MatrixStride 16
               OpMemberDecorate %UBO_type 2 Offset 128
               OpMemberDecorate %UBO_type 3 Offset 192
               OpMemberDecorate %UBO_type 4 Offset 208
               OpMemberDecorate %UBO_type 5 Offset 224

; === Types ===
%void           = OpTypeVoid
%func_type      = OpTypeFunction %void
%float          = OpTypeFloat 32
%v2float        = OpTypeVector %float 2
%v3float        = OpTypeVector %float 3
%v4float        = OpTypeVector %float 4
%bool           = OpTypeBool
%int            = OpTypeInt 32 1
%mat4           = OpTypeMatrix %v4float 4

; UBO struct: model, view, projection, lightPos, cameraPos, params
%UBO_type       = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Pointer types
%ptr_input_v3float   = OpTypePointer Input %v3float
%ptr_input_v2float   = OpTypePointer Input %v2float
%ptr_output_v4float  = OpTypePointer Output %v4float
%ptr_uniform_UBO     = OpTypePointer Uniform %UBO_type
%ptr_uniform_v4float = OpTypePointer Uniform %v4float

; === Constants ===
%float_0        = OpConstant %float 0.0
%float_01       = OpConstant %float 0.1
%float_03       = OpConstant %float 0.3
%float_05       = OpConstant %float 0.5
%float_07       = OpConstant %float 0.7
%float_08       = OpConstant %float 0.8
%float_1        = OpConstant %float 1.0
%float_6        = OpConstant %float 6.0
%int_3          = OpConstant %int 3

%v2const_05     = OpConstantComposite %v2float %float_05 %float_05
%v3const_yellow = OpConstantComposite %v3float %float_1 %float_08 %float_0
%v3const_blue   = OpConstantComposite %v3float %float_01 %float_01 %float_03

; === Variables ===
%in_worldPos    = OpVariable %ptr_input_v3float Input
%in_normal      = OpVariable %ptr_input_v3float Input
%in_uv          = OpVariable %ptr_input_v2float Input
%outColor       = OpVariable %ptr_output_v4float Output
%ubo            = OpVariable %ptr_uniform_UBO Uniform

; === Main function ===
%main           = OpFunction %void None %func_type
%entry          = OpLabel

; Load UV coordinates
%uv             = OpLoad %v2float %in_uv

; Scale to 6x6 grid and take fract for cell-local position
%uv_scaled      = OpVectorTimesScalar %v2float %uv %float_6
%uv_fract       = OpExtInst %v2float %glslstd450 Fract %uv_scaled

; Center each cell so dot center is at (0.5, 0.5)
%uv_centered    = OpFSub %v2float %uv_fract %v2const_05

; Compute Euclidean distance from cell center
%c_x            = OpCompositeExtract %float %uv_centered 0
%c_y            = OpCompositeExtract %float %uv_centered 1
%c_x2           = OpFMul %float %c_x %c_x
%c_y2           = OpFMul %float %c_y %c_y
%dist2          = OpFAdd %float %c_x2 %c_y2
%dist           = OpExtInst %float %glslstd450 Sqrt %dist2

; Test if fragment is inside a dot (radius 0.3)
%is_dot         = OpFOrdLessThanEqual %bool %dist %float_03

; Select pattern color: yellow dot or dark blue background
%base_color     = OpSelect %v3float %is_dot %v3const_yellow %v3const_blue

; --- Lambertian shading ---
; Load world-space position
%worldPos       = OpLoad %v3float %in_worldPos

; Load and normalize world-space normal
%normal_raw     = OpLoad %v3float %in_normal
%N              = OpExtInst %v3float %glslstd450 Normalize %normal_raw

; Load light position from UBO member 3 (lightPos)
%lightPos_ptr   = OpAccessChain %ptr_uniform_v4float %ubo %int_3
%lightPos       = OpLoad %v4float %lightPos_ptr
%L_pos          = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2

; Compute light direction and normalize
%to_light       = OpFSub %v3float %L_pos %worldPos
%L              = OpExtInst %v3float %glslstd450 Normalize %to_light

; Diffuse term: max(dot(N, L), 0.0)
%NdotL          = OpDot %float %N %L
%diffuse        = OpExtInst %float %glslstd450 FMax %NdotL %float_0

; Combined shading factor: 0.3 + 0.7 * diffuse
%scaled_diff    = OpFMul %float %float_07 %diffuse
%shade          = OpFAdd %float %float_03 %scaled_diff

; Apply shading to pattern color
%shaded_color   = OpVectorTimesScalar %v3float %base_color %shade

; Construct final RGBA output with alpha = 1.0
%output         = OpCompositeConstruct %v4float %shaded_color %float_1
               OpStore %outColor %output

               OpReturn
               OpFunctionEnd
```
