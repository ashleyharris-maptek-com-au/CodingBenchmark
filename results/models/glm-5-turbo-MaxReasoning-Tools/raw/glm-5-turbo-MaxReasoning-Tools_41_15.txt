; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; Decorations
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

; Types and constants
       %void = OpTypeVoid
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
   %mat4v4f = OpTypeMatrix %v4float 4

       %UBO = OpTypeStruct %mat4v4f %mat4v4f %mat4v4f %v4float %v4float %v4float

%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_ubo   = OpTypePointer Uniform %UBO
%ptr_ubo_v4 = OpTypePointer Uniform %v4float

  %func_ty  = OpTypeFunction %void

       %f0   = OpConstant %float 0
      %f05   = OpConstant %float 0.5
       %f1   = OpConstant %float 1
      %f02   = OpConstant %float 0.2
      %f04   = OpConstant %float 0.4
      %f07   = OpConstant %float 0.7
     %f075   = OpConstant %float 0.75
        %i3  = OpConstant %int 3

      %cool  = OpConstantComposite %v3float %f02 %f02 %f075
      %warm  = OpConstantComposite %v3float %f07 %f07 %f04

; Variables
  %worldPos  = OpVariable %ptr_in_v3 Input
    %normal  = OpVariable %ptr_in_v3 Input
   %outColor = OpVariable %ptr_out_v4 Output
       %ubo  = OpVariable %ptr_ubo Uniform

; Main function
       %main = OpFunction %void None %func_ty
      %entry = OpLabel

; Load inputs
      %N_in = OpLoad %v3float %normal
         %P = OpLoad %v3float %worldPos

; Normalize normal
         %N = OpExtInst %v3float %1 Normalize %N_in

; Load lightPos from UBO (member 3)
  %lightPtr = OpAccessChain %ptr_ubo_v4 %ubo %i3
 %lightPos4 = OpLoad %v4float %lightPtr
  %lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

; Light direction L = normalize(lightPos - worldPos)
     %L_vec = OpFSub %v3float %lightPos %P
         %L = OpExtInst %v3float %1 Normalize %L_vec

; t = dot(N, L) * 0.5 + 0.5
       %NdL = OpDot %float %N %L
   %NdL_half = OpFMul %float %NdL %f05
         %t = OpFAdd %float %NdL_half %f05

; color = mix(cool, warm, t) = cool*(1-t) + warm*t
 %one_minus_t = OpFSub %float %f1 %t
 %cool_scaled = OpVectorTimesScalar %v3float %cool %one_minus_t
 %warm_scaled = OpVectorTimesScalar %v3float %warm %t
     %color3 = OpFAdd %v3float %cool_scaled %warm_scaled

; Output vec4(color, 1.0)
     %color4 = OpCompositeConstruct %v4float %color3 %f1
              OpStore %outColor %color4

              OpReturn
              OpFunctionEnd
