```spirv
; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %outColor Location 0
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224

%void       = OpTypeVoid
%func       = OpTypeFunction %void
%float      = OpTypeFloat 32
%int        = OpTypeInt 32 1
%vec3       = OpTypeVector %float 3
%vec4       = OpTypeVector %float 4
%mat4       = OpTypeMatrix %vec4 4
%UBO        = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_in_v3  = OpTypePointer Input %vec3
%ptr_out_v4 = OpTypePointer Output %vec4
%ptr_uni    = OpTypePointer Uniform %UBO
%ptr_uni_v4 = OpTypePointer Uniform %vec4

%c0         = OpConstant %float 0.0
%c03        = OpConstant %float 0.3
%c07        = OpConstant %float 0.7
%c1         = OpConstant %float 1.0
%c15        = OpConstant %float 1.5
%c3f        = OpConstant %float 3.0
%c08        = OpConstant %float 0.8
%c02        = OpConstant %float 0.2
%int_3      = OpConstant %int 3
%int_4      = OpConstant %int 4

%cBaseColor = OpConstantComposite %vec3 %c08 %c02 %c02
%cFogColor  = OpConstantComposite %vec3 %c07 %c07 %c08

%worldPos   = OpVariable %ptr_in_v3 Input
%normal     = OpVariable %ptr_in_v3 Input
%outColor   = OpVariable %ptr_out_v4 Output
%ubo        = OpVariable %ptr_uni Uniform

%main       = OpFunction %void None %func
%entry      = OpLabel

%wp         = OpLoad %vec3 %worldPos
%n          = OpLoad %vec3 %normal
%n_norm     = OpExtInst %vec3 %1 Normalize %n

%lightPtr   = OpAccessChain %ptr_uni_v4 %ubo %int_3
%camPtr     = OpAccessChain %ptr_uni_v4 %ubo %int_4
%light      = OpLoad %vec4 %lightPtr
%cam        = OpLoad %vec4 %camPtr

%light_xyz  = OpVectorShuffle %vec3 %light %light 0 1 2
%cam_xyz    = OpVectorShuffle %vec3 %cam %cam 0 1 2

%toLight    = OpFSub %vec3 %light_xyz %wp
%ldir       = OpExtInst %vec3 %1 Normalize %toLight
%diff       = OpDot %float %n_norm %ldir
%diff_c     = OpExtInst %float %1 FClamp %diff %c0 %c1

%factor_mul = OpFMul %float %c07 %diff_c
%factor     = OpFAdd %float %c03 %factor_mul

%litColor   = OpVectorTimesScalar %vec3 %cBaseColor %factor

%dist_val   = OpExtInst %float %1 Distance %cam_xyz %wp

%fog_num    = OpFSub %float %dist_val %c15
%fog_div    = OpFDiv %float %fog_num %c3f
%fog_fac    = OpExtInst %float %1 FClamp %fog_div %c0 %c1

%one_minus  = OpFSub %float %c1 %fog_fac
%lit_w      = OpVectorTimesScalar %vec3 %litColor %one_minus
%fog_w      = OpVectorTimesScalar %vec3 %cFogColor %fog_fac
%fogged     = OpFAdd %vec3 %lit_w %fog_w

%r          = OpCompositeExtract %float %fogged 0
%g          = OpCompositeExtract %float %fogged 1
%b          = OpCompositeExtract %float %fogged 2
%result     = OpCompositeConstruct %vec4 %r %g %b %c1

               OpStore %outColor %result
               OpReturn
               OpFunctionEnd
```
