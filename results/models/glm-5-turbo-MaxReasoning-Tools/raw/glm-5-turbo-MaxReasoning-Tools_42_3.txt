```spirv
               OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               ; --- decorations ------------------------------------------------
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpMemberDecorate %gl_PerVertex_in 0 Offset 0
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpMemberDecorate %gl_PerVertex_out 0 Offset 0
               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal   Location 1
               OpDecorate %inColor    Location 2
               OpDecorate %outColor   Location 0

               ; --- types ------------------------------------------------------
       %void = OpTypeVoid
     %fn_type = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1
    %uint_2 = OpConstant %uint 2
   %v2float = OpTypeVector %float 2
   %v3float = OpTypeVector %float 3
   %v4float = OpTypeVector %float 4
   %float_0 = OpConstant %float 0
  %float_003 = OpConstant %float 0.03

               ; --- gl_PerVertex structs (input / output) ----------------------
%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

               ; --- per-vertex input arrays (size 2 for InputLines) ------------
 %arr_in_pv = OpTypeArray %gl_PerVertex_in %uint_2
  %arr_in_v3 = OpTypeArray %v3float        %uint_2

               ; --- pointer types ----------------------------------------------
%ptr_in_arr_pv = OpTypePointer Input  %arr_in_pv
%ptr_in_arr_v3 = OpTypePointer Input  %arr_in_v3
   %ptr_in_v4  = OpTypePointer Input  %v4float
   %ptr_in_v3  = OpTypePointer Input  %v3float
 %ptr_out_pv   = OpTypePointer Output %gl_PerVertex_out
  %ptr_out_v4  = OpTypePointer Output %v4float
  %ptr_out_v3  = OpTypePointer Output %v3float

               ; --- interface variables ----------------------------------------
      %gl_in    = OpVariable %ptr_in_arr_pv Input          ; gl_in[]
 %inWorldPos    = OpVariable %ptr_in_arr_v3 Input          ; location 0
   %inNormal    = OpVariable %ptr_in_arr_v3 Input          ; location 1
    %inColor    = OpVariable %ptr_in_arr_v3 Input          ; location 2
     %gl_out    = OpVariable %ptr_out_pv   Output         ; gl_PerVertex
   %outColor    = OpVariable %ptr_out_v3   Output         ; location 0

; ===========================================================================
; main
; ===========================================================================
       %main = OpFunction %void None %fn_type
   %entry   = OpLabel

               ; ---- load gl_Position of both line endpoints -------------------
 %p0_ptr     = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
 %p0         = OpLoad %v4float %p0_ptr

 %p1_ptr     = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
 %p1         = OpLoad %v4float %p1_ptr

               ; ---- load per-vertex colors ------------------------------------
 %c0_ptr     = OpAccessChain %ptr_in_v3 %inColor %int_0
 %c0         = OpLoad %v3float %c0_ptr

 %c1_ptr     = OpAccessChain %ptr_in_v3 %inColor %int_1
 %c1         = OpLoad %v3float %c1_ptr

               ; ---- direction = p1 - p0 --------------------------------------
 %dir        = OpFSub %v4float %p1 %p0

               ; ---- keep only xy components -----------------------------------
 %dir_xy     = OpVectorShuffle %v2float %dir %dir 0 1

               ; ---- perpendicular = normalize(-dir.y, dir.x) * thickness -----
 %dir_x      = OpCompositeExtract %float %dir_xy 0
 %dir_y      = OpCompositeExtract %float %dir_xy 1
 %neg_dir_y  = OpFNegate %float %dir_y
 %perp_raw   = OpCompositeConstruct %v2float %neg_dir_y %dir_x
 %perp_norm  = OpExtInst %v2float %1 Normalize %perp_raw
 %perp       = OpExtInst %v2float %1 VectorTimesScalar %perp_norm %float_003

               ; ---- build vec4 offsets (+perp and -perp) ----------------------
 %perp_x     = OpCompositeExtract %float %perp 0
 %perp_y     = OpCompositeExtract %float %perp 1
 %neg_px     = OpFNegate %float %perp_x
 %neg_py     = OpFNegate %float %perp_y

 %perp_v4    = OpCompositeConstruct %v4float %perp_x %perp_y %float_0 %float_0
 %neg_perp_v4= OpCompositeConstruct %v4float %neg_px %neg_py %float_0 %float_0

               ; ---- quad vertex positions -------------------------------------
 %v0_pos     = OpFAdd %v4float %p0 %perp_v4       ; p0 + perp
 %v1_pos     = OpFAdd %v4float %p0 %neg_perp_v4   ; p0 - perp
 %v2_pos     = OpFAdd %v4float %p1 %perp_v4       ; p1 + perp
 %v3_pos     = OpFAdd %v4float %p1 %neg_perp_v4   ; p1 - perp

               ; ---- pointer to gl_out.gl_Position ----------------------------
 %pos_ptr    = OpAccessChain %ptr_out_v4 %gl_out %int_0

               ; ---- emit triangle-strip quad ---------------------------------
               ; vertex 0: p0 + perp, color = c0
               OpStore %pos_ptr %v0_pos
               OpStore %outColor %c0
               OpEmitVertex

               ; vertex 1: p0 - perp, color = c0
               OpStore %pos_ptr %v1_pos
               OpStore %outColor %c0
               OpEmitVertex

               ; vertex 2: p1 + perp, color = c1
               OpStore %pos_ptr %v2_pos
               OpStore %outColor %c1
               OpEmitVertex

               ; vertex 3: p1 - perp, color = c1
               OpStore %pos_ptr %v3_pos
               OpStore %outColor %c1
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
```
