```spirv
; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Types
%void    = OpTypeVoid
%bool    = OpTypeBool
%i32     = OpTypeInt 32 1
%f32     = OpTypeFloat 32
%v2f     = OpTypeVector %f32 2
%v3f     = OpTypeVector %f32 3
%v4f     = OpTypeVector %f32 4
%m4f     = OpTypeMatrix %v4f 4
%UBO     = OpTypeStruct %m4f %m4f %m4f %v4f %v4f %v4f
%pInV3   = OpTypePointer Input %v3f
%pInV2   = OpTypePointer Input %v2f
%pOutV4  = OpTypePointer Output %v4f
%pUnUBO  = OpTypePointer Uniform %UBO
%pUnV4   = OpTypePointer Uniform %v4f
%fnty    = OpTypeFunction %void

; Constants
%c0      = OpConstant %f32 0
%c1      = OpConstant %f32 1
%c32     = OpConstant %f32 32
%ca_r    = OpConstant %f32 0.1
%ca_g    = OpConstant %f32 0.05
%ca_b    = OpConstant %f32 0.05
%ambient = OpConstantComposite %v3f %ca_r %ca_g %ca_b
%white3  = OpConstantComposite %v3f %c1 %c1 %c1
%ci0     = OpConstant %i32 0
%ci3     = OpConstant %i32 3
%ci4     = OpConstant %i32 4

; Variables
%inPos    = OpVariable %pInV3 Input
%inNormal = OpVariable %pInV3 Input
%inUV     = OpVariable %pInV2 Input
%inTangent= OpVariable %pInV3 Input
%inColor  = OpVariable %pInV3 Input
%outColor = OpVariable %pOutV4 Output
%ubo      = OpVariable %pUnUBO Uniform

; Function
%main = OpFunction %void None %fnty
%bb   = OpLabel

; --- Load inputs ---
%worldPos    = OpLoad %v3f %inPos
%normalRaw   = OpLoad %v3f %inNormal
%vertexColor = OpLoad %v3f %inColor

; --- Normalize interpolated normal ---
%N = OpExtInst %v3f %glsl450 Normalize %normalRaw

; --- Load lightPos and cameraPos from UBO ---
%lpPtr = OpAccessChain %pUnV4 %ubo %ci3
%cpPtr = OpAccessChain %pUnV4 %ubo %ci4
%lp    = OpLoad %v4f %lpPtr
%cp    = OpLoad %v4f %cpPtr

; xyz swizzle
%lp3 = OpVectorShuffle %v3f %lp %lp 0 1 2
%cp3 = OpVectorShuffle %v3f %cp %cp 0 1 2

; --- Light vector L = normalize(lightPos.xyz - worldPos) ---
%toLight = OpFSub %v3f %lp3 %worldPos
%L       = OpExtInst %v3f %glsl450 Normalize %toLight

; --- Diffuse: vertexColor * max(dot(N, L), 0) ---
%NdL        = OpDot %f32 %N %L
%NdL_clamp  = OpExtInst %f32 %glsl450 FMax %NdL %c0
%diffuse    = OpVectorTimesScalar %v3f %vertexColor %NdL_clamp

; --- Specular: reflect(-L, N), view V = normalize(camPos.xyz - worldPos) ---
%negL   = OpFNegate %v3f %L
%R      = OpExtInst %v3f %glsl450 Reflect %negL %N

%toView = OpFSub %v3f %cp3 %worldPos
%V      = OpExtInst %v3f %glsl450 Normalize %toView

%RdV        = OpDot %f32 %R %V
%RdV_clamp  = OpExtInst %f32 %glsl450 FMax %RdV %c0
%specPow    = OpExtInst %f32 %glsl450 Pow %RdV_clamp %c32
%specular   = OpVectorTimesScalar %v3f %white3 %specPow

; --- Combine: ambient + diffuse + specular, clamp [0,1], alpha=1 ---
%sum1  = OpFAdd %v3f %ambient %diffuse
%sum2  = OpFAdd %v3f %sum1 %specular
%rgb   = OpExtInst %v3f %glsl450 FClamp %sum2 %c0 %c1
%final = OpCompositeConstruct %v4f %rgb %c1

OpStore %outColor %final
OpReturn
OpFunctionEnd
```
