```spirv
; SPIR-V
; Version: 1.0
               OpCapability Shader
        %glsl450 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %inPos %inNormal %inUV %inTangent %inColor %outColor
               OpExecutionMode %main OriginUpperLeft

; ----- Input Locations -----
               OpDecorate %inPos    Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %inUV     Location 2
               OpDecorate %inTangent Location 3
               OpDecorate %inColor  Location 4

; ----- Output Location -----
               OpDecorate %outColor Location 0

; ----- Uniform Buffer (UBO) -----
               OpDecorate %_UBO     DescriptorSet 0
               OpDecorate %_UBO     Binding 0
               OpDecorate %UboType  Block
               OpMemberDecorate %UboType 0 Offset 0
               OpMemberDecorate %UboType 1 Offset 64
               OpMemberDecorate %UboType 2 Offset 128
               OpMemberDecorate %UboType 3 Offset 192
               OpMemberDecorate %UboType 4 Offset 208
               OpMemberDecorate %UboType 5 Offset 224
               OpMemberDecorate %UboType 0 ColMajor
               OpMemberDecorate %UboType 0 MatrixStride 16
               OpMemberDecorate %UboType 1 ColMajor
               OpMemberDecorate %UboType 1 MatrixStride 16
               OpMemberDecorate %UboType 2 ColMajor
               OpMemberDecorate %UboType 2 MatrixStride 16

; ==================== Types ====================
%void       = OpTypeVoid
%float      = OpTypeFloat 32
%uint       = OpTypeInt 32 0
%v2float    = OpTypeVector %float 2
%v3float    = OpTypeVector %float 3
%v4float    = OpTypeVector %float 4
%mat4       = OpTypeMatrix %v4float 4
%UboType    = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%ptrInV3    = OpTypePointer Input %v3float
%ptrInV2    = OpTypePointer Input %v2float
%ptrOutV4   = OpTypePointer Output %v4float
%ptrUbo     = OpTypePointer Uniform %UboType
%ptrUboV4   = OpTypePointer Uniform %v4float

%fnType     = OpTypeFunction %void

; ==================== Constants ====================
%c0         = OpConstant %float 0
%c005       = OpConstant %float 0.05
%c015       = OpConstant %float 0.15
%c03        = OpConstant %float 0.3
%c04        = OpConstant %float 0.4
%c05        = OpConstant %float 0.5
%c06        = OpConstant %float 0.6
%c1         = OpConstant %float 1

%cGround    = OpConstantComposite %v3float %c03 %c015 %c005
%cSky       = OpConstantComposite %v3float %c04 %c06 %c1

%i3         = OpConstant %uint 3

; ==================== Variables ====================
%inPos      = OpVariable %ptrInV3 Input
%inNormal   = OpVariable %ptrInV3 Input
%inUV       = OpVariable %ptrInV2 Input
%inTangent  = OpVariable %ptrInV3 Input
%inColor    = OpVariable %ptrInV3 Input
%outColor   = OpVariable %ptrOutV4 Output
%_UBO       = OpVariable %ptrUbo Uniform

; ==================== Main ====================
%main       = OpFunction %void None %fnType
%entry      = OpLabel

; --- Normalize the interpolated normal ---
%rawN       = OpLoad %v3float %inNormal
%N          = OpExtInst %v3float %glsl450 Normalize %rawN

; --- Hemisphere blend factor: N.y * 0.5 + 0.5 ---
%Ny         = OpCompositeExtract %float %N 1
%halfNy     = OpFMul %float %Ny %c05
%blend      = OpFAdd %float %halfNy %c05
%invBlend   = OpFSub %float %c1 %blend

; --- Hemisphere ambient = groundColor*(1-blend) + skyColor*blend ---
%gndMix     = OpVectorTimesScalar %v3float %cGround %invBlend
%skyMix     = OpVectorTimesScalar %v3float %cSky    %blend
%ambient    = OpFAdd %v3float %gndMix %skyMix

; --- Diffuse term: 0.3 * max(dot(N, L), 0.0) ---
%wPos       = OpLoad %v3float %inPos
%lpPtr      = OpAccessChain %ptrUboV4 %_UBO %i3
%lp4        = OpLoad %v4float %lpPtr
%lp3        = OpVectorShuffle %v3float %lp4 %lp4 0 1 2
%toLight    = OpFSub %v3float %lp3 %wPos
%L          = OpExtInst %v3float %glsl450 Normalize %toLight
%NdotL      = OpDot %float %N %L
%d          = OpExtInst %float %glsl450 FMax %NdotL %c0
%diffS      = OpFMul %float %c03 %d
%diffV      = OpCompositeConstruct %v3float %diffS %diffS %diffS

; --- Final color = ambient + diffuse, alpha = 1.0 ---
%rgb        = OpFAdd %v3float %ambient %diffV
%r          = OpCompositeExtract %float %rgb 0
%g          = OpCompositeExtract %float %rgb 1
%b          = OpCompositeExtract %float %rgb 2
%rgba       = OpCompositeConstruct %v4float %r %g %b %c1

               OpStore %outColor %rgba
               OpReturn
               OpFunctionEnd
```
