%pAInV3 = OpTypePointer Input %aV3
%pInv3  = OpTypePointer Input %v3f
%inWorldPos = OpVariable %pAInV3 Input

  %c0ac = OpAccessChain %pInv3 %inColor %ei

  %psy   = OpCompositeExtract %f32 %ps2 1
  %pv3   = OpCompositeConstruct %v3f %psx %psy %cf0
  %npv3  = OpFNegate %v3f %pv3

  %q2 = OpFAdd %v3f %n1   %pv3
  %q3 = OpFAdd %v3f %n1   %npv3

  %o1y   = OpCompositeExtract %f32 %o1xyz 1

  %o2x   = OpCompositeExtract %f32 %o2xyz 0
  %o2y   = OpCompositeExtract %f32 %o2xyz 1
  %o2z   = OpCompositeExtract %f32 %o2xyz 2
  %o2    = OpCompositeConstruct %v4f %o2x %o2y %o2z %p1w
