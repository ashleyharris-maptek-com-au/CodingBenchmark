```
; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_in_v3 = OpTypePointer Input %vec3
%ptr_out_v4 = OpTypePointer Output %vec4
%fn_type = OpTypeFunction %void

; Constants
%c0 = OpConstant %float 0.0
%c0_1 = OpConstant %float 0.1
%c0_4 = OpConstant %float 0.4
%c0_5 = OpConstant %float 0.5
%c1_0 = OpConstant %float 1.0
%c_bottom = OpConstantComposite %vec3 %c0_1 %c0_4 %c0_1
%c_top = OpConstantComposite %vec3 %c1_0 %c1_0 %c1_0

; Variables
%worldPos = OpVariable %ptr_in_v3 Input
%outColor = OpVariable %ptr_out_v4 Output

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0

; Main function
%main = OpFunction %void None %fn_type
%entry = OpLabel
  %wp = OpLoad %vec3 %worldPos
  %y = OpCompositeExtract %float %wp 1
  %y_half = OpFMul %float %y %c0_5
  %t = OpFAdd %float %y_half %c0_5
  %t_clamped = OpExtInst %float %1 FClamp %t %c0 %c1_0
  %one_minus_t = OpFSub %float %c1_0 %t_clamped
  %bot_scaled = OpVectorTimesScalar %vec3 %c_bottom %one_minus_t
  %top_scaled = OpVectorTimesScalar %vec3 %c_top %t_clamped
  %mixed = OpFAdd %vec3 %bot_scaled %top_scaled
  %r = OpCompositeExtract %float %mixed 0
  %g = OpCompositeExtract %float %mixed 1
  %b = OpCompositeExtract %float %mixed 2
  %result = OpCompositeConstruct %vec4 %r %g %b %c1_0
  OpStore %outColor %result
  OpReturn
OpFunctionEnd
```
