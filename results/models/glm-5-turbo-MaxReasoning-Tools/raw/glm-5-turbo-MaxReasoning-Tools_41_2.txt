```spirv
; SPIR-V
; Version: 1.0

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %uv %outColor
               OpExecutionMode %main OriginUpperLeft

               OpSource GLSL 450

               OpName %main "main"
               OpName %uv "uv"
               OpName %outColor "outColor"

               OpDecorate %uv Location 2
               OpDecorate %outColor Location 0

      %void = OpTypeVoid
%funcType = OpTypeFunction %void
    %float = OpTypeFloat 32
     %v2f = OpTypeVector %float 2
     %v4f = OpTypeVector %float 4

%inPtrV2f = OpTypePointer Input %v2f
%outPtrV4f = OpTypePointer Output %v4f

   %fZero = OpConstant %float 0.0
   %fOne = OpConstant %float 1.0

       %uv = OpVariable %inPtrV2f Input
  %outColor = OpVariable %outPtrV4f Output

     %main = OpFunction %void None %funcType
   %entry = OpLabel

   %uvVal = OpLoad %v2f %uv
     %uvX = OpCompositeExtract %float %uvVal 0
     %uvY = OpCompositeExtract %float %uvVal 1
   %color = OpCompositeConstruct %v4f %uvX %uvY %fZero %fOne

               OpStore %outColor %color
               OpReturn
               OpFunctionEnd
```
