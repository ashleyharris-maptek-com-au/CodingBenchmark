```spirv
; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%bool = OpTypeBool
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Pointer types
%ptrInV3 = OpTypePointer Input %v3float
%ptrInV2 = OpTypePointer Input %v2float
%ptrOutV4 = OpTypePointer Output %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformV4 = OpTypePointer Uniform %v4float

; Function type
%fnType = OpTypeFunction %void

; Constants
%c_f0 = OpConstant %float 0.0
%c_f0_2 = OpConstant %float 0.2
%c_f0_5 = OpConstant %float 0.5
%c_f0_9 = OpConstant %float 0.9
%c_f1 = OpConstant %float 1.0
%c_f2 = OpConstant %float 2.0
%c_f8 = OpConstant %float 8.0
%c_i3 = OpConstant %int 3

; Variables
%inWorldPos = OpVariable %ptrInV3 Input
%inNormal = OpVariable %ptrInV3 Input
%inUV = OpVariable %ptrInV2 Input
%outColor = OpVariable %ptrOutV4 Output
%ubo = OpVariable %ptrUniformUBO Uniform

; Main function
%main = OpFunction %void None %fnType
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %inWorldPos
%normal = OpLoad %v3float %inNormal
%uv = OpLoad %v2float %inUV

; Normalize the normal
%normal_n = OpExtInst %v3float %1 Normalize %normal

; --- Checkerboard ---
; Scale UV by 8 and floor each component
%uv_scaled = OpVectorTimesScalar %v2float %uv %c_f8
%uv_floored = OpExtInst %v2float %1 Floor %uv_scaled
; Sum the two floored components
%x_floored = OpCompositeExtract %float %uv_floored 0
%y_floored = OpCompositeExtract %float %uv_floored 1
%sum = OpFAdd %float %x_floored %y_floored
; Mod 2 gives 0.0 (even) or 1.0 (odd)
%sum_mod2 = OpExtInst %float %1 FMod %sum %c_f2

; Mix between dark (0.2) and bright (0.9)
%dark = OpCompositeConstruct %v3float %c_f0_2 %c_f0_2 %c_f0_2
%bright = OpCompositeConstruct %v3float %c_f0_9 %c_f0_9 %c_f0_9
%checker_color = OpExtInst %v3float %1 FMix %dark %bright %sum_mod2

; --- Lambertian shading ---
; Get light position from UBO member 3 (offset 192)
%lightPos_ptr = OpAccessChain %ptrUniformV4 %ubo %c_i3
%lightPos = OpLoad %v4float %lightPos_ptr
%lightPos_v3 = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2

; Compute light direction and normalize
%lightDir = OpFSub %v3float %lightPos_v3 %worldPos
%lightDir_n = OpExtInst %v3float %1 Normalize %lightDir

; N dot L clamped to [0,1]
%NdotL = OpExtInst %float %1 Dot %normal_n %lightDir_n
%NdotL_c = OpExtInst %float %1 FClamp %NdotL %c_f0 %c_f1

; Shading factor: 0.5 + 0.5 * diffuse
%shade_mul = OpFMul %float %c_f0_5 %NdotL_c
%shade = OpFAdd %float %c_f0_5 %shade_mul

; Apply shading to checkerboard color
%final_color = OpVectorTimesScalar %v3float %checker_color %shade

; Construct output: RGB from shaded checker, alpha = 1.0
%fr = OpCompositeExtract %float %final_color 0
%fg = OpCompositeExtract %float %final_color 1
%fb = OpCompositeExtract %float %final_color 2
%outRGBA = OpCompositeConstruct %v4float %fr %fg %fb %c_f1

OpStore %outColor %outRGBA
OpReturn
OpFunctionEnd
```
