```
; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %normal %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %normal Location 1
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%func_type = OpTypeFunction %void
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float

; Constants
%c0_5 = OpConstant %float 0.5
%c1_0 = OpConstant %float 1.0
%c0_5_v3 = OpConstantComposite %v3float %c0_5 %c0_5 %c0_5
%c1_0_v4 = OpConstantComposite %v4float %c1_0 %c1_0 %c1_0 %c1_0

; Variables
%normal = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel
  %n = OpLoad %v3float %normal
  %normalized = OpExtInst %v3float %1 Normalize %n
  %scaled = OpVectorTimesScalar %v3float %normalized %c0_5
  %mapped = OpFAdd %v3float %scaled %c0_5_v3
  %color = OpCompositeConstruct %v4float %mapped %c1_0
  OpStore %outColor %color
  OpReturn
OpFunctionEnd
```
