; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
               OpExecutionMode %main OriginUpperLeft

; Decorations
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %outColor Location 0
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

; Types
      %void = OpTypeVoid
      %func = OpTypeFunction %void
       %f32 = OpTypeFloat 32
       %i32 = OpTypeInt 32 0
     %v3f32 = OpTypeVector %f32 3
     %v4f32 = OpTypeVector %f32 4
    %mat4v4 = OpTypeMatrix %v4f32 4
       %UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4f32 %v4f32 %v4f32
%ptr_in_v3 = OpTypePointer Input %v3f32
%ptr_out_v4 = OpTypePointer Output %v4f32
%ptr_uni = OpTypePointer Uniform %UBO
%ptr_uni_v4 = OpTypePointer Uniform %v4f32

; Constants
     %c0 = OpConstant %f32 0.0
   %c0_2 = OpConstant %f32 0.2
   %c0_5 = OpConstant %f32 0.5
   %c0_8 = OpConstant %f32 0.8
   %c1_0 = OpConstant %f32 1.0
  %c32_0 = OpConstant %f32 32.0
   %ci_3 = OpConstant %i32 3
   %ci_4 = OpConstant %i32 4

; Variables
   %worldPos = OpVariable %ptr_in_v3 Input
     %normal = OpVariable %ptr_in_v3 Input
   %outColor = OpVariable %ptr_out_v4 Output
       %ubo = OpVariable %ptr_uni Uniform

; Main function
       %main = OpFunction %void None %func
      %entry = OpLabel

; Load vertex inputs
%worldPos_val = OpLoad %v3f32 %worldPos
 %normal_val = OpLoad %v3f32 %normal

; Normalize the interpolated normal
          %N = OpExtInst %v3f32 %1 Normalize %normal_val

; Access lightPos (UBO member 3) and extract xyz
   %ptr_lp = OpAccessChain %ptr_uni_v4 %ubo %ci_3
      %lp4 = OpLoad %v4f32 %ptr_lp
    %lp_xyz = OpVectorShuffle %v3f32 %lp4 %lp4 0 1 2

; L = normalize(lightPos.xyz - worldPos)
   %lp_sub = OpFSub %v3f32 %lp_xyz %worldPos_val
        %L = OpExtInst %v3f32 %1 Normalize %lp_sub

; Access cameraPos (UBO member 4) and extract xyz
   %ptr_cp = OpAccessChain %ptr_uni_v4 %ubo %ci_4
      %cp4 = OpLoad %v4f32 %ptr_cp
    %cp_xyz = OpVectorShuffle %v3f32 %cp4 %cp4 0 1 2

; V = normalize(cameraPos.xyz - worldPos)
   %cp_sub = OpFSub %v3f32 %cp_xyz %worldPos_val
        %V = OpExtInst %v3f32 %1 Normalize %cp_sub

; R = reflect(-L, N)
      %negL = OpFNegate %v3f32 %L
         %R = OpExtInst %v3f32 %1 Reflect %negL %N

; diffuse = max(dot(N, L), 0.0)
     %NdotL = OpDot %f32 %N %L
   %diffuse = OpFMax %f32 %NdotL %c0

; specular = pow(max(dot(R, V), 0.0), 32.0)
     %RdotV = OpDot %f32 %R %V
 %spec_base = OpFMax %f32 %RdotV %c0
  %specular = OpExtInst %f32 %1 Pow %spec_base %c32_0

; intensity = 0.2 + 0.5*diffuse + 0.8*specular
       %a = OpFMul %f32 %c0_5 %diffuse
       %b = OpFMul %f32 %c0_8 %specular
       %c = OpFAdd %f32 %c0_2 %a
%intensity = OpFAdd %f32 %c %b

; outColor = vec4(intensity, intensity, intensity, 1.0)
   %out_v4 = OpCompositeConstruct %v4f32 %intensity %intensity %intensity %c1_0
             OpStore %outColor %out_v4

               OpReturn
               OpFunctionEnd
