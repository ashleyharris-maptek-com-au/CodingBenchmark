import math

def read_input():
    lines = [line.strip() for line in sys.stdin.read().strip().split('\n')]
    if not lines:
        return None, None, None, None, None, None
    
    # First line: N K CAPACITY
    first = lines[0].split()
    N = int(first[0])
    K = int(first[1])
    CAPACITY = int(first[2])
    
    # Second line: depot
    depot_line = lines[1].split()
    depot = (float(depot_line[0]), float(depot_line[1]))
    
    # Customers
    customers = []
    for i in range(N):
        parts = lines[2 + i].split()
        x = float(parts[0])
        y = float(parts[1])
        demand = int(parts[2])
        customers.append({
            'id': i,
            'x': x,
            'y': y,
            'demand': demand
        })
    
    return N, K, CAPACITY, depot, customers
