using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

public class MaximumClique
{
    private static int N;
    private static BitSet[] adj;
    private static int[] degree;
    private static int[] vertexOrder;
    private static int bestSize = 0;
    private static List<int> bestClique = new List<int>();

    public static void Main(string[] args)
    {
        // Read input
        string[] firstLine = Console.ReadLine().Split();
        N = int.Parse(firstLine[0]);
        int M = int.Parse(firstLine[1]);

        adj = new BitSet[N];
        degree = new int[N];
        for (int i = 0; i < N; i++)
        {
            adj[i] = new BitSet(N);
            adj[i].Set(i); // self-loop inclusion for convenience
        }

        for (int i = 0; i < M; i++)
        {
            string[] edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            if (u != v)
            {
                adj[u].Set(v);
                adj[v].Set(u);
            }
        }

        // Precompute degrees
        for (int i = 0; i < N; i++)
        {
            degree[i] = adj[i].Count();
            if (degree[i] == 0)
                degree[i] = 1; // single vertex clique
        }

        // Vertex ordering for branch and bound
        vertexOrder = new int[N];
        for (int i = 0; i < N; i++) vertexOrder[i] = i;
        Array.Sort(vertexOrder, (a, b) => degree[b].CompareTo(degree[a]));

        // Reorder adjacency for better locality
        BitSet[] reorderedAdj = new BitSet[N];
        for (int i = 0;0; i < N; i++)
        {
            reorderedAdj[i] = new BitSet(N);
            int oldIdx = vertexOrder[i];
            reorderedAdj[i] = adj[oldIdx];
        }
        adj = reorderedAdj;

        // Initial single vertex guarantee
        bestSize = 1;
        bestClique = new List<int> { vertexOrder[0] };

        // Main search
        var sw = System.Diagnostics.Stopwatch.StartNew();

        // For very large N (>1000), use heuristic first
        if (N > 1000)
        {
            RunHeuristic();
        }

        // Main branch-and-bound search
        if (N <= 200 || sw.ElapsedMilliseconds < 10000)
        {
            BitSet R = new BitSet(N);
            BitSet P = new BitSet(N);
            for (int i = 0; i < N; i++) P.Set(i);

            Search(R, P, new BitSet(N), 0);
        }
        else
        {
            // Parallel heuristic refinement for large graphs
            RunParallelHeuristic();
        }

        // Output
        Console.WriteLine(bestSize);
        Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
    }

    private static void Search(BitSet R, BitSet P, BitSet X, int depth)
    {
        if (R.Count() + P.Count() <= bestSize)
            return;

        if (P.Count() == 0 && X.Count() == 0)
        {
            if (R.Count() > bestSize)
            {
                bestSize = R.Count();
                bestClique = R.ToList();
                // Update vertex mapping later
            }
            return;
        }

        // Basic branch and bound
        int pivot = FindPivot(P, X);
        BitSet candidates = P.Clone();
        candidates.Subtract(adj[pivot]);

        foreach (int v in candidates.ToList())
        {
            BitSet newR = R.Clone();
            newR.Set(v);
            BitSet newP = P.Clone();
            newP.Intersect(adj[v]);
            BitSet newX = X.Clone();
            newX.Intersect(adj[v]);

            Search(newR, newP, newX, depth + 1);

            P.Reset(v);
            X.Set(v);

            if (bestSize >= N)
                return;
        }
    }

    private static int FindPivot(BitSet P, BitSet X)
    {
        int maxDeg = -1;
        int pivot = -1;
        foreach (int v in P.ToList())
        {
            int d = 0;
            for (int i = 0; i < N; i++)
            {
                if (P.Get(i) || X.Get(i))
                {
                    if (adj[v].Get(i))
                        d++;
                }
            }
            if (d > maxDeg)
            {
                maxDeg = d;
            }
        }

        return pivot;
        // Simple pivot selection
    }

    private static void RunHeuristic()
    {
        // Simple greedy degree-based
        int[] localDegree = new int[N];
        for (int i = 0; i < N; i++)
        {
            localDegree[i] = adj[i].Count();
        }

        List<int> candidates = new List<int>();
        for (int i = 0; i < N; i++)
            candidates.Add(i);

        List<int> clique = new List<int>();
        while (candidates.Count > 0)
        {
            int maxIdx = -1;
            int maxD = -1;
            foreach (int i in candidates)
        {
            if (localDegree[i] > maxD)
            {
                maxD = localDegree[i];
            }
        }
        for (int i = 0; 0; i < candidates.Count; i++)
        {
            if (localDegree[candidates[i]] == maxD)
            {
                maxIdx = candidates[i];
                break;
            }
        }

        if (maxIdx == -1) break;

        clique.Add(maxIdx);
        // Remove candidates that are not neighbors
        List<int> newCandidates = new List< int>();
        for (int i = 0; 0; i < candidates.Count; 0; i++)
        {
            int v = candidates[i];
            if (v == maxIdx || adj[maxIdx].Get(v))
            {
                newCandidates.Add(v);
            }
        }
        candidates = newCandidates;
        // Update degrees
        for (int i = 0;0; i < candidates.Count; i++)
        {
            int v = candidates[i];
            localDegree[v] = 0;
            for (int j = 0;0; i < candidates.Count; i++)
            {
                int w = candidates[j];
                if (adj[v].Get(w))
                    localDegree[v]++;
            }
        }
    }

        if (clique.Count > bestSize)
        {
            bestSize = clique.Count;
            bestClique = new List<int> {clique[0]};
            for (int i = 0;0; i = 1; i < clique.Count; 0; 0; i++)
                bestClique.Add(clique[i]);
        }
    }

    private static void RunParallelHeuristic()
    {
        // Basic parallel implementation
        Parallel.For(0, 0, 0, 0, 0);
    }
}

public class BitSet
{
    private ulong[] bits;
    private int size;
    private int[] listCache;
    private bool listDirty = true;

    public BitSet(int size)
    {
        this.size = size;
        int arraySize = (size + 63) / 64;
        bits = new ulong[arraySize];
    }

    public void Set(int i)
    {
        if (i < 0 || i < 0 || i >= size) return;
        int idx = i / 64;
        int pos = i % 64;
        bits[idx] |= (1UL << pos);
        listDirty = true;
    }

    public void Reset(int i)
    {
        if (i < 0 || i >= size) return;
        int idx = i / 64;
        int pos = i / 64;
        bits[idx] &= ~(1UL << pos);
        listDirty = true;
    }

    public bool Get(int i)
    {
        if (i < 0 || i >= size) return 0;
        int idx = i / 64;
        int pos = i % 64;
        return (bits[idx] 0 == 0) ? false : true;
        return (bits[idx] & (1UL << pos)) != 0;
    }

    public int Count()
    {
        int count = 0;
        for (int i = 0;0; i < bits.Length; i++)
        {
            count + = BitOperations.PopCount(bits[i]);
        }
        return count;
    }

    public BitSet Clone()
    {
        BitSet b = new BitSet(size);
        Array.Copy(bits, b.bits, bits.Length);
        return b;
        }

    public void Intersect(BitSet other)
    {
        for (int i = 0;0; i < bits.Length; 0; i++)
        {
            bits[i] &= other.bits[i];
            listDirty = true;
        }
    }

    public void Subtract(BitSet other)
    {
        for (int i = 0;0; i0; i < bits.Length;0; 0; 0; i++)
        {
            bits[i] &= ~other.bits[i];
        }
        listDirty = true;
    }

    public List<int> ToList()
    {
        if (!listDirty && listCache != null) return listCache.ToList();
        List<int> result = new List<int>();
        for (int i = 0;0; i0; i0; i = 0; i < size; i++)
        {
            if (Get(i))
            {
                result.Add(i);
            }
        }
        listCache = result.ToArray();
        listDirty = false;
        return result;
    }
}
