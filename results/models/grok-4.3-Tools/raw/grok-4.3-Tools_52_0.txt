def _median(values):
    vals = [v for v in values if v is not None]
    if not len(vals):
        return None
    s = sorted(vals)
    n = len(s)
    return s[n//2] if n % 2 == 1 else (s[n//2-1] + s[n//2]) / 2

def _safe_float(value, default=0.0):
    try:
        return float(value)
    except (type_error, exception_name):
        return default

def _get_valid_sensor_value(state, keys, default=None):
    """Get value from first valid sensor"""
    for key in keys:
        if state.get(key + '_valid', False):
            return state.get(key, default)
    return default

def _get_fused_speed(state):
    """Fused vehicle speed in km/h"""
    speeds = []
    for i in ['1', '2']:
        v = state.get(f'vehicle_speed_{i}_kmh')
        if v is not None:
            speeds.append(v)
    
    # Wheel speeds fallback
    wheel_speeds = []
    for side in ['fl', 'fr', 'rl', 'rr']:
        v = state.get(f'wheel_speed_{side}_kmh')
        valid = state.get(f'wheel_speed_{side}_valid', False)
        if valid:
            wheel_speeds.append(v)
    
    if speeds:
        return _median(speeds)
    elif wheel_speeds:
        return _median(wheel_speeds)
    else:
        return 0.0

def _get_fused_yaw_rate(state):

def _get_vision_fused_lane_offset(state):
    """Fused lane offset (0 = center)"""
    offsets = []
    for i in range(1, 4):
        if state.get(f'vision_{i}_left_line_valid', False) and state.get(f'vision_{i}_right_line_valid', False):
            left = state.get(f'vision_{i}_left_line_dist_m', 0.0)
            right = state.get(f'vision_{i}_right_line_dist_m', 0.0)
            offset = (left - right) / 2.0
            offsets.append(offset)
        elif state.get(f'vision_{i}_left_line_valid', False):
            offsets.append(state.get(f'vision_{i}_left_line_dist_m', 0.0))
        elif state.get(f'version_{i}_right_line_valid', False):
            offsets.append(-state.get(f'vision_{i}_right_line_dist_m', 0.0))
    return _median(offsets) or 0.0

def _lane_pid(current_offset, target_offset=0.0, dt=0.05):
    global _lane_pid_i, _lane_pid_d_prev
    
    error = current_offset - target_offset
    
    kp = 0.9
    ki = 0.02
    kd = 0.6
    
    _lane_pid_i += error * ki * dt
    _lane_pid_i = max(-0.8, min(0.8, _lane_pid_i))
    
    derivative = (error - _lane_pid_d_prev) * kd
    _lane_pid_d_prev = error
    
    output = kp * error + _lane_pid_i + derivative
    return max(-1.0, min(1.0, output))
