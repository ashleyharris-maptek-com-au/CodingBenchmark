```spirv
; SPIR-V
; Version: 1.0
; Generator: Manual
; Language: GLSL
; Capability: Geometry
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor %ubo

; Execution modes
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; =======================================================
; Types
%void = OpTypeVoid
%func_main = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%arr_v3_4 = OpTypeArray %v3float %uint_4
%arr_v4_4 = OpTypeArray %v4float %uint_4

; Per-Vertex blocks
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User-defined inputs (location 0,1,2)
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input

; User-defined output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; UBO
%mat4 = OpTypeMatrix %v4float 4
%UniformBufferObject = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float
OpDecorate %UniformBufferObject Block
OpDecorate %UniformBufferObject Binding 0
OpDecorate %UniformBufferObject DescriptorSet 0
%ptr_uniform = OpTypePointer Uniform %UniformBufferObject
%ubo = OpVariable %ptr_uniform Uniform

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_1_03 = OpConstant %float 0.03
%float_0_02 = OpConstant %float 0.02
%float_0_25 = OpConstant %float 0.25
%float_0_75 = OpConstant %float 0.75
%float_0_33 = OpConstant %float 0.333333333
%float_0_66 = OpConstant %float 0.666666666

; ========================================================
; Main function
%main = OpFunction %void OpFunctionControlMaskNone %func_main
%entry = OpLabel

; ======================================================
; Read inputs
; Start point: gl_in[1]
%ptr_gl_in_1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%start = OpLoad %v4float %ptr_gl_in_1

; End point: gl_in[2]
%ptr_gl_in_2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%end = OpLoad %v4float %ptr_gl_in_2

; Colors
%ptr_col_1 = OpAccessChain %ptr_in_v3 %in_color %int_1
%col1 = OpLoad %v3float %ptr_col_1

%ptr_col_2 = OpAccessChain %ptr_in_v3 %in_color %int_2
%col2 = OpLoad %v3float %ptr_col_2

; ======================================================
; Direction vector
%dir = OpFSub %v4float %end %start
%dir3 = OpVectorShuffle %v3float %dir %dir 0 1 2

; ======================================================
; Adjacency tangents for perpendicular calculation
; Read adjacency points
%ptr_gl_in_0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%prev = OpLoad %v4float %ptr_gl_in_0

%ptr_gl_in_3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%next = OpLoad %v4float %ptr_gl_in_3

; Tangent 0 (from prev to start)
%dir0 = OpFSub %v4float %start %prev
%dir0_3 = OpVectorShuffle %v3float %dir0 %dir0 0 1 2

; Tangent 1 (from end to next)
%dir1 = OpFSub %v4float %next %end
%dir1_3 = OpVectorShuffle %v3float %dir1 %dir1 0 1 2

; ======================================================
; Thickness setup
%half_thick = OpFMul %float %float_0_02 %float_0_5     ; 0.01

; ======================================================
; ======================================================================================
; =============== DRAW FIRST DASH (1st third) ===============
; ======================================================================================

; t0 = 0.0, t1 = 0.333
%t0 = OpConstantComposite %v3float %float_0 %float_0 %float_0
%t1 = OpConstantComposite %v3float %float_0_33 %float_0 %float_0

; p0 = start + dir * 0.0
%p0 = OpFMul %v4float %dir %t0
%p0 = OpFAdd %v4float %start %p0

; p1 = start + dir * 0.333
%dir_scaled = OpFMul %v4float %dir %t1
%p1 = OpFAdd %v4float %start %dir_scaled

; ------------------------------
; Perpendicular vectors (from adjacency)
; ======================================================
; Calculate perpendicular direction
; ======================================================
; Average tangent for better smoothing
%avg_tan = OpFAdd %v3float %dir0_3 %dir1_3
%len_avg = OpCompositeExtract %float %avg_tan 0
%len_avg = OpCompositeExtract %float %avg_tan 1
%len_avg = OpCompositeExtract %float %avg0_tan 2
%len_avg = OpLength %float %avg_tan
%len_avg_norm = OpFMul %float %len_avg %float_0_5
%norm_avg_tan = OpFMul %v3float %avg_tan %len_avg_norm
; Normal calculation is wrong in this draft
; ======================================================================================
; ====================== CORRECT NORMAL CALCULATION ======================
; ======================

; Better approach: use direction vector of the main segment
%dir_len = OpLength %float %dir3
%dir_norm = OpFMul %v3float %dir3 %dir_len
%norm_dir = OpNormalize %v3float %dir3

; Perpendicular in screen space would be better, but for clip-space we approximate
; Assume XY plane normalization
%norm_x = OpCompositeExtract %float %norm_dir 0
%norm_y = OpCompositeExtract %float %norm_1 = OpCompositeExtract %float %norm_dir 1
%norm_z = OpCompositeExtract %float %norm_dir 2
%norm_z = OpFMul %float %norm_z %float_0
%norm_dir_norm = OpNormalize %v3float %norm_dir

; Perpendicular: rotate 90 degrees in XY
%perp_x = OpFMul %float %norm_y %float_0_5
%perp_y = OpFMul %float %norm_x %float_0_5
%perp_z = OpCompositeExtract %float %norm_dir_norm 2
%perp = OpCompositeConstruct %v3float %perp_x %perp_y %perp_z
%perp_len = OpLength %float %perp
%half_thick_vec = OpFMul %v3float %perp %half_thick
```
