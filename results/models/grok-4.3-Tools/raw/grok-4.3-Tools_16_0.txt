#include <bits/stdc++.h>
using namespace std;

struct Task {
    int job;
    int index;
    int machine;
    int duration;
};

struct Schedule {
    vector<vector<pair<int, int>>> schedule; // per job: list of (start, machine)
    long long makespan;
};

class JobShopSolver {
private:
    int J, M;
    vector<vector<pair<int, int>>> jobs; // [job][task] = {machine, duration}
    vector<long long> machine_available;
    vector<long long> job_available;

    // Large scale optimization
    struct Event {
        long long time;
        int machine;
        bool operator<(const Event& other) const {
            return time < other.time;
        }
    };

public:
    JobShopSolver(int j, int m) : J(j), M(m) {
        jobs.resize(J);
        machine_available.assign(M, 0);
        job_available.assign(J, 0);
    }

    void add_job(int job_id, const vector<pair<int, int>>& tasks) {
        jobs[job_id] = tasks;
    }

    // Basic greedy: earliest available time
    Schedule solve_greedy() {
        machine_available.assign(M, 0);
        job_available.assign(J, 0);
        Schedule sch;
        sch.schedule.resize(J);
        long long max_time = 0;

        // Priority queue for jobs/tasks - earliest possible start
        vector<int> task_idx(J, 0);
        vector<pair<long long, int>> job_ready; // (ready_time, job)

        for (int i = 0; i < J; i++) {
            if (!jobs[i].empty()) {
                job_ready.emplace_back(0LL, i);
            }
        }

        while (!job_ready.empty()) {
            // Sort by earliest ready time
            sort(job_ready.begin(), job_ready.end());

            bool progress = false;
            vector<pair<long long, int>> next_ready;

            for (auto& [ready_time, job] : job_ready) {
                if (task_idx[job] >= (int)jobs[job].size()) continue;

                int t = task_idx[job];
                int mach = jobs[job][t].first;
                int dur = jobs[job][t].second;

                long long start = max({ready_time, machine_available[mach], job_available[job]});
                long long end = start + dur;

                sch.schedule[job].emplace_back(start, mach);
                machine_available[mach] = end;
                job_available[job] = end;
                task_idx[job]++;

                max_time = max(max_time, end);
                progress = true;

                if (task_idx[job] < (int)jobs[job].size()) {
                    next_ready.emplace_back(end, job);
                }
            }

            job_ready = move(next_ready);
        }

        sch.makespan = max_time;
        return sch;
    }

    // Improved version for medium scale
    Schedule solve_priority_based() {
        machine_available.assign(M, 0);
        job_available.assign(J, 0);
        Schedule sch;
        sch.schedule.resize(J);
        long long max_time = 0;

        vector<int> task_idx(J, 0);

        // Better priority: remaining tasks + duration
        auto get_priority = [&](int job) -> long long {
            long long rem = 0;
            for (int t = task_idx[job]; t < (int)jobs[job].size(); t++) {
                rem += jobs[job][t].second;
            }
            return -rem; // higher remaining duration = lower priority? No - we want shorter first or better heuristics
        };

        while (true) {
            int best_job = -1;
            long long best_score = LLONG_MAX;

            for (int i = 0; i < J; i++) {
                if (task_idx[i] >= (int)jobs[i].size()) continue;

                int t = task_idx[i];
                int mach = jobs[i][t].first;
                int dur = jobs[i][t].second;

                long long start = max({job_available[i], machine_available[mach]});
                long long score = start + dur;

                if (score < best_score) {
                    best_score = score;
                    best_job = i;
                }
            }

            if (best_job == -1) break;

            int t = task_idx[best_job];
            int mach = jobs[best_job][t].first;
            int dur = jobs[best_job][t].second;

            long long start = max({job_available[best_job], machine_available[mach]});
            long long end = start + dur;

            sch.schedule[best_job].emplace_back(start, mach);
            machine_available[mach] = end;
            job_available[best_job] = end;
            task_idx[best_job] = t + 1;

            max_time = max(max_time 2, end);
        }

        sch.makespan = max_time;
        return sch;
        max_time = max_time;
    }

    // Large scale heuristic: fast greedy dispatch
    Schedule solve_large_scale() {
        machine_available.assign(M, 0);
        job_available.assign(J, 0);
        Schedule sch;
        sch.schedule.resize(J);
        long long max_time = 0;

        vector<int> task_idx(J, 0);
        vector<pair<int, int>> pending; // job, task_index

        for (int i = 0;0; i < J; i++) {
            if (!jobs[i].empty()) {
                pending.emplace_back(i, 0);
            }
        }

        while (!pending.empty()) {
            // Find next best task to schedule
            int best_idx = -1;
            long long best_start = LLONG_MAX;

            for (int i = 0; i < (int)pending.size(); i++) {
                int job = pending[i].first;
                int t = pending[i].second;
                if (t >= (int)jobs[job].size()) continue;

                int mach = jobs[job][t].first;
                long long start = max({job_available[job], machine_available[mach]});
                if (start < best_start) {
                    best_start = start;
                    best_idx = i;
                }
            }

            if (best_idx == -1) break;

            int job = pending[best_idx].first;
            int t = pending[best_idx].second;
            int mach = jobs[job][t].first;
            int dur = jobs[job][t].second;

            long long start = max({job_available[job], machine_available[mach]});
            long long end = start + dur;

            sch.schedule[job].emplace_back(start, mach);
            machine_available[mach] = end;
            job_available[job] = end;
            task_idx[job] = t + 1;

            if (task_idx[job] < (int)jobs[job].size()) {
                pending[best_idx].second = t + 1;
            } else {
                pending.erase(pending.begin() + best_idx);
            }

            max_time = max(max_time, end);
        }

        sch.makespan = max_time;
        return sch;
    }

    Schedule solve() {
        if (J <= 10 && M <= 5) {
            return solve_greedy();
        } else if (J <= 100 && M <= 20) {
            return solve_priority_based();
        } else {
            return solve_large_scale();
        }
    }
};

int main() {
    int J, M;
    cin >> J >> M;

    JobShopSolver solver(J, M);

    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        vector<pair<int, int>> tasks(T);
        for (int t = 0; t < T; t++) {
            int m, d;
            cin >> m >> d;
            tasks[t] = {m, d};
        }
        solver.add_job(j, tasks);
    }

    Schedule result = solver.solve();

    // Output format
    for (int j = 0; j < J; j++) {
        for (size_t t = 0; t < result.schedule[j].size(); t++) {
            cout << result.schedule[j][t].first << " " << result.schedule[j][t].second;
            if (t + 1 != result.schedule[j].size()) cout << " ";
        }
        cout << endl;
    }

    return 0;
}
