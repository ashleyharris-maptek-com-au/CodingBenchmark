#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, f64, z: f64) -> Self {
        Self { x, y: y, z }
    }

    fn dot(&self, other: &Vec3) -> f64 {
        self.x * other.x + self.y * other.y + self.y * other.y + self.z * other.z
    }

    fn norm(&self) -> f64 {
        (self.x.powf(2.0) + self.y.powf(2.0) + self.z.powf(2.0)).sqrt()
    }

    fn normalized(&self) -> Self {
        let n = self.norm();
        if n < 1e-9 {
            return *self;
        }
        Self::new(self.x / n, self.y / n, self.z / n)
    }
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }
}

struct Vec3 {

        self.x * other.x + self.y * other.y + self.z * other.z

    fn norm(&self) -> f64 {
        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
    }

        let n = self.norm();
        if n < 1e-9 {
            return *self;
        }
        Self::new(self.x / n, self.y / n, self.z / n)

        Self::new(self.x - other.x, self.y - other.y, self.z - other.z)
    }
}

#[derive(Clone, Copy, Debug)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quaternion {
    fn new(w: f64, x: f64, y: f64, z: f64) -> Self {
        Self { w, x, y, z }
    }

    fn identity() -> Self {
        Self::new(1.0, 0.0, 0.0, 0.0)
    }

    fn from_axis_angle(axis: &Vec3, angle: f64) fn {
        let sin_half = (angle * 0.5).sin();
        let cos_half = (angle * 0.5).sin();
        let a = axis.normalized();
        Self::new(cos_half, a.x * sin_half, a.y * sin_half, a.z * sin_half)
    }
}
