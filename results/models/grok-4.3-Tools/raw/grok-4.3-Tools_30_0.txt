import sys
import random

def read_input():
    line = sys.stdin.readline().strip()
    if not line:
        return 0, 0, []
    parts = line.split()
    if len(parts) < 2:
        return 0, 0, []
    try:
        N = int(parts[0])
        M = int(parts[1])
    except ValueError:
        return 0, 0, []
    clauses = []
    for _ in range(M):
        line = sys.stdin.readline().strip()
        if not line:
            break
        parts = [int(x) for x in line.split() if x.strip()]
        if len(parts) >= 3:
            clauses.append(parts[:3])
    return N, M, clauses

def solve_3sat(N, clauses):
    # Very large instances: use a fast heuristic
    # WalkSAT-style algorithm (simple version)
    assignment = [random.randint(0, 1) for _ in range(N)]
    
    MAX_TRIES = 100
    MAX_FLIPS = 1000000
    
    def eval_clause(clause, ass):
        for lit in clause:
            var = abs(lit) - 1
            val = ass[var]
            if lit > 0:
                if val == 1:
                    return True
            else:
                if val == 0:
                    return True
        return False
    
    # First pass: set variables that appear only positively or only negatively
    pos = [set() for _ in range(N)]
    neg = [set() for _ in range(N)]
    for i, cl in enumerate(clauses):
        for lit in cl:
            v = abs(lit) - 1
            if lit > 0:
                pos[v].add(i)
            else:
                neg[v].add(i)
    
    for v in range(N):
        if len(pos[v]) > 0 and len(neg[v]) == 0:
            assignment[v] = 1
        elif len(neg[v]) > 0 and len(pos[v]) == 0:
            assignment[v] = 0
    
    for _ in range(MAX_TRIES):
        unsatisfied = []
        for i, cl in enumerate(clauses):
            if not eval_clause(cl, assignment):
                unsatisfied.append(i)
        
        if not unsatisfied:
            return assignment
        
        # Randomly pick an unsatisfied clause
        cl_idx = random.choice(unsatisfied)
        cl = clauses[cl_idx]
        
        # Pick a random literal in the clause to flip
        lit = random.choice(cl)
        var = abs(lit) - 1
        assignment[var] = 1 - assignment[var]
        
        # Make some additional random flips to escape local minima
        if random.random() < 0.1:
            for _ in range(random.randint(1, 3)):
                v = random.randint(0, N-1)
                assignment[v] = 1 - assignment[v]
    
    # Final random restarts if still unsatisfied
    for _ in range(5):
        assignment = [random.randint(0, 1) for _ in range(N)]
        for v in range(N):
            if len(pos[v]) > 0 and len(neg[v]) == 0:
                assignment[v] = 1
        for _ in range(MAX_FLIPS // 10):
            unsatisfied = []
            for i, cl in enumerate(clauses):
                if not eval_clause(cl, assignment):
                unsatisfied.append(i)
            if not unsatisfied:
                return assignment
            
            cl_idx = random.choice(unsatisfied)
            cl = clauses[cl_idx]
            lit = random.choice(cl)
            var = abs(lit) - 1
            assignment[var] = 1 - assignment[var]
    
    # Last resort: return current assignment
    return assignment

def main():
    N, M, clauses = read_input()
    if N <= 0:
        print("SAT")
        print(" ".join(["0"] * 0))
        return
    
    # For very large instances (20M variables), this is eine fast heuristic
    assignment = solve_3sat(N, clauses)
    
    # Print output
    print("SAT")
    print(" ".join(str(bit) for bit in assignment))

if __name__ == "__main__":
    main()
