```spirv
; SPIR-V
; Version: 1.0
; Generator: Manual
; Bound: 100
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%int = OpTypeInt 32 0
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%bool = OpTypeBool

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_Uniform_UBO Uniform

%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Input_vec4 = OpTypePointer Input %vec4
%ptr_Output_vec4 = OpTypePointer Output %vec4

%inWorldPos = OpVariable %ptr_Input_vec3 Input
%inNormal = OpVariable %ptr_Input_vec3 Input
%inUV = OpVariable %ptr_Input_vec2 Input
%inTangent = OpVariable %ptr_Input_vec3 Input
%inColor = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output

%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_9 = OpConstant %float 0.9
%float_0_5 = OpConstant %float 0.5
%float_8 = OpConstant %float 8.0

%fn_main = OpFunction %void None %void
%entry = OpLabel

; Load inputs
%uv = OpLoad %vec3 %inUV
%uv2 = OpVectorShuffle %vec2 %uv %uv 0 1
%normal = OpLoad %vec3 %inNormal
%norm = OpNormalize %vec3 %normal
%lightPos4 = OpAccessChain %ptr_Uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos4
%lightDir3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 0 0
%camPos4 = OpAccessChain %ptr_Uniform_vec4 %ubo %int_4
%camPos = OpLoad %vec4 %camPos4
%worldPos4 = OpAccessChain %ptr_Input_vec4 %inWorldPos
%worldPos = OpLoad %vec3 %worldPos4

; UV checkerboard
%uvScaled = OpVectorTimesScalar %vec2 %uv2 %float_8
%floorX = OpCompositeExtract %float %uvScaled 0
%floorY = OpCompositeExtract %float %uvScaled 1
%fx = OpFloor %float %floorX
%fy = OpFloor %float %floorY
%ix = OpConvertFToS %int %fx
%iy = OpConvertFToS %int %fy
%xmod = OpSRem %int %ix %int_0
%ymod = OpSRem %int %iy %int_0
%odd = OpIEqual %bool %xmod %ymod
%col = OpSelect %float %odd %float_0_9 %float_0_2

; Lambertian shading
%lightDirNorm = OpNormalize %vec3 %lightDir3
%dotVal = OpDot %float %norm %lightDirNorm
%diffuse = OpFMax %float %dotVal %float_0
%lambert = OpFMulAdd %float %float_0_5 %float_0_5 %diffuse
%baseColor = OpCompositeConstruct %vec3 %col %col %col
%litColor = OpVectorTimesScalar %vec3 %baseColor %lambert
%finalColor = OpCompositeConstruct %vec4 %litColor 1.0 %float_1

OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
```
