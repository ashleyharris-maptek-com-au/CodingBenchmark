const fs = require('fs');

// Read input
const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const width = data.width;
const height = data.height;
const grid = data.grid;
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = data.cutBudget || 0;

// Directions for spreading
const dirs = [[0, 1], [1, 0], [0, -1], [-1, 0]];

// Helper functions
function isValid(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function isBurnable(x, y) {
  if (!isValid(x, y)) return false;
  return grid[y][x] === '.';
}

function getKey(x, y) {
  return `${x},${y}`;
}

// 1. Simulate initial fire spread (before cuts)
function simulateFire(cutsSet) {
  const visited = new Set();
  const queue = [];
  
  for (const [x, y] of ignitions) {
    if (isBurnable(x, y) && !cutsSet.has(getKey(x, y))) {
      const key = getKey(x, y);
      visited.add(key);
      queue.push([x, y]);
    }
  }
  
  while (queue.length > 0) {
    const [x, y] = queue.shift();
    
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      const nkey = getKey(nx, ny);
      
      if (isBurnable(nx, ny) && 
          !cutsSet.has(nkey) && 
          !visited.has(nkey)) {
        visited.add(nkey);
        queue.push([nx, ny]);
      }
    }
  }
  
  return visited;
}

// 4. Priority queue for best cuts (greedy approach)
class PriorityQueue {
  constructor() {
    this.items = [];
  }
  
  enqueue(item, priority) {
    this.items.push({ item, priority });
    this.items.sort((a, b) => b.priority - a.priority);
  }
  
  dequeue() {
    return this.items.shift();
  }
  
  isEmpty() {
    return this.items.length === 0;
  }
}

// 5. Heuristic: Find good places to cut
function findBestCuts(budget) {
  const cuts = [];
  const used = new Set();
  
  // Strategy: Find bottlenecks and high-value towns' access points
  // Better approach: Model as multi-source BFS and find critical paths
  
  // First, compute distance from fire front
  const dist = Array.from({ length: height }, () => Array(width).fill(Infinity));
  const fireQueue = [];
  
  // Initialize fire front
  const tempCuts = new Set();
  const fireFront = simulateFire(tempCuts);
  
  for (const key of fireFront) {
    const [x, y] = key.split(',').map(Number);
    dist[y][x] = 0;
    fireQueue.push([x, y]);
  }
  
  // Multi-source BFS for distance from fire
  let idx = 0;
  while (idx < fireQueue.length) {
    const [x, y] = fireQueue[idx++];
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (isBurnable(nx, ny) && dist[ny][nx] > dist[y][x] + 1) {
        dist[ny][nx] = dist[y][x] + 1;
        fireQueue.push([nx, ny]);
      }
    }
  }
  
  // Collect potential cut candidates
  const candidates = [];
  
  for (let y = 0; y < height; y++) {
    for (let x = 0; x < width; x++) {
      if (!isBurnable(x, y)) continue;
      const key = getKey(x, y);
      
      if (used.has(key)) continue;
      
      // Skip ignitions
      let isIgnition = false;
      for (const [ix, iy] of ignitions) {
        if (ix === x && iy === y) {
          isIgnition = true;
          break;
        }
      }
      if (isIgnition) continue;
      
      let score = 0;
      
      // Prefer cells near current fire front
      if (dist[y][x] < Infinity) {
        score += 100 - Math.min(dist[y][x], 100);
      }
      
      // Count neighbors that could be blocked
      let neighborCount = 0;
      for (const [dx, dy] of dirs) {
        const nx = x + dx;
        const ny = y + dy;
        if (isBurnable(nx, ny)) {
          neighborCount++;
        }
      }
      
      // Prefer narrow passes / high connectivity cells
      score += neighborCount * 3;
      
      // Check if this cell is on path to any town
      let townScore = 0;
      for (const town of towns) {
        const townKey = getKey(town.x, town.y);
        if (!baselineBurned.has(townKey)) continue; // Only towns at risk
        
        // Simple proximity to town
        const dx = Math.abs(x - town.x);
        const dy = Math.abs(y - town.y);
        const distance = dx + dx + dy + dy;
        if (distance < 30) {
          townScore += town.value * (1.0 / Math.max(1, distance));
        }
      }
      
      score += townScore;
      
      candidates.push({ x, y, score });
    }
  }
  
  // Sort by score descending
  candidates.sort((a, b) => b.score - a.score);
  
  // Pick top budget candidates
  const selectedCuts = [];
  const cutsSet = new Set();
  
  for (const cand of candidates) {
    if (selectedCuts.length >= budget) break;
    const key = getKey(cand.x, cand.y);
    if (!cutsSet.has(key)) {
      selectedCuts.push([cand.x, cand.y]);
      cutsSet.add(key);
    }
  }
  
  return selectedCuts;
}
