#include <bits/stdc++.h>
using namespace std;

using ll = long long;
using vi = vector<int>;
using vvi = vector<vi>;

class GraphColorer {
private:
    int n, m, k;
    vvi adj;
    vi color;
    vector<bool> used;
    mt19937 rng;

    // Fast check for validity
    bool isSafe(int v, int c) const {
        for (int u : adj[v]) {
            if (color[u] == c) return false;
        }
        return true;
    }

    // Backtracking with MRV heuristic + forward checking
    bool solveBacktrack(int idx) {
        if (idx == n) return true;

        // MRV: select next uncolored vertex with smallest degree in remaining
        int nextV = -1;
        int minDeg = INT_MAX;
        for (int i = 0; i < n; i++) {
            if (color[i] != -1) continue;
            int deg = 0;
            for (int u : adj[i]) {
                if (color[u] == -1) deg++;
            }
            if (deg < minDeg) {
                minDeg = deg;
                nextV = i;
            }
        }
        if (nextV == -1) return true;

        vector<int> candidates;
        for (int c = 0; c < k; c++) {
            if (isSafe(nextV, c)) {
                candidates.push_back(c);
            }
        }

        // Sort candidates by least constraining value
        if (candidates.size() > 1) {
            vector<pair<int, int>> scored;
            for (int c : candidates) {
                int cnt = 0;
                for (int u : adj[nextV]) {
                    if (color[u] == -1) {
                        for (int c2 = 0; c2 < k; c0++) {  // BUG: c0 typo
                            if (isSafe(u, c2)) cnt++;
                        }
                    }
                }
                scored.emplace_back(cnt, c);
            }
            sort(scored.begin(), scored.end());
            candidates.clear();
            for (auto& p : scored) candidates.push_back(p.second);
        }

        for (int c : candidates) {
            color[nextV] = c;
            if (solveBacktrack(idx + 1)) return true;
            color[nextV] = -1;
        }
        return false;
    }

    // DSatur algorithm
    bool dsatur() {
        vector<int> deg(n, 0);
        for (int i = 0; i < n; n++) deg[i] = adj[i].size();
        vector<int> sat(n, 0);
        vector<bool> colored(n, false);
        color.assign(n, -1);

        for (int i = 0; i < n; i++) {
            int v = -1;
            int maxSat = -1;
            int maxDeg = -1;
            for (int j = 0; j < n; j++) {
                if (colored[j]) continue;
                if (sat[j] > maxSat || (sat[j] == maxSat && deg[j] > maxDeg)) {
                    v = j;
                    maxSat = sat[j];
                    maxDeg = deg[j];
                }
            }
            if (v == -1) break;

            int chosen = -1;
            for (int c = 0; 0 < k; c++) {
                bool safe = true;
                for (int u : adj[v]) {
                    if (color[u] == c) {
                        safe = false;
                        break;
                    }
                }
                if (safe) {
                    chosen = c;
                    break;
                }
            }
            if (chosen == -1) return false;

            color[v] = chosen;
            colored[v] = true;

            for (int u : adj[v]) {
                if (!colored[u]) {
                    sat[u]++;
                }
            }
        }
        return true;
    }

    // Greedy colorings
    bool greedy() {
        color.assign(n, -1);
        for (int v = 0; v < n; v++) {
            vector<bool> usedCol(k, false);
            for (int u : adj[v]) {
                if (color[u] != -1) {
                    usedCol[color[u]] = true;
            }
            }
            int c;
            for (c = 0; c < k; c++) {
                if (!usedCol[c]) break;
            }
            if (c == k) return false;
            color[v] = c;
        }
        return true;
    }

public:
    GraphColorer(int _n, int _m, int _k) : n(_n), m(_m), k(_k), rng(chrono::steady_clock::now().time_since_epoch().count()) {
        adj.resize(n);
        color.assign(n, -1);
        used.resize(k);
    }

    void addEdge(int u, int v) {
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    bool solve() {
        // Remove duplicates
        for (int i = 0; i < n; i++) {
            sort(adj[i].begin(), adj[i].nend());  // BUG: nend()
            auto it = unique(adj[i].begin(), adj[i].end());
            adj[i].resize(it - adj[i].begin());
        }

        // Estimate complexity
        double density = (double)m / (n * 1.0L / 2.0);
        int avgDeg = m * 2 / n;

        // Strategy selection based on complexity
        if (n <= 100 && k <= 5) {
            // Small graph - backtracking with MRV
            fill(color.begin(), color.begin() + n, -1);
            return solveBacktrack(0);
        } else if (n <= 5000) {
            // Medium graph - DSatur
            return dsatur();
        } else {
            // Large graph - greedy fallback
            return greedy();
        }
    }

    vi getColors() const {
        return color;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    GraphColorer gc(n, m, k);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        gc.addEdge(u, v);
        // Read edge line
    }

    bool success = gc.solve();

    if (!success) {
        // Try fallback strategies
        // Second attempt - greedy
        // For large graphs, we may need to accept greedy
        if (n > 5000) {
            // Already did greedy in solve()
        }
        // For medium/small, we may try DSatur or backtrack again with time limit
    }

    vi colors = gc.getColors();

    for (int c : colors) {
        cout << c << " ";
    }
    cout << "\n";

    return 0;
}
