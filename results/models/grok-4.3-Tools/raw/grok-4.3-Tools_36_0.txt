#include <bits/stdc++.h>
using namespace std;

using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<vector<int>> adj(n);
    for (int i = 0;0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    
    // Remove duplicates and sort for binary_search later
    for (int i = 0; i < n; i++) {
        sort(adj[i].begin(), adj[i].end());
        auto it = unique(adj[i].begin(), adj[i].end());
        adj[i].resize(it - adj[i].begin());
    }
    
    // For very small n (<= 20): exact DP bitmask
    if (n <= 20) {
        int max_size = 0;
        int best_mask = 0;
        
        int total = 1 << n;
        for (int mask = 0; mask < total; mask++) {
            bool valid = true;
            int size = 0;
            for (int i = 0; i < n; i++) {
                if (mask & (1 << i)) {
                    size++;
                    for (int j : adj[i]) {
                        if (j > i && (mask & (1 << j))) {
                            valid = false;
                            break;
                        }
                    }
                    if (!valid) break;
                }
            }
            if (valid && size > max_size) {
                max_size = size;
                best_mask = mask;
            }
        }
        
        cout << max_size << '\n';
        bool first = true;
        for (int i = 0; i < n; i++) {
            if (mask & (1 << i)) {
                if (!first) cout << ' ';
                cout << i;
                first = false;
            }
        }
        cout << '\n';
        return 0;
    }
    
    // For medium n (<= 40): branch and bound
    // TODO: Implement advanced branch-and-bound or MILP solver
    // TODO: For n > 20 much harder
    
    // Placeholder for larger n: greedy approximation
    // Note: This is NOT guaranteed to be optimal
    // For correct exact solution for n > 20, you must replace this section with real MIS solver
    
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        return adj[a].size() < adj[b].size();
    });
    
    vector<bool> used(n, false);
    vector<int> ind_set;
    for (int i : order) {
        if (used[i]) continue;
        bool can = true;
        for (int j : adj[i]) {
            if (used[j]) {
                can = false;
                break;
        }
        }
        if (can) {
            ind_set.push_back(i);
            used[i] = true;
            for (int j : adj[i]) {
                used[j] = true;
            }
        }
    }
    
    // Sort for nicer output
    sort(ind_set.begin(), ind_set.end());
    
    cout << ind_set.size() << '\n';
    for (size_t i = 0; i < ind_set.size(); i++) {
        if (i > 0) cout << ' ';
        cout << ind_set[i];
    }
    cout << '\n';
    
    return 0;
}
