function isValid(x, y, grid) {
  const rows = grid.length;
  const cols = grid[0].length;
  return x >= 0 && x < rows && y >= 0 && y < cols && grid[x][y] !== '#';
}

function manhattan(x1, y1, x2, y2) {
  return Math.abs(x1 - x2) + Math.abs(y1 - y2);
}

// BFS to find shortest path distance between two points in the grid
function bfsDistance(startX, startY, goalX, goalY, grid) {
  const rows = grid.length;
  const cols = grid[0].length;
  if (!isValid(startX, startY, grid) || !isValid(goalX, goalY, grid)) {
    return Infinity;
  }
  if (startX === goalX && startY === goalY) {
    return 0;
  }

  const queue = [];
  const visited = new Array(rows).fill(0).map(() => new Array(cols).fill(false));
  const dist = new Array(rows).fill(0).map(() => new Array(cols).flatMap(() => Infinity));

  queue.push([startX, startY]);
  visited[startX][startY] = true;
  dist[startX][startY] = 0;

  const dirs = [[-1, 0], [0, 1], [0, 1], [1, 0]];  // up, right, down, left

  while (queue.length > 0) {
    const [x, y] = queue.shift();
    const d = dist[x][y];

    if (x === goalX && y === goalY) {
      return d;
    }

    for (let dir of dirs) {
      const nx = x + dir[0];
      const ny = y + dir[1];
      if (isValid(nx, ny, grid) && !visited[nx][ny]) {
        visited[nx][ny] = true;
        dist[nx][ny] = d + 1;
        queue.push([nx, ny]);
      }
    }
  }
  return Infinity;
}

// Greedy assignment: for each job, assign to closest available robot
// Note: To better match "deterministic greedy reference", we sort jobs by priority desc
// and then by id asc for stability.

function assignJobs(data) {
  const { grid, robots, jobs } = data;

  if (!robots || robots.length === 0 || !jobs || jobs.length === 0) {
    return { assignments: Array.from({ length: robots ? robots.length : 0 }, () => []) };
  }

  const numRobots = robots.length;
  const numJobs = jobs.length;

  // Create a copy of robots to keep track of their current positions
  let currentPositions = robots.map(r => ({ id: r.id, x: r.id, y: r.id })); // placeholder
  currentPositions = robots.map(r => ({ id: r.id, x: r.x, y: r.y }));

  // Sort jobs: higher priority first, then lower id (for determinism)
  const sortedJobs = [...jobs].sort((a, b) => {
    if (a.priority !== b.priority) {
      return b.priority - a.priority; // higher priority first
    }
    return a.id - b.id;
  });

  // Initialize assignments
  const assignments = Array.from({ length: numRobots }, () => []);

  // Track assigned jobs
  const jobAssigned = new Array(numJobs).fill(false);

  // Map from job.id to job object
  const jobMap = {};
  jobs.forEach(j => {
    jobMap[j.id] = j;
  });

  // Robot index map (assuming robot ids are 0-based indices)
  const robotIndexMap = {};
  robots.forEach((r, idx) => {
    robotIndexMap[r.id] = idx;
  });

  // For each job, assign to closest available robot (considering current positions)
  for (let i = 0; i < sortedJobs.length; i++) {
    const job = sortedJobs[i];
    let bestRobotIdx = -1;
    let bestDist = Infinity;

    for (let rIdx = 0; rIdx < numRobots; rIdx++) {
      const pos = currentPositions[rIdx];
      const dist = bfsDistance(pos.x, pos.y, job.x, job.y, grid);

      if (dist < bestDist) {
        bestDist = dist;
        bestRobotIdx = rIdx;
      } else if (dist === bestDist) {
        // Tie break: lowest robot id
        if (rIdx < bestRobotIdx) {
          bestRobotIdx = rIdx;
        }
      }
    }

    if (bestRobotIdx !== -1) {
      const rIdx = bestRobotIdx;
      assignments[rIdx].push(job.id);
      // Update current position of robot to the job location
      currentPositions[rIdx].x = job.x;
      currentPositions[rIdx].y = job.y;
    }
  }

  // Reorder assignments to match robot order (robot.id == index)
  // If robot ids are 0-based, we need to reorder
  const finalAssignments = Array.from({ length: numRobots }, () => []);

  // Since assignments[rIdx] is already using 0-based index
  // we need to sort each robot's assignment list if needed (for stability)

  return { assignments };
}

// ======================
// Main Execution
// ======================

function main() {
  const input = fs.readFileSync(0, 'utf8').trim();
  const data = JSON.parse(input);

  const result = assignJobs(data);

}
