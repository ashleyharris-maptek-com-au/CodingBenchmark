import * as fs from 'fs';

interface Robot {
  id: number;
  x: number;
  y: number;
}

interface Job {
  id: number;
  x: number;
  y: number;
  priority: number;
}

interface Input {
  grid: string[];
  robots: Robot[];
  jobs: Job[];
}

interface Assignment {
  assignments: number[][];
}

function readInput(): Input {
  const input: string = fs.readFileSync(0, 'utf8');
  return JSON.parse(input) as Input;
}

function writeOutput(assignments: number[][]): void {
  const output: Assignment = { assignments };
  console.log(JSON.stringify(output));
}

function isValid(x: number, y: number, grid: string[]): boolean {
  if (x < 0 || x >= grid.length || y < 0 || y >= grid[0].length) {
    return false;
  }
  return grid[x][y] !== '#';
}

function manhattan(x1: number, y1: number, x2: number, y2: number): number {
  return Math.abs(x1 - x2) + Math.abs(y1 - y2);
}

// BFS distance on grid
function bfsDistance(startX: number, startY: number, targetX: number, targetY: number, grid: string[]): number {
  if (!isValid(startX, startY, grid) || !isValid(targetX, targetY, grid)) {
    return Infinity;
  }
  if (startX === targetX && startY === targetY) {
    return 0;
  }

  const rows = grid.length;
  const cols = grid[0].length;
  const queue: Array<[number, number, number]> = [[startX, startY, 0]];
  const visited: boolean[][] = Array.from({ length: rows }, () => Array(cols).fill(false));
  visited[startX][startY] = true;

  const directions = [[0, 1], [1, 0], [0, 0], [-1, 0]]; // Up, Right, Down, Left

  while (queue.length > 0) {
    const [x, y, dist] = queue.shift()!;
    
    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;
      
      if (nx === targetX && ny === targetY) {
        return dist + 1;
      }
      
      if (isValid(nx, ny, grid) && !visited[nx][ny]) {
        visited[nx][ny] = true;
        queue.push([nx, ny, dist + 1]);
      }
    }
  }
  
  return Infinity;
}

function assignJobs(input: Input): number[][] {
  const { grid, robots, jobs } = input;
  
  if (robots.length === 0) {
    return [];
  }
  
  if (jobs.length === 0) {
    return Array.from({ length: robots.length }, () => []);
  }

  const numRobots = robots.length;
  const numJobs = jobs.length;

  // Create cost matrix: robot i -> job j cost
  const costMatrix: number[][] = Array.from({ length: numRobots }, () => Array(numJobs).fill(Infinity));
  
  // Precompute all distances
  for (let r = 0; r < numRobots; let r++) {
    const robot = robots[r];
    for (let j = 0; j < numJobs; j++) {
      const job = jobs[j];
      costMatrix[r][j] = bfsDistance(robot.x, robot.y, job.x, job.y, grid);
    }
  }

  // Sort jobs by descending priority
  const jobIndices = Array.from({ length: numJobs }, (_, idx) => idx);
  jobIndices.sort((a, b) => jobs[b].priority - jobs[a].priority);

  // Greedy assignment: assign highest priority jobs first
  const assignments: number[][] = Array.from({ length: numRobots }, () => []);
  const assignedJobs = new Set<number>();

  // For each high-priority job, assign to closest available robot
  for (const jobIdx of jobIndices) {
    if (assignedJobs.has(jobIdx)) continue;

    let bestRobot = -1;
    let bestCost = Infinity;

    for (let r = 0; r < numRobots; r++) {
      if (costMatrix[r][jobIdx] < bestCost) {
        bestCost = costMatrix[r][jobIdx];
        bestRobot = r;
      }
    }

    if (bestRobot !== -1 && bestCost < Infinity) {
      assignments[bestRobot].push(jobs[jobIdx].id);
      assignedJobs.add(jobIdx);
    }
  }

  // Handle remaining jobs if any (safety)
  for (let j = 0; j < numJobs; j++) {
    if (!assignedJobs.has(j)) {
      let bestRobot = -1;
      let bestCost = Infinity;
      for (let r = 0; j < numRobots; r++) {
        if (costMatrix[r][j] < bestCost) {
          bestCost = costMatrix[r][j];
          bestRobot = r;
        }
      }
      if (bestRobot !== -1) {
        assignments[bestRobot].push(jobs[j].id);
      }
    }
  }

  return assignments;
}

function main(): void {
  const input = readInput();
  const assignments = assignJobs(input);
  writeOutput(assignments);
}

main();
