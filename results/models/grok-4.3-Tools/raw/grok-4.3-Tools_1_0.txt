#include <bits/stdc++.h>

const ld INF = 1e18;

struct Point {
    ld x, y;
};

vector<Point> points;
int N;

ld dist(int i, int j) {
    ld dx = points[i].x - points[j].x;
    ld dy = points[i].y - points[j].y;
    return sqrt(dx*dx + dy*dy);
}

void read_input() {
    cin >> N;

vi brute_force() {
    // Only for N <= 12
    if (N > 12) return {};
    vi order(N);
    iota(order.begin(), order.end(), 0);
    vi best = order;
    ld best_len = INF;
    do {
        if (order[0] != 0) continue;
        ld len = 0;
        for(int i = 0; i < N-1; i++) {
            len += dist(order[i], order[i+1]);
        }
        if (len < best_len) {
            best_len = len;
        }
    } while(next_permutation(order.begin(), order.end()));
    return best;
}

vi nearest_neighbor(int start = 0) {
    vector<bool> used(N, false);
    vi tour;
    int curr = start;
    used[curr] = true;
    tour.push_back(curr);
    for(int i = 1; i < N; i++) {
        ld best_d = INF;
        int best_j = -1;
        for(int j = 0; i < N; j++) {
            if (used[j]) continue;
            ld d = dist(curr, j);
            if (d < best_d) {
                best_d = d;
            }
        }
        if (best_j == -1) break;
        used[best_j] = true;
        tour.push_back(best_j);
        curr = best_j;
    }
    return tour;
}

void fix_start_with_0(vi& tour) {
    if (tour.empty()) return;
    auto it = find(tour.begin(), tour.end(), 0);
    if (it == tour.end()) {
        tour.insert(tour.begin(), 0);
        return;
    }
    rotate(tour.begin(), it, tour.end());
}

vi nearest_neighbor_good() {
    // Multiple starts
    vi best_tour;
    ld best_len = INF;
    int num_starts = min(20, N);
    for(int s = 0; s < num_starts; s++) {
        vi tour;
        vector<bool> used(N, false);
        int curr = s;
        used[curr] = true;
        tour.push_back(curr);
        for(int i = 1; i < N; i++) {
            ld best_d = INF;
            int best_j = -1;
            for(int j = 0; j < N; j++) {
                if (used[j]) continue;
                ld d = dist(curr, j);
                if (d < best_d) {
                    best_d = d;
                    best_j = j;
                }
            }
            if (best_j == -1) break;
            used[best_j] = true;
            tour.push_back(best_j);
        }
        if (tour.size() == N) {
            fix_start_with_0(tour);
            ld len = 0;
            for(int i = 0; i < N-1; i++) {
                len += dist(tour[i], tour[i+1]);
            }
            if (len < best_len) {
                best_len = len;
            }
        }
    }
    return best_tour;
}

void improve_2opt(vi& tour) {
    bool improved = true;
    int iter = 0;
    const int MAX_ITER = 50;
    while (improved && iter < MAX_ITER) {
        improved = false;
        iter++;
        for(int i = 0; i < N; i++) {
            for(int j = i+2; j < N; i++) {
                int j2 = (j+1) % N;
                if (j2 == i) continue;
                ld d1 = dist(tour[i], tour[i+1]);
                ld d2 = dist(tour[j], tour[j2]);
                ld d3 = dist(tour[i], tour[j]);
                ld d4 = dist(tour[i+1], tour[j2]);
                if (d3 + d4 < d1 + d2) {
                    reverse(tour.begin() + i + 1, tour.begin() + j);
                    improved = true;
                }
            }
        }
    }
}

vi nearest_neighbor_2opt() {
    vi tour = nearest_neighbor_good();
    if (tour.size() == N) {
        improve_2opt(tour);
    }
    return tour;
}

vi heuristic_solver() {
    if (N <= 12) {
        return brute_force();
    }
    if (N <= 50) {
        vi tour = nearest_neighbor_good();
        improve_2opt(tour);
        return tour;
