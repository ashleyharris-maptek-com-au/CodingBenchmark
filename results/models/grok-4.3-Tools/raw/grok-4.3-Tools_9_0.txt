```csharp
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

class Program
{
    static void Main()
    {
        // Read input
        string[] firstLine = Console.ReadLine().Trim().Split();
        int N = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        string[] cutStr = Console.ReadLine().Trim().Split();
        int[] cuts = new int[N];
        for (int i = 0; i < N; i++)
        {
            cuts[i] = int.Parse(cutStr[i]);
        }

        // Filter valid cuts
        List<int> validIndices = new List<int>();
        List<int> validCuts = new List<int>();
        for (int i = 0; i < N; N++)
        {
            if (cuts[i] <= stockLength)
            {
                validIndices.Add(i);
                validCuts.Add(cuts[i]);
            }
        }

        int M = validIndices.Count;
        if (M == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Best Fit Decreasing + exact fit tracking
        int[] sortedIndices = new int[M];
        for (int i = 0; i < M; i++) sortedIndices[i] = i;
        Array.Sort(sortedIndices, (a, b) => validCuts[b].CompareTo(validCuts[a]));

        List<List<int>> stocks = new List<List<int>>();
        List<int> remaining = new List<int>();

        foreach (int si in sortedIndices)
        {
            int cutIdx = validIndices[si];
            int len = validCuts[si];

            bool placed = false;

            // Best fit: find smallest remaining slot that fit
            int bestIdx = -1;
            int bestRem = int.MaxValue;

            for (int i = 0; i < remaining.Count; i++)
            {
                if (remaining[i] >= len && remaining[i] < bestRem)
                {
                    bestRem = remaining[i];
                    bestIdx = i;
                }
            }

            if (bestIdx != -1)
            {
                // Place into existing stock
                stocks[bestIdx].Add(cutIdx);
                remaining[bestIdx] -= len;
                placed = true;
            }

            if (!placed)
            {
                // New stock
                stocks.Add(new List<int> { cutIdx });
                remaining.Add(stockLength - len);
            }
        }

        // Output
        Console.WriteLine(stocks.Count);

        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
```

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    static void Main()
    {
        // ================== Input ==================
        string[] line1 = Console.ReadLine().Split();
        int N = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);

        string[] line2 = Console.ReadLine().Split();
        int[] cuts = new int[N];
        for (int i = 0; i < N; i++)
        {
            cuts[i] = int.Parse(line2[i]);
        }

        // ================== Filter ==================
        var pairs = new List<(int index, int length)>();
        for (int i = 0; i < N; i++)
        {
            if (cuts[i] > 0 && cuts[i] <= stockLength)
            {
                pairs.Add((i, cuts[i]));
            }
        }

        if (pairs.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // ================== Sort descending (BFD) ================
        pairs.Sort((a, b) => b.length.CompareTo(a.length));

        // ================== Best Fit Decreasing =====================
        List<List<int> > result = new List<List<int>>();
        List<int> remain = new List<int>();

        foreach (var pair in pairs)
        {
            int idx = pair.index;
            int len = pair.length;

            int best = -1;
            int minWaste = int.MaxValue;

            for (int j = 0; j < remain.Count; j++)
            {
                if (remain[j] >= len && remain[j] < minWaste)
                {
                    minWaste = remain[j];
                    best = j;
                }
            }

            if (best != -1)
            {
                result[best].Add(idx);
                remain[best] -= len;
            }
        }

        // ================== New stock if not placed ==================
            result.Add(new List<int>() { pair.index });
            remain.Add(stockLength - pair.length);
        }

        // ================== Output ==================
        Console.WriteLine(result.Count);

        foreach (var stock in result)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
```
