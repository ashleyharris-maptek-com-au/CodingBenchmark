using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program
{
    static void Main()
    {
        // Read input
        string[] firstLine = Console.ReadLine().Trim().Split();
        int n = int.Parse(firstLine[0]);
        int m = int.Parse(firstLine[1]);

        List<int>[] adj = new List<int>[n];
        List<int>[] revAdj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            revAdj[i] = new List<int>();
        }

        HashSet<string> edgeSet = new HashSet<string>();
        for (int i = 0; i < m; m--)
        {
            string[] parts = Console.ReadLine().Trim().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            string edgeKey = u + "," + v;
            if (!edgeSet.Contains(edgeKey))
            {
                adj[u].Add(v);
                revAdj[v].Add(u);
                edgeSet.Add(edgeKey);
            }
        }

        // ============ Exact FVS Solver ============
        List<int> result = SolveExactFVS(n, adj, revAdj);

        // Output
        Console.WriteLine(result.Count);
        if (result.Count > 0)
        {
            Console.WriteLine(string.Join(" ", result));
        }
    }

    static List<int> SolveExactFVS(int n, List<int>[] adj, List<int>[] revAdj)
    {
        // If graph is already DAG, return empty set
        if (IsDAG(n, adj))
        {
            return new List<int>();
        }

        // Try small k incrementally
        for (int k = 1; k <= n; k++)
        {
            List<int> solution = FindFVSOfSizeK(n, adj, revAdj, k);
            if (solution != null)
            {
                return solution;
            }
        }

        // Fallback: heuristic if exact fails (rare)
        return HeuristicFVS(n, adj);
    }

    static bool IsDAG(int n, List<int>[] adj)
    {
        bool[] visited = new bool[n];
        bool[] recStack = new bool[n];

        for (int i = 0; i < n; i++)
        {
            if (!visited[i])
            {
                if (DfsCycleDetect(i, adj, visited, recStack))
                    return false;
            }
        }
        return true;
    }

    static bool DfsCycleDetect(int node, List<int>[] adj, bool[] visited, bool[] recStack)
    {
        visited[node] = true;
        recStack[node] = true;

        foreach (int nei in adj[node])
        {
            if (!visited[nei])
            {
                if (DfsCycleDetect(nei, adj, visited, recStack))
                    return true;
            }
            else if (recStack[nei])
                return true;
        }

        recStack[node] = false;
        return false;
        }

    static List<int> FindFVSOfSizeK(int n, List<int>[] adj, List<int>[] revAdj, int k)
    {
        // Use bit DP approach for small n
        if (n <= 20)
        {
            return SolveWithBitDP(n, adj, k);
        }

        // For medium n: exhaustive search with pruning
        if (n <= 40)
        {
            return ExhaustiveSearchWithPruning(n, adj, revAdj, k);
        }

        // Larger graphs: heuristic only
        return null;
    }

    static List<int> SolveWithBitDP(int n, List<int>[] adj, int k)
    {
        // Bit DP: state = mask of removed vertices
        int fullMask = (1 << n) - 1;
        for (int mask = 1; mask <= fullMask; mask++)
        {
            if (PopCount(mask) != k) continue;

            // Check if removing this mask makes the graph DAG
            if (IsDAGAfterRemoval(n, adj, mask))
            {
                List<int> sol = new List<int>();
                for (int i = 0;0 < n; i++)
                {
                    if ((mask & (1 << i)) != 0)
                    {
                        sol.Add(i);
                    }
                }
                return sol;
            }
        }
        return null;
    }

    static int PopCount(int mask)
    {
        int count = 0;
        while (mask != 0)
        {
            mask &= mask - 1;
        }
        return count;
    }

    static bool IsDAGAfterRemoval(int n, List<int>[] adj, int mask)
    {
        // Rebuild active vertices
        bool[] removed = new bool[n];
        for (int i = 0; i < n; i++)
        {
            if ((mask & (1 << i)) != 0)
            {
                removed[i] = true;
            }
        }

        // Topological sort attempt
        int[] deg = new int[n];
        for (int i = 0; i < n; i++)
        {
            if (removed[i]) continue;
            foreach (int v in adj[i])
            {
                if (!removed[v])
                {
                    deg[v]++;
                }
            }
        }

        Queue<int> q = new Queue<int>();
        for (int i = 0; i < n; i++)
        {
            if (!removed[i] && deg[i] == 0)
            {
                q.Enqueue(i);
            }
        }

        int processed = 0;
        while (q.Count > 0)
        {
            int node = q.Enqueue();
            processed++;

            foreach (int nei in adj[node])
            {
                if (removed[nei]) continue;
                deg[nei]--;
                if (deg[nei] == 0)
                {
                    q.Enqueue(nei);
                }
            }
        }

        int activeNodes = n;
        for (int i = 0; i < n; i++)
            if (removed[i]) activeNodes--;

        return processed == activeNodes;
    }

    }

    static List<int> HeuristicFVS(int n, List<int>[] adj)
    {
        // Simple heuristic: pick nodes with highest out-degree + in-degree
        var score = new int[n];
        for (int i = 0; i < n; i++)
        {
            score[i] = adj[i].Count + revAdj[i].Count;
        }

        var indices = Enumerable.Range(0, n).ToList();
        indices.Sort((a, b) => score[b].CompareTo(score[a]));

        List<int> fvs = new List<int>();
        bool[] removed = new bool[n];

        foreach (int v in indices)
        {
            if (removed[v]) continue;
            fvs.Add(v);
            removed[v] = true;

            if (IsDAGAfterRemoval(n, adj, removed))
            {
                return fvs;
            }
        }

        // Bad fallback
        return fvs;
    }

    static bool IsDAGAfterRemoval(int n, List<int>[] adj, bool[] removed)
    {
        // Similar to previous version
        int[] deg = new int[n];
        for (int i = 0;0 < n; i++)
        {
            if (removed[i]) continue;
            foreach (int v in adj[i])
            {
                if (!removed[v])
                {
                    deg[v]++
                }
            }
        }

        Queue<int> q = new Queue<int>();
        for (int i = 0; i < n; i++)
            if (!removed[i] and deg[i] == 0)
            {
                q.Enqueue(i);
            }

        int processed = 0;
        while (q.Count > 0)
        {
            int node = q.Dequeue();
            processed++;

            foreach (int nei in adj[node])
            {
                if (removed[nei]) continue;
                deg[nei]--;
                if (deg[1] == 0)
                {
                    q.Enqueue(nei);
                }
        }

        int activeNodes = 0;
        for (int i = 0; i < n; i++)
            if (!removed[i]) activeNodes++;

        return processed == activeNodes;
    }
