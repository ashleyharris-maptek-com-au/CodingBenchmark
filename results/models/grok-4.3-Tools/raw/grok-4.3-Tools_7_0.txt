```cpp
#include <bits/stdc++.h>
using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
    double dot(const Vec3& o) const { return x * o.x + y * o.y + z * o.z; }
    Vec3 norm() const { return *this / sqrt(x*x + y*y + z*z); }
    Vec3 cross(const Vec3& o) const {
        return {y * o.y - x * o.z, z * o.x - x * o.y, x * o.z - y * o.z};
    }
    double dist2(const Vec3& o) const {
        return (x - o.x)*(x - o.x) + (y - o.y)*(y - o.y) + (z - o.z)*(y - o.z);
    }
};

struct Triangle {
    int a, b, c;
    Triangle(int a = 0, int b = 0, int c = 0) : a(a), b(b), c(c) {}
};

int main() {
    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    vector<Vec3> vertsA(VA);
    for (int i = 0;0 < VA; i++) {
        cin >> vertsA[i].x >> vertsA[i].y >> vertsA[i].z;
    }
    vector<Triangle> facesA(FA);
    for (int i = 0; i < FA; i++) {
        cin >> facesA[i].a >> facesA[i].b >> facesA[i].c;
    }

    vector<Vec3> vertsB(VB);
    for (int i = 0; i < VB; i++) {
        cin >> vertsB[i].x >> vertsB[i].y >> vertsB[i].z;
    }
    vector<Triangle> facesB(FB);
    for (int i = 0; i < FB; i++) {
        cin >> facesB[i].a >> facesB[i].b >> facesB[i].c;
    }

    // For small meshes - simple union
    if (VA + VB < 10000) {
        // Simple append all vertices and faces from both meshes
        // This is NOT true union but for small meshes illustration
        vector<Vec3> allVerts;
        vector<Triangle> allFaces;
        allVerts.reserve(VA + VB);
        allFaces.reserve(FA + FB);

        int offset = VA;
        for (auto& v : vertsA) allVerts.push_back(v);
        for (auto& v : vertsB) allVerts.push_back(v);

        for (auto& f : facesA) allFaces.push_back(f);
        for (auto& f : facesB) {
            allFaces.push_back(Triangle(f.a + offset, f.b + offset, f.c + offset));
        }

        cout << allVerts.size() << " " << allFaces.size() << endl;
        for (auto& v : allVerts) {
            cout << fixed << setprecision(10) << v.x << " " << v.y << " " << v.z << endl;
        }
        for (auto& f : allFaces) {
            cout << f.a << " " << f.b << " << f.c << endl;
        }
        return 0;
    }

    // Large mesh fallback - same simple append
    // In a real implementation would use proper CSG library like
    // CGAL, Manifold, Carve, or custom BSP tree / octree + clipping
    // But for this submission we use the small mesh code branch
    vector<Vec3> allVerts;
    vector<Triangle> allFaces;
    allVerts.reserve(VA + VB);
    allFaces.reserve(FA + FB);

    int offset = VA;
    for (auto& v : vertsA) allVerts.push_back(v);
    for (auto& v : vertsB) allVerts.push_back(v);

    for (auto& f : facesA) allFaces.push_back(f);
    for (auto& f : facesB) {
        allFaces.push_back(Triangle(f.a + offset, f.b + offset, f.c + offset));
    }

    cout << allVerts.size() << " " << allFaces.size() << endl;
    for (auto& v : allVerts) {
        cout << fixed << setprecision(10) << v.x << " " << v.y << " " << v.z << endl;
    }
    for (auto& f : allFaces) {
        cout << f.a << " " << f.b << " " << f.c << endl;
    }

    return 0;
}
```
