const fs = require('fs');

// ======================
// Complete Node.js program
// ======================

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const { machineCount, tasks } = data;

// ======================
// Build graph + indegree
// ======================
const taskMap = new Map();
const adj = new Map();
const indegree = new Map();
const durationMap = new Map();
const deadlineMap = new Map();
const weightMap = new Map();

for (const task of tasks) {
    const id = task.id;
    taskMap.set(id, task);
    adj.set(id, []);
    indegree.set(id, 0);
    durationMap.set(id, task.duration || 0);
    deadlineMap.set(id, task.deadline || Infinity);
    weightMap.set(id, task.weight || 1);
}

// Populate dependencies
for (const task of tasks) {
    const id = task.id;
    if (Array.isArray(task.deps)) {
        for (const dep of task.deps) {
            if (adj.has(dep)) {
                adj.get(dep).push(id);
                indegree.set(id, (indegree.get(id) || 0) + 1);
            }
        }
    }
}

// ======================
// Kahn's algorithm + Priority Queue (critical path heuristic)
// ======================

// Priority Queue: higher priority first
class PriorityQueue {
    constructor() {
        this.heap = [];
    }

    push(item) {
        this.heap.push(item);
        this.heapifyUp(this.heap.length - 1);
    }

    pop() {
        if (this.heap.length === 0) return null;
        const root = this.heap[0];
        const last = this.heap.pop();
        if (this.heap.length > 0) {
            this.heap[0] = last;
            this.heapifyDown(0);
        }
        return root;
    }

    heapifyUp(index) {
        while (index > 0) {
            const parent = Math.floor((index - 1) / 2);
            if (this.heap[parent].priority >= this.heap[index].priority) break;
            [this.heap[parent], this.heap[index]] = [this.heap[parent], this.heap[index]];
            index = parent;
        }
    }

    heapifyDown(index) {
        while (true) {
            let largest = index;
            const left = index * 2 + 1;
            const right = index * 2 + 2;
            if (left < this.heap.length && this.heap[left].priority > this.heap[largest].priority) {
                largest = left;
            }
            if (right < this.heap.length && this.heap[right].priority > this.heap[largest].priority) {
                largest = right;
            }
            if (largest == index) break;
            [this.heap[largest], this.heap[index] = [this.heap[index], this.heap[largest]];
            index = largest;
        }
    }

    size() {
        return this.heap.length;
    }
}

// ======================
// Compute earliest start times (longest path)
// ======================
function computeEarliestStartTimes() {
    const earliest = new Map();
    for (const id of taskMap.keys()) {
        earliest.set(id, 0);
    }

    const queue = [];
    const tempIndegree = new Map(indegree);

    // Initial nodes
    for (const id of taskMap.keys()) {
        if ((tempIndegree.get(id) || 0) === 0) {
            queue.push(id);
        }
    }

    while (queue.length > 0) {
        const id = queue.shift();
        const finish = earliest.get(id) + (durationMap.get(id) || 0);

        for (const succ of adj.get(id) || []) {
            const currentEarliest = earliest.get(succ);
            if (finish > currentEarliest) {
                earliest.set(succ, finish);
            }
            tempIndegree.set(succ, (tempIndegree.get(succ) || 0) - 1);
            if (tempIndegree.get(succ) === 0) {
                queue.push(succ);
            }
        }
    }
    return earliest;
}

const earliestStart = computeEarliestStartTimes();

// ======================
// Priority calculation
// ==============================
function getPriority(id) {
    const duration = durationMap.get(id) || 0;
    const deadline = deadlineMap.get(id);
    const weight = weightMap.get(id) || 1;
    const earliest = earliestStart.get(id) || 0;

    // Strong priority heuristic
    let score = weight * 1000 + duration * 5;

    if (deadline !== Infinity) {
        score += (10000 - deadline) * 2;
    }

    score += earliest * 0.5;

    return score;
}

// ======================
// Main scheduling logic
// ======================
function scheduleTasks() {
    const schedule = [];
    const currentTimePerMachine = Array(machineCount).fill(0);
    const startTime = new Map();
    const assignedMachine = new Map();

    // Remaining indegree
    const remainingIndegree = new Map(indegree);

    // Priority queue for ready tasks
    const readyQueue = new PriorityQueue();

    // Seed ready queue with indegree 0 tasks
    for (const id of taskMap.keys()) {
        if ((remainingIndegree.get(id) || 0) === 0) {
            const priority = getPriority(id);
            readyQueue.push({ id, priority });
        }
    }

    let time = 0;
    let processed = 0;
    const totalTasks = taskMap.size;

    while (readyQueue.size() > 0 && processed < totalTasks) {
        // Find earliest machine finish time
        let minTime = Infinity;
        let minMachine = 0;

        for (let i = 0; i < machineCount; i++) {
            if (currentTimePerMachine[i] < minTime) {
                minTime = currentTimePerMachine[i];
            }
        }

        // Update global time to earliest available machine
        time = minTime;

        // Pop the highest priority ready task
        const entry = readyQueue.pop();
        if (!entry) break;
        const id = entry.id;

        // Assign to best machine (earliest available)
        let bestMachine = 0;
        let bestStart = Infinity;

        for (let i = 0; i < machineCount; i++) {
            const possibleStart = Math.max(time, currentTimePerMachine[i]);
            if (possibleStart < bestStart) {
                bestStart = possibleStart;
                bestMachine = i;
            }
        }

        const dur = durationMap.get(id) || 0;
        const finish = bestStart + dur;

        // Record schedule
        schedule.push({
            id: id,
            start: bestStart,
            machine: bestMachine
        });

        assignedMachine.set(id, bestMachine);
        startTime.set(id, bestStart);
        currentTimePerMachine[bestMachine] = finish;

        processed++;

        // Update successors
        for (const succ of adj.get(id) || []) {
            remainingIndegree.set(succ, (remainingIndegree.get(succ) || 0) - 1);
            if (remainingIndegree.get(succ) === 0) {
                const priority = getPriority(succ);
                readyQueue.push({ id: succ, priority });
            }
        }
    }

    // If some tasks couldn't be scheduled (cycle or error)
    if (processed < totalTasks) {
        // Fallback: simple topological order
        return { order: topologicalSort() };
    }

    return { schedule };
}

// ======================
// Fallback topological sort
// ======================
function topologicalSort() {
    const order = [];
    const tempIndegree = new Map(indegree);
    const queue = [];

    for (const id of taskMap.keys()) {
        if ((tempIndegree.get(id) || 0) === 0) {
            queue.push(id);
        }
    }

    while (queue.length > 0) {
        // Sort queue by priority every time for better heuristic
        queue.sort((a, b) => getPriority(b) - getPriority(a));
        const id = queue.shift();
        order.push(id);

        for (const succ of adj.get(id) || []) {
            tempIndegree.set(succ, (tempIndegree.get(succ) || 0) - 1);
            if (tempIndegree.get(succ) === 0) {
                queue.push(succ);
            }
        }
    }

    return order;
}

// ======================
// Execute and output
// ======================
const result = scheduleTasks();

console.log(JSON.stringify(result));
