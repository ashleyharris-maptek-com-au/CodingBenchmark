```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x = dt, radii.y = sepRadius, radii.z = alignRadius, radii.w = cohRadius
    vec4 weights;      // weights.x = sepWeight, weights.y = alignWeight, weights.z = cohWeight, weights.w = maxSpeed
    vec4 bounds;       // bounds.x = boundSize, bounds.y = boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) {
        return;
    }

    // Read current boid
    vec3 pos = inData[i * 2 + 0].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    // Unpack parameters
    float dt             = radii.x;
    float sepRadius      = radii.y;
    float alignRadius    = radii.z;
    float cohRadius      = radii.w;

    float sepWeight      = weights.x;
    float alignWeight    = weights.y;
    float cohWeight      = weights.z;
    float maxSpeed       = weights.w;

    float boundSize      = bounds.x;
    float boundForce     = bounds.y;

    float boundThreshold = boundSize * 0.4;
    float boundCenter    = boundSize * 0.5;

    // Accumulated forces
    vec3 sepForce   = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce   = vec3(0.0);

    int sepCount    = 0;
    int alignCount  = 0;
    int cohCount    = 0;

    // Loop over all boids
    for (uint j = 0; j < N; j++) {
        if (i == j) continue;

        vec3 jPos = inData[j * 2 + 0].xyz;
        vec3 diff = pos - jPos;
        float distSq = dot(diff, diff);

        if (distSq < 0.0001) continue;  // skip zero-distance

        float dist = sqrt(distSq);

        // Separation
        if (dist < sepRadius) {
            sepForce += diff / dist;  // direction away
            sepCount++;
        }

        // Alignment & Cohesion
        if (dist < alignRadius) {
            vec3 jVel = inData[j * 2 + 1].xyz;
            alignForce += jVel;
            alignCount++;
        }

        if (dist < cohRadius) {
            cohForce += jPos;
            cohCount++;
        }
    }

    // Average the forces
    vec3 accel = vec3(0.0);

    // Separation
    if (sepCount > 0) {
        sepForce = sepForce / float(sepCount);
        accel += sepForce * sepWeight;
    }

    // Alignment
    if (alignCount > 0) {
        alignForce = alignForce / float(alignCount);
        accel += alignForce * alignWeight;
    }

    // Cohesion
    if (cohCount > 0) {
        cohForce = cohForce / float(cohCount);
        cohForce = normalize(cohForce - pos);  // direction to center of mass
        accel += cohForce * cohWeight;
    }

    // Boundary avoidance
    vec3 boundaryAccel = vec3(0.0);
    if (abs(pos.x) > boundThreshold) {
        boundaryAccel.x = sign(pos.x) * -boundForce * (boundSize * 0.5 - abs(pos.x)) / boundSize;
    }
    if (abs(pos.y) > boundThreshold) {
        boundaryAccel.y = sign(pos.y) * -boundForce * (boundSize * 0.5 - abs(pos.x)) / boundSize;
    }
    if (abs(pos.z) > boundThreshold) {
        boundaryAccel.z = sign(pos.z) * -boundForce * (boundSize * 0.5 - abs(pos.z)) / boundSize;
    }

    accel += boundaryAccel;

    // Update velocity
    vel += accel * dt;

    // Clamp speed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = vel / speed * maxSpeed;
    }

    // Update position
    pos += vel * dt;

    // Write back
    outData[i * 2 + 0] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
