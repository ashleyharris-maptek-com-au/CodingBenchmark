```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const prob = data.probability;
const drones = data.drones;
const maxStepsPerDrone = data.maxStepsPerDrone;

const rows = prob.length;
const cols = prob[0].length;

// Directions: up, down, left, right, stay
const dirs = [
  [-1, 0], // up
  [1, 0],  // down
  [0, -1], // left
  [0, 1],  // right
  [0, 0]   // stay
];

// Helper function to check if move is valid
function isValid(x, y) {
  return x >= 0 && x < rows && y >= 0 && y < cols && prob[x][y] !== -1;
}

// Create visited coverage map
const covered = Array.from({ length: rows }, () => Array(cols).fill(false));

// Mark obstacles as covered (they can't be covered anyway)
for (let i = 0; i < rows; i++) {
  for (let j = 0; j < cols; j++) {
    if (prob[i][j] === -1) {
      covered[i][j] = true;
    }
  }
}

// Track current positions
const currentPositions = drones.map(d => ({ x: d.x, y: d.y }));

// Paths for each drone
const paths = [];
for (let i = 0; i < drones.length; i++) {
  paths.push([]); // Each path starts empty initially
}

// For each drone individually
for (let d = 0; d < drones.length; d++) {
  const dronePath = [];
  let posX = currentPositions[d].x;
  let posY = currentPositions[d].y;
  
  // Record starting position (step 0)
  dronePath.push([posX, posY]);
  
  // Mark starting position as covered
  if (!covered[posX][posY]) {
    covered[posX][posY] = true;
  }
  
  // Generate greedy best-first path
  for (let step = 0; step < maxStepsPerDrone; step++) {
    let bestScore = -1;
    let bestDir = null;
    let bestX = posX;
    let bestY = posY;
    
    // Look ahead one step for best move
    for (let dirIdx = 0; dirIdx < dirs.length; dirIdx++) {
      const [dx, dy] = dirs[dirIdx];
      const nx = posX + dx;
      const ny = posY + dy;
      
      if (!isValid(nx, ny)) continue;
      
      let score = 0;
      if (!covered[nx][ny]) {
        score = prob[nx][ny] || 0;
      }
      
      if (score > bestScore) {
        bestScore = score;
        bestDir = [dx, dy];
        bestX = nx;
        bestY = ny;
      }
    }
    
    // If no better move found, stay
    if (bestDir === null) {
      bestX = posX;
      bestY = posY;
    }
    
    // Perform move
    posX = bestX;
    posY = bestY;
    
    // Record move
    dronePath.push([posX, posY]);
    
    // Mark as covered
    if (!covered[posX][posY]) {
      covered[posX][posY] = true;
    }
  }
  
  paths[d] = dronePath;
}

const output = { paths };
console.log(JSON.stringify(output));
```
