```hlsl
[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize normal
    float3 N = normalize(input.normal);
    
    // Compute light direction
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Lambertian diffuse term
    float diff = max(dot(N, L), 0.0);
    
    // UV checkerboard at 8x8 frequency
    float2 scaledUV = input.uv * 8.0;
    int checkX = int(scaledUV.x);
    int checkY = int(scaledUV.y);
    float check = ((checkX + checkY) % 2 == 0) ? 0.9 : 0.2;
    
    // Base color from checkerboard
    float3 baseColor = check.xxx;
    
    // Lambertian shading factor
    float shade = 0.5 + 0.5 * diff;
    
    // Final color with shading
    float3 finalColor = baseColor * shade;
    
    return float4(finalColor, 1.0);
}
```
