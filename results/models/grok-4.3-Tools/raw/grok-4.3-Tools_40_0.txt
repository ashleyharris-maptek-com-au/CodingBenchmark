#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<long long> c(n);
    for (int i = 0; i < n; i++) {
        cin >> c[i];
    }

    // Read constraints: adjacency list for the conflict graph
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Maximum Independent Set via DP on connected components
    // Since n can be small (implicitly), but we use bit DP if n <= 20

    long long best_value = 0;
    vector<int> best_x(n, 0);

    // First, we assume n is small enough for exact solution (common in such problems)
    // If n > 20, we still can use DP on tree / components, but for now we implement bitmask DP for safety
    // But to be general, we detect connected components and solve MIS on each

    // Build undirected graph
    vector<vector<int>> graph(n);
    vector<bool> visited(n, false);
    for (int i = 0; i < n; i++) {
        for (int j : adj[i]) {
            if (find(graph[i].begin(), graph[i].end(), j) == graph[i].end()) {
                graph[i].push_back(j);
                graph[j].push_back(i);  // already bidirectional
            }
        }
    }

    // Remove duplicates and sort
    for (int i = 0; i < n; i++) {
        sort(graph[i].begin(), graph[i].end());
        auto it = unique(graph[i].begin(), graph[i].dbegin(), graph[i].end());
        graph[i].resize(it - graph[i].begin());
    }

    // Actually simpler approach: since constraints are only pairwise exclusions, it's exactly Maximum Weight Independent Set on conflict graph

    // For correctness and generality, since n is likely small (1 < n < 25 or so) , use bit DP on the helpline

    // But to handle larger n (up to 40?), we can use Bron-Kerbosch or branch and bound, but simplest for correct exact solution is:

    // We will use dynamic programming over subsets for small n, but to make it reliable, we use a recursive branch-and-bound with pruning

    // Let's implement a recursive backtracking with pruning that finds the maximum weight independent set

    long long current = 0;
    vector<int> x(n, 0);
    vector<bool> used(n, false);

    function<void(int, long long)> dfs = [&](int idx, long long val) {
        if (idx == n) {
            if (val > best_value) {
                best_value = val;
                best_x = x;
            }
            return;
        }

        // Prune if impossible to beat best
        if (val + accumulate(c.begin() + idx, c.end(), 0LL) <= best_value) {
            return;
        }

        // Option 1: skip current
        dfs(idx + 1, val);

        // Option 2: take current if possible
        bool can_take = true;
        for (int nei : adj[idx]) {
            if (nei < idx && used[nei]) {
                can_take = false;
                break;
            }
        }
        if (can_take) {
            x[idx] = 1;
            used[idx] = true;
            dfs(idx + 1, val + c[idx]);
            used[idx] = false;
            x[idx] = 0;
        }
    };

    // Initialize best to 0
    best_value = 0;
    for (int i = 0; i < n; all() {
        best_value += c[i];
    }

    dfs(0, 0);

    // Output
    cout << best_value << '\n';
    for (int i = 0; i < n; i++) {
        cout << best_x[i];
        if (i + 1 != n) cout << " ";
    }
    cout << '\n';

    return 0;
}
