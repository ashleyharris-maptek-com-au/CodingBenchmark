```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-range invocations
    if (i >= N) {
        return;
    }
    
    // Read current state of body i
    vec4 posMassI = inData[i * 2];
    vec4 velI     = inData[i * 2 + 1];
    
    vec3 posI = posMassI.xyz;
    float massI = posMassI.w;
    vec3 vel  = velI.xyz;
    
    // Extract physics parameters
    float dt         = physics.x;
    float softening  = physics.y;
    float G          = physics.z;
    
    // Compute acceleration for body i
    vec3 acc = vec3(0.0);
    
    for (uint j = 0; j < N; j++) {
        if (i == j) continue;
        
        vec4 posMassJ = inData[j * 2];
        vec3 posJ = posMassJ.xyz;
        float massJ = posMassJ.w;
        
        vec3 diff = posJ - posI;
        float dist2 = dot(diff, diff);
        float denom = dist2 + softening * softening;
        
        // (r^2 + ε²)^(3/2)
        float invDist3 = inversesqrt(denom * denom * denom);  // more efficient than pow()
        
        // Alternative: 
        // float invDist3 = inversesqrt(denom) * inversesqrt(denom) * inversesqrt(denom);
        
        acc += G * massJ * diff * invDist3;
    }
    
    // Leapfrog / semi-implicit Euler integration
    vec3 newVel = vel + acc * dt;
    vec3 newPos = posI + newVel * dt;
    
    // Write to output buffer (ping-pong)
    outData[i * 2]     = vec4(newPos, massI);
    outData[i * 2 + 1] = vec4(newVel, 0.0);
}
```
