```spirv
; SPIR-V
; Version: 1.0
; Generator: Manual
; Language: GLSL450
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor %ubo

; Execution modes
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorates
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_in DescriptorSet 0
OpDecorate %gl_in Binding 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out DescriptorSet 0
OpDecorate %gl_out Binding 0

; User-defined inputs
OpDecorate %in_worldPos Location 0
OpDecorate %in_worldPos Component 0
OpDecorate %arr_worldPos ArrayStride 12
OpDecorate %in_normal Location 1
OpDecorate %in_normal Component 0
OpDecorate %arr_normal ArrayStride 12
OpDecorate %in_color Location 2
OpDecorate %in_color Component 0
OpDecorate %arr_color ArrayStride 12

; User-defined output
OpDecorate %outColor Location 0
OpDecorate %outColor Component 0

; UBO
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 BuiltIn 0
OpMemberDecorate %UBO 1 Offset 48
OpMemberDecorate %UBO 1 BuiltIn 0
OpMemberDecorate %UBO 2 Offset 64
OpMemberDecorate %UBO 2 BuiltIn 0
OpMemberDecorate %UBO 3 Offset 80
OpMemberOpDecorate %UBO 3 Offset 80

OpDecorate %v4float Input
OpDecorate %v3float Output

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%bool = OpTypeBool

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_worldPos = OpTypeArray %v3float %uint_4
%ptr_in_worldPos = OpTypePointer Input %arr_worldPos
%in_worldPos = OpVariable %ptr_in_worldPos Input

%arr_normal = OpTypeArray %v3float %uint_4
%ptr_in_normal = OpTypePointer Input %arr_normal
%in_normal = OpVariable %ptr_in_normal Input

%arr_color = OpTypeArray %v3float %uint_4
%ptr_in_color = OpTypePointer Input %arr_color
%in_color = OpVariable %ptr_in_color Input

%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

; UBO struct
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform

; Main function
%main = OpFunction %void OpFunctionParameterless %func
%entry = OpLabel

; ========================================================
; Read 4 positions from gl_in
; ========================================================
; p0 = gl_in[0].gl_Position.xyz
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0v4 = OpLoad %v4float %ptr_p0
%p0 = OpVectorShuffle %v3float %p0v4 %p0v4 0 1 2

; p1 = gl_in[1].gl_Position.xyz
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1v4 = OpLoad %v4float %ptr_p1
%p1 = OpVectorShuffle %v3float %p1v4 %p1v4 0 1 2

; p2 = gl_in[2].gl_Position.xyz
%ptr_p2 = OpAccessChain %p1v4 %gl_in %int_2 %int_0
%p2v4 = OpLoad %v4float %ptr_p2
%p2 = OpVectorShuffle %v3float %p2v4 %p2v4 0 1 2

; p3 = gl_in[3].gl_Position.xyz
%ptr_p3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3v4 = OpLoad %v4float %ptr_p3
%p3 = OpVectorShuffle %v3float %p3v4 %p3v4 0 1 2

; ========================================================
; Read colors
; ========================================================
%ptr_c1 = OpAccessChain %ptr_in_color %in_color %int_1
%c1 = OpLoad %v3float %ptr_c1

%ptr_c2 = OpAccessChain %ptr_in_color %in_color %int_2
%c2 = OpLoad %v3float %ptr_c2

; ========================================================
; 2. Compute tangents
; ========================================================
; t0 = normalize(p2 - p0)
%vec_t0 = OpFSub %v3float %p2 %p0
%len_t0 = OpExtInst %float %glsl450 Length %vec_t0
%len_t0_recip = OpFDiv %float %len_t0_recip %len_t0
%t0 = OpVectorTimesScalar %v3float %vec_t0 %len_t0_recip

; t1 = normalize(p3 - p1)
%vec_t1 = OpFSub %v3float %p3 %p1
%len_t1 = OpExtInst %float %glsl450 Length %vec_t1
%len_t1_recip = OpFDiv %float %len_t0_recip %len_t1
%t0 = OpVectorTimesScalar %v3float %vec_t1 %len_t1_recip

; ========================================================
; Perpendiculars (2D)
; =======================================================
; 4. perp0 = vec2(-t0.y, t0.x)
%t0x = OpCompositeExtract %float %t0 0
%t0y = OpCompositeExtract %float %t0 1
%perp0x = OpFNEG %float %t0y
%perp0y = OpCompositeExtract %float %t0 0
%perp0 = OpCompositeConstruct %v2float %perp0x %perp0y

; 4. perp1 = vec2(-t1.y, t1.x)
%t1x = OpCompositeExtract %float %t1 0
%t1y = OpCompositeExtract %float %t1 1
%perp1x = OpFNEG %float %t1y
%perp1y = OpCompositeExtract %float %t2 0
%perp1 = OpCompositeConstruct %v2float %perp1x %perp1y

; ========================================================
; Thickness
; ========================================================
%thickness = OpConstant %float 0.025

; ====================================<fim-middle>====================
; 6. Emit 4 vertices as TriangleStrip
; ========================================================
; --- Vertex 0: start + perp0 * thickness
%scale0 = OpVectorTimesScalar %v2float %perp0 %thickness
%pos0_2d = OpFAdd %v2float %p1 %scale0   ; start + perp0*thickness
%pos0v4 = OpCompositeConstruct %v4float %pos0_2d 0 1 0 1
%ptr_out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out0 %pos0v4
OpStore %outColor %c1
OpEmitVertex

; --- Vertex 1: start - perp0 * thickness
%scale1 = OpVectorTimesScalar %v2float %perp0 %thickness
%pos1_2d = OpFSub %p1 %scale1
%pos1v4 = OpCompositeConstruct %v4float %pos1_2d 0 1 0 1
%ptr_out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out1 %pos1v4
OpStore %outColor %c1
OpEmitVertex

; --- Vertex 2: end + perp1 * thickness
%scale2 = OpVectorTimesScalar %v2float %perp1 %thickness
%pos2_2d = OpFAdd %p2 %scale2
%pos2v4 = OpCompositeConstruct %v4float %pos2_2d 0 1 0 1
%ptr_out2 = OpAccessChain %ptr_out_v4 %gl2_out %gl_out %int_0
%ptr_out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out2 %pos2v4
OpStore %outColor %c2
OpEmitVertex

; --- Vertex 3: end - perp1 * thickness
%scale3 = OpVectorTimes<fim-suffix>2float %perp1 %thickness
%pos3_2d = OpFSub %p2 %scale3
%pos3v4 = OpCompositeConstruct %v4float %pos3_2d 0 1 0 1
%ptr_out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out3 %pos3v4
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
