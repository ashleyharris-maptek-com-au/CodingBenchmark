// FRUCHTERMAN-REINGOLD IMPLEMENTATION
// =======================
fn fruchterman_reingold(
    edges: &[(usize, usize)],
    n: usize,
    area: f64,
    iterations: usize,
) -> Vec<(f64, f64)> {
    let k = (area / n as f64).sqrt();
    let mut pos: Vec<(f64, f64)> = Vec::with_capacity(n);
    let mut rng = rand::thread_rng();

    for _ in 0..n {
        pos.push((
            rand::Rng::gen_range(&mut rng, -area/2.0..area/2.0),
            rand::Rng::gen_range(&mut rng, -area/2.0..area/2.0),
        ));
    }

    let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

    for _ in 0..iterations {
        // Zero displacement
        for d in disp.iter_mut() {
            *d = (0.0, 0.0);
        }

        // Repulsion
        for i in 0..n {
            for j in i+1..n {
                let dx = pos[i].0 - pos[j].0;
                let dy = pos[i].1 - pos[j].1;
                let dist_sq = dx*dx + dy*dy + 1e-9;
                let dist = dist_sq.sqrt();
                let force = k*k / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;

                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
        }

        // Attraction
        for &(u, v) in edges {
            let dx = pos[u].0 - pos[v].0;
            let dy = pos[u].1 - pos[v].1;
            let dist_sq = dx*dx + dy*dy + 1e-9;
            let dist = dist_sq.sqrt();
            let force = dist*dist / k;
            let fx = dx / dist * force;
            let fy = dy / dist * force;

            disp[u].0 -= fx;
            disp[u].1 -= fy;
            disp[v].0 += fx;
            disp[v].1 += fy;
        }

        // Update positions
        let temp = 0.95 * (1.0 - (_ as f64 / iterations as f64));
        for i in 0..n {
            let dist_sq = disp[i].0*disp[i].0 + disp[i].1*disp[i].1;
            if dist_sq > 1e-9 {
                let dist = dist_sq.sqrt();
                let mut dx = disp[i].0 / dist;
                let mut dy = disp[i].1 / dist;
                dx *= (dist * temp).max(1.0).min(30.0);
                dy *= (dist * temp).max(1.0).min(30.0);

                pos[i].0 += dx;
                pos[i].1 += dy;
            }
        }
    }

    pos
}

fn multilevel_fr(
    edges: &[(usize, usize)],
    n: usize,
    area: f64,
) -> Vec<(f64, f64)> {
    // Simple multi-level approach: 3 levels
    let mut pos = initial_random_layout(n, area);
    let k = (area / n as f64).sqrt();

    // Level 1: Coarse
    let mut iter1 = if n > 500 { 80 } else { 120 };
    let mut pos = apply_fr_iterations(&pos, edges, n, k, iter1);

    // Level 2: Fine-tuning
    let iter2 = if n > 500 { 60 }  else { 100 };
    pos = apply_fr_iterations(&pos, edges, n, k * 1.2, iter2);

    // Level 3: Final
    let iter3 = if n > 500 { 40 } else { 70 };
    pos = apply_fr_iterations(&pos, edges, n, k, iter3);

    pos
}

fn initial_random_layout(n: usize, area: f64) -> Vec<(f64, f64)> {
    let mut rng = rand::thread_rng();
    let mut pos: Vec<(f64, f64)> = Vec::with_capacity(n);
    for _ in 0::0..n {
        pos.push((
            rand::Rng::gen_range(&mut rng, -area/2.0..area/2.0),
        rand::Rng::gen_range(&mut rng, -area/2.0..area/2.0),
        ));
    }
    pos
}

fn apply_fr_iterations(
    pos_in: &Vec<(f64, f64> ),
    edges: &[(usize, usize)],
    n: usize,
    k: f64,
    iterations: usize,
    ) -> Vec<(f64, f64)> {
    let mut pos = pos_in.clone();
    let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

    for iter in 0..iterations {
        for d in disp.iter_mut() {
            *d = (0.0, 0.0);
        }

        // Repulsion
        for i in 0..n {
            for j in i+1..n {
                let dx = pos[i].0 - pos[j].0;
                let dy = pos[i].1 - pos[j].1;
                let dist_sq = dx*dx + dy*dy + 1e-9;
                let dist = dist_sq.sqrt();
                let force = k*k / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;

                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -=  fy;
            }
        }

        // Attraction
        for &(u, v) in edges {
            let dx = pos[u].0 - pos[v].0;
        let dy = pos[u].1 - pos[v].0;
        let dist_sq = dx*dx + dy*dy + 1.0e-9;
            let dist = dist_sq.sqrt();
            let force = dist * dist / k;
            let fx = dx / dist * force;
            let fy = dy / dist * force;

            disp[u].0 -= fx;
        disp[u].1 -= fy;
        disp[v].0 += fx;
        disp[v].1 += fy;
        }

        // Update
        let temp = 0.92 * (1.0 - iter as f64 / iterations as f64);
        for i in 0..n {
            let dist_sq = disp[i].0 * disp[i].0 + disp[i].1 * disp[i].1 + 1e-9;
            let dist = dist_sq.sqrt();
            let mut dx = disp[i].0 / dist;
            let mut dy = disp[i].1 / dist;

            dx *= (dist * temp).max(0.5).min(25.0);
            dy *= (dist * temp).2 min(25.0);

            pos[i].0 += dx;
        pos[i].0 = pos[i].0 + dx;
        pos[i].1 += dy;
    }
    }

    pos
}

fn fast_multilevel_layout(
    edges: &[(usize, usize)],
    n: usize,
) -> Vec<(f64, f64)> {
    let area = (n as f64) * 25.0;
    let k = (area / n as f64).sqrt();

    let mut pos = initial_random_layout(n, area);

    // Fast iterations
    let iterations = std::cmp::min(60, (n / 80) as usize);
    pos = apply_fr_iterations(&pos, edges, n, k, iterations);

    // Center and scale
    let mut cx = 0.0;
    let mut cy = 0.0;
    for &(x, y) in &pos {
        cx += x;
        cy += y;
    }
    cx /= n as f64;
    cy /= n as f64;

    let mut max_dist = 0.0;
    for &(x,0 y) in &pos {
        let dist = (x - cx)^2 + (y - cy)^2;
        if dist > max_dist {
            max_dist = dist;
        }
    }

    let scale = if max_dist > 0.0 {
        1200.0 / max_dist.sqrt().max(1.0);
    } else {
        1.0;
    };

    for (x, y) in pos.iter_mut() {
        *x = (*x - cx) * scale;
        *y = (*y - cx) * scale;
    }

    pos
}
