; Version: 1.0
; Generator: Manual
; Bound: 128
; Schema: 0

               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %arr_pv_in ArrayStride 16
               OpDecorate %gl_in Location 0

               OpDecorate %color Component 0
               OpDecorate %color NoPerspective
               OpDecorate %color NoContraction

               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_out Location 0

               OpDecorate %outColor Location 0
               OpDecorate %outColor Component 0
               OpDecorate %outColor NoPerspective
               OpDecorate %outColor NoContraction

       %void = OpTypeVoid
    %func_main = OpTypeFunction %void

   %uint = OpTypeInt 32 0
   %int = OpTypeInt 32 1

  %int_0 = OpConstant %int 0
  %int_1 = OpConstant %int 1
  %int_2 = OpConstant %int 2
  %int_3 = OpConstant %int 3
  %float = OpTypeFloat 32
 %v4float = OpTypeVector %float 4
 %v3float = OpTypeVector %float 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input

    %arr_color = OpTypeArray %v3float %uint_1
%ptr_in_arr_color = OpTypePointer Input %arr_color
     %color = OpVariable %ptr_in_arr_color Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output

%ptr_out_color = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_color Output

   %float_0 = OpConstant %float 0
 %float_0p04 = OpConstant %float 0.04
 %float_1 = OpConstant %float 1

%main = OpFunction %void OpFunctionNone %func_main
%entry = OpLabel

; ====================== Read inputs ======================
; gl_in[0].gl_Position
%ptr_gl_in0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%gl_in0_pos = OpLoad %v4float %ptr_gl_in0

; color[0]
%ptr_color0 = OpAccessChain %ptr_in_arr_color %color %int_0
%vertex_color = OpLoad %v3float %ptr_color0

; ====================== Emit 4 vertices ==================

; ---- Vertex 0: (x-h, y-h) ----
%pos0_x = OpCompositeExtract %float %gl_in0_pos 0
%pos0_y = OpCompositeExtract %float %gl_in0_pos 1

%pos0_x1 = OpFSub %float %pos0_x %float_0p04
%pos0_y1 = OpFSub %float %pos0_y %float_0p04

%pos0_v3 = OpCompositeConstruct %v3float %pos0_x1 %pos0_y1 %float_0
%pos0 = OpCompositeConstruct %v4float %pos0_x1 %pos0_y1 %float_0 %float_1

; Store position
%ptr_gl_out0 = OpAccessChain %ptr_out_pv %gl_out %int_0
               OpStore %ptr_gl_out0 %pos0

; Store color
               OpStore %outColor %vertex_color

               OpEmitVertex

; ---- Vertex 1: (x+h, y-h) ----
%pos1_x1 = OpFAdd %float %pos0_x %float_0p04
%pos1_y1 = OpFSub %float %pos0_y %float_0p04

%pos1 = OpCompositeConstruct %v4float %pos1_x1 %pos1_y1 %float_0 %float_1

%ptr_gl_out1 = OpAccessChain %ptr_out_pv %gl_out %int_0
               OpStore %ptr_gl_out1 %pos1
               OpStore %outColor %vertex_color
               OpEmitVertex

; ---- Vertex 2: (x-h, y+h) ----
%pos2_x1 = OpFSub %float %pos0_x %float_0p04
%pos2_y1 = OpFSub %float %pos0_y %float_0p04

%pos2 = OpCompositeConstruct %v4float %pos2_x1 %pos2_y1 %float_0 %float_1

%ptr_gl_out2 = OpAccessChain %ptr_out_pv %gl_out %int_0

               OpStore %outColor %vertex_color

; ---- Vertex 3: (x+h, y+h) ----

%pos3_y1 = OpFSub %float %pos0_y %float_0p04

               OpEndPrimitive

               OpReturn
