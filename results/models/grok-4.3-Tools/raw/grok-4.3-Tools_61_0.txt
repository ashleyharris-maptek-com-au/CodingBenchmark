import * as fs from 'fs';

interface Package {
  id: string;
  x: number;
  y: number;
}

interface Input {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}

interface Output {
  routes: string[][];
}

function distance(a: [number, number], b: [number, number]): number {
  const dx = a[0] - b[0];
  const dy = a[1] - b[1];
  return Math.sqrt(dx * dx + dy * dy);
}

function calculateRouteDistance(route: Package[], depot: [number, number]): number {
  if (route.length === 0) return 0;
  let dist = distance(depot, [route[0].x, route[0].y]);
  for (let i = 0; i < route.length - 1; i++) {
    dist += distance([route[i].x, route[i].y], [route[i + 1].x, route[i + 1].y]);
  }
  dist += distance([route[route.length - 1].x, route[route.length - 1].y], depot);
  return dist;
}

function nearestNeighborClustering(
  packages: Package[],
  depot: [number, number],
  truckCount: number
): Package[][] {
  const remaining = [...packages];
  const routes: Package[][] = [];

  // Sort packages by distance from depot
  remaining.sort((a, b) => 
    distance(depot, [a.x, a.y]) - distance(depot, [b.x, b.y])
  );

  for (let i = 0; i < truckCount && remaining.length > 0; i++) {
    const route: Package[] = [];
    let current = depot;

    while (remaining.length > 0) {
      // Find nearest package to current point
      let minDist = Infinity;
      let bestIndex = -1;

      for (let j = 0; j < remaining.length; j++) {
        const p = remaining[j];
        const d = distance(current, [p.x, p.y]);
        if (d < minDist) {
          minDist = d;
          bestIndex = j;
        }
      }

      if (bestIndex === -1) break;

      const nextPkg = remaining[bestIndex];
      route.push(nextPkg);
      current = [nextPkg.x, nextPkg.y];
      remaining.splice(bestIndex, 1);
    }

    if (route.length > 0) {
      routes.push(route);
    }
  }

  // Distribute any leftover packages evenly among existing routes
  let routeIndex = 0;
  while (remaining.length > 0) {
    if (routes.length === 0) {
      // Rare case: too few packages
      routes.push([remaining.pop()!]);
      continue;
    }
    
    const nextPkg = remaining.pop()!;
    routes[routeIndex % routes.length].push(nextPkg);
    routeIndex++;
  }

  return routes;
}

function optimizeRoute(route: Package[], depot: [number, number]): Package[] {
  if (route.length <= 2) return route;

  const n = route.length;
  let bestRoute = [...route];
  let bestDist = calculateRouteDistance(bestRoute, depot);

  // Simple 2-opt improvement
  let improved = true;
  let iterations = 0;
  const maxIterations = 50;

  while (improved && iterations < maxIterations) {
    improved = false;
    iterations++;

    for (let i = 1; i < n - 1; i++) {
      for (let j = i + 1; j < n; j++) {
        // 2-opt swap: reverse segment [i..j]
        const newRoute = [...bestRoute];
        newRoute.splice(i, j - i + 1, ...newRoute.slice(i, j + 1).reverse());

        const newDist = calculateRouteDistance(newRoute, depot);
        if (newDist < bestDist) {
          bestRoute = newRoute;
          bestDist = newDist;
          improved = true;
          break;
        }
      }
      if (improved) break;
    }
  }

  return bestRoute;
}

function planRoutes(input: Input): Output {
  const { depot, truckCount, packages } = input;

  if (packages.length === 0) {
    return { routes: [] };
  }

  // Step 1: Cluster packages into routes using nearest neighbor
  let routes = nearestNeighborClustering(packages, depot, truckCount);

  // Step 2: Optimize each route individually with 2-opt
  routes = routes.map(route => optimizeRoute(route, depot));

  // Step 3: Convert to output format (only IDs)
  const outputRoutes: string[][] = routes.map(route =>
    route.map(pkg => pkg.id)
  );

  return { routes: outputRoutes };
}

function main() {
  const inputStr: string = fs.readFileSync(0, 'utf8').trim();
  const data: Input = JSON.parse(inputStr);

  const output: Output = planRoutes(data);

  console.log(JSON.stringify(output));
}

main();
