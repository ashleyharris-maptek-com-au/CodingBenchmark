; SPIR-V
; Version: 1.0
; Generator: Manual
; Bound: 128
; Schema: 0

               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpDecorate %UBO Block
               OpDecorate %UBO DescriptorSet 0
               OpDecorate %UBO Binding 0
               OpDecorate %ubo PointerUniform

               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

               OpDecorate %worldPos Input
               OpDecorate %normal Input
               OpDecorate %uv Input
               OpDecorate %tangent Input
               OpDecorate %color Input
               OpDecorate %outColor Output

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%ptr_Uniform_v3float = OpTypePointer Uniform %v3float
%ptr_Uniform_mat4v4float = OpTypePointer Uniform %mat4v4float

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float

%worldPos = OpVariable %ptr_Input_v3float Input
%normal = OpVariable %ptr_Input_v3float Input
%uv = OpVariable %ptr_Input_v2float Input
%tangent = OpTypeVariable %ptr_Input_v3float Input
%color = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform

%100 = OpFunction %void OpFunctionNone %void
%101 = OpLabel

; Load inputs
%world = OpLoad %v3float %worldPos
%Nin = OpLoad %v3float %normal
%N = OpNormalize %v3float %Nin
%col = OpLoad %v3float %color
%Ldir4 = OpAccessChain %ptr_Uniform_v4float %ubo 3
%Ldir4v = OpLoad %v4float %Ldir4
%L3 = OpVectorShuffle %v3float %Ldir4v %Ldir4v 0 1 2
%CamPos4 = OpAccessChain %ptr_Uniform_v4float %ubo 4
%CamPosv = OpLoad %v4float %CamPos4
%Cam3 = OpVectorShuffle %v3float %CamPosv %CamPosv 0 1 2
%V = OpNormalize %v3float %OpSub %v3float %Cam3 %world

; Light vector from fragment to light
%Lvec = OpSub %v3float %L3 %world
%L = OpNormalize %v3float %Lvec

; Ambient
%amb = OpConstantComposite %v3float %float_0_1 %float_0_05 %float_0_05
%float_0_1 = OpConstant %float 0.1
%float_0_05 = OpConstant %float 0.05

; Diffuse
%dotNL = OpDot %float %N %L
%maxNL = OpExtInst %float %2 FMax %dotNL %float_0
%diff = OpVectorTimesScalar %v3float %col %maxNL

; Specular
%Lneg = OpFMul %v3float %L %float_n1
%Rvec = OpExtInst %float %2 Reflect %Lneg %N
%R = OpVectorShuffle %v3float %Rvec %Rvec 0 1 2
%VdotR = OpDot %float %V %R
%maxVR = OpExtInst %float %2 FMax %VdotR %float_0
%shininess = OpConstant %float 32
%specPow = OpExtInst %float %2 Pow %maxVR %shininess
%specCol = OpConstantComposite %v3float %float_1 %float_1 %float_1
%spec = OpVectorTimesScalar %v3float %specCol %specPow

; Sum
%diffSpec = OpFAdd %v3float %diff %spec
%finalCol = OpFAdd %v3float %amb %diffSpec

; Clamp
%one = OpConstant %float 1
%zero = OpConstant %float 0
%clamp0 = OpExtInst %float %2 FClamp %finalCol %float_0 %float_1
%clamp1 = OpExtInst %float %2 FClamp %clamp0 %float_0 %float_1
%clamp2 = OpExtInst %float %2 FClamp %clamp1 %float_0 %float_1
%finalv3 = OpFMul %v3float %finalCol %one     ; dummy
%final = OpCompositeConstruct %v4float %finalCol %one

OpStore %outColor %final
OpReturn
OpFunctionEnd
