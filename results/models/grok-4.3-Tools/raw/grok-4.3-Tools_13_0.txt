#include <bits/stdc++.h>
using namespace std;

string longestCommonSubstring(const vector<string>& strs) {
    if (strs.empty()) return "";
    
    int n = strs.size();
    if (n == 1) return strs[0];
    
    // Find the shortest string as starting point
    string shortest = strs[0];
    for (const auto& s : strs) {
        if (s.length() < shortest.length()) {
            shortest = s;
        }
    }
    
    int shortLen = shortest.length();
    if (shortLen == 0) return "";
    
    // Adaptive strategy based on scale
    size_t totalSize = 0;
    for (const auto& s : strs) totalSize += s.length();
    bool isExtreme = (n >= 20 || totalSize > 1000000); // 1MB+
    
    string best = "";
    
    // Strategy 1: For small/medium problems - DP approach (accurate)
    if (!isExtreme && n <= 5 && totalSize < 50000) {
        // Use generalized LCS for multiple strings (DP)
        // For simplicity and correctness, we use brute force binary search on length + hashing
        
        // Binary search on length
        int low = 0;
        int high = shortLen;
        
        while (low <= high) {
            int mid = (low + high) / 2;
            bool found = false;
            string candidate = "";
            
            for (int i = 0; i <= shortLen - mid; ++i) {
                string sub = shortest.substr(i, mid);
                bool isCommon = true;
                for (const auto& s : strs) {
                    if (s.find(sub) == string::npos) {
                        isCommon = false;
                        break;
                    }
                }
                if (isCommon) {
                    found = true;
                    candidate = sub;
                    break;
                }
            }
            
            if (found) {
                best = candidate;
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
    } else {
        // Strategy 2: Binary search + efficient verification for larger problems
        // Binary search on length of LCS
        int low = 0;
        int high = shortLen;
        string result = "";
        
        while (low <= high) {
            int mid = (low + high) / 2;
            bool found = false;
            string candidate = "";
            
            // Check existence of common substring of length mid
            for (int i = 0; i <= shortLen - mid; ++i) {
                string sub = shortest.substr(i, mid);
                bool isCommon = true;
                for (const auto& s : strs) {
                    if (s.find(sub) == string::npos) {
                        isCommon = false;
                    }
                }
                if (isCommon) {
                    found = true;
                    candidate = sub;
                    break;
                }
            }
            
            if (found) {
                result = candidate;
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        best = result;
    }
    
    return best;
}

int main() {
    string line;
    getline(cin, line);
    stringstream ss(line);
    int N;
    ss >> N;
    
    vector<string> strs(N);
    for (int i = 0) {
        getline(cin, strs[i]);
    }
    
    string result = longestCommonSubstring(strs);
    cout << result << endl;
    return 0;
}
