%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_pv_out = OpTypeArray %gl_PerVertex_out %uint_1
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

%arr_v3_in = OpTypeArray %v3float %uint_1

%inWorldPos = OpVariable %ptr_in_v3 Input
%inNormal = OpVariable %ptr_in_v3 Input
%inColor = OpVariable %ptr_in_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%gl_PerVertex_out = OpTypeStruct %v4float

               OpFunction %void %main None

; ======================== Read input ==============================
; gl_in[0].gl_Position
%ptr_gl_in = OpAccessChain %ptr_in_pv %gl_in %int_0 %int_0
%inPos = OpLoad %v4float %ptr_gl_in

; inColor[0]
%ptr_in_col = OpAccessChain %ptr_in_v3 %inColor %int_0
%vertexColor = OpLoad %v3float %ptr_in_col

; ======================== Define offsets ==============================
%h = OpLoad %float %float_0_05
%zero = OpLoad %float %float_0

%pos_x = OpCompositeExtract %float %inPos 0
%pos_y = OpCompositeExtract %float %inPos 1
%pos_z = OpCompositeExtract %float %inPos 2
%pos_w = OpCompositeExtract %float %inPos 3

; Top: (x, y + h)
%top_x = %pos_x
%top_y = OpFAdd %float %pos_y %h
%top = OpCompositeConstruct %v4float %top_x %top_y %pos_z %pos_w

; Right: (x + h, y)
%right_x = OpFAdd %float %pos_x %h
%right_y = %pos_y
%right = OpCompositeConstruct %v4float %right_x %right_y %pos_z %pos_w

; Left: (x - h, y)
%left_x = OpFSub %float %pos_x %h
%left_y = %pos_y
%left = OpCompositeConstruct %v4float %left_x %left_y %pos_z %pos_w

; Bottom: (x, y - h)
%bottom_x = %pos_x
%bottom_y = OpFSub %float %pos_y %h
%bottom = OpCompositeConstruct %v4float %bottom_x %bottom_y %pos_z %pos_w

; ======================== Emit Triangle Strip ==============================
; Top
%ptr_out_top = OpAccessChain %ptr_out_pv2 %gl_out %int_0
OpStore %ptr_out_top %top
%ptr_out_col_top = OpAccessChain %ptr_out_v3 %outColor
OpStore %ptr_out_col_top %vertexColor

; Right
%ptr_out_right = OpAccessChain %ptr_out_pv2 %gl_out %gl_out %int_0
%ptr_out_col_right = OpAccessChain %ptr_out_v3 %outColor
OpStore %ptr_out_col_right %vertexColor
OpStore %ptr_out_right %right
OpEmitVertex

; Left
%ptr_out_left = OpAccessChain %ptr_out_pv2 %gl_out %int_0
OpStore %ptr_out_left %left
%ptr_out_col_left = OpAccessChain %ptr_out_v3 %outColor
OpStore %ptr_out_col_left %vertexColor

; Bottom
%ptr_out_bottom = OpAccessChain %ptr_out_pv2 %gl_out %int_0
OpStore %ptr_out_bottom %bottom
%ptr_out_col_bottom = OpAccessChain %ptr_out_v3 %outColor
