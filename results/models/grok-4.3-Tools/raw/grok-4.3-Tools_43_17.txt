float4 main(PSInput input) : SV_Target0
{
    float2 uv = input.uv * float2(8.0, 16.0);
    float2 f = frac(uv);
    float row = floor(uv.y);
    float offset = fmod(row, 2.0) * 0.5;
    float2 p = f - float2(offset, 0.0);
    float mx = 0.05;
    float my = 0.1;
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 color = mortarColor;
    if (p.x > mx && p.y > my) {
        color = brickColor;
    }
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diff = max(dot(N, L), 0.0);
    float lighting = 0.4 + 0.6 * diff;
    color = color * lighting;
    return float4(color, 1.0);
}
