bool is_valid(const pos_t& p, const vector<int>& bounds) {
    if (p.size() != bounds.size()) return false;
    for (size_t i = 0; i < p.size(); i++) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

size_t hash_pos(const pos_t& p) {
    size_t h = 0;

        h = h * 31 + (x + 1000); // arbitrary offset
    }
    return h;
}

path_t find_path(const State& st, pos_t target) {
    int D = st.dim;
    int max_size = 1;
    for (int b : st.bounds) max_size = max(max_size, b);
    if (max_size > 12 || D > 5) {
        // Heuristic for large dimensions
        return {};
    }

    queue<pair<pos_t, path_t>> q;
    unordered_set<string> visited;

    auto to_str = [](const pos_t& p) {
        string s;
        for (int x : p) s += to_string(x) + ",";
        return s;
    };

    q.push({st.snake[0], {}});
    visited.insert(to_str(st.snake[0]));

    vector<pair<int, int>> dirs;
    for (int a = 0; a < D; a++) {
        dirs.emplace_back(a, -1);
        dirs.emplace_back(a, 1);
    }

    while (!q.empty()) {
        auto [cur, path] = q.front(); q.pop();

        if (cur == target) {
            return path;
        }

        for (auto [ax, dr] : dirs) {
            pos_t next = cur;
            next[ax] += dr;
            if (!is_valid(next, st.bounds)) continue;

            // crude occupied check
            bool occ = false;
            for (size_t i = 0; i < st.snake.size(); i++) {
                if (st.snake[i] == next) {
                    occ = true;
                    break;
                }
            }
            if (occ) continue;

            string key = to_str(next);
            if (visited.count(key)) continue;
            visited.insert(key);

            path_t np = path;
            np.emplace_back(ax, dr);
            q.push({next, np});

            if (q.size() > 10000) break;
        }
    }
    return {};
}
