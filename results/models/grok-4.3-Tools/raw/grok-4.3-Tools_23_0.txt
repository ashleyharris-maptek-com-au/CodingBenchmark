```rust
use std::io::{self, BufRead, BufReader, BufWriter, Write};
use std::sync::mpsc;
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone, Copy, Debug)]
struct Config {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    initial_fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
}

fn main() {
    let stdin = io::stdin();
    let reader = BufReader::new(stdin);
    let mut lines = reader.lines();

    // === READ INITIAL MAP ===
    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => return,
    };

    let mut nums: Vec<f64> = first_line
        .split_whitespace()
        .map(|s| s.clone().parse().unwrap_or(0.0))
        .collect();

    if nums.len() < 5 {
        eprintln!("Invalid initial line");
        return;
    }

    let config = Config {
        width: nums[0] as usize,
        height: nums[1] as usize,
        gravity: nums[2],
        max_thrust: nums[3],
        initial_fuel: nums[4],
        start_x: 0.0,
        start_y: 0.0,
        target_x: 0.0,
        target_y: 0.0,
    };

    let second_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => return,
    };

    let nums2: Vec<f64> = second_line
        .split_whitespace()
        .map(|s| s.clone().parse().unwrap_or(0.0))
        .collect();

    if nums2.len() < 4 {
        eprintln!("Invalid second line");
        return;
    }

    let mut config = config;
    config.start_x = nums2[0];
    config.start_y = nums2[1];
    config.target_x = nums2[2];
    config.target_y = nums2[3];

    // Skip map rows (consume them)
    for _ in 0..config.height {
        if let Some(Ok(_line)) = lines.next() {
            // Optionally parse map if needed
        } else {
            break;
        }
    }

    // === COMMUNICATION CHANNEL ===
    let (tx_state, rx_state) = mpsc::channel::<State>();
    let (tx_cmd, rx_cmd) = mpsc::channel::<(f64, f64)>();

    // === CONTROLLER THREAD ===
    let controller_thread = thread::spawn(move || {
        let mut last_x = config.start_x;
        let mut last_y = config.start_y;
        let mut last_vx = 0.0;
        let mut last_vx_time = Instant::now();

        loop {
            let state = match rx_state.recv() {
                Ok(s) => s,
                Err(_) => break,
            };

            // Very simple but functional controller
            let target_x = config.target_x;
            let target_y = config.target_y;

            let dx = target_x - state.x;
            let dy = target_y - state.y;

            // Horizontal correction
            let desired_angle = if dx.abs() > 1.0 {
                let sign = if dx > 0.0 { 1.0 } else { -1.0 };
                (sign * 0.6).min(0.9).max(-0.9)
            } else {
                0.0
            };

            let mut angle_error = desired_angle - state.angle;

            // Clamp angle_error to [-π/2, π/2] approx
            if angle_error > std::f64::consts::PI / 2.0 {
                angle_error = std::f64::consts::PI / 2.0;
            } else if angle_error > std::f64::consts::PI / 2.0 - 2.0 {
                angle_error = 0.0;
            }

            let turn_rate = (angle_error * 3.0).min(1.0).max(-1.0);

            // Thrust decision
            let mut thrust = 0.0;

            let vertical_speed = state.vy;
            let height_above_target = (target_y - state.y).max(0.0);

            if height_above_target < 25.0 {
                // Close to landing
                if vertical_speed > 1.5 {
                    thrust = 0.85;
                } else if vertical_speed > 0.5 {
                    thrust = 0.6;
                } else {
                    thrust = 0.4;
                }
            } else {
                // Mid-flight
                if vertical_speed > 10.0 {
                    thrust = 0.75;
                    // rotate to upright
                    if state.angle.abs() > 0.2 {
                        // already handled by turn
                    }
                } else {
                    thrust = 0.0;
                }
            }

            // Fuel safety
            if state.fuel < 10.0 {
                thrust = thrust.min(0.3);
            }

            // Better horizontal stabilization
            if dx.abs() > 2.0 && state.vy < 10.0 {
                if state.angle.abs() < 0.1 {
                    thrust = 0.75;
                }
            }

            let _ = tx_cmd.send((thrust, turn_rate));
        }
    });

    // === MAIN IO THREAD ===
    let mut writer = BufWriter::new(io::stdout());

    // Wait for first state
    let mut current_state = State {
        x: config.start_x,
        y: config.start_y,
        vx: 0.0,
        vy: config.gravity,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: config.initial_fuel,
    };

    // Send initial state trigger
    let _ = tx_state.send(current_state);

    for line in lines {
        let line = match line {
            Ok(l) => l,
            Err(_) => break,
        };

        if line.trim().is_empty() {
            continue;
        }

        let nums: Vec<f64> = line
            .split_whitespace()
            .map(|s0) => s0.clone().parse().unwrap_or(0.0))
            .collect();

        if nums.len() < 7 {
            continue;
        }

        current_state = State {
            x: nums[0],
            y: nums[1],
            vx: nums[2],
            vy: nums[3],
            angle: nums[4],
            angular_vel: nums[5],
            fuel: nums[6],
        };

        // Send state to controller thread
        if tx_state.send(current_state).is_err() {
            break;
        }

        // Receive command from controller
        if let Ok((thrust, turn)) = rx_cmd.recv_timeout(Duration::from_millis(30)) {
            let output = format!("{:.4} {:.4}\n", thrust, turn);
            let _ = writer.write_all(output.as_bytes());
            let _ = writer.flush();
        } else {
            // Default safe command if timeout
            let output = format!("0.0 0.0\n");
            let _ = writer.write_all(output.as_bytes());
            let _ = writer.flush();
        }
    }

    // Cleanup
    let _ = controller_thread.join();
}
```
