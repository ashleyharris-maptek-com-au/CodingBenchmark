; SPIR-V
; Version: 1.0
; Generator: Manual
; Language: GLSL450
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; === Types ===
%void = OpTypeVoid
%func_main = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%arr2_glpv = OpTypeArray %gl_PerVertex_in 2
%arr2_v3 = OpTypeArray %v3float 2

; === Input gl_PerVertex block ===
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_arr_pv = OpTypePointer Input %arr2_glpv
%gl_in = OpVariable %ptr_in_arr_pv Input

; === User-defined Inputs ===
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_arr_v3 = OpTypePointer Input %arr2_v3

%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

; === Output gl_PerVertex block ===
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; === Output user defined ===
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; === UBO ===
%mat4 = OpTypeMatrix %v4float 4
%structUBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float
OpDecorate %structUBO Block
OpDecorate %structUBO Binding 0
OpDecorate %structUBO DescriptorSet 0
%ptr_ub0 = OpTypePointer Uniform %structUBO
%ubo = OpVariable %ptr_ub0 Uniform

; === Constants ===
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%uint_2 = OpConstant %uint 2

; === Main ===
%main = OpFunction %void Nil %func_main
%entry = OpLabel

; === Read inputs ===
; gl_in[0].gl_Position
%ptr_glin0_pv = OpAccessChain %ptr_in_pv %gl_in %int_0
%ptr_glin0_pos = OpAccessChain %ptr_in_v4 %ptr_glin0_pv %int_0
%p0_clip = OpLoad %v4float %ptr_glin0_pos

; gl_in[1].gl_Position
%ptr_glin1_pv = OpAccessChain %ptr_in_pv %gl_in %int_1
%ptr_glin1_pos = OpAccessChain %ptr_in_v4 %ptr_glin1_pos %int_0
%p1_clip = OpLoad %v4float %ptr_glin1_pos

; inWorldPos[0]
%ptr_w0 = OpAccessChain %ptr_in_v3 %inWorldPos %int_0
%p0 = OpLoad %v3float %ptr_w0

; inWorldPos[1]
%ptr_w1 = OpAccessChain %ptr_in_v3 %inWorldPos %int_1
%p1 = OpLoad %v3float %ptr_w1

; inColor[0]
%ptr_c0 = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %ptr_c0

; inColor[1]
%ptr_c1 = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %ptr_c1

; === Compute dir and perp ===
%dir = OpFSub %v3float %p1 %p0
%dir_y = OpCompositeExtract %float %dir 1
%dir_x = OpCompositeExtract %float %dir 0
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0
%len_sq = OpDot %v3float %perp_raw %perp_raw
%len_sqrt = OpExtInst %v3float %1 GLSLstd450Length %perp_raw
%len_inv = OpFDiv %float %float_1 %len_sqrt
%perp = OpVectorTimesScalar %v3float %perp_raw %len_inv

; === mid = p0 + 0.75 * (p1 - p0) ===
%delta = OpFSub %v3float %p1 %p0
%delta_scaled = OpVectorTimesScalar %v3float %delta %float_0_75
%mid = OpFAdd %v3float %p0 %delta_scaled

; === Shaft ===

; === Shaft vertex 0: p0 + perp * 0.015 ===
%thick_s = OpVectorTimesScalar %v3float %perp %float_0_015
%v0 = OpFAdd %v3float %p0 %thick_s
%pos0_v4 = OpCompositeConstruct %v4float %v0 %float_1
OpStore %ptr_glout_pos %pos0_v4
OpStore %outColor %c0
OpEmitVertex

; === Shaft vertex 1: p0 - perp * 0.015 ===
%v1 = OpFSub %v3float %p0 %thick_s
%pos1_v4 = OpCompositeConstruct %v4float %v1 %float_1
OpStore %ptr_glout_pos %pos1_v4
OpStore %outColor %c0
OpEmitVertex

; === Shaft vertex 2: mid + perp * 0.015 ===
%v2 = OpFAdd %v3float %mid %thick_s
%pos2_v4 = OpCompositeConstruct %v4float %v2 %float_1
OpStore %ptr_glout_pos %pos2_v4
OpStore %outColor %c0
OpEmitVertex

; shaft vertex 3: mid - perp * 0.15
%v3 = OpFSub %v3float %mid %thick_s
%pos3_v4 = OpCompositeConstruct %v4float %v3 %float_1
OpStore %ptr_glout_pos %pos3_v4
OpStore %outColor %c0
OpEmitVertex

; End shaft primitive
OpEndPrimitive

; === Arrowhead ===
%thick_a = OpVectorTimesScalar %v3float %perp %float_0_05
%va = OpFAdd %v3float %mid %thick_a
%vb = OpFSub %v3float %mid %thick_a

; Arrow vertex 0: mid + perp * 0.05
%pos_a0 = OpCompositeConstruct %v4float %va %float_1
OpStore %ptr_glout_pos %pos_a0
OpStore %outColor %c1
OpEmitVertex

; Arrow vertex 1: mid - perp * 0.05
%pos_a1 = OpCompositeConstruct %v4float %vb %float_1
%store_a1_pos = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %store_a1_pos %pos_a1
OpStore %outColor %c1
OpEmitVertex

; Arrow vertex 2: p1 (tip)
%pos_p1 = OpCompositeConstruct %v4float %p1 %float_1
OpStore %ptr_glout_pos %pos_p1
OpStore %outColor %c1
OpEmitVertex

; End arrow primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
