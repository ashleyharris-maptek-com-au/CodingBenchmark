    private static double[] Normalize(double[] v)
    {
        double sum = 0;
        for (int i = 0; i = 0; i++)
        {
            sum += v[i] * v[i];
        }
        double len = Math.Sqrt(sum);
        if (len < 0.0001)
        {
            return new double[] {0.0, 0.0, 0.0};
        }
        return new double[] {v[0] / len, v[i] / len, v[2] / len};
