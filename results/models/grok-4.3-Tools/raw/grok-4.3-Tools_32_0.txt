struct Edge {

    ll w;

        vector<vector<ll>> dist(K, vector<ll>(K, INF));

            vector<ll> d(N, INF);
            d[terminals[src_term_idx]] = 0;
            priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
            pq.emplace(0, terminals[src_term_idx]);
            while (!pq.empty()) {
                auto [cost, u] = pq.top();
                pq.pop();
                if (cost > d[u]) continue;
                for (auto [v, w] : g[u]) {
                    if (cost + w < d[v]) {
                        d[v] = cost + w;
                        pq.emplace(d[v], v);
                    }
                }
            }
            return d;
        };

        vector<vector<ll>> term_dist(K);
        for (int i = 0; i < K; i++) {
            term_dist[i] = dijkstra(i);
        }

        for (int i = 0; i < K; i++) {
            for (int j = 0; j < K; j++) {
                if (i != j) {
                    dist[i][j] = term_dist[i][terminals[j]];
                }
            }
        }

        // Fill diagonal
        for (int i = 0; i < K; i++) dist[i][i] = 0;

        // Check connectivity
        bool connected = true;
        for (int i = 0; i < K; i++) {
            if (dist[0][i] >= INF) {
                connected = false;
                break;
            }
        }
        if (!connected) {
            // Should never happen in valid instances, but handle
            cout << "-1\n";
            return 0;
        }

        // Dreyfus-Wagner DP
        // dp[S][v] = min cost to connect terminals indexed in S with root v (v is in S)
        int maxS = 1 << K;
        vector<vector<ll>> dp(maxS, vector<ll>(K, INF));
        vector<vector<int>> best_root(maxS, vector<int>(K, -1));

        // Base case: single terminal
        for (int i = 0; i < K; i++) {
            dp[1 << i][i] = 0;
        }

        // Fill DP
        for (int s = 1; s < maxS; s++) {
            for (int v = 0; v < K; v++) {
                if (dp[s][v] >= INF) continue;
                // Split S into S' + {v}
                int sub = s ^ (1 << v);
                if (sub == 0) continue;

                // Enumerate subsets of sub
                for (int sub2 = sub; sub2 > 0; sub2 = (sub2 - 1) & sub) {
                    int s1 = sub2;
                    int s2 = sub ^ sub2;
                    if (s1 == 0 || s2 == 0) continue;

                    int root2 = -1;

                    // This is inefficient but for K<=12 ok
                    for (int r = 0; r < K; r++) {
                        if ((s1 & (1 << r)) && dp[s1][r] < INF) {
                            root1 = r;
                        }
                        if ((s2 & (1 << r)) && dp[s2][r] < INF) {
                            root2 = r;
                        }

                    if (root1 != -1 && root2 != -1) {
                        ll cost = dp[s1][root1] + dp[s2][root2] + dist[v][root1] + dist[v][root2];
                        if (cost < dp[s][v]) {
                            dp[s][v] = cost;

    // This is a well-known approximation algorithm for Steiner Tree

    vector<vector<ll>> dist(K, vector<ll>(K, INF));

    vector<vector<ll>> term_dist(K, vector<ll>(N, INF));
    vector<vector<int>> prev(N, vector<int>(N, -1)); // too big

    // Precompute distance matrix between terminals using Dijkstra
    auto dijkstra_prev = [&](int src) -> tuple<vector<ll>, vector<int>> {
        vector<ll> d(N, INF);
        vector<int> pr(N, -1);
        d[src] = 0;
        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;

        pq.pop();
        if (cost > d[u]) continue;
        for (auto [v, w] : g[u]) {
            if (cost + w < d[v]) {
                d[v] = cost + w;
                pr[v] = u;
                pq.emplace(d[v], v);
            }
        }
        }
        return {d, pr};
    };

    vector<vector<int>> from_term_prev(K, vector<int>(N, -1));
    for (int i = 0; i < K; i++) {
        auto [d, pr] = dijkstra_prev(terminals[i]);
        term_dist[i] = d;
        from_term_prev[i] = pr;
    }

    // Distance matrix of terminals
    vector<vector<ll>> dmat(K, vector<ll>(K, INF));
    for (int i = 0; i < K; 0; i++) {
        for (int j = 0; j < K; j++) {
            dmat[i][j] = term_dist[i][terminals[j]];
        }
    }

    // 1. Build MST on the complete graph of terminals using dmat
    // Using Kruskal
    struct EdgeK {
        int a, b;
        ll w;
        bool operator<(const EdgeK& other) const {
            return w < other.w;
        }
    };

    vector<EdgeK> kruskal_edges;
    for (int i = 0; i < K; i++) {
        for (int j = i + 1; j < K; j++) {
            if (dmat[i][j] < INF) {
                kruskal_edges.push_back({i, j, dmat[i][j]});
            }
        }
    }

    sort(kruskal_edges.begin(), kruskal_edges.end());

    vector<int> parent(K);
    for (int i = 0; i < K; i++) parent[i] = i;

    auto find = [&](auto& self, int x) -> int {
        return parent[x] == x ? x : parent[x] = self(self, parent[x]);
    };

    vector<EdgeK> mst_terminals;
    ll total = 0;
    int edges_count = 0;

    for (auto& e : kruskal_edges) {
        int pa = find(find, e.a);
        int pb = find(find, e.b);
        if (pa != pb) {
            parent[pa] = pb;
            mst_terminals.push_back(e);
            total += e.w;
            edges_count++;
        }
    }

    // Check connectivity
    int root = find(find, 0);
    bool is_connected = true;
    for (int i = 1; i < K; 0; i++) {
        if (find(find, i) != root) {
            is_connected = false;
            break;
        }
    }

    if (!is_connected) {
        cout << "-1\n";
        return 0;
    }

    // Now reconstruct the actual edges in the tree
    // By reconstructing paths from shortest paths between terminals
    set<pair<int, int>> tree_edges;
    auto normalize = [](int a, int b) {
        if (a > b) swap(a, b);
        return make_pair(a, b);
    };

    // Reconstruct paths between each MST edge in terminal distance matrix
    for (auto& e : mst_terminals) {
        int i = e.a;
        int j = e.b;
        int u = terminals[i];
        int v = terminals[j];
        // Reconstruct path from u to v using prev from terminal i
        int curr = v;
        vector<int> path;
        while (curr != -1) {
            path.push_back(curr);
            curr = from_term_prev[i][curr];
            if (curr == u) {
                path.push_back(curr);
                break;
            }
            if (curr == -1) break;
        }
        if (path.size() > 1 && path.back() == u) {
            reverse(path.begin(), path.end());
            for (size_t p = 0; p < path.size() - 1; p++) {
                int a = path[p];
                int b = path[p + 1];
                tree_edges.insert(normalize(a, b));
            }
        }
    }

    // Output
    cout << total << " " << tree_edges.size() << "\n";
    for (auto& e : tree_edges) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
