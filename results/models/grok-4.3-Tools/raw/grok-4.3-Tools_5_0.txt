use std::io::{self, BufRead, BufReader};
use std::time::{Instant, Duration};
use std::collections::VecDeque;

#[derive(Clone, Copy, PartialEq, Eq, Hash)]
struct Point {
    x: i32,
    y: i32,
}

impl Point {
    fn new(x: i32, y: i32) -> Self {
        Self { x, y }
    }
}

#[derive(Clone)]
struct Grid {
    width: i32,
    height: i32,
    blocked: Vec<Vec<bool>>,
    total_free: usize,
}

impl Grid {
    fn is_valid(&self, p: Point) -> bool {
        p.x >= 0 && p.x < self.width && p.y >= 0 && p.y < self.height && !self.blocked[p.y as usize][p.x as usize]
    }

    fn neighbors(&self, p: Point) -> Vec<Point> {
        let mut res = Vec::with_capacity(4);
        let dirs = [(0, 1), (-1, 0), (0, 2), (1, 0)];
        for &(dx, dy) in &dirs {
            let np = Point::new(p.x + dx, p.y + dy);
            if self.is_valid(np) {
                res.push(np);
            }
        }
        res
    }
}

fn parse_input() -> (Grid, bool) {
    let reader = BufReader::new(io::stdin());
    let mut lines = reader.lines();
    
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    
    let width: i32 = parts[0].parse().unwrap();
    let height: i32 = parts[1].parse().unwrap();
    let obstacle_count: usize = parts[2].parse().unwrap();
    let require_cycle: bool = parts[3].parse::<i32>().unwrap() == 1;
    
    let mut blocked = vec![vec![false; width as usize]; height as usize];
    
    for _ in 0 {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<&str> = line.split_whitespace().collect();
        if coords.len() == 2 {
            let x: i32 = coords[0].parse().unwrap();
            let y: i32 = coords[1].parse().unwrap();
            if x >= 0 && x < width && y >= 0 && y < height {
                blocked[y as usize][x as usize] = true;
            }
        }
    }
    
    let mut total_free = 0;
    for y in 0..height {
        for x in 0..width {
            if !blocked[y as usize][x as usize] {
                total_free += 1;
            }
        }
    }
    
    let grid = Grid {
        width,
        height,
        blocked,
        total_free,
    };
    
    (grid, require_cycle)
}

fn backtrack(grid: &Grid, current: Point, visited: &mut Vec<Vec<bool>>, path: &mut Vec<Point>, require_cycle: bool, start_time: Instant, time_limit: Duration) -> bool {
    if path.len() == grid.total_free {
        if !require_cycle {
            return true;
        }
        // Check if last cell is adjacent to (0,0)
        let last = path[path.len() - 1];
        let dx = (last.x - 0).abs();
        let dy = (last.y - 0).abs();
        return dx + dy == 1;
    }
    
    if path.len() > 0 && path.len() % 100 == 0 {
        if start_time.elapsed() > time_limit {
            return false;
        }
    }
    
    let mut neigh = grid.neighbors(current);
    // Sort neighbors for heuristic better branching
    neigh.sort_by(|a, b| {
        let a_count = grid.neighbors(*a).len();
        let b_count = grid.neighbors(*b).len();
        b_count.cmp(&a_count) // prefer higher degree first? reverse for dead-end first
    });
    
    for &next in &neigh {
        if visited[next.y as usize][next.x as usize] {
            continue;
        }
        
        visited[next.y as usize][next.x as usize] = true;
        path.push(next);
        
        if backtrack(grid, next, visited, path, require_cycle, start_time, time_limit) {
            return true;
        }
        
        path.pop();
        visited[next.y as usize][next.x as usize] = false;
    }
    
    false
}

fn solve_hamiltonian(grid: &Grid, require_cycle: bool) -> Option<Vec<Point>> {
    let start = Point::new(0, 0);
    if !grid.is_valid(start) {
        return None;
    }
    
    let mut visited = vec![vec![false; grid.width as usize]; grid.height as usize];
    let mut path: Vec<Point> = Vec::with_capacity(grid.total_free);
    
    visited[start.y as usize][start.x as usize] = true;
    path.push(start);
    
    let time_limit = Duration::from_secs(290); // ~5 minutes buffer
    let start_time = Instant::now();
    
    if backtrack(grid, start, visited, path, require_cycle, start_time, time_limit) {
        Some(path)
    } else {
        None
    }
}

fn main() {
    let (grid, require_cycle) = parse_input();
    
    // Special cases for small grids
    if grid.total_free == 0 {
        println!("0");
        return;
    }
    
    if grid.total_free == 1 {
        if require_cycle {
            println!("0");
        } else {
            println!("1");
            println!("0 0");
        }
        return;
    }
    
    if grid.width * grid.height > 10000 {
        // Very large grid - warn but still try
        eprintln!("Warning: Large grid detected");
    }
    
    match solve_hamiltonian(&grid, require_cycle) {
        Some(path) {
            println!("{}", path.len());
            for p in &path {
                println!("{} {}", p.x, p.y);
            }
        }
        None => {
            println!("0");
        }
    }
}
