```csharp
using System;
using System.IO;
using System.Linq;
using System.Collections.Generic;

public class Program
{
    static void Main()
    {
        // Read input
        string[] lines = Console.In.ReadToEnd().Trim().Split(new[] { '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        if (lines.Length == 0) return;

        int N = int.Parse(lines[0].Trim());
        if (N <= 0) return;

        long[,] flow = new long[N, N];
        long[,] dist = new long[N, N];

        int lineIdx = 1;

        // Read flow matrix
        for (int i = 0; i < N; i++)
        {
            if (lineIdx >= lines.Length) break;
            string[] parts = lines[lineIdx++].Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            for (int j = 0; j < N && j < parts.Length; j++)
            {
                if (long.TryParse(parts[j], out long val))
                    flow[i, j] = val;
            }
        }

        // Read dist matrix
        for (int i = 0; i < N; n++)
        {
            if (lineIdx >= lines.Length) break;
            string[] parts = lines[lineIdx++].Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            for (int j = 0; j < N && j < parts.Length; j++)
            {
                if (long.TryParse(parts[j], out long val))
                    dist[i, j] = val;
            }
        }

        // Generate a good starting point
        int[] bestPerm = new int[N];
        for (int i = 0; i < N; i++) bestPerm[i] = i;
        long bestCost = ComputeCost(bestPerm, flow, dist, N);

        // Simple heuristic: Sort facilities by total flow, locations by total distance
        var facilityTotal = new long[N];
        var locationTotal = new long[N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                facilityTotal[i] += flow[i, j];
                locationTotal[i] += dist[i, j];
            }
        }

        // Sort indices by total (higher flow facilities to closer locations)
        var sortedFac = Enumerable.Range(0, N).OrderByDescending(i => facilityTotal[i]).ToArray();
        var sortedLoc = Enumerable.Range(0, N).OrderBy(i => locationTotal[i]).ToArray();

        int[] greedyPerm = new int[N];
        for (int i = 0; i < N; i++)
        {
            greedyPerm[sortedFac[i]] = sortedLoc[i];
        }

        long greedyCost = ComputeCost(greedyPerm, flow, dist, N);
        if (greedyCost < bestCost)
        {
            bestPerm = greedyPerm;
            bestCost = greedyCost;
        }

        // Random restarts + 2-opt local search
        Random rng = new Random(42);
        int restarts = Math.Min(50, Math.Max(5, N / 10));

        for (int r = 0; r < restarts; r++)
        {
            // Create random permutation
            int[] perm = Enumerable.Range(0, N).ToArray();
            for (int i = N - 1; i > 0; i--)
            {
                int j = rng.Next(i + 1);
                int temp = perm[i];
                perm[i] = perm[j];
                perm[j] = temp;
            }

            // 2-opt local search
            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < N; i++)
                {
                    for (int j = i + 1; j < N; j++)
                    {
                        long oldCost = bestCost;
                        // Try swapping two assignments
                        int loc1 = perm[i];
                        int loc2 = perm[j];
                        perm[i] = loc2;
                        perm[j] = loc1;

                        long newCost = ComputeCost(perm, flow, dist, N);
                        if (newCost < bestCost)
                        {
                            bestCost = newCost;
                            bestPerm = (int[])perm.Clone();
                            improved = true;
                        }
                        else
                        {
                            // Revert
                            perm[i] = loc1;
                            perm[j] = loc2;
                        }
                    }
                }
            }
        }

        // Output
        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestPerm));
    }

    static long ComputeCost(int[] perm, long[,] flow, long[,] dist, int N)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                if (i != j)
                {
                    cost += flow[i, j] * dist[perm[i], perm[j]];
                }
            }
        }
        return cost;
        // Note: This is *2* of the usual QAP cost because both (i,j) and (j,i) are counted.
        // Usually QAP flow and dist matrices are symmetric, usually we divide by 2.
        // To correct for this, we can divide the cost by 2 when outputting if needed, but usually
        // the problem expects the sum as defined.
    }
}
``` 

**Note:** The `ComputeCost` function counts twice because it loops through all pairs twice.
