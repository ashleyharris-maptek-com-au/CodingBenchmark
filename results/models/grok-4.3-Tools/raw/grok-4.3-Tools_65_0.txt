import * as fs from 'fs';

interface Task {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
}

interface Data {
  machineCount: number;
  tasks: Task[];
}

interface ScheduleEntry {
  id: string;
  start: number;
  machine: number;
}

interface ScheduleOutput {
  schedule: ScheduleEntry[];
}

interface OrderOutput {
  order: string[];
}

type Output = ScheduleOutput | OrderOutput;

class TaskScheduler {
  private data: Data;
  private taskMap: Map<string, Task>;
  private adj: Map<string, Set<string>>;
  private indegree: Map<string, number>;

  constructor(data: Data) {
    this.data = data;
    this.taskMap = new Map();
    this.adj = new Map();
    this.indegree = new Map();
    this.buildGraph();
  }

  private buildGraph(): void {
    const { tasks } = this.data;

    for (const task of tasks) {
      this.taskMap.set(task.id, task);
      this.adj.set(task.id, new Set());
      this.indegree.set(task.id, 0);
    }

    for (const task of tasks) {
      if (task.deps && task.deps.length > 0) {
        for (const depId of task.deps) {
          if (this.adj.has(depId)) {
            this.adj.get(depId)!.add(task.id);
            const currentIndegree = this.indegree.get(task.id)!;
            this.indegree.set(task.id, currentIndegree + 1);
          }
        }
      }
    }
  }

  private topologicalSort(): string[] {
    const queue: string[] = [];
    const order: string[] = [];
    const indegreeCopy = new Map(this.indegree);

    for (const [id, deg] of indegreeCopy) {
      if (deg === 0) {
        queue.push(id);
      }
    }

    while (queue.length > 0) {
      // Sort queue by earliest deadline first (EDD rule)
      queue.sort((a, b) => {
        const taskA = this.taskMap.get(a)!;
        const taskB = this.taskMap.get(b)!;
        return taskA.deadline - taskB.deadline;
      });

      const current = queue.shift()!;
      order.push(current);

      for (const neighbor of this.adj.get(current)!) {
        const currentDeg = indegreeCopy.get(neighbor)!;
        const newDeg = currentDeg - 1;
        indegreeCopy.set(neighbor, newDeg);
        if (newDeg === 0) {
          queue.push(neighbor);
        }
      }
    }

    if (order.length !== this.data.tasks.length) {
      // Cycle detected - fallback to any order
      return Array.from(this.taskMap.keys());
    }

    return order;
  }

  public generateSchedule(): Output {
    const machineCount = this.data.machineCount;
    if (machineCount <= 0) {
      return { order: this.topologicalSort() };
    }

    const order = this.topologicalSort();
    const schedule: ScheduleEntry[] = [];

    // Machine finish times
    const machineFinish: number[] = new Array(machineCount).fill(0);

    // Track start time for each task
    const startTimes: Map<string, number> = new Map();

    // First pass: compute earliest start time considering dependencies
    const earliestStart: Map<string, number> = new Map();
    for (const id of order) {
      const task = this.taskMap.get(id)!;
      let earliest = 0;

      if (task.deps && task.deps.length > 0) {
        for (const depId of task.deps) {
          const depStart = earliestStart.get(depId) || 0;
          const depTask = this.taskMap.get(depId)!;
          earliest = Math.max(earliest, depStart + depTask.duration);
        }
      }
      earliestStart.set(id, earliest);
    }

    // Second pass: assign to machines greedily
    for (const id of order) {
      const task = this.taskMap.get(id)!;
      const depFinish = earliestStart.get(id) || 0;

      // Find machine with earliest finish time
      let bestMachine = 0;
      let bestFinish = machineFinish[0];

      for (let m = 1; m < machineCount; m++) {
        if (machineFinish[m] < bestFinish) {
          bestFinish = machineFinish[m];
          bestMachine = m;
        }
      }

      const start = Math.max(depFinish, bestFinish);
      const finish = start + task.duration;

      schedule.push({
        id: id,
        start: start,
        machine: bestMachine
      });

      machineFinish[bestMachine] = finish;
      startTimes.set(id, start);
    }

    return { schedule };
  }
}

function main(): void {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: Data = JSON.parse(input);

  const scheduler = new TaskScheduler(data);
  const output: Output = scheduler.generateSchedule();

  console.log(JSON.stringify(output));
}

main();
