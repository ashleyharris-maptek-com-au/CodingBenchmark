```rust
use std::io::{self, BufRead, BufReader};
use std::collections::{VecDeque, HashSet};

#[derive(Clone, Copy, PartialEq, Eq, Hash)]
struct Point {
    x: usize,
    y: usize,
}

fn main() {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin);

    let mut line = String::new();
    if reader.read_line(&mut line).is_err() {
        println!("0");
        return;
    }
    let line = line.trim();
    let mut parts: Vec<&str> = line.split_whitespace().collect();
    if parts.len() < 2 {
        println!("0");
        return;
    }

    let h: usize = match parts[0].parse() {
        // Fix: Parse height and width properly
        Ok(val) => val,
        Err(_) => {
            println!("0");
        return;
        }
    };
    let w: usize = match parts[1].parse() {
        Ok(val) => val,
        Err(_) => {
            println!("0");
            return;
        }
    };

    let mut maze: Vec<Vec<char>> = Vec::with_capacity(h);
    let mut start = Point { x: 0, y: 0 };
    let mut goal = Point { x: 0, y: 0 };

    for y in 0..h {
        line.clear();
        if reader.read_line(&mut line).is_err() {
            println!("0");
            return;
        }
        let mut row: Vec<char> = line.trim_end_matches(['\n', '\r']).chars().collect();
        if row.len() != w {
            // Pad or adjust if needed, but assume correct input
            while row.len() < w {
                row.push(' ');
            }
            row.truncate(w);
        }

        for x in 0..w {
            if row[x] == 'A' {
                start = Point { x, y };
            } else if row[x] = 'B' {
                goal = Point { x, y };
            }
        }
        maze.push(row);
    }

    // BFS for shortest path
    let mut queue: VecDeque<Point> = VecDeque::new();
    let mut visited: HashSet<Point> = HashSet::new();
    let mut parent: Vec<Vec<Point>> = vec![vec![Point { x: usize::MAX, y: usize::MAX}; w]; h];
    let mut came_from: Vec<Vec<bool>> = vec![vec![false]; w]; // unused

    queue.push_back(start);
    visited.insert(start);
    parent[start.y][start.x] = start;

    let dirs = [(0, -1i32), (0, 1i32), (-1i32, 0), (1i32, 0)];

    let mut found = false;

    while let Some(current) = queue.pop_front() {
        if current.x == goal.x && current.y == goal.y {
            found = true;
            break;
        }

        for (dx, dy) in dirs.iter() {
            let nx = current.x as i32 + dx;
            let ny = current.y as i32 + dy;

            if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                let npx = nx as usize;
                let npy = ny as usize;

                let ch = maze[npy][npx];
                if (ch == ' ' || ch == 'A' || ch == 'B') && !visited.contains(&Point { x: npx, y: npy }) {
                    queue.push_back(Point { x: npx, y: npy });
                    visited.insert(Point { x: npx, y: npy });
                    parent[npy][npx] = current;
                }
            }
        }
    }

    if !found {
        println!("0");
        return;
    }

    // Reconstruct path
    let mut path: Vec<Point> = Vec::new();
    let mut current = goal;
    while current != start {
        path.push(current);
        current = parent[current.y][current.x];
        if current.x == usize::MAX {
            println!("0");
            return;
        }
    }
    path.push(start);
    path.reverse();

    // Output
    println!("{}", path.len());
    for p in &path {
        println!("{} {}", p.x, p.y);
    }
}
```
