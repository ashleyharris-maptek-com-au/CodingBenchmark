; SPIR-V
; Version: 1.0
; Generator: Manual
; Bound: 128
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %ubo_struct Block
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
OpDecorate %worldPos Input
OpDecorate %normal Input
OpDecorate %uv Input
OpDecorate %tangent Input
OpDecorate %color Input
OpDecorate %outColor Output
OpDecorate %ubo Uniform
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%bool = OpTypeBool
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 =0
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_0_1 = OpConstant %float 0.1
%float_0_3_bg = OpConstant %float 0.3
%float_0_1_bg = OpConstant %float 0.1
%float_0_3_bg2 = OpConstant %float 0.3
%float_0_1_3 = OpConstant %float 0.1
%float_0_3_2 = OpConstant %float 0.3
%float_0_1_3_2 = OpConstant %float 0.1
%float_0_8_2 = OpConstant %float 0.8
%float_0_02 = OpConstant %float 0.02
%float_0_12 = OpLength %float 0.1
OpTypePointer Input %vec3
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_uniform_mat4 = OpTypePointer Uniform %mat4
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_uniform_ubo Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%tangent = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%main = OpFunction %void OpFunctionControlMaskNone %void
%entry = OpLabel
; Load inputs
%uv_val = OpLoad %vec2 %uv
%norm_in = OpLoad %vec3 %normal
%norm_normalized = OpNormalize %vec3 %norm_in
; Compute center of nearest cell
%uvx = OpCompositeExtract %float %uv_val 0
%uvy = OpCompositeExtract %float %uv_val 1
%uv_scaled = OpFMul %vec2 %uv_val %float_0_5
%mod_x = OpFMod %float %uvx OpConstant %float 1.0
%mod_y = OpFMod %float %uvy OpConstant %float 1.0
%center_x = OpFSub %float %uvx %mod_x
%center_y = OpFSub %float %uvy %mod_y
%center = OpCompositeConstruct %vec2 %center_x %center_y
; Distance to center
%delta = OpFSub %vec2 %uv_val %center
%delta_x = OpCompositeExtract %float %delta 0
%delta_y = OpCompositeExtract %float %delta 0
%delta_y2 = OpFReplicate %float %delta_y
%delta_squared = OpFMul %float %delta_x %delta_x
%delta_squared_y = OpFMul %float %delta_y %delta_y
%dist_sq = OpFAdd %float %delta_squared %delta_squared_y
%radius_sq = OpConstant %float 0.09
%is_dot = OpFOrdLessThan %bool %dist_sq %radius_sq
; Colors
%bg_color_vec3 = OpCompositeConstruct %vec3 %float_0_1_bg %float_0_1_bg %float_0_3_bg
%dot_color_vec3 = OpCompositeConstruct %vec3 %float_1 %float_0_8 %float_0_0
%base_color_vec3 = OpSelect %vec3 %is_dot %dot_color_vec3 %bg_color_vec3
; Lambertian shading
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightDir = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%lightLen_sq = OpDot %vec3 %lightDir %lightDir
%lightLen = OpSqrt %float %lightLen_sq
%lightDir_norm = OpFDiv %vec3 %lightDir %lightLen
%dot_prod = OpDot %vec3 %norm_normalized %lightDir_norm
%diffuse = OpFMax %float %dot_prod %float_0
%diffuse_factor = OpFMul %float %float_0_7 %diffuse
%lambert = OpFAdd %float %float_0_3 %diffuse_factor
%final_color_vec3 = OpFMul %vec3 %base_color_vec3 %lambert
%out_col = OpCompositeConstruct %vec4 %final_color_vec3 %float_1
OpStore %outColor %out_col
OpReturn
OpFunctionEnd
