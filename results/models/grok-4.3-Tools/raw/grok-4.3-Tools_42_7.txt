%e2y2x = OpCompositeExtract %float %e2 2

%e2x2y = OpCompositeExtract %f

%e1_correct = OpFSub %v3float %p1 %p0

%e2_correct = OpFSub %v3float %p2 %p0

%y1 = OpCompositeExtract %float %e1_correct 1
%z1 = OpCompositeExtract %float %e1_correct 2
%z2 = OpCompositeExtract %float %e2_correct 2

%x2 = OpCompositeExtract %v3float %e2_correct 0
%x1 = OpCompositeExtract %float %e1_correct 0

; This is broken. Below is the correct line-by-line cross product
%bf = OpFMul %float %y1 %z2

%cd = OpFMul %float %z1 %x2
