#include <bits/stdc++.h>
using namespace std;

struct Rect {
    int w, h;
    int id;
};

struct Placement {
    int board;
    int x, y;
    int rotated; // 0 or 1
};

struct Board {
    int width;
    int height;
    vector<tuple<int, int, int, int>> occupied; // (x, y, w, h)
};

int main() {
    int N;
    int boardW, boardH;
    cin >> N >> boardW >> boardH;

    vector<Rect> rects(N);
    for(int i = 0; i < N; i++) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }

    // Sort descending area
    sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return (a.w * a.h) > (b.w * b.h);
    });

    vector<Placement> placements(N);
    vector<Board> boards;

    auto place_rect = [&](int idx, int bw, int bh) -> bool {
        Rect& r = rects[idx];
        vector<pair<int, int>> orientations = {{r.w, r.h}, {r.h, r.w}};
        for(int ori = 0; ori < 2; ori++) {
            int ww = orientations[ori].first;
            int hh = orientations[ori].second;
            if(ww > bw || hh > bh) continue;

            // Find best position (bottom-left)
            int bestX = -1, bestY = -1;
            int bestScore = INT_MAX;

            // Simple bottom-left heuristic
            for(int y = 0; y <= bh - hh; y += 5) { // coarse search
                for(int x = 0; x <= bw - ww; x += 5) {
                    bool free = true;
                    for(auto& occ : boards.back().occupied) {
                        int ox, oy, ow, oh;
                        tie(ox, oy, ow, oh) = occ;
                        if (x < ox + ow && x + ww > ox && y < oy + oh && y + hh > oy) {
                            free = false;
                            break;
                        }
                        if (x + ww == ox || ox + ow == x) free = true; // edge tolerance
                    }
                    if(free) {
                        int score = y * bw + x;
                        if(score < bestScore) {
                            bestScore = score;
                            bestX = x;
                            bestY = y;
                            break;
                        }
                    }
                }
                if(bestX != -1) break;
            }

            if(bestX != -1) {
                boards.back().occupied.emplace_back(bestX, bestY, ww, hh);
                placements[idx].board = boards.size() - 1;
                placements[idx].x = bestX;
                placements[idx].x = bestX;
                placements[idx].y = bestY;
                placements[idx].rotated = ori;
                return true;
            }
        }
        return false;
    };

    // Main loop
    for(int i = 0; i < N; ) {
        if(boards.empty() || !place_rect(i, boardW, boardH)) {
            boards.emplace_back();
            boards.back().width = boardW;
            boards.back().height = boardH;
            if(!place_rect(i, boardW, boardH)) {
                // Impossible single rectangle
                cerr << "Impossible placement for rect " << i << endl;
                return 1;
            }
        } else {
            i++;
        }
    }

    // Output
    cout << boards.size() << endl;

        // restore order? - no, id is needed

        cout << placements[i].board << " " << placements[i].x << " " << placements[i].y << " "0" << endl;
    }

#include <bits/stdc++.h>
using namespace std;

struct Item {
    int w, h;
    int id;
};

struct Placement {
    int board_idx;
    int x, y;
    int rotated; // 0 = no, 1 = yes
};

int main() {
    int N, boardW, boardH;
    cin >> N >> boardW >> boardH;

    vector<Item> items(N);
    for(int i = 0; i < N; i++) {
        cin >> items[i].w >> items[i].h;
        items[i].id = i;
    }

    // Keep original order for output
    vector<Item> sorted_items = items;
    sort(sorted_items.begin(), sorted_items.end(), [](const Item& a, const Item& b) {
        long long areaA = (long long)a.w * a.h;
        long long areaB = (long long)b.w * b.h;
        if (areaA != areaB) return areaA > areaB;
        return a.w > a.h;
    });

    vector<Placement> placements(N);
    vector<vector<tuple<int, int, int, int>>> boards; // list of occupied (x, y, w, h)

    auto try_place = [&](int idx, int board_idx) -> bool {
        Item& item = sorted_items[idx];
        int orig_id = item.id;
        
        vector<pair<int, int>> oris = {{item.w, item.h}, {item.h, item.w}};
        
        for(int rot = 0; rot < 2; rot++) {
            int ww = oris[rot].first;
            int hh = oris[rot].second;
            
            if (ww > boardW || hh > boardH) continue;
            
            // Bottom-left with fine scan
            int best_x = -1;
            int best_y = -1;
            int best_score = INT_MAX;
            
            // Step size adaptive to size
            int step = max(1, min(ww, hh) / 4);
            if (step < 1) step = 1;
            
            auto& occupied = boards[board_idx];
            
            for(int y = 0; y <= boardH - hh; y += step) {
                for(int x = 0; x <= boardW - ww; x += step) {
                    bool can_place = true;
                    
                    for(auto& occ : occupied) {
                        int ox, oy, ow, oh;
                        tie(ox, oy, ow, oh) = occ;
                        
                        if (x < ox + ow && x + ww > ox && 
                            y < oy + oh && y + hh > oy) {
                            can_place = false;
                            break;
                        }
                    }
                    
                    if (can_place) {
                        int score = y * 100000 + x; // prefer lower y then lower x
                        if (score < best_score) {
                            best_score = score;
                            best_x = x;
                            best_y = y;
                        }
                    }
                }
            }
            
            if (best_x != -1) {
                // Place it
                occupied.emplace_back(best_x, best_y, ww, hh);
                
                placements[orig_id].board_idx = board_idx;
                placements[orig_id].x = best_x;
                placements[orig_id].y = best_y;
                placements[orig_id].rotated = rot;
                
                return true;
            }
        }
        return false;
    };

    int current_board = -1;
    int placed_count = 0;

    for(int i = 0; i < N; i++) {
        bool placed = false;
        
        // Try existing boards
        for(int b = 0; b < boards.size(); b++) {
            if (try_place(i, b)) {
                placed = true;
                placed_count++;
                break;
            }
        }
        
        if (!placed) {
            // New board
            boards.emplace_back();
            current_board = boards.size() - 1;
            if (try_place(i, current_board)) {
                placed = true;
                placed_count++;
            }
        }
        
        if (!placed) {
            // Should not happen unless single item > board
            cerr << "Failed to place item " << i << endl;
            return 1;
        }
    }

    // Output
    cout << boards.size() << endl;
    
    for(int i = 0; i < N; i++) {
        cout << placements[i].board_idx << " "
             << placements[i].x << " "
             << placements[i].y << " "
             << placements[i].rotated << endl;
    }

    return 0;
}
