```python
import sys

def solve_exact_cover(U, S, sets):
    # Bitmasks for each set
    masks = [0] * S
    for i in range(S):
        for elem in sets[i]:
            if 0 <= elem < U:
                masks[i] |= (1 << elem)
    
    # DP: dp[state] = list of set indices that cover state exactly
    # Use a list of tuples or track previous choices
    dp = [None] * (1 << U)
    dp[0] = []
    
    for i in range(S):
        mask = masks[i]
        # Iterate backwards to avoid using same set twice
        for state in range((1 << U) - 1, -1, -1):
            if dp[state] is not None and (state & mask) == 0:
                new_state = state | mask
                if dp[new_state] is None:
                    dp[new_state] = dp[state] + [i]
    
    full = (1 << U) - 1
    if dp[full] is not None:
        solution = dp[full]
        return solution
    return None

def main():
    input_lines = sys.stdin.read().strip().split('\n')
    if not input_lines:
        print("NO SOLUTION")
        return
    
    # Parse header
    header = input_lines[0].strip().split()
    if len(header) < 2:
        print("NO SOLUTION")
        return
    
    try:
        U = int(header[0])
        S = int(header[1])
    except ValueError:
        print("NO SOLUTION")
        return
    
    sets = []
    for line in input_lines[1:]:
        line = line.strip()
        if not line:
            continue
        try:
            row = [int(x) for x in line.split()]
            sets.append(row)
        except ValueError:
            continue
    
    # Pad or trim if needed
    while len(sets) < S:
        sets.append([])
    sets = sets[:S]
    
    solution = solve_exact_cover(U, S, sets)
    
    if solution:
        print("SOLUTION")
        print(" ".join(map(str, solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
```

**Explanation of approach**: 
- We use bitmask DP where each `state` represents the covered elements.
- We iterate over sets and update DP table by adding a set only if it does not overlap with current state.
- Since we iterate over sets in order and add them sequentially, the solution list will be in ascending order.
- This works for typical small constraints (U ≤ 20-22).
