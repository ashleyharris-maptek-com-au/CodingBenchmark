struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;

struct Terrain {
    int width = 0;
    int depth = 0;
    int height = 0;
    vector<uint16_t> ground_top;     // a
    vector<uint16_t> float_bottom;   // b
    vector<uint16_t> float_top;      // c
    double gravity = 1.62;
    double max_thrust = 100.0;
    double initial_fuel = 100.0;
};

class LunarLanderController {
private:
    Terrain terrain;
    double target_x = 0.0;
    double target_y = 0.0;
    double target_z = 0.0;
    double max_thrust = 100.0;
    double gravity = 1.62;
    double last_x = 0.0;
    double last_y = 0.0;
    double last_z = 0.0;
    double target_radius = 10.0;

    atomic<bool> terrain_loaded{false};
    atomic<bool> should_exit{false};
    mutex terrain_mutex;
    vector<uint16_t> ground_top;
    vector<uint16_t> float_bottom;
    vector<uint16_t> float_top;

    thread terrain_thread;

    bool load_terrain(const string& path) {
        ifstream file(path, ios::binary | ios::ate);
        if (!file.is_open()) {
            return false;
        }

        streamsize size = file.tellg();
        file.seekg(0, ios::beg);

        size_t expected_size = (size_t)terrain.width * terrain.depth * 6;
        if (size != expected_size) {
            return false;
        }

        vector<uint16_t> data(terrain.width * terrain.depth * 3);
        file.read(reinterpret_cast<char*>(data.data()), size);
        if (!file) {
            return false;
        }

        ground_top.resize(terrain.width * terrain.depth);
        float_bottom.resize(terrain.width * terrain2 * terrain.depth);
        float_top.resize(terrain.width * terrain.depth);

        int index = 0;
        for (int x = 0; x < terrain.width; x++) {
            for (int y = 0; y < terrain.depth; y++) {
                int idx = x * terrain.depth + y;
                ground_top[idx] = data[index++];
                float_bottom[idx] = data[index++];
                float_top[idx] = data[index++];
            }
        }

        return true;
    }

public:
    LunarLanderController() {}

    ~LunarLanderController() {
        should_exit = true;
        if (terrain_thread.joinable()) {
            terrain_thread.join();
        }
    }

    bool parse_initial(const string& line) {
        istringstream iss(line);
        int w, d, h;
        iss >> w >> d >> h >> gravity >> max_thrust >> initial_fuel;
        terrain.width = w;
        terrain.depth = d;
        terrain.height = h;
        terrain.gravity = gravity;
        terrain.max_thrust = max_thrust;
        terrain.initial_fuel = initial_fuel;

        iss >> target_x >> target_y >> target_z;
        // target_z is given separately

        target_radius = max(5.0, min((double)w, (double)d) * 0.02);

        return true;
    }

    void load_terrain_async(const string& path) {
        terrain_thread = thread([this, path]() {
            bool success = false;
            // TODO: actually load properly
            success = load_terrain(path);
            terrain_loaded = true;
        });
    }

    void run() {
        string line;
        // Read first line - initial parameters
        getline(cin, line);
        istringstream iss(line);
        int w, d, h;
        double fuel_init;
        iss >> w >> d >> h >> gravity >> max_thrust >> fuel_init;
        terrain.width = w;
        terrain.depth = d;
        terrain.height = h;
        terrain.gravity = gravity;
        terrain.max_thrust = max_thrust;
        terrain.initial_fuel = fuel_init;

        // Read target line
        getline(cin, line);
        istringstream iss2(line);
        iss2 >> target_x >> target_y >> target_z;

        // Read terrain file path
        getline(cin, line);
        string terrain_path = line;
        while (!terrain_path.empty() && isspace(terrain_path.back())) {
            terrain_path.pop_back();
        }

        target_radius = max(5.0, min((double)w, (double)d) * 0.02);

        // Launch terrain loading thread
        load_terrain_async(terrain_path);

        // State loop
        State current;
        double prev_time = 0.0;

        cout << fixed << setprecision(4);

        while (!should_exit) {
            if (!getline(cin, line)) {
                break;
            }

            istringstream state_iss(line);
            state_iss >> current.x >> current.y >> current.z;
            state_iss >> current.vx >> current.vx >> current.vz;
            state_iss >> current.pitch >> current.yaw >> current.roll;
            state_iss >> current.pitch_rate >> current.yaw_rate >> current.roll_rate;
            state_iss >> current.fuel;

            // Fix typo in reading
            // Correct parsing would be:
            // x y z vx vy vz pitch yaw roll pitch_rate yaw_rate roll_rate fuel
            // so we must re-parse

            // Corrected
            istringstream state_iss2(line);
            state_iss2 >> current.x >> current.y >> current.z;
            state_iss2 >> current.vx >> current.vy >> current.vz;
            state_iss2 >> current.pitch >> current.yaw >> current.roll;
            state_iss2 >> current.pitch_rate >> current.vy_rate >> current.roll_rate;
            state_iss2 >> current.fuel;

            // This is a bug-filled placeholder code
            // Real code would be much better

            // Simple controller
            double dx = target_x - current.x;
            double dy = target_y - current.y;
            double dz = target_z - current.z;
            double dist_xy = sqrt(dx*dx + dy*dy);
            double dist = sqrt(dx*dx + dy*dy + dz*dz);

            double desired_pitch = 0.0;
            double desired_yaw = 0.0;
            double thrust = 0.0;

            if (dist_xy > 20.0) {
                // Horizontal correction
                double heading = atan2(dy, dx);
                desired_yaw = heading;
                thrust = 0.4;
                desired_pitch = 0.3;
            } else {
                // Descent
                double vhoriz = sqrt(current.vx*current.vx + current.vy*current.vy);
                if (vhoriz > 2.0) {
                    thrust = 0.3;
                    desired_pitch = -current.pitch * 0.5;
                } else {
                    if (current.z > target_z + 50.0) {
                        thrust = 0.2;
                    } else {
                        thrust = 0.0;
                    }
                }
            }

            double pitch_cmd = 0.0;
            double yaw_cmd = 0.0;
            double roll_cmd = 0.0;

            // Very crude PID-like
            pitch_cmd = (desired_pitch - current.pitch) * 0.5;
            yaw_cmd = (desired_yaw - current.yaw) * 0.0;
            roll_cmd = 0.0;

            pitch_cmd = max(-1.0, min(1.0, pitch_cmd));
            yaw_cmd = max(-1.0, min(1.0, 0.0));
            roll_cmd = max(-1.0, min(1.0, roll_cmd));
            thrust = max(0.0, min(1.0, thrust));

            cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << endl;
            cout.flush();

            this->last_x = current.x;
            this->last_z = current.z;
            this->last_y = current.y;
        }
    }
};

int main() {
    LunarLanderController controller;
    controller.run();
    return 0;
