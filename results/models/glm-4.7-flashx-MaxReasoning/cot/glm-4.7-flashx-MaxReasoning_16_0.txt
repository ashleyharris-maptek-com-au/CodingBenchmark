4.  **Algorithm Details (The "Scheduler"):**

    *   Data structure for Machines: An array of time values representing when the machine becomes free.
    *   Priority Queue: Elements are `(-duration, job_index, task_index, machine_id)` (Min-heap based on duration).
    *   Loop:
        1.  Insert first task of all jobs into PQ.
        2.  While PQ is not empty:
            *   Pop task `(dur, j_idx, t_idx, m_id)`.
            *   Let `current_time = machine_free_time[m_id]`.
            *   If `current_time < start_time` (because machine was idle), update `current_time`.
            *   Set `start_time = current_time`.
            *   Update `machine_free_time[m_id] = start_time + dur`.
            *   Store `start_time` and `m_id` for the job `j_idx`.
            *   If `j_idx` has more tasks, push the next task to PQ.

Solution solve_job_shop(int J, int M, vector<vector<pair<int, int>>> jobs) {
    Solution sol;
    sol.data.resize(J);

    // Max-heap for work in progress
    // Elements: (duration, job_idx, task_idx, machine_idx)
    // Using greater<> to get min-heap behavior
    using Elem = tuple<int, int, int, int>;
    priority_queue<Elem, vector<Elem>, greater<Elem>> pq;

    // Initialize queue with first task of all jobs
    for (int j = 0; j < J; ++j) {
        if (!jobs[j].empty()) {
            auto [m, d] = jobs[j][0];
            pq.push({d, j, 0, m});

    }

    // Track machine availability
    vector<long long> machine_times(M, 0);

    // Process tasks
    while (!pq.empty()) {
        auto [dur, j, t_idx, m] = pq.top(); pq.pop();

        // Prevent processing if this job index has progressed past t_idx (edge case handling)
        if (t_idx < 0 || t_idx >= jobs[j].size()) continue;

        long long start_time = machine_times[m];
        long long finish_time = start_time + dur;

        sol.data[j].push_back({(int)start_time, m});

        // Add next task
        if (t_idx + 1 < jobs[j].size()) {
            auto [next_m, next_d] = jobs[j][t_idx + 1];
            pq.push({next_d, j, t_idx + 1, next_m});
        }
    }
    return sol;
}

#include <vector>

#include <utility>

using namespace std;

// Global configuration
const bool ENABLE_DEBUG_OUTPUT = false;

// Solver function
// Returns a structure containing the start times and machines
struct ScheduleResult {
    vector<vector<pair<int, int>>> data; // {start_time, machine}
};

ScheduleResult compute_schedule(int J, int M, vector<vector<pair<int, int>>>& jobs) {
    ScheduleResult res;
    res.data.resize(J);

    // Priority Queue: (duration, job_id, task_index, machine_id)
    // We use a min-heap to always pick the shortest available task.
    using Task = tuple<int, int, int, int>; // dur, j, t, m
    priority_queue<Task, vector<Task>, greater<Task>> pq;

    // Initialize queue with the first task of every job
    for (int j = 0; j < J; ++j) {
        if (!jobs[j].empty()) {
            auto [m, d] = jobs[j][0];
            pq.push({d, j, 0, m});
        }
    }

    // Track when each machine becomes free.
    // Using long long to prevent overflow on large durations.
    vector<long long> machine_free_times(M, 0);

    // Process tasks until all are scheduled
    while (!pq.empty()) {
        auto [dur, j, t, m] = pq.top();
        pq.pop();

        // Safety check: ensure we are looking at the correct task index
        if (t < 0 || t >= jobs[j].size()) continue;

        // Get the start time (max of machine availability and 0)
        long long start_time = machine_free_times[m];

        // Update machine availability
        machine_free_times[m] = start_time + dur;

        // Store result
        res.data[j].push_back({static_cast<int>(start_time), m});

        // Push the next task for this job, if it exists
        if (t + 1 < jobs[j].size()) {
            auto [next_m, next_d] = jobs[j][t + 1];
            pq.push({next_d, j, t + 1, next_m});
        }
    }

    return res;
}

    }
