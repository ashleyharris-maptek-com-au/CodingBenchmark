fn count_reachable(grid: &[Vec<Cell>], x: usize, y: usize, visited: &mut [u32], width: usize) -> usize {
    let bit = 1u32 << x;
    
    if x >= width || grid[y][x] == Cell::Obstacle || (visited[y] & bit) != 0 {
        return 0;
    }
    
    visited[y] |= bit;
    let mut count = 1;
    
    count += count_reachable(grid, x + 1, y, visited, width);
    count += count_reachable(grid, x - 1, y, visited, width);
    count += count_reachable(grid, x, y + 1, visited, width);
    count += count_reachable(grid, x, y - 1, visited, width);
    
    count
}

fn find_hamiltonian_path(
    grid: &[Vec<Cell>],
    visited: &mut [u32],
    x: usize,
    y: usize,
    require_cycle: bool,
    total_cells: usize,
    width: usize,
) -> Option<Vec<(usize, usize)>> {
    let bit = 1u32 << x;
    visited[y] |= bit;
    let mut path = vec![(x, y)];
    
    if path.len() == total_cells {
        if require_cycle && !is_adjacent(x, y, 0, 0, width) {
            visited[y] &= !bit;
            path.pop();
            return None;

        }

    None
}
