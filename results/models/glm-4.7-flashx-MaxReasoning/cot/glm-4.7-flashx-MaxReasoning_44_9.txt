5.  **Final Code Structure:**

    ```glsl
    #version 450

    layout(location = 0) in vec3 worldPos;
    layout(location = 1) in vec3 normal;
    layout(location = 2) in vec2 uv;
    layout(location = 3) in vec3 tangent;
    layout(location = 4) in vec3 color;

    layout(set = 0, binding = 0) uniform UBO {
        mat4 model;
        mat4 view;
        mat4 projection;
        vec4 lightPos;
        vec4 cameraPos;
        vec4 params;
    } ubo;

    layout(location = 0) out vec4 outColor;

    void main() {
        // Normalize interpolated normal
        vec3 n = normalize(normal);

        // UV Checkerboard
        // 8x8 frequency
        float scale = 8.0;
        // Use sum of integer parts for parity (diagonal checkerboard)
        // Or product for orthogonal? "UV checkerboard" usually implies orthogonal.
        // Let's try: fract(uv.x * 8.0) < 0.5 ? mix(val1, val2, ...) ?
        // Actually, the simplest parity check for a checkerboard is floor(x) + floor(y).
        float fX = floor(uv.x * scale);
        float fY = floor(uv.y * scale);
        float parity = mod(fX + fY, 2.0);
        vec3 baseColor = mix(vec3(0.9), vec3(0.2), parity < 1.0);

        // Lambertian
        vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
        float diffuse = max(dot(n, lightDir), 0.0);
        float shadingFactor = 0.5 + 0.5 * diffuse;

        // Combine
        vec3 finalColor = baseColor * shadingFactor;

        outColor = vec4(finalColor, 1.0);
    }
    ```
