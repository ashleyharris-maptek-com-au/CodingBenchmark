struct State {

    int len;

class SuffixAutomaton {
public:
    vector<State> st;
    int last;

    SuffixAutomaton() {
        st.resize(1);
        st[0].len = 0;
        st[0].link = -1;
        last = 0;
        // Initialize next to 0
        for(int i=0; i<256; i++) st[0].next[i] = 0;
    }

    void extend(int c) {
        int cur = st.size();
        st.emplace_back();
        st[cur].len = st[last].len + 1;
        // Initialize next to 0
        for(int i=0; i<256; i++) st[cur].next[i] = 0;

        int p = last;
        while (p != -1 && !st[p].next[c]) {
            st[p].next[c] = cur;
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = st.size();
                st.emplace_back();
                st[clone].len = st[p].len + 1;
                for(int i=0; i<256; i++) st[clone].next[i] = st[q].next[i];
                st[clone].link = st[q].link;

                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }

    int getLCS(const string& s) {
        int v = 0;
        int l = 0;
        int best = 0;
        for (char c : s) {
            int ch = (unsigned char)c; // Handle potential issues with negative chars if any
            while (v && !st[v].next[ch]) {
                v = st[v].link;
                l = st[v].len;
            }
            if (st[v].next[ch]) {
                v = st[v].next[ch];
                l++;
            } else {
                v = 0;
                l = 0;
            }
            if (l > best) best = l;
        }
        return best;
    }
};
