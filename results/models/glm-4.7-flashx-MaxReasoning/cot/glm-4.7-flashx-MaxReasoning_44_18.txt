```glsl
#version 450

layout(location = 0) out vec4 outColor;

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

void main() {
    // Calculate View Vector (V)
    // V is direction from surface to camera
    vec3 viewDir = normalize(ubo.cameraPos.xyz - worldPos);
    
    // Normalize normal (ensure valid input)
    vec3 N = normalize(normal);
    
    // Calculate NdotV
    float NdotV = max(dot(N, viewDir), 0.0);
    
    vec3 heatmapColor = vec3(0.0);
    
    if (NdotV < 0.5) {
        // Segment 1: Blue to Green
        heatmapColor.r = 1.0;
        heatmapColor.g = 2.0 * NdotV;
        heatmapColor.b = 0.0;
    } else {
        // Segment 2: Green to Yellow
        float t = NdotV - 0.5;
        heatmapColor.r = 1.0 - 2.0 * t;
        heatmapColor.g = 1.0 - 2.0 * t;
        heatmapColor.b = 2.0 * t;
    }
    
    outColor = vec4(heatmapColor, 1.0);
}
```

    #version 450

    layout(location = 0) out vec4 outColor;

    layout(location = 0) in vec3 worldPos;
    layout(location = 1) in vec3 normal;
    layout(location = 2) in vec2 uv;
    layout(location = 3) in vec3 tangent;
    layout(location = 4) in vec3 color;

    layout(set = 0, binding = 0) uniform UBO {
        mat4 model;
        mat4 view;
        mat4 projection;
        vec4 lightPos;
        vec4 cameraPos;
        vec4 params;
    } ubo;

    void main() {
        // Calculate the View Vector (V) from the surface to the camera
        vec3 V = normalize(ubo.cameraPos.xyz - worldPos);

        // Ensure the normal is normalized
        vec3 N = normalize(normal);

        // Calculate NdotV: maximum of dot product and 0 (clamped to 0)
        float NdotV = max(dot(N, V), 0.0);

        vec3 resultColor = vec3(0.0);

        if (NdotV < 0.5) {
            // Segment 1: Red to Yellow
            resultColor.r = 1.0;
            resultColor.g = 2.0 * NdotV;
            resultColor.b = 0.0;
        } else {
            // Segment 2: Yellow to Blue
            float t = NdotV - 0.5;
            resultColor.r = 1.0 - (t * 2.0);
            resultColor.g = 1.0 - (t * 2.0);
            resultColor.b = t * 2.0;
        }

        outColor = vec4(resultColor, 1.0);
    }
