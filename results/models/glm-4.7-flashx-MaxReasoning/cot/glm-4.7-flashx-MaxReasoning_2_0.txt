1.  **Understand the Problem:** The goal is to solve the Chinese Postman Problem (CPP) in C#. The CPP asks for the shortest closed walk that traverses every edge at least once in a graph.

2.  **Analyze the Requirements:**
    *   **Input:** $N$ nodes, $M$ edges, 0-indexed. $M$ lines of $u, v, w$.
    *   **Output:** A sequence of nodes representing the route (starting and ending at 0) traversing every edge at least once. Optionally prefix with length $L$.
    *   **Scale:** Must handle 6 to 100,000 nodes and up to 1,000,000 edges.
    *   **Performance:** 30 seconds limit.
    *   **Environment:** .NET 5.3.0 (lang 14), 64-bit, 64 cores, 128GB RAM. SIMD instructions available (though C# JIT doesn't use AVX/AVX2 intrinsics for algorithmic logic automatically, we can use `Span`, `Memory`, or careful array management).

3.  **Algorithm Selection:**
    *   *Small Graphs (6-20 nodes):* An Eulerian circuit exists only if the graph is Eulerian (all nodes have even degree). If not, we need to pair odd-degree nodes. With $N \le 20$, we can check all permutations of odd nodes. Complexity: $O(N!)$ is too slow for naive checks. But with $N=20$, number of odd nodes $k$ is at most 20. Finding Minimum Weight Perfect Matching (MWPM) is the standard way. The Hopcroft-Karp algorithm is efficient. After MWPM, we double the shortest path between pairs.
    *   *Medium/Large/Extreme Graphs (35-100000 nodes):* The number of odd-degree nodes $k$ is usually small in sparse graphs (average degree is low). However, in dense graphs, $k$ can be large.
        *   If $k$ is small (e.g., $< 50$), we solve MWPM on the subgraph of odd nodes using Dijkstra (or Floyd-Warshall if $N$ is small, but Dijkstra is better here).
        *   If $k$ is large, MWPM becomes expensive. However, for CPP, the "Odd-Vertex Problem" (MWPM) is the bottleneck.
        *   The solution involves:
            1. Check Eulerian. If yes, find Eulerian circuit.
            2. If not, find all odd-degree vertices. Let $k$ be the count.
            3. Compute shortest path distances between all pairs of odd vertices.
            4. Solve Minimum Weight Perfect Matching on these $k$ vertices.
            5. For each matched pair $(u, v)$, add the shortest path from $u$ to $v$ to the edge list (duplicate edges).
            6. Find Eulerian circuit in the modified graph.

        List<(long d, int v)> data;
        public MinHeap() { data = new List<(long d, int v)>(); }
        public int Count => data.Count;
        public void Push(long d, int v) {
            data.Add((d, v));
            SiftUp(data.Count - 1);
        }
        public (long d, int v) Pop() {
            var res = data[0];
            int last = data.Count - 1;
            data[0] = data[last];
            data.RemoveAt(last);
            if (data.Count > 0) SiftDown(0);
            return res;
        }
        void SiftUp(int i) {
            while (i > 0) {
                int p = (i - 1) / 2;
                if (data[i].d >= data[p].d) break;
                var tmp = data[i]; data[i] = data[p]; data[p] = tmp;
                i = p;
            }
        }
        void SiftDown(int i) {
            while (true) {
                int l = 2 * i + 1, r = 2 * i + 2;
                int smallest = i;
                if (l < data.Count && data[l].d < data[smallest].d) smallest = l;
                if (r < data.Count && data[r].d < data[smallest].d) smallest = r;
                if (smallest == i) break;
                var tmp = data[i]; data[i] = data[smallest]; data[smallest] = tmp;
                i = smallest;
            }
        }

        // Greedy Heuristic (Repeated Nearest Neighbor)
        // Scales well to large k
        List<(int a, int b)> pairs = new List<(int, int)>(k / 2);
        bool[] used = new bool[k];
        
        for (int i = 0; i < k; i++) {
            if (used[i]) continue;
            
            int minJ = -1;
            long minW = long.MaxValue;
            
            for (int j = 0; j < k; j++) {
                if (i == j || used[j]) continue;
                if (dist[i, j] < minW) {
                    minW = dist[i, j];
                    minJ = j;
                }
            }
            
            if (minJ != -1) {
                used[i] = true;
                used[minJ] = true;
                pairs.Add((i, minJ));
            }
        }
        
        // Return pairs list
        return pairs;
    }

    static List<int> EulerCircuit(List<(int to, long w)>[] adj) {
        // Hierholzer's Algorithm
        List<int> circuit = new List<int>();
        Stack<int> stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0) {
            int u = stack.Peek();
            if (adj[u].Count == 0) {
                stack.Pop();
                circuit.Add(u);
            } else {
                var e = adj[u].Pop();
                stack.Push(e.to);
            }
        }
        return circuit;
    }
}
