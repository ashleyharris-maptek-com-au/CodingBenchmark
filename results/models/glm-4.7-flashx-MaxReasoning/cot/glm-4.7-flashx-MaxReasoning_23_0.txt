struct LanderState {
    x: f64, y: f64,
    vx: f64, vy: f64,
    angle: f64, angular_vel: f64,
    fuel: f64,
    dt: f64,
}

struct Control {
    thrust: f64,
    turn: f64,
}

fn calculate_controls(state: &LanderState, sx: f64, sy: f64, tx: f64, ty: f64, gravity: f64, max_thrust: f64) -> Control {
    // 1. Orientation
    // We want to point to the target.
    let dx = tx - state.x;
    let dy = ty - state.y;
    
    // If close, we might want to face down (90 deg) to engage braking.
    let dist = (dx*dx + dy*dy).sqrt();
    
    let target_angle = if dist < 10.0 {
        std::f64::consts::PI / 2.0 // Face down
    } else {
        (dy as f64).atan2(dx as f64) // Point to target
    };

    let mut turn = target_angle - state.angle;
    // Normalize turn to -PI..PI
    while turn < -std::f64::consts::PI { turn += 2.0 * std::f64::consts::PI; }
    while turn > std::f64::consts::PI { turn -= 2.0 * std::f64::consts::PI; }

    // Clamp turn
    let turn_rate = turn.max(-1.0).min(1.0);

    // 2. Thrust
    // Vertical control: Try to stop moving down (vy > 0).
    // Target vertical velocity: 0.
    // Error: -vy.
    // We want to counteract gravity and velocity.
    // Required acceleration to hover: gravity.
    // Required acceleration to stop at 0: gravity + vy.
    // Clamp to max_thrust.
    
    let desired_vy = 0.0;
    let v_error = desired_vy - state.vy; // If falling (+vy), error is negative.
    
    // PD controller for vertical
    // k_p = 0.5
    let kp = 0.5;
    let thrust = v_error * kp;
    
    // Clamp
    let thrust = thrust.max(0.0).min(1.0);
    
    // Extra safety: If velocity is high, max out thrust
    if state.vy > 5.0 {
        let thrust = (state.vy - 2.0).max(0.0).min(1.0);
    }

    Control { thrust, turn: turn_rate }
}

fn physics_step(state: &mut LanderState, thrust: f64, turn_rate: f64, gravity: f64, dt: f64) {
    // Update Angle
    state.angle += turn_rate * dt;
    
    // Angular friction? Prompt says angular_vel is input. No damping mentioned.
    // But output turn is max 1.0. We assume the environment applies the turn.
    state.angular_vel = turn_rate; // Assuming we can control it directly or it acts instantly.
    // Actually, angular_vel is part of input. We apply turn_rate to change it.
    
    // Physics
    let ax = thrust * state.angle.cos();
    let ay = thrust * state.angle.sin();

    state.vx += ax * dt;
    state.vy += (ay + gravity) * dt;
    
    state.x += state.vx * dt;
    state.y += state.vy * dt;
    
    state.fuel -= thrust * dt; // Burn fuel
}

use std::io::{self, BufRead, Write, BufWriter};
use std::time::{Instant, Duration};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    // Use iterator for efficient line reading
    let mut lines = stdin.lock().lines();

    // Read Map Dimensions
    let dims: Vec<f64> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();
    
    // Map dimensions, Gravity, Max Thrust, Initial Fuel
    // We don't need width/height for logic, just to skip map.
    let _width = dims[0] as usize;
    let _height = dims[1] as usize;
    let gravity = dims[2];
    let _max_thrust = dims[3];
    let _initial_fuel = dims[4];

    // Read Start and Target
    let pos: Vec<f64> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();
    
    let start_x = pos[0];
    let start_y = pos[1];
    let target_x = pos[2];
    let target_y = pos[3];

    // Read Map Rows (Discard to save memory)
    // The map is 1m x 1m, so width lines.
    for _ in 0.._width {
        lines.next().unwrap_or(Ok(String::new())).unwrap();
    }

    // Physics State
    // x y vx vy angle angular_vel fuel
    let mut state: Vec<f64> = vec![0.0; 7];
    let dt = 0.1; // Physics update interval

    // Timing
    let interval = Duration::from_millis(50); // 20Hz output
    let mut last_output = Instant::now();

    for line in lines {
        let line = line.unwrap();
        let data: Vec<f64> = line.split_whitespace()
            .map(|x| x.parse().unwrap())
            .collect();

        if data.len() < 7 {
            continue; // Should not happen based on format
        }

        state = data; // Update state

        // Time management
        let now = Instant::now();
        let elapsed = now.duration_since(last_output);
        if elapsed < interval {
            std::thread::sleep(interval - elapsed);
        }

        // Control Logic
        let (thrust, turn) = compute_controls(
            state[0], state[1], state[2], state[3], state[4], state[5],
            start_x, start_y, target_x, target_y, gravity
        );

        // Output
        writeln!(out, "{} {}", thrust, turn).unwrap();
        out.flush().unwrap();

        last_output = Instant::now();

        // Physics Update for next step
        // x += vx * dt
        // y += vy * dt
        // vx += thrust_x * dt
        // vy += (gravity + thrust_y) * dt
        // angle += angular_vel * dt
        // fuel -= thrust * dt
        
        let x = state[0];
        let y = state[1];
        let vx = state[2];
        let vy = state[3];
        let angle = state[4];
        let angular_vel = state[5];
        let fuel = state[6];

        let new_vx = vx + (thrust * angle.cos()) * dt;
        let new_vy = vy + (gravity + thrust * angle.sin()) * dt;
        let new_x = x + new_vx * dt;
        let new_y = y + new_vy * dt;
        
        // Fuel consumption: max_thrust is 1.0, we output 0-1. 
        // "100% engine burn burns 1 unit of fuel per second"
        // Our dt is 0.1. So if thrust is 1.0, we burn 0.1.
        let new_fuel = fuel - (thrust * dt).max(0.0);

        // Update state vector (we reuse the vec for simplicity in this logic, 
        // though strictly we should copy to avoid aliasing issues in a real engine, 
        // but here we overwrite the whole thing).
        state[0] = new_x;
        state[1] = new_y;
        state[2] = new_vx;
        state[3] = new_vy;
        state[4] = angle + angular_vel * dt; // Simplify angular step
        state[5] = angular_vel; // Angular velocity might be clamped in env
        state[6] = new_fuel;
    }
}

fn compute_controls(
    x: f64, y: f64, vx: f64, vy: f64, angle: f64, angular_vel: f64,
    sx: f64, sy: f64, tx: f64, ty: f64, gravity: f64
) -> (f64, f64) {
    
    // 1. Horizontal Position and Angle Control
    // We want to face the target.
    // dx, dy relative to target
    let dx = tx - x;
    let dy = ty - y;
    let dist = (dx*dx + dy*dy).sqrt();
    
    let target_angle;
    // If we are very close, try to face DOWN (90 deg / PI/2) to prepare for landing.
    // If we are far, point at target.
    if dist < 20.0 {
        target_angle = std::f64::consts::PI / 2.0;
    } else {
        target_angle = dy.atan2(dx);
    }
    
    // Turn towards target_angle
    let mut turn = target_angle - angle;
    // Normalize to [-PI, PI]
    while turn <= -std::f64::consts::PI { turn += 2.0 * std::f64::consts::PI; }
    while turn > std::f64::consts::PI { turn -= 2.0 * std::f64::consts::PI; }
    
    // Clamp turn rate to [-1, 1]
    let turn_rate = turn.max(-1.0).min(1.0);

    // 2. Vertical Control
    // Desired vertical velocity: 0 (to land softly).
    // PD Controller: Error = Desired - Current.
    let desired_vy = 0.0;
    let v_error = desired_vy - vy;
    
    // Proportional gain
    let kp = 0.5;
    let mut thrust = v_error * kp;
    
    // Clamp
    thrust = thrust.max(0.0).min(1.0);
    
    // Safety override: If falling very fast, max thrust
    if vy > 5.0 {
        thrust = 1.0;
    }
    // Safety override: If rising very fast, no thrust
    if vy < -5.0 {
        thrust = 0.0;
    }

    (thrust, turn_rate)
}
