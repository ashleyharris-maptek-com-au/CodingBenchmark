    ```hlsl
    struct PSInput {
        [[vk::location(0)]] float3 worldPos : TEXCOORD0;
        [[vk::location(1)]] float3 normal   : TEXCOORD1;
        [[vk::location(2)]] float2 uv       : TEXCOORD2;
        [[vk::location(3)]] float3 tangent  : TEXCOORD3;
        [[vk::location(4)]] float3 color    : TEXCOORD4;
    };

    [[vk::binding(0, 0)]]
    cbuffer UBO : register(b0) {
        float4x4 model;
        float4x4 view;
        float4x4 projection;
        float4 lightPos;
        float4 cameraPos;
        float4 params;
    };

    float4 main(PSInput input) : SV_Target0 {
        // 1. Stripe Logic
        // sin(uv.y * 3.14159 * 16.0) oscillates between -1 and 1.
        // We want alternating red and blue.
        // floor(sin(...)) will be 0 in positive half, -1 in negative half.
        float sinVal = sin(input.uv.y * 3.14159 * 16.0);
        float isRed = step(0.0, floor(sinVal)); // Returns 1 if floor >= 0 (Red), 0 if < 0 (Blue)
        float3 stripeColor = lerp(float3(0.1, 0.1, 0.9), float3(0.9, 0.1, 0.1), isRed);

        // 2. Lighting
        float3 N = normalize(input.normal);
        float3 L = normalize(lightPos.xyz - input.worldPos);
        float diffuse = max(0.0, dot(N, L));
        float lightingFactor = 0.4 + 0.6 * diffuse;

        // 3. Final Color
        float4 finalColor;
        finalColor.rgb = stripeColor * lightingFactor;
        finalColor.a = 1.0;

        return finalColor;
    }
    ```
