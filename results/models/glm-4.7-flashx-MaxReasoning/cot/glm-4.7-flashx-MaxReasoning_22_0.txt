        for (int x = 0; x < width; x++)

                if (board[x, y] == 1)

                    colHeights[x] = height - y;
                    break;
                }
            }
        }

        double aggregateHeight = colHeights.Sum();
        
        // Calculate bumpiness
        double bumpiness = 0;
        for (int x = 0; x < width - 1; x++)

            bumpiness += Math.Abs((double)colHeights[x] - colHeights[x + 1]);
        }

                if (board[x, y] == 1)

                }
                else if (foundFilled)
                {

                }
            }
        }

        int completeLines = 0;
        for (int y = 0; y < height; y++)
        {

                {
                    full = false;
                    break;
                }
            }

        double wells = 0;
        for (int x = 1; x < width - 1; x++)
        {
            if (colHeights[x] < colHeights[x - 1] && colHeights[x] < colHeights[x + 1])
            {
                wells += Math.Min(colHeights[x - 1], colHeights[x + 1]) - colHeights[x];
            }
        }
        
        // Calculate maximum height
        int maxHeight = colHeights.Max();

        return WEI_HEIGHT[0] * aggregateHeight +
               WEI_HEIGHT[1] * bumpiness +
               WEI_HEIGHT[2] * holes +
               WEI_HEIGHT[3] * wells +
               WEI_HEIGHT[4] * maxHeight +
               WEI_HEIGHT[5] * completeLines +
               WEI_HEIGHT[6] * (double)holes / (aggregateHeight + 1);

    static void RemoveLine(int[,] board, int width, int height)
    {
        int maxColHeight = 0;
        
        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < height; y++)
            {
                if (board[x, y] == 1)
                    maxColHeight = Math.Max(maxColHeight, y + 1);
            }
        }
        
        // Find and clear a full line (highest one first)
        int lineToClear = -1;
        for (int y = maxColHeight - 1; y >= 0; y--)
        {
            bool full = true;
            for (int x = 0; x < width; x++)
            {
                if (board[x, y] == 0)
                {
                    full = false;
                    break;
                }
            }
            
            if (full)
            {
                lineToClear = y;
                break;
            }
        }
        
        if (lineToClear >= 0)
        {
            // Shift everything down
            for (int y = lineToClear; y < maxColHeight - 1; y++)
            {
                for (int x = 0; x < width; x++)
                    board[x, y] = board[x, y + 1];
            }
            
            // Clear the top line
            for (int x = 0; x < width; x++)
                board[x, maxColHeight - 1] = 0;
        }
    }
    
    static double EvaluateBoard(int[,] board, int width, int height)
    {
        // Calculate column heights
        int[] colHeights = new int[width];
        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < height; y++)
            {
                if (board[x, y] == 1)
                {
                    colHeights[x] = height - y;
                    break;
                }
            }
        }
        
        // Aggregate height
        double aggregateHeight = colHeights.Sum();
        
        // Bumpiness
        double bumpiness = 0;
        for (int x = 0; x < width - 1; x++)
            bumpiness += Math.Abs(colHeights[x] - colHeights[x + 1]);
        
        // Holes
        int holes = 0;
        for (int x = 0; x < width; x++)
        {
            bool foundFilled = false;
            for (int y = 0; y < height; y++)
            {
                if (board[x, y] == 1)
                    foundFilled = true;
                else if (foundFilled)
                    holes++;
            }
        }
        
        // Complete lines
        int completeLines = 0;
        for (int y = 0; y < height; y++)
        {
            bool full = true;
            for (int x = 0; x < width; x++)
            {
                if (board[x, y] == 0)
                {
                    full = false;
                    break;
                }
            }
            
            if (full)
                completeLines++;
        }
        
        // Wells
        double wells = 0;
        for (int x = 1; x < width - 1; x++)
        {
            if (colHeights[x] < colHeights[x - 1] && colHeights[x] < colHeights[x + 1])
            {
                wells += Math.Min(colHeights[x - 1], colHeights[x + 1]) - colHeights[x];
            }
        }
        
        // Maximum height
        int maxHeight = colHeights.Max();
        
        // Average height
        double avgHeight = aggregateHeight / width;
        
        return HEIGHT_WEIGHT[0] * aggregateHeight +
               HEIGHT_WEIGHT[1] * bumpiness +
               HEIGHT_WEIGHT[2] * holes +
               HEIGHT_WEIGHT[3] * wells +
               HEIGHT_WEIGHT[4] * maxHeight +
               HEIGHT_WEIGHT[5] * completeLines +
               HEIGHT_WEIGHT[6] * avgHeight;
    }
