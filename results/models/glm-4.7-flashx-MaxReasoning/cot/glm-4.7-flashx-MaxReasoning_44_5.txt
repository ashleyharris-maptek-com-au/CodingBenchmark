3.  **Refining the Code:**

    ```glsl
    #version 450

    layout(location = 0) in vec3 worldPos;
    layout(location = 1) in vec3 normal;
    layout(location = 2) in vec2 uv;
    layout(location = 3) in vec3 tangent;
    layout(location = 4) in vec3 color;

    layout(set = 0, binding = 0) uniform UBO {
        mat4 model;
        mat4 view;
        mat4 projection;
        vec4 lightPos;
        vec4 cameraPos;
        vec4 params;
    } ubo;

    layout(location = 0) out vec4 outColor;

    void main() {
        // Vectors
        vec3 N = normalize(normal);
        vec3 L = normalize(ubo.lightPos.xyz - worldPos);
        vec3 V = normalize(ubo.cameraPos.xyz - worldPos);

        // Phong
        float NdotL = max(dot(N, L), 0.0);
        vec3 R = reflect(-L, N);
        float RdotV = max(dot(R, V), 0.0);
        float specular = pow(RdotV, 32.0);
        float diffuse = NdotL;

        // Intensity
        float intensity = 0.2 + 0.5 * diffuse + 0.8 * specular;

        // Output (grayscale + alpha 1.0)
        outColor = vec4(intensity, intensity, intensity, 1.0);
    }
    ```
