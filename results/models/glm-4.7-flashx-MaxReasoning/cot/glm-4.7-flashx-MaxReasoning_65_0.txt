interface Task {
    id: string | number;
    duration: number;
    deps: Array<string | number>;
    deadline: number;
    weight: number;
}

// Since TS 6.0, standard library might be less needed, but interfaces are good.
// I'll stick to simple declarations for the node environment.

declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount: number = data.machineCount;
const tasks: Task[] = data.tasks;

// Build structures
const adj: Map<string|number, Set<string|number>> = new Map();
const inDegree: Map<string|number, number> = new Map();
const taskMap: Map<string|number, Task> = new Map();
const ids: string|number[] = [];

for (const t of tasks) {
    taskMap.set(t.id, t);
    inDegree.set(t.id, 0);
    adj.set(t.id, new Set());
    ids.push(t.id);
}

for (const t of tasks) {
    for (const depId of t.deps) {
        adj.get(depId)!.add(t.id);
        inDegree.set(t.id, inDegree.get(t.id)! + 1);
    }
}

// Topological Sort with Priority Queue (Min Heap by Deadline)
interface PQNode {
    id: string|number;
    deadline: number;
}
// Simple Min-Heap implementation
const heap: PQNode[] = [];

const addToHeap = (id: string|number) => {
    heap.push({ id, deadline: taskMap.get(id)!.deadline });
    // Sift up
    let idx = heap.length - 1;
    while (idx > 0) {
        const parent = Math.floor((idx - 1) / 2);
        if (heap[parent].deadline <= heap[idx].deadline) break;
        [heap[parent], heap[idx]] = [heap[idx], heap[parent]];
        idx = parent;
    }
};

const popHeap = (): PQNode | null => {
    if (heap.length === 0) return null;
    const min = heap[0];
    const end = heap.pop()!;
    if (heap.length > 0) {
        heap[0] = end;
        // Sift down
        let idx = 0;
        const length = heap.length;
        while (true) {
            const left = 2 * idx + 1;
            const right = 2 * idx + 2;
            let swapIdx = -1;
            if (left < length && heap[left].deadline < heap[idx].deadline) swapIdx = left;
            if (right < length && (swapIdx === -1 || heap[right].deadline < heap[left].deadline)) swapIdx = right;
            if (swapIdx === -1) break;
            [heap[idx], heap[swapIdx]] = [heap[swapIdx], heap[idx]];
            idx = swapIdx;
        }
    }
    return min;
};

const order: (string|number)[] = [];
for (const id of ids) {
    if (inDegree.get(id) === 0) {
        addToHeap(id);
    }
}

while (heap.length > 0) {
    const u = popHeap()!;
    order.push(u.id);
    for (const v of adj.get(u.id)!) {
        inDegree.set(v, inDegree.get(v)! - 1);
        if (inDegree.get(v) === 0) {
            addToHeap(v);
        }
    }
}

// Schedule Generation (List Scheduling)
// Strategy: Earliest Finish Time (EFT) for placement
const machineFinishTimes = new Array<number>(machineCount).fill(0);
const schedule: Array<{id: string|number, start: number, machine: number}> = [];

// We need to ensure we handle dependencies correctly in the schedule generation?
// The topological sort guarantees dependencies are satisfied in the sequence.
// The schedule generation assumes tasks run back-to-back on machines.
// Since we selected tasks respecting dependencies, they can be scheduled in that order.

for (const taskId of order) {
    const task = taskMap.get(taskId)!;
    
    // Find machine with earliest finish time
    let minIndex = 0;
    for (let i = 1; i < machineCount; i++) {
        if (machineFinishTimes[i] < machineFinishTimes[minIndex]) {
            minIndex = i;
        }
    }

    const start = machineFinishTimes[minIndex];
    const finish = start + task.duration;
    machineFinishTimes[minIndex] = finish;

    schedule.push({
        id: taskId,
        start: start,
        machine: minIndex
    });
}

// Output
console.log(JSON.stringify({ schedule: schedule }));

const machineCount: number = data.machineCount;
const tasks = data.tasks;

// We need to map IDs to indices or keep them as keys. Keeping as keys (string/number) is safer for dynamic IDs.
interface Task {
    id: string | number;
    duration: number;
    deps: Array<string | number>;
    deadline: number;
    weight: number;
}

const taskMap: Map<string | number, Task> = new Map();
const inDegree: Map<string | number, number> = new Map();
const adj: Map<string | number, Set<string | number>> = new Map();

for (const t of tasks) {
    taskMap.set(t.id, t);
    inDegree.set(t.id, 0);
    adj.set(t.id, new Set<string | number>());
}

    for (const dep of t.deps) {
        adj.get(dep)!.add(t.id);
        inDegree.set(t.id, inDegree.get(t.id)! + 1);
    }
}

// Min-Heap for Topological Sort (by Deadline)
interface HeapNode {
    id: string | number;
    deadline: number;
}
const heap: HeapNode[] = [];

const push = (node: HeapNode) => {
    heap.push(node);
    let idx = heap.length - 1;
    while (idx > 0) {
        const parent = Math.floor((idx - 1) / 2);
        if (heap[parent].deadline <= heap[idx].deadline) break;
        [heap[parent], heap[idx]] = [heap[idx], heap[parent]];
        idx = parent;
    }
};

const pop = (): HeapNode | null => {
    if (heap.length === 0) return null;
    const min = heap[0];
    const end = heap.pop()!;
    if (heap.length > 0) {
        heap[0] = end;
        let idx = 0;
        while (true) {
            const left = 2 * idx + 1;
            const right = 2 * idx + 2;
            let swap = -1;
            if (left < heap.length && heap[left].deadline < heap[idx].deadline) swap = left;
            if (right < heap.length && heap[right].deadline < heap[idx].deadline && (swap === -1 || heap[right].deadline < heap[left].deadline)) swap = right;
            if (swap === -1) break;
            [heap[idx], heap[swap]] = [heap[swap], heap[idx]];
            idx = swap;
        }
    }
    return min;
};

// Topological Sort
const order: (string | number)[] = [];

// Initialize heap with 0 in-degree nodes
for (const [id, deg] of inDegree) {
    if (deg === 0) {
        push({ id, deadline: taskMap.get(id)!.deadline });
    }
}

while (heap.length > 0) {
    const u = pop()!;
    order.push(u.id);
    const uId = u.id;
    const uTask = taskMap.get(uId)!;
    
    for (const vId of adj.get(uId)!) {
        inDegree.set(vId, inDegree.get(vId)! - 1);
        if (inDegree.get(vId) === 0) {
            push({ id: vId, deadline: taskMap.get(vId)!.deadline });
        }
    }
}

// Schedule Generation
// List Scheduling: Earliest Finish Time (EFT)
const machineFinishes = new Array<number>(machineCount).fill(0);
const schedule = [];

for (const taskId of order) {
    const task = taskMap.get(taskId)!;
    
    // Find machine with smallest finish time
    let minMachine = 0;
    for (let i = 1; i < machineCount; i++) {
        if (machineFinishes[i] < machineFinishes[minMachine]) {
            minMachine = i;
        }
    }
    
    const start = machineFinishes[minMachine];
    machineFinishes[minMachine] = start + task.duration;
    
    schedule.push({
        id: taskId,
        start: start,
        machine: minMachine
    });
}

    if (heap.length === 0) return null;
    const minNode = heap[0];
    const endNode = heap.pop()!;
    if (heap.length > 0) {
        heap[0] = endNode;
        let idx = 0;
        const len = heap.length;

If Late finishes, Early finishes late?

Earliest first: Finishes 100. Lateness 0. Good.
So EDF ensures tasks with *small deadlines* are scheduled earlier, which is the primary defense against lateness.
