3.  **Drafting the Code:**

    ```hlsl
    struct PSInput {
        [[vk::location(0)]] float3 worldPos : TEXCOORD0;
        [[vk::location(1)]] float3 normal   : TEXCOORD1;
        [[vk::location(2)]] float2 uv       : TEXCOORD2;
        [[vk::location(3)]] float3 tangent  : TEXCOORD3;
        [[vk::location(4)]] float3 color    : TEXCOORD4;
    };

    [[vk::binding(0, 0)]]
    cbuffer UBO : register(b0) {
        float4x4 model;
        float4x4 view;
        float4x4 projection;
        float4 lightPos;
        float4 cameraPos;
        float4 params;
    };

    float4 main(PSInput input) : SV_Target0 {
        // 1. Normalize inputs
        float3 N = normalize(input.normal);
        float3 V = normalize(cameraPos.xyz - worldPos.xyz);

        // 2. Light direction L
        // Formula: max(dot(N, L), 0) where L points towards the light source.
        float3 L = normalize(lightPos.xyz - input.worldPos.xyz);

        // 3. Half vector H
        float3 H = normalize(L + V);

        // 4. Blinn-Phong calculations
        float diffuse = max(dot(N, L), 0.0);
        float specular = pow(max(dot(N, H), 0.0), 64.0);

        // 5. Final Intensity
        // Formula: 0.1 + 0.6 * diffuse + 0.8 * specular
        float intensity = 0.1 + 0.6 * diffuse + 0.8 * specular;

        // 6. Output
        return float4(intensity, intensity, intensity, 1.0);
    }
    ```
