        [[vk::location(0)]] float3 worldPos : TEXCOORD0;
        [[vk::location(1)]] float3 normal   : TEXCOORD1;
        [[vk::location(2)]] float2 uv       : TEXCOORD2;

    float4 main(PSInput input) : SV_Target0 {
        // 1. Map UV to Mandelbrot coordinates
        // c = (uv.x*3-2, uv.y*2-1)

        float cy = input.uv.y * 2.0 - 1.0;

        // 2. Iterate Mandelbrot
        float zx = 0.0;
        float zy = 0.0;
        float iter = 0.0;
        const int maxIter = 20;

        while (iter < maxIter) {
            // z = z^2 + c
            // (x + yi)^2 = x^2 - y^2 + 2xyi
            float xtemp = zx * zx - zy * zy + cx;
            zy = 2.0 * zx * zy + cy;
            zx = xtemp;

            // Check escape |z|^2 > 4
            if ((zx * zx + zy * zy) > 4.0) {
                break;

            iter++;
        }

        // 3. Color
        // t = iter / 20
        float t = iter / 20.0;

        // Color = (t, t*0.5, 1.0-t)
        float3 color = float3(t, t * 0.5, 1.0 - t);

        return float4(color, 1.0);
    }
