5.  **Drafting the Code:**

    ```hlsl
    struct PSInput {
        [[vk::location(0)]] float3 worldPos : TEXCOORD0;
        [[vk::location(1)]] float3 normal   : TEXCOORD1;
        [[vk::location(2)]] float2 uv       : TEXCOORD2;
        [[vk::location(3)]] float3 tangent  : TEXCOORD3;
        [[vk::location(4)]] float3 color    : TEXCOORD4;
    };

    [[vk::binding(0, 0)]]
    cbuffer UBO : register(b0) {
        float4x4 model;
        float4x4 view;
        float4x4 projection;
        float4 lightPos;
        float4 cameraPos;
        float4 params;
    };

    float4 main(PSInput input) : SV_Target0 {
        // 1. Normalization
        float3 N = normalize(input.normal);

        // 2. Light Direction
        float3 L = normalize(lightPos.xyz - input.worldPos);
        float NdotL = dot(N, L);

        // 3. Brick Pattern Logic
        // Scale 8x16
        const float width = 8.0;
        const float height = 16.0;
        float u_step = 1.0 / width;
        float v_step = 1.0 / height;

        // Determine row and col
        float col = floor(input.uv.x / u_step);
        float row = floor(input.uv.y / v_step);

        // Local coordinates within the cell
        float u_local = frac(input.uv.x / u_step) * u_step;
        float v_local = frac(input.uv.y / v_step) * v_step;

        // Stagger rows (half offset every other row)
        if (mod(row, 2.0) > 0.5) {
            u_local = (u_local + u_step * 0.5) % u_step;
        }

        // Mortar thickness
        float t_x = 0.05;
        float t_y = 0.1;

        // Check for mortar
        // Since u_step = 0.125 and t_x = 0.05, u_local < 0.05 is valid.
        // Since v_step = 0.0625 and t_y = 0.1, v_local < 0.1 is ALWAYS true (since v_local max is 0.0625).
        // This results in a solid mortar color strip every row.
        bool isMortarX = u_local < t_x || u_local > (u_step - t_x);
        bool isMortarY = v_local < t_y || v_local > (v_step - t_y);

        // Colors
        float3 brickColor = float3(0.7, 0.2, 0.1);
        float3 mortarColor = float3(0.7, 0.7, 0.7);

        float3 finalColor = brickColor;
        if (isMortarX || isMortarY) {
            finalColor = mortarColor;
        }

        // 4. Lighting
        // Apply Lambertian shading with factor (0.4 + 0.6*diffuse)
        // Here "diffuse" likely refers to the NdotL term.
        float intensity = 0.4 + 0.6 * NdotL;
        finalColor *= intensity;

        return float4(finalColor, 1.0);
    }
    ```
