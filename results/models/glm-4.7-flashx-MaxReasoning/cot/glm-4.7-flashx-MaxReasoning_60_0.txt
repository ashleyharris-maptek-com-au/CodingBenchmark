    const fs = require('fs');
    const input = fs.readFileSync(0, 'utf8');
    const { machineCount, tasks } = JSON.parse(input);

    // Map tasks for easy access
    const taskMap = {};
    const inDegree = {};
    // Initialize inDegree and map
    tasks.forEach(t => {
        taskMap[t.id] = { ...t, successors: [] }; // clone to avoid mutation
        inDegree[t.id] = t.deps ? t.deps.length : 0;
    });

    // Build adjacency list for successors
    tasks.forEach(t => {
        if (t.deps) {
            t.deps.forEach(depId => {
                taskMap[depId].successors.push(t.id);
            });
        }
    });

    // Machines
    const machines = Array.from({ length: machineCount }, (_, i) => ({ id: i, finish: 0 }));
    const ready = [];
    const schedule = [];

    // Initial Ready Tasks
    for (const [id, task] of Object.entries(taskMap)) {
        if (inDegree[id] === 0) {
            ready.push(task);
        }
    }

    // Sort ready tasks by deadline (Earliest Deadline First)
    // This is a standard heuristic for minimizing lateness
    ready.sort((a, b) => a.deadline - b.deadline);

    // Main Loop
    while (ready.length > 0) {
        // Select task (Tie-breaker: Heavier tasks or something? EDD is good)
        const task = ready.shift(); 

        // Find machine with earliest finish time
        const machine = machines.reduce((prev, curr) => prev.finish < curr.finish ? prev : curr);

        const start = machine.finish;
        const end = start + task.duration;
        
        // Record
        schedule.push({
            id: task.id,
            start: start,
            machine: machine.id
        });

        // Update machine
        machine.finish = end;

        // Update dependencies for successors
        for (const succId of task.successors) {
            inDegree[succId]--;
            if (inDegree[succId] === 0) {
                ready.push(taskMap[succId]);
            }
        }

        // Re-sort ready queue occasionally or every time?
        // Re-sorting every time is $O(N \log N)$ per task. Total $O(N^2 \log N)$.
        // With high parallelism, $N$ could be large.
        // Optimize: Just sort at the start of the loop?
        // Actually, since we add new tasks, we should re-sort.
        // Optimization: Use a Priority Queue? In JS, manual array sort is usually fast enough for $N < 10,000$. For larger, a heap is better. But standard sort is concise and standard for these benchmarks unless N is huge.
        // Let's use sort. It's simple and readable.

    }
