    #version 450
    layout(local_size_x = 256) in;

    layout(set = 0, binding = 0) readonly buffer InputBoids {
        vec4 inData[];
    };

    layout(set = 0, binding = 1) buffer OutputBoids {
        vec4 outData[];
    };

    layout(set = 0, binding = 2) uniform Params {
        uvec4 counts;
        vec4 radii;
        vec4 weights;
        vec4 bounds;
    };

    void main() {
        uint idx = gl_GlobalInvocationID.x;
        if (idx >= counts.x) return;

        // Data
        vec4 posData = inData[idx * 2];
        vec4 velData = inData[idx * 2 + 1];
        vec3 pos = posData.xyz;
        vec3 vel = velData.xyz;

        vec3 sep = vec3(0.0);
        vec3 ali = vec3(0.0);
        vec3 coh = vec3(0.0);

        int sepCnt = 0;
        int aliCnt = 0;
        int cohCnt = 0;

        // Radii constants for clarity (though could be from uniform)
        // sep: 2.0, align: 5.0, coh: 8.0
        float sepRad = radii.y;
        float aliRad = radii.z;
        float cohRad = radii.w;

        for (uint j = 0; j < counts.x; j++) {
            if (j == idx) continue;

            vec4 otherPos = inData[j * 2];
            vec3 otherPosVec = otherPos.xyz;

            vec3 d = pos - otherPosVec;
            float distSq = dot(d, d);

            // Zero distance check
            if (distSq < 0.00001) continue;

            // Separation (only needs distSq if using inverse, but we need dist for checking radii usually)
            // Optimization: check radii against distSq
            float dist = sqrt(distSq);

            if (dist < sepRad) {
                sep += d / distSq; // 1/distance steering
                sepCnt++;
            }

            if (dist < aliRad) {
                vec3 otherVel = inData[j * 2 + 1].xyz;
                ali += otherVel;
                aliCnt++;
            }

            if (dist < cohRad) {
                coh += otherPosVec;
                cohCnt++;
            }
        }

        // Normalize Forces
        // Separation
        if (sepCnt > 0) {
            sep = normalize(sep);
        }

        // Alignment
        if (aliCnt > 0) {
            ali = ali / float(aliCnt);
            ali -= vel;
            ali = normalize(ali) * weights.w; // limit magnitude to max speed
        }

        // Cohesion
        if (cohCnt > 0) {
            coh = coh / float(cohCnt);
            coh -= pos;
            coh = normalize(coh) * weights.w;
        }

        // Boundary Logic
        // "when a component exceeds boundSize*0.4"
        // Let's assume a soft boundary force towards 0,0,0 if near edge.
        // Hard to interpret "boundSize*0.5 scaling".
        // Let's try: if abs(pos) > boundSize * 0.4, add boundForce * normalize(-pos)
        vec3 boundForce = vec3(0.0);
        float boundSize = bounds.x;
        float boundThreshold = boundSize * 0.4;

        if (abs(pos.x) > boundThreshold || abs(pos.y) > boundThreshold || abs(pos.z) > boundThreshold) {
            // Push toward center
            // Using boundForce magnitude (0.5)
            // Scale factor? Maybe 1.0?
            boundForce = normalize(-pos) * bounds.y; // bounds.y is boundForce
        }

        // Weighted Accumulation
        vec3 acc = sep * weights.x; // sepWeight
        acc += ali * weights.y; // alignWeight
        acc += coh * weights.z; // cohWeight
        acc += boundForce;

        // Update Velocity
        vel += acc * radii.x; // dt

        // Clamp Speed
        float speed = length(vel);
        if (speed > weights.w) { // maxSpeed
            vel = normalize(vel) * weights.w;
        }

        // Update Position
        pos += vel * radii.x; // dt

        // Write Output
        outData[idx * 2] = vec4(pos, 0.0);
        outData[idx * 2 + 1] = vec4(vel, 0.0);
    }

    layout(local_size_x = 256) in;

    layout(set = 0, binding = 0) readonly buffer InputBoids {
        vec4 inData[];
    };

    layout(set = 0, binding = 1) buffer OutputBoids {
        vec4 outData[];
    };

    layout(set = 0, binding = 2) uniform Params {
        uvec4 counts;
        vec4 radii; // .x=dt
        vec4 weights; // .x=sep, .y=align, .z=coh, .w=maxSpeed
        vec4 bounds; // .x=boundSize, .y=boundForce
    };

    void main() {
        uint idx = gl_GlobalInvocationID.x;
        if (idx >= counts.x) return;

        vec4 p1 = inData[idx * 2];
        vec4 p2 = inData[idx * 2 + 1];
        vec3 pos = p1.xyz;
        vec3 vel = p2.xyz;

        vec3 sep = vec3(0.0);
        vec3 ali = vec3(0.0);
        vec3 coh = vec3(0.0);

        int sepCnt = 0;
        int aliCnt = 0;
        int cohCnt = 0;

        float sepRadSq = radii.y * radii.y;
        float aliRadSq = radii.z * radii.z;
        float cohRadSq = radii.w * radii.w;
        float boundLimit = bounds.x * 0.4;

        // Use booleans for safety against division by zero? 
        // No, simply checking if cnt > 0 is enough.

        for (uint j = 0; j < counts.x; j++) {
            if (j == idx) continue;

            vec4 j1 = inData[j * 2];
            vec4 j2 = inData[j * 2 + 1];
            vec3 jPos = j1.xyz;
            vec3 jVel = j2.xyz;

            vec3 d = pos - jPos;
            float dSq = dot(d, d);
            if (dSq < 1e-6) continue; // Zero distance check

            // Optimization: avoid sqrt for radius checks if possible, 
            // but we need direction later. 
            // We can normalize later.

            // Separation
            if (dSq < sepRadSq) {
                // Steer away: use d. Normalize later or scale by inverse?
                // Normalize(d) gives direction.
                // sum += d / distSq gives 1/dist scaling.
                // Standard is often just normalize(d).
                // Let's use normalize(d).
                sep += normalize(d);
                sepCnt++;
            }

            // Alignment
            if (dSq < aliRadSq) {
                ali += jVel;
                aliCnt++;
            }

            // Cohesion
            if (dSq < cohRadSq) {
                coh += jPos;
                cohCnt++;
            }
        }

        vec3 acc = vec3(0.0);

        // Separation
        if (sepCnt > 0) {
            sep = normalize(sep);
            acc += sep * weights.x;
        }

        // Alignment
        if (aliCnt > 0) {
            ali = ali / float(aliCnt);
            ali -= vel;
            ali = normalize(ali);
            acc += ali * weights.y;
        }

        // Cohesion
        if (cohCnt > 0) {
            coh = coh / float(cohCnt);
            coh -= pos;
            coh = normalize(coh);
            acc += coh * weights.z;
        }

        // Boundary
        if (abs(pos.x) > boundLimit || abs(pos.y) > boundLimit || abs(pos.z) > boundLimit) {
            // Push toward center
            // Using boundForce (bounds.y)
            acc += normalize(-pos) * bounds.y;
        }

        // Integration
        vel += acc * radii.x; // dt

        // Clamp Speed
        float speed = length(vel);
        if (speed > weights.w) {
            vel = normalize(vel) * weights.w;
        }

        // Position Update
        pos += vel * radii.x;

        // Output
        outData[idx * 2] = vec4(pos, 0.0);
        outData[idx * 2 + 1] = vec4(vel, 0.0);
    }
