            this.packages.sort((a, b) => distSq(b, this.depot) - distSq(a, this.depot));

                this.routes[idx % this.truckCount].push(pkg);
            });

            this.routes.forEach(route => {
                this.solveTSP(route);
            });

            // Nearest Neighbor
            let current = this.depot;
            let unvisited = [...route];
            const order = [current, ...unvisited, current];

                let minDist = Infinity;

                        nearestIdx = i;

                }
