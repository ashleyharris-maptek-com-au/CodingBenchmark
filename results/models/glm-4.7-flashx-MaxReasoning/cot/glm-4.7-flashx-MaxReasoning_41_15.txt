2.  **Shader Logic Implementation (Mental Draft):**

    ```glsl
    // Inputs
    layout(location = 0) in vec3 worldPos;
    layout(location = 1) in vec3 normal;
    layout(location = 2) in vec2 uv;
    layout(location = 3) in vec3 tangent;
    layout(location = 4) in vec3 color;

    // Uniforms
    layout(set = 0, binding = 0) uniform UBO {
        mat4 model;
        mat4 view;
        mat4 projection;
        vec4 lightPos;
        vec4 cameraPos;
        vec4 params;
    };

    layout(location = 0) out vec4 outColor;

    void main() {
        // Normalizing normal (World space)
        vec3 N = normalize(normal);

        // Light vector L (from surface to light)
        // lightPos is usually in world space in this layout
        vec3 L = normalize(lightPos.xyz - worldPos);

        // Diffuse factor (dot product)
        float NdotL = dot(N, L);

        // Gooch shading parameter t
        // t = dot(N,L) * 0.5 + 0.5
        float t = NdotL * 0.5 + 0.5;

        // Define colors
        vec3 coolColor = vec3(0.2, 0.2, 0.75);
        vec3 warmColor = vec3(0.7, 0.7, 0.4);

        // Interpolate
        vec3 finalColor = (1.0 - t) * coolColor + t * warmColor;

        // Output
        outColor = vec4(finalColor, 1.0);
    }
    ```
