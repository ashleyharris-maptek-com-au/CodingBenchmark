        for (int i = 0; i < n; ++i) {
            if (order[i] == u) {
                su = i;
                break;
            }
        }
        for (int v : adj[u]) {
            if (u < v) { // Add undirected edge once
                int sv = -1;
                for (int i = 0; i < n; ++i) {
                    if (order[i] == v) {
                        sv = i;
                        break;
                    }
                }
                adj_bits[su][sv] = 1;
            }
        }
    }

    // 2. Greedy Initial Solution
    vector<int> best_set;
    bitset<2000> greedy_taken;
    
    // To improve greedy, sort by degree in the remaining graph?
    // Simple greedy:
    for (int i = 0; i < n; ++i) {
        if (!greedy_taken[i]) {
            greedy_taken[i] = 1;
            greedy_taken |= adj_bits[i];
        }
    }
    
    for (int i = 0; i < n; ++i) {
        if (greedy_taken[i]) {
            best_set.push_back(sorted_to_original[i]);
        }
    }

    // 3. Recursive DFS
    // We search for a set S in the sorted space.
    // S is a bitset of remaining candidates.
    // We pass the current path as a vector to construct the solution.
    vector<int> current_path;
    
    // Use a fixed size bitset constant.
    const int N = 2000; 

    function<void(bitset<N>, int)> dfs;
    dfs = [&](bitset<N> S, int current_size) {
        if (S.empty()) {
            if (current_size > best_set.size()) {
                best_set = current_path;
            }
            return;
        }

        // Prune: If even taking all remaining vertices can't beat best
        if (current_size + S.count() <= best_set.size()) return;

        // Heuristic: Pick a vertex with the minimum degree in the remaining graph.
        // Since we sorted by degree initially, the first set bit is a good candidate for "min degree" in the *original* ordering.
        // In the *remaining* graph, degrees might vary.
        // But finding min degree in S on the fly is expensive (O(N^2)).
        // Optimization: Just use the first set bit (lowest index).
        // Since we sorted by degree, this is often the vertex with the smallest degree.
        
        int v = (int)S.find_first_set();
        if (v == N) return;

        // Branch 1: Include v
        current_path.push_back(sorted_to_original[v]);
        // Remove v and its neighbors
        dfs(S & ~adj_bits[v], current_size + 1);
        current_path.pop_back();

        // Branch 2: Exclude v
        // Remove v only
        dfs(S & ~(bitset<N>(1) << v), current_size);
    };

    bitset<N> all_bits;
    all_bits.set();
    dfs(all_bits, 0);

    // Output
    cout << best_set.size() << endl;
    for (int v : best_set) cout << v << " ";
    cout << endl;

    return 0;
}
