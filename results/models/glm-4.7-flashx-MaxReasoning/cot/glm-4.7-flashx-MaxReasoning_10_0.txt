// Guillotine cut node
struct Node {
    int x, y, w, h;
    bool is_leaf;
    bool is_horizontal;
    int rect_id = -1;
    unique_ptr<Node> left, right;
    
    int area() const { return w * h; }
};

class GuillotinePacker {
public:
    void reset(int board_w, int board_h) {
        board_w_ = board_w;
        board_h_ = board_h;
        root_ = make_unique<Node>();
        root_->x = root_->y = 0;
        root_->w = board_w;
        root_->h = board_h;
        root_->is_leaf = true;
        rectangles_.clear();
    }
    
    bool tryPlace(int w, int h, int& x, int& y, bool allow_rot) {
        // Try without rotation first
        if (findBestSpot(w, h, x, y)) return true;
        
        // Try with rotation
        if (allow_rot) {
            if (findBestSpot(h, w, x, y)) return true;
        }
        
        return false;
    }
    
private:
    bool findBestSpot(int w, int h, int& x, int& y) {
        Node* best = nullptr;
        int best_score = -1;
        
        // Search all leaf nodes
        searchLeaf(root_.get(), w, h, best, best_score);
        
        if (best) {
            x = best->x;
            y = best->y;
            
            // Create cut
            if (best->w >= w && best->h >= h) {
                // Choose orientation: prioritize horizontal for balance
                if (best->h - h >= best->w - w) {
                    best->is_horizontal = true;
                } else {
                    best->is_horizontal = false;
                }
            }
            
            // Split
            split(best, w, h);
            return true;
        }
        
        return false;
    }
    
    void searchLeaf(Node* node, int w, int h, Node*& best, int& best_score) {
        if (!node) return;
        
        if (node->is_leaf) {
            // Score based on remaining space and orientation
            int remaining = node->area() - w * h;
            int waste = max(0, max(node->w - w, node->h - h));
            
            // Prefer exact fit or less waste
            int score = remaining - waste;
            
            if (score > best_score) {
                best = node;
                best_score = score;
            }
        } else {
            searchLeaf(node->left.get(), w, h, best, best_score);
            searchLeaf(node->right.get(), w, h, best, best_score);
        }
    }
    
    void split(Node* node, int w, int h) {
        node->is_leaf = false;
        node->rect_id = rectangles_.size();
        rectangles_.push_back({node->x, node->y, w, h});
        
        if (node->is_horizontal) {
            // Split vertically
            int mid = node->w / 2;
            node->left = make_unique<Node>();
            node->left->x = node->x;
            node->left->y = node->y;
            node->left->w = mid;
            node->left->h = node->h;
            node->left->is_leaf = true;
            
            node->right = make_unique<Node>();
            node->right->x = node->x + mid;
            node->right->y = node->y;
            node->right->w = node->w - mid;
            node->right->h = node->h;
            node->right->is_leaf = true;
        } else {

            int mid = node->h / 2;
            node->left = make_unique<Node>();
            node->left->x = node->x;
            node->left->y = node->y;
            node->left->w = node->w;
            node->left->h = mid;
            node->left->is_leaf = true;
            
            node->right = make_unique<Node>();
            node->right->x = node->x;
            node->right->y = node->y + mid;
            node->right->w = node->w;
            node->right->h = node->h - mid;
            node->right->is_leaf = true;
        }
    }

    unique_ptr<Node> root_;
