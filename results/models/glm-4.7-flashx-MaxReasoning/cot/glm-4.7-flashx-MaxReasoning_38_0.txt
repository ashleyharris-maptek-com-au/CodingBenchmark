class DLX:
    def __init__(self, U, sets):
        self.U = U
        self.S = len(sets)
        self.cols = [] # column headers
        self.root = DLXNode(-1) # root node with col=-1
        # Create column headers
        for c in range(U):
            col_node = DLXNode(c) # we'll treat as header; but we need to link them
            col_node.C = col_node
            col_node.row = -1
            col_node.col = -1
            col_node.size = 0
            self.cols.append(col_node)
        # Link column headers in a circle
        for c in range(U):
            self.link_circular(c, (c-1) % U)
        # Add rows
        for r, elems in enumerate(sets):
            if not elems:
                continue  # empty set cannot be used
            # create a linked list for this row
            first = None
            prev = None
            for e in elems:
                if 0 <= e < U:
                    node = DLXNode(e, r, e)
                    node.C = self.cols[e]
                    # Link to column header
                    # Insert into column header's list: before current head
                    c_node = self.cols[e]
                    node.U = c_node
                    node.D = c_node.D
                    c_node.D.U = node
                    c_node.D = node
                    c_node.size += 1
                    # Link within row
                    if prev is None:
                        first = node
                        node.L = node
                        node.R = node
                    else:
                        node.L = prev
                        node.R = prev.R
                        prev.R.L = node
                        prev.R = node
                    prev = node
            # Ensure row is linked (if row had no elements, skip)
        self.solution = []

    def link_circular(self, c1, c2):
        # link c1 and c2 in order
        left = self.cols[c1]
        right = self.cols[c2]
        left.R = right
        right.L = left

    def print_solution(self):
        print("SOLUTION")
        print(" ".join(map(str, self.solution)))

    def search(self, depth=0):
        if self.root.R == self.root:  # all columns covered
            return True
        # Choose column with smallest size (heuristic)
        c = self.choose_column()
        # Cover column
        self.cover(c)
        # Iterate rows
        r = c.D
        while r != c:
            # choose this row
            self.solution.append(r.row)
            # cover all columns in this row
            r = r.R
            while r != c:
                self.cover(r.C)
                r = r.R
            # recurse
            if self.search(depth+1):
                return True
            # un-choose row
            r = r.R
            while r != c:
                self.uncover(r.C)
                r = r.R
            self.solution.pop()
            r = r.D
        # uncover column
        self.uncover(c)
        return False

    def choose_column(self):
        # find column with minimum size
        min_col = self.root.R
        min_size = min_col.size
        c = min_col.R
        while c != self.root:
            if c.size < min_size:
                min_size = c.size
                min_col = c
            c = c.R
        return min_col

    def cover(self, c):
        # remove column c from column list
        c.L.R = c.R
        c.R.L = c.L
        # cover rows
        r = c.D
        while r != c:
            j = r.R
            while j != r:
                # remove node j from its column
                j.U.D = j.D
                j.D.U = j.U
                j.C.size -= 1
                j = j.R
            r = r.D

    def uncover(self, c):

        r = c.D

            j = r.R
            while j != r:
                j.C.size += 1
                j.U.D = j
                j.D.U = j
                j = j.R
            r = r.D
        # restore column
        c.L.R = c
        c.R.L = c

    def __init__(self, U, sets):
        self.U = U
        self.S = len(sets)
        self.cols = [Node(i) for i in range(U)]  # column headers
        self.root = Node(-1)  # root sentinel
        # link root to column headers
        # initialize column headers
        for i in range(U):
            self.cols[i].L = self.cols[i]
            self.cols[i].R = self.cols[i]
            self.cols[i].U = self.cols[i]
            self.cols[i].D = self.cols[i]
            self.cols[i].size = 0
        # link root to first and last column
        if U > 0:
            self.root.L = self.cols[U-1]
            self.root.R = self.cols[0]
            self.cols[U-1].R = self.root
            self.cols[0].L = self.root
        else:
            # no columns, root alone
            self.root.L = self.root
            self.root.R = self.root

        # Build rows
        for r, elems in enumerate(sets):
            if not elems:
                continue
            # Deduplicate elements in this set
            seen = set()
            first = None
            prev = None
            for e in elems:
                if e < 0 or e >= U:
                    continue  # ignore invalid elements
                if e in seen:
                    continue
                seen.add(e)
                node = Node(self.cols[e], row=r, col=e)
                # Insert into column e
                c = self.cols[e]
                node.U = c
                node.D = c.D
                c.D.U = node
                c.D = node
                c.size += 1
                # Link within row
                if prev is None:
                    first = node
                    node.L = node
                    node.R = node
                else:
                    node.L = prev
                    node.R = prev.R
                    prev.R.L = node
                    prev.R = node
                prev = node

    def choose_column(self):
        # find column with minimum size
        min_col = self.root.R
        min_size = min_col.size
        c = min_col.R
        while c != self.root:
            if c.size < min_size:
                min_size = c.size
                min_col = c
            c = c.R
        return min_col

    def cover(self, col):
        # Remove column from list of columns
        col.L.R = col.R
        col.R.L = col.L
        # Cover rows in this column
        r = col.D
        while r != col:
            j = r.R
            while j != r:
                # Remove j from its column
                j.U.D = j.D
                j.D.U = j.U
                j.C.size -= 1
                j = j.R
            r = r.D

    def uncover(self, col):
        # Uncover rows
        r = col.D
        while r != col:
            j = r.R
            while j != r:
                j.C.size += 1
                j.U.D = j
                j.D.U = j
                j = j.R
            r = r.D
        # Restore column
        col.L.R = col
        col.R.L = col

    def search(self):
        self.solution = []
        if self.root.R == self.root:  # no columns left
            return True
        col = self.choose_column()
        self.cover(col)
        r = col.D
        while r != col:
            self.solution.append(r.row)
            # cover all columns in this row
            j = r.R
            while j != r:
                self.cover(j.C)
                j = j.R
            if self.search():
                return True
            # un-choose
            j = r.R
            while j != r:
                self.uncover(j.C)
                j = j.R
            self.solution.pop()
            r = r.D
        self.uncover(col)
        return False

    def print_solution(self):
        if self.solution:
            print("SOLUTION")
            print(" ".join(map(str, self.solution)))
        else:
            print("NO SOLUTION")

def main():
    import sys
    # read first line, skip empty lines
    first_line = sys.stdin.readline()
    while first_line.strip() == '':
