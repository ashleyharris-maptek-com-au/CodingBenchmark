fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse n, m
    let first_line = lines.next().unwrap().unwrap();
    let mut iter = first_line.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let nums: Vec<usize> = line.split_whitespace()
            .map(|x| x.parse().unwrap())
            .collect();
        edges.push((nums[0], nums[1]));
    }

    // Prepare adjacency list for fast lookup during recursion
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for (u, v) in edges {
        adj[u].push(v);
        adj[v].push(u);
    }

    let half = n / 2;
    let mut best_cut = u64::MAX;
    let mut best_assignments: Vec<i32> = vec![0; n];
    let mut assignments: Vec<i32> = vec![0; n]; // 0 for A, 1 for B
    let mut current_cut: usize = 0;

    let mut vertex_order: Vec<usize> = (0..n).collect();
    vertex_order.sort_by_key(|&i| adj[i].len());

    // We can fix the first vertex in the order to A to reduce symmetry
    // e.g., set assignments[vertex_order[0]] = 0 before starting recursion.
    // Then we just assign the rest.

    // To make it cleaner, let's just iterate recursively. 
    // We pass the index in the sorted order.

    // Pruning helper
    fn recurse(

        n: usize,
        half: usize,
        adj: &Vec<Vec<usize>>,
        current_cut: usize,
        best_cut: &mut u64,
        assignments: &mut Vec<i32>,
        best_assignments: &mut Vec<i32>,
        vertex_order: &[usize]
    ) {
        // If we processed all vertices
        if pos == n {
            if current_cut < *best_cut {
                *best_cut = current_cut;
                best_assignments.clone_from_slice(assignments);
            }
            return;
        }

        let u = vertex_order[pos];
        
        // If u is already assigned (we didn't reset for symmetry breaking, 
        // but we usually don't need to reset assignments array if we overwrite indices.
        // Actually, in a simple bitmask/recursion, we just fill the array).
        
        // Symmetry breaking: If we have assigned exactly 'half' vertices to A (0),
        // and we have more vertices to go, we must assign them to B (1).
        // But wait, if we assign to B, we increase the cut?
        // No, cut is edges (A, B). Assigning to B only adds edges (u, A).

        // Count cuts added: edges from u to vertices already in B (1)
        let mut cuts_added = 0;
        let neighbor_indices: Vec<usize> = adj[u].iter().map(|&v| v).collect();
        
        // Check neighbors
        for &v in &neighbor_indices {
            if assignments[v] == 1 {
                cuts_added += 1;
            }
        }

        if current_cut + cuts_added < *best_cut {
            assignments[u] = 0;
            recurse(pos + 1, n, half, adj, current_cut + cuts_added, best_cut, assignments, best_assignments, vertex_order);
        }

        // Branch 2: Assign u to B
        cuts_added = 0;
        for &v in &neighbor_indices {
            if assignments[v] == 0 {
                cuts_added += 1;
            }
        }

        if current_cut + cuts_added < *best_cut {
            assignments[u] = 1;
            recurse(pos + 1, n, half, adj, current_cut + cuts_added, best_cut, assignments, best_assignments, vertex_order);
        }
        
        // Clean up? Not strictly necessary for local variables, but good practice
        assignments[u] = -1; 
    }

    // To fix symmetry (A vs B), we can just ensure that assignments[0] is 0.
    // But we sorted the order.
    // Let's just start recursion from index 0.
    // Since we need exactly n/2 in A, we can treat the "remaining" calculation implicitly by counting 0s.
    // A more efficient way: 
    // Choose n/2 vertices. The rest is B.
    // Iterate from 0 to n.
    // At step k, we have chosen k vertices for A. Next vertex must be A or B.
    // Wait, "Choose n/2 vertices" is better than "Assign n vertices to 0/1".
    // It saves checking count >= n/2.
    
    // Algorithm: 
    // pos: current vertex index (0 to n)
    // cnt_a: number of vertices assigned to A so far.
    // We must ensure cnt_a <= n/2.
    // At pos, we decide if vertex[pos] goes to A or B.
    // If we assign to A, cnt_a increases.
    
    // Optimization: sort vertices.
    
    // Pruning: if current_cut >= best_cut, stop.
    
    let mut cnt_a = 0;
    let mut current_cut = 0;
    let mut best_cut = u64::MAX;
    let mut best_partition: Vec<usize> = Vec::new();

    assignments[vertex_order[0]] = 0;
    cnt_a += 1;

    recurse_rec(1, n, half, adj, &mut assignments, &mut current_cut, &mut best_cut, &mut best_partition, &mut cnt_a, &mut vertex_order, 0);

            print!("{} ", i);
        }

    println!();

fn recurse_rec(
    pos: usize,
    n: usize,
    half: usize,
    adj: &Vec<Vec<usize>>,
    assignments: &mut Vec<i32>,
    current_cut: &mut usize,
    best_cut: &mut u64,
    best_partition: &mut Vec<usize>,
    cnt_a: &mut usize,
    vertex_order: &[usize]
) {
    if pos == n {
        if *current_cut < *best_cut {
            *best_cut = *current_cut;
            best_partition.clone_from_slice(assignments);
        }
        return;
    }

    let u = vertex_order[pos];
    let neighbors = &adj[u];
    
    // Try putting u in A
    if *cnt_a < half {
        let mut cut_inc = 0;
        for &v in neighbors {
            if assignments[v] == 1 {
                cut_inc += 1;
            }
        }
        
        let new_cut = *current_cut + cut_inc;
        if new_cut < *best_cut {
            assignments[u] = 0;
            *cnt_a += 1;
            recurse_rec(pos + 1, n, half, adj, assignments, current_cut, best_cut, best_partition, cnt_a, vertex_order);
            *cnt_a -= 1;
        }
    }

    // Try putting u in B
    let mut cut_inc = 0;
    for &v in neighbors {
        if assignments[v] == 0 {
            cut_inc += 1;
        }
    }

    let new_cut = *current_cut + cut_inc;
    if new_cut < *best_cut {
        assignments[u] = 1;
        recurse_rec(pos + 1, n, half, adj, assignments, current_cut, best_cut, best_partition, cnt_a, vertex_order);
        assignments[u] = -1; // Reset
    }
}

            return;
        }

        let u = indices[pos];
        let neighbors = &adj[u];

        // Try assign to A
        if *cnt_a < half {
            let mut inc = 0;
            for &v in neighbors {
                if assignments[v] == 1 {
                    inc += 1;
                }
            }

            if *current_cut + inc < *best_cut {
                assignments[u] = 0;
                *cnt_a += 1;
                recurse(pos + 1, adj, assignments, current_cut, best_cut, best_partition, cnt_a, indices);
                *cnt_a -= 1;
            }
        }

        // Try assign to B
        let mut inc = 0;
        for &v in neighbors {
            if assignments[v] == 0 {
                inc += 1;
            }
        }

        if *current_cut + inc < *best_cut {
            assignments[u] = 1;
            recurse(pos + 1, adj, assignments, current_cut, best_cut, best_partition, cnt_a, indices);
            assignments[u] = -1; 
        }
    };

    // Initialize count for fixed vertex
    let mut cnt_a = 1;
    recurse(1, &adj, &mut assignments, &mut current_cut, &mut best_cut, &mut best_partition, &mut cnt_a, &indices);

    // Output
    writeln!(out, "{}", best_cut).unwrap();
    let mut out_str = String::new();
    for v in &best_partition {
        out_str.push_str(&format!("{} ", v));
    }
    // Remove trailing space if any
    if out_str.ends_with(' ') {
        out_str.pop();
    }
    writeln!(out, "{}", out_str).unwrap();
}
