    if (!isFree(rx, ry) || !isFree(jx, jy)) return Infinity;
    if (rx === jx && ry === jy) return 0;

    const queue = [[rx, ry, 0]];
    // Use Uint8Array for visited for performance.

    // However, resetting a large Uint8Array in JS is fast enough for moderate grids.

    // Given the environment spec (64 cores, 128GB RAM), brute force BFS is likely expected or sufficient.
    // Let's stick to the simple visited array approach for readability and correctness.

    // Since JS array access is faster than Set, let's use a 2D array or flat array.
    const visited = new Uint8Array(W * H);
    visited[ry * W + rx] = 1;

    while (queue.length > 0) {
        const [cx, cy, dist] = queue.shift();
        for (const [dx, dy] of dirs) {
            const nx = cx + dx;
            const ny = cy + dy;
            const nIdx = ny * W + nx;

            if (nx === jx && ny === jy) return dist + 1;

            if (isFree(nx, ny) && visited[nIdx] === 0) {
                visited[nIdx] = 1;
                queue.push([nx, ny, dist + 1]);
            }
        }
    }
    return Infinity;
}

// 1. Sort jobs by priority (highest first).
// 2. Assign each job to the closest available robot.
//    (Deterministic: find min, assign, remove robot from list).
//    Note: The standard greedy "greedy" often refers to sorting jobs or sorting resources.
//    Assigning job -> closest robot is a valid approximation for the assignment problem.

jobs.sort((a, b) => b.priority - a.priority);

        if (usedRobots.has(i)) continue;

            bestIdx = i;
        }
    }

    if (bestIdx !== -1) {
        assignments[bestIdx] = job.id;
        usedRobots.add(bestIdx);
    }

        const W = grid[0].length;
        const H = grid.length;
        const dirs = [[0, 1], [1, 0], [0, -1], [-1, 0]];

        // Optimization for BFS visited states to avoid GC
        let bfsGen = 0;
        const visited = new Uint8Array(W * H);

        const isFree = (x, y) => {
            return x >= 0 && x < W && y >= 0 && y < H && grid[y][x] === '.';
        };

        const getDist = (x1, y1, x2, y2) => {
            if (!isFree(x1, y1) || !isFree(x2, y2)) return Infinity;
            if (x1 === x2 && y1 === y2) return 0;

            const queue = [[x1, y1, 0]];
            const startIdx = y1 * W + x1;
            
            bfsGen++;
            visited[startIdx] = bfsGen;

            while (queue.length) {
                const [cx, cy, dist] = queue.shift();
                for (const [dx, dy] of dirs) {
                    const nx = cx + dx;
                    const ny = cy + dy;
                    const nIdx = ny * W + nx;

                    if (nx === x2 && ny === y2) return dist + 1;

                    if (isFree(nx, ny) && visited[nIdx] !== bfsGen) {
                        visited[nIdx] = bfsGen;
                        queue.push([nx, ny, dist + 1]);
                    }
                }
            }
            return Infinity;
        };

        // Greedy: Highest priority jobs first
        jobs.sort((a, b) => b.priority - a.priority);

        const assignments = new Array(robots.length).fill(null);
        const used = new Set();

        for (const job of jobs) {
            let bestIdx = -1;
            let minD = Infinity;

            for (let i = 0; i < robots.length; i++) {
                if (used.has(i)) continue;
                const d = getDist(robots[i].x, robots[i].y, job.x, job.y);
                if (d < minD) {
                    minD = d;
                    bestIdx = i;
                }
            }

            if (bestIdx !== -1) {
                assignments[bestIdx] = job.id;
                used.add(bestIdx);
            }
        }

        console.log(JSON.stringify({ assignments }));
