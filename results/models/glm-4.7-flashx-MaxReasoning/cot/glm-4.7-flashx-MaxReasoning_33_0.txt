    using System.Collections.Generic;
    using System.IO;

    public class QAP
    {
        public static void Main()
        {
            using (StreamReader sr = new StreamReader(Console.OpenStandardInput()))
            using (StreamWriter sw = new StreamWriter(Console.OpenStandardOutput()))
            {
                int N = int.Parse(sr.ReadLine().Trim());
                
                long[,] flow = new long[N, N];
                long[,] dist = new long[N, N];

                for (int i = 0; i < N; i++)
                {
                    string[] parts = sr.ReadLine().Split();
                    for (int j = 0; j < N; j++)
                    {
                        flow[i, j] = long.Parse(parts[j]);
                    }
                }

                for (int i = 0; i < N; i++)
                {
                    string[] parts = sr.ReadLine().Split();
                    for (int j = 0; j < N; j++)
                    {
                        dist[i, j] = long.Parse(parts[j]);
                    }
                }

                int[] perm = SolveQAP(N, flow, dist);
                
                long totalCost = 0;
                for (int i = 0; i < N; i++)
                {
                    for (int j = 0; j < N; j++)
                    {
                        totalCost += flow[i, j] * dist[perm[i], perm[j]];
                    }
                }

                sw.WriteLine(totalCost);
                sw.WriteLine(string.Join(" ", perm));
            }
        }

        static int[] SolveQAP(int N, long[,] flow, long[,] dist)
        {
            // 1. Initialize with a greedy heuristic (High Flow -> High Distance)
            // Calculate total flow for each facility
            long[] facilityInfluence = new long[N];
            for (int i = 0; i < N; i++)
            {
                facilityInfluence[i] = 0;
                for (int j = 0; j < N; j++) facilityInfluence[i] += flow[i, j];
            }

            // Calculate total distance for each location
            long[] locationInfluence = new long[N];
            for (int j = 0; j < N; j++)
            {
                locationInfluence[j] = 0;
                for (int i = 0; i < N; i++) locationInfluence[j] += dist[i, j];
            }

            // Indices of facilities sorted by influence (descending)
            int[] facilityOrder = new int[N];
            for (int i = 0; i < N; i++) facilityOrder[i] = i;
            Array.Sort(facilityOrder, (a, b) => facilityInfluence[b].CompareTo(facilityInfluence[a]));

            // Indices of locations sorted by influence (descending)
            int[] locationOrder = new int[N];
            for (int i = 0; i < N; i++) locationOrder[i] = i;
            Array.Sort(locationOrder, (a, b) => locationInfluence[b].CompareTo(locationInfluence[a]));

            // Initial permutation
            int[] perm = new int[N];
            for (int i = 0; i < N; i++)
            {
                perm[facilityOrder[i]] = locationOrder[i];
            }

            // 2. Simulated Annealing
            // Pre-calculate row totals of dist and flow for quick access if needed, 
            // but 2D arrays are fine.
            
            // Optimization: Use `long` for delta to avoid overflow
            long currentCost = CalculateCost(N, flow, dist, perm);
            Random rnd = new Random();
            
            double temp = 100000000.0; // High initial temperature
            double coolingRate = 0.9995;
            double minTemp = 0.1;

            while (temp > minTemp)
            {
                for (int iter = 0; iter < N * 5; iter++) // Loop multiple times per temp step
                {
                    int i = rnd.Next(N);
                    int j = rnd.Next(N);
                    if (i == j) continue;

                    long delta = CalculateDeltaSwap(N, flow, dist, perm, i, j);
                    
                    // Always accept if better
                    if (delta < 0 || (double)rnd.Next(int.MaxValue) / int.MaxValue < Math.Exp(-delta / temp))
                    {
                        // Apply swap
                        Swap(perm, i, j);
                        currentCost += delta;
                    }
                }
                temp *= coolingRate;
            }

            return perm;
        }

        static long CalculateCost(int N, long[,] flow, long[,] dist, int[] perm)
        {
            long cost = 0;
            for (int i = 0; i < N; i++)
            {
                for (int j = 0; j < N; j++)
                {
                    cost += flow[i, j] * dist[perm[i], perm[j]];
                }
            }
            return cost;
        }

        static long CalculateDeltaSwap(int N, long[,] flow, long[,] dist, int[] perm, int i, int j)
        {
            long delta = 0;
            long pi = perm[i];
            long pj = perm[j];
            
            // Note: We calculate change for pairs involving i and j.
            // Pairs (i, k) and (j, k) for all k.
            // k ranges 0..N-1.
            
            for (int k = 0; k < N; k++)
            {
                if (k == i || k == j) continue;
                
                // Terms involving (i, k) and (j, k)
                // Term (i, k): flow[i][k] * dist[p_i][p_k] -> flow[i][k] * dist[pj][p_k]
                // Term (j, k): flow[j][k] * dist[p_j][p_k] -> flow[j][k] * dist[pi][p_k]
                
                long newTerm = flow[i, k] * dist[pj, perm[k]] + flow[j, k] * dist[pi, perm[k]];
                long oldTerm = flow[i, k] * dist[pi, perm[k]] + flow[j, k] * dist[pj, perm[k]];
                
                delta += newTerm - oldTerm;
            }
            
            // What about pairs (k, i) and (k, j)? 
            // The outer loop sums over all pairs (i, j) in CalculateCost.
            // Since CalculateCost iterates over all i and all j, (k, i) is covered when outer loop is k and inner loop is i.
            // Wait, inside CalculateCost:
            // i=0, j=0..N-1 -> covers (0,0)...(0,N-1)
            // i=1, j=0..N-1 -> covers (1,0)...(1,N-1) -> includes (0,1) and (1,0)
            // So pairs are fully covered by iterating k over all N indices.
            
            return delta;
        }

        static void Swap(int[] arr, int i, int j)
        {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }

    **Reviewing the Delta Logic again:**
    Original Cost: $\sum_{u,v} flow[u][v] \cdot dist[p_u][p_v]$.
    Swap $i \leftrightarrow j$.
    Affected terms are those where $u=i$ or $v=i$ or $u=j$ or $v=j$.
    The set $\{u,v\}$ contains $i$ or $j$.
    My loop iterates $k$ from $0 \dots N-1$.
    Terms considered: $(i, k)$ and $(j, k)$.
    Since the set of all pairs is $P = \{(u,v) | 0 \le u, v < N\}$.
    The affected set $A = \{(u,v) \in P | u \in \{i,j\} \lor v \in \{i,j\}\}$.
    Terms in $A$:
    1. $u=i, v \in 0..N-1$ (N terms)
    2. $u=j, v \in 0..N-1$ (N terms)
    Intersection: $u=i, v=j$ (counted twice).
    Total: $2N - 1$ terms.
    My loop $k=0 \dots N-1$ covers:
    $u=i, v=k$ (N terms).
    $u=j, v=k$ (N terms).
    It covers all terms.
    Logic holds.
