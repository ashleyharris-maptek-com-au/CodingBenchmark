```spirv
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; -- Types --

OpTypeVoid
OpTypeFloat 32
OpTypeVector %float 2
OpTypeVector %float 3
OpTypeVector %float 4
OpTypeMatrix %float 4x4

; gl_PerVertex Input
%gl_PerVertex_in = OpTypeStruct %float_4
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User Inputs
%ptr_in_vec3 = OpTypePointer Input %float_3
%arr_vec3 = OpTypeArray %float_3 %uint_2
%worldPos = OpVariable %ptr_in_vec3 Input
%normal = OpVariable %ptr_in_vec3 Input
%color = OpVariable %ptr_in_vec3 Input

; gl_PerVertex Output
%gl_PerVertex_out = OpTypeStruct %float_4
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_struct_pv_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_struct_pv_out Output
%ptr_out_vec3 = OpTypePointer Output %float_3
%outColor = OpVariable %ptr_out_vec3 Output

; Uniforms (Test 41 Layout: mvp, mvp_inv, light, cam, params)
%ubo_struct = OpTypeStruct %mat4 %mat4 %float_4 %float_4 %float_4
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_uniform_ubo Uniform
%mvp = OpMemberConstant %ubo_struct 0 %uint 0

; -- Constants --
%int_zero = OpConstant %int 0
%int_one = OpConstant %int 1
%int_two = OpConstant %int 2
%int_three = OpConstant %int 3
%float_one = OpConstant %float 1.0
%float_zero = OpConstant %float 0.0
%float_neg_one = OpConstant %float -1.0
%vec3_one = OpConstantComposite %float_3 %float_one %float_one %float_one
%vec3_zero = OpConstantComposite %float_3 %float_zero %float_zero %float_zero
%vec2_one = OpConstantComposite %float_2 %float_one %float_one
%vec2_x = OpConstantComposite %float_2 %float_one %float_zero
%vec2_y = OpConstantComposite %float_2 %float_zero %float_one
%float_0_75 = OpConstant %float 0.75
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05

; -- Function --

%main_func = OpFunction %void None %void
%lbl_start = OpLabel

; Load Inputs
%p0 = OpLoad %float_3 %worldPos %int_zero
%p1 = OpLoad %float_3 %worldPos %int_one
%c0 = OpLoad %float_3 %color %int_zero
%c1 = OpLoad %float_3 %color %int_one

; Transform to Clip Space (MVP)
%p0_vec4 = OpCompositeConstruct %float_4 %p0 %float_one
%p1_vec4 = OpCompositeConstruct %float_4 %p1 %float_one
%p0_clip = OpMatrixTimesVector %float_4 %mvp %p0_vec4
%p1_clip = OpMatrixTimesVector %float_4 %mvp %p1_vec4

; Geometry Logic
%dir = OpFSub %float_3 %p1 %p0
%mid = OpFAdd %float_3 %p0 OpFMul %float_3 %dir %vec3_0_75

; Perpendicular (normalize(-dir.y, dir.x))
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%dir_x = OpCompositeExtract %float %dir 0
%vec2_perp_raw = OpCompositeConstruct %float_2 %neg_dir_y %dir_x
%vec3_perp_raw = OpCompositeConstruct %float_3 OpVectorShuffle %vec2_perp_raw %vec2_perp_raw 0 1 0
%perp = OpNormalize %float_3 %vec3_perp_raw

; Offsets
%off_t0 = OpFMul %float_3 %perp %float_0_015
%off_t1 = OpFMul %float_3 %perp %float_0_05

; Shaft Vertices (4)
; Order: p0+, p0-, mid+, mid-
; Note: p0+ means p0 + off_t0, p0- means p0 - off_t0. 
; We define v0-v3 based on the strip order.
%v0_shaft = OpFAdd %float_3 %p0 %off_t0
%v1_shaft = OpFSub %float_3 %p0 %off_t0
%v2_shaft = OpFAdd %float_3 %mid %off_t0
%v3_shaft = OpFSub %float_3 %mid %off_t0

; Arrowhead Vertices (3)
; Order: mid+, mid-, p1
%v0_arrow = OpFAdd %float_3 %mid %off_t1
%v1_arrow = OpFSub %float_3 %mid %off_t1
%v2_arrow = %p1

; Transform Shaft to Clip
%v0_clip = OpFAdd %float_3 %p0 %off_t0
%v1_clip = OpFSub %float_3 %p0 %off_t0
%v2_clip = OpFAdd %float_3 %mid %off_t0
%v3_clip = OpFSub %float_3 %mid %off_t0

%v0_clip4 = OpCompositeConstruct %float_4 %v0_clip %float_one
%v1_clip4 = OpCompositeConstruct %float_4 %v1_clip %float_one
%v2_clip4 = OpCompositeConstruct %float_4 %v2_clip %float_one
%v3_clip4 = OpCompositeConstruct %float_4 %v3_clip %float_one

%ac_0 = OpAccessChain %ptr_out_struct_pv_out %gl_out %int_zero %int_zero
%ac_c0 = OpAccessChain %ptr_out_vec3 %outColor %int_zero
OpStore %ac_0 %v0_clip4
OpStore %ac_c0 %c0
OpEmitVertex

%ac_1 = OpAccessChain %ptr_out_struct_pv_out %gl_out %int_zero %int_zero
OpStore %ac_1 %v1_clip4
OpStore %ac_c0 %c0
OpEmitVertex

%ac_2 = OpAccessChain %ptr_out_struct_pv_out %gl_out %int_zero %int_zero
OpStore %ac_2 %v2_clip4
OpStore %ac_c0 %c0
OpEmitVertex

%ac_3 = OpAccessChain %ptr_out_struct_pv_out %gl_out %int_zero %int_zero
OpStore %ac_3 %v3_clip4
OpStore %ac_c0 %c0
OpEmitVertex
OpEndPrimitive

; Transform Arrowhead to Clip
%v0_arrow_clip = OpFAdd %float_3 %mid %off_t1
%v1_arrow_clip = OpFSub %float_3 %mid %off_t1
%v0_arrow_clip4 = OpCompositeConstruct %float_4 %v0_arrow_clip %float_one
%v1_arrow_clip4 = OpCompositeConstruct %float_4 %v1_arrow_clip %float_one

%ac_4 = OpAccessChain %ptr_out_struct_pv_out %gl_out %int_zero %int_zero
%ac_c1 = OpAccessChain %ptr_out_vec3 %outColor %int_zero
OpStore %ac_4 %v0_arrow_clip4
OpStore %ac_c1 %c1
OpEmitVertex

%ac_5 = OpAccessChain %ptr_out_struct_pv_out %gl_out %int_zero %int_zero
OpStore %ac_5 %v1_arrow_clip4
OpStore %ac_c1 %c1
OpEmitVertex

%ac_6 = OpAccessChain %ptr_out_struct_pv_out %gl_out %int_zero %int_zero
OpStore %ac_6 %p1_clip
OpStore %ac_c1 %c1
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
