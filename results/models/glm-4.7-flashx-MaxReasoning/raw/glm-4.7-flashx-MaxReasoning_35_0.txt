```rust
use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    
    // Parse dimensions
    let first_line = lines.next().unwrap().unwrap();
    let mut iter = first_line.split_whitespace();
    let num_vertices: usize = iter.next().unwrap().parse().unwrap();
    let num_edges: usize = iter.next().unwrap().parse().unwrap();

    if num_vertices == 0 {
        writeln!(out, "0").unwrap();
        return;
    }
    if num_vertices == 1 {
        writeln!(out, "1\n0").unwrap();
        return;
    }

    // 1. Build adjacency list and calculate degrees
    // Using Vec<Vec<usize>> for sparse representation
    let mut adj: Vec<Vec<usize>> = vec![vec![]; num_vertices];
    let mut degrees: Vec<usize> = vec![0; num_vertices];

    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let mut iter = line.split_whitespace();
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();

        // Undirected graph
        adj[u].push(v);
        adj[v].push(u);
        degrees[u] += 1;
        degrees[v] += 1;
    }

    // 2. Greedy Heuristic: Sort vertices by degree (descending)
    // We want to pick vertices that likely cover the most neighbors.
    // This heuristic runs in O(n log n + m).
    
    let mut indices: Vec<usize> = (0..num_vertices).collect();
    // Sort indices by degree in descending order
    indices.sort_by_key(|&i| degrees[i]);

    let mut dominated = vec![false; num_vertices];
    let mut result: Vec<usize> = Vec::new();

    for v in indices {
        if !dominated[v] {
            // v is not yet dominated. It must be in the set.
            result.push(v);
            dominated[v] = true;

            // Mark all neighbors as dominated
            // This loop iterates over all neighbors of v
            // The total work across all iterations is O(m)
            let neighbors = &adj[v];
            for &u in neighbors {
                dominated[u] = true;
            }
        }
    }

    // 3. Output
    writeln!(out, "{}", result.len()).unwrap();
    if !result.is_empty() {
        let result_str: Vec<String> = result.iter().map(|x| x.to_string()).collect();
        writeln!(out, "{}", result_str.join(" ")).unwrap();
    }
}
```
