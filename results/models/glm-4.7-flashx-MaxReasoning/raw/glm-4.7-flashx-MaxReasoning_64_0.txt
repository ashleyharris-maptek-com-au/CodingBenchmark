// src/index.ts

/**
 * Represents a position in the grid.
 */
interface Position {
    x: number;
    y: number;
}

/**
 * Represents a drone starting point.
 */
interface Drone {
    id: number;
    x: number;
    y: number;
}

/**
 * Represents the input data structure.
 */
interface InputData {
    probability: number[][];
    drones: Drone[];
    maxStepsPerDrone: number;
}

/**
 * Represents a node in the Priority Queue for pathfinding.
 * g: accumulated steps (cost)
 * h: heuristic (estimated gain)
 * f: g + h
 */
interface PriorityQueueNode {
    x: number;
    y: number;
    f: number;
    g: number;
}

// Simple Priority Queue implementation using an array
class PriorityQueue {
    private items: PriorityQueueNode[] = [];

    enqueue(node: PriorityQueueNode): void {
        this.items.push(node);
        this.items.sort((a, b) => a.f - b.f);
    }

    dequeue(): PriorityQueueNode | undefined {
        return this.items.shift();
    }

    isEmpty(): boolean {
        return this.items.length === 0;
    }
}

function solveDrone(
    grid: number[][],
    start: Position,
    maxSteps: number
): Position[] {
    const visited = new Set<string>();
    const queue = new PriorityQueue();
    const cameFrom = new Map<string, Position>();
    const costSoFar = new Map<string, number>();

    const startKey = `${start.x},${start.y}`;
    visited.add(startKey);
    costSoFar.set(startKey, 0);
    
    // Initial cost is 0 (start cell).
    // We want to maximize probability, so we try to minimize the negative probability (1 - prob).
    // For the start node, let's assume cost 0.
    queue.enqueue({ x: start.x, y: start.y, f: 0, g: 0 });

    // We stop when we find a path that accumulates "enough" steps, 
    // or simply run Dijkstra until we have a path.
    // Since we don't have a specific target but a step budget, 
    // we interpret the problem as finding the best path within the step limit.
    // A* with a heuristic is best here.
    // Heuristic: The max value in the neighborhood. 
    // Since we want to maximize value, we look for nodes with high "potential".
    
    const rows = grid.length;
    const cols = grid[0].length;
    let bestNode: {x: number, y: number} = start;
    let bestCost = 0;

    while (!queue.isEmpty()) {
        const current = queue.dequeue()!;

        // If we have walked maxSteps, we stop exploring further (reconstruct path to best point found)
        if (current.g >= maxSteps) {
            bestNode = current;
            break;
        }

        // Explore neighbors: Up, Down, Left, Right
        const neighbors = [
            { x: current.x, y: current.y - 1 },
            { x: current.x, y: current.y + 1 },
            { x: current.x - 1, y: current.y },
            { x: current.x + 1, y: current.y }
        ];

        for (const next of neighbors) {
            // Check bounds
            if (next.x < 0 || next.x >= cols || next.y < 0 || next.y >= rows) continue;
            
            // Check obstacles
            if (grid[next.y][next.x] === -1) continue;

            const nextKey = `${next.x},${next.y}`;
            const newCost = current.g + 1;

            // If we haven't visited, or found a cheaper path to a cell
            // (Note: In a purely maximizing reward scenario, we might revisit cells with lower cost, 
            // but standard A* typically avoids cycles. Here, 1-step cycles are inefficient).
            if (!visited.has(nextKey)) {
                visited.add(nextKey);
                cameFrom.set(nextKey, current);
                
                // Cost calculation:
                // We want to minimize steps to reach cells with high probability?
                // No, we want to maximize probability.
                // Let's use a heuristic: 
                // h = grid[next.y][next.x] (Simple immediate gain heuristic)
                // g = newCost (Steps taken)
                // f = g + h
                
                // Actually, to strictly maximize probability within steps, 
                // a simpler approach is: pick the neighbor with the highest probability that we can reach.
                // This is effectively a "Greedy Best-First Search" where the cost is ignored, 
                // but we bound it by steps.
                
                // Let's use the standard A* structure but the heuristic is very aggressive towards high value.
                const h = grid[next.y][next.x]; 
                const f = newCost + h;
                
                queue.enqueue({ x: next.x, y: next.y, f, g: newCost });
            }
        }
    }

    // Reconstruct path
    const path: Position[] = [];
    let curr: {x: number, y: number} = bestNode;
    
    while (curr && (curr.x !== start.x || curr.y !== start.y)) {
        path.unshift(curr);
        const key = `${curr.x},${curr.y}`;
        curr = cameFrom.get(key);
    }
    path.unshift(start); // Add start

    return path;
}

// Main execution logic
function main(): void {
    // Node.js v24.1.1
    // Using the specific require pattern requested, but typed with 'any' as suggested in pattern
    declare const require: any;
    const fs = require('fs');
    const input: string = fs.readFileSync(0, 'utf8');
    
    const data: InputData = JSON.parse(input);

    const rows = data.probability.length;
    const cols = data.probability[0].length;
    
    const paths: Position[][] = [];

    for (const drone of data.drones) {
        const dronePath = solveDrone(
            data.probability,
            { x: drone.x, y: drone.y },
            data.maxStepsPerDrone
        );
        paths.push(dronePath);
    }

    const output = {
        paths: paths.map(path => path.map(p => [p.x, p.y]))
    };

    console.log(JSON.stringify(output));
}

main();
