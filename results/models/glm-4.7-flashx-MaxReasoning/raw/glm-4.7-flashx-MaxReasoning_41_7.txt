; SPIR-V
Version: 1.0
Source: HLSL
Extension: SPV_KHR_shader_ballot

OpCapability Shader
OpCapability GLSL450
OpCapability StorageBuffer
OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_tangent %in_color %ubo
OpExecutionMode %main OriginUpperLeft

; Types
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%void = OpTypeVoid
%func_void = OpTypeFunction %void

; Matrices
%m4float = OpTypeMatrix %v4float 4

; Pointer Types
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Input_v3float_1 = OpTypePointer Input %v3float
%ptr_Input_v3float_2 = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_m4float = OpTypePointer Uniform %m4float
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%ptr_Uniform_v4float_1 = OpTypePointer Uniform %v4float

; Uniform Buffer Structure
%ubo = OpTypeStruct %m4float %m4float %m4float %v4float %v4float %v4float
%ptr_Uniform_ubo = OpTypePointer Uniform %ubo

; Constants
%const_0_2 = OpConstant %float 0.200000003
%const_0_3 = OpConstant %float 0.300000012
%const_0_8 = OpConstant %float 0.800000012
%const_1_0 = OpConstant %float 1.0
%const_0_5 = OpConstant %float 0.5
%const_0_0 = OpConstant %float 0.0

%base_color = OpConstantComposite %v3float %const_0_2 %const_0_3 %const_0_8
%rim_tint = OpConstantComposite %v3float %const_1_0 %const_0_8 %const_0_5

; Variables
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_tangent = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input

%outColor = OpVariable %ptr_Output_v4float Output

%ubo_inst = OpVariable %ptr_Uniform_ubo Uniform
%ubo_ptr_model = OpMemberPointer %ubo 0 %ptr_Uniform_ubo
%ubo_ptr_view = OpMemberPointer %ubo 1 %ptr_Uniform_ubo
%ubo_ptr_proj = OpMemberPointer %ubo 2 %ptr_Uniform_ubo
%ubo_ptr_lightPos = OpMemberPointer %ubo 3 %ptr_Uniform_ubo
%ubo_ptr_cameraPos = OpMemberPointer %ubo 4 %ptr_Uniform_ubo
%ubo_ptr_params = OpMemberPointer %ubo 5 %ptr_Uniform_ubo

; Decoration
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo_inst DescriptorSet 0
OpDecorate %ubo_inst Binding 0
OpDecorate %ubo Block
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224

; Function
%main = OpFunction %void None %func_void
%label = OpLabel

; Calculate V (View Direction)
%cam_x = OpLoad %float (OpCompositeExtract %v4float (OpLoad %v4float %ubo_ptr_cameraPos) 0)
%cam_y = OpCompositeExtract %float (OpLoad %v4float %ubo_ptr_cameraPos) 1
%cam_z = OpCompositeExtract %float (OpLoad %v4float %ubo_ptr_cameraPos) 2
%cam_w = OpCompositeExtract %float (OpLoad %v4float %ubo_ptr_cameraPos) 3

%pos_x = OpCompositeExtract %float (OpLoad %v3float %in_worldPos) 0
%pos_y = OpCompositeExtract %float (OpLoad %v3float %in_worldPos) 1
%pos_z = OpCompositeExtract %float (OpLoad %v3float %in_worldPos) 2

%v_dir_x = OpFSub %float %cam_x %pos_x
%v_dir_y = OpFSub %float %cam_y %pos_y
%v_dir_z = OpFSub %float %cam_z %pos_z
%v_dir = OpCompositeConstruct %v3float %v_dir_x %v_dir_y %v_dir_z

%v_len = OpLength %v3float %v_dir
%v = OpFDiv %v3float %v_dir %v_len

; Calculate L (Light Direction)
%light_x = OpCompositeExtract %float (OpLoad %v4float %ubo_ptr_lightPos) 0
%light_y = OpCompositeExtract %float (OpLoad %v4float %ubo_ptr_lightPos) 1
%light_z = OpCompositeExtract %float (OpLoad %v4float %ubo_ptr_lightPos) 2
%light_w = OpCompositeExtract %float (OpLoad %v4float %ubo_ptr_lightPos) 3

%l_dir_x = OpFSub %float %light_x %pos_x
%l_dir_y = OpFSub %float %light_y %pos_y
%l_dir_z = OpFSub %float %light_z %pos_z
%l_dir = OpCompositeConstruct %v3float %l_dir_x %l_dir_y %l_dir_z

%l_len = OpLength %v3float %l_dir
%l = OpFDiv %v3float %l_dir %l_len

; Calculate Normal
%norm_x = OpCompositeExtract %float (OpLoad %v3float %in_normal) 0
%norm_y = OpCompositeExtract %float (OpLoad %v3float %in_normal) 1
%norm_z = OpCompositeExtract %float (OpLoad %v3float %in_normal) 2
%norm = OpCompositeConstruct %v3float %norm_x %norm_y %norm_z

%n_len = OpLength %v3float %norm
%n = OpFDiv %v3float %norm %n_len

; Calculate Diffuse
%diff = OpDot %v3float %n %l
%diff_clamped = OpNMin %float %diff %const_0_0

; Base Color Modulation
%diff_mul = OpFMul %float %diff_clamped %const_0_5
%diff_add = OpFAdd %float %const_0_3 %diff_mul
%base_mod = OpVectorTimesScalar %v3float %base_color %diff_add

; Calculate Rim
%dot_nv = OpDot %v3float %n %v
%rim_val = OpFSub %float %const_1_0 %dot_nv
%rim_clamped = OpNMin %float %rim_val %const_0_0
%rim_pow = OpFMul %float %rim_clamped %rim_clamped
%rim_pow_2 = OpFMul %float %rim_pow %rim_clamped

; Rim Tint
%rim_tint_col = OpVectorTimesScalar %v3float %rim_tint %rim_pow_2

; Final Color
%final_col = OpFAdd %v3float %base_mod %rim_tint_col

; Store Output
%out_vec = OpCompositeConstruct %v4float %final_col.x %final_col.y %final_col.z %const_1_0
OpStore %outColor %out_vec

OpReturn
OpFunctionEnd
