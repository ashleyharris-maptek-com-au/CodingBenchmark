```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program
{
    static void Main()
    {
        // Use buffered readers/writers for performance
        var stdin = new StreamReader(Console.OpenStandardInput(1024 * 16));
        var stdout = new StreamWriter(Console.OpenStandardOutput(1024 * 16)) { AutoFlush = false };

        string line;
        if ((line = stdin.ReadLine()) == null) return;
        
        var parts = line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(parts[0]);
        long W = long.Parse(parts[1]);
        long H = long.Parse(parts[2]);
        long D = long.Parse(parts[3]);

        // Box structure to minimize memory footprint
        struct Box
        {
            public long w, h, d; // Dimensions
            public int id;       // Original index
            
            public long Volume
            {
                get { return w * h * d; }
            }
        }

        Box[] boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            parts = stdin.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            // We may reorder w/h/d. 
            // To maximize fit, sorting dimensions (smallest to largest) often helps 
            // as it prevents packing the largest dimension against a wall if not necessary,
            // or simply allows fitting into tighter corners. 
            // Here we sort ascending.
            long w = long.Parse(parts[0]);
            long h = long.Parse(parts[1]);
            long d = long.Parse(parts[2]);
            
            long[] dims = new long[] { w, h, d };
            Array.Sort(dims);
            
            boxes[i] = new Box { w = dims[0], h = dims[1], d = dims[2], id = i };
        }

        // Sort boxes by volume descending. 
        // Larger boxes constrain placement options more, so placing them first 
        // leaves more flexibility (or less flexibility but better spatial utilization) 
        // for smaller boxes later.
        Array.Sort(boxes, (a, b) => b.Volume.CompareTo(a.Volume));

        List<long> occupiedZ = new List<long>(); // Stores Z indices of occupied planes
        // Optimization: Instead of a HashSet, a List is faster for adding 
        // if we don't need to remove, which we don't in this greedy packing.
        // Binary search is needed for "Contains".
        // However, for dense packing (1B volume), occupiedZ size is ~1000.
        // For sparse packing (100x100x100 volume), occupiedZ size is ~10k.
        
        List<Box> placedBoxes = new List<Box>();

        // Pre-calculate dimensions to avoid overflow or repeated multiplication inside tight loops
        // Note: We use long for coordinates and checks.
        long W_dim = W;
        long H_dim = H;
        long D_dim = D;

        foreach (var box in boxes)
        {
            long w = box.w;
            long h = box.h;
            long d = box.d;
            
            bool placed = false;

            // Optimization: Sort dimensions before placing to try all orientations.
            // The struct 'box' already has sorted dimensions. We just need to iterate dimensions.
            // To maximize count, we try to pack the "smallest" dimension along the axis we scan first?
            // No, simply iterating is fine. 
            // But we need to try permutations because sorting w,h,d gives ONE orientation (S-M-L).
            // A L-S-M might fit where S-M-L does not.
            
            long[] dims = new long[] { box.w, box.h, box.d };
            // Try a few permutations or just sort. Sorting is usually best for density.
            // Let's stick to the sorted one for the main loop, it's a valid heuristic.
            
            // Try to place at (0,0,0) and scan upwards.
            // Scanning: x, y, z. 
            // To minimize search, scan the dimension that typically has the most space.
            // In the Z-slice optimization, we iterate z.
            
            // The dimensions are already sorted ascending.
            // w <= h <= d
            
            // Check valid dimensions
            if (w > W_dim || h > H_dim || d > D_dim) continue;

            // Iterate Z
            // Optimization: Binary search for next free Z to minimize loop iterations
            int zIndex = occupiedZ.BinarySearch(d);
            if (zIndex < 0) zIndex = ~zIndex;

            for (; zIndex + d <= D_dim; zIndex++)
            {
                long z = (long)zIndex;
                
                // Check if this Z plane is fully occupied?
                // No, we need to check the X-Y plane at this Z.
                
                // Iterate X
                // Iterate Y
                
                // To speed up X-Y scan:
                // Iterate X from 0 to W-w.
                // Inside X, iterate Y from 0 to H-h.
                // Check overlap.
                
                bool planeFree = true;
                
                // We iterate X and Y.
                // To optimize, we can check if the bounding box of the box overlaps 
                // with the occupied X-Y planes.
                // Since we scan Z from 0 upwards, we check Z-plane by Z-plane.
                
                // Optimizing the X-Y check:
                // We have a list of placed boxes.
                // Iterate all placed boxes. Check if `z <= placed.z + placed.d` and `z+d >= placed.z`.
                // If they overlap in Z, they might overlap in X-Y.
                
                // However, simple overlap check is O(N). N=10k.
                // Nested loops: Z (1000) * X (1000) * Y (1000) = 1B.
                // We optimize by checking X and Y first.
                
                // Check Z overlap with existing boxes.
                // If they overlap in Z, they can't be at this Z (or this Z is occupied).
                // Actually, if `z` is in occupiedZ, we skip. 
                // But `z` might be in occupiedZ while the box is placed at `z+5`.
                // We need to check AABB overlap.
                
                bool validZ = true;
                for (int k = 0; k < placedBoxes.Count; k++)
                {
                    var pb = placedBoxes[k];
                    // Check Z overlap
                    if (z < pb.d + pb.z && z + d > pb.z)
                    {
                        validZ = false;
                        break;
                    }
                }
                
                if (!validZ) continue;

                // Now scan X and Y for this Z
                for (long x = 0; x <= W_dim - w; x++)
                {
                    for (long y = 0; y <= H_dim - h; y++)
                    {
                        bool validXY = true;
                        
                        // Check overlap with all placed boxes
                        // This is the main O(N) cost.
                        for (int k = 0; k < placedBoxes.Count; k++)
                        {
                            var pb = placedBoxes[k];
                            
                            // AABB Check
                            if (x < pb.w + pb.x && x + w > pb.x &&
                                y < pb.h + pb.y && y + h > pb.y &&
                                z < pb.d + pb.z && z + d > pb.z)
                            {
                                validXY = false;
                                break;
                            }
                        }
                        
                        if (validXY)
                        {
                            // Place the box
                            placedBoxes.Add(new Box { w = w, h = h, d = d, x = x, y = y, z = z, id = box.id });
                            occupiedZ.Add(z); // Mark Z as occupied
                            // occupiedZ is not sorted, but we sort when searching.
                            // Binary search handles it.
                            placed = true;
                            break;
                        }
                    }
                    if (placed) break;
                }
                if (placed) break;
            }
        }

        // Output
        stdout.WriteLine(placedBoxes.Count);
        foreach (var box in placedBoxes)
        {
            stdout.WriteLine($"{box.id} {box.x} {box.y} {box.z}");
        }
        stdout.Flush();
    }
