```rust
use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    // Read entire input into memory for maximum throughput
    let mut buffer = String::new();
    stdin.lock().read_to_string(&mut buffer).unwrap();

    let mut iter = buffer.split_whitespace();
    
    // Parse dimensions
    let h: usize = iter.next().unwrap().parse().unwrap();
    let w: usize = iter.next().unwrap().parse().unwrap();
    
    let total_cells = h * w;
    
    // Grid encoding:
    // 0: Empty path (' ')
    // 1: Wall ('#')
    // We don't strictly need to differentiate 'A' and 'B' in the grid for logic,
    // but we track their indices separately.
    let mut grid: Vec<u8> = vec![0; total_cells];
    
    // Parent tracking: stores the index of the previous cell in the path
    let mut parents = vec![usize::MAX; total_cells];
    
    let mut start_idx = 0;
    let mut goal_idx = 0;

    // Parse grid rows
    for y in 0..h {
        if let Some(line) = iter.next() {
            // Input lines should be exactly w characters long
            let line = line.as_bytes();
            for (x, &byte) in line.iter().enumerate().take(w) {
                let idx = y * w + x;
                match byte {
                    b'A' => { 
                        grid[idx] = 0; 
                        start_idx = idx; 
                        parents[start_idx] = start_idx; 
                    } 
                    b'B' => { 
                        grid[idx] = 0; 
                        goal_idx = idx; 
                    } 
                    b' ' => { grid[idx] = 0; } 
                    b'#' => { grid[idx] = 1; } 
                    _ => {} // Ignore unexpected characters
                }
            }
        } else {
            panic!("Invalid input format: missing maze rows");
        }
    }

    // Breadth-First Search (BFS)
    let mut queue = VecDeque::new();
    queue.push_back(start_idx);
    
    // Directions: Down, Up, Right, Left
    let dirs = [(1i32, 0i32), (-1i32, 0i32), (0i32, 1i32), (0i32, -1i32)];
    let mut found_goal = false;

    while let Some(idx) = queue.pop_front() {
        let y = idx / w;
        let x = idx % w;

        for &(dy, dx) in dirs.iter() {
            let ny = y as i32 + dy;
            let nx = x as i32 + dx;

            if ny >= 0 && ny < h as i32 && nx >= 0 && nx < w as i32 {
                let nidx = (ny as usize) * w + (nx as usize);
                
                // Check if walkable and not visited
                if grid[nidx] != 1 && parents[nidx] == usize::MAX {
                    parents[nidx] = idx;
                    
                    // Check if this cell is the goal
                    if nidx == goal_idx {
                        found_goal = true;
                        break;
                    }
                    
                    queue.push_back(nidx);
                }
            }
        }
        if found_goal { break; }
    }

    // If no path is found, output nothing
    if !found_goal {
        return;
    }

    // Reconstruct path from Goal to Start
    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut current = goal_idx;
    
    loop {
        let (cy, cx) = (current / w, current % w);
        path.push((cx, cy)); // x y (0-indexed)
        
        if current == start_idx { break; }
        current = parents[current];
    }

    // Reverse to get Start -> Goal order
    path.reverse();

    // Output
    writeln!(out, "{}", path.len()).unwrap();
    for (x, y) in path {
        writeln!(out, "{} {}", x, y).unwrap();
    }
}
```
