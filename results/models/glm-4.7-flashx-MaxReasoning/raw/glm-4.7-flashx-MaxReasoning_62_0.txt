/**
 * Firebreak Optimization
 * 
 * Strategy:
 * 1. Simulate initial fire spread from all ignitions.
 * 2. Identify "Threatened Towns" (reachable from ignitions via burnable cells).
 * 3. If no towns are threatened, return empty list.
 * 4. Perform BFS from each threatened town to find the "Danger Zone" (cells that connect the town to the fire).
 * 5. Treat this Danger Zone as a set cover problem:
 *    - Each cell protects a subset of Threatened Towns.
 *    - We want to select K cells to maximize the total protected town value.
 *    - Use a greedy algorithm: pick the cell covering the most *new* value.
 */

const fs = require('fs');

interface InputData {
    width: number;
    height: number;
    grid: string[];
    ignitions: number[][];
    towns: { id: string; x: number; y: number; value: number }[];
    cutBudget: number;
}

class FirebreakPlanner {
    private data: InputData;
    private grid: (string | undefined)[][]; // 2D array for easier access
    private width: number;
    private height: number;
    private townMap: Map<string, { x: number; y: number; value: number; index: number }>;
    private threatenedTowns: number[]; // Indices of towns that are burning
    private cutBudget: number;
    
    // Danger zone tracking
    // Using a Map for sparse storage if the danger zone is small, 
    // or we could use an array if we knew the range. Given large grids, Map/Set is safer.
    private dangerCells: Map<number, Set<number>>; // cellIndex -> Set of town indices
    
    constructor(data: InputData) {
        this.data = data;
        this.width = data.width;
        this.height = data.height;
        this.cutBudget = data.cutBudget;
        this.grid = [];
        this.townMap = new Map();
        this.dangerCells = new Map();
        this.threatenedTowns = [];

        // Initialize Grid
        for (let y = 0; y < this.height; y++) {
            const row: (string | undefined)[] = [];
            for (let x = 0; x < this.width; x++) {
                row.push(data.grid[y][x]);
            }
            this.grid.push(row);
        }

        // Initialize Towns
        data.towns.forEach((town, idx) => {
            this.townMap.set(town.id, { x: town.x, y: town.y, value: town.value, index: idx });
        });
    }

    solve(): { firebreaks: number[][] } {
        // Step 1: Determine which towns are threatened
        this.identifyThreatenedTowns();

        if (this.threatenedTowns.length === 0) {
            return { firebreaks: [] };
        }

        // Step 2: Determine the Danger Zone
        // Find all cells that connect a threatened town to the fire
        this.findDangerZone();

        // Step 3: Greedy Selection of Firebreaks
        return this.selectFirebreaks();
    }

    private identifyThreatenedTowns(): void {
        const width = this.width;
        const height = this.height;
        
        // 0 = Unburned, 1 = Burning
        const burned = new Uint8Array(width * height);

        // Directions
        const dirs = [
            [1, 0], [-1, 0], [0, 1], [0, -1]
        ];

        // Helper to get index
        const idx = (x: number, y: number) => y * width + x;

        // Initialize queue with all ignitions
        const queue: number[] = [];
        
        for (const [x, y] of this.data.ignitions) {
            const i = idx(x, y);
            if (this.grid[y][x] !== '~' && burned[i] === 0) {
                burned[i] = 1;
                queue.push(i);
            }
        }

        // BFS to spread fire
        while (queue.length > 0) {
            const curr = queue.shift()!;
            const cx = curr % width;
            const cy = Math.floor(curr / width);

            for (const [dx, dy] of dirs) {
                const nx = cx + dx;
                const ny = cy + dy;
                
                if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
                    const ni = idx(nx, ny);
                    const cellType = this.grid[ny][nx];
                    
                    // Fire spreads through '.' and ' ' (if any), blocked by '~' and already burned
                    if (cellType !== '~' && burned[ni] === 0) {
                        // Treat towns as just cells for the spreading mechanic, 
                        // but we mark them burned.
                        burned[ni] = 1;
                        queue.push(ni);
                    }
                }
            }
        }

        // Identify threatened towns (towns that are burned)
        for (const [id, town] of this.townMap) {
            const i = idx(town.x, town.y);
            if (burned[i] === 1) {
                this.threatenedTowns.push(town.index);
            }
        }
    }

    private findDangerZone(): void {
        const width = this.width;
        const height = this.height;
        const dirs = [[1, 0], [-1, 0], [0, 1], [0, -1]];
        
        // Helper to get index
        const idx = (x: number, y: number) => y * width + x;

        // We want to find cells that are NOT water and connect a threatened town to the fire.
        // Effectively, we run BFS from each threatened town towards the fire.
        // We stop at water, cut cells, and boundary.
        
        for (const townIdx of this.threatenedTowns) {
            const town = this.townMap.get(this.data.towns[townIdx].id)!;
            const startIdx = idx(town.x, town.y);
            
            const visited = new Set<number>();
            const queue: number[] = [startIdx];
            visited.add(startIdx);

            while (queue.length > 0) {
                const curr = queue.shift()!;
                const cx = curr % width;
                const cy = Math.floor(curr / width);

                for (const [dx, dy] of dirs) {
                    const nx = cx + dx;
                    const ny = cy + dy;
                    
                    if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
                        const ni = idx(nx, ny);
                        
                        // We only care about burnable cells ('.') that are NOT the town itself
                        const cellType = this.grid[ny][nx];
                        
                        if (cellType === '.' && !visited.has(ni)) {
                            // Add to danger zone
                            // We aggregate by checking if we've added this cell before
                            // To avoid storing the whole map in a simple BFS, we process updates carefully.
                            // Actually, we can just check if it exists in the map. If not, add townIdx.
                            if (!this.dangerCells.has(ni)) {
                                this.dangerCells.set(ni, new Set<number>());
                            }
                            this.dangerCells.get(ni)!.add(townIdx);
                            
                            visited.add(ni);
                            queue.push(ni);
                        }
                    }
                }
            }
        }
    }

    private selectFirebreaks(): { firebreaks: number[][] } {
        const firebreaks: number[][] = [];
        const savedTowns = new Set<number>(); // Indices of towns we have already saved

        // We need to iterate cutBudget times.
        // Since dangerCells might be large, we re-calculate scores? 
        // Better: Sort candidates by initial value, then remove overlaps.
        
        // Flatten candidates
        const candidates: Array<{ cell: number; value: number; townSet: Set<number> }> = [];

        for (const [cellIdx, townIndices] of this.dangerCells) {
            // Calculate total value this cell protects
            let totalVal = 0;
            for (const t of townIndices) {
                totalVal += this.data.towns[t].value;
            }
            candidates.push({
                cell: cellIdx,
                value: totalVal,
                townSet: townIndices
            });
        }

        // Sort by value descending
        candidates.sort((a, b) => b.value - a.value);

        for (const cand of candidates) {
            if (firebreaks.length >= this.cutBudget) break;

            // Calculate new value gained
            let gain = 0;
            for (const t of cand.townSet) {
                if (!savedTowns.has(t)) {
                    gain += this.data.towns[t].value;
                    savedTowns.add(t);
                }
            }

            if (gain > 0) {
                const cx = cand.cell % this.width;
                const cy = Math.floor(cand.cell / this.height);
                firebreaks.push([cx, cy]);
            } else {
                // If this cell doesn't save anything new (shouldn't happen if value>0, 
                // but possible if towns share strictly identical immediate paths or if budget is tiny relative to granularity),
                // we can break early or continue. 
                // Actually, if gain is 0, we might as well stop because remaining cells are equally useless (assuming identical scores).
                break; 
            }
        }

        return { firebreaks };
    }
}

// --- Main Execution ---

declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const planner = new FirebreakPlanner(data);
const result = planner.solve();

console.log(JSON.stringify(result));
