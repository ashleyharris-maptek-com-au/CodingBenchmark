```spirv
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Imports
%100 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Input Per-Vertex
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%uint_3 = OpConstant %uint 3
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User Inputs
%ptr_in_v3 = OpTypePointer Input %v3float
%p_pos = OpVariable %ptr_in_v3 Input
%p_norm = OpVariable %ptr_in_v3 Input
%p_color = OpVariable %ptr_in_v3 Input
OpDecorate %p_color Location 2

; Output Per-Vertex
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%arr_pv_out = OpTypeArray %gl_PerVertex_out %uint_3
%ptr_out_arr_pv = OpTypePointer Output %arr_pv_out
%gl_out = OpVariable %ptr_out_arr_pv Output

; User Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; Loop Indices and Constants
%uint_1 = OpConstant %uint 1
%uint_0 = OpConstant %uint 0
%uint_2 = OpConstant %uint 2
%i = OpVariable %ptr_in_uint Input
OpDecorate %i Location 0
%float_1 = OpConstant %float 1
%float_3 = OpConstant %float 3
%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.08

; Pointers
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

; Main Function
%main = OpFunction %void None %func
%label = OpLabel

; --- Calculate Face Normal (once) ---
OpBranch %calc_normal

%calc_normal = OpLabel
%idx_0 = OpConstant %uint 0
%idx_1 = OpConstant %uint 1
%idx_2 = OpConstant %uint 2

%ptr_p0 = OpAccessChain %ptr_in_pv %gl_in %idx_0
%val_p0 = OpLoad %gl_PerVertex_in %ptr_p0
%ptr_pos0 = OpAccessChain %ptr_in_v4 %val_p0 %uint_0
%p0 = OpLoad %v4float %ptr_pos0

%ptr_p1 = OpAccessChain %ptr_in_pv %gl_in %idx_1
%val_p1 = OpLoad %gl_PerVertex_in %ptr_p1
%ptr_pos1 = OpAccessChain %ptr_in_v4 %val_p1 %uint_0
%p1 = OpLoad %v4float %ptr_pos1

%ptr_p2 = OpAccessChain %ptr_in_pv %gl_in %idx_2
%val_p2 = OpLoad %gl_PerVertex_in %ptr_p2
%ptr_pos2 = OpAccessChain %ptr_in_v4 %val_p2 %uint_0
%p2 = OpLoad %v4float %ptr_pos2

%e1 = OpFSub %v4float %p1 %p0
%e2 = OpFSub %v4float %p2 %p0
%e1_xyz = OpVectorShuffle %v3float %e1 %e1 0 1 2
%e2_xyz = OpVectorShuffle %v3float %e2 %e2 0 1 2
%N = OpExtInst %v3float 68 %e1_xyz %e2_xyz
%N_norm = OpExtInst %v3float 69 %N

%ptr_N = OpVariable %ptr_out_v3 Output
OpStore %ptr_N %N_norm
OpBranch %loop_cond

%loop_cond = OpLabel
OpBranch %loop_body

%loop_body = OpLabel
%i_phi = OpPhi %uint %uint_0 %calc_normal %i_next %loop_inc
OpBranch %calc_emit

%calc_emit = OpLabel
; --- Load N ---
%N_val = OpLoad %v3float %ptr_N
%offset = OpFMul %v3float %N_val %float_0_08
%offset_x = OpCompositeExtract %float %offset 0
%offset_y = OpCompositeExtract %float %offset 1
%offset_z = OpCompositeExtract %float %offset 2
%offset_vec4 = OpCompositeConstruct %v4float %offset_x %offset_y %offset_z %float_0

; --- Load Position ---
%ptr_cur = OpAccessChain %ptr_in_pv %gl_in %i_phi
%val_cur = OpLoad %gl_PerVertex_in %ptr_cur
%ptr_cur_pos = OpAccessChain %ptr_in_v4 %val_cur %uint_0
%p_i = OpLoad %v4float %ptr_cur_pos

; --- New Position ---
%gl_newPos = OpFAdd %v4float %p_i %offset_vec4

; --- Store ---
%ptr_gl_out = OpAccessChain %ptr_out_pv %gl_out %i_phi
%val_gl_out = OpLoad %gl_PerVertex_out %ptr_gl_out
%ptr_gl_out_pos = OpAccessChain %ptr_out_v4 %val_gl_out %uint_0
OpStore %ptr_gl_out_pos %gl_newPos

; --- Store Color ---
%ptr_col = OpAccessChain %ptr_in_v3 %p_color %i_phi
%col = OpLoad %v3float %ptr_col
OpStore %outColor %col

OpEmitVertex
OpBranch %loop_inc

%loop_inc = OpLabel
%i_next = OpIAdd %uint %i_phi %uint_1
OpBranch %loop_cond

%loop_merge = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
```
