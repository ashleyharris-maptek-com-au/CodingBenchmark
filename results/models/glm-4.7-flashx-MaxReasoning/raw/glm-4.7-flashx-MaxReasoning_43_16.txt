```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Map UVs to Mandelbrot complex plane coordinates
    // c = (u*3 - 2, v*2 - 1)
    float cx = input.uv.x * 3.0 - 2.0;
    float cy = input.uv.y * 2.0 - 1.0;

    float zx = 0.0;
    float zy = 0.0;
    float iter = 0.0;
    const int MAX_ITER = 20;

    while (iter < MAX_ITER) {
        // z = z^2 + c
        float xtemp = zx * zx - zy * zy + cx;
        zy = 2.0 * zx * zy + cy;
        zx = xtemp;

        // Escape when |z|^2 > 4
        if ((zx * zx + zy * zy) > 4.0) {
            break;
        }
        iter++;
    }

    // Calculate normalized iteration count t
    float t = iter / 20.0;

    // Color with (t, t*0.5, 1.0-t), alpha 1.0
    float3 color = float3(t, t * 0.5, 1.0 - t);

    return float4(color, 1.0);
}
```
