import math

# --- Persistence Variables (Module Level) ---
# These allow the PID controllers to remember state across function calls.
class PIDController:
    def __init__(self, kp, ki, kd, max_output=1.0):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.max_output = max_output
        self.integral = 0.0
        self.prev_error = 0.0

    def update(self, error, dt):
        self.integral += error * dt
        self.integral = max(min(self.integral, 10.0), -10.0) # Anti-windup
        derivative = (error - self.prev_error) / dt if dt > 0 else 0
        output = self.kp * error + self.ki * self.integral + self.kd * derivative
        self.prev_error = error
        return max(min(output, self.max_output), -self.max_output)

# Initialize PIDs
pid_pitch = PIDController(0.8, 0.05, 1.5, 1.0)      # Vertical speed / Pitch
pid_roll = PIDController(1.2, 0.0, 3.0, 1.0)        # Heading / Roll
pid_yaw = PIDController(0.5, 0.01, 4.0, 1.0)        # Yaw damping / Heading

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Main Autopilot Logic.
    """
    
    # --- 1. Sensor Fusion & State Estimation ---
    
    def get_sensors_3(sensor_base, suffix, valid_suffix):
        """Fuses 3 redundant sensors. Returns (value, valid_count, confidence)."""
        vals = [getattr(state, f"{sensor_base}{i}{suffix}", 0) for i in range(1, 4)]
        valids = [getattr(state, f"{sensor_base}{i}{valid_suffix}", True) for i in range(1, 4)]
        
        # Filter out known failures (valid flag usually reliable, but we handle silent fail by median)
        valid_vals = [v for v, ok in zip(vals, valids) if ok]
        
        if len(valid_vals) >= 2:
            # 2 or 3 valid: Average
            return sum(valid_vals) / len(valid_vals), len(valid_vals)
        elif len(valid_vals) == 1:
            # 1 valid: Return it (conf=1)
            return valid_vals[0], 1
        else:
            # 0 valid: Median of all raw values (robust to silent failures)
            return sorted(vals)[1], 0

    # Estimate Primary Flight Data
    ias, ias_c = get_sensors_3("ias", "_kt", "_valid")
    aoa, aoa_c = get_sensors_3("aoa", "_deg", "_valid")
    pitch, pitch_c = get_sensors_3("irs", "_pitch_deg", "_valid")
    roll, roll_c = get_sensors_3("irs", "_roll_deg", "_valid")
    pitch_rate, pr_c = get_sensors_3("gyro", "_pitch_rate_dps", "_valid")
    roll_rate, rr_c = get_sensors_3("gyro", "_roll_rate_dps", "_valid")
    
    # Alternative sensors for cross-check
    tas, tas_c = get_sensors_3("tas", "_kt", None)
    mag_hdg, hdg_c = get_sensors_3("irs", "_heading_deg", "_valid")
    vs, vs_c = get_sensors_3("irs", "_vs_fpm", "_valid")
    
    # Radar Altitude (Ground proximity)
    radar_alt, _ = get_sensors_3("radar_alt", "_ft", "_valid")
    
    # Thrust (Engine status)
    n1_1 = getattr(state, "engine_1_n1_pct", 0)
    n1_2 = getattr(state, "engine_2_n1_pct", 0)
    eng1_run = getattr(state, "engine_1_running", False)
    eng2_run = getattr(state, "engine_2_running", False)

    # --- 2. Failure Detection & Accident Recovery Logic ---
    
    is_stalled = (aoa > 20) or (ias < 80)
    is_double_engine_failure = (not eng1_run) and (not eng2_run)
    is_single_engine_failure = (not eng1_run) or (not eng2_run)
    is_ground_proximity = (radar_alt < 500) and (radar_alt > 0)
    is_upset = (roll_rate > 10 or roll_rate < -10) and (abs(roll) > 45)

    # --- 3. Control Surface Logic ---

    # --- Throttles ---
    # Logic: IAS Hold + V/S Hold. 
    # Accident Logic: Double Engine Failure -> 0 Thrust.
    throttle_1 = getattr(state, "throttle_1_pos", 0.0)
    throttle_2 = getattr(state, "throttle_2_pos", 0.0)
    target_ias = getattr(state, "target_ias_kt", 240)
    
    if is_double_engine_failure:
        throttle_1 = 0.0
        throttle_2 = 0.0
        # Low speed recovery for engine failure: Pitch down to regain IAS, Flaps Down
    else:
        # Simple IAS control (PID-like)
        ias_error = target_ias - ias
        throttle_inc = ias_error * 0.01
        
        # Add thrust if falling too fast
        target_vs = getattr(state, "target_vs_fpm", 0)
        vs_error = target_vs - vs
        
        # If sinking and throttle is maxed, pitch down logic is handled elsewhere
        # If sinking and throttle available, add it
        if vs_error > 0 and (throttle_1 + throttle_2) < 1.9:
             throttle_1 += min(throttle_inc, 0.1)
             throttle_2 += min(throttle_inc, 0.1)

        throttle_1 = max(0.0, min(1.0, throttle_1))
        throttle_2 = max(0.0, min(1.0, throttle_2))

    # --- Flaps & Gear ---
    gear_handle = getattr(state, "gear_handle_down", False)
    gear_state = 1 if gear_handle else 0
    
    # Auto-flap logic based on speed and altitude
    target_flaps = 0
    if gear_handle:
        target_flaps = 2 # Approach config
        if radar_alt < 500:
            target_flaps = 5
    elif not gear_handle:
        if ias > 280:
            target_flaps = 0
        elif ias > 250:
            target_flaps = 1
        elif ias > 200:
            target_flaps = 2
        elif ias > 180:
            target_flaps = 5
            
    # Speedbrake logic
    target_speedbrake = 0.0
    if ias > target_ias * 1.2:
        target_speedbrake = 1.0
    if ias < target_ias * 0.8 and not is_double_engine_failure:
        target_speedbrake = 0.0

    # --- Elevator (Pitch) ---
    # Logic: Pitch Rate + Angle of Attack + Vertical Speed
    # Recover from Stall (High AOA)
    
    elevator_output = 0.0
    
    if is_stalled:
        # Aggressive recovery: Pull nose up (Negative elevator relative to trim)
        elevator_output = -1.0 # Commanding -1 implies full nose up relative to current setup
        # If engine failure, pitch down to recover airspeed
        if is_double_engine_failure:
             elevator_output = 0.2 # Nose down
            
    else:
        # Normal PID control
        # Desired Pitch: Derived from Vertical Speed Target
        # V/S ~ q * cos(alpha). Simplified: Pitch rate setpoint
        desired_pitch_rate = 0.5 # Default moderate climb
        
        # Override pitch command if we are sinking or climbing fast
        current_vs = vs / 60.0 # convert fpm to deg/s approx
        if current_vs < 0.5: # Climbing too fast
             desired_pitch_rate = 0.2
        elif current_vs > 5.0: # Falling fast
             desired_pitch_rate = 0.8
             
        pitch_error = desired_pitch_rate - pitch_rate
        elevator_output = pid_pitch.update(pitch_error, dt)
        
        # Feedback Trim adjust (Autopilot compensation)
        elevator_trim = getattr(state, "elevator_trim_deg", 0.0)
        
    # --- Aileron (Roll) ---
    # Logic: Roll Rate Damping + Heading Hold
    
    target_hdg = getattr(state, "target_hdg_deg", mag_hdg)
    heading_error = target_hdg - mag_hdg
    heading_error = (heading_error + 180) % 360 - 180 # Normalize to -180, 180
    
    roll_error = 0.0 # Target bank angle
    
    # Accident Recovery: Level Wings
    if is_upset or is_double_engine_failure:
        roll_error = 0 - roll
    else:
        # Desired bank for heading change: 1 deg per degree heading error approx
        # Avoid over-banking in cruise
        desired_bank = min(max(heading_error * 2, 5), 45) 
        roll_error = desired_bank - roll

    # Crosswind / Yaw Asymmetry correction (Simplified)
    yaw_error = getattr(state, "drift_angle_deg", 0) * 5 # Exaggerate for effect
    
    aileron_output = pid_roll.update(roll_error, dt)
    
    # Limit aileron rate for realism
    aileron_output = max(min(aileron_output, 1.0), -1.0)

    # --- Rudder (Yaw) ---
    # Logic: Yaw Rate Damping + Sideslip compensation
    
    yaw_rate, yr_c = get_sensors_3("gyro", "_yaw_rate_dps", "_valid")
    
    rudder_output = pid_yaw.update(yaw_rate, dt)
    
    # Add crosswind correction
    # If we have speed error, we use elevator. If we have sideslip, rudder.
    rudder_output += yaw_error * 0.1

    # --- Final Output Assembly ---
    # Elevator is relative to trim (-1 to 1).
    # If we are at limit (-1), it effectively overrides trim.
    
    # Map generic -1..1 elevator logic to output
    # Note: The prompt asks for elevator relative to trim.
    # We compute the raw elevator command. 
    # The system (sim) subtracts elevator_trim to get the actual deflection.
    # So we output the deviation.
    
    elevator_command = elevator_output 

    return {
        "elevator": elevator_command,
        "aileron": aileron_output,
        "rudder": rudder_output,
        "throttle_1": throttle_1,
        "throttle_2": throttle_2,
        "elevator_trim": getattr(state, "elevator_trim_deg", 0.0),
        "flaps": target_flaps,
        "speedbrake": target_speedbrake,
        "gear": gear_state
    }
