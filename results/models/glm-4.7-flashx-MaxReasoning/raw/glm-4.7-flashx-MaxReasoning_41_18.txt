; SPIR-V
Version 1.3
Generator 239
Bound 60
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float 4
%ubo = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %ubo
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_in_vec2 = OpTypePointer Input %vec4
%ptr_in_vec3_t = OpTypePointer Input %vec3
%ptr_in_vec3_c = OpTypePointer Input %vec3
%ptr_out_vec4 = OpTypePointer Output %vec4
%func = OpTypeFunction void

%c0 = OpConstant %float 0
%c1 = OpConstant %float 1
%c2 = OpConstant %float 2
%c05 = OpConstant %float 0.5
%ui0 = OpConstant %uint 0
%ui4 = OpConstant %uint 4

%in_worldPos = OpVariable %ptr_in_vec3 Input
%in_normal = OpVariable %ptr_in_vec3 Input
%in_uv = OpVariable %ptr_in_vec2 Input
%in_tangent = OpVariable %ptr_in_vec3_t Input
%in_color = OpVariable %ptr_in_vec3_c Input
%ubo_var = OpVariable %ptr_ubo Uniform
%outColor = OpVariable %ptr_out_vec4 Output

%main = OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main OriginUpperLeft

%main_body = OpFunction void None %func
%bb0 = OpLabel
        %cam_ptr = OpAccessChain %ptr_in_vec2 %ubo_var %ui4
        %cam = OpLoad %vec4 %cam_ptr
        %cam_vec = OpVectorShuffle %vec3 %cam %cam 0 1 2

        %pos = OpLoad %vec3 %in_worldPos
        %nrm = OpLoad %vec3 %in_normal

        %dir = OpFSub %vec3 %cam_vec %pos
        %dir = OpNormalize %vec3 %dir

        %ndotv = OpDot %float %nrm %dir
        %ndotv = OpFMax %float %ndotv %c0

        %cond = OpFOrdLessThan %bool %ndotv %c05
        OpBranchConditional %cond %bb1 %bb2

%bb1 = OpLabel
        %val = OpCompositeConstruct %vec4 %c1 %c2 %c0 %c1
        OpBranch %bb3

%bb2 = OpLabel
        %t = OpFSub %float %ndotv %c05
        %t2 = OpFMul %float %t %c2
        %val = OpCompositeConstruct %vec4 %c1 %c1 %t2 %c1
        OpBranch %bb3

%bb3 = OpLabel
        %final = OpPhi %vec4 %val %bb1 %val %bb2
        OpStore %outColor %final
        OpReturn
OpFunctionEnd
