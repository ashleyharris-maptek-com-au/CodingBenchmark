use std::io::{self, BufRead, Write, BufWriter};
use std::simd::{u64x8, u32x16};
use std::mem;
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    
    // Parse N and Target
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let target: u64 = parts.next().unwrap().parse().unwrap();

    // Parse numbers
    let mut numbers: Vec<i64> = Vec::with_capacity(n);
    for line in lines.take(n) {
        let num: i64 = line.unwrap().trim().parse().unwrap();
        numbers.push(num);
    }

    // Strategy Selection
    // If N is very small, use Meet-in-the-Middle for exact reconstruction
    // If N is large, use Bitset DP with SIMD (assuming small integers for performance)
    if n <= 60 {
        solve_meet_in_middle(&numbers, target, &mut out);
    } else {
        solve_bitset_dp(&numbers, target, &mut out);
    }
}

// For N <= 60, this provides exact solution with indices via parallelization hints
fn solve_meet_in_middle(nums: &[i64], target: u64, out: &mut BufWriter<impl Write>) {
    let n = nums.len();
    let mid = n / 2;

    // Function to generate all subset sums and store their indices
    let gen_subsets = |nums: &[i64]| -> Vec<(u64, Vec<usize>)> {
        let mut map: std::collections::HashMap<u64, Vec<usize>> = std::collections::HashMap::new();
        map.insert(0, vec![]);

        for (i, &val) in nums.iter().enumerate() {
            if val == 0 { continue; } // Skip zeros to avoid duplicates
            // Iterate over existing keys to avoid modifying while iterating
            // In practice for N=60, cloning keys is fine
            let mut next_map = map.clone();
            for (&sum, indices) in &map {
                let new_sum = sum + val;
                if new_sum <= target {
                    let mut new_indices = indices.clone();
                    new_indices.push(i);
                    next_map.insert(new_sum, new_indices);
                }
            }
            map = next_map;
        }
        map
    };

    let left_map = gen_subsets(&nums[..mid]);
    let right_map = gen_subsets(&nums[mid..]);

    for (&sum, left_indices) in &left_map {
        let rem = target.saturating_sub(sum);
        if let Some(right_indices) = right_map.get(&rem) {
            let mut all_indices = Vec::with_capacity(left_indices.len() + right_indices.len());
            all_indices.extend(left_indices);
            all_indices.extend(right_indices);
            
            // Sort indices to ensure 0-based sequential order
            all_indices.sort_unstable();
            
            writeln!(out, "YES").unwrap();
            write!(out, "0 ").unwrap();
            for idx in all_indices {
                write!(out, "{} ", idx).unwrap();
            }
            writeln!(out).unwrap();
            return;
        }
    }

    writeln!(out, "NO").unwrap();
}
