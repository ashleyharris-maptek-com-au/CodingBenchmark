```csharp
using System;
using System.Text;

class Program
{
    // Struct to hold the cut length and its original index
    struct Cut
    {
        public int Length;
        public int OriginalIndex;
    }

    static void Main()
    {
        // 1. Read Input
        // Line 1: N stock_length
        string line1 = Console.ReadLine();
        if (string.IsNullOrEmpty(line1))
            return;

        string[] parts = line1.Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(parts[0]);
        int stockLength = int.Parse(parts[1]);

        // Handle edge case of 0 cuts
        if (n == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Line 2: N cut lengths separated by spaces
        string line2 = Console.ReadLine();
        string[] cutStrings = line2.Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        
        Cut[] cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i].Length = int.Parse(cutStrings[i]);
            cuts[i].OriginalIndex = i;
        }

        // 2. Sort Cuts Descending (First Fit Decreasing Heuristic)
        // Comparison delegate to sort by Length descending
        Comparison<Cut> cmp = delegate(Cut c1, Cut c2)
        {
            return c2.Length.CompareTo(c1.Length);
        };

        Array.Sort(cuts, cmp);

        // 3. Pack Cuts into Stocks
        int[] stockAssignment = new int[n]; // stores which stock (1-based) each original index belongs to
        int currentStock = 1;
        int remainingSpace = stockLength;

        for (int i = 0; i < n; i++)
        {
            if (cuts[i].Length > remainingSpace)
            {
                // Start a new stock
                currentStock++;
                remainingSpace = stockLength - cuts[i].Length;
            }

            // Assign the original index to the current stock
            stockAssignment[cuts[i].OriginalIndex] = currentStock;
        }

        // 4. Output Results
        // We need to output: num_stocks, then lists of indices for each stock
        // We assume stock numbers are contiguous starting from 1.
        
        int stocksUsed = currentStock;
        
        // StringBuilder for efficient output
        StringBuilder sb = new StringBuilder(stocksUsed * 10);
        sb.Append(stocksUsed).AppendLine();

        int prevStock = -1;
        for (int i = 0; i < n; i++)
        {
            int stockId = stockAssignment[i];
            
            if (stockId != prevStock)
            {
                // New stock line
                if (prevStock != -1)
                {
                    sb.AppendLine();
                }
                sb.Append(i.ToString());
                prevStock = stockId;
            }
            else
            {
                // Continue current stock line
                sb.Append(" ").Append(i.ToString());
            }
        }

        Console.Write(sb);
    }
}
```
