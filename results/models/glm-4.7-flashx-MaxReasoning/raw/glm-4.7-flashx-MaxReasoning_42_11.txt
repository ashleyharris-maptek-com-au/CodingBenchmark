```
OpCapability Geometry
OpCapability Shader
OpCapability StorageBufferStorageClass
OpExtension "SPV_KHR_vulkan_memory_model"
OpMemoryModel Logical VulkanKHR GLSL450

%void = OpTypeVoid
%func = OpTypeFunction %void %main

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4x4 = OpTypeMatrix %float 4 4

%int = OpTypeInt 32 0
%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%zero = OpConstant %float 0
%one = OpConstant %float 1
%half_thickness = OpConstant %float 0.01
%one_third = OpConstant %float 0.333333343
%two_thirds = OpConstant %float 0.666666687

; Input Struct
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input Arrays
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_pv Input

%ptr_in_arr = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr Input
%normal = OpVariable %ptr_in_arr Input
%color = OpVariable %ptr_in_arr Input

; Output Struct
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_3 Output

; UBO
%ubo_type = OpTypeStruct %mat4x4 %mat4x4 %mat4x4 %v3float %v3float %v3float
OpDecorate %ubo_type Block
OpMemberDecorate %ubo_type 0 Offset 0
OpMemberDecorate %ubo_type 1 Offset 64
OpMemberDecorate %ubo_type 2 Offset 128
OpMemberDecorate %ubo_type 3 Offset 192
OpMemberDecorate %ubo_type 4 Offset 208
OpMemberDecorate %ubo_type 5 Offset 224
OpDecorate %ubo_type Binding 0
OpDecorate %ubo_type Set 0
%ptr_ubo = OpTypePointer Uniform %ubo_type
%ubo = OpVariable %ptr_ubo Uniform

; Vectors
%v3_zero = OpConstantComposite %v3float %zero %zero %zero
%v3_one = OpConstantComposite %v3float %one %one %one
%v3_up = OpConstantComposite %v3float %zero %one %zero

%main = OpFunction %void None %func
%entry = OpLabel

; Access Inputs
%pos_1_ptr = OpAccessChain %ptr_in_pv %gl_in %int_1 %int_0
%pos_1 = OpLoad %v4float %pos_1_ptr

%pos_2_ptr = OpAccessChain %ptr_in_pv %gl_in %int_2 %int_0
%pos_2 = OpLoad %v4float %pos_2_ptr

%col_1_ptr = OpAccessChain %ptr_in_arr %color %int_1
%col_1 = OpLoad %v3float %col_1_ptr

%col_2_ptr = OpAccessChain %ptr_in_arr %color %int_2
%col_2 = OpLoad %v3float %col_2_ptr

%cam_pos_ptr = OpAccessChain %ptr_ubo %ubo %int_4
%cam_pos = OpLoad %v3float %cam_pos_ptr

; Calculate Direction and Tangent
%tangent_vec = OpFSub %v4float %pos_2 %pos_1
%tangent_vec_3d = OpVectorShuffle %v3float %tangent_vec %tangent_vec 0 1 2 0
%dir = OpNormalize %v3float %tangent_vec_3d

; Calculate Normal (perpendicular to tangent and camera)
%diff_cam = OpFSub %v3float %cam_pos %tangent_vec_3d
%norm_cam = OpNormalize %v3float %diff_cam
%curve_normal = OpNormalize %v3float OpCross %dir %norm_cam

; --- Segment 1: 0 to 1/3 ---
OpBranch %seg1_loop

%seg1_loop = OpLabel
OpSelectionMerge %seg1_end None
OpBranchConditional %true %seg1_t0 %seg1_t13

%seg1_t0 = OpLabel
; t = 0
%pos_center = OpFAdd %v3float %pos_1 %tangent_vec_3d
%col_center = OpFAdd %v3float %col_1 OpFMul %v3float OpFSub %v3float %col_2 %col_1 %zero
%off_left = OpFMul %v3float %curve_normal OpFMul %v3float %v3_one OpFMul %v3float %half_thickness
%off_right = OpFMul %v3float %curve_normal OpFMul %v3float %v3_one OpFMul %v3float %half_thickness

; Emit Bottom (Left)
%pos_emit = OpFAdd %v3float %pos_center %off_left
; Store Position
%out_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos_ptr %pos_emit
; Store Color
%out_col_ptr = OpAccessChain %ptr_out_3 %outColor %int_0
OpStore %out_col_ptr %col_center
OpEmitVertex

; Emit Top (Right)
%pos_emit = OpFAdd %v3float %pos_center %off_right
OpStore %out_pos_ptr %pos_emit
OpStore %out_col_ptr %col_center
OpEmitVertex

OpBranch %seg1_t13

%seg1_t13 = OpLabel
; t = 1/3
%t_vec = OpFMul %v3float %dir %one_third
%pos_center = OpFAdd %v3float %pos_1 %t_vec
%col_center = OpFAdd %v3float %col_1 OpFMul %v3float OpFSub %v3float %col_2 %col_1 %one_third

%pos_emit = OpFAdd %v3float %pos_center %off_left
OpStore %out_pos_ptr %pos_emit
OpStore %out_col_ptr %col_center
OpEmitVertex

%pos_emit = OpFAdd %v3float %pos_center %off_right
OpStore %out_pos_ptr %pos_emit
OpStore %out_col_ptr %col_center
OpEmitVertex

%seg1_end = OpLabel
OpBranch %split

; --- Split ---
%split = OpLabel
OpEndPrimitive
OpBranch %seg2_loop

; --- Segment 2: 2/3 to 1 ---
%seg2_loop = OpLabel
OpSelectionMerge %seg2_end None
OpBranchConditional %true %seg2_t23 %seg2_t1

%seg2_t23 = OpLabel
; t = 2/3
%t_vec = OpFMul %v3float %dir %two_thirds
%pos_center = OpFAdd %v3float %pos_1 %t_vec
%col_center = OpFAdd %v3float %col_1 OpFMul %v3float OpFSub %v3float %col_2 %col_1 %two_thirds

%pos_emit = OpFAdd %v3float %pos_center %off_left
OpStore %out_pos_ptr %pos_emit
OpStore %out_col_ptr %col_center
OpEmitVertex

%pos_emit = OpFAdd %v3float %pos_center %off_right
OpStore %out_pos_ptr %pos_emit
OpStore %out_col_ptr %col_center
OpEmitVertex

OpBranch %seg2_t1

%seg2_t1 = OpLabel
; t = 1
%pos_center = OpFAdd %v3float %pos_1 %tangent_vec_3d
%col_center = OpFAdd %v3float %col_1 OpFMul %v3float OpFSub %v3float %col_2 %col_1 %one_third

%pos_emit = OpFAdd %v3float %pos_center %off_left
OpStore %out_pos_ptr %pos_emit
OpStore %out_col_ptr %col_center
OpEmitVertex

%pos_emit = OpFAdd %v3float %pos_center %off_right
OpStore %out_pos_ptr %pos_emit
OpStore %out_col_ptr %col_center
OpEmitVertex

%seg2_end = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
```
