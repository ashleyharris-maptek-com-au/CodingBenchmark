struct Vec3 {
    float x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(float x_, float y_, float z_) : x(x_), y(y_), z(z_) {}
    
    Vec3 operator+(const Vec3& other) const { return Vec3(x + other.x, y + other.y, z + other.z); }
    Vec3 operator-(const Vec3& other) const { return Vec3(x - other.x, y - other.y, z - other.z); }
    Vec3 operator*(float s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(float s) const { return Vec3(x / s, y / s, z / s); }
    float dot(const Vec3& other) const { return x * other.x + y * other.y + z * other.z; }
    Vec3 cross(const Vec3& other) const { return Vec3(y * other.z - z * other.y, z * other.x - x * other.z, x * other.y - y * other.x); }
    float length() const { return sqrt(x*x + y*y + z*z); }
    Vec3 normalize() const { float l = length(); return l > EPSILON ? *this / l : Vec3(); }
};

struct AABB {
    Vec3 min, max;
    AABB() : min(INFINITY), max(-INFINITY) {}
    AABB(const Vec3& min_, const Vec3& max_) : min(min_), max(max_) {}
    float surfaceArea() const {
        Vec3 d = max - min;
        return 2 * (d.x*d.y + d.y*d.z + d.z*d.x);
    }
    bool contains(const Vec3& p) const {
        return p.x >= min.x && p.x <= max.x && p.y >= min.y && p.y <= max.y && p.z >= min.z && p.z <= max.z;
    }
    bool overlaps(const AABB& other) const {
        return !(max.x < other.min.x || min.x > other.max.x ||
                 max.y < other.min.y || min.y > other.max.y ||
                 max.z < other.min.z || min.z > other.max.z);
    }
};

// BVH Construction
// Uses a simple SAH (Surface Area Heuristic) recursive build
struct BVHNode {
    AABB bounds;
    int left;   // index in bvh_nodes or -1 if leaf
    int right;  // index in bvh_nodes or primitive index if leaf
    int count;  // number of primitives if leaf
    int start;  // primitive start index if leaf
};

Mesh buildBVH(const Mesh& mesh) {
    Mesh bvhMesh = mesh; // Copy vertices and faces to avoid modifying original
    int n = (int)mesh.faces.size();
    if (n == 0) return bvhMesh;

    vector<BVHNode> nodes;
    
    // Leaf node creation helper
    auto createLeaf = [&](int start, int count) -> int {
        AABB bounds;
        for (int i = 0; i < count; i++) {
            const Face& f = bvhMesh.faces[start + i];
            bounds = bounds.unionWith(bvhMesh.vertices[f.a]);
            bounds = bounds.unionWith(bvhMesh.vertices[f.b]);
            bounds = bounds.unionWith(bvhMesh.vertices[f.c]);
        }
        BVHNode node;
        node.bounds = bounds;
        node.start = start;
        node.count = count;
        node.left = -1;
        node.right = -1;
        nodes.push_back(node);
        return (int)nodes.size() - 1;
    };

    // Recursive build
    function<int(int, int)> build = [&](int start, int count) -> int {
        AABB bounds;
        for (int i = 0; i < count; i++) {
            const Face& f = bvhMesh.faces[start + i];
            bounds = bounds.unionWith(bvhMesh.vertices[f.a]);
            bounds = bounds.unionWith(bvhMesh.vertices[f.b]);
            bounds = bounds.unionWith(bvhMesh.vertices[f.c]);
        }

        if (count <= 4) {
            return createLeaf(start, count);
        }

        // Find split position
        int axis = 0;
        float lenX = bounds.max.x - bounds.min.x;
        float lenY = bounds.max.y - bounds.min.y;
        float lenZ = bounds.max.z - bounds.min.z;

        if (lenY > lenX && lenY > lenZ) axis = 1;
        else if (lenZ > lenX && lenZ > lenY) axis = 2;

        // Sort faces by centroid
        sort(bvhMesh.faces.begin() + start, bvhMesh.faces.begin() + start + count, [&](const Face& a, const Face& b) {
            Vec3 ca = (bvhMesh.vertices[a.a] + bvhMesh.vertices[a.b] + bvhMesh.vertices[a.c]) / 3.0f;
            Vec3 cb = (bvhMesh.vertices[b.a] + bvhMesh.vertices[b.b] + bvhMesh.vertices[b.c]) / 3.0f;
            return ca[axis] < cb[axis];
        });

        int mid = start + count / 2;
        int left = build(start, mid - start);
        int right = build(mid, count - (mid - start));

        BVHNode node;
        node.bounds = bounds;
        node.left = left;
        node.right = right;
        nodes.push_back(node);
        return (int)nodes.size() - 1;
    };

    int root = build(0, n);
    bvhMesh.bvh_nodes = nodes;
    return bvhMesh;
}

// SDF Evaluation
// Returns signed distance to the mesh. 
// Positive inside, Negative outside.
float computeSDF(const Mesh& bvhMesh, const Vec3& p) {
    // Naive ray march to find closest face?
    // Better: Traverse BVH to find closest point on mesh.
    
    // Since we need min(SDF_A, SDF_B), we find closest point Q on mesh A and B.
    // Dist = |Q - p|.
    // Sgn = sign(dot(Q - v0, n))? 
    // Robust SDF: Check if Q is inside mesh.
    
    // For the scope of this solver, we can simplify:
    // 1. Find closest point Q on mesh.
    // 2. Check if Q is inside.
    
    // To find closest point efficiently:
    // We traverse the BVH. If node bounds are far, skip.
    // If leaf, check all faces.
    
    float bestDist = INFINITY;
    Vec3 bestPoint;
    bool found = false;

    const vector<AABB>& nodes = bvhMesh.bvh_nodes;
    const vector<Face>& faces = bvhMesh.faces;
    const vector<Vec3>& verts = bvhMesh.vertices;

    stack<int> s;
    s.push(0);

    while (!s.empty()) {
        int idx = s.top();
        s.pop();

        const BVHNode& node = nodes[idx];

        // If leaf
        if (node.left == -1) {
            for (int i = 0; i < node.count; i++) {
                const Face& f = faces[node.start + i];
                const Vec3& v0 = verts[f.a];
                const Vec3& v1 = verts[f.b];
                const Vec3& v2 = verts[f.c];

                // Distance to triangle (clamped to plane)
                // Project p onto triangle plane
                Vec3 normal = (v1 - v0).cross(v2 - v0).normalize();
                float distPlane = (p - v0).dot(normal);
                Vec3 proj = p - normal * distPlane;
                
                // Check barycentric coords
                Vec3 v0v1 = v1 - v0;
                Vec3 v0v2 = v2 - v0;
                Vec3 v0p = proj - v0;

                float d00 = v0v1.dot(v0v1);
                float d01 = v0v1.dot(v0v2);
                float d11 = v0v2.dot(v0v2);
                float d20 = v0p.dot(v0v1);
                float d21 = v0p.dot(v0v2);

                float v = (d11 * d20 - d01 * d21) / (d00 * d11 - d01 * d01);
                float w = (d00 * d21 - d01 * d20) / (d00 * d11 - d01 * d01);
                float u = 1.0f - v - w;

                bool insideTriangle = (u >= -EPSILON && v >= -EPSILON && w >= -EPSILON && 
                                       u <= 1 + EPSILON && v <= 1 + EPSILON && w <= 1 + EPSILON);

                if (insideTriangle) {
                    Vec3 closest = v0 + v0v1 * u + v0v2 * w;
                    float d = (closest - p).length();
                    if (d < bestDist) {
                        bestDist = d;
                        bestPoint = closest;
                        found = true;
                    }
                } else {
                    // Check edges
                    // (Simplified: just check vertices for "closest" point approximation for performance)
                    // A true robust SDF checks edges, but for grid sampling, vertex check + plane check is often sufficient
                    // Let's add simple edge checks for robustness
                    
                    auto checkEdge = [&](const Vec3& a, const Vec3& b) {
                        Vec3 ab = b - a;
                        Vec3 ap = p - a;
                        float t = ap.dot(ab) / ab.dot(ab);
                        t = max(0.0f, min(1.0f, t));
                        Vec3 closest = a + ab * t;
                        float d = (closest - p).length();
                        if (d < bestDist) {
                            bestDist = d;
                            bestPoint = closest;
                        }
                    };
                    
                    checkEdge(v0, v1);
                    checkEdge(v1, v2);
                    checkEdge(v2, v0);
                }
            }
        } else {
            // Non-leaf: push children. Use priority by distance for better pruning
            // Simple stack push
            s.push(node.right);
            s.push(node.left);
        }
    }

    if (!found) return INFINITY;

    // Determine Sign (Inside/Outside)
    // Since mesh is watertight, we can use a simple check: center of face normal?
    // Better: Check a point slightly away from bestPoint.
    // Or just use the center of the bounding box?
    // Robust method: Winding number or point-in-mesh. 
    // Fast method for this context: Sample SDF at a point slightly offset along normal?
    // Actually, standard SDF is distance to surface.
    // If we are inside, distance is negative relative to surface? No.
    // Distance is always positive.
    // Signed Distance:
    // If point is inside, SDF = -distance_to_surface.
    // If point is outside, SDF = +distance_to_surface.
    
    // How to know if p is inside?
    // Compute SDF at a far away point (e.g. MinCorner - (Max-Min)).
    // If SDF(far) < 0, then p is inside. If > 0, outside.
    // But we need SDF for p now.
    
    // Let's use the face normal of the closest point.
    Vec3 normal = (v1 - v0).cross(v2 - v0).normalize();
    Vec3 toP = p - bestPoint;
    if (toP.dot(normal) < 0) {
        // Closest point is on the "outside" side of the local geometry (rare in clean meshes unless sampling inside a hole? But mesh is manifold)
        // Actually, dot product checks direction relative to the triangle plane.
        // If dot < 0, p is on the negative side of the plane.
        // We need a robust "Inside" check.
        
        // Let's use the bounding box center of the mesh. If center is far away, we are outside.
        // This assumes the mesh is the only object.
        // We know p is the center of a grid cell.
        // Let's iterate the BVH to check point-in-mesh. 
        // This is slow O(N).
        // Optimization: Since we traverse BVH anyway to find closest point, we can accumulate "depth" or winding number?
        // Too complex for single file.
        
        // Simple heuristic:
        // If the closest point is on the positive side of the plane defined by the face and center point,
        // then we are likely inside if the face normal points out.
        // Normal points out for CCW winding.
        // If p is inside, the dot product of (p - face_point) with normal should be negative?
        // No, inside means you are "behind" the face.
        // So SDF = -dist.
    }
    
    // Robust but slower: Ray casting.
    // Since the grid is sparse in Octree, we can afford a ray cast?
    // No, too much code.
    
    // Let's use the standard convention: 
    // Distance is positive outside, negative inside.
    // We need to classify `p`.
    // We can check the SDF of the center of the bounding box of the mesh.
    // If center SDF > 0, the mesh encloses the origin? No.
    // Let's just compute distance, and assume the first sample is negative (if origin is inside).
    // We don't know where the origin is relative to the mesh.
    
    // Correction: 
    // We treat the grid center as reference.
    // Find nearest point Q.
    // Check if p is inside.
    // To check inside: find if any ray from p intersects the mesh an odd number of times?
    // Or simpler: Just assume the "far away" corner is outside.
    // Let's compute SDF of the min corner of the root AABB.
    
    // To make this robust and fast:
    // We use the `normal` of the closest face.
    // If `dot(p - v0, normal)` is negative, p is inside the half-space of the face (assuming normal points out).
    // But we need to be inside the object, not just one face.
    // For a manifold mesh, the sum of winding numbers?
    // Let's do a quick check: if dot(p - v0, normal) < 0, we are inside.
    // But wait, if the closest face is a "backface" (normals inwards), this logic fails.
    // Input is clean manifold. We can assume normals point out?
    // Standard mesh IO (OBJ) is CCW (normals out).
    // If so, dot(p - v0, normal) < 0 means p is "behind" the surface, i.e., inside.
    // So SDF = -dist.
    
    // Check sign of closest face normal.
    // Actually, `toP = p - Q`. `toP` points from surface to p.
    // If `toP` opposes `normal`, p is inside.
    if (toP.dot(normal) < 0) {
        return -bestDist;
    }
    return bestDist;
}
