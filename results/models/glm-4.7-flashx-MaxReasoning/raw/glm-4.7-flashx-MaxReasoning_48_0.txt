```spirv
; SPIR-V
; Version: 1.3
; Generator: Unknown
; Bound: 60
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint GLCompute %main "main"
OpExecutionMode %main LocalSize 16 16 1

; --- Decorations ---
OpDecorate %inF Block
OpMemberDecorate %inF 0 Offset 0
OpDecorate %inF ArrayStride 4
OpDecorate %outF Block
OpMemberDecorate %outF 0 Offset 0
OpDecorate %outF ArrayStride 4
OpDecorate %params Block
OpMemberDecorate %params 0 Offset 0
OpMemberDecorate %params 1 Offset 16
OpMemberDecorate %params 2 Offset 32
OpMemberDecorate %params 3 Offset 48
OpDecorate %inF DescriptorSet 0
OpDecorate %inF Binding 0
OpDecorate %outF DescriptorSet 0
OpDecorate %outF Binding 1
OpDecorate %params DescriptorSet 0
OpDecorate %params Binding 2

; --- Types ---
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%int = OpTypeInt 32 0
%uint = OpTypeInt 32 1
%v2int = OpTypeVector %int 2
%v2uint = OpTypeVector %uint 2
%v4uint = OpTypeVector %uint 4

%inF = OpTypeStruct %v4float
%inF_ptr = OpTypePointer StorageBuffer %inF
%inF_arr = OpTypeRuntimeArray %v4float

%outF = OpTypeStruct %v4float
%outF_ptr = OpTypePointer StorageBuffer %outF
%outF_arr = OpTypeRuntimeArray %v4float

%params = OpTypeStruct %v4uint %v4float
%params_ptr = OpTypePointer Uniform %params

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_3 = OpConstant %float 3
%float_4_5 = OpConstant %float 4.5
%float_1_5 = OpConstant %float 1.5
%float_4_9 = OpConstant %float 0.4444444444
%float_1_9 = OpConstant %float 0.1111111111
%float_1_36 = OpConstant %float 0.0277777778

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%int_6 = OpConstant %int 6
%int_7 = OpConstant %int 7
%int_8 = OpConstant %int 8
%int_9 = OpConstant %int 9
%int_neg1 = OpConstant %int -1
%int_pos1 = OpConstant %int 1

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1

; --- Constants for Directions (cx, cy, w) ---
; k=0: (0,0, 4/9)
; k=1: (1,0, 1/9)
; k=2: (0,1, 1/9)
; k=3: (-1,0, 1/9)
; k=4: (0,-1, 1/9)
; k=5: (1,1, 1/36)
; k=6: (-1,1, 1/36)
; k=7: (-1,-1, 1/36)
; k=8: (1,-1, 1/36)

%float_cx0 = OpConstant %float 0
%float_cy0 = OpConstant %float 0
%float_w0  = OpConstant %float 0.4444444444

%float_cx1 = OpConstant %float 1
%float_cy1 = OpConstant %float 0
%float_w1  = OpConstant %float 0.1111111111

%float_cx2 = OpConstant %float 0
%float_cy2 = OpConstant %float 1
%float_w2  = OpConstant %float 0.1111111111

%float_cx3 = OpConstant %float -1
%float_cy3 = OpConstant %float 0
%float_w3  = OpConstant %float 0.1111111111

%float_cx4 = OpConstant %float 0
%float_cy4 = OpConstant %float -1
%float_w4  = OpConstant %float 0.1111111111

%float_cx5 = OpConstant %float 1
%float_cy5 = OpConstant %float 1
%float_w5  = OpConstant %float 0.0277777778

%float_cx6 = OpConstant %float -1
%float_cy6 = OpConstant %float 1
%float_w6  = OpConstant %float 0.0277777778

%float_cx7 = OpConstant %float -1
%float_cy7 = OpConstant %float -1
%float_w7  = OpConstant %float 0.0277777778

%float_cx8 = OpConstant %float 1
%float_cy8 = OpConstant %float -1
%float_w8  = OpConstant %float 0.0277777778

; --- Variables ---
%gl_GlobalInvocationID = OpVariable %v2uint Input

%inF_var = OpVariable %inF_ptr StorageBuffer
%outF_var = OpVariable %outF_ptr StorageBuffer
%params_var = OpVariable %params_ptr Uniform

; --- Main ---
%main = OpFunction %void None %void_fn
%label = OpLabel

; --- Load Inputs ---
%gid = OpLoad %v2uint %gl_GlobalInvocationID
%x = OpCompositeExtract %uint %gid 0
%y = OpCompositeExtract %uint %gid 1

%params_load = OpLoad %params %params_var
%dim = OpCompositeExtract %v4uint %params_load 0
%constants = OpCompositeExtract %v4float %params_load 1

%W = OpCompositeExtract %uint %dim 0
%H = OpCompositeExtract %uint %dim 1
%omega = OpCompositeExtract %float %constants 0

; --- Initialize Macroscopic ---
%rho = OpFAdd %float %float_0 %float_0
%ux = OpFAdd %float %float_0 %float_0
%uy = OpFAdd %float %float_0 %float_0

; --- Loop 1: Calculate Rho, Ux, Uy ---
%k_init = OpConstant %int 0
OpBranch %loop1_header

%loop1_header = OpLabel
%k_loop = OpPhi %int %k_init %start_loop1 %k_inc %loop1_merge

%continue_loop1 = OpSLessThan %bool %k_loop %int_9
OpBranchConditional %continue_loop1 %loop1_body %loop1_merge

%loop1_body = OpLabel
; Calculate Indices for Input Buffer: (y * W + x) * 9 + k
%y_mul_w = OpIMul %int %y %W
%flat_idx_base = OpIAdd %int %y_mul_w %x
%cell_offset = OpIMul %int %flat_idx_base %int_9
%curr_idx = OpIAdd %int %cell_offset %k_loop

; Load f
%in_ptr = OpAccessChain %inF_ptr %inF_var %curr_idx
%f_val = OpLoad %float %in_ptr

%rho_add = OpFAdd %float %rho %f_val
%ux_add = OpFAdd %float %ux %f_val
%uy_add = OpFAdd %float %uy %f_val

; Store loop vars
%rho = OpFAdd %float %rho %rho_add
%ux = OpFAdd %float %ux %ux_add
%uy = OpFAdd %float %uy %uy_add

%k_inc = OpIAdd %int %k_loop %int_1
OpBranch %loop1_header

%start_loop1 = OpLabel OpBranch %loop1_header
%loop1_merge = OpLabel

; Normalize Velocity
%rho_div = OpFDiv %float %rho %rho ; Div by self? No, need to store rho first.
; Wait, loop uses previous rho. Need to handle division outside loop or accumulate.
; Let's redo the phi.
%rho_accum = OpFAdd %float %rho %rho_add
%ux_accum = OpFAdd %float %ux %ux_add
%uy_accum = OpFAdd %float %uy %uy_add
%rho = OpFAdd %float %rho %rho_accum ; Reset/Store correct rho
%ux = OpFAdd %float %ux %ux_accum
%uy = OpFAdd %float %uy %uy_add

%ux_norm = OpFDiv %float %ux %rho
%uy_norm = OpFDiv %float %uy %rho

; Precompute usq = ux*ux + uy*uy
%ux_sq = OpFMul %float %ux_norm %ux_norm
%uy_sq = OpFMul %float %uy_norm %uy_norm
%usq = OpFAdd %float %ux_sq %uy_sq

; --- Loop 2: Collision & Streaming ---
%k_loop_2 = OpConstant %int 0
OpBranch %loop2_header

%loop2_header = OpLabel
%k_loop2 = OpPhi %int %k_loop_2 %start_loop2 %k_inc_2 %loop2_merge

%continue_loop2 = OpSLessThan %bool %k_loop2 %int_9
OpBranchConditional %continue_loop2 %loop2_body %loop2_merge

%start_loop2 = OpLabel OpBranch %loop2_header
%loop2_merge = OpLabel OpReturn

%loop2_body = OpLabel
; Select Constants based on k_loop2
%cx = OpSelect %float 
    %float_cx1 %float_cx0
    %float_cx2 %float_cx1
    %float_cx3 %float_cx2
    %float_cx4 %float_cx3
    %float_cx5 %float_cx4
    %float_cx6 %float_cx5
    %float_cx7 %float_cx6
    %float_cx8 %float_cx7
    %float_cx0 %float_cx8 ; default for k=0

%cy = OpSelect %float 
    %float_cy1 %float_cy0
    %float_cy2 %float_cy1
    %float_cy3 %float_cy2
    %float_cy4 %float_cy3
    %float_cy5 %float_cy4
    %float_cy6 %float_cy5
    %float_cy7 %float_cy6
    %float_cy8 %float_cy7
    %float_cy0 %float_cy8 ; default for k=0

%w = OpSelect %float 
    %float_w1 %float_w0
    %float_w2 %float_w1
    %float_w3 %float_w2
    %float_w4 %float_w3
    %float_w5 %float_w4
    %float_w6 %float_w5
    %float_w7 %float_w6
    %float_w8 %float_w7
    %float_w0 %float_w8 ; default for k=0

; Load f again
%in_ptr2 = OpAccessChain %inF_ptr %inF_var %curr_idx
%f_val2 = OpLoad %float %in_ptr2

; Calculate cu = cx*ux + cy*uy
%cu = OpFAdd %float %cx %ux_norm
%cu_mul = OpFMul %float %cu %cx
%cu_mul_2 = OpFAdd %float %cu_mul %uy_norm
%cu_mul_3 = OpFMul %float %cu_mul_2 %cy

; Calculate feq = w * rho * (1 + 3*cu + 4.5*cu^2 - 1.5*usq)
%cu_sq = OpFMul %float %cu_mul_2 %cu_mul_2
%feq_term1 = OpFAdd %float %float_1 %cu_mul_3
%feq_term2 = OpFAdd %float %feq_term1 %OpFMul %float_4_5 %cu_sq
%feq_term3 = OpFSub %float %feq_term2 %OpFMul %float_1_5 %usq
%feq = OpFMul %float %w %rho
%feq = OpFMul %float %feq %feq_term3

; Collision: f_out = f + omega * (feq - f)
%feq_minus_f = OpFSub %float %feq %f_val2
%omega_mul = OpFMul %float %feq_minus_f %omega
%f_out = OpFAdd %float %f_val2 %omega_mul

; Streaming: Write to (y+cy, x+cx) in Output Buffer
%y_stream = OpIAdd %int %y %int_neg1
%y_stream = OpSelect %int %y_stream %y %int_neg1 ; Ensure y is signed? No, OpIAdd works on signed int operands if used as signed.
; Wait, y is uint. y+(-1) is 0xFFFFFFFF.
; Use OpSMod logic: y + cy. cy is float.
%y_int_stream = OpConvertFToS %int %y_stream
%x_int_stream = OpConvertFToS %int %x_int_stream

; Wrap
%y_mod = OpSMod %int %y_int_stream %H
%x_mod = OpSMod %int %x_int_stream %W

%y_mul_w_out = OpIMul %int %y_mod %W
%flat_out = OpIAdd %int %y_mul_w_out %x_mod
%cell_offset_out = OpIMul %int %flat_out %int_9
%out_idx = OpIAdd %int %cell_offset_out %k_loop2

; Store to Output Buffer
%out_ptr = OpAccessChain %outF_ptr %outF_var %out_idx
OpStore %out_ptr %f_out

%k_inc_2 = OpIAdd %int %k_loop2 %int_1
OpBranch %loop2_header
```
