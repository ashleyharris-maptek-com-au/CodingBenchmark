declare const require: any; 
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid: string[][] = data.grid as string[][];
const robots: { id: number; x: number; y: number }[] = data.robots as { id: number; x: number; y: number }[];
const jobs: { id: number; x: number; y: number; priority: number }[] = data.jobs as { id: number; x: number; y: number; priority: number }[];

const h = grid.length;
const w = grid[0].length;
const dirs = [[1, 0], [-1, 0], [0, 1], [0, -1]];

function getDist(x1: number, y1: number, x2: number, y2: number): number {
    if (x1 === x2 && y1 === y2) return 0;
    
    const gScore = new Int32Array(w * h).fill(Infinity);
    const fScore = new Int32Array(w * h).fill(Infinity);
    const openSet: { x: number; y: number; f: number }[] = [];
    
    const startIdx = y1 * w + x1;
    const endIdx = y2 * w + x2;
    
    gScore[startIdx] = 0;
    const startH = Math.abs(x1 - x2) + Math.abs(y1 - y2);
    fScore[startIdx] = startH;
    openSet.push({ x: x1, y: y1, f: startH });

    while (openSet.length > 0) {
        openSet.sort((a, b) => a.f - b.f);
        const current = openSet.shift()!;
        
        if (current.x === x2 && current.y === y2) return gScore[current.y * w + current.x];
        
        const cx = current.x, cy = current.y;
        const currentIdx = cy * w + cx;
        
        for (const [dx, dy] of dirs) {
            const nx = cx + dx, ny = cy + dy;
            if (nx >= 0 && nx < w && ny >= 0 && ny < h && grid[ny][nx] !== '#') {
                const neighborIdx = ny * w + nx;
                const tentativeG = gScore[currentIdx] + 1;
                if (tentativeG < gScore[neighborIdx]) {
                    gScore[neighborIdx] = tentativeG;
                    const hVal = Math.abs(nx - x2) + Math.abs(ny - y2);
                    fScore[neighborIdx] = tentativeG + hVal;
                    const exists = openSet.some(n => n.x === nx && n.y === ny);
                    if (!exists) {
                        openSet.push({ x: nx, y: ny, f: tentativeG + hVal });
                    }
                }
            }
        }
    }
    return 999999;
}

const J = jobs.length;
const K = robots.length;
const jobDist = Array(J).fill(0).map(() => Array(J).fill(0));

for (let i = 0; i < J; i++) {
    jobDist[i][i] = 0;
    for (let j = i + 1; j < J; j++) {
        const d = getDist(jobs[i].x, jobs[i].y, jobs[j].x, jobs[j].y);
        jobDist[i][j] = d;
        jobDist[j][i] = d;
    }
}

const sortedJobs = [...jobs].sort((a, b) => {
    if (b.priority !== a.priority) return b.priority - a.priority;
    return a.id - b.id;
});

const assignments: number[][] = Array(K).fill(null).map(() => []);
const lastJobIdx = Array(K).fill(-1);
const lastRobotPos = robots.map(r => ({ x: r.x, y: r.y }));

for (let j = 0; j < J; j++) {
    const job = sortedJobs[j];
    let bestRobotIdx = 0;
    let bestCost = Infinity;

    for (let r = 0; r < K; r++) {
        let cost: number;
        if (assignments[r].length === 0) {
            cost = getDist(lastRobotPos[r].x, lastRobotPos[r].y, job.x, job.y);
        } else {
            cost = jobDist[lastJobIdx[r]][j];
        }

        if (cost < bestCost) {
            bestCost = cost;
            bestRobotIdx = r;
        }
    }

    assignments[bestRobotIdx].push(job.id);
    lastJobIdx[bestRobotIdx] = j;
    lastRobotPos[bestRobotIdx] = { x: job.x, y: job.y };
}

console.log(JSON.stringify({ assignments }));
