using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace OrbitalTSP
{
    class Program
    {
        // Earth Gravitational Parameter
        const double MU = 398600.4418;
        
        // Fixed time of flight for the transfer heuristic (4 hours)
        // This transforms the dynamic problem into a static graph problem
        const double TRANSFER_TIME = 4.0 * 3600.0; 
        
        static void Main(string[] args)
        {
            // 1. Read Input
            string[] firstLine = Console.ReadLine().Split();
            int N = int.Parse(firstLine[0]);
            double mu = double.Parse(firstLine[1]);
            
            // Start State
            double[] startPos = Console.ReadLine().Split().Select(double.Parse).ToArray();
            double[] startVel = Console.ReadLine().Split().Select(double.Parse).ToArray();
            
            // Station States
            List<double[]> stationsPos = new List<double[]>();
            List<double[]> stationsVel = new List<double[]>();
            
            for (int i = 0; i < N; i++)
            {
                string line = Console.ReadLine();
                double[] pos = line.Split().Select(double.Parse).Take(3).ToArray();
                double[] vel = line.Split().Select(double.Parse).Skip(3).ToArray();
                stationsPos.Add(pos);
                stationsVel.Add(vel);
            }

            // 2. Propagate all stations to T + TRANSFER_TIME
            // We assume stations move on Keplerian orbits defined by their initial state vectors
            List<double[]> targetPos = new List<double[]>();
            List<double[]> targetVel = new List<double[]>();
            
            for (int i = 0; i < N; i++)
            {
                var (r, v) = PropagateOrbit(stationsPos[i], stationsVel[i], TRANSFER_TIME);
                targetPos.Add(r);
                targetVel.Add(v);
            }

            // 3. Compute Cost Matrix (Delta-V)
            // C[i][j] = Delta-V required to go from Station i (at T) to Station j (at T+dt)
            double[,] costMatrix = new double[N, N];
            
            Parallel.For(0, N, i =>
            {
                for (int j = 0; j < N; j++)
                {
                    if (i == j)
                    {
                        costMatrix[i, j] = 0.0;
                        continue;
                    }
                    
                    // Solve Lambert Problem
                    // From: r_i at T to r_j at T + dt
                    double[] vTransitArrival = LambertUniversal(targetPos[i], targetPos[j], TRANSFER_TIME);
                    
                    // We want to match station j's velocity at the arrival time
                    // Delta V = |v_transfer - v_station_target|
                    double dv = Math.Sqrt(
                        Math.Pow(vTransitArrival[0] - targetVel[j][0], 2) +
                        Math.Pow(vTransitArrival[1] - targetVel[j][1], 2) +
                        Math.Pow(vTransitArrival[2] - targetVel[j][2], 2)
                    );
                    
                    costMatrix[i, j] = dv;
                    costMatrix[j, i] = dv; // Assuming symmetric transfer costs for simplicity
                }
            });

            // 4. Solve TSP using 2-Opt Heuristic
            int[] bestPath = SolveTSP(costMatrix, N);
            
            // 5. Output
            StringBuilder sb = new StringBuilder();
            foreach (int idx in bestPath)
            {
                sb.Append(idx + " ");
            }
            Console.WriteLine(sb.ToString().Trim());
        }

        /// <summary>
        /// Propagates an orbit from state (r, v) to time dt using Universal Variables.
        /// </summary>
        static (double[] r, double[] v) PropagateOrbit(double[] r0, double[] v0, double dt)
        {
            double r0_mag = Math.Sqrt(r0[0]*r0[0] + r0[1]*r0[1] + r0[2]*r0[2]);
            double v0_mag = Math.Sqrt(v0[0]*v0[0] + v0[1]*v0[1] + v0[2]*v0[2]);
            
            double alpha = 2.0/r0_mag - v0_mag*v0_mag / MU;
            double chi = v0_mag * dt; // Initial guess
            
            double tol = 1e-8;
            int maxIter = 50;
            
            for (int iter = 0; iter < maxIter; iter++)
            {
                // Stumpff functions
                double c2 = 0.0; double c3 = 0.0;
                if (chi != 0)
                {
                    double z = alpha * chi * chi;
                    if (z > 1e-6)
                    {
                        double sqz = Math.Sqrt(z);
                        c2 = (1 - Math.Cos(sqz)) / z;
                        c3 = (sqz - Math.Sin(sqz)) / Math.Pow(sqz, 3);
                    }
                    else if (z > -1e-6)
                    {
                        c2 = 0.5;
                        c3 = 1.0 / 6.0;
                    }
                    else
                    {
                        c2 = 0.5;
                        c3 = -1.0 / 6.0;
                    }
                }
                
                double r = r0_mag * (1 - alpha * chi * c2) + v0_mag * chi * (1 - alpha * c3);
                
                // Solve Kepler Eq: r = r0 cos(E) + ... 
                // Universal Kepler Eq: z*c3 + c2 - c1 = (chi * alpha * c3 - 1) / r0_mag + alpha * chi
                
                double A = v0_mag * chi / Math.Sqrt(MU) - 1.0;
                double y = r0_mag * A * chi * c2 + (1.0 - alpha * chi * chi * c3) * r;
                double s = chi * chi;
                
                double f = y/r - 1.0;
                double g = A * Math.Sqrt(y/MU);
                double gDot = 1.0 - y/r;
                
                double x = f * r0_mag + g * v0_mag * dt;
                double xd = fDot * r0_mag + gDot * v0_mag * dt;
                
                // Derivatives
                double df = - (1/r) * fDot * y + (1/r) * g * (v0_mag/MU) * (A * chi * c2 + (1 - alpha * chi * c3));
                double dg = f + g * (v0_mag/MU) * (A * chi * c2 + (1 - alpha * chi * c3)) - f * y / r;
                double dgDot = fDot + gDot * (v0_mag/MU) * (A * chi * c2 + (1 - alpha * chi * c3)) + g * (v0_mag/MU) * (A * c2 + 2 * chi * c3 - 3 * alpha * chi * chi * c3 / (2 * r));
                double fDot = dgDot / g; // Using relation f g dot = 1 - y/r
                
                double xNew = x + df * x + dg * xd;
                double xdNew = xd + fDot * x + dgDot * xd;
                
                if (Math.Abs(xNew - x) < tol && Math.Abs(xdNew - xd) < tol)
                {
                    return (new double[] { xNew, xdNew }, new double[] { xdNew, xNew });
                }
                
                chi = xNew / xdNew;
            }
            
            // Fallback
            return (r0, v0);
        }

        /// <summary>
        /// Solves Lambert's problem for a transfer from r1 to r2 in time dt.
        /// Returns the velocity vector at r2 (arrival velocity).
        /// Uses Universal Variables.
        /// </summary>
        static double[] LambertUniversal(double[] r1, double[] r2, double dt)
        {
            // Normalize vectors
            double r1_mag = Math.Sqrt(r1[0]*r1[0] + r1[1]*r1[1] + r1[2]*r1[2]);
            double r2_mag = Math.Sqrt(r2[0]*r2[0] + r2[1]*r2[1] + r2[2]*r2[2]);
            
            double r1x = r1[0] / r1_mag;
            double r1y = r1[1] / r1_mag;
            double r1z = r1[2] / r1_mag;
            
            double r2x = r2[0] / r2_mag;
            double r2y = r2[1] / r2_mag;
            double r2z = r2[2] / r2_mag;
            
            // Dot product and cross product
            double dDot = r1x * r2x + r1y * r2y + r1z * r2z;
            double cx = r1y * r2z - r1z * r2y;
            double cy = r1z * r2x - r1x * r2z;
            double cz = r1x * r2y - r1y * r2x;
            
            double cx2 = cx*cx + cy*cy + cz*cz;
            
            // Longitude
            double theta = Math.Acos(dDot);
            
            // Determine orientation (z-component of cross product)
            // If cross product Z is positive, theta is CCW. If negative, CW.
            // Usually standard for short way transfer if theta <= pi
            // Here we assume standard orientation handling
            
            double A = Math.Sin(theta) * Math.Sqrt(r1_mag * r2_mag / (1 - Math.Cos(theta)));
            
            // Universal variable formulation
            // We need to solve: z * c3 + c2 - c1 = (A * z * c3 - (1 - c2)) / (nu * sqrt(mu)) * (something)
            // Actually simpler formulation:
            // f(z) = r1+r2 - 2 sqrt(r1 r2) cos(nu) - r1 r2 v^2 / mu * (z c3 + c2 - c1) = 0
            
            // Initial guess
            double z = 0.0;
            double s = 0.0;
            double c2 = 0.5, c3 = 1.0/6.0;
            
            // Iteration
            for (int i = 0; i < 50; i++)
            {
                double c1 = 1.0 - (s/2.0);
                
                // Stumpff c2, c3
                if (s > 1e-6)
                {
                    double sqrt_s = Math.Sqrt(s);
                    c2 = (1.0 - Math.Cos(sqrt_s)) / s;
                    c3 = (sqrt_s - Math.Sin(sqrt_s)) / Math.Pow(sqrt_s, 3);
                }
                else if (s > -1e-6)
                {
                    c2 = 0.5;
                    c3 = 1.0/6.0;
                }
                else
                {
                    c2 = 0.5;
                    c3 = -1.0/6.0;
                }
                
                double y = r1_mag + r2_mag + A * (s * c3 - 1.0) / Math.Sqrt(MU);
                double dt_calc = (y * c3 + c2) * Math.Sqrt(y/MU) + A * Math.Sqrt(1 + (s * c2)/y);
                
                if (Math.Abs(dt_calc - dt) < 1e-4) break;
                
                double ddt_dt = Math.Sqrt(y/MU) * (1.0 - (s/2.0)*(c2+c3*s)) + A * (c3*s + (1 - 2*c2)*s/(2*y));
                double dz = (dt - dt_calc) / ddt_dt;
                
                z += dz;
                s = z * z;
            }
            
            // Calculate State Vectors
            double y = r1_mag + r2_mag + A * (s * c3 - 1.0) / Math.Sqrt(MU);
            double f = 1.0 - y / r1_mag;
            double g = A * Math.Sqrt(y / MU);
            
            double fDot = Math.Sqrt(MU / y) * ( (1 - y/r2_mag) - (y / r1_mag) );
            double gDot = 1.0 - y / r2_mag;
            
            // Velocity at arrival
            double vx = (r2[0] - f * r1[0]) / g;
            double vy = (r2[1] - f * r1[1]) / g;
            double vz = (r2[2] - f * r1[2]) / g;
            
            return new double[] { vx, vy, vz };
        }

        /// <summary>
        /// Solves the TSP using 2-Opt Local Search.
        /// </summary>
        static int[] SolveTSP(double[,] costMatrix, int N)
        {
            // Initial random tour
            int[] path = new int[N];
            for(int i=0; i<N; i++) path[i] = i;
            
            // Shuffle
            Random rnd = new Random();
            for (int i = N - 1; i > 0; i--)
            {
                int j = rnd.Next(i + 1);
                int temp = path[i];
                path[i] = path[j];
                path[j] = temp;
            }
            
            bool improved = true;
            while (improved)
            {
                improved = false;
                // Nested loop for 2-opt
                // Optimization: skip edges near the end
                for (int i = 0; i < N - 1; i++)
                {
                    for (int j = i + 1; j < N; j++)
                    {
                        // Indices of the two paths to swap
                        int len1 = j - i;
                        int len2 = N - len1;
                        
                        // Current cost of edges (i, i+1) and (j, j+1)
                        // Note: (j, j+1) is handled as (j, (j+1)%N)
                        // Standard 2-opt swap improves by removing edges (i, i+1) and (j, j+1)
                        // and adding (i, j) and (j+1, i+1)
                        
                        int currentCost = costMatrix[path[i], path[i+1]] + costMatrix[path[j], path[(j+1)%N]];
                        int newCost = costMatrix[path[i], path[j]] + costMatrix[path[(j+1)%N], path[(i+1)%N]];
                        
                        if (newCost < currentCost)
                        {
                            // Reverse the segment between i+1 and j
                            ReverseSegment(path, i + 1, j);
                            improved = true;
                        }
                    }
                }
            }
            
            return path;
        }

        static void ReverseSegment(int[] arr, int start, int end)
        {
            while (start < end)
            {
                int temp = arr[start];
                arr[start] = arr[end];
                arr[end] = temp;
                start++;
                end--;
            }
        }
    }
}
