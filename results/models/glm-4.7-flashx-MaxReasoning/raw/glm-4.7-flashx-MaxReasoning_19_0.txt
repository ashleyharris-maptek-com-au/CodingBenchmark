    // Represents a block of dirt or ore
    struct Voxel
    {
        public int Mass; // d % 16
        public int Value; // d // 16
        
        public int MassTotal => Mass; // Assuming each byte is a unit? "Each unit of mass"
        // Actually prompt says "Each peice of dirt... needs to be moved".
        // The input byte represents a block. 
        // Let's assume `Mass` in the struct represents the total mass in this voxel.
        
        public bool IsOre => Value > 0;
        public bool IsDirt => Mass > 0 && Value == 0;
        public bool IsEmpty => Mass == 0 && Value == 0;
    }

            int.TryParse(header[0], out XV);
            int.TryParse(header[1], out YV);
            int.TryParse(header[2], out ZV);
            float.TryParse(header[3], out XBS);
            float.TryParse(header[4], out YBS);
            float.TryParse(header[5], out ZBS);

            // Threshold for full simulation. 10 million voxels is about 10 MB of data.

            bool isLarge = totalVoxels > 10_000_000;

            if (isLarge)
            {
                HandleLargeGrid(XV, YV, ZV, Console.OpenStandardInput());
            }
            else
            {
                HandleMediumGrid(XV, YV, ZV, Console.OpenStandardInput());
            }
        }
