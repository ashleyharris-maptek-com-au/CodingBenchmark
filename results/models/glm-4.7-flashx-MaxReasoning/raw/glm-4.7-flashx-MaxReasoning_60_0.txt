```javascript
const fs = require('fs');

// Read input from stdin
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { machineCount, tasks } = data;

// Data structures
const taskMap = {}; // Map taskId -> task object
const inDegree = {}; // Map taskId -> number of remaining dependencies
const successors = {}; // Map taskId -> array of dependent taskIds

// Initialize data structures
tasks.forEach(task => {
    taskMap[task.id] = { ...task };
    inDegree[task.id] = task.deps ? task.deps.length : 0;
    successors[task.id] = [];
});

// Build successor adjacency list
tasks.forEach(task => {
    if (task.deps) {
        task.deps.forEach(depId => {
            successors[depId].push(task.id);
        });
    }
});

// Min-Heap implementation for Priority Queue (based on Deadline)
class MinHeap {
    constructor() {
        this.heap = [];
    }

    push(item) {
        this.heap.push(item);
        this.bubbleUp(this.heap.length - 1);
    }

    pop() {
        if (this.heap.length === 0) return null;
        const min = this.heap[0];
        const end = this.heap.pop();
        if (this.heap.length > 0) {
            this.heap[0] = end;
            this.sinkDown(0);
        }
        return min;
    }

    peek() {
        return this.heap[0];
    }

    bubbleUp(n) {
        const element = this.heap[n];
        while (n > 0) {
            let parentN = Math.floor((n + 1) / 2) - 1;
            let parent = this.heap[parentN];
            if (element.deadline >= parent.deadline) break;
            this.heap[parentN] = element;
            this.heap[n] = parent;
            n = parentN;
        }
    }

    sinkDown(n) {
        const length = this.heap.length;
        const element = this.heap[n];
        while (true) {
            let child2N = (n + 1) * 2;
            let child1N = child2N - 1;
            let swap = null;
            if (child1N < length) {
                let child1 = this.heap[child1N];
                if (child1.deadline < element.deadline) {
                    swap = child1N;
                }
            }
            if (child2N < length) {
                let child2 = this.heap[child2N];
                if (
                    (swap === null && child2.deadline < element.deadline) ||
                    (swap !== null && child2.deadline < this.heap[swap].deadline)
                ) {
                    swap = child2N;
                }
            }
            if (swap === null) break;
            this.heap[n] = this.heap[swap];
            this.heap[swap] = element;
            n = swap;
        }
    }
}

// Initialize Machines
// Machine object: { id: index, finishTime: number }
const machines = Array.from({ length: machineCount }, (_, i) => ({
    id: i,
    finishTime: 0
}));

// Initialize Ready Queue with tasks that have no dependencies
const readyQueue = new MinHeap();
Object.keys(inDegree).forEach(taskId => {
    if (inDegree[taskId] === 0) {
        readyQueue.push(taskMap[taskId]);
    }
});

// Schedule
const schedule = [];

// Process tasks
while (readyQueue.heap.length > 0) {
    const currentTask = readyQueue.pop();

    // Find machine with the earliest finish time
    const assignedMachine = machines.reduce((prev, curr) => 
        prev.finishTime < curr.finishTime ? prev : curr
    );

    const startTime = assignedMachine.finishTime;
    const finishTime = startTime + currentTask.duration;

    // Record schedule
    schedule.push({
        id: currentTask.id,
        start: startTime,
        machine: assignedMachine.id
    });

    // Update machine finish time
    assignedMachine.finishTime = finishTime;

    // Check successors
    const succIds = successors[currentTask.id];
    if (succIds) {
        succIds.forEach(succId => {
            inDegree[succId]--;
            if (inDegree[succId] === 0) {
                readyQueue.push(taskMap[succId]);
            }
        });
    }
}

// Output result
console.log(JSON.stringify({ schedule }, null, 0));
```
