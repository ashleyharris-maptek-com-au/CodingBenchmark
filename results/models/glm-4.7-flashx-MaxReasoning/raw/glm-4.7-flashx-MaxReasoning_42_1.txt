```spirv
OpCapability Geometry
OpCapability StorageBuffer
OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Input Decoration
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Output Decoration
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; User Inputs Decoration
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Output User Variable Decoration
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%f32 = OpTypeFloat 32
%v3f = OpTypeVector %f32 3
%v4f = OpTypeVector %f32 4

%uint = OpTypeInt 32 0
%uint1 = OpConstant %uint 1

; Constants
%f32_zero = OpConstant %f32 0.0
%f32_one = OpConstant %f32 1.0
%f32_h = OpConstant %f32 0.05
%vec_zero = OpConstantComposite %v3f %f32_zero %f32_zero %f32_zero
%vec_h = OpConstantComposite %v3f %f32_zero %f32_h %f32_zero
%vec_neg_h = OpConstantComposite %v3f %f32_zero %f32_zero %f32_zero ; Placeholders

; Define negative constant
%float_neg_h = OpConstant %f32 -0.05
%vec_neg_h = OpConstantComposite %v3f %float_neg_h %f32_zero %f32_zero

; Input Types
%gl_PerVertex_in = OpTypeStruct %v4f
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v3 = OpTypePointer Input %v3f
%worldPos = OpVariable %ptr_in_v3 Input
%normal = OpVariable %ptr_in_v3 Input
%color = OpVariable %ptr_in_v3 Input

; Output Types
%gl_PerVertex_out = OpTypeStruct %v4f
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3f
%outColor = OpVariable %ptr_out_v3 Output

; Main Function
%main = OpFunction %void None %void_fn
%main_label = OpLabel

%uint0 = OpConstant %uint 0
%uint1 = OpConstant %uint 1
%uint2 = OpConstant %uint 2
%uint3 = OpConstant %uint 3

; Load position and color
%ptr_wpos = OpAccessChain %ptr_in_v3 %worldPos %uint0
%pos = OpLoad %v3f %ptr_wpos
%ptr_col = OpAccessChain %ptr_in_v3 %color %uint0
%col = OpLoad %v3f %ptr_col

; Emit Vertex 0 (Top)
%top = OpFAdd %v3f %pos %vec_h
%v4_top = OpCompositeConstruct %v4f %top %f32_one
%ptr_out_0 = OpAccessChain %ptr_out_pv %gl_out %uint0
OpStore %ptr_out_0 %v4_top
OpStore %outColor %col
OpEmitVertex

; Emit Vertex 1 (Right)
%right = OpFAdd %v3f %pos %vec_h
%v4_right = OpCompositeConstruct %v4f %right %f32_one
%ptr_out_1 = OpAccessChain %ptr_out_pv %gl_out %uint1
OpStore %ptr_out_1 %v4_right
OpStore %outColor %col
OpEmitVertex

; Emit Vertex 2 (Left)
%left = OpFAdd %v3f %pos %vec_neg_h
%v4_left = OpCompositeConstruct %v4f %left %f32_one
%ptr_out_2 = OpAccessChain %ptr_out_pv %gl_out %uint2
OpStore %ptr_out_2 %v4_left
OpStore %outColor %col
OpEmitVertex

; Emit Vertex 3 (Bottom)
%bot = OpFAdd %v3f %pos %vec_neg_h
%v4_bot = OpCompositeConstruct %v4f %bot %f32_one
%ptr_out_3 = OpAccessChain %ptr_out_pv %gl_out %uint3
OpStore %ptr_out_3 %v4_bot
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
