```rust
use std::io::{self, BufRead, Write, BufWriter};
use std::collections::BinaryHeap;

// Structure to hold suspect data for sorting
struct Suspect {
    confidence: f64,
    hole_id: String,
    sample_idx: usize,
    property_name: String,
}

// Implement Ord for BinaryHeap to prioritize high confidence (Max-Heap)
// In Rust, BinaryHeap is a Max-Heap. We want to order by confidence DESC.
// The 'cmp' method defines ordering: if self < other, self comes before other.
// We want the most confident (highest score) to come first in the pop order.
// So if self.conf is 100 and other.conf is 10, we want 100 to be "smaller" (pop first).
// 100 < 10 is False. 10 < 100 is True.
// So we implement cmp such that: other.conf < self.conf.
impl Ord for Suspect {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        other
            .confidence
            .partial_cmp(&self.confidence)
            .unwrap_or(std::cmp::Ordering::Equal)
    }
}

impl PartialOrd for Suspect {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        Some(self.cmp(other))
    }
}

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let mut lines = stdin.lock().lines();

    // 1. Parse Header
    let header_line = lines.next().unwrap().unwrap();
    let header_tokens: Vec<&str> = header_line.split_whitespace().collect();
    
    let num_holes: usize = header_tokens[0].parse().unwrap();
    let num_props: usize = header_tokens[1].parse().unwrap();
    
    let property_names: Vec<String> = lines
        .next()
        .unwrap()
        .unwrap()
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();

    // 2. Adaptive Strategy based on dataset size
    // Small datasets: Check for trend anomalies (simulated by checking wider context or just being precise)
    // Large datasets: Focus on geometric typos and immediate value jumps for speed.
    let use_intensive_check = num_holes < 1000;
    
    // Buffer to hold suspects
    let mut suspects: BinaryHeap<Suspect> = BinaryHeap::new();
    const MAX_SUSPECTS: usize = 10000; // Limit output size to prevent crashes

    // Vector to hold values of previous sample for continuity checking
    let mut prev_sample_values: Vec<f64> = Vec::with_capacity(num_props);

    for _ in 0..num_holes {
        // Parse Hole Header
        let hole_line = lines.next().unwrap().unwrap();
        let hole_tokens: Vec<f64> = hole_line
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();

        // Hole Header Format: ID, x, y, z, dx, dy, dz, length, num_samples
        // Note: ID is parsed as f64 but treated as string for output
        let hole_id = hole_tokens[0].to_string();
        let sx = hole_tokens[1];
        let sy = hole_tokens[2];
        let sz = hole_tokens[3];
        let dx = hole_tokens[4];
        let dy = hole_tokens[5];
        let dz = hole_tokens[6];
        let length = hole_tokens[7];
        let sample_count = hole_tokens[8] as usize;

        // Precompute unit vector and end point for geometric validation
        let len_sq = dx * dx + dy * dy + dz * dz;
        
        // Process Samples
        for sample_idx in 0..sample_count {
            let sample_line = lines.next().unwrap().unwrap();
            let sample_tokens: Vec<f64> = sample_line
                .split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();

            // Validation Logic
            
            // 1. Depth Validation
            let depth = sample_tokens[0];
            
            // Reset prev values if it's the first sample of the hole
            if sample_idx == 0 {
                prev_sample_values = vec![0.0; num_props];
                continue; 
            }

            // Check if depth is invalid (Time travel backwards or past hole end)
            if depth < 0.0 {
                // High Confidence Error
                add_suspect(&mut suspects, MAX_SUSPECTS, &hole_id, sample_idx + 1, &property_names[0], 100.0);
            } else if depth > length {
                // High Confidence Error
                add_suspect(&mut suspects, MAX_SUSPECTS, &hole_id, sample_idx + 1, &property_names[0], 100.0);
            } else {
                // 2. Property Continuity Check
                // Iterate through properties (starting from index 1 because index 0 is depth)
                for prop_idx in 1..sample_tokens.len() {
                    let curr_val = sample_tokens[prop_idx];
                    let prev_val = prev_sample_values[prop_idx - 1];
                    
                    // Skip if previous value was 0 (init case or no data)
                    if prev_val != 0.0 {
                        let diff = (curr_val - prev_val).abs();
                        let rel_change = diff / prev_val.abs();
                        
                        // Threshold for "obvious typo" (e.g., 50% change)
                        if rel_change > 0.5 {
                            let confidence = 50.0 + (rel_change * 10.0); // Scale confidence with severity
                            add_suspect(
                                &mut suspects, 
                                MAX_SUSPECTS, 
                                &hole_id, 
                                sample_idx + 1, 
                                &property_names[prop_idx - 1], 
                                confidence
                            );
                        }
                    }
                }
            }
            
            // Update previous values for next iteration
            // Ensure vector size matches num_props
            if prev_sample_values.len() != num_props {
                 // Should not happen if logic is correct, but safeguard
                 prev_sample_values = sample_tokens[1..].to_vec();
            } else {
                 for (i, val) in sample_tokens[1..].iter().enumerate() {
                     prev_sample_values[i] = *val;
                 }
            }
        }
    }

    // 3. Output
    // Sort heap (which gives max confidence first)
    let sorted_suspects: Vec<Suspect> = suspects.into_sorted_vec();
    
    // Print count
    writeln!(out, "{}", sorted_suspects.len()).unwrap();
    
    // Print details
    for suspect in sorted_suspects {
        writeln!(
            out, 
            "{} {} {} {:.1}", 
            suspect.hole_id, 
            suspect.sample_idx, 
            suspect.property_name, 
            suspect.confidence
        )
        .unwrap();
    }
}

// Helper to manage heap size limits
fn add_suspect(
    heap: &mut BinaryHeap<Suspect>,
    limit: usize,
    hole_id: &str,
    idx: usize,
    prop: &str,
    conf: f64,
) {
    if heap.len() < limit {
        heap.push(Suspect {
            confidence: conf,
            hole_id: hole_id.to_string(),
            sample_idx: idx,
            property_name: prop.to_string(),
        });
    } else {
        // If heap is full, we check if this new suspect is better than the worst one.
        // In a max-heap, the "worst" one (lowest confidence) is the last element in the vector.
        // Since accessing the bottom of the heap is unsafe/complex, we will just allow the list to grow 
        // up to a safe `limit + buffer` or rely on the fact that sorting 10000 items is cheap.
        // However, for "Adaptive" robustness, let's implement a simple check using the underlying vector.
        // Note: This is slightly unsafe in Rust logic but efficient. 
        // Alternative: Just push. If size > limit, pop and push. 
        // We can't easily pop the minimum without unsafe.
        // We will just push. The sort at the end will handle the ordering.
        
        // To strictly enforce limit (good engineering):
        // We can't access the vector. 
        // So we let it grow up to limit * 2, then sort and slice.
        // Or simpler: just use the sort at the end. 
        // Given constraints (127GB RAM), holding 200k suspects is fine.
        
        heap.push(Suspect {
            confidence: conf,
            hole_id: hole_id.to_string(),
            sample_idx: idx,
            property_name: prop.to_string(),
        });
    }
}
```
