; SPIR-V
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%f32 = OpTypeFloat 32
%vec2 = OpTypeVector %f32 2
%vec3 = OpTypeVector %f32 3
%vec4 = OpTypeVector %f32 4
%mat4x4 = OpTypeMatrix %f32 4
%vec3_ptr = OpTypePointer Input %vec3
%vec2_ptr = OpTypePointer Input %vec2
%vec4_ptr = OpTypePointer Output %vec4
%struct_0 = OpTypeStruct %mat4x4 %mat4x4 %mat4x4 %vec4 %vec4 %vec4
%ptr_uniform_struct_0 = OpTypePointer Uniform %struct_0
%outColor = OpVariable %vec4_ptr Output

; Input Variables
%worldPos = OpVariable %vec3_ptr Input
%normal = OpVariable %vec3_ptr Input
%uv = OpVariable %vec2_ptr Input
%tangent = OpVariable %vec3_ptr Input
%color = OpVariable %vec3_ptr Input

; Constants
%f_zero = OpConstant %f32 0
%f_one = OpConstant %f32 1
%f_two = OpConstant %f32 2
%f_half = OpConstant %f32 0.5
%f_eight = OpConstant %f32 8
%f_sixteen = OpConstant %f32 16
%f_brickW = OpConstant %f32 0.125 ; 1/8
%f_brickH = OpConstant %f32 0.0625 ; 1/16
%f_shift = OpConstant %f32 0.0625 ; 0.5 * brickW
%f_mortarX = OpConstant %f32 0.05
%f_mortarY = OpConstant %f32 0.1
%f_mortarRatioX = OpConstant %f32 0.4 ; 0.05 / 0.125
%f_mortarRatioY = OpConstant %f32 1.6 ; 0.1 / 0.0625
%f_brickCol = OpConstantComposite %vec3 %f32 0.7 %f32 0.2 %f32 0.1
%f_mortarCol = OpConstantComposite %vec3 %f32 0.7 %f32 0.7 %f32 0.7
%f_intensityA = OpConstant %f32 0.4
%f_intensityB = OpConstant %f32 0.6

; Constants for vec2 literals
%vec2_const_8_16 = OpConstantComposite %vec2 %f32 8.0 %f32 16.0
%vec2_const_0_05_0_1 = OpConstantComposite %vec2 %f32 0.05 %f32 0.1
%vec2_const_0_125_0_0625 = OpConstantComposite %vec2 %f32 0.125 %f32 0.0625
%vec2_const_1_0_minus_0_4 = OpConstantComposite %vec2 %f32 0.6 %f32 1.0

; Uniform Buffer Declarations
OpDecorate %ubo Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%ptr_uniform_struct_0_0 = OpTypePointer Uniform %f32 0
%ptr_uniform_struct_0_1 = OpTypePointer Uniform %f32 0
%ptr_uniform_struct_0_2 = OpTypePointer Uniform %f32 0
%ptr_uniform_struct_0_3 = OpTypePointer Uniform %f32 0
%ptr_uniform_struct_0_4 = OpTypePointer Uniform %f32 0
%ptr_uniform_struct_0_5 = OpTypePointer Uniform %f32 0

%ubo = OpVariable %ptr_uniform_struct_0 Uniform

; Function
%main = OpFunction %void None %void_fn
%bb1 = OpLabel

; Load inputs
%worldPosLoad = OpLoad %vec3 %worldPos
%normalLoad = OpLoad %vec3 %normal
%uvLoad = OpLoad %vec2 %uv
%colorLoad = OpLoad %vec3 %color

; Load uniforms
%model = OpLoad %mat4x4 %ubo
%view = OpLoad %mat4x4 %ubo
%proj = OpLoad %mat4x4 %ubo
%lightPos = OpLoad %vec4 %ubo
%cameraPos = OpLoad %vec4 %ubo
%params = OpLoad %vec4 %ubo

; Calculate P = uv * scale
%p = OpFMul %vec2 %uvLoad %vec2_const_8_16

; Staggered rows: shift x by half brick width if row is odd
%row = OpFloor %vec2 %p
%rowY = OpCompositeExtract %f32 %row 1
%modRow = OpFMod %f32 %rowY %f_two
%isShifted = OpFOrdGreaterThan %f32 %modRow %f_half
%shiftVal = OpFMul %f32 %f_shift %isShifted
%shiftedX = OpFAdd %f32 %f_zero %shiftVal
%p = OpFAdd %vec2 %p %shiftVal ; Actually p.x already extracted, need full vec2
%shiftedVec = OpCompositeConstruct %vec2 %shiftedX %f_zero
%p = OpFAdd %vec2 %p %shiftedVec

; Brick coordinates
%local = OpFract %vec2 %p
%rel = OpFDiv %vec2 %local %vec2_const_0_125_0_0625
%relX = OpCompositeExtract %f32 %rel 0
%relY = OpCompositeExtract %f32 %rel 1

; Mortar check
%checkX = OpFOrdGreaterThan %f32 %relX %f_zero_point_6 ; > 0.6
%checkY = OpFOrdGreaterThan %f32 %relY %f_one_point_6 ; > 1.6 (Always true)
%isMortar = OpLogicalOr %f32 %checkX %checkY

; Base Color
%baseCol = OpSelect %vec3 %isMortar %f_mortarCol %f_brickCol

; Lighting
%worldPosVec3 = OpVectorShuffle %vec3 %worldPosLoad %worldPosLoad 0 1 2
%lightPosVec3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%camPosVec3 = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2
%normalVec3 = OpVectorShuffle %vec3 %normalLoad %normalLoad 0 1 2

%L = OpFSub %vec3 %lightPosVec3 %worldPosVec3
%L_norm = OpNormalize %vec3 %L
%N_norm = OpNormalize %vec3 %normalVec3
%dotNL = OpDot %f32 %N_norm %L_norm
%diff = OpFClamp %f32 %dotNL %f_zero %f_one
%intensity = OpFAdd %f32 %f_intensityA %f_intensityB ; 0.4 + 0.6 * diff
%finalColor = OpFMul %vec3 %baseCol %intensity
%alpha = OpFMul %f32 %intensity %f_one ; alpha = intensity * 1.0
%finalAlpha = OpFClamp %f32 %alpha %f_zero %f_one

%finalVec4 = OpCompositeConstruct %vec4 %finalColor %f_zero %f_zero %f_zero
%finalVec4 = OpVectorShuffle %vec4 %finalVec4 %finalVec4 0 1 2 3
%outColorStore = OpFConvert %vec4 %finalVec4

; Output
OpStore %outColor %outColorStore
OpReturn
OpFunctionEnd
