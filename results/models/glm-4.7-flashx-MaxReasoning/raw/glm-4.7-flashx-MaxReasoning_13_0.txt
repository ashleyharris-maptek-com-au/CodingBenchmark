```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

// Suffix Automaton State structure
// Stores transitions (next) for all possible byte values (0-255) to keep it generic
struct State {
    int next[256]; 
    int link;
    int len;
};

class SuffixAutomaton {
private:
    vector<State> st;
    int last;

public:
    SuffixAutomaton() {
        st.resize(1);
        st[0].len = 0;
        st[0].link = -1;
        last = 0;
        // Initialize next array to 0
        for (int i = 0; i < 256; ++i) st[0].next[i] = 0;
    }

    // Adds a character to the automaton
    void extend(int c) {
        int cur = (int)st.size();
        st.emplace_back();
        st[cur].len = st[last].len + 1;
        // Initialize next array for the new state to 0
        for (int i = 0; i < 256; ++i) st[cur].next[i] = 0;

        int p = last;
        // Add transitions from states that don't have 'c' to the new state 'cur'
        while (p != -1 && st[p].next[c] == 0) {
            st[p].next[c] = cur;
            p = st[p].link;
        }

        if (p == -1) {
            // p is root, set link of cur to root
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                // Direct link is valid
                st[cur].link = q;
            } else {
                // Need to clone state q
                int clone = (int)st.size();
                st.emplace_back();
                st[clone].len = st[p].len + 1;
                // Copy transitions from q to clone
                for (int i = 0; i < 256; ++i) st[clone].next[i] = st[q].next[i];
                st[clone].link = st[q].link;

                // Update links pointing to q to point to clone
                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }

    // Calculates the length of the longest common substring between the reference string
    // (the one the SAM was built on) and 's'
    int getLCS(const string& s) {
        int v = 0; // Current state
        int l = 0; // Current length of match
        int best = 0; // Best match length found so far

        for (char c : s) {
            int ch = (unsigned char)c; // Handle any byte value
            // Follow transitions, fallback to link if needed
            while (v != 0 && st[v].next[ch] == 0) {
                v = st[v].link;
                l = st[v].len;
            }
            if (st[v].next[ch] != 0) {
                v = st[v].next[ch];
                l++;
            } else {
                // No match starting at this position
                v = 0;
                l = 0;
            }
            if (l > best) {
                best = l;
            }
        }
        return best;
    }
};

int main() {
    // Optimizations for large I/O
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    if (!(cin >> n)) return 0;
    cin.ignore();

    vector<string> strings;
    strings.reserve(n);

    // Read all strings
    for (int i = 0; i < n; ++i) {
        string s;
        getline(cin, s);
        strings.push_back(s);
    }

    // If only one string, it is the longest common substring
    if (n == 1) {
        cout << strings[0] << endl;
        return 0;
    }

    // Strategy: Build Suffix Automaton on the SHORTEST string.
    // Any common substring must be a substring of the shortest string.
    // The LCS of the shortest string and the other strings gives the LCS of the whole set.
    
    int shortest_idx = 0;
    for (int i = 1; i < n; ++i) {
        if (strings[i].size() < strings[shortest_idx].size()) {
            shortest_idx = i;
        }
    }

    const string& ref = strings[shortest_idx];
    SuffixAutomaton sam;

    // Build automaton
    for (char c : ref) {
        sam.extend((unsigned char)c);
    }

    int best_len = 0;

    // Iterate through other strings to find the maximum match length
    for (int i = 0; i < n; ++i) {
        if (i == shortest_idx) continue;
        
        int current_len = sam.getLCS(strings[i]);
        
        if (current_len > best_len) {
            best_len = current_len;
            // Early exit: we cannot find a substring longer than the reference itself
            if (best_len == (int)ref.size()) break;
        }
    }

    // Output result
    if (best_len == 0) {
        cout << endl;
    } else {
        // Extract the substring from the reference string
        // Since we know best_len is valid, any occurrence will do.
        // We return the first one found to be deterministic.
        for (size_t i = 0; i + best_len <= ref.size(); ++i) {
            string candidate = ref.substr(i, best_len);
            cout << candidate << endl;
            return 0;
        }
    }

    return 0;
}
```
