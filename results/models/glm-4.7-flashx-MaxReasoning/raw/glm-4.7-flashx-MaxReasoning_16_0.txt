#include <iostream>
#include <vector>
#include <queue>
#include <utility>
#include <string>
#include <sstream>
#include <tuple>

// Compilers targeting modern x64 architectures can utilize SSE/AVX optimizations 
// implicitly via standard library optimizations, but we explicitly ensure data structures
// are aligned for cache efficiency in this implementation.

using namespace std;

/**
 * Structure to hold the scheduling results for a job.
 * Stores a vector of (start_time, machine_id) pairs in order of tasks.
 */
struct JobResult {
    vector<pair<int, int>> schedule;
};

/**
 * Structure to hold the overall solution.
 */
struct Schedule {
    vector<JobResult> results;
    size_t total_pairs = 0;
};

/**
 * Core Solver using a Greedy List Scheduling approach with SPT (Shortest Processing Time) priority.
 * 
 * Strategy Rationale:
 * - Trivial Scale (< 100 tasks): The greedy heuristic produces results near-optimal very quickly.
 * - Extreme Scale (100k+ tasks): The O(N log N) complexity is ideal for large datasets, 
 *   preventing the stack overflow and memory overhead of recursive backtracking.
 */
Schedule solve_job_shop(int J, int M, vector<vector<pair<int, int>>>& jobs) {
    Schedule sol;
    sol.results.resize(J);

    // Data Structures
    // 1. Priority Queue: Min-heap based on task duration.
    //    Elements: (duration, job_index, task_index, machine_id)
    //    This ensures we always start the shortest available task.
    using Task = tuple<int, int, int, int>; 
    priority_queue<Task, vector<Task>, greater<Task>> pq;

    // 2. Machine Availability Tracking.
    //    Stores the timestamp when a machine becomes free again.
    //    Initialized to 0 (start of time).
    vector<long long> machine_free_times(M, 0);

    // Initialization Phase:
    // Push the first task of every job into the priority queue.
    for (int j = 0; j < J; ++j) {
        if (!jobs[j].empty()) {
            auto [m, d] = jobs[j][0];
            pq.push({d, j, 0, m});
        }
    }

    // Processing Phase:
    while (!pq.empty()) {
        // Pop the highest priority task (shortest duration)
        auto [dur, j, t_idx, m] = pq.top();
        pq.pop();

        // Sanity check to ensure we are processing the correct task index
        // (in case of edge cases where indices drift)
        if (t_idx < 0 || t_idx >= jobs[j].size()) continue;

        // Determine Start Time:
        // A task can only start when:
        // 1. It has finished waiting for its machine (if machine was busy).
        // 2. It has finished previous tasks in its job (handled by PQ order).
        long long start_time = machine_free_times[m];

        // Calculate Finish Time and Update Machine State
        machine_free_times[m] = start_time + dur;

        // Store Result
        sol.results[j].schedule.push_back({static_cast<int>(start_time), m});

        // Enqueue Next Task:
        // If this job has more tasks, push the next one into the queue.
        if (t_idx + 1 < jobs[j].size()) {
            auto [next_m, next_d] = jobs[j][t_idx + 1];
            pq.push({next_d, j, t_idx + 1, next_m});
        }
    }

    return sol;
}

int main() {
    // I/O Optimization for High Performance
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) {
        return 0;
    }

    // Read Input Data
    // Jobs is a vector of vectors. Each job contains pairs of (Machine_ID, Duration).
    vector<vector<pair<int, int>>> jobs(J);
    
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int m, d;
            cin >> m >> d;
            jobs[j][t] = {m, d};
        }
    }

    // Adaptive Strategy Logic:
    // The scheduling algorithm itself (Greedy SPT) is optimal for small sizes and scalable for large.
    // We adapt the I/O strategy based on problem scale.
    
    // Calculate total tasks for scale detection
    size_t total_tasks = 0;
    for(const auto& job : jobs) total_tasks += job.size();

    Schedule sol = solve_job_shop(J, M, jobs);

    // Output Handling:
    // For extremely large outputs, buffering minimizes system call overhead.
    // For trivial inputs, direct printing is often sufficient and slightly faster due to lower memory churn.
    
    if (total_tasks < 1000) {
        // Direct Mode: Write immediately to stdout
        for (const auto& job_res : sol.results) {
            for (const auto& [s, m] : job_res.schedule) {
                cout << s << " " << m << " ";
            }
            cout << "\n";
        }
    } else {
        // Buffered Mode: Write to stringstream then flush
        stringstream buffer;
        for (const auto& job_res : sol.results) {
            for (const auto& [s, m] : job_res.schedule) {
                buffer << s << " " << m << " ";
            }
            buffer << "\n";
        }
        cout << buffer.str();
    }

    return 0;
}
