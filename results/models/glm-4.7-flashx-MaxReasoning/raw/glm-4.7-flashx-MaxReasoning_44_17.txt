#version 450
layout(location = 0) out vec4 outColor;

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

void main() {
    // 1. Lighting Calculations (Lambertian)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    vec3 N = normalize(normal);
    float diffuse = max(dot(N, L), 0.0);
    float lightFactor = 0.4 + 0.6 * diffuse;

    // 2. Procedural Brick Pattern
    // Scale UVs to 8x16
    vec2 scaledUV = uv * vec2(8.0, 16.0);

    // Staggered rows (half-brick offset every other row)
    // Check if row index is odd
    if (mod(int(scaledUV.y), 2) == 1) {
        scaledUV.x -= 0.5;
    }

    // Calculate UV inside the current brick cell (0.0 to 1.0)
    vec2 brickUV = fract(scaledUV);

    // Define mortar thickness
    float xMortarThick = 0.05;
    float yMortarThick = 0.10;

    // Calculate distance from nearest edge for current brick
    float distX = min(brickUV.x, 1.0 - brickUV.x);
    float distY = min(brickUV.y, 1.0 - brickUV.y);

    // Determine if pixel is inside mortar
    // step(a, b) returns 1 if b >= a. 
    // We check if dist is less than thick.
    float isMortar = step(xMortarThick, distX) * step(yMortarThick, distY);

    // 3. Base Colors
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);

    vec3 baseColor = mix(brickColor, mortarColor, float(isMortar));

    // 4. Combine Lighting and Color
    vec3 finalColor = baseColor * lightFactor;

    // 5. Output
    outColor = vec4(finalColor, 1.0);
}
