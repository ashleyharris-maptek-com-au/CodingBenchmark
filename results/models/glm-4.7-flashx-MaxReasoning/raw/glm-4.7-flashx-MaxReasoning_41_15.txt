%v3_0_2_0_2_0_75 = OpConstantComposite %v3float %float_0_2 %float_0_2 %float_0_75
%v3_0_7_0_7_0_4 = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_4

OpEntryPoint Fragment %main "main" %in0 %in1 %in2 %in3 %in4 %out0 %ubo_var

%main = OpFunction %void None %3
%label = OpLabel

%worldPos = OpLoad %v3float %in0
%normal = OpLoad %v3float %in1
%uv = OpLoad %v2float %in2
%tangent = OpLoad %v3float %in3
%color = OpLoad %v3float %in4

%n = OpNormalize %v3float %normal

%lightPos_ptr = OpAccessChain %21 %ubo_var %uint_0 %uint_0 %uint_0 %uint_3
%lightPos = OpLoad %v4float %lightPos_ptr

%L = OpVectorTimesScalar %v3float %lightPos %float_1_0
%L = OpFSub %v3float %L %worldPos
%L = OpNormalize %v3float %L

%NdotL = OpDot %float %n %L

%t = OpFMul %float %NdotL %float_0_5
%t = OpFAdd %float %t %float_0_5

%t_v3 = OpCompositeConstruct %v3float %t %t %t
%one_minus_t = OpFSub %v3float %v4_1_0_0_0 %t_v3

%res = OpFMul %v3float %one_minus_t %v3_0_2_0_2_0_75
%res = OpFAdd %v3float %res %OpFMul %v3float %t_v3 %v3_0_7_0_7_0_4

%res_4 = OpCompositeConstruct %v4float %res %float_1_0
OpStore %out0 %res_4

OpReturn
OpEndFunction
