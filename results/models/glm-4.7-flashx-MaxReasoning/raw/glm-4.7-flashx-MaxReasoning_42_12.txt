```spirv
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; gl_PerVertex Input
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%uint_6 = OpConstant %uint 6
%_arr_gl_PerVertex_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%_ptr_Input__arr_gl_PerVertex_in_6 = OpTypePointer Input %_arr_gl_PerVertex_in_6
%gl_in = OpVariable %_ptr_Input__arr_gl_PerVertex_in_6 Input

; gl_PerVertex Output
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%_ptr_Output__ptr_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output__ptr_gl_PerVertex_out Output

; Inputs
%_ptr_Input_v3float = OpTypePointer Input %v3float
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input

; Output
%_ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %_ptr_Output_v3float Output

; UBO
%_struct_15 = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float
%_ptr_Uniform__struct_15 = OpTypePointer Uniform %_struct_15
%ubo = OpVariable %_ptr_Uniform__struct_15 Uniform
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%uint_6 = OpConstant %uint 6
%one = OpConstant %float 1.0
%zero = OpConstant %float 0.0
%0_5 = OpConstant %float 0.5
%dim_c = OpConstant %float 0.3
%width = OpConstant %float 0.015
%one_vec = OpConstantComposite %v3float %one %one %one
%zero_vec = OpConstantComposite %v3float %zero %zero %zero

; Entry Point
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor %ubo

; Execution Modes
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

%main = OpFunction %void None %3
%label = OpLabel

; --- Load MVP Matrix ---
%ptr_ubo_0 = OpAccessChain %_ptr_Uniform_mat4 %ubo %uint_0
%mvp = OpLoad %mat4 %ptr_ubo_0
%ptr_ubo_1 = OpAccessChain %_ptr_Uniform_mat4 %ubo %uint_1
%view = OpLoad %mat4 %ptr_ubo_1
%ptr_ubo_2 = OpAccessChain %_ptr_Uniform_mat4 %ubo %uint_2
%proj = OpLoad %mat4 %ptr_ubo_2
; Model * View * Proj
%mvp_full = OpCompositeConstruct %mat4 %proj %view %mvp

; --- Emit Main Triangle (Indices 0, 2, 4) ---
; Vert 0
%pos_0 = OpCompositeExtract %v4float %gl_in %uint_0 0
%col_ptr_0 = OpAccessChain %_ptr_Input_v3float %color %uint_0
%col_0 = OpLoad %v3float %col_ptr_0
%dim_col_0 = OpVectorTimesScalar %v3float %col_0 %dim_c
%gl_out_pos_0 = OpAccessChain %_ptr_Output__ptr_gl_PerVertex_out %gl_out %uint_0
OpStore %gl_out_pos_0 %pos_0
%out_col_ptr_0 = OpAccessChain %_ptr_Output_v3float %outColor %uint_0
OpStore %out_col_ptr_0 %dim_col_0
OpEmitVertex

; Vert 2
%pos_2 = OpCompositeExtract %v4float %gl_in %uint_2 0
%col_ptr_2 = OpAccessChain %_ptr_Input_v3float %color %uint_2
%col_2 = OpLoad %v3float %col_ptr_2
%dim_col_2 = OpVectorTimesScalar %v3float %col_2 %dim_c
%gl_out_pos_2 = OpAccessChain %_ptr_Output__ptr_gl_PerVertex_out %gl_out %uint_0
OpStore %gl_out_pos_2 %pos_2
%out_col_ptr_2 = OpAccessChain %_ptr_Output_v3float %outColor %uint_0
OpStore %out_col_ptr_2 %dim_col_2
OpEmitVertex

; Vert 4
%pos_4 = OpCompositeExtract %v4float %gl_in %uint_4 0
%col_ptr_4 = OpAccessChain %_ptr_Input_v3float %color %uint_4
%col_4 = OpLoad %v3float %col_ptr_4
%dim_col_4 = OpVectorTimesScalar %v3float %col_4 %dim_c
%gl_out_pos_4 = OpAccessChain %_ptr_Output__ptr_gl_PerVertex_out %gl_out %uint_0
OpStore %gl_out_pos_4 %pos_4
%out_col_ptr_4 = OpAccessChain %_ptr_Output_v3float %outColor %uint_0
OpStore %out_col_ptr_4 %dim_col_4
OpEmitVertex

; --- Emit Edge 0-2 (Quad) ---
; Indices 0, 2 (Adj is 1, not used for position calculation here)
; 1. Load World Pos
%ptr_w_0 = OpAccessChain %_ptr_Input_v3float %worldPos %uint_0
%w_0 = OpLoad %v3float %ptr_w_0
%ptr_w_2 = OpAccessChain %_ptr_Input_v3float %worldPos %uint_2
%w_2 = OpLoad %v3float %ptr_w_2
; 2. Vector V = W2 - W0
%vec_v = OpFma %v3float %w_2 %one %w_0
%vec_v_neg = OpFma %v3float %vec_v %zero %vec_v ; Identity to help vector ops? No, vec_v is already V. %vec_v_neg = OpVectorTimesScalar %v3float %vec_v %zero is identity.
%vec_v_neg = OpVectorTimesScalar %v3float %vec_v %zero ; Fix: negate vector. No, FMA A*1 + B -> B-A.
; %vec_v = w_2 * 1 + w_0 * (-1) = w_2 - w_0
%neg_one = OpVectorTimesScalar %v3float %one_vec %zero
%vec_v = OpFma %v3float %w_2 %one_vec %neg_one
%vec_v_neg = OpFma %v3float %vec_v %one_vec %vec_v ; Identity.

; 3. Normal = Cross(V, Up(0,1,0))
%up_vec = OpConstantComposite %v3float %zero %one %zero
%edge_norm = OpCross %v3float %vec_v %up_vec
%edge_norm_neg = OpCross %v3float %up_vec %vec_v ; Opposite direction

; 4. Offset = Normal * width
%off_vec = OpVectorTimesScalar %v3float %edge_norm %width
%off_vec_neg = OpVectorTimesScalar %v3float %edge_norm_neg %width

; 5. Offsets for World
%w_0_plus = OpFma %v3float %w_0 %one_vec %off_vec
%w_2_plus = OpFma %v3float %w_2 %one_vec %off_vec
%w_2_minus = OpFma %v3float %w_2 %one_vec %off_vec_neg
%w_0_minus = OpFma %v3float %w_0 %one_vec %off_vec_neg

; 6. Transform to Clip Space
%clip_0_plus = OpVectorTimesMatrix %v4float %mvp_full %w_0_plus
%clip_2_plus = OpVectorTimesMatrix %v4float %mvp_full %w_2_plus
%clip_2_minus = OpVectorTimesMatrix %v4float %mvp_full %w_2_minus
%clip_0_minus = OpVectorTimesMatrix %v4float %mvp_full %w_0_minus

; 7. Store and Emit
%gl_out_ptr = OpAccessChain %_ptr_Output__ptr_gl_PerVertex_out %gl_out %uint_0
OpStore %gl_out_ptr %clip_0_plus
%out_col_ptr = OpAccessChain %_ptr_Output_v3float %outColor %uint_0
OpStore %out_col_ptr %one_vec
OpEmitVertex

OpStore %gl_out_ptr %clip_2_plus
OpStore %out_col_ptr %one_vec
OpEmitVertex

OpStore %gl_out_ptr %clip_2_minus
OpStore %out_col_ptr %one_vec
OpEmitVertex

OpStore %gl_out_ptr %clip_0_minus
OpStore %out_col_ptr %one_vec
OpEmitVertex

; --- Emit Edge 2-4 (Quad) ---
; Indices 2, 4
%ptr_w_2_2 = OpAccessChain %_ptr_Input_v3float %worldPos %uint_2
%w_2_load = OpLoad %v3float %ptr_w_2_2
%ptr_w_4 = OpAccessChain %_ptr_Input_v3float %worldPos %uint_4
%w_4 = OpLoad %v3float %ptr_w_4

%vec_v_2_4 = OpFma %v3float %w_4 %one_vec %neg_one
%edge_norm_2_4 = OpCross %v3float %vec_v_2_4 %up_vec
%off_vec_2_4 = OpVectorTimesScalar %v3float %edge_norm_2_4 %width
%off_vec_2_4_neg = OpVectorTimesScalar %v3float %edge_norm_2_4 %zero ; Fix neg
%off_vec_2_4_neg = OpVectorTimesScalar %v3float %edge_norm_2_4 %zero ; Hack for neg (cross changes sign).
%edge_norm_2_4_neg = OpCross %v3float %up_vec %vec_v_2_4
%off_vec_2_4_neg = OpVectorTimesScalar %v3float %edge_norm_2_4_neg %width

%w_2_plus_2 = OpFma %v3float %w_2_load %one_vec %off_vec_2_4
%w_4_plus = OpFma %v3float %w_4 %one_vec %off_vec_2_4
%w_4_minus = OpFma %v3float %w_4 %one_vec %off_vec_2_4_neg
%w_2_minus_2 = OpFma %v3float %w_2_load %one_vec %off_vec_2_4_neg

%clip_2_plus_2 = OpVectorTimesMatrix %v4float %mvp_full %w_2_plus_2
%clip_4_plus = OpVectorTimesMatrix %v4float %mvp_full %w_4_plus
%clip_4_minus = OpVectorTimesMatrix %v4float %mvp_full %w_4_minus
%clip_2_minus_2 = OpVectorTimesMatrix %v4float %mvp_full %w_2_minus_2

OpStore %gl_out_ptr %clip_2_plus_2
OpStore %out_col_ptr %one_vec
OpEmitVertex

OpStore %gl_out_ptr %clip_4_plus
OpStore %out_col_ptr %one_vec
OpEmitVertex

OpStore %gl_out_ptr %clip_4_minus
OpStore %out_col_ptr %one_vec
OpEmitVertex

OpStore %gl_out_ptr %clip_2_minus_2
OpStore %out_col_ptr %one_vec
OpEmitVertex

; --- Emit Edge 4-0 (Quad) ---
; Indices 4, 0
%ptr_w_0_4 = OpAccessChain %_ptr_Input_v3float %worldPos %uint_0
%w_0_load = OpLoad %v3float %ptr_w_0_4
%ptr_w_4_4 = OpAccessChain %_ptr_Input_v3float %worldPos %uint_4
%w_4_load = OpLoad %v3float %ptr_w_4_4

%vec_v_4_0 = OpFma %v3float %w_0_load %one_vec %neg_one
%edge_norm_4_0 = OpCross %v3float %vec_v_4_0 %up_vec
%off_vec_4_0 = OpVectorTimesScalar %v3float %edge_norm_4_0 %width
%edge_norm_4_0_neg = OpCross %v3float %up_vec %vec_v_4_0
%off_vec_4_0_neg = OpVectorTimesScalar %v3float %edge_norm_4_0_neg %width

%w_4_plus_4 = OpFma %v3float %w_4_load %one_vec %off_vec_4_0
%w_0_plus_4 = OpFma %v3float %w_0_load %one_vec %off_vec_4_0
%w_0_minus_4 = OpFma %v3float %w_0_load %one_vec %off_vec_4_0_neg
%w_4_minus_4 = OpFma %v3float %w_4_load %one_vec %off_vec_4_0_neg

%clip_4_plus_4 = OpVectorTimesMatrix %v4float %mvp_full %w_4_plus_4
%clip_0_plus_4 = OpVectorTimesMatrix %v4float %mvp_full %w_0_plus_4
%clip_0_minus_4 = OpVectorTimesMatrix %v4float %mvp_full %w_0_minus_4
%clip_4_minus_4 = OpVectorTimesMatrix %v4float %mvp_full %w_4_minus_4

OpStore %gl_out_ptr %clip_4_plus_4
OpStore %out_col_ptr %one_vec
OpEmitVertex

OpStore %gl_out_ptr %clip_0_plus_4
OpStore %out_col_ptr %one_vec
OpEmitVertex

OpStore %gl_out_ptr %clip_0_minus_4
OpStore %out_col_ptr %one_vec
OpEmitVertex

OpStore %gl_out_ptr %clip_4_minus_4
OpStore %out_col_ptr %one_vec
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
