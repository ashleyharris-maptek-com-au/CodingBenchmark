```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <queue>
#include <functional>

using namespace std;

long long global_best_val = 0;
vector<bool> global_x;

void solve_component(vector<int>& indices, vector<long long>& comp_costs, vector<vector<int>>& comp_adj) {
    int k = indices.size();
    if (k == 0) return;

    // Heuristic: Greedy
    vector<int> order(k);
    iota(order.begin(), order.end(), 0); 
    sort(order.begin(), order.end(), [&](int a, int b) { return comp_costs[a] > comp_costs[b]; });
    
    vector<bool> greedy_selected(k, false);
    long long current_sum = 0;
    for (int id : order) {
        if (!greedy_selected[id]) {
            current_sum += comp_costs[id];
            greedy_selected[id] = true;
            for (int nb : comp_adj[id]) greedy_selected[nb] = true;
        }
    }
    if (current_sum > global_best_val) {
        global_best_val = current_sum;
        for (int i = 0; i < k; ++i) global_x[indices[i]] = greedy_selected[i];
    }

    // Precompute prefix sums for component costs for efficient bound calculation
    vector<long long> prefix(k + 1, 0);
    for (int i = 0; i < k; ++i) prefix[i+1] = prefix[i] + comp_costs[i];

    // Recursive Branch and Bound
    // val: current accumulated cost
    // rem: list of local indices still available (will be sorted by degree at start of call)
    // chosen: list of local indices selected so far
    function<void(long long, vector<int>, vector<int>)> dfs = [&](long long val, vector<int> rem, vector<int> chosen) {
        // Pruning: If even adding all remaining weights doesn't beat global best, stop.
        if (!rem.empty()) {
            // rem is sorted by local index in previous calls (since rem1 and rem2 are copies/processed)
            // However, we need to ensure prefix indices are correct.
            // Prefix[i] = sum of costs[0..i-1].
            // Sum of costs for indices l0, l1... = prefix[last+1] - prefix[first].
            long long rem_sum = prefix[rem.back() + 1] - prefix[rem[0]];
            if (val + rem_sum <= global_best_val) return;
        }

        // Base Case: No remaining vertices to consider
        if (rem.empty()) {
            if (val > global_best_val) {
                global_best_val = val;
                for (int id : chosen) {
                    global_x[indices[id]] = true;
                }
            }
            return;
        }

        // Sort remaining vertices by degree descending to pick the most constrained one first
        // This is a heuristic to prune the search tree effectively
        sort(rem.begin(), rem.end(), [&](int a, int b) {
            return comp_adj[a].size() > comp_adj[b].size();
        });

        int v = rem[0];

        // Branch 1: Include vertex v (Pick v)
        // We can include v, so we must exclude v and all its neighbors
        vector<int> rem1;
        for (int u : rem) {
            // Check if u is not v and not a neighbor of v
            if (u != v && find(comp_adj[v].begin(), comp_adj[v].end(), u) == comp_adj[v].end()) {
                rem1.push_back(u);
            }
        }
        vector<int> chosen1 = chosen;
        chosen1.push_back(v);
        dfs(val + comp_costs[v], rem1, chosen1);

        // Branch 2: Exclude vertex v
        // We exclude v, so we can include its neighbors in the future (just not v itself)
        vector<int> rem2(rem.begin() + 1, rem.end());
        dfs(val, rem2, chosen);
    };

    dfs(0, order, vector<int>());
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> costs(n);
    for (int i = 0; i < n; ++i) {
        cin >> costs[i];
    }

    vector<vector<int>> adj(n);
    // If a constraint u=v exists (x_u + x_u <= 1), it implies 2*x_u <= 1, so x_u must be 0.
    vector<bool> is_fixed_zero(n, false);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) {
            is_fixed_zero[u] = true;
        } else {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    global_best_val = 0;
    global_x.assign(n, false);

    // Find connected components of the graph
    vector<vector<int>> components;
    vector<bool> visited(n, false);
    for (int i = 0; i < n; ++i) {
        // Skip fixed-zero nodes (loops) as they are not part of the feasible independent set problem
        // They are effectively removed from the graph.
        if (is_fixed_zero[i]) continue;
        if (!visited[i]) {
            queue<int> q;
            q.push(i);
            visited[i] = true;
            components.push_back(vector<int>());
            while (!q.empty()) {
                int curr = q.front(); q.pop();
                components.back().push_back(curr);
                for (int nb : adj[curr]) {
                    if (!visited[nb]) {
                        visited[nb] = true;
                        q.push(nb);
                    }
                }
            }
        }
    }

    // Solve each component independently
    for (auto& comp : components) {
        if (comp.empty()) continue;
        
        int k = comp.size();
        vector<long long> comp_costs(k);
        vector<vector<int>> comp_adj(k);
        unordered_map<int, int> map_id; // global index -> local index

        for (int i = 0; i < k; ++i) {
            int g_id = comp[i];
            map_id[g_id] = i;
            comp_costs[i] = costs[g_id];
            for (int nb : adj[g_id]) {
                // Check if neighbor is in the same component (always true here, but good practice)
                if (map_id.count(nb)) {
                    comp_adj[i].push_back(map_id[nb]);
                }
            }
        }
        
        solve_component(comp, comp_costs, comp_adj);
    }

    // Output
    cout << global_best_val << "\n";
    for (int i = 0; i < n; ++i) {
        if (i != 0) cout << " ";
        cout << global_x[i];
    }
    cout << "\n";

    return 0;
}
```
