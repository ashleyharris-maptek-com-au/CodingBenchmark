```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);

    // Calculate light direction
    vec3 L = ubo.lightPos.xyz - worldPos;
    L = normalize(L);

    // Calculate t as specified: dot(N, L) * 0.5 + 0.5
    float t = dot(N, L) * 0.5 + 0.5;

    // Define cool and warm colors
    vec3 cool = vec3(0.2, 0.2, 0.75);
    vec3 warm = vec3(0.7, 0.7, 0.4);

    // Apply Gooch shading blend
    vec3 goochColor = mix(cool, warm, t);

    // Output the result with alpha 1.0
    outColor = vec4(goochColor, 1.0);
}
```
