import numpy as np

class Autopilot:
    """
    Module-level Autopilot class to persist state and filters across calls.
    """
    def __init__(self):
        # History buffers for median filtering to handle flickering/sensor noise
        self.wheel_speed_hist = []
        self.lane_offset_hist = []
        
        # PID Integrators (persistent error accumulation)
        self.steer_err = 0.0
        self.speed_err = 0.0
        self.brake_err = 0.0
        
        # Controller Gains
        # Lateral: Proportional + Derivative (Integral often causes oscillation in steering)
        self.steer_kp = 0.5   # rad/m approx
        self.steer_kd = 0.3
        
        # Longitudinal: Full PID
        self.throttle_kp = 0.5
        self.throttle_ki = 0.01
        self.throttle_kd = 0.1
        
        # Braking: Aggressive P and I
        self.brake_kp = 3.0
        self.brake_ki = 0.5
        self.brake_kd = 0.0
        
        # World Model Constants
        self.lane_width = 3.7  # meters

    def get_median(self, lst):
        """Helper to safely calculate median."""
        if not lst:
            return 0.0
        return float(np.median(lst))

    def median_filter(self, new_val, history, max_size=3):
        """Rolling median filter to smooth flickering data."""
        history.append(new_val)
        if len(history) > max_size:
            history.pop(0)
        return self.get_median(history)

    def autopilot_step(self, state: dict, dt: float) -> dict:
        # --- 1. Data Extraction & Sensor Fusion ---
        
        # 1.1 Speed Fusion (Wheel Speed + GPS + Accel)
        # Check 4 wheel speeds
        ws_candidates = []
        for w in ['wheel_speed_fl', 'wheel_speed_fr', 'wheel_speed_rl', 'wheel_speed_rr']:
            if state.get(f"{w}_valid", False):
                ws_candidates.append(state[w])
        
        # Check GPS speed
        gps_speed = state.get('gps_1_speed_kmh', 0)
        if state.get('gps_1_valid', False):
            ws_candidates.append(gps_speed)
        
        # Fuse using Median
        current_speed = self.median_filter(
            self.get_median(ws_candidates) if ws_candidates else 0.0,
            self.wheel_speed_hist
        )

        # 1.2 Yaw Rate Fusion (Gyroscopes)
        gyros = []
        for i in range(1, 4):
            if state.get(f'gyro_{i}_valid', False):
                gyros.append(state[f'gyro_{i}_yaw_rate_dps'])
        current_yaw_rate = self.get_median(gyros) if gyros else 0.0

        # 1.3 Steering Angle Fusion (Sensors)
        steers = []
        if state.get('steering_angle_1_valid', False):
            steers.append(state['steering_angle_1_deg'])
        if state.get('steering_angle_2_valid', False):
            steers.append(state['steering_angle_2_deg'])
        current_steer_angle = self.get_median(steers) if steers else 0.0

        # 1.4 Lateral Error (Vision Lane Offset)
        vision_offsets = []
        for i in range(1, 4):
            if state.get(f'vision_{i}_valid', False):
                vision_offsets.append(state[f'vision_{i}_lane_offset_m'])
        
        lane_offset = self.median_filter(
            self.get_median(vision_offsets) if vision_offsets else 0.0,
            self.lane_offset_hist
        )
        
        # 1.5 Obstacle Detection
        obstacle_detected = False
        min_obstacle_dist = float('inf')
        
        # Check Vision Systems first (most reliable for "obstacle")
        for i in range(1, 4):
            if state.get(f'vision_{i}_valid', False) and state.get(f'vision_{i}_obstacle_detected', False):
                d = state.get(f'vision_{i}_obstacle_dist_m', float('inf'))
                if d < min_obstacle_dist:
                    min_obstacle_dist = d
                    obstacle_detected = True
        
        # 1.6 Environment Data
        mu = state.get('estimated_mu', 1.0) # Friction coefficient
        target_lane = state.get('target_lane', 0)
        current_lane = state.get('current_lane', 0)
        target_speed = state.get('target_speed_kmh', 100)
        
        # Speed Limit Detection
        speed_limit = 140 # Default Highway limit
        for i in range(1, 4):
             if state.get(f'vision_{i}_valid', False) and state.get(f'vision_{i}_speed_limit_valid', False):
                 speed_limit = state[f'vision_{i}_speed_limit_kmh']
                 break

        # --- 2. Control Logic ---
        
        # 2.1 Lateral Control (Steering)
        # Calculate desired lateral error based on lane change target
        lane_diff = target_lane - current_lane
        desired_offset = lane_diff * self.lane_width
        
        # Obstacle Avoidance: If obstacle is close, prioritize staying in current lane
        # (Simple strategy to prevent "cutting" into obstacle)
        if obstacle_detected and min_obstacle_dist < 60:
            desired_offset = 0.0
            
        # PID Calculation
        err_lateral = desired_offset - lane_offset
        d_err = (err_lateral - self.steer_err) / dt if dt > 0 else 0
        self.steer_err += err_lateral * dt
        
        # Scale PID output to [-1.0, 1.0]
        # The gain 0.5 maps 2m offset to approx 1.0 steer (reasonable)
        steer_cmd = (self.steer_kp * err_lateral) + (self.steer_kd * d_err)
        steer_cmd = np.clip(steer_cmd, -1.0, 1.0)

        # 2.2 Longitudinal Control (Throttle/Brake)
        throttle_cmd = 0.0
        brake_cmd = 0.0
        
        # Braking Logic
        if obstacle_detected:
            # Determine relative speed to obstacle
            obs_rel_speed = 0
            for i in range(1, 4):
                if state.get(f'vision_{i}_valid', False) and state.get(f'vision_{i}_obstacle_detected', False):
                    obs_rel_speed = state.get(f'vision_{i}_obstacle_rel_speed_kmh', 0)
                    break
            
            # If we are significantly faster than obstacle, brake
            if current_speed > obs_rel_speed + 5: # Difference > 5 km/h
                brake_cmd = 1.0 # Emergency
            elif current_speed > obs_rel_speed + 2:
                brake_cmd = 0.5 # Moderate

        # Throttle Logic (if no immediate need to brake)
        if brake_cmd == 0.0:
            # Respect Speed Limit
            final_target_speed = speed_limit if current_speed > speed_limit else target_speed
            
            # Adjust gains based on Traction (mu)
            # Low mu -> softer throttle, prevent wheelspin
            mu_factor = max(0.3, mu) 
            
            err_speed = final_target_speed - current_speed
            self.speed_err += err_speed * dt
            d_speed = (err_speed - self.speed_err) / dt if dt > 0 else 0
            
            throttle_cmd = (self.throttle_kp * mu_factor * err_speed) + (self.throttle_ki * self.speed_err) + (self.throttle_kd * mu_factor * d_speed)
            throttle_cmd = np.clip(throttle_cmd, 0.0, 1.0)

        # Return Control Dict
        return {
            'steering': steer_cmd,
            'throttle': throttle_cmd,
            'brake': brake_cmd
        }
