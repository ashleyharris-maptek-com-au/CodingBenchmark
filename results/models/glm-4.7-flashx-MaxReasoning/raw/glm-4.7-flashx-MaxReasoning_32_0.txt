```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <tuple>
#include <set>
#include <algorithm>
#include <climits>
#include <iomanip>

using namespace std;

// Fast I/O
ios_base::sync_with_stdio(false);
cin.tie(NULL);

struct Edge {
    int to;
    long long weight;
};

int main() {
    int N, M, K;
    if (!(cin >> N >> M >> K)) return 0;

    // Build adjacency list
    vector<vector<Edge>> adj(N);
    for (int i = 0; i < M; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    // Read terminal vertices
    vector<int> terminals(K);
    for (int i = 0; i < K; ++i) {
        cin >> terminals[i];
    }

    // Map original terminal IDs to 0..K-1 indices
    map<int, int> terminal_map;
    for (int i = 0; i < K; ++i) {
        terminal_map[terminals[i]] = i;
    }

    // 1. Precompute all-pairs shortest paths (Dijkstra from every node)
    const long long INF = 1e18;
    vector<vector<long long>> dist(N, vector<long long>(N, INF));

    for (int s = 0; s < N; ++s) {
        dist[s][s] = 0;
        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        pq.push({0, s});
        
        while (!pq.empty()) {
            long long d_u = pq.top().first;
            int u = pq.top().second;
            pq.pop();

            if (d_u > dist[s][u]) continue;

            for (const auto& e : adj[u]) {
                int v = e.to;
                long long w = e.weight;
                if (dist[s][u] + w < dist[s][v]) {
                    dist[s][v] = dist[s][u] + w;
                    pq.push({dist[s][v], v});
                }
            }
        }
    }

    // 2. Exact DP (Dreyfus-Wagner)
    int full_mask = (1 << K) - 1;
    
    // dp[mask][v]: Min weight to connect terminals in 'mask' ending at vertex 'v'
    vector<vector<long long>> dp(1 << K, vector<long long>(N, INF));
    // par_node[mask][v]: Previous Steiner node (used for reconstruction)
    vector<vector<int>> par_node(1 << K, vector<int>(N, -1));
    // par_term[mask][v]: Previous terminal index (used for reconstruction)
    vector<vector<int>> par_term(1 << K, vector<int>(N, -1));

    // Initialize base cases (single terminal)
    for (int t = 0; t < K; ++t) {
        int orig_node = terminals[t];
        dp[1 << t][orig_node] = 0;
        par_term[1 << t][orig_node] = t;
    }

    // Iterate over all masks
    for (int mask = 1; mask < (1 << K); ++mask) {
        for (int v = 0; v < N; ++v) {
            if (dp[mask][v] >= INF) continue;

            // A. Submask Merge (Merge terminal t into current component at v)
            // Transition: dp[mask][v] = min_{t in mask} (dp[mask ^ {t}][v] + dist[v][terminals[t]])
            for (int t = 0; t < K; ++t) {
                if (mask & (1 << t)) {
                    int nmask = mask ^ (1 << t);
                    int w = dist[v][terminals[t]];
                    long long new_dist = dp[mask][v] + w;
                    if (new_dist < dp[nmask][v]) {
                        dp[nmask][v] = new_dist;
                        par_term[nmask][v] = t;
                    }
                }
            }

            // B. Steiner Merge (Connect current component at v to Steiner node u)
            // Transition: dp[mask][v] = min_{u} (dp[mask][u] + dist[u][v])
            for (int u = 0; u < N; ++u) {
                long long new_dist = dp[mask][u] + dist[u][v];
                if (new_dist < dp[mask][v]) {
                    dp[mask][v] = new_dist;
                    par_node[mask][v] = u;
                }
            }
        }
    }

    // 3. Find best solution
    int best_v = -1;
    long long best_weight = INF;
    for (int v = 0; v < N; ++v) {
        if (dp[full_mask][v] < best_weight) {
            best_weight = dp[full_mask][v];
            best_v = v;
        }
    }

    // 4. Reconstruction
    set<pair<int, int>> edges;
    int cur_mask = full_mask;
    int cur_v = best_v;

    while (cur_mask) {
        // Check if we connected via a terminal
        if (par_term[cur_mask][cur_v] != -1) {
            int t = par_term[cur_mask][cur_v];
            int t_node = terminals[t];
            edges.insert({cur_v, t_node});
            // Move to the subproblem that connected t
            // The edge connects cur_v to t. t is the "parent" in the hierarchy of this merge.
            cur_mask ^= (1 << t);
            cur_v = t_node; 
        } 
        // Otherwise connected via a Steiner node
        else if (par_node[cur_mask][cur_v] != -1) {
            int u = par_node[cur_mask][cur_v];
            edges.insert({cur_v, u});
            // Move to the Steiner node u
            cur_v = u;
        } 
        // Fallback (should not happen with correct initialization)
        else {
            break;
        }
    }

    // Output
    cout << best_weight << " " << edges.size() << "\n";
    for (const auto& e : edges) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
```
