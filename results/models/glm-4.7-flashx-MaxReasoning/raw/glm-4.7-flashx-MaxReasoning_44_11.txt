#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 1. Polka Dot Pattern (6x6)
    // Calculate position within the UV grid (6 cells in each direction)
    vec2 cellUV = uv * 6.0;
    // Get coordinates relative to the center of the current cell ([-0.5, 0.5])
    vec2 centerUV = fract(cellUV) - 0.5;
    // Calculate distance from center
    float dist = length(centerUV);
    // Determine if inside the circle (radius 0.3)
    // Using smoothstep for a soft edge or step for a hard edge. 
    // A hard edge fits "radius 0.3" definition better.
    float dotMask = step(0.3, dist); 

    vec3 bgColor = vec3(0.1, 0.1, 0.3);
    vec3 dotColor = vec3(1.0, 0.8, 0.0);

    vec3 patternColor = mix(bgColor, dotColor, dotMask);

    // 2. Lambertian Shading
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    vec3 norm = normalize(normal);
    float diffuse = max(dot(norm, lightDir), 0.0);
    vec3 shading = 0.3 + 0.7 * diffuse;

    // 3. Output
    outColor = vec4(patternColor * shading, 1.0);
}
