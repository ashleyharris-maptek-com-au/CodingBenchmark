; SPIR-V
1 0 0 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %ubo "ubo"
OpName %outColor "outColor"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %normal Normal
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float 4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%v3_ptr = OpTypePointer Input %vec3
%v4_ptr_ubo = OpTypePointer Uniform %vec4
%ubo_ptr = OpTypePointer Uniform %UBO
%v4_ptr_out = OpTypePointer Output %vec4
%bool = OpTypeBool
%int = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_192 = OpConstant %int 192
%int_224 = OpConstant %int 224
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1

%worldPos = OpVariable %v3_ptr Input
%normal = OpVariable %v3_ptr Input
%ubo = OpVariable %ubo_ptr Uniform
%outColor = OpVariable %v4_ptr_out Output

%main = OpFunction %void None %void
%entry = OpLabel

%light_pos_ptr = OpAccessChain %v4_ptr_ubo %ubo %int_192
%light_pos = OpLoad %vec4 %light_pos_ptr
%light_dir = OpVectorShuffle %vec3 %light_pos %light_pos 0 1 2
%world_pos = OpLoad %vec3 %worldPos

%norm_len = OpLength %float %normal
%norm_inv_len = OpFDiv %float %float_1 %norm_len
%norm = OpFMul %vec3 %normal %norm_inv_len

%l_vec = OpFSub %vec3 %light_dir %world_pos
%l_len = OpLength %float %l_vec
%l_inv_len = OpFDiv %float %float_1 %l_len
%l = OpFMul %vec3 %l_vec %l_inv_len

%dot_val = OpDot %float %norm %l
%is_negative = OpFOrdLessThan %bool %dot_val %float_0
%diffuse = OpSelect %float %is_negative %float_0 %dot_val

%res = OpVectorShuffle %vec4 %diffuse %diffuse 0 0 0 0
%final_col = OpFAdd %vec4 %res %float_1

OpStore %outColor %final_col

OpReturn
OpFunctionEnd
