struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);

    // Calculate the light vector in world space
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Calculate Lambertian diffuse term
    float d = max(dot(N, L), 0.0);

    // Quantize into bands based on thresholds 0.75, 0.5, 0.25
    // Corresponding values: 1.0, 0.7, 0.4, 0.2
    float shade = 0.2;
    if (d > 0.25)
    {
        shade = 0.4;
        if (d > 0.5)
        {
            shade = 0.7;
            if (d > 0.75)
            {
                shade = 1.0;
            }
        }
    }

    // Apply shade to vertex color
    float3 finalColor = input.color * shade;

    return float4(finalColor, 1.0);
}
