```csharp
using System;
using System.Collections.Generic;
using System.IO;

class MinesweeperSolver
{
    // Cell type constants
    const byte UNKNOWN = 0;
    const byte SAFE = 1;   // Contains a value 0-8
    const byte MINE = 2;   // Flagged mine

    // Neighbor offsets (NW, N, NE, W, E, SW, S, SE)
    // Pre-calculated to avoid recalculating (-1, 0, 1) loops repeatedly
    static readonly int[] Offsets = {
        -1, 0, 1,
        -W - 1, -W, -W + 1,
        W - 1, W, W + 1
    };

    static int W; // Width
    static int H; // Height
    static byte[] Board;
    static Queue<int> Queue;

    static void Main()
    {
        // Configure buffering for fast output
        // While we collect moves in a list, we should write efficiently
        using (var writer = new StreamWriter(Console.OpenStandardOutput()))
        {
            // Read Input
            // Line 1: H W
            string[] dims = Console.ReadLine().Split();
            H = int.Parse(dims[0]);
            W = int.Parse(dims[1]);

            Board = new byte[H * W];
            Queue = new Queue<int>();
            List<int> safeMoves = new List<int>();

            // Parse board
            for (int y = 0; y < H; y++)
            {
                string line = Console.ReadLine();
                for (int x = 0; x < W; x++)
                {
                    int idx = y * W + x;
                    char c = line[x];
                    if (c >= '0' && c <= '8')
                    {
                        Board[idx] = (byte)(c - '0' + SAFE);
                        Queue.Enqueue(idx);
                    }
                    else
                    {
                        if (c == '*')
                        {
                            Board[idx] = MINE;
                        }
                        else
                        {
                            Board[idx] = UNKNOWN;
                        }
                    }
                }
            }

            // Solver Loop
            while (Queue.Count > 0)
            {
                int idx = Queue.Dequeue();
                int y = idx / W;
                int x = idx % W;
                int val = Board[idx] - SAFE;

                int unknownCount = 0;
                int flagCount = 0;

                // Iterate 8 neighbors
                for (int i = 0; i < 8; i++)
                {
                    int nIdx = idx + Offsets[i];
                    
                    // Bounds check (offsets are relative to x/y, need to check local bounds)
                    // Optimization: Instead of checking bounds per neighbor, we can check the global bounds of the tile.
                    // However, simple modulo check is safer and fast enough for C# JIT.
                    if (nIdx < 0 || nIdx >= Board.Length) continue;

                    byte nType = Board[nIdx];
                    if (nType == UNKNOWN) unknownCount++;
                    else if (nType == MINE) flagCount++;
                }

                // Logic 1: Cell is '0' (Empty)
                // All unrevealed neighbors are safe.
                if (val == 0)
                {
                    for (int i = 0; i < 8; i++)
                    {
                        int nIdx = idx + Offsets[i];
                        if (nIdx < 0 || nIdx >= Board.Length) continue;

                        if (Board[nIdx] == UNKNOWN)
                        {
                            Board[nIdx] = SAFE; // Reveal
                            Queue.Enqueue(nIdx);
                            safeMoves.Add(nIdx);
                        }
                    }
                }
                // Logic 2: Unrevealed + Flagged == Value
                // The mines are all accounted for by flags. Remaining unknowns must be safe.
                else if (unknownCount + flagCount == val)
                {
                    for (int i = 0; i < 8; i++)
                    {
                        int nIdx = idx + Offsets[i];
                        if (nIdx < 0 || nIdx >= Board.Length) continue;

                        if (Board[nIdx] == UNKNOWN)
                        {
                            Board[nIdx] = SAFE;
                            Queue.Enqueue(nIdx);
                            safeMoves.Add(nIdx);
                        }
                    }
                }
                // Logic 3: Unrevealed == Value
                // No flags left to place, only unknowns remain. They must be mines.
                else if (unknownCount == val)
                {
                    for (int i = 0; i < 8; i++)
                    {
                        int nIdx = idx + Offsets[i];
                        if (nIdx < 0 || nIdx >= Board.Length) continue;

                        if (Board[nIdx] == UNKNOWN)
                        {
                            Board[nIdx] = MINE; // Flag
                        }
                    }
                }
            }

            // Output
            writer.WriteLine(safeMoves.Count);
            foreach (int idx in safeMoves)
            {
                int x = idx % W;
                int y = idx / W;
                writer.WriteLine($"{x} {y}");
            }
        }
    }
}
```
