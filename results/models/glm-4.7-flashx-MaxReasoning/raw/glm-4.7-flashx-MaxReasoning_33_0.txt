```csharp
using System;
using System.Collections.Generic;
using System.IO;

public class QAP
{
    public static void Main()
    {
        using (StreamReader sr = new StreamReader(Console.OpenStandardInput()))
        using (StreamWriter sw = new StreamWriter(Console.OpenStandardOutput()))
        {
            // 1. Read Input
            if (sr.EndOfStream) return;
            int N = int.Parse(sr.ReadLine().Trim());

            long[,] flow = new long[N, N];
            for (int i = 0; i < N; i++)
            {
                var parts = sr.ReadLine().Split();
                for (int j = 0; j < N; j++)
                {
                    flow[i, j] = long.Parse(parts[j]);
                }
            }

            long[,] dist = new long[N, N];
            for (int i = 0; i < N; i++)
            {
                var parts = sr.ReadLine().Split();
                for (int j = 0; j < N; j++)
                {
                    dist[i, j] = long.Parse(parts[j]);
                }
            }

            // 2. Solve using Simulated Annealing
            int[] perm = SolveQAP(N, flow, dist);

            // 3. Calculate Final Cost
            long totalCost = 0;
            for (int i = 0; i < N; i++)
            {
                for (int j = 0; j < N; j++)
                {
                    // Cast to long to prevent overflow during multiplication if inputs are max int
                    totalCost += flow[i, j] * (long)dist[perm[i], perm[j]];
                }
            }

            // 4. Output
            sw.WriteLine(totalCost);
            sw.WriteLine(string.Join(" ", perm));
        }
    }

    // Uses a greedy heuristic (High Flow -> High Distance) as the initial solution
    // and then refines it using Simulated Annealing.
    static int[] SolveQAP(int N, long[,] flow, long[,] dist)
    {
        // --- Initialization ---
        // Calculate total flow for each facility (influence)
        long[] facilityInfluence = new long[N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
                facilityInfluence[i] += flow[i, j];
        }

        // Calculate total distance for each location
        long[] locationInfluence = new long[N];
        for (int j = 0; j < N; j++)
        {
            for (int i = 0; i < N; i++)
                locationInfluence[j] += dist[i, j];
        }

        // Sort facilities by influence (descending)
        int[] facilityOrder = new int[N];
        for (int i = 0; i < N; i++) facilityOrder[i] = i;
        Array.Sort(facilityOrder, (a, b) => facilityInfluence[b].CompareTo(facilityInfluence[a]));

        // Sort locations by influence (descending)
        int[] locationOrder = new int[N];
        for (int i = 0; i < N; i++) locationOrder[i] = i;
        Array.Sort(locationOrder, (a, b) => locationInfluence[b].CompareTo(locationInfluence[a]));

        // Create initial permutation: assign most influential facility to most influential location
        int[] perm = new int[N];
        for (int i = 0; i < N; i++)
        {
            perm[facilityOrder[i]] = locationOrder[i];
        }

        // --- Simulated Annealing ---
        Random rnd = new Random();
        
        // Precompute initial cost
        long currentCost = CalculateCost(N, flow, dist, perm);
        
        // Parameters
        double temperature = 100000000.0; // High initial temperature
        double coolingRate = 0.9995;       // Cooling schedule
        double minTemp = 0.1;              // Termination threshold

        while (temperature > minTemp)
        {
            // Perform multiple moves at the current temperature to balance exploration
            int movesPerStep = N * 5;
            for (int iter = 0; iter < movesPerStep; iter++)
            {
                // Select two random facilities to swap
                int i = rnd.Next(N);
                int j = rnd.Next(N);
                if (i == j) continue;

                // Calculate change in cost (Delta) if we swap i and j
                long delta = CalculateDeltaSwap(N, flow, dist, perm, i, j);

                // Acceptance criteria
                // Always accept if it improves the solution
                if (delta < 0 || (double)rnd.Next(int.MaxValue) / int.MaxValue < Math.Exp(-delta / temperature))
                {
                    // Apply the swap
                    int tempSwap = perm[i];
                    perm[i] = perm[j];
                    perm[j] = tempSwap;
                    currentCost += delta;
                }
            }
            // Cool down
            temperature *= coolingRate;
        }

        return perm;
    }

    // Calculates total cost of a permutation
    static long CalculateCost(int N, long[,] flow, long[,] dist, int[] perm)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                cost += flow[i, j] * dist[perm[i], perm[j]];
            }
        }
        return cost;
    }

    // Calculates the delta in cost resulting from swapping facilities i and j
    // Optimized to O(N) instead of O(N^2)
    static long CalculateDeltaSwap(int N, long[,] flow, long[,] dist, int[] perm, int i, int j)
    {
        long delta = 0;
        
        // Indices of the current locations
        long locI = perm[i];
        long locJ = perm[j];
        
        // We iterate over all potential partners k. 
        // The pairs affected by swapping i and j are (i, k) and (j, k) for all k.
        // The term (k, i) is covered when the outer loop variable is k and inner loop is i.
        for (int k = 0; k < N; k++)
        {
            if (k == i || k == j) continue;

            // Old terms involving (i, k) and (j, k)
            long oldTerm = flow[i, k] * dist[locI, perm[k]] + flow[j, k] * dist[locJ, perm[k]];
            
            // New terms involving (i, k) and (j, k) after swap
            long newTerm = flow[i, k] * dist[locJ, perm[k]] + flow[j, k] * dist[locI, perm[k]];
            
            delta += newTerm - oldTerm;
        }

        return delta;
    }
}
```
