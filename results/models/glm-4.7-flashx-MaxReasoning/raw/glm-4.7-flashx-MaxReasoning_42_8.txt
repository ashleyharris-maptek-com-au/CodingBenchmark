```text
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %ubo
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

; gl_PerVertex_in
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex_out
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Inputs
%worldPos = OpTypeArray %v3float %uint_3
%ptr_in_worldPos = OpTypePointer Input %worldPos
OpDecorate %worldPos Location 0

%normal = OpTypeArray %v3float %uint_3
%ptr_in_normal = OpTypePointer Input %normal
OpDecorate %normal Location 1

%color = OpTypeArray %v3float %uint_3
%ptr_in_color = OpTypePointer Input %color
OpDecorate %color Location 2

; Output
%outColor = OpTypePointer Output %v3float
OpDecorate %outColor Location 0

; UBO
%ubo_t = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float
OpDecorate %ubo_t Block
OpMemberDecorate %ubo_t 0 Offset 0
OpMemberDecorate %ubo_t 1 Offset 64
OpMemberDecorate %ubo_t 2 Offset 128
OpMemberDecorate %ubo_t 3 Offset 192
OpMemberDecorate %ubo_t 4 Offset 208
%ptr_ubo = OpTypePointer Uniform %ubo_t
%ubo = OpVariable %ptr_ubo Uniform

; Constants
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_7 = OpConstant %float 0.700000017
%float_3 = OpConstant %float 3.0
%float_0 = OpConstant %float 0.0

; Pointers for access
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv_v4 = OpTypePointer Output %v4float
%ptr_out_col = OpTypePointer Output %v3float

%main = OpFunction %void None %void_fn
%label = OpLabel

; Load positions from gl_in
%glin0 = OpAccessChain %ptr_in_pv %gl_in %int_0
%glin0_pos = OpAccessChain %ptr_in_v4 %glin0 %int_0
%p0 = OpLoad %v4float %glin0_pos

%glin1 = OpAccessChain %ptr_in_pv %gl_in %int_1
%glin1_pos = OpAccessChain %ptr_in_v4 %glin1 %int_0
%p1 = OpLoad %v4float %glin1_pos

%glin2 = OpAccessChain %ptr_in_pv %gl_in %int_2
%glin2_pos = OpAccessChain %ptr_in_v4 %glin2 %int_0
%p2 = OpLoad %v4float %glin2_pos

; Compute Centroid
%p_sum = OpFAdd %v4float %p0 %p1
%p_sum2 = OpFAdd %v4float %p_sum %p2
%c = OpFDiv %v4float %p_sum2 %float_3

; Prepare output pointers
%glout_struct = OpAccessChain %ptr_out_pv %gl_out %int_0
%glout_v4 = OpAccessChain %ptr_out_pv_v4 %glout_struct %int_0
%outcol_scalar = OpAccessChain %ptr_out_col %outColor %int_0

; Emit Vertex 0
%diff0 = OpFSub %v4float %p0 %c
%scaled0 = OpFMul %v4float %diff0 %float_0_7
%pos0 = OpFAdd %v4float %c %scaled0
OpStore %glout_v4 %pos0

%col0 = OpAccessChain %ptr_in_v3 %color %int_0
%load_col0 = OpLoad %v3float %col0
OpStore %outcol_scalar %load_col0
OpEmitVertex

; Emit Vertex 1
%diff1 = OpFSub %v4float %p1 %c
%scaled1 = OpFMul %v4float %diff1 %float_0_7
%pos1 = OpFAdd %v4float %c %scaled1
OpStore %glout_v4 %pos1

%col1 = OpAccessChain %ptr_in_v3 %color %int_1
%load_col1 = OpLoad %v3float %col1
OpStore %outcol_scalar %load_col1
OpEmitVertex

; Emit Vertex 2
%diff2 = OpFSub %v4float %p2 %c
%scaled2 = OpFMul %v4float %diff2 %float_0_7
%pos2 = OpFAdd %v4float %c %scaled2
OpStore %glout_v4 %pos2

%col2 = OpAccessChain %ptr_in_v3 %color %int_2
%load_col2 = OpLoad %v3float %col2
OpStore %outcol_scalar %load_col2
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
```
