```cpp
#include <iostream>
#include <vector>
#include <unordered_set>
#include <algorithm>
#include <cmath>
#include <limits>
#include <sstream>

using namespace std;

// Custom hash for std::vector<int> to use in unordered_set
struct VectorHash {
    size_t operator()(const std::vector<int>& v) const {
        size_t seed = 0;
        for (int c : v) {
            // Combines hash of individual integers
            seed ^= std::hash<int>{}(c) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
        }
        return seed;
    }
};

// Wrapper struct for coordinates to facilitate hashing
struct Coord {
    vector<int> pos;

    Coord(vector<int> p) : pos(p) {}

    bool operator==(const Coord& other) const {
        return pos == other.pos;
    }
};

// Specialize std::hash for Coord
namespace std {
    template<>
    struct hash<Coord> {
        size_t operator()(const Coord& c) const {
            return VectorHash()(c.pos);
        }
    };
}

int main() {
    // Optimize I/O for large input sizes and high frequency output
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int D, max_turns;
    if (!(cin >> D >> max_turns)) return 0;

    // Bounds for each dimension
    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) {
        cin >> bounds[i];
    }

    // Snake initial state
    int length;
    vector<int> start_pos(D);
    cin >> length;
    for (int i = 0; i < D; ++i) {
        cin >> start_pos[i];
    }

    // Food items
    int F;
    cin >> F;
    vector<vector<int>> food(F, vector<int>(D));
    for (int i = 0; i < F; ++i) {
        for (int d = 0; d < D; ++d) {
            cin >> food[i][d];
        }
    }

    // Obstacles
    int O;
    cin >> O;
    unordered_set<vector<int>, VectorHash> obstacles;
    for (int i = 0; i < O; ++i) {
        vector<int> obs(D);
        for (int d = 0; d < D; ++d) {
            cin >> obs[d];
        }
        obstacles.insert(obs);
    }

    // Initialize Snake State
    // We store snake body as a vector of Coords, Head at index 0, Tail at end
    vector<Coord> snake;
    snake.reserve(length);
    
    // Initialize head
    snake.emplace_back(start_pos);

    // Initialize body: Assume snake is a straight line extending from head in negative X direction (axis 0)
    // This is a standard deterministic initialization for N-d snake.
    vector<int> current = start_pos;
    for (int i = 0; i < length - 1; ++i) {
        current[0]--; 
        snake.emplace_back(current);
    }

    // Store occupied cells for O(1) collision detection
    unordered_set<Coord, hash<Coord>> occupied;
    for (const auto& s : snake) {
        occupied.insert(s);
    }

    int remaining_food = F;
    int turns = 0;

    // Game Loop
    while (turns < max_turns && remaining_food > 0) {
        const Coord& head = snake.front();
        const vector<int>& h_pos = head.pos;

        int best_axis = -1;
        int best_dir = 0;
        double min_dist = numeric_limits<double>::max();

        // Evaluate all 2 * D possible moves
        for (int axis = 0; axis < D; ++axis) {
            for (int dir = -1; dir <= 1; dir += 2) {
                // Proposed new position
                vector<int> new_pos = h_pos;
                new_pos[axis] += dir;

                // 1. Check Bounds
                bool in_bounds = true;
                for (int d = 0; d < D; ++d) {
                    if (new_pos[d] < 0 || new_pos[d] >= bounds[d]) {
                        in_bounds = false;
                        break;
                    }
                }
                if (!in_bounds) continue;

                // 2. Check Obstacles
                if (obstacles.count(new_pos)) continue;

                // 3. Check Self Collision
                Coord new_pt(new_pos);
                if (occupied.find(new_pt) != occupied.end()) {
                    // Exception: If we move to the tail position, it's safe (snake slides)
                    if (new_pos != snake.back().pos) continue;
                }

                // 4. Calculate Heuristic Score (Distance to nearest food)
                double score = 0;
                for (const auto& f_pos : food) {
                    double dist = 0;
                    for (int d = 0; d < D; ++d) {
                        dist += abs(new_pos[d] - f_pos[d]);
                    }
                    score += dist; // Minimizing sum of distances acts as gravity
                }

                if (score < min_dist) {
                    min_dist = score;
                    best_axis = axis;
                    best_dir = dir;
                }
            }
        }

        // If no valid move found (likely dead), we pick a fallback to satisfy the turn
        if (best_axis == -1) {
            // Try to find a valid bound neighbor (defaulting to axis 0)
            best_axis = 0;
            best_dir = 1; 
            // Ideally we would search for a valid move, but if game is over, outputting anything is compliant.
        }

        // Execute Move
        vector<int> new_head = h_pos;
        new_head[best_axis] += best_dir;
        
        // Add new head to snake and occupied set
        snake.insert(snake.begin(), Coord(new_head));
        occupied.insert(Coord(new_head));

        bool ate = false;
        // Check for food
        for (int f = 0; f < F; ++f) {
            if (food[f] == new_head) {
                ate = true;
                remaining_food--;
                // If ate, we do NOT pop the tail (snake grows)
                break;
            }
        }

        if (!ate) {
            // If not ate, remove tail
            const Coord& tail = snake.back();
            occupied.erase(tail);
            snake.pop_back();
        }

        // Output Move
        cout << best_axis << " " << best_dir << "\n";
        turns++;
    }

    return 0;
}
```
