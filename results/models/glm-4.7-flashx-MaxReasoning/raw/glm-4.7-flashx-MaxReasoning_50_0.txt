```glsl
#version 450

layout(local_size_x = 256) in;

// Input and output buffers
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

// Uniform parameters
layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    
    // Guard against N
    if (idx >= counts.x) return;

    // Read current position and velocity
    // Index mapping: i*2 and i*2+1
    vec4 posData = inData[idx * 2];
    vec4 velData = inData[idx * 2 + 1];
    vec3 pos = posData.xyz;
    vec3 vel = velData.xyz;

    // Accumulators for flocking forces
    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    // Pre-calculate squared radii for optimization
    float sepRadSq = radii.y * radii.y;
    float alignRadSq = radii.z * radii.z;
    float cohRadSq = radii.w * radii.w;
    float boundLimit = bounds.x * 0.4; // BoundSize * 0.4

    // Iterate over all boids to find neighbors
    for (uint j = 0; j < counts.x; j++) {
        if (j == idx) continue;

        vec4 jPosData = inData[j * 2];
        vec4 jVelData = inData[j * 2 + 1];
        vec3 jPos = jPosData.xyz;
        vec3 jVel = jVelData.xyz;

        // Vector from neighbor to self
        vec3 diff = pos - jPos;
        float distSq = dot(diff, diff);

        // Guard against zero distance neighbors
        if (distSq < 0.00001) continue;

        // Separation (only check if distance squared is smaller than radius squared)
        if (distSq < sepRadSq) {
            // Add normalized direction vector away from neighbor
            sepForce += normalize(diff);
            sepCount++;
        }

        // Alignment (check radius)
        if (distSq < alignRadSq) {
            alignForce += jVel;
            alignCount++;
        }

        // Cohesion (check radius)
        if (distSq < cohRadSq) {
            cohForce += jPos;
            cohCount++;
        }
    }

    // Apply Separation
    if (sepCount > 0) {
        sepForce = normalize(sepForce); // Normalize average direction
    }

    // Apply Alignment
    if (alignCount > 0) {
        alignForce = alignForce / float(alignCount); // Average velocity
        alignForce -= vel; // Steering = desired - current
        alignForce = normalize(alignForce);
    }

    // Apply Cohesion
    if (cohCount > 0) {
        cohForce = cohForce / float(cohCount); // Average position
        cohForce -= pos; // Steering = desired - current
        cohForce = normalize(cohForce);
    }

    // Boundary Force
    // Apply when any component exceeds boundSize * 0.4
    // Push toward center using boundForce
    if (abs(pos.x) > boundLimit || abs(pos.y) > boundLimit || abs(pos.z) > boundLimit) {
        // Push towards (0,0,0)
        vec3 dirToCenter = normalize(-pos);
        // Add force vector
        sepForce += dirToCenter * bounds.y;
    }

    // Calculate total acceleration (weighted sum)
    // We apply weights to the unit steering vectors
    vec3 accel = vec3(0.0);
    accel += sepForce * weights.x;
    accel += alignForce * weights.y;
    accel += cohForce * weights.z;

    // Update velocity: v = v + accel * dt
    vel += accel * radii.x;

    // Clamp speed to maxSpeed
    float speed = length(vel);
    if (speed > weights.w) {
        vel = normalize(vel) * weights.w;
    }

    // Update position: p = p + vel * dt
    pos += vel * radii.x;

    // Write to output buffer
    outData[idx * 2] = vec4(pos, 0.0);
    outData[idx * 2 + 1] = vec4(vel, 0.0);
}
```
