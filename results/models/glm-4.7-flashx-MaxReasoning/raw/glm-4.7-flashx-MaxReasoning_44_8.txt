```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize interpolated normal
    vec3 N = normalize(normal);
    
    // Calculate light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate Lambertian diffuse term
    float d = max(dot(N, L), 0.0);

    // Quantize diffuse term into bands based on thresholds
    // Thresholds: 0.75, 0.5, 0.25
    // Corresponding values: 1.0, 0.7, 0.4, 0.2
    float level;
    if (d > 0.75) {
        level = 1.0;
    } else if (d > 0.5) {
        level = 0.7;
    } else if (d > 0.25) {
        level = 0.4;
    } else {
        level = 0.2;
    }

    // Multiply vertex color by the quantized level and output
    outColor = vec4(color * level, 1.0);
}
```
