```javascript
const fs = require('fs');

// Read input
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.probability;
const drones = data.drones;
const maxSteps = data.maxStepsPerDrone;

const H = grid.length;
const W = grid[0].length;
const TOTAL_CELLS = H * W;

function solve(startX, startY) {
    // Use flat arrays for memory efficiency and speed
    let currentValues = new Float32Array(TOTAL_CELLS).fill(-Infinity);
    let parents = new Int32Array(TOTAL_CELLS).fill(-1);

    const startIdx = startY * W + startX;
    currentValues[startIdx] = grid[startY][startX];

    // Iterate through steps
    for (let step = 1; step <= maxSteps; step++) {
        let nextValues = new Float32Array(TOTAL_CELLS).fill(-Infinity);
        let nextParents = new Int32Array(TOTAL_CELLS).fill(-1);

        for (let y = 0; y < H; y++) {
            const rowOffset = y * W;
            for (let x = 0; x < W; x++) {
                const currIdx = rowOffset + x;
                const curVal = currentValues[currIdx];

                // Skip unreachable cells
                if (curVal === -Infinity) continue;

                // Option 1: Stay in place
                const stayVal = curVal + grid[y][x];
                if (stayVal > nextValues[currIdx]) {
                    nextValues[currIdx] = stayVal;
                    nextParents[currIdx] = currIdx;
                }

                // Option 2: Move Up
                if (y > 0 && grid[y-1][x] !== -1) {
                    const nIdx = (y - 1) * W + x;
                    const nVal = curVal + grid[y-1][x];
                    if (nVal > nextValues[nIdx]) {
                        nextValues[nIdx] = nVal;
                        nextParents[nIdx] = currIdx;
                    }
                }
                // Option 3: Move Down
                if (y < H - 1 && grid[y+1][x] !== -1) {
                    const nIdx = (y + 1) * W + x;
                    const nVal = curVal + grid[y+1][x];
                    if (nVal > nextValues[nIdx]) {
                        nextValues[nIdx] = nVal;
                        nextParents[nIdx] = currIdx;
                    }
                }
                // Option 4: Move Left
                if (x > 0 && grid[y][x-1] !== -1) {
                    const nIdx = rowOffset + (x - 1);
                    const nVal = curVal + grid[y][x-1];
                    if (nVal > nextValues[nIdx]) {
                        nextValues[nIdx] = nVal;
                        nextParents[nIdx] = currIdx;
                    }
                }
                // Option 5: Move Right
                if (x < W - 1 && grid[y][x+1] !== -1) {
                    const nIdx = rowOffset + (x + 1);
                    const nVal = curVal + grid[y][x+1];
                    if (nVal > nextValues[nIdx]) {
                        nextValues[nIdx] = nVal;
                        nextParents[nIdx] = currIdx;
                    }
                }
            }
        }

        // Swap buffers for next iteration
        currentValues = nextValues;
        parents = nextParents;
    }

    // Reconstruct path
    // Find the cell with the maximum value at the final step
    let bestIdx = startIdx;
    let bestVal = currentValues[startIdx];
    for (let i = 0; i < TOTAL_CELLS; i++) {
        if (currentValues[i] > bestVal) {
            bestVal = currentValues[i];
            bestIdx = i;
        }
    }

    const path = [];
    let curr = bestIdx;
    while (curr !== -1) {
        const x = curr % W;
        const y = Math.floor(curr / W);
        path.unshift([x, y]);
        curr = parents[curr];
    }

    return path;
}

const paths = drones.map(d => solve(d.x, d.y));
console.log(JSON.stringify({ paths: paths }));
```
