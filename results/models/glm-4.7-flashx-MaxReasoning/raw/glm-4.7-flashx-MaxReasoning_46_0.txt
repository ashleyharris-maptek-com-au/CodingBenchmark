```glsl
#version 450

layout(local_size_x = 256) in;

// Input buffer layout: Each body is stored as two consecutive vec4s
// Body i: inData[i*2] (pos.xyz, mass) and inData[i*2+1] (vel.xyz, 0.0)
layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

// Output buffer layout
layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

// Parameters
layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint idx = gl_GlobalInvocationID.x;

    // Guard against gl_GlobalInvocationID.x >= N
    if (idx >= counts.x) {
        return;
    }

    // Read current position and mass
    vec4 posMass = inData[idx * 2];
    vec4 vel = inData[idx * 2 + 1];

    vec3 pos = posMass.xyz;
    float mass = posMass.w;
    vec3 velocity = vel.xyz;

    // Extract physics parameters
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    uint N = counts.x;
    vec3 acceleration = vec3(0.0);

    // Compute acceleration from all other bodies
    // Loop through all bodies j
    for (uint j = 0u; j < N; ++j) {
        // Skip self-interaction
        if (j == idx) {
            continue;
        }

        // Read body j's position and mass
        vec4 otherPosMass = inData[j * 2];

        // Vector from i to j: r = pos_j - pos_i
        vec3 r = otherPosMass.xyz - pos;

        // Distance squared + softening^2
        float r2 = dot(r, r) + (softening * softening);

        // Inverse distance cubed: 1 / (r^2 + e^2)^(3/2)
        float invDistCube = 1.0 / pow(r2, 1.5);

        // Acc = G * mass_j * (r_vec) / (r^2 + e^2)^(3/2)
        acceleration += G * otherPosMass.w * r * invDistCube;
    }

    // Integration: Semi-Implicit Euler
    // 1. Update velocity: v = v + acc * dt
    velocity += acceleration * dt;

    // 2. Update position: x = x + v * dt
    pos += velocity * dt;

    // Write to output buffers
    // Position (xyz) and Mass (w) -> Mass remains constant
    outData[idx * 2] = vec4(pos, mass);
    
    // Velocity (xyz) and dummy (w)
    outData[idx * 2 + 1] = vec4(velocity, 0.0);
}
```
