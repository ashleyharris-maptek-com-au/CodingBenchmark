struct TTEntry {
    uint64_t hash;
    int bestMove;
    int depth;
    int bound; // 0=exact, 1=alpha, 2=beta
};

class TranspositionTable {
private:
    TTEntry table[TT_SIZE];
public:
    void clear() { memset(table, 0, sizeof(table)); }
    
    int probe(uint64_t h, int depth, int alpha, int beta) {
        int idx = (h ^ (h >> 32)) % TT_SIZE;
        TTEntry *e = &table[idx];
        if (e->hash == h && e->depth >= depth) {
            if (e->bound == 0) return e->bestMove;
            if (e->bound == 1 && e->bestMove <= alpha) return e->bestMove;
            if (e->bound == 2 && e->bestMove >= beta) return e->bestMove;
        }
        return -1;
    }
    
    void store(uint64_t h, int move, int depth, int bound) {
        int idx = (h ^ (h >> 32)) % TT_SIZE;
        table[idx].hash = h;
        table[idx].bestMove = move;
        table[idx].depth = depth;
        table[idx].bound = bound;
    }
};

void makeMove(Move m, bool undo = false) {
    uint64_t from = 1ULL << m.src;
    uint64_t to = 1ULL << m.dst;
    int color = board.turn;
    int type = m.type;
    int flags = m.flags;
    
    // Remove from from, add to to
    uint64_t pieceMask = from;
    uint64_t destMask = to;
    
    if (m.piece == PIECE_PAWN) {
        if (color == COLOR_WHITE) board.pawns[0] &= ~from;
        else board.pawns[1] &= ~from;
        
        if (flags & 2) { // Promotion
            destMask = 0;
            if (color == COLOR_WHITE) board.queens[0] |= to;
            else board.queens[1] |= to;
        } else {
            if (color == COLOR_WHITE) board.pawns[0] |= to;
            else board.pawns[1] |= to;
        }
    }
    else if (m.piece == PIECE_KNIGHT) {
        if (color == COLOR_WHITE) board.knights[0] &= ~from;
        else board.knights[1] &= ~from;
        if (color == COLOR_WHITE) board.knights[0] |= to;
        else board.knights[1] |= to;
    }
    else if (m.piece == PIECE_BISHOP) {
        if (color == COLOR_WHITE) board.bishops[0] &= ~from;
        else board.bishops[1] &= ~from;
        if (color == COLOR_WHITE) board.bishops[0] |= to;
        else board.bishops[1] |= to;
    }
    else if (m.piece == PIECE_ROOK) {
        if (color == COLOR_WHITE) board.rooks[0] &= ~from;
        else board.rooks[1] &= ~from;
        if (color == COLOR_WHITE) board.rooks[0] |= to;
        else board.rooks[1] |= to;
    }
    else if (m.piece == PIECE_QUEEN) {
        if (color == COLOR_WHITE) board.queens[0] &= ~from;
        else board.queens[1] &= ~from;
        if (color == COLOR_WHITE) board.queens[0] |= to;
        else board.queens[1] |= to;
    }
    else if (m.piece == PIECE_KING) {
        if (color == COLOR_WHITE) board.kings[0] &= ~from;
        else board.kings[1] &= ~from;
        if (color == COLOR_WHITE) board.kings[0] |= to;
        else board.kings[1] |= to;
    }
    
    board.occ[color] = (board.occ[color] & ~from) | to;
    
    // Castling move
    if (type == 1 || type == 2) {
        if (type == 1) { // Kingside
            board.rooks[color] = (board.rooks[color] & ~(1ULL << 7)) | (1ULL << 5);
            board.occ[color] = (board.occ[color] & ~(1ULL << 7)) | (1ULL << 5);
        } else { // Queenside
            board.rooks[color] = (board.rooks[color] & ~(1ULL << 0)) | (1ULL << 3);
            board.occ[color] = (board.occ[color] & ~(1ULL << 0)) | (1ULL << 3);
        }
    }
    
    // En Passant capture
    if (flags & 4) {
        int capSq = m.src ^ 8; // The square "behind" the pawn moving
        if (color == COLOR_WHITE) board.pawns[1] &= ~(1ULL << capSq);
        else board.pawns[0] &= ~(1ULL << capSq);
        board.occ[1 - color] &= ~(1ULL << capSq);
    }
    
    // EP handling
    if (!undo) {
        board.ep_sq = -1;
        int r = m.dst / 8;
        if (r == 2 || r == 5) { // In middle files?
            // Set EP square
            board.ep_sq = (color == COLOR_WHITE) ? (m.dst + 8) : (m.dst - 8);
        }
        
        // Castling rights
        if (m.piece == PIECE_KING) {
            if (color == COLOR_WHITE) {
                if (m.src == 60) board.castling &= ~3; // Remove WK, WQ
                else if (m.dst == 62) board.castling &= ~1; // Remove WK
                else if (m.dst == 58) board.castling &= ~2; // Remove WQ
            } else {
                if (m.src == 4) board.castling &= ~12; // Remove BK, BQ
                else if (m.dst == 6) board.castling &= ~4; // Remove BK
                else if (m.dst == 2) board.castling &= ~8; // Remove BQ
            }
        }
        if (m.piece == PIECE_ROOK) {
            if (color == COLOR_WHITE) {
                if (m.src == 56) board.castling &= ~2;
                if (m.src == 63) board.castling &= ~1;
            } else {
                if (m.src == 0) board.castling &= ~8;
                if (m.src == 7) board.castling &= ~4;
            }
        }
    }
    
    board.turn = 1 - board.turn;
    
    // Hash update
    // ... simplified hash update omitted for brevity in thought trace, 
    // but essential for TT. I will perform the XOR operations based on piece moved.
}

// --- Search ---

int search(int depth, int alpha, int beta, bool isRoot) {
    if (depth == 0) return evaluate();
    
    auto moves = generateMoves();
    if (moves.empty()) {
        return isKingInCheck(board.turn) ? -30000 : 0; // Mate or Draw
    }
    
    int bestMove = -1;
    
    // Move ordering
    // Simple: Capture first
    std::sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
        return (a.flags & 1) > (b.flags & 1);
    });
    
    for (auto m : moves) {
        makeMove(m);
        int score = -search(depth - 1, -beta, -alpha, false);
        makeMove(m, true);
        
        if (score >= beta) return beta;
        if (score > alpha) {
            alpha = score;
            bestMove = m.dst | (m.src << 6);
        }
    }
    
    return alpha;
}

int qsearch(int alpha, int beta) {
    if (alpha >= beta) return alpha;
    // Material gain only
    return evaluate();
}
