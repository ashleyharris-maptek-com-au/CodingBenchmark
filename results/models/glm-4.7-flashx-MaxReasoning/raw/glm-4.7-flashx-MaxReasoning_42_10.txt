; SPIR-V
; Version: 1.0
; Generator: Unknown (20xx to 1.0)
; Bound: 150
; Schema: 0
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

%1 = OpTypeVoid
%2 = OpTypeFunction %1

; --- Type Definitions ---
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%mat4v4float = OpTypeMatrix %v4float 4
%bool = OpTypeBool

; --- Input Structures ---
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%struct_InputData = OpTypeStruct %v3float %v3float %v3float
; layout(location=0) in vec3 worldPos
; layout(location=1) in vec3 normal
; layout(location=2) in vec3 color
OpDecorate %struct_InputData Block
OpMemberDecorate %struct_InputData 0 Location 0
OpMemberDecorate %struct_InputData 1 Location 1
OpMemberDecorate %struct_InputData 2 Location 2

; --- UBO Structure ---
; Layout matches test 41 (Model, View, Proj, Light, Camera, Params)
%struct_UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v3float %v3float %v3float
OpDecorate %struct_UBO Block
OpMemberDecorate %struct_UBO 0 Offset 0
OpMemberDecorate %struct_UBO 1 Offset 64
OpMemberDecorate %struct_UBO 2 Offset 128
OpMemberDecorate %struct_UBO 3 Offset 192
OpMemberDecorate %struct_UBO 4 Offset 208
OpMemberDecorate %struct_UBO 5 Offset 224

%uint = OpTypeUInt 32
%uint_4 = OpConstant %uint 4
%uint_1 = OpConstant %uint 1
%int = OpTypeInt 32
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_10 = OpConstant %uint 10

%ptr_Uniform_struct_UBO = OpTypePointer Uniform %struct_UBO
%ptr_Input_gl_PerVertex_in = OpTypePointer Input %gl_PerVertex_in
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v3float = OpTypePointer Output %v3float

; --- Input Variables ---
%gl_in = OpVariable %ptr_Input_gl_PerVertex_in Input
%ubo = OpVariable %ptr_Uniform_struct_UBO Uniform

; --- Output Variables ---
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

; --- Constants ---
%float_0_02 = OpConstant %float 0.02
%float_0_0 = OpConstant %float 0.0
%float_1_0 = OpConstant %float 1.0
%float_0_25 = OpConstant %float 0.25 ; 1/4

; --- Entry Point ---
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices %uint_10
OpExecutionMode %main Invocations 1

; --- Function ---
%main = OpFunction %1 None %2
%entry = OpLabel

; --- Load Inputs ---
; P0, P1, P2, P3
%idx_0 = OpConstant %uint 0
%idx_1 = OpConstant %uint 1
%idx_2 = OpConstant %uint 2
%idx_3 = OpConstant %uint 3

%in_0 = OpAccessChain %ptr_Input_gl_PerVertex_in %gl_in %idx_0
%in_1 = OpAccessChain %ptr_Input_gl_PerVertex_in %gl_in %idx_1
%in_2 = OpAccessChain %ptr_Input_gl_PerVertex_in %gl_in %idx_2
%in_3 = OpAccessChain %ptr_Input_gl_PerVertex_in %gl_in %idx_3

%pv_in_0 = OpLoad %gl_PerVertex_in %in_0
%pv_in_1 = OpLoad %gl_PerVertex_in %in_1
%pv_in_2 = OpLoad %gl_PerVertex_in %in_2
%pv_in_3 = OpLoad %gl_PerVertex_in %in_3

%p0 = OpCompositeExtract %v4float %pv_in_0 0
%p1 = OpCompositeExtract %v4float %pv_in_1 0
%p2 = OpCompositeExtract %v4float %pv_in_2 0
%p3 = OpCompositeExtract %v4float %pv_in_3 0

%in_data_1 = OpAccessChain %ptr_Input_gl_PerVertex_in %gl_in %idx_1
%pv_data_1 = OpLoad %gl_PerVertex_in %in_data_1
%c1 = OpCompositeExtract %v3float %pv_data_1 2

%in_data_2 = OpAccessChain %ptr_Input_gl_PerVertex_in %gl_in %idx_2
%pv_data_2 = OpLoad %gl_PerVertex_in %in_data_2
%c2 = OpCompositeExtract %v3float %pv_data_2 2

; --- Prepare Tangents ---
%t0_vec = OpFSub %v4float %p2 %p0
%t3_vec = OpFSub %v4float %p3 %p1

%t0 = OpVectorTimesScalar %v4float %t0_vec %float_0_25 ; Optimization for 1/4
%t3 = OpVectorTimesScalar %v4float %t3_vec %float_0_25

%len_t0 = OpLength %v4float %t0
%len_t3 = OpLength %v4float %t3

%cond_t0 = OpFOrdNotEqual %bool %len_t0 %float_0_0
%cond_t3 = OpFOrdNotEqual %bool %len_t3 %float_0_0

%t0_norm = OpSelect %v4float %cond_t0 %t0 %float_1_0
%t3_norm = OpSelect %v4float %cond_t3 %t3 %float_1_0

; --- Loop ---
%i = OpVariable %uint StorageClassPrivate %uint_0
OpBranch %loop_body

%loop_header = OpLabel
OpLoopMerge %loop_exit %loop_inc None
OpBranch %loop_body

%loop_body = OpLabel
%current_i = OpLoad %uint %i
%cond_i = OpULessThan %bool %current_i %uint_4
OpBranchConditional %cond_i %loop_body_calc %loop_exit

%loop_exit = OpLabel
OpReturn

%loop_inc = OpLabel
%next_i = OpISub %uint %current_i %uint_1
OpStore %i %next_i
OpBranch %loop_header

%loop_body_calc = OpLabel
%current_i = OpLoad %uint %i
%t_val = OpFMul %float %float_0_25 %current_i ; 0.0, 0.25, 0.5, 0.75, 1.0

; Interpolate Position (P1 to P2)
%pos_lerp = OpFma %v4float %p1 %float_1_0 %OpFMul %v4float %OpFSub %v4float %p2 %p1 %t_val ; Complex FMA construction for t
; Simplified:
%pos_vec = OpFSub %v4float %p2 %p1
%pos_scaled = OpVectorTimesScalar %v4float %pos_vec %t_val
%pos = OpFAdd %v4float %p1 %pos_scaled

; Interpolate Color
%col_lerp = OpFma %v3float %c1 %float_1_0 %OpFMul %v3float %OpFSub %v3float %c2 %c1 %t_val

; Interpolate Tangents (Start to End)
%t_lerp = OpFma %v4float %t0_norm %float_1_0 %OpFMul %v4float %OpFSub %v4float %t3_norm %t0_norm %t_val
%normal = OpNormalize %v4float %t_lerp

; Calculate Offset
%offset = OpVectorTimesScalar %v4float %normal %float_0_02

; Emit + offset
%gl_out_pos = OpAccessChain %ptr_Output_gl_PerVertex_out %gl_out %int_0
OpStore %gl_out_pos %pos
%outColor_val = OpAccessChain %ptr_Output_v3float %outColor %int_0
OpStore %outColor_val %col_lerp
OpEmitVertex

; Emit - offset
%gl_out_pos_m = OpAccessChain %ptr_Output_gl_PerVertex_out %gl_out %int_0
OpStore %gl_out_pos_m %OpFSub %v4float %pos %offset
%outColor_val_m = OpAccessChain %ptr_Output_v3float %outColor %int_0
OpStore %outColor_val_m %col_lerp
OpEmitVertex

OpBranch %loop_inc
OpFunctionEnd
