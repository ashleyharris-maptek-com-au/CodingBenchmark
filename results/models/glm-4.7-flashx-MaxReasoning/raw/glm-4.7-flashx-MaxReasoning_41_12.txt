; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color %ubo_var
OpExecutionMode %main OriginUpperLeft

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 Signed
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%v2float = OpTypeVector %float 2
%mat4 = OpTypeMat %float 4
%struct_ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_Uniform_struct = OpTypePointer Uniform %struct_ubo
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_v4float = OpTypePointer Output %v4float
%func = OpTypeFunction %void

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%const_0_5 = OpConstant %float 0.5
%const_0_3 = OpConstant %float 0.3
%const_0_0 = OpConstant %float 0.0
%const_1_0 = OpConstant %float 1.0
%const_0_4 = OpConstant %float 0.4
%const_0_6 = OpConstant %float 0.6
%const_0_3_ground = OpConstant %float 0.3
%const_0_15 = OpConstant %float 0.15
%const_0_05 = OpConstant %float 0.05

%skyColor_vec3 = OpConstantComposite %v3float %const_0_4 %const_0_6 %const_1_0
%groundColor_vec3 = OpConstantComposite %v3float %const_0_3_ground %const_0_15 %const_0_05

OpDecorate %struct_ubo Block
OpMemberDecorate %struct_ubo 0 Offset 0
OpMemberDecorate %struct_ubo 1 Offset 64
OpMemberDecorate %struct_ubo 2 Offset 128
OpMemberDecorate %struct_ubo 3 Offset 192
OpMemberDecorate %struct_ubo 4 Offset 208
OpMemberDecorate %struct_ubo 5 Offset 224
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%ubo_var = OpVariable %ptr_Uniform_struct Uniform
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input
%in_tangent = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v4float Input
%out_color = OpVariable %ptr_Output_v4float Output

%main = OpFunction %void None %func
%entry_label = OpLabel

%worldPos_val = OpLoad %v3float %in_worldPos
%normal_val = OpLoad %v3float %in_normal
%color_val = OpLoad %v4float %in_color
%uv_val = OpLoad %v2float %in_uv
%tangent_val = OpLoad %v3float %in_tangent

%light_ptr = OpAccessChain %ptr_Uniform_v4float %ubo_var %int_3
%light_vec4 = OpLoad %v4float %light_ptr
%lightPos_val = OpVectorShuffle %v3float %light_vec4 %light_vec4 0 1 2

%normal_norm = OpFNormalize %v3float %normal_val

%norm_y = OpCompositeExtract %float %normal_norm 1
%factor = OpFAdd %float %norm_y %const_0_5
%factor_mul = OpFMul %float %factor %const_0_5

%sky = OpLoad %v3float %skyColor_vec3
%ground = OpLoad %v3float %groundColor_vec3

%hemisphereColor = OpFMul %v3float %sky %factor
%hemisphereColor = OpFAdd %v3float %hemisphereColor %OpFMul %v3float %ground %OpFSub %float %const_1_0 %factor

%lightDir = OpFSub %v3float %lightPos_val %worldPos_val
%lightDir = OpFNormalize %v3float %lightDir
%diff = OpDot %float %normal_norm %lightDir
%diff = OpFMax %float %diff %const_0_0
%diff_val = OpFMul %float %diff %const_0_3

%base = OpFAdd %v3float %hemisphereColor %diff_val
%alpha = OpLoad %float %const_1_0
%final_color = OpCompositeConstruct %v4float %base %alpha

OpStore %out_color %final_color
OpReturn
OpFunctionEnd
