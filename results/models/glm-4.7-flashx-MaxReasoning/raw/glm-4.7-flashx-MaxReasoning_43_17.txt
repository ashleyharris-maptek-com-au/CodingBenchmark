```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);
    
    // Calculate light direction (L)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Lambertian diffuse term
    float diffuse = max(dot(N, L), 0.0);
    
    // Apply factor (0.4 + 0.6 * diffuse)
    float intensity = 0.4 + 0.6 * diffuse;

    // Brick Pattern Logic
    const float rows = 16.0;
    const float cols = 8.0;
    float u_step = 1.0 / cols;
    float v_step = 1.0 / rows;
    
    float col = floor(input.uv.x / u_step);
    float row = floor(input.uv.y / v_step);
    
    // Local UV coordinates within the current cell
    float u_local = frac(input.uv.x / u_step) * u_step;
    float v_local = frac(input.uv.y / v_step) * v_step;
    
    // Staggered rows (half offset every other row)
    if (mod(row, 2.0) > 0.5) {
        u_local = (u_local + u_step * 0.5) % u_step;
    }
    
    // Mortar thickness
    float t_x = 0.05;
    float t_y = 0.1;
    
    // Check if pixel is in mortar
    bool isMortarX = (u_local < t_x) || (u_local > (u_step - t_x));
    bool isMortarY = (v_local < t_y) || (v_local > (v_step - t_y));
    
    // Colors
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    
    float3 finalColor = brickColor;
    if (isMortarX || isMortarY) {
        finalColor = mortarColor;
    }
    
    // Apply lighting
    finalColor *= intensity;
    
    return float4(finalColor, 1.0);
}
```
