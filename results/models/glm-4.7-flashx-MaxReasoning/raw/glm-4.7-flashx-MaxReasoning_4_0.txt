// Function to check if a point is inside a convex polyhedron
bool is_point_inside(const vector<Point>& vertices, double px, double py, double pz) {
    int n = vertices.size();
    if (n < 4) return false;

    // We determine if the point is consistently on one side of all faces.
    // We calculate the signed distance for each face (using cross product for normal).
    // The normals point outward. If dot(Normal, Point - FacePoint) >= 0 for all, it's inside.
    // To be robust against vertex ordering, we check if all distances have the same sign.
    
    bool all_positive = false;
    bool all_negative = false;

    for (int i = 0; i < n; ++i) {
        Point p0 = vertices[i];
        Point p1 = vertices[(i + 1) % n];
        Point p2 = vertices[(i + 2) % n];

        // Compute outward facing normal using (v1 - v0) x (v2 - v0)
        double nx = (p1.y - p0.y) * (p2.z - p0.z) - (p1.z - p0.z) * (p2.y - p0.y);
        double ny = (p1.z - p0.z) * (p2.x - p0.x) - (p1.x - p0.x) * (p2.z - p0.z);
        double nz = (p1.x - p0.x) * (p2.y - p0.y) - (p1.y - p0.y) * (p2.x - p0.x);

        // Signed distance
        double dot = nx * (px - p0.x) + ny * (py - p0.y) + nz * (pz - p0.z);

        if (dot > 1e-12) all_positive = true;
        else if (dot < -1e-12) all_negative = true;
