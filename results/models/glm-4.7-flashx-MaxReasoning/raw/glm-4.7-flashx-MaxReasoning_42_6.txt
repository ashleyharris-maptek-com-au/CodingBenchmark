OpCapability Geometry

OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpName %main "main"
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%int = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%lt = OpSLessThan %bool %int %int_3
%main = OpFunction %void None %void_fn
%header = OpLabel
%i = OpPhi %int %int_0 %main %next %header
%cond = OpSLessThan %bool %i %int_3
OpBranchConditional %cond %body %merge
%body = OpLabel

%gl_out_ptr = OpAccessChain %ptr_out_pv %gl_out %i

%gl_out_pos_val = OpCompositeInsert %gl_PerVertex_out %pos %gl_out_val 0
OpStore %gl_out_ptr %gl_out_pos_val
OpEmitVertex
%next = OpIAdd %int %i %int_1
OpBranch %header

OpEndPrimitive
