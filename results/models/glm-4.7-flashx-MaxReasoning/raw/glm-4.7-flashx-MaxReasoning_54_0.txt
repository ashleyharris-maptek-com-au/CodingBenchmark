```python
import math

# -----------------------------------------------------------------------------
# PID Controller Class
# -----------------------------------------------------------------------------
class PID:
    def __init__(self, kp, ki, kd, out_min=-1.0, out_max=1.0):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.out_min = out_min
        self.out_max = out_max
        self.prev_error = 0.0
        self.integral = 0.0

    def reset(self):
        self.prev_error = 0.0
        self.integral = 0.0

    def update(self, target, measured):
        error = target - measured
        self.integral += error
        derivative = error - self.prev_error
        self.prev_error = error

        output = (self.kp * error) + (self.ki * self.integral) + (self.kd * derivative)
        
        # Clamp output
        if output > self.out_max: output = self.out_max
        if output < self.out_min: output = self.out_min
        return output

# -----------------------------------------------------------------------------
# Autopilot Logic
# -----------------------------------------------------------------------------
class ILS_Autopilot:
    def __init__(self):
        self.reset()

    def reset(self):
        self.go_arounds = 0
        self.phase = "intercept" # intercept, approach, flare, rollout, go_around
        self.go_around_timer = 0.0
        
        # Configuration State
        self.flaps = 0.0
        self.gear = 0.0
        self.spoilers = 0.0
        self.reverse = 0.0
        self.autobrake = 0.0

        # PID Controllers
        # Roll: Aim for 0 roll. Strong gain needed for LOC.
        self.pid_roll = PID(0.15, 0.0, 0.02, -1.0, 1.0)
        # Pitch: Aim for 0 pitch deviation on GS.
        self.pid_pitch = PID(0.10, 0.0, 0.05, -1.0, 1.0)
        # IAS: Maintain approach speed.
        self.pid_ias = PID(0.08, 0.01, 0.0, 0.0, 1.0)

    def _get_valid_avg(self, state, base_names):
        """Fuses redundant sensors (Loc 1, 2, 3)."""
        vals = []
        for name in base_names:
            valid_key = name + "_valid"
            val_key = name
            if valid_key in state and state[valid_key]:
                vals.append(state[val_key])
        
        if not vals:
            return 0.0
        return sum(vals) / len(vals)

    def _is_stable(self, state, alt, loc, gs, ias):
        """Check if conditions are safe for approach/flare."""
        vref = state.get('target_vref_kt', 137)
        min_ias = vref - 20
        max_ias = vref + 10
        
        if alt < 500: return False # Too low
        if abs(loc) > 2: return False # Way off centerline
        if abs(gs) > 2: return False # Way off glideslope
        if ias < min_ias or ias > max_ias: return False # Speed unstable
        return True

    def _go_around(self, state):
        """Execute go-around sequence."""
        self.go_arounds += 1
        self.phase = "go_around"
        self.go_around_timer = 0.0
        
        # Throttle up
        throttle_cmd = 1.0
        
        # Pitch up
        pitch_cmd = 20.0 # Pitch up 20 degrees
        
        # Configuration for climb
        self.flaps = 30.0
        self.gear = 0.0
        self.spoilers = 0.0
        self.reverse = 0.0
        self.autobrake = 0.0

        # Return command for the immediate step
        # We return a partial state dict, controls will be filled in the main loop
        return {
            'throttle_1': throttle_cmd,
            'throttle_2': throttle_cmd,
            'elevator': 0.8, # Force pitch up
            'aileron': 0,
            'rudder': 0,
            'flaps': 30.0,
            'gear': 0.0
        }

    def step(self, state, dt):
        # 1. Extract Critical Data
        loc_dots = self._get_valid_avg(state, ['localizer_1_dots', 'localizer_2_dots', 'localizer_3_dots'])
        gs_dots = self._get_valid_avg(state, ['glideslope_1_dots', 'glideslope_2_dots', 'glideslope_3_dots'])
        
        # Radio altimeter: 0 if failed
        radio_alt = state.get('radio_alt_ft', 0)
        
        # Check weight on wheels
        on_ground = state.get('weight_on_wheels', False)
        
        # IAS: Use target if available, otherwise target_ias
        ias = state.get('target_ias_kt', 137) 
        vref = state.get('target_vref_kt', 137)

        # Engine Status
        eng1_running = state.get('engine_1_running', 1)
        eng2_running = state.get('engine_2_running', 1)

        # 2. Logic State Machine
        
        # A. Go-Around Logic
        if self.phase == "go_around":
            self.go_around_timer += dt
            # Continue pitch up, then return to intercept config
            elevator_cmd = 0.5
            throttle_cmd = 1.0
            
            # Climb to 1500ft AGL (approximated by radio alt, though radio alt 0 on climbout)
            # Use GPS Alt for height check if Radio Alt is 0 (failed)
            gps_alt = state.get('gps_1_alt_ft', 0)
            
            # We assume 1500ft is roughly safe. If we've been climbing for a while or 
            # altitude is rising, reset.
            # Simplified: If we have passed decision height (radio < 200) and are unstable,
            # we trigger this. Resetting phase happens when we are clearly climbing.
            
            # To keep it simple: If we have been in go-around for ~10s and altitude is rising
            if self.go_around_timer > 10.0 and (gps_alt > 1000):
                self.phase = "intercept"
                self.reset() # Resets control signals but keeps go-around count

            return {
                'elevator': elevator_cmd,
                'throttle_1': throttle_cmd if eng1_running else 0.0,
                'throttle_2': throttle_cmd if eng2_running else 0.0,
                'flaps': 30.0,
                'gear': 0.0,
                'spoiler': 0.0,
                'autobrake': 0.0,
                'reverse_1': 0.0,
                'reverse_2': 0.0
            }

        # B. Ground Logic
        if on_ground:
            self.phase = "rollout"
            self.target_pitch = 0
            self.target_roll = 0
            self.target_ias = 0 # Slow down
            
            # Final config
            self.flaps = 40.0
            self.gear = 1.0
            self.spoilers = 1.0
            self.reverse = 1.0
            self.autobrake = 1.0
            
            # Hold pitch/roll at 0
            return {
                'elevator': 0,
                'aileron': 0,
                'rudder': 0,
                'throttle_1': 0,
                'throttle_2': 0,
                'flaps': 40.0,
                'gear': 1.0,
                'spoiler': 1.0,
                'speedbrake': 0.0,
                'autobrake': 1.0,
                'reverse_1': 1.0,
                'reverse_2': 1.0
            }

        # C. Flare Logic
        if radio_alt < 50.0:
            self.phase = "flare"
            self.gear = 1.0 # Keep gear down
            
            # Pull up
            self.target_pitch = 25.0 
            self.target_roll = 0
            self.target_ias = 0 # Let speed bleed off
            
            # Check for hard landing (sink rate) - simulated here by "Too low, not stabilized"
            # If we are in flare and sink rate is high (simulated by being too low too fast)
            # Ideally, we use IAS to pitch up. IAS drops fast in flare.
            # We add a bit of throttle to maintain speed if it drops too much.
            
            throttle_cmd = self.pid_ias.update(0, ias)
            if throttle_cmd > 0.2: throttle_cmd = 0.2 # Minimal throttle to prevent stall
            
            # Force pitch up
            return {
                'elevator': 1.0, 
                'aileron': 0, 
                'rudder': 0,
                'throttle_1': throttle_cmd if eng1_running else 0.0,
                'throttle_2': throttle_cmd if eng2_running else 0.0,
                'flaps': 40.0,
                'gear': 1.0,
                'spoiler': 0.0
            }

        # D. Approach & Intercept (> 50ft, < 1000ft)
        if radio_alt < 1000.0:
            self.phase = "approach"
            
            # Unstability check
            if not self._is_stable(state, radio_alt, loc_dots, gs_dots, ias):
                return self._go_around(state)

            # Steering to Centerline and Glideslope
            # Target Roll: Center on Localizer. Positive dot = left? usually - is left.
            # Let's assume dots: + is left, - is right? Or vice versa. 
            # Usually on ILS display, Dot is deviation. "Right" means plane needs to go right.
            # If dot is + (Right), target roll should be Right (+).
            self.target_roll = loc_dots * 2.0 # Gain
            
            # Pitch: Center on Glideslope.
            # Positive dot = High (Above). Need Pitch Down. Negative = Low.
            self.target_pitch = -gs_dots * 2.0 # Gain
            
            # Speed: Maintain Vref - 10
            self.target_ias = vref - 10.0

        # E. Intercept Phase (> 1000ft)
        else:
            self.phase = "intercept"
            self.target_roll = loc_dots * 2.0
            self.target_pitch = -gs_dots * 2.0
            self.target_ias = vref - 10.0

        # 3. Calculate Control Outputs
        
        # Roll (Aileron)
        # Compensate for Jammed controls (e.g. Scenario 16). 
        # If state shows plane drifting despite command 0, command positive to stop drift.
        # We rely on IRS roll data to see the effect.
        aileron = self.pid_roll.update(self.target_roll, state.get('irs_1_roll_deg', 0))
        
        # Pitch (Elevator)
        elevator = self.pid_pitch.update(self.target_pitch, state.get('irs_1_pitch_deg', 0))
        
        # Yaw (Rudder)
        # Maintain heading (Track vs Heading)
        heading = state.get('irs_1_heading_deg', 0)
        track = state.get('irs_1_track_deg', 0)
        # Yaw error
        yaw_error = track - heading 
        rudder = self.pid_roll.update(yaw_error, 0) # Simple heading hold
        
        # Throttle
        throttle = self.pid_ias.update(self.target_ias, ias)
        
        # 4. Apply Limits & Engine Status
        
        # Throttle limit (0-1) + Engine fail safety
        throttle = max(0.0, min(1.0, throttle))
        if not eng1_running: throttle = 0.0
        if not eng2_running: throttle = 0.0
        
        # Elevator/Aileron/Rudder limits
        elevator = max(-1.0, min(1.0, elevator))
        aileron = max(-1.0, min(1.0, aileron))
        rudder = max(-1.0, min(1.0, rudder))

        # 5. Return Control Dict
        return {
            'elevator': elevator,
            'aileron': aileron,
            'rudder': rudder,
            'throttle_1': throttle,
            'throttle_2': throttle,
            'elevator_trim': 0.0,
            'flaps': self.flaps,
            'gear': self.gear,
            'speedbrake': 0.0,
            'spoiler': self.spoilers,
            'autobrake': self.autobrake,
            'reverse_1': self.reverse,
            'reverse_2': self.reverse
        }

# -----------------------------------------------------------------------------
# Global Instance
# -----------------------------------------------------------------------------
autopilot = ILS_Autopilot()

def autopilot_step(state: dict, dt: float) -> dict:
    return autopilot.step(state, dt)
```
