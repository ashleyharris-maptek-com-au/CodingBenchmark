```spirv
; SPIR-V Assembly Geometry Shader
; Converts input lines into tapered ribbons.

OpCapability Geometry
OpCapability Shader
OpCapability StorageBuffer

OpMemoryModel Logical GLSL450

; Types
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_pv_out = OpTypeArray %gl_PerVertex_out %uint_4
%struct_ubo = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float %float
%void = OpTypeVoid

; Function Signature
%func_void = OpTypeFunction %void

; Constants
%uint_2 = OpConstant %uint 2
%uint_4 = OpConstant %uint 4
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2_const = OpConstant %uint 2
%uint_3_const = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%1.0 = OpConstant %float 1.0
%0.04 = OpConstant %float 0.04
%0.005 = OpConstant %float 0.005
%0 = OpConstant %float 0
%1 = OpConstant %float 1

; Pointers
%ptr_gl_pv_in = OpTypePointer Input %gl_PerVertex_in
%ptr_gl_pv_out = OpTypePointer Output %gl_PerVertex_out
%ptr_arr_pv_out = OpTypePointer Output %arr_pv_out
%ptr_v3_in = OpTypePointer Input %v3float
%ptr_v3_out = OpTypePointer Output %v3float
%ptr_ubo = OpTypePointer Uniform %struct_ubo

; Variables
%gl_in = OpVariable %ptr_gl_pv_in Input
%gl_out = OpVariable %ptr_arr_pv_out Output
%worldPos_in = OpVariable %ptr_v3_in Input
%normal_in = OpVariable %ptr_v3_in Input
%color_in = OpVariable %ptr_v3_in Input
%outColor_var = OpVariable %ptr_v3_out Output
%ubo = OpVariable %ptr_ubo Uniform

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %struct_ubo Block
OpDecorateMember %struct_ubo 0 Location 0
OpDecorateMember %struct_ubo 1 Location 1
OpDecorateMember %struct_ubo 2 Location 2
OpDecorateMember %struct_ubo 3 Location 3
OpDecorateMember %struct_ubo 4 Location 4
OpDecorateMember %struct_ubo 5 Location 5
OpDecorateMember %struct_ubo 6 Location 6

%main = OpFunction %void None %func_void
%label = OpLabel

; Load Inputs
%pv0 = OpAccessChain %ptr_gl_pv_in %gl_in %uint_0
%pos0 = OpCompositeExtract %v4float %pv0 0
%col0 = OpAccessChain %ptr_v3_in %color_in %uint_0
%c0 = OpLoad %v3float %col0

%pv1 = OpAccessChain %ptr_gl_pv_in %gl_in %uint_1
%pos1 = OpCompositeExtract %v4float %pv1 0
%col1 = OpAccessChain %ptr_v3_in %color_in %uint_1
%c1 = OpLoad %v3float %col1

; Compute Geometry (InputLines)
; p1 - p0
%diff = OpFSub %v4float %pos1 %pos0
%diff_xyz = OpVectorShuffle %v3float %diff %diff 0 1 2
%len_sq = OpDot %float %diff_xyz %diff_xyz
%inv_len = OpFDiv %float %1.0 %len_sq
%dir = OpFMul %v3float %diff_xyz %inv_len

; Normalize perpendicular (up vector)
%up = OpConstantComposite %v3float %0 %1 %0
%perp = OpCross %v3float %dir %up
%perp_len_sq = OpDot %float %perp %perp
%perp_inv_len = OpFDiv %float %1.0 %perp_len_sq
%perp_norm = OpFMul %v3float %perp %perp_inv_len

; Calculate positions for the ribbon
; Start vertices (thick 0.04)
%pos_start_plus = OpFAdd %v4float %pos0 %perp_norm
%pos_start_minus = OpFSub %v4float %pos0 %perp_norm

; End vertices (thin 0.005)
%pos_end_plus = OpFAdd %v4float %pos1 %perp_norm
%pos_end_minus = OpFSub %v4float %pos1 %perp_norm

; Emit Triangle Strip (4 vertices)
; Vertex 0: P0 + Perp*0.04
%gl_pv_out0 = OpAccessChain %ptr_gl_pv_out %gl_out %uint_0
%gl_pv_struct0 = OpCompositeInsert %gl_PerVertex_out %pos_start_plus %gl_pv_out0 0
OpStore %gl_pv_out0 %gl_pv_struct0
%out_col0 = OpAccessChain %ptr_v3_out %outColor_var %uint_0
OpStore %out_col0 %c0
OpEmitVertex

; Vertex 1: P0 - Perp*0.04
%gl_pv_out1 = OpAccessChain %ptr_gl_pv_out %gl_out %uint_1
%gl_pv_struct1 = OpCompositeInsert %gl_PerVertex_out %pos_start_minus %gl_pv_out1 0
OpStore %gl_pv_out1 %gl_pv_struct1
%out_col1 = OpAccessChain %ptr_v3_out %outColor_var %uint_1
OpStore %out_col1 %c0
OpEmitVertex

; Vertex 2: P1 + Perp*0.005
%gl_pv_out2 = OpAccessChain %ptr_gl_pv_out %gl_out %uint_2
%gl_pv_struct2 = OpCompositeInsert %gl_PerVertex_out %pos_end_plus %gl_pv_out2 0
OpStore %gl_pv_out2 %gl_pv_struct2
%out_col2 = OpAccessChain %ptr_v3_out %outColor_var %uint_2
OpStore %out_col2 %c1
OpEmitVertex

; Vertex 3: P1 - Perp*0.005
%gl_pv_out3 = OpAccessChain %ptr_gl_pv_out %gl_out %uint_3
%gl_pv_struct3 = OpCompositeInsert %gl_PerVertex_out %pos_end_minus %gl_pv_out3 0
OpStore %gl_pv_out3 %gl_pv_struct3
%out_col3 = OpAccessChain %ptr_v3_out %outColor_var %uint_3
OpStore %out_col3 %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
