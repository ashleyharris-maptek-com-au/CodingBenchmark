struct PSInput
{
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0)
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // 1. UV Checkerboard 8x8
    float2 grid = floor(input.uv * 8.0);
    float isEven = step(0.5, grid.x + grid.y);
    float3 baseColor = float3(0.2);
    float3 checkColor = lerp(baseColor, float3(0.9), isEven);

    // 2. Lambertian Shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = dot(N, L);

    // 3. Factor (0.5 + 0.5 * diffuse)
    float shadingFactor = 0.5 + 0.5 * diffuse;

    // 4. Apply shading to checkerboard
    float3 finalColor = checkColor * shadingFactor;

    return float4(finalColor, 1.0);
}
