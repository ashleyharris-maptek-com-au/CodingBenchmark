```spirv
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint = OpTypeInt 32 0
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%float = OpTypeFloat 32
%float_0_03 = OpConstant %float 0.03

%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2

%pv_in = OpTypeStruct %v4float
OpDecorate %pv_in Block
OpMemberDecorate %pv_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %pv_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%pv_out = OpTypeStruct %v4float
OpDecorate %pv_out Block
OpMemberDecorate %pv_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %pv_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_pv_pos = OpTypePointer Output %v4float

%arr_v3f = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3f
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

%func_void = OpTypeFunction %void
%main = OpFunction %void None %func_void
%entry = OpLabel

%ptr_pos0 = OpAccessChain %ptr_in_v3 %pos0
%pos0 = OpLoad %v4float %ptr_pos0
%ptr_pos1 = OpAccessChain %ptr_in_v3 %pos1
%pos1 = OpLoad %v4float %ptr_pos1

%ptr_c0 = OpAccessChain %ptr_in_v3 %color %uint_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c1 = OpAccessChain %ptr_in_v3 %color %uint_1
%c1 = OpLoad %v3float %ptr_c1

%dir = OpFSub %v4float %pos1 %pos0
%dir_xy = OpVectorShuffle %v2float %dir %dir 0 1
%rot_xy = OpVectorShuffle %v2float %dir_xy %dir_xy 1 0
%perp_xy = OpExtInst %v2float GLSL.std.450 Normalize %rot_xy
%perp_scaled = OpFMul %v2float %perp_xy %float_0_03
%perp_4d = OpVectorShuffle %v4float %perp_scaled %perp_scaled 0 1 2 2

%pos_plus = OpFAdd %v4float %pos0 %perp_4d
%ptr_out = OpAccessChain %ptr_out_pv_pos %gl_out %uint_0
OpStore %ptr_out %pos_plus
OpStore %outColor %c0
OpEmitVertex

%pos_minus = OpFSub %v4float %pos0 %perp_4d
OpStore %ptr_out %pos_minus
OpStore %outColor %c0
OpEmitVertex

%pos_plus = OpFAdd %v4float %pos1 %perp_4d
OpStore %ptr_out %pos_plus
OpStore %outColor %c1
OpEmitVertex

%pos_minus = OpFSub %v4float %pos1 %perp_4d
OpStore %ptr_out %pos_minus
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
