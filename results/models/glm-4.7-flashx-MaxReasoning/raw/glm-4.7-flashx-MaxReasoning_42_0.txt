```text
OpCapability Geometry
OpCapability Shader
OpCapability StorageBuffer
OpCapability UniformBuffer
OpCapability InputAttachment
OpCapability GLSL450
OpExtension "SPV_KHR_vulkan_memory_model"
OpMemoryModel Logical GLSL450

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%bool = OpTypeBool
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_neg0_04 = OpConstant %float -0.04
%float_0_04 = OpConstant %float 0.04

; Input/Output Structs
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input Variables
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_arr_v3_in Input
%normal = OpVariable %ptr_arr_v3_in Input
%color = OpVariable %ptr_arr_v3_in Input

; Output Variables
%arr_pv_out = OpTypeArray %gl_PerVertex_out %uint_4
%ptr_arr_pv_out = OpTypePointer Output %arr_pv_out
%gl_out = OpVariable %ptr_arr_pv_out Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; UBO (Set 0, Binding 0)
; Layout: Model(4x32), View(4x32), Proj(4x32), LightPos(12), CamPos(12), Params(12)
%struct_UBO = OpTypeStruct %v4float %v4float %v4float %v3float %v3float %v3float
OpMemberDecorate %struct_UBO 0 Offset 0
OpMemberDecorate %struct_UBO 1 Offset 16
OpMemberDecorate %struct_UBO 2 Offset 32
OpMemberDecorate %struct_UBO 3 Offset 48
OpMemberDecorate %struct_UBO 4 Offset 64
OpMemberDecorate %struct_UBO 5 Offset 80
OpDecorate %struct_UBO Block
%ptr_uniform_UBO = OpTypePointer Uniform %struct_UBO
%ubo = OpVariable %ptr_uniform_UBO Uniform

%main = OpFunction %void None %void_func
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Body
%pv_in = OpLoad %gl_PerVertex_in %gl_in
%col = OpLoad %v3float %color
%pos = OpCompositeExtract %v4float %pv_in 0
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%h = OpConstant %float 0.04

; Loop counter
%loop_i = OpVariable %ptr_uniform_uint Input %uint_0
%label_start = OpLabel

%i = OpLoad %uint %loop_i
%cond = OpULessThan %bool %i %uint_4
OpBranchConditional %cond %label_body %label_end

%label_body = OpLabel
; X offset: +h if i is odd, -h if even
%i_mod_2 = OpBitwiseAnd %uint %i %uint_1
%is_x_plus = OpUGreaterThanEqual %bool %i_mod_2 %uint_1
%offset_x = OpSelect %float %float_0_04 %is_x_plus %float_neg0_04

; Y offset: +h if i >= 2, -h if i < 2
%i_div_2 = OpBitwiseShiftRightLogical %uint %i %uint_2
%is_y_plus = OpUGreaterThanEqual %bool %i_div_2 %uint_1
%offset_y = OpSelect %float %float_0_04 %is_y_plus %float_neg0_04

%local_x = OpFAdd %float %x %offset_x
%local_y = OpFAdd %float %y %offset_y
%new_pos = OpCompositeConstruct %v4float %local_x %local_y %float_0 %float_1

%store_ptr = OpAccessChain %ptr_out_pv %gl_out %i %int_0
OpStore %store_ptr %new_pos
OpStore %outColor %col
OpEmitVertex

%i_next = OpIAdd %uint %i %uint_1
OpStore %loop_i %i_next
OpBranch %label_start

%label_end = OpLabel
OpEndPrimitive
OpReturn
OpEndFunction
```
