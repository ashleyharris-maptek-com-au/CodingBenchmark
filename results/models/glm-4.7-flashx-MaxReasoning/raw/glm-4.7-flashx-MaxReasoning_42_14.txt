OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3

; --- gl_PerVertex types ---
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%uint_6 = OpConstant %uint 6
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_arr_gl_PerVertex_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%arr_gl_PerVertex_out = OpTypeArray %gl_PerVertex_out %uint_1
%ptr_arr_gl_PerVertex_out = OpTypePointer Output %arr_gl_PerVertex_out
%gl_out = OpVariable %ptr_arr_gl_PerVertex_out Output

; --- User Input types ---
%ptr_in_v3float = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_v3float Input
%normal = OpVariable %ptr_in_v3float Input
%color = OpVariable %ptr_in_v3float Input

; --- Output types ---
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

; --- Function ---
%func = OpTypeFunction %void %ptr_arr_gl_PerVertex_in %ptr_arr_gl_PerVertex_out %ptr_in_v3float %ptr_in_v3float %ptr_in_v3float %ptr_out_v3float

; --- Constants ---
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%int_6 = OpConstant %int 6
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_1_0 = OpConstant %float 1.0
%float_0_0 = OpConstant %float 0.0
%vec3_1_1_1 = OpConstantComposite %v3float %float_1_0 %float_1_0 %float_1_0

%main = OpFunction %void None %func
%label = OpLabel

; --- Extract Triangle Vertices (Indices 0, 2, 4) ---
%ptr_p0 = OpAccessChain %ptr_arr_gl_PerVertex_in %gl_in %int_0
%p0_struct = OpLoad %gl_PerVertex_in %ptr_p0
%p0_pos = OpCompositeExtract %v4float %p0_struct 0

%ptr_p2 = OpAccessChain %ptr_arr_gl_PerVertex_in %gl_in %int_2
%p2_struct = OpLoad %gl_PerVertex_in %ptr_p2
%p2_pos = OpCompositeExtract %v4float %p2_struct 0

%ptr_p4 = OpAccessChain %ptr_arr_gl_PerVertex_in %gl_in %int_4
%p4_struct = OpLoad %gl_PerVertex_in %ptr_p4
%p4_pos = OpCompositeExtract %v4float %p4_struct 0

; Extract XYZ for math
%p0_xyz = OpVectorShuffle %v3float %p0_pos %p0_pos 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2_pos %p2_pos 0 1 2
%p4_xyz = OpVectorShuffle %v3float %p4_pos %p4_pos 0 1 2

; --- Extract Colors ---
%ptr_c0 = OpAccessChain %ptr_in_v3float %color %int_0
%c0 = OpLoad %v3float %ptr_c0

%ptr_c2 = OpAccessChain %ptr_in_v3float %color %int_2
%c2 = OpLoad %v3float %ptr_c2

%ptr_c4 = OpAccessChain %ptr_in_v3float %color %int_4
%c4 = OpLoad %v3float %ptr_c4

; --- Average Color ---
%avg_color = OpFDiv %v3float (OpFAdd %v3float (OpFAdd %v3float %c0 %c2) %c4) %float_1_0

; --- Calculate Face Normal (Manual Cross Product) ---
; v0 = p2 - p0
%v0 = OpFSub %v3float %p2_xyz %p0_xyz
; v1 = p4 - p0
%v1 = OpFSub %v3float %p4_xyz %p0_xyz

; cross.x = v0.y * v1.z - v0.z * v1.y
%cx = OpFSub %float (OpFMul %float (OpCompositeExtract %float %v0 1) (OpCompositeExtract %float %v1 2)) (OpFMul %float (OpCompositeExtract %float %v0 2) (OpCompositeExtract %float %v1 1))
%cy = OpFSub %float (OpFMul %float (OpCompositeExtract %float %v0 2) (OpCompositeExtract %float %v1 0)) (OpFMul %float (OpCompositeExtract %float %v0 0) (OpCompositeExtract %float %v1 2))
%cz = OpFSub %float (OpFMul %float (OpCompositeExtract %float %v0 0) (OpCompositeExtract %float %v1 1)) (OpFMul %float (OpCompositeExtract %float %v0 1) (OpCompositeExtract %float %v1 0))
%face_normal = OpCompositeConstruct %v3float %cx %cy %cz

; --- Lighting ---
; Normalize
%face_normal_norm = OpExtInst %v3float GLSL.std.450 Normalize %face_normal

; Dot product with light direction (1, 1, 1)
%diffuse = OpDot %float %face_normal_norm %vec3_1_1_1

; Max(0.0, diffuse)
%diffuse = OpExtInst %float GLSL.std.450 FMax %diffuse %float_0_0

; Attenuation = 0.2 + 0.8 * diffuse
%attenuation = OpFAdd %float %float_0_2 (OpFMul %float %float_0_8 %diffuse)

; Final Color = avg_color * attenuation
%final_color = OpFMul %v3float %avg_color %attenuation

OpStore %outColor %final_color

; --- Emit Vertices ---
; Emit Vertex 0 (p0)
%ptr_gout_p0 = OpAccessChain %ptr_arr_gl_PerVertex_out %gl_out %int_0
OpStore %ptr_gout_p0 %p0_pos
OpEmitVertex

; Emit Vertex 2 (p2)
%ptr_gout_p2 = OpAccessChain %ptr_arr_gl_PerVertex_out %gl_out %int_0
OpStore %ptr_gout_p2 %p2_pos
OpEmitVertex

; Emit Vertex 4 (p4)
%ptr_gout_p4 = OpAccessChain %ptr_arr_gl_PerVertex_out %gl_out %int_0
OpStore %ptr_gout_p4 %p4_pos
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
