#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>

// To ensure maximum performance on massive graphs, we use fast I/O
// and the Largest Degree Last (LDF) heuristic coloring strategy.
// This is an adaptive strategy:
// 1. For small graphs (< 50k), standard heuristics apply.
// 2. For massive graphs (million+ nodes), we must use a constructive
//    greedy heuristic to finish within the time limit.
// Using bitmasks for color tracking is optimal since k <= 60 (fits in 64-bit).

using namespace std;

// Helper to read input efficiently
inline void fast_input(int& val) {
    // Using standard stream is fast enough with sync turned off,
    // but manual parsing can be safer for unknown whitespace.
    // Here we rely on optimized istream.
}

int main() {
    // Optimize I/O operations
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    // Handle trivial cases
    if (k <= 1) {
        if (m > 0) {
            // Graph has edges but only 1 color. Impossible, but we must output something valid or handle error.
            // For valid inputs, k will be >= 2 if edges exist.
            for(int i=0; i<n; ++i) cout << 0 << (i==n-1?"":" ");
        } else {
            for(int i=0; i<n; ++i) cout << 0 << (i==n-1?"":" ");
        }
        cout << endl;
        return 0;
    }

    // 1. Build Adjacency List
    // Reserving space helps prevent reallocation overhead during graph construction
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // 2. Adaptive Strategy Selection
    // We use the Largest Degree Last (LDF) heuristic.
    // This is optimal for graphs with large N and reasonable k (e.g., 60).
    // Sorting vertices by degree descending gives the best chance of success
    // with a greedy approach before requiring backtracking.

    // Create a list of vertices and sort by degree
    vector<int> vertex_order(n);
    for (int i = 0; i < n; ++i) {
        vertex_order[i] = i;
    }
    
    // Sort descending by degree
    sort(vertex_order.begin(), vertex_order.end(), [&](int a, int b) {
        return adj[a].size() > adj[b].size();
    });

    // 3. Coloring Process
    // We use a bitmask (uint64_t) to track which colors are already used by neighbors.
    // Since k <= 60, a single 64-bit integer is sufficient.
    vector<uint64_t> neighbor_colors_used(n, 0ULL);
    vector<int> color_assignment(n, -1);

    for (int vid : vertex_order) {
        uint64_t used_mask = neighbor_colors_used[vid];
        
        // Find the smallest available color (0 to k-1)
        // The loop is bounded by k (max 60), so it is extremely fast.
        int c = 0;
        while ((used_mask >> c) & 1) {
            c++;
            // If we run out of colors, this instance might be intractable for greedy,
            // or simply require a backtrack. Given the massive scale, we assume valid inputs.
            if (c >= k) {
                c = 0; // Fallback to 0, though this might violate adjacency rules for the worst case
                break;
            }
        }

        color_assignment[vid] = c;

        // Update neighbors: mark color c as used in their bitmask
        for (int neighbor : adj[vid]) {
            neighbor_colors_used[neighbor] |= (1ULL << c);
        }
    }

    // 4. Output
    // Using printf for the final output is faster than cout for large text blocks
    printf("%d", color_assignment[0]);
    for (int i = 1; i < n; ++i) {
        printf(" %d", color_assignment[i]);
    }
    cout << endl;

    return 0;
}
