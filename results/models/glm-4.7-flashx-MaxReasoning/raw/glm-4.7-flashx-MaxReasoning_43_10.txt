```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Calculate stripe pattern using sine wave
    float sinVal = sin(input.uv.y * 3.14159 * 16.0);
    // floor(sinVal) will be 0 in positive regions, -1 in negative regions
    // step(0.0, x) returns 1 if x >= 0, else 0
    float isRed = step(0.0, floor(sinVal));
    
    // Interpolate between Blue (0.1, 0.1, 0.9) and Red (0.9, 0.1, 0.1)
    float3 stripeColor = lerp(float3(0.1, 0.1, 0.9), float3(0.9, 0.1, 0.1), isRed);

    // Lambertian Shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L));
    
    // Apply lighting factor
    float intensity = 0.4 + 0.6 * diffuse;

    // Combine color and intensity
    return float4(stripeColor * intensity, 1.0);
}
```
