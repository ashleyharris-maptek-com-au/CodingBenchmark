; SPIR-V 1.0
OpCapability Shader
OpCapability GLSL450
OpMemoryModel Logical GLSL450

%float = type float
%vec2 = type vector %float 2
%vec4 = type vector %float 4
%int = type int
%void = type void
%_ptr_Input_vec2 = type %ptr Input %vec2
%_ptr_Output_vec4 = type %ptr Output %vec4
%_ptr_Uniform_UBO = type %ptr Uniform %UBO
%UBO = type struct { %mat4, %mat4, %mat4, %vec4, %vec4, %vec4 }
%mat4 = type matrix %vec4 4
%func = type %func %void

%const_0 = OpConstant %float 0
%const_1 = OpConstant %float 1
%const_2 = OpConstant %float 2
%const_3 = OpConstant %float 3
%const_4 = OpConstant %float 4
%const_half = OpConstant %float 0.5
%const_20 = OpConstant %int 20

%in_loc2 = OpVariable %_ptr_Input_vec2 Input
%out_loc0 = OpVariable %_ptr_Output_vec4 Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

OpDecorate %ubo Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %in_loc2 Location 2
OpDecorate %out_loc0 Location 0

OpEntryPoint Fragment %main "main" %in_loc2 %out_loc0
OpExecutionMode %main OriginUpperLeft

%main = OpFunction %void None %func
%1 = OpLabel
%uv = OpLoad %vec2 %in_loc2

; c = (uv.x * 3 - 2, uv.y * 2 - 1)
%c_x = OpCompositeExtract %float %uv 0
%c_x = OpFMul %float %c_x %const_3
%c_x = OpFSub %float %c_x %const_2

%c_y = OpCompositeExtract %float %uv 1
%c_y = OpFMul %float %c_y %const_2
%c_y = OpFSub %float %c_y %const_1

%z = OpConstantComposite %vec2 %const_0 %const_0
%iter = OpConstant %int 0

%loop_header = OpLabel
%iter_lt_20 = OpSLessThan %bool %iter %const_20
OpBranchConditional %iter_lt_20 %check_escape %loop_merge

%check_escape = OpLabel
%sq_x = OpFMul %float %z.x %z.x
%sq_y = OpFMul %float %z.y %z.y
%dist_sq = OpFAdd %float %sq_x %sq_y
%escaped = OpFOrdGreaterThan %bool %dist_sq %const_4
OpBranchConditional %escaped %loop_merge %loop_body

%loop_body = OpLabel
%x2 = OpFMul %float %z.x %z.x
%y2 = OpFMul %float %z.y %z.y
%xy = OpFMul %float %z.x %z.y
%y_new = OpFMul %float %xy %const_2
%x_new = OpFSub %float %x2 %y2
%x_new = OpFAdd %float %x_new %c_x
%y_new = OpFAdd %float %y_new %c_y

%z = OpCompositeConstruct %vec2 %x_new %y_new
%iter = OpIAdd %int %iter %const_20
OpBranch %loop_header

%loop_merge = OpLabel

; Color: (t, t*0.5, 1-t, 1)
; t = iter / 20
%iter_f = OpConvertSToF %float %iter
%scale = OpFDiv %float %iter_f %const_20
%color_x = OpFAdd %float %scale %const_0
%color_y = OpFMul %float %scale %const_half
%color_z = OpFSub %float %const_1 %scale
%color_w = OpFAdd %float %const_1 %const_0
%color = OpCompositeConstruct %vec4 %color_x %color_y %color_z %color_w

OpStore %out_loc0 %color
OpReturn
OpFunctionEnd
