```asm
; SPIR-V Assembly for Geometry Shader
; Input: InputLinesAdjacency
; Output: OutputTriangleStrip
; Task: Render sine wave curve with adjacency

OpCapability Geometry
OpCapability Shader
OpCapability Float16
OpCapability Float64
OpCapability F32
OpCapability F16
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v4float = OpTypeVector %float 4
%vec3 = OpTypeVector %float 3
%vec2 = OpTypeVector %float 2
%mat4 = OpTypeFloatMat4

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%uint_4 = OpConstant %uint 4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_arr_vec3 = OpTypePointer Input %arr_vec3
%worldPos = OpVariable %ptr_in_arr_vec3 Input
%normal = OpVariable %ptr_in_arr_vec3 Input
%color = OpVariable %ptr_in_arr_vec3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_vec3 = OpTypePointer Output %vec3
%outColor = OpVariable %ptr_out_vec3 Output

%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %vec3 %vec3 %vec3
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo Uniform

; Execution Modes
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations for user inputs
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%float_0_025 = OpConstant %float 0.025
%float_0 = OpConstant %float 0.0

%main = OpFunction %void None %void_func
%void_func = OpTypeFunction %void

; Function body
%label = OpLabel

; --- Load Inputs ---
; gl_in indices: 0=prev, 1=start, 2=end, 3=next
%ptr_v4_pv = OpTypePointer Input %v4float
%ptr_v3_vec = OpTypePointer Input %vec3

; Positions
%p0 = OpAccessChain %ptr_v4_pv %gl_in %uint_0 %int_0
%p1 = OpAccessChain %ptr_v4_pv %gl_in %uint_1 %int_0
%p2 = OpAccessChain %ptr_v4_pv %gl_in %uint_2 %int_0
%p3 = OpAccessChain %ptr_v4_pv %gl_in %uint_3 %int_0

; Colors
%c1 = OpAccessChain %ptr_v3_vec %color %uint_1
%c2 = OpAccessChain %ptr_v3_vec %color %uint_2

; --- Math: Tangents ---
; t0 = normalize(end - prev) => gl_in[2] - gl_in[0]
%diff0 = OpFSub %v4float %p2 %p0
%t0 = OpNormalize %v4float %diff0

; t1 = normalize(next - start) => gl_in[3] - gl_in[1]
%diff1 = OpFSub %v4float %p3 %p1
%t1 = OpNormalize %v4float %diff1

; Extract X and Y for perpendiculars
%tx0 = OpCompositeExtract %float %t0 0
%ty0 = OpCompositeExtract %float %t0 1
%tx1 = OpCompositeExtract %float %t1 0
%ty1 = OpCompositeExtract %float %t1 1

; Perpendiculars
%vec2_0 = OpCompositeConstruct %vec2 %float_0 %float_0
%perp0_vec = OpCompositeConstruct %vec2 (-ty0, tx0)
%perp1_vec = OpCompositeConstruct %vec2 (-ty1, tx1)

%perp0 = OpVectorTimesScalar %vec2 %perp0_vec %float_0_025
%perp1 = OpVectorTimesScalar %vec2 %perp1_vec %float_0_025

; --- Calculate Output Vertices ---
; Strip order: Start+, Start-, End+, End-

; Vertex 1: Start + perp0
%vec4_perp0 = OpCompositeConstruct %vec4 %perp0 %float_0
%pos_v1 = OpFAdd %v4float %p1 %vec4_perp0
%ptr_out_v4 = OpTypePointer Output %v4float
%out_pv_v1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pv_v1 %pos_v1
%out_col_v1 = OpAccessChain %ptr_out_vec3 %outColor %int_0
OpStore %out_col_v1 %c1
OpEmitVertex

; Vertex 2: Start - perp0
%vec4_nperp0 = OpCompositeConstruct %vec4 (-%perp0) %float_0
%pos_v2 = OpFAdd %v4float %p1 %vec4_nperp0
%out_pv_v2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pv_v2 %pos_v2
OpStore %out_col_v1 %c1
OpEmitVertex

; Vertex 3: End + perp1
%vec4_perp1 = OpCompositeConstruct %vec4 %perp1 %float_0
%pos_v3 = OpFAdd %v4float %p2 %vec4_perp1
%out_pv_v3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pv_v3 %pos_v3
%out_col_v3 = OpAccessChain %ptr_out_vec3 %outColor %int_0
OpStore %out_col_v3 %c2
OpEmitVertex

; Vertex 4: End - perp1
%vec4_nperp1 = OpCompositeConstruct %vec4 (-%perp1) %float_0
%pos_v4 = OpFAdd %v4float %p2 %vec4_nperp1
%out_pv_v4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pv_v4 %pos_v4
OpStore %out_col_v3 %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
