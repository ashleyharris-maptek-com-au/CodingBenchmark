```csharp
using System;
using System.Collections.Generic;
using System.IO;

public class Program {
    // Global best solution to minimize
    private static HashSet<int> bestSolution = new HashSet<int>();

    // Input data
    private static int n;
    private static List<HashSet<int>> graph; // Adjacency list

    public static void Main() {
        // Using StreamReader for faster input
        using (var sr = new StreamReader(System.Console.OpenStandardInput())) {
            string line;
            if ((line = sr.ReadLine()) == null) return;

            string[] parts = line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            n = int.Parse(parts[0]);
            int m = int.Parse(parts[1]);

            graph = new List<HashSet<int>>(n);
            for (int i = 0; i < n; i++) {
                graph.Add(new HashSet<int>());
            }

            for (int i = 0; i < m; i++) {
                line = sr.ReadLine();
                parts = line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
                int u = int.Parse(parts[0]);
                int v = int.Parse(parts[1]);
                graph[u].Add(v);
            }
        }

        // Step 1: Find Strongly Connected Components (SCCs)
        // We only need to solve the problem on non-trivial SCCs (size > 1)
        // Since edges between SCCs go one way, solving them independently sums up to the global optimal.
        int[] comp = TarjanSCC(n, graph);

        bool hasNonTrivialSCC = false;
        for (int i = 0; i < n; i++) {
            // We need to find the size of the component for vertex i
            int currentComp = comp[i];
            // Count occurrences of currentComp
            int count = 0;
            for (int j = 0; j < n; j++) {
                if (comp[j] == currentComp) count++;
            }
            if (count > 1) {
                hasNonTrivialSCC = true;
                // Step 2: Solve FVS for this SCC
                // We pass a subgraph containing only vertices of this component
                SolveComponent(i, count);
            }
        }

        if (!hasNonTrivialSCC) {
            Console.WriteLine(0);
        } else {
            // Output
            Console.WriteLine(bestSolution.Count);
            // Sort for consistent output (though any order is valid)
            var sorted = new List<int>(bestSolution);
            sorted.Sort();
            Console.WriteLine(string.Join(" ", sorted));
        }
    }

    // Recursive Search for FVS
    private static void SolveComponent(int startVertexIndex, int componentSize) {
        // 'active' stores vertices currently in the graph (not removed)
        HashSet<int> active = new HashSet<int>();
        // Initialize active with all vertices in this component
        for (int i = 0; i < n; i++) {
            if (comp[i] == comp[startVertexIndex]) {
                active.Add(i);
            }
        }
        
        // We don't need to pass 'startVertexIndex' as a parameter to the recursive function
        // because 'active' already defines the current subgraph.
        // However, to support SCC decomposition, we structure it so it only touches vertices in this component.
        
        // Note: The recursive logic below needs to handle the "active" set efficiently.
        // We use a global 'comp' map to check if we are still inside the component.
        
        DFS(active);
    }

    private static void DFS(HashSet<int> active) {
        // Pruning: If current solution is already worse than best, stop
        if (bestSolution.Count <= active.Count) return;

        // Find a cycle in the current active graph
        // To ensure we find *any* cycle, we iterate through active vertices and try to find a path back to themselves.
        // Note: In a pure DFS, we stop at the first found cycle.
        List<int> cycle = FindCycle(active);
        
        if (cycle == null) {
            // No cycles found, we found a minimum solution for this branch
            bestSolution = new HashSet<int>(active); // Actually, active now contains the vertices NOT removed
            return;
        }

        // Pick a vertex to branch on.
        // Ideally, pick the one that reduces the graph most, or just the smallest index for consistency.
        int v = cycle[0]; 
        // Or v = cycle[cycle.Count / 2]; // Pick middle
        // Or v = cycle[cycle.Count - 1];
        // Standard heuristic: pick smallest index to keep recursion search space deterministic
        v = cycle[0]; 

        // Branch 1: Remove v
        HashSet<int> active1 = new HashSet<int>(active);
        active1.Remove(v);
        DFS(active1);

        // Branch 2: Do not remove v, remove all vertices reachable from v
        // This is sound: if we keep v, it cannot be in any cycle. 
        // Therefore, all paths starting from v must be cut.
        HashSet<int> reachable = GetReachable(active, v);
        HashSet<int> active2 = new HashSet<int>(active);
        foreach (int u in reachable) {
            active2.Remove(u);
        }
        DFS(active2);
    }

    // Find a cycle in the graph defined by 'active'
    private static List<int> FindCycle(HashSet<int> active) {
        foreach (int start in active) {
            Dictionary<int, int> parent = new Dictionary<int, int>();
            Stack<int> stack = new Stack<int>();
            stack.Push(start);
            parent[start] = -1; // Mark visited

            while (stack.Count > 0) {
                int u = stack.Pop();

                foreach (int v in graph[u]) {
                    if (!active.Contains(v)) continue;
                    
                    if (v == start) {
                        // Found a cycle
                        List<int> cycle = new List<int>();
                        cycle.Add(v); // Add the target
                        int curr = u;
                        while (curr != start) {
                            cycle.Add(curr);
                            curr = parent[curr];
                        }
                        cycle.Add(start); // Close the loop for clarity, though we just need vertices
                        return cycle;
                    }

                    if (!parent.ContainsKey(v)) {
                        parent[v] = u;
                        stack.Push(v);
                    }
                }
            }
        }
        return null;
    }

    // Get all vertices reachable from 'v' within 'active' set
    private static HashSet<int> GetReachable(HashSet<int> active, int v) {
        HashSet<int> visited = new HashSet<int>();
        Queue<int> queue = new Queue<int>();
        queue.Enqueue(v);
        visited.Add(v);

        while (queue.Count > 0) {
            int u = queue.Dequeue();
            foreach (int w in graph[u]) {
                if (active.Contains(w) && !visited.Contains(w)) {
                    visited.Add(w);
                    queue.Enqueue(w);
                }
            }
        }
        return visited;
    }

    // Tarjan's SCC Algorithm
    private static int[] TarjanSCC(int n, List<HashSet<int>> graph) {
        int index = 0;
        Stack<int> stack = new Stack<int>();
        int[] indices = new int[n];
        int[] lowlinks = new int[n];
        bool[] onStack = new bool[n];
        HashSet<int>[] compMap = new HashSet<int>[n]; // Maps original vertex -> comp index
        for (int i = 0; i < n; i++) compMap[i] = new HashSet<int>();

        int sccCount = 0;

        for (int i = 0; i < n; i++) {
            if (indices[i] == 0) {
                StrongConnect(i, index, indices, lowlinks, onStack, stack, compMap, ref sccCount);
            }
        }

        // Assign new IDs to components (0..k-1) to make the array compact and sequential
        // compMap is sparse. We need to collect unique components first.
        var uniqueComps = new HashSet<HashSet<int>>();
        for (int i = 0; i < n; i++) uniqueComps.Add(compMap[i]);
        
        int[] compId = new int[n];
        int currentId = 0;
        foreach (var c in uniqueComps) {
            foreach (int v in c) compId[v] = currentId;
            currentId++;
        }

        return compId;
    }

    private static void StrongConnect(int v, int index, int[] indices, int[] lowlinks, bool[] onStack, Stack<int> stack, HashSet<int>[] compMap, ref int sccCount) {
        indices[v] = index;
        lowlinks[v] = index;
        index++;
        stack.Push(v);
        onStack[v] = true;

        foreach (int w in graph[v]) {
            if (indices[w] == 0) {
                StrongConnect(w, index, indices, lowlinks, onStack, stack, compMap, ref sccCount);
                lowlinks[v] = Math.Min(lowlinks[v], lowlinks[w]);
            } else if (onStack[w]) {
                lowlinks[v] = Math.Min(lowlinks[v], indices[w]);
            }
        }

        if (lowlinks[v] == indices[v]) {
            HashSet<int> component = new HashSet<int>();
            int w;
            do {
                w = stack.Pop();
                onStack[w] = false;
                component.Add(w);
            } while (w != v);
            
            compMap[v] = component;
        }
    }
}
```
