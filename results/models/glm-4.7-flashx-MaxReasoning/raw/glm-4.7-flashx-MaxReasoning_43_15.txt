```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);
    
    // Calculate light direction L = lightPos - worldPos
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Calculate t as specified: dot(N, L) * 0.5 + 0.5
    // Using params.y (0.5) here matches the specification
    float t = dot(N, L) * params.y + 0.5;
    
    // Define Gooch warm and cool colors
    float3 cool = float3(0.2, 0.2, 0.75);
    float3 warm = float3(0.7, 0.7, 0.4);
    
    // Calculate Gooch color: (1-t)*Cool + t*Warm
    float3 gooColor = lerp(cool, warm, t);
    
    // Apply the vertex color (which varies across the unit sphere)
    float3 finalColor = gooColor * input.color;
    
    return float4(finalColor, 1.0);
}
```
