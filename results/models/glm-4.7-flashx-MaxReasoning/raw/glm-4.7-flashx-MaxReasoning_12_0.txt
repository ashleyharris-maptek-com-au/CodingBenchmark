using System;
using System.Collections.Generic;
using System.Linq;

namespace BinPackingSolver

    // Core Math Structures
    public struct Vec3
    {
        public double x, y, z;
        public Vec3(double x, double y, double z) { this.x = x; this.y = y; this.z = z; }
        public double DistSq(Vec3 v) => (x - v.x) * (x - v.x) + (y - v.y) * (y - v.y) + (z - v.z) * (z - v.z);
    }

    public struct Quaternion
    {
        public double w, x, y, z;
        public Quaternion(double w, double x, double y, double z) { this.w = w; this.x = x; this.y = y; this.z = z; }
    }

    // Represents a single packed item
    public class PackItem
    {
        public Vec3 Position;
        public Quaternion Rotation;
        public Vec3 Dimensions; // Width, Height, Depth (Local AABB)
    }

    // Polyhedron Data
    public class Polyhedron
    {
        public List<Vec3> Vertices;
        public List<(int, int, int)> Faces;
        public Vec3 BBoxMin;
        public Vec3 BBoxMax;
        public Vec3[] ConvexHull; // Simplified shape for packing

        public Polyhedron(List<Vec3> v, List<(int,int,int)> f)
        {
            Vertices = v;
            Faces = f;
            ComputeBoundingVolume();
        }

        private void ComputeBoundingVolume()
        {
            BBoxMin = new Vec3(double.MaxValue, double.MaxValue, double.MaxValue);
            BBoxMax = new Vec3(double.MinValue, double.MinValue, double.MinValue);

            foreach (var v in Vertices)
            {
                if (v.x < BBoxMin.x) BBoxMin.x = v.x;
                if (v.y < BBoxMin.y) BBoxMin.y = v.y;
                if (v.z < BBoxMin.z) BBoxMin.z = v.z;
                if (v.x > BBoxMax.x) BBoxMax.x = v.x;
                if (v.y > BBoxMax.y) BBoxMax.y = v.y;
                if (v.z > BBoxMax.z) BBoxMax.z = v.z;
            }

            // Compute Convex Hull approximation (Monotone Chain) to get valid shape
            ConvexHull = QuickHull(Vertices);
        }

        // Simplified QuickHull to get approximate shape for packing
        private Vec3[] QuickHull(List<Vec3> points)
        {
            if (points.Count == 0) return new Vec3[0];
            // Simplification: Just return the vertices if hull is too complex, 
            // but we use the Bounding Box for packing anyway for performance.
            return points.ToArray(); 
        }
    }

    // Strategy Configuration
    enum ScaleLevel { Trivial, Medium, Large, Extreme }
    enum PackingStrategy { GreedyGrid, RandomizedParallel }

    class Program

        // Rotation Matrices for 90 degrees

            new Quaternion(0.7071, 0, 1, 0), // 90 X
        };

        static void Main(string[] args)
        {
            using var reader = new StreamReader(Console.OpenStandardInput());
            using var writer = new StreamWriter(Console.OpenStandardOutput());

            // 1. Parse Input
            if (reader.EndOfStream) return;
            var firstLine = reader.ReadLine().Split();
            int V = int.Parse(firstLine[0]);
            int F = int.Parse(firstLine[1]);

            var vertices = new List<Vec3>();
            for (int i = 0; i < V; i++)
            {
                var parts = reader.ReadLine().Split();
                vertices.Add(new Vec3(double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2])));
            }

            var faces = new List<(int, int, int)>();
            for (int i = 0; i < F; i++)
            {
                var parts = reader.ReadLine().Split();
                faces.Add((int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2])));
            }

            var lastLine = reader.ReadLine().Split();
            double W = double.Parse(lastLine[0]);
            double H = double.Parse(lastLine[1]);
            double D = double.Parse(lastLine[2]);

            // 2. Process Polyhedron
            var poly = new Polyhedron(vertices, faces);

            // Determine Scale
            double containerVol = W * H * D;
            double objectVol = (poly.BBoxMax.x - poly.BBoxMin.x) * 
                               (poly.BBoxMax.y - poly.BBoxMin.y) * 
                               (poly.BBoxMax.z - poly.BBoxMin.z);

            ScaleLevel level = ScaleLevel.Medium;
            if (containerVol < 10000) level = ScaleLevel.Trivial;
            else if (containerVol > 10000000) level = ScaleLevel.Extreme;

            // 3. Solve
            List<PackItem> result;
            if (level == ScaleLevel.Extreme)
            {
                result = SolveExtreme(poly, W, H, D, containerVol);
            }
            else
            {
                result = SolveStandard(poly, W, H, D);
            }

            // 4. Output
            writer.WriteLine(result.Count);
            foreach (var item in result)
            {
                writer.WriteLine($"{item.Position.x:0.000} {item.Position.y:0.000} {item.Position.z:0.000} " +
                                 $"{item.Rotation.w:0.000} {item.Rotation.x:0.000} {item.Rotation.y:0.000} {item.Rotation.z:0.000}");
            }
            writer.Flush();
        }

        // Optimization: Randomized Parallel Search for Extreme Scale
        static List<PackItem> SolveExtreme(Polyhedron poly, double W, double H, double D, double vol)
        {
            int bestCount = 0;
            List<PackItem> bestResult = new List<PackItem>();

            // Determine object dimensions
            double wx = poly.BBoxMax.x - poly.BBoxMin.x;
            double wy = poly.BBoxMax.y - poly.BBoxMin.y;
            double wz = poly.BBoxMax.z - poly.BBoxMin.z;
            
            // Choose best orientation for packing (smallest height is usually best for layering)
            double[] dims = new double[3] { wx, wy, wz };
            Array.Sort(dims); // dim0 <= dim1 <= dim2

            // Best orientation dimensions
            double bestW = dims[0]; double bestH = dims[1]; double bestD = dims[2];
            
            // Setup Grid
            // Cell size = object size / 2 to allow "floating" placement
            double cellSize = Math.Max(bestW, Math.Max(bestH, bestD)) / 2.0;
            int gridW = (int)Math.Ceiling(W / cellSize);
            int gridH = (int)Math.Ceiling(H / cellSize);
            int gridD = (int)Math.Ceiling(D / cellSize);
            
            // Use a bit array or byte array for grid. 
            // Since C# lists are slow, we use a simple heuristic search in parallel.
            
            int runs = 8; // Parallel runs
            var tasks = new Task<List<PackItem>>[runs];

            for (int i = 0; i < runs; i++)
            {
                tasks[i] = Task.Run(() => RunSingleSearch(poly, W, H, D, bestW, bestH, bestD, cellSize, gridW, gridH, gridD, seed: i));
            }

            Task.WaitAll(tasks);

            for (int i = 0; i < runs; i++)
            {
                if (tasks[i].Result.Count > bestCount)
                {
                    bestCount = tasks[i].Result.Count;
                    bestResult = tasks[i].Result;
                }
            }

            return bestResult;
        }

        static List<PackItem> RunSingleSearch(Polyhedron poly, double W, double H, double D, double bestW, double bestH, double bestD, 
            double cellSize, int gridW, int gridH, int gridD, int seed)
        {
            Random rnd = new Random(seed);
            // Positions will be mapped to grid cells for fast collision
            // Using a byte array for occupancy
            byte[,] occupied = new byte[gridW, gridH]; // 2D slice approach is faster than 3D in C# loops? 
            // Let's use a 1D flattened array for speed
            byte[] grid = new byte[gridW * gridH * gridD];
            
            List<PackItem> items = new List<PackItem>();
            
            // Heuristic: Fill the grid randomly but try to maintain stability?
            // Simple Random Fill:
            int maxAttempts = 100000; // Don't run forever
            int attempts = 0;
            int currentGridIndex = 0;

            // Start with a seed position (0,0,0)
            PlaceItem(0, 0, 0, 0, cellSize, grid, gridW, gridH, gridD, items);

            while (attempts < maxAttempts)
            {
                attempts++;
                
                // Find an empty spot. To be efficient, we look at neighbors of placed items.
                // Randomly pick an item and expand around it
                if (items.Count == 0) break;
                
                PackItem refItem = items[rnd.Next(items.Count)];
                
                // Try to place a box (reference item dimensions) adjacent to refItem
                // Dimensions of reference item
                double iw = bestW, ih = bestH, id = bestD;

                // Expansion directions
                int dir = rnd.Next(6); // 0-2 X, 3-5 Y, etc.
                int gx = 0, gy = 0, gz = 0;
                
                if (dir == 0) gx = 1; if (dir == 1) gx = -1;
                else if (dir == 2) gy = 1; else if (dir == 3) gy = -1;
                else if (dir == 4) gz = 1; else if (dir == 5) gz = -1;

                // Calculate grid cell index of the reference item's center (approximate)
                // We assume items are placed at (x*cellSize, y*cellSize, z*cellSize)
                // We need to know where the item is.
                // Since we don't store exact positions in grid, we infer:
                // We rely on the fact that we placed the reference item.
                // We just need a neighbor index.
                
                // Simple hack: we iterate randomly through the grid, checking validity.
                // This is O(N_cells * N_items) which is heavy.
                // Better: The reference item logic implies we know its grid pos?
                // No, we didn't store it.
                
                // Revert to Random Grid Sampling with rejection for this thread (simpler, robust for parallel)
                // It's slower but guarantees coverage.
                int cx = rnd.Next(gridW);
                int cy = rnd.Next(gridH);
                int cz = rnd.Next(gridD);
                
                // Check if fits (AABB overlap check is handled by grid occupancy check logic)
                // But grid occupancy is just "is cell occupied".
                // We need to check if the volume of the new object overlaps with any occupied cells.
                // This is hard without storing object grid maps.
                
                // For extreme scale, we must simplify.
                // ASSUMPTION: The polyhedron approximates a box.
                // If we place it on a grid cell (cx, cy, cz), it occupies that cell and neighbors?
                // Let's use "Block Packing" logic on the grid.
                
                // Re-approach: We are just filling the grid with blocks of size (w/cellSize) x (h/cellSize) x (d/cellSize)
                int blkW = (int)Math.Ceiling(bestW / cellSize);
                int blkH = (int)Math.Ceiling(bestH / cellSize);
                int blkD = (int)Math.Ceiling(bestD / cellSize);
                
                if (CanPlace(cx, cy, cz, blkW, blkH, blkD, grid, gridW, gridH, gridD))
                {
                    // Place
                    PlaceItem(cx, cy, cz, blkW, cellSize, grid, gridW, gridH, gridD, items);
                }
            }

            return items;
        }

        static bool CanPlace(int gx, int gy, int gz, int bw, int bh, int bd, byte[] grid, int gw, int gh, int gd)
        {
            for (int x = 0; x < bw; x++)
                for (int y = 0; y < bh; y++)
                    for (int z = 0; z < bd; z++)
                        if (gx + x < gw && gy + y < gh && gz + z < gd)
                            if (grid[(gx + x) + (gy + y) * gw + (gz + z) * gw * gh] != 0)
                                return false;
            return true;
        }

        static void PlaceItem(int gx, int gy, int gz, int bw, double cellSize, byte[] grid, int gw, int gh, int gd, List<PackItem> items)
        {
            for (int x = 0; x < bw; x++)
                for (int y = 0; y < bh; y++) // Height of block
                    for (int z = 0; z < bd; z++) // Depth of block
                        if (gx + x < gw && gy + y < gh && gz + z < gd)
                            grid[(gx + x) + (gy + y) * gw + (gz + z) * gw * gh] = 1;

            // Store Position
            double px = gx * cellSize;
            double py = gy * cellSize;
            double pz = gz * cellSize;
            
            items.Add(new PackItem 
            { 
                Position = new Vec3(px, py, pz),
                Rotation = new Quaternion(1, 0, 0, 0),
                Dimensions = new Vec3(bestW, bestH, bestD)
            });
        }

        // Standard Grid Solver for Medium/Small
        static List<PackItem> SolveStandard(Polyhedron poly, double W, double H, double D)
        {
            List<PackItem> items = new List<PackItem>();

            double wx = poly.BBoxMax.x - poly.BBoxMin.x;
            double wy = poly.BBoxMax.y - poly.BBoxMin.y;
            double wz = poly.BBoxMax.z - poly.BBoxMin.z;
            
            // Try to pack based on height to fill container bottom first
            // Sort dimensions to get target dimensions
            double[] dims = new double[3] { wx, wy, wz };
            Array.Sort(dims); 
            
            double bestW = dims[0]; double bestH = dims[1]; double bestD = dims[2];

            double cellSize = Math.Max(bestW, Math.Max(bestH, bestD)) / 2.0;
            int gridW = (int)Math.Ceiling(W / cellSize);
            int gridH = (int)Math.Ceiling(H / cellSize);
            int gridD = (int)Math.Ceiling(D / cellSize);
            
            int blkW = (int)Math.Ceiling(bestW / cellSize);
            int blkH = (int)Math.Ceiling(bestH / cellSize);
            int blkD = (int)Math.Ceiling(bestD / cellSize);

            byte[] grid = new byte[gridW * gridH * gridD];

            // Simple Greedy Layering
            // Iterate Z (Depth) first, then Y (Height), then X (Width)
            // Actually, to maximize, we want to fill gaps.
            // Simple nested loop is often the most robust for AABB packing.
            
            for (int z = 0; z < gridD; z++)
            {
                for (int y = 0; y < gridH; y++)
                {
                    for (int x = 0; x < gridW; x++)
                    {
                        if (CanPlace(x, y, z, blkW, blkH, blkD, grid, gridW, gridH, gridD))
                        {
                            PlaceItem(x, y, z, blkW, cellSize, grid, gridW, gridH, gridD, items);
                        }
                    }
                }
            }

            return items;
        }
    }
