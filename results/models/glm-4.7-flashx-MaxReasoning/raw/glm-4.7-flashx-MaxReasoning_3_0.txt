use std::io::{self, BufRead, BufWriter, Write};
use std::collections::HashMap;
use std::f64::consts::PI;

const COOLING_FACTOR: f64 = 0.995;
const MAX_ITERATIONS: usize = 200;
const DAMPING: f64 = 0.85; // Velocity damping
const CENTERING_STRENGTH: f64 = 0.005;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse Header
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    // Parse Edges
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        
        if u != v {
            // Add to edges list
            edges.push((u, v));
            // Add to adjacency list
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    if n == 0 {
        return;
    }

    // Initialize Layout (Spiral)
    let mut positions: Vec<[f64; 2]> = vec![[0.0; 2]; n];
    let radius = 10.0 * (n as f64).sqrt();
    
    for i in 0..n {
        let angle = i as f64 * 0.5; // Arc length constant
        let r = radius * (angle / (2.0 * PI)).sqrt(); // Spiral growth
        positions[i][0] = r * angle.cos();
        positions[i][1] = r * angle.sin();
    }

    // Pre-compute Canvas Size for Optimal Spring Length (K)
    // We fix a virtual canvas size that adapts to N to prevent nodes from flying too far
    let canvas_size = 1000.0; 
    let k = canvas_size / ((n as f64).sqrt() / 2.0); // Optimal edge length
    let k_sq = k * k;

    let mut velocities: Vec<[f64; 2]> = vec![[0.0; 2]; n];

    // Grid parameters for Spatial Partitioning (Repulsion)
    // Grid size is slightly larger than k to optimize
    let grid_size = k * 1.5; 

    // Solver Loop
    for iter in 0..MAX_ITERATIONS {
        // Clear forces (reset to 0)
        // We use velocities as temporary storage for forces in this simple implementation
        // or maintain a separate forces vector. For clarity, we use a local force vector 
        // but parallelizing the accumulation is tricky without Mutexes or Atomics.
        // Since N can be large, sequential accumulation for forces is acceptable.
        
        let mut forces: Vec<[f64; 2]> = vec![[0.0; 2]; n];

        // 1. Repulsion (Coulomb's Law)
        // Optimization: Spatial Grid
        let mut grid: HashMap<(usize, usize), Vec<usize>> = HashMap::new();

        // Populate grid
        for i in 0..n {
            let x = positions[i][0];
            let y = positions[i][1];
            let gx = (x / grid_size) as usize;
            let gy = (y / grid_size) as usize;
            grid.entry((gx, gy)).or_insert_with(Vec::new).push(i);
        }

        // Compute repulsion
        for (gx, gy, cell_nodes) in grid.iter() {
            // Check all nodes in this cell
            for &i in cell_nodes {
                let fx = &mut forces[i];
                let pos_i = positions[i];
                
                // Check 9 surrounding cells (Moore neighborhood)
                for dx in -1i32..=1 {
                    for dy in -1i32..=1 {
                        let key = (gx as i32 + dx, gy as i32 + dy);
                        if let Some(neighbors) = grid.get(&key) {
                            for &j in neighbors {
                                if i == j { continue; }
                                
                                let dx = pos_i[0] - positions[j][0];
                                let dy = pos_i[1] - positions[j][1];
                                let dist_sq = dx*dx + dy*dy;
                                
                                // Clamp distance to avoid division by zero/infinity
                                let dist_sq = dist_sq.max(1.0); 
                                
                                // F = k^2 / dist
                                let force = k_sq / dist_sq;
                                
                                let dist = dist_sq.sqrt();
                                fx[0] += force * (dx / dist);
                                fx[1] += force * (dy / dist);
                            }
                        }
                    }
                }
            }
        }

        // 2. Attraction (Hooke's Law)
        // This is parallelizable since edges are sparse
        edges.par_iter().for_each(|(u, v)| {
            let pos_u = positions[*u];
            let pos_v = positions[*v];
            
            let dx = pos_v[0] - pos_u[0];
            let dy = pos_v[1] - pos_u[1];
            let dist_sq = dx*dx + dy*dy;
            
            // Clamp dist_sq for safety
            let dist = dist_sq.sqrt();
            if dist < 1e-5 { return; }

            // F = dist^2 / k
            let force = dist_sq / k;

            // Apply force to both nodes
            let fu = &mut forces[*u];
            let fv = &mut forces[*v];
            
            fu[0] -= force * (dx / dist);
            fu[1] -= force * (dy / dist);
            
            fv[0] += force * (dx / dist);
            fv[1] += force * (dy / dist);
        });

        // 3. Centering Force (Pull towards origin to prevent drift)
        for i in 0..n {
            let x = positions[i][0];
            let y = positions[i][1];
            forces[i][0] -= x * CENTERING_STRENGTH;
            forces[i][1] -= y * CENTERING_STRENGTH;
        }

        // 4. Integration (Update Positions)
        let current_temp = 1.0 - (iter as f64 / MAX_ITERATIONS as f64); // Temperature cooling
        let current_temp = current_temp.max(0.01);

        for i in 0..n {
            let vel = &mut velocities[i];
            let force = forces[i];

            // Update Velocity: v = (v + F) * friction
            vel[0] += force[0];
            vel[1] += force[1];
            
            vel[0] *= DAMPING;
            vel[1] *= DAMPING;

            // Move
            positions[i][0] += vel[0] * current_temp;
            positions[i][1] += vel[1] * current_temp;
        }
    }

    // Output
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    // Optional: Output N first
    writeln!(out, "{}", n).unwrap();

    for pos in positions {
        writeln!(out, "{} {}", pos[0], pos[1]).unwrap();
    }
}
