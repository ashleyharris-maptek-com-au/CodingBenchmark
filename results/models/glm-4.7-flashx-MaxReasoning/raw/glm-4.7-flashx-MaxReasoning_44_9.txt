```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize interpolated normal to ensure it's a unit vector
    vec3 n = normalize(normal);

    // UV Checkerboard at 8x8 frequency, alternating colors 0.9 and 0.2 by parity
    float scale = 8.0;
    // Use floor to get the integer index, sum them for parity (checkerboard pattern)
    float sum = floor(uv.x * scale) + floor(uv.y * scale);
    // If sum is even (0), use 0.9, if odd (1), use 0.2
    vec3 checkerboard = mix(vec3(0.9), vec3(0.2), mod(sum, 2.0) < 1.0);

    // Lambertian Shading
    // Light direction from surface to light
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    // Diffuse intensity (dot product)
    float diff = max(dot(n, lightDir), 0.0);
    // Apply factor (0.5 + 0.5 * diffuse)
    float shadingFactor = 0.5 + 0.5 * diff;

    // Combine checkerboard base with lighting factor
    vec3 finalColor = checkerboard * shadingFactor;

    outColor = vec4(finalColor, 1.0);
}
```
