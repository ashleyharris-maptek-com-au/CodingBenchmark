; SPIR-V
1.0.0.0
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%v4struct = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%_ptr_Uniform_v4struct = OpTypePointer Uniform %v4struct
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%func = OpTypeFunction %void

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1

OpDecorate %v4struct Block
OpMemberDecorate %v4struct 0 Offset 0
OpMemberDecorate %v4struct 1 Offset 64
OpMemberDecorate %v4struct 2 Offset 128
OpMemberDecorate %v4struct 3 Offset 192
OpMemberDecorate %v4struct 4 Offset 208
OpMemberDecorate %v4struct 5 Offset 224

%ubo = OpVariable %_ptr_Uniform_v4struct Uniform
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%tangent = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

%main = OpFunction %void None %func
%entry = OpLabel
%uv_load = OpLoad %v2float %uv
%x = OpCompositeExtract %v3float %uv_load 0
%y = OpCompositeExtract %v3float %uv_load 1
%z = OpCompositeConstruct %v3float %float_0 %float_0 %float_0
%vec_rgba = OpCompositeConstruct %v4float %x %y %z %float_1
OpStore %outColor %vec_rgba
OpReturn
OpFunctionEnd
