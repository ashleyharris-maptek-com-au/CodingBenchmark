```text
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %gl_out_0 %gl_out_1 %gl_out_2 %gl_out_3 %gl_out_4 %gl_out_5 %gl_out_6 %gl_out_7 %gl_out_8 %gl_out_9 %gl_out_10 %gl_out_11 %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

%float = OpTypeFloat 32
%void = OpTypeVoid
%func = OpTypeFunction %void
%int = OpTypeInt 32 1
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_6 = OpConstant %uint 6

; Input Struct
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input Arrays
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%gl_in = OpVariable %ptr_in_pv Input

%ptr_in_v3 = OpTypePointer Input %v3float
%gl_in_0 = OpVariable %ptr_in_v3 Input
OpDecorate %gl_in_0 Location 0

%gl_in_1 = OpVariable %ptr_in_v3 Input
OpDecorate %gl_in_1 Location 1

%gl_in_2 = OpVariable %ptr_in_v3 Input
OpDecorate %gl_in_2 Location 2

; Output Struct
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%uint_12 = OpConstant %uint 12
%arr_pv_out = OpTypeArray %gl_PerVertex_out %uint_12
%ptr_out_pv = OpTypePointer Output %arr_pv_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%gl_out_0 = OpVariable %ptr_out_v3 Output
OpDecorate %gl_out_0 Location 0

; UBO Layout (Model, View, Proj, Params)
%ubo_struct = OpTypeStruct %v4float %v4float %v4float %v4float
OpDecorate %ubo_struct Block
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 1 Offset 16
OpMemberDecorate %ubo_struct 2 Offset 32
OpMemberDecorate %ubo_struct 3 Offset 48
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo Uniform

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0_008 = OpConstant %float 0.008
%float_0_5 = OpConstant %float 0.5

%main = OpFunction %void None %func
%label = OpLabel

; Indices for main triangle vertices in adjacency (0, 2, 4)
%idx_a = OpLoad %int %int_0
%idx_b = OpLoad %int %int_2
%idx_c = OpLoad %int %int_4
%idx_prev = OpLoad %int %int_0
%idx_curr = OpLoad %int %int_2
%idx_next = OpLoad %int %int_4

; Get positions and colors
%p_a = OpAccessChain %ptr_in_pv %gl_in %idx_a %int_0
%pos_a = OpLoad %v4float %p_a
%p_b = OpAccessChain %ptr_in_pv %gl_in %idx_b %int_0
%pos_b = OpLoad %v4float %p_b
%p_c = OpAccessChain %ptr_in_pv %gl_in %idx_c %int_0
%pos_c = OpLoad %v4float %p_c

%c_a = OpAccessChain %ptr_in_v3 %gl_in_2 %idx_a
%col_a = OpLoad %v3float %c_a
%c_b = OpAccessChain %ptr_in_v3 %gl_in_2 %idx_b
%col_b = OpLoad %v3float %c_b
%c_c = OpAccessChain %ptr_in_v3 %gl_in_2 %idx_c
%col_c = OpLoad %v3float %c_c

; Calculate average colors
%avg_ab = OpFAdd %v3float %col_a %col_b
%avg_ab = OpFMul %v3float %avg_ab %float_0_5
%avg_bc = OpFAdd %v3float %col_b %col_c
%avg_bc = OpFMul %v3float %avg_bc %float_0_5
%avg_ca = OpFAdd %v3float %col_c %col_a
%avg_ca = OpFMul %v3float %avg_ca %float_0_5

; ---------------- EDGE 0: A -> B ----------------
; Direction
%d_ab = OpFSub %v4float %pos_b %pos_a
; Perpendicular in XY clip space
%perp_ab = OpVectorShuffle %v4float %d_ab %d_ab 1 0 2 3
%perp_ab = OpVectorTimesScalar %v4float %perp_ab %float_0_008
; Offset
%pos0_a = OpFAdd %v4float %pos_a %perp_ab
%pos0_b = OpFAdd %v4float %pos_b %perp_ab
%pos0_c = OpFSub %v4float %pos_b %perp_ab
%pos0_d = OpFSub %v4float %pos_a %perp_ab
; Store and emit
%store0 = OpAccessChain %ptr_out_pv %gl_out %int_0 %int_0
OpStore %store0 %pos0_a
%store1 = OpAccessChain %ptr_out_pv %gl_out %int_1 %int_0
OpStore %store1 %pos0_b
%store2 = OpAccessChain %ptr_out_pv %gl_out %int_2 %int_0
OpStore %store2 %pos0_c
%store3 = OpAccessChain %ptr_out_pv %gl_out %int_3 %int_0
OpStore %store3 %pos0_d
%storec0 = OpAccessChain %ptr_out_v3 %gl_out_0 %int_0
OpStore %storec0 %avg_ab
OpEmitVertex
OpEndPrimitive

; ---------------- EDGE 1: B -> C ----------------
%d_bc = OpFSub %v4float %pos_c %pos_b
%perp_bc = OpVectorShuffle %v4float %d_bc %d_bc 1 0 2 3
%perp_bc = OpVectorTimesScalar %v4float %perp_bc %float_0_008
%pos1_a = OpFAdd %v4float %pos_b %perp_bc
%pos1_b = OpFAdd %v4float %pos_c %perp_bc
%pos1_c = OpFSub %v4float %pos_c %perp_bc
%pos1_d = OpFSub %v4float %pos_b %perp_bc
%store4 = OpAccessChain %ptr_out_pv %gl_out %int_4 %int_0
OpStore %store4 %pos1_a
%store5 = OpAccessChain %ptr_out_pv %gl_out %int_5 %int_0
OpStore %store5 %pos1_b
%store6 = OpAccessChain %ptr_out_pv %gl_out %int_6 %int_0
OpStore %store6 %pos1_c
%store7 = OpAccessChain %ptr_out_pv %gl_out %int_7 %int_0
OpStore %store7 %pos1_d
%storec1 = OpAccessChain %ptr_out_v3 %gl_out_0 %int_4
OpStore %storec1 %avg_bc
OpEmitVertex
OpEndPrimitive

; ---------------- EDGE 2: C -> A ----------------
%d_ca = OpFSub %v4float %pos_a %pos_c
%perp_ca = OpVectorShuffle %v4float %d_ca %d_ca 1 0 2 3
%perp_ca = OpVectorTimesScalar %v4float %perp_ca %float_0_008
%pos2_a = OpFAdd %v4float %pos_c %perp_ca
%pos2_b = OpFAdd %v4float %pos_a %perp_ca
%pos2_c = OpFSub %v4float %pos_a %perp_ca
%pos2_d = OpFSub %v4float %pos_c %perp_ca
%store8 = OpAccessChain %ptr_out_pv %gl_out %int_8 %int_0
OpStore %store8 %pos2_a
%store9 = OpAccessChain %ptr_out_pv %gl_out %int_9 %int_0
OpStore %store9 %pos2_b
%store10 = OpAccessChain %ptr_out_pv %gl_out %int_10 %int_0
OpStore %store10 %pos2_c
%store11 = OpAccessChain %ptr_out_pv %gl_out %int_11 %int_0
OpStore %store11 %pos2_d
%storec2 = OpAccessChain %ptr_out_v3 %gl_out_0 %int_8
OpStore %storec2 %avg_ca
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
