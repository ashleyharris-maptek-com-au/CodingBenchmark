def main():
    # Setup recursion limit (not used, but good practice)

    # Parse and Solve
    # We'll do it inside main to capture return
    
    data = sys.stdin.buffer.read().split()
    if not data: 
        return
    it = iter(data)
    try:
        N = int(next(it))
        M = int(next(it))

    neg_watches = [array.array('I') for _ in range(MAX_LIT + 2)] 
    
    sat_counts = array.array('B', [0]) * M 
    
    for cid in range(M):
        l1 = clauses_data[3 * cid]

        l3 = clauses_data[3 * cid + 2]
        
        if l1 > 0: pos_watches[l1].append(cid)
        else: neg_watches[-l1].append(cid)
        
        if l2 > 0: pos_watches[l2].append(cid)
        else: neg_watches[-l2].append(cid)

    assignment = bytearray(N + 1)
    unit_queue = deque()
    decision_stack = [] # (var, val)
    
    def assign_var(lit):
        if lit < 0:
            var = -lit
            val = False
            watch_list = neg_watches[-lit]
        else:
            var = lit
            val = True
            watch_list = pos_watches[lit]
            
        assignment[var] = val
        
        for cid in watch_list:
            sc = sat_counts[cid]
            if sc >= 2: continue
            
            c0 = clauses_data[3*cid]
            c1 = clauses_data[3*cid+1]
            c2 = clauses_data[3*cid+2]
            
            # We just set `lit` to True.
            # We need to count how many others are True.
            # If exactly one other is True -> Unit.
            # If two others are True -> Satisfied.
            
            true_cnt = 1
            
            # Check c0
            if c0 > 0 and assignment[c0]: true_cnt += 1
            elif c0 < 0 and not assignment[-c0]:
                unit_queue.append(c0)
            # else unassigned or false, continue
            
            if true_cnt >= 2:
                sat_counts[cid] = true_cnt
                continue
            
            # Check c1
            if c1 > 0 and assignment[c1]: true_cnt += 1
            elif c1 < 0 and not assignment[-c1]:
                unit_queue.append(c1)
            
            if true_cnt >= 2:
                sat_counts[cid] = true_cnt
                continue
                
            # Check c2
            if c2 > 0 and assignment[c2]: true_cnt += 1
            elif c2 < 0 and not assignment[-c2]:
                unit_queue.append(c2)
            
            if true_cnt >= 2:
                sat_counts[cid] = true_cnt
            else:
                sat_counts[cid] = true_cnt # set to 1 if unit, 0 otherwise

    def propagate():
        # Returns True if conflict found
        # Conflict: unit_queue contains a literal that is already assigned False
        # (handled in assign_var when we see a false literal in unit_queue)
        # But we need to propagate through the queue.
        
        # If unit_queue contains a literal that is already False (due to previous branch), conflict.

        # If we see c0 < 0 and not assignment[-c0], we add to queue.

            lit = unit_queue.popleft()
            # Check if lit is satisfied
            if lit > 0 and assignment[lit]: continue
            if lit < 0 and not assignment[-lit]: 
                # Conflicting assignment: trying to set negative literal to False? 
                # Or trying to set negative literal to True?
                # If lit < 0, it means "variable var must be False".
                # If we check `not assignment[-lit]`, that means assignment[var] is False.

                return True
            if lit > 0 and not assignment[lit]:
                # We are trying to set var to True, but var is False. Conflict.
