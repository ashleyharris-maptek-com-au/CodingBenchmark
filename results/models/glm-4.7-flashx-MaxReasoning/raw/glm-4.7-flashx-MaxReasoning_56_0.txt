const fs = require('fs');
const input = JSON.parse(fs.readFileSync(0, 'utf8'));

// Helper: Euclidean distance squared (avoids sqrt for comparisons)
const distSq = (a, b) => (a.x - b.x) ** 2 + (a.y - b.y) ** 2;

// Helper: Calculate distance between two points
const dist = (a, b) => Math.sqrt(distSq(a, b));

// Helper: Calculate route length
const routeLength = (route, depot) => {
    let sum = 0;
    for (let i = 0; i < route.length; i++) {
        const current = i === 0 ? depot : route[i - 1];
        const next = route[i];
        sum += dist(current, next);
    }
    // Return to depot
    sum += dist(route[route.length - 1], depot);
    return sum;
};

// Helper: Calculate insertion cost for adding package p to route r at index i
// Route r is [p0, p1, ...]
// Insert p between p_prev and p_next
const insertionCost = (route, depot, p, i) => {
    let cost = 0;
    if (i === 0) {
        // Add to start (after depot)
        cost += dist(depot, p);
        if (route.length > 0) cost += dist(p, route[0]);
        else cost += dist(p, depot);
    } else {
        const pPrev = route[i - 1];
        const pNext = route[i] || depot; // If i is last index, next is depot
        cost += dist(pPrev, p);
        cost += dist(p, pNext);
    }
    return cost;
};

// Helper: Nearest Neighbor + 2-Opt for a single route
const solveTSP = (packages, depot) => {
    if (packages.length === 0) return [];
    
    // 1. Nearest Neighbor construction
    let route = [];
    let unvisited = [...packages];
    
    // Start from depot, pick nearest
    let current = depot;
    
    // Find first package
    let nearest = null;
    let minD = Infinity;
    let minIdx = -1;

    for(let i=0; i<unvisited.length; i++) {
        const d = distSq(unvisited[i], current);
        if(d < minD) {
            minD = d;
            nearest = unvisited[i];
            minIdx = i;
        }
    }
    if (nearest) {
        route.push(nearest);
        unvisited.splice(minIdx, 1);
        current = nearest;
    }

    while(unvisited.length > 0) {
        nearest = null;
        minD = Infinity;
        minIdx = -1;

        for(let i=0; i<unvisited.length; i++) {
            const d = distSq(unvisited[i], current);
            if(d < minD) {
                minD = d;
                nearest = unvisited[i];
                minIdx = i;
            }
        }
        if (nearest) {
            route.push(nearest);
            unvisited.splice(minIdx, 1);
            current = nearest;
        }
    }

    // 2. 2-Opt Local Optimization
    // Basic 2-opt swap
    let improved = true;
    while (improved) {
        improved = false;
        const n = route.length;
        // Optimization: Check edge case
        if (n <= 2) break;

        // We iterate. A full double-loop is O(N^2), but N is usually small per route (<100).
        for (let i = 0; i < n - 1; i++) {
            const pA = route[i];
            const pB = route[i + 1];

            for (let j = i + 2; j < n; j++) {
                // Standard 2-opt: reversing segment [i+1 ... j]
                // We need to calculate delta distance
                const pC = route[j];
                const pD = route[(j + 1) % n];

                // Current edge weights:
                // A-B, C-D
                // Proposed weights:
                // A-C, B-D
                
                const oldDist = distSq(pA, pB) + distSq(pC, pD);
                const newDist = distSq(pA, pC) + distSq(pB, pD);

                if (newDist < oldDist) {
                    // Perform swap
                    // Reverse segment between i+1 and j
                    // Copy route to avoid mutation issues in complex loops (though not strictly needed here)
                    const newRoute = [...route];
                    
                    // Reverse indices i+1 to j
                    const start = i + 1;
                    const end = j;
                    while (start < end) {
                        newRoute[start] = route[end];
                        newRoute[end] = route[start];
                        start++;
                        end--;
                    }
                    
                    route = newRoute;
                    improved = true;
                    
                    // Break inner loop if improved to restart and catch new improvements
                    break; 
                }
            }
            if (improved) break;
        }
    }

    return route;
};

const { depot, truckCount, packages } = input;
const numPackages = packages.length;

// If we have enough trucks for everyone, one route is optimal
if (truckCount >= numPackages) {
    const route = solveTSP(packages, depot);
    const result = { routes: [ route.map(p => p.id) ] };
    console.log(JSON.stringify(result));
} else {
    // 1. Sort packages by distance from depot (furthest first helps balance load)
    // We sort by squared distance to avoid sqrt
    const sortedPackages = [...packages].sort((a, b) => distSq(b, depot) - distSq(a, depot));
    
    // 2. Greedy Assignment to Trucks
    const routes = Array.from({ length: truckCount }, () => []);
    
    // We iterate through packages
    for (const pkg of sortedPackages) {
        let bestRouteIdx = -1;
        let bestInsertCost = Infinity;
        let bestInsertIdx = -1;

        // Check insertion in existing routes
        for (let r = 0; r < truckCount; r++) {
            const currentRoute = routes[r];
            // Insert at end is usually best for initial greedy if we don't optimize insertion position
            // But let's try to find the exact best position to minimize cost increase
            
            // Optimization: Only check first and last for large routes? 
            // For this scale, linear search is fine.
            for (let i = 0; i <= currentRoute.length; i++) {
                const cost = insertionCost(currentRoute, depot, pkg, i);
                if (cost < bestInsertCost) {
                    bestInsertCost = cost;
                    bestRouteIdx = r;
                    bestInsertIdx = i;
                }
            }
        }

        if (bestRouteIdx !== -1) {
            routes[bestRouteIdx].splice(bestInsertIdx, 0, pkg);
        } else {
            // Fallback (should not happen)
            routes[0].push(pkg);
        }
    }

    // 3. Optimize each route (TSP)
    const optimizedRoutes = routes.map(route => solveTSP(route, depot));

    const result = { routes: optimizedRoutes.map(r => r.map(p => p.id)) };
    console.log(JSON.stringify(result));
}
