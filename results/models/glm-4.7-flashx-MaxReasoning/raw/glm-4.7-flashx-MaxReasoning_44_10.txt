```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 1. Render horizontal stripes
    float stripeFactor = sin(uv.y * 3.14159 * 16.0);
    
    // Alternating red (0.9, 0.1, 0.1) and blue (0.1, 0.1, 0.9)
    vec3 baseColor = mix(vec3(0.1, 0.1, 0.9), vec3(0.9, 0.1, 0.1), step(0.0, stripeFactor));

    // 2. Lambertian Shading
    vec3 norm = normalize(normal);
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(norm, lightDir), 0.0);

    // 3. Apply factor (0.4 + 0.6 * diffuse)
    float shadingFactor = 0.4 + 0.6 * diffuse;

    // 4. Output
    outColor = vec4(baseColor * shadingFactor, 1.0);
}
```
