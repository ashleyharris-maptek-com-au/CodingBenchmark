; SPIR-V
OpCapability Shader
OpExtension "SPV_KHR_storage_buffer_storage_class"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float 4
%struct_ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_uniform_struct_ubo = OpTypePointer Uniform %struct_ubo
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_input_v2float = OpTypePointer Input %v2float
%ptr_output_v4float = OpTypePointer Output %v4float
%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%label = OpLabel

%one = OpConstant %float 1
%zero = OpConstant %float 0
%vec4_red = OpConstantComposite %v4float %one %zero %zero %one

%ubo = OpVariable %ptr_uniform_struct_ubo Uniform
%in_worldPos = OpVariable %ptr_input_v3float Input
%in_normal = OpVariable %ptr_input_v3float Input
%in_uv = OpVariable %ptr_input_v2float Input
%in_tangent = OpVariable %ptr_input_v3float Input
%in_color = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output

OpDecorate %struct_ubo Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %struct_ubo 0 Offset 0
OpMemberDecorate %struct_ubo 0 MatrixStride 16 ColMajor
OpMemberDecorate %struct_ubo 1 Offset 64
OpMemberDecorate %struct_ubo 1 MatrixStride 16 ColMajor
OpMemberDecorate %struct_ubo 2 Offset 128
OpMemberDecorate %struct_ubo 2 MatrixStride 16 ColMajor
OpMemberDecorate %struct_ubo 3 Offset 192
OpMemberDecorate %struct_ubo 4 Offset 208
OpMemberDecorate %struct_ubo 5 Offset 224

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpStore %outColor %vec4_red
OpReturn
OpFunctionEnd
