; SPIR-V
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %out_outColor Location 0
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %uniform_ubo DescriptorSet 0
OpDecorate %uniform_ubo Binding 0
OpDecorate %ubo_struct Block
OpDecorate %ubo_model Offset 0
OpDecorate %ubo_view Offset 64
OpDecorate %ubo_proj Offset 128
OpDecorate %ubo_light Offset 192
OpDecorate %ubo_cam Offset 208
OpDecorate %ubo_params Offset 224

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float 4
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_v3float = OpTypePointer Input %v3float
%ptr_v2float = OpTypePointer Input %v2float
%ptr_v3float_2 = OpTypePointer Input %v3float
%ptr_v4float = OpTypePointer Input %v4float
%ptr_v4float_out = OpTypePointer Output %v4float
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%GLSL450 = OpExtInstImport "GLSL.std.450"
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%float_0_8 = OpConstant %float 0.8
%float_0_2 = OpConstant %float 0.2
%float_1_0 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_1_5 = OpConstant %float 1.5
%float_3_0 = OpConstant %float 3.0
%float_0_0 = OpConstant %float 0.0
%float_0_7_fog = OpConstant %float 0.7
%float_0_8_fog = OpConstant %float 0.8

%in_worldPos = OpVariable %ptr_v3float Input
%in_normal = OpVariable %ptr_v3float Input
%in_uv = OpVariable %ptr_v2float Input
%in_tangent = OpVariable %ptr_v3float_2 Input
%in_color = OpVariable %ptr_v3float_2 Input
%out_outColor = OpVariable %ptr_v4float_out Output
%uniform_ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %func
%label = OpLabel

%worldPos = OpLoad %v3float %in_worldPos
%normal = OpLoad %v3float %in_normal
%normal_norm = OpNormalize %v3float %normal

%ubo_light_ptr = OpAccessChain %ptr_v4float %uniform_ubo %uint_0 %uint_3
%ubo_light_vec = OpLoad %v4float %ubo_light_ptr
%lightPos = OpCompositeExtract %v3float %ubo_light_vec 0

%ubo_cam_ptr = OpAccessChain %ptr_v4float %uniform_ubo %uint_0 %uint_4
%ubo_cam_vec = OpLoad %v4float %ubo_cam_ptr
%camPos = OpCompositeExtract %v3float %ubo_cam_vec 0

%lightDir = OpFSub %v3float %lightPos %worldPos
%lightDirNorm = OpNormalize %v3float %lightDir
%diffuse = OpDot %float %normal_norm %lightDirNorm
%diffuse_clamped = OpExtInst %float %GLSL450 FClamp %diffuse %uint_0 %uint_0
%lambertianFactor = OpFAdd %float %float_0_3 (OpFMul %float %float_0_7 %diffuse_clamped)

%dist = OpLength %v3float (OpFSub %v3float %worldPos %camPos)
%fogFactor = OpFDiv %float (OpFSub %float %dist %float_1_5) %float_3_0
%fogFactorClamped = OpExtInst %float %GLSL450 FClamp %fogFactor %float_0_0 %float_1_0

%baseColor = OpCompositeConstruct %v4float %float_0_8 %float_0_2 %float_0_2 %float_1_0
%fogColor = OpCompositeConstruct %v4float %float_0_7_fog %float_0_7_fog %float_0_8_fog %float_1_0
%fogFactorInv = OpFSub %float %float_1_0 %fogFactorClamped

%finalColor = OpFAdd %v4float (OpFMul %v4float %baseColor %fogFactorInv) (OpFMul %v4float %fogColor %fogFactorClamped)

OpStore %out_outColor %finalColor
OpReturn
OpFunctionEnd
