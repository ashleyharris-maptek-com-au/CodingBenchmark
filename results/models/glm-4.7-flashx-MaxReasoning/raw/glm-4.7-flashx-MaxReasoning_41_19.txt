; SPIR-V 1.6
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%mat4 = OpTypeMatrix %float 4
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%uint = OpTypeInt 32 0
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Input_v3float_1 = OpTypePointer Input %v3float
%_ptr_Input_v3float_2 = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%main = OpFunction %void None %void
%main_body = OpLabel
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%tangent = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%float_0_1 = OpConstant %float 0.1
%float_0_05 = OpConstant %float 0.05
%float_32 = OpConstant %float 32
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%const_ambient = OpConstantComposite %v3float %float_0_1 %float_0_05 %float_0_1
%const_specular_color = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1
%const_vec3_zero = OpConstantComposite %v3float %float_0 %float_0 %float_0
%const_vec4_zero = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0
%const_float4_one = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5

OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %uv %tangent %color
OpExecutionMode %main OriginUpperLeft

%ubolight = OpAccessChain %_ptr_Uniform_v4float %ubo %uint_3
%ubocam = OpAccessChain %_ptr_Uniform_v4float %ubo %uint_4
%uboparams = OpAccessChain %_ptr_Uniform_v4float %ubo %uint_5
%lightPos = OpLoad %v4float %ubolight
%camPos = OpLoad %v4float %ubocam
%params = OpLoad %v4float %uboparams

%worldPosVal = OpLoad %v3float %worldPos
%normalVal = OpLoad %v3float %normal
%colorVal = OpLoad %v3float %color

%normalDir = OpNormalize %v3float %normalVal
%L = OpFSub %v3float %lightPos %worldPosVal
%viewDir = OpFSub %v3float %camPos %worldPosVal
%viewDir = OpNormalize %v3float %viewDir

%diff = OpDot %v3float %normalDir %L
%diff = OpFSub %float %diff %float_0
%diffColor = OpFMul %v3float %diff %colorVal

%incident = OpFSub %v3float %worldPosVal %lightPos
%reflectDir = OpReflect %v3float %incident %normalDir
%spec = OpDot %v3float %reflectDir %viewDir
%spec = OpFSub %float %spec %float_0
%spec = OpFMul %float %spec %float_32

%specColor = OpFMul %v3float %spec %const_specular_color

%finalColor = OpFAdd %v3float %const_ambient %diffColor
%finalColor = OpFAdd %v3float %finalColor %specColor
%finalColor = OpExtInst %v3float %Glsl450 64 %finalColor %const_vec3_zero %const_vec3_zero

%finalVec4 = OpCompositeConstruct %v4float %finalColor %float_1
OpStore %outColor %finalVec4
OpReturn
OpFunctionEnd
