import sys
import math

def solve() -> None:
    # Read all input from stdin
    data = sys.stdin.read().strip().split()
    if not data:
        return

    it = iter(data)
    try:
        N = int(next(it))
        K = int(next(it))
        CAPACITY = int(next(it))
    except StopIteration:
        return

    # Read Depot coordinates
    try:
        depot_x = float(next(it))
        depot_y = float(next(it))
    except StopIteration:
        return

    # Read Customers
    customers = []
    for _ in range(N):
        try:
            x = float(next(it))
            y = float(next(it))
            d = int(next(it))
            customers.append((x, y, d))
        except StopIteration:
            break

    # Data Structures
    # Node 0: Depot
    # Node 1..N: Customers
    n_nodes = N + 1
    demand = [0] * n_nodes
    for i in range(1, n_nodes):
        demand[i] = customers[i-1][2]

    # Distance Matrix
    dist = [[0.0] * n_nodes for _ in range(n_nodes)]
    for i in range(n_nodes):
        for j in range(n_nodes):
            if i == j:
                dist[i][j] = 0.0
                continue
            
            # Get coordinates
            if i == 0:
                xi, yi = depot_x, depot_y
            else:
                xi, yi = customers[i-1][0], customers[i-1][1]
            
            if j == 0:
                xj, yj = depot_x, depot_y
            else:
                xj, yj = customers[j-1][0], customers[j-1][1]
                
            dist[i][j] = math.hypot(xi - xj, yi - yj)

    # --- Clark-Wright Savings Algorithm ---
    
    # Initial State: Each customer is a separate route
    # routes: list of lists of node indices (1..N)
    routes = []
    route_demand = []
    route_head = [] # First customer in the route
    route_tail = [] # Last customer in the route
    node_to_route = {} # Map node -> route_id

    for i in range(1, n_nodes):
        rid = len(routes)
        routes.append([i]) # [node_i]
        route_demand.append(demand[i])
        route_head.append(i)
        route_tail.append(i)
        node_to_route[i] = rid

    # Calculate Savings
    # Savings[i][j] = d(0,i) + d(0,j) - d(i,j)
    # We store them in a list of tuples to sort
    savings_pairs = []
    for i in range(1, n_nodes):
        for j in range(i + 1, n_nodes):
            s = dist[0][i] + dist[0][j] - dist[i][j]
            # Store negative for max-heap behavior or sort reverse
            savings_pairs.append((-s, i, j))
    
    savings_pairs.sort() # Sort by highest savings (most negative first)

    # Greedy Merge
    for neg_s, i, j in savings_pairs:
        # Optimization: Stop if we reach the vehicle limit
        if len(routes) <= K:
            break

        r1 = node_to_route.get(i)
        r2 = node_to_route.get(j)

        if r1 is None or r2 is None or r1 == r2:
            continue

        # Try merging Route r1 into Route r2
        # Case 1: r1 ends with i, r2 starts with j
        if route_tail[r1] == i and route_head[r2] == j:
            if route_demand[r1] + route_demand[r2] <= CAPACITY:
                _merge_routes(r1, r2, routes, route_demand, route_head, route_tail, node_to_route)
                continue

        # Case 2: r2 ends with j, r1 starts with i
        if route_tail[r2] == j and route_head[r1] == i:
            if route_demand[r1] + route_demand[r2] <= CAPACITY:
                _merge_routes(r2, r1, routes, route_demand, route_head, route_tail, node_to_route)
                continue

    # --- Output Preparation ---
    total_distance = 0.0
    output_routes = []

    for nodes in routes:
        if not nodes:
            continue
        
        # Calculate distance: 0 -> nodes[0] -> ... -> nodes[-1] -> 0
        d = dist[0][nodes[0]]
        for k in range(len(nodes) - 1):
            d += dist[nodes[k]][nodes[k+1]]
        d += dist[nodes[-1]][0]
        total_distance += d

        output_routes.append(nodes)

    # Output
    # Printing total distance
    print(total_distance)
    # Printing number of routes
    print(len(routes))
    # Printing routes
    # Nodes are integers 1..N
    for r in output_routes:
        print(" ".join(map(str, r)))

def _merge_routes(r_src, r_dst, routes, rd, rh, rt, n2r):
    """
    Merges route r_src into r_dst.
    r_src is consumed and removed.
    """
    # Combine node lists
    merged_nodes = routes[r_src] + routes[r_dst]
    routes[r_dst] = merged_nodes
    rd[r_dst] = rd[r_src] + rd[r_dst]

    # Update node_to_route mapping
    # Remove old mappings for r_dst nodes, add new ones
    for n in routes[r_dst]:
        del n2r[n]
    for n in merged_nodes:
        n2r[n] = r_dst

    # Update head and tail
    # Head is the head of the source (r_src)
    rh[r_dst] = rh[r_src]
    # Tail is the tail of the destination (r_dst)
    rt[r_dst] = rt[r_dst]

    # Remove r_src from the end of the lists
    routes.pop()
    rd.pop()
    rh.pop()
    rt.pop()
    
    # Update mappings for the moved route (last element)
    # Indices shifted by 1.
    # Let the new route index be r_src (the old r_dst position)
    moved_rid = len(routes) # The index where the popped route went
    
    for n in routes[moved_rid]:
        n2r[n] = moved_rid

if __name__ == "__main__":
    solve()
