using System;
using System.IO;
using System.Collections.Generic;

public static class Program
{
    private struct Edge
    {
        public int U;
        public int V;
        public int W;

        public Edge(int u, int v, int w)
        {
            U = u;
            V = v;
            W = w;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }

            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }

            return val * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        if (n == int.MinValue)
        {
            return;
        }

        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        if (m < 0)
        {
            Console.WriteLine("0");
            return;
        }

        var edges = new Edge[m];
        var degree = new int[n];

        int baseArcCap = Math.Max(1, 2 * m);
        var head = new int[n];
        Array.Fill(head, -1);
        var to = new int[baseArcCap];
        var next = new int[baseArcCap];
        var arcEdgeId = new int[baseArcCap];
        int arcPtr = 0;

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();

            if ((uint)u >= (uint)n || (uint)v >= (uint)n)
            {
                Console.WriteLine("0");
                return;
            }

            edges[i] = new Edge(u, v, w);
            degree[u]++;
            degree[v]++;

            to[arcPtr] = v;
            arcEdgeId[arcPtr] = i;
            next[arcPtr] = head[u];
            head[u] = arcPtr++;

            to[arcPtr] = u;
            arcEdgeId[arcPtr] = i;
            next[arcPtr] = head[v];
            head[v] = arcPtr++;
        }

        if (m == 0)
        {
            Console.WriteLine("0");
            return;
        }

        // DFS from node 0 to verify all edges are reachable and build a spanning tree.
        var visited = new bool[n];
        var parent = new int[n];
        var parentEdge = new int[n];
        Array.Fill(parent, -1);
        Array.Fill(parentEdge, -1);

        var dfsStack = new int[n];
        var order = new int[n];
        int sp = 0;
        int ordCount = 0;

        visited[0] = true;
        dfsStack[sp++] = 0;

        while (sp > 0)
        {
            int v = dfsStack[--sp];
            order[ordCount++] = v;

            for (int e = head[v]; e != -1; e = next[e])
            {
                int u = to[e];
                if (visited[u]) continue;

                visited[u] = true;
                parent[u] = v;
                parentEdge[u] = arcEdgeId[e];
                dfsStack[sp++] = u;
            }
        }

        for (int i = 0; i < m; i++)
        {
            if (!visited[edges[i].U] || !visited[edges[i].V])
            {
                // No feasible closed walk from node 0 can cover all edges.
                Console.WriteLine("0");
                return;
            }
        }

        // Eulerize by duplicating selected spanning-tree edges based on parity.
        var extraCountOnEdge = new int[m];
        var odd = new int[n];
        for (int i = 0; i < n; i++)
        {
            odd[i] = degree[i] & 1;
        }

        for (int i = ordCount - 1; i >= 1; i--)
        {
            int v = order[i];
            if (odd[v] == 0) continue;

            int pe = parentEdge[v];
            if (pe < 0)
            {
                Console.WriteLine("0");
                return;
            }

            extraCountOnEdge[pe]++;
            odd[v] ^= 1;
            odd[parent[v]] ^= 1;
        }

        if (odd[0] != 0)
        {
            Console.WriteLine("0");
            return;
        }

        long extraTotalLong = 0;
        for (int i = 0; i < m; i++)
        {
            extraTotalLong += extraCountOnEdge[i];
        }

        long totalEdgesLong = (long)m + extraTotalLong;
        if (totalEdgesLong > int.MaxValue)
        {
            Console.WriteLine("0");
            return;
        }

        int totalEdges = (int)totalEdgesLong;

        var mu = new int[totalEdges];
        var mv = new int[totalEdges];
        int p = 0;

        for (int i = 0; i < m; i++)
        {
            mu[p] = edges[i].U;
            mv[p] = edges[i].V;
            p++;
        }

        for (int i = 0; i < m; i++)
        {
            int add = extraCountOnEdge[i];
            for (int c = 0; c < add; c++)
            {
                mu[p] = edges[i].U;
                mv[p] = edges[i].V;
                p++;
            }
        }

        int arcCap = Math.Max(1, 2 * totalEdges);
        var eHead = new int[n];
        Array.Fill(eHead, -1);
        var eTo = new int[arcCap];
        var eNext = new int[arcCap];
        var eId = new int[arcCap];
        int ep = 0;

        for (int i = 0; i < totalEdges; i++)
        {
            int a = mu[i];
            int b = mv[i];

            eTo[ep] = b;
            eId[ep] = i;
            eNext[ep] = eHead[a];
            eHead[a] = ep++;

            eTo[ep] = a;
            eId[ep] = i;
            eNext[ep] = eHead[b];
            eHead[b] = ep++;
        }

        // Hierholzer on Eulerian multigraph.
        var iter = new int[n];
        Array.Copy(eHead, iter, n);

        var used = new bool[totalEdges];
        var pathStack = new int[totalEdges + 1];
        var circuit = new int[totalEdges + 1];
        int pathSp = 0;
        int circuitLen = 0;

        pathStack[pathSp++] = 0;

        while (pathSp > 0)
        {
            int v = pathStack[pathSp - 1];
            int e = iter[v];

            while (e != -1 && used[eId[e]])
            {
                e = eNext[e];
            }

            iter[v] = e;

            if (e == -1)
            {
                circuit[circuitLen++] = v;
                pathSp--;
            }
            else
            {
                int id = eId[e];
                used[id] = true;
                iter[v] = eNext[e];
                pathStack[pathSp++] = eTo[e];
            }
        }

        if (circuitLen != totalEdges + 1)
        {
            Console.WriteLine("0");
            return;
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput());
        for (int i = circuitLen - 1; i >= 0; i--)
        {
            if (i != circuitLen - 1) writer.Write(' ');
            writer.Write(circuit[i]);
        }
        writer.WriteLine();
    }
}
