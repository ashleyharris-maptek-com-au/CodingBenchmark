#include <algorithm>
#include <array>
#include <cstdint>
#include <iostream>
#include <limits>
#include <set>
#include <unordered_set>
#include <utility>
#include <vector>

struct Item {
    int id;
    int w;
    int h;
    long long area;
    int maxSide;
    int minSide;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rot = 0;
};

struct PlanResult {
    bool feasible = true;
    int boardsUsed = 0;
    std::vector<Placement> placement;
};

struct MaxSegTree {
    int size = 0;
    int n = 1;
    std::vector<int> tree;

    void init(int size_) {
        size = size_;
        n = 1;
        while (n < size) n <<= 1;
        tree.assign(2 * n, -1);
    }

    void setValue(int idx, int val) {
        int p = idx + n;
        tree[p] = val;
        p >>= 1;
        while (p > 0) {
            tree[p] = std::max(tree[p << 1], tree[(p << 1) | 1]);
            p >>= 1;
        }
    }

    int firstAtLeastRec(int p, int l, int r, int ql, int need) const {
        if (r < ql || tree[p] < need) return -1;
        if (l == r) return l;
        int m = (l + r) >> 1;
        int left = firstAtLeastRec(p << 1, l, m, ql, need);
        if (left != -1) return left;
        return firstAtLeastRec((p << 1) | 1, m + 1, r, ql, need);
    }

    int firstAtLeast(int ql, int need) const {
        if (ql < 0) ql = 0;
        if (ql >= size) return -1;
        return firstAtLeastRec(1, 0, n - 1, ql, need);
    }
};

struct Shelf {
    int board = -1;
    int y = 0;
    int h = 0;
    int xUsed = 0;
    int rem = 0;
};

struct Board {
    int usedH = 0;
    int remH = 0;
};

struct Cand {
    bool valid = false;
    int type = 3;   // 0: existing shelf, 1: new shelf in existing board, 2: new board
    int board = -1;
    int shelf = -1;
    int w = 0;
    int h = 0;
    int rot = 0;
    long long k1 = 0;
    long long k2 = 0;
    long long k3 = 0;
};

static bool betterCand(const Cand& a, const Cand& b) {
    if (!a.valid) return false;
    if (!b.valid) return true;
    if (a.type != b.type) return a.type < b.type;
    if (a.k1 != b.k1) return a.k1 < b.k1;
    if (a.k2 != b.k2) return a.k2 < b.k2;
    if (a.k3 != b.k3) return a.k3 < b.k3;
    if (a.rot != b.rot) return a.rot < b.rot;
    if (a.board != b.board) return a.board < b.board;
    return a.shelf < b.shelf;
}

PlanResult packWithShelfBestFit(const std::vector<Item>& itemsOrdered, int N, int BW, int BH) {
    PlanResult result;
    result.feasible = true;
    result.placement.assign(N, Placement{});

    std::vector<Shelf> shelves;
    shelves.reserve(N);
    std::vector<Board> boards;
    boards.reserve(std::max(8, N / 8));

    std::vector<std::multiset<std::pair<int, int>>> shelfBuckets(BH + 1);
    MaxSegTree shelfTree;
    shelfTree.init(BH + 1);

    std::vector<std::unordered_set<int>> boardRemBuckets(BH + 1);
    std::set<int> boardRemKeys;

    auto refreshShelfHeight = [&](int h) {
        if (h < 0 || h > BH) return;
        if (shelfBuckets[h].empty()) {
            shelfTree.setValue(h, -1);
        } else {
            shelfTree.setValue(h, shelfBuckets[h].rbegin()->first);
        }
    };

    auto addBoard = [&]() -> int {
        int id = static_cast<int>(boards.size());
        boards.push_back(Board{0, BH});
        boardRemBuckets[BH].insert(id);
        boardRemKeys.insert(BH);
        return id;
    };

    auto moveBoardRem = [&](int boardId, int oldRem, int newRem) {
        if (oldRem == newRem) return;
        auto& oldBucket = boardRemBuckets[oldRem];
        oldBucket.erase(boardId);
        if (oldBucket.empty()) boardRemKeys.erase(oldRem);

        boardRemBuckets[newRem].insert(boardId);
        boardRemKeys.insert(newRem);
    };

    auto evalCandidate = [&](int w, int h, int rot) {
        Cand c;
        if (w > BW || h > BH) return c;

        int shelfH = shelfTree.firstAtLeast(h, w);
        if (shelfH != -1) {
            auto& bucket = shelfBuckets[shelfH];
            auto it = bucket.lower_bound({w, -1});
            if (it != bucket.end()) {
                c.valid = true;
                c.type = 0;
                c.shelf = it->second;
                c.board = shelves[c.shelf].board;
                c.w = w;
                c.h = h;
                c.rot = rot;
                c.k1 = static_cast<long long>(it->first - w);
                c.k2 = static_cast<long long>(shelfH - h);
                c.k3 = c.board;
                return c;
            }
        }

        auto bit = boardRemKeys.lower_bound(h);
        if (bit != boardRemKeys.end()) {
            int remH = *bit;
            int board = *boardRemBuckets[remH].begin();
            c.valid = true;
            c.type = 1;
            c.board = board;
            c.w = w;
            c.h = h;
            c.rot = rot;
            c.k1 = static_cast<long long>(remH - h);
            c.k2 = static_cast<long long>(BW - w);
            c.k3 = board;
            return c;
        }

        c.valid = true;
        c.type = 2;
        c.w = w;
        c.h = h;
        c.rot = rot;
        c.k1 = static_cast<long long>(BH - h);
        c.k2 = static_cast<long long>(BW - w);
        c.k3 = 0;
        return c;
    };

    for (const Item& it : itemsOrdered) {
        Cand best;
        Cand c0 = evalCandidate(it.w, it.h, 0);
        if (betterCand(c0, best)) best = c0;
        if (it.w != it.h) {
            Cand c1 = evalCandidate(it.h, it.w, 1);
            if (betterCand(c1, best)) best = c1;
        }

        if (!best.valid) {
            result.feasible = false;
            break;
        }

        if (best.type == 0) {
            Shelf& s = shelves[best.shelf];
            auto& bucket = shelfBuckets[s.h];
            auto eraseIt = bucket.find({s.rem, best.shelf});
            if (eraseIt != bucket.end()) bucket.erase(eraseIt);
            refreshShelfHeight(s.h);

            int x = s.xUsed;
            int y = s.y;
            s.xUsed += best.w;
            s.rem -= best.w;

            bucket.insert({s.rem, best.shelf});
            refreshShelfHeight(s.h);

            result.placement[it.id] = Placement{s.board, x, y, best.rot};
        } else {
            int boardId;
            if (best.type == 1) {
                boardId = best.board;
            } else {
                boardId = addBoard();
            }

            Board& b = boards[boardId];
            int oldRem = b.remH;
            int y = b.usedH;
            b.usedH += best.h;
            b.remH -= best.h;
            moveBoardRem(boardId, oldRem, b.remH);

            int shelfId = static_cast<int>(shelves.size());
            shelves.push_back(Shelf{boardId, y, best.h, best.w, BW - best.w});
            shelfBuckets[best.h].insert({BW - best.w, shelfId});
            refreshShelfHeight(best.h);

            result.placement[it.id] = Placement{boardId, 0, y, best.rot};
        }
    }

    if (result.feasible) {
        result.boardsUsed = static_cast<int>(boards.size());
    }

    return result;
}

std::vector<Item> makeOrder(const std::vector<Item>& items, int mode) {
    std::vector<Item> v = items;
    if (mode == 0) {
        std::sort(v.begin(), v.end(), [](const Item& a, const Item& b) {
            if (a.area != b.area) return a.area > b.area;
            if (a.maxSide != b.maxSide) return a.maxSide > b.maxSide;
            if (a.minSide != b.minSide) return a.minSide > b.minSide;
            return a.id < b.id;
        });
    } else if (mode == 1) {
        std::sort(v.begin(), v.end(), [](const Item& a, const Item& b) {
            if (a.maxSide != b.maxSide) return a.maxSide > b.maxSide;
            if (a.area != b.area) return a.area > b.area;
            if (a.minSide != b.minSide) return a.minSide > b.minSide;
            return a.id < b.id;
        });
    } else {
        std::sort(v.begin(), v.end(), [](const Item& a, const Item& b) {
            if (a.minSide != b.minSide) return a.minSide > b.minSide;
            if (a.area != b.area) return a.area > b.area;
            if (a.maxSide != b.maxSide) return a.maxSide > b.maxSide;
            return a.id < b.id;
        });
    }
    return v;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, BW, BH;
    if (!(std::cin >> N >> BW >> BH)) {
        return 0;
    }

    std::vector<Item> items;
    items.reserve(N);

    bool impossible = false;
    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        Item it;
        it.id = i;
        it.w = w;
        it.h = h;
        it.area = 1LL * w * h;
        it.maxSide = std::max(w, h);
        it.minSide = std::min(w, h);
        items.push_back(it);

        bool fit0 = (w <= BW && h <= BH);
        bool fit1 = (h <= BW && w <= BH);
        if (!fit0 && !fit1) impossible = true;
    }

    if (impossible) {
        std::cout << -1 << '\n';
        for (int i = 0; i < N; ++i) {
            std::cout << -1 << " 0 0 0\n";
        }
        return 0;
    }

    std::vector<int> modes;
    if (N <= 2000) {
        modes = {0, 1, 2};
    } else if (N <= 15000) {
        modes = {0, 1};
    } else {
        modes = {0};
    }

    PlanResult best;
    best.feasible = false;
    best.boardsUsed = std::numeric_limits<int>::max();

    for (int mode : modes) {
        std::vector<Item> order = makeOrder(items, mode);
        PlanResult cur = packWithShelfBestFit(order, N, BW, BH);
        if (!cur.feasible) continue;
        if (!best.feasible || cur.boardsUsed < best.boardsUsed) {
            best = std::move(cur);
        }
    }

    if (!best.feasible) {
        std::cout << -1 << '\n';
        for (int i = 0; i < N; ++i) {
            std::cout << -1 << " 0 0 0\n";
        }
        return 0;
    }

    std::cout << best.boardsUsed << '\n';
    for (int i = 0; i < N; ++i) {
        const Placement& p = best.placement[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rot << '\n';
    }

    return 0;
}
