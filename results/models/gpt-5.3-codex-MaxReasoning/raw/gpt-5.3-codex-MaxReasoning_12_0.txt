using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Text;

public static class Program
{
    private const double Eps = 1e-9;

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        if (!scanner.TryNextInt(out int vertexCount))
        {
            return;
        }

        int faceCount = scanner.NextInt();

        var vertices = new Vec3[vertexCount];
        for (int i = 0; i < vertexCount; i++)
        {
            double x = scanner.NextDouble();
            double y = scanner.NextDouble();
            double z = scanner.NextDouble();
            vertices[i] = new Vec3(x, y, z);
        }

        // Faces are not directly needed for this fast packing heuristic, but we parse them to consume input correctly.
        for (int i = 0; i < faceCount; i++)
        {
            scanner.NextInt();
            scanner.NextInt();
            scanner.NextInt();
        }

        double containerW = scanner.NextDouble();
        double containerH = scanner.NextDouble();
        double containerD = scanner.NextDouble();

        var container = new Container(containerW, containerH, containerD);
        var scale = ProblemScaleClassifier.Classify(container, vertexCount, faceCount);

        var orientations = OrientationGenerator.Generate(vertices, scale);
        if (orientations.Count == 0)
        {
            Console.WriteLine("0");
            return;
        }

        var planner = new Planner(container, scale);
        PackingPlan bestPlan = planner.BuildBestPlan(orientations);

        if (bestPlan == null || bestPlan.Count <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        bestPlan.Write(Console.Out);
    }

    private static long FloorToLong(double value)
    {
        if (value <= 0.0)
        {
            return 0;
        }

        long n = (long)Math.Floor(value + Eps);
        return n < 0 ? 0 : n;
    }

    private enum ProblemScale
    {
        Trivial,
        Medium,
        Large,
        Extreme
    }

    private readonly struct Container
    {
        public readonly double W;
        public readonly double H;
        public readonly double D;

        public Container(double w, double h, double d)
        {
            W = Math.Max(0.0, w);
            H = Math.Max(0.0, h);
            D = Math.Max(0.0, d);
        }

        public double AxisLength(int axis)
        {
            if (axis == 0) return W;
            if (axis == 1) return H;
            return D;
        }
    }

    private static class ProblemScaleClassifier
    {
        public static ProblemScale Classify(Container c, int vertexCount, int faceCount)
        {
            double maxDim = Math.Max(c.W, Math.Max(c.H, c.D));
            double volume = c.W * c.H * c.D;

            if (maxDim <= 25.0 && volume <= 20_000.0 && vertexCount <= 500 && faceCount <= 1_000)
            {
                return ProblemScale.Trivial;
            }

            if (maxDim <= 120.0 && volume <= 4_000_000.0 && vertexCount <= 5_000)
            {
                return ProblemScale.Medium;
            }

            if (maxDim <= 260.0 && volume <= 40_000_000.0)
            {
                return ProblemScale.Large;
            }

            return ProblemScale.Extreme;
        }
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }
    }

    private readonly struct Mat3
    {
        public readonly double M00;
        public readonly double M01;
        public readonly double M02;
        public readonly double M10;
        public readonly double M11;
        public readonly double M12;
        public readonly double M20;
        public readonly double M21;
        public readonly double M22;

        public Mat3(
            double m00, double m01, double m02,
            double m10, double m11, double m12,
            double m20, double m21, double m22)
        {
            M00 = m00; M01 = m01; M02 = m02;
            M10 = m10; M11 = m11; M12 = m12;
            M20 = m20; M21 = m21; M22 = m22;
        }

        public Vec3 Mul(in Vec3 v)
        {
            double x = M00 * v.X + M01 * v.Y + M02 * v.Z;
            double y = M10 * v.X + M11 * v.Y + M12 * v.Z;
            double z = M20 * v.X + M21 * v.Y + M22 * v.Z;
            return new Vec3(x, y, z);
        }
    }

    private readonly struct QuaternionD
    {
        public readonly double W;
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public QuaternionD(double w, double x, double y, double z)
        {
            W = w;
            X = x;
            Y = y;
            Z = z;
        }

        public QuaternionD Normalize()
        {
            double n2 = W * W + X * X + Y * Y + Z * Z;
            if (n2 <= Eps)
            {
                return new QuaternionD(1.0, 0.0, 0.0, 0.0);
            }

            double inv = 1.0 / Math.Sqrt(n2);
            return new QuaternionD(W * inv, X * inv, Y * inv, Z * inv);
        }

        public Mat3 ToMatrix()
        {
            double xx = X * X;
            double yy = Y * Y;
            double zz = Z * Z;
            double xy = X * Y;
            double xz = X * Z;
            double yz = Y * Z;
            double wx = W * X;
            double wy = W * Y;
            double wz = W * Z;

            return new Mat3(
                1.0 - 2.0 * (yy + zz), 2.0 * (xy - wz), 2.0 * (xz + wy),
                2.0 * (xy + wz), 1.0 - 2.0 * (xx + zz), 2.0 * (yz - wx),
                2.0 * (xz - wy), 2.0 * (yz + wx), 1.0 - 2.0 * (xx + yy)
            );
        }

        public static QuaternionD FromMatrix(in Mat3 m)
        {
            double trace = m.M00 + m.M11 + m.M22;
            double w, x, y, z;

            if (trace > 0.0)
            {
                double s = Math.Sqrt(trace + 1.0) * 2.0;
                w = 0.25 * s;
                x = (m.M21 - m.M12) / s;
                y = (m.M02 - m.M20) / s;
                z = (m.M10 - m.M01) / s;
            }
            else if (m.M00 > m.M11 && m.M00 > m.M22)
            {
                double s = Math.Sqrt(1.0 + m.M00 - m.M11 - m.M22) * 2.0;
                w = (m.M21 - m.M12) / s;
                x = 0.25 * s;
                y = (m.M01 + m.M10) / s;
                z = (m.M02 + m.M20) / s;
            }
            else if (m.M11 > m.M22)
            {
                double s = Math.Sqrt(1.0 + m.M11 - m.M00 - m.M22) * 2.0;
                w = (m.M02 - m.M20) / s;
                x = (m.M01 + m.M10) / s;
                y = 0.25 * s;
                z = (m.M12 + m.M21) / s;
            }
            else
            {
                double s = Math.Sqrt(1.0 + m.M22 - m.M00 - m.M11) * 2.0;
                w = (m.M10 - m.M01) / s;
                x = (m.M02 + m.M20) / s;
                y = (m.M12 + m.M21) / s;
                z = 0.25 * s;
            }

            return new QuaternionD(w, x, y, z).Normalize();
        }
    }

    private sealed class Orientation
    {
        public QuaternionD Q;
        public Mat3 M;
        public double MinX;
        public double MaxX;
        public double MinY;
        public double MaxY;
        public double MinZ;
        public double MaxZ;
        public double Dx;
        public double Dy;
        public double Dz;

        public string QwText = "1";
        public string QxText = "0";
        public string QyText = "0";
        public string QzText = "0";

        public double AxisDim(int axis)
        {
            if (axis == 0) return Dx;
            if (axis == 1) return Dy;
            return Dz;
        }

        public static Orientation? Create(QuaternionD q, Vec3[] vertices)
        {
            q = q.Normalize();
            Mat3 m = q.ToMatrix();

            double minX = double.PositiveInfinity;
            double minY = double.PositiveInfinity;
            double minZ = double.PositiveInfinity;
            double maxX = double.NegativeInfinity;
            double maxY = double.NegativeInfinity;
            double maxZ = double.NegativeInfinity;

            for (int i = 0; i < vertices.Length; i++)
            {
                Vec3 rv = m.Mul(vertices[i]);
                if (rv.X < minX) minX = rv.X;
                if (rv.Y < minY) minY = rv.Y;
                if (rv.Z < minZ) minZ = rv.Z;
                if (rv.X > maxX) maxX = rv.X;
                if (rv.Y > maxY) maxY = rv.Y;
                if (rv.Z > maxZ) maxZ = rv.Z;
            }

            double dx = maxX - minX;
            double dy = maxY - minY;
            double dz = maxZ - minZ;
            if (dx <= Eps || dy <= Eps || dz <= Eps)
            {
                return null;
            }

            var o = new Orientation
            {
                Q = q,
                M = m,
                MinX = minX,
                MaxX = maxX,
                MinY = minY,
                MaxY = maxY,
                MinZ = minZ,
                MaxZ = maxZ,
                Dx = dx,
                Dy = dy,
                Dz = dz
            };

            o.QwText = FormatFloat(o.Q.W);
            o.QxText = FormatFloat(o.Q.X);
            o.QyText = FormatFloat(o.Q.Y);
            o.QzText = FormatFloat(o.Q.Z);
            return o;
        }

        private static string FormatFloat(double value)
        {
            if (Math.Abs(value) < 5e-16) value = 0.0;
            return value.ToString("G17", CultureInfo.InvariantCulture);
        }

        public string SignatureKey()
        {
            return string.Concat(
                Quant(Dx), "|", Quant(Dy), "|", Quant(Dz), "|",
                Quant(MinX), "|", Quant(MinY), "|", Quant(MinZ)
            );
        }

        private static string Quant(double x)
        {
            long q = (long)Math.Round(x * 1_000_000.0);
            return q.ToString(CultureInfo.InvariantCulture);
        }
    }

    private static class OrientationGenerator
    {
        public static List<Orientation> Generate(Vec3[] vertices, ProblemScale scale)
        {
            var result = new List<Orientation>(256);
            var seen = new HashSet<string>(StringComparer.Ordinal);

            AddAxisAligned24(vertices, result, seen);

            int randomTarget = scale switch
            {
                ProblemScale.Trivial => 180,
                ProblemScale.Medium => 96,
                ProblemScale.Large => 36,
                _ => 0,
            };

            if (randomTarget > 0)
            {
                var rng = new Random(123456789);
                int attempts = randomTarget * 6;
                int added = 0;

                for (int i = 0; i < attempts && added < randomTarget; i++)
                {
                    QuaternionD q = RandomUnitQuaternion(rng);
                    Orientation? o = Orientation.Create(q, vertices);
                    if (o == null)
                    {
                        continue;
                    }

                    string key = o.SignatureKey();
                    if (seen.Add(key))
                    {
                        result.Add(o);
                        added++;
                    }
                }
            }

            return result;
        }

        private static void AddAxisAligned24(Vec3[] vertices, List<Orientation> result, HashSet<string> seen)
        {
            int[][] perms =
            {
                new[] { 0, 1, 2 },
                new[] { 0, 2, 1 },
                new[] { 1, 0, 2 },
                new[] { 1, 2, 0 },
                new[] { 2, 0, 1 },
                new[] { 2, 1, 0 }
            };

            int[] signs = { -1, 1 };
            foreach (int[] p in perms)
            {
                int permParity = PermutationParity(p);
                foreach (int sx in signs)
                {
                    foreach (int sy in signs)
                    {
                        foreach (int sz in signs)
                        {
                            int det = permParity * sx * sy * sz;
                            if (det != 1)
                            {
                                continue;
                            }

                            double m00 = p[0] == 0 ? sx : 0.0;
                            double m01 = p[0] == 1 ? sx : 0.0;
                            double m02 = p[0] == 2 ? sx : 0.0;

                            double m10 = p[1] == 0 ? sy : 0.0;
                            double m11 = p[1] == 1 ? sy : 0.0;
                            double m12 = p[1] == 2 ? sy : 0.0;

                            double m20 = p[2] == 0 ? sz : 0.0;
                            double m21 = p[2] == 1 ? sz : 0.0;
                            double m22 = p[2] == 2 ? sz : 0.0;

                            var m = new Mat3(m00, m01, m02, m10, m11, m12, m20, m21, m22);
                            QuaternionD q = QuaternionD.FromMatrix(m);

                            Orientation? o = Orientation.Create(q, vertices);
                            if (o == null)
                            {
                                continue;
                            }

                            string key = o.SignatureKey();
                            if (seen.Add(key))
                            {
                                result.Add(o);
                            }
                        }
                    }
                }
            }
        }

        private static int PermutationParity(int[] p)
        {
            int inv = 0;
            for (int i = 0; i < p.Length; i++)
            {
                for (int j = i + 1; j < p.Length; j++)
                {
                    if (p[i] > p[j]) inv++;
                }
            }
            return (inv & 1) == 0 ? 1 : -1;
        }

        private static QuaternionD RandomUnitQuaternion(Random rng)
        {
            double u1 = rng.NextDouble();
            double u2 = rng.NextDouble();
            double u3 = rng.NextDouble();

            double sqrt1 = Math.Sqrt(1.0 - u1);
            double sqrt2 = Math.Sqrt(u1);
            double t1 = 2.0 * Math.PI * u2;
            double t2 = 2.0 * Math.PI * u3;

            // Shoemake method, returned as (w, x, y, z).
            double x = sqrt1 * Math.Sin(t1);
            double y = sqrt1 * Math.Cos(t1);
            double z = sqrt2 * Math.Sin(t2);
            double w = sqrt2 * Math.Cos(t2);

            return new QuaternionD(w, x, y, z).Normalize();
        }
    }

    private readonly struct SingleScore
    {
        public readonly Orientation Orientation;
        public readonly long Nx;
        public readonly long Ny;
        public readonly long Nz;
        public readonly long Count;

        public SingleScore(Orientation orientation, long nx, long ny, long nz, long count)
        {
            Orientation = orientation;
            Nx = nx;
            Ny = ny;
            Nz = nz;
            Count = count;
        }
    }

    private sealed class Planner
    {
        private readonly Container _container;
        private readonly ProblemScale _scale;

        public Planner(Container container, ProblemScale scale)
        {
            _container = container;
            _scale = scale;
        }

        public PackingPlan BuildBestPlan(List<Orientation> orientations)
        {
            var singleScores = new List<SingleScore>(orientations.Count);
            PackingPlan? bestPlan = null;
            long bestCount = 0;

            for (int i = 0; i < orientations.Count; i++)
            {
                Orientation o = orientations[i];
                long nx = FloorToLong(_container.W / o.Dx);
                long ny = FloorToLong(_container.H / o.Dy);
                long nz = FloorToLong(_container.D / o.Dz);

                if (nx <= 0 || ny <= 0 || nz <= 0)
                {
                    continue;
                }

                long count = SafeMul3(nx, ny, nz);
                if (count <= 0)
                {
                    continue;
                }

                singleScores.Add(new SingleScore(o, nx, ny, nz, count));

                if (count > bestCount)
                {
                    bestCount = count;
                    bestPlan = new SinglePlan(o, nx, ny, nz, count);
                }
            }

            if (singleScores.Count == 0)
            {
                return new EmptyPlan();
            }

            singleScores.Sort((a, b) => b.Count.CompareTo(a.Count));

            int topK = _scale switch
            {
                ProblemScale.Trivial => 28,
                ProblemScale.Medium => 22,
                ProblemScale.Large => 12,
                _ => 6
            };

            if (topK > singleScores.Count)
            {
                topK = singleScores.Count;
            }

            // Medium and smaller problems use extra combinational search.
            if (_scale != ProblemScale.Extreme)
            {
                int exhaustiveLimit = _scale switch
                {
                    ProblemScale.Trivial => 5000,
                    ProblemScale.Medium => 2000,
                    _ => 600
                };

                int sampleCount = _scale switch
                {
                    ProblemScale.Trivial => 512,
                    ProblemScale.Medium => 384,
                    _ => 192
                };

                for (int axis = 0; axis < 3; axis++)
                {
                    for (int i = 0; i < topK; i++)
                    {
                        for (int j = 0; j < topK; j++)
                        {
                            Orientation a = singleScores[i].Orientation;
                            Orientation b = singleScores[j].Orientation;
                            SplitPlan? candidate = TryBuildBestSplit(axis, a, b, exhaustiveLimit, sampleCount);
                            if (candidate != null && candidate.Count > bestCount)
                            {
                                bestCount = candidate.Count;
                                bestPlan = candidate;
                            }
                        }
                    }
                }
            }

            return bestPlan ?? new EmptyPlan();
        }

        private SplitPlan? TryBuildBestSplit(int axis, Orientation a, Orientation b, int exhaustiveLimit, int sampleCount)
        {
            double containerAxis = _container.AxisLength(axis);
            double aLen = a.AxisDim(axis);
            double bLen = b.AxisDim(axis);
            if (aLen <= Eps || bLen <= Eps)
            {
                return null;
            }

            long crossA = CrossCount(a, axis);
            long crossB = CrossCount(b, axis);
            if (crossA <= 0 && crossB <= 0)
            {
                return null;
            }

            long nAMax = FloorToLong(containerAxis / aLen);
            if (nAMax < 0)
            {
                return null;
            }

            long bestNA = 0;
            long bestNB = FloorToLong(containerAxis / bLen);
            long bestTotal = SafeAdd(SafeMul(bestNA, crossA), SafeMul(bestNB, crossB));

            if (nAMax <= exhaustiveLimit)
            {
                for (long nA = 0; nA <= nAMax; nA++)
                {
                    EvaluateSplitCandidate(nA);
                }
            }
            else
            {
                var tested = new HashSet<long>();

                // Coarse scan across the whole range.
                long step = Math.Max(1L, nAMax / Math.Max(1, sampleCount));
                for (long nA = 0; nA <= nAMax; nA += step)
                {
                    if (tested.Add(nA))
                    {
                        EvaluateSplitCandidate(nA);
                    }
                }

                // Refine near boundaries where floor transitions usually matter most.
                long window = Math.Min(256L, nAMax);
                for (long i = 0; i <= window; i++)
                {
                    long low = i;
                    long high = nAMax - i;
                    if (tested.Add(low)) EvaluateSplitCandidate(low);
                    if (tested.Add(high)) EvaluateSplitCandidate(high);
                }
            }

            if (bestTotal <= 0)
            {
                return null;
            }

            return new SplitPlan(_container, axis, a, b, bestNA, bestNB, bestTotal);

            void EvaluateSplitCandidate(long nA)
            {
                double used = nA * aLen;
                double rem = containerAxis - used;
                if (rem < -Eps)
                {
                    return;
                }

                if (rem < 0.0)
                {
                    rem = 0.0;
                }

                long nB = FloorToLong(rem / bLen);
                long total = SafeAdd(SafeMul(nA, crossA), SafeMul(nB, crossB));
                if (total > bestTotal)
                {
                    bestTotal = total;
                    bestNA = nA;
                    bestNB = nB;
                }
            }
        }

        private long CrossCount(Orientation o, int axis)
        {
            if (axis == 0)
            {
                long ny = FloorToLong(_container.H / o.Dy);
                long nz = FloorToLong(_container.D / o.Dz);
                return SafeMul(ny, nz);
            }

            if (axis == 1)
            {
                long nx = FloorToLong(_container.W / o.Dx);
                long nz = FloorToLong(_container.D / o.Dz);
                return SafeMul(nx, nz);
            }

            {
                long nx = FloorToLong(_container.W / o.Dx);
                long ny = FloorToLong(_container.H / o.Dy);
                return SafeMul(nx, ny);
            }
        }

        private static long SafeMul(long a, long b)
        {
            if (a == 0 || b == 0)
            {
                return 0;
            }

            if (a > long.MaxValue / b)
            {
                return long.MaxValue;
            }

            return a * b;
        }

        private static long SafeMul3(long a, long b, long c)
        {
            return SafeMul(SafeMul(a, b), c);
        }

        private static long SafeAdd(long a, long b)
        {
            if (long.MaxValue - a < b)
            {
                return long.MaxValue;
            }

            return a + b;
        }
    }

    private abstract class PackingPlan
    {
        public long Count { get; protected set; }

        public abstract void Write(TextWriter writer);

        protected static void EmitRegion(
            PlacementWriter pw,
            Orientation o,
            double regionStartX,
            double regionStartY,
            double regionStartZ,
            long nx,
            long ny,
            long nz)
        {
            if (nx <= 0 || ny <= 0 || nz <= 0)
            {
                return;
            }

            double baseX = regionStartX - o.MinX;
            double baseY = regionStartY - o.MinY;
            double baseZ = regionStartZ - o.MinZ;

            for (long ix = 0; ix < nx; ix++)
            {
                double tx = baseX + ix * o.Dx;
                for (long iy = 0; iy < ny; iy++)
                {
                    double ty = baseY + iy * o.Dy;
                    for (long iz = 0; iz < nz; iz++)
                    {
                        double tz = baseZ + iz * o.Dz;
                        pw.WritePlacement(tx, ty, tz, o);
                    }
                }
            }
        }
    }

    private sealed class EmptyPlan : PackingPlan
    {
        public EmptyPlan()
        {
            Count = 0;
        }

        public override void Write(TextWriter writer)
        {
            writer.WriteLine("0");
        }
    }

    private sealed class SinglePlan : PackingPlan
    {
        private readonly Orientation _orientation;
        private readonly long _nx;
        private readonly long _ny;
        private readonly long _nz;

        public SinglePlan(Orientation orientation, long nx, long ny, long nz, long count)
        {
            _orientation = orientation;
            _nx = nx;
            _ny = ny;
            _nz = nz;
            Count = count;
        }

        public override void Write(TextWriter writer)
        {
            writer.WriteLine(Count.ToString(CultureInfo.InvariantCulture));
            var pw = new PlacementWriter(writer);
            EmitRegion(pw, _orientation, 0.0, 0.0, 0.0, _nx, _ny, _nz);
            pw.Flush();
        }
    }

    private sealed class SplitPlan : PackingPlan
    {
        private readonly Container _container;
        private readonly int _axis;
        private readonly Orientation _a;
        private readonly Orientation _b;
        private readonly long _nA;
        private readonly long _nB;

        public SplitPlan(Container container, int axis, Orientation a, Orientation b, long nA, long nB, long count)
        {
            _container = container;
            _axis = axis;
            _a = a;
            _b = b;
            _nA = nA;
            _nB = nB;
            Count = count;
        }

        public override void Write(TextWriter writer)
        {
            writer.WriteLine(Count.ToString(CultureInfo.InvariantCulture));
            var pw = new PlacementWriter(writer);

            if (_axis == 0)
            {
                long nyA = FloorToLong(_container.H / _a.Dy);
                long nzA = FloorToLong(_container.D / _a.Dz);
                EmitRegion(pw, _a, 0.0, 0.0, 0.0, _nA, nyA, nzA);

                double startX = _nA * _a.Dx;
                long nyB = FloorToLong(_container.H / _b.Dy);
                long nzB = FloorToLong(_container.D / _b.Dz);
                EmitRegion(pw, _b, startX, 0.0, 0.0, _nB, nyB, nzB);
            }
            else if (_axis == 1)
            {
                long nxA = FloorToLong(_container.W / _a.Dx);
                long nzA = FloorToLong(_container.D / _a.Dz);
                EmitRegion(pw, _a, 0.0, 0.0, 0.0, nxA, _nA, nzA);

                double startY = _nA * _a.Dy;
                long nxB = FloorToLong(_container.W / _b.Dx);
                long nzB = FloorToLong(_container.D / _b.Dz);
                EmitRegion(pw, _b, 0.0, startY, 0.0, nxB, _nB, nzB);
            }
            else
            {
                long nxA = FloorToLong(_container.W / _a.Dx);
                long nyA = FloorToLong(_container.H / _a.Dy);
                EmitRegion(pw, _a, 0.0, 0.0, 0.0, nxA, nyA, _nA);

                double startZ = _nA * _a.Dz;
                long nxB = FloorToLong(_container.W / _b.Dx);
                long nyB = FloorToLong(_container.H / _b.Dy);
                EmitRegion(pw, _b, 0.0, 0.0, startZ, nxB, nyB, _nB);
            }

            pw.Flush();
        }
    }

    private sealed class PlacementWriter
    {
        private readonly TextWriter _writer;
        private readonly StringBuilder _sb;

        public PlacementWriter(TextWriter writer)
        {
            _writer = writer;
            _sb = new StringBuilder(1 << 20);
        }

        public void WritePlacement(double tx, double ty, double tz, Orientation o)
        {
            AppendFloat(tx);
            _sb.Append(' ');
            AppendFloat(ty);
            _sb.Append(' ');
            AppendFloat(tz);
            _sb.Append(' ');
            _sb.Append(o.QwText);
            _sb.Append(' ');
            _sb.Append(o.QxText);
            _sb.Append(' ');
            _sb.Append(o.QyText);
            _sb.Append(' ');
            _sb.Append(o.QzText);
            _sb.Append('\n');

            if (_sb.Length >= (1 << 20))
            {
                Flush();
            }
        }

        public void Flush()
        {
            if (_sb.Length == 0)
            {
                return;
            }

            _writer.Write(_sb.ToString());
            _sb.Clear();
        }

        private void AppendFloat(double value)
        {
            if (Math.Abs(value) < 5e-16)
            {
                value = 0.0;
            }

            _sb.Append(value.ToString("G17", CultureInfo.InvariantCulture));
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
            _len = 0;
            _ptr = 0;
        }

        public bool TryNextInt(out int value)
        {
            string? token = NextToken();
            if (token == null)
            {
                value = 0;
                return false;
            }

            value = int.Parse(token, CultureInfo.InvariantCulture);
            return true;
        }

        public int NextInt()
        {
            string token = NextToken() ?? throw new EndOfStreamException();
            return int.Parse(token, CultureInfo.InvariantCulture);
        }

        public double NextDouble()
        {
            string token = NextToken() ?? throw new EndOfStreamException();
            return double.Parse(token, CultureInfo.InvariantCulture);
        }

        private string? NextToken()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0)
                {
                    return null;
                }
            }
            while (c <= 32);

            var sb = new StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = Read();
                if (c < 0)
                {
                    break;
                }
            }

            return sb.ToString();
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_ptr++];
        }
    }
