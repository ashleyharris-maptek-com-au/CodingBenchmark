OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpMemberDecorate %gl_PerVertex 0 BuiltIn Position
OpDecorate %gl_PerVertex Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_02 = OpConstant %float 0.02
%float_0_25 = OpConstant %float 0.25
%float_n1 = OpConstant %float -1
%perpSign = OpConstantComposite %v2float %float_n1 %float_1
%gl_PerVertex = OpTypeStruct %v4float
%_arr_gl_PerVertex_uint_4 = OpTypeArray %gl_PerVertex %uint_4
%_arr_v3float_uint_4 = OpTypeArray %v3float %uint_4
%_ptr_Input__arr_gl_PerVertex_uint_4 = OpTypePointer Input %_arr_gl_PerVertex_uint_4
%_ptr_Input__arr_v3float_uint_4 = OpTypePointer Input %_arr_v3float_uint_4
%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%_ptr_Function_int = OpTypePointer Function %int
%gl_in = OpVariable %_ptr_Input__arr_gl_PerVertex_uint_4 Input
%worldPos = OpVariable %_ptr_Input__arr_v3float_uint_4 Input
%normal = OpVariable %_ptr_Input__arr_v3float_uint_4 Input
%color = OpVariable %_ptr_Input__arr_v3float_uint_4 Input
%gl_out = OpVariable %_ptr_Output_gl_PerVertex Output
%outColor = OpVariable %_ptr_Output_v3float Output
%main = OpFunction %void None %fn_void
%entry = OpLabel
%i = OpVariable %_ptr_Function_int Function
%p0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_3 %int_0
%P0 = OpLoad %v4float %p0_ptr
%P1 = OpLoad %v4float %p1_ptr
%P2 = OpLoad %v4float %p2_ptr
%P3 = OpLoad %v4float %p3_ptr
%c1_ptr = OpAccessChain %_ptr_Input_v3float %color %int_1
%c2_ptr = OpAccessChain %_ptr_Input_v3float %color %int_2
%C1 = OpLoad %v3float %c1_ptr
%C2 = OpLoad %v3float %c2_ptr
%P0xy = OpVectorShuffle %v2float %P0 %P0 0 1
%P1xy = OpVectorShuffle %v2float %P1 %P1 0 1
%P2xy = OpVectorShuffle %v2float %P2 %P2 0 1
%P3xy = OpVectorShuffle %v2float %P3 %P3 0 1
%tanStart = OpFSub %v2float %P2xy %P0xy
%tanEnd = OpFSub %v2float %P3xy %P1xy
%colorDelta = OpFSub %v3float %C2 %C1
OpStore %i %int_0
OpBranch %loop_header
%loop_header = OpLabel
OpLoopMerge %loop_merge %loop_continue None
OpBranch %loop_cond
%loop_cond = OpLabel
%iv = OpLoad %int %i
%cond = OpSLessThan %bool %iv %int_5
OpBranchConditional %cond %loop_body %loop_merge
%loop_body = OpLabel
%ivf = OpConvertSToF %float %iv
%t = OpFMul %float %ivf %float_0_25
%posDelta = OpFSub %v4float %P2 %P1
%posStep = OpVectorTimesScalar %v4float %posDelta %t
%pos = OpFAdd %v4float %P1 %posStep
%tanDelta = OpFSub %v2float %tanEnd %tanStart
%tanStep = OpVectorTimesScalar %v2float %tanDelta %t
%tan = OpFAdd %v2float %tanStart %tanStep
%tanN = OpExtInst %v2float %1 Normalize %tan
%perpBase = OpVectorShuffle %v2float %tanN %tanN 1 0
%perpSwizzled = OpFMul %v2float %perpBase %perpSign
%perp = OpVectorTimesScalar %v2float %perpSwizzled %float_0_02
%px = OpCompositeExtract %float %perp 0
%py = OpCompositeExtract %float %perp 1
%offset = OpCompositeConstruct %v4float %px %py %float_0 %float_0
%colStep = OpVectorTimesScalar %v3float %colorDelta %t
%col = OpFAdd %v3float %C1 %colStep
%posPlus = OpFAdd %v4float %pos %offset
%posMinus = OpFSub %v4float %pos %offset
%outPosPtrA = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %outPosPtrA %posPlus
OpStore %outColor %col
OpEmitVertex
%outPosPtrB = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %outPosPtrB %posMinus
OpStore %outColor %col
OpEmitVertex
OpBranch %loop_continue
%loop_continue = OpLabel
%next = OpIAdd %int %iv %int_1
OpStore %i %next
OpBranch %loop_header
%loop_merge = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
