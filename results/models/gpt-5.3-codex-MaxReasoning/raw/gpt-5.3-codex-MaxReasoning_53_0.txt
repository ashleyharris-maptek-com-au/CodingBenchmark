import math

_MEM = {
    "init": False,
    "pos": [0.0, 0.0, 0.0],
    "vel": [0.0, 0.0, 0.0],
    "int_pos": [0.0, 0.0, 0.0],
    "phase": "far",
}


def _finite(x):
    return isinstance(x, (int, float)) and math.isfinite(x)


def _clamp(x, lo, hi):
    if x < lo:
        return lo
    if x > hi:
        return hi
    return x


def _norm(v):
    return math.sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2])


def _median(vals):
    if not vals:
        return None
    s = sorted(vals)
    n = len(s)
    m = n // 2
    if n % 2:
        return s[m]
    return 0.5 * (s[m - 1] + s[m])


def _valid_value(state, value_key, valid_key):
    if not bool(state.get(valid_key, False)):
        return None
    v = state.get(value_key)
    if not _finite(v):
        return None
    return float(v)


def _fuse_axis(candidates, prev, max_jump):
    vals = [v for v in candidates if v is not None]
    if not vals:
        return None
    if _finite(prev):
        near = [v for v in vals if abs(v - prev) <= max_jump]
        if near:
            vals = near
    return sum(vals) / float(len(vals))


def _camera_bearing(state):
    if not bool(state.get("dock_camera_active", False)):
        return None, None
    bx = _valid_value(state, "dock_camera_bearing_x", "dock_camera_bearing_x_valid")
    by = _valid_value(state, "dock_camera_bearing_y", "dock_camera_bearing_y_valid")
    bz = _valid_value(state, "dock_camera_bearing_z", "dock_camera_bearing_z_valid")
    if bx is None or by is None or bz is None:
        return None, None
    b = [bx, by, bz]
    bn = _norm(b)
    if bn < 1e-8:
        return None, None
    b = [b[0] / bn, b[1] / bn, b[2] / bn]
    r = state.get("dock_camera_range_m")
    if _finite(r) and r > 0.0:
        return b, float(r)
    return b, None


def _fuse_gyro_dps(state, axis):
    vals = []
    for i in (1, 2, 3):
        v = _valid_value(state, f"gyro_{i}_{axis}_dps", f"gyro_{i}_{axis}_valid")
        if v is not None:
            vals.append(v)
    return _median(vals)


def autopilot_step(state: dict, dt: float) -> dict:
    global _MEM

    if not _finite(dt):
        dt = 0.2
    dt = _clamp(float(dt), 0.05, 2.0)

    prev_pos = _MEM["pos"]
    prev_vel = _MEM["vel"]

    axes = ("x", "y", "z")

    radar_pos = [
        _valid_value(state, f"radar_rel_pos_{a}_m", f"radar_rel_{a}_valid") for a in axes
    ]
    lidar_pos = [
        _valid_value(state, f"lidar_rel_pos_{a}_m", f"lidar_rel_{a}_valid") for a in axes
    ]
    radar_vel = [
        _valid_value(state, f"radar_rel_vel_{a}_ms", f"radar_relv_{a}_valid") for a in axes
    ]
    lidar_vel = [
        _valid_value(state, f"lidar_rel_vel_{a}_ms", f"lidar_relv_{a}_valid") for a in axes
    ]

    pos_meas = [None, None, None]
    vel_meas = [None, None, None]

    for i in range(3):
        max_jump_pos = 80.0 * max(1.0, dt) + 0.35 * abs(prev_pos[i]) + 2.0
        max_jump_vel = 4.0 + 0.6 * abs(prev_vel[i]) + 0.02 * abs(prev_pos[i])

        pos_meas[i] = _fuse_axis([radar_pos[i], lidar_pos[i]], prev_pos[i], max_jump_pos)
        vel_meas[i] = _fuse_axis([radar_vel[i], lidar_vel[i]], prev_vel[i], max_jump_vel)

    have_pos = all(v is not None for v in pos_meas)
    have_vel = all(v is not None for v in vel_meas)

    ranges = []
    for i in (1, 2, 3):
        rv = _valid_value(state, f"range_{i}_m", f"range_{i}_valid")
        if rv is not None and rv > 0.0:
            ranges.append(rv)
    range_scalar = _median(ranges)

    cam_dir, cam_range = _camera_bearing(state)
    if cam_dir is not None and cam_range is not None and not have_pos:
        pos_meas = [cam_dir[0] * cam_range, cam_dir[1] * cam_range, cam_dir[2] * cam_range]
        have_pos = True
    elif cam_dir is not None and cam_range is not None and have_pos and cam_range < 400.0:
        r_est = _norm(pos_meas)
        if r_est > 1e-6:
            mag = cam_range if abs(cam_range - r_est) < 150.0 else r_est
            blend = 0.3 if cam_range > 80.0 else 0.55
            cam_pos = [cam_dir[0] * mag, cam_dir[1] * mag, cam_dir[2] * mag]
            pos_meas = [
                (1.0 - blend) * pos_meas[0] + blend * cam_pos[0],
                (1.0 - blend) * pos_meas[1] + blend * cam_pos[1],
                (1.0 - blend) * pos_meas[2] + blend * cam_pos[2],
            ]

    if not have_vel and have_pos and _MEM["init"]:
        vel_meas = [
            (pos_meas[0] - prev_pos[0]) / dt,
            (pos_meas[1] - prev_pos[1]) / dt,
            (pos_meas[2] - prev_pos[2]) / dt,
        ]
        have_vel = True

    if not _MEM["init"]:
        if have_pos:
            _MEM["pos"] = [float(pos_meas[0]), float(pos_meas[1]), float(pos_meas[2])]
        if have_vel:
            _MEM["vel"] = [float(vel_meas[0]), float(vel_meas[1]), float(vel_meas[2])]
        _MEM["init"] = True
        prev_pos = _MEM["pos"]
        prev_vel = _MEM["vel"]

    pred_pos = [
        prev_pos[0] + prev_vel[0] * dt,
        prev_pos[1] + prev_vel[1] * dt,
        prev_pos[2] + prev_vel[2] * dt,
    ]

    r_guess = _norm(pos_meas) if have_pos else (range_scalar if range_scalar is not None else _norm(pred_pos))

    if have_pos:
        alpha_pos = 0.78 if r_guess < 60.0 else (0.5 if r_guess > 300.0 else 0.62)
        pos_est = [
            alpha_pos * pos_meas[0] + (1.0 - alpha_pos) * pred_pos[0],
            alpha_pos * pos_meas[1] + (1.0 - alpha_pos) * pred_pos[1],
            alpha_pos * pos_meas[2] + (1.0 - alpha_pos) * pred_pos[2],
        ]
    else:
        pos_est = pred_pos

    if have_vel:
        alpha_vel = 0.72 if r_guess < 120.0 else 0.52
        vel_est = [
            alpha_vel * vel_meas[0] + (1.0 - alpha_vel) * prev_vel[0],
            alpha_vel * vel_meas[1] + (1.0 - alpha_vel) * prev_vel[1],
            alpha_vel * vel_meas[2] + (1.0 - alpha_vel) * prev_vel[2],
        ]
    else:
        raw_vel = [
            (pos_est[0] - prev_pos[0]) / dt,
            (pos_est[1] - prev_pos[1]) / dt,
            (pos_est[2] - prev_pos[2]) / dt,
        ]
        vel_est = [
            0.6 * raw_vel[0] + 0.4 * prev_vel[0],
            0.6 * raw_vel[1] + 0.4 * prev_vel[1],
            0.6 * raw_vel[2] + 0.4 * prev_vel[2],
        ]

    if range_scalar is not None:
        r_est = _norm(pos_est)
        if r_est > 1e-6 and abs(range_scalar - r_est) < 150.0:
            gain = 0.05 if r_est > 200.0 else 0.12
            scale = (r_est + gain * (range_scalar - r_est)) / r_est
            pos_est = [pos_est[0] * scale, pos_est[1] * scale, pos_est[2] * scale]

    r = _norm(pos_est)
    speed = _norm(vel_est)

    if r < 1e-6:
        r = 1e-6

    los = [pos_est[0] / r, pos_est[1] / r, pos_est[2] / r]
    closing = -(pos_est[0] * vel_est[0] + pos_est[1] * vel_est[1] + pos_est[2] * vel_est[2]) / r

    phase = _MEM["phase"]
    if phase == "far":
        if r < 180.0:
            phase = "medium"
    elif phase == "medium":
        if r > 260.0:
            phase = "far"
        elif r < 45.0:
            phase = "close"
    elif phase == "close":
        if r > 90.0:
            phase = "medium"
        elif r < 8.0:
            phase = "final"
    else:
        if r > 20.0:
            phase = "close"

    fuel_pct = state.get("fuel_remaining_pct")
    if not _finite(fuel_pct):
        fuel_pct = 100.0

    if phase == "far":
        target_closing = 1.1
        v_ref = [
            _clamp(-0.006 * pos_est[0], -0.35, 0.35),
            _clamp(-0.002 * pos_est[1], -0.40, 0.40),
            _clamp(-0.006 * pos_est[2], -0.35, 0.35),
        ]
        orbit_bias_y = 0.38 * math.tanh(pos_est[1] / 180.0)

        u_x = 0.55 * (v_ref[0] - vel_est[0]) - 0.0015 * pos_est[0]
        u_y = -0.35 * vel_est[1] - 0.0004 * pos_est[1] + orbit_bias_y
        u_z = 0.55 * (v_ref[2] - vel_est[2]) - 0.0015 * pos_est[2]

        _MEM["int_pos"] = [
            0.98 * _MEM["int_pos"][0],
            0.98 * _MEM["int_pos"][1],
            0.98 * _MEM["int_pos"][2],
        ]
    else:
        if phase == "medium":
            k_pos = 0.014
            k_vel = 0.92
            k_int = 0.00025
            target_closing = 0.45
        elif phase == "close":
            k_pos = 0.025
            k_vel = 1.18
            k_int = 0.0005
            target_closing = 0.15
        else:
            k_pos = 0.035
            k_vel = 1.35
            k_int = 0.0008
            target_closing = 0.05

        if fuel_pct < 20.0:
            target_closing *= 0.8
        if fuel_pct < 10.0:
            target_closing *= 0.75

        v_ref = [
            -k_pos * pos_est[0],
            -k_pos * pos_est[1],
            -k_pos * pos_est[2],
        ]

        if phase == "medium":
            v_cap = 0.65
        elif phase == "close":
            v_cap = 0.25
        else:
            v_cap = 0.09

        v_ref = [
            _clamp(v_ref[0], -v_cap, v_cap),
            _clamp(v_ref[1], -v_cap, v_cap),
            _clamp(v_ref[2], -v_cap, v_cap),
        ]

        _MEM["int_pos"][0] = _clamp(_MEM["int_pos"][0] + pos_est[0] * dt, -200.0, 200.0)
        _MEM["int_pos"][1] = _clamp(_MEM["int_pos"][1] + pos_est[1] * dt, -200.0, 200.0)
        _MEM["int_pos"][2] = _clamp(_MEM["int_pos"][2] + pos_est[2] * dt, -200.0, 200.0)

        u_x = k_vel * (v_ref[0] - vel_est[0]) - k_int * _MEM["int_pos"][0]
        u_y = k_vel * (v_ref[1] - vel_est[1]) - k_int * _MEM["int_pos"][1]
        u_z = k_vel * (v_ref[2] - vel_est[2]) - k_int * _MEM["int_pos"][2]

    if r < 120.0:
        if closing > 1.5 * target_closing:
            brake = _clamp((closing - target_closing) * 0.9, 0.0, 0.9)
            u_x += brake * los[0]
            u_y += brake * los[1]
            u_z += brake * los[2]
        elif closing < -0.2:
            push = _clamp((-closing - 0.2) * 0.5, 0.0, 0.35)
            u_x -= push * los[0]
            u_y -= push * los[1]
            u_z -= push * los[2]

    if abs(pos_est[2]) > 30.0 and phase != "far":
        u_z += -0.002 * pos_est[2]

    max_cmd = 1.0
    if fuel_pct < 20.0:
        max_cmd = 0.75
    if fuel_pct < 10.0:
        max_cmd = 0.50
    if fuel_pct < 5.0:
        max_cmd = 0.35

    if phase == "close":
        max_cmd = min(max_cmd, 0.55)
    if phase == "final":
        max_cmd = min(max_cmd, 0.35)

    if r < 2.2 and speed < 0.25:
        u_x *= 0.4
        u_y *= 0.4
        u_z *= 0.4

    thrust_x = _clamp(u_x, -max_cmd, max_cmd)
    thrust_y = _clamp(u_y, -max_cmd, max_cmd)
    thrust_z = _clamp(u_z, -max_cmd, max_cmd)

    roll_dps = _fuse_gyro_dps(state, "roll")
    pitch_dps = _fuse_gyro_dps(state, "pitch")
    yaw_dps = _fuse_gyro_dps(state, "yaw")

    d2r = math.pi / 180.0
    roll_rate = (roll_dps * d2r) if roll_dps is not None else 0.0
    pitch_rate = (pitch_dps * d2r) if pitch_dps is not None else 0.0
    yaw_rate = (yaw_dps * d2r) if yaw_dps is not None else 0.0

    att_rate_roll = _clamp(-0.8 * roll_rate, -0.035, 0.035)
    att_rate_pitch = _clamp(-0.8 * pitch_rate, -0.035, 0.035)
    att_rate_yaw = _clamp(-0.8 * yaw_rate, -0.035, 0.035)

    _MEM["pos"] = [pos_est[0], pos_est[1], pos_est[2]]
    _MEM["vel"] = [vel_est[0], vel_est[1], vel_est[2]]
    _MEM["phase"] = phase

    return {
        "thrust_x": thrust_x,
        "thrust_y": thrust_y,
        "thrust_z": thrust_z,
        "att_rate_roll": att_rate_roll,
        "att_rate_pitch": att_rate_pitch,
        "att_rate_yaw": att_rate_yaw,
    }
