OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Invocations 1
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_gl_PerVertex_in_3 = OpTypePointer Input %arr_gl_PerVertex_in_3
%ptr_Input_v4float = OpTypePointer Input %v4float
%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3float_3 = OpTypePointer Input %arr_v3float_3
%ptr_Input_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_3 Input
%worldPos = OpVariable %ptr_Input_arr_v3float_3 Input
%normal = OpVariable %ptr_Input_arr_v3float_3 Input
%color = OpVariable %ptr_Input_arr_v3float_3 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %func
%entry = OpLabel
%in_pos_ptr_0 = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%in_pos_0 = OpLoad %v4float %in_pos_ptr_0
%out_pos_ptr_0 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr_0 %in_pos_0
%in_col_ptr_0 = OpAccessChain %ptr_Input_v3float %color %int_0
%in_col_0 = OpLoad %v3float %in_col_ptr_0
OpStore %outColor %in_col_0
OpEmitVertex

%in_pos_ptr_1 = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%in_pos_1 = OpLoad %v4float %in_pos_ptr_1
%out_pos_ptr_1 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr_1 %in_pos_1
%in_col_ptr_1 = OpAccessChain %ptr_Input_v3float %color %int_1
%in_col_1 = OpLoad %v3float %in_col_ptr_1
OpStore %outColor %in_col_1
OpEmitVertex

%in_pos_ptr_2 = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%in_pos_2 = OpLoad %v4float %in_pos_ptr_2
%out_pos_ptr_2 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr_2 %in_pos_2
%in_col_ptr_2 = OpAccessChain %ptr_Input_v3float %color %int_2
%in_col_2 = OpLoad %v3float %in_col_ptr_2
OpStore %outColor %in_col_2
OpEmitVertex

OpEndPrimitive
OpReturn
