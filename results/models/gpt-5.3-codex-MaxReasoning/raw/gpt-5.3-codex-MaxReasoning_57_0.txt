const fs = require('fs');

function outputEmpty() {
  process.stdout.write('{"firebreaks":[]}');
}

const raw = fs.readFileSync(0, 'utf8');
let data;
try {
  data = JSON.parse(raw);
} catch {
  outputEmpty();
  process.exit(0);
}

const width = Number(data.width);
const height = Number(data.height);
const grid = Array.isArray(data.grid) ? data.grid : [];
const budget = Math.max(0, Math.floor(Number(data.cutBudget) || 0));

if (!Number.isInteger(width) || !Number.isInteger(height) || width <= 0 || height <= 0 || grid.length !== height) {
  outputEmpty();
  process.exit(0);
}

const area = width * height;
if (!Number.isFinite(area) || area <= 0) {
  outputEmpty();
  process.exit(0);
}

const posToGlobal = new Int32Array(area);
posToGlobal.fill(-1);

const cellX = [];
const cellY = [];
let cellCount = 0;

for (let y = 0; y < height; y++) {
  const row = typeof grid[y] === 'string' ? grid[y] : '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) {
      posToGlobal[y * width + x] = cellCount;
      cellX.push(x);
      cellY.push(y);
      cellCount++;
    }
  }
}

if (cellCount === 0 || budget === 0) {
  outputEmpty();
  process.exit(0);
}

const ignitionGlobals = [];
const ignitionMark = new Uint8Array(cellCount);
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];

for (let i = 0; i < ignitions.length; i++) {
  const it = ignitions[i];
  if (!Array.isArray(it) || it.length < 2) continue;
  const x = it[0] | 0;
  const y = it[1] | 0;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const g = posToGlobal[y * width + x];
  if (g < 0) continue;
  if (ignitionMark[g] === 0) {
    ignitionMark[g] = 1;
    ignitionGlobals.push(g);
  }
}

if (ignitionGlobals.length === 0) {
  outputEmpty();
  process.exit(0);
}

const reachable = new Uint8Array(cellCount);
const dist = new Int32Array(cellCount);
dist.fill(-1);
const parent = new Int32Array(cellCount);
parent.fill(-1);
const bfsOrder = new Int32Array(cellCount);
let orderLen = 0;

const q = new Int32Array(cellCount);
let qh = 0;
let qt = 0;

for (let i = 0; i < ignitionGlobals.length; i++) {
  const g = ignitionGlobals[i];
  reachable[g] = 1;
  dist[g] = 0;
  q[qt++] = g;
}

while (qh < qt) {
  const g = q[qh++];
  bfsOrder[orderLen++] = g;
  const x = cellX[g];
  const y = cellY[g];

  let nx = x + 1;
  let ny = y;
  if (nx < width) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng] === 0) {
      reachable[ng] = 1;
      dist[ng] = dist[g] + 1;
      parent[ng] = g;
      q[qt++] = ng;
    }
  }

  nx = x - 1;
  ny = y;
  if (nx >= 0) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng] === 0) {
      reachable[ng] = 1;
      dist[ng] = dist[g] + 1;
      parent[ng] = g;
      q[qt++] = ng;
    }
  }

  nx = x;
  ny = y + 1;
  if (ny < height) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng] === 0) {
      reachable[ng] = 1;
      dist[ng] = dist[g] + 1;
      parent[ng] = g;
      q[qt++] = ng;
    }
  }

  nx = x;
  ny = y - 1;
  if (ny >= 0) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng] === 0) {
      reachable[ng] = 1;
      dist[ng] = dist[g] + 1;
      parent[ng] = g;
      q[qt++] = ng;
    }
  }
}

if (orderLen === 0) {
  outputEmpty();
  process.exit(0);
}

const relFromGlobal = new Int32Array(cellCount);
relFromGlobal.fill(-1);
const relGlobal = [];
for (let g = 0; g < cellCount; g++) {
  if (reachable[g]) {
    relFromGlobal[g] = relGlobal.length;
    relGlobal.push(g);
  }
}
const R = relGlobal.length;
if (R === 0) {
  outputEmpty();
  process.exit(0);
}

const deg = new Int32Array(R);
for (let r = 0; r < R; r++) {
  const g = relGlobal[r];
  const x = cellX[g];
  const y = cellY[g];

  let nx = x + 1;
  let ny = y;
  if (nx < width) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng]) deg[r]++;
  }

  nx = x - 1;
  ny = y;
  if (nx >= 0) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng]) deg[r]++;
  }

  nx = x;
  ny = y + 1;
  if (ny < height) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng]) deg[r]++;
  }

  nx = x;
  ny = y - 1;
  if (ny >= 0) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng]) deg[r]++;
  }
}

const nbrStart = new Int32Array(R + 1);
for (let r = 0; r < R; r++) nbrStart[r + 1] = nbrStart[r] + deg[r];
const nbr = new Int32Array(nbrStart[R]);
const fillPtr = new Int32Array(R);
for (let r = 0; r < R; r++) fillPtr[r] = nbrStart[r];

for (let r = 0; r < R; r++) {
  const g = relGlobal[r];
  const x = cellX[g];
  const y = cellY[g];

  let nx = x + 1;
  let ny = y;
  if (nx < width) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng]) nbr[fillPtr[r]++] = relFromGlobal[ng];
  }

  nx = x - 1;
  ny = y;
  if (nx >= 0) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng]) nbr[fillPtr[r]++] = relFromGlobal[ng];
  }

  nx = x;
  ny = y + 1;
  if (ny < height) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng]) nbr[fillPtr[r]++] = relFromGlobal[ng];
  }

  nx = x;
  ny = y - 1;
  if (ny >= 0) {
    const ng = posToGlobal[ny * width + nx];
    if (ng >= 0 && reachable[ng]) nbr[fillPtr[r]++] = relFromGlobal[ng];
  }
}

const ignitionRel = [];
for (let i = 0; i < ignitionGlobals.length; i++) {
  const r = relFromGlobal[ignitionGlobals[i]];
  if (r >= 0) ignitionRel.push(r);
}
if (ignitionRel.length === 0) {
  outputEmpty();
  process.exit(0);
}

const townValueGlobal = new Float64Array(cellCount);
const towns = Array.isArray(data.towns) ? data.towns : [];
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (!t || typeof t !== 'object') continue;
  const x = t.x | 0;
  const y = t.y | 0;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const g = posToGlobal[y * width + x];
  if (g < 0 || reachable[g] === 0) continue;
  const v = Number(t.value);
  if (v > 0 && Number.isFinite(v)) townValueGlobal[g] += v;
}

const townValueRel = new Float64Array(R);
let totalProtectable = 0;
let maxTownCellValue = 0;
const townRelIds = [];
const townRelVals = [];
for (let r = 0; r < R; r++) {
  const g = relGlobal[r];
  const v = townValueGlobal[g];
  if (v > 0) {
    townValueRel[r] = v;
    townRelIds.push(r);
    townRelVals.push(v);
    totalProtectable += v;
    if (v > maxTownCellValue) maxTownCellValue = v;
  }
}

if (totalProtectable <= 0) {
  outputEmpty();
  process.exit(0);
}

const importanceGlobal = new Float64Array(cellCount);
for (let r = 0; r < R; r++) {
  const g = relGlobal[r];
  const v = townValueGlobal[g];
  if (v > 0) importanceGlobal[g] = v;
}
for (let i = orderLen - 1; i >= 0; i--) {
  const g = bfsOrder[i];
  const p = parent[g];
  if (p >= 0) importanceGlobal[p] += importanceGlobal[g];
}

const scoreRel = new Float64Array(R);
for (let r = 0; r < R; r++) {
  const g = relGlobal[r];
  const d = dist[g] <= 0 ? 0 : dist[g];
  scoreRel[r] = importanceGlobal[g] / (1 + 0.15 * d);
}

const cutStamp = new Int32Array(R);
const burnStamp = new Int32Array(R);
const seenStamp = new Int32Array(R);
let stampCounter = 1;
const evalQueue = new Int32Array(R);

function simulateSaved(cutList) {
  stampCounter++;
  const cutS = stampCounter;
  for (let i = 0; i < cutList.length; i++) cutStamp[cutList[i]] = cutS;

  stampCounter++;
  const burnS = stampCounter;

  let h = 0;
  let t = 0;
  for (let i = 0; i < ignitionRel.length; i++) {
    const r = ignitionRel[i];
    if (cutStamp[r] !== cutS && burnStamp[r] !== burnS) {
      burnStamp[r] = burnS;
      evalQueue[t++] = r;
    }
  }

  while (h < t) {
    const r = evalQueue[h++];
    for (let p = nbrStart[r], pe = nbrStart[r + 1]; p < pe; p++) {
      const nr = nbr[p];
      if (cutStamp[nr] === cutS || burnStamp[nr] === burnS) continue;
      burnStamp[nr] = burnS;
      evalQueue[t++] = nr;
    }
  }

  let burned = 0;
  for (let i = 0; i < townRelIds.length; i++) {
    if (burnStamp[townRelIds[i]] === burnS) burned += townRelVals[i];
  }
  return totalProtectable - burned;
}

function normalizeCuts(raw) {
  stampCounter++;
  const s = stampCounter;
  const out = [];
  for (let i = 0; i < raw.length; i++) {
    const r = raw[i] | 0;
    if (r < 0 || r >= R) continue;
    if (seenStamp[r] === s) continue;
    seenStamp[r] = s;
    out.push(r);
    if (out.length >= budget) break;
  }
  return out;
}

function topByScore(raw, k) {
  if (k <= 0) return [];
  const arr = [];
  stampCounter++;
  const s = stampCounter;
  for (let i = 0; i < raw.length; i++) {
    const r = raw[i] | 0;
    if (r < 0 || r >= R || seenStamp[r] === s) continue;
    seenStamp[r] = s;
    arr.push(r);
  }
  arr.sort((a, b) => scoreRel[b] - scoreRel[a]);
  if (arr.length > k) arr.length = k;
  return arr;
}

const candidates = [];
candidates.push([]);

const ignitionCandidate = normalizeCuts(ignitionRel);
if (ignitionCandidate.length > 0) candidates.push(ignitionCandidate);

const sortedRel = new Array(R);
for (let r = 0; r < R; r++) sortedRel[r] = r;
sortedRel.sort((a, b) => scoreRel[b] - scoreRel[a]);
const scoreCandidate = sortedRel.slice(0, Math.min(budget, sortedRel.length));
if (scoreCandidate.length > 0) candidates.push(scoreCandidate);

const MAX_FLOW_CELLS = 25000;
const EPS = 1e-9;

class Dinic {
  constructor(n) {
    this.n = n;
    this.head = new Int32Array(n);
    this.head.fill(-1);
    this.to = [];
    this.cap = [];
    this.next = [];
  }

  addEdge(u, v, c) {
    const ei = this.to.length;
    this.to.push(v);
    this.cap.push(c);
    this.next.push(this.head[u]);
    this.head[u] = ei;

    this.to.push(u);
    this.cap.push(0);
    this.next.push(this.head[v]);
    this.head[v] = ei + 1;
  }

  maxFlow(s, t) {
    const n = this.n;
    const level = new Int32Array(n);
    const q2 = new Int32Array(n);
    const stackNodes = new Int32Array(n);
    const stackEdges = new Int32Array(n);
    let flow = 0;

    while (true) {
      level.fill(-1);
      let h = 0;
      let tt = 0;
      level[s] = 0;
      q2[tt++] = s;

      while (h < tt) {
        const u = q2[h++];
        for (let e = this.head[u]; e !== -1; e = this.next[e]) {
          if (this.cap[e] <= EPS) continue;
          const v = this.to[e];
          if (level[v] !== -1) continue;
          level[v] = level[u] + 1;
          q2[tt++] = v;
        }
      }

      if (level[t] === -1) break;

      const it = level.slice();
      for (let i = 0; i < n; i++) it[i] = this.head[i];

      let depth = 0;
      stackNodes[0] = s;
      let u = s;

      while (true) {
        if (u === t) {
          let add = Infinity;
          for (let i = 0; i < depth; i++) {
            const c = this.cap[stackEdges[i]];
            if (c < add) add = c;
          }
          if (!(add > EPS) || !Number.isFinite(add)) break;

          for (let i = 0; i < depth; i++) {
            const e = stackEdges[i];
            this.cap[e] -= add;
            this.cap[e ^ 1] += add;
          }
          flow += add;

          let newDepth = depth - 1;
          for (let i = depth - 1; i >= 0; i--) {
            const e = stackEdges[i];
            if (this.cap[e] <= EPS) {
              it[stackNodes[i]] = this.next[e];
              newDepth = i - 1;
            }
          }
          if (newDepth < 0) {
            depth = 0;
            u = s;
          } else {
            depth = newDepth + 1;
            u = stackNodes[depth];
          }
          continue;
        }

        let e = it[u];
        while (e !== -1) {
          const v = this.to[e];
          if (this.cap[e] > EPS && level[v] === level[u] + 1) break;
          e = this.next[e];
        }
        it[u] = e;

        if (e === -1) {
          if (depth === 0) break;
          depth--;
          const p = stackNodes[depth];
          const pe = stackEdges[depth];
          it[p] = this.next[pe];
          u = p;
        } else {
          const v = this.to[e];
          stackEdges[depth] = e;
          depth++;
          stackNodes[depth] = v;
          u = v;
        }
      }
    }

    return flow;
  }

  reachableFrom(s) {
    const vis = new Uint8Array(this.n);
    const qq = new Int32Array(this.n);
    let h = 0;
    let t = 0;
    vis[s] = 1;
    qq[t++] = s;
    while (h < t) {
      const u = qq[h++];
      for (let e = this.head[u]; e !== -1; e = this.next[e]) {
        if (this.cap[e] <= EPS) continue;
        const v = this.to[e];
        if (vis[v]) continue;
        vis[v] = 1;
        qq[t++] = v;
      }
    }
    return vis;
  }
}

if (R <= MAX_FLOW_CELLS && maxTownCellValue > 0) {
  const pairA = [];
  const pairB = [];
  for (let r = 0; r < R; r++) {
    for (let p = nbrStart[r], pe = nbrStart[r + 1]; p < pe; p++) {
      const nr = nbr[p];
      if (nr > r) {
        pairA.push(r);
        pairB.push(nr);
      }
    }
  }

  const cache = new Map();

  function solveLambda(lambda) {
    const key = lambda.toFixed(6);
    if (cache.has(key)) return cache.get(key);

    const nNodes = 2 * R + 2;
    const s = 2 * R;
    const t = s + 1;

    let INF = totalProtectable + lambda * (R + 1) + 1;
    if (!Number.isFinite(INF) || INF <= 0) INF = 1e30;

    const mf = new Dinic(nNodes);

    for (let r = 0; r < R; r++) {
      mf.addEdge(2 * r, 2 * r + 1, lambda);
    }

    for (let i = 0; i < pairA.length; i++) {
      const a = pairA[i];
      const b = pairB[i];
      mf.addEdge(2 * a + 1, 2 * b, INF);
      mf.addEdge(2 * b + 1, 2 * a, INF);
    }

    for (let i = 0; i < ignitionRel.length; i++) {
      mf.addEdge(s, 2 * ignitionRel[i], INF);
    }

    for (let i = 0; i < townRelIds.length; i++) {
      const r = townRelIds[i];
      mf.addEdge(2 * r + 1, t, townRelVals[i]);
    }

    mf.maxFlow(s, t);
    const vis = mf.reachableFrom(s);

    const cuts = [];
    let burned = 0;

    for (let r = 0; r < R; r++) {
      if (vis[2 * r] && !vis[2 * r + 1]) cuts.push(r);
    }

    for (let i = 0; i < townRelIds.length; i++) {
      const r = townRelIds[i];
      if (vis[2 * r + 1]) burned += townRelVals[i];
    }

    const result = {
      lambda,
      k: cuts.length,
      cuts,
      saved: totalProtectable - burned,
    };
    cache.set(key, result);
    return result;
  }

  const highLambda = totalProtectable + 1;
  let low = 0;
  let high = highLambda;

  const baseLow = solveLambda(low);
  const baseHigh = solveLambda(high);

  if (baseLow.k <= budget) {
    candidates.push(normalizeCuts(baseLow.cuts));
  } else {
    const iter = R <= 10000 ? 8 : R <= 18000 ? 6 : 4;
    for (let i = 0; i < iter; i++) {
      const mid = (low + high) * 0.5;
      const res = solveLambda(mid);
      if (res.k > budget) low = mid;
      else high = mid;
    }
  }

  const probe = [0, highLambda, low, high, (low + high) * 0.5, (2 * low + high) / 3, (low + 2 * high) / 3];
  for (let i = 0; i < probe.length; i++) {
    const lam = probe[i];
    if (!(lam >= 0) || !Number.isFinite(lam)) continue;
    const res = solveLambda(lam);
    if (res.k <= budget) {
      candidates.push(normalizeCuts(res.cuts));
    } else if (budget > 0) {
      candidates.push(topByScore(res.cuts, budget));
    }
  }
}

let bestCuts = [];
let bestSaved = -1;

for (let i = 0; i < candidates.length; i++) {
  const cuts = candidates[i];
  if (!cuts || cuts.length > budget) continue;
  const saved = simulateSaved(cuts);
  if (saved > bestSaved || (saved === bestSaved && cuts.length < bestCuts.length)) {
    bestSaved = saved;
    bestCuts = cuts;
  }
}

const firebreaks = new Array(bestCuts.length);
for (let i = 0; i < bestCuts.length; i++) {
  const g = relGlobal[bestCuts[i]];
  firebreaks[i] = [cellX[g], cellY[g]];
}

process.stdout.write(JSON.stringify({ firebreaks }));
