OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0_7 = OpConstant %float 0.699999988
%float_0_33333334 = OpConstant %float 0.333333343

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_gl_PerVertex_in_3 = OpTypePointer Input %arr_gl_PerVertex_in_3
%ptr_Input_v4float = OpTypePointer Input %v4float

%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3float_3 = OpTypePointer Input %arr_v3float_3
%ptr_Input_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_3 Input
%inWorldPos = OpVariable %ptr_Input_arr_v3float_3 Input
%inNormal = OpVariable %ptr_Input_arr_v3float_3 Input
%inColor = OpVariable %ptr_Input_arr_v3float_3 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%centroid = OpVectorTimesScalar %v4float %sum012 %float_0_33333334

%delta0 = OpFSub %v4float %p0 %centroid
%scaled0 = OpVectorTimesScalar %v4float %delta0 %float_0_7
%new0 = OpFAdd %v4float %centroid %scaled0
%gl_out_pos0 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_pos0 %new0
%color0_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_0
%color0 = OpLoad %v3float %color0_ptr
OpStore %outColor %color0
OpEmitVertex

%delta1 = OpFSub %v4float %p1 %centroid
%scaled1 = OpVectorTimesScalar %v4float %delta1 %float_0_7
%new1 = OpFAdd %v4float %centroid %scaled1
%gl_out_pos1 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_pos1 %new1
%color1_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_1
%color1 = OpLoad %v3float %color1_ptr
OpStore %outColor %color1
OpEmitVertex

%delta2 = OpFSub %v4float %p2 %centroid
%scaled2 = OpVectorTimesScalar %v4float %delta2 %float_0_7
%new2 = OpFAdd %v4float %centroid %scaled2
%gl_out_pos2 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_pos2 %new2
%color2_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_2
%color2 = OpLoad %v3float %color2_ptr
OpStore %outColor %color2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
