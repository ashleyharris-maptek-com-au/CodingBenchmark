use std::collections::HashMap;
use std::io::{self, Read};
use std::time::{Duration, Instant};

const MITM_MAX_N: usize = 46;
const BITSET_TARGET_MAX: usize = 6_000_000;
const BITSET_WORK_MAX: u128 = 320_000_000;
const SPARSE_N_MAX: usize = 6_000;
const SPARSE_STATE_MAX: usize = 2_500_000;
const HEURISTIC_BUDGET_MS: u64 = 1_200;

#[derive(Clone, Copy)]
struct Item {
    value: i64,
    index: usize,
}

#[derive(Clone, Copy)]
struct PosItem {
    value: u64,
    index: usize,
}

struct Scanner<'a> {
    data: &'a [u8],
    pos: usize,
}

impl<'a> Scanner<'a> {
    fn new(data: &'a [u8]) -> Self {
        Self { data, pos: 0 }
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.data.len() {
            return None;
        }

        let mut sign: i128 = 1;
        if self.data[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        } else if self.data[self.pos] == b'+' {
            self.pos += 1;
        }

        let mut value: i128 = 0;
        let mut any = false;
        while self.pos < self.data.len() {
            let b = self.data[self.pos];
            if b.is_ascii_digit() {
                any = true;
                value = value.saturating_mul(10).saturating_add((b - b'0') as i128);
                self.pos += 1;
            } else {
                break;
            }
        }

        if !any {
            return None;
        }

        let signed = value.saturating_mul(sign);
        if signed < i64::MIN as i128 || signed > i64::MAX as i128 {
            return None;
        }
        Some(signed as i64)
    }
}

struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let state = if seed == 0 { 0x9E37_79B9_7F4A_7C15 } else { seed };
        Self { state }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
}

fn main() {
    let mut input = Vec::new();
    if io::stdin().read_to_end(&mut input).is_err() {
        return;
    }

    let mut scanner = Scanner::new(&input);
    let n_raw = match scanner.next_i64() {
        Some(v) => v,
        None => return,
    };
    let target = match scanner.next_i64() {
        Some(v) => v,
        None => return,
    };

    if n_raw < 0 {
        println!("NO");
        return;
    }

    let n = n_raw as usize;
    let mut items = Vec::with_capacity(n);
    for idx in 0..n {
        match scanner.next_i64() {
            Some(v) => items.push(Item { value: v, index: idx }),
            None => break,
        }
    }

    match solve(&items, target) {
        Some(mut indices) => {
            indices.sort_unstable();
            println!("YES");
            let mut line = String::new();
            for (i, idx) in indices.iter().enumerate() {
                if i > 0 {
                    line.push(' ');
                }
                line.push_str(&idx.to_string());
            }
            println!("{line}");
        }
        None => {
            println!("NO");
        }
    }
}

fn solve(items: &[Item], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    if items.is_empty() {
        return None;
    }

    for item in items {
        if item.value == target {
            return Some(vec![item.index]);
        }
    }

    let mut min_sum: i128 = 0;
    let mut max_sum: i128 = 0;
    let mut all_nonneg = true;
    let mut all_nonpos = true;

    for item in items {
        if item.value > 0 {
            all_nonpos = false;
            max_sum += item.value as i128;
        } else if item.value < 0 {
            all_nonneg = false;
            min_sum += item.value as i128;
        }
    }

    let t = target as i128;
    if t < min_sum || t > max_sum {
        return None;
    }

    if items.len() <= MITM_MAX_N {
        return meet_in_middle(items, target);
    }

    if all_nonneg && target > 0 {
        let mut pos = Vec::with_capacity(items.len());
        for item in items {
            if item.value > 0 {
                pos.push(PosItem {
                    value: item.value as u64,
                    index: item.index,
                });
            }
        }
        return solve_nonnegative(&pos, target as u64);
    }

    if all_nonpos && target < 0 {
        let mut pos = Vec::with_capacity(items.len());
        for item in items {
            if item.value < 0 {
                if item.value == i64::MIN {
                    return solve_mixed(items, target);
                }
                pos.push(PosItem {
                    value: (-item.value) as u64,
                    index: item.index,
                });
            }
        }
        return solve_nonnegative(&pos, (-target) as u64);
    }

    solve_mixed(items, target)
}

fn solve_nonnegative(items: &[PosItem], target: u64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }

    let mut filtered = Vec::with_capacity(items.len());
    let mut total: u128 = 0;
    let mut g: u64 = 0;

    for item in items {
        if item.value == 0 {
            continue;
        }
        if item.value == target {
            return Some(vec![item.index]);
        }
        if item.value < target {
            filtered.push(*item);
            total += item.value as u128;
            g = gcd_u64(g, item.value);
        }
    }

    if filtered.is_empty() {
        return None;
    }
    if total < target as u128 {
        return None;
    }
    if g != 0 && target % g != 0 {
        return None;
    }

    if filtered.len() <= MITM_MAX_N {
        let mut tmp = Vec::with_capacity(filtered.len());
        for it in &filtered {
            tmp.push(Item {
                value: it.value as i64,
                index: it.index,
            });
        }
        return meet_in_middle(&tmp, target as i64);
    }

    if let Some(ans) = two_sum_exact(&filtered, target) {
        return Some(ans);
    }

    let words = (target as usize / 64) + 1;
    let work = (filtered.len() as u128) * (words as u128);
    if target as usize <= BITSET_TARGET_MAX && work <= BITSET_WORK_MAX {
        if let Some(ans) = bitset_dp_with_parents(&filtered, target as usize) {
            return Some(ans);
        }
        return None;
    }

    if filtered.len() <= SPARSE_N_MAX {
        if let Some(ans) = sparse_dp_nonnegative(&filtered, target, SPARSE_STATE_MAX) {
            return Some(ans);
        }
    }

    heuristic_nonnegative(
        &filtered,
        target,
        Duration::from_millis(HEURISTIC_BUDGET_MS),
    )
}

fn solve_mixed(items: &[Item], target: i64) -> Option<Vec<usize>> {
    if items.len() <= MITM_MAX_N {
        return meet_in_middle(items, target);
    }

    if items.len() <= SPARSE_N_MAX {
        if let Some(ans) = sparse_dp_signed(items, target, SPARSE_STATE_MAX) {
            return Some(ans);
        }
    }

    None
}

fn gcd_u64(mut a: u64, mut b: u64) -> u64 {
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

fn meet_in_middle(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let n = items.len();
    let mid = n / 2;
    let left = &items[..mid];
    let right = &items[mid..];

    let left_count = 1usize << left.len();
    let right_count = 1usize << right.len();

    let mut left_sums: Vec<(i128, u64)> = Vec::with_capacity(left_count);
    for mask in 0..left_count {
        let mut sum: i128 = 0;
        for (bit, item) in left.iter().enumerate() {
            if ((mask >> bit) & 1) != 0 {
                sum += item.value as i128;
            }
        }
        left_sums.push((sum, mask as u64));
    }

    let mut right_sums: Vec<(i128, u64)> = Vec::with_capacity(right_count);
    for mask in 0..right_count {
        let mut sum: i128 = 0;
        for (bit, item) in right.iter().enumerate() {
            if ((mask >> bit) & 1) != 0 {
                sum += item.value as i128;
            }
        }
        right_sums.push((sum, mask as u64));
    }

    right_sums.sort_unstable_by_key(|x| x.0);
    let target128 = target as i128;

    for (left_sum, left_mask) in left_sums {
        let need = target128 - left_sum;
        if let Ok(pos) = right_sums.binary_search_by_key(&need, |x| x.0) {
            let right_mask = right_sums[pos].1;
            let mut indices = Vec::new();

            for (bit, item) in left.iter().enumerate() {
                if ((left_mask >> bit) & 1) != 0 {
                    indices.push(item.index);
                }
            }
            for (bit, item) in right.iter().enumerate() {
                if ((right_mask >> bit) & 1) != 0 {
                    indices.push(item.index);
                }
            }
            return Some(indices);
        }
    }

    None
}

fn two_sum_exact(items: &[PosItem], target: u64) -> Option<Vec<usize>> {
    let mut seen: HashMap<u64, usize> = HashMap::with_capacity(items.len().min(1_000_000));
    for item in items {
        if item.value > target {
            continue;
        }
        let need = target - item.value;
        if let Some(other_idx) = seen.get(&need) {
            if *other_idx != item.index {
                return Some(vec![*other_idx, item.index]);
            }
        }
        seen.entry(item.value).or_insert(item.index);
    }
    None
}

fn sparse_dp_nonnegative(items: &[PosItem], target: u64, state_limit: usize) -> Option<Vec<usize>> {
    let mut parent: HashMap<u64, (u64, usize)> = HashMap::new();
    parent.insert(0, (u64::MAX, usize::MAX));

    for item in items {
        let mut sums: Vec<u64> = parent.keys().copied().collect();
        sums.sort_unstable();

        for s in sums {
            if s > target || item.value > target - s {
                continue;
            }
            let ns = s + item.value;
            if let std::collections::hash_map::Entry::Vacant(slot) = parent.entry(ns) {
                slot.insert((s, item.index));
                if ns == target {
                    return Some(reconstruct_u64(&parent, target));
                }
            }
        }

        if parent.len() > state_limit {
            return None;
        }
    }

    if parent.contains_key(&target) {
        Some(reconstruct_u64(&parent, target))
    } else {
        None
    }
}

fn sparse_dp_signed(items: &[Item], target: i64, state_limit: usize) -> Option<Vec<usize>> {
    let mut parent: HashMap<i64, (i64, usize)> = HashMap::new();
    parent.insert(i64::MIN, (i64::MIN, usize::MAX));
    parent.insert(0, (i64::MIN, usize::MAX));

    for item in items {
        let sums: Vec<i64> = parent.keys().copied().filter(|s| *s != i64::MIN).collect();

        for s in sums {
            if let Some(ns) = s.checked_add(item.value) {
                if let std::collections::hash_map::Entry::Vacant(slot) = parent.entry(ns) {
                    slot.insert((s, item.index));
                    if ns == target {
                        return Some(reconstruct_i64(&parent, target));
                    }
                }
            }
        }

        if parent.len() > state_limit {
            return None;
        }
    }

    if parent.contains_key(&target) {
        Some(reconstruct_i64(&parent, target))
    } else {
        None
    }
}

fn reconstruct_u64(parent: &HashMap<u64, (u64, usize)>, mut sum: u64) -> Vec<usize> {
    let mut indices = Vec::new();
    while sum != 0 {
        let &(prev, idx) = parent
            .get(&sum)
            .expect("subset reconstruction failed for nonnegative DP");
        indices.push(idx);
        sum = prev;
    }
    indices
}

fn reconstruct_i64(parent: &HashMap<i64, (i64, usize)>, mut sum: i64) -> Vec<usize> {
    let mut indices = Vec::new();
    while sum != 0 {
        let &(prev, idx) = parent
            .get(&sum)
            .expect("subset reconstruction failed for signed DP");
        indices.push(idx);
        sum = prev;
    }
    indices
}

fn bitset_dp_with_parents(items: &[PosItem], target: usize) -> Option<Vec<usize>> {
    let words = (target / 64) + 1;
    let mut bits = vec![0u64; words];
    bits[0] = 1;

    let mut parent_idx = vec![usize::MAX; target + 1];
    let mut parent_prev = vec![u64::MAX; target + 1];
    parent_idx[0] = 0;
    parent_prev[0] = 0;

    for item in items {
        let shift_words = (item.value / 64) as usize;
        let shift_bits = (item.value % 64) as u32;
        if shift_words >= words {
            continue;
        }

        for w in (shift_words..words).rev() {
            let src = w - shift_words;
            let base = bits[src];
            let shifted = if shift_bits == 0 {
                base
            } else {
                let hi = base << shift_bits;
                let lo = if src > 0 {
                    bits[src - 1] >> (64 - shift_bits)
                } else {
                    0
                };
                hi | lo
            };

            let old = bits[w];
            let mut new_bits = shifted & !old;
            bits[w] = old | shifted;

            while new_bits != 0 {
                let tz = new_bits.trailing_zeros() as usize;
                let sum = w * 64 + tz;
                if sum <= target {
                    parent_idx[sum] = item.index;
                    parent_prev[sum] = (sum as u64) - item.value;
                }
                new_bits &= new_bits - 1;
            }
        }

        if ((bits[target / 64] >> (target % 64)) & 1) != 0 {
            break;
        }
    }

    if ((bits[target / 64] >> (target % 64)) & 1) == 0 {
        return None;
    }

    let mut indices = Vec::new();
    let mut sum = target as u64;
    while sum != 0 {
        let s = sum as usize;
        let idx = parent_idx[s];
        let prev = parent_prev[s];
        if idx == usize::MAX || prev == u64::MAX {
            return None;
        }
        indices.push(idx);
        sum = prev;
    }

    Some(indices)
}

fn heuristic_nonnegative(items: &[PosItem], target: u64, budget: Duration) -> Option<Vec<usize>> {
    let mut sorted = items.to_vec();
    sorted.sort_unstable_by(|a, b| b.value.cmp(&a.value));

    if let Some(ans) = greedy_once(&sorted, target, true) {
        return Some(ans);
    }

    let mut best_residual = u64::MAX;
    let mut best_indices: Vec<usize> = Vec::new();
    let mut rng = XorShift64::new((target ^ (items.len() as u64).wrapping_mul(0xA24B_AED4_963E_E407)).wrapping_add(1));
    let start = Instant::now();

    while start.elapsed() < budget {
        let take_large_bias = (rng.next_u64() & 1) == 0;
        let mut sum = 0u64;
        let mut picked = Vec::new();

        for item in &sorted {
            if item.value > target - sum {
                continue;
            }

            if sum + item.value == target {
                picked.push(item.index);
                return Some(picked);
            }

            let coin = (rng.next_u64() % 100) as u32;
            let remaining = target - sum;
            let pick = if take_large_bias {
                coin < 55 || item.value > remaining / 2
            } else {
                coin < 45 || item.value > remaining / 3
            };

            if pick {
                sum += item.value;
                picked.push(item.index);
            }
        }

        if sum == target {
            return Some(picked);
        }

        let residual = target - sum;
        if residual < best_residual {
            best_residual = residual;
            best_indices = picked;
        }
    }

    let _ = best_indices;
    None
}

fn greedy_once(sorted: &[PosItem], target: u64, deterministic: bool) -> Option<Vec<usize>> {
    let mut sum = 0u64;
    let mut indices = Vec::new();

    for item in sorted {
        if item.value > target - sum {
            continue;
        }
        sum += item.value;
        indices.push(item.index);
        if sum == target {
            return Some(indices);
        }
        if !deterministic && sum > target {
            break;
        }
    }

    None
}
