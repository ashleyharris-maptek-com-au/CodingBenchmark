use std::io::{self, Read, Write};

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data)?;
        Ok(Self { data, idx: 0 })
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.data.len() {
            return None;
        }

        let mut value: usize = 0;
        let mut found_digit = false;
        while self.idx < self.data.len() {
            let b = self.data[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            found_digit = true;
            value = value.saturating_mul(10).saturating_add((b - b'0') as usize);
            self.idx += 1;
        }

        if !found_digit {
            None
        } else {
            Some(value)
        }
    }
}

fn build_initial_matching_cover(
    n: usize,
    edges_u: &[u32],
    edges_v: &[u32],
    forced: &[u8],
) -> Vec<u8> {
    let mut selected = vec![0u8; n];
    selected.copy_from_slice(forced);

    for i in 0..edges_u.len() {
        let u = edges_u[i] as usize;
        let v = edges_v[i] as usize;

        if u == v {
            selected[u] = 1;
            continue;
        }

        if selected[u] == 0 && selected[v] == 0 {
            selected[u] = 1;
            selected[v] = 1;
        }
    }

    selected
}

fn build_initial_degree_cover(
    n: usize,
    edges_u: &[u32],
    edges_v: &[u32],
    deg: &[u32],
    forced: &[u8],
) -> Vec<u8> {
    let mut selected = vec![0u8; n];
    selected.copy_from_slice(forced);

    for i in 0..edges_u.len() {
        let u = edges_u[i] as usize;
        let v = edges_v[i] as usize;

        if u == v {
            selected[u] = 1;
            continue;
        }

        if selected[u] == 0 && selected[v] == 0 {
            if deg[u] >= deg[v] {
                selected[u] = 1;
            } else {
                selected[v] = 1;
            }
        }
    }

    selected
}

fn prune_cover(
    edges_u: &[u32],
    edges_v: &[u32],
    offsets: &[usize],
    adj_edges: &[u32],
    selected: &mut [u8],
) -> usize {
    let n = selected.len();
    let m = edges_u.len();

    let mut selected_count = selected.iter().map(|&x| x as usize).sum::<usize>();
    let mut edge_cov = vec![0u8; m];
    let mut unique = vec![0u32; n];

    for i in 0..m {
        let u = edges_u[i] as usize;
        let v = edges_v[i] as usize;

        let cov = if u == v {
            selected[u]
        } else {
            selected[u] + selected[v]
        };
        edge_cov[i] = cov;

        if cov == 1 {
            if u == v {
                unique[u] = unique[u].saturating_add(1);
            } else if selected[u] == 1 {
                unique[u] = unique[u].saturating_add(1);
            } else {
                unique[v] = unique[v].saturating_add(1);
            }
        }
    }

    let mut stack = Vec::new();
    for v in 0..n {
        if selected[v] == 1 && unique[v] == 0 {
            stack.push(v);
        }
    }

    while let Some(v) = stack.pop() {
        if selected[v] == 0 || unique[v] != 0 {
            continue;
        }

        selected[v] = 0;
        selected_count -= 1;

        let start = offsets[v];
        let end = offsets[v + 1];
        for idx in start..end {
            let e = adj_edges[idx] as usize;

            if edge_cov[e] == 2 {
                edge_cov[e] = 1;

                let a = edges_u[e] as usize;
                let b = edges_v[e] as usize;
                let other = if a == v { b } else { a };
                if selected[other] == 1 {
                    unique[other] = unique[other].saturating_add(1);
                }
            }
        }
    }

    selected_count
}

fn is_valid_cover(edges_u: &[u32], edges_v: &[u32], selected: &[u8]) -> bool {
    for i in 0..edges_u.len() {
        let u = edges_u[i] as usize;
        let v = edges_v[i] as usize;
        if selected[u] == 0 && selected[v] == 0 {
            return false;
        }
    }
    true
}

fn main() {
    let mut input = match FastInput::new() {
        Ok(v) => v,
        Err(_) => return,
    };

    let n = match input.next_usize() {
        Some(v) => v,
        None => {
            let stdout = io::stdout();
            let mut out = io::BufWriter::new(stdout.lock());
            if writeln!(out, "0").is_err() {
                return;
            }
            let _ = writeln!(out);
            return;
        }
    };

    let m_input = match input.next_usize() {
        Some(v) => v,
        None => {
            let stdout = io::stdout();
            let mut out = io::BufWriter::new(stdout.lock());
            if writeln!(out, "0").is_err() {
                return;
            }
            let _ = writeln!(out);
            return;
        }
    };

    let mut edges_u: Vec<u32> = Vec::with_capacity(m_input);
    let mut edges_v: Vec<u32> = Vec::with_capacity(m_input);
    let mut deg = vec![0u32; n];
    let mut forced = vec![0u8; n];

    for _ in 0..m_input {
        let u_opt = input.next_usize();
        let v_opt = input.next_usize();

        let (u, v) = match (u_opt, v_opt) {
            (Some(uu), Some(vv)) => (uu, vv),
            _ => break,
        };

        if u >= n || v >= n {
            continue;
        }

        edges_u.push(u as u32);
        edges_v.push(v as u32);

        if u == v {
            forced[u] = 1;
            deg[u] = deg[u].saturating_add(1);
        } else {
            deg[u] = deg[u].saturating_add(1);
            deg[v] = deg[v].saturating_add(1);
        }
    }

    let m = edges_u.len();

    if m == 0 {
        let stdout = io::stdout();
        let mut out = io::BufWriter::new(stdout.lock());
        if writeln!(out, "0").is_err() {
            return;
        }
        let _ = writeln!(out);
        return;
    }

    let mut offsets = vec![0usize; n + 1];
    for v in 0..n {
        offsets[v + 1] = offsets[v] + deg[v] as usize;
    }

    let mut adj_edges = vec![0u32; offsets[n]];
    let mut cursor = offsets[..n].to_vec();

    for e in 0..m {
        let u = edges_u[e] as usize;
        let v = edges_v[e] as usize;

        let cu = cursor[u];
        adj_edges[cu] = e as u32;
        cursor[u] += 1;

        if u != v {
            let cv = cursor[v];
            adj_edges[cv] = e as u32;
            cursor[v] += 1;
        }
    }

    let mut cover_match = build_initial_matching_cover(n, &edges_u, &edges_v, &forced);
    let size_match = prune_cover(&edges_u, &edges_v, &offsets, &adj_edges, &mut cover_match);
    let valid_match = is_valid_cover(&edges_u, &edges_v, &cover_match);

    let mut cover_degree = build_initial_degree_cover(n, &edges_u, &edges_v, &deg, &forced);
    let size_degree = prune_cover(&edges_u, &edges_v, &offsets, &adj_edges, &mut cover_degree);
    let valid_degree = is_valid_cover(&edges_u, &edges_v, &cover_degree);

    let final_cover = if valid_match {
        if valid_degree && size_degree < size_match {
            cover_degree
        } else {
            cover_match
        }
    } else if valid_degree {
        cover_degree
    } else {
        vec![1u8; n]
    };

    let final_size = final_cover.iter().map(|&x| x as usize).sum::<usize>();

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());

    if writeln!(out, "{}", final_size).is_err() {
        return;
    }

    if final_size > 0 {
        let mut first = true;
        for (v, &pick) in final_cover.iter().enumerate() {
            if pick == 1 {
                if !first && write!(out, " ").is_err() {
                    return;
                }
                if write!(out, "{}", v).is_err() {
                    return;
                }
                first = false;
            }
        }
    }

    let _ = writeln!(out);
}
