declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type JobId = number | string;

interface RobotInput {
  id: number;
  x: number;
  y: number;
}

interface JobInput {
  id: JobId;
  x: number;
  y: number;
  priority: number;
}

interface RobotState {
  id: number;
  x: number;
  y: number;
  posIdx: number;
}

const grid: string[] = Array.isArray(data.grid) ? data.grid.map((row: unknown) => String(row)) : [];
const robotsInput: RobotInput[] = Array.isArray(data.robots)
  ? data.robots.map((r: any) => ({ id: Number(r?.id), x: Number(r?.x), y: Number(r?.y) }))
  : [];
const jobsInput: JobInput[] = Array.isArray(data.jobs)
  ? data.jobs.map((j: any) => ({
      id: j?.id as JobId,
      x: Number(j?.x),
      y: Number(j?.y),
      priority: Number(j?.priority),
    }))
  : [];

const height = grid.length;
let width = 0;
for (const row of grid) {
  if (row.length > width) width = row.length;
}
const cellCount = width * height;

const open = new Uint8Array(cellCount);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    if (x < row.length && row.charCodeAt(x) === 46) {
      open[y * width + x] = 1;
    }
  }
}

const inBounds = (x: number, y: number): boolean => x >= 0 && y >= 0 && x < width && y < height;
const toIndex = (x: number, y: number): number => y * width + x;

function cellIndexIfOpen(x: number, y: number): number {
  if (!inBounds(x, y)) return -1;
  const idx = toIndex(x, y);
  return open[idx] === 1 ? idx : -1;
}

function compareIds(a: JobId, b: JobId): number {
  if (typeof a === 'number' && typeof b === 'number') {
    return a - b;
  }
  const as = String(a);
  const bs = String(b);
  if (as < bs) return -1;
  if (as > bs) return 1;
  return 0;
}

const robots: RobotState[] = robotsInput
  .filter((r) => Number.isFinite(r.id) && Number.isFinite(r.x) && Number.isFinite(r.y))
  .sort((a, b) => a.id - b.id)
  .map((r) => ({
    id: Math.trunc(r.id),
    x: Math.trunc(r.x),
    y: Math.trunc(r.y),
    posIdx: cellIndexIfOpen(Math.trunc(r.x), Math.trunc(r.y)),
  }));

const maxRobotId = robots.reduce((mx, r) => (r.id > mx ? r.id : mx), -1);
const assignments: JobId[][] = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);

const jobs: JobInput[] = jobsInput
  .filter((j) => Number.isFinite(j.x) && Number.isFinite(j.y) && Number.isFinite(j.priority))
  .map((j) => ({ id: j.id, x: Math.trunc(j.x), y: Math.trunc(j.y), priority: j.priority }))
  .sort((a, b) => {
    if (a.priority !== b.priority) return b.priority - a.priority;
    const idCmp = compareIds(a.id, b.id);
    if (idCmp !== 0) return idCmp;
    if (a.y !== b.y) return a.y - b.y;
    return a.x - b.x;
  });

const canRoute = robots.length > 0 && width > 0 && height > 0;
const seen = new Int32Array(cellCount);
const dist = new Int32Array(cellCount);
const queue = new Int32Array(cellCount);
let stamp = 1;

function nextStamp(): number {
  stamp += 1;
  if (stamp === 2147483647) {
    seen.fill(0);
    stamp = 1;
  }
  return stamp;
}

function bfsFromTargetToRobotCells(startIdx: number, targetCells: number[]): Map<number, number> {
  const found = new Map<number, number>();
  if (startIdx < 0 || startIdx >= cellCount || open[startIdx] === 0 || targetCells.length === 0) {
    return found;
  }

  const uniqueTargets = new Set<number>();
  for (const cell of targetCells) {
    if (cell >= 0 && cell < cellCount) uniqueTargets.add(cell);
  }
  if (uniqueTargets.size === 0) return found;

  const currentStamp = nextStamp();
  let head = 0;
  let tail = 0;

  queue[tail++] = startIdx;
  seen[startIdx] = currentStamp;
  dist[startIdx] = 0;

  if (uniqueTargets.has(startIdx)) {
    found.set(startIdx, 0);
    uniqueTargets.delete(startIdx);
  }

  while (head < tail && uniqueTargets.size > 0) {
    const cur = queue[head++];
    const baseDist = dist[cur] + 1;

    const up = cur - width;
    if (up >= 0 && open[up] === 1 && seen[up] !== currentStamp) {
      seen[up] = currentStamp;
      dist[up] = baseDist;
      queue[tail++] = up;
      if (uniqueTargets.has(up)) {
        found.set(up, baseDist);
        uniqueTargets.delete(up);
      }
    }

    const down = cur + width;
    if (down < cellCount && open[down] === 1 && seen[down] !== currentStamp) {
      seen[down] = currentStamp;
      dist[down] = baseDist;
      queue[tail++] = down;
      if (uniqueTargets.has(down)) {
        found.set(down, baseDist);
        uniqueTargets.delete(down);
      }
    }

    if (cur % width !== 0) {
      const left = cur - 1;
      if (open[left] === 1 && seen[left] !== currentStamp) {
        seen[left] = currentStamp;
        dist[left] = baseDist;
        queue[tail++] = left;
        if (uniqueTargets.has(left)) {
          found.set(left, baseDist);
          uniqueTargets.delete(left);
        }
      }
    }

    if (cur % width !== width - 1) {
      const right = cur + 1;
      if (open[right] === 1 && seen[right] !== currentStamp) {
        seen[right] = currentStamp;
        dist[right] = baseDist;
        queue[tail++] = right;
        if (uniqueTargets.has(right)) {
          found.set(right, baseDist);
          uniqueTargets.delete(right);
        }
      }
    }
  }

  return found;
}

function manhattan(ax: number, ay: number, bx: number, by: number): number {
  return Math.abs(ax - bx) + Math.abs(ay - by);
}

if (canRoute) {
  for (const job of jobs) {
    const jobIdx = cellIndexIfOpen(job.x, job.y);
    const robotCells = robots.map((r) => r.posIdx);
    const distancesByCell = bfsFromTargetToRobotCells(jobIdx, robotCells);

    let bestRobotIndex = 0;
    let bestDistance = Number.POSITIVE_INFINITY;

    for (let i = 0; i < robots.length; i++) {
      const robot = robots[i];
      const d = robot.posIdx >= 0 ? distancesByCell.get(robot.posIdx) : undefined;
      const distance = d === undefined ? Number.POSITIVE_INFINITY : d;

      if (
        distance < bestDistance ||
        (distance === bestDistance && robots[i].id < robots[bestRobotIndex].id)
      ) {
        bestDistance = distance;
        bestRobotIndex = i;
      }
    }

    if (!Number.isFinite(bestDistance)) {
      let fallbackIndex = 0;
      let fallbackCost = Number.POSITIVE_INFINITY;
      for (let i = 0; i < robots.length; i++) {
        const r = robots[i];
        const cost = manhattan(r.x, r.y, job.x, job.y);
        if (cost < fallbackCost || (cost === fallbackCost && r.id < robots[fallbackIndex].id)) {
          fallbackCost = cost;
          fallbackIndex = i;
        }
      }
      bestRobotIndex = fallbackIndex;
    }

    const chosen = robots[bestRobotIndex];
    if (chosen.id >= 0 && chosen.id < assignments.length) {
      assignments[chosen.id].push(job.id);
    }

    chosen.x = job.x;
    chosen.y = job.y;
    chosen.posIdx = jobIdx;
  }
}

const output = { assignments };
fs.writeSync(1, JSON.stringify(output));
