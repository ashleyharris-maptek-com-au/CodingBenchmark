struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;
};

struct Plane {
    Vec3 n;
    double d = 0.0;
};

struct BBox {
    Vec3 mn;
    Vec3 mx;
};

static inline Vec3 operator+(const Vec3& a, const Vec3& b) {
    return {a.x + b.x, a.y + b.y, a.z + b.z};
}

static inline Vec3 operator-(const Vec3& a, const Vec3& b) {
    return {a.x - b.x, a.y - b.y, a.z - b.z};
}

static inline Vec3 operator*(const Vec3& a, double s) {
    return {a.x * s, a.y * s, a.z * s};
}

static inline Vec3 operator/(const Vec3& a, double s) {
    return {a.x / s, a.y / s, a.z / s};
}

static inline Vec3& operator+=(Vec3& a, const Vec3& b) {
    a.x += b.x;
    a.y += b.y;
    a.z += b.z;
    return a;
}

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double norm2(const Vec3& v) {
    return dot(v, v);
}

static inline double norm(const Vec3& v) {
    return std::sqrt(norm2(v));
}

static BBox computeBBox(const std::vector<Vec3>& pts) {
    const double inf = std::numeric_limits<double>::infinity();
    BBox box{{inf, inf, inf}, {-inf, -inf, -inf}};
    for (const auto& p : pts) {
        box.mn.x = std::min(box.mn.x, p.x);
        box.mn.y = std::min(box.mn.y, p.y);
        box.mn.z = std::min(box.mn.z, p.z);
        box.mx.x = std::max(box.mx.x, p.x);
        box.mx.y = std::max(box.mx.y, p.y);
        box.mx.z = std::max(box.mx.z, p.z);
    }
    return box;
}

static double diagonalLength(const BBox& box) {
    return norm(box.mx - box.mn);
}

static std::vector<Vec3> pickRepresentativePoints(const std::vector<Vec3>& pts) {
    const std::size_t n = pts.size();
    if (n <= 140) {
        return pts;
    }

    std::vector<char> used(n, 0);
    std::vector<std::size_t> ids;
    ids.reserve(220);

    auto add = [&](std::size_t idx) {
        if (!used[idx]) {
            used[idx] = 1;
            ids.push_back(idx);
        }
    };

    std::size_t minX = 0, maxX = 0;
    std::size_t minY = 0, maxY = 0;
    std::size_t minZ = 0, maxZ = 0;

    for (std::size_t i = 1; i < n; ++i) {
        if (pts[i].x < pts[minX].x) minX = i;
        if (pts[i].x > pts[maxX].x) maxX = i;
        if (pts[i].y < pts[minY].y) minY = i;
        if (pts[i].y > pts[maxY].y) maxY = i;
        if (pts[i].z < pts[minZ].z) minZ = i;
        if (pts[i].z > pts[maxZ].z) maxZ = i;
    }

    add(minX); add(maxX);
    add(minY); add(maxY);
    add(minZ); add(maxZ);

    Vec3 centroid{};
    for (const auto& p : pts) {
        centroid += p;
    }
    centroid = centroid / static_cast<double>(n);

    int dirCount = 36;
    if (n < 5000) dirCount = 84;
    else if (n < 50000) dirCount = 56;

    const double goldenAngle = 2.39996322972865332223;
    for (int k = 0; k < dirCount; ++k) {
        const double z = 1.0 - 2.0 * ((static_cast<double>(k) + 0.5) / static_cast<double>(dirCount));
        const double r = std::sqrt(std::max(0.0, 1.0 - z * z));
        const double phi = goldenAngle * static_cast<double>(k);
        const Vec3 d{r * std::cos(phi), r * std::sin(phi), z};

        std::size_t maxIdx = 0;
        std::size_t minIdx = 0;
        double maxVal = -std::numeric_limits<double>::infinity();
        double minVal =  std::numeric_limits<double>::infinity();

        for (std::size_t i = 0; i < n; ++i) {
            const double v = dot(pts[i] - centroid, d);
            if (v > maxVal) {
                maxVal = v;
                maxIdx = i;
            }
            if (v < minVal) {
                minVal = v;
                minIdx = i;
            }
        }

        add(maxIdx);
        add(minIdx);
    }

    constexpr std::size_t kMaxRep = 170;
    if (ids.size() > kMaxRep) {
        std::vector<std::pair<double, std::size_t>> scored;
        scored.reserve(ids.size());
        for (std::size_t idx : ids) {
            scored.push_back({norm2(pts[idx] - centroid), idx});
        }
        std::sort(scored.begin(), scored.end(), [](const auto& a, const auto& b) {
            return a.first > b.first;
        });

        std::vector<char> keep(n, 0);
        std::vector<std::size_t> trimmed;
        trimmed.reserve(kMaxRep);

        auto addKeep = [&](std::size_t idx) {
            if (!keep[idx] && trimmed.size() < kMaxRep) {
                keep[idx] = 1;
                trimmed.push_back(idx);
            }
        };

        addKeep(minX); addKeep(maxX);
        addKeep(minY); addKeep(maxY);
        addKeep(minZ); addKeep(maxZ);

        for (const auto& item : scored) {
            if (trimmed.size() >= kMaxRep) break;
            addKeep(item.second);
        }

        ids.swap(trimmed);
    }

    std::vector<Vec3> out;
    out.reserve(ids.size());
    for (std::size_t idx : ids) {
        out.push_back(pts[idx]);
    }
    return out;
}

static bool almostSamePlane(const Plane& a, const Plane& b, double normalTol, double dTol) {
    const double align = dot(a.n, b.n);
    return align > (1.0 - normalTol) && std::fabs(a.d - b.d) <= dTol;
}

static std::vector<Plane> buildConvexPlanes(const std::vector<Vec3>& pts) {
    std::vector<Plane> planes;
    const std::size_t n = pts.size();
    if (n < 4) {
        return planes;
    }

    const BBox box = computeBBox(pts);
    const double scale = std::max(1.0, diagonalLength(box));
    const double eps = 1e-9 * scale;
    const double dTol = 1e-7 * scale;
    const double normalTol = 1e-6;

    planes.reserve(256);

    for (std::size_t i = 0; i + 2 < n; ++i) {
        for (std::size_t j = i + 1; j + 1 < n; ++j) {
            for (std::size_t k = j + 1; k < n; ++k) {
                Vec3 normal = cross(pts[j] - pts[i], pts[k] - pts[i]);
                const double len = norm(normal);
                if (len <= eps) {
                    continue;
                }
                normal = normal / len;
                double d = dot(normal, pts[i]);

                bool seenPos = false;
                bool seenNeg = false;
                double maxSide = -std::numeric_limits<double>::infinity();

                for (std::size_t t = 0; t < n; ++t) {
                    const double s = dot(normal, pts[t]) - d;
                    maxSide = std::max(maxSide, s);
                    if (s > eps) seenPos = true;
                    if (s < -eps) seenNeg = true;
                    if (seenPos && seenNeg) break;
                }

                if (seenPos && seenNeg) {
                    continue;
                }

                if (maxSide > eps) {
                    normal = normal * -1.0;
                    d = -d;
                }

                const Plane candidate{normal, d};
                bool duplicate = false;
                for (const auto& p : planes) {
                    if (almostSamePlane(candidate, p, normalTol, dTol)) {
                        duplicate = true;
                        break;
                    }
                }
                if (!duplicate) {
                    planes.push_back(candidate);
                }
            }
        }
    }

    return planes;
}

static bool insideShrunkenPoly(const Vec3& c, const std::vector<Plane>& planes, double shrink, double eps) {
    for (const auto& pl : planes) {
        if (dot(pl.n, c) > (pl.d - shrink + eps)) {
            return false;
        }
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V = 0;
    double edge = 0.0;
    if (!(std::cin >> V >> edge)) {
        std::cout << 0 << '\n';
        return 0;
    }

    if (V <= 0 || edge <= 0.0) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::vector<Vec3> vertices;
    vertices.resize(static_cast<std::size_t>(V));
    for (int i = 0; i < V; ++i) {
        std::cin >> vertices[static_cast<std::size_t>(i)].x
                 >> vertices[static_cast<std::size_t>(i)].y
                 >> vertices[static_cast<std::size_t>(i)].z;
    }

    if (vertices.size() < 4) {
        std::cout << 0 << '\n';
        return 0;
    }

    const auto tStart = std::chrono::steady_clock::now();
    const auto tDeadline = tStart + std::chrono::milliseconds(28500);

    const std::vector<Vec3> reduced = pickRepresentativePoints(vertices);
    std::vector<Plane> planes = buildConvexPlanes(reduced);

    if (planes.size() < 4 && reduced.size() < vertices.size() && vertices.size() <= 500) {
        planes = buildConvexPlanes(vertices);
    }

    if (planes.size() < 4) {
        std::cout << 0 << '\n';
        return 0;
    }

    const BBox box = computeBBox(reduced);
    const double boxScale = std::max(1.0, diagonalLength(box));

    // Circumradius of regular tetrahedron with edge length e.
    const double R = edge * std::sqrt(3.0 / 8.0);
    const double safety = 1.00001;
    const double inflatedR = R * safety;

    // FCC lattice with nearest-neighbor distance 2R.
    const double fccBaseStep = 2.0 * std::sqrt(2.0) * inflatedR;
    if (!(fccBaseStep > 0.0)) {
        std::cout << 0 << '\n';
        return 0;
    }

    const double dx = std::max(0.0, box.mx.x - box.mn.x);
    const double dy = std::max(0.0, box.mx.y - box.mn.y);
    const double dz = std::max(0.0, box.mx.z - box.mn.z);

    const double approxRawCandidates = 4.0 *
        (dx / fccBaseStep + 1.0) *
        (dy / fccBaseStep + 1.0) *
        (dz / fccBaseStep + 1.0);

    constexpr double kTargetCandidates = 2400000.0;
    double scaleFactor = 1.0;
    if (approxRawCandidates > kTargetCandidates) {
        scaleFactor = std::cbrt(approxRawCandidates / kTargetCandidates);
    }

    const double step = fccBaseStep * scaleFactor;
    if (!(step > 0.0)) {
        std::cout << 0 << '\n';
        return 0;
    }

    const double epsInside = 1e-9 * boxScale;

    constexpr std::size_t kHardCap = 250000;
    std::vector<Vec3> bestCenters;
    bestCenters.reserve(65536);

    std::array<Vec3, 4> shiftFractions{{
        {0.00, 0.00, 0.00},
        {0.25, 0.25, 0.25},
        {0.50, 0.00, 0.50},
        {0.00, 0.50, 0.50}
    }};

    int shiftsToTry = (approxRawCandidates <= 700000.0) ? 4 : 1;

    const std::array<Vec3, 4> basis{{
        {0.0, 0.0, 0.0},
        {0.0, 0.5, 0.5},
        {0.5, 0.0, 0.5},
        {0.5, 0.5, 0.0}
    }};

    for (int s = 0; s < shiftsToTry; ++s) {
        if (std::chrono::steady_clock::now() >= tDeadline) break;

        const Vec3 shift = {
            shiftFractions[static_cast<std::size_t>(s)].x * step,
            shiftFractions[static_cast<std::size_t>(s)].y * step,
            shiftFractions[static_cast<std::size_t>(s)].z * step
        };

        const Vec3 origin = {box.mn.x + shift.x, box.mn.y + shift.y, box.mn.z + shift.z};

        const int ix0 = static_cast<int>(std::floor((box.mn.x - origin.x) / step)) - 1;
        const int iy0 = static_cast<int>(std::floor((box.mn.y - origin.y) / step)) - 1;
        const int iz0 = static_cast<int>(std::floor((box.mn.z - origin.z) / step)) - 1;

        const int ix1 = static_cast<int>(std::ceil((box.mx.x - origin.x) / step)) + 1;
        const int iy1 = static_cast<int>(std::ceil((box.mx.y - origin.y) / step)) + 1;
        const int iz1 = static_cast<int>(std::ceil((box.mx.z - origin.z) / step)) + 1;

        std::vector<Vec3> current;
        current.reserve(bestCenters.size());

        std::uint64_t iterCounter = 0;
        bool timeUp = false;

        for (int ix = ix0; ix <= ix1 && !timeUp; ++ix) {
            const double bx = origin.x + step * static_cast<double>(ix);
            for (int iy = iy0; iy <= iy1 && !timeUp; ++iy) {
                const double by = origin.y + step * static_cast<double>(iy);
                for (int iz = iz0; iz <= iz1; ++iz) {
                    const double bz = origin.z + step * static_cast<double>(iz);

                    for (const auto& b : basis) {
                        const Vec3 c{bx + step * b.x, by + step * b.y, bz + step * b.z};
                        if (insideShrunkenPoly(c, planes, inflatedR, epsInside)) {
                            current.push_back(c);
                            if (current.size() >= kHardCap) {
                                timeUp = true;
                                break;
                            }
                        }
                    }

                    ++iterCounter;
                    if ((iterCounter & 0x3FFFu) == 0u) {
                        if (std::chrono::steady_clock::now() >= tDeadline) {
                            timeUp = true;
                            break;
                        }
                    }
                }
            }
        }

        if (current.size() > bestCenters.size()) {
            bestCenters.swap(current);
        }
    }

    std::cout << bestCenters.size() << '\n';
    std::cout << std::fixed << std::setprecision(9);
    for (const auto& c : bestCenters) {
        // We use one fixed orientation; geometric validity comes from circumsphere-based constraints.
        std::cout << c.x << ' ' << c.y << ' ' << c.z << ' '
                  << 0.0 << ' ' << 0.0 << ' ' << 0.0 << '\n';
    }

    return 0;
}
