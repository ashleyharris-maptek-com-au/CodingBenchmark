; SPIR-V
; Version: 1.0
OpCapability Shader
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1_5 = OpConstant %float 1.5
%float_3 = OpConstant %float 3
%baseColor = OpConstantComposite %v3float %float_0_8 %float_0_2 %float_0_2
%fogColor = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_8
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%_ptr_Input_v3float = OpTypePointer Input %v3float
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%_ptr_Output_v4float = OpTypePointer Output %v4float
%outColor = OpVariable %_ptr_Output_v4float Output
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%main = OpFunction %void None %fn
%entry = OpLabel
%n_in = OpLoad %v3float %normal
%n = OpExtInst %v3float %ext Normalize %n_in
%wp = OpLoad %v3float %worldPos
%lightPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos4 = OpLoad %v4float %lightPosPtr
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%lvec = OpFSub %v3float %lightPos3 %wp
%l = OpExtInst %v3float %ext Normalize %lvec
%ndotl = OpDot %float %n %l
%diffuse = OpExtInst %float %ext FMax %ndotl %float_0
%scaledDiffuse = OpFMul %float %float_0_7 %diffuse
%lambert = OpFAdd %float %float_0_3 %scaledDiffuse
%baseLit = OpVectorTimesScalar %v3float %baseColor %lambert
%camPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%camPos4 = OpLoad %v4float %camPosPtr
%camPos3 = OpVectorShuffle %v3float %camPos4 %camPos4 0 1 2
%viewVec = OpFSub %v3float %camPos3 %wp
%dist = OpExtInst %float %ext Length %viewVec
%distShift = OpFSub %float %dist %float_1_5
%fogRaw = OpFDiv %float %distShift %float_3
%fogFactor = OpExtInst %float %ext FClamp %fogRaw %float_0 %float_1
%invFog = OpFSub %float %float_1 %fogFactor
%basePart = OpVectorTimesScalar %v3float %baseLit %invFog
%fogPart = OpVectorTimesScalar %v3float %fogColor %fogFactor
%finalRgb = OpFAdd %v3float %basePart %fogPart
%r = OpCompositeExtract %float %finalRgb 0
%g = OpCompositeExtract %float %finalRgb 1
%b = OpCompositeExtract %float %finalRgb 2
%finalColor = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
