use std::cmp::Ordering;
use std::collections::{HashMap, HashSet};
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
}

#[derive(Clone, Copy, Debug)]
struct Aabb {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
}

impl Aabb {
    fn empty() -> Self {
        Self {
            minx: f64::INFINITY,
            miny: f64::INFINITY,
            maxx: f64::NEG_INFINITY,
            maxy: f64::NEG_INFINITY,
        }
    }

    fn from_points(pts: &[Pt]) -> Self {
        let mut b = Self::empty();
        for &p in pts {
            if p.x < b.minx {
                b.minx = p.x;
            }
            if p.y < b.miny {
                b.miny = p.y;
            }
            if p.x > b.maxx {
                b.maxx = p.x;
            }
            if p.y > b.maxy {
                b.maxy = p.y;
            }
        }
        b
    }

    fn shift(&self, tx: f64, ty: f64) -> Self {
        Self {
            minx: self.minx + tx,
            miny: self.miny + ty,
            maxx: self.maxx + tx,
            maxy: self.maxy + ty,
        }
    }

    fn overlaps(&self, other: &Self) -> bool {
        !(self.maxx < other.minx - EPS
            || other.maxx < self.minx - EPS
            || self.maxy < other.miny - EPS
            || other.maxy < self.miny - EPS)
    }

    fn within(&self, outer: &Self) -> bool {
        self.minx >= outer.minx - EPS
            && self.miny >= outer.miny - EPS
            && self.maxx <= outer.maxx + EPS
            && self.maxy <= outer.maxy + EPS
    }

    fn width(&self) -> f64 {
        self.maxx - self.minx
    }

    fn height(&self) -> f64 {
        self.maxy - self.miny
    }
}

#[derive(Clone)]
struct Piece {
    id: usize,
    verts: Vec<Pt>,
    area: f64,
    centroid: Pt,
}

#[derive(Clone)]
struct Variant {
    angle: f64,
    verts: Vec<Pt>,
    bbox: Aabb,
    centroid: Pt,
    refs: Vec<Pt>,
}

#[derive(Clone)]
struct PlacedPoly {
    verts: Vec<Pt>,
    bbox: Aabb,
}

#[derive(Clone, Copy)]
struct Placement {
    stock: usize,
    tx: f64,
    ty: f64,
    rot: f64,
}

struct StockGeom {
    poly: Vec<Pt>,
    edges: Vec<(Pt, Pt)>,
    vertices: Vec<Pt>,
    bbox: Aabb,
    area: f64,
    convex: bool,
    orient_sign: f64,
    sample_points: Vec<Pt>,
}

struct StockState {
    placed: Vec<PlacedPoly>,
    anchors: Vec<Pt>,
    anchor_set: HashSet<(i64, i64)>,
    grid: HashMap<u64, Vec<usize>>,
    cell_size: f64,
    origin: Pt,
    used_area: f64,
    max_anchors: usize,
}

struct Params {
    base_angles: usize,
    stock_try_limit: usize,
    anchor_try: usize,
    max_anchors: usize,
    sample_grid: usize,
    random_samples: usize,
    max_refs: usize,
}

struct Scanner {
    data: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            data: input.into_bytes(),
            idx: 0,
        }
    }

    fn skip_ws(&mut self) {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }

    fn next_usize(&mut self) -> usize {
        self.skip_ws();
        let start = self.idx;
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        std::str::from_utf8(&self.data[start..self.idx])
            .unwrap()
            .parse::<usize>()
            .unwrap()
    }

    fn next_f64(&mut self) -> f64 {
        self.skip_ws();
        let start = self.idx;
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        std::str::from_utf8(&self.data[start..self.idx])
            .unwrap()
            .parse::<f64>()
            .unwrap()
    }
}

struct Lcg {
    state: u64,
}

impl Lcg {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }

    fn next_u64(&mut self) -> u64 {
        self.state = self
            .state
            .wrapping_mul(6364136223846793005)
            .wrapping_add(1442695040888963407);
        self.state
    }

    fn next_f64(&mut self) -> f64 {
        let v = self.next_u64() >> 11;
        (v as f64) * (1.0 / ((1u64 << 53) as f64))
    }
}

fn sub(a: Pt, b: Pt) -> Pt {
    Pt::new(a.x - b.x, a.y - b.y)
}

fn cross(a: Pt, b: Pt) -> f64 {
    a.x * b.y - a.y * b.x
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    cross(sub(b, a), sub(c, a))
}

fn dot(a: Pt, b: Pt) -> f64 {
    a.x * b.x + a.y * b.y
}

fn cmp_f64(a: f64, b: f64) -> Ordering {
    if a < b - EPS {
        Ordering::Less
    } else if a > b + EPS {
        Ordering::Greater
    } else {
        Ordering::Equal
    }
}

fn cmp_pt(a: Pt, b: Pt) -> Ordering {
    let oy = cmp_f64(a.y, b.y);
    if oy != Ordering::Equal {
        oy
    } else {
        cmp_f64(a.x, b.x)
    }
}

fn point_on_segment(p: Pt, a: Pt, b: Pt) -> bool {
    let ap = sub(p, a);
    let bp = sub(p, b);
    orient(a, b, p).abs() <= EPS && dot(ap, bp) <= EPS
}

fn point_in_polygon_inclusive(p: Pt, poly: &[Pt]) -> bool {
    let n = poly.len();
    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b) {
            return true;
        }
        let intersects = (a.y > p.y) != (b.y > p.y)
            && p.x < (b.x - a.x) * (p.y - a.y) / ((b.y - a.y).abs().max(EPS)) + a.x;
        if intersects {
            inside = !inside;
        }
    }
    inside
}

fn point_in_polygon_strict(p: Pt, poly: &[Pt]) -> bool {
    let n = poly.len();
    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b) {
            return false;
        }
        let intersects = (a.y > p.y) != (b.y > p.y)
            && p.x < (b.x - a.x) * (p.y - a.y) / ((b.y - a.y).abs().max(EPS)) + a.x;
        if intersects {
            inside = !inside;
        }
    }
    inside
}

fn segments_proper_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    ((o1 > EPS && o2 < -EPS) || (o1 < -EPS && o2 > EPS))
        && ((o3 > EPS && o4 < -EPS) || (o3 < -EPS && o4 > EPS))
}

fn polygon_signed_area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        s += cross(a, b);
    }
    0.5 * s
}

fn polygon_centroid(poly: &[Pt]) -> Pt {
    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut twice_area = 0.0;

    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let cr = cross(a, b);
        twice_area += cr;
        cx += (a.x + b.x) * cr;
        cy += (a.y + b.y) * cr;
    }

    if twice_area.abs() <= EPS {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for &p in poly {
            sx += p.x;
            sy += p.y;
        }
        let n = poly.len() as f64;
        return Pt::new(sx / n, sy / n);
    }

    Pt::new(cx / (3.0 * twice_area), cy / (3.0 * twice_area))
}

fn is_convex_polygon(poly: &[Pt]) -> bool {
    if poly.len() < 4 {
        return true;
    }
    let mut has_pos = false;
    let mut has_neg = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let c = poly[(i + 2) % poly.len()];
        let o = orient(a, b, c);
        if o > EPS {
            has_pos = true;
        } else if o < -EPS {
            has_neg = true;
        }
        if has_pos && has_neg {
            return false;
        }
    }
    true
}

fn point_in_convex_inclusive(p: Pt, poly: &[Pt], orient_sign: f64) -> bool {
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let v = orient(a, b, p) * orient_sign;
        if v < -EPS {
            return false;
        }
    }
    true
}

fn polygon_within_stock(poly: &[Pt], stock: &StockGeom) -> bool {
    if stock.convex {
        for &p in poly {
            if !point_in_convex_inclusive(p, &stock.poly, stock.orient_sign) {
                return false;
            }
        }
        return true;
    }

    for &p in poly {
        if !point_in_polygon_inclusive(p, &stock.poly) {
            return false;
        }
    }

    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        for &(c, d) in &stock.edges {
            if segments_proper_intersect(a, b, c, d) {
                return false;
            }
        }
    }

    true
}

fn polygons_overlap(a: &[Pt], b: &[Pt]) -> bool {
    for i in 0..a.len() {
        let a0 = a[i];
        let a1 = a[(i + 1) % a.len()];
        for j in 0..b.len() {
            let b0 = b[j];
            let b1 = b[(j + 1) % b.len()];
            if segments_proper_intersect(a0, a1, b0, b1) {
                return true;
            }
        }
    }

    for &p in a {
        if point_in_polygon_strict(p, b) {
            return true;
        }
    }

    for &p in b {
        if point_in_polygon_strict(p, a) {
            return true;
        }
    }

    false
}

fn rotate_point(p: Pt, cos_a: f64, sin_a: f64) -> Pt {
    Pt::new(cos_a * p.x - sin_a * p.y, sin_a * p.x + cos_a * p.y)
}

fn dedup_refs(mut refs: Vec<Pt>, max_refs: usize) -> Vec<Pt> {
    let mut out: Vec<Pt> = Vec::with_capacity(refs.len());
    for p in refs.drain(..) {
        let mut dup = false;
        for q in &out {
            if (p.x - q.x).abs() <= 1e-8 && (p.y - q.y).abs() <= 1e-8 {
                dup = true;
                break;
            }
        }
        if !dup {
            out.push(p);
            if out.len() >= max_refs {
                break;
            }
        }
    }
    out
}

fn generate_variants(piece: &Piece, angles: &[f64], max_refs: usize) -> Vec<Variant> {
    let mut vars = Vec::with_capacity(angles.len());

    for &ang in angles {
        let (sin_a, cos_a) = ang.sin_cos();
        let mut verts = Vec::with_capacity(piece.verts.len());
        for &p in &piece.verts {
            verts.push(rotate_point(p, cos_a, sin_a));
        }
        let bbox = Aabb::from_points(&verts);
        let centroid = rotate_point(piece.centroid, cos_a, sin_a);

        let mut low_idx = 0usize;
        let mut left_idx = 0usize;
        for i in 1..verts.len() {
            if cmp_pt(verts[i], verts[low_idx]) == Ordering::Less {
                low_idx = i;
            }
            let lx = cmp_f64(verts[i].x, verts[left_idx].x);
            if lx == Ordering::Less
                || (lx == Ordering::Equal && cmp_f64(verts[i].y, verts[left_idx].y) == Ordering::Less)
            {
                left_idx = i;
            }
        }

        let mut refs = Vec::with_capacity(4);
        refs.push(Pt::new(bbox.minx, bbox.miny));
        if max_refs >= 2 {
            refs.push(verts[low_idx]);
        }
        if max_refs >= 3 {
            refs.push(verts[left_idx]);
        }
        refs = dedup_refs(refs, max_refs.max(1));

        vars.push(Variant {
            angle: ang,
            verts,
            bbox,
            centroid,
            refs,
        });
    }

    vars.sort_by(|a, b| {
        let ah = a.bbox.height();
        let bh = b.bbox.height();
        let oy = cmp_f64(ah, bh);
        if oy != Ordering::Equal {
            oy
        } else {
            cmp_f64(a.bbox.width(), b.bbox.width())
        }
    });

    vars
}

fn quantize_anchor(p: Pt) -> (i64, i64) {
    let s = 1_000_000.0;
    ((p.x * s).round() as i64, (p.y * s).round() as i64)
}

fn pack_cell(ix: i32, iy: i32) -> u64 {
    ((ix as u32 as u64) << 32) | (iy as u32 as u64)
}

impl StockState {
    fn new(stock: &StockGeom, max_anchors: usize) -> Self {
        let bbox = stock.bbox;
        let diag_scale = (bbox.width().abs() * bbox.height().abs()).sqrt();
        let cell_size = (diag_scale / 40.0).max(1e-6);
        let mut s = Self {
            placed: Vec::new(),
            anchors: Vec::new(),
            anchor_set: HashSet::new(),
            grid: HashMap::new(),
            cell_size,
            origin: Pt::new(bbox.minx, bbox.miny),
            used_area: 0.0,
            max_anchors,
        };

        s.add_anchor(Pt::new(bbox.minx, bbox.miny));
        for &v in &stock.vertices {
            s.add_anchor(v);
        }

        s
    }

    fn add_anchor(&mut self, p: Pt) {
        if !p.x.is_finite() || !p.y.is_finite() {
            return;
        }
        if self.anchors.len() >= self.max_anchors {
            return;
        }
        let k = quantize_anchor(p);
        if !self.anchor_set.insert(k) {
            return;
        }
        let pos = self
            .anchors
            .binary_search_by(|q| cmp_pt(*q, p))
            .unwrap_or_else(|e| e);
        self.anchors.insert(pos, p);
    }

    fn cell_range(&self, bbox: &Aabb) -> (i32, i32, i32, i32) {
        let inv = 1.0 / self.cell_size;
        let ix0 = ((bbox.minx - self.origin.x) * inv).floor() as i32;
        let iy0 = ((bbox.miny - self.origin.y) * inv).floor() as i32;
        let ix1 = ((bbox.maxx - self.origin.x) * inv).floor() as i32;
        let iy1 = ((bbox.maxy - self.origin.y) * inv).floor() as i32;
        (ix0, iy0, ix1, iy1)
    }

    fn query_candidates(&self, bbox: &Aabb) -> Vec<usize> {
        let (ix0, iy0, ix1, iy1) = self.cell_range(bbox);
        let mut out = Vec::new();
        for ix in ix0..=ix1 {
            for iy in iy0..=iy1 {
                if let Some(v) = self.grid.get(&pack_cell(ix, iy)) {
                    out.extend_from_slice(v);
                }
            }
        }
        if out.len() > 1 {
            out.sort_unstable();
            out.dedup();
        }
        out
    }

    fn insert_placed(&mut self, placed: PlacedPoly) {
        let idx = self.placed.len();
        let (ix0, iy0, ix1, iy1) = self.cell_range(&placed.bbox);
        for ix in ix0..=ix1 {
            for iy in iy0..=iy1 {
                self.grid.entry(pack_cell(ix, iy)).or_default().push(idx);
            }
        }
        self.placed.push(placed);
    }
}

fn has_overlap(stock_state: &StockState, poly: &[Pt], bbox: &Aabb) -> bool {
    if stock_state.placed.is_empty() {
        return false;
    }
    let candidates = stock_state.query_candidates(bbox);
    for idx in candidates {
        let other = &stock_state.placed[idx];
        if !bbox.overlaps(&other.bbox) {
            continue;
        }
        if polygons_overlap(poly, &other.verts) {
            return true;
        }
    }
    false
}

fn transform_variant(variant: &Variant, tx: f64, ty: f64, out: &mut Vec<Pt>) {
    out.clear();
    if out.capacity() < variant.verts.len() {
        out.reserve(variant.verts.len() - out.capacity());
    }
    for &p in &variant.verts {
        out.push(Pt::new(p.x + tx, p.y + ty));
    }
}

fn add_new_anchors(stock_state: &mut StockState, poly: &[Pt], bbox: &Aabb) {
    stock_state.add_anchor(Pt::new(bbox.maxx, bbox.miny));
    stock_state.add_anchor(Pt::new(bbox.minx, bbox.maxy));
    stock_state.add_anchor(Pt::new(bbox.maxx, bbox.maxy));

    let stride = (poly.len() / 6).max(1);
    for i in (0..poly.len()).step_by(stride) {
        let p = poly[i];
        stock_state.add_anchor(p);
        stock_state.add_anchor(Pt::new(p.x + 1e-7, p.y));
        stock_state.add_anchor(Pt::new(p.x, p.y + 1e-7));
    }
}

fn try_candidate(
    stock_geom: &StockGeom,
    stock_state: &StockState,
    variant: &Variant,
    tx: f64,
    ty: f64,
    temp_poly: &mut Vec<Pt>,
) -> Option<Aabb> {
    let bbox = variant.bbox.shift(tx, ty);
    if !bbox.within(&stock_geom.bbox) {
        return None;
    }

    transform_variant(variant, tx, ty, temp_poly);
    if !polygon_within_stock(temp_poly, stock_geom) {
        return None;
    }

    if has_overlap(stock_state, temp_poly, &bbox) {
        return None;
    }

    Some(bbox)
}

fn try_place_in_stock(
    stock_geom: &StockGeom,
    stock_state: &mut StockState,
    piece: &Piece,
    variants: &[Variant],
    params: &Params,
    rng: &mut Lcg,
) -> Option<Placement> {
    let mut temp_poly = Vec::<Pt>::with_capacity(piece.verts.len());

    let anchor_limit = stock_state.anchors.len().min(params.anchor_try);
    for ai in 0..anchor_limit {
        let anchor = stock_state.anchors[ai];
        for v in variants {
            let ref_limit = v.refs.len().min(params.max_refs.max(1));
            for ri in 0..ref_limit {
                let rp = v.refs[ri];
                let tx = anchor.x - rp.x;
                let ty = anchor.y - rp.y;

                if let Some(bbox) = try_candidate(stock_geom, stock_state, v, tx, ty, &mut temp_poly) {
                    let placed = PlacedPoly {
                        verts: temp_poly.clone(),
                        bbox,
                    };
                    stock_state.insert_placed(placed);
                    stock_state.used_area += piece.area;
                    add_new_anchors(stock_state, &temp_poly, &bbox);
                    return Some(Placement {
                        stock: 0,
                        tx,
                        ty,
                        rot: v.angle,
                    });
                }
            }
        }
    }

    if stock_state.placed.is_empty() {
        for v in variants {
            for &sp in &stock_geom.sample_points {
                let tx = sp.x - v.centroid.x;
                let ty = sp.y - v.centroid.y;
                if let Some(bbox) = try_candidate(stock_geom, stock_state, v, tx, ty, &mut temp_poly) {
                    let placed = PlacedPoly {
                        verts: temp_poly.clone(),
                        bbox,
                    };
                    stock_state.insert_placed(placed);
                    stock_state.used_area += piece.area;
                    add_new_anchors(stock_state, &temp_poly, &bbox);
                    return Some(Placement {
                        stock: 0,
                        tx,
                        ty,
                        rot: v.angle,
                    });
                }
            }
        }

        for v in variants {
            let step = (v.verts.len() / 8).max(1);
            for &sv in &stock_geom.vertices {
                for pi in (0..v.verts.len()).step_by(step) {
                    let pv = v.verts[pi];
                    let tx = sv.x - pv.x;
                    let ty = sv.y - pv.y;
                    if let Some(bbox) = try_candidate(stock_geom, stock_state, v, tx, ty, &mut temp_poly) {
                        let placed = PlacedPoly {
                            verts: temp_poly.clone(),
                            bbox,
                        };
                        stock_state.insert_placed(placed);
                        stock_state.used_area += piece.area;
                        add_new_anchors(stock_state, &temp_poly, &bbox);
                        return Some(Placement {
                            stock: 0,
                            tx,
                            ty,
                            rot: v.angle,
                        });
                    }
                }
            }
        }

        for _ in 0..params.random_samples {
            let x = stock_geom.bbox.minx + rng.next_f64() * stock_geom.bbox.width();
            let y = stock_geom.bbox.miny + rng.next_f64() * stock_geom.bbox.height();
            let p = Pt::new(x, y);
            if !point_in_polygon_inclusive(p, &stock_geom.poly) {
                continue;
            }

            for v in variants {
                let tx = p.x - v.centroid.x;
                let ty = p.y - v.centroid.y;
                if let Some(bbox) = try_candidate(stock_geom, stock_state, v, tx, ty, &mut temp_poly) {
                    let placed = PlacedPoly {
                        verts: temp_poly.clone(),
                        bbox,
                    };
                    stock_state.insert_placed(placed);
                    stock_state.used_area += piece.area;
                    add_new_anchors(stock_state, &temp_poly, &bbox);
                    return Some(Placement {
                        stock: 0,
                        tx,
                        ty,
                        rot: v.angle,
                    });
                }
            }
        }
    }

    None
}

fn build_angles(k: usize) -> Vec<f64> {
    let kk = k.max(1);
    let mut out = Vec::with_capacity(kk);
    for i in 0..kk {
        out.push((2.0 * PI * (i as f64)) / (kk as f64));
    }
    out
}

fn choose_params(np: usize, avg_verts: f64) -> Params {
    let mut base_angles = if np <= 20 {
        36
    } else if np <= 100 {
        24
    } else if np <= 400 {
        16
    } else if np <= 1200 {
        10
    } else if np <= 3000 {
        6
    } else {
        4
    };

    if avg_verts > 120.0 && base_angles > 4 {
        base_angles -= 2;
    }

    let stock_try_limit = if np <= 100 {
        40
    } else if np <= 500 {
        24
    } else if np <= 2000 {
        12
    } else {
        8
    };

    let anchor_try = if np <= 100 {
        600
    } else if np <= 500 {
        250
    } else if np <= 2000 {
        80
    } else {
        25
    };

    let max_anchors = if np <= 100 {
        4000
    } else if np <= 500 {
        1800
    } else if np <= 2000 {
        600
    } else {
        250
    };

    let sample_grid = if np <= 100 {
        24
    } else if np <= 500 {
        16
    } else if np <= 2000 {
        10
    } else {
        8
    };

    let random_samples = if np <= 500 { 256 } else { 96 };

    let max_refs = if np <= 500 {
        3
    } else if np <= 2000 {
        2
    } else {
        1
    };

    Params {
        base_angles,
        stock_try_limit,
        anchor_try,
        max_anchors,
        sample_grid,
        random_samples,
        max_refs,
    }
}

fn angle_count_for_piece(base_k: usize, np: usize, verts: usize) -> usize {
    let mut k = base_k;
    if verts > 120 && k > 6 {
        k -= 2;
    }
    if verts > 300 && k > 4 {
        k = 4;
    }
    if np > 3000 && k > 4 {
        k = 4;
    }
    k.max(1)
}

fn build_sample_points(stock_poly: &[Pt], bbox: Aabb, grid: usize) -> Vec<Pt> {
    let mut pts = Vec::new();
    let mut seen: HashSet<(i64, i64)> = HashSet::new();

    let add_pt = |p: Pt, pts: &mut Vec<Pt>, seen: &mut HashSet<(i64, i64)>| {
        if !p.x.is_finite() || !p.y.is_finite() {
            return;
        }
        if !point_in_polygon_inclusive(p, stock_poly) {
            return;
        }
        let k = quantize_anchor(p);
        if seen.insert(k) {
            pts.push(p);
        }
    };

    add_pt(polygon_centroid(stock_poly), &mut pts, &mut seen);
    for &v in stock_poly {
        add_pt(v, &mut pts, &mut seen);
    }

    let gx = grid.max(2);
    let gy = grid.max(2);
    let w = bbox.width().max(EPS);
    let h = bbox.height().max(EPS);

    for iy in 0..gy {
        for ix in 0..gx {
            let x = bbox.minx + (ix as f64 + 0.5) * w / (gx as f64);
            let y = bbox.miny + (iy as f64 + 0.5) * h / (gy as f64);
            add_pt(Pt::new(x, y), &mut pts, &mut seen);
        }
    }

    pts.sort_by(|a, b| cmp_pt(*a, *b));
    pts
}

fn build_stock_geom(stock_poly: Vec<Pt>, sample_grid: usize) -> StockGeom {
    let bbox = Aabb::from_points(&stock_poly);
    let area_signed = polygon_signed_area(&stock_poly);
    let area = area_signed.abs();
    let convex = is_convex_polygon(&stock_poly);
    let orient_sign = if area_signed >= 0.0 { 1.0 } else { -1.0 };

    let mut edges = Vec::with_capacity(stock_poly.len());
    for i in 0..stock_poly.len() {
        edges.push((stock_poly[i], stock_poly[(i + 1) % stock_poly.len()]));
    }

    let sample_points = build_sample_points(&stock_poly, bbox, sample_grid);

    StockGeom {
        poly: stock_poly.clone(),
        edges,
        vertices: stock_poly,
        bbox,
        area,
        convex,
        orient_sign,
        sample_points,
    }
}

fn main() {
    let mut sc = Scanner::new();
    let sv = sc.next_usize();
    let np = sc.next_usize();

    let mut stock_poly = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x = sc.next_f64();
        let y = sc.next_f64();
        stock_poly.push(Pt::new(x, y));
    }

    let mut pieces = Vec::with_capacity(np);
    let mut total_piece_vertices = 0usize;

    for id in 0..np {
        let pv = sc.next_usize();
        total_piece_vertices += pv;
        let mut verts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x = sc.next_f64();
            let y = sc.next_f64();
            verts.push(Pt::new(x, y));
        }
        let area = polygon_signed_area(&verts).abs();
        let centroid = polygon_centroid(&verts);
        pieces.push(Piece {
            id,
            verts,
            area,
            centroid,
        });
    }

    let avg_verts = if np == 0 {
        0.0
    } else {
        total_piece_vertices as f64 / np as f64
    };

    let params = choose_params(np, avg_verts);
    let stock_geom = build_stock_geom(stock_poly, params.sample_grid);

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        let oa = cmp_f64(pieces[b].area, pieces[a].area);
        if oa != Ordering::Equal {
            oa
        } else {
            pieces[a].id.cmp(&pieces[b].id)
        }
    });

    let seed = (np as u64)
        ^ (sv as u64).wrapping_mul(0x9E3779B97F4A7C15)
        ^ (stock_geom.area.to_bits());
    let mut rng = Lcg::new(seed);

    let mut stocks: Vec<StockState> = Vec::new();
    let mut placements: Vec<Option<Placement>> = vec![None; np];

    for pid in order {
        let piece = &pieces[pid];
        let k = angle_count_for_piece(params.base_angles, np, piece.verts.len());
        let angles = build_angles(k);
        let variants = generate_variants(piece, &angles, params.max_refs);

        let mut placed = false;
        let stock_limit = stocks.len().min(params.stock_try_limit);

        for si in 0..stock_limit {
            if stocks[si].used_area + piece.area > stock_geom.area * 1.01 {
                continue;
            }

            if let Some(mut pl) = try_place_in_stock(
                &stock_geom,
                &mut stocks[si],
                piece,
                &variants,
                &params,
                &mut rng,
            ) {
                pl.stock = si;
                placements[piece.id] = Some(pl);
                placed = true;
                break;
            }
        }

        if !placed {
            let mut new_stock = StockState::new(&stock_geom, params.max_anchors);
            if let Some(mut pl) =
                try_place_in_stock(&stock_geom, &mut new_stock, piece, &variants, &params, &mut rng)
            {
                let si = stocks.len();
                pl.stock = si;
                placements[piece.id] = Some(pl);
                stocks.push(new_stock);
            } else {
                let si = stocks.len();
                stocks.push(new_stock);
                placements[piece.id] = Some(Placement {
                    stock: si,
                    tx: 0.0,
                    ty: 0.0,
                    rot: 0.0,
                });
            }
        }
    }

    let num_stocks = stocks.len().max(1);
    let mut out = String::new();
    out.push_str(&format!("{}\n", num_stocks));

    for i in 0..np {
        let p = placements[i].unwrap_or(Placement {
            stock: 0,
            tx: 0.0,
            ty: 0.0,
            rot: 0.0,
        });
        out.push_str(&format!("{} {:.9} {:.9} {:.9}\n", p.stock, p.tx, p.ty, p.rot));
    }

    print!("{}", out);
}
