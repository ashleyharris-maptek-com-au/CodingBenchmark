using System;
using System.Collections.Generic;
using System.Text;

public static class Program
{
    private readonly struct Clue
    {
        public readonly int Target;
        public readonly int StaticFlags;
        public readonly int[] Unknowns;

        public Clue(int target, int staticFlags, int[] unknowns)
        {
            Target = target;
            StaticFlags = staticFlags;
            Unknowns = unknowns;
        }
    }

    private sealed class Constraint
    {
        public readonly int Need;
        public readonly int[] Vars;

        public Constraint(int need, int[] vars)
        {
            Need = need;
            Vars = vars;
        }
    }

    private static int _h;
    private static int _w;
    private static string[] _board = Array.Empty<string>();

    public static void Main()
    {
        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.Write("0\n");
            return;
        }

        string[] dims = first.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
        if (dims.Length < 2 || !int.TryParse(dims[0], out _h) || !int.TryParse(dims[1], out _w) || _h <= 0 || _w <= 0)
        {
            Console.Write("0\n");
            return;
        }

        _board = new string[_h];
        for (int r = 0; r < _h; r++)
        {
            string line = Console.ReadLine() ?? string.Empty;
            if (line.Length < _w)
            {
                line = line.PadRight(_w, ' ');
            }
            else if (line.Length > _w)
            {
                line = line.Substring(0, _w);
            }

            _board[r] = line;
        }

        List<Clue> clues = BuildClues();
        var knownSafe = new HashSet<int>();
        var knownMines = new HashSet<int>();

        long cells = (long)_h * _w;

        // Adaptive strategy selection based on board scale.
        bool enableSubset = cells <= 50_000_000L && clues.Count > 0;
        int subsetConstraintLimit = cells <= 1_000_000L ? 200_000 : (cells <= 25_000_000L ? 80_000 : 20_000);

        bool enableExact = cells <= 4_000_000L && clues.Count <= 300_000;
        int maxExactVars = cells <= 400_000L ? 26 : (cells <= 1_500_000L ? 22 : 18);
        int maxExactConstraints = maxExactVars * 3;
        int exactNodeLimit = 2_000_000;

        int rounds = enableExact ? 6 : 1;
        for (int round = 0; round < rounds; round++)
        {
            RunLocalToFixpoint(clues, knownSafe, knownMines, enableSubset, subsetConstraintLimit);

            if (!enableExact)
            {
                break;
            }

            bool exactChanged = ApplyExactComponents(clues, knownSafe, knownMines, maxExactVars, maxExactConstraints, exactNodeLimit);
            if (!exactChanged)
            {
                break;
            }
        }

        RunLocalToFixpoint(clues, knownSafe, knownMines, enableSubset, subsetConstraintLimit);
        WriteSafeCells(knownSafe);
    }

    private static List<Clue> BuildClues()
    {
        var clues = new List<Clue>();

        for (int r = 0; r < _h; r++)
        {
            string row = _board[r];
            for (int c = 0; c < _w; c++)
            {
                char ch = row[c];
                if (!IsDigit(ch))
                {
                    continue;
                }

                int target = ch - '0';
                int staticFlags = 0;
                int[] tempUnknowns = new int[8];
                int count = 0;

                for (int dr = -1; dr <= 1; dr++)
                {
                    int nr = r + dr;
                    if ((uint)nr >= (uint)_h)
                    {
                        continue;
                    }

                    for (int dc = -1; dc <= 1; dc++)
                    {
                        if (dr == 0 && dc == 0)
                        {
                            continue;
                        }

                        int nc = c + dc;
                        if ((uint)nc >= (uint)_w)
                        {
                            continue;
                        }

                        char n = _board[nr][nc];
                        if (n == '*')
                        {
                            staticFlags++;
                        }
                        else if (n == ' ')
                        {
                            tempUnknowns[count++] = ToId(nr, nc);
                        }
                    }
                }

                if (count == 0)
                {
                    continue;
                }

                int[] unknowns = new int[count];
                for (int i = 0; i < count; i++)
                {
                    unknowns[i] = tempUnknowns[i];
                }

                clues.Add(new Clue(target, staticFlags, unknowns));
            }
        }

        return clues;
    }

    private static void RunLocalToFixpoint(
        List<Clue> clues,
        HashSet<int> knownSafe,
        HashSet<int> knownMines,
        bool enableSubset,
        int subsetConstraintLimit)
    {
        while (true)
        {
            bool changed = ApplyLocalAndSubset(clues, knownSafe, knownMines, enableSubset, subsetConstraintLimit);
            if (!changed)
            {
                return;
            }
        }
    }

    private static bool ApplyLocalAndSubset(
        List<Clue> clues,
        HashSet<int> knownSafe,
        HashSet<int> knownMines,
        bool enableSubset,
        int subsetConstraintLimit)
    {
        bool changed = false;
        var active = enableSubset ? new List<Constraint>(Math.Min(clues.Count, subsetConstraintLimit)) : null;
        bool collectForSubset = enableSubset;

        foreach (Clue clue in clues)
        {
            int need = clue.Target - clue.StaticFlags;
            int[] pending = new int[8];
            int pCount = 0;

            int[] unknowns = clue.Unknowns;
            for (int i = 0; i < unknowns.Length; i++)
            {
                int id = unknowns[i];
                if (knownSafe.Contains(id))
                {
                    continue;
                }

                if (knownMines.Contains(id))
                {
                    need--;
                    continue;
                }

                pending[pCount++] = id;
            }

            if (pCount == 0)
            {
                continue;
            }

            if (need == 0)
            {
                for (int i = 0; i < pCount; i++)
                {
                    changed |= MarkSafe(pending[i], knownSafe, knownMines);
                }

                continue;
            }

            if (need == pCount)
            {
                for (int i = 0; i < pCount; i++)
                {
                    changed |= MarkMine(pending[i], knownSafe, knownMines);
                }

                continue;
            }

            if (need <= 0 || need >= pCount || !collectForSubset || active is null)
            {
                continue;
            }

            int[] vars = new int[pCount];
            for (int i = 0; i < pCount; i++)
            {
                vars[i] = pending[i];
            }

            Array.Sort(vars);
            active.Add(new Constraint(need, vars));

            if (active.Count > subsetConstraintLimit)
            {
                active.Clear();
                collectForSubset = false;
            }
        }

        if (collectForSubset && active is not null && active.Count > 1)
        {
            changed |= ApplySubset(active, knownSafe, knownMines);
        }

        return changed;
    }

    private static bool ApplySubset(List<Constraint> active, HashSet<int> knownSafe, HashSet<int> knownMines)
    {
        bool changed = false;
        var varToConstraints = new Dictionary<int, List<int>>(active.Count * 2);

        for (int i = 0; i < active.Count; i++)
        {
            int[] vars = active[i].Vars;
            for (int j = 0; j < vars.Length; j++)
            {
                int v = vars[j];
                if (!varToConstraints.TryGetValue(v, out List<int>? list))
                {
                    list = new List<int>(4);
                    varToConstraints[v] = list;
                }

                list.Add(i);
            }
        }

        var seenPairs = new HashSet<long>();
        foreach (KeyValuePair<int, List<int>> kv in varToConstraints)
        {
            List<int> list = kv.Value;
            for (int i = 0; i < list.Count; i++)
            {
                for (int j = i + 1; j < list.Count; j++)
                {
                    int a = list[i];
                    int b = list[j];
                    if (a > b)
                    {
                        (a, b) = (b, a);
                    }

                    long key = ((long)a << 32) ^ (uint)b;
                    if (!seenPairs.Add(key))
                    {
                        continue;
                    }

                    Constraint c1 = active[a];
                    Constraint c2 = active[b];
                    changed |= DeduceFromSubsetPair(c1, c2, knownSafe, knownMines);
                }
            }
        }

        return changed;
    }

    private static bool DeduceFromSubsetPair(Constraint a, Constraint b, HashSet<int> knownSafe, HashSet<int> knownMines)
    {
        bool changed = false;

        Span<int> diff = stackalloc int[8];

        if (TrySubset(a.Vars, b.Vars, diff, out int diffCountAB))
        {
            int mineDiff = b.Need - a.Need;
            if (mineDiff == 0)
            {
                for (int i = 0; i < diffCountAB; i++)
                {
                    changed |= MarkSafe(diff[i], knownSafe, knownMines);
                }
            }
            else if (mineDiff == diffCountAB)
            {
                for (int i = 0; i < diffCountAB; i++)
                {
                    changed |= MarkMine(diff[i], knownSafe, knownMines);
                }
            }
        }

        if (TrySubset(b.Vars, a.Vars, diff, out int diffCountBA))
        {
            int mineDiff = a.Need - b.Need;
            if (mineDiff == 0)
            {
                for (int i = 0; i < diffCountBA; i++)
                {
                    changed |= MarkSafe(diff[i], knownSafe, knownMines);
                }
            }
            else if (mineDiff == diffCountBA)
            {
                for (int i = 0; i < diffCountBA; i++)
                {
                    changed |= MarkMine(diff[i], knownSafe, knownMines);
                }
            }
        }

        return changed;
    }

    private static bool TrySubset(int[] maybeSubset, int[] maybeSuperset, Span<int> diff, out int diffCount)
    {
        diffCount = 0;
        if (maybeSubset.Length > maybeSuperset.Length)
        {
            return false;
        }

        int i = 0;
        int j = 0;

        while (i < maybeSubset.Length && j < maybeSuperset.Length)
        {
            int a = maybeSubset[i];
            int b = maybeSuperset[j];

            if (a == b)
            {
                i++;
                j++;
            }
            else if (a > b)
            {
                diff[diffCount++] = b;
                j++;
            }
            else
            {
                return false;
            }
        }

        if (i != maybeSubset.Length)
        {
            return false;
        }

        while (j < maybeSuperset.Length)
        {
            diff[diffCount++] = maybeSuperset[j++];
        }

        return true;
    }

    private static bool ApplyExactComponents(
        List<Clue> clues,
        HashSet<int> knownSafe,
        HashSet<int> knownMines,
        int maxExactVars,
        int maxExactConstraints,
        int nodeLimit)
    {
        var active = BuildUnresolvedConstraints(clues, knownSafe, knownMines);
        if (active.Count == 0)
        {
            return false;
        }

        var varToConstraints = new Dictionary<int, List<int>>(active.Count * 2);
        for (int ci = 0; ci < active.Count; ci++)
        {
            int[] vars = active[ci].Vars;
            for (int i = 0; i < vars.Length; i++)
            {
                int v = vars[i];
                if (!varToConstraints.TryGetValue(v, out List<int>? list))
                {
                    list = new List<int>(4);
                    varToConstraints[v] = list;
                }

                list.Add(ci);
            }
        }

        bool changed = false;
        bool[] visitedConstraints = new bool[active.Count];
        var queue = new Queue<int>();

        for (int start = 0; start < active.Count; start++)
        {
            if (visitedConstraints[start])
            {
                continue;
            }

            queue.Clear();
            queue.Enqueue(start);
            visitedConstraints[start] = true;

            var componentConstraints = new List<int>();
            var componentVarSet = new HashSet<int>();

            while (queue.Count > 0)
            {
                int cIdx = queue.Dequeue();
                componentConstraints.Add(cIdx);

                int[] vars = active[cIdx].Vars;
                for (int i = 0; i < vars.Length; i++)
                {
                    int v = vars[i];
                    if (!componentVarSet.Add(v))
                    {
                        continue;
                    }

                    if (!varToConstraints.TryGetValue(v, out List<int>? neighbors))
                    {
                        continue;
                    }

                    for (int k = 0; k < neighbors.Count; k++)
                    {
                        int next = neighbors[k];
                        if (!visitedConstraints[next])
                        {
                            visitedConstraints[next] = true;
                            queue.Enqueue(next);
                        }
                    }
                }
            }

            if (componentVarSet.Count == 0)
            {
                continue;
            }

            if (componentVarSet.Count > maxExactVars || componentConstraints.Count > maxExactConstraints)
            {
                continue;
            }

            int[] varIds = new int[componentVarSet.Count];
            componentVarSet.CopyTo(varIds);
            Array.Sort(varIds);

            var localIndex = new Dictionary<int, int>(varIds.Length);
            for (int i = 0; i < varIds.Length; i++)
            {
                localIndex[varIds[i]] = i;
            }

            var localConstraints = new List<Constraint>(componentConstraints.Count);
            for (int i = 0; i < componentConstraints.Count; i++)
            {
                Constraint g = active[componentConstraints[i]];
                int[] localVars = new int[g.Vars.Length];
                for (int k = 0; k < g.Vars.Length; k++)
                {
                    localVars[k] = localIndex[g.Vars[k]];
                }

                Array.Sort(localVars);
                localConstraints.Add(new Constraint(g.Need, localVars));
            }

            if (!EnumerateComponent(varIds.Length, localConstraints, nodeLimit, out bool[] canBeSafe, out bool[] canBeMine))
            {
                continue;
            }

            for (int i = 0; i < varIds.Length; i++)
            {
                int id = varIds[i];
                if (canBeSafe[i] && !canBeMine[i])
                {
                    changed |= MarkSafe(id, knownSafe, knownMines);
                }
                else if (canBeMine[i] && !canBeSafe[i])
                {
                    changed |= MarkMine(id, knownSafe, knownMines);
                }
            }
        }

        return changed;
    }

    private static List<Constraint> BuildUnresolvedConstraints(List<Clue> clues, HashSet<int> knownSafe, HashSet<int> knownMines)
    {
        var active = new List<Constraint>(Math.Min(clues.Count, 200_000));

        foreach (Clue clue in clues)
        {
            int need = clue.Target - clue.StaticFlags;
            int[] pending = new int[8];
            int count = 0;

            int[] unknowns = clue.Unknowns;
            for (int i = 0; i < unknowns.Length; i++)
            {
                int id = unknowns[i];
                if (knownSafe.Contains(id))
                {
                    continue;
                }

                if (knownMines.Contains(id))
                {
                    need--;
                    continue;
                }

                pending[count++] = id;
            }

            if (count <= 1)
            {
                continue;
            }

            if (need <= 0 || need >= count)
            {
                continue;
            }

            int[] vars = new int[count];
            for (int i = 0; i < count; i++)
            {
                vars[i] = pending[i];
            }

            Array.Sort(vars);
            active.Add(new Constraint(need, vars));
        }

        return active;
    }

    private static bool EnumerateComponent(
        int variableCount,
        List<Constraint> constraints,
        int nodeLimit,
        out bool[] canBeSafe,
        out bool[] canBeMine)
    {
        bool[] safePossible = new bool[variableCount];
        bool[] minePossible = new bool[variableCount];

        int constraintCount = constraints.Count;
        var varToConstraints = new List<int>[variableCount];
        for (int i = 0; i < variableCount; i++)
        {
            varToConstraints[i] = new List<int>(4);
        }

        int[] need = new int[constraintCount];
        int[] remaining = new int[constraintCount];
        int[] mines = new int[constraintCount];

        for (int c = 0; c < constraintCount; c++)
        {
            Constraint constraint = constraints[c];
            need[c] = constraint.Need;
            remaining[c] = constraint.Vars.Length;

            int[] vars = constraint.Vars;
            for (int i = 0; i < vars.Length; i++)
            {
                varToConstraints[vars[i]].Add(c);
            }
        }

        int[] order = new int[variableCount];
        for (int i = 0; i < variableCount; i++)
        {
            order[i] = i;
        }

        Array.Sort(order, (a, b) => varToConstraints[b].Count.CompareTo(varToConstraints[a].Count));

        int[] assignment = new int[variableCount];
        Array.Fill(assignment, -1);

        long nodes = 0;
        int solutions = 0;
        bool aborted = false;

        void Dfs(int pos)
        {
            if (aborted)
            {
                return;
            }

            nodes++;
            if (nodes > nodeLimit)
            {
                aborted = true;
                return;
            }

            if (pos == variableCount)
            {
                for (int c = 0; c < constraintCount; c++)
                {
                    if (mines[c] != need[c])
                    {
                        return;
                    }
                }

                solutions++;
                for (int v = 0; v < variableCount; v++)
                {
                    if (assignment[v] == 1)
                    {
                        minePossible[v] = true;
                    }
                    else
                    {
                        safePossible[v] = true;
                    }
                }

                return;
            }

            int var = order[pos];

            TryAssign(pos, var, 0);
            if (!aborted)
            {
                TryAssign(pos, var, 1);
            }
        }

        void TryAssign(int pos, int var, int value)
        {
            assignment[var] = value;
            List<int> links = varToConstraints[var];
            bool valid = true;

            for (int i = 0; i < links.Count; i++)
            {
                int c = links[i];
                remaining[c]--;
                if (value == 1)
                {
                    mines[c]++;
                }

                if (mines[c] > need[c] || mines[c] + remaining[c] < need[c])
                {
                    valid = false;
                }
            }

            if (valid)
            {
                Dfs(pos + 1);
            }

            for (int i = links.Count - 1; i >= 0; i--)
            {
                int c = links[i];
                if (value == 1)
                {
                    mines[c]--;
                }

                remaining[c]++;
            }

            assignment[var] = -1;
        }

        Dfs(0);

        if (aborted || solutions == 0)
        {
            canBeSafe = safePossible;
            canBeMine = minePossible;
            return false;
        }

        canBeSafe = safePossible;
        canBeMine = minePossible;
        return true;
    }

    private static bool MarkSafe(int id, HashSet<int> knownSafe, HashSet<int> knownMines)
    {
        if (knownMines.Contains(id))
        {
            return false;
        }

        return knownSafe.Add(id);
    }

    private static bool MarkMine(int id, HashSet<int> knownSafe, HashSet<int> knownMines)
    {
        if (knownSafe.Contains(id))
        {
            return false;
        }

        return knownMines.Add(id);
    }

    private static void WriteSafeCells(HashSet<int> knownSafe)
    {
        var safeIds = new List<int>(knownSafe.Count);
        foreach (int id in knownSafe)
        {
            int r = id / _w;
            int c = id % _w;
            if ((uint)r < (uint)_h && (uint)c < (uint)_w && _board[r][c] == ' ')
            {
                safeIds.Add(id);
            }
        }

        safeIds.Sort();

        var sb = new StringBuilder();
        sb.AppendLine(safeIds.Count.ToString());
        for (int i = 0; i < safeIds.Count; i++)
        {
            int id = safeIds[i];
            int r = id / _w;
            int c = id % _w;
            sb.Append(c).Append(' ').Append(r).Append('\n');
        }

        Console.Write(sb.ToString());
    }

    private static bool IsDigit(char ch)
    {
        return ch >= '0' && ch <= '8';
    }

    private static int ToId(int r, int c)
    {
        return r * _w + c;
    }
}
