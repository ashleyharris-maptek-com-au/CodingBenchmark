from math import isfinite

_AP = None


def _clip(x, lo, hi):
    if x < lo:
        return lo
    if x > hi:
        return hi
    return x


def _f(x):
    try:
        v = float(x)
    except Exception:
        return None
    if not isfinite(v):
        return None
    return v


def _median(values):
    if not values:
        return None
    vals = sorted(values)
    return vals[len(vals) // 2]


def _norm_deg(x):
    while x > 180.0:
        x -= 360.0
    while x < -180.0:
        x += 360.0
    return x


def _voted(state, value_keys, valid_keys=None, default=0.0, outlier=None):
    vals = []
    for i, key in enumerate(value_keys):
        v = _f(state.get(key))
        if v is None:
            continue
        if valid_keys is not None:
            vk = valid_keys[i]
            if vk is not None and not bool(state.get(vk, False)):
                continue
        vals.append(v)
    if not vals:
        return float(default), 0.0

    med = _median(vals)
    if outlier is not None and len(vals) >= 3:
        filt = [v for v in vals if abs(v - med) <= outlier]
        if filt:
            vals = filt
            med = _median(vals)
    conf = min(1.0, len(vals) / 3.0)
    return med, conf


def _reset_state():
    return {
        "mode": "APPROACH",
        "loc_i": 0.0,
        "gs_i": 0.0,
        "spd_i": 0.0,
        "trim_i": 0.0,
        "prev_loc": 0.0,
        "prev_gs": 0.0,
        "hold_loc": 0.0,
        "hold_gs": 0.0,
        "hold_loc_age": 999.0,
        "hold_gs_age": 999.0,
        "runway_elev_ft": None,
        "last_on_ground": False,
    }


def autopilot_step(state: dict, dt: float) -> dict:
    global _AP
    if _AP is None:
        _AP = _reset_state()

    dt_v = _f(dt)
    if dt_v is None:
        dt_v = 0.05
    dt_v = _clip(dt_v, 0.01, 0.2)

    vref = _f(state.get("target_vref_kt"))
    if vref is None or vref < 90.0 or vref > 220.0:
        vref = 137.0
    target_ias = _f(state.get("target_ias_kt"))
    if target_ias is None or target_ias < 90.0 or target_ias > 260.0:
        target_ias = vref + 10.0

    loc_raw, loc_conf = _voted(
        state,
        ["localizer_1_dots", "localizer_2_dots", "localizer_3_dots"],
        ["localizer_1_valid", "localizer_2_valid", "localizer_3_valid"],
        default=0.0,
        outlier=1.5,
    )
    gs_raw, gs_conf = _voted(
        state,
        ["glideslope_1_dots", "glideslope_2_dots", "glideslope_3_dots"],
        ["glideslope_1_valid", "glideslope_2_valid", "glideslope_3_valid"],
        default=0.0,
        outlier=1.5,
    )

    baro_alt, baro_conf = _voted(
        state,
        ["baro_alt_1_ft", "baro_alt_2_ft", "baro_alt_3_ft"],
        ["baro_alt_1_valid", "baro_alt_2_valid", "baro_alt_3_valid"],
        default=0.0,
        outlier=500.0,
    )
    gps_alt, gps_alt_conf = _voted(
        state,
        ["gps_1_alt_ft", "gps_2_alt_ft", "gps_3_alt_ft"],
        ["gps_1_valid", "gps_2_valid", "gps_3_valid"],
        default=baro_alt,
        outlier=500.0,
    )
    if baro_conf <= 0.0 and gps_alt_conf > 0.0:
        baro_alt = gps_alt
        baro_conf = gps_alt_conf

    ias, _ = _voted(
        state,
        ["ias_1_kt", "ias_2_kt", "ias_3_kt"],
        ["ias_1_valid", "ias_2_valid", "ias_3_valid"],
        default=target_ias,
        outlier=25.0,
    )
    vs, _ = _voted(
        state,
        ["irs_1_vs_fpm", "irs_2_vs_fpm", "irs_3_vs_fpm"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        default=-700.0,
        outlier=2000.0,
    )
    roll, _ = _voted(
        state,
        ["irs_1_roll_deg", "irs_2_roll_deg", "irs_3_roll_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        default=0.0,
        outlier=25.0,
    )
    pitch, _ = _voted(
        state,
        ["irs_1_pitch_deg", "irs_2_pitch_deg", "irs_3_pitch_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        default=2.0,
        outlier=20.0,
    )
    heading, _ = _voted(
        state,
        ["irs_1_heading_deg", "irs_2_heading_deg", "irs_3_heading_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        default=0.0,
        outlier=60.0,
    )
    track, _ = _voted(
        state,
        ["irs_1_track_deg", "irs_2_track_deg", "irs_3_track_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        default=heading,
        outlier=60.0,
    )

    ground_speed = _f(state.get("groundspeed_kt"))
    if ground_speed is None:
        gps_gs, gps_gs_conf = _voted(
            state,
            ["gps_1_gs_kt", "gps_2_gs_kt", "gps_3_gs_kt"],
            ["gps_1_valid", "gps_2_valid", "gps_3_valid"],
            default=max(ias - 5.0, 80.0),
            outlier=50.0,
        )
        irs_gs, irs_gs_conf = _voted(
            state,
            ["irs_1_gs_kt", "irs_2_gs_kt", "irs_3_gs_kt"],
            ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
            default=gps_gs,
            outlier=50.0,
        )
        if gps_gs_conf > 0.0 and irs_gs_conf > 0.0:
            ground_speed = 0.5 * (gps_gs + irs_gs)
        elif gps_gs_conf > 0.0:
            ground_speed = gps_gs
        else:
            ground_speed = irs_gs

    dmes = []
    d1 = _f(state.get("dme_1_nm"))
    d2 = _f(state.get("dme_2_nm"))
    if d1 is not None and d1 >= 0.0:
        dmes.append(d1)
    if d2 is not None and d2 >= 0.0:
        dmes.append(d2)
    dme_nm = _median(dmes)

    ra = _f(state.get("radio_alt_ft"))
    ra_valid = ra is not None and -20.0 <= ra <= 5000.0

    if ra_valid and baro_conf > 0.0:
        runway_est = baro_alt - ra
        if _AP["runway_elev_ft"] is None:
            _AP["runway_elev_ft"] = runway_est
        else:
            _AP["runway_elev_ft"] = 0.985 * _AP["runway_elev_ft"] + 0.015 * runway_est

    if not ra_valid and _AP["runway_elev_ft"] is not None and baro_conf > 0.0:
        ra = max(0.0, baro_alt - _AP["runway_elev_ft"])
        ra_valid = True

    if not ra_valid and dme_nm is not None:
        est = dme_nm * 318.0
        if gs_conf > 0.2:
            est += gs_raw * 100.0
        ra = max(0.0, est)
        ra_valid = True

    if not ra_valid and baro_conf > 0.0:
        if _AP["runway_elev_ft"] is None:
            _AP["runway_elev_ft"] = baro_alt - 1000.0
        ra = max(0.0, baro_alt - _AP["runway_elev_ft"])
        ra_valid = True

    if not ra_valid:
        ra = 1000.0

    if loc_conf > 0.0:
        _AP["hold_loc"] = loc_raw
        _AP["hold_loc_age"] = 0.0
    else:
        _AP["hold_loc_age"] += dt_v
    if gs_conf > 0.0:
        _AP["hold_gs"] = gs_raw
        _AP["hold_gs_age"] = 0.0
    else:
        _AP["hold_gs_age"] += dt_v

    loc = loc_raw
    gs = gs_raw
    if loc_conf <= 0.0 and _AP["hold_loc_age"] < 6.0:
        loc = _AP["hold_loc"]
        loc_conf = max(0.1, 1.0 - (_AP["hold_loc_age"] / 6.0))
    if gs_conf <= 0.0 and _AP["hold_gs_age"] < 6.0:
        gs = _AP["hold_gs"]
        gs_conf = max(0.1, 1.0 - (_AP["hold_gs_age"] / 6.0))

    eng1_run = bool(state.get("engine_1_running", True)) and not bool(state.get("engine_1_fire", False))
    eng2_run = bool(state.get("engine_2_running", True)) and not bool(state.get("engine_2_fire", False))
    n1_1 = _f(state.get("engine_1_n1_pct"))
    n1_2 = _f(state.get("engine_2_n1_pct"))
    if n1_1 is not None and n1_1 < 5.0:
        eng1_run = False
    if n1_2 is not None and n1_2 < 5.0:
        eng2_run = False
    dual_engine_out = (not eng1_run) and (not eng2_run)

    on_ground = bool(state.get("on_ground", False) or state.get("weight_on_wheels", False))
    if not on_ground and ra < 1.5 and ground_speed < 40.0:
        on_ground = True

    if _AP["last_on_ground"] and (not on_ground) and ra > 80.0:
        _AP = _reset_state()

    below_dh = bool(state.get("below_decision_height", False)) or (ra < 200.0)
    crab = _norm_deg(heading - track)

    gust_add = 5.0 if abs(crab) > 10.0 else 0.0
    if dual_engine_out:
        if ra > 800.0:
            speed_ref = max(vref + 25.0, target_ias + 5.0)
        elif ra > 300.0:
            speed_ref = max(vref + 15.0, target_ias)
        else:
            speed_ref = vref + 10.0
    else:
        if ra > 1200.0:
            speed_ref = max(target_ias, vref + 20.0)
        elif ra > 600.0:
            speed_ref = max(vref + 12.0, target_ias - 5.0)
        elif ra > 200.0:
            speed_ref = vref + 8.0
        else:
            speed_ref = vref + 4.0
    speed_ref += gust_add

    if on_ground:
        _AP["mode"] = "ROLLOUT"
    elif _AP["mode"] != "GA":
        if ra < 55.0 and vs < -80.0:
            _AP["mode"] = "FLARE"
        else:
            _AP["mode"] = "APPROACH"

    unstable = False
    if (not on_ground) and (_AP["mode"] != "GA") and (not dual_engine_out):
        if ra > 80.0 and (abs(loc) > 2.8 or abs(gs) > 2.4):
            unstable = True
        if ra < 700.0 and (abs(loc) > 1.7 or abs(gs) > 1.7):
            unstable = True
        if ra < 350.0 and (abs(loc) > 1.1 or abs(gs) > 1.1):
            unstable = True
        if ra < 400.0 and (ias > speed_ref + 25.0 or ias < max(vref - 8.0, 105.0)):
            unstable = True
        if ra < 300.0 and vs < -1700.0:
            unstable = True
        if below_dh and (loc_conf < 0.2 or gs_conf < 0.2 or abs(loc) > 1.2 or abs(gs) > 1.2):
            unstable = True
        if ra < 250.0 and ias < vref - 12.0 and vs < -1300.0:
            unstable = True
    if unstable and ra > 40.0:
        _AP["mode"] = "GA"

    cmd = {
        "elevator": 0.0,
        "aileron": 0.0,
        "rudder": 0.0,
        "throttle_1": 0.0,
        "throttle_2": 0.0,
        "elevator_trim": 0.0,
        "flaps": 5.0,
        "gear": 1.0,
        "speedbrake": 0.0,
        "spoiler": 0.0,
        "autobrake": 0.0,
        "reverse_1": 0.0,
        "reverse_2": 0.0,
    }

    if dual_engine_out:
        if ra > 1200.0:
            cmd["flaps"] = 5.0
            cmd["gear"] = 0.0
        elif ra > 700.0:
            cmd["flaps"] = 10.0
            cmd["gear"] = 0.0
        elif ra > 350.0:
            cmd["flaps"] = 15.0
            cmd["gear"] = 1.0
        elif ra > 150.0:
            cmd["flaps"] = 25.0
            cmd["gear"] = 1.0
        else:
            cmd["flaps"] = 30.0
            cmd["gear"] = 1.0
    else:
        if ra > 1500.0:
            cmd["flaps"] = 5.0
            cmd["gear"] = 0.0
        elif ra > 900.0:
            cmd["flaps"] = 15.0
            cmd["gear"] = 1.0
        elif ra > 350.0:
            cmd["flaps"] = 25.0
            cmd["gear"] = 1.0
        else:
            cmd["flaps"] = 30.0
            cmd["gear"] = 1.0

    if _AP["mode"] == "GA":
        pitch_target = 10.0
        cmd["elevator"] = _clip(0.07 * (pitch_target - pitch) + 0.002 * ((speed_ref + 10.0) - ias), -0.2, 0.85)
        cmd["aileron"] = _clip(-0.14 * loc - 0.05 * roll, -1.0, 1.0)
        cmd["rudder"] = _clip(-0.08 * loc - 0.05 * crab, -1.0, 1.0)
        cmd["throttle_1"] = 1.0 if eng1_run else 0.0
        cmd["throttle_2"] = 1.0 if eng2_run else 0.0
        cmd["flaps"] = max(cmd["flaps"], 15.0)
        if (ra > 120.0) and (not dual_engine_out):
            cmd["gear"] = 0.0
        cmd["elevator_trim"] = _clip(0.02 * (pitch_target - pitch), -0.3, 0.3)
        _AP["last_on_ground"] = on_ground
        return cmd

    if on_ground:
        cmd["gear"] = 1.0
        cmd["flaps"] = max(cmd["flaps"], 25.0)
        cmd["spoiler"] = 1.0
        cmd["speedbrake"] = 1.0
        cmd["autobrake"] = 1.0 if ground_speed > 20.0 else 0.3
        cmd["elevator"] = 0.2
        cmd["aileron"] = _clip(-0.06 * roll, -0.5, 0.5)
        cmd["rudder"] = _clip(-0.2 * loc - 0.04 * crab, -1.0, 1.0)
        rev = 0.0
        if not dual_engine_out:
            if ground_speed > 90.0:
                rev = 1.0
            elif ground_speed > 60.0:
                rev = 0.7
            elif ground_speed > 35.0:
                rev = 0.35
        cmd["reverse_1"] = rev if eng1_run else 0.0
        cmd["reverse_2"] = rev if eng2_run else 0.0
        _AP["last_on_ground"] = on_ground
        return cmd

    d_loc = (loc - _AP["prev_loc"]) / dt_v
    d_gs = (gs - _AP["prev_gs"]) / dt_v
    _AP["prev_loc"] = loc
    _AP["prev_gs"] = gs

    if loc_conf > 0.2:
        _AP["loc_i"] = _clip(_AP["loc_i"] + loc * dt_v, -2.5, 2.5)
    else:
        _AP["loc_i"] *= 0.98
    if gs_conf > 0.2 and ra > 80.0:
        _AP["gs_i"] = _clip(_AP["gs_i"] + gs * dt_v, -2.0, 2.0)
    else:
        _AP["gs_i"] *= 0.98

    aileron = _clip(-0.30 * loc - 0.06 * d_loc - 0.035 * roll - 0.015 * _AP["loc_i"], -1.0, 1.0)
    rudder = _clip(-0.14 * loc - (0.03 if ra > 120.0 else 0.08) * crab, -1.0, 1.0)

    gs_for_path = max(ground_speed, 90.0)
    vs_nominal = -5.2 * gs_for_path
    vs_target = vs_nominal - 280.0 * gs

    if _AP["mode"] == "FLARE":
        flare = _clip((50.0 - ra) / 50.0, 0.0, 1.0)
        pitch_target = 2.0 + 3.5 * flare
        sink_target = -220.0 + 130.0 * flare
        elevator = 0.10 * (pitch_target - pitch) + 0.0009 * (sink_target - vs)
        aileron *= 0.7
        rudder *= 1.2
    else:
        elevator = -0.22 * gs - 0.045 * d_gs + 0.0005 * (vs_target - vs) - 0.03 * pitch - 0.02 * _AP["gs_i"]
        if ias < speed_ref - 8.0:
            elevator -= 0.10
        if ias < vref - 3.0:
            elevator -= 0.08
        if ias > speed_ref + 12.0 and ra > 100.0:
            elevator += 0.05

    elevator = _clip(elevator, -1.0, 1.0)

    throttle = 0.0
    if not dual_engine_out:
        spd_err = speed_ref - ias
        _AP["spd_i"] = _clip(_AP["spd_i"] + spd_err * dt_v * 0.03, -0.25, 0.35)
        throttle = 0.46 + 0.017 * spd_err + _AP["spd_i"]
        throttle += _clip(((-vs) - abs(vs_target)) * 0.00035, -0.10, 0.25)
        if _AP["mode"] == "FLARE":
            throttle *= _clip(ra / 60.0, 0.0, 1.0)
            if ra < 15.0:
                throttle = min(throttle, 0.08)
        elif ra < 80.0:
            throttle = min(throttle, 0.35)

        if ra < 400.0 and vs < -1200.0 and ias < speed_ref - 5.0:
            throttle = 1.0
            elevator = max(elevator, 0.20)
        throttle = _clip(throttle, 0.0, 1.0)

    cmd["elevator"] = elevator
    cmd["aileron"] = aileron
    cmd["rudder"] = rudder
    cmd["throttle_1"] = throttle if eng1_run else 0.0
    cmd["throttle_2"] = throttle if eng2_run else 0.0
    _AP["trim_i"] = _clip(0.98 * _AP["trim_i"] + 0.02 * _clip(-0.08 * gs, -0.3, 0.3), -0.4, 0.4)
    cmd["elevator_trim"] = _AP["trim_i"]

    _AP["last_on_ground"] = on_ground
    return cmd
