#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <utility>
#include <vector>

using ll = long long;

struct Task {
    int machine;
    int duration;
};

struct Instance {
    int jobsCount = 0;
    int machinesCount = 0;
    int totalTasks = 0;
    std::vector<std::vector<Task>> jobs;
    std::vector<ll> machineTotalLoad;
    std::vector<std::vector<ll>> remJobDur;
};

struct ScheduleResult {
    ll makespan = std::numeric_limits<ll>::max();
    std::vector<std::vector<ll>> startTimes;
};

static std::pair<ll, ll> priorityKeys(
    const Instance& ins,
    int job,
    int taskIndex,
    int policy
) {
    const Task& t = ins.jobs[job][taskIndex];
    const ll rem = ins.remJobDur[job][taskIndex];
    const ll machineLoad = ins.machineTotalLoad[t.machine];
    const ll opsLeft = static_cast<ll>(ins.jobs[job].size() - taskIndex);

    switch (policy) {
        case 1:
            // LRPT tie-break: larger remaining work first.
            return {-rem, static_cast<ll>(t.duration)};
        case 2:
            // Bottleneck-first tie-break: prefer globally busy machines first.
            return {-machineLoad, static_cast<ll>(t.duration)};
        case 3:
            // More operations left first.
            return {-opsLeft, static_cast<ll>(t.duration)};
        case 0:
        default:
            // SPT tie-break.
            return {static_cast<ll>(t.duration), rem};
    }
}

static ScheduleResult buildHeuristicSchedule(const Instance& ins, int policy) {
    struct Node {
        ll earliestStart;
        ll p2;
        ll p3;
        int job;
        int ticket;
    };

    struct Cmp {
        bool operator()(const Node& a, const Node& b) const {
            if (a.earliestStart != b.earliestStart) return a.earliestStart > b.earliestStart;
            if (a.p2 != b.p2) return a.p2 > b.p2;
            if (a.p3 != b.p3) return a.p3 > b.p3;
            return a.job > b.job;
        }
    };

    const int J = ins.jobsCount;
    const int M = ins.machinesCount;

    std::vector<int> nextTask(J, 0);
    std::vector<int> ticket(J, 0);
    std::vector<ll> jobReady(J, 0);
    std::vector<ll> machineReady(M, 0);

    std::vector<std::vector<ll>> startTimes(J);
    for (int j = 0; j < J; ++j) {
        startTimes[j].assign(ins.jobs[j].size(), 0);
    }

    std::priority_queue<Node, std::vector<Node>, Cmp> pq;

    auto pushJob = [&](int j) {
        const int k = nextTask[j];
        if (k >= static_cast<int>(ins.jobs[j].size())) return;
        const Task& t = ins.jobs[j][k];
        const ll est = std::max(jobReady[j], machineReady[t.machine]);
        const auto [p2, p3] = priorityKeys(ins, j, k, policy);
        pq.push(Node{est, p2, p3, j, ticket[j]});
    };

    for (int j = 0; j < J; ++j) {
        if (!ins.jobs[j].empty()) {
            pushJob(j);
        }
    }

    int scheduled = 0;
    while (scheduled < ins.totalTasks) {
        if (pq.empty()) {
            // Should never happen for valid input, but keep behavior safe.
            break;
        }

        Node cur = pq.top();
        pq.pop();

        const int j = cur.job;
        if (cur.ticket != ticket[j]) continue;

        const int k = nextTask[j];
        if (k >= static_cast<int>(ins.jobs[j].size())) continue;

        const Task& t = ins.jobs[j][k];
        const ll est = std::max(jobReady[j], machineReady[t.machine]);

        // Lazy key repair when machine/job readiness changed.
        if (est != cur.earliestStart) {
            pq.push(Node{est, cur.p2, cur.p3, j, cur.ticket});
            continue;
        }

        startTimes[j][k] = est;
        const ll finish = est + static_cast<ll>(t.duration);

        jobReady[j] = finish;
        machineReady[t.machine] = finish;

        ++nextTask[j];
        ++ticket[j];
        ++scheduled;

        if (nextTask[j] < static_cast<int>(ins.jobs[j].size())) {
            pushJob(j);
        }
    }

    ll makespan = 0;
    for (ll x : machineReady) makespan = std::max(makespan, x);

    ScheduleResult out;
    out.makespan = makespan;
    out.startTimes = std::move(startTimes);
    return out;
}

class ExactTinySolver {
public:
    explicit ExactTinySolver(const Instance& ins)
        : ins_(ins),
          J_(ins.jobsCount),
          M_(ins.machinesCount),
          N_(ins.totalTasks),
          nextTask_(J_, 0),
          jobReady_(J_, 0),
          machineReady_(M_, 0),
          remMachine_(ins.machineTotalLoad),
          currentStart_(J_),
          bestStart_(J_) {
        for (int j = 0; j < J_; ++j) {
            currentStart_[j].assign(ins_.jobs[j].size(), -1);
            bestStart_[j].assign(ins_.jobs[j].size(), 0);
        }
    }

    ScheduleResult solve() {
        bestMakespan_ = std::numeric_limits<ll>::max();
        dfs(0, 0);
        ScheduleResult out;
        out.makespan = bestMakespan_;
        out.startTimes = std::move(bestStart_);
        return out;
    }

private:
    void dfs(int scheduled, ll currentMaxFinish) {
        if (scheduled == N_) {
            if (currentMaxFinish < bestMakespan_) {
                bestMakespan_ = currentMaxFinish;
                bestStart_ = currentStart_;
            }
            return;
        }

        ll lb = currentMaxFinish;

        for (int m = 0; m < M_; ++m) {
            lb = std::max(lb, machineReady_[m] + remMachine_[m]);
        }

        for (int j = 0; j < J_; ++j) {
            const int k = nextTask_[j];
            if (k < static_cast<int>(ins_.jobs[j].size())) {
                lb = std::max(lb, jobReady_[j] + ins_.remJobDur[j][k]);
            }
        }

        if (lb >= bestMakespan_) return;

        std::vector<int> candidates;
        candidates.reserve(J_);
        for (int j = 0; j < J_; ++j) {
            if (nextTask_[j] < static_cast<int>(ins_.jobs[j].size())) {
                candidates.push_back(j);
            }
        }

        std::sort(candidates.begin(), candidates.end(), [&](int a, int b) {
            const int ka = nextTask_[a];
            const int kb = nextTask_[b];
            const Task& ta = ins_.jobs[a][ka];
            const Task& tb = ins_.jobs[b][kb];
            const ll ea = std::max(jobReady_[a], machineReady_[ta.machine]);
            const ll eb = std::max(jobReady_[b], machineReady_[tb.machine]);
            if (ea != eb) return ea < eb;
            if (ta.duration != tb.duration) return ta.duration < tb.duration;
            return a < b;
        });

        for (int j : candidates) {
            const int k = nextTask_[j];
            const Task& t = ins_.jobs[j][k];

            const ll oldJR = jobReady_[j];
            const ll oldMR = machineReady_[t.machine];
            const ll oldRemMachine = remMachine_[t.machine];

            const ll start = std::max(jobReady_[j], machineReady_[t.machine]);
            const ll finish = start + static_cast<ll>(t.duration);

            currentStart_[j][k] = start;
            ++nextTask_[j];
            jobReady_[j] = finish;
            machineReady_[t.machine] = finish;
            remMachine_[t.machine] -= t.duration;

            dfs(scheduled + 1, std::max(currentMaxFinish, finish));

            remMachine_[t.machine] = oldRemMachine;
            machineReady_[t.machine] = oldMR;
            jobReady_[j] = oldJR;
            --nextTask_[j];
            currentStart_[j][k] = -1;
        }
    }

    const Instance& ins_;
    int J_;
    int M_;
    int N_;

    std::vector<int> nextTask_;
    std::vector<ll> jobReady_;
    std::vector<ll> machineReady_;
    std::vector<ll> remMachine_;

    std::vector<std::vector<ll>> currentStart_;
    std::vector<std::vector<ll>> bestStart_;
    ll bestMakespan_ = std::numeric_limits<ll>::max();
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    Instance ins;
    if (!(std::cin >> ins.jobsCount >> ins.machinesCount)) {
        return 0;
    }

    const int J = ins.jobsCount;
    const int M = ins.machinesCount;

    ins.jobs.assign(J, {});
    ins.machineTotalLoad.assign(M, 0);

    for (int j = 0; j < J; ++j) {
        int T;
        std::cin >> T;
        ins.jobs[j].reserve(T);
        for (int k = 0; k < T; ++k) {
            int m, d;
            std::cin >> m >> d;
            ins.jobs[j].push_back(Task{m, d});
            if (m >= 0 && m < M) {
                ins.machineTotalLoad[m] += d;
            }
            ++ins.totalTasks;
        }
    }

    ins.remJobDur.assign(J, {});
    for (int j = 0; j < J; ++j) {
        const int T = static_cast<int>(ins.jobs[j].size());
        ins.remJobDur[j].assign(T + 1, 0);
        for (int k = T - 1; k >= 0; --k) {
            ins.remJobDur[j][k] = ins.remJobDur[j][k + 1] + ins.jobs[j][k].duration;
        }
    }

    ScheduleResult best;

    const bool tinyInstance =
        (ins.totalTasks <= 18 && ins.jobsCount <= 8 && ins.machinesCount <= 8);

    if (ins.totalTasks == 0) {
        best.makespan = 0;
        best.startTimes.assign(J, {});
    } else if (tinyInstance) {
        ExactTinySolver solver(ins);
        best = solver.solve();
    } else {
        std::vector<int> policies;
        if (ins.totalTasks <= 4000) {
            policies = {1, 0, 2, 3};
        } else if (ins.totalTasks <= 50000) {
            policies = {1, 0, 2};
        } else {
            policies = {0};
        }

        for (int policy : policies) {
            ScheduleResult cur = buildHeuristicSchedule(ins, policy);
            if (cur.makespan < best.makespan) {
                best = std::move(cur);
            }
        }
    }

    for (int j = 0; j < J; ++j) {
        for (int k = 0; k < static_cast<int>(ins.jobs[j].size()); ++k) {
            if (k > 0) std::cout << ' ';
            std::cout << best.startTimes[j][k] << ' ' << ins.jobs[j][k].machine;
        }
        std::cout << '\n';
    }

    return 0;
}
