const fs = require('fs');

const raw = fs.readFileSync(0, 'utf8');
const data = raw.trim().length > 0 ? JSON.parse(raw) : {};

const depot = Array.isArray(data.depot) && data.depot.length >= 2 ? data.depot : [0, 0];
const depotX = Number.isFinite(Number(depot[0])) ? Number(depot[0]) : 0;
const depotY = Number.isFinite(Number(depot[1])) ? Number(depot[1]) : 0;

const packages = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

if (n === 0) {
  process.stdout.write('{"routes":[]}');
  process.exit(0);
}

let maxRoutes = Number.isFinite(Number(data.truckCount)) ? Math.floor(Number(data.truckCount)) : 1;
if (maxRoutes < 1) maxRoutes = 1;
if (maxRoutes > n) maxRoutes = n;

const ids = new Array(n);
const xs = new Float64Array(n);
const ys = new Float64Array(n);
const angles = new Float64Array(n);
const depotDist = new Float64Array(n);

let sumDepotDist = 0;
for (let i = 0; i < n; i++) {
  const p = packages[i] || {};
  ids[i] = Object.prototype.hasOwnProperty.call(p, 'id') ? p.id : i;

  const px = Number(p.x);
  const py = Number(p.y);
  const x = Number.isFinite(px) ? px : depotX;
  const y = Number.isFinite(py) ? py : depotY;

  xs[i] = x;
  ys[i] = y;

  const dx = x - depotX;
  const dy = y - depotY;
  angles[i] = Math.atan2(dy, dx);
  const d = Math.hypot(dx, dy);
  depotDist[i] = d;
  sumDepotDist += d;
}

const ordered = Array.from({ length: n }, (_, i) => i);
ordered.sort((a, b) => {
  const da = angles[a] - angles[b];
  if (da !== 0) return da;
  return depotDist[a] - depotDist[b];
});

if (n > 1) {
  const TWO_PI = Math.PI * 2;
  let bestGap = -1;
  let gapIdx = n - 1;

  for (let i = 0; i < n; i++) {
    const a = angles[ordered[i]];
    let b = angles[ordered[(i + 1) % n]];
    if (i === n - 1) b += TWO_PI;
    const gap = b - a;
    if (gap > bestGap) {
      bestGap = gap;
      gapIdx = i;
    }
  }

  const start = (gapIdx + 1) % n;
  if (start !== 0) {
    const rotated = ordered.slice(start).concat(ordered.slice(0, start));
    for (let i = 0; i < n; i++) ordered[i] = rotated[i];
  }
}

const avgDepotDist = sumDepotDist / n;
const distScale = avgDepotDist > 1e-9 ? avgDepotDist : 1;
const alpha = 0.35;

const prefix = new Float64Array(n);
let totalWeight = 0;
for (let i = 0; i < n; i++) {
  const idx = ordered[i];
  const w = 1 + alpha * (depotDist[idx] / distScale);
  totalWeight += w;
  prefix[i] = totalWeight;
}

const cuts = [];
let ptr = 0;
let prevCut = -1;
for (let r = 1; r < maxRoutes; r++) {
  const target = (totalWeight * r) / maxRoutes;
  const minPos = prevCut + 1;
  const maxPos = n - (maxRoutes - r) - 1;

  if (ptr < minPos) ptr = minPos;
  while (ptr < maxPos && prefix[ptr] < target) ptr++;

  let cut = ptr;
  if (cut > minPos) {
    const prev = cut - 1;
    const diffCur = Math.abs(prefix[cut] - target);
    const diffPrev = Math.abs(prefix[prev] - target);
    if (prev >= minPos && diffPrev <= diffCur) {
      cut = prev;
    }
  }

  if (cut < minPos) cut = minPos;
  if (cut > maxPos) cut = maxPos;

  cuts.push(cut);
  prevCut = cut;
}

const routeSegments = [];
let startPos = 0;
for (let i = 0; i < cuts.length; i++) {
  const endPos = cuts[i];
  routeSegments.push(ordered.slice(startPos, endPos + 1));
  startPos = endPos + 1;
}
routeSegments.push(ordered.slice(startPos));

function distNodeNode(a, b) {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return Math.hypot(dx, dy);
}

function distDepotOrNode(a, b) {
  if (a === -1 && b === -1) return 0;
  if (a === -1) return depotDist[b];
  if (b === -1) return depotDist[a];
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return Math.hypot(dx, dy);
}

function buildRouteFromSweepCycle(segment) {
  const m = segment.length;
  if (m <= 1) return segment.slice();

  let cycleLen = 0;
  for (let i = 0; i < m; i++) {
    const a = segment[i];
    const b = segment[(i + 1) % m];
    cycleLen += distNodeNode(a, b);
  }

  let bestEdge = 0;
  let bestCost = Infinity;
  for (let i = 0; i < m; i++) {
    const a = segment[i];
    const b = segment[(i + 1) % m];
    const removed = distNodeNode(a, b);
    const cost = cycleLen - removed + depotDist[a] + depotDist[b];
    if (cost < bestCost) {
      bestCost = cost;
      bestEdge = i;
    }
  }

  const route = new Array(m);
  let k = 0;
  for (let t = 1; t <= m; t++) {
    route[k++] = segment[(bestEdge + t) % m];
  }
  return route;
}

function twoOptWindow(route, window, maxPasses) {
  const m = route.length;
  if (m < 4) return;

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;

    for (let i = -1; i < m - 1; i++) {
      const a = i >= 0 ? route[i] : -1;
      const bi = i + 1;
      const b = route[bi];
      const maxJ = Math.min(m - 1, i + 1 + window);

      for (let j = i + 2; j <= maxJ; j++) {
        if (i === -1 && j === m - 1) continue;

        const c = route[j];
        const d = j + 1 < m ? route[j + 1] : -1;

        const oldCost = distDepotOrNode(a, b) + distDepotOrNode(c, d);
        const newCost = distDepotOrNode(a, c) + distDepotOrNode(b, d);

        if (newCost + 1e-9 < oldCost) {
          let l = bi;
          let r = j;
          while (l < r) {
            const tmp = route[l];
            route[l] = route[r];
            route[r] = tmp;
            l++;
            r--;
          }
          improved = true;
          break;
        }
      }

      if (improved) break;
    }

    if (!improved) break;
  }
}

const routes = [];
for (let i = 0; i < routeSegments.length; i++) {
  const segment = routeSegments[i];
  if (segment.length === 0) continue;

  const routeIdx = buildRouteFromSweepCycle(segment);
  const passes = routeIdx.length > 120 ? 1 : 2;
  twoOptWindow(routeIdx, 24, passes);

  const routeIds = new Array(routeIdx.length);
  for (let j = 0; j < routeIdx.length; j++) {
    routeIds[j] = ids[routeIdx[j]];
  }

  routes.push(routeIds);
}

process.stdout.write(JSON.stringify({ routes }));
