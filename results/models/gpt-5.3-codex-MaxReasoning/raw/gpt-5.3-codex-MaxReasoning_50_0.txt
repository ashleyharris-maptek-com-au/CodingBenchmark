#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

const float EPS = 1e-6;

vec3 computeBoundaryForce(vec3 p, float boundSize, float boundForce)
{
    float limit = boundSize * 0.4;
    float scale = max(boundSize * 0.5, EPS);
    vec3 f = vec3(0.0);

    if (p.x > limit) {
        f.x -= boundForce * ((p.x - limit) / scale);
    } else if (p.x < -limit) {
        f.x += boundForce * ((-limit - p.x) / scale);
    }

    if (p.y > limit) {
        f.y -= boundForce * ((p.y - limit) / scale);
    } else if (p.y < -limit) {
        f.y += boundForce * ((-limit - p.y) / scale);
    }

    if (p.z > limit) {
        f.z -= boundForce * ((p.z - limit) / scale);
    } else if (p.z < -limit) {
        f.z += boundForce * ((-limit - p.z) / scale);
    }

    return f;
}

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint n = counts.x;

    // Critical guard for over-dispatch.
    if (i >= n) {
        return;
    }

    uint base = i * 2u;
    vec3 pos = inData[base].xyz;
    vec3 vel = inData[base + 1u].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    float sepR2 = sepRadius * sepRadius;
    float alignR2 = alignRadius * alignRadius;
    float cohR2 = cohRadius * cohRadius;

    vec3 sepAccum = vec3(0.0);
    vec3 alignAccum = vec3(0.0);
    vec3 cohAccum = vec3(0.0);

    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < n; ++j) {
        if (j == i) {
            continue; // skip self
        }

        uint jBase = j * 2u;
        vec3 otherPos = inData[jBase].xyz;
        vec3 offset = pos - otherPos;
        float dist2 = dot(offset, offset);

        if (dist2 <= EPS) {
            continue; // skip zero-distance neighbors
        }

        if (dist2 < sepR2) {
            // Stronger repulsion for closer neighbors.
            sepAccum += offset / dist2;
            sepCount++;
        }

        if (dist2 < alignR2) {
            vec3 otherVel = inData[jBase + 1u].xyz;
            alignAccum += otherVel;
            alignCount++;
        }

        if (dist2 < cohR2) {
            cohAccum += otherPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        vec3 sepAvg = sepAccum / float(sepCount);
        accel += sepAvg * sepWeight;
    }

    if (alignCount > 0u) {
        vec3 avgVel = alignAccum / float(alignCount);
        vec3 alignSteer = avgVel - vel;
        accel += alignSteer * alignWeight;
    }

    if (cohCount > 0u) {
        vec3 center = cohAccum / float(cohCount);
        vec3 cohSteer = center - pos;
        accel += cohSteer * cohWeight;
    }

    accel += computeBoundaryForce(pos, boundSize, boundForce);

    vel += accel * dt;

    float speed = length(vel);
    if (speed > maxSpeed && speed > EPS) {
        vel = (vel / speed) * maxSpeed;
    }

    pos += vel * dt;

    outData[base] = vec4(pos, 0.0);
    outData[base + 1u] = vec4(vel, 0.0);
}
