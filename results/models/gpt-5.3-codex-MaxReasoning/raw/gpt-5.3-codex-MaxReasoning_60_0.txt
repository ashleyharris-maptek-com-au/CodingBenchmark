'use strict';

const fs = require('fs');

function toNonNegInt(value, fallback) {
  const x = Number(value);
  if (!Number.isFinite(x)) return fallback;
  if (x <= 0) return 0;
  return Math.floor(x);
}

class BinaryHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp;
  }

  size() {
    return this.a.length;
  }

  peek() {
    return this.a[0];
  }

  push(v) {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[i], a[p]) >= 0) break;
      const t = a[i];
      a[i] = a[p];
      a[p] = t;
      i = p;
    }
  }

  pop() {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop();
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      for (;;) {
        let l = i * 2 + 1;
        let r = l + 1;
        let best = i;
        if (l < a.length && this.cmp(a[l], a[best]) < 0) best = l;
        if (r < a.length && this.cmp(a[r], a[best]) < 0) best = r;
        if (best === i) break;
        const t = a[i];
        a[i] = a[best];
        a[best] = t;
        i = best;
      }
    }
    return top;
  }
}

function cmpAsc(a, b) {
  return a < b ? -1 : a > b ? 1 : 0;
}

function cmpDesc(a, b) {
  return a > b ? -1 : a < b ? 1 : 0;
}

function buildGraph(data) {
  const taskList = Array.isArray(data.tasks) ? data.tasks : [];
  const n = taskList.length;

  const ids = new Array(n);
  const duration = new Array(n);
  const weight = new Array(n);
  const deadline = new Array(n);
  const depsRaw = new Array(n);

  const idToIndex = new Map();

  for (let i = 0; i < n; i++) {
    const t = taskList[i] || {};
    let id = Object.prototype.hasOwnProperty.call(t, 'id') ? t.id : i;

    if (idToIndex.has(id)) {
      id = String(id) + '__' + i;
    }
    ids[i] = id;
    idToIndex.set(id, i);

    duration[i] = toNonNegInt(t.duration, 0);

    const w = Number(t.weight);
    weight[i] = Number.isFinite(w) ? Math.max(0, w) : 1;

    const d = Number(t.deadline);
    deadline[i] = Number.isFinite(d) ? Math.floor(d) : Infinity;

    depsRaw[i] = Array.isArray(t.deps) ? t.deps : [];
  }

  const succ = Array.from({ length: n }, () => []);
  const indeg = new Array(n).fill(0);

  for (let i = 0; i < n; i++) {
    const deps = depsRaw[i];
    const seen = new Set();

    for (let k = 0; k < deps.length; k++) {
      const depId = deps[k];
      if (seen.has(depId)) continue;
      seen.add(depId);

      const p = idToIndex.get(depId);
      if (p === undefined || p === i) continue;
      succ[p].push(i);
      indeg[i]++;
    }
  }

  return { n, ids, duration, weight, deadline, succ, indeg };
}

function topoSort(n, succ, indeg) {
  const indegWork = indeg.slice();
  const q = [];
  let head = 0;

  for (let i = 0; i < n; i++) {
    if (indegWork[i] === 0) q.push(i);
  }

  while (head < q.length) {
    const u = q[head++];
    const out = succ[u];
    for (let k = 0; k < out.length; k++) {
      const v = out[k];
      indegWork[v]--;
      if (indegWork[v] === 0) q.push(v);
    }
  }

  return { order: q, acyclic: q.length === n };
}

function computeMetrics(graph, topoOrder) {
  const { n, succ, duration, weight, deadline } = graph;

  const critLen = duration.slice();
  const chainWeight = weight.slice();
  const depth = new Array(n).fill(1);
  const weightPerTime = new Array(n);
  const latestStart = new Array(n);

  for (let i = topoOrder.length - 1; i >= 0; i--) {
    const u = topoOrder[i];
    const out = succ[u];

    let bestCrit = duration[u];
    let bestWeight = weight[u];
    let bestDepth = 1;

    for (let k = 0; k < out.length; k++) {
      const v = out[k];

      const cCrit = duration[u] + critLen[v];
      if (cCrit > bestCrit) bestCrit = cCrit;

      const cWeight = weight[u] + chainWeight[v];
      if (cWeight > bestWeight) bestWeight = cWeight;

      const cDepth = 1 + depth[v];
      if (cDepth > bestDepth) bestDepth = cDepth;
    }

    critLen[u] = bestCrit;
    chainWeight[u] = bestWeight;
    depth[u] = bestDepth;
  }

  for (let i = 0; i < n; i++) {
    weightPerTime[i] = weight[i] / (duration[i] + 1);
    latestStart[i] = Number.isFinite(deadline[i]) ? deadline[i] - critLen[i] : Infinity;
  }

  return { critLen, chainWeight, depth, weightPerTime, latestStart };
}

function makeReadyComparator(rule, metrics, graph) {
  const { duration, weight, deadline } = graph;
  const { critLen, chainWeight, depth, weightPerTime, latestStart } = metrics;

  return function compare(i, j) {
    let c;

    switch (rule) {
      case 0:
        c = cmpAsc(latestStart[i], latestStart[j]);
        if (c) return c;
        c = cmpDesc(weightPerTime[i], weightPerTime[j]);
        if (c) return c;
        c = cmpDesc(critLen[i], critLen[j]);
        if (c) return c;
        c = cmpAsc(duration[i], duration[j]);
        if (c) return c;
        break;

      case 1:
        c = cmpDesc(weightPerTime[i], weightPerTime[j]);
        if (c) return c;
        c = cmpAsc(latestStart[i], latestStart[j]);
        if (c) return c;
        c = cmpDesc(chainWeight[i], chainWeight[j]);
        if (c) return c;
        c = cmpAsc(duration[i], duration[j]);
        if (c) return c;
        break;

      case 2:
        c = cmpDesc(critLen[i], critLen[j]);
        if (c) return c;
        c = cmpAsc(latestStart[i], latestStart[j]);
        if (c) return c;
        c = cmpDesc(weight[i], weight[j]);
        if (c) return c;
        c = cmpAsc(duration[i], duration[j]);
        if (c) return c;
        break;

      case 3:
        c = cmpAsc(deadline[i], deadline[j]);
        if (c) return c;
        c = cmpDesc(weight[i], weight[j]);
        if (c) return c;
        c = cmpDesc(depth[i], depth[j]);
        if (c) return c;
        c = cmpAsc(duration[i], duration[j]);
        if (c) return c;
        break;

      default:
        c = cmpDesc(chainWeight[i], chainWeight[j]);
        if (c) return c;
        c = cmpAsc(latestStart[i], latestStart[j]);
        if (c) return c;
        c = cmpDesc(weightPerTime[i], weightPerTime[j]);
        if (c) return c;
        c = cmpDesc(critLen[i], critLen[j]);
        if (c) return c;
        c = cmpAsc(duration[i], duration[j]);
        if (c) return c;
        break;
    }

    return i - j;
  };
}

function buildSchedule(graph, metrics, machineCount, rule) {
  const { n, duration, deadline, weight, succ, indeg } = graph;

  const indegWork = indeg.slice();
  const release = new Array(n).fill(0);
  const start = new Array(n).fill(0);
  const finish = new Array(n).fill(0);
  const machine = new Array(n).fill(0);

  const readyCmp = makeReadyComparator(rule, metrics, graph);

  const readyHeap = new BinaryHeap((a, b) => readyCmp(a, b));
  const futureHeap = new BinaryHeap((a, b) => {
    if (a.release !== b.release) return a.release - b.release;
    const c = readyCmp(a.task, b.task);
    if (c !== 0) return c;
    return a.task - b.task;
  });
  const machineHeap = new BinaryHeap((a, b) => {
    if (a.time !== b.time) return a.time - b.time;
    return a.machine - b.machine;
  });

  for (let m = 0; m < machineCount; m++) {
    machineHeap.push({ time: 0, machine: m });
  }

  for (let i = 0; i < n; i++) {
    if (indegWork[i] === 0) {
      futureHeap.push({ release: 0, task: i });
    }
  }

  let scheduled = 0;
  let globalTime = 0;

  while (scheduled < n) {
    if (machineHeap.size() === 0) break;

    const mac = machineHeap.pop();
    let t = mac.time > globalTime ? mac.time : globalTime;

    while (futureHeap.size() > 0 && futureHeap.peek().release <= t) {
      readyHeap.push(futureHeap.pop().task);
    }

    if (readyHeap.size() === 0) {
      if (futureHeap.size() === 0) {
        break;
      }

      const nextRelease = futureHeap.peek().release;
      if (nextRelease > t) t = nextRelease;
      if (t > globalTime) globalTime = t;

      while (futureHeap.size() > 0 && futureHeap.peek().release <= t) {
        readyHeap.push(futureHeap.pop().task);
      }

      if (readyHeap.size() === 0) {
        machineHeap.push({ time: t, machine: mac.machine });
        continue;
      }
    }

    if (t > globalTime) globalTime = t;

    const task = readyHeap.pop();
    const st = t;
    const fn = st + duration[task];

    start[task] = st;
    finish[task] = fn;
    machine[task] = mac.machine;

    scheduled++;
    machineHeap.push({ time: fn, machine: mac.machine });

    const out = succ[task];
    for (let k = 0; k < out.length; k++) {
      const v = out[k];
      if (fn > release[v]) release[v] = fn;
      indegWork[v]--;
      if (indegWork[v] === 0) {
        futureHeap.push({ release: release[v], task: v });
      }
    }
  }

  if (scheduled !== n) {
    return { ok: false };
  }

  let makespan = 0;
  let weightedTardiness = 0;
  let weightedCompletion = 0;

  for (let i = 0; i < n; i++) {
    const c = finish[i];
    if (c > makespan) makespan = c;

    if (Number.isFinite(deadline[i])) {
      const late = c - deadline[i];
      if (late > 0) weightedTardiness += late * weight[i];
    }

    weightedCompletion += c * weight[i];
  }

  const objective = weightedTardiness + makespan;

  return {
    ok: true,
    start,
    machine,
    makespan,
    weightedTardiness,
    weightedCompletion,
    objective,
  };
}

function compareSchedules(a, b) {
  if (a.objective !== b.objective) return a.objective < b.objective;
  if (a.weightedTardiness !== b.weightedTardiness) return a.weightedTardiness < b.weightedTardiness;
  if (a.makespan !== b.makespan) return a.makespan < b.makespan;
  return a.weightedCompletion < b.weightedCompletion;
}

function main() {
  const raw = fs.readFileSync(0, 'utf8');
  if (!raw.trim()) {
    process.stdout.write('{"schedule":[]}');
    return;
  }

  const data = JSON.parse(raw);
  const graph = buildGraph(data || {});

  if (graph.n === 0) {
    process.stdout.write('{"schedule":[]}');
    return;
  }

  const machineCountRaw = Number((data || {}).machineCount);
  const machineCount = Number.isFinite(machineCountRaw)
    ? Math.max(1, Math.floor(machineCountRaw))
    : 1;

  const topo = topoSort(graph.n, graph.succ, graph.indeg);
  if (!topo.acyclic) {
    const used = new Array(graph.n).fill(false);
    const order = [];

    for (let i = 0; i < topo.order.length; i++) {
      const v = topo.order[i];
      used[v] = true;
      order.push(graph.ids[v]);
    }

    for (let i = 0; i < graph.n; i++) {
      if (!used[i]) order.push(graph.ids[i]);
    }

    process.stdout.write(JSON.stringify({ order }));
    return;
  }

  const metrics = computeMetrics(graph, topo.order);

  const rules = [0, 1, 2, 3, 4];
  let best = null;

  for (let r = 0; r < rules.length; r++) {
    const cand = buildSchedule(graph, metrics, machineCount, rules[r]);
    if (!cand.ok) continue;
    if (best === null || compareSchedules(cand, best)) {
      best = cand;
    }
  }

  if (best === null) {
    const order = topo.order.map((idx) => graph.ids[idx]);
    process.stdout.write(JSON.stringify({ order }));
    return;
  }

  const schedule = new Array(graph.n);
  for (let i = 0; i < graph.n; i++) {
    schedule[i] = {
      id: graph.ids[i],
      start: best.start[i],
      machine: best.machine[i],
    };
  }

  schedule.sort((a, b) => {
    if (a.start !== b.start) return a.start - b.start;
    if (a.machine !== b.machine) return a.machine - b.machine;
    const sa = String(a.id);
    const sb = String(b.id);
    return sa < sb ? -1 : sa > sb ? 1 : 0;
  });

  process.stdout.write(JSON.stringify({ schedule }));
}

main();
