use std::collections::{HashSet, VecDeque};
use std::io::{self, Read};
use std::time::{Duration, Instant};

const DIRS: [(i32, i32); 4] = [(1, 0), (-1, 0), (0, 1), (0, -1)];

#[derive(Clone, Copy)]
struct SplitMix64 {
    state: u64,
}

impl SplitMix64 {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }

    fn next_u64(&mut self) -> u64 {
        let mut z = self.state.wrapping_add(0x9e3779b97f4a7c15);
        self.state = z;
        z = (z ^ (z >> 30)).wrapping_mul(0xbf58476d1ce4e5b9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94d049bb133111eb);
        z ^ (z >> 31)
    }

    fn next_u32(&mut self) -> u32 {
        (self.next_u64() >> 32) as u32
    }
}

struct GridSolver {
    w: usize,
    h: usize,
    require_cycle: bool,
    n: usize,
    start: usize,
    coords: Vec<(i32, i32)>,
    adj: Vec<Vec<usize>>,
    color: Vec<u8>,
    adj_to_start: Vec<bool>,
    obstacle_count: usize,
    started_at: Instant,
    time_limit: Duration,
    rng: SplitMix64,
}

impl GridSolver {
    fn new(w: usize, h: usize, require_cycle: bool, blocked: &[(usize, usize)]) -> Option<Self> {
        if w == 0 || h == 0 {
            return None;
        }

        let total = w.checked_mul(h)?;
        let mut open = vec![true; total];
        for &(x, y) in blocked {
            if x >= w || y >= h {
                continue;
            }
            open[y * w + x] = false;
        }

        if !open[0] {
            return None;
        }

        let mut id_of_cell = vec![usize::MAX; total];
        let mut coords = Vec::new();
        let mut color = Vec::new();

        for y in 0..h {
            for x in 0..w {
                if open[y * w + x] {
                    let id = coords.len();
                    id_of_cell[y * w + x] = id;
                    coords.push((x as i32, y as i32));
                    color.push(((x + y) & 1) as u8);
                }
            }
        }

        let n = coords.len();
        let start = id_of_cell[0];
        if start == usize::MAX || n == 0 {
            return None;
        }

        let mut adj = vec![Vec::new(); n];
        for y in 0..h {
            for x in 0..w {
                if !open[y * w + x] {
                    continue;
                }
                let u = id_of_cell[y * w + x];
                for (dx, dy) in DIRS {
                    let nx = x as i32 + dx;
                    let ny = y as i32 + dy;
                    if nx < 0 || ny < 0 {
                        continue;
                    }
                    let nxu = nx as usize;
                    let nyu = ny as usize;
                    if nxu >= w || nyu >= h {
                        continue;
                    }
                    if !open[nyu * w + nxu] {
                        continue;
                    }
                    let v = id_of_cell[nyu * w + nxu];
                    if v != usize::MAX {
                        adj[u].push(v);
                    }
                }
            }
        }

        let mut adj_to_start = vec![false; n];
        for &v in &adj[start] {
            adj_to_start[v] = true;
        }

        Some(Self {
            w,
            h,
            require_cycle,
            n,
            start,
            coords,
            adj,
            color,
            adj_to_start,
            obstacle_count: blocked.len(),
            started_at: Instant::now(),
            time_limit: Duration::from_secs(295),
            rng: SplitMix64::new(
                ((w as u64) << 48)
                    ^ ((h as u64) << 32)
                    ^ (blocked.len() as u64)
                    ^ if require_cycle { 0x9e3779b97f4a7c15 } else { 0 },
            ),
        })
    }

    fn solve(&mut self) -> Option<Vec<usize>> {
        if self.n == 1 {
            if self.require_cycle {
                return None;
            }
            return Some(vec![self.start]);
        }

        if self.obstacle_count == 0 && self.n == self.w * self.h {
            if self.require_cycle {
                return self.construct_open_cycle();
            }
            return Some(self.construct_open_snake());
        }

        if !self.static_checks() {
            return None;
        }

        let deadline = self.started_at + self.time_limit;

        if self.n <= 62 {
            if let Some(path) = self.solve_exact_u64(deadline) {
                return Some(path);
            }
        }

        let deterministic_cap = if self.n <= 260 {
            120_000_000u64
        } else {
            40_000_000u64
        };

        if let Some(path) = self.search_iterative(false, true, deterministic_cap, deadline) {
            return Some(path);
        }

        let mut attempts = if self.n <= 300 { 16 } else { 40 };
        while attempts > 0 && Instant::now() < deadline {
            attempts -= 1;
            let cap = if self.n <= 300 {
                15_000_000
            } else if self.n <= 700 {
                8_000_000
            } else {
                3_000_000
            };
            let strong = self.n <= 450;
            if let Some(path) = self.search_iterative(true, strong, cap, deadline) {
                return Some(path);
            }
        }

        None
    }

    fn construct_open_snake(&self) -> Vec<usize> {
        let mut out = Vec::with_capacity(self.w * self.h);
        for y in 0..self.h {
            if (y & 1) == 0 {
                for x in 0..self.w {
                    out.push(y * self.w + x);
                }
            } else {
                for x in (0..self.w).rev() {
                    out.push(y * self.w + x);
                }
            }
        }
        out
    }

    fn construct_open_cycle(&self) -> Option<Vec<usize>> {
        let mut coords = if self.w % 2 == 0 {
            self.cycle_even_width(self.w, self.h)?
        } else if self.h % 2 == 0 {
            let transposed = self.cycle_even_width(self.h, self.w)?;
            transposed.into_iter().map(|(x, y)| (y, x)).collect()
        } else {
            return None;
        };

        if coords.len() != self.w * self.h {
            return None;
        }

        if coords[0] != (0, 0) {
            return None;
        }

        let (lx, ly) = coords[coords.len() - 1];
        if (lx as i32 - 0).abs() + (ly as i32 - 0).abs() != 1 {
            return None;
        }

        let mut seen = vec![false; self.w * self.h];
        let mut ids = Vec::with_capacity(coords.len());
        for (x, y) in coords.drain(..) {
            if x >= self.w || y >= self.h {
                return None;
            }
            let id = y * self.w + x;
            if seen[id] {
                return None;
            }
            seen[id] = true;
            ids.push(id);
        }
        Some(ids)
    }

    fn cycle_even_width(&self, width: usize, height: usize) -> Option<Vec<(usize, usize)>> {
        if width % 2 != 0 {
            return None;
        }
        if width == 0 || height == 0 {
            return None;
        }

        if height == 1 {
            if width == 2 {
                return Some(vec![(0, 0), (1, 0)]);
            }
            return None;
        }

        let mut out = Vec::with_capacity(width * height);

        for y in 0..height {
            out.push((0, y));
        }

        for x in 1..width {
            if (x & 1) == 1 {
                for y in (1..height).rev() {
                    out.push((x, y));
                }
            } else {
                for y in 1..height {
                    out.push((x, y));
                }
            }
        }

        for x in (1..width).rev() {
            out.push((x, 0));
        }

        if out.len() != width * height {
            return None;
        }

        Some(out)
    }

    fn static_checks(&self) -> bool {
        if !self.is_connected() {
            return false;
        }

        let mut count = [0usize; 2];
        for &c in &self.color {
            count[c as usize] += 1;
        }

        let start_color = self.color[self.start] as usize;
        if self.require_cycle {
            if self.n % 2 == 1 {
                return false;
            }
            if count[0] != count[1] {
                return false;
            }
        } else if self.n % 2 == 0 {
            if count[0] != count[1] {
                return false;
            }
        } else {
            let major = if count[0] > count[1] { 0 } else { 1 };
            if start_color != major {
                return false;
            }
        }

        if self.n > 1 && self.adj[self.start].is_empty() {
            return false;
        }

        if self.require_cycle {
            if self.n > 2 && self.adj[self.start].len() < 2 {
                return false;
            }
            if self.n > 2 {
                for v in 0..self.n {
                    if self.adj[v].len() < 2 {
                        return false;
                    }
                }
            }
        } else {
            let mut leaf_not_start = 0usize;
            for v in 0..self.n {
                let d = self.adj[v].len();
                if d == 0 {
                    return false;
                }
                if v != self.start && d == 1 {
                    leaf_not_start += 1;
                    if leaf_not_start > 1 {
                        return false;
                    }
                }
            }
        }

        let comps_after = self.components_after_removal();
        if self.require_cycle {
            for &c in &comps_after {
                if c > 1 {
                    return false;
                }
            }
        } else {
            if comps_after[self.start] > 1 {
                return false;
            }
            for v in 0..self.n {
                if v != self.start && comps_after[v] > 2 {
                    return false;
                }
            }
        }

        true
    }

    fn is_connected(&self) -> bool {
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        seen[self.start] = true;
        q.push_back(self.start);
        let mut got = 0usize;

        while let Some(u) = q.pop_front() {
            got += 1;
            for &v in &self.adj[u] {
                if !seen[v] {
                    seen[v] = true;
                    q.push_back(v);
                }
            }
        }

        got == self.n
    }

    fn components_after_removal(&self) -> Vec<usize> {
        let n = self.n;
        let mut disc = vec![usize::MAX; n];
        let mut low = vec![0usize; n];
        let mut parent = vec![usize::MAX; n];
        let mut comps = vec![0usize; n];
        let mut time = 0usize;

        fn dfs(
            u: usize,
            time: &mut usize,
            disc: &mut [usize],
            low: &mut [usize],
            parent: &mut [usize],
            comps: &mut [usize],
            adj: &[Vec<usize>],
        ) {
            disc[u] = *time;
            low[u] = *time;
            *time += 1;

            let mut child_count = 0usize;
            let mut split_children = 0usize;

            for &v in &adj[u] {
                if disc[v] == usize::MAX {
                    parent[v] = u;
                    child_count += 1;
                    dfs(v, time, disc, low, parent, comps, adj);
                    low[u] = low[u].min(low[v]);
                    if parent[u] != usize::MAX && low[v] >= disc[u] {
                        split_children += 1;
                    }
                } else if v != parent[u] {
                    low[u] = low[u].min(disc[v]);
                }
            }

            if parent[u] == usize::MAX {
                comps[u] = child_count;
            } else {
                comps[u] = 1 + split_children;
            }
        }

        dfs(
            self.start,
            &mut time,
            &mut disc,
            &mut low,
            &mut parent,
            &mut comps,
            &self.adj,
        );

        comps
    }

    fn solve_exact_u64(&self, deadline: Instant) -> Option<Vec<usize>> {
        if self.n == 0 || self.n > 62 {
            return None;
        }

        let mut adj_mask = vec![0u64; self.n];
        for u in 0..self.n {
            let mut m = 0u64;
            for &v in &self.adj[u] {
                m |= 1u64 << v;
            }
            adj_mask[u] = m;
        }

        let full = if self.n == 64 {
            u64::MAX
        } else {
            (1u64 << self.n) - 1
        };

        let mut path = Vec::with_capacity(self.n);
        path.push(self.start);
        let start_mask = 1u64 << self.start;
        let mut dead = HashSet::<u128>::new();
        let mut timed_out = false;

        let ok = self.dfs_exact_u64(
            self.start,
            start_mask,
            full,
            &adj_mask,
            &mut path,
            &mut dead,
            deadline,
            &mut timed_out,
        );

        if ok {
            Some(path)
        } else {
            None
        }
    }

    fn dfs_exact_u64(
        &self,
        cur: usize,
        visited: u64,
        full: u64,
        adj_mask: &[u64],
        path: &mut Vec<usize>,
        dead: &mut HashSet<u128>,
        deadline: Instant,
        timed_out: &mut bool,
    ) -> bool {
        if Instant::now() >= deadline {
            *timed_out = true;
            return false;
        }

        if path.len() == self.n {
            if self.require_cycle {
                return self.adj_to_start[cur];
            }
            return true;
        }

        let key = ((visited as u128) << 7) | (cur as u128);
        if dead.contains(&key) {
            return false;
        }

        let mut cands_mask = adj_mask[cur] & !visited;
        if cands_mask == 0 {
            dead.insert(key);
            return false;
        }

        let mut cands = Vec::new();
        while cands_mask != 0 {
            let b = cands_mask.trailing_zeros() as usize;
            cands_mask &= cands_mask - 1;

            if self.require_cycle && path.len() + 1 == self.n && !self.adj_to_start[b] {
                continue;
            }

            let next_visited = visited | (1u64 << b);
            if !self.prune_u64(b, next_visited, full, adj_mask) {
                continue;
            }

            let onward = (adj_mask[b] & !next_visited).count_ones() as i32;
            cands.push((onward, b));
        }

        cands.sort_by(|a, b| b.0.cmp(&a.0));

        for (_, nxt) in cands {
            path.push(nxt);
            let next_visited = visited | (1u64 << nxt);
            if self.dfs_exact_u64(
                nxt,
                next_visited,
                full,
                adj_mask,
                path,
                dead,
                deadline,
                timed_out,
            ) {
                return true;
            }
            path.pop();
            if *timed_out {
                return false;
            }
        }

        dead.insert(key);
        false
    }

    fn prune_u64(&self, cur: usize, visited: u64, full: u64, adj_mask: &[u64]) -> bool {
        let rem = full & !visited;
        let rem_count = rem.count_ones() as usize;

        if rem_count == 0 {
            return !self.require_cycle || self.adj_to_start[cur];
        }

        if (adj_mask[cur] & rem) == 0 {
            return false;
        }

        let mut leaf_count = 0usize;
        let mut leaf_id = usize::MAX;

        let mut bits = rem;
        while bits != 0 {
            let v = bits.trailing_zeros() as usize;
            bits &= bits - 1;
            let mut deg = (adj_mask[v] & rem).count_ones() as usize;
            if (adj_mask[v] & (1u64 << cur)) != 0 {
                deg += 1;
            }
            if deg == 0 {
                return false;
            }
            if deg == 1 {
                leaf_count += 1;
                leaf_id = v;
                if leaf_count > 1 {
                    return false;
                }
            }
        }

        if self.require_cycle {
            let mut possible_end = false;
            let mut bits2 = rem;
            while bits2 != 0 {
                let v = bits2.trailing_zeros() as usize;
                bits2 &= bits2 - 1;
                if self.adj_to_start[v] {
                    possible_end = true;
                    break;
                }
            }
            if !possible_end {
                return false;
            }
            if leaf_count == 1 && !self.adj_to_start[leaf_id] {
                return false;
            }
        }

        if rem_count > 1 {
            let seed = rem.trailing_zeros() as usize;
            let mut seen = 0u64;
            let mut stack = 1u64 << seed;
            while stack != 0 {
                let v = stack.trailing_zeros() as usize;
                stack &= stack - 1;
                if (seen & (1u64 << v)) != 0 {
                    continue;
                }
                seen |= 1u64 << v;
                let nexts = adj_mask[v] & rem & !seen;
                stack |= nexts;
            }
            if seen != rem {
                return false;
            }
        }

        true
    }

    fn search_iterative(
        &mut self,
        randomize: bool,
        strong_prune: bool,
        max_steps: u64,
        deadline: Instant,
    ) -> Option<Vec<usize>> {
        let mut visited = vec![false; self.n];
        let mut path = Vec::with_capacity(self.n);
        let mut choices: Vec<Vec<usize>> = Vec::with_capacity(self.n);

        visited[self.start] = true;
        path.push(self.start);

        let mut steps = 0u64;

        loop {
            if (steps & 4095) == 0 && Instant::now() >= deadline {
                return None;
            }
            if steps >= max_steps {
                return None;
            }

            let cur = *path.last().unwrap();
            if path.len() == self.n {
                if !self.require_cycle || self.adj_to_start[cur] {
                    return Some(path);
                }
                if !Self::backtrack(&mut path, &mut visited, &mut choices) {
                    return None;
                }
                continue;
            }

            let depth = path.len() - 1;
            if choices.len() == depth {
                let mut cands = self.build_candidates(
                    cur,
                    path.len(),
                    &mut visited,
                    randomize,
                    strong_prune,
                    deadline,
                    &mut steps,
                );
                choices.push(std::mem::take(&mut cands));
            }

            if let Some(next) = choices[depth].pop() {
                if visited[next] {
                    continue;
                }
                visited[next] = true;
                path.push(next);
                steps += 1;
            } else {
                if !Self::backtrack(&mut path, &mut visited, &mut choices) {
                    return None;
                }
                steps += 1;
            }
        }
    }

    fn backtrack(
        path: &mut Vec<usize>,
        visited: &mut [bool],
        choices: &mut Vec<Vec<usize>>,
    ) -> bool {
        if path.len() <= 1 {
            return false;
        }
        let popped = path.pop().unwrap();
        visited[popped] = false;
        while choices.len() > path.len() {
            choices.pop();
        }
        true
    }

    fn build_candidates(
        &mut self,
        cur: usize,
        path_len: usize,
        visited: &mut [bool],
        randomize: bool,
        strong_prune: bool,
        deadline: Instant,
        steps: &mut u64,
    ) -> Vec<usize> {
        let mut scored = Vec::<(i32, usize)>::new();

        for &next in &self.adj[cur] {
            if visited[next] {
                continue;
            }

            if self.require_cycle && path_len + 1 == self.n && !self.adj_to_start[next] {
                continue;
            }

            let onward = self.count_unvisited_neighbors(next, visited);
            if path_len + 1 < self.n && onward == 0 {
                continue;
            }

            if strong_prune {
                visited[next] = true;
                let ok = self.prune_state(next, visited, path_len + 1, deadline, steps);
                visited[next] = false;
                if !ok {
                    continue;
                }
            }

            let mut score = (onward as i32) * 8 + self.adj[next].len() as i32;
            if self.require_cycle {
                if self.adj_to_start[next] && path_len + 1 < self.n - 1 {
                    score += 3;
                }
                if !self.adj_to_start[next] && path_len + 2 == self.n {
                    score += 1000;
                }
            }

            if randomize {
                score = score * 16 + (self.rng.next_u32() as i32 & 15);
            }

            scored.push((score, next));
        }

        scored.sort_by(|a, b| b.0.cmp(&a.0));
        scored.into_iter().map(|(_, v)| v).collect()
    }

    fn count_unvisited_neighbors(&self, u: usize, visited: &[bool]) -> usize {
        let mut cnt = 0usize;
        for &v in &self.adj[u] {
            if !visited[v] {
                cnt += 1;
            }
        }
        cnt
    }

    fn prune_state(
        &self,
        cur: usize,
        visited: &[bool],
        path_len: usize,
        deadline: Instant,
        steps: &mut u64,
    ) -> bool {
        if (*steps & 2047) == 0 && Instant::now() >= deadline {
            return false;
        }

        if path_len == self.n {
            return !self.require_cycle || self.adj_to_start[cur];
        }

        let rem = self.n - path_len;

        if self.count_unvisited_neighbors(cur, visited) == 0 {
            return false;
        }

        let mut first_unvisited = usize::MAX;
        let mut leaf_count = 0usize;
        let mut leaf_id = usize::MAX;

        let mut even_count = if self.color[cur] == 0 { 1 } else { 0 };
        let mut odd_count = if self.color[cur] == 1 { 1 } else { 0 };

        let m = rem + 1;
        let expected_end_color = if (m & 1) == 0 {
            1 - self.color[cur]
        } else {
            self.color[cur]
        };

        let mut has_adj_start = false;
        let mut has_adj_start_expected_color = false;

        for v in 0..self.n {
            if visited[v] {
                continue;
            }

            if first_unvisited == usize::MAX {
                first_unvisited = v;
            }

            if self.color[v] == 0 {
                even_count += 1;
            } else {
                odd_count += 1;
            }

            let mut deg = 0usize;
            for &nb in &self.adj[v] {
                if !visited[nb] || nb == cur {
                    deg += 1;
                }
            }

            if deg == 0 {
                return false;
            }

            if deg == 1 {
                leaf_count += 1;
                leaf_id = v;
                if leaf_count > 1 {
                    return false;
                }
            }

            if self.adj_to_start[v] {
                has_adj_start = true;
                if self.color[v] == expected_end_color {
                    has_adj_start_expected_color = true;
                }
            }
        }

        if (m & 1) == 0 {
            if even_count != odd_count {
                return false;
            }
        } else if self.color[cur] == 0 {
            if even_count != odd_count + 1 {
                return false;
            }
        } else if odd_count != even_count + 1 {
            return false;
        }

        if self.require_cycle {
            if !has_adj_start || !has_adj_start_expected_color {
                return false;
            }
            if leaf_count == 1 && !self.adj_to_start[leaf_id] {
                return false;
            }
        }

        if rem > 1 {
            let seed = first_unvisited;
            if seed == usize::MAX {
                return false;
            }
            let mut seen = vec![false; self.n];
            let mut q = VecDeque::new();
            seen[seed] = true;
            q.push_back(seed);
            let mut reached = 0usize;

            while let Some(u) = q.pop_front() {
                reached += 1;
                for &v in &self.adj[u] {
                    if !visited[v] && !seen[v] {
                        seen[v] = true;
                        q.push_back(v);
                    }
                }
            }

            if reached != rem {
                return false;
            }
        }

        true
    }

    fn output_path(&self, path: &[usize]) {
        let mut out = String::new();
        out.push_str(&format!("{}\n", path.len()));
        for &id in path {
            let (x, y) = if self.obstacle_count == 0 && self.n == self.w * self.h {
                ((id % self.w) as i32, (id / self.w) as i32)
            } else {
                self.coords[id]
            };
            out.push_str(&format!("{} {}\n", x, y));
        }
        print!("{}", out);
    }
}

fn parse_input() -> Option<(usize, usize, usize, bool, Vec<(usize, usize)>)> {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).ok()?;
    let mut it = input.split_whitespace();

    let w: usize = it.next()?.parse().ok()?;
    let h: usize = it.next()?.parse().ok()?;
    let obstacle_count: usize = it.next()?.parse().ok()?;
    let require_cycle_i: i32 = it.next()?.parse().ok()?;
    let require_cycle = require_cycle_i != 0;

    let mut blocked = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        let x: usize = match it.next() {
            Some(s) => s.parse().ok()?,
            None => return None,
        };
        let y: usize = match it.next() {
            Some(s) => s.parse().ok()?,
            None => return None,
        };
        blocked.push((x, y));
    }

    Some((w, h, obstacle_count, require_cycle, blocked))
}

fn main() {
    let parsed = parse_input();
    if parsed.is_none() {
        println!("0");
        return;
    }

    let (w, h, _obstacle_count, require_cycle, blocked) = parsed.unwrap();

    let mut solver = match GridSolver::new(w, h, require_cycle, &blocked) {
        Some(s) => s,
        None => {
            println!("0");
            return;
        }
    };

    match solver.solve() {
        Some(path) => solver.output_path(&path),
        None => println!("0"),
    }
}
