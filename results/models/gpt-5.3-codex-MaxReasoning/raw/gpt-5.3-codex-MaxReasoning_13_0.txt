#include <algorithm>
#include <array>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <vector>

namespace {

void trim_cr(std::string& s) {
    if (!s.empty() && s.back() == '\r') {
        s.pop_back();
    }
}

std::string lcs_two_small_dp(const std::string& a, const std::string& b) {
    if (a.empty() || b.empty()) {
        return "";
    }

    std::vector<int> prev(b.size() + 1, 0), cur(b.size() + 1, 0);
    std::size_t best_len = 0;
    std::size_t best_end = 0;

    for (std::size_t i = 1; i <= a.size(); ++i) {
        for (std::size_t j = 1; j <= b.size(); ++j) {
            if (a[i - 1] == b[j - 1]) {
                cur[j] = prev[j - 1] + 1;
                if (static_cast<std::size_t>(cur[j]) > best_len) {
                    best_len = static_cast<std::size_t>(cur[j]);
                    best_end = i;
                }
            } else {
                cur[j] = 0;
            }
        }
        std::swap(prev, cur);
        std::fill(cur.begin(), cur.end(), 0);
    }

    return a.substr(best_end - best_len, best_len);
}

class DenseSAM {
public:
    explicit DenseSAM(const std::string& base) : base_(base) {
        states_.reserve(base.size() * 2 + 1);
        states_.push_back(State{});
        last_ = 0;

        for (unsigned char c : base) {
            extend(c);
        }

        build_order();
        for (auto& st : states_) {
            st.best = st.len;
        }
    }

    void process_string(const std::string& t) {
        int v = 0;
        int l = 0;

        for (unsigned char c : t) {
            while (v != 0 && states_[v].next[c] == -1) {
                v = states_[v].link;
                l = states_[v].len;
            }

            int nx = states_[v].next[c];
            if (nx != -1) {
                v = nx;
                ++l;
            } else {
                v = 0;
                l = 0;
            }

            if (l > states_[v].cur) {
                states_[v].cur = l;
            }
        }

        for (int i = static_cast<int>(order_.size()) - 1; i > 0; --i) {
            int s = order_[i];
            int p = states_[s].link;
            if (p >= 0) {
                int push_len = std::min(states_[s].cur, states_[p].len);
                if (push_len > states_[p].cur) {
                    states_[p].cur = push_len;
                }
            }
        }

        for (auto& st : states_) {
            st.best = std::min(st.best, st.cur);
            st.cur = 0;
        }
    }

    std::string answer() const {
        int best_len = 0;
        int best_state = 0;
        for (int i = 0; i < static_cast<int>(states_.size()); ++i) {
            if (states_[i].best > best_len) {
                best_len = states_[i].best;
                best_state = i;
            }
        }

        if (best_len <= 0) {
            return "";
        }

        int end_pos = states_[best_state].first_pos;
        int start = end_pos - best_len + 1;
        if (start < 0) {
            start = 0;
        }
        return base_.substr(static_cast<std::size_t>(start), static_cast<std::size_t>(best_len));
    }

private:
    struct State {
        int len = 0;
        int link = -1;
        int best = 0;
        int cur = 0;
        int first_pos = -1;
        std::array<int, 256> next{};

        State() {
            next.fill(-1);
        }
    };

    std::string base_;
    std::vector<State> states_;
    std::vector<int> order_;
    int last_ = 0;

    int make_state(int len) {
        states_.push_back(State{});
        states_.back().len = len;
        return static_cast<int>(states_.size()) - 1;
    }

    void extend(unsigned char c) {
        int cur = make_state(states_[last_].len + 1);
        states_[cur].first_pos = states_[cur].len - 1;

        int p = last_;
        while (p != -1 && states_[p].next[c] == -1) {
            states_[p].next[c] = cur;
            p = states_[p].link;
        }

        if (p == -1) {
            states_[cur].link = 0;
        } else {
            int q = states_[p].next[c];
            if (states_[p].len + 1 == states_[q].len) {
                states_[cur].link = q;
            } else {
                int clone = make_state(states_[p].len + 1);
                states_[clone].next = states_[q].next;
                states_[clone].link = states_[q].link;
                states_[clone].first_pos = states_[q].first_pos;

                while (p != -1 && states_[p].next[c] == q) {
                    states_[p].next[c] = clone;
                    p = states_[p].link;
                }

                states_[q].link = clone;
                states_[cur].link = clone;
            }
        }

        last_ = cur;
    }

    void build_order() {
        int max_len = 0;
        for (const auto& st : states_) {
            if (st.len > max_len) {
                max_len = st.len;
            }
        }

        std::vector<int> cnt(static_cast<std::size_t>(max_len) + 1, 0);
        for (const auto& st : states_) {
            ++cnt[static_cast<std::size_t>(st.len)];
        }

        for (int i = 1; i <= max_len; ++i) {
            cnt[static_cast<std::size_t>(i)] += cnt[static_cast<std::size_t>(i - 1)];
        }

        order_.assign(states_.size(), 0);
        for (int i = static_cast<int>(states_.size()) - 1; i >= 0; --i) {
            int len = states_[i].len;
            order_[--cnt[static_cast<std::size_t>(len)]] = i;
        }
    }
};

class SparseSAM {
public:
    explicit SparseSAM(const std::string& base) : base_(base) {
        states_.reserve(base.size() * 2 + 1);
        edges_.reserve(base.size() * 3 + 5);

        states_.push_back(State{});
        last_ = 0;

        for (unsigned char c : base) {
            extend(c);
        }

        build_order();
        for (auto& st : states_) {
            st.best = st.len;
        }
    }

    void process_string(const std::string& t) {
        int v = 0;
        int l = 0;

        for (unsigned char c : t) {
            while (v != 0 && get_next(v, c) == -1) {
                v = states_[v].link;
                l = states_[v].len;
            }

            int nx = get_next(v, c);
            if (nx != -1) {
                v = nx;
                ++l;
            } else {
                v = 0;
                l = 0;
            }

            if (l > states_[v].cur) {
                states_[v].cur = l;
            }
        }

        for (int i = static_cast<int>(order_.size()) - 1; i > 0; --i) {
            int s = order_[i];
            int p = states_[s].link;
            if (p >= 0) {
                int push_len = std::min(states_[s].cur, states_[p].len);
                if (push_len > states_[p].cur) {
                    states_[p].cur = push_len;
                }
            }
        }

        for (auto& st : states_) {
            st.best = std::min(st.best, st.cur);
            st.cur = 0;
        }
    }

    std::string answer() const {
        int best_len = 0;
        int best_state = 0;
        for (int i = 0; i < static_cast<int>(states_.size()); ++i) {
            if (states_[i].best > best_len) {
                best_len = states_[i].best;
                best_state = i;
            }
        }

        if (best_len <= 0) {
            return "";
        }

        int end_pos = states_[best_state].first_pos;
        int start = end_pos - best_len + 1;
        if (start < 0) {
            start = 0;
        }
        return base_.substr(static_cast<std::size_t>(start), static_cast<std::size_t>(best_len));
    }

private:
    struct Edge {
        int to = -1;
        int next = -1;
        unsigned char ch = 0;
    };

    struct State {
        int len = 0;
        int link = -1;
        int head = -1;
        int best = 0;
        int cur = 0;
        int first_pos = -1;
    };

    std::string base_;
    std::vector<State> states_;
    std::vector<Edge> edges_;
    std::vector<int> order_;
    int last_ = 0;

    int make_state(int len) {
        states_.push_back(State{});
        states_.back().len = len;
        return static_cast<int>(states_.size()) - 1;
    }

    int get_next(int s, unsigned char c) const {
        for (int e = states_[s].head; e != -1; e = edges_[e].next) {
            if (edges_[e].ch == c) {
                return edges_[e].to;
            }
        }
        return -1;
    }

    void set_next(int s, unsigned char c, int to) {
        for (int e = states_[s].head; e != -1; e = edges_[e].next) {
            if (edges_[e].ch == c) {
                edges_[e].to = to;
                return;
            }
        }
        Edge ne;
        ne.to = to;
        ne.ch = c;
        ne.next = states_[s].head;
        edges_.push_back(ne);
        states_[s].head = static_cast<int>(edges_.size()) - 1;
    }

    void clone_transitions(int from, int to) {
        std::vector<std::pair<unsigned char, int>> tmp;
        for (int e = states_[from].head; e != -1; e = edges_[e].next) {
            tmp.push_back({edges_[e].ch, edges_[e].to});
        }
        for (const auto& kv : tmp) {
            set_next(to, kv.first, kv.second);
        }
    }

    void extend(unsigned char c) {
        int cur = make_state(states_[last_].len + 1);
        states_[cur].first_pos = states_[cur].len - 1;

        int p = last_;
        while (p != -1 && get_next(p, c) == -1) {
            set_next(p, c, cur);
            p = states_[p].link;
        }

        if (p == -1) {
            states_[cur].link = 0;
        } else {
            int q = get_next(p, c);
            if (states_[p].len + 1 == states_[q].len) {
                states_[cur].link = q;
            } else {
                int clone = make_state(states_[p].len + 1);
                states_[clone].link = states_[q].link;
                states_[clone].first_pos = states_[q].first_pos;
                clone_transitions(q, clone);

                while (p != -1 && get_next(p, c) == q) {
                    set_next(p, c, clone);
                    p = states_[p].link;
                }

                states_[q].link = clone;
                states_[cur].link = clone;
            }
        }

        last_ = cur;
    }

    void build_order() {
        int max_len = 0;
        for (const auto& st : states_) {
            if (st.len > max_len) {
                max_len = st.len;
            }
        }

        std::vector<int> cnt(static_cast<std::size_t>(max_len) + 1, 0);
        for (const auto& st : states_) {
            ++cnt[static_cast<std::size_t>(st.len)];
        }

        for (int i = 1; i <= max_len; ++i) {
            cnt[static_cast<std::size_t>(i)] += cnt[static_cast<std::size_t>(i - 1)];
        }

        order_.assign(states_.size(), 0);
        for (int i = static_cast<int>(states_.size()) - 1; i >= 0; --i) {
            int len = states_[i].len;
            order_[--cnt[static_cast<std::size_t>(len)]] = i;
        }
    }
};

} // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    if (!(std::cin >> n)) {
        return 0;
    }

    std::string dummy;
    std::getline(std::cin, dummy);

    if (n <= 0) {
        std::cout << '\n';
        return 0;
    }

    std::vector<std::string> s(static_cast<std::size_t>(n));
    for (int i = 0; i < n; ++i) {
        std::getline(std::cin, s[static_cast<std::size_t>(i)]);
        trim_cr(s[static_cast<std::size_t>(i)]);
    }

    int shortest = 0;
    for (int i = 1; i < n; ++i) {
        if (s[static_cast<std::size_t>(i)].size() < s[static_cast<std::size_t>(shortest)].size()) {
            shortest = i;
        }
    }

    if (s[static_cast<std::size_t>(shortest)].empty()) {
        std::cout << '\n';
        return 0;
    }

    if (shortest != 0) {
        std::swap(s[0], s[static_cast<std::size_t>(shortest)]);
    }

    if (n == 1) {
        std::cout << s[0] << '\n';
        return 0;
    }

    const std::size_t len0 = s[0].size();
    const std::size_t len1 = s[1].size();

    // Strategy A: tiny 2-string case -> classic DP.
    if (n == 2 && len0 <= 10000 && len1 <= 10000 && len0 * len1 <= 25000000ULL) {
        std::cout << lcs_two_small_dp(s[0], s[1]) << '\n';
        return 0;
    }

    // Strategy B/C: scalable suffix automaton, dense for moderate base size, sparse for very large base.
    const std::size_t dense_threshold = 180000;

    std::string ans;
    if (s[0].size() <= dense_threshold) {
        DenseSAM sam(s[0]);
        for (int i = 1; i < n; ++i) {
            sam.process_string(s[static_cast<std::size_t>(i)]);
        }
        ans = sam.answer();
    } else {
        SparseSAM sam(s[0]);
        for (int i = 1; i < n; ++i) {
            sam.process_string(s[static_cast<std::size_t>(i)]);
        }
        ans = sam.answer();
    }

    std::cout << ans << '\n';
    return 0;
}
