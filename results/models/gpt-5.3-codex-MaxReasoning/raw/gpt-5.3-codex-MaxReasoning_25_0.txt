using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private const double G = 6.67430e-11;
    private const double DAY = 86400.0;
    private const double EARTH_RADIUS = 6371e3;
    private const double INTERCEPT_SUCCESS_METERS = 1_000_000.0; // 1000 km

    public static void Main()
    {
        string input = Console.In.ReadToEnd();
        var fs = new FastScanner(input);
        if (!fs.HasNext)
        {
            return;
        }

        int numBodies = fs.NextInt();
        double warningDays = Math.Max(0.01, fs.NextDouble());
        double impactorMass = Math.Max(1.0, fs.NextDouble());
        double deltaVBudget = Math.Max(0.0, fs.NextDouble());
        double asteroidMass = Math.Max(1.0, fs.NextDouble());
        double t0UnixUtc = fs.NextDouble();

        var bodies = new BodyInit[numBodies];
        for (int i = 0; i < numBodies; i++)
        {
            bodies[i] = new BodyInit
            {
                Name = fs.Next(),
                Mass = Math.Max(0.0, fs.NextDouble()),
                Pos = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble()),
                Vel = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble())
            };
        }

        string bspPath = fs.HasNext ? fs.Next() : string.Empty;
        _ = bspPath;
        _ = t0UnixUtc;

        double durationSec = warningDays * DAY;
        double bodyStepSec = ChooseBodyStepDays(warningDays) * DAY;
        Ephemeris eph = Ephemeris.Build(bodies, durationSec, bodyStepSec);

        int earthIdx = eph.FindBodyIndex("Earth");
        int asteroidIdx = eph.FindBodyIndex("Asteroid");

        MissionPlan plan;
        if (earthIdx < 0 || asteroidIdx < 0 || deltaVBudget <= 0.0)
        {
            plan = BuildFallbackPlan(warningDays);
        }
        else
        {
            plan = PlanMission(
                eph,
                earthIdx,
                asteroidIdx,
                warningDays,
                impactorMass,
                asteroidMass,
                deltaVBudget);
        }

        EmitPlan(plan);
    }

    private static double ChooseBodyStepDays(double warningDays)
    {
        if (warningDays <= 30.0) return 1.0 / 12.0;   // 2h
        if (warningDays <= 365.0) return 0.25;        // 6h
        if (warningDays <= 4000.0) return 1.0;        // 1d
        return 2.0;                                   // 2d
    }

    private static MissionPlan BuildFallbackPlan(double warningDays)
    {
        double impact = Math.Max(0.1, Math.Min(warningDays, warningDays * 0.8));
        double correction = Math.Max(0.05, impact * 0.5);
        return new MissionPlan
        {
            LaunchDay = 0.0,
            CorrectionDay = correction,
            ImpactDay = impact,
            LaunchDv = Vec3.Zero,
            CorrectionDv = Vec3.Zero,
            ImpactDv = Vec3.Zero,
            MissDistance = double.MaxValue,
            TotalDv = 0.0,
            RelativeSpeed = 0.0,
            Score = double.NegativeInfinity
        };
    }

    private static MissionPlan PlanMission(
        Ephemeris eph,
        int earthIdx,
        int asteroidIdx,
        double warningDays,
        double impactorMass,
        double asteroidMass,
        double deltaVBudget)
    {
        double impactDay = EstimateImpactDay(eph, earthIdx, asteroidIdx, warningDays);
        double leadGuard = Math.Max(0.15, Math.Min(5.0, 0.02 * impactDay));
        double latestIntercept = Math.Min(warningDays, impactDay - leadGuard);
        if (latestIntercept <= 0.2)
        {
            latestIntercept = Math.Min(warningDays, Math.Max(0.25, impactDay * 0.9));
        }

        double minTof = Math.Max(0.2, Math.Min(4.0, warningDays * 0.004));
        double launchUpper = Math.Max(0.0, latestIntercept - minTof);

        int launchSamples = warningDays < 40.0 ? 6 : warningDays < 365.0 ? 10 : 14;
        int tofSamples = warningDays < 40.0 ? 10 : warningDays < 365.0 ? 14 : 18;

        MissionPlan? best = null;

        for (int i = 0; i < launchSamples; i++)
        {
            double u = launchSamples == 1 ? 0.0 : (double)i / (launchSamples - 1);
            // Bias early launches (useful for deflection leverage).
            double launchDay = launchUpper * u * u;

            double interceptMin = launchDay + minTof;
            double interceptMax = latestIntercept;
            if (interceptMax <= interceptMin)
            {
                continue;
            }

            for (int j = 0; j < tofSamples; j++)
            {
                double v = tofSamples == 1 ? 0.0 : (double)j / (tofSamples - 1);
                double interceptDay = interceptMin + (interceptMax - interceptMin) * v;

                if (!TrySolveDirectIntercept(
                    eph,
                    earthIdx,
                    asteroidIdx,
                    launchDay,
                    interceptDay,
                    deltaVBudget,
                    out Vec3 launchDv,
                    out double missDirect,
                    out double relSpeedDirect,
                    out State directEndState))
                {
                    continue;
                }
                _ = directEndState;

                double correctionDay = launchDay + 0.65 * (interceptDay - launchDay);
                Vec3 correctionDv = Vec3.Zero;
                double missFinal = missDirect;
                double relSpeedFinal = relSpeedDirect;

                if (correctionDay > launchDay + 0.02)
                {
                    if (TrySolveCorrectionBurn(
                        eph,
                        earthIdx,
                        asteroidIdx,
                        launchDay,
                        correctionDay,
                        interceptDay,
                        launchDv,
                        out Vec3 dv2,
                        out double missCorrected,
                        out double relSpeedCorrected))
                    {
                        // Keep correction only if it improves terminal miss significantly.
                        if (missCorrected < missFinal * 0.92)
                        {
                            correctionDv = dv2;
                            missFinal = missCorrected;
                            relSpeedFinal = relSpeedCorrected;
                        }
                    }
                }

                double totalDv = launchDv.Norm() + correctionDv.Norm();
                double score = ScorePlan(
                    missFinal,
                    relSpeedFinal,
                    totalDv,
                    deltaVBudget,
                    impactorMass,
                    asteroidMass,
                    impactDay,
                    interceptDay);

                var candidate = new MissionPlan
                {
                    LaunchDay = launchDay,
                    CorrectionDay = correctionDay,
                    ImpactDay = interceptDay,
                    LaunchDv = launchDv,
                    CorrectionDv = correctionDv,
                    ImpactDv = Vec3.Zero,
                    MissDistance = missFinal,
                    TotalDv = totalDv,
                    RelativeSpeed = relSpeedFinal,
                    Score = score
                };

                if (best == null || candidate.Score > best.Value.Score)
                {
                    best = candidate;
                }
            }
        }

        if (best == null)
        {
            return BuildFallbackPlan(warningDays);
        }

        MissionPlan chosen = best.Value;

        // Final pass: refine correction with more propagation steps.
        if (chosen.CorrectionDv.Norm() > 0.0)
        {
            if (TrySolveCorrectionBurn(
                eph,
                earthIdx,
                asteroidIdx,
                chosen.LaunchDay,
                chosen.CorrectionDay,
                chosen.ImpactDay,
                chosen.LaunchDv,
                out Vec3 refinedDv2,
                out double refinedMiss,
                out double refinedRelSpeed,
                highFidelity: true))
            {
                double refinedTotalDv = chosen.LaunchDv.Norm() + refinedDv2.Norm();
                if (refinedMiss <= chosen.MissDistance * 1.05)
                {
                    chosen.CorrectionDv = refinedDv2;
                    chosen.MissDistance = refinedMiss;
                    chosen.RelativeSpeed = refinedRelSpeed;
                    chosen.TotalDv = refinedTotalDv;
                }
            }
        }

        // Enforce budget hard by scaling burns if needed (direction preserved).
        if (chosen.TotalDv > deltaVBudget && chosen.TotalDv > 0.0)
        {
            double scale = deltaVBudget / chosen.TotalDv;
            chosen.LaunchDv *= scale;
            chosen.CorrectionDv *= scale;
            chosen.TotalDv = deltaVBudget;

            // Re-evaluate miss after scaling to keep consistent output.
            EvaluateTrajectory(
                eph,
                earthIdx,
                asteroidIdx,
                chosen.LaunchDay,
                chosen.CorrectionDay,
                chosen.ImpactDay,
                chosen.LaunchDv,
                chosen.CorrectionDv,
                highFidelity: true,
                out double missScaled,
                out double relSpeedScaled);

            chosen.MissDistance = missScaled;
            chosen.RelativeSpeed = relSpeedScaled;
        }

        // Keep times in legal launch window and monotonic order.
        chosen.LaunchDay = Clamp(chosen.LaunchDay, 0.0, warningDays);
        chosen.CorrectionDay = Clamp(chosen.CorrectionDay, chosen.LaunchDay, chosen.ImpactDay);
        chosen.ImpactDay = Clamp(chosen.ImpactDay, chosen.CorrectionDay, warningDays);

        return chosen;
    }

    private static double EstimateImpactDay(Ephemeris eph, int earthIdx, int asteroidIdx, double warningDays)
    {
        double bestT = warningDays;
        double bestD2 = double.MaxValue;

        int samples = Math.Max(200, Math.Min(4000, eph.Steps));
        for (int i = 0; i < samples; i++)
        {
            double tDay = warningDays * i / Math.Max(1, samples - 1);
            double tSec = tDay * DAY;
            Vec3 re = eph.GetPosition(earthIdx, tSec);
            Vec3 ra = eph.GetPosition(asteroidIdx, tSec);
            double d2 = (ra - re).NormSq();
            if (d2 < bestD2)
            {
                bestD2 = d2;
                bestT = tDay;
            }
        }

        // Small local refinement around the best coarse sample.
        double span = Math.Min(5.0, Math.Max(0.2, warningDays * 0.01));
        double t0 = Math.Max(0.0, bestT - span);
        double t1 = Math.Min(warningDays, bestT + span);
        int fine = 60;
        for (int i = 0; i <= fine; i++)
        {
            double tDay = t0 + (t1 - t0) * i / fine;
            double tSec = tDay * DAY;
            Vec3 re = eph.GetPosition(earthIdx, tSec);
            Vec3 ra = eph.GetPosition(asteroidIdx, tSec);
            double d2 = (ra - re).NormSq();
            if (d2 < bestD2)
            {
                bestD2 = d2;
                bestT = tDay;
            }
        }

        // If the closest approach is still very large, use late horizon as pseudo-impact.
        double bestD = Math.Sqrt(bestD2);
        if (bestD > 15.0 * EARTH_RADIUS)
        {
            bestT = Math.Min(warningDays, Math.Max(bestT, 0.85 * warningDays));
        }

        return bestT;
    }

    private static double ScorePlan(
        double miss,
        double relSpeed,
        double totalDv,
        double budget,
        double impactorMass,
        double asteroidMass,
        double earthImpactDay,
        double interceptDay)
    {
        double leadSec = Math.Max(0.0, (earthImpactDay - interceptDay) * DAY);
        double asteroidKick = (impactorMass / asteroidMass) * relSpeed;
        double deflectionProxy = asteroidKick * leadSec;

        double score = deflectionProxy;

        // Strongly reward successful terminal intercept.
        if (miss <= INTERCEPT_SUCCESS_METERS)
        {
            score += 1e12;
        }
        else
        {
            score -= miss * 50.0;
        }

        // Prefer higher impact speed for kinetic momentum transfer.
        score += relSpeed * 1e5;

        // Keep within dv budget.
        if (totalDv <= budget)
        {
            score += 5e10;
        }
        else
        {
            score -= (totalDv - budget) * 1e8;
        }

        // Mild preference for lower consumption when tied.
        score -= totalDv * 2e5;

        return score;
    }

    private static bool TrySolveDirectIntercept(
        Ephemeris eph,
        int earthIdx,
        int asteroidIdx,
        double launchDay,
        double interceptDay,
        double budget,
        out Vec3 launchDv,
        out double miss,
        out double relSpeed,
        out State finalState)
    {
        launchDv = Vec3.Zero;
        miss = double.MaxValue;
        relSpeed = 0.0;
        finalState = default;

        double tLaunch = launchDay * DAY;
        double tIntercept = interceptDay * DAY;
        if (tIntercept <= tLaunch + 10.0)
        {
            return false;
        }

        State earthState = eph.GetState(earthIdx, tLaunch);
        State asteroidImpact = eph.GetState(asteroidIdx, tIntercept);

        double dt = tIntercept - tLaunch;
        Vec3 guessV = (asteroidImpact.Pos - earthState.Pos) / dt;
        Vec3 dv = guessV - earthState.Vel;

        double cap = Math.Max(200.0, budget * 1.2);
        double dvMag = dv.Norm();
        if (dvMag > cap)
        {
            dv *= cap / dvMag;
        }

        int steps = SelectPropSteps(dt, highFidelity: false);

        for (int iter = 0; iter < 8; iter++)
        {
            State scEnd = PropagateSpacecraft(eph, tLaunch, tIntercept, earthState.Pos, earthState.Vel + dv, steps);
            Vec3 err = scEnd.Pos - asteroidImpact.Pos;
            double curMiss = err.Norm();

            if (curMiss < miss)
            {
                miss = curMiss;
                finalState = scEnd;
                launchDv = dv;
            }

            if (curMiss < 1000.0)
            {
                break;
            }

            double eps = Math.Max(0.2, Math.Min(8.0, 0.005 * Math.Max(1.0, dv.Norm())));
            double[,] J = new double[3, 3];

            for (int axis = 0; axis < 3; axis++)
            {
                Vec3 delta = axis == 0
                    ? new Vec3(eps, 0.0, 0.0)
                    : axis == 1
                        ? new Vec3(0.0, eps, 0.0)
                        : new Vec3(0.0, 0.0, eps);

                State pertEnd = PropagateSpacecraft(eph, tLaunch, tIntercept, earthState.Pos, earthState.Vel + dv + delta, steps);
                Vec3 derr = (pertEnd.Pos - scEnd.Pos) / eps;

                J[0, axis] = derr.X;
                J[1, axis] = derr.Y;
                J[2, axis] = derr.Z;
            }

            Vec3 step = DampedLeastSquaresStep(J, err * -1.0);
            if (step.Norm() < 1e-4)
            {
                break;
            }

            // Keep Newton jumps controlled.
            double maxStep = Math.Max(50.0, budget * 0.25 + 100.0);
            double stepMag = step.Norm();
            if (stepMag > maxStep)
            {
                step *= maxStep / stepMag;
            }

            // Backtracking line-search.
            bool accepted = false;
            Vec3 baseDv = dv;
            double baseMiss = curMiss;
            for (int ls = 0; ls < 5; ls++)
            {
                double alpha = 1.0 / (1 << ls);
                Vec3 trialDv = baseDv + step * alpha;
                State trialEnd = PropagateSpacecraft(eph, tLaunch, tIntercept, earthState.Pos, earthState.Vel + trialDv, steps);
                double trialMiss = (trialEnd.Pos - asteroidImpact.Pos).Norm();
                if (trialMiss < baseMiss)
                {
                    dv = trialDv;
                    accepted = true;
                    break;
                }
            }

            if (!accepted)
            {
                break;
            }
        }

        // Final evaluation with slightly higher fidelity.
        State final = PropagateSpacecraft(
            eph,
            tLaunch,
            tIntercept,
            earthState.Pos,
            earthState.Vel + launchDv,
            SelectPropSteps(dt, highFidelity: true));

        State asteroidFinal = eph.GetState(asteroidIdx, tIntercept);
        miss = (final.Pos - asteroidFinal.Pos).Norm();
        relSpeed = (final.Vel - asteroidFinal.Vel).Norm();
        finalState = final;

        return true;
    }

    private static bool TrySolveCorrectionBurn(
        Ephemeris eph,
        int earthIdx,
        int asteroidIdx,
        double launchDay,
        double correctionDay,
        double interceptDay,
        Vec3 launchDv,
        out Vec3 correctionDv,
        out double miss,
        out double relSpeed,
        bool highFidelity = false)
    {
        correctionDv = Vec3.Zero;
        miss = double.MaxValue;
        relSpeed = 0.0;

        double tLaunch = launchDay * DAY;
        double tCorr = correctionDay * DAY;
        double tImpact = interceptDay * DAY;

        if (tCorr <= tLaunch + 10.0 || tImpact <= tCorr + 10.0)
        {
            return false;
        }

        State earthLaunch = eph.GetState(earthIdx, tLaunch);
        State atCorr = PropagateSpacecraft(
            eph,
            tLaunch,
            tCorr,
            earthLaunch.Pos,
            earthLaunch.Vel + launchDv,
            SelectPropSteps(tCorr - tLaunch, highFidelity));

        Vec3 dv2 = Vec3.Zero;
        int steps = SelectPropSteps(tImpact - tCorr, highFidelity);

        for (int iter = 0; iter < 6; iter++)
        {
            State scEnd = PropagateSpacecraft(eph, tCorr, tImpact, atCorr.Pos, atCorr.Vel + dv2, steps);
            State astEnd = eph.GetState(asteroidIdx, tImpact);
            Vec3 err = scEnd.Pos - astEnd.Pos;
            double curMiss = err.Norm();

            if (curMiss < miss)
            {
                miss = curMiss;
                relSpeed = (scEnd.Vel - astEnd.Vel).Norm();
                correctionDv = dv2;
            }

            if (curMiss < 300.0)
            {
                break;
            }

            double eps = Math.Max(0.05, Math.Min(5.0, 0.005 * Math.Max(1.0, dv2.Norm() + 1.0)));
            double[,] J = new double[3, 3];

            for (int axis = 0; axis < 3; axis++)
            {
                Vec3 delta = axis == 0
                    ? new Vec3(eps, 0.0, 0.0)
                    : axis == 1
                        ? new Vec3(0.0, eps, 0.0)
                        : new Vec3(0.0, 0.0, eps);

                State pert = PropagateSpacecraft(eph, tCorr, tImpact, atCorr.Pos, atCorr.Vel + dv2 + delta, steps);
                Vec3 derr = (pert.Pos - scEnd.Pos) / eps;

                J[0, axis] = derr.X;
                J[1, axis] = derr.Y;
                J[2, axis] = derr.Z;
            }

            Vec3 step = DampedLeastSquaresStep(J, err * -1.0);
            double stepMag = step.Norm();
            if (stepMag < 1e-4)
            {
                break;
            }

            if (stepMag > 800.0)
            {
                step *= 800.0 / stepMag;
            }

            bool accepted = false;
            double baseMiss = curMiss;
            Vec3 baseDv = dv2;

            for (int ls = 0; ls < 5; ls++)
            {
                double alpha = 1.0 / (1 << ls);
                Vec3 trialDv = baseDv + step * alpha;
                State trial = PropagateSpacecraft(eph, tCorr, tImpact, atCorr.Pos, atCorr.Vel + trialDv, steps);
                double trialMiss = (trial.Pos - astEnd.Pos).Norm();
                if (trialMiss < baseMiss)
                {
                    dv2 = trialDv;
                    accepted = true;
                    break;
                }
            }

            if (!accepted)
            {
                break;
            }
        }

        return miss < double.MaxValue / 2.0;
    }

    private static void EvaluateTrajectory(
        Ephemeris eph,
        int earthIdx,
        int asteroidIdx,
        double launchDay,
        double correctionDay,
        double impactDay,
        Vec3 launchDv,
        Vec3 correctionDv,
        bool highFidelity,
        out double miss,
        out double relSpeed)
    {
        double tLaunch = launchDay * DAY;
        double tCorr = correctionDay * DAY;
        double tImpact = impactDay * DAY;

        State earthLaunch = eph.GetState(earthIdx, tLaunch);

        State atCorr = PropagateSpacecraft(
            eph,
            tLaunch,
            tCorr,
            earthLaunch.Pos,
            earthLaunch.Vel + launchDv,
            SelectPropSteps(tCorr - tLaunch, highFidelity));

        State atImpact = PropagateSpacecraft(
            eph,
            tCorr,
            tImpact,
            atCorr.Pos,
            atCorr.Vel + correctionDv,
            SelectPropSteps(tImpact - tCorr, highFidelity));

        State asteroidImpact = eph.GetState(asteroidIdx, tImpact);
        miss = (atImpact.Pos - asteroidImpact.Pos).Norm();
        relSpeed = (atImpact.Vel - asteroidImpact.Vel).Norm();
    }

    private static int SelectPropSteps(double durationSec, bool highFidelity)
    {
        double days = durationSec / DAY;
        int baseSteps;

        if (days <= 2.0) baseSteps = 500;
        else if (days <= 20.0) baseSteps = 700;
        else if (days <= 180.0) baseSteps = 900;
        else if (days <= 2000.0) baseSteps = 1100;
        else baseSteps = 1300;

        if (highFidelity)
        {
            baseSteps = (int)(baseSteps * 1.6);
        }

        return Math.Max(200, Math.Min(2600, baseSteps));
    }

    private static State PropagateSpacecraft(
        Ephemeris eph,
        double tStart,
        double tEnd,
        Vec3 r0,
        Vec3 v0,
        int steps)
    {
        if (tEnd <= tStart)
        {
            return new State(r0, v0);
        }

        steps = Math.Max(1, steps);
        double total = tEnd - tStart;
        double dt = total / steps;

        Vec3 r = r0;
        Vec3 v = v0;
        double t = tStart;

        Vec3 a = SpacecraftAcceleration(eph, r, t);

        for (int i = 0; i < steps; i++)
        {
            double h = (i == steps - 1) ? (tEnd - t) : dt;
            if (h <= 0.0)
            {
                break;
            }

            Vec3 vHalf = v + a * (0.5 * h);
            Vec3 rNew = r + vHalf * h;
            double tNew = t + h;

            Vec3 aNew = SpacecraftAcceleration(eph, rNew, tNew);
            Vec3 vNew = vHalf + aNew * (0.5 * h);

            r = rNew;
            v = vNew;
            a = aNew;
            t = tNew;
        }

        return new State(r, v);
    }

    private static Vec3 SpacecraftAcceleration(Ephemeris eph, Vec3 spacecraftPos, double tSec)
    {
        Vec3 a = Vec3.Zero;
        for (int i = 0; i < eph.BodyCount; i++)
        {
            Vec3 rb = eph.GetPosition(i, tSec);
            Vec3 dr = rb - spacecraftPos;
            double r2 = dr.NormSq();

            // Softening for robustness very near any body center.
            r2 = Math.Max(r2, 1e6);
            double invR = 1.0 / Math.Sqrt(r2);
            double invR3 = invR * invR * invR;
            a += dr * (G * eph.Masses[i] * invR3);
        }

        return a;
    }

    private static Vec3 DampedLeastSquaresStep(double[,] J, Vec3 rhs)
    {
        // Solve (J^T J + lambda I) * x = J^T * rhs
        double lambda = 1e-6;
        for (int k = 0; k < 6; k++)
        {
            double[,] A = new double[3, 3];
            double[] b = new double[3];

            for (int r = 0; r < 3; r++)
            {
                for (int c = 0; c < 3; c++)
                {
                    double sum = 0.0;
                    for (int m = 0; m < 3; m++)
                    {
                        sum += J[m, r] * J[m, c];
                    }
                    A[r, c] = sum;
                }
            }

            A[0, 0] += lambda;
            A[1, 1] += lambda;
            A[2, 2] += lambda;

            b[0] = J[0, 0] * rhs.X + J[1, 0] * rhs.Y + J[2, 0] * rhs.Z;
            b[1] = J[0, 1] * rhs.X + J[1, 1] * rhs.Y + J[2, 1] * rhs.Z;
            b[2] = J[0, 2] * rhs.X + J[1, 2] * rhs.Y + J[2, 2] * rhs.Z;

            if (SolveLinear3x3(A, b, out Vec3 x))
            {
                return x;
            }

            lambda *= 100.0;
        }

        return Vec3.Zero;
    }

    private static bool SolveLinear3x3(double[,] A, double[] b, out Vec3 x)
    {
        // Gaussian elimination with partial pivoting on augmented matrix [A|b].
        double[,] m = new double[3, 4]
        {
            { A[0, 0], A[0, 1], A[0, 2], b[0] },
            { A[1, 0], A[1, 1], A[1, 2], b[1] },
            { A[2, 0], A[2, 1], A[2, 2], b[2] }
        };

        for (int col = 0; col < 3; col++)
        {
            int pivot = col;
            double best = Math.Abs(m[col, col]);
            for (int r = col + 1; r < 3; r++)
            {
                double val = Math.Abs(m[r, col]);
                if (val > best)
                {
                    best = val;
                    pivot = r;
                }
            }

            if (best < 1e-18)
            {
                x = Vec3.Zero;
                return false;
            }

            if (pivot != col)
            {
                for (int c = col; c < 4; c++)
                {
                    (m[col, c], m[pivot, c]) = (m[pivot, c], m[col, c]);
                }
            }

            double diag = m[col, col];
            for (int c = col; c < 4; c++)
            {
                m[col, c] /= diag;
            }

            for (int r = 0; r < 3; r++)
            {
                if (r == col) continue;
                double factor = m[r, col];
                if (Math.Abs(factor) < 1e-20) continue;
                for (int c = col; c < 4; c++)
                {
                    m[r, c] -= factor * m[col, c];
                }
            }
        }

        x = new Vec3(m[0, 3], m[1, 3], m[2, 3]);
        return true;
    }

    private static double Clamp(double value, double min, double max)
    {
        if (value < min) return min;
        if (value > max) return max;
        return value;
    }

    private readonly struct State
    {

        public readonly Vec3 Vel;

        public State(Vec3 pos, Vec3 vel)

            Vel = vel;

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public static readonly Vec3 Zero = new Vec3(0.0, 0.0, 0.0);

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double NormSq() => X * X + Y * Y + Z * Z;
        public double Norm() => Math.Sqrt(NormSq());

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
    }

    private sealed class Ephemeris
    {
        private readonly Vec3[] _positions; // flattened [step * bodyCount + body]
        private readonly Vec3[] _velocities; // flattened [step * bodyCount + body]
        private readonly Dictionary<string, int> _nameToIndex;

        public int BodyCount { get; }
        public int Steps { get; }
        public double DtSec { get; }
        public double DurationSec { get; }
        public double[] Masses { get; }

        private Ephemeris(
            int bodyCount,
            int steps,
            double dtSec,
            double durationSec,
            double[] masses,
            Vec3[] positions,
            Vec3[] velocities,
            Dictionary<string, int> nameToIndex)
        {
            BodyCount = bodyCount;
            Steps = steps;
            DtSec = dtSec;
            DurationSec = durationSec;
            Masses = masses;
            _positions = positions;
            _velocities = velocities;
            _nameToIndex = nameToIndex;
        }

        public static Ephemeris Build(BodyInit[] bodies, double durationSec, double dtSec)
        {
            int n = bodies.Length;
            int steps = Math.Max(2, (int)Math.Ceiling(durationSec / dtSec) + 1);

            var masses = new double[n];
            var names = new Dictionary<string, int>(StringComparer.OrdinalIgnoreCase);
            var pos = new Vec3[steps * n];
            var vel = new Vec3[steps * n];

            for (int i = 0; i < n; i++)
            {
                masses[i] = bodies[i].Mass;
                names[bodies[i].Name] = i;
                pos[i] = bodies[i].Pos;
                vel[i] = bodies[i].Vel;
            }

            Vec3[] r = new Vec3[n];
            Vec3[] v = new Vec3[n];
            Vec3[] a = new Vec3[n];
            Vec3[] vHalf = new Vec3[n];
            Vec3[] rNew = new Vec3[n];
            Vec3[] aNew = new Vec3[n];

            for (int i = 0; i < n; i++)
            {
                r[i] = bodies[i].Pos;
                v[i] = bodies[i].Vel;
            }

            ComputeAccelerations(r, masses, a);

            for (int s = 0; s < steps - 1; s++)
            {
                double tHere = s * dtSec;
                double h = Math.Min(dtSec, durationSec - tHere);
                if (h <= 0.0)
                {
                    for (int i = 0; i < n; i++)
                    {
                        pos[(s + 1) * n + i] = r[i];
                        vel[(s + 1) * n + i] = v[i];
                    }
                    continue;
                }

                for (int i = 0; i < n; i++)
                {
                    vHalf[i] = v[i] + a[i] * (0.5 * h);
                    rNew[i] = r[i] + vHalf[i] * h;
                }

                ComputeAccelerations(rNew, masses, aNew);

                for (int i = 0; i < n; i++)
                {
                    Vec3 vNext = vHalf[i] + aNew[i] * (0.5 * h);
                    r[i] = rNew[i];
                    v[i] = vNext;
                    a[i] = aNew[i];

                    pos[(s + 1) * n + i] = r[i];
                    vel[(s + 1) * n + i] = v[i];
                }
            }

            return new Ephemeris(n, steps, dtSec, durationSec, masses, pos, vel, names);
        }

        public int FindBodyIndex(string name)
        {
            return _nameToIndex.TryGetValue(name, out int idx) ? idx : -1;
        }

        public Vec3 GetPosition(int bodyIndex, double tSec)
        {
            if (bodyIndex < 0 || bodyIndex >= BodyCount)
            {
                return Vec3.Zero;
            }

            double tc = Clamp(tSec, 0.0, DurationSec);
            double f = tc / DtSec;
            int i0 = (int)f;
            if (i0 >= Steps - 1)
            {
                return _positions[(Steps - 1) * BodyCount + bodyIndex];
            }

            double u = f - i0;
            Vec3 p0 = _positions[i0 * BodyCount + bodyIndex];
            Vec3 p1 = _positions[(i0 + 1) * BodyCount + bodyIndex];
            return p0 * (1.0 - u) + p1 * u;
        }

        public Vec3 GetVelocity(int bodyIndex, double tSec)
        {
            if (bodyIndex < 0 || bodyIndex >= BodyCount)
            {
                return Vec3.Zero;
            }

            double tc = Clamp(tSec, 0.0, DurationSec);
            double f = tc / DtSec;
            int i0 = (int)f;
            if (i0 >= Steps - 1)
            {
                return _velocities[(Steps - 1) * BodyCount + bodyIndex];
            }

            double u = f - i0;
            Vec3 v0 = _velocities[i0 * BodyCount + bodyIndex];
            Vec3 v1 = _velocities[(i0 + 1) * BodyCount + bodyIndex];
            return v0 * (1.0 - u) + v1 * u;
        }

        public State GetState(int bodyIndex, double tSec)
        {
            return new State(GetPosition(bodyIndex, tSec), GetVelocity(bodyIndex, tSec));
        }

        private static void ComputeAccelerations(Vec3[] r, double[] masses, Vec3[] a)
        {
            int n = r.Length;
            for (int i = 0; i < n; i++) a[i] = Vec3.Zero;

            for (int i = 0; i < n; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    Vec3 dr = r[j] - r[i];
                    double r2 = dr.NormSq();
                    r2 = Math.Max(r2, 1e6);
                    double invR = 1.0 / Math.Sqrt(r2);
                    double invR3 = invR * invR * invR;
                    Vec3 acc = dr * (G * invR3);
                    a[i] += acc * masses[j];
                    a[j] -= acc * masses[i];
                }
            }
        }
    }

    private sealed class FastScanner
    {
        private readonly string[] _tokens;
        private int _index;

        public FastScanner(string input)
        {
            _tokens = input.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
            _index = 0;
        }

        public bool HasNext => _index < _tokens.Length;

        public string Next()
        {
            if (_index >= _tokens.Length)
            {
                throw new InvalidOperationException("Unexpected end of input.");
            }
            return _tokens[_index++];
        }

        public int NextInt() => int.Parse(Next(), NumberStyles.Integer, CultureInfo.InvariantCulture);

        public double NextDouble() => double.Parse(Next(), NumberStyles.Float | NumberStyles.AllowThousands, CultureInfo.InvariantCulture);
    }
