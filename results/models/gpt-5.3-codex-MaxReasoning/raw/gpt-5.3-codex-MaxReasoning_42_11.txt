OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_third = OpConstant %float 0.33333334
%float_two_third = OpConstant %float 0.66666669
%float_thickness = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_gl_PerVertex_in_4 = OpTypePointer Input %arr_gl_PerVertex_in_4
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex_in_4 Input

%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3float_4 = OpTypePointer Input %arr_v3float_4
%inWorldPos = OpVariable %ptr_in_arr_v3float_4 Input
%inNormal = OpVariable %ptr_in_arr_v3float_4 Input
%inColor = OpVariable %ptr_in_arr_v3float_4 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex_out Output
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %fn_void
%entry = OpLabel

%ptr_prev = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p_prev = OpLoad %v4float %ptr_prev
%ptr_p0 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_next = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
%p_next = OpLoad %v4float %ptr_next

%ptr_col0 = OpAccessChain %ptr_in_v3float %inColor %int_1
%col0 = OpLoad %v3float %ptr_col0
%ptr_col1 = OpAccessChain %ptr_in_v3float %inColor %int_2
%col1 = OpLoad %v3float %ptr_col1

%prev_xy = OpVectorShuffle %v2float %p_prev %p_prev 0 1
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%next_xy = OpVectorShuffle %v2float %p_next %p_next 0 1

%prev_w = OpCompositeExtract %float %p_prev 3
%p0_w = OpCompositeExtract %float %p0 3
%p1_w = OpCompositeExtract %float %p1 3
%next_w = OpCompositeExtract %float %p_next 3

%prev_w2 = OpCompositeConstruct %v2float %prev_w %prev_w
%p0_w2 = OpCompositeConstruct %v2float %p0_w %p0_w
%p1_w2 = OpCompositeConstruct %v2float %p1_w %p1_w
%next_w2 = OpCompositeConstruct %v2float %next_w %next_w

%prev_ndc = OpFDiv %v2float %prev_xy %prev_w2
%p0_ndc = OpFDiv %v2float %p0_xy %p0_w2
%p1_ndc = OpFDiv %v2float %p1_xy %p1_w2
%next_ndc = OpFDiv %v2float %next_xy %next_w2

%dir_prev = OpFSub %v2float %p0_ndc %prev_ndc
%dir_mid = OpFSub %v2float %p1_ndc %p0_ndc
%dir_next = OpFSub %v2float %next_ndc %p1_ndc

%ndir_prev = OpExtInst %v2float %glsl450 Normalize %dir_prev
%ndir_mid = OpExtInst %v2float %glsl450 Normalize %dir_mid
%ndir_next = OpExtInst %v2float %glsl450 Normalize %dir_next

%tan_start_sum = OpFAdd %v2float %ndir_prev %ndir_mid
%tan_end_sum = OpFAdd %v2float %ndir_mid %ndir_next
%tan_start = OpExtInst %v2float %glsl450 Normalize %tan_start_sum
%tan_end = OpExtInst %v2float %glsl450 Normalize %tan_end_sum

%pos_delta = OpFSub %v4float %p1 %p0
%col_delta = OpFSub %v3float %col1 %col0
%tan_delta = OpFSub %v2float %tan_end %tan_start

%seg1_pos_off = OpVectorTimesScalar %v4float %pos_delta %float_third
%seg1_pos = OpFAdd %v4float %p0 %seg1_pos_off
%seg1_col_off = OpVectorTimesScalar %v3float %col_delta %float_third
%seg1_col = OpFAdd %v3float %col0 %seg1_col_off
%seg1_tan_off = OpVectorTimesScalar %v2float %tan_delta %float_third
%seg1_tan_pre = OpFAdd %v2float %tan_start %seg1_tan_off
%seg1_tan = OpExtInst %v2float %glsl450 Normalize %seg1_tan_pre

%seg2_pos_off = OpVectorTimesScalar %v4float %pos_delta %float_two_third
%seg2_pos = OpFAdd %v4float %p0 %seg2_pos_off
%seg2_col_off = OpVectorTimesScalar %v3float %col_delta %float_two_third
%seg2_col = OpFAdd %v3float %col0 %seg2_col_off
%seg2_tan_off = OpVectorTimesScalar %v2float %tan_delta %float_two_third
%seg2_tan_pre = OpFAdd %v2float %tan_start %seg2_tan_off
%seg2_tan = OpExtInst %v2float %glsl450 Normalize %seg2_tan_pre

%tan0_x = OpCompositeExtract %float %tan_start 0
%tan0_y = OpCompositeExtract %float %tan_start 1
%neg_tan0_y = OpFNegate %float %tan0_y
%perp0 = OpCompositeConstruct %v2float %neg_tan0_y %tan0_x
%off0_ndc = OpVectorTimesScalar %v2float %perp0 %float_thickness
%p0_out_w = OpCompositeExtract %float %p0 3
%off0_clip = OpVectorTimesScalar %v2float %off0_ndc %p0_out_w
%off0_x = OpCompositeExtract %float %off0_clip 0
%off0_y = OpCompositeExtract %float %off0_clip 1
%off0_4 = OpCompositeConstruct %v4float %off0_x %off0_y %float_0 %float_0
%p0_plus = OpFAdd %v4float %p0 %off0_4
%p0_minus = OpFSub %v4float %p0 %off0_4

%tan1_x = OpCompositeExtract %float %seg1_tan 0
%tan1_y = OpCompositeExtract %float %seg1_tan 1
%neg_tan1_y = OpFNegate %float %tan1_y
%perp1 = OpCompositeConstruct %v2float %neg_tan1_y %tan1_x
%off1_ndc = OpVectorTimesScalar %v2float %perp1 %float_thickness
%seg1_w = OpCompositeExtract %float %seg1_pos 3
%off1_clip = OpVectorTimesScalar %v2float %off1_ndc %seg1_w
%off1_x = OpCompositeExtract %float %off1_clip 0
%off1_y = OpCompositeExtract %float %off1_clip 1
%off1_4 = OpCompositeConstruct %v4float %off1_x %off1_y %float_0 %float_0
%seg1_plus = OpFAdd %v4float %seg1_pos %off1_4
%seg1_minus = OpFSub %v4float %seg1_pos %off1_4

%ptr_out_pos = OpAccessChain %ptr_out_v4float %gl_out %int_0

OpStore %ptr_out_pos %p0_plus
OpStore %outColor %col0
OpEmitVertex

OpStore %ptr_out_pos %p0_minus
OpStore %outColor %col0
OpEmitVertex

OpStore %ptr_out_pos %seg1_plus
OpStore %outColor %seg1_col
OpEmitVertex

OpStore %ptr_out_pos %seg1_minus
OpStore %outColor %seg1_col
OpEmitVertex

OpEndPrimitive

%tan2_x = OpCompositeExtract %float %seg2_tan 0
%tan2_y = OpCompositeExtract %float %seg2_tan 1
%neg_tan2_y = OpFNegate %float %tan2_y
%perp2 = OpCompositeConstruct %v2float %neg_tan2_y %tan2_x
%off2_ndc = OpVectorTimesScalar %v2float %perp2 %float_thickness
%seg2_w = OpCompositeExtract %float %seg2_pos 3
%off2_clip = OpVectorTimesScalar %v2float %off2_ndc %seg2_w
%off2_x = OpCompositeExtract %float %off2_clip 0
%off2_y = OpCompositeExtract %float %off2_clip 1
%off2_4 = OpCompositeConstruct %v4float %off2_x %off2_y %float_0 %float_0
%seg2_plus = OpFAdd %v4float %seg2_pos %off2_4
%seg2_minus = OpFSub %v4float %seg2_pos %off2_4

%tan3_x = OpCompositeExtract %float %tan_end 0
%tan3_y = OpCompositeExtract %float %tan_end 1
%neg_tan3_y = OpFNegate %float %tan3_y
%perp3 = OpCompositeConstruct %v2float %neg_tan3_y %tan3_x
%off3_ndc = OpVectorTimesScalar %v2float %perp3 %float_thickness
%p1_out_w = OpCompositeExtract %float %p1 3
%off3_clip = OpVectorTimesScalar %v2float %off3_ndc %p1_out_w
%off3_x = OpCompositeExtract %float %off3_clip 0
%off3_y = OpCompositeExtract %float %off3_clip 1
%off3_4 = OpCompositeConstruct %v4float %off3_x %off3_y %float_0 %float_0
%p1_plus = OpFAdd %v4float %p1 %off3_4
%p1_minus = OpFSub %v4float %p1 %off3_4

OpStore %ptr_out_pos %seg2_plus
OpStore %outColor %seg2_col
OpEmitVertex

OpStore %ptr_out_pos %seg2_minus
OpStore %outColor %seg2_col
OpEmitVertex

OpStore %ptr_out_pos %p1_plus
OpStore %outColor %col1
OpEmitVertex

OpStore %ptr_out_pos %p1_minus
OpStore %outColor %col1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
