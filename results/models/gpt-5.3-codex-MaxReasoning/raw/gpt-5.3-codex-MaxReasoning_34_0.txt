import math
import random
import sys


def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return

    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = int(next(it))

    depot_x = float(next(it))
    depot_y = float(next(it))

    coords = []
    demands = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        coords.append((x, y))
        demands.append(d)

    if n == 0:
        sys.stdout.write("0.000000\n0\n")
        return

    total_demand = sum(demands)
    if any(d > capacity for d in demands) or total_demand > k * capacity:
        sys.stdout.write("-1\n0\n")
        return

    random.seed(0)

    d0 = [math.hypot(x - depot_x, y - depot_y) for x, y in coords]

    use_matrix = n <= 700
    dist_matrix = None
    if use_matrix:
        dist_matrix = [[0.0] * n for _ in range(n)]
        for i in range(n):
            xi, yi = coords[i]
            row = dist_matrix[i]
            for j in range(i + 1, n):
                xj, yj = coords[j]
                d = math.hypot(xi - xj, yi - yj)
                row[j] = d
                dist_matrix[j][i] = d

    def dist(i: int, j: int) -> float:
        if i == j:
            return 0.0
        if dist_matrix is not None:
            return dist_matrix[i][j]
        xi, yi = coords[i]
        xj, yj = coords[j]
        return math.hypot(xi - xj, yi - yj)

    def edge(u: int, v: int) -> float:
        if u == -1:
            return d0[v]
        if v == -1:
            return d0[u]
        return dist(u, v)

    def route_load(route):
        return sum(demands[c] for c in route)

    def route_cost(route) -> float:
        if not route:
            return 0.0
        total = d0[route[0]] + d0[route[-1]]
        for i in range(len(route) - 1):
            total += dist(route[i], route[i + 1])
        return total

    def total_cost(routes) -> float:
        return sum(route_cost(r) for r in routes)

    def validate(routes) -> bool:
        if len(routes) > k:
            return False
        seen = [0] * n
        for r in routes:
            if not r:
                return False
            if route_load(r) > capacity:
                return False
            for c in r:
                if c < 0 or c >= n:
                    return False
                seen[c] += 1
                if seen[c] > 1:
                    return False
        return all(v == 1 for v in seen)

    def nearest_neighbor_order(nodes):
        if len(nodes) <= 2:
            return nodes[:]
        unvisited = set(nodes)
        route = []
        current = -1
        while unvisited:
            if current == -1:
                nxt = min(unvisited, key=lambda c: d0[c])
            else:
                nxt = min(unvisited, key=lambda c: dist(current, c))
            route.append(nxt)
            unvisited.remove(nxt)
            current = nxt
        return route

    def two_opt(route):
        m = len(route)
        if m < 4 or m > 80:
            return route
        improved = True
        passes = 0
        while improved and passes < 2:
            improved = False
            passes += 1
            for i in range(m - 1):
                a = -1 if i == 0 else route[i - 1]
                b = route[i]
                for j in range(i + 1, m):
                    c = route[j]
                    d = -1 if j == m - 1 else route[j + 1]
                    old = edge(a, b) + edge(c, d)
                    new = edge(a, c) + edge(b, d)
                    if new + 1e-12 < old:
                        route[i : j + 1] = reversed(route[i : j + 1])
                        improved = True
        return route

    def clarke_wright_routes():
        routes_nodes = {i: [i] for i in range(n)}
        routes_load = {i: demands[i] for i in range(n)}
        route_of = list(range(n))

        savings = []
        for i in range(n):
            for j in range(i + 1, n):
                s = d0[i] + d0[j] - dist(i, j)
                savings.append((s, i, j))
        savings.sort(reverse=True)

        for _, i, j in savings:
            ri = route_of[i]
            rj = route_of[j]
            if ri == rj:
                continue

            if routes_load[ri] + routes_load[rj] > capacity:
                continue

            a = routes_nodes[ri]
            b = routes_nodes[rj]

            merged = None
            if a[0] == i and b[0] == j:
                merged = a[::-1] + b
            elif a[0] == i and b[-1] == j:
                merged = b + a
            elif a[-1] == i and b[0] == j:
                merged = a + b
            elif a[-1] == i and b[-1] == j:
                merged = a + b[::-1]

            if merged is None:
                continue

            routes_nodes[ri] = merged
            routes_load[ri] += routes_load[rj]

            for c in b:
                route_of[c] = ri

            del routes_nodes[rj]
            del routes_load[rj]

        routes = list(routes_nodes.values())
        loads = [route_load(r) for r in routes]
        return routes, loads

    def sweep_routes():
        order = sorted(
            range(n),
            key=lambda i: math.atan2(coords[i][1] - depot_y, coords[i][0] - depot_x),
        )
        routes = []
        loads = []
        cur = []
        cur_load = 0
        for c in order:
            d = demands[c]
            if cur and cur_load + d > capacity:
                routes.append(cur)
                loads.append(cur_load)
                cur = [c]
                cur_load = d
            else:
                cur.append(c)
                cur_load += d
        if cur:
            routes.append(cur)
            loads.append(cur_load)
        return routes, loads

    def force_reduce(routes, loads):
        routes = [r[:] for r in routes]
        loads = loads[:]

        while len(routes) > k:
            best = None
            m = len(routes)
            for a in range(m):
                ra = routes[a]
                a0, a1 = ra[0], ra[-1]
                for b in range(a + 1, m):
                    if loads[a] + loads[b] > capacity:
                        continue
                    rb = routes[b]
                    b0, b1 = rb[0], rb[-1]

                    options = [
                        (dist(a1, b0) - d0[a1] - d0[b0], ra + rb),
                        (dist(a1, b1) - d0[a1] - d0[b1], ra + rb[::-1]),
                        (dist(a0, b0) - d0[a0] - d0[b0], ra[::-1] + rb),
                        (dist(a0, b1) - d0[a0] - d0[b1], ra[::-1] + rb[::-1]),
                    ]
                    delta, merged = min(options, key=lambda x: x[0])

                    if best is None or delta < best[0]:
                        best = (delta, a, b, merged)

            if best is None:
                break

            _, a, b, merged = best
            routes[a] = merged
            loads[a] += loads[b]
            routes.pop(b)
            loads.pop(b)

        return routes, loads

    def randomized_repack(attempts):
        best = None
        base_order = sorted(range(n), key=lambda c: demands[c], reverse=True)

        for t in range(attempts):
            order = base_order[:]
            if t > 0:
                for i in range(len(order) - 1, 0, -1):
                    j = random.randint(max(0, i - 4), i)
                    order[i], order[j] = order[j], order[i]

            bins = [[] for _ in range(k)]
            rem = [capacity] * k
            ok = True

            for c in order:
                dc = demands[c]
                candidates = [b for b in range(k) if rem[b] >= dc]
                if not candidates:
                    ok = False
                    break

                if len(candidates) > 1 and random.random() < 0.3:
                    ranked = sorted(candidates, key=lambda b: rem[b] - dc)
                    pick = random.choice(ranked[: min(3, len(ranked))])
                else:
                    pick = min(candidates, key=lambda b: rem[b] - dc)

                bins[pick].append(c)
                rem[pick] -= dc

            if not ok:
                continue

            routes = [b for b in bins if b]
            for i, r in enumerate(routes):
                ordered = nearest_neighbor_order(r)
                ordered = two_opt(ordered)
                routes[i] = ordered

            if not validate(routes):
                continue

            cost = total_cost(routes)
            if best is None or cost < best[0]:
                best = (cost, routes)

        return best

    candidates = []

    if n <= 700:
        base_routes, base_loads = clarke_wright_routes()
    else:
        base_routes, base_loads = sweep_routes()

    for i in range(len(base_routes)):
        base_routes[i] = two_opt(base_routes[i])

    reduced_routes, reduced_loads = force_reduce(base_routes, base_loads)
    if validate(reduced_routes):
        candidates.append((total_cost(reduced_routes), reduced_routes))

    need_fallback = (not candidates) or (len(reduced_routes) > k)
    if need_fallback:
        attempts = 220 if n <= 250 else 120
        packed = randomized_repack(attempts)
        if packed is not None:
            candidates.append(packed)

    if not candidates:
        sys.stdout.write("-1\n0\n")
        return

    best_cost, best_routes = min(candidates, key=lambda x: x[0])

    out = [f"{best_cost:.6f}", str(len(best_routes))]
    out.extend(" ".join(map(str, r)) for r in best_routes)
    sys.stdout.write("\n".join(out) + "\n")


if __name__ == "__main__":
    solve()
