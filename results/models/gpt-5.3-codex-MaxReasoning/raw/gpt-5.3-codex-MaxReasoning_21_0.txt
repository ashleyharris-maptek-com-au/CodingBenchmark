#include <algorithm>
#include <array>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using Point = std::vector<int>;

struct PointHash {
    std::size_t operator()(const Point& p) const noexcept {
        std::size_t h = 1469598103934665603ull;
        for (int x : p) {
            std::size_t v = static_cast<std::size_t>(static_cast<std::uint64_t>(x) ^ 0x9e3779b97f4a7c15ull);
            h ^= v + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        }
        return h;
    }
};

using PointSet = std::unordered_set<Point, PointHash>;
using PointCountMap = std::unordered_map<Point, int, PointHash>;

static inline bool samePoint(const Point& a, const Point& b) {
    return a == b;
}

static inline bool inBounds(const Point& p, const std::vector<int>& bounds) {
    for (std::size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

static inline int manhattan(const Point& a, const Point& b) {
    long long sum = 0;
    for (std::size_t i = 0; i < a.size(); ++i) {
        sum += std::llabs(static_cast<long long>(a[i]) - static_cast<long long>(b[i]));
    }
    if (sum > std::numeric_limits<int>::max()) return std::numeric_limits<int>::max();
    return static_cast<int>(sum);
}

struct AStarNode {
    int f;
    int g;
    Point p;
};

struct AStarCmp {
    bool operator()(const AStarNode& a, const AStarNode& b) const {
        if (a.f != b.f) return a.f > b.f;
        return a.g < b.g;
    }
};

static std::vector<Point> reconstructPath(
    const Point& start,
    const Point& goal,
    const std::unordered_map<Point, Point, PointHash>& parent
) {
    std::vector<Point> path;
    Point cur = goal;
    path.push_back(cur);
    while (!samePoint(cur, start)) {
        auto it = parent.find(cur);
        if (it == parent.end()) {
            return {};
        }
        cur = it->second;
        path.push_back(cur);
    }
    std::reverse(path.begin(), path.end());
    return path;
}

static std::vector<Point> boundedAStar(
    const Point& start,
    const Point& goal,
    const std::vector<int>& bounds,
    const PointSet& obstacles,
    const PointCountMap& occ,
    const Point& tail,
    int nodeLimit
) {
    if (samePoint(start, goal)) return {start};

    std::priority_queue<AStarNode, std::vector<AStarNode>, AStarCmp> pq;
    std::unordered_map<Point, int, PointHash> gScore;
    std::unordered_map<Point, Point, PointHash> parent;
    gScore.reserve(static_cast<std::size_t>(nodeLimit) * 2ull + 64ull);
    parent.reserve(static_cast<std::size_t>(nodeLimit) * 2ull + 64ull);

    gScore[start] = 0;
    pq.push(AStarNode{manhattan(start, goal), 0, start});

    int expanded = 0;
    const int D = static_cast<int>(start.size());

    while (!pq.empty() && expanded < nodeLimit) {
        AStarNode cur = pq.top();
        pq.pop();

        auto git = gScore.find(cur.p);
        if (git == gScore.end() || git->second != cur.g) continue;
        if (samePoint(cur.p, goal)) {
            return reconstructPath(start, goal, parent);
        }

        ++expanded;

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Point np = cur.p;
                np[axis] += dir;
                if (!inBounds(np, bounds)) continue;
                if (obstacles.find(np) != obstacles.end()) continue;

                auto oit = occ.find(np);
                if (oit != occ.end() && !samePoint(np, tail) && !samePoint(np, goal)) {
                    continue;
                }

                int ng = cur.g + 1;
                auto ngit = gScore.find(np);
                if (ngit == gScore.end() || ng < ngit->second) {
                    gScore[np] = ng;
                    parent[np] = cur.p;
                    int h = manhattan(np, goal);
                    int f = (h >= std::numeric_limits<int>::max() - ng) ? std::numeric_limits<int>::max() : (ng + h);
                    pq.push(AStarNode{f, ng, std::move(np)});
                }
            }
        }
    }

    return {};
}

static std::vector<Point> topKNearestFoods(const Point& head, const PointSet& foods, int k) {
    std::vector<std::pair<int, Point>> best;
    best.reserve(static_cast<std::size_t>(k));

    for (const Point& f : foods) {
        int d = manhattan(head, f);
        if (static_cast<int>(best.size()) < k) {
            best.push_back({d, f});
            continue;
        }

        int worstIdx = 0;
        for (int i = 1; i < k; ++i) {
            if (best[i].first > best[worstIdx].first) worstIdx = i;
        }
        if (d < best[worstIdx].first) {
            best[worstIdx] = {d, f};
        }
    }

    std::sort(best.begin(), best.end(), [](const auto& a, const auto& b) {
        return a.first < b.first;
    });

    std::vector<Point> out;
    out.reserve(best.size());
    for (auto& p : best) out.push_back(std::move(p.second));
    return out;
}

static int nearestFoodDistance(const Point& p, const PointSet& foods) {
    if (foods.empty()) return 0;
    int best = std::numeric_limits<int>::max();
    for (const Point& f : foods) {
        best = std::min(best, manhattan(p, f));
    }
    return best;
}

static bool occupiedAfterMove(
    const Point& cell,
    const Point& nextHead,
    bool grow,
    const Point& oldTail,
    const PointCountMap& occ
) {
    if (samePoint(cell, nextHead)) return true;
    auto it = occ.find(cell);
    if (it == occ.end()) return false;
    if (!grow && samePoint(cell, oldTail)) return false;
    return true;
}

static int floodFreedomAfterMove(
    const Point& nextHead,
    bool grow,
    const std::vector<int>& bounds,
    const PointSet& obstacles,
    const PointCountMap& occ,
    const Point& oldTail,
    int cap
) {
    std::queue<Point> q;
    PointSet seen;
    seen.reserve(static_cast<std::size_t>(cap) * 2ull + 64ull);

    q.push(nextHead);
    seen.insert(nextHead);

    const int D = static_cast<int>(nextHead.size());

    while (!q.empty() && static_cast<int>(seen.size()) < cap) {
        Point cur = q.front();
        q.pop();

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Point np = cur;
                np[axis] += dir;
                if (!inBounds(np, bounds)) continue;
                if (obstacles.find(np) != obstacles.end()) continue;
                if (occupiedAfterMove(np, nextHead, grow, oldTail, occ)) continue;
                if (seen.insert(np).second) {
                    q.push(std::move(np));
                    if (static_cast<int>(seen.size()) >= cap) break;
                }
            }
            if (static_cast<int>(seen.size()) >= cap) break;
        }
    }

    return static_cast<int>(seen.size());
}

static int openNeighborsAfterMove(
    const Point& nextHead,
    bool grow,
    const std::vector<int>& bounds,
    const PointSet& obstacles,
    const PointCountMap& occ,
    const Point& oldTail
) {
    int count = 0;
    const int D = static_cast<int>(nextHead.size());

    for (int axis = 0; axis < D; ++axis) {
        for (int dir : {-1, 1}) {
            Point np = nextHead;
            np[axis] += dir;
            if (!inBounds(np, bounds)) continue;
            if (obstacles.find(np) != obstacles.end()) continue;
            if (occupiedAfterMove(np, nextHead, grow, oldTail, occ)) continue;
            ++count;
        }
    }

    return count;
}

struct MoveOption {
    int axis;
    int dir;
    Point next;
    int score;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int D = 0;
    int maxTurns = 0;
    if (!(std::cin >> D >> maxTurns)) {
        return 0;
    }

    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];

    int snakeLengthInput = 1;
    Point start(D);
    std::cin >> snakeLengthInput;
    for (int i = 0; i < D; ++i) std::cin >> start[i];

    int F = 0;
    std::cin >> F;
    PointSet foods;
    foods.reserve(static_cast<std::size_t>(F) * 2ull + 64ull);
    for (int i = 0; i < F; ++i) {
        Point p(D);
        for (int j = 0; j < D; ++j) std::cin >> p[j];
        foods.insert(std::move(p));
    }

    int O = 0;
    std::cin >> O;
    PointSet obstacles;
    obstacles.reserve(static_cast<std::size_t>(O) * 2ull + 64ull);
    for (int i = 0; i < O; ++i) {
        Point p(D);
        for (int j = 0; j < D; ++j) std::cin >> p[j];
        obstacles.insert(std::move(p));
    }

    // Build a deterministic initial body from head + length when only head is provided.
    // If a full initial body is not available in input, this gives us a valid internal state.
    std::deque<Point> snake;
    PointCountMap occ;
    occ.reserve(static_cast<std::size_t>(std::max(16, snakeLengthInput * 2 + 16)));

    snake.push_back(start);
    occ[start] = 1;

    for (int seg = 1; seg < snakeLengthInput; ++seg) {
        const Point& tail = snake.back();
        bool placed = false;

        for (int axis = 0; axis < D && !placed; ++axis) {
            for (int dir : {-1, 1}) {
                Point np = tail;
                np[axis] += dir;
                if (!inBounds(np, bounds)) continue;
                if (obstacles.find(np) != obstacles.end()) continue;
                if (occ.find(np) != occ.end()) continue;

                snake.push_back(np);
                occ[np] = 1;
                placed = true;
                break;
            }
        }

        if (!placed) break;
    }

    // Any food currently occupied by body is considered unavailable.
    for (const Point& part : snake) {
        auto it = foods.find(part);
        if (it != foods.end()) foods.erase(it);
    }

    std::vector<std::pair<int, int>> outputs;
    outputs.reserve(static_cast<std::size_t>(std::max(0, maxTurns)));

    for (int turn = 0; turn < maxTurns; ++turn) {
        if (foods.empty()) break;

        const Point& head = snake.front();
        const Point& oldTail = snake.back();

        std::vector<MoveOption> options;
        options.reserve(static_cast<std::size_t>(2 * D));

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Point np = head;
                np[axis] += dir;

                if (!inBounds(np, bounds)) continue;
                if (obstacles.find(np) != obstacles.end()) continue;

                bool grow = (foods.find(np) != foods.end());
                auto oit = occ.find(np);
                if (oit != occ.end()) {
                    bool canUseTail = (!grow && samePoint(np, oldTail));
                    if (!canUseTail) continue;
                }

                options.push_back(MoveOption{axis, dir, std::move(np), std::numeric_limits<int>::min()});
            }
        }

        if (options.empty()) break;

        // Try to get a planned first step from bounded A* toward one of the nearest foods.
        bool hasPathStep = false;
        Point plannedStep;
        {
            int k = std::min<int>(6, static_cast<int>(foods.size()));
            if (k > 0) {
                std::vector<Point> candidates = topKNearestFoods(head, foods, k);
                int nodeLimit = 2000 + D * 300 + static_cast<int>(snake.size()) * 8;
                if (nodeLimit > 50000) nodeLimit = 50000;

                for (const Point& goal : candidates) {
                    std::vector<Point> path = boundedAStar(head, goal, bounds, obstacles, occ, oldTail, nodeLimit);
                    if (path.size() >= 2) {
                        plannedStep = path[1];
                        hasPathStep = true;
                        break;
                    }
                }
            }
        }

        int bestIdx = 0;
        int bestScore = std::numeric_limits<int>::min();

        for (int i = 0; i < static_cast<int>(options.size()); ++i) {
            MoveOption& m = options[i];
            bool grow = (foods.find(m.next) != foods.end());

            int freedomCap = std::max(80, std::min(8000, static_cast<int>(snake.size()) * 4 + 256));
            int freedom = floodFreedomAfterMove(m.next, grow, bounds, obstacles, occ, oldTail, freedomCap);
            int degree = openNeighborsAfterMove(m.next, grow, bounds, obstacles, occ, oldTail);
            int dist = nearestFoodDistance(m.next, foods);

            int score = 0;
            if (grow) score += 200000;
            if (hasPathStep && samePoint(m.next, plannedStep)) score += 60000;
            score += freedom * 8;
            score += degree * 1200;
            score -= dist * 25;

            if (!grow && degree == 0) score -= 120000;
            if (!grow && freedom < static_cast<int>(snake.size() / 2)) score -= 20000;

            m.score = score;

            if (score > bestScore) {
                bestScore = score;
                bestIdx = i;
            }
        }

        const MoveOption& chosen = options[bestIdx];
        outputs.push_back({chosen.axis, chosen.dir});

        bool grow = (foods.find(chosen.next) != foods.end());

        snake.push_front(chosen.next);
        ++occ[chosen.next];

        if (grow) {
            foods.erase(chosen.next);
        } else {
            Point tail = snake.back();
            snake.pop_back();
            auto it = occ.find(tail);
            if (it != occ.end()) {
                --(it->second);
                if (it->second <= 0) occ.erase(it);
            }
        }
    }

    for (const auto& mv : outputs) {
        std::cout << mv.first << ' ' << mv.second << '\n';
    }

    return 0;
}
