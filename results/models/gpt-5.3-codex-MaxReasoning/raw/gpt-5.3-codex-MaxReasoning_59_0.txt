'use strict';

const fs = require('fs');

const rawInput = fs.readFileSync(0, 'utf8');
let data;
try {
  data = rawInput.trim() ? JSON.parse(rawInput) : {};
} catch {
  data = {};
}

const gridRaw = Array.isArray(data.probability) ? data.probability : [];
const H = gridRaw.length;
const W = H > 0 && Array.isArray(gridRaw[0]) ? gridRaw[0].length : 0;
const dronesRaw = Array.isArray(data.drones) ? data.drones : [];
const maxStepsRaw = Number(data.maxStepsPerDrone);
const maxSteps = Number.isFinite(maxStepsRaw) ? Math.max(0, Math.floor(maxStepsRaw)) : 0;

if (H === 0 || W === 0) {
  process.stdout.write(JSON.stringify({ paths: dronesRaw.map(() => []) }));
  process.exit(0);
}

const N = H * W;
const obstacle = new Uint8Array(N);
const values = new Float64Array(N);
const validIndices = [];
let positiveCount = 0;

for (let y = 0; y < H; y++) {
  const row = Array.isArray(gridRaw[y]) ? gridRaw[y] : [];
  for (let x = 0; x < W; x++) {
    const idx = y * W + x;
    const v = Number(row[x]);
    if (!Number.isFinite(v) || v < 0) {
      obstacle[idx] = 1;
      continue;
    }
    validIndices.push(idx);
    if (v > 0) {
      values[idx] = v;
      positiveCount++;
    }
  }
}

if (validIndices.length === 0) {
  process.stdout.write(JSON.stringify({ paths: dronesRaw.map(() => []) }));
  process.exit(0);
}

const prefixW = W + 1;
const prefix = new Float64Array((H + 1) * prefixW);
for (let y = 0; y < H; y++) {
  let rowSum = 0;
  const pRow = (y + 1) * prefixW;
  const pPrev = y * prefixW;
  const base = y * W;
  for (let x = 0; x < W; x++) {
    rowSum += values[base + x];
    prefix[pRow + x + 1] = prefix[pPrev + x + 1] + rowSum;
  }
}

function squareSum(x1, y1, x2, y2) {
  const a = y1 * prefixW + x1;
  const b = y1 * prefixW + (x2 + 1);
  const c = (y2 + 1) * prefixW + x1;
  const d = (y2 + 1) * prefixW + (x2 + 1);
  return prefix[d] - prefix[b] - prefix[c] + prefix[a];
}

const potential = new Float64Array(N);
const radius = 2;
for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    const x1 = x - radius < 0 ? 0 : x - radius;
    const y1 = y - radius < 0 ? 0 : y - radius;
    const x2 = x + radius >= W ? W - 1 : x + radius;
    const y2 = y + radius >= H ? H - 1 : y + radius;
    potential[y * W + x] = squareSum(x1, y1, x2, y2);
  }
}

function idxToCoord(idx) {
  return [idx % W, (idx / W) | 0];
}

function isInside(x, y) {
  return x >= 0 && y >= 0 && x < W && y < H;
}

function nearestValidIndex(x, y) {
  if (isInside(x, y)) {
    const idx = y * W + x;
    if (!obstacle[idx]) {
      return idx;
    }
  }

  let bestIdx = validIndices[0];
  let bestDist = Number.POSITIVE_INFINITY;
  for (let i = 0; i < validIndices.length; i++) {
    const idx = validIndices[i];
    const vx = idx % W;
    const vy = (idx / W) | 0;
    const d = Math.abs(vx - x) + Math.abs(vy - y);
    if (d < bestDist) {
      bestDist = d;
      bestIdx = idx;
    }
  }
  return bestIdx;
}

function asIntCoord(value) {
  const n = Number(value);
  return Number.isFinite(n) ? Math.round(n) : 0;
}

const covered = new Uint8Array(N);
const visits = new Uint16Array(N);
let uncoveredPositive = positiveCount;

function coverCell(idx) {
  if (values[idx] > 0 && covered[idx] === 0) {
    covered[idx] = 1;
    uncoveredPositive--;
  }
}

const states = dronesRaw.map((d) => {
  const sx = asIntCoord(d && d.x);
  const sy = asIntCoord(d && d.y);
  const start = nearestValidIndex(sx, sy);
  visits[start] = visits[start] < 65535 ? visits[start] + 1 : 65535;
  coverCell(start);
  return {
    cur: start,
    remaining: maxSteps,
    goal: -1,
    plan: [],
    cooldown: 0,
    path: [idxToCoord(start)],
  };
});

if (states.length === 0) {
  process.stdout.write(JSON.stringify({ paths: [] }));
  process.exit(0);
}

const bfsSeen = new Int32Array(N);
const bfsDist = new Int32Array(N);
const bfsParent = new Int32Array(N);
const bfsQueue = new Int32Array(N);
let bfsToken = 1;

function runBfs(startIdx, maxDepth) {
  bfsToken++;
  if (bfsToken === 2147483647) {
    bfsSeen.fill(0);
    bfsToken = 1;
  }
  const token = bfsToken;

  let head = 0;
  let tail = 0;
  bfsSeen[startIdx] = token;
  bfsDist[startIdx] = 0;
  bfsParent[startIdx] = -1;
  bfsQueue[tail++] = startIdx;

  while (head < tail) {
    const cur = bfsQueue[head++];
    const d = bfsDist[cur];
    if (d >= maxDepth) {
      continue;
    }

    const x = cur % W;
    const y = (cur / W) | 0;

    if (x > 0) {
      const n = cur - 1;
      if (!obstacle[n] && bfsSeen[n] !== token) {
        bfsSeen[n] = token;
        bfsDist[n] = d + 1;
        bfsParent[n] = cur;
        bfsQueue[tail++] = n;
      }
    }
    if (x + 1 < W) {
      const n = cur + 1;
      if (!obstacle[n] && bfsSeen[n] !== token) {
        bfsSeen[n] = token;
        bfsDist[n] = d + 1;
        bfsParent[n] = cur;
        bfsQueue[tail++] = n;
      }
    }
    if (y > 0) {
      const n = cur - W;
      if (!obstacle[n] && bfsSeen[n] !== token) {
        bfsSeen[n] = token;
        bfsDist[n] = d + 1;
        bfsParent[n] = cur;
        bfsQueue[tail++] = n;
      }
    }
    if (y + 1 < H) {
      const n = cur + W;
      if (!obstacle[n] && bfsSeen[n] !== token) {
        bfsSeen[n] = token;
        bfsDist[n] = d + 1;
        bfsParent[n] = cur;
        bfsQueue[tail++] = n;
      }
    }
  }

  return { token, count: tail };
}

function frontierCount(idx) {
  let c = 0;
  const x = idx % W;
  const y = (idx / W) | 0;

  if (x > 0) {
    const n = idx - 1;
    if (values[n] > 0 && !covered[n]) c++;
  }
  if (x + 1 < W) {
    const n = idx + 1;
    if (values[n] > 0 && !covered[n]) c++;
  }
  if (y > 0) {
    const n = idx - W;
    if (values[n] > 0 && !covered[n]) c++;
  }
  if (y + 1 < H) {
    const n = idx + W;
    if (values[n] > 0 && !covered[n]) c++;
  }

  return c;
}

function isNeighborOrSame(a, b) {
  if (a === b) return true;
  if (a + W === b || a - W === b) return true;
  const ax = a % W;
  if (a + 1 === b && ax + 1 < W) return true;
  if (a - 1 === b && ax > 0) return true;
  return false;
}

function chooseLocalMove(cur) {
  let best = cur;
  let bestScore = -Infinity;

  function consider(idx) {
    if (obstacle[idx]) return;
    let score = 0;
    if (!covered[idx]) score += values[idx] * 4.0;
    score += potential[idx] * 0.05;
    score += frontierCount(idx) * 0.3;
    score -= visits[idx] * 0.05;

    if (score > bestScore + 1e-12) {
      bestScore = score;
      best = idx;
    } else if (Math.abs(score - bestScore) <= 1e-12) {
      if (visits[idx] < visits[best]) {
        best = idx;
      }
    }
  }

  const x = cur % W;
  const y = (cur / W) | 0;

  consider(cur);
  if (x > 0) consider(cur - 1);
  if (x + 1 < W) consider(cur + 1);
  if (y > 0) consider(cur - W);
  if (y + 1 < H) consider(cur + W);

  return best;
}

function buildReservations(excludeIdx) {
  const map = new Map();
  for (let i = 0; i < states.length; i++) {
    if (i === excludeIdx) continue;
    const g = states[i].goal;
    if (g >= 0) {
      map.set(g, (map.get(g) || 0) + 1);
    }
  }
  return map;
}

function planForState(state, stateIndex) {
  if (state.remaining <= 0 || uncoveredPositive <= 0) {
    return false;
  }

  const reservations = buildReservations(stateIndex);
  const bfs = runBfs(state.cur, state.remaining);

  let bestIdx = -1;
  let bestScore = -1;
  let bestDist = 0;

  const POT_W = 0.18;
  const DIST_BIAS = 1.2;
  const RESERVE_W = 2.3;
  const VISIT_W = 0.06;

  for (let i = 0; i < bfs.count; i++) {
    const idx = bfsQueue[i];
    const d = bfsDist[idx];

    if (d === 0) continue;
    if (values[idx] <= 0 || covered[idx]) continue;

    let score = (values[idx] + POT_W * potential[idx]) / (d + DIST_BIAS);

    const r = reservations.get(idx) || 0;
    if (r > 0) {
      score /= 1 + RESERVE_W * r;
    }

    const v = visits[idx];
    if (v > 0) {
      score /= 1 + VISIT_W * v;
    }

    if (score > bestScore + 1e-12) {
      bestScore = score;
      bestIdx = idx;
      bestDist = d;
    } else if (Math.abs(score - bestScore) <= 1e-12 && bestIdx >= 0) {
      if (d < bestDist) {
        bestIdx = idx;
        bestDist = d;
      }
    }
  }

  if (bestIdx < 0) {
    return false;
  }

  const plan = [];
  let cur = bestIdx;
  while (cur !== state.cur) {
    plan.push(cur);
    const p = bfsParent[cur];
    if (p === -1) {
      return false;
    }
    cur = p;
  }

  if (plan.length === 0) {
    return false;
  }

  plan.reverse();
  state.plan = plan;
  state.goal = bestIdx;
  return true;
}

for (let step = 0; step < maxSteps; step++) {
  let active = false;

  for (let offset = 0; offset < states.length; offset++) {
    const i = (step + offset) % states.length;
    const state = states[i];

    if (state.remaining <= 0) continue;
    active = true;

    if (state.goal >= 0 && covered[state.goal]) {
      state.goal = -1;
      state.plan.length = 0;
    }

    if (state.plan.length === 0) {
      if (state.cooldown > 0) {
        state.cooldown--;
      } else {
        const planned = planForState(state, i);
        if (!planned) {
          state.cooldown = state.remaining > 3 ? 3 : state.remaining;
        }
      }
    }

    let next = -1;
    if (state.plan.length > 0) {
      next = state.plan.shift();
      if (!isNeighborOrSame(state.cur, next) || obstacle[next]) {
        next = -1;
        state.goal = -1;
        state.plan.length = 0;
      }
    }

    if (next === -1) {
      next = chooseLocalMove(state.cur);
      if (state.goal >= 0 && next !== state.cur) {
        state.goal = -1;
        state.plan.length = 0;
      }
    }

    state.cur = next;
    state.remaining--;
    visits[next] = visits[next] < 65535 ? visits[next] + 1 : 65535;
    coverCell(next);
    state.path.push(idxToCoord(next));

    if (state.goal === next) {
      state.goal = -1;
      state.plan.length = 0;
    }
  }

  if (!active) break;
}

process.stdout.write(JSON.stringify({ paths: states.map((s) => s.path) }));
