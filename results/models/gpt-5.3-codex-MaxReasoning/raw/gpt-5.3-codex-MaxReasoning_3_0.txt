use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read, Write};

const EPS: f64 = 1e-9;
const UNSET: usize = usize::MAX;

struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut val: usize = 0;
        let mut found = false;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if b.is_ascii_digit() {
                found = true;
                val = val.saturating_mul(10).saturating_add((b - b'0') as usize);
                self.idx += 1;
            } else {
                break;
            }
        }
        if !found {
            return None;
        }
        Some(val)
    }
}

#[derive(Clone, Copy)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let s = if seed == 0 {
            0x9E37_79B9_7F4A_7C15
        } else {
            seed
        };
        Self { state: s }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }

    fn next_f64(&mut self) -> f64 {
        ((self.next_u64() >> 11) as f64) * (1.0 / ((1u64 << 53) as f64))
    }

    fn next_usize(&mut self, upper: usize) -> usize {
        if upper <= 1 {
            0
        } else {
            (self.next_u64() as usize) % upper
        }
    }
}

fn splitmix64(mut x: u64) -> u64 {
    x = x.wrapping_add(0x9E37_79B9_7F4A_7C15);
    x = (x ^ (x >> 30)).wrapping_mul(0xBF58_476D_1CE4_E5B9);
    x = (x ^ (x >> 27)).wrapping_mul(0x94D0_49BB_1331_11EB);
    x ^ (x >> 31)
}

fn normalize_center(v: &mut [f64]) -> f64 {
    if v.is_empty() {
        return 0.0;
    }
    let mean = v.iter().sum::<f64>() / (v.len() as f64);
    let mut norm2 = 0.0;
    for x in v.iter_mut() {
        *x -= mean;
        norm2 += *x * *x;
    }
    let norm = norm2.sqrt();
    if norm > EPS {
        let inv = 1.0 / norm;
        for x in v.iter_mut() {
            *x *= inv;
        }
    }
    norm
}

fn orthogonalize(v: &mut [f64], base: &[f64]) {
    let mut dot = 0.0;
    for i in 0..v.len() {
        dot += v[i] * base[i];
    }
    for i in 0..v.len() {
        v[i] -= dot * base[i];
    }
}

fn build_local_edges(
    nodes: &[usize],
    offsets: &[usize],
    nbrs: &[usize],
    inv: &[usize],
) -> Vec<(u32, u32)> {
    let mut edges = Vec::with_capacity(nodes.len().saturating_mul(2));
    for (li, &v) in nodes.iter().enumerate() {
        for &u in &nbrs[offsets[v]..offsets[v + 1]] {
            let lj = inv[u];
            if lj != UNSET && li < lj {
                edges.push((li as u32, lj as u32));
            }
        }
    }
    edges
}

fn init_positions(n: usize, seed: u64) -> (Vec<f64>, Vec<f64>) {
    let mut x = vec![0.0; n];
    let mut y = vec![0.0; n];
    if n == 0 {
        return (x, y);
    }
    let mut rng = XorShift64::new(seed);
    let r = (n as f64).sqrt().max(1.0);
    for i in 0..n {
        let a = 2.0 * PI * (i as f64) / (n as f64);
        x[i] = r * a.cos() + (rng.next_f64() - 0.5) * 0.2;
        y[i] = r * a.sin() + (rng.next_f64() - 0.5) * 0.2;
    }
    (x, y)
}

fn full_force_layout(n: usize, edges: &[(u32, u32)], seed: u64) -> (Vec<f64>, Vec<f64>) {
    if n <= 1 {
        return (vec![0.0; n], vec![0.0; n]);
    }
    let (mut x, mut y) = init_positions(n, seed);
    let mut fx = vec![0.0; n];
    let mut fy = vec![0.0; n];
    let iters = 220usize;
    let ideal = 2.2;
    let spring = 0.14;
    let repulse = 1.25;
    let gravity = 0.02 / (n as f64).sqrt().max(1.0);
    let mut step = 2.5 + (n as f64).sqrt() * 0.45;

    for it in 0..iters {
        fx.fill(0.0);
        fy.fill(0.0);

        for i in 0..n {
            for j in (i + 1)..n {
                let dx = x[i] - x[j];
                let dy = y[i] - y[j];
                let d2 = dx * dx + dy * dy + EPS;
                let f = repulse / d2;
                let rx = dx * f;
                let ry = dy * f;
                fx[i] += rx;
                fy[i] += ry;
                fx[j] -= rx;
                fy[j] -= ry;
            }
        }

        for &(a, b) in edges {
            let i = a as usize;
            let j = b as usize;
            let dx = x[j] - x[i];
            let dy = y[j] - y[i];
            let d2 = dx * dx + dy * dy + EPS;
            let d = d2.sqrt();
            let f = spring * (d - ideal);
            let ex = dx / d * f;
            let ey = dy / d * f;
            fx[i] += ex;
            fy[i] += ey;
            fx[j] -= ex;
            fy[j] -= ey;
        }

        for i in 0..n {
            fx[i] -= x[i] * gravity;
            fy[i] -= y[i] * gravity;
        }

        let cool = 1.0 - (it as f64) / (iters as f64);
        let temp = step * cool;
        for i in 0..n {
            let d2 = fx[i] * fx[i] + fy[i] * fy[i];
            if d2 > EPS {
                let d = d2.sqrt();
                let mv = temp.min(d);
                x[i] += fx[i] * (mv / d);
                y[i] += fy[i] * (mv / d);
            }
        }
        step *= 0.985;
    }

    (x, y)
}

fn sampled_force_layout(
    n: usize,
    edges: &[(u32, u32)],
    seed: u64,
    density: f64,
) -> (Vec<f64>, Vec<f64>) {
    if n <= 1 {
        return (vec![0.0; n], vec![0.0; n]);
    }
    let (mut x, mut y) = init_positions(n, seed ^ 0xA5A5_A5A5_A5A5_A5A5);
    let mut fx = vec![0.0; n];
    let mut fy = vec![0.0; n];
    let mut rng = XorShift64::new(seed ^ 0xD1B5_4A32_D192_ED03);

    let iters = if n <= 300 {
        180
    } else if n <= 1200 {
        120
    } else {
        84
    };
    let samples = if n <= 700 {
        28
    } else if n <= 2000 {
        20
    } else {
        14
    };

    let edge_count = edges.len().max(1) as f64;
    let ideal = ((n as f64 / edge_count).sqrt() * 2.7).clamp(0.9, 6.0);
    let spring = 0.085;
    let repulse = 1.1;
    let gravity = (0.006 + density * 0.01).clamp(0.006, 0.03);
    let scale = (n as f64) / (samples as f64);
    let mut step = ideal * 2.4 + 1.0;

    for it in 0..iters {
        fx.fill(0.0);
        fy.fill(0.0);

        for i in 0..n {
            for _ in 0..samples {
                let mut j = rng.next_usize(n - 1);
                if j >= i {
                    j += 1;
                }
                let dx = x[i] - x[j];
                let dy = y[i] - y[j];
                let d2 = dx * dx + dy * dy + EPS;
                let f = repulse * scale / d2;
                fx[i] += dx * f;
                fy[i] += dy * f;
            }
        }

        for &(a, b) in edges {
            let i = a as usize;
            let j = b as usize;
            let dx = x[j] - x[i];
            let dy = y[j] - y[i];
            let d2 = dx * dx + dy * dy + EPS;
            let d = d2.sqrt();
            let f = spring * (d - ideal);
            let ex = dx / d * f;
            let ey = dy / d * f;
            fx[i] += ex;
            fy[i] += ey;
            fx[j] -= ex;
            fy[j] -= ey;
        }

        for i in 0..n {
            fx[i] -= x[i] * gravity;
            fy[i] -= y[i] * gravity;
        }

        let cool = 1.0 - (it as f64) / (iters as f64);
        let temp = step * (0.25 + 0.75 * cool);
        for i in 0..n {
            let d2 = fx[i] * fx[i] + fy[i] * fy[i];
            if d2 > EPS {
                let d = d2.sqrt();
                let mv = temp.min(d);
                x[i] += fx[i] * (mv / d);
                y[i] += fy[i] * (mv / d);
            }
        }
        step *= 0.987;
    }

    (x, y)
}

fn matvec_norm_adj(
    nodes: &[usize],
    offsets: &[usize],
    nbrs: &[usize],
    inv: &[usize],
    inv_sqrt_deg: &[f64],
    src: &[f64],
    dst: &mut [f64],
) {
    for (li, &v) in nodes.iter().enumerate() {
        let mut sum = 0.0;
        let sv = inv_sqrt_deg[v];
        for &u in &nbrs[offsets[v]..offsets[v + 1]] {
            let lj = inv[u];
            if lj != UNSET {
                sum += src[lj] * inv_sqrt_deg[u];
            }
        }
        dst[li] = sum * sv;
    }
}

fn spectral_layout(
    nodes: &[usize],
    offsets: &[usize],
    nbrs: &[usize],
    inv: &[usize],
    inv_sqrt_deg: &[f64],
    seed: u64,
    m_edges: usize,
) -> (Vec<f64>, Vec<f64>) {
    let n = nodes.len();
    if n <= 1 {
        return (vec![0.0; n], vec![0.0; n]);
    }

    let mut v1 = vec![0.0; n];
    let mut v2 = vec![0.0; n];
    let mut w = vec![0.0; n];

    const INV53: f64 = 1.0 / ((1u64 << 53) as f64);

    for i in 0..n {
        let h = splitmix64(seed ^ (nodes[i] as u64).wrapping_mul(0x9E37_79B9_7F4A_7C15));
        v1[i] = ((h >> 11) as f64) * INV53 - 0.5;
    }
    if normalize_center(&mut v1) <= EPS {
        for i in 0..n {
            v1[i] = (i as f64) / (n as f64) - 0.5;
        }
        normalize_center(&mut v1);
    }

    let m = m_edges.max(1);
    let mut iters = (120_000_000usize / m.saturating_mul(2)).clamp(4, 24);
    if n > 80_000 {
        iters = iters.min(10);
    }
    if n > 150_000 {
        iters = iters.min(8);
    }
    if m > n.saturating_mul(40) {
        iters = iters.min(6);
    }

    for _ in 0..iters {
        matvec_norm_adj(nodes, offsets, nbrs, inv, inv_sqrt_deg, &v1, &mut w);
        if normalize_center(&mut w) <= EPS {
            break;
        }
        std::mem::swap(&mut v1, &mut w);
    }

    for i in 0..n {
        let h =
            splitmix64(seed ^ 0xD1B5_4A32_D192_ED03 ^ (nodes[i] as u64).wrapping_mul(0x94D0_49BB_1331_11EB));
        v2[i] = ((h >> 11) as f64) * INV53 - 0.5;
    }
    orthogonalize(&mut v2, &v1);
    if normalize_center(&mut v2) <= EPS {
        for i in 0..n {
            v2[i] = ((i * 13 % (n + 1)) as f64) / (n as f64) - 0.5;
        }
        orthogonalize(&mut v2, &v1);
        normalize_center(&mut v2);
    }

    for _ in 0..iters {
        matvec_norm_adj(nodes, offsets, nbrs, inv, inv_sqrt_deg, &v2, &mut w);
        orthogonalize(&mut w, &v1);
        if normalize_center(&mut w) <= EPS {
            break;
        }
        std::mem::swap(&mut v2, &mut w);
    }

    if normalize_center(&mut v2) <= EPS {
        for i in 0..n {
            let h = splitmix64(seed ^ 0xA5A5_A5A5_A5A5_A5A5 ^ (nodes[i] as u64));
            v2[i] = ((h >> 11) as f64) * INV53 - 0.5;
        }
        orthogonalize(&mut v2, &v1);
        normalize_center(&mut v2);
    }

    let spread = (n as f64).sqrt().max(1.0) * 2.0;
    for i in 0..n {
        let h1 = splitmix64(seed ^ (nodes[i] as u64).wrapping_mul(0xBF58_476D_1CE4_E5B9));
        let h2 = splitmix64(seed ^ (nodes[i] as u64).wrapping_mul(0x94D0_49BB_1331_11EB));
        let jx = (((h1 >> 11) as f64) * INV53 - 0.5) * 0.02;
        let jy = (((h2 >> 11) as f64) * INV53 - 0.5) * 0.02;
        v1[i] = v1[i] * spread + jx;
        v2[i] = v2[i] * spread + jy;
    }

    (v1, v2)
}

fn component_bbox(nodes: &[usize], x: &[f64], y: &[f64]) -> (f64, f64, f64, f64) {
    if nodes.is_empty() {
        return (0.0, 0.0, 0.0, 0.0);
    }
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for &v in nodes {
        min_x = min_x.min(x[v]);
        max_x = max_x.max(x[v]);
        min_y = min_y.min(y[v]);
        max_y = max_y.max(y[v]);
    }
    (min_x, max_x, min_y, max_y)
}

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = input.next_usize().unwrap_or(0);

    if n == 0 {
        return;
    }

    let mut edges = Vec::<(usize, usize)>::with_capacity(m);
    let mut deg = vec![0usize; n];

    for _ in 0..m {
        let u = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            continue;
        }
        deg[u] += 1;
        deg[v] += 1;
        edges.push((u, v));
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + deg[i];
    }

    let mut nbrs = vec![0usize; offsets[n]];
    let mut cur = offsets[..n].to_vec();
    for &(u, v) in &edges {
        nbrs[cur[u]] = v;
        cur[u] += 1;
        nbrs[cur[v]] = u;
        cur[v] += 1;
    }

    let mut inv_sqrt_deg = vec![0.0; n];
    for i in 0..n {
        inv_sqrt_deg[i] = if deg[i] == 0 {
            0.0
        } else {
            1.0 / (deg[i] as f64).sqrt()
        };
    }

    let mut comp_id = vec![UNSET; n];
    let mut components: Vec<Vec<usize>> = Vec::new();
    let mut comp_edges: Vec<usize> = Vec::new();
    let mut queue = Vec::<usize>::new();

    for start in 0..n {
        if comp_id[start] != UNSET {
            continue;
        }
        let cid = components.len();
        queue.clear();
        queue.push(start);
        comp_id[start] = cid;
        let mut head = 0usize;
        let mut nodes = Vec::new();
        let mut deg_sum = 0usize;

        while head < queue.len() {
            let v = queue[head];
            head += 1;
            nodes.push(v);
            let s = offsets[v];
            let e = offsets[v + 1];
            deg_sum += e - s;
            for &u in &nbrs[s..e] {
                if comp_id[u] == UNSET {
                    comp_id[u] = cid;
                    queue.push(u);
                }
            }
        }

        components.push(nodes);
        comp_edges.push(deg_sum / 2);
    }

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];
    let mut inv = vec![UNSET; n];
    let mut comp_boxes = vec![(0.0, 0.0, 0.0, 0.0); components.len()];

    for cid in 0..components.len() {
        let nodes = &components[cid];
        let nc = nodes.len();
        for (li, &v) in nodes.iter().enumerate() {
            inv[v] = li;
        }

        if nc == 1 {
            let v = nodes[0];
            x[v] = 0.0;
            y[v] = 0.0;
            comp_boxes[cid] = (-0.5, 0.5, -0.5, 0.5);
            inv[v] = UNSET;
            continue;
        }

        let mc = comp_edges[cid];
        let density = (2.0 * mc as f64) / ((nc * (nc - 1)) as f64);
        let seed = splitmix64((cid as u64).wrapping_mul(0x9E37_79B9_7F4A_7C15) ^ (nc as u64));

        let (lx, ly) = if nc <= 80 {
            let local_edges = build_local_edges(nodes, &offsets, &nbrs, &inv);
            full_force_layout(nc, &local_edges, seed)
        } else if nc <= 4000 && mc <= nc.saturating_mul(50) {
            let local_edges = build_local_edges(nodes, &offsets, &nbrs, &inv);
            sampled_force_layout(nc, &local_edges, seed, density)
        } else {
            spectral_layout(nodes, &offsets, &nbrs, &inv, &inv_sqrt_deg, seed, mc)
        };

        for (i, &v) in nodes.iter().enumerate() {
            x[v] = lx[i];
            y[v] = ly[i];
        }

        comp_boxes[cid] = component_bbox(nodes, &x, &y);

        for &v in nodes {
            inv[v] = UNSET;
        }
    }

    let mut order: Vec<usize> = (0..components.len()).collect();
    let area = |b: (f64, f64, f64, f64)| {
        let w = (b.1 - b.0).max(1.0);
        let h = (b.3 - b.2).max(1.0);
        w * h
    };
    order.sort_by(|&a, &b| {
        area(comp_boxes[b])
            .partial_cmp(&area(comp_boxes[a]))
            .unwrap_or(Ordering::Equal)
    });

    let pad = 4.0;
    let total_area: f64 = comp_boxes
        .iter()
        .map(|&b| {
            let w = (b.1 - b.0).max(1.0) + pad;
            let h = (b.3 - b.2).max(1.0) + pad;
            w * h
        })
        .sum();
    let target_row_width = total_area.sqrt().max(10.0);

    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut row_h: f64 = 0.0;

    for &cid in &order {
        let b = comp_boxes[cid];
        let w = (b.1 - b.0).max(1.0);
        let h = (b.3 - b.2).max(1.0);

        if cx > 0.0 && cx + w > target_row_width {
            cx = 0.0;
            cy += row_h + pad;
            row_h = 0.0;
        }

        let ox = cx - b.0;
        let oy = cy - b.2;
        for &v in &components[cid] {
            x[v] += ox;
            y[v] += oy;
        }

        cx += w + pad;
        row_h = row_h.max(h);
    }

    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for i in 0..n {
        min_x = min_x.min(x[i]);
        max_x = max_x.max(x[i]);
        min_y = min_y.min(y[i]);
        max_y = max_y.max(y[i]);
    }
    let mid_x = 0.5 * (min_x + max_x);
    let mid_y = 0.5 * (min_y + max_y);
    for i in 0..n {
        x[i] -= mid_x;
        y[i] -= mid_y;
    }

    let mut out = io::BufWriter::new(io::stdout().lock());
    for i in 0..n {
        let _ = writeln!(out, "{:.6} {:.6}", x[i], y[i]);
    }
}
