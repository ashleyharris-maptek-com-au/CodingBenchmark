static int popcount64(uint64_t x) {
    int c = 0;
    while (x) {
        x &= (x - 1ULL);
        ++c;
    }
    return c;
}

static vector<int> solvePathOrder(const vector<int>& order) {
    const int n = static_cast<int>(order.size());
    if (n == 0) return {};
    vector<int> dp(n + 1, 0);
    dp[1] = 1;
    for (int i = 2; i <= n; ++i) {
        dp[i] = max(dp[i - 1], dp[i - 2] + 1);
    }

    vector<int> chosen;
    for (int i = n; i >= 1;) {
        if (i == 1) {
            if (dp[1] > dp[0]) chosen.push_back(order[0]);
            break;
        }
        if (dp[i - 1] >= dp[i - 2] + 1) {
            --i;
        } else {
            chosen.push_back(order[i - 1]);
            i -= 2;
        }
    }
    return chosen;
}

static vector<int> solveMaxDegTwo(const vector<vector<int>>& adj) {
    const int n = static_cast<int>(adj.size());
    vector<int> result;
    vector<char> seen(n, 0);

    for (int s = 0; s < n; ++s) {
        if (seen[s]) continue;

        vector<int> comp;
        queue<int> q;
        q.push(s);
        seen[s] = 1;
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            comp.push_back(v);
            for (int to : adj[v]) {
                if (!seen[to]) {
                    seen[to] = 1;
                    q.push(to);
                }
            }
        }

        if (comp.size() == 1) {
            result.push_back(comp[0]);
            continue;
        }

        bool allDegTwo = true;
        for (int v : comp) {
            if (static_cast<int>(adj[v].size()) != 2) {
                allDegTwo = false;
                break;
            }
        }

        if (!allDegTwo) {
            int start = -1;
            for (int v : comp) {
                if (static_cast<int>(adj[v].size()) <= 1) {
                    start = v;
                    break;
                }
            }
            if (start == -1) start = comp[0];

            vector<int> order;
            order.reserve(comp.size());
            int prev = -1;
            int cur = start;
            while (cur != -1) {
                order.push_back(cur);
                int nxt = -1;
                for (int to : adj[cur]) {
                    if (to != prev) {
                        nxt = to;
                        break;
                    }
                }
                prev = cur;
                cur = nxt;
            }

            vector<int> take = solvePathOrder(order);
            result.insert(result.end(), take.begin(), take.end());
        } else {
            vector<int> order;
            order.reserve(comp.size());
            int start = comp[0];
            int prev = -1;
            int cur = start;
            do {
                order.push_back(cur);
                int a = adj[cur][0];
                int b = adj[cur][1];
                int nxt = (a == prev ? b : a);
                prev = cur;
                cur = nxt;
            } while (cur != start);

            vector<int> caseAPath(order.begin() + 1, order.end());
            vector<int> a = solvePathOrder(caseAPath);

            vector<int> b;
            b.push_back(order[0]);
            if (order.size() > 3) {
                vector<int> caseBPath(order.begin() + 2, order.end() - 1);
                vector<int> tail = solvePathOrder(caseBPath);
                b.insert(b.end(), tail.begin(), tail.end());
            }

            if (b.size() > a.size()) {
                result.insert(result.end(), b.begin(), b.end());
            } else {
                result.insert(result.end(), a.begin(), a.end());
            }
        }
    }

    return result;
}

static bool bipartiteColoring(const vector<vector<int>>& adj, vector<int>& part) {
    const int n = static_cast<int>(adj.size());
    part.assign(n, -1);
    queue<int> q;

    for (int i = 0; i < n; ++i) {
        if (part[i] != -1) continue;
        part[i] = 0;
        q.push(i);

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (int to : adj[v]) {
                if (part[to] == -1) {
                    part[to] = part[v] ^ 1;
                    q.push(to);
                } else if (part[to] == part[v]) {
                    return false;
                }
            }
        }
    }

    return true;
}

static vector<int> solveBipartiteMIS(const vector<vector<int>>& adj, const vector<int>& part) {
    const int n = static_cast<int>(adj.size());
    vector<int> leftId(n, -1), rightId(n, -1);
    int L = 0, R = 0;

    for (int i = 0; i < n; ++i) {
        if (part[i] == 0) leftId[i] = L++;
        else rightId[i] = R++;
    }

    vector<vector<int>> ladj(L);
    for (int u = 0; u < n; ++u) {
        if (part[u] != 0) continue;
        int lu = leftId[u];
        for (int v : adj[u]) {
            if (part[v] == 1) ladj[lu].push_back(rightId[v]);
        }
    }

    vector<int> matchL(L, -1), matchR(R, -1), dist(L, -1);

    auto bfs = [&]() -> bool {
        queue<int> q;
        bool foundFree = false;
        for (int u = 0; u < L; ++u) {
            if (matchL[u] == -1) {
                dist[u] = 0;
                q.push(u);
            } else {
                dist[u] = -1;
            }
        }

        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : ladj[u]) {
                int u2 = matchR[v];
                if (u2 == -1) {
                    foundFree = true;
                } else if (dist[u2] == -1) {
                    dist[u2] = dist[u] + 1;
                    q.push(u2);
                }
            }
        }
        return foundFree;
    };

    function<bool(int)> dfs = [&](int u) -> bool {
        for (int v : ladj[u]) {
            int u2 = matchR[v];
            if (u2 == -1 || (dist[u2] == dist[u] + 1 && dfs(u2))) {
                matchL[u] = v;
                matchR[v] = u;
                return true;
            }
        }
        dist[u] = -1;
        return false;
    };

    while (bfs()) {
        for (int u = 0; u < L; ++u) {
            if (matchL[u] == -1) dfs(u);
        }
    }

    vector<char> visL(L, 0), visR(R, 0);
    queue<int> q;
    for (int u = 0; u < L; ++u) {
        if (matchL[u] == -1) {
            visL[u] = 1;
            q.push(u);
        }
    }

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : ladj[u]) {
            if (matchL[u] == v) continue;
            if (!visR[v]) {
                visR[v] = 1;
                int u2 = matchR[v];
                if (u2 != -1 && !visL[u2]) {
                    visL[u2] = 1;
                    q.push(u2);
                }
            }
        }
    }

    vector<int> independent;
    independent.reserve(n);
    for (int i = 0; i < n; ++i) {
        bool inCover;
        if (part[i] == 0) {
            inCover = !visL[leftId[i]];
        } else {
            inCover = visR[rightId[i]];
        }
        if (!inCover) independent.push_back(i);
    }

    return independent;
}

class MaxCliqueOnComplement {
public:
    explicit MaxCliqueOnComplement(const vector<vector<int>>& originalAdj)
        : n_(static_cast<int>(originalAdj.size())), words_((n_ + 63) / 64), compDeg_(n_, 0) {

        vector<vector<uint64_t>> originalBits(n_, vector<uint64_t>(words_, 0ULL));
        compBits_.assign(n_, vector<uint64_t>(words_, 0ULL));

        for (int i = 0; i < n_; ++i) {
            for (int to : originalAdj[i]) {
                originalBits[i][to >> 6] |= (1ULL << (to & 63));
            }
        }

        uint64_t lastMask = ~0ULL;
        if ((n_ & 63) != 0) {
            lastMask = (1ULL << (n_ & 63)) - 1ULL;
        }

        for (int i = 0; i < n_; ++i) {
            for (int w = 0; w < words_; ++w) {
                compBits_[i][w] = ~originalBits[i][w];
            }
            compBits_[i][i >> 6] &= ~(1ULL << (i & 63));
            compBits_[i][words_ - 1] &= lastMask;

            int deg = 0;
            for (int w = 0; w < words_; ++w) deg += popcount64(compBits_[i][w]);
            compDeg_[i] = deg;
        }
    }

    vector<int> solve(const vector<int>& initialClique) {
        best_ = initialClique;

        vector<int> P(n_);
        iota(P.begin(), P.end(), 0);
        sort(P.begin(), P.end(), [&](int a, int b) {
            if (compDeg_[a] != compDeg_[b]) return compDeg_[a] > compDeg_[b];
            return a < b;
        });

        vector<int> cur;
        expand(cur, P);
        return best_;
    }

private:
    int n_;
    int words_;
    vector<vector<uint64_t>> compBits_;
    vector<int> compDeg_;
    vector<int> best_;

    inline bool compAdjacent(int a, int b) const {
        return ((compBits_[a][b >> 6] >> (b & 63)) & 1ULL) != 0ULL;
    }

    void colorSort(const vector<int>& P, vector<int>& order, vector<int>& bound) {
        vector<int> verts = P;
        sort(verts.begin(), verts.end(), [&](int a, int b) {
            if (compDeg_[a] != compDeg_[b]) return compDeg_[a] > compDeg_[b];
            return a < b;
        });

        const int m = static_cast<int>(verts.size());
        vector<int> colorOf(m, 0);
        vector<int> mark(m + 1, 0);
        int maxColor = 0;
        int stamp = 1;

        for (int i = 0; i < m; ++i) {
            int v = verts[i];
            if (stamp == numeric_limits<int>::max()) {
                fill(mark.begin(), mark.end(), 0);
                stamp = 1;
            }

            for (int j = 0; j < i; ++j) {
                int u = verts[j];
                if (compAdjacent(v, u)) {
                    mark[colorOf[j]] = stamp;
                }
            }

            int c = 1;
            while (c <= maxColor && mark[c] == stamp) ++c;
            if (c > maxColor) maxColor = c;
            colorOf[i] = c;
            ++stamp;
        }

        vector<vector<int>> buckets(maxColor + 1);
        for (int i = 0; i < m; ++i) {
            buckets[colorOf[i]].push_back(verts[i]);
        }

        order.clear();
        bound.clear();
        order.reserve(m);
        bound.reserve(m);
        for (int c = 1; c <= maxColor; ++c) {
            for (int v : buckets[c]) {
                order.push_back(v);
                bound.push_back(c);
            }
        }
    }

    void expand(vector<int>& cur, const vector<int>& P) {
        if (P.empty()) {
            if (cur.size() > best_.size()) best_ = cur;
            return;
        }
        if (cur.size() + P.size() <= best_.size()) return;

        vector<int> order;
        vector<int> bound;
        colorSort(P, order, bound);

        for (int i = static_cast<int>(order.size()) - 1; i >= 0; --i) {
            if (cur.size() + static_cast<size_t>(bound[i]) <= best_.size()) return;

            int v = order[i];
            cur.push_back(v);

            vector<int> P2;
            P2.reserve(i);
            for (int j = 0; j < i; ++j) {
                int u = order[j];
                if (compAdjacent(v, u)) P2.push_back(u);
            }

            if (P2.empty()) {
                if (cur.size() > best_.size()) best_ = cur;
            } else {
                expand(cur, P2);
            }

            cur.pop_back();
        }
    }
};

static vector<int> greedyIndependentLowerBound(const vector<vector<int>>& adj) {
    const int n = static_cast<int>(adj.size());
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (adj[a].size() != adj[b].size()) return adj[a].size() < adj[b].size();
        return a < b;
    });

    vector<char> blocked(n, 0);
    vector<int> indep;
    indep.reserve(n);

    for (int v : order) {
        if (blocked[v]) continue;
        indep.push_back(v);
        blocked[v] = 1;
        for (int to : adj[v]) blocked[to] = 1;
    }

    return indep;
}

static vector<int> solveGeneralComponent(const vector<int>& compGlobal,
                                         const vector<vector<int>>& g,
                                         vector<int>& localOfGlobal) {
    const int s = static_cast<int>(compGlobal.size());
    vector<vector<int>> adj(s);

    for (int i = 0; i < s; ++i) {
        localOfGlobal[compGlobal[i]] = i;
    }

    for (int i = 0; i < s; ++i) {
        int gv = compGlobal[i];
        for (int to : g[gv]) {
            int j = localOfGlobal[to];
            if (j != -1) adj[i].push_back(j);
        }
    }

    int maxDeg = 0;
    for (int i = 0; i < s; ++i) {
        int d = static_cast<int>(adj[i].size());
        if (d > maxDeg) maxDeg = d;
    }

    vector<int> localAns;
    if (maxDeg <= 2) {
        localAns = solveMaxDegTwo(adj);
    } else {
        vector<int> part;
        if (bipartiteColoring(adj, part)) {
            localAns = solveBipartiteMIS(adj, part);
        } else {
            vector<int> initial = greedyIndependentLowerBound(adj);
            MaxCliqueOnComplement solver(adj);
            localAns = solver.solve(initial);
        }
    }

    vector<int> globalAns;
    globalAns.reserve(localAns.size());
    for (int v : localAns) {
        globalAns.push_back(compGlobal[v]);
    }

    for (int i = 0; i < s; ++i) {
        localOfGlobal[compGlobal[i]] = -1;
    }

    return globalAns;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    long long m;
    if (!(cin >> n >> m)) {
        return 0;
    }

    vector<vector<int>> g(n);
    vector<char> forbidden(n, 0);

    for (long long i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            forbidden[u] = 1;
            continue;
        }
        g[u].push_back(v);
        g[v].push_back(u);
    }

    for (int i = 0; i < n; ++i) {
        auto& vec = g[i];
        sort(vec.begin(), vec.end());
        vec.erase(unique(vec.begin(), vec.end()), vec.end());
    }

    vector<char> seen(n, 0);
    vector<int> localOfGlobal(n, -1);
    vector<int> answer;

    for (int s = 0; s < n; ++s) {
        if (forbidden[s] || seen[s]) continue;

        vector<int> comp;
        queue<int> q;
        q.push(s);
        seen[s] = 1;

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            comp.push_back(v);
            for (int to : g[v]) {
                if (forbidden[to] || seen[to]) continue;
                seen[to] = 1;
                q.push(to);
            }
        }

        vector<int> partAns = solveGeneralComponent(comp, g, localOfGlobal);
        answer.insert(answer.end(), partAns.begin(), partAns.end());
    }

    sort(answer.begin(), answer.end());

    cout << answer.size() << '\n';
    for (size_t i = 0; i < answer.size(); ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';

    return 0;
}
