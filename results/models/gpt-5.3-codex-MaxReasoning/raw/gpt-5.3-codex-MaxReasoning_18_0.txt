#include <algorithm>
#include <array>
#include <cstdio>
#include <cstdint>
#include <fstream>
#include <iostream>
#include <limits>
#include <numeric>
#include <string>
#include <unordered_map>
#include <utility>
#include <vector>

using namespace std;

struct EdgePair {
    int u;
    int v;
};

struct WeightedEdge {
    int u;
    int v;
    long long w;
};

static uint64_t pack_key(int a, int b) {
    if (a > b) {
        std::swap(a, b);
    }
    return (static_cast<uint64_t>(static_cast<uint32_t>(a)) << 32) |
           static_cast<uint32_t>(b);
}

static pair<long long, vector<int>> stoer_wagner_dense(int n, const vector<EdgePair>& edges) {
    vector<vector<long long>> w(n, vector<long long>(n, 0));
    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        ++w[e.u][e.v];
        ++w[e.v][e.u];
    }

    vector<int> verts(n);
    iota(verts.begin(), verts.end(), 0);

    vector<vector<int>> groups(n);
    for (int i = 0; i < n; ++i) {
        groups[i].push_back(i);
    }

    long long best = numeric_limits<long long>::max();
    vector<int> best_side;

    int active_count = n;
    while (active_count > 1) {
        vector<long long> weight(n, 0);
        vector<char> added(n, 0);
        int prev = -1;

        for (int it = 0; it < active_count; ++it) {
            int sel = -1;
            for (int i = 0; i < active_count; ++i) {
                int v = verts[i];
                if (!added[v] && (sel == -1 || weight[v] > weight[sel])) {
                    sel = v;
                }
            }

            if (it == active_count - 1) {
                long long cut_value = weight[sel];
                if (cut_value < best) {
                    best = cut_value;
                    best_side = groups[sel];
                }

                if (prev != -1) {
                    for (int i = 0; i < active_count; ++i) {
                        int v = verts[i];
                        if (v == sel || v == prev) continue;
                        w[prev][v] += w[sel][v];
                        w[v][prev] = w[prev][v];
                    }

                    groups[prev].insert(groups[prev].end(), groups[sel].begin(), groups[sel].end());
                    groups[sel].clear();

                    auto pos_it = find(verts.begin(), verts.begin() + active_count, sel);
                    iter_swap(pos_it, verts.begin() + active_count - 1);
                    --active_count;
                }
                break;
            }

            added[sel] = 1;
            prev = sel;

            for (int i = 0; i < active_count; ++i) {
                int v = verts[i];
                if (!added[v]) {
                    weight[v] += w[sel][v];
                }
            }
        }
    }

    if (best_side.empty() && n > 0) {
        best_side.push_back(0);
        best = 0;
    }
    return {best, best_side};
}

static pair<long long, vector<int>> stoer_wagner_sparse(int n, const vector<WeightedEdge>& wedges) {
    vector<unordered_map<int, long long>> adj(n);
    adj.reserve(n);

    for (const auto& e : wedges) {
        if (e.u == e.v || e.w == 0) continue;
        adj[e.u][e.v] += e.w;
        adj[e.v][e.u] += e.w;
    }

    vector<char> alive(n, 1);
    int active_count = n;

    vector<vector<int>> groups(n);
    for (int i = 0; i < n; ++i) groups[i].push_back(i);

    long long best = numeric_limits<long long>::max();
    vector<int> best_side;

    while (active_count > 1) {
        vector<long long> weight(n, 0);
        vector<char> added(n, 0);
        int prev = -1;

        for (int it = 0; it < active_count; ++it) {
            int sel = -1;
            for (int v = 0; v < n; ++v) {
                if (!alive[v] || added[v]) continue;
                if (sel == -1 || weight[v] > weight[sel]) sel = v;
            }

            if (sel == -1) {
                active_count = 1;
                break;
            }

            added[sel] = 1;

            if (it == active_count - 1) {
                long long cut_value = weight[sel];
                if (cut_value < best) {
                    best = cut_value;
                    best_side = groups[sel];
                }

                if (prev != -1) {
                    for (const auto& kv : adj[sel]) {
                        int to = kv.first;
                        long long cap = kv.second;
                        if (!alive[to] || to == prev) continue;

                        adj[prev][to] += cap;
                        adj[to][prev] += cap;
                        adj[to].erase(sel);
                    }
                    adj[prev].erase(sel);
                    adj[sel].clear();

                    groups[prev].insert(groups[prev].end(), groups[sel].begin(), groups[sel].end());
                    groups[sel].clear();

                    alive[sel] = 0;
                    --active_count;
                }
                break;
            }

            prev = sel;
            for (const auto& kv : adj[sel]) {
                int to = kv.first;
                if (alive[to] && !added[to]) {
                    weight[to] += kv.second;
                }
            }
        }
    }

    if (best_side.empty() && n > 0) {
        best_side.push_back(0);
        best = 0;
    }
    return {best, best_side};
}

static vector<EdgePair> build_cut_from_side(const vector<EdgePair>& edges, const vector<int>& side, int n) {
    vector<char> in_side(n, 0);
    for (int v : side) {
        if (v >= 0 && v < n) in_side[v] = 1;
    }

    vector<EdgePair> cut;
    cut.reserve(edges.size() / 8 + 8);
    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        if (in_side[e.u] != in_side[e.v]) {
            cut.push_back(e);
        }
    }
    return cut;
}

static vector<EdgePair> cut_by_isolating_vertex_from_file(const string& path, int isolate_v) {
    vector<EdgePair> cut;
    ifstream fin(path, ios::binary);
    if (!fin) return cut;

    EdgePair e{};
    while (fin.read(reinterpret_cast<char*>(&e), sizeof(e))) {
        if (e.u == e.v) continue;
        if (e.u == isolate_v || e.v == isolate_v) {
            cut.push_back(e);
        }
    }
    return cut;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n = 0;
    uint64_t m = 0;
    if (!(cin >> n >> m)) {
        return 0;
    }

    if (n <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    const uint64_t MAX_IN_MEMORY_EDGES = 20000000ULL;
    const int DENSE_N_LIMIT = 900;
    const uint64_t MAX_EXACT_EDGES = 3000000ULL;
    const int MAX_EXACT_N = 7000;

    const bool spill_to_disk = (m > MAX_IN_MEMORY_EDGES);

    vector<int> degree(n, 0);
    int min_deg_vertex = 0;

    vector<EdgePair> edges;
    edges.reserve(spill_to_disk ? 1024 : static_cast<size_t>(m));

    string temp_file = "edges_spill.bin";
    ofstream spill_out;
    if (spill_to_disk) {
        spill_out.open(temp_file, ios::binary | ios::trunc);
        if (!spill_out) {
            return 1;
        }
    }

    for (uint64_t i = 0; i < m; ++i) {
        int u = 0, v = 0;
        cin >> u >> v;
        EdgePair e{u, v};

        if (u >= 0 && u < n && v >= 0 && v < n && u != v) {
            ++degree[u];
            ++degree[v];
        }

        if (spill_to_disk) {
            spill_out.write(reinterpret_cast<const char*>(&e), sizeof(e));
        } else {
            edges.push_back(e);
        }
    }

    if (spill_to_disk) {
        spill_out.close();
    }

    for (int i = 1; i < n; ++i) {
        if (degree[i] < degree[min_deg_vertex]) {
            min_deg_vertex = i;
        }
    }

    vector<EdgePair> answer_cut;

    const bool use_fallback = spill_to_disk || n > MAX_EXACT_N || m > MAX_EXACT_EDGES;

    if (use_fallback) {
        if (spill_to_disk) {
            answer_cut = cut_by_isolating_vertex_from_file(temp_file, min_deg_vertex);
            std::remove(temp_file.c_str());
        } else {
            answer_cut.reserve(static_cast<size_t>(degree[min_deg_vertex]));
            for (const auto& e : edges) {
                if (e.u == e.v) continue;
                if (e.u == min_deg_vertex || e.v == min_deg_vertex) {
                    answer_cut.push_back(e);
                }
            }
        }
    } else {
        vector<int> best_side;

        if (n <= DENSE_N_LIMIT) {
            auto [best_value, side] = stoer_wagner_dense(n, edges);
            (void)best_value;
            best_side = std::move(side);
        } else {
            unordered_map<uint64_t, long long> cap;
            cap.reserve(edges.size() * 2 + 1);
            for (const auto& e : edges) {
                if (e.u == e.v) continue;
                ++cap[pack_key(e.u, e.v)];
            }

            vector<WeightedEdge> wedges;
            wedges.reserve(cap.size());
            for (const auto& kv : cap) {
                int u = static_cast<int>(kv.first >> 32);
                int v = static_cast<int>(kv.first & 0xffffffffu);
                wedges.push_back({u, v, kv.second});
            }

            auto [best_value, side] = stoer_wagner_sparse(n, wedges);
            (void)best_value;
            best_side = std::move(side);
        }

        answer_cut = build_cut_from_side(edges, best_side, n);

        if (answer_cut.empty()) {
            answer_cut.reserve(static_cast<size_t>(degree[min_deg_vertex]));
            for (const auto& e : edges) {
                if (e.u == e.v) continue;
                if (e.u == min_deg_vertex || e.v == min_deg_vertex) {
                    answer_cut.push_back(e);
                }
            }
        }
    }

    cout << answer_cut.size() << '\n';
    for (const auto& e : answer_cut) {
        cout << e.u << ' ' << e.v << '\n';
    }

    return 0;
}
