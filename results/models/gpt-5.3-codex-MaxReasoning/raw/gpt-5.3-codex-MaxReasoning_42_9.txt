OpCapability Shader
OpCapability Geometry
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0_025 = OpConstant %float 0.025
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%_ptr_Input_arr_gl_PerVertex_in_4 = OpTypePointer Input %arr_gl_PerVertex_in_4
%_ptr_Input_v4float = OpTypePointer Input %v4float
%arr_v3float_4 = OpTypeArray %v3float %uint_4
%_ptr_Input_arr_v3float_4 = OpTypePointer Input %arr_v3float_4
%_ptr_Input_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in_4 Input
%inWorldPos = OpVariable %_ptr_Input_arr_v3float_4 Input
%inNormal = OpVariable %_ptr_Input_arr_v3float_4 Input
%inColor = OpVariable %_ptr_Input_arr_v3float_4 Input
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %_ptr_Output_v3float Output
%main = OpFunction %void None %fn_void
%entry = OpLabel
%p0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr
%col1_ptr = OpAccessChain %_ptr_Input_v3float %inColor %int_1
%col2_ptr = OpAccessChain %_ptr_Input_v3float %inColor %int_2
%col1 = OpLoad %v3float %col1_ptr
%col2 = OpLoad %v3float %col2_ptr
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1
%t0_dir = OpFSub %v2float %p2_xy %p0_xy
%t1_dir = OpFSub %v2float %p3_xy %p1_xy
%t0 = OpExtInst %v2float %ext Normalize %t0_dir
%t1 = OpExtInst %v2float %ext Normalize %t1_dir
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t0_y = OpFNegate %float %t0_y
%neg_t1_y = OpFNegate %float %t1_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x
%off0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
%off1 = OpVectorTimesScalar %v2float %perp1 %float_0_025
%start_plus_xy = OpFAdd %v2float %p1_xy %off0
%start_minus_xy = OpFSub %v2float %p1_xy %off0
%end_plus_xy = OpFAdd %v2float %p2_xy %off1
%end_minus_xy = OpFSub %v2float %p2_xy %off1
%start_plus = OpVectorShuffle %v4float %p1 %start_plus_xy 4 5 2 3
%start_minus = OpVectorShuffle %v4float %p1 %start_minus_xy 4 5 2 3
%end_plus = OpVectorShuffle %v4float %p2 %end_plus_xy 4 5 2 3
%end_minus = OpVectorShuffle %v4float %p2 %end_minus_xy 4 5 2 3
%out_pos_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr %start_plus
OpStore %outColor %col1
OpEmitVertex
OpStore %out_pos_ptr %start_minus
OpStore %outColor %col1
OpEmitVertex
OpStore %out_pos_ptr %end_plus
OpStore %outColor %col2
OpEmitVertex
OpStore %out_pos_ptr %end_minus
OpStore %outColor %col2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
