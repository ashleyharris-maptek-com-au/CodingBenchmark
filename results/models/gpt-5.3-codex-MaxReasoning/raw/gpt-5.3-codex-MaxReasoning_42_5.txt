OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpSource GLSL 450

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_75 = OpConstant %float 0.75
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_gl_PerVertex_in_2 = OpTypePointer Input %arr_gl_PerVertex_in_2
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_2 Input

%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3float_2 = OpTypePointer Input %arr_v3float_2
%ptr_Input_v3float = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_Input_arr_v3float_2 Input
%normal = OpVariable %ptr_Input_arr_v3float_2 Input
%color = OpVariable %ptr_Input_arr_v3float_2 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v3float %worldPos %int_0
%p1_ptr = OpAccessChain %ptr_Input_v3float %worldPos %int_1
%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1

%p0 = OpLoad %v3float %p0_ptr
%p1 = OpLoad %v3float %p1_ptr
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy

%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %glsl450 Normalize %perp_raw

%shaft_half = OpVectorTimesScalar %v2float %perp %float_0_015
%arrow_half = OpVectorTimesScalar %v2float %perp %float_0_05

%dir_75 = OpVectorTimesScalar %v2float %dir %float_0_75
%mid_xy = OpFAdd %v2float %p0_xy %dir_75

%p0_z = OpCompositeExtract %float %p0 2
%p1_z = OpCompositeExtract %float %p1 2
%dz = OpFSub %float %p1_z %p0_z
%dz_75 = OpFMul %float %dz %float_0_75
%mid_z = OpFAdd %float %p0_z %dz_75

%gl_out_pos = OpAccessChain %ptr_Output_v4float %gl_out %int_0

%shaft_v0_xy = OpFAdd %v2float %p0_xy %shaft_half
%shaft_v0_x = OpCompositeExtract %float %shaft_v0_xy 0
%shaft_v0_y = OpCompositeExtract %float %shaft_v0_xy 1
%shaft_v0 = OpCompositeConstruct %v4float %shaft_v0_x %shaft_v0_y %p0_z %float_1
OpStore %gl_out_pos %shaft_v0
OpStore %outColor %c0
OpEmitVertex

%shaft_v1_xy = OpFSub %v2float %p0_xy %shaft_half
%shaft_v1_x = OpCompositeExtract %float %shaft_v1_xy 0
%shaft_v1_y = OpCompositeExtract %float %shaft_v1_xy 1
%shaft_v1 = OpCompositeConstruct %v4float %shaft_v1_x %shaft_v1_y %p0_z %float_1
OpStore %gl_out_pos %shaft_v1
OpStore %outColor %c0
OpEmitVertex

%shaft_v2_xy = OpFAdd %v2float %mid_xy %shaft_half
%shaft_v2_x = OpCompositeExtract %float %shaft_v2_xy 0
%shaft_v2_y = OpCompositeExtract %float %shaft_v2_xy 1
%shaft_v2 = OpCompositeConstruct %v4float %shaft_v2_x %shaft_v2_y %mid_z %float_1
OpStore %gl_out_pos %shaft_v2
OpStore %outColor %c0
OpEmitVertex

%shaft_v3_xy = OpFSub %v2float %mid_xy %shaft_half
%shaft_v3_x = OpCompositeExtract %float %shaft_v3_xy 0
%shaft_v3_y = OpCompositeExtract %float %shaft_v3_xy 1
%shaft_v3 = OpCompositeConstruct %v4float %shaft_v3_x %shaft_v3_y %mid_z %float_1
OpStore %gl_out_pos %shaft_v3
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

%arrow_v0_xy = OpFAdd %v2float %mid_xy %arrow_half
%arrow_v0_x = OpCompositeExtract %float %arrow_v0_xy 0
%arrow_v0_y = OpCompositeExtract %float %arrow_v0_xy 1
%arrow_v0 = OpCompositeConstruct %v4float %arrow_v0_x %arrow_v0_y %mid_z %float_1
OpStore %gl_out_pos %arrow_v0
OpStore %outColor %c1
OpEmitVertex

%arrow_v1_xy = OpFSub %v2float %mid_xy %arrow_half
%arrow_v1_x = OpCompositeExtract %float %arrow_v1_xy 0
%arrow_v1_y = OpCompositeExtract %float %arrow_v1_xy 1
%arrow_v1 = OpCompositeConstruct %v4float %arrow_v1_x %arrow_v1_y %mid_z %float_1
OpStore %gl_out_pos %arrow_v1
OpStore %outColor %c1
OpEmitVertex

%tip_x = OpCompositeExtract %float %p1_xy 0
%tip_y = OpCompositeExtract %float %p1_xy 1
%tip = OpCompositeConstruct %v4float %tip_x %tip_y %p1_z %float_1
OpStore %gl_out_pos %tip
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
