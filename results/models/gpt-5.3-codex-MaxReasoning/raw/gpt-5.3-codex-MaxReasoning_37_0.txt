using System;
using System.Collections.Generic;
using System.IO;

public class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n < 0) return;
        int m = fs.NextInt();

        var outAdj = new List<int>[n];
        var inAdj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            outAdj[i] = new List<int>();
            inAdj[i] = new List<int>();
        }

        var hasSelfLoop = new bool[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n) continue;

            outAdj[u].Add(v);
            inAdj[v].Add(u);
            if (u == v) hasSelfLoop[u] = true;
        }

        var active = new bool[n];
        Array.Fill(active, true);
        var answer = new List<int>();

        ReduceGraph(active, outAdj, inAdj, hasSelfLoop, answer);

        if (HasAnyActive(active))
        {
            var cyclicSccs = GetCyclicSccs(active, outAdj, inAdj, hasSelfLoop);
            for (int i = 0; i < cyclicSccs.Count; i++)
            {
                var solver = new ComponentSolver(cyclicSccs[i], outAdj, inAdj);
                var part = solver.Solve();
                answer.AddRange(part);
            }
        }

        var seen = new bool[n];
        var final = new List<int>(answer.Count);
        for (int i = 0; i < answer.Count; i++)
        {
            int v = answer[i];
            if ((uint)v < (uint)n && !seen[v])
            {
                seen[v] = true;
                final.Add(v);
            }
        }

        final.Sort();

        var sw = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        sw.WriteLine(final.Count);
        if (final.Count > 0)
            sw.WriteLine(string.Join(" ", final));
        else
            sw.WriteLine();
        sw.Flush();
    }

    static bool HasAnyActive(bool[] active)
    {
        for (int i = 0; i < active.Length; i++)
            if (active[i]) return true;
        return false;
    }

    static void ReduceGraph(bool[] active, List<int>[] outAdj, List<int>[] inAdj, bool[] hasSelfLoop, List<int> forcedOut)
    {
        int n = active.Length;
        var indeg = new int[n];
        var outdeg = new int[n];

        for (int v = 0; v < n; v++)
        {
            if (!active[v]) continue;
            var outs = outAdj[v];
            for (int i = 0; i < outs.Count; i++)
            {
                int to = outs[i];
                if (!active[to]) continue;
                outdeg[v]++;
                indeg[to]++;
            }
        }

        var q = new Queue<int>();
        var mark = new byte[n]; // 0 none, 1 peel, 2 forced(self-loop)

        for (int v = 0; v < n; v++)
        {
            if (!active[v]) continue;
            if (hasSelfLoop[v])
            {
                mark[v] = 2;
                q.Enqueue(v);
            }
        }

        for (int v = 0; v < n; v++)
        {
            if (!active[v] || mark[v] != 0) continue;
            if (indeg[v] == 0 || outdeg[v] == 0)
            {
                mark[v] = 1;
                q.Enqueue(v);
            }
        }

        while (q.Count > 0)
        {
            int v = q.Dequeue();
            if (!active[v]) continue;

            bool forced = mark[v] == 2;
            active[v] = false;
            if (forced) forcedOut.Add(v);

            var outs = outAdj[v];
            for (int i = 0; i < outs.Count; i++)
            {
                int to = outs[i];
                if (!active[to]) continue;
                indeg[to]--;
                if (mark[to] == 0 && (indeg[to] == 0 || outdeg[to] == 0))
                {
                    mark[to] = 1;
                    q.Enqueue(to);
                }
            }

            var ins = inAdj[v];
            for (int i = 0; i < ins.Count; i++)
            {
                int from = ins[i];
                if (!active[from]) continue;
                outdeg[from]--;
                if (mark[from] == 0 && (indeg[from] == 0 || outdeg[from] == 0))
                {
                    mark[from] = 1;
                    q.Enqueue(from);
                }
            }
        }
    }

    static List<int[]> GetCyclicSccs(bool[] active, List<int>[] outAdj, List<int>[] inAdj, bool[] hasSelfLoop)
    {
        int n = active.Length;
        var visited = new bool[n];
        var order = new List<int>(n);
        var nextIdx = new int[n];

        for (int s = 0; s < n; s++)
        {
            if (!active[s] || visited[s]) continue;

            var st = new Stack<int>();
            st.Push(s);
            visited[s] = true;
            nextIdx[s] = 0;

            while (st.Count > 0)
            {
                int v = st.Peek();
                var outs = outAdj[v];
                bool advanced = false;

                while (nextIdx[v] < outs.Count)
                {
                    int to = outs[nextIdx[v]++];
                    if (!active[to] || visited[to]) continue;
                    visited[to] = true;
                    nextIdx[to] = 0;
                    st.Push(to);
                    advanced = true;
                    break;
                }

                if (!advanced)
                {
                    st.Pop();
                    order.Add(v);
                }
            }
        }

        var compVisited = new bool[n];
        var cyclic = new List<int[]>();

        for (int idx = order.Count - 1; idx >= 0; idx--)
        {
            int s = order[idx];
            if (!active[s] || compVisited[s]) continue;

            var comp = new List<int>();
            var st = new Stack<int>();
            st.Push(s);
            compVisited[s] = true;

            while (st.Count > 0)
            {
                int v = st.Pop();
                comp.Add(v);
                var ins = inAdj[v];
                for (int i = 0; i < ins.Count; i++)
                {
                    int u = ins[i];
                    if (!active[u] || compVisited[u]) continue;
                    compVisited[u] = true;
                    st.Push(u);
                }
            }

            bool isCyclic;
            if (comp.Count > 1)
                isCyclic = true;
            else
                isCyclic = hasSelfLoop[comp[0]];

            if (isCyclic)
                cyclic.Add(comp.ToArray());
        }

        return cyclic;
    }

    sealed class ComponentSolver
    {
        private readonly int n;
        private readonly int[] localToGlobal;
        private readonly int[][] outAdj;
        private readonly int[][] inAdj;
        private readonly bool[] hasSelfLoop;

        private readonly List<int> current = new List<int>();
        private List<int> best = new List<int>();
        private int bestCount;

        public ComponentSolver(int[] globalVertices, List<int>[] globalOut, List<int>[] globalIn)
        {
            n = globalVertices.Length;
            localToGlobal = new int[n];
            Array.Copy(globalVertices, localToGlobal, n);

            var globalToLocal = new Dictionary<int, int>(n);
            for (int i = 0; i < n; i++) globalToLocal[globalVertices[i]] = i;

            var outTmp = new List<int>[n];
            var inTmp = new List<int>[n];
            hasSelfLoop = new bool[n];

            for (int i = 0; i < n; i++)
            {
                outTmp[i] = new List<int>();
                inTmp[i] = new List<int>();
            }

            for (int li = 0; li < n; li++)
            {
                int gv = localToGlobal[li];
                var outs = globalOut[gv];
                for (int k = 0; k < outs.Count; k++)
                {
                    int gto = outs[k];
                    if (!globalToLocal.TryGetValue(gto, out int lto)) continue;

                    outTmp[li].Add(lto);
                    inTmp[lto].Add(li);
                    if (li == lto) hasSelfLoop[li] = true;
                }
            }

            outAdj = new int[n][];
            inAdj = new int[n][];
            for (int i = 0; i < n; i++)
            {
                outAdj[i] = outTmp[i].ToArray();
                inAdj[i] = inTmp[i].ToArray();
            }
        }

        public List<int> Solve()
        {
            var active = new bool[n];
            Array.Fill(active, true);

            var ub = GreedyCompletion(active);
            best = new List<int>(ub);
            bestCount = ub.Count;

            Search(active, 0, 0);

            var result = new List<int>(bestCount);
            for (int i = 0; i < best.Count; i++)
                result.Add(localToGlobal[best[i]]);
            return result;
        }

        private void Search(bool[] active, int currentCount, int depth)
        {
            if (currentCount >= bestCount) return;

            var work = (bool[])active.Clone();
            int baseLen = current.Count;

            int forced = ApplyReduction(work, current);
            int newCount = currentCount + forced;

            if (newCount >= bestCount)
            {
                TrimCurrent(baseLen);
                return;
            }

            var cycle = FindCycle(work);
            if (cycle == null)
            {
                if (newCount < bestCount)
                {
                    bestCount = newCount;
                    best = new List<int>(current);
                }
                TrimCurrent(baseLen);
                return;
            }

            if (depth <= 2 && newCount + 1 < bestCount)
            {
                var completion = GreedyCompletion(work);
                int candidate = newCount + completion.Count;
                if (candidate < bestCount)
                {
                    bestCount = candidate;
                    best = new List<int>(current);
                    best.AddRange(completion);
                }
            }

            int cutoff = bestCount - newCount - 1;
            if (cutoff <= 0)
            {
                TrimCurrent(baseLen);
                return;
            }

            int lb = LowerBoundByDisjointCycles(work, cutoff);
            if (newCount + lb >= bestCount)
            {
                TrimCurrent(baseLen);
                return;
            }

            cycle.Sort((a, b) => ScoreVertex(work, b).CompareTo(ScoreVertex(work, a)));

            for (int i = 0; i < cycle.Count; i++)
            {
                int v = cycle[i];
                if (!work[v]) continue;

                var next = (bool[])work.Clone();
                next[v] = false;

                current.Add(v);
                Search(next, newCount + 1, depth + 1);
                current.RemoveAt(current.Count - 1);
            }

            TrimCurrent(baseLen);
        }

        private void TrimCurrent(int len)
        {
            if (current.Count > len)
                current.RemoveRange(len, current.Count - len);
        }

        private int ApplyReduction(bool[] active, List<int> forcedInto)
        {
            var indeg = new int[n];
            var outdeg = new int[n];

            for (int v = 0; v < n; v++)
            {
                if (!active[v]) continue;
                var outs = outAdj[v];
                for (int i = 0; i < outs.Length; i++)
                {
                    int to = outs[i];
                    if (!active[to]) continue;
                    outdeg[v]++;
                    indeg[to]++;
                }
            }

            var q = new Queue<int>();
            var mark = new byte[n]; // 0 none, 1 peel, 2 forced

            for (int v = 0; v < n; v++)
            {
                if (!active[v]) continue;
                if (hasSelfLoop[v])
                {
                    mark[v] = 2;
                    q.Enqueue(v);
                }
            }

            for (int v = 0; v < n; v++)
            {
                if (!active[v] || mark[v] != 0) continue;
                if (indeg[v] == 0 || outdeg[v] == 0)
                {
                    mark[v] = 1;
                    q.Enqueue(v);
                }
            }

            int added = 0;

            while (q.Count > 0)
            {
                int v = q.Dequeue();
                if (!active[v]) continue;

                bool forced = mark[v] == 2;
                active[v] = false;
                if (forced)
                {
                    forcedInto.Add(v);
                    added++;
                }

                var outs = outAdj[v];
                for (int i = 0; i < outs.Length; i++)
                {
                    int to = outs[i];
                    if (!active[to]) continue;
                    indeg[to]--;
                    if (mark[to] == 0 && (indeg[to] == 0 || outdeg[to] == 0))
                    {
                        mark[to] = 1;
                        q.Enqueue(to);
                    }
                }

                var ins = inAdj[v];
                for (int i = 0; i < ins.Length; i++)
                {
                    int from = ins[i];
                    if (!active[from]) continue;
                    outdeg[from]--;
                    if (mark[from] == 0 && (indeg[from] == 0 || outdeg[from] == 0))
                    {
                        mark[from] = 1;
                        q.Enqueue(from);
                    }
                }
            }

            return added;
        }

        private List<int> FindCycle(bool[] active)
        {
            var color = new byte[n]; // 0 white, 1 gray, 2 black
            var parent = new int[n];
            var nextIdx = new int[n];
            Array.Fill(parent, -1);

            var st = new Stack<int>();

            for (int s = 0; s < n; s++)
            {
                if (!active[s] || color[s] != 0) continue;

                st.Push(s);
                color[s] = 1;
                nextIdx[s] = 0;
                parent[s] = -1;

                while (st.Count > 0)
                {
                    int v = st.Peek();
                    var outs = outAdj[v];
                    bool advanced = false;

                    while (nextIdx[v] < outs.Length)
                    {
                        int to = outs[nextIdx[v]++];
                        if (!active[to]) continue;

                        if (color[to] == 0)
                        {
                            color[to] = 1;
                            parent[to] = v;
                            nextIdx[to] = 0;
                            st.Push(to);
                            advanced = true;
                            break;
                        }

                        if (color[to] == 1)
                        {
                            var cycle = new List<int>();
                            cycle.Add(to);
                            int x = v;
                            while (x != to && x != -1)
                            {
                                cycle.Add(x);
                                x = parent[x];
                            }
                            return cycle;
                        }
                    }

                    if (!advanced)
                    {
                        color[v] = 2;
                        st.Pop();
                    }
                }
            }

            return null;
        }

        private int LowerBoundByDisjointCycles(bool[] active, int cutoff)
        {
            var temp = (bool[])active.Clone();
            int lb = 0;

            while (lb < cutoff)
            {
                var cycle = FindCycle(temp);
                if (cycle == null) break;
                lb++;
                for (int i = 0; i < cycle.Count; i++)
                    temp[cycle[i]] = false;
            }

            return lb;
        }

        private List<int> GreedyCompletion(bool[] active)
        {
            var temp = (bool[])active.Clone();
            var pick = new List<int>();

            while (true)
            {
                ApplyReduction(temp, pick);
                var cycle = FindCycle(temp);
                if (cycle == null) break;

                int bestV = cycle[0];
                int bestScore = ScoreVertex(temp, bestV);

                for (int i = 1; i < cycle.Count; i++)
                {
                    int v = cycle[i];
                    int sc = ScoreVertex(temp, v);
                    if (sc > bestScore)
                    {
                        bestScore = sc;
                        bestV = v;
                    }
                }

                temp[bestV] = false;
                pick.Add(bestV);
            }

            return pick;
        }

        private int ScoreVertex(bool[] active, int v)
        {
            int inD = 0;
            int outD = 0;

            var ins = inAdj[v];
            for (int i = 0; i < ins.Length; i++)
                if (active[ins[i]]) inD++;

            var outs = outAdj[v];
            for (int i = 0; i < outs.Length; i++)
                if (active[outs[i]]) outD++;

            return inD * outD + inD + outD;
        }
    }

    sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len;
        private int ptr;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return -1;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }
    }
}
