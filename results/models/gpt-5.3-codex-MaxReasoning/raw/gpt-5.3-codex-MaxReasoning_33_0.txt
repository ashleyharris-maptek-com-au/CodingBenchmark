using System;
using System.Diagnostics;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var timer = Stopwatch.StartNew();
        var scanner = new FastScanner(Console.OpenStandardInput());

        if (!scanner.TryNextInt(out int n) || n <= 0)
        {
            return;
        }

        int nn = n * n;
        var flow = new int[nn];
        var dist = new int[nn];

        var flowOut = new long[n];
        var flowIn = new long[n];
        var distOut = new long[n];
        var distIn = new long[n];

        for (int i = 0; i < n; i++)
        {
            int row = i * n;
            for (int j = 0; j < n; j++)
            {
                int v = scanner.NextInt();
                flow[row + j] = v;
                flowOut[i] += v;
                flowIn[j] += v;
            }
        }

        for (int i = 0; i < n; i++)
        {
            int row = i * n;
            for (int j = 0; j < n; j++)
            {
                int v = scanner.NextInt();
                dist[row + j] = v;
                distOut[i] += v;
                distIn[j] += v;
            }
        }

        var facilityScore = new long[n];
        var locationScore = new long[n];

        for (int i = 0; i < n; i++)
        {
            facilityScore[i] = flowOut[i] + flowIn[i];
            locationScore[i] = distOut[i] + distIn[i];
        }

        int[] facilitiesByImportance = OrderByScore(facilityScore, descending: true);
        int[] locationsByCentrality = OrderByScore(locationScore, descending: false);

        var p = new int[n];
        for (int rank = 0; rank < n; rank++)
        {
            p[facilitiesByImportance[rank]] = locationsByCentrality[rank];
        }

        long cost = ComputeCost(n, flow, dist, p);

        // Keep search within ~5 minutes total wall time.
        const long totalBudgetMs = 290000;
        LocalSearch(n, flow, dist, p, ref cost, facilitiesByImportance, timer, totalBudgetMs);

        // Recompute exactly for robust output.
        cost = ComputeCost(n, flow, dist, p);

        var sb = new StringBuilder(Math.Max(32, n * 6));
        sb.Append(cost).Append('\n');
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(p[i]);
        }
        sb.Append('\n');

        Console.Write(sb.ToString());
    }

    private static int[] OrderByScore(long[] score, bool descending)
    {
        int n = score.Length;
        var keys = new long[n];
        var idx = new int[n];
        for (int i = 0; i < n; i++)
        {
            keys[i] = score[i];
            idx[i] = i;
        }

        Array.Sort(keys, idx);
        if (descending)
        {
            Array.Reverse(idx);
        }

        return idx;
    }

    private static void LocalSearch(
        int n,
        int[] flow,
        int[] dist,
        int[] p,
        ref long cost,
        int[] facilitiesByImportance,
        Stopwatch timer,
        long deadlineMs)
    {
        if (n < 2 || timer.ElapsedMilliseconds >= deadlineMs)
        {
            return;
        }

        if (n <= 220)
        {
            while (timer.ElapsedMilliseconds < deadlineMs)
            {
                long bestDelta = 0;
                int bestA = -1;
                int bestB = -1;

                for (int a = 0; a < n - 1; a++)
                {
                    for (int b = a + 1; b < n; b++)
                    {
                        long delta = ComputeSwapDelta(n, flow, dist, p, a, b);
                        if (delta < bestDelta)
                        {
                            bestDelta = delta;
                            bestA = a;
                            bestB = b;
                        }
                    }
                }

                if (bestA < 0)
                {
                    break;
                }

                Swap(ref p[bestA], ref p[bestB]);
                cost += bestDelta;
            }

            return;
        }

        var rng = new XorShift64(0x9E3779B97F4A7C15UL ^ (ulong)n);

        int sampleSize;
        int iterations;
        if (n <= 600)
        {
            sampleSize = 14;
            iterations = n * 30;
        }
        else if (n <= 1200)
        {
            sampleSize = 10;
            iterations = n * 20;
        }
        else
        {
            sampleSize = 8;
            iterations = n * 12;
        }

        int topCount = Math.Max(2, n / 4);
        int stagnant = 0;

        for (int it = 0; it < iterations && timer.ElapsedMilliseconds < deadlineMs; it++)
        {
            int a;
            if ((it & 3) != 0)
            {
                a = facilitiesByImportance[rng.NextInt(topCount)];
            }
            else
            {
                a = rng.NextInt(n);
            }

            long bestDelta = 0;
            int bestB = -1;

            for (int s = 0; s < sampleSize; s++)
            {
                int b = rng.NextInt(n - 1);
                if (b >= a) b++;

                long delta = ComputeSwapDelta(n, flow, dist, p, a, b);
                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestB = b;
                }
            }

            if (bestB >= 0)
            {
                Swap(ref p[a], ref p[bestB]);
                cost += bestDelta;
                stagnant = 0;
            }
            else
            {
                stagnant++;

                // Mild perturbation when stuck, to escape poor local minima.
                if (stagnant > n && (it & 31) == 31)
                {
                    int x = rng.NextInt(n);
                    int y = rng.NextInt(n - 1);
                    if (y >= x) y++;

                    long kickDelta = ComputeSwapDelta(n, flow, dist, p, x, y);
                    if (kickDelta <= 0 || rng.NextInt(100) < 5)
                    {
                        Swap(ref p[x], ref p[y]);
                        cost += kickDelta;
                        stagnant = 0;
                    }
                }
            }
        }

        // Short polishing pass: first-improvement with random pairs.
        int polishTrials = Math.Min(10000, n * 5);
        for (int t = 0; t < polishTrials && timer.ElapsedMilliseconds < deadlineMs; t++)
        {
            int a = facilitiesByImportance[t % topCount];
            int b = rng.NextInt(n - 1);
            if (b >= a) b++;

            long delta = ComputeSwapDelta(n, flow, dist, p, a, b);
            if (delta < 0)
            {
                Swap(ref p[a], ref p[b]);
                cost += delta;
            }
        }
    }

    private static long ComputeCost(int n, int[] flow, int[] dist, int[] p)
    {
        long cost = 0;

        for (int i = 0; i < n; i++)
        {
            int flowRow = i * n;
            int distRow = p[i] * n;
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[flowRow + j] * dist[distRow + p[j]];
            }
        }

        return cost;
    }

    private static long ComputeSwapDelta(int n, int[] flow, int[] dist, int[] p, int a, int b)
    {
        int pa = p[a];
        int pb = p[b];

        int rowA = a * n;
        int rowB = b * n;
        int paN = pa * n;
        int pbN = pb * n;

        long delta = 0;

        delta += (long)flow[rowA + a] * (dist[pbN + pb] - dist[paN + pa]);
        delta += (long)flow[rowB + b] * (dist[paN + pa] - dist[pbN + pb]);
        delta += (long)flow[rowA + b] * (dist[pbN + pa] - dist[paN + pb]);
        delta += (long)flow[rowB + a] * (dist[paN + pb] - dist[pbN + pa]);

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;

            int pk = p[k];
            int pkN = pk * n;

            delta += (long)flow[rowA + k] * (dist[pbN + pk] - dist[paN + pk]);
            delta += (long)flow[rowB + k] * (dist[paN + pk] - dist[pbN + pk]);
            delta += (long)flow[k * n + a] * (dist[pkN + pb] - dist[pkN + pa]);
            delta += (long)flow[k * n + b] * (dist[pkN + pa] - dist[pkN + pb]);
        }

        return delta;
    }

    private static void Swap(ref int a, ref int b)
    {
        int t = a;
        a = b;
        b = t;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
            _len = 0;
            _ptr = 0;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public bool TryNextInt(out int value)
        {
            int c;
            do
            {
                c = Read();
                if (c == -1)
                {
                    value = 0;
                    return false;
                }
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = Read();
            }

            value = v * sign;
            return true;
        }

        public int NextInt()
        {
            return TryNextInt(out int v) ? v : 0;
        }
    }

    private struct XorShift64
    {
        private ulong _state;

        public XorShift64(ulong seed)
        {
            _state = seed == 0 ? 0xA0761D6478BD642FUL : seed;
        }

        private uint NextU32()
        {
            ulong x = _state;
            x ^= x >> 12;
            x ^= x << 25;
            x ^= x >> 27;
            _state = x;
            return (uint)((x * 2685821657736338717UL) >> 32);
        }

        public int NextInt(int bound)
        {
            return (int)(NextU32() % (uint)bound);
        }
    }
}
