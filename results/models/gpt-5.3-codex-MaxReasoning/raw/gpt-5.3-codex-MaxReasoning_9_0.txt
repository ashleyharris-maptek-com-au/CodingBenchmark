using System;
using System.IO;
using System.Text;

public static class Program
{
    private struct Cut
    {
        public int Length;
        public int Index;
    }

    private sealed class FastReader
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer;
        private int _length;
        private int _position;

        public FastReader(int bufferSize = 1 << 20)
        {
            _buffer = new byte[bufferSize];
        }

        private int ReadByte()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1)
                {
                    return int.MinValue;
                }
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }

            return value * sign;
        }
    }

    private sealed class BinTreap
    {
        private sealed class Node
        {
            public int Remaining;
            public int BinId;
            public uint Priority;
            public Node? Left;
            public Node? Right;

            public Node(int remaining, int binId, uint priority)
            {
                Remaining = remaining;
                BinId = binId;
                Priority = priority;
            }
        }

        private Node? _root;
        private uint _seed = 2463534242u;

        private uint NextPriority()
        {
            uint x = _seed;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _seed = x;
            return x;
        }

        private static int Compare(int remA, int idA, int remB, int idB)
        {
            if (remA != remB)
            {
                return remA < remB ? -1 : 1;
            }

            if (idA == idB)
            {
                return 0;
            }

            return idA < idB ? -1 : 1;
        }

        private static Node RotateLeft(Node node)
        {
            Node pivot = node.Right!;
            node.Right = pivot.Left;
            pivot.Left = node;
            return pivot;
        }

        private static Node RotateRight(Node node)
        {
            Node pivot = node.Left!;
            node.Left = pivot.Right;
            pivot.Right = node;
            return pivot;
        }

        private Node Insert(Node? root, Node node)
        {
            if (root == null)
            {
                return node;
            }

            int cmp = Compare(node.Remaining, node.BinId, root.Remaining, root.BinId);
            if (cmp < 0)
            {
                root.Left = Insert(root.Left, node);
                if (root.Left!.Priority > root.Priority)
                {
                    root = RotateRight(root);
                }
            }
            else
            {
                root.Right = Insert(root.Right, node);
                if (root.Right!.Priority > root.Priority)
                {
                    root = RotateLeft(root);
                }
            }

            return root;
        }

        private Node? Remove(Node? root, int remaining, int binId)
        {
            if (root == null)
            {
                return null;
            }

            int cmp = Compare(remaining, binId, root.Remaining, root.BinId);
            if (cmp < 0)
            {
                root.Left = Remove(root.Left, remaining, binId);
            }
            else if (cmp > 0)
            {
                root.Right = Remove(root.Right, remaining, binId);
            }
            else
            {
                if (root.Left == null)
                {
                    return root.Right;
                }

                if (root.Right == null)
                {
                    return root.Left;
                }

                if (root.Left.Priority > root.Right.Priority)
                {
                    root = RotateRight(root);
                    root.Right = Remove(root.Right, remaining, binId);
                }
                else
                {
                    root = RotateLeft(root);
                    root.Left = Remove(root.Left, remaining, binId);
                }
            }

            return root;
        }

        public void Insert(int remaining, int binId)
        {
            _root = Insert(_root, new Node(remaining, binId, NextPriority()));
        }

        public void Remove(int remaining, int binId)
        {
            _root = Remove(_root, remaining, binId);
        }

        public bool TryGetLowerBound(int required, out int remaining, out int binId)
        {
            Node? current = _root;
            Node? candidate = null;

            while (current != null)
            {
                if (current.Remaining >= required)
                {
                    candidate = current;
                    current = current.Left;
                }
                else
                {
                    current = current.Right;
                }
            }

            if (candidate == null)
            {
                remaining = 0;
                binId = 0;
                return false;
            }

            remaining = candidate.Remaining;
            binId = candidate.BinId;
            return true;
        }
    }

    private static void AddCutToBin(int binId, int cutIndex, int[] heads, int[] tails, int[] next)
    {
        if (heads[binId] == -1)
        {
            heads[binId] = cutIndex;
            tails[binId] = cutIndex;
        }
        else
        {
            next[tails[binId]] = cutIndex;
            tails[binId] = cutIndex;
        }

        next[cutIndex] = -1;
    }

    public static void Main()
    {
        var reader = new FastReader();

        int n = reader.NextInt();
        if (n == int.MinValue)
        {
            return;
        }

        int stockLength = reader.NextInt();
        if (stockLength == int.MinValue)
        {
            return;
        }

        if (n < 0)
        {
            return;
        }

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int len = reader.NextInt();
            if (len == int.MinValue)
            {
                len = 0;
            }

            if (len > stockLength)
            {
                Console.WriteLine("-1");
                return;
            }

            cuts[i] = new Cut { Length = len, Index = i };
        }

        Array.Sort(cuts, static (a, b) => b.Length.CompareTo(a.Length));

        int size = Math.Max(1, n);
        var heads = new int[size];
        var tails = new int[size];
        var next = new int[size];
        var remainings = new int[size];

        Array.Fill(heads, -1);
        Array.Fill(tails, -1);
        Array.Fill(next, -1);

        var treap = new BinTreap();
        int binCount = 0;

        for (int i = 0; i < n; i++)
        {
            int len = cuts[i].Length;
            int index = cuts[i].Index;

            if (treap.TryGetLowerBound(len, out int oldRemaining, out int binId))
            {
                treap.Remove(oldRemaining, binId);
                AddCutToBin(binId, index, heads, tails, next);

                int newRemaining = oldRemaining - len;
                remainings[binId] = newRemaining;
                treap.Insert(newRemaining, binId);
            }
            else
            {
                int newBinId = binCount++;
                AddCutToBin(newBinId, index, heads, tails, next);

                int remaining = stockLength - len;
                remainings[newBinId] = remaining;
                treap.Insert(remaining, newBinId);
            }
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        writer.WriteLine(binCount);

        var line = new StringBuilder(64);
        for (int b = 0; b < binCount; b++)
        {
            int cur = heads[b];
            if (cur == -1)
            {
                writer.WriteLine();
                continue;
            }

            line.Clear();
            line.Append(cur);
            while ((cur = next[cur]) != -1)
            {
                line.Append(' ');
                line.Append(cur);
            }

            writer.WriteLine(line);
        }
    }
}
