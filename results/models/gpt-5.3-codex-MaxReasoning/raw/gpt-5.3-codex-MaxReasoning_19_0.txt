using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private readonly struct Point
    {
        public readonly int X;
        public readonly int Y;

        public Point(int x, int y)
        {
            X = x;
            Y = y;
        }
    }

    private readonly struct QueueItem
    {
        public readonly int Index;
        public readonly int Version;

        public QueueItem(int index, int version)
        {
            Index = index;
            Version = version;
        }
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _pos;
        private int _len;

        public FastInput(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
            _pos = 0;
            _len = 0;
        }

        public int ReadByteRaw()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_pos++];
        }

        public byte ReadByteRequired()
        {
            int b = ReadByteRaw();
            if (b < 0)
            {
                throw new EndOfStreamException("Unexpected end of input while reading voxel bytes.");
            }

            return (byte)b;
        }

        public string ReadAsciiToken()
        {
            int b;
            do
            {
                b = ReadByteRaw();
                if (b < 0)
                {
                    return string.Empty;
                }
            }
            while (b <= 32);

            var sb = new StringBuilder(32);
            while (b > 32)
            {
                sb.Append((char)b);
                b = ReadByteRaw();
                if (b < 0)
                {
                    break;
                }
            }

            return sb.ToString();
        }
    }

    public static void Main()
    {
        try
        {
            Solve();
        }
        catch
        {
            Console.WriteLine("0");
        }
    }

    private static void Solve()
    {
        var input = new FastInput(Console.OpenStandardInput());

        string sx = input.ReadAsciiToken();
        string sy = input.ReadAsciiToken();
        string sz = input.ReadAsciiToken();
        string sbx = input.ReadAsciiToken();
        string sby = input.ReadAsciiToken();
        string sbz = input.ReadAsciiToken();

        if (string.IsNullOrEmpty(sx) || string.IsNullOrEmpty(sy) || string.IsNullOrEmpty(sz))
        {
            Console.WriteLine("0");
            return;
        }

        int xCount = int.Parse(sx, CultureInfo.InvariantCulture);
        int yCount = int.Parse(sy, CultureInfo.InvariantCulture);
        int zCount = int.Parse(sz, CultureInfo.InvariantCulture);

        // Parsed to consume full header format even though strategy uses voxel index space.
        _ = double.Parse(sbx, CultureInfo.InvariantCulture);
        _ = double.Parse(sby, CultureInfo.InvariantCulture);
        _ = double.Parse(sbz, CultureInfo.InvariantCulture);

        if (xCount <= 0 || yCount <= 0 || zCount <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        long xyLong = (long)xCount * yCount;
        long totalVoxels = xyLong * zCount;

        if (xyLong > int.MaxValue)
        {
            // Extremely large XY would not be practical with current memory model.
            ConsumeRemaining(input, totalVoxels);
            Console.WriteLine("0");
            return;
        }

        int xy = (int)xyLong;

        bool smallDetailed = totalVoxels <= 40_000_000L && xyLong <= 2_000_000L;
        bool mediumSurface = !smallDetailed && xyLong <= 8_000_000L;

        int[] topZ = new int[xy];
        byte[] topMass = new byte[xy];
        byte[] topValue = new byte[xy];

        Array.Fill(topZ, -1);

        byte[]? voxels = null;
        if (smallDetailed)
        {
            if (totalVoxels > int.MaxValue)
            {
                smallDetailed = false;
            }
            else
            {
                voxels = new byte[(int)totalVoxels];
            }
        }

        ReadWorld(input, xCount, yCount, zCount, topZ, topMass, topValue, voxels);

        var sw = Stopwatch.StartNew();
        var body = new StringBuilder(1 << 20);
        int opCount = 0;

        if (smallDetailed && voxels != null)
        {
            opCount = RunSmallDetailedStrategy(
                xCount,
                yCount,
                zCount,
                voxels,
                topZ,
                topMass,
                topValue,
                body,
                sw);
        }
        else
        {
            opCount = RunSurfaceStrategy(
                xCount,
                yCount,
                topZ,
                topMass,
                topValue,
                mediumSurface,
                body,
                sw);
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        writer.Write(opCount.ToString(CultureInfo.InvariantCulture));
        writer.Write('\n');
        writer.Write(body.ToString());
    }

    private static void ConsumeRemaining(FastInput input, long total)
    {
        for (long i = 0; i < total; i++)
        {
            if (input.ReadByteRaw() < 0)
            {
                break;
            }
        }
    }

    private static void ReadWorld(
        FastInput input,
        int xCount,
        int yCount,
        int zCount,
        int[] topZ,
        byte[] topMass,
        byte[] topValue,
        byte[]? voxels)
    {
        for (int x = 0; x < xCount; x++)
        {
            int xBase = x * yCount;
            for (int y = 0; y < yCount; y++)
            {
                int idx = xBase + y;
                int top = -1;
                byte tMass = 0;
                byte tValue = 0;

                long columnBase = (long)idx * zCount;
                for (int z = 0; z < zCount; z++)
                {
                    byte d = input.ReadByteRequired();

                    if (voxels != null)
                    {
                        voxels[columnBase + z] = d;
                    }

                    int mass = d & 0x0F;
                    if (mass != 0)
                    {
                        top = z;
                        tMass = (byte)mass;
                        tValue = (byte)(d >> 4);
                    }
                }

                topZ[idx] = top;
                topMass[idx] = tMass;
                topValue[idx] = tValue;
            }
        }
    }

    private static int RunSmallDetailedStrategy(
        int xCount,
        int yCount,
        int zCount,
        byte[] voxels,
        int[] topZ,
        byte[] topMass,
        byte[] topValue,
        StringBuilder body,
        Stopwatch sw)
    {
        int maxOps = Math.Min(200_000, xCount * yCount * 4);
        int[] versions = new int[xCount * yCount];
        var pq = new PriorityQueue<QueueItem, long>();
        var path = new List<Point>(256);

        for (int x = 0; x < xCount; x++)
        {
            int xBase = x * yCount;
            for (int y = 0; y < yCount; y++)
            {
                int idx = xBase + y;
                if (topZ[idx] >= 0 && topMass[idx] > 0 && topValue[idx] > 0)
                {
                    long approx = ApproxScore(x, y, topMass[idx], topValue[idx]);
                    pq.Enqueue(new QueueItem(idx, 0), -approx);
                }
            }
        }

        int ops = 0;
        while (pq.Count > 0 && ops < maxOps)
        {
            if (sw.ElapsedMilliseconds > 270_000)
            {
                break;
            }

            QueueItem item = pq.Dequeue();
            int idx = item.Index;
            if (item.Version != versions[idx])
            {
                continue;
            }

            int z = topZ[idx];
            if (z < 0)
            {
                continue;
            }

            byte d = voxels[(long)idx * zCount + z];
            int mass = d & 0x0F;
            int value = d >> 4;
            if (mass <= 0 || value <= 0)
            {
                continue;
            }

            int sx = idx / yCount;
            int sy = idx % yCount;

            if (!TryBestPath(sx, sy, 0, 0, topZ, yCount, z, mass, out bool xFirst, out long moveCost))
            {
                continue;
            }

            long revenue = 1000L * mass * value;
            long profit = revenue - moveCost;
            if (profit <= 0)
            {
                continue;
            }

            BuildPathPoints(sx, sy, 0, 0, xFirst, path);
            AppendOperation(body, sx, sy, z, path);
            ops++;

            // Remove this top voxel.
            topZ[idx] = FindNextSolidZ(voxels, idx, z - 1, zCount);
            versions[idx]++;

            int newTop = topZ[idx];
            if (newTop >= 0)
            {
                byte nd = voxels[(long)idx * zCount + newTop];
                int nmass = nd & 0x0F;
                int nvalue = nd >> 4;
                topMass[idx] = (byte)nmass;
                topValue[idx] = (byte)nvalue;

                if (nmass > 0 && nvalue > 0)
                {
                    long approx = ApproxScore(sx, sy, (byte)nmass, (byte)nvalue);
                    pq.Enqueue(new QueueItem(idx, versions[idx]), -approx);
                }
            }
            else
            {
                topMass[idx] = 0;
                topValue[idx] = 0;
            }
        }

        return ops;
    }

    private static int RunSurfaceStrategy(
        int xCount,
        int yCount,
        int[] topZ,
        byte[] topMass,
        byte[] topValue,
        bool mediumSurface,
        StringBuilder body,
        Stopwatch sw)
    {
        int maxOps = mediumSurface ? 60_000 : 20_000;
        var candidates = new List<(int idx, long score)>(200_000);

        if (mediumSurface)
        {
            for (int x = 0; x < xCount; x++)
            {
                int xBase = x * yCount;
                for (int y = 0; y < yCount; y++)
                {
                    int idx = xBase + y;
                    if (topZ[idx] >= 0 && topMass[idx] > 0 && topValue[idx] > 0)
                    {
                        long s = ApproxScore(x, y, topMass[idx], topValue[idx]);
                        candidates.Add((idx, s));
                    }
                }
            }
        }
        else
        {
            // Large-world mode: prioritize a near-plant window and coarse global samples.
            int window = Math.Min(700, Math.Min(xCount - 1, yCount - 1));
            for (int x = 0; x <= window; x++)
            {
                int xBase = x * yCount;
                for (int y = 0; y <= window; y++)
                {
                    int idx = xBase + y;
                    if (topZ[idx] >= 0 && topMass[idx] > 0 && topValue[idx] > 0)
                    {
                        long s = ApproxScore(x, y, topMass[idx], topValue[idx]);
                        candidates.Add((idx, s));
                    }
                }
            }

            int stride = Math.Max(8, Math.Max(xCount, yCount) / 256);
            for (int x = 0; x < xCount; x += stride)
            {
                int xBase = x * yCount;
                for (int y = 0; y < yCount; y += stride)
                {
                    int idx = xBase + y;
                    if (topZ[idx] >= 0 && topMass[idx] > 0 && topValue[idx] > 0)
                    {
                        long s = ApproxScore(x, y, topMass[idx], topValue[idx]);
                        candidates.Add((idx, s));
                    }
                }
            }
        }

        candidates.Sort((a, b) => b.score.CompareTo(a.score));

        var path = new List<Point>(256);
        int ops = 0;

        for (int i = 0; i < candidates.Count && ops < maxOps; i++)
        {
            if (sw.ElapsedMilliseconds > 270_000)
            {
                break;
            }

            int idx = candidates[i].idx;
            int z = topZ[idx];
            if (z < 0)
            {
                continue;
            }

            int mass = topMass[idx];
            int value = topValue[idx];
            if (mass <= 0 || value <= 0)
            {
                continue;
            }

            int sx = idx / yCount;
            int sy = idx % yCount;

            if (!TryBestPath(sx, sy, 0, 0, topZ, yCount, z, mass, out bool xFirst, out long moveCost))
            {
                continue;
            }

            long revenue = 1000L * mass * value;
            long profit = revenue - moveCost;
            if (profit <= 0)
            {
                continue;
            }

            BuildPathPoints(sx, sy, 0, 0, xFirst, path);
            AppendOperation(body, sx, sy, z, path);
            ops++;

            // We only know top layer in this mode. Lower it by one and avoid revisiting.
            topZ[idx] = z - 1;
            topMass[idx] = 0;
            topValue[idx] = 0;
        }

        return ops;
    }

    private static long ApproxScore(int x, int y, byte mass, byte value)
    {
        long revenue = 1000L * mass * value;
        long distancePenalty = mass * (long)(x + y);
        return revenue - distancePenalty;
    }

    private static int FindNextSolidZ(byte[] voxels, int idx, int z, int zCount)
    {
        long basePos = (long)idx * zCount;
        for (int k = z; k >= 0; k--)
        {
            if ((voxels[basePos + k] & 0x0F) != 0)
            {
                return k;
            }
        }

        return -1;
    }

    private static bool TryBestPath(
        int sx,
        int sy,
        int dx,
        int dy,
        int[] heights,
        int yCount,
        int startZ,
        int mass,
        out bool xFirst,
        out long bestCost)
    {
        bool okX = TryEvaluatePath(sx, sy, dx, dy, true, heights, yCount, startZ, mass, out long costX);
        bool okY = TryEvaluatePath(sx, sy, dx, dy, false, heights, yCount, startZ, mass, out long costY);

        if (!okX && !okY)
        {
            xFirst = true;
            bestCost = long.MaxValue;
            return false;
        }

        if (okX && (!okY || costX <= costY))
        {
            xFirst = true;
            bestCost = costX;
            return true;
        }

        xFirst = false;
        bestCost = costY;
        return true;
    }

    private static bool TryEvaluatePath(
        int sx,
        int sy,
        int dx,
        int dy,
        bool xFirst,
        int[] heights,
        int yCount,
        int startZ,
        int mass,
        out long cost)
    {
        int cx = sx;
        int cy = sy;
        int currentH = startZ;
        cost = 0L;

        if (xFirst)
        {
            while (cx != dx)
            {
                cx += dx > cx ? 1 : -1;
                int nh = heights[cx * yCount + cy];
                if (Math.Abs(nh - currentH) > 1)
                {
                    return false;
                }

                if (nh > currentH)
                {
                    cost += 5L * mass;
                }
                else if (nh == currentH)
                {
                    cost += 1L * mass;
                }

                currentH = nh;
            }

            while (cy != dy)
            {
                cy += dy > cy ? 1 : -1;
                int nh = heights[cx * yCount + cy];
                if (Math.Abs(nh - currentH) > 1)
                {
                    return false;
                }

                if (nh > currentH)
                {
                    cost += 5L * mass;
                }
                else if (nh == currentH)
                {
                    cost += 1L * mass;
                }

                currentH = nh;
            }
        }
        else
        {
            while (cy != dy)
            {
                cy += dy > cy ? 1 : -1;
                int nh = heights[cx * yCount + cy];
                if (Math.Abs(nh - currentH) > 1)
                {
                    return false;
                }

                if (nh > currentH)
                {
                    cost += 5L * mass;
                }
                else if (nh == currentH)
                {
                    cost += 1L * mass;
                }

                currentH = nh;
            }

            while (cx != dx)
            {
                cx += dx > cx ? 1 : -1;
                int nh = heights[cx * yCount + cy];
                if (Math.Abs(nh - currentH) > 1)
                {
                    return false;
                }

                if (nh > currentH)
                {
                    cost += 5L * mass;
                }
                else if (nh == currentH)
                {
                    cost += 1L * mass;
                }

                currentH = nh;
            }
        }

        return true;
    }

    private static void BuildPathPoints(int sx, int sy, int dx, int dy, bool xFirst, List<Point> path)
    {
        path.Clear();

        int cx = sx;
        int cy = sy;

        if (xFirst)
        {
            while (cx != dx)
            {
                cx += dx > cx ? 1 : -1;
                path.Add(new Point(cx, cy));
            }

            while (cy != dy)
            {
                cy += dy > cy ? 1 : -1;
                path.Add(new Point(cx, cy));
            }
        }
        else
        {
            while (cy != dy)
            {
                cy += dy > cy ? 1 : -1;
                path.Add(new Point(cx, cy));
            }

            while (cx != dx)
            {
                cx += dx > cx ? 1 : -1;
                path.Add(new Point(cx, cy));
            }
        }
    }

    private static void AppendOperation(StringBuilder body, int digX, int digY, int digZ, List<Point> path)
    {
        int n = path.Count > 0 ? path.Count - 1 : 0;
        int destX = path.Count > 0 ? path[path.Count - 1].X : 0;
        int destY = path.Count > 0 ? path[path.Count - 1].Y : 0;

        body.Append(n.ToString(CultureInfo.InvariantCulture));
        body.Append(' ');
        body.Append(digX.ToString(CultureInfo.InvariantCulture));
        body.Append(' ');
        body.Append(digY.ToString(CultureInfo.InvariantCulture));
        body.Append(' ');
        body.Append(digZ.ToString(CultureInfo.InvariantCulture));

        for (int i = 0; i < n; i++)
        {
            body.Append(' ');
            body.Append(path[i].X.ToString(CultureInfo.InvariantCulture));
            body.Append(' ');
            body.Append(path[i].Y.ToString(CultureInfo.InvariantCulture));
        }

        body.Append(' ');
        body.Append(destX.ToString(CultureInfo.InvariantCulture));
        body.Append(' ');
        body.Append(destY.ToString(CultureInfo.InvariantCulture));
        body.Append('\n');
    }
}
