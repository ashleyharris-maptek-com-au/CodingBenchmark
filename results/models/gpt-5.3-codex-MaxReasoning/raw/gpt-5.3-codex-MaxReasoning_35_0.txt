use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Eq, PartialEq)]
struct HeapEntry {
    score: usize,
    vertex: usize,
    version: u64,
}

impl Ord for HeapEntry {
    fn cmp(&self, other: &Self) -> Ordering {
        self.score
            .cmp(&other.score)
            .then_with(|| self.vertex.cmp(&other.vertex))
            .then_with(|| self.version.cmp(&other.version))
    }
}

impl PartialOrd for HeapEntry {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn add_vertex_to_solution(
    v: usize,
    closed: &[Vec<usize>],
    covered: &mut [bool],
    in_set: &mut [bool],
    score: &mut [usize],
    version: &mut [u64],
    heap: &mut BinaryHeap<HeapEntry>,
    remaining: &mut usize,
    dominating_set: &mut Vec<usize>,
) {
    if in_set[v] {
        return;
    }

    in_set[v] = true;
    dominating_set.push(v);

    for &u in &closed[v] {
        if covered[u] {
            continue;
        }

        covered[u] = true;
        *remaining -= 1;

        // Vertex u is no longer uncovered, so every x whose closed neighborhood
        // contains u loses one unit of marginal gain.
        for &x in &closed[u] {
            if score[x] > 0 {
                score[x] -= 1;
                version[x] += 1;
                heap.push(HeapEntry {
                    score: score[x],
                    vertex: x,
                    version: version[x],
                });
            }
        }
    }
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        println!("0\n");
        return;
    }

    let mut it = input.split_whitespace();

    let n: usize = match it.next().and_then(|x| x.parse::<usize>().ok()) {
        Some(v) => v,
        None => {
            println!("0\n");
            return;
        }
    };
    let m: usize = it
        .next()
        .and_then(|x| x.parse::<usize>().ok())
        .unwrap_or(0);

    if n == 0 {
        println!("0\n");
        return;
    }

    let mut adj = vec![Vec::<usize>::new(); n];

    for _ in 0..m {
        let Some(us) = it.next() else { break };
        let Some(vs) = it.next() else { break };

        let Some(u) = us.parse::<usize>().ok() else {
            continue;
        };
        let Some(v) = vs.parse::<usize>().ok() else {
            continue;
        };

        if u >= n || v >= n {
            continue;
        }

        adj[u].push(v);
        if u != v {
            adj[v].push(u);
        }
    }

    for neigh in &mut adj {
        neigh.sort_unstable();
        neigh.dedup();
    }

    let mut closed: Vec<Vec<usize>> = Vec::with_capacity(n);
    for v in 0..n {
        let mut c = Vec::with_capacity(adj[v].len() + 1);
        c.push(v);
        c.extend_from_slice(&adj[v]);
        c.sort_unstable();
        c.dedup();
        closed.push(c);
    }

    let mut covered = vec![false; n];
    let mut in_set = vec![false; n];
    let mut score = vec![0usize; n];
    let mut version = vec![0u64; n];
    let mut heap = BinaryHeap::<HeapEntry>::new();
    let mut dominating_set = Vec::<usize>::new();

    for v in 0..n {
        score[v] = closed[v].len();
        heap.push(HeapEntry {
            score: score[v],
            vertex: v,
            version: version[v],
        });
    }

    let mut remaining = n;

    while remaining > 0 {
        let mut chosen = None;

        while let Some(top) = heap.pop() {
            if top.version != version[top.vertex] || top.score != score[top.vertex] {
                continue;
            }
            if in_set[top.vertex] {
                continue;
            }
            chosen = Some(top.vertex);
            break;
        }

        let v = if let Some(v) = chosen {
            if score[v] == 0 {
                covered.iter().position(|&x| !x).unwrap_or(0)
            } else {
                v
            }
        } else {
            covered.iter().position(|&x| !x).unwrap_or(0)
        };

        add_vertex_to_solution(
            v,
            &closed,
            &mut covered,
            &mut in_set,
            &mut score,
            &mut version,
            &mut heap,
            &mut remaining,
            &mut dominating_set,
        );
    }

    // Redundancy pruning: remove any selected vertex whose closed neighborhood
    // remains dominated by other selected vertices.
    let mut dom_count = vec![0u32; n];
    for &v in &dominating_set {
        for &u in &closed[v] {
            dom_count[u] += 1;
        }
    }

    let mut order = dominating_set.clone();
    order.sort_by_key(|&v| {
        let unique = closed[v].iter().filter(|&&u| dom_count[u] == 1).count();
        (unique, closed[v].len())
    });

    for &v in &order {
        if !in_set[v] {
            continue;
        }
        let removable = closed[v].iter().all(|&u| dom_count[u] >= 2);
        if removable {
            in_set[v] = false;
            for &u in &closed[v] {
                dom_count[u] -= 1;
            }
        }
    }

    // Safety fallback (should almost never trigger): ensure all vertices are dominated.
    for u in 0..n {
        if closed[u].iter().any(|&x| in_set[x]) {
            continue;
        }
        in_set[u] = true;
    }

    let mut answer = Vec::new();
    for (v, &take) in in_set.iter().enumerate() {
        if take {
            answer.push(v);
        }
    }
    answer.sort_unstable();

    println!("{}", answer.len());
    if answer.is_empty() {
        println!();
    } else {
        for (i, v) in answer.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    }
}
