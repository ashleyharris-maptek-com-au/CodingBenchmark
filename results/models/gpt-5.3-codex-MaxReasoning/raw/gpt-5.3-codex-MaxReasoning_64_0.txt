declare const require: any;
const fs = require('fs');

type DroneInput = {
  id?: string | number;
  x: number;
  y: number;
};

type InputData = {
  probability: number[][];
  drones: DroneInput[];
  maxStepsPerDrone: number;
};

class BoundedBFS {
  private readonly width: number;
  private readonly height: number;
  private readonly n: number;
  private readonly passable: Uint8Array;
  readonly seen: Int32Array;
  readonly dist: Int32Array;
  readonly parent: Int32Array;
  readonly queue: Int32Array;
  private stamp: number;

  constructor(passable: Uint8Array, width: number, height: number) {
    this.passable = passable;
    this.width = width;
    this.height = height;
    this.n = width * height;
    this.seen = new Int32Array(this.n);
    this.dist = new Int32Array(this.n);
    this.parent = new Int32Array(this.n);
    this.queue = new Int32Array(this.n);
    this.stamp = 1;
  }

  run(start: number, maxDist: number): number {
    if (maxDist < 0 || start < 0 || start >= this.n || this.passable[start] === 0) {
      return 0;
    }

    this.stamp += 1;
    if (this.stamp >= 2147483640) {
      this.seen.fill(0);
      this.stamp = 1;
    }

    const stamp = this.stamp;
    const seen = this.seen;
    const dist = this.dist;
    const parent = this.parent;
    const queue = this.queue;
    const passable = this.passable;
    const width = this.width;

    let head = 0;
    let tail = 0;

    seen[start] = stamp;
    dist[start] = 0;
    parent[start] = -1;
    queue[tail++] = start;

    while (head < tail) {
      const v = queue[head++];
      const d = dist[v];
      if (d >= maxDist) {
        continue;
      }

      const x = v % width;
      const y = (v / width) | 0;
      const nd = d + 1;

      let nb = v - 1;
      if (x > 0 && passable[nb] !== 0 && seen[nb] !== stamp) {
        seen[nb] = stamp;
        dist[nb] = nd;
        parent[nb] = v;
        queue[tail++] = nb;
      }

      nb = v + 1;
      if (x + 1 < width && passable[nb] !== 0 && seen[nb] !== stamp) {
        seen[nb] = stamp;
        dist[nb] = nd;
        parent[nb] = v;
        queue[tail++] = nb;
      }

      nb = v - width;
      if (y > 0 && passable[nb] !== 0 && seen[nb] !== stamp) {
        seen[nb] = stamp;
        dist[nb] = nd;
        parent[nb] = v;
        queue[tail++] = nb;
      }

      nb = v + width;
      if (y + 1 < this.height && passable[nb] !== 0 && seen[nb] !== stamp) {
        seen[nb] = stamp;
        dist[nb] = nd;
        parent[nb] = v;
        queue[tail++] = nb;
      }
    }

    return tail;
  }
}

type Candidate = {
  id: number;
  dist: number;
  score: number;
};

function toInt(value: unknown, fallback: number): number {
  const n = Number(value);
  return Number.isFinite(n) ? Math.floor(n) : fallback;
}

function clamp(v: number, lo: number, hi: number): number {
  if (v < lo) {
    return lo;
  }
  if (v > hi) {
    return hi;
  }
  return v;
}

function claimCell(id: number, claimed: Uint8Array, value: Float64Array): void {
  if (id >= 0 && claimed[id] === 0 && value[id] > 0) {
    claimed[id] = 1;
  }
}

function findNearestPassable(
  startX: number,
  startY: number,
  passable: Uint8Array,
  width: number,
  height: number
): number {
  let bestId = -1;
  let bestDist = Number.MAX_SAFE_INTEGER;

  for (let y = 0; y < height; y++) {
    const rowBase = y * width;
    for (let x = 0; x < width; x++) {
      const id = rowBase + x;
      if (passable[id] === 0) {
        continue;
      }
      const d = Math.abs(x - startX) + Math.abs(y - startY);
      if (d < bestDist) {
        bestDist = d;
        bestId = id;
        if (d === 0) {
          return bestId;
        }
      }
    }
  }

  return bestId;
}

function chooseBestTarget(
  droneIndex: number,
  currentId: number,
  remaining: number,
  tail: number,
  bfs: BoundedBFS,
  owner: Int32Array,
  claimed: Uint8Array,
  value: Float64Array,
  density: Float64Array
): Candidate {
  const q = bfs.queue;
  const dist = bfs.dist;

  let bestId = -1;
  let bestDist = 0;
  let bestScore = 0;

  for (let i = 0; i < tail; i++) {
    const id = q[i];
    if (id === currentId || claimed[id] !== 0) {
      continue;
    }

    const v = value[id];
    if (v <= 0) {
      continue;
    }

    const d = dist[id];
    if (d <= 0 || d > remaining) {
      continue;
    }

    let base = v + 0.28 * density[id];
    if (owner[id] === droneIndex) {
      base *= 1.12;
    } else if (owner[id] !== -1) {
      base *= 0.93;
    }

    const score = base / Math.pow(d + 1, 1.12);
    if (score > bestScore || (score === bestScore && (bestId === -1 || d < bestDist))) {
      bestScore = score;
      bestDist = d;
      bestId = id;
    }
  }

  return { id: bestId, dist: bestDist, score: bestScore };
}

function solve(): void {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: Partial<InputData> = JSON.parse(input || '{}');

  const gridRaw = Array.isArray(data.probability) ? data.probability : [];
  const height = gridRaw.length;
  const width =
    height > 0 && Array.isArray(gridRaw[0])
      ? (gridRaw[0] as unknown[]).length
      : 0;

  const drones: DroneInput[] = Array.isArray(data.drones) ? data.drones : [];
  const maxSteps = Number.isFinite(data.maxStepsPerDrone)
    ? Math.max(0, Math.floor(Number(data.maxStepsPerDrone)))
    : 0;

  if (height <= 0 || width <= 0) {
    const fallbackPaths = drones.map((d) => {
      const sx = toInt(d?.x, 0);
      const sy = toInt(d?.y, 0);
      const p: number[][] = [[sx, sy]];
      for (let i = 0; i < maxSteps; i++) {
        p.push([sx, sy]);
      }
      return p;
    });

    console.log(JSON.stringify({ paths: fallbackPaths }));
    return;
  }

  const n = width * height;
  const passable = new Uint8Array(n);
  const value = new Float64Array(n);
  let passableCount = 0;

  for (let y = 0; y < height; y++) {
    const row = Array.isArray(gridRaw[y]) ? gridRaw[y] : [];
    const rowAny = row as unknown[];
    const base = y * width;

    for (let x = 0; x < width; x++) {
      const id = base + x;
      const raw = Number(rowAny[x]);
      if (!Number.isFinite(raw) || raw <= -1) {
        passable[id] = 0;
        value[id] = 0;
      } else {
        passable[id] = 1;
        passableCount += 1;
        value[id] = raw > 0 ? raw : 0;
      }
    }
  }

  const droneCount = drones.length;
  const paths: number[][][] = new Array(droneCount);

  if (passableCount === 0) {
    for (let i = 0; i < droneCount; i++) {
      const sx = clamp(toInt(drones[i]?.x, 0), 0, width - 1);
      const sy = clamp(toInt(drones[i]?.y, 0), 0, height - 1);
      const p: number[][] = [[sx, sy]];
      for (let step = 0; step < maxSteps; step++) {
        p.push([sx, sy]);
      }
      paths[i] = p;
    }

    console.log(JSON.stringify({ paths }));
    return;
  }

  const density = new Float64Array(n);
  const offsets: Array<[number, number, number]> = [];
  for (let dy = -2; dy <= 2; dy++) {
    for (let dx = -2; dx <= 2; dx++) {
      const md = Math.abs(dx) + Math.abs(dy);
      if (md <= 2) {
        offsets.push([dx, dy, 1 / (md + 1)]);
      }
    }
  }

  for (let y = 0; y < height; y++) {
    const base = y * width;
    for (let x = 0; x < width; x++) {
      const id = base + x;
      if (passable[id] === 0) {
        continue;
      }

      let sum = 0;
      for (let k = 0; k < offsets.length; k++) {
        const dx = offsets[k][0];
        const dy = offsets[k][1];
        const w = offsets[k][2];
        const nx = x + dx;
        const ny = y + dy;
        if (nx < 0 || ny < 0 || nx >= width || ny >= height) {
          continue;
        }

        const nid = ny * width + nx;
        if (passable[nid] !== 0 && value[nid] > 0) {
          sum += value[nid] * w;
        }
      }

      density[id] = sum;
    }
  }

  const startIds = new Int32Array(droneCount);
  const currentIds = new Int32Array(droneCount);
  const remaining = new Int32Array(droneCount);
  const claimed = new Uint8Array(n);

  for (let i = 0; i < droneCount; i++) {
    const rawX = clamp(toInt(drones[i]?.x, 0), 0, width - 1);
    const rawY = clamp(toInt(drones[i]?.y, 0), 0, height - 1);

    let sid = rawY * width + rawX;
    if (passable[sid] === 0) {
      const nearest = findNearestPassable(rawX, rawY, passable, width, height);
      if (nearest !== -1) {
        sid = nearest;
      }
    }

    startIds[i] = sid;
    currentIds[i] = sid;
    remaining[i] = maxSteps;
    paths[i] = [[sid % width, (sid / width) | 0]];
  }

  for (let i = 0; i < droneCount; i++) {
    claimCell(startIds[i], claimed, value);
  }

  const owner = new Int32Array(n);
  owner.fill(-1);
  {
    const q = new Int32Array(n);
    let head = 0;
    let tail = 0;

    for (let i = 0; i < droneCount; i++) {
      const sid = startIds[i];
      if (passable[sid] === 0) {
        continue;
      }
      if (owner[sid] === -1) {
        owner[sid] = i;
        q[tail++] = sid;
      }
    }

    while (head < tail) {
      const v = q[head++];
      const o = owner[v];
      const x = v % width;
      const y = (v / width) | 0;

      let nb = v - 1;
      if (x > 0 && passable[nb] !== 0 && owner[nb] === -1) {
        owner[nb] = o;
        q[tail++] = nb;
      }

      nb = v + 1;
      if (x + 1 < width && passable[nb] !== 0 && owner[nb] === -1) {
        owner[nb] = o;
        q[tail++] = nb;
      }

      nb = v - width;
      if (y > 0 && passable[nb] !== 0 && owner[nb] === -1) {
        owner[nb] = o;
        q[tail++] = nb;
      }

      nb = v + width;
      if (y + 1 < height && passable[nb] !== 0 && owner[nb] === -1) {
        owner[nb] = o;
        q[tail++] = nb;
      }
    }
  }

  const order = Array.from({ length: droneCount }, (_, i) => i);
  order.sort((a, b) => {
    const pa = value[startIds[a]] + 0.2 * density[startIds[a]];
    const pb = value[startIds[b]] + 0.2 * density[startIds[b]];
    return pb - pa;
  });

  const bfs = new BoundedBFS(passable, width, height);
  const localRadius = 28;

  for (let ord = 0; ord < order.length; ord++) {
    const droneIndex = order[ord];
    const path = paths[droneIndex];

    while (remaining[droneIndex] > 0) {
      const cur = currentIds[droneIndex];
      claimCell(cur, claimed, value);

      const firstLimit = remaining[droneIndex] < localRadius ? remaining[droneIndex] : localRadius;
      let tail = bfs.run(cur, firstLimit);
      let best = chooseBestTarget(
        droneIndex,
        cur,
        remaining[droneIndex],
        tail,
        bfs,
        owner,
        claimed,
        value,
        density
      );

      if (best.id === -1 && firstLimit < remaining[droneIndex]) {
        tail = bfs.run(cur, remaining[droneIndex]);
        best = chooseBestTarget(
          droneIndex,
          cur,
          remaining[droneIndex],
          tail,
          bfs,
          owner,
          claimed,
          value,
          density
        );
      }

      if (best.id === -1) {
        break;
      }

      const dist = best.dist;
      const steps = new Int32Array(dist);
      let t = best.id;
      for (let k = dist - 1; k >= 0; k--) {
        steps[k] = t;
        t = bfs.parent[t];
      }

      for (let s = 0; s < steps.length && remaining[droneIndex] > 0; s++) {
        const nid = steps[s];
        currentIds[droneIndex] = nid;
        remaining[droneIndex] -= 1;
        path.push([nid % width, (nid / width) | 0]);
        claimCell(nid, claimed, value);
      }
    }

    while (remaining[droneIndex] > 0) {
      const stay = currentIds[droneIndex];
      path.push([stay % width, (stay / width) | 0]);
      remaining[droneIndex] -= 1;
    }
  }

  console.log(JSON.stringify({ paths }));
}

solve();
