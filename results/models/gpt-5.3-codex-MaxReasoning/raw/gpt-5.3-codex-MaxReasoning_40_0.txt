#include <bits/stdc++.h>
using namespace std;

class MaxWeightCliqueSolver {
public:
    MaxWeightCliqueSolver(const vector<long long>& weights,
                          const vector<vector<uint64_t>>& adjBits)
        : n((int)weights.size()), w(weights), adj(adjBits), words((n + 63) >> 6),
          degree(n, 0), bestWeight(0) {
        for (int v = 0; v < n; ++v) {
            int d = 0;
            for (uint64_t x : adj[v]) d += __builtin_popcountll(x);
            degree[v] = d;
        }
    }

    pair<long long, vector<int>> solve() {
        vector<int> P(n);
        iota(P.begin(), P.end(), 0);

        // Greedy initial incumbent.
        vector<int> order = P;
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (w[a] != w[b]) return w[a] > w[b];
            return degree[a] > degree[b];
        });
        vector<int> greedyClique;
        long long greedyWeight = 0;
        for (int v : order) {
            bool ok = true;
            for (int u : greedyClique) {
                if (!isAdjacent(u, v)) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                greedyClique.push_back(v);
                greedyWeight += w[v];
            }
        }
        bestWeight = greedyWeight;
        bestClique = greedyClique;

        currentClique.clear();
        expand(P, 0LL);
        return {bestWeight, bestClique};
    }

private:
    int n;
    vector<long long> w;
    vector<vector<uint64_t>> adj;
    int words;
    vector<int> degree;

    long long bestWeight;
    vector<int> bestClique;
    vector<int> currentClique;

    bool isAdjacent(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    bool intersects(const vector<uint64_t>& a, const vector<uint64_t>& b) const {
        for (int i = 0; i < words; ++i) {
            if (a[i] & b[i]) return true;
        }
        return false;
    }

    void colorSort(const vector<int>& P, vector<int>& ordered, vector<long long>& bounds) {
        vector<int> verts = P;
        sort(verts.begin(), verts.end(), [&](int a, int b) {
            if (w[a] != w[b]) return w[a] > w[b];
            return degree[a] > degree[b];
        });

        vector<vector<int>> classes;
        vector<vector<uint64_t>> classMasks;
        vector<long long> classMax;

        classes.reserve(verts.size());
        classMasks.reserve(verts.size());
        classMax.reserve(verts.size());

        for (int v : verts) {
            int assigned = -1;
            for (int c = 0; c < (int)classes.size(); ++c) {
                if (!intersects(adj[v], classMasks[c])) {
                    assigned = c;
                    break;
                }
            }
            if (assigned == -1) {
                assigned = (int)classes.size();
                classes.push_back({});
                classMasks.push_back(vector<uint64_t>(words, 0ULL));
                classMax.push_back(0);
            }
            classes[assigned].push_back(v);
            classMasks[assigned][v >> 6] |= (1ULL << (v & 63));
            classMax[assigned] = max(classMax[assigned], w[v]);
        }

        ordered.clear();
        bounds.clear();
        ordered.reserve(P.size());
        bounds.reserve(P.size());

        long long pref = 0;
        for (int c = 0; c < (int)classes.size(); ++c) {
            auto& cls = classes[c];
            sort(cls.begin(), cls.end(), [&](int a, int b) {
                if (w[a] != w[b]) return w[a] > w[b];
                return degree[a] > degree[b];
            });
            pref += classMax[c];
            for (int v : cls) {
                ordered.push_back(v);
                bounds.push_back(pref);
            }
        }
    }

    void expand(const vector<int>& P, long long currentWeight) {
        if (P.empty()) {
            if (currentWeight > bestWeight) {
                bestWeight = currentWeight;
                bestClique = currentClique;
            }
            return;
        }

        vector<int> order;
        vector<long long> bounds;
        colorSort(P, order, bounds);

        for (int i = (int)order.size() - 1; i >= 0; --i) {
            if (currentWeight + bounds[i] <= bestWeight) {
                return;
            }

            int v = order[i];
            long long nextWeight = currentWeight + w[v];

            currentClique.push_back(v);

            vector<int> nextP;
            nextP.reserve(i);
            for (int j = 0; j < i; ++j) {
                int u = order[j];
                if (isAdjacent(u, v)) nextP.push_back(u);
            }

            if (nextP.empty()) {
                if (nextWeight > bestWeight) {
                    bestWeight = nextWeight;
                    bestClique = currentClique;
                }
            } else {
                expand(nextP, nextWeight);
            }

            currentClique.pop_back();
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) {
        return 0;
    }

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<pair<int, int>> edges;
    edges.reserve(m);
    vector<char> banned(n, 0);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) banned[u] = 1;
        edges.push_back({u, v});
    }

    vector<int> activeOrig;
    activeOrig.reserve(n);
    for (int i = 0; i < n; ++i) {
        if (!banned[i] && c[i] > 0) activeOrig.push_back(i);
    }

    int k = (int)activeOrig.size();
    vector<int> idx(n, -1);
    for (int i = 0; i < k; ++i) idx[activeOrig[i]] = i;

    vector<pair<int, int>> activeEdges;
    activeEdges.reserve(edges.size());
    for (auto [u, v] : edges) {
        int a = (u >= 0 && u < n) ? idx[u] : -1;
        int b = (v >= 0 && v < n) ? idx[v] : -1;
        if (a == -1 || b == -1 || a == b) continue;
        if (a > b) swap(a, b);
        activeEdges.push_back({a, b});
    }
    sort(activeEdges.begin(), activeEdges.end());
    activeEdges.erase(unique(activeEdges.begin(), activeEdges.end()), activeEdges.end());

    vector<vector<int>> conflictAdj(k);
    for (auto [a, b] : activeEdges) {
        conflictAdj[a].push_back(b);
        conflictAdj[b].push_back(a);
    }

    vector<int> finalX(n, 0);

    // Positive-weight isolated active vertices are always selected.
    vector<int> compId(k, -1);
    int compCnt = 0;
    for (int s = 0; s < k; ++s) {
        if (compId[s] != -1) continue;
        queue<int> q;
        q.push(s);
        compId[s] = compCnt;
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (int u : conflictAdj[v]) {
                if (compId[u] == -1) {
                    compId[u] = compCnt;
                    q.push(u);
                }
            }
        }
        ++compCnt;
    }

    vector<vector<int>> comps(compCnt);
    for (int v = 0; v < k; ++v) comps[compId[v]].push_back(v);

    for (const auto& comp : comps) {
        int sz = (int)comp.size();
        if (sz == 0) continue;
        if (sz == 1) {
            int a = comp[0];
            finalX[activeOrig[a]] = 1;
            continue;
        }

        vector<int> localOfActive(k, -1);
        for (int i = 0; i < sz; ++i) localOfActive[comp[i]] = i;

        int words = (sz + 63) >> 6;
        vector<vector<uint64_t>> conflictBits(sz, vector<uint64_t>(words, 0ULL));

        for (int i = 0; i < sz; ++i) {
            int a = comp[i];
            for (int b : conflictAdj[a]) {
                int j = localOfActive[b];
                if (j == -1) continue;
                conflictBits[i][j >> 6] |= (1ULL << (j & 63));
            }
        }

        vector<uint64_t> fullMask(words, ~0ULL);
        if (sz % 64) fullMask.back() = (1ULL << (sz % 64)) - 1ULL;

        vector<vector<uint64_t>> compatBits(sz, vector<uint64_t>(words, 0ULL));
        for (int i = 0; i < sz; ++i) {
            for (int widx = 0; widx < words; ++widx) {
                compatBits[i][widx] = (~conflictBits[i][widx]) & fullMask[widx];
            }
            compatBits[i][i >> 6] &= ~(1ULL << (i & 63));
        }

        vector<long long> localW(sz, 0);
        for (int i = 0; i < sz; ++i) {
            localW[i] = c[activeOrig[comp[i]]];
        }

        MaxWeightCliqueSolver solver(localW, compatBits);
        auto [bestW, localClique] = solver.solve();
        (void)bestW;

        for (int li : localClique) {
            int activeV = comp[li];
            finalX[activeOrig[activeV]] = 1;
        }
    }

    long long objective = 0;
    for (int i = 0; i < n; ++i) {
        objective += c[i] * (long long)finalX[i];
    }

    cout << objective << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << finalX[i];
    }
    cout << '\n';

    return 0;
}
