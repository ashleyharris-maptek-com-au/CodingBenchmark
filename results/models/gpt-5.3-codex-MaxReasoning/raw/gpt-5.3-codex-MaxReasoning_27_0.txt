using namespace std;
using Clock = chrono::steady_clock;

struct Graph {
    int n = 0;
    int k = 0;
    long long m = 0;
    vector<int> offset;
    vector<int> adj;
    vector<int> degree;
};

static inline double elapsedSeconds(const Clock::time_point& start) {
    return chrono::duration<double>(Clock::now() - start).count();
}

static inline int firstSetBit64(uint64_t x) {
#ifdef _MSC_VER
#if defined(_M_X64) || defined(_M_ARM64)
    unsigned long idx = 0;
    _BitScanForward64(&idx, x);
    return static_cast<int>(idx);
#else
    unsigned long idx = 0;
    const unsigned int low = static_cast<unsigned int>(x & 0xFFFFFFFFULL);
    if (low != 0U) {
        _BitScanForward(&idx, low);
        return static_cast<int>(idx);
    }
    _BitScanForward(&idx, static_cast<unsigned int>(x >> 32));
    return static_cast<int>(idx) + 32;
#endif
#else
    return __builtin_ctzll(x);
#endif
}

static bool readGraph(Graph& g, bool& hasSelfLoop) {
    int n, k;
    long long mInput;
    if (!(cin >> n >> mInput >> k)) {
        return false;
    }

    if (n < 0) n = 0;
    if (k < 0) k = 0;
    if (mInput < 0) mInput = 0;

    g.n = n;
    g.k = k;
    hasSelfLoop = false;

    vector<pair<int, int>> edges;
    edges.reserve(static_cast<size_t>(mInput));
    g.degree.assign(n, 0);

    for (long long i = 0; i < mInput; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            hasSelfLoop = true;
            continue;
        }
        edges.emplace_back(u, v);
        g.degree[u]++;
        g.degree[v]++;
    }

    g.m = static_cast<long long>(edges.size());
    g.offset.assign(n + 1, 0);
    for (int v = 0; v < n; ++v) {
        g.offset[v + 1] = g.offset[v] + g.degree[v];
    }

    g.adj.assign(static_cast<size_t>(g.offset[n]), 0);
    vector<int> cur = g.offset;
    for (const auto& e : edges) {
        const int u = e.first;
        const int v = e.second;
        g.adj[cur[u]++] = v;
        g.adj[cur[v]++] = u;
    }

    return true;
}

static vector<int> largestFirstOrder(const Graph& g) {
    vector<int> order(g.n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (g.degree[a] != g.degree[b]) return g.degree[a] > g.degree[b];
        return a < b;
    });
    return order;
}

static vector<int> smallestLastOrder(const Graph& g) {
    vector<int> curDeg = g.degree;
    vector<char> removed(g.n, 0);
    using Node = pair<int, int>;  // {degree, vertex}
    priority_queue<Node, vector<Node>, greater<Node>> pq;
    for (int v = 0; v < g.n; ++v) pq.push({curDeg[v], v});

    vector<int> peel;
    peel.reserve(g.n);

    while (static_cast<int>(peel.size()) < g.n) {
        auto [d, v] = pq.top();
        pq.pop();
        if (removed[v] || d != curDeg[v]) continue;

        removed[v] = 1;
        peel.push_back(v);

        for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
            int u = g.adj[i];
            if (removed[u]) continue;
            --curDeg[u];
            pq.push({curDeg[u], u});
        }
    }

    reverse(peel.begin(), peel.end());
    return peel;
}

static vector<int> perturbedOrder(const vector<int>& base, mt19937& rng, int swaps) {
    vector<int> order = base;
    if (order.empty()) return order;
    uniform_int_distribution<int> dist(0, static_cast<int>(order.size() - 1));
    for (int i = 0; i < swaps; ++i) {
        int a = dist(rng);
        int b = dist(rng);
        swap(order[a], order[b]);
    }
    return order;
}

static bool strictGreedyColor(const Graph& g, const vector<int>& order, vector<int>& color) {
    color.assign(g.n, -1);

    if (g.k <= 64) {
        const uint64_t allMask = (g.k == 64) ? ~0ULL : ((1ULL << g.k) - 1ULL);
        for (int v : order) {
            uint64_t forbidden = 0ULL;
            for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
                int u = g.adj[i];
                int cu = color[u];
                if (cu >= 0) forbidden |= (1ULL << cu);
            }
            uint64_t freeMask = allMask & ~forbidden;
            if (freeMask == 0ULL) return false;
            color[v] = firstSetBit64(freeMask);
        }
        return true;
    }

    vector<int> seen(g.k, -1);
    int stamp = 0;
    for (int v : order) {
        ++stamp;
        for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
            int u = g.adj[i];
            int cu = color[u];
            if (cu >= 0) seen[cu] = stamp;
        }
        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (seen[c] != stamp) {
                chosen = c;
                break;
            }
        }
        if (chosen < 0) return false;
        color[v] = chosen;
    }

    return true;
}

static void greedyWithConflicts(const Graph& g, const vector<int>& order, vector<int>& color, mt19937& rng) {
    color.assign(g.n, 0);
    vector<int> cnt(g.k, 0);
    vector<int> touched;
    touched.reserve(64);

    for (int v : order) {
        touched.clear();
        for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
            int u = g.adj[i];
            int cu = color[u];
            if (cu < 0 || cu >= g.k) continue;
            if (cnt[cu] == 0) touched.push_back(cu);
            cnt[cu]++;
        }

        int bestColor = 0;
        int bestConf = cnt[0];
        for (int c = 1; c < g.k; ++c) {
            int conf = cnt[c];
            if (conf < bestConf) {
                bestConf = conf;
                bestColor = c;
            }
        }

        if (bestConf > 0) {
            for (int c = 0; c < g.k; ++c) {
                if (cnt[c] == 0) {
                    bestColor = c;
                    bestConf = 0;
                    break;
                }
            }
            if (bestConf > 0) {
                vector<int> ties;
                for (int c = 0; c < g.k; ++c) {
                    if (cnt[c] == bestConf) ties.push_back(c);
                }
                if (!ties.empty()) {
                    bestColor = ties[rng() % ties.size()];
                }
            }
        }

        color[v] = bestColor;
        for (int c : touched) cnt[c] = 0;
    }
}

static long long countConflicts(const Graph& g, const vector<int>& color) {
    long long bad = 0;
    for (int v = 0; v < g.n; ++v) {
        for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
            int u = g.adj[i];
            if (u > v && color[u] == color[v]) ++bad;
        }
    }
    return bad;
}

static bool isProperColoring(const Graph& g, const vector<int>& color) {
    if (static_cast<int>(color.size()) != g.n) return false;
    for (int v = 0; v < g.n; ++v) {
        if (color[v] < 0 || color[v] >= g.k) return false;
        for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
            int u = g.adj[i];
            if (u > v && color[u] == color[v]) return false;
        }
    }
    return true;
}

static void tryUseMoreColors(const Graph& g, vector<int>& color) {
    if (g.k <= 0 || g.n <= 0) return;

    vector<int> freq(g.k, 0);
    for (int v = 0; v < g.n; ++v) {
        if (color[v] >= 0 && color[v] < g.k) freq[color[v]]++;
    }

    for (int c = 0; c < g.k; ++c) {
        if (freq[c] != 0) continue;

        bool placed = false;
        for (int v = 0; v < g.n && !placed; ++v) {
            int old = color[v];
            if (old == c) {
                placed = true;
                break;
            }
            if (old < 0 || old >= g.k) continue;
            if (freq[old] <= 1) continue;

            bool ok = true;
            for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
                int u = g.adj[i];
                if (color[u] == c) {
                    ok = false;
                    break;
                }
            }

            if (ok) {
                color[v] = c;
                freq[old]--;
                freq[c]++;
                placed = true;
            }
        }

        if (!placed) {
            for (int v = 0; v < g.n && !placed; ++v) {
                int old = color[v];
                if (old == c || old < 0 || old >= g.k) continue;

                bool ok = true;
                for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
                    int u = g.adj[i];
                    if (color[u] == c) {
                        ok = false;
                        break;
                    }
                }

                if (ok) {
                    color[v] = c;
                    freq[old]--;
                    freq[c]++;
                    placed = true;
                }
            }
        }
    }
}

static bool minConflictsRepair(const Graph& g, vector<int>& color, mt19937& rng,
                               const Clock::time_point& start, double timeLimitSec) {
    vector<int> conflictCount(g.n, 0);
    long long badEdges = 0;

    for (int v = 0; v < g.n; ++v) {
        for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
            int u = g.adj[i];
            if (u > v && color[u] == color[v]) {
                conflictCount[v]++;
                conflictCount[u]++;
                ++badEdges;
            }
        }
    }

    if (badEdges == 0) return true;

    vector<int> pos(g.n, -1);
    vector<int> conflicted;
    conflicted.reserve(g.n);

    auto addConflicted = [&](int v) {
        if (pos[v] == -1) {
            pos[v] = static_cast<int>(conflicted.size());
            conflicted.push_back(v);
        }
    };

    auto removeConflicted = [&](int v) {
        int p = pos[v];
        if (p == -1) return;
        int last = conflicted.back();
        conflicted[p] = last;
        pos[last] = p;
        conflicted.pop_back();
        pos[v] = -1;
    };

    for (int v = 0; v < g.n; ++v) {
        if (conflictCount[v] > 0) addConflicted(v);
    }

    vector<int> cnt(g.k, 0);
    vector<int> touched;
    touched.reserve(64);
    uniform_real_distribution<double> prob01(0.0, 1.0);

    long long bestBad = badEdges;
    vector<int> bestColor = color;

    while (badEdges > 0 && elapsedSeconds(start) < timeLimitSec) {
        if (conflicted.empty()) break;

        int v = conflicted[rng() % conflicted.size()];
        int oldColor = color[v];

        touched.clear();
        for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
            int u = g.adj[i];
            int cu = color[u];
            if (cnt[cu] == 0) touched.push_back(cu);
            cnt[cu]++;
        }

        int oldConf = cnt[oldColor];
        int minConf = oldConf;
        vector<int> bestChoices;
        bestChoices.push_back(oldColor);

        for (int c = 0; c < g.k; ++c) {
            int conf = cnt[c];
            if (conf < minConf) {
                minConf = conf;
                bestChoices.clear();
                bestChoices.push_back(c);
            } else if (conf == minConf) {
                bestChoices.push_back(c);
            }
        }

        int newColor = oldColor;
        if (minConf < oldConf) {
            newColor = bestChoices[rng() % bestChoices.size()];
        } else if (g.k > 1 && prob01(rng) < 0.08) {
            newColor = static_cast<int>(rng() % g.k);
        }

        for (int c : touched) cnt[c] = 0;

        if (newColor == oldColor) continue;

        for (int i = g.offset[v]; i < g.offset[v + 1]; ++i) {
            int u = g.adj[i];
            if (color[u] == oldColor) {
                --conflictCount[v];
                --conflictCount[u];
                --badEdges;
                if (conflictCount[u] == 0) removeConflicted(u);
            }
            if (color[u] == newColor) {
                ++conflictCount[v];
                ++conflictCount[u];
                ++badEdges;
                if (conflictCount[u] == 1) addConflicted(u);
            }
        }

        color[v] = newColor;
        if (conflictCount[v] == 0) removeConflicted(v);
        else addConflicted(v);

        if (badEdges < bestBad) {
            bestBad = badEdges;
            bestColor = color;
            if (bestBad == 0) return true;
        }
    }

    if (bestBad < badEdges) {
        color = bestColor;
        badEdges = bestBad;
    }

    return badEdges == 0;
}

static bool dsaturBacktracking(const Graph& g, vector<int>& color,
                               const Clock::time_point& start, double timeLimitSec) {
    if (g.k <= 0) return g.n == 0;
    if (g.k > 63) return false;

    vector<uint64_t> satMask(g.n, 0ULL);
    vector<int> satDeg(g.n, 0);
    vector<char> uncolored(g.n, 1);
    color.assign(g.n, -1);

    const uint64_t allMask = (g.k == 64) ? ~0ULL : ((1ULL << g.k) - 1ULL);

    function<bool(int)> dfs = [&](int coloredCount) -> bool {
        if (elapsedSeconds(start) >= timeLimitSec) return false;
        if (coloredCount == g.n) return true;

        int pick = -1;
        int bestSat = -1;
        int bestDeg = -1;

        for (int v = 0; v < g.n; ++v) {
            if (!uncolored[v]) continue;
            if (satDeg[v] > bestSat || (satDeg[v] == bestSat && g.degree[v] > bestDeg)) {
                bestSat = satDeg[v];
                bestDeg = g.degree[v];
                pick = v;
            }
        }

        if (pick < 0) return false;

        uint64_t forbidden = 0ULL;
        for (int i = g.offset[pick]; i < g.offset[pick + 1]; ++i) {
            int u = g.adj[i];
            int cu = color[u];
            if (cu >= 0) forbidden |= (1ULL << cu);
        }
        uint64_t avail = allMask & ~forbidden;
        if (avail == 0ULL) return false;

        while (avail) {
            uint64_t bit = avail & (~avail + 1ULL);
            int c = firstSetBit64(bit);
            avail ^= bit;

            color[pick] = c;
            uncolored[pick] = 0;

            vector<int> changed;
            changed.reserve(g.degree[pick]);

            for (int i = g.offset[pick]; i < g.offset[pick + 1]; ++i) {
                int u = g.adj[i];
                if (!uncolored[u]) continue;
                uint64_t before = satMask[u];
                uint64_t after = before | (1ULL << c);
                if (after != before) {
                    satMask[u] = after;
                    satDeg[u]++;
                    changed.push_back(u);
                }
            }

            bool impossible = false;
            for (int u : changed) {
                if ((satMask[u] & allMask) == allMask) {
                    impossible = true;
                    break;
                }
            }

            if (!impossible && dfs(coloredCount + 1)) return true;

            for (int u : changed) {
                satMask[u] &= ~(1ULL << c);
                satDeg[u]--;
            }

            uncolored[pick] = 1;
            color[pick] = -1;
        }

        return false;
    };

    return dfs(0);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Graph g;
    bool hasSelfLoop = false;
    if (!readGraph(g, hasSelfLoop)) return 0;

    if (g.n == 0) {
        cout << '\n';
        return 0;
    }

    vector<int> color(g.n, 0);

    if (g.k <= 0 || hasSelfLoop) {
        for (int i = 0; i < g.n; ++i) {
            if (i) cout << ' ';
            cout << 0;
        }
        cout << '\n';
        return 0;
    }

    const Clock::time_point start = Clock::now();
    const double hardLimit = 295.0;

    random_device rd;
    mt19937 rng(rd() ^ static_cast<unsigned>(g.n * 911382323u + g.k * 972663749u));

    bool solved = false;

    if (g.n <= 220 && g.k <= 63) {
        solved = dsaturBacktracking(g, color, start, min(8.0, hardLimit));
    }

    vector<int> bestColor = color;
    long long bestBad = numeric_limits<long long>::max();

    auto attempt = [&](const vector<int>& order, double localBudgetSec) {
        vector<int> candidate;

        if (strictGreedyColor(g, order, candidate)) {
            long long bad = countConflicts(g, candidate);
            if (bad < bestBad) {
                bestBad = bad;
                bestColor = candidate;
            }
            if (bad == 0) {
                color = move(candidate);
                solved = true;
                return;
            }
        }

        greedyWithConflicts(g, order, candidate, rng);
        minConflictsRepair(g, candidate, rng, start, min(hardLimit, elapsedSeconds(start) + localBudgetSec));
        long long bad = countConflicts(g, candidate);
        if (bad < bestBad) {
            bestBad = bad;
            bestColor = candidate;
        }
        if (bad == 0) {
            color = move(candidate);
            solved = true;
        }
    };

    if (!solved) {
        vector<int> orderLF = largestFirstOrder(g);
        attempt(orderLF, 40.0);

        if (!solved && g.n <= 250000 && g.m <= 6000000 && elapsedSeconds(start) < 120.0) {
            vector<int> orderSL = smallestLastOrder(g);
            attempt(orderSL, 60.0);
        }

        int tries = (g.n < 200000) ? 6 : 3;
        for (int t = 0; t < tries && !solved; ++t) {
            if (elapsedSeconds(start) >= hardLimit) break;
            int swaps = max(1, g.n / 200);
            swaps = min(swaps, 30000);
            vector<int> p = perturbedOrder(orderLF, rng, swaps);
            double budget = (g.n < 200000) ? 40.0 : 25.0;
            attempt(p, budget);
        }

        if (!solved && bestBad != numeric_limits<long long>::max()) {
            color = bestColor;
            solved = (bestBad == 0);
        }
    }

    if (isProperColoring(g, color)) {
        tryUseMoreColors(g, color);
    } else {
        vector<int> order(g.n);
        iota(order.begin(), order.end(), 0);
        strictGreedyColor(g, order, color);
    }

    for (int i = 0; i < g.n; ++i) {
        if (i) cout << ' ';
        int c = color[i];
        if (c < 0 || c >= g.k) c = 0;
        cout << c;
    }
    cout << '\n';

    return 0;
}
