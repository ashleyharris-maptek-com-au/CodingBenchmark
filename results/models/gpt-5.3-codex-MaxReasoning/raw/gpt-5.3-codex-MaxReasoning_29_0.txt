using System;
using System.Buffers;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Numerics;
using System.Threading;
using System.Threading.Tasks;

public static class Program
{
    private const int MaxHeuristicSeeds = 96;
    private const int ExactKernelLimit = 2200;
    private static readonly TimeSpan TotalTimeLimit = TimeSpan.FromMinutes(4.9);

    public static void Main()
    {
        var stopwatch = Stopwatch.StartNew();
        var scanner = new FastScanner(Console.OpenStandardInput());

        if (!scanner.TryReadInt(out int n) || !scanner.TryReadInt(out int mInput))
        {
            return;
        }

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        if (mInput < 0)
        {
            mInput = 0;
        }

        int[] inputDegrees = new int[n];
        int[] us = new int[mInput];
        int[] vs = new int[mInput];
        int m = 0;

        for (int i = 0; i < mInput; i++)
        {
            if (!scanner.TryReadInt(out int u) || !scanner.TryReadInt(out int v))
            {
                break;
            }

            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v)
            {
                continue;
            }

            us[m] = u;
            vs[m] = v;
            m++;
            inputDegrees[u]++;
            inputDegrees[v]++;
        }

        Graph graph = Graph.Build(n, m, us, vs, inputDegrees);

        int[] core;
        int maxCore = ComputeCoreNumbers(graph, out core);
        int upperBound = maxCore + 1;

        int[] degreeOrder = new int[n];
        for (int i = 0; i < n; i++)
        {
            degreeOrder[i] = i;
        }

        Array.Sort(degreeOrder, (a, b) =>
        {
            int cmp = graph.Degree[b].CompareTo(graph.Degree[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        int[] bestClique = GreedyFromOrder(graph, degreeOrder);
        if (bestClique.Length == 0)
        {
            bestClique = new[] { degreeOrder[0] };
        }

        RunSeedHeuristicsParallel(graph, degreeOrder, stopwatch, TotalTimeLimit, ref bestClique);

        int bestSize = bestClique.Length;

        if (bestSize < upperBound && stopwatch.Elapsed < TotalTimeLimit)
        {
            int[] kernel = BuildKernelForLargerCliques(graph, bestSize);

            if (kernel.Length > 0 && kernel.Length <= ExactKernelLimit && stopwatch.Elapsed < TotalTimeLimit - TimeSpan.FromSeconds(2))
            {
                BuildKernelBitsetGraph(graph, kernel, out ulong[] bitAdj, out int[] kernelDegree);
                int[] mappedInitialClique = MapCliqueToKernel(bestClique, kernel, graph.N);

                var exact = new ExactCliqueSolver(kernel.Length, bitAdj, kernelDegree, mappedInitialClique, stopwatch, TotalTimeLimit);
                exact.Solve();

                if (exact.BestSize > bestSize)
                {
                    int[] mappedBest = exact.GetBestClique();
                    int[] lifted = new int[mappedBest.Length];
                    for (int i = 0; i < mappedBest.Length; i++)
                    {
                        lifted[i] = kernel[mappedBest[i]];
                    }

                    bestClique = lifted;
                    bestSize = bestClique.Length;
                }
            }
        }

        Array.Sort(bestClique);
        Console.WriteLine(bestClique.Length);
        Console.WriteLine(bestClique.Length == 0 ? string.Empty : string.Join(" ", bestClique));
    }

    private static int[] GreedyFromOrder(Graph g, int[] order)
    {
        var clique = new List<int>(Math.Min(64, g.N));

        for (int i = 0; i < order.Length; i++)
        {
            int v = order[i];
            bool ok = true;

            for (int j = 0; j < clique.Count; j++)
            {
                if (!g.IsAdjacent(v, clique[j]))
                {
                    ok = false;
                    break;
                }
            }

            if (ok)
            {
                clique.Add(v);
            }
        }

        return clique.ToArray();
    }

    private static void RunSeedHeuristicsParallel(Graph g, int[] degreeOrder, Stopwatch stopwatch, TimeSpan limit, ref int[] bestClique)
    {
        int seedCount = Math.Min(MaxHeuristicSeeds, g.N);
        if (seedCount <= 0)
        {
            return;
        }

        int[] seeds = new int[seedCount];
        Array.Copy(degreeOrder, seeds, seedCount);

        object bestLock = new object();
        int[] currentBest = bestClique;

        var markArrays = new ThreadLocal<int[]>(() => new int[g.N]);
        var stampValues = new ThreadLocal<int>(() => 1);

        Parallel.For(0, seedCount, new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, sidx =>
        {
            if (stopwatch.Elapsed >= limit)
            {
                return;
            }

            int seed = seeds[sidx];
            int[] marks = markArrays.Value!;
            int stamp = stampValues.Value;

            int[] clique = GreedyFromSeed(g, seed, marks, ref stamp);
            stampValues.Value = stamp;

            lock (bestLock)
            {
                if (clique.Length > currentBest.Length)
                {
                    currentBest = clique;
                }
            }
        });

        bestClique = currentBest;
    }

    private static int[] GreedyFromSeed(Graph g, int seed, int[] marks, ref int stamp)
    {
        var clique = new List<int>(32) { seed };

        int start = g.Offset[seed];
        int end = g.Offset[seed + 1];
        int candCount = end - start;
        if (candCount <= 0)
        {
            return clique.ToArray();
        }

        int[] candidates = new int[candCount];
        Array.Copy(g.Adj, start, candidates, 0, candCount);

        while (candCount > 0)
        {
            int bestIdx = 0;
            int bestScore = -1;

            for (int i = 0; i < candCount; i++)
            {
                int v = candidates[i];
                int score = g.Degree[v];
                if (score > bestScore)
                {
                    bestScore = score;
                    bestIdx = i;
                }
            }

            int chosen = candidates[bestIdx];
            clique.Add(chosen);

            stamp++;
            if (stamp == int.MaxValue)
            {
                Array.Clear(marks, 0, marks.Length);
                stamp = 1;
            }

            int nStart = g.Offset[chosen];
            int nEnd = g.Offset[chosen + 1];
            for (int p = nStart; p < nEnd; p++)
            {
                marks[g.Adj[p]] = stamp;
            }

            int nextCount = 0;
            for (int i = 0; i < candCount; i++)
            {
                int v = candidates[i];
                if (v != chosen && marks[v] == stamp)
                {
                    candidates[nextCount++] = v;
                }
            }

            candCount = nextCount;
        }

        return clique.ToArray();
    }

    private static int[] BuildKernelForLargerCliques(Graph g, int currentBestSize)
    {
        int threshold = currentBestSize;
        if (threshold <= 0)
        {
            int[] all = new int[g.N];
            for (int i = 0; i < g.N; i++) all[i] = i;
            return all;
        }

        int n = g.N;
        int[] degree = new int[n];
        Array.Copy(g.Degree, degree, n);
        bool[] active = new bool[n];
        Array.Fill(active, true);

        var q = new Queue<int>();
        for (int v = 0; v < n; v++)
        {
            if (degree[v] < threshold)
            {
                active[v] = false;
                q.Enqueue(v);
            }
        }

        while (q.Count > 0)
        {
            int v = q.Dequeue();
            int start = g.Offset[v];
            int end = g.Offset[v + 1];

            for (int p = start; p < end; p++)
            {
                int u = g.Adj[p];
                if (!active[u]) continue;

                degree[u]--;
                if (degree[u] < threshold)
                {
                    active[u] = false;
                    q.Enqueue(u);
                }
            }
        }

        int count = 0;
        for (int i = 0; i < n; i++)
        {
            if (active[i]) count++;
        }

        int[] kernel = new int[count];
        int idx = 0;
        for (int i = 0; i < n; i++)
        {
            if (active[i]) kernel[idx++] = i;
        }

        return kernel;
    }

    private static void BuildKernelBitsetGraph(Graph g, int[] kernel, out ulong[] bitAdj, out int[] kernelDegree)
    {
        int k = kernel.Length;
        int words = (k + 63) >> 6;

        bitAdj = new ulong[k * words];
        kernelDegree = new int[k];

        int[] map = new int[g.N];
        Array.Fill(map, -1);
        for (int i = 0; i < k; i++)
        {
            map[kernel[i]] = i;
        }

        for (int i = 0; i < k; i++)
        {
            int old = kernel[i];
            int row = i * words;
            int start = g.Offset[old];
            int end = g.Offset[old + 1];
            int deg = 0;

            for (int p = start; p < end; p++)
            {
                int mapped = map[g.Adj[p]];
                if (mapped >= 0 && mapped != i)
                {
                    bitAdj[row + (mapped >> 6)] |= 1UL << (mapped & 63);
                    deg++;
                }
            }

            kernelDegree[i] = deg;
        }
    }

    private static int[] MapCliqueToKernel(int[] clique, int[] kernel, int originalN)
    {
        int[] map = new int[originalN];
        Array.Fill(map, -1);
        for (int i = 0; i < kernel.Length; i++)
        {
            map[kernel[i]] = i;
        }

        var mapped = new List<int>(clique.Length);
        for (int i = 0; i < clique.Length; i++)
        {
            int m = map[clique[i]];
            if (m >= 0)
            {
                mapped.Add(m);
            }
        }

        return mapped.ToArray();
    }

    private static int ComputeCoreNumbers(Graph g, out int[] core)
    {
        int n = g.N;
        int[] deg = new int[n];
        Array.Copy(g.Degree, deg, n);

        int maxDeg = 0;
        for (int i = 0; i < n; i++)
        {
            if (deg[i] > maxDeg) maxDeg = deg[i];
        }

        int[] bin = new int[maxDeg + 1];
        for (int i = 0; i < n; i++)
        {
            bin[deg[i]]++;
        }

        int start = 0;
        for (int d = 0; d <= maxDeg; d++)
        {
            int num = bin[d];
            bin[d] = start;
            start += num;
        }

        int[] vert = new int[n];
        int[] pos = new int[n];

        for (int v = 0; v < n; v++)
        {
            int d = deg[v];
            int p = bin[d]++;
            pos[v] = p;
            vert[p] = v;
        }

        for (int d = maxDeg; d >= 1; d--)
        {
            bin[d] = bin[d - 1];
        }
        bin[0] = 0;

        int maxCore = 0;
        for (int i = 0; i < n; i++)
        {
            int v = vert[i];
            int dv = deg[v];
            if (dv > maxCore) maxCore = dv;

            int s = g.Offset[v];
            int e = g.Offset[v + 1];
            for (int p = s; p < e; p++)
            {
                int u = g.Adj[p];
                if (deg[u] > dv)
                {
                    int du = deg[u];
                    int pu = pos[u];
                    int pw = bin[du];
                    int w = vert[pw];

                    if (u != w)
                    {
                        vert[pu] = w;
                        vert[pw] = u;
                        pos[u] = pw;
                        pos[w] = pu;
                    }

                    bin[du]++;
                    deg[u]--;
                }
            }
        }

        core = deg;
        return maxCore;
    }

    private sealed class Graph
    {
        public int N { get; }
        public int[] Offset { get; }
        public int[] Adj { get; }
        public int[] Degree { get; }

        private Graph(int n, int[] offset, int[] adj, int[] degree)
        {
            N = n;
            Offset = offset;
            Adj = adj;
            Degree = degree;
        }

        public static Graph Build(int n, int m, int[] us, int[] vs, int[] degree)
        {
            int[] offset = new int[n + 1];
            for (int i = 0; i < n; i++)
            {
                offset[i + 1] = offset[i] + degree[i];
            }

            int[] adj = new int[offset[n]];
            int[] cursor = new int[n];
            Array.Copy(offset, cursor, n);

            for (int i = 0; i < m; i++)
            {
                int u = us[i];
                int v = vs[i];
                adj[cursor[u]++] = v;
                adj[cursor[v]++] = u;
            }

            if (n >= 10000)
            {
                Parallel.For(0, n, i =>
                {
                    int s = offset[i];
                    int len = offset[i + 1] - s;
                    if (len > 1)
                    {
                        Array.Sort(adj, s, len);
                    }
                });
            }
            else
            {
                for (int i = 0; i < n; i++)
                {
                    int s = offset[i];
                    int len = offset[i + 1] - s;
                    if (len > 1)
                    {
                        Array.Sort(adj, s, len);
                    }
                }
            }

            return new Graph(n, offset, adj, degree);
        }

        public bool IsAdjacent(int u, int v)
        {
            int su = Offset[u];
            int eu = Offset[u + 1];
            int sv = Offset[v];
            int ev = Offset[v + 1];

            if (eu - su > ev - sv)
            {
                int t = u; u = v; v = t;
                su = Offset[u];
                eu = Offset[u + 1];
            }

            int lo = su;
            int hi = eu - 1;
            while (lo <= hi)
            {
                int mid = lo + ((hi - lo) >> 1);
                int x = Adj[mid];
                if (x == v) return true;
                if (x < v) lo = mid + 1;
                else hi = mid - 1;
            }

            return false;
        }
    }

    private sealed class ExactCliqueSolver
    {
        private readonly int n;
        private readonly int words;
        private readonly ulong[] adj;
        private readonly int[] degree;
        private readonly Stopwatch stopwatch;
        private readonly long deadlineTicks;

        private bool timedOut;
        private long nodes;
        private int bestSize;
        private int[] bestClique;
        private readonly int[] cliqueStack;

        public bool TimedOut => timedOut;
        public int BestSize => bestSize;

        public ExactCliqueSolver(int n, ulong[] adj, int[] degree, int[] initialClique, Stopwatch stopwatch, TimeSpan totalLimit)
        {
            this.n = n;
            this.adj = adj;
            this.degree = degree;
            this.words = (n + 63) >> 6;
            this.stopwatch = stopwatch;
            deadlineTicks = (long)(totalLimit.TotalSeconds * Stopwatch.Frequency);

            bestSize = initialClique.Length;
            bestClique = new int[Math.Max(1, n)];
            for (int i = 0; i < initialClique.Length; i++)
            {
                bestClique[i] = initialClique[i];
            }

            cliqueStack = new int[Math.Max(1, n)];
        }

        public int[] GetBestClique()
        {
            int[] outClique = new int[bestSize];
            Array.Copy(bestClique, outClique, bestSize);
            return outClique;
        }

        public void Solve()
        {
            int[] candidates = new int[n];
            for (int i = 0; i < n; i++)
            {
                candidates[i] = i;
            }

            Array.Sort(candidates, (a, b) =>
            {
                int cmp = degree[b].CompareTo(degree[a]);
                return cmp != 0 ? cmp : a.CompareTo(b);
            });

            Expand(candidates, n, 0);
        }

        private void Expand(int[] candidates, int count, int depth)
        {
            if (timedOut)
            {
                return;
            }

            if (count == 0)
            {
                if (depth > bestSize)
                {
                    bestSize = depth;
                    Array.Copy(cliqueStack, bestClique, depth);
                }
                return;
            }

            if ((++nodes & 2047) == 0 && stopwatch.ElapsedTicks >= deadlineTicks)
            {
                timedOut = true;
                return;
            }

            int[] ordered = ArrayPool<int>.Shared.Rent(count);
            int[] bounds = ArrayPool<int>.Shared.Rent(count);
            try
            {
                ColorSort(candidates, count, ordered, bounds);

                for (int i = count - 1; i >= 0; i--)
                {
                    if (depth + bounds[i] <= bestSize)
                    {
                        break;
                    }

                    int v = ordered[i];
                    cliqueStack[depth] = v;

                    int[] next = ArrayPool<int>.Shared.Rent(i == 0 ? 1 : i);
                    int nextCount = 0;
                    for (int j = 0; j < i; j++)
                    {
                        int u = ordered[j];
                        if (HasEdge(v, u))
                        {
                            next[nextCount++] = u;
                        }
                    }

                    if (nextCount == 0)
                    {
                        int newSize = depth + 1;
                        if (newSize > bestSize)
                        {
                            bestSize = newSize;
                            Array.Copy(cliqueStack, bestClique, newSize);
                        }
                    }
                    else
                    {
                        Expand(next, nextCount, depth + 1);
                    }

                    ArrayPool<int>.Shared.Return(next);

                    if (timedOut)
                    {
                        return;
                    }
                }
            }
            finally
            {
                ArrayPool<int>.Shared.Return(ordered);
                ArrayPool<int>.Shared.Return(bounds);
            }
        }

        private void ColorSort(int[] candidates, int count, int[] ordered, int[] bounds)
        {
            int[] rem = ArrayPool<int>.Shared.Rent(count);
            int[] colorClass = ArrayPool<int>.Shared.Rent(count);
            try
            {
                Array.Copy(candidates, rem, count);
                int remCount = count;
                int outIdx = 0;
                int color = 0;

                while (remCount > 0)
                {
                    color++;
                    int classCount = 0;

                    for (int i = 0; i < remCount; i++)
                    {
                        int v = rem[i];
                        bool ok = true;
                        for (int j = 0; j < classCount; j++)
                        {
                            if (HasEdge(v, colorClass[j]))
                            {
                                ok = false;
                                break;
                            }
                        }

                        if (ok)
                        {
                            colorClass[classCount++] = v;
                            ordered[outIdx] = v;
                            bounds[outIdx] = color;
                            outIdx++;

                            rem[i] = rem[remCount - 1];
                            remCount--;
                            i--;
                        }
                    }
                }
            }
            finally
            {
                ArrayPool<int>.Shared.Return(rem);
                ArrayPool<int>.Shared.Return(colorClass);
            }
        }

        private bool HasEdge(int a, int b)
        {
            int idx = a * words + (b >> 6);
            ulong mask = 1UL << (b & 63);
            return (adj[idx] & mask) != 0;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer;
        private int length;
        private int ptr;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
            buffer = new byte[1 << 16];
        }

        public bool TryReadInt(out int value)
        {
            value = 0;
            int c;

            do
            {
                c = Read();
                if (c == -1) return false;
            }
            while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
                if (c == -1)
                {
                    value = 0;
                    return true;
                }
            }

            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = Read();
                if (c == -1) break;
            }

            value = x * sign;
            return true;
        }

        private int Read()
        {
            if (ptr >= length)
            {
                length = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (length == 0)
                {
                    return -1;
                }
            }

            return buffer[ptr++];
        }
    }
