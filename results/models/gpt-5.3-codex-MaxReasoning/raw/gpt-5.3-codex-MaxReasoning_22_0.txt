using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Numerics;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        if (!scanner.TryNextInt(out int width))
        {
            return;
        }

        if (!scanner.TryNextInt(out int height) || !scanner.TryNextInt(out int numPieces))
        {
            return;
        }

        var board = new Board(width, height);
        var ai = new TetrisAi();

        for (int i = 0; i < numPieces; i++)
        {
            string pieceToken = scanner.NextToken();
            if (pieceToken == null)
            {
                break;
            }

            char pieceType = char.ToUpperInvariant(pieceToken[0]);
            Move move = ai.ChooseMove(board, pieceType, numPieces - i);

            Console.WriteLine(move.Rotation + " " + move.Column);
            Console.Out.Flush();

            if (board.TryPlace(pieceType, move.Rotation, move.Column, out Board nextBoard, out _))
            {
                board = nextBoard;
            }

            string resultToken = scanner.NextToken();
            if (resultToken == null)
            {
                break;
            }

            if (resultToken.StartsWith("gameover", StringComparison.OrdinalIgnoreCase))
            {
                break;
            }
        }
    }
}

public sealed class TetrisAi
{
    private static readonly char[] PieceOrder = { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };

    public Move ChooseMove(Board board, char pieceType, int piecesRemaining)
    {
        List<Orientation> orientations = PieceLibrary.Get(pieceType);
        var strategy = Strategy.For(board, piecesRemaining);
        long startTicks = Stopwatch.GetTimestamp();

        var candidates = new List<Candidate>(48);

        for (int oi = 0; oi < orientations.Count; oi++)
        {
            Orientation orientation = orientations[oi];
            int maxCol = board.Width - orientation.Width;

            for (int col = 0; col <= maxCol; col++)
            {
                if (!board.TryPlace(orientation, col, out Board nextBoard, out int cleared))
                {
                    continue;
                }

                double immediateScore = Evaluate(nextBoard, cleared, strategy);
                candidates.Add(new Candidate(new Move(orientation.RotationIndex, col), immediateScore, nextBoard));
            }
        }

        if (candidates.Count == 0)
        {
            return new Move(0, 0);
        }

        candidates.Sort((a, b) => b.Score.CompareTo(a.Score));

        if (strategy.LookaheadBeam <= 0 || strategy.FutureSamples <= 0)
        {
            return candidates[0].Move;
        }

        int beam = Math.Min(strategy.LookaheadBeam, candidates.Count);
        Move bestMove = candidates[0].Move;
        double bestTotal = double.NegativeInfinity;

        for (int i = 0; i < beam; i++)
        {
            if (ElapsedMs(startTicks) > strategy.TimeBudgetMs)
            {
                break;
            }

            Candidate candidate = candidates[i];
            double future = ExpectedFutureValue(candidate.Board, strategy, startTicks);
            double total = candidate.Score + strategy.FutureWeight * future;

            if (total > bestTotal)
            {
                bestTotal = total;
                bestMove = candidate.Move;
            }
        }

        return bestMove;
    }

    private double ExpectedFutureValue(Board board, Strategy strategy, long startTicks)
    {
        int samples = Math.Min(strategy.FutureSamples, PieceOrder.Length);
        double sum = 0.0;

        for (int pi = 0; pi < samples; pi++)
        {
            if (ElapsedMs(startTicks) > strategy.TimeBudgetMs)
            {
                break;
            }

            char piece = PieceOrder[pi];
            List<Orientation> orientations = PieceLibrary.Get(piece);
            double best = double.NegativeInfinity;

            for (int oi = 0; oi < orientations.Count; oi++)
            {
                Orientation orientation = orientations[oi];
                int maxCol = board.Width - orientation.Width;

                for (int col = 0; col <= maxCol; col++)
                {
                    if (!board.TryPlace(orientation, col, out Board nextBoard, out int cleared))
                    {
                        continue;
                    }

                    double score = Evaluate(nextBoard, cleared, strategy);
                    if (score > best)
                    {
                        best = score;
                    }
                }
            }

            if (double.IsNegativeInfinity(best))
            {
                best = -1_000_000.0;
            }

            sum += best;
        }

        return sum / samples;
    }

    private static double Evaluate(Board board, int cleared, Strategy s)
    {
        Metrics m = board.ComputeMetrics(s.UseTransitions);

        double score =
            cleared * s.LineWeight +
            m.AggregateHeight * s.AggregateHeightWeight +
            m.Holes * s.HoleWeight +
            m.Bumpiness * s.BumpinessWeight +
            m.MaxHeight * s.MaxHeightWeight +
            m.Wells * s.WellWeight;

        if (s.UseTransitions)
        {
            score += m.RowTransitions * s.RowTransitionWeight;
            score += m.ColumnTransitions * s.ColumnTransitionWeight;
        }

        if (s.PreferRightWell && board.Width >= 2)
        {
            int wellCol = board.Width - 1;
            int neighbor = board.ColumnHeights[board.Width - 2];
            int wellDepth = Math.Max(0, neighbor - board.ColumnHeights[wellCol]);
            score += s.RightWellWeight * wellDepth;
            score -= 0.10 * board.ColumnHeights[wellCol];
        }

        return score;
    }

    private static double ElapsedMs(long startTicks)
    {
        long dt = Stopwatch.GetTimestamp() - startTicks;
        return (1000.0 * dt) / Stopwatch.Frequency;
    }

    private readonly struct Candidate
    {
        public Candidate(Move move, double score, Board board)
        {
            Move = move;
            Score = score;
            Board = board;
        }

        public Move Move { get; }
        public double Score { get; }
        public Board Board { get; }
    }
}

public readonly struct Strategy
{
    public readonly double LineWeight;
    public readonly double AggregateHeightWeight;
    public readonly double HoleWeight;
    public readonly double BumpinessWeight;
    public readonly double MaxHeightWeight;
    public readonly double WellWeight;
    public readonly double RowTransitionWeight;
    public readonly double ColumnTransitionWeight;
    public readonly double FutureWeight;
    public readonly double RightWellWeight;
    public readonly int LookaheadBeam;
    public readonly int FutureSamples;
    public readonly int TimeBudgetMs;
    public readonly bool UseTransitions;
    public readonly bool PreferRightWell;

    private Strategy(
        double lineWeight,
        double aggregateHeightWeight,
        double holeWeight,
        double bumpinessWeight,
        double maxHeightWeight,
        double wellWeight,
        double rowTransitionWeight,
        double columnTransitionWeight,
        double futureWeight,
        double rightWellWeight,
        int lookaheadBeam,
        int futureSamples,
        int timeBudgetMs,
        bool useTransitions,
        bool preferRightWell)
    {
        LineWeight = lineWeight;
        AggregateHeightWeight = aggregateHeightWeight;
        HoleWeight = holeWeight;
        BumpinessWeight = bumpinessWeight;
        MaxHeightWeight = maxHeightWeight;
        WellWeight = wellWeight;
        RowTransitionWeight = rowTransitionWeight;
        ColumnTransitionWeight = columnTransitionWeight;
        FutureWeight = futureWeight;
        RightWellWeight = rightWellWeight;
        LookaheadBeam = lookaheadBeam;
        FutureSamples = futureSamples;
        TimeBudgetMs = timeBudgetMs;
        UseTransitions = useTransitions;
        PreferRightWell = preferRightWell;
    }

    public static Strategy For(Board board, int piecesRemaining)
    {
        double danger = board.MaxHeight / (double)Math.Max(1, board.Height);
        double fill = board.FilledCells / (double)Math.Max(1, board.Width * board.Height);
        bool wideBoard = board.Width >= 14;

        if (danger >= 0.78 || fill >= 0.58)
        {
            return new Strategy(
                lineWeight: 2.10,
                aggregateHeightWeight: -0.62,
                holeWeight: -3.80,
                bumpinessWeight: -0.42,
                maxHeightWeight: -2.20,
                wellWeight: -0.20,
                rowTransitionWeight: -0.55,
                columnTransitionWeight: -0.50,
                futureWeight: 0.0,
                rightWellWeight: 0.0,
                lookaheadBeam: 0,
                futureSamples: 0,
                timeBudgetMs: 25,
                useTransitions: true,
                preferRightWell: false);
        }

        if (wideBoard || piecesRemaining > 600)
        {
            return new Strategy(
                lineWeight: 1.90,
                aggregateHeightWeight: -0.40,
                holeWeight: -2.70,
                bumpinessWeight: -0.30,
                maxHeightWeight: -1.20,
                wellWeight: -0.06,
                rowTransitionWeight: -0.30,
                columnTransitionWeight: -0.25,
                futureWeight: 0.12,
                rightWellWeight: 0.0,
                lookaheadBeam: 2,
                futureSamples: 4,
                timeBudgetMs: 40,
                useTransitions: false,
                preferRightWell: false);
        }

        bool lowDanger = danger < 0.45;
        bool shortRun = piecesRemaining < 180;

        if (lowDanger && shortRun)
        {
            return new Strategy(
                lineWeight: 2.80,
                aggregateHeightWeight: -0.28,
                holeWeight: -2.25,
                bumpinessWeight: -0.24,
                maxHeightWeight: -0.90,
                wellWeight: 0.02,
                rowTransitionWeight: -0.30,
                columnTransitionWeight: -0.25,
                futureWeight: 0.28,
                rightWellWeight: 0.18,
                lookaheadBeam: 6,
                futureSamples: 7,
                timeBudgetMs: 120,
                useTransitions: false,
                preferRightWell: board.Width >= 10);
        }

        return new Strategy(
            lineWeight: 2.20,
            aggregateHeightWeight: -0.35,
            holeWeight: -2.50,
            bumpinessWeight: -0.28,
            maxHeightWeight: -1.05,
            wellWeight: -0.05,
            rowTransitionWeight: -0.45,
            columnTransitionWeight: -0.35,
            futureWeight: 0.18,
            rightWellWeight: 0.08,
            lookaheadBeam: 4,
            futureSamples: 6,
            timeBudgetMs: 80,
            useTransitions: true,
            preferRightWell: board.Width >= 10 && danger < 0.55);
    }
}

public sealed class Board
{
    public int Width { get; }
    public int Height { get; }
    public int FilledCells { get; }
    public int[] ColumnHeights { get; }
    public int MaxHeight { get; }

    private ulong FullMask { get; }
    private ulong[] Rows { get; }

    public Board(int width, int height)
    {
        Width = Math.Max(1, width);
        Height = Math.Max(1, height);

        if (Width >= 63)
        {
            throw new ArgumentException("Board width must be <= 62 for bitboard representation.");
        }

        FullMask = (1UL << Width) - 1UL;
        Rows = new ulong[Height];
        ColumnHeights = new int[Width];
        FilledCells = 0;
        MaxHeight = 0;
    }

    private Board(int width, int height, ulong fullMask, ulong[] rows, int[] columnHeights, int filledCells)
    {
        Width = width;
        Height = height;
        FullMask = fullMask;
        Rows = rows;
        ColumnHeights = columnHeights;
        FilledCells = filledCells;

        int max = 0;
        for (int i = 0; i < columnHeights.Length; i++)
        {
            if (columnHeights[i] > max)
            {
                max = columnHeights[i];
            }
        }

        MaxHeight = max;
    }

    public bool TryPlace(char pieceType, int rotation, int column, out Board next, out int cleared)
    {
        Orientation? orientation = PieceLibrary.GetByRotation(pieceType, rotation);
        if (orientation == null)
        {
            next = this;
            cleared = 0;
            return false;
        }

        return TryPlace(orientation, column, out next, out cleared);
    }

    public bool TryPlace(Orientation orientation, int column, out Board next, out int cleared)
    {
        next = this;
        cleared = 0;

        if (column < 0 || column + orientation.Width > Width)
        {
            return false;
        }

        int landingY = 0;
        for (int lx = 0; lx < orientation.Width; lx++)
        {
            int boardHeight = ColumnHeights[column + lx];
            int y = boardHeight - orientation.BottomOffsets[lx];
            if (y > landingY)
            {
                landingY = y;
            }
        }

        if (landingY + orientation.Height > Height)
        {
            return false;
        }

        ulong[] newRows = (ulong[])Rows.Clone();

        for (int i = 0; i < orientation.Blocks.Length; i++)
        {
            Block b = orientation.Blocks[i];
            int gx = column + b.X;
            int gy = landingY + b.Y;

            if ((uint)gy >= (uint)Height)
            {
                return false;
            }

            ulong bit = 1UL << gx;
            if ((newRows[gy] & bit) != 0)
            {
                return false;
            }

            newRows[gy] |= bit;
        }

        int write = 0;
        for (int read = 0; read < Height; read++)
        {
            if (newRows[read] == FullMask)
            {
                cleared++;
            }
            else
            {
                newRows[write++] = newRows[read];
            }
        }

        while (write < Height)
        {
            newRows[write++] = 0UL;
        }

        int[] heights = RecomputeHeights(newRows);
        int newFilled = FilledCells + 4 - cleared * Width;
        next = new Board(Width, Height, FullMask, newRows, heights, newFilled);
        return true;
    }

    public Metrics ComputeMetrics(bool includeTransitions)
    {
        int aggregateHeight = 0;
        int holes = 0;
        int bumpiness = 0;
        int wells = 0;

        for (int x = 0; x < Width; x++)
        {
            int h = ColumnHeights[x];
            aggregateHeight += h;

            bool seenBlock = false;
            for (int y = Height - 1; y >= 0; y--)
            {
                bool occupied = ((Rows[y] >> x) & 1UL) != 0;
                if (occupied)
                {
                    seenBlock = true;
                }
                else if (seenBlock)
                {
                    holes++;
                }
            }

            if (x + 1 < Width)
            {
                bumpiness += Math.Abs(ColumnHeights[x] - ColumnHeights[x + 1]);
            }

            int left = x == 0 ? Height : ColumnHeights[x - 1];
            int right = x == Width - 1 ? Height : ColumnHeights[x + 1];
            int minNeighbor = Math.Min(left, right);
            if (minNeighbor > h)
            {
                int depth = minNeighbor - h;
                wells += depth * (depth + 1) / 2;
            }
        }

        int rowTransitions = 0;
        int colTransitions = 0;

        if (includeTransitions)
        {
            for (int y = 0; y < Height; y++)
            {
                bool prev = true;
                for (int x = 0; x < Width; x++)
                {
                    bool occupied = ((Rows[y] >> x) & 1UL) != 0;
                    if (occupied != prev)
                    {
                        rowTransitions++;
                    }
                    prev = occupied;
                }
                if (!prev)
                {
                    rowTransitions++;
                }
            }

            for (int x = 0; x < Width; x++)
            {
                bool prev = true;
                for (int y = Height - 1; y >= 0; y--)
                {
                    bool occupied = ((Rows[y] >> x) & 1UL) != 0;
                    if (occupied != prev)
                    {
                        colTransitions++;
                    }
                    prev = occupied;
                }
                if (!prev)
                {
                    colTransitions++;
                }
            }
        }

        return new Metrics(
            aggregateHeight,
            holes,
            bumpiness,
            MaxHeight,
            wells,
            rowTransitions,
            colTransitions);
    }

    private int[] RecomputeHeights(ulong[] rows)
    {
        int[] heights = new int[Width];

        for (int x = 0; x < Width; x++)
        {
            int h = 0;
            for (int y = Height - 1; y >= 0; y--)
            {
                if (((rows[y] >> x) & 1UL) != 0)
                {
                    h = y + 1;
                    break;
                }
            }
            heights[x] = h;
        }

        return heights;
    }
}

public readonly struct Metrics
{
    public Metrics(int aggregateHeight, int holes, int bumpiness, int maxHeight, int wells, int rowTransitions, int columnTransitions)
    {
        AggregateHeight = aggregateHeight;
        Holes = holes;
        Bumpiness = bumpiness;
        MaxHeight = maxHeight;
        Wells = wells;
        RowTransitions = rowTransitions;
        ColumnTransitions = columnTransitions;
    }

    public int AggregateHeight { get; }
    public int Holes { get; }
    public int Bumpiness { get; }
    public int MaxHeight { get; }
    public int Wells { get; }
    public int RowTransitions { get; }
    public int ColumnTransitions { get; }
}

public readonly struct Block
{
    public Block(int x, int y)
    {
        X = x;
        Y = y;
    }

    public int X { get; }
    public int Y { get; }
}

public sealed class Orientation
{
    public Orientation(int rotationIndex, Block[] blocks)
    {
        RotationIndex = rotationIndex;
        Blocks = blocks;

        int maxX = 0;
        int maxY = 0;
        for (int i = 0; i < blocks.Length; i++)
        {
            if (blocks[i].X > maxX) maxX = blocks[i].X;
            if (blocks[i].Y > maxY) maxY = blocks[i].Y;
        }

        Width = maxX + 1;
        Height = maxY + 1;
        BottomOffsets = new int[Width];
        for (int i = 0; i < Width; i++)
        {
            BottomOffsets[i] = int.MaxValue;
        }

        for (int i = 0; i < blocks.Length; i++)
        {
            Block b = blocks[i];
            if (b.Y < BottomOffsets[b.X])
            {
                BottomOffsets[b.X] = b.Y;
            }
        }
    }

    public int RotationIndex { get; }
    public Block[] Blocks { get; }
    public int Width { get; }
    public int Height { get; }
    public int[] BottomOffsets { get; }
}

public static class PieceLibrary
{
    private static readonly Dictionary<char, List<Orientation>> All = new Dictionary<char, List<Orientation>>();
    private static readonly Dictionary<char, Dictionary<int, Orientation>> ByRotation = new Dictionary<char, Dictionary<int, Orientation>>();

    static PieceLibrary()
    {
        Add('I', new[] { new Block(0, 0), new Block(1, 0), new Block(2, 0), new Block(3, 0) });
        Add('O', new[] { new Block(0, 0), new Block(1, 0), new Block(0, 1), new Block(1, 1) });
        Add('T', new[] { new Block(0, 0), new Block(1, 0), new Block(2, 0), new Block(1, 1) });
        Add('S', new[] { new Block(1, 0), new Block(2, 0), new Block(0, 1), new Block(1, 1) });
        Add('Z', new[] { new Block(0, 0), new Block(1, 0), new Block(1, 1), new Block(2, 1) });
        Add('J', new[] { new Block(0, 0), new Block(1, 0), new Block(2, 0), new Block(0, 1) });
        Add('L', new[] { new Block(0, 0), new Block(1, 0), new Block(2, 0), new Block(2, 1) });
    }

    public static List<Orientation> Get(char piece)
    {
        char key = char.ToUpperInvariant(piece);
        if (!All.TryGetValue(key, out List<Orientation>? value))
        {
            return All['I'];
        }

        return value;
    }

    public static Orientation? GetByRotation(char piece, int rotation)
    {
        char key = char.ToUpperInvariant(piece);
        if (!ByRotation.TryGetValue(key, out Dictionary<int, Orientation>? map))
        {
            return null;
        }

        map.TryGetValue((rotation % 4 + 4) % 4, out Orientation? orientation);
        return orientation;
    }

    private static void Add(char piece, Block[] baseBlocks)
    {
        var list = new List<Orientation>(4);
        var map = new Dictionary<int, Orientation>(4);
        var seen = new HashSet<string>(StringComparer.Ordinal);

        Block[] current = (Block[])baseBlocks.Clone();

        for (int rotation = 0; rotation < 4; rotation++)
        {
            Block[] normalized = Normalize(current);
            string sig = Signature(normalized);

            if (seen.Add(sig))
            {
                var orientation = new Orientation(rotation, normalized);
                list.Add(orientation);
                map[rotation] = orientation;
            }

            current = RotateCW(current);
        }

        All[piece] = list;
        ByRotation[piece] = map;
    }

    private static Block[] RotateCW(Block[] blocks)
    {
        Block[] outBlocks = new Block[blocks.Length];
        for (int i = 0; i < blocks.Length; i++)
        {
            // (x, y) -> (y, -x)
            outBlocks[i] = new Block(blocks[i].Y, -blocks[i].X);
        }
        return outBlocks;
    }

    private static Block[] Normalize(Block[] blocks)
    {
        int minX = int.MaxValue;
        int minY = int.MaxValue;

        for (int i = 0; i < blocks.Length; i++)
        {
            if (blocks[i].X < minX) minX = blocks[i].X;
            if (blocks[i].Y < minY) minY = blocks[i].Y;
        }

        Block[] normalized = new Block[blocks.Length];
        for (int i = 0; i < blocks.Length; i++)
        {
            normalized[i] = new Block(blocks[i].X - minX, blocks[i].Y - minY);
        }

        Array.Sort(normalized, (a, b) =>
        {
            int cx = a.X.CompareTo(b.X);
            return cx != 0 ? cx : a.Y.CompareTo(b.Y);
        });

        return normalized;
    }

    private static string Signature(Block[] blocks)
    {
        var chars = new char[blocks.Length * 6];
        int p = 0;

        for (int i = 0; i < blocks.Length; i++)
        {
            int x = blocks[i].X;
            int y = blocks[i].Y;

            chars[p++] = (char)('0' + x);
            chars[p++] = ',';
            chars[p++] = (char)('0' + y);
            chars[p++] = ';';
        }

        return new string(chars, 0, p);
    }
}

public sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _len;
    private int _ptr;

    public FastScanner(Stream stream)
    {
        _stream = stream;
    }

    public string NextToken()
    {
        int c;
        do
        {
            c = Read();
            if (c < 0)
            {
                return null;
            }
        } while (c <= 32);

        var chars = new char[32];
        int n = 0;

        while (c > 32)
        {
            if (n == chars.Length)
            {
                Array.Resize(ref chars, chars.Length * 2);
            }

            chars[n++] = (char)c;
            c = Read();
        }

        return new string(chars, 0, n);
    }

    public bool TryNextInt(out int value)
    {
        value = 0;
        string token = NextToken();
        if (token == null)
        {
            return false;
        }

        return int.TryParse(token, out value);
    }

    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len == 0)
            {
                return -1;
            }
        }

        return _buffer[_ptr++];
    }
}
