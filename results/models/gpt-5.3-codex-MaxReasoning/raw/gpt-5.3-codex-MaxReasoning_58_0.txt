const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const grid = Array.isArray(data.grid) ? data.grid : [];
const robotsInput = Array.isArray(data.robots) ? data.robots : [];
const jobsInput = Array.isArray(data.jobs) ? data.jobs : [];

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const cellCount = width * height;

const blocked = new Uint8Array(cellCount);
for (let y = 0; y < height; y++) {
  const row = String(grid[y] ?? '');
  for (let x = 0; x < width; x++) {
    const c = row[x] || '#';
    if (c === '#') blocked[y * width + x] = 1;
  }
}

function isInBounds(x, y) {
  return x >= 0 && y >= 0 && x < width && y < height;
}

function toIndex(x, y) {
  return y * width + x;
}

function compareIds(a, b) {
  if (typeof a === 'number' && typeof b === 'number') return a - b;
  const sa = String(a);
  const sb = String(b);
  if (sa < sb) return -1;
  if (sa > sb) return 1;
  return 0;
}

const distCache = new Map();

function bfsFrom(srcIdx) {
  const dist = new Int32Array(cellCount);
  dist.fill(-1);

  if (cellCount === 0 || srcIdx < 0 || srcIdx >= cellCount || blocked[srcIdx]) {
    return dist;
  }

  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;

  dist[srcIdx] = 0;
  queue[tail++] = srcIdx;

  while (head < tail) {
    const cur = queue[head++];
    const baseDist = dist[cur] + 1;

    const x = cur % width;

    let nb;

    if (x > 0) {
      nb = cur - 1;
      if (!blocked[nb] && dist[nb] === -1) {
        dist[nb] = baseDist;
        queue[tail++] = nb;
      }
    }

    if (x + 1 < width) {
      nb = cur + 1;
      if (!blocked[nb] && dist[nb] === -1) {
        dist[nb] = baseDist;
        queue[tail++] = nb;
      }
    }

    if (cur >= width) {
      nb = cur - width;
      if (!blocked[nb] && dist[nb] === -1) {
        dist[nb] = baseDist;
        queue[tail++] = nb;
      }
    }

    if (cur + width < cellCount) {
      nb = cur + width;
      if (!blocked[nb] && dist[nb] === -1) {
        dist[nb] = baseDist;
        queue[tail++] = nb;
      }
    }
  }

  return dist;
}

function getDistance(srcIdx, dstIdx) {
  let distArr = distCache.get(srcIdx);
  if (!distArr) {
    distArr = bfsFrom(srcIdx);
    distCache.set(srcIdx, distArr);
  }
  if (dstIdx < 0 || dstIdx >= cellCount) return -1;
  return distArr[dstIdx];
}

const robots = [];
for (let i = 0; i < robotsInput.length; i++) {
  const r = robotsInput[i] || {};
  const id = Number(r.id);
  const x = Number(r.x);
  const y = Number(r.y);
  if (!Number.isInteger(id) || !Number.isInteger(x) || !Number.isInteger(y)) continue;

  const idx = isInBounds(x, y) ? toIndex(x, y) : -1;
  robots.push({ id, x, y, posIdx: idx, travel: 0 });
}

robots.sort((a, b) => compareIds(a.id, b.id));

let maxRobotId = -1;
for (let i = 0; i < robots.length; i++) {
  if (robots[i].id > maxRobotId) maxRobotId = robots[i].id;
}

const assignments = Array.from({ length: maxRobotId + 1 }, () => []);

const jobs = [];
for (let i = 0; i < jobsInput.length; i++) {
  const j = jobsInput[i] || {};
  const rawX = Number(j.x);
  const rawY = Number(j.y);
  const x = Number.isInteger(rawX) ? rawX : 0;
  const y = Number.isInteger(rawY) ? rawY : 0;

  const idx = Number.isInteger(rawX) && Number.isInteger(rawY) && isInBounds(x, y) ? toIndex(x, y) : -1;
  jobs.push({
    id: j.id,
    x,
    y,
    idx,
    priority: Number(j.priority) || 0,
    order: i,
  });
}

jobs.sort((a, b) => {
  if (a.priority !== b.priority) return b.priority - a.priority;
  const idCmp = compareIds(a.id, b.id);
  if (idCmp !== 0) return idCmp;
  return a.order - b.order;
});

if (robots.length > 0) {
  for (let ji = 0; ji < jobs.length; ji++) {
    const job = jobs[ji];

    let best = null;
    let bestDist = Infinity;

    for (let ri = 0; ri < robots.length; ri++) {
      const robot = robots[ri];
      const d = getDistance(robot.posIdx, job.idx);
      if (d < 0) continue;

      if (
        d < bestDist ||
        (d === bestDist && (best === null || robot.travel < best.travel)) ||
        (d === bestDist && best !== null && robot.travel === best.travel && compareIds(robot.id, best.id) < 0)
      ) {
        best = robot;
        bestDist = d;
      }
    }

    if (best === null) {
      let bestFallback = null;
      let bestManhattan = Infinity;
      for (let ri = 0; ri < robots.length; ri++) {
        const robot = robots[ri];
        const manhattan = Math.abs(robot.x - job.x) + Math.abs(robot.y - job.y);
        if (
          manhattan < bestManhattan ||
          (manhattan === bestManhattan && (bestFallback === null || robot.travel < bestFallback.travel)) ||
          (manhattan === bestManhattan && bestFallback !== null && robot.travel === bestFallback.travel && compareIds(robot.id, bestFallback.id) < 0)
        ) {
          bestFallback = robot;
          bestManhattan = manhattan;
        }
      }
      best = bestFallback;
      bestDist = -1;
    }

    if (best) {
      if (best.id >= 0 && best.id < assignments.length) {
        assignments[best.id].push(job.id);
      }
      if (bestDist > 0) best.travel += bestDist;
      best.x = job.x;
      best.y = job.y;
      best.posIdx = job.idx;
    }
  }
}

const output = { assignments };
process.stdout.write(JSON.stringify(output));
