declare const require: any;
declare const process: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data: any = JSON.parse(input);

type Town = { id?: string | number; x: number; y: number; value: number };

const width: number = Number.isFinite(data?.width) ? (data.width | 0) : 0;
const height: number = Number.isFinite(data?.height) ? (data.height | 0) : 0;
const grid: string[] = Array.isArray(data?.grid) ? data.grid : [];
const ignitionsIn: any[] = Array.isArray(data?.ignitions) ? data.ignitions : [];
const townsIn: Town[] = Array.isArray(data?.towns) ? data.towns : [];
const cutBudgetRaw: number = Number.isFinite(data?.cutBudget) ? Number(data.cutBudget) : 0;
const cutBudget: number = Math.max(0, Math.floor(cutBudgetRaw));

function outputFirebreaks(cells: number[][]): void {
  process.stdout.write(JSON.stringify({ firebreaks: cells }));
}

if (width <= 0 || height <= 0 || grid.length < height) {
  outputFirebreaks([]);
  process.exit(0);
}

const n: number = width * height;
const burnable: Uint8Array = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row: string = typeof grid[y] === 'string' ? grid[y] : '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) {
      burnable[y * width + x] = 1;
    }
  }
}

const ignitionMark: Uint8Array = new Uint8Array(n);
const ignitionList: number[] = [];
for (const it of ignitionsIn) {
  if (!Array.isArray(it) || it.length < 2) continue;
  const x: number = it[0] | 0;
  const y: number = it[1] | 0;
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  const idx: number = y * width + x;
  if (!burnable[idx]) continue;
  if (!ignitionMark[idx]) {
    ignitionMark[idx] = 1;
    ignitionList.push(idx);
  }
}

if (ignitionList.length === 0 || cutBudget === 0) {
  outputFirebreaks([]);
  process.exit(0);
}

const hasTown: Uint8Array = new Uint8Array(n);
for (const t of townsIn) {
  if (!t) continue;
  const x: number = t.x | 0;
  const y: number = t.y | 0;
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  const v: number = Number(t.value);
  if (!(v > 0)) continue;
  hasTown[y * width + x] = 1;
}

// Baseline reachability from ignitions without cuts.
const dist: Int32Array = new Int32Array(n);
for (let i = 0; i < n; i++) dist[i] = -1;
const q0: Int32Array = new Int32Array(n);
let h0 = 0;
let t0 = 0;
for (const s of ignitionList) {
  if (dist[s] === -1) {
    dist[s] = 0;
    q0[t0++] = s;
  }
}
while (h0 < t0) {
  const cur: number = q0[h0++];
  const d: number = dist[cur] + 1;
  const x: number = cur % width;

  let nb: number;
  if (x > 0) {
    nb = cur - 1;
    if (burnable[nb] && dist[nb] === -1) {
      dist[nb] = d;
      q0[t0++] = nb;
    }
  }
  if (x + 1 < width) {
    nb = cur + 1;
    if (burnable[nb] && dist[nb] === -1) {
      dist[nb] = d;
      q0[t0++] = nb;
    }
  }
  if (cur >= width) {
    nb = cur - width;
    if (burnable[nb] && dist[nb] === -1) {
      dist[nb] = d;
      q0[t0++] = nb;
    }
  }
  if (cur + width < n) {
    nb = cur + width;
    if (burnable[nb] && dist[nb] === -1) {
      dist[nb] = d;
      q0[t0++] = nb;
    }
  }
}

const atRiskTownValue: Float64Array = new Float64Array(n);
let totalAtRisk = 0;
for (const t of townsIn) {
  if (!t) continue;
  const x: number = t.x | 0;
  const y: number = t.y | 0;
  if (x < 0 || y < 0 || x >= width || y >= height) continue;
  const v: number = Number(t.value);
  if (!(v > 0)) continue;
  const idx: number = y * width + x;
  if (burnable[idx] && dist[idx] >= 0) {
    atRiskTownValue[idx] += v;
    totalAtRisk += v;
  }
}

if (!(totalAtRisk > 0)) {
  outputFirebreaks([]);
  process.exit(0);
}

// Cells that can reach at least one at-risk town.
const canReachTown: Uint8Array = new Uint8Array(n);
const q1: Int32Array = new Int32Array(n);
let h1 = 0;
let t1 = 0;
for (let i = 0; i < n; i++) {
  if (atRiskTownValue[i] > 0) {
    canReachTown[i] = 1;
    q1[t1++] = i;
  }
}
while (h1 < t1) {
  const cur: number = q1[h1++];
  const x: number = cur % width;
  let nb: number;

  if (x > 0) {
    nb = cur - 1;
    if (burnable[nb] && !canReachTown[nb]) {
      canReachTown[nb] = 1;
      q1[t1++] = nb;
    }
  }
  if (x + 1 < width) {
    nb = cur + 1;
    if (burnable[nb] && !canReachTown[nb]) {
      canReachTown[nb] = 1;
      q1[t1++] = nb;
    }
  }
  if (cur >= width) {
    nb = cur - width;
    if (burnable[nb] && !canReachTown[nb]) {
      canReachTown[nb] = 1;
      q1[t1++] = nb;
    }
  }
  if (cur + width < n) {
    nb = cur + width;
    if (burnable[nb] && !canReachTown[nb]) {
      canReachTown[nb] = 1;
      q1[t1++] = nb;
    }
  }
}

// Active cells: potentially relevant to saving at-risk towns.
const localOfGlobal: Int32Array = new Int32Array(n);
for (let i = 0; i < n; i++) localOfGlobal[i] = -1;
const globalOfLocal: number[] = [];
for (let i = 0; i < n; i++) {
  if (burnable[i] && dist[i] >= 0 && canReachTown[i]) {
    localOfGlobal[i] = globalOfLocal.length;
    globalOfLocal.push(i);
  }
}
const m: number = globalOfLocal.length;
if (m === 0) {
  outputFirebreaks([]);
  process.exit(0);
}

const cuttableLocal: Uint8Array = new Uint8Array(m);
const ignitionLocal: Uint8Array = new Uint8Array(m);
const townValueLocal: Float64Array = new Float64Array(m);
const cuttableGlobal: Uint8Array = new Uint8Array(n);
let cuttableCount = 0;
let ignitionActiveCount = 0;
let townEdgeCount = 0;
for (let u = 0; u < m; u++) {
  const g: number = globalOfLocal[u];
  const isIgn: number = ignitionMark[g];
  ignitionLocal[u] = isIgn;
  if (isIgn) ignitionActiveCount++;
  const tv: number = atRiskTownValue[g];
  townValueLocal[u] = tv;
  if (tv > 0) townEdgeCount++;
  if (!isIgn && !hasTown[g]) {
    cuttableLocal[u] = 1;
    cuttableGlobal[g] = 1;
    cuttableCount++;
  }
}

if (cuttableCount === 0) {
  outputFirebreaks([]);
  process.exit(0);
}

const pairU: number[] = [];
const pairV: number[] = [];
for (let u = 0; u < m; u++) {
  const g: number = globalOfLocal[u];
  const x: number = g % width;
  const y: number = (g / width) | 0;

  if (x + 1 < width) {
    const v: number = localOfGlobal[g + 1];
    if (v !== -1) {
      pairU.push(u);
      pairV.push(v);
    }
  }
  if (y + 1 < height) {
    const v: number = localOfGlobal[g + width];
    if (v !== -1) {
      pairU.push(u);
      pairV.push(v);
    }
  }
}

const activeSorted: number[] = globalOfLocal.slice();
activeSorted.sort((a: number, b: number) => dist[b] - dist[a]);
const potential: Float64Array = new Float64Array(n);
for (let i = 0; i < n; i++) {
  potential[i] = atRiskTownValue[i];
}
for (const g of activeSorted) {
  const val: number = potential[g];
  if (!(val > 0)) continue;
  const d: number = dist[g];
  if (d <= 0) continue;

  const x: number = g % width;
  let p0 = -1;
  let p1 = -1;
  let p2 = -1;
  let p3 = -1;
  let c = 0;

  let nb: number;
  if (x > 0) {
    nb = g - 1;
    if (localOfGlobal[nb] !== -1 && dist[nb] === d - 1) {
      if (c === 0) p0 = nb;
      else if (c === 1) p1 = nb;
      else if (c === 2) p2 = nb;
      else p3 = nb;
      c++;
    }
  }
  if (x + 1 < width) {
    nb = g + 1;
    if (localOfGlobal[nb] !== -1 && dist[nb] === d - 1) {
      if (c === 0) p0 = nb;
      else if (c === 1) p1 = nb;
      else if (c === 2) p2 = nb;
      else p3 = nb;
      c++;
    }
  }
  if (g >= width) {
    nb = g - width;
    if (localOfGlobal[nb] !== -1 && dist[nb] === d - 1) {
      if (c === 0) p0 = nb;
      else if (c === 1) p1 = nb;
      else if (c === 2) p2 = nb;
      else p3 = nb;
      c++;
    }
  }
  if (g + width < n) {
    nb = g + width;
    if (localOfGlobal[nb] !== -1 && dist[nb] === d - 1) {
      if (c === 0) p0 = nb;
      else if (c === 1) p1 = nb;
      else if (c === 2) p2 = nb;
      else p3 = nb;
      c++;
    }
  }

  if (c > 0) {
    const share: number = val / c;
    if (p0 >= 0) potential[p0] += share;
    if (p1 >= 0) potential[p1] += share;
    if (p2 >= 0) potential[p2] += share;
    if (p3 >= 0) potential[p3] += share;
  }
}

class Dinic {
  private n: number;
  private head: Int32Array;
  private to: Int32Array;
  private next: Int32Array;
  private cap: Float64Array;
  private level: Int32Array;
  private it: Int32Array;
  private queue: Int32Array;
  private vStack: Int32Array;
  private eStack: Int32Array;
  private minStack: Float64Array;
  private edgeCount: number;

  constructor(n: number, maxDirectedEdges: number) {
    this.n = n;
    this.head = new Int32Array(n);
    this.head.fill(-1);
    this.to = new Int32Array(maxDirectedEdges);
    this.next = new Int32Array(maxDirectedEdges);
    this.cap = new Float64Array(maxDirectedEdges);
    this.level = new Int32Array(n);
    this.it = new Int32Array(n);
    this.queue = new Int32Array(n);
    this.vStack = new Int32Array(n);
    this.eStack = new Int32Array(n);
    this.minStack = new Float64Array(n);
    this.edgeCount = 0;
  }

  addEdge(u: number, v: number, c: number): number {
    let e: number = this.edgeCount;
    this.to[e] = v;
    this.cap[e] = c;
    this.next[e] = this.head[u];
    this.head[u] = e;
    e++;

    this.to[e] = u;
    this.cap[e] = 0;
    this.next[e] = this.head[v];
    this.head[v] = e;
    e++;

    this.edgeCount = e;
    return e - 2;
  }

  private bfs(s: number, t: number, eps: number): boolean {
    this.level.fill(-1);
    let h = 0;
    let q = 0;
    this.level[s] = 0;
    this.queue[q++] = s;

    while (h < q) {
      const v: number = this.queue[h++];
      for (let e = this.head[v]; e !== -1; e = this.next[e]) {
        if (this.cap[e] <= eps) continue;
        const u: number = this.to[e];
        if (this.level[u] !== -1) continue;
        this.level[u] = this.level[v] + 1;
        if (u === t) return true;
        this.queue[q++] = u;
      }
    }

    return this.level[t] !== -1;
  }

  private sendOne(s: number, t: number, inf: number, eps: number): number {
    let top = 0;
    this.vStack[0] = s;
    this.minStack[0] = inf;

    while (true) {
      const v: number = this.vStack[top];
      if (v === t) {
        const pushed: number = this.minStack[top];
        for (let i = 0; i < top; i++) {
          const e: number = this.eStack[i];
          this.cap[e] -= pushed;
          this.cap[e ^ 1] += pushed;
        }
        return pushed;
      }

      let e: number = this.it[v];
      while (e !== -1) {
        const u: number = this.to[e];
        if (this.cap[e] > eps && this.level[u] === this.level[v] + 1) break;
        e = this.next[e];
      }
      this.it[v] = e;

      if (e === -1) {
        if (top === 0) return 0;
        top--;
        continue;
      }

      this.it[v] = this.next[e];
      const u: number = this.to[e];
      this.eStack[top] = e;
      const pm: number = this.minStack[top];
      const ce: number = this.cap[e];
      this.minStack[top + 1] = ce < pm ? ce : pm;
      this.vStack[top + 1] = u;
      top++;
    }
  }

  maxFlow(s: number, t: number): number {
    const eps = 1e-9;
    let flow = 0;
    while (this.bfs(s, t, eps)) {
      this.it.set(this.head);
      while (true) {
        const f: number = this.sendOne(s, t, Number.POSITIVE_INFINITY, eps);
        if (f <= eps) break;
        flow += f;
      }
    }
    return flow;
  }

  minCutReachable(s: number): Uint8Array {
    const eps = 1e-9;
    const seen: Uint8Array = new Uint8Array(this.n);
    let h = 0;
    let q = 0;
    seen[s] = 1;
    this.queue[q++] = s;

    while (h < q) {
      const v: number = this.queue[h++];
      for (let e = this.head[v]; e !== -1; e = this.next[e]) {
        if (this.cap[e] <= eps) continue;
        const u: number = this.to[e];
        if (seen[u]) continue;
        seen[u] = 1;
        this.queue[q++] = u;
      }
    }

    return seen;
  }
}

const maxAddEdges: number = m + pairU.length * 2 + ignitionActiveCount + townEdgeCount;
const maxDirectedEdges: number = maxAddEdges * 2 + 8;

function solveForLambda(lambda: number): number[] {
  let infCap: number = totalAtRisk + lambda * (cuttableCount + 1) + 1;
  if (!Number.isFinite(infCap) || infCap > 1e15) infCap = 1e15;

  const nodeCount: number = m * 2 + 2;
  const source: number = nodeCount - 2;
  const sink: number = nodeCount - 1;

  const dinic: Dinic = new Dinic(nodeCount, maxDirectedEdges);

  for (let u = 0; u < m; u++) {
    const inNode: number = u * 2;
    const outNode: number = inNode + 1;
    dinic.addEdge(inNode, outNode, cuttableLocal[u] ? lambda : infCap);

    if (ignitionLocal[u]) {
      dinic.addEdge(source, inNode, infCap);
    }

    const tv: number = townValueLocal[u];
    if (tv > 0) {
      dinic.addEdge(outNode, sink, tv);
    }
  }

  for (let i = 0; i < pairU.length; i++) {
    const u: number = pairU[i];
    const v: number = pairV[i];
    dinic.addEdge(u * 2 + 1, v * 2, infCap);
    dinic.addEdge(v * 2 + 1, u * 2, infCap);
  }

  dinic.maxFlow(source, sink);
  const reach: Uint8Array = dinic.minCutReachable(source);

  const cuts: number[] = [];
  for (let u = 0; u < m; u++) {
    if (!cuttableLocal[u]) continue;
    const inNode: number = u * 2;
    const outNode: number = inNode + 1;
    if (reach[inNode] && !reach[outNode]) {
      cuts.push(globalOfLocal[u]);
    }
  }
  return cuts;
}

const visitedMark: Int32Array = new Int32Array(n);
const blockedMark: Int32Array = new Int32Array(n);
const burnQueue: Int32Array = new Int32Array(n);
let visitToken = 1;
let blockToken = 1;

function nextVisitToken(): number {
  visitToken++;
  if (visitToken === 0x7fffffff) {
    visitedMark.fill(0);
    visitToken = 1;
  }
  return visitToken;
}

function nextBlockToken(): number {
  blockToken++;
  if (blockToken === 0x7fffffff) {
    blockedMark.fill(0);
    blockToken = 1;
  }
  return blockToken;
}

type EvalResult = { saved: number; burned: number; reachCount: number };

function evaluateCuts(cuts: number[]): EvalResult {
  const vTok: number = nextVisitToken();
  const bTok: number = nextBlockToken();

  for (const c of cuts) blockedMark[c] = bTok;

  let burned = 0;
  let h = 0;
  let q = 0;

  for (const s of ignitionList) {
    if (!burnable[s]) continue;
    if (blockedMark[s] === bTok) continue;
    if (visitedMark[s] === vTok) continue;
    visitedMark[s] = vTok;
    burnQueue[q++] = s;
    burned += atRiskTownValue[s];
  }

  while (h < q) {
    const cur: number = burnQueue[h++];
    const x: number = cur % width;
    let nb: number;

    if (x > 0) {
      nb = cur - 1;
      if (burnable[nb] && blockedMark[nb] !== bTok && visitedMark[nb] !== vTok) {
        visitedMark[nb] = vTok;
        burnQueue[q++] = nb;
        burned += atRiskTownValue[nb];
      }
    }
    if (x + 1 < width) {
      nb = cur + 1;
      if (burnable[nb] && blockedMark[nb] !== bTok && visitedMark[nb] !== vTok) {
        visitedMark[nb] = vTok;
        burnQueue[q++] = nb;
        burned += atRiskTownValue[nb];
      }
    }
    if (cur >= width) {
      nb = cur - width;
      if (burnable[nb] && blockedMark[nb] !== bTok && visitedMark[nb] !== vTok) {
        visitedMark[nb] = vTok;
        burnQueue[q++] = nb;
        burned += atRiskTownValue[nb];
      }
    }
    if (cur + width < n) {
      nb = cur + width;
      if (burnable[nb] && blockedMark[nb] !== bTok && visitedMark[nb] !== vTok) {
        visitedMark[nb] = vTok;
        burnQueue[q++] = nb;
        burned += atRiskTownValue[nb];
      }
    }
  }

  return { saved: totalAtRisk - burned, burned, reachCount: q };
}

const lambdaCache: Map<string, number[]> = new Map();
function lambdaKey(x: number): string {
  return x.toFixed(6);
}
function getCutsForLambda(lambda: number): number[] {
  const key: string = lambdaKey(lambda);
  const cached: number[] | undefined = lambdaCache.get(key);
  if (cached) return cached;
  const cuts: number[] = solveForLambda(lambda);
  lambdaCache.set(key, cuts);
  return cuts;
}

let bestCuts: number[] = [];
let bestSaved = -1;

function considerLambda(lambda: number): number {
  const cuts: number[] = getCutsForLambda(lambda);
  const cnt: number = cuts.length;
  if (cnt <= cutBudget) {
    const ev: EvalResult = evaluateCuts(cuts);
    const saved: number = ev.saved;
    if (saved > bestSaved + 1e-9 || (Math.abs(saved - bestSaved) <= 1e-9 && cnt > bestCuts.length)) {
      bestSaved = saved;
      bestCuts = cuts.slice();
    }
  }
  return cnt;
}

let lo = 0;
let hi = totalAtRisk + 1;
const iters: number = m > 60000 ? 5 : (m > 25000 ? 6 : 8);
for (let i = 0; i < iters; i++) {
  const mid: number = (lo + hi) * 0.5;
  const cnt: number = considerLambda(mid);
  if (cnt > cutBudget) lo = mid;
  else hi = mid;
}

considerLambda(hi);
considerLambda((lo + hi) * 0.5);
considerLambda(totalAtRisk / Math.max(1, cutBudget));
if (cutBudget >= cuttableCount) considerLambda(0);

if (bestCuts.length > cutBudget) {
  bestCuts = bestCuts.slice(0, cutBudget);
}

const chosen: Uint8Array = new Uint8Array(n);
for (const c of bestCuts) chosen[c] = 1;

let currentCuts: number[] = bestCuts.slice();
let currentEval: EvalResult = evaluateCuts(currentCuts);
const maxAugment: number = Math.min(cutBudget - currentCuts.length, 30);

for (let step = 0; step < maxAugment; step++) {
  const poolLimit = 64;
  const evalLimit = 12;
  const poolCells: number[] = [];
  const poolScores: number[] = [];
  let minPos = -1;

  // Use cells currently reachable by fire as candidate area.
  for (let i = 0; i < currentEval.reachCount; i++) {
    const g: number = burnQueue[i];
    if (!cuttableGlobal[g] || chosen[g]) continue;
    const s: number = potential[g];
    if (!(s > 0)) continue;

    if (poolCells.length < poolLimit) {
      poolCells.push(g);
      poolScores.push(s);
      if (minPos === -1 || s < poolScores[minPos]) minPos = poolScores.length - 1;
    } else if (s > poolScores[minPos]) {
      poolCells[minPos] = g;
      poolScores[minPos] = s;
      minPos = 0;
      for (let j = 1; j < poolScores.length; j++) {
        if (poolScores[j] < poolScores[minPos]) minPos = j;
      }
    }
  }

  if (poolCells.length === 0) break;

  const order: number[] = poolCells.map((_, i) => i);
  order.sort((a: number, b: number) => poolScores[b] - poolScores[a]);

  let bestCand = -1;
  let bestCandSaved = currentEval.saved;

  const tries = Math.min(evalLimit, order.length);
  for (let i = 0; i < tries; i++) {
    const g: number = poolCells[order[i]];
    currentCuts.push(g);
    const ev: EvalResult = evaluateCuts(currentCuts);
    currentCuts.pop();

    if (ev.saved > bestCandSaved + 1e-9) {
      bestCandSaved = ev.saved;
      bestCand = g;
    }
  }

  if (bestCand === -1) break;

  chosen[bestCand] = 1;
  currentCuts.push(bestCand);
  currentEval = evaluateCuts(currentCuts);
}

if (currentCuts.length > cutBudget) {
  currentCuts = currentCuts.slice(0, cutBudget);
}

currentCuts.sort((a: number, b: number) => a - b);
const firebreaks: number[][] = [];
for (const idx of currentCuts) {
  const x: number = idx % width;
  const y: number = (idx / width) | 0;
  firebreaks.push([x, y]);
}

outputFirebreaks(firebreaks);
