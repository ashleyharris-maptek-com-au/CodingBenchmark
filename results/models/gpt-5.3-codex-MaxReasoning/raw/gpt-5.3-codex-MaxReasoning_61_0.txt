declare const require: any;
const fs = require('fs');

interface InputPackage {
  id: unknown;
  x: number;
  y: number;
}

interface InputData {
  depot: [number, number];
  truckCount: number;
  packages: InputPackage[];
}

interface Point {
  id: unknown;
  x: number;
  y: number;
  angle: number;
  radius2: number;
}

interface BoundaryDelta {
  index: number;
  delta: number;
}

function num(v: unknown): number {
  const n = Number(v);
  return Number.isFinite(n) ? n : 0;
}

function dist(x1: number, y1: number, x2: number, y2: number): number {
  return Math.hypot(x1 - x2, y1 - y2);
}

function reverseRange<T>(arr: T[], lo: number, hi: number): void {
  let i = lo;
  let j = hi;
  while (i < j) {
    const tmp = arr[i];
    arr[i] = arr[j];
    arr[j] = tmp;
    i++;
    j--;
  }
}

function improveRoute2Opt(route: Point[], depotX: number, depotY: number): void {
  const m = route.length;
  if (m < 4) return;

  const window = m > 4000 ? 6 : m > 2000 ? 10 : m > 800 ? 16 : 28;
  const maxPasses = m > 3000 ? 1 : m > 1200 ? 2 : 3;

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;

    for (let i = 0; i < m - 1; i++) {
      const aX = i === 0 ? depotX : route[i - 1].x;
      const aY = i === 0 ? depotY : route[i - 1].y;
      const b = route[i];

      const maxJ = Math.min(m - 1, i + window);
      for (let j = i + 1; j <= maxJ; j++) {
        const c = route[j];
        const dX = j === m - 1 ? depotX : route[j + 1].x;
        const dY = j === m - 1 ? depotY : route[j + 1].y;

        const before = dist(aX, aY, b.x, b.y) + dist(c.x, c.y, dX, dY);
        const after = dist(aX, aY, c.x, c.y) + dist(b.x, b.y, dX, dY);

        if (after + 1e-12 < before) {
          reverseRange(route, i, j);
          improved = true;
          break;
        }
      }
    }

    if (!improved) break;
  }
}

function solve(data: InputData): { routes: unknown[][] } {
  const depotX = num(data.depot?.[0]);
  const depotY = num(data.depot?.[1]);

  const rawPackages = Array.isArray(data.packages) ? data.packages : [];
  const n = rawPackages.length;
  if (n === 0) return { routes: [] };

  const requestedTrucks = Math.floor(num(data.truckCount));
  const routeBudget = Math.max(1, Math.min(n, requestedTrucks > 0 ? requestedTrucks : 1));

  const points: Point[] = rawPackages.map((p) => {
    const x = num(p.x);
    const y = num(p.y);
    const dx = x - depotX;
    const dy = y - depotY;
    return {
      id: p.id,
      x,
      y,
      angle: Math.atan2(dy, dx),
      radius2: dx * dx + dy * dy,
    };
  });

  points.sort((a, b) => {
    if (a.angle !== b.angle) return a.angle - b.angle;
    if (a.radius2 !== b.radius2) return a.radius2 - b.radius2;
    const as = String(a.id);
    const bs = String(b.id);
    if (as < bs) return -1;
    if (as > bs) return 1;
    return 0;
  });

  const deltas: BoundaryDelta[] = [];
  for (let i = 0; i < n; i++) {
    const j = (i + 1) % n;
    const a = points[i];
    const b = points[j];
    const delta = dist(depotX, depotY, a.x, a.y) + dist(depotX, depotY, b.x, b.y) - dist(a.x, a.y, b.x, b.y);
    deltas.push({ index: i, delta });
  }

  deltas.sort((a, b) => {
    if (a.delta !== b.delta) return a.delta - b.delta;
    return a.index - b.index;
  });

  const negativeCount = deltas.findIndex((d) => d.delta >= -1e-12);
  const usableNegative = negativeCount === -1 ? deltas.length : negativeCount;

  let cutCount: number;
  if (usableNegative > 0) {
    cutCount = Math.min(routeBudget, usableNegative);
  } else {
    cutCount = 1;
  }

  const chosen = deltas.slice(0, cutCount);
  const cutBoundary = new Array<boolean>(n).fill(false);
  for (const c of chosen) cutBoundary[c.index] = true;

  let startBoundary = chosen[0].index;
  for (let i = 1; i < chosen.length; i++) {
    if (chosen[i].index < startBoundary) startBoundary = chosen[i].index;
  }

  const routes: Point[][] = [];
  let current: Point[] = [];
  let idx = (startBoundary + 1) % n;

  for (let visited = 0; visited < n; visited++) {
    current.push(points[idx]);

    if (cutBoundary[idx]) {
      routes.push(current);
      current = [];
    }

    idx = (idx + 1) % n;
  }

  if (current.length > 0) routes.push(current);

  for (const route of routes) {
    improveRoute2Opt(route, depotX, depotY);
  }

  return {
    routes: routes.filter((r) => r.length > 0).map((r) => r.map((p) => p.id)),
  };
}

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input) as InputData;
const result = solve(data);
fs.writeFileSync(1, JSON.stringify(result));
