; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inColor %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450
               OpName %main "main"
               OpName %inWorldPos "worldPos"
               OpName %inNormal "normal"
               OpName %inColor "color"
               OpName %outColor "outColor"
               OpName %UBO "UBO"
               OpMemberName %UBO 0 "model"
               OpMemberName %UBO 1 "view"
               OpMemberName %UBO 2 "projection"
               OpMemberName %UBO 3 "lightPos"
               OpMemberName %UBO 4 "cameraPos"
               OpMemberName %UBO 5 "params"
               OpName %uboVar "ubo"
               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %inColor Location 4
               OpDecorate %outColor Location 0
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %uboVar DescriptorSet 0
               OpDecorate %uboVar Binding 0
       %void = OpTypeVoid
         %fn = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4float 4
        %int = OpTypeInt 32 1
        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%float_0 = OpConstant %float 0
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_1 = OpConstant %float 1
%float_32 = OpConstant %float 32
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
   %ambientC = OpConstantComposite %v3float %float_0_1 %float_0_05 %float_0_05
     %whiteC = OpConstantComposite %v3float %float_1 %float_1 %float_1
   %v3_zeroC = OpConstantComposite %v3float %float_0 %float_0 %float_0
    %v3_oneC = OpConstantComposite %v3float %float_1 %float_1 %float_1
 %inWorldPos = OpVariable %_ptr_Input_v3float Input
   %inNormal = OpVariable %_ptr_Input_v3float Input
    %inColor = OpVariable %_ptr_Input_v3float Input
   %outColor = OpVariable %_ptr_Output_v4float Output
     %uboVar = OpVariable %_ptr_Uniform_UBO Uniform
       %main = OpFunction %void None %fn
      %entry = OpLabel
     %worldP = OpLoad %v3float %inWorldPos
    %normal0 = OpLoad %v3float %inNormal
          %N = OpExtInst %v3float %1 Normalize %normal0
   %lightPtr = OpAccessChain %_ptr_Uniform_v4float %uboVar %int_3
     %light4 = OpLoad %v4float %lightPtr
     %light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
       %Lraw = OpFSub %v3float %light3 %worldP
          %L = OpExtInst %v3float %1 Normalize %Lraw
  %cameraPtr = OpAccessChain %_ptr_Uniform_v4float %uboVar %int_4
       %cam4 = OpLoad %v4float %cameraPtr
       %cam3 = OpVectorShuffle %v3float %cam4 %cam4 0 1 2
       %Vraw = OpFSub %v3float %cam3 %worldP
          %V = OpExtInst %v3float %1 Normalize %Vraw
      %NdotL = OpDot %float %N %L
  %NdotLPos = OpExtInst %float %1 FMax %NdotL %float_0
      %albedo = OpLoad %v3float %inColor
    %diffuse = OpVectorTimesScalar %v3float %albedo %NdotLPos
       %negL = OpFNegate %v3float %L
          %R = OpExtInst %v3float %1 Reflect %negL %N
      %RdotV = OpDot %float %R %V
  %RdotVPos = OpExtInst %float %1 FMax %RdotV %float_0
  %specTerm = OpExtInst %float %1 Pow %RdotVPos %float_32
   %specular = OpVectorTimesScalar %v3float %whiteC %specTerm
       %sum0 = OpFAdd %v3float %ambientC %diffuse
       %sum1 = OpFAdd %v3float %sum0 %specular
  %clamped3 = OpExtInst %v3float %1 FClamp %sum1 %v3_zeroC %v3_oneC
         %cx = OpCompositeExtract %float %clamped3 0
         %cy = OpCompositeExtract %float %clamped3 1
         %cz = OpCompositeExtract %float %clamped3 2
   %outVec4 = OpCompositeConstruct %v4float %cx %cy %cz %float_1
               OpStore %outColor %outVec4
               OpReturn
               OpFunctionEnd
