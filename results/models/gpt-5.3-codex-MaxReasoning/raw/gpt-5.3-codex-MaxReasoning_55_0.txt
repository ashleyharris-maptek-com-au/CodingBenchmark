namespace {

constexpr int EMPTY = 0;
constexpr int PAWN = 1;
constexpr int KNIGHT = 2;
constexpr int BISHOP = 3;
constexpr int ROOK = 4;
constexpr int QUEEN = 5;
constexpr int KING = 6;

constexpr int WK_CASTLE = 1;
constexpr int WQ_CASTLE = 2;
constexpr int BK_CASTLE = 4;
constexpr int BQ_CASTLE = 8;

constexpr int CAPTURE = 1;
constexpr int EN_PASSANT = 2;
constexpr int CASTLE = 4;

constexpr int INF = 1'000'000'000;
constexpr int MATE_SCORE = 30'000;
constexpr int MAX_PLY = 128;

struct Move {
    int from = -1;
    int to = -1;
    int promo = 0;
    int flags = 0;
};

struct Undo {
    int captured = 0;
    int castlingRights = 0;
    int enPassant = -1;
    int halfmoveClock = 0;
    int movedPiece = 0;
    int whiteKingSq = 4;
    int blackKingSq = 60;
};

struct Position {
    std::array<int, 64> board{};
    bool whiteToMove = true;
    int castlingRights = 0;
    int enPassant = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
    int whiteKingSq = 4;
    int blackKingSq = 60;
};

inline int sq(int rank, int file) {
    return rank * 8 + file;
}

inline int fileOf(int s) {
    return s & 7;
}

inline int rankOf(int s) {
    return s >> 3;
}

inline int pieceType(int p) {
    return p < 0 ? -p : p;
}

inline int colorIndex(bool white) {
    return white ? 0 : 1;
}

int pieceValue(int type) {
    switch (type) {
        case PAWN:
            return 100;
        case KNIGHT:
            return 320;
        case BISHOP:
            return 330;
        case ROOK:
            return 500;
        case QUEEN:
            return 900;
        default:
            return 0;
    }
}

bool sameMove(const Move& a, const Move& b) {
    return a.from == b.from && a.to == b.to && a.promo == b.promo && a.flags == b.flags;
}

std::string trim(const std::string& s) {
    const std::string ws = " \t\r\n";
    const size_t start = s.find_first_not_of(ws);
    if (start == std::string::npos) {
        return "";
    }
    const size_t end = s.find_last_not_of(ws);
    return s.substr(start, end - start + 1);
}

int squareFromString(const std::string& s) {
    if (s.size() != 2) {
        return -1;
    }
    const char fileChar = static_cast<char>(std::tolower(static_cast<unsigned char>(s[0])));
    const char rankChar = s[1];
    if (fileChar < 'a' || fileChar > 'h' || rankChar < '1' || rankChar > '8') {
        return -1;
    }
    const int file = fileChar - 'a';
    const int rank = rankChar - '1';
    return sq(rank, file);
}

std::string squareToString(int s) {
    std::string out;
    out.push_back(static_cast<char>('a' + fileOf(s)));
    out.push_back(static_cast<char>('1' + rankOf(s)));
    return out;
}

char notationPieceLetter(int type) {
    switch (type) {
        case PAWN:
            return 'p';
        case KNIGHT:
            return 'N';
        case BISHOP:
            return 'B';
        case ROOK:
            return 'R';
        case QUEEN:
            return 'Q';
        case KING:
            return 'K';
        default:
            return '?';
    }
}

int pieceFromFenChar(char c) {
    switch (c) {
        case 'P':
            return PAWN;
        case 'N':
            return KNIGHT;
        case 'B':
            return BISHOP;
        case 'R':
            return ROOK;
        case 'Q':
            return QUEEN;
        case 'K':
            return KING;
        case 'p':
            return -PAWN;
        case 'n':
            return -KNIGHT;
        case 'b':
            return -BISHOP;
        case 'r':
            return -ROOK;
        case 'q':
            return -QUEEN;
        case 'k':
            return -KING;
        default:
            return 0;
    }
}

bool parseFEN(const std::string& fen, Position& pos) {
    pos = Position{};
    std::istringstream iss(fen);
    std::string boardToken;
    std::string sideToken;
    std::string castlingToken;
    std::string epToken;
    std::string halfToken;
    std::string fullToken;
    if (!(iss >> boardToken >> sideToken >> castlingToken >> epToken)) {
        return false;
    }
    iss >> halfToken >> fullToken;

    int rank = 7;
    int file = 0;
    bool whiteKingFound = false;
    bool blackKingFound = false;

    for (char c : boardToken) {
        if (c == '/') {
            if (file != 8 || rank == 0) {
                return false;
            }
            --rank;
            file = 0;
            continue;
        }
        if (std::isdigit(static_cast<unsigned char>(c))) {
            const int skip = c - '0';
            if (skip < 1 || skip > 8 || file + skip > 8) {
                return false;
            }
            file += skip;
            continue;
        }
        const int piece = pieceFromFenChar(c);
        if (piece == 0 || file >= 8 || rank < 0) {
            return false;
        }
        const int s = sq(rank, file);
        pos.board[s] = piece;
        if (piece == KING) {
            pos.whiteKingSq = s;
            whiteKingFound = true;
        } else if (piece == -KING) {
            pos.blackKingSq = s;
            blackKingFound = true;
        }
        ++file;
    }

    if (rank != 0 || file != 8 || !whiteKingFound || !blackKingFound) {
        return false;
    }

    if (sideToken == "w") {
        pos.whiteToMove = true;
    } else if (sideToken == "b") {
        pos.whiteToMove = false;
    } else {
        return false;
    }

    pos.castlingRights = 0;
    if (castlingToken != "-") {
        for (char c : castlingToken) {
            switch (c) {
                case 'K':
                    pos.castlingRights |= WK_CASTLE;
                    break;
                case 'Q':
                    pos.castlingRights |= WQ_CASTLE;
                    break;
                case 'k':
                    pos.castlingRights |= BK_CASTLE;
                    break;
                case 'q':
                    pos.castlingRights |= BQ_CASTLE;
                    break;
                default:
                    return false;
            }
        }
    }

    if (epToken == "-") {
        pos.enPassant = -1;
    } else {
        pos.enPassant = squareFromString(epToken);
        if (pos.enPassant < 0) {
            return false;
        }
    }

    if (!halfToken.empty()) {
        try {
            pos.halfmoveClock = std::stoi(halfToken);
        } catch (...) {
            pos.halfmoveClock = 0;
        }
    }
    if (!fullToken.empty()) {
        try {
            pos.fullmoveNumber = std::stoi(fullToken);
        } catch (...) {
            pos.fullmoveNumber = 1;
        }
    }
    return true;
}

bool isSquareAttacked(const Position& pos, int targetSq, bool byWhite) {
    const int tr = rankOf(targetSq);
    const int tf = fileOf(targetSq);

    if (byWhite) {
        if (tr > 0) {
            if (tf > 0 && pos.board[targetSq - 9] == PAWN) {
                return true;
            }
            if (tf < 7 && pos.board[targetSq - 7] == PAWN) {
                return true;
            }
        }
    } else {
        if (tr < 7) {
            if (tf > 0 && pos.board[targetSq + 7] == -PAWN) {
                return true;
            }
            if (tf < 7 && pos.board[targetSq + 9] == -PAWN) {
                return true;
            }
        }
    }

    static const int knr[8] = {2, 2, 1, 1, -1, -1, -2, -2};
    static const int knf[8] = {1, -1, 2, -2, 2, -2, 1, -1};
    for (int i = 0; i < 8; ++i) {
        const int nr = tr + knr[i];
        const int nf = tf + knf[i];
        if (nr < 0 || nr > 7 || nf < 0 || nf > 7) {
            continue;
        }
        const int p = pos.board[sq(nr, nf)];
        if (byWhite && p == KNIGHT) {
            return true;
        }
        if (!byWhite && p == -KNIGHT) {
            return true;
        }
    }

    for (int dr = -1; dr <= 1; ++dr) {
        for (int df = -1; df <= 1; ++df) {
            if (dr == 0 && df == 0) {
                continue;
            }
            const int nr = tr + dr;
            const int nf = tf + df;
            if (nr < 0 || nr > 7 || nf < 0 || nf > 7) {
                continue;
            }
            const int p = pos.board[sq(nr, nf)];
            if (byWhite && p == KING) {
                return true;
            }
            if (!byWhite && p == -KING) {
                return true;
            }
        }
    }

    static const int bdr[4] = {1, 1, -1, -1};
    static const int bdf[4] = {1, -1, 1, -1};
    for (int i = 0; i < 4; ++i) {
        int nr = tr + bdr[i];
        int nf = tf + bdf[i];
        while (nr >= 0 && nr <= 7 && nf >= 0 && nf <= 7) {
            const int p = pos.board[sq(nr, nf)];
            if (p != EMPTY) {
                if (byWhite && p > 0 && (pieceType(p) == BISHOP || pieceType(p) == QUEEN)) {
                    return true;
                }
                if (!byWhite && p < 0 && (pieceType(p) == BISHOP || pieceType(p) == QUEEN)) {
                    return true;
                }
                break;
            }
            nr += bdr[i];
            nf += bdf[i];
        }
    }

    static const int rdr[4] = {1, -1, 0, 0};
    static const int rdf[4] = {0, 0, 1, -1};
    for (int i = 0; i < 4; ++i) {
        int nr = tr + rdr[i];
        int nf = tf + rdf[i];
        while (nr >= 0 && nr <= 7 && nf >= 0 && nf <= 7) {
            const int p = pos.board[sq(nr, nf)];
            if (p != EMPTY) {
                if (byWhite && p > 0 && (pieceType(p) == ROOK || pieceType(p) == QUEEN)) {
                    return true;
                }
                if (!byWhite && p < 0 && (pieceType(p) == ROOK || pieceType(p) == QUEEN)) {
                    return true;
                }
                break;
            }
            nr += rdr[i];
            nf += rdf[i];
        }
    }

    return false;
}

bool isInCheck(const Position& pos, bool whiteKing) {
    const int ksq = whiteKing ? pos.whiteKingSq : pos.blackKingSq;
    return isSquareAttacked(pos, ksq, !whiteKing);
}

void makeMove(Position& pos, const Move& m, Undo& u) {
    u.castlingRights = pos.castlingRights;
    u.enPassant = pos.enPassant;
    u.halfmoveClock = pos.halfmoveClock;
    u.whiteKingSq = pos.whiteKingSq;
    u.blackKingSq = pos.blackKingSq;
    u.movedPiece = pos.board[m.from];

    const int moving = u.movedPiece;
    const int side = moving > 0 ? 1 : -1;
    pos.enPassant = -1;

    if (pieceType(moving) == PAWN || (m.flags & CAPTURE) != 0) {
        pos.halfmoveClock = 0;
    } else {
        ++pos.halfmoveClock;
    }

    if ((m.flags & EN_PASSANT) != 0) {
        const int capSq = m.to + (side == 1 ? -8 : 8);
        u.captured = pos.board[capSq];
        pos.board[capSq] = EMPTY;
    } else {
        u.captured = pos.board[m.to];
    }

    pos.board[m.from] = EMPTY;

    if ((m.flags & CASTLE) != 0) {
        if (moving > 0) {
            if (m.to == 6) {
                pos.board[5] = pos.board[7];
                pos.board[7] = EMPTY;
            } else if (m.to == 2) {
                pos.board[3] = pos.board[0];
                pos.board[0] = EMPTY;
            }
        } else {
            if (m.to == 62) {
                pos.board[61] = pos.board[63];
                pos.board[63] = EMPTY;
            } else if (m.to == 58) {
                pos.board[59] = pos.board[56];
                pos.board[56] = EMPTY;
            }
        }
    }

    int placed = moving;
    if (m.promo != 0) {
        placed = side * m.promo;
    }
    pos.board[m.to] = placed;

    if (moving == KING) {
        pos.whiteKingSq = m.to;
    } else if (moving == -KING) {
        pos.blackKingSq = m.to;
    }

    if (pieceType(moving) == PAWN && std::abs(m.to - m.from) == 16) {
        pos.enPassant = m.from + (side == 1 ? 8 : -8);
    }

    if (moving == KING) {
        pos.castlingRights &= ~(WK_CASTLE | WQ_CASTLE);
    } else if (moving == -KING) {
        pos.castlingRights &= ~(BK_CASTLE | BQ_CASTLE);
    }
    if (m.from == 0 || m.to == 0) {
        pos.castlingRights &= ~WQ_CASTLE;
    }
    if (m.from == 7 || m.to == 7) {
        pos.castlingRights &= ~WK_CASTLE;
    }
    if (m.from == 56 || m.to == 56) {
        pos.castlingRights &= ~BQ_CASTLE;
    }
    if (m.from == 63 || m.to == 63) {
        pos.castlingRights &= ~BK_CASTLE;
    }

    pos.whiteToMove = !pos.whiteToMove;
}

void unmakeMove(Position& pos, const Move& m, const Undo& u) {
    pos.whiteToMove = !pos.whiteToMove;
    pos.castlingRights = u.castlingRights;
    pos.enPassant = u.enPassant;
    pos.halfmoveClock = u.halfmoveClock;
    pos.whiteKingSq = u.whiteKingSq;
    pos.blackKingSq = u.blackKingSq;

    const int moving = u.movedPiece;
    const int side = moving > 0 ? 1 : -1;

    if ((m.flags & CASTLE) != 0) {
        if (moving > 0) {
            if (m.to == 6) {
                pos.board[7] = pos.board[5];
                pos.board[5] = EMPTY;
            } else if (m.to == 2) {
                pos.board[0] = pos.board[3];
                pos.board[3] = EMPTY;
            }
        } else {
            if (m.to == 62) {
                pos.board[63] = pos.board[61];
                pos.board[61] = EMPTY;
            } else if (m.to == 58) {
                pos.board[56] = pos.board[59];
                pos.board[59] = EMPTY;
            }
        }
    }

    if ((m.flags & EN_PASSANT) != 0) {
        pos.board[m.from] = moving;
        pos.board[m.to] = EMPTY;
        const int capSq = m.to + (side == 1 ? -8 : 8);
        pos.board[capSq] = u.captured;
    } else {
        pos.board[m.from] = moving;
        pos.board[m.to] = u.captured;
    }
}

void generatePseudoMoves(const Position& pos, std::vector<Move>& moves, bool capturesOnly) {
    const bool white = pos.whiteToMove;
    const int sideSign = white ? 1 : -1;
    static const int promotions[4] = {QUEEN, ROOK, BISHOP, KNIGHT};

    for (int from = 0; from < 64; ++from) {
        const int piece = pos.board[from];
        if (piece == EMPTY || (piece > 0) != white) {
            continue;
        }
        const int type = pieceType(piece);
        const int r = rankOf(from);
        const int f = fileOf(from);

        if (type == PAWN) {
            const int forward = white ? 8 : -8;
            const int startRank = white ? 1 : 6;
            const int promoRank = white ? 6 : 1;

            const int oneStep = from + forward;
            if (oneStep >= 0 && oneStep < 64 && pos.board[oneStep] == EMPTY) {
                if (r == promoRank) {
                    for (int promo : promotions) {
                        moves.push_back(Move{from, oneStep, promo, 0});
                    }
                } else if (!capturesOnly) {
                    moves.push_back(Move{from, oneStep, 0, 0});
                    if (r == startRank) {
                        const int twoStep = from + 2 * forward;
                        if (twoStep >= 0 && twoStep < 64 && pos.board[twoStep] == EMPTY) {
                            moves.push_back(Move{from, twoStep, 0, 0});
                        }
                    }
                }
            }

            for (int df : {-1, 1}) {
                const int nf = f + df;
                if (nf < 0 || nf > 7) {
                    continue;
                }
                const int to = from + forward + df;
                if (to < 0 || to >= 64) {
                    continue;
                }
                const int target = pos.board[to];
                if (target != EMPTY && (target > 0) != white) {
                    if (r == promoRank) {
                        for (int promo : promotions) {
                            moves.push_back(Move{from, to, promo, CAPTURE});
                        }
                    } else {
                        moves.push_back(Move{from, to, 0, CAPTURE});
                    }
                } else if (to == pos.enPassant) {
                    moves.push_back(Move{from, to, 0, CAPTURE | EN_PASSANT});
                }
            }
            continue;
        }

        if (type == KNIGHT) {
            static const int knr[8] = {2, 2, 1, 1, -1, -1, -2, -2};
            static const int knf[8] = {1, -1, 2, -2, 2, -2, 1, -1};
            for (int i = 0; i < 8; ++i) {
                const int nr = r + knr[i];
                const int nf = f + knf[i];
                if (nr < 0 || nr > 7 || nf < 0 || nf > 7) {
                    continue;
                }
                const int to = sq(nr, nf);
                const int target = pos.board[to];
                if (target == EMPTY) {
                    if (!capturesOnly) {
                        moves.push_back(Move{from, to, 0, 0});
                    }
                } else if ((target > 0) != white) {
                    moves.push_back(Move{from, to, 0, CAPTURE});
                }
            }
            continue;
        }

        if (type == BISHOP || type == ROOK || type == QUEEN) {
            static const int bdr[4] = {1, 1, -1, -1};
            static const int bdf[4] = {1, -1, 1, -1};
            static const int rdr[4] = {1, -1, 0, 0};
            static const int rdf[4] = {0, 0, 1, -1};

            if (type == BISHOP || type == QUEEN) {
                for (int i = 0; i < 4; ++i) {
                    int nr = r + bdr[i];
                    int nf = f + bdf[i];
                    while (nr >= 0 && nr <= 7 && nf >= 0 && nf <= 7) {
                        const int to = sq(nr, nf);
                        const int target = pos.board[to];
                        if (target == EMPTY) {
                            if (!capturesOnly) {
                                moves.push_back(Move{from, to, 0, 0});
                            }
                        } else {
                            if ((target > 0) != white) {
                                moves.push_back(Move{from, to, 0, CAPTURE});
                            }
                            break;
                        }
                        nr += bdr[i];
                        nf += bdf[i];
                    }
                }
            }
            if (type == ROOK || type == QUEEN) {
                for (int i = 0; i < 4; ++i) {
                    int nr = r + rdr[i];
                    int nf = f + rdf[i];
                    while (nr >= 0 && nr <= 7 && nf >= 0 && nf <= 7) {
                        const int to = sq(nr, nf);
                        const int target = pos.board[to];
                        if (target == EMPTY) {
                            if (!capturesOnly) {
                                moves.push_back(Move{from, to, 0, 0});
                            }
                        } else {
                            if ((target > 0) != white) {
                                moves.push_back(Move{from, to, 0, CAPTURE});
                            }
                            break;
                        }
                        nr += rdr[i];
                        nf += rdf[i];
                    }
                }
            }
            continue;
        }

        if (type == KING) {
            for (int dr = -1; dr <= 1; ++dr) {
                for (int df = -1; df <= 1; ++df) {
                    if (dr == 0 && df == 0) {
                        continue;
                    }
                    const int nr = r + dr;
                    const int nf = f + df;
                    if (nr < 0 || nr > 7 || nf < 0 || nf > 7) {
                        continue;
                    }
                    const int to = sq(nr, nf);
                    const int target = pos.board[to];
                    if (target == EMPTY) {
                        if (!capturesOnly) {
                            moves.push_back(Move{from, to, 0, 0});
                        }
                    } else if ((target > 0) != white) {
                        moves.push_back(Move{from, to, 0, CAPTURE});
                    }
                }
            }

            if (!capturesOnly) {
                if (white && from == 4) {
                    if ((pos.castlingRights & WK_CASTLE) != 0 &&
                        pos.board[5] == EMPTY && pos.board[6] == EMPTY && pos.board[7] == ROOK &&
                        !isSquareAttacked(pos, 4, false) &&
                        !isSquareAttacked(pos, 5, false) &&
                        !isSquareAttacked(pos, 6, false)) {
                        moves.push_back(Move{4, 6, 0, CASTLE});
                    }
                    if ((pos.castlingRights & WQ_CASTLE) != 0 &&
                        pos.board[1] == EMPTY && pos.board[2] == EMPTY && pos.board[3] == EMPTY &&
                        pos.board[0] == ROOK &&
                        !isSquareAttacked(pos, 4, false) &&
                        !isSquareAttacked(pos, 3, false) &&
                        !isSquareAttacked(pos, 2, false)) {
                        moves.push_back(Move{4, 2, 0, CASTLE});
                    }
                } else if (!white && from == 60) {
                    if ((pos.castlingRights & BK_CASTLE) != 0 &&
                        pos.board[61] == EMPTY && pos.board[62] == EMPTY && pos.board[63] == -ROOK &&
                        !isSquareAttacked(pos, 60, true) &&
                        !isSquareAttacked(pos, 61, true) &&
                        !isSquareAttacked(pos, 62, true)) {
                        moves.push_back(Move{60, 62, 0, CASTLE});
                    }
                    if ((pos.castlingRights & BQ_CASTLE) != 0 &&
                        pos.board[57] == EMPTY && pos.board[58] == EMPTY && pos.board[59] == EMPTY &&
                        pos.board[56] == -ROOK &&
                        !isSquareAttacked(pos, 60, true) &&
                        !isSquareAttacked(pos, 59, true) &&
                        !isSquareAttacked(pos, 58, true)) {
                        moves.push_back(Move{60, 58, 0, CASTLE});
                    }
                }
            }
        }
    }
}

void generateLegalMoves(const Position& pos, std::vector<Move>& legal, bool capturesOnly) {
    std::vector<Move> pseudo;
    pseudo.reserve(128);
    generatePseudoMoves(pos, pseudo, capturesOnly);

    Position temp = pos;
    const bool moverWhite = pos.whiteToMove;
    for (const Move& m : pseudo) {
        Undo u;
        makeMove(temp, m, u);
        if (!isInCheck(temp, moverWhite)) {
            legal.push_back(m);
        }
        unmakeMove(temp, m, u);
    }
}

class Engine {
   public:
    Engine() {
        for (auto& p : killers_) {
            p[0] = Move{};
            p[1] = Move{};
        }
    }

    Move findBestMove(Position& pos, double timeoutSeconds) {
        nodes_ = 0;
        stopped_ = false;
        for (auto& colorLayer : history_) {
            for (auto& fromLayer : colorLayer) {
                fromLayer.fill(0);
            }
        }

        std::vector<Move> rootMoves;
        generateLegalMoves(pos, rootMoves, false);
        if (rootMoves.empty()) {
            return Move{};
        }

        Move bestMove = rootMoves[0];
        Move pvMove = bestMove;
        int bestScore = -INF;

        const auto now = std::chrono::steady_clock::now();
        const double totalMs = timeoutSeconds * 1000.0;
        const double safetyMs = std::clamp(totalMs * 0.05, 30.0, 250.0);
        const long long searchMs = static_cast<long long>(std::max(1.0, totalMs - safetyMs));
        deadline_ = now + std::chrono::milliseconds(searchMs);

        for (int depth = 1; depth <= 64; ++depth) {
            Move iterBest = bestMove;
            const int score = searchRoot(pos, depth, iterBest, &pvMove);
            if (stopped_) {
                break;
            }
            bestMove = iterBest;
            pvMove = bestMove;
            bestScore = score;

            if (bestScore > MATE_SCORE - 100 || bestScore < -MATE_SCORE + 100) {
                break;
            }
        }
        return bestMove;
    }

   private:
    std::array<std::array<Move, 2>, MAX_PLY> killers_{};
    std::array<std::array<std::array<int, 64>, 64>, 2> history_{};
    std::chrono::steady_clock::time_point deadline_{};
    unsigned long long nodes_ = 0;
    bool stopped_ = false;

    bool checkTime() {
        if (stopped_) {
            return true;
        }
        if ((nodes_ & 2047ULL) == 0ULL && std::chrono::steady_clock::now() >= deadline_) {
            stopped_ = true;
            return true;
        }
        return false;
    }

    int pstValue(int type, int sqWhitePerspective) const {
        static const int pawnPst[64] = {

            10, 10, 20, 30, 30, 20, 10, 10,
            5, 5, 10, 25, 25, 10, 5, 5,
            0, 0, 0, 20, 20, 0, 0, 0,
            5, -5, -10, 0, 0, -10, -5, 5,
            5, 10, 10, -20, -20, 10, 10, 5,
            0, 0, 0, 0, 0, 0, 0, 0};

        static const int knightPst[64] = {
            -50, -40, -30, -30, -30, -30, -40, -50,
            -40, -20, 0, 5, 5, 0, -20, -40,
            -30, 5, 10, 15, 15, 10, 5, -30,
            -30, 0, 15, 20, 20, 15, 0, -30,
            -30, 5, 15, 20, 20, 15, 5, -30,
            -30, 0, 10, 15, 15, 10, 0, -30,
            -40, -20, 0, 0, 0, 0, -20, -40,
            -50, -40, -30, -30, -30, -30, -40, -50};

        static const int bishopPst[64] = {
            -20, -10, -10, -10, -10, -10, -10, -20,
            -10, 0, 0, 0, 0, 0, 0, -10,
            -10, 0, 5, 10, 10, 5, 0, -10,
            -10, 5, 5, 10, 10, 5, 5, -10,
            -10, 0, 10, 10, 10, 10, 0, -10,
            -10, 10, 10, 10, 10, 10, 10, -10,
            -10, 5, 0, 0, 0, 0, 5, -10,
            -20, -10, -10, -10, -10, -10, -10, -20};

        static const int rookPst[64] = {
            0, 0, 0, 0, 0, 0, 0, 0,
            5, 10, 10, 10, 10, 10, 10, 5,
            -5, 0, 0, 0, 0, 0, 0, -5,
            -5, 0, 0, 0, 0, 0, 0, -5,
            -5, 0, 0, 0, 0, 0, 0, -5,
            -5, 0, 0, 0, 0, 0, 0, -5,
            -5, 0, 0, 0, 0, 0, 0, -5,
            0, 0, 0, 5, 5, 0, 0, 0};

        static const int queenPst[64] = {
            -20, -10, -10, -5, -5, -10, -10, -20,
            -10, 0, 0, 0, 0, 0, 0, -10,
            -10, 0, 5, 5, 5, 5, 0, -10,
            -5, 0, 5, 5, 5, 5, 0, -5,
            0, 0, 5, 5, 5, 5, 0, -5,
            -10, 5, 5, 5, 5, 5, 0, -10,
            -10, 0, 5, 0, 0, 0, 0, -10,
            -20, -10, -10, -5, -5, -10, -10, -20};

        static const int kingPst[64] = {
            -30, -40, -40, -50, -50, -40, -40, -30,
            -30, -40, -40, -50, -50, -40, -40, -30,
            -30, -40, -40, -50, -50, -40, -40, -30,
            -30, -40, -40, -50, -50, -40, -40, -30,
            -20, -30, -30, -40, -40, -30, -30, -20,

            20, 20, 0, 0, 0, 0, 20, 20,
            20, 30, 10, 0, 0, 10, 30, 20};

        switch (type) {
            case PAWN:
                return pawnPst[sqWhitePerspective];
            case KNIGHT:
                return knightPst[sqWhitePerspective];
            case BISHOP:
                return bishopPst[sqWhitePerspective];
            case ROOK:
                return rookPst[sqWhitePerspective];
            case QUEEN:
                return queenPst[sqWhitePerspective];
            case KING:

    }

    int evaluate(const Position& pos) const {
        int score = 0;
        int whiteBishops = 0;
        int blackBishops = 0;

        for (int s = 0; s < 64; ++s) {
            const int p = pos.board[s];
            if (p == EMPTY) {
                continue;
            }
            const int type = pieceType(p);
            const int val = pieceValue(type);
            if (p > 0) {
                score += val + pstValue(type, s);
                if (type == BISHOP) {
                    ++whiteBishops;
                }
            } else {
                score -= val + pstValue(type, s ^ 56);
                if (type == BISHOP) {
                    ++blackBishops;
                }
            }
        }

        if (whiteBishops >= 2) {
            score += 25;
        }
        if (blackBishops >= 2) {
            score -= 25;
        }

        return pos.whiteToMove ? score : -score;
    }

    int scoreMove(const Position& pos, const Move& m, int ply, const Move* pvMove) const {
        int score = 0;
        if (pvMove != nullptr && sameMove(m, *pvMove)) {
            score += 2'000'000;
        }

        const int moving = pos.board[m.from];
        const int attackerType = pieceType(moving);

        if ((m.flags & CAPTURE) != 0) {
            const int victimType = ((m.flags & EN_PASSANT) != 0) ? PAWN : pieceType(pos.board[m.to]);
            score += 100'000 + pieceValue(victimType) * 10 - pieceValue(attackerType);
        }

        if (m.promo != 0) {
            score += 90'000 + pieceValue(m.promo);
        }

        if ((m.flags & CAPTURE) == 0 && m.promo == 0 && ply < MAX_PLY) {
            if (sameMove(m, killers_[ply][0])) {
                score += 85'000;
            } else if (sameMove(m, killers_[ply][1])) {
                score += 84'000;
            }
            score += history_[colorIndex(pos.whiteToMove)][m.from][m.to];
        }
        return score;
    }

    void orderMoves(const Position& pos, std::vector<Move>& moves, int ply, const Move* pvMove) const {
        std::vector<std::pair<int, Move>> scored;
        scored.reserve(moves.size());
        for (const Move& m : moves) {
            scored.push_back(std::make_pair(scoreMove(pos, m, ply, pvMove), m));
        }
        std::sort(scored.begin(), scored.end(),
                  [](const std::pair<int, Move>& a, const std::pair<int, Move>& b) { return a.first > b.first; });
        for (size_t i = 0; i < moves.size(); ++i) {
            moves[i] = scored[i].second;
        }
    }

    int quiescence(Position& pos, int alpha, int beta, int ply) {
        ++nodes_;
        if (checkTime()) {
            return 0;
        }

        const bool inCheck = isInCheck(pos, pos.whiteToMove);
        if (!inCheck) {
            const int standPat = evaluate(pos);
            if (standPat >= beta) {
                return beta;
            }
            if (standPat > alpha) {
                alpha = standPat;
            }
        }

        std::vector<Move> moves;
        moves.reserve(64);
        generateLegalMoves(pos, moves, !inCheck);
        if (moves.empty()) {
            if (inCheck) {
                return -MATE_SCORE + ply;
            }
            return alpha;
        }

        orderMoves(pos, moves, ply, nullptr);
        for (const Move& m : moves) {
            Undo u;
            makeMove(pos, m, u);
            const int score = -quiescence(pos, -beta, -alpha, ply + 1);
            unmakeMove(pos, m, u);
            if (stopped_) {
                return 0;
            }
            if (score >= beta) {
                return beta;
            }
            if (score > alpha) {
                alpha = score;
            }
        }
        return alpha;
    }

    int negamax(Position& pos, int depth, int alpha, int beta, int ply) {
        ++nodes_;
        if (checkTime()) {
            return 0;
        }
        if (depth <= 0) {
            return quiescence(pos, alpha, beta, ply);
        }

        const bool inCheck = isInCheck(pos, pos.whiteToMove);
        std::vector<Move> moves;
        moves.reserve(128);
        generateLegalMoves(pos, moves, false);

        if (moves.empty()) {
            if (inCheck) {
                return -MATE_SCORE + ply;
            }
            return 0;
        }

        orderMoves(pos, moves, ply, nullptr);
        int best = -INF;
        const bool moverWhite = pos.whiteToMove;

        for (const Move& m : moves) {
            Undo u;
            makeMove(pos, m, u);
            const int score = -negamax(pos, depth - 1, -beta, -alpha, ply + 1);
            unmakeMove(pos, m, u);

            if (stopped_) {
                return 0;
            }
            if (score > best) {
                best = score;
            }
            if (score > alpha) {
                alpha = score;
                if (alpha >= beta) {
                    if ((m.flags & CAPTURE) == 0 && m.promo == 0 && ply < MAX_PLY) {
                        if (!sameMove(m, killers_[ply][0])) {
                            killers_[ply][1] = killers_[ply][0];
                            killers_[ply][0] = m;
                        }
                        history_[colorIndex(moverWhite)][m.from][m.to] += depth * depth;
                    }
                    break;
                }
            }
        }

        return best;
    }

    int searchRoot(Position& pos, int depth, Move& bestMove, const Move* pvMove) {
        std::vector<Move> moves;
        moves.reserve(128);
        generateLegalMoves(pos, moves, false);
        if (moves.empty()) {
            return isInCheck(pos, pos.whiteToMove) ? -MATE_SCORE : 0;
        }

        orderMoves(pos, moves, 0, pvMove);
        int alpha = -INF;
        int bestScore = -INF;
        Move localBest = moves[0];

        for (const Move& m : moves) {
            Undo u;
            makeMove(pos, m, u);
            const int score = -negamax(pos, depth - 1, -INF, INF, 1);
            unmakeMove(pos, m, u);

            if (stopped_) {
                break;
            }
            if (score > bestScore) {
                bestScore = score;
                localBest = m;
            }
            if (score > alpha) {
                alpha = score;
            }
        }

        if (!stopped_) {
            bestMove = localBest;
        }
        return bestScore;
    }
};

std::string moveToNotation(Position& pos, const Move& m) {
    if (m.from < 0 || m.to < 0) {
        return "0000";
    }
    std::string out;
    if ((m.flags & CASTLE) != 0) {
        out = (fileOf(m.to) == 6) ? "O-O" : "O-O-O";
    } else {
        const int moving = pos.board[m.from];
        out.push_back(notationPieceLetter(pieceType(moving)));
        out += squareToString(m.from);
        if ((m.flags & CAPTURE) != 0) {
            out.push_back('x');
        }
        out += squareToString(m.to);
        if (m.promo != 0) {
            out.push_back('/');
            out.push_back(notationPieceLetter(m.promo));
        }
        if ((m.flags & EN_PASSANT) != 0) {
            out += "ep";
        }
    }

    Undo u;
    makeMove(pos, m, u);
    const bool oppInCheck = isInCheck(pos, pos.whiteToMove);
    if (oppInCheck) {
        std::vector<Move> replies;
        replies.reserve(64);
        generateLegalMoves(pos, replies, false);
        out += replies.empty() ? "#" : "+";
    }
    unmakeMove(pos, m, u);
    return out;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::string timeoutLine;
    std::string fenLine;
    if (!std::getline(std::cin, timeoutLine)) {
        return 0;
    }
    if (!std::getline(std::cin, fenLine)) {
        return 0;
    }

    timeoutLine = trim(timeoutLine);
    fenLine = trim(fenLine);

    double timeoutSeconds = 5.0;
    try {
        timeoutSeconds = std::stod(timeoutLine);
    } catch (...) {
        timeoutSeconds = 5.0;
    }
    if (timeoutSeconds <= 0.0) {
        timeoutSeconds = 0.1;
    }

    Position pos;
    if (!parseFEN(fenLine, pos)) {
        std::cout << "0000\n";
        return 0;
    }

    Engine engine;
    Move best = engine.findBestMove(pos, timeoutSeconds);
    if (best.from < 0 || best.to < 0) {
        std::cout << "0000\n";
        return 0;
    }

    std::cout << moveToNotation(pos, best) << '\n';
    return 0;
}
