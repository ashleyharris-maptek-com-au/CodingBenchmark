import sys
from functools import lru_cache


def parse_input() -> tuple[int, list[int]]:
    data = sys.stdin.read()
    lines = data.splitlines()
    if not lines:
        return 0, []

    first = lines[0].strip().split()
    if len(first) < 2:
        return 0, []

    u = int(first[0])
    s = int(first[1])

    # Expected format has a universe line after the header.
    # Be tolerant if it is missing.
    if len(lines) - 1 == s:
        set_start = 1
    else:
        set_start = 2

    set_lines = lines[set_start:set_start + s]
    if len(set_lines) < s:
        set_lines += [""] * (s - len(set_lines))

    row_masks: list[int] = []
    full_mask = (1 << u) - 1

    for line in set_lines:
        mask = 0
        tokens = line.strip().split()
        valid = True
        for tok in tokens:
            e = int(tok)
            if e < 0 or e >= u:
                valid = False
                break
            mask |= 1 << e
        # Invalid sets are made unusable.
        if not valid:
            mask = full_mask + 1
        row_masks.append(mask)

    return u, row_masks


def solve_exact_cover(u: int, row_masks: list[int]) -> list[int] | None:
    full_mask = (1 << u) - 1

    rows_by_elem: list[list[int]] = [[] for _ in range(u)]
    for idx, mask in enumerate(row_masks):
        if mask <= 0 or mask > full_mask:
            continue
        m = mask
        while m:
            bit = m & -m
            elem = bit.bit_length() - 1
            rows_by_elem[elem].append(idx)
            m ^= bit

    for e in range(u):
        if not rows_by_elem[e]:
            return None

    @lru_cache(maxsize=None)
    def dfs(remaining: int):
        if remaining == 0:
            return ()

        best_elem = -1
        best_candidates = None

        m = remaining
        while m:
            bit = m & -m
            elem = bit.bit_length() - 1

            candidates = []
            for r in rows_by_elem[elem]:
                rm = row_masks[r]
                # Row must be fully inside remaining uncovered elements.
                if rm & remaining == rm:
                    candidates.append(r)

            if not candidates:
                return None

            if best_candidates is None or len(candidates) < len(best_candidates):
                best_elem = elem
                best_candidates = candidates
                if len(best_candidates) == 1:
                    break

            m ^= bit

        for r in best_candidates:
            rm = row_masks[r]
            nxt = remaining & ~rm
            sub = dfs(nxt)
            if sub is not None:
                return (r,) + sub

        return None

    ans = dfs(full_mask)
    if ans is None:
        return None

    return sorted(ans)


def main() -> None:
    u, row_masks = parse_input()
    solution = solve_exact_cover(u, row_masks)

    out = []
    if solution is None:
        out.append("NO SOLUTION")
    else:
        out.append("SOLUTION")
        out.append(" ".join(map(str, solution)))

    sys.stdout.write("\n".join(out))


if __name__ == "__main__":
    main()
