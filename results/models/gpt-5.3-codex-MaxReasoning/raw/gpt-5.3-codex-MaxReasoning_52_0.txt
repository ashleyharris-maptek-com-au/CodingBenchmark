import math
from statistics import median

# Persistent controller memory across calls.
_AP = {
    "speed_kmh": 0.0,
    "yaw_dps": 0.0,
    "lane_offset_m": 0.0,
    "curvature_1pm": 0.0,
    "speed_limit_kmh": 130.0,
    "speed_limit_hold_s": 0.0,
    "speed_i": 0.0,
    "lane_i": 0.0,
    "last_steering": 0.0,
    "last_throttle": 0.0,
    "last_brake": 0.0,
    "lane_change_timer_s": 0.0,
    "lane_change_dir": 0,
    "hazard_hold_s": 0.0,
    "brake_fail_timer_s": 0.0,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _as_float(v):
    try:
        f = float(v)
    except (TypeError, ValueError):
        return None
    if not math.isfinite(f):
        return None
    return f


def _boolish(v):
    if isinstance(v, str):
        s = v.strip().lower()
        if s in ("false", "0", "no", "off", ""):
            return False
        if s in ("true", "1", "yes", "on"):
            return True
    return bool(v)


def _robust_fuse(cands, prev, jump_limit=None):
    vals = [v for v, _w in cands if v is not None and math.isfinite(v)]
    if not vals:
        return prev, 0.0

    med = median(vals)
    devs = [abs(v - med) for v in vals]
    mad = median(devs) + 1e-6
    tol = max(3.0 * mad, 1.0)

    inliers = [(v, max(0.05, w)) for v, w in cands if v is not None and abs(v - med) <= tol]
    if not inliers:
        inliers = [(med, 1.0)]

    wsum = sum(w for _v, w in inliers)
    fused = sum(v * w for v, w in inliers) / max(wsum, 1e-6)
    conf = _clamp(wsum / max(1.0, float(len(cands))), 0.0, 1.0)

    if prev is None:
        return fused, conf

    if jump_limit is not None and conf < 0.65:
        fused = _clamp(fused, prev - jump_limit, prev + jump_limit)

    alpha = 0.15 + 0.55 * conf
    out = prev + alpha * (fused - prev)
    return out, conf


def _lidar_keys(angle_deg):
    sign = "+" if angle_deg >= 0 else "-"
    mag = abs(int(angle_deg))
    stem = f"lidar_{sign}{mag:03d}_deg"
    return stem + "_range_m", stem + "_valid"


def _median_or(default, arr):
    vals = [x for x in arr if x is not None and math.isfinite(x)]
    return median(vals) if vals else default


def autopilot_step(state: dict, dt: float) -> dict:
    global _AP

    dt = _clamp(_as_float(dt) or _as_float(state.get("dt")) or 0.05, 0.01, 0.2)

    mu = _clamp(_as_float(state.get("estimated_mu")) or 0.9, 0.1, 1.0)

    # ---------- Speed fusion ----------
    speed_cands = []

    for corner in ("fl", "fr", "rl", "rr"):
        k = f"wheel_speed_{corner}_kmh"
        vk = _as_float(state.get(k))
        if vk is None or vk < 0.0 or vk > 320.0:
            continue
        valid = _boolish(state.get(f"wheel_speed_{corner}_valid", True))
        speed_cands.append((vk, 1.0 if valid else 0.35))

    for idx in (1, 2):
        vk = _as_float(state.get(f"vehicle_speed_{idx}_kmh"))
        if vk is not None and 0.0 <= vk <= 320.0:
            speed_cands.append((vk, 0.9))

    for idx in (1, 2):
        vk = _as_float(state.get(f"gps_{idx}_speed_kmh"))
        if vk is None or vk < 0.0 or vk > 320.0:
            continue
        valid = _boolish(state.get(f"gps_{idx}_valid", True))
        speed_cands.append((vk, 0.8 if valid else 0.25))

    speed_kmh, _speed_conf = _robust_fuse(speed_cands, _AP["speed_kmh"], jump_limit=max(3.0, 20.0 * dt))
    speed_kmh = max(0.0, speed_kmh)
    speed_ms = speed_kmh / 3.6

    # ---------- Gyro + vision curvature fusion ----------
    yaw_cands = []
    for idx in (1, 2, 3):
        gy = _as_float(state.get(f"gyro_{idx}_yaw_rate_dps"))
        if gy is None or abs(gy) > 150.0:
            continue
        valid = _boolish(state.get(f"gyro_{idx}_valid", True))
        yaw_cands.append((gy, 1.0 if valid else 0.3))

    # ---------- Vision fusion ----------
    lane_cands = []
    curvature_cands = []
    speed_limit_cands = []
    obs_dist_cands = []
    obs_rel_cands = []

    for idx in (1, 2, 3):
        lane_offset = _as_float(state.get(f"vision_{idx}_lane_offset_m"))
        if lane_offset is not None and abs(lane_offset) > 6.0:
            lane_offset = None

        left_d = _as_float(state.get(f"vision_{idx}_left_line_dist_m"))
        right_d = _as_float(state.get(f"vision_{idx}_right_line_dist_m"))
        left_ok = _boolish(state.get(f"vision_{idx}_left_line_valid", False)) and left_d is not None and 0.0 < left_d < 8.0
        right_ok = _boolish(state.get(f"vision_{idx}_right_line_valid", False)) and right_d is not None and 0.0 < right_d < 8.0

        derived_offset = None
        if left_ok and right_ok:
            # Positive means vehicle is to the right of center if right distance grows.
            derived_offset = 0.5 * (right_d - left_d)

        local_estimates = []
        if lane_offset is not None:
            local_estimates.append(lane_offset)
        if derived_offset is not None:
            local_estimates.append(derived_offset)

        if local_estimates:
            local_lane = sum(local_estimates) / float(len(local_estimates))
            conf = 0.35
            if lane_offset is not None:
                conf += 0.25
            if left_ok and right_ok:
                conf += 0.40
            lane_cands.append((local_lane, _clamp(conf, 0.1, 1.0)))

        curv = _as_float(state.get(f"vision_{idx}_road_curvature_1pm"))
        if curv is not None and abs(curv) < 0.35:
            curvature_cands.append((curv, 0.75))
            yaw_from_curv = curv * speed_ms * (180.0 / math.pi)
            if abs(yaw_from_curv) < 150.0:
                yaw_cands.append((yaw_from_curv, 0.45))

        sl = _as_float(state.get(f"vision_{idx}_speed_limit_kmh"))
        sl_ok = _boolish(state.get(f"vision_{idx}_speed_limit_valid", False))
        if sl is not None and 20.0 <= sl <= 180.0:
            speed_limit_cands.append((sl, 0.9 if sl_ok else 0.35))

        obs_detected = _boolish(state.get(f"vision_{idx}_obstacle_detected", False))
        obs_valid = _boolish(state.get(f"vision_{idx}_obstacle_valid", True))
        obs_dist = _as_float(state.get(f"vision_{idx}_obstacle_dist_m"))
        obs_rel = _as_float(state.get(f"vision_{idx}_obstacle_rel_speed_kmh"))

        if obs_dist is not None and 0.2 <= obs_dist <= 250.0 and (obs_detected or obs_valid):
            w = (0.9 if obs_detected else 0.5) * (1.0 if obs_valid else 0.5)
            obs_dist_cands.append((obs_dist, _clamp(w, 0.1, 1.0)))
            if obs_rel is not None and abs(obs_rel) <= 220.0:
                obs_rel_cands.append((obs_rel, _clamp(w, 0.1, 1.0)))

    lane_jump = max(0.3, (0.8 + 0.1 * speed_ms) * dt)
    lane_offset_m, lane_conf = _robust_fuse(lane_cands, _AP["lane_offset_m"], jump_limit=lane_jump)

    curvature_1pm, _curv_conf = _robust_fuse(curvature_cands, _AP["curvature_1pm"], jump_limit=0.03)
    yaw_dps, _yaw_conf = _robust_fuse(yaw_cands, _AP["yaw_dps"], jump_limit=max(5.0, 60.0 * dt))

    # ---------- Lidar fusion ----------
    lidar = {}
    for a in (-90, -60, -45, -30, -15, -5, 5, 15, 30, 45, 60, 90):
        rk, vk = _lidar_keys(a)
        rr = _as_float(state.get(rk))
        if rr is None or rr < 0.1 or rr > 250.0:
            continue
        valid = _boolish(state.get(vk, True))
        lidar[a] = (rr, 1.0 if valid else 0.35)

    def lidar_min(angles, default=999.0):
        vals = [lidar[a][0] for a in angles if a in lidar]
        return min(vals) if vals else default

    front_min = lidar_min((-15, -5, 5, 15), default=999.0)
    if front_min >= 999.0:
        front_min = lidar_min((-30, 30), default=999.0)

    left_clear = lidar_min((30, 45, 60, 90), default=999.0)
    right_clear = lidar_min((-30, -45, -60, -90), default=999.0)

    # Slight lidar side-balance correction when vision confidence is weak.
    if lane_conf < 0.35 and left_clear < 999.0 and right_clear < 999.0:
        side_bias = _clamp((right_clear - left_clear) * 0.03, -0.8, 0.8)
        lane_offset_m = 0.8 * lane_offset_m + 0.2 * side_bias

    # ---------- Speed limit hold/filter ----------
    if speed_limit_cands:
        raw_limit, _ = _robust_fuse(speed_limit_cands, _AP["speed_limit_kmh"], jump_limit=25.0)
        # React quickly to lower limits, slowly to higher limits.
        if raw_limit < _AP["speed_limit_kmh"]:
            _AP["speed_limit_kmh"] = 0.65 * _AP["speed_limit_kmh"] + 0.35 * raw_limit
        else:
            _AP["speed_limit_kmh"] = 0.85 * _AP["speed_limit_kmh"] + 0.15 * raw_limit
        _AP["speed_limit_hold_s"] = 3.0
    else:
        _AP["speed_limit_hold_s"] = max(0.0, _AP["speed_limit_hold_s"] - dt)
        if _AP["speed_limit_hold_s"] <= 0.0:
            _AP["speed_limit_kmh"] = 0.995 * _AP["speed_limit_kmh"] + 0.005 * 130.0

    # ---------- Obstacle fusion ----------
    vis_obs_dist = _median_or(999.0, [v for v, _w in obs_dist_cands])
    front_dist = min(front_min, vis_obs_dist)

    rel_kmh = _median_or(None, [v for v, _w in obs_rel_cands])
    if rel_kmh is None:
        closing_speed_ms = speed_ms
    else:
        # Negative relative speed means we are closing.
        closing_speed_ms = max(0.0, -rel_kmh / 3.6)
        if closing_speed_ms < 0.3 and front_dist < 50.0:
            closing_speed_ms = speed_ms

    ttc_s = front_dist / max(0.2, closing_speed_ms)
    obstacle_now = front_dist < 120.0 and (front_min < 120.0 or len(obs_dist_cands) >= 1)

    if obstacle_now:
        _AP["hazard_hold_s"] = 0.8
    else:
        _AP["hazard_hold_s"] = max(0.0, _AP["hazard_hold_s"] - dt)
    obstacle_active = obstacle_now or _AP["hazard_hold_s"] > 0.0

    # ---------- Lane change management ----------
    cur_lane = int(round(_as_float(state.get("current_lane")) or 1.0))
    tgt_lane = int(round(_as_float(state.get("target_lane")) or float(cur_lane)))
    cur_lane = int(_clamp(cur_lane, 0, 2))
    tgt_lane = int(_clamp(tgt_lane, 0, 2))

    lane_delta = tgt_lane - cur_lane
    lane_dir_req = -1 if lane_delta < 0 else (1 if lane_delta > 0 else 0)

    safe_gap = max(8.0, 4.0 + 0.6 * speed_ms)
    safe_left = left_clear > safe_gap
    safe_right = right_clear > safe_gap

    if _AP["lane_change_timer_s"] > 0.0:
        _AP["lane_change_timer_s"] = max(0.0, _AP["lane_change_timer_s"] - dt)
    else:
        _AP["lane_change_dir"] = 0

    if _AP["lane_change_timer_s"] <= 0.0 and lane_dir_req != 0:
        if lane_dir_req < 0 and safe_left:
            _AP["lane_change_dir"] = -1
            _AP["lane_change_timer_s"] = 2.8
        elif lane_dir_req > 0 and safe_right:
            _AP["lane_change_dir"] = 1
            _AP["lane_change_timer_s"] = 2.8

    # If obstacle is close and lane change is possible, prefer evasive lateral move.
    if front_dist < 35.0 and _AP["lane_change_timer_s"] <= 0.2:
        if safe_left and (left_clear >= right_clear or not safe_right):
            _AP["lane_change_dir"] = -1
            _AP["lane_change_timer_s"] = 2.8
        elif safe_right:
            _AP["lane_change_dir"] = 1
            _AP["lane_change_timer_s"] = 2.8

    # ---------- Traction-aware target speed ----------
    target_speed_kmh = _as_float(state.get("target_speed_kmh"))
    if target_speed_kmh is None:
        target_speed_kmh = 100.0

    speed_cap = min(target_speed_kmh, _AP["speed_limit_kmh"] + 2.0)

    # Reduce speed for low friction.
    mu_scale = max(0.35, 0.55 + 0.45 * mu)
    speed_cap *= mu_scale

    # Curvature-based speed cap.
    ca = abs(curvature_1pm)
    if ca > 1e-4:
        max_lat_acc = 1.2 + 3.2 * mu  # m/s^2
        curve_speed_ms = math.sqrt(max(0.2, max_lat_acc / ca))
        speed_cap = min(speed_cap, curve_speed_ms * 3.6)

    # Car-following cap for obstacle ahead.
    if obstacle_active and front_dist < 100.0:
        follow_speed_ms = max(0.0, (front_dist - 8.0) / 2.0)
        speed_cap = min(speed_cap, follow_speed_ms * 3.6)

    speed_cap = _clamp(speed_cap, 0.0, 180.0)

    # ---------- Longitudinal control ----------
    speed_err = speed_cap - speed_kmh
    _AP["speed_i"] = _clamp(_AP["speed_i"] + speed_err * dt * 0.10, -12.0, 12.0)

    throttle = _clamp(0.030 * speed_err + 0.020 * _AP["speed_i"], 0.0, 1.0)
    brake = 0.0

    if speed_err < -1.0:
        throttle = 0.0
        brake = _clamp((-speed_err) / 28.0, 0.0, 0.75)

    d_stop = (speed_ms * speed_ms) / (2.0 * max(1.5, 7.5 * mu)) + speed_ms * 0.35

    emergency = False
    if obstacle_active and front_dist < 120.0:
        comfort_ttc = 2.0 + (1.0 - mu) * 1.8
        risk = 0.0

        comfort_gap = d_stop + 6.0
        if front_dist < comfort_gap:
            risk = max(risk, (comfort_gap - front_dist) / max(comfort_gap, 1.0))

        if ttc_s < comfort_ttc:
            risk = max(risk, (comfort_ttc - ttc_s) / max(comfort_ttc, 0.5))

        brake = max(brake, _clamp(risk, 0.0, 1.0))
        throttle = 0.0

        if front_dist < max(4.0, 0.5 * speed_ms) or ttc_s < 0.7:
            emergency = True
            brake = 1.0
            throttle = 0.0

    # If lane change is active to avoid obstacle, cap non-emergency braking a bit to keep maneuverability.
    if _AP["lane_change_timer_s"] > 0.0 and not emergency and front_dist > 8.0:
        brake = min(brake, 0.65)

    # ---------- Lateral control ----------
    desired_offset = 0.0
    if _AP["lane_change_timer_s"] > 0.0:
        desired_offset = 1.2 * float(_AP["lane_change_dir"])

    lane_err = lane_offset_m - desired_offset
    _AP["lane_i"] = _clamp(_AP["lane_i"] + lane_err * dt, -2.0, 2.0)

    kp_lane = _clamp(0.38 + 0.006 * speed_kmh, 0.35, 0.95)
    ki_lane = 0.06
    yaw_damp = 0.018

    steer = 0.0
    steer += -kp_lane * lane_err
    steer += -ki_lane * _AP["lane_i"]
    steer += -yaw_damp * yaw_dps
    steer += _clamp(curvature_1pm * 10.0, -0.30, 0.30)

    if _AP["lane_change_timer_s"] > 0.0:
        steer += 0.08 * float(_AP["lane_change_dir"])

    # Lateral acceleration damping helps crosswind/gust recovery.
    lat_acc_cands = []
    for idx in (1, 2, 3):
        la = _as_float(state.get(f"accel_{idx}_lateral_ms2"))
        if la is None or abs(la) > 30.0:
            continue
        valid = _boolish(state.get(f"accel_{idx}_valid", True))
        lat_acc_cands.append((la, 1.0 if valid else 0.35))
    lat_acc, _ = _robust_fuse(lat_acc_cands, 0.0, jump_limit=4.0)
    steer += _clamp(-0.03 * lat_acc, -0.15, 0.15)

    # Steering sensor disagreement -> reduce aggressiveness.
    s1 = _as_float(state.get("steering_angle_1_deg"))
    s2 = _as_float(state.get("steering_angle_2_deg"))
    s1_ok = _boolish(state.get("steering_angle_1_valid", True)) and s1 is not None
    s2_ok = _boolish(state.get("steering_angle_2_valid", True)) and s2 is not None
    if s1_ok and s2_ok and abs(s1 - s2) > 8.0:
        steer *= 0.85

    # ---------- Brake failure inference ----------
    bp_vals = []
    for idx in (1, 2):
        bp = _as_float(state.get(f"brake_pressure_{idx}_pct"))
        bp_ok = _boolish(state.get(f"brake_pressure_{idx}_valid", True))
        if bp is not None and -5.0 <= bp <= 120.0:
            bp_vals.append(bp if bp_ok else 0.6 * bp)
    bp_avg = _median_or(0.0, bp_vals)

    if brake > 0.6 and bp_avg < 10.0:
        _AP["brake_fail_timer_s"] += dt
    else:
        _AP["brake_fail_timer_s"] = max(0.0, _AP["brake_fail_timer_s"] - 2.0 * dt)

    if _AP["brake_fail_timer_s"] > 0.8:
        # Suspected brake authority loss: cut throttle hard and keep requesting max brake.
        throttle = 0.0
        brake = 1.0

    # ---------- Traction shaping + rate limits ----------
    steer_limit = 0.95 if mu > 0.7 else (0.75 if mu > 0.4 else 0.55)
    steer = _clamp(steer, -steer_limit, steer_limit)

    throttle_cap = _clamp(0.15 + 0.85 * mu, 0.22, 1.0)
    throttle = min(throttle, throttle_cap)

    if brake > 0.03:
        throttle = 0.0

    if not emergency:
        brake = min(brake, 0.55 + 0.9 * mu)

    # Rate limiting for smoothness under flicker and low friction.
    max_steer_step = dt * (0.8 + 0.8 * mu)
    steer = _clamp(steer, _AP["last_steering"] - max_steer_step, _AP["last_steering"] + max_steer_step)

    max_thr_up = dt * (0.20 + 0.70 * mu)
    max_thr_down = dt * 1.8
    throttle = _clamp(throttle, _AP["last_throttle"] - max_thr_down, _AP["last_throttle"] + max_thr_up)

    if emergency:
        brake = 1.0
    else:
        max_brk_up = dt * (1.2 + 2.0 * mu)
        max_brk_down = dt * 1.5
        brake = _clamp(brake, _AP["last_brake"] - max_brk_down, _AP["last_brake"] + max_brk_up)

    throttle = _clamp(throttle, 0.0, 1.0)
    brake = _clamp(brake, 0.0, 1.0)
    steer = _clamp(steer, -1.0, 1.0)

    # ---------- Persist ----------
    _AP["speed_kmh"] = speed_kmh
    _AP["yaw_dps"] = yaw_dps
    _AP["lane_offset_m"] = lane_offset_m
    _AP["curvature_1pm"] = curvature_1pm
    _AP["last_steering"] = steer
    _AP["last_throttle"] = throttle
    _AP["last_brake"] = brake

    return {
        "steering": float(round(steer, 6)),
        "throttle": float(round(throttle, 6)),
        "brake": float(round(brake, 6)),
    }
