#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <vector>

using namespace std;

struct Edge {
    int u;
    int v;
    long long w;
};

struct AdjEdge {
    int to;
    long long w;
    int id;
};

struct DSU {
    vector<int> p;
    vector<int> r;

    explicit DSU(int n = 0) {
        init(n);
    }

    void init(int n) {
        p.resize(n);
        r.assign(n, 0);
        iota(p.begin(), p.end(), 0);
    }

    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }

    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

struct Candidate {
    int a;
    int b;
    long long cost;
    int u;
    int v;
    int bridgeEdge;
};

static const long long INF = numeric_limits<long long>::max() / 4;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, kInput;
    if (!(cin >> n >> m >> kInput)) {
        return 0;
    }

    vector<Edge> edges;
    edges.reserve(m);
    vector<vector<AdjEdge>> g(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges.push_back({u, v, w});
        g[u].push_back({v, w, i});
        g[v].push_back({u, w, i});
    }

    vector<int> terminals(kInput);
    for (int i = 0; i < kInput; ++i) {
        cin >> terminals[i];
    }

    sort(terminals.begin(), terminals.end());
    terminals.erase(unique(terminals.begin(), terminals.end()), terminals.end());
    const int k = static_cast<int>(terminals.size());

    if (k <= 1) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    vector<char> isTerminal(n, false);
    for (int t : terminals) isTerminal[t] = true;

    vector<long long> dist(n, INF);
    vector<int> owner(n, -1);
    vector<int> parent(n, -1);
    vector<int> parentEdge(n, -1);

    using State = tuple<long long, int, int>; // dist, owner, vertex
    priority_queue<State, vector<State>, greater<State>> pq;

    for (int i = 0; i < k; ++i) {
        int s = terminals[i];
        if (0 < dist[s] || (dist[s] == 0 && (owner[s] == -1 || i < owner[s]))) {
            dist[s] = 0;
            owner[s] = i;
            parent[s] = -1;
            parentEdge[s] = -1;
            pq.push({0, i, s});
        }
    }

    while (!pq.empty()) {
        auto [d, own, v] = pq.top();
        pq.pop();

        if (d != dist[v] || own != owner[v]) continue;

        for (const auto& e : g[v]) {
            if (d > INF - e.w) continue;
            long long nd = d + e.w;
            int to = e.to;
            if (nd < dist[to] || (nd == dist[to] && (owner[to] == -1 || own < owner[to]))) {
                dist[to] = nd;
                owner[to] = own;
                parent[to] = v;
                parentEdge[to] = e.id;
                pq.push({nd, own, to});
            }
        }
    }

    vector<Candidate> candidates;
    candidates.reserve(m);

    for (int id = 0; id < m; ++id) {
        int u = edges[id].u;
        int v = edges[id].v;
        int a = owner[u];
        int b = owner[v];
        if (a < 0 || b < 0 || a == b) continue;
        if (dist[u] >= INF || dist[v] >= INF) continue;

        int uu = u;
        int vv = v;
        if (a > b) {
            swap(a, b);
            swap(uu, vv);
        }

        long long cost = dist[uu] + edges[id].w + dist[vv];
        candidates.push_back({a, b, cost, uu, vv, id});
    }

    sort(candidates.begin(), candidates.end(), [](const Candidate& x, const Candidate& y) {
        if (x.a != y.a) return x.a < y.a;
        if (x.b != y.b) return x.b < y.b;
        if (x.cost != y.cost) return x.cost < y.cost;
        return x.bridgeEdge < y.bridgeEdge;
    });

    vector<Candidate> pairEdges;
    pairEdges.reserve(candidates.size());
    for (size_t i = 0; i < candidates.size();) {
        size_t j = i;
        Candidate best = candidates[i];
        while (j < candidates.size() && candidates[j].a == candidates[i].a && candidates[j].b == candidates[i].b) {
            if (candidates[j].cost < best.cost ||
                (candidates[j].cost == best.cost && candidates[j].bridgeEdge < best.bridgeEdge)) {
                best = candidates[j];
            }
            ++j;
        }
        pairEdges.push_back(best);
        i = j;
    }

    sort(pairEdges.begin(), pairEdges.end(), [](const Candidate& x, const Candidate& y) {
        if (x.cost != y.cost) return x.cost < y.cost;
        if (x.a != y.a) return x.a < y.a;
        if (x.b != y.b) return x.b < y.b;
        return x.bridgeEdge < y.bridgeEdge;
    });

    DSU dsuTerm(k);
    vector<Candidate> terminalMST;
    terminalMST.reserve(k - 1);

    for (const auto& c : pairEdges) {
        if (dsuTerm.unite(c.a, c.b)) {
            terminalMST.push_back(c);
            if (static_cast<int>(terminalMST.size()) == k - 1) break;
        }
    }

    if (static_cast<int>(terminalMST.size()) != k - 1) {
        cout << -1 << ' ' << 0 << '\n';
        return 0;
    }

    vector<char> inSubgraph(m, false);

    auto addPathToOwner = [&](int x) {
        while (x >= 0 && parent[x] != -1) {
            int eid = parentEdge[x];
            if (eid >= 0) inSubgraph[eid] = true;
            x = parent[x];
        }
    };

    for (const auto& c : terminalMST) {
        addPathToOwner(c.u);
        inSubgraph[c.bridgeEdge] = true;
        addPathToOwner(c.v);
    }

    vector<int> subEdges;
    subEdges.reserve(m);
    for (int id = 0; id < m; ++id) {
        if (inSubgraph[id]) subEdges.push_back(id);
    }

    sort(subEdges.begin(), subEdges.end(), [&](int x, int y) {
        if (edges[x].w != edges[y].w) return edges[x].w < edges[y].w;
        return x < y;
    });

    DSU dsuNode(n);
    vector<int> treeEdges;
    treeEdges.reserve(subEdges.size());

    for (int id : subEdges) {
        if (dsuNode.unite(edges[id].u, edges[id].v)) {
            treeEdges.push_back(id);
        }
    }

    int rootComp = dsuNode.find(terminals[0]);
    for (int t : terminals) {
        if (dsuNode.find(t) != rootComp) {
            cout << -1 << ' ' << 0 << '\n';
            return 0;
        }
    }

    vector<char> active(m, false);
    vector<vector<pair<int, int>>> treeAdj(n);
    vector<int> degree(n, 0);

    for (int id : treeEdges) {
        active[id] = true;
        int u = edges[id].u;
        int v = edges[id].v;
        treeAdj[u].push_back({v, id});
        treeAdj[v].push_back({u, id});
        ++degree[u];
        ++degree[v];
    }

    queue<int> q;
    for (int v = 0; v < n; ++v) {
        if (!isTerminal[v] && degree[v] == 1) {
            q.push(v);
        }
    }

    while (!q.empty()) {
        int v = q.front();
        q.pop();

        if (isTerminal[v] || degree[v] != 1) continue;

        int remEdge = -1;
        int nei = -1;
        for (const auto& [to, eid] : treeAdj[v]) {
            if (active[eid]) {
                remEdge = eid;
                nei = to;
                break;
            }
        }

        if (remEdge == -1) {
            degree[v] = 0;
            continue;
        }

        active[remEdge] = false;
        --degree[v];
        --degree[nei];

        if (!isTerminal[nei] && degree[nei] == 1) {
            q.push(nei);
        }
    }

    vector<int> finalEdges;
    finalEdges.reserve(treeEdges.size());
    long long totalWeight = 0;

    for (int id : treeEdges) {
        if (active[id]) {
            finalEdges.push_back(id);
            totalWeight += edges[id].w;
        }
    }

    DSU finalCheck(n);
    for (int id : finalEdges) {
        finalCheck.unite(edges[id].u, edges[id].v);
    }

    int r0 = finalCheck.find(terminals[0]);
    for (int t : terminals) {
        if (finalCheck.find(t) != r0) {
            cout << -1 << ' ' << 0 << '\n';
            return 0;
        }
    }

    cout << totalWeight << ' ' << finalEdges.size() << '\n';
    for (int id : finalEdges) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }

    return 0;
}
