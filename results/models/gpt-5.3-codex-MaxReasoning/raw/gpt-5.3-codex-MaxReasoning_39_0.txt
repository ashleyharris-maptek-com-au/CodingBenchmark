use std::cmp::Ordering;
use std::io::{self, Read};

fn compute_cut(side_a: &[bool], edges: &[(usize, usize)]) -> i64 {
    let mut cut = 0_i64;
    for &(u, v) in edges {
        if side_a[u] != side_a[v] {
            cut += 1;
        }
    }
    cut
}

fn improve_by_swaps(side_a: &mut [bool], adj: &[Vec<usize>], edge_exists: &[Vec<bool>]) -> i64 {
    let n = side_a.len();
    let mut cut = 0_i64;
    for u in 0..n {
        if side_a[u] {
            for &v in &adj[u] {
                if !side_a[v] {
                    cut += 1;
                }
            }
        }
    }

    loop {
        let mut best_delta = 0_i64;
        let mut best_a = usize::MAX;
        let mut best_b = usize::MAX;

        for a in 0..n {
            if !side_a[a] {
                continue;
            }
            let mut deg_a_to_a = 0_i64;
            let mut deg_a_to_b = 0_i64;
            for &x in &adj[a] {
                if side_a[x] {
                    deg_a_to_a += 1;
                } else {
                    deg_a_to_b += 1;
                }
            }
            let delta_a = deg_a_to_a - deg_a_to_b;

            for b in 0..n {
                if side_a[b] {
                    continue;
                }
                let mut deg_b_to_a = 0_i64;
                let mut deg_b_to_b = 0_i64;
                for &x in &adj[b] {
                    if side_a[x] {
                        deg_b_to_a += 1;
                    } else {
                        deg_b_to_b += 1;
                    }
                }
                let delta_b = deg_b_to_b - deg_b_to_a;

                let has_ab = if edge_exists[a][b] { 1_i64 } else { 0_i64 };
                let delta_swap = delta_a + delta_b + 2 * has_ab;

                if delta_swap < best_delta {
                    best_delta = delta_swap;
                    best_a = a;
                    best_b = b;
                }
            }
        }

        if best_delta < 0 {
            side_a[best_a] = false;
            side_a[best_b] = true;
            cut += best_delta;
        } else {
            break;
        }
    }

    cut
}

struct Solver {
    n: usize,
    target: usize,
    adj: Vec<Vec<usize>>,
    order: Vec<usize>,

    side: Vec<i8>, // -1 free, 0 -> A, 1 -> B
    to_a: Vec<i32>,
    to_b: Vec<i32>,

    current_cut: i64,
    best_cut: i64,
    best_side_a: Vec<bool>,
}

impl Solver {
    fn new(n: usize, adj: Vec<Vec<usize>>, initial_best_cut: i64, initial_best_side_a: Vec<bool>) -> Self {
        let mut order: Vec<usize> = (0..n).collect();
        order.sort_by(|&u, &v| {
            let du = adj[u].len();
            let dv = adj[v].len();
            dv.cmp(&du).then_with(|| u.cmp(&v))
        });

        Self {
            n,
            target: n / 2,
            adj,
            order,
            side: vec![-1; n],
            to_a: vec![0; n],
            to_b: vec![0; n],
            current_cut: 0,
            best_cut: initial_best_cut,
            best_side_a: initial_best_side_a,
        }
    }

    fn assign_vertex(&mut self, v: usize, place_a: bool) {
        self.side[v] = if place_a { 0 } else { 1 };

        for &u in &self.adj[v] {
            match self.side[u] {
                -1 => {
                    if place_a {
                        self.to_a[u] += 1;
                    } else {
                        self.to_b[u] += 1;
                    }
                }
                0 => {
                    if !place_a {
                        self.current_cut += 1;
                    }
                }
                1 => {
                    if place_a {
                        self.current_cut += 1;
                    }
                }
                _ => {}
            }
        }
    }

    fn unassign_vertex(&mut self, v: usize, place_a: bool) {
        for &u in &self.adj[v] {
            match self.side[u] {
                -1 => {
                    if place_a {
                        self.to_a[u] -= 1;
                    } else {
                        self.to_b[u] -= 1;
                    }
                }
                0 => {
                    if !place_a {
                        self.current_cut -= 1;
                    }
                }
                1 => {
                    if place_a {
                        self.current_cut -= 1;
                    }
                }
                _ => {}
            }
        }

        self.side[v] = -1;
    }

    fn lower_bound(&self, pos: usize, a_count: usize) -> i64 {
        let remaining = self.n - pos;
        if a_count > self.target {
            return i64::MAX / 4;
        }
        let need_a = self.target - a_count;
        if need_a > remaining {
            return i64::MAX / 4;
        }

        let mut base = 0_i64;
        let mut deltas: Vec<i32> = Vec::with_capacity(remaining);

        for i in pos..self.n {
            let v = self.order[i];
            let cost_a = self.to_b[v];
            let cost_b = self.to_a[v];
            base += cost_b as i64;
            deltas.push(cost_a - cost_b);
        }

        if need_a > 0 {
            let k = need_a;
            let kth_index = k - 1;
            deltas.select_nth_unstable(kth_index);
            for &d in &deltas[..k] {
                base += d as i64;
            }
        }

        self.current_cut + base
    }

    fn maybe_record_best(&mut self) {
        if self.current_cut < self.best_cut {
            self.best_cut = self.current_cut;
            for v in 0..self.n {
                self.best_side_a[v] = self.side[v] == 0;
            }
        }
    }

    fn dfs(&mut self, pos: usize, a_count: usize, b_count: usize) {
        if self.current_cut >= self.best_cut {
            return;
        }

        let lb = self.lower_bound(pos, a_count);
        if lb >= self.best_cut {
            return;
        }

        if pos == self.n {
            if a_count == self.target && b_count == self.target {
                self.maybe_record_best();
            }
            return;
        }

        let remaining = self.n - pos;
        let need_a = self.target.saturating_sub(a_count);
        let need_b = self.target.saturating_sub(b_count);

        if need_a > remaining || need_b > remaining {
            return;
        }

        let v = self.order[pos];

        if need_a == remaining {
            self.assign_vertex(v, true);
            self.dfs(pos + 1, a_count + 1, b_count);
            self.unassign_vertex(v, true);
            return;
        }

        if need_b == remaining {
            self.assign_vertex(v, false);
            self.dfs(pos + 1, a_count, b_count + 1);
            self.unassign_vertex(v, false);
            return;
        }

        // Branching heuristic: try the side with lower immediate added cut first.
        let add_if_a = self.to_b[v] as i64;
        let add_if_b = self.to_a[v] as i64;

        let mut branches = [(true, add_if_a), (false, add_if_b)];
        branches.sort_by(|x, y| {
            let c = x.1.cmp(&y.1);
            if c == Ordering::Equal {
                if x.0 == y.0 {
                    Ordering::Equal
                } else if x.0 {
                    Ordering::Less
                } else {
                    Ordering::Greater
                }
            } else {
                c
            }
        });

        for (place_a, _) in branches {
            if place_a {
                if a_count < self.target {
                    self.assign_vertex(v, true);
                    self.dfs(pos + 1, a_count + 1, b_count);
                    self.unassign_vertex(v, true);
                }
            } else if b_count < self.target {
                self.assign_vertex(v, false);
                self.dfs(pos + 1, a_count, b_count + 1);
                self.unassign_vertex(v, false);
            }
        }
    }

    fn solve(&mut self) {
        if self.n == 0 {
            self.best_cut = 0;
            return;
        }

        // Symmetry breaking: force one vertex into A.
        let root = self.order[0];
        self.assign_vertex(root, true);
        self.dfs(1, 1, 0);
        self.unassign_vertex(root, true);
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => {
            return;
        }
    };
    let m: usize = it.next().unwrap_or("0").parse().unwrap();

    let mut adj = vec![Vec::<usize>::new(); n];
    let mut edge_exists = vec![vec![false; n]; n];
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);

    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();

        if u >= n || v >= n || u == v {
            continue;
        }

        adj[u].push(v);
        adj[v].push(u);
        edge_exists[u][v] = true;
        edge_exists[v][u] = true;
        edges.push((u, v));
    }

    let target = n / 2;

    // Build an initial feasible bisection and improve it with local swaps.
    let mut initial_side_a = vec![false; n];
    let mut order0: Vec<usize> = (0..n).collect();
    order0.sort_by_key(|&v| std::cmp::Reverse(adj[v].len()));
    for &v in order0.iter().take(target) {
        initial_side_a[v] = true;
    }

    // Ensure symmetry-breaking root can be in A without worsening the initial bound.
    if n > 0 {
        let root = order0[0];
        if !initial_side_a[root] {
            for x in &mut initial_side_a {
                *x = !*x;
            }
        }
    }

    let mut improved_side = initial_side_a.clone();
    let improved_cut = improve_by_swaps(&mut improved_side, &adj, &edge_exists);

    let initial_cut = compute_cut(&initial_side_a, &edges);
    let (best_init_cut, best_init_side) = if improved_cut <= initial_cut {
        (improved_cut, improved_side)
    } else {
        (initial_cut, initial_side_a)
    };

    let mut solver = Solver::new(n, adj, best_init_cut, best_init_side);
    solver.solve();

    let mut part_a: Vec<usize> = (0..n).filter(|&v| solver.best_side_a[v]).collect();

    if part_a.len() != target {
        // Fallback repair (should not happen, but keeps output valid).
        part_a.clear();
        for v in 0..target {
            part_a.push(v);
        }
    }

    part_a.sort_unstable();

    println!("{}", solver.best_cut);
    if part_a.is_empty() {
        println!();
    } else {
        for (i, v) in part_a.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    }
}
