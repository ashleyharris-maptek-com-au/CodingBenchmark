#include <algorithm>
#include <cmath>
#include <cstdint>
#include <cstdio>
#include <limits>
#include <numeric>
#include <string>
#include <vector>

class FastInput {
public:
    FastInput() : pos_(0), len_(0) {}

    bool readInt(int& out) {
        char c = nextNonSpace();
        if (!c) return false;

        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = readChar();
        } else if (c == '+') {
            c = readChar();
        }

        int val = 0;
        while (c >= '0' && c <= '9') {
            val = val * 10 + (c - '0');
            c = readChar();
        }
        out = sign * val;
        return true;
    }

    bool readDouble(double& out) {
        char c = nextNonSpace();
        if (!c) return false;

        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = readChar();
        } else if (c == '+') {
            c = readChar();
        }

        double val = 0.0;
        while (c >= '0' && c <= '9') {
            val = val * 10.0 + static_cast<double>(c - '0');
            c = readChar();
        }

        if (c == '.') {
            double place = 0.1;
            c = readChar();
            while (c >= '0' && c <= '9') {
                val += static_cast<double>(c - '0') * place;
                place *= 0.1;
                c = readChar();
            }
        }

        if (c == 'e' || c == 'E') {
            int expSign = 1;
            int expVal = 0;
            c = readChar();
            if (c == '-') {
                expSign = -1;
                c = readChar();
            } else if (c == '+') {
                c = readChar();
            }
            while (c >= '0' && c <= '9') {
                expVal = expVal * 10 + (c - '0');
                c = readChar();
            }
            val *= std::pow(10.0, static_cast<double>(expSign * expVal));
        }

        out = sign * val;
        return true;
    }

private:
    static constexpr size_t kBufSize = 1u << 20;
    char buffer_[kBufSize];
    size_t pos_;
    size_t len_;

    char readChar() {
        if (pos_ >= len_) {
            len_ = std::fread(buffer_, 1, kBufSize, stdin);
            pos_ = 0;
            if (len_ == 0) return 0;
        }
        return buffer_[pos_++];
    }

    char nextNonSpace() {
        char c = readChar();
        while (c && c <= ' ') c = readChar();
        return c;
    }
};

static inline double dist2(const std::vector<float>& xs, const std::vector<float>& ys, int a, int b) {
    const double dx = static_cast<double>(xs[a]) - static_cast<double>(xs[b]);
    const double dy = static_cast<double>(ys[a]) - static_cast<double>(ys[b]);
    return dx * dx + dy * dy;
}

std::vector<int> solveExactBruteforce(const std::vector<float>& xs, const std::vector<float>& ys) {
    const int n = static_cast<int>(xs.size());
    if (n <= 1) return {0};

    std::vector<std::vector<double>> d(n, std::vector<double>(n, 0.0));
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            const double dist = std::sqrt(dist2(xs, ys, i, j));
            d[i][j] = dist;
            d[j][i] = dist;
        }
    }

    std::vector<int> perm(n - 1);
    std::iota(perm.begin(), perm.end(), 1);
    std::vector<int> bestPerm = perm;
    double bestLen = std::numeric_limits<double>::infinity();

    do {
        double len = d[0][perm[0]];
        for (int i = 0; i + 1 < n - 1; ++i) {
            len += d[perm[i]][perm[i + 1]];
            if (len >= bestLen) break;
        }
        if (len < bestLen) {
            len += d[perm.back()][0];
            if (len < bestLen) {
                bestLen = len;
                bestPerm = perm;
            }
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    route.insert(route.end(), bestPerm.begin(), bestPerm.end());
    return route;
}

std::vector<int> solveNearestNeighbor(const std::vector<float>& xs, const std::vector<float>& ys) {
    const int n = static_cast<int>(xs.size());
    std::vector<int> route;
    route.reserve(n);

    std::vector<unsigned char> used(n, 0);
    route.push_back(0);
    used[0] = 1;

    for (int step = 1; step < n; ++step) {
        const int cur = route.back();
        int best = -1;
        double bestD = std::numeric_limits<double>::infinity();

        for (int j = 1; j < n; ++j) {
            if (used[j]) continue;
            const double d = dist2(xs, ys, cur, j);
            if (d < bestD) {
                bestD = d;
                best = j;
            }
        }

        used[best] = 1;
        route.push_back(best);
    }

    return route;
}

void twoOptLocal(std::vector<int>& route, const std::vector<float>& xs, const std::vector<float>& ys,
                 int window, int maxPasses) {
    const int n = static_cast<int>(route.size());
    if (n < 4) return;

    for (int pass = 0; pass < maxPasses; ++pass) {
        bool improved = false;

        for (int i = 1; i + 1 < n; ++i) {
            int a = route[i - 1];
            int b = route[i];
            const int maxK = std::min(n - 1, i + window);

            for (int k = i + 1; k <= maxK; ++k) {
                const int c = route[k];
                const int d = (k + 1 < n) ? route[k + 1] : route[0];

                const double oldDist = dist2(xs, ys, a, b) + dist2(xs, ys, c, d);
                const double newDist = dist2(xs, ys, a, c) + dist2(xs, ys, b, d);

                if (newDist + 1e-12 < oldDist) {
                    std::reverse(route.begin() + i, route.begin() + (k + 1));
                    improved = true;
                    a = route[i - 1];
                    b = route[i];
                }
            }
        }

        if (!improved) break;
    }
}

static inline uint64_t splitBy1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return splitBy1(x) | (splitBy1(y) << 1);
}

struct KeyIndex {
    uint64_t key;
    uint32_t idx;
};

std::vector<int> solveMorton(const std::vector<float>& xs, const std::vector<float>& ys) {
    const int n = static_cast<int>(xs.size());
    if (n <= 1) return {0};

    float minX = xs[0], maxX = xs[0], minY = ys[0], maxY = ys[0];
    for (int i = 1; i < n; ++i) {
        minX = std::min(minX, xs[i]);
        maxX = std::max(maxX, xs[i]);
        minY = std::min(minY, ys[i]);
        maxY = std::max(maxY, ys[i]);
    }

    constexpr double kMaxU32 = 4294967295.0;
    const double scaleX = (maxX > minX) ? (kMaxU32 / static_cast<double>(maxX - minX)) : 0.0;
    const double scaleY = (maxY > minY) ? (kMaxU32 / static_cast<double>(maxY - minY)) : 0.0;

    std::vector<KeyIndex> keys;
    keys.reserve(static_cast<size_t>(n - 1));

    for (int i = 1; i < n; ++i) {
        double tx = (static_cast<double>(xs[i]) - static_cast<double>(minX)) * scaleX;
        double ty = (static_cast<double>(ys[i]) - static_cast<double>(minY)) * scaleY;

        if (tx < 0.0) tx = 0.0;
        if (tx > kMaxU32) tx = kMaxU32;
        if (ty < 0.0) ty = 0.0;
        if (ty > kMaxU32) ty = kMaxU32;

        const uint32_t qx = static_cast<uint32_t>(tx);
        const uint32_t qy = static_cast<uint32_t>(ty);
        keys.push_back(KeyIndex{morton2D(qx, qy), static_cast<uint32_t>(i)});
    }

    std::sort(keys.begin(), keys.end(), [](const KeyIndex& a, const KeyIndex& b) {
        if (a.key != b.key) return a.key < b.key;
        return a.idx < b.idx;
    });

    std::vector<int> route;
    route.resize(n);
    route[0] = 0;
    for (int i = 1; i < n; ++i) {
        route[i] = static_cast<int>(keys[i - 1].idx);
    }

    return route;
}

static inline void appendUInt(std::string& out, uint32_t v) {
    char tmp[16];
    int len = 0;
    do {
        tmp[len++] = static_cast<char>('0' + (v % 10));
        v /= 10;
    } while (v > 0);

    for (int i = len - 1; i >= 0; --i) out.push_back(tmp[i]);
}

void writeRoute(const std::vector<int>& route) {
    std::string out;
    out.reserve(1u << 20);

    for (size_t i = 0; i < route.size(); ++i) {
        if (i) out.push_back(' ');
        appendUInt(out, static_cast<uint32_t>(route[i]));

        if (out.size() >= (1u << 20)) {
            std::fwrite(out.data(), 1, out.size(), stdout);
            out.clear();
        }
    }

    out.push_back('\n');
    std::fwrite(out.data(), 1, out.size(), stdout);
}

int main() {
    FastInput in;

    int n = 0;
    if (!in.readInt(n)) return 0;
    if (n <= 0) {
        std::puts("");
        return 0;
    }

    std::vector<float> xs(static_cast<size_t>(n));
    std::vector<float> ys(static_cast<size_t>(n));

    for (int i = 0; i < n; ++i) {
        double x = 0.0, y = 0.0;
        if (!in.readDouble(x)) x = 0.0;
        if (!in.readDouble(y)) y = 0.0;
        xs[i] = static_cast<float>(x);
        ys[i] = static_cast<float>(y);
    }

    std::vector<int> route;

    if (n <= 11) {
        route = solveExactBruteforce(xs, ys);
    } else if (n <= 4000) {
        route = solveNearestNeighbor(xs, ys);
        const int window = (n <= 1000) ? 56 : 28;
        const int passes = (n <= 1500) ? 3 : 2;
        twoOptLocal(route, xs, ys, window, passes);
    } else {
        route = solveMorton(xs, ys);
        if (n <= 200000) {
            const int window = (n <= 50000) ? 24 : 12;
            twoOptLocal(route, xs, ys, window, 1);
        }
    }

    writeRoute(route);
    return 0;
}
