declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type TaskInput = {
  id: any;
  duration?: number;
  deps?: any[];
  deadline?: number;
  weight?: number;
};

type ScheduleItem = { id: any; start: number; machine: number };

type RunResult = {
  ok: boolean;
  score: number;
  makespan: number;
  starts: number[];
  machines: number[];
};

const LARGE_DEADLINE = 1_000_000_000_000;

class BinaryHeap {
  private data: number[] = [];
  private readonly higherPriority: (a: number, b: number) => boolean;

  constructor(higherPriority: (a: number, b: number) => boolean) {
    this.higherPriority = higherPriority;
  }

  size(): number {
    return this.data.length;
  }

  push(value: number): void {
    const a = this.data;
    a.push(value);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.higherPriority(a[i], a[p])) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }

  pop(): number | undefined {
    const a = this.data;
    const n = a.length;
    if (n === 0) return undefined;
    const top = a[0];
    const last = a.pop() as number;
    if (n > 1) {
      a[0] = last;
      this.siftDown(0);
    }
    return top;
  }

  private siftDown(i: number): void {
    const a = this.data;
    const n = a.length;
    while (true) {
      let best = i;
      const l = i * 2 + 1;
      const r = l + 1;
      if (l < n && this.higherPriority(a[l], a[best])) best = l;
      if (r < n && this.higherPriority(a[r], a[best])) best = r;
      if (best === i) break;
      const tmp = a[i];
      a[i] = a[best];
      a[best] = tmp;
      i = best;
    }
  }
}

function safeInt(v: any, fallback: number): number {
  const n = Number(v);
  if (!Number.isFinite(n)) return fallback;
  return Math.floor(n);
}

function safeNonNegNumber(v: any, fallback: number): number {
  const n = Number(v);
  if (!Number.isFinite(n)) return fallback;
  return n < 0 ? 0 : n;
}

const machineCountRaw = safeInt(data?.machineCount, 1);
const machineCount = Math.max(1, machineCountRaw);
const taskInputs: TaskInput[] = Array.isArray(data?.tasks) ? data.tasks : [];
const n = taskInputs.length;

if (n === 0) {
  process.stdout.write('{"schedule":[]}');
  process.exit(0);
}

const ids: any[] = new Array(n);
const durations: number[] = new Array(n);
const deadlines: number[] = new Array(n);
const weights: number[] = new Array(n);
const depsRaw: any[][] = new Array(n);

for (let i = 0; i < n; i++) {
  const t = taskInputs[i] || ({} as TaskInput);
  ids[i] = t.id;
  durations[i] = Math.max(0, safeInt(t.duration, 0));
  deadlines[i] = safeInt(t.deadline, LARGE_DEADLINE);
  weights[i] = safeNonNegNumber(t.weight, 1);
  depsRaw[i] = Array.isArray(t.deps) ? t.deps : [];
}

const idToIndex = new Map<any, number>();
for (let i = 0; i < n; i++) {
  // If duplicate ids exist, keep first occurrence as the dependency anchor.
  if (!idToIndex.has(ids[i])) idToIndex.set(ids[i], i);
}

const preds: number[][] = Array.from({ length: n }, () => []);
const succs: number[][] = Array.from({ length: n }, () => []);
const baseIndeg: number[] = new Array(n).fill(0);

for (let i = 0; i < n; i++) {
  const seen = new Set<number>();
  for (const depId of depsRaw[i]) {
    const p = idToIndex.get(depId);
    if (p === undefined) continue;
    if (p === i) continue;
    if (seen.has(p)) continue;
    seen.add(p);
    preds[i].push(p);
    succs[p].push(i);
  }
  baseIndeg[i] = preds[i].length;
}

function topologicalOrder(): number[] {
  const indeg = baseIndeg.slice();
  const q: number[] = [];
  let head = 0;
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);

  const order: number[] = [];
  while (head < q.length) {
    const u = q[head++];
    order.push(u);
    for (const v of succs[u]) {
      indeg[v]--;
      if (indeg[v] === 0) q.push(v);
    }
  }
  return order;
}

const topo = topologicalOrder();
if (topo.length !== n) {
  // Fallback for invalid cyclic input: return deterministic original order.
  const orderIds = ids.slice();
  process.stdout.write(JSON.stringify({ order: orderIds }));
  process.exit(0);
}

const criticalDur: number[] = new Array(n).fill(0);
const criticalWeight: number[] = new Array(n).fill(0);
for (let k = n - 1; k >= 0; k--) {
  const u = topo[k];
  let bestDurTail = 0;
  let bestWeightTail = 0;
  for (const v of succs[u]) {
    if (criticalDur[v] > bestDurTail) bestDurTail = criticalDur[v];
    if (criticalWeight[v] > bestWeightTail) bestWeightTail = criticalWeight[v];
  }
  criticalDur[u] = durations[u] + bestDurTail;
  criticalWeight[u] = weights[u] + bestWeightTail;
}

enum Strategy {
  UrgentWeighted = 0,
  CriticalPath = 1,
  DensityDeadline = 2,
  HeavyFirst = 3,
}

function compareDensity(a: number, b: number): number {
  // Compare weights[a] / durations[a] vs weights[b] / durations[b] without division when possible.
  const da = Math.max(1, durations[a]);
  const db = Math.max(1, durations[b]);
  const left = weights[a] * db;
  const right = weights[b] * da;
  if (left > right) return -1;
  if (left < right) return 1;
  return 0;
}

function makeComparator(strategy: Strategy, release: number[]): (a: number, b: number) => boolean {
  return (a: number, b: number): boolean => {
    const ra = release[a];
    const rb = release[b];

    const tardyA = ra + durations[a] - deadlines[a];
    const tardyB = rb + durations[b] - deadlines[b];
    const posTardyA = tardyA > 0 ? tardyA : 0;
    const posTardyB = tardyB > 0 ? tardyB : 0;

    const slackA = deadlines[a] - (ra + criticalDur[a]);
    const slackB = deadlines[b] - (rb + criticalDur[b]);

    if (strategy === Strategy.UrgentWeighted) {
      if (posTardyA !== posTardyB) return posTardyA > posTardyB;
      if (slackA !== slackB) return slackA < slackB;
      if (weights[a] !== weights[b]) return weights[a] > weights[b];
      const d = compareDensity(a, b);
      if (d !== 0) return d < 0;
      if (criticalDur[a] !== criticalDur[b]) return criticalDur[a] > criticalDur[b];
      return a < b;
    }

    if (strategy === Strategy.CriticalPath) {
      if (slackA !== slackB) return slackA < slackB;
      if (criticalDur[a] !== criticalDur[b]) return criticalDur[a] > criticalDur[b];
      if (criticalWeight[a] !== criticalWeight[b]) return criticalWeight[a] > criticalWeight[b];
      if (weights[a] !== weights[b]) return weights[a] > weights[b];
      if (deadlines[a] !== deadlines[b]) return deadlines[a] < deadlines[b];
      return a < b;
    }

    if (strategy === Strategy.DensityDeadline) {
      const d = compareDensity(a, b);
      if (d !== 0) return d < 0;
      if (deadlines[a] !== deadlines[b]) return deadlines[a] < deadlines[b];
      if (slackA !== slackB) return slackA < slackB;
      if (criticalDur[a] !== criticalDur[b]) return criticalDur[a] > criticalDur[b];
      return a < b;
    }

    // HeavyFirst
    if (weights[a] !== weights[b]) return weights[a] > weights[b];
    if (posTardyA !== posTardyB) return posTardyA > posTardyB;
    if (deadlines[a] !== deadlines[b]) return deadlines[a] < deadlines[b];
    if (criticalDur[a] !== criticalDur[b]) return criticalDur[a] > criticalDur[b];
    return a < b;
  };
}

function chooseMachine(releaseTime: number, machineAvail: number[]): { machine: number; start: number } {
  let bestMachine = 0;
  let bestStart = machineAvail[0] > releaseTime ? machineAvail[0] : releaseTime;

  for (let m = 1; m < machineAvail.length; m++) {
    const s = machineAvail[m] > releaseTime ? machineAvail[m] : releaseTime;
    if (s < bestStart || (s === bestStart && m < bestMachine)) {
      bestStart = s;
      bestMachine = m;
    }
  }

  return { machine: bestMachine, start: bestStart };
}

function runStrategy(strategy: Strategy): RunResult {
  const indeg = baseIndeg.slice();
  const release: number[] = new Array(n).fill(0);
  const machineAvail: number[] = new Array(machineCount).fill(0);
  const starts: number[] = new Array(n).fill(0);
  const machines: number[] = new Array(n).fill(0);
  const finishes: number[] = new Array(n).fill(0);

  const heap = new BinaryHeap(makeComparator(strategy, release));
  for (let i = 0; i < n; i++) if (indeg[i] === 0) heap.push(i);

  let scheduled = 0;
  let tardiness = 0;
  let makespan = 0;

  while (heap.size() > 0) {
    const u = heap.pop() as number;
    const r = release[u];

    const picked = chooseMachine(r, machineAvail);
    const start = picked.start;
    const finish = start + durations[u];

    starts[u] = start;
    machines[u] = picked.machine;
    finishes[u] = finish;
    machineAvail[picked.machine] = finish;

    if (finish > makespan) makespan = finish;
    if (finish > deadlines[u]) {
      tardiness += weights[u] * (finish - deadlines[u]);
    }

    scheduled++;

    for (const v of succs[u]) {
      if (finish > release[v]) release[v] = finish;
      indeg[v]--;
      if (indeg[v] === 0) heap.push(v);
    }
  }

  if (scheduled !== n) {
    return {
      ok: false,
      score: Number.POSITIVE_INFINITY,
      makespan: Number.POSITIVE_INFINITY,
      starts,
      machines,
    };
  }

  return {
    ok: true,
    score: tardiness + makespan,
    makespan,
    starts,
    machines,
  };
}

const strategies: Strategy[] = [
  Strategy.UrgentWeighted,
  Strategy.CriticalPath,
  Strategy.DensityDeadline,
  Strategy.HeavyFirst,
];

let best: RunResult | null = null;
for (const s of strategies) {
  const result = runStrategy(s);
  if (!result.ok) continue;
  if (
    best === null ||
    result.score < best.score ||
    (result.score === best.score && result.makespan < best.makespan)
  ) {
    best = result;
  }
}

if (best === null) {
  const orderIds = topo.map((i) => ids[i]);
  process.stdout.write(JSON.stringify({ order: orderIds }));
  process.exit(0);
}

const schedule: ScheduleItem[] = new Array(n);
for (let i = 0; i < n; i++) {
  schedule[i] = { id: ids[i], start: best.starts[i], machine: best.machines[i] };
}

schedule.sort((a, b) => {
  if (a.start !== b.start) return a.start - b.start;
  if (a.machine !== b.machine) return a.machine - b.machine;
  const ia = idToIndex.get(a.id) ?? 0;
  const ib = idToIndex.get(b.id) ?? 0;
  return ia - ib;
});

process.stdout.write(JSON.stringify({ schedule }));
