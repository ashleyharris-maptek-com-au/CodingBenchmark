use std::f64::consts::PI;
use std::fmt::Write as _;
use std::io::{self, Read};

const MAX_SCALE: f64 = 10.0;
const EPS: f64 = 1e-9;

#[derive(Clone, Copy)]
struct Vec2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

struct Template {
    hull: Vec<Vec2>,
    center: Vec2,
    radius: f64,
    area: f64,
    bbox_min: Vec2,
    bbox_max: Vec2,
}

fn cross(o: Vec2, a: Vec2, b: Vec2) -> f64 {
    (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x)
}

fn polygon_area_abs(poly: &[Vec2]) -> f64 {
    if poly.len() < 3 {
        return 0.0;
    }
    let mut s = 0.0;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        s += poly[i].x * poly[j].y - poly[i].y * poly[j].x;
    }
    0.5 * s.abs()
}

fn convex_hull(points: &[Vec2]) -> Vec<Vec2> {
    let mut pts = points.to_vec();
    pts.sort_by(|a, b| {
        let cx = a.x.total_cmp(&b.x);
        if cx.is_eq() {
            a.y.total_cmp(&b.y)
        } else {
            cx
        }
    });

    let mut uniq = Vec::with_capacity(pts.len());
    for p in pts {
        if uniq
            .last()
            .map(|q: &Vec2| (q.x - p.x).abs() <= 1e-12 && (q.y - p.y).abs() <= 1e-12)
            .unwrap_or(false)
        {
            continue;
        }
        uniq.push(p);
    }

    if uniq.len() <= 1 {
        return uniq;
    }

    let mut lower: Vec<Vec2> = Vec::new();
    for p in &uniq {
        while lower.len() >= 2 {
            let n = lower.len();
            if cross(lower[n - 2], lower[n - 1], *p) <= 0.0 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(*p);
    }

    let mut upper: Vec<Vec2> = Vec::new();
    for p in uniq.iter().rev() {
        while upper.len() >= 2 {
            let n = upper.len();
            if cross(upper[n - 2], upper[n - 1], *p) <= 0.0 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(*p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn average_point(poly: &[Vec2]) -> Vec2 {
    if poly.is_empty() {
        return Vec2 { x: 0.0, y: 0.0 };
    }
    let mut sx = 0.0;
    let mut sy = 0.0;
    for p in poly {
        sx += p.x;
        sy += p.y;
    }
    let inv = 1.0 / (poly.len() as f64);
    Vec2 {
        x: sx * inv,
        y: sy * inv,
    }
}

fn signed_dist_to_line(p: Vec2, a: Vec2, b: Vec2) -> f64 {
    let dx = b.x - a.x;
    let dy = b.y - a.y;
    let len = (dx * dx + dy * dy).sqrt();
    if len < EPS {
        0.0
    } else {
        ((dx) * (p.y - a.y) - (dy) * (p.x - a.x)) / len
    }
}

fn point_in_convex(poly: &[Vec2], p: Vec2) -> bool {
    if poly.len() < 3 {
        return false;
    }
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        if cross(poly[i], poly[j], p) < -1e-9 {
            return false;
        }
    }
    true
}

fn quat_normalize(q: Quat) -> Quat {
    let n = (q.w * q.w + q.x * q.x + q.y * q.y + q.z * q.z).sqrt();
    if n < EPS {
        Quat {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        }
    } else {
        let inv = 1.0 / n;
        Quat {
            w: q.w * inv,
            x: q.x * inv,
            y: q.y * inv,
            z: q.z * inv,
        }
    }
}

fn quat_from_euler(roll: f64, pitch: f64, yaw: f64) -> Quat {
    let (sr, cr) = (0.5 * roll).sin_cos();
    let (sp, cp) = (0.5 * pitch).sin_cos();
    let (sy, cy) = (0.5 * yaw).sin_cos();

    quat_normalize(Quat {
        w: cr * cp * cy + sr * sp * sy,
        x: sr * cp * cy - cr * sp * sy,
        y: cr * sp * cy + sr * cp * sy,
        z: cr * cp * sy - sr * sp * cy,
    })
}

fn quat_to_matrix(q: Quat) -> [[f64; 3]; 3] {
    let q = quat_normalize(q);
    let xx = q.x * q.x;
    let yy = q.y * q.y;
    let zz = q.z * q.z;
    let xy = q.x * q.y;
    let xz = q.x * q.z;
    let yz = q.y * q.z;
    let wx = q.w * q.x;
    let wy = q.w * q.y;
    let wz = q.w * q.z;

    [
        [1.0 - 2.0 * (yy + zz), 2.0 * (xy - wz), 2.0 * (xz + wy)],
        [2.0 * (xy + wz), 1.0 - 2.0 * (xx + zz), 2.0 * (yz - wx)],
        [2.0 * (xz - wy), 2.0 * (yz + wx), 1.0 - 2.0 * (xx + yy)],
    ]
}

fn rotate(m: &[[f64; 3]; 3], v: Vec3) -> Vec3 {
    Vec3 {
        x: m[0][0] * v.x + m[0][1] * v.y + m[0][2] * v.z,
        y: m[1][0] * v.x + m[1][1] * v.y + m[1][2] * v.z,
        z: m[2][0] * v.x + m[2][1] * v.y + m[2][2] * v.z,
    }
}

fn tetra_vertices() -> [Vec3; 4] {
    [
        Vec3 {
            x: 1.0,
            y: 0.0,
            z: -0.707,
        },
        Vec3 {
            x: -1.0,
            y: 0.0,
            z: -0.707,
        },
        Vec3 {
            x: 0.0,
            y: 1.0,
            z: 0.707,
        },
        Vec3 {
            x: 0.0,
            y: -1.0,
            z: 0.707,
        },
    ]
}

fn build_template(q: Quat, sun: Vec3, scale: f64) -> Template {
    let safe_sz = if sun.z.abs() < 1e-9 {
        if sun.z >= 0.0 {
            -1e-9
        } else {
            1e-9
        }
    } else {
        sun.z
    };
    let kx = sun.x / safe_sz;
    let ky = sun.y / safe_sz;

    let m = quat_to_matrix(q);
    let mut pts = Vec::with_capacity(4);
    for v in tetra_vertices() {
        let r = rotate(&m, v);
        let rx = r.x * scale;
        let ry = r.y * scale;
        let rz = r.z * scale;
        pts.push(Vec2 {
            x: rx - rz * kx,
            y: ry - rz * ky,
        });
    }

    let hull = convex_hull(&pts);
    if hull.len() < 3 {
        return Template {
            hull,
            center: Vec2 { x: 0.0, y: 0.0 },
            radius: 0.0,
            area: 0.0,
            bbox_min: Vec2 { x: 0.0, y: 0.0 },
            bbox_max: Vec2 { x: 0.0, y: 0.0 },
        };
    }

    let center = average_point(&hull);
    let mut radius = f64::INFINITY;
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;

    for i in 0..hull.len() {
        let j = (i + 1) % hull.len();
        radius = radius.min(signed_dist_to_line(center, hull[i], hull[j]));
    }
    for p in &hull {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }

    Template {
        hull,
        center,
        radius,
        area: polygon_area_abs(&pts),
        bbox_min: Vec2 { x: minx, y: miny },
        bbox_max: Vec2 { x: maxx, y: maxy },
    }
}

fn rand01(state: &mut u64) -> f64 {
    *state = state
        .wrapping_mul(6364136223846793005)
        .wrapping_add(1442695040888963407);
    let bits = *state >> 11;
    (bits as f64) * (1.0 / ((1u64 << 53) as f64))
}

fn random_quat(state: &mut u64) -> Quat {
    let u1 = rand01(state);
    let u2 = rand01(state);
    let u3 = rand01(state);

    let sq1 = (1.0 - u1).sqrt();
    let sq2 = u1.sqrt();
    let t1 = 2.0 * PI * u2;
    let t2 = 2.0 * PI * u3;

    quat_normalize(Quat {
        x: sq1 * t1.sin(),
        y: sq1 * t1.cos(),
        z: sq2 * t2.sin(),
        w: sq2 * t2.cos(),
    })
}

fn choose_orientation(sun: Vec3) -> (Quat, Template) {
    let mut best_q = Quat {
        w: 1.0,
        x: 0.0,
        y: 0.0,
        z: 0.0,
    };
    let mut best_t = build_template(best_q, sun, MAX_SCALE);
    let mut best_r = best_t.radius;
    let mut best_a = best_t.area;

    let eval = |q: Quat,
                sun: Vec3,
                best_q: &mut Quat,
                best_t: &mut Template,
                best_r: &mut f64,
                best_a: &mut f64| {
        let t = build_template(q, sun, MAX_SCALE);
        if t.hull.len() < 3 || t.radius <= 1e-10 || t.area <= 1e-10 {
            return;
        }
        let better_r = t.radius > *best_r + 1e-12;
        let tie_better_a = (t.radius - *best_r).abs() <= 1e-12 && t.area > *best_a;
        if better_r || tie_better_a {
            *best_q = q;
            *best_t = t;
            *best_r = best_t.radius;
            *best_a = best_t.area;
        }
    };

    let angles = [0.0, PI / 6.0, PI / 4.0, PI / 3.0, PI / 2.0];
    for &ax in &angles {
        for &ay in &angles {
            for &az in &angles {
                eval(
                    quat_from_euler(ax, ay, az),
                    sun,
                    &mut best_q,
                    &mut best_t,
                    &mut best_r,
                    &mut best_a,
                );
            }
        }
    }

    let mut state = 0x9E3779B97F4A7C15u64;
    for _ in 0..320 {
        let q = random_quat(&mut state);
        eval(q, sun, &mut best_q, &mut best_t, &mut best_r, &mut best_a);
    }

    if best_t.hull.len() < 3 || best_t.radius <= 0.0 {
        best_q = quat_normalize(Quat {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        });
        best_t = build_template(best_q, sun, MAX_SCALE);
    }

    (best_q, best_t)
}

fn translated_vertex_inside(template_hull: &[Vec2], translation: Vec2, p: Vec2) -> bool {
    point_in_convex(
        template_hull,
        Vec2 {
            x: p.x - translation.x,
            y: p.y - translation.y,
        },
    )
}

fn bbox_of_polygon(poly: &[Vec2]) -> (Vec2, Vec2) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (
        Vec2 { x: minx, y: miny },
        Vec2 { x: maxx, y: maxy },
    )
}

fn generate_square_grid(min: Vec2, max: Vec2, r: f64, template_center: Vec2) -> Vec<Vec2> {
    let step = (2.0f64).sqrt() * r * 0.98;
    let pad = 2.0 * r + step;
    let mut out = Vec::new();

    let y0 = min.y - pad;
    let y1 = max.y + pad;
    let x0 = min.x - pad;
    let x1 = max.x + pad;

    let mut y = y0;
    while y <= y1 + EPS {
        let mut x = x0;
        while x <= x1 + EPS {
            out.push(Vec2 {
                x: x - template_center.x,
                y: y - template_center.y,
            });
            x += step;
        }
        y += step;
    }

    out
}

fn generate_hex_grid(min: Vec2, max: Vec2, r: f64, template_center: Vec2) -> Vec<Vec2> {
    let dx = (3.0f64).sqrt() * r * 0.98;
    let dy = 1.5 * r * 0.98;
    let pad = 3.0 * r + dx;
    let mut out = Vec::new();

    let y0 = min.y - pad;
    let y1 = max.y + pad;
    let x0 = min.x - pad;
    let x1 = max.x + pad;

    let mut row: i64 = 0;
    let mut y = y0;
    while y <= y1 + EPS {
        let offset = if row & 1 == 0 { 0.0 } else { 0.5 * dx };
        let mut x = x0 + offset;
        while x <= x1 + EPS {
            out.push(Vec2 {
                x: x - template_center.x,
                y: y - template_center.y,
            });
            x += dx;
        }
        y += dy;
        row += 1;
    }

    out
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    if nv < 3 {
        println!("0");
        return;
    }

    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let y = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        poly.push(Vec2 { x, y });
    }

    let sx = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let sy = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let mut sz = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(-1.0);
    if sz.abs() < 1e-9 {
        sz = -1e-9;
    }

    let sun = Vec3 { x: sx, y: sy, z: sz };
    let (q, template) = choose_orientation(sun);

    let (bmin, bmax) = bbox_of_polygon(&poly);
    let mut effective_r = template.radius;
    if !effective_r.is_finite() || effective_r <= 1e-6 {
        effective_r = 0.25;
    }

    // Adaptive strategy by scale:
    // 1) Try single tetra for tiny cases.
    // 2) Use square grid for moderate coverage.
    // 3) Use hex grid for large-scale coverage density.
    let bbox_center = Vec2 {
        x: 0.5 * (bmin.x + bmax.x),
        y: 0.5 * (bmin.y + bmax.y),
    };
    let single_translation = Vec2 {
        x: bbox_center.x - template.center.x,
        y: bbox_center.y - template.center.y,
    };

    let mut translations: Vec<Vec2> = Vec::new();
    let mut single_ok = true;
    for p in &poly {
        if !translated_vertex_inside(&template.hull, single_translation, *p) {
            single_ok = false;
            break;
        }
    }

    if single_ok {
        translations.push(single_translation);
    } else {
        let bw = (bmax.x - bmin.x).max(0.0);
        let bh = (bmax.y - bmin.y).max(0.0);
        let bbox_area = bw * bh;
        let shadow_w = template.bbox_max.x - template.bbox_min.x;
        let shadow_h = template.bbox_max.y - template.bbox_min.y;
        let shadow_area_hint = (shadow_w * shadow_h).max(1e-9);
        let est = bbox_area / shadow_area_hint;

        if est < 2500.0 && nv < 200 {
            translations = generate_square_grid(bmin, bmax, effective_r, template.center);
        } else {
            translations = generate_hex_grid(bmin, bmax, effective_r, template.center);
        }
    }

    let safe_sz = if sun.z.abs() < 1e-9 { -1e-9 } else { sun.z };
    let kx = sun.x / safe_sz;
    let ky = sun.y / safe_sz;

    let max_vertex_norm = (1.0_f64 + 0.707_f64 * 0.707_f64).sqrt();
    let z_step = 2.5 * max_vertex_norm * MAX_SCALE;

    let mut out = String::new();
    writeln!(&mut out, "{}", translations.len()).unwrap();

    let qn = quat_normalize(q);
    for (i, t) in translations.iter().enumerate() {
        let z = z_step * (i as f64);
        let px = t.x + z * kx;
        let py = t.y + z * ky;
        let pz = z;
        writeln!(
            &mut out,
            "{:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9} {:.9}",
            px, py, pz, qn.w, qn.x, qn.y, qn.z, MAX_SCALE
        )
        .unwrap();
    }

    print!("{}", out);
}
