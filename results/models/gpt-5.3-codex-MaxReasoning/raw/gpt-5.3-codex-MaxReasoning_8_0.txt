use std::io::{self, BufRead, BufReader, BufWriter, Write};

struct BitSet {
    words: Vec<u64>,
}

impl BitSet {
    fn new(bits: usize) -> Self {
        Self {
            words: vec![0u64; (bits + 63) / 64],
        }
    }

    #[inline]
    fn get(&self, idx: usize) -> bool {
        ((self.words[idx >> 6] >> (idx & 63)) & 1) != 0
    }

    #[inline]
    fn set(&mut self, idx: usize) {
        self.words[idx >> 6] |= 1u64 << (idx & 63);
    }

    #[inline]
    fn set_if_unset(&mut self, idx: usize) -> bool {
        let w = idx >> 6;
        let bit = 1u64 << (idx & 63);
        let old = self.words[w];
        if (old & bit) != 0 {
            false
        } else {
            self.words[w] = old | bit;
            true
        }
    }
}

struct PackedDirs {
    bytes: Vec<u8>,
}

impl PackedDirs {
    fn new(n: usize) -> Self {
        Self {
            bytes: vec![0u8; (n + 3) / 4],
        }
    }

    #[inline]
    fn set(&mut self, idx: usize, dir: u8) {
        let b = idx >> 2;
        let shift = ((idx & 3) * 2) as u8;
        let clear_mask = !(0b11u8 << shift);
        self.bytes[b] = (self.bytes[b] & clear_mask) | ((dir & 0b11) << shift);
    }

    #[inline]
    fn get(&self, idx: usize) -> u8 {
        let b = idx >> 2;
        let shift = ((idx & 3) * 2) as u8;
        (self.bytes[b] >> shift) & 0b11
    }
}

#[inline]
fn step(idx: usize, dir: u8, w: usize) -> usize {
    match dir {
        0 => idx - w,
        1 => idx + w,
        2 => idx - 1,
        _ => idx + 1,
    }
}

#[inline]
fn reverse_dir(d: u8) -> u8 {
    d ^ 1
}

#[inline]
fn try_enqueue(
    v: usize,
    rev_dir_to_parent: u8,
    walls: &BitSet,
    seen_self: &mut BitSet,
    seen_other: &BitSet,
    parent_self: &mut PackedDirs,
    next: &mut Vec<u32>,
) -> Option<u32> {
    if walls.get(v) {
        return None;
    }
    if !seen_self.set_if_unset(v) {
        return None;
    }

    parent_self.set(v, rev_dir_to_parent);

    if seen_other.get(v) {
        return Some(v as u32);
    }

    next.push(v as u32);
    None
}

fn expand_layer(
    frontier: &[u32],
    next: &mut Vec<u32>,
    walls: &BitSet,
    seen_self: &mut BitSet,
    seen_other: &BitSet,
    parent_self: &mut PackedDirs,
    w: usize,
    n: usize,
) -> Option<u32> {
    next.clear();

    for &u32_idx in frontier {
        let idx = u32_idx as usize;

        if idx >= w {
            let v = idx - w;
            if let Some(m) = try_enqueue(v, reverse_dir(0), walls, seen_self, seen_other, parent_self, next)
            {
                return Some(m);
            }
        }

        if idx + w < n {
            let v = idx + w;
            if let Some(m) = try_enqueue(v, reverse_dir(1), walls, seen_self, seen_other, parent_self, next)
            {
                return Some(m);
            }
        }

        let col = idx % w;
        if col > 0 {
            let v = idx - 1;
            if let Some(m) = try_enqueue(v, reverse_dir(2), walls, seen_self, seen_other, parent_self, next)
            {
                return Some(m);
            }
        }

        if col + 1 < w {
            let v = idx + 1;
            if let Some(m) = try_enqueue(v, reverse_dir(3), walls, seen_self, seen_other, parent_self, next)
            {
                return Some(m);
            }
        }
    }

    None
}

fn parse_dims<R: BufRead>(reader: &mut R) -> io::Result<(usize, usize)> {
    let mut line = String::new();

    loop {
        line.clear();
        let got = reader.read_line(&mut line)?;
        if got == 0 {
            return Err(io::Error::new(
                io::ErrorKind::UnexpectedEof,
                "missing H W line",
            ));
        }

        if line.trim().is_empty() {
            continue;
        }

        let mut it = line.split_whitespace();
        let h = it
            .next()
            .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "missing H"))?
            .parse::<usize>()
            .map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "invalid H"))?;
        let w = it
            .next()
            .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "missing W"))?
            .parse::<usize>()
            .map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "invalid W"))?;

        return Ok((h, w));
    }
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    let (h, w) = parse_dims(&mut reader)?;
    if h == 0 || w == 0 {
        let mut out = BufWriter::new(io::stdout().lock());
        writeln!(out, "0")?;
        return Ok(());
    }

    let n = h
        .checked_mul(w)
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "maze too large"))?;

    if n > (u32::MAX as usize) {
        return Err(io::Error::new(
            io::ErrorKind::InvalidInput,
            "maze has too many cells for this implementation",
        ));
    }

    let mut walls = BitSet::new(n);
    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    let mut row = Vec::<u8>::with_capacity(w + 2);

    for r in 0..h {
        row.clear();
        let got = reader.read_until(b'\n', &mut row)?;
        if got == 0 {
            break;
        }

        if row.last() == Some(&b'\n') {
            row.pop();
        }
        if row.last() == Some(&b'\r') {
            row.pop();
        }

        if row.len() < w {
            row.resize(w, b' ');
        }

        let base = r * w;
        for c in 0..w {
            let ch = row[c];
            let idx = base + c;
            match ch {
                b'#' => walls.set(idx),
                b'A' => {
                    if start.is_none() {
                        start = Some(idx);
                    }
                }
                b'B' => {
                    if goal.is_none() {
                        goal = Some(idx);
                    }
                }
                _ => {}
            }
        }
    }

    let mut out = BufWriter::new(io::stdout().lock());

    let s = match start {
        Some(v) => v,
        None => {
            writeln!(out, "0")?;
            return Ok(());
        }
    };

    let t = match goal {
        Some(v) => v,
        None => {
            writeln!(out, "0")?;
            return Ok(());
        }
    };

    if s == t {
        writeln!(out, "1")?;
        writeln!(out, "{} {}", s / w, s % w)?;
        return Ok(());
    }

    if walls.get(s) || walls.get(t) {
        writeln!(out, "0")?;
        return Ok(());
    }

    let mut seen_s = BitSet::new(n);
    let mut seen_t = BitSet::new(n);
    seen_s.set(s);
    seen_t.set(t);

    let mut parent_s = PackedDirs::new(n);
    let mut parent_t = PackedDirs::new(n);

    let mut frontier_s = vec![s as u32];
    let mut frontier_t = vec![t as u32];
    let mut next_s: Vec<u32> = Vec::new();
    let mut next_t: Vec<u32> = Vec::new();

    let meet = loop {
        if frontier_s.is_empty() || frontier_t.is_empty() {
            break None;
        }

        if frontier_s.len() <= frontier_t.len() {
            if let Some(m) = expand_layer(
                &frontier_s,
                &mut next_s,
                &walls,
                &mut seen_s,
                &seen_t,
                &mut parent_s,
                w,
                n,
            ) {
                break Some(m as usize);
            }
            std::mem::swap(&mut frontier_s, &mut next_s);
        } else {
            if let Some(m) = expand_layer(
                &frontier_t,
                &mut next_t,
                &walls,
                &mut seen_t,
                &seen_s,
                &mut parent_t,
                w,
                n,
            ) {
                break Some(m as usize);
            }
            std::mem::swap(&mut frontier_t, &mut next_t);
        }
    };

    let meet = match meet {
        Some(v) => v,
        None => {
            writeln!(out, "0")?;
            return Ok(());
        }
    };

    let mut left: Vec<usize> = Vec::new();
    let mut cur = meet;
    left.push(cur);
    while cur != s {
        let d = parent_s.get(cur);
        cur = step(cur, d, w);
        left.push(cur);
    }
    left.reverse();

    let mut right: Vec<usize> = Vec::new();
    cur = meet;
    while cur != t {
        let d = parent_t.get(cur);
        cur = step(cur, d, w);
        right.push(cur);
    }

    let total_len = left.len() + right.len();
    writeln!(out, "{}", total_len)?;

    for idx in left {
        writeln!(out, "{} {}", idx / w, idx % w)?;
    }
    for idx in right {
        writeln!(out, "{} {}", idx / w, idx % w)?;
    }

    Ok(())
}
