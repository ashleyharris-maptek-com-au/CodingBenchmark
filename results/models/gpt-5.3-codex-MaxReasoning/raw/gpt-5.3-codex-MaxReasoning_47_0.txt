@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;
// params.x = difficulty, params.y = range_start, params.z = range_size

fn rotl32(x: u32, shift: u32) -> u32 {
    return (x << shift) | (x >> (32u - shift));
}

fn ctz32(x: u32) -> u32 {
    var n: u32 = 0u;
    var v: u32 = x;
    loop {
        if (((v & 1u) != 0u) || (n == 32u)) {
            break;
        }
        n = n + 1u;
        v = v >> 1u;
    }
    return n;
}

fn trailing_zeros_64(h0: u32, h1: u32) -> u32 {
    if (h0 != 0u) {
        return ctz32(h0);
    }
    if (h1 != 0u) {
        return 32u + ctz32(h1);
    }
    return 64u;
}

fn custom_hash(nonce: u32) -> vec2<u32> {
    var state: array<u32, 16>;

    for (var i: u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    // 8 rounds of quarter-round mixing (ChaCha-like)
    for (var round: u32 = 0u; round < 8u; round = round + 1u) {
        // state[0..3]
        state[0u] = state[0u] + state[1u];
        state[3u] = rotl32(state[3u] ^ state[0u], 16u);
        state[2u] = state[2u] + state[3u];
        state[1u] = rotl32(state[1u] ^ state[2u], 12u);
        state[0u] = state[0u] + state[1u];
        state[3u] = rotl32(state[3u] ^ state[0u], 8u);
        state[2u] = state[2u] + state[3u];
        state[1u] = rotl32(state[1u] ^ state[2u], 7u);

        // state[4..7]
        state[4u] = state[4u] + state[5u];
        state[7u] = rotl32(state[7u] ^ state[4u], 16u);
        state[6u] = state[6u] + state[7u];
        state[5u] = rotl32(state[5u] ^ state[6u], 12u);
        state[4u] = state[4u] + state[5u];
        state[7u] = rotl32(state[7u] ^ state[4u], 8u);
        state[6u] = state[6u] + state[7u];
        state[5u] = rotl32(state[5u] ^ state[6u], 7u);

        // state[8..11]
        state[8u]  = state[8u] + state[9u];
        state[11u] = rotl32(state[11u] ^ state[8u], 16u);
        state[10u] = state[10u] + state[11u];
        state[9u]  = rotl32(state[9u] ^ state[10u], 12u);
        state[8u]  = state[8u] + state[9u];
        state[11u] = rotl32(state[11u] ^ state[8u], 8u);
        state[10u] = state[10u] + state[11u];
        state[9u]  = rotl32(state[9u] ^ state[10u], 7u);

        // state[12..15]
        state[12u] = state[12u] + state[13u];
        state[15u] = rotl32(state[15u] ^ state[12u], 16u);
        state[14u] = state[14u] + state[15u];
        state[13u] = rotl32(state[13u] ^ state[14u], 12u);
        state[12u] = state[12u] + state[13u];
        state[15u] = rotl32(state[15u] ^ state[12u], 8u);
        state[14u] = state[14u] + state[15u];
        state[13u] = rotl32(state[13u] ^ state[14u], 7u);
    }

    let h0: u32 = state[0u] ^ state[2u] ^ state[4u] ^ state[6u] ^
                  state[8u] ^ state[10u] ^ state[12u] ^ state[14u];
    let h1: u32 = state[1u] ^ state[3u] ^ state[5u] ^ state[7u] ^
                  state[9u] ^ state[11u] ^ state[13u] ^ state[15u];

    return vec2<u32>(h0, h1);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid: vec3<u32>) {
    // Guard: out-of-range thread
    if (gid.x >= params.z) {
        return;
    }

    let nonce: u32 = params.y + gid.x;
    let h: vec2<u32> = custom_hash(nonce);
    let h0: u32 = h.x;
    let h1: u32 = h.y;

    let tz: u32 = trailing_zeros_64(h0, h1);

    if (tz >= params.x) {
        result[0u] = 1u;
        result[1u] = nonce;
        result[2u] = h0;
        result[3u] = h1;
    }
}
