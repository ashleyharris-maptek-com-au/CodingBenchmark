using System;
using System.Diagnostics;
using System.Globalization;
using System.IO;
using System.Numerics;
using System.Text;

public static class Program
{
    private const double Eps = 1e-12;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        string first = fs.Next();
        if (first == null)
        {
            return;
        }

        int n = int.Parse(first, CultureInfo.InvariantCulture);
        double mu = fs.NextDouble();

        State start = ReadState(fs);
        State[] stations = new State[n];
        for (int i = 0; i < n; i++)
        {
            stations[i] = ReadState(fs);
        }

        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }

        OrbitFeature[] features = new OrbitFeature[n + 1];
        features[0] = OrbitFeature.FromState(start, mu);
        for (int i = 0; i < n; i++)
        {
            features[i + 1] = OrbitFeature.FromState(stations[i], mu);
        }

        double[,] cost = BuildCostMatrix(features, mu);

        int[] order = n <= 12 ? SolveExact(n, cost) : SolveHeuristic(n, cost);
        Console.WriteLine(string.Join(" ", order));
    }

    private static State ReadState(FastScanner fs)
    {
        double x = fs.NextDouble();
        double y = fs.NextDouble();
        double z = fs.NextDouble();
        double vx = fs.NextDouble();
        double vy = fs.NextDouble();
        double vz = fs.NextDouble();
        return new State(new Vec3(x, y, z), new Vec3(vx, vy, vz));
    }

    private static double[,] BuildCostMatrix(OrbitFeature[] features, double mu)
    {
        int m = features.Length;
        var c = new double[m, m];
        for (int i = 0; i < m; i++)
        {
            c[i, i] = 0.0;
            for (int j = i + 1; j < m; j++)
            {
                double v = EstimateTransferCost(features[i], features[j], mu);
                c[i, j] = v;
                c[j, i] = v;
            }
        }
        return c;
    }

    private static double EstimateTransferCost(OrbitFeature a, OrbitFeature b, double mu)
    {
        double r1 = Math.Max(3000.0, a.RepresentativeRadius);
        double r2 = Math.Max(3000.0, b.RepresentativeRadius);

        double vc1 = Math.Sqrt(Math.Max(0.0, mu / r1));
        double vc2 = Math.Sqrt(Math.Max(0.0, mu / r2));
        double at = 0.5 * (r1 + r2);

        double term1 = 2.0 / r1 - 1.0 / at;
        double term2 = 2.0 / r2 - 1.0 / at;
        double vt1 = term1 > 0.0 ? Math.Sqrt(mu * term1) : vc1;
        double vt2 = term2 > 0.0 ? Math.Sqrt(mu * term2) : vc2;

        double dvHohmann = Math.Abs(vt1 - vc1) + Math.Abs(vc2 - vt2);
        if (!double.IsFinite(dvHohmann))
        {
            dvHohmann = Math.Abs(vc1 - vc2);
        }

        double planeDot = Clamp(Vec3.Dot(a.Normal, b.Normal), -1.0, 1.0);
        double planeAngle = Math.Acos(planeDot);
        double vMean = 0.5 * (vc1 + vc2);
        double dvPlane = 2.0 * vMean * Math.Sin(0.5 * planeAngle);

        double dvEcc = vMean * Math.Abs(a.Eccentricity - b.Eccentricity) * 0.5;

        double dr = (a.R - b.R).Norm();
        double dv = (a.V - b.V).Norm();
        double rScale = Math.Max(1.0, Math.Max(a.RMag, b.RMag));
        double dvGeom = vMean * (dr / rScale) * 0.08 + dv * 0.03;

        double dEnergy = Math.Abs(a.SpecificEnergy - b.SpecificEnergy);
        double energyScale = Math.Max(1.0, 0.5 * (vc1 * vc1 + vc2 * vc2));
        double dvEnergy = vMean * Math.Min(1.0, dEnergy / energyScale) * 0.08;

        double cost = 0.50 * dvHohmann
                    + 0.30 * dvPlane
                    + 0.10 * dvEcc
                    + 0.06 * dvGeom
                    + 0.04 * dvEnergy;

        return Math.Max(1e-6, cost);
    }

    private static int[] SolveExact(int n, double[,] c)
    {
        int states = 1 << n;
        double[,] dp = new double[states, n];
        int[,] parent = new int[states, n];

        for (int mask = 0; mask < states; mask++)
        {
            for (int j = 0; j < n; j++)
            {
                dp[mask, j] = double.PositiveInfinity;
                parent[mask, j] = -1;
            }
        }

        for (int i = 0; i < n; i++)
        {
            dp[1 << i, i] = c[0, i + 1];
        }

        for (int mask = 1; mask < states; mask++)
        {
            int bits = mask;
            while (bits != 0)
            {
                int last = BitOperations.TrailingZeroCount((uint)bits);
                bits &= bits - 1;

                int prevMask = mask ^ (1 << last);
                if (prevMask == 0)
                {
                    continue;
                }

                double best = double.PositiveInfinity;
                int bestPrev = -1;
                int prevBits = prevMask;
                while (prevBits != 0)
                {
                    int prev = BitOperations.TrailingZeroCount((uint)prevBits);
                    prevBits &= prevBits - 1;
                    double baseCost = dp[prevMask, prev];
                    if (!double.IsFinite(baseCost))
                    {
                        continue;
                    }

                    double cand = baseCost + c[prev + 1, last + 1];
                    if (cand < best)
                    {
                        best = cand;
                        bestPrev = prev;
                    }
                }

                dp[mask, last] = best;
                parent[mask, last] = bestPrev;
            }
        }

        int full = states - 1;
        double bestTotal = double.PositiveInfinity;
        int bestLast = 0;
        for (int last = 0; last < n; last++)
        {
            if (dp[full, last] < bestTotal)
            {
                bestTotal = dp[full, last];
                bestLast = last;
            }
        }

        int[] order = new int[n];
        int cur = bestLast;
        int curMask = full;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            order[pos] = cur;
            int prev = parent[curMask, cur];
            curMask ^= 1 << cur;
            cur = prev;
            if (cur < 0 && pos > 0)
            {
                break;
            }
        }

        return order;
    }

    private static int[] SolveHeuristic(int n, double[,] c)
    {
        long budgetMs = n <= 20 ? 1200 : n <= 40 ? 3000 : n <= 70 ? 7000 : 14000;
        var sw = Stopwatch.StartNew();

        int seed = unchecked(146959810 + n * 16777619);
        var rnd = new Random(seed);

        int[] best = BuildGreedyRoute(c, n, rnd, deterministic: true);
        LocalSearch(best, c, sw, budgetMs, n <= 40 ? 350 : 450);
        double bestCost = RouteCost(best, c);

        int iter = 0;
        while (sw.ElapsedMilliseconds < budgetMs)
        {
            int[] cand;
            if (iter % 6 == 0)
            {
                cand = (int[])best.Clone();
                Perturb(cand, rnd);
            }
            else
            {
                cand = BuildGreedyRoute(c, n, rnd, deterministic: false);
            }

            int maxPasses = n <= 30 ? 220 : (n <= 60 ? 140 : 90);
            LocalSearch(cand, c, sw, budgetMs, maxPasses);
            double candCost = RouteCost(cand, c);
            if (candCost + Eps < bestCost)
            {
                best = cand;
                bestCost = candCost;
            }

            iter++;
        }

        return best;
    }

    private static int[] BuildGreedyRoute(double[,] c, int n, Random rnd, bool deterministic)
    {
        int[] route = new int[n];
        bool[] used = new bool[n];
        int currentNode = 0;

        int topK = n <= 20 ? 3 : (n <= 60 ? 4 : 6);
        double[] bestCosts = new double[topK];
        int[] bestIds = new int[topK];

        for (int pos = 0; pos < n; pos++)
        {
            for (int k = 0; k < topK; k++)
            {
                bestCosts[k] = double.PositiveInfinity;
                bestIds[k] = -1;
            }

            int available = 0;
            for (int s = 0; s < n; s++)
            {
                if (used[s])
                {
                    continue;
                }

                available++;
                double v = c[currentNode, s + 1];
                if (v >= bestCosts[topK - 1])
                {
                    continue;
                }

                int ins = topK - 1;
                while (ins > 0 && v < bestCosts[ins - 1])
                {
                    bestCosts[ins] = bestCosts[ins - 1];
                    bestIds[ins] = bestIds[ins - 1];
                    ins--;
                }

                bestCosts[ins] = v;
                bestIds[ins] = s;
            }

            int count = Math.Min(topK, available);
            int chosen;
            if (deterministic || count <= 1)
            {
                chosen = bestIds[0];
            }
            else
            {
                double r = rnd.NextDouble();
                int pick = (int)(r * r * count); // bias toward the best candidates
                if (pick >= count)
                {
                    pick = count - 1;
                }
                chosen = bestIds[pick];
            }

            if (chosen < 0)
            {
                for (int s = 0; s < n; s++)
                {
                    if (!used[s])
                    {
                        chosen = s;
                        break;
                    }
                }
            }

            route[pos] = chosen;
            used[chosen] = true;
            currentNode = chosen + 1;
        }

        return route;
    }

    private static void Perturb(int[] route, Random rnd)
    {
        int n = route.Length;
        if (n < 4)
        {
            return;
        }

        int ops = 2 + n / 25;
        for (int t = 0; t < ops; t++)
        {
            if (rnd.NextDouble() < 0.5)
            {
                int i = rnd.Next(n);
                int j = rnd.Next(n);
                if (i == j)
                {
                    continue;
                }

                int tmp = route[i];
                route[i] = route[j];
                route[j] = tmp;
            }
            else
            {
                int i = rnd.Next(n - 1);
                int j = i + 1 + rnd.Next(n - i - 1);
                Array.Reverse(route, i, j - i + 1);
            }
        }
    }

    private static void LocalSearch(int[] route, double[,] c, Stopwatch sw, long budgetMs, int maxPasses)
    {
        for (int pass = 0; pass < maxPasses && sw.ElapsedMilliseconds < budgetMs; pass++)
        {
            if (TwoOptFirstImprovement(route, c, sw, budgetMs))
            {
                continue;
            }

            if (RelocateFirstImprovement(route, c, sw, budgetMs))
            {
                continue;
            }

            break;
        }
    }

    private static bool TwoOptFirstImprovement(int[] route, double[,] c, Stopwatch sw, long budgetMs)
    {
        int n = route.Length;
        for (int i = 0; i < n - 1; i++)
        {
            int pre = i == 0 ? 0 : route[i - 1] + 1;
            int a = route[i] + 1;

            for (int j = i + 1; j < n; j++)
            {
                if (sw.ElapsedMilliseconds >= budgetMs)
                {
                    return false;
                }

                int b = route[j] + 1;
                int post = j == n - 1 ? -1 : route[j + 1] + 1;

                double oldCost = c[pre, a] + (post >= 0 ? c[b, post] : 0.0);
                double newCost = c[pre, b] + (post >= 0 ? c[a, post] : 0.0);

                if (newCost + Eps < oldCost)
                {
                    Array.Reverse(route, i, j - i + 1);
                    return true;
                }
            }
        }

        return false;
    }

    private static bool RelocateFirstImprovement(int[] route, double[,] c, Stopwatch sw, long budgetMs)
    {
        int n = route.Length;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j <= n; j++)
            {
                if (sw.ElapsedMilliseconds >= budgetMs)
                {
                    return false;
                }

                if (j == i || j == i + 1)
                {
                    continue;
                }

                double delta = RelocateDelta(route, c, i, j);
                if (delta + Eps < 0.0)
                {
                    ApplyRelocate(route, i, j);
                    return true;
                }
            }
        }

        return false;
    }

    private static double RelocateDelta(int[] route, double[,] c, int i, int j)
    {
        int n = route.Length;

        int aPrev = i == 0 ? 0 : route[i - 1] + 1;
        int a = route[i] + 1;
        int aNext = i == n - 1 ? -1 : route[i + 1] + 1;

        double beforeRemoval = c[aPrev, a] + (aNext >= 0 ? c[a, aNext] : 0.0);
        double afterRemoval = aNext >= 0 ? c[aPrev, aNext] : 0.0;
        double delta = afterRemoval - beforeRemoval;

        int reducedLen = n - 1;
        int j2 = j > i ? j - 1 : j;

        int left = 0;
        if (j2 > 0)
        {
            int leftIdxReduced = j2 - 1;
            int leftIdxOriginal = leftIdxReduced < i ? leftIdxReduced : leftIdxReduced + 1;
            left = route[leftIdxOriginal] + 1;
        }

        int right = -1;
        if (j2 < reducedLen)
        {
            int rightIdxReduced = j2;
            int rightIdxOriginal = rightIdxReduced < i ? rightIdxReduced : rightIdxReduced + 1;
            right = route[rightIdxOriginal] + 1;
        }

        double beforeInsert = right >= 0 ? c[left, right] : 0.0;
        double afterInsert = c[left, a] + (right >= 0 ? c[a, right] : 0.0);
        delta += afterInsert - beforeInsert;

        return delta;
    }

    private static void ApplyRelocate(int[] route, int i, int j)
    {
        int n = route.Length;
        int node = route[i];

        if (j > i)
        {
            int count = j - i - 1;
            if (count > 0)
            {
                Array.Copy(route, i + 1, route, i, count);
            }
            route[j - 1] = node;
        }
        else
        {
            int count = i - j;
            if (count > 0)
            {
                Array.Copy(route, j, route, j + 1, count);
            }
            route[j] = node;
        }
    }

    private static double RouteCost(int[] route, double[,] c)
    {
        int n = route.Length;
        if (n == 0)
        {
            return 0.0;
        }

        double total = c[0, route[0] + 1];
        for (int i = 0; i < n - 1; i++)
        {
            total += c[route[i] + 1, route[i + 1] + 1];
        }

        return total;
    }

    private static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }

    private readonly struct State
    {
        public readonly Vec3 R;
        public readonly Vec3 V;

        public State(Vec3 r, Vec3 v)
        {
            R = r;
            V = v;
        }
    }

    private readonly struct OrbitFeature
    {
        public readonly Vec3 R;
        public readonly Vec3 V;
        public readonly Vec3 Normal;
        public readonly double RMag;
        public readonly double Eccentricity;
        public readonly double SpecificEnergy;
        public readonly double RepresentativeRadius;

        public OrbitFeature(
            Vec3 r,
            Vec3 v,
            Vec3 normal,
            double rMag,
            double eccentricity,
            double specificEnergy,
            double representativeRadius)
        {
            R = r;
            V = v;
            Normal = normal;
            RMag = rMag;
            Eccentricity = eccentricity;
            SpecificEnergy = specificEnergy;
            RepresentativeRadius = representativeRadius;
        }

        public static OrbitFeature FromState(State s, double mu)
        {
            Vec3 r = s.R;
            Vec3 v = s.V;

            double rMag = Math.Max(1e-9, r.Norm());
            double vMag = v.Norm();

            Vec3 h = Vec3.Cross(r, v);
            double hMag = h.Norm();
            Vec3 normal = hMag > 1e-12 ? h / hMag : new Vec3(0.0, 0.0, 1.0);

            double specificEnergy = 0.5 * vMag * vMag - mu / rMag;
            double a = Math.Abs(specificEnergy) > 1e-12 ? -mu / (2.0 * specificEnergy) : rMag;

            Vec3 eVec = Vec3.Cross(v, h) * (1.0 / mu) - r * (1.0 / rMag);
            double ecc = eVec.Norm();

            double repRadius = a;
            if (!double.IsFinite(repRadius) || repRadius <= 1000.0)
            {
                repRadius = rMag;
            }

            if (!double.IsFinite(repRadius) || repRadius <= 1000.0)
            {
                repRadius = 1000.0;
            }

            return new OrbitFeature(r, v, normal, rMag, ecc, specificEnergy, repRadius);
        }
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double Norm()
        {
            return Math.Sqrt(X * X + Y * Y + Z * Z);
        }

        public static Vec3 operator +(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        }

        public static Vec3 operator -(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        }

        public static Vec3 operator *(Vec3 a, double k)
        {
            return new Vec3(a.X * k, a.Y * k, a.Z * k);
        }

        public static Vec3 operator /(Vec3 a, double k)
        {
            return new Vec3(a.X / k, a.Y / k, a.Z / k);
        }

        public static double Dot(Vec3 a, Vec3 b)
        {
            return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        }

        public static Vec3 Cross(Vec3 a, Vec3 b)
        {
            return new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_ptr++];
        }

        public string Next()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1)
                {
                    return null;
                }
            } while (c <= 32);

            var sb = new StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = Read();
                if (c == -1)
                {
                    break;
                }
            }

            return sb.ToString();
        }

        public double NextDouble()
        {
            string s = Next();
            if (s == null)
            {
                return 0.0;
            }
            return double.Parse(s, CultureInfo.InvariantCulture);
        }
    }
}
