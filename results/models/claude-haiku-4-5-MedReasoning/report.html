            while let Some(u) = stack.pop() {
                if visited[u] {
                    continue;

                visited[u] = true;
                component_id[u] = comp_count;

                for &(a, b) in edges {
                    let neighbor = if a == u { b } else if b == u { a } else { continue };

                    }

        comp_nodes[component_id[i]].push(i);

fn construct_snake_path(width: usize, height: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    let mut path = Vec::new();

    // Create snake pattern: zigzag through grid
    for y in 0..height {
        if y % 2 == 0 {
            // Left to right
            for x in 0..width {
                path.push((x, y));
            }
        } else {
            // Right to left
            for x in (0..width).rev() {
                path.push((x, y));
            }
        }

    if require_cycle {
        // Check if last cell is adjacent to (0, 0)
        let (lx, ly) = path[path.len() - 1];
        let is_adjacent = (lx == 1 && ly == 0) || (lx == 0 && ly == 1);
        if !is_adjacent {
            return None;
        }
    }

    Some(path)
}

fn backtrack(
    path: &mut Vec<(usize, usize)>,
    visited: &mut HashSet<(usize, usize)>,
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
    total_cells: usize,
) -> bool {
    if visited.len() == total_cells {
        if require_cycle {
            let (lx, ly) = path[path.len() - 1];
            return (lx == 1 && ly == 0) || (lx == 0 && ly == 1);
        }
        return true;
    }

    let (x, y) = path[path.len() - 1];

    // Get and sort candidate neighbors
    let mut candidates = Vec::new();
    if x > 0 { candidates.push((x - 1, y)); }
    if x < width - 1 { candidates.push((x + 1, y)); }
    if y > 0 { candidates.push((x, y - 1)); }
    if y < height - 1 { candidates.push((x, y + 1)); }

    // Heuristic: prioritize moves to cells with more unvisited neighbors
    // This helps avoid dead-ends early
    candidates.sort_by_key(|&(nx, ny)| {
        if visited.contains(&(nx, ny)) || obstacles.contains(&(nx, ny)) {
            return std::usize::MAX;
        }

        let mut unvisited_count = 0;

        let neighbors = vec![
            (nx.saturating_sub(1), ny),
            (nx + 1, ny),
            (nx, ny.saturating_sub(1)),
            (nx, ny + 1),
        ];

        for (nnx, nny) in neighbors {
            if nnx < width && nny < height && !visited.contains(&(nnx, nny)) && !obstacles.contains(&(nnx, nny)) {
                unvisited_count += 1;
            }
        }

        // Return negative so higher counts are tried first
        std::cmp::Reverse(unvisited_count)
    });

    for (nx, ny) in candidates {
        if !visited.contains(&(nx, ny)) && !obstacles.contains(&(nx, ny)) {
            path.push((nx, ny));
            visited.insert((nx, ny));

            if backtrack(path, visited, width, height, obstacles, require_cycle, total_cells) {
                return true;
            }

            path.pop();
            visited.remove(&(nx, ny));
        }
    }

    false
}

int computeWindingNumber(const Vec3& point, const Mesh& mesh) {
    double angle_sum = 0.0;

    for (const auto& face : mesh.faces) {
        const Vec3& v0 = mesh.vertices[face.a];
        const Vec3& v1 = mesh.vertices[face.b];
        const Vec3& v2 = mesh.vertices[face.c];

        // Vectors from point to vertices
        Vec3 a = v0 - point;
        Vec3 b = v1 - point;
        Vec3 c = v2 - point;

        double alen = a.length();
        double blen = b.length();
        double clen = c.length();

        // Skip degenerate triangles or if point is too close to vertex
        if (alen < EPS || blen < EPS || clen < EPS) continue;

        // Normalize vectors
        a = a / alen;
        b = b / blen;
        c = c / clen;

        // Compute signed solid angle
        double numerator = a.dot(b.cross(c));
        double denominator = 1.0 + a.dot(b) + b.dot(c) + c.dot(a);

#include <vector>
#include <algorithm>
#include <cstring>

struct Rect {
    int w, h, id;
};

struct Placement {
    int board, x, y, rotated;
};

struct FreeRect {
    int x, y, w, h;
};

struct Board {
    int width, height;
    std::vector<FreeRect> freeRects;

    Board(int w, int h) : width(w), height(h) {
        freeRects.push_back({0, 0, w, h});
    }

    bool place(int w, int h, int& px, int& py) {
        // Find first free rectangle that fits
        for (int i = 0; i < (int)freeRects.size(); i++) {
            FreeRect& fr = freeRects[i];
            if (fr.w >= w && fr.h >= h) {
                px = fr.x;
                py = fr.y;

                // Create new free rectangles from the split
                FreeRect right{fr.x + w, fr.y, fr.w - w, h};
                FreeRect top{fr.x, fr.y + h, fr.w, fr.h - h};

                // Remove the used rectangle
                freeRects.erase(freeRects.begin() + i);

                // Add new free rectangles if they have area
                if (right.w > 0 && right.h > 0) {
                    freeRects.push_back(right);
                }
                if (top.w > 0 && top.h > 0) {
                    freeRects.push_back(top);
                }

                return true;
            }
        }
        return false;
    }
};

struct Point {

    y: f64,
}

#[derive(Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

#[derive(Clone)]
struct PlacedPiece {
    polygon: Polygon,
    rotation: f64,
    tx: f64,
    ty: f64,
}

impl Polygon {
    fn from_coords(coords: Vec<(f64, f64)>) -> Self {
        Polygon {
            vertices: coords.into_iter().map(|(x, y)| Point { x, y }).collect(),
        }
    }

    fn bbox(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            min_x = min_x.min(v.x);
            max_x = max_x.max(v.x);
            min_y = min_y.min(v.y);
            max_y = max_y.max(v.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn transformed_bbox(&self, rotation: f64, tx: f64, ty: f64) -> (f64, f64, f64, f64) {
        let cos_a = rotation.cos();
        let sin_a = rotation.sin();
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            let rx = v.x * cos_a - v.y * sin_a + tx;
            let ry = v.x * sin_a + v.y * cos_a + ty;
            min_x = min_x.min(rx);
            max_x = max_x.max(rx);
            min_y = min_y.min(ry);
            max_y = max_y.max(ry);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn get_transformed_vertices(&self, rotation: f64, tx: f64, ty: f64) -> Vec<Point> {
        let cos_a = rotation.cos();
        let sin_a = rotation.sin();
        self.vertices
            .iter()
            .map(|v| {
                let rx = v.x * cos_a - v.y * sin_a + tx;
                let ry = v.x * sin_a + v.y * cos_a + ty;
                Point { x: rx, y: ry }
            })
            .collect()
    }

    fn contains_point(&self, p: &Point) -> bool {
        point_in_polygon(&self.vertices, p)
    }

    fn contains_polygon(&self, other: &[Point]) -> bool {
        other.iter().all(|p| self.contains_point(p))

    fn bbox(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            min_x = min_x.min(v.x);
            max_x = max_x.max(v.x);
            min_y = min_y.min(v.y);
            max_y = max_y.max(v.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn transformed_bbox(&self, rotation: f64, tx: f64, ty: f64) -> (f64, f64, f64, f64) {
        let cos_a = rotation.cos();
        let sin_a = rotation.sin();
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            let rx = v.x * cos_a - v.y * sin_a + tx;
            let ry = v.x * sin_a + v.y * cos_a + ty;
            min_x = min_x.min(rx);
            max_x = max_x.max(rx);
            min_y = min_y.min(ry);
            max_y = max_y.max(ry);
        }

        (min_x, min_y, max_x, max_y)
    }

public struct Quaternion
{
    public float W, X, Y, Z;

    public Quaternion(float w, float x, float y, float z) { W = w; X = x; Y = y; Z = z; }

    public static Quaternion Identity => new Quaternion(1, 0, 0, 0);

    public Quaternion Normalized()
    {
        float len = (float)Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
        if (len < 0.0001f) return Identity;
        return new Quaternion(W / len, X / len, Y / len, Z / len);
    }

    public Vector3 RotateVector(Vector3 v)
    {
        // Rotation formula: p' = q * p * q^(-1)
        // Using: v' = v + 2*w*(q_xyz × v) + 2*(q_xyz × (q_xyz × v))
        float qw = W, qx = X, qy = Y, qz = Z;
        Vector3 qxyz = new Vector3(qx, qy, qz);
        Vector3 cross1 = qxyz.Cross(v);
        Vector3 cross2 = qxyz.Cross(cross1);
        return v + cross1 * (2 * qw) + cross2 * 2;
    }
}

fn point_in_polygon(point: (f64, f64), polygon: &[(f64, f64)]) -> bool {
    let (px, py) = point;
    let mut inside = false;
    let mut j = polygon.len() - 1;

    for i in 0..polygon.len() {
        let (x1, y1) = polygon[i];
        let (x2, y2) = polygon[j];

        if ((y1 > py) != (y2 > py)) &&
           (px < (x2 - x1) * (py - y1) / (y2 - y1) + x1) {
            inside = !inside;
        }
        j = i;
    }

    inside
}

fn point_near_polygon(point: (f64, f64), polygon: &[(f64, f64)], margin: f64) -> bool {
    let (px, py) = point;

    for i in 0..polygon.len() {
        let (x1, y1) = polygon[i];
        let (x2, y2) = polygon[(i + 1) % polygon.len()];

        let dx = x2 - x1;
        let dy = y2 - y1;
        let len_sq = dx * dx + dy * dy;

        if len_sq > 0.0 {
            let t = ((px - x1) * dx + (py - y1) * dy) / len_sq;
            let t = t.max(0.0).min(1.0);

            let closest_x = x1 + t * dx;
            let closest_y = y1 + t * dy;

            let dist_sq = (px - closest_x) * (px - closest_x) + (py - closest_y) * (py - closest_y);
            if dist_sq < margin * margin {
                return true;
            }
        }
    }

    false
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    // Store job tasks: each job has tasks with (machine, duration)
    vector<vector<pair<int, int>>> jobs(J);

    // Read input
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; t++) {
            cin >> jobs[j][t].first >> jobs[j][t].second; // machine, duration
        }
    }

    // Track when each machine becomes free
    vector<long long> machineFreeTimes(M, 0);

    // Store scheduled start times for each task
    vector<vector<long long>> schedule(J);

    // Greedy scheduling: for each job, schedule tasks in order
    for (int j = 0; j < J; j++) {
        schedule[j].resize(jobs[j].size());
        long long jobTime = 0; // When previous task in this job finishes

        for (int t = 0; t < jobs[j].size(); t++) {
            int machine = jobs[j][t].first;
            long long duration = jobs[j][t].second;

            // Task can start at the earliest time when:
            // 1. Previous task in this job has finished (jobTime)
            // 2. The machine is available (machineFreeTimes[machine])
            long long startTime = max(jobTime, machineFreeTimes[machine]);

            schedule[j][t] = startTime;

            // Update when this machine will be free
            machineFreeTimes[machine] = startTime + duration;

            // Update when next task in this job can start
            jobTime = startTime + duration;
        }
    }

    // Output scheduling solution
    for (int j = 0; j < J; j++) {
        for (int t = 0; t < jobs[j].size(); t++) {
            if (t > 0) cout << " ";
            cout << schedule[j][t] << " " << jobs[j][t].first;
        }
        cout << "\n";
    }

    return 0;

const int INF = 1e9;

struct Edge {
    int to, cap, flow;
    size_t rev;
};

class MaxFlow {
public:
    vector<vector<Edge>> graph;
    vector<int> level, iter;
    vector<bool> visited;
    int n;

    MaxFlow(int n) : n(n), graph(n), level(n), iter(n), visited(n) {}

    void add_edge(int from, int to, int cap) {
        graph[from].push_back({to, cap, 0, graph[to].size()});
        graph[to].push_back({from, 0, 0, graph[from].size() - 1});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& e : graph[v]) {
                if (level[e.to] < 0 && e.cap > e.flow) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }

        return level[t] >= 0;
    }

    int dfs(int v, int t, int pushed) {
        if (v == t || pushed == 0) return pushed;

        for (int& cid = iter[v]; cid < (int)graph[v].size(); cid++) {
            Edge& e = graph[v][cid];
            if (level[v] + 1 != level[e.to] || e.cap <= e.flow)
                continue;

            int tr = dfs(e.to, t, min(pushed, e.cap - e.flow));
            if (tr > 0) {
                e.flow += tr;
                graph[e.to][e.rev].flow -= tr;
                return tr;
            }
        }

        return 0;
    }

    int max_flow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            while (int pushed = dfs(s, t, INF)) {
                flow += pushed;
            }
        }
        return flow;
    }

    void find_reachable(int s) {
        fill(visited.begin(), visited.end(), false);
        queue<int> q;
        q.push(s);
        visited[s] = true;

        while (!q.empty()) {

                if (!visited[e.to] && e.cap > e.flow) {
                    visited[e.to] = true;
                    q.push(e.to);
                }
            }
        }
    }
};

#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

const int INF = 1e9;

struct Edge {
    int to, cap, flow;
    size_t rev;
};

class MaxFlow {
public:
    vector<vector<Edge>> graph;
    vector<int> level, iter;
    vector<bool> visited;
    int n;

    MaxFlow(int n) : n(n), graph(n), level(n), iter(n), visited(n) {}

    void add_edge(int from, int to, int cap) {
        graph[from].push_back({to, cap, 0, graph[to].size()});
        graph[to].push_back({from, 0, 0, graph[from].size() - 1});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& e : graph[v]) {
                if (level[e.to] < 0 && e.cap > e.flow) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }

        return level[t] >= 0;
    }

    int dfs(int v, int t, int pushed) {
        if (v == t || pushed == 0) return pushed;

        for (int& cid = iter[v]; cid < (int)graph[v].size(); cid++) {
            Edge& e = graph[v][cid];
            if (level[v] + 1 != level[e.to] || e.cap <= e.flow)
                continue;

            int tr = dfs(e.to, t, min(pushed, e.cap - e.flow));
            if (tr > 0) {
                e.flow += tr;
                graph[e.to][e.rev].flow -= tr;
                return tr;
            }
        }

        return 0;
    }

    int max_flow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            while (int pushed = dfs(s, t, INF)) {
                flow += pushed;
            }
        }
        return flow;
    }

    void find_reachable(int s) {
        fill(visited.begin(), visited.end(), false);
        queue<int> q;
        q.push(s);
        visited[s] = true;

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& e : graph[v]) {
                if (!visited[e.to] && e.cap > e.flow) {
                    visited[e.to] = true;
                    q.push(e.to);
                }
            }
        }
    }
};

using System;
using System.Collections.Generic;
using System.Linq;

class MiningOptimizer
{
    static void Main()
    {
        string header = Console.ReadLine();
        var parts = header.Split(' ');

        int xCount = int.Parse(parts[0]);
        int yCount = int.Parse(parts[1]);
        int zCount = int.Parse(parts[2]);

        // Read voxel data from stdin
        byte[,,] voxels = new byte[xCount, yCount, zCount];
        var input = Console.OpenStandardInput();

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    int b = input.ReadByte();
                    if (b >= 0) voxels[x, y, z] = (byte)b;
                }
            }
        }

        var operations = new List<string>();

        // Collect ore with profitability info
        var oreList = new List<(int x, int y, int z, int mass, int value, long profit)>();

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    byte v = voxels[x, y, z];
                    int mass = v % 16;
                    int valuePerTonne = v / 16;

                    if (valuePerTonne > 0 && mass > 0)
                    {
                        // Calculate profit: revenue - transport cost
                        // Revenue: mass * valuePerTonne * $1000 per unit
                        long revenue = (long)mass * valuePerTonne * 1000;

                        // Transport cost: manhattan distance * mass * $1 per voxel (flat)
                        int distance = x + y;
                        long transportCost = (long)distance * mass;

                        long profit = revenue - transportCost;
                        oreList.Add((x, y, z, mass, valuePerTonne, profit));
                    }
                }
            }
        }

        // Sort by profit descending and mine profitable ore
        var profitable = oreList.Where(o => o.profit > 0).OrderByDescending(o => o.profit).ToList();

        foreach (var (x, y, z, mass, value, _) in profitable)
        {
            // Generate Manhattan path from (x,y) to (0,0)
            var path = new List<(int, int)>();
            int cx = x, cy = y;

            while (cx != 0 || cy != 0)
            {
                if (cx > 0)
                    cx--;
                else if (cx < 0)
                    cx++;
                else if (cy > 0)
                    cy--;
                else if (cy < 0)
                    cy++;

                path.Add((cx, cy));
            }

            // Format operation: N digX digY digZ step1X step1Y ... stepNX stepNY destX destY
            string op = $"{path.Count} {x} {y} {z}";
            foreach (var (px, py) in path)
            {
                op += $" {px} {py}";
            }
            op += " 0 0";
            operations.Add(op);
        }

        // Output operations
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op);
        }
    }
}
