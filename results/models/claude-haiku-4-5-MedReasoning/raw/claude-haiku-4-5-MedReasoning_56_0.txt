const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { depot, truckCount, packages } = data;

function distance(p1, p2) {
  const dx = p1[0] - p2[0];
  const dy = p1[1] - p2[1];
  return Math.sqrt(dx * dx + dy * dy);
}

function routeDistance(route, packageMap, depotLoc) {
  let dist = distance(depotLoc, packageMap[route[0]]);
  for (let i = 1; i < route.length; i++) {
    dist += distance(packageMap[route[i-1]], packageMap[route[i]]);
  }
  dist += distance(packageMap[route[route.length-1]], depotLoc);
  return dist;
}

function totalDistance(routes, packageMap, depotLoc) {
  let total = 0;
  for (let route of routes) {
    total += routeDistance(route, packageMap, depotLoc);
  }
  return total;
}

// Create package map by ID
const packageMap = {};
for (let pkg of packages) {
  packageMap[pkg.id] = [pkg.x, pkg.y];
}

// Nearest Neighbor construction
function constructRoutes(packageIds, truckCount, packageMap, depot) {
  const routes = [];
  const used = new Set();

  for (let truck = 0; truck < truckCount && used.size < packageIds.length; truck++) {
    const route = [];
    let current = null;
    let currentPos = depot;

    // Start with nearest unvisited package
    let minDist = Infinity;
    for (let id of packageIds) {
      if (!used.has(id)) {
        const d = distance(currentPos, packageMap[id]);
        if (d < minDist) {
          minDist = d;
          current = id;
        }
      }
    }

    while (current !== null) {
      route.push(current);
      used.add(current);
      currentPos = packageMap[current];

      // Find nearest unvisited package
      let nextId = null;
      minDist = Infinity;
      for (let id of packageIds) {
        if (!used.has(id)) {
          const d = distance(currentPos, packageMap[id]);
          if (d < minDist) {
            minDist = d;
            nextId = id;
          }
        }
      }
      current = nextId;
    }

    if (route.length > 0) {
      routes.push(route);
    }
  }

  // If there are remaining packages, add them to the last route
  for (let id of packageIds) {
    if (!used.has(id)) {
      if (routes.length === 0) {
        routes.push([id]);
      } else {
        routes[routes.length - 1].push(id);
      }
      used.add(id);
    }
  }

  return routes;
}

// 2-opt improvement
function improve2Opt(routes, packageMap, depot) {
  let improved = true;
  let iterations = 0;
  const maxIterations = Math.min(1000, Math.max(100, routes.length * 50));

  while (improved && iterations < maxIterations) {
    improved = false;
    iterations++;

    for (let i = 0; i < routes.length; i++) {
      for (let j = i + 1; j < routes.length; j++) {
        // Try 2-opt between routes
        const route1 = routes[i];
        const route2 = routes[j];

        if (route1.length < 2 || route2.length < 2) continue;

        // Cost of reversing a segment in route1
        for (let a = 1; a < route1.length; a++) {
          const currentCost1 = distance(packageMap[route1[a-1]], packageMap[route1[a]]);

          for (let b = 1; b < route2.length; b++) {
            const currentCost2 = distance(packageMap[route2[b-1]], packageMap[route2[b]]);
            const newCost1 = distance(packageMap[route1[a-1]], packageMap[route2[b-1]]);
            const newCost2 = distance(packageMap[route1[a]], packageMap[route2[b]]);

            const delta = (newCost1 + newCost2) - (currentCost1 + currentCost2);

            if (delta < -1e-9) {
              const newRoute1 = route1.slice(0, a).concat(route2.slice(0, b).reverse());
              const newRoute2 = route1.slice(a).concat(route2.slice(b));
              routes[i] = newRoute1;
              routes[j] = newRoute2;
              improved = true;
              break;
            }
          }
          if (improved) break;
        }
        if (improved) break;
      }
      if (improved) break;
    }

    // Within-route 2-opt
    if (!improved) {
      for (let i = 0; i < routes.length; i++) {
        const route = routes[i];
        for (let a = 0; a < route.length - 2; a++) {
          for (let b = a + 2; b < route.length; b++) {
            const p1 = packageMap[route[a]];
            const p2 = packageMap[route[a + 1]];
            const p3 = packageMap[route[b]];
            const p4 = packageMap[route[(b + 1) % route.length]];

            const currentCost = distance(p1, p2) + distance(p3, p4);
            const newCost = distance(p1, p3) + distance(p2, p4);

            if (newCost < currentCost - 1e-9) {
              const newRoute = route.slice(0, a + 1).concat(route.slice(a + 1, b + 1).reverse()).concat(route.slice(b + 1));
              routes[i] = newRoute;
              improved = true;
              break;
            }
          }
          if (improved) break;
        }
        if (improved) break;
      }
    }
  }

  return routes;
}

// Main algorithm
const packageIds = packages.map(p => p.id);
let routes = constructRoutes(packageIds, truckCount, packageMap, depot);

// Remove empty routes
routes = routes.filter(r => r.length > 0);

// Improve with 2-opt
routes = improve2Opt(routes, packageMap, depot);

// Final output
const output = { routes };
console.log(JSON.stringify(output));
