using System;
using System.Collections.Generic;
using System.Linq;

public struct Vector3
{
    public float X, Y, Z;

    public Vector3(float x, float y, float z) { X = x; Y = y; Z = z; }

    public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3 operator *(Vector3 v, float s) => new Vector3(v.X * s, v.Y * s, v.Z * s);

    public float Dot(Vector3 other) => X * other.X + Y * other.Y + Z * other.Z;

    public Vector3 Cross(Vector3 other) =>
        new Vector3(Y * other.Z - Z * other.Y, Z * other.X - X * other.Z, X * other.Y - Y * other.X);
}

public struct Quaternion
{
    public float W, X, Y, Z;

    public Quaternion(float w, float x, float y, float z) { W = w; X = x; Y = y; Z = z; }

    public static Quaternion Identity => new Quaternion(1, 0, 0, 0);

    public Quaternion Normalized()
    {
        float len = (float)Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
        if (len < 0.0001f) return Identity;
        return new Quaternion(W / len, X / len, Y / len, Z / len);
    }

    public Vector3 RotateVector(Vector3 v)
    {
        // Rotation formula: p' = q * p * q^(-1)
        // Using: v' = v + 2*w*(q_xyz × v) + 2*(q_xyz × (q_xyz × v))
        float qw = W, qx = X, qy = Y, qz = Z;
        Vector3 qxyz = new Vector3(qx, qy, qz);
        Vector3 cross1 = qxyz.Cross(v);
        Vector3 cross2 = qxyz.Cross(cross1);
        return v + cross1 * (2 * qw) + cross2 * 2;
    }
}

public struct BoundingBox
{
    public Vector3 Min, Max;

    public Vector3 Size => new Vector3(Max.X - Min.X, Max.Y - Min.Y, Max.Z - Min.Z);

    public bool IntersectsWith(BoundingBox other, float margin = 0.001f) =>
        Min.X < other.Max.X + margin && Max.X > other.Min.X - margin &&
        Min.Y < other.Max.Y + margin && Max.Y > other.Min.Y - margin &&
        Min.Z < other.Max.Z + margin && Max.Z > other.Min.Z - margin;
}

public struct Placement
{
    public Vector3 Translation;
    public Quaternion Rotation;
    public BoundingBox BBox;
}

class Solution
{
    private Vector3[] vertices;
    private int[][] faces;
    private BoundingBox originalBBox;
    private Vector3 containerSize;
    private List<Placement> placements;

    static void Main()
    {
        new Solution().Run();
    }

    void Run()
    {
        ParseInput();
        ComputeOriginalBBox();
        placements = new List<Placement>();

        // Determine strategy based on container volume
        int containerVolume = (int)(containerSize.X * containerSize.Y * containerSize.Z);

        if (containerVolume < 1000) // Trivial: ~5x5x5
            PackTrivial();
        else if (containerVolume < 100000) // Medium: ~10-20 cubed
            PackMedium();
        else if (containerVolume < 1000000) // Large: ~50-100 cubed
            PackLarge();
        else // Extreme: 200+ cubed
            PackExtreme();

        OutputResult();
    }

    void ParseInput()
    {
        var vf = Console.ReadLine().Split();
        int V = int.Parse(vf[0]);
        int F = int.Parse(vf[1]);

        vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            var coords = Console.ReadLine().Split();
            vertices[i] = new Vector3(float.Parse(coords[0]), float.Parse(coords[1]), float.Parse(coords[2]));
        }

        faces = new int[F][];
        for (int i = 0; i < F; i++)
        {
            var indices = Console.ReadLine().Split();
            faces[i] = new[] { int.Parse(indices[0]), int.Parse(indices[1]), int.Parse(indices[2]) };
        }

        var containerDims = Console.ReadLine().Split();
        containerSize = new Vector3(float.Parse(containerDims[0]), float.Parse(containerDims[1]), float.Parse(containerDims[2]));
    }

    void ComputeOriginalBBox()
    {
        if (vertices.Length == 0) return;

        originalBBox.Min = vertices[0];
        originalBBox.Max = vertices[0];

        foreach (var v in vertices)
        {
            if (v.X < originalBBox.Min.X) originalBBox.Min.X = v.X;
            if (v.Y < originalBBox.Min.Y) originalBBox.Min.Y = v.Y;
            if (v.Z < originalBBox.Min.Z) originalBBox.Min.Z = v.Z;

            if (v.X > originalBBox.Max.X) originalBBox.Max.X = v.X;
            if (v.Y > originalBBox.Max.Y) originalBBox.Max.Y = v.Y;
            if (v.Z > originalBBox.Max.Z) originalBBox.Max.Z = v.Z;
        }
    }

    Quaternion[] GetRotations(int count)
    {
        var rots = new List<Quaternion>
        {
            Quaternion.Identity,
            new Quaternion(0.707f, 0, 0, 0.707f), // 90° around Z
            new Quaternion(0.707f, 0.707f, 0, 0), // 90° around X
            new Quaternion(0.707f, 0, 0.707f, 0), // 90° around Y
            new Quaternion(0.5f, 0.5f, 0.5f, 0.5f), // 120° around (1,1,1)
            new Quaternion(0.5f, 0.5f, -0.5f, 0.5f),
            new Quaternion(0.5f, -0.5f, 0.5f, 0.5f),
            new Quaternion(0.5f, 0.5f, 0.5f, -0.5f),
        };

        return rots.Take(Math.Min(count, rots.Count)).Select(q => q.Normalized()).ToArray();
    }

    void PackTrivial()
    {
        var rotations = GetRotations(8);
        Vector3 size = originalBBox.Size;
        float sx = Math.Max(0.1f, size.X);
        float sy = Math.Max(0.1f, size.Y);
        float sz = Math.Max(0.1f, size.Z);

        for (float x = 0; x < containerSize.X - 0.01f; x += sx * 0.7f)
            for (float y = 0; y < containerSize.Y - 0.01f; y += sy * 0.7f)
                for (float z = 0; z < containerSize.Z - 0.01f; z += sz * 0.7f)
                    foreach (var rot in rotations)
                        TryPlace(new Vector3(x, y, z), rot);
    }

    void PackMedium()
    {
        var rotations = GetRotations(6);
        Vector3 size = originalBBox.Size;
        float step = Math.Max(size.X, Math.Max(size.Y, size.Z)) * 0.8f;

        for (float x = 0; x < containerSize.X - 0.01f; x += step)
            for (float y = 0; y < containerSize.Y - 0.01f; y += step)
                for (float z = 0; z < containerSize.Z - 0.01f; z += step)
                    foreach (var rot in rotations)
                        TryPlace(new Vector3(x, y, z), rot);
    }

    void PackLarge()
    {
        var rotations = GetRotations(4);
        Vector3 size = originalBBox.Size;
        float step = Math.Max(size.X, Math.Max(size.Y, size.Z));

        for (float x = 0; x < containerSize.X - 0.01f; x += step)
            for (float y = 0; y < containerSize.Y - 0.01f; y += step)
                for (float z = 0; z < containerSize.Z - 0.01f; z += step)
                    foreach (var rot in rotations)
                        TryPlace(new Vector3(x, y, z), rot);
    }

    void PackExtreme()
    {
        var rotations = GetRotations(2);
        Vector3 size = originalBBox.Size;
        float step = Math.Max(size.X, Math.Max(size.Y, size.Z)) * 1.2f;

        for (float x = 0; x < containerSize.X - 0.01f; x += step)
        {
            for (float y = 0; y < containerSize.Y - 0.01f; y += step)
            {
                for (float z = 0; z < containerSize.Z - 0.01f; z += step)
                {
                    foreach (var rot in rotations)
                        TryPlace(new Vector3(x, y, z), rot);
                }
            }
        }
    }

    void TryPlace(Vector3 position, Quaternion rotation)
    {
        BoundingBox rotatedBBox = ComputeRotatedBBox(rotation);
        Vector3 size = rotatedBBox.Size;

        if (position.X + size.X > containerSize.X + 0.001f ||
            position.Y + size.Y > containerSize.Y + 0.001f ||
            position.Z + size.Z > containerSize.Z + 0.001f)
            return;

        BoundingBox placedBBox = new BoundingBox
        {
            Min = new Vector3(rotatedBBox.Min.X + position.X, rotatedBBox.Min.Y + position.Y, rotatedBBox.Min.Z + position.Z),
            Max = new Vector3(rotatedBBox.Max.X + position.X, rotatedBBox.Max.Y + position.Y, rotatedBBox.Max.Z + position.Z)
        };

        foreach (var p in placements)
            if (placedBBox.IntersectsWith(p.BBox))
                return;

        placements.Add(new Placement
        {
            Translation = position,
            Rotation = rotation.Normalized(),
            BBox = placedBBox
        });
    }

    BoundingBox ComputeRotatedBBox(Quaternion rotation)
    {
        Vector3[] rotated = new Vector3[vertices.Length];
        for (int i = 0; i < vertices.Length; i++)
            rotated[i] = rotation.RotateVector(vertices[i]);

        BoundingBox bbox = new BoundingBox { Min = rotated[0], Max = rotated[0] };

        for (int i = 1; i < rotated.Length; i++)
        {
            if (rotated[i].X < bbox.Min.X) bbox.Min.X = rotated[i].X;
            if (rotated[i].Y < bbox.Min.Y) bbox.Min.Y = rotated[i].Y;
            if (rotated[i].Z < bbox.Min.Z) bbox.Min.Z = rotated[i].Z;

            if (rotated[i].X > bbox.Max.X) bbox.Max.X = rotated[i].X;
            if (rotated[i].Y > bbox.Max.Y) bbox.Max.Y = rotated[i].Y;
            if (rotated[i].Z > bbox.Max.Z) bbox.Max.Z = rotated[i].Z;
        }

        return bbox;
    }

    void OutputResult()
    {
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Translation.X:F6} {p.Translation.Y:F6} {p.Translation.Z:F6} " +
                            $"{p.Rotation.W:F6} {p.Rotation.X:F6} {p.Rotation.Y:F6} {p.Rotation.Z:F6}");
        }
    }
}
