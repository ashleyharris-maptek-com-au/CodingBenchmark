using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    const double MU = 398600.4418;

    struct OrbitalState
    {
        public double X, Y, Z, VX, VY, VZ;
    }

    static void Main()
    {
        var firstLine = Console.ReadLine().Split();
        int numStations = int.Parse(firstLine[0]);
        double mu = double.Parse(firstLine[1]);

        var orbits = new OrbitalState[numStations + 1];

        var startLine = Console.ReadLine().Split();
        orbits[0] = ParseOrbitalState(startLine);

        for (int i = 1; i <= numStations; i++)
        {
            var line = Console.ReadLine().Split();
            orbits[i] = ParseOrbitalState(line);
        }

        var costs = ComputeCostMatrix(orbits, numStations + 1, mu);

        int[] fullRoute = SolveTSP(costs, numStations + 1);

        // Convert full route to station indices
        var stationRoute = new List<int>();
        for (int i = 0; i < fullRoute.Length; i++)
        {
            if (fullRoute[i] != 0)
            {
                stationRoute.Add(fullRoute[i] - 1);
            }
        }

        Console.WriteLine(string.Join(" ", stationRoute));
    }

    static OrbitalState ParseOrbitalState(string[] parts)
    {
        return new OrbitalState
        {
            X = double.Parse(parts[0]),
            Y = double.Parse(parts[1]),
            Z = double.Parse(parts[2]),
            VX = double.Parse(parts[3]),
            VY = double.Parse(parts[4]),
            VZ = double.Parse(parts[5])
        };
    }

    static double[,] ComputeCostMatrix(OrbitalState[] orbits, int numPositions, double mu)
    {
        var costs = new double[numPositions, numPositions];
        for (int i = 0; i < numPositions; i++)
        {
            for (int j = 0; j < numPositions; j++)
            {
                if (i != j)
                {
                    costs[i, j] = TransferCost(orbits[i], orbits[j], mu);
                }
            }
        }
        return costs;
    }

    static double TransferCost(OrbitalState from, OrbitalState to, double mu)
    {
        double dx = to.X - from.X;
        double dy = to.Y - from.Y;
        double dz = to.Z - from.Z;
        double posDist = Math.Sqrt(dx*dx + dy*dy + dz*dz);

        double dvx = to.VX - from.VX;
        double dvy = to.VY - from.VY;
        double dvz = to.VZ - from.VZ;
        double velDiff = Math.Sqrt(dvx*dvx + dvy*dvy + dvz*dvz);

        return velDiff + posDist / 10000.0;
    }

    static int[] SolveTSP(double[,] costs, int numPositions)
    {
        int[] route;

        if (numPositions <= 10)
        {
            route = SolveBruteForce(costs, numPositions);
        }
        else if (numPositions <= 12)
        {
            route = SolveNearestNeighbor(costs, numPositions);
        }
        else
        {
            route = BuildNearestNeighborRoute(costs, numPositions, 0);
            route = Improve2Opt(route, costs);
        }

        return route;
    }

    static int[] SolveBruteForce(double[,] costs, int n)
    {
        int[] bestRoute = Enumerable.Range(0, n).ToArray();
        double bestCost = RouteCost(bestRoute, costs);

        var indices = Enumerable.Range(0, n).ToList();

        while (NextPermutation(indices))
        {
            var route = indices.ToArray();
            double cost = RouteCost(route, costs);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestRoute = (int[])route.Clone();
            }
        }

        return bestRoute;
    }

    static int[] SolveNearestNeighbor(double[,] costs, int n)
    {
        int[] bestRoute = null;
        double bestCost = double.MaxValue;

        for (int start = 0; start < Math.Min(n, 10); start++)
        {
            int[] route = BuildNearestNeighborRoute(costs, n, start);
            route = Improve2Opt(route, costs);
            double cost = RouteCost(route, costs);

            if (cost < bestCost)
            {
                bestCost = cost;
                bestRoute = route;
            }
        }

        return bestRoute ?? Enumerable.Range(0, n).ToArray();
    }

    static int[] BuildNearestNeighborRoute(double[,] costs, int n, int start)
    {
        int[] route = new int[n];
        bool[] visited = new bool[n];

        route[0] = start;
        visited[start] = true;

        for (int i = 1; i < n; i++)
        {
            int current = route[i - 1];
            int nearest = -1;
            double minCost = double.MaxValue;

            for (int j = 0; j < n; j++)
            {
                if (!visited[j] && costs[current, j] < minCost)
                {
                    minCost = costs[current, j];
                    nearest = j;
                }
            }

            if (nearest == -1) nearest = FindUnvisited(visited, n);
            route[i] = nearest;
            visited[nearest] = true;
        }

        return route;
    }

    static int FindUnvisited(bool[] visited, int n)
    {
        for (int i = 0; i < n; i++)
            if (!visited[i]) return i;
        return 0;
    }

    static int[] Improve2Opt(int[] route, double[,] costs)
    {
        int n = route.Length;
        bool improved = true;
        int iterations = 0;
        int maxIterations = n < 50 ? n * n : n * 10;

        while (improved && iterations < maxIterations)
        {
            improved = false;
            iterations++;

            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    if (j == n - 1 && i == 0) continue;

                    double delta = Calculate2OptDelta(route, costs, i, j);
                    if (delta < -1e-9)
                    {
                        Reverse(route, i + 1, j);
                        improved = true;
                        break;
                    }
                }
                if (improved) break;
            }
        }

        return route;
    }

    static double Calculate2OptDelta(int[] route, double[,] costs, int i, int j)
    {
        int n = route.Length;
        int a = route[i];
        int b = route[(i + 1) % n];
        int c = route[j];
        int d = route[(j + 1) % n];

        double currentCost = costs[a, b] + costs[c, d];
        double newCost = costs[a, c] + costs[b, d];

        return newCost - currentCost;
    }

    static void Reverse(int[] route, int start, int end)
    {
        while (start < end)
        {
            int temp = route[start];
            route[start] = route[end];
            route[end] = temp;
            start++;
            end--;
        }
    }

    static double RouteCost(int[] route, double[,] costs)
    {
        double total = 0;
        for (int i = 0; i < route.Length - 1; i++)
        {
            total += costs[route[i], route[i + 1]];
        }
        total += costs[route[route.Length - 1], route[0]];
        return total;
    }

    static bool NextPermutation<T>(List<T> list) where T : IComparable<T>
    {
        int i = list.Count - 2;
        while (i >= 0 && list[i].CompareTo(list[i + 1]) >= 0) i--;
        if (i < 0) return false;

        int j = list.Count - 1;
        while (j > i && list[j].CompareTo(list[i]) <= 0) j--;

        var temp = list[i];
        list[i] = list[j];
        list[j] = temp;

        int left = i + 1;
        int right = list.Count - 1;
        while (left < right)
        {
            temp = list[left];
            list[left] = list[right];
            list[right] = temp;
            left++;
            right--;
        }

        return true;
    }
}
