#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (idx >= N) return;

    vec3 pos = inData[idx * 2].xyz;
    vec3 vel = inData[idx * 2 + 1].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    vec3 sepForce = vec3(0.0);
    vec3 avgAlignVel = vec3(0.0);
    vec3 avgCohPos = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    for (uint j = 0; j < N; j++) {
        if (j == idx) continue;

        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;

        vec3 diff = otherPos - pos;
        float dist = length(diff);

        if (dist == 0.0) continue;

        if (dist < sepRadius) {
            sepForce -= normalize(diff);
            sepCount++;
        }

        if (dist < alignRadius) {
            avgAlignVel += otherVel;
            alignCount++;
        }

        if (dist < cohRadius) {
            avgCohPos += otherPos;
            cohCount++;
        }
    }

    vec3 acceleration = vec3(0.0);

    if (sepCount > 0) {
        acceleration += (sepForce / float(sepCount)) * sepWeight;
    }

    if (alignCount > 0) {
        vec3 desiredVel = avgAlignVel / float(alignCount);
        acceleration += (desiredVel - vel) * alignWeight;
    }

    if (cohCount > 0) {
        vec3 desiredPos = avgCohPos / float(cohCount);
        acceleration += (desiredPos - pos) * cohWeight;
    }

    float threshold = boundSize * 0.4;

    if (abs(pos.x) > threshold) {
        acceleration.x += -sign(pos.x) * boundForce;
    }
    if (abs(pos.y) > threshold) {
        acceleration.y += -sign(pos.y) * boundForce;
    }
    if (abs(pos.z) > threshold) {
        acceleration.z += -sign(pos.z) * boundForce;
    }

    vel += acceleration * dt;

    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = (vel / speed) * maxSpeed;
    }

    pos += vel * dt;

    outData[idx * 2] = vec4(pos, 0.0);
    outData[idx * 2 + 1] = vec4(vel, 0.0);
}
