use std::f64::consts::PI;
use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Duration;

#[derive(Default, Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read and parse map header
    let map_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };

    let map_parts: Vec<&str> = map_line.split_whitespace().collect();
    if map_parts.len() < 5 {
        return;
    }

    let height: usize = match map_parts[1].parse() {
        Ok(h) => h,
        Err(_) => return,
    };
    let gravity: f64 = match map_parts[2].parse() {
        Ok(g) => g,
        Err(_) => return,
    };
    let max_thrust: f64 = match map_parts[3].parse() {
        Ok(t) => t,
        Err(_) => return,
    };

    // Read target location
    let start_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let start_parts: Vec<&str> = start_line.split_whitespace().collect();
    if start_parts.len() < 4 {
        return;
    }

    let target_x: f64 = match start_parts[2].parse() {
        Ok(x) => x,
        Err(_) => return,
    };
    let target_y: f64 = match start_parts[3].parse() {
        Ok(y) => y,
        Err(_) => return,
    };

    // Skip map rows
    for _ in 0..height {
        lines.next();
    }

    // Shared mutable state for lander
    let state = Arc::new(Mutex::new(State::default()));
    let state_reader = Arc::clone(&state);

    // Spawn dedicated reader thread
    thread::spawn(move || {
        for line in lines {
            if let Ok(line) = line {
                let parts: Vec<&str> = line.split_whitespace().collect();
                if parts.len() >= 7 {
                    if let (Ok(x), Ok(y), Ok(vx), Ok(vy), Ok(angle), Ok(angular_vel), Ok(fuel)) = (
                        parts[0].parse::<f64>(),
                        parts[1].parse::<f64>(),
                        parts[2].parse::<f64>(),
                        parts[3].parse::<f64>(),
                        parts[4].parse::<f64>(),
                        parts[5].parse::<f64>(),
                        parts[6].parse::<f64>(),
                    ) {
                        if let Ok(mut s) = state_reader.lock() {
                            s.x = x;
                            s.y = y;
                            s.vx = vx;
                            s.vy = vy;
                            s.angle = angle;
                            s.angular_vel = angular_vel;
                            s.fuel = fuel;
                        }
                    }
                }
            }
        }
    });

    // Main control loop at 20Hz
    let mut stdout = io::stdout();
    let tick = Duration::from_millis(50);

    loop {
        let current_state = {
            let s = state.lock().unwrap();
            *s
        };

        let (thrust, turn) = compute_control(&current_state, target_x, target_y, gravity, max_thrust);

        writeln!(stdout, "{} {}", thrust, turn).ok();
        stdout.flush().ok();

        thread::sleep(tick);
    }
}

fn compute_control(state: &State, target_x: f64, target_y: f64, gravity: f64, max_thrust: f64) -> (f64, f64) {
    // Calculate position errors
    let dx = target_x - state.x;
    let dy = target_y - state.y;

    // Determine desired angle based on current situation
    let desired_angle = if state.vy > 0.5 || dy > 15.0 {
        // Descending rapidly or far above target - face up to apply upward thrust
        0.0
    } else if state.vx.abs() > 0.5 {
        // Significant horizontal velocity - face opposite direction to decelerate
        if state.vx > 0.0 { PI } else { 0.0 }
    } else if dx.abs() > 3.0 {
        // Offset from target - angle to move towards it
        if dx > 0.0 { 0.15 } else { -0.15 }
    } else {
        // Close to target and low velocity - minimal angle
        0.0
    };

    // Calculate angle error with proper wrapping
    let angle_diff = desired_angle - state.angle;
    let angle_error = if angle_diff > PI {
        angle_diff - 2.0 * PI
    } else if angle_diff < -PI {
        angle_diff + 2.0 * PI
    } else {
        angle_diff
    };

    // PD control for rotation
    let turn = (angle_error * 3.0 - state.angular_vel * 0.4).clamp(-1.0, 1.0);

    // Thrust control
    let thrust = if state.vy > 1.0 {
        // Falling fast - maximum thrust to counteract gravity
        1.0
    } else if state.vy > 0.1 {
        // Moderate descent - ramp up thrust
        let hover_thrust = gravity / max_thrust;
        (hover_thrust + state.vy * 0.3).clamp(0.0, 1.0)
    } else {
        // Minimal descent or ascending - maintain hover thrust
        (gravity / max_thrust * 0.9).clamp(0.0, 1.0)
    };

    (thrust, turn)
}
