OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %color_in

OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %gl_out BuiltIn
OpMemberDecorate %gl_out 0 BuiltIn Position

OpDecorate %outColor Location 0
OpDecorate %color_in Location 0

; Type definitions
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%bool = OpTypeBool

; Integer constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%int_6 = OpConstant %int 6

; Float constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015
%float_neg_0_015 = OpConstant %float -0.015

; Color constants
%v3white = OpConstantComposite %v3float %float_1 %float_1 %float_1
%v3zero = OpConstantComposite %v3float %float_0 %float_0 %float_0

; gl_PerVertex input block
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %int_6
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_Input_arr_pv Input

; gl_PerVertex output block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_pv Output

; Input color array
%arr_v3 = OpTypeArray %v3float %int_6
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3
%color_in = OpVariable %ptr_Input_arr_v3 Input

; Output color variable
%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output

; Pointer types
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float

; Function type
%func_void = OpTypeFunction %void

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load main triangle positions (indices 0, 2, 4)
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

%p2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

%p4_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr

; Load adjacent positions (indices 1, 3, 5)
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%p3_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%p5_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_5 %int_0
%p5 = OpLoad %v4float %p5_ptr

; Load colors for main triangle vertices
%c0_ptr = OpAccessChain %ptr_Input_v3 %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr

%c2_ptr = OpAccessChain %ptr_Input_v3 %color_in %int_2
%c2 = OpLoad %v3float %c2_ptr

%c4_ptr = OpAccessChain %ptr_Input_v3 %color_in %int_4
%c4 = OpLoad %v3float %c4_ptr

; === Emit the original triangle (dimmed, 0.3x brightness) ===
; Vertex 0
%out_pos = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_pos %p0
%c0_dimmed = OpVectorTimesScalar %v3float %c0 %float_0_3
OpStore %outColor %c0_dimmed
OpEmitVertex

; Vertex 2
OpStore %out_pos %p2
%c2_dimmed = OpVectorTimesScalar %v3float %c2 %float_0_3
OpStore %outColor %c2_dimmed
OpEmitVertex

; Vertex 4
OpStore %out_pos %p4
%c4_dimmed = OpVectorTimesScalar %v3float %c4 %float_0_3
OpStore %outColor %c4_dimmed
OpEmitVertex

OpEndPrimitive

; === Emit thick lines for edges ===
; Edge 0→2 with adjacent vertex 1
; Check if edge is boundary: if p1 == p0 or p1 == p2
%p1_eq_p0 = OpFOrdEqual %v4float %p1 %p0
%p1_eq_p2 = OpFOrdEqual %v4float %p1 %p2
%any1_0 = OpAny %bool %p1_eq_p0
%any1_2 = OpAny %bool %p1_eq_p2
%is_boundary_edge_1 = OpLogicalOr %bool %any1_0 %any1_2

; For now, emit edge as white line quad (simplified approach)
OpStore %outColor %v3white
OpStore %out_pos %p0
OpEmitVertex
%p0_offset = OpConstantComposite %v4float %float_0_015 %float_0 %float_0 %float_0
%p0_off = OpFAdd %v4float %p0 %p0_offset
OpStore %out_pos %p0_off
OpEmitVertex
OpStore %out_pos %p2
OpEmitVertex
%p2_off = OpFAdd %v4float %p2 %p0_offset
OpStore %out_pos %p2_off
OpEmitVertex
OpEndPrimitive

; Edge 2→4 with adjacent vertex 3
%p3_eq_p2 = OpFOrdEqual %v4float %p3 %p2
%p3_eq_p4 = OpFOrdEqual %v4float %p3 %p4
%any3_2 = OpAny %bool %p3_eq_p2
%any3_4 = OpAny %bool %p3_eq_p4
%is_boundary_edge_3 = OpLogicalOr %bool %any3_2 %any3_4

OpStore %outColor %v3white
OpStore %out_pos %p2
OpEmitVertex
%p2_off_2 = OpFAdd %v4float %p2 %p0_offset
OpStore %out_pos %p2_off_2
OpEmitVertex
OpStore %out_pos %p4
OpEmitVertex
%p4_off = OpFAdd %v4float %p4 %p0_offset
OpStore %out_pos %p4_off
OpEmitVertex
OpEndPrimitive

; Edge 4→0 with adjacent vertex 5
%p5_eq_p4 = OpFOrdEqual %v4float %p5 %p4
%p5_eq_p0 = OpFOrdEqual %v4float %p5 %p0
%any5_4 = OpAny %bool %p5_eq_p4
%any5_0 = OpAny %bool %p5_eq_p0
%is_boundary_edge_5 = OpLogicalOr %bool %any5_4 %any5_0

OpStore %outColor %v3white
OpStore %out_pos %p4
OpEmitVertex
%p4_off_2 = OpFAdd %v4float %p4 %p0_offset
OpStore %out_pos %p4_off_2
OpEmitVertex
OpStore %out_pos %p0
OpEmitVertex
%p0_off_2 = OpFAdd %v4float %p0 %p0_offset
OpStore %out_pos %p0_off_2
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
