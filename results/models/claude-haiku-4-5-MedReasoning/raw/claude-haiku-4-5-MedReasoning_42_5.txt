OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%gl_PerVertex_in = OpTypeStruct %v4float
%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%color_in = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%func_type = OpTypeFunction %void
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%glsl_set = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0_v4 = OpLoad %v4float %p0_ptr
%p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1_v4 = OpLoad %v4float %p1_ptr
%p1 = OpVectorShuffle %v3float %p1_v4 %p1_v4 0 1 2

; Load colors
%c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr

%c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr

; Compute direction: dir = p1 - p0
%dir = OpFSub %v3float %p1 %p0

; Extract direction components for perpendicular computation
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

; Compute perpendicular: perp = normalize(vec3(-dir.y, dir.x, 0))
%neg_dir_y = OpFNegate %float %dir_y
%perp_unnorm = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0

; Normalize perpendicular
%perp = OpExtInst %v3float %glsl_set 69 %perp_unnorm

; Compute mid = p0 + 0.75 * (p1 - p0)
%scale_vec = OpVectorTimesScalar %v3float %dir %float_0_75
%mid = OpFAdd %v3float %p0 %scale_vec

; Compute perpendicular offsets
%perp_015 = OpVectorTimesScalar %v3float %perp %float_0_015
%perp_05 = OpVectorTimesScalar %v3float %perp %float_0_05

; Negate perpendicular offsets for opposite side
%neg_perp_015 = OpFNegate %v3float %perp_015
%neg_perp_05 = OpFNegate %v3float %perp_05

; Emit shaft - 4 vertices as triangle strip
; Vertex 1: p0 + perp * 0.015
%v1 = OpFAdd %v3float %p0 %perp_015
%v1_x = OpCompositeExtract %float %v1 0
%v1_y = OpCompositeExtract %float %v1 1
%v1_z = OpCompositeExtract %float %v1 2
%v1_v4 = OpCompositeConstruct %v4float %v1_x %v1_y %v1_z %float_1
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %v1_v4
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: p0 - perp * 0.015
%v2 = OpFAdd %v3float %p0 %neg_perp_015
%v2_x = OpCompositeExtract %float %v2 0
%v2_y = OpCompositeExtract %float %v2 1
%v2_z = OpCompositeExtract %float %v2 2
%v2_v4 = OpCompositeConstruct %v4float %v2_x %v2_y %v2_z %float_1
OpStore %out_pos_ptr %v2_v4
OpStore %outColor %c0
OpEmitVertex

; Vertex 3: mid + perp * 0.015
%v3 = OpFAdd %v3float %mid %perp_015
%v3_x = OpCompositeExtract %float %v3 0
%v3_y = OpCompositeExtract %float %v3 1
%v3_z = OpCompositeExtract %float %v3 2
%v3_v4 = OpCompositeConstruct %v4float %v3_x %v3_y %v3_z %float_1
OpStore %out_pos_ptr %v3_v4
OpStore %outColor %c0
OpEmitVertex

; Vertex 4: mid - perp * 0.015
%v4 = OpFAdd %v3float %mid %neg_perp_015
%v4_x = OpCompositeExtract %float %v4 0
%v4_y = OpCompositeExtract %float %v4 1
%v4_z = OpCompositeExtract %float %v4 2
%v4_v4 = OpCompositeConstruct %v4float %v4_x %v4_y %v4_z %float_1
OpStore %out_pos_ptr %v4_v4
OpStore %outColor %c0
OpEmitVertex

; End primitive
OpEndPrimitive

; Emit arrowhead - 3 vertices as triangle strip
; Vertex 5: mid + perp * 0.05
%v5 = OpFAdd %v3float %mid %perp_05
%v5_x = OpCompositeExtract %float %v5 0
%v5_y = OpCompositeExtract %float %v5 1
%v5_z = OpCompositeExtract %float %v5 2
%v5_v4 = OpCompositeConstruct %v4float %v5_x %v5_y %v5_z %float_1
OpStore %out_pos_ptr %v5_v4
OpStore %outColor %c1
OpEmitVertex

; Vertex 6: mid - perp * 0.05
%v6 = OpFAdd %v3float %mid %neg_perp_05
%v6_x = OpCompositeExtract %float %v6 0
%v6_y = OpCompositeExtract %float %v6 1
%v6_z = OpCompositeExtract %float %v6 2
%v6_v4 = OpCompositeConstruct %v4float %v6_x %v6_y %v6_z %float_1
OpStore %out_pos_ptr %v6_v4
OpStore %outColor %c1
OpEmitVertex

; Vertex 7: p1 (tip)
%v7_x = OpCompositeExtract %float %p1 0
%v7_y = OpCompositeExtract %float %p1 1
%v7_z = OpCompositeExtract %float %p1 2
%v7_v4 = OpCompositeConstruct %v4float %v7_x %v7_y %v7_z %float_1
OpStore %out_pos_ptr %v7_v4
OpStore %outColor %c1
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
