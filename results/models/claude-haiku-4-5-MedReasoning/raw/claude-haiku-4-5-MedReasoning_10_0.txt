#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>

struct Rect {
    int w, h, id;
};

struct Placement {
    int board, x, y, rotated;
};

struct FreeRect {
    int x, y, w, h;
};

struct Board {
    int width, height;
    std::vector<FreeRect> freeRects;

    Board(int w, int h) : width(w), height(h) {
        freeRects.push_back({0, 0, w, h});
    }

    bool place(int w, int h, int& px, int& py) {
        // Find first free rectangle that fits
        for (int i = 0; i < (int)freeRects.size(); i++) {
            FreeRect& fr = freeRects[i];
            if (fr.w >= w && fr.h >= h) {
                px = fr.x;
                py = fr.y;

                // Create new free rectangles from the split
                FreeRect right{fr.x + w, fr.y, fr.w - w, h};
                FreeRect top{fr.x, fr.y + h, fr.w, fr.h - h};

                // Remove the used rectangle
                freeRects.erase(freeRects.begin() + i);

                // Add new free rectangles if they have area
                if (right.w > 0 && right.h > 0) {
                    freeRects.push_back(right);
                }
                if (top.w > 0 && top.h > 0) {
                    freeRects.push_back(top);
                }

                return true;
            }
        }
        return false;
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, bw, bh;
    std::cin >> N >> bw >> bh;

    std::vector<Rect> rects(N);
    for (int i = 0; i < N; i++) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }

    // Sort by area descending for better packing
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return (long long)a.w * a.h > (long long)b.w * b.h;
    });

    std::vector<Board> boards;
    std::vector<Placement> result(N);

    for (const auto& rect : rects) {
        bool placed = false;

        // Try to place in existing boards
        for (int b = 0; b < (int)boards.size() && !placed; b++) {
            int x, y;

            // Try without rotation
            if (boards[b].place(rect.w, rect.h, x, y)) {
                result[rect.id] = {b, x, y, 0};
                placed = true;
            }
            // Try with rotation
            else if (boards[b].place(rect.h, rect.w, x, y)) {
                result[rect.id] = {b, x, y, 1};
                placed = true;
            }
        }

        // Create new board if needed
        if (!placed) {
            boards.push_back(Board(bw, bh));
            int x, y;

            if (boards.back().place(rect.w, rect.h, x, y)) {
                result[rect.id] = {(int)boards.size() - 1, x, y, 0};
            } else if (boards.back().place(rect.h, rect.w, x, y)) {
                result[rect.id] = {(int)boards.size() - 1, x, y, 1};
            }
        }
    }

    // Output
    std::cout << boards.size() << "\n";
    for (int i = 0; i < N; i++) {
        std::cout << result[i].board << " " << result[i].x << " "
                  << result[i].y << " " << result[i].rotated << "\n";
    }

    return 0;
}
