using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        string[] line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);

        int[] cuts = Console.ReadLine().Split(' ');
        int[] cutValues = new int[cuts.Length];
        for (int i = 0; i < cuts.Length; i++)
        {
            cutValues[i] = int.Parse(cuts[i]);
        }

        var stocks = new List<List<int>>();
        var remaining = new List<int>();

        for (int i = 0; i < cutValues.Length; i++)
        {
            int cutLength = cutValues[i];

            int binIndex = -1;
            for (int j = 0; j < stocks.Count; j++)
            {
                if (remaining[j] >= cutLength)
                {
                    binIndex = j;
                    break;
                }
            }

            if (binIndex == -1)
            {
                stocks.Add(new List<int>());
                remaining.Add(stockLength);
                binIndex = stocks.Count - 1;
            }

            stocks[binIndex].Add(i);
            remaining[binIndex] -= cutLength;
        }

        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            for (int i = 0; i < stock.Count; i++)
            {
                if (i > 0) Console.Write(" ");
                Console.Write(stock[i]);
            }
            Console.WriteLine();
        }
    }
}
