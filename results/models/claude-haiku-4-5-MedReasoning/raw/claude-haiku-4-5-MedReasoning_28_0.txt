use std::io::{self, Read};
use std::collections::HashSet;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();

    let mut lines = input.lines();
    let first_line = match lines.next() {
        Some(line) => line,
        None => {
            println!("0");
            return;
        }
    };

    let parts: Vec<&str> = first_line.split_whitespace().collect();
    if parts.len() < 2 {
        println!("0");
        return;
    }

    let n: usize = parts[0].parse().unwrap_or(0);
    let _m: usize = parts[1].parse().unwrap_or(0);

    let mut adj: Vec<HashSet<usize>> = vec![HashSet::new(); n];

    for line in lines {
        if line.is_empty() {
            continue;
        }
        let parts: Vec<&str> = line.split_whitespace().collect();
        if parts.len() >= 2 {
            if let (Ok(u), Ok(v)) = (parts[0].parse::<usize>(), parts[1].parse::<usize>()) {
                if u < n && v < n && u != v {
                    adj[u].insert(v);
                    adj[v].insert(u);
                }
            }
        }
    }

    let has_edges = adj.iter().any(|neighbors| !neighbors.is_empty());
    if !has_edges {
        println!("0");
        return;
    }

    let mut cover = solve_vertex_cover(&adj);

    println!("{}", cover.len());
    if !cover.is_empty() {
        cover.sort_unstable();
        println!("{}", cover.iter().map(|v| v.to_string()).collect::<Vec<_>>().join(" "));
    }
}

fn solve_vertex_cover(original_adj: &[HashSet<usize>]) -> Vec<usize> {
    let mut cover = Vec::new();
    let mut adj: Vec<HashSet<usize>> = original_adj.to_vec();
    let n = adj.len();

    loop {
        let mut best_vertex = 0;
        let mut best_degree = 0;

        for v in 0..n {
            if adj[v].len() > best_degree {
                best_degree = adj[v].len();
                best_vertex = v;
            }
        }

        if best_degree == 0 {
            break;
        }

        cover.push(best_vertex);

        let neighbors: Vec<usize> = adj[best_vertex].iter().copied().collect();
        for u in neighbors {
            adj[u].remove(&best_vertex);
        }
        adj[best_vertex].clear();
    }

    cover
}
