const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { machineCount, tasks } = data;

// Build dependency graph
const taskMap = new Map(tasks.map(t => [t.id, t]));
const dependents = new Map(); // task -> list of tasks depending on it
const inDegree = new Map(); // task -> count of dependencies

for (const task of tasks) {
  inDegree.set(task.id, (task.deps || []).length);
  if (!dependents.has(task.id)) {
    dependents.set(task.id, []);
  }
}

for (const task of tasks) {
  for (const dep of task.deps || []) {
    if (!dependents.has(dep)) {
      dependents.set(dep, []);
    }
    dependents.get(dep).push(task.id);
  }
}

// Compute earliest start time and critical path info
const earliestStart = new Map();
const earliestEnd = new Map();

const computeEarliestTimes = () => {
  const queue = [];
  const tempInDegree = new Map(inDegree);
  for (const task of tasks) {
    if (tempInDegree.get(task.id) === 0) {
      queue.push(task.id);
      earliestStart.set(task.id, 0);
    }
  }

  while (queue.length > 0) {
    const taskId = queue.shift();
    const task = taskMap.get(taskId);
    const start = earliestStart.get(taskId);
    const end = start + task.duration;
    earliestEnd.set(taskId, end);

    for (const dependent of dependents.get(taskId)) {
      const depTask = taskMap.get(dependent);
      const currentEarliest = earliestStart.get(dependent) || 0;
      earliestStart.set(dependent, Math.max(currentEarliest, end));

      tempInDegree.set(dependent, tempInDegree.get(dependent) - 1);
      if (tempInDegree.get(dependent) === 0) {
        queue.push(dependent);
      }
    }
  }
};

computeEarliestTimes();

// Greedy scheduling with priority based on deadline, weight, and criticality
const machines = Array.from({ length: machineCount }, () => ({ endTime: 0, tasks: [] }));
const schedule = [];
const scheduled = new Set();

const getTaskPriority = (taskId) => {
  const task = taskMap.get(taskId);
  const slack = task.deadline - earliestEnd.get(taskId);
  return {
    deadline: task.deadline,
    weight: -task.weight,
    slack: slack,
    earliestEnd: earliestEnd.get(taskId)
  };
};

const compareTask = (a, b) => {
  const aPriority = getTaskPriority(a);
  const bPriority = getTaskPriority(b);

  // First: tighter deadlines
  if (aPriority.deadline !== bPriority.deadline) {
    return aPriority.deadline - bPriority.deadline;
  }
  // Second: higher weight
  if (aPriority.weight !== bPriority.weight) {
    return aPriority.weight - bPriority.weight;
  }
  // Third: less slack (more critical)
  if (aPriority.slack !== bPriority.slack) {
    return aPriority.slack - bPriority.slack;
  }
  // Finally: earlier end time
  return aPriority.earliestEnd - bPriority.earliestEnd;
};

while (scheduled.size < tasks.length) {
  // Find ready tasks (all dependencies scheduled)
  const ready = [];
  for (const task of tasks) {
    if (!scheduled.has(task.id)) {
      const depsReady = (task.deps || []).every(dep => scheduled.has(dep));
      if (depsReady) {
        ready.push(task.id);
      }
    }
  }

  if (ready.length === 0) break;

  // Sort ready tasks by priority
  ready.sort(compareTask);

  // Schedule the first ready task on the earliest available machine
  const taskId = ready[0];
  const task = taskMap.get(taskId);

  // Find the machine that finishes earliest but respects earliest start time
  let bestMachine = 0;
  let bestTime = Math.max(machines[0].endTime, earliestStart.get(taskId));

  for (let i = 1; i < machineCount; i++) {
    const time = Math.max(machines[i].endTime, earliestStart.get(taskId));
    if (time < bestTime) {
      bestTime = time;
      bestMachine = i;
    }
  }

  const startTime = bestTime;
  const endTime = startTime + task.duration;

  schedule.push({
    id: taskId,
    start: startTime,
    machine: bestMachine
  });

  machines[bestMachine].endTime = endTime;
  machines[bestMachine].tasks.push(taskId);
  scheduled.add(taskId);
}

// Output the schedule
console.log(JSON.stringify({ schedule }, null, 0));
