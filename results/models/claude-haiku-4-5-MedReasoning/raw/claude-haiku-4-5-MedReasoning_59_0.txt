const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const probability = data.probability;
const drones = data.drones;
const maxStepsPerDrone = data.maxStepsPerDrone;

const rows = probability.length;
const cols = probability[0].length;
const covered = new Array(rows).fill(null).map(() => new Array(cols).fill(false));

function isValid(x, y) {
  return x >= 0 && x < rows && y >= 0 && y < cols && probability[x][y] !== -1;
}

function manhattanDistance(x1, y1, x2, y2) {
  return Math.abs(x1 - x2) + Math.abs(y1 - y2);
}

function findBestPath(startX, startY, maxSteps) {
  const path = [[startX, startY]];
  let currentX = startX;
  let currentY = startY;
  let stepsLeft = maxSteps;

  while (stepsLeft > 0) {
    // Find the best uncovered cell reachable with remaining steps
    let bestCell = null;
    let bestScore = -1;
    let bestDist = 0;

    for (let i = 0; i < rows; i++) {
      for (let j = 0; j < cols; j++) {
        if (!covered[i][j] && probability[i][j] > 0) {
          const dist = manhattanDistance(currentX, currentY, i, j);
          if (dist <= stepsLeft && dist > 0) {
            // Score = value per step efficiency
            const score = probability[i][j] / dist;
            if (score > bestScore || (score === bestScore && dist < bestDist)) {
              bestScore = score;
              bestCell = [i, j];
              bestDist = dist;
            }
          }
        }
      }
    }

    if (!bestCell) {
      // No uncovered cells found, try staying or moving randomly
      break;
    }

    // Move towards best cell using greedy pathfinding
    const [targetX, targetY] = bestCell;
    while ((currentX !== targetX || currentY !== targetY) && stepsLeft > 0) {
      let moved = false;

      // Try to move closer to target
      if (currentX < targetX && isValid(currentX + 1, currentY)) {
        currentX++;
        moved = true;
      } else if (currentX > targetX && isValid(currentX - 1, currentY)) {
        currentX--;
        moved = true;
      } else if (currentY < targetY && isValid(currentX, currentY + 1)) {
        currentY++;
        moved = true;
      } else if (currentY > targetY && isValid(currentX, currentY - 1)) {
        currentY--;
        moved = true;
      }

      if (moved) {
        stepsLeft--;
        path.push([currentX, currentY]);
      } else {
        break;
      }
    }
  }

  return path;
}

function markCovered(path) {
  for (const [x, y] of path) {
    covered[x][y] = true;
  }
}

const paths = [];
for (const drone of drones) {
  const path = findBestPath(drone.x, drone.y, maxStepsPerDrone);
  paths.push(path);
  markCovered(path);
}
