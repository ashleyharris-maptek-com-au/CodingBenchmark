@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

fn rotl(x : u32, shift : u32) -> u32 {
    return (x << shift) | (x >> (32u - shift));
}

fn count_trailing_zeros(h0 : u32, h1 : u32) -> u32 {
    if h0 != 0u {
        return countTrailingZeros(h0);
    } else {
        return 32u + countTrailingZeros(h1);
    }
}

fn custom_hash(data : array<u32, 16>, nonce : u32) -> vec2<u32> {
    var state : array<u32, 16>;

    // Initialize state with data XOR (nonce + i)
    for (var i : u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = data[i] ^ (nonce + i);
    }

    // 8 rounds of quarter-round mixing
    for (var round : u32 = 0u; round < 8u; round = round + 1u) {
        // Quarter-round on state[0..3]
        var a : u32 = state[0];
        var b : u32 = state[1];
        var c : u32 = state[2];
        var d : u32 = state[3];

        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);

        state[0] = a;
        state[1] = b;
        state[2] = c;
        state[3] = d;

        // Quarter-round on state[4..7]
        a = state[4];
        b = state[5];
        c = state[6];
        d = state[7];

        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);

        state[4] = a;
        state[5] = b;
        state[6] = c;
        state[7] = d;

        // Quarter-round on state[8..11]
        a = state[8];
        b = state[9];
        c = state[10];
        d = state[11];

        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);

        state[8] = a;
        state[9] = b;
        state[10] = c;
        state[11] = d;

        // Quarter-round on state[12..15]
        a = state[12];
        b = state[13];
        c = state[14];
        d = state[15];

        a = a + b; d = d ^ a; d = rotl(d, 16u);
        c = c + d; b = b ^ c; b = rotl(b, 12u);
        a = a + b; d = d ^ a; d = rotl(d, 8u);
        c = c + d; b = b ^ c; b = rotl(b, 7u);

        state[12] = a;
        state[13] = b;
        state[14] = c;
        state[15] = d;
    }

    // Fold 16 words to 2
    var h0 : u32 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    var h1 : u32 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    return vec2<u32>(h0, h1);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    // Check if this thread is in range
    if gid.x >= params.z {
        return;
    }

    // Calculate the nonce for this thread
    let my_nonce : u32 = params.y + gid.x;

    // Compute hash
    let hash : vec2<u32> = custom_hash(baseData, my_nonce);
    let h0 : u32 = hash.x;
    let h1 : u32 = hash.y;

    // Count trailing zeros
    let trailing_zeros : u32 = count_trailing_zeros(h0, h1);

    // Check if we found a valid nonce
    if trailing_zeros >= params.x {
        result[0] = 1u;
        result[1] = my_nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
