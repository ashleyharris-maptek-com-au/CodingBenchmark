using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    private int height;
    private int width;
    private char[,] board;
    private bool[,] isSafe;
    private bool[,] isMine;
    private bool[,] isKnown;

    static void Main()
    {
        var solver = new MinesweeperSolver();
        solver.Run();
    }

    void Run()
    {
        ReadInput();
        SolveMinesweeper();
        OutputResults();
    }

    void ReadInput()
    {
        var line = Console.ReadLine().Split();
        height = int.Parse(line[0]);
        width = int.Parse(line[1]);

        board = new char[height, width];
        isSafe = new bool[height, width];
        isMine = new bool[height, width];
        isKnown = new bool[height, width];

        for (int i = 0; i < height; i++)
        {
            string row = Console.ReadLine();
            for (int j = 0; j < width; j++)
            {
                board[i, j] = j < row.Length ? row[j] : ' ';
                if (board[i, j] == '*')
                {
                    isMine[i, j] = true;
                    isKnown[i, j] = true;
                }
                else if (board[i, j] != ' ')
                {
                    isKnown[i, j] = true;
                }
            }
        }
    }

    void SolveMinesweeper()
    {
        bool changed = true;
        while (changed)
        {
            changed = false;

            for (int i = 0; i < height; i++)
            {
                for (int j = 0; j < width; j++)
                {
                    if (board[i, j] >= '0' && board[i, j] <= '8')
                    {
                        if (ProcessNumberCell(i, j))
                            changed = true;
                    }
                }
            }
        }
    }

    bool ProcessNumberCell(int row, int col)
    {
        int mineCount = board[row, col] - '0';
        var neighbors = GetNeighbors(row, col);

        int flaggedCount = 0;
        int unknownCount = 0;
        var unknownCells = new List<(int, int)>();

        foreach (var (r, c) in neighbors)
        {
            if (isMine[r, c])
            {
                flaggedCount++;
            }
            else if (!isKnown[r, c])
            {
                unknownCount++;
                unknownCells.Add((r, c));
            }
        }

        bool changed = false;
        int minesNeeded = mineCount - flaggedCount;

        if (minesNeeded == 0)
        {
            foreach (var (r, c) in unknownCells)
            {
                if (!isSafe[r, c])
                {
                    isSafe[r, c] = true;
                    isKnown[r, c] = true;
                    changed = true;
                }
            }
        }
        else if (minesNeeded == unknownCount && unknownCount > 0)
        {
            foreach (var (r, c) in unknownCells)
            {
                if (!isMine[r, c])
                {
                    isMine[r, c] = true;
                    isKnown[r, c] = true;
                    changed = true;
                }
            }
        }

        return changed;
    }

    List<(int, int)> GetNeighbors(int row, int col)
    {
        var neighbors = new List<(int, int)>();
        for (int dr = -1; dr <= 1; dr++)
        {
            for (int dc = -1; dc <= 1; dc++)
            {
                if (dr == 0 && dc == 0) continue;
                int nr = row + dr;
                int nc = col + dc;
                if (nr >= 0 && nr < height && nc >= 0 && nc < width)
                {
                    neighbors.Add((nr, nc));
                }
            }
        }
        return neighbors;
    }

    void OutputResults()
    {
        var safeCells = new List<(int, int)>();

        for (int i = 0; i < height; i++)
        {
            for (int j = 0; j < width; j++)
            {
                if (board[i, j] == ' ' && isSafe[i, j])
                {
                    safeCells.Add((j, i));
                }
            }
        }

        safeCells.Sort();
        Console.WriteLine(safeCells.Count);
        foreach (var (x, y) in safeCells)
        {
            Console.WriteLine($"{x} {y}");
        }
    }
}
