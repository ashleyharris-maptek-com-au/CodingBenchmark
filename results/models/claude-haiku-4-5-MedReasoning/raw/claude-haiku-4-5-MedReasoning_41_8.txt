OpCapability Shader
OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations for UBO structure
OpDecorate %ubo Block
OpMemberDecorate %ubo 0 ColMajor
OpMemberDecorate %ubo 0 MatrixStride 16
OpMemberDecorate %ubo 1 ColMajor
OpMemberDecorate %ubo 1 MatrixStride 16
OpMemberDecorate %ubo 2 ColMajor
OpMemberDecorate %ubo 2 MatrixStride 16
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

; Decorations for input/output variables
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

; Type definitions
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; UBO structure
%ubo = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Pointer types
%ubo_ptr = OpTypePointer Uniform %ubo
%vec3_input = OpTypePointer Input %vec3
%vec2_input = OpTypePointer Input %vec2
%vec4_output = OpTypePointer Output %vec4
%vec4_uniform = OpTypePointer Uniform %vec4

; Function type
%func_void = OpTypeFunction %void

; Constants
%c_float_0 = OpConstant %float 0.0
%c_float_1 = OpConstant %float 1.0
%c_float_0_2 = OpConstant %float 0.2
%c_float_0_4 = OpConstant %float 0.4
%c_float_0_7 = OpConstant %float 0.7
%c_float_0_25 = OpConstant %float 0.25
%c_float_0_5 = OpConstant %float 0.5
%c_float_0_75 = OpConstant %float 0.75
%c_int_3 = OpConstant %int 3

; Global variables
%in_worldPos = OpVariable %vec3_input Input
%in_normal = OpVariable %vec3_input Input
%in_uv = OpVariable %vec2_input Input
%in_tangent = OpVariable %vec3_input Input
%in_color = OpVariable %vec3_input Input
%out_color = OpVariable %vec4_output Output
%uboVar = OpVariable %ubo_ptr Uniform

%glsl = OpExtInstImport "GLSL.std.450"

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load input values
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%color = OpLoad %vec3 %in_color

; Normalize the normal vector
%normalNorm = OpExtInst %vec3 %glsl 65 %normal

; Get light position from UBO (member 3, offset 192)
%lightPtr = OpAccessChain %vec4_uniform %uboVar %c_int_3
%lightPos = OpLoad %vec4 %lightPtr

; Extract xyz components from light position
%lightXyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

; Calculate light direction (L = normalize(lightPos - worldPos))
%lightDir = OpFSub %vec3 %lightXyz %worldPos
%lightDirNorm = OpExtInst %vec3 %glsl 65 %lightDir

; Calculate Lambertian diffuse term: d = dot(N, L)
%diffuse = OpDot %float %normalNorm %lightDirNorm

; Clamp to [0, 1]: d = max(d, 0)
%diffuseMax = OpExtInst %float %glsl 40 %diffuse %c_float_0

; Quantize into bands with thresholds 0.75, 0.5, 0.25 and values 1.0, 0.7, 0.4, 0.2
%cmp075 = OpFOrdGreaterThanEqual %bool %diffuseMax %c_float_0_75
%cmp05 = OpFOrdGreaterThanEqual %bool %diffuseMax %c_float_0_5
%cmp025 = OpFOrdGreaterThanEqual %bool %diffuseMax %c_float_0_25

; Nested select to assign band value
%band_level1 = OpSelect %float %cmp025 %c_float_0_4 %c_float_0_2
%band_level2 = OpSelect %float %cmp05 %c_float_0_7 %band_level1
%band = OpSelect %float %cmp075 %c_float_1 %band_level2

; Multiply vertex color by the band value
%colorFinal = OpVectorTimesScalar %vec3 %color %band

; Create output color as vec4 with alpha = 1.0
%outColorVal = OpCompositeConstruct %vec4 %colorFinal %c_float_1

; Store the result
OpStore %out_color %outColorVal

OpReturn
OpFunctionEnd
