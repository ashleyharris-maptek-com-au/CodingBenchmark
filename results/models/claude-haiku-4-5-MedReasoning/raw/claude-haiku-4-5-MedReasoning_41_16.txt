; Bound: 150

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4float = OpTypeMatrix %v4float 4

; UBO struct
%UBO = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float

; Pointer types
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Function_float = OpTypePointer Function %float
%ptr_Function_int = OpTypePointer Function %int

; Function types
%functype = OpTypeFunction %void

; Constants
%c_0 = OpConstant %float 0.0
%c_1 = OpConstant %float 1.0
%c_2 = OpConstant %float 2.0
%c_3 = OpConstant %float 3.0
%c_4 = OpConstant %float 4.0
%c_0_5 = OpConstant %float 0.5
%c_20 = OpConstant %float 20.0
%i_0 = OpConstant %int 0
%i_1 = OpConstant %int 1
%i_20 = OpConstant %int 20

; Global variables
%ubo_var = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input
%in_tangent = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v3float Input
%out_color = OpVariable %ptr_Output_v4float Output

; Main function
%main = OpFunction %void None %functype
%entry = OpLabel

; Allocate local variables
%z_real = OpVariable %ptr_Function_float Function
%z_imag = OpVariable %ptr_Function_float Function
%iter = OpVariable %ptr_Function_int Function
%c_real = OpVariable %ptr_Function_float Function
%c_imag = OpVariable %ptr_Function_float Function

; Load UV
%uv = OpLoad %v2float %in_uv

; Extract UV coordinates
%uv_x = OpCompositeExtract %float %uv 0
%uv_y = OpCompositeExtract %float %uv 1

; Compute c_real = uv.x * 3 - 2
%uv_x_3 = OpFMul %float %uv_x %c_3
%cr_val = OpFSub %float %uv_x_3 %c_2
OpStore %c_real %cr_val

; Compute c_imag = uv.y * 2 - 1
%uv_y_2 = OpFMul %float %uv_y %c_2
%ci_val = OpFSub %float %uv_y_2 %c_1
OpStore %c_imag %ci_val

; Initialize z = 0 + 0i
OpStore %z_real %c_0
OpStore %z_imag %c_0

; Initialize iteration counter
OpStore %iter %i_0

; Loop entry
OpBranch %loop_check

%loop_check = OpLabel
%iter_val = OpLoad %int %iter
%cond = OpSLessThan %bool %iter_val %i_20
OpLoopMerge %loop_exit %loop_cont None
OpBranchConditional %cond %loop_body %loop_exit

%loop_body = OpLabel
%zr = OpLoad %float %z_real
%zi = OpLoad %float %z_imag
%cr = OpLoad %float %c_real
%ci = OpLoad %float %c_imag

; Compute |z|^2
%zr_sq = OpFMul %float %zr %zr
%zi_sq = OpFMul %float %zi %zi
%mag_sq = OpFAdd %float %zr_sq %zi_sq

; Check if |z|^2 > 4
%escape_cond = OpFOrdGreaterThan %bool %mag_sq %c_4
OpSelectionMerge %escape_merge None
OpBranchConditional %escape_cond %loop_exit %escape_merge

%escape_merge = OpLabel
; Compute z = z*z + c
; Real part: zr^2 - zi^2 + cr
%zr_sq_zi_sq = OpFSub %float %zr_sq %zi_sq
%new_zr = OpFAdd %float %zr_sq_zi_sq %cr
OpStore %z_real %new_zr

; Imaginary part: 2*zr*zi + ci
%two_zr = OpFMul %float %c_2 %zr
%two_zr_zi = OpFMul %float %two_zr %zi
%new_zi = OpFAdd %float %two_zr_zi %ci
OpStore %z_imag %new_zi

OpBranch %loop_cont

%loop_cont = OpLabel
%iter_old = OpLoad %int %iter
%iter_new = OpIAdd %int %iter_old %i_1
OpStore %iter %iter_new
OpBranch %loop_check

%loop_exit = OpLabel
%iter_final = OpLoad %int %iter
%iter_float = OpConvertSToF %float %iter_final
%t = OpFDiv %float %iter_float %c_20

; Compute color components
%color_g = OpFMul %float %t %c_0_5
%color_b = OpFSub %float %c_1 %t

; Construct output color: (t, t*0.5, 1.0-t, 1.0)
%out_rgba = OpCompositeConstruct %v4float %t %color_g %color_b %c_1
OpStore %out_color %out_rgba

OpReturn
OpFunctionEnd
