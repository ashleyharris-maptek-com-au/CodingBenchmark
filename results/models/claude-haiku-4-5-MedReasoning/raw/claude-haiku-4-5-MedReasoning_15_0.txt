use std::io::{self, BufRead};

#[derive(Debug, Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Vec3 { x, y, z }
    }

    fn add(self, other: Vec3) -> Vec3 {
        Vec3::new(self.x + other.x, self.y + other.y, self.z + other.z)
    }

    fn mul(self, scalar: f64) -> Vec3 {
        Vec3::new(self.x * scalar, self.y * scalar, self.z * scalar)
    }

    fn len(self) -> f64 {
        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
    }

    fn normalize(self) -> Vec3 {
        let l = self.len();
        if l > 1e-10 {
            self.mul(1.0 / l)
        } else {
            self
        }
    }
}

#[derive(Debug, Clone, Copy)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quat {
    fn identity() -> Self {
        Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 }
    }

    fn rotate_vec3(self, v: Vec3) -> Vec3 {
        let qv_w = -self.x * v.x - self.y * v.y - self.z * v.z;
        let qv_x = self.w * v.x + self.y * v.z - self.z * v.y;
        let qv_y = self.w * v.y + self.z * v.x - self.x * v.z;
        let qv_z = self.w * v.z + self.x * v.y - self.y * v.x;

        let result_x = -qv_w * self.x + qv_x * self.w - qv_y * self.z + qv_z * self.y;
        let result_y = -qv_w * self.y + qv_y * self.w - qv_z * self.x + qv_x * self.z;
        let result_z = -qv_w * self.z + qv_z * self.w - qv_x * self.y + qv_y * self.x;

        Vec3::new(result_x, result_y, result_z)
    }
}

fn project_shadow(point: Vec3, sun_dir: Vec3) -> (f64, f64) {
    if sun_dir.z.abs() < 1e-10 {
        (point.x, point.y)
    } else {
        let t = point.z / sun_dir.z;
        (point.x - t * sun_dir.x, point.y - t * sun_dir.y)
    }
}

fn point_in_polygon(point: (f64, f64), polygon: &[(f64, f64)]) -> bool {
    let (px, py) = point;
    let mut inside = false;
    let mut j = polygon.len() - 1;

    for i in 0..polygon.len() {
        let (x1, y1) = polygon[i];
        let (x2, y2) = polygon[j];

        if ((y1 > py) != (y2 > py)) &&
           (px < (x2 - x1) * (py - y1) / (y2 - y1) + x1) {
            inside = !inside;
        }
        j = i;
    }

    inside
}

fn point_near_polygon(point: (f64, f64), polygon: &[(f64, f64)], margin: f64) -> bool {
    let (px, py) = point;

    for i in 0..polygon.len() {
        let (x1, y1) = polygon[i];
        let (x2, y2) = polygon[(i + 1) % polygon.len()];

        let dx = x2 - x1;
        let dy = y2 - y1;
        let len_sq = dx * dx + dy * dy;

        if len_sq > 0.0 {
            let t = ((px - x1) * dx + (py - y1) * dy) / len_sq;
            let t = t.max(0.0).min(1.0);

            let closest_x = x1 + t * dx;
            let closest_y = y1 + t * dy;

            let dist_sq = (px - closest_x) * (px - closest_x) + (py - closest_y) * (py - closest_y);
            if dist_sq < margin * margin {
                return true;
            }
        }
    }

    false
}

fn tet_reference_vertices() -> Vec<Vec3> {
    vec![
        Vec3::new(1.0, 0.0, -0.707),
        Vec3::new(-1.0, 0.0, -0.707),
        Vec3::new(0.0, 1.0, 0.707),
        Vec3::new(0.0, -1.0, 0.707),
    ]
}

fn compute_shadow_bounds(center: Vec3, quat: Quat, scale: f64, sun_dir: Vec3) -> ((f64, f64), (f64, f64)) {
    let verts = tet_reference_vertices();
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for vert in verts {
        let rotated = quat.rotate_vec3(vert);
        let scaled = rotated.mul(scale);
        let pos = center.add(scaled);
        let (sx, sy) = project_shadow(pos, sun_dir);

        min_x = min_x.min(sx);
        max_x = max_x.max(sx);
        min_y = min_y.min(sy);
        max_y = max_y.max(sy);
    }

    ((min_x, min_y), (max_x, max_y))
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();

    let mut vertices = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.trim().split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        vertices.push((x, y));
    }

    let sun_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = sun_line.trim().split_whitespace().collect();
    let sx: f64 = parts[0].parse().unwrap();
    let sy: f64 = parts[1].parse().unwrap();
    let sz: f64 = parts[2].parse().unwrap();

    let sun_dir = Vec3::new(sx, sy, sz).normalize();

    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for (x, y) in &vertices {
        min_x = min_x.min(*x);
        max_x = max_x.max(*x);
        min_y = min_y.min(*y);
        max_y = max_y.max(*y);
    }

    let width = (max_x - min_x).abs();
    let height = (max_y - min_y).abs();
    let bbox_size = width.max(height).max(0.01);

    let mut tetrahedrons = Vec::new();
    let _tet_verts = tet_reference_vertices();

    let grid_divisions = if bbox_size < 1.0 {
        2
    } else if bbox_size < 5.0 {
        3
    } else if bbox_size < 20.0 {
        4
    } else {
        5
    };

    let cell_size = bbox_size / grid_divisions as f64;

    for i in 0..grid_divisions {
        for j in 0..grid_divisions {
            let px = min_x + (i as f64 + 0.5) * cell_size;
            let py = min_y + (j as f64 + 0.5) * cell_size;
            let pz = bbox_size * 1.5;

            let quat = Quat::identity();
            let scale = (cell_size * 0.8).min(10.0);

            let center = Vec3::new(px, py, pz);
            let ((shadow_min_x, shadow_min_y), (shadow_max_x, shadow_max_y)) =
                compute_shadow_bounds(center, quat, scale, sun_dir);

            let mut covers_area = false;
            let test_points = [
                (shadow_min_x, shadow_min_y),
                (shadow_max_x, shadow_min_y),
                (shadow_min_x, shadow_max_y),
                (shadow_max_x, shadow_max_y),
                ((shadow_min_x + shadow_max_x) / 2.0, (shadow_min_y + shadow_max_y) / 2.0),
            ];

            for test_pt in &test_points {
                if point_in_polygon(*test_pt, &vertices) ||
                   point_near_polygon(*test_pt, &vertices, cell_size * 0.3) {
                    covers_area = true;
                    break;
                }
            }

            if covers_area {
                tetrahedrons.push((px, py, pz, quat.w, quat.x, quat.y, quat.z, scale));
            }
        }
    }

    println!("{}", tetrahedrons.len());
    for (px, py, pz, qw, qx, qy, qz, scale) in tetrahedrons {
        println!("{} {} {} {} {} {} {} {}", px, py, pz, qw, qx, qy, qz, scale);
    }
}
