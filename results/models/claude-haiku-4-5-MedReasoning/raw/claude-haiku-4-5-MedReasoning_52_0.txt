import math
from collections import deque

# Module-level state for filtering and control
_state_filters = {
    'speed_history': deque(maxlen=5),
    'steering_history': deque(maxlen=5),
    'lane_offset_history': deque(maxlen=3),
    'obstacle_distance_history': deque(maxlen=3),
    'lidar_center_history': deque(maxlen=3),
}

_pid_states = {
    'speed_error_integral': 0.0,
    'steering_error_integral': 0.0,
}

_last_valid = {
    'speed': 0.0,
    'heading': 0.0,
    'position_x': 0.0,
    'position_y': 0.0,
    'lane_offset': 0.0,
    'road_curvature': 0.0,
}

def _get_valid_sensor(values, valids, prefer_first=True):
    """Get the most reliable sensor reading using voting/median."""
    valid_readings = []
    for v, valid in zip(values, valids):
        if valid and v is not None:
            valid_readings.append(v)

    if not valid_readings:
        return None

    if len(valid_readings) == 1:
        return valid_readings[0]

    # Use median for stability
    valid_readings.sort()
    return valid_readings[len(valid_readings) // 2]

def _filter_value(value, history, max_change=None):
    """Apply low-pass filter to smooth noisy sensor readings."""
    if value is None:
        return None

    history.append(value)

    if max_change is not None and len(history) > 1:
        prev_avg = sum(list(history)[:-1]) / (len(history) - 1)
        if abs(value - prev_avg) > max_change:
            return prev_avg

    return sum(history) / len(history)

def _get_vehicle_speed(state):
    """Fuse vehicle speed from multiple sources (km/h)."""
    wheel_speeds = [
        state.get('wheel_speed_fl_kmh'),
        state.get('wheel_speed_fr_kmh'),
        state.get('wheel_speed_rl_kmh'),
        state.get('wheel_speed_rr_kmh'),
    ]
    wheel_valids = [
        state.get('wheel_speed_fl_valid'),
        state.get('wheel_speed_fr_valid'),
        state.get('wheel_speed_rl_valid'),
        state.get('wheel_speed_rr_valid'),
    ]

    wheel_speed = _get_valid_sensor(wheel_speeds, wheel_valids)

    speeds = [
        state.get('vehicle_speed_1_kmh'),
        state.get('vehicle_speed_2_kmh'),
        wheel_speed,
    ]

    gps_speeds = [
        state.get('gps_1_speed_kmh') if state.get('gps_1_valid') else None,
        state.get('gps_2_speed_kmh') if state.get('gps_2_valid') else None,
    ]
    speeds.extend(gps_speeds)

    speeds = [s for s in speeds if s is not None and s >= 0]

    if not speeds:
        return _last_valid['speed']

    speeds.sort()
    speed = speeds[len(speeds) // 2]

    filtered = _filter_value(speed, _state_filters['speed_history'], max_change=10.0)
    _last_valid['speed'] = filtered if filtered is not None else _last_valid['speed']

    return _last_valid['speed']

def _get_heading(state):
    """Fuse heading from GPS and gyro."""
    gps_headings = []
    if state.get('gps_1_valid'):
        gps_headings.append(state.get('gps_1_heading_deg'))
    if state.get('gps_2_valid'):
        gps_headings.append(state.get('gps_2_heading_deg'))

    if gps_headings:
        heading = sum(gps_headings) / len(gps_headings)
    else:
        heading = _last_valid['heading']

    _last_valid['heading'] = heading
    return heading

def _get_road_curvature(state):
    """Fuse road curvature from vision systems."""
    curvatures = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_road_curvature_1pm'):
            curvatures.append(state.get(f'vision_{i}_road_curvature_1pm'))

    if not curvatures:
        return _last_valid['road_curvature']

    curvatures.sort()
    curvature = curvatures[len(curvatures) // 2]
    _last_valid['road_curvature'] = curvature
    return curvature

def _get_lane_offset(state):
    """Fuse lane offset from vision systems."""
    offsets = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_lane_offset_m') is not None:
            offsets.append(state.get(f'vision_{i}_lane_offset_m'))

    if offsets:
        offsets.sort()
        offset = offsets[len(offsets) // 2]
        filtered = _filter_value(offset, _state_filters['lane_offset_history'], max_change=0.5)
        _last_valid['lane_offset'] = filtered if filtered is not None else _last_valid['lane_offset']
        return _last_valid['lane_offset']

    return _last_valid['lane_offset']

def _get_lidar_center(state):
    """Get center lidar reading from forward-facing beams."""
    center_readings = []
    for angle in ['+005_deg', '-005_deg', '+015_deg', '-015_deg']:
        key = f'lidar_{angle}_range_m'
        valid_key = f'lidar_{angle}_valid'
        if state.get(valid_key) and state.get(key) is not None:
            center_readings.append(state.get(key))

    if center_readings:
        center_readings.sort()
        center = center_readings[len(center_readings) // 2]
        filtered = _filter_value(center, _state_filters['lidar_center_history'], max_change=2.0)
        return filtered if filtered is not None else center

    return None

def _get_obstacle_info(state):
    """Fuse obstacle detection from vision systems."""
    obstacles = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_obstacle_valid') and state.get(f'vision_{i}_obstacle_detected'):
            dist = state.get(f'vision_{i}_obstacle_dist_m')
            rel_speed = state.get(f'vision_{i}_obstacle_rel_speed_kmh')
            if dist is not None and dist > 0:
                obstacles.append((dist, rel_speed if rel_speed is not None else 0))

    if obstacles:
        obstacles.sort(key=lambda x: x[0])
        dist, rel_speed = obstacles[0]
        filtered = _filter_value(dist, _state_filters['obstacle_distance_history'], max_change=5.0)
        return (filtered if filtered is not None else dist, rel_speed)

    return None, None

def _get_speed_limit(state):
    """Get speed limit from vision systems."""
    limits = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_speed_limit_valid'):
            limit = state.get(f'vision_{i}_speed_limit_kmh')
            if limit is not None and limit > 0:
                limits.append(limit)

    if limits:
        limits.sort()
        return limits[len(limits) // 2]

    return None

def autopilot_step(state: dict, dt: float) -> dict:
    """Single autopilot control step."""

    # Get fused sensor readings
    current_speed = _get_vehicle_speed(state)
    heading = _get_heading(state)
    lane_offset = _get_lane_offset(state)
    road_curvature = _get_road_curvature(state)
    lidar_center = _get_lidar_center(state)
    obstacle_dist, obstacle_rel_speed = _get_obstacle_info(state)
    speed_limit = _get_speed_limit(state)
    mu = state.get('estimated_mu', 1.0)

    # Get targets
    target_speed = state.get('target_speed_kmh', 100.0)
    target_lane = state.get('target_lane', 1)
    current_lane = state.get('current_lane', 1)

    # Respect speed limit
    if speed_limit is not None:
        target_speed = min(target_speed, speed_limit)

    # Adjust target speed based on traction
    if mu < 0.5:
        target_speed = min(target_speed, 60.0)
    elif mu < 0.3:
        target_speed = min(target_speed, 40.0)

    # Obstacle avoidance
    obstacle_throttle = 1.0
    obstacle_brake = 0.0
    obstacle_steering = 0.0

    if obstacle_dist is not None and obstacle_dist < 50.0:
        if obstacle_dist < 20.0:
            obstacle_brake = min(1.0, (20.0 - obstacle_dist) / 20.0)
            obstacle_throttle = 0.0

            if obstacle_rel_speed is None or obstacle_rel_speed < 0:
                target_speed = min(current_speed - 10, target_speed)
        elif obstacle_dist < 40.0:
            obstacle_throttle = 0.5
            target_speed = min(current_speed - 5, target_speed)

    # Speed control with PID
    speed_error = target_speed - current_speed
    _pid_states['speed_error_integral'] += speed_error * dt
    _pid_states['speed_error_integral'] = max(-50, min(50, _pid_states['speed_error_integral']))

    kp_speed = 0.02
    ki_speed = 0.001

    if mu < 0.3:
        kp_speed *= 0.5
        ki_speed *= 0.5

    throttle = 0.5 + kp_speed * speed_error + ki_speed * _pid_states['speed_error_integral']
    throttle = max(0.0, min(1.0, throttle))

    brake = 0.0
    if speed_error < -5.0:
        brake = min(1.0, (-speed_error) / 20.0)
        throttle = 0.0

    # Lane keeping with vision + lidar fusion
    steering_error = 0.0

    if lane_offset is not None:
        steering_error = lane_offset

    if lidar_center is not None and lidar_center < 15.0:
        lidar_center_offset = 7.5 - lidar_center
        steering_error = steering_error * 0.6 + lidar_center_offset * 0.4

    # Add road curvature feedforward
    curve_steering = 0.0
    if road_curvature is not None:
        curve_steering = road_curvature * 0.05

    _pid_states['steering_error_integral'] += steering_error * dt
    _pid_states['steering_error_integral'] = max(-2.0, min(2.0, _pid_states['steering_error_integral']))

    kp_steering = 0.1
    ki_steering = 0.02

    if mu < 0.3:
        kp_steering *= 0.5
        ki_steering *= 0.5

    steering = kp_steering * steering_error + ki_steering * _pid_states['steering_error_integral'] + curve_steering
    steering = max(-1.0, min(1.0, steering))

    # Apply obstacle avoidance overrides
    if obstacle_brake > brake:
        brake = obstacle_brake
    if obstacle_throttle < throttle:
        throttle = obstacle_throttle

    # Reduce aggressive inputs on low-traction surfaces
    if mu < 0.3:
        steering *= 0.7
        throttle = min(throttle, 0.6)
        brake = min(brake, 0.7)
    elif mu < 0.5:
        steering *= 0.85
        throttle = min(throttle, 0.8)

    # Handle stuck throttle/brake (clamp to prevent full acceleration/braking)
    # This is handled by the clipping above

    # Ensure mutually exclusive throttle/brake
    if brake > 0.1:
        throttle = 0.0

    return {
        'steering': steering,
        'throttle': throttle,
        'brake': brake,
    }
