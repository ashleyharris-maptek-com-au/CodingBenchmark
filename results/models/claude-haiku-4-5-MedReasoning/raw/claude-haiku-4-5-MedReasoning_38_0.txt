def solve():
    # Read input
    U, S = map(int, input().split())

    sets = []
    for _ in range(S):
        line = input().strip()
        if line:
            elements = list(map(int, line.split()))
        else:
            elements = []
        sets.append(elements)

    # Target: cover all elements 0..U-1
    target = set(range(U))

    # Backtracking function to find exact cover
    def backtrack(set_index, covered, selected):
        # Found a solution if all elements are covered
        if covered == target:
            return selected[:]

        # No more sets to try
        if set_index >= S:
            return None

        # Convert current set for operations
        current_set = set(sets[set_index])

        # Try including this set if it doesn't overlap with covered elements
        if covered.isdisjoint(current_set):
            selected.append(set_index)
            result = backtrack(set_index + 1, covered | current_set, selected)
            if result is not None:
                return result
            selected.pop()

        # Try excluding this set
        result = backtrack(set_index + 1, covered, selected)
        return result

    result = backtrack(0, set(), [])

    if result is None:
        print("NO SOLUTION")
    else:
        print("SOLUTION")
        print(" ".join(map(str, result)))

solve()
