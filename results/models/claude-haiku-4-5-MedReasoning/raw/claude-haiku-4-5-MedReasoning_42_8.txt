OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
               OpDecorate %color Location 2

        %void = OpTypeVoid
       %float = OpTypeFloat 32
         %int = OpTypeInt 32 1
       %v3float = OpTypeVector %float 3
       %v4float = OpTypeVector %float 4

  %gl_PerVertex_in = OpTypeStruct %v4float
      %uint_3 = OpConstant %int 3
   %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
       %gl_in = OpVariable %ptr_in_arr_pv Input

   %arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
       %color = OpVariable %ptr_in_arr_v3 Input

 %gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
      %gl_out = OpVariable %ptr_out_pv Output

 %ptr_out_v3 = OpTypePointer Output %v3float
     %outColor = OpVariable %ptr_out_v3 Output

    %func_ty = OpTypeFunction %void

         %int_0 = OpConstant %int 0
         %int_1 = OpConstant %int 1
         %int_2 = OpConstant %int 2
       %float_0_7 = OpConstant %float 0.7
       %float_1 = OpConstant %float 1.0
       %float_3 = OpConstant %float 3.0

       %ptr_in_v4float = OpTypePointer Input %v4float
       %ptr_in_v3float = OpTypePointer Input %v3float
       %ptr_out_v4float = OpTypePointer Output %v4float

        %main = OpFunction %void None %func_ty
       %entry = OpLabel

       ; Load positions
       %p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
       %p0 = OpLoad %v4float %p0_ptr
       %p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
       %p1 = OpLoad %v4float %p1_ptr
       %p2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
       %p2 = OpLoad %v4float %p2_ptr

       ; Compute centroid: C = (p0 + p1 + p2) / 3
       %sum01 = OpFAdd %v4float %p0 %p1
       %sum = OpFAdd %v4float %sum01 %p2
       %inv_3 = OpFDiv %float %float_1 %float_3
       %centroid = OpVectorTimesScalar %v4float %sum %inv_3

       ; Emit vertex 0
       %diff0 = OpFSub %v4float %p0 %centroid
       %scaled0 = OpVectorTimesScalar %v4float %diff0 %float_0_7
       %new_pos0 = OpFAdd %v4float %centroid %scaled0

       %color0_ptr = OpAccessChain %ptr_in_v3float %color %int_0
       %color0 = OpLoad %v3float %color0_ptr

       %pos_out_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
       OpStore %pos_out_ptr %new_pos0
       OpStore %outColor %color0
       OpEmitVertex

       ; Emit vertex 1
       %diff1 = OpFSub %v4float %p1 %centroid
       %scaled1 = OpVectorTimesScalar %v4float %diff1 %float_0_7
       %new_pos1 = OpFAdd %v4float %centroid %scaled1

       %color1_ptr = OpAccessChain %ptr_in_v3float %color %int_1
       %color1 = OpLoad %v3float %color1_ptr

       OpStore %pos_out_ptr %new_pos1
       OpStore %outColor %color1
       OpEmitVertex

       ; Emit vertex 2
       %diff2 = OpFSub %v4float %p2 %centroid
       %scaled2 = OpVectorTimesScalar %v4float %diff2 %float_0_7
       %new_pos2 = OpFAdd %v4float %centroid %scaled2

       %color2_ptr = OpAccessChain %ptr_in_v3float %color %int_2
       %color2 = OpLoad %v3float %color2_ptr

       OpStore %pos_out_ptr %new_pos2
       OpStore %outColor %color2
       OpEmitVertex

       OpEndPrimitive
       OpReturn
       OpFunctionEnd
