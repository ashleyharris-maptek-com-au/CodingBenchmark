using System;
using System.Collections.Generic;
using System.Linq;

class Solution {
    static void Main() {
        string[] firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        int m = int.Parse(firstLine[1]);

        var graph = new List<int>[n];
        for (int i = 0; i < n; i++) {
            graph[i] = new List<int>();
        }

        for (int i = 0; i < m; i++) {
            string[] edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            graph[u].Add(v);
        }

        var solution = SolveFVS(n, graph);

        Console.WriteLine(solution.Count);
        if (solution.Count > 0) {
            Console.WriteLine(string.Join(" ", solution));
        }
    }

    static List<int> SolveFVS(int n, List<int>[] graph) {
        var removed = new HashSet<int>();
        var remaining = new HashSet<int>(Enumerable.Range(0, n));

        while (!IsAcyclic(graph, remaining)) {
            // Find vertices involved in back edges (cycle indicators)
            var backEdgeTarget = new Dictionary<int, int>();
            for (int i = 0; i < n; i++) {
                backEdgeTarget[i] = 0;
            }

            CountBackEdgeTargets(graph, remaining, backEdgeTarget);

            // Remove vertex with most back edges pointing to it
            // Use in-degree and out-degree as tiebreakers
            var toRemove = remaining
                .OrderByDescending(v => backEdgeTarget[v])
                .ThenByDescending(v => graph[v].Count(u => remaining.Contains(u)))
                .ThenByDescending(v => remaining.Count(u => graph[u].Contains(v)))
                .First();

            removed.Add(toRemove);
            remaining.Remove(toRemove);
        }

        return removed.ToList();
    }

    static void CountBackEdgeTargets(List<int>[] graph, HashSet<int> remaining, Dictionary<int, int> count) {
        var visited = new HashSet<int>();
        var inStack = new HashSet<int>();

        foreach (var start in remaining.Where(v => !visited.Contains(v))) {
            DFSCountBackEdges(start, graph, remaining, visited, inStack, count);
        }
    }

    static void DFSCountBackEdges(int u, List<int>[] graph, HashSet<int> remaining,
        HashSet<int> visited, HashSet<int> inStack, Dictionary<int, int> count) {
        visited.Add(u);
        inStack.Add(u);

        foreach (var v in graph[u]) {
            if (!remaining.Contains(v)) continue;

            if (inStack.Contains(v)) {
                // Back edge found: u -> v
                count[v]++;
            } else if (!visited.Contains(v)) {
                DFSCountBackEdges(v, graph, remaining, visited, inStack, count);
            }
        }

        inStack.Remove(u);
    }

    static bool IsAcyclic(List<int>[] graph, HashSet<int> remaining) {
        var visited = new HashSet<int>();
        var inStack = new HashSet<int>();

        foreach (var v in remaining.Where(v => !visited.Contains(v))) {
            if (HasCycleDFS(v, graph, remaining, visited, inStack)) {
                return false;
            }
        }

        return true;
    }

    static bool HasCycleDFS(int u, List<int>[] graph, HashSet<int> remaining,
        HashSet<int> visited, HashSet<int> inStack) {
        visited.Add(u);
        inStack.Add(u);

        foreach (var v in graph[u]) {
            if (!remaining.Contains(v)) continue;

            if (!visited.Contains(v)) {
                if (HasCycleDFS(v, graph, remaining, visited, inStack)) {
                    return true;
                }
            } else if (inStack.Contains(v)) {
                return true;
            }
        }

        inStack.Remove(u);
        return false;
    }
}
