#include <iostream>
#include <string>
#include <vector>
#include <cstring>
#include <algorithm>
#include <chrono>
#include <atomic>
#include <thread>
#include <mutex>

using namespace std;

struct Move {
    int from, to;
    int piece;
    int capturedPiece;
    bool isEnPassant;
    bool isCastle;
    bool isPromotion;
    int promotionPiece;

    Move() : from(-1), to(-1), piece(0), capturedPiece(0), isEnPassant(false), isCastle(false), isPromotion(false), promotionPiece(0) {}
    Move(int f, int t, int p) : from(f), to(t), piece(p), capturedPiece(0), isEnPassant(false), isCastle(false), isPromotion(false), promotionPiece(0) {}
};

class ChessBot {
private:
    static const int EMPTY = 0;
    static const int PAWN = 1, KNIGHT = 2, BISHOP = 3, ROOK = 4, QUEEN = 5, KING = 6;
    static const int WHITE = 8, BLACK = 16;

    int board[64];
    bool whiteTurn;
    bool whiteCanCastleKingside, whiteCanCastleQueenside;
    bool blackCanCastleKingside, blackCanCastleQueenside;
    int enPassantSquare;

    chrono::time_point<chrono::high_resolution_clock> startTime;
    int timeoutMs;

    static const int PIECE_VALUES[7];

public:
    ChessBot() : whiteTurn(true), whiteCanCastleKingside(true), whiteCanCastleQueenside(true),
                 blackCanCastleKingside(true), blackCanCastleQueenside(true), enPassantSquare(-1) {
        memset(board, 0, sizeof(board));
    }

    void parseFEN(const string& fen) {
        memset(board, 0, sizeof(board));
        int sq = 0;
        int pos = 0;

        while (pos < fen.length() && fen[pos] != ' ') {
            char c = fen[pos];
            if (c == '/') {
                pos++;
                continue;
            }
            if (isdigit(c)) {
                sq += c - '0';
            } else {
                int piece = charToPiece(c);
                board[sq] = piece;
                sq++;
            }
            pos++;
        }

        while (pos < fen.length() && fen[pos] == ' ') pos++;
        whiteTurn = (fen[pos] == 'w');

        while (pos < fen.length() && fen[pos] == ' ') pos++;
        pos++;

        whiteCanCastleKingside = whiteCanCastleQueenside = false;
        blackCanCastleKingside = blackCanCastleQueenside = false;
        while (pos < fen.length() && fen[pos] != ' ') {
            if (fen[pos] == 'K') whiteCanCastleKingside = true;
            if (fen[pos] == 'Q') whiteCanCastleQueenside = true;
            if (fen[pos] == 'k') blackCanCastleKingside = true;
            if (fen[pos] == 'q') blackCanCastleQueenside = true;
            if (fen[pos] == '-') break;
            pos++;
        }

        while (pos < fen.length() && fen[pos] == ' ') pos++;
        enPassantSquare = -1;
        if (pos < fen.length() && fen[pos] != '-') {
            int file = fen[pos] - 'a';
            int rank = fen[pos+1] - '1';
            enPassantSquare = rank * 8 + file;
        }
    }

    int charToPiece(char c) {
        switch(c) {
            case 'P': return PAWN | WHITE;
            case 'N': return KNIGHT | WHITE;
            case 'B': return BISHOP | WHITE;
            case 'R': return ROOK | WHITE;
            case 'Q': return QUEEN | WHITE;
            case 'K': return KING | WHITE;
            case 'p': return PAWN | BLACK;
            case 'n': return KNIGHT | BLACK;
            case 'b': return BISHOP | BLACK;
            case 'r': return ROOK | BLACK;
            case 'q': return QUEEN | BLACK;
            case 'k': return KING | BLACK;
            default: return EMPTY;
        }
    }

    char pieceToChar(int piece) {
        if (piece == EMPTY) return '.';
        int color = piece & (WHITE | BLACK);
        int type = piece & 7;

        if (color == BLACK) {
            switch(type) {
                case PAWN: return 'p';
                case KNIGHT: return 'n';
                case BISHOP: return 'b';
                case ROOK: return 'r';
                case QUEEN: return 'q';
                case KING: return 'k';
            }
        } else {
            switch(type) {
                case PAWN: return 'P';
                case KNIGHT: return 'N';
                case BISHOP: return 'B';
                case ROOK: return 'R';
                case QUEEN: return 'Q';
                case KING: return 'K';
            }
        }
        return '?';
    }

    bool isWhitePiece(int piece) {
        return (piece & WHITE) != 0;
    }

    bool isBlackPiece(int piece) {
        return (piece & BLACK) != 0;
    }

    void generateMoves(vector<Move>& moves) {
        int color = whiteTurn ? WHITE : BLACK;

        for (int sq = 0; sq < 64; sq++) {
            int piece = board[sq];
            if (piece == EMPTY || ((piece & (WHITE | BLACK)) != color)) continue;

            int type = piece & 7;

            if (type == PAWN) generatePawnMoves(sq, moves);
            else if (type == KNIGHT) generateKnightMoves(sq, moves);
            else if (type == BISHOP) generateBishopMoves(sq, moves);
            else if (type == ROOK) generateRookMoves(sq, moves);
            else if (type == QUEEN) generateQueenMoves(sq, moves);
            else if (type == KING) generateKingMoves(sq, moves);
        }
    }

    void generatePawnMoves(int sq, vector<Move>& moves) {
        int rank = sq / 8;
        int file = sq % 8;
        int forward = whiteTurn ? 8 : -8;
        int startRank = whiteTurn ? 1 : 6;
        int promoteRank = whiteTurn ? 7 : 0;

        int newSq = sq + forward;
        if (newSq >= 0 && newSq < 64 && board[newSq] == EMPTY) {
            if (rank == promoteRank) {
                moves.push_back(Move(sq, newSq, board[sq]));
                moves.back().isPromotion = true;
                moves.back().promotionPiece = QUEEN;
            } else {
                moves.push_back(Move(sq, newSq, board[sq]));
            }

            if (rank == startRank) {
                int doubleSq = sq + 2 * forward;
                if (board[doubleSq] == EMPTY) {
                    moves.push_back(Move(sq, doubleSq, board[sq]));
                }
            }
        }

        for (int df : {-1, 1}) {
            int newFile = file + df;
            if (newFile >= 0 && newFile < 8) {
                int captSq = sq + forward + df;
                if (board[captSq] != EMPTY && !isSameColor(board[sq], board[captSq])) {
                    if (rank == promoteRank) {
                        moves.push_back(Move(sq, captSq, board[sq]));
                        moves.back().isPromotion = true;
                        moves.back().promotionPiece = QUEEN;
                        moves.back().capturedPiece = board[captSq];
                    } else {
                        moves.push_back(Move(sq, captSq, board[sq]));
                        moves.back().capturedPiece = board[captSq];
                    }
                }

                if (captSq == enPassantSquare) {
                    moves.push_back(Move(sq, captSq, board[sq]));
                    moves.back().isEnPassant = true;
                    moves.back().capturedPiece = board[sq - forward];
                }
            }
        }
    }

    void generateKnightMoves(int sq, vector<Move>& moves) {
        int deltas[] = {-17, -15, -10, -6, 6, 10, 15, 17};
        for (int d : deltas) {
            int newSq = sq + d;
            if (newSq >= 0 && newSq < 64 && abs(newSq % 8 - sq % 8) <= 2) {
                if (board[newSq] == EMPTY || !isSameColor(board[sq], board[newSq])) {
                    moves.push_back(Move(sq, newSq, board[sq]));
                    if (board[newSq] != EMPTY) moves.back().capturedPiece = board[newSq];
                }
            }
        }
    }

    void addSlidingMoves(int sq, const int* deltas, vector<Move>& moves) {
        for (int d : deltas) {
            for (int newSq = sq + d; newSq >= 0 && newSq < 64; newSq += d) {
                if (abs(newSq % 8 - sq % 8) > 8) break;

                if (board[newSq] == EMPTY) {
                    moves.push_back(Move(sq, newSq, board[sq]));
                } else if (!isSameColor(board[sq], board[newSq])) {
                    moves.push_back(Move(sq, newSq, board[sq]));
                    moves.back().capturedPiece = board[newSq];
                    break;
                } else {
                    break;
                }
            }
        }
    }

    void generateBishopMoves(int sq, vector<Move>& moves) {
        int deltas[] = {-9, -7, 7, 9, 0};
        addSlidingMoves(sq, deltas, moves);
    }

    void generateRookMoves(int sq, vector<Move>& moves) {
        int deltas[] = {-8, -1, 1, 8, 0};
        addSlidingMoves(sq, deltas, moves);
    }

    void generateQueenMoves(int sq, vector<Move>& moves) {
        int deltas[] = {-9, -8, -7, -1, 1, 7, 8, 9, 0};
        addSlidingMoves(sq, deltas, moves);
    }

    void generateKingMoves(int sq, vector<Move>& moves) {
        for (int newSq = sq - 9; newSq <= sq + 9; newSq++) {
            if (newSq == sq) continue;
            if (newSq < 0 || newSq >= 64) continue;
            if (abs(newSq % 8 - sq % 8) > 1) continue;

            if (board[newSq] == EMPTY || !isSameColor(board[sq], board[newSq])) {
                moves.push_back(Move(sq, newSq, board[sq]));
                if (board[newSq] != EMPTY) moves.back().capturedPiece = board[newSq];
            }
        }

        int rank = sq / 8;
        if (whiteTurn && rank == 0 && sq == 4) {
            if (whiteCanCastleKingside && board[5] == EMPTY && board[6] == EMPTY) {
                moves.push_back(Move(sq, 6, board[sq]));
                moves.back().isCastle = true;
            }
            if (whiteCanCastleQueenside && board[3] == EMPTY && board[2] == EMPTY && board[1] == EMPTY) {
                moves.push_back(Move(sq, 2, board[sq]));
                moves.back().isCastle = true;
            }
        } else if (!whiteTurn && rank == 7 && sq == 60) {
            if (blackCanCastleKingside && board[61] == EMPTY && board[62] == EMPTY) {
                moves.push_back(Move(sq, 62, board[sq]));
                moves.back().isCastle = true;
            }
            if (blackCanCastleQueenside && board[59] == EMPTY && board[58] == EMPTY && board[57] == EMPTY) {
                moves.push_back(Move(sq, 58, board[sq]));
                moves.back().isCastle = true;
            }
        }
    }

    bool isSameColor(int p1, int p2) {
        return ((p1 & (WHITE | BLACK)) == (p2 & (WHITE | BLACK)));
    }

    void makeMove(const Move& m) {
        board[m.to] = board[m.from];
        board[m.from] = EMPTY;

        if (m.isCastle) {
            if (m.to == 6) {
                board[5] = board[7];
                board[7] = EMPTY;
            } else if (m.to == 2) {
                board[3] = board[0];
                board[0] = EMPTY;
            } else if (m.to == 62) {
                board[61] = board[63];
                board[63] = EMPTY;
            } else if (m.to == 58) {
                board[59] = board[56];
                board[56] = EMPTY;
            }
        }

        if (m.isEnPassant) {
            int captSq = whiteTurn ? m.to - 8 : m.to + 8;
            board[captSq] = EMPTY;
        }

        if (m.isPromotion) {
            board[m.to] = m.promotionPiece | (whiteTurn ? WHITE : BLACK);
        }

        whiteTurn = !whiteTurn;
    }

    void unmakeMove(const Move& m) {
        whiteTurn = !whiteTurn;

        board[m.from] = board[m.to];
        board[m.to] = m.capturedPiece;

        if (m.isCastle) {
            if (m.to == 6) {
                board[7] = board[5];
                board[5] = EMPTY;
            } else if (m.to == 2) {
                board[0] = board[3];
                board[3] = EMPTY;
            } else if (m.to == 62) {
                board[63] = board[61];
                board[61] = EMPTY;
            } else if (m.to == 58) {
                board[56] = board[59];
                board[59] = EMPTY;
            }
        }

        if (m.isEnPassant) {
            int captSq = whiteTurn ? m.to - 8 : m.to + 8;
            board[captSq] = m.capturedPiece;
        }
    }

    int evaluate() {
        int score = 0;
        for (int sq = 0; sq < 64; sq++) {
            int piece = board[sq];
            if (piece == EMPTY) continue;

            int type = piece & 7;
            int value = PIECE_VALUES[type];

            if (isWhitePiece(piece)) score += value;
            else score -= value;

            score += positionBonus(sq, piece);
        }
        return whiteTurn ? score : -score;
    }

    int positionBonus(int sq, int piece) {
        int type = piece & 7;
        int rank = sq / 8;

        if (type == PAWN) {
            if (isWhitePiece(piece)) return rank * 10;
            else return (7 - rank) * 10;
        } else if (type == KNIGHT || type == BISHOP) {
            return 5;
        }
        return 0;
    }

    int search(int depth, int alpha, int beta) {
        if (depth == 0) return evaluate();

        auto now = chrono::high_resolution_clock::now();
        int elapsed = chrono::duration_cast<chrono::milliseconds>(now - startTime).count();
        if (elapsed > timeoutMs * 0.8) return evaluate();

        vector<Move> moves;
        generateMoves(moves);

        if (moves.empty()) return whiteTurn ? -10000 : 10000;

        for (const Move& m : moves) {
            makeMove(m);
            int score = -search(depth - 1, -beta, -alpha);
            unmakeMove(m);

            alpha = max(alpha, score);
            if (alpha >= beta) break;
        }

        return alpha;
    }

    Move findBestMove() {
        startTime = chrono::high_resolution_clock::now();

        vector<Move> bestMoves;
        int bestScore = whiteTurn ? -100000 : 100000;

        for (int depth = 1; depth <= 20; depth++) {
            vector<Move> moves;
            generateMoves(moves);

            bestMoves.clear();
            bestScore = whiteTurn ? -100000 : 100000;

            for (const Move& m : moves) {
                makeMove(m);
                int score = -search(depth - 1, -100000, 100000);
                unmakeMove(m);

                if (whiteTurn) {
                    if (score > bestScore) {
                        bestScore = score;
                        bestMoves.clear();
                        bestMoves.push_back(m);
                    } else if (score == bestScore) {
                        bestMoves.push_back(m);
                    }
                } else {
                    if (score < bestScore) {
                        bestScore = score;
                        bestMoves.clear();
                        bestMoves.push_back(m);
                    } else if (score == bestScore) {
                        bestMoves.push_back(m);
                    }
                }
            }

            auto now = chrono::high_resolution_clock::now();
            int elapsed = chrono::duration_cast<chrono::milliseconds>(now - startTime).count();
            if (elapsed > timeoutMs * 0.9) break;
        }

        return bestMoves.empty() ? Move() : bestMoves[0];
    }

    string moveToString(const Move& m) {
        string result;

        int type = board[m.from] & 7;
        char pieceChar = 'p';
        if (type == KNIGHT) pieceChar = 'N';
        else if (type == BISHOP) pieceChar = 'B';
        else if (type == ROOK) pieceChar = 'R';
        else if (type == QUEEN) pieceChar = 'Q';
        else if (type == KING) pieceChar = 'K';

        if (m.isCastle) {
            if (m.to == 6 || m.to == 62) return "O-O";
            else return "O-O-O";
        }

        result += pieceChar;
        result += (char)('a' + m.from % 8);
        result += (char)('1' + m.from / 8);

        if (m.capturedPiece != EMPTY && !m.isEnPassant) {
            result += 'x';
        } else if (m.isEnPassant) {
            result += 'x';
        }

        result += (char)('a' + m.to % 8);
        result += (char)('1' + m.to / 8);

        if (m.isEnPassant) result += "ep";

        if (m.isPromotion) {
            result += "/Q";
        }

        return result;
    }

    void solve(int timeoutSeconds) {
        timeoutMs = timeoutSeconds * 1000;
        Move best = findBestMove();
        cout << moveToString(best) << endl;
    }
};

const int ChessBot::PIECE_VALUES[] = {0, 100, 300, 300, 500, 900, 10000};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int timeout;
    string fen;
    cin >> timeout;
    cin.ignore();
    getline(cin, fen);

    ChessBot bot;
    bot.parseFEN(fen);
    bot.solve(timeout);

    return 0;
}
