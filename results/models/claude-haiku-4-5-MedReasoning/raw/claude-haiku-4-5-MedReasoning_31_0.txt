use std::io::{self, BufRead};
use std::collections::HashMap;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line: num_numbers target
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let _n: usize = parts[0].parse().unwrap();
    let target: i64 = parts[1].parse().unwrap();

    // Read second line: numbers
    let second_line = lines.next().unwrap().unwrap();
    let numbers: Vec<i64> = second_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    // Solve subset sum
    match solve_subset_sum(&numbers, target) {
        Some(indices) => {
            println!("YES");
            println!("{}", indices.iter().map(|i| i.to_string()).collect::<Vec<_>>().join(" "));
        }
        None => {
            println!("NO");
        }
    }
}

fn solve_subset_sum(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(vec![]);
    }

    let n = numbers.len();
    if n == 0 {
        return None;
    }

    let mut dp: HashMap<i64, (i64, usize)> = HashMap::new();
    dp.insert(0, (0, n));

    for (i, &num) in numbers.iter().enumerate() {
        let old_sums: Vec<i64> = dp.keys().cloned().collect();
        for sum in old_sums {
            let new_sum = sum + num;
            if new_sum == target {
                let mut indices = vec![i];
                let mut current = sum;

                while current != 0 {
                    let (prev, idx) = dp[&current];
                    if idx < n {
                        indices.push(idx);
                    }
                    current = prev;
                }

                indices.reverse();
                return Some(indices);
            }
            if !dp.contains_key(&new_sum) {
                dp.insert(new_sum, (sum, i));
            }
        }
    }

    None
}
