OpCapability Geometry
OpCapability Shader
OpExtension "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %outColor %gl_out %worldPos %normal %color
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0.0
%float_0_08 = OpConstant %float 0.08

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%arr_v3_in = OpTypeArray %v3float %uint_3

%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v3 = OpTypePointer Output %v3float
%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float

%void_func = OpTypeFunction %void

%gl_in = OpVariable %ptr_input_arr_pv Input
%worldPos = OpVariable %ptr_input_arr_v3 Input
%normal = OpVariable %ptr_input_arr_v3 Input
%color = OpVariable %ptr_input_arr_v3 Input
%gl_out = OpVariable %ptr_output_pv Output
%outColor = OpVariable %ptr_output_v3 Output

%main = OpFunction %void None %void_func
%entry = OpLabel

%glsl = OpExtInstImport "GLSL.std.450"

%pos0_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos0_xyz = OpVectorShuffle %v3float %pos0 %pos0 0 1 2

%pos1_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%pos1_xyz = OpVectorShuffle %v3float %pos1 %pos1 0 1 2

%pos2_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos2_xyz = OpVectorShuffle %v3float %pos2 %pos2 0 1 2

%e1 = OpFSub %v3float %pos1_xyz %pos0_xyz
%e2 = OpFSub %v3float %pos2_xyz %pos0_xyz

%cross_result = OpExtInst %v3float %glsl Cross %e1 %e2
%face_normal = OpExtInst %v3float %glsl Normalize %cross_result

%offset = OpVectorTimesScalar %v3float %face_normal %float_0_08

%color0_ptr = OpAccessChain %ptr_input_v3 %color %int_0
%color0 = OpLoad %v3float %color0_ptr

%color1_ptr = OpAccessChain %ptr_input_v3 %color %int_1
%color1 = OpLoad %v3float %color1_ptr

%color2_ptr = OpAccessChain %ptr_input_v3 %color %int_2
%color2 = OpLoad %v3float %color2_ptr

%out_pos_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0

%pos0_x = OpCompositeExtract %float %pos0 0
%pos0_y = OpCompositeExtract %float %pos0 1
%pos0_z = OpCompositeExtract %float %pos0 2
%pos0_w = OpCompositeExtract %float %pos0 3
%off_x = OpCompositeExtract %float %offset 0
%off_y = OpCompositeExtract %float %offset 1
%off_z = OpCompositeExtract %float %offset 2
%new0_x = OpFAdd %float %pos0_x %off_x
%new0_y = OpFAdd %float %pos0_y %off_y
%new0_z = OpFAdd %float %pos0_z %off_z
%new_pos0 = OpCompositeConstruct %v4float %new0_x %new0_y %new0_z %pos0_w
OpStore %out_pos_ptr %new_pos0
OpStore %outColor %color0
OpEmitVertex

%pos1_x = OpCompositeExtract %float %pos1 0
%pos1_y = OpCompositeExtract %float %pos1 1
%pos1_z = OpCompositeExtract %float %pos1 2
%pos1_w = OpCompositeExtract %float %pos1 3
%new1_x = OpFAdd %float %pos1_x %off_x
%new1_y = OpFAdd %float %pos1_y %off_y
%new1_z = OpFAdd %float %pos1_z %off_z
%new_pos1 = OpCompositeConstruct %v4float %new1_x %new1_y %new1_z %pos1_w
OpStore %out_pos_ptr %new_pos1
OpStore %outColor %color1
OpEmitVertex

%pos2_x = OpCompositeExtract %float %pos2 0
%pos2_y = OpCompositeExtract %float %pos2 1
%pos2_z = OpCompositeExtract %float %pos2 2
%pos2_w = OpCompositeExtract %float %pos2 3
%new2_x = OpFAdd %float %pos2_x %off_x
%new2_y = OpFAdd %float %pos2_y %off_y
%new2_z = OpFAdd %float %pos2_z %off_z
%new_pos2 = OpCompositeConstruct %v4float %new2_x %new2_y %new2_z %pos2_w
OpStore %out_pos_ptr %new_pos2
OpStore %outColor %color2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
