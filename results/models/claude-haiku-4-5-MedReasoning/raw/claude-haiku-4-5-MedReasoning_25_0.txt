using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class AsteroidInterceptionSolver
{
    const double G = 6.67430e-11;
    const double SECONDS_PER_DAY = 86400.0;
    const double AU = 1.496e11;

    struct Body
    {
        public string Name;
        public double Mass;
        public double[] Pos;
        public double[] Vel;
    }

    struct Burn
    {
        public double Time;
        public double[] DV;
    }

    static void Main()
    {
        try
        {
            // Parse input
            string[] line1 = Console.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            int numBodies = int.Parse(line1[0]);
            double warningDays = double.Parse(line1[1]);
            double impactorMass = double.Parse(line1[2]);
            double deltaVBudget = double.Parse(line1[3]);
            double asteroidMass = double.Parse(line1[4]);
            long t0Unix = long.Parse(line1[5]);

            var bodies = new Dictionary<string, Body>();
            for (int i = 0; i < numBodies; i++)
            {
                string[] parts = Console.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
                string name = parts[0];
                double mass = double.Parse(parts[1]);
                double[] pos = new[] { double.Parse(parts[2]), double.Parse(parts[3]), double.Parse(parts[4]) };
                double[] vel = new[] { double.Parse(parts[5]), double.Parse(parts[6]), double.Parse(parts[7]) };
                bodies[name] = new Body { Name = name, Mass = mass, Pos = pos, Vel = vel };
            }

            string de421Path = Console.ReadLine().Trim();

            // Solve
            var burns = SolveInterception(bodies, warningDays, deltaVBudget, asteroidMass, impactorMass);

            // Output
            Console.WriteLine(burns.Count);
            foreach (var b in burns)
                Console.WriteLine($"{b.Time:F6} {b.DV[0]:E10} {b.DV[1]:E10} {b.DV[2]:E10}");
        }
        catch (Exception ex)
        {
            Console.Error.WriteLine($"Error: {ex.Message}");
            Console.WriteLine(0);
        }
    }

    static List<Burn> SolveInterception(
        Dictionary<string, Body> bodies,
        double warningDays, double deltaVBudget, double asteroidMass, double impactorMass)
    {
        var burns = new List<Burn>();

        Body ast = bodies["Asteroid"];
        Body earth = bodies["Earth"];

        // Find impact time (search when asteroid reaches Earth proximity)
        double impactTime = FindImpactTime(ast, earth, warningDays);

        // Initial spacecraft state (from Earth)
        double[] scPos = (double[])earth.Pos.Clone();
        double[] scVel = (double[])earth.Vel.Clone();

        // Launch burn (t=0): transfer to intercept
        double[] launchBurn = PlanInterceptBurn(scPos, scVel, ast.Pos, ast.Vel, impactTime, deltaVBudget);
        burns.Add(new Burn { Time = 0, DV = launchBurn });

        // Apply burn
        for (int i = 0; i < 3; i++) scVel[i] += launchBurn[i];
        double remainDV = deltaVBudget - Mag(launchBurn);

        // Mid-course correction at 40% of transfer
        if (remainDV > 50 && impactTime > 10)
        {
            double corrTime = impactTime * 0.4;
            Propagate(scPos, scVel, corrTime);
            Propagate(ast.Pos, ast.Vel, corrTime);

            double[] corrBurn = RefineBurn(scPos, scVel, ast.Pos, ast.Vel, impactTime - corrTime, remainDV * 0.4);
            if (Mag(corrBurn) > 1)
            {
                burns.Add(new Burn { Time = corrTime, DV = corrBurn });
                for (int i = 0; i < 3; i++) scVel[i] += corrBurn[i];
                remainDV -= Mag(corrBurn);
            }
        }

        // Final approach burn
        Propagate(scPos, scVel, impactTime);
        Propagate(ast.Pos, ast.Vel, impactTime);

        double[] finalBurn = CalculateImpactBurn(scPos, scVel, ast.Pos, ast.Vel, remainDV);
        burns.Add(new Burn { Time = impactTime, DV = finalBurn });

        return burns;
    }

    static double FindImpactTime(Body ast, Body earth, double maxDays)
    {
        double bestTime = maxDays * 0.5;
        double bestDist = double.MaxValue;

        // Sample forward in time
        for (double t = 0; t <= maxDays; t += Math.Max(1, maxDays / 100))
        {
            double[] ap = (double[])ast.Pos.Clone();
            double[] av = (double[])ast.Vel.Clone();
            double[] ep = (double[])earth.Pos.Clone();
            double[] ev = (double[])earth.Vel.Clone();

            Propagate(ap, av, t);
            Propagate(ep, ev, t);

            double d = Dist(ap, ep);
            if (d < bestDist)
            {
                bestDist = d;
                bestTime = t;
            }
            if (d < 1e7) break; // Found close approach
        }

        return Math.Max(0.1, bestTime);
    }

    static double[] PlanInterceptBurn(
        double[] scPos, double[] scVel, double[] astPos, double[] astVel,
        double timeToImpact, double budget)
    {
        double[] rel = Sub(astPos, scPos);
        double dist = Mag(rel);

        if (dist < 1e3) return new[] { 0.0, 0.0, 0.0 };

        // Target velocity to meet asteroid
        double[] targetVel = new double[3];
        for (int i = 0; i < 3; i++)
            targetVel[i] = (rel[i] / dist) * (dist / Math.Max(1, timeToImpact)) + astVel[i];

        double[] dv = Sub(targetVel, scVel);
        double mag = Mag(dv);

        if (mag > budget) for (int i = 0; i < 3; i++) dv[i] *= budget / mag;
        return dv;
    }

    static double[] RefineBurn(
        double[] scPos, double[] scVel, double[] astPos, double[] astVel,
        double timeRemaining, double budget)
    {
        double[] rel = Sub(astPos, scPos);
        double dist = Mag(rel);
        if (dist < 1e3) return new[] { 0.0, 0.0, 0.0 };

        double[] dv = new double[3];
        double burnMag = Math.Min(100, budget * 0.5);
        for (int i = 0; i < 3; i++)
            dv[i] = (rel[i] / dist) * burnMag;

        return dv;
    }

    static double[] CalculateImpactBurn(
        double[] scPos, double[] scVel, double[] astPos, double[] astVel, double remainDV)
    {
        // Match velocity with asteroid
        double[] dv = Sub(astVel, scVel);
        double mag = Mag(dv);

        if (mag > remainDV) for (int i = 0; i < 3; i++) dv[i] *= remainDV / mag;
        return dv;
    }

    static void Propagate(double[] pos, double[] vel, double days)
    {
        double secs = days * SECONDS_PER_DAY;
        double dt = 1000; // 1000s steps

        for (double t = 0; t < secs; t += dt)
        {
            double step = Math.Min(dt, secs - t);
            for (int i = 0; i < 3; i++)
                pos[i] += vel[i] * step;
        }
    }

    static double Dist(double[] a, double[] b) => Mag(Sub(a, b));
    static double Mag(double[] v) => Math.Sqrt(v[0]*v[0] + v[1]*v[1] + v[2]*v[2]);
    static double[] Sub(double[] a, double[] b) => new[] { a[0]-b[0], a[1]-b[1], a[2]-b[2] };
}
