const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { grid, robots, jobs } = data;

// BFS to find shortest distance between two points on the grid
function bfs(grid, start, end) {
  if (start.x === end.x && start.y === end.y) return 0;

  const rows = grid.length;
  const cols = grid[0].length;
  const visited = new Set();
  const queue = [{ x: start.x, y: start.y, dist: 0 }];
  visited.add(`${start.x},${start.y}`);

  while (queue.length > 0) {
    const { x, y, dist } = queue.shift();

    // Check all 4 adjacent cells
    const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]];

    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < rows && ny >= 0 && ny < cols &&
          grid[nx][ny] === '.' && !visited.has(`${nx},${ny}`)) {

        if (nx === end.x && ny === end.y) {
          return dist + 1;
        }

        visited.add(`${nx},${ny}`);
        queue.push({ x: nx, y: ny, dist: dist + 1 });
      }
    }
  }

  return Infinity;
}

const numRobots = robots.length;
const numJobs = jobs.length;

// Pre-compute distances between all robot-job pairs
const distances = Array(numRobots).fill(0).map(() => Array(numJobs).fill(0));

for (let r = 0; r < numRobots; r++) {
  for (let j = 0; j < numJobs; j++) {
    distances[r][j] = bfs(grid, robots[r], jobs[j]);
  }
}

// Greedy assignment: create sorted list of (distance, robot, job) tuples
const pairs = [];
for (let r = 0; r < numRobots; r++) {
  for (let j = 0; j < numJobs; j++) {
    pairs.push({ dist: distances[r][j], robot: r, job: j });
  }
}
pairs.sort((a, b) => a.dist - b.dist);

// Assign jobs in order of increasing distance
const assignments = Array(numRobots).fill(0).map(() => []);
const assigned = new Set();

for (const { robot, job } of pairs) {
  if (!assigned.has(job)) {
    assignments[robot].push(job);
    assigned.add(job);
  }
}

// Format output: convert job indices to job IDs
const result = {
  assignments: assignments.map(jobIndices => jobIndices.map(j => jobs[j].id))
};

console.log(JSON.stringify(result));
