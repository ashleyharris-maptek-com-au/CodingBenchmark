import math

# Module-level state for persistent tracking
state_mem = {
    'last_pos': [0, 0, 0],
    'last_vel': [0, 0, 0],
    'gyro_bias': [0, 0, 0],
    'phase': 'far',  # far, medium, close, final
    'fuel_reserve': 50,  # kg, keep minimum reserve
    'int_error_y': 0,  # integral error for y-axis
    'int_error_x': 0,  # integral error for x-axis
    'int_error_z': 0,  # integral error for z-axis
}

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Spacecraft docking autopilot with orbital mechanics awareness.
    Handles sensor fusion, redundancy, and counter-intuitive orbital dynamics.
    """

    # Extract and validate sensor data with redundancy
    rel_pos = fuse_position(state)
    rel_vel = fuse_velocity(state)
    attitude = fuse_attitude(state)
    distance = math.sqrt(rel_pos[0]**2 + rel_pos[1]**2 + rel_pos[2]**2) if rel_pos else 1000
    speed = math.sqrt(rel_vel[0]**2 + rel_vel[1]**2 + rel_vel[2]**2) if rel_vel else 1.0

    # Determine phase based on range
    if distance < 10:
        state_mem['phase'] = 'final'
    elif distance < 50:
        state_mem['phase'] = 'close'
    elif distance < 200:
        state_mem['phase'] = 'medium'
    else:
        state_mem['phase'] = 'far'

    # Get fuel available for maneuvers (keep 50kg reserve)
    fuel_available = max(0, state['fuel_remaining_kg'] - state_mem['fuel_reserve'])

    # Generate thrust command based on phase
    if distance < 2.0 and speed < 0.3:
        # Already docked or very close - minimal corrections only
        thrust = [0, 0, 0]
    else:
        thrust = compute_thrust(rel_pos, rel_vel, distance, state_mem['phase'], fuel_available)

    # Attitude control to align with velocity vector for efficient maneuvering
    att_rates = compute_attitude_rates(rel_vel, attitude, state_mem['phase'])

    # Update state memory for next iteration
    if rel_pos:
        state_mem['last_pos'] = rel_pos[:]
        state_mem['last_vel'] = rel_vel[:]

    return {
        'thrust_x': thrust[0],
        'thrust_y': thrust[1],
        'thrust_z': thrust[2],
        'att_rate_roll': att_rates[0],
        'att_rate_pitch': att_rates[1],
        'att_rate_yaw': att_rates[2],
    }


def fuse_position(state):
    """Fuse multiple position sources with redundancy and validation."""
    candidates = []

    # Lidar (preferred if valid)
    if (state.get('lidar_rel_x_valid') and state.get('lidar_rel_y_valid') and
        state.get('lidar_rel_z_valid')):
        candidates.append([
            state['lidar_rel_pos_x_m'],
            state['lidar_rel_pos_y_m'],
            state['lidar_rel_pos_z_m']
        ])

    # Radar (backup)
    if (state.get('radar_rel_x_valid') and state.get('radar_rel_y_valid') and
        state.get('radar_rel_z_valid')):
        candidates.append([
            state['radar_rel_pos_x_m'],
            state['radar_rel_pos_y_m'],
            state['radar_rel_pos_z_m']
        ])

    # Docking camera (only near, provides bearing + range)
    if (state.get('dock_camera_active') and
        state.get('dock_camera_bearing_x_valid') and
        state.get('dock_camera_range_m')):
        rng = state['dock_camera_range_m']
        if rng > 0:
            # bearing_* are unit vectors, multiply by range
            candidates.append([
                state['dock_camera_bearing_x'] * rng,
                state['dock_camera_bearing_y'] * rng,
                state['dock_camera_bearing_z'] * rng,
            ])

    if candidates:
        # Average multiple valid sources for robustness
        avg = [sum(c[i] for c in candidates) / len(candidates) for i in range(3)]
        return avg

    # Fallback: use range finders to estimate distance, maintain direction
    rng_vals = []
    for i in [1, 2, 3]:
        if state.get(f'range_{i}_valid'):
            rng_vals.append(state[f'range_{i}_m'])

    if rng_vals:
        avg_range = sum(rng_vals) / len(rng_vals)
        # Assume mostly behind (negative y) by default
        return [0, -avg_range, 0]

    return None


def fuse_velocity(state):
    """Fuse velocity from radar/lidar with redundancy."""
    candidates = []

    # Lidar velocity
    if (state.get('lidar_relv_x_valid') and state.get('lidar_relv_y_valid') and
        state.get('lidar_relv_z_valid')):
        candidates.append([
            state['lidar_rel_vel_x_ms'],
            state['lidar_rel_vel_y_ms'],
            state['lidar_rel_vel_z_ms']
        ])

    # Radar velocity
    if (state.get('radar_relv_x_valid') and state.get('radar_relv_y_valid') and
        state.get('radar_relv_z_valid')):
        candidates.append([
            state['radar_rel_vel_x_ms'],
            state['radar_rel_vel_y_ms'],
            state['radar_rel_vel_z_ms']
        ])

    if candidates:
        avg = [sum(c[i] for c in candidates) / len(candidates) for i in range(3)]
        return avg

    return [0, 0, 0]


def fuse_attitude(state):
    """Extract spacecraft attitude from redundant sources."""
    # Star trackers are most reliable for attitude
    if (state.get('star_tracker_1_pitch_valid') and
        state.get('star_tracker_1_roll_valid') and
        state.get('star_tracker_1_yaw_valid')):
        return [
            math.radians(state['star_tracker_1_roll_deg']),
            math.radians(state['star_tracker_1_pitch_deg']),
            math.radians(state['star_tracker_1_yaw_deg']),
        ]

    if (state.get('star_tracker_2_pitch_valid') and
        state.get('star_tracker_2_roll_valid') and
        state.get('star_tracker_2_yaw_valid')):
        return [
            math.radians(state['star_tracker_2_roll_deg']),
            math.radians(state['star_tracker_2_pitch_deg']),
            math.radians(state['star_tracker_2_yaw_deg']),
        ]

    # Fallback: assume neutral attitude
    return [0, 0, 0]


def compute_thrust(rel_pos, rel_vel, distance, phase, fuel_available):
    """
    Compute thrust command based on phase and orbital mechanics.
    Key insight: counter-intuitive orbital dynamics apply.
    """
    thrust = [0, 0, 0]

    if not rel_pos or not rel_vel:
        return thrust

    # Scaling: limit thrust magnitude based on fuel
    thrust_scale = min(1.0, fuel_available / 100.0) if fuel_available > 0 else 0

    if phase == 'final':
        # Very close (<10m): use gentle linear control
        # Just oppose relative velocity to bring closure rate to near-zero
        kp = 0.05
        kd = 0.5

        thrust[0] = -kp * rel_pos[0] - kd * rel_vel[0]  # radial
        thrust[1] = -kp * rel_pos[1] - kd * rel_vel[1]  # along-track
        thrust[2] = -kp * rel_pos[2] - kd * rel_vel[2]  # cross-track

        # Clamp to gentle thrusts
        thrust = [max(-0.05, min(0.05, t)) for t in thrust]

    elif phase == 'close':
        # 10-50m: linear regime (CW equations approximation)
        kp = 0.1
        kd = 0.3

        thrust[0] = -kp * rel_pos[0] - kd * rel_vel[0]
        thrust[1] = -kp * rel_pos[1] - kd * rel_vel[1]
        thrust[2] = -kp * rel_pos[2] - kd * rel_vel[2]

        thrust = [max(-0.3, min(0.3, t)) for t in thrust]

    elif phase == 'medium':
        # 50-200m: V-bar approach preferred
        # Correct along-track (y) gently, use radial to adjust

        # For y (along-track): if behind (neg y), we want positive velocity
        # In orbit: if we're behind, we need to lower orbit (retrograde thrust) to go faster
        # But at this range we use direct approach
        y_error = rel_pos[1]  # negative if behind
        y_vel = rel_vel[1]

        # Gentle proportional control on along-track
        thrust[1] = -0.15 * y_error - 0.2 * y_vel

        # Radial corrections (damping only for stability)
        thrust[0] = -0.1 * rel_vel[0]

        # Cross-track corrections
        thrust[2] = -0.1 * rel_pos[2] - 0.15 * rel_vel[2]

        thrust = [max(-0.3, min(0.3, t)) for t in thrust]

    else:  # far phase (>200m)
        # Use Hohmann-like strategy
        distance_xy = math.sqrt(rel_pos[1]**2 + rel_pos[0]**2)

        if abs(rel_pos[1]) > 100 and rel_vel[1] < 0.1:
            # Behind and not catching up: use retrograde burn
            # Lower orbit to go faster - but subtle at this range
            thrust[1] = -0.15  # retrograde to lower orbit
        else:
            # General approach: gentle correction
            thrust[1] = -0.1 * rel_vel[1]

        # Radial adjustment if too high/low
        if rel_pos[0] > 50:
            thrust[0] = -0.1  # lower orbit
        elif rel_pos[0] < -50:
            thrust[0] = 0.1   # raise orbit

        # Cross-track correction
        if abs(rel_pos[2]) > 50:
            thrust[2] = -0.08 * (rel_pos[2] / abs(rel_pos[2]))

    # Apply fuel scaling
    thrust = [t * thrust_scale for t in thrust]

    # Hard limits
    thrust = [max(-1.0, min(1.0, t)) for t in thrust]

    return thrust


def compute_attitude_rates(rel_vel, attitude, phase):
    """
    Control attitude to align with desired thrust direction.
    Slow, deliberate rate commands (max 0.035 rad/s).
    """
    roll_rate = 0
    pitch_rate = 0
    yaw_rate = 0

    # For docking, we want to be aligned with velocity vector
    # Use gyro cross-check to validate attitude sensors
    # Typically small attitude corrections are needed

    if phase == 'final':
        # Minimal attitude changes when very close
        max_rate = 0.005
    elif phase == 'close':
        max_rate = 0.01
    else:
        max_rate = 0.03

    # Gentle attitude control: point along velocity vector
    # This is oversimplified but safe - full 3-axis attitude control
    # would require quaternions and is complex

    # Keep attitude near zero (aligned with orbit frame)
    roll_rate = max(-max_rate, min(max_rate, -attitude[0] * 0.01))
    pitch_rate = max(-max_rate, min(max_rate, -attitude[1] * 0.01))
    yaw_rate = max(-max_rate, min(max_rate, -attitude[2] * 0.01))

    return [roll_rate, pitch_rate, yaw_rate]
