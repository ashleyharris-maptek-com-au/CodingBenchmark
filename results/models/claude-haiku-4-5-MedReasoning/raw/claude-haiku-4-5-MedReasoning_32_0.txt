#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <set>

using namespace std;

const long long INF = 1e18;

struct State {
    int mask;
    int vertex;
    long long cost;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int V, E, K;
    cin >> V >> E >> K;

    vector<vector<pair<int, long long>>> adj(V);

    for (int i = 0; i < E; i++) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<int> terminals(K);
    for (int i = 0; i < K; i++) {
        cin >> terminals[i];
    }

    if (K == 1) {
        cout << "0 0\n";
        return 0;
    }

    // Compute shortest paths between all pairs
    vector<vector<long long>> dist(V, vector<long long>(V, INF));

    for (int s = 0; s < V; s++) {
        dist[s][s] = 0;
        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;
        pq.push({0, s});

        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();

            if (d > dist[s][u]) continue;

            for (auto [v, w] : adj[u]) {
                if (dist[s][u] + w < dist[s][v]) {
                    dist[s][v] = dist[s][u] + w;
                    pq.push({dist[s][v], v});
                }
            }
        }
    }

    // Dreyfus-Wagner DP: dp[mask][v] = min cost to connect terminals in mask with tree containing v
    vector<vector<long long>> dp(1 << K, vector<long long>(V, INF));
    vector<vector<int>> parent_v(1 << K, vector<int>(V, -1));
    vector<vector<int>> parent_mask(1 << K, vector<int>(V, -1));

    // Base case: single terminals
    for (int i = 0; i < K; i++) {
        dp[1 << i][terminals[i]] = 0;
    }

    // Fill DP table for all subsets
    for (int mask = 1; mask < (1 << K); mask++) {
        if (__builtin_popcount(mask) == 1) continue;

        // Try combining two disjoint subsets at each vertex
        for (int submask = (mask - 1) & mask; submask > 0; submask = (submask - 1) & mask) {
            int complement = mask ^ submask;
            for (int v = 0; v < V; v++) {
                long long combined = dp[submask][v] + dp[complement][v];
                if (combined < dp[mask][v]) {
                    dp[mask][v] = combined;
                    parent_mask[mask][v] = submask;
                }
            }
        }

        // Extend using shortest paths (Dijkstra-like relaxation)
        bool changed = true;
        for (int iter = 0; iter < V && changed; iter++) {
            changed = false;
            for (int u = 0; u < V; u++) {
                if (dp[mask][u] < INF) {
                    for (auto [v, w] : adj[u]) {
                        if (dp[mask][u] + w < dp[mask][v]) {
                            dp[mask][v] = dp[mask][u] + w;
                            parent_v[mask][v] = u;
                            parent_mask[mask][v] = -1;
                            changed = true;
                        }
                    }
                }
            }
        }
    }

    // Find minimum cost root
    long long minCost = INF;
    int root = -1;
    int fullMask = (1 << K) - 1;
    for (int v = 0; v < V; v++) {
        if (dp[fullMask][v] < minCost) {
            minCost = dp[fullMask][v];
            root = v;
        }
    }

    if (minCost >= INF) {
        cout << "0 0\n";
        return 0;
    }

    // Reconstruct the tree
    set<pair<int, int>> edges;

    function<void(int, int)> reconstruct = [&](int mask, int v) {
        if (mask == 0 || dp[mask][v] >= INF) return;

        if (__builtin_popcount(mask) == 1) return;

        // Check if from combining subsets
        if (parent_mask[mask][v] != -1) {
            int submask = parent_mask[mask][v];
            int complement = mask ^ submask;
            reconstruct(submask, v);
            reconstruct(complement, v);
        } else if (parent_v[mask][v] != -1) {
            // From extension
            int u = parent_v[mask][v];
            reconstruct(mask, u);
            int a = min(u, v), b = max(u, v);
            edges.insert({a, b});
        }
    };

    reconstruct(fullMask, root);

    cout << minCost << " " << edges.size() << "\n";
    for (auto [u, v] : edges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
