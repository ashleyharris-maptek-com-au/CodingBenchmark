declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Robot {
  id: number;
  x: number;
  y: number;
}

interface Job {
  id: number;
  x: number;
  y: number;
  priority: number;
}

interface InputData {
  grid: string[];
  robots: Robot[];
  jobs: Job[];
}

function bfs(grid: string[], start: [number, number], end: [number, number]): number {
  const rows = grid.length;
  if (rows === 0) return Infinity;
  const cols = grid[0].length;

  const sx = start[0], sy = start[1];
  const ex = end[0], ey = end[1];

  if (sx < 0 || sx >= rows || sy < 0 || sy >= cols) return Infinity;
  if (ex < 0 || ex >= rows || ey < 0 || ey >= cols) return Infinity;
  if (grid[sx][sy] === '#' || grid[ex][ey] === '#') return Infinity;
  if (sx === ex && sy === ey) return 0;

  const visited = new Set<string>();
  const queue: Array<[number, number, number]> = [[sx, sy, 0]];
  visited.add(`${sx},${sy}`);

  while (queue.length > 0) {
    const [x, y, dist] = queue.shift()!;

    const directions = [[0, 1], [1, 0], [0, -1], [-1, 0]];
    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx === ex && ny === ey) {
        return dist + 1;
      }

      const key = `${nx},${ny}`;
      if (nx >= 0 && nx < rows && ny >= 0 && ny < cols &&
          !visited.has(key) && grid[nx][ny] === '.') {
        visited.add(key);
        queue.push([nx, ny, dist + 1]);
      }
    }
  }

  return Infinity;
}

function solve(data: InputData): number[][] {
  const { grid, robots, jobs } = data;

  const assignments: number[][] = robots.map(() => []);
  const assignedJobs = new Set<number>();

  const robotPositions: Map<number, [number, number]> = new Map();
  for (const robot of robots) {
    robotPositions.set(robot.id, [robot.x, robot.y]);
  }

  while (assignedJobs.size < jobs.length) {
    let bestJob = -1;
    let bestRobot = -1;
    let bestDist = Infinity;

    for (const job of jobs) {
      if (assignedJobs.has(job.id)) continue;

      for (const robot of robots) {
        const pos = robotPositions.get(robot.id)!;
        const dist = bfs(grid, pos, [job.x, job.y]);

        if (dist < bestDist) {
          bestDist = dist;
          bestJob = job.id;
          bestRobot = robot.id;
        }
      }
    }

    if (bestRobot !== -1 && bestDist !== Infinity) {
      assignments[bestRobot].push(bestJob);
      assignedJobs.add(bestJob);

      const job = jobs.find(j => j.id === bestJob)!;
      robotPositions.set(bestRobot, [job.x, job.y]);
    } else {
      break;
    }
  }

  return assignments;
}

const result = solve(data);
