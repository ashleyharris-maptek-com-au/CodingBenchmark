#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>

using namespace std;

const double EPS = 1e-10;

struct Vec3 {
    double x, y, z;

    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }

    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }

    double len() const { return sqrt(x * x + y * y + z * z); }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge_length;
    cin >> V >> edge_length;

    vector<Vec3> container(V);
    for (int i = 0; i < V; i++) {
        cin >> container[i].x >> container[i].y >> container[i].z;
    }

    // Compute container center
    Vec3 center;
    for (const auto& v : container) {
        center = center + v;
    }
    center = center * (1.0 / V);

    // Compute bounding box
    double minX = container[0].x, maxX = container[0].x;
    double minY = container[0].y, maxY = container[0].y;
    double minZ = container[0].z, maxZ = container[0].z;

    for (const auto& v : container) {
        minX = min(minX, v.x); maxX = max(maxX, v.x);
        minY = min(minY, v.y); maxY = max(maxY, v.y);
        minZ = min(minZ, v.z); maxZ = max(maxZ, v.z);
    }

    // Regular tetrahedron vertices at origin with edge length 1.0
    // Standard form: vertices at (±1, ±1, ±1) scaled appropriately
    double scale = edge_length / sqrt(8.0 / 3.0);
    vector<Vec3> tet_verts = {
        Vec3(1, 1, 1) * scale,
        Vec3(1, -1, -1) * scale,
        Vec3(-1, 1, -1) * scale,
        Vec3(-1, -1, 1) * scale
    };

    // Lambda to check if point is inside convex polyhedron
    auto pointInContainer = [&](const Vec3& p) -> bool {
        // For each triangle face formed by 3 vertices,
        // check if p is on same side as container center
        for (int i = 0; i < V; i++) {
            for (int j = i + 1; j < V; j++) {
                for (int k = j + 1; k < V; k++) {
                    Vec3 v1 = container[j] - container[i];
                    Vec3 v2 = container[k] - container[i];
                    Vec3 normal = v1.cross(v2);

                    double d_p = normal.dot(p - container[i]);
                    double d_c = normal.dot(center - container[i]);

                    // If signs differ significantly, p is outside
                    if (d_p * d_c < -EPS) {
                        return false;
                    }
                }
            }
        }
        return true;
    };

    vector<Vec3> placed;

    // Circumradius for edge length = 1.0
    double circumradius = edge_length * sqrt(3.0 / 8.0);
    // Minimum distance between tetrahedron centers (conservative estimate)
    double minDist = 1.95 * circumradius;

    // Grid spacing
    double spacing = edge_length;

    // Adaptive grid resolution based on container size
    double containerSize = max({maxX - minX, maxY - minY, maxZ - minZ});
    int maxIterations = (int)(containerSize / spacing) + 2;
    maxIterations = min(maxIterations * maxIterations * maxIterations, 1000000); // Limit iterations
    int iterations = 0;

    // Grid-based placement
    for (double x = minX; x <= maxX && iterations < maxIterations; x += spacing) {
        for (double y = minY; y <= maxY && iterations < maxIterations; y += spacing) {
            for (double z = minZ; z <= maxZ && iterations < maxIterations; z += spacing) {
                iterations++;
                Vec3 pos(x, y, z);

                // Check if all vertices of tetrahedron fit inside container
                bool fits = true;
                for (const auto& v : tet_verts) {
                    if (!pointInContainer(v + pos)) {
                        fits = false;
                        break;
                    }
                }

                if (!fits) continue;

                // Check for overlaps with already placed tetrahedrons
                bool overlaps = false;
                for (const auto& p : placed) {
                    if ((pos - p).len() < minDist) {
                        overlaps = true;
                        break;
                    }
                }

                if (!overlaps) {
                    placed.push_back(pos);
                }
            }
        }
    }

    // Output results
    cout << placed.size() << "\n";
    cout << fixed << setprecision(6);
    for (const auto& p : placed) {
        cout << p.x << " " << p.y << " " << p.z << " 0 0 0\n";
    }

    return 0;
}
