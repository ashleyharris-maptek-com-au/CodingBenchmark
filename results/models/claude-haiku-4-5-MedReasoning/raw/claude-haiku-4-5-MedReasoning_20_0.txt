use std::io::{self, BufRead};
use std::collections::HashMap;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse header
    let header = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = header.split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    // Parse property names
    let prop_names: Vec<String> = lines
        .next()
        .unwrap()
        .unwrap()
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();

    // Parse holes and samples
    let mut samples = Vec::new();

    for _ in 0..num_holes {
        let hole_line = lines.next().unwrap().unwrap();
        let hole_parts: Vec<&str> = hole_line.split_whitespace().collect();

        let hole_id = hole_parts[0].parse::<usize>().unwrap();
        let start_x: f64 = hole_parts[1].parse().unwrap();
        let start_y: f64 = hole_parts[2].parse().unwrap();
        let start_z: f64 = hole_parts[3].parse().unwrap();
        let dir_x: f64 = hole_parts[4].parse().unwrap();
        let dir_y: f64 = hole_parts[5].parse().unwrap();
        let dir_z: f64 = hole_parts[6].parse().unwrap();
        let _length: f64 = hole_parts[7].parse().unwrap();
        let num_samples: usize = hole_parts[8].parse().unwrap();

        // Normalize direction vector
        let dir_len = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let (dir_x, dir_y, dir_z) = if dir_len > 0.0 {
            (dir_x / dir_len, dir_y / dir_len, dir_z / dir_len)
        } else {
            (0.0, 0.0, 1.0)
        };

        // Parse samples
        for sample_idx in 0..num_samples {
            let sample_line = lines.next().unwrap().unwrap();
            let sample_parts: Vec<&str> = sample_line.split_whitespace().collect();

            let depth: f64 = sample_parts[0].parse().unwrap();
            let properties: Vec<f64> = sample_parts[1..]
                .iter()
                .map(|s| s.parse().unwrap())
                .collect();

            // Calculate 3D position
            let x = start_x + dir_x * depth;
            let y = start_y + dir_y * depth;
            let z = start_z + dir_z * depth;

            samples.push((hole_id, sample_idx, x, y, z, properties));
        }
    }

    // Adaptive parameter selection based on dataset size
    let sample_count = samples.len();
    let (grid_size, search_radius, z_threshold) = if sample_count < 5000 {
        (30.0, 40.0, 2.0)
    } else if sample_count < 50000 {
        (50.0, 50.0, 2.5)
    } else {
        (100.0, 100.0, 3.0)
    };

    // Detect outliers using grid-based spatial indexing
    let mut outliers = Vec::new();
    let search_radius_sq = search_radius * search_radius;

    // Build spatial grid for efficient neighbor finding
    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();

    for (i, (_, _, x, y, z, _)) in samples.iter().enumerate() {
        let grid_x = (x / grid_size).floor() as i32;
        let grid_y = (y / grid_size).floor() as i32;
        let grid_z = (z / grid_size).floor() as i32;

        grid.entry((grid_x, grid_y, grid_z))
            .or_insert_with(Vec::new)
            .push(i);
    }

    for i in 0..samples.len() {
        let (hole_id, sample_idx, x, y, z, props) = &samples[i];

        // Find neighbors using grid lookup
        let grid_x = (x / grid_size).floor() as i32;
        let grid_y = (y / grid_size).floor() as i32;
        let grid_z = (z / grid_size).floor() as i32;

        let mut neighbors = Vec::new();

        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    if let Some(cell) = grid.get(&(grid_x + dx, grid_y + dy, grid_z + dz)) {
                        for &j in cell {
                            if i == j {
                                continue;
                            }
                            let (_, _, nx, ny, nz, _) = &samples[j];
                            let dist_sq = (x - nx).powi(2) + (y - ny).powi(2) + (z - nz).powi(2);
                            if dist_sq <= search_radius_sq {
                                neighbors.push(j);
                            }
                        }
                    }
                }
            }
        }

        if neighbors.len() < 2 {
            continue;
        }

        // Check each property for outliers
        for prop_idx in 0..props.len() {
            let value = props[prop_idx];

            // Collect neighbor values for this property
            let neighbor_values: Vec<f64> = neighbors
                .iter()
                .map(|&j| samples[j].5[prop_idx])
                .collect();

            // Calculate mean and standard deviation
            let mean = neighbor_values.iter().sum::<f64>() / neighbor_values.len() as f64;
            let variance = neighbor_values
                .iter()
                .map(|v| (v - mean).powi(2))
                .sum::<f64>()
                / neighbor_values.len() as f64;
            let std_dev = variance.sqrt();

            // Calculate z-score
            if std_dev > 1e-10 {
                let z_score = (value - mean).abs() / std_dev;
                if z_score > z_threshold {
                    let confidence = (z_score / 10.0).min(1.0);
                    outliers.push((
                        *hole_id,
                        *sample_idx,
                        prop_names[prop_idx].clone(),
                        confidence,
                        z_score,
                    ));
                }
            }
        }
    }

    // Sort by confidence (descending), then by z_score as tiebreaker
    outliers.sort_by(|a, b| {
        b.3.partial_cmp(&a.3)
            .unwrap_or(std::cmp::Ordering::Equal)
            .then_with(|| b.4.partial_cmp(&a.4).unwrap_or(std::cmp::Ordering::Equal))
    });

    // Output results
    println!("{}", outliers.len());
    for (hole_id, sample_idx, prop_name, confidence, _) in outliers {
        println!("{} {} {} {:.4}", hole_id, sample_idx, prop_name, confidence);
    }
}
