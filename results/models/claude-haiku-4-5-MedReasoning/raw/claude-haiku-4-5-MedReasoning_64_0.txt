declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Drone {
  id: number;
  x: number;
  y: number;
}

interface Point {
  x: number;
  y: number;
}

function isValid(x: number, y: number, rows: number, cols: number, grid: number[][]): boolean {
  return x >= 0 && x < rows && y >= 0 && y < cols && grid[x][y] !== -1;
}

function getManhattanDistance(p1: Point, p2: Point): number {
  return Math.abs(p1.x - p2.x) + Math.abs(p1.y - p2.y);
}

function findPath(start: Point, target: Point, grid: number[][], maxSteps: number): Point[] | null {
  if (start.x === target.x && start.y === target.y) return [start];

  const rows = grid.length;
  const cols = grid[0].length;
  const heap: { cost: number; steps: number; pos: Point; path: Point[] }[] = [
    { cost: getManhattanDistance(start, target), steps: 0, pos: start, path: [start] }
  ];
  const visited = new Map<string, number>();
  visited.set(`${start.x},${start.y}`, 0);

  while (heap.length > 0) {
    heap.sort((a, b) => a.cost - b.cost);
    const { pos, path, steps } = heap.shift()!;

    if (pos.x === target.x && pos.y === target.y) return path;
    if (steps >= maxSteps) continue;

    const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]];
    for (const [dx, dy] of directions) {
      const nx = pos.x + dx;
      const ny = pos.y + dy;
      const key = `${nx},${ny}`;
      const newSteps = steps + 1;

      if (isValid(nx, ny, rows, cols, grid)) {
        const visited_steps = visited.get(key);
        if (visited_steps === undefined || visited_steps > newSteps) {
          visited.set(key, newSteps);
          const newPos = { x: nx, y: ny };
          const heuristic = getManhattanDistance(newPos, target);
          heap.push({
            cost: newSteps + heuristic,
            steps: newSteps,
            pos: newPos,
            path: [...path, newPos]
          });
        }
      }
    }
  }

  return null;
}

function planDronePaths(
  probability: number[][],
  drones: Drone[],
  maxStepsPerDrone: number
): Point[][] {
  const rows = probability.length;
  const cols = probability[0].length;
  const paths: Point[][] = [];
  const covered = new Set<string>();

  drones.sort((a, b) => a.id - b.id);

  for (const drone of drones) {
    const path: Point[] = [{ x: drone.x, y: drone.y }];
    let current = { x: drone.x, y: drone.y };
    let stepsRemaining = maxStepsPerDrone;
    const localCovered = new Set<string>();

    while (stepsRemaining > 0) {
      let bestScore = -Infinity;
      let bestTarget: Point | null = null;
      let bestDist = Infinity;

      const maxSearchDistance = Math.min(stepsRemaining, Math.ceil(Math.sqrt(rows * cols)));

      for (let i = 0; i < rows; i++) {
        for (let j = 0; j < cols; j++) {
          if (probability[i][j] > 0) {
            const key = `${i},${j}`;
            const dist = getManhattanDistance(current, { x: i, y: j });

            if (dist > 0 && dist <= Math.min(stepsRemaining, maxSearchDistance)) {
              const isCovered = covered.has(key);
              const isLocalCovered = localCovered.has(key);
              const value = probability[i][j];
              const score = (!isCovered && !isLocalCovered ? value : value * 0.1) - dist * 0.05;

              if (score > bestScore || (score === bestScore && dist < bestDist)) {
                bestScore = score;
                bestTarget = { x: i, y: j };
                bestDist = dist;
              }
            }
          }
        }
      }

      if (!bestTarget) break;

      const pathToTarget = findPath(current, bestTarget, probability, stepsRemaining);
      if (!pathToTarget || pathToTarget.length === 1) break;

      for (let k = 1; k < pathToTarget.length; k++) {
        const p = pathToTarget[k];
        path.push(p);
        const key = `${p.x},${p.y}`;
        covered.add(key);
        localCovered.add(key);
      }

      current = bestTarget;
      stepsRemaining -= pathToTarget.length - 1;
    }

    paths.push(path);
  }

  return paths;
}

const probability = data.probability;
const drones: Drone[] = data.drones;
const maxStepsPerDrone = data.maxStepsPerDrone;

const paths = planDronePaths(probability, drones, maxStepsPerDrone);

const result = {
  paths: paths.map(path => path.map(p => [p.x, p.y]))
};
