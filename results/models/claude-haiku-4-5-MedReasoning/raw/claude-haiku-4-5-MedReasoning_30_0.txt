#!/usr/bin/env python3
import sys
import random

def read_input():
    """Read SAT instance from stdin"""
    line = sys.stdin.readline().strip()
    if not line:
        return None, []

    num_vars, num_clauses = map(int, line.split())
    clauses = []

    for _ in range(num_clauses):
        line = sys.stdin.readline().strip()
        if line:
            clause = tuple(map(int, line.split()))
            clauses.append(clause)

    return num_vars, clauses

def evaluate_clause(clause, assignment):
    """Check if a clause is satisfied by the assignment"""
    for lit in clause:
        var_idx = abs(lit) - 1
        var_val = assignment[var_idx]

        if (lit > 0 and var_val == 1) or (lit < 0 and var_val == 0):
            return True

    return False

def find_unsatisfied_clauses(clauses, assignment):
    """Return list of unsatisfied clause indices"""
    unsatisfied = []
    for i, clause in enumerate(clauses):
        if not evaluate_clause(clause, assignment):
            unsatisfied.append(i)
    return unsatisfied

def solve_sat(num_vars, clauses):
    """Solve SAT using WalkSAT algorithm"""
    max_iterations = 10000000
    max_tries = 20

    for attempt in range(max_tries):
        assignment = [random.randint(0, 1) for _ in range(num_vars)]

        for iteration in range(max_iterations):
            unsatisfied = find_unsatisfied_clauses(clauses, assignment)

            if not unsatisfied:
                return assignment

            clause = clauses[random.choice(unsatisfied)]
            lit = random.choice(clause)
            var_idx = abs(lit) - 1
            assignment[var_idx] = 1 - assignment[var_idx]

    return assignment

def main():
    random.seed(42)
    num_vars, clauses = read_input()

    if num_vars is None:
        sys.exit(1)

    assignment = solve_sat(num_vars, clauses)

    print("SAT")
    print(" ".join(map(str, assignment)))

if __name__ == "__main__":
    main()
