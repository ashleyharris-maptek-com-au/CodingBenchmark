OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_color %out_color
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%input_type = OpTypePointer Input %v3float
%output_type = OpTypePointer Output %v4float

%in_color = OpVariable %input_type Input
%out_color = OpVariable %output_type Output

%main_type = OpTypeFunction %void
%main = OpFunction %void None %main_type
%entry = OpLabel

%color_vec3 = OpLoad %v3float %in_color
%one = OpConstant %float 1.0

%r = OpCompositeExtract %float %color_vec3 0
%g = OpCompositeExtract %float %color_vec3 1
%b = OpCompositeExtract %float %color_vec3 2

%result = OpCompositeConstruct %v4float %r %g %b %one

OpStore %out_color %result

OpReturn
OpFunctionEnd
