using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    struct Placement
    {
        public int BoxIndex;
        public int X, Y, Z;
        public int W, H, D;
    }

    static void Main()
    {
        var line = Console.ReadLine().Split(' ');
        int n = int.Parse(line[0]);
        int W = int.Parse(line[1]);
        int H = int.Parse(line[2]);
        int D = int.Parse(line[3]);

        var boxes = new (int w, int h, int d)[n];
        for (int i = 0; i < n; i++)
        {
            var dims = Console.ReadLine().Split(' ');
            boxes[i] = (int.Parse(dims[0]), int.Parse(dims[1]), int.Parse(dims[2]));
        }

        var placements = PackBoxes(boxes, n, W, H, D);

        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.BoxIndex} {p.X} {p.Y} {p.Z}");
        }
    }

    static List<Placement> PackBoxes((int w, int h, int d)[] boxes, int n, int W, int H, int D)
    {
        var indices = Enumerable.Range(0, n).ToArray();
        Array.Sort(indices, (i, j) =>
            ((long)boxes[j].w * boxes[j].h * boxes[j].d).CompareTo((long)boxes[i].w * boxes[i].h * boxes[i].d));

        var placements = new List<Placement>();
        var spatial = new SpatialIndex(W, H, D);

        foreach (var i in indices)
        {
            var (w, h, d) = boxes[i];
            var orientations = GetOrientations(w, h, d);
            bool placed = false;

            foreach (var (ow, oh, od) in orientations)
            {
                if (placed) break;

                var pos = FindBestPosition(ow, oh, od, W, H, D, spatial, placements);

                if (pos.HasValue)
                {
                    var (x, y, z) = pos.Value;
                    var p = new Placement { BoxIndex = i, X = x, Y = y, Z = z, W = ow, H = oh, D = od };
                    placements.Add(p);
                    spatial.Add(p);
                    placed = true;
                }
            }
        }

        return placements;
    }

    class SpatialIndex
    {
        private int W, H, D;
        private Dictionary<long, List<Placement>> cells = new Dictionary<long, List<Placement>>();
        private const int CELL = 50;

        public SpatialIndex(int w, int h, int d)
        {
            W = w;
            H = h;
            D = d;
        }

        public void Add(Placement p)
        {
            int x1 = p.X / CELL;
            int y1 = p.Y / CELL;
            int z1 = p.Z / CELL;
            int x2 = (p.X + p.W - 1) / CELL;
            int y2 = (p.Y + p.H - 1) / CELL;
            int z2 = (p.Z + p.D - 1) / CELL;

            for (int x = x1; x <= x2; x++)
                for (int y = y1; y <= y2; y++)
                    for (int z = z1; z <= z2; z++)
                    {
                        long key = ((long)x << 40) | ((long)y << 20) | (long)z;
                        if (!cells.ContainsKey(key))
                            cells[key] = new List<Placement>();
                        cells[key].Add(p);
                    }
        }

        public bool CanPlace(int x, int y, int z, int w, int h, int d)
        {
            int x1 = x / CELL;
            int y1 = y / CELL;
            int z1 = z / CELL;
            int x2 = (x + w - 1) / CELL;
            int y2 = (y + h - 1) / CELL;
            int z2 = (z + d - 1) / CELL;

            var checked_set = new HashSet<int>();
            for (int xi = x1; xi <= x2; xi++)
                for (int yi = y1; yi <= y2; yi++)
                    for (int zi = z1; zi <= z2; zi++)
                    {
                        long key = ((long)xi << 40) | ((long)yi << 20) | (long)zi;
                        if (cells.ContainsKey(key))
                        {
                            foreach (var p in cells[key])
                            {
                                if (checked_set.Add(p.BoxIndex))
                                {
                                    if (Overlaps(x, y, z, w, h, d, p.X, p.Y, p.Z, p.W, p.H, p.D))
                                        return false;
                                }
                            }
                        }
                    }
            return true;
        }
    }

    static (int, int, int)? FindBestPosition(int w, int h, int d, int W, int H, int D, SpatialIndex spatial, List<Placement> placements)
    {
        var candidates = new List<(int, int, int)>();
        candidates.Add((0, 0, 0));

        int limit = Math.Min(placements.Count, 200);
        var seen = new HashSet<(int, int, int)>();

        for (int i = Math.Max(0, placements.Count - limit); i < placements.Count; i++)
        {
            var box = placements[i];
            AddIfValid((box.X + box.W, box.Y, box.Z), W, H, D, w, h, d, seen, candidates);
            AddIfValid((box.X, box.Y + box.H, box.Z), W, H, D, w, h, d, seen, candidates);
            AddIfValid((box.X, box.Y, box.Z + box.D), W, H, D, w, h, d, seen, candidates);
        }

        foreach (var (x, y, z) in candidates)
        {
            if (spatial.CanPlace(x, y, z, w, h, d))
                return (x, y, z);
        }

        return null;
    }

    static void AddIfValid(int x, int y, int z, int W, int H, int D, int w, int h, int d, HashSet<(int, int, int)> seen, List<(int, int, int)> list)
    {
        if (x >= 0 && y >= 0 && z >= 0 && x + w <= W && y + h <= H && z + d <= D && seen.Add((x, y, z)))
            list.Add((x, y, z));
    }

    static bool Overlaps(int x1, int y1, int z1, int w1, int h1, int d1,
                         int x2, int y2, int z2, int w2, int h2, int d2)
    {
        return !(x1 + w1 <= x2 || x2 + w2 <= x1 ||
                 y1 + h1 <= y2 || y2 + h2 <= y1 ||
                 z1 + d1 <= z2 || z2 + d2 <= z1);
    }

    static List<(int, int, int)> GetOrientations(int w, int h, int d)
    {
        var orientations = new HashSet<(int, int, int)>()
        {
            (w, h, d),
            (w, d, h),
            (h, w, d),
            (h, d, w),
            (d, w, h),
            (d, h, w)
        };
        return orientations.ToList();
    }
}
