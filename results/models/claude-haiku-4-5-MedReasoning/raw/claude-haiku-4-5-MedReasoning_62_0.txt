declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Town {
  id: number;
  x: number;
  y: number;
  value: number;
}

const { width, height, grid, ignitions, towns, cutBudget }: {
  width: number;
  height: number;
  grid: string[];
  ignitions: [number, number][];
  towns: Town[];
  cutBudget: number;
} = data;

function simulateFire(firebreaks: Set<string>): Set<string> {
  const burned = new Set<string>();
  const queue: [number, number][] = [];

  // Initialize with ignition points
  for (const [x, y] of ignitions) {
    const key = `${x},${y}`;
    if (grid[y][x] === '.' && !firebreaks.has(key)) {
      queue.push([x, y]);
      burned.add(key);
    }
  }

  // BFS to spread fire in 4 directions
  while (queue.length > 0) {
    const [x, y] = queue.shift()!;

    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
        const key = `${nx},${ny}`;
        if (!burned.has(key) && !firebreaks.has(key) && grid[ny][nx] === '.') {
          burned.add(key);
          queue.push([nx, ny]);
        }
      }
    }
  }

  return burned;
}

function calculateSavedValue(burned: Set<string>): number {
  let value = 0;
  for (const town of towns) {
    const key = `${town.x},${town.y}`;
    if (!burned.has(key)) {
      value += town.value;
    }
  }
  return value;
}

function getCandidateCells(burned: Set<string>): Set<string> {
  const candidates = new Set<string>();

  // Include cells adjacent to burned areas (fire front)
  for (const key of burned) {
    const [x, y] = key.split(',').map(Number);
    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;
      if (nx >= 0 && nx < width && ny >= 0 && ny < height && grid[ny][nx] === '.') {
        candidates.add(`${nx},${ny}`);
      }
    }
  }

  // Include cells adjacent to unburned towns (protect them)
  for (const town of towns) {
    const key = `${town.x},${town.y}`;
    if (!burned.has(key)) {
      for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
        const nx = town.x + dx;
        const ny = town.y + dy;
        if (nx >= 0 && nx < width && ny >= 0 && ny < height && grid[ny][nx] === '.') {
          candidates.add(`${nx},${ny}`);
        }
      }
    }
  }

  return candidates;
}

const firebreaks = new Set<string>();

for (let iteration = 0; iteration < cutBudget; iteration++) {
  const currentBurned = simulateFire(firebreaks);
  const currentValue = calculateSavedValue(currentBurned);

  let bestPos: [number, number] | null = null;
  let bestImprovement = 0;

  // Get candidate cells to try (adjacent to fire front and unburned towns)
  const candidates = getCandidateCells(currentBurned);

  for (const key of candidates) {
    if (!firebreaks.has(key)) {
      const [x, y] = key.split(',').map(Number);

      firebreaks.add(key);
      const newBurned = simulateFire(firebreaks);
      const newValue = calculateSavedValue(newBurned);
      const improvement = newValue - currentValue;

      if (improvement > bestImprovement) {
        bestImprovement = improvement;
        bestPos = [x, y];
      }

      firebreaks.delete(key);
    }
  }

  if (bestPos === null) {
    break; // No improvement possible
  }

  firebreaks.add(`${bestPos[0]},${bestPos[1]}`);
}

// Output result
const result = Array.from(firebreaks).map(key => {
  const [x, y] = key.split(',').map(Number);
  return [x, y];
});
