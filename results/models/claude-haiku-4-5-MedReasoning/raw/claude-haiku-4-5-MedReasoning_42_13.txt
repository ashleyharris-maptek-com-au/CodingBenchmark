       OpStore %c4 %c4_load

       %col_c2_scale = OpVectorTimesScalar %v3float %c2_e1 %float_0_5

       %c0_e3 = OpLoad %v3float %c0

       %col_c0_scale2 = OpVectorTimesScalar %v3float %c0_e3 %float_0_5
