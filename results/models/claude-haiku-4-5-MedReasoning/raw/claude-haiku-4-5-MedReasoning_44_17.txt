#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Brick pattern parameters
    float scaleX = 8.0;
    float scaleY = 16.0;
    float mortarX = 0.05;
    float mortarY = 0.1;

    // Scale the UV coordinates
    vec2 scaledUv = uv * vec2(scaleX, scaleY);

    // Apply staggered offset for every other row
    float row = floor(scaledUv.y);
    vec2 brickUv = scaledUv;
    if (mod(row, 2.0) > 0.5) {
        brickUv.x += 0.5;
    }

    // Get the position within the brick
    vec2 brickLocal = fract(brickUv);

    // Determine if we're in mortar
    bool inMortarX = brickLocal.x < mortarX || brickLocal.x > (1.0 - mortarX);
    bool inMortarY = brickLocal.y < mortarY || brickLocal.y > (1.0 - mortarY);
    bool inMortar = inMortarX || inMortarY;

    // Choose color based on mortar or brick
    vec3 baseColor = inMortar ? vec3(0.7, 0.7, 0.7) : vec3(0.7, 0.2, 0.1);

    // Lambertian shading
    vec3 norm = normalize(normal);
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(norm, lightDir), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;

    // Apply shading
    vec3 finalColor = baseColor * shadingFactor;

    outColor = vec4(finalColor, 1.0);
}
