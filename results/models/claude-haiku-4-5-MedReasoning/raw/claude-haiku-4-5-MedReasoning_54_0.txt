def autopilot_step(state: dict, dt: float) -> dict:
    global _phase, _go_around_count, _pitch_integral, _roll_integral, _throttle_integral

    # Initialize state on first call
    if '_phase' not in globals():
        globals()['_phase'] = 'INTERCEPT'
        globals()['_go_around_count'] = 0
        globals()['_pitch_integral'] = 0
        globals()['_roll_integral'] = 0
        globals()['_throttle_integral'] = 0

    _phase = globals()['_phase']
    _go_around_count = globals()['_go_around_count']

    # Get fused sensor values
    radio_alt = state.get('radio_alt_ft', 0)
    ias = get_fused_ias(state)
    glideslope_dots = get_fused_glideslope(state)
    localizer_dots = get_fused_localizer(state)
    target_ias = state.get('target_ias_kt', 137)
    on_ground = state.get('weight_on_wheels', False)

    # Phase transitions
    if _phase == 'INTERCEPT' and radio_alt < 1000:
        _phase = 'APPROACH'
    elif _phase == 'APPROACH' and radio_alt < 50:
        _phase = 'FLARE'
    elif _phase == 'FLARE' and on_ground:
        _phase = 'TOUCHDOWN'

    # Check for go-around condition
    if should_go_around(state, _phase, ias, target_ias, radio_alt, glideslope_dots):
        _go_around_count += 1
        globals()['_go_around_count'] = _go_around_count
        globals()['_phase'] = _phase

        if _go_around_count >= 10:
            return {}  # Scenario failed

        # Execute go-around: TOGA thrust, pitch up, retract speedbrake
        globals()['_pitch_integral'] = 0
        globals()['_roll_integral'] = 0
        globals()['_throttle_integral'] = 0
        return {
            'throttle_1': 1.0,
            'throttle_2': 1.0,
            'elevator': 0.3,
            'speedbrake': 0.0,
            'flaps': 15,
            'gear': 1.0
        }

    # Reset go-around count on successful landing
    if _phase == 'TOUCHDOWN' and state.get('groundspeed_kt', 0) < 50:
        _go_around_count = 0

    # Generate control commands based on phase
    cmd = {}

    if _phase == 'INTERCEPT':
        cmd = control_intercept(state, glideslope_dots, localizer_dots, ias, target_ias, radio_alt)
    elif _phase == 'APPROACH':
        cmd = control_approach(state, glideslope_dots, localizer_dots, ias, target_ias, radio_alt)
    elif _phase == 'FLARE':
        cmd = control_flare(state, radio_alt, ias, localizer_dots)
    elif _phase == 'TOUCHDOWN':
        cmd = control_touchdown(state)

    globals()['_phase'] = _phase
    globals()['_go_around_count'] = _go_around_count
    globals()['_pitch_integral'] = globals()['_pitch_integral']
    globals()['_roll_integral'] = globals()['_roll_integral']
    globals()['_throttle_integral'] = globals()['_throttle_integral']

    return cmd

def get_fused_ias(state):
    """Sensor fusion for airspeed: vote from 3 sources using valid flags"""
    candidates = []
    for i in [1, 2, 3]:
        if state.get(f'ias_{i}_valid', False):
            candidates.append(state.get(f'ias_{i}_kt', 0))

    if len(candidates) >= 2:
        # Use median of valid sources
        return sorted(candidates)[len(candidates) // 2]
    elif candidates:
        return candidates[0]
    else:
        return state.get('target_ias_kt', 137)

def get_fused_glideslope(state):
    """Sensor fusion for glideslope: average valid sources"""
    candidates = []
    for i in [1, 2, 3]:
        if state.get(f'glideslope_{i}_valid', False):
            candidates.append(state.get(f'glideslope_{i}_dots', 0))

    if candidates:
        return sum(candidates) / len(candidates)
    return 0

def get_fused_localizer(state):
    """Sensor fusion for localizer: average valid sources"""
    candidates = []
    for i in [1, 2, 3]:
        if state.get(f'localizer_{i}_valid', False):
            candidates.append(state.get(f'localizer_{i}_dots', 0))

    if candidates:
        return sum(candidates) / len(candidates)
    return 0

def should_go_around(state, phase, ias, target_ias, radio_alt, glideslope_dots):
    """Detect unstable approach requiring go-around"""

    # Never go around if already touching down
    if state.get('weight_on_wheels', False):
        return False

    # Too fast at any point
    if ias > target_ias + 25:
        return True

    # Glideslope unstable below 500ft
    if radio_alt < 500 and abs(glideslope_dots) > 2.5:
        return True

    # Too fast below decision height (200ft)
    if radio_alt < 200 and ias > target_ias + 15:
        return True

    # Too fast in flare
    if radio_alt < 50 and ias > target_ias + 20:
        return True

    # Excessive descent rate below 500ft (conservative check via glideslope)
    if radio_alt < 500 and glideslope_dots < -3:
        return True

    return False

def control_intercept(state, glideslope_dots, localizer_dots, ias, target_ias, radio_alt):
    """Intercept phase: >1000ft, track ILS, configure aircraft"""

    # Pitch control: track 3° glideslope (300 dots full scale)
    # Negative glideslope_dots means below glideslope (need pitch up)
    gs_error = -glideslope_dots / 300.0 * 3.0  # Convert to degrees
    pitch_cmd = max(-0.5, min(0.5, gs_error * 0.15))

    # Roll control: track localizer (300 dots full scale)
    # Positive localizer_dots means right of center (need left roll)
    loc_error = -localizer_dots / 300.0 * 3.0  # Convert to degrees
    roll_cmd = max(-1.0, min(1.0, loc_error * 0.1))

    # Throttle: maintain approach speed
    speed_error = ias - target_ias
    throttle = 0.5
    if speed_error > 10:
        throttle = 0.3
    elif speed_error < -10:
        throttle = 0.7

    # Gradual flap extension (5° at 2000ft, 15° at 1000ft)
    if radio_alt > 2000:
        flaps = 5
    elif radio_alt > 1000:
        flaps = 10
    else:
        flaps = 15

    return {
        'elevator': pitch_cmd,
        'aileron': roll_cmd,
        'rudder': 0.0,
        'throttle_1': throttle,
        'throttle_2': throttle,
        'flaps': flaps,
        'gear': 1.0,
        'speedbrake': 0.0
    }

def control_approach(state, glideslope_dots, localizer_dots, ias, target_ias, radio_alt):
    """Approach phase: 1000ft-200ft, stabilize on ILS"""

    # More aggressive glideslope tracking
    gs_error = -glideslope_dots / 300.0 * 3.0
    pitch_cmd = max(-0.6, min(0.6, gs_error * 0.2))

    # Localizer tracking
    loc_error = -localizer_dots / 300.0 * 3.0
    roll_cmd = max(-1.0, min(1.0, loc_error * 0.12))

    # Reduce speed toward landing speed
    speed_error = ias - target_ias
    if speed_error > 10:
        throttle = 0.2
    elif speed_error < -10:
        throttle = 0.5
    else:
        throttle = 0.35

    # Full flap extension (25-30 degrees)
    flaps = 30

    # Deploy minimal speedbrake for descent control
    speedbrake = 0.1 if radio_alt > 500 else 0.05

    return {
        'elevator': pitch_cmd,
        'aileron': roll_cmd,
        'rudder': 0.0,
        'throttle_1': throttle,
        'throttle_2': throttle,
        'flaps': flaps,
        'gear': 1.0,
        'speedbrake': speedbrake
    }

def control_flare(state, radio_alt, ias, localizer_dots):
    """Flare phase: <50ft, reduce descent rate, transition to touchdown"""

    # Progressive pitch up as we descend
    # At 50ft: slight pitch, at 10ft: more pitch
    flare_gradient = 1.0 - max(0, (radio_alt - 10) / 40)  # 0 at 50ft, 1 at 10ft
    pitch_cmd = 0.2 + flare_gradient * 0.5

    # Maintain localizer
    loc_error = -localizer_dots / 300.0 * 3.0
    roll_cmd = max(-0.5, min(0.5, loc_error * 0.15))

    # Idle thrust in flare
    throttle = max(0.0, 0.1 - flare_gradient * 0.1)

    return {
        'elevator': pitch_cmd,
        'aileron': roll_cmd,
        'rudder': 0.0,
        'throttle_1': throttle,
        'throttle_2': throttle,
        'flaps': 40,
        'gear': 1.0,
        'speedbrake': 0.0
    }

def control_touchdown(state):
    """Touchdown phase: deploy spoilers, reverse thrust, autobrake"""

    gs = state.get('groundspeed_kt', 0)

    # Progressive reverse thrust deployment
    reverse_factor = min(1.0, max(0.0, (gs - 50) / 100))
    reverse = 0.3 + reverse_factor * 0.7

    # Autobrake proportional to speed
    autobrake = min(1.0, max(0.0, (gs - 30) / 100))

    return {
        'elevator': -0.1,  # Slight pitch down to maintain weight on wheels
        'aileron': 0.0,
        'rudder': 0.0,
        'throttle_1': 0.0,
        'throttle_2': 0.0,
        'flaps': 40,
        'gear': 1.0,
        'spoiler': 1.0,
        'reverse_1': reverse,
        'reverse_2': reverse,
        'autobrake': autobrake
    }
