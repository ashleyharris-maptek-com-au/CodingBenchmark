OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %out_color
OpExecutionMode %main OriginUpperLeft

%glsl = OpExtInstImport "GLSL.std.450"

; Decorations for uniform block
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16

; Decorations for I/O
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

; Type definitions
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec4 = OpTypePointer Input %vec4
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Function_vec2 = OpTypePointer Function %vec2
%ptr_Function_vec3 = OpTypePointer Function %vec3
%ptr_Function_float = OpTypePointer Function %float

%func_type = OpTypeFunction %void

; Global variables
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec4 Input
%out_color = OpVariable %ptr_Output_vec4 Output

; Constants
%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_6 = OpConstant %float 6.0
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3

; Composite constants
%yellow = OpConstantComposite %vec3 %float_1 %float_0_8 %float_0
%darkblue = OpConstantComposite %vec3 %float_0_1 %float_0_1 %float_0_3
%center = OpConstantComposite %vec2 %float_0_5 %float_0_5

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load inputs
%uv = OpLoad %vec2 %in_uv
%normal = OpLoad %vec3 %in_normal
%worldPos = OpLoad %vec3 %in_worldPos

; Normalize normal vector
%normal_norm = OpExtInst %vec3 %glsl Normalize %normal

; Scale UV coordinates to 6x6 grid
%uv_scaled = OpVectorTimesScalar %vec2 %uv %float_6

; Compute local coordinates within cell and cell index
%local_uv = OpExtInst %vec2 %glsl Fract %uv_scaled

; Compute distance from cell center (0.5, 0.5)
%diff = OpFSub %vec2 %local_uv %center
%dist = OpExtInst %float %glsl Length %diff

; Check if distance < 0.3 (inside dot)
%is_dot = OpFOrdLessThan %bool %dist %float_0_3

; Select color: yellow if in dot, dark blue if background
%base_color = OpSelect %vec3 %is_dot %yellow %darkblue

; Access light position from uniform buffer (offset 3, which is vec4 at 192 bytes)
%light_addr = OpAccessChain %ptr_Uniform_vec4 %ubo %int_3
%light_pos4 = OpLoad %vec4 %light_addr

; Extract xyz from light position
%light_pos = OpVectorShuffle %vec3 %light_pos4 %light_pos4 0 1 2

; Compute light direction (from fragment to light)
%to_light = OpFSub %vec3 %light_pos %worldPos
%to_light_norm = OpExtInst %vec3 %glsl Normalize %to_light

; Compute Lambertian diffuse: dot product with light direction
%dot_product = OpDot %float %normal_norm %to_light_norm

; Clamp to [0, 1]
%diffuse = OpExtInst %float %glsl FMax %dot_product %float_0

; Compute shading factor: 0.3 + 0.7 * diffuse
%scaled_diffuse = OpFMul %float %float_0_7 %diffuse
%shading_factor = OpFAdd %float %float_0_3 %scaled_diffuse

; Apply shading to base color
%shaded_color = OpVectorTimesScalar %vec3 %base_color %shading_factor

; Construct final output: (RGB shaded color, alpha = 1.0)
%output_color = OpCompositeConstruct %vec4 %shaded_color %float_1

; Store to output
OpStore %out_color %output_color

OpReturn
OpFunctionEnd
