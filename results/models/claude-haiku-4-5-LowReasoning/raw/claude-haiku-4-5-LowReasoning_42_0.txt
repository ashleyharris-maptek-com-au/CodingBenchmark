OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_0 = OpConstant %float 0.0
%float_0_04 = OpConstant %float 0.04
%float_1 = OpConstant %float 1.0

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3 = OpTypeArray %v3float %uint_1

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%func_type = OpTypeFunction %void

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func_type
%entry = OpLabel

%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%col_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%col = OpLoad %v3float %col_ptr

%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1

%x0 = OpFSub %float %x %float_0_04
%y0 = OpFSub %float %y %float_0_04
%v0 = OpCompositeConstruct %v4float %x0 %y0 %float_0 %float_1
%pos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos0_ptr %v0
OpStore %outColor %col
OpEmitVertex

%x1 = OpFAdd %float %x %float_0_04
%y1 = OpFSub %float %y %float_0_04
%v1 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
%pos1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos1_ptr %v1
OpStore %outColor %col
OpEmitVertex

%x2 = OpFSub %float %x %float_0_04
%y2 = OpFAdd %float %y %float_0_04
%v2 = OpCompositeConstruct %v4float %x2 %y2 %float_0 %float_1
%pos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos2_ptr %v2
OpStore %outColor %col
OpEmitVertex

%x3 = OpFAdd %float %x %float_0_04
%y3 = OpFAdd %float %y %float_0_04
%v3 = OpCompositeConstruct %v4float %x3 %y3 %float_0 %float_1
%pos3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos3_ptr %v3
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
