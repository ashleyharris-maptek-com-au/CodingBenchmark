use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_vertices: usize = parts[0].parse().unwrap();
    let num_edges: usize = parts[1].parse().unwrap();

    // Build adjacency list
    let mut adj: Vec<HashSet<usize>> = vec![HashSet::new(); num_vertices];

    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u: usize = parts[0].parse().unwrap();
        let v: usize = parts[1].parse().unwrap();
        adj[u].insert(v);
        adj[v].insert(u);
    }

    // Greedy dominating set algorithm
    let mut dominating_set = HashSet::new();
    let mut covered = HashSet::new();

    while covered.len() < num_vertices {
        // Find vertex not yet in dominating set that covers most uncovered vertices
        let mut best_vertex = 0;
        let mut best_count = 0;

        for v in 0..num_vertices {
            if dominating_set.contains(&v) {
                continue;
            }

            let mut count = 0;
            if !covered.contains(&v) {
                count += 1; // The vertex itself
            }
            for neighbor in &adj[v] {
                if !covered.contains(neighbor) {
                    count += 1;
                }
            }

            if count > best_count {
                best_count = count;
                best_vertex = v;
            }
        }

        // Add best vertex to dominating set
        dominating_set.insert(best_vertex);
        covered.insert(best_vertex);
        for neighbor in &adj[best_vertex] {
            covered.insert(*neighbor);
        }
    }

    // Output
    let mut result: Vec<usize> = dominating_set.iter().cloned().collect();
    result.sort();

    println!("{}", result.len());
    if !result.is_empty() {
        let output = result.iter()
            .map(|x| x.to_string())
            .collect::<Vec<_>>()
            .join(" ");
        println!("{}", output);
    }
}
