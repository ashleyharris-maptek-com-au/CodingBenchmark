declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Town {
  id: string;
  x: number;
  y: number;
  value: number;
}

interface InputData {
  width: number;
  height: number;
  grid: string[];
  ignitions: [number, number][];
  towns: Town[];
  cutBudget: number;
}

function simulateFireSpread(
  grid: string[],
  ignitions: [number, number][],
  firebreaks: Set<string>,
  width: number,
  height: number
): Set<string> {
  const visited = new Set<string>();
  const queue: [number, number][] = [];

  for (const [x, y] of ignitions) {
    const key = `${x},${y}`;
    if (!firebreaks.has(key)) {
      queue.push([x, y]);
      visited.add(key);
    }
  }

  while (queue.length > 0) {
    const [x, y] = queue.shift()!;

    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;
      const key = `${nx},${ny}`;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height && !visited.has(key) && !firebreaks.has(key)) {
        if (grid[ny][nx] === '.') {
          visited.add(key);
          queue.push([nx, ny]);
        }
      }
    }
  }

  return visited;
}

function solve(data: InputData): [number, number][] {
  const { width, height, grid, ignitions, towns, cutBudget } = data;

  const firebreaks = new Set<string>();

  for (let iteration = 0; iteration < cutBudget; iteration++) {
    const burned = simulateFireSpread(grid, ignitions, firebreaks, width, height);

    let currentSaved = 0;
    for (const town of towns) {
      if (!burned.has(`${town.x},${town.y}`)) {
        currentSaved += town.value;
      }
    }

    let bestCell: string | null = null;
    let bestImprovement = 0;

    const candidates = new Set<string>();
    for (const key of burned) {
      const [x, y] = key.split(',').map(Number);
      for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
        const nx = x + dx;
        const ny = y + dy;
        const cKey = `${nx},${ny}`;

        if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
            grid[ny][nx] === '.' && !firebreaks.has(cKey) && !burned.has(cKey)) {
          candidates.add(cKey);
        }
      }
    }

    for (const cKey of candidates) {
      const testFirebreaks = new Set(firebreaks);
      testFirebreaks.add(cKey);
      const testBurned = simulateFireSpread(grid, ignitions, testFirebreaks, width, height);

      let testSaved = 0;
      for (const town of towns) {
        if (!testBurned.has(`${town.x},${town.y}`)) {
          testSaved += town.value;
        }
      }

      const improvement = testSaved - currentSaved;
      if (improvement > bestImprovement) {
        bestImprovement = improvement;
        bestCell = cKey;
      }
    }

    if (bestCell) {
      firebreaks.add(bestCell);
    } else {
      break;
    }
  }

  return Array.from(firebreaks).map(key => {
    const [x, y] = key.split(',').map(Number);
    return [x, y];
  });
}

const result = solve(data as InputData);
