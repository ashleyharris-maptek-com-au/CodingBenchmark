#include <iostream>
#include <thread>
#include <mutex>
#include <cmath>
#include <vector>
#include <queue>
#include <atomic>
#include <chrono>
#include <fstream>
#include <iomanip>
#include <cstring>

using namespace std;

// State structure matching input format
struct State {
    float x, y, z;
    float vx, vy, vz;
    float pitch, yaw, roll;
    float pitch_rate, yaw_rate, roll_rate;
    float fuel_remaining;
};

struct TerrainCell {
    uint16_t ground_top;
    uint16_t floating_bottom;
    uint16_t floating_top;
};

// Global configuration
int terrain_width, terrain_depth, terrain_height;
float gravity, max_thrust;
float target_x, target_y, target_z;
vector<vector<TerrainCell>> terrain;

// Threading and synchronization
State current_state = {};
mutex state_mutex;
atomic<bool> terrain_loaded(false);
atomic<bool> running(true);
queue<State> state_queue;
mutex queue_mutex;

// Read little-endian uint16
uint16_t read_u16_le(FILE* f) {
    unsigned char lo = fgetc(f);
    unsigned char hi = fgetc(f);
    return ((uint16_t)hi << 8) | (uint16_t)lo;
}

// Load terrain from binary file
void load_terrain(const string& filename) {
    FILE* f = fopen(filename.c_str(), "rb");
    if (!f) return;

    terrain.assign(terrain_width, vector<TerrainCell>(terrain_depth));

    for (int x = 0; x < terrain_width; x++) {
        for (int y = 0; y < terrain_depth; y++) {
            uint16_t a = read_u16_le(f);
            uint16_t b = read_u16_le(f);
            uint16_t c = read_u16_le(f);
            terrain[x][y] = {a, b, c};
        }
    }

    fclose(f);
    terrain_loaded = true;
}

// Check if position is solid terrain
bool is_solid(float px, float py, float pz) {
    // Out of bounds check
    if (px < 0 || px >= terrain_width || py < 0 || py >= terrain_depth ||
        pz < 0 || pz >= terrain_height) {
        return true;
    }

    if (!terrain_loaded) return false;

    int ix = (int)floor(px);
    int iy = (int)floor(py);
    int iz = (int)floor(pz);

    if (ix < 0 || ix >= terrain_width || iy < 0 || iy >= terrain_depth)
        return true;

    const TerrainCell& cell = terrain[ix][iy];

    // Check ground rock
    if (iz < cell.ground_top) return true;
    // Check floating rock
    if (iz >= cell.floating_bottom && iz < cell.floating_top) return true;

    return false;
}

// Thread: Read stdin and populate state queue
void stdin_reader() {
    string line;

    // Read header line 1: width depth height gravity max_thrust
    cin >> terrain_width >> terrain_depth >> terrain_height >> gravity >> max_thrust;

    // Read header line 2: start_x start_y start_z target_x target_y target_z
    float start_x, start_y, start_z;
    cin >> start_x >> start_y >> start_z;
    cin >> target_x >> target_y >> target_z;

    // Read header line 3: terrain file path
    cin.ignore(1);
    getline(cin, line);
    string terrain_path = line;

    // Start terrain loading in background thread
    thread terrain_loader(load_terrain, terrain_path);
    terrain_loader.detach();

    // Read STATE marker
    string marker;
    cin >> marker;

    // Read state updates continuously
    while (running) {
        State s;
        if (!(cin >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
                  >> s.pitch >> s.yaw >> s.roll
                  >> s.pitch_rate >> s.yaw_rate >> s.roll_rate
                  >> s.fuel_remaining)) {
            break;
        }

        {
            lock_guard<mutex> lock(queue_mutex);
            state_queue.push(s);
        }
    }

    running = false;
}

// Thread: Control algorithm
void controller() {
    auto last_output_time = chrono::high_resolution_clock::now();

    while (running) {
        // Get latest state from queue
        State s;
        {
            lock_guard<mutex> lock(queue_mutex);
            if (state_queue.empty()) {
                this_thread::sleep_for(chrono::milliseconds(2));
                continue;
            }
            s = state_queue.front();
            state_queue.pop();
        }

        // Update global state
        {
            lock_guard<mutex> lock(state_mutex);
            current_state = s;
        }

        // Calculate relative position to target
        float dx = target_x - s.x;
        float dy = target_y - s.y;
        float dz = target_z - s.z;
        float horizontal_dist = sqrt(dx * dx + dy * dy);

        float thrust = 0.5f;
        float pitch_cmd = 0.0f;
        float yaw_cmd = 0.0f;
        float roll_cmd = 0.0f;

        // Phase 1: High altitude descent (dz > 30m or still falling fast)
        if (dz > 30.0f || s.vz < -5.0f) {
            // Maneuver horizontally toward target
            if (horizontal_dist > 5.0f) {
                float target_heading = atan2(dy, dx);
                float heading_error = target_heading - s.yaw;

                // Normalize heading error to [-pi, pi]
                while (heading_error > M_PI) heading_error -= 2 * M_PI;
                while (heading_error < -M_PI) heading_error += 2 * M_PI;

                yaw_cmd = heading_error * 0.3f;
                yaw_cmd = max(-1.0f, min(1.0f, yaw_cmd));
            }

            // Control descent rate: aim for about -2 m/s
            if (s.vz > -1.5f) {
                thrust = 0.4f;  // Let gravity pull down
            } else if (s.vz < -3.0f) {
                thrust = 0.65f;  // Slow down descent
            } else {
                thrust = 0.5f;
            }
        }
        // Phase 2: Medium altitude stabilization
        else if (dz > 10.0f) {
            // Fine horizontal approach
            pitch_cmd = dx * 0.03f;
            yaw_cmd = -dy * 0.03f;

            pitch_cmd = max(-0.5f, min(0.5f, pitch_cmd));
            yaw_cmd = max(-0.5f, min(0.5f, yaw_cmd));

            // Gentle descent
            if (s.vz > -1.0f) {
                thrust = 0.5f;
            } else if (s.vz < -2.0f) {
                thrust = 0.6f;
            } else {
                thrust = 0.55f;
            }
        }
        // Phase 3: Final approach (landing)
        else {
            // Counteract horizontal velocity
            pitch_cmd = s.vx * 0.12f;
            yaw_cmd = -s.vy * 0.08f;

            // Very gentle descent - minimum thrust variation
            if (abs(s.vz) < 0.2f) {
                thrust = 0.5f;
            } else if (s.vz > -0.3f) {
                thrust = 0.48f;
            } else {
                thrust = 0.52f;
            }
        }

        // Clamp all commands to valid ranges
        thrust = max(0.0f, min(1.0f, thrust));
        pitch_cmd = max(-1.0f, min(1.0f, pitch_cmd));
        yaw_cmd = max(-1.0f, min(1.0f, yaw_cmd));
        roll_cmd = max(-1.0f, min(1.0f, roll_cmd));

        // Output control command with proper precision
        cout << fixed << setprecision(4)
             << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        cout.flush();

        // Rate limiting to maintain ~10Hz output
        auto now = chrono::high_resolution_clock::now();
        auto ms_elapsed = chrono::duration_cast<chrono::milliseconds>(now - last_output_time).count();
        if (ms_elapsed < 100) {
            this_thread::sleep_for(chrono::milliseconds(100 - ms_elapsed));
        }
        last_output_time = chrono::high_resolution_clock::now();
    }
}

int main() {
    // Disable synchronization between C and C++ stdio for faster I/O
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    // Start reader thread (handles stdin)
    thread reader_thread(stdin_reader);

    // Start controller thread (handles control algorithm)
    thread control_thread(controller);

    // Wait for both threads to complete
    reader_thread.join();
    control_thread.join();

    return 0;
}
