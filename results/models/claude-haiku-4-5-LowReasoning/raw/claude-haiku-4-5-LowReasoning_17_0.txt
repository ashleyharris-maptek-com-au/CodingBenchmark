using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    struct Box
    {
        public int Index, W, H, D;
        public long Volume => (long)W * H * D;
    }

    struct FreeRect
    {
        public int X, Y, Z, W, H, D;
    }

    struct Placement
    {
        public int Index, X, Y, Z;
    }

    static void Main()
    {
        var line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);
        int containerW = int.Parse(line1[1]);
        int containerH = int.Parse(line1[2]);
        int containerD = int.Parse(line1[3]);

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            var dims = Console.ReadLine().Split();
            boxes[i] = new Box
            {
                Index = i,
                W = int.Parse(dims[0]),
                H = int.Parse(dims[1]),
                D = int.Parse(dims[2])
            };
        }

        var placements = new List<Placement>();
        var freeRects = new List<FreeRect>
        {
            new FreeRect { X = 0, Y = 0, Z = 0, W = containerW, H = containerH, D = containerD }
        };

        // Sort boxes by volume (descending) for better packing
        var sortedBoxes = boxes.OrderByDescending(b => b.Volume).ToList();

        foreach (var box in sortedBoxes)
        {
            // Generate all unique orientations
            var orientations = GetOrientations(box.W, box.H, box.D);
            int bestRectIdx = -1;
            (int w, int h, int d) bestDims = (0, 0, 0);
            int bestWaste = int.MaxValue;

            // Find best fitting rectangle and orientation
            for (int rectIdx = 0; rectIdx < freeRects.Count; rectIdx++)
            {
                var rect = freeRects[rectIdx];

                foreach (var (w, h, d) in orientations)
                {
                    if (w <= rect.W && h <= rect.H && d <= rect.D)
                    {
                        // Prefer rectangle with least waste
                        int waste = rect.W * rect.H * rect.D - w * h * d;
                        if (waste < bestWaste)
                        {
                            bestWaste = waste;
                            bestRectIdx = rectIdx;
                            bestDims = (w, h, d);
                        }
                    }
                }
            }

            // Place the box if we found a suitable rectangle
            if (bestRectIdx >= 0)
            {
                var rect = freeRects[bestRectIdx];
                placements.Add(new Placement { Index = box.Index, X = rect.X, Y = rect.Y, Z = rect.Z });

                // Remove the used rectangle
                freeRects.RemoveAt(bestRectIdx);

                // Add new free rectangles from guillotine splits
                if (bestDims.w < rect.W)
                    freeRects.Add(new FreeRect
                    {
                        X = rect.X + bestDims.w,
                        Y = rect.Y,
                        Z = rect.Z,
                        W = rect.W - bestDims.w,
                        H = rect.H,
                        D = rect.D
                    });

                if (bestDims.h < rect.H)
                    freeRects.Add(new FreeRect
                    {
                        X = rect.X,
                        Y = rect.Y + bestDims.h,
                        Z = rect.Z,
                        W = bestDims.w,
                        H = rect.H - bestDims.h,
                        D = rect.D
                    });

                if (bestDims.d < rect.D)
                    freeRects.Add(new FreeRect
                    {
                        X = rect.X,
                        Y = rect.Y,
                        Z = rect.Z + bestDims.d,
                        W = bestDims.w,
                        H = bestDims.h,
                        D = rect.D - bestDims.d
                    });

                // Remove invalid rectangles (those that don't fit the constraints)
                freeRects = freeRects.Where(r => r.W > 0 && r.H > 0 && r.D > 0).ToList();
            }
        }

        // Output results
        Console.WriteLine(placements.Count);
        foreach (var p in placements.OrderBy(x => x.Index))
        {
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
        }
    }

    static List<(int, int, int)> GetOrientations(int w, int h, int d)
    {
        var orientations = new HashSet<(int, int, int)>();
        var dims = new[] { w, h, d };

        // Generate all permutations
        var perms = new[] {
            (dims[0], dims[1], dims[2]),
            (dims[0], dims[2], dims[1]),
            (dims[1], dims[0], dims[2]),
            (dims[1], dims[2], dims[0]),
            (dims[2], dims[0], dims[1]),
            (dims[2], dims[1], dims[0])
        };

        foreach (var perm in perms)
            orientations.Add(perm);

        return orientations.ToList();
    }
}
