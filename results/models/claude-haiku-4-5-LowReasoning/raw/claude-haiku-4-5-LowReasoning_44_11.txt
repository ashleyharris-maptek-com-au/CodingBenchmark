#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Scale UV coordinates to 6x6 grid
    vec2 gridUV = uv * 6.0;
    vec2 cellPos = fract(gridUV); // Position within cell [0, 1)

    // Distance from cell center (0.5, 0.5)
    vec2 fromCenter = cellPos - vec2(0.5);
    float distance = length(fromCenter);

    // Determine base color based on distance from cell center
    vec3 baseColor;
    if (distance < 0.3) {
        baseColor = vec3(1.0, 0.8, 0.0); // Yellow dot
    } else {
        baseColor = vec3(0.1, 0.1, 0.3); // Dark blue background
    }

    // Lambertian shading calculation
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shadingFactor = 0.3 + 0.7 * diffuse;

    // Apply shading to base color
    vec3 finalColor = baseColor * shadingFactor;

    outColor = vec4(finalColor, 1.0);
}
