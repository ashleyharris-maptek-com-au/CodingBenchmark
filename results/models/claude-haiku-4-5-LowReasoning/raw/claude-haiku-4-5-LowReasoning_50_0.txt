#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-bounds invocation
    if (i >= N) {
        return;
    }

    // Read current boid state
    vec3 pos = inData[i*2].xyz;
    vec3 vel = inData[i*2+1].xyz;

    // Extract parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    // Accumulate forces and counts for each behavior
    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohPos = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    // Iterate through all other boids
    for (uint j = 0; j < N; j++) {
        // Skip self
        if (i == j) continue;

        vec3 neighborPos = inData[j*2].xyz;
        vec3 neighborVel = inData[j*2+1].xyz;

        // Calculate displacement and distance
        vec3 diff = neighborPos - pos;
        float distance = length(diff);

        // Skip zero-distance neighbors
        if (distance == 0.0) continue;

        // Separation: steer to avoid crowding local flockmates
        if (distance < sepRadius) {
            sepForce -= normalize(diff);
            sepCount++;
        }

        // Alignment: steer towards the average heading of local flockmates
        if (distance < alignRadius) {
            alignForce += neighborVel;
            alignCount++;
        }

        // Cohesion: steer to move toward the average location of local flockmates
        if (distance < cohRadius) {
            cohPos += neighborPos;
            cohCount++;
        }
    }

    // Apply averaged forces with weights
    vec3 accel = vec3(0.0);

    if (sepCount > 0) {
        accel += (sepForce / float(sepCount)) * sepWeight;
    }

    if (alignCount > 0) {
        vec3 avgVel = alignForce / float(alignCount);
        accel += (avgVel - vel) * alignWeight;
    }

    if (cohCount > 0) {
        vec3 avgPos = cohPos / float(cohCount);
        accel += (avgPos - pos) * cohWeight;
    }

    // Apply boundary force when exceeding bounds
    float boundLimit = boundSize * 0.4;
    for (int axis = 0; axis < 3; axis++) {
        if (abs(pos[axis]) > boundLimit) {
            // Push back toward center (0)
            accel[axis] -= sign(pos[axis]) * boundForce;
        }
    }

    // Update velocity: v = v + a*dt
    vel += accel * dt;

    // Clamp velocity magnitude to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    // Update position: p = p + v*dt
    pos += vel * dt;

    // Write updated boid state to output buffer
    outData[i*2] = vec4(pos, 0.0);
    outData[i*2+1] = vec4(vel, 0.0);
}
