using System;
using System.Collections.Generic;
using System.Linq;

class Solution
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        int m = int.Parse(firstLine[1]);

        var edges = new List<(int u, int v, int w)>();
        var adj = new List<(int to, int edgeId)>[n];

        for (int i = 0; i < n; i++)
            adj[i] = new List<(int, int)>();

        for (int i = 0; i < m; i++)
        {
            string[] parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);

            edges.Add((u, v, w));
            adj[u].Add((v, i));
            if (u != v)
                adj[v].Add((u, i));
        }

        var oddVerts = Enumerable.Range(0, n).Where(i => adj[i].Count % 2 == 1).ToList();

        if (oddVerts.Count == 0)
        {
            var path = Hierholzer(adj, edges, n);
            Console.WriteLine(path.Count);
            Console.WriteLine(string.Join(" ", path));
            return;
        }

        var dist = ComputeShortestPaths(edges, n);
        var matching = GreedyMatching(oddVerts, dist);

        foreach (var (u, v) in matching)
        {
            var pathNodes = ReconstructShortestPath(u, v, dist, n);
            for (int i = 0; i < pathNodes.Count - 1; i++)
            {
                int a = pathNodes[i], b = pathNodes[i + 1];
                int w = (int)dist[a][b];
                adj[a].Add((b, edges.Count));
                if (a != b)
                    adj[b].Add((a, edges.Count));
                edges.Add((a, b, w));
            }
        }

        var result = Hierholzer(adj, edges, n);
        Console.WriteLine(result.Count);
        Console.WriteLine(string.Join(" ", result));
    }

    static long[][] ComputeShortestPaths(List<(int, int, int)> edges, int n)
    {
        const long INF = long.MaxValue / 2;
        var dist = new long[n][];
        for (int i = 0; i < n; i++)
        {
            dist[i] = new long[n];
            for (int j = 0; j < n; j++)
                dist[i][j] = i == j ? 0 : INF;
        }

        foreach (var (u, v, w) in edges)
        {
            dist[u][v] = Math.Min(dist[u][v], w);
            dist[v][u] = Math.Min(dist[v][u], w);
        }

        for (int k = 0; k < n; k++)
            for (int i = 0; i < n; i++)
                for (int j = 0; j < n; j++)
                    if (dist[i][k] != INF && dist[k][j] != INF)
                        dist[i][j] = Math.Min(dist[i][j], dist[i][k] + dist[k][j]);

        return dist;
    }

    static List<(int, int)> GreedyMatching(List<int> odds, long[][] dist)
    {
        var result = new List<(int, int)>();
        var used = new bool[odds.Count];

        for (int i = 0; i < odds.Count; i++)
        {
            if (used[i]) continue;

            long best = long.MaxValue;
            int bestj = -1;

            for (int j = i + 1; j < odds.Count; j++)
            {
                if (used[j]) continue;
                if (dist[odds[i]][odds[j]] < best)
                {
                    best = dist[odds[i]][odds[j]];
                    bestj = j;
                }
            }

            if (bestj >= 0)
            {
                result.Add((odds[i], odds[bestj]));
                used[i] = used[bestj] = true;
            }
        }

        return result;
    }

    static List<int> ReconstructShortestPath(int s, int t, long[][] dist, int n)
    {
        var path = new List<int> { s };
        int curr = s;

        while (curr != t)
        {
            for (int next = 0; next < n; next++)
            {
                if (dist[curr][next] < long.MaxValue / 2 &&
                    dist[curr][next] + dist[next][t] == dist[curr][t])
                {
                    path.Add(next);
                    curr = next;
                    break;
                }
            }
        }

        return path;
    }

    static List<int> Hierholzer(List<(int, int)>[] adj, List<(int, int, int)> edges, int n)
    {
        var g = new List<int>[n];
        var used = new bool[edges.Count];

        for (int i = 0; i < n; i++)
            g[i] = new List<int>();

        for (int u = 0; u < n; u++)
            foreach (var (v, id) in adj[u])
                g[u].Add(id);

        var stack = new Stack<int>();
        var path = new List<int>();

        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;

            for (int i = g[u].Count - 1; i >= 0; i--)
            {
                int id = g[u][i];
                if (!used[id])
                {
                    used[id] = true;
                    var (a, b, w) = edges[id];
                    int v = (a == u) ? b : a;
                    g[u].RemoveAt(i);
                    stack.Push(v);
                    found = true;
                    break;
                }
            }

            if (!found)
                path.Add(stack.Pop());
        }

        path.Reverse();
        return path;
    }
}
