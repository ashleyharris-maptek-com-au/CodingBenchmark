use std::io::{self, BufRead};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read H and W
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let h: usize = parts[0].parse().unwrap();
    let w: usize = parts[1].parse().unwrap();

    // Read the maze and find start/goal positions
    let mut maze = Vec::new();
    let mut start = (0, 0);
    let mut goal = (0, 0);

    for i in 0..h {
        let line = lines.next().unwrap().unwrap();
        maze.push(line);

        for (j, ch) in maze[i].chars().enumerate() {
            if ch == 'A' {
                start = (i, j);
            } else if ch == 'B' {
                goal = (i, j);
            }
        }
    }

    // BFS to find shortest path
    let mut queue = VecDeque::new();
    let mut visited = vec![vec![false; w]; h];
    let mut parent = vec![vec![None; w]; h];

    queue.push_back(start);
    visited[start.0][start.1] = true;

    let directions = [(0, 1), (0, -1), (1, 0), (-1, 0)];

    while let Some((r, c)) = queue.pop_front() {
        if (r, c) == goal {
            break;
        }

        for (dr, dc) in &directions {
            let nr = (r as i32 + dr) as usize;
            let nc = (c as i32 + dc) as usize;

            if nr < h && nc < w && !visited[nr][nc] {
                let cell = maze[nr].chars().nth(nc).unwrap();
                if cell != '#' {
                    visited[nr][nc] = true;
                    parent[nr][nc] = Some((r, c));
                    queue.push_back((nr, nc));
                }
            }
        }
    }

    // Reconstruct path from goal to start
    let mut path = Vec::new();

    if visited[goal.0][goal.1] {
        let mut current = goal;
        while current != start {
            path.push(current);
            current = parent[current.0][current.1].unwrap();
        }
        path.push(start);
        path.reverse();
    }

    // Output path
    println!("{}", path.len());
    for (r, c) in path {
        println!("{} {}", c, r);
    }
}
