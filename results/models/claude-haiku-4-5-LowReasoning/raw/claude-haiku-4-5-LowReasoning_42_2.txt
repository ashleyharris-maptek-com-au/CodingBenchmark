OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0p02 = OpConstant %float 0.02
%float_0p04 = OpConstant %float 0.04
%float_0p5 = OpConstant %float 0.5
%float_0p034641 = OpConstant %float 0.034641
%float_neg_0p02 = OpConstant %float -0.02
%float_neg_0p04 = OpConstant %float -0.04
%float_neg_0p034641 = OpConstant %float -0.034641

%gl_PerVertex_in = OpTypeStruct %v4float
%array_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %int_1
%ptr_Input_array = OpTypePointer Input %array_gl_PerVertex_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_struct = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float

%fn_void = OpTypeFunction %void

%gl_in = OpVariable %ptr_Input_array Input
%gl_out = OpVariable %ptr_Output_struct Output
%inWorldPos = OpVariable %ptr_Input_v3 Input
%inNormal = OpVariable %ptr_Input_v3 Input
%inColor = OpVariable %ptr_Input_v3 Input
%outColor = OpVariable %ptr_Output_v3 Output

%main = OpFunction %void None %fn_void
%label_entry = OpLabel

%pos_addr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_addr
%color = OpLoad %v3float %inColor

%edge_color = OpVectorTimesScalar %v3float %color %float_0p5

%out_pos_addr = OpAccessChain %ptr_Output_v4 %gl_out %int_0

; Triangle 1: center, e0, e1
OpStore %out_pos_addr %pos
OpStore %outColor %color
OpEmitVertex

%e0_offset = OpConstantComposite %v4float %float_0p04 %float_0 %float_0 %float_0
%e0_pos = OpFAdd %v4float %pos %e0_offset
OpStore %out_pos_addr %e0_pos
OpStore %outColor %edge_color
OpEmitVertex

%e1_offset = OpConstantComposite %v4float %float_0p02 %float_0p034641 %float_0 %float_0
%e1_pos = OpFAdd %v4float %pos %e1_offset
OpStore %out_pos_addr %e1_pos
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

; Triangle 2: center, e1, e2
OpStore %out_pos_addr %pos
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_addr %e1_pos
OpStore %outColor %edge_color
OpEmitVertex

%e2_offset = OpConstantComposite %v4float %float_neg_0p02 %float_0p034641 %float_0 %float_0
%e2_pos = OpFAdd %v4float %pos %e2_offset
OpStore %out_pos_addr %e2_pos
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

; Triangle 3: center, e2, e3
OpStore %out_pos_addr %pos
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_addr %e2_pos
OpStore %outColor %edge_color
OpEmitVertex

%e3_offset = OpConstantComposite %v4float %float_neg_0p04 %float_0 %float_0 %float_0
%e3_pos = OpFAdd %v4float %pos %e3_offset
OpStore %out_pos_addr %e3_pos
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

; Triangle 4: center, e3, e4
OpStore %out_pos_addr %pos
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_addr %e3_pos
OpStore %outColor %edge_color
OpEmitVertex

%e4_offset = OpConstantComposite %v4float %float_neg_0p02 %float_neg_0p034641 %float_0 %float_0
%e4_pos = OpFAdd %v4float %pos %e4_offset
OpStore %out_pos_addr %e4_pos
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

; Triangle 5: center, e4, e5
OpStore %out_pos_addr %pos
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_addr %e4_pos
OpStore %outColor %edge_color
OpEmitVertex

%e5_offset = OpConstantComposite %v4float %float_0p02 %float_neg_0p034641 %float_0 %float_0
%e5_pos = OpFAdd %v4float %pos %e5_offset
OpStore %out_pos_addr %e5_pos
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

; Triangle 6: center, e5, e0
OpStore %out_pos_addr %pos
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos_addr %e5_pos
OpStore %outColor %edge_color
OpEmitVertex

OpStore %out_pos_addr %e0_pos
OpStore %outColor %edge_color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
