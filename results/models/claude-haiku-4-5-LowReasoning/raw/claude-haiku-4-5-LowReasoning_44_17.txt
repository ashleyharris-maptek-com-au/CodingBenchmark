#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Brick pattern parameters
    float mortarThicknessX = 0.05;
    float mortarThicknessY = 0.1;
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 brickColor = vec3(0.7, 0.2, 0.1);

    // Apply UV scaling: 8x16 bricks
    vec2 scaledUv = uv * vec2(8.0, 16.0);

    // Apply stagger: every other row is offset by 0.5
    int row = int(floor(scaledUv.y));
    float adjustedX = scaledUv.x;
    if ((row & 1) == 1) {
        adjustedX += 0.5;
    }

    // Get position within brick
    vec2 brickUv = fract(vec2(adjustedX, scaledUv.y));

    // Determine if in mortar
    bool isMortarX = brickUv.x < mortarThicknessX || brickUv.x > (1.0 - mortarThicknessX);
    bool isMortarY = brickUv.y < mortarThicknessY || brickUv.y > (1.0 - mortarThicknessY);
    bool isMortar = isMortarX || isMortarY;

    // Select color
    vec3 baseColor = isMortar ? mortarColor : brickColor;

    // Apply Lambertian shading
    vec3 norm = normalize(normal);
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(norm, lightDir), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;

    // Output
    vec3 finalColor = baseColor * shadingFactor;
    outColor = vec4(finalColor, 1.0);
}
