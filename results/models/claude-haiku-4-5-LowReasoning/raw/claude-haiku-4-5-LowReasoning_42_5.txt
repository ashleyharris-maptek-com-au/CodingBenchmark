OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor

OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %outColor Location 0

%GLSL_450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%uint_2 = OpConstant %uint 2

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

; Input gl_PerVertex
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

; Input arrays
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

; Output gl_PerVertex
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Function
%func_void = OpTypeFunction %void

%main = OpFunction %void None %func_void
%entry = OpLabel

; Load input positions
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1

; Load colors
%ptr_c0 = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c1 = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %ptr_c1

; Extract XY for 2D calculations
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1

; Compute direction
%dir = OpFSub %v2float %p1_xy %p0_xy

; Compute perpendicular: (-dir.y, dir.x)
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_unnorm = OpCompositeConstruct %v2float %neg_dir_y %dir_x

; Normalize perpendicular
%perp_len_sq = OpDot %float %perp_unnorm %perp_unnorm
%perp_len = OpExtInst %float %GLSL_450 Sqrt %perp_len_sq
%perp_len_inv = OpFDiv %float %float_1 %perp_len
%perp = OpVectorTimesScalar %v2float %perp_unnorm %perp_len_inv

; Compute mid = p0 + 0.75 * (p1 - p0)
%diff = OpFSub %v2float %p1_xy %p0_xy
%diff_scaled = OpVectorTimesScalar %v2float %diff %float_0_75
%mid = OpFAdd %v2float %p0_xy %diff_scaled

; Extract Z and W for interpolation
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

; Interpolate Z and W at mid
%diff_z = OpFSub %float %p1_z %p0_z
%diff_z_scaled = OpFMul %float %diff_z %float_0_75
%mid_z = OpFAdd %float %p0_z %diff_z_scaled

%diff_w = OpFSub %float %p1_w %p0_w
%diff_w_scaled = OpFMul %float %diff_w %float_0_75
%mid_w = OpFAdd %float %p0_w %diff_w_scaled

; Compute shaft positions
%shaft_offset = OpVectorTimesScalar %v2float %perp %float_0_015

%p0_left = OpFSub %v2float %p0_xy %shaft_offset
%p0_right = OpFAdd %v2float %p0_xy %shaft_offset
%mid_left = OpFSub %v2float %mid %shaft_offset
%mid_right = OpFAdd %v2float %mid %shaft_offset

; Extract components for v4 construction
%p0_left_x = OpCompositeExtract %float %p0_left 0
%p0_left_y = OpCompositeExtract %float %p0_left 1
%p0_right_x = OpCompositeExtract %float %p0_right 0
%p0_right_y = OpCompositeExtract %float %p0_right 1
%mid_left_x = OpCompositeExtract %float %mid_left 0
%mid_left_y = OpCompositeExtract %float %mid_left 1
%mid_right_x = OpCompositeExtract %float %mid_right 0
%mid_right_y = OpCompositeExtract %float %mid_right 1

; Construct shaft vertex positions
%v1_pos = OpCompositeConstruct %v4float %p0_left_x %p0_left_y %p0_z %p0_w
%v2_pos = OpCompositeConstruct %v4float %p0_right_x %p0_right_y %p0_z %p0_w
%v3_pos = OpCompositeConstruct %v4float %mid_left_x %mid_left_y %mid_z %mid_w
%v4_pos = OpCompositeConstruct %v4float %mid_right_x %mid_right_y %mid_z %mid_w

; Emit shaft (4 vertices)
%out_v4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %out_v4_ptr %v1_pos
OpStore %outColor %c0
OpEmitVertex

OpStore %out_v4_ptr %v2_pos
OpStore %outColor %c0
OpEmitVertex

OpStore %out_v4_ptr %v3_pos
OpStore %outColor %c0
OpEmitVertex

OpStore %out_v4_ptr %v4_pos
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; Compute arrowhead positions
%arrow_offset = OpVectorTimesScalar %v2float %perp %float_0_05

%mid_arrow_left = OpFSub %v2float %mid %arrow_offset
%mid_arrow_right = OpFAdd %v2float %mid %arrow_offset

; Extract components
%mid_al_x = OpCompositeExtract %float %mid_arrow_left 0
%mid_al_y = OpCompositeExtract %float %mid_arrow_left 1
%mid_ar_x = OpCompositeExtract %float %mid_arrow_right 0
%mid_ar_y = OpCompositeExtract %float %mid_arrow_right 1

; Construct arrowhead vertex positions
%v5_pos = OpCompositeConstruct %v4float %mid_al_x %mid_al_y %mid_z %mid_w
%v6_pos = OpCompositeConstruct %v4float %mid_ar_x %mid_ar_y %mid_z %mid_w

; Emit arrowhead (3 vertices)
OpStore %out_v4_ptr %v5_pos
OpStore %outColor %c1
OpEmitVertex

OpStore %out_v4_ptr %v6_pos
OpStore %outColor %c1
OpEmitVertex

OpStore %out_v4_ptr %p1
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
