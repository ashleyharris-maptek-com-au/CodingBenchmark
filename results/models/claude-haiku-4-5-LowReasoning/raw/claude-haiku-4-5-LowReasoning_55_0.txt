#include <iostream>
#include <string>
#include <vector>
#include <cstring>
#include <algorithm>
#include <chrono>
#include <thread>
#include <atomic>
#include <unordered_map>
#include <iomanip>

using namespace std;

struct Move {
    int from, to;
    int piece;
    int captured;
    bool is_capture;
    bool is_promotion;
    int promotion_piece;
    bool is_castling;
    bool is_en_passant;
    bool gives_check;
};

struct Position {
    unsigned char board[64];
    bool white_to_move;
    int castling_rights;
    int en_passant_square;
    int halfmove_clock;
    int fullmove_number;
};

enum Piece : unsigned char {
    EMPTY = 0,
    WHITE_PAWN = 1, WHITE_KNIGHT = 2, WHITE_BISHOP = 3, WHITE_ROOK = 4, WHITE_QUEEN = 5, WHITE_KING = 6,
    BLACK_PAWN = 7, BLACK_KNIGHT = 8, BLACK_BISHOP = 9, BLACK_ROOK = 10, BLACK_QUEEN = 11, BLACK_KING = 12
};

inline bool is_white(Piece p) { return p >= 1 && p <= 6; }
inline bool is_black(Piece p) { return p >= 7 && p <= 12; }
inline Piece black_equivalent(Piece p) { return (Piece)(p + 6); }
inline Piece white_equivalent(Piece p) { return (Piece)(p - 6); }

class ChessEngine {
private:
    Position pos;
    chrono::steady_clock::time_point start_time;
    int timeout_ms;
    atomic<bool> stop_search{false};
    unordered_map<unsigned long long, int> transposition_table;

    unsigned long long hash_position() {
        unsigned long long h = 0;
        for (int i = 0; i < 64; i++) {
            h = h * 31 + pos.board[i];
        }
        h = h * 2 + pos.white_to_move;
        h = h * 16 + pos.castling_rights;
        h = h * 65 + pos.en_passant_square + 1;
        return h;
    }

    bool is_time_up() {
        auto elapsed = chrono::duration_cast<chrono::milliseconds>(
            chrono::steady_clock::now() - start_time).count();
        return elapsed > timeout_ms;
    }

    bool is_white_square(int sq) { return ((sq / 8) + (sq % 8)) % 2 == 0; }

    int piece_value(Piece p) {
        switch(p) {
            case WHITE_PAWN: case BLACK_PAWN: return 100;
            case WHITE_KNIGHT: case BLACK_KNIGHT: return 320;
            case WHITE_BISHOP: case BLACK_BISHOP: return 330;
            case WHITE_ROOK: case BLACK_ROOK: return 500;
            case WHITE_QUEEN: case BLACK_QUEEN: return 900;
            default: return 0;
        }
    }

    int psqt_value(Piece p, int sq) {
        if (p == EMPTY) return 0;

        int rank = sq / 8;
        int file = sq % 8;

        if (is_black(p)) rank = 7 - rank;

        int pst = 0;
        switch(white_equivalent(p)) {
            case WHITE_PAWN:
                pst = (rank > 3) ? (rank - 3) * 10 : 0;
                break;
            case WHITE_KNIGHT:
                pst = (rank == 3 || rank == 4) ? 5 : 0;
                pst += (file >= 2 && file <= 5) ? 5 : 0;
                break;
            case WHITE_BISHOP:
                pst = (rank == 3 || rank == 4) ? 5 : 0;
                break;
            case WHITE_ROOK:
                pst = (rank >= 4) ? (rank - 4) * 3 : 0;
                break;
            case WHITE_QUEEN:
                pst = (rank >= 3) ? 5 : 0;
                break;
            case WHITE_KING:
                pst = (rank <= 1) ? 10 : -5;
                break;
            default: break;
        }
        return pst;
    }

    int evaluate() {
        int score = 0;
        int white_pieces = 0, black_pieces = 0;

        for (int sq = 0; sq < 64; sq++) {
            Piece p = (Piece)pos.board[sq];
            if (p == EMPTY) continue;

            int val = piece_value(p) + psqt_value(p, sq);
            if (is_white(p)) {
                score += val;
                white_pieces++;
            } else {
                score -= val;
                black_pieces++;
            }
        }

        if (white_pieces + black_pieces < 10) {
            score += (pos.white_to_move ? 25 : -25);
        }

        return pos.white_to_move ? score : -score;
    }

    bool is_attacked(int sq, bool by_white) {
        for (int i = 0; i < 64; i++) {
            Piece p = (Piece)pos.board[i];
            if (p == EMPTY || is_white(p) != by_white) continue;

            vector<Move> moves;
            generate_moves_from(i, moves, false);
            for (auto& m : moves) {
                if (m.to == sq) return true;
            }
        }
        return false;
    }

    void generate_moves_from(int from, vector<Move>& moves, bool check_legality) {
        Piece p = (Piece)pos.board[from];
        if (p == EMPTY) return;

        bool is_white = ::is_white(p);
        Piece king = is_white ? WHITE_KING : BLACK_KING;

        auto add_move = [&](int to, bool capture = false, bool promo = false,
                            int promo_piece = 0, bool castle = false, bool ep = false) {
            if (to < 0 || to >= 64) return;
            if (!capture && pos.board[to] != EMPTY) return;
            if (capture && (pos.board[to] == EMPTY || is_white(p) == is_white((Piece)pos.board[to]))) return;

            Move m;
            m.from = from;
            m.to = to;
            m.piece = p;
            m.captured = pos.board[to];
            m.is_capture = capture || (pos.board[to] != EMPTY);
            m.is_promotion = promo;
            m.promotion_piece = promo_piece;
            m.is_castling = castle;
            m.is_en_passant = ep;
            m.gives_check = false;

            if (check_legality) {
                Position backup = pos;
                make_move(m);
                bool king_safe = !is_attacked(find_king(is_white), !is_white);
                pos = backup;
                if (!king_safe) return;
            }

            moves.push_back(m);
        };

        if (p == WHITE_PAWN || p == BLACK_PAWN) {
            int dir = is_white ? -8 : 8;
            int start_rank = is_white ? 6 : 1;
            int promo_rank = is_white ? 0 : 7;

            int rank = from / 8;
            if (rank == promo_rank) return;

            int to = from + dir;
            if (to >= 0 && to < 64 && pos.board[to] == EMPTY) {
                if (rank + (is_white ? -1 : 1) == promo_rank) {
                    for (int np : {2, 3, 4, 5}) add_move(to, false, true, np);
                } else {
                    add_move(to);
                    if (rank == start_rank && pos.board[to + dir] == EMPTY) {
                        add_move(to + dir);
                    }
                }
            }

            for (int dc : {-1, 1}) {
                int dc_sq = from + dir + dc;
                if (dc_sq >= 0 && dc_sq < 64 && pos.board[dc_sq] != EMPTY) {
                    if (is_white(p) != is_white((Piece)pos.board[dc_sq])) {
                        if (rank + (is_white ? -1 : 1) == promo_rank) {
                            for (int np : {2, 3, 4, 5}) add_move(dc_sq, true, true, np);
                        } else {
                            add_move(dc_sq, true);
                        }
                    }
                }

                if (pos.en_passant_square == dc_sq) {
                    add_move(dc_sq, true, false, 0, false, true);
                }
            }
        } else if (p == WHITE_KNIGHT || p == BLACK_KNIGHT) {
            int knight_moves[] = {-17, -15, -10, -6, 6, 10, 15, 17};
            for (int delta : knight_moves) {
                add_move(from + delta);
            }
        } else if (p == WHITE_BISHOP || p == BLACK_BISHOP) {
            for (int delta : {-9, -7, 7, 9}) {
                for (int to = from + delta; to >= 0 && to < 64; to += delta) {
                    add_move(to);
                    if (pos.board[to] != EMPTY) break;
                    if ((from / 8 == 7 && delta < 0) || (from / 8 == 0 && delta > 0)) break;
                    if ((from % 8 == 0 && (delta == -9 || delta == 7)) ||
                        (from % 8 == 7 && (delta == -7 || delta == 9))) break;
                }
            }
        } else if (p == WHITE_ROOK || p == BLACK_ROOK) {
            for (int delta : {-8, -1, 1, 8}) {
                for (int to = from + delta; to >= 0 && to < 64; to += delta) {
                    add_move(to);
                    if (pos.board[to] != EMPTY) break;
                    if ((delta == -1 && from % 8 == 0) || (delta == 1 && from % 8 == 7)) break;
                    if ((delta == -8 && from / 8 == 0) || (delta == 8 && from / 8 == 7)) break;
                }
            }
        } else if (p == WHITE_QUEEN || p == BLACK_QUEEN) {
            for (int delta : {-9, -8, -7, -1, 1, 7, 8, 9}) {
                for (int to = from + delta; to >= 0 && to < 64; to += delta) {
                    add_move(to);
                    if (pos.board[to] != EMPTY) break;
                    if (delta % 8 == 0) {
                        if ((delta == -8 && from / 8 == 0) || (delta == 8 && from / 8 == 7)) break;
                    } else if (delta % 9 == 0) {
                        if ((from / 8 == 0 && delta < 0) || (from / 8 == 7 && delta > 0)) break;
                        if ((from % 8 == 0 && delta < 0) || (from % 8 == 7 && delta > 0)) break;
                    } else {
                        if ((delta == -1 && from % 8 == 0) || (delta == 1 && from % 8 == 7)) break;
                    }
                }
            }
        } else if (p == WHITE_KING || p == BLACK_KING) {
            for (int delta : {-9, -8, -7, -1, 1, 7, 8, 9}) {
                add_move(from + delta);
            }

            if (is_white && (pos.castling_rights & 1) && pos.board[5] == EMPTY && pos.board[6] == EMPTY) {
                if (!is_attacked(4, true) && !is_attacked(5, true)) {
                    add_move(6, false, false, 0, true);
                }
            }
            if (is_white && (pos.castling_rights & 2) && pos.board[1] == EMPTY && pos.board[2] == EMPTY && pos.board[3] == EMPTY) {
                if (!is_attacked(4, true) && !is_attacked(3, true)) {
                    add_move(2, false, false, 0, true);
                }
            }
            if (!is_white && (pos.castling_rights & 4) && pos.board[61] == EMPTY && pos.board[62] == EMPTY) {
                if (!is_attacked(60, false) && !is_attacked(61, false)) {
                    add_move(62, false, false, 0, true);
                }
            }
            if (!is_white && (pos.castling_rights & 8) && pos.board[57] == EMPTY && pos.board[58] == EMPTY && pos.board[59] == EMPTY) {
                if (!is_attacked(60, false) && !is_attacked(59, false)) {
                    add_move(58, false, false, 0, true);
                }
            }
        }
    }

    int find_king(bool white) {
        Piece king = white ? WHITE_KING : BLACK_KING;
        for (int i = 0; i < 64; i++) {
            if (pos.board[i] == king) return i;
        }
        return -1;
    }

    void make_move(const Move& m) {
        if (m.is_castling) {
            pos.board[m.from] = EMPTY;
            pos.board[m.to] = m.piece;
            if (is_white(m.piece)) {
                if (m.to == 6) {
                    pos.board[7] = EMPTY;
                    pos.board[5] = WHITE_ROOK;
                    pos.castling_rights &= ~1;
                    pos.castling_rights &= ~2;
                } else {
                    pos.board[0] = EMPTY;
                    pos.board[3] = WHITE_ROOK;
                    pos.castling_rights &= ~1;
                    pos.castling_rights &= ~2;
                }
            } else {
                if (m.to == 62) {
                    pos.board[63] = EMPTY;
                    pos.board[61] = BLACK_ROOK;
                    pos.castling_rights &= ~4;
                    pos.castling_rights &= ~8;
                } else {
                    pos.board[56] = EMPTY;
                    pos.board[59] = BLACK_ROOK;
                    pos.castling_rights &= ~4;
                    pos.castling_rights &= ~8;
                }
            }
        } else if (m.is_en_passant) {
            pos.board[m.from] = EMPTY;
            pos.board[m.to] = m.piece;
            pos.board[m.to + (is_white(m.piece) ? 8 : -8)] = EMPTY;
        } else {
            pos.board[m.from] = EMPTY;
            pos.board[m.to] = m.is_promotion ?
                (is_white(m.piece) ? m.promotion_piece : m.promotion_piece + 6) : m.piece;

            if (m.piece == WHITE_KING || m.piece == BLACK_KING) {
                if (is_white(m.piece)) {
                    pos.castling_rights &= ~1;
                    pos.castling_rights &= ~2;
                } else {
                    pos.castling_rights &= ~4;
                    pos.castling_rights &= ~8;
                }
            }
            if (m.from == 0) pos.castling_rights &= ~2;
            if (m.from == 7) pos.castling_rights &= ~1;
            if (m.from == 56) pos.castling_rights &= ~8;
            if (m.from == 63) pos.castling_rights &= ~4;
        }

        pos.en_passant_square = -1;
        if ((m.piece == WHITE_PAWN || m.piece == BLACK_PAWN) &&
            abs(m.to - m.from) == 16) {
            pos.en_passant_square = (m.from + m.to) / 2;
        }

        pos.white_to_move = !pos.white_to_move;
        pos.halfmove_clock++;
        if (pos.white_to_move) pos.fullmove_number++;
    }

    int quiesce(int alpha, int beta, int depth) {
        if (is_time_up() || stop_search) return evaluate();

        int stand_pat = evaluate();
        if (stand_pat >= beta) return beta;
        alpha = max(alpha, stand_pat);

        vector<Move> moves;
        bool white = pos.white_to_move;
        for (int i = 0; i < 64; i++) {
            Piece p = (Piece)pos.board[i];
            if (p != EMPTY && is_white(p) == white) {
                generate_moves_from(i, moves, true);
            }
        }

        for (auto& m : moves) {
            if (!m.is_capture && !m.is_promotion) continue;

            Position backup = pos;
            make_move(m);
            int score = -quiesce(-beta, -alpha, depth + 1);
            pos = backup;

            if (score >= beta) return beta;
            alpha = max(alpha, score);
        }

        return alpha;
    }

    int search(int depth, int alpha, int beta) {
        if (is_time_up() || stop_search) return evaluate();

        if (depth == 0) return quiesce(alpha, beta, 0);

        unsigned long long h = hash_position();
        auto it = transposition_table.find(h);
        if (it != transposition_table.end() && depth > 0) {
            return it->second;
        }

        vector<Move> moves;
        bool white = pos.white_to_move;
        for (int i = 0; i < 64; i++) {
            Piece p = (Piece)pos.board[i];
            if (p != EMPTY && is_white(p) == white) {
                generate_moves_from(i, moves, true);
            }
        }

        if (moves.empty()) {
            bool in_check = is_attacked(find_king(white), !white);
            return in_check ? -50000 : 0;
        }

        sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            if (a.is_capture != b.is_capture) return a.is_capture;
            if (a.is_promotion != b.is_promotion) return a.is_promotion;
            return piece_value((Piece)a.captured) > piece_value((Piece)b.captured);
        });

        for (auto& m : moves) {
            Position backup = pos;
            make_move(m);
            int score = -search(depth - 1, -beta, -alpha);
            pos = backup;

            if (score >= beta) {
                transposition_table[h] = beta;
                return beta;
            }
            alpha = max(alpha, score);
        }

        transposition_table[h] = alpha;
        return alpha;
    }

public:
    void parse_fen(const string& fen_str) {
        memset(pos.board, 0, 64);
        pos.castling_rights = 0;
        pos.en_passant_square = -1;
        pos.halfmove_clock = 0;
        pos.fullmove_number = 1;

        size_t pos_idx = 0;
        int sq = 0;

        for (char c : fen_str.substr(0, fen_str.find(' '))) {
            if (c == '/') continue;
            if (isdigit(c)) {
                sq += c - '0';
            } else {
                const char pieces[] = "pnbrqk";
                for (int i = 0; i < 6; i++) {
                    if (tolower(c) == pieces[i]) {
                        pos.board[sq++] = isupper(c) ? i + 7 : i + 1;
                        break;
                    }
                }
            }
        }

        size_t next_space = fen_str.find(' ');
        size_t curr = next_space + 1;
        next_space = fen_str.find(' ', curr);

        pos.white_to_move = fen_str[curr] == 'w';

        curr = next_space + 1;
        next_space = fen_str.find(' ', curr);
        string castling = fen_str.substr(curr, next_space - curr);
        if (castling.find('K') != string::npos) pos.castling_rights |= 1;
        if (castling.find('Q') != string::npos) pos.castling_rights |= 2;
        if (castling.find('k') != string::npos) pos.castling_rights |= 4;
        if (castling.find('q') != string::npos) pos.castling_rights |= 8;

        curr = next_space + 1;
        next_space = fen_str.find(' ', curr);
        string ep = fen_str.substr(curr, next_space - curr);
        if (ep != "-") {
            pos.en_passant_square = (ep[0] - 'a') + (ep[1] - '1') * 8;
        }
    }

    Move find_best_move(int timeout_seconds) {
        timeout_ms = timeout_seconds * 1000 - 100;
        start_time = chrono::steady_clock::now();
        stop_search = false;

        vector<Move> root_moves;
        bool white = pos.white_to_move;
        for (int i = 0; i < 64; i++) {
            Piece p = (Piece)pos.board[i];
            if (p != EMPTY && is_white(p) == white) {
                generate_moves_from(i, root_moves, true);
            }
        }

        Move best_move = root_moves.empty() ? Move() : root_moves[0];
        int best_score = -1000000;

        for (int depth = 1; depth <= 100; depth++) {
            if (is_time_up()) break;

            int alpha = -1000000, beta = 1000000;
            for (auto& m : root_moves) {
                Position backup = pos;
                make_move(m);
                int score = -search(depth - 1, -beta, -alpha);
                pos = backup;

                if (score > best_score) {
                    best_score = score;
                    best_move = m;
                }

                alpha = max(alpha, score);

                if (is_time_up() || stop_search) break;
            }
        }

        return best_move;
    }

    string move_to_string(const Move& m) {
        string notation;

        if (m.is_castling) {
            return (m.to == 6 || m.to == 62) ? "O-O" : "O-O-O";
        }

        const char pieces[] = "pnbrqk";
        char piece_char = pieces[m.piece <= 6 ? m.piece - 1 : m.piece - 7];

        if (piece_char == 'p') {
            notation += char('a' + (m.from % 8));
            notation += char('1' + (m.from / 8));
        } else {
            notation += piece_char;
            notation += char('a' + (m.from % 8));
            notation += char('1' + (m.from / 8));
        }

        if (m.is_capture) notation += 'x';

        notation += char('a' + (m.to % 8));
        notation += char('1' + (m.to / 8));

        if (m.is_en_passant) notation += "ep";

        if (m.is_promotion) {
            notation += '/';
            notation += pieces[m.promotion_piece - 1];
        }

        return notation;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int timeout;
    string fen;
    cin >> timeout;
    cin.ignore();
    getline(cin, fen);

    ChessEngine engine;
    engine.parse_fen(fen);
    Move best_move = engine.find_best_move(timeout);
    cout << engine.move_to_string(best_move) << endl;

    return 0;
}
