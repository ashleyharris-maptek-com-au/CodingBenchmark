OpCapability Shader
OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations for inputs
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4

; Decorations for output
OpDecorate %outColor Location 0

; Decorations for UBO
OpDecorate %ubo Block
OpDecorate %uboInstance DescriptorSet 0
OpDecorate %uboInstance Binding 0

; Member offsets for UBO
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224

; Type declarations
%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; UBO struct type
%ubo = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Pointer types
%ptr_Uniform_ubo = OpTypePointer Uniform %ubo
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4

; Function type
%void_fn = OpTypeFunction %void

; Integer constants
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

; Float constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_6 = OpConstant %float 0.6
%float_0_8 = OpConstant %float 0.8
%float_64 = OpConstant %float 64.0

; Variable declarations
%inWorldPos = OpVariable %ptr_Input_vec3 Input
%inNormal = OpVariable %ptr_Input_vec3 Input
%inUV = OpVariable %ptr_Input_vec2 Input
%inTangent = OpVariable %ptr_Input_vec3 Input
%inColor = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%uboInstance = OpVariable %ptr_Uniform_ubo Uniform

; Extended instruction set import
%glsl = OpExtInstImport "GLSL.std.450"

; Main function definition
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load input values
%worldPos = OpLoad %vec3 %inWorldPos
%normal = OpLoad %vec3 %inNormal
%uv = OpLoad %vec2 %inUV
%tangent = OpLoad %vec3 %inTangent
%color = OpLoad %vec3 %inColor

; Normalize the normal vector
%N = OpExtInst %vec3 %glsl Normalize %normal

; Access lightPos from UBO (member 3)
%lightPosPtr = OpAccessChain %ptr_Uniform_vec4 %uboInstance %int_3
%lightPosVec4 = OpLoad %vec4 %lightPosPtr
%lightPos = OpVectorShuffle %vec3 %lightPosVec4 %lightPosVec4 0 1 2

; Access cameraPos from UBO (member 4)
%cameraPosPtr = OpAccessChain %ptr_Uniform_vec4 %uboInstance %int_4
%cameraPosVec4 = OpLoad %vec4 %cameraPosPtr
%cameraPos = OpVectorShuffle %vec3 %cameraPosVec4 %cameraPosVec4 0 1 2

; Calculate L (normalized light direction)
%lightDir = OpFSub %vec3 %lightPos %worldPos
%L = OpExtInst %vec3 %glsl Normalize %lightDir

; Calculate V (normalized view direction)
%viewDir = OpFSub %vec3 %cameraPos %worldPos
%V = OpExtInst %vec3 %glsl Normalize %viewDir

; Calculate H (half-vector): normalize(L + V)
%LplusV = OpFAdd %vec3 %L %V
%H = OpExtInst %vec3 %glsl Normalize %LplusV

; Calculate diffuse component: max(dot(N, L), 0)
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %dotNL %float_0

; Calculate specular component: pow(max(dot(N, H), 0), 64)
%dotNH = OpDot %float %N %H
%maxNH = OpExtInst %float %glsl FMax %dotNH %float_0
%specular = OpExtInst %float %glsl Pow %maxNH %float_64

; Calculate final intensity: 0.1 + 0.6*diffuse + 0.8*specular
%diffuseTerm = OpFMul %float %float_0_6 %diffuse
%temp1 = OpFAdd %float %float_0_1 %diffuseTerm
%specularTerm = OpFMul %float %float_0_8 %specular
%intensity = OpFAdd %float %temp1 %specularTerm

; Construct output color as grayscale: (intensity, intensity, intensity, 1.0)
%outColorValue = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1

; Store the result
OpStore %outColor %outColorValue

; End function
OpReturn
OpFunctionEnd
