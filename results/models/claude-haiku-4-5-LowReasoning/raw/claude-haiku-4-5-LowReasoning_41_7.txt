%void = OpTypeVoid

%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%ubo_block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ubo_ptr_type = OpTypePointer Uniform %ubo_block
%vec3_input_ptr = OpTypePointer Input %vec3
%vec4_output_ptr = OpTypePointer Output %vec4
%vec4_uniform_ptr = OpTypePointer Uniform %vec4

%main_func_type = OpTypeFunction %void

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_3 = OpConstant %float 3.0

%float_0p3 = OpConstant %float 0.3
%float_0p5 = OpConstant %float 0.5
%float_0p8 = OpConstant %float 0.8

%base_color = OpConstantComposite %vec3 %float_0p2 %float_0p3 %float_0p8
%rim_tint_color = OpConstantComposite %vec3 %float_1 %float_0p8 %float_0p5
%light_direction = OpConstantComposite %vec3 %float_1 %float_1 %float_1

%ubo_var = OpVariable %ubo_ptr_type Uniform
%in_worldPos = OpVariable %vec3_input_ptr Input
%in_normal = OpVariable %vec3_input_ptr Input
%in_color = OpVariable %vec3_input_ptr Input
%out_color = OpVariable %vec4_output_ptr Output

%glsl_std = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %main_func_type
%entry_label = OpLabel

%worldPos = OpLoad %vec3 %in_worldPos
%normal_raw = OpLoad %vec3 %in_normal
%color_unused = OpLoad %vec3 %in_color

%camera_access = OpAccessChain %vec4_uniform_ptr %ubo_var %int_4
%camera_vec4 = OpLoad %vec4 %camera_access
%cameraPos = OpVectorShuffle %vec3 %camera_vec4 %camera_vec4 0 1 2

%to_camera = OpFSub %vec3 %cameraPos %worldPos

%view_dir = OpExtInst %vec3 %glsl_std %Normalize %to_camera

%normal = OpExtInst %vec3 %glsl_std %Normalize %normal_raw

%light_dir_norm = OpExtInst %vec3 %glsl_std %Normalize %light_direction

%dot_n_v = OpDot %float %normal %view_dir
%dot_n_v_clamped = OpExtInst %float %glsl_std %FMax %dot_n_v %float_0

%one_minus_dot = OpFSub %float %float_1 %dot_n_v_clamped
%rim = OpExtInst %float %glsl_std %Pow %one_minus_dot %float_3

%dot_n_l = OpDot %float %normal %light_dir_norm
%dot_n_l_clamped = OpExtInst %float %glsl_std %FMax %dot_n_l %float_0

%diffuse = OpFMul %float %dot_n_l_clamped %float_0p5
%diffuse_contrib = OpFMul %float %float_0p5 %diffuse
%modulation_factor = OpFAdd %float %float_0p3 %diffuse_contrib

%base_modulated = OpVectorTimesScalar %vec3 %base_color %modulation_factor

%rim_contribution = OpVectorTimesScalar %vec3 %rim_tint_color %rim

%final_rgb = OpFAdd %vec3 %base_modulated %rim_contribution

%final_color = OpCompositeConstruct %vec4 %final_rgb %float_1

OpStore %out_color %final_color

OpReturn
