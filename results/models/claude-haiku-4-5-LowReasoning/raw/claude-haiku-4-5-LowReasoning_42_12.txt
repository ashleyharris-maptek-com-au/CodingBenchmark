OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Flat

OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %outColor Location 0

%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_6 = OpConstant %uint 6

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0.0
%float_0_001 = OpConstant %float 0.001
%float_0_015 = OpConstant %float 0.015
%float_0_3 = OpConstant %float 0.3
%float_1 = OpConstant %float 1.0
%float_minus1 = OpConstant %float -1.0

%v3_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_v4_6 = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_Input_arr_v4 = OpTypePointer Input %arr_v4_6
%gl_in = OpVariable %ptr_Input_arr_v4 Input

%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_6
%worldPos_in = OpVariable %ptr_Input_arr_v3 Input
%normal_in = OpVariable %ptr_Input_arr_v3 Input
%color_in = OpVariable %ptr_Input_arr_v3 Input

%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output

%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output
%ptr_Output_v4 = OpTypePointer Output %v4float

%func_main = OpTypeFunction %void
%main = OpFunction %void None %func_main
%label = OpLabel

; Load main triangle vertices (indices 0, 2, 4)
%ac0 = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %ac0
%ac_c0 = OpAccessChain %ptr_Input_v3 %color_in %int_0
%col0 = OpLoad %v3float %ac_c0

%ac2 = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %ac2
%ac_c2 = OpAccessChain %ptr_Input_v3 %color_in %int_2
%col2 = OpLoad %v3float %ac_c2

%ac4 = OpAccessChain %ptr_Input_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %ac4
%ac_c4 = OpAccessChain %ptr_Input_v3 %color_in %int_4
%col4 = OpLoad %v3float %ac_c4

; Load adjacent vertices (indices 1, 3, 5)
%ac1 = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %ac1

%ac3 = OpAccessChain %ptr_Input_v4 %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %ac3

%ac5 = OpAccessChain %ptr_Input_v4 %gl_in %int_5 %int_0
%pos5 = OpLoad %v4float %ac5

; Extract xyz for boundary detection
%p0xyz = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
%p1xyz = OpVectorShuffle %v3float %pos1 %pos1 0 1 2
%p2xyz = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
%p3xyz = OpVectorShuffle %v3float %pos3 %pos3 0 1 2
%p4xyz = OpVectorShuffle %v3float %pos4 %pos4 0 1 2
%p5xyz = OpVectorShuffle %v3float %pos5 %pos5 0 1 2

; Check Edge 0→2 (adjacent is 1) - boundary if pos1 matches pos0 or pos2
%e0_d1 = OpFSub %v3float %p1xyz %p0xyz
%e0_l1 = OpDot %float %e0_d1 %e0_d1
%e0_c1 = OpFOrdLessThan %bool %e0_l1 %float_0_001

%e0_d2 = OpFSub %v3float %p1xyz %p2xyz
%e0_l2 = OpDot %float %e0_d2 %e0_d2
%e0_c2 = OpFOrdLessThan %bool %e0_l2 %float_0_001

%e0_is_bound = OpLogicalOr %bool %e0_c1 %e0_c2

; Check Edge 2→4 (adjacent is 3)
%e1_d1 = OpFSub %v3float %p3xyz %p2xyz
%e1_l1 = OpDot %float %e1_d1 %e1_d1
%e1_c1 = OpFOrdLessThan %bool %e1_l1 %float_0_001

%e1_d2 = OpFSub %v3float %p3xyz %p4xyz
%e1_l2 = OpDot %float %e1_d2 %e1_d2
%e1_c2 = OpFOrdLessThan %bool %e1_l2 %float_0_001

%e1_is_bound = OpLogicalOr %bool %e1_c1 %e1_c2

; Check Edge 4→0 (adjacent is 5)
%e2_d1 = OpFSub %v3float %p5xyz %p4xyz
%e2_l1 = OpDot %float %e2_d1 %e2_d1
%e2_c1 = OpFOrdLessThan %bool %e2_l1 %float_0_001

%e2_d2 = OpFSub %v3float %p5xyz %p0xyz
%e2_l2 = OpDot %float %e2_d2 %e2_d2
%e2_c2 = OpFOrdLessThan %bool %e2_l2 %float_0_001

%e2_is_bound = OpLogicalOr %bool %e2_c1 %e2_c2

; Emit triangle with dimmed colors
%col0_dim = OpVectorTimesScalar %v3float %col0 %float_0_3
%pos_store = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %pos_store %pos0
OpStore %outColor %col0_dim
OpEmitVertex

%col2_dim = OpVectorTimesScalar %v3float %col2 %float_0_3
OpStore %pos_store %pos2
OpStore %outColor %col2_dim
OpEmitVertex

%col4_dim = OpVectorTimesScalar %v3float %col4 %float_0_3
OpStore %pos_store %pos4
OpStore %outColor %col4_dim
OpEmitVertex

OpEndPrimitive

; Emit Edge 0→2 if boundary
OpSelectionMerge %merge0 None
OpBranchConditional %e0_is_bound %then0 %merge0
%then0 = OpLabel
  OpStore %pos_store %pos0
  OpStore %outColor %v3_white
  OpEmitVertex
  OpStore %pos_store %pos2
  OpStore %outColor %v3_white
  OpEmitVertex
  OpStore %pos_store %pos0
  OpStore %outColor %v3_white
  OpEmitVertex
  OpStore %pos_store %pos2
  OpStore %outColor %v3_white
  OpEmitVertex
  OpEndPrimitive
  OpBranch %merge0
%merge0 = OpLabel

; Emit Edge 2→4 if boundary
OpSelectionMerge %merge1 None
OpBranchConditional %e1_is_bound %then1 %merge1
%then1 = OpLabel
  OpStore %pos_store %pos2
  OpStore %outColor %v3_white
  OpEmitVertex
  OpStore %pos_store %pos4
  OpStore %outColor %v3_white
  OpEmitVertex
  OpStore %pos_store %pos2
  OpStore %outColor %v3_white
  OpEmitVertex
  OpStore %pos_store %pos4
  OpStore %outColor %v3_white
  OpEmitVertex
  OpEndPrimitive
  OpBranch %merge1
%merge1 = OpLabel

; Emit Edge 4→0 if boundary
OpSelectionMerge %merge2 None
OpBranchConditional %e2_is_bound %then2 %merge2
%then2 = OpLabel
  OpStore %pos_store %pos4
  OpStore %outColor %v3_white
  OpEmitVertex
  OpStore %pos_store %pos0
  OpStore %outColor %v3_white
  OpEmitVertex
  OpStore %pos_store %pos4
  OpStore %outColor %v3_white
  OpEmitVertex
  OpStore %pos_store %pos0
  OpStore %outColor %v3_white
  OpEmitVertex
  OpEndPrimitive
  OpBranch %merge2
%merge2 = OpLabel

OpReturn
OpFunctionEnd
