OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_m1 = OpConstant %float -1.0
%float_0p04 = OpConstant %float 0.04
%float_0p005 = OpConstant %float 0.005

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_gl_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_color = OpTypeArray %v3float %uint_2

%ptr_in_arr_gl_pv = OpTypePointer Input %arr_gl_pv_in
%ptr_in_arr_color = OpTypePointer Input %arr_color
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_gl_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_gl_pv Input
%color_in = OpVariable %ptr_in_arr_color Input
%gl_out = OpVariable %ptr_out_gl_pv Output
%outColor = OpVariable %ptr_out_v3float Output

%fn_type = OpTypeFunction %void
%main = OpFunction %void None %fn_type
%entry = OpLabel

; Load input positions
%p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load input colors
%c0_ptr = OpAccessChain %ptr_in_v3float %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3float %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr

; Extract xyz components and w
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_w = OpCompositeExtract %float %p1 3

; Compute direction vector
%dir = OpFSub %v3float %p1_xyz %p0_xyz

; Compute perpendicular: cross(dir, (0,0,1))
%z_axis = OpConstantComposite %v3float %float_0 %float_0 %float_1
%perp_unnorm = OpCross %v3float %dir %z_axis
%perp = OpNormalize %v3float %perp_unnorm

; Emit vertex 0: p0 + perp * 0.04
%v0_offset = OpVectorTimesScalar %v3float %perp %float_0p04
%v0_xyz = OpFAdd %v3float %p0_xyz %v0_offset
%v0_x = OpCompositeExtract %float %v0_xyz 0
%v0_y = OpCompositeExtract %float %v0_xyz 1
%v0_z = OpCompositeExtract %float %v0_xyz 2
%v0 = OpCompositeConstruct %v4float %v0_x %v0_y %v0_z %p0_w
%pos_ptr_0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %pos_ptr_0 %v0
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 1: p0 - perp * 0.04
%v1_neg_offset = OpVectorTimesScalar %v3float %perp %float_m1
%v1_offset = OpVectorTimesScalar %v3float %v1_neg_offset %float_0p04
%v1_xyz = OpFAdd %v3float %p0_xyz %v1_offset
%v1_x = OpCompositeExtract %float %v1_xyz 0
%v1_y = OpCompositeExtract %float %v1_xyz 1
%v1_z = OpCompositeExtract %float %v1_xyz 2
%v1 = OpCompositeConstruct %v4float %v1_x %v1_y %v1_z %p0_w
OpStore %pos_ptr_0 %v1
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 2: p1 + perp * 0.005
%v2_offset = OpVectorTimesScalar %v3float %perp %float_0p005
%v2_xyz = OpFAdd %v3float %p1_xyz %v2_offset
%v2_x = OpCompositeExtract %float %v2_xyz 0
%v2_y = OpCompositeExtract %float %v2_xyz 1
%v2_z = OpCompositeExtract %float %v2_xyz 2
%v2 = OpCompositeConstruct %v4float %v2_x %v2_y %v2_z %p1_w
OpStore %pos_ptr_0 %v2
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 3: p1 - perp * 0.005
%v3_neg_offset = OpVectorTimesScalar %v3float %perp %float_m1
%v3_offset = OpVectorTimesScalar %v3float %v3_neg_offset %float_0p005
%v3_xyz = OpFAdd %v3float %p1_xyz %v3_offset
%v3_x = OpCompositeExtract %float %v3_xyz 0
%v3_y = OpCompositeExtract %float %v3_xyz 1
%v3_z = OpCompositeExtract %float %v3_xyz 2
%v3 = OpCompositeConstruct %v4float %v3_x %v3_y %v3_z %p1_w
OpStore %pos_ptr_0 %v3
OpStore %outColor %c1
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
