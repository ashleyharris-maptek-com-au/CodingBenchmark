               OpExecutionMode %main OutputVertices 3

               OpDecorate %color Location 2

               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0

         %int = OpTypeInt 32 1

    %v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
       %int_0 = OpConstant %int 0
       %int_1 = OpConstant %int 1
       %int_2 = OpConstant %int 2
       %int_3 = OpConstant %int 3
     %float_0 = OpConstant %float 0
  %float_0_08 = OpConstant %float 0.08

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
  %ptr_in_v4 = OpTypePointer Input %v4float
  %ptr_in_v3 = OpTypePointer Input %v3float
 %ptr_out_v3 = OpTypePointer Output %v3float
 %ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
  %func_void = OpTypeFunction %void
        %glsl = OpExtInstImport "GLSL.std.450"
       %gl_in = OpVariable %ptr_in_arr_pv Input
      %color = OpVariable %ptr_in_arr_v3 Input
      %gl_out = OpVariable %ptr_out_pv Output
   %outColor = OpVariable %ptr_out_v3 Output
        %main = OpFunction %void None %func_void
       %entry = OpLabel
    %pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %pos0 = OpLoad %v4float %pos0_ptr
    %pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %pos1 = OpLoad %v4float %pos1_ptr
    %pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %pos2 = OpLoad %v4float %pos2_ptr
          %p0 = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
          %p1 = OpVectorShuffle %v3float %pos1 %pos1 0 1 2
          %p2 = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
          %e1 = OpFSub %v3float %p1 %p0
          %e2 = OpFSub %v3float %p2 %p0
       %cross = OpExtInst %v3float %glsl Cross %e1 %e2
      %normal = OpExtInst %v3float %glsl Normalize %cross
      %offset = OpVectorTimesScalar %v3float %normal %float_0_08
      %c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
          %c0 = OpLoad %v3float %c0_ptr
      %c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
          %c1 = OpLoad %v3float %c1_ptr
      %c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
          %c2 = OpLoad %v3float %c2_ptr
       %off_x = OpCompositeExtract %float %offset 0
       %off_y = OpCompositeExtract %float %offset 1
       %off_z = OpCompositeExtract %float %offset 2
   %offset_v4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0
   %new_pos0 = OpFAdd %v4float %pos0 %offset_v4
   %new_pos1 = OpFAdd %v4float %pos1 %offset_v4
   %new_pos2 = OpFAdd %v4float %pos2 %offset_v4
 %out_pos_ptr_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %out_pos_ptr_0 %new_pos0
                OpStore %outColor %c0
                OpEmitVertex
 %out_pos_ptr_1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %out_pos_ptr_1 %new_pos1
                OpStore %outColor %c1
                OpEmitVertex
 %out_pos_ptr_2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %out_pos_ptr_2 %new_pos2
                OpStore %outColor %c2

                OpEndPrimitive
                OpReturn
