use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read number of vertices
    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();

    // Read polygon vertices
    let mut polygon = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        polygon.push((x, y));
    }

    // Read sun direction
    let line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = line.split_whitespace().collect();
    let sun_x: f64 = parts[0].parse().unwrap();
    let sun_y: f64 = parts[1].parse().unwrap();
    let sun_z: f64 = parts[2].parse().unwrap();

    // Normalize sun direction
    let sun_len = (sun_x * sun_x + sun_y * sun_y + sun_z * sun_z).sqrt();
    let _sun = (sun_x / sun_len, sun_y / sun_len, sun_z / sun_len);

    // Standard tetrahedron vertices (before transform)
    let _tet_verts = [
        (1.0, 0.0, -0.707),
        (-1.0, 0.0, -0.707),
        (0.0, 1.0, 0.707),
        (0.0, -1.0, 0.707),
    ];

    // Find polygon bounding box and centroid
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    let mut center_x = 0.0;
    let mut center_y = 0.0;

    for (x, y) in &polygon {
        min_x = min_x.min(*x);
        max_x = max_x.max(*x);
        min_y = min_y.min(*y);
        max_y = max_y.max(*y);
        center_x += x;
        center_y += y;
    }

    center_x /= polygon.len() as f64;
    center_y /= polygon.len() as f64;

    let width = max_x - min_x;
    let height = max_y - min_y;
    let diag = (width * width + height * height).sqrt();

    // Determine number of tetrahedrons based on polygon complexity
    let num_tets = if polygon.len() <= 4 {
        1
    } else if polygon.len() <= 8 {
        1
    } else if diag > 50.0 {
        ((diag / 30.0).ceil() as usize).min(5)
    } else {
        1
    };

    let mut tetrahedrons = Vec::new();

    if num_tets == 1 {
        // Single tetrahedron positioned above polygon center
        let z_pos = (diag / 2.0).max(2.0);
        let scale = (diag / 2.5).min(10.0).max(0.5);

        tetrahedrons.push((center_x, center_y, z_pos, 1.0, 0.0, 0.0, 0.0, scale));
    } else {
        // Multiple tetrahedrons distributed around polygon
        let grid_size = (num_tets as f64).sqrt().ceil() as usize;
        let step_x = (width + 4.0) / (grid_size as f64);
        let step_y = (height + 4.0) / (grid_size as f64);
        let scale = (diag / (3.0 * (grid_size as f64))).min(10.0).max(0.3);

        let mut count = 0;
        for i in 0..grid_size {
            for j in 0..grid_size {
                if count >= num_tets {
                    break;
                }
                let x = min_x - 2.0 + (i as f64) * step_x;
                let y = min_y - 2.0 + (j as f64) * step_y;
                let z = (diag / 3.0).max(1.5);

                // Vary quaternion for different rotations based on grid position
                let angle = (i as f64 + j as f64) * 0.5;
                let qw = angle.cos();
                let qx = (angle * 0.7).sin();
                let qy = (angle * 0.3).sin();
                let qz = (angle * 0.5).sin();

                let len = (qw * qw + qx * qx + qy * qy + qz * qz).sqrt();
                tetrahedrons.push((x, y, z, qw / len, qx / len, qy / len, qz / len, scale));

                count += 1;
            }
        }
    }

    // Output result
    println!("{}", tetrahedrons.len());
    for (px, py, pz, qw, qx, qy, qz, scale) in tetrahedrons {
        println!("{} {} {} {} {} {} {} {}", px, py, pz, qw, qx, qy, qz, scale);
    }
}
