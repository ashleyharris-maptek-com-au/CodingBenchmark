using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static Random rand = new Random(42);

    static void Main()
    {
        int n = int.Parse(Console.ReadLine());

        int[][] flow = new int[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = Console.ReadLine().Split().Select(int.Parse).ToArray();
        }

        int[][] distance = new int[n][];
        for (int i = 0; i < n; i++)
        {
            distance[i] = Console.ReadLine().Split().Select(int.Parse).ToArray();
        }

        int[] bestPerm = new int[n];
        long bestCost;

        if (n <= 8)
        {
            bestPerm = BruteForceQAP(n, flow, distance, out bestCost);
        }
        else
        {
            bestCost = long.MaxValue;
            int numTries = n <= 100 ? 10 : n <= 500 ? 5 : 2;

            for (int t = 0; t < numTries; t++)
            {
                int[] perm = (t == 0) ? GreedyQAP(n, flow, distance) : RandomPermutation(n);
                long cost = CalculateCost(perm, flow, distance);

                LocalSearchSwap(perm, flow, distance, ref cost);

                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestPerm = perm;
                }
            }
        }

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestPerm));
    }

    static int[] BruteForceQAP(int n, int[][] flow, int[][] distance, out long bestCost)
    {
        int[] bestPerm = Enumerable.Range(0, n).ToArray();
        bestCost = CalculateCost(bestPerm, flow, distance);

        int[] perm = (int[])bestPerm.Clone();
        while (NextPermutation(perm))
        {
            long cost = CalculateCost(perm, flow, distance);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestPerm = (int[])perm.Clone();
            }
        }

        return bestPerm;
    }

    static int[] GreedyQAP(int n, int[][] flow, int[][] distance)
    {
        int[] perm = new int[n];
        bool[] used = new bool[n];

        perm[0] = 0;
        used[0] = true;

        for (int i = 1; i < n; i++)
        {
            int bestLoc = -1;
            long bestCost = long.MaxValue;

            for (int loc = 0; loc < n; loc++)
            {
                if (used[loc]) continue;

                long totalCost = 0;
                for (int j = 0; j < i; j++)
                {
                    totalCost += (long)flow[j][i] * distance[perm[j]][loc];
                    totalCost += (long)flow[i][j] * distance[loc][perm[j]];
                }

                if (totalCost < bestCost)
                {
                    bestCost = totalCost;
                    bestLoc = loc;
                }
            }

            perm[i] = bestLoc;
            used[bestLoc] = true;
        }

        return perm;
    }

    static int[] RandomPermutation(int n)
    {
        int[] perm = Enumerable.Range(0, n).ToArray();
        for (int i = n - 1; i > 0; i--)
        {
            int j = rand.Next(i + 1);
            int temp = perm[i];
            perm[i] = perm[j];
            perm[j] = temp;
        }
        return perm;
    }

    static void LocalSearchSwap(int[] perm, int[][] flow, int[][] distance, ref long currentCost)
    {
        int n = perm.Length;
        bool improved = true;
        int iterations = 0;
        int maxIterations = n > 500 ? 5 : n > 100 ? 20 : 100;

        while (improved && iterations < maxIterations)
        {
            improved = false;
            iterations++;

            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    int temp = perm[i];
                    perm[i] = perm[j];
                    perm[j] = temp;

                    long newCost = CalculateCost(perm, flow, distance);
                    if (newCost < currentCost)
                    {
                        currentCost = newCost;
                        improved = true;
                        break;
                    }
                    else
                    {
                        perm[j] = perm[i];
                        perm[i] = temp;
                    }
                }
                if (improved) break;
            }
        }
    }

    static long CalculateCost(int[] perm, int[][] flow, int[][] distance)
    {
        long cost = 0;
        int n = perm.Length;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[i][j] * distance[perm[i]][perm[j]];
            }
        }
        return cost;
    }

    static void Reverse(int[] arr, int start, int end)
    {
        while (start < end)
        {
            int temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
            start++;
            end--;
        }
    }

    static bool NextPermutation(int[] arr)
    {
        int i = arr.Length - 2;
        while (i >= 0 && arr[i] >= arr[i + 1]) i--;

        if (i < 0) return false;

        int j = arr.Length - 1;
        while (j > i && arr[j] <= arr[i]) j--;

        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;

        Reverse(arr, i + 1, arr.Length - 1);
        return true;
    }
}
