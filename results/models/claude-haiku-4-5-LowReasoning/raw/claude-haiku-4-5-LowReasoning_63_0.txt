declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid: string[] = data.grid;
const robots = data.robots;
const jobs = data.jobs;

const distCache = new Map<string, number>();

function bfs(startX: number, startY: number, endX: number, endY: number): number {
  const cacheKey = `${startX},${startY},${endX},${endY}`;
  if (distCache.has(cacheKey)) {
    return distCache.get(cacheKey)!;
  }

  if (startX === endX && startY === endY) {
    return 0;
  }

  const height = grid.length;
  const width = grid[0].length;
  const visited: boolean[][] = Array(height).fill(null).map(() => Array(width).fill(false));
  const queue: Array<[number, number, number]> = [[startX, startY, 0]];
  visited[startY][startX] = true;

  while (queue.length > 0) {
    const [x, y, dist] = queue.shift()!;

    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
          !visited[ny][nx] && grid[ny][nx] === '.') {
        if (nx === endX && ny === endY) {
          distCache.set(cacheKey, dist + 1);
          return dist + 1;
        }
        visited[ny][nx] = true;
        queue.push([nx, ny, dist + 1]);
      }
    }
  }

  distCache.set(cacheKey, Infinity);
  return Infinity;
}

const maxRobotId = Math.max(...robots.map((r: any) => r.id));
const assignments: number[][] = Array(maxRobotId + 1).fill(null).map(() => []);

// Greedy: for each job in order, assign to nearest robot
const sortedJobs = [...jobs].sort((a: any, b: any) => a.id - b.id);

for (const job of sortedJobs) {
  let bestRobot = null;
  let bestDist = Infinity;

  for (const robot of robots) {
    const dist = bfs(robot.x, robot.y, job.x, job.y);

    if (dist < bestDist || (dist === bestDist && (bestRobot === null || robot.id < bestRobot.id))) {
      bestDist = dist;
      bestRobot = robot;
    }
  }

  if (bestRobot !== null) {
    assignments[bestRobot.id].push(job.id);
  }
}
