const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { tasks } = data;

// Build dependency graph
const graph = {};
const inDegree = {};
const outgoing = {};

tasks.forEach(task => {
    graph[task.id] = task;
    inDegree[task.id] = 0;
    outgoing[task.id] = [];
});

tasks.forEach(task => {
    if (task.deps && task.deps.length > 0) {
        task.deps.forEach(dep => {
            outgoing[dep].push(task.id);
            inDegree[task.id]++;
        });
    }
});

// Calculate critical path length (longest path from task to end)
const cp = {};
function criticalPath(id) {
    if (id in cp) return cp[id];
    const task = graph[id];
    let max = task.duration;
    outgoing[id].forEach(next => {
        max = Math.max(max, task.duration + criticalPath(next));
    });
    return cp[id] = max;
}

tasks.forEach(t => criticalPath(t.id));

// Topological sort with priority heuristic
const order = [];
const deg = { ...inDegree };
const queue = [];

Object.keys(deg).forEach(id => {
    if (deg[id] === 0) queue.push(id);
});

while (queue.length > 0) {
    // Prioritize by:
    // 1. Critical path (descending) - tasks on critical path reduce makespan
    // 2. Deadline (ascending) - earlier deadlines have less slack
    // 3. Weight (descending) - higher weight tasks should not be late
    queue.sort((a, b) => {
        if (cp[a] !== cp[b]) return cp[b] - cp[a];
        if (graph[a].deadline !== graph[b].deadline) return graph[a].deadline - graph[b].deadline;
        return graph[b].weight - graph[a].weight;
    });

    const id = queue.shift();
    order.push(id);

    outgoing[id].forEach(next => {
        deg[next]--;
        if (deg[next] === 0) queue.push(next);
    });
}
