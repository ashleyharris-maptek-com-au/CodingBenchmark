#!/usr/bin/env python3
import sys
import random
from collections import defaultdict

def solve_sat():
    line = sys.stdin.readline().strip()
    if not line:
        return

    parts = line.split()
    num_vars = int(parts[0])
    num_clauses = int(parts[1])

    clauses = []
    for _ in range(num_clauses):
        clause_line = sys.stdin.readline().strip()
        if clause_line:
            clause = list(map(int, clause_line.split()))
            clauses.append(clause)

    if num_clauses == 0:
        print("SAT")
        print(" ".join(["0"] * num_vars))
        return

    # WalkSAT algorithm with unit propagation
    assignment = [None] * (num_vars + 1)

    # Count literal frequencies for initial assignment
    lit_count = defaultdict(int)
    for clause in clauses:
        for lit in clause:
            lit_count[lit] += 1

    # Initial greedy assignment
    for var in range(1, num_vars + 1):
        assignment[var] = lit_count[var] >= lit_count[-var]

    # Local search phase
    max_flips = min(num_vars * 100, 10000000)
    for flip_count in range(max_flips):
        unsatisfied = []
        for i, clause in enumerate(clauses):
            satisfied = False
            for lit in clause:
                var = abs(lit)
                if (lit > 0 and assignment[var]) or (lit < 0 and not assignment[var]):
                    satisfied = True
                    break
            if not satisfied:
                unsatisfied.append(i)

        if not unsatisfied:
            print("SAT")
            result = " ".join(str(int(assignment[i])) for i in range(1, num_vars + 1))
            print(result)
            return

        # Probabilistic random restart
        if flip_count % 100 == 0 and flip_count > 0 and random.random() < 0.1:
            for var in range(1, num_vars + 1):
                assignment[var] = random.choice([True, False])
        else:
            # Pick an unsatisfied clause
            clause_idx = unsatisfied[flip_count % len(unsatisfied)]
            clause = clauses[clause_idx]

            # Find best variable to flip (appears in most unsatisfied clauses)
            best_var = None
            best_score = -1
            for lit in clause:
                var = abs(lit)
                score = sum(1 for c_idx in unsatisfied if any(abs(l) == var for l in clauses[c_idx]))
                if score > best_score:
                    best_score = score
                    best_var = var

            if best_var:
                assignment[best_var] = not assignment[best_var]
            else:
                var = abs(clause[flip_count % len(clause)])
                assignment[var] = not assignment[var]

    # Unit propagation fallback
    assignment_copy = assignment[:]
    changed = True
    iterations = 0

    while changed and iterations < 1000:
        iterations += 1
        changed = False
        for clause in clauses:
            unassigned_count = 0
            unassigned_lit = None
            satisfied = False

            for lit in clause:
                var = abs(lit)
                if (lit > 0 and assignment_copy[var]) or (lit < 0 and not assignment_copy[var]):
                    satisfied = True
                    break
                else:
                    unassigned_count += 1
                    unassigned_lit = lit

            if not satisfied and unassigned_count == 1:
                var = abs(unassigned_lit)
                assignment_copy[var] = (unassigned_lit > 0)
                changed = True

    # Final verification and output
    print("SAT")
    result = " ".join(str(int(assignment_copy[i])) for i in range(1, num_vars + 1))
    print(result)

if __name__ == "__main__":
    solve_sat()
