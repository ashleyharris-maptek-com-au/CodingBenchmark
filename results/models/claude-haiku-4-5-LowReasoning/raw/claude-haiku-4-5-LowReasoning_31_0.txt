use std::collections::{HashMap, HashSet};
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let _num_numbers: usize = parts[0].parse().unwrap();
    let target: i64 = parts[1].parse().unwrap();

    let second_line = lines.next().unwrap().unwrap();
    let numbers: Vec<i64> = second_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    match solve_subset_sum(&numbers, target) {
        Some(mut indices) => {
            println!("YES");
            indices.sort_unstable();
            let indices_str: Vec<String> = indices.iter().map(|i| i.to_string()).collect();
            println!("[{}]", indices_str.join(" "));
        }
        None => println!("NO"),
    }
}

fn solve_subset_sum(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(vec![]);
    }

    if numbers.is_empty() {
        return None;
    }

    for (i, &num) in numbers.iter().enumerate() {
        if num == target {
            return Some(vec![i]);
        }
    }

    let mut dp: HashMap<i64, (i64, usize)> = HashMap::new();
    let mut reachable: HashSet<i64> = HashSet::new();
    reachable.insert(0);

    for (i, &num) in numbers.iter().enumerate() {
        let current_reachable: Vec<i64> = reachable.iter().copied().collect();

        for sum in current_reachable {
            let new_sum = sum + num;

            if new_sum == target {
                let mut result = vec![i];
                let mut cur_sum = sum;
                while cur_sum != 0 {
                    let (prev_sum, idx) = dp[&cur_sum];
                    result.push(idx);
                    cur_sum = prev_sum;
                }
                return Some(result);
            }

            if !reachable.contains(&new_sum) {
                reachable.insert(new_sum);
                dp.insert(new_sum, (sum, i));
            }
        }
    }

    None
}
