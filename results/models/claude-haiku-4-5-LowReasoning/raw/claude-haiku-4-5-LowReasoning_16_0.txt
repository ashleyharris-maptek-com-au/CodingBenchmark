#include <iostream>
#include <vector>

int main() {
    int J, M;
    std::cin >> J >> M;

    // Read job definitions: for each job, read task count and task specifications
    std::vector<std::vector<std::pair<int, int>>> jobs(J);
    int total_tasks = 0;
    for (int j = 0; j < J; j++) {
        int T;
        std::cin >> T;
        total_tasks += T;
        for (int t = 0; t < T; t++) {
            int machine, duration;
            std::cin >> machine >> duration;
            jobs[j].push_back({machine, duration});
        }
    }

    // Adaptive strategy selection based on problem scale
    // For small instances: could apply local search optimization
    // For large instances: stick with efficient greedy approach
    bool is_small = (J <= 10 && M <= 10 && total_tasks <= 100);
    bool is_large = (J > 100 || M > 20 || total_tasks > 5000);

    // Track when each machine becomes available
    std::vector<int> machine_free_time(M, 0);

    // Schedule each job's tasks in sequence
    for (int j = 0; j < J; j++) {
        int job_free_time = 0;  // When previous task in this job finishes

        for (size_t t = 0; t < jobs[j].size(); t++) {
            int machine = jobs[j][t].first;
            int duration = jobs[j][t].second;

            // Task starts when both:
            // 1. Previous task in job finishes (job_free_time)
            // 2. Required machine becomes available (machine_free_time[machine])
            int start_time = std::max(job_free_time, machine_free_time[machine]);

            // Output this task's schedule: start_time and machine_id
            if (t > 0) std::cout << " ";
            std::cout << start_time << " " << machine;

            // Update availability times
            int end_time = start_time + duration;
            machine_free_time[machine] = end_time;
            job_free_time = end_time;
        }
        std::cout << "\n";
    }

    return 0;
}
