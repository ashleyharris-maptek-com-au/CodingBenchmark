@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

fn rotl(x: u32, n: u32) -> u32 {
    return (x << n) | (x >> (32u - n));
}

fn custom_hash(data: array<u32, 16>, nonce: u32) -> vec2<u32> {
    var state: array<u32, 16>;

    // Initialize state with data XORed with (nonce + i)
    for (var i: u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = data[i] ^ (nonce + i);
    }

    // 8 rounds of quarter-round mixing
    for (var round: u32 = 0u; round < 8u; round = round + 1u) {
        // Group 1: indices 0, 1, 2, 3
        var a = state[0];
        var b = state[1];
        var c = state[2];
        var d = state[3];

        a = a + b;
        d = d ^ a;
        d = rotl(d, 16u);
        c = c + d;
        b = b ^ c;
        b = rotl(b, 12u);
        a = a + b;
        d = d ^ a;
        d = rotl(d, 8u);
        c = c + d;
        b = b ^ c;
        b = rotl(b, 7u);

        state[0] = a;
        state[1] = b;
        state[2] = c;
        state[3] = d;

        // Group 2: indices 4, 5, 6, 7
        a = state[4];
        b = state[5];
        c = state[6];
        d = state[7];

        a = a + b;
        d = d ^ a;
        d = rotl(d, 16u);
        c = c + d;
        b = b ^ c;
        b = rotl(b, 12u);
        a = a + b;
        d = d ^ a;
        d = rotl(d, 8u);
        c = c + d;
        b = b ^ c;
        b = rotl(b, 7u);

        state[4] = a;
        state[5] = b;
        state[6] = c;
        state[7] = d;

        // Group 3: indices 8, 9, 10, 11
        a = state[8];
        b = state[9];
        c = state[10];
        d = state[11];

        a = a + b;
        d = d ^ a;
        d = rotl(d, 16u);
        c = c + d;
        b = b ^ c;
        b = rotl(b, 12u);
        a = a + b;
        d = d ^ a;
        d = rotl(d, 8u);
        c = c + d;
        b = b ^ c;
        b = rotl(b, 7u);

        state[8] = a;
        state[9] = b;
        state[10] = c;
        state[11] = d;

        // Group 4: indices 12, 13, 14, 15
        a = state[12];
        b = state[13];
        c = state[14];
        d = state[15];

        a = a + b;
        d = d ^ a;
        d = rotl(d, 16u);
        c = c + d;
        b = b ^ c;
        b = rotl(b, 12u);
        a = a + b;
        d = d ^ a;
        d = rotl(d, 8u);
        c = c + d;
        b = b ^ c;
        b = rotl(b, 7u);

        state[12] = a;
        state[13] = b;
        state[14] = c;
        state[15] = d;
    }

    // Fold 16 words to 2
    let h0 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    return vec2<u32>(h0, h1);
}

fn count_trailing_zeros(h0: u32, h1: u32) -> u32 {
    if (h0 == 0u) {
        if (h1 == 0u) {
            return 64u;
        }
        return 32u + countTrailingZeros(h1);
    }
    return countTrailingZeros(h0);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let my_nonce = params.y + gid.x;

    // Guard: if out of range, return
    if (gid.x >= params.z) {
        return;
    }

    // Compute hash
    let hash = custom_hash(baseData, my_nonce);
    let h0 = hash.x;
    let h1 = hash.y;

    // Count trailing zeros
    let trailing_zeros = count_trailing_zeros(h0, h1);

    // Check if meets difficulty
    if (trailing_zeros >= params.x) {
        result[0] = 1u;
        result[1] = my_nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
