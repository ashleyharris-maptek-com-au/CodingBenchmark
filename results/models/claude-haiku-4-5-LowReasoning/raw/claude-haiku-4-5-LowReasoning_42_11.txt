OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExtInstImport "GLSL.std.450"
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_333 = OpConstant %float 0.333333333
%float_0_667 = OpConstant %float 0.666666667
%float_0_02 = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%func_type = OpTypeFunction %void
%glsl = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func_type
%entry = OpLabel

%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2
%ptr_p3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %ptr_p3

%xyz0 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%xyz1 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%xyz2 = OpVectorShuffle %v3float %p2 %p2 0 1 2
%xyz3 = OpVectorShuffle %v3float %p3 %p3 0 1 2

%tan_in = OpFSub %v3float %xyz1 %xyz0
%tan_in_norm = OpExtInst %v3float %glsl Normalize %tan_in
%tan_out = OpFSub %v3float %xyz3 %xyz2
%tan_out_norm = OpExtInst %v3float %glsl Normalize %tan_out

%up = OpCompositeConstruct %v3float %float_0 %float_1 %float_0

%perp_in = OpExtInst %v3float %glsl Cross %tan_in_norm %up
%perp_in_norm = OpExtInst %v3float %glsl Normalize %perp_in
%perp_out = OpExtInst %v3float %glsl Cross %tan_out_norm %up
%perp_out_norm = OpExtInst %v3float %glsl Normalize %perp_out

%perp_in_scaled = OpVectorTimesScalar %v3float %perp_in_norm %float_0_02
%perp_out_scaled = OpVectorTimesScalar %v3float %perp_out_norm %float_0_02

%seg1_mid = OpVectorTimesScalar %v3float %xyz1 %float_0_333
%seg1_mid2 = OpVectorTimesScalar %v3float %xyz2 %float_0_667
%seg1_center = OpFAdd %v3float %seg1_mid %seg1_mid2

%ptr_c1 = OpAccessChain %ptr_in_arr_v3 %color %int_1
%c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_in_arr_v3 %color %int_2
%c2 = OpLoad %v3float %ptr_c2

%seg1_col_a = OpVectorTimesScalar %v3float %c1 %float_0_667
%seg1_col_b = OpVectorTimesScalar %v3float %c2 %float_0_333
%seg1_color = OpFAdd %v3float %seg1_col_a %seg1_col_b

%seg1_v0 = OpFAdd %v3float %seg1_center %perp_in_scaled
%seg1_v0_x = OpCompositeExtract %float %seg1_v0 0
%seg1_v0_y = OpCompositeExtract %float %seg1_v0 1
%seg1_v0_z = OpCompositeExtract %float %seg1_v0 2
%seg1_v0_pos = OpCompositeConstruct %v4float %seg1_v0_x %seg1_v0_y %seg1_v0_z %float_1

%seg1_v1 = OpFSub %v3float %seg1_center %perp_in_scaled
%seg1_v1_x = OpCompositeExtract %float %seg1_v1 0
%seg1_v1_y = OpCompositeExtract %float %seg1_v1 1
%seg1_v1_z = OpCompositeExtract %float %seg1_v1 2
%seg1_v1_pos = OpCompositeConstruct %v4float %seg1_v1_x %seg1_v1_y %seg1_v1_z %float_1

%seg1_v2 = OpFAdd %v3float %seg1_center %perp_out_scaled
%seg1_v2_x = OpCompositeExtract %float %seg1_v2 0
%seg1_v2_y = OpCompositeExtract %float %seg1_v2 1
%seg1_v2_z = OpCompositeExtract %float %seg1_v2 2
%seg1_v2_pos = OpCompositeConstruct %v4float %seg1_v2_x %seg1_v2_y %seg1_v2_z %float_1

%seg1_v3 = OpFSub %v3float %seg1_center %perp_out_scaled
%seg1_v3_x = OpCompositeExtract %float %seg1_v3 0
%seg1_v3_y = OpCompositeExtract %float %seg1_v3 1
%seg1_v3_z = OpCompositeExtract %float %seg1_v3 2
%seg1_v3_pos = OpCompositeConstruct %v4float %seg1_v3_x %seg1_v3_y %seg1_v3_z %float_1

%pos_out_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_0 %seg1_v0_pos
OpStore %outColor %seg1_color
OpEmitVertex

OpStore %pos_out_0 %seg1_v1_pos
OpStore %outColor %seg1_color
OpEmitVertex

OpStore %pos_out_0 %seg1_v2_pos
OpStore %outColor %seg1_color
OpEmitVertex

OpStore %pos_out_0 %seg1_v3_pos
OpStore %outColor %seg1_color
OpEmitVertex

OpEndPrimitive

%seg2_mid = OpVectorTimesScalar %v3float %xyz1 %float_0_333
%seg2_mid2 = OpVectorTimesScalar %v3float %xyz2 %float_0_667
%seg2_center = OpFAdd %v3float %seg2_mid %seg2_mid2

%seg2_col_a = OpVectorTimesScalar %v3float %c1 %float_0_333
%seg2_col_b = OpVectorTimesScalar %v3float %c2 %float_0_667
%seg2_color = OpFAdd %v3float %seg2_col_a %seg2_col_b

%perp_avg = OpFAdd %v3float %perp_in_norm %perp_out_norm
%perp_avg_norm = OpExtInst %v3float %glsl Normalize %perp_avg
%perp_avg_scaled = OpVectorTimesScalar %v3float %perp_avg_norm %float_0_02

%seg2_v0 = OpFAdd %v3float %seg2_center %perp_avg_scaled
%seg2_v0_x = OpCompositeExtract %float %seg2_v0 0
%seg2_v0_y = OpCompositeExtract %float %seg2_v0 1
%seg2_v0_z = OpCompositeExtract %float %seg2_v0 2
%seg2_v0_pos = OpCompositeConstruct %v4float %seg2_v0_x %seg2_v0_y %seg2_v0_z %float_1

%seg2_v1 = OpFSub %v3float %seg2_center %perp_avg_scaled
%seg2_v1_x = OpCompositeExtract %float %seg2_v1 0
%seg2_v1_y = OpCompositeExtract %float %seg2_v1 1
%seg2_v1_z = OpCompositeExtract %float %seg2_v1 2
%seg2_v1_pos = OpCompositeConstruct %v4float %seg2_v1_x %seg2_v1_y %seg2_v1_z %float_1

%seg2_v2 = OpFAdd %v3float %seg2_center %perp_out_scaled
%seg2_v2_x = OpCompositeExtract %float %seg2_v2 0
%seg2_v2_y = OpCompositeExtract %float %seg2_v2 1
%seg2_v2_z = OpCompositeExtract %float %seg2_v2 2
%seg2_v2_pos = OpCompositeConstruct %v4float %seg2_v2_x %seg2_v2_y %seg2_v2_z %float_1

%seg2_v3 = OpFSub %v3float %seg2_center %perp_out_scaled
%seg2_v3_x = OpCompositeExtract %float %seg2_v3 0
%seg2_v3_y = OpCompositeExtract %float %seg2_v3 1
%seg2_v3_z = OpCompositeExtract %float %seg2_v3 2
%seg2_v3_pos = OpCompositeConstruct %v4float %seg2_v3_x %seg2_v3_y %seg2_v3_z %float_1

OpStore %pos_out_0 %seg2_v0_pos
OpStore %outColor %seg2_color
OpEmitVertex

OpStore %pos_out_0 %seg2_v1_pos
OpStore %outColor %seg2_color
OpEmitVertex

OpStore %pos_out_0 %seg2_v2_pos
OpStore %outColor %seg2_color
OpEmitVertex

OpStore %pos_out_0 %seg2_v3_pos
OpStore %outColor %seg2_color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
