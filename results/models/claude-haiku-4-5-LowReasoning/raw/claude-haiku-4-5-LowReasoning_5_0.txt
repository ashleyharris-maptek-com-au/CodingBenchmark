use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let width: usize = parts[0].parse().unwrap();
    let height: usize = parts[1].parse().unwrap();
    let obstacle_count: usize = parts[2].parse().unwrap();
    let require_cycle: bool = parts[3].parse::<u8>().unwrap() == 1;

    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let x: usize = parts[0].parse().unwrap();
        let y: usize = parts[1].parse().unwrap();
        obstacles.insert((x, y));
    }

    // Count non-obstacle cells and their bipartite colors
    let mut total_cells = 0;
    let mut color0 = 0; // (x + y) % 2 == 0
    let mut color1 = 0; // (x + y) % 2 == 1

    for y in 0..height {
        for x in 0..width {
            if !obstacles.contains(&(x, y)) {
                total_cells += 1;
                if (x + y) % 2 == 0 {
                    color0 += 1;
                } else {
                    color1 += 1;
                }
            }
        }
    }

    // Quick validity check using bipartite coloring
    if require_cycle && color0 != color1 {
        println!("0");
        return;
    }
    if !require_cycle && (color0 as i32 - color1 as i32).abs() > 1 {
        println!("0");
        return;
    }

    let mut visited = vec![vec![false; width]; height];
    let mut path = Vec::new();
    visited[0][0] = true;
    path.push((0, 0));

    if find_path(&obstacles, &mut visited, &mut path, total_cells, width, height, require_cycle) {
        println!("{}", path.len());
        for (x, y) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn count_unvisited_neighbors(
    x: usize,
    y: usize,
    visited: &[Vec<bool>],
    obstacles: &HashSet<(usize, usize)>,
    width: usize,
    height: usize,
) -> usize {
    const DIRECTIONS: &[(i32, i32)] = &[(0, 1), (1, 0), (0, -1), (-1, 0)];
    let mut count = 0;
    for &(dx, dy) in DIRECTIONS {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;
        if nx >= 0 && nx < width as i32 && ny >= 0 && ny < height as i32 {
            let nx = nx as usize;
            let ny = ny as usize;
            if !visited[ny][nx] && !obstacles.contains(&(nx, ny)) {
                count += 1;
            }
        }
    }
    count
}

fn find_path(
    obstacles: &HashSet<(usize, usize)>,
    visited: &mut [Vec<bool>],
    path: &mut Vec<(usize, usize)>,
    total_cells: usize,
    width: usize,
    height: usize,
    require_cycle: bool,
) -> bool {
    if path.len() == total_cells {
        if require_cycle {
            let (x0, y0) = path[0];
            let (lx, ly) = path[path.len() - 1];
            let dx = (x0 as i32 - lx as i32).abs();
            let dy = (y0 as i32 - ly as i32).abs();
            return dx + dy == 1;
        } else {
            return true;
        }
    }

    let (x, y) = path[path.len() - 1];

    // Count remaining unvisited cells by bipartite color
    let mut color0_left = 0;
    let mut color1_left = 0;
    for yy in 0..height {
        for xx in 0..width {
            if !visited[yy][xx] && !obstacles.contains(&(xx, yy)) {
                if (xx + yy) % 2 == 0 {
                    color0_left += 1;
                } else {
                    color1_left += 1;
                }
            }
        }
    }

    // Prune if bipartite constraint violated
    if require_cycle && color0_left != color1_left {
        return false;
    }
    if !require_cycle && (color0_left as i32 - color1_left as i32).abs() > 1 {
        return false;
    }

    // Get neighbors and sort by heuristic (prefer constrained nodes)
    const DIRECTIONS: &[(i32, i32)] = &[(0, 1), (1, 0), (0, -1), (-1, 0)];
    let mut neighbors = Vec::new();

    for &(dx, dy) in DIRECTIONS {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;
        if nx >= 0 && nx < width as i32 && ny >= 0 && ny < height as i32 {
            let nx = nx as usize;
            let ny = ny as usize;
            if !visited[ny][nx] && !obstacles.contains(&(nx, ny)) {
                let unvisited_neighbors = count_unvisited_neighbors(nx, ny, visited, obstacles, width, height);
                neighbors.push((unvisited_neighbors, nx, ny));
            }
        }
    }

    // Sort by number of unvisited neighbors (prefer more constrained nodes first)
    neighbors.sort_by_key(|&(count, _, _)| std::cmp::Reverse(count));

    for (_, nx, ny) in neighbors {
        visited[ny][nx] = true;
        path.push((nx, ny));

        if find_path(obstacles, visited, path, total_cells, width, height, require_cycle) {
            return true;
        }

        path.pop();
        visited[ny][nx] = false;
    }

    false
}
