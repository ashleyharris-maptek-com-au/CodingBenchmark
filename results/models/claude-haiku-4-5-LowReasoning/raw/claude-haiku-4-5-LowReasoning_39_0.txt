use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse input
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    // Build adjacency list
    let mut adj = vec![Vec::new(); n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let edge: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let u = edge[0];
        let v = edge[1];
        adj[u].push(v);
        adj[v].push(u);
    }

    // Find best partition using multiple random restarts and local search
    let mut best_partition = Vec::new();
    let mut best_cut = usize::MAX;

    for seed in 0..100 {
        let partition = make_initial_partition(n, seed);
        let partition = local_search(partition, &adj, n);
        let cut = count_cut(&partition, &adj);

        if cut < best_cut {
            best_cut = cut;
            best_partition = partition;
        }
    }

    // Output result
    println!("{}", best_cut);
    let output = best_partition.iter()
        .map(|x| x.to_string())
        .collect::<Vec<_>>()
        .join(" ");
    println!("{}", output);
}

fn make_initial_partition(n: usize, seed: usize) -> Vec<usize> {
    // Create pseudo-random initial partition
    let mut partition = Vec::new();
    let mut used = vec![false; n];

    // Use a simple hash to select vertices
    for i in 0..n {
        if partition.len() < n / 2 {
            let hash = ((i as u64).wrapping_mul(73856093) ^ (seed as u64).wrapping_mul(19349663)) as usize;
            if (hash % 2) == 0 {
                partition.push(i);
                used[i] = true;
            }
        }
    }

    // Fill remaining slots
    for i in 0..n {
        if partition.len() < n / 2 && !used[i] {
            partition.push(i);
        }
    }

    partition
}

fn local_search(mut partition: Vec<usize>, adj: &[Vec<usize>], n: usize) -> Vec<usize> {
    let mut improved = true;
    let mut iterations = 0;

    while improved && iterations < 5000 {
        improved = false;
        iterations += 1;

        let set_a: HashSet<usize> = partition.iter().copied().collect();

        // Try swapping each vertex in partition A with each in B
        for i_idx in 0..partition.len() {
            let i_vertex = partition[i_idx];

            for j_vertex in 0..n {
                if set_a.contains(&j_vertex) {
                    continue;
                }

                // Calculate the change in cut size if we swap i_vertex and j_vertex
                let delta = calculate_swap_delta(i_vertex, j_vertex, &set_a, adj);

                if delta < 0 {
                    // Swap improves the cut
                    partition[i_idx] = j_vertex;
                    improved = true;
                    break;
                }
            }

            if improved {
                break;
            }
        }
    }

    partition
}

fn calculate_swap_delta(i_vertex: usize, j_vertex: usize, set_a: &HashSet<usize>, adj: &[Vec<usize>]) -> i32 {
    let mut delta = 0i32;

    // Count edges from i_vertex
    for &neighbor in &adj[i_vertex] {
        if set_a.contains(&neighbor) && neighbor != i_vertex {
            // Internal edge: removing i breaks it
            delta += 1;
        } else if !set_a.contains(&neighbor) {
            // External edge: removing i loses it from cut
            delta -= 1;
        }
    }

    // Count edges from j_vertex
    for &neighbor in &adj[j_vertex] {
        if set_a.contains(&neighbor) {
            // Edge to partition A: adding j creates cut edge
            delta += 1;
        } else if !set_a.contains(&neighbor) && neighbor != j_vertex {
            // Internal edge: adding j breaks it
            delta -= 1;
        }
    }

    // Handle direct edge between i_vertex and j_vertex
    if adj[i_vertex].contains(&j_vertex) {
        // This edge becomes internal, reduces cut by 1
        delta -= 1;
    }

    delta
}

fn count_cut(partition: &[usize], adj: &[Vec<usize>]) -> usize {
    let set_a: HashSet<usize> = partition.iter().copied().collect();
    let mut cut = 0;

    for &v in partition {
        for &neighbor in &adj[v] {
            if !set_a.contains(&neighbor) {
                cut += 1;
            }
        }
    }

    cut
}
