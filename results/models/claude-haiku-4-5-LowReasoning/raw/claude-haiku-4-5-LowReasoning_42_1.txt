OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3_1 = OpTypeArray %v3float %uint_1

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1
%ptr_in_v3float = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float

%func_void = OpTypeFunction %void

%gl_in = OpVariable %ptr_in_arr_pv Input
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3float Output

%main = OpFunction %void None %func_void
%entry = OpLabel

%pos_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr

%color_ptr = OpAccessChain %ptr_in_v3float %inColor %int_0
%color = OpLoad %v3float %color_ptr

%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

%y_plus = OpFAdd %float %y %float_0_05
%top_v = OpCompositeConstruct %v4float %x %y_plus %z %w

%x_plus = OpFAdd %float %x %float_0_05
%right_v = OpCompositeConstruct %v4float %x_plus %y %z %w

%x_minus = OpFSub %float %x %float_0_05
%left_v = OpCompositeConstruct %v4float %x_minus %y %z %w

%y_minus = OpFSub %float %y %float_0_05
%bottom_v = OpCompositeConstruct %v4float %x %y_minus %z %w

%out_pos = OpAccessChain %ptr_out_v4float %gl_out %int_0

OpStore %out_pos %top_v
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos %right_v
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos %left_v
OpStore %outColor %color
OpEmitVertex

OpStore %out_pos %bottom_v
OpStore %outColor %color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
