#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

bool existsInAll(const vector<string>& strings, const string& candidate) {
    for (const auto& s : strings) {
        if (s.find(candidate) == string::npos) {
            return false;
        }
    }
    return true;
}

string findLCS(const vector<string>& strings) {
    if (strings.empty()) return "";
    if (strings.size() == 1) return strings[0];

    // Find the shortest string (limits maximum LCS length)
    size_t minLen = strings[0].size();
    int minIdx = 0;
    for (size_t i = 1; i < strings.size(); i++) {
        if (strings[i].size() < minLen) {
            minLen = strings[i].size();
            minIdx = i;
        }
    }

    const string& refString = strings[minIdx];
    string best = "";

    // Binary search on the length of the result
    // This reduces iterations from O(n) to O(log n)
    size_t left = 0, right = minLen;
    while (left <= right) {
        size_t mid = (left + right) / 2;
        bool found = false;

        // Try all substrings of length mid in the reference string
        for (size_t start = 0; start + mid <= minLen; start++) {
            string candidate = refString.substr(start, mid);
            if (existsInAll(strings, candidate)) {
                best = candidate;
                found = true;
                break;  // Found one valid substring of this length
            }
        }

        if (found) {
            // Try longer substrings
            left = mid + 1;
        } else {
            // Try shorter substrings
            right = (mid == 0) ? 0 : mid - 1;
        }
    }

    return best;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }

    string result = findLCS(strings);
    cout << result << endl;

    return 0;
}
