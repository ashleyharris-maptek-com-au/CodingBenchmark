using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    static void Main()
    {
        var line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        var adj = new List<HashSet<int>>(n);
        for (int i = 0; i < n; i++)
            adj.Add(new HashSet<int>());

        for (int i = 0; i < m; i++)
        {
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            adj[u].Add(v);
            adj[v].Add(u);
        }

        var cts = new CancellationTokenSource(TimeSpan.FromSeconds(270));
        List<int> bestClique = new();
        object lockObj = new();

        int numStarts = n < 10000 ? 200 : (n < 100000 ? 100 : 50);
        numStarts = Math.Min(numStarts, n);
        var tasks = new Task[numStarts];

        for (int i = 0; i < numStarts; i++)
        {
            int start = i;
            tasks[i] = Task.Run(() =>
            {
                var clique = GreedyClique(adj, start, cts.Token);
                lock (lockObj)
                {
                    if (clique.Count > bestClique.Count)
                        bestClique = clique;
                }
            }, cts.Token);
        }

        try
        {
            Task.WaitAll(tasks, cts.Token);
        }
        catch { }

        Console.WriteLine(bestClique.Count);
        if (bestClique.Count > 0)
        {
            Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
        }
    }

    static List<int> GreedyClique(List<HashSet<int>> adj, int start, CancellationToken ct)
    {
        var clique = new List<int> { start };
        var candidates = new HashSet<int>(adj[start]);

        while (candidates.Count > 0 && !ct.IsCancellationRequested)
        {
            int best = -1;
            int bestScore = -1;

            foreach (var v in candidates)
            {
                int score = 0;
                foreach (var u in candidates)
                {
                    if (u != v && adj[v].Contains(u))
                        score++;
                }

                if (score > bestScore)
                {
                    bestScore = score;
                    best = v;
                }
            }

            clique.Add(best);
            candidates.RemoveWhere(v => v == best || !adj[best].Contains(v));
        }

        return clique;
    }
}
