OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Function_float = OpTypePointer Function %float
%ptr_Function_vec3 = OpTypePointer Function %vec3
%ptr_Function_vec4 = OpTypePointer Function %vec4

%ubo = OpVariable %ptr_Uniform_UBO Uniform
%inWorldPos = OpVariable %ptr_Input_vec3 Input
%inNormal = OpVariable %ptr_Input_vec3 Input
%inUV = OpVariable %ptr_Input_vec2 Input
%inTangent = OpVariable %ptr_Input_vec3 Input
%inColor = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output

%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_1 = OpConstant %float 1.0
%float_32 = OpConstant %float 32.0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%vec3_ambient = OpConstantComposite %vec3 %float_0_1 %float_0_05 %float_0_05
%vec3_specular = OpConstantComposite %vec3 %float_1 %float_1 %float_1

%fn_void = OpTypeFunction %void

%main = OpFunction %void None %fn_void
%entry = OpLabel

%worldPos = OpVariable %ptr_Function_vec3 Function
%normal = OpVariable %ptr_Function_vec3 Function
%vertexColor = OpVariable %ptr_Function_vec3 Function
%lightPos = OpVariable %ptr_Function_vec3 Function
%cameraPos = OpVariable %ptr_Function_vec3 Function
%N = OpVariable %ptr_Function_vec3 Function
%L = OpVariable %ptr_Function_vec3 Function
%V = OpVariable %ptr_Function_vec3 Function
%R = OpVariable %ptr_Function_vec3 Function
%negL = OpVariable %ptr_Function_vec3 Function
%diffuseFactor = OpVariable %ptr_Function_float Function
%specularFactor = OpVariable %ptr_Function_float Function
%specularPower = OpVariable %ptr_Function_float Function
%ambientComponent = OpVariable %ptr_Function_vec3 Function
%diffuseComponent = OpVariable %ptr_Function_vec3 Function
%specularComponent = OpVariable %ptr_Function_vec3 Function
%sum = OpVariable %ptr_Function_vec3 Function
%clamped = OpVariable %ptr_Function_vec3 Function
%result = OpVariable %ptr_Function_vec4 Function

; Load inputs
%wp = OpLoad %vec3 %inWorldPos
OpStore %worldPos %wp
%n = OpLoad %vec3 %inNormal
OpStore %normal %n
%vc = OpLoad %vec3 %inColor
OpStore %vertexColor %vc

; Load light and camera from UBO
%ptr_lp = OpAccessChain %ptr_Uniform_vec4 %ubo %int_3
%lpv = OpLoad %vec4 %ptr_lp
%lp = OpVectorShuffle %vec3 %lpv %lpv 0 1 2
OpStore %lightPos %lp

%ptr_cp = OpAccessChain %ptr_Uniform_vec4 %ubo %int_4
%cpv = OpLoad %vec4 %ptr_cp
%cp = OpVectorShuffle %vec3 %cpv %cpv 0 1 2
OpStore %cameraPos %cp

; Normalize normal
%loaded_normal = OpLoad %vec3 %normal
%normalized_normal = OpExtInst %vec3 %ext GLSLStd450 Normalize %loaded_normal
OpStore %N %normalized_normal

; Calculate L = normalize(lightPos - worldPos)
%loaded_lp = OpLoad %vec3 %lightPos
%loaded_wp = OpLoad %vec3 %worldPos
%L_diff = OpFSub %vec3 %loaded_lp %loaded_wp
%L_norm = OpExtInst %vec3 %ext GLSLStd450 Normalize %L_diff
OpStore %L %L_norm

; Calculate V = normalize(cameraPos - worldPos)
%loaded_cp = OpLoad %vec3 %cameraPos
%V_diff = OpFSub %vec3 %loaded_cp %loaded_wp
%V_norm = OpExtInst %vec3 %ext GLSLStd450 Normalize %V_diff
OpStore %V %V_norm

; Calculate R = reflect(-L, N)
%loaded_L = OpLoad %vec3 %L
%neg_L = OpFNegate %vec3 %loaded_L
OpStore %negL %neg_L
%loaded_N = OpLoad %vec3 %N
%loaded_negL = OpLoad %vec3 %negL
%R_val = OpExtInst %vec3 %ext GLSLStd450 Reflect %loaded_negL %loaded_N
OpStore %R %R_val

; Ambient component = (0.1, 0.05, 0.05)
OpStore %ambientComponent %vec3_ambient

; Diffuse: vertex_color * max(dot(N,L), 0)
%loaded_N2 = OpLoad %vec3 %N
%loaded_L2 = OpLoad %vec3 %L
%dot_NL = OpDot %float %loaded_N2 %loaded_L2
%diff_factor = OpExtInst %float %ext GLSLStd450 FMax %dot_NL %float_0
OpStore %diffuseFactor %diff_factor
%loaded_vc = OpLoad %vec3 %vertexColor
%loaded_df = OpLoad %float %diffuseFactor
%diff_comp = OpVectorTimesScalar %vec3 %loaded_vc %loaded_df
OpStore %diffuseComponent %diff_comp

; Specular: (1,1,1) * max(dot(R,V), 0)^32
%loaded_R = OpLoad %vec3 %R
%loaded_V = OpLoad %vec3 %V
%dot_RV = OpDot %float %loaded_R %loaded_V
%spec_factor = OpExtInst %float %ext GLSLStd450 FMax %dot_RV %float_0
OpStore %specularFactor %spec_factor
%loaded_sf = OpLoad %float %specularFactor
%spec_pow = OpExtInst %float %ext GLSLStd450 Pow %loaded_sf %float_32
OpStore %specularPower %spec_pow
%loaded_sp = OpLoad %float %specularPower
%spec_comp = OpVectorTimesScalar %vec3 %vec3_specular %loaded_sp
OpStore %specularComponent %spec_comp

; Sum = ambient + diffuse + specular
%loaded_amb = OpLoad %vec3 %ambientComponent
%loaded_diff = OpLoad %vec3 %diffuseComponent
%sum1 = OpFAdd %vec3 %loaded_amb %loaded_diff
%loaded_spec = OpLoad %vec3 %specularComponent
%sum_all = OpFAdd %vec3 %sum1 %loaded_spec
OpStore %sum %sum_all

; Clamp to [0,1]
%loaded_sum = OpLoad %vec3 %sum
%clamped_val = OpExtInst %vec3 %ext GLSLStd450 FClamp %loaded_sum %float_0 %float_1
OpStore %clamped %clamped_val

; Create output vec4 with alpha 1.0
%loaded_clamped = OpLoad %vec3 %clamped
%final = OpCompositeConstruct %vec4 %loaded_clamped %float_1
OpStore %result %final

; Store to output
%loaded_result = OpLoad %vec4 %result
OpStore %outColor %loaded_result

OpReturn
%end = OpFunctionEnd
