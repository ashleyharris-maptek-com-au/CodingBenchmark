; SPIR-V
; Version: 1.0
; Generator: Manual
; Bound: 200
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations for UBO
OpDecorate %ubo_struct Block
OpMemberDecorate %ubo_struct 0 ColMajor
OpMemberDecorate %ubo_struct 0 MatrixStride 16
OpMemberDecorate %ubo_struct 1 ColMajor
OpMemberDecorate %ubo_struct 1 MatrixStride 16
OpMemberDecorate %ubo_struct 2 ColMajor
OpMemberDecorate %ubo_struct 2 MatrixStride 16
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
OpDecorate %ubo_buffer DescriptorSet 0
OpDecorate %ubo_buffer Binding 0

; Input/Output decorations
OpDecorate %in_uv Location 2
OpDecorate %out_color Location 0

; Type definitions
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; UBO struct and pointer
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ubo_ptr_type = OpTypePointer Uniform %ubo_struct
%ubo_buffer = OpVariable %ubo_ptr_type Uniform

; Input/Output pointers and variables
%in_ptr_vec2 = OpTypePointer Input %vec2
%out_ptr_vec4 = OpTypePointer Output %vec4
%in_uv = OpVariable %in_ptr_vec2 Input
%out_color = OpVariable %out_ptr_vec4 Output

; Function types and constants
%main_func_type = OpTypeFunction %void
%const_float_0 = OpConstant %float 0.0
%const_float_1 = OpConstant %float 1.0
%const_float_2 = OpConstant %float 2.0
%const_float_3 = OpConstant %float 3.0
%const_float_4 = OpConstant %float 4.0
%const_float_20 = OpConstant %float 20.0
%const_float_0_5 = OpConstant %float 0.5
%const_int_0 = OpConstant %int 0
%const_int_1 = OpConstant %int 1
%const_int_20 = OpConstant %int 20

; Main function
%main = OpFunction %void None %main_func_type
%entry = OpLabel

; Load UV coordinates
%uv = OpLoad %vec2 %in_uv
%uv_x = OpCompositeExtract %float %uv 0
%uv_y = OpCompositeExtract %float %uv 1

; Map to Mandelbrot plane: c = (uv.x*3-2, uv.y*2-1)
%c_real_1 = OpFMul %float %uv_x %const_float_3
%c_real = OpFSub %float %c_real_1 %const_float_2

%c_imag_1 = OpFMul %float %uv_y %const_float_2
%c_imag = OpFSub %float %c_imag_1 %const_float_1

; Initialize z = 0, iteration = 0
%z_real_init = OpConstant %float 0.0
%z_imag_init = OpConstant %float 0.0
%iter_init = OpConstant %int 0

; Loop setup
OpBranch %loop_header

%loop_header = OpLabel
%z_real = OpPhi %float %z_real_init %entry %z_real_new %loop_body
%z_imag = OpPhi %float %z_imag_init %entry %z_imag_new %loop_body
%iter = OpPhi %int %iter_init %entry %iter_inc %loop_body

%iter_cmp = OpSLessThan %bool %iter %const_int_20
OpLoopMerge %loop_exit %loop_body None
OpBranchConditional %iter_cmp %loop_body %loop_exit

%loop_body = OpLabel

; Check escape condition: |z|^2 > 4
%z_real_sq = OpFMul %float %z_real %z_real
%z_imag_sq = OpFMul %float %z_imag %z_imag
%mag_sq = OpFAdd %float %z_real_sq %z_imag_sq
%escape = OpFOrdGreaterThan %bool %mag_sq %const_float_4

OpSelectionMerge %escape_check None
OpBranchConditional %escape %loop_exit %escape_check

%escape_check = OpLabel

; Mandelbrot iteration: z = z^2 + c
; z_real_new = z_real^2 - z_imag^2 + c_real
%z_real_sq_2 = OpFMul %float %z_real %z_real
%z_imag_sq_2 = OpFMul %float %z_imag %z_imag
%z_real_part = OpFSub %float %z_real_sq_2 %z_imag_sq_2
%z_real_new = OpFAdd %float %z_real_part %c_real

; z_imag_new = 2*z_real*z_imag + c_imag
%z_cross = OpFMul %float %z_real %z_imag
%z_cross_2 = OpFMul %float %z_cross %const_float_2
%z_imag_new = OpFAdd %float %z_cross_2 %c_imag

; Increment iteration counter
%iter_inc = OpIAdd %int %iter %const_int_1

OpBranch %loop_header

%loop_exit = OpLabel

; Convert iteration count to normalized value for coloring
%iter_float = OpConvertSToF %float %iter
%t = OpFDiv %float %iter_float %const_float_20

; Color calculation: (t, t*0.5, 1.0-t, 1.0)
%color_r = %t
%color_g = OpFMul %float %t %const_float_0_5
%color_b = OpFSub %float %const_float_1 %t
%color_a = OpConstant %float 1.0

; Construct output color
%out_vec = OpCompositeConstruct %vec4 %color_r %color_g %color_b %color_a

; Write output
OpStore %out_color %out_vec

OpReturn
OpFunctionEnd
