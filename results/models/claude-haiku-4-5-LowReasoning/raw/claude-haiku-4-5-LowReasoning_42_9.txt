OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0.0
%float_025 = OpConstant %float 0.025
%float_neg025 = OpConstant %float -0.025

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %int_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %int_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float

%func_type = OpTypeFunction %void
%glsl = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input positions
%pos0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%pos2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos3_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %pos3_ptr

; Load colors
%color1_ptr = OpAccessChain %ptr_in_v3float %color %int_1
%color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_in_v3float %color %int_2
%color2 = OpLoad %v3float %color2_ptr

; Extract xyz components for tangent calculation
%pos0_xyz = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
%pos1_xyz = OpVectorShuffle %v3float %pos1 %pos1 0 1 2
%pos2_xyz = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
%pos3_xyz = OpVectorShuffle %v3float %pos3 %pos3 0 1 2

; Compute tangent at start: t0 = normalize(pos2 - pos0)
%diff0 = OpFSub %v3float %pos2_xyz %pos0_xyz
%t0 = OpExtInst %v3float %glsl Normalize %diff0

; Compute tangent at end: t1 = normalize(pos3 - pos1)
%diff1 = OpFSub %v3float %pos3_xyz %pos1_xyz
%t1 = OpExtInst %v3float %glsl Normalize %diff1

; Extract 2D tangents (xy components)
%t0_2d = OpVectorShuffle %v2float %t0 %t0 0 1
%t1_2d = OpVectorShuffle %v2float %t1 %t1 0 1

; Create perpendiculars: perp = (-t.y, t.x)
%t0_x = OpCompositeExtract %float %t0_2d 0
%t0_y = OpCompositeExtract %float %t0_2d 1
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%t1_x = OpCompositeExtract %float %t1_2d 0
%t1_y = OpCompositeExtract %float %t1_2d 1
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Get start and end XY positions
%pos1_2d = OpVectorShuffle %v2float %pos1_xyz %pos1_xyz 0 1
%pos2_2d = OpVectorShuffle %v2float %pos2_xyz %pos2_xyz 0 1

; Extract z and w from start and end positions
%pos1_z = OpCompositeExtract %float %pos1 2
%pos1_w = OpCompositeExtract %float %pos1 3
%pos2_z = OpCompositeExtract %float %pos2 2
%pos2_w = OpCompositeExtract %float %pos2 3

; Scale perpendiculars by thickness
%perp0_thick = OpVectorTimesScalar %v2float %perp0 %float_025
%perp0_neg_thick = OpVectorTimesScalar %v2float %perp0 %float_neg025
%perp1_thick = OpVectorTimesScalar %v2float %perp1 %float_025
%perp1_neg_thick = OpVectorTimesScalar %v2float %perp1 %float_neg025

; Vertex 1: start + perp0*thickness
%v1_2d = OpFAdd %v2float %pos1_2d %perp0_thick
%v1_x = OpCompositeExtract %float %v1_2d 0
%v1_y = OpCompositeExtract %float %v1_2d 1
%v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %pos1_z %pos1_w
%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos_ptr %v1_pos
OpStore %outColor %color1
OpEmitVertex

; Vertex 2: start - perp0*thickness
%v2_2d = OpFAdd %v2float %pos1_2d %perp0_neg_thick
%v2_x = OpCompositeExtract %float %v2_2d 0
%v2_y = OpCompositeExtract %float %v2_2d 1
%v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %pos1_z %pos1_w
OpStore %out_pos_ptr %v2_pos
OpStore %outColor %color1
OpEmitVertex

; Vertex 3: end + perp1*thickness
%v3_2d = OpFAdd %v2float %pos2_2d %perp1_thick
%v3_x = OpCompositeExtract %float %v3_2d 0
%v3_y = OpCompositeExtract %float %v3_2d 1
%v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %pos2_z %pos2_w
OpStore %out_pos_ptr %v3_pos
OpStore %outColor %color2
OpEmitVertex

; Vertex 4: end - perp1*thickness
%v4_2d = OpFAdd %v2float %pos2_2d %perp1_neg_thick
%v4_x = OpCompositeExtract %float %v4_2d 0
%v4_y = OpCompositeExtract %float %v4_2d 1
%v4_pos = OpCompositeConstruct %v4float %v4_x %v4_y %pos2_z %pos2_w
OpStore %out_pos_ptr %v4_pos
OpStore %outColor %color2
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
