#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int num_vertices, num_edges, num_colors;
    cin >> num_vertices >> num_edges >> num_colors;

    // Build adjacency list
    vector<vector<int>> adj(num_vertices);
    for (int i = 0; i < num_edges; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> color(num_vertices, -1);
    vector<bool> usedColors(num_colors);

    // Adaptive strategy: Welsh-Powell for medium graphs, simple greedy for very large
    if (num_vertices <= 100000) {
        // Welsh-Powell: Order vertices by degree (descending) for better heuristic
        vector<int> order(num_vertices);
        for (int i = 0; i < num_vertices; i++) {
            order[i] = i;
        }

        // Sort by degree (descending)
        sort(order.begin(), order.end(), [&](int a, int b) {
            return adj[a].size() > adj[b].size();
        });

        // Greedy coloring with sorted order
        for (int v : order) {
            fill(usedColors.begin(), usedColors.end(), false);
            for (int u : adj[v]) {
                if (color[u] != -1) {
                    usedColors[color[u]] = true;
                }
            }

            for (int c = 0; c < num_colors; c++) {
                if (!usedColors[c]) {
                    color[v] = c;
                    break;
                }
            }
        }
    } else {
        // Simple greedy for very large graphs (faster, no sorting overhead)
        for (int v = 0; v < num_vertices; v++) {
            fill(usedColors.begin(), usedColors.end(), false);
            for (int u : adj[v]) {
                if (color[u] != -1) {
                    usedColors[color[u]] = true;
                }
            }

            for (int c = 0; c < num_colors; c++) {
                if (!usedColors[c]) {
                    color[v] = c;
                    break;
                }
            }
        }
    }

    // Output coloring
    for (int i = 0; i < num_vertices; i++) {
        if (i > 0) cout << " ";
        cout << color[i];
    }
    cout << "\n";

    return 0;
}
