#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // Create 8x8 checkerboard pattern
    int gridX = int(uv.x * 8.0);
    int gridY = int(uv.y * 8.0);
    int parity = (gridX + gridY) & 1;

    // Alternate between 0.9 and 0.2 based on parity
    float checkerColor = (parity == 0) ? 0.9 : 0.2;

    // Normalize the normal
    vec3 N = normalize(normal);

    // Calculate light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Lambertian diffuse shading
    float diffuse = max(dot(N, L), 0.0);

    // Apply shading factor: (0.5 + 0.5*diffuse)
    float shadingFactor = 0.5 + 0.5 * diffuse;

    // Combine checkerboard with shading
    float finalColor = checkerColor * shadingFactor;

    // Output with alpha 1.0
    outColor = vec4(finalColor, finalColor, finalColor, 1.0);
}
