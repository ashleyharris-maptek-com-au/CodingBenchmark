declare const require: any;
const fs = require('fs');

interface Point {
  x: number;
  y: number;
}

interface Package extends Point {
  id: number;
}

interface Input {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}

interface Output {
  routes: number[][];
}

function distance(p1: Point, p2: Point): number {
  const dx = p1.x - p2.x;
  const dy = p1.y - p2.y;
  return Math.sqrt(dx * dx + dy * dy);
}

function solve(input: Input): Output {
  const { depot, truckCount, packages } = input;
  const depotPoint: Point = { x: depot[0], y: depot[1] };

  if (packages.length === 0) return { routes: [] };

  const k = Math.min(truckCount, packages.length);
  const n = packages.length;

  // Initialize centroids with random packages
  const centroids: Point[] = [];
  const used = new Set<number>();
  for (let i = 0; i < k; i++) {
    let idx = Math.floor(Math.random() * n);
    while (used.has(idx) && used.size < n) {
      idx = Math.floor(Math.random() * n);
    }
    used.add(idx);
    centroids.push({ x: packages[idx].x, y: packages[idx].y });
  }

  let assignment: number[] = Array(n).fill(0);

  // K-means clustering iterations
  for (let iter = 0; iter < 10; iter++) {
    // Assign each package to nearest centroid
    for (let i = 0; i < n; i++) {
      let nearest = 0;
      let minDist = distance(packages[i], centroids[0]);

      for (let j = 1; j < k; j++) {
        const d = distance(packages[i], centroids[j]);
        if (d < minDist) {
          minDist = d;
          nearest = j;
        }
      }
      assignment[i] = nearest;
    }

    // Update centroid positions
    for (let j = 0; j < k; j++) {
      let sumX = 0, sumY = 0, count = 0;
      for (let i = 0; i < n; i++) {
        if (assignment[i] === j) {
          sumX += packages[i].x;
          sumY += packages[i].y;
          count++;
        }
      }
      if (count > 0) {
        centroids[j] = { x: sumX / count, y: sumY / count };
      }
    }
  }

  // Build clusters from assignments
  const clusters: number[][] = Array(k).fill(null).map(() => []);
  for (let i = 0; i < n; i++) {
    clusters[assignment[i]].push(i);
  }

  // Create routes using nearest neighbor within each cluster
  const routes: number[][] = [];
  for (const cluster of clusters) {
    if (cluster.length === 0) continue;

    const route: number[] = [];
    const remaining = new Set(cluster);
    let current = depotPoint;

    while (remaining.size > 0) {
      let nearest = -1;
      let minDist = Infinity;

      for (const idx of remaining) {
        const d = distance(current, packages[idx]);
        if (d < minDist) {
          minDist = d;
          nearest = idx;
        }
      }

      route.push(packages[nearest].id);
      remaining.delete(nearest);
      current = packages[nearest];
    }

    routes.push(route);
  }

  return { routes };
}

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);
const result: Output = solve(data);
