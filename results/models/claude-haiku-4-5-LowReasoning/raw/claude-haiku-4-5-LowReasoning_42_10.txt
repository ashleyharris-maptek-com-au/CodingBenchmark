OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%arr4_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr4_gl_PerVertex = OpTypePointer Input %arr4_gl_PerVertex_in
%ptr_Input_v4float = OpTypePointer Input %v4float

%arr4_v3float = OpTypeArray %v3float %uint_4
%ptr_Input_arr4_v3 = OpTypePointer Input %arr4_v3float
%ptr_Input_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%ptr_Function_v3float = OpTypePointer Function %v3float
%ptr_Function_v4float = OpTypePointer Function %v4float
%ptr_Function_float = OpTypePointer Function %float
%ptr_Function_int = OpTypePointer Function %int

%func_void = OpTypeFunction %void

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_3 = OpConstant %float 3.0
%float_4 = OpConstant %float 4.0
%float_5 = OpConstant %float 5.0
%float_8 = OpConstant %float 8.0
%float_9 = OpConstant %float 9.0
%float_10 = OpConstant %float 10.0
%float_0_02 = OpConstant %float 0.02
%float_0_5 = OpConstant %float 0.5

%gl_in = OpVariable %ptr_Input_arr4_gl_PerVertex Input
%worldPos = OpVariable %ptr_Input_arr4_v3 Input
%color = OpVariable %ptr_Input_arr4_v3 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output
%outColor = OpVariable %ptr_Output_v3float Output

%glsl_450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func_void
%entry = OpLabel

%P0 = OpVariable %ptr_Function_v3float Function
%P1 = OpVariable %ptr_Function_v3float Function
%P2 = OpVariable %ptr_Function_v3float Function
%P3 = OpVariable %ptr_Function_v3float Function
%pos = OpVariable %ptr_Function_v3float Function
%tang = OpVariable %ptr_Function_v3float Function
%perp = OpVariable %ptr_Function_v3float Function
%i = OpVariable %ptr_Function_int Function
%t = OpVariable %ptr_Function_float Function
%t2 = OpVariable %ptr_Function_float Function
%t3 = OpVariable %ptr_Function_float Function
%col_interp = OpVariable %ptr_Function_v3float Function

%ac0 = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%P0_v4 = OpLoad %v4float %ac0
%P0_v3 = OpVectorShuffle %v3float %P0_v4 %P0_v4 0 1 2
OpStore %P0 %P0_v3

%ac1 = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%P1_v4 = OpLoad %v4float %ac1
%P1_v3 = OpVectorShuffle %v3float %P1_v4 %P1_v4 0 1 2
OpStore %P1 %P1_v3

%ac2 = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%P2_v4 = OpLoad %v4float %ac2
%P2_v3 = OpVectorShuffle %v3float %P2_v4 %P2_v4 0 1 2
OpStore %P2 %P2_v3

%ac3 = OpAccessChain %ptr_Input_v4float %gl_in %int_3 %int_0
%P3_v4 = OpLoad %v4float %ac3
%P3_v3 = OpVectorShuffle %v3float %P3_v4 %P3_v4 0 1 2
OpStore %P3 %P3_v3

OpStore %i %int_0
%loop_header = OpLabel
%i_val = OpLoad %int %i
%cond = OpSLessThan %bool %i_val %int_5
OpLoopMerge %loop_end %loop_cont None
OpBranchConditional %cond %loop_body %loop_end

%loop_body = OpLabel

%i_f = OpConvertSToF %float %i_val
%t_val = OpFDiv %float %i_f %float_4
OpStore %t %t_val

%t_load = OpLoad %float %t
%t2_val = OpFMul %float %t_load %t_load
OpStore %t2 %t2_val

%t3_val = OpFMul %float %t2_val %t_load
OpStore %t3 %t3_val

%P0_load = OpLoad %v3float %P0
%P1_load = OpLoad %v3float %P1
%P2_load = OpLoad %v3float %P2
%P3_load = OpLoad %v3float %P3

%P1_2 = OpVectorTimesScalar %v3float %P1_load %float_2

%neg_P0 = OpFNegate %v3float %P0_load
%sum_P0_P2 = OpFAdd %v3float %neg_P0 %P2_load
%t_load2 = OpLoad %float %t
%term_t = OpVectorTimesScalar %v3float %sum_P0_P2 %t_load2

%P0_2x = OpVectorTimesScalar %v3float %P0_load %float_2
%P1_5x = OpVectorTimesScalar %v3float %P1_load %float_5
%P2_4x = OpVectorTimesScalar %v3float %P2_load %float_4
%neg_P3 = OpFNegate %v3float %P3_load

%d1 = OpFSub %v3float %P0_2x %P1_5x
%d2 = OpFAdd %v3float %d1 %P2_4x
%d3 = OpFAdd %v3float %d2 %neg_P3
%t2_load = OpLoad %float %t2
%term_t2 = OpVectorTimesScalar %v3float %d3 %t2_load

%P1_3x = OpVectorTimesScalar %v3float %P1_load %float_3
%P2_3x = OpVectorTimesScalar %v3float %P2_load %float_3
%P3_1x = OpVectorTimesScalar %v3float %P3_load %float_1

%d4 = OpFAdd %v3float %neg_P0 %P1_3x
%d5 = OpFSub %v3float %d4 %P2_3x
%d6 = OpFAdd %v3float %d5 %P3_1x
%t3_load = OpLoad %float %t3
%term_t3 = OpVectorTimesScalar %v3float %d6 %t3_load

%s1 = OpFAdd %v3float %P1_2 %term_t
%s2 = OpFAdd %v3float %s1 %term_t2
%s3 = OpFAdd %v3float %s2 %term_t3
%pos_result = OpVectorTimesScalar %v3float %s3 %float_0_5
OpStore %pos %pos_result

%P0_load2 = OpLoad %v3float %P0
%P2_load2 = OpLoad %v3float %P2
%tang_base = OpFAdd %v3float %neg_P0 %P2_load2

%P0_4x = OpVectorTimesScalar %v3float %P0_load2 %float_4
%P1_10x = OpVectorTimesScalar %v3float %P1_load %float_10
%P2_8x = OpVectorTimesScalar %v3float %P2_load2 %float_8
%P3_2x = OpVectorTimesScalar %v3float %P3_load %float_2

%dt1 = OpFSub %v3float %P0_4x %P1_10x
%dt2 = OpFAdd %v3float %dt1 %P2_8x
%dt3 = OpFSub %v3float %dt2 %P3_2x
%t_load3 = OpLoad %float %t
%tang_t = OpVectorTimesScalar %v3float %dt3 %t_load3

%P0_3 = OpVectorTimesScalar %v3float %P0_load2 %float_3
%neg_P0_3 = OpFNegate %v3float %P0_3
%P1_9x = OpVectorTimesScalar %v3float %P1_load %float_9
%P2_9x = OpVectorTimesScalar %v3float %P2_load2 %float_9
%P3_3x = OpVectorTimesScalar %v3float %P3_load %float_3

%dt4 = OpFAdd %v3float %neg_P0_3 %P1_9x
%dt5 = OpFSub %v3float %dt4 %P2_9x
%dt6 = OpFAdd %v3float %dt5 %P3_3x
%t2_load2 = OpLoad %float %t2
%tang_t2 = OpVectorTimesScalar %v3float %dt6 %t2_load2

%tang_s1 = OpFAdd %v3float %tang_base %tang_t
%tang_final = OpFAdd %v3float %tang_s1 %tang_t2
OpStore %tang %tang_final

%tang_load = OpLoad %v3float %tang
%tang_norm = OpExtInst %v3float %glsl_450 Normalize %tang_load

%tang_y = OpCompositeExtract %float %tang_norm 1
%tang_x = OpCompositeExtract %float %tang_norm 0

%neg_tang_y = OpFNegate %float %tang_y
%perp_unscaled = OpCompositeConstruct %v3float %neg_tang_y %tang_x %float_0
%perp_norm = OpExtInst %v3float %glsl_450 Normalize %perp_unscaled
%perp_final = OpVectorTimesScalar %v3float %perp_norm %float_0_02
OpStore %perp %perp_final

%col1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
%col2 = OpLoad %v3float %col2_ptr

%col_diff = OpFSub %v3float %col2 %col1
%t_load4 = OpLoad %float %t
%col_delta = OpVectorTimesScalar %v3float %col_diff %t_load4
%col_result = OpFAdd %v3float %col1 %col_delta
OpStore %col_interp %col_result

%pos_load = OpLoad %v3float %pos
%perp_load = OpLoad %v3float %perp
%v1_pos = OpFAdd %v3float %pos_load %perp_load
%v1_v4 = OpCompositeConstruct %v4float %v1_pos %float_1

%out_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_ptr %v1_v4

%col_load = OpLoad %v3float %col_interp
OpStore %outColor %col_load
OpEmitVertex

%v2_pos = OpFSub %v3float %pos_load %perp_load
%v2_v4 = OpCompositeConstruct %v4float %v2_pos %float_1
OpStore %out_ptr %v2_v4
OpStore %outColor %col_load
OpEmitVertex

%loop_cont = OpLabel
%i_load = OpLoad %int %i
%i_next = OpIAdd %int %i_load %int_1
OpStore %i %i_next
OpBranch %loop_header

%loop_end = OpLabel
OpEndPrimitive
OpReturn

OpFunctionEnd
