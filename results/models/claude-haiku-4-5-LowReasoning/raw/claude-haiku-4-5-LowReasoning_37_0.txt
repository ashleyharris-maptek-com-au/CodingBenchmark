using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string[] line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        List<int>[] adj = new List<int>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            string[] edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
        }

        List<int> fvs = SolveDFVS(n, adj);

        Console.WriteLine(fvs.Count);
        if (fvs.Count > 0)
            Console.WriteLine(string.Join(" ", fvs));
        else
            Console.WriteLine();
    }

    static List<int> SolveDFVS(int n, List<int>[] adj)
    {
        List<HashSet<int>> sccs = FindSCCs(n, adj);

        HashSet<int> result = new HashSet<int>();

        foreach (var scc in sccs)
        {
            if (scc.Count > 1)
            {
                List<int> fvsForScc = FindMinimalFVSForSCC(scc.ToList(), adj);
                foreach (var v in fvsForScc)
                    result.Add(v);
            }
        }

        return result.OrderBy(x => x).ToList();
    }

    static List<HashSet<int>> FindSCCs(int n, List<int>[] adj)
    {
        bool[] visited = new bool[n];
        Stack<int> finish = new Stack<int>();

        for (int i = 0; i < n; i++)
        {
            if (!visited[i])
                DFS1(i, adj, visited, finish);
        }

        List<int>[] radj = new List<int>[n];
        for (int i = 0; i < n; i++)
            radj[i] = new List<int>();

        for (int u = 0; u < n; u++)
        {
            foreach (var v in adj[u])
                radj[v].Add(u);
        }

        visited = new bool[n];
        List<HashSet<int>> sccs = new List<HashSet<int>>();

        while (finish.Count > 0)
        {
            int u = finish.Pop();
            if (!visited[u])
            {
                HashSet<int> scc = new HashSet<int>();
                DFS2(u, radj, visited, scc);
                sccs.Add(scc);
            }
        }

        return sccs;
    }

    static void DFS1(int u, List<int>[] adj, bool[] visited, Stack<int> finish)
    {
        visited[u] = true;
        foreach (var v in adj[u])
        {
            if (!visited[v])
                DFS1(v, adj, visited, finish);
        }
        finish.Push(u);
    }

    static void DFS2(int u, List<int>[] radj, bool[] visited, HashSet<int> scc)
    {
        visited[u] = true;
        scc.Add(u);
        foreach (var v in radj[u])
        {
            if (!visited[v])
                DFS2(v, radj, visited, scc);
        }
    }

    static List<int> FindMinimalFVSForSCC(List<int> scc, List<int>[] adj)
    {
        foreach (var v in scc)
        {
            if (IsAcyclicAfterRemoval(scc, new List<int> { v }, adj))
                return new List<int> { v };
        }

        if (scc.Count <= 15)
        {
            for (int i = 0; i < scc.Count; i++)
            {
                for (int j = i + 1; j < scc.Count; j++)
                {
                    var removal = new List<int> { scc[i], scc[j] };
                    if (IsAcyclicAfterRemoval(scc, removal, adj))
                        return removal;
                }
            }
        }

        return GreedyRemoval(scc, adj);
    }

    static bool IsAcyclicAfterRemoval(List<int> scc, List<int> removed, List<int>[] adj)
    {
        HashSet<int> removedSet = new HashSet<int>(removed);
        HashSet<int> sccSet = new HashSet<int>(scc);

        HashSet<int> visited = new HashSet<int>();
        HashSet<int> recStack = new HashSet<int>();

        foreach (var u in scc)
        {
            if (!removedSet.Contains(u) && !visited.Contains(u))
            {
                if (HasCycleDFS(u, adj, sccSet, removedSet, visited, recStack))
                    return false;
            }
        }

        return true;
    }

    static bool HasCycleDFS(int u, List<int>[] adj, HashSet<int> sccSet, HashSet<int> removed, HashSet<int> visited, HashSet<int> recStack)
    {
        visited.Add(u);
        recStack.Add(u);

        foreach (var v in adj[u])
        {
            if (!sccSet.Contains(v) || removed.Contains(v))
                continue;

            if (!visited.Contains(v))
            {
                if (HasCycleDFS(v, adj, sccSet, removed, visited, recStack))
                    return true;
            }
            else if (recStack.Contains(v))
            {
                return true;
            }
        }

        recStack.Remove(u);
        return false;
    }

    static List<int> GreedyRemoval(List<int> scc, List<int>[] adj)
    {
        List<int> result = new List<int>();
        HashSet<int> remaining = new HashSet<int>(scc);

        while (remaining.Count > 0)
        {
            if (IsAcyclicAfterRemoval(remaining.ToList(), new List<int>(), adj))
                break;

            int maxDegree = -1;
            int maxVertex = -1;

            foreach (var u in remaining)
            {
                int degree = 0;
                foreach (var v in adj[u])
                {
                    if (remaining.Contains(v))
                        degree++;
                }
                foreach (var v in remaining)
                {
                    if (v != u && adj[v].Contains(u))
                        degree++;
                }

                if (degree > maxDegree)
                {
                    maxDegree = degree;
                    maxVertex = u;
                }
            }

            if (maxVertex >= 0)
            {
                result.Add(maxVertex);
                remaining.Remove(maxVertex);
            }
            else
            {
                break;
            }
        }

        return result;
    }
}
