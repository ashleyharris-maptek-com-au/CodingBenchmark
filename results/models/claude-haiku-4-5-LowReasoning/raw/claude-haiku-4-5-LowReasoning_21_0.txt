#include <iostream>
#include <vector>
#include <set>
#include <cmath>

using namespace std;

int main() {
    int D, max_turns;
    cin >> D >> max_turns;

    vector<int> bounds(D);
    for (int i = 0; i < D; i++) {
        cin >> bounds[i];
    }

    int initial_length;
    cin >> initial_length;

    vector<int> head(D);
    for (int i = 0; i < D; i++) {
        cin >> head[i];
    }

    vector<vector<int>> snake;
    snake.push_back(head);

    int F;
    cin >> F;
    vector<vector<int>> food(F);
    for (int i = 0; i < F; i++) {
        food[i].resize(D);
        for (int j = 0; j < D; j++) {
            cin >> food[i][j];
        }
    }

    int O;
    cin >> O;
    set<vector<int>> obstacles;
    for (int i = 0; i < O; i++) {
        vector<int> obs(D);
        for (int j = 0; j < D; j++) {
            cin >> obs[j];
        }
        obstacles.insert(obs);
    }

    int current_length = initial_length;

    for (int turn = 0; turn < max_turns && !food.empty(); turn++) {
        // Find nearest food using Manhattan distance
        int target_idx = 0;
        double min_dist = 1e18;

        for (size_t i = 0; i < food.size(); i++) {
            double dist = 0;
            for (int j = 0; j < D; j++) {
                dist += abs(head[j] - food[i][j]);
            }
            if (dist < min_dist) {
                min_dist = dist;
                target_idx = i;
            }
        }

        vector<int> target = food[target_idx];

        // Find best valid move (greedy: closest to target food)
        int best_axis = 0;
        int best_dir = 1;
        double best_dist = 1e18;
        bool found_move = false;

        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                vector<int> new_head = head;
                new_head[axis] += dir;

                // Check bounds
                bool valid = true;
                for (int j = 0; j < D; j++) {
                    if (new_head[j] < 0 || new_head[j] >= bounds[j]) {
                        valid = false;
                        break;
                    }
                }
                if (!valid) continue;

                // Check obstacles
                if (obstacles.count(new_head)) continue;

                // Check self collision (all body except tail which moves away)
                bool collision = false;
                for (size_t i = 0; i < snake.size() - 1; i++) {
                    if (snake[i] == new_head) {
                        collision = true;
                        break;
                    }
                }
                if (collision) continue;

                found_move = true;

                // Distance to target food
                double dist = 0;
                for (int j = 0; j < D; j++) {
                    dist += abs(new_head[j] - target[j]);
                }

                if (dist < best_dist) {
                    best_dist = dist;
                    best_axis = axis;
                    best_dir = dir;
                }
            }
        }

        if (!found_move) {
            // No move towards food found, try any valid move
            for (int axis = 0; axis < D && !found_move; axis++) {
                for (int dir : {-1, 1}) {
                    vector<int> new_head = head;
                    new_head[axis] += dir;

                    bool valid = true;
                    for (int j = 0; j < D; j++) {
                        if (new_head[j] < 0 || new_head[j] >= bounds[j]) {
                            valid = false;
                            break;
                        }
                    }
                    if (!valid) continue;

                    if (obstacles.count(new_head)) continue;

                    bool collision = false;
                    for (size_t i = 0; i < snake.size() - 1; i++) {
                        if (snake[i] == new_head) {
                            collision = true;
                            break;
                        }
                    }
                    if (collision) continue;

                    best_axis = axis;
                    best_dir = dir;
                    found_move = true;
                    break;
                }
            }
        }

        if (!found_move) {
            break;
        }

        // Execute move
        vector<int> new_head = head;
        new_head[best_axis] += best_dir;

        snake.insert(snake.begin(), new_head);
        head = new_head;

        // Check if food was eaten
        for (size_t i = 0; i < food.size(); i++) {
            if (food[i] == head) {
                food.erase(food.begin() + i);
                current_length++;
                break;
            }
        }

        // Remove tail if not growing
        if ((int)snake.size() > current_length) {
            snake.pop_back();
        }

        cout << best_axis << " " << best_dir << "\n";
    }

    return 0;
}
