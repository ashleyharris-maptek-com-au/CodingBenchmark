use std::io::{self, BufRead, Write, BufReader};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Duration;
use std::f64::consts::PI;

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

fn main() {
    // Read header and map in initial scope so locks are released
    let (gravity, max_thrust, target_x, target_y) = {
        let stdin = io::stdin();
        let mut reader = BufReader::new(stdin.lock());
        let mut line = String::new();

        // Read header
        reader.read_line(&mut line).expect("Failed to read header");
        let header: Vec<f64> = line.split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();
        if header.len() < 5 {
            panic!("Invalid header");
        }
        let gravity = header[2];
        let max_thrust = header[3];

        // Read coords
        line.clear();
        reader.read_line(&mut line).expect("Failed to read coords");
        let coords: Vec<f64> = line.split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();
        if coords.len() < 4 {
            panic!("Invalid coords");
        }
        let target_x = coords[2];
        let target_y = coords[3];

        // Read and skip map
        let height = header[1] as usize;
        for _ in 0..height {
            line.clear();
            reader.read_line(&mut line).expect("Failed to read map");
        }

        (gravity, max_thrust, target_x, target_y)
    }; // Lock is released here

    let state = Arc::new(Mutex::new(None::<State>));
    let state_clone = state.clone();

    // Spawn reader thread to continuously update state
    thread::spawn(move || {
        let stdin = io::stdin();
        let reader = BufReader::new(stdin.lock());

        for line in reader.lines() {
            if let Ok(line) = line {
                let parts: Vec<f64> = line.split_whitespace()
                    .filter_map(|s| s.parse().ok())
                    .collect();

                if parts.len() >= 7 {
                    let new_state = State {
                        x: parts[0],
                        y: parts[1],
                        vx: parts[2],
                        vy: parts[3],
                        angle: parts[4],
                        angular_vel: parts[5],
                        fuel: parts[6],
                    };

                    if let Ok(mut state_lock) = state_clone.lock() {
                        *state_lock = Some(new_state);
                    }
                }
            } else {
                break;
            }
        }
    });

    // Main control loop - output at 20Hz (50ms per iteration)
    loop {
        if let Ok(state_lock) = state.lock() {
            if let Some(current_state) = *state_lock {
                let (thrust, turn) = calculate_command(
                    current_state,
                    target_x,
                    target_y,
                    gravity,
                    max_thrust,
                );

                println!("{:.2} {:.2}", thrust, turn);
                let _ = io::stdout().flush();
            }
        }

        thread::sleep(Duration::from_millis(50));
    }
}

fn calculate_command(
    state: State,
    target_x: f64,
    target_y: f64,
    _gravity: f64,
    _max_thrust: f64,
) -> (f64, f64) {
    let dx = target_x - state.x;
    let dy = target_y - state.y;
    let dist = (dx * dx + dy * dy).sqrt();
    let speed = (state.vx * state.vx + state.vy * state.vy).sqrt();

    // Thrust control based on descent rate
    let mut thrust = 0.5;

    if state.vy > 5.0 {
        // Falling very fast - full thrust
        thrust = 1.0;
    } else if state.vy > 3.0 {
        thrust = 0.8;
    } else if state.vy > 1.5 {
        thrust = 0.6;
    } else if state.vy > 0.5 {
        thrust = 0.4;
    } else if state.vy > 0.0 {
        thrust = 0.2;
    } else if speed < 2.0 && dist < 10.0 {
        // Near target with low speed - minimal thrust to avoid bouncing
        thrust = 0.1;
    } else {
        thrust = 0.3;
    }

    // Turn control to face target
    // angle = 0 is up, angle = π/2 is right, angle = π is down
    // In screen coords where y increases downward, target angle is:
    let desired_angle = dx.atan2(-dy);
    let angle_diff = normalize_angle(desired_angle - state.angle);
    let turn = (angle_diff * 0.8).clamp(-1.0, 1.0);

    (thrust, turn)
}

fn normalize_angle(mut angle: f64) -> f64 {
    while angle > PI {
        angle -= 2.0 * PI;
    }
    while angle < -PI {
        angle += 2.0 * PI;
    }
    angle
}
