use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Debug, Clone)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

#[derive(Debug, Clone)]
struct Placement {
    stock_index: usize,
    x: f64,
    y: f64,
    rotation: f64,
}

fn main() {
    let (stock, pieces) = read_input().expect("Failed to read input");

    let mut placements: Vec<Option<Placement>> = vec![None; pieces.len()];
    let mut stocks: Vec<Vec<usize>> = Vec::new();

    for piece_idx in 0..pieces.len() {
        let piece = &pieces[piece_idx];
        let mut placed = false;

        for stock_idx in 0..stocks.len() {
            if let Some(placement) = try_place_on_stock(
                &stock,
                piece,
                stock_idx,
                &pieces,
                &placements,
                &stocks,
            ) {
                placements[piece_idx] = Some(placement);
                stocks[stock_idx].push(piece_idx);
                placed = true;
                break;
            }
        }

        if !placed {
            let stock_idx = stocks.len();
            stocks.push(Vec::new());
            if let Some(placement) = try_place_on_stock(
                &stock,
                piece,
                stock_idx,
                &pieces,
                &placements,
                &stocks,
            ) {
                placements[piece_idx] = Some(placement);
                stocks[stock_idx].push(piece_idx);
            }
        }
    }

    println!("{}", stocks.len());
    for placement in placements {
        if let Some(p) = placement {
            println!("{} {} {} {}", p.stock_index, p.x, p.y, p.rotation);
        }
    }
}

fn read_input() -> io::Result<(Polygon, Vec<Polygon>)> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap()?;
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap()?;
        let parts: Vec<&str> = line.split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        stock_vertices.push(Point { x, y });
    }
    let stock = Polygon { vertices: stock_vertices };

    let mut pieces = Vec::new();
    for _ in 0..np {
        let line = lines.next().unwrap()?;
        let pv: usize = line.trim().parse().unwrap();
        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap()?;
            let parts: Vec<&str> = line.split_whitespace().collect();
            let x: f64 = parts[0].parse().unwrap();
            let y: f64 = parts[1].parse().unwrap();
            piece_vertices.push(Point { x, y });
        }
        pieces.push(Polygon { vertices: piece_vertices });
    }

    Ok((stock, pieces))
}

fn point_in_polygon(point: &Point, polygon: &Polygon) -> bool {
    let mut inside = false;
    let n = polygon.vertices.len();

    for i in 0..n {
        let p1 = &polygon.vertices[i];
        let p2 = &polygon.vertices[(i + 1) % n];

        if ((p1.y > point.y) != (p2.y > point.y)) &&
           (point.x < (p2.x - p1.x) * (point.y - p1.y) / (p2.y - p1.y) + p1.x) {
            inside = !inside;
        }
    }

    inside
}

fn polygon_in_polygon(piece: &Polygon, stock: &Polygon) -> bool {
    for vertex in &piece.vertices {
        if !point_in_polygon(vertex, stock) {
            return false;
        }
    }
    true
}

fn rotate_point(point: &Point, angle: f64) -> Point {
    let cos_a = angle.cos();
    let sin_a = angle.sin();
    Point {
        x: point.x * cos_a - point.y * sin_a,
        y: point.x * sin_a + point.y * cos_a,
    }
}

fn translate_polygon(polygon: &Polygon, dx: f64, dy: f64) -> Polygon {
    Polygon {
        vertices: polygon.vertices.iter()
            .map(|p| Point { x: p.x + dx, y: p.y + dy })
            .collect(),
    }
}

fn rotate_polygon(polygon: &Polygon, angle: f64) -> Polygon {
    Polygon {
        vertices: polygon.vertices.iter()
            .map(|p| rotate_point(p, angle))
            .collect(),
    }
}

fn get_bounds(polygon: &Polygon) -> (Point, Point) {
    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;

    for v in &polygon.vertices {
        min_x = min_x.min(v.x);
        max_x = max_x.max(v.x);
        min_y = min_y.min(v.y);
        max_y = max_y.max(v.y);
    }

    (Point { x: min_x, y: min_y }, Point { x: max_x, y: max_y })
}

fn bounding_boxes_overlap(min1: &Point, max1: &Point, min2: &Point, max2: &Point) -> bool {
    min1.x < max2.x && max1.x > min2.x && min1.y < max2.y && max1.y > min2.y
}

fn segments_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {
    fn ccw(a: &Point, b: &Point, c: &Point) -> bool {
        (c.y - a.y) * (b.x - a.x) > (b.y - a.y) * (c.x - a.x)
    }
    ccw(p1, p3, p4) != ccw(p2, p3, p4) && ccw(p1, p2, p3) != ccw(p1, p2, p4)
}

fn polygons_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    let (min1, max1) = get_bounds(poly1);
    let (min2, max2) = get_bounds(poly2);

    if !bounding_boxes_overlap(&min1, &max1, &min2, &max2) {
        return false;
    }

    let n1 = poly1.vertices.len();
    let n2 = poly2.vertices.len();

    for v in &poly1.vertices {
        if point_in_polygon(v, poly2) {
            return true;
        }
    }

    for v in &poly2.vertices {
        if point_in_polygon(v, poly1) {
            return true;
        }
    }

    for i in 0..n1 {
        for j in 0..n2 {
            if segments_intersect(
                &poly1.vertices[i],
                &poly1.vertices[(i + 1) % n1],
                &poly2.vertices[j],
                &poly2.vertices[(j + 1) % n2],
            ) {
                return true;
            }
        }
    }

    false
}

fn try_place_on_stock(
    stock: &Polygon,
    piece: &Polygon,
    stock_idx: usize,
    pieces: &Vec<Polygon>,
    placements: &Vec<Option<Placement>>,
    stocks: &Vec<Vec<usize>>,
) -> Option<Placement> {
    let num_pieces = pieces.len();
    let (min_s, max_s) = get_bounds(stock);
    let stock_width = max_s.x - min_s.x;
    let stock_height = max_s.y - min_s.y;

    let num_rotations = if num_pieces > 2000 { 4 } else if num_pieces > 500 { 8 } else { 16 };

    for rot_idx in 0..num_rotations {
        let angle = (rot_idx as f64) * 2.0 * PI / (num_rotations as f64);
        let rotated = rotate_polygon(piece, angle);
        let (min_b, max_b) = get_bounds(&rotated);
        let piece_width = max_b.x - min_b.x;
        let piece_height = max_b.y - min_b.y;

        let step = if num_pieces > 2000 { 10.0 } else if num_pieces > 500 { 5.0 } else { 2.0 };
        let mut x = min_s.x;
        while x <= max_s.x - piece_width {
            let mut y = min_s.y;
            while y <= max_s.y - piece_height {
                let translated = translate_polygon(&rotated, x - min_b.x, y - min_b.y);

                if polygon_in_polygon(&translated, stock) {
                    let mut overlaps = false;

                    if stock_idx < stocks.len() {
                        for &other_idx in &stocks[stock_idx] {
                            if let Some(other_placement) = &placements[other_idx] {
                                let other_rotated = rotate_polygon(&pieces[other_idx], other_placement.rotation);
                                let other_translated = translate_polygon(&other_rotated, other_placement.x, other_placement.y);

                                if polygons_overlap(&translated, &other_translated) {
                                    overlaps = true;
                                    break;
                                }
                            }
                        }
                    }

                    if !overlaps {
                        return Some(Placement {
                            stock_index: stock_idx,
                            x: x - min_b.x,
                            y: y - min_b.y,
                            rotation: angle,
                        });
                    }
                }

                y += step;
            }
            x += step;
        }
    }

    None
}
