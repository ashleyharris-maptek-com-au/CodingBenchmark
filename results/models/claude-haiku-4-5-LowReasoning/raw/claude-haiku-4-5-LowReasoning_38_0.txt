# Read input
line = input().split()
U = int(line[0])
S = int(line[1])

sets = []
for _ in range(S):
    elements = list(map(int, input().split()))
    sets.append(set(elements))

# Backtracking to find exact cover
def backtrack(index, covered, selected):
    # Check if all elements are covered
    if len(covered) == U:
        return selected[:]

    # Pruning: if we've considered all sets and haven't found solution
    if index == S:
        return None

    # Try including the current set if it doesn't overlap
    if not (sets[index] & covered):
        selected.append(index)
        result = backtrack(index + 1, covered | sets[index], selected)
        if result is not None:
            return result
        selected.pop()

    # Try not including the current set
    result = backtrack(index + 1, covered, selected)
    if result is not None:
        return result

    return None

result = backtrack(0, set(), [])

if result is None:
    print("NO SOLUTION")
else:
    print("SOLUTION")
    print(' '.join(map(str, result)))
