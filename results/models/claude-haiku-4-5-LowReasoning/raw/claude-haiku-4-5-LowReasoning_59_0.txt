const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { probability, drones, maxStepsPerDrone } = data;
const rows = probability.length;
const cols = probability[0].length;

const covered = new Set();
const directions = [[-1, 0], [1, 0], [0, -1], [0, 1], [0, 0]];

function isValid(x, y) {
  return x >= 0 && x < rows && y >= 0 && y < cols && probability[x][y] !== -1;
}

function findBestTargets(limit = 15) {
  const targets = [];
  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      const key = `${i},${j}`;
      if (probability[i][j] > 0 && !covered.has(key)) {
        targets.push([i, j, probability[i][j]]);
      }
    }
  }
  targets.sort((a, b) => b[2] - a[2]);
  return targets.slice(0, limit).map(t => [t[0], t[1]]);
}

function manhattan(x1, y1, x2, y2) {
  return Math.abs(x1 - x2) + Math.abs(y1 - y2);
}

function planPath(startX, startY, maxSteps) {
  const path = [[startX, startY]];
  let x = startX, y = startY;
  const visited = new Set();
  visited.add(`${x},${y}`);

  for (let step = 0; step < maxSteps; step++) {
    const targets = findBestTargets(20);
    let bestNext = null;
    let bestScore = -Infinity;

    for (const [dx, dy] of directions) {
      const nx = x + dx, ny = y + dy;
      if (isValid(nx, ny)) {
        const key = `${nx},${ny}`;
        const isCovered = covered.has(key);
        const cellValue = probability[nx][ny] || 0;

        let score = 0;
        if (!isCovered && cellValue > 0) {
          score += cellValue * 10;
        }

        if (targets.length > 0) {
          const minDist = Math.min(...targets.map(t => manhattan(nx, ny, t[0], t[1])));
          score += Math.max(0, 100 - minDist * 5);
        }

        if (score > bestScore) {
          bestScore = score;
          bestNext = [nx, ny];
        }
      }
    }

    if (bestNext && bestScore >= 0) {
      x = bestNext[0];
      y = bestNext[1];
      const key = `${x},${y}`;
      if (!visited.has(key)) {
        visited.add(key);
        path.push([x, y]);
      }
    } else {
      break;
    }
  }

  for (const [px, py] of path) {
    if (probability[px][py] > 0) {
      covered.add(`${px},${py}`);
    }
  }

  return path;
}

const paths = [];
for (const drone of drones) {
  const path = planPath(drone.x, drone.y, maxStepsPerDrone);
  paths.push(path);
}
