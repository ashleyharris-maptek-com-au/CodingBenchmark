const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { grid, robots, jobs } = data;

function bfsDistance(grid, startX, startY, endX, endY) {
  if (startX === endX && startY === endY) return 0;

  const rows = grid.length;
  const cols = grid[0].length;
  const visited = new Set();
  const queue = [[startX, startY, 0]];
  visited.add(`${startX},${startY}`);

  while (queue.length > 0) {
    const [x, y, dist] = queue.shift();

    const directions = [[0, 1], [1, 0], [0, -1], [-1, 0]];
    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < cols && ny >= 0 && ny < rows &&
          grid[ny][nx] === '.' && !visited.has(`${nx},${ny}`)) {

        if (nx === endX && ny === endY) {
          return dist + 1;
        }

        visited.add(`${nx},${ny}`);
        queue.push([nx, ny, dist + 1]);
      }
    }
  }

  return Infinity;
}

const maxRobotId = Math.max(...robots.map(r => r.id));
const assignments = new Array(maxRobotId + 1).fill(null).map(() => []);
const assignedJobs = new Set();
const robotPositions = new Map(robots.map(r => [r.id, { x: r.x, y: r.y }]));

for (const robot of robots) {
  while (true) {
    let closestJob = null;
    let closestDistance = Infinity;

    for (const job of jobs) {
      if (!assignedJobs.has(job.id)) {
        const pos = robotPositions.get(robot.id);
        const dist = bfsDistance(grid, pos.x, pos.y, job.x, job.y);

        if (dist < closestDistance || (dist === closestDistance && (closestJob === null || job.id < closestJob.id))) {
          closestDistance = dist;
          closestJob = job;
        }
      }
    }

    if (!closestJob || closestDistance === Infinity) break;

    assignedJobs.add(closestJob.id);
    assignments[robot.id].push(closestJob.id);
    robotPositions.set(robot.id, { x: closestJob.x, y: closestJob.y });
  }
}
