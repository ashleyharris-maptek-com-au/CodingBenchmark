struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal : TEXCOORD1;
    [[vk::location(2)]] float2 uv : TEXCOORD2;
    [[vk::location(3)]] float3 tangent : TEXCOORD3;
    [[vk::location(4)]] float3 color : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Scale UV coordinates for 8x16 brick pattern
    float2 scaledUV = input.uv * float2(8.0, 16.0);

    // Get the row index for staggering
    int rowIndex = int(floor(scaledUV.y));

    // Stagger: offset by 0.5 for odd rows
    if ((rowIndex % 2) == 1) {
        scaledUV.x += 0.5;
    }

    // Get fractional part within each brick cell
    float2 fracUV = frac(scaledUV);

    // Mortar thickness
    float mortarX = 0.05;
    float mortarY = 0.1;

    // Determine if this pixel is in mortar or brick
    bool inMortarX = fracUV.x < mortarX || fracUV.x > (1.0 - mortarX);
    bool inMortarY = fracUV.y < mortarY || fracUV.y > (1.0 - mortarY);
    bool inMortar = inMortarX || inMortarY;

    // Colors
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);

    // Select color based on mortar/brick
    float3 finalColor = inMortar ? mortarColor : brickColor;

    // Apply Lambertian shading
    float3 normal = normalize(input.normal);
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(normal, lightDir));
    float shadingFactor = 0.4 + 0.6 * diffuse;

    // Apply shading to color
    finalColor *= shadingFactor;

    // Return with full alpha
    return float4(finalColor, 1.0);
}
