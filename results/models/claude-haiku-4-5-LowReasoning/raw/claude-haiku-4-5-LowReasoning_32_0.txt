#include <iostream>
#include <vector>
#include <queue>
#include <set>

using namespace std;

const long long INF = 1e18;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<pair<int, long long>>> adj(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; i++) {
        cin >> terminals[i];
    }

    if (k <= 1) {
        cout << "0 0\n";
        return 0;
    }

    // Dreyfus-Wagner algorithm with edge tracking
    vector<vector<long long>> dp(1 << k, vector<long long>(n, INF));
    vector<vector<set<pair<int, int>>>> edges(1 << k, vector<set<pair<int, int>>>(n));

    // Initialize: single terminals have cost 0
    for (int i = 0; i < k; i++) {
        dp[1 << i][terminals[i]] = 0;
    }

    // Process all subsets of terminals
    for (int mask = 1; mask < (1 << k); mask++) {
        // Step 1: Combine two non-empty subsets at each vertex
        for (int v = 0; v < n; v++) {
            for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
                int comp = mask ^ sub;
                if (dp[sub][v] < INF && dp[comp][v] < INF) {
                    long long cost = dp[sub][v] + dp[comp][v];
                    if (cost < dp[mask][v]) {
                        dp[mask][v] = cost;
                        edges[mask][v].clear();
                        edges[mask][v].insert(edges[sub][v].begin(), edges[sub][v].end());
                        edges[mask][v].insert(edges[comp][v].begin(), edges[comp][v].end());
                    }
                }
            }
        }

        // Step 2: Dijkstra relaxation for this subset
        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;
        vector<bool> visited(n, false);

        for (int v = 0; v < n; v++) {
            if (dp[mask][v] < INF) {
                pq.push({dp[mask][v], v});
            }
        }

        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();

            if (visited[u]) continue;
            visited[u] = true;

            for (auto [v, w] : adj[u]) {
                if (dp[mask][u] + w < dp[mask][v]) {
                    dp[mask][v] = dp[mask][u] + w;
                    edges[mask][v] = edges[mask][u];
                    int a = min(u, v), b = max(u, v);
                    edges[mask][v].insert({a, b});
                    pq.push({dp[mask][v], v});
                }
            }
        }
    }

    // Find the minimum cost Steiner tree
    int full_mask = (1 << k) - 1;
    long long min_cost = INF;
    int best_root = -1;
    for (int v = 0; v < n; v++) {
        if (dp[full_mask][v] < min_cost) {
            min_cost = dp[full_mask][v];
            best_root = v;
        }
    }

    auto& result_edges = edges[full_mask][best_root];

    cout << min_cost << " " << result_edges.size() << "\n";
    for (auto [u, v] : result_edges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
