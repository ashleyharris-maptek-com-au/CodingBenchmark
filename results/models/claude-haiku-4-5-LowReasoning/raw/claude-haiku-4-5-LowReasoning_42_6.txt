OpCapability Geometry
OpCapability Shader

OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor_out

OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex block
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for input color array
OpDecorate %color_in Location 2

; Decorations for output gl_PerVertex block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Decorations for output color
OpDecorate %outColor_out Location 0

; Type declarations
%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

; Vector types
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Struct types
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Array types
%arr_3_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %int_3
%arr_3_v3float = OpTypeArray %v3float %int_3

; Pointer types
%ptr_Input_arr_3_gl_PerVertex_in = OpTypePointer Input %arr_3_gl_PerVertex_in
%ptr_Input_arr_3_v3float = OpTypePointer Input %arr_3_v3float
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float

%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%func_void = OpTypeFunction %void

; Variable declarations
%gl_in = OpVariable %ptr_Input_arr_3_gl_PerVertex_in Input
%color_in = OpVariable %ptr_Input_arr_3_v3float Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor_out = OpVariable %ptr_Output_v3float Output

; Function definition
%main = OpFunction %void None %func_void
%entry = OpLabel

; Get pointer to gl_out.gl_Position (reused for all iterations)
%gl_out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

; Iteration 0: i = 0
%gl_in_0_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%gl_in_0_pos = OpLoad %v4float %gl_in_0_pos_ptr
OpStore %gl_out_pos_ptr %gl_in_0_pos

%color_0_ptr = OpAccessChain %ptr_Input_v3float %color_in %int_0
%color_0 = OpLoad %v3float %color_0_ptr
OpStore %outColor_out %color_0

OpEmitVertex

; Iteration 1: i = 1
%gl_in_1_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%gl_in_1_pos = OpLoad %v4float %gl_in_1_pos_ptr
OpStore %gl_out_pos_ptr %gl_in_1_pos

%color_1_ptr = OpAccessChain %ptr_Input_v3float %color_in %int_1
%color_1 = OpLoad %v3float %color_1_ptr
OpStore %outColor_out %color_1

OpEmitVertex

; Iteration 2: i = 2
%gl_in_2_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%gl_in_2_pos = OpLoad %v4float %gl_in_2_pos_ptr
OpStore %gl_out_pos_ptr %gl_in_2_pos

%color_2_ptr = OpAccessChain %ptr_Input_v3float %color_in %int_2
%color_2 = OpLoad %v3float %color_2_ptr
OpStore %outColor_out %color_2

OpEmitVertex

OpEndPrimitive
OpReturn

OpFunctionEnd
