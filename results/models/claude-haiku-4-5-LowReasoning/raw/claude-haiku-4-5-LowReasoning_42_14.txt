OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpExtInstImport %glsl "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor_var %worldPos_var %normal_var %color_var
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos_var Location 0
OpDecorate %normal_var Location 1
OpDecorate %color_var Location 2
OpDecorate %outColor_var Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_6 = OpConstant %uint 6
%uint_0 = OpConstant %uint 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0p2 = OpConstant %float 0.2
%float_0p8 = OpConstant %float 0.8
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0p333333 = OpConstant %float 0.333333

%v1_1_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_6_gl_PerVertex = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_Input_arr = OpTypePointer Input %arr_6_gl_PerVertex
%gl_in = OpVariable %ptr_Input_arr Input

%arr_6_v3 = OpTypeArray %v3float %uint_6
%ptr_Input_arr_v3 = OpTypePointer Input %arr_6_v3
%worldPos_var = OpVariable %ptr_Input_arr_v3 Input
%normal_var = OpVariable %ptr_Input_arr_v3 Input
%color_var = OpVariable %ptr_Input_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output

%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor_var = OpVariable %ptr_Output_v3 Output

%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float

%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%entry_block = OpLabel

; Load main triangle vertices (indices 0, 2, 4)
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p4_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr

; Extract XYZ components
%v0 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%v2 = OpVectorShuffle %v3float %p2 %p2 0 1 2
%v4 = OpVectorShuffle %v3float %p4 %p4 0 1 2

; Compute edge vectors: p2-p0 and p4-p0
%edge1 = OpFSub %v3float %v2 %v0
%edge2 = OpFSub %v3float %v4 %v0

; Compute cross product: cross(edge1, edge2)
%cross = OpExtInst %v3float %glsl Cross %edge1 %edge2

; Normalize to get face normal
%face_normal = OpExtInst %v3float %glsl Normalize %cross

; Normalize light direction
%light_norm = OpExtInst %v3float %glsl Normalize %v1_1_1

; Compute diffuse: max(dot(N, L), 0)
%dot_val = OpDot %float %face_normal %light_norm
%diffuse = OpExtInst %float %glsl FMax %dot_val %float_0

; Load colors and compute average
%c0_ptr = OpAccessChain %ptr_Input_v3 %color_var %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_Input_v3 %color_var %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_Input_v3 %color_var %int_4
%c4 = OpLoad %v3float %c4_ptr

%sum1 = OpFAdd %v3float %c0 %c2
%sum2 = OpFAdd %v3float %sum1 %c4
%base_color = OpVectorTimesScalar %v3float %sum2 %float_0p333333

; Compute shading factor: 0.2 + 0.8 * diffuse
%diffuse_scaled = OpFMul %float %float_0p8 %diffuse
%shade_factor = OpFAdd %float %float_0p2 %diffuse_scaled

; Apply shading to base color
%final_color = OpVectorTimesScalar %v3float %base_color %shade_factor

; Emit vertex 0
%out_pos_ptr_0 = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_pos_ptr_0 %p0
OpStore %outColor_var %final_color
OpEmitVertex

; Emit vertex 2
%out_pos_ptr_2 = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_pos_ptr_2 %p2
OpStore %outColor_var %final_color
OpEmitVertex

; Emit vertex 4
%out_pos_ptr_4 = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_pos_ptr_4 %p4
OpStore %outColor_var %final_color
OpEmitVertex

; End the triangle strip
OpEndPrimitive

OpReturn
OpFunctionEnd
