#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <numeric>
#include <limits>
#include <cstring>

using namespace std;

struct City {
    double x, y;
    int id;
};

inline double dist_sq(const City& a, const City& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// Nearest neighbor for small-medium problems
vector<int> nearestNeighbor(const vector<City>& cities) {
    int n = cities.size();
    vector<bool> visited(n, false);
    vector<int> tour;
    tour.push_back(0);
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        int last = tour.back();
        int nearest = -1;
        double minDist = numeric_limits<double>::max();

        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                double d = dist_sq(cities[last], cities[j]);
                if (d < minDist) {
                    minDist = d;
                    nearest = j;
                }
            }
        }

        tour.push_back(nearest);
        visited[nearest] = true;
    }

    return tour;
}

// Grid-based approach for large problems
vector<int> gridBasedTour(const vector<City>& cities) {
    int n = cities.size();

    // Find bounds
    double minX = cities[0].x, maxX = cities[0].x;
    double minY = cities[0].y, maxY = cities[0].y;
    for (const auto& c : cities) {
        minX = min(minX, c.x);
        maxX = max(maxX, c.x);
        minY = min(minY, c.y);
        maxY = max(maxY, c.y);
    }

    // Adaptive grid size
    int gridSize = max(1, (int)sqrt(n / 100));
    double cellWidth = (maxX - minX + 1e-9) / gridSize;
    double cellHeight = (maxY - minY + 1e-9) / gridSize;

    // Assign cities to grid cells
    vector<pair<pair<int, int>, int>> gridCities;
    for (int i = 0; i < n; i++) {
        int gx = (int)((cities[i].x - minX) / cellWidth);
        int gy = (int)((cities[i].y - minY) / cellHeight);
        gx = min(gx, gridSize - 1);
        gy = min(gy, gridSize - 1);
        gridCities.push_back({{gx * gridSize + gy, i}, i});
    }

    // Sort by grid cell (space-filling order)
    sort(gridCities.begin(), gridCities.end());

    // Extract sorted indices
    vector<int> tour;
    tour.reserve(n);
    for (const auto& p : gridCities) {
        tour.push_back(p.second);
    }

    return tour;
}

// Simple 2-opt for small problems only
void twoOpt(vector<int>& tour, const vector<City>& cities, int maxIter) {
    int n = tour.size();
    bool improved = true;
    int iter = 0;

    while (improved && iter < maxIter) {
        improved = false;
        iter++;

        for (int i = 1; i < n - 2; i++) {
            for (int j = i + 2; j < n; j++) {
                double delta =
                    dist_sq(cities[tour[i-1]], cities[tour[j]]) +
                    dist_sq(cities[tour[i]], cities[tour[(j+1)%n]]) -
                    dist_sq(cities[tour[i-1]], cities[tour[i]]) -
                    dist_sq(cities[tour[j]], cities[tour[(j+1)%n]]);

                if (delta < -1e-9) {
                    reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<City> cities(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].id = i;
    }

    vector<int> tour;

    if (n <= 100) {
        // Trivial: nearest neighbor + aggressive 2-opt
        tour = nearestNeighbor(cities);
        twoOpt(tour, cities, n * n);
    } else if (n <= 5000) {
        // Small-Medium: nearest neighbor + limited 2-opt
        tour = nearestNeighbor(cities);
        twoOpt(tour, cities, min(n, 1000));
    } else if (n <= 50000) {
        // Medium-Large: nearest neighbor only
        tour = nearestNeighbor(cities);
    } else {
        // Large-Extreme-Epic: grid-based fast approximation
        tour = gridBasedTour(cities);
    }

    // Verify tour starts with 0
    if (tour[0] != 0) {
        int pos = 0;
        for (int i = 0; i < n; i++) {
            if (tour[i] == 0) {
                pos = i;
                break;
            }
        }
        rotate(tour.begin(), tour.begin() + pos, tour.end());
    }

    // Output
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << tour[i];
    }
    cout << "\n";

    return 0;
}
