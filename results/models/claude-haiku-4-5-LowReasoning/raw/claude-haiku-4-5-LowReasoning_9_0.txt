using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        // Read input
        var line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);
        int stock_length = int.Parse(line1[1]);

        var cuts_input = Console.ReadLine().Split();
        var cuts = new List<(int length, int index)>();

        for (int i = 0; i < n; i++)
        {
            int length = int.Parse(cuts_input[i]);
            cuts.Add((length, i));
        }

        // Sort by length in descending order (First Fit Decreasing)
        cuts.Sort((a, b) => b.length.CompareTo(a.length));

        // Bin packing using First Fit Decreasing
        var stocks = new List<List<int>>();
        var stock_remaining = new List<int>();

        foreach (var (length, index) in cuts)
        {
            // Find first stock with enough space
            int stock_idx = -1;
            for (int i = 0; i < stock_remaining.Count; i++)
            {
                if (stock_remaining[i] >= length)
                {
                    stock_idx = i;
                    break;
                }
            }

            // If not found, create new stock
            if (stock_idx == -1)
            {
                stocks.Add(new List<int>());
                stock_remaining.Add(stock_length);
                stock_idx = stocks.Count - 1;
            }

            // Add cut to stock
            stocks[stock_idx].Add(index);
            stock_remaining[stock_idx] -= length;
        }

        // Output
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
