const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { width, height, grid, ignitions, towns, cutBudget } = data;

// Convert grid strings to 2D array for easier access
const gridArray = grid.map(row => row.split(''));

// Simulate fire spread with given cuts
function fireReach(cuts) {
  const burned = new Set();
  const cutSet = new Set(cuts.map(c => `${c[0]},${c[1]}`));

  const queue = [];
  for (const [x, y] of ignitions) {
    const key = `${x},${y}`;
    burned.add(key);
    queue.push([x, y]);
  }

  let idx = 0;
  while (idx < queue.length) {
    const [x, y] = queue[idx++];

    // Fire spreads in 4 directions
    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
        const key = `${nx},${ny}`;
        // Spread to forest cells that aren't burned, cut, or water
        if (!burned.has(key) && gridArray[ny][nx] === '.' && !cutSet.has(key)) {
          burned.add(key);
          queue.push([nx, ny]);
        }
      }
    }
  }

  return burned;
}

// Calculate total value of towns that survive (not burned)
function getSavedValue(burned) {
  let saved = 0;
  for (const town of towns) {
    if (!burned.has(`${town.x},${town.y}`)) {
      saved += town.value;
    }
  }
  return saved;
}

// Get cells at the frontier - adjacent to burned cells but not burned themselves
function getFrontierCells(burned) {
  const frontier = new Set();

  for (const cellKey of burned) {
    const [x, y] = cellKey.split(',').map(Number);

    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
        const key = `${nx},${ny}`;
        if (!burned.has(key) && gridArray[ny][nx] === '.') {
          frontier.add(key);
        }
      }
    }
  }

  return frontier;
}

// Greedy algorithm: iteratively place cuts that save the most value
let cuts = [];
let currentBurned = fireReach([]);
let currentSaved = getSavedValue(currentBurned);

for (let iter = 0; iter < cutBudget; iter++) {
  const frontier = getFrontierCells(currentBurned);

  if (frontier.size === 0) break; // No unprotected forest cells to cut

  let bestCut = null;
  let bestSaved = currentSaved;
  let bestBurned = null;

  // Evaluate each frontier cell as a potential cut
  for (const cellKey of frontier) {
    const [x, y] = cellKey.split(',').map(Number);
    const testCuts = [...cuts, [x, y]];
    const testBurned = fireReach(testCuts);
    const testSaved = getSavedValue(testBurned);

    if (testSaved > bestSaved) {
      bestSaved = testSaved;
      bestCut = [x, y];
      bestBurned = testBurned;
    }
  }

  if (bestCut) {
    cuts.push(bestCut);
    currentBurned = bestBurned;
    currentSaved = bestSaved;
  } else {
    break; // No improvement possible
  }
}

console.log(JSON.stringify({ firebreaks: cuts }));
