use std::io::{self, BufRead};

#[derive(Debug, Clone)]
struct Sample {
    hole_id: usize,
    sample_idx: usize,
    properties: Vec<f64>,
    position: (f64, f64, f64),
}

#[derive(Debug)]
struct Suspect {
    hole_id: usize,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read header
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let _num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    // Read property names
    let property_names: Vec<String> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|s| s.to_string())
        .take(num_properties)
        .collect();

    let mut samples = Vec::new();

    // Parse holes and samples
    while let Some(Ok(line)) = lines.next() {
        let parts: Vec<&str> = line.split_whitespace().collect();

        if parts.len() >= 9 {
            // This is a hole header
            let hole_id: usize = parts[0].parse().unwrap();
            let start_x: f64 = parts[1].parse().unwrap();
            let start_y: f64 = parts[2].parse().unwrap();
            let start_z: f64 = parts[3].parse().unwrap();
            let dir_x: f64 = parts[4].parse().unwrap();
            let dir_y: f64 = parts[5].parse().unwrap();
            let dir_z: f64 = parts[6].parse().unwrap();
            let num_samples: usize = parts[8].parse().unwrap();

            // Normalize direction vector
            let dir_len = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
            let (dir_x, dir_y, dir_z) = if dir_len > 0.0 {
                (dir_x / dir_len, dir_y / dir_len, dir_z / dir_len)
            } else {
                (0.0, 0.0, 1.0)
            };

            for sample_idx in 0..num_samples {
                if let Some(Ok(sample_line)) = lines.next() {
                    let sample_parts: Vec<&str> = sample_line.split_whitespace().collect();
                    if sample_parts.len() >= 1 {
                        let depth: f64 = sample_parts[0].parse().unwrap_or(0.0);
                        let properties: Vec<f64> = sample_parts[1..]
                            .iter()
                            .map(|s| s.parse().unwrap_or(0.0))
                            .take(num_properties)
                            .collect();

                        // Calculate 3D position along the drillhole
                        let x = start_x + dir_x * depth;
                        let y = start_y + dir_y * depth;
                        let z = start_z + dir_z * depth;

                        samples.push(Sample {
                            hole_id,
                            sample_idx,
                            properties,
                            position: (x, y, z),
                        });
                    }
                }
            }
        }
    }

    // Find outliers
    let suspects = find_outliers(&samples, &property_names);

    // Sort by confidence descending
    let mut suspects = suspects;
    suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    // Output
    println!("{}", suspects.len());
    for suspect in suspects {
        println!("{} {} {} {}", suspect.hole_id, suspect.sample_idx,
                 suspect.property_name, suspect.confidence);
    }
}

fn find_outliers(samples: &[Sample], property_names: &[String]) -> Vec<Suspect> {
    let mut suspects = Vec::new();

    // Adaptive parameters based on dataset size
    let total_samples = samples.len();
    let (k_neighbors, use_grid) = if total_samples > 5000 {
        (3, true)  // Use spatial grid for large datasets
    } else {
        (5, false) // Brute force for small datasets
    };

    let z_threshold = 2.0;

    if use_grid && total_samples > 100 {
        analyze_with_grid(samples, property_names, k_neighbors, z_threshold, &mut suspects);
    } else {
        analyze_brute_force(samples, property_names, k_neighbors, z_threshold, &mut suspects);
    }

    suspects
}

fn analyze_brute_force(samples: &[Sample], property_names: &[String],
                       k_neighbors: usize, z_threshold: f64, suspects: &mut Vec<Suspect>) {
    for (idx, sample) in samples.iter().enumerate() {
        let neighbors = find_neighbors_brute(samples, idx, k_neighbors);

        for prop_idx in 0..sample.properties.len() {
            if prop_idx < property_names.len() {
                if let Some(suspect) = check_property(samples, &neighbors, idx, prop_idx,
                                                        z_threshold, &property_names[prop_idx], sample) {
                    suspects.push(suspect);
                }
            }
        }
    }
}

fn analyze_with_grid(samples: &[Sample], property_names: &[String],
                     k_neighbors: usize, z_threshold: f64, suspects: &mut Vec<Suspect>) {
    // Determine grid size dynamically
    let grid_size = ((samples.len() as f64 / 50.0).sqrt().ceil() as usize).max(3).min(100);

    // Find bounds
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    let mut min_z = f64::INFINITY;
    let mut max_z = f64::NEG_INFINITY;

    for sample in samples {
        min_x = min_x.min(sample.position.0);
        max_x = max_x.max(sample.position.0);
        min_y = min_y.min(sample.position.1);
        max_y = max_y.max(sample.position.1);
        min_z = min_z.min(sample.position.2);
        max_z = max_z.max(sample.position.2);
    }

    let eps = 1e-10;
    let cell_x = ((max_x - min_x).abs() / grid_size as f64).max(eps);
    let cell_y = ((max_y - min_y).abs() / grid_size as f64).max(eps);
    let cell_z = ((max_z - min_z).abs() / grid_size as f64).max(eps);

    // Create grid
    let mut grid = vec![vec![vec![Vec::new(); grid_size]; grid_size]; grid_size];

    // Index samples into grid
    for (idx, sample) in samples.iter().enumerate() {
        let gx = (((sample.position.0 - min_x) / cell_x) as usize).min(grid_size - 1);
        let gy = (((sample.position.1 - min_y) / cell_y) as usize).min(grid_size - 1);
        let gz = (((sample.position.2 - min_z) / cell_z) as usize).min(grid_size - 1);
        grid[gx][gy][gz].push(idx);
    }

    // Analyze samples
    for (idx, sample) in samples.iter().enumerate() {
        let gx = (((sample.position.0 - min_x) / cell_x) as usize).min(grid_size - 1);
        let gy = (((sample.position.1 - min_y) / cell_y) as usize).min(grid_size - 1);
        let gz = (((sample.position.2 - min_z) / cell_z) as usize).min(grid_size - 1);

        let neighbors = find_neighbors_grid(&grid, gx, gy, gz, grid_size, samples, idx, k_neighbors);

        for prop_idx in 0..sample.properties.len() {
            if prop_idx < property_names.len() {
                if let Some(suspect) = check_property(samples, &neighbors, idx, prop_idx,
                                                        z_threshold, &property_names[prop_idx], sample) {
                    suspects.push(suspect);
                }
            }
        }
    }
}

fn find_neighbors_brute(samples: &[Sample], idx: usize, k: usize) -> Vec<usize> {
    let sample = &samples[idx];
    let mut distances: Vec<(f64, usize)> = samples.iter()
        .enumerate()
        .filter(|(i, _)| *i != idx)
        .map(|(i, s)| (distance(&sample.position, &s.position), i))
        .collect();

    distances.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap());
    distances.iter().take(k).map(|(_, i)| *i).collect()
}

fn find_neighbors_grid(grid: &[Vec<Vec<Vec<usize>>>], gx: usize, gy: usize, gz: usize,
                       grid_size: usize, samples: &[Sample], idx: usize, k: usize) -> Vec<usize> {
    let mut candidates = Vec::new();
    let range = 1;

    let gx_min = if gx >= range { gx - range } else { 0 };
    let gx_max = (gx + range).min(grid_size - 1);
    let gy_min = if gy >= range { gy - range } else { 0 };
    let gy_max = (gy + range).min(grid_size - 1);
    let gz_min = if gz >= range { gz - range } else { 0 };
    let gz_max = (gz + range).min(grid_size - 1);

    for dx in gx_min..=gx_max {
        for dy in gy_min..=gy_max {
            for dz in gz_min..=gz_max {
                candidates.extend(&grid[dx][dy][dz]);
            }
        }
    }

    let sample = &samples[idx];
    let mut distances: Vec<(f64, usize)> = candidates.iter()
        .filter(|&&i| i != idx)
        .map(|&i| (distance(&sample.position, &samples[i].position), i))
        .collect();

    distances.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap());
    distances.iter().take(k).map(|(_, i)| *i).collect()
}

fn check_property(samples: &[Sample], neighbors: &[usize], idx: usize, prop_idx: usize,
                  z_threshold: f64, property_name: &str, sample: &Sample) -> Option<Suspect> {
    if neighbors.is_empty() || prop_idx >= sample.properties.len() {
        return None;
    }

    let value = sample.properties[prop_idx];
    let neighbor_values: Vec<f64> = neighbors.iter()
        .filter_map(|&n| {
            if prop_idx < samples[n].properties.len() {
                Some(samples[n].properties[prop_idx])
            } else {
                None
            }
        })
        .collect();

    if neighbor_values.is_empty() {
        return None;
    }

    let mean = neighbor_values.iter().sum::<f64>() / neighbor_values.len() as f64;
    let variance = neighbor_values.iter()
        .map(|v| (v - mean).powi(2))
        .sum::<f64>() / neighbor_values.len() as f64;
    let std_dev = variance.sqrt();

    if std_dev > 1e-10 {
        let z_score = (value - mean).abs() / std_dev;
        if z_score > z_threshold {
            // Confidence increases with z-score, capped at 1.0
            let confidence = ((z_score / z_threshold - 1.0) / 2.0).min(1.0);
            return Some(Suspect {
                hole_id: sample.hole_id,
                sample_idx: sample.sample_idx,
                property_name: property_name.to_string(),
                confidence,
            });
        }
    }

    None
}

fn distance(p1: &(f64, f64, f64), p2: &(f64, f64, f64)) -> f64 {
    let dx = p1.0 - p2.0;
    let dy = p1.1 - p2.1;
    let dz = p1.2 - p2.2;
    (dx * dx + dy * dy + dz * dz).sqrt()
}
