using System;
using System.Collections.Generic;
using System.IO;

class MiningOptimizer
{
    private int xCount, yCount, zCount;
    private double xBlockSize, yBlockSize, zBlockSize;
    private byte[,,] voxels;
    private bool[,,] dug;

    static void Main()
    {
        var optimizer = new MiningOptimizer();
        optimizer.Run();
    }

    void Run()
    {
        ReadInput();
        var operations = GenerateOperations();
        WriteOutput(operations);
    }

    void ReadInput()
    {
        var header = Console.ReadLine().Split();
        xCount = int.Parse(header[0]);
        yCount = int.Parse(header[1]);
        zCount = int.Parse(header[2]);
        xBlockSize = double.Parse(header[3]);
        yBlockSize = double.Parse(header[4]);
        zBlockSize = double.Parse(header[5]);

        voxels = new byte[xCount, yCount, zCount];
        dug = new bool[xCount, yCount, zCount];

        var stdin = Console.OpenStandardInput();
        int totalBytes = xCount * yCount * zCount;
        byte[] buffer = new byte[totalBytes];
        stdin.Read(buffer, 0, totalBytes);

        int idx = 0;
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    voxels[x, y, z] = buffer[idx++];
                }
            }
        }
    }

    List<string> GenerateOperations()
    {
        var operations = new List<string>();

        // Collect all ore with value > 0
        var oreVoxels = new List<(int x, int y, int z, int mass, int value, double profitScore)>();

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    if (dug[x, y, z]) continue;

                    byte voxel = voxels[x, y, z];
                    int mass = voxel % 16;
                    int value = voxel / 16;

                    if (value > 0 && mass > 0)
                    {
                        // Profit score: (mass * value * 1000) / (transport_cost_estimate)
                        // Transport cost estimate: manhattan distance * cost_per_step
                        int manDist = Math.Abs(x) + Math.Abs(y);
                        double costEst = Math.Max(1, manDist); // Avoid division by zero
                        double profitScore = (mass * value * 1000.0) / costEst;

                        oreVoxels.Add((x, y, z, mass, value, profitScore));
                    }
                }
            }
        }

        // Sort by profit score (descending)
        oreVoxels.Sort((a, b) => b.profitScore.CompareTo(a.profitScore));

        // Process highest profit ore first
        foreach (var (x, y, z, mass, value, profitScore) in oreVoxels)
        {
            if (dug[x, y, z]) continue; // Already processed

            // Check if ore is exposed (at surface level or has empty space nearby)
            if (IsExposed(x, y, z))
            {
                // Calculate actual movement cost
                int manDist = Math.Abs(x) + Math.Abs(y);
                double revenue = mass * value * 1000.0;
                double cost = mass * manDist; // Simplified: $1 per unit mass per step

                // Only move if profitable
                if (revenue > cost * 1.5) // 50% profit margin threshold
                {
                    var path = GeneratePath(x, y, 0, 0);
                    if (path.Count > 0)
                    {
                        var operation = FormatOperation(x, y, z, path);
                        if (operation != null)
                        {
                            operations.Add(operation);
                            dug[x, y, z] = true;
                        }
                    }
                }
            }
        }

        return operations;
    }

    bool IsExposed(int x, int y, int z)
    {
        // Check if voxel is at the surface (top of stack at this x,y)
        // or has empty space (dug area) adjacent

        // Check above
        if (z == zCount - 1) return true;
        if (dug[x, y, z + 1]) return true;

        // Check sides (4-connected)
        int[] dx = { 1, -1, 0, 0 };
        int[] dy = { 0, 0, 1, -1 };

        for (int i = 0; i < 4; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];

            if (nx >= 0 && nx < xCount && ny >= 0 && ny < yCount)
            {
                if (dug[nx, ny, z])
                    return true;
            }
            else
            {
                // Outside boundary = exposed to the edge
                return true;
            }
        }

        return false;
    }

    List<(int x, int y)> GeneratePath(int startX, int startY, int endX, int endY)
    {
        var path = new List<(int, int)>();

        int x = startX, y = startY;

        // Generate manhattan path to destination
        while (x != endX || y != endY)
        {
            if (x > endX) x--;
            else if (x < endX) x++;
            else if (y > endY) y--;
            else if (y < endY) y++;

            path.Add((x, y));
        }

        return path;
    }

    string FormatOperation(int digX, int digY, int digZ, List<(int x, int y)> path)
    {
        if (path.Count == 0) return null;

        var sb = new System.Text.StringBuilder();
        sb.Append($"{path.Count} {digX} {digY} {digZ}");

        // Add all path steps
        foreach (var (x, y) in path)
        {
            sb.Append($" {x} {y}");
        }

        // Add destination (last position in path)
        var lastPos = path[path.Count - 1];
        sb.Append($" {lastPos.x} {lastPos.y}");

        return sb.ToString();
    }

    void WriteOutput(List<string> operations)
    {
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op);
        }
    }
}
