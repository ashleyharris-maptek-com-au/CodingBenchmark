[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // 6x6 grid pattern
    float2 cellCoord = input.uv * 6.0;
    float2 cellCenter = floor(cellCoord) + 0.5;
    float2 cellOffset = cellCoord - cellCenter;

    // Distance from cell center
    float dist = length(cellOffset);

    // Determine color based on distance from cell center
    float3 baseColor;
    if (dist < 0.3) {
        baseColor = float3(1.0, 0.8, 0.0); // yellow dots
    } else {
        baseColor = float3(0.1, 0.1, 0.3); // dark blue background
    }

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L));
    float shadingFactor = 0.3 + 0.7 * diffuse;

    // Apply shading and output
    float3 finalColor = baseColor * shadingFactor;
    return float4(finalColor, 1.0);
}
