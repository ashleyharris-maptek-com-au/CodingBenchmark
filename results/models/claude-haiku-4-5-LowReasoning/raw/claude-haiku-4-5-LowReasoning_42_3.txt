OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
     %uint_2 = OpConstant %uint 2
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
 %float_0_03 = OpConstant %float 0.03
 %float_0_5 = OpConstant %float 0.5
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_input_arr = OpTypePointer Input %arr_gl_PerVertex
       %gl_in = OpVariable %ptr_input_arr Input
%arr_color_in = OpTypeArray %v3float %uint_2
%ptr_input_color = OpTypePointer Input %arr_color_in
    %color_in = OpVariable %ptr_input_color Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_struct = OpTypePointer Output %gl_PerVertex_out
      %gl_out = OpVariable %ptr_output_struct Output
%ptr_output_v3 = OpTypePointer Output %v3float
     %outColor = OpVariable %ptr_output_v3 Output
%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_output_v4 = OpTypePointer Output %v4float
 %fn_type = OpTypeFunction %void
       %main = OpFunction %void None %fn_type
      %label = OpLabel
; Load input positions
      %addr0 = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
        %p0 = OpLoad %v4float %addr0
      %addr1 = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
        %p1 = OpLoad %v4float %addr1
; Load input colors
   %c0_addr = OpAccessChain %ptr_input_color %color_in %int_0
        %c0 = OpLoad %v3float %c0_addr
   %c1_addr = OpAccessChain %ptr_input_color %color_in %int_1
        %c1 = OpLoad %v3float %c1_addr
; Extract xy coordinates
    %p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
    %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
; Compute direction: dir = p1 - p0
      %dir = OpFSub %v2float %p1_xy %p0_xy
; Perpendicular: (-dir.y, dir.x)
    %dir_x = OpVectorShuffle %float %dir %dir 0 0
    %dir_y = OpVectorShuffle %float %dir %dir 1 1
  %neg_y = OpFNegate %float %dir_y
 %perp_unnorm = OpCompositeConstruct %v2float %neg_y %dir_x
; Compute length: sqrt(x^2 + y^2)
    %px = OpVectorShuffle %float %perp_unnorm %perp_unnorm 0 0
    %py = OpVectorShuffle %float %perp_unnorm %perp_unnorm 1 1
   %px2 = OpFMul %float %px %px
   %py2 = OpFMul %float %py %py
  %len2 = OpFAdd %float %px2 %py2
   %len = OpExtInst %float %1 Sqrt %len2
; Normalize: perp_norm = perp_unnorm / len
 %norm_x = OpFDiv %float %px %len
 %norm_y = OpFDiv %float %py %len
%norm_vec = OpCompositeConstruct %v2float %norm_x %norm_y
; Scale by thickness: perp = norm_vec * 0.03
   %perp = OpVectorTimesScalar %v2float %norm_vec %float_0_03
; Extract z, w from p0 and p1
  %p0_z = OpVectorShuffle %float %p0 %p0 2 2
  %p0_w = OpVectorShuffle %float %p0 %p0 3 3
  %p1_z = OpVectorShuffle %float %p1 %p1 2 2
  %p1_w = OpVectorShuffle %float %p1 %p1 3 3
; Compute v0 = (p0.xy + perp, p0.z, p0.w)
  %v0_xy = OpFAdd %v2float %p0_xy %perp
   %v0_xy_zw = OpCompositeConstruct %v4float %float_0 %float_0 %p0_z %p0_w
      %v0 = OpVectorShuffle %v4float %v0_xy %v0_xy_zw 0 1 2 3
; Compute v1 = (p0.xy - perp, p0.z, p0.w)
  %v1_xy = OpFSub %v2float %p0_xy %perp
   %v1_xy_zw = OpCompositeConstruct %v4float %float_0 %float_0 %p0_z %p0_w
      %v1 = OpVectorShuffle %v4float %v1_xy %v1_xy_zw 0 1 2 3
; Compute v2 = (p1.xy + perp, p1.z, p1.w)
  %v2_xy = OpFAdd %v2float %p1_xy %perp
   %v2_xy_zw = OpCompositeConstruct %v4float %float_0 %float_0 %p1_z %p1_w
      %v2 = OpVectorShuffle %v4float %v2_xy %v2_xy_zw 0 1 2 3
; Compute v3 = (p1.xy - perp, p1.z, p1.w)
  %v3_xy = OpFSub %v2float %p1_xy %perp
   %v3_xy_zw = OpCompositeConstruct %v4float %float_0 %float_0 %p1_z %p1_w
      %v3 = OpVectorShuffle %v4float %v3_xy %v3_xy_zw 0 1 2 3
; Emit vertex 0: v0 with c0
      %gl_pos0 = OpAccessChain %ptr_output_v4 %gl_out %int_0
               OpStore %gl_pos0 %v0
               OpStore %outColor %c0
               OpEmitVertex
; Emit vertex 1: v1 with c0
      %gl_pos1 = OpAccessChain %ptr_output_v4 %gl_out %int_0
               OpStore %gl_pos1 %v1
               OpStore %outColor %c0
               OpEmitVertex
; Emit vertex 2: v2 with c1
      %gl_pos2 = OpAccessChain %ptr_output_v4 %gl_out %int_0
               OpStore %gl_pos2 %v2
               OpStore %outColor %c1
               OpEmitVertex
; Emit vertex 3: v3 with c1
      %gl_pos3 = OpAccessChain %ptr_output_v4 %gl_out %int_0
               OpStore %gl_pos3 %v3
               OpStore %outColor %c1
               OpEmitVertex
; End primitive
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
