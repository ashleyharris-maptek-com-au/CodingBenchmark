#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>
#include <queue>

using namespace std;

struct Point {
    double x, y, z;
    Point() : x(0), y(0), z(0) {}
    Point(double x, double y, double z) : x(x), y(y), z(z) {}
    Point operator+(const Point& p) const { return Point(x+p.x, y+p.y, z+p.z); }
    Point operator-(const Point& p) const { return Point(x-p.x, y-p.y, z-p.z); }
    Point operator*(double s) const { return Point(x*s, y*s, z*s); }
    double dot(const Point& p) const { return x*p.x + y*p.y + z*p.z; }
    Point cross(const Point& p) const {
        return Point(y*p.z - z*p.y, z*p.x - x*p.z, x*p.y - y*p.x);
    }
    double length() const { return sqrt(x*x + y*y + z*z); }
    double length2() const { return x*x + y*y + z*z; }
    Point normalized() const {
        double len = length();
        if (len < 1e-10) return Point(0, 0, 0);
        return Point(x/len, y/len, z/len);
    }
};

struct Tetrahedron {
    Point center;
    double rx, ry, rz;
};

struct Plane {
    Point normal;
    double d;
    Plane() {}
    Plane(const Point& a, const Point& b, const Point& c) {
        Point ba = b - a;
        Point ca = c - a;
        normal = ba.cross(ca).normalized();
        d = normal.dot(a);
    }
    double distance(const Point& p) const {
        return normal.dot(p) - d;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge_length;
    cin >> V >> edge_length;

    vector<Point> vertices(V);
    for (int i = 0; i < V; i++) {
        cin >> vertices[i].x >> vertices[i].y >> vertices[i].z;
    }

    Point minp = vertices[0];
    Point maxp = vertices[0];
    for (const auto& v : vertices) {
        minp.x = min(minp.x, v.x);
        minp.y = min(minp.y, v.y);
        minp.z = min(minp.z, v.z);
        maxp.x = max(maxp.x, v.x);
        maxp.y = max(maxp.y, v.y);
        maxp.z = max(maxp.z, v.z);
    }

    double circumradius = edge_length * sqrt(3.0/8.0);
    double min_dist = 2.0 * circumradius * 0.99;

    vector<Tetrahedron> placed;
    vector<Plane> planes;
    for (int i = 0; i < V; i++) {
        for (int j = i+1; j < V; j++) {
            for (int k = j+1; k < V; k++) {
                planes.push_back(Plane(vertices[i], vertices[j], vertices[k]));
            }
        }
    }

    // Compute container center and bounds
    Point center_point = (minp + maxp) * 0.5;
    Point extent = maxp - minp;
    double max_extent = max({extent.x, extent.y, extent.z});
    double grid_spacing = edge_length * 1.15;

    // Adaptive grid placement
    int grid_dim = (int)(max_extent / grid_spacing) + 2;

    for (int gi = 0; gi <= grid_dim; gi++) {
        for (int gj = 0; gj <= grid_dim; gj++) {
            for (int gk = 0; gk <= grid_dim; gk++) {
                Point candidate(
                    minp.x + gi * grid_spacing,
                    minp.y + gj * grid_spacing,
                    minp.z + gk * grid_spacing
                );

                if (candidate.x > maxp.x + circumradius ||
                    candidate.y > maxp.y + circumradius ||
                    candidate.z > maxp.z + circumradius ||
                    candidate.x < minp.x - circumradius ||
                    candidate.y < minp.y - circumradius ||
                    candidate.z < minp.z - circumradius) {
                    continue;
                }

                bool inside = true;
                for (const auto& p : planes) {
                    if (p.distance(candidate) < -circumradius) {
                        inside = false;
                        break;
                    }
                }

                if (!inside) continue;

                bool collision = false;
                for (const auto& tet : placed) {
                    double dist2 = (candidate - tet.center).length2();
                    if (dist2 < min_dist * min_dist) {
                        collision = true;
                        break;
                    }
                }

                if (!collision) {
                    Tetrahedron tet;
                    tet.center = candidate;
                    tet.rx = 0;
                    tet.ry = 0;
                    tet.rz = 0;
                    placed.push_back(tet);
                }

                if (placed.size() > 1000000) break;
            }
            if (placed.size() > 1000000) break;
        }
        if (placed.size() > 1000000) break;
    }

    cout << placed.size() << "\n";
    cout << fixed << setprecision(10);
    for (const auto& tet : placed) {
        cout << tet.center.x << " " << tet.center.y << " " << tet.center.z << " "
             << tet.rx << " " << tet.ry << " " << tet.rz << "\n";
    }

    return 0;
}
