#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out of bounds
    if (idx >= N) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Read current body data
    uint bodyIdx = idx * 2;
    vec4 posAndMass = inData[bodyIdx];
    vec4 velData = inData[bodyIdx + 1];

    vec3 pos = posAndMass.xyz;
    float mass = posAndMass.w;
    vec3 vel = velData.xyz;

    // Compute acceleration from all other bodies
    vec3 acc = vec3(0.0);

    for (uint j = 0; j < N; ++j) {
        if (j == idx) continue;  // Skip self-interaction

        uint otherBodyIdx = j * 2;
        vec4 otherPosAndMass = inData[otherBodyIdx];

        vec3 otherPos = otherPosAndMass.xyz;
        float otherMass = otherPosAndMass.w;

        vec3 delta = otherPos - pos;
        float distSq = dot(delta, delta) + softening * softening;
        float distCubed = distSq * sqrt(distSq);

        acc += G * otherMass * delta / distCubed;
    }

    // Update velocity and position using Euler method
    vel += acc * dt;
    pos += vel * dt;

    // Write updated data
    outData[bodyIdx] = vec4(pos, mass);
    outData[bodyIdx + 1] = vec4(vel, 0.0);
}
