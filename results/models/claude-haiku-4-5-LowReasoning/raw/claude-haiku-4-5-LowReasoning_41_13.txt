%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%ptr_Input_vec2 = OpTypePointer Input %vec2

%main_func_type = OpTypeFunction %void

%worldPos = OpVariable %ptr_Input_vec3 Input
%normal = OpVariable %ptr_Input_vec3 Input

%tangent = OpVariable %ptr_Input_vec3 Input

%main = OpFunction %void None %main_func_type
%entry = OpLabel

%wp = OpLoad %vec3 %worldPos
%y = OpCompositeExtract %float %wp 1

%scaled = OpFMul %float %y %float_0_5
%shifted = OpFAdd %float %scaled %float_0_5
%t = OpExtInst %float %glsl FClamp %shifted %float_0 %float_1_0

%bottom = OpCompositeConstruct %vec3 %float_0_1 %float_0_4 %float_0_1
%top = OpCompositeConstruct %vec3 %float_1_0 %float_1_0 %float_1_0

%diff = OpFSub %vec3 %top %bottom
%scaled_diff = OpVectorTimesScalar %vec3 %diff %t
%result_rgb = OpFAdd %vec3 %bottom %scaled_diff

%result_vec4 = OpCompositeConstruct %vec4 %result_rgb %float_1_0

OpStore %outColor %result_vec4

OpReturn
