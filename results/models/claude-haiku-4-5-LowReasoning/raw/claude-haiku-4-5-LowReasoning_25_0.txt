using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;

class AsteroidInterceptionSolver
{
    const double G = 6.67430e-11;
    const double AU = 1.496e11;
    const double EARTH_MASS = 5.972e24;
    const double SUN_MASS = 1.989e30;
    const double IMPACT_DISTANCE = 1000000; // 1000 km in meters

    struct Body
    {
        public string name;
        public double mass;
        public Vector3 pos;
        public Vector3 vel;
    }

    struct Burn
    {
        public double time;
        public Vector3 dv;
    }

    struct Vector3
    {
        public double x, y, z;

        public Vector3(double x = 0, double y = 0, double z = 0)
        {
            this.x = x;
            this.y = y;
            this.z = z;
        }

        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.x + b.x, a.y + b.y, a.z + b.z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.x - b.x, a.y - b.y, a.z - b.z);
        public static Vector3 operator *(Vector3 a, double s) => new Vector3(a.x * s, a.y * s, a.z * s);
        public static Vector3 operator *(double s, Vector3 a) => new Vector3(a.x * s, a.y * s, a.z * s);

        public double Magnitude() => Math.Sqrt(x * x + y * y + z * z);
        public Vector3 Normalized() { var m = Magnitude(); return m > 1e-10 ? new Vector3(x / m, y / m, z / m) : new Vector3(); }
        public double Dot(Vector3 b) => x * b.x + y * b.y + z * b.z;
        public Vector3 Cross(Vector3 b) => new Vector3(y * b.z - z * b.y, z * b.x - x * b.z, x * b.y - y * b.x);
    }

    static void Main()
    {
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        var idx = 0;

        int numBodies = int.Parse(tokens[idx++]);
        int warningDays = int.Parse(tokens[idx++]);
        double impactorMass = double.Parse(tokens[idx++]);
        double deltaVBudget = double.Parse(tokens[idx++]);
        double asteroidMass = double.Parse(tokens[idx++]);
        long t0UnixUtc = long.Parse(tokens[idx++]);

        var bodies = new List<Body>();
        Body earth = default;
        Body asteroid = default;

        for (int i = 0; i < numBodies; i++)
        {
            var body = new Body();
            body.name = tokens[idx++];
            body.mass = double.Parse(tokens[idx++]);
            body.pos = new Vector3(double.Parse(tokens[idx++]), double.Parse(tokens[idx++]), double.Parse(tokens[idx++]));
            body.vel = new Vector3(double.Parse(tokens[idx++]), double.Parse(tokens[idx++]), double.Parse(tokens[idx++]));

            bodies.Add(body);
            if (body.name == "Earth") earth = body;
            if (body.name == "Asteroid") asteroid = body;
        }

        string de421Path = idx < tokens.Length ? tokens[idx] : "";

        var burns = PlanTrajectory(bodies, earth, asteroid, warningDays, deltaVBudget, asteroidMass, impactorMass);

        Console.WriteLine(burns.Count);
        foreach (var burn in burns)
        {
            Console.WriteLine($"{burn.time} {burn.dv.x} {burn.dv.y} {burn.dv.z}");
        }
    }

    static List<Burn> PlanTrajectory(List<Body> bodies, Body earth, Body asteroid, int warningDays, double deltaVBudget, double asteroidMass, double impactorMass)
    {
        var burns = new List<Burn>();

        // Find collision time (when asteroid reaches Earth's position)
        double collisionTime = FindCollisionTime(earth, asteroid, warningDays);

        if (collisionTime <= 0 || collisionTime > warningDays)
        {
            // Asteroid not on collision course or won't collide in warning window
            return burns;
        }

        // Plan intercept
        double interceptTime = collisionTime * 0.95; // Intercept slightly before collision

        // Launch burn: accelerate from Earth toward asteroid
        var earthToAsteroid = asteroid.pos - earth.pos;
        double distance = earthToAsteroid.Magnitude();

        // Estimate required delta-v for Hohmann-like transfer
        double launchDv = EstimateLaunchDv(earth, asteroid, interceptTime);

        if (launchDv > deltaVBudget)
        {
            // Try direct ascent with all available delta-v
            launchDv = deltaVBudget * 0.6; // Reserve some for corrections and impact
        }

        // Launch immediately
        Vector3 launchDirection = earthToAsteroid.Normalized();
        var launchBurn = new Burn
        {
            time = 1.0,
            dv = launchDirection * launchDv
        };
        burns.Add(launchBurn);

        double usedDv = launchDv;
        double remainingDv = deltaVBudget - usedDv;

        // Mid-course corrections
        // Simple approach: one correction halfway to intercept
        if (remainingDv > 100 && interceptTime > 5)
        {
            double correctionTime = interceptTime * 0.5;
            Vector3 correctionDv = PlanCorrectionBurn(earth, asteroid, launchBurn, correctionTime, interceptTime);
            double correctionMag = correctionDv.Magnitude();

            if (correctionMag > 0 && correctionMag < remainingDv * 0.5)
            {
                var correctionBurn = new Burn { time = correctionTime, dv = correctionDv };
                burns.Add(correctionBurn);
                usedDv += correctionMag;
                remainingDv -= correctionMag;
            }
        }

        // Final impact burn
        // Calculate deflection needed based on kinetic impact
        double deflectionNeeded = CalculateDeflectionNeeded(earth, asteroid, asteroidMass, impactorMass);
        Vector3 impactDv = CalculateImpactBurn(asteroid, deflectionNeeded, remainingDv * 0.4);

        if (impactDv.Magnitude() > 0.1)
        {
            var impactBurn = new Burn
            {
                time = interceptTime,
                dv = impactDv
            };
            burns.Add(impactBurn);
        }

        return burns;
    }

    static double FindCollisionTime(Body earth, Body asteroid, int maxDays)
    {
        const double dt = 0.1; // 0.1 day steps
        Vector3 earthPos = earth.pos;
        Vector3 earthVel = earth.vel;
        Vector3 asteroidPos = asteroid.pos;
        Vector3 asteroidVel = asteroid.vel;

        for (double t = 0; t <= maxDays; t += dt)
        {
            earthPos = earthPos + earthVel * dt;
            asteroidPos = asteroidPos + asteroidVel * dt;

            var diff = asteroidPos - earthPos;
            double distance = diff.Magnitude();

            if (distance < AU * 0.1) // Within ~15 million km (rough collision zone)
            {
                if (t > 0.1) return t;
            }
        }

        return maxDays + 1; // Not on collision course
    }

    static double EstimateLaunchDv(Body earth, Body asteroid, double interceptTime)
    {
        var rel = asteroid.pos - earth.pos;
        double distance = rel.Magnitude();
        double timeToIntercept = interceptTime;

        // Simple estimate: need to cover distance in given time
        double requiredVel = distance / (timeToIntercept * 86400);
        double earthOrbitVel = earth.vel.Magnitude();

        return Math.Sqrt(requiredVel * requiredVel + earthOrbitVel * earthOrbitVel) - earthOrbitVel;
    }

    static Vector3 PlanCorrectionBurn(Body earth, Body asteroid, Burn previousBurn, double correctionTime, double targetTime)
    {
        // Estimate spacecraft position after launch
        Vector3 spacecraftPos = earth.pos;
        Vector3 spacecraftVel = earth.vel + previousBurn.dv;

        double dt = correctionTime;
        spacecraftPos = spacecraftPos + spacecraftVel * dt;

        // Target position
        Vector3 targetPos = asteroid.pos + asteroid.vel * dt;

        // Calculate correction needed
        var diff = targetPos - spacecraftPos;
        double distance = diff.Magnitude();

        if (distance < 1e9) return new Vector3(); // Already close enough

        // Apply simple correction
        double correctionMag = Math.Min(distance / (targetTime - correctionTime) / 86400, 500);
        return diff.Normalized() * correctionMag;
    }

    static double CalculateDeflectionNeeded(Body earth, Body asteroid, double asteroidMass, double impactorMass)
    {
        var relPos = asteroid.pos - earth.pos;
        var relVel = asteroid.vel - earth.vel;

        double distance = relPos.Magnitude();
        double relSpeed = relVel.Magnitude();

        // Needed velocity change to avoid collision (simplified)
        // Assume collision zone is ~100,000 km
        double collisionZone = 1e8;
        double deflectionNeeded = (collisionZone / distance) * relSpeed;

        // Momentum transfer efficiency
        double efficiency = impactorMass / asteroidMass;

        return deflectionNeeded / efficiency;
    }

    static Vector3 CalculateImpactBurn(Body asteroid, double deflectionNeeded, double maxDv)
    {
        // Direction roughly perpendicular to Earth impact trajectory
        Vector3 impactDir = new Vector3(1, 0.5, 0.2).Normalized();
        double magnitude = Math.Min(deflectionNeeded * 0.1, maxDv);
        return impactDir * magnitude;
    }
}
