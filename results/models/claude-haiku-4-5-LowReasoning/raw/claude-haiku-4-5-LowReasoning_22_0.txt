using System;
using System.Linq;

class TetrisAI
{
    // Piece rotations defined as arrays of (col, row) offsets
    static readonly int[][][] PIECES = {
        // I - 4x1 line piece
        new[] { new[] { 0, 0, 1, 0, 2, 0, 3, 0 }, new[] { 0, 0, 0, 1, 0, 2, 0, 3 } },
        // O - 2x2 square piece
        new[] { new[] { 0, 0, 1, 0, 0, 1, 1, 1 } },
        // T - T-shaped piece
        new[] {
            new[] { 0, 0, 1, 0, 2, 0, 1, 1 },
            new[] { 0, 0, 0, 1, 1, 1, 0, 2 },
            new[] { 0, 1, 1, 0, 1, 1, 2, 1 },
            new[] { 1, 0, 0, 1, 1, 1, 1, 2 }
        },
        // S - S-shaped piece
        new[] { new[] { 1, 0, 2, 0, 0, 1, 1, 1 }, new[] { 0, 0, 0, 1, 1, 1, 1, 2 } },
        // Z - Z-shaped piece
        new[] { new[] { 0, 0, 1, 0, 1, 1, 2, 1 }, new[] { 1, 0, 0, 1, 1, 1, 0, 2 } },
        // J - J-shaped piece (left-leaning L)
        new[] {
            new[] { 0, 0, 0, 1, 1, 1, 2, 1 },
            new[] { 1, 0, 2, 0, 1, 1, 1, 2 },
            new[] { 0, 1, 1, 1, 2, 1, 2, 0 },
            new[] { 0, 0, 0, 1, 0, 2, 1, 2 }
        },
        // L - L-shaped piece (right-leaning L)
        new[] {
            new[] { 2, 0, 0, 1, 1, 1, 2, 1 },
            new[] { 0, 0, 1, 0, 1, 1, 1, 2 },
            new[] { 0, 0, 0, 1, 1, 1, 2, 1 },
            new[] { 0, 0, 0, 1, 0, 2, 1, 2 }
        }
    };

    int width, height;
    int[] col_heights;

    static void Main()
    {
        new TetrisAI().Run();
    }

    void Run()
    {
        // Parse game parameters
        string[] input = Console.ReadLine().Split();
        width = int.Parse(input[0]);
        height = int.Parse(input[1]);
        int num_pieces = int.Parse(input[2]);

        col_heights = new int[width];

        // Process each piece
        for (int i = 0; i < num_pieces; i++)
        {
            char piece_type = Console.ReadLine().Trim()[0];
            int piece_idx = "IOTSZJL".IndexOf(piece_type);

            // Find and output best placement
            var best = FindBestPlacement(piece_idx);
            Console.WriteLine($"{best.rotation} {best.column}");
            Console.Out.Flush();

            // Read game result
            string result = Console.ReadLine();
            if (result == "gameover")
                break;

            // Update board state with placement
            PlacePiece(piece_idx, best.rotation, best.column);

            // Handle line clears
            if (result.StartsWith("cleared_"))
            {
                int cleared_lines = int.Parse(result.Substring(8));
                for (int j = 0; j < width; j++)
                    col_heights[j] = Math.Max(0, col_heights[j] - cleared_lines);
            }
        }
    }

    (int rotation, int column) FindBestPlacement(int piece_idx)
    {
        int best_rotation = 0, best_column = 0;
        int best_height = int.MaxValue;

        int[][] rotations = PIECES[piece_idx];

        for (int rot = 0; rot < rotations.Length; rot++)
        {
            int[] shape = rotations[rot];

            // Calculate piece bounding box
            int min_col = int.MaxValue, max_col = int.MinValue, max_row = int.MinValue;
            for (int i = 0; i < shape.Length; i += 2)
            {
                min_col = Math.Min(min_col, shape[i]);
                max_col = Math.Max(max_col, shape[i]);
                max_row = Math.Max(max_row, shape[i + 1]);
            }

            int piece_width = max_col - min_col + 1;

            // Try each valid column position
            for (int col = 0; col <= width - piece_width; col++)
            {
                int landing_height = 0;

                // Calculate where piece lands for this position
                for (int i = 0; i < shape.Length; i += 2)
                {
                    int actual_col = col + shape[i] - min_col;
                    int cell_row = shape[i + 1];
                    int final_height = col_heights[actual_col] + max_row - cell_row + 1;
                    landing_height = Math.Max(landing_height, final_height);
                }

                // Check if placement is valid and better than current best
                if (landing_height <= height && landing_height < best_height)
                {
                    best_height = landing_height;
                    best_rotation = rot;
                    best_column = col;
                }
            }
        }

        return (best_rotation, best_column);
    }

    void PlacePiece(int piece_idx, int rotation, int column)
    {
        int[] shape = PIECES[piece_idx][rotation];

        // Calculate bounding box
        int min_col = int.MaxValue, max_row = int.MinValue;
        for (int i = 0; i < shape.Length; i += 2)
        {
            min_col = Math.Min(min_col, shape[i]);
            max_row = Math.Max(max_row, shape[i + 1]);
        }

        // Find landing height
        int landing_height = 0;
        for (int i = 0; i < shape.Length; i += 2)
        {
            int actual_col = column + shape[i] - min_col;
            int cell_row = shape[i + 1];
            int final_height = col_heights[actual_col] + max_row - cell_row + 1;
            landing_height = Math.Max(landing_height, final_height);
        }

        // Update column heights
        for (int i = 0; i < shape.Length; i += 2)
        {
            int actual_col = column + shape[i] - min_col;
            int cell_row = shape[i + 1];
            int block_height = landing_height - max_row + cell_row;
            col_heights[actual_col] = Math.Max(col_heights[actual_col], block_height);
        }
    }
}
