OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpName %main "main"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_3 = OpConstant %float 3.0
%float_0_7 = OpConstant %float 0.7

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_input_arr = OpTypePointer Input %arr_pv_in_3
%gl_in = OpVariable %ptr_input_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_pv Output

%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_3
%inColor = OpVariable %ptr_input_arr_v3 Input

%ptr_output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3 Output

%void_fn = OpTypeFunction %void
%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_input_v3 = OpTypePointer Input %v3float

%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load positions from gl_in[0..2]
%p0_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

; Load colors from inColor[0..2]
%c0_ptr = OpAccessChain %ptr_input_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_input_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_input_v3 %inColor %int_2
%c2 = OpLoad %v3float %c2_ptr

; Compute centroid: C = (p0 + p1 + p2) / 3
%p01 = OpFAdd %v4float %p0 %p1
%p012 = OpFAdd %v4float %p01 %p2
%centroid = OpFDiv %v4float %p012 %float_3

; Shrink vertex 0: new_pos = C + 0.7 * (p0 - C)
%d0 = OpFSub %v4float %p0 %centroid
%sd0 = OpVectorTimesScalar %v4float %d0 %float_0_7
%s0 = OpFAdd %v4float %centroid %sd0

; Shrink vertex 1: new_pos = C + 0.7 * (p1 - C)
%d1 = OpFSub %v4float %p1 %centroid
%sd1 = OpVectorTimesScalar %v4float %d1 %float_0_7
%s1 = OpFAdd %v4float %centroid %sd1

; Shrink vertex 2: new_pos = C + 0.7 * (p2 - C)
%d2 = OpFSub %v4float %p2 %centroid
%sd2 = OpVectorTimesScalar %v4float %d2 %float_0_7
%s2 = OpFAdd %v4float %centroid %sd2

; Emit vertex 0 with shrunk position and original color
%out_ptr0 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %out_ptr0 %s0
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 1 with shrunk position and original color
%out_ptr1 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %out_ptr1 %s1
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 2 with shrunk position and original color
%out_ptr2 = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %out_ptr2 %s2
OpStore %outColor %c2
OpEmitVertex

; End the output primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
