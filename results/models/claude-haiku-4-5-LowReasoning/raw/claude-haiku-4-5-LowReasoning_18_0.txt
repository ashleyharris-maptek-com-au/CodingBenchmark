#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <cstring>

using namespace std;

const int INF = 1e9;

struct Edge {
    int to, cap, flow;
};

class MaxFlow {
public:
    vector<Edge> edges;
    vector<vector<int>> g;
    vector<int> level, iter;
    int n;

    MaxFlow(int n) : n(n), g(n), level(n), iter(n) {}

    void addEdge(int from, int to, int cap) {
        g[from].push_back(edges.size());
        edges.push_back({to, cap, 0});
        g[to].push_back(edges.size());
        edges.push_back({from, 0, 0});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        level[s] = 0;
        queue<int> q;
        q.push(s);

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (int id : g[v]) {
                if (level[edges[id].to] < 0 && edges[id].flow < edges[id].cap) {
                    level[edges[id].to] = level[v] + 1;
                    q.push(edges[id].to);
                }
            }
        }

        return level[t] >= 0;
    }

    int dfs(int v, int t, int pushed) {
        if (v == t || pushed == 0) return pushed;

        for (int& cid = iter[v]; cid < (int)g[v].size(); cid++) {
            int id = g[v][cid];
            int to = edges[id].to;

            if (level[v] + 1 != level[to] || edges[id].cap <= edges[id].flow)
                continue;

            int tr = dfs(to, t, min(pushed, edges[id].cap - edges[id].flow));
            if (tr > 0) {
                edges[id].flow += tr;
                edges[id ^ 1].flow -= tr;
                return tr;
            }
        }

        return 0;
    }

    int maxflow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            while (int pushed = dfs(s, t, INF)) {
                flow += pushed;
            }
        }
        return flow;
    }

    vector<bool> getSourceSide(int s) {
        vector<bool> reachable(n, false);
        queue<int> q;
        q.push(s);
        reachable[s] = true;

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (int id : g[v]) {
                int to = edges[id].to;
                if (!reachable[to] && edges[id].flow < edges[id].cap) {
                    reachable[to] = true;
                    q.push(to);
                }
            }
        }

        return reachable;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<pair<int, int>> edges(m);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        edges[i] = {u, v};
    }

    if (n <= 1) {
        cout << 0 << "\n";
        return 0;
    }

    int minCut = INF;
    int bestTarget = 1;

    for (int t = 1; t < n; t++) {
        MaxFlow mf(n);
        for (int i = 0; i < m; i++) {
            mf.addEdge(edges[i].first, edges[i].second, 1);
            mf.addEdge(edges[i].second, edges[i].first, 1);
        }

        int flow = mf.maxflow(0, t);
        if (flow < minCut) {
            minCut = flow;
            bestTarget = t;
        }

        if (minCut == 1) break;
    }

    MaxFlow mf(n);
    for (int i = 0; i < m; i++) {
        mf.addEdge(edges[i].first, edges[i].second, 1);
        mf.addEdge(edges[i].second, edges[i].first, 1);
    }

    mf.maxflow(0, bestTarget);

    vector<bool> sourceSide = mf.getSourceSide(0);

    vector<pair<int, int>> cutEdges;
    for (auto [u, v] : edges) {
        if (sourceSide[u] && !sourceSide[v]) {
            cutEdges.push_back({u, v});
        } else if (!sourceSide[u] && sourceSide[v]) {
            cutEdges.push_back({v, u});
        }
    }

    cout << minCut << "\n";
    for (auto [u, v] : cutEdges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
