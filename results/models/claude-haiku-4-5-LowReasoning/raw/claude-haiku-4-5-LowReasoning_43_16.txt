struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Map UVs to complex plane: c = (uv.x*3-2, uv.y*2-1)
    float2 c = float2(input.uv.x * 3.0 - 2.0, input.uv.y * 2.0 - 1.0);

    // Mandelbrot iteration
    float2 z = float2(0.0, 0.0);
    int iterations = 0;

    for (int i = 0; i < 20; i++) {
        // Check escape condition: |z|^2 > 4
        float mag_sq = dot(z, z);
        if (mag_sq > 4.0) {
            break;
        }

        // z = z^2 + c
        float2 z_sq = float2(z.x * z.x - z.y * z.y, 2.0 * z.x * z.y);
        z = z_sq + c;

        iterations++;
    }

    // Color with t = iterations/20 as (t, t*0.5, 1.0-t), alpha 1.0
    float t = float(iterations) / 20.0;
    float r = t;
    float g = t * 0.5;
    float b = 1.0 - t;

    return float4(r, g, b, 1.0);
}
