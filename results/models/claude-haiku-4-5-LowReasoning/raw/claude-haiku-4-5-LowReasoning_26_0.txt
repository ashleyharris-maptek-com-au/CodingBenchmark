use std::io::{self, BufReader, BufWriter, Read, Write};

#[derive(Clone, Debug)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    fn distance_squared(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }

    fn from_bytes(bytes: &[u8]) -> Self {
        let x = f32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
        let y = f32::from_le_bytes([bytes[4], bytes[5], bytes[6], bytes[7]]);
        let z = f32::from_le_bytes([bytes[8], bytes[9], bytes[10], bytes[11]]);
        Point { x, y, z }
    }
}

struct Centroid {
    x: f64,
    y: f64,
    z: f64,
    count: u64,
}

impl Centroid {
    fn new(p: &Point) -> Self {
        Centroid {
            x: p.x as f64,
            y: p.y as f64,
            z: p.z as f64,
            count: 1,
        }
    }

    fn add(&mut self, p: &Point) {
        self.x += p.x as f64;
        self.y += p.y as f64;
        self.z += p.z as f64;
        self.count += 1;
    }

    fn to_point(&self) -> Point {
        if self.count == 0 {
            Point { x: 0.0, y: 0.0, z: 0.0 }
        } else {
            let c = self.count as f64;
            Point {
                x: (self.x / c) as f32,
                y: (self.y / c) as f32,
                z: (self.z / c) as f32,
            }
        }
    }

    fn reset(&mut self) {
        self.x = 0.0;
        self.y = 0.0;
        self.z = 0.0;
        self.count = 0;
    }
}

fn find_nearest_centroid(point: &Point, centroids: &[Point]) -> usize {
    let mut best_idx = 0;
    let mut best_dist = f32::INFINITY;
    for (i, centroid) in centroids.iter().enumerate() {
        let dist = point.distance_squared(centroid);
        if dist < best_dist {
            best_dist = dist;
            best_idx = i;
        }
    }
    best_idx
}

fn initialize_centroids(k: usize, first_points: &[Point]) -> Vec<Point> {
    let mut centroids = Vec::new();

    if k == 0 {
        return centroids;
    }

    if first_points.is_empty() {
        for i in 0..k {
            let angle = (i as f32) * 2.0 * std::f32::consts::PI / (k as f32);
            centroids.push(Point {
                x: angle.cos(),
                y: angle.sin(),
                z: 0.0,
            });
        }
        return centroids;
    }

    centroids.push(first_points[0].clone());
    let mut rng = 12345u64;

    while centroids.len() < k {
        rng = rng.wrapping_mul(1103515245).wrapping_add(12345);
        let idx = (rng as usize) % first_points.len();
        let candidate = &first_points[idx];

        let min_dist = centroids
            .iter()
            .map(|c| candidate.distance_squared(c))
            .fold(f32::INFINITY, f32::min);

        if min_dist > 1e-6 {
            centroids.push(candidate.clone());
        } else if centroids.len() < k {
            let angle = (centroids.len() as f32) * 2.0 * std::f32::consts::PI / (k as f32);
            centroids.push(Point {
                x: angle.cos() + candidate.x,
                y: angle.sin() + candidate.y,
                z: candidate.z,
            });
        }
    }

    centroids.truncate(k);
    centroids
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut input = BufReader::new(stdin.lock());

    let mut header = [0u8; 8];
    input.read_exact(&mut header)?;

    let num_points = u32::from_le_bytes([header[0], header[1], header[2], header[3]]) as usize;
    let num_clusters = u32::from_le_bytes([header[4], header[5], header[6], header[7]]) as usize;

    if num_clusters == 0 || num_points == 0 {
        return Ok(());
    }

    let stdout = io::stdout();
    let mut output = BufWriter::new(stdout.lock());

    let mut cluster_assignments = vec![0u32; num_points];

    let sample_size = (100_000).min(num_points);
    let mut initial_points = Vec::new();
    let mut buffer = vec![0u8; 12 * 1024];

    let step = if num_points > sample_size {
        num_points / sample_size
    } else {
        1
    };

    let mut sampled = 0;
    let mut total_read = 0;

    loop {
        match input.read(&mut buffer)? {
            0 => break,
            bytes_read => {
                for chunk in buffer[..bytes_read].chunks(12) {
                    if chunk.len() == 12 {
                        if sampled % step == 0 && initial_points.len() < sample_size {
                            initial_points.push(Point::from_bytes(chunk));
                        }
                        sampled += 1;
                        total_read += 1;
                    }
                }
                if initial_points.len() >= sample_size {
                    break;
                }
            }
        }
    }

    let mut centroids = initialize_centroids(num_clusters, &initial_points);

    let batch_size = if num_points <= 10_000_000 {
        1_000_000
    } else if num_points <= 100_000_000 {
        500_000
    } else {
        100_000
    };

    let max_iterations = if num_points <= 10_000_000 {
        20
    } else if num_points <= 100_000_000 {
        10
    } else {
        5
    };

    for iteration in 0..max_iterations {
        let mut centroid_accum: Vec<Centroid> = Vec::new();
        for c in &centroids {
            let mut acc = Centroid::new(c);
            acc.reset();
            centroid_accum.push(acc);
        }

        let stdin = io::stdin();
        let mut input = BufReader::new(stdin.lock());
        let mut buffer = vec![0u8; 12 * batch_size];

        let mut point_idx = 0;
        loop {
            match input.read(&mut buffer)? {
                0 => break,
                bytes_read => {
                    for chunk in buffer[..bytes_read].chunks(12) {
                        if chunk.len() == 12 && point_idx < num_points {
                            let point = Point::from_bytes(chunk);
                            let nearest = find_nearest_centroid(&point, &centroids);
                            centroid_accum[nearest].add(&point);
                            cluster_assignments[point_idx] = nearest as u32;
                            point_idx += 1;
                        }
                    }
                }
            }
        }

        let old_centroids = centroids.clone();
        centroids.clear();

        for (i, acc) in centroid_accum.iter().enumerate() {
            if acc.count > 0 {
                centroids.push(acc.to_point());
            } else {
                if i < old_centroids.len() {
                    centroids.push(old_centroids[i].clone());
                } else {
                    centroids.push(Point { x: 0.0, y: 0.0, z: 0.0 });
                }
            }
        }

        let mut converged = true;
        if iteration > 1 {
            for (old, new) in old_centroids.iter().zip(centroids.iter()) {
                if old.distance_squared(new) > 1e-5 {
                    converged = false;
                    break;
                }
            }
        }

        if converged && iteration > 2 {
            break;
        }
    }

    let stdin = io::stdin();
    let mut input = BufReader::new(stdin.lock());
    let mut buffer = vec![0u8; 12 * batch_size];
    let mut point_idx = 0;

    loop {
        match input.read(&mut buffer)? {
            0 => break,
            bytes_read => {
                for chunk in buffer[..bytes_read].chunks(12) {
                    if chunk.len() == 12 && point_idx < num_points {
                        let point = Point::from_bytes(chunk);
                        let nearest = find_nearest_centroid(&point, &centroids);
                        cluster_assignments[point_idx] = nearest as u32;
                        point_idx += 1;
                    }
                }
            }
        }
    }

    for &cluster_id in &cluster_assignments {
        output.write_all(&cluster_id.to_le_bytes())?;
    }
    output.flush()?;

    Ok(())
}
