#include <iostream>
#include <vector>
#include <array>
#include <cmath>
#include <unordered_map>
#include <algorithm>
#include <set>
#include <queue>

using namespace std;

struct Vec3 {
    double x, y, z;

    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double t) const { return Vec3(x * t, y * t, z * t); }
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const { return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x); }
    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const { double len = length(); if (len < 1e-10) return Vec3(0,0,0); return *this * (1.0 / len); }
};

struct Face {
    int a, b, c;
};

struct Mesh {
    vector<Vec3> vertices;
    vector<Face> faces;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh meshA, meshB;

    // Read mesh A
    meshA.vertices.resize(VA);
    for (int i = 0; i < VA; i++) {
        cin >> meshA.vertices[i].x >> meshA.vertices[i].y >> meshA.vertices[i].z;
    }
    meshA.faces.resize(FA);
    for (int i = 0; i < FA; i++) {
        cin >> meshA.faces[i].a >> meshA.faces[i].b >> meshA.faces[i].c;
    }

    // Read mesh B
    meshB.vertices.resize(VB);
    for (int i = 0; i < VB; i++) {
        cin >> meshB.vertices[i].x >> meshB.vertices[i].y >> meshB.vertices[i].z;
    }
    meshB.faces.resize(FB);
    for (int i = 0; i < FB; i++) {
        cin >> meshB.faces[i].a >> meshB.faces[i].b >> meshB.faces[i].c;
    }

    // Compute bounding boxes
    auto getBbox = [](const Mesh& m) -> pair<Vec3, Vec3> {
        if (m.vertices.empty()) return {Vec3(0,0,0), Vec3(0,0,0)};
        Vec3 mn = m.vertices[0], mx = m.vertices[0];
        for (const auto& v : m.vertices) {
            mn.x = min(mn.x, v.x); mn.y = min(mn.y, v.y); mn.z = min(mn.z, v.z);
            mx.x = max(mx.x, v.x); mx.y = max(mx.y, v.y); mx.z = max(mx.z, v.z);
        }
        return {mn, mx};
    };

    auto [bmin_a, bmax_a] = getBbox(meshA);
    auto [bmin_b, bmax_b] = getBbox(meshB);

    // For union: combine all vertices from both meshes that are on the boundary
    vector<Vec3> result_verts;
    vector<Face> result_faces;

    // Simple approach: keep all faces from both meshes (this gives a union approximation)
    // For a proper CSG union, we'd need intersection detection and face culling

    // Add all vertices and faces from A
    int offset = 0;
    for (const auto& v : meshA.vertices) {
        result_verts.push_back(v);
    }

    for (const auto& f : meshA.faces) {
        result_faces.push_back(f);
    }

    // Add all vertices and faces from B (with offset)
    offset = meshA.vertices.size();
    for (const auto& v : meshB.vertices) {
        result_verts.push_back(v);
    }

    for (const auto& f : meshB.faces) {
        result_faces.push_back({f.a + offset, f.b + offset, f.c + offset});
    }

    // Remove duplicate vertices (within epsilon)
    const double EPS = 1e-9;
    vector<int> vert_map(result_verts.size());
    vector<Vec3> unique_verts;
    iota(vert_map.begin(), vert_map.end(), 0);

    for (int i = 0; i < (int)result_verts.size(); i++) {
        if (vert_map[i] != i) continue;

        for (int j = i + 1; j < (int)result_verts.size(); j++) {
            if (vert_map[j] != j) continue;

            Vec3 diff = result_verts[i] - result_verts[j];
            if (diff.dot(diff) < EPS * EPS) {
                vert_map[j] = i;
            }
        }
    }

    // Remap faces and collect unique vertices
    vector<int> new_idx(result_verts.size(), -1);
    int vid = 0;
    for (int i = 0; i < (int)result_verts.size(); i++) {
        if (new_idx[vert_map[i]] == -1) {
            new_idx[vert_map[i]] = vid++;
            unique_verts.push_back(result_verts[vert_map[i]]);
        }
    }

    vector<Face> final_faces;
    for (auto& f : result_faces) {
        int na = new_idx[vert_map[f.a]];
        int nb = new_idx[vert_map[f.b]];
        int nc = new_idx[vert_map[f.c]];
        if (na != nb && nb != nc && nc != na) {
            final_faces.push_back({na, nb, nc});
        }
    }

    // Output
    cout << unique_verts.size() << " " << final_faces.size() << "\n";

    for (const auto& v : unique_verts) {
        cout << fixed;
        cout.precision(10);
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    for (const auto& f : final_faces) {
        cout << f.a << " " << f.b << " " << f.c << "\n";
    }

    return 0;
}
