using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;

class OrbitalTSP
{
    const double MU = 398600.4418;

    struct Orbit
    {
        public double X, Y, Z, VX, VY, VZ;

        public double Distance(Orbit other)
        {
            double dx = X - other.X, dy = Y - other.Y, dz = Z - other.Z;
            return Math.Sqrt(dx*dx + dy*dy + dz*dz);
        }

        public double Speed()
        {
            return Math.Sqrt(VX*VX + VY*VY + VZ*VZ);
        }
    }

    static double CalcDeltaV(Orbit from, Orbit to)
    {
        double dist = from.Distance(to);
        double v1 = from.Speed(), v2 = to.Speed();
        return Math.Abs(v2 - v1) + dist / 5000.0;
    }

    static int[] SolveTSP(double[][] cost, int n)
    {
        if (n <= 8)
            return BruteForce(cost, n);
        if (n <= 15)
            return NearestNeighbor2Opt(cost, n);
        return NearestNeighbor(cost, n);
    }

    static int[] BruteForce(double[][] cost, int n)
    {
        int[] best = null;
        double bestCost = double.MaxValue;

        var perm = Enumerable.Range(0, n).ToList();
        PermGen(perm, 0, cost, ref best, ref bestCost);
        return best;
    }

    static void PermGen(List<int> p, int k, double[][] cost, ref int[] best, ref double bestCost)
    {
        if (k == p.Count)
        {
            double c = CalcTourCost(p, cost);
            if (c < bestCost)
            {
                bestCost = c;
                best = p.ToArray();
            }
            return;
        }
        for (int i = k; i < p.Count; i++)
        {
            Swap(p, k, i);
            PermGen(p, k + 1, cost, ref best, ref bestCost);
            Swap(p, k, i);
        }
    }

    static void Swap(List<int> p, int i, int j)
    {
        int t = p[i]; p[i] = p[j]; p[j] = t;
    }

    static double CalcTourCost(List<int> tour, double[][] cost)
    {
        double c = 0;
        for (int i = 0; i < tour.Count - 1; i++)
            c += cost[tour[i]][tour[i+1]];
        c += cost[tour[tour.Count-1]][tour[0]];
        return c;
    }

    static int[] NearestNeighbor(double[][] cost, int n)
    {
        bool[] vis = new bool[n];
        int[] tour = new int[n];
        tour[0] = 0;
        vis[0] = true;

        for (int i = 1; i < n; i++)
        {
            int cur = tour[i-1], best = -1;
            double minC = double.MaxValue;
            for (int j = 0; j < n; j++)
                if (!vis[j] && cost[cur][j] < minC)
                {
                    minC = cost[cur][j];
                    best = j;
                }
            tour[i] = best;
            vis[best] = true;
        }
        return tour;
    }

    static int[] NearestNeighbor2Opt(double[][] cost, int n)
    {
        int[] tour = NearestNeighbor(cost, n);

        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < n-1; i++)
            {
                for (int j = i+2; j < n; j++)
                {
                    int a = tour[i], b = tour[i+1], c = tour[j], d = tour[(j+1)%n];
                    double delta = cost[a][c] + cost[b][d] - cost[a][b] - cost[c][d];
                    if (delta < -1e-9)
                    {
                        Array.Reverse(tour, i+1, j-i);
                        improved = true;
                        break;
                    }
                }
                if (improved) break;
            }
        }
        return tour;
    }

    static void Main()
    {
        string[] lines = File.ReadAllLines("question.txt");
        var p1 = lines[0].Split();
        int n = int.Parse(p1[0]);

        var start = ParseOrbit(lines[1].Split());
        var orbits = new Orbit[n];
        for (int i = 0; i < n; i++)
            orbits[i] = ParseOrbit(lines[2+i].Split());

        double[][] cost = new double[n][];
        for (int i = 0; i < n; i++)
        {
            cost[i] = new double[n];
            for (int j = 0; j < n; j++)
                cost[i][j] = (i == j) ? 0 : CalcDeltaV(orbits[i], orbits[j]);
        }

        int[] tour = SolveTSP(cost, n);
        File.WriteAllText("answer.txt", string.Join(" ", tour));
    }

    static Orbit ParseOrbit(string[] parts)
    {
        return new Orbit { X = double.Parse(parts[0]), Y = double.Parse(parts[1]),
            Z = double.Parse(parts[2]), VX = double.Parse(parts[3]),
            VY = double.Parse(parts[4]), VZ = double.Parse(parts[5]) };
    }
}
