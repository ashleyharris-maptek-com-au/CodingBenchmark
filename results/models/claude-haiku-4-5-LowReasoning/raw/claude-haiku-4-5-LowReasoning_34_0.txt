import sys
import math
from typing import List, Tuple

def distance(p1: Tuple[float, float], p2: Tuple[float, float]) -> float:
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def calculate_route_distance(route: List[int], depot: Tuple[float, float], customers: List[Tuple[float, float]]) -> float:
    total = distance(depot, customers[route[0]])
    for i in range(len(route) - 1):
        total += distance(customers[route[i]], customers[route[i+1]])
    total += distance(customers[route[-1]], depot)
    return total

def calculate_total_distance(routes: List[List[int]], depot: Tuple[float, float], customers: List[Tuple[float, float]]) -> float:
    return sum(calculate_route_distance(route, depot, customers) for route in routes)

def nearest_neighbor_cvrp(depot: Tuple[float, float], customers: List[Tuple[float, float]], demands: List[int], capacity: int, num_vehicles: int) -> List[List[int]]:
    n = len(customers)
    unvisited = set(range(n))
    routes = []

    while unvisited:
        route = []
        current_load = 0
        current_pos = depot

        while unvisited:
            nearest = None
            nearest_dist = float('inf')

            for customer in unvisited:
                if current_load + demands[customer] <= capacity:
                    dist = distance(current_pos, customers[customer])
                    if dist < nearest_dist:
                        nearest_dist = dist
                        nearest = customer

            if nearest is None:
                break

            route.append(nearest)
            current_load += demands[nearest]
            current_pos = customers[nearest]
            unvisited.remove(nearest)

        if route:
            routes.append(route)

        if not unvisited:
            break

    return routes

def two_opt_within_route(route: List[int], depot: Tuple[float, float], customers: List[Tuple[float, float]]) -> List[int]:
    improved = True
    best_route = route[:]

    while improved:
        improved = False
        for i in range(len(best_route) - 1):
            for j in range(i + 2, len(best_route)):
                new_route = best_route[:i+1] + best_route[i+1:j+1][::-1] + best_route[j+1:]

                old_dist = calculate_route_distance(best_route, depot, customers)
                new_dist = calculate_route_distance(new_route, depot, customers)

                if new_dist < old_dist:
                    best_route = new_route
                    improved = True
                    break
            if improved:
                break

    return best_route

def two_opt_improve(routes: List[List[int]], depot: Tuple[float, float], customers: List[Tuple[float, float]]) -> List[List[int]]:
    improved_routes = [two_opt_within_route(route, depot, customers) for route in routes]
    return improved_routes

def merge_small_routes(routes: List[List[int]], depot: Tuple[float, float], customers: List[Tuple[float, float]], demands: List[int], capacity: int) -> List[List[int]]:
    if len(routes) <= 1:
        return routes

    merged = True
    while merged:
        merged = False
        for i in range(len(routes)):
            for j in range(i + 1, len(routes)):
                route_i_load = sum(demands[c] for c in routes[i])
                route_j_load = sum(demands[c] for c in routes[j])

                if route_i_load + route_j_load <= capacity:
                    merged_route = routes[i] + routes[j]
                    merged_route = two_opt_within_route(merged_route, depot, customers)

                    old_dist = calculate_route_distance(routes[i], depot, customers) + calculate_route_distance(routes[j], depot, customers)
                    new_dist = calculate_route_distance(merged_route, depot, customers)

                    if new_dist < old_dist:
                        routes[i] = merged_route
                        routes.pop(j)
                        merged = True
                        break
            if merged:
                break

    return routes

def solve_cvrp(n: int, k: int, capacity: int, depot: Tuple[float, float], customers: List[Tuple[float, float]], demands: List[int]) -> Tuple[float, List[List[int]]]:
    best_routes = nearest_neighbor_cvrp(depot, customers, demands, capacity, k)
    best_routes = two_opt_improve(best_routes, depot, customers)
    best_routes = merge_small_routes(best_routes, depot, customers, demands, capacity)
    best_distance = calculate_total_distance(best_routes, depot, customers)

    return best_distance, best_routes

def main():
    lines = sys.stdin.read().strip().split('\n')
    n, k, capacity = map(int, lines[0].split())
    depot_x, depot_y = map(float, lines[1].split())
    depot = (depot_x, depot_y)

    customers = []
    demands = []

    for i in range(2, 2 + n):
        parts = lines[i].split()
        x, y = float(parts[0]), float(parts[1])
        demand = int(parts[2])
        customers.append((x, y))
        demands.append(demand)

    total_distance, routes = solve_cvrp(n, k, capacity, depot, customers, demands)

    print(f"{total_distance:.2f}")
    print(len(routes))
    for route in routes:
        print(' '.join(map(str, route)))

if __name__ == "__main__":
    main()
