using System;
using System.Collections.Generic;
using System.Linq;
using System.Diagnostics;
using System.IO;

public class Vector3
{
    public double X, Y, Z;
    public Vector3(double x = 0, double y = 0, double z = 0) { X = x; Y = y; Z = z; }
    public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3 operator *(double s, Vector3 v) => new Vector3(s * v.X, s * v.Y, s * v.Z);
    public static Vector3 operator *(Vector3 v, double s) => s * v;
    public double Dot(Vector3 b) => X * b.X + Y * b.Y + Z * b.Z;
    public Vector3 Cross(Vector3 b) => new Vector3(Y * b.Z - Z * b.Y, Z * b.X - X * b.Z, X * b.Y - Y * b.X);
    public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
    public Vector3 Normalized() { var l = Length(); return l > 1e-10 ? (1.0 / l) * this : this; }
    public override string ToString() => $"{X:F3},{Y:F3},{Z:F3}";
}

public class Quaternion
{
    public double W, X, Y, Z;
    public Quaternion(double w = 1, double x = 0, double y = 0, double z = 0) { W = w; X = x; Y = y; Z = z; }
    public Quaternion Normalized()
    {
        var len = Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
        return len > 1e-10 ? new Quaternion(W / len, X / len, Y / len, Z / len) : this;
    }
    public Vector3 RotatePoint(Vector3 p)
    {
        var q = Normalized();
        var qc = new Quaternion(q.W, -q.X, -q.Y, -q.Z);
        var qp = new Quaternion(0, p.X, p.Y, p.Z);
        var res = Mul(Mul(q, qp), qc);
        return new Vector3(res.X, res.Y, res.Z);
    }
    static Quaternion Mul(Quaternion a, Quaternion b) => new Quaternion(
        a.W * b.W - a.X * b.X - a.Y * b.Y - a.Z * b.Z,
        a.W * b.X + a.X * b.W + a.Y * b.Z - a.Z * b.Y,
        a.W * b.Y - a.X * b.Z + a.Y * b.W + a.Z * b.X,
        a.W * b.Z + a.X * b.Y - a.Y * b.X + a.Z * b.W);
    public override string ToString() => $"{W:F6} {X:F6} {Y:F6} {Z:F6}";
}

public class Polyhedron
{
    public List<Vector3> Vertices;
    public List<(int, int, int)> Faces;
    public Vector3 Min, Max;
    public Polyhedron(List<Vector3> vertices, List<(int, int, int)> faces)
    {
        Vertices = vertices;
        Faces = faces;
        ComputeBounds();
    }
    void ComputeBounds()
    {
        Min = new Vector3(Vertices[0].X, Vertices[0].Y, Vertices[0].Z);
        Max = new Vector3(Vertices[0].X, Vertices[0].Y, Vertices[0].Z);
        foreach (var v in Vertices)
        {
            Min.X = Math.Min(Min.X, v.X); Max.X = Math.Max(Max.X, v.X);
            Min.Y = Math.Min(Min.Y, v.Y); Max.Y = Math.Max(Max.Y, v.Y);
            Min.Z = Math.Min(Min.Z, v.Z); Max.Z = Math.Max(Max.Z, v.Z);
        }
    }
    public (Vector3, Vector3) GetTransformedBounds(Vector3 pos, Quaternion rot)
    {
        var min = new Vector3(double.MaxValue, double.MaxValue, double.MaxValue);
        var max = new Vector3(double.MinValue, double.MinValue, double.MinValue);
        foreach (var v in Vertices)
        {
            var rotated = rot.RotatePoint(v);
            var transformed = pos + rotated;
            min.X = Math.Min(min.X, transformed.X); max.X = Math.Max(max.X, transformed.X);
            min.Y = Math.Min(min.Y, transformed.Y); max.Y = Math.Max(max.Y, transformed.Y);
            min.Z = Math.Min(min.Z, transformed.Z); max.Z = Math.Max(max.Z, transformed.Z);
        }
        return (min, max);
    }
}

public class Placement
{
    public Vector3 Position;
    public Quaternion Rotation;
    public Placement(Vector3 pos, Quaternion rot) { Position = pos; Rotation = rot; }
}

class Program
{
    static void Main()
    {
        var timer = Stopwatch.StartNew();

        var input = Console.In.ReadToEnd().Trim().Split('\n');
        var parts = input[0].Split(' ');
        int V = int.Parse(parts[0]), F = int.Parse(parts[1]);

        var vertices = new List<Vector3>();
        for (int i = 0; i < V; i++)
        {
            var coords = input[1 + i].Split(' ');
            vertices.Add(new Vector3(double.Parse(coords[0]), double.Parse(coords[1]), double.Parse(coords[2])));
        }

        var faces = new List<(int, int, int)>();
        for (int i = 0; i < F; i++)
        {
            var indices = input[1 + V + i].Split(' ');
            faces.Add((int.Parse(indices[0]), int.Parse(indices[1]), int.Parse(indices[2])));
        }

        var dimParts = input[1 + V + F].Split(' ');
        int W = int.Parse(dimParts[0]), H = int.Parse(dimParts[1]), D = int.Parse(dimParts[2]);

        var poly = new Polyhedron(vertices, faces);
        var placements = new List<Placement>();

        var rotations = GenerateRotations(W, H, D);
        var step = EstimateGridStep(W, H, D);

        for (double x = 0; x < W && timer.ElapsedMilliseconds < 290000; x += step)
            for (double y = 0; y < H && timer.ElapsedMilliseconds < 290000; y += step)
                for (double z = 0; z < D && timer.ElapsedMilliseconds < 290000; z += step)
                    foreach (var rot in rotations)
                    {
                        var pos = new Vector3(x, y, z);
                        if (CanPlace(poly, pos, rot, placements, W, H, D))
                            placements.Add(new Placement(pos, rot));
                    }

        Console.WriteLine(placements.Count);
        foreach (var p in placements)
            Console.WriteLine($"{p.Position.X:F1} {p.Position.Y:F1} {p.Position.Z:F1} {p.Rotation}");
    }

    static List<Quaternion> GenerateRotations(int W, int H, int D)
    {
        var rots = new List<Quaternion> { new Quaternion(1, 0, 0, 0) };
        var size = Math.Max(W, Math.Max(H, D));

        if (size < 50)
        {
            for (int deg = 45; deg < 360; deg += 45)
            {
                double rad = deg * Math.PI / 180;
                double s = Math.Sin(rad / 2);
                rots.Add(new Quaternion(Math.Cos(rad / 2), s, 0, 0));
                rots.Add(new Quaternion(Math.Cos(rad / 2), 0, s, 0));
                rots.Add(new Quaternion(Math.Cos(rad / 2), 0, 0, s));
            }
        }
        else if (size < 100)
        {
            for (int deg = 90; deg < 360; deg += 90)
            {
                double rad = deg * Math.PI / 180;
                double s = Math.Sin(rad / 2);
                rots.Add(new Quaternion(Math.Cos(rad / 2), s, 0, 0));
                rots.Add(new Quaternion(Math.Cos(rad / 2), 0, s, 0));
                rots.Add(new Quaternion(Math.Cos(rad / 2), 0, 0, s));
            }
        }

        return rots;
    }

    static double EstimateGridStep(int W, int H, int D)
    {
        var maxDim = Math.Max(W, Math.Max(H, D));
        if (maxDim < 20) return 1.0;
        if (maxDim < 50) return 2.0;
        if (maxDim < 100) return 5.0;
        return Math.Max(10.0, maxDim / 20.0);
    }

    static bool CanPlace(Polyhedron poly, Vector3 pos, Quaternion rot, List<Placement> placed, int W, int H, int D)
    {
        var (min, max) = poly.GetTransformedBounds(pos, rot);

        if (min.X < 0 || max.X > W || min.Y < 0 || max.Y > H || min.Z < 0 || max.Z > D)
            return false;

        foreach (var p in placed)
        {
            if (Overlaps(poly, pos, rot, p.Position, p.Rotation))
                return false;
        }

        return true;
    }

    static bool Overlaps(Polyhedron poly, Vector3 pos1, Quaternion rot1, Vector3 pos2, Quaternion rot2)
    {
        var (min1, max1) = poly.GetTransformedBounds(pos1, rot1);
        var (min2, max2) = poly.GetTransformedBounds(pos2, rot2);

        return !(max1.X < min2.X || max2.X < min1.X ||
                 max1.Y < min2.Y || max2.Y < min1.Y ||
                 max1.Z < min2.Z || max2.Z < min1.Z);
    }
}
