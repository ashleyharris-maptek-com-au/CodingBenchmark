using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    private char[,] board;
    private bool[,] isSafe;
    private bool[,] isMine;
    private int height;
    private int width;

    static void Main()
    {
        var solver = new MinesweeperSolver();
        solver.Solve();
    }

    void Solve()
    {
        // Read input
        string[] firstLine = Console.ReadLine().Split();
        height = int.Parse(firstLine[0]);
        width = int.Parse(firstLine[1]);

        board = new char[height, width];
        isSafe = new bool[height, width];
        isMine = new bool[height, width];

        for (int i = 0; i < height; i++)
        {
            string line = Console.ReadLine();
            for (int j = 0; j < width; j++)
            {
                board[i, j] = j < line.Length ? line[j] : ' ';
                if (board[i, j] == '*')
                    isMine[i, j] = true;
            }
        }

        // Constraint propagation
        ConstraintPropagation();

        // Collect safe cells that are unknown and provably safe
        var safeCells = new List<(int, int)>();
        for (int i = 0; i < height; i++)
        {
            for (int j = 0; j < width; j++)
            {
                if (board[i, j] == ' ' && isSafe[i, j])
                    safeCells.Add((j, i)); // Output as (x, y) = (col, row)
            }
        }

        // Output
        Console.WriteLine(safeCells.Count);
        foreach (var (x, y) in safeCells.OrderBy(c => c.Item1).ThenBy(c => c.Item2))
        {
            Console.WriteLine($"{x} {y}");
        }
    }

    void ConstraintPropagation()
    {
        bool changed = true;
        int iterations = 0;
        int maxIterations = height + width;

        while (changed && iterations < maxIterations)
        {
            changed = false;
            iterations++;

            // Process each numbered cell
            for (int i = 0; i < height; i++)
            {
                for (int j = 0; j < width; j++)
                {
                    if (char.IsDigit(board[i, j]))
                    {
                        if (ProcessCell(i, j))
                            changed = true;
                    }
                }
            }
        }
    }

    bool ProcessCell(int row, int col)
    {
        int requiredMines = board[row, col] - '0';
        var neighbors = GetNeighbors(row, col);

        int knownMines = 0;
        int unknowns = 0;
        var unknownCells = new List<(int, int)>();

        // Count known mines and unknown cells
        foreach (var (nr, nc) in neighbors)
        {
            if (isMine[nr, nc])
            {
                knownMines++;
            }
            else if (board[nr, nc] == ' ' && !isSafe[nr, nc])
            {
                unknowns++;
                unknownCells.Add((nr, nc));
            }
        }

        bool changed = false;
        int remainingMines = requiredMines - knownMines;

        // If all mines are accounted for, all remaining unknowns are safe
        if (remainingMines == 0)
        {
            foreach (var (nr, nc) in unknownCells)
            {
                if (!isSafe[nr, nc])
                {
                    isSafe[nr, nc] = true;
                    changed = true;
                }
            }
        }
        // If unknowns exactly equal remaining mines, all unknowns are mines
        else if (remainingMines == unknowns && unknowns > 0)
        {
            foreach (var (nr, nc) in unknownCells)
            {
                if (!isMine[nr, nc])
                {
                    isMine[nr, nc] = true;
                    changed = true;
                }
            }
        }

        return changed;
    }

    List<(int, int)> GetNeighbors(int row, int col)
    {
        var neighbors = new List<(int, int)>();
        for (int dr = -1; dr <= 1; dr++)
        {
            for (int dc = -1; dc <= 1; dc++)
            {
                if (dr == 0 && dc == 0) continue;
                int nr = row + dr;
                int nc = col + dc;
                if (nr >= 0 && nr < height && nc >= 0 && nc < width)
                    neighbors.Add((nr, nc));
            }
        }
        return neighbors;
    }
}
