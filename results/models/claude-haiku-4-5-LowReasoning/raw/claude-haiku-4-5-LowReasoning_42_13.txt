OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %color_in %gl_out %outColor

OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos_in Location 0
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_5 = OpConstant %float 0.5
%float_0_008 = OpConstant %float 0.008
%float_neg_0_008 = OpConstant %float -0.008

%fn_main = OpTypeFunction %void

%main = OpFunction %void None %fn_main
%entry_label = OpLabel

; Load main triangle vertices (indices 0, 2, 4)
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr

; Load colors
%col0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_in_v3 %color_in %int_4
%col4 = OpLoad %v3float %col4_ptr

; Edge 0→2: compute edge vector and perpendicular
%edge_02 = OpFSub %v4float %pos2 %pos0
%edge_02_xy = OpVectorShuffle %v2float %edge_02 %edge_02 0 1
%edge_02_len_sq = OpDot %float %edge_02_xy %edge_02_xy
%edge_02_len = OpSqrt %float %edge_02_len_sq
%edge_02_normalized = OpFDiv %v2float %edge_02_xy %edge_02_len
%edge_02_perp = OpVectorShuffle %v2float %edge_02_normalized %edge_02_normalized 1 0
%edge_02_perp_x_neg = OpFNegate %float %edge_02_perp
%edge_02_perp = OpVectorShuffle %v2float %edge_02_perp_x_neg %edge_02_perp 0 1
%edge_02_perp_thick_pos = OpVectorTimesScalar %v2float %edge_02_perp %float_0_008
%edge_02_perp_thick_neg = OpVectorTimesScalar %v2float %edge_02_perp %float_neg_0_008

%pos0_w = OpVectorShuffle %float %pos0 %pos0 3 3
%pos2_w = OpVectorShuffle %float %pos2 %pos2 3 3
%perp_0_pos_scaled = OpVectorTimesScalar %v2float %edge_02_perp_thick_pos %pos0_w
%perp_0_neg_scaled = OpVectorTimesScalar %v2float %edge_02_perp_thick_neg %pos0_w

; Emit 4 vertices for edge 0→2
%v0_xy_part = OpVectorShuffle %v2float %pos0 %pos0 0 1
%v0_xy_new = OpFAdd %v2float %v0_xy_part %perp_0_pos_scaled
%v0_new = OpVectorShuffle %v4float %v0_xy_new %pos0 2 3 2 3
%col02 = OpFMul %v3float %col0 %float_0_5
%col02_tmp = OpFMul %v3float %col2 %float_0_5
%col02_avg = OpFAdd %v3float %col02 %col02_tmp
OpStore %outColor %col02_avg
%v0_pos_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v0_pos_out %v0_new
OpEmitVertex

%v1_xy_new = OpFAdd %v2float %v0_xy_part %perp_0_neg_scaled
%v1_new = OpVectorShuffle %v4float %v1_xy_new %pos0 2 3 2 3
OpStore %outColor %col02_avg
OpStore %v0_pos_out %v1_new
OpEmitVertex

%pos2_w_val = OpVectorShuffle %float %pos2 %pos2 3 3
%perp_2_pos_scaled = OpVectorTimesScalar %v2float %edge_02_perp_thick_pos %pos2_w_val
%v2_xy_part = OpVectorShuffle %v2float %pos2 %pos2 0 1
%v2_xy_new = OpFAdd %v2float %v2_xy_part %perp_2_pos_scaled
%v2_new = OpVectorShuffle %v4float %v2_xy_new %pos2 2 3 2 3
OpStore %outColor %col02_avg
OpStore %v0_pos_out %v2_new
OpEmitVertex

%perp_2_neg_scaled = OpVectorTimesScalar %v2float %edge_02_perp_thick_neg %pos2_w_val
%v3_xy_new = OpFAdd %v2float %v2_xy_part %perp_2_neg_scaled
%v3_new = OpVectorShuffle %v4float %v3_xy_new %pos2 2 3 2 3
OpStore %outColor %col02_avg
OpStore %v0_pos_out %v3_new
OpEmitVertex

OpEndPrimitive

; Edge 2→4
%edge_24 = OpFSub %v4float %pos4 %pos2
%edge_24_xy = OpVectorShuffle %v2float %edge_24 %edge_24 0 1
%edge_24_len_sq = OpDot %float %edge_24_xy %edge_24_xy
%edge_24_len = OpSqrt %float %edge_24_len_sq
%edge_24_normalized = OpFDiv %v2float %edge_24_xy %edge_24_len
%edge_24_perp_swap = OpVectorShuffle %v2float %edge_24_normalized %edge_24_normalized 1 0
%edge_24_perp_x_neg = OpFNegate %float %edge_24_perp_swap
%edge_24_perp = OpVectorShuffle %v2float %edge_24_perp_x_neg %edge_24_perp_swap 0 1
%edge_24_perp_thick_pos = OpVectorTimesScalar %v2float %edge_24_perp %float_0_008
%edge_24_perp_thick_neg = OpVectorTimesScalar %v2float %edge_24_perp %float_neg_0_008

%pos2_w_2 = OpVectorShuffle %float %pos2 %pos2 3 3
%pos4_w = OpVectorShuffle %float %pos4 %pos4 3 3
%perp_2_pos_scaled_24 = OpVectorTimesScalar %v2float %edge_24_perp_thick_pos %pos2_w_2
%perp_2_neg_scaled_24 = OpVectorTimesScalar %v2float %edge_24_perp_thick_neg %pos2_w_2

%v4_xy_part = OpVectorShuffle %v2float %pos2 %pos2 0 1
%v4_xy_new = OpFAdd %v2float %v4_xy_part %perp_2_pos_scaled_24
%v4_new = OpVectorShuffle %v4float %v4_xy_new %pos2 2 3 2 3
%col24 = OpFMul %v3float %col2 %float_0_5
%col24_tmp = OpFMul %v3float %col4 %float_0_5
%col24_avg = OpFAdd %v3float %col24 %col24_tmp
OpStore %outColor %col24_avg
OpStore %v0_pos_out %v4_new
OpEmitVertex

%v5_xy_new = OpFAdd %v2float %v4_xy_part %perp_2_neg_scaled_24
%v5_new = OpVectorShuffle %v4float %v5_xy_new %pos2 2 3 2 3
OpStore %outColor %col24_avg
OpStore %v0_pos_out %v5_new
OpEmitVertex

%perp_4_pos_scaled = OpVectorTimesScalar %v2float %edge_24_perp_thick_pos %pos4_w
%v6_xy_part = OpVectorShuffle %v2float %pos4 %pos4 0 1
%v6_xy_new = OpFAdd %v2float %v6_xy_part %perp_4_pos_scaled
%v6_new = OpVectorShuffle %v4float %v6_xy_new %pos4 2 3 2 3
OpStore %outColor %col24_avg
OpStore %v0_pos_out %v6_new
OpEmitVertex

%perp_4_neg_scaled = OpVectorTimesScalar %v2float %edge_24_perp_thick_neg %pos4_w
%v7_xy_new = OpFAdd %v2float %v6_xy_part %perp_4_neg_scaled
%v7_new = OpVectorShuffle %v4float %v7_xy_new %pos4 2 3 2 3
OpStore %outColor %col24_avg
OpStore %v0_pos_out %v7_new
OpEmitVertex

OpEndPrimitive

; Edge 4→0
%edge_40 = OpFSub %v4float %pos0 %pos4
%edge_40_xy = OpVectorShuffle %v2float %edge_40 %edge_40 0 1
%edge_40_len_sq = OpDot %float %edge_40_xy %edge_40_xy
%edge_40_len = OpSqrt %float %edge_40_len_sq
%edge_40_normalized = OpFDiv %v2float %edge_40_xy %edge_40_len
%edge_40_perp_swap = OpVectorShuffle %v2float %edge_40_normalized %edge_40_normalized 1 0
%edge_40_perp_x_neg = OpFNegate %float %edge_40_perp_swap
%edge_40_perp = OpVectorShuffle %v2float %edge_40_perp_x_neg %edge_40_perp_swap 0 1
%edge_40_perp_thick_pos = OpVectorTimesScalar %v2float %edge_40_perp %float_0_008
%edge_40_perp_thick_neg = OpVectorTimesScalar %v2float %edge_40_perp %float_neg_0_008

%pos4_w_2 = OpVectorShuffle %float %pos4 %pos4 3 3
%pos0_w_2 = OpVectorShuffle %float %pos0 %pos0 3 3
%perp_4_pos_scaled_40 = OpVectorTimesScalar %v2float %edge_40_perp_thick_pos %pos4_w_2
%perp_4_neg_scaled_40 = OpVectorTimesScalar %v2float %edge_40_perp_thick_neg %pos4_w_2

%v8_xy_part = OpVectorShuffle %v2float %pos4 %pos4 0 1
%v8_xy_new = OpFAdd %v2float %v8_xy_part %perp_4_pos_scaled_40
%v8_new = OpVectorShuffle %v4float %v8_xy_new %pos4 2 3 2 3
%col40 = OpFMul %v3float %col4 %float_0_5
%col40_tmp = OpFMul %v3float %col0 %float_0_5
%col40_avg = OpFAdd %v3float %col40 %col40_tmp
OpStore %outColor %col40_avg
OpStore %v0_pos_out %v8_new
OpEmitVertex

%v9_xy_new = OpFAdd %v2float %v8_xy_part %perp_4_neg_scaled_40
%v9_new = OpVectorShuffle %v4float %v9_xy_new %pos4 2 3 2 3
OpStore %outColor %col40_avg
OpStore %v0_pos_out %v9_new
OpEmitVertex

%perp_0_pos_scaled_40 = OpVectorTimesScalar %v2float %edge_40_perp_thick_pos %pos0_w_2
%v10_xy_part = OpVectorShuffle %v2float %pos0 %pos0 0 1
%v10_xy_new = OpFAdd %v2float %v10_xy_part %perp_0_pos_scaled_40
%v10_new = OpVectorShuffle %v4float %v10_xy_new %pos0 2 3 2 3
OpStore %outColor %col40_avg
OpStore %v0_pos_out %v10_new
OpEmitVertex

%perp_0_neg_scaled_40 = OpVectorTimesScalar %v2float %edge_40_perp_thick_neg %pos0_w_2
%v11_xy_new = OpFAdd %v2float %v10_xy_part %perp_0_neg_scaled_40
%v11_new = OpVectorShuffle %v4float %v11_xy_new %pos0 2 3 2 3
OpStore %outColor %col40_avg
OpStore %v0_pos_out %v11_new
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
