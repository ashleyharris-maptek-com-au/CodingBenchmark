#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<vector<int>> adj;
vector<int> best_set;

// Greedy upper bound: compute maximum greedy independent set on remaining vertices
int greedy_ub(const vector<bool>& removed) {
    vector<bool> used(n, false);
    int count = 0;
    for (int u = 0; u < n; u++) {
        if (!removed[u] && !used[u]) {
            count++;
            used[u] = true;
            for (int v : adj[u]) {
                used[v] = true;
            }
        }
    }
    return count;
}

void solve(vector<int>& current, vector<bool>& removed) {
    // Pruning: if current size + upper bound <= best found, return
    int potential = (int)current.size() + greedy_ub(removed);
    if (potential <= (int)best_set.size()) return;

    // Find next vertex with minimum degree for better branching
    int next = -1;
    int min_degree = n + 1;
    for (int i = 0; i < n; i++) {
        if (!removed[i]) {
            int degree = 0;
            for (int j : adj[i]) {
                if (!removed[j]) degree++;
            }
            if (degree < min_degree) {
                min_degree = degree;
                next = i;
            }
        }
    }

    // Base case: all vertices considered
    if (next == -1) {
        if (current.size() > best_set.size()) {
            best_set = current;
        }
        return;
    }

    // Branch 1: include next vertex (remove it and all neighbors)
    current.push_back(next);
    vector<bool> new_removed = removed;
    new_removed[next] = true;
    for (int neighbor : adj[next]) {
        new_removed[neighbor] = true;
    }
    solve(current, new_removed);
    current.pop_back();

    // Branch 2: exclude next vertex
    removed[next] = true;
    solve(current, removed);
    removed[next] = false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    adj.assign(n, vector<int>());

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Special case: empty graph (no edges)
    if (m == 0) {
        for (int i = 0; i < n; i++) {
            best_set.push_back(i);
        }
    } else {
        vector<int> current;
        vector<bool> removed(n, false);
        solve(current, removed);
    }

    // Output
    cout << best_set.size() << "\n";
    for (int i = 0; i < best_set.size(); i++) {
        if (i > 0) cout << " ";
        cout << best_set[i];
    }
    cout << "\n";

    return 0;
}
