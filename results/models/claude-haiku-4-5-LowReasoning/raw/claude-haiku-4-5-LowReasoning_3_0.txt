use std::io::{self, BufRead};

#[derive(Clone)]
struct Node {
    x: f64,
    y: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();

    let mut line = String::new();
    reader.read_line(&mut line).unwrap();
    let parts: Vec<&str> = line.trim().split_whitespace().collect();
    let n: usize = parts[0].parse().unwrap();
    let m: usize = parts[1].parse().unwrap();

    let mut edges = Vec::new();
    for _ in 0..m {
        line.clear();
        reader.read_line(&mut line).unwrap();
        let parts: Vec<&str> = line.trim().split_whitespace().collect();
        if parts.len() >= 2 {
            let u: usize = parts[0].parse().unwrap_or(0);
            let v: usize = parts[1].parse().unwrap_or(0);
            if u < n && v < n && u != v {
                edges.push((u, v));
            }
        }
    }

    let mut nodes = vec![Node { x: 0.0, y: 0.0 }; n];
    let radius = ((n as f64).sqrt()) * 2.0;

    // Initialize positions in a circle
    for i in 0..n {
        let angle = 2.0 * std::f64::consts::PI * (i as f64 / n as f64);
        nodes[i].x = radius * angle.cos();
        nodes[i].y = radius * angle.sin();
    }

    // Adaptive iteration count based on graph size
    let iterations = if n <= 100 {
        100
    } else if n <= 1000 {
        50
    } else if n <= 10000 {
        20
    } else {
        5
    };

    let k = (radius * 2.0).sqrt();
    let c_attr = 0.05;
    let c_rep = 0.3;

    for iter in 0..iterations {
        let mut fx = vec![0.0; n];
        let mut fy = vec![0.0; n];

        // Repulsive forces (O(n²) - only for graphs with ≤ 2000 nodes)
        if n <= 2000 {
            for i in 0..n {
                for j in (i+1)..n {
                    let dx = nodes[j].x - nodes[i].x;
                    let dy = nodes[j].y - nodes[i].y;
                    let dist_sq = dx * dx + dy * dy + 0.001;
                    let dist = dist_sq.sqrt();
                    let repel = -c_rep * k * k / (dist * dist);

                    let fx_contrib = repel * dx / dist;
                    let fy_contrib = repel * dy / dist;

                    fx[i] += fx_contrib;
                    fy[i] += fy_contrib;
                    fx[j] -= fx_contrib;
                    fy[j] -= fy_contrib;
                }
            }
        }

        // Attractive forces along edges (O(m))
        for (u, v) in &edges {
            let dx = nodes[*v].x - nodes[*u].x;
            let dy = nodes[*v].y - nodes[*u].y;
            let dist = (dx * dx + dy * dy).sqrt();

            if dist > 1e-6 {
                let attract = c_attr * dist / k;
                let fx_contrib = attract * dx / dist;
                let fy_contrib = attract * dy / dist;

                fx[*u] += fx_contrib;
                fy[*u] += fy_contrib;
                fx[*v] -= fx_contrib;
                fy[*v] -= fy_contrib;
            }
        }

        // Update positions with temperature-based cooling
        let cooling_factor = 1.0 - iter as f64 / iterations as f64;
        let max_displacement = radius * cooling_factor * 0.1;

        for i in 0..n {
            let mag = (fx[i] * fx[i] + fy[i] * fy[i]).sqrt();
            if mag > 1e-6 {
                let displacement = mag.min(max_displacement);
                nodes[i].x += (fx[i] / mag) * displacement;
                nodes[i].y += (fy[i] / mag) * displacement;
            }
        }
    }

    // Output
    println!("{}", n);
    for node in nodes {
        println!("{} {}", node.x, node.y);
    }
}
