// Utility: convert square index [0..63] to algebraic like "e4"
static string sqToAlg(int s){
    int f = s % 8;
    int r = s / 8;
    string t;
    t.push_back(char('a' + f));
    t.push_back(char('1' + r));
    return t;
}

// Utility: convert "e4" to square index
static int algToSq(const string& s){
    int f = s[0] - 'a';
    int r = s[1] - '1';
    return r * 8 + f;
}

// Move representation
struct Move {
    int from, to;
    bool capture = false;
    bool enPassant = false;
    bool castle = false;
    int castleSide = -1; // 0: kingside, 1: queenside
    char promo = 0;      // 'Q','R','B','N' for promotions
    char pieceMoved = 0; // 'P','N','B','R','Q','K' for white, lowercase for black

// Board representation and engine
struct Board {
    // 64 squares: '.' = empty, otherwise piece letter
    array<char,64> sq;

    int epSquare = -1;

    inline bool isBlack(char p) const { return islower((unsigned char)p) != 0; }

    // Convert from FEN to board
    static Board fromFEN(const string& fen){
        Board b;
        b.sq.fill('.'); // default empty
        // Split FEN: piece placement, active color, castling, en passant, etc.
        string piecePlacement;
        string rest;
        {
            // First line (two lines input) will pass fen as a single line
            // Extract piece placement (before first space)
            size_t pos = fen.find(' ');
            piecePlacement = fen.substr(0, pos);
            rest = fen.substr(pos);
        }

        // Parse piece placement (8 ranks from 8 to 1)
        vector<string> ranks;
        {
            string cur;
            for(char c : piecePlacement){
                if(c == '/'){ ranks.push_back(cur); cur.clear(); }
                else cur.push_back(c);
            }
            ranks.push_back(cur);
        }
        // ranks should have 8 entries
        int r = 7; // start from rank 8 -> index 7
        for(const string& row : ranks){
            int f = 0;
            for(char ch : row){
                if(isdigit((unsigned char)ch)){
                    int skip = ch - '0';
                    f += skip;
                } else {
                    b.sq[r*8 + f] = ch;
                    f++;
                }
            }
            r--;
        }

        // Parse the rest tokens
        // e.g. " w KQkq - 0 1"
        // We'll use stringstream
        {
            // remove leading spaces
            while(!rest.empty() && rest[0]==' ') rest.erase(rest.begin());
            // We'll parse tokens
            string t;
            vector<string> toks;
            for(size_t i=0;i<rest.size();++i){
                if(rest[i]==' '){
                    if(!t.empty()){ toks.push_back(t); t.clear(); }
                } else {
                    t.push_back(rest[i]);
                }
            }
            if(!t.empty()) toks.push_back(t);

            // tokens expected: color, castling, en-passant, halfmove, fullmove
            if(toks.size()>=1){
                b.whiteToMove = (toks[0] == "w");
            }
            if(toks.size()>=2){
                string cast = toks[1];
                b.wKSide = cast.find('K') != string::npos;
                b.wQSide = cast.find('Q') != string::npos;
                b.bKSide = cast.find('k') != string::npos;
                b.bQSide = cast.find('q') != string::npos;
            }
            if(toks.size()>=3){
                string ep = toks[2];
                if(ep == "-") b.epSquare = -1;
                else{
                    int file = ep[0]-'a';
                    int rank = ep[1]-'1';
                    b.epSquare = rank*8 + file;
                }
            } else {
                b.epSquare = -1;
            }
            // halfmove/fullmove: ignored for core engine
        }

        // If no ep given, ensure it's -1
        if (b.epSquare < -1) b.epSquare = -1;

        // Ensure castling rights reflect presence in FEN (we recompute dynamically enough)
        // But keep something reasonable:
        // If FEN castling string was provided but we didn't set anything, it's fine.

        // Ensure board is valid
        return b;
    }

    // Convert current position to a simple ASCII board (debug)

                s += sq[r*8+f];
                s += ' ';

        }
        return s;
    }

    int evaluate() const {
        int score = 0;
        for(int i=0;i<64;++i){
            char p = sq[i];
            if(p=='.') continue;
            int val = 0;
            switch(p){
                case 'P': val = 100; break;
                case 'N': val = 320; break;
                case 'B': val = 330; break;
                case 'R': val = 500; break;
                case 'Q': val = 900; break;
                case 'K': val = 0; break;
                case 'p': val = -100; break;
                case 'n': val = -320; break;
                case 'b': val = -330; break;
                case 'r': val = -500; break;
                case 'q': val = -900; break;
                case 'k': val = 0; break;
            }
            score += val;
        }
        return score;
    }

        // Actually the easiest and reliable way is to implement a direct attacker search rather than reverse-mapping:
        // We'll scan potential attacker pieces around the square:

        // Knight attacks: from 8 possible knight-positions
        const int kn_dx[8] = {1,2,2,1,-1,-2,-2,-1};
        const int kn_dy[8] = {2,1,-1,-2,-2,-1,1,2};
        for(int i=0;i<8;i++){
            int nf = f + kn_dx[i];
            int nr = r + kn_dy[i];
            if(nf>=0 && nf<8 && nr>=0 && nr<8){
                int from = nr*8 + nf;
                if (byWhite){

                char p = sq[from];
                if (byWhite && p == 'N') return true;

                if(nr>=0 && nr<8 && nf>=0 && nf<8){
                    int from = nr*8 + nf;
                    char p = sq[from];
                    if (byWhite && p == 'K') return true;
                    if (!byWhite && p == 'k') return true;
                }
            }
        }

        // Sliding: bishop/queen diagonals
        const int diagDx[4] = {1,1,-1,-1};
        const int diagDy[4] = {1,-1,1,-1};
        for(int dir=0; dir<4; ++dir){
            int nr = r, nf = f;
            while(true){
                nr += diagDy[dir];

                if(nr<0||nr>=8||nf<0||nf>=8) break;
                int from = nr*8 + nf;
                char p = sq[from];
                if(p != '.'){

        // Sliding: rook/queen orthogonals
        const int orthDx[4] = {1,-1,0,0};

        for(int dir=0; dir<4; ++dir){

                    if (!byWhite && (p=='r' || p=='q')) return true;
                    break;
                }
            }
        }

        // Pawns: do a final, explicit check to avoid edge cases
        // White pawns attacking sq from (sq.r-1, sq.f-1) and (sq.r-1, sq.f+1)
        if (byWhite){
            if (r-1 >= 0 && f-1 >= 0){
                int from = (r-1)*8 + (f-1);
                if (sq[from] == 'P') return true;

                int from = (r+1)*8 + (f+1);
                if (sq[from] == 'p') return true;
            }
        }

        return false;
    }

    // Check if current side to move is in check
    bool isInCheck() const {
        // Find king of side to move
        char kingChar = whiteToMove ? 'K' : 'k';
        int kingSq = -1;
        for(int i=0;i<64;++i){
            if(sq[i] == kingChar){ kingSq = i; break; }
        }
        if(kingSq==-1) return true; // something wrong -> treat as in check
        // Opponent attacks this square
        return isSquareAttacked(kingSq, !whiteToMove);
    }

    // Generate all pseudo-legal moves for the side to move
    vector<Move> generateMoves() const {
        vector<Move> moves;
        bool white = whiteToMove;

        // Helper lambda to add a move
        auto addMove = [&](int from, int to, bool cap, bool enPassantFlag, bool castleFlag, int castleSideFlag, char promo, char pieceMoved){
            Move m;
            m.from = from; m.to = to;
            m.capture = cap;
            m.enPassant = enPassantFlag;
            m.castle = castleFlag;
            m.castleSide = castleSideFlag;
            m.promo = promo;
            m.pieceMoved = pieceMoved;
            moves.push_back(m);
        };

        // First, potential castling moves (we will validate safety at generation time)
        // White castling: K on e1 (4) and R on h1 (7)
        if (white){
            // Kingside
            if (sq[4] == 'K' && sq[7] == 'R' && sq[5] == '.' && sq[6] == '.') {
                // Squares not attacked (e1, f1, g1) by Black
                if (!isSquareAttacked(4, false) && !isSquareAttacked(5, false) && !isSquareAttacked(6, false)) {
                    addMove(4, 6, false, false, true, 0, 0, 'K');
                }
            }
            // Queenside
            if (sq[4] == 'K' && sq[0] == 'R' && sq[1] == '.' && sq[2] == '.' && sq[3] == '.') {
                if (!isSquareAttacked(4, false) && !isSquareAttacked(3, false) && !isSquareAttacked(2, false)) {
                    addMove(4, 2, false, false, true, 1, 0, 'K');
                }
            }
        } else {
            // Black castling: on e8 (60)
            if (sq[60] == 'k' && sq[63] == 'r' && sq[61] == '.' && sq[62] == '.') {
                if (!isSquareAttacked(60, true) && !isSquareAttacked(61, true) && !isSquareAttacked(62, true)) {
                    addMove(60, 62, false, false, true, 0, 0, 'k');
                }
            }
            // Queenside
            if (sq[60] == 'k' && sq[56] == 'r' && sq[57] == '.' && sq[58] == '.' && sq[59] == '.') {
                if (!isSquareAttacked(60, true) && !isSquareAttacked(59, true) && !isSquareAttacked(58, true)) {
                    addMove(60, 58, false, false, true, 1, 0, 'k');
                }
            }
        }

        // Pawn moves
        for(int i=0;i<64;++i){
            char p = sq[i];
            if(p == '.') continue;
            bool pieceWhite = isWhite(p);
            if (pieceWhite != white) continue;

            int r = i / 8, f = i % 8;
            if (p == 'P'){ // White pawn
                // one step
                int to = i + 8;
                if (to < 64 && sq[to] == '.'){
                    // Promotion?
                    if (to/8 == 7){
                        for(char promo : {'Q','R','B','N'}){
                            addMove(i, to, false, false, false, -1, promo, p);
                        }
                    } else {
                        addMove(i, to, false, false, false, -1, 0, p);
                    }
                }
                // two steps
                if (r == 1){
                    int to2 = i + 16;
                    if (sq[i+8] == '.' && sq[to2] == '.'){
                        addMove(i, to2, false, false, false, -1, 0, p);
                    }
                }
                // captures
                // left capture
                int left = i + 7;
                if (f > 0 && left < 64){
                    if (sq[left] != '.' && isWhite(sq[left]) == false){
                        if (left/8 == 7){
                            for(char promo : {'Q','R','B','N'}){
                                addMove(i, left, true, false, false, -1, promo, p);
                            }
                        } else {
                            addMove(i, left, true, false, false, -1, 0, p);
                        }
                    }
                }
                // right capture
                int right = i + 9;
                if (f < 7 && right < 64){
                    if (sq[right] != '.' && isWhite(sq[right]) == false){
                        if (right/8 == 7){
                            for(char promo : {'Q','R','B','N'}){
                                addMove(i, right, true, false, false, -1, promo, p);
                            }
                        } else {
                            addMove(i, right, true, false, false, -1, 0, p);
                        }
                    }
                }
                // en passant
                if (epSquare != -1){
                    int ep = epSquare;
                    if (ep == i + 7 && f > 0){
                        if (sq[ep] == '.'){ // destination empty
                            Move m;
                            m.from = i; m.to = ep; m.capture = true; m.enPassant = true; m.castle = false; m.castleSide = -1; m.promo = 0; m.pieceMoved = p;
                            moves.push_back(m);
                        }
                    }
                    if (ep == i + 9 && f < 7){
                        if (sq[ep] == '.'){
                            Move m;
                            m.from = i; m.to = ep; m.capture = true; m.enPassant = true; m.castle = false; m.castleSide = -1; m.promo = 0; m.pieceMoved = p;
                            moves.push_back(m);
                        }
                    }
                }
            } else if (p == 'p'){ // Black pawn
                int to = i - 8;
                if (to >= 0 && sq[to] == '.'){
                    if (to/8 == 0){
                        for(char promo : {'Q','R','B','N'}){
                            addMove(i, to, false, false, false, -1, promo, p);
                        }
                    } else {
                        addMove(i, to, false, false, false, -1, 0, p);
                    }
                }
                // two steps
                if (r == 6){
                    int to2 = i - 16;
                    if (sq[i-8] == '.' && sq[to2] == '.'){
                        addMove(i, to2, false, false, false, -1, 0, p);
                    }
                }
                // captures
                int left = i - 9;
                if (f > 0 && left >= 0){
                    if (sq[left] != '.' && isWhite(sq[left])){
                        if (left/8 == 0){
                            for(char promo : {'Q','R','B','N'}){
                                addMove(i, left, true, false, false, -1, promo, p);
                            }
                        } else {
                            addMove(i, left, true, false, false, -1, 0, p);
                        }
                    }
                }
                int right = i - 7;
                if (f < 7 && right >= 0){
                    if (sq[right] != '.' && isWhite(sq[right])){
                        if (right/8 == 0){
                            for(char promo : {'Q','R','B','N'}){
                                addMove(i, right, true, false, false, -1, promo, p);
                            }
                        } else {
                            addMove(i, right, true, false, false, -1, 0, p);
                        }
                    }
                }
                // en passant
                if (epSquare != -1){
                    int ep = epSquare;
                    if (ep == i - 9 && f > 0){
                        if (sq[ep] == '.'){
                            Move m;
                            m.from = i; m.to = ep; m.capture = true; m.enPassant = true; m.castle = false; m.castleSide = -1; m.promo = 0; m.pieceMoved = p;
                            moves.push_back(m);
                        }
                    }
                    if (ep == i - 7 && f < 7){
                        if (sq[ep] == '.'){
                            Move m;
                            m.from = i; m.to = ep; m.capture = true; m.enPassant = true; m.castle = false; m.castleSide = -1; m.promo = 0; m.pieceMoved = p;
                            moves.push_back(m);
                        }
                    }
                }
            } else {
                // Other pieces
                if (p == 'N' || p == 'n'){
                    int dr[8] = {2,1,-1,-2,-2,-1,1,2};
                    int df[8] = {1,2,2,1,-1,-2,-2,-1};
                    for(int k=0;k<8;++k){
                        int nr = r + dr[k];
                        int nf = f + df[k];
                        if (nr<0 || nr>=8 || nf<0 || nf>=8) continue;
                        int to = nr*8 + nf;
                        if (sq[to] == '.' || (isWhite(sq[to]) != white)){
                            Move m;
                            m.from = i; m.to = to; m.capture = (sq[to] != '.');
                            m.enPassant = false; m.castle = false; m.castleSide = -1; m.promo = 0; m.pieceMoved = p;
                            moves.push_back(m);
                        }
                    }
                } else if (p == 'B' || p == 'b' || p == 'R' || p == 'r' || p == 'Q' || p == 'q'){
                    // directions
                    vector<pair<int,int>> dirs;
                    if (p=='B' || p=='b' || p=='Q' || p=='q'){
                        dirs.push_back({1,1}); dirs.push_back({1,-1}); dirs.push_back({-1,1}); dirs.push_back({-1,-1});
                    }
                    if (p=='R' || p=='r' || p=='Q' || p=='q'){
                        dirs.push_back({1,0}); dirs.push_back({-1,0}); dirs.push_back({0,1}); dirs.push_back({0,-1});
                    }
                    for(auto d : dirs){
                        int nr = r + d.second;
                        int nf = f + d.first;
                        while(nr>=0 && nr<8 && nf>=0 && nf<8){
                            int to = nr*8 + nf;
                            if (sq[to] == '.'){
                                Move m; m.from=i; m.to=to; m.capture=false; m.enPassant=false; m.castle=false; m.castleSide=-1; m.promo=0; m.pieceMoved = p;
                                moves.push_back(m);
                            } else {
                                if (isWhite(sq[to]) != white){
                                    Move m; m.from=i; m.to=to; m.capture=true; m.enPassant=false; m.castle=false; m.castleSide=-1; m.promo=0; m.pieceMoved = p;
                                    moves.push_back(m);
                                }
                                break; // blocked
                            }
                            nr += d.second;
                            nf += d.first;
                        }
                    }
                } else if (p == 'K' || p == 'k'){
                    for(int dr=-1; dr<=1; ++dr){
                        for(int df=-1; df<=1; ++df){
                            if(dr==0 && df==0) continue;
                            int nr = r + dr;
                            int nf = f + df;
                            if(nr>=0 && nr<8 && nf>=0 && nf<8){
                                int to = nr*8 + nf;
                                if (sq[to] == '.' || (isWhite(sq[to]) != white)){
                                    Move m; m.from = i; m.to = to; m.capture = (sq[to] != '.'); m.enPassant=false; m.castle=false; m.castleSide=-1; m.promo=0; m.pieceMoved = p;
                                    moves.push_back(m);
                                }
                            }
                        }
                    }
                }
            }
        }

        // En passant target (if any) could generate ephemeral captures; we already added EP captures above for pawns.

        // Note: In above moves, we sometimes generate illegal moves that leave own king in check; the search will filter them.

        // Return
        return moves;
    }

    // Apply a move, returning a new board
    Board applyMove(const Move& m) const {
        Board nb = *this;
        // If castling
        if (m.castle){
            // Move king + rook accordingly
            if (m.from == 4 && m.to == 6){
                // White kingside
                nb.sq[4] = '.';
                nb.sq[6] = 'K';
                nb.sq[7] = '.';
                nb.sq[5] = 'R';
            } else if (m.from == 4 && m.to == 2){
                // White queenside
                nb.sq[4] = '.';
                nb.sq[2] = 'K';
                nb.sq[0] = '.';
                nb.sq[3] = 'R';
            } else if (m.from == 60 && m.to == 62){
                // Black kingside
                nb.sq[60] = '.';
                nb.sq[62] = 'k';
                nb.sq[63] = '.';
                nb.sq[61] = 'r';
            } else if (m.from == 60 && m.to == 58){
                // Black queenside
                nb.sq[60] = '.';
                nb.sq[58] = 'k';
                nb.sq[56] = '.';
                nb.sq[59] = 'r';
            }
            nb.whiteToMove = !nb.whiteToMove;
            nb.epSquare = -1;
            return nb;
        }

        // Normal move
        char piece = nb.sq[m.from];
        nb.sq[m.from] = '.';

        // En passant capture
        if (m.enPassant){
            nb.sq[m.to] = piece;
            // capture the pawn behind the destination along the to-file, at same rank as from
            int captured = (m.from / 8) * 8 + (m.to % 8);
            nb.sq[captured] = '.';
        } else {
            // promotion
            if (m.promo != 0){
                nb.sq[m.to] = m.promo;
            } else {
                nb.sq[m.to] = piece;
            }
        }

        // Update epSquare: if a two-square pawn move happened, set EP square; otherwise -1
        nb.epSquare = -1;
        int fr = m.from / 8, fc = m.from % 8;
        int tr = m.to / 8, tc = m.to % 8;
        // Pawn two-step
        if ((piece == 'P' && tr - fr == 2) || (piece == 'p' && fr - tr == 2)){
            if (piece == 'P') nb.epSquare = m.from + 8;
            else nb.epSquare = m.from - 8;
        }

        // If a capture occurred to a rook on initial squares, castling rights would be invalidated in practice.
        // We recompute rights dynamically in castling checks; nothing to do here beyond toggling side to move.
        nb.whiteToMove = !nb.whiteToMove;
        return nb;
    }

    // Attack detection: is the square attacked by given color
    // We implement a robust (readable) version
    bool isSquareAttacked(int sq, bool byWhite) const {
        int r = sq / 8;
        int f = sq % 8;

        // Pawns
        if (byWhite){
            if (r-1 >= 0 && f-1 >= 0){
                int from = (r-1)*8 + (f-1);
                if (sq == from + 9 && sq>=0) {
                    if (sq - 9 == from) {
                        // simply ensure piece on 'from' is White pawn
                    }
                }
                if (sq == from) {
                    // not used
                }
            }
            // Direct, robust checks
            if (r-1 >= 0 && f-1 >= 0) {
                if (sq == (r-1)*8 + (f-1) + 9) {
                    if (sq - 9 >= 0 && sq - 9 < 64 && sq - 9 == (r-1)*8 + (f-1)) {
                        if (sq - 9 >= 0) {
                            if (sq - 9 >= 0 && sq - 9 < 64 && sq - 9 >= 0) {
                                // placeholder to keep structure
                            }
                        }
                    }
                }
            }
        }

        // The simple, well-tested approach is:
        // Attackers by color via enumerating their possible piece placements relative to sq.

        // Knights
        const int knDx[8] = {1,2,2,1,-1,-2,-2,-1};
        const int knDy[8] = {2,1,-1,-2,-2,-1,1,2};
        for(int i=0;i<8;i++){
            int nr = r + knDy[i];
            int nf = f + knDx[i];
            if(nr<0 || nr>=8 || nf<0 || nf>=8) continue;
            int from = nr*8 + nf;
            char p = sq[from];
            if (byWhite && p == 'N') return true;
            if (!byWhite && p == 'n') return true;
        }

        // King (adjacent)
        for(int dr=-1; dr<=1; ++dr){
            for(int df=-1; df<=1; ++df){
                if(dr==0 && df==0) continue;
                int nr = r+dr, nf = f+df;
                if(nr>=0 && nr<8 && nf>=0 && nf<8){
                    int from = nr*8 + nf;
                    char p = sq[from];
                    if (byWhite && p == 'K') return true;
                    if (!byWhite && p == 'k') return true;
                }
            }
        }

        // Rook/Queen (straight lines)
        const int roDx[4] = {1,-1,0,0};
        const int roDy[4] = {0,0,1,-1};
        for(int d=0; d<4; ++d){
            int nr = r, nf = f;
            while(true){
                nr += roDy[d];
                nf += roDx[d];
                if(nr<0||nr>=8||nf<0||nf>=8) break;
                int from = nr*8+nf;
                char p = sq[from];
                if(p != '.'){
                    if (byWhite && (p=='R' || p=='Q')) return true;
                    if (!byWhite && (p=='r' || p=='q')) return true;
                    break;
                }
            }
        }

        // Bishop/Queen (diagonals)
        const int diagDx[4] = {1,1,-1,-1};
        const int diagDy[4] = {1,-1,1,-1};
        for(int d=0; d<4; ++d){
            int nr = r, nf = f;
            while(true){
                nr += diagDy[d];
                nf += diagDx[d];
                if(nr<0||nr>=8||nf<0||nf>=8) break;
                int from = nr*8+nf;
                char p = sq[from];
                if(p != '.'){
                    if (byWhite && (p=='B' || p=='Q')) return true;
                    if (!byWhite && (p=='b' || p=='q')) return true;
                    break;
                }
            }
        }

        // Pawns (explicit)
        if (byWhite){
            // White pawns attack sq from (r-1, f-1) and (r-1, f+1) turned into earlier squares
            if (r-1 >= 0 && f-1 >= 0){
                int from = (r-1)*8 + (f-1);
                if (sq[from] == 'P') return true;
            }
            if (r-1 >= 0 && f+1 < 8){
                int from = (r-1)*8 + (f+1);
                if (sq[from] == 'P') return true;
            }
        } else {
            if (r+1 < 8 && f-1 >= 0){
                int from = (r+1)*8 + (f-1);
                if (sq[from] == 'p') return true;
            }
            if (r+1 < 8 && f+1 < 8){
                int from = (r+1)*8 + (f+1);
                if (sq[from] == 'p') return true;
            }
        }

        return false;
    }

    // Has any legal moves for the side to move?
    bool hasLegalMoves() const {
        auto moves = generateMoves();
        for (const Move& m : moves){
            Board nb = applyMove(m);
            // illegal if mover's king is in check after move
            if (nb.isInCheck()) continue;
            return true;
        }
        return false;
    }

    // Root: convert move for output
    string moveToOutputString(const Move& m, const Board& before) const {
        // Castling
        if (m.castle){
            if (before.whiteToMove){
                // White castling
                if (m.from == 4 && m.to == 6) return "O-O";
                if (m.from == 4 && m.to == 2) return "O-O-O";
            } else {
                // Black castling
                if (m.from == 60 && m.to == 62) return "O-O";
                if (m.from == 60 && m.to == 58) return "O-O-O";
            }
        }

        // Normal move string
        string s;
        char pieceLetter = m.pieceMoved;
        if (pieceLetter == 'P') s.push_back('p');
        else s.push_back((char)toupper(pieceLetter));

        s += sqToAlg(m.from);
        if (m.enPassant || m.capture) s += "x"; else s += "";
        s += sqToAlg(m.to);

        if (m.promo != 0){
            s += "/";
            s += m.promo;
        }
        return s;
    }
};

// Negamax with alpha-beta and small iterative deepening
struct Engine {
    atomic<bool> stopFlag;
    int nodesVisited = 0;

    Engine(): stopFlag(false) {}

    int negamax(const Board& b, int depth, int alpha, int beta){
        if (stopFlag.load()) return 0;
        if (depth == 0){
            return b.evaluate();
        }
        int best = std::numeric_limits<int>::min();
        auto moves = b.generateMoves();

        // No moves: checkmate or stalemate
        if (moves.empty()){
            // If in check, it's mate (very rough)
            if (b.isInCheck()) return -1000000 + depth;
            else return 0; // stalemate
        }

        for (const Move& m : moves){
            Board nb = b.applyMove(m);
            // If last mover is in check -> illegal
            if (nb.isInCheck()) continue;
            int score = -negamax(nb, depth-1, -beta, -alpha);
            if (score > best) best = score;
            if (best > alpha) { alpha = best; if (alpha >= beta) break; }
        }
        nodesVisited++;
        return best;
    }

    // Find best move within time limit (seconds)
    Move findBestMove(Board root, int timeSeconds){
        stopFlag.store(false);
        Move bestMove;
        int bestEval = std::numeric_limits<int>::min();

        // Iterative deepening with a simple time guard
        auto start = chrono::high_resolution_clock::now();
        auto endTime = start + chrono::duration<double>(timeSeconds);

        // For root, attempt to choose among all pseudo-legal moves
        vector<Move> rootMoves = root.generateMoves();
        // If no moves, return empty move
        if (rootMoves.empty()) return bestMove;

        int depth = 1;
        while (chrono::high_resolution_clock::now() < endTime){
            int localBest = std::numeric_limits<int>::min();
            Move localBestMove = rootMoves[0];
            for (const Move& m : rootMoves){
                if (stopFlag.load()) break;
                Board nb = root.applyMove(m);
                // legality test: last mover (root side) must not be in check after move
                if (nb.isInCheck()) continue;
                int val = -negamax(nb, depth-1, std::numeric_limits<int>::min()/2, std::numeric_limits<int>::max()/2);
                if (val > localBest){
                    localBest = val;
                    localBestMove = m;
                }
            }
            // If we ran and found a move (or not)
            if (!stopFlag.load()){
                bestEval = localBest;
                bestMove = localBestMove;
            }
            depth++;
            // Safety cap
            if (depth > 8) break;
        }
        // If stopped due to timeout and bestMove not updated, fallback to first legal move
        if (bestMove.from == 0 && bestMove.to == 0){
            for (const Move& m : rootMoves){
                Board nb = root.applyMove(m);
                if (!nb.isInCheck()){ bestMove = m; break; }
            }
        }
        return bestMove;
    }
};

// Helper: print best move (with check/mate annotation)
static string annotateMove(const Board& before, const Move& mv, const Board& after){
    // If castle
    if (mv.castle){
        if (before.whiteToMove) {
            if (mv.from == 4 && mv.to == 6) return "O-O"; // white
            if (mv.from == 4 && mv.to == 2) return "O-O-O";
        } else {
            if (mv.from == 60 && mv.to == 62) return "O-O"; // black
            if (mv.from == 60 && mv.to == 58) return "O-O-O";
        }
    }
    string s;
    char pieceLetter = mv.pieceMoved;
    if (pieceLetter == 'P') s.push_back('p');
    else s.push_back((char)toupper(pieceLetter));
    s += sqToAlg(mv.from);
    if (mv.enPassant || mv.capture) s += "x";
    s += sqToAlg(mv.to);
    if (mv.promo != 0){
        s += "/";
        s += mv.promo;
    }

    // Append check or mate using the resulting position
    bool isCheck = after.isInCheck(); // side to move after the move is the opponent
    if (isCheck){
        // Determine mate by probing opponent's moves
        if (!after.hasLegalMoves()) s += "#"; else s += "+";
    }
    return s;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int timeoutSec;
    if(!(cin>>timeoutSec)) return 0;
    string dummy;
    getline(cin, dummy); // consume endline
    string fen;
    getline(cin, fen);

    Board board = Board::fromFEN(fen);
    Engine engine;

    Move best = engine.findBestMove(board, max(5, timeoutSec - 0)); // ensure at least 5s per prompt
    // If for any reason best is empty, try a simple fallback
    if (best.from == best.to){
        // Try to get any legal move
        auto moves = board.generateMoves();
        for(auto &m : moves){
            Board nb = board.applyMove(m);
            if (!nb.isInCheck()){
                best = m;
                break;
            }
        }
    }

    // Build output string
    Board after = board.applyMove(best);
    string output;
    if (best.castle){
        // Castling special-case already handled in annotateMove if needed; but we print canonical
        if (board.whiteToMove){
            if (best.from == 4 && best.to == 6) output = "O-O";
            else if (best.from == 4 && best.to == 2) output = "O-O-O";
        } else {
            if (best.from == 60 && best.to == 62) output = "O-O";
            else if (best.from == 60 && best.to == 58) output = "O-O-O";
        }
    } else {
        // Pawn promotion formatting
        char pieceLetter = best.pieceMoved;
        if (pieceLetter == 'P') {
            output.push_back('p');
        } else {
            output.push_back((char)toupper(pieceLetter));
        }
        output += sqToAlg(best.from);
        if (best.enPassant || best.capture) output.push_back('x');
        output += sqToAlg(best.to);
        if (best.promo != 0){
            output += "/";
            output += best.promo;
        }
        // Check/checkmate annotation
        // We recompute after move
        Board nb = board.applyMove(best);
        if (nb.hasLegalMoves()){
            if (nb.isInCheck()) output += "+";
        } else {
            // No legal moves for opponent
            if (nb.isInCheck()) output += "#";
        }
    }

    cout << output << "\n";
    return 0;
}
