#include <algorithm>
#include <cstdint>
#include <functional>
#include <iostream>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

class MaximumIndependentSetSolver {
public:
    MaximumIndependentSetSolver(int n, vector<vector<int>> graph, vector<char> enabled)
        : n_(n), graph_(move(graph)), enabled_(move(enabled)), position_(n_, -1) {}

    vector<int> solve() {
        vector<int> answer;
        vector<char> seen(n_, false);
        queue<int> q;

        for (int start = 0; start < n_; ++start) {
            if (!enabled_[start] || seen[start]) {
                continue;
            }

            seen[start] = true;
            q.push(start);

            vector<int> component;
            while (!q.empty()) {
                int v = q.front();
                q.pop();
                component.push_back(v);
                for (int u : graph_[v]) {
                    if (!seen[u]) {
                        seen[u] = true;
                        q.push(u);
                    }
                }
            }

            vector<int> part = solveComponent(component);
            answer.insert(answer.end(), part.begin(), part.end());
        }

        sort(answer.begin(), answer.end());
        return answer;
    }

private:
    struct LocalGraph {
        vector<int> global;
        vector<vector<int>> adj;
    };

    class MaxCliqueSolver {
    public:
        explicit MaxCliqueSolver(vector<vector<uint64_t>> adj)
            : adj_(move(adj)), n_(static_cast<int>(adj_.size())), degree_(n_, 0) {
            for (int v = 0; v < n_; ++v) {
                int deg = 0;
                for (uint64_t word : adj_[v]) {
                    deg += popcount(word);
                }
                degree_[v] = deg;
            }
        }

        vector<int> solve() {
            vector<int> candidates(n_);
            iota(candidates.begin(), candidates.end(), 0);
            sort(candidates.begin(), candidates.end(), [&](int a, int b) {
                if (degree_[a] != degree_[b]) {
                    return degree_[a] > degree_[b];
                }
                return a < b;
            });

            best_ = greedyClique(candidates);
            current_.reserve(n_);
            expand(candidates);
            return best_;
        }

    private:
        static int popcount(uint64_t x) {
            int count = 0;
            while (x != 0) {
                x &= (x - 1);
                ++count;
            }
            return count;
        }

        bool adjacent(int u, int v) const {
            return ((adj_[u][v >> 6] >> (v & 63)) & 1ULL) != 0;
        }

        vector<int> greedyClique(vector<int> candidates) const {
            vector<int> clique;
            while (!candidates.empty()) {
                int v = candidates.front();
                clique.push_back(v);

                vector<int> next;
                next.reserve(candidates.size());
                for (size_t i = 1; i < candidates.size(); ++i) {
                    int u = candidates[i];
                    if (adjacent(v, u)) {
                        next.push_back(u);
                    }
                }

                sort(next.begin(), next.end(), [&](int a, int b) {
                    if (degree_[a] != degree_[b]) {
                        return degree_[a] > degree_[b];
                    }
                    return a < b;
                });
                candidates.swap(next);
            }
            return clique;
        }

        void colorSort(const vector<int>& candidates, vector<int>& ordered, vector<int>& bounds) const {
            vector<int> remaining = candidates;
            ordered.clear();
            bounds.clear();
            ordered.reserve(candidates.size());
            bounds.reserve(candidates.size());

            int color = 0;
            vector<int> colorClass;
            vector<int> next;

            while (!remaining.empty()) {
                ++color;
                colorClass.clear();
                next.clear();

                for (int v : remaining) {
                    bool fits = true;
                    for (int u : colorClass) {
                        if (adjacent(v, u)) {
                            fits = false;
                            break;
                        }
                    }
                    if (fits) {
                        colorClass.push_back(v);
                    } else {
                        next.push_back(v);
                    }
                }

                for (int v : colorClass) {
                    ordered.push_back(v);
                    bounds.push_back(color);
                }

                remaining.swap(next);
            }
        }

        void expand(const vector<int>& candidates) {
            if (candidates.empty()) {
                if (current_.size() > best_.size()) {
                    best_ = current_;
                }
                return;
            }

            if (current_.size() + candidates.size() <= best_.size()) {
                return;
            }

            vector<int> ordered;
            vector<int> bounds;
            colorSort(candidates, ordered, bounds);

            for (int i = static_cast<int>(ordered.size()) - 1; i >= 0; --i) {
                if (current_.size() + static_cast<size_t>(bounds[i]) <= best_.size()) {
                    return;
                }

                int v = ordered[i];
                current_.push_back(v);

                vector<int> next;
                next.reserve(i);
                for (int j = 0; j < i; ++j) {
                    int u = ordered[j];
                    if (adjacent(v, u)) {
                        next.push_back(u);
                    }
                }

                if (next.empty()) {
                    if (current_.size() > best_.size()) {
                        best_ = current_;
                    }
                } else {
                    expand(next);
                }

                current_.pop_back();
            }
        }

        vector<vector<uint64_t>> adj_;
        int n_;
        vector<int> degree_;
        vector<int> best_;
        vector<int> current_;
    };

    LocalGraph buildLocalGraph(const vector<int>& vertices) {
        LocalGraph local;
        local.global = vertices;
        local.adj.assign(vertices.size(), {});

        for (size_t i = 0; i < vertices.size(); ++i) {
            position_[vertices[i]] = static_cast<int>(i);
        }

        for (size_t i = 0; i < vertices.size(); ++i) {
            int v = vertices[i];
            vector<int>& out = local.adj[i];
            out.reserve(graph_[v].size());
            for (int u : graph_[v]) {
                int pos = position_[u];
                if (pos != -1) {
                    out.push_back(pos);
                }
            }
        }

        for (int v : vertices) {
            position_[v] = -1;
        }

        return local;
    }

    bool isBipartite(const LocalGraph& local, vector<int>& color) const {
        int size = static_cast<int>(local.global.size());
        color.assign(size, -1);
        queue<int> q;

        for (int start = 0; start < size; ++start) {
            if (color[start] != -1) {
                continue;
            }

            color[start] = 0;
            q.push(start);

            while (!q.empty()) {
                int v = q.front();
                q.pop();
                for (int u : local.adj[v]) {
                    if (color[u] == -1) {
                        color[u] = color[v] ^ 1;
                        q.push(u);
                    } else if (color[u] == color[v]) {
                        return false;
                    }
                }
            }
        }

        return true;
    }

    vector<int> solveBipartite(const LocalGraph& local, const vector<int>& color) const {
        int size = static_cast<int>(local.global.size());
        vector<int> leftOf(size, -1);
        vector<int> rightOf(size, -1);
        vector<int> leftLocal;
        vector<int> rightLocal;

        for (int v = 0; v < size; ++v) {
            if (color[v] == 0) {
                leftOf[v] = static_cast<int>(leftLocal.size());
                leftLocal.push_back(v);
            } else {
                rightOf[v] = static_cast<int>(rightLocal.size());
                rightLocal.push_back(v);
            }
        }

        const int leftSize = static_cast<int>(leftLocal.size());
        const int rightSize = static_cast<int>(rightLocal.size());

        vector<vector<int>> adjLeft(leftSize);
        for (int i = 0; i < leftSize; ++i) {
            int v = leftLocal[i];
            vector<int>& out = adjLeft[i];
            out.reserve(local.adj[v].size());
            for (int u : local.adj[v]) {
                out.push_back(rightOf[u]);
            }
        }

        vector<int> pairU(leftSize, -1);
        vector<int> pairV(rightSize, -1);
        vector<int> dist(leftSize, -1);

        auto bfs = [&]() -> bool {
            queue<int> q;
            fill(dist.begin(), dist.end(), -1);

            for (int u = 0; u < leftSize; ++u) {
                if (pairU[u] == -1) {
                    dist[u] = 0;
                    q.push(u);
                }
            }

            bool foundFree = false;
            while (!q.empty()) {
                int u = q.front();
                q.pop();
                for (int v : adjLeft[u]) {
                    int matched = pairV[v];
                    if (matched == -1) {
                        foundFree = true;
                    } else if (dist[matched] == -1) {
                        dist[matched] = dist[u] + 1;
                        q.push(matched);
                    }
                }
            }

            return foundFree;
        };

        function<bool(int)> dfs = [&](int u) -> bool {
            for (int v : adjLeft[u]) {
                int matched = pairV[v];
                if (matched == -1 || (dist[matched] == dist[u] + 1 && dfs(matched))) {
                    pairU[u] = v;
                    pairV[v] = u;
                    return true;
                }
            }
            dist[u] = -1;
            return false;
        };

        while (bfs()) {
            for (int u = 0; u < leftSize; ++u) {
                if (pairU[u] == -1) {
                    dfs(u);
                }
            }
        }

        vector<char> visLeft(leftSize, false);
        vector<char> visRight(rightSize, false);
        queue<int> q;

        for (int u = 0; u < leftSize; ++u) {
            if (pairU[u] == -1) {
                visLeft[u] = true;
                q.push(u);
            }
        }

        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : adjLeft[u]) {
                if (pairU[u] == v) {
                    continue;
                }
                if (!visRight[v]) {
                    visRight[v] = true;
                    int matched = pairV[v];
                    if (matched != -1 && !visLeft[matched]) {
                        visLeft[matched] = true;
                        q.push(matched);
                    }
                }
            }
        }

        vector<char> inCover(size, false);
        for (int i = 0; i < leftSize; ++i) {
            if (!visLeft[i]) {
                inCover[leftLocal[i]] = true;
            }
        }
        for (int i = 0; i < rightSize; ++i) {
            if (visRight[i]) {
                inCover[rightLocal[i]] = true;
            }
        }

        vector<int> answer;
        answer.reserve(size);
        for (int v = 0; v < size; ++v) {
            if (!inCover[v]) {
                answer.push_back(local.global[v]);
            }
        }
        return answer;
    }

    vector<int> solveByClique(const LocalGraph& local) const {
        int size = static_cast<int>(local.global.size());
        int blocks = (size + 63) >> 6;
        uint64_t lastMask = (size % 64 == 0) ? ~0ULL : ((1ULL << (size % 64)) - 1ULL);

        vector<vector<uint64_t>> complement(size, vector<uint64_t>(blocks, ~0ULL));
        if (blocks > 0) {
            for (int v = 0; v < size; ++v) {
                complement[v][blocks - 1] &= lastMask;
            }
        }

        for (int v = 0; v < size; ++v) {
            complement[v][v >> 6] &= ~(1ULL << (v & 63));
            for (int u : local.adj[v]) {
                complement[v][u >> 6] &= ~(1ULL << (u & 63));
            }
        }

        MaxCliqueSolver solver(move(complement));
        vector<int> clique = solver.solve();

        vector<int> answer;
        answer.reserve(clique.size());
        for (int v : clique) {
            answer.push_back(local.global[v]);
        }
        return answer;
    }

    vector<int> solveGeneral(const LocalGraph& local) {
        const int size = static_cast<int>(local.global.size());
        const int blocks = (size + 63) >> 6;
        const int simplicialLimit = 32;

        vector<vector<uint64_t>> adjBits(size, vector<uint64_t>(blocks, 0ULL));
        for (int v = 0; v < size; ++v) {
            for (int u : local.adj[v]) {
                adjBits[v][u >> 6] |= 1ULL << (u & 63);
            }
        }

        vector<char> active(size, true);
        vector<int> degree(size, 0);
        queue<int> q;

        for (int v = 0; v < size; ++v) {
            degree[v] = static_cast<int>(local.adj[v].size());
            q.push(v);
        }

        auto hasEdge = [&](int a, int b) -> bool {
            return ((adjBits[a][b >> 6] >> (b & 63)) & 1ULL) != 0;
        };

        auto removeVertex = [&](int v) {
            if (!active[v]) {
                return;
            }
            active[v] = false;
            for (int u : local.adj[v]) {
                if (active[u]) {
                    --degree[u];
                    q.push(u);
                }
            }
            degree[v] = 0;
        };

        auto findSingleActiveNeighbor = [&](int v) -> int {
            for (int u : local.adj[v]) {
                if (active[u]) {
                    return u;
                }
            }
            return -1;
        };

        auto neighborhoodIsClique = [&](int v) -> bool {
            vector<int> neighbors;
            neighbors.reserve(degree[v]);
            for (int u : local.adj[v]) {
                if (active[u]) {
                    neighbors.push_back(u);
                }
            }

            for (size_t i = 0; i < neighbors.size(); ++i) {
                for (size_t j = i + 1; j < neighbors.size(); ++j) {
                    if (!hasEdge(neighbors[i], neighbors[j])) {
                        return false;
                    }
                }
            }
            return true;
        };

        auto removeClosedNeighborhood = [&](int v) {
            vector<int> toRemove;
            toRemove.reserve(local.adj[v].size() + 1);
            toRemove.push_back(v);
            for (int u : local.adj[v]) {
                if (active[u]) {
                    toRemove.push_back(u);
                }
            }
            for (int u : toRemove) {
                removeVertex(u);
            }
        };

        vector<int> forced;
        bool changed = false;

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            if (!active[v]) {
                continue;
            }

            if (degree[v] == 0) {
                forced.push_back(local.global[v]);
                removeVertex(v);
                changed = true;
                continue;
            }

            if (degree[v] == 1) {
                int u = findSingleActiveNeighbor(v);
                if (u != -1) {
                    forced.push_back(local.global[v]);
                    removeVertex(v);
                    removeVertex(u);
                    changed = true;
                }
                continue;
            }

            if (degree[v] <= simplicialLimit && neighborhoodIsClique(v)) {
                forced.push_back(local.global[v]);
                removeClosedNeighborhood(v);
                changed = true;
            }
        }

        if (!changed) {
            return solveByClique(local);
        }

        vector<int> answer = forced;
        vector<char> seen(size, false);
        queue<int> bfs;

        for (int start = 0; start < size; ++start) {
            if (!active[start] || seen[start]) {
                continue;
            }

            seen[start] = true;
            bfs.push(start);

            vector<int> subcomponent;
            while (!bfs.empty()) {
                int v = bfs.front();
                bfs.pop();
                subcomponent.push_back(local.global[v]);
                for (int u : local.adj[v]) {
                    if (active[u] && !seen[u]) {
                        seen[u] = true;
                        bfs.push(u);
                    }
                }
            }

            vector<int> part = solveComponent(subcomponent);
            answer.insert(answer.end(), part.begin(), part.end());
        }

        return answer;
    }

    vector<int> solveComponent(const vector<int>& vertices) {
        if (vertices.empty()) {
            return {};
        }
        if (vertices.size() == 1) {
            return vertices;
        }

        LocalGraph local = buildLocalGraph(vertices);
        vector<int> color;
        if (isBipartite(local, color)) {
            return solveBipartite(local, color);
        }
        return solveGeneral(local);
    }

    int n_;
    vector<vector<int>> graph_;
    vector<char> enabled_;
    vector<int> position_;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n = 0;
    int m = 0;
    if (!(cin >> n >> m)) {
        return 0;
    }

    vector<char> blocked(n, false);
    vector<pair<int, int>> edges;
    edges.reserve(max(0, m));

    for (int i = 0; i < m; ++i) {
        int u = 0;
        int v = 0;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) {
            continue;
        }
        if (u == v) {
            blocked[u] = true;
        } else {
            edges.emplace_back(u, v);
        }
    }

    vector<vector<int>> graph(n);
    for (const auto& edge : edges) {
        int u = edge.first;
        int v = edge.second;
        if (blocked[u] || blocked[v]) {
            continue;
        }
        graph[u].push_back(v);
        graph[v].push_back(u);
    }

    for (int v = 0; v < n; ++v) {
        if (blocked[v]) {
            graph[v].clear();
            continue;
        }
        sort(graph[v].begin(), graph[v].end());
        graph[v].erase(unique(graph[v].begin(), graph[v].end()), graph[v].end());
    }

    vector<char> enabled(n, false);
    for (int v = 0; v < n; ++v) {
        if (!blocked[v]) {
            enabled[v] = true;
        }
    }

    vector<int> answer;
    if (find(enabled.begin(), enabled.end(), true) != enabled.end()) {
        MaximumIndependentSetSolver solver(n, move(graph), move(enabled));
        answer = solver.solve();
    }

    sort(answer.begin(), answer.end());
    cout << answer.size() << '\n';
    for (size_t i = 0; i < answer.size(); ++i) {
        if (i != 0) {
            cout << ' ';
        }
        cout << answer[i];
    }
    cout << '\n';
    return 0;
}
