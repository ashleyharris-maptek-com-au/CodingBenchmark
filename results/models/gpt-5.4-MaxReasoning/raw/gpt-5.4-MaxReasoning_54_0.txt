import math


_AP = {}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _finite(x):
    return isinstance(x, (int, float)) and math.isfinite(x)


def _median(values):
    values = sorted(values)
    n = len(values)
    if n == 0:
        return None
    mid = n // 2
    if n % 2:
        return values[mid]
    return 0.5 * (values[mid - 1] + values[mid])


def _angdiff(target, current):
    return (target - current + 180.0) % 360.0 - 180.0


def _angle_mean(values):
    if not values:
        return None
    s = sum(math.sin(math.radians(v)) for v in values)
    c = sum(math.cos(math.radians(v)) for v in values)
    if abs(s) < 1e-9 and abs(c) < 1e-9:
        return values[0] % 360.0
    return (math.degrees(math.atan2(s, c)) + 360.0) % 360.0


def _fuse_scalar(values, last=None, outlier=1.0):
    values = [float(v) for v in values if _finite(v)]
    if not values:
        return last, False
    if len(values) == 1:
        return values[0], True
    med = _median(values)
    band = [v for v in values if abs(v - med) <= outlier]
    if len(band) >= 2:
        return sum(band) / len(band), True
    if last is not None:
        best = min(values, key=lambda v: abs(v - last))
        return best, True
    return med, True


def _fuse_angle(values, last=None):
    values = [float(v) % 360.0 for v in values if _finite(v)]
    if not values:
        return last, False
    if len(values) == 1:
        return values[0], True
    if last is None:
        return _angle_mean(values), True
    unwrapped = [last + _angdiff(v, last) for v in values]
    return sum(unwrapped) / len(unwrapped) % 360.0, True


def _triplet(state, prefix, field, last=None, outlier=1.0, angle=False):
    values = []
    for i in (1, 2, 3):
        key = f"{prefix}_{i}_{field}"
        if key not in state:
            continue
        if bool(state.get(f"{prefix}_{i}_valid", True)):
            values.append(state.get(key))
    if angle:
        return _fuse_angle(values, last)
    return _fuse_scalar(values, last, outlier)


def _lp(name, value, dt, tau):
    if value is None:
        return _AP.get(name)
    prev = _AP.get(name)
    if prev is None or not _finite(prev):
        _AP[name] = float(value)
        return _AP[name]
    alpha = _clamp(dt / max(tau, 1e-3), 0.0, 1.0)
    _AP[name] = prev + alpha * (float(value) - prev)
    return _AP[name]


def _slew(prev, target, up_rate, down_rate, dt):
    if target > prev:
        return min(target, prev + up_rate * dt)
    return max(target, prev - down_rate * dt)


def _engine_live(state, idx):
    running = state.get(f"engine_{idx}_running")
    fire = bool(state.get(f"engine_{idx}_fire"))
    n1 = state.get(f"engine_{idx}_n1_pct")
    thrust = state.get(f"engine_{idx}_thrust_pct")
    if fire:
        return False
    if running is False:
        if _finite(n1) and float(n1) > 55.0:
            return True
        if _finite(thrust) and float(thrust) > 30.0:
            return True
        return False
    return True


def _init():
    if _AP.get("booted"):
        return
    _AP.clear()
    _AP.update(
        {
            "booted": True,
            "t": 0.0,
            "mode": "intercept",
            "loc_age": 99.0,
            "gs_age": 99.0,
            "unstable_t": 0.0,
            "ra_freeze_t": 0.0,
            "trim": 0.0,
            "loc_i": 0.0,
            "gs_i": 0.0,
            "spd_i": 0.0,
            "agl": 1500.0,
            "runway_elev": None,
            "last_ra": None,
            "last_baro": None,
            "heading": 0.0,
            "track": 0.0,
            "last_throttle1": 0.55,
            "last_throttle2": 0.55,
        }
    )


def autopilot_step(state: dict, dt: float) -> dict:
    _init()
    dt = float(dt) if _finite(dt) else 0.1
    dt = _clamp(dt, 0.02, 0.5)
    _AP["t"] += dt

    loc_raw, loc_fresh = _triplet(
        state, "localizer", "dots", _AP.get("loc_raw"), outlier=0.7
    )
    gs_raw, gs_fresh = _triplet(
        state, "glideslope", "dots", _AP.get("gs_raw"), outlier=0.7
    )
    baro_raw, _ = _triplet(state, "baro_alt", "ft", _AP.get("baro_raw"), outlier=120.0)
    ias_raw, _ = _triplet(state, "ias", "kt", _AP.get("ias_raw"), outlier=12.0)
    aoa_raw, _ = _triplet(state, "aoa", "deg", _AP.get("aoa_raw"), outlier=2.5)
    gps_gs_raw, _ = _triplet(state, "gps", "gs_kt", _AP.get("gps_gs"), outlier=12.0)
    irs_gs_raw, _ = _triplet(state, "irs", "gs_kt", _AP.get("irs_gs"), outlier=12.0)
    vs_raw, _ = _triplet(state, "irs", "vs_fpm", _AP.get("vs_raw"), outlier=350.0)
    pitch_raw, _ = _triplet(state, "irs", "pitch_deg", _AP.get("pitch_raw"), outlier=2.5)
    roll_raw, _ = _triplet(state, "irs", "roll_deg", _AP.get("roll_raw"), outlier=4.0)
    heading_raw, _ = _triplet(
        state, "irs", "heading_deg", _AP.get("heading"), angle=True
    )
    track_gps_raw, track_gps_fresh = _triplet(
        state, "gps", "track_deg", _AP.get("track"), angle=True
    )
    track_irs_raw, track_irs_fresh = _triplet(
        state, "irs", "track_deg", _AP.get("track"), angle=True
    )

    loc = _lp("loc", 0.0 if loc_raw is None else loc_raw, dt, 0.45)
    gs = _lp("gs", 0.0 if gs_raw is None else gs_raw, dt, 0.45)
    baro = _lp("baro", baro_raw, dt, 0.8) if baro_raw is not None else _AP.get("baro")
    ias = _lp("ias", ias_raw, dt, 0.45) if ias_raw is not None else _AP.get("ias")
    aoa = _lp("aoa", 4.0 if aoa_raw is None else aoa_raw, dt, 0.35)
    vs = _lp("vs", -700.0 if vs_raw is None else vs_raw, dt, 0.35)
    pitch = _lp("pitch", 2.5 if pitch_raw is None else pitch_raw, dt, 0.3)
    roll = _lp("roll", 0.0 if roll_raw is None else roll_raw, dt, 0.25)
    heading = heading_raw if heading_raw is not None else _AP.get("heading", 0.0)
    _AP["heading"] = heading % 360.0

    if track_gps_fresh and track_gps_raw is not None:
        track = track_gps_raw
    elif track_irs_fresh and track_irs_raw is not None:
        track = track_irs_raw
    else:
        track = _AP.get("track", heading)
    _AP["track"] = track % 360.0

    dme_raw, _ = _fuse_scalar(
        [state.get("dme_1_nm"), state.get("dme_2_nm")], _AP.get("dme"), outlier=0.8
    )
    dme = _lp("dme", dme_raw, dt, 1.0) if dme_raw is not None else _AP.get("dme")
    if _finite(dme):
        dme = max(float(dme), 0.0)
        _AP["dme"] = dme

    groundspeed_raw, _ = _fuse_scalar(
        [state.get("groundspeed_kt"), gps_gs_raw, irs_gs_raw, ias],
        _AP.get("groundspeed"),
        outlier=25.0,
    )
    groundspeed = (
        _lp("groundspeed", groundspeed_raw, dt, 0.6)
        if groundspeed_raw is not None
        else _AP.get("groundspeed")
    )

    ra = state.get("radio_alt_ft")
    ra_valid = _finite(ra) and -20.0 <= float(ra) <= 5000.0
    if ra_valid:
        ra = float(ra)
        if _finite(baro) and _finite(_AP.get("last_baro")) and _finite(_AP.get("last_ra")):
            baro_delta = float(baro) - float(_AP["last_baro"])
            ra_delta = ra - float(_AP["last_ra"])
            if abs(vs) > 300.0 and abs(baro_delta) > 3.0 and abs(ra_delta) < 0.2:
                _AP["ra_freeze_t"] += dt
            else:
                _AP["ra_freeze_t"] = max(0.0, _AP["ra_freeze_t"] - 2.0 * dt)
        if _AP["ra_freeze_t"] > 2.0:
            ra_valid = False
    else:
        _AP["ra_freeze_t"] = min(10.0, _AP["ra_freeze_t"] + dt)

    if ra_valid and _finite(baro):
        est = float(baro) - ra
        if _AP["runway_elev"] is None:
            _AP["runway_elev"] = est
        else:
            alpha = _clamp(dt / 5.0, 0.01, 0.2)
            _AP["runway_elev"] += alpha * (est - _AP["runway_elev"])
    elif _finite(baro) and _finite(dme) and dme < 15.0:
        est = float(baro) - (50.0 + 318.0 * dme)
        if _AP["runway_elev"] is None:
            _AP["runway_elev"] = est
        else:
            alpha = _clamp(dt / 10.0, 0.005, 0.08)
            _AP["runway_elev"] += alpha * (est - _AP["runway_elev"])

    if ra_valid:
        agl = max(ra, 0.0)
    elif _finite(baro) and _finite(_AP.get("runway_elev")):
        agl = max(float(baro) - float(_AP["runway_elev"]), 0.0)
    elif _finite(dme):
        agl = max(50.0 + 318.0 * dme, 0.0)
    else:
        agl = max(float(_AP.get("agl", 1500.0)) + float(vs) * dt / 60.0, 0.0)
    _AP["agl"] = agl

    if ra_valid:
        _AP["last_ra"] = ra
    if _finite(baro):
        _AP["last_baro"] = float(baro)

    _AP["loc_age"] = 0.0 if loc_fresh else _AP["loc_age"] + dt
    _AP["gs_age"] = 0.0 if gs_fresh else _AP["gs_age"] + dt
    reliable_loc = loc_fresh or _AP["loc_age"] < 1.5
    reliable_gs = gs_fresh or _AP["gs_age"] < 1.5

    eng1_live = _engine_live(state, 1)
    eng2_live = _engine_live(state, 2)
    live_engines = int(eng1_live) + int(eng2_live)

    vref = state.get("target_vref_kt")
    vref = float(vref) if _finite(vref) else 137.0
    target_ias = state.get("target_ias_kt")
    target_ias = float(target_ias) if _finite(target_ias) else vref + 10.0
    target_vs = state.get("target_vs_fpm")
    target_vs = float(target_vs) if _finite(target_vs) else None

    if not _finite(groundspeed):
        groundspeed = ias if _finite(ias) else target_ias
    if not _finite(ias):
        ias = groundspeed if _finite(groundspeed) else target_ias
    if not _finite(vs):
        vs = -5.2 * max(groundspeed, target_ias, 130.0)
    if not _finite(pitch):
        pitch = 2.5
    if not _finite(roll):
        roll = 0.0
    if not _finite(aoa):
        aoa = 4.0

    runway_heading = state.get("target_hdg_deg")
    runway_heading = float(runway_heading) % 360.0 if _finite(runway_heading) else heading

    track_err = _angdiff(runway_heading, track)
    heading_err = _angdiff(runway_heading, heading)
    crab_angle = abs(_angdiff(heading, track))

    wow = bool(state.get("weight_on_wheels"))
    on_ground = bool(state.get("on_ground")) or wow or (agl < 2.0 and groundspeed < 120.0)

    if on_ground:
        mode = "rollout"
    elif _AP["mode"] == "go_around":
        mode = "go_around"
    elif agl < 50.0:
        mode = "flare"
    elif agl < 1000.0:
        mode = "approach"
    else:
        mode = "intercept"

    wind_buffer = 5.0 + min(crab_angle * 0.6, 8.0)
    if _AP["loc_age"] > 0.5 or _AP["gs_age"] > 0.5:
        wind_buffer += 2.0
    if live_engines == 1:
        wind_buffer = max(wind_buffer, 10.0)
    elif live_engines == 0:
        wind_buffer = max(wind_buffer, 14.0)
    if agl < 150.0:
        wind_buffer = min(wind_buffer, 8.0 if live_engines > 0 else 12.0)

    if mode == "go_around":
        flaps = 20 if agl < 100.0 else 15
        gear = 1.0 if agl < 100.0 else 0.0
    else:
        if live_engines == 0:
            if agl > 1800.0 and (not _finite(dme) or dme > 7.0):
                flaps = 5
            elif agl > 900.0 and (not _finite(dme) or dme > 4.0):
                flaps = 15
            elif agl > 300.0 and (not _finite(dme) or dme > 1.8):
                flaps = 20
            else:
                flaps = 30
        elif live_engines == 1:
            if agl > 2200.0:
                flaps = 5
            elif agl > 1300.0:
                flaps = 15
            elif agl > 350.0:
                flaps = 20
            else:
                flaps = 30
        else:
            if agl > 2600.0:
                flaps = 5
            elif agl > 1600.0:
                flaps = 15
            elif agl > 800.0:
                flaps = 20
            else:
                flaps = 30
        gear = 1.0 if (agl < 2500.0 or (_finite(dme) and dme < 10.0) or mode in ("approach", "flare")) else 0.0

    if mode == "intercept":
        speed_target = max(target_ias, vref + (30.0 if live_engines == 0 else 25.0 if live_engines == 1 else 20.0))
    elif mode == "approach":
        speed_target = max(vref + wind_buffer, min(target_ias, vref + wind_buffer + 5.0))
    elif mode == "flare":
        speed_target = vref + (5.0 if live_engines == 0 else 2.0)
    elif mode == "go_around":
        speed_target = max(vref + 25.0, 160.0 if live_engines == 1 else 170.0)
    else:
        speed_target = 0.0

    if flaps <= 5:
        speed_target = max(speed_target, 175.0 if live_engines == 0 else 170.0)
    elif flaps <= 15:
        speed_target = max(speed_target, 155.0 if live_engines == 0 else 150.0)
    elif flaps <= 20:
        speed_target = max(speed_target, 150.0 if live_engines == 0 else 145.0)
    elif mode not in ("flare", "rollout"):
        speed_target = max(speed_target, vref + (8.0 if live_engines == 0 else 5.0))

    base_vs = (
        target_vs
        if target_vs is not None and -2500.0 < target_vs < -100.0
        else -5.2 * max(groundspeed, speed_target, 130.0)
    )

    loc_ctl = loc if reliable_loc else 0.0
    gs_ctl = gs if reliable_gs else 0.0

    if on_ground or mode == "go_around":
        _AP["loc_i"] *= max(0.0, 1.0 - 2.0 * dt)
        _AP["gs_i"] *= max(0.0, 1.0 - 2.0 * dt)
    else:
        _AP["loc_i"] = _clamp(_AP["loc_i"] + loc_ctl * dt, -2.5, 2.5)
        _AP["gs_i"] = _clamp(_AP["gs_i"] + gs_ctl * dt, -2.0, 2.0)

    unstable_now = False
    if mode in ("intercept", "approach", "flare") and live_engines > 0 and agl < 600.0:
        if agl < 500.0:
            unstable_now = unstable_now or abs(loc_ctl) > 3.0 or abs(gs_ctl) > 2.5
            unstable_now = unstable_now or ias > speed_target + 22.0 or ias < vref - 10.0
            unstable_now = unstable_now or vs < -1600.0 or abs(roll) > 25.0
        if agl < 200.0:
            unstable_now = unstable_now or abs(loc_ctl) > 2.2 or abs(gs_ctl) > 1.8
            unstable_now = unstable_now or ias > speed_target + 18.0 or ias < vref - 8.0
            unstable_now = unstable_now or vs < -1200.0 or abs(roll) > 15.0
            unstable_now = unstable_now or not reliable_loc or (not reliable_gs and agl < 120.0)
        if bool(state.get("below_decision_height")) and agl < 250.0:
            unstable_now = unstable_now or abs(loc_ctl) > 2.0 or abs(gs_ctl) > 1.5

    if unstable_now:
        _AP["unstable_t"] += dt
    else:
        _AP["unstable_t"] = max(0.0, _AP["unstable_t"] - 2.0 * dt)

    if (
        _AP["unstable_t"] > 1.5
        and live_engines > 0
        and agl > 35.0
        and mode != "go_around"
    ):
        _AP["mode"] = "go_around"
        mode = "go_around"
        flaps = 20 if agl < 100.0 else 15
        gear = 1.0 if agl < 100.0 else 0.0
        speed_target = max(vref + 25.0, 160.0 if live_engines == 1 else 170.0)

    if mode != "go_around":
        _AP["mode"] = mode

    desired_bank = _clamp(-7.0 * loc_ctl - 2.0 * _AP["loc_i"] + 0.35 * track_err, -20.0, 20.0)
    if mode == "flare":
        desired_bank = _clamp(desired_bank, -5.0, 5.0)
    elif mode == "approach":
        desired_bank = _clamp(desired_bank, -15.0, 15.0)
    elif mode == "rollout":
        desired_bank = _clamp(-4.0 * loc_ctl + 0.15 * track_err, -8.0, 8.0)

    aileron = _clamp(0.07 * (desired_bank - roll), -1.0, 1.0)

    if mode == "rollout":
        rudder = _clamp(-0.12 * loc_ctl + 0.08 * heading_err, -1.0, 1.0)
    elif mode == "flare":
        rudder = _clamp(-0.05 * loc_ctl + 0.06 * heading_err, -1.0, 1.0)
    elif mode == "go_around":
        rudder = _clamp(-0.03 * loc_ctl + 0.04 * heading_err, -1.0, 1.0)
    else:
        rudder = _clamp(-0.06 * loc_ctl + 0.02 * heading_err, -1.0, 1.0)

    speed_error = speed_target - ias

    if mode == "go_around":
        throttle_cmd = 1.0
    elif mode == "rollout" or live_engines == 0:
        throttle_cmd = 0.0
    elif mode == "flare":
        throttle_cmd = 0.08 if agl > 20.0 else 0.02
        if vs < -900.0 and agl > 15.0:
            throttle_cmd = max(throttle_cmd, 0.25)
    else:
        _AP["spd_i"] = _clamp(_AP["spd_i"] + speed_error * dt * 0.08, -10.0, 10.0)
        throttle_cmd = 0.55 + 0.018 * speed_error + 0.012 * _AP["spd_i"] + 0.00018 * (base_vs - vs)
        if agl < 300.0 and vs < -900.0:
            throttle_cmd += 0.15
        if agl < 300.0 and speed_error > 8.0:
            throttle_cmd += 0.12
        if live_engines == 1:
            throttle_cmd = max(throttle_cmd, 0.65)

    if aoa > 10.0 and mode != "rollout":
        throttle_cmd = max(throttle_cmd, 0.85 if mode != "flare" else 0.3)

    throttle_cmd = _clamp(throttle_cmd, 0.0, 1.0)
    throttle_1_tgt = throttle_cmd if eng1_live else 0.0
    throttle_2_tgt = throttle_cmd if eng2_live else 0.0

    throttle_1 = _slew(_AP["last_throttle1"], throttle_1_tgt, 0.9, 1.2, dt)
    throttle_2 = _slew(_AP["last_throttle2"], throttle_2_tgt, 0.9, 1.2, dt)
    _AP["last_throttle1"] = throttle_1
    _AP["last_throttle2"] = throttle_2

    rudder += 0.35 * (throttle_2 - throttle_1)
    rudder = _clamp(rudder, -1.0, 1.0)

    if mode == "go_around":
        desired_pitch = 9.0 if live_engines == 1 else 10.0
        desired_vs = 1200.0
        pitch_error = desired_pitch - pitch
        vs_error = desired_vs - vs
        elevator = _clamp(0.09 * pitch_error + 0.00035 * vs_error, -1.0, 1.0)
        _AP["trim"] = _clamp(_AP["trim"] + dt * 0.04 * pitch_error, -1.0, 1.0)
    elif mode == "rollout":
        elevator = -0.05
        _AP["trim"] *= max(0.0, 1.0 - 1.5 * dt)
    elif mode == "flare":
        flare_height = 45.0 if ra_valid else 55.0
        flare_frac = _clamp((flare_height - agl) / flare_height, 0.0, 1.0)
        desired_pitch = 2.5 + 2.5 * flare_frac + (0.5 if live_engines == 0 else 0.0)
        desired_vs = _clamp(-220.0 + 100.0 * flare_frac, -260.0, -90.0)
        pitch_error = desired_pitch - pitch
        vs_error = desired_vs - vs
        elevator = _clamp(0.12 * pitch_error + 0.00045 * vs_error, -1.0, 1.0)
        if aoa > 12.0:
            elevator = min(elevator, 0.15)
        _AP["trim"] = _clamp(
            _AP["trim"] + dt * (0.04 * pitch_error + 0.00006 * vs_error), -1.0, 1.0
        )
    else:
        desired_vs = base_vs - 300.0 * gs_ctl - 80.0 * _AP["gs_i"]
        if mode == "intercept" and gs_ctl > 1.2:
            desired_vs -= 200.0
        elif mode == "intercept" and gs_ctl < -1.2:
            desired_vs += 120.0
        desired_vs = _clamp(desired_vs, -1800.0, -150.0 if agl > 80.0 else -80.0)
        desired_pitch = _clamp(2.8 - 1.1 * gs_ctl + 0.03 * speed_error, 0.5, 7.0)
        if agl < 200.0 and vs < -900.0:
            desired_pitch += 1.5
        if ias < vref - 5.0 or aoa > 11.0:
            desired_pitch = min(desired_pitch, 2.0)
        pitch_error = desired_pitch - pitch
        vs_error = desired_vs - vs
        elevator = _clamp(0.10 * pitch_error + 0.00042 * vs_error, -1.0, 1.0)
        _AP["trim"] = _clamp(
            _AP["trim"] + dt * (0.035 * pitch_error + 0.00008 * vs_error), -1.0, 1.0
        )

    elevator_trim = _clamp(_AP["trim"], -1.0, 1.0)

    if mode == "rollout":
        spoiler = 1.0
        speedbrake = 1.0
        autobrake = 1.0 if groundspeed > 5.0 else 0.2
        if live_engines == 2 and groundspeed > 40.0:
            reverse_1 = 1.0 if groundspeed > 80.0 else 0.6
            reverse_2 = 1.0 if groundspeed > 80.0 else 0.6
        else:
            reverse_1 = 0.0
            reverse_2 = 0.0
        throttle_1 = 0.0
        throttle_2 = 0.0
    elif mode == "go_around":
        spoiler = 0.0
        speedbrake = 0.0
        autobrake = 0.0
        reverse_1 = 0.0
        reverse_2 = 0.0
    else:
        spoiler = 0.0
        speedbrake = 0.25 if (live_engines == 0 and reliable_gs and gs_ctl > 1.5 and _finite(dme) and dme < 4.0) else 0.0
        autobrake = 0.0
        reverse_1 = 0.0
        reverse_2 = 0.0

    return {
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": _clamp(throttle_1, 0.0, 1.0),
        "throttle_2": _clamp(throttle_2, 0.0, 1.0),
        "elevator_trim": elevator_trim,
        "flaps": float(_clamp(flaps, 0.0, 40.0)),
        "gear": float(1.0 if gear >= 0.5 else 0.0),
        "speedbrake": _clamp(speedbrake, 0.0, 1.0),
