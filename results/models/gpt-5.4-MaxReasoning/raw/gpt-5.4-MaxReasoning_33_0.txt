using System;
using System.Diagnostics;
using System.IO;
using System.Text;

public static class Program
{
    private const long TimeBudgetMs = 280_000;

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        if (!scanner.TryNextInt(out int n))
        {
            return;
        }

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int size = checked(n * n);
        var flow = new int[size];
        var dist = new int[size];
        var flowStrength = new long[n];
        var distCloseness = new long[n];

        for (int i = 0; i < n; i++)
        {
            int rowBase = i * n;
            for (int j = 0; j < n; j++)
            {
                int value = scanner.NextInt();
                flow[rowBase + j] = value;
                long magnitude = Math.Abs((long)value);
                flowStrength[i] += magnitude;
                flowStrength[j] += magnitude;
            }
        }

        for (int i = 0; i < n; i++)
        {
            int rowBase = i * n;
            for (int j = 0; j < n; j++)
            {
                int value = scanner.NextInt();
                dist[rowBase + j] = value;
                long weight = value;
                distCloseness[i] += weight;
                distCloseness[j] += weight;
            }
        }

        var stopwatch = Stopwatch.StartNew();
        var rng = new Random(1234567 + n);

        int[] facilityOrder = SortIndicesDescending(flowStrength);
        int[] locationOrder = SortIndicesAscending(distCloseness);

        int[] bestPermutation = BuildPermutationFromOrders(n, facilityOrder, locationOrder);
        long bestCost = ComputeCost(n, flow, dist, bestPermutation);

        if (n <= 90)
        {
            ImproveSmallInstance(n, flow, dist, ref bestPermutation, ref bestCost, stopwatch, rng);
        }
        else
        {
            ImproveLargeInstance(
                n,
                flow,
                dist,
                facilityOrder,
                locationOrder,
                ref bestPermutation,
                ref bestCost,
                stopwatch,
                rng);
        }

        bestCost = ComputeCost(n, flow, dist, bestPermutation);

        var output = new StringBuilder();
        output.Append(bestCost).Append('\n');
        for (int i = 0; i < n; i++)
        {
            if (i > 0)
            {
                output.Append(' ');
            }

            output.Append(bestPermutation[i]);
        }

        Console.Write(output.ToString());
    }

    private static void ImproveSmallInstance(
        int n,
        int[] flow,
        int[] dist,
        ref int[] bestPermutation,
        ref long bestCost,
        Stopwatch stopwatch,
        Random rng)
    {
        int[] current = (int[])bestPermutation.Clone();
        int[] inverse = new int[n];
        BuildInverse(current, inverse);
        long currentCost = bestCost;

        ExhaustiveSwapLocalSearch(n, flow, dist, current, inverse, ref currentCost, stopwatch);
        currentCost = ComputeCost(n, flow, dist, current);
        if (currentCost < bestCost)
        {
            bestCost = currentCost;
            bestPermutation = current;
        }

        int failures = 0;
        int shakeSize = 2;
        while (stopwatch.ElapsedMilliseconds < TimeBudgetMs && failures < 8)
        {
            int[] work = (int[])bestPermutation.Clone();
            int[] workInverse = new int[n];
            BuildInverse(work, workInverse);
            long workCost = bestCost;

            ShakeSolution(n, flow, dist, work, workInverse, ref workCost, rng, Math.Min(Math.Max(1, n - 1), shakeSize), stopwatch);
            ExhaustiveSwapLocalSearch(n, flow, dist, work, workInverse, ref workCost, stopwatch);
            workCost = ComputeCost(n, flow, dist, work);

            if (workCost < bestCost)
            {
                bestCost = workCost;
                bestPermutation = work;
                failures = 0;
                shakeSize = 2;
            }
            else
            {
                failures++;
                shakeSize = Math.Min(shakeSize + 1, Math.Max(3, n / 2));
            }
        }
    }

    private static void ImproveLargeInstance(
        int n,
        int[] flow,
        int[] dist,
        int[] facilityOrder,
        int[] locationOrder,
        ref int[] bestPermutation,
        ref long bestCost,
        Stopwatch stopwatch,
        Random rng)
    {
        int facilityK = Math.Min(n - 1, n <= 200 ? 20 : n <= 800 ? 12 : 8);
        int locationK = Math.Min(n - 1, n <= 200 ? 20 : n <= 800 ? 12 : 8);

        int[] facilityCandidates = BuildFacilityCandidates(n, flow, facilityK);
        int[] locationCandidates = BuildLocationCandidates(n, dist, locationK);

        int[] current = (int[])bestPermutation.Clone();
        int[] inverse = new int[n];
        BuildInverse(current, inverse);
        long currentCost = bestCost;

        CandidateLocalSearch(
            n,
            flow,
            dist,
            facilityCandidates,
            facilityK,
            locationCandidates,
            locationK,
            current,
            inverse,
            ref currentCost,
            stopwatch,
            rng);

        currentCost = ComputeCost(n, flow, dist, current);
        if (currentCost < bestCost)
        {
            bestCost = currentCost;
            bestPermutation = current;
        }

        int randomWindow = Math.Min(n, n <= 120 ? 6 : n <= 500 ? 12 : 20);
        int failures = 0;
        int shakeSize = 2;

        while (stopwatch.ElapsedMilliseconds < TimeBudgetMs)
        {
            bool useRestart = failures > 0 && failures % 4 == 0;
            int[] work = useRestart
                ? BuildRandomizedInitialPermutation(n, facilityOrder, locationOrder, rng, randomWindow)
                : (int[])bestPermutation.Clone();

            int[] workInverse = new int[n];
            BuildInverse(work, workInverse);

            long workCost = useRestart ? ComputeCost(n, flow, dist, work) : bestCost;
            if (!useRestart)
            {
                int appliedShake = Math.Min(Math.Max(2, Math.Min(12, n - 1)), shakeSize);
                ShakeSolution(n, flow, dist, work, workInverse, ref workCost, rng, appliedShake, stopwatch);
            }

            CandidateLocalSearch(
                n,
                flow,
                dist,
                facilityCandidates,
                facilityK,
                locationCandidates,
                locationK,
                work,
                workInverse,
                ref workCost,
                stopwatch,
                rng);

            workCost = ComputeCost(n, flow, dist, work);
            if (workCost < bestCost)
            {
                bestCost = workCost;
                bestPermutation = work;
                failures = 0;
                shakeSize = 2;
            }
            else
            {
                failures++;
                shakeSize = Math.Min(shakeSize + 1, 10);
            }
        }
    }

    private static void ExhaustiveSwapLocalSearch(
        int n,
        int[] flow,
        int[] dist,
        int[] permutation,
        int[] inverse,
        ref long cost,
        Stopwatch stopwatch)
    {
        while (stopwatch.ElapsedMilliseconds < TimeBudgetMs)
        {
            long bestDelta = 0;
            int bestI = -1;
            int bestJ = -1;

            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    long delta = ComputeSwapDelta(n, flow, dist, permutation, i, j);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestI = i;
                        bestJ = j;
                    }
                }

                if ((i & 7) == 0 && stopwatch.ElapsedMilliseconds >= TimeBudgetMs)
                {
                    break;
                }
            }

            if (bestI < 0)
            {
                break;
            }

            cost += bestDelta;
            ApplySwap(permutation, inverse, bestI, bestJ);
        }
    }

    private static void CandidateLocalSearch(
        int n,
        int[] flow,
        int[] dist,
        int[] facilityCandidates,
        int facilityK,
        int[] locationCandidates,
        int locationK,
        int[] permutation,
        int[] inverse,
        ref long cost,
        Stopwatch stopwatch,
        Random rng)
    {
        int[] order = new int[n];
        int[] seen = new int[n];
        for (int i = 0; i < n; i++)
        {
            order[i] = i;
        }

        int stamp = 1;
        int randomTrials = n <= 200 ? 6 : n <= 800 ? 4 : 3;

        while (stopwatch.ElapsedMilliseconds < TimeBudgetMs)
        {
            Shuffle(order, rng);
            bool improved = false;

            for (int idx = 0; idx < n; idx++)
            {
                int facility = order[idx];
                long bestDelta = 0;
                int bestPartner = -1;

                if (++stamp == int.MaxValue)
                {
                    Array.Clear(seen, 0, seen.Length);
                    stamp = 1;
                }

                int facilityOffset = facility * facilityK;
                for (int t = 0; t < facilityK; t++)
                {
                    int other = facilityCandidates[facilityOffset + t];
                    if (other < 0 || other == facility || seen[other] == stamp)
                    {
                        continue;
                    }

                    seen[other] = stamp;
                    long delta = ComputeSwapDelta(n, flow, dist, permutation, facility, other);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestPartner = other;
                    }
                }

                int location = permutation[facility];
                int locationOffset = location * locationK;
                for (int t = 0; t < locationK; t++)
                {
                    int otherLocation = locationCandidates[locationOffset + t];
                    if (otherLocation < 0)
                    {
                        continue;
                    }

                    int other = inverse[otherLocation];
                    if (other == facility || seen[other] == stamp)
                    {
                        continue;
                    }

                    seen[other] = stamp;
                    long delta = ComputeSwapDelta(n, flow, dist, permutation, facility, other);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestPartner = other;
                    }
                }

                for (int t = 0; t < randomTrials; t++)
                {
                    int other = rng.Next(n - 1);
                    if (other >= facility)
                    {
                        other++;
                    }

                    if (seen[other] == stamp)
                    {
                        continue;
                    }

                    seen[other] = stamp;
                    long delta = ComputeSwapDelta(n, flow, dist, permutation, facility, other);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestPartner = other;
                    }
                }

                if (bestPartner >= 0)
                {
                    cost += bestDelta;
                    ApplySwap(permutation, inverse, facility, bestPartner);
                    improved = true;
                }

                if ((idx & 31) == 0 && stopwatch.ElapsedMilliseconds >= TimeBudgetMs)
                {
                    break;
                }
            }

            if (!improved)
            {
                break;
            }
        }
    }

    private static int[] BuildFacilityCandidates(int n, int[] flow, int k)
    {
        var result = new int[n * k];
        if (k == 0)
        {
            return result;
        }

        var bestScores = new long[k];
        var bestIndices = new int[k];

        for (int i = 0; i < n; i++)
        {
            for (int t = 0; t < k; t++)
            {
                bestScores[t] = long.MinValue;
                bestIndices[t] = -1;
            }

            int rowBase = i * n;
            for (int j = 0; j < n; j++)
            {
                if (j == i)
                {
                    continue;
                }

                int colBase = j * n;
                long score = Math.Abs((long)flow[rowBase + j]) + Math.Abs((long)flow[colBase + i]);
                if (score <= bestScores[k - 1])
                {
                    continue;
                }

                int pos = k - 1;
                while (pos > 0 && score > bestScores[pos - 1])
                {
                    bestScores[pos] = bestScores[pos - 1];
                    bestIndices[pos] = bestIndices[pos - 1];
                    pos--;
                }

                bestScores[pos] = score;
                bestIndices[pos] = j;
            }

            int offset = i * k;
            for (int t = 0; t < k; t++)
            {
                result[offset + t] = bestIndices[t];
            }
        }

        return result;
    }

    private static int[] BuildLocationCandidates(int n, int[] dist, int k)
    {
        var result = new int[n * k];
        if (k == 0)
        {
            return result;
        }

        var bestScores = new long[k];
        var bestIndices = new int[k];

        for (int i = 0; i < n; i++)
        {
            for (int t = 0; t < k; t++)
            {
                bestScores[t] = long.MaxValue;
                bestIndices[t] = -1;
            }

            int rowBase = i * n;
            for (int j = 0; j < n; j++)
            {
                if (j == i)
                {
                    continue;
                }

                long score = (long)dist[rowBase + j] + dist[j * n + i];
                if (score >= bestScores[k - 1])
                {
                    continue;
                }

                int pos = k - 1;
                while (pos > 0 && score < bestScores[pos - 1])
                {
                    bestScores[pos] = bestScores[pos - 1];
                    bestIndices[pos] = bestIndices[pos - 1];
                    pos--;
                }

                bestScores[pos] = score;
                bestIndices[pos] = j;
            }

            int offset = i * k;
            for (int t = 0; t < k; t++)
            {
                result[offset + t] = bestIndices[t];
            }
        }

        return result;
    }

    private static int[] BuildPermutationFromOrders(int n, int[] facilityOrder, int[] locationOrder)
    {
        var permutation = new int[n];
        for (int i = 0; i < n; i++)
        {
            permutation[facilityOrder[i]] = locationOrder[i];
        }

        return permutation;
    }

    private static int[] BuildRandomizedInitialPermutation(
        int n,
        int[] facilityOrder,
        int[] locationOrder,
        Random rng,
        int window)
    {
        int[] facilities = (int[])facilityOrder.Clone();
        int[] locations = (int[])locationOrder.Clone();

        window = Math.Max(1, window);
        for (int i = 0; i < n; i++)
        {
            int upper = Math.Min(n - 1, i + window - 1);
            int facilityPick = i + rng.Next(upper - i + 1);
            int locationPick = i + rng.Next(upper - i + 1);
            SwapValues(facilities, i, facilityPick);
            SwapValues(locations, i, locationPick);
        }

        return BuildPermutationFromOrders(n, facilities, locations);
    }

    private static void ShakeSolution(
        int n,
        int[] flow,
        int[] dist,
        int[] permutation,
        int[] inverse,
        ref long cost,
        Random rng,
        int swaps,
        Stopwatch stopwatch)
    {
        if (n < 2)
        {
            return;
        }

        for (int step = 0; step < swaps && stopwatch.ElapsedMilliseconds < TimeBudgetMs; step++)
        {
            int i = rng.Next(n);
            int j = rng.Next(n - 1);
            if (j >= i)
            {
                j++;
            }

            long delta = ComputeSwapDelta(n, flow, dist, permutation, i, j);
            cost += delta;
            ApplySwap(permutation, inverse, i, j);
        }
    }

    private static long ComputeCost(int n, int[] flow, int[] dist, int[] permutation)
    {
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int flowRow = i * n;
            int distRow = permutation[i] * n;
            for (int j = 0; j < n; j++)
            {
                total += (long)flow[flowRow + j] * dist[distRow + permutation[j]];
            }
        }

        return total;
    }

    // Delta for swapping facilities r and s in an ordered-pairs QAP objective.
    private static long ComputeSwapDelta(int n, int[] flow, int[] dist, int[] permutation, int r, int s)
    {
        if (r == s)
        {
            return 0;
        }

        int locationR = permutation[r];
        int locationS = permutation[s];

        int flowRowR = r * n;
        int flowRowS = s * n;
        int distRowR = locationR * n;
        int distRowS = locationS * n;

        long delta =
            (((long)flow[flowRowR + r] - flow[flowRowS + s]) * ((long)dist[distRowS + locationS] - dist[distRowR + locationR])) +
            (((long)flow[flowRowR + s] - flow[flowRowS + r]) * ((long)dist[distRowS + locationR] - dist[distRowR + locationS]));

        for (int k = 0; k < n; k++)
        {
            if (k == r || k == s)
            {
                continue;
            }

            int pk = permutation[k];
            int flowRowK = k * n;
            int distRowK = pk * n;

            delta +=
                (((long)flow[flowRowR + k] - flow[flowRowS + k]) * ((long)dist[distRowS + pk] - dist[distRowR + pk])) +
                (((long)flow[flowRowK + r] - flow[flowRowK + s]) * ((long)dist[distRowK + locationS] - dist[distRowK + locationR]));
        }

        return delta;
    }

    private static void ApplySwap(int[] permutation, int[] inverse, int i, int j)
    {
        int locationI = permutation[i];
        int locationJ = permutation[j];
        permutation[i] = locationJ;
        permutation[j] = locationI;
        inverse[locationI] = j;
        inverse[locationJ] = i;
    }

    private static void BuildInverse(int[] permutation, int[] inverse)
    {
        for (int i = 0; i < permutation.Length; i++)
        {
            inverse[permutation[i]] = i;
        }
    }

    private static int[] SortIndicesAscending(long[] values)
    {
        int n = values.Length;
        var keys = new long[n];
        var indices = new int[n];
        for (int i = 0; i < n; i++)
        {
            keys[i] = values[i];
            indices[i] = i;
        }

        Array.Sort(keys, indices);
        return indices;
    }

    private static int[] SortIndicesDescending(long[] values)
    {
        int[] indices = SortIndicesAscending(values);
        Array.Reverse(indices);
        return indices;
    }

    private static void Shuffle(int[] values, Random rng)
    {
        for (int i = values.Length - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            int tmp = values[i];
            values[i] = values[j];
            values[j] = tmp;
        }
    }

    private static void SwapValues(int[] values, int i, int j)
    {
        if (i == j)
        {
            return;
        }

        int tmp = values[i];
        values[i] = values[j];
        values[j] = tmp;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _position;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            if (!TryNextInt(out int value))
            {
                throw new EndOfStreamException();
            }

            return value;
        }

        public bool TryNextInt(out int value)
        {
            int c;
            do
            {
                c = Read();
                if (c < 0)
                {
                    value = 0;
                    return false;
                }
            }
            while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int result = 0;
            while (c > 32)
            {
                result = checked(result * 10 + (c - '0'));
                c = Read();
            }

            value = result * sign;
            return true;
        }

        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }
    }
}
