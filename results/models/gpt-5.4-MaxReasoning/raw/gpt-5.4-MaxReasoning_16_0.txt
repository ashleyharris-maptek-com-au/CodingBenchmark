#include <algorithm>
#include <cstddef>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <random>
#include <utility>
#include <vector>

using i64 = long long;

namespace {

constexpr i64 INF = std::numeric_limits<i64>::max() / 4;

struct Operation {
    int machine = 0;
    i64 duration = 0;
};

struct Instance {
    int jobs = 0;
    int machines = 0;
    int total_tasks = 0;
    std::vector<std::vector<Operation>> job_ops;
    std::vector<i64> job_total_work;
    std::vector<i64> machine_total_work;
    std::vector<int> machine_task_count;
};

struct Schedule {
    std::vector<std::vector<i64>> starts;
    i64 makespan = INF;
};

enum class Mode {
    ExactSmall,
    MediumInsertion,
    LargeAppend
};

enum class Rule {
    EstSpt,
    EstLrpt,
    Mwkr,
    Mopnr,
    MachineLoad,
    Balanced
};

struct Candidate {
    int job = -1;
    int task_idx = -1;
    int machine = -1;
    i64 dur = 0;
    i64 est = 0;
    i64 job_remaining = 0;
    i64 machine_remaining = 0;
    int ops_left = 0;
};

struct Interval {
    i64 start = 0;
    i64 end = 0;
};

std::vector<std::vector<i64>> blank_starts(const Instance& ins) {
    std::vector<std::vector<i64>> starts(ins.jobs);
    for (int j = 0; j < ins.jobs; ++j) {
        starts[j].assign(ins.job_ops[j].size(), -1);
    }
    return starts;
}

void maybe_take(Schedule& best, Schedule cand) {
    if (cand.makespan < best.makespan) {
        best = std::move(cand);
    }
}

std::vector<int> make_bias(int n, std::uint64_t seed) {
    std::vector<int> order(n);
    std::iota(order.begin(), order.end(), 0);
    std::mt19937_64 rng(seed);
    std::shuffle(order.begin(), order.end(), rng);
    std::vector<int> bias(n, 0);
    for (int i = 0; i < n; ++i) {
        bias[order[i]] = i;
    }
    return bias;
}

bool better_candidate(const Candidate& a, const Candidate& b, Rule rule,
                      const std::vector<int>& bias) {
    switch (rule) {
        case Rule::EstSpt:
            if (a.est != b.est) return a.est < b.est;
            if (a.dur != b.dur) return a.dur < b.dur;
            if (a.job_remaining != b.job_remaining) return a.job_remaining > b.job_remaining;
            if (a.ops_left != b.ops_left) return a.ops_left > b.ops_left;
            break;
        case Rule::EstLrpt:
            if (a.est != b.est) return a.est < b.est;
            if (a.job_remaining != b.job_remaining) return a.job_remaining > b.job_remaining;
            if (a.dur != b.dur) return a.dur > b.dur;
            if (a.machine_remaining != b.machine_remaining) {
                return a.machine_remaining > b.machine_remaining;
            }
            break;
        case Rule::Mwkr:
            if (a.job_remaining != b.job_remaining) return a.job_remaining > b.job_remaining;
            if (a.est != b.est) return a.est < b.est;
            if (a.dur != b.dur) return a.dur > b.dur;
            if (a.ops_left != b.ops_left) return a.ops_left > b.ops_left;
            break;
        case Rule::Mopnr:
            if (a.ops_left != b.ops_left) return a.ops_left > b.ops_left;
            if (a.est != b.est) return a.est < b.est;
            if (a.dur != b.dur) return a.dur < b.dur;
            if (a.job_remaining != b.job_remaining) return a.job_remaining > b.job_remaining;
            break;
        case Rule::MachineLoad:
            if (a.machine_remaining != b.machine_remaining) {
                return a.machine_remaining > b.machine_remaining;
            }
            if (a.est != b.est) return a.est < b.est;
            if (a.job_remaining != b.job_remaining) return a.job_remaining > b.job_remaining;
            if (a.dur != b.dur) return a.dur > b.dur;
            break;
        case Rule::Balanced:
            if (a.est != b.est) return a.est < b.est;
            if (a.machine_remaining != b.machine_remaining) {
                return a.machine_remaining > b.machine_remaining;
            }
            if (a.ops_left != b.ops_left) return a.ops_left > b.ops_left;
            if (a.dur != b.dur) return a.dur < b.dur;
            break;
    }

    if (bias[a.job] != bias[b.job]) return bias[a.job] < bias[b.job];
    return a.job < b.job;
}

Candidate build_candidate(const Instance& ins, int job, const std::vector<int>& next_task,
                          const std::vector<i64>& job_ready,
                          const std::vector<i64>& machine_proxy_ready,
                          const std::vector<i64>& job_remaining,
                          const std::vector<i64>& machine_remaining) {
    const int idx = next_task[job];
    const Operation& op = ins.job_ops[job][idx];
    Candidate cand;
    cand.job = job;
    cand.task_idx = idx;
    cand.machine = op.machine;
    cand.dur = op.duration;
    cand.est = std::max(job_ready[job], machine_proxy_ready[op.machine]);
    cand.job_remaining = job_remaining[job];
    cand.machine_remaining = machine_remaining[op.machine];
    cand.ops_left = static_cast<int>(ins.job_ops[job].size()) - idx;
    return cand;
}

Schedule run_append_scheduler(const Instance& ins, Rule rule, std::uint64_t seed) {
    Schedule sched;
    sched.starts = blank_starts(ins);

    std::vector<int> next_task(ins.jobs, 0);
    std::vector<i64> job_ready(ins.jobs, 0);
    std::vector<i64> machine_ready(ins.machines, 0);
    std::vector<i64> job_remaining = ins.job_total_work;
    std::vector<i64> machine_remaining = ins.machine_total_work;
    std::vector<int> bias = make_bias(ins.jobs, seed);

    i64 makespan = 0;
    for (int done = 0; done < ins.total_tasks; ++done) {
        bool found = false;
        Candidate best;

        for (int j = 0; j < ins.jobs; ++j) {
            if (next_task[j] >= static_cast<int>(ins.job_ops[j].size())) {
                continue;
            }
            Candidate cand = build_candidate(ins, j, next_task, job_ready, machine_ready,
                                             job_remaining, machine_remaining);
            if (!found || better_candidate(cand, best, rule, bias)) {
                best = cand;
                found = true;
            }
        }

        const int j = best.job;
        const int idx = next_task[j];
        const Operation& op = ins.job_ops[j][idx];
        const i64 start = std::max(job_ready[j], machine_ready[op.machine]);
        const i64 end = start + op.duration;

        sched.starts[j][idx] = start;
        next_task[j]++;
        job_ready[j] = end;
        machine_ready[op.machine] = end;
        job_remaining[j] -= op.duration;
        machine_remaining[op.machine] -= op.duration;
        if (end > makespan) makespan = end;
    }

    sched.makespan = makespan;
    return sched;
}

i64 insert_earliest(std::vector<Interval>& machine_line, i64 release, i64 duration) {
    i64 t = release;
    std::size_t pos = 0;
    while (pos < machine_line.size()) {
        if (t + duration <= machine_line[pos].start) {
            break;
        }
        if (t < machine_line[pos].end) {
            t = machine_line[pos].end;
        }
        ++pos;
    }
    machine_line.insert(machine_line.begin() + static_cast<std::ptrdiff_t>(pos),
                        Interval{t, t + duration});
    return t;
}

Schedule run_insertion_scheduler(const Instance& ins, Rule rule, std::uint64_t seed) {
    Schedule sched;
    sched.starts = blank_starts(ins);

    std::vector<int> next_task(ins.jobs, 0);
    std::vector<i64> job_ready(ins.jobs, 0);
    std::vector<i64> machine_tail(ins.machines, 0);
    std::vector<i64> job_remaining = ins.job_total_work;
    std::vector<i64> machine_remaining = ins.machine_total_work;
    std::vector<int> bias = make_bias(ins.jobs, seed);
    std::vector<std::vector<Interval>> machine_lines(ins.machines);

    for (int m = 0; m < ins.machines; ++m) {
        machine_lines[m].reserve(ins.machine_task_count[m]);
    }

    i64 makespan = 0;
    for (int done = 0; done < ins.total_tasks; ++done) {
        bool found = false;
        Candidate best;

        for (int j = 0; j < ins.jobs; ++j) {
            if (next_task[j] >= static_cast<int>(ins.job_ops[j].size())) {
                continue;
            }
            Candidate cand = build_candidate(ins, j, next_task, job_ready, machine_tail,
                                             job_remaining, machine_remaining);
            if (!found || better_candidate(cand, best, rule, bias)) {
                best = cand;
                found = true;
            }
        }

        const int j = best.job;
        const int idx = next_task[j];
        const Operation& op = ins.job_ops[j][idx];
        const i64 start = insert_earliest(machine_lines[op.machine], job_ready[j], op.duration);
        const i64 end = start + op.duration;

        sched.starts[j][idx] = start;
        next_task[j]++;
        job_ready[j] = end;
        if (end > machine_tail[op.machine]) machine_tail[op.machine] = end;
        job_remaining[j] -= op.duration;
        machine_remaining[op.machine] -= op.duration;
        if (end > makespan) makespan = end;
    }

    sched.makespan = makespan;
    return sched;
}

class SmallExactSearch {
public:
    SmallExactSearch(const Instance& instance, Schedule incumbent)
        : ins(instance),
          next_task(ins.jobs, 0),
          job_ready(ins.jobs, 0),
          machine_ready(ins.machines, 0),
          job_remaining(ins.job_total_work),
          machine_remaining(ins.machine_total_work),
          current_starts(blank_starts(ins)),
          best(std::move(incumbent)),
          bias(make_bias(ins.jobs, 0xC0FFEEULL)) {
        node_limit = (ins.total_tasks <= 12) ? 4000000LL : 1500000LL;
    }

    Schedule solve() {
        dfs(0, 0);
        return best;
    }

private:
    const Instance& ins;
    std::vector<int> next_task;
    std::vector<i64> job_ready;
    std::vector<i64> machine_ready;
    std::vector<i64> job_remaining;
    std::vector<i64> machine_remaining;
    std::vector<std::vector<i64>> current_starts;
    Schedule best;
    std::vector<int> bias;
    i64 node_limit = 0;
    i64 nodes = 0;
    bool aborted = false;

    i64 lower_bound(i64 current_makespan) const {
        i64 lb = current_makespan;
        for (int j = 0; j < ins.jobs; ++j) {
            if (next_task[j] < static_cast<int>(ins.job_ops[j].size())) {
                lb = std::max(lb, job_ready[j] + job_remaining[j]);
            }
        }
        for (int m = 0; m < ins.machines; ++m) {
            if (machine_remaining[m] > 0) {
                lb = std::max(lb, machine_ready[m] + machine_remaining[m]);
            }
        }
        return lb;
    }

    void dfs(int scheduled, i64 current_makespan) {
        if (aborted) return;
        if (++nodes > node_limit) {
            aborted = true;
            return;
        }

        if (scheduled == ins.total_tasks) {
            if (current_makespan < best.makespan) {
                best.starts = current_starts;
                best.makespan = current_makespan;
            }
            return;
        }

        if (lower_bound(current_makespan) >= best.makespan) {
            return;
        }

        std::vector<Candidate> choices;
        choices.reserve(ins.jobs);
        for (int j = 0; j < ins.jobs; ++j) {
            if (next_task[j] >= static_cast<int>(ins.job_ops[j].size())) {
                continue;
            }
            choices.push_back(build_candidate(ins, j, next_task, job_ready, machine_ready,
                                              job_remaining, machine_remaining));
        }

        std::sort(choices.begin(), choices.end(),
                  [&](const Candidate& lhs, const Candidate& rhs) {
                      return better_candidate(lhs, rhs, Rule::EstLrpt, bias);
                  });

        for (const Candidate& cand : choices) {
            const int j = cand.job;
            const int idx = next_task[j];
            const Operation& op = ins.job_ops[j][idx];
            const int m = op.machine;
            const i64 start = std::max(job_ready[j], machine_ready[m]);
            const i64 end = start + op.duration;

            const i64 old_job_ready = job_ready[j];
            const i64 old_machine_ready = machine_ready[m];
            const i64 old_job_remaining = job_remaining[j];
            const i64 old_machine_remaining = machine_remaining[m];

            current_starts[j][idx] = start;
            next_task[j]++;
            job_ready[j] = end;
            machine_ready[m] = end;
            job_remaining[j] -= op.duration;
            machine_remaining[m] -= op.duration;

            dfs(scheduled + 1, std::max(current_makespan, end));

            current_starts[j][idx] = -1;
            next_task[j]--;
            job_ready[j] = old_job_ready;
            machine_ready[m] = old_machine_ready;
            job_remaining[j] = old_job_remaining;
            machine_remaining[m] = old_machine_remaining;

            if (aborted) return;
        }
    }
};

Mode choose_mode(const Instance& ins) {
    if (ins.total_tasks <= 18 && ins.jobs <= 12) {
        return Mode::ExactSmall;
    }

    i64 insertion_cost = 0;
    for (int cnt : ins.machine_task_count) {
        insertion_cost += static_cast<i64>(cnt) * static_cast<i64>(cnt);
    }

    if (ins.total_tasks <= 8000 && insertion_cost <= 12000000LL) {
        return Mode::MediumInsertion;
    }

    return Mode::LargeAppend;
}

Schedule solve_exact_small(const Instance& ins) {
    Schedule incumbent;
    incumbent.makespan = INF;

    maybe_take(incumbent, run_insertion_scheduler(ins, Rule::EstLrpt, 11));
    maybe_take(incumbent, run_insertion_scheduler(ins, Rule::Mwkr, 23));
    maybe_take(incumbent, run_append_scheduler(ins, Rule::EstLrpt, 37));
    maybe_take(incumbent, run_append_scheduler(ins, Rule::EstSpt, 53));

    SmallExactSearch search(ins, std::move(incumbent));
    return search.solve();
}

Schedule solve_medium(const Instance& ins) {
    Schedule best;
    best.makespan = INF;

    const std::vector<Rule> rules = {
        Rule::EstLrpt, Rule::Mwkr, Rule::EstSpt,
        Rule::Mopnr, Rule::MachineLoad, Rule::Balanced
    };

    int repeats = (ins.total_tasks <= 1000) ? 2 : 1;
    std::uint64_t seed = 0x9E3779B97F4A7C15ULL;
    for (int rep = 0; rep < repeats; ++rep) {
        for (Rule rule : rules) {
            maybe_take(best, run_insertion_scheduler(ins, rule, seed));
            seed += 0xBF58476D1CE4E5B9ULL;
        }
    }

    maybe_take(best, run_append_scheduler(ins, Rule::EstLrpt, seed + 1));
    return best;
}

Schedule solve_large(const Instance& ins) {
    Schedule best;
    best.makespan = INF;

    const std::vector<Rule> rules = {
        Rule::EstLrpt, Rule::Mwkr, Rule::EstSpt, Rule::Balanced
    };

    int repeats = (ins.total_tasks > 30000 || ins.jobs > 700) ? 1 : 2;
    std::uint64_t seed = 0xD6E8FEB86659FD93ULL;
    for (int rep = 0; rep < repeats; ++rep) {
        for (Rule rule : rules) {
            maybe_take(best, run_append_scheduler(ins, rule, seed));
            seed += 0x94D049BB133111EBULL;
        }
    }

    return best;
}

Schedule solve_instance(const Instance& ins) {
    if (ins.total_tasks == 0) {
        Schedule empty;
        empty.starts = blank_starts(ins);
        empty.makespan = 0;
        return empty;
    }

    switch (choose_mode(ins)) {
        case Mode::ExactSmall:
            return solve_exact_small(ins);
        case Mode::MediumInsertion:
            return solve_medium(ins);
        case Mode::LargeAppend:
            return solve_large(ins);
    }
    return solve_large(ins);
}

}  // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    Instance ins;
    if (!(std::cin >> ins.jobs >> ins.machines)) {
        return 0;
    }

    ins.job_ops.resize(ins.jobs);
    ins.job_total_work.assign(ins.jobs, 0);
    ins.machine_total_work.assign(ins.machines, 0);
    ins.machine_task_count.assign(ins.machines, 0);

    for (int j = 0; j < ins.jobs; ++j) {
        int task_count = 0;
        std::cin >> task_count;
        ins.job_ops[j].resize(task_count);
        for (int t = 0; t < task_count; ++t) {
            int machine = 0;
            i64 duration = 0;
            std::cin >> machine >> duration;
            ins.job_ops[j][t] = Operation{machine, duration};
            ins.job_total_work[j] += duration;
            ins.machine_total_work[machine] += duration;
            ins.machine_task_count[machine] += 1;
            ins.total_tasks += 1;
        }
    }

    Schedule best = solve_instance(ins);

    for (int j = 0; j < ins.jobs; ++j) {
        for (int t = 0; t < static_cast<int>(ins.job_ops[j].size()); ++t) {
            if (t > 0) std::cout << ' ';
            std::cout << best.starts[j][t] << ' ' << ins.job_ops[j][t].machine;
        }
        std::cout << '\n';
    }

    return 0;
}
