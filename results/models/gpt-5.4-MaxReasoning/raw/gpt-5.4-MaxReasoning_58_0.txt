const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = input.trim() ? JSON.parse(input) : {};

const rawGrid = Array.isArray(data.grid) ? data.grid.map((row) => String(row)) : [];
const robotsInput = Array.isArray(data.robots) ? data.robots : [];
const jobsInput = Array.isArray(data.jobs) ? data.jobs : [];

let maxX = -1;
let maxY = -1;

for (let i = 0; i < robotsInput.length; i++) {
  const robot = robotsInput[i];
  if (robot && Number.isFinite(robot.x) && Number.isFinite(robot.y)) {
    if (robot.x > maxX) maxX = robot.x;
    if (robot.y > maxY) maxY = robot.y;
  }
}

for (let i = 0; i < jobsInput.length; i++) {
  const job = jobsInput[i];
  if (job && Number.isFinite(job.x) && Number.isFinite(job.y)) {
    if (job.x > maxX) maxX = job.x;
    if (job.y > maxY) maxY = job.y;
  }
}

let width = 0;
for (let y = 0; y < rawGrid.length; y++) {
  if (rawGrid[y].length > width) width = rawGrid[y].length;
}
if (maxX + 1 > width) width = maxX + 1;
if (width < 1) width = 1;

let height = rawGrid.length;
if (maxY + 1 > height) height = maxY + 1;
if (height < 1) height = 1;

const cellCount = width * height;
const special = new Uint8Array(cellCount);

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < width && y < height;
}

function cellIndex(x, y) {
  return y * width + x;
}

for (let i = 0; i < robotsInput.length; i++) {
  const robot = robotsInput[i];
  if (robot && inBounds(robot.x, robot.y)) {
    special[cellIndex(robot.x, robot.y)] = 1;
  }
}

for (let i = 0; i < jobsInput.length; i++) {
  const job = jobsInput[i];
  if (job && inBounds(job.x, job.y)) {
    special[cellIndex(job.x, job.y)] = 1;
  }
}

const passable = new Uint8Array(cellCount);
for (let y = 0; y < height; y++) {
  const row = y < rawGrid.length ? rawGrid[y] : '';
  for (let x = 0; x < width; x++) {
    const idx = cellIndex(x, y);
    const ch = x < row.length ? row.charCodeAt(x) : 35;
    passable[idx] = special[idx] || ch !== 35 ? 1 : 0;
  }
}

const neighbors = new Int32Array(cellCount * 4);
const neighborCount = new Uint8Array(cellCount);

for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const idx = cellIndex(x, y);
    if (!passable[idx]) continue;

    let count = 0;
    const base = idx << 2;

    if (x > 0) {
      const left = idx - 1;
      if (passable[left]) neighbors[base + count++] = left;
    }
    if (x + 1 < width) {
      const right = idx + 1;
      if (passable[right]) neighbors[base + count++] = right;
    }
    if (y > 0) {
      const up = idx - width;
      if (passable[up]) neighbors[base + count++] = up;
    }
    if (y + 1 < height) {
      const down = idx + width;
      if (passable[down]) neighbors[base + count++] = down;
    }

    neighborCount[idx] = count;
  }
}

const robotStates = [];
let maxRobotId = -1;

for (let i = 0; i < robotsInput.length; i++) {
  const robot = robotsInput[i];
  if (!robot) continue;

  const id = Number(robot.id);
  const x = Number(robot.x);
  const y = Number(robot.y);

  if (!Number.isInteger(id) || !inBounds(x, y)) continue;

  if (id > maxRobotId) maxRobotId = id;
  robotStates.push({
    id,
    posIdx: cellIndex(x, y),
    candidateJobIndex: -1,
    candidateDist: 0,
  });
}

if (maxRobotId < 0) {
  process.stdout.write(JSON.stringify({ assignments: [] }));
  process.exit(0);
}

robotStates.sort((a, b) => a.id - b.id);

const assignments = Array.from({ length: maxRobotId + 1 }, () => []);

const jobs = [];
for (let i = 0; i < jobsInput.length; i++) {
  const job = jobsInput[i];
  if (!job) continue;

  const id = Number(job.id);
  const x = Number(job.x);
  const y = Number(job.y);
  const priority = Number(job.priority);

  if (!Number.isFinite(id) || !inBounds(x, y)) continue;

  jobs.push({
    id,
    x,
    y,
    priority: Number.isFinite(priority) ? priority : 0,
    cellIdx: cellIndex(x, y),
  });
}

jobs.sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  if (a.id !== b.id) return a.id - b.id;
  if (a.y !== b.y) return a.y - b.y;
  return a.x - b.x;
});

if (jobs.length === 0) {
  process.stdout.write(JSON.stringify({ assignments }));
  process.exit(0);
}

const assigned = new Uint8Array(jobs.length);
const jobsAtCell = new Array(cellCount);
const cellPtr = new Int32Array(cellCount);
const queue = new Int32Array(cellCount);
const seen = new Int32Array(cellCount);
let seenToken = 1;

function currentCellJob(cell) {
  const list = jobsAtCell[cell];
  if (list === undefined) return -1;

  let ptr = cellPtr[cell];
  while (ptr < list.length && assigned[list[ptr]]) ptr++;
  cellPtr[cell] = ptr;

  return ptr < list.length ? list[ptr] : -1;
}

function findNearestJob(posIdx) {
  if (seenToken === 2147483647) {
    seen.fill(0);
    seenToken = 1;
  }

  const token = seenToken++;
  let head = 0;
  let tail = 0;
  let depth = 0;
  let layerSize = 1;

  queue[tail++] = posIdx;
  seen[posIdx] = token;

  while (head < tail) {
    let nextLayerSize = 0;
    let bestJobIndex = -1;

    for (let i = 0; i < layerSize; i++) {
      const cell = queue[head++];
      const jobIndex = currentCellJob(cell);
      if (jobIndex !== -1) {
        if (bestJobIndex === -1 || jobs[jobIndex].id < jobs[bestJobIndex].id) {
          bestJobIndex = jobIndex;
        }
      }

      const base = cell << 2;
      const count = neighborCount[cell];
      for (let n = 0; n < count; n++) {
        const next = neighbors[base + n];
        if (seen[next] === token) continue;
        seen[next] = token;
        queue[tail++] = next;
        nextLayerSize++;
      }
    }

    if (bestJobIndex !== -1) {
      return { jobIndex: bestJobIndex, dist: depth };
    }

    layerSize = nextLayerSize;
    depth++;
  }

  return { jobIndex: -1, dist: 0 };
}

function updateCandidate(robot) {
  const result = findNearestJob(robot.posIdx);
  robot.candidateJobIndex = result.jobIndex;
  robot.candidateDist = result.dist;
}

for (let start = 0; start < jobs.length; ) {
  let end = start + 1;
  while (end < jobs.length && jobs[end].priority === jobs[start].priority) end++;

  const touchedCells = [];
  for (let i = start; i < end; i++) {
    const cell = jobs[i].cellIdx;
    if (jobsAtCell[cell] === undefined) {
      jobsAtCell[cell] = [i];
      cellPtr[cell] = 0;
      touchedCells.push(cell);
    } else {
      jobsAtCell[cell].push(i);
    }
  }

  for (let i = 0; i < robotStates.length; i++) {
    updateCandidate(robotStates[i]);
  }

  let remaining = end - start;
  while (remaining > 0) {
    let bestRobot = null;

    for (let i = 0; i < robotStates.length; i++) {
      const robot = robotStates[i];
      if (robot.candidateJobIndex !== -1 && assigned[robot.candidateJobIndex]) {
        updateCandidate(robot);
      }
      if (robot.candidateJobIndex === -1) continue;

      if (
        bestRobot === null ||
        robot.candidateDist < bestRobot.candidateDist ||
        (robot.candidateDist === bestRobot.candidateDist &&
          (jobs[robot.candidateJobIndex].id < jobs[bestRobot.candidateJobIndex].id ||
            (jobs[robot.candidateJobIndex].id === jobs[bestRobot.candidateJobIndex].id &&
              robot.id < bestRobot.id)))
      ) {
        bestRobot = robot;
      }
    }

    if (bestRobot === null) {
      let fallbackRobot = robotStates[0];
      for (let i = 1; i < robotStates.length; i++) {
        if (robotStates[i].id < fallbackRobot.id) fallbackRobot = robotStates[i];
      }

      for (let i = start; i < end; i++) {
        if (assigned[i]) continue;
        assigned[i] = 1;
        assignments[fallbackRobot.id].push(jobs[i].id);
        fallbackRobot.posIdx = jobs[i].cellIdx;
        remaining--;
      }
      break;
    }

    const jobIndex = bestRobot.candidateJobIndex;
    assigned[jobIndex] = 1;
    assignments[bestRobot.id].push(jobs[jobIndex].id);
    bestRobot.posIdx = jobs[jobIndex].cellIdx;
    remaining--;

    const list = jobsAtCell[jobs[jobIndex].cellIdx];
    if (list !== undefined) {
      let ptr = cellPtr[jobs[jobIndex].cellIdx];
      while (ptr < list.length && assigned[list[ptr]]) ptr++;
      cellPtr[jobs[jobIndex].cellIdx] = ptr;
    }

    updateCandidate(bestRobot);
  }

  for (let i = 0; i < touchedCells.length; i++) {
    const cell = touchedCells[i];
    jobsAtCell[cell] = undefined;
    cellPtr[cell] = 0;
  }

  start = end;
}

process.stdout.write(JSON.stringify({ assignments }));
