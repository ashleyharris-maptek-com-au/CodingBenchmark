OpCapability Shader
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.200000003
%float_0_8 = OpConstant %float 0.800000012
%float_1 = OpConstant %float 1
%float_0_33333334 = OpConstant %float 0.333333343
%light_ones = OpConstantComposite %v3float %float_1 %float_1 %float_1
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_Input_arr_gl_PerVertex_in_6 = OpTypePointer Input %arr_gl_PerVertex_in_6
%ptr_Input_v4float = OpTypePointer Input %v4float
%arr_v3float_6 = OpTypeArray %v3float %uint_6
%ptr_Input_arr_v3float_6 = OpTypePointer Input %arr_v3float_6
%ptr_Input_v3float = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_6 Input
%worldPos = OpVariable %ptr_Input_arr_v3float_6 Input
%normal = OpVariable %ptr_Input_arr_v3float_6 Input
%color = OpVariable %ptr_Input_arr_v3float_6 Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%main = OpFunction %void None %fn
%entry = OpLabel
%wp0_ptr = OpAccessChain %ptr_Input_v3float %worldPos %int_0
%wp2_ptr = OpAccessChain %ptr_Input_v3float %worldPos %int_2
%wp4_ptr = OpAccessChain %ptr_Input_v3float %worldPos %int_4
%p0 = OpLoad %v3float %wp0_ptr
%p2 = OpLoad %v3float %wp2_ptr
%p4 = OpLoad %v3float %wp4_ptr
%n0_ptr = OpAccessChain %ptr_Input_v3float %normal %int_0
%unused_n0 = OpLoad %v3float %n0_ptr
%edge0 = OpFSub %v3float %p2 %p0
%edge1 = OpFSub %v3float %p4 %p0
%edge0_x = OpCompositeExtract %float %edge0 0
%edge0_y = OpCompositeExtract %float %edge0 1
%edge0_z = OpCompositeExtract %float %edge0 2
%edge1_x = OpCompositeExtract %float %edge1 0
%edge1_y = OpCompositeExtract %float %edge1 1
%edge1_z = OpCompositeExtract %float %edge1 2
%cross_x_a = OpFMul %float %edge0_y %edge1_z
%cross_x_b = OpFMul %float %edge0_z %edge1_y
%cross_x = OpFSub %float %cross_x_a %cross_x_b
%cross_y_a = OpFMul %float %edge0_z %edge1_x
%cross_y_b = OpFMul %float %edge0_x %edge1_z
%cross_y = OpFSub %float %cross_y_a %cross_y_b
%cross_z_a = OpFMul %float %edge0_x %edge1_y
%cross_z_b = OpFMul %float %edge0_y %edge1_x
%cross_z = OpFSub %float %cross_z_a %cross_z_b
%face = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z
%N = OpExtInst %v3float %1 Normalize %face
%L = OpExtInst %v3float %1 Normalize %light_ones
%ndotl = OpDot %float %N %L
%diffuse = OpExtInst %float %1 FMax %ndotl %float_0
%c0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
%c4_ptr = OpAccessChain %ptr_Input_v3float %color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%c02 = OpFAdd %v3float %c0 %c2
%c024 = OpFAdd %v3float %c02 %c4
%base = OpVectorTimesScalar %v3float %c024 %float_0_33333334
%lit_scale_a = OpFMul %float %float_0_8 %diffuse
%lit_scale = OpFAdd %float %float_0_2 %lit_scale_a
%finalColor = OpVectorTimesScalar %v3float %base %lit_scale
%gl_out_pos = OpAccessChain %ptr_Output_v4float %gl_out %int_0
%in_pos0 = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %in_pos0
OpStore %gl_out_pos %pos0
OpStore %outColor %finalColor
OpEmitVertex
%in_pos2 = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %in_pos2
OpStore %gl_out_pos %pos2
OpStore %outColor %finalColor
OpEmitVertex
%in_pos4 = OpAccessChain %ptr_Input_v4float %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %in_pos4
OpStore %gl_out_pos %pos4
OpStore %outColor %finalColor
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
