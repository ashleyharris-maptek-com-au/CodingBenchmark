int absoluteValue(int value) {
    return value < 0 ? -value : value;
}

int colorOf(int piece) {
    return piece > 0 ? WHITE : BLACK;
}

int rankOf(int square) {
    return square / 8;
}

int fileOf(int square) {
    return square % 8;
}

int makeSquare(int file, int rank) {
    return rank * 8 + file;
}

int mirrorSquare(int square) {
    return square ^ 56;
}

int pieceIndex(int piece) {
    const int type = absoluteValue(piece);
    const int colorBase = piece > 0 ? 0 : 6;
    return colorBase + (type - 1);
}

char pieceLetter(int pieceType) {
    switch (pieceType) {
        case PAWN: return 'p';
        case KNIGHT: return 'N';
        case BISHOP: return 'B';
        case ROOK: return 'R';
        case QUEEN: return 'Q';
        case KING: return 'K';
        default: return '?';
    }
}

std::string trim(const std::string& input) {
    std::size_t begin = 0;
    while (begin < input.size() && std::isspace(static_cast<unsigned char>(input[begin])) != 0) {
        ++begin;
    }
    std::size_t end = input.size();
    while (end > begin && std::isspace(static_cast<unsigned char>(input[end - 1])) != 0) {
        --end;
    }
    return input.substr(begin, end - begin);
}

std::string squareToString(int square) {
    std::string result(2, ' ');
    result[0] = static_cast<char>('a' + fileOf(square));
    result[1] = static_cast<char>('1' + rankOf(square));
    return result;
}

uint32_t packMove(const Move& move) {
    return static_cast<uint32_t>(move.from)
        | (static_cast<uint32_t>(move.to) << 6)
        | (static_cast<uint32_t>(move.promo) << 12)
        | (static_cast<uint32_t>(move.flags) << 16);
}

Move unpackMove(uint32_t packed) {
    Move move;
    move.from = static_cast<int>(packed & 63U);
    move.to = static_cast<int>((packed >> 6) & 63U);
    move.promo = static_cast<int>((packed >> 12) & 15U);
    move.flags = static_cast<int>((packed >> 16) & 255U);
    return move;
}

bool sameMove(const Move& lhs, const Move& rhs) {
    return lhs.from == rhs.from && lhs.to == rhs.to && lhs.promo == rhs.promo && lhs.flags == rhs.flags;
}

uint64_t computeHash(const Board& board) {
    uint64_t hash = 0;
    for (int square = 0; square < 64; ++square) {
        const int piece = board.squares[square];
        if (piece != EMPTY) {
            hash ^= ZOBRIST_PIECES[pieceIndex(piece)][square];
        }
    }
    hash ^= ZOBRIST_CASTLING[board.castling];
    if (board.epSquare != -1) {
        hash ^= ZOBRIST_EP[fileOf(board.epSquare)];
    }
    if (!board.whiteToMove) {
        hash ^= ZOBRIST_SIDE;
    }
    return hash;
}

void initZobrist() {
    std::mt19937_64 rng(0xD0C0FFEE12345678ULL);
    for (auto& pieceArray : ZOBRIST_PIECES) {
        for (auto& value : pieceArray) {
            value = rng();
        }
    }
    for (auto& value : ZOBRIST_CASTLING) {
        value = rng();
    }
    for (auto& value : ZOBRIST_EP) {
        value = rng();
    }
    ZOBRIST_SIDE = rng();
}

bool parseFen(const std::string& fen, Board& board) {
    board = Board{};
    std::vector<std::string> parts;
    std::string current;
    for (char ch : fen) {
        if (std::isspace(static_cast<unsigned char>(ch)) != 0) {
            if (!current.empty()) {
                parts.push_back(current);
                current.clear();
            }
        } else {
            current.push_back(ch);
        }
    }
    if (!current.empty()) {
        parts.push_back(current);
    }
    if (parts.size() < 4) {
        return false;
    }

    int square = 56;
    for (char ch : parts[0]) {
        if (ch == '/') {
            square -= 16;
            continue;
        }
        if (std::isdigit(static_cast<unsigned char>(ch)) != 0) {
            square += ch - '0';
            continue;
        }
        int piece = 0;
        switch (std::tolower(static_cast<unsigned char>(ch))) {
            case 'p': piece = PAWN; break;
            case 'n': piece = KNIGHT; break;
            case 'b': piece = BISHOP; break;
            case 'r': piece = ROOK; break;
            case 'q': piece = QUEEN; break;
            case 'k': piece = KING; break;
            default: return false;
        }
        if (std::isupper(static_cast<unsigned char>(ch)) == 0) {
            piece = -piece;
        }
        if (square < 0 || square >= 64) {
            return false;
        }
        board.squares[square] = piece;
        if (absoluteValue(piece) == KING) {
            board.kingSquares[piece > 0 ? WHITE : BLACK] = square;
        }
        ++square;
    }

    board.whiteToMove = (parts[1] == "w");
    board.castling = 0;
    if (parts[2] != "-") {
        for (char ch : parts[2]) {
            switch (ch) {
                case 'K': board.castling |= 1; break;
                case 'Q': board.castling |= 2; break;
                case 'k': board.castling |= 4; break;
                case 'q': board.castling |= 8; break;
                default: break;
            }
        }
    }
    if (parts[3] != "-" && parts[3].size() == 2) {
        const int file = parts[3][0] - 'a';
        const int rank = parts[3][1] - '1';
        if (file >= 0 && file < 8 && rank >= 0 && rank < 8) {
            board.epSquare = makeSquare(file, rank);
        }
    }
    if (parts.size() >= 5) {
        board.halfmoveClock = std::max(0, std::atoi(parts[4].c_str()));
    }
    if (parts.size() >= 6) {
        board.fullmoveNumber = std::max(1, std::atoi(parts[5].c_str()));
    }

    board.hash = computeHash(board);
    return true;
}

bool isSquareAttacked(const Board& board, int square, int byColor) {
    const int file = fileOf(square);
    const int rank = rankOf(square);

    if (byColor == WHITE) {
        if (file > 0 && rank > 0 && board.squares[square - 9] == PAWN) {
            return true;
        }
        if (file < 7 && rank > 0 && board.squares[square - 7] == PAWN) {
            return true;
        }
    } else {
        if (file > 0 && rank < 7 && board.squares[square + 7] == -PAWN) {
            return true;
        }
        if (file < 7 && rank < 7 && board.squares[square + 9] == -PAWN) {
            return true;
        }
    }

    for (int i = 0; i < 8; ++i) {
        const int nf = file + KNIGHT_DF[i];
        const int nr = rank + KNIGHT_DR[i];
        if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) {
            continue;
        }
        const int piece = board.squares[makeSquare(nf, nr)];
        if (piece == (byColor == WHITE ? KNIGHT : -KNIGHT)) {
            return true;
        }
    }

    for (int dir = 0; dir < 4; ++dir) {
        int nf = file + BISHOP_DF[dir];
        int nr = rank + BISHOP_DR[dir];
        while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            const int piece = board.squares[makeSquare(nf, nr)];
            if (piece != EMPTY) {
                if (piece == (byColor == WHITE ? BISHOP : -BISHOP)
                    || piece == (byColor == WHITE ? QUEEN : -QUEEN)) {
                    return true;
                }
                break;
            }
            nf += BISHOP_DF[dir];
            nr += BISHOP_DR[dir];
        }
    }

    for (int dir = 0; dir < 4; ++dir) {
        int nf = file + ROOK_DF[dir];
        int nr = rank + ROOK_DR[dir];
        while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            const int piece = board.squares[makeSquare(nf, nr)];
            if (piece != EMPTY) {
                if (piece == (byColor == WHITE ? ROOK : -ROOK)
                    || piece == (byColor == WHITE ? QUEEN : -QUEEN)) {
                    return true;
                }
                break;
            }
            nf += ROOK_DF[dir];
            nr += ROOK_DR[dir];
        }
    }

    for (int i = 0; i < 8; ++i) {
        const int nf = file + KING_DF[i];
        const int nr = rank + KING_DR[i];
        if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) {
            continue;
        }
        const int piece = board.squares[makeSquare(nf, nr)];
        if (piece == (byColor == WHITE ? KING : -KING)) {
            return true;
        }
    }

    return false;
}

bool inCheck(const Board& board, int side) {
    return isSquareAttacked(board, board.kingSquares[side], side ^ 1);
}

void addPromotionMoves(MoveList& moves, int from, int to, int flags) {
    moves.add(from, to, QUEEN, flags | FLAG_PROMOTION);
    moves.add(from, to, KNIGHT, flags | FLAG_PROMOTION);
    moves.add(from, to, ROOK, flags | FLAG_PROMOTION);
    moves.add(from, to, BISHOP, flags | FLAG_PROMOTION);
}

void generatePseudoMoves(const Board& board, MoveList& moves, bool capturesOnly) {
    moves.clear();
    const int side = board.whiteToMove ? WHITE : BLACK;
    const int forward = side == WHITE ? 8 : -8;
    const int pawnStartRank = side == WHITE ? 1 : 6;
    const int promotionRank = side == WHITE ? 6 : 1;

    for (int from = 0; from < 64; ++from) {
        const int piece = board.squares[from];
        if (piece == EMPTY || colorOf(piece) != side) {
            continue;
        }

        const int type = absoluteValue(piece);
        const int file = fileOf(from);
        const int rank = rankOf(from);

        if (type == PAWN) {
            const int oneForward = from + forward;
            if (rank == promotionRank) {
                if (oneForward >= 0 && oneForward < 64 && board.squares[oneForward] == EMPTY) {
                    addPromotionMoves(moves, from, oneForward, 0);
                }
            } else if (!capturesOnly && oneForward >= 0 && oneForward < 64 && board.squares[oneForward] == EMPTY) {
                moves.add(from, oneForward, 0, 0);
                if (rank == pawnStartRank) {
                    const int twoForward = from + forward * 2;
                    if (board.squares[twoForward] == EMPTY) {
                        moves.add(from, twoForward, 0, FLAG_DOUBLE_PUSH);
                    }
                }
            }

            for (int df : {-1, 1}) {
                const int nf = file + df;
                if (nf < 0 || nf >= 8) {
                    continue;
                }
                const int to = side == WHITE ? from + 8 + df : from - 8 + df;
                if (to < 0 || to >= 64) {
                    continue;
                }
                const int target = board.squares[to];
                if (target != EMPTY && colorOf(target) != side) {
                    if (rank == promotionRank) {
                        addPromotionMoves(moves, from, to, FLAG_CAPTURE);
                    } else {
                        moves.add(from, to, 0, FLAG_CAPTURE);
                    }
                } else if (to == board.epSquare) {
                    moves.add(from, to, 0, FLAG_CAPTURE | FLAG_EN_PASSANT);
                }
            }
            continue;
        }

        if (type == KNIGHT) {
            for (int i = 0; i < 8; ++i) {
                const int nf = file + KNIGHT_DF[i];
                const int nr = rank + KNIGHT_DR[i];
                if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) {
                    continue;
                }
                const int to = makeSquare(nf, nr);
                const int target = board.squares[to];
                if (target == EMPTY) {
                    if (!capturesOnly) {
                        moves.add(from, to, 0, 0);
                    }
                } else if (colorOf(target) != side) {
                    moves.add(from, to, 0, FLAG_CAPTURE);
                }
            }
            continue;
        }

        if (type == BISHOP || type == ROOK || type == QUEEN) {
            const bool useBishopDirs = (type == BISHOP || type == QUEEN);
            const bool useRookDirs = (type == ROOK || type == QUEEN);
            if (useBishopDirs) {
                for (int dir = 0; dir < 4; ++dir) {
                    int nf = file + BISHOP_DF[dir];
                    int nr = rank + BISHOP_DR[dir];
                    while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                        const int to = makeSquare(nf, nr);
                        const int target = board.squares[to];
                        if (target == EMPTY) {
                            if (!capturesOnly) {
                                moves.add(from, to, 0, 0);
                            }
                        } else {
                            if (colorOf(target) != side) {
                                moves.add(from, to, 0, FLAG_CAPTURE);
                            }
                            break;
                        }
                        nf += BISHOP_DF[dir];
                        nr += BISHOP_DR[dir];
                    }
                }
            }
            if (useRookDirs) {
                for (int dir = 0; dir < 4; ++dir) {
                    int nf = file + ROOK_DF[dir];
                    int nr = rank + ROOK_DR[dir];
                    while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                        const int to = makeSquare(nf, nr);
                        const int target = board.squares[to];
                        if (target == EMPTY) {
                            if (!capturesOnly) {
                                moves.add(from, to, 0, 0);
                            }
                        } else {
                            if (colorOf(target) != side) {
                                moves.add(from, to, 0, FLAG_CAPTURE);
                            }
                            break;
                        }
                        nf += ROOK_DF[dir];
                        nr += ROOK_DR[dir];
                    }
                }
            }
            continue;
        }

        if (type == KING) {
            for (int i = 0; i < 8; ++i) {
                const int nf = file + KING_DF[i];
                const int nr = rank + KING_DR[i];
                if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) {
                    continue;
                }
                const int to = makeSquare(nf, nr);
                const int target = board.squares[to];
                if (target == EMPTY) {
                    if (!capturesOnly) {
                        moves.add(from, to, 0, 0);
                    }
                } else if (colorOf(target) != side) {
                    moves.add(from, to, 0, FLAG_CAPTURE);
                }
            }

            if (capturesOnly || inCheck(board, side)) {
                continue;
            }

            if (side == WHITE) {
                if ((board.castling & 1) != 0
                    && board.squares[5] == EMPTY
                    && board.squares[6] == EMPTY
                    && board.squares[7] == ROOK
                    && !isSquareAttacked(board, 5, BLACK)
                    && !isSquareAttacked(board, 6, BLACK)) {
                    moves.add(4, 6, 0, FLAG_CASTLE);
                }
                if ((board.castling & 2) != 0
                    && board.squares[1] == EMPTY
                    && board.squares[2] == EMPTY
                    && board.squares[3] == EMPTY
                    && board.squares[0] == ROOK
                    && !isSquareAttacked(board, 3, BLACK)
                    && !isSquareAttacked(board, 2, BLACK)) {
                    moves.add(4, 2, 0, FLAG_CASTLE);
                }
            } else {
                if ((board.castling & 4) != 0
                    && board.squares[61] == EMPTY
                    && board.squares[62] == EMPTY
                    && board.squares[63] == -ROOK
                    && !isSquareAttacked(board, 61, WHITE)
                    && !isSquareAttacked(board, 62, WHITE)) {
                    moves.add(60, 62, 0, FLAG_CASTLE);
                }
                if ((board.castling & 8) != 0
                    && board.squares[57] == EMPTY
                    && board.squares[58] == EMPTY
                    && board.squares[59] == EMPTY
                    && board.squares[56] == -ROOK
                    && !isSquareAttacked(board, 59, WHITE)
                    && !isSquareAttacked(board, 58, WHITE)) {
                    moves.add(60, 58, 0, FLAG_CASTLE);
                }
            }
        }
    }
}

bool makeMove(Board& board, const Move& move, Undo& undo) {
    const int side = board.whiteToMove ? WHITE : BLACK;
    const int movingPiece = board.squares[move.from];
    const int movingType = absoluteValue(movingPiece);

    undo.movedPiece = movingPiece;
    undo.previousEpSquare = board.epSquare;
    undo.previousCastling = board.castling;
    undo.previousHalfmove = board.halfmoveClock;
    undo.previousFullmove = board.fullmoveNumber;
    undo.previousHash = board.hash;
    undo.previousKingSquares = board.kingSquares;

    int capturedPiece = EMPTY;
    if ((move.flags & FLAG_EN_PASSANT) != 0) {
        const int captureSquare = side == WHITE ? move.to - 8 : move.to + 8;
        capturedPiece = board.squares[captureSquare];
    } else {
        capturedPiece = board.squares[move.to];
    }
    undo.capturedPiece = capturedPiece;

    if (board.epSquare != -1) {
        board.hash ^= ZOBRIST_EP[fileOf(board.epSquare)];
    }
    board.hash ^= ZOBRIST_CASTLING[board.castling];
    board.hash ^= ZOBRIST_PIECES[pieceIndex(movingPiece)][move.from];

    board.squares[move.from] = EMPTY;
    board.epSquare = -1;

    if ((move.flags & FLAG_EN_PASSANT) != 0) {
        const int captureSquare = side == WHITE ? move.to - 8 : move.to + 8;
        board.hash ^= ZOBRIST_PIECES[pieceIndex(capturedPiece)][captureSquare];
        board.squares[captureSquare] = EMPTY;
    } else if (capturedPiece != EMPTY) {
        board.hash ^= ZOBRIST_PIECES[pieceIndex(capturedPiece)][move.to];
    }

    int placedPiece = movingPiece;
    if ((move.flags & FLAG_PROMOTION) != 0) {
        placedPiece = side == WHITE ? move.promo : -move.promo;
    }

    if ((move.flags & FLAG_CASTLE) != 0) {
        if (move.to == 6) {
            board.squares[7] = EMPTY;
            board.squares[5] = ROOK;
            board.hash ^= ZOBRIST_PIECES[pieceIndex(ROOK)][7];
            board.hash ^= ZOBRIST_PIECES[pieceIndex(ROOK)][5];
        } else if (move.to == 2) {
            board.squares[0] = EMPTY;
            board.squares[3] = ROOK;
            board.hash ^= ZOBRIST_PIECES[pieceIndex(ROOK)][0];
            board.hash ^= ZOBRIST_PIECES[pieceIndex(ROOK)][3];
        } else if (move.to == 62) {
            board.squares[63] = EMPTY;
            board.squares[61] = -ROOK;
            board.hash ^= ZOBRIST_PIECES[pieceIndex(-ROOK)][63];
            board.hash ^= ZOBRIST_PIECES[pieceIndex(-ROOK)][61];
        } else if (move.to == 58) {
            board.squares[56] = EMPTY;
            board.squares[59] = -ROOK;
            board.hash ^= ZOBRIST_PIECES[pieceIndex(-ROOK)][56];
            board.hash ^= ZOBRIST_PIECES[pieceIndex(-ROOK)][59];
        }
    }

    board.squares[move.to] = placedPiece;
    board.hash ^= ZOBRIST_PIECES[pieceIndex(placedPiece)][move.to];

    if (movingType == KING) {
        board.kingSquares[side] = move.to;
        if (side == WHITE) {
            board.castling &= ~3;
        } else {
            board.castling &= ~12;
        }
    } else if (movingType == ROOK) {
        if (move.from == 0) board.castling &= ~2;
        if (move.from == 7) board.castling &= ~1;
        if (move.from == 56) board.castling &= ~8;
        if (move.from == 63) board.castling &= ~4;
    }

    if (capturedPiece != EMPTY && absoluteValue(capturedPiece) == ROOK) {
        if (move.to == 0) board.castling &= ~2;
        if (move.to == 7) board.castling &= ~1;
        if (move.to == 56) board.castling &= ~8;
        if (move.to == 63) board.castling &= ~4;
    }

    if ((move.flags & FLAG_DOUBLE_PUSH) != 0) {
        board.epSquare = side == WHITE ? move.from + 8 : move.from - 8;
    }

    board.hash ^= ZOBRIST_CASTLING[board.castling];
    if (board.epSquare != -1) {
        board.hash ^= ZOBRIST_EP[fileOf(board.epSquare)];
    }

    board.halfmoveClock = (movingType == PAWN || capturedPiece != EMPTY) ? 0 : board.halfmoveClock + 1;
    if (side == BLACK) {
        ++board.fullmoveNumber;
    }

    board.whiteToMove = !board.whiteToMove;
    board.hash ^= ZOBRIST_SIDE;
    return true;
}

void undoMove(Board& board, const Move& move, const Undo& undo) {
    board.whiteToMove = !board.whiteToMove;
    board.castling = undo.previousCastling;
    board.epSquare = undo.previousEpSquare;
    board.halfmoveClock = undo.previousHalfmove;
    board.fullmoveNumber = undo.previousFullmove;
    board.kingSquares = undo.previousKingSquares;
    board.hash = undo.previousHash;

    board.squares[move.from] = undo.movedPiece;

    if ((move.flags & FLAG_CASTLE) != 0) {
        if (move.to == 6) {
            board.squares[7] = ROOK;
            board.squares[5] = EMPTY;
        } else if (move.to == 2) {
            board.squares[0] = ROOK;
            board.squares[3] = EMPTY;
        } else if (move.to == 62) {
            board.squares[63] = -ROOK;
            board.squares[61] = EMPTY;
        } else if (move.to == 58) {
            board.squares[56] = -ROOK;
            board.squares[59] = EMPTY;
        }
    }

    if ((move.flags & FLAG_EN_PASSANT) != 0) {
        board.squares[move.to] = EMPTY;
        const int captureSquare = board.whiteToMove ? move.to - 8 : move.to + 8;
        board.squares[captureSquare] = undo.capturedPiece;
    } else {
        board.squares[move.to] = undo.capturedPiece;
    }
}

void generateLegalMoves(Board& board, MoveList& legalMoves, bool capturesOnly) {
    MoveList pseudoMoves;
    generatePseudoMoves(board, pseudoMoves, capturesOnly);
    legalMoves.clear();
    const int side = board.whiteToMove ? WHITE : BLACK;
    for (int i = 0; i < pseudoMoves.size; ++i) {
        const Move move = pseudoMoves.moves[i];
        Undo undo;
        makeMove(board, move, undo);
        if (!inCheck(board, side)) {
            legalMoves.moves[legalMoves.size++] = move;
        }
        undoMove(board, move, undo);
    }
}

int countMobility(const Board& board, int square, int type, int side) {
    const int file = fileOf(square);
    const int rank = rankOf(square);
    int mobility = 0;

    if (type == KNIGHT) {
        for (int i = 0; i < 8; ++i) {
            const int nf = file + KNIGHT_DF[i];
            const int nr = rank + KNIGHT_DR[i];
            if (nf < 0 || nf >= 8 || nr < 0 || nr >= 8) {
                continue;
            }
            const int target = board.squares[makeSquare(nf, nr)];
            if (target == EMPTY || colorOf(target) != side) {
                ++mobility;
            }
        }
        return mobility;
    }

    const std::array<int, 4>* dfArray = nullptr;
    const std::array<int, 4>* drArray = nullptr;
    int loops = 0;
    if (type == BISHOP) {
        dfArray = &BISHOP_DF;
        drArray = &BISHOP_DR;
        loops = 4;
    } else if (type == ROOK) {
        dfArray = &ROOK_DF;
        drArray = &ROOK_DR;
        loops = 4;
    } else if (type == QUEEN) {
        for (int dir = 0; dir < 4; ++dir) {
            int nf = file + BISHOP_DF[dir];
            int nr = rank + BISHOP_DR[dir];
            while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                const int target = board.squares[makeSquare(nf, nr)];
                if (target == EMPTY) {
                    ++mobility;
                } else {
                    if (colorOf(target) != side) {
                        ++mobility;
                    }
                    break;
                }
                nf += BISHOP_DF[dir];
                nr += BISHOP_DR[dir];
            }
        }
        dfArray = &ROOK_DF;
        drArray = &ROOK_DR;
        loops = 4;
    } else {
        return 0;
    }

    for (int dir = 0; dir < loops; ++dir) {
        int nf = file + (*dfArray)[dir];
        int nr = rank + (*drArray)[dir];
        while (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            const int target = board.squares[makeSquare(nf, nr)];
            if (target == EMPTY) {
                ++mobility;
            } else {
                if (colorOf(target) != side) {
                    ++mobility;
                }
                break;
            }
            nf += (*dfArray)[dir];
            nr += (*drArray)[dir];
        }
    }
    return mobility;
}

bool insufficientMaterial(const Board& board) {
    int whiteMinor = 0;
    int blackMinor = 0;
    int bishops = 0;
    for (int piece : board.squares) {
        const int type = absoluteValue(piece);
        if (type == EMPTY || type == KING) {
            continue;
        }
        if (type == PAWN || type == ROOK || type == QUEEN) {
            return false;
        }
        if (piece > 0) {
            ++whiteMinor;
        } else {
            ++blackMinor;
        }
        if (type == BISHOP) {
            ++bishops;
        }
    }

    if (whiteMinor + blackMinor == 0) {
        return true;
    }
    if (whiteMinor + blackMinor == 1) {
        return true;
    }
    if (whiteMinor + blackMinor == 2 && bishops == 0) {
        return true;
    }
    return false;
}

int evaluate(const Board& board) {
    int whiteScore = 0;
    int blackScore = 0;
    std::array<int, 8> whitePawnFiles{};
    std::array<int, 8> blackPawnFiles{};
    int whiteBishops = 0;
    int blackBishops = 0;
    int nonPawnMaterial = 0;

    for (int square = 0; square < 64; ++square) {
        const int piece = board.squares[square];
        if (piece == EMPTY) {
            continue;
        }
        const int type = absoluteValue(piece);
        const int pstSquare = piece > 0 ? square : mirrorSquare(square);
        int pstValue = 0;
        switch (type) {
            case PAWN: pstValue = PAWN_PST[pstSquare]; break;
            case KNIGHT: pstValue = KNIGHT_PST[pstSquare]; break;
            case BISHOP: pstValue = BISHOP_PST[pstSquare]; break;
            case ROOK: pstValue = ROOK_PST[pstSquare]; break;
            case QUEEN: pstValue = QUEEN_PST[pstSquare]; break;
            default: break;
        }

        const int side = piece > 0 ? WHITE : BLACK;
        const int material = MATERIAL[type];
        if (type != PAWN && type != KING) {
            nonPawnMaterial += material;
        }

        int contribution = material + pstValue;
        if (type == KNIGHT) {
            contribution += 4 * countMobility(board, square, type, side);
        } else if (type == BISHOP) {
            contribution += 5 * countMobility(board, square, type, side);
        } else if (type == ROOK) {
            contribution += 2 * countMobility(board, square, type, side);
        } else if (type == QUEEN) {
            contribution += countMobility(board, square, type, side);
        }

        if (piece > 0) {
            whiteScore += contribution;
            if (type == PAWN) {
                ++whitePawnFiles[fileOf(square)];
            } else if (type == BISHOP) {
                ++whiteBishops;
            }
        } else {
            blackScore += contribution;
            if (type == PAWN) {
                ++blackPawnFiles[fileOf(square)];
            } else if (type == BISHOP) {
                ++blackBishops;
            }
        }
    }

    const bool endgame = nonPawnMaterial <= 2400;
    whiteScore += (endgame ? KING_END_PST[board.kingSquares[WHITE]] : KING_MID_PST[board.kingSquares[WHITE]]);
    blackScore += (endgame ? KING_END_PST[mirrorSquare(board.kingSquares[BLACK])]
                           : KING_MID_PST[mirrorSquare(board.kingSquares[BLACK])]);

    if (whiteBishops >= 2) {
        whiteScore += 30;
    }
    if (blackBishops >= 2) {
        blackScore += 30;
    }

    for (int square = 0; square < 64; ++square) {
        const int piece = board.squares[square];
        if (piece == EMPTY) {
            continue;
        }
        const int type = absoluteValue(piece);
        if (type != PAWN && type != ROOK) {
            continue;
        }
        const int side = piece > 0 ? WHITE : BLACK;
        const int file = fileOf(square);
        const int rank = rankOf(square);

        if (type == PAWN) {
            const std::array<int, 8>& ownFiles = side == WHITE ? whitePawnFiles : blackPawnFiles;
            const std::array<int, 8>& enemyFiles = side == WHITE ? blackPawnFiles : whitePawnFiles;
            int& scoreRef = side == WHITE ? whiteScore : blackScore;

            if (ownFiles[file] > 1) {
                scoreRef -= 10 * (ownFiles[file] - 1);
            }
            const bool leftSupport = file > 0 && ownFiles[file - 1] > 0;
            const bool rightSupport = file < 7 && ownFiles[file + 1] > 0;
            if (!leftSupport && !rightSupport) {
                scoreRef -= 12;
            }

            bool passed = true;
            for (int df = -1; df <= 1 && passed; ++df) {
                const int nf = file + df;
                if (nf < 0 || nf >= 8) {
                    continue;
                }
                if (enemyFiles[nf] == 0) {
                    continue;
                }
                for (int r = side == WHITE ? rank + 1 : rank - 1;
                     side == WHITE ? r < 8 : r >= 0;
                     side == WHITE ? ++r : --r) {
                    const int target = board.squares[makeSquare(nf, r)];
                    if (target == (side == WHITE ? -PAWN : PAWN)) {
                        passed = false;
                        break;
                    }
                }
            }
            if (passed) {
                const int advance = side == WHITE ? rank : 7 - rank;
                scoreRef += 12 + advance * 10;
            }
        } else if (type == ROOK) {
            const std::array<int, 8>& ownFiles = side == WHITE ? whitePawnFiles : blackPawnFiles;
            const std::array<int, 8>& enemyFiles = side == WHITE ? blackPawnFiles : whitePawnFiles;
            int& scoreRef = side == WHITE ? whiteScore : blackScore;
            if (ownFiles[file] == 0) {
                scoreRef += 12;
                if (enemyFiles[file] == 0) {
                    scoreRef += 10;
                }
            }
        }
    }

    int score = whiteScore - blackScore;
    score += board.whiteToMove ? 8 : -8;
    return board.whiteToMove ? score : -score;
}

class Searcher {
public:
    Searcher()
        : tt_(TT_SIZE) {}

    Move findBestMove(Board& board, double timeLimitSeconds) {
        start_ = Clock::now();
        const double safeTime = std::max(0.05, timeLimitSeconds - 0.05);
        deadline_ = start_ + std::chrono::duration_cast<Clock::duration>(std::chrono::duration<double>(safeTime * 0.98));
        nodes_ = 0;
        repetition_.clear();
        repetition_.push_back(board.hash);

        MoveList legalMoves;
        generateLegalMoves(board, legalMoves, false);
        if (legalMoves.size == 0) {
            return Move{};
        }
        if (legalMoves.size == 1) {
            return legalMoves.moves[0];
        }

        Move bestMove = legalMoves.moves[0];
        int bestScore = -INF;
        for (int depth = 1; depth <= 64; ++depth) {
            try {
                MoveList rootMoves = legalMoves;
                sortRootMoves(board, rootMoves, bestMove);

                Move localBest = bestMove;
                int localBestScore = -INF;
                int alpha = -INF;
                int beta = INF;

                for (int i = 0; i < rootMoves.size; ++i) {
                    checkTime();
                    Move move = rootMoves.moves[i];
                    Undo undo;
                    makeMove(board, move, undo);
                    repetition_.push_back(board.hash);

                    int score;
                    try {
                        if (i == 0) {
                            score = -search(board, depth - 1, -beta, -alpha, 1);
                        } else {
                            score = -search(board, depth - 1, -alpha - 1, -alpha, 1);
                            if (score > alpha && score < beta) {
                                score = -search(board, depth - 1, -beta, -alpha, 1);
                            }
                        }
                    } catch (const SearchStop&) {
                        repetition_.pop_back();
                        undoMove(board, move, undo);
                        throw;
                    }

                    repetition_.pop_back();
                    undoMove(board, move, undo);

                    if (score > localBestScore) {
                        localBestScore = score;
                        localBest = move;
                    }
                    if (score > alpha) {
                        alpha = score;
                    }
                }

                bestMove = localBest;
                bestScore = localBestScore;
                if (bestScore > MATE_SCORE - 2000 || bestScore < -MATE_SCORE + 2000) {
                    break;
                }
            } catch (const SearchStop&) {
                break;
            }
        }

        return bestMove;
    }

private:
    std::vector<TTEntry> tt_;
    Clock::time_point start_{};
    Clock::time_point deadline_{};
    uint64_t nodes_ = 0;
    std::array<std::array<std::array<int, 64>, 64>, 2> history_{};
    std::array<std::array<uint32_t, 2>, MAX_PLY> killers_{};
    std::vector<uint64_t> repetition_;

    void checkTime() {
        ++nodes_;
        if ((nodes_ & 2047ULL) == 0 && Clock::now() >= deadline_) {
            throw SearchStop{};
        }
    }

    bool isRepetitionOrDraw(const Board& board) const {
        if (board.halfmoveClock >= 100) {
            return true;
        }
        if (insufficientMaterial(board)) {
            return true;
        }
        const int limit = static_cast<int>(repetition_.size()) - board.halfmoveClock - 1;
        for (int i = static_cast<int>(repetition_.size()) - 3; i >= 0 && i >= limit; i -= 2) {
            if (repetition_[i] == board.hash) {
                return true;
            }
        }
        return false;
    }

    int scoreToTT(int score, int ply) const {
        if (score > MATE_SCORE - 1000) {
            return score + ply;
        }
        if (score < -MATE_SCORE + 1000) {
            return score - ply;
        }
        return score;
    }

    int scoreFromTT(int score, int ply) const {
        if (score > MATE_SCORE - 1000) {
            return score - ply;
        }
        if (score < -MATE_SCORE + 1000) {
            return score + ply;
        }
        return score;
    }

    TTEntry* probe(uint64_t key) {
        TTEntry& entry = tt_[key & (TT_SIZE - 1)];
        return entry.key == key ? &entry : nullptr;
    }

    void store(uint64_t key, int depth, int score, int flag, uint32_t packedMove, int ply) {
        TTEntry& entry = tt_[key & (TT_SIZE - 1)];
        if (entry.key != key || depth >= entry.depth || flag == TT_EXACT) {
            entry.key = key;
            entry.depth = depth;
            entry.score = scoreToTT(score, ply);
            entry.flag = static_cast<uint8_t>(flag);
            entry.packedMove = packedMove;
        }
    }

    int captureType(const Board& board, const Move& move) const {
        if ((move.flags & FLAG_EN_PASSANT) != 0) {
            return PAWN;
        }
        const int piece = board.squares[move.to];
        return piece == EMPTY ? 0 : absoluteValue(piece);
    }

    bool isQuietMove(const Move& move) const {
        return (move.flags & (FLAG_CAPTURE | FLAG_EN_PASSANT | FLAG_PROMOTION)) == 0;
    }

    void scoreMoves(Board& board, MoveList& moves, int ply, uint32_t ttMove) {
        const int side = board.whiteToMove ? WHITE : BLACK;
        for (int i = 0; i < moves.size; ++i) {
            Move& move = moves.moves[i];
            move.score = 0;
            const uint32_t packed = packMove(move);
            if (packed == ttMove) {
                move.score = 10'000'000;
                continue;
            }
            if ((move.flags & (FLAG_CAPTURE | FLAG_EN_PASSANT)) != 0) {
                const int victim = captureType(board, move);
                const int attacker = absoluteValue(board.squares[move.from]);
                move.score = 1'000'000 + victim * 16 - attacker;
            } else if ((move.flags & FLAG_PROMOTION) != 0) {
                move.score = 900'000 + move.promo * 10;
            } else if (ply < MAX_PLY && killers_[ply][0] == packed) {
                move.score = 800'000;
            } else if (ply < MAX_PLY && killers_[ply][1] == packed) {
                move.score = 790'000;
            } else {
                move.score = history_[side][move.from][move.to];
            }
        }
        std::sort(moves.moves.begin(), moves.moves.begin() + moves.size, [](const Move& lhs, const Move& rhs) {
            return lhs.score > rhs.score;
        });
    }

    void sortRootMoves(Board& board, MoveList& moves, const Move& previousBest) {
        const uint32_t rootBest = packMove(previousBest);
        for (int i = 0; i < moves.size; ++i) {
            Move& move = moves.moves[i];
            move.score = 0;
            if (packMove(move) == rootBest) {
                move.score = 10'000'000;
            } else if ((move.flags & (FLAG_CAPTURE | FLAG_EN_PASSANT)) != 0) {
                move.score = 1'000'000 + captureType(board, move) * 16 - absoluteValue(board.squares[move.from]);
            } else if ((move.flags & FLAG_PROMOTION) != 0) {
                move.score = 900'000 + move.promo * 10;
            } else {
                const int side = board.whiteToMove ? WHITE : BLACK;
                move.score = history_[side][move.from][move.to];
            }
        }
        std::sort(moves.moves.begin(), moves.moves.begin() + moves.size, [](const Move& lhs, const Move& rhs) {
            return lhs.score > rhs.score;
        });
    }

    int quiescence(Board& board, int alpha, int beta, int ply) {
        checkTime();
        if (isRepetitionOrDraw(board)) {
            return 0;
        }

        const int side = board.whiteToMove ? WHITE : BLACK;
        const bool check = inCheck(board, side);

        int standPat = evaluate(board);
        if (!check) {
            if (standPat >= beta) {
                return beta;
            }
            if (standPat > alpha) {
                alpha = standPat;
            }
        }

        MoveList moves;
        generateLegalMoves(board, moves, !check);
        if (moves.size == 0) {
            return check ? -MATE_SCORE + ply : standPat;
        }

        scoreMoves(board, moves, ply, 0);
        for (int i = 0; i < moves.size; ++i) {
            const Move move = moves.moves[i];
            Undo undo;
            makeMove(board, move, undo);
            repetition_.push_back(board.hash);
            int score;
            try {
                score = -quiescence(board, -beta, -alpha, ply + 1);
            } catch (const SearchStop&) {
                repetition_.pop_back();
                undoMove(board, move, undo);
                throw;
            }
            repetition_.pop_back();
            undoMove(board, move, undo);

            if (score >= beta) {
                return beta;
            }
            if (score > alpha) {
                alpha = score;
            }
        }
        return alpha;
    }

    int search(Board& board, int depth, int alpha, int beta, int ply) {
        checkTime();
        if (ply >= MAX_PLY - 1) {
            return evaluate(board);
        }
        if (isRepetitionOrDraw(board)) {
            return 0;
        }

        const int originalAlpha = alpha;
        const int side = board.whiteToMove ? WHITE : BLACK;
        const bool check = inCheck(board, side);

        if (check) {
            ++depth;
        }
        if (depth <= 0) {
            return quiescence(board, alpha, beta, ply);
        }

        TTEntry* entry = probe(board.hash);
        uint32_t ttMove = 0;
        if (entry != nullptr) {
            ttMove = entry->packedMove;
            if (entry->depth >= depth) {
                const int ttScore = scoreFromTT(entry->score, ply);
                if (entry->flag == TT_EXACT) {
                    return ttScore;
                }
                if (entry->flag == TT_LOWER && ttScore >= beta) {
                    return ttScore;
                }
                if (entry->flag == TT_UPPER && ttScore <= alpha) {
                    return ttScore;
                }
            }
        }

        MoveList moves;
        generateLegalMoves(board, moves, false);
        if (moves.size == 0) {
            return check ? -MATE_SCORE + ply : 0;
        }

        scoreMoves(board, moves, ply, ttMove);

        Move bestMove;
        int bestScore = -INF;
        int legalCount = 0;

        for (int i = 0; i < moves.size; ++i) {
            const Move move = moves.moves[i];
            Undo undo;
            makeMove(board, move, undo);
            repetition_.push_back(board.hash);

            int score;
            const bool quiet = isQuietMove(move);
            const bool reduce = !check && quiet && depth >= 3 && legalCount >= 4;

            try {
                if (legalCount == 0) {
                    score = -search(board, depth - 1, -beta, -alpha, ply + 1);
                } else if (reduce) {
                    score = -search(board, depth - 2, -alpha - 1, -alpha, ply + 1);
                    if (score > alpha) {
                        score = -search(board, depth - 1, -alpha - 1, -alpha, ply + 1);
                        if (score > alpha && score < beta) {
                            score = -search(board, depth - 1, -beta, -alpha, ply + 1);
                        }
                    }
                } else {
                    score = -search(board, depth - 1, -alpha - 1, -alpha, ply + 1);
                    if (score > alpha && score < beta) {
                        score = -search(board, depth - 1, -beta, -alpha, ply + 1);
                    }
                }
            } catch (const SearchStop&) {
                repetition_.pop_back();
                undoMove(board, move, undo);
                throw;
            }

            repetition_.pop_back();
            undoMove(board, move, undo);
            ++legalCount;

            if (score > bestScore) {
                bestScore = score;
                bestMove = move;
            }
            if (score > alpha) {
                alpha = score;
            }
            if (alpha >= beta) {
                if (quiet && ply < MAX_PLY) {
                    const uint32_t packed = packMove(move);
                    if (killers_[ply][0] != packed) {
                        killers_[ply][1] = killers_[ply][0];
                        killers_[ply][0] = packed;
                    }
                    history_[side][move.from][move.to] += depth * depth;
                }
                store(board.hash, depth, alpha, TT_LOWER, packMove(move), ply);
                return alpha;
            }
        }

        int flag = TT_EXACT;
        if (bestScore <= originalAlpha) {
            flag = TT_UPPER;
        } else if (bestScore >= beta) {
            flag = TT_LOWER;
        }
        store(board.hash, depth, bestScore, flag, packMove(bestMove), ply);
        return bestScore;
    }
};

std::string moveToNotation(Board& board, const Move& move) {
    if ((move.flags & FLAG_CASTLE) != 0) {
        std::string notation = (fileOf(move.to) == 6) ? "O-O" : "O-O-O";
        Undo undo;
        makeMove(board, move, undo);
        const int sideToMove = board.whiteToMove ? WHITE : BLACK;
        const bool check = inCheck(board, sideToMove);
        MoveList replies;
        generateLegalMoves(board, replies, false);
        if (check) {
            notation.push_back(replies.size == 0 ? '#' : '+');
        }
        undoMove(board, move, undo);
        return notation;
    }

    const int piece = board.squares[move.from];
    std::string notation;
    notation.push_back(pieceLetter(absoluteValue(piece)));
    notation += squareToString(move.from);
    if ((move.flags & (FLAG_CAPTURE | FLAG_EN_PASSANT)) != 0) {
        notation.push_back('x');
    }
    notation += squareToString(move.to);
    if ((move.flags & FLAG_EN_PASSANT) != 0) {
        notation += "ep";
    }
    if ((move.flags & FLAG_PROMOTION) != 0) {
        notation.push_back('/');
        notation.push_back(pieceLetter(move.promo));
    }

    Undo undo;
    makeMove(board, move, undo);
    const int sideToMove = board.whiteToMove ? WHITE : BLACK;
    const bool check = inCheck(board, sideToMove);
    MoveList replies;
    generateLegalMoves(board, replies, false);
    if (check) {
        notation.push_back(replies.size == 0 ? '#' : '+');
    }
    undoMove(board, move, undo);

    return notation;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    initZobrist();

    std::string timeoutLine;
    std::string fenLine;
    if (!std::getline(std::cin, timeoutLine)) {
        return 0;
    }
    if (!std::getline(std::cin, fenLine)) {
        return 0;
    }

    timeoutLine = trim(timeoutLine);
    fenLine = trim(fenLine);
    if (timeoutLine.empty() || fenLine.empty()) {
        return 0;
    }

    const double timeoutSeconds = std::max(0.01, std::atof(timeoutLine.c_str()));

    Board board;
    if (!parseFen(fenLine, board)) {
        std::cout << "0000\n";
        return 0;
    }

    Searcher searcher;
    Move bestMove = searcher.findBestMove(board, timeoutSeconds);
    MoveList legalMoves;
    generateLegalMoves(board, legalMoves, false);
    if (legalMoves.size == 0) {
        std::cout << "0000\n";
        return 0;
    }
    bool found = false;
    for (int i = 0; i < legalMoves.size; ++i) {
        if (sameMove(bestMove, legalMoves.moves[i])) {
            found = true;
            break;
        }
    }
    if (!found) {
        bestMove = legalMoves.moves[0];
    }

    std::cout << moveToNotation(board, bestMove) << '\n';
    return 0;
}
