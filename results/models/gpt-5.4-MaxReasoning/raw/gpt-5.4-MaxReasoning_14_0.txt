using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private const byte UnknownCell = 9;
    private const byte FlaggedCell = 10;

    private const byte StateUnknown = 0;
    private const byte StateSafe = 1;
    private const byte StateMine = 2;

    private const int GraphClueLimit = 300_000;
    private const int GraphVarLimit = 600_000;
    private const int GraphEdgeLimit = 2_400_000;

    private const int ExactVarLimit = 22;
    private const int ExactConstraintLimit = 64;
    private const int ExactNodeBudget = 2_000_000;

    private const int SubsetVarLimit = 160;
    private const int SubsetConstraintLimit = 256;
    private const int AdvancedRounds = 6;

    public static void Main()
    {
        var input = new FastInput();
        int height = input.ReadInt();
        if (height == int.MinValue)
        {
            return;
        }

        int width = input.ReadInt();
        int cellCount = checked(height * width);

        var board = new byte[cellCount];
        input.ReadBoard(board, height, width);

        var state = new byte[cellCount];
        var relevantClues = BuildRelevantClues(board, height, width);

        if (relevantClues.Count > 0)
        {
            if (relevantClues.Count <= GraphClueLimit &&
                TryBuildConstraintModel(board, height, width, relevantClues, out var model))
            {
                SolveWithConstraintModel(model, state);
            }
            else
            {
                SolveLargeBoard(board, state, relevantClues, height, width);
            }
        }

        WriteAnswer(board, state, height, width);
    }

    private static void SolveWithConstraintModel(ConstraintModel model, byte[] state)
    {
        model.EnqueueAllConstraints();
        model.Propagate();

        for (int round = 0; round < AdvancedRounds; round++)
        {
            bool changed = false;

            changed |= ApplySubsetReasoning(model);
            model.Propagate();

            changed |= ApplyExactReasoning(model);
            model.Propagate();

            if (!changed)
            {
                break;
            }
        }

        model.CopyAssignmentsToState(state);
    }

    private static bool ApplySubsetReasoning(ConstraintModel model)
    {
        bool changed = false;
        var components = BuildActiveComponents(model, SubsetVarLimit, SubsetConstraintLimit);

        foreach (var component in components)
        {
            var views = BuildConstraintViews(model, component.ConstraintIds);
            int viewCount = views.Count;

            for (int i = 0; i < viewCount; i++)
            {
                var a = views[i];
                for (int j = i + 1; j < viewCount; j++)
                {
                    var b = views[j];

                    if (TryApplySubset(model, a, b))
                    {
                        changed = true;
                    }

                    if (TryApplySubset(model, b, a))
                    {
                        changed = true;
                    }
                }
            }
        }

        return changed;
    }

    private static bool TryApplySubset(ConstraintModel model, ConstraintView subset, ConstraintView superset)
    {
        if (subset.Vars.Length >= superset.Vars.Length)
        {
            return false;
        }

        if (!IsSubset(subset.Vars, superset.Vars))
        {
            return false;
        }

        int diffSize = superset.Vars.Length - subset.Vars.Length;
        int diffMines = superset.Required - subset.Required;

        if (diffMines < 0 || diffMines > diffSize)
        {
            return false;
        }

        bool changed = false;

        if (diffMines == 0 || diffMines == diffSize)
        {
            int forcedState = diffMines == 0 ? 0 : 1;
            int i = 0;
            int j = 0;

            while (j < superset.Vars.Length)
            {
                int value = superset.Vars[j];
                if (i < subset.Vars.Length && subset.Vars[i] == value)
                {
                    i++;
                    j++;
                    continue;
                }

                if (model.AssignVar(value, forcedState))
                {
                    changed = true;
                }

                j++;
            }
        }

        return changed;
    }

    private static bool ApplyExactReasoning(ConstraintModel model)
    {
        bool changed = false;
        var components = BuildActiveComponents(model, ExactVarLimit, ExactConstraintLimit);

        foreach (var component in components)
        {
            int variableCount = component.VarIds.Length;
            if (variableCount == 0 || variableCount > ExactVarLimit)
            {
                continue;
            }

            var localMap = new Dictionary<int, int>(variableCount);
            for (int i = 0; i < variableCount; i++)
            {
                localMap[component.VarIds[i]] = i;
            }

            var localConstraints = new List<ConstraintView>(component.ConstraintIds.Length);
            foreach (int constraintId in component.ConstraintIds)
            {
                var vars = FilterActiveVars(model, constraintId);
                if (vars.Length == 0)
                {
                    continue;
                }

                var remapped = new int[vars.Length];
                for (int i = 0; i < vars.Length; i++)
                {
                    remapped[i] = localMap[vars[i]];
                }

                Array.Sort(remapped);
                localConstraints.Add(new ConstraintView(remapped, model.ConstraintRemaining[constraintId]));
            }

            if (localConstraints.Count == 0 || localConstraints.Count > ExactConstraintLimit)
            {
                continue;
            }

            var solver = new ExactComponentSolver(variableCount, localConstraints, ExactNodeBudget);
            if (!solver.TrySolve(out ulong alwaysSafeMask, out ulong alwaysMineMask))
            {
                continue;
            }

            for (int i = 0; i < variableCount; i++)
            {
                ulong bit = 1UL << i;
                if ((alwaysSafeMask & bit) != 0 && model.AssignVar(component.VarIds[i], 0))
                {
                    changed = true;
                }
                else if ((alwaysMineMask & bit) != 0 && model.AssignVar(component.VarIds[i], 1))
                {
                    changed = true;
                }
            }
        }

        return changed;
    }

    private static List<ConstraintView> BuildConstraintViews(ConstraintModel model, int[] constraintIds)
    {
        var views = new List<ConstraintView>(constraintIds.Length);
        foreach (int constraintId in constraintIds)
        {
            var vars = FilterActiveVars(model, constraintId);
            if (vars.Length == 0)
            {
                continue;
            }

            views.Add(new ConstraintView(vars, model.ConstraintRemaining[constraintId]));
        }

        return views;
    }

    private static int[] FilterActiveVars(ConstraintModel model, int constraintId)
    {
        int[] source = model.ConstraintVars[constraintId];
        int count = model.ConstraintUnknowns[constraintId];
        var result = new int[count];
        int write = 0;

        for (int i = 0; i < source.Length; i++)
        {
            int varId = source[i];
            if (model.VarAssignments[varId] == -1)
            {
                result[write++] = varId;
            }
        }

        if (write != result.Length)
        {
            Array.Resize(ref result, write);
        }

        Array.Sort(result);
        return result;
    }

    private static List<Component> BuildActiveComponents(ConstraintModel model, int maxVars, int maxConstraints)
    {
        int varCount = model.VarAssignments.Length;
        int constraintCount = model.ConstraintVars.Length;

        var activeConstraint = new bool[constraintCount];
        for (int i = 0; i < constraintCount; i++)
        {
            activeConstraint[i] = model.ConstraintUnknowns[i] > 0 &&
                                  model.ConstraintRemaining[i] > 0 &&
                                  model.ConstraintRemaining[i] < model.ConstraintUnknowns[i];
        }

        var visitedVars = new bool[varCount];
        var visitedConstraints = new bool[constraintCount];
        var components = new List<Component>();

        for (int startVar = 0; startVar < varCount; startVar++)
        {
            if (model.VarAssignments[startVar] != -1 || visitedVars[startVar])
            {
                continue;
            }

            bool touchesActiveConstraint = false;
            var varLinks = model.VarToConstraints[startVar];
            for (int i = 0; i < varLinks.Length; i++)
            {
                if (activeConstraint[varLinks[i]])
                {
                    touchesActiveConstraint = true;
                    break;
                }
            }

            if (!touchesActiveConstraint)
            {
                continue;
            }

            var queue = new Queue<int>();
            var vars = new List<int>();
            var constraints = new List<int>();

            visitedVars[startVar] = true;
            queue.Enqueue(startVar);

            while (queue.Count > 0)
            {
                int varId = queue.Dequeue();
                vars.Add(varId);

                int[] links = model.VarToConstraints[varId];
                for (int i = 0; i < links.Length; i++)
                {
                    int constraintId = links[i];
                    if (!activeConstraint[constraintId] || visitedConstraints[constraintId])
                    {
                        continue;
                    }

                    visitedConstraints[constraintId] = true;
                    constraints.Add(constraintId);

                    int[] varsInConstraint = model.ConstraintVars[constraintId];
                    for (int j = 0; j < varsInConstraint.Length; j++)
                    {
                        int nextVar = varsInConstraint[j];
                        if (model.VarAssignments[nextVar] == -1 && !visitedVars[nextVar])
                        {
                            visitedVars[nextVar] = true;
                            queue.Enqueue(nextVar);
                        }
                    }
                }
            }

            if (vars.Count <= maxVars && constraints.Count <= maxConstraints)
            {
                components.Add(new Component(vars.ToArray(), constraints.ToArray()));
            }
        }

        return components;
    }

    private static void SolveLargeBoard(byte[] board, byte[] state, List<int> relevantClues, int height, int width)
    {
        int passLimit;
        if (relevantClues.Count <= 50_000)
        {
            passLimit = int.MaxValue;
        }
        else if (relevantClues.Count <= 250_000)
        {
            passLimit = 32;
        }
        else if (relevantClues.Count <= 1_000_000)
        {
            passLimit = 8;
        }
        else
        {
            passLimit = 4;
        }

        for (int pass = 0; pass < passLimit; pass++)
        {
            bool changed = false;

            for (int i = 0; i < relevantClues.Count; i++)
            {
                int clueIndex = relevantClues[i];
                int y = clueIndex / width;
                int x = clueIndex - (y * width);
                int clue = board[clueIndex];

                int unknownCount = 0;
                int mineCount = 0;
                Span<int> unknownBuffer = stackalloc int[8];

                int y0 = y > 0 ? y - 1 : y;
                int y1 = y + 1 < height ? y + 1 : y;
                int x0 = x > 0 ? x - 1 : x;
                int x1 = x + 1 < width ? x + 1 : x;

                for (int ny = y0; ny <= y1; ny++)
                {
                    int rowStart = ny * width;
                    for (int nx = x0; nx <= x1; nx++)
                    {
                        if (nx == x && ny == y)
                        {
                            continue;
                        }

                        int neighborIndex = rowStart + nx;
                        byte cell = board[neighborIndex];

                        if (cell == FlaggedCell || state[neighborIndex] == StateMine)
                        {
                            mineCount++;
                        }
                        else if (cell == UnknownCell && state[neighborIndex] == StateUnknown)
                        {
                            unknownBuffer[unknownCount++] = neighborIndex;
                        }
                    }
                }

                int remaining = clue - mineCount;
                if (remaining < 0 || remaining > unknownCount || unknownCount == 0)
                {
                    continue;
                }

                if (remaining == 0)
                {
                    for (int j = 0; j < unknownCount; j++)
                    {
                        int idx = unknownBuffer[j];
                        if (state[idx] == StateUnknown)
                        {
                            state[idx] = StateSafe;
                            changed = true;
                        }
                    }
                }
                else if (remaining == unknownCount)
                {
                    for (int j = 0; j < unknownCount; j++)
                    {
                        int idx = unknownBuffer[j];
                        if (state[idx] == StateUnknown)
                        {
                            state[idx] = StateMine;
                            changed = true;
                        }
                    }
                }
            }

            if (!changed)
            {
                break;
            }
        }
    }

    private static bool TryBuildConstraintModel(
        byte[] board,
        int height,
        int width,
        List<int> relevantClues,
        out ConstraintModel model)
    {
        var varMap = new Dictionary<int, int>(Math.Min(GraphVarLimit, relevantClues.Count * 2));
        var varCells = new List<int>();
        var constraints = new List<int[]>();
        var required = new List<int>();
        int edgeCount = 0;

        for (int i = 0; i < relevantClues.Count; i++)
        {
            int clueIndex = relevantClues[i];
            int y = clueIndex / width;
            int x = clueIndex - (y * width);
            int clue = board[clueIndex];

            Span<int> buffer = stackalloc int[8];
            int varCount = 0;
            int flaggedCount = 0;

            int y0 = y > 0 ? y - 1 : y;
            int y1 = y + 1 < height ? y + 1 : y;
            int x0 = x > 0 ? x - 1 : x;
            int x1 = x + 1 < width ? x + 1 : x;

            for (int ny = y0; ny <= y1; ny++)
            {
                int rowStart = ny * width;
                for (int nx = x0; nx <= x1; nx++)
                {
                    if (nx == x && ny == y)
                    {
                        continue;
                    }

                    int neighborIndex = rowStart + nx;
                    byte cell = board[neighborIndex];
                    if (cell == FlaggedCell)
                    {
                        flaggedCount++;
                    }
                    else if (cell == UnknownCell)
                    {
                        if (!varMap.TryGetValue(neighborIndex, out int varId))
                        {
                            varId = varCells.Count;
                            if (varId >= GraphVarLimit)
                            {
                                model = null!;
                                return false;
                            }

                            varMap.Add(neighborIndex, varId);
                            varCells.Add(neighborIndex);
                        }

                        buffer[varCount++] = varId;
                    }
                }
            }

            int remaining = clue - flaggedCount;
            if (varCount == 0 || remaining < 0 || remaining > varCount)
            {
                continue;
            }

            var vars = new int[varCount];
            for (int j = 0; j < varCount; j++)
            {
                vars[j] = buffer[j];
            }

            Array.Sort(vars);
            constraints.Add(vars);
            required.Add(remaining);
            edgeCount += varCount;

            if (edgeCount > GraphEdgeLimit)
            {
                model = null!;
                return false;
            }
        }

        int variableCount = varCells.Count;
        int constraintCount = constraints.Count;

        var adjacencyBuilder = new List<int>[variableCount];
        for (int i = 0; i < variableCount; i++)
        {
            adjacencyBuilder[i] = new List<int>(4);
        }

        for (int constraintId = 0; constraintId < constraintCount; constraintId++)
        {
            int[] vars = constraints[constraintId];
            for (int i = 0; i < vars.Length; i++)
            {
                adjacencyBuilder[vars[i]].Add(constraintId);
            }
        }

        var varToConstraints = new int[variableCount][];
        for (int i = 0; i < variableCount; i++)
        {
            varToConstraints[i] = adjacencyBuilder[i].ToArray();
        }

        model = new ConstraintModel(
            board,
            varCells.ToArray(),
            constraints.ToArray(),
            required.ToArray(),
            varToConstraints);
        return true;
    }

    private static List<int> BuildRelevantClues(byte[] board, int height, int width)
    {
        var clues = new List<int>();
        for (int y = 0; y < height; y++)
        {
            int rowStart = y * width;
            for (int x = 0; x < width; x++)
            {
                int idx = rowStart + x;
                byte cell = board[idx];
                if (cell > 8)
                {
                    continue;
                }

                int y0 = y > 0 ? y - 1 : y;
                int y1 = y + 1 < height ? y + 1 : y;
                int x0 = x > 0 ? x - 1 : x;
                int x1 = x + 1 < width ? x + 1 : x;

                bool touchesUnknown = false;
                for (int ny = y0; ny <= y1 && !touchesUnknown; ny++)
                {
                    int neighborRow = ny * width;
                    for (int nx = x0; nx <= x1; nx++)
                    {
                        if (nx == x && ny == y)
                        {
                            continue;
                        }

                        if (board[neighborRow + nx] == UnknownCell)
                        {
                            touchesUnknown = true;
                            break;
                        }
                    }
                }

                if (touchesUnknown)
                {
                    clues.Add(idx);
                }
            }
        }

        return clues;
    }

    private static void WriteAnswer(byte[] board, byte[] state, int height, int width)
    {
        int safeCount = 0;
        for (int i = 0; i < board.Length; i++)
        {
            if (board[i] == UnknownCell && state[i] == StateSafe)
            {
                safeCount++;
            }
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput())
        {
            AutoFlush = false
        };

        writer.WriteLine(safeCount);
        if (safeCount > 0)
        {
            for (int y = 0; y < height; y++)
            {
                int rowStart = y * width;
                for (int x = 0; x < width; x++)
                {
                    int idx = rowStart + x;
                    if (board[idx] == UnknownCell && state[idx] == StateSafe)
                    {
                        writer.Write(x);
                        writer.Write(' ');
                        writer.WriteLine(y);
                    }
                }
            }
        }

        writer.Flush();
    }

    private static bool IsSubset(int[] a, int[] b)
    {
        int i = 0;
        int j = 0;
        while (i < a.Length && j < b.Length)
        {
            if (a[i] == b[j])
            {
                i++;
                j++;
            }
            else if (a[i] > b[j])
            {
                j++;
            }
            else
            {
                return false;
            }
        }

        return i == a.Length;
    }

    private sealed class ConstraintModel
    {
        public byte[] Board { get; }
        public int[] VarCellIndices { get; }
        public int[][] ConstraintVars { get; }
        public int[] ConstraintRemaining { get; }
        public int[] ConstraintUnknowns { get; }
        public int[][] VarToConstraints { get; }
        public sbyte[] VarAssignments { get; }

        private readonly Queue<int> _queue;
        private readonly bool[] _inQueue;

        public ConstraintModel(
            byte[] board,
            int[] varCellIndices,
            int[][] constraintVars,
            int[] initialRequired,
            int[][] varToConstraints)
        {
            Board = board;
            VarCellIndices = varCellIndices;
            ConstraintVars = constraintVars;
            ConstraintRemaining = (int[])initialRequired.Clone();
            ConstraintUnknowns = new int[constraintVars.Length];
            VarToConstraints = varToConstraints;
            VarAssignments = new sbyte[varCellIndices.Length];
            Array.Fill(VarAssignments, (sbyte)-1);

            for (int i = 0; i < constraintVars.Length; i++)
            {
                ConstraintUnknowns[i] = constraintVars[i].Length;
            }

            _queue = new Queue<int>(constraintVars.Length);
            _inQueue = new bool[constraintVars.Length];
        }

        public void EnqueueAllConstraints()
        {
            for (int i = 0; i < ConstraintVars.Length; i++)
            {
                EnqueueConstraint(i);
            }
        }

        public void EnqueueConstraint(int constraintId)
        {
            if (!_inQueue[constraintId])
            {
                _inQueue[constraintId] = true;
                _queue.Enqueue(constraintId);
            }
        }

        public bool AssignVar(int varId, int value)
        {
            sbyte current = VarAssignments[varId];
            if (current == value)
            {
                return false;
            }

            if (current != -1)
            {
                return false;
            }

            VarAssignments[varId] = (sbyte)value;
            int[] constraints = VarToConstraints[varId];
            for (int i = 0; i < constraints.Length; i++)
            {
                int constraintId = constraints[i];
                ConstraintUnknowns[constraintId]--;
                if (value == 1)
                {
                    ConstraintRemaining[constraintId]--;
                }

                EnqueueConstraint(constraintId);
            }

            return true;
        }

        public void Propagate()
        {
            while (_queue.Count > 0)
            {
                int constraintId = _queue.Dequeue();
                _inQueue[constraintId] = false;

                int remaining = ConstraintRemaining[constraintId];
                int unknowns = ConstraintUnknowns[constraintId];

                if (remaining < 0 || remaining > unknowns || unknowns == 0)
                {
                    continue;
                }

                if (remaining != 0 && remaining != unknowns)
                {
                    continue;
                }

                int forcedState = remaining == 0 ? 0 : 1;
                int[] vars = ConstraintVars[constraintId];
                for (int i = 0; i < vars.Length; i++)
                {
                    int varId = vars[i];
                    if (VarAssignments[varId] == -1)
                    {
                        AssignVar(varId, forcedState);
                    }
                }
            }
        }

        public void CopyAssignmentsToState(byte[] state)
        {
            for (int varId = 0; varId < VarAssignments.Length; varId++)
            {
                sbyte assignment = VarAssignments[varId];
                if (assignment == 0)
                {
                    state[VarCellIndices[varId]] = StateSafe;
                }
                else if (assignment == 1)
                {
                    state[VarCellIndices[varId]] = StateMine;
                }
            }
        }
    }

    private readonly struct ConstraintView
    {
        public readonly int[] Vars;
        public readonly int Required;

        public ConstraintView(int[] vars, int required)
        {
            Vars = vars;
            Required = required;
        }
    }

    private readonly struct Component
    {
        public readonly int[] VarIds;
        public readonly int[] ConstraintIds;

        public Component(int[] varIds, int[] constraintIds)
        {
            VarIds = varIds;
            ConstraintIds = constraintIds;
        }
    }

    private sealed class ExactComponentSolver
    {
        private readonly int _varCount;
        private readonly int[][] _constraintVars;
        private readonly int[][] _varToConstraints;
        private readonly int[] _remaining;
        private readonly int[] _unassigned;
        private readonly sbyte[] _assignment;
        private readonly int _nodeBudget;

        private readonly int[] _varUndoIndex;
        private readonly sbyte[] _varUndoOld;
        private int _varUndoCount;

        private readonly int[] _constraintUndoIndex;
        private readonly int[] _constraintUndoRemaining;
        private readonly int[] _constraintUndoUnassigned;
        private int _constraintUndoCount;

        private readonly ulong _fullMask;
        private ulong _commonSafeMask;
        private ulong _commonMineMask;
        private int _solutionCount;
        private int _nodes;
        private bool _aborted;

        public ExactComponentSolver(int varCount, List<ConstraintView> constraints, int nodeBudget)
        {
            _varCount = varCount;
            _nodeBudget = nodeBudget;
            _assignment = new sbyte[varCount];
            Array.Fill(_assignment, (sbyte)-1);

            _constraintVars = new int[constraints.Count][];
            _remaining = new int[constraints.Count];
            _unassigned = new int[constraints.Count];

            var varLinks = new List<int>[varCount];
            for (int i = 0; i < varCount; i++)
            {
                varLinks[i] = new List<int>(4);
            }

            int edgeCount = 0;
            for (int i = 0; i < constraints.Count; i++)
            {
                _constraintVars[i] = constraints[i].Vars;
                _remaining[i] = constraints[i].Required;
                _unassigned[i] = constraints[i].Vars.Length;
                edgeCount += constraints[i].Vars.Length;

                for (int j = 0; j < constraints[i].Vars.Length; j++)
                {
                    varLinks[constraints[i].Vars[j]].Add(i);
                }
            }

            _varToConstraints = new int[varCount][];
            for (int i = 0; i < varCount; i++)
            {
                _varToConstraints[i] = varLinks[i].ToArray();
            }

            _varUndoIndex = new int[varCount * 2 + 16];
            _varUndoOld = new sbyte[varCount * 2 + 16];
            _constraintUndoIndex = new int[edgeCount * 2 + 16];
            _constraintUndoRemaining = new int[edgeCount * 2 + 16];
            _constraintUndoUnassigned = new int[edgeCount * 2 + 16];

            _fullMask = varCount == 64 ? ulong.MaxValue : ((1UL << varCount) - 1UL);
        }

        public bool TrySolve(out ulong alwaysSafeMask, out ulong alwaysMineMask)
        {
            alwaysSafeMask = 0;
            alwaysMineMask = 0;

            if (!PropagateForced())
            {
                return false;
            }

            Search();
            if (_aborted || _solutionCount == 0)
            {
                return false;
            }

            alwaysSafeMask = _commonSafeMask;
            alwaysMineMask = _commonMineMask;
            return true;
        }

        private void Search()
        {
            if (_aborted)
            {
                return;
            }

            _nodes++;
            if (_nodes > _nodeBudget)
            {
                _aborted = true;
                return;
            }

            int nextVar = ChooseBranchVariable();
            if (nextVar < 0)
            {
                RecordSolution();
                return;
            }

            int varCheckpoint = _varUndoCount;
            int constraintCheckpoint = _constraintUndoCount;
            if (Assign(nextVar, 0) && PropagateForced())
            {
                Search();
            }
            Restore(varCheckpoint, constraintCheckpoint);

            if (_aborted)
            {
                return;
            }

            varCheckpoint = _varUndoCount;
            constraintCheckpoint = _constraintUndoCount;
            if (Assign(nextVar, 1) && PropagateForced())
            {
                Search();
            }
            Restore(varCheckpoint, constraintCheckpoint);
        }

        private int ChooseBranchVariable()
        {
            int bestVar = -1;
            int bestSlack = int.MaxValue;
            int bestDegree = -1;

            for (int constraintId = 0; constraintId < _constraintVars.Length; constraintId++)
            {
                int unknowns = _unassigned[constraintId];
                if (unknowns <= 1)
                {
                    continue;
                }

                int slack = Math.Min(_remaining[constraintId], unknowns - _remaining[constraintId]);
                int[] vars = _constraintVars[constraintId];
                for (int i = 0; i < vars.Length; i++)
                {
                    int varId = vars[i];
                    if (_assignment[varId] != -1)
                    {
                        continue;
                    }

                    int degree = _varToConstraints[varId].Length;
                    if (slack < bestSlack || (slack == bestSlack && degree > bestDegree))
                    {
                        bestSlack = slack;
                        bestDegree = degree;
                        bestVar = varId;
                    }
                }
            }

            if (bestVar >= 0)
            {
                return bestVar;
            }

            for (int varId = 0; varId < _varCount; varId++)
            {
                if (_assignment[varId] == -1)
                {
                    return varId;
                }
            }

            return -1;
        }

        private bool PropagateForced()
        {
            bool changed;
            do
            {
                changed = false;
                for (int constraintId = 0; constraintId < _constraintVars.Length; constraintId++)
                {
                    int remaining = _remaining[constraintId];
                    int unknowns = _unassigned[constraintId];

                    if (remaining < 0 || remaining > unknowns)
                    {
                        return false;
                    }

                    if (unknowns == 0)
                    {
                        continue;
                    }

                    if (remaining != 0 && remaining != unknowns)
                    {
                        continue;
                    }

                    int force = remaining == 0 ? 0 : 1;
                    int[] vars = _constraintVars[constraintId];
                    for (int i = 0; i < vars.Length; i++)
                    {
                        int varId = vars[i];
                        if (_assignment[varId] == -1)
                        {
                            if (!Assign(varId, force))
                            {
                                return false;
                            }

                            changed = true;
                        }
                        else if (_assignment[varId] != force)
                        {
                            return false;
                        }
                    }
                }
            }
            while (changed);

            return true;
        }

        private bool Assign(int varId, int value)
        {
            sbyte current = _assignment[varId];
            if (current == value)
            {
                return true;
            }

            if (current != -1)
            {
                return false;
            }

            _varUndoIndex[_varUndoCount] = varId;
            _varUndoOld[_varUndoCount] = current;
            _varUndoCount++;
            _assignment[varId] = (sbyte)value;

            int[] constraints = _varToConstraints[varId];
            for (int i = 0; i < constraints.Length; i++)
            {
                int constraintId = constraints[i];
                _constraintUndoIndex[_constraintUndoCount] = constraintId;
                _constraintUndoRemaining[_constraintUndoCount] = _remaining[constraintId];
                _constraintUndoUnassigned[_constraintUndoCount] = _unassigned[constraintId];
                _constraintUndoCount++;

                _unassigned[constraintId]--;
                if (value == 1)
                {
                    _remaining[constraintId]--;
                }
            }

            return true;
        }

        private void Restore(int varCheckpoint, int constraintCheckpoint)
        {
            while (_constraintUndoCount > constraintCheckpoint)
            {
                _constraintUndoCount--;
                int constraintId = _constraintUndoIndex[_constraintUndoCount];
                _remaining[constraintId] = _constraintUndoRemaining[_constraintUndoCount];
                _unassigned[constraintId] = _constraintUndoUnassigned[_constraintUndoCount];
            }

            while (_varUndoCount > varCheckpoint)
            {
                _varUndoCount--;
                int varId = _varUndoIndex[_varUndoCount];
                _assignment[varId] = _varUndoOld[_varUndoCount];
            }
        }

        private void RecordSolution()
        {
            ulong mines = 0;
            for (int i = 0; i < _varCount; i++)
            {
                if (_assignment[i] == 1)
                {
                    mines |= 1UL << i;
                }
            }

            ulong safe = _fullMask & ~mines;
            if (_solutionCount == 0)
            {
                _commonMineMask = mines;
                _commonSafeMask = safe;
            }
            else
            {
                _commonMineMask &= mines;
                _commonSafeMask &= safe;
            }

            _solutionCount++;
        }
    }

    private sealed class FastInput
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _position;
        private int _pushback = int.MinValue;

        public int ReadInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c < 0)
                {
                    return int.MinValue;
                }
            }
            while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32)
            {
                value = (value * 10) + (c - '0');
                c = ReadByte();
            }

            return value * sign;
        }

        public void ReadBoard(byte[] board, int height, int width)
        {
            for (int y = 0; y < height; y++)
            {
                int rowStart = y * width;
                int x = 0;
                while (x < width)
                {
                    int c = ReadByte();
                    if (c < 0)
                    {
                        board[rowStart + x] = UnknownCell;
                        x++;
                        continue;
                    }

                    if (c == '\r' || c == '\n')
                    {
                        continue;
                    }

                    board[rowStart + x] = EncodeCell(c);
                    x++;
                }

                int tail = ReadByte();
                if (tail == '\r')
                {
                    int next = ReadByte();
                    if (next != '\n' && next >= 0)
                    {
                        Unread(next);
                    }
                }
                else if (tail != '\n' && tail >= 0)
                {
                    Unread(tail);
                }
            }
        }

        private static byte EncodeCell(int c)
        {
            if (c == ' ')
            {
                return UnknownCell;
            }

            if (c == '*')
            {
                return FlaggedCell;
            }

            return (byte)(c - '0');
        }

        private int ReadByte()
        {
            if (_pushback != int.MinValue)
            {
                int c = _pushback;
                _pushback = int.MinValue;
                return c;
            }

            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }

        private void Unread(int c)
        {
            _pushback = c;
        }
    }
