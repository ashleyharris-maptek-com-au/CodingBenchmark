declare const require: any;
declare const process: any;

const fs = require('fs');

type TaskId = string | number;

type RawTask = {
  id: TaskId;
  duration: number;
  deps?: TaskId[];
  deadline?: number;
  weight?: number;
};

type ScheduleItem = {
  id: TaskId;
  start: number;
  machine: number;
};

type RunningItem = {
  finish: number;
  machine: number;
  task: number;
};

type CandidateResult = {
  name: string;
  schedule: ScheduleItem[];
  tardyScore: number;
  completionScore: number;
};

class Heap<T> {
  private readonly data: T[] = [];
  private readonly better: (a: T, b: T) => boolean;

  constructor(better: (a: T, b: T) => boolean) {
    this.better = better;
  }

  size(): number {
    return this.data.length;
  }

  peek(): T | undefined {
    return this.data[0];
  }

  push(value: T): void {
    const data = this.data;
    data.push(value);
    let index = data.length - 1;
    while (index > 0) {
      const parent = (index - 1) >> 1;
      if (!this.better(data[index], data[parent])) {
        break;
      }
      const tmp = data[index];
      data[index] = data[parent];
      data[parent] = tmp;
      index = parent;
    }
  }

  pop(): T | undefined {
    const data = this.data;
    if (data.length === 0) {
      return undefined;
    }
    const top = data[0];
    const last = data.pop()!;
    if (data.length > 0) {
      data[0] = last;
      let index = 0;
      const length = data.length;
      while (true) {
        const left = (index << 1) + 1;
        const right = left + 1;
        let best = index;
        if (left < length && this.better(data[left], data[best])) {
          best = left;
        }
        if (right < length && this.better(data[right], data[best])) {
          best = right;
        }
        if (best === index) {
          break;
        }
        const tmp = data[index];
        data[index] = data[best];
        data[best] = tmp;
        index = best;
      }
    }
    return top;
  }
}

function asNonNegativeInt(value: unknown): number {
  const n = Number(value);
  if (!Number.isFinite(n)) {
    return 0;
  }
  if (n <= 0) {
    return 0;
  }
  return Math.floor(n);
}

function asWeight(value: unknown): number {
  const n = Number(value);
  if (!Number.isFinite(n)) {
    return 1;
  }
  return n > 0 ? n : 0;
}

function asDeadline(value: unknown): number {
  const n = Number(value);
  if (!Number.isFinite(n)) {
    return 1e15;
  }
  return Math.floor(n);
}

function lessNumber(a: number, b: number): boolean {
  return a < b;
}

function betterRunning(a: RunningItem, b: RunningItem): boolean {
  if (a.finish !== b.finish) {
    return a.finish < b.finish;
  }
  if (a.machine !== b.machine) {
    return a.machine < b.machine;
  }
  return a.task < b.task;
}

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input || '{}');
const rawTasks: RawTask[] = Array.isArray(data.tasks) ? data.tasks : [];
const rawMachineCount = Number(data.machineCount);
const machineCount = Number.isFinite(rawMachineCount) ? Math.floor(rawMachineCount) : 1;
const taskCount = rawTasks.length;

if (taskCount === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const ids: TaskId[] = new Array(taskCount);
const duration: number[] = new Array(taskCount);
const deadline: number[] = new Array(taskCount);
const weight: number[] = new Array(taskCount);
const successors: number[][] = Array.from({ length: taskCount }, () => []);
const indegree: number[] = new Array(taskCount).fill(0);
const idToIndex = new Map<TaskId, number>();

for (let i = 0; i < taskCount; i++) {
  const task = rawTasks[i] || ({} as RawTask);
  const id = task.id ?? i;
  ids[i] = id;
  duration[i] = asNonNegativeInt(task.duration);
  deadline[i] = asDeadline(task.deadline);
  weight[i] = asWeight(task.weight);
  if (!idToIndex.has(id)) {
    idToIndex.set(id, i);
  }
}

for (let i = 0; i < taskCount; i++) {
  const deps = Array.isArray(rawTasks[i]?.deps) ? rawTasks[i]!.deps! : [];
  const seen = deps.length > 1 ? new Set<number>() : null;
  for (let j = 0; j < deps.length; j++) {
    const depIndex = idToIndex.get(deps[j]);
    if (depIndex === undefined || depIndex === i) {
      continue;
    }
    if (seen) {
      if (seen.has(depIndex)) {
        continue;
      }
      seen.add(depIndex);
    }
    successors[depIndex].push(i);
    indegree[i]++;
  }
}

const topo: number[] = new Array(taskCount);
const earliestStart: number[] = new Array(taskCount).fill(0);
const indegreeTmp = indegree.slice();
const queue: number[] = [];
let topoLength = 0;

for (let i = 0; i < taskCount; i++) {
  if (indegreeTmp[i] === 0) {
    queue.push(i);
  }
}

for (let head = 0; head < queue.length; head++) {
  const u = queue[head];
  topo[topoLength++] = u;
  const finish = earliestStart[u] + duration[u];
  const outs = successors[u];
  for (let k = 0; k < outs.length; k++) {
    const v = outs[k];
    if (finish > earliestStart[v]) {
      earliestStart[v] = finish;
    }
    indegreeTmp[v]--;
    if (indegreeTmp[v] === 0) {
      queue.push(v);
    }
  }
}

if (topoLength !== taskCount || machineCount < 1) {
  const order = topoLength === taskCount ? topo.map((i) => ids[i]) : ids.slice();
  process.stdout.write(JSON.stringify({ order }));
  process.exit(0);
}

const tail: number[] = duration.slice();
const latestStart: number[] = new Array(taskCount);
const chainWeight: number[] = new Array(taskCount);
const successorCount: number[] = new Array(taskCount);

for (let i = 0; i < taskCount; i++) {
  latestStart[i] = deadline[i] - duration[i];
  chainWeight[i] = weight[i];
  successorCount[i] = successors[i].length;
}

for (let pos = taskCount - 1; pos >= 0; pos--) {
  const u = topo[pos];
  const outs = successors[u];
  let bestTail = duration[u];
  let bestLatestStart = deadline[u] - duration[u];
  let bestChainWeight = weight[u];
  for (let k = 0; k < outs.length; k++) {
    const v = outs[k];
    const candTail = duration[u] + tail[v];
    if (candTail > bestTail) {
      bestTail = candTail;
    }
    const candLatestStart = latestStart[v] - duration[u];
    if (candLatestStart < bestLatestStart) {
      bestLatestStart = candLatestStart;
    }
    const candChainWeight = weight[u] + chainWeight[v];
    if (candChainWeight > bestChainWeight) {
      bestChainWeight = candChainWeight;
    }
  }
  tail[u] = bestTail;
  latestStart[u] = bestLatestStart;
  chainWeight[u] = bestChainWeight;
}

const density: number[] = new Array(taskCount);
const chainDensity: number[] = new Array(taskCount);
const staticSlack: number[] = new Array(taskCount);

for (let i = 0; i < taskCount; i++) {
  const durationBase = duration[i] > 0 ? duration[i] : 0.5;
  const tailBase = tail[i] > 0 ? tail[i] : 0.5;
  density[i] = weight[i] / durationBase;
  chainDensity[i] = chainWeight[i] / tailBase;
  staticSlack[i] = latestStart[i] - earliestStart[i];
}

function compareByUrgency(a: number, b: number): boolean {
  if (latestStart[a] !== latestStart[b]) {
    return latestStart[a] < latestStart[b];
  }
  if (chainWeight[a] !== chainWeight[b]) {
    return chainWeight[a] > chainWeight[b];
  }
  if (tail[a] !== tail[b]) {
    return tail[a] > tail[b];
  }
  if (density[a] !== density[b]) {
    return density[a] > density[b];
  }
  if (successorCount[a] !== successorCount[b]) {
    return successorCount[a] > successorCount[b];
  }
  return a < b;
}

function compareByDensity(a: number, b: number): boolean {
  if (chainDensity[a] !== chainDensity[b]) {
    return chainDensity[a] > chainDensity[b];
  }
  if (density[a] !== density[b]) {
    return density[a] > density[b];
  }
  if (latestStart[a] !== latestStart[b]) {
    return latestStart[a] < latestStart[b];
  }
  if (tail[a] !== tail[b]) {
    return tail[a] > tail[b];
  }
  if (successorCount[a] !== successorCount[b]) {
    return successorCount[a] > successorCount[b];
  }
  return a < b;
}

function compareByCriticalPath(a: number, b: number): boolean {
  if (tail[a] !== tail[b]) {
    return tail[a] > tail[b];
  }
  if (latestStart[a] !== latestStart[b]) {
    return latestStart[a] < latestStart[b];
  }
  if (chainWeight[a] !== chainWeight[b]) {
    return chainWeight[a] > chainWeight[b];
  }
  if (density[a] !== density[b]) {
    return density[a] > density[b];
  }
  if (successorCount[a] !== successorCount[b]) {
    return successorCount[a] > successorCount[b];
  }
  return a < b;
}

function compareByChainWeight(a: number, b: number): boolean {
  if (chainWeight[a] !== chainWeight[b]) {
    return chainWeight[a] > chainWeight[b];
  }
  if (latestStart[a] !== latestStart[b]) {
    return latestStart[a] < latestStart[b];
  }
  if (tail[a] !== tail[b]) {
    return tail[a] > tail[b];
  }
  if (density[a] !== density[b]) {
    return density[a] > density[b];
  }
  if (successorCount[a] !== successorCount[b]) {
    return successorCount[a] > successorCount[b];
  }
  return a < b;
}

function compareBySlack(a: number, b: number): boolean {
  if (staticSlack[a] !== staticSlack[b]) {
    return staticSlack[a] < staticSlack[b];
  }
  if (latestStart[a] !== latestStart[b]) {
    return latestStart[a] < latestStart[b];
  }
  if (chainWeight[a] !== chainWeight[b]) {
    return chainWeight[a] > chainWeight[b];
  }
  if (tail[a] !== tail[b]) {
    return tail[a] > tail[b];
  }
  if (density[a] !== density[b]) {
    return density[a] > density[b];
  }
  return a < b;
}

function buildSchedule(
  name: string,
  betterTask: (a: number, b: number) => boolean,
): CandidateResult {
  const remaining = indegree.slice();
  const readyTime: number[] = new Array(taskCount).fill(0);
  const ready = new Heap<number>(betterTask);
  const freeMachines = new Heap<number>(lessNumber);
  const running = new Heap<RunningItem>(betterRunning);
  const schedule: ScheduleItem[] = new Array(taskCount);

  for (let machine = 0; machine < machineCount; machine++) {
    freeMachines.push(machine);
  }
  for (let i = 0; i < taskCount; i++) {
    if (remaining[i] === 0) {
      ready.push(i);
    }
  }

  let currentTime = 0;
  let placed = 0;
  let makespan = 0;
  let weightedTardiness = 0;
  let weightedCompletion = 0;

  while (placed < taskCount) {
    while (freeMachines.size() > 0 && ready.size() > 0) {
      const task = ready.pop()!;
      const machine = freeMachines.pop()!;
      const start = currentTime > readyTime[task] ? currentTime : readyTime[task];
      const finish = start + duration[task];
      schedule[placed++] = { id: ids[task], start, machine };
      if (finish > makespan) {
        makespan = finish;
      }
      weightedCompletion += weight[task] * finish;
      if (finish > deadline[task]) {
        weightedTardiness += weight[task] * (finish - deadline[task]);
      }

      if (duration[task] === 0) {
        freeMachines.push(machine);
        const outs = successors[task];
        for (let k = 0; k < outs.length; k++) {
          const v = outs[k];
          if (finish > readyTime[v]) {
            readyTime[v] = finish;
          }
          remaining[v]--;
          if (remaining[v] === 0) {
            ready.push(v);
          }
        }
      } else {
        running.push({ finish, machine, task });
      }
    }

    if (placed >= taskCount) {
      break;
    }

    const next = running.pop();
    if (!next) {
      break;
    }

    currentTime = next.finish;
    freeMachines.push(next.machine);
    const completedTasks: number[] = [next.task];

    while (running.size() > 0 && running.peek()!.finish === currentTime) {
      const item = running.pop()!;
      freeMachines.push(item.machine);
      completedTasks.push(item.task);
    }

    for (let i = 0; i < completedTasks.length; i++) {
      const task = completedTasks[i];
      const outs = successors[task];
      for (let k = 0; k < outs.length; k++) {
        const v = outs[k];
        if (currentTime > readyTime[v]) {
          readyTime[v] = currentTime;
        }
        remaining[v]--;
        if (remaining[v] === 0) {
          ready.push(v);
        }
      }
    }
  }

  if (placed !== taskCount) {
    const fallback = topo.map((idx) => ids[idx]);
    process.stdout.write(JSON.stringify({ order: fallback }));
    process.exit(0);
  }

  return {
    name,
    schedule,
    tardyScore: weightedTardiness + makespan,
    completionScore: weightedCompletion + makespan,
  };
}

const strategies: Array<{ name: string; cmp: (a: number, b: number) => boolean }> = [
  { name: 'urgent', cmp: compareByUrgency },
  { name: 'density', cmp: compareByDensity },
  { name: 'critical', cmp: compareByCriticalPath },
];

if (taskCount <= 75000) {
  strategies.push({ name: 'chainWeight', cmp: compareByChainWeight });
}
if (taskCount <= 30000) {
  strategies.push({ name: 'slack', cmp: compareBySlack });
}

let best = buildSchedule(strategies[0].name, strategies[0].cmp);

for (let i = 1; i < strategies.length; i++) {
  const candidate = buildSchedule(strategies[i].name, strategies[i].cmp);
  if (candidate.tardyScore < best.tardyScore) {
    best = candidate;
    continue;
  }
  if (candidate.tardyScore === best.tardyScore && candidate.completionScore < best.completionScore) {
    best = candidate;
  }
}

process.stdout.write(JSON.stringify({ schedule: best.schedule }));
