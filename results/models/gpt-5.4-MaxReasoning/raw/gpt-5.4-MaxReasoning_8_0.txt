use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read, Write};

const DIR_LEFT: u8 = 0;
const DIR_RIGHT: u8 = 1;
const DIR_UP: u8 = 2;
const DIR_DOWN: u8 = 3;

const BFS_AREA_LIMIT: usize = 8_000_000;
const BFS_WALLY_AREA_LIMIT: usize = 32_000_000;
const WALL_DENSITY_BFS_PERCENT: u64 = 35;

struct BitGrid {
    bits: Vec<u64>,
}

impl BitGrid {
    fn new(len: usize) -> Self {
        Self {
            bits: vec![0; len.div_ceil(64)],
        }
    }

    #[inline]
    fn get(&self, idx: usize) -> bool {
        ((self.bits[idx >> 6] >> (idx & 63)) & 1) != 0
    }

    #[inline]
    fn set(&mut self, idx: usize) {
        self.bits[idx >> 6] |= 1u64 << (idx & 63);
    }
}

struct PackedDirs {
    data: Vec<u8>,
}

impl PackedDirs {
    fn new(len: usize) -> Self {
        Self {
            data: vec![0; len.div_ceil(4)],
        }
    }

    #[inline]
    fn get(&self, idx: usize) -> u8 {
        let shift = (idx & 3) * 2;
        (self.data[idx >> 2] >> shift) & 0b11
    }

    #[inline]
    fn set(&mut self, idx: usize, dir: u8) {
        let shift = (idx & 3) * 2;
        let mask = !(0b11 << shift);
        let cell = &mut self.data[idx >> 2];
        *cell = (*cell & mask) | (dir << shift);
    }
}

struct Maze {
    h: usize,
    w: usize,
    w_u32: u32,
    blocked: BitGrid,
    start: u32,
    goal: u32,
    wall_count: usize,
}

#[derive(Copy, Clone, Eq, PartialEq)]
struct State {
    f: u32,
    g: u32,
    idx: u32,
}

impl Ord for State {
    fn cmp(&self, other: &Self) -> Ordering {
        other
            .f
            .cmp(&self.f)
            .then_with(|| other.g.cmp(&self.g))
            .then_with(|| other.idx.cmp(&self.idx))
    }
}

impl PartialOrd for State {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

#[inline]
fn heuristic(idx: u32, goal: u32, width: u32) -> u32 {
    let x1 = idx % width;
    let y1 = idx / width;
    let x2 = goal % width;
    let y2 = goal / width;
    x1.abs_diff(x2) + y1.abs_diff(y2)
}

#[inline]
fn step_to_parent(idx: u32, dir: u8, width: u32) -> u32 {
    match dir {
        DIR_LEFT => idx - 1,
        DIR_RIGHT => idx + 1,
        DIR_UP => idx - width,
        DIR_DOWN => idx + width,
        _ => unreachable!("invalid parent direction"),
    }
}

fn reconstruct_path(start: u32, goal: u32, parents: &PackedDirs, width: u32) -> Vec<u32> {
    let mut path = Vec::new();
    let mut cur = goal;
    loop {
        path.push(cur);
        if cur == start {
            break;
        }
        cur = step_to_parent(cur, parents.get(cur as usize), width);
    }
    path.reverse();
    path
}

fn bfs(maze: &Maze) -> Option<Vec<u32>> {
    let area = maze.h * maze.w;
    let mut seen = BitGrid::new(area);
    let mut parents = PackedDirs::new(area);
    let mut queue = Vec::with_capacity((maze.w + maze.h).max(1024));
    let mut head = 0usize;

    seen.set(maze.start as usize);
    queue.push(maze.start);

    while head < queue.len() {
        let idx = queue[head];
        head += 1;

        let x = idx % maze.w_u32;

        if x > 0 {
            let next = idx - 1;
            let next_usize = next as usize;
            if !maze.blocked.get(next_usize) && !seen.get(next_usize) {
                seen.set(next_usize);
                parents.set(next_usize, DIR_RIGHT);
                if next == maze.goal {
                    return Some(reconstruct_path(maze.start, maze.goal, &parents, maze.w_u32));
                }
                queue.push(next);
            }
        }

        if x + 1 < maze.w_u32 {
            let next = idx + 1;
            let next_usize = next as usize;
            if !maze.blocked.get(next_usize) && !seen.get(next_usize) {
                seen.set(next_usize);
                parents.set(next_usize, DIR_LEFT);
                if next == maze.goal {
                    return Some(reconstruct_path(maze.start, maze.goal, &parents, maze.w_u32));
                }
                queue.push(next);
            }
        }

        if idx >= maze.w_u32 {
            let next = idx - maze.w_u32;
            let next_usize = next as usize;
            if !maze.blocked.get(next_usize) && !seen.get(next_usize) {
                seen.set(next_usize);
                parents.set(next_usize, DIR_DOWN);
                if next == maze.goal {
                    return Some(reconstruct_path(maze.start, maze.goal, &parents, maze.w_u32));
                }
                queue.push(next);
            }
        }

        let next = idx + maze.w_u32;
        if next < area as u32 {
            let next_usize = next as usize;
            if !maze.blocked.get(next_usize) && !seen.get(next_usize) {
                seen.set(next_usize);
                parents.set(next_usize, DIR_UP);
                if next == maze.goal {
                    return Some(reconstruct_path(maze.start, maze.goal, &parents, maze.w_u32));
                }
                queue.push(next);
            }
        }
    }

    None
}

fn astar(maze: &Maze) -> Option<Vec<u32>> {
    let area = maze.h * maze.w;
    let mut g_score = vec![u32::MAX; area];
    let mut parents = PackedDirs::new(area);
    let mut open = BinaryHeap::with_capacity((maze.w + maze.h).max(1024));

    g_score[maze.start as usize] = 0;
    open.push(State {
        f: heuristic(maze.start, maze.goal, maze.w_u32),
        g: 0,
        idx: maze.start,
    });

    while let Some(State { g, idx, .. }) = open.pop() {
        if g != g_score[idx as usize] {
            continue;
        }

        let x = idx % maze.w_u32;
        let next_g = g + 1;

        if x > 0 {
            let next = idx - 1;
            let next_usize = next as usize;
            if !maze.blocked.get(next_usize) && next_g < g_score[next_usize] {
                g_score[next_usize] = next_g;
                parents.set(next_usize, DIR_RIGHT);
                if next == maze.goal {
                    return Some(reconstruct_path(maze.start, maze.goal, &parents, maze.w_u32));
                }
                open.push(State {
                    f: next_g + heuristic(next, maze.goal, maze.w_u32),
                    g: next_g,
                    idx: next,
                });
            }
        }

        if x + 1 < maze.w_u32 {
            let next = idx + 1;
            let next_usize = next as usize;
            if !maze.blocked.get(next_usize) && next_g < g_score[next_usize] {
                g_score[next_usize] = next_g;
                parents.set(next_usize, DIR_LEFT);
                if next == maze.goal {
                    return Some(reconstruct_path(maze.start, maze.goal, &parents, maze.w_u32));
                }
                open.push(State {
                    f: next_g + heuristic(next, maze.goal, maze.w_u32),
                    g: next_g,
                    idx: next,
                });
            }
        }

        if idx >= maze.w_u32 {
            let next = idx - maze.w_u32;
            let next_usize = next as usize;
            if !maze.blocked.get(next_usize) && next_g < g_score[next_usize] {
                g_score[next_usize] = next_g;
                parents.set(next_usize, DIR_DOWN);
                if next == maze.goal {
                    return Some(reconstruct_path(maze.start, maze.goal, &parents, maze.w_u32));
                }
                open.push(State {
                    f: next_g + heuristic(next, maze.goal, maze.w_u32),
                    g: next_g,
                    idx: next,
                });
            }
        }

        let next = idx + maze.w_u32;
        if next < area as u32 {
            let next_usize = next as usize;
            if !maze.blocked.get(next_usize) && next_g < g_score[next_usize] {
                g_score[next_usize] = next_g;
                parents.set(next_usize, DIR_UP);
                if next == maze.goal {
                    return Some(reconstruct_path(maze.start, maze.goal, &parents, maze.w_u32));
                }
                open.push(State {
                    f: next_g + heuristic(next, maze.goal, maze.w_u32),
                    g: next_g,
                    idx: next,
                });
            }
        }
    }

    None
}

fn parse_maze(input: &[u8]) -> Maze {
    let header_end = input.iter().position(|&b| b == b'\n').unwrap_or(input.len());
    let mut header = &input[..header_end];
    if matches!(header.last(), Some(&b'\r')) {
        header = &header[..header.len() - 1];
    }

    let header_str = std::str::from_utf8(header).expect("header must be valid UTF-8");
    let mut header_parts = header_str.split_ascii_whitespace();
    let h: usize = header_parts
        .next()
        .expect("missing height")
        .parse()
        .expect("invalid height");
    let w: usize = header_parts
        .next()
        .expect("missing width")
        .parse()
        .expect("invalid width");

    let area = h.checked_mul(w).expect("maze dimensions overflow");
    assert!(area <= u32::MAX as usize, "maze exceeds supported cell count");

    let mut blocked = BitGrid::new(area);
    let mut start = None;
    let mut goal = None;
    let mut wall_count = 0usize;

    let mut pos = if header_end < input.len() {
        header_end + 1
    } else {
        header_end
    };

    for y in 0..h {
        let row_start = pos;
        while pos < input.len() && input[pos] != b'\n' {
            pos += 1;
        }

        let mut row = &input[row_start..pos];
        if matches!(row.last(), Some(&b'\r')) {
            row = &row[..row.len() - 1];
        }
        assert!(row.len() >= w, "maze row {} is shorter than width", y);

        for (x, &cell) in row.iter().take(w).enumerate() {
            let idx = y * w + x;
            match cell {
                b'#' => {
                    blocked.set(idx);
                    wall_count += 1;
                }
                b'A' => start = Some(idx as u32),
                b'B' => goal = Some(idx as u32),
                _ => {}
            }
        }

        if pos < input.len() && input[pos] == b'\n' {
            pos += 1;
        }
    }

    Maze {
        h,
        w,
        w_u32: w as u32,
        blocked,
        start: start.expect("missing start cell 'A'"),
        goal: goal.expect("missing goal cell 'B'"),
        wall_count,
    }
}

fn main() {
    let mut input = Vec::new();
    io::stdin()
        .read_to_end(&mut input)
        .expect("failed to read stdin");
    if input.is_empty() {
        return;
    }

    let maze = parse_maze(&input);
    let area = maze.h * maze.w;

    let use_bfs = area <= BFS_AREA_LIMIT
        || (area <= BFS_WALLY_AREA_LIMIT
            && (maze.wall_count as u64) * 100 >= (area as u64) * WALL_DENSITY_BFS_PERCENT);

    let path = if maze.start == maze.goal {
        Some(vec![maze.start])
    } else if use_bfs {
        bfs(&maze)
    } else {
        astar(&maze)
    };

    let mut out = io::BufWriter::new(io::stdout().lock());
    match path {
        Some(path) => {
            writeln!(out, "{}", path.len()).unwrap();
            for idx in path {
                let x = idx % maze.w_u32;
                let y = idx / maze.w_u32;
                writeln!(out, "{} {}", x, y).unwrap();
            }
        }
        None => {
            writeln!(out, "0").unwrap();
        }
    }
}
