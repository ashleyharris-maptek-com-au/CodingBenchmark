#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

vec3 computeBoundaryForce(vec3 pos, float boundSize, float boundForce)
{
    vec3 force = vec3(0.0);
    float edge = boundSize * 0.4;
    float scale = max(boundSize * 0.5, 1e-5);

    if (pos.x > edge) {
        force.x -= boundForce * (pos.x - edge) / scale;
    } else if (pos.x < -edge) {
        force.x += boundForce * (-edge - pos.x) / scale;
    }

    if (pos.y > edge) {
        force.y -= boundForce * (pos.y - edge) / scale;
    } else if (pos.y < -edge) {
        force.y += boundForce * (-edge - pos.y) / scale;
    }

    if (pos.z > edge) {
        force.z -= boundForce * (pos.z - edge) / scale;
    } else if (pos.z < -edge) {
        force.z += boundForce * (-edge - pos.z) / scale;
    }

    return force;
}

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint count = counts.x;

    if (i >= count) {
        return;
    }

    uint baseIndex = i * 2u;
    vec3 pos = inData[baseIndex].xyz;
    vec3 vel = inData[baseIndex + 1u].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepRadius2 = sepRadius * sepRadius;
    float alignRadius2 = alignRadius * alignRadius;
    float cohRadius2 = cohRadius * cohRadius;

    vec3 sepSum = vec3(0.0);
    vec3 alignSum = vec3(0.0);
    vec3 cohSum = vec3(0.0);

    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < count; ++j) {
        if (j == i) {
            continue;
        }

        uint otherBase = j * 2u;
        vec3 otherPos = inData[otherBase].xyz;
        vec3 otherVel = inData[otherBase + 1u].xyz;

        vec3 offset = pos - otherPos;
        float dist2 = dot(offset, offset);

        if (dist2 <= 0.0) {
            continue;
        }

        if (dist2 < sepRadius2) {
            sepSum += offset / max(dist2, 1e-6);
            ++sepCount;
        }

        if (dist2 < alignRadius2) {
            alignSum += otherVel;
            ++alignCount;
        }

        if (dist2 < cohRadius2) {
            cohSum += otherPos;
            ++cohCount;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        accel += (sepSum / float(sepCount)) * weights.x;
    }

    if (alignCount > 0u) {
        vec3 avgVel = alignSum / float(alignCount);
        accel += (avgVel - vel) * weights.y;
    }

    if (cohCount > 0u) {
        vec3 center = cohSum / float(cohCount);
        accel += (center - pos) * weights.z;
    }

    accel += computeBoundaryForce(pos, bounds.x, bounds.y);

    vec3 newVel = vel + accel * dt;
    float speed = length(newVel);
    float maxSpeed = weights.w;

    if (speed > maxSpeed && speed > 0.0) {
        newVel = (newVel / speed) * maxSpeed;
    }

    vec3 newPos = pos + newVel * dt;

    outData[baseIndex] = vec4(newPos, 0.0);
    outData[baseIndex + 1u] = vec4(newVel, 0.0);
}
