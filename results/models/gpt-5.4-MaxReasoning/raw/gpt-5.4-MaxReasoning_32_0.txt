using namespace std;

namespace {

constexpr long long INF = numeric_limits<long long>::max() / 4;

struct Edge {
    int u = 0;
    int v = 0;
    long long w = 0;
};

struct AdjEdge {
    int to = 0;
    int id = 0;
    long long w = 0;
};

struct DSU {
    vector<int> parent;
    vector<int> rankv;

    explicit DSU(int n) : parent(n), rankv(n, 0) {
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        while (parent[x] != x) {
            parent[x] = parent[parent[x]];
            x = parent[x];
        }
        return x;
    }

    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) {
            return false;
        }
        if (rankv[a] < rankv[b]) {
            swap(a, b);
        }
        parent[b] = a;
        if (rankv[a] == rankv[b]) {
            ++rankv[a];
        }
        return true;
    }
};

void dijkstra_single(const vector<vector<AdjEdge>>& graph,
                     int source,
                     vector<long long>& dist,
                     vector<int>* parent_v = nullptr,
                     vector<int>* parent_e = nullptr) {
    const int n = static_cast<int>(graph.size());
    dist.assign(n, INF);
    if (parent_v != nullptr) {
        parent_v->assign(n, -1);
    }
    if (parent_e != nullptr) {
        parent_e->assign(n, -1);
    }

    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    dist[source] = 0;
    pq.push({0, source});

    while (!pq.empty()) {
        const auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) {
            continue;
        }
        for (const auto& e : graph[u]) {
            const long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                if (parent_v != nullptr) {
                    (*parent_v)[e.to] = u;
                }
                if (parent_e != nullptr) {
                    (*parent_e)[e.to] = e.id;
                }
                pq.push({nd, e.to});
            }
        }
    }
}

int dijkstra_to_remaining(const vector<vector<AdjEdge>>& graph,
                          const vector<int>& sources,
                          const vector<char>& remaining_terminal,
                          vector<int>& parent_v,
                          vector<int>& parent_e) {
    const int n = static_cast<int>(graph.size());
    vector<long long> dist(n, INF);
    parent_v.assign(n, -1);
    parent_e.assign(n, -1);

    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    for (int s : sources) {
        if (dist[s] == 0) {
            continue;
        }
        dist[s] = 0;
        pq.push({0, s});
    }

    while (!pq.empty()) {
        const auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) {
            continue;
        }
        if (remaining_terminal[u]) {
            return u;
        }
        for (const auto& e : graph[u]) {
            const long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parent_v[e.to] = u;
                parent_e[e.to] = e.id;
                pq.push({nd, e.to});
            }
        }
    }

    return -1;
}

vector<int> exact_steiner(const vector<vector<AdjEdge>>& graph, const vector<int>& terminals) {
    const int n = static_cast<int>(graph.size());
    const int k = static_cast<int>(terminals.size());
    const size_t mask_count = size_t{1} << k;
    const size_t total_states = mask_count * static_cast<size_t>(n);

    vector<long long> dp(total_states, INF);
    vector<int> split(total_states, -1);
    vector<int> prev_v(total_states, -1);
    vector<int> prev_e(total_states, -1);

    const auto idx = [n](size_t mask, int v) -> size_t {
        return mask * static_cast<size_t>(n) + static_cast<size_t>(v);
    };

    for (int i = 0; i < k; ++i) {
        dp[idx(size_t{1} << i, terminals[i])] = 0;
    }

    for (size_t mask = 1; mask < mask_count; ++mask) {
        if ((mask & (mask - 1)) != 0) {
            for (size_t sub = (mask - 1) & mask; sub != 0; sub = (sub - 1) & mask) {
                const size_t other = mask ^ sub;
                if (sub >= other) {
                    continue;
                }
                const size_t base_mask = mask * static_cast<size_t>(n);
                const size_t base_sub = sub * static_cast<size_t>(n);
                const size_t base_other = other * static_cast<size_t>(n);
                for (int v = 0; v < n; ++v) {
                    const long long a = dp[base_sub + static_cast<size_t>(v)];
                    const long long b = dp[base_other + static_cast<size_t>(v)];
                    if (a >= INF / 2 || b >= INF / 2) {
                        continue;
                    }
                    const long long cand = a + b;
                    const size_t state = base_mask + static_cast<size_t>(v);
                    if (cand < dp[state]) {
                        dp[state] = cand;
                        split[state] = static_cast<int>(sub);
                        prev_v[state] = -1;
                        prev_e[state] = -1;
                    }
                }
            }
        }

        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        const size_t base = mask * static_cast<size_t>(n);
        for (int v = 0; v < n; ++v) {
            const long long d = dp[base + static_cast<size_t>(v)];
            if (d < INF / 2) {
                pq.push({d, v});
            }
        }

        while (!pq.empty()) {
            const auto [d, u] = pq.top();
            pq.pop();
            if (d != dp[base + static_cast<size_t>(u)]) {
                continue;
            }
            for (const auto& e : graph[u]) {
                const long long nd = d + e.w;
                const size_t state = base + static_cast<size_t>(e.to);
                if (nd < dp[state]) {
                    dp[state] = nd;
                    split[state] = -1;
                    prev_v[state] = u;
                    prev_e[state] = e.id;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    const size_t full = mask_count - 1;
    int root = -1;
    long long best = INF;
    for (int v = 0; v < n; ++v) {
        const long long cur = dp[idx(full, v)];
        if (cur < best) {
            best = cur;
            root = v;
        }
    }
    if (root < 0 || best >= INF / 2) {
        return {};
    }

    vector<char> seen(total_states, 0);
    vector<int> candidate_edges;
    candidate_edges.reserve(n);

    function<void(size_t, int)> build = [&](size_t mask, int v) {
        const size_t state = idx(mask, v);
        if (seen[state]) {
            return;
        }
        seen[state] = 1;

        if (prev_v[state] != -1) {
            candidate_edges.push_back(prev_e[state]);
            build(mask, prev_v[state]);
            return;
        }

        if (split[state] != -1) {
            const size_t left = static_cast<size_t>(split[state]);
            build(left, v);
            build(mask ^ left, v);
        }
    };

    build(full, root);
    return candidate_edges;
}

vector<int> heuristic_kmb(const vector<vector<AdjEdge>>& graph, const vector<int>& terminals) {
    const int k = static_cast<int>(terminals.size());
    vector<vector<long long>> metric(k, vector<long long>(k, INF));
    vector<long long> dist;

    for (int i = 0; i < k; ++i) {
        dijkstra_single(graph, terminals[i], dist);
        for (int j = 0; j < k; ++j) {
            metric[i][j] = dist[terminals[j]];
        }
    }

    vector<long long> best(k, INF);
    vector<int> parent(k, -1);
    vector<char> used(k, 0);
    best[0] = 0;

    for (int iter = 0; iter < k; ++iter) {
        int u = -1;
        for (int i = 0; i < k; ++i) {
            if (!used[i] && (u == -1 || best[i] < best[u])) {
                u = i;
            }
        }
        if (u == -1 || best[u] >= INF / 2) {
            return {};
        }
        used[u] = 1;
        for (int v = 0; v < k; ++v) {
            if (!used[v] && metric[u][v] < best[v]) {
                best[v] = metric[u][v];
                parent[v] = u;
            }
        }
    }

    vector<vector<int>> children(k);
    for (int v = 1; v < k; ++v) {
        if (parent[v] < 0) {
            return {};
        }
        children[parent[v]].push_back(v);
    }

    vector<int> candidate_edges;
    vector<int> parent_v;
    vector<int> parent_e;

    for (int src = 0; src < k; ++src) {
        if (children[src].empty()) {
            continue;
        }
        dijkstra_single(graph, terminals[src], dist, &parent_v, &parent_e);
        for (int child : children[src]) {
            int cur = terminals[child];
            if (dist[cur] >= INF / 2) {
                return {};
            }
            while (cur != terminals[src]) {
                const int eid = parent_e[cur];
                if (eid < 0) {
                    return {};
                }
                candidate_edges.push_back(eid);
                cur = parent_v[cur];
            }
        }
    }

    return candidate_edges;
}

vector<int> heuristic_greedy(const vector<vector<AdjEdge>>& graph, const vector<int>& terminals) {
    const int n = static_cast<int>(graph.size());
    vector<char> in_tree(n, 0);
    vector<char> remaining_terminal(n, 0);
    vector<int> tree_vertices;
    tree_vertices.reserve(n);

    in_tree[terminals[0]] = 1;
    tree_vertices.push_back(terminals[0]);

    int remaining_count = 0;
    for (size_t i = 1; i < terminals.size(); ++i) {
        if (!remaining_terminal[terminals[i]]) {
            remaining_terminal[terminals[i]] = 1;
            ++remaining_count;
        }
    }

    vector<int> candidate_edges;
    vector<int> parent_v;
    vector<int> parent_e;

    while (remaining_count > 0) {
        const int target = dijkstra_to_remaining(graph, tree_vertices, remaining_terminal, parent_v, parent_e);
        if (target < 0) {
            return {};
        }

        int cur = target;
        while (!in_tree[cur]) {
            const int eid = parent_e[cur];
            if (eid < 0) {
                return {};
            }
            candidate_edges.push_back(eid);
            in_tree[cur] = 1;
            tree_vertices.push_back(cur);
            if (remaining_terminal[cur]) {
                remaining_terminal[cur] = 0;
                --remaining_count;
            }
            cur = parent_v[cur];
            if (cur < 0) {
                return {};
            }
        }
    }

    return candidate_edges;
}

vector<int> finalize_tree(int n,
                          const vector<Edge>& edges,
                          const vector<int>& terminals,
                          vector<int> candidate_edges) {
    if (terminals.size() <= 1) {
        return {};
    }
    if (candidate_edges.empty()) {
        return {};
    }

    sort(candidate_edges.begin(), candidate_edges.end());
    candidate_edges.erase(unique(candidate_edges.begin(), candidate_edges.end()), candidate_edges.end());
    stable_sort(candidate_edges.begin(), candidate_edges.end(), [&](int a, int b) {
        if (edges[a].w != edges[b].w) {
            return edges[a].w < edges[b].w;
        }
        return a < b;
    });

    DSU dsu(n);
    vector<int> tree_edges;
    tree_edges.reserve(candidate_edges.size());
    for (int id : candidate_edges) {
        if (dsu.unite(edges[id].u, edges[id].v)) {
            tree_edges.push_back(id);
        }
    }

    const int root = dsu.find(terminals[0]);
    for (int t : terminals) {
        if (dsu.find(t) != root) {
            return {};
        }
    }

    vector<char> is_terminal(n, 0);
    for (int t : terminals) {
        is_terminal[t] = 1;
    }

    vector<vector<int>> incident(n);
    vector<int> degree(n, 0);
    for (int i = 0; i < static_cast<int>(tree_edges.size()); ++i) {
        const auto& e = edges[tree_edges[i]];
        incident[e.u].push_back(i);
        incident[e.v].push_back(i);
        ++degree[e.u];
        ++degree[e.v];
    }

    vector<char> active(tree_edges.size(), 1);
    queue<int> q;
    for (int v = 0; v < n; ++v) {
        if (degree[v] == 1 && !is_terminal[v]) {
            q.push(v);
        }
    }

    while (!q.empty()) {
        const int v = q.front();
        q.pop();
        if (degree[v] != 1 || is_terminal[v]) {
            continue;
        }

        int edge_pos = -1;
        for (int pos : incident[v]) {
            if (active[pos]) {
                edge_pos = pos;
                break;
            }
        }
        if (edge_pos < 0) {
            continue;
        }

        active[edge_pos] = 0;
        const auto& e = edges[tree_edges[edge_pos]];
        const int other = (e.u == v ? e.v : e.u);
        --degree[v];
        --degree[other];
        if (degree[other] == 1 && !is_terminal[other]) {
            q.push(other);
        }
    }

    vector<int> result;
    result.reserve(tree_edges.size());
    for (int i = 0; i < static_cast<int>(tree_edges.size()); ++i) {
        if (active[i]) {
            result.push_back(tree_edges[i]);
        }
    }
    sort(result.begin(), result.end());
    return result;
}

}  // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int num_vertices = 0;
    int num_edges = 0;
    int num_terminals = 0;
    if (!(cin >> num_vertices >> num_edges >> num_terminals)) {
        return 0;
    }

    vector<Edge> edges(num_edges);
    vector<vector<AdjEdge>> graph(num_vertices);
    for (int i = 0; i < num_edges; ++i) {
        int u = 0;
        int v = 0;
        long long w = 0;
        cin >> u >> v >> w;
        edges[i] = {u, v, w};
        graph[u].push_back({v, i, w});
        graph[v].push_back({u, i, w});
    }

    vector<int> terminals(num_terminals);
    for (int i = 0; i < num_terminals; ++i) {
        cin >> terminals[i];
    }
    sort(terminals.begin(), terminals.end());
    terminals.erase(unique(terminals.begin(), terminals.end()), terminals.end());

    if (terminals.size() <= 1) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    vector<int> result;

    const int k = static_cast<int>(terminals.size());
    if (k <= 12) {
        const size_t mask_count = size_t{1} << k;
        const unsigned long long state_budget =
            static_cast<unsigned long long>(mask_count) * static_cast<unsigned long long>(num_vertices);
        const unsigned long long relax_budget =
            static_cast<unsigned long long>(mask_count) * static_cast<unsigned long long>(num_edges);
        bool use_exact = false;
        if (k <= 10) {
            use_exact = (state_budget <= 4000000ULL && relax_budget <= 50000000ULL);
        } else {
            use_exact = (state_budget <= 1500000ULL && relax_budget <= 8000000ULL);
        }
        if (use_exact) {
            result = finalize_tree(num_vertices, edges, terminals, exact_steiner(graph, terminals));
        }
    }

    if (result.empty()) {
        if (k <= 200) {
            result = finalize_tree(num_vertices, edges, terminals, heuristic_kmb(graph, terminals));
            if (result.empty()) {
                result = finalize_tree(num_vertices, edges, terminals, heuristic_greedy(graph, terminals));
            }
        } else {
            result = finalize_tree(num_vertices, edges, terminals, heuristic_greedy(graph, terminals));
            if (result.empty() && k <= 400) {
                result = finalize_tree(num_vertices, edges, terminals, heuristic_kmb(graph, terminals));
            }
        }
    }

    if (result.empty()) {
        cout << -1 << ' ' << 0 << '\n';
        return 0;
    }

    long long total_weight = 0;
    for (int id : result) {
        total_weight += edges[id].w;
    }

    cout << total_weight << ' ' << result.size() << '\n';
    for (int id : result) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }

    return 0;
}
