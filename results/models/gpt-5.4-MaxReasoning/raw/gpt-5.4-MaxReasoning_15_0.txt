use std::collections::VecDeque;
use std::io::{self, Read};

const MAX_SCALE: f64 = 10.0;
const LAYER_MARGIN: f64 = 1.0;
const GROUND_MARGIN: f64 = 1.0;
const ROOT_EPS: f64 = 1.0e-12;

#[derive(Clone, Copy)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn sub(self, other: Self) -> Self {
        Self {
            x: self.x - other.x,
            y: self.y - other.y,
        }
    }

    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y
    }

    fn cross(self, other: Self) -> f64 {
        self.x * other.y - self.y * other.x
    }

    fn norm2(self) -> f64 {
        self.dot(self)
    }
}

#[derive(Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn add(self, other: Self) -> Self {
        Self {
            x: self.x + other.x,
            y: self.y + other.y,
            z: self.z + other.z,
        }
    }

    fn sub(self, other: Self) -> Self {
        Self {
            x: self.x - other.x,
            y: self.y - other.y,
            z: self.z - other.z,
        }
    }

    fn mul(self, factor: f64) -> Self {
        Self {
            x: self.x * factor,
            y: self.y * factor,
            z: self.z * factor,
        }
    }

    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(self, other: Self) -> Self {
        Self {
            x: self.y * other.z - self.z * other.y,
            y: self.z * other.x - self.x * other.z,
            z: self.x * other.y - self.y * other.x,
        }
    }

    fn norm2(self) -> f64 {
        self.dot(self)
    }

    fn norm(self) -> f64 {
        self.norm2().sqrt()
    }

    fn normalize(self) -> Self {
        let len = self.norm();
        if len <= ROOT_EPS {
            Self {
                x: 0.0,
                y: 0.0,
                z: 1.0,
            }
        } else {
            self.mul(1.0 / len)
        }
    }
}

#[derive(Clone, Copy)]
struct Mat3 {
    m: [[f64; 3]; 3],
}

impl Mat3 {
    fn from_cols(c0: Vec3, c1: Vec3, c2: Vec3) -> Self {
        Self {
            m: [
                [c0.x, c1.x, c2.x],
                [c0.y, c1.y, c2.y],
                [c0.z, c1.z, c2.z],
            ],
        }
    }

    fn transpose(self) -> Self {
        let m = self.m;
        Self {
            m: [
                [m[0][0], m[1][0], m[2][0]],
                [m[0][1], m[1][1], m[2][1]],
                [m[0][2], m[1][2], m[2][2]],
            ],
        }
    }

    fn mul_mat(self, other: Self) -> Self {
        let mut out = [[0.0; 3]; 3];
        let a = self.m;
        let b = other.m;
        let mut r = 0;
        while r < 3 {
            let mut c = 0;
            while c < 3 {
                out[r][c] = a[r][0] * b[0][c] + a[r][1] * b[1][c] + a[r][2] * b[2][c];
                c += 1;
            }
            r += 1;
        }
        Self { m: out }
    }

    fn mul_vec3(self, v: Vec3) -> Vec3 {
        Vec3 {
            x: self.m[0][0] * v.x + self.m[0][1] * v.y + self.m[0][2] * v.z,
            y: self.m[1][0] * v.x + self.m[1][1] * v.y + self.m[1][2] * v.z,
            z: self.m[2][0] * v.x + self.m[2][1] * v.y + self.m[2][2] * v.z,
        }
    }
}

struct ReferenceGeom {
    ref_a: Vec3,
    edge_ab: f64,
    edge_ac: f64,
    edge_bc: f64,
    basis_t: Mat3,
    radius: f64,
}

struct Tetra {

}

fn triangle_area2(a: Vec2, b: Vec2, c: Vec2) -> f64 {
    b.sub(a).cross(c.sub(a))
}

fn signed_area2(poly: &[Vec2]) -> f64 {
    let n = poly.len();
    let mut acc = 0.0;
    let mut i = 0;
    while i < n {
        let j = (i + 1) % n;
        acc += poly[i].cross(poly[j]);
        i += 1;
    }
    acc
}

fn bbox_diag(poly: &[Vec2]) -> f64 {
    if poly.is_empty() {
        return 1.0;
    }
    let mut min_x = poly[0].x;
    let mut min_y = poly[0].y;
    let mut max_x = poly[0].x;
    let mut max_y = poly[0].y;
    for p in poly {
        min_x = min_x.min(p.x);
        min_y = min_y.min(p.y);
        max_x = max_x.max(p.x);
        max_y = max_y.max(p.y);
    }
    let dx = max_x - min_x;
    let dy = max_y - min_y;
    (dx * dx + dy * dy).sqrt().max(1.0)
}

fn point_in_triangle(p: Vec2, a: Vec2, b: Vec2, c: Vec2, eps: f64) -> bool {
    let c1 = b.sub(a).cross(p.sub(a));
    let c2 = c.sub(b).cross(p.sub(b));
    let c3 = a.sub(c).cross(p.sub(c));
    c1 >= -eps && c2 >= -eps && c3 >= -eps
}

fn sanitize_polygon(points: &[Vec2]) -> Vec<Vec2> {
    if points.is_empty() {
        return Vec::new();
    }
    let diag = bbox_diag(points);
    let len_eps = 1.0e-10 * diag;
    let len_eps2 = len_eps * len_eps;
    let area_eps = len_eps * diag;

    let mut filtered = Vec::with_capacity(points.len());
    for &p in points {
        let keep = match filtered.last() {
            Some(&last) => p.sub(last).norm2() > len_eps2,
            None => true,
        };
        if keep {
            filtered.push(p);
        }
    }

    if filtered.len() > 1 && filtered[0].sub(*filtered.last().unwrap()).norm2() <= len_eps2 {
        filtered.pop();
    }

    loop {
        let n = filtered.len();
        if n < 3 {
            break;
        }
        let mut changed = false;
        let mut next_points = Vec::with_capacity(n);
        let mut i = 0;
        while i < n {
            let prev = filtered[(i + n - 1) % n];
            let cur = filtered[i];
            let next = filtered[(i + 1) % n];
            let v1 = cur.sub(prev);
            let v2 = next.sub(cur);
            if v1.norm2() <= len_eps2 {
                changed = true;
                i += 1;
                continue;
            }
            if v1.cross(v2).abs() <= area_eps && v1.dot(v2) >= -len_eps2 {
                changed = true;
                i += 1;
                continue;
            }
            next_points.push(cur);
            i += 1;
        }
        filtered = next_points;
        if !changed {
            break;
        }
    }

    if filtered.len() >= 3 && signed_area2(&filtered) < 0.0 {
        filtered.reverse();
    }

    filtered
}

fn is_convex_polygon(poly: &[Vec2], eps: f64) -> bool {
    let n = poly.len();
    if n < 3 {
        return false;
    }
    let mut i = 0;
    while i < n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let c = poly[(i + 2) % n];
        if triangle_area2(a, b, c) <= eps {
            return false;
        }
        i += 1;
    }
    true
}

fn triangulate_convex_fan(n: usize) -> Vec<[usize; 3]> {
    let mut tris = Vec::with_capacity(n.saturating_sub(2));
    let mut i = 1;
    while i + 1 < n {
        tris.push([0, i, i + 1]);
        i += 1;
    }
    tris
}

fn is_reflex_vertex(i: usize, poly: &[Vec2], prev: &[usize], next: &[usize], eps: f64) -> bool {
    triangle_area2(poly[prev[i]], poly[i], poly[next[i]]) <= eps
}

fn is_ear_small(
    i: usize,
    poly: &[Vec2],
    prev: &[usize],
    next: &[usize],
    alive: &[bool],
    eps: f64,
) -> bool {
    let a = prev[i];
    let c = next[i];
    let pa = poly[a];
    let pb = poly[i];
    let pc = poly[c];
    if triangle_area2(pa, pb, pc) <= eps {
        return false;
    }
    let mut j = 0;
    while j < poly.len() {
        if alive[j] && j != a && j != i && j != c && point_in_triangle(poly[j], pa, pb, pc, eps) {
            return false;
        }
        j += 1;
    }
    true
}

fn is_ear_large(
    i: usize,
    poly: &[Vec2],
    prev: &[usize],
    next: &[usize],
    alive: &[bool],
    reflex: &[bool],
    eps: f64,
) -> bool {
    let a = prev[i];
    let c = next[i];
    let pa = poly[a];
    let pb = poly[i];
    let pc = poly[c];
    if triangle_area2(pa, pb, pc) <= eps {
        return false;
    }
    let mut j = 0;
    while j < poly.len() {
        if alive[j]
            && reflex[j]
            && j != a
            && j != i
            && j != c
            && point_in_triangle(poly[j], pa, pb, pc, eps)
        {
            return false;
        }
        j += 1;
    }
    true
}

fn clip_ear(
    i: usize,
    prev: &mut [usize],
    next: &mut [usize],
    alive: &mut [bool],
    remaining: &mut usize,
    out: &mut Vec<[usize; 3]>,
) -> (usize, usize) {
    let a = prev[i];
    let c = next[i];
    out.push([a, i, c]);
    alive[i] = false;
    next[a] = c;
    prev[c] = a;
    *remaining -= 1;
    (a, c)
}

fn triangulate_small_ear(poly: &[Vec2], eps: f64) -> Vec<[usize; 3]> {
    let n = poly.len();
    let mut tris = Vec::with_capacity(n.saturating_sub(2));
    let mut prev = vec![0usize; n];
    let mut next = vec![0usize; n];
    let mut alive = vec![true; n];
    let mut i = 0;
    while i < n {
        prev[i] = if i == 0 { n - 1 } else { i - 1 };
        next[i] = if i + 1 == n { 0 } else { i + 1 };
        i += 1;
    }

    let mut remaining = n;
    let mut cursor = 0usize;
    let mut guard = 0usize;
    let guard_limit = n.saturating_mul(n).saturating_mul(8).max(32);

    while remaining > 3 && guard < guard_limit {
        if !alive[cursor] {
            cursor = next[cursor];
            guard += 1;
            continue;
        }
        if is_ear_small(cursor, poly, &prev, &next, &alive, eps) {
            let (a, c) = clip_ear(
                cursor,
                &mut prev,
                &mut next,
                &mut alive,
                &mut remaining,
                &mut tris,
            );
            cursor = a;
            if !alive[cursor] {
                cursor = c;
            }
        } else {
            cursor = next[cursor];
        }
        guard += 1;
    }

    let mut final_indices = Vec::with_capacity(3);
    let mut k = 0;
    while k < n {
        if alive[k] {
            final_indices.push(k);
        }
        k += 1;
    }
    if final_indices.len() == 3 {
        let a = final_indices[0];
        let b = final_indices[1];
        let c = final_indices[2];
        if triangle_area2(poly[a], poly[b], poly[c]) > 0.0 {
            tris.push([a, b, c]);
        } else {
            tris.push([a, c, b]);
        }
    } else {
        tris.clear();
    }
    tris
}

fn triangulate_large_ear(poly: &[Vec2], eps: f64) -> Vec<[usize; 3]> {
    let n = poly.len();
    let mut tris = Vec::with_capacity(n.saturating_sub(2));
    let mut prev = vec![0usize; n];
    let mut next = vec![0usize; n];
    let mut alive = vec![true; n];
    let mut reflex = vec![false; n];
    let mut queue = VecDeque::new();

    let mut i = 0;
    while i < n {
        prev[i] = if i == 0 { n - 1 } else { i - 1 };
        next[i] = if i + 1 == n { 0 } else { i + 1 };
        i += 1;
    }

    let mut j = 0;
    while j < n {
        reflex[j] = is_reflex_vertex(j, poly, &prev, &next, eps);
        if !reflex[j] {
            queue.push_back(j);
        }
        j += 1;
    }

    let mut remaining = n;
    let mut guard = 0usize;
    let guard_limit = n.saturating_mul(n).saturating_mul(8).max(64);

    while remaining > 3 && guard < guard_limit {
        let mut clipped = false;
        let passes = queue.len().max(remaining);
        let mut pass = 0usize;
        while pass < passes {
            let maybe_idx = queue.pop_front();
            if let Some(idx) = maybe_idx {
                if alive[idx] {
                    reflex[idx] = is_reflex_vertex(idx, poly, &prev, &next, eps);
                    if !reflex[idx] && is_ear_large(idx, poly, &prev, &next, &alive, &reflex, eps) {
                        let (a, c) = clip_ear(
                            idx,
                            &mut prev,
                            &mut next,
                            &mut alive,
                            &mut remaining,
                            &mut tris,
                        );
                        reflex[a] = is_reflex_vertex(a, poly, &prev, &next, eps);
                        reflex[c] = is_reflex_vertex(c, poly, &prev, &next, eps);
                        queue.push_back(a);
                        queue.push_back(c);
                        clipped = true;
                        break;
                    }
                }
            } else {
                break;
            }
            pass += 1;
        }

        if !clipped {
            let mut idx = 0usize;
            while idx < n {
                if alive[idx] {
                    reflex[idx] = is_reflex_vertex(idx, poly, &prev, &next, eps);
                    if !reflex[idx] && is_ear_large(idx, poly, &prev, &next, &alive, &reflex, eps) {
                        let (a, c) = clip_ear(
                            idx,
                            &mut prev,
                            &mut next,
                            &mut alive,
                            &mut remaining,
                            &mut tris,
                        );
                        reflex[a] = is_reflex_vertex(a, poly, &prev, &next, eps);
                        reflex[c] = is_reflex_vertex(c, poly, &prev, &next, eps);
                        queue.push_back(a);
                        queue.push_back(c);
                        clipped = true;
                        break;
                    }
                }
                idx += 1;
            }
        }

        if !clipped {
            tris.clear();
            return tris;
        }

        guard += 1;
    }

    let mut final_indices = Vec::with_capacity(3);
    let mut k = 0;
    while k < n {
        if alive[k] {
            final_indices.push(k);
        }
        k += 1;
    }
    if final_indices.len() == 3 {
        let a = final_indices[0];
        let b = final_indices[1];
        let c = final_indices[2];
        if triangle_area2(poly[a], poly[b], poly[c]) > 0.0 {
            tris.push([a, b, c]);
        } else {
            tris.push([a, c, b]);
        }
    } else {
        tris.clear();
    }
    tris
}

fn face_basis(a: Vec3, b: Vec3, c: Vec3) -> Mat3 {
    let x = b.sub(a).normalize();
    let y_raw = c.sub(a);
    let y = y_raw.sub(x.mul(y_raw.dot(x))).normalize();
    let z = x.cross(y);
    Mat3::from_cols(x, y, z)
}

fn quat_from_mat3(m: Mat3) -> [f64; 4] {
    let tr = m.m[0][0] + m.m[1][1] + m.m[2][2];
    let mut q = [0.0; 4];
    if tr > 0.0 {
        let s = (tr + 1.0).sqrt() * 2.0;
        q[0] = 0.25 * s;
        q[1] = (m.m[2][1] - m.m[1][2]) / s;
        q[2] = (m.m[0][2] - m.m[2][0]) / s;
        q[3] = (m.m[1][0] - m.m[0][1]) / s;
    } else if m.m[0][0] > m.m[1][1] && m.m[0][0] > m.m[2][2] {
        let s = (1.0 + m.m[0][0] - m.m[1][1] - m.m[2][2]).sqrt() * 2.0;
        q[0] = (m.m[2][1] - m.m[1][2]) / s;
        q[1] = 0.25 * s;
        q[2] = (m.m[0][1] + m.m[1][0]) / s;
        q[3] = (m.m[0][2] + m.m[2][0]) / s;
    } else if m.m[1][1] > m.m[2][2] {
        let s = (1.0 + m.m[1][1] - m.m[0][0] - m.m[2][2]).sqrt() * 2.0;
        q[0] = (m.m[0][2] - m.m[2][0]) / s;
        q[1] = (m.m[0][1] + m.m[1][0]) / s;
        q[2] = 0.25 * s;
        q[3] = (m.m[1][2] + m.m[2][1]) / s;
    } else {
        let s = (1.0 + m.m[2][2] - m.m[0][0] - m.m[1][1]).sqrt() * 2.0;
        q[0] = (m.m[1][0] - m.m[0][1]) / s;
        q[1] = (m.m[0][2] + m.m[2][0]) / s;
        q[2] = (m.m[1][2] + m.m[2][1]) / s;
        q[3] = 0.25 * s;
    }
    let len = (q[0] * q[0] + q[1] * q[1] + q[2] * q[2] + q[3] * q[3]).sqrt();
    if len > 0.0 {
        let inv = 1.0 / len;
        q[0] *= inv;
        q[1] *= inv;
        q[2] *= inv;
        q[3] *= inv;
    } else {
        q = [1.0, 0.0, 0.0, 0.0];
    }
    if q[0] < 0.0 {
        q[0] = -q[0];
        q[1] = -q[1];
        q[2] = -q[2];
        q[3] = -q[3];
    }
    q
}

fn build_reference() -> ReferenceGeom {
    let ref_verts = [
        Vec3 {
            x: 1.0,
            y: 0.0,
            z: -0.707,
        },
        Vec3 {
            x: -1.0,
            y: 0.0,
            z: -0.707,
        },
        Vec3 {
            x: 0.0,
            y: 1.0,
            z: 0.707,
        },
        Vec3 {
            x: 0.0,
            y: -1.0,
            z: 0.707,
        },
    ];

    let basis = face_basis(ref_verts[0], ref_verts[1], ref_verts[2]);
    let basis_t = basis.transpose();
    let edge_ab = ref_verts[1].sub(ref_verts[0]).norm();
    let edge_ac = ref_verts[2].sub(ref_verts[0]).norm();
    let edge_bc = ref_verts[2].sub(ref_verts[1]).norm();
    let mut radius = 0.0_f64;
    for v in ref_verts {
        radius = radius.max(v.norm());
    }

    ReferenceGeom {
        ref_a: ref_verts[0],
        edge_ab,
        edge_ac,
        edge_bc,
        basis_t,
        radius,
    }
}

fn solve_face_projection(
    a2: Vec2,
    b2: Vec2,
    c2: Vec2,
    sun: Vec3,
    reference: &ReferenceGeom,
) -> Option<([Vec3; 3], f64)> {
    let ab2 = b2.sub(a2);
    let ac2 = c2.sub(a2);
    let bc2 = c2.sub(b2);

    let proj_dot = |v: Vec2| v.x * sun.x + v.y * sun.y;

    let a = proj_dot(ab2);
    let b = proj_dot(ac2);
    let c = proj_dot(bc2);

    let ab_perp = (ab2.norm2() - a * a).max(0.0);
    let ac_perp = (ac2.norm2() - b * b).max(0.0);
    let bc_perp = (bc2.norm2() - c * c).max(0.0);

    let scale_min = (ab_perp.sqrt() / reference.edge_ab)
        .max(ac_perp.sqrt() / reference.edge_ac)
        .max(bc_perp.sqrt() / reference.edge_bc);

    let mut best: Option<(f64, f64, f64, f64)> = None;
    let signs = [-1.0_f64, 1.0_f64];

    for &s1 in &signs {
        for &s2 in &signs {
            for &s3 in &signs {
                let f = |scale: f64| -> f64 {
                    let p = (scale * scale * reference.edge_ab * reference.edge_ab - ab_perp)
                        .max(0.0)
                        .sqrt();
                    let q = (scale * scale * reference.edge_ac * reference.edge_ac - ac_perp)
                        .max(0.0)
                        .sqrt();
                    let r = (scale * scale * reference.edge_bc * reference.edge_bc - bc_perp)
                        .max(0.0)
                        .sqrt();
                    s2 * q - s1 * p - s3 * r
                };

                let mut lo = scale_min;
                let mut hi = scale_min;
                let mut flo = f(lo);
                let mut found = flo.abs() <= ROOT_EPS * (1.0 + scale_min);

                if !found {
                    hi = (scale_min + 1.0).max(1.0);
                    let mut fhi = f(hi);
                    let mut step = 0usize;
                    while step < 96 {
                        if flo * fhi <= 0.0 {
                            found = true;
                            break;
                        }
                        hi *= 2.0;
                        fhi = f(hi);
                        step += 1;
                    }
                }

                if !found {
                    continue;
                }

                let mut iter = 0usize;
                while iter < 96 {
                    let mid = 0.5 * (lo + hi);
                    let fm = f(mid);
                    if flo * fm <= 0.0 {
                        hi = mid;
                    } else {
                        lo = mid;
                        flo = fm;
                    }
                    iter += 1;
                }

                let scale = 0.5 * (lo + hi);
                let p = (scale * scale * reference.edge_ab * reference.edge_ab - ab_perp)
                    .max(0.0)
                    .sqrt();
                let q = (scale * scale * reference.edge_ac * reference.edge_ac - ac_perp)
                    .max(0.0)
                    .sqrt();
                let alpha = -a + s1 * p;
                let beta = -b + s2 * q;

                let a3 = Vec3 {
                    x: a2.x,
                    y: a2.y,
                    z: 0.0,
                };
                let b3 = Vec3 {
                    x: b2.x,
                    y: b2.y,
                    z: 0.0,
                }
                .add(sun.mul(alpha));
                let c3 = Vec3 {
                    x: c2.x,
                    y: c2.y,
                    z: 0.0,
                }
                .add(sun.mul(beta));

                let err_ab = (b3.sub(a3).norm() - scale * reference.edge_ab).abs();
                let err_ac = (c3.sub(a3).norm() - scale * reference.edge_ac).abs();
                let err_bc = (c3.sub(b3).norm() - scale * reference.edge_bc).abs();
                let err = err_ab.max(err_ac.max(err_bc));
                if err > 1.0e-7 * (1.0 + scale) {
                    continue;
                }

                let span_high = alpha.max(beta.max(0.0));
                let span_low = alpha.min(beta.min(0.0));
                let span = span_high - span_low;

                match best {
                    Some((best_span, best_scale, _, _)) if span > best_span + ROOT_EPS => {}
                    Some((best_span, best_scale, _, _))
                        if (span - best_span).abs() <= ROOT_EPS && scale >= best_scale =>
                    {}
                    _ => {
                        best = Some((span, scale, alpha, beta));
                    }
                }
            }
        }
    }

    best.map(|(_, scale, alpha, beta)| {
        let a3 = Vec3 {
            x: a2.x,
            y: a2.y,
            z: 0.0,
        };
        let b3 = Vec3 {
            x: b2.x,
            y: b2.y,
            z: 0.0,
        }
        .add(sun.mul(alpha));
        let c3 = Vec3 {
            x: c2.x,
            y: c2.y,
            z: 0.0,
        }
        .add(sun.mul(beta));
        ([a3, b3, c3], scale)
    })
}

fn build_tetra_from_face(face: [Vec3; 3], scale: f64, reference: &ReferenceGeom) -> Tetra {
    let world_basis = face_basis(face[0], face[1], face[2]);
    let rotation = world_basis.mul_mat(reference.basis_t);
    let rotated_ref_a = rotation.mul_vec3(reference.ref_a);
    let center = face[0].sub(rotated_ref_a.mul(scale));
    let quat = quat_from_mat3(rotation);
    Tetra {
        center,
        quat,
        scale,
        radius: reference.radius * scale,
    }
}

fn bary3(a: Vec3, b: Vec3, c: Vec3, u: f64, v: f64) -> Vec3 {
    a.add(b.sub(a).mul(u)).add(c.sub(a).mul(v))
}

fn emit_face_cover(face: [Vec3; 3], scale: f64, reference: &ReferenceGeom, out: &mut Vec<Tetra>) {
    if scale <= MAX_SCALE + 1.0e-9 {
        out.push(build_tetra_from_face(face, scale, reference));
        return;
    }

    let pieces = (scale / MAX_SCALE).ceil() as usize;
    let pieces_f = pieces as f64;
    let child_scale = scale / pieces_f;

    let mut i = 0usize;
    while i < pieces {
        let mut j = 0usize;
        while j + i < pieces {
            let u0 = i as f64 / pieces_f;
            let v0 = j as f64 / pieces_f;
            let u1 = (i + 1) as f64 / pieces_f;
            let v1 = (j + 1) as f64 / pieces_f;

            let q00 = bary3(face[0], face[1], face[2], u0, v0);
            let q10 = bary3(face[0], face[1], face[2], u1, v0);
            let q01 = bary3(face[0], face[1], face[2], u0, v1);
            out.push(build_tetra_from_face([q00, q10, q01], child_scale, reference));

            if i + j + 1 < pieces {
                let q11 = bary3(face[0], face[1], face[2], u1, v1);
                out.push(build_tetra_from_face([q11, q01, q10], child_scale, reference));
            }

            j += 1;
        }
        i += 1;
    }
}

fn cover_triangle(
    a: Vec2,
    b: Vec2,
    c: Vec2,
    sun: Vec3,
    reference: &ReferenceGeom,
    out: &mut Vec<Tetra>,
) {
    if triangle_area2(a, b, c).abs() <= 1.0e-18 {
        return;
    }
    if let Some((face, scale)) = solve_face_projection(a, b, c, sun, reference) {
        emit_face_cover(face, scale, reference, out);
    }
}

fn separate_tetrahedra(tetras: &mut [Tetra], sun: Vec3) {
    let mut cursor = 0.0_f64;
    for tetra in &mut *tetras {
        cursor += tetra.radius + LAYER_MARGIN;
        let delta = cursor - tetra.center.dot(sun);
        tetra.center = tetra.center.add(sun.mul(delta));
        cursor += tetra.radius;
    }

    if sun.z.abs() > ROOT_EPS {
        let mut min_z = f64::INFINITY;
        for tetra in &*tetras {
            min_z = min_z.min(tetra.center.z - tetra.radius);
        }
        if min_z < GROUND_MARGIN {
            let shift = (GROUND_MARGIN - min_z) / sun.z;
            for tetra in &mut *tetras {
                tetra.center = tetra.center.add(sun.mul(shift));
            }
        }
    }
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        println!("0");
        return;
    }

    let mut it = input.split_whitespace();
    let nv = match it.next().and_then(|v| v.parse::<usize>().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut polygon = Vec::with_capacity(nv);
    let mut idx = 0usize;
    while idx < nv {
        let x = match it.next().and_then(|v| v.parse::<f64>().ok()) {
            Some(v) => v,
            None => {
                println!("0");
                return;
            }
        };
        let y = match it.next().and_then(|v| v.parse::<f64>().ok()) {
            Some(v) => v,
            None => {
                println!("0");
                return;
            }
        };
        polygon.push(Vec2 { x, y });
        idx += 1;
    }

    let sx = match it.next().and_then(|v| v.parse::<f64>().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let sy = match it.next().and_then(|v| v.parse::<f64>().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let sz = match it.next().and_then(|v| v.parse::<f64>().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut sun = Vec3 { x: sx, y: sy, z: sz };
    if sun.norm2() <= ROOT_EPS || sun.z.abs() <= ROOT_EPS {
        sun = Vec3 {
            x: 0.0,
            y: 0.0,
            z: -1.0,
        };
    } else {
        sun = sun.normalize();
    }

    let polygon = sanitize_polygon(&polygon);
    if polygon.len() < 3 || signed_area2(&polygon).abs() <= 1.0e-18 {
        println!("0");
        return;
    }

    let diag = bbox_diag(&polygon);
    let area_eps = 1.0e-10 * diag * diag;

    let mut triangles = if is_convex_polygon(&polygon, area_eps) {
        triangulate_convex_fan(polygon.len())
    } else if polygon.len() <= 96 {
        let small = triangulate_small_ear(&polygon, area_eps);
        if small.is_empty() {
            triangulate_large_ear(&polygon, area_eps)
        } else {
            small
        }
    } else {
        let large = triangulate_large_ear(&polygon, area_eps);
        if large.is_empty() {
            triangulate_small_ear(&polygon, area_eps)
        } else {
            large
        }
    };

    if triangles.is_empty() {
        println!("0");
        return;
    }

    let reference = build_reference();
    let mut tetras = Vec::new();
    for tri in triangles.drain(..) {
        cover_triangle(
            polygon[tri[0]],
            polygon[tri[1]],
            polygon[tri[2]],
            sun,
            &reference,
            &mut tetras,
        );
    }

    separate_tetrahedra(&mut tetras, sun);

    println!("{}", tetras.len());
    for tetra in &tetras {
        println!(
            "{:.12} {:.12} {:.12} {:.12} {:.12} {:.12} {:.12} {:.12}",
            tetra.center.x,
            tetra.center.y,
            tetra.center.z,
            tetra.quat[0],
            tetra.quat[1],
            tetra.quat[2],
            tetra.quat[3],
            tetra.scale
        );
    }
}
