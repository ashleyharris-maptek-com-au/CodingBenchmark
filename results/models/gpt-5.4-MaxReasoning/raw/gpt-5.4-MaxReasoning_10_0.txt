struct Rect {
    int id = -1;
    int w = 0;
    int h = 0;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Solution {
    bool feasible = false;
    int boards = 0;
    long long secondary = 0;
    std::vector<Placement> placements;
};

static bool better_solution(const Solution& lhs, const Solution& rhs) {
    if (!lhs.feasible) {
        return false;
    }
    if (!rhs.feasible) {
        return true;
    }
    if (lhs.boards != rhs.boards) {
        return lhs.boards < rhs.boards;
    }
    if (lhs.secondary != rhs.secondary) {
        return lhs.secondary < rhs.secondary;
    }
    return false;
}

enum class OrderMode {
    MaxSideArea,
    AreaMaxSide
};

static std::vector<int> build_order(const std::vector<Rect>& rects, OrderMode mode) {
    std::vector<int> order(rects.size());
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int a, int b) {
        const Rect& ra = rects[a];
        const Rect& rb = rects[b];
        const int a_long = std::max(ra.w, ra.h);
        const int b_long = std::max(rb.w, rb.h);
        const int a_short = std::min(ra.w, ra.h);
        const int b_short = std::min(rb.w, rb.h);
        const long long a_area = 1LL * ra.w * ra.h;
        const long long b_area = 1LL * rb.w * rb.h;

        if (mode == OrderMode::MaxSideArea) {
            if (a_long != b_long) {
                return a_long > b_long;
            }
            if (a_area != b_area) {
                return a_area > b_area;
            }
            if (a_short != b_short) {
                return a_short > b_short;
            }
        } else {
            if (a_area != b_area) {
                return a_area > b_area;
            }
            if (a_long != b_long) {
                return a_long > b_long;
            }
            if (a_short != b_short) {
                return a_short > b_short;
            }
        }

        return ra.id < rb.id;
    });
    return order;
}

class SegmentTreeMax {
public:
    explicit SegmentTreeMax(int n = 0) {
        init(n);
    }

    void init(int n) {
        n_ = n;
        tree_.assign(n_ > 0 ? 4 * n_ : 1, 0);
    }

    void update(int index, int value) {
        if (n_ == 0) {
            return;
        }
        update_impl(1, 0, n_ - 1, index, value);
    }

    int find_first_at_least(int from_index, int min_value) const {
        if (n_ == 0 || from_index >= n_ || tree_[1] < min_value) {
            return -1;
        }
        return find_impl(1, 0, n_ - 1, from_index, min_value);
    }

private:
    int n_ = 0;
    std::vector<int> tree_;

    void update_impl(int node, int left, int right, int index, int value) {
        if (left == right) {
            tree_[node] = value;
            return;
        }
        const int mid = left + (right - left) / 2;
        if (index <= mid) {
            update_impl(node * 2, left, mid, index, value);
        } else {
            update_impl(node * 2 + 1, mid + 1, right, index, value);
        }
        tree_[node] = std::max(tree_[node * 2], tree_[node * 2 + 1]);
    }

    int find_impl(int node, int left, int right, int from_index, int min_value) const {
        if (right < from_index || tree_[node] < min_value) {
            return -1;
        }
        if (left == right) {
            return left;
        }
        const int mid = left + (right - left) / 2;
        const int left_result = find_impl(node * 2, left, mid, from_index, min_value);
        if (left_result != -1) {
            return left_result;
        }
        return find_impl(node * 2 + 1, mid + 1, right, from_index, min_value);
    }
};

struct ShelfBoard {
    int used_height = 0;
};

struct ShelfState {
    int board = -1;
    int y = 0;
    int height = 0;
    int used_width = 0;
};

struct ShelfCandidate {
    bool valid = false;
    bool use_existing_shelf = false;
    bool use_existing_board = false;
    int shelf_id = -1;
    int board_id = -1;
    int placed_w = 0;
    int placed_h = 0;
    int rotated = 0;
    std::array<int, 6> score = {
        std::numeric_limits<int>::max(),
        std::numeric_limits<int>::max(),
        std::numeric_limits<int>::max(),
        std::numeric_limits<int>::max(),
        std::numeric_limits<int>::max(),
        std::numeric_limits<int>::max()
    };
};

static bool better_candidate(const ShelfCandidate& lhs, const ShelfCandidate& rhs) {
    if (!lhs.valid) {
        return false;
    }
    if (!rhs.valid) {
        return true;
    }
    return lhs.score < rhs.score;
}

static Solution pack_with_shelves(const std::vector<Rect>& rects, int board_w, int board_h, OrderMode mode) {
    Solution solution;
    solution.feasible = true;
    solution.placements.assign(rects.size(), Placement{});

    if (rects.empty()) {
        return solution;
    }

    std::vector<int> heights;
    heights.reserve(rects.size() * 2);
    for (const Rect& rect : rects) {
        heights.push_back(rect.h);
        heights.push_back(rect.w);
    }
    std::sort(heights.begin(), heights.end());
    heights.erase(std::unique(heights.begin(), heights.end()), heights.end());

    SegmentTreeMax tree(static_cast<int>(heights.size()));
    std::vector<std::set<std::pair<int, int>>> bucket_by_height(heights.size());
    std::set<std::pair<int, int>> boards_by_remaining_height;
    std::vector<ShelfBoard> boards;
    std::vector<ShelfState> shelves;
    boards.reserve(rects.size());
    shelves.reserve(rects.size());

    auto update_bucket = [&](int height_idx) {
        const auto& bucket = bucket_by_height[height_idx];
        const int max_remaining = bucket.empty() ? 0 : bucket.rbegin()->first;
        tree.update(height_idx, max_remaining);
    };

    auto remove_shelf_capacity = [&](int shelf_id) {
        const ShelfState& shelf = shelves[shelf_id];
        const int remaining = board_w - shelf.used_width;
        if (remaining <= 0) {
            return;
        }
        const int height_idx = static_cast<int>(std::lower_bound(heights.begin(), heights.end(), shelf.height) - heights.begin());
        auto& bucket = bucket_by_height[height_idx];
        auto it = bucket.find({remaining, shelf_id});
        if (it != bucket.end()) {
            bucket.erase(it);
            update_bucket(height_idx);
        }
    };

    auto add_shelf_capacity = [&](int shelf_id) {
        const ShelfState& shelf = shelves[shelf_id];
        const int remaining = board_w - shelf.used_width;
        if (remaining <= 0) {
            return;
        }
        const int height_idx = static_cast<int>(std::lower_bound(heights.begin(), heights.end(), shelf.height) - heights.begin());
        auto& bucket = bucket_by_height[height_idx];
        bucket.insert({remaining, shelf_id});
        update_bucket(height_idx);
    };

    auto evaluate_orientation = [&](int rect_w, int rect_h, int rotated_flag) {
        ShelfCandidate best;
        if (rect_w > board_w || rect_h > board_h) {
            return best;
        }

        const int start_idx = static_cast<int>(std::lower_bound(heights.begin(), heights.end(), rect_h) - heights.begin());
        if (start_idx < static_cast<int>(heights.size())) {
            const int height_idx = tree.find_first_at_least(start_idx, rect_w);
            if (height_idx != -1) {
                const auto& bucket = bucket_by_height[height_idx];
                auto it = bucket.lower_bound({rect_w, std::numeric_limits<int>::min()});
                if (it != bucket.end()) {
                    const int shelf_id = it->second;
                    const ShelfState& shelf = shelves[shelf_id];
                    const int board_id = shelf.board;
                    const int remaining_width = it->first;
                    const int remaining_height = board_h - boards[board_id].used_height;

                    ShelfCandidate candidate;
                    candidate.valid = true;
                    candidate.use_existing_shelf = true;
                    candidate.use_existing_board = true;
                    candidate.shelf_id = shelf_id;
                    candidate.board_id = board_id;
                    candidate.placed_w = rect_w;
                    candidate.placed_h = rect_h;
                    candidate.rotated = rotated_flag;
                    candidate.score = {
                        0,
                        0,
                        shelf.height - rect_h,
                        remaining_width - rect_w,
                        remaining_height,
                        shelf_id
                    };
                    best = candidate;
                }
            }
        }

        auto board_it = boards_by_remaining_height.lower_bound({rect_h, std::numeric_limits<int>::min()});
        if (board_it != boards_by_remaining_height.end()) {
            const int remaining_height = board_it->first;
            const int board_id = board_it->second;

            ShelfCandidate candidate;
            candidate.valid = true;
            candidate.use_existing_shelf = false;
            candidate.use_existing_board = true;
            candidate.board_id = board_id;
            candidate.placed_w = rect_w;
            candidate.placed_h = rect_h;
            candidate.rotated = rotated_flag;
            candidate.score = {
                0,
                1,
                remaining_height - rect_h,
                board_w - rect_w,
                boards[board_id].used_height,
                board_id
            };
            if (better_candidate(candidate, best)) {
                best = candidate;
            }
        }

        ShelfCandidate new_board_candidate;
        new_board_candidate.valid = true;
        new_board_candidate.use_existing_shelf = false;
        new_board_candidate.use_existing_board = false;
        new_board_candidate.placed_w = rect_w;
        new_board_candidate.placed_h = rect_h;
        new_board_candidate.rotated = rotated_flag;
        new_board_candidate.score = {
            1,
            1,
            board_h - rect_h,
            board_w - rect_w,
            0,
            0
        };
        if (better_candidate(new_board_candidate, best)) {
            best = new_board_candidate;
        }

        return best;
    };

    const std::vector<int> order = build_order(rects, mode);
    for (int order_index : order) {
        const Rect& rect = rects[order_index];

        ShelfCandidate best = evaluate_orientation(rect.w, rect.h, 0);
        if (rect.w != rect.h) {
            ShelfCandidate rotated = evaluate_orientation(rect.h, rect.w, 1);
            if (better_candidate(rotated, best)) {
                best = rotated;
            }
        }

        if (!best.valid) {
            solution.feasible = false;
            return solution;
        }

        int board_id = best.board_id;
        int x = 0;
        int y = 0;

        if (best.use_existing_shelf) {
            ShelfState& shelf = shelves[best.shelf_id];
            remove_shelf_capacity(best.shelf_id);
            board_id = shelf.board;
            x = shelf.used_width;
            y = shelf.y;
            shelf.used_width += best.placed_w;
            add_shelf_capacity(best.shelf_id);
        } else {
            if (best.use_existing_board) {
                auto it = boards_by_remaining_height.find({board_h - boards[best.board_id].used_height, best.board_id});
                if (it != boards_by_remaining_height.end()) {
                    boards_by_remaining_height.erase(it);
                }
                board_id = best.board_id;
            } else {
                board_id = static_cast<int>(boards.size());
                boards.push_back(ShelfBoard{});
            }

            ShelfBoard& board = boards[board_id];
            y = board.used_height;
            x = 0;
            board.used_height += best.placed_h;

            shelves.push_back(ShelfState{board_id, y, best.placed_h, best.placed_w});
            add_shelf_capacity(static_cast<int>(shelves.size()) - 1);

            const int remaining_height = board_h - board.used_height;
            if (remaining_height > 0) {
                boards_by_remaining_height.insert({remaining_height, board_id});
            }
        }

        solution.placements[rect.id] = Placement{board_id, x, y, best.rotated};
    }

    solution.boards = static_cast<int>(boards.size());
    solution.secondary = static_cast<long long>(shelves.size());
    return solution;
}

struct FreeRect {
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
};

struct MaxRectChoice {
    bool valid = false;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int rotated = 0;
    int short_fit = std::numeric_limits<int>::max();
    int long_fit = std::numeric_limits<int>::max();
    int area_fit = std::numeric_limits<int>::max();
};

static bool better_maxrect_choice(const MaxRectChoice& lhs, const MaxRectChoice& rhs) {
    if (!lhs.valid) {
        return false;
    }
    if (!rhs.valid) {
        return true;
    }
    return std::tie(lhs.short_fit, lhs.long_fit, lhs.area_fit, lhs.y, lhs.x) <
           std::tie(rhs.short_fit, rhs.long_fit, rhs.area_fit, rhs.y, rhs.x);
}

class MaxRectsBoard {
public:
    MaxRectsBoard(int width, int height) : width_(width), height_(height) {
        free_rects_.push_back({0, 0, width_, height_});
    }

    MaxRectChoice find_best(int rect_w, int rect_h) const {
        MaxRectChoice best;
        for (const FreeRect& free_rect : free_rects_) {
            evaluate_fit(free_rect, rect_w, rect_h, 0, best);
            if (rect_w != rect_h) {
                evaluate_fit(free_rect, rect_h, rect_w, 1, best);
            }
        }
        return best;
    }

    void place(const MaxRectChoice& choice) {
        std::vector<FreeRect> next;
        next.reserve(free_rects_.size() * 2 + 4);
        const FreeRect used{choice.x, choice.y, choice.w, choice.h};

        for (const FreeRect& free_rect : free_rects_) {
            if (!intersects(free_rect, used)) {
                next.push_back(free_rect);
                continue;
            }

            if (used.x > free_rect.x) {
                next.push_back({free_rect.x, free_rect.y, used.x - free_rect.x, free_rect.h});
            }
            if (used.x + used.w < free_rect.x + free_rect.w) {
                next.push_back({
                    used.x + used.w,
                    free_rect.y,
                    free_rect.x + free_rect.w - (used.x + used.w),
                    free_rect.h
                });
            }
            if (used.y > free_rect.y) {
                next.push_back({free_rect.x, free_rect.y, free_rect.w, used.y - free_rect.y});
            }
            if (used.y + used.h < free_rect.y + free_rect.h) {
                next.push_back({
                    free_rect.x,
                    used.y + used.h,
                    free_rect.w,
                    free_rect.y + free_rect.h - (used.y + used.h)
                });
            }
        }

        prune(next);
        free_rects_.swap(next);
    }

    std::size_t free_count() const {
        return free_rects_.size();
    }

private:
    int width_ = 0;
    int height_ = 0;
    std::vector<FreeRect> free_rects_;

    static bool intersects(const FreeRect& a, const FreeRect& b) {
        return !(a.x >= b.x + b.w || a.x + a.w <= b.x || a.y >= b.y + b.h || a.y + a.h <= b.y);
    }

    static bool contains(const FreeRect& outer, const FreeRect& inner) {
        return outer.x <= inner.x && outer.y <= inner.y &&
               outer.x + outer.w >= inner.x + inner.w &&
               outer.y + outer.h >= inner.y + inner.h;
    }

    static void prune(std::vector<FreeRect>& rects) {
        std::vector<char> removed(rects.size(), 0);
        for (std::size_t i = 0; i < rects.size(); ++i) {
            if (removed[i] || rects[i].w <= 0 || rects[i].h <= 0) {
                removed[i] = 1;
                continue;
            }
            for (std::size_t j = i + 1; j < rects.size(); ++j) {
                if (removed[j] || rects[j].w <= 0 || rects[j].h <= 0) {
                    removed[j] = 1;
                    continue;
                }
                if (contains(rects[i], rects[j])) {
                    removed[j] = 1;
                } else if (contains(rects[j], rects[i])) {
                    removed[i] = 1;
                    break;
                }
            }
        }

        std::vector<FreeRect> compacted;
        compacted.reserve(rects.size());
        for (std::size_t i = 0; i < rects.size(); ++i) {
            if (!removed[i] && rects[i].w > 0 && rects[i].h > 0) {
                compacted.push_back(rects[i]);
            }
        }
        rects.swap(compacted);
    }

    static void evaluate_fit(const FreeRect& free_rect, int rect_w, int rect_h, int rotated, MaxRectChoice& best) {
        if (rect_w > free_rect.w || rect_h > free_rect.h) {
            return;
        }
        const int leftover_h = free_rect.h - rect_h;
        const int leftover_w = free_rect.w - rect_w;
        MaxRectChoice candidate;
        candidate.valid = true;
        candidate.x = free_rect.x;
        candidate.y = free_rect.y;
        candidate.w = rect_w;
        candidate.h = rect_h;
        candidate.rotated = rotated;
        candidate.short_fit = std::min(leftover_h, leftover_w);
        candidate.long_fit = std::max(leftover_h, leftover_w);
        candidate.area_fit = free_rect.w * free_rect.h - rect_w * rect_h;
        if (better_maxrect_choice(candidate, best)) {
            best = candidate;
        }
    }
};

static Solution pack_with_maxrects(const std::vector<Rect>& rects, int board_w, int board_h, OrderMode mode) {
    Solution solution;
    solution.feasible = true;
    solution.placements.assign(rects.size(), Placement{});

    if (rects.empty()) {
        return solution;
    }

    const std::vector<int> order = build_order(rects, mode);
    std::vector<MaxRectsBoard> boards;
    boards.reserve(rects.size());

    for (int order_index : order) {
        const Rect& rect = rects[order_index];

        MaxRectChoice best_choice;
        int best_board = -1;
        for (int board_id = 0; board_id < static_cast<int>(boards.size()); ++board_id) {
            MaxRectChoice choice = boards[board_id].find_best(rect.w, rect.h);
            if (better_maxrect_choice(choice, best_choice)) {
                best_choice = choice;
                best_board = board_id;
            }
        }

        if (best_board == -1) {
            boards.emplace_back(board_w, board_h);
            best_board = static_cast<int>(boards.size()) - 1;
            best_choice = boards.back().find_best(rect.w, rect.h);
        }

        if (!best_choice.valid) {
            solution.feasible = false;
            return solution;
        }

        boards[best_board].place(best_choice);
        solution.placements[rect.id] = Placement{
            best_board,
            best_choice.x,
            best_choice.y,
            best_choice.rotated
        };
    }

    solution.boards = static_cast<int>(boards.size());
    long long free_rects = 0;
    for (const MaxRectsBoard& board : boards) {
        free_rects += static_cast<long long>(board.free_count());
    }
    solution.secondary = free_rects;
    return solution;
}

static Solution pack_shelves_with_axis_mode(const std::vector<Rect>& original_rects,
                                            int board_w,
                                            int board_h,
                                            OrderMode mode,
                                            bool transposed_axes) {
    std::vector<Rect> transformed = original_rects;
    int packed_board_w = board_w;
    int packed_board_h = board_h;

    if (transposed_axes) {
        packed_board_w = board_h;
        packed_board_h = board_w;
        for (Rect& rect : transformed) {
            std::swap(rect.w, rect.h);
        }
    }

    Solution packed = pack_with_shelves(transformed, packed_board_w, packed_board_h, mode);
    if (!packed.feasible || !transposed_axes) {
        return packed;
    }

    for (Placement& placement : packed.placements) {
        const int old_x = placement.x;
        placement.x = placement.y;
        placement.y = old_x;
    }

    return packed;
}

static Solution solve(const std::vector<Rect>& rects, int board_w, int board_h) {
    Solution best;
    best.feasible = false;

    for (const Rect& rect : rects) {
        const bool fits = (rect.w <= board_w && rect.h <= board_h) ||
                          (rect.h <= board_w && rect.w <= board_h);
        if (!fits) {
            return best;
        }
    }

    const std::array<OrderMode, 2> order_modes = {OrderMode::MaxSideArea, OrderMode::AreaMaxSide};
    for (OrderMode mode : order_modes) {
        Solution direct = pack_shelves_with_axis_mode(rects, board_w, board_h, mode, false);
        if (better_solution(direct, best)) {
            best = std::move(direct);
        }

        Solution transposed = pack_shelves_with_axis_mode(rects, board_w, board_h, mode, true);
        if (better_solution(transposed, best)) {
            best = std::move(transposed);
        }
    }

    if (rects.size() <= 2500) {
        for (OrderMode mode : order_modes) {
            Solution exactish = pack_with_maxrects(rects, board_w, board_h, mode);
            if (better_solution(exactish, best)) {
                best = std::move(exactish);
            }
        }
    }

    return best;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    int board_w = 0;
    int board_h = 0;
    if (!(std::cin >> n >> board_w >> board_h)) {
        return 0;
    }

    std::vector<Rect> rects;
    rects.reserve(n);
    for (int i = 0; i < n; ++i) {
        Rect rect;
        rect.id = i;
        std::cin >> rect.w >> rect.h;
        rects.push_back(rect);
    }

    const Solution solution = solve(rects, board_w, board_h);
    if (!solution.feasible) {
        std::cout << -1 << '\n';
        return 0;
    }

    std::cout << solution.boards << '\n';
    for (const Placement& placement : solution.placements) {
        std::cout << placement.board << ' '
                  << placement.x << ' '
                  << placement.y << ' '
                  << placement.rotated << '\n';
    }
    return 0;
}
