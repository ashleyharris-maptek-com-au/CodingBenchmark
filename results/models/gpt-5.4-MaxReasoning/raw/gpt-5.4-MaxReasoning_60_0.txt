const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = input.trim() ? JSON.parse(input) : {};
const tasks = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasks.length;
const machineCount = Math.max(1, toInt(data.machineCount, 1));
const INF = Number.POSITIVE_INFINITY;

if (n === 0) {
  process.stdout.write('{"schedule":[]}');
  process.exit(0);
}

const ids = new Array(n);
const durations = new Float64Array(n);
const weights = new Float64Array(n);
const deadlines = new Float64Array(n);
const succs = Array.from({ length: n }, () => []);
const indeg = new Int32Array(n);
const idToIndex = new Map();

for (let i = 0; i < n; i++) {
  ids[i] = tasks[i].id;
  idToIndex.set(ids[i], i);
}

for (let i = 0; i < n; i++) {
  const task = tasks[i] || {};
  durations[i] = toNonNegativeInt(task.duration, 0);
  weights[i] = toFiniteNumber(task.weight, 1);
  deadlines[i] = Number.isFinite(task.deadline) ? Math.floor(task.deadline) : INF;
  const deps = Array.isArray(task.deps) ? task.deps : [];
  for (let j = 0; j < deps.length; j++) {
    const depIndex = idToIndex.get(deps[j]);
    if (depIndex === undefined || depIndex === i) continue;
    succs[depIndex].push(i);
    indeg[i]++;
  }
}

const topo = new Int32Array(n);
const indegForTopo = indeg.slice();
const topoQueue = new Int32Array(n);
let qHead = 0;
let qTail = 0;
for (let i = 0; i < n; i++) {
  if (indegForTopo[i] === 0) topoQueue[qTail++] = i;
}

let topoCount = 0;
while (qHead < qTail) {
  const v = topoQueue[qHead++];
  topo[topoCount++] = v;
  const nexts = succs[v];
  for (let k = 0; k < nexts.length; k++) {
    const u = nexts[k];
    indegForTopo[u]--;
    if (indegForTopo[u] === 0) topoQueue[qTail++] = u;
  }
}

if (topoCount !== n) {
  process.stdout.write(JSON.stringify({ order: ids }));
  process.exit(0);
}

const est = new Float64Array(n);
for (let i = 0; i < n; i++) {
  const v = topo[i];
  const finish = est[v] + durations[v];
  const nexts = succs[v];
  for (let k = 0; k < nexts.length; k++) {
    const u = nexts[k];
    if (finish > est[u]) est[u] = finish;
  }
}

const bottomLevel = new Float64Array(n);
const pathWeight = new Float64Array(n);
const effDeadline = new Float64Array(n);
for (let i = n - 1; i >= 0; i--) {
  const v = topo[i];
  let maxTail = 0;
  let bestPathWeight = 0;
  let bestEffDeadline = deadlines[v];
  const nexts = succs[v];
  for (let k = 0; k < nexts.length; k++) {
    const u = nexts[k];
    if (bottomLevel[u] > maxTail) maxTail = bottomLevel[u];
    if (pathWeight[u] > bestPathWeight) bestPathWeight = pathWeight[u];
    const propagatedDeadline = effDeadline[u] - durations[u];
    if (propagatedDeadline < bestEffDeadline) bestEffDeadline = propagatedDeadline;
  }
  bottomLevel[v] = durations[v] + maxTail;
  pathWeight[v] = weights[v] + bestPathWeight;
  effDeadline[v] = bestEffDeadline;
}

const ownSlack = new Float64Array(n);
const effSlack = new Float64Array(n);
const chainSlack = new Float64Array(n);
const density = new Float64Array(n);
const criticalDensity = new Float64Array(n);
for (let i = 0; i < n; i++) {
  ownSlack[i] = deadlines[i] - est[i] - durations[i];
  effSlack[i] = effDeadline[i] - est[i] - durations[i];
  chainSlack[i] = effDeadline[i] - est[i] - bottomLevel[i];
  const denom = durations[i] > 0 ? durations[i] : 1;
  density[i] = weights[i] / denom;
  criticalDensity[i] = pathWeight[i] / denom;
}

function buildSchedule(strategyId) {
  const ready = new BinaryHeap((a, b) => betterTask(a, b, strategyId));
  const freeMachines = new BinaryHeap((a, b) => a < b);
  const running = new BinaryHeap((a, b) => {
    if (a[0] !== b[0]) return a[0] < b[0];
    if (a[1] !== b[1]) return a[1] < b[1];
    return a[2] < b[2];
  });
  const indegRem = indeg.slice();
  const starts = new Float64Array(n);
  const machines = new Int32Array(n);
  for (let i = 0; i < machineCount; i++) freeMachines.push(i);
  for (let i = 0; i < n; i++) {
    if (indegRem[i] === 0) ready.push(i);
  }

  let currentTime = 0;
  let scheduled = 0;
  let penalty = 0;
  let makespan = 0;

  while (scheduled < n) {
    while (freeMachines.size() > 0 && ready.size() > 0) {
      const task = ready.pop();
      const machine = freeMachines.pop();
      const start = currentTime;
      const finish = start + durations[task];
      starts[task] = start;
      machines[task] = machine;
      running.push([finish, machine, task]);
      if (finish > makespan) makespan = finish;
      const deadline = deadlines[task];
      if (deadline !== INF && finish > deadline) {
        penalty += (finish - deadline) * weights[task];
      }
      scheduled++;
    }

    if (scheduled >= n) break;
    if (running.size() === 0) return null;

    currentTime = running.peek()[0];
    while (running.size() > 0 && running.peek()[0] === currentTime) {
      const [_, machine, task] = running.pop();
      freeMachines.push(machine);
      const nexts = succs[task];
      for (let k = 0; k < nexts.length; k++) {
        const u = nexts[k];
        indegRem[u]--;
        if (indegRem[u] === 0) ready.push(u);
      }
    }
  }

  return {
    starts,
    machines,
    makespan,
    score: penalty + makespan,
  };
}

function betterTask(a, b, strategyId) {
  let cmp;
  switch (strategyId) {
    case 0:
      cmp = compareAsc(effDeadline, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(chainSlack, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(pathWeight, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(bottomLevel, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(density, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(durations, a, b);
      if (cmp) return cmp < 0;
      return a < b;
    case 1:
      cmp = compareAsc(ownSlack, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(deadlines, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(density, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(bottomLevel, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(pathWeight, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(durations, a, b);
      if (cmp) return cmp < 0;
      return a < b;
    case 2:
      cmp = compareDesc(criticalDensity, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(pathWeight, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(effDeadline, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(bottomLevel, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(durations, a, b);
      if (cmp) return cmp < 0;
      return a < b;
    case 3:
      cmp = compareDesc(bottomLevel, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(effDeadline, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(pathWeight, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(density, a, b);
      if (cmp) return cmp < 0;
      return a < b;
    case 4:
      cmp = compareDesc(density, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(effDeadline, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(bottomLevel, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(pathWeight, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(durations, a, b);
      if (cmp) return cmp < 0;
      return a < b;
    default:
      cmp = compareAsc(effSlack, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(criticalDensity, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(deadlines, a, b);
      if (cmp) return cmp < 0;
      cmp = compareDesc(bottomLevel, a, b);
      if (cmp) return cmp < 0;
      cmp = compareAsc(durations, a, b);
      if (cmp) return cmp < 0;
      return a < b;
  }
}

function compareAsc(values, a, b) {
  const av = values[a];
  const bv = values[b];
  if (av < bv) return -1;
  if (av > bv) return 1;
  return 0;
}

function compareDesc(values, a, b) {
  const av = values[a];
  const bv = values[b];
  if (av > bv) return -1;
  if (av < bv) return 1;
  return 0;
}

function chooseStrategies(taskCount) {
  if (taskCount <= 2000) return [0, 1, 2, 3, 4, 5];
  if (taskCount <= 20000) return [0, 1, 2, 3];
  if (taskCount <= 100000) return [0, 2, 3];
  return [0, 3];
}

function toInt(value, fallback) {
  const num = Number(value);
  return Number.isFinite(num) ? Math.floor(num) : fallback;
}

function toNonNegativeInt(value, fallback) {
  const num = Number(value);
  return Number.isFinite(num) && num >= 0 ? Math.floor(num) : fallback;
}

function toFiniteNumber(value, fallback) {
  const num = Number(value);
  return Number.isFinite(num) ? num : fallback;
}

function BinaryHeap(compare) {
  this.data = [];
  this.compare = compare;
}

BinaryHeap.prototype.size = function size() {
  return this.data.length;
};

BinaryHeap.prototype.peek = function peek() {
  return this.data[0];
};

BinaryHeap.prototype.push = function push(value) {
  const data = this.data;
  const compare = this.compare;
  let index = data.length;
  data.push(value);
  while (index > 0) {
    const parent = (index - 1) >> 1;
    if (!compare(value, data[parent])) break;
    data[index] = data[parent];
    index = parent;
  }
  data[index] = value;
};

BinaryHeap.prototype.pop = function pop() {
  const data = this.data;
  const top = data[0];
  const last = data.pop();
  if (data.length === 0) return top;
  const compare = this.compare;
  let index = 0;
  while (true) {
    let left = (index << 1) + 1;
    if (left >= data.length) break;
    let child = left;
    const right = left + 1;
    if (right < data.length && compare(data[right], data[left])) child = right;
    if (!compare(data[child], last)) break;
    data[index] = data[child];
    index = child;
  }
  data[index] = last;
  return top;
};

function main() {
  const strategyIds = chooseStrategies(n);
  let best = null;
  for (let i = 0; i < strategyIds.length; i++) {
    const candidate = buildSchedule(strategyIds[i]);
    if (!candidate) continue;
    if (
      best === null ||
      candidate.score < best.score ||
      (candidate.score === best.score && candidate.makespan < best.makespan)
    ) {
      best = candidate;
    }
  }

  if (!best) {
    process.stdout.write(JSON.stringify({ order: Array.from(topo, v => ids[v]) }));
    return;
  }

  const schedule = new Array(n);
  for (let i = 0; i < n; i++) {
    schedule[i] = {
      id: ids[i],
      start: best.starts[i],
      machine: best.machines[i],
    };
  }

  process.stdout.write(JSON.stringify({ schedule }));
}

main();
