use std::cmp::Reverse;
use std::collections::HashMap;
use std::io::{self, Read, Write};
use std::thread;

const MITM_LIMIT: usize = 42;
const BRANCH_AND_BOUND_LIMIT: usize = 80;
const POSITIVE_DENSE_TARGET_LIMIT: i128 = 12_000_000;
const MIXED_RANGE_LIMIT: i128 = 8_000_000;
const BITSET_WORK_LIMIT: u128 = 250_000_000;
const SPARSE_STATE_LIMIT: usize = 400_000;
const TWO_SUM_LIMIT: usize = 1_000_000;
const BEAM_WIDTH: usize = 4_096;
const BEAM_ITEM_LIMIT: usize = 4_096;
const BRANCH_NODE_LIMIT: u64 = 5_000_000;

struct FastInput {
    buf: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, pos: 0 })
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.buf.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.buf[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        }
        let mut value = 0i64;
        let mut saw_digit = false;
        while self.pos < self.buf.len() {
            let byte = self.buf[self.pos];
            if !byte.is_ascii_digit() {
                break;
            }
            saw_digit = true;
            value = value.saturating_mul(10).saturating_add((byte - b'0') as i64);
            self.pos += 1;
        }
        if !saw_digit {
            None
        } else {
            Some(value.saturating_mul(sign))
        }
    }
}

#[derive(Clone, Copy)]
struct BeamNode {
    prev: usize,
    idx: usize,
}

fn abs_i128(value: i128) -> i128 {
    if value < 0 { -value } else { value }
}

fn gcd_i128(mut a: i128, mut b: i128) -> i128 {
    a = abs_i128(a);
    b = abs_i128(b);
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

fn final_word_mask(highest_bit: usize) -> u64 {
    let used_bits = (highest_bit % 64) + 1;
    if used_bits == 64 {
        u64::MAX
    } else {
        (1u64 << used_bits) - 1
    }
}

fn shift_left_bits(src: &[u64], dst: &mut [u64], shift: usize) {
    dst.fill(0);
    if shift == 0 {
        dst.copy_from_slice(src);
        return;
    }
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    for (src_index, &word) in src.iter().enumerate() {
        if word == 0 {
            continue;
        }
        let dst_index = src_index + word_shift;
        if dst_index >= dst.len() {
            break;
        }
        dst[dst_index] |= word << bit_shift;
        if bit_shift != 0 {
            let next_index = dst_index + 1;
            if next_index < dst.len() {
                dst[next_index] |= word >> (64 - bit_shift);
            }
        }
    }
}

fn shift_right_bits(src: &[u64], dst: &mut [u64], shift: usize) {
    dst.fill(0);
    if shift == 0 {
        dst.copy_from_slice(src);
        return;
    }
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    for src_index in word_shift..src.len() {
        let word = src[src_index];
        if word == 0 {
            continue;
        }
        let dst_index = src_index - word_shift;
        dst[dst_index] |= word >> bit_shift;
        if bit_shift != 0 && dst_index > 0 {
            dst[dst_index - 1] |= word << (64 - bit_shift);
        }
    }
}

fn reconstruct_from_parent_map(
    parent: &HashMap<i128, (i128, usize)>,
    mut target: i128,
) -> Option<Vec<usize>> {
    let mut answer = Vec::new();
    while target != 0 {
        let &(prev, idx) = parent.get(&target)?;
        answer.push(idx);
        target = prev;
    }
    answer.sort_unstable();
    Some(answer)
}

fn solve_two_sum(values: &[i64], target: i128) -> Option<Vec<usize>> {
    let mut seen = HashMap::with_capacity(values.len().saturating_mul(2).min(2_000_000));
    for (idx, &value) in values.iter().enumerate() {
        let needed = target - value as i128;
        if needed >= i64::MIN as i128 && needed <= i64::MAX as i128 {
            if let Some(&other) = seen.get(&(needed as i64)) {
                let mut answer = vec![other, idx];
                answer.sort_unstable();
                return Some(answer);
            }
        }
        seen.entry(value).or_insert(idx);
    }
    None
}

fn gen_sums(
    items: &[(i64, usize)],
    at: usize,
    current_sum: i128,
    mask: u64,
    out: &mut Vec<(i128, u64)>,
) {
    if at == items.len() {
        out.push((current_sum, mask));
        return;
    }
    gen_sums(items, at + 1, current_sum, mask, out);
    gen_sums(
        items,
        at + 1,
        current_sum + items[at].0 as i128,
        mask | (1u64 << at),
        out,
    );
}

fn solve_mitm(items: &[(i64, usize)], target: i128) -> Option<Vec<usize>> {
    let mid = items.len() / 2;
    let (left_items, right_items) = items.split_at(mid);

    let (left_sums, mut right_sums) = thread::scope(|scope| {
        let left_handle = scope.spawn(|| {
            let mut out = Vec::with_capacity(1usize << left_items.len());
            gen_sums(left_items, 0, 0, 0, &mut out);
            out
        });
        let right_handle = scope.spawn(|| {
            let mut out = Vec::with_capacity(1usize << right_items.len());
            gen_sums(right_items, 0, 0, 0, &mut out);
            out
        });
        let left = left_handle.join().expect("left subset worker panicked");
        let right = right_handle.join().expect("right subset worker panicked");
        (left, right)
    });

    right_sums.sort_unstable_by_key(|entry| entry.0);
    for &(left_sum, left_mask) in &left_sums {
        let needed = target - left_sum;
        if let Ok(pos) = right_sums.binary_search_by_key(&needed, |entry| entry.0) {
            let right_mask = right_sums[pos].1;
            let mut answer = Vec::new();
            for bit in 0..left_items.len() {
                if (left_mask >> bit) & 1 == 1 {
                    answer.push(left_items[bit].1);
                }
            }
            for bit in 0..right_items.len() {
                if (right_mask >> bit) & 1 == 1 {
                    answer.push(right_items[bit].1);
                }
            }
            answer.sort_unstable();
            return Some(answer);
        }
    }
    None
}

fn solve_positive_branch_and_bound(items: &[(i64, usize)], target: i128) -> Option<Vec<usize>> {
    fn dfs(
        at: usize,
        remaining: i128,
        items: &[(i64, usize)],
        suffix_sum: &[i128],
        chosen: &mut [bool],
        visited: &mut u64,
        answer: &mut Option<Vec<usize>>,
    ) {
        if answer.is_some() || *visited >= BRANCH_NODE_LIMIT {
            return;
        }
        *visited += 1;
        if remaining == 0 {
            let mut subset = Vec::new();
            for (idx, &take) in chosen.iter().enumerate() {
                if take {
                    subset.push(items[idx].1);
                }
            }
            subset.sort_unstable();
            *answer = Some(subset);
            return;
        }
        if at == items.len() || remaining < 0 || suffix_sum[at] < remaining {
            return;
        }
        let value = items[at].0 as i128;
        if value <= remaining {
            chosen[at] = true;
            dfs(
                at + 1,
                remaining - value,
                items,
                suffix_sum,
                chosen,
                visited,
                answer,
            );
            chosen[at] = false;
        }
        if answer.is_some() || *visited >= BRANCH_NODE_LIMIT {
            return;
        }
        dfs(
            at + 1,
            remaining,
            items,
            suffix_sum,
            chosen,
            visited,
            answer,
        );
    }

    let mut sorted = items.to_vec();
    sorted.sort_unstable_by_key(|&(value, _)| Reverse(value));
    let mut suffix_sum = vec![0i128; sorted.len() + 1];
    for idx in (0..sorted.len()).rev() {
        suffix_sum[idx] = suffix_sum[idx + 1] + sorted[idx].0 as i128;
    }
    let mut chosen = vec![false; sorted.len()];
    let mut visited = 0u64;
    let mut answer = None;
    dfs(
        0,
        target,
        &sorted,
        &suffix_sum,
        &mut chosen,
        &mut visited,
        &mut answer,
    );
    answer
}

fn solve_positive_sparse(
    items: &[(i64, usize)],
    target: i128,
    state_limit: usize,
) -> Option<Vec<usize>> {
    let mut states = vec![0i128];
    let mut parent: HashMap<i128, (i128, usize)> =
        HashMap::with_capacity(state_limit.min(65_536));
    parent.insert(0, (0, usize::MAX));

    for &(value, orig_idx) in items {
        let snapshot_len = states.len();
        let mut additions = Vec::new();
        for state_idx in 0..snapshot_len {
            let next = states[state_idx] + value as i128;
            if next > target {
                continue;
            }
            if let std::collections::hash_map::Entry::Vacant(slot) = parent.entry(next) {
                slot.insert((states[state_idx], orig_idx));
                if next == target {
                    return reconstruct_from_parent_map(&parent, target);
                }
                additions.push(next);
            }
        }
        if states.len() + additions.len() > state_limit {
            return None;
        }
        states.extend(additions);
    }
    reconstruct_from_parent_map(&parent, target)
}

fn solve_positive_dense(
    value_source: &[i64],
    items: &[(i64, usize)],
    target: usize,
) -> Option<Vec<usize>> {
    let words = (target / 64) + 1;
    let mut reachable = vec![0u64; words];
    let mut shifted = vec![0u64; words];
    let mut parent = vec![usize::MAX; target + 1];
    let target_word = target / 64;
    let target_bit = target % 64;
    let last_mask = final_word_mask(target);

    reachable[0] = 1;
    for &(value, orig_idx) in items {
        shift_left_bits(&reachable, &mut shifted, value as usize);
        shifted[words - 1] &= last_mask;
        for word_idx in 0..words {
            let new_bits = shifted[word_idx] & !reachable[word_idx];
            let mut pending = new_bits;
            while pending != 0 {
                let bit = pending.trailing_zeros() as usize;
                let sum = word_idx * 64 + bit;
                if sum <= target {
                    parent[sum] = orig_idx;
                }
                pending &= pending - 1;
            }
            reachable[word_idx] |= shifted[word_idx];
        }
        if ((reachable[target_word] >> target_bit) & 1) == 1 {
            break;
        }
    }

    if ((reachable[target_word] >> target_bit) & 1) == 0 {
        return None;
    }

    let mut sum = target;
    let mut answer = Vec::new();
    while sum > 0 {
        let idx = parent[sum];
        if idx == usize::MAX {
            return None;
        }
        answer.push(idx);
        sum -= value_source[idx] as usize;
    }
    answer.sort_unstable();
    Some(answer)
}

fn solve_bounded_range(
    value_source: &[i64],
    items: &[(i64, usize)],
    target: i128,
    min_sum: i128,
    max_sum: i128,
) -> Option<Vec<usize>> {
    let width = usize::try_from(max_sum - min_sum).ok()?;
    let offset = usize::try_from(-min_sum).ok()?;
    let target_pos = usize::try_from(target - min_sum).ok()?;
    let words = (width / 64) + 1;
    let last_mask = final_word_mask(width);
    let target_word = target_pos / 64;
    let target_bit = target_pos % 64;

    let mut reachable = vec![0u64; words];
    let mut shifted = vec![0u64; words];
    let mut parent = vec![usize::MAX; width + 1];
    reachable[offset / 64] |= 1u64 << (offset % 64);

    for &(value, orig_idx) in items {
        if value >= 0 {
            shift_left_bits(&reachable, &mut shifted, value as usize);
        } else {
            shift_right_bits(&reachable, &mut shifted, (-value) as usize);
        }
        shifted[words - 1] &= last_mask;
        for word_idx in 0..words {
            let new_bits = shifted[word_idx] & !reachable[word_idx];
            let mut pending = new_bits;
            while pending != 0 {
                let bit = pending.trailing_zeros() as usize;
                let pos = word_idx * 64 + bit;
                if pos <= width {
                    parent[pos] = orig_idx;
                }
                pending &= pending - 1;
            }
            reachable[word_idx] |= shifted[word_idx];
        }
        if ((reachable[target_word] >> target_bit) & 1) == 1 {
            break;
        }
    }

    if ((reachable[target_word] >> target_bit) & 1) == 0 {
        return None;
    }

    let mut current = target;
    let mut answer = Vec::new();
    while current != 0 {
        let pos = usize::try_from(current - min_sum).ok()?;
        let idx = parent[pos];
        if idx == usize::MAX {
            return None;
        }
        answer.push(idx);
        current -= value_source[idx] as i128;
    }
    answer.sort_unstable();
    Some(answer)
}

fn solve_sparse_mixed(
    items: &[(i64, usize)],
    target: i128,
    state_limit: usize,
) -> Option<Vec<usize>> {
    let mut states = vec![0i128];
    let mut parent: HashMap<i128, (i128, usize)> =
        HashMap::with_capacity(state_limit.min(65_536));
    parent.insert(0, (0, usize::MAX));

    for &(value, orig_idx) in items {
        let snapshot_len = states.len();
        let mut additions = Vec::new();
        for state_idx in 0..snapshot_len {
            let next = states[state_idx] + value as i128;
            if let std::collections::hash_map::Entry::Vacant(slot) = parent.entry(next) {
                slot.insert((states[state_idx], orig_idx));
                if next == target {
                    return reconstruct_from_parent_map(&parent, target);
                }
                additions.push(next);
            }
        }
        if states.len() + additions.len() > state_limit {
            return None;
        }
        states.extend(additions);
    }
    reconstruct_from_parent_map(&parent, target)
}

fn reconstruct_beam(nodes: &[BeamNode], mut node: usize) -> Vec<usize> {
    let mut answer = Vec::new();
    while node != 0 {
        let current = nodes[node];
        answer.push(current.idx);
        node = current.prev;
    }
    answer.sort_unstable();
    answer
}

fn solve_beam(items: &[(i64, usize)], target: i128, width: usize) -> Option<Vec<usize>> {
    let mut nodes = vec![BeamNode {
        prev: 0,
        idx: usize::MAX,
    }];
    let mut states = vec![(0i128, 0usize)];

    for &(value, orig_idx) in items {
        let mut next_map: HashMap<i128, usize> =
            HashMap::with_capacity(states.len().saturating_mul(2).saturating_add(1));
        for &(sum, node) in &states {
            next_map.entry(sum).or_insert(node);
        }
        for &(sum, node) in &states {
            let next_sum = sum + value as i128;
            if let std::collections::hash_map::Entry::Vacant(slot) = next_map.entry(next_sum) {
                nodes.push(BeamNode { prev: node, idx: orig_idx });
                slot.insert(nodes.len() - 1);
            }
        }
        if let Some(&node) = next_map.get(&target) {
            return Some(reconstruct_beam(&nodes, node));
        }

        let mut next_states: Vec<(i128, usize)> = next_map.into_iter().collect();
        next_states.sort_unstable_by(|a, b| {
            let da = abs_i128(a.0 - target);
            let db = abs_i128(b.0 - target);
            da.cmp(&db).then_with(|| b.0.cmp(&a.0))
        });
        if next_states.len() > width {
            next_states.truncate(width);
        }
        states = next_states;
    }
    None
}

fn solve_positive_subset(values: &[i64], target: i128) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    let mut items = Vec::new();
    let mut total = 0i128;
    for (idx, &value) in values.iter().enumerate() {
        if value <= 0 {
            continue;
        }
        let value_i128 = value as i128;
        if value_i128 > target {
            continue;
        }
        total += value_i128;
        items.push((value, idx));
    }

    if total < target {
        return None;
    }
    if total == target {
        let mut answer: Vec<usize> = items.iter().map(|&(_, idx)| idx).collect();
        answer.sort_unstable();
        return Some(answer);
    }
    if items.len() <= MITM_LIMIT {
        return solve_mitm(&items, target);
    }
    if items.len() <= BRANCH_AND_BOUND_LIMIT {
        if let Some(answer) = solve_positive_branch_and_bound(&items, target) {
            return Some(answer);
        }
    }
    if items.len() <= 4_000 {
        if let Some(answer) = solve_positive_sparse(&items, target, SPARSE_STATE_LIMIT) {
            return Some(answer);
        }
    }
    if target <= POSITIVE_DENSE_TARGET_LIMIT {
        let target_usize = usize::try_from(target).ok()?;
        let words = (target_usize / 64) + 1;
        let work = words as u128 * items.len() as u128;
        if work <= BITSET_WORK_LIMIT {
            if let Some(answer) = solve_positive_dense(values, &items, target_usize) {
                return Some(answer);
            }
        }
    }

    let mut beam_items = items;
    beam_items.sort_unstable_by_key(|&(value, _)| Reverse(value));
    if beam_items.len() > BEAM_ITEM_LIMIT {
        beam_items.truncate(BEAM_ITEM_LIMIT);
    }
    solve_beam(&beam_items, target, BEAM_WIDTH)
}

fn solve_mixed_subset(
    values: &[i64],
    target: i128,
    min_sum: i128,
    max_sum: i128,
) -> Option<Vec<usize>> {
    let mut items = Vec::new();
    for (idx, &value) in values.iter().enumerate() {
        if value != 0 {
            items.push((value, idx));
        }
    }

    if items.len() <= MITM_LIMIT {
        return solve_mitm(&items, target);
    }

    let width_i128 = max_sum - min_sum;
    if width_i128 <= MIXED_RANGE_LIMIT {
        let width = usize::try_from(width_i128).ok()?;
        let words = (width / 64) + 1;
        let work = words as u128 * items.len() as u128;
        if work <= BITSET_WORK_LIMIT {
            if let Some(answer) = solve_bounded_range(values, &items, target, min_sum, max_sum) {
                return Some(answer);
            }
        }
    }

    if items.len() <= 2_000 {
        if let Some(answer) = solve_sparse_mixed(&items, target, SPARSE_STATE_LIMIT) {
            return Some(answer);
        }
    }

    items.sort_unstable_by(|a, b| {
        let a_abs = abs_i128(a.0 as i128);
        let b_abs = abs_i128(b.0 as i128);
        b_abs.cmp(&a_abs).then_with(|| b.0.cmp(&a.0))
    });
    if items.len() > BEAM_ITEM_LIMIT {
        items.truncate(BEAM_ITEM_LIMIT);
    }
    solve_beam(&items, target, BEAM_WIDTH)
}

fn solve_subset_sum(values: &[i64], target: i128) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    if values.is_empty() {
        return None;
    }

    let mut gcd = 0i128;
    let mut sum_all = 0i128;
    let mut min_sum = 0i128;
    let mut max_sum = 0i128;
    let mut all_non_negative = true;
    let mut all_non_positive = true;

    for (idx, &value) in values.iter().enumerate() {
        let value_i128 = value as i128;
        if value_i128 == target {
            return Some(vec![idx]);
        }
        if value_i128 != 0 {
            gcd = if gcd == 0 {
                abs_i128(value_i128)
            } else {
                gcd_i128(gcd, value_i128)
            };
        }
        sum_all += value_i128;
        if value > 0 {
            max_sum += value_i128;
            all_non_positive = false;
        } else if value < 0 {
            min_sum += value_i128;
            all_non_negative = false;
        }
    }

    if gcd != 0 && target % gcd != 0 {
        return None;
    }
    if target < min_sum || target > max_sum {
        return None;
    }
    if sum_all == target {
        return Some((0..values.len()).collect());
    }
    if target == max_sum {
        let mut answer = Vec::new();
        for (idx, &value) in values.iter().enumerate() {
            if value > 0 {
                answer.push(idx);
            }
        }
        return Some(answer);
    }
    if target == min_sum {
        let mut answer = Vec::new();
        for (idx, &value) in values.iter().enumerate() {
            if value < 0 {
                answer.push(idx);
            }
        }
        return Some(answer);
    }
    if values.len() <= TWO_SUM_LIMIT {
        if let Some(answer) = solve_two_sum(values, target) {
            return Some(answer);
        }
    }

    if all_non_negative {
        return solve_positive_subset(values, target);
    }
    if all_non_positive {
        let transformed: Vec<i64> = values.iter().map(|&value| -value).collect();
        return solve_positive_subset(&transformed, -target);
    }
    solve_mixed_subset(values, target, min_sum, max_sum)
}

fn main() -> io::Result<()> {
    let mut input = FastInput::new()?;
    let Some(raw_n) = input.next_i64() else {
        return Ok(());
    };
    let Some(raw_target) = input.next_i64() else {
        return Ok(());
    };

    let n = if raw_n <= 0 { 0usize } else { raw_n as usize };
    let target = raw_target as i128;
    let mut values = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(value) = input.next_i64() {
            values.push(value);
        } else {
            break;
        }
    }

    let solution = solve_subset_sum(&values, target);
    let mut out = io::BufWriter::new(io::stdout().lock());
    match solution {
        Some(indices) => {
            writeln!(out, "YES")?;
            if indices.is_empty() {
                writeln!(out)?;
            } else {
                for (pos, idx) in indices.iter().enumerate() {
                    if pos > 0 {
                        write!(out, " ")?;
                    }
                    write!(out, "{idx}")?;
                }
                writeln!(out)?;
            }
        }
        None => {
            writeln!(out, "NO")?;
        }
    }
    Ok(())
}
